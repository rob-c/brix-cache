1/* File: gsi_outbound_common.c — Credentialed XRootD login for native TPC pulls
2 * WHAT: After kXR_login returns kXR_authmore, completes the authentication handshake using either WLCG JWT (ztn) from brix_tpc_outbound_bearer_file or GSI cert chain when the server advertises &P=ztn / &P=gsi in the login parameter block. Provides wire helpers (tpc_put_u32 for big-endian encoding, tpc_send_kxr_auth for ClientRequestHdr construction + send_all) and public auth path functions (tpc_outbound_ztn for JWT bearer, tpc_read_bearer_token for file read delegation).
3 *
4 * WHY: Native TPC pull connects directly to a remote xrootd server; after anonymous handshake (bootstrap.c), authenticated fetch requires sending credentials on the outbound socket. The server's kXR_login response parameter block advertises which auth method it accepts (&P=ztn for JWT, &P=gsi for GSI). This file provides both wire-level helpers and the credential assembly + send logic for each path — ztn reads token from config file or delegated_token buffer, builds "ztn\x00" + token payload, sends kXR_auth ClientRequestHdr. GSI path (certreq chain) lives in gsi_outbound_certreq.c and gsi_outbound_exchange.c.
5 *
6 * HOW: tpc_put_u32 — htonl(v) → ngx_memcpy to output buffer for big-endian wire encoding; tpc_send_kxr_auth — zero ClientRequestHdr, set streamid[1] = seq + requestid = kXR_auth, memcpy ctype from cred_payload into hdr body offset 12, set dlen via htonl, send_all(hdr) then send_all(cred_payload); tpc_read_bearer_token — read path from t->conf->tpc_outbound_bearer_file, delegate to brix_token_read_file with "TPC outbound" label; tpc_outbound_ztn — check delegated_token[0] != '\0' (OAuth2/OIDC exchange result) → strlen → malloc(4+token_len) → memcpy("ztn\x00") + token → tpc_send_kxr_auth → recv_response checking status == kXR_ok → free(cred/body). 7 * */

32/* WHAT: Big-endian uint32 wire encoding helper — htonl(v) → ngx_memcpy to output buffer. */
33
34void
35tpc_put_u32(u_char *p, uint32_t v)
36{
37 uint32_t be;
38
39 /*
40 * XRootD security buckets carry length/tag fields as network-order
41 * (big-endian) uint32. htonl() normalises from host order, then we copy the
42 * raw 4 bytes — ngx_memcpy (not a *(uint32_t *)p store) so callers can write
43 * to unaligned cursor positions inside a packed wire buffer.
44 */
45 be = htonl(v);
46 ngx_memcpy(p, &be, sizeof(be));
47}

49/* WHAT: Construct kXR_auth ClientRequestHdr (streamid + requestid + ctype from cred_payload + dlen via htonl), send_all(hdr) then send_all(cred_payload). Returns 0 or -1 with error code. */
50
51int
52tpc_send_kxr_auth(brix_tpc_pull_t *t, int fd, u_char stream_seq,
53 const u_char *cred_payload, uint32_t cred_len)
54{
55 ClientRequestHdr hdr;
56 u_char ctype[4];
57
58 /*
59 * Build the fixed 24-byte kXR_auth request header. ClientRequestHdr is the
60 * generic overlay: streamid[2] | requestid (u16) | body[16] | dlen (u32).
61 * For kXR_auth that body region maps to ClientAuthRequest as
62 * reserved[12] | credtype[4], so the credtype lands at body offset 12
63 * (= absolute byte offset 16 in the header).
64 */
65 ngx_memzero(&hdr, sizeof(hdr));
66 /* streamid[1] is the low byte; we use it as the handshake sequence number
67 * (streamid[0] stays 0). Matches the seq the server echoes in replies. */
68 hdr.streamid[1] = stream_seq;
69 hdr.requestid = htons(kXR_auth); /* network order on the wire */
70 /*
71 * The credtype field advertises the auth protocol name as the first 4 bytes
72 * of the payload, NUL-padded — "gsi\0", "ztn\0", etc. Mirror those bytes
73 * verbatim into the header's credtype slot; the full name (and its trailing
74 * data) is also sent in the payload below. cred_payload must be >= 4 bytes.
75 */
76 ngx_memcpy(ctype, cred_payload, sizeof(ctype));
77 ngx_memcpy(hdr.body + 12, ctype, 4); /* credtype field (offset 16) */
78 hdr.dlen = htonl((kXR_int32) cred_len); /* payload length, net order */
79
80 /*
81 * Wire order: header first, then the cred_len-byte payload. Both go through
82 * tpc_send_all which loops over partial writes; a short/failed send here
83 * leaves the socket mid-message, so the caller must treat -1 as fatal for
84 * this connection (it does — every caller aborts the pull).
85 */
86 if (tpc_send_all(t, fd, &hdr, sizeof(hdr)) != 0) {
87 snprintf(t->err_msg, sizeof(t->err_msg), "TPC kXR_auth send hdr failed");
88 t->xrd_error = kXR_ServerError;
89 return -1;
90 }
91 if (tpc_send_all(t, fd, cred_payload, cred_len) != 0) {
92 snprintf(t->err_msg, sizeof(t->err_msg), "TPC kXR_auth send body failed");
93 t->xrd_error = kXR_ServerError;
94 return -1;
95 }
96
97 return 0;
98}

129/* WHAT: JWT bearer auth path — check delegated_token[0] != '\0' (OAuth2/OIDC exchange result) → strlen → malloc(4+token_len) → memcpy("ztn\x00") + token → tpc_send_kxr_auth(kXR_auth, seq=3) → recv_response checking status == kXR_ok → free(cred/body). Returns 0 or -1 with error code. Caller: tpc/thread.c (auth path dispatch based on login parameter block &P=ztn). */
130
131int
132tpc_outbound_ztn(brix_tpc_pull_t *t, int fd)
133{
134 u_char *cred;
135 uint32_t cred_len;
136 uint16_t status;
137 uint32_t dlen;
138 u_char *body;
139 size_t token_len;
140
141 /*
142 * Use the delegated token from OAuth2/OIDC token exchange when available;
143 * otherwise fall back to reading from the bearer file.
144 */
145 if (t->delegated_token[0] != '\0') {
146 /* Already populated by an earlier OAuth2/OIDC exchange for this pull. */
147 token_len = strlen(t->delegated_token);
148 } else {
149 u_char token_buf[TPC_BEARER_MAX];
150 if (tpc_read_bearer_token(t, token_buf, sizeof(token_buf), &token_len)
151 != 0)
152 {
153 return -1;
154 }
155 /*
156 * Cache the file-sourced token in t->delegated_token so any later auth
157 * round reuses it. token_len + 1 copies the trailing NUL that
158 * tpc_read_bearer_token guarantees; re-derive token_len from the cached
159 * copy so a NUL embedded in the file truncates here, not on the wire.
160 */
161 ngx_memcpy(t->delegated_token, token_buf, token_len + 1);
162 token_len = strlen(t->delegated_token);
163 }
164
165 /*
166 * ztn credential layout: 4-byte protocol tag "ztn\0" followed by the raw
167 * token bytes (no NUL, no length prefix — dlen in the header bounds it).
168 */
169 cred_len = (uint32_t) (4 + token_len);
170 cred = malloc((size_t) cred_len);
171 if (cred == NULL) {
172 snprintf(t->err_msg, sizeof(t->err_msg), "TPC ztn cred OOM");
173 t->xrd_error = kXR_NoMemory;
174 return -1;
175 }
176
177 ngx_memcpy(cred, "ztn\x00", 4); /* protocol tag, slots 0..3 */
178 ngx_memcpy(cred + 4, t->delegated_token, token_len);
179
180 /*
181 * seq=3 is the handshake sequence following bootstrap (protocol=1, login=2),
182 * so this is the first authenticated request on the outbound socket. cred is
183 * a malloc()'d temporary owned here — free it on every exit path (the send
184 * helper copies what it needs into the header and writes the body inline).
185 */
186 if (tpc_send_kxr_auth(t, fd, 3, cred, cred_len) != 0) {
187 free(cred);
188 return -1;
189 }
190 free(cred);
191
192 /*
193 * tpc_recv_response allocates *body for the server reply; we own it from
194 * here. The reply payload is unused for ztn (success is signalled by status
195 * alone), so free it unconditionally before inspecting status.
196 */
197 body = NULL;
198 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
199 snprintf(t->err_msg, sizeof(t->err_msg), "TPC ztn auth recv failed");
200 t->xrd_error = kXR_ServerError;
201 return -1;
202 }
203 free(body);
204
205 if (status != kXR_ok) {
206 snprintf(t->err_msg, sizeof(t->err_msg),
207 "TPC ztn auth rejected (status=%u)", (unsigned) status);
208 t->xrd_error = kXR_AuthFailed;
209 return -1;
210 }
211
212 return 0;
213}
