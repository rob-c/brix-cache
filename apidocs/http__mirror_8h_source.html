1/*
2 * http_mirror.h — Phase 24 HTTP/WebDAV traffic mirror.
3 *
4 * Off by default. When `brix_mirror_url http[s]://host:port` is configured on
5 * a WebDAV location, a qualifying READ request (GET/HEAD/PROPFIND/OPTIONS, per
6 * brix_mirror_methods) spawns one fire-and-forget background subrequest per
7 * shadow target. The primary request is served to the client normally and is
8 * never delayed; the shadow response body is discarded and its status compared
9 * to the primary to detect divergence.
10 *
11 * Two phase handlers implement this:
12 * - brix_http_mirror_precontent_handler (NGX_HTTP_PRECONTENT_PHASE) does
13 * both jobs: on the MAIN request it fires the background subrequests, and
14 * on each mirror SUBREQUEST it takes the request over and proxies it to the
15 * shadow target with credentials stripped. Running the takeover before the
16 * content phase avoids any dependence on content-phase handler ordering.
17 * - brix_http_mirror_log_handler (NGX_HTTP_LOG_PHASE) records the primary's
18 * final status for the divergence comparison.
19 * Both are registered from src/webdav/postconfig.c.
20 */

26/* PRECONTENT_PHASE handler with two roles keyed on r vs r->main:
27 * - MAIN request: if mirroring is enabled, the method/write/loop/sample gates
28 * pass, and it has not already fired (ctx->mirror_fired), fire one background
29 * subrequest per shadow target. For body methods (PUT) it first buffers the
30 * client body (returns NGX_DONE to suspend the main request, fires from the
31 * body handler); otherwise it fires inline and returns NGX_DECLINED so the
32 * main request proceeds to its content handler — the client is never delayed.
33 * - mirror SUBREQUEST (ctx->is_mirror): takes the request over and proxies it
34 * to the shadow with credentials stripped.
35 * Returns NGX_DECLINED to pass control on (the common case), NGX_DONE while the
36 * body is being read, the proxy rc on a shadow subreq, or an HTTP error status.
37 * May allocate and attach the webdav req ctx on r->pool. */
38ngx_int_t brix_http_mirror_precontent_handler(ngx_http_request_t *r);
39
40/* LOG_PHASE handler: on the MAIN request only, if a mirror was fired, stamp the
41 * primary's final HTTP status into ctx->primary_status for divergence
42 * comparison. Always returns NGX_OK; no-op for subrequests. */
43ngx_int_t brix_http_mirror_log_handler(ngx_http_request_t *r);
