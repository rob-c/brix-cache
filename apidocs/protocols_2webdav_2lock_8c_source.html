1/*
2 * lock.c — WebDAV LOCK handler (RFC 4918 §9.10).
3 *
4 * Lock state is stored as a single xattr on the locked resource
5 * (see prop_xattr.c). No shared memory or mutex is needed:
6 * XATTR_CREATE provides kernel-serialized atomic lock creation;
7 * expiry cleanup is lazy (on next access to that path).
8 *
9 * The lock-conflict walk (webdav_check_locks / _tree) lives in lock_check.c;
10 * UNLOCK, the startup sweep and PROPFIND lock-discovery XML live in
11 * lock_discovery.c. The confined-ctx helper (webdav_lock_vfs_ctx) and the
12 * lock-null reaper (webdav_lock_reap_null) are defined here and shared with
13 * those siblings via lock_internal.h.

39/*
40 * Generate Version 4 UUID for WebDAV lock token (`opaquelocktoken:` prefix).
41 * OpenSSL RAND_bytes provides collision-resistant identifiers.
42 */
43static void
44webdav_generate_uuid(char *buf, size_t bufsz)
45{
46 u_char bytes[16];
47 (void) RAND_bytes(bytes, 16);
48 bytes[6] = (bytes[6] & 0x0f) | 0x40; /* Version 4 */
49 bytes[8] = (bytes[8] & 0x3f) | 0x80; /* Variant */
50
51 snprintf(buf, bufsz,
52 "opaquelocktoken:%02x%02x%02x%02x-%02x%02x-%02x%02x"
53 "-%02x%02x-%02x%02x%02x%02x%02x%02x",
54 bytes[0], bytes[1], bytes[2], bytes[3],
55 bytes[4], bytes[5],
56 bytes[6], bytes[7],
57 bytes[8], bytes[9],
58 bytes[10], bytes[11], bytes[12], bytes[13], bytes[14], bytes[15]);
59}

61/*
62 * Build and send the RFC 4918 §9.10 LOCK response body (activelock XML)
63 * plus Lock-Token response header.
64 */
65static void
66webdav_lock_xml_response(ngx_http_request_t *r, webdav_lock_xattr_t *e)
67{
68 ngx_chain_t *head = NULL, *tail = NULL;
69 char timeout_buf[32];
70 char *safe_owner;
71 int64_t now = (int64_t) ngx_time();
72 ngx_uint_t remaining;
73 off_t total_len = 0;
74 ngx_chain_t *lc;
75 ngx_table_elt_t *h;
76
77 /* expires is absolute wall-clock seconds → remaining is a plain subtraction. */
78 remaining = (e->expires > now) ? (ngx_uint_t) (e->expires - now) : 0;
79 ngx_sprintf((u_char *) timeout_buf, "Second-%ui", remaining);
80
81 safe_owner = webdav_escape_xml_text(r->pool, e->owner);
82 if (safe_owner == NULL) {
83 safe_owner = "anonymous";
84 }
85
86 brix_http_chain_appendf(r->pool, &head, &tail,
87 "<?xml version=\"1.0\" encoding=\"utf-8\" ?>"
88 "<D:prop xmlns:D=\"DAV:\">"
89 "<D:lockdiscovery>"
90 "<D:activelock>"
91 "<D:locktype><D:write/></D:locktype>"
92 "<D:lockscope>%s</D:lockscope>"
93 "<D:depth>%s</D:depth>"
94 "<D:owner>%s</D:owner>"
95 "<D:timeout>%s</D:timeout>"
96 "<D:locktoken><D:href>%s</D:href></D:locktoken>"
97 "</D:activelock>"
98 "</D:lockdiscovery>"
99 "</D:prop>",
100 e->exclusive ? "<D:exclusive/>" : "<D:shared/>",
101 e->depth_infinity ? "infinity" : "0",
102 safe_owner, timeout_buf, e->token);
103
104 if (tail != NULL) {
105 tail->buf->last_buf = 1;
106 tail->buf->last_in_chain = 1;
107 }
108
109 for (lc = head; lc != NULL; lc = lc->next) {
110 total_len += lc->buf->last - lc->buf->pos;
111 }
112
113 r->headers_out.content_length_n = total_len;
114
115 h = ngx_list_push(&r->headers_out.headers);
116 if (h != NULL) {
117 h->hash = 1;
118 ngx_str_set(&h->key, "Content-Type");
119 ngx_str_set(&h->value, "application/xml; charset=\"utf-8\"");
120 }
121
122 h = ngx_list_push(&r->headers_out.headers);
123 if (h != NULL) {
124 h->hash = 1;
125 ngx_str_set(&h->key, "Lock-Token");
126 h->value.len = strlen(e->token) + 2;
127 h->value.data = ngx_pnalloc(r->pool, h->value.len);
128 if (h->value.data != NULL) {
129 ngx_sprintf(h->value.data, "<%s>", e->token);
130 }
131 }
132
133 ngx_http_send_header(r);
134 ngx_http_output_filter(r, head);
135}

137/*
138 * webdav_lock_reap_null — release a lock-null placeholder.
139 *
140 * WHAT: If `e` recorded a lock-null lock (the LOCK created a zero-byte resource
141 * to reserve a name, RFC 4918 §9.10.1) AND the resource is still an empty
142 * regular file, unlink it so the reserved name disappears with the lock.
143 * WHY: Without this, UNLOCK (or expiry) of a never-PUT lock-null resource would
144 * leave an orphaned 0-byte file behind. A resource the client PUT data into
145 * is non-empty and is never reaped, so the is_null flag becomes inert once
146 * real content exists — no need to clear it on PUT.
147 * HOW: Confined no-follow lstat; only a regular, zero-length file is unlinked,
148 * via the confined unlink. Best-effort — failures are ignored. MUST NOT be
149 * called on a path that is about to be re-locked (it would remove the file
150 * out from under a follow-up XATTR_CREATE).
151 */
152/* Build a confined VFS ctx for a lock-DB namespace op on `path` (mirrors the
153 * canonical webdav construction). Identity comes from the request ctx so the op
154 * runs as the mapped user under impersonation. */
155void
156webdav_lock_vfs_ctx(ngx_http_request_t *r, const char *path,
157 brix_vfs_ctx_t *vctx)
158{
159 ngx_http_brix_webdav_loc_conf_t *conf =
160 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
161 ngx_http_brix_webdav_req_ctx_t *rx =
162 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
163 int is_tls = 0;
164
165#if (NGX_HTTP_SSL)
166 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
167#endif
168
169 brix_vfs_ctx_init(vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
170 conf->common.root_canon, conf->cache_root_canon, conf->common.allow_write,
171 is_tls, (rx != NULL) ? rx->identity : NULL, path);
172}

174void
175webdav_lock_reap_null(ngx_http_request_t *r, const char *path,
176 const webdav_lock_xattr_t *e)
177{
178 brix_vfs_ctx_t vctx;
179 brix_vfs_stat_t vst;
180
181 if (!e->is_null) {
182 return;
183 }
184
185 /* Confined no-follow probe; only a regular, zero-length file is unlinked,
186 * through the VFS. Best-effort — failures are ignored. */
187 webdav_lock_vfs_ctx(r, path, &vctx);
188 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) == NGX_OK

191 (void) brix_vfs_unlink(&vctx);
192 }

197/*
198 * LOCK body-ready callback. Phase 40: the lockinfo body is read asynchronously,
199 * so the dispatch wrapper already cleared the impersonation principal.
200 * Re-establish it (mirrors PUT/PROPFIND/PROPPATCH) so the lock xattr AND any
201 * zero-byte resource creation happen AS THE MAPPED USER via the broker; without
202 * it the worker (svc) cannot setxattr the user-owned lock target (EACCES) and
203 * LOCK fails 500. No-op unless map mode is active.
204 */
205void
206webdav_handle_lock(ngx_http_request_t *r)
207{
208 ngx_http_brix_webdav_req_ctx_t *rx =
209 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
210
211 brix_imp_request_begin(rx != NULL ? rx->identity : NULL);
212 webdav_handle_lock_inner(r);
213 brix_imp_request_end();
214}

216/*
217 * webdav_lock_create_null — RFC 4918 §9.10.1 lock-null placeholder creation.
218 *
219 * WHAT: If `path` does not exist, atomically create a zero-byte resource to
220 * host the lock, and report via *created_null whether we made it.
221 * WHY: A LOCK on a non-existent resource MUST reserve the name; the placeholder
222 * is reaped on UNLOCK/expiry if never converted by a PUT/MKCOL.
223 * HOW: Confined no-follow probe; on ENOENT do an O_CREATE|O_EXCL open. A race
224 * loser (EEXIST) is not an error and leaves created_null clear. Returns
225 * NGX_OK on success (whether or not a file was created) or
226 * NGX_HTTP_INTERNAL_SERVER_ERROR when the create fails for another reason.
227 */
228static ngx_int_t
229webdav_lock_create_null(ngx_http_request_t *r, const char *path,
230 int *created_null)
231{
232 brix_vfs_ctx_t vctx;
233 brix_vfs_stat_t vst;
234 brix_vfs_file_t *fh;
235 int verr = 0;
236
237 *created_null = 0;
238
239 webdav_lock_vfs_ctx(r, path, &vctx);
240 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) != NGX_DECLINED
241 || errno != ENOENT)
242 {
243 return NGX_OK;
244 }
245
246 fh = brix_vfs_open(&vctx,
247 BRIX_VFS_O_WRITE | BRIX_VFS_O_CREATE | BRIX_VFS_O_EXCL, &verr);
248 if (fh == NULL) {
249 return (verr == EEXIST) ? NGX_OK : NGX_HTTP_INTERNAL_SERVER_ERROR;
250 }
251
252 brix_vfs_close(fh, r->connection->log);
253 /* We created the resource solely to host this lock: it is a lock-null
254 * placeholder, reaped on UNLOCK/expiry if still empty. */
255 *created_null = 1;
256 return NGX_OK;
257}

268/*
269 * webdav_lock_owner_set — choose the <D:owner> value for a fresh lock.
270 *
271 * WHAT: Copies, in priority order, the request-supplied owner, else the
272 * authenticated DN, else the literal "anonymous", into e->owner.
273 * WHY: Keeps owner-selection out of the create path so its intent is explicit
274 * and unchanged. HOW: bounded ngx_cpystrn into the fixed owner buffer.
275 */
276static void
277webdav_lock_owner_set(webdav_lock_xattr_t *e, const char *owner,
278 const ngx_http_brix_webdav_req_ctx_t *ctx)
279{
280 if (owner[0] != '\0') {
281 ngx_cpystrn((u_char *) e->owner, (u_char *) owner, sizeof(e->owner));
282 } else if (ctx != NULL && ctx->dn[0] != '\0') {
283 ngx_cpystrn((u_char *) e->owner, (u_char *) ctx->dn, sizeof(e->owner));
284 } else {
285 ngx_cpystrn((u_char *) e->owner, (u_char *) "anonymous",
286 sizeof(e->owner));
287 }

290/*
291 * webdav_lock_write_status — map an XATTR_CREATE lock-write result to a status.
292 *
293 * WHAT: Translates the write rc/errno into the wire status for the new-lock
294 * path: NGX_HTTP_LOCKED on a lost race (NGX_DECLINED), 403 vs 500 on
295 * failure, or NGX_OK on success.
296 * WHY: The two-EACCES-source disambiguation is subtle and load-bearing; giving
297 * it its own function keeps the create path readable while preserving the
298 * exact policy-vs-syscall distinction documented below.
299 * HOW: See the inline rationale — a remote-backed export's credential gate
300 * (drv != NULL) yields 403; a plain local-syscall EACCES stays 500.
301 */
302static ngx_int_t
303webdav_lock_write_status(ngx_http_request_t *r,
304 ngx_http_brix_webdav_loc_conf_t *conf, ngx_int_t rc)
305{
306 if (rc == NGX_DECLINED) {
307 /* Another worker created the lock between our read and write. */
308 return NGX_HTTP_LOCKED;
309 }
310 if (rc == NGX_OK) {
311 return NGX_OK;
312 }
313
314 /* Two DIFFERENT EACCES sources land here, and they must map to
315 * different statuses (phase-3 T1):
316 * (a) the per-user backend credential gate (brix_vfs_ns_cred,
317 * src/fs/vfs/vfs_xattr.c) denying a no-cred user in deny
318 * mode on a REMOTE-BACKED export — the gate runs and fails
319 * BEFORE brix_vfs_setxattr ever reaches a syscall, and only
320 * engages when the lock ctx's vctx->sd is a non-NULL backend
321 * instance (webdav_lock_vfs_ctx_init in prop_xattr.c binds
322 * it via brix_webdav_backend_instance). This is a policy
323 * refusal -> 403.
324 * (b) the impersonation broker's setxattr on a user-owned file
325 * failing EACCES on a LOCAL POSIX export (vctx->sd stays
326 * NULL there, so the gate path above is never taken and
327 * this EACCES can only come from the raw syscall) -- the
328 * service worker genuinely could not perform the op. This
329 * is intentionally left 500 (documented at
330 * webdav_handle_lock's WHAT/WHY/HOW comment above).
331 * The two cases are structurally distinguishable at this callsite
332 * because vfs_xattr.c sets errno = EACCES from the gate check
333 * (source a) using a DIFFERENT code path (drv != NULL) than the
334 * plain-syscall EACCES (source b, drv == NULL) -- but distinguishing
335 * *here* would require re-deriving whether the export is remote-
336 * backed, duplicating prop_xattr.c's internal ctx-driver logic.
337 * brix_webdav_backend_instance(conf, log) is idempotent and cheap
338 * (a config-scoped cached instance, not a fresh connection), so
339 * calling it again here to re-derive "remote-backed?" is safe and
340 * avoids threading a new out-parameter through webdav_lock_xattr_write. */
341 if ((errno == EACCES || errno == EPERM)
342 && brix_webdav_backend_instance(conf, r->connection->log) != NULL)
343 {
344 return NGX_HTTP_FORBIDDEN;
345 }
346 return NGX_HTTP_INTERNAL_SERVER_ERROR;
347}

349/*
350 * webdav_lock_refresh — refresh an existing active lock (RFC 4918 §9.10).
351 *
352 * WHAT: If the client owns the lock, extend its timeout and rewrite the xattr;
353 * otherwise report 423 Locked.
354 * WHY: Splits the "existing lock" branch out of the handler so both branches
355 * read as a single decision each. HOW: token match via the If header,
356 * XATTR_REPLACE rewrite; sets 200 OK on success. Returns NGX_OK when the
357 * caller should proceed to emit the XML body, else a finalized status.
358 */
359static ngx_int_t
360webdav_lock_refresh(ngx_http_request_t *r,
361 ngx_http_brix_webdav_loc_conf_t *conf, const char *path,
362 webdav_lock_xattr_t *e)
363{
364 if (!webdav_lock_if_header_matches(r, e->token)) {
365 return NGX_HTTP_LOCKED;
366 }
367 e->expires = webdav_lock_parse_timeout(r, conf);
368 if (webdav_lock_xattr_write(r, path, e, XATTR_REPLACE) != NGX_OK) {
369 return NGX_HTTP_INTERNAL_SERVER_ERROR;
370 }
371 r->headers_out.status = NGX_HTTP_OK;
372 return NGX_OK;
373}

375/*
376 * webdav_lock_create — create a new lock atomically (RFC 4918 §9.10).
377 *
378 * WHAT: Builds a fresh lock record (new token, owner, scope, depth, timeout,
379 * lock-null flag) and writes it with XATTR_CREATE.
380 * WHY: Splits the "no existing lock" branch out of the handler. HOW: delegates
381 * owner selection and write-result mapping to helpers; sets 201 Created on
382 * success. Returns NGX_OK when the caller should emit the XML body, else a
383 * finalized status.
384 */
385static ngx_int_t
386webdav_lock_create(ngx_http_request_t *r,
387 ngx_http_brix_webdav_loc_conf_t *conf, const char *path,
388 const webdav_lock_new_t *nl, webdav_lock_xattr_t *e)
389{
390 ngx_http_brix_webdav_req_ctx_t *ctx =
391 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
392 ngx_int_t rc;
393
394 ngx_memzero(e, sizeof(*e));
395 webdav_generate_uuid(e->token, sizeof(e->token));
396 webdav_lock_owner_set(e, nl->owner, ctx);
397
398 e->exclusive = nl->exclusive;
399 e->depth_infinity = nl->depth_infinity;
400 e->is_null = nl->created_null;
401 e->expires = webdav_lock_parse_timeout(r, conf);
402
403 rc = webdav_lock_xattr_write(r, path, e, XATTR_CREATE);
404 rc = webdav_lock_write_status(r, conf, rc);
405 if (rc != NGX_OK) {
406 return rc;
407 }
408
409 r->headers_out.status = NGX_HTTP_CREATED;
410 return NGX_OK;
411}

413static void
414webdav_handle_lock_inner(ngx_http_request_t *r)
415{
416 ngx_http_brix_webdav_loc_conf_t *conf;
417 char path[WEBDAV_MAX_PATH];
418 ngx_int_t rc;
419 webdav_lock_xattr_t e;
420 webdav_lock_new_t nl;
421 char owner[256];
422
423 ngx_memzero(&nl, sizeof(nl));
424 nl.depth_infinity = 1;
425 nl.exclusive = 1;
426 nl.owner = owner;
427
428 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
429
430 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
431 path, sizeof(path));
432 if (rc != NGX_OK) {
433 webdav_metrics_finalize_request(r, rc);
434 return;
435 }
436
437 rc = webdav_lock_parse_depth(r, &nl.depth_infinity);
438 if (rc != NGX_OK) {
439 webdav_metrics_finalize_request(r, rc);
440 return;
441 }
442
443 webdav_lock_parse_body(r, owner, sizeof(owner), &nl.exclusive);
444
445 /* RFC 4918 §9.10.1: LOCK on non-existent resource MUST create zero-byte resource. */
446 rc = webdav_lock_create_null(r, path, &nl.created_null);
447 if (rc != NGX_OK) {
448 webdav_metrics_finalize_request(r, rc);
449 return;
450 }
451
452 rc = webdav_lock_xattr_read(r, path, &e);
453 if (rc == NGX_ERROR) {
454 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
455 return;
456 }
457
458 /* Treat expired lock as absent. */
459 if (rc == NGX_OK && e.expires <= (int64_t) ngx_time()) {
460 (void) webdav_lock_xattr_delete(r, path);
461 rc = NGX_DECLINED;
462 }
463
464 if (rc == NGX_OK) {
465 /* Existing active lock: refresh if token matches, else 423. */
466 rc = webdav_lock_refresh(r, conf, path, &e);
467 } else {
468 /* No lock: create atomically with XATTR_CREATE. */
469 rc = webdav_lock_create(r, conf, path, &nl, &e);
470 }
471
472 if (rc != NGX_OK) {
473 webdav_metrics_finalize_request(r, rc);
474 return;
475 }
476
477 webdav_lock_xml_response(r, &e);
478}
