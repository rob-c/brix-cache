1/*
2 * lock.c — WebDAV LOCK and UNLOCK handler (RFC 4918 §9.10, §9.11).
3 *
4 * Lock state is stored as a single xattr on the locked resource
5 * (see prop_xattr.c). No shared memory or mutex is needed:
6 * XATTR_CREATE provides kernel-serialized atomic lock creation;
7 * expiry cleanup is lazy (on next access to that path).
8 *
9 * Lock check walks from the target path up to the export root,
10 * O(path_depth) xattr reads rather than O(1024) SHM slot scans.
11 */

34/*
35 * Generate Version 4 UUID for WebDAV lock token (`opaquelocktoken:` prefix).
36 * OpenSSL RAND_bytes provides collision-resistant identifiers.
37 */
38static void
39webdav_generate_uuid(char *buf, size_t bufsz)
40{
41 u_char bytes[16];
42 (void) RAND_bytes(bytes, 16);
43 bytes[6] = (bytes[6] & 0x0f) | 0x40; /* Version 4 */
44 bytes[8] = (bytes[8] & 0x3f) | 0x80; /* Variant */
45
46 snprintf(buf, bufsz,
47 "opaquelocktoken:%02x%02x%02x%02x-%02x%02x-%02x%02x"
48 "-%02x%02x-%02x%02x%02x%02x%02x%02x",
49 bytes[0], bytes[1], bytes[2], bytes[3],
50 bytes[4], bytes[5],
51 bytes[6], bytes[7],
52 bytes[8], bytes[9],
53 bytes[10], bytes[11], bytes[12], bytes[13], bytes[14], bytes[15]);
54}

56/*
57 * Startup lock sweep — remove every persisted lock xattr under the export
58 * root. Locks are stored as the WEBDAV_LOCK_XATTR_KEY xattr on each resource
59 * (prop_xattr.c), so they normally survive an nginx restart. When
60 * brix_webdav_lock_startup_sweep is on, this walk restores ephemeral
61 * RFC 4918 §10.1 semantics by clearing them once at startup.
62 *
63 * removexattr is idempotent: a node with no lock returns ENODATA/ENOATTR,
64 * which is not counted and not an error. The walk continues past per-node
65 * failures so one unreadable subtree cannot abort the sweep.
66 */
67/* Sweep state threaded to the walk callback so each node can build a transient,
68 * confined VFS ctx for its lock-xattr removal (the sweep runs at config-merge
69 * time, so it carries its own pool/log/root_canon rather than a request ctx). */
70typedef struct {
71 ngx_uint_t removed;
72 ngx_pool_t *pool;
73 ngx_log_t *log;
74 const char *root_canon;
75} webdav_lock_sweep_ctx_t;
76
77/* Remove the lock xattr on one absolute path through the VFS seam. removexattr
78 * via the VFS is idempotent for our purposes: ENODATA/ENOATTR (no lock) returns
79 * NGX_ERROR but is simply not counted, so a node without a lock is a no-op. */
80static void
81webdav_lock_sweep_remove(webdav_lock_sweep_ctx_t *sw, const char *path)
82{
83 brix_vfs_ctx_t vctx;
84
85 brix_vfs_ctx_init(&vctx, sw->pool, sw->log, BRIX_PROTO_WEBDAV,
86 sw->root_canon, NULL, 1 /* allow_write */, 0 /* is_tls */, NULL, path);
87
88 if (brix_vfs_removexattr(&vctx, WEBDAV_LOCK_XATTR_KEY) == NGX_OK) {
89 sw->removed++;
90 }
91}
92
93static ngx_int_t
94webdav_lock_sweep_cb(const brix_fs_walk_entry_t *entry, void *data)
95{
96 webdav_lock_sweep_remove(data, entry->path);
97 return NGX_OK; /* continue regardless of per-node result */
98}

100ngx_uint_t
101webdav_lock_startup_sweep(ngx_pool_t *pool, ngx_log_t *log,
102 const char *root_canon)
103{
104 brix_fs_walk_options_t opts;
105 webdav_lock_sweep_ctx_t sw;
106
107 if (root_canon == NULL || root_canon[0] == '\0') {
108 return 0;
109 }
110
111 ngx_memzero(&sw, sizeof(sw));
112 sw.pool = pool;
113 sw.log = log;
114 sw.root_canon = root_canon;
115
116 /* The tree walk visits children only; clear a lock on the root itself. */
117 webdav_lock_sweep_remove(&sw, root_canon);
118
119 ngx_memzero(&opts, sizeof(opts));
120 opts.include_files = 1; /* locks may sit on files ... */
121 opts.include_dirs = 1; /* ... and on collections (Depth: infinity) */
122
123 (void) brix_fs_walk(log, root_canon, &opts, webdav_lock_sweep_cb, &sw);
124 return sw.removed;
125}

127/*
128 * Build and send the RFC 4918 §9.10 LOCK response body (activelock XML)
129 * plus Lock-Token response header.
130 */
131static void
132webdav_lock_xml_response(ngx_http_request_t *r, webdav_lock_xattr_t *e)
133{
134 ngx_chain_t *head = NULL, *tail = NULL;
135 char timeout_buf[32];
136 char *safe_owner;
137 int64_t now = (int64_t) ngx_time();
138 ngx_uint_t remaining;
139 off_t total_len = 0;
140 ngx_chain_t *lc;
141 ngx_table_elt_t *h;
142
143 /* expires is absolute wall-clock seconds → remaining is a plain subtraction. */
144 remaining = (e->expires > now) ? (ngx_uint_t) (e->expires - now) : 0;
145 ngx_sprintf((u_char *) timeout_buf, "Second-%ui", remaining);
146
147 safe_owner = webdav_escape_xml_text(r->pool, e->owner);
148 if (safe_owner == NULL) {
149 safe_owner = "anonymous";
150 }
151
152 brix_http_chain_appendf(r->pool, &head, &tail,
153 "<?xml version=\"1.0\" encoding=\"utf-8\" ?>"
154 "<D:prop xmlns:D=\"DAV:\">"
155 "<D:lockdiscovery>"
156 "<D:activelock>"
157 "<D:locktype><D:write/></D:locktype>"
158 "<D:lockscope>%s</D:lockscope>"
159 "<D:depth>%s</D:depth>"
160 "<D:owner>%s</D:owner>"
161 "<D:timeout>%s</D:timeout>"
162 "<D:locktoken><D:href>%s</D:href></D:locktoken>"
163 "</D:activelock>"
164 "</D:lockdiscovery>"
165 "</D:prop>",
166 e->exclusive ? "<D:exclusive/>" : "<D:shared/>",
167 e->depth_infinity ? "infinity" : "0",
168 safe_owner, timeout_buf, e->token);
169
170 if (tail != NULL) {
171 tail->buf->last_buf = 1;
172 tail->buf->last_in_chain = 1;
173 }
174
175 for (lc = head; lc != NULL; lc = lc->next) {
176 total_len += lc->buf->last - lc->buf->pos;
177 }
178
179 r->headers_out.content_length_n = total_len;
180
181 h = ngx_list_push(&r->headers_out.headers);
182 if (h != NULL) {
183 h->hash = 1;
184 ngx_str_set(&h->key, "Content-Type");
185 ngx_str_set(&h->value, "application/xml; charset=\"utf-8\"");
186 }
187
188 h = ngx_list_push(&r->headers_out.headers);
189 if (h != NULL) {
190 h->hash = 1;
191 ngx_str_set(&h->key, "Lock-Token");
192 h->value.len = strlen(e->token) + 2;
193 h->value.data = ngx_pnalloc(r->pool, h->value.len);
194 if (h->value.data != NULL) {
195 ngx_sprintf(h->value.data, "<%s>", e->token);
196 }
197 }
198
199 ngx_http_send_header(r);
200 ngx_http_output_filter(r, head);
201}

203/*
204 * webdav_lock_reap_null — release a lock-null placeholder.
205 *
206 * WHAT: If `e` recorded a lock-null lock (the LOCK created a zero-byte resource
207 * to reserve a name, RFC 4918 §9.10.1) AND the resource is still an empty
208 * regular file, unlink it so the reserved name disappears with the lock.
209 * WHY: Without this, UNLOCK (or expiry) of a never-PUT lock-null resource would
210 * leave an orphaned 0-byte file behind. A resource the client PUT data into
211 * is non-empty and is never reaped, so the is_null flag becomes inert once
212 * real content exists — no need to clear it on PUT.
213 * HOW: Confined no-follow lstat; only a regular, zero-length file is unlinked,
214 * via the confined unlink. Best-effort — failures are ignored. MUST NOT be
215 * called on a path that is about to be re-locked (it would remove the file
216 * out from under a follow-up XATTR_CREATE).
217 */
218/* Build a confined VFS ctx for a lock-DB namespace op on `path` (mirrors the
219 * canonical webdav construction). Identity comes from the request ctx so the op
220 * runs as the mapped user under impersonation. */
221static void
222webdav_lock_vfs_ctx(ngx_http_request_t *r, const char *path,
223 brix_vfs_ctx_t *vctx)
224{
225 ngx_http_brix_webdav_loc_conf_t *conf =
226 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
227 ngx_http_brix_webdav_req_ctx_t *rx =
228 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
229 int is_tls = 0;
230
231#if (NGX_HTTP_SSL)
232 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
233#endif
234
235 brix_vfs_ctx_init(vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
236 conf->common.root_canon, conf->cache_root_canon, conf->common.allow_write,
237 is_tls, (rx != NULL) ? rx->identity : NULL, path);
238}

240static void
241webdav_lock_reap_null(ngx_http_request_t *r, const char *path,
242 const webdav_lock_xattr_t *e)
243{
244 brix_vfs_ctx_t vctx;
245 brix_vfs_stat_t vst;
246
247 if (!e->is_null) {
248 return;
249 }
250
251 /* Confined no-follow probe; only a regular, zero-length file is unlinked,
252 * through the VFS. Best-effort — failures are ignored. */
253 webdav_lock_vfs_ctx(r, path, &vctx);
254 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) == NGX_OK
255 && vst.is_regular && vst.size == 0)
256 {
257 (void) brix_vfs_unlink(&vctx);
258 }
259}

261/*
262 * webdav_check_locks — walk from path up to export root, checking for active
263 * xattr locks at each level. O(path_depth) xattr reads.
264 *
265 * Returns NGX_HTTP_LOCKED if an active, client-unowned lock blocks this
266 * operation; NGX_OK otherwise.
267 */
268ngx_int_t
269webdav_check_locks(ngx_http_request_t *r, const char *path, int need_write)
270{
271 ngx_http_brix_webdav_loc_conf_t *conf;
272 char check[PATH_MAX];
273 char *slash;
274 size_t root_len, check_len, path_len;
275 webdav_lock_xattr_t e;
276 ngx_int_t rc;
277
278 (void) need_write;
279
280 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
281 root_len = strlen(conf->common.root_canon);
282 path_len = strlen(path);
283
284 ngx_cpystrn((u_char *) check, (u_char *) path, sizeof(check));
285
286 for (;;) {
287 check_len = strlen(check);
288
289 rc = webdav_lock_xattr_read(r, check, &e);
290 if (rc == NGX_ERROR) {
291 return NGX_HTTP_INTERNAL_SERVER_ERROR;
292 }
293
294 if (rc == NGX_OK) {
295 if (e.expires > (int64_t) ngx_time()) {
296 /* Lock covers path if: exact match or depth-infinity ancestor. */
297 int covers = (check_len == path_len)
298 || (e.depth_infinity
299 && check_len < path_len
300 && (check[check_len - 1] == '/'
301 || path[check_len] == '/'));
302 if (covers && !webdav_lock_if_header_matches(r, e.token)) {
303 return NGX_HTTP_LOCKED;
304 }
305 } else {
306 (void) webdav_lock_xattr_delete(r, check);
307 webdav_lock_reap_null(r, check, &e);
308 }
309 }
310
311 /* Stop at or above export root. */
312 if (check_len <= root_len) {
313 break;
314 }
315
316 /* Strip trailing slashes then the last path component. */
317 slash = check + check_len - 1;
318 while (slash > check && *slash == '/') {
319 slash--;
320 }
321 while (slash > check && *slash != '/') {
322 slash--;
323 }
324 if (*slash != '/') {
325 break;
326 }
327 *slash = '\0';
328
329 if (strlen(check) < root_len) {
330 break;
331 }
332 }
333
334 return NGX_OK;
335}

337/*
338 * check_locks_descendants — scan direct and recursive children of a directory
339 * for unexpired xattr locks not covered by the client's If header.
340 * Does NOT walk upward; the caller is responsible for checking ancestors.
341 */
342static ngx_int_t
343check_locks_descendants(ngx_http_request_t *r, const char *dir)
344{
345 brix_vfs_ctx_t vctx;
346 brix_vfs_dir_t *dp;
347 char child[PATH_MAX];
348 size_t dir_len;
349 webdav_lock_xattr_t e;
350 ngx_int_t rc;
351
352 /* Confined NON-metered opendir (recursive lock scan; a planted in-export
353 * symlink cannot redirect it out of the export root). */
354 webdav_lock_vfs_ctx(r, dir, &vctx);
355 dp = brix_vfs_opendir_quiet(&vctx, NULL);
356 if (dp == NULL) {
357 return NGX_OK;
358 }
359
360 dir_len = strlen(dir);
361 rc = NGX_OK;
362
363 for ( ;; ) {
364 ngx_str_t name;
365 brix_vfs_dirent_kind_t dkind;
366 const char *dname;
367 int is_dir;
368 ngx_int_t xrc;
369
370 /* "."/".." filtered by readdir_kind; kind from d_type. */
371 if (brix_vfs_readdir_kind(dp, &name, &dkind) != NGX_OK) {
372 break; /* NGX_DONE (end) or error → stop */
373 }
374 dname = (const char *) name.data;
375
376 if (dir_len + 1 + name.len + 1 > sizeof(child)) {
377 continue; /* path too long — skip */
378 }
379 snprintf(child, sizeof(child), "%s/%s", dir, dname);
380
381 /* Check lock xattr directly on this child. */
382 xrc = webdav_lock_xattr_read(r, child, &e);
383 if (xrc == NGX_ERROR) {
384 rc = NGX_HTTP_INTERNAL_SERVER_ERROR;
385 break;
386 }
387
388 if (xrc == NGX_OK) {
389 if (e.expires > (int64_t) ngx_time()) {
390 if (!webdav_lock_if_header_matches(r, e.token)) {
391 rc = NGX_HTTP_LOCKED;
392 break;
393 }
394 } else {
395 (void) webdav_lock_xattr_delete(r, child);
396 webdav_lock_reap_null(r, child, &e);
397 }
398 }
399 /* xrc == NGX_DECLINED: no lock xattr on this child — OK */
400
401 /* Recurse into subdirectories. d_type gives the answer directly; on a
402 * DT_UNKNOWN filesystem fall back to a confined no-follow probe so a
403 * trailing symlink is never followed into recursion. */
404 is_dir = (dkind == BRIX_VFS_DT_DIR);
405 if (dkind == BRIX_VFS_DT_UNKNOWN) {
406 brix_vfs_ctx_t cctx;
407 brix_vfs_stat_t cvst;
408
409 webdav_lock_vfs_ctx(r, child, &cctx);
410 is_dir = (brix_vfs_probe(&cctx, 1 /* no-follow */, &cvst) == NGX_OK
411 && cvst.is_directory);
412 }
413 if (is_dir) {
414 rc = check_locks_descendants(r, child);
415 if (rc != NGX_OK) {
416 break;
417 }
418 }
419 }
420
421 brix_vfs_closedir(dp, r->connection->log);
422 return rc;
423}

425/*
426 * webdav_check_locks_tree — check locks on a path and all its descendants.
427 * Use this instead of webdav_check_locks() for DELETE/COPY/MOVE on
428 * collections, where a lock on any child must block the operation.
429 */
430ngx_int_t
431webdav_check_locks_tree(ngx_http_request_t *r, const char *path)
432{
433 brix_vfs_ctx_t vctx;
434 brix_vfs_stat_t vst;
435 ngx_int_t rc;
436
437 rc = webdav_check_locks(r, path, 1);
438 if (rc != NGX_OK) {
439 return rc;
440 }
441
442 /* Only recurse for a directory (confined no-follow probe). */
443 webdav_lock_vfs_ctx(r, path, &vctx);
444 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) == NGX_OK
445 && vst.is_directory)
446 {
447 rc = check_locks_descendants(r, path);
448 }
449
450 return rc;
451}

455/*

469 brix_imp_request_begin(rx != NULL ? rx->identity : NULL);
470 webdav_handle_lock_inner(r);

474static void
475webdav_handle_lock_inner(ngx_http_request_t *r)
476{
477 ngx_http_brix_webdav_loc_conf_t *conf;
478 char path[WEBDAV_MAX_PATH];
479 ngx_int_t rc;
480 webdav_lock_xattr_t e;
481 ngx_http_brix_webdav_req_ctx_t *ctx;
482 int depth_infinity = 1;
483 char owner[256];
484 int exclusive = 1;
485 int created_null = 0; /* lock-null placeholder */
486
487 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
488
489 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
490 path, sizeof(path));
491 if (rc != NGX_OK) {
492 webdav_metrics_finalize_request(r, rc);
493 return;
494 }
495
496 rc = webdav_lock_parse_depth(r, &depth_infinity);
497 if (rc != NGX_OK) {
498 webdav_metrics_finalize_request(r, rc);
499 return;
500 }
501
502 webdav_lock_parse_body(r, owner, sizeof(owner), &exclusive);
503
504 /* RFC 4918 §9.10.1: LOCK on non-existent resource MUST create zero-byte resource. */
505 {
506 brix_vfs_ctx_t vctx;
507 brix_vfs_stat_t vst;
508
509 webdav_lock_vfs_ctx(r, path, &vctx);
510 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) == NGX_DECLINED
511 && errno == ENOENT)
512 {
513 int verr = 0;
514 brix_vfs_file_t *fh = brix_vfs_open(&vctx,
515 BRIX_VFS_O_WRITE | BRIX_VFS_O_CREATE | BRIX_VFS_O_EXCL,
516 &verr);
517 if (fh == NULL && verr != EEXIST) {
518 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
519 return;
520 }
521 if (fh != NULL) {
522 brix_vfs_close(fh, r->connection->log);
523 /* We created the resource solely to host this lock: it is a
524 * lock-null placeholder, reaped on UNLOCK/expiry if still empty. */
525 created_null = 1;
526 }
527 }
528 }
529
530 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
531
532 rc = webdav_lock_xattr_read(r, path, &e);
533 if (rc == NGX_ERROR) {
534 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
535 return;
536 }
537
538 /* Treat expired lock as absent. */
539 if (rc == NGX_OK && e.expires <= (int64_t) ngx_time()) {
540 (void) webdav_lock_xattr_delete(r, path);
541 rc = NGX_DECLINED;
542 }
543
544 if (rc == NGX_OK) {
545 /* Existing active lock: refresh if token matches, else 423. */
546 if (!webdav_lock_if_header_matches(r, e.token)) {
547 webdav_metrics_finalize_request(r, NGX_HTTP_LOCKED);
548 return;
549 }
550 e.expires = webdav_lock_parse_timeout(r, conf);
551 if (webdav_lock_xattr_write(r, path, &e,
552 XATTR_REPLACE) != NGX_OK) {
553 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
554 return;
555 }
556 r->headers_out.status = NGX_HTTP_OK;
557
558 } else {
559 /* No lock: create atomically with XATTR_CREATE. */
560 ngx_memzero(&e, sizeof(e));
561 webdav_generate_uuid(e.token, sizeof(e.token));
562
563 if (owner[0] != '\0') {
564 ngx_cpystrn((u_char *) e.owner, (u_char *) owner, sizeof(e.owner));
565 } else if (ctx != NULL && ctx->dn[0] != '\0') {
566 ngx_cpystrn((u_char *) e.owner, (u_char *) ctx->dn, sizeof(e.owner));
567 } else {
568 ngx_cpystrn((u_char *) e.owner, (u_char *) "anonymous",
569 sizeof(e.owner));
570 }
571
572 e.exclusive = exclusive;
573 e.depth_infinity = depth_infinity;
574 e.is_null = created_null;
575 e.expires = webdav_lock_parse_timeout(r, conf);
576
577 rc = webdav_lock_xattr_write(r, path, &e, XATTR_CREATE);
578 if (rc == NGX_DECLINED) {
579 /* Another worker created the lock between our read and write. */
580 webdav_metrics_finalize_request(r, NGX_HTTP_LOCKED);
581 return;
582 }
583 if (rc != NGX_OK) {
584 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
585 return;
586 }
587
588 r->headers_out.status = NGX_HTTP_CREATED;
589 }
590
591 webdav_lock_xml_response(r, &e);
592}

594ngx_int_t
595webdav_handle_unlock(ngx_http_request_t *r)
596{
597 ngx_http_brix_webdav_loc_conf_t *conf;
598 char path[WEBDAV_MAX_PATH];
599 ngx_int_t rc;
600 ngx_table_elt_t *h;
601 webdav_lock_xattr_t e;
602 const u_char *lock_val;
603 size_t lock_len, token_len;
604
605 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
606
607 h = webdav_tpc_find_header(r, "Lock-Token", sizeof("Lock-Token") - 1);
608 if (h == NULL) {
609 return NGX_HTTP_BAD_REQUEST;
610 }
611
612 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
613 path, sizeof(path));
614 if (rc != NGX_OK) {
615 return rc;
616 }
617
618 rc = webdav_lock_xattr_read(r, path, &e);
619 if (rc == NGX_ERROR) {
620 return NGX_HTTP_INTERNAL_SERVER_ERROR;
621 }
622 if (rc == NGX_DECLINED || e.expires <= (int64_t) ngx_time()) {
623 /* No active lock on this path. */
624 if (rc == NGX_OK) {
625 (void) webdav_lock_xattr_delete(r, path);
626 webdav_lock_reap_null(r, path, &e);
627 }
628 return NGX_HTTP_CONFLICT;
629 }
630
631 /* Strip angle brackets from Lock-Token header: <opaquelocktoken:...> → token */
632 lock_val = h->value.data;
633 lock_len = (size_t) h->value.len;
634 if (lock_len > 0 && lock_val[0] == '<') {
635 lock_val++;
636 lock_len--;
637 if (lock_len > 0 && lock_val[lock_len - 1] == '>') {
638 lock_len--;
639 }
640 }
641
642 token_len = strlen(e.token);
643 if (lock_len != token_len
644 || CRYPTO_memcmp(lock_val, (u_char *) e.token, token_len) != 0)
645 {
646 return NGX_HTTP_CONFLICT; /* token mismatch — not our lock */
647 }
648
649 /*
650 * Remove the lock xattr and HONOUR the result. Under impersonation the
651 * removexattr runs (via the broker) as the requester's mapped user, so a
652 * NON-OWNER who presents a valid (e.g. stolen/leaked) lock token cannot
653 * actually clear the lock on another user's file — the broker returns EACCES.
654 * Returning 204 unconditionally would falsely report success for that denied
655 * removal; surface it as 403 instead so the lock state and the response agree.
656 * (ENODATA/ENOENT collapse to NGX_OK inside the helper — idempotent unlock.)
657 */
658 if (webdav_lock_xattr_delete(r, path) != NGX_OK) {
659 return NGX_HTTP_FORBIDDEN;
660 }
661
662 /* RFC 4918 §9.10.1: a lock-null resource that was never converted by a PUT/
663 * MKCOL disappears when its lock is removed. Reap the empty placeholder. */
664 webdav_lock_reap_null(r, path, &e);
665
666 return webdav_send_no_body(r, NGX_HTTP_NO_CONTENT);
667}

695/*
696 * webdav_lock_append_discovery — emit <D:lockdiscovery> XML for PROPFIND.
697 * Reads the xattr on path; emits an empty element if no active lock.
698 */
699ngx_int_t
700webdav_lock_append_discovery(ngx_http_request_t *r, const char *path,
701 ngx_chain_t **head, ngx_chain_t **tail)
702{
703 webdav_lock_xattr_t e;
704 ngx_int_t rc;
705 int64_t now;
706 ngx_uint_t remaining;
707 char *safe_owner;
708
709 if (brix_http_chain_appendf(r->pool, head, tail,
710 "<D:lockdiscovery>") == NULL)
711 {
712 return NGX_ERROR;
713 }
714
715 rc = webdav_lock_xattr_read(r, path, &e);
716 if (rc == NGX_ERROR) {
717 return NGX_ERROR;
718 }
719
720 now = (int64_t) ngx_time();
721
722 if (rc == NGX_OK && e.expires > now) {
723 remaining = (ngx_uint_t) (e.expires - now);
724 safe_owner = webdav_escape_xml_text(r->pool,
725 (e.owner[0] != '\0') ? e.owner : "anonymous");
726 if (safe_owner == NULL) {
727 safe_owner = "anonymous";
728 }
729
730 if (brix_http_chain_appendf(r->pool, head, tail,
731 "<D:activelock>"
732 "<D:locktype><D:write/></D:locktype>"
733 "<D:lockscope>%s</D:lockscope>"
734 "<D:depth>%s</D:depth>"
735 "<D:owner>%s</D:owner>"
736 "<D:timeout>Second-%lu</D:timeout>"
737 "<D:locktoken>"
738 "<D:href>%s</D:href>"
739 "</D:locktoken>"
740 "</D:activelock>",
741 e.exclusive ? "<D:exclusive/>" : "<D:shared/>",
742 e.depth_infinity ? "infinity" : "0",
743 safe_owner, (unsigned long) remaining, e.token) == NULL)
744 {
745 return NGX_ERROR;
746 }
747 } else if (rc == NGX_OK) {
748 /* Expired lock — clean up lazily (and reap a lock-null placeholder). */
749 (void) webdav_lock_xattr_delete(r, path);
750 webdav_lock_reap_null(r, path, &e);
751 }
752
753 if (brix_http_chain_appendf(r->pool, head, tail,
754 "</D:lockdiscovery>") == NULL)
755 {
756 return NGX_ERROR;
757 }
758
759 return NGX_OK;
760}
