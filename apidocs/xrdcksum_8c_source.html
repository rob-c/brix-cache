1/*
2 * xrdcksum.c — multi-call front-end for the checksum tool family.
3 *
4 * WHAT: One binary serving xrdcrc32c / xrdcrc64 / xrdadler32 / xrdckverify /
5 * xrdcinfo (via argv[0] symlinks, stock-name compatible) and the
6 * subcommand form `xrdcksum <crc32c|crc64|adler32|verify|info> …`.
7 * WHY: Five installed binaries collapse into one link target — one binary to
8 * audit and ship; per-tool behavior, options and exit codes are
9 * unchanged (the checksum personalities keep their stock error exits:
10 * xrdcrc32c=3, xrdcrc64/xrdadler32=1).
11 * HOW: basename(argv[0]) picks the personality; the bare `xrdcksum` name
12 * shifts argv by one and dispatches on the subcommand so each
13 * personality still sees its own name at argv[0]. Checksum
14 * personalities delegate to brix_cli_cksum_main() exactly as the old
15 * thin wrappers did; verify/info call the renamed tool mains.
16 */

54/* Run the personality named `name`; -1 = no such personality. */
55static int
56dispatch(const char *name, int argc, char **argv)
57{
58 size_t i;
59
60 for (i = 0; i < sizeof(CKSUM_TOOLS) / sizeof(CKSUM_TOOLS[0]); i++) {
61 const cksum_tool *t = &CKSUM_TOOLS[i];
62
63 if (strcmp(name, t->name) == 0) {
64 return brix_cli_cksum_main(argv[0], t->algo_name, t->algo,
65 argc == 2 ? argv[1] : NULL,
66 t->err_exit);
67 }
68 }
69 if (strcmp(name, "xrdckverify") == 0 || strcmp(name, "verify") == 0) {
70 return brix_xrdckverify_main(argc, argv);
71 }
72 if (strcmp(name, "xrdcinfo") == 0 || strcmp(name, "info") == 0) {
73 return brix_xrdcinfo_main(argc, argv);
74 }
75 if (strcmp(name, "xrdcktree") == 0 || strcmp(name, "tree") == 0) {
76 return brix_xrdcktree_main(argc, argv);
77 }
78 if (strcmp(name, "xrdckcheck") == 0 || strcmp(name, "check") == 0) {
79 return brix_xrdckcheck_main(argc, argv);
80 }
81 return -1;
82}

134 if (suggestion != NULL && arg != NULL) {

137 }
