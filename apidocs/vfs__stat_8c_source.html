1/*
2 * vfs_stat.c — VFS path stat.
3 *
4 * WHAT: Implements brix_vfs_stat(), which lstat()s the resolved ctx path and
5 * fills an brix_vfs_stat_t (size/mtime/ctime/mode/ino plus the
6 * is_directory / is_regular flags).
7 *
8 * WHY: kXR_stat / kXR_statx, WebDAV PROPFIND on a single resource, and S3 HEAD
9 * all need one confined, metered stat with consistent error mapping
10 * instead of each protocol calling stat(2) directly.
11 *
12 * HOW: Re-verifies confinement (and a non-NULL stat_out), uses lstat() so
13 * symlinks are reported rather than followed, converts via
14 * brix_vfs_fill_stat(), and emits an BRIX_METRIC_OP_STAT
15 * metric/access-log line on every path through
16 * brix_vfs_observe_ctx_op().
17 */
18#include "vfs_internal.h"
19#include "fs/backend/cache/sd_cache.h"
20#include "fs/backend/stage/sd_stage.h"

22/*
23 * Shared confined-stat body for both the lstat (no-follow) and stat (follow)
24 * entrypoints. nofollow!=0 reports a trailing symlink as itself; nofollow=0
25 * follows it chroot-style within the export. Both run AS THE MAPPED USER under
26 * impersonation (broker-routed) — otherwise a metadata op (WebDAV
27 * HEAD/DELETE/PROPFIND-on-file, kXR_stat/statx, S3 HEAD) whose target sits
28 * inside a directory the unprivileged worker cannot traverse (a 0700 user-owned
29 * subdir, or a group-restricted 0770 dir the mapped user reaches only via a
30 * supplementary group) would EACCES and fail (500) for the legitimate
31 * owner/group-member. Off impersonation this is the same bare lstat/stat.
32 */
33static ngx_int_t
34brix_vfs_stat_impl(brix_vfs_ctx_t *ctx, brix_vfs_stat_t *stat_out,
35 int nofollow)
36{
37 struct stat st;
38 const char *path;
39 uint64_t start;
40 int saved_errno;
41 const brix_sd_driver_t *drv;
42
43 start = brix_vfs_now_ns();
44 path = brix_vfs_ctx_path(ctx);
45
46 if (stat_out == NULL || brix_vfs_require_confined(ctx) != NGX_OK) {
47 errno = EINVAL;
48 saved_errno = errno;
49 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
50 NGX_ERROR, saved_errno, start);
51 return NGX_ERROR;
52 }
53
54 /* Non-POSIX backend: the namespace lives in the driver, not the export tree.
55 * nofollow is moot (an object/block backend has no symlinks). */
56 drv = brix_vfs_ctx_driver(ctx);
57 if (drv != NULL) {
58 brix_sd_stat_t sd_st;
59
60 if (drv->stat == NULL
61 || drv->stat(ctx->sd, brix_vfs_export_relative(ctx, path), &sd_st)
62 != NGX_OK)
63 {
64 saved_errno = errno;
65 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
66 NGX_ERROR, saved_errno, start);
67 return NGX_ERROR;
68 }
69 brix_vfs_sd_stat_fill(&sd_st, stat_out);
70 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
71 NGX_OK, 0, start);
72 return NGX_OK;
73 }
74
75 if (brix_lstat_confined_canon(ctx->log, ctx->root_canon, path, &st,
76 nofollow) != 0)
77 {
78 saved_errno = errno;
79 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
80 NGX_ERROR, saved_errno, start);
81 return NGX_ERROR;
82 }
83
84 brix_vfs_fill_stat(&st, stat_out);
85 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
86 NGX_OK, 0, start);
87 return NGX_OK;
88}

90/* lstat the resolved ctx path into *stat_out (no symlink follow). Confined and
91 * metered as OP_STAT; NGX_ERROR with errno set on guard failure or lstat error. */
92ngx_int_t
93brix_vfs_stat(brix_vfs_ctx_t *ctx, brix_vfs_stat_t *stat_out)
94{
95 return brix_vfs_stat_impl(ctx, stat_out, 1 /* nofollow */);

107/* Descend one cache/stage decorator to its wrapped source (NULL at a leaf or a
108 * non-decorator instance). Both unwrap helpers self-guard, so this is safe on any
109 * instance and lets the residency seam find a nearline driver buried under a
110 * read-cache and/or write-stage tier. */
111static brix_sd_instance_t *
112brix_vfs_decorator_source(const brix_sd_instance_t *inst)
113{
114 brix_sd_instance_t *s = brix_sd_cache_source_instance(inst);
115
116 return (s != NULL) ? s : brix_sd_stage_source_instance(inst);
117}

119/* Classify the resolved ctx path's nearline residency (online/nearline/offline/
120 * lost) so protocol handlers can advertise tape state — the HTTP Tape REST API, S3
121 * InvalidObjectState / x-amz-storage-class, root:// stat's nearline flag — WITHOUT
122 * forcing a recall. Walks any cache/stage decorators down to the CAP_NEARLINE
123 * driver and reads its residency model; an export with no nearline tier is always
124 * ONLINE (a plain disk/object store is resident). Emits no metric (it is an
125 * internal classification the caller's own op accounts for). NGX_OK with *out set,
126 * or NGX_ERROR (errno) on a guard failure or a driver error. */
127ngx_int_t
128brix_vfs_residency(brix_vfs_ctx_t *ctx, brix_sd_residency_t *out,
129 int *nearline_export)
130{
131 brix_sd_instance_t *inst;
132
133 if (nearline_export != NULL) {
134 *nearline_export = 0;
135 }
136 if (out == NULL || brix_vfs_require_confined(ctx) != NGX_OK) {
137 errno = EINVAL;
138 return NGX_ERROR;
139 }
140
141 *out = BRIX_SD_RES_ONLINE;
142
143 for (inst = ctx->sd; inst != NULL;
144 inst = brix_vfs_decorator_source(inst))
145 {
146 if ((brix_sd_caps(inst) & BRIX_SD_CAP_NEARLINE) != 0
147 && inst->driver->residency != NULL)
148 {
149 const char *path = brix_vfs_ctx_path(ctx);
150
151 if (nearline_export != NULL) {
152 *nearline_export = 1;
153 }
154 return inst->driver->residency(
155 inst, brix_vfs_export_relative(ctx, path), out);
156 }
157 }
158 return NGX_OK;
159}

161/*
162 * brix_vfs_probe — confined existence/type probe for pre-op resolution and
163 * ACL gates. Unlike brix_vfs_stat/statf this emits NO OP_STAT metric or
164 * access-log line: it is an internal namespace pre-check that the caller's own
165 * operation accounts for (routing it through the metered stat would record a
166 * phantom STAT for every rm/chmod/mkdir/mv). nofollow selects lstat vs stat
167 * semantics. Returns NGX_OK with *stat_out filled when the path is present,
168 * NGX_DECLINED when it is absent (errno preserved from the underlying stat), or
169 * NGX_ERROR on a confinement-guard failure.
170 */
171ngx_int_t
172brix_vfs_probe(brix_vfs_ctx_t *ctx, int nofollow,
173 brix_vfs_stat_t *stat_out)
174{
175 struct stat st;
176 const brix_sd_driver_t *drv;
177
178 if (stat_out == NULL || brix_vfs_require_confined(ctx) != NGX_OK) {
179 errno = EINVAL;
180 return NGX_ERROR;
181 }
182
183 drv = brix_vfs_ctx_driver(ctx);
184 if (drv != NULL) {
185 brix_sd_stat_t sd_st;
186
187 if (drv->stat == NULL
188 || drv->stat(ctx->sd,
189 brix_vfs_export_relative(ctx, brix_vfs_ctx_path(ctx)),
190 &sd_st) != NGX_OK)
191 {
192 return NGX_DECLINED; /* absent (or unsupported) — caller's errno */
193 }
194 brix_vfs_sd_stat_fill(&sd_st, stat_out);
195 return NGX_OK;
196 }
197
198 if (brix_lstat_confined_canon(ctx->log, ctx->root_canon,
199 brix_vfs_ctx_path(ctx), &st,
200 nofollow) != 0)
201 {
202 return NGX_DECLINED;
203 }
204
205 brix_vfs_fill_stat(&st, stat_out);
206 return NGX_OK;
207}
