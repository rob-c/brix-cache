1/*
2 * vfs_stat.c — VFS path stat.
3 *
4 * WHAT: Implements brix_vfs_stat(), which lstat()s the resolved ctx path and
5 * fills an brix_vfs_stat_t (size/mtime/ctime/mode/ino plus the
6 * is_directory / is_regular flags).
7 *
8 * WHY: kXR_stat / kXR_statx, WebDAV PROPFIND on a single resource, and S3 HEAD
9 * all need one confined, metered stat with consistent error mapping
10 * instead of each protocol calling stat(2) directly.
11 *
12 * HOW: Re-verifies confinement (and a non-NULL stat_out), uses lstat() so
13 * symlinks are reported rather than followed, converts via
14 * brix_vfs_fill_stat(), and emits an BRIX_METRIC_OP_STAT
15 * metric/access-log line on every path through
16 * brix_vfs_observe_ctx_op().
17 */
18#include "vfs_internal.h"
19#include "fs/backend/cache/sd_cache.h"
20#include "fs/backend/stage/sd_stage.h"

22/*
23 * Shared confined-stat body for both the lstat (no-follow) and stat (follow)
24 * entrypoints. nofollow!=0 reports a trailing symlink as itself; nofollow=0
25 * follows it chroot-style within the export. Both run AS THE MAPPED USER under
26 * impersonation (broker-routed) — otherwise a metadata op (WebDAV
27 * HEAD/DELETE/PROPFIND-on-file, kXR_stat/statx, S3 HEAD) whose target sits
28 * inside a directory the unprivileged worker cannot traverse (a 0700 user-owned
29 * subdir, or a group-restricted 0770 dir the mapped user reaches only via a
30 * supplementary group) would EACCES and fail (500) for the legitimate
31 * owner/group-member. Off impersonation this is the same bare lstat/stat.
32 */
33static ngx_int_t
34brix_vfs_stat_impl(brix_vfs_ctx_t *ctx, brix_vfs_stat_t *stat_out,
35 int nofollow)
36{
37 struct stat st;
38 const char *path;
39 uint64_t start;
40 int saved_errno;
41 const brix_sd_driver_t *drv;
42
43 start = brix_vfs_now_ns();
44 path = brix_vfs_ctx_path(ctx);
45
46 if (stat_out == NULL || brix_vfs_require_confined(ctx) != NGX_OK) {
47 errno = EINVAL;
48 saved_errno = errno;
49 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
50 NGX_ERROR, saved_errno, start);
51 return NGX_ERROR;
52 }
53
54 /* Non-POSIX backend: the namespace lives in the driver, not the export tree.
55 * nofollow is moot (an object/block backend has no symlinks). */
56 drv = brix_vfs_ctx_driver(ctx);
57 if (drv != NULL) {
58 brix_sd_ucred_t store;
59 brix_sd_cred_t cred;
60 brix_sd_stat_t sd_st;
61 int use_cred = 0, cred_err = 0;
62 ngx_int_t grc;
63
64 /* Zero the cred before the gate: the gate fills only the ACTIVE kind
65 * (an x509 proxy path OR a bearer OR an s3/ceph tuple) and leaves the
66 * inactive pointers untouched, so an unzeroed struct hands a garbage
67 * pointer to the driver's stat_cred slot (a bearer-only PASSTHROUGH
68 * cred left a live-stack x509_proxy that sd_xroot dereferenced → SIGSEGV
69 * on the two-hop token read). Mirrors the data-plane callers, which all
70 * memzero their ucred before brix_vfs_backend_cred. */
71 ngx_memzero(&cred, sizeof(cred));
72
73 if (brix_vfs_cred_gate_active(ctx)) {
74 grc = brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err);
75 if (grc != NGX_OK) {
76 saved_errno = cred_err ? cred_err : EACCES;
77 errno = saved_errno;
78 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL,
79 0, NGX_ERROR, saved_errno, start);
80 return NGX_ERROR;
81 }
82 }
83
84 /* Dispatch on the leaf instance so brix_sd_stat_maybe_cred finds the
85 * leaf driver's stat_cred slot (decorators have only plain relays). */
86 if (drv->stat == NULL
87 || brix_sd_stat_maybe_cred(brix_vfs_ns_leaf(ctx->sd),
88 brix_vfs_export_relative(ctx, path), &sd_st,
89 use_cred ? &cred : NULL) != NGX_OK)
90 {
91 saved_errno = errno;
92 brix_sd_ucred_wipe(&store); /* secret consumed; erase (A-4/T4) */
93 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
94 NGX_ERROR, saved_errno, start);
95 return NGX_ERROR;
96 }
97 brix_sd_ucred_wipe(&store); /* secret consumed; erase (A-4/T4) */
98 brix_vfs_sd_stat_fill(&sd_st, stat_out);
99 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
100 NGX_OK, 0, start);
101 return NGX_OK;
102 }
103
104 if (brix_lstat_confined_canon(ctx->log, ctx->root_canon, path, &st,
105 nofollow) != 0)
106 {
107 saved_errno = errno;
108 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
109 NGX_ERROR, saved_errno, start);
110 return NGX_ERROR;
111 }
112
113 brix_vfs_fill_stat(&st, stat_out);
114 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_STAT, NULL, 0,
115 NGX_OK, 0, start);
116 return NGX_OK;
117}

136/* Descend one cache/stage decorator to its wrapped source (NULL at a leaf or a
137 * non-decorator instance). Both unwrap helpers self-guard, so this is safe on any
138 * instance and lets the residency seam find a nearline driver buried under a
139 * read-cache and/or write-stage tier. */
140static brix_sd_instance_t *
141brix_vfs_decorator_source(const brix_sd_instance_t *inst)
142{
143 brix_sd_instance_t *s = brix_sd_cache_source_instance(inst);
144
145 return (s != NULL) ? s : brix_sd_stage_source_instance(inst);
146}

148/* Classify the resolved ctx path's nearline residency (online/nearline/offline/
149 * lost) so protocol handlers can advertise tape state — the HTTP Tape REST API, S3
150 * InvalidObjectState / x-amz-storage-class, root:// stat's nearline flag — WITHOUT
151 * forcing a recall. Walks any cache/stage decorators down to the CAP_NEARLINE
152 * driver and reads its residency model; an export with no nearline tier is always
153 * ONLINE (a plain disk/object store is resident). Emits no metric (it is an
154 * internal classification the caller's own op accounts for). NGX_OK with *out set,
155 * or NGX_ERROR (errno) on a guard failure or a driver error. */
156ngx_int_t
157brix_vfs_residency(brix_vfs_ctx_t *ctx, brix_sd_residency_t *out,
158 int *nearline_export)
159{
160 brix_sd_instance_t *inst;
161
162 if (nearline_export != NULL) {
163 *nearline_export = 0;
164 }
165 if (out == NULL || brix_vfs_require_confined(ctx) != NGX_OK) {
166 errno = EINVAL;
167 return NGX_ERROR;
168 }
169
170 *out = BRIX_SD_RES_ONLINE;
171
172 for (inst = ctx->sd; inst != NULL;
173 inst = brix_vfs_decorator_source(inst))
174 {
175 if ((brix_sd_caps(inst) & BRIX_SD_CAP_NEARLINE) != 0
176 && inst->driver->residency != NULL)
177 {
178 const char *path = brix_vfs_ctx_path(ctx);
179
180 if (nearline_export != NULL) {
181 *nearline_export = 1;
182 }
183 return inst->driver->residency(
184 inst, brix_vfs_export_relative(ctx, path), out);
185 }
186 }
187 return NGX_OK;
188}

190/* Driver-reported export space (phase-83 F5). Same decorator walk as residency:
191 * the first tier implementing the optional `space` slot answers with its
192 * quota-aware logical view; exports whose drivers have no space model decline so
193 * the caller keeps its statvfs(2) fallback. */
194ngx_int_t
195brix_vfs_space(brix_vfs_ctx_t *ctx, brix_sd_space_t *out)
196{
197 brix_sd_instance_t *inst;
198
199 if (out == NULL || ctx == NULL) {
200 errno = EINVAL;
201 return NGX_ERROR;
202 }
203
204 for (inst = ctx->sd; inst != NULL;
205 inst = brix_vfs_decorator_source(inst))
206 {
207 if (inst->driver->space != NULL) {
208 return inst->driver->space(inst, out);
209 }
210 }
211 return NGX_DECLINED;
212}

214/*
215 * brix_vfs_probe — confined existence/type probe for pre-op resolution and
216 * ACL gates. Unlike brix_vfs_stat/statf this emits NO OP_STAT metric or
217 * access-log line: it is an internal namespace pre-check that the caller's own
218 * operation accounts for (routing it through the metered stat would record a
219 * phantom STAT for every rm/chmod/mkdir/mv). nofollow selects lstat vs stat
220 * semantics. Returns NGX_OK with *stat_out filled when the path is present,
221 * NGX_DECLINED when it is absent (errno preserved from the underlying stat), or
222 * NGX_ERROR on a confinement-guard failure.
223 */
224ngx_int_t
225brix_vfs_probe(brix_vfs_ctx_t *ctx, int nofollow,
226 brix_vfs_stat_t *stat_out)
227{
228 struct stat st;
229 const brix_sd_driver_t *drv;
230
231 if (stat_out == NULL || brix_vfs_require_confined(ctx) != NGX_OK) {
232 errno = EINVAL;
233 return NGX_ERROR;
234 }
235
236 drv = brix_vfs_ctx_driver(ctx);
237 if (drv != NULL) {
238 brix_sd_ucred_t store;
239 brix_sd_cred_t cred;
240 brix_sd_stat_t sd_st;
241 int use_cred = 0, cred_err = 0;
242
243 /* Zero before the gate: it fills only the active credential kind and
244 * leaves the inactive pointers as-is, so an unzeroed cred would pass a
245 * garbage x509_proxy/bearer pointer to the driver (see the companion
246 * note in brix_vfs_stat_impl). */
247 ngx_memzero(&cred, sizeof(cred));
248
249 /* Credential gate for the probe: a denied pre-flight MUST return
250 * NGX_ERROR (EACCES), not NGX_DECLINED — a denied probe must not be
251 * silently treated as "absent", which would let the caller proceed.
252 * The caller maps EACCES → 403. */
253 if (brix_vfs_cred_gate_active(ctx)) {
254 if (brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err)
255 != NGX_OK)
256 {
257 errno = cred_err ? cred_err : EACCES;
258 return NGX_ERROR;
259 }
260 }
261
262 /* Dispatch on the leaf instance so brix_sd_stat_maybe_cred finds the
263 * leaf driver's stat_cred slot (decorators have only plain relays). */
264 if (drv->stat == NULL
265 || brix_sd_stat_maybe_cred(brix_vfs_ns_leaf(ctx->sd),
266 brix_vfs_export_relative(ctx, brix_vfs_ctx_path(ctx)),
267 &sd_st, use_cred ? &cred : NULL) != NGX_OK)
268 {
269 brix_sd_ucred_wipe(&store); /* secret consumed; erase (A-4/T4) */
270 return NGX_DECLINED; /* absent (or unsupported) — caller's errno */
271 }
272 brix_sd_ucred_wipe(&store); /* secret consumed; erase (A-4/T4) */
273 brix_vfs_sd_stat_fill(&sd_st, stat_out);
274 return NGX_OK;
275 }
276
277 if (brix_lstat_confined_canon(ctx->log, ctx->root_canon,
278 brix_vfs_ctx_path(ctx), &st,
279 nofollow) != 0)
280 {
281 return NGX_DECLINED;
282 }
283
284 brix_vfs_fill_stat(&st, stat_out);
285 return NGX_OK;
286}
