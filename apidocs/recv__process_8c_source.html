21/* brix_max_payload_for_request — per-opcode payload size limit, checked BEFORE
22 * any allocation so an oversized dlen is rejected without allocating. */
23static uint32_t
24brix_max_payload_for_request(uint16_t reqid)
25{
26 if (reqid == kXR_pgwrite || reqid == kXR_write || reqid == kXR_writev
27 || reqid == kXR_chkpoint) {
28 return BRIX_MAX_WRITE_PAYLOAD;
29 }
30
31 if (reqid == kXR_readv) {
32 /* Each segment is BRIX_READV_SEGSIZE (16) bytes. */
33 return BRIX_READV_MAXSEGS * BRIX_READV_SEGSIZE;
34 }
35
36 if (reqid == kXR_auth) {
37 return BRIX_MAX_AUTH_PAYLOAD;
38 }
39
40 if (reqid == kXR_prepare) {
41 return BRIX_MAX_PREPARE_PAYLOAD;
42 }
43
44 /* All other requests carry only a path (BRIX_MAX_PATH) plus a small
45 * fixed-size body. The +64 covers opcode-specific extras (e.g. the
46 * kXR_login info field that follows the username in the payload). */
47 return BRIX_MAX_PATH + 64;
48}

50/* brix_ensure_payload_buffer: ensure payload_buf holds dlen (+1 NUL) bytes at
51 * request start (free-then-alloc on resize — the buffer is empty here). */
52static ngx_int_t
53brix_ensure_payload_buffer(brix_ctx_t *ctx, ngx_connection_t *c,
54 uint32_t dlen)
55{
56 u_char *buf;
57 size_t need;
58
59 if (dlen > (uint32_t) (SIZE_MAX - 1)) {
60 return NGX_ERROR;
61 }
62 need = (size_t) dlen + 1;
63
64 if (ctx->recv.payload_buf != NULL && ctx->recv.payload_buf_size >= need) {
65 ctx->recv.payload = ctx->recv.payload_buf;
66 ctx->recv.payload[dlen] = '\0';
67 return NGX_OK;
68 }
69
70 buf = ngx_alloc(need, c->log);
71 if (buf == NULL) {
72 return NGX_ERROR;
73 }
74
75 if (ctx->recv.payload_buf != NULL) {
76 ngx_free(ctx->recv.payload_buf);
77 }
78
79 ctx->recv.payload_buf = buf;
80 ctx->recv.payload_buf_size = need;
81 ctx->recv.payload = buf;
82 ctx->recv.payload[dlen] = '\0';

126/*
127 * Phase 29 drain barrier condition: a non-read/write opcode arriving while
128 * reads (out.count) or writes (wr_inflight) are still in flight must run with
129 * the connection quiescent (a kXR_close could free a handle an in-flight
130 * sendfile chain or pwrite still references). kXR_read and kXR_write both
131 * pipeline and are never deferred.
132 */
133static ngx_flag_t
134brix_recv_should_defer(brix_ctx_t *ctx)
135{
136 return (ctx->out.count > 0 || ctx->out.wr_inflight > 0)
137 && ctx->recv.cur_reqid != kXR_read
138 && ctx->recv.cur_reqid != kXR_write;
139}
140
141/*
142 * Phase 29 pipelining condition for a parked cleartext sendfile kXR_read: keep
143 * reading so the next read's sendfile span queues behind this one while the
144 * prior response drains. All four conjuncts are load-bearing: resp_pipelinable
145 * = a single self-contained sendfile span (safe to queue another behind);
146 * !rd.win_active excludes a multi-window read still streaming out of the shared
147 * read_scratch (which must stay serial); out.count < depth is the in-flight cap.
148 */
149static ngx_flag_t
150brix_recv_try_pipeline_read(brix_ctx_t *ctx)
151{
152 return ctx->state == XRD_ST_SENDING
153 && ctx->recv.cur_reqid == kXR_read
154 && ctx->out.resp_pipelinable
155 && !ctx->rd.win_active
156 && ctx->out.count < ctx->out.pipeline_depth;
157}

159/*
160 * kXR_writev / kXR_chkpoint stock wire framing: the header dlen covers only the
161 * descriptor block (writev) or the embedded 24-byte sub-header (chkpoint/ckpXeq);
162 * the trailing segment/sub-request data streams after the frame. Just after
163 * that first block lands, raise the read obligation by the trailing length so
164 * descriptors + data land contiguously before dispatch. A malformed block is
165 * deliberately NOT rejected here — dispatch runs the login/auth/write gates
166 * first and the handler emits the stock-parity error and drops the link. Both
167 * stages are bounded (BRIX_MAX_WRITE_PAYLOAD), so the grow is safe. CONTINUE to
168 * keep receiving the extended body, NEXT when no extension applies, BREAK on a
169 * grow failure.
170 */
171static brix_recv_step_t
172brix_recv_extend_body(brix_ctx_t *ctx, ngx_connection_t *c)
173{
174 if (ctx->recv.cur_reqid == kXR_writev && !ctx->recv.cur_body_extended) {
175 uint32_t extra;
176
177 ctx->recv.cur_body_extended = 2;
178
179 if (brix_writev_body_extra(ctx->recv.payload, ctx->recv.cur_dlen,
180 &extra) == NGX_OK
181 && extra > 0)
182 {
183 if (brix_grow_payload_buffer(ctx, c,
184 ctx->recv.cur_dlen + extra) != NGX_OK)
185 {
186 return BRIX_RECV_STEP_BREAK;
187 }
188 ctx->recv.cur_body_extra = extra;
189 return BRIX_RECV_STEP_CONTINUE; /* receive the streamed segments */
190 }
191 }
192
193 if (ctx->recv.cur_reqid == kXR_chkpoint
194 && ctx->recv.cur_body[15] == kXR_ckpXeq
195 && ctx->recv.cur_dlen == XRD_REQUEST_HDR_LEN
196 && ctx->recv.cur_body_extended < 2)
197 {
198 uint32_t extra = 0;
199 unsigned final = 1;
200
201 (void) brix_ckpxeq_body_extra(ctx->recv.payload,
202 ctx->recv.cur_dlen + ctx->recv.cur_body_extra,
203 &extra, &final);
204 ctx->recv.cur_body_extended = final ? 2 : 1;
205
206 if (extra > 0) {
207 if (brix_grow_payload_buffer(ctx, c,
208 ctx->recv.cur_dlen + ctx->recv.cur_body_extra + extra)
209 != NGX_OK)
210 {
211 return BRIX_RECV_STEP_BREAK;
212 }
213 ctx->recv.cur_body_extra += extra;
214 return BRIX_RECV_STEP_CONTINUE; /* receive the streamed sub-body */
215 }
216 }
217
218 return BRIX_RECV_STEP_NEXT;
219}

221/*
222 * A fully-received 24-byte ClientRequestHdr just landed: decode it, reject an
223 * oversized dlen before any allocation, and either accept a payload-bearing
224 * request (→ REQ_PAYLOAD) or dispatch a dlen==0 request. CONTINUE/RETURN/BREAK.
225 */
226static brix_recv_step_t
227brix_recv_after_header(ngx_stream_session_t *s, ngx_connection_t *c,
228 ngx_stream_brix_srv_conf_t *conf, brix_ctx_t *ctx, ngx_event_t *rev,
229 size_t *rx_pending)
230{
231 ClientRequestHdr *hdr = (ClientRequestHdr *) ctx->recv.hdr_buf;
232 uint32_t max_pl;
233 ngx_int_t rc;
234
235 /*
236 * Decode the fixed 24-byte ClientRequestHdr (wire layout: streamid[2] @0,
237 * requestid @2, body[16] @4, dlen @20). streamid is an opaque client tag
238 * echoed verbatim — copied byte-for-byte, never byte-swapped. requestid and
239 * dlen are big-endian on the wire, so ntohs/ntohl to host order. cur_body is
240 * the raw 16-byte opcode argument block, handed to dispatch untouched.
241 */
242 ctx->recv.cur_streamid[0] = hdr->streamid[0];
243 ctx->recv.cur_streamid[1] = hdr->streamid[1];
244 ctx->recv.cur_reqid = ntohs(hdr->requestid);
245 ngx_memcpy(ctx->recv.cur_body, hdr->body, 16);
246 ctx->recv.cur_dlen = (uint32_t) ntohl(hdr->dlen);
247 ctx->recv.cur_body_extra = 0;
248 ctx->recv.cur_body_extended = 0;
249 BRIX_SRV_METRIC_INC(ctx, request_frames_total);
250 BRIX_SRV_METRIC_ADD(ctx, request_payload_bytes_total, ctx->recv.cur_dlen);
251
252 ngx_log_debug(NGX_LOG_DEBUG_STREAM, c->log, 0,
253 "brix: req sid=[%02xd%02xd] reqid=%04xd dlen=%uz"
254 " avail=%d ready=%d",
255 (int) ctx->recv.cur_streamid[0],
256 (int) ctx->recv.cur_streamid[1],
257 (int) ctx->recv.cur_reqid, (size_t) ctx->recv.cur_dlen,
258 c->read->available, (int) c->read->ready);
259
260 /* dlen is untrusted client input — reject before any allocation. */
261 max_pl = brix_max_payload_for_request(ctx->recv.cur_reqid);
262 if (ctx->recv.cur_dlen > max_pl) {
263 ngx_log_error(NGX_LOG_WARN, c->log, 0,
264 "brix: payload too large (%uz), closing",
265 (size_t) ctx->recv.cur_dlen);
266 BRIX_SRV_METRIC_INC(ctx, oversized_payloads_total);
267 return BRIX_RECV_STEP_BREAK;
268 }
269
270 if (ctx->recv.cur_dlen > 0) {
271 if (brix_ensure_payload_buffer(ctx, c, ctx->recv.cur_dlen) != NGX_OK) {
272 return BRIX_RECV_STEP_BREAK;
273 }
274 ctx->recv.payload_pos = 0;
275 ctx->state = XRD_ST_REQ_PAYLOAD;
276 ctx->recv.hdr_pos = 0;
277 return BRIX_RECV_STEP_CONTINUE;
278 }
279
280 ctx->recv.payload = NULL;
281
282 /*
283 * Phase 29 drain barrier (extended for write pipelining): a non-read/write
284 * opcode arriving while reads or writes are still in flight must run with the
285 * connection quiescent. Defer it until both out.count and wr_inflight drain;
286 * the recv loop re-dispatches it then.
287 */
288 if (brix_recv_should_defer(ctx)) {
289 ctx->out.recv_deferred = 1;
290 ctx->state = XRD_ST_SENDING;
291 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, *rx_pending);
292 return BRIX_RECV_STEP_RETURN;
293 }
294
295 /*
296 * Reset the pipelinable marker before dispatch; only the single-chunk
297 * sendfile read builder sets it back to 1. A read served from the
298 * memory/window path thus stays non-pipelinable (its header/data live in the
299 * shared scratch buffers).
300 */
301 ctx->out.resp_pipelinable = 0;
302
303 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, *rx_pending);
304 *rx_pending = 0;
305 rc = brix_dispatch(ctx, c, conf);
306 if (rc == NGX_ERROR) {
307 return BRIX_RECV_STEP_BREAK;
308 }
309
310 if (ctx->state == XRD_ST_AIO) {
311 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
312 return BRIX_RECV_STEP_BREAK;
313 }
314 return BRIX_RECV_STEP_RETURN;
315 }
316
317 /*
318 * Phase 29 pipelining: a cleartext sendfile kXR_read parked its response
319 * (state SENDING) and we are below the in-flight cap. Keep reading so the
320 * next read's sendfile span queues behind this one.
321 */
322 if (brix_recv_try_pipeline_read(ctx)) {
323 ctx->state = XRD_ST_REQ_HEADER;
324 ctx->recv.hdr_pos = 0;
325 }
326
327 if (ctx->state != XRD_ST_SENDING) {
328 if (ctx->tls_pending) {
329 brix_start_tls(ctx, c, conf);
330 return BRIX_RECV_STEP_RETURN;
331 }
332 ctx->state = XRD_ST_REQ_HEADER;
333 ctx->recv.hdr_pos = 0;
334 }
335
336 return BRIX_RECV_STEP_CONTINUE;
337}

339/*
340 * A fully-received payload body just landed: run the kXR_writev / kXR_chkpoint
341 * body extension (which may keep receiving), apply the drain barrier, then
342 * dispatch — handling the write-pipelining and read-pipelining continuations.
343 * CONTINUE/RETURN/BREAK.
344 */
345static brix_recv_step_t
346brix_recv_after_payload(ngx_stream_session_t *s, ngx_connection_t *c,
347 ngx_stream_brix_srv_conf_t *conf, brix_ctx_t *ctx, ngx_event_t *rev,
348 size_t *rx_pending)
349{
350 brix_recv_step_t ext;
351 ngx_int_t rc;
352
353 ext = brix_recv_extend_body(ctx, c);
354 if (ext != BRIX_RECV_STEP_NEXT) {
355 return ext; /* CONTINUE (receiving the streamed body) or BREAK */
356 }
357
358 /*
359 * Phase 29 drain barrier (extended for write pipelining): every opcode other
360 * than kXR_read / kXR_write must run with the connection quiescent, so defer
361 * it until both the response/ack queue and the in-flight pwrites have drained
362 * — e.g. a kXR_close must not retire a handle a pwrite is still writing.
363 */
364 if (brix_recv_should_defer(ctx)) {
365 ctx->out.recv_deferred = 1;
366 ctx->state = XRD_ST_SENDING;
367 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, *rx_pending);
368 return BRIX_RECV_STEP_RETURN;
369 }
370
371 ctx->state = XRD_ST_REQ_HEADER;
372 ctx->recv.hdr_pos = 0;
373
374 /* Only the single-chunk sendfile read builder re-sets this. */
375 ctx->out.resp_pipelinable = 0;
376
377 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, *rx_pending);
378 *rx_pending = 0;
379 rc = brix_dispatch(ctx, c, conf);
380 if (rc == NGX_ERROR) {
381 return BRIX_RECV_STEP_BREAK;
382 }
383
384 if (ctx->state == XRD_ST_AIO) {
385 /*
386 * Write pipelining: a plain kXR_write just posted its pwrite to the
387 * thread pool (wr_inflight bumped in brix_handle_write). Instead of
388 * suspending until it completes — which would serialize the next chunk's
389 * network receive behind this chunk's disk write — keep receiving. The
390 * backpressure boundary caps the depth; the completion callback queues
391 * the ack asynchronously. Every other AIO op still suspends.
392 */
393 if (ctx->recv.cur_reqid == kXR_write && ctx->out.wr_inflight > 0) {
394 ctx->state = XRD_ST_REQ_HEADER;
395 ctx->recv.hdr_pos = 0;
396 return BRIX_RECV_STEP_CONTINUE;
397 }
398 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
399 return BRIX_RECV_STEP_BREAK;
400 }
401 return BRIX_RECV_STEP_RETURN;
402 }
403
404 /*
405 * Phase 29 pipelining (payload-bearing kXR_read with a read-ahead list):
406 * identical to the dlen==0 read path — keep reading so the next read's
407 * sendfile span queues behind this one.
408 */
409 if (brix_recv_try_pipeline_read(ctx)) {
410 ctx->state = XRD_ST_REQ_HEADER;
411 ctx->recv.hdr_pos = 0;
412 return BRIX_RECV_STEP_CONTINUE;
413 }
414
415 if (ctx->state == XRD_ST_SENDING) {
416 return BRIX_RECV_STEP_RETURN;
417 }
418
419 return BRIX_RECV_STEP_CONTINUE;
420}
