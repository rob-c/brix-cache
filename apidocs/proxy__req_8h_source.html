38/*
39 * brix_gsi_build_pxyreq — build the server→client kXGS_pxyreq payload: an
40 * RFC-3820 proxy-certificate REQUEST for `parent_pem` (the peer's EEC/proxy, PEM).
41 *
42 * Generates a fresh RSA key (>= parent bits, min 2048, exp 65537), names the
43 * subject `<parent subject>/CN=<random serial>`, attaches a CRITICAL
44 * proxyCertInfo extension (impersonation policy, pathlen derived from the parent),
45 * copies the parent's other extensions (minus subjectAltName / proxyCertInfo), and
46 * self-signs the request with SHA-256.
47 *
48 * On success returns 0 with *newkey = the fresh keypair (caller EVP_PKEY_free) and
49 * req_der = a malloc'd DER X509_REQ (caller free()). On failure returns -1 and,
50 * if err != NULL, writes a short reason into err->buf; no output is set.
51 */
52int brix_gsi_build_pxyreq(const brix_gsi_blob_t *parent_pem, EVP_PKEY **newkey,
53 brix_gsi_buf_t *req_der, const brix_gsi_err_t *err);

55/*
56 * brix_gsi_sign_pxyreq — ISSUE a proxy by signing a request (the delegating
57 * client's role, kXGC_sigpxy). `signer_pem` is the issuer EEC/proxy (PEM, leaf
58 * first), `signer_key` its private key, `req_der` the DER request from
59 * brix_gsi_build_pxyreq.
60 *
61 * Validates the request subject is `<signer subject>/CN=<serial>`, builds an
62 * X.509 v3 proxy (serial = the request CN, subject = request subject, issuer =
63 * signer subject, pubkey = request pubkey, validity = the signer's remaining
64 * lifetime), copies the signer's non-SAN/non-pci extensions (keyUsage required;
65 * a subjectAltName is rejected), adds a CRITICAL proxyCertInfo with the correctly
66 * decremented path-length, and signs with `signer_key` (SHA-256).
67 *
68 * On success returns 0 with proxy_pem = the malloc'd signed proxy certificate
69 * (PEM, caller free()). -1 with err->buf set on failure.
70 */
71int brix_gsi_sign_pxyreq(const brix_gsi_blob_t *signer_pem, EVP_PKEY *signer_key,
72 const brix_gsi_blob_t *req_der,
73 brix_gsi_buf_t *proxy_pem, const brix_gsi_err_t *err);
