17/*
18 * brix_gsi_build_pxyreq — build the server→client kXGS_pxyreq payload: an
19 * RFC-3820 proxy-certificate REQUEST for `parent_pem` (the peer's EEC/proxy, PEM).
20 *
21 * Generates a fresh RSA key (>= parent bits, min 2048, exp 65537), names the
22 * subject `<parent subject>/CN=<random serial>`, attaches a CRITICAL
23 * proxyCertInfo extension (impersonation policy, pathlen derived from the parent),
24 * copies the parent's other extensions (minus subjectAltName / proxyCertInfo), and
25 * self-signs the request with SHA-256.
26 *
27 * On success returns 0 with *newkey = the fresh keypair (caller EVP_PKEY_free) and
28 * *req_der / *req_len = a malloc'd DER X509_REQ (caller free()). On failure returns
29 * -1 and, if err != NULL, writes a short reason into err[errcap]; no output is set.
30 */
31int brix_gsi_build_pxyreq(const uint8_t *parent_pem, size_t parent_pem_len,
32 EVP_PKEY **newkey, uint8_t **req_der, size_t *req_len,
33 char *err, size_t errcap);

35/*
36 * brix_gsi_sign_pxyreq — ISSUE a proxy by signing a request (the delegating
37 * client's role, kXGC_sigpxy). `signer_pem` is the issuer EEC/proxy (PEM, leaf
38 * first), `signer_key` its private key, `req_der`/`req_len` the request from
39 * brix_gsi_build_pxyreq.
40 *
41 * Validates the request subject is `<signer subject>/CN=<serial>`, builds an
42 * X.509 v3 proxy (serial = the request CN, subject = request subject, issuer =
43 * signer subject, pubkey = request pubkey, validity = the signer's remaining
44 * lifetime), copies the signer's non-SAN/non-pci extensions (keyUsage required;
45 * a subjectAltName is rejected), adds a CRITICAL proxyCertInfo with the correctly
46 * decremented path-length, and signs with `signer_key` (SHA-256).
47 *
48 * On success returns 0 with proxy_pem + proxy_len = the malloc'd signed proxy
49 * certificate (PEM, caller free()). -1 with err[errcap] on failure.
50 */
51int brix_gsi_sign_pxyreq(const uint8_t *signer_pem, size_t signer_pem_len,
