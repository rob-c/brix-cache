22static void
23test_loadavg(void)
24{
25 uint8_t pct = 77;
26
27 CHECK(brix_cms_meter_parse_loadavg("2.00 1.50 1.00 2/345 6789\n", 4,
28 &pct) == 0);
29 CHECK(pct == 50); /* 2.0 load on 4 cpus = 50% */
30
31 CHECK(brix_cms_meter_parse_loadavg("8.00 0 0", 4, &pct) == 0);
32 CHECK(pct == 100); /* clamped at 100 */
33
34 CHECK(brix_cms_meter_parse_loadavg("0.00 0 0", 4, &pct) == 0);
35 CHECK(pct == 0);
36
37 /* error legs: garbage / no cpus / NULL */
38 pct = 77;
39 CHECK(brix_cms_meter_parse_loadavg("garbage", 4, &pct) == -1);
40 CHECK(brix_cms_meter_parse_loadavg("1.0 0 0", 0, &pct) == -1);
41 CHECK(brix_cms_meter_parse_loadavg(NULL, 4, &pct) == -1);
42 CHECK(pct == 77); /* out untouched on failure */
43}

47test_meminfo(void)
48{
49 uint8_t pct = 77;
50 const char *mi =
51 "MemTotal: 16000000 kB\n"
52 "MemFree: 1000000 kB\n"
53 "MemAvailable: 4000000 kB\n"
54 "Buffers: 500000 kB\n";
55
56 CHECK(brix_cms_meter_parse_meminfo(mi, &pct) == 0);
57 CHECK(pct == 75); /* (16M-4M)/16M */
58
59 /* avail > total (procfs oddity) clamps to 0 used, not underflow */
60 CHECK(brix_cms_meter_parse_meminfo(
61 "MemTotal: 100 kB\nMemAvailable: 200 kB\n", &pct) == 0);
62 CHECK(pct == 0);
63
64 /* error legs: key missing / zero total / NULL */
65 pct = 77;
66 CHECK(brix_cms_meter_parse_meminfo("MemTotal: 100 kB\n", &pct) == -1);
67 CHECK(brix_cms_meter_parse_meminfo(
68 "MemTotal: 0 kB\nMemAvailable: 0 kB\n", &pct) == -1);
69 CHECK(brix_cms_meter_parse_meminfo(NULL, &pct) == -1);
70 CHECK(pct == 77);
71}

115static void
116test_rate_pct(void)
117{
118 /* 125 MB over 1s against a 125 MB/s reference = 100% */
119 CHECK(brix_cms_meter_rate_pct(125000000, 1000, 125000000) == 100);
120 /* half rate = 50% */
121 CHECK(brix_cms_meter_rate_pct(62500000, 1000, 125000000) == 50);
122 /* over-reference clamps at 100 */
123 CHECK(brix_cms_meter_rate_pct(999999999, 1000, 125000000) == 100);
124 /* zero elapsed / zero reference never divide */
125 CHECK(brix_cms_meter_rate_pct(1000, 0, 125000000) == 0);
126 CHECK(brix_cms_meter_rate_pct(1000, 1000, 0) == 0);
127}

130static void
131test_sample_live(void)
132{
133 /* Smoke on the real /proc: bytes must be percentages and the call must
134 * not crash; the first (unprimed) sample reports 0 for net/pag. */
135 brix_cms_meter_t m;
136 uint8_t out5[5] = {1, 2, 3, 4, 5};
137
138 memset(&m, 0, sizeof(m));
139 brix_cms_meter_sample(&m, 1000, out5);
140 CHECK(out5[0] <= 100);
141 CHECK(out5[1] == 0); /* unprimed delta */
142 CHECK(out5[2] == 0); /* xeq always 0 */
143 CHECK(out5[3] <= 100);
144 CHECK(out5[4] == 0); /* unprimed delta */
145
146 brix_cms_meter_sample(&m, 2000, out5); /* primed second sample */
147 CHECK(out5[1] <= 100);
148 CHECK(out5[4] <= 100);
149}
