1/* ---- File: src/core/seccomp/seccomp.h — per-worker seccomp-BPF syscall filter ----
2 *
3 * WHAT: Declares brix_seccomp_install(), the single entry point that installs a
4 * libseccomp allowlist filter on the calling worker process. Called once
5 * per worker from the tail of ngx_stream_brix_init_process (process.c),
6 * after every setup syscall has run.
7 *
8 * WHY: Defence-in-depth (hyper-hardening-plan §D-3). The privilege model is
9 * already strong — unprivileged operation, PR_SET_NO_NEW_PRIVS, full
10 * capability drop — but the impersonation broker necessarily retains
11 * CAP_SETUID, so a worker-code exploit is root-equivalent if the broker is
12 * reached. A syscall allowlist is the most direct mitigation for that
13 * residual: it removes execve/ptrace/process_vm_* from the worker's reach
14 * so a hijacked worker cannot spawn a shell, attach to another process, or
15 * peer into the broker's address space.
16 *
17 * HOW: Tri-state, operator opt-in (`brix_seccomp off|audit|enforce`, default
18 * off). AUDIT loads a log-only filter to converge the allowlist without
19 * risk; ENFORCE kills the named-dangerous set and EPERMs any other
20 * non-allowlisted syscall. When the binary was built without libseccomp
21 * (-DBRIX_HAVE_SECCOMP unset) the implementation is a stub that fails
22 * closed for audit/enforce and is a no-op for off — the operator never
23 * silently runs unfiltered while believing otherwise.
24 */

31/*
32 * Install the seccomp filter for `mode` (BRIX_SECCOMP_OFF/AUDIT/ENFORCE) on the
33 * calling process. Returns NGX_OK when the requested policy is in force (OFF is
34 * always OK); NGX_ERROR when audit/enforce was requested but the filter could
35 * not be built and loaded — the caller MUST fail the worker on NGX_ERROR rather
36 * than serve unfiltered.
37 */
38ngx_int_t brix_seccomp_install(ngx_cycle_t *cycle, ngx_uint_t mode,
39 ngx_uint_t allow_exec);

41/*
42 * Process-global "allow execve under enforce" flag (0/1), set by
43 * `brix_seccomp_allow_exec on`. When 1, an ENFORCE filter allowlists
44 * execve/execveat (so OIDC token fetch / native-TPC token-exchange / the
45 * kXR_prepare hook can fork+exec) while STILL killing ptrace/process_vm_*.
