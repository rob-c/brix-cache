1/*
2 * root_prepare.c — shared export-root validation and canonicalization.
3 *
4 * All three protocol surfaces (native root://, WebDAV, S3) need the same
5 * startup check on their configured export root: verify it exists, is a
6 * readable directory with the right permissions, and resolve it through
7 * realpath(3) to eliminate symlinks from the confinement boundary.
8 *
9 * This helper centralises that logic so the behaviour is identical across
10 * all surfaces. It must be called during merge_loc_conf / postconfiguration
11 * so that nginx -t catches misconfigured roots before traffic is accepted.
12 */
13

22char *
23brix_prepare_export_root(ngx_conf_t *cf,
24 const ngx_str_t *root, const brix_export_root_opts_t *opts,
25 char *root_canon)
26{
27 char root_buf[PATH_MAX];
28 int access_mode;
29 ngx_str_t root_str;
30
31 /* Empty root — either silently skip or hard fail depending on opts. */
32 if (root == NULL || root->len == 0) {
33 if (opts->required) {
34 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
35 "%s is required but not set",
36 opts->directive_name);
37 return NGX_CONF_ERROR;
38 }
39 return NGX_CONF_OK;
40 }
41
42 /* Guard against paths that would overflow the canonical buffer. */
43 if (root->len >= opts->canon_size || root->len >= sizeof(root_buf)) {
44 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
45 "%s path is too long", opts->directive_name);
46 return NGX_CONF_ERROR;
47 }
48
49 /* Build a NUL-terminated copy for stat/access/realpath. */
50 ngx_memcpy(root_buf, root->data, root->len);
51 root_buf[root->len] = '\0';
52
53 /* Validate existence, kind (directory), and access permissions. */
54 root_str.data = (u_char *) root_buf;
55 root_str.len = root->len;
56 access_mode = opts->allow_write ? (R_OK | W_OK | X_OK) : (R_OK | X_OK);
57
58 if (brix_validate_path(cf, opts->directive_name, &root_str,
59 BRIX_PATH_DIRECTORY, access_mode) != NGX_OK)
60 {
61 return NGX_CONF_ERROR;
62 }
63
64 /* Resolve symlinks and normalise the path for the confinement boundary. */
65 if (realpath(root_buf, root_canon) == NULL) {
66 ngx_conf_log_error(NGX_LOG_EMERG, cf, errno,
67 "%s: cannot resolve canonical path for \"%s\"",
68 opts->directive_name, root_buf);
69 return NGX_CONF_ERROR;
70 }
71
72 return NGX_CONF_OK;
73}
