21static void rec_alert(void *ctx, const char *msg)
22{
23 (void) ctx;
24 if (g_nalerts < 8) {
25 size_t n = strlen(msg);
26 if (n >= sizeof(g_alerts[0])) n = sizeof(g_alerts[0]) - 1;
27 memcpy(g_alerts[g_nalerts], msg, n);
28 g_alerts[g_nalerts][n] = '\0';
29 g_nalerts++;
30 }
31}
