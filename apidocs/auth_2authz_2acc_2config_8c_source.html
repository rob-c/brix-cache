1/*
2 * config.c — xrdacc engine lifecycle: per-worker build + hot-reload timer.
3 *
4 * WHAT: brix_acc_build() parses an authdb file into a fresh table generation
5 * and installs the OS/NIS group tunables + resolvers. brix_acc_init_server()
6 * builds the per-worker tables for a server running `brix_authdb_format
7 * xrdacc` and, when a refresh interval is set, arms a timer that re-reads the
8 * file on mtime change and atomically swaps the live tables.
9 *
10 * WHY: XrdAcc keeps authorization per-process and reloads the authdb without a
11 * restart (acc.authrefresh). Per-worker tables + a single-threaded event-loop
12 * swap give the same behaviour with no locking.
13 *
14 * HOW: modelled on the CRL reload timer — ngx_pcalloc an ngx_event_t in the
15 * worker pool, point it at the refresh handler, and re-arm after each fire.
16 */
17

55/* Re-read the authdb when its mtime changes; swap the live tables atomically. */
56static void
57brix_acc_refresh_handler(ngx_event_t *ev)
58{
59 ngx_stream_brix_srv_conf_t *xcf = ev->data;
60 ngx_file_info_t fi;
61
62 if (xcf->authdb.len > 0
63 && ngx_file_info(xcf->authdb.data, &fi) != NGX_FILE_ERROR)
64 {
65 time_t mt = ngx_file_mtime(&fi);
66 time_t cur = (xcf->acc.tables != NULL) ? xcf->acc.tables->mtime : 0;
67
68 if (mt != cur) {
69 brix_acc_tables_t *nt =
70 brix_acc_build((const char *) xcf->authdb.data,
71 xcf->acc.gidlifetime, xcf->acc.pgo,
72 (xcf->acc.nisdomain.len > 0)
73 ? (const char *) xcf->acc.nisdomain.data
74 : NULL,
75 (xcf->acc.gidretran.len > 0)
76 ? (const char *) xcf->acc.gidretran.data
77 : NULL,
78 (xcf->acc.spacechar.len > 0)
79 ? (char) xcf->acc.spacechar.data[0] : 0,
80 xcf->acc.encoding,
81 ev->log);
82 if (nt != NULL) {
83 brix_acc_tables_t *old = xcf->acc.tables;
84 xcf->acc.tables = nt; /* single-threaded: safe swap */
85 brix_acc_tables_free(old);
86 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
87 "xrootd authdb reloaded: %s", xcf->authdb.data);
88 }
89 }
90 }
91
92 if (xcf->acc.refresh > 0 && !ngx_exiting) {
93 ngx_add_timer(ev, (ngx_msec_t) xcf->acc.refresh * 1000);
94 }
95}

113/*
114 * HTTP hot-reload timer — the loc-conf analogue of brix_acc_refresh_handler.
115 * ev->data is the brix_acc_http_t (embedded in the loc-conf, so it outlives
116 * every request); on mtime change it rebuilds and atomically swaps acc->tables.
117 */
118static void
119brix_acc_http_refresh_handler(ngx_event_t *ev)
120{
121 brix_acc_http_t *acc = ev->data;
122 ngx_file_info_t fi;
123
124 if (acc->authdb.len > 0
125 && ngx_file_info(acc->authdb.data, &fi) != NGX_FILE_ERROR)
126 {
127 time_t mt = ngx_file_mtime(&fi);
128 time_t cur = (acc->tables != NULL) ? acc->tables->mtime : 0;
129
130 if (mt != cur) {
131 brix_acc_tables_t *nt = brix_acc_http_build(acc, ev->log);
132 if (nt != NULL) {
133 brix_acc_tables_t *old = acc->tables;
134 acc->tables = nt; /* single-threaded: safe swap */
135 brix_acc_tables_free(old);
136 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
137 "xrootd authdb reloaded: %s", acc->authdb.data);
138 }
139 }
140 }
141
142 if (acc->refresh > 0 && !ngx_exiting) {
143 ngx_add_timer(ev, (ngx_msec_t) acc->refresh * 1000);
144 }
145}

147/* Arm this worker's refresh timer on first use (no HTTP per-location init hook,
148 * so the lazy build is the natural place — see XrdAcc acc.authrefresh). */
149static void
150brix_acc_http_arm_timer(brix_acc_http_t *acc)
151{
152 if (acc->timer_armed || acc->refresh <= 0) {
153 return;
154 }
155 acc->timer_armed = 1;
156 acc->timer.handler = brix_acc_http_refresh_handler;
157 acc->timer.data = acc;
158 acc->timer.log = ngx_cycle->log;
159 ngx_add_timer(&acc->timer, (ngx_msec_t) acc->refresh * 1000);
160}

194ngx_int_t
195brix_acc_http_authorize(ngx_pool_t *pool, ngx_log_t *log,
196 brix_acc_http_t *acc, const char *name, const char *host,
197 const char *vorg, const char *role, const char *grp,
198 brix_acc_op_t op, const char *path)
199{
200 brix_acc_entity_t *ent;
201 brix_acc_privs_t privs;
202
203 if (acc->format != BRIX_AUTHDB_FORMAT_XRDACC) {
204 return NGX_DECLINED; /* engine not selected — caller's checks stand */
205 }
206
207 /* Lazy per-worker build: the first request in a worker parses the authdb;
208 * the write is COW-private so each worker keeps its own table set. */
209 if (acc->tables == NULL) {
210 if (acc->authdb.len == 0) {
211 return NGX_ERROR; /* xrdacc selected without an authdb -> deny */
212 }
213 acc->tables = brix_acc_http_build(acc, log);
214 if (acc->tables == NULL) {
215 return NGX_ERROR;
216 }
217 }
218 brix_acc_http_arm_timer(acc); /* hot-reload, once per worker */
219
220 if (name == NULL) { name = ""; }
221 if (host == NULL || *host == '\0') { host = "?"; }
222 if (path == NULL) { path = "/"; }
223
224 ent = brix_acc_entity_build(pool, name, host, (name[0] != '\0'),
225 vorg ? vorg : "", role ? role : "",
226 grp ? grp : "");
227 if (ent == NULL) {
228 return NGX_ERROR;
229 }
230
231 privs = brix_acc_access(acc->tables, ent, path, op);
232 brix_acc_audit(log, acc->audit, privs != BRIX_ACC_PRIV_NONE,
233 brix_acc_op_name(op), name, host, path);
234
235 return (privs != BRIX_ACC_PRIV_NONE) ? NGX_OK : NGX_ERROR;
236}

238ngx_int_t
239brix_acc_init_server(ngx_stream_brix_srv_conf_t *xcf, ngx_cycle_t *cycle)
240{
241 /* Only servers using the xrdacc engine with an authdb file. */
242 if (xcf->acc.format != BRIX_AUTHDB_FORMAT_XRDACC || xcf->authdb.len == 0) {
243 return NGX_OK;
244 }
245
246 xcf->acc.tables = brix_acc_build(
247 (const char *) xcf->authdb.data, xcf->acc.gidlifetime, xcf->acc.pgo,
248 (xcf->acc.nisdomain.len > 0) ? (const char *) xcf->acc.nisdomain.data
249 : NULL,
250 (xcf->acc.gidretran.len > 0) ? (const char *) xcf->acc.gidretran.data
251 : NULL,
252 (xcf->acc.spacechar.len > 0) ? (char) xcf->acc.spacechar.data[0] : 0,
253 xcf->acc.encoding,
254 cycle->log);
255 if (xcf->acc.tables == NULL) {
256 BRIX_DIAG_EMERG(cycle->log, 0,
257 "brix: failed to load authorization database \"%V\"",
258 "the authdb could not be opened or parsed",
259 "see the specific \"xrootd authdb:\" error logged just above for "
260 "the exact file/line and how to fix it",
261 &xcf->authdb);
262 return NGX_ERROR;
263 }
264
265 if (xcf->acc.refresh > 0) {
266 xcf->acc.timer = ngx_pcalloc(cycle->pool, sizeof(ngx_event_t));
267 if (xcf->acc.timer == NULL) {
268 return NGX_ERROR;
269 }
270 xcf->acc.timer->handler = brix_acc_refresh_handler;
271 xcf->acc.timer->data = xcf;
272 xcf->acc.timer->log = cycle->log;
273 ngx_add_timer(xcf->acc.timer, (ngx_msec_t) xcf->acc.refresh * 1000);
274 }
275
276 return NGX_OK;
277}
