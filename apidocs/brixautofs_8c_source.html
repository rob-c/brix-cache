1/*
2 * brixautofs.c — CVMFS-brix-autofs: the /cvmfs automount umbrella daemon.
3 *
4 * WHAT: `brixMount autofs <etc-root|-> <mountdir>` mounts a FUSE umbrella on
5 * <mountdir> (normally /cvmfs). Resolving a valid repository FQRN
6 * fork/execs `brixMount cvmfs <fqrn> <mntbase>/<fqrn> -f -o …` as a real
7 * FUSE mount in an EXTERNAL mount farm and answers the name with a
8 * symlink to it — the first path resolution blocks until the repo is
9 * live (autofs UX), every later access follows the link straight into
10 * the mounted fs.
11 * WHY: the stock CVMFS client's autofs experience with zero dependency on
12 * autofs or systemd — the out-of-the-box answer for WSL2 and containers,
13 * and a plain daemon for everything else. One process per repo also
14 * isolates a hung stratum to that repo.
15 * HOW: children CANNOT mount over the umbrella's own virtual subdir: the
16 * kernel serializes every lookup of one name on the in-lookup dentry
17 * (d_alloc_parallel), so while our LOOKUP handler waits for the child,
18 * the child's own mount(2) path walk of the same name waits on US — a
19 * guaranteed deadlock the autofs kernel module exists to solve. A plain
20 * FUSE umbrella therefore mounts children in a farm dir it never path-
21 * walks itself (default <cachebase>/.mnt) and presents each repo as a
22 * symlink; readlink is the (blocking) mount trigger, lstat/readdir never
23 * mount. Works fully unprivileged (no bind mounts needed). Idle repos
24 * expire via umount2(MNT_EXPIRE) two-phase (root only); SIGTERM unmounts
25 * every child before the umbrella itself goes away. Honors
26 * CVMFS_REPOSITORIES / CVMFS_STRICT_MOUNT from the stock config cascade.
27 * Pure parts (FQRN gate, slot table) live in brixautofs.h and compile
28 * without libfuse for the unit test (-DBRIXAUTOFS_UNIT).
29 */

37int brixautofs_valid_fqrn(const char *name) {
38 if (name == NULL || name[0] == '\0') return 0;
39 size_t len = strlen(name);
40 if (len > BRIXAUTOFS_FQRN_MAX - 1) return 0;
41
42 int labels = 0;
43 size_t lab = 0; /* current label length */
44 for (size_t i = 0; i <= len; i++) {
45 char c = name[i];
46 if (c == '.' || c == '\0') {
47 if (lab == 0 || lab > 63) return 0; /* empty/overlong label */
48 if (name[i - 1] == '-') return 0; /* label ends with '-' */
49 labels++;
50 lab = 0;
51 continue;
52 }
53 int ok = (c >= 'a' && c <= 'z') || (c >= '0' && c <= '9') || c == '-';
54 if (!ok) return 0; /* '/', '_', upper, meta… */
55 if (lab == 0 && c == '-') return 0; /* label starts with '-' */
56 lab++;
57 }
58 return labels >= 2; /* needs an org AND a domain part */
59}

61int brixautofs_repo_listed(const char *list, const char *fqrn) {
62 if (list == NULL || fqrn == NULL) return 0;
63 size_t fl = strlen(fqrn);
64 const char *p = list;
65 while (*p) {
66 while (*p == ',' || *p == ':' || *p == ' ' || *p == '\t') p++;
67 const char *tok = p;
68 while (*p && *p != ',' && *p != ':' && *p != ' ' && *p != '\t') p++;
69 if ((size_t)(p - tok) == fl && strncmp(tok, fqrn, fl) == 0) return 1;
70 }
71 return 0;
72}

174/* 1 iff `path` is a mount point per /proc/self/mountinfo (field 5). FQRNs
175 * never contain the \040-style escapes mountinfo uses, so plain compare. */
176static int af_is_mounted(const char *path) {
177 FILE *f = fopen("/proc/self/mountinfo", "r");
178 if (f == NULL) return 0;
179 char line[1024];
180 int found = 0;
181 while (!found && fgets(line, sizeof(line), f) != NULL) {
182 /* fields: id parent maj:min root mountpoint … */
183 char *save = NULL, *tok = strtok_r(line, " ", &save);
184 for (int i = 0; tok != NULL && i < 4; i++) tok = strtok_r(NULL, " ", &save);
185 if (tok != NULL && strcmp(tok, path) == 0) found = 1;
186 }
187 fclose(f);
188 return found;
189}

197/* mkdir -p (two fixed levels are enough: farm base, then the per-repo dir). */
198static int af_mkdir_p(const char *path) {
199 char buf[768];
200 snprintf(buf, sizeof(buf), "%s", path);
201 for (char *p = buf + 1; *p; p++) {
202 if (*p != '/') continue;
203 *p = '\0';
204 if (mkdir(buf, 0755) != 0 && errno != EEXIST) return -1;
205 *p = '/';
206 }
207 if (mkdir(buf, 0755) != 0 && errno != EEXIST) return -1;
208 return 0;
209}

211/* Unmount `path`: umount2 when root (MNT_DETACH on EBUSY), else fusermount3
212 * (-u, then lazy -uz). Best-effort; returns 0 if the mount is gone. */
213static int af_umount_path(const char *path) {
214 if (geteuid() == 0) {
215 if (umount2(path, 0) == 0) return 0;
216 if (errno == EBUSY && umount2(path, MNT_DETACH) == 0) return 0;
217 if (!af_is_mounted(path)) return 0;
218 }
219 const char *modes[] = { "-u", "-uz" };
220 for (int m = 0; m < 2; m++) {
221 pid_t pid = fork();
222 if (pid == 0) {
223 int devnull = open("/dev/null", O_RDWR);
224 if (devnull >= 0) { dup2(devnull, 1); dup2(devnull, 2); }
225 execlp("fusermount3", "fusermount3", modes[m], path, (char *) NULL);
226 _exit(127);
227 }
228 if (pid > 0) {
229 int st = 0;
230 waitpid(pid, &st, 0);
231 }
232 if (!af_is_mounted(path)) return 0;
233 }
234 return af_is_mounted(path) ? -1 : 0;
235}

237/* Build the child's `-o` string. `cache=` pins a per-repo cache dir (which
238 * also disables brixcvmfs "clever" mode — it must never write onto the
239 * mountpoint under the umbrella); nodev,nosuid harden the nested mount.
240 * $BRIXAUTOFS_CHILD_OPTS appends extra tokens (tests). Must stay < 512
241 * (brixcvmfs parse_opts buffer cap). */
242static void af_child_opts(const char *fqrn, char *out, size_t cap) {
243 const char *extra = getenv("BRIXAUTOFS_CHILD_OPTS");
244 snprintf(out, cap, "cache=%s/%s,nodev,nosuid%s%s%s",
245 g_af.o.cache_base, fqrn,
246 g_af.o.allow_other ? ",allow_other" : "",
247 extra ? "," : "", extra ? extra : "");
248}

250/* Fork/exec the per-repo mount (`brixMount cvmfs <fqrn> <mnt>/<fqrn> -f -o …`)
251 * and poll /proc/self/mountinfo until the nested mount is live. The child runs
252 * -f (no daemonize) so its pid stays ours to reap and its exit before the
253 * mount lands is a definitive failure. Returns the child pid, or -1. */
254static pid_t af_spawn_child(const char *fqrn) {
255 char mntpath[768], opts[512], exe[512];
256 af_child_path(fqrn, mntpath, sizeof(mntpath));
257 af_child_opts(fqrn, opts, sizeof(opts));
258
259 ssize_t el = readlink("/proc/self/exe", exe, sizeof(exe) - 1);
260 if (el > 0) {
261 exe[el] = '\0';
262 } else {
263 const char *env = getenv("BRIXMOUNT_BIN");
264 if (env == NULL) { af_log("cannot resolve own binary path"); return -1; }
265 snprintf(exe, sizeof(exe), "%s", env);
266 }
267
268 if (af_mkdir_p(mntpath) != 0) {
269 af_log("cannot create mountpoint %s: %s", mntpath, strerror(errno));
270 return -1;
271 }
272
273 pid_t pid = fork();
274 if (pid < 0) { af_log("fork: %s", strerror(errno)); return -1; }
275 if (pid == 0) {
276 int devnull = open("/dev/null", O_RDONLY);
277 if (devnull >= 0) dup2(devnull, 0);
278 execl(exe, exe, "cvmfs", fqrn, mntpath, "-f", "-o", opts, (char *) NULL);
279 _exit(127);
280 }
281
282 int deadline_ms = g_af.o.spawn_timeout_s * 1000;
283 for (int waited = 0; waited < deadline_ms; waited += 100) {
284 if (af_is_mounted(mntpath)) return pid;
285 int st = 0;
286 if (waitpid(pid, &st, WNOHANG) == pid) {
287 af_log("mount of %s failed (child exit %d)", fqrn,
288 WIFEXITED(st) ? WEXITSTATUS(st) : -1);
289 return -1;
290 }
291 struct timespec ts = { 0, 100 * 1000 * 1000 };
292 nanosleep(&ts, NULL);
293 }
294 af_log("mount of %s timed out after %ds", fqrn, g_af.o.spawn_timeout_s);
295 kill(pid, SIGTERM);
296 waitpid(pid, NULL, 0);
297 af_umount_path(mntpath); /* in case it landed between poll and kill */
298 return -1;
299}

301/* Admission gate shared by every op: is `name` a repo this umbrella may show
302 * at all? (validity + strict-mount policy; never touches the slot table). */
303static int af_admit(const char *name) {
304 if (!brixautofs_valid_fqrn(name)) return 0;
305 if (g_af.strict && !brixautofs_repo_listed(g_af.repos, name)) return 0;
306 return !g_af.shutting_down;
307}

309/* The mount trigger: drive the slot state machine and block until the child
310 * mount is live in the farm. Called from readlink only — the triggering path
311 * resolution waits (autofs UX); the child's own mount(2) walk happens in the
312 * farm, far away from the umbrella's dentries, so nothing re-enters us.
313 * Returns 0 when <farm>/<name> is a live mount, else -ENOENT. */
314static int af_ensure_repo(const char *name) {
315 if (!af_admit(name)) return -ENOENT;
316
317 pthread_mutex_lock(&g_af.tab.mu);
318 int idx = brixautofs_find_locked(&g_af.tab, name);
319 if (idx >= 0) {
320 /* MOUNTED: done. MOUNTING: another resolution is bringing it up; its
321 * kernel-side lookup already serializes ours, so just answer. */
322 pthread_mutex_unlock(&g_af.tab.mu);
323 return 0;
324 }
325 idx = brixautofs_claim_locked(&g_af.tab, name);
326 pthread_mutex_unlock(&g_af.tab.mu);
327 if (idx < 0) {
328 af_log("repo table full (%d), refusing %s", BRIXAUTOFS_MAX_REPOS, name);
329 return -ENOENT;
330 }
331
332 pid_t pid = af_spawn_child(name); /* blocking: the autofs UX — the
333 * triggering resolution waits */
334 pthread_mutex_lock(&g_af.tab.mu);
335 if (pid > 0) brixautofs_commit_locked(&g_af.tab, idx, pid);
336 else brixautofs_release_locked(&g_af.tab, idx);
337 pthread_mutex_unlock(&g_af.tab.mu);
338 return pid > 0 ? 0 : -ENOENT;
339}

364/* getattr NEVER mounts: an admissible name is a symlink into the farm, so a
365 * bare lstat / colorized ls of ghosts stays cheap. Following the link (stat,
366 * open, cd) goes through readlink — that is where the mount happens. */
367static int af_getattr(const char *path, struct stat *st, struct fuse_file_info *fi) {
368 (void) fi;
369 if (strcmp(path, "/") == 0) { af_fill_dir_stat(st); return 0; }
370 const char *name = path + 1;
371 if (strchr(name, '/') != NULL) return -ENOENT; /* depth 1 only */
372 if (!af_admit(name)) return -ENOENT;
373 af_fill_link_stat(st, name);
374 return 0;
375}

377/* readlink IS the mount trigger: block until the repo is live, then hand the
378 * kernel the farm path to walk into. */
379static int af_readlink(const char *path, char *buf, size_t cap) {
380 const char *name = path + 1;
381 if (strchr(name, '/') != NULL) return -ENOENT;
382 int rc = af_ensure_repo(name);
383 if (rc != 0) return rc;
384 char target[768];
385 af_child_path(name, target, sizeof(target));
386 snprintf(buf, cap, "%s", target);
387 return 0;
388}

396/* readdir("/") = "." ".." ∪ mounted repos ∪ configured repos (ghost listing:
397 * CVMFS_REPOSITORIES + the config.d .conf entries). NEVER spawns a mount. */
398static int af_readdir(const char *path, void *buf, fuse_fill_dir_t fill,
399 off_t off, struct fuse_file_info *fi,
400 enum fuse_readdir_flags flags) {
401 (void) off; (void) fi; (void) flags;
402 fill(buf, ".", NULL, 0, 0);
403 fill(buf, "..", NULL, 0, 0);
404 if (strcmp(path, "/") != 0) return 0; /* unreachable: opendir admits "/" only */
405
406 char seen[BRIXAUTOFS_MAX_REPOS * 2][BRIXAUTOFS_FQRN_MAX];
407 int nseen = 0;
408
409 pthread_mutex_lock(&g_af.tab.mu);
410 for (int i = 0; i < BRIXAUTOFS_MAX_REPOS; i++) {
411 if (g_af.tab.slot[i].st == BRIXAUTOFS_FREE) continue;
412 snprintf(seen[nseen], BRIXAUTOFS_FQRN_MAX, "%s", g_af.tab.slot[i].fqrn);
413 fill(buf, seen[nseen++], NULL, 0, 0);
414 }
415 pthread_mutex_unlock(&g_af.tab.mu);
416
417 /* ghost entries: configured but not (yet) mounted */
418 for (int src = 0; src < 2; src++) {
419 for (int i = 0; ; i++) {
420 char name[BRIXAUTOFS_FQRN_MAX] = "";
421 if (src == 0) {
422 if (i >= g_af.nghost) break;
423 snprintf(name, sizeof(name), "%s", g_af.ghost[i]);
424 } else {
425 /* i-th token of the repos list */
426 const char *p = g_af.repos;
427 int tok = 0, got = 0;
428 while (*p && !got) {
429 while (*p == ',' || *p == ':' || *p == ' ') p++;
430 const char *s = p;
431 while (*p && *p != ',' && *p != ':' && *p != ' ') p++;
432 if (p == s) break;
433 if (tok++ == i) {
434 snprintf(name, sizeof(name), "%.*s", (int)(p - s), s);
435 got = 1;
436 }
437 }
438 if (!got) break;
439 }
440 int dup = 0;
441 for (int k = 0; k < nseen && !dup; k++)
442 if (strcmp(seen[k], name) == 0) dup = 1;
443 if (dup || !brixautofs_valid_fqrn(name)) continue;
444 if (nseen >= (int)(sizeof(seen) / sizeof(seen[0]))) break;
445 snprintf(seen[nseen], BRIXAUTOFS_FQRN_MAX, "%s", name);
446 fill(buf, seen[nseen++], NULL, 0, 0);
447 }
448 }
449 return 0;
450}

467/* Reap dead children; a slot whose child died gets its (possibly wedged)
468 * mount detached and the slot freed so the repo can remount on next access. */
469static void af_reap_children(void) {
470 for (;;) {
471 int st = 0;
472 pid_t pid = waitpid(-1, &st, WNOHANG);
473 if (pid <= 0) return;
474 pthread_mutex_lock(&g_af.tab.mu);
475 int idx = brixautofs_find_pid_locked(&g_af.tab, pid);
476 char fqrn[BRIXAUTOFS_FQRN_MAX] = "";
477 if (idx >= 0) {
478 snprintf(fqrn, sizeof(fqrn), "%s", g_af.tab.slot[idx].fqrn);
479 brixautofs_release_locked(&g_af.tab, idx);
480 }
481 pthread_mutex_unlock(&g_af.tab.mu);
482 if (fqrn[0] == '\0') continue;
483 char mntpath[768];
484 af_child_path(fqrn, mntpath, sizeof(mntpath));
485 if (af_is_mounted(mntpath)) af_umount_path(mntpath);
486 if (!g_af.shutting_down) af_log("child for %s exited", fqrn);
487 }
488}

490/* Unmount every child, then reap. Idempotent — runs from the control thread
491 * on SIGTERM (while the umbrella's worker threads still serve the path walks)
492 * and again after fuse_loop returns. */
493static void af_teardown_children(void) {
494 g_af.shutting_down = 1;
495 for (int i = 0; i < BRIXAUTOFS_MAX_REPOS; i++) {
496 pthread_mutex_lock(&g_af.tab.mu);
497 char fqrn[BRIXAUTOFS_FQRN_MAX] = "";
498 pid_t pid = 0;
499 if (g_af.tab.slot[i].st != BRIXAUTOFS_FREE) {
500 snprintf(fqrn, sizeof(fqrn), "%s", g_af.tab.slot[i].fqrn);
501 pid = g_af.tab.slot[i].pid;
502 brixautofs_release_locked(&g_af.tab, i);
503 }
504 pthread_mutex_unlock(&g_af.tab.mu);
505 if (fqrn[0] == '\0') continue;
506
507 char mntpath[768];
508 af_child_path(fqrn, mntpath, sizeof(mntpath));
509 af_umount_path(mntpath);
510 if (pid > 0) {
511 int st = 0;
512 for (int w = 0; w < 30 && waitpid(pid, &st, WNOHANG) == 0; w++) {
513 struct timespec ts = { 0, 100 * 1000 * 1000 };
514 nanosleep(&ts, NULL);
515 }
516 if (waitpid(pid, &st, WNOHANG) == 0) {
517 kill(pid, SIGTERM);
518 waitpid(pid, &st, 0);
519 }
520 }
521 }
522}

524/* Control thread: owns the self-pipe. 'C' → reap; 'T' → teardown children
525 * (worker threads still live to serve the umount path walks), then end the
526 * umbrella session; unmounting our own mountpoint makes every worker's
527 * /dev/fuse read return ENODEV, so the loop exits deterministically. */
528static void *af_control_thread(void *arg) {
529 (void) arg;
530 for (;;) {
531 char c = 0;
532 ssize_t n = read(g_af.sigpipe[0], &c, 1);
533 if (n <= 0) {
534 if (n < 0 && errno == EINTR) continue;
535 return NULL;
536 }
537 if (c == 'C') { af_reap_children(); continue; }
538 af_teardown_children();
539 fuse_exit(g_af.fuse);
540 af_umount_path(g_af.mnt);
541 return NULL;
542 }
543}

545/* Idle expiry (root only): umount2(MNT_EXPIRE) two-phase — first call marks
546 * (EAGAIN), an untouched repo expires on the next tick (0), any access in
547 * between clears the mark in-kernel. The reaper frees the slot when the
548 * child exits after its session ends. EPERM ⇒ no CAP_SYS_ADMIN ⇒ disable. */
549static void *af_idle_thread(void *arg) {
550 (void) arg;
551 unsigned tick = (unsigned) g_af.o.idle_s / 2;
552 if (tick == 0) tick = 1;
553 for (;;) {
554 sleep(tick);
555 if (g_af.shutting_down) return NULL;
556 char repos[BRIXAUTOFS_MAX_REPOS][BRIXAUTOFS_FQRN_MAX];
557 int n = 0;
558 pthread_mutex_lock(&g_af.tab.mu);
559 for (int i = 0; i < BRIXAUTOFS_MAX_REPOS; i++)
560 if (g_af.tab.slot[i].st == BRIXAUTOFS_MOUNTED)
561 snprintf(repos[n++], BRIXAUTOFS_FQRN_MAX, "%s", g_af.tab.slot[i].fqrn);
562 pthread_mutex_unlock(&g_af.tab.mu);
563 for (int i = 0; i < n; i++) {
564 char mntpath[768];
565 af_child_path(repos[i], mntpath, sizeof(mntpath));
566 if (umount2(mntpath, MNT_EXPIRE) == 0) {
567 af_log("idle-expired %s", repos[i]);
568 } else if (errno == EPERM) {
569 af_log("idle expiry needs CAP_SYS_ADMIN — disabled");
570 return NULL;
571 }
572 /* EAGAIN = marked for next tick; EBUSY = in use: both fine */
573 }
574 }
575}

579static void af_opts_o_list(char *list, autofs_opts_t *o) {
580 char *save = NULL;
581 for (char *t = strtok_r(list, ",", &save); t; t = strtok_r(NULL, ",", &save)) {

585 snprintf(o->cache_base, sizeof(o->cache_base), "%s", t + 10);
586 else if (strncmp(t, "mntbase=", 8) == 0)

588 else if (strncmp(t, "repos=", 6) == 0)
589 snprintf(o->repos, sizeof(o->repos), "%s", t + 6);
590 else if (strcmp(t, "allow_other") == 0) o->allow_other = 1;
591 else { /* forward to the umbrella's libfuse */
592 size_t cur = strlen(o->fuse_extra);
593 snprintf(o->fuse_extra + cur, sizeof(o->fuse_extra) - cur,
594 "%s%s", cur ? "," : "", t);
595 }
596 }

600 memset(o, 0, sizeof(*o));

612 snprintf(obuf, sizeof(obuf), "%s", argv[i] + 2);
613 af_opts_o_list(obuf, o);
614 } else if (strcmp(argv[i], "-f") == 0) {
615 o->foreground = 1;
616 } else if (strcmp(argv[i], "-d") == 0) {
617 o->foreground = 1;
618 o->debug = 1;
619 }

623/* Ghost list: config.d .conf basenames are the operator-configured repos. */
624static void af_scan_configured(const char *etc_root) {
625 char dd[512];
626 snprintf(dd, sizeof(dd), "%s/config.d", etc_root[0] ? etc_root : "/etc/cvmfs");
627 DIR *d = opendir(dd);
628 if (d == NULL) return;
629 struct dirent *e;
630 while ((e = readdir(d)) != NULL && g_af.nghost < BRIXAUTOFS_MAX_REPOS) {
631 const char *dot = strrchr(e->d_name, '.');
632 if (dot == NULL || strcmp(dot, ".conf") != 0) continue;
633 char name[BRIXAUTOFS_FQRN_MAX];
634 snprintf(name, sizeof(name), "%.*s", (int)(dot - e->d_name), e->d_name);
635 if (brixautofs_valid_fqrn(name))
636 snprintf(g_af.ghost[g_af.nghost++], BRIXAUTOFS_FQRN_MAX, "%s", name);
637 }
638 closedir(d);
639}

641/* brixautofs entry — dispatched by the brixMount umbrella:
642 * brixMount autofs <etc-root|-> <mountdir> [-f|-d] [-o idle=…,…] */
643int brixautofs_main(int argc, char **argv) {
644 if (argc < 3) {
645 fprintf(stderr,
646 "usage: brixMount autofs <etc-root|-> <mountdir> [-f|-d] [-o opts]\n"
647 " -o idle=<s> unmount idle repos after <s> seconds (default 600, 0=off)\n"
648 " -o timeout=<s> per-repo mount bring-up timeout (default 60)\n"
649 " -o cachebase=<d> child cache dirs under <d>/<fqrn> (default /var/lib/brixcvmfs)\n"
650 " -o mntbase=<d> child mount farm; repos appear as symlinks to <d>/<fqrn>\n"
651 " (default <cachebase>/.mnt)\n"
652 " -o repos=a:b restrict/ghost-list repos (overrides CVMFS_REPOSITORIES)\n"
653 " -o allow_other let other users read the mounts (needs root or user_allow_other)\n");
654 return 2;
655 }
656
657 memset(&g_af.tab, 0, sizeof(g_af.tab));
658 brixautofs_table_init(&g_af.tab);
659 if (strcmp(argv[1], "-") != 0)
660 snprintf(g_af.etc, sizeof(g_af.etc), "%s", argv[1]);
661 snprintf(g_af.mnt, sizeof(g_af.mnt), "%s", argv[2]);
662 size_t ml = strlen(g_af.mnt);
663 while (ml > 1 && g_af.mnt[ml - 1] == '/') g_af.mnt[--ml] = '\0';
664 af_parse_opts(argc, argv, 3, &g_af.o);
665
666 /* child mount farm: an absolute dir the umbrella never path-walks itself */
667 char farm_raw[512];
668 if (g_af.o.mnt_base[0])
669 snprintf(farm_raw, sizeof(farm_raw), "%s", g_af.o.mnt_base);
670 else
671 snprintf(farm_raw, sizeof(farm_raw), "%s/.mnt", g_af.o.cache_base);
672 if (af_mkdir_p(farm_raw) != 0) {
673 af_log("cannot create mount farm %s: %s", farm_raw, strerror(errno));
674 return 1;
675 }
676 char farm_rp[PATH_MAX];
677 const char *farm_src =
678 realpath(farm_raw, farm_rp) != NULL ? farm_rp : farm_raw;
679 if (strlen(farm_src) >= sizeof(g_af.farm)) {
680 af_log("mount farm path too long: %s", farm_src);
681 return 1;
682 }
683 snprintf(g_af.farm, sizeof(g_af.farm), "%.*s",
684 (int)(sizeof(g_af.farm) - 1), farm_src);
685 size_t mnl = strlen(g_af.mnt);
686 if (strncmp(g_af.farm, g_af.mnt, mnl) == 0
687 && (g_af.farm[mnl] == '/' || g_af.farm[mnl] == '\0')) {
688 af_log("mount farm %s must not live under the umbrella %s", g_af.farm, g_af.mnt);
689 return 1;
690 }
691
692 /* children resolve their own repo config from the same etc root */
693 if (g_af.etc[0]) setenv("BRIXCVMFS_ETC", g_af.etc, 1);
694
695 cvmfs_conf_t cf;
696 cvmfs_conf_init(&cf);
697 cvmfs_conf_load_cascade(&cf, g_af.etc[0] ? g_af.etc : NULL, "");
698 const char *strict = cvmfs_conf_get(&cf, "CVMFS_STRICT_MOUNT");
699 g_af.strict = strict != NULL
700 && (strcmp(strict, "yes") == 0 || strcmp(strict, "on") == 0
701 || strcmp(strict, "1") == 0);
702 const char *repos = cvmfs_conf_get(&cf, "CVMFS_REPOSITORIES");
703 snprintf(g_af.repos, sizeof(g_af.repos), "%s",
704 g_af.o.repos[0] ? g_af.o.repos : (repos ? repos : ""));
705 if (g_af.strict && g_af.repos[0] == '\0')
706 af_log("warning: CVMFS_STRICT_MOUNT set with no CVMFS_REPOSITORIES — "
707 "every mount will be refused");
708 af_scan_configured(g_af.etc);
709
710 mkdir(g_af.mnt, 0755); /* mount(2) will fail loudly if unusable */
711
712 /* umbrella libfuse args (mountpoint goes to fuse_mount, not the arg list) */
713 char oarg[600];
714 snprintf(oarg, sizeof(oarg), "fsname=brixautofs,subtype=cvmfs%s%s%s",
715 g_af.o.allow_other ? ",allow_other" : "",
716 g_af.o.fuse_extra[0] ? "," : "", g_af.o.fuse_extra);
717 char *fargv[8];
718 int fargc = 0;
719 fargv[fargc++] = argv[0];
720 if (g_af.o.debug) fargv[fargc++] = (char *) "-d";
721 fargv[fargc++] = (char *) "-o";
722 fargv[fargc++] = oarg;
723 struct fuse_args fargs = FUSE_ARGS_INIT(fargc, fargv);
724
725 g_af.fuse = fuse_new(&fargs, &af_ops, sizeof(af_ops), NULL);
726 if (g_af.fuse == NULL) { af_log("fuse_new failed"); return 1; }
727 if (fuse_mount(g_af.fuse, g_af.mnt) != 0) {
728 af_log("cannot mount umbrella on %s", g_af.mnt);
729 fuse_destroy(g_af.fuse);
730 return 1;
731 }
732 if (fuse_daemonize(g_af.o.foreground) != 0) {
733 af_log("daemonize failed");
734 fuse_unmount(g_af.fuse);
735 fuse_destroy(g_af.fuse);
736 return 1;
737 }
738
739 /* threads AFTER daemonize (fork drops them); self-pipe carries signals */
740 if (pipe(g_af.sigpipe) != 0) {
741 af_log("pipe: %s", strerror(errno));
742 fuse_unmount(g_af.fuse);
743 fuse_destroy(g_af.fuse);
744 return 1;
745 }
746 struct sigaction sa;
747 memset(&sa, 0, sizeof(sa));
748 sa.sa_handler = af_signal;
749 sa.sa_flags = SA_RESTART | SA_NOCLDSTOP;
750 sigaction(SIGTERM, &sa, NULL);
751 sigaction(SIGINT, &sa, NULL);
752 sigaction(SIGCHLD, &sa, NULL);
753 signal(SIGPIPE, SIG_IGN);
754
755 pthread_t ctl, idle;
756 pthread_create(&ctl, NULL, af_control_thread, NULL);
757 int have_idle = g_af.o.idle_s > 0;
758 if (have_idle) pthread_create(&idle, NULL, af_idle_thread, NULL);
759
760 af_log("serving %s (etc=%s idle=%ds strict=%d)", g_af.mnt,
761 g_af.etc[0] ? g_af.etc : "/etc/cvmfs", g_af.o.idle_s, g_af.strict);
762 int rc = fuse_loop_mt(g_af.fuse, 0);
763
764 /* normal exit path (external unmount): tear children down too */
765 af_teardown_children();
766 close(g_af.sigpipe[1]); /* control thread read() returns 0 → exits */
767 pthread_join(ctl, NULL);
768 if (have_idle) { pthread_cancel(idle); pthread_join(idle, NULL); }
769 fuse_unmount(g_af.fuse);
770 fuse_destroy(g_af.fuse);
771 return rc == 0 ? 0 : 1;
772}
