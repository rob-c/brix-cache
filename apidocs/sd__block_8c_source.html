1/*
2 * sd_block.c — block-device storage driver (shared by the nginx server and the
3 * userland clients).
4 *
5 * WHAT: brix_sd_block_driver — a backend for writing/reading a raw block
6 * device (or a file used as one) in place. The raw byte I/O is identical to
7 * POSIX, so the pread/pwrite/preadv/fsync slots delegate to the POSIX
8 * driver; the only block-specific behaviour is `fstat`, which reports the
9 * true device capacity via BLKGETSIZE64 (a block device's struct.st_size is
10 * 0), and `open`, which never creates or truncates the device.
11 * WHY: block was previously implemented only in client/lib/vfs_block.c — a
12 * second storage driver outside src/. This is the single home: both the
13 * client (block:// copy endpoints) and, in future, a block-backed server
14 * export use the same driver. ngx-free (dual-build via sd.h's
15 * XRDPROTO_NO_NGX fallback) so the client links it from libxrdproto.
16 * HOW: flat, POD-pointer-only vtable; no instance/namespace ops (a block device
17 * has no directory namespace). The unconfined open helper mirrors
18 * brix_sd_posix_open_unconfined but is exposed under the block name for
19 * call-site clarity.
20 */

67/* sd_block_fstat — like POSIX fstat, but a block device reports st_size == 0, so
68 * query the real capacity via BLKGETSIZE64 when the fd is a block device. */
69static ngx_int_t
70sd_block_fstat(brix_sd_obj_t *obj, brix_sd_stat_t *out)
71{
72 struct stat sb;
73
74 if (fstat(obj->fd, &sb) != 0) {
75 return NGX_ERROR;
76 }
77 ngx_memzero(out, sizeof(*out));
78 out->size = sb.st_size;
79 out->mtime = sb.st_mtime;
80 out->ctime = sb.st_ctime;
81 out->mode = sb.st_mode;
82 out->ino = sb.st_ino;
83 out->is_dir = S_ISDIR(sb.st_mode) ? 1 : 0;
84 out->is_reg = S_ISREG(sb.st_mode) ? 1 : 0;
85
86#ifdef BLKGETSIZE64
87 if (S_ISBLK(sb.st_mode)) {
88 uint64_t sz = 0;
89 if (ioctl(obj->fd, BLKGETSIZE64, &sz) == 0) {
90 out->size = (off_t) sz;
91 }
92 }
93#endif
94 return NGX_OK;
95}

97/* brix_sd_block_open_unconfined — open a block device (no O_CREAT/O_TRUNC: the
98 * device exists and must not be re-created or zeroed). Returns an fd or -1. */
99int
100brix_sd_block_open_unconfined(const char *path, int sd_flags, mode_t mode)
101{
102 /* Strip create/truncate intent — a block device is opened in place. */
103 sd_flags &= ~(BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC);
104 return brix_sd_posix_open_unconfined(path, sd_flags, mode);
