9/*
10 * Locate a CGI opaque key on a token boundary (start / '&' / '?') and return its
11 * value span [*val, *val+*vlen). Shared by the compression and ZIP-member
12 * negotiations, which both scan the same path?opaque carrier. Returns 1 when the
13 * key is present on a boundary, 0 otherwise.
14 */
15static int
16open_cgi_find_value(const char *opaque, const char *key,
17 const char **val, size_t *vlen)
18{
19 const char *p = strstr(opaque, key);
20 const char *v, *end;
21
22 if (p == NULL) {
23 return 0;
24 }
25 if (p != opaque && p[-1] != '&' && p[-1] != '?') {
26 return 0; /* not on a key boundary */
27 }
28
29 v = p + strlen(key);
30 end = v;
31 while (*end != '\0' && *end != '&') {
32 end++;
33 }
34 *val = v;
35 *vlen = (size_t) (end - v);
36 return 1;
37}

42 * opaque. Returns the codec ordinal (brix_codec_id_t) to use for this
43 * handle, or BRIX_CODEC_IDENTITY (0) when compression is disabled for the
44 * direction, no "?xrootd.compress=" opaque is present, or the requested

114/* Intra-archive traversal / absolute-path guard for a ZIP member name: reject a
115 * leading '/', a bare "..", a leading "../", any embedded "/../", or a trailing
116 * "/..". Returns 1 (unsafe → caller rejects with kXR_ArgInvalid), else 0. */
117static int
118open_zip_member_path_unsafe(const char *out, size_t vlen)
119{
120 if (out[0] == '/') {
121 return 1;

123 if (ngx_strcmp(out, "..") == 0) {
124 return 1;

130 return 1;

133 return 1; /* trailing "/.." */
134 }
