1/*
2 * sock.c — blocking TCP with poll(2) deadlines.
3 *
4 * WHAT: connect-with-timeout (happy-path v4/v6 via getaddrinfo) plus full-read /
5 * full-write helpers that bound each syscall with poll(2).
6 * WHY: The client is deliberately synchronous (no nginx event loop, no XrdCl
7 * PostMaster); poll gives us per-operation timeouts without going async.
8 * HOW: The socket is non-blocking only during connect (so we can time it out),
9 * then switched back to blocking; reads/writes poll first, then do the
10 * blocking syscall, looping on partial transfers and EINTR.
11 */

27static int
28set_blocking(int fd, int blocking)
29{
30 int flags = fcntl(fd, F_GETFL, 0);
31 if (flags < 0) {
32 return -1;
33 }
34 if (blocking) {
35 flags &= ~O_NONBLOCK;
36 } else {
37 flags |= O_NONBLOCK;
38 }
39 return fcntl(fd, F_SETFL, flags);
40}

42/*
43 * Apply latency + liveness socket options (best-effort; failures are ignored).
44 * TCP_NODELAY kills Nagle so small request frames go out immediately (matters on a
45 * pipelined, high-RTT link). SO_KEEPALIVE + the keep* triad let the OS notice a
46 * silently-dead peer (the "wifi vanished" case) as a backstop to the app-level
47 * kXR_ping heartbeat. Tunables are conservative so a brief stall is not mistaken
48 * for a dead peer.
49 */
50void
51brix_sock_tune(int fd)
52{
53 int on = 1;
54 setsockopt(fd, IPPROTO_TCP, TCP_NODELAY, &on, sizeof(on));
55 setsockopt(fd, SOL_SOCKET, SO_KEEPALIVE, &on, sizeof(on));
56#ifdef TCP_KEEPIDLE
57 int idle = 30, intvl = 10, cnt = 3; /* ~60s to declare a dead peer */
58 setsockopt(fd, IPPROTO_TCP, TCP_KEEPIDLE, &idle, sizeof(idle));
59 setsockopt(fd, IPPROTO_TCP, TCP_KEEPINTVL, &intvl, sizeof(intvl));
60 setsockopt(fd, IPPROTO_TCP, TCP_KEEPCNT, &cnt, sizeof(cnt));
61#endif
62}

64/*
65 * Per-candidate connect cap (ms). When more resolved addresses remain to try,
66 * a single attempt is bounded by this so a black-holed family (the classic
67 * broken-IPv6-on-dual-stack case) yields quickly to the next family instead of
68 * burning the caller's full (30s) budget on a dead address — RFC 8305 Happy
69 * Eyeballs in spirit, sequential rather than racing. The FINAL candidate always
70 * gets the full timeout, so a single-address host's semantics are unchanged and
71 * a merely-slow (not dead) sole path is never abandoned early.
72 */
73#define XRDC_CONNECT_ATTEMPT_MS 5000
74
75/*
76 * Open one connected, tuned, blocking socket to a single resolved address.
77 * Returns the fd on success, or -1 on any failure (socket / connect / timeout) —
78 * the caller simply walks to the next candidate. The socket is non-blocking only
79 * for the connect (so poll(2) can time it out), then switched back to blocking.
80 * timeout_ms bounds the EINPROGRESS settle.
81 */
82static int
83connect_one(const struct addrinfo *ai, int timeout_ms)
84{
85 int fd = socket(ai->ai_family, ai->ai_socktype, ai->ai_protocol);
86 if (fd < 0) {
87 return -1;
88 }
89 if (set_blocking(fd, 0) < 0) {
90 close(fd);
91 return -1;
92 }
93
94 /* connect()==0 means it settled immediately; only EINPROGRESS needs poll. */
95 if (connect(fd, ai->ai_addr, ai->ai_addrlen) != 0) {
96 if (errno != EINPROGRESS) {
97 close(fd);
98 return -1;
99 }
100
101 struct pollfd pfd;
102 int pr;
103 pfd.fd = fd;
104 pfd.events = POLLOUT;
105 do {
106 pr = poll(&pfd, 1, timeout_ms);
107 } while (pr < 0 && errno == EINTR);
108 if (pr <= 0) {
109 int e = (pr == 0) ? ETIMEDOUT : errno; /* timeout vs poll error */
110 close(fd);
111 errno = e; /* report the real cause (close may clobber) */
112 return -1;
113 }
114
115 int err = 0;
116 socklen_t elen = sizeof(err);
117 if (getsockopt(fd, SOL_SOCKET, SO_ERROR, &err, &elen) < 0) {
118 close(fd);
119 return -1;
120 }
121 if (err != 0) {
122 close(fd);
123 errno = err; /* propagate SO_ERROR (ECONNREFUSED/…) so callers
124 * can distinguish a definitive refusal from a stall */
125 return -1;
126 }
127 }
128
129 if (set_blocking(fd, 1) < 0) {
130 close(fd);
131 return -1;
132 }
133 brix_sock_tune(fd); /* TCP_NODELAY + keepalive (best-effort) */
134 return fd;
135}

137/*
138 * Address family (AF_INET / AF_INET6) of a connected socket, via getsockname;
139 * AF_UNSPEC if unknown. Lets a wire error be attributed to the right family so
140 * only an IPv6 failure trips the downgrade.
141 */
142static int
143fd_family(int fd)
144{
145 struct sockaddr_storage ss;
146 socklen_t sl = sizeof(ss);
147 if (fd < 0 || getsockname(fd, (struct sockaddr *) &ss, &sl) != 0) {
148 return AF_UNSPEC;
149 }
150 return ss.ss_family;
151}

153/*
154 * Resolve host with the given family hint and connect to the first candidate
155 * that answers (happy-eyeballs, bounded per non-final attempt). On a working
156 * IPv4 reached only after an IPv6 candidate failed, demote the session to
157 * IPv4-only (the connect-time trigger). Returns the fd, or -1 with *st set.
158 */
159static int
160connect_resolved(const char *host, int port, int timeout_ms, int family,
161 brix_status *st)
162{
163 struct addrinfo hints, *res = NULL, *ai;
164 char portstr[16];
165 int gai;
166 int v6_failed = 0; /* an IPv6 candidate failed this resolve */
167 int last_errno = 0; /* real errno of the last failed candidate */
168
169 memset(&hints, 0, sizeof(hints));
170 hints.ai_family = family;
171 hints.ai_socktype = SOCK_STREAM;
172 /* AI_NUMERICSERV: the port is always a decimal string, so tell the resolver
173 * never to consult /etc/services (saves an NSS lookup on every connect).
174 * AI_ADDRCONFIG: only return an address family the host actually has
175 * configured, so a v4-only box is not handed a AAAA candidate it would burn
176 * a full connect attempt failing on before falling back to IPv4. */
177 hints.ai_flags = AI_NUMERICSERV | AI_ADDRCONFIG;
178 snprintf(portstr, sizeof(portstr), "%d", port);
179
180 gai = getaddrinfo(host, portstr, &hints, &res);
181 if (gai != 0) {
182 /* EAI_AGAIN is a transient resolver hiccup (keep it retryable as a
183 * transport fault); every other failure means the name has no address
184 * and never will on a retry, so mark it permanent (XRDC_ERESOLVE) and
185 * let the resilient loop fail fast instead of spinning the stall window
186 * on a dead endpoint. */
187 int code = (gai == EAI_AGAIN) ? XRDC_ESOCK : XRDC_ERESOLVE;
188 brix_status_set(st, code, 0, "resolve %s:%d: %s",
189 host, port, gai_strerror(gai));
190 return -1;
191 }
192
193 for (ai = res; ai != NULL; ai = ai->ai_next) {
194 /* Bound a non-final attempt so a dead family yields promptly; the last
195 * candidate keeps the full budget (see XRDC_CONNECT_ATTEMPT_MS). */
196 int attempt_ms =
197 (ai->ai_next != NULL && timeout_ms > XRDC_CONNECT_ATTEMPT_MS)
198 ? XRDC_CONNECT_ATTEMPT_MS : timeout_ms;
199 int fd = connect_one(ai, attempt_ms);
200
201 if (fd >= 0) {
202 if (ai->ai_family == AF_INET && v6_failed) {
203 brix_netpref_demote_ipv6(host);
204 }
205 freeaddrinfo(res);
206 return fd;
207 }
208 last_errno = errno; /* capture before the next attempt / freeaddrinfo clobbers it */
209 if (ai->ai_family == AF_INET6) {
210 v6_failed = 1;
211 }
212 }
213
214 freeaddrinfo(res);
215 brix_status_set(st, XRDC_ESOCK, last_errno, "connect %s:%d failed", host, port);
216 return -1;
217}

219int
220brix_tcp_connect(const char *host, int port, int timeout_ms, brix_status *st)
221{
222 /* Env-proxy tunnelling: if http_proxy/https_proxy/all_proxy names a proxy for
223 * this target (and no_proxy does not exempt it), connect to the proxy and open
224 * an HTTP CONNECT tunnel to host:port, then hand the tunnelled fd back so TLS /
225 * the XRootD handshake run over it unchanged. Env proxy overrides any in-config
226 * proxy (agreed policy). No env proxy → the direct path below, untouched. */
227 brix_proxy_t px;
228 if (brix_proxy_resolve(NULL, host, port, &px)) {
229 brix_proxy_report(&px, host, port);
230 int pfd = connect_resolved(px.host, px.port, timeout_ms, brix_netpref_family(), st);
231 if (pfd < 0)
232 pfd = connect_resolved(px.host, px.port, timeout_ms, AF_UNSPEC, st);
233 if (pfd < 0) {
234 brix_status_set(st, XRDC_ESOCK, 0, "connect proxy %s:%d failed",
235 px.host, px.port);
236 return -1;
237 }
238 char perr[160];
239 if (brix_proxy_connect_tunnel(pfd, host, port, timeout_ms, perr, sizeof(perr)) != 0) {
240 close(pfd);
241 brix_status_set(st, XRDC_ESOCK, 0, "proxy CONNECT %s:%d via %s:%d: %s",
242 host, port, px.host, px.port, perr);
243 return -1;
244 }
245 brix_sock_tune(pfd);
246 return pfd;
247 }
248
249 /* AF_UNSPEC normally; AF_INET once this session has demoted to IPv4-only
250 * (netpref.c) — either because an IPv6 connect failed where IPv4 worked, or
251 * because an established IPv6 connection failed over the wire. Demotion makes
252 * the resolver omit v6 records, so no further v6 connect timeout is paid. */
253 int family = brix_netpref_family();
254 int fd = connect_resolved(host, port, timeout_ms, family, st);
255
256 /* Self-heal: we are demoted to IPv4-only but IPv4 will not connect here (no
257 * A record, or v4 refused/timed out). That happens if a transient IPv6 wire
258 * error optimistically demoted what is really an IPv6-only host. Revert to
259 * dual-stack and retry so the connection still comes up. */
260 if (fd < 0 && family == AF_INET) {
261 brix_netpref_undo_demote("the IPv4-only path failed");
262 fd = connect_resolved(host, port, timeout_ms, AF_UNSPEC, st);
263 }
264 return fd;
265}

267static int
268plain_read_full(brix_io *io, void *buf, size_t n, brix_status *st)
269{
270 int fd = io->fd;
271 int timeout_ms = io->timeout_ms;
272 uint8_t *p = (uint8_t *) buf;
273 size_t got = 0;
274
275 while (got < n) {
276 struct pollfd pfd;
277 ssize_t r;
278 int pr;
279
280 pfd.fd = fd;
281 pfd.events = POLLIN;
282 do {
283 /* Phase 40 (a): observe a cooperative cancel even when blocked on a
284 * stalled peer — a SIGINT interrupts poll (EINTR), and the next pass
285 * bails promptly instead of silently re-arming for the full timeout. */
286 if (brix_copy_quit_requested()) {
287 brix_status_set(st, XRDC_ESOCK, EINTR,
288 "transfer cancelled (signal)");
289 return -1;
290 }
291 pr = poll(&pfd, 1, timeout_ms);
292 } while (pr < 0 && errno == EINTR);
293
294 if (pr == 0) {
295 brix_status_set(st, XRDC_ESOCK, ETIMEDOUT, "read timed out");
296 return -1;
297 }
298 if (pr < 0) {
299 brix_status_set(st, XRDC_ESOCK, errno, "poll(read): %s", strerror(errno));
300 return -1;
301 }
302
303 r = read(fd, p + got, n - got);
304 if (r == 0) {
305 brix_status_set(st, XRDC_ESOCK, 0,
306 "connection closed by peer (read %zu/%zu)", got, n);
307 return -1;
308 }
309 if (r < 0) {
310 if (errno == EINTR || errno == EAGAIN) {
311 continue;
312 }
313 brix_status_set(st, XRDC_ESOCK, errno, "read: %s", strerror(errno));
314 return -1;
315 }
316 got += (size_t) r;
317 }
318 return 0;
319}

321static int
322plain_write_full(brix_io *io, const void *buf, size_t n, brix_status *st)
323{
324 int fd = io->fd;
325 int timeout_ms = io->timeout_ms;
326 const uint8_t *p = (const uint8_t *) buf;
327 size_t sent = 0;
328
329 while (sent < n) {
330 struct pollfd pfd;
331 ssize_t w;
332 int pr;
333
334 pfd.fd = fd;
335 pfd.events = POLLOUT;
336 do {
337 /* Phase 40 (a): prompt cooperative cancel (see brix_read_full). */
338 if (brix_copy_quit_requested()) {
339 brix_status_set(st, XRDC_ESOCK, EINTR,
340 "transfer cancelled (signal)");
341 return -1;
342 }
343 pr = poll(&pfd, 1, timeout_ms);
344 } while (pr < 0 && errno == EINTR);
345
346 if (pr == 0) {
347 brix_status_set(st, XRDC_ESOCK, ETIMEDOUT, "write timed out");
348 return -1;
349 }
350 if (pr < 0) {
351 brix_status_set(st, XRDC_ESOCK, errno, "poll(write): %s", strerror(errno));
352 return -1;
353 }
354
355 /* send() with MSG_NOSIGNAL: a write to a peer that has gone away returns
356 * EPIPE instead of raising SIGPIPE, which would otherwise KILL a one-shot
357 * tool (xrdcp) on a flaky/severed link. Treated as a normal transport
358 * error so the caller's reconnect/retry logic handles it. */
359 w = send(fd, p + sent, n - sent, MSG_NOSIGNAL);
360 if (w < 0) {
361 if (errno == EINTR || errno == EAGAIN) {
362 continue;
363 }
364 brix_status_set(st, XRDC_ESOCK, errno, "write: %s", strerror(errno));
365 return -1;
366 }
367 sent += (size_t) w;
368 }
369 return 0;
370}

372/*
373 * A transport failure (XRDC_ESOCK) that is NOT a user-driven cancel (EINTR) is
374 * evidence the wire is bad; if this connection is IPv6 it trips the session-wide
375 * IPv4 downgrade (netpref.c) so the reconnect comes back over IPv4. This is the
376 * single chokepoint for both cleartext and TLS byte I/O, so it covers every
377 * tool — root://, WebDAV, and S3 alike.
378 */
379static void
380note_io_failure(brix_io *io, int rc, const brix_status *st)
381{
382 if (rc < 0 && st->kxr == XRDC_ESOCK && st->sys_errno != EINTR) {
383 brix_netpref_note_wire_error(fd_family(io->fd));
384 }
385}
