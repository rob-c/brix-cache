19put_be(unsigned char *p, unsigned long long v, int width)
20{
21 for (int i = width - 1; i >= 0; i--) {

37int
38fp_replay_record_start(const char *path)
39{
40 pthread_mutex_lock(&g_rec_lock);
41 if (g_rec_fd >= 0) {
42 close(g_rec_fd);
43 g_rec_fd = -1;
44 }

76fp_replay_record(int is_up, unsigned long long ts_ms,
77 const unsigned char *b, size_t n)
78{
79 pthread_mutex_lock(&g_rec_lock);
80 if (g_rec_fd >= 0) {
81 unsigned char h[13];
82 h[0] = is_up ? 'u' : 'd';
83 put_be(h + 1, ts_ms, 8);
84 put_be(h + 9, (unsigned long long) n, 4);
85 if (write(g_rec_fd, h, sizeof(h)) == (ssize_t) sizeof(h) && n > 0) {
86 ssize_t w = write(g_rec_fd, b, n);
87 (void) w;
88 }

93int
94fp_replay_load(const char *path, fp_replay_store *s)
95{
96 memset(s, 0, sizeof(*s));
97 FILE *f = fopen(path, "rb");
98 if (!f) {
99 return -1;
100 }
101 unsigned char magic[5];
102 if (fread(magic, 1, sizeof(magic), f) != sizeof(magic) ||
103 memcmp(magic, REC_MAGIC, sizeof(magic)) != 0) {
104 fclose(f);
105 return -1;
106 }
107 size_t cap = 0;
108 for (;;) {
109 unsigned char h[13];
110 if (fread(h, 1, sizeof(h), f) != sizeof(h)) {
111 break;
112 }
113 size_t len = (size_t) get_be(h + 9, 4);
114 unsigned char *bytes = NULL;
115 if (len > 0) {
116 bytes = malloc(len);
117 if (!bytes || fread(bytes, 1, len, f) != len) {
118 free(bytes);
119 break;
120 }
121 }
122 if (s->n == cap) {
123 size_t ncap = cap ? cap * 2 : 64;
124 fp_replay_rec *nr = realloc(s->recs, ncap * sizeof(*nr));
125 if (!nr) {
126 free(bytes);
127 break;
128 }
129 s->recs = nr;
130 cap = ncap;
131 }
132 s->recs[s->n].is_up = (h[0] == 'u');
133 s->recs[s->n].ts_ms = get_be(h + 1, 8);
134 s->recs[s->n].bytes = bytes;
135 s->recs[s->n].len = len;
136 s->n++;
137 }
138 fclose(f);
139 return 0;
140}
