4/*
5 * wire_vendor_ext.h — request structures for the nginx-xrootd vendor extension
6 * opcodes (kXR_setattr / kXR_symlink / kXR_readlink / kXR_link, opcodes.h).
7 *
8 * WHAT: 24-byte ClientRequestHdr-shaped structs for the POSIX-completeness ops
9 * the base XRootD protocol lacks. All are capability-negotiated (the server
10 * advertises "xrdfs.ext" via kXR_Qconfig; the native client only emits these
11 * when advertised), so a stock server never sees them.
12 * WHY: Lets a FUSE mount honour `cp -p` / `touch -d` (mtime), chown, and
13 * `ln`/`ln -s` (hard/symbolic links) against this gateway.
14 * HOW: Same framing as every other request — 2B streamid + 2B requestid + 16B
15 * opcode body + 4B dlen (big-endian), then a payload. These structs use the
16 * project's natural-alignment convention (no #pragma pack): every field sits
17 * on its natural boundary so each struct is exactly 24 bytes with no padding,
18 * matching ClientRmRequest / ClientMvRequest. The variable attribute data for
19 * setattr lives entirely in the PAYLOAD (parsed field-by-field with
20 * be64toh/ntohl) so no misaligned int64 appears in a struct.
21 *
22 * Included by wire.h (after types.h, so the kXR_* type aliases are in scope).
23 */
24

27/* ---- kXR_setattr (3500) — set modification/access times and/or owner ----
28 *
29 * WHAT: Set a path's timestamps (utimens) and/or owner (chown). kXR_chmod already
30 * covers mode, so this opcode deliberately does NOT carry mode.
31 *
32 * Payload layout (big-endian fixed prefix, then a NUL-terminated path):
33 * off 0: int32 flags (kXR_sa_times | kXR_sa_owner)
34 * off 4: int64 atime_sec
35 * off 12: int64 atime_nsec (UTIME_OMIT / UTIME_NOW honoured per utimensat(2))
36 * off 20: int64 mtime_sec
37 * off 28: int64 mtime_nsec
38 * off 36: int32 uid (-1 = leave unchanged)
39 * off 40: int32 gid (-1 = leave unchanged)
40 * off 44: path[] (NUL-terminated; dlen - 44 bytes)
41 */
42#define BRIX_SETATTR_PREFIX_LEN 44 /* fixed binary prefix before the path */
43#define kXR_sa_times 0x01 /* apply atime/mtime via utimensat */
44#define kXR_sa_owner 0x02 /* apply uid/gid via fchownat */
45
46typedef struct {
47 kXR_char streamid[2];
48 kXR_unt16 requestid; /* kXR_setattr */
49 kXR_char reserved[16];
50 kXR_int32 dlen; /* BRIX_SETATTR_PREFIX_LEN + path length */
51 /* payload: 44-byte big-endian attribute prefix + null-terminated path */
52} ClientSetattrRequest; /* 24 bytes */

55 *
56 * WHAT: Create a symlink named <linkpath> whose target is the literal string
