34typedef struct {
35 char root[128]; /* the mkdtemp dir (plays <mnt>/.brixwrites) */
36 int writes_fd;
37 brix_overlay ov;
38} fix_t;

40static int fix_up(fix_t *f) {
41 snprintf(f->root, sizeof(f->root), "/tmp/ovut.XXXXXX");
42 if (mkdtemp(f->root) == NULL) return -1;
43 f->writes_fd = open(f->root, O_RDONLY | O_DIRECTORY);
44 if (f->writes_fd < 0) return -1;
45 return brix_overlay_init(&f->ov, f->writes_fd);
46}
47
48static void fix_down(fix_t *f) {
49 char cmd[160];
50 brix_overlay_close(&f->ov);
51 close(f->writes_fd);
52 snprintf(cmd, sizeof(cmd), "rm -rf '%s'", f->root);
53 if (system(cmd) != 0) { /* best-effort teardown */ }
54}

150/* ---- mkdirs + create + mkdir + opaque + unlink ----
151 *
152 * WHAT: asserts the directory-creating and file-creating mutation primitives —
153 * brix_overlay_mkdirs (default + mode_fn), brix_overlay_open (O_CREAT and
154 * missing-parent -ENOENT), brix_overlay_mkdir (mode + -EEXIST on repeat),
155 * set_opaque masking, and unlink_upper. Populates the fixture upper tree
156 * (under d1/d2 and m1) that later mutation groups depend on.
157 * WHY: split out of test_mutations so each named mutation family stays under
158 * the per-function complexity cap while every assertion and its ordering
159 * are preserved exactly.
160 * HOW: (1) mkdirs a deep chain, then one honoring the mode callback; (2) open
161 * O_CREAT|O_EXCL, write, close, and re-classify the new file; (3) reject a
162 * missing-parent create; (4) mkdir with an explicit mode, then twice;
163 * (5) set_opaque and confirm it masks below; (6) unlink_upper the created
164 * file back to NONE.
165 */
166static void test_mut_mkdirs_create(fix_t *f) {
167 struct stat st;
168 CHECK(brix_overlay_mkdirs(&f->ov, "d1/d2/d3", NULL, NULL) == 0
169 && cls(f, "d1/d2/d3", &st) == BRIX_OV_UPPER && S_ISDIR(st.st_mode),
170 "mkdirs deep chain");
171 CHECK(brix_overlay_mkdirs(&f->ov, "m1/m2", mode_0700, NULL) == 0
172 && cls(f, "m1", &st) == BRIX_OV_UPPER && (st.st_mode & 07777) == 0700,
173 "mkdirs honors mode_fn");
174
175 int fd = brix_overlay_open(&f->ov, "d1/d2/new", O_WRONLY | O_CREAT | O_EXCL, 0644);
176 CHECK(fd >= 0, "open O_CREAT|O_EXCL");
177 CHECK(write(fd, "abc", 3) == 3, "write via returned fd");
178 close(fd);
179 CHECK(cls(f, "d1/d2/new", &st) == BRIX_OV_UPPER && st.st_size == 3,
180 "created file classifies UPPER, size 3");
181 CHECK(brix_overlay_open(&f->ov, "noparent/x", O_WRONLY | O_CREAT, 0644) == -ENOENT,
182 "open with missing parent → -ENOENT"); /* neg */
183
184 CHECK(brix_overlay_mkdir(&f->ov, "d1/sub", 0750) == 0
185 && cls(f, "d1/sub", &st) == BRIX_OV_UPPER && (st.st_mode & 07777) == 0750,
186 "mkdir with mode");
187 CHECK(brix_overlay_mkdir(&f->ov, "d1/sub", 0750) == -EEXIST, "mkdir twice → -EEXIST");
188
189 CHECK(brix_overlay_set_opaque(&f->ov, "d1/sub") == 0, "set_opaque");
190 CHECK(cls(f, "d1/sub/anything", NULL) == BRIX_OV_MASKED, "opaque masks below");
191
192 CHECK(brix_overlay_unlink_upper(&f->ov, "d1/d2/new") == 0
193 && cls(f, "d1/d2/new", NULL) == BRIX_OV_NONE, "unlink_upper");
194}

315#define CU_SIZE (3u * 1024u * 1024u) /* 3 MiB → exercises the chunk loop */
316
317typedef struct { int fail_at_mib; } cu_src_t;
318
319/* mock lower reader: byte i = (i * 7) & 0xff; optional -EIO at a boundary */
320static int cu_read(void *ud, const char *rel, uint64_t off, size_t len,
321 unsigned char *buf, size_t *outlen) {
322 cu_src_t *s = ud;
323 (void) rel;
324 if (s->fail_at_mib >= 0 && off >= (uint64_t) s->fail_at_mib * 1024 * 1024)
325 return -EIO;
326 if (off >= CU_SIZE) { *outlen = 0; return 0; }
327 size_t n = len;
328 if (off + n > CU_SIZE) n = CU_SIZE - off;
329 for (size_t i = 0; i < n; i++) buf[i] = (unsigned char) (((off + i) * 7) & 0xff);
330 *outlen = n;
331 return 0;
332}
