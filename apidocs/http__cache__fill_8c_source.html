1/*
2 * http_cache_fill.c - off-event-loop cache-miss fill for the HTTP read plane.
3 * See http_cache_fill.h for the WHAT/WHY/HOW.
4 *
5 * Phase-68 additions:
6 * - COALESCING: every concurrent request for one (inst,key) parks on a
7 * single heap-owned fill — a 40-request stampede is exactly ONE origin
8 * fetch. The waiter list is event-loop-only (no locks).
9 * - NEVER-DROP (T20): with a client-hold configured, the fill worker
10 * retries transient origin failures with jittered backoff until the
11 * hold deadline; a waiter whose hold expires detaches and receives
12 * 504 + Retry-After on a KEPT-ALIVE connection (a TCP close is never
13 * an error signal — convention #6). A client abort detaches its waiter
14 * but never cancels the fill: the fill keeps retrying (max-life
15 * deadline) and publishes so the client's retry is a hit.
16 *
17 * Ownership: the fill ctx (+ its ngx_thread_task_t) is one calloc block,
18 * freed in the done handler. Each waiter is freed by ITS REQUEST's pool
19 * cleanup (which always fires exactly once), so a late cleanup can never
20 * use-after-free a waiter the done handler already resolved.
21 */

94/* Unlink `w` from its fill (idempotent; both the hold timer and the request
95 * cleanup may race to it on the event loop). */
96static void
97brix_http_fill_detach(brix_http_fill_waiter_t *w)
98{
99 brix_http_fill_waiter_t **pp;
100
101 if (w->resolved) {
102 return;
103 }
104 w->resolved = 1;
105 /* Un-arm the parked-window client-abort handler: past this point the
106 * request is either served (reenter), sent a 504, or being finalized —
107 * downstream read handling reverts to nginx's default, exactly as before
108 * the request parked. */
109 if (w->r != NULL) {
110 w->r->read_event_handler = ngx_http_block_reading;
111 }
112 if (w->hold.timer_set) {
113 ngx_del_timer(&w->hold);
114 }
115 for (pp = &w->owner->waiters; *pp != NULL; pp = &(*pp)->next) {
116 if (*pp == w) {
117 *pp = w->next;
118 break;
119 }
120 }
121 atomic_fetch_sub_explicit(&w->owner->waiters_n, 1, memory_order_relaxed);
122}

124/* 504 + Retry-After on a KEPT-ALIVE connection (convention #6: origin
125 * trouble is a well-formed HTTP answer, never a broken socket). */
126static void
127brix_http_fill_send_retry_later(ngx_http_request_t *r)
128{
129 static u_char body[] = "origin temporarily unreachable; retrying - "
130 "please retry\n";
131 ngx_table_elt_t *h;
132 ngx_buf_t *b;
133 ngx_chain_t out;
134
135 h = ngx_list_push(&r->headers_out.headers);
136 if (h == NULL) {
137 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
138 return;
139 }
140 h->hash = 1;
141 ngx_str_set(&h->key, "Retry-After");
142 ngx_str_set(&h->value, "2");
143
144 r->headers_out.status = NGX_HTTP_GATEWAY_TIME_OUT;
145 r->headers_out.content_length_n = sizeof(body) - 1;
146 r->keepalive = 1; /* NEVER close on origin trouble */
147 if (ngx_http_send_header(r) != NGX_OK || r->header_only) {
148 ngx_http_finalize_request(r, NGX_OK);
149 return;
150 }
151 b = ngx_pcalloc(r->pool, sizeof(*b));
152 if (b == NULL) {
153 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
154 return;
155 }
156 b->pos = b->start = body;
157 b->last = b->end = body + sizeof(body) - 1;
158 b->memory = 1;
159 b->last_buf = 1;
160 out.buf = b;
161 out.next = NULL;
162 ngx_http_finalize_request(r, ngx_http_output_filter(r, &out));
163}

165/* T20 hold timer: the client has waited long enough — answer 504 on the
166 * kept-alive connection and leave the fill running (its own retry, arriving
167 * on the same warm socket, coalesces onto this still-running fill). */
168static void
169brix_http_fill_hold_expire(ngx_event_t *ev)
170{
171 brix_http_fill_waiter_t *w = ev->data;
172 ngx_http_request_t *r = w->r;
173 ngx_connection_t *c = r->connection;
174
175 /* Diagnosis line for hold/CVMFS_TIMEOUT misalignment: WHO waited HOW
176 * LONG on WHAT while the origin still hadn't answered. */
177 if (w->owner != NULL) {
178 char kbuf[256];
179
180 ngx_log_error(NGX_LOG_WARN, c->log, 0,
181 "xrootd-fill: event=hold-expired key=\"%s\" client=%V "
182 "held_ms=%M fill_elapsed_ms=%M waiters_left=%d "
183 "action=\"504+Retry-After sent on the kept-alive connection; "
184 "the fill keeps running\"",
185 brix_http_fill_log_key(w->owner->key, kbuf, sizeof(kbuf)),
186 &c->addr_text,
187 (ngx_msec_t) (ngx_current_msec - w->parked_ms),
188 (ngx_msec_t) (ngx_current_msec - w->owner->started_ms),
189 atomic_load_explicit(&w->owner->waiters_n,
190 memory_order_relaxed) - 1);
191 }
192
193 brix_http_fill_detach(w);
194 brix_http_fill_send_retry_later(r);
195 ngx_http_run_posted_requests(c);
196}

198/* Request-pool cleanup: the waiter's single owner for freeing. Fires exactly
199 * once per parked request — after the done handler resolved it (resolved=1,
200 * just free) or on a client abort mid-fill (detach, then free; the fill
201 * itself is NEVER cancelled). */
202static void
203brix_http_fill_waiter_cleanup(void *data)
204{
205 brix_http_fill_waiter_t *w = data;
206
207 /* Still attached at teardown = an ABNORMAL finalize the read-abort
208 * handler and the hold timer both missed (worker shutdown mid-fill, or
209 * nginx forcibly closing the request). The routine client-abort case is
210 * logged promptly by brix_http_fill_client_read; this is the safety
211 * net so a parked request never disappears without a trace. */
212 if (!w->resolved && w->owner != NULL) {
213 ngx_connection_t *c = w->r->connection;
214 char kbuf[256];
215
216 ngx_log_error(NGX_LOG_WARN, c->log, 0,
217 "xrootd-fill: event=parked-teardown key=\"%s\" client=%V "
218 "parked_ms=%M fill_elapsed_ms=%M "
219 "cause=\"request finalized while parked (worker shutdown or "
220 "forced close); the fill continues detached\"",
221 brix_http_fill_log_key(w->owner->key, kbuf, sizeof(kbuf)),
222 &c->addr_text,
223 (ngx_msec_t) (ngx_current_msec - w->parked_ms),
224 (ngx_msec_t) (ngx_current_msec - w->owner->started_ms));
225 }
226
227 brix_http_fill_detach(w);
228 free(w);
229}

231/* Find the waiter parked for request `r` (a handful in flight; linear is
232 * fine). NULL if `r` is not currently parked on any fill. */
233static brix_http_fill_waiter_t *
234brix_http_fill_waiter_for(ngx_http_request_t *r)
235{
236 brix_http_cache_fill_ctx_t *t;
237 brix_http_fill_waiter_t *w;
238
239 for (t = brix_http_fills; t != NULL; t = t->next) {
240 for (w = t->waiters; w != NULL; w = w->next) {
241 if (w->r == r) {
242 return w;
243 }
244 }
245 }
246 return NULL;
247}

249/* Read-event handler for a PARKED request: nginx normally ignores downstream
250 * reads while a request is suspended (r->main->count raised), so a client
251 * that breaks its connection mid-fill would go unnoticed until its hold timer
252 * fires. Detecting it here lets us log the broken client, reclaim its parked
253 * slot immediately, and finalize — while the fill keeps running detached
254 * (never-drop: one client hanging up must not cancel the origin fetch that
255 * other clients, or the client's own retry, still need). */
256static void
257brix_http_fill_client_read(ngx_http_request_t *r)
258{
259 ngx_connection_t *c = r->connection;
260 ngx_event_t *rev = c->read;
261 brix_http_fill_waiter_t *w;
262 u_char probe;
263 ssize_t n;
264
265 if (!rev->ready && !rev->eof) {
266 return; /* spurious wakeup */
267 }
268 /* Peek: EOF (0) or a hard error means the client is gone; EAGAIN means it
269 * merely sent data we don't want (ignore — never-drop keeps waiting). */
270 n = recv(c->fd, &probe, 1, MSG_PEEK);
271 if (n > 0) {
272 return;
273 }
274 if (n < 0 && (ngx_errno == NGX_EAGAIN || ngx_errno == NGX_EINTR)) {
275 return;
276 }
277
278 w = brix_http_fill_waiter_for(r);
279 if (w != NULL && w->owner != NULL) {
280 char kbuf[256];
281
282 ngx_log_error(NGX_LOG_WARN, c->log, 0,
283 "xrootd-fill: event=client-gone key=\"%s\" client=%V "
284 "parked_ms=%M fill_elapsed_ms=%M waiters_left=%d "
285 "hint=\"client broke the connection before the origin answered; "
286 "the fill continues detached. If this recurs the client timeout "
287 "is shorter than the fill latency (CVMFS_TIMEOUT vs client_hold)\"",
288 brix_http_fill_log_key(w->owner->key, kbuf, sizeof(kbuf)),
289 &c->addr_text,
290 (ngx_msec_t) (ngx_current_msec - w->parked_ms),
291 (ngx_msec_t) (ngx_current_msec - w->owner->started_ms),
292 atomic_load_explicit(&w->owner->waiters_n,
293 memory_order_relaxed) - 1);
294 brix_http_fill_detach(w); /* resolved=1: no dup log */
295 r->main->count--; /* release the parked ref */
296 }
297 c->error = 1;
298 ngx_http_finalize_request(r, NGX_HTTP_CLIENT_CLOSED_REQUEST);
299}

301/* Park `r` on fill `t` (r->main->count++ balanced by the finalize in done /
302 * hold-expiry / abort teardown). */
303static ngx_int_t
304brix_http_fill_attach(brix_http_cache_fill_ctx_t *t,
305 ngx_http_request_t *r, brix_http_cache_reenter_pt reenter,
306 void *reenter_data)
307{
308 brix_http_fill_waiter_t *w = calloc(1, sizeof(*w));
309 ngx_pool_cleanup_t *cln;
310
311 if (w == NULL) {
312 return NGX_ERROR;
313 }
314 cln = ngx_pool_cleanup_add(r->pool, 0);
315 if (cln == NULL) {
316 free(w);
317 return NGX_ERROR;
318 }
319 w->r = r;
320 w->reenter = reenter;
321 w->reenter_data = reenter_data;
322 w->parked_ms = ngx_current_msec;
323 w->owner = t;
324 w->next = t->waiters;
325 t->waiters = w;
326 atomic_fetch_add_explicit(&t->waiters_n, 1, memory_order_relaxed);
327
328 cln->handler = brix_http_fill_waiter_cleanup;
329 cln->data = w;
330
331 if (t->client_hold > 0) {
332 w->hold.handler = brix_http_fill_hold_expire;
333 w->hold.data = w;
334 w->hold.log = r->connection->log;
335 ngx_add_timer(&w->hold, (ngx_msec_t) t->client_hold * 1000);
336 }
337
338 /* Notice a client that breaks its connection while parked (above): the
339 * read handler logs it, reclaims this slot, and finalizes — the fill
340 * runs on. Arm the read event so the abort actually wakes us. */
341 r->read_event_handler = brix_http_fill_client_read;
342 if (ngx_handle_read_event(r->connection->read, 0) != NGX_OK) {
343 /* non-fatal: we simply fall back to the hold timer for teardown */
344 ngx_log_debug0(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
345 "xrootd-fill: could not arm client-abort read event");
346 }
347
348 r->main->count++;
349 return NGX_DONE;
350}

352/* Worker thread: run the blocking source->store fill off the event loop,
353 * retrying transient origin failures to the T20 deadlines (single-pass when
354 * no hold is configured — the pre-phase-68 semantics for other protocols). */
355static void
356brix_http_cache_fill_thread(void *data, ngx_log_t *log)
357{
358 brix_http_cache_fill_ctx_t *t = data;
359 brix_fill_retry_t rs;
360 unsigned n_eps;
361 char kbuf[256];
362
363 n_eps = (unsigned) sd_http_n_endpoints(
364 brix_sd_cache_source_instance(t->inst));
365 brix_fill_retry_init(&rs, t->client_hold, t->max_life, &t->waiters_n,
366 n_eps);
367 for ( ;; ) {
368 errno = 0;
369 t->attempts++;
370 t->result = brix_sd_cache_fill_key(t->inst, t->key);
371 t->err = errno;
372 switch (brix_fill_classify(t->result, t->err, &rs)) {
373 case BRIX_FILL_OK:
374 if (t->attempts > 1) {
375 ngx_log_error(NGX_LOG_WARN, log, 0,
376 "xrootd-fill: event=recovered key=\"%s\" attempts=%ud "
377 "elapsed_ms=%M — origin answered after transient failures",
378 brix_http_fill_log_key(t->key, kbuf, sizeof(kbuf)),
379 t->attempts,
380 (ngx_msec_t) (ngx_current_msec - t->started_ms));
381 }
382 return;
383 case BRIX_FILL_DEFINITIVE:
384 return;
385 case BRIX_FILL_RETRY:
386 break;
387 }
388 /* One line per failed attempt: WHICH object, WHICH attempt, WHY —
389 * the trail that locates a flapping origin or a lossy WAN window.
390 * Bounded by the backoff schedule (a handful per fill, worst case). */
391 ngx_log_error(NGX_LOG_WARN, log, t->err,
392 "xrootd-fill: event=retry key=\"%s\" attempt=%ud rc=%i "
393 "elapsed_ms=%M endpoints=%ud waiters=%d next_backoff_ms=%M",
394 brix_http_fill_log_key(t->key, kbuf, sizeof(kbuf)),
395 t->attempts, t->result,
396 (ngx_msec_t) (ngx_current_msec - t->started_ms), n_eps,
397 atomic_load_explicit(&t->waiters_n, memory_order_relaxed),
398 rs.backoff_ms);
399 if (!brix_fill_retry_wait(&rs)) {
400 if (t->err == EBADMSG) {
401 return; /* proven-bad data: 502, not "later" */
402 }
403 t->err = ETIMEDOUT; /* deadline exhausted: 504 to waiters */
404 return;
405 }
406 }
407}

409/* Resolve one waiter with the fill outcome (event loop). */
410static void
411brix_http_fill_resolve_waiter(brix_http_cache_fill_ctx_t *t,
412 brix_http_fill_waiter_t *w)
413{
414 ngx_http_request_t *r = w->r;
415 ngx_connection_t *c = r->connection;
416 ngx_int_t rc;
417
418 if (t->result == NGX_OK) {
419 rc = w->reenter(r, w->reenter_data); /* hit -> serve (zero-copy) */
420 } else if (t->result == NGX_DECLINED) {
421 ngx_log_error(NGX_LOG_WARN, c->log, 0,
422 "brix: cache fill declined for \"%s\" (object not cacheable and "
423 "remote-source streaming is not yet supported) - returning 502",
424 t->key);
425 rc = NGX_HTTP_BAD_GATEWAY;
426 } else if (t->err == ENOENT || t->err == ENOTDIR) {
427 /* The origin's definitive answer: the object does not exist. */
428 rc = NGX_HTTP_NOT_FOUND;
429 } else if (t->err == EACCES || t->err == EPERM) {
430 rc = NGX_HTTP_FORBIDDEN;
431 } else if (t->err == ETIMEDOUT) {
432 /* T20: deadline exhausted while this waiter was still attached —
433 * same keep-alive 504 the hold timer sends. */
434 brix_http_fill_send_retry_later(r);
435 ngx_http_run_posted_requests(c);
436 return;
437 } else {
438 /* per-waiter detail at debug; the fill-level ERR line in done()
439 * carries the full story once, not once per parked client */
440 ngx_log_debug2(NGX_LOG_DEBUG_HTTP, c->log, 0,
441 "xrootd-fill: resolving waiter with 502 (key \"%s\", err %d)",
442 t->key, t->err);
443 rc = NGX_HTTP_BAD_GATEWAY;
444 }
445
446 ngx_http_finalize_request(r, rc);
447 ngx_http_run_posted_requests(c);
448}

450/* Event loop: resolve EVERY still-attached waiter with the one fill outcome
451 * (the stampede coalescing contract), then release the fill. Waiter memory
452 * stays with its request's pool cleanup. */
453static void
454brix_http_cache_fill_done(ngx_event_t *ev)
455{
456 ngx_thread_task_t *task = ev->data;
457 brix_http_cache_fill_ctx_t *t = task->ctx;
458 brix_http_cache_fill_ctx_t **pp;
459 brix_http_fill_waiter_t *w;
460 char errscratch[BRIX_SESSLOG_ERR_MAX];
461 const char *err;
462 int ok;
463
464 /* Unlink from the in-flight list FIRST so a re-entered handler that
465 * misses again starts a fresh fill rather than attaching to this one. */
466 for (pp = &brix_http_fills; *pp != NULL; pp = &(*pp)->next) {
467 if (*pp == t) {
468 *pp = t->next;
469 break;
470 }
471 }
472
473 /* ONE outcome line per fill — the anchor every other event (retry /
474 * hold-expired / client-gone) correlates with via the key. */
475 {
476 int waiters = atomic_load_explicit(&t->waiters_n,
477 memory_order_relaxed);
478 char kbuf[256];
479 const char *k = brix_http_fill_log_key(t->key, kbuf, sizeof(kbuf));
480 ngx_msec_t elapsed = ngx_current_msec - t->started_ms;
481
482 if (t->result == NGX_OK) {
483 ngx_uint_t lvl = (waiters > 0) ? NGX_LOG_INFO : NGX_LOG_NOTICE;
484
485 ngx_log_error(lvl, ev->log, 0,
486 "xrootd-fill: event=%s key=\"%s\" attempts=%ud "
487 "elapsed_ms=%M waiters=%d",
488 (waiters > 0) ? "done" : "done-detached", k,
489 t->attempts, elapsed, waiters);
490 } else if (t->err == ENOENT || t->err == ENOTDIR) {
491 ngx_log_error(NGX_LOG_INFO, ev->log, 0,
492 "xrootd-fill: event=not-found key=\"%s\" attempts=%ud "
493 "elapsed_ms=%M waiters=%d — the origin's definitive 404",
494 k, t->attempts, elapsed, waiters);
495 } else if (t->err == ETIMEDOUT) {
496 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
497 "xrootd-fill: event=exhausted key=\"%s\" attempts=%ud "
498 "elapsed_ms=%M waiters=%d verdict=\"504 retry-later to every "
499 "waiter; no origin answered within the deadline\"",
500 k, t->attempts, elapsed, waiters);
501 } else {
502 ngx_log_error(NGX_LOG_ERR, ev->log, t->err,
503 "xrootd-fill: event=failed key=\"%s\" attempts=%ud "
504 "elapsed_ms=%M waiters=%d verdict=%s",
505 k, t->attempts, elapsed, waiters,
506 (t->err == EBADMSG)
507 ? "\"502; every endpoint served corrupt data "
508 "(verify quarantined the evidence)\""
509 : "\"502 to every waiter\"");
510 }
511 }
512
513 ok = (t->result == NGX_OK);
514 err = ok ? NULL : brix_sesslog_err_from_errno(t->err, errscratch,
515 sizeof(errscratch));
516 brix_sess_result(t->sess, ok, t->key, BRIX_SESS_MODE_READ, err);
517 brix_sess_xfer_end(t->sess, &t->sess_xfer,
518 ok ? BRIX_SESS_XFER_COMPLETE
519 : BRIX_SESS_XFER_ABORTED);
520 brix_sess_end(t->sess, ok ? BRIX_SESS_END_SERVER
521 : BRIX_SESS_END_ERROR);
522
523 while ((w = t->waiters) != NULL) {
524 brix_http_fill_detach(w); /* unlink + stop the hold */
525 brix_http_fill_resolve_waiter(t, w);
526 }
527 free(t->task); /* the calloc'd task+ctx block (task is first) */
528}

554 ngx_http_brix_shared_conf_t *common,
555 brix_http_cache_reenter_pt reenter, void *reenter_data)
556{
557 ngx_thread_task_t *task;
558 brix_http_cache_fill_ctx_t *t;
559 ngx_thread_pool_t *pool;
560 u_char *block;
561
562 if (inst == NULL || key == NULL || reenter == NULL || common == NULL
563 || !brix_sd_cache_fill_needs_offload(inst, key))
564 {
565 return NGX_DECLINED; /* serve inline (hit / local / none) */
566 }
567
568 /* Coalesce onto an in-flight fill of the same object: the stampede case
569 * (N concurrent cold reads) is exactly ONE origin fetch. */
570 t = brix_http_fill_find(inst, key);
571 if (t != NULL) {
572 return brix_http_fill_attach(t, r, reenter, reenter_data);
573 }
574
575 pool = brix_http_cache_fill_pool(common);
576 if (pool == NULL) {
577 /* No pool: nothing can run off-loop, so fall through to the inline path
578 * (preserves the pre-SP2 behaviour - a remote miss may stall/fail). */
579 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
580 "brix: cache miss on \"%s\" needs an async thread pool to fill a "
581 "remote tier; none configured - serving inline (may stall)", key);
582 return NGX_DECLINED;
583 }
584
585 /* Heap-owned task + ctx: the fill is shared by every parked request and
586 * must not live in any one request's pool (an aborted first requester
587 * would otherwise free the memory under the running task). */
588 block = calloc(1, sizeof(ngx_thread_task_t)
589 + sizeof(brix_http_cache_fill_ctx_t));
590 if (block == NULL) {
591 return NGX_ERROR;
592 }
593 task = (ngx_thread_task_t *) block;
594 task->ctx = block + sizeof(ngx_thread_task_t);
595 t = task->ctx;
596 t->inst = inst;
597 t->task = task;
598 t->result = NGX_ERROR;
599 t->client_hold = common->cache_client_hold; /* 0 = single-pass fill */
600 t->max_life = common->cache_fill_max_life;
601 t->started_ms = ngx_current_msec;
602 ngx_cpystrn((u_char *) t->key, (u_char *) key, sizeof(t->key));
603 t->sess = brix_sess_begin(common->session_log,
604 brix_http_shared_access_log_fd(common), BRIX_SESS_PROTO_FILL,
605 BRIX_SESS_DIR_OUT, "cache-origin", sizeof("cache-origin") - 1,
606 BRIX_SESS_AM_ANON, NULL);
607 brix_sess_auth_once(t->sess, BRIX_SESS_AM_ANON, "-", "-");
608 brix_sess_attempt(t->sess, t->key, BRIX_SESS_MODE_READ);
609 brix_sess_xfer_start(t->sess, &t->sess_xfer, t->key,
610 BRIX_SESS_MODE_READ, -1);
611
612 brix_task_bind(task, brix_http_cache_fill_thread,
613 brix_http_cache_fill_done);
614 task->event.log = r->connection->log;
615
616 if (brix_http_fill_attach(t, r, reenter, reenter_data) != NGX_DONE) {
617 brix_sess_end(t->sess, BRIX_SESS_END_ERROR);
618 free(block);
619 return NGX_ERROR;
620 }
621
622 if (ngx_thread_task_post(pool, task) != NGX_OK) {
623 brix_http_fill_detach(t->waiters); /* undo the attach */
624 r->main->count--;
625 brix_sess_end(t->sess, BRIX_SESS_END_ERROR);
626 free(block);
627 return NGX_ERROR;
628 }
629
630 t->next = brix_http_fills; /* publish for coalescing */
631 brix_http_fills = t;
632
633 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
634 "brix: offloaded cache fill of \"%s\" to the thread pool", key);
635 return NGX_DONE;
636}
637
638#else /* !NGX_THREADS */

641 * inline open/fill path (correct for a local tier; a remote tier is unsupported
642 * without threads, exactly as before SP2). */

650 (void) inst;
651 (void) key;

653 (void) reenter;
654 (void) reenter_data;
655 return NGX_DECLINED;
656}
