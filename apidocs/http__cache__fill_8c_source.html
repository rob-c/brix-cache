1/*
2 * http_cache_fill.c - off-event-loop cache-miss fill for the HTTP read plane.
3 * See http_cache_fill.h for the WHAT/WHY/HOW.
4 *
5 * This unit is the public entry point + async-pool resolver. The concurrency
6 * machinery it drives was split out mechanically (zero behaviour change) for
7 * auditability — see http_cache_fill_internal.h:
8 * http_cache_fill_registry.c coalescing waiter registry (attach/detach/hold/abort)
9 * http_cache_fill_worker.c worker thread + resolve + done finalize
10 *
11 * Phase-68 additions:
12 * - COALESCING: every concurrent request for one (inst,key) parks on a
13 * single heap-owned fill — a 40-request stampede is exactly ONE origin
14 * fetch. The waiter list is event-loop-only (no locks).
15 * - NEVER-DROP (T20): with a client-hold configured, the fill worker
16 * retries transient origin failures with jittered backoff until the
17 * hold deadline; a waiter whose hold expires detaches and receives
18 * 504 + Retry-After on a KEPT-ALIVE connection (a TCP close is never
19 * an error signal — convention #6). A client abort detaches its waiter
20 * but never cancels the fill: the fill keeps retrying (max-life
21 * deadline) and publishes so the client's retry is a hit.
22 *
23 * Ownership: the fill ctx (+ its ngx_thread_task_t) is one calloc block,
24 * freed in the done handler. Each waiter is freed by ITS REQUEST's pool
25 * cleanup (which always fires exactly once), so a late cleanup can never
26 * use-after-free a waiter the done handler already resolved.
27 */

45/* Lazily resolve the export's async thread pool: server postconfig fills only the
46 * server-level loc_conf, so a nested location block resolves on first use (the
47 * webdav/copy.c idiom). NULL when no pool is configured. */
48static ngx_thread_pool_t *
49brix_http_cache_fill_pool(ngx_http_brix_shared_conf_t *common)
50{
51 ngx_thread_pool_t *pool = common->thread_pool;
52
53 if (pool == NULL) {
54 static ngx_str_t default_name = ngx_string("default");
55 ngx_str_t *pname = common->thread_pool_name.len > 0
56 ? &common->thread_pool_name : &default_name;
57
58 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
59 if (pool != NULL) {
60 common->thread_pool = pool;
61 }
62 }
63 return pool;
64}

66ngx_int_t
67brix_http_cache_fill_if_needed(ngx_http_request_t *r,
68 brix_sd_instance_t *inst, const char *key,
69 ngx_http_brix_shared_conf_t *common,
70 brix_http_cache_reenter_pt reenter, void *reenter_data)
71{
72 ngx_thread_task_t *task;
73 brix_http_cache_fill_ctx_t *t;
74 ngx_thread_pool_t *pool;
75 u_char *block;
76
77 if (inst == NULL || key == NULL || reenter == NULL || common == NULL
78 || !brix_sd_cache_fill_needs_offload(inst, key))
79 {
80 return NGX_DECLINED; /* serve inline (hit / local / none) */
81 }
82
83 /* Coalesce onto an in-flight fill of the same object: the stampede case
84 * (N concurrent cold reads) is exactly ONE origin fetch. */
85 t = brix_http_fill_find(inst, key);
86 if (t != NULL) {
87 return brix_http_fill_attach(t, r, reenter, reenter_data);
88 }
89
90 pool = brix_http_cache_fill_pool(common);
91 if (pool == NULL) {
92 /* No pool: nothing can run off-loop, so fall through to the inline path
93 * (preserves the pre-SP2 behaviour - a remote miss may stall/fail). */
94 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
95 "brix: cache miss on \"%s\" needs an async thread pool to fill a "
96 "remote tier; none configured - serving inline (may stall)", key);
97 return NGX_DECLINED;
98 }
99
100 /* Heap-owned task + ctx: the fill is shared by every parked request and
101 * must not live in any one request's pool (an aborted first requester
102 * would otherwise free the memory under the running task). */
103 block = calloc(1, sizeof(ngx_thread_task_t)
104 + sizeof(brix_http_cache_fill_ctx_t));
105 if (block == NULL) {
106 return NGX_ERROR;
107 }
108 task = (ngx_thread_task_t *) block;
109 task->ctx = block + sizeof(ngx_thread_task_t);
110 t = task->ctx;
111 t->inst = inst;
112 t->task = task;
113 t->result = NGX_ERROR;
114 t->client_hold = common->cache_client_hold; /* 0 = single-pass fill */
115 t->max_life = common->cache_fill_max_life;
116 t->started_ms = ngx_current_msec;
117 ngx_cpystrn((u_char *) t->key, (u_char *) key, sizeof(t->key));
118 t->sess = brix_sess_begin(common->session_log,
119 brix_http_shared_access_log_fd(common), BRIX_SESS_PROTO_FILL,
120 BRIX_SESS_DIR_OUT, "cache-origin", sizeof("cache-origin") - 1,
121 BRIX_SESS_AM_ANON, NULL);
122 brix_sess_auth_once(t->sess, BRIX_SESS_AM_ANON, "-", "-");
123 brix_sess_attempt(t->sess, t->key, BRIX_SESS_MODE_READ);
124 brix_sess_xfer_start(t->sess, &t->sess_xfer, t->key,
125 BRIX_SESS_MODE_READ, -1);
126
127 brix_task_bind(task, brix_http_cache_fill_thread,
128 brix_http_cache_fill_done);
129 task->event.log = r->connection->log;
130
131 if (brix_http_fill_attach(t, r, reenter, reenter_data) != NGX_DONE) {
132 brix_sess_end(t->sess, BRIX_SESS_END_ERROR);
133 free(block);
134 return NGX_ERROR;
135 }
136
137 if (ngx_thread_task_post(pool, task) != NGX_OK) {
138 brix_http_fill_detach(t->waiters); /* undo the attach */
139 r->main->count--;
140 brix_sess_end(t->sess, BRIX_SESS_END_ERROR);
141 free(block);
142 return NGX_ERROR;
143 }
144
145 t->next = brix_http_fills; /* publish for coalescing */
146 brix_http_fills = t;
147
148 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
149 "brix: offloaded cache fill of \"%s\" to the thread pool", key);
150 return NGX_DONE;
151}
