1/*
2 * cephfs_denc.h — read-only cursor over a Ceph-encoded buffer.
3 *
4 * WHAT: The low-level decode primitives for Ceph's wire/disk encoding —
5 * fixed-width little-endian integers, length-prefixed strings/byte runs,
6 * and the ENCODE_START/DECODE_FINISH struct framing (struct_v / compat_v /
7 * byte-length) used by every versioned Ceph type. Bytes in, scalars out.
8 *
9 * WHY: Reading a CephFS directly from RADOS means decoding the MDS's on-disk
10 * structures (dentries, inodes, file layouts, fragtrees). Those are all
11 * built from this same handful of primitives. Isolating them here — with
12 * no RADOS and no nginx dependency — lets the typed decoders in
13 * cephfs_layout.* be written and unit-tested against byte fixtures with a
14 * plain compiler, and keeps every bounds check in one audited place.
15 *
16 * HOW: A cephfs_denc_t holds (cursor, end, sticky-error). Every accessor checks
17 * remaining space before reading; on overrun it sets the sticky error and
18 * returns a zero/NULL value, so a caller can decode a whole struct and test
19 * cephfs_denc_ok() once at the end rather than after every field. The frame
20 * helpers record where a struct's payload ends so callers can forward-skip
21 * fields they don't understand (forward compatibility across versions).
22 *
23 * This header is pure C with no nginx/RADOS dependency by design.
24 */
25#ifndef BRIX_CEPHFS_DENC_H
26#define BRIX_CEPHFS_DENC_H
27
28#include <stddef.h>
29#include <stdint.h>

31/* A read-only cursor. `err` is sticky: once set, every further accessor is a
32 * no-op that returns zero/NULL, so partial decodes never read out of bounds. */
33typedef struct {
34 const uint8_t *p; /* next byte to read */
35 const uint8_t *end; /* one past the last valid byte */
36 int err; /* 0 = ok; non-zero once an overrun/decode error hits */
37} cephfs_denc_t;
38
39/* A decoded ENCODE_START frame header (struct version, compat version, and the
40 * absolute end of this struct's payload — used by cephfs_denc_finish). */
41typedef struct {
42 uint8_t struct_v;
43 uint8_t struct_compat;
44 const uint8_t *payload_end;
45} cephfs_denc_frame_t;
46
47/* Bind a cursor to [buf, buf+len). */
48void cephfs_denc_init(cephfs_denc_t *d, const void *buf, size_t len);
49
50/* 1 when no error has been recorded, else 0. */
51int cephfs_denc_ok(const cephfs_denc_t *d);
52
53/* Bytes left between the cursor and end (0 once errored or exhausted). */
54size_t cephfs_denc_remaining(const cephfs_denc_t *d);
55
56/* Fixed-width little-endian integers. On overrun: set err, return 0. */
57uint8_t cephfs_denc_u8 (cephfs_denc_t *d);
58uint16_t cephfs_denc_u16(cephfs_denc_t *d);
59uint32_t cephfs_denc_u32(cephfs_denc_t *d);
60uint64_t cephfs_denc_u64(cephfs_denc_t *d);
61int64_t cephfs_denc_s64(cephfs_denc_t *d);

63/* A run of n raw bytes: returns a pointer into the buffer (no copy) and advances
64 * the cursor; returns NULL (and sets err) on overrun. */
65const uint8_t *cephfs_denc_bytes(cephfs_denc_t *d, size_t n);
66
67/* Advance the cursor by n bytes (clamped; sets err on overrun). */
68void cephfs_denc_skip(cephfs_denc_t *d, size_t n);
69
70/* A Ceph string/bufferlist: u32 length prefix + that many bytes. On success sets
71 * *s to a pointer into the buffer (NOT null-terminated) and *len to the length,
72 * and returns 0; on overrun sets err and returns -1. Either out-param may be
73 * NULL to decode-and-discard. */
74int cephfs_denc_str(cephfs_denc_t *d, const char **s, uint32_t *len);
75
76/* Read an ENCODE_START header: struct_v(u8), struct_compat(u8), struct_len(u32).
77 * Records the payload bounds in *f and returns struct_v. On overrun, or if the
78 * declared length runs past the buffer, sets err and returns 0. */
79uint8_t cephfs_denc_start(cephfs_denc_t *d, cephfs_denc_frame_t *f);
80
81/* Jump the cursor to the end of a frame's payload (skipping any trailing fields a
82 * newer encoder added that this decoder did not read). No-op if errored or if the
83 * cursor is already at/past payload_end. */
84void cephfs_denc_finish(cephfs_denc_t *d, const cephfs_denc_frame_t *f);
