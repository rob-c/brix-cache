7/*
8 * kXR_fattr — XRootD file-attribute (extended attribute) operations.
9 *
10 * XRootD file attributes are mapped to POSIX extended attributes under the
11 * "user." namespace, with 'U' as an internal sub-namespace marker (mirroring the
12 * reference's FATTR_NAMESPACE='U'). The XRootD wire name (e.g. "mykey") is
13 * stored as the xattr key "user.U.mykey" on the filesystem. On retrieval the
14 * ENTIRE internal prefix is stripped so the client sees the VERBATIM name it set
15 * ("mykey") — matching the reference XeqFALsd, which keeps 'U' internal only.
16 *
17 * Key prefixes:
18 * BRIX_FATTR_XKEY_PFX "user.U." — filesystem xattr key prefix
19 * BRIX_FATTR_XKEY_PFX_LEN 7 — byte length stripped to recover the
20 * bare client-visible name
21 * BRIX_FATTR_MAX_VBUF 65536 — maximum xattr value buffer
22 */
23#define BRIX_FATTR_XKEY_PFX "user.U."
24#define BRIX_FATTR_XKEY_PFX_LEN 7
25#define BRIX_FATTR_MAX_VBUF 65536

27/*
28 * Parsed view of one fattr name-vector (nvec) entry.
29 *
30 * XRootD fattr requests carry a name vector: a sequence of
31 * [uint16_t rc = 0][uint16_t nlen][nlen bytes name]
32 * entries. After per-attribute operations complete, rc_ptr is overwritten with
33 * the XRootD status code in-place so the same nvec buffer is used for the
34 * kXR_fattr status response without rebuilding the wire frame.
35 *
36 * Lifetime: stack-allocated by the dispatch functions; valid only for the
37 * duration of one kXR_fattr request.
38 */
39typedef struct {
40 u_char *rc_ptr; /* pointer into nvec_copy where the 2-byte rc lives */
41 char *name; /* NUL-terminated attribute name from the nvec */
42 size_t nlen; /* byte count of name (not including NUL) */
43 char xkey[512]; /* full "user.U.<name>" xattr key, NUL-terminated */
44 u_char *value; /* value for set ops (NULL for get/del/list) */
45 ssize_t vlen; /* length of value (-1 = not set) */
46 uint16_t errcode; /* per-attribute XRootD error code (0 = success) */
47} brix_fattr_entry_t;
