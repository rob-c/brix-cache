1/*
2 * vfs_rename.c — VFS rename / move.
3 *
4 * WHAT: Implements brix_vfs_rename(), which moves the resolved ctx (source)
5 * path to a caller-supplied, already-resolved destination
6 * brix_path_result_t.
7 *
8 * WHY: kXR_mv and WebDAV MOVE need a single write-gated, confined rename whose
9 * destination is verified to be inside the export root before the syscall,
10 * with one metric/access-log line — not an ad-hoc rename(2) per protocol.
11 *
12 * HOW: Enforces brix_vfs_require_write(), then demands a non-NULL root_canon
13 * and a destination that is itself is_confined with a non-empty resolved
14 * path. It delegates the actual move to brix_ns_rename() (namespace
15 * layer), maps the returned status back to errno (sys_errno or EIO), and
16 * observes the operation as BRIX_METRIC_OP_RENAME on every path.
17 */
18#include "vfs_internal.h"
19
20/* Thread-safe confined rename (no pool alloc, no metric) — relocates the
21 * namespace rename into the VFS layer for off-thread / pool-less callers (kXR_mv,
22 * WebDAV MOVE collection offload). Maps the namespace status to errno + an
23 * optional was_dir flag; see vfs.h. */
24ngx_int_t
25brix_vfs_rename_path(brix_sd_instance_t *sd, ngx_log_t *log,
26 const char *root_canon, const char *src, const char *dst,
27 unsigned overwrite, int *was_dir_out)
28{
29 brix_ns_result_t res;
30
31 /* Non-POSIX backend: rename in the driver namespace (export-relative keys).
32 * was_dir is derived from a post-rename stat of the destination. */
33 if (sd != NULL && sd->driver != brix_sd_default_driver()
34 && sd->driver->rename != NULL)
35 {
36 const char *s = brix_vfs_export_relative_root(src, root_canon);
37 const char *d = brix_vfs_export_relative_root(dst, root_canon);
38 ngx_int_t rc = sd->driver->rename(sd, s, d, overwrite ? 0 : 1);
39
40 if (was_dir_out != NULL) {
41 brix_sd_stat_t st;
42
43 *was_dir_out = (rc == NGX_OK && sd->driver->stat != NULL
44 && sd->driver->stat(sd, d, &st) == NGX_OK)
45 ? st.is_dir : 0;
46 }
47 return rc;
48 }
49
50 res = brix_ns_rename(log, root_canon, src, dst, overwrite ? 1 : 0);
51 if (was_dir_out != NULL) {
52 *was_dir_out = res.was_dir;
53 }
54 if (res.status == BRIX_NS_OK) {
55 return NGX_OK;
56 }
57 errno = res.sys_errno != 0 ? res.sys_errno
58 : brix_vfs_ns_status_errno(res.status);
59 return NGX_ERROR;
60}

62/* Move the resolved ctx source path to the confined destination `dst`.
63 * Write-gated; both endpoints must be confined. Metered as OP_RENAME. */
64ngx_int_t
65brix_vfs_rename(brix_vfs_ctx_t *ctx, const brix_path_result_t *dst)
66{
67 brix_ns_result_t res;
68 const char *path;
69 uint64_t start;
70 int saved_errno;
71 const brix_sd_driver_t *drv;
72
73 start = brix_vfs_now_ns();
74 path = brix_vfs_ctx_path(ctx);
75
76 if (brix_vfs_require_write(ctx) != NGX_OK) {
77 saved_errno = errno;
78 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
79 NGX_ERROR, saved_errno, start);
80 return NGX_ERROR;
81 }
82
83 if (ctx->root_canon == NULL || dst == NULL || !dst->is_confined
84 || dst->resolved.data == NULL)
85 {
86 errno = EINVAL;
87 saved_errno = errno;
88 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
89 NGX_ERROR, saved_errno, start);
90 return NGX_ERROR;
91 }
92
93 /* Non-POSIX backend: rename within the driver namespace (both endpoints are
94 * keyed export-relative; the move carries content via the catalog). */
95 drv = brix_vfs_ctx_driver(ctx);
96 if (drv != NULL) {
97 ngx_int_t rc = (drv->rename != NULL)
98 ? drv->rename(ctx->sd, brix_vfs_export_relative(ctx, path),
99 brix_vfs_export_relative(ctx,
100 (const char *) dst->resolved.data), 0)
101 : (errno = ENOSYS, NGX_ERROR);
102
103 saved_errno = errno;
104 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
105 rc, saved_errno, start);
106 return rc;
107 }
108
109 res = brix_ns_rename(ctx->log, ctx->root_canon,
110 brix_vfs_ctx_path(ctx),
111 (const char *) dst->resolved.data, 0);
112 if (res.status == BRIX_NS_OK) {
113 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
114 NGX_OK, 0, start);
115 return NGX_OK;
116 }
117
118 errno = res.sys_errno != 0 ? res.sys_errno : EIO;
119 saved_errno = errno;
120 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
121 NGX_ERROR, saved_errno, start);
122 return NGX_ERROR;
123}
