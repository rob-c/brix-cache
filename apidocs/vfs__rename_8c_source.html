1/*
2 * vfs_rename.c — VFS rename / move.
3 *
4 * WHAT: Implements brix_vfs_rename(), which moves the resolved ctx (source)
5 * path to a caller-supplied, already-resolved destination
6 * brix_path_result_t.
7 *
8 * WHY: kXR_mv and WebDAV MOVE need a single write-gated, confined rename whose
9 * destination is verified to be inside the export root before the syscall,
10 * with one metric/access-log line — not an ad-hoc rename(2) per protocol.
11 *
12 * HOW: Enforces brix_vfs_require_write(), then demands a non-NULL root_canon
13 * and a destination that is itself is_confined with a non-empty resolved
14 * path. It delegates the actual move to brix_ns_rename() (namespace
15 * layer), maps the returned status back to errno (sys_errno or EIO), and
16 * observes the operation as BRIX_METRIC_OP_RENAME on every path.
17 */
18#include "vfs_internal.h"
19#include "fs/backend/cache/sd_cache.h" /* brix_sd_cache_evict: leaf-dispatch bypasses the decorator's own evict */
20
21/* Thread-safe confined rename (no pool alloc, no metric) — relocates the
22 * namespace rename into the VFS layer for off-thread / pool-less callers (kXR_mv,
23 * WebDAV MOVE collection offload). Maps the namespace status to errno + an
24 * optional was_dir flag; see vfs.h. */
25ngx_int_t
26brix_vfs_rename_path(brix_sd_instance_t *sd, ngx_log_t *log,
27 const char *root_canon, const char *src, const char *dst,
28 unsigned overwrite, int *was_dir_out)
29{
30 brix_ns_result_t res;
31
32 /* Non-POSIX backend: rename in the driver namespace (export-relative keys).
33 * was_dir is derived from a post-rename stat of the destination. */
34 if (sd != NULL && sd->driver != brix_sd_default_driver()
35 && sd->driver->rename != NULL)
36 {
37 const char *s = brix_vfs_export_relative_root(src, root_canon);
38 const char *d = brix_vfs_export_relative_root(dst, root_canon);
39 ngx_int_t rc = sd->driver->rename(sd, s, d, overwrite ? 0 : 1);
40
41 if (was_dir_out != NULL) {
42 brix_sd_stat_t st;
43
44 *was_dir_out = (rc == NGX_OK && sd->driver->stat != NULL
45 && sd->driver->stat(sd, d, &st) == NGX_OK)
46 ? st.is_dir : 0;
47 }
48 return rc;
49 }
50
51 res = brix_ns_rename(log, root_canon, src, dst, overwrite ? 1 : 0);
52 if (was_dir_out != NULL) {
53 *was_dir_out = res.was_dir;
54 }
55 if (res.status == BRIX_NS_OK) {
56 return NGX_OK;
57 }
58 errno = res.sys_errno != 0 ? res.sys_errno
59 : brix_vfs_ns_status_errno(res.status);
60 return NGX_ERROR;
61}
62
63/* Move the resolved ctx source path to the confined destination `dst`.
64 * Write-gated; both endpoints must be confined. Metered as OP_RENAME.
65 * `overwrite_dirs` lets the rename replace an existing directory destination
66 * (its tree is removed first — WebDAV MOVE Overwrite:T); 0 keeps kXR_mv
67 * semantics where an existing dir dest fails with EEXIST. */
68ngx_int_t
69brix_vfs_rename(brix_vfs_ctx_t *ctx, const brix_path_result_t *dst,
70 unsigned overwrite_dirs)
71{
72 brix_ns_result_t res;
73 const char *path;
74 uint64_t start;
75 int saved_errno;
76 const brix_sd_driver_t *drv;
77
78 start = brix_vfs_now_ns();
79 path = brix_vfs_ctx_path(ctx);
80
81 if (brix_vfs_require_write(ctx) != NGX_OK) {
82 saved_errno = errno;
83 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
84 NGX_ERROR, saved_errno, start);
85 return NGX_ERROR;
86 }
87
88 if (ctx->root_canon == NULL || dst == NULL || !dst->is_confined
89 || dst->resolved.data == NULL)
90 {
91 errno = EINVAL;
92 saved_errno = errno;
93 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
94 NGX_ERROR, saved_errno, start);
95 return NGX_ERROR;
96 }
97
98 /* Non-POSIX backend: rename within the driver namespace (both endpoints are
99 * keyed export-relative; the move carries content via the catalog).
100 * Dispatch on the leaf instance so brix_sd_rename_maybe_cred finds the
101 * leaf driver's rename_cred slot (decorators have only plain relays). */
102 drv = brix_vfs_ctx_driver(ctx);
103 if (drv != NULL) {
104 brix_sd_instance_t *leaf = brix_vfs_ns_leaf(ctx->sd);
105 brix_sd_ucred_t store;
106 brix_sd_cred_t cred;
107 ngx_int_t rc;
108 int use_cred = 0, cred_err = 0;
109
110 /* Zero before the gate: it fills only the active credential kind; an
111 * unzeroed cred hands a garbage inactive pointer to the driver's
112 * cred slot (bearer PASSTHROUGH would leave x509_proxy dangling). */
113 ngx_memzero(&cred, sizeof(cred));
114
115 /* phase-71: catalog-mutation capability gate — a read-only catalog
116 * (CAP_DIRS without CAP_DIRS_WRITE) rejects rename uniformly. */
117 if (!(drv->caps & BRIX_SD_CAP_DIRS_WRITE)) {
118 saved_errno = EPERM;
119 errno = saved_errno;
120 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
121 NGX_ERROR, saved_errno, start);
122 return NGX_ERROR;
123 }
124 if (brix_vfs_cred_gate_active(ctx)) {
125 if (brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err)
126 != NGX_OK)
127 {
128 saved_errno = cred_err ? cred_err : EACCES;
129 errno = saved_errno;
130 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME,
131 NULL, 0, NGX_ERROR, saved_errno,
132 start);
133 return NGX_ERROR;
134 }
135 }
136
137 const char *src_key = brix_vfs_export_relative(ctx, path);
138 const char *dst_key = brix_vfs_export_relative(ctx,
139 (const char *) dst->resolved.data);
140
141 rc = (drv->rename != NULL)
142 ? brix_sd_rename_maybe_cred(leaf, src_key, dst_key, 0,
143 use_cred ? &cred : NULL)
144 : (errno = ENOTSUP, NGX_ERROR);
145
146 saved_errno = errno;
147 brix_sd_ucred_wipe(&store); /* secret consumed by rename; erase (A-4/T4) */
148
149 /* The leaf dispatch above skips the cache decorator's own rename evict,
150 * so drop both endpoints from the store here to keep it coherent. */
151 if (rc == NGX_OK) {
152 brix_sd_cache_evict(ctx->sd, src_key);
153 brix_sd_cache_evict(ctx->sd, dst_key);
154 }
155
156 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
157 rc, saved_errno, start);
158 return rc;
159 }
160
161 res = brix_ns_rename(ctx->log, ctx->root_canon,
162 brix_vfs_ctx_path(ctx),
163 (const char *) dst->resolved.data,
164 overwrite_dirs ? 1 : 0);
165 if (res.status == BRIX_NS_OK) {
166 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
167 NGX_OK, 0, start);
168 return NGX_OK;
169 }
170
171 /* A pre-checked refusal (e.g. destination is an existing directory →
172 * BRIX_NS_EXISTS) carries no syscall errno; map the status instead of
173 * collapsing it to EIO so mv/MOVE report the true category. */
174 errno = res.sys_errno != 0 ? res.sys_errno
175 : brix_vfs_ns_status_errno(res.status);
176 saved_errno = errno;
177 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_RENAME, NULL, 0,
178 NGX_ERROR, saved_errno, start);
179 return NGX_ERROR;
180}
