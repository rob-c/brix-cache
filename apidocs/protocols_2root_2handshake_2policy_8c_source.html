3/*
4 * brix_check_token_scope — enforce WLCG token scopes on a path operation.
5 *
6 * Only active when ctx->token.auth == 1 (bearer-token session). GSI and
7 * anonymous sessions are not restricted by this function — the brix_allow_write
8 * directive and VO ACL remain their access gate.
9 *
10 * logical_path: the client-facing XRootD path (e.g. "/cms/store/file.root"),
11 * NOT the resolved filesystem path.
12 * need_write: 1 for mutations (write/create/delete/chmod/mv/mkdir/rmdir),
13 * 0 for reads (stat/dirlist/locate/open-read/fattr-get).
14 *
15 * Returns NGX_OK if access is granted, NGX_ERROR if denied (the caller must
16 * send the error response and log the access failure).
17 */
18ngx_int_t
19brix_check_token_scope(brix_ctx_t *ctx, const char *logical_path,
20 int need_write)
21{
22 if (!ctx->token.auth) {
23 return NGX_OK;
24 }
25
26 if (ctx->identity != NULL) {
27 return brix_identity_check_token_scope(ctx->identity, logical_path,
28 need_write);
29 }
30
31 if (need_write) {
32 return brix_token_check_write(ctx->token.scopes,
33 ctx->token.scope_count, logical_path)
34 ? NGX_OK : NGX_ERROR;
35 }
36
37 return brix_token_check_read(ctx->token.scopes,
38 ctx->token.scope_count, logical_path)
39 ? NGX_OK : NGX_ERROR;
40}

43brix_min_sec_enforce(brix_ctx_t *ctx, ngx_connection_t *c,
44 ngx_stream_brix_srv_conf_t *conf)
45{
46 if (conf->min_sec_level == BRIX_MIN_SEC_NONE) {
47 return BRIX_DISPATCH_CONTINUE;
48 }
49
50 /*
51 * compat and intense both require a TLS-encrypted transport. brix_tls only
52 * ADVERTISES kXR_ableTLS; a client is free to finish login/auth in cleartext
53 * and reach here — that is precisely the walked-down session this floor
54 * refuses. We reject every data/metadata opcode (session opcodes were
55 * dispatched before this gate) with kXR_TLSRequired; A-1 pairs the upstream
56 * leg so neither side can be downgraded independently.
57 */
58 if (c->ssl == NULL || c->ssl->connection == NULL) {
59 ngx_log_error(NGX_LOG_WARN, c->log, 0,
60 "brix: cleartext session refused by brix_min_sec_level=%ui (opcode=%d)",
61 conf->min_sec_level, (int) ctx->recv.cur_reqid);
62 return brix_send_error(ctx, c, kXR_TLSRequired,
63 "server security policy requires a TLS-encrypted session");
64 }
65
66 /*
67 * intense additionally requires a non-anonymous identity. An auth=none
68 * listener authenticates nobody (anonymous login sets auth_done=1 with no
69 * credential), so even over TLS it is below the floor.
70 */
71 if (conf->min_sec_level >= BRIX_MIN_SEC_INTENSE
72 && conf->auth == BRIX_AUTH_NONE)
73 {
74 ngx_log_error(NGX_LOG_WARN, c->log, 0,
75 "brix: anonymous session refused by brix_min_sec_level=intense (opcode=%d)",
76 (int) ctx->recv.cur_reqid);
77 return brix_send_error(ctx, c, kXR_NotAuthorized,
78 "server security policy requires an authenticated identity");
79 }
80
81 return BRIX_DISPATCH_CONTINUE;
82}
