1/*
2 * kv.h — generic cross-worker key/value store in nginx shared memory.
3 *
4 * An open-addressed (linear-probing) hash table laid out in a single shared
5 * memory segment. Used as the substrate for the JWT validation cache, the
6 * auth-result cache, and the token-bucket rate limiter — each of which is just
7 * a thin typed wrapper over an brix_kv_t zone.
8 *
9 * Design notes:
10 * - Keys and values are opaque byte buffers bounded by key_max / val_max,
11 * fixed at zone-configuration time.
12 * - FNV-1a 64-bit hash, linear probing with backward-shift deletion so the
13 * probe chains stay contiguous (no tombstones).
14 * - Load factor capped at 0.5: set() refuses to insert a *new* key once the
15 * live count reaches capacity/2, bounding worst-case probe distance.
16 * - Expiry is lazy: get() evicts an entry it finds expired and reports a
17 * miss. No background sweeper.
18 * - One spinlock per zone; every op holds it for a single O(1) probe
19 * sequence with no I/O or allocation inside the critical section.
20 */
21#ifndef BRIX_SHM_KV_H
22#define BRIX_SHM_KV_H

34/*
35 * Process-local handle to a shared zone. Allocated from the configuration
36 * pool by the brix_kv_zone directive and pointed at by consumer conf fields
37 * (token_cache_kv, auth_cache_kv, rate-limit kv). The shared table is
38 * slab-allocated and published via kv->zone->data (NOT at shm.addr, which holds
39 * nginx's slab-pool header); mutex is a per-process handle to the shared spinlock.
40 */
41typedef struct {
42 ngx_str_t name; /* zone name — used for Prometheus labels */
43 ngx_shm_zone_t *zone; /* nginx shared-memory zone */
44 ngx_shmtx_t mutex; /* process-local handle to the shared lock */
45 size_t size; /* configured zone size in bytes */
46 size_t key_max; /* max key length per entry */
47 size_t val_max; /* max value length per entry */
48 uint32_t capacity; /* bucket count (power of two), computed at configure */
49 size_t table_bytes; /* header + capacity*stride, slab-allocated at init */
50} brix_kv_t;
