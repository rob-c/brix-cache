1/*
2 * stream_mirror.c — Phase 24 XRootD stream traffic mirror (see stream_mirror.h).
3 *
4 * The mirror is a self-contained async XRootD client connection whose wire
5 * framing reuses the proven upstream bootstrap, exactly like the Phase 22
6 * health-check probe (src/manager/health_check.c): the bootstrap write buffer
7 * is built by brix_upstream_build_bootstrap(), and responses are read as
8 * uniform 8-byte ServerResponseHdr + dlen-byte body frames. Unlike the health
9 * probe, on bootstrap completion it sends the SAVED primary request frame
10 * (header + payload, copied at launch with the streamid rewritten to 0x0002),
11 * reads exactly one response, compares its status to the primary, and discards
12 * the body.
13 *
14 * Lifetime: ctx + connection live in their own pool created from cycle->pool, so
15 * the mirror outlives the client connection that triggered it. All logging uses
16 * ngx_cycle->log, never the (possibly freed) client connection log.
17 */

88/* opcode filter */
89static ngx_uint_t
90brix_mirror_opcode_bit(uint16_t reqid)
91{
92 switch (reqid) {
93 case kXR_stat: return BRIX_MIRROR_OP_STAT;
94 case kXR_locate: return BRIX_MIRROR_OP_LOCATE;
95 case kXR_open: return BRIX_MIRROR_OP_OPEN;
96 case kXR_read: return BRIX_MIRROR_OP_READ;
97 case kXR_readv: return BRIX_MIRROR_OP_READV;
98 case kXR_dirlist: return BRIX_MIRROR_OP_DIRLIST;
99 case kXR_statx: return BRIX_MIRROR_OP_STATX;
100 case kXR_query: return BRIX_MIRROR_OP_QUERY;
101 /* Write/mutation opcodes (Phase 24 write mirroring, gated by
102 * brix_mirror_writes). mkdir/rm/rmdir/mv/truncate/chmod are self-contained
103 * path-based metadata ops replayable by this stateless mirror; data writes
104 * (open-write/write/pgwrite/close) map to OP_WRITE but are handled by the
105 * stateful write-mirror (W3), not here. */
106 case kXR_mkdir: return BRIX_MIRROR_OP_MKDIR;
107 case kXR_rm: return BRIX_MIRROR_OP_RM;
108 case kXR_rmdir: return BRIX_MIRROR_OP_RMDIR;
109 case kXR_mv: return BRIX_MIRROR_OP_MV;
110 case kXR_truncate: return BRIX_MIRROR_OP_TRUNCATE;
111 case kXR_chmod: return BRIX_MIRROR_OP_CHMOD;
112 default: return 0;
113 }
114}

116/* Write-implying open flags. A read-path mirror must never replay these to a
117 * shadow: they would create/truncate/append/mkdir on the official server. */
118#define BRIX_MIRROR_OPEN_WRITE_FLAGS \
119 (kXR_delete | kXR_new | kXR_open_updt | kXR_open_apnd | kXR_mkpath)
120
121/*
122 * Can this request be faithfully replayed to a fresh shadow session?
123 *
124 * The mirror is stateless — it opens a new connection, bootstraps, and sends ONE
125 * saved request frame. That only works for SELF-CONTAINED, side-effect-free
126 * requests whose target lives entirely in the frame:
127 *
128 * - locate / dirlist / query (incl. Qcksum) : path + args are in the payload.
129 * - stat / statx with dlen>0 : path-based (dlen==0 is by open
130 * handle and cannot be replayed).
131 * - open WITHOUT write flags : a read-only open of a path.
132 *
133 * Handle-based ops (read, readv, stat/statx by handle) carry the CLIENT's file
134 * handle, which is meaningless on the shadow's separate session, and write/
135 * create opens would mutate the official server. Neither is replayable by a
136 * stateless mirror, so they are skipped — this is what lets the mirror "just
137 * work" in front of an official xrootd instead of spuriously diverging.
138 */
139static int
140brix_mirror_request_replayable(brix_ctx_t *ctx)
141{
142 switch (ctx->recv.cur_reqid) {
143 case kXR_locate:
144 case kXR_dirlist:
145 case kXR_query:
146 return 1;
147 case kXR_stat:
148 case kXR_statx:
149 return ctx->recv.cur_dlen > 0;
150 case kXR_open: {
151 uint16_t options; /* ClientOpenRequest.options — header byte 6 */
152 ngx_memcpy(&options, ctx->recv.hdr_buf + 6, sizeof(options));
153 options = ntohs(options);
154 return (options & BRIX_MIRROR_OPEN_WRITE_FLAGS) == 0;
155 }
156 /* Metadata mutations (W1): self-contained and path-based — the path[s] live
157 * in the payload, so a fresh shadow session replays them faithfully.
158 * truncate may instead be by open-handle (dlen==0); that form is not
159 * replayable, same as handle-based stat. These reach here only when
160 * brix_mirror_writes is on (gated in brix_stream_mirror_maybe). */
161 case kXR_mkdir:
162 case kXR_rm:
163 case kXR_rmdir:
164 case kXR_mv:
165 case kXR_chmod:
166 case kXR_truncate:
167 return ctx->recv.cur_dlen > 0;
168 default:
169 return 0;
170 }
171}

184/* Bootstrap complete: send the saved primary request frame to the shadow. */
185static void
186brix_mir_send_request(brix_stream_mirror_t *mir)
187{
188 size_t total = (size_t) 24 + mir->saved_dlen;
189 u_char *buf = ngx_palloc(mir->pool, total);
190
191 if (buf == NULL) {
192 brix_mir_finish(mir, 0);
193 return;
194 }
195
196 ngx_memcpy(buf, mir->saved_hdr, 24);
197 buf[0] = 0; buf[1] = 2; /* streamid 0x0002 marks a mirror request */
198 if (mir->saved_dlen > 0 && mir->saved_payload != NULL) {
199 ngx_memcpy(buf + 24, mir->saved_payload, mir->saved_dlen);
200 }
201
202 mir->wbuf = buf;
203 mir->wbuf_len = total;
204 mir->wbuf_pos = 0;
205 mir->rhdr_pos = 0;
206 mir->resp_dlen = 0;
207 mir->resp_body = NULL;
208 mir->resp_body_pos = 0;
209 mir->phase = XRD_MIR_REQUEST;
210
211 if (brix_mir_flush(mir) == NGX_ERROR) {
212 brix_mir_finish(mir, 0);
213 }
214}

216static void
217brix_mir_write_handler(ngx_event_t *wev)
218{
219 ngx_connection_t *c = wev->data;
220 brix_stream_mirror_t *mir = c->data;
221
222 if (wev->timedout) {
223 brix_mir_finish(mir, 0);
224 return;
225 }
226
227 if (mir->connecting) {
228 int err = 0;
229 socklen_t len = sizeof(err);
230 if (getsockopt(c->fd, SOL_SOCKET, SO_ERROR, (char *) &err, &len) == -1
231 || err != 0)
232 {
233 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, mir->log, 0,
234 "xrootd mirror: %s:%d connect failed",
235 mir->host, (int) mir->port);
236 brix_mir_finish(mir, 0);
237 return;
238 }
239 mir->connecting = 0;
240 }
241
242 if (brix_mir_flush(mir) == NGX_ERROR) {
243 brix_mir_finish(mir, 0);
244 }
245}

259/* The shadow answered our replayed request: compare status, count divergence. */
260static void
261brix_mir_on_response(brix_stream_mirror_t *mir)
262{
263 int shadow_ok = (mir->resp_status == kXR_ok
264 || mir->resp_status == kXR_oksofar
265 || mir->resp_status == kXR_redirect);
266
267 /*
268 * A shadow "operation not implemented" is a benign FEATURE-support
269 * difference, not an nginx defect: e.g. mirroring a Qcksum to an official
270 * xrootd that has no checksum configured returns kXR_Unsupported. The
271 * mirror must "just work" in front of any server, so this is never counted
272 * as a divergence — nginx is free to support more than the server it
273 * mirrors. (kXR_error body is errnum[4] + message.)
274 */
275 int shadow_unsupported = 0;
276 if (!shadow_ok && mir->resp_status == kXR_error
277 && mir->resp_body != NULL && mir->resp_dlen >= 4)
278 {
279 uint32_t errnum;
280 ngx_memcpy(&errnum, mir->resp_body, 4);
281 if (ntohl(errnum) == (uint32_t) kXR_Unsupported) {
282 shadow_unsupported = 1;
283 }
284 }
285
286 BRIX_MIR_METRIC_INC(mirror_stream_total);
287
288 if (!shadow_unsupported && shadow_ok != mir->primary_ok) {
289 BRIX_MIR_METRIC_INC(mirror_stream_divergence_total);
290 if (mir->log_diverge) {
291 ngx_log_error(NGX_LOG_NOTICE, mir->log, 0,
292 "xrootd mirror divergence: %s:%d op=%d "
293 "primary_ok=%d shadow_kxr_status=%d",
294 mir->host, (int) mir->port, (int) mir->saved_opcode,
295 mir->primary_ok, (int) mir->resp_status);
296 }
297 }
298
299 brix_mir_finish(mir, 1);
300}

302static void
303brix_mir_dispatch(brix_stream_mirror_t *mir)
304{
305 switch (mir->phase) {
306
307 case XRD_MIR_HANDSHAKE:
308 if (mir->resp_status != kXR_ok) { brix_mir_finish(mir, 0); return; }
309 mir->phase = XRD_MIR_PROTOCOL;
310 break;
311
312 case XRD_MIR_PROTOCOL:
313 if (mir->resp_status != kXR_ok) { brix_mir_finish(mir, 0); return; }
314 /* If the shadow demands TLS we can't continue the cleartext replay; the
315 * server is alive though, so just stop without counting an error. */
316 if (mir->resp_dlen >= 8) {
317 uint32_t flags_be;
318 ngx_memcpy(&flags_be, mir->resp_body + 4, sizeof(flags_be));
319 if (ntohl(flags_be) & kXR_gotoTLS) {
320 brix_mir_finish(mir, 1);
321 return;
322 }
323 }
324 mir->phase = XRD_MIR_LOGIN;
325 break;
326
327 case XRD_MIR_LOGIN:
328 /* authmore => shadow wants credentials we cannot replay; stop quietly. */
329 if (mir->resp_status == kXR_authmore) {
330 brix_mir_finish(mir, 1);
331 return;
332 }
333 if (mir->resp_status != kXR_ok) { brix_mir_finish(mir, 0); return; }
334 brix_mir_send_request(mir); /* sets phase = REQUEST */
335 return;
336
337 case XRD_MIR_REQUEST:
338 brix_mir_on_response(mir);
339 return;
340 }
341
342 /* Reset the accumulator and re-post the read so pipelined bootstrap frames
343 * already in the socket buffer are processed in this cycle. */
344 mir->rhdr_pos = 0;
345 mir->resp_dlen = 0;
346 mir->resp_body = NULL;
347 mir->resp_body_pos = 0;
348
349 if (ngx_handle_read_event(mir->conn->read, 0) != NGX_OK) {
350 brix_mir_finish(mir, 0);
351 return;
352 }
353 ngx_post_event(mir->conn->read, &ngx_posted_events);
354}

356static void
357brix_mir_read_handler(ngx_event_t *rev)
358{
359 ngx_connection_t *c = rev->data;
360 brix_stream_mirror_t *mir = c->data;
361 ngx_int_t rc;
362
363 if (rev->timedout) {
364 brix_mir_finish(mir, 0);
365 return;
366 }
367
368 rc = brix_mir_recv_frame(mir);
369 if (rc == NGX_AGAIN) {
370 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
371 brix_mir_finish(mir, 0);
372 }
373 return;
374 }
375 if (rc == NGX_ERROR) {
376 brix_mir_finish(mir, 0);
377 return;
378 }
379 brix_mir_dispatch(mir);
380}

394static void
395brix_mir_finish(brix_stream_mirror_t *mir, int sent)
396{
397 ngx_pool_t *pool = mir->pool;
398
399 if (mir->tev.timer_set) {
400 ngx_del_timer(&mir->tev);
401 }
402 if (mir->conn != NULL) {
403 ngx_close_connection(mir->conn);
404 mir->conn = NULL;
405 }
406 if (!sent) {
407 BRIX_MIR_METRIC_INC(mirror_stream_errors_total);
408 }
409 ngx_destroy_pool(pool); /* frees mir itself (last use above) */

412static void
413brix_mir_start(brix_stream_mirror_t *mir, ngx_msec_t timeout_ms)
414{
415 ngx_connection_t *c;
416 ngx_socket_t fd;
417 size_t bslen;
418 int rc;
419
420 fd = ngx_socket(mir->sockaddr.ss_family, SOCK_STREAM, 0);
421 if (fd == (ngx_socket_t) -1 || ngx_nonblocking(fd) == -1) {
422 if (fd != (ngx_socket_t) -1) { ngx_close_socket(fd); }
423 brix_mir_finish(mir, 0);
424 return;
425 }
426
427 c = ngx_get_connection(fd, mir->log);
428 if (c == NULL) {
429 ngx_close_socket(fd);
430 brix_mir_finish(mir, 0);
431 return;
432 }
433 c->pool = mir->pool;
434 c->data = mir;
435 c->recv = ngx_recv;
436 c->send = ngx_send;
437 c->read->handler = brix_mir_read_handler;
438 c->write->handler = brix_mir_write_handler;
439 c->read->log = c->write->log = mir->log;
440 mir->conn = c;
441
442 /* Pipelined bootstrap: handshake + protocol + login. */
443 bslen = XRD_HANDSHAKE_LEN + sizeof(ClientProtocolRequest)
444 + sizeof(ClientLoginRequest);
445 mir->wbuf = ngx_palloc(mir->pool, bslen);
446 if (mir->wbuf == NULL) {
447 brix_mir_finish(mir, 0);
448 return;
449 }
450 brix_upstream_build_bootstrap(mir->wbuf);
451 mir->wbuf_len = bslen;
452 mir->wbuf_pos = 0;
453
454 mir->tev.handler = brix_mir_timeout_handler;
455 mir->tev.data = mir;
456 mir->tev.log = mir->log;
457 ngx_add_timer(&mir->tev, timeout_ms);
458
459 rc = connect(fd, (struct sockaddr *) &mir->sockaddr, mir->socklen);
460
461 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
462 brix_mir_finish(mir, 0);
463 return;
464 }
465 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
466 brix_mir_finish(mir, 0);
467 return;
468 }
469
470 if (rc == 0) {
471 mir->connecting = 0;
472 if (brix_mir_flush(mir) == NGX_ERROR) {
473 brix_mir_finish(mir, 0);
474 }
475 } else {
476 mir->connecting = 1; /* completes in the write handler */
477 }
478}

481/* launch hook (called from dispatch.c) */
482void
483brix_stream_mirror_maybe(brix_ctx_t *ctx, ngx_connection_t *c,
484 ngx_stream_brix_srv_conf_t *conf, ngx_int_t primary_rc)
485{
486 brix_mirror_target_t *targets;
487 ngx_uint_t opbit, i;
488 int primary_ok;
489
490 if (!conf->mirror.enabled || conf->mirror.targets == NULL) {
491 return;
492 }
493
494 opbit = brix_mirror_opcode_bit(ctx->recv.cur_reqid);
495 /* Mirror all ops in opcode_mask (default ALL) that are not de-selected via
496 * brix_mirror_exclude_opcodes. */
497 if (opbit == 0
498 || (conf->mirror.opcode_mask & opbit) == 0
499 || (conf->mirror.opcode_exclude_mask & opbit) != 0)
500 {
501 return;
502 }
503
504 /* Second, independent guard for write/mutation opcodes: even if an operator
505 * lists e.g. "mkdir" in brix_mirror_opcodes, it stays inert unless
506 * brix_mirror_writes is explicitly on (and the shadow is an isolated
507 * namespace). OP_WRITE (data writes) is handled by the stateful write-mirror,
508 * not this one-shot path, so it never proceeds here. */
509 if ((opbit & BRIX_MIRROR_OP_WRITE_ALL) != 0) {
510 if (!conf->mirror.mirror_writes || opbit == BRIX_MIRROR_OP_WRITE) {
511 return;
512 }
513 }
514
515 /* Only replay requests the shadow can answer standalone (path-based reads,
516 * read-only opens, query/Qcksum). Handle-based reads/readv/handle-stat and
517 * write opens are skipped so the mirror never spuriously diverges against an
518 * official xrootd — see brix_mirror_request_replayable(). */
519 if (!brix_mirror_request_replayable(ctx)) {
520 return;
521 }
522
523 /* Skip oversized payloads (write bodies); also a sanity guard. */
524 if (ctx->recv.cur_dlen > BRIX_MIRROR_MAX_PAYLOAD) {
525 return;
526 }
527
528 if (!brix_mirror_should_sample(conf->mirror.sample_pct)) {
529 BRIX_MIR_METRIC_INC(mirror_stream_dropped_total);
530 return;
531 }
532
533 primary_ok = (primary_rc != NGX_ERROR);
534 targets = conf->mirror.targets->elts;
535
536 for (i = 0; i < conf->mirror.targets->nelts; i++) {
537 ngx_pool_t *pool;
538 brix_stream_mirror_t *mir;
539
540 if (targets[i].socklen == 0) {
541 continue; /* unresolved target */
542 }
543
544 pool = ngx_create_pool(2048, ngx_cycle->log);
545 if (pool == NULL) {
546 continue;
547 }
548 mir = ngx_pcalloc(pool, sizeof(*mir));
549 if (mir == NULL) {
550 ngx_destroy_pool(pool);
551 continue;
552 }
553 mir->pool = pool;
554 mir->log = ngx_cycle->log; /* outlives the client connection */
555 mir->phase = XRD_MIR_HANDSHAKE;
556 mir->primary_ok = primary_ok;
557 mir->log_diverge = conf->mirror.log_diverge;
558 mir->port = targets[i].port;
559 mir->socklen = targets[i].socklen;
560 ngx_memcpy(&mir->sockaddr, &targets[i].sockaddr, targets[i].socklen);
561 ngx_cpystrn((u_char *) mir->host,
562 targets[i].host.data ? targets[i].host.data
563 : (u_char *) "?",
564 sizeof(mir->host));
565
566 /* Snapshot the request now — ctx->recv.payload may be reused by the next
567 * request before the shadow exchange completes. */
568 ngx_memcpy(mir->saved_hdr, ctx->recv.hdr_buf, 24);
569 mir->saved_opcode = ctx->recv.cur_reqid;
570 if (ctx->recv.payload != NULL && ctx->recv.cur_dlen > 0) {
571 mir->saved_payload = ngx_palloc(pool, ctx->recv.cur_dlen);
572 if (mir->saved_payload != NULL) {
573 ngx_memcpy(mir->saved_payload, ctx->recv.payload, ctx->recv.cur_dlen);
574 mir->saved_dlen = ctx->recv.cur_dlen;
575 }
576 }
577
578 brix_mir_start(mir, conf->mirror.timeout_ms);
579 }
580}

583/* directive setters */
584/*
585 * brix_stream_mirror_url host:port — append one shadow target, resolved at
586 * configuration time so request handlers never call getaddrinfo on the event
587 * loop. Up to BRIX_MIRROR_MAX_TARGETS may be configured.
588 */
589char *
590brix_stream_mirror_set_url(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
591{
592 ngx_stream_brix_srv_conf_t *xcf = conf;
593 ngx_str_t *value = cf->args->elts;
594 brix_mirror_target_t *t;
595 ngx_url_t u;
596 u_char *colon;
597 ngx_str_t hostport = value[1];
598
599 (void) cmd;
600
601 if (xcf->mirror.targets == NULL) {
602 xcf->mirror.targets = ngx_array_create(cf->pool,
603 BRIX_MIRROR_MAX_TARGETS, sizeof(brix_mirror_target_t));
604 if (xcf->mirror.targets == NULL) {
605 return NGX_CONF_ERROR;
606 }
607 }
608 if (xcf->mirror.targets->nelts >= BRIX_MIRROR_MAX_TARGETS) {
609 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
610 "brix_stream_mirror_url: at most %d targets supported",
611 BRIX_MIRROR_MAX_TARGETS);
612 return NGX_CONF_ERROR;
613 }
614
615 if (hostport.len == 0
616 || ngx_strlchr(hostport.data, hostport.data + hostport.len, ':')
617 == NULL)
618 {
619 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
620 "brix_stream_mirror_url: expected host:port, got \"%V\"",
621 &hostport);
622 return NGX_CONF_ERROR;
623 }
624
625 ngx_memzero(&u, sizeof(ngx_url_t));
626 u.url = hostport;
627 u.default_port = 1094;
628 if (ngx_parse_url(cf->pool, &u) != NGX_OK || u.naddrs == 0) {
629 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
630 "brix_stream_mirror_url: cannot resolve \"%V\"%s%s", &hostport,
631 u.err ? ": " : "", u.err ? u.err : "");
632 return NGX_CONF_ERROR;
633 }
634
635 t = ngx_array_push(xcf->mirror.targets);
636 if (t == NULL) {
637 return NGX_CONF_ERROR;
638 }
639 ngx_memzero(t, sizeof(*t));
640 t->url = hostport;
641 t->host = u.host;
642 t->port = u.port;
643 if (u.addrs[0].socklen > sizeof(t->sockaddr)) {
644 return NGX_CONF_ERROR;
645 }
646 ngx_memcpy(&t->sockaddr, u.addrs[0].sockaddr, u.addrs[0].socklen);
647 t->socklen = u.addrs[0].socklen;
648
649 /* keep the host:port colon split out of the host label for logging */
650 colon = ngx_strlchr(hostport.data, hostport.data + hostport.len, ':');
651 if (colon != NULL && t->host.len == 0) {
652 t->host.data = hostport.data;
653 t->host.len = (size_t) (colon - hostport.data);
654 }
655
656 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
657 "brix: stream mirror target %V", &hostport);
658 return NGX_CONF_OK;
659}

661/*
662 * Parse opcode name args (cf->args[1..]) into a bitmask. "all" expands to
663 * BRIX_MIRROR_OP_ALL. Shared by the allowlist and exclude setters.
664 */
665static char *
666brix_mirror_parse_opcode_args(ngx_conf_t *cf, const char *directive,
667 ngx_uint_t *mask_out)
668{
669 ngx_str_t *value = cf->args->elts;
670 ngx_uint_t i, mask = 0;
671
672 for (i = 1; i < cf->args->nelts; i++) {
673 ngx_str_t *v = &value[i];
674 if (ngx_strcmp(v->data, "all") == 0) mask |= BRIX_MIRROR_OP_ALL;
675 else if (ngx_strcmp(v->data, "stat") == 0) mask |= BRIX_MIRROR_OP_STAT;
676 else if (ngx_strcmp(v->data, "locate") == 0) mask |= BRIX_MIRROR_OP_LOCATE;
677 else if (ngx_strcmp(v->data, "open") == 0) mask |= BRIX_MIRROR_OP_OPEN;
678 else if (ngx_strcmp(v->data, "read") == 0) mask |= BRIX_MIRROR_OP_READ;
679 else if (ngx_strcmp(v->data, "readv") == 0) mask |= BRIX_MIRROR_OP_READV;
680 else if (ngx_strcmp(v->data, "dirlist") == 0) mask |= BRIX_MIRROR_OP_DIRLIST;
681 else if (ngx_strcmp(v->data, "statx") == 0) mask |= BRIX_MIRROR_OP_STATX;
682 else if (ngx_strcmp(v->data, "query") == 0) mask |= BRIX_MIRROR_OP_QUERY;
683 /* Write/mutation opcodes (require brix_mirror_writes on). */
684 else if (ngx_strcmp(v->data, "mkdir") == 0) mask |= BRIX_MIRROR_OP_MKDIR;
685 else if (ngx_strcmp(v->data, "rm") == 0) mask |= BRIX_MIRROR_OP_RM;
686 else if (ngx_strcmp(v->data, "rmdir") == 0) mask |= BRIX_MIRROR_OP_RMDIR;
687 else if (ngx_strcmp(v->data, "mv") == 0) mask |= BRIX_MIRROR_OP_MV;
688 else if (ngx_strcmp(v->data, "truncate") == 0) mask |= BRIX_MIRROR_OP_TRUNCATE;
689 else if (ngx_strcmp(v->data, "chmod") == 0) mask |= BRIX_MIRROR_OP_CHMOD;
690 else if (ngx_strcmp(v->data, "write") == 0) mask |= BRIX_MIRROR_OP_WRITE;
691 else {
692 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
693 "%s: unknown opcode \"%V\" (expected one of"
694 " all stat locate open read readv dirlist statx query"
695 " mkdir rm rmdir mv truncate chmod write)",
696 directive, v);
697 return NGX_CONF_ERROR;
698 }
699 }
700 *mask_out = mask;
701 return NGX_CONF_OK;
702}

704/*
705 * brix_mirror_opcodes stat locate open ... — RESTRICT mirroring to exactly
706 * the named opcodes (allowlist; overrides the default-all). Most operators
707 * want the default (mirror everything) or brix_mirror_exclude_opcodes instead.
708 */
709char *
710brix_stream_mirror_set_opcodes(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
