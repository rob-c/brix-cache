1/*
2 * stream_mirror.c — Phase 24 XRootD stream traffic mirror: the async shadow-
3 * connection state machine (see stream_mirror.h / stream_mirror_internal.h).
4 *
5 * The mirror is a self-contained async XRootD client connection whose wire
6 * framing reuses the proven upstream bootstrap, exactly like the Phase 22
7 * health-check probe (src/manager/health_check.c): the bootstrap write buffer
8 * is built by brix_upstream_build_bootstrap(), and responses are read as
9 * uniform 8-byte ServerResponseHdr + dlen-byte body frames. Unlike the health
10 * probe, on bootstrap completion it sends the SAVED primary request frame
11 * (header + payload, copied at launch with the streamid rewritten to 0x0002),
12 * reads exactly one response, compares its status to the primary, and discards
13 * the body.
14 *
15 * Lifetime: ctx + connection live in their own pool created from cycle->pool, so
16 * the mirror outlives the client connection that triggered it. All logging uses
17 * ngx_cycle->log, never the (possibly freed) client connection log.
18 *
19 * Split (phase-79 file-size cap) into three focused files:
20 * stream_mirror.c — this file: the shadow-connection state machine
21 * (bootstrap/replay send, frame read, phase dispatch,
22 * divergence compare, lifecycle start/finish).
23 * stream_mirror_launch.c — eligibility gates + per-target launch + the public
24 * brix_stream_mirror_maybe() request-path hook.
25 * stream_mirror_config.c — configuration-time directive setters.
26 * The context struct, phase enum, metric macro, replay-payload bound, and the
27 * single cross-file entry point brix_mir_start() live in stream_mirror_internal.h.
28 */

57/* Bootstrap complete: send the saved primary request frame to the shadow. */
58static void
59brix_mir_send_request(brix_stream_mirror_t *mir)
60{
61 size_t total = (size_t) 24 + mir->saved_dlen;
62 u_char *buf = ngx_palloc(mir->pool, total);
63
64 if (buf == NULL) {
65 brix_mir_finish(mir, 0);
66 return;
67 }
68
69 ngx_memcpy(buf, mir->saved_hdr, 24);
70 buf[0] = 0; buf[1] = 2; /* streamid 0x0002 marks a mirror request */
71 if (mir->saved_dlen > 0 && mir->saved_payload != NULL) {
72 ngx_memcpy(buf + 24, mir->saved_payload, mir->saved_dlen);
73 }
74
75 mir->wbuf = buf;
76 mir->wbuf_len = total;
77 mir->wbuf_pos = 0;
78 mir->rhdr_pos = 0;
79 mir->resp_dlen = 0;
80 mir->resp_body = NULL;
81 mir->resp_body_pos = 0;
82 mir->phase = XRD_MIR_REQUEST;
83
84 if (brix_mir_flush(mir) == NGX_ERROR) {
85 brix_mir_finish(mir, 0);
86 }
87}

89static void
90brix_mir_write_handler(ngx_event_t *wev)
91{
92 ngx_connection_t *c = wev->data;
93 brix_stream_mirror_t *mir = c->data;
94
95 if (wev->timedout) {
96 brix_mir_finish(mir, 0);
97 return;
98 }
99
100 if (mir->connecting) {
101 int err = 0;
102 socklen_t len = sizeof(err);
103 if (getsockopt(c->fd, SOL_SOCKET, SO_ERROR, (char *) &err, &len) == -1
104 || err != 0)
105 {
106 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, mir->log, 0,
107 "xrootd mirror: %s:%d connect failed",
108 mir->host, (int) mir->port);
109 brix_mir_finish(mir, 0);
110 return;
111 }
112 mir->connecting = 0;
113 }
114
115 if (brix_mir_flush(mir) == NGX_ERROR) {
116 brix_mir_finish(mir, 0);
117 }
118}

132/* The shadow answered our replayed request: compare status, count divergence. */
133static void
134brix_mir_on_response(brix_stream_mirror_t *mir)
135{
136 int shadow_ok = (mir->resp_status == kXR_ok
137 || mir->resp_status == kXR_oksofar
138 || mir->resp_status == kXR_redirect);
139
140 /*
141 * A shadow "operation not implemented" is a benign FEATURE-support
142 * difference, not an nginx defect: e.g. mirroring a Qcksum to an official
143 * xrootd that has no checksum configured returns kXR_Unsupported. The
144 * mirror must "just work" in front of any server, so this is never counted
145 * as a divergence — nginx is free to support more than the server it
146 * mirrors. (kXR_error body is errnum[4] + message.)
147 */
148 int shadow_unsupported = 0;
149 if (!shadow_ok && mir->resp_status == kXR_error
150 && mir->resp_body != NULL && mir->resp_dlen >= 4)
151 {
152 uint32_t errnum;
153 ngx_memcpy(&errnum, mir->resp_body, 4);
154 if (ntohl(errnum) == (uint32_t) kXR_Unsupported) {
155 shadow_unsupported = 1;
156 }
157 }
158
159 BRIX_MIR_METRIC_INC(mirror_stream_total);
160
161 if (!shadow_unsupported && shadow_ok != mir->primary_ok) {
162 BRIX_MIR_METRIC_INC(mirror_stream_divergence_total);
163 if (mir->log_diverge) {
164 ngx_log_error(NGX_LOG_NOTICE, mir->log, 0,
165 "xrootd mirror divergence: %s:%d op=%d "
166 "primary_ok=%d shadow_kxr_status=%d",
167 mir->host, (int) mir->port, (int) mir->saved_opcode,
168 mir->primary_ok, (int) mir->resp_status);
169 }
170 }
171
172 brix_mir_finish(mir, 1);
173}

175static void
176brix_mir_dispatch(brix_stream_mirror_t *mir)
177{
178 switch (mir->phase) {
179
180 case XRD_MIR_HANDSHAKE:
181 if (mir->resp_status != kXR_ok) { brix_mir_finish(mir, 0); return; }
182 mir->phase = XRD_MIR_PROTOCOL;
183 break;
184
185 case XRD_MIR_PROTOCOL:
186 if (mir->resp_status != kXR_ok) { brix_mir_finish(mir, 0); return; }
187 /* If the shadow demands TLS we can't continue the cleartext replay; the
188 * server is alive though, so just stop without counting an error. */
189 if (mir->resp_dlen >= 8) {
190 uint32_t flags_be;
191 ngx_memcpy(&flags_be, mir->resp_body + 4, sizeof(flags_be));
192 if (ntohl(flags_be) & kXR_gotoTLS) {
193 brix_mir_finish(mir, 1);
194 return;
195 }
196 }
197 mir->phase = XRD_MIR_LOGIN;
198 break;
199
200 case XRD_MIR_LOGIN:
201 /* authmore => shadow wants credentials we cannot replay; stop quietly. */
202 if (mir->resp_status == kXR_authmore) {
203 brix_mir_finish(mir, 1);
204 return;
205 }
206 if (mir->resp_status != kXR_ok) { brix_mir_finish(mir, 0); return; }
207 brix_mir_send_request(mir); /* sets phase = REQUEST */
208 return;
209
210 case XRD_MIR_REQUEST:
211 brix_mir_on_response(mir);
212 return;
213 }
214
215 /* Reset the accumulator and re-post the read so pipelined bootstrap frames
216 * already in the socket buffer are processed in this cycle. */
217 mir->rhdr_pos = 0;
218 mir->resp_dlen = 0;
219 mir->resp_body = NULL;
220 mir->resp_body_pos = 0;
221
222 if (ngx_handle_read_event(mir->conn->read, 0) != NGX_OK) {
223 brix_mir_finish(mir, 0);
224 return;
225 }
226 ngx_post_event(mir->conn->read, &ngx_posted_events);
227}

229static void
230brix_mir_read_handler(ngx_event_t *rev)
231{
232 ngx_connection_t *c = rev->data;
233 brix_stream_mirror_t *mir = c->data;
234 ngx_int_t rc;
235
236 if (rev->timedout) {
237 brix_mir_finish(mir, 0);
238 return;
239 }
240
241 rc = brix_mir_recv_frame(mir);
242 if (rc == NGX_AGAIN) {
243 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
244 brix_mir_finish(mir, 0);
245 }
246 return;
247 }
248 if (rc == NGX_ERROR) {
249 brix_mir_finish(mir, 0);
250 return;
251 }
252 brix_mir_dispatch(mir);
253}

267static void
268brix_mir_finish(brix_stream_mirror_t *mir, int sent)
269{
270 ngx_pool_t *pool = mir->pool;
271
272 if (mir->tev.timer_set) {
273 ngx_del_timer(&mir->tev);
274 }
275 if (mir->conn != NULL) {
276 ngx_close_connection(mir->conn);
277 mir->conn = NULL;
278 }
279 if (!sent) {
280 BRIX_MIR_METRIC_INC(mirror_stream_errors_total);
281 }
282 ngx_destroy_pool(pool); /* frees mir itself (last use above) */
283}

285void
286brix_mir_start(brix_stream_mirror_t *mir, ngx_msec_t timeout_ms)
287{
288 ngx_connection_t *c;
289 ngx_socket_t fd;
290 size_t bslen;
291 int rc;
292
293 fd = ngx_socket(mir->sockaddr.ss_family, SOCK_STREAM, 0);
294 if (fd == (ngx_socket_t) -1 || ngx_nonblocking(fd) == -1) {
295 if (fd != (ngx_socket_t) -1) { ngx_close_socket(fd); }
296 brix_mir_finish(mir, 0);
297 return;
298 }
299
300 c = ngx_get_connection(fd, mir->log);
301 if (c == NULL) {
302 ngx_close_socket(fd);
303 brix_mir_finish(mir, 0);
304 return;
305 }
306 c->pool = mir->pool;
307 c->data = mir;
308 c->recv = ngx_recv;
309 c->send = ngx_send;
310 c->read->handler = brix_mir_read_handler;
311 c->write->handler = brix_mir_write_handler;
312 c->read->log = c->write->log = mir->log;
313 mir->conn = c;
314
315 /* Pipelined bootstrap: handshake + protocol + login. */
316 bslen = XRD_HANDSHAKE_LEN + sizeof(ClientProtocolRequest)
317 + sizeof(ClientLoginRequest);
318 mir->wbuf = ngx_palloc(mir->pool, bslen);
319 if (mir->wbuf == NULL) {
320 brix_mir_finish(mir, 0);
321 return;
322 }
323 brix_upstream_build_bootstrap(mir->wbuf);
324 mir->wbuf_len = bslen;
325 mir->wbuf_pos = 0;
326
327 mir->tev.handler = brix_mir_timeout_handler;
328 mir->tev.data = mir;
329 mir->tev.log = mir->log;
330 ngx_add_timer(&mir->tev, timeout_ms);
331
332 rc = connect(fd, (struct sockaddr *) &mir->sockaddr, mir->socklen);
333
334 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
335 brix_mir_finish(mir, 0);
336 return;
337 }
338 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
339 brix_mir_finish(mir, 0);
340 return;
341 }
342
343 if (rc == 0) {
344 mir->connecting = 0;
345 if (brix_mir_flush(mir) == NGX_ERROR) {
346 brix_mir_finish(mir, 0);
347 }
348 } else {
349 mir->connecting = 1; /* completes in the write handler */
350 }
