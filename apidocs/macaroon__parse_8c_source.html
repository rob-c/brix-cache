1/* Macaroon HMAC-SHA256 signature chain — packet framing, per-type chain updates, third-party caveat capture, and final signature verification.
2 *
3 * WHAT: Parses one macaroon binary as a sequence of length-prefixed packets, reconstructing the HMAC-SHA256 chain
4 * (sig = HMAC(sig_prev, packet_data)) across identifier/location/cid/vid/signature packets, capturing third-party
5 * caveats (cid+vid+sig_before) into the caller's tp_arr, verifying the final signature in constant time, enforcing
6 * fail-closed expiry, and finalizing the extracted scopes into claims.
7 *
8 * WHY: Split out of macaroon.c (phase-79 file-size split). The HMAC chain is the integrity core of the macaroon
9 * security model — any tampered or reordered caveat yields a mismatched final signature. Keeping the exact HMAC
10 * ordering (identifier seeds the chain; cid/vid each fold in before capture; signature compared with CRYPTO_memcmp)
11 * in one file makes that guarantee auditable, separate from caveat interpretation (macaroon_caveats.c) and vid
12 * crypto (macaroon_crypto.c).
13 *
14 * HOW: brix_macaroon_packet_len() reads the 4-char hex length. macaroon_parse_core() loops packets, dispatching each
15 * via macaroon_dispatch_packet() to the per-type handlers: identifier seeds sig=HMAC(key,id); cid saves sig_before_cid
16 * then folds the caveat and hands first-party caveats to macaroon_parse_first_party_caveat(); vid folds and records the
17 * third-party triple; signature compares. After the loop it checks found_sig/expiry and finalizes scopes. */

32int
33brix_macaroon_packet_len(const u_char *p)
34/* WHAT: Parse a 4-character hex-encoded packet length from macaroon binary data.
35 * WHY: Macaroon packets are prefixed with a hex-encoded 32-bit length field (8 hex chars → uint32). This helper converts the first 4 hex characters into an integer for bounds checking before reading packet data.
36 * HOW: Call brix_hex_from_char() on each of p[0..3], reject if any nibble invalid (<0); combine via bit shifts (v0<<12)|(v1<<8)|(v2<<4)|v3 to form uint32; return value or -1 on invalid nibble. */
37{
38 int v0, v1, v2, v3;
39 v0 = brix_hex_from_char(p[0]);
40 v1 = brix_hex_from_char(p[1]);
41 v2 = brix_hex_from_char(p[2]);
42 v3 = brix_hex_from_char(p[3]);
43 if (v0 < 0 || v1 < 0 || v2 < 0 || v3 < 0) return -1;
44 return (v0 << 12) | (v1 << 8) | (v2 << 4) | v3;
45}

47static ngx_int_t
48macaroon_packet_identifier(brix_macaroon_parse_state_t *state,
49 const u_char *data, size_t data_len)
50{
51 const u_char *identifier = data + 11;
52 size_t identifier_len = data_len - 11;
53
54 HMAC(EVP_sha256(), state->key, state->key_len, identifier,
55 identifier_len, state->sig, &state->sig_out_len);
56 ngx_memcpy(state->claims->sub, identifier,
57 identifier_len < sizeof(state->claims->sub)
58 ? identifier_len : sizeof(state->claims->sub) - 1);
59 state->found_id = 1;
60 state->have_last_cid = 0;
61 return NGX_OK;
62}

76static ngx_int_t
77macaroon_packet_cid(brix_macaroon_parse_state_t *state,
78 const u_char *data, size_t data_len)
79{
80 const u_char *caveat = data + 4;
81 size_t caveat_len = data_len - 4;
82 u_char next_sig[32];
83
84 if (!state->found_id) {
85 ngx_log_error(NGX_LOG_WARN, state->log, 0,
86 "brix_macaroon: caveat before identifier");
87 return NGX_ERROR;
88 }
89
90 ngx_memcpy(state->sig_before_cid, state->sig, 32);
91 HMAC(EVP_sha256(), state->sig, 32, caveat, caveat_len, next_sig,
92 &state->sig_out_len);
93 ngx_memcpy(state->sig, next_sig, 32);
94
95 state->last_cid_len = caveat_len < sizeof(state->last_cid)
96 ? caveat_len : sizeof(state->last_cid) - 1;
97 ngx_memcpy(state->last_cid, caveat, state->last_cid_len);
98 state->have_last_cid = 1;
99 macaroon_parse_first_party_caveat(state, caveat, caveat_len);
100 return NGX_OK;
101}

105 const u_char *vid_data, size_t vid_len)
106{
107 brix_macaroon_tp_t *tp;
108
109 if (!state->have_last_cid || state->tp_arr == NULL || state->n_tp == NULL
110 || *state->n_tp >= state->max_tp) {
111 return;
112 }
113
114 tp = &state->tp_arr[(*state->n_tp)++];
115 ngx_memcpy(tp->cid, state->last_cid, state->last_cid_len);
116 tp->cid_len = state->last_cid_len;
117 tp->vid_len = vid_len < sizeof(tp->vid) ? vid_len : sizeof(tp->vid) - 1;
118 ngx_memcpy(tp->vid, vid_data, tp->vid_len);
119 ngx_memcpy(tp->sig_before, state->sig_before_cid, 32);

171 if (data_len >= 10 && memcmp(data, "signature ", 10) == 0) {
172 return macaroon_packet_signature(state, data, data_len);

227/* WHAT: Parse one macaroon binary, reconstruct HMAC-SHA256 signature chain across all packets, verify final signature, and extract WLCG caveats into claims.
228 * WHY: The macaroon security model requires each caveat to deterministically modify the HMAC chain — sig = HMAC(sig_prev, caveat_data). This ensures any tampered or reordered caveat produces a mismatched final signature. Extracting activity:/path:/before: caveats converts raw binary authorization into structured claims for access control decisions.
229 * HOW: Initialize sig=HMAC(key, identifier), scope_buf="", path_caveats[], last_cid/sig_before_cid state; loop packets (p+4≤end): brix_macaroon_packet_len(p)→plen; data=p+4,dlen=plen-4; strip trailing newline if present; process packet types: "identifier " → HMAC(EVP_sha256,key,identifier)→sig, copy to claims->sub; "location " → copy to claims->iss; "cid " → save sig_before_cid, HMAC(sig,cid)→next_sig→sig, track last_cid for vid pairing; parse first-party caveats within cid data (activity:→scope mapping, before:→parse_iso8601→claims->exp min, path:→path_caveats array); "vid " → HMAC(sig,vid_data)→sig, record (cid+vid+sig_before) triple into tp_arr if available; "signature " → compare provided 32-byte sig against computed sig, reject mismatch; after loop: check found_sig and found_id, validate expiry (now>claims->exp), finalize scopes from scope_buf via brix_token_parse_scopes(), apply path caveats via macaroon_apply_path_caveats(); return 0 success or -1 failure. */
230int
231macaroon_parse_core(const macaroon_parse_input_t *in,
232 const u_char *bin, size_t bin_len)
233{
234 brix_macaroon_parse_state_t state;
235 ngx_log_t *log = in->log;
236 const u_char *p, *end;
237
238 macaroon_parse_state_init(&state, in);
239 p = bin;
240 end = bin + bin_len;
241
242 while (p + 4 <= end) {
243 int plen = brix_macaroon_packet_len(p);
244 u_char *data;
245 size_t dlen;
246
247 if (plen < 4 || p + plen > end) {
248 ngx_log_error(NGX_LOG_WARN, log, 0,
249 "brix_macaroon: malformed packet length: %d", plen);
250 return -1;
251 }
252
253 data = (u_char *)(p + 4);
254 dlen = (size_t)(plen - 4);
255
256 /* Strip trailing newline if present */
257 if (dlen > 0 && data[dlen - 1] == '\n') {
258 dlen--;
259 }
260
261 if (macaroon_dispatch_packet(&state, data, dlen) != NGX_OK) {
262 return -1;
263 }
264
265 p += plen;
266 }
267
268 if (!state.found_sig) {
269 ngx_log_error(NGX_LOG_WARN, log, 0,
270 "brix_macaroon: no valid signature found");
271 return -1;
272 }
273
274 /*
275 * Expiry handling (fail-closed). A macaroon is a bearer credential: a root
276 * with no before: caveat leaves claims->exp == 0 and would otherwise be valid
277 * forever, so a single leak would never lapse. We therefore REQUIRE an
278 * expiry on the root/standalone macaroon (tp_arr != NULL identifies that
279 * context; discharges are validated with tp_arr == NULL). dCache/WLCG
280 * macaroons always carry a before: caveat on the root, so this rejects only
281 * malformed or deliberately-unbounded tokens. Discharge macaroons may
282 * legitimately omit before: — their lifetime is governed by the root and
283 * intersected by the caller — so we only enforce "not already expired" for
284 * them. claims->exp is the earliest before: caveat seen in the packet loop
285 * above (macaroons only narrow, never widen).
286 */
287 if (macaroon_check_expiry(&state) != NGX_OK) {
288 return -1;
289 }
290
291 macaroon_finalize_scopes(&state);
292 return 0;
293}
