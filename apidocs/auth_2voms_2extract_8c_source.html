24/*
25 * WHAT: Public entry point for extracting VOMS virtual organisation membership
26 * from an x509 proxy certificate. Called by the GSI authentication path after
27 * the proxy chain is verified to populate ctx->primary_vo and ctx->vo_list.
28 *
29 * WHY: VO-based ACL enforcement (brix_require_vo) requires knowledge of which
30 * virtual organisations the user belongs to. VOMS extensions embedded in the
31 * certificate provide this information. This function encapsulates the entire
32 * extraction pipeline — chain preparation, API call, and result collection.
33 *
34 * HOW: Four-phase flow:
35 * 1. Pre-checks — return NGX_DECLINED if VOMS library not loaded or parameters
36 * missing (graceful degradation per INVARIANT #8: metric labels low-cardinality)
37 * 2. Buffer preparation — convert ngx_str_t paths to NUL-terminated buffers with
38 * bounds validation against PATH_MAX
39 * 3. VOMS extraction — initialise API, duplicate certificate chain and remove the
40 * leaf (VOMS needs parent certificates for extension lookup), call retrieve()
41 * with VOMS_RECURSE_CHAIN flag; non-critical errors (VOMS_VERR_NOEXT /
42 * VOMS_VERR_NODATA) are silently skipped per GSI auth convention
43 * 4. Result collection — delegate to brix_collect_voms_vos(); cleanup chain and
44 * API state regardless of outcome
45 *
46 * INVARIANT: All wire paths → resolve_path() before open(). This function operates
47 * on certificate data, not filesystem paths, so the invariant does not apply directly.
48 */

50/*
51 * WHAT: Validate that VOMS extraction can proceed and that the caller-supplied
52 * directory paths fit their fixed on-stack buffers.
53 *
54 * WHY: Splits the pre-flight guard ladder out of the main entry point so the
55 * latter stays under the complexity cap. Preserves the original two-tier
56 * disposition exactly: unavailable/missing inputs degrade gracefully
57 * (NGX_DECLINED) while over-length paths are a hard error (NGX_ERROR).
58 *
59 * HOW: Mirror the original guard order — VOMS library loaded, mandatory
60 * non-empty inputs present, then path lengths bounded below the destination
61 * buffer size. Returns NGX_OK when extraction may continue.
62 */
63static ngx_int_t
64brix_voms_precheck(X509 *leaf, const ngx_str_t *vomsdir,
65 const ngx_str_t *cert_dir, size_t buf_sz)
66{
67 if (!brix_voms_loaded) {
68 return NGX_DECLINED;
69 }
70
71 if (leaf == NULL || vomsdir == NULL || cert_dir == NULL
72 || vomsdir->len == 0 || cert_dir->len == 0)
73 {
74 return NGX_DECLINED;
75 }
76
77 if (vomsdir->len >= buf_sz || cert_dir->len >= buf_sz) {
78 return NGX_ERROR;
79 }
80
81 return NGX_OK;
82}

119/*
120 * WHAT: Duplicate the verified certificate chain and drop the leaf, producing
121 * the parent-only stack that the VOMS retrieve() call expects.
122 *
123 * WHY: VOMS looks up the AC extension in the parent certificates, so the leaf
124 * (index 0, matching *leaf) must be removed. Isolating this keeps the pointer
125 * juggling and its NULL/empty edge cases out of the main flow.
126 *
127 * HOW: For a non-empty chain, sk_X509_dup() the stack; on allocation failure
128 * signal via *ok = 0 (caller must clean up the API handle). If the duplicated
129 * head equals the leaf, delete it. A NULL/empty input yields a NULL chain,
130 * which is valid input to retrieve(). Sets *ok = 1 on success.
131 */
132static STACK_OF(X509) *
133brix_voms_build_parent_chain(STACK_OF(X509) *chain, X509 *leaf, int *ok)
134{
135 STACK_OF(X509) *voms_chain = NULL;
136
137 *ok = 1;
138
139 if (chain == NULL || sk_X509_num(chain) <= 0) {
140 return NULL;
141 }
142
143 voms_chain = sk_X509_dup(chain);
144 if (voms_chain == NULL) {
145 *ok = 0;
146 return NULL;
147 }
148
149 if (sk_X509_num(voms_chain) > 0
150 && X509_cmp(sk_X509_value(voms_chain, 0), leaf) == 0)
151 {
152 sk_X509_delete(voms_chain, 0);
153 }
154
155 return voms_chain;
156}

158/*
159 * WHAT: Perform the VOMS retrieve() call and, on success, collect the VO/FQAN
160 * results into the caller's buffers.
161 *
162 * WHY: Concentrates the retrieve/error/collect decision in one helper so the
163 * entry point reads as a linear pipeline. Preserves the original disposition:
164 * VOMS_VERR_NOEXT / VOMS_VERR_NODATA are silently ignored (NGX_DECLINED per
165 * initial rc); other retrieve failures WARN-log and return NGX_ERROR.
166 *
167 * HOW: Call retrieve() with VOMS_RECURSE_CHAIN. On failure, classify the error;
168 * on success, delegate to brix_collect_voms_vos(). Uses a local error buffer for
169 * the human-readable message.
170 */
171static ngx_int_t
172brix_voms_retrieve_and_collect(ngx_log_t *log, X509 *leaf,
173 STACK_OF(X509) *voms_chain, struct voms_data *vd,
174 const brix_voms_out_t *out)
175{
176 char errbuf[512];
177 int error = 0;
178
179 if (!brix_voms_api.retrieve(leaf, voms_chain, VOMS_RECURSE_CHAIN, vd,
180 &error))
181 {
182 if (error != VOMS_VERR_NOEXT && error != VOMS_VERR_NODATA) {
183 ngx_log_error(NGX_LOG_WARN, log, 0,
184 "brix: VOMS extraction failed: %s",
185 brix_voms_api.error_message(vd, error, errbuf,
186 (int) sizeof(errbuf)));
187 return NGX_ERROR;
188 }
189
190 return NGX_DECLINED;
191 }
192
193 return brix_collect_voms_vos(vd, out->primary_vo, out->primary_vo_sz,
194 out->vo_list, out->vo_list_sz);
195}
