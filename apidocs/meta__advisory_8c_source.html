22static int
23adv_append(char *out, size_t cap, int *n, const char *fmt, ...)
24{
25 va_list ap;
26 int r;
27
28 va_start(ap, fmt);
29 r = vsnprintf(out + *n, cap - (size_t) *n, fmt, ap);
30 va_end(ap);
31 if (r < 0 || (size_t) *n + (size_t) r >= cap) {
32 return -1;
33 }
34 *n += r;
35 return 0;
36}

68brix_meta_advisory_decode(const char *blob, size_t len, brix_meta_advisory_t *m)
69{
70 char tmp[512];
71 size_t n;
72 char *tok, *save = NULL;
73 int uid_seen = 0, gid_seen = 0;
74
75 if (m == NULL) {
76 return -1;
77 }
78 memset(m, 0, sizeof(*m));
79 if (blob == NULL || len == 0) {
80 return 0;
81 }
82
83 n = (len < sizeof(tmp) - 1) ? len : sizeof(tmp) - 1;
84 memcpy(tmp, blob, n);
85 tmp[n] = '\0';
86
87 for (tok = strtok_r(tmp, " ", &save); tok != NULL;
88 tok = strtok_r(NULL, " ", &save))
89 {
90 char *eq = strchr(tok, '=');
91 const char *key, *val;
92
93 if (eq == NULL) {
94 continue; /* version token ("v1"/"v9") — skip */
95 }
96 *eq = '\0';
97 key = tok;
98 val = eq + 1;
99
100 if (strcmp(key, "mode") == 0) {
101 m->mode = (mode_t) strtol(val, NULL, 8);
102 m->have_mode = 1;
103 } else if (strcmp(key, "uid") == 0) {
104 m->uid = (uid_t) strtoul(val, NULL, 10);
105 uid_seen = 1;
106 } else if (strcmp(key, "gid") == 0) {
107 m->gid = (gid_t) strtoul(val, NULL, 10);
108 gid_seen = 1;
109 } else if (strcmp(key, "mtime") == 0) {
110 m->mtime = (time_t) strtoll(val, NULL, 10);
111 m->have_mtime = 1;
112 } else if (strcmp(key, "mtime_ns") == 0) {
113 m->mtime_ns = strtol(val, NULL, 10);
114 }
115 /* unknown key → ignored (forward-compatible) */
116 }
117
118 m->have_owner = (uid_seen && gid_seen) ? 1 : 0;
119 return 0;
120}
