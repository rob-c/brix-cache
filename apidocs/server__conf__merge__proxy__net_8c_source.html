1/*
2 * server_conf_merge_proxy_net.c — server-block merge for the upstream/proxy and
3 * network-fault configuration area.
4 *
5 * WHAT: Owns brix_merge_srv_proxy_net() (upstream TLS + token, transparent proxy
6 * mode, the write-through origin + prefix rules + decision struct, OCSP,
7 * the Phase-39 network-fault deadlines, and rate-limit rule inheritance),
8 * together with the file-local per-concern helpers it delegates to.
9 * WHY: Split (phase-79 file-size cap) out of the former 1249-line
10 * server_conf.c. The entry point is non-static (declared in
11 * server_conf_internal.h) because the top-level orchestrator in
12 * server_conf.c calls it last in linear order — after the cache fields it
13 * depends on (origin host/port, cache size cap, include-regex) have
14 * settled; every sub-helper stays file-local.
15 * HOW: Standard nginx parent->child inheritance via ngx_conf_merge_* and the
16 * BRIX_MERGE_* macros. The write-through decision struct rebuilds from the
17 * just-merged prefix arrays/size cap/regex; the manager-staleness publish
18 * is a config-time side effect that runs once its input settles, exactly
19 * as before. No behaviour change from the split.
20 */

26/*
27 * WHAT: merge the upstream-TLS + transparent-proxy group (relay guard, proxy
28 * enable/port/auth/login, TLS material, rewrite rules, and the connect/
29 * read/write/keepalive timeouts).
30 * WHY: the proxy login-user name is a fixed char buffer copied by ngx_cpystrn,
31 * not a mergeable str; grouping keeps that special case local.
32 * HOW: inherit the upstream/proxy scalars child<-parent and copy the parent's
33 * login-user name only when the child left it empty.

42 ngx_conf_merge_str_value(conf->upstream_token_file,
43 prev->upstream_token_file, "");

79/*
80 * WHAT: merge the write-through group — enable/mode, origin host/port (falling
81 * back to the cache origin), the deny/allow prefix arrays, and rebuild the
82 * write-through decision struct from the merged inputs.
83 * WHY: the decision struct references the just-merged prefix arrays, cache size
84 * cap, and include-regex, so it must rebuild after they settle.
85 * HOW: merge the wt scalars + prefix arrays (NULL result + non-empty parent/
86 * child ⇒ NGX_CONF_ERROR), then point the decision fn/data/limits at them.
87 */
88static char *
89brix_merge_srv_writethrough(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *conf,
90 ngx_stream_brix_srv_conf_t *prev)
91{
92 ngx_array_t *child_wt_deny_prefixes;
93 ngx_array_t *child_wt_allow_prefixes;
94
95 ngx_conf_merge_value(conf->wt.enable, prev->wt.enable, 0);
96 BRIX_MERGE_ENUM(conf, prev, wt.mode, BRIX_WT_MODE_UNSET, BRIX_WT_MODE_SYNC);
97 BRIX_MERGE_HOSTPORT(conf, prev, wt.origin_host, wt.origin_port);
98 if (conf->wt.origin_host.len == 0 && conf->cache_origin_host.len > 0) {
99 conf->wt.origin_host = conf->cache_origin_host;
100 conf->wt.origin_port = conf->cache_origin_port;
101 }
102
103 child_wt_deny_prefixes = conf->wt.deny_prefixes;
104 conf->wt.deny_prefixes = brix_merge_arrays(cf, prev->wt.deny_prefixes,
105 child_wt_deny_prefixes,
106 sizeof(brix_wt_prefix_entry_t));
107 if (conf->wt.deny_prefixes == NULL
108 && (prev->wt.deny_prefixes != NULL || child_wt_deny_prefixes != NULL))
109 {
110 return NGX_CONF_ERROR;
111 }
112
113 child_wt_allow_prefixes = conf->wt.allow_prefixes;
114 conf->wt.allow_prefixes = brix_merge_arrays(cf, prev->wt.allow_prefixes,
115 child_wt_allow_prefixes,
116 sizeof(brix_wt_prefix_entry_t));
117 if (conf->wt.allow_prefixes == NULL
118 && (prev->wt.allow_prefixes != NULL || child_wt_allow_prefixes != NULL))
119 {
120 return NGX_CONF_ERROR;
121 }
122
123 conf->wt.decision.fn = brix_wt_default_decide;
124 conf->wt.decision.user_data = &conf->wt.decision;
125 conf->wt.decision.deny_prefixes = conf->wt.deny_prefixes;
126 conf->wt.decision.allow_prefixes = conf->wt.allow_prefixes;
127 conf->wt.decision.max_write_through_bytes = conf->cache_max_file_size;
128 conf->wt.decision.include_regex = conf->include_regex.re;
129 conf->wt.decision.include_regex_set = conf->include_regex.set;
130
131 return NGX_CONF_OK;
132}

134/*
135 * WHAT: merge OCSP, the Phase-39 network-fault deadlines, and inherit the
136 * advanced rate-limit rule array; publish the manager staleness threshold.
137 * WHY: the staleness publish is a config-time side effect (before fork) that
138 * must run once inputs settle; grouping keeps it beside its merge.
139 * HOW: delegate OCSP, inherit the deadline scalars (0 == disabled), publish the
140 * staleness threshold when > 0, and inherit rl_rules when the child is NULL.
141 */
142static void
143brix_merge_srv_net_fault(ngx_stream_brix_srv_conf_t *conf,
144 ngx_stream_brix_srv_conf_t *prev)
145{
146 brix_ocsp_conf_merge(&conf->ocsp, &prev->ocsp);
147
148 /* ocsp.staple_data/len are populated at init_process, not here */
149
150 /* Phase 39: network-fault resilience — default OFF (0) = no behaviour change.
151 * 0 means "disabled"; arm sites all guard on `> 0` (ngx_msec_t is unsigned). */
152 ngx_conf_merge_msec_value(conf->read_timeout, prev->read_timeout, 0);
153 ngx_conf_merge_msec_value(conf->handshake_timeout, prev->handshake_timeout, 0);
154 ngx_conf_merge_msec_value(conf->send_timeout, prev->send_timeout, 0);
155 ngx_conf_merge_msec_value(conf->tcp_user_timeout, prev->tcp_user_timeout, 0);
156 ngx_conf_merge_value(conf->tcp_keepalive, prev->tcp_keepalive, 0);
157 ngx_conf_merge_str_value(conf->tcp_congestion, prev->tcp_congestion, "");
158 ngx_conf_merge_uint_value(conf->max_connections, prev->max_connections, 0);
159 ngx_conf_merge_msec_value(conf->manager_stale_after,
160 prev->manager_stale_after, 0);
161 /* Phase 39 (WS7): publish the data-server staleness threshold to the shared
162 * cluster registry (config-time, before fork). Guarded so a 0-default block
163 * does not disable a sibling that enabled it. */
164 if (conf->manager_stale_after > 0) {
165 brix_srv_set_stale_after(conf->manager_stale_after);
166 }
167
168 /* Phase 39 (WS9): a child server block with no advanced rate-limit rules of
169 * its own inherits the parent's (the Phase-20 kv caches above already do;
170 * the Phase-25 rl_rules array was previously never inherited). */
171 if (conf->rl_rules == NULL) {
172 conf->rl_rules = prev->rl_rules;
173 }
174}

176/* Upstream/proxy & network: upstream TLS + token, transparent proxy mode
177 * (auth/login/timeouts/rewrite), the write-through origin + prefix rules +
178 * decision struct, OCSP, the Phase-39 network-fault deadlines, and rate-limit
179 * rule inheritance. */
180char *
181brix_merge_srv_proxy_net(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *conf,
182 ngx_stream_brix_srv_conf_t *prev)
183{
184 brix_merge_srv_proxy(conf, prev);
185 if (brix_merge_srv_writethrough(cf, conf, prev) != NGX_CONF_OK) {
186 return NGX_CONF_ERROR;
187 }
188 brix_merge_srv_net_fault(conf, prev);
189
190 return NGX_CONF_OK;
191}
