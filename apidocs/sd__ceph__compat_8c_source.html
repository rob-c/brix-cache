62int
63sd_ceph_oid_is_stripe_data(const char *oid)
64{
65 size_t n, i;
66
67 if (oid == NULL) {
68 return 0;
69 }
70 n = strlen(oid);
71 /* must end in '.' + exactly 16 lowercase-hex digits to be ANY stripe object */
72 if (n < SUFFIX_LEN || oid[n - SUFFIX_LEN] != '.') {
73 return 0;
74 }
75 for (i = n - SUFFIX_LEN + 1; i < n; i++) {
76 char c = oid[i];
77 if (!((c >= '0' && c <= '9') || (c >= 'a' && c <= 'f'))) {
78 return 0; /* not a 16-hex stripe suffix */
79 }
80 }
81 /* a striper data stripe iff it is NOT the all-zero first stripe */
82 return sd_ceph_oid_is_first_stripe(oid) ? 0 : 1;
83}
