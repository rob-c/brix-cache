6/*
7 * relay.c — transparent pass-through relay: a bidirectional buffered TCP relay
8 * from the stream xrootd listener to an upstream XRootD server, with a
9 * non-consuming tap that decodes the cleartext frames in flight. See relay.h
10 * for WHAT/WHY. Modeled on src/handoff/handoff.c — the difference is the
11 * upstream is the configured XRootD server, the relay engages before any frame
12 * is read, and each freshly-recv'd chunk is fed to a per-direction tap decoder.
13 */
14

16#define BRIX_RELAY_IDLE 75000 /* ms; drop a relay that stalls both ways */

154/*
155 * Move bytes one direction: drain the pending window [off,end) into `to`, then
156 * read more from `from` and loop. Each freshly-read chunk is fed to `dec` (the
157 * direction's tap decoder) exactly once — before it is queued for send, never on
158 * a re-send after NGX_AGAIN — so the tap observes every byte once, in order.
159 * When the guard sink flags a frame in that chunk, the chunk is NOT forwarded
160 * and the whole relay comes down (bounce = connection drop).
161 * Arms the appropriate event (and an idle timer) when either side would block;
162 * returns NGX_ERROR on EOF/error so the caller closes the whole relay.
163 */
164static ngx_int_t
165relay_pump(brix_relay_t *r, ngx_connection_t *from, ngx_connection_t *to,
166 u_char *buf, size_t *off, size_t *end, brix_tap_stream_t *dec)
167{
168 ssize_t n;
169
170 for ( ;; ) {
171 while (*off < *end) {
172 n = to->send(to, buf + *off, *end - *off);
173 if (n > 0) {
174 *off += (size_t) n;
175 continue;
176 }
177 if (n == NGX_AGAIN) {
178 if (ngx_handle_write_event(to->write, 0) != NGX_OK) {
179 return NGX_ERROR;
180 }
181 ngx_add_timer(to->write, BRIX_RELAY_IDLE);
182 return NGX_OK;
183 }
184 return NGX_ERROR; /* 0 or NGX_ERROR on the write side */
185 }
186 *off = *end = 0;
187
188 n = from->recv(from, buf, BRIX_RELAY_BUF);
189 if (n > 0) {
190 if (from == r->client) {
191 /* First client bytes: reject a peer not speaking root before
192 * it is forwarded (the tap only sees decoded kXR frames). */
193 brix_relay_guard_handshake(&r->guard, buf, (size_t) n);
194 }
195 brix_tap_stream_feed(dec, buf, (size_t) n); /* tap once */
196 if (brix_relay_guard_should_drop(&r->guard)) {
197 return NGX_ERROR; /* BOUNCE: never forward this chunk */
198 }
199 *end = (size_t) n;
200 *off = 0;
201 continue;
202 }
203 if (n == NGX_AGAIN) {
204 if (ngx_handle_read_event(from->read, 0) != NGX_OK) {
205 return NGX_ERROR;
206 }
207 ngx_add_timer(from->read, BRIX_RELAY_IDLE);
208 return NGX_OK;
209 }
210 return NGX_ERROR; /* EOF (0) or error: tear the relay down */
211 }
212}

297static void
298relay_connect_done(ngx_event_t *ev)
299{
300 ngx_connection_t *u = ev->data;
301 brix_relay_t *r = u->data;
302 int err;
303 socklen_t len = sizeof(err);
304
305 if (ev->timedout) {
306 ngx_log_error(NGX_LOG_ERR, r->client->log, NGX_ETIMEDOUT,
307 "xrootd relay: connect to %V timed out", r->peer.name);
308 relay_close(r);
309 return;
310 }
311 if (getsockopt(u->fd, SOL_SOCKET, SO_ERROR, (void *) &err, &len) == -1
312 || err != 0)
313 {
314 ngx_log_error(NGX_LOG_ERR, r->client->log, err,
315 "xrootd relay: connect to upstream failed");
316 relay_close(r);
317 return;
318 }
319 if (u->write->timer_set) {
320 ngx_del_timer(u->write);
321 }
322 relay_begin(r);
323}

327brix_relay_start(ngx_stream_session_t *s, ngx_connection_t *c, void *srv_conf)
328{
329 ngx_stream_brix_srv_conf_t *conf = srv_conf;
330 brix_ctx_t *ctx;
331 brix_relay_t *r;
332 ngx_connection_t *u;
333 ngx_int_t rc;
334
335 ctx = ngx_stream_get_module_ctx(s, ngx_stream_brix_module);
336
337 r = ngx_pcalloc(c->pool, sizeof(brix_relay_t));
338 if (r == NULL) {
339 return NGX_ERROR;
340 }
341 r->cu = ngx_pnalloc(c->pool, BRIX_RELAY_BUF);
342 r->uc = ngx_pnalloc(c->pool, BRIX_RELAY_BUF);
343 if (r->cu == NULL || r->uc == NULL) {
344 return NGX_ERROR;
345 }
346
347 r->s = s;
348 r->client = c;
349 r->metrics = ctx->metrics;
350
351 /* No pre-read prefix in transparent mode: the pump reads the client hello. */
352 r->cu_off = r->cu_end = 0;
353 r->uc_off = r->uc_end = 0;
354
355 /* Tap: a stable log copy (server error-log destination, no stale session
356 * appender), then register the audit sink + a decoder per direction. */
357 r->log = *c->log;
358 r->log.handler = NULL;
359 r->log.data = NULL;
360 r->log.action = NULL;
361 brix_tap_register_sink(&r->tap, relay_audit_sink, &r->log);
362 brix_tap_stream_init(&r->cu_dec, &r->tap, BRIX_TAP_C2U);
363 brix_tap_stream_init(&r->uc_dec, &r->tap, BRIX_TAP_U2C);
364
365 /* Bad-actor guard (brix_guard_stream): classify each tapped frame and
366 * let the pump drop the connection on a BOUNCE verdict. */
367 brix_relay_guard_init(&r->guard, conf->relay_guard_enable == 1,
368 &c->addr_text, &r->log);
369 if (r->guard.enable) {
370 brix_tap_register_sink(&r->tap, brix_relay_guard_sink, &r->guard);
371 }
372
373 /* The xrootd recv/send handlers and any deadline timer must not fire on this
374 * connection again — the relay owns it now. */
375 if (c->read->timer_set) {
376 ngx_del_timer(c->read);
377 }
378 if (c->write->timer_set) {
379 ngx_del_timer(c->write);
380 }
381 ctx->state = XRD_ST_PROXY;
382 ctx->relay = r;
383
384 r->peer.sockaddr = conf->relay_addr->sockaddr;
385 r->peer.socklen = conf->relay_addr->socklen;
386 r->peer.name = &conf->relay_name;
387 r->peer.get = relay_peer_get;
388 r->peer.free = relay_peer_free;
389 r->peer.log = c->log;
390 r->peer.log_error = NGX_ERROR_ERR;
391
392 rc = ngx_event_connect_peer(&r->peer);
393 if (rc == NGX_ERROR || rc == NGX_BUSY || rc == NGX_DECLINED) {
394 ngx_log_error(NGX_LOG_ERR, c->log, 0,
395 "xrootd relay: cannot connect to %V", &conf->relay_name);
396 return NGX_ERROR;
397 }
398
399 u = r->peer.connection;
400 r->upstream = u;
401 u->data = r;
402 u->pool = c->pool;
403 u->read->handler = relay_upstream_read;
404 u->write->handler = relay_connect_done;
405
406 if (rc == NGX_OK) {
407 /* Connected synchronously (loopback) — start relaying immediately. */
408 relay_begin(r);
409 } else {
410 /* NGX_AGAIN: connect in progress; arm a connect deadline. */
411 ngx_add_timer(u->write, BRIX_RELAY_IDLE);
412 }
413
414 return NGX_OK;
415}
