36const char *
37dashboard_state_name(const ngx_http_brix_dashboard_loc_conf_t *conf,
38 uint8_t state, int64_t idle_ms, int moving)
39{
40 if (state == BRIX_XFER_STATE_ERROR) { return "error"; }
41 if (state == BRIX_XFER_STATE_CLOSING) { return "closing"; }
42 /*
43 * A transfer idle past the stalled threshold is normally "stalled". But a
44 * transfer that has moved data overall (moving != 0) and is merely between
45 * client-imposed rate-limit bursts — xrdcp --xrate sleeps ~84 s then bursts
46 * 8 MiB at 100 kB/s — is making scheduled forward progress, not stuck. Show
47 * it as "throttled" so a paced transfer reads as a steady slow row instead
48 * of flapping into the red "stalled" state every inter-burst gap. The
49 * "idle" band (shorter pauses) is unchanged so ordinary transfers that just
50 * paused briefly are not mislabelled.
51 */
52 if (idle_ms >= (int64_t) conf->stalled_threshold_ms) {
53 return moving ? "throttled" : "stalled";
54 }
55 if (idle_ms >= (int64_t) conf->idle_threshold_ms) { return "idle"; }
56 return "active";
57}

109/*
110 * WHAT: Fold a 16-byte session id into a 32-bit value (FNV-1a hash).
111 * WHY: The dashboard exposes a short "session_hash" so an operator can
112 * correlate rows without leaking the raw session id. NOT for security —
113 * it is non-cryptographic and only needs to be stable per session.
114 * HOW: Standard FNV-1a: 2166136261 offset basis, 16777619 prime.
115 */
116uint32_t
117dashboard_session_hash(const u_char sessid[16])
118{
119 uint32_t h = 2166136261u;
120 ngx_uint_t i;
121
122 for (i = 0; i < 16; i++) {
123 h ^= sessid[i];
124 h *= 16777619u;
125 }
126 return h;
127}

130/* Average bytes/sec over a transfer's lifetime. Guards against a zero/negative
131 * elapsed window (clock not advanced, or start in the future) to avoid divide-
132 * by-zero; the *1000 converts the ms denominator to a per-second rate. */
133uint64_t
134dashboard_avg_bps(int64_t bytes, int64_t start_ms, int64_t now_ms)
135{
136 int64_t elapsed_ms;
137
138 elapsed_ms = (start_ms > 0 && now_ms > start_ms) ? now_ms - start_ms : 0;
139 return elapsed_ms > 0 ? (uint64_t) ((bytes * 1000) / elapsed_ms) : 0;
140}

144/*
145 * WHAT: Entry point for every JSON dashboard endpoint.
146 * HOW: Enforce auth FIRST (before touching SHM or building any payload), reject

180/*
181 * WHAT: Route a resolved dashboard endpoint to its JSON builder and return the
182 * built root object (or NULL on OOM/truncation from the builder). Sets
183 * *status for the two endpoints that override the default 200: the
184 * transfer-detail builder writes its own status through the pointer, and
185 * the not-found endpoint forces 404. All other endpoints leave *status
186 * untouched (the caller pre-seeds it to 200).
187 * WHY: Isolating the per-endpoint dispatch keeps the api_handler orchestrator
188 * flat and under the complexity cap while preserving the exact builder
189 * call, argument set, and status semantics for every endpoint.
190 * HOW: 1. Switch on the endpoint enum. 2. Invoke the matching builder with the
191 * same arguments each formerly received (some take r, some take totals,
192 * transfer-detail takes &status, cache/ratelimit/cvmfs omit r/totals).
193 * 3. For the not-found/default arm, set *status to 404 before building.
194 * 4. Return the builder result unchanged.
195 */
196static json_t *
197dashboard_dispatch_builder(ngx_http_request_t *r,
198 brix_dashboard_api_endpoint_e endpoint, int64_t now_ms,
199 ngx_http_brix_dashboard_loc_conf_t *conf,
200 brix_dashboard_totals_t *totals, ngx_uint_t redact, ngx_int_t *status)
201{
202 switch (endpoint) {
203 case BRIX_DASHBOARD_API_COMPAT_TRANSFERS:
204 return dashboard_build_compat_transfers(now_ms, conf, totals, redact);
205 case BRIX_DASHBOARD_API_V1_TRANSFERS:
206 return dashboard_build_v1_transfers(r, now_ms, conf, totals, redact);
207 case BRIX_DASHBOARD_API_V1_TRANSFER_DETAIL:
208 return dashboard_build_v1_transfer_detail(r, now_ms, conf, status);
209 case BRIX_DASHBOARD_API_V1_SNAPSHOT:
210 return dashboard_build_v1_snapshot(r, now_ms, conf, totals, redact);
211 case BRIX_DASHBOARD_API_V1_EVENTS:
212 return dashboard_build_v1_events(r, now_ms, conf, redact);
213 case BRIX_DASHBOARD_API_V1_HISTORY:
214 return dashboard_build_v1_history(r, now_ms, conf, redact);
215 case BRIX_DASHBOARD_API_V1_CLUSTER:
216 return dashboard_build_v1_cluster(r, now_ms, conf, redact);
217 case BRIX_DASHBOARD_API_V1_CACHE:
218 return dashboard_build_v1_cache(now_ms, conf, redact);
219 case BRIX_DASHBOARD_API_V1_RATELIMIT:
220 return dashboard_build_v1_ratelimit(now_ms, conf, redact);
221 case BRIX_DASHBOARD_API_V1_CVMFS:
222 return dashboard_build_v1_cvmfs(now_ms, conf, redact);
223 case BRIX_DASHBOARD_API_V1_NOT_FOUND:
224 default:
225 *status = NGX_HTTP_NOT_FOUND;
226 return dashboard_build_v1_not_found(now_ms, conf, redact);
227 }

231ngx_int_t
232ngx_http_brix_dashboard_api_handler(ngx_http_request_t *r,
233 brix_dashboard_api_endpoint_e endpoint)
234{
235 ngx_http_brix_dashboard_loc_conf_t *conf;
236 brix_dashboard_totals_t totals;
237 json_t *root = NULL;
238 int64_t now_ms;
239 ngx_int_t status = NGX_HTTP_OK;
240 ngx_uint_t redact = 0;
241
242 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
243
244 /* Three-way auth gate. Authenticated -> full payload. Unauthenticated but
245 * brix_dashboard_anonymous is on AND this is an anon-allowed read endpoint
246 * -> serve a PII/secret-redacted payload (redact=1). Otherwise -> 401.
247 * (check_auth suppresses its "missing cookie" event when anonymous is on,
248 * so anonymous polls do not spam the event ring.) */
249 {
250 ngx_int_t auth_rc = ngx_http_brix_dashboard_check_auth(r, conf,
251 conf->anonymous);
252 if (auth_rc == NGX_OK) {
253 redact = 0;
254 } else if (conf->anonymous
255 && dashboard_endpoint_is_anon_allowed(endpoint)) {
256 redact = 1;
257 } else {
258 return auth_rc;
259 }
260 }
261
262 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
263 return NGX_HTTP_NOT_ALLOWED;
264 }
265
266 now_ms = (int64_t) ngx_current_msec;
267 brix_dashboard_history_sample(now_ms);
268 dashboard_collect_totals(&totals);
269
270 root = dashboard_dispatch_builder(r, endpoint, now_ms, conf, &totals,
271 redact, &status);
272
273 if (root == NULL) {
274 brix_dashboard_event_add(BRIX_DASH_EVENT_DASHBOARD, 0, 507,
275 "dashboard JSON response truncated", NULL);
276 status = NGX_HTTP_INSUFFICIENT_STORAGE;
277 root = dashboard_build_v1_truncated(now_ms, conf);
278 }
279
280 return dashboard_json_send(r, status, root);
281}
