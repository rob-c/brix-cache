1/*
2 * progname.h — runtime program-name identity for the client tools.
3 *
4 * WHAT: helpers that let one name-agnostic binary discover, at run time from
5 * argv[0], (a) its own display name, (b) whether it was invoked under the
6 * co-install "brix-" prefix, and (c) the name of a sibling tool to exec.
7 * Plus brix_usage_footer(), the runtime replacement for the old
8 * BRIX_USAGE_FOOTER() string-literal macro.
9 * WHY: the brix-cache-client-compat package installs the exact same binaries as
10 * brix-cache-client but under a "brix-" prefix (brix-xrdcp, brix-xrdfs …)
11 * so they co-install with the official xrootd tools. A prefixed tool must
12 * print its own name — never the stock one — in usage/version/footer, and
13 * must dispatch multi-call personalities and exec sibling tools under the
14 * matching name. Deriving everything from argv[0] keeps ONE binary (git /
15 * busybox model) with no second compile and no drift.
16 * HOW: header-only inline helpers (no new translation unit, so the build
17 * source-list is untouched). basename is computed with strrchr rather
18 * than libgen basename(3) to avoid mutating the caller's string.
19 */
20#ifndef BRIX_CORE_PROGNAME_H
21#define BRIX_CORE_PROGNAME_H
22
23#include <stdio.h>
24#include <string.h>
25
26/* The co-install prefix carried by the brix-cache-client-compat tools. */
27#define BRIX_COMPAT_PREFIX "brix-"
28#define BRIX_COMPAT_PREFIX_LEN 5
29
30/*
31 * brix_prog_base — basename of an argv[0]/path (pointer into the same string).
32 * WHY: argv[0] may be a bare name ("brix-xrdcp") or a path ("./bin/brix-xrdcp");
33 * the display/identity logic wants just the final component.
34 */
35static inline const char *
36brix_prog_base(const char *argv0)
37{
38 const char *slash;
39
40 if (argv0 == NULL || *argv0 == '\0') {
41 return "?";
42 }
43 slash = strrchr(argv0, '/');
44 return slash != NULL ? slash + 1 : argv0;
45}
46
47/*
48 * brix_prog_strip_compat — the basename with any leading "brix-" removed.
49 * WHY: multi-call dispatchers (xrdcksum, xrddiag) pick a personality from
50 * basename(argv[0]); invoked as "brix-xrdadler32" they must still match the
51 * "xrdadler32" personality, so strip the co-install prefix before matching.
52 */
53static inline const char *
54brix_prog_strip_compat(const char *base)
55{
56 if (strncmp(base, BRIX_COMPAT_PREFIX, BRIX_COMPAT_PREFIX_LEN) == 0) {
57 return base + BRIX_COMPAT_PREFIX_LEN;
58 }
59 return base;
60}
61
62/*
63 * brix_prog_prefix — "brix-" when the basename carries the co-install prefix,
64 * else "". WHY: an umbrella tool (xrd) must exec siblings under the same
65 * flavour it was invoked as — "brix-xrd cp" -> "brix-xrdcp".
66 */
67static inline const char *
68brix_prog_prefix(const char *base)
69{
70 return strncmp(base, BRIX_COMPAT_PREFIX, BRIX_COMPAT_PREFIX_LEN) == 0
71 ? BRIX_COMPAT_PREFIX
72 : "";
73}
74
75/*
76 * brix_usage_footer — the two-line footer appended to every usage block.
77 * WHY: replaces the old BRIX_USAGE_FOOTER(TOOL) string-literal macro so the
78 * "man <tool>" line names the tool AS INVOKED (brix-xrdcp under the compat
79 * package, xrdcp otherwise) rather than a baked-in stock literal.
80 * HOW: prints basename(argv0); the middle dot (U+00B7) is the same UTF-8 byte
81 * sequence the old macro emitted.
82 */
83static inline void
84brix_usage_footer(FILE *out, const char *argv0)
85{
86 fprintf(out,
87 "config: ~/.xrdrc defines endpoint aliases and credentials "
88 "(see brix-env(7))\n"
89 "docs: man %s \xc2\xb7 exit codes are listed at the end of "
90 "the man page\n",
91 brix_prog_base(argv0));
92}
