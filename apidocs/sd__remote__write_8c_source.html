2 * sd_remote_write.c — write path (SP3) for the remote-origin (s3://) storage
3 * driver: staged whole-object uploads (.staged_* → single PUT or multipart
4 * upload) plus .unlink (DELETE). Split out of sd_remote.c verbatim; the driver
5 * table lives there and references these via sd_remote_internal.h. Shared
6 * path/param/cred helpers stay in sd_remote.c.

8 * A staged write delegates to sd_s3's single-PUT/multipart upload; the object

20/* Per-staged-write state: the delegated S3 write handle, plus the composed
21 * object path so a noreplace commit can HEAD the destination (P80.2). When the
22 * upload was opened under a per-user credential (P80.3) the triple is copied
23 * here — the caller's cred store does not outlive the open call, and the
24 * noreplace HEAD must present the same identity as the upload itself. */
25typedef struct {
26 sd_s3_file *s3;
27 char objpath[768];
28 int has_cred;
29 char ak[128];
30 char sk[256];
31 char region[64];
32} sd_remote_staged_state;
33
34/* Multipart part size for a staged upload of unknown final size (S3's 5 MiB
35 * minimum for non-final parts; 16 MiB balances request count vs. buffering). */
36#define SD_REMOTE_PART_SIZE (16 * 1024 * 1024)
37
38/* Shared staged-open body: start the upload, optionally signing with a
39 * per-user ak/sk/region override (NULL = the static service credential). The
40 * override triple is copied into the staged state so the noreplace commit's
41 * HEAD (P80.2) presents the same identity as the upload (P80.3). */
42static brix_sd_staged_t *
43sd_remote_staged_open_impl(brix_sd_instance_t *inst, const char *final_path,
44 const char *ak, const char *sk, const char *region, int *err_out)
45{
46 const brix_sd_remote_cfg_t *cfg = inst->state;
47 sd_s3_open_params p;
48 char objpath[768];
49 char errbuf[256];
50 sd_s3_file *s3;
51 sd_remote_staged_state *ss;
52 brix_sd_staged_t *h;
53
54 sd_remote_s3_key(cfg, final_path, objpath, sizeof(objpath));
55 sd_remote_s3_params(cfg, objpath, &p);
56 if (ak != NULL) { p.ak = ak; }
57 if (sk != NULL) { p.sk = sk; }
58 if (region != NULL) { p.region = region; }
59
60 /* Unknown final size: sd_s3 buffers a single PUT and lazily upgrades to a
61 * multipart upload only past SD_REMOTE_PART_SIZE (P80.2), so small objects
62 * cost one request while any size still works. */
63 s3 = sd_s3_open_write(&p, -1, SD_REMOTE_PART_SIZE, errbuf, sizeof(errbuf));
64 if (s3 == NULL) {
65 if (err_out) { *err_out = EIO; }
66 return NULL;
67 }
68 ss = calloc(1, sizeof(*ss));
69 h = calloc(1, sizeof(*h));
70 if (ss == NULL || h == NULL) {
71 free(ss);
72 free(h);
73 sd_s3_abort(s3);
74 sd_s3_close(s3);
75 if (err_out) { *err_out = ENOMEM; }
76 return NULL;
77 }
78 ss->s3 = s3;
79 snprintf(ss->objpath, sizeof(ss->objpath), "%s", objpath);
80 if (ak != NULL && sk != NULL) {
81 ss->has_cred = 1;
82 snprintf(ss->ak, sizeof(ss->ak), "%s", ak);
83 snprintf(ss->sk, sizeof(ss->sk), "%s", sk);
84 snprintf(ss->region, sizeof(ss->region), "%s",
85 (region != NULL) ? region : "");
86 }
87 h->inst = inst;
88 h->state = ss;
89 return h;
90}

101/* Cred-scoped staged open (P80.3): a write whose identity resolved to a
102 * `<key>.s3` credential uploads to the origin as THAT user — every leg of the
103 * upload (CreateMPU/UploadPart/PUT/Complete, and the noreplace HEAD via the
104 * copied triple) signs with the per-user keys. Gate semantics identical to
105 * sd_remote_open_cred. */
106brix_sd_staged_t *
107sd_remote_staged_open_cred(brix_sd_instance_t *inst, const char *final_path,
108 mode_t mode, const brix_sd_cred_t *cred, int *err_out)
109{
110 int gate = sd_remote_cred_gate(cred);
111
112 (void) mode;
113
114 if (gate > 0) {
115 return sd_remote_staged_open_impl(inst, final_path,
116 cred->s3_ak, cred->s3_sk, cred->s3_region, err_out);
117 }
118 if (gate < 0) {
119 if (err_out) { *err_out = EACCES; }
120 errno = EACCES;
121 return NULL;
122 }
123 return sd_remote_staged_open_impl(inst, final_path, NULL, NULL, NULL,
124 err_out);
125}

141ngx_int_t
142sd_remote_staged_commit(brix_sd_staged_t *h, int noreplace)
143{
144 sd_remote_staged_state *ss = h->state;
145 char errbuf[256];
146 int rc;
147
148 /* Exclusive publish (P80.2): S3 PUT/MPU-complete always replaces, so
149 * noreplace is a HEAD-before-publish existence check. This is check-then-
150 * act — RACY against a concurrent external writer landing the object
151 * between the HEAD and the PUT — but honest O_EXCL/POSC semantics for
152 * everything going through this gateway, versus silently overwriting. */
153 if (noreplace) {
154 const brix_sd_remote_cfg_t *cfg = h->inst->state;
155 sd_s3_open_params p;
156 sd_s3_file *probe;
157 int64_t size = 0;
158
159 sd_remote_s3_params(cfg, ss->objpath, &p);
160 if (ss->has_cred) {
161 /* P80.3: the existence probe must present the same identity as
162 * the upload it gates — never the shared service credential. */
163 p.ak = ss->ak;
164 p.sk = ss->sk;
165 if (ss->region[0] != '\0') { p.region = ss->region; }
166 }
167 probe = sd_s3_open_read(&p, errbuf, sizeof(errbuf));
168 if (probe != NULL) {
169 int exists = sd_s3_size(probe, &size, errbuf, sizeof(errbuf)) == 0;
170
171 sd_s3_close(probe);
172 if (exists) {
173 /* Failure contract: leave the staged handle intact — the
174 * caller's staged_abort discards the upload and frees it. */
175 errno = EEXIST;
176 return NGX_ERROR;
177 }
178 }
179 }
180
181 rc = sd_s3_commit(ss->s3, errbuf, sizeof(errbuf));
182 sd_s3_close(ss->s3);
183 free(ss);
184 free(h);
185 if (rc != 0) {
186 errno = EIO;
187 return NGX_ERROR;
188 }
189 return NGX_OK;
190}

203/* Shared unlink body: DELETE the object, optionally signing with a per-user
204 * ak/sk/region override (NULL = the instance's static service credential). */
205static ngx_int_t
206sd_remote_unlink_impl(brix_sd_instance_t *inst, const char *path,
207 const char *ak, const char *sk, const char *region)
208{
209 const brix_sd_remote_cfg_t *cfg = inst->state;
210 sd_s3_open_params p;
211 char objpath[768];
212 char errbuf[256];
213
214 sd_remote_s3_key(cfg, path, objpath, sizeof(objpath));
215 sd_remote_s3_params(cfg, objpath, &p);
216 if (ak != NULL) { p.ak = ak; }
217 if (sk != NULL) { p.sk = sk; }
218 if (region != NULL) { p.region = region; }
219
220 if (sd_s3_delete(&p, errbuf, sizeof(errbuf)) != 0) {
221 if (errno == 0) { errno = EIO; }
222 return NGX_ERROR;
223 }
224 return NGX_OK;
225}
