1/*
2 * backend_async_queue.c — durable write-behind queue for backend namespace
3 * mutations (brix_backend_async). See backend_async_queue.h for the contract.
4 *
5 * WHAT: Per-worker coalescing + durable journalling of unlink/rmdir/rename/mkdir,
6 * with in-bulk drain against the backend and a protocol-supplied waker.
7 *
8 * WHY: Trade per-op latency for bulk backend efficiency without ever losing an
9 * acknowledged mutation across a crash. Split from the stage engine on
10 * purpose: a namespace mutation is not a byte transfer.
11 *
12 * HOW: A fixed-size pending array (per worker), each row carrying the durable
13 * record + the adapter's completion callback + coalescing thresholds. The
14 * size trigger fires inline in enqueue; the time trigger fires on the worker
15 * timer tick. Durability reuses the stage engine's journal DIRECTORY root
16 * (stage_journal_dir) but a private "backend/" subdir and a private record
17 * type + reconcile, so the two subsystems never scan each other's records.
18 */
19

36/* The worker's pending batch. Bounded: at capacity, enqueue drains first so a
37 * burst never grows unboundedly (and never silently drops). 4096 rows is far
38 * above any sane coalescing batch while staying a small fixed footprint. */
39#define BRIX_BAQ_MAX_PENDING 4096
40
41typedef struct {
42 brix_baq_rec_t rec; /* the durable record (also what we journal) */
43 brix_baq_done_pt done; /* adapter waker (NULL = replayed, no client) */
44 void *client; /* opaque conn/request token for `done` */
45 ngx_uint_t batch; /* this op's size trigger */
46 ngx_msec_t wait_ms; /* this op's time trigger */
47 ngx_msec_t enqueued_ms; /* monotonic enqueue stamp (time trigger) */
48} brix_baq_pending_t;

87/* fsync a QUEUED record to <baq_journal_dir>/<reqid>.req. Best-effort: a failure
88 * costs durability for that one op, never correctness of the in-memory drain. */
89static void
90baq_journal_write(const brix_baq_rec_t *rec)
91{
92 char path[1300];
93 int fd;
94
95 if (baq_journal_dir[0] == '\0') {
96 return;
97 }
98 if ((size_t) snprintf(path, sizeof(path), "%s/%s.req",
99 baq_journal_dir, rec->reqid) >= sizeof(path))
100 {
101 return;
102 }
103 fd = open(path, O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC, 0600);
104 if (fd < 0) {
105 return;
106 }
107 if (write(fd, rec, sizeof(*rec)) == (ssize_t) sizeof(*rec)) {
108 (void) fsync(fd);
109 }
110 (void) close(fd);
111}

130/*
131 * Drive ONE record against its backend. Returns the raw POSIX errno (0 == ok).
132 *
133 * WHAT: Dispatches on rec->op to the matching pool-less VFS path primitive, all
134 * of which confine to root_canon and are safe to call inline on the event
135 * loop (no pool alloc, no metric).
136 *
137 * WHY: Shared by the live drain (which delivers the raw errno to the client as
138 * the true result) and reconcile (which additionally squashes idempotent-
139 * benign errnos — see baq_apply_idempotent).
140 *
141 * HOW: RENAME needs the resolved backend instance (brix_vfs_rename_path takes an
142 * sd); the others take root_canon directly.
143 */
144static int
145baq_apply(const brix_baq_rec_t *rec, ngx_log_t *log)
146{
147 switch ((brix_baq_op_t) rec->op) {
148
149 case BRIX_BAQ_UNLINK:
150 if (brix_vfs_unlink_path(log, rec->root_canon, rec->src_key) == 0) {
151 return 0;
152 }
153 /* kXR_rm parity: "unlink a file, rmdir a directory" non-recursively. The
154 * file-unlink primitive returns EISDIR on a directory target, so fall back
155 * to the empty-dir removal (which itself returns ENOTEMPTY on a non-empty
156 * dir) — matching brix_vfs_unlink's file-or-empty-dir behaviour exactly. */
157 if (errno == EISDIR
158 && brix_vfs_rmdir_path(log, rec->root_canon, rec->src_key) == 0)
159 {
160 return 0;
161 }
162 return errno ? errno : EIO;
163
164 case BRIX_BAQ_RMDIR:
165 return brix_vfs_rmdir_path(log, rec->root_canon, rec->src_key) == 0
166 ? 0 : (errno ? errno : EIO);
167
168 case BRIX_BAQ_MKDIR:
169 return brix_vfs_mkdir_path(log, rec->root_canon, rec->src_key,
170 (mode_t) rec->mode) == 0
171 ? 0 : (errno ? errno : EIO);
172
173 case BRIX_BAQ_RENAME: {
174 brix_sd_instance_t *sd = brix_vfs_backend_resolve(rec->root_canon, log);
175
176 errno = 0;
177 if (brix_vfs_rename_path(sd, log, rec->root_canon,
178 rec->src_key, rec->dst_key,
179 0 /* no overwrite */, NULL) == NGX_OK)
180 {
181 return 0;
182 }
183 return errno ? errno : EIO;
184 }
185
186 default:
187 return EINVAL; /* unknown op = corrupt record */
188 }
189}

191/* Reconcile-time idempotency: a mutation the client was told to wait for has
192 * already taken effect if the target is now in the desired end-state. Squash the
193 * benign "already done" errnos to success so the replay removes the record; any
194 * other errno is a real transient failure and the record is kept for a later try. */
195static int
196baq_apply_idempotent(const brix_baq_rec_t *rec, ngx_log_t *log)
197{
198 int e = baq_apply(rec, log);
199
200 switch ((brix_baq_op_t) rec->op) {
201 case BRIX_BAQ_UNLINK:
202 case BRIX_BAQ_RMDIR:
203 return (e == ENOENT) ? 0 : e; /* already gone = done */
204 case BRIX_BAQ_MKDIR:
205 return (e == EEXIST) ? 0 : e; /* already present = done */
206 case BRIX_BAQ_RENAME:
207 return (e == ENOENT) ? 0 : e; /* source already moved = done */
208 default:
209 return e;
210 }
211}

215/* Drain the WHOLE batch: apply each op, deliver its real errno to its client, drop
216 * its durable record. Clears the pending list. Live path (has clients to wake). */
217static void
218baq_drain_all(void)
219{
220 ngx_log_t *log = (ngx_cycle != NULL) ? ngx_cycle->log : NULL;
221 ngx_uint_t i;
222
223 for (i = 0; i < baq_count; i++) {
224 brix_baq_pending_t *p = &baq_pending[i];
225 int e = baq_apply(&p->rec, log);
226
227 baq_journal_remove(p->rec.reqid);
228 if (p->done != NULL) {
229 p->done(p->client, e); /* wake the parked client (real result) */
230 }
231 }
232 baq_count = 0;
233}

235/*
236 * Deferred-drain event: the size trigger posts this instead of draining inline.
237 *
238 * WHAT: An nginx posted event whose handler runs baq_drain_all() after the current
239 * stack unwinds (processed from ngx_posted_events by the event loop).
240 *
241 * WHY: A batch of 1 (or any op that completes the batch) would otherwise drain
242 * INSIDE brix_baq_enqueue, firing the done() callback for the very client the
243 * caller is still in the middle of parking — the reply would race the caller's
244 * own state transition. Deferring guarantees every parked connection has fully
245 * reached its wait state before its waker fires.
246 *
247 * HOW: ngx_post_event is idempotent via ev->posted, so multiple enqueues in one
248 * loop iteration coalesce into a single drain.
249 */
250static void
251baq_drain_event_handler(ngx_event_t *ev)
252{
253 (void) ev;
254 baq_drain_all();
255}
256
257static void
258baq_post_drain(void)
259{
260 if (baq_drain_ev.handler == NULL) {
261 baq_drain_ev.handler = baq_drain_event_handler;
262 baq_drain_ev.data = NULL;
263 baq_drain_ev.log = (ngx_cycle != NULL) ? ngx_cycle->log : NULL;
264 }
265 ngx_post_event((&baq_drain_ev), (&ngx_posted_events));
266}

268/* Would the batch flush on the SIZE trigger? True once the queue depth reaches the
269 * smallest per-op batch threshold currently parked (a shared worker may hold ops
270 * from servers configured with different batch sizes). */
271static ngx_uint_t
272baq_size_triggered(void)
273{
274 ngx_uint_t i, min_batch = 0;
275
276 if (baq_count == 0) {
277 return 0;
278 }
279 for (i = 0; i < baq_count; i++) {
280 if (min_batch == 0 || baq_pending[i].batch < min_batch) {
281 min_batch = baq_pending[i].batch;
282 }
283 }
284 return (min_batch != 0 && baq_count >= min_batch);
285}
286
287/* Would the batch flush on the TIME trigger? True once any parked op has waited at
288 * least its own wait_ms (ngx_current_msec is the worker's monotonic clock). */
289static ngx_uint_t
290baq_time_triggered(void)
291{
292 ngx_uint_t i;
293
294 for (i = 0; i < baq_count; i++) {
295 if (baq_pending[i].wait_ms == 0) {
296 return 1; /* 0ms wait = flush at the next tick */
297 }
298 if (ngx_current_msec - baq_pending[i].enqueued_ms
299 >= baq_pending[i].wait_ms)
300 {
301 return 1;
302 }
303 }
304 return 0;
305}

309ngx_int_t
310brix_baq_enqueue(brix_baq_op_t op, const char *root_canon,
311 const char *src_key, const char *dst_key, uint32_t mode,
312 ngx_uint_t batch, ngx_msec_t wait_ms,
313 brix_baq_done_pt done, void *client)
314{
315 brix_baq_pending_t *p;
316
317 if (root_canon == NULL || src_key == NULL || done == NULL) {
318 return NGX_ERROR;
319 }
320
321 /* Reject any path that would not fit the fixed record fields rather than
322 * silently truncating it — a truncated path at drain would target the WRONG
323 * object. The caller falls back to running the op inline. */
324 if (strlen(root_canon) >= sizeof(baq_pending[0].rec.root_canon)
325 || strlen(src_key) >= sizeof(baq_pending[0].rec.src_key)
326 || (dst_key != NULL
327 && strlen(dst_key) >= sizeof(baq_pending[0].rec.dst_key)))
328 {
329 return NGX_ERROR;
330 }
331
332 /* At capacity: drain now to make room. If a full-queue drain still leaves us
333 * at capacity (should not happen — drain empties it), fail so the caller runs
334 * the op inline rather than overwrite a parked row. */
335 if (baq_count >= BRIX_BAQ_MAX_PENDING) {
336 baq_drain_all();
337 }
338 if (baq_count >= BRIX_BAQ_MAX_PENDING) {
339 return NGX_ERROR;
340 }
341
342 p = &baq_pending[baq_count];
343 ngx_memzero(&p->rec, sizeof(p->rec));
344 baq_reqid_mint(p->rec.reqid);
345 p->rec.op = (uint32_t) op;
346 snprintf(p->rec.root_canon, sizeof(p->rec.root_canon), "%s", root_canon);
347 snprintf(p->rec.src_key, sizeof(p->rec.src_key), "%s", src_key);
348 if (dst_key != NULL) {
349 snprintf(p->rec.dst_key, sizeof(p->rec.dst_key), "%s", dst_key);
350 }
351 p->rec.mode = mode;
352 p->rec.enqueued_at = (int64_t) time(NULL);
353 p->done = done;
354 p->client = client;
355 p->batch = (batch == 0) ? 1 : batch;
356 p->wait_ms = wait_ms;
357 p->enqueued_ms = ngx_current_msec;
358
359 baq_journal_write(&p->rec); /* durable BEFORE the client parks */
360 baq_count++;
361
362 /* Size trigger: post a deferred drain rather than draining inline, so this op's
363 * waker never fires before the caller has finished parking its client (a batch
364 * of 1 still behaves like a synchronous mutation with a durable record, just one
365 * event-loop hop later). */
366 if (baq_size_triggered()) {
367 baq_post_drain();
368 }
369 return NGX_OK;
370}

380void
381brix_baq_drop_client(void *client)
382{
383 ngx_uint_t i = 0;
384
385 /* Compact the array, dropping rows for `client`. Their durable records are
386 * intentionally LEFT on disk so an un-drained mutation is replayed at boot. */
387 while (i < baq_count) {
388 if (baq_pending[i].client == client) {
389 baq_pending[i] = baq_pending[baq_count - 1];
390 baq_count--; /* re-test the swapped-in row */
391 continue;
392 }
393 i++;
394 }
395}

399/* Re-apply ONE persisted record idempotently; remove it on success. Returns
400 * 1 replayed / 0 kept (transient failure) / -1 dropped (corrupt). */
401static int
402baq_reconcile_one(const char *path, ngx_log_t *log)
403{
404 brix_baq_rec_t rec;
405 char rbuf[sizeof(brix_baq_rec_t)];
406 int fd, e;
407 ssize_t n;
408
409 fd = open(path, O_RDONLY | O_CLOEXEC);
410 if (fd < 0) {
411 return 0;
412 }
413 n = read(fd, rbuf, sizeof(rbuf));
414 (void) close(fd);
415
416 if (n != (ssize_t) sizeof(rec)) {
417 (void) unlink(path); /* short/corrupt/oversized — drop */
418 return -1;
419 }
420 ngx_memcpy(&rec, rbuf, sizeof(rec));
421 if (rec.op > BRIX_BAQ_MKDIR) {
422 (void) unlink(path); /* unknown op — drop */
423 return -1;
424 }
425
426 e = baq_apply_idempotent(&rec, log);
427 if (e == 0) {
428 (void) unlink(path);
429 return 1;
430 }
431 ngx_log_error(NGX_LOG_WARN, log, e,
432 "xrootd backend-async: reconcile of \"%s\" (op %uD root \"%s\") kept "
433 "for retry", rec.src_key, rec.op, rec.root_canon);
434 return 0;
435}

437void
438brix_baq_reconcile(void)
439{
440 char names[1024][256];
441 DIR *d;
442 struct dirent *de;
443 ngx_uint_t ncount = 0, i, replayed = 0, kept = 0, dropped = 0;
444 ngx_log_t *log;
445
446 if (baq_journal_dir[0] == '\0' || ngx_cycle == NULL) {
447 return;
448 }
449 log = ngx_cycle->log;
450
451 /* Snapshot the *.req names first (we unlink while driving). */
452 d = opendir(baq_journal_dir);
453 if (d == NULL) {
454 return;
455 }
456 while ((de = readdir(d)) != NULL && ncount < 1024) {
457 size_t nlen = strlen(de->d_name);
458
459 if (nlen > 4 && nlen < 256
460 && strcmp(de->d_name + nlen - 4, ".req") == 0)
461 {
462 memcpy(names[ncount], de->d_name, nlen + 1);
463 ncount++;
464 }
465 }
466 closedir(d);
467
468 for (i = 0; i < ncount; i++) {
469 char path[1400];
470 int r;
471
472 if ((size_t) snprintf(path, sizeof(path), "%s/%s",
473 baq_journal_dir, names[i]) >= sizeof(path))
474 {
475 continue;
476 }
477 r = baq_reconcile_one(path, log);
478 if (r > 0) { replayed++; }
479 else if (r < 0) { dropped++; }
480 else { kept++; }
481 }
482
483 if (replayed > 0 || kept > 0 || dropped > 0) {
484 ngx_log_error(NGX_LOG_NOTICE, log, 0,
485 "xrootd backend-async: restart reconcile — %ui mutation(s) replayed, "
486 "%ui kept for retry, %ui dropped", replayed, kept, dropped);
487 }
488}
