13/*
14 * dashboard/module.c - nginx HTTP module definition for the live transfer monitor.
15 *
16 * WHAT: Defines the ngx_http_brix_dashboard_module nginx HTTP module.
17 * Provides two location-level directives:
18 * brix_dashboard on|off - enable the dashboard at this location
19 * brix_dashboard_password "secret" - set the admin password
20 * When enabled, the content handler is installed and dispatches all
21 * requests under the location to the appropriate handler.
22 *
23 * ROUTING:
24 * /xrootd/ -> page_handler (HTML dashboard, after auth)
25 * /xrootd/login -> auth_login_handler (GET form + POST verify)
26 * /xrootd/transfers -> compatibility JSON transfer snapshot
27 * /xrootd/api/v1/<endpoint> -> versioned JSON API
28 *
29 * SECURITY NOTE:
30 * The dashboard exposes client identities, file paths, and IP addresses.
31 * Operators MUST restrict access to trusted networks via nginx allow/deny
32 * rules or firewall - the module provides no IP-level access control.
33 * The password is stored in nginx.conf in plaintext; use file permissions
34 * to protect the config file.
35 */

70/*
71 * One confinement-root canonicalization request: the merged `root` value, this
72 * level's `canon` buffer to fill, the parent's `prev_canon` to inherit from, the
73 * directive name for the "not accessible" log line, and the config-error string
74 * to surface when the path is too long. `canon`/`prev_canon` are char[PATH_MAX].
75 */
76typedef struct {
77 ngx_str_t *root;
78 char *canon;
79 const char *prev_canon;
80 const char *directive;
81 char *too_long_msg;
82} dashboard_canon_root_t;
83
84/*
85 * WHAT: Resolve a confinement root directive's realpath into req->canon, or
86 * inherit the parent's already-resolved canon when this level left it unset.
87 * WHY: browse_root and scan_root share identical canonicalize-once-or-inherit
88 * logic (a configured-but-bad path fails config loudly, like an export
89 * root); one helper removes the duplicate block and its branch count.
90 * HOW: If canon is empty and root is set, copy root to a NUL-terminated buffer
91 * and realpath() it into canon; on failure log with directive + root and
92 * return NGX_CONF_ERROR (via *err). If canon is empty but the parent's is
93 * set, inherit it. Returns NGX_OK on success; NGX_ERROR with *err set to
94 * the config-error string on failure (too_long_msg for an over-length
95 * path). Empty root => feature off.
96 */
97static ngx_int_t
98dashboard_merge_canon_root(ngx_conf_t *cf, const dashboard_canon_root_t *req,
99 char **err)
100{
101 if (req->canon[0] == '\0' && req->root->len > 0) {
102 char tmp[PATH_MAX];
103
104 if (brix_str_cbuf(tmp, sizeof(tmp), req->root) == NULL) {
105 *err = req->too_long_msg;
106 return NGX_ERROR;
107 }
108 if (realpath(tmp, req->canon) == NULL) {
109 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
110 "%s \"%V\" is not accessible", req->directive, req->root);
111 *err = NGX_CONF_ERROR;
112 return NGX_ERROR;
113 }
114 return NGX_OK;
115 }
116 if (req->canon[0] == '\0' && req->prev_canon[0] != '\0') {
117 ngx_memcpy(req->canon, req->prev_canon, PATH_MAX);
118 }
119 return NGX_OK;
120}

122static char *
123ngx_http_brix_dashboard_merge_loc_conf(ngx_conf_t *cf,
124 void *parent, void *child)
125{
126 ngx_http_brix_dashboard_loc_conf_t *prev = parent;
127 ngx_http_brix_dashboard_loc_conf_t *conf = child;
128 char *err = NULL;
129
130 ngx_conf_merge_value(conf->enable, prev->enable, 0);
131 ngx_conf_merge_value(conf->anonymous, prev->anonymous, 0);
132 ngx_conf_merge_uint_value(conf->session_ttl, prev->session_ttl, 28800);
133 ngx_conf_merge_msec_value(conf->idle_threshold_ms,
134 prev->idle_threshold_ms, 5000);
135 ngx_conf_merge_msec_value(conf->stalled_threshold_ms,
136 prev->stalled_threshold_ms, 60000);
137 ngx_conf_merge_msec_value(conf->cluster_stale_after_ms,
138 prev->cluster_stale_after_ms, 90000);
139 ngx_conf_merge_str_value(conf->password, prev->password, "");
140 ngx_conf_merge_str_value(conf->cookie_path, prev->cookie_path, "/brix");
141 if (conf->users == NULL) {
142 conf->users = prev->users;
143 }
144 /* Phase 23 admin API: inherit allow/secret; require_both defaults off. */
145 if (conf->admin_allow == NULL) {
146 conf->admin_allow = prev->admin_allow;
147 }
148 ngx_conf_merge_str_value(conf->admin_secret, prev->admin_secret, "");
149 ngx_conf_merge_value(conf->admin_require_both, prev->admin_require_both, 0);
150
151 /* Admin file-browser root: canonicalize once (realpath) into browse_root_canon
152 * so request-time confinement has a stable anchor. Empty => feature off. A
153 * configured-but-bad path fails config loudly (like an export root). */
154 ngx_conf_merge_str_value(conf->browse_root, prev->browse_root, "");
155 {
156 dashboard_canon_root_t browse = {
157 &conf->browse_root, conf->browse_root_canon,
158 prev->browse_root_canon, "brix_dashboard_browse_root",
159 "brix_dashboard_browse_root path too long"
160 };
161 if (dashboard_merge_canon_root(cf, &browse, &err) != NGX_OK) {
162 return err;
163 }
164 }
165
166 /* Storage-scan root: canonicalize once (realpath) into scan_root_canon, the
167 * confinement anchor for the /scan endpoint. Empty => feature off (404). */
168 ngx_conf_merge_str_value(conf->scan_root, prev->scan_root, "");
169 ngx_conf_merge_value(conf->vfs_browse, prev->vfs_browse, 0);
170 ngx_conf_merge_uint_value(conf->scan_max_files, prev->scan_max_files, 100000);
171 {
172 dashboard_canon_root_t scan = {
173 &conf->scan_root, conf->scan_root_canon, prev->scan_root_canon,
174 "brix_scan_root", "brix_scan_root path too long"
175 };
176 if (dashboard_merge_canon_root(cf, &scan, &err) != NGX_OK) {
177 return err;
178 }
179 }
180 /* Cross-field invariant: a transfer cannot become "stalled" before it is
181 * even "idle", so reject a config that inverts the two thresholds. */
182 if (conf->stalled_threshold_ms < conf->idle_threshold_ms) {
183 return "brix_dashboard_stalled_threshold must be greater than or equal to brix_dashboard_idle_threshold";
184 }
185 return NGX_CONF_OK;
186}

250/*
251 * Validate a cookie Path attribute. Must be a non-empty absolute path, and must
252 * contain no control bytes or ';' — those would let the value break out of the
253 * Set-Cookie attribute and inject further attributes (header/cookie injection).
254 */
255static ngx_int_t
256dashboard_cookie_path_valid(ngx_str_t *path)
257{
258 ngx_uint_t i;
259
260 if (path->len == 0 || path->data == NULL || path->data[0] != '/') {
261 return NGX_DECLINED;
262 }
263
264 for (i = 0; i < path->len; i++) {
265 if (path->data[i] < 0x20 || path->data[i] == ';') {
266 return NGX_DECLINED;
267 }
268 }
269
270 return NGX_OK;
271}
