14int
15brix_csi_open(brix_csi_t *c, const char *abs_path,
16 uint32_t granule, int writable)
17{
18 size_t plen;
19
20 if (c == NULL || abs_path == NULL || granule == 0) {
21 errno = EINVAL;
22 return BRIX_CSI_ERR;
23 }
24 plen = strlen(abs_path);
25 if (plen == 0 || plen >= sizeof(c->path)) {
26 errno = ENAMETOOLONG;
27 return BRIX_CSI_ERR;
28 }
29
30 memset(c, 0, sizeof(*c));
31 memcpy(c->path, abs_path, plen + 1);
32 c->granule = granule;
33 c->writable = writable ? 1 : 0;
34
35 if (writable) {
36 return BRIX_CSI_OK; /* the record is created/merged at flush */
37 }
38
39 /* Read handle: load the record ONCE and snapshot it on the handle. The
40 * at-rest record is immutable for a read handle's lifetime, so verify reads
41 * the cached copy instead of re-loading (getxattr + parse + malloc) on every
42 * read job — the difference between O(1) and O(reads) filesystem/heap work
43 * on the multi-threaded read hot path. A verifiable record also reports OK
44 * so the caller can enforce csi_require at open. */
45 {
46 brix_xmeta_t xm;
47 int rc = brix_xmeta_path_load(c->path, &xm);
48
49 if (rc == BRIX_XMETA_ERR) {
50 return BRIX_CSI_ERR;
51 }
52 if (rc == BRIX_XMETA_FOREIGN) {
53 return BRIX_CSI_NOTAGS;
54 }
55 if (!xm.have_blockcrc || xm.blockcrc == NULL) {
56 brix_xmeta_free(&xm);
57 return BRIX_CSI_NOTAGS;
58 }
59 c->record = malloc(sizeof(*c->record));
60 if (c->record == NULL) {
61 brix_xmeta_free(&xm);
62 errno = ENOMEM;
63 return BRIX_CSI_ERR;
64 }
65 *c->record = xm; /* move ownership (bitmap/blockcrc) */
66 return BRIX_CSI_OK;
67 }
68}

70void
71brix_csi_close(brix_csi_t *c)
72{
73 if (c == NULL) {
74 return;
75 }
76 if (c->writable && c->dirty) {
77 (void) brix_csi_flush(c); /* best-effort; tags stay unset on
78 failure (never falsely failing) */
79 }
80 if (c->record != NULL) {
81 brix_xmeta_free(c->record);
82 free(c->record);
83 }
84 free(c->local);
85 memset(c, 0, sizeof(*c));
86}
