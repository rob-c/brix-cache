1/*
2 * cephfs_layout.h — typed decoders for CephFS on-RADOS metadata structures.
3 *
4 * WHAT: Decodes the Ceph MDS's on-disk encodings — the directory-entry value
5 * (primary vs remote dentry), the embedded inode (inode_t), and the file
6 * layout (file_layout_t) — into plain C structs the read-only driver and
7 * recovery tools consume. Built entirely on cephfs_denc primitives.
8 *
9 * WHY: Serving a CephFS by reading RADOS directly means parsing exactly what the
10 * MDS wrote into the metadata-pool omaps. These encodings are versioned
11 * (struct_v grows as Ceph adds fields); decoding here is driven by the same
12 * version guards the Ceph source uses, so one decoder spans many releases
13 * and refuses cleanly on an encoding it does not understand.
14 *
15 * HOW: Every Ceph struct is wrapped in an ENCODE_START frame (struct_v / compat /
16 * byte-length). We decode the leading fields we need, honour each
17 * `if (struct_v >= N)` guard exactly as Ceph does, then jump to the frame's
18 * end (forward-compat) — so newer trailing fields are skipped, not
19 * mis-read. The inode is itself framed, so we only decode up to the fields
20 * the driver needs (ino/mode/size/mtime/layout) and frame-skip the rest.
21 *
22 * VERSION COVERAGE: inode_t decode honours every guard from struct_v 2 through 19
23 * (reef). file_layout_t is the long-stable framed v2 form. The byte layout
24 * is verified against captured reef-18.2.4 fixtures (tests/ceph/fixtures/)
25 * and the field order/guards are taken verbatim from the Ceph v18.2.4
26 * source (src/include/cephfs/types.h, src/mds/CInode.cc, src/mds/CDir.cc).
27 *
28 * SCOPE (this layer): primary inodes (files, dirs, symlinks) + file layout +
29 * remote-dentry ino/d_type, plus the directory fragment tree and inode
30 * xattr map that sit after the inode inside the InodeStore.
31 *
32 * Pure C; no RADOS, no nginx.
33 */

35#define BRIX_CEPHFS_LAYOUT_H

39/* Bounds on the per-inode collections we materialize. A directory with more leaf
40 * fragments, or an inode with more xattrs, than these is decoded up to the cap
41 * and flagged truncated (see cephfs_dentry_t.frags_truncated / xattrs_truncated)
42 * rather than overrunning. */
43#define CEPHFS_MAX_FRAGS 256
44#define CEPHFS_MAX_XATTRS 64

46/* Ceph st_mode type bits (independent of host <sys/stat.h>, which the decoder
47 * must not depend on for a portable, testable parse). */
48#define CEPHFS_S_IFMT 0170000u
49#define CEPHFS_S_IFDIR 0040000u
50#define CEPHFS_S_IFREG 0100000u
51#define CEPHFS_S_IFLNK 0120000u
52
53static inline int cephfs_mode_is_dir (uint32_t m) { return (m & CEPHFS_S_IFMT) == CEPHFS_S_IFDIR; }
54static inline int cephfs_mode_is_reg (uint32_t m) { return (m & CEPHFS_S_IFMT) == CEPHFS_S_IFREG; }
55static inline int cephfs_mode_is_link(uint32_t m) { return (m & CEPHFS_S_IFMT) == CEPHFS_S_IFLNK; }

57/* Decoded file/dir striping layout (file_layout_t). pool_ns is decoded but

91/* What a directory-entry value resolved to. */
92typedef enum {
93 CEPHFS_DENTRY_PRIMARY = 1, /* embeds a full inode (the common case) */
94 CEPHFS_DENTRY_REMOTE = 2, /* a hardlink: only ino + d_type are stored */
95} cephfs_dentry_kind_t;

97typedef struct {

100 /* REMOTE only: */

102 uint8_t remote_d_type;
103
104 /* PRIMARY only: */
105 cephfs_inode_t inode;

107 * null-terminated); NULL unless a symlink */
108 uint32_t symlink_len;
109
110 /* directory leaf fragments — the object-name suffixes ("%08x" of each leaf
111 * frag) whose omaps hold this dir's entries; for an unsplit dir this is the
112 * single value 0 ("00000000"). Meaningful for directories. */
113 uint32_t frag_enc[CEPHFS_MAX_FRAGS];
114 uint32_t nfrags;
115 int frags_truncated;
116
117 /* inode xattrs (read-only). */
118 cephfs_xattr_t xattrs[CEPHFS_MAX_XATTRS];
119 uint32_t nxattrs;
120 int xattrs_truncated;
121} cephfs_dentry_t;

123/* Decode a file_layout_t (framed) at the cursor into *out. 0 on success; -1 (and
124 * the cursor's sticky error set) on a malformed/unsupported encoding. */
125int cephfs_decode_file_layout(cephfs_denc_t *d, cephfs_layout_t *out);

127/* Decode an inode_t (framed) at the cursor into *out, honouring all version
128 * guards and frame-skipping fields past what we need. 0 / -1. */
129int cephfs_decode_inode(cephfs_denc_t *d, cephfs_inode_t *out);

131/* Decode a fragtree_t (an UNframed compact_map<frag_t,int32_t> as written inside
132 * the InodeStore) at the cursor and resolve it to its LEAF fragments, writing up
133 * to `max` leaf `_enc` values to `leaves` and the count to `*count`. An empty
134 * split map yields the single leaf 0. Sets *truncated when there are more leaves
135 * than `max`. 0 / -1 (cursor error). */
136int cephfs_decode_fragtree(cephfs_denc_t *d, uint32_t *leaves, uint32_t max,
137 uint32_t *count, int *truncated);
138
139/* Decode an xattr map (u32 count + (string name, bufferptr value) pairs, as
140 * written by InodeStore) at the cursor, writing up to `max` entries to `out`
141 * (pointers into the cursor's buffer). Sets *truncated when there are more than
142 * `max`. 0 / -1. */
143int cephfs_decode_xattrs(cephfs_denc_t *d, cephfs_xattr_t *out, uint32_t max,
144 uint32_t *count, int *truncated);
145
146/* Decode a whole directory-entry omap value (`<name>_head`) into *out. Handles
147 * primary ('I'/'i') and remote ('L'/'l') dentries. 0 on success; -1 on a
148 * malformed value or an encoding this decoder does not understand. */
149int cephfs_decode_dentry(const void *buf, size_t len, cephfs_dentry_t *out);
