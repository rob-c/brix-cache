12/* Wrapper: maps brix_http_resolve_path() return codes to NGX_HTTP_*,
13 * using 409 Conflict when the destination parent cannot be resolved
14 * (RFC 4918 §9.8 — parent collection does not exist). */
15static ngx_int_t
16map_resolve_rc(int rc)
17{
18 switch (rc) {
19 case 0: return NGX_OK;
20 case 403: return NGX_HTTP_FORBIDDEN;
21 case 414: return NGX_HTTP_REQUEST_URI_TOO_LARGE;
22 case 500: return NGX_HTTP_INTERNAL_SERVER_ERROR;
23 default: return NGX_HTTP_INTERNAL_SERVER_ERROR;
24 }
25}

35
36 (void) log;
37 (void) op_label;
38
39 dlen = strlen(decoded_path);
40 while (dlen > 1 && decoded_path[dlen - 1] == '/')
41 dlen--;
42
43 if (dlen >= sizeof(stripped))
44 return NGX_HTTP_REQUEST_URI_TOO_LARGE;
45 memcpy(stripped, decoded_path, dlen);
46 stripped[dlen] = '\0';
47
48 rc = brix_http_resolve_path_ex(root_canon, stripped, out, outsz,
49 allow_internal ? 1u : 0u);
50 if (rc == 0)
51 return NGX_OK;
52 /* 404 from shared resolver → parent doesn't exist → 409 Conflict */
53 if (rc == 404)
54 return NGX_HTTP_CONFLICT;
55 return map_resolve_rc(rc);
