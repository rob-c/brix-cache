1/*
2 * firefly.c — SciTags "firefly" UDP flow reporting.
3 *
4 * WHAT: The out-of-band SciTags mechanism: for each marked flow emit a
5 * RFC5424-syslog-wrapped JSON "firefly" document (start / ongoing / end) to the
6 * configured collector(s) and, optionally, to the client origin. Byte-for-byte
7 * compatible with XRootD's XrdNetPMarkFF (template at XrdNetPMarkFF.cc:60-99).
8 * This file owns the per-flow lifecycle entry points (flow_begin / flow_end /
9 * flow_echo) and the per-worker UDP sender.
10 *
11 * WHY: Firefly is the interop path — it is what site flowd collectors and the
12 * SciTags dashboards consume. It is also the begin/end site where the in-band
13 * IPv6 flow label is applied (flowlabel.c), so one call marks a flow both ways.
14 *
15 * HOW: Codes are resolved via mapping.c; the datagram is built with libc snprintf
16 * (only the parsed integers + numeric IPs reach it — never raw client bytes);
17 * delivery is a non-blocking, fire-and-forget sendto on a per-worker UDP socket
18 * (one AF_INET + one AF_INET6, created lazily). Everything is fail-open: a
19 * send error is dropped, never surfaced, and never blocks the data plane.
20 */

38static ngx_socket_t
39pmark_worker_socket(int family)
40{
41 ngx_socket_t *slot = (family == AF_INET6) ? &pmark_fd6 : &pmark_fd4;
42 ngx_socket_t fd;
43
44 if (*slot != (ngx_socket_t) -1) {
45 return *slot;
46 }
47 fd = ngx_socket(family, SOCK_DGRAM, 0);
48 if (fd == (ngx_socket_t) -1) {
49 return (ngx_socket_t) -1;
50 }
51 if (ngx_nonblocking(fd) == -1) {
52 ngx_close_socket(fd);
53 return (ngx_socket_t) -1;
54 }
55 *slot = fd;
56 return fd;
57}

60/* Build the firefly datagram into buf; returns its length (0 on failure). */
61static size_t
62pmark_build(brix_pmark_flow_t *f, const char *state, int with_end,
63 u_char *buf, size_t buflen)
64{
65 brix_pmark_sockstats_t st;
66 char now[48], endt[80];
67 uint64_t recv, sent;
68 ngx_str_t host;
69 int n;
70
71 host = ngx_cycle->hostname;
72
73 brix_pmark_sockstats((int) f->fd, &st);
74 /* "supplier is source": for a PUT the client supplies the data, so report
75 * bytes from the supplier's perspective (swap acked/received). */
76 if (f->is_put) { recv = st.bytes_sent; sent = st.bytes_recv; }
77 else { recv = st.bytes_recv; sent = st.bytes_sent; }
78
79 brix_pmark_iso8601_now(now, sizeof(now));
80
81 endt[0] = '\0';
82 if (with_end) {
83 char e[48];
84 brix_pmark_iso8601_now(e, sizeof(e));
85 (void) snprintf(endt, sizeof(endt), ",\"end-time\":\"%s\"", e);
86 }
87
88 n = snprintf((char *) buf, buflen,
89 "<134>1 - %.*s xrootd - firefly-json - "
90 "{\"version\":1,"
91 "\"flow-lifecycle\":{\"state\":\"%s\",\"current-time\":\"%s\","
92 "\"start-time\":\"%s\"%s},"
93 "\"usage\":{\"received\":%llu,\"sent\":%llu},"
94 "\"netlink\":{\"rtt\":%u.%03u},"
95 "\"context\":{\"experiment-id\":%u,\"activity-id\":%u,"
96 "\"application\":\"%s\"},"
97 "\"flow-id\":{\"afi\":\"ipv%c\",\"src-ip\":\"%s\",\"dst-ip\":\"%s\","
98 "\"protocol\":\"tcp\",\"src-port\":%d,\"dst-port\":%d}}",
99 (int) host.len, (const char *) host.data,
100 state, now, f->start_iso, endt,
101 (unsigned long long) recv, (unsigned long long) sent,
102 (unsigned) (st.rtt_us / 1000), (unsigned) (st.rtt_us % 1000),
103 (unsigned) f->exp, (unsigned) f->act, f->app,
104 f->afi, f->src_ip, f->dst_ip, f->src_port, f->dst_port);
105
106 if (n < 0) {
107 return 0;
108 }
109 return ((size_t) n < buflen) ? (size_t) n : buflen;
110}

113/* Fire one datagram to every configured collector + optional origin. */
114static void
115pmark_emit(brix_pmark_flow_t *f, const char *state, int with_end,
116 ngx_log_t *log)
117{
118 u_char buf[1280];
119 size_t n;
120 brix_pmark_conf_t *pm = f->pm;
121 ngx_uint_t i;
122
123 n = pmark_build(f, state, with_end, buf, sizeof(buf));
124 if (n == 0) {
125 return;
126 }
127
128 if (pm->dest_sa) {
129 brix_pmark_dest_t *d = pm->dest_sa->elts;
130 for (i = 0; i < pm->dest_sa->nelts; i++) {
131 ngx_socket_t fd = pmark_worker_socket(d[i].family);
132 if (fd == (ngx_socket_t) -1) {
133 continue;
134 }
135 if (sendto(fd, buf, n, 0, (struct sockaddr *) &d[i].ss, d[i].len)
136 < 0)
137 {
138 BRIX_PMARK_METRIC_INC(pmark_firefly_dropped_total);
139 ngx_log_debug1(NGX_LOG_DEBUG_CORE, log, ngx_errno,
140 "pmark: firefly sendto failed (state=%s)", state);
141 } else {
142 BRIX_PMARK_METRIC_INC(pmark_firefly_sent_total);
143 }
144 }
145 }
146
147 /* Report to the client origin (its IP at the firefly port). */
148 if (f->want_origin && f->peer_len) {
149 struct sockaddr_storage o;
150 ngx_socket_t fd;
151
152 ngx_memcpy(&o, &f->peer_ss, f->peer_len);
153 if (o.ss_family == AF_INET) {
154 ((struct sockaddr_in *) &o)->sin_port = htons(BRIX_PMARK_FF_PORT);
155 } else if (o.ss_family == AF_INET6) {
156 ((struct sockaddr_in6 *) &o)->sin6_port =
157 htons(BRIX_PMARK_FF_PORT);
158 }
159 fd = pmark_worker_socket(o.ss_family);
160 if (fd != (ngx_socket_t) -1) {
161 (void) sendto(fd, buf, n, 0, (struct sockaddr *) &o, f->peer_len);
162 }
163 }
164}

167brix_pmark_flow_t *
168brix_pmark_flow_begin(brix_pmark_conf_t *pm, ngx_pool_t *pool,
169 ngx_connection_t *c, int is_write, const char *vo_csv, const char *user,
170 const char *path, const char *cgi, ngx_log_t *log)
171{
172 brix_pmark_flow_t *f;
173 ngx_uint_t exp, act;
174 char pip[64], sip[64];
175 int pport, sport;
176 char pafi, safi;
177 size_t an;
178
179 if (pm == NULL || !pm->enable || c == NULL) {
180 return NULL;
181 }
182 if (brix_pmark_runtime_ensure(pm, ngx_cycle->pool, log) != NGX_OK) {
183 return NULL;
184 }
185 if (brix_pmark_map_codes(pm, vo_csv, user, path, cgi, &exp, &act)
186 != NGX_OK)
187 {
188 BRIX_PMARK_METRIC_INC(pmark_map_unresolved_total);
189 return NULL; /* nothing maps → not marked */
190 }
191
192 BRIX_PCALLOC_OR_RETURN(f, pool, sizeof(*f), NULL);
193 f->pm = pm;
194 f->exp = exp;
195 f->act = act;
196 f->fd = c->fd;
197 f->is_put = is_write ? 1 : 0;
198 f->want_origin = pm->firefly_origin ? 1 : 0;
199
200 an = pm->appname.len < sizeof(f->app) - 1
201 ? pm->appname.len : sizeof(f->app) - 1;
202 if (an) {
203 ngx_memcpy(f->app, pm->appname.data, an);
204 }
205 f->app[an] = '\0';
206
207 brix_pmark_iso8601_now(f->start_iso, sizeof(f->start_iso));
208
209 /* Endpoints: 0 = peer (client), 1 = local (server). */
210 brix_pmark_endpoint((int) c->fd, 0, pip, sizeof(pip), &pport, &pafi);
211 brix_pmark_endpoint((int) c->fd, 1, sip, sizeof(sip), &sport, &safi);
212 f->afi = pafi;
213 if (f->is_put) { /* client supplies → client is src */
214 ngx_memcpy(f->src_ip, pip, sizeof(f->src_ip));
215 ngx_memcpy(f->dst_ip, sip, sizeof(f->dst_ip));
216 f->src_port = pport;
217 f->dst_port = sport;
218 } else { /* server supplies → server is src */
219 ngx_memcpy(f->src_ip, sip, sizeof(f->src_ip));
220 ngx_memcpy(f->dst_ip, pip, sizeof(f->dst_ip));
221 f->src_port = sport;
222 f->dst_port = pport;
223 }
224
225 /* Capture the client peer sockaddr for origin firefly reports. */
226 f->peer_len = sizeof(f->peer_ss);
227 if (getpeername((int) c->fd, (struct sockaddr *) &f->peer_ss, &f->peer_len)
228 != 0)
229 {
230 f->peer_len = 0;
231 }
232
233 /* In-band IPv6 flow label (no-op on IPv4/mapped/disabled — see flowlabel.c). */
234 if (pm->flowlabel) {
235 (void) brix_pmark_flowlabel_apply(c, (int) c->fd, exp, act, log);
236 }
237
238 /* Out-of-band firefly start. */
239 if (pm->firefly) {
240 pmark_emit(f, "start", 0, log);
241 f->firefly_started = 1;
242 }
243
244 BRIX_PMARK_METRIC_INC(pmark_flows_started_total);
245 return f;
246}

281void
282brix_pmark_firefly_oneshot(brix_pmark_conf_t *pm, int fd, ngx_uint_t exp,
283 ngx_uint_t act, int peer_is_src, const char *app, ngx_log_t *log)
284{
285 brix_pmark_flow_t f;
286 char pip[64], sip[64];
287 int pport, sport;
288 char pafi, safi;
289 size_t an;
290
291 if (pm == NULL || !pm->firefly || fd < 0) {
292 return;
293 }
294
295 ngx_memzero(&f, sizeof(f));
296 f.pm = pm;
297 f.exp = exp;
298 f.act = act;
299 f.fd = fd;
300 f.is_put = peer_is_src ? 1 : 0;
301 f.firefly_started = 1;
302
303 if (app != NULL) {
304 an = ngx_strlen(app);
305 if (an >= sizeof(f.app)) { an = sizeof(f.app) - 1; }
306 ngx_memcpy(f.app, app, an);
307 f.app[an] = '\0';
308 }
309
310 brix_pmark_iso8601_now(f.start_iso, sizeof(f.start_iso));
311 brix_pmark_endpoint(fd, 0, pip, sizeof(pip), &pport, &pafi);
312 brix_pmark_endpoint(fd, 1, sip, sizeof(sip), &sport, &safi);
313 f.afi = pafi;
314 if (f.is_put) { /* peer is source */
315 ngx_memcpy(f.src_ip, pip, sizeof(f.src_ip)); f.src_port = pport;
316 ngx_memcpy(f.dst_ip, sip, sizeof(f.dst_ip)); f.dst_port = sport;
317 } else { /* local is source */
318 ngx_memcpy(f.src_ip, sip, sizeof(f.src_ip)); f.src_port = sport;
319 ngx_memcpy(f.dst_ip, pip, sizeof(f.dst_ip)); f.dst_port = pport;
320 }
321
322 /* Outbound sockets are short-lived to us (curl owns the connection); emit a
323 * start + end pair at close, when the socket is connected and TCP_INFO is
324 * available. */
325 pmark_emit(&f, "start", 0, log);
326 pmark_emit(&f, "end", 1, log);
327 BRIX_PMARK_METRIC_INC(pmark_flows_started_total);
328 BRIX_PMARK_METRIC_INC(pmark_flows_ended_total);
329}
