55 *
56 * WHY: Using this legacy format with v5 clients causes parsing errors — status=0x0008 / dlen=1312, causing the client to stall.

59/*---- ServerInitHandShake struct ----
60 *
61 * HOW: Layout is fixed: 4B msglen + 4B protover + 4B msgval = 12 bytes total, all big-endian.
62 * msglen indicates 8 more bytes follow; protover is server protocol version; msgval identifies server type (0=LB, 1=DataServer). */

64/*

66 * This is NOT standard ServerResponseHdr framing — it has its own layout:
67 * msglen[4] + protover[4] + msgval[4]

88/* ---- ClientRequestHdr — all client requests share this header ----
89 *

92 *

96 * HOW: Layout is fixed: 2B streamid + 2B reqid (kXR_* from opcodes.h) + 16B body + 4B dlen.
97 * Always big-endian. Use brix_dispatch_opcode() to route based on requestid field. */

109 * kXR_ok → result body · kXR_error → errnum[4] + errmsg · kXR_status → CRC32c body.

114 * HOW: Layout is fixed: 2B streamid (echoed) + 2B status (kXR_* from opcodes.h) + 4B dlen.
115 * Always big-endian. Use brix_build_resp_hdr() to construct, brix_queue_response() to send. */

117typedef struct {
118 kXR_char streamid[2]; /* echoed from request */
119 kXR_unt16 status; /* kXR_ok / kXR_error / ... */
120 kXR_int32 dlen; /* response body length */
121} ServerResponseHdr; /* 8 bytes; body follows inline */

123/* ---- kXR_status (4007) extended response — pgwrite/pgread integrity checking ----
124 *
125 * WHAT: Extended status frame used exclusively for kXR_pgwrite and kXR_pgread.
126 * Provides CRC32c integrity verification across all page transfers.
127 *
128 * WHY: Ensures data integrity in paged operations where multiple pages are sent/received.
129 * The CRC covers everything from streamID to end of body — any bit corruption detected.
130 *
131 * HOW: Full wire layout for pgwrite success (no bad pages), 32 bytes total:
132 * [ServerResponseHdr 8B] status=kXR_status, dlen=24
133 * [ServerResponseBody_Status 16B] crc32c, streamID, requestid, resptype, reserved, dlen=0
134 * [ServerResponseBody_pgWrite 8B] offset (last written)
135 *
136 * INVARIANT: kXR_pgwrite/pgwrite MUST use this framing + per-page CRC32c checksum. */
137
138typedef struct {
139 kXR_unt32 crc32c; /* CRC32c of everything from &streamID to end */
140 kXR_char streamID[2]; /* echo of request streamid */
141 kXR_char requestid; /* requestcode - kXR_1stRequest */
142 kXR_char resptype; /* 0=kXR_FinalResult, 1=kXR_PartialResult */
143 kXR_char reserved[4];
144 kXR_int32 dlen; /* size of bad-page list (0 = no bad pages) */
145} ServerResponseBody_Status; /* 16 bytes */
146
147typedef struct {
148 kXR_int64 offset; /* file offset of read data */
149} ServerResponseBody_pgRead; /* 8 bytes */
150
151typedef struct {
152 kXR_int64 offset; /* file offset of written data */
153} ServerResponseBody_pgWrite; /* 8 bytes */

155/* ---- kXR_pgwrite CSE (checksum-error) retransmit trailer ----
156 *
157 * Appended to ServerResponseBody_pgWrite when one or more pages failed CRC32c
158 * verification. The server replies with a SUCCESS kXR_status frame (not an
159 * error) whose bdy.dlen = sizeof(pgWrCSE) + n*8, then the client resends each
160 * listed page with reqflags |= kXR_pgRetry. Followed inline by a big-endian
161 * vector `kXR_int64 bof[n]` of the corrupt pages' file offsets.
162 *
163 * Wire layout for a CSE reply (n bad pages) — stock srsComplete convention:
164 * hdr.dlen counts ONLY the fixed status body (24); the CSE trailer follows as
165 * separate `data` of length bdy.dlen, exactly like pgread page data.
166 * [ServerResponseHdr 8B] status=kXR_status, dlen = 24
167 * [ServerResponseBody_Status] dlen = 8 + n*8 (size of the CSE trailer below)
168 * [ServerResponseBody_pgWrite] offset
169 * ---- the following bytes are NOT counted in hdr.dlen ----
170 * [ServerResponseBody_pgWrCSE 8B] cseCRC, dlFirst, dlLast
171 * [kXR_int64 bof[n]] corrupt-page file offsets (big-endian)
172 *
173 * body crc32c covers only the 20-byte fixed head (streamID..pgw.offset);
174 * cseCRC = CRC32c of every byte AFTER cseCRC (dlFirst..end of bof[]). */
175typedef struct {
176 kXR_unt32 cseCRC; /* CRC32c of all following bytes (dlFirst..bof end) */
177 kXR_int16 dlFirst; /* fragment length of the first bad page */
178 kXR_int16 dlLast; /* fragment length of the last bad page */
179} ServerResponseBody_pgWrCSE; /* 8 bytes; kXR_int64 bof[n] follows */
