1/*
2 * zip_dir.h — ZIP central-directory reader for member access (phase-57 W2).
3 *
4 * WHAT: Resolves a single member of a ZIP archive (by name) to its on-disk byte
5 * range and metadata, by reading the End-Of-Central-Directory, optional
6 * ZIP64 records, the central directory, and the member's local file header.
7 * WHY: ZIP member access (the xrdcl.unzip= opaque) serves one file inside an
8 * archive as a standalone object across root://, WebDAV and S3 GET. This is
9 * the read-only locator; byte serving lives in zip_member.c. Matches the
10 * XrdZip semantics (stored + deflate members).
11 * HOW: Pure C (no nginx, no OpenSSL): pread-only, fully bounds-checked against
12 * the archive size. Tail-scan for the EOCD signature; promote to ZIP64 when
13 * the 32-bit fields are saturated (0xFFFFFFFF / 0xFFFF); walk
14 * signature-checked CDFH records; reject data-descriptor entries (size
15 * unknown at open); read the LFH to compute the true first-data offset.
16 * The caller (zip_member.c) maps the int result to wire/HTTP errors and
17 * does all logging at the edge.
18 */

49/*
50 * Resolve `member` within the archive open on `fd` (size `archive_size`).
51 * `cd_max` caps the central-directory read (bomb guard on a hostile directory).
52 * Never reads outside [0, archive_size). Returns one of BRIX_ZIP_*.
53 * Rejects (BRIX_ZIP_ECORRUPT): encrypted entries, methods other than 0/8,
54 * data-descriptor entries (size not known at open), and any out-of-bounds field.
55 * On duplicate names the LAST central-directory entry wins (XrdZip semantics).
56 */
57int brix_zip_find_member(int fd, off_t archive_size, const char *member,
58 size_t cd_max, brix_zip_member_t *out);

64 * Returns the number of bytes produced (== m->uncomp_size on success) or -1 on a
65 * pread/inflate/allocation error. Intended for one-shot serving (HTTP/S3 GET);
66 * the streaming reader in zip_member.c is used for the root:// handle path.
67 */
68ssize_t brix_zip_extract_full(int fd, const brix_zip_member_t *m,
