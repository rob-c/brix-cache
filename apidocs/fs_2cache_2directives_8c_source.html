42/* brix_conf_set_cache_eviction_threshold — parse the eviction threshold as a
43 * decimal (0.95) or percentage (95%), stored as parts-per-million. */
44
45char *
46brix_conf_set_cache_eviction_threshold(ngx_conf_t *cf, ngx_command_t *cmd,
47 void *conf)
48{
49 ngx_stream_brix_srv_conf_t *xcf = conf;
50 ngx_str_t *value;
51 char *copy;
52 char *endp;
53 double ratio;
54 ngx_uint_t ppm;
55
56 value = cf->args->elts;
57 (void) cmd;
58
59 if (xcf->cache_eviction_threshold != NGX_CONF_UNSET_UINT) {
60 return "is duplicate";
61 }
62
63 copy = brix_pstrdup_z(cf->pool, &value[1]);
64 if (copy == NULL) {
65 return NGX_CONF_ERROR;
66 }
67
68 errno = 0;
69 ratio = strtod(copy, &endp);
70 if (endp == copy || errno != 0) {
71 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
72 "brix_cache_eviction_threshold: invalid value \"%V\"",
73 &value[1]);
74 return NGX_CONF_ERROR;
75 }
76
77 if (*endp == '%') {
78 endp++;
79 ratio /= 100.0;
80 } else if (ratio > 1.0) {
81 ratio /= 100.0;
82 }
83
84 if (*endp != '\0' || ratio <= 0.0 || ratio >= 1.0) {
85 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
86 "brix_cache_eviction_threshold must be greater than 0 "
87 "and less than 1.0 (or 1%% and 100%%)");
88 return NGX_CONF_ERROR;
89 }
90
91 ppm = (ngx_uint_t) (ratio * 1000000.0 + 0.5);
92 if (ppm == 0 || ppm >= 1000000) {
93 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
94 "brix_cache_eviction_threshold is out of range");
95 return NGX_CONF_ERROR;
96 }
97
98 xcf->cache_eviction_threshold = ppm;
99
100 return NGX_CONF_OK;
101}

163/* brix_conf_set_cache_max_file_size — parse the byte limit (raw bytes or k/m/g
164 * suffixes) into an off_t. Admission control: files larger than this are served
165 * from origin only, never cached. */
166
167char *
168brix_conf_set_cache_max_file_size(ngx_conf_t *cf, ngx_command_t *cmd,
169 void *conf)
170{
171 ngx_stream_brix_srv_conf_t *xcf = conf;
172 ngx_str_t *value;
173 char copy[64];
174 char *endp;
175 unsigned long long raw;
176 off_t bytes;
177
178 value = cf->args->elts;
179 (void) cmd;
180
181 if (xcf->cache_max_file_size != NGX_CONF_UNSET) {
182 return "is duplicate";
183 }
184
185 if (value[1].len == 0
186 || brix_str_cbuf(copy, sizeof(copy), &value[1]) == NULL)
187 {
188 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
189 "brix_cache_max_file_size: invalid value \"%V\"", &value[1]);
190 return NGX_CONF_ERROR;
191 }
192
193 errno = 0;
194 raw = strtoull(copy, &endp, 10);
195 if (endp == copy || errno != 0) {
196 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
197 "brix_cache_max_file_size: invalid number \"%V\"", &value[1]);
198 return NGX_CONF_ERROR;
199 }
200
201 switch (*endp) {
202 case 'k': case 'K': raw *= 1024ULL; endp++; break;
203 case 'm': case 'M': raw *= 1024ULL * 1024; endp++; break;
204 case 'g': case 'G': raw *= 1024ULL * 1024 * 1024; endp++; break;
205 case '\0': break;
206 default:
207 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
208 "brix_cache_max_file_size: unknown suffix in \"%V\" "
209 "(use k/m/g)", &value[1]);
210 return NGX_CONF_ERROR;
211 }
212
213 if (*endp != '\0') {
214 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
215 "brix_cache_max_file_size: trailing garbage in \"%V\"",
216 &value[1]);
217 return NGX_CONF_ERROR;
218 }
219
220 bytes = (off_t) raw;
221 if (bytes < 0) {
222 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
223 "brix_cache_max_file_size: value \"%V\" overflows off_t",
224 &value[1]);
225 return NGX_CONF_ERROR;
226 }
227
228 xcf->cache_max_file_size = bytes;
229
230 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
231 "brix: cache admission limit: %llu bytes",
232 (unsigned long long) bytes);
233
234 return NGX_CONF_OK;
235}

237/* brix_conf_set_cache_include_regex — compile the POSIX extended pattern
238 * (regcomp REG_EXTENDED|REG_NOSUB). Admission filter: only matching paths are
239 * cached; all others serve from origin. */
240
241char *
242brix_conf_set_cache_include_regex(ngx_conf_t *cf, ngx_command_t *cmd,
243 void *conf)
244{
245 ngx_stream_brix_srv_conf_t *xcf = conf;
246 ngx_str_t *value;
247 char *pattern;
248 int rc;
249 char errbuf[256];
250
251 value = cf->args->elts;
252 (void) cmd;
253
254 if (xcf->include_regex.set) {
255 return "is duplicate";
256 }
257
258 /* Copy to NUL-terminated buffer for regcomp */
259 pattern = brix_pstrdup_z(cf->pool, &value[1]);
260 if (pattern == NULL) {
261 return NGX_CONF_ERROR;
262 }
263
264 rc = regcomp(&xcf->include_regex.re, pattern, REG_EXTENDED | REG_NOSUB);
265 if (rc != 0) {
266 regerror(rc, &xcf->include_regex.re, errbuf, sizeof(errbuf));
267 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
268 "brix_cache_include_regex: invalid pattern \"%V\": %s",
269 &value[1], errbuf);
270 return NGX_CONF_ERROR;
271 }
272
273 xcf->include_regex.str.data = (u_char *) pattern;
274 xcf->include_regex.str.len = value[1].len;
275 xcf->include_regex.set = 1;
276
277 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
278 "brix: cache include regex: \"%V\"", &value[1]);
279
280 return NGX_CONF_OK;
281}

455/* brix_conf_set_wt_origin — parse brix_wt_origin (same format as cache_origin:
456 * plain host:port or root://host:port, IPv6 brackets supported), the destination
457 * for write-through propagation when wt_enable=on. */
458
459char *
460brix_conf_set_wt_origin(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
461{
462 ngx_stream_brix_srv_conf_t *xcf = conf;
463 ngx_str_t *value;
464 char *addr_copy, *colon, *endp;
465 const u_char *addr_data;
466 size_t addr_len;
467 long pnum;
468
469 value = cf->args->elts;
470 (void) cmd;
471
472 if (xcf->wt.origin_host.len > 0) {
473 return "is duplicate";
474 }
475
476 /* Accept plain host:port or root://host:port (same as cache_origin). */
477 addr_data = value[1].data;
478 addr_len = value[1].len;
479
480 if (addr_len > sizeof("root://") - 1
481 && ngx_strncmp(addr_data, "root://", sizeof("root://") - 1) == 0)
482 {
483 addr_data += sizeof("root://") - 1;
484 addr_len -= sizeof("root://") - 1;
485 }
486
487 BRIX_PNALLOC_OR_RETURN(addr_copy, cf->pool, addr_len + 1, NGX_CONF_ERROR);
488 ngx_memcpy(addr_copy, addr_data, addr_len);
489 addr_copy[addr_len] = '\0';
490
491 /* Parse host:port — supports IPv6 bracket notation. */
492 if (addr_copy[0] == '[') {
493 char *rb = strchr(addr_copy, ']');
494 if (rb == NULL || *(rb + 1) != ':') {
495 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
496 "brix_wt_origin: invalid address \"%V\"", &value[1]);
497 return NGX_CONF_ERROR;
498 }
499 size_t hostlen = (size_t)(rb - addr_copy - 1);
500 if (brix_pstrdupz(cf->pool, &xcf->wt.origin_host,
501 (u_char *) addr_copy + 1, hostlen) != NGX_OK) {
502 return NGX_CONF_ERROR;
503 }
504 pnum = strtol(rb + 2, &endp, 10);
505 } else {
506 colon = strrchr(addr_copy, ':');
507 if (colon == NULL) {
508 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
509 "brix_wt_origin: missing port in \"%V\"", &value[1]);
510 return NGX_CONF_ERROR;
511 }
512 size_t hostlen = (size_t)(colon - addr_copy);
513 if (brix_pstrdupz(cf->pool, &xcf->wt.origin_host,
514 (u_char *) addr_copy, hostlen) != NGX_OK) {
515 return NGX_CONF_ERROR;
516 }
517 pnum = strtol(colon + 1, &endp, 10);
518 }
519
520 if (*endp != '\0' || pnum <= 0 || pnum > 65535) {
521 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
522 "brix_wt_origin: invalid port in \"%V\"", &value[1]);
523 return NGX_CONF_ERROR;
524 }
525 xcf->wt.origin_port = (uint16_t) pnum;
526
527 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
528 "brix: write-through origin: %s:%d",
529 (char *) xcf->wt.origin_host.data, (int) xcf->wt.origin_port);
530
531 return NGX_CONF_OK;
532}

536/* brix_conf_add_wt_prefix — push a parsed prefix onto the WT allow or deny
537 * ngx_array_t (created on first use, capacity 4, cf->pool-lived across merges).

539
540/* Push one cf->args[1] prefix onto *target (created on first use, cf->pool-lived
541 * so it persists across merges). Shared by the write-through AND read-cache
542 * allow/deny directives so both lists are parsed identically. */
543static char *
544brix_conf_push_prefix(ngx_conf_t *cf, ngx_array_t **target,
545 const char *log_label)
546{

548 brix_wt_prefix_entry_t *entry;
549 char *prefix_copy;
550

553 if (*target == NULL) {
554 return NGX_CONF_ERROR;
555 }
556 }
557 entry = ngx_array_push(*target);
558 if (entry == NULL) {
559 return NGX_CONF_ERROR;

562 if (prefix_copy == NULL) {
563 return NGX_CONF_ERROR;
564 }

566 entry->prefix.len = value[1].len;
567
568 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0, "%s: \"%V\"", log_label, &value[1]);
