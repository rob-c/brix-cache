54/* Open helper shared by open()/create(): writable + force tri-state. */
55int
56afh_open(const char *path, int writable, int force, struct fuse_file_info *fi)
57{
58 brix_status st;
59 afh *h = calloc(1, sizeof(*h));
60 if (h == NULL) {
61 return -ENOMEM;
62 }
63 pthread_mutex_init(&h->lock, NULL);
64 h->writable = writable;
65 brix_iobuf_init(&h->io, h, afh_io_pread, afh_io_pwrite, writable,
66 g_readahead, g_writeback);
67 brix_status_clear(&st);
68
69 /* HTTP(S)/WebDAV mounts are READ-ONLY (Range GET); reject write opens. */
70 if (g_web) {
71 if (writable) {
72 afh_free(h);
73 return -EROFS;
74 }
75 char pbuf[XRDC_PATH_MAX];
76 h->wf = brix_webfile_open(&g_weburl, srv_path(path, pbuf, sizeof(pbuf)),
77 g_bearer, g_web_verify, g_web_ca, g_max_stall,
78 NULL, &st);
79 if (h->wf == NULL) {
80 afh_free(h);
81 return xfs_err(&st);
82 }
83 fi->fh = (uint64_t) (uintptr_t) h;
84 return 0;
85 }
86
87 /* phase-42 W4/W5: request inline compression when configured — on read opens
88 * the server compresses responses (W4), on write opens it decompresses our
89 * payloads (W5). A server that doesn't support it returns plaintext (the
90 * mfile learns read_codec/write_codec == 0). */
91 char opq[48];
92 const char *opaque = NULL;
93 if (g_compress[0] != '\0') {
94 snprintf(opq, sizeof(opq), "xrootd.compress=%s", g_compress);
95 opaque = opq;
96 }
97 char pbuf[XRDC_PATH_MAX];
98 h->mf = brix_mfile_open(brix_mgr_pick(g_mgr), srv_path(path, pbuf, sizeof(pbuf)),
99 writable, force, 0, opaque,
100 g_max_stall, g_max_retries, &st);
101 if (h->mf == NULL) {
102 afh_free(h);
103 return xfs_err(&st);
104 }
105 fi->fh = (uint64_t) (uintptr_t) h;
106 return 0;
107}

151int
152xfs_write(const char *path, const char *buf, size_t size, off_t offset,
153 struct fuse_file_info *fi)
154{
155 if (g_web) return -EROFS;
156 afh *h = (afh *) (uintptr_t) fi->fh;
157 brix_status st;
158 int rc;
159
160 (void) path;
161 if (h == NULL || !h->writable) {
162 return -EBADF;
163 }
164 brix_status_clear(&st);
165 pthread_mutex_lock(&h->lock);
166 rc = brix_iobuf_write(&h->io, offset, buf, size, &st);
167 if (rc == 0 && (int64_t) offset + (int64_t) size > h->wsize) {
168 h->wsize = (int64_t) offset + (int64_t) size; /* track logical EOF */
169 }
170 pthread_mutex_unlock(&h->lock);
171 return rc != 0 ? xfs_err(&st) : (int) size;
172}
