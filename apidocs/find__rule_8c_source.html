5/*
6 * WHAT: Longest-prefix rule matching — find the most specific policy rule for a resolved path.
7 * WHY: Policy rules (VO, group, manager-map) are configured as prefixes that may overlap (e.g., `/data` and `/data/atlas`).
8 * The longest prefix match ensures the most specific rule wins when multiple prefixes cover the same path.
9 * This is critical for VO membership checks where broader groups inherit from narrower ones. INVARIANT:
10 * all paths passed here must be canonical (normalized via normalize.c) before matching.
11 * HOW: Static helper brix_path_prefix_match validates prefix containment (must match up to prefix_len AND either end-of-string or next slash).
12 * Three public functions iterate their respective rule arrays using this helper, tracking best_len to select longest-matching rule.
13 * Returns NULL if no match found or input is invalid.
14 */
15
16/* brix_path_prefix_match — boundary-aware prefix test: path starts with prefix
17 * (prefix_len bytes) AND the next byte is '\0' or '/', so `/data` matches
18 * `/data/atlas` but not `/data-atlas`. The caller passes prefix_len (it knows the
19 * rule length), avoiding a redundant strlen per request in the match loop. */
20
21static ngx_flag_t
22brix_path_prefix_match(const char *prefix, size_t prefix_len, const char *path)
23{
24 if (prefix == NULL || path == NULL) {
25 return 0;
26 }
27
28 if (strncmp(prefix, path, prefix_len) != 0) {
29 return 0;
30 }
31
32 /*
33 * The match must land on a path-component boundary so "/foo" does not match
34 * "/foobar". A prefix that itself ends in '/' is already at a separator — the
35 * root prefix "/" (which cannot be stripped) and any explicit "/dir/" prefix
36 * therefore match everything beneath them. Without this, a manager_map / VO /
37 * authdb rule on "/" matched only the literal "/" and not "/file" (e.g. a
38 * static-map redirector failed to redirect a stat of "/blob.bin").
39 */
40 if (prefix_len > 0 && prefix[prefix_len - 1] == '/') {
41 return 1;
42 }
43
44 return path[prefix_len] == '\0' || path[prefix_len] == '/';
45}

47/* brix_find_vo_rule — longest-prefix-match the VO (Virtual Organization) rule for
48 * a resolved path (a more specific `/data/atlas/run3` rule wins over `/data/atlas`):
49 * scan the vo_rule_t array via brix_path_prefix_match tracking best_len. NULL if
50 * none match. */
51
52const brix_vo_rule_t *
53brix_find_vo_rule(const char *resolved_path, ngx_array_t *rules)
54{
55 const brix_vo_rule_t *best = NULL;
56 brix_vo_rule_t *rule;
57 size_t best_len = 0;
58 ngx_uint_t i;
59
60 if (resolved_path == NULL || rules == NULL) {
61 return NULL;
62 }
63
64 rule = rules->elts;
65 for (i = 0; i < rules->nelts; i++) {
66 size_t rule_len = strlen(rule[i].resolved);
67
68 if (!brix_path_prefix_match(rule[i].resolved, rule_len,
69 resolved_path))
70 {
71 continue;
72 }
73
74 if (rule_len >= best_len) {
75 best = &rule[i];
76 best_len = rule_len;
77 }
78 }
79
80 return best;
81}
