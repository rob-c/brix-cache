30/*
31 * Confinement model for the open/op helpers in this file: defence-in-depth in
32 * two layers — (1) canonical-path resolution by the caller (no symlinks, no
33 * ".."), and (2) kernel-level confinement via openat2 RESOLVE_BENEATH (Linux
34 * 5.6+) or an O_NOFOLLOW fallback. Both must pass, so a symlink swapped in
35 * between resolve and open still cannot escape the export root.
36 */

38/*
39 * brix_dirlist_access_ok — under impersonation, verify the mapped user may
40 * enumerate <resolved> by asking the broker to open it O_RDONLY|O_DIRECTORY as
41 * that user (EACCES ⇒ not entitled). NGX_OK when impersonation is off or the
42 * open succeeds; fail-closed (NGX_ERROR) when map mode is on but no principal is
43 * set for the request.
44 */
45
46ngx_int_t
47brix_dirlist_access_ok(ngx_log_t *log, const char *root_canon,
48 const char *resolved)
49{
50 char rel[PATH_MAX];
51 int fd;
52
53 if (!brix_imp_enabled()) {
54 return NGX_OK; /* impersonation off — existing gate holds */
55 }
56 if (!brix_imp_client_active()) {
57 /*
58 * Map mode is active but this request carries no per-request principal
59 * (e.g. the principal was cleared by an async body read and not yet
60 * re-established, or an anonymous request reached here). We cannot
61 * determine the mapped user's DAC, so FAIL CLOSED — refuse to enumerate
62 * the directory rather than list it with the privileged worker's
63 * credentials (which leaked entries of dirs the mapped user cannot read).
64 */
65 ngx_log_error(NGX_LOG_WARN, log, 0,
66 "impersonate: dirlist access check with no principal set"
67 " — refusing to enumerate \"%s\" (fail closed)", resolved);
68 return NGX_ERROR;
69 }
70 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
71 rel, sizeof(rel)))
72 {
73 return NGX_ERROR;
74 }
75 /* Ask the broker to open the dir for READING as the mapped user. readdir
76 * needs read permission on the directory, so a successful O_RDONLY|O_DIRECTORY
77 * open means the user is entitled to list it; EACCES means they are not. */
78 fd = brix_imp_open(rel, O_RDONLY | O_DIRECTORY, 0);
79 if (fd < 0) {
80 return NGX_ERROR;
81 }
82 close(fd);
83 return NGX_OK;
84}

86/*
87 * brix_open_confined_canon — open <resolved> (already canonical, under
88 * root_canon) with kernel confinement: openat2(RESOLVE_BENEATH), or on an older
89 * kernel a confined parent open + O_NOFOLLOW openat of the final component.
90 * Under impersonation the open is delegated to the broker (as the mapped user).
91 * Returns a NON-pool fd (caller closes) or -1. Arg order is (…, flags, mode):
92 * never pass permission bits in the flags slot (0644 sets O_EXCL).
93 */
94int
95brix_open_confined_canon(ngx_log_t *log, const char *root_canon,
96 const char *resolved, int flags, mode_t mode)
97{
98 char rel[PATH_MAX];
99 char parent[PATH_MAX];
100 char base[NAME_MAX + 1];
101 int rootfd;
102 int parentfd;
103 int fd;
104
105 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
106 rel, sizeof(rel)))
107 {
108 return -1;
109 }
110
111 /*
112 * Phase 40: when per-request impersonation is active (map mode + a principal
113 * set for this request), delegate the open to the privileged broker, which
114 * performs it as the mapped UNIX user under its own rootfd. `rel` is already
115 * the export-root-relative path the broker expects. This is the HTTP/S3
116 * (legacy confined-open) counterpart to the seam in beneath.c; off-path it is
117 * an inert flag check.
118 */
119 if (brix_imp_client_active()) {
120 return brix_imp_open(rel, flags, mode);
121 }
122
123 rootfd = brix_open_root_fd(log, root_canon);
124 if (rootfd < 0) {
125 return -1;
126 }
127
128#if (BRIX_HAVE_OPENAT2)
129 fd = brix_openat2_confined(rootfd, rel, flags, mode);
130 if (fd >= 0
131 || (errno != ENOSYS && errno != EINVAL && errno != EOPNOTSUPP))
132 {
133 close(rootfd);
134 return fd;
135 }
136#endif
137
138 /*
139 * Root-directory case: rel="." means the target IS the export root.
140 * brix_split_relative_parent refuses "." (no parent to navigate to),
141 * so open the current directory of rootfd directly with O_NOFOLLOW.
142 */
143 if (rel[0] == '.' && rel[1] == '\0') {
144 fd = openat(rootfd, ".", flags | O_CLOEXEC | O_NOFOLLOW, mode);
145 close(rootfd);
146 return fd;
147 }
148
149 if (!brix_split_relative_parent(rel, parent, sizeof(parent),
150 base, sizeof(base)))
151 {
152 close(rootfd);
153 return -1;
154 }
155
156 parentfd = brix_open_confined_parent_fallback(rootfd, parent);
157 close(rootfd);
158 if (parentfd < 0) {
159 return -1;
160 }
161
162 fd = openat(parentfd, base, flags | O_CLOEXEC | O_NOFOLLOW, mode);
163 close(parentfd);
164 return fd;
165}

254/* Shared body for the plain and create-if-absent confined renames. When
255 * `noreplace` is set, uses renameat2(RENAME_NOREPLACE) and falls back to a plain
256 * renameat (logged once) on kernels/filesystems without the flag. */
257static int
258rename_confined_canon_impl(ngx_log_t *log, const char *root_canon,
259 const char *src_resolved, const char *dst_resolved, int noreplace)
260{
261 char src_base[NAME_MAX + 1];
262 char dst_base[NAME_MAX + 1];
263 int src_parentfd;
264 int dst_parentfd;
265 int rc;
266
267 /* Phase 40: route through the broker (as the mapped user) when active. */
268 if (brix_imp_client_active()) {
269 char rsrc[PATH_MAX], rdst[PATH_MAX];
270 if (!brix_resolved_relative_to_root(log, root_canon, src_resolved,
271 rsrc, sizeof(rsrc))
272 || !brix_resolved_relative_to_root(log, root_canon, dst_resolved,
273 rdst, sizeof(rdst)))
274 {
275 return -1;
276 }
277 return noreplace ? brix_imp_rename_noreplace(rsrc, rdst)
278 : brix_imp_rename(rsrc, rdst);
279 }
280
281 src_parentfd = brix_open_confined_parent_canon(log, root_canon,
282 src_resolved, src_base,
283 sizeof(src_base));
284 if (src_parentfd < 0) {
285 return -1;
286 }
287
288 dst_parentfd = brix_open_confined_parent_canon(log, root_canon,
289 dst_resolved, dst_base,
290 sizeof(dst_base));
291 if (dst_parentfd < 0) {
292 close(src_parentfd);
293 return -1;
294 }
295
296 if (noreplace) {
297 rc = (int) syscall(SYS_renameat2, src_parentfd, src_base,
298 dst_parentfd, dst_base,
299 (unsigned int) RENAME_NOREPLACE);
300 if (rc != 0 && (errno == ENOSYS || errno == EINVAL)) {
301 static int warned = 0;
302 if (!warned) {
303 warned = 1;
304 ngx_log_error(NGX_LOG_WARN, log, errno,
305 "brix: renameat2(RENAME_NOREPLACE) unsupported; "
306 "create-if-absent falls back to non-atomic rename");
307 }
308 rc = renameat(src_parentfd, src_base, dst_parentfd, dst_base);
309 }
310 } else {
311 rc = renameat(src_parentfd, src_base, dst_parentfd, dst_base);
312 }
313 {
314 int e = errno;
315 close(src_parentfd);
316 close(dst_parentfd);
317 errno = e;
318 }
319 return rc;
320}

387/* Confined directory open — broker-routed fd + fdopendir under impersonation.
388 *
389 * open <resolved> as a directory stream that, under impersonation map mode,
390 * is opened BY THE BROKER as the mapped user (O_RDONLY|O_DIRECTORY via
391 * RESOLVE_BENEATH) and handed back as an fd that we fdopendir(). readdir on
392 * that stream then enumerates the directory with the mapped user's read
393 * permission — so a worker that cannot itself opendir a user-private dir
394 * (e.g. a 0700 multipart staging dir owned by the mapped user) can still
395 * list it on the user's behalf. Off impersonation it is a plain opendir().
396 *
397 * WHY: S3 multipart ListParts / AbortMultipartUpload (and the staging-dir cleanup)
398 * enumerate a staging directory created+owned by the mapped user. Done as a
399 * raw worker-side opendir() they fail EACCES under impersonation; routing the
400 * open through the broker fixes that while keeping per-entry removal on the
401 * already-brokered brix_unlink_confined_canon path. Returns a DIR* (caller
402 * closedir()s it) or NULL (errno set). */
403DIR *
404brix_opendir_confined_canon(ngx_log_t *log, const char *root_canon,
405 const char *resolved)
406{
407 char rel[PATH_MAX];
408 int rootfd;
409 DIR *d;
410
411 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
412 rel, sizeof(rel)))
413 {
414 return NULL;
415 }
416
417 if (brix_imp_client_active()) {
418 int fd = brix_imp_open(rel, O_RDONLY | O_DIRECTORY, 0); /* mapped user */
419 if (fd < 0) {
420 return NULL;
421 }
422 d = fdopendir(fd);
423 if (d == NULL) {
424 close(fd);
425 }
426 return d; /* closedir() closes the fd */
427 }
428
429 /*
430 * Off impersonation, open the directory chroot-style under an O_PATH rootfd
431 * (openat2 RESOLVE_IN_ROOT) rather than a bare opendir() on the canonical
432 * path: a bare opendir() follows a trailing in-export symlink with an
433 * outward target (e.g. /_sym_dir -> /etc) straight out of the export root
434 * and enumerates it — a confinement escape. RESOLVE_IN_ROOT confines the
435 * resolution so an escaping target is refused.
436 */
437 rootfd = brix_open_root_fd(log, root_canon);
438 if (rootfd < 0) {
439 return NULL;
440 }
441 d = brix_opendir_beneath(rootfd, rel);
442 close(rootfd);
443 return d;
444}

446/* brix_lstat_confined_canon — lstat()/stat() a path *as the mapped user* under
447 * impersonation, else a bare lstat/stat. WHY: recursive walks (COPY/MOVE
448 * collections, S3 multipart, remove-tree) lstat children of a directory owned
449 * 0700 by the mapped user; a raw worker lstat would EACCES on the parent's
450 * search bit. Routes through the broker (which stat()s as the mapped user) so
451 * the walk sees what that user can see. nofollow!=0 → lstat semantics (do not
452 * follow a trailing symlink — confinement: never resolve a link out of the
453 * export). Returns 0 on success, -1 on error (errno set). */
454int
455brix_lstat_confined_canon(ngx_log_t *log, const char *root_canon,
456 const char *resolved, struct stat *st, int nofollow)
457{
458 char rel[PATH_MAX];
459 int rootfd;
460 int rc;
461
462 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
463 rel, sizeof(rel)))
464 {
465 return -1;
466 }
467
468 if (brix_imp_client_active()) {
469 return brix_imp_stat(rel, st, nofollow); /* as mapped user */
470 }
471
472 /*
473 * Off impersonation, resolve chroot-style under an O_PATH rootfd via openat2
474 * RESOLVE_IN_ROOT (brix_{,l}stat_beneath) rather than a bare stat()/lstat()
475 * on the canonical path. A bare follow-stat() dereferences a trailing
476 * symlink against the REAL filesystem, so a planted in-export link with an
477 * outward absolute target (e.g. /_sym -> /etc/passwd) would be followed
478 * straight out of the export root — a confinement escape. RESOLVE_IN_ROOT
479 * resolves "/" and ".." relative to rootfd, so an outward target lands on a
480 * non-existent in-root path (ENOENT) instead, which the stat handler's
481 * realpath-confined fallback then rejects. nofollow=1 still reports a
482 * trailing link as itself (O_PATH|O_NOFOLLOW) without following it.
483 */
484 rootfd = brix_open_root_fd(log, root_canon);
485 if (rootfd < 0) {
486 return -1;
487 }
488 rc = nofollow ? brix_lstat_beneath(rootfd, rel, st)
489 : brix_stat_beneath(rootfd, rel, st);
490 close(rootfd);
491 return rc;
492}
