30/*
31 * Confinement model for the open/op helpers in this file: defence-in-depth in
32 * two layers — (1) canonical-path resolution by the caller (no symlinks, no
33 * ".."), and (2) kernel-level confinement via openat2 RESOLVE_BENEATH (Linux
34 * 5.6+) or an O_NOFOLLOW fallback. Both must pass, so a symlink swapped in
35 * between resolve and open still cannot escape the export root.
36 */
37
38/*
39 * brix_dirlist_access_ok — under impersonation, verify the mapped user may
40 * enumerate <resolved> by asking the broker to open it O_RDONLY|O_DIRECTORY as
41 * that user (EACCES ⇒ not entitled). NGX_OK when impersonation is off or the
42 * open succeeds; fail-closed (NGX_ERROR) when map mode is on but no principal is
43 * set for the request.
44 */
45
46ngx_int_t
47brix_dirlist_access_ok(ngx_log_t *log, const char *root_canon,
48 const char *resolved)
49{
50 char rel[PATH_MAX];
51 int fd;
52
53 if (!brix_imp_enabled()) {
54 return NGX_OK; /* impersonation off — existing gate holds */
55 }
56 if (!brix_imp_client_active()) {
57 /*
58 * Map mode is active but this request carries no per-request principal
59 * (e.g. the principal was cleared by an async body read and not yet
60 * re-established, or an anonymous request reached here). We cannot
61 * determine the mapped user's DAC, so FAIL CLOSED — refuse to enumerate
62 * the directory rather than list it with the privileged worker's
63 * credentials (which leaked entries of dirs the mapped user cannot read).
64 */
65 ngx_log_error(NGX_LOG_WARN, log, 0,
66 "impersonate: dirlist access check with no principal set"
67 " — refusing to enumerate \"%s\" (fail closed)", resolved);
68 return NGX_ERROR;
69 }
70 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
71 rel, sizeof(rel)))
72 {
73 return NGX_ERROR;
74 }
75 /* Ask the broker to open the dir for READING as the mapped user. readdir
76 * needs read permission on the directory, so a successful O_RDONLY|O_DIRECTORY
77 * open means the user is entitled to list it; EACCES means they are not. */
78 fd = brix_imp_open(rel, O_RDONLY | O_DIRECTORY, 0);
79 if (fd < 0) {
80 return NGX_ERROR;
81 }
82 close(fd);
83 return NGX_OK;
84}

86/*
87 * brix_open_confined_canon — open <resolved> (already canonical, under
88 * root_canon) with kernel confinement: openat2(RESOLVE_BENEATH), or on an older
89 * kernel a confined parent open + O_NOFOLLOW openat of the final component.
90 * Under impersonation the open is delegated to the broker (as the mapped user).
91 * Returns a NON-pool fd (caller closes) or -1. Arg order is (…, flags, mode):
92 * never pass permission bits in the flags slot (0644 sets O_EXCL).
93 */
94int
95brix_open_confined_canon(ngx_log_t *log, const char *root_canon,
96 const char *resolved, int flags, mode_t mode)
97{
98 char rel[PATH_MAX];
99 char parent[PATH_MAX];
100 char base[NAME_MAX + 1];
101 int rootfd;
102 int parentfd;
103 int fd;
104
105 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
106 rel, sizeof(rel)))
107 {
108 return -1;
109 }
110
111 /*
112 * Phase 40: when per-request impersonation is active (map mode + a principal
113 * set for this request), delegate the open to the privileged broker, which
114 * performs it as the mapped UNIX user under its own rootfd. `rel` is already
115 * the export-root-relative path the broker expects. This is the HTTP/S3
116 * (legacy confined-open) counterpart to the seam in beneath.c; off-path it is
117 * an inert flag check.
118 */
119 if (brix_imp_client_active()) {
120 return brix_imp_open(rel, flags, mode);
121 }
122
123 rootfd = brix_open_root_fd(log, root_canon);
124 if (rootfd < 0) {
125 return -1;
126 }
127
128#if (BRIX_HAVE_OPENAT2)
129 fd = brix_openat2_confined(rootfd, rel, flags, mode);
130 if (fd >= 0
131 || (errno != ENOSYS && errno != EINVAL && errno != EOPNOTSUPP))
132 {
133 close(rootfd);
134 return fd;
135 }
136#endif
137
138 /*
139 * Root-directory case: rel="." means the target IS the export root.
140 * brix_split_relative_parent refuses "." (no parent to navigate to),
141 * so open the current directory of rootfd directly with O_NOFOLLOW.
142 */
143 if (rel[0] == '.' && rel[1] == '\0') {
144 fd = openat(rootfd, ".", flags | O_CLOEXEC | O_NOFOLLOW, mode);
145 close(rootfd);
146 return fd;
147 }
148
149 if (!brix_split_relative_parent(rel, parent, sizeof(parent),
150 base, sizeof(base)))
151 {
152 close(rootfd);
153 return -1;
154 }
155
156 parentfd = brix_open_confined_parent_fallback(rootfd, parent);
157 close(rootfd);
158 if (parentfd < 0) {
159 return -1;
160 }
161
162 fd = openat(parentfd, base, flags | O_CLOEXEC | O_NOFOLLOW, mode);
163 close(parentfd);
164 return fd;
165}

254/* Shared body for the plain and create-if-absent confined renames. When
255 * `noreplace` is set, uses renameat2(RENAME_NOREPLACE) and falls back to a plain
256 * renameat (logged once) on kernels/filesystems without the flag. */
257static int
258rename_confined_canon_impl(ngx_log_t *log, const char *root_canon,
259 const char *src_resolved, const char *dst_resolved, int noreplace)
260{
261 char src_base[NAME_MAX + 1];
262 char dst_base[NAME_MAX + 1];
263 int src_parentfd;
264 int dst_parentfd;
265 int rc;
266
267 /* Phase 40: route through the broker (as the mapped user) when active. */
268 if (brix_imp_client_active()) {
269 char rsrc[PATH_MAX], rdst[PATH_MAX];
270 if (!brix_resolved_relative_to_root(log, root_canon, src_resolved,
271 rsrc, sizeof(rsrc))
272 || !brix_resolved_relative_to_root(log, root_canon, dst_resolved,
273 rdst, sizeof(rdst)))
274 {
275 return -1;
276 }
277 return noreplace ? brix_imp_rename_noreplace(rsrc, rdst)
278 : brix_imp_rename(rsrc, rdst);
279 }
280
281 src_parentfd = brix_open_confined_parent_canon(log, root_canon,
282 src_resolved, src_base,
283 sizeof(src_base));
284 if (src_parentfd < 0) {
285 return -1;
286 }
287
288 dst_parentfd = brix_open_confined_parent_canon(log, root_canon,
289 dst_resolved, dst_base,
290 sizeof(dst_base));
291 if (dst_parentfd < 0) {
292 close(src_parentfd);
293 return -1;
294 }
295
296 if (noreplace) {
297 rc = (int) syscall(SYS_renameat2, src_parentfd, src_base,
298 dst_parentfd, dst_base,
299 (unsigned int) RENAME_NOREPLACE);
300 if (rc != 0 && (errno == ENOSYS || errno == EINVAL)) {
301 static int warned = 0;
302 if (!warned) {
303 warned = 1;
304 ngx_log_error(NGX_LOG_WARN, log, errno,
305 "brix: renameat2(RENAME_NOREPLACE) unsupported; "
306 "create-if-absent falls back to non-atomic rename");
307 }
308 rc = renameat(src_parentfd, src_base, dst_parentfd, dst_base);
309 }
310 } else {
311 rc = renameat(src_parentfd, src_base, dst_parentfd, dst_base);
312 }
313 {
314 int e = errno;
315 close(src_parentfd);
316 close(dst_parentfd);
317 errno = e;
318 }
319 return rc;
320}

384/* Confined setattr — utimensat + fchownat under parent confinement.
385 *
386 * Apply timestamps (set_times → utimensat) and/or owner (set_owner →
387 * fchownat) to <resolved>, both *at() syscalls anchored at the confined
388 * parent fd so a symlink/parent swap cannot redirect the change outside the
389 * root. AT_SYMLINK_NOFOLLOW is used so the operation never follows a final
390 * symlink. uid/gid of (uid_t)-1 / (gid_t)-1 leave that id unchanged.
391 * kXR_chmod already covers mode, so mode is intentionally not handled here.
392 * Returns 0 on success, -1 with errno set.
393 */
394int
395brix_setattr_confined_canon(ngx_log_t *log, const char *root_canon,
396 const char *resolved, int set_times, const struct timespec times[2],
397 int set_owner, uid_t uid, gid_t gid)
398{
399 char base[NAME_MAX + 1];
400 int parentfd;
401 int rc = 0;
402 int saved_errno;
403
404 /* Phase 40: route through the broker (as the mapped user) when active. */
405 if (brix_imp_client_active()) {
406 char rel[PATH_MAX];
407 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
408 rel, sizeof(rel)))
409 {
410 return -1;
411 }
412 return brix_imp_setattr(rel, set_times, times, set_owner, uid, gid);
413 }
414
415 parentfd = brix_open_confined_parent_canon(log, root_canon, resolved,
416 base, sizeof(base));
417 if (parentfd < 0) {
418 return -1;
419 }
420 if (set_times && utimensat(parentfd, base, times, AT_SYMLINK_NOFOLLOW) != 0) {
421 rc = -1;
422 }

425 rc = -1;
426 }

428 close(parentfd);
429 errno = saved_errno;
430 return rc;

556 *

558 * lexically-confined) path. When impersonation map mode is active the op

570 * that and keeps the on-disk metadata owned by the right identity.

640/* Confined directory open — broker-routed fd + fdopendir under impersonation.
641 *
642 * open <resolved> as a directory stream that, under impersonation map mode,
643 * is opened BY THE BROKER as the mapped user (O_RDONLY|O_DIRECTORY via
644 * RESOLVE_BENEATH) and handed back as an fd that we fdopendir(). readdir on
645 * that stream then enumerates the directory with the mapped user's read
646 * permission — so a worker that cannot itself opendir a user-private dir
647 * (e.g. a 0700 multipart staging dir owned by the mapped user) can still
648 * list it on the user's behalf. Off impersonation it is a plain opendir().
649 *
650 * WHY: S3 multipart ListParts / AbortMultipartUpload (and the staging-dir cleanup)
651 * enumerate a staging directory created+owned by the mapped user. Done as a
652 * raw worker-side opendir() they fail EACCES under impersonation; routing the
653 * open through the broker fixes that while keeping per-entry removal on the
654 * already-brokered brix_unlink_confined_canon path. Returns a DIR* (caller
655 * closedir()s it) or NULL (errno set). */
656DIR *
657brix_opendir_confined_canon(ngx_log_t *log, const char *root_canon,
658 const char *resolved)
659{
660 char rel[PATH_MAX];
661 int rootfd;
662 DIR *d;
663
664 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
665 rel, sizeof(rel)))
666 {
667 return NULL;
668 }
669
670 if (brix_imp_client_active()) {
671 int fd = brix_imp_open(rel, O_RDONLY | O_DIRECTORY, 0); /* mapped user */
672 if (fd < 0) {
673 return NULL;
674 }
675 d = fdopendir(fd);
676 if (d == NULL) {
677 close(fd);
678 }
679 return d; /* closedir() closes the fd */
680 }
681
682 /*
683 * Off impersonation, open the directory chroot-style under an O_PATH rootfd
684 * (openat2 RESOLVE_IN_ROOT) rather than a bare opendir() on the canonical
685 * path: a bare opendir() follows a trailing in-export symlink with an
686 * outward target (e.g. /_sym_dir -> /etc) straight out of the export root
687 * and enumerates it — a confinement escape. RESOLVE_IN_ROOT confines the
688 * resolution so an escaping target is refused.
689 */
690 rootfd = brix_open_root_fd(log, root_canon);
691 if (rootfd < 0) {
692 return NULL;
693 }
694 d = brix_opendir_beneath(rootfd, rel);
695 close(rootfd);
696 return d;
697}

699/* brix_lstat_confined_canon — lstat()/stat() a path *as the mapped user* under
700 * impersonation, else a bare lstat/stat. WHY: recursive walks (COPY/MOVE
701 * collections, S3 multipart, remove-tree) lstat children of a directory owned
702 * 0700 by the mapped user; a raw worker lstat would EACCES on the parent's
703 * search bit. Routes through the broker (which stat()s as the mapped user) so
704 * the walk sees what that user can see. nofollow!=0 → lstat semantics (do not
705 * follow a trailing symlink — confinement: never resolve a link out of the
706 * export). Returns 0 on success, -1 on error (errno set). */
707int
708brix_lstat_confined_canon(ngx_log_t *log, const char *root_canon,
709 const char *resolved, struct stat *st, int nofollow)
710{
711 char rel[PATH_MAX];
712 int rootfd;
713 int rc;
714
715 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
716 rel, sizeof(rel)))
717 {
718 return -1;
719 }
720
721 if (brix_imp_client_active()) {
722 return brix_imp_stat(rel, st, nofollow); /* as mapped user */
723 }
724
725 /*
726 * Off impersonation, resolve chroot-style under an O_PATH rootfd via openat2
727 * RESOLVE_IN_ROOT (brix_{,l}stat_beneath) rather than a bare stat()/lstat()
728 * on the canonical path. A bare follow-stat() dereferences a trailing
729 * symlink against the REAL filesystem, so a planted in-export link with an
730 * outward absolute target (e.g. /_sym -> /etc/passwd) would be followed
731 * straight out of the export root — a confinement escape. RESOLVE_IN_ROOT
732 * resolves "/" and ".." relative to rootfd, so an outward target lands on a
733 * non-existent in-root path (ENOENT) instead, which the stat handler's
734 * realpath-confined fallback then rejects. nofollow=1 still reports a
735 * trailing link as itself (O_PATH|O_NOFOLLOW) without following it.
736 */
737 rootfd = brix_open_root_fd(log, root_canon);
738 if (rootfd < 0) {
739 return -1;
740 }
741 rc = nofollow ? brix_lstat_beneath(rootfd, rel, st)
742 : brix_stat_beneath(rootfd, rel, st);
743 close(rootfd);
744 return rc;
745}
