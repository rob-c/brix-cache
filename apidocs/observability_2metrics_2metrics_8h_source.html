112/*
113 * Why the stale-dirty cache reaper removed a cached file — the `reason` label on
114 * brix_cache_dirty_reaped_total. Derived per file from its .cinfo write-back
115 * state (brix_cache_cinfo_state):
116 * ABANDONED — DIRTY, aged past brix_cache_dirty_max_age, and NEVER written
117 * back (flush_gen==0): the un-flushed bytes are discarded → loss.
118 * INCOMPLETE — DIRTY, aged, but the file HAD a prior successful write-back
119 * (flush_gen>0) and was re-dirtied: only the trailing dirty
120 * episode is discarded (earlier data reached the origin).
121 * COMPLETED — CLEAN and fully written back (flush_gen>0), last flush aged out:
122 * a finished write-back staging copy reclaimed with NO data loss.
123 * (A read-through fill has flush_gen==0 and is left for eviction.)
124 * Keep COUNT last; it sizes the per-reason counter array below.
125 */
126typedef enum {
127 BRIX_CACHE_REAP_ABANDONED = 0,
128 BRIX_CACHE_REAP_INCOMPLETE,
129 BRIX_CACHE_REAP_COMPLETED,
130 BRIX_CACHE_REAP_REASON_COUNT
131} brix_cache_reap_reason_t;
