14/* Credentials are few (one per upstream identity); a small fixed, process-wide
15 * table avoids allocation and is scanned linearly by name. Interned strings live
16 * on cf->pool (stable for the process lifetime). */
17#define BRIX_CREDENTIAL_MAX 32
18static brix_credential_t brix_credentials[BRIX_CREDENTIAL_MAX];
19static ngx_uint_t brix_credential_count;

21const brix_credential_t *
22brix_credential_lookup(const char *name)
23{
24 ngx_uint_t i;
25
26 if (name == NULL || name[0] == '\0') {
27 return NULL;
28 }
29 for (i = 0; i < brix_credential_count; i++) {
30 if (brix_credentials[i].name.len == ngx_strlen(name)
31 && ngx_strncmp(brix_credentials[i].name.data, name,
32 brix_credentials[i].name.len) == 0)
33 {
34 return &brix_credentials[i];
35 }
36 }
37 return NULL;
38}

71/* Map a `mode` token to its brix_cred_mode value, or -1 when unrecognised. The
72 * tokens name the delegation strategy the consuming subsystem uses to present
73 * this identity to the upstream (sd.h §"How the per-open credential was
74 * obtained"): "passthrough" replays the user's own credential unmodified (a full
75 * X.509 proxy is a passthrough credential), "exchange" trades it for a backend-
76 * audienced one, "delegate"/"mint" obtain a fresh proxy, "select" forces the
77 * directory-based service credential, "auto" dispatches by the identity's auth
78 * method. */
79static ngx_int_t
80brix_credential_mode_token(const ngx_str_t *tok)
81{
82 static const struct { const char *k; enum brix_cred_mode v; } modes[] = {
83 { "select", BRIX_CRED_SELECT },
84 { "passthrough", BRIX_CRED_PASSTHROUGH },
85 { "exchange", BRIX_CRED_EXCHANGE },
86 { "delegate", BRIX_CRED_DELEGATE },
87 { "mint", BRIX_CRED_MINT },
88 { "auto", BRIX_CRED_AUTO },
89 };
90 ngx_uint_t i;
91
92 for (i = 0; i < sizeof(modes) / sizeof(modes[0]); i++) {
93 if (tok->len == ngx_strlen(modes[i].k)
94 && ngx_strncmp(tok->data, modes[i].k, tok->len) == 0)
95 {
96 return (ngx_int_t) modes[i].v;
97 }
98 }
99 return -1;
100}

102static char *
103brix_credential_line(ngx_conf_t *cf, ngx_command_t *dummy, void *conf)
104{
105 brix_credential_t *cred = conf;
106 ngx_str_t *value = cf->args->elts;
107 const brix_credential_field_t *fld;
108
109 /* `mode` is not a str/flag field — it parses a fixed token set into an enum. */
110 if (value[0].len == 4 && ngx_strncmp(value[0].data, "mode", 4) == 0) {
111 ngx_int_t m;
112
113 if (cf->args->nelts != 2) {
114 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
115 "brix_credential: \"mode\" takes exactly one argument");
116 return NGX_CONF_ERROR;
117 }
118 m = brix_credential_mode_token(&value[1]);
119 if (m < 0) {
120 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
121 "brix_credential: unknown mode \"%V\" (want select|passthrough|"
122 "exchange|delegate|mint|auto)", &value[1]);
123 return NGX_CONF_ERROR;
124 }
125 cred->mode = m;
126 return NGX_CONF_OK;
127 }
128
129 for (fld = brix_credential_fields; fld->key != NULL; fld++) {
130 if (value[0].len != ngx_strlen(fld->key)
131 || ngx_strncmp(value[0].data, fld->key, value[0].len) != 0)
132 {
133 continue;
134 }
135
136 if (cf->args->nelts != 2) {
137 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
138 "brix_credential: \"%V\" takes exactly one argument",
139 &value[0]);
140 return NGX_CONF_ERROR;
141 }
142
143 if (fld->flag_off != (size_t) -1) {
144 ngx_flag_t *flag = (ngx_flag_t *) ((char *) cred + fld->flag_off);
145
146 if (value[1].len == 2 && ngx_strncmp(value[1].data, "on", 2) == 0) {
147 *flag = 1;
148 } else if (value[1].len == 3
149 && ngx_strncmp(value[1].data, "off", 3) == 0) {
150 *flag = 0;
151 } else {
152 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
153 "brix_credential: \"%V\" expects on|off", &value[0]);
154 return NGX_CONF_ERROR;
155 }
156 return NGX_CONF_OK;
157 }
158
159 {
160 ngx_str_t *dst = (ngx_str_t *) ((char *) cred + fld->str_off);
161
162 dst->data = (u_char *) brix_pstrdup_z(cf->pool, &value[1]);
163 if (dst->data == NULL) {
164 return NGX_CONF_ERROR;
165 }
166 dst->len = value[1].len;
167 }
168 return NGX_CONF_OK;
169 }
170
171 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
172 "brix_credential: unknown directive \"%V\"", &value[0]);
173 return NGX_CONF_ERROR;
174}

176char *
177brix_conf_credential_block(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
178{
179 ngx_str_t *value = cf->args->elts;
180 brix_credential_t *cred;
181 ngx_conf_t save;
182 char *rv;
183 ngx_flag_t dup_in_this_config = 0;
184
185 /* Dedup by name (last write wins), so a reload re-parsing the whole conf
186 * updates the credential in place rather than appending or erroring.
187 *
188 * A name defined AGAIN within the SAME config parse (same cf->cycle) is a
189 * genuine duplicate, NOT a reload update: the second block silently zeroes
190 * and overrides the first, so any field the first set (e.g. x509_proxy) is
191 * lost if the second omits it. brix_credential is a single GLOBAL,
192 * name-keyed registry — a block in stream{} and another in http{}/conf.d
193 * with the same name collapse to ONE entry. This exact shape (stream block
194 * with x509_proxy, http/conf.d block with x509_cert+x509_key) silently broke
195 * remote-origin GSI auth for hours; warn loudly so it never recurs. A
196 * reload re-parse has a different cf->cycle, so it does NOT warn. */
197 {
198 char name_z[256];
199 const brix_credential_t *existing;
200
201 ngx_cpystrn((u_char *) name_z, value[1].data,
202 ngx_min(value[1].len + 1, sizeof(name_z)));
203 existing = brix_credential_lookup(name_z);
204 if (existing != NULL) {
205 cred = (brix_credential_t *) existing;
206 dup_in_this_config = (existing->last_def_cycle == (void *) cf->cycle);
207 } else if (brix_credential_count >= BRIX_CREDENTIAL_MAX) {
208 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
209 "brix_credential: too many credential blocks (max %d)",
210 BRIX_CREDENTIAL_MAX);
211 return NGX_CONF_ERROR;
212 } else {
213 cred = &brix_credentials[brix_credential_count++];
214 }
215 }
216
217 if (dup_in_this_config) {
218 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
219 "brix_credential \"%V\" is defined more than once in this "
220 "configuration. brix_credential is a single global name-keyed "
221 "registry (shared across stream{} and http{}), so the LAST "
222 "definition silently overrides the earlier one(s) — fields the "
223 "earlier block set are LOST if this block omits them. Make every "
224 "\"brix_credential %V { ... }\" block IDENTICAL, or give them "
225 "distinct names.", &value[1], &value[1]);
226 }
227
228 ngx_memzero(cred, sizeof(*cred));
229 cred->mode = NGX_CONF_UNSET; /* distinguish "unset" from an explicit select */
230 cred->name.data = (u_char *) brix_pstrdup_z(cf->pool, &value[1]);
231 if (cred->name.data == NULL) {
232 return NGX_CONF_ERROR;
233 }
234 cred->name.len = value[1].len;
235 cred->last_def_cycle = (void *) cf->cycle; /* stamp this parse's cycle */
236
237 save = *cf;
238 cf->handler = brix_credential_line;
239 cf->handler_conf = (char *) cred;
240
241 rv = ngx_conf_parse(cf, NULL);
242
243 *cf = save;
244 return rv;
245}

247/* ---- NUL-terminated view of a credential field, or NULL when unset ----
248 *
249 * WHAT: Returns s->data as a C string when the field is set, NULL otherwise.
250 *
251 * WHY: The backend-cred struct wants "unset" as NULL (the registry setter
252 * treats NULL as clear-to-empty); credential fields are interned with a
253 * trailing NUL (brix_pstrdup_z), so the cast is safe.
254 *
255 * HOW: 1. len > 0 ⇒ the interned, NUL-terminated data pointer; else NULL.
256 */
257static const char *
258brix_credential_str_or_null(const ngx_str_t *s)
259{
260 return (s->len > 0) ? (const char *) s->data : NULL;
261}

263/* ---- THE credential_t → backend_cred_t mapper (P80.1) ----
264 *
265 * WHAT: Maps every consumable field of a brix_credential into a
266 * brix_vfs_backend_cred_t for brix_vfs_backend_set_credential. Returns NGX_OK,
267 * or NGX_ERROR when the credential's token_file is set but unreadable.
268 *
269 * WHY: Four call sites (WebDAV parse, stream parse, S3-front parse, stream
270 * worker replay) each hand-copied this mapping and drifted: the worker replay
271 * dropped bearer + all three s3 fields, and because the registry setter
272 * unconditionally overwrites all 8 slots, the parse-time S3 keys were wiped to
273 * "" on every worker spawn (phase-80 finding 1.1). One mapper makes that drift
274 * class impossible.
275 *
276 * HOW:
277 * 1. Zero *out; NULL cred ⇒ anonymous, done.
278 * 2. Derive the bearer (inline token or token_file) into the caller's
279 * bearer_buf; error out if a configured token_file is unreadable.
280 * 3. x509: prefer the proxy PEM (key bundled inside); else cert + key.
281 * 4. Straight str-or-NULL for ca_dir, the three s3 fields, and sss_keytab.
282 */
283ngx_int_t
284brix_credential_to_backend_cred(const brix_credential_t *cred,
285 char *bearer_buf, size_t bearer_cap,
286 struct brix_vfs_backend_cred_s *out, ngx_log_t *log)
287{
288 ngx_memzero(out, sizeof(*out));
289
290 if (cred == NULL) {
291 return NGX_OK; /* anonymous */
292 }
293 if (brix_credential_bearer(cred, bearer_buf, bearer_cap, log) != NGX_OK) {
294 return NGX_ERROR;
295 }
296 out->bearer = (bearer_buf[0] != '\0') ? bearer_buf : NULL;
297
298 if (cred->x509_proxy.len > 0) {
299 out->x509_proxy = (const char *) cred->x509_proxy.data;
300 } else {
301 out->x509_proxy = brix_credential_str_or_null(&cred->x509_cert);
302 out->x509_key = brix_credential_str_or_null(&cred->x509_key);
303 }
304 out->ca_dir = brix_credential_str_or_null(&cred->ca_dir);
305 out->s3_access_key = brix_credential_str_or_null(&cred->s3_access_key);
306 out->s3_secret_key = brix_credential_str_or_null(&cred->s3_secret_key);
307 out->s3_region = brix_credential_str_or_null(&cred->s3_region);
308 out->sss_keytab = brix_credential_str_or_null(&cred->sss_keytab);
309 return NGX_OK;
310}

312ngx_int_t
313brix_credential_bearer(const brix_credential_t *cred, char *out, size_t cap,
314 ngx_log_t *log)
315{
316 int fd;
317 ssize_t n;
318 size_t i;
319
320 if (cap == 0) {
321 return NGX_ERROR;
322 }
323 out[0] = '\0';
324
325 if (cred == NULL) {
326 return NGX_OK; /* anonymous */
327 }
328 if (cred->token.len > 0) {
329 ngx_cpystrn((u_char *) out, cred->token.data,
330 ngx_min(cred->token.len + 1, cap));
331 return NGX_OK;
332 }
333 if (cred->token_file.len == 0) {
334 return NGX_OK; /* no bearer configured */
335 }
336
337 /* Read the bearer from token_file (a trusted config path, not export data:
338 * O_NOFOLLOW so a planted symlink cannot redirect it). First line only. */
339 fd = open((const char *) cred->token_file.data, /* vfs-seam-allow: config-domain bearer token file (not export storage) */
340 O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
341 if (fd < 0) {
342 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
343 "brix_credential: cannot open token_file \"%V\"",
344 &cred->token_file);
345 return NGX_ERROR;
346 }
347 n = read(fd, out, cap - 1);
348 close(fd);
349 if (n < 0) {
350 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
351 "brix_credential: cannot read token_file \"%V\"",
352 &cred->token_file);
353 out[0] = '\0';
354 return NGX_ERROR;
355 }
356 out[n] = '\0';
357 for (i = 0; i < (size_t) n; i++) { /* trim at the first CR/LF/space */
358 if (out[i] == '\r' || out[i] == '\n' || out[i] == ' '
359 || out[i] == '\t')
360 {
361 out[i] = '\0';
362 break;
363 }
364 }
365 return NGX_OK;
366}
