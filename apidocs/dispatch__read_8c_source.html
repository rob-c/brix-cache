8static ngx_int_t

23/* Gate macros: check auth (and bound for non-read ops), then invoke handler.
24 * ctx/c/conf/rc are from the enclosing brix_dispatch_read_opcode scope. */
25#define DISPATCH_RD(fn, ...) \
26 rc = brix_dispatch_require_auth(ctx, c); \
27 if (rc != BRIX_DISPATCH_CONTINUE) { return rc; } \
28 return fn(ctx, c, ##__VA_ARGS__)
29
30#define DISPATCH_RD_BOUND(verb, fn, ...) \
31 rc = brix_dispatch_require_auth(ctx, c); \
32 if (rc != BRIX_DISPATCH_CONTINUE) { return rc; } \
33 rc = brix_reject_bound_nonread_file_op(ctx, c, verb); \
34 if (rc != BRIX_DISPATCH_CONTINUE) { return rc; } \
35 return fn(ctx, c, ##__VA_ARGS__)

37/* brix_dispatch_read_opcode — phase 2 routing (from handshake/dispatch.c) for the
38 * non-mutating opcodes: a switch on ctx->recv.cur_reqid over stat/statx, open/read/close,
39 * dirlist/query/locate, readv/pgread, fattr, prepare, and clone. Each case passes
40 * the DISPATCH_RD gate (brix_dispatch_require_auth) — and DISPATCH_RD_BOUND adds
41 * the bound-stream check for handle-affecting ops — before its handler. Returns the
42 * handler result, or BRIX_DISPATCH_CONTINUE when not a read opcode. */
43ngx_int_t
