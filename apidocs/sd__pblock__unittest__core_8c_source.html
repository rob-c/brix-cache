49
50 /* zero-copy parity: a real blob fd is sendfile-able */

53 pb_close(o);
54

269typedef struct {
270 brix_sd_instance_t *inst;
271 int id;
272 int ops;
273 int ok;
274} thread_arg;
275
276/* Each thread writes + reads back its own private files through the SHARED
277 * instance (shared SQLite connection + blob dir), proving the per-worker handle
278 * is safe across a thread pool and that blob I/O runs in parallel. */
279static void *
280thread_body(void *p)
281{
282 thread_arg *a = p;
283 int i;
284
285 for (i = 0; i < a->ops; i++) {
286 char path[64];
287 char buf[32];
288 char want[32];
289 int wn = snprintf(want, sizeof(want), "t%d-op%d", a->id, i);
290
291 snprintf(path, sizeof(path), "/conc/t%d-%d", a->id, i);
292 if (write_file(a->inst, path, want, (size_t) wn) != 0) {
293 return NULL;
294 }
295 if (read_file(a->inst, path, buf, sizeof(buf)) != wn
296 || memcmp(buf, want, (size_t) wn) != 0)
297 {
298 return NULL;
299 }
300 }
301 a->ok = 1;
302 return NULL;
303}

305void
306test_threads(brix_sd_instance_t *inst)
307{
308 enum { NTHREADS = 8, NOPS = 40 };
309 pthread_t tids[NTHREADS];
310 thread_arg args[NTHREADS];
311 int i;
312
313 CHECK(D->mkdir(inst, "/conc", 0755) == NGX_OK, "mkdir conc");
314 for (i = 0; i < NTHREADS; i++) {
315 args[i].inst = inst;
316 args[i].id = i;
317 args[i].ops = NOPS;
318 args[i].ok = 0;
319 CHECK(pthread_create(&tids[i], NULL, thread_body, &args[i]) == 0,
320 "pthread_create %d", i);
321 }
322 for (i = 0; i < NTHREADS; i++) {
323 pthread_join(tids[i], NULL);
324 CHECK(args[i].ok == 1, "thread %d failed", i);
325 }
326}

328/* Each forked child opens its OWN instance on the same root and writes a batch;
329 * the parent then verifies every child's files landed. Exercises cross-process
330 * WAL locking + the busy-timeout retry path. */
331void
332test_processes(const char *root, brix_sd_instance_t *inst)
333{
334 enum { NPROC = 4, NOPS = 25 };
335 pid_t pids[NPROC];
336 int i, total = 0, rc;
337 brix_sd_dir_t *dir;
338 brix_sd_dirent_t de;
339 int err = 0;
340
341 CHECK(D->mkdir(inst, "/mp", 0755) == NGX_OK, "mkdir mp");
342
343 for (i = 0; i < NPROC; i++) {
344 pid_t pid = fork();
345
346 if (pid == 0) {
347 brix_sd_instance_t cinst = {0};
348 brix_sd_pblock_conf_t conf = { root, 5000, 0 };
349 int j, bad = 0;
350
351 cinst.driver = D;
352 if (D->init(&cinst, &conf) != NGX_OK) {
353 _exit(2);
354 }
355 for (j = 0; j < NOPS; j++) {
356 char path[64];
357
358 snprintf(path, sizeof(path), "/mp/p%d-%d", (int) getpid(), j);
359 if (write_file(&cinst, path, "x", 1) != 0) {
360 bad = 1;
361 break;
362 }
363 }
364 D->cleanup(&cinst);
365 _exit(bad ? 1 : 0);
366 }
367 CHECK(pid > 0, "fork %d", i);
368 pids[i] = pid;
369 }
370
371 for (i = 0; i < NPROC; i++) {
372 int status = 0;
373
374 waitpid(pids[i], &status, 0);
375 CHECK(WIFEXITED(status) && WEXITSTATUS(status) == 0,
376 "child %d exit status %d", i, status);
377 }
378
379 dir = D->opendir(inst, "/mp", &err);
380 CHECK(dir != NULL, "opendir mp");
381 if (dir != NULL) {
382 while ((rc = D->readdir(dir, &de)) == NGX_OK) {
383 total++;
384 }
385 D->closedir(dir);
386 }
387 CHECK(total == NPROC * NOPS, "expected %d files, found %d", NPROC * NOPS,
388 total);
389}
