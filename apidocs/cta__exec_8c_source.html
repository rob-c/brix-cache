1/*
2 * cta_exec.c — CTA request executor. See cta_exec.h.
3 *
4 * The test executor walks the lifecycle deterministically and emits a progress
5 * alert at each step, so the whole SSI archive/retrieve flow (waitresp → alerts →
6 * response) is exercised without a tape system. The production executor is the
7 * documented seam where archive/retrieve hook the real tier/frm engine.
8 */
9
10#include "cta_exec.h"
11#include <stddef.h>

13static void
14emit(cta_progress_t *p, const char *msg)
15{
16 if (p != NULL && p->alert != NULL) {
17 p->alert(p->ctx, msg);
18 }
19}

21/* Advance through QUEUED → ACTIVE → COMPLETE, alerting at each step. */
22static int
23sim_lifecycle(cta_req_t *e, cta_progress_t *p, const char *queued_msg,
24 const char *active_msg)
25{
26 if (cta_queue_transition(e, CTA_ST_QUEUED) != 0) {
27 return -1;
28 }
29 emit(p, queued_msg);
30 if (cta_queue_transition(e, CTA_ST_ACTIVE) != 0) {
31 return -1;
32 }
33 emit(p, active_msg);
34 if (cta_queue_transition(e, CTA_ST_COMPLETE) != 0) {
35 return -1;
36 }
37 return 0;
38}

68/*
69 * Production executor seam. In a build without a configured nearline/tape backend
70 * archive and retrieve cannot do real work, so they fail cleanly. The real wiring
71 * (retrieve → fs/tier + fs/xfer/stage_engine + frm to recall an object online;
72 * archive → tape-write hook) lands here when a NEARLINE backend exists; it must
73 * drive the same cta_queue transitions + progress alerts as the test executor.
74 */
75static int
76prod_archive(cta_req_t *e, cta_progress_t *p)
77{
78 (void) e;
79 emit(p, "no nearline backend configured");

81}

101int
102cta_exec_run(const cta_exec_vtbl_t *vt, cta_req_t *e, cta_progress_t *p)
103{
104 switch (e->req.op) {
105 case CTA_OP_ARCHIVE:
106 return vt->archive(e, p);
107 case CTA_OP_RETRIEVE:
108 return vt->retrieve(e, p);
109 case CTA_OP_CANCEL:
110 return vt->cancel(e);
111 case CTA_OP_QUERY:
112 case CTA_OP_UNKNOWN:
113 default:
114 return -1; /* not an executor op (query is handled by the service) */
115 }
116}
