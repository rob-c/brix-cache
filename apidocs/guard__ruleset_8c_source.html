1/*
2 * guard_ruleset.c — guard ruleset construction.
3 *
4 * WHAT: builders that populate a guard_ruleset_t: zero-init, the built-in
5 * junk-scanner signature set, operator-supplied signatures/prefixes, and
6 * per-profile grammar defaults ("arc" | "xrdhttp" | "root").
7 * WHY: adapters (nginx http module, stream relay) assemble rulesets at config
8 * time from directives; the assembly logic itself is protocol-agnostic and
9 * lives here, next to the classifier that consumes it.
10 * HOW: pure C, no allocation — patterns are borrowed pointers that must
11 * outlive the ruleset (string literals or nginx conf-pool strings).
12 */
13#include "guard.h"
14#include <string.h>

16/* ---- Zero a ruleset ----
17 *
18 * WHAT: resets every field of *rs to zero (no signatures, no prefixes, no ops
19 * allowed, grammar not enforced, outcome flags off).
20 *
21 * WHY: gives adapters one canonical empty state to build on, so a
22 * half-initialized ruleset can never classify.
23 *
24 * HOW: 1. memset the whole struct.
25 */
26void
27guard_ruleset_init(guard_ruleset_t *rs)
28{
29 memset(rs, 0, sizeof(*rs));
30}

32/* ---- Add one signature to the blocklist ----
33 *
34 * WHAT: appends (kind, pat, pat_len) to rs->sigs. Returns 1 on success, 0 if
35 * the fixed table is full.
36 *
37 * WHY: operator directives and the built-in defaults share one append path,
38 * with a hard cap instead of allocation (pure-C core rule).
39 *
40 * HOW: 1. Refuse when n_sigs is at GUARD_MAX_SIGS.
41 * 2. Store the borrowed pattern pointer + length and bump the count.
42 */
43int
44guard_ruleset_add_signature(guard_ruleset_t *rs, guard_sig_kind_t kind,
45 const char *pat, size_t pat_len)
46{
47 if (rs->n_sigs >= GUARD_MAX_SIGS) {
48 return 0;
49 }
50 rs->sigs[rs->n_sigs].kind = kind;
51 rs->sigs[rs->n_sigs].pat = pat;
52 rs->sigs[rs->n_sigs].pat_len = pat_len;
53 rs->n_sigs++;
54 return 1;
55}

57/* ---- Append the built-in junk-scanner signature set ----
58 *
59 * WHAT: adds the default web-scanner corpus (php/asp/cgi extension probes,
60 * wordpress/cgi-bin/vendor/.git prefixes, .env//../ /phpMyAdmin/.aws
61 * substrings) to rs.
62 *
63 * WHY: these patterns never appear in legitimate grid traffic (ARC REST,
64 * WebDAV data paths, root:// namespaces) but dominate internet scanner
65 * noise — a curated default beats making every operator assemble one.
66 *
67 * HOW: 1. Walk a static descriptor table (string literals — they outlive any
68 * ruleset by definition).
69 * 2. Append each via guard_ruleset_add_signature().
70 */
71void
72guard_ruleset_add_default_signatures(guard_ruleset_t *rs)
73{
74 static const struct { guard_sig_kind_t kind; const char *pat; } defaults[] = {
75 { GUARD_SIG_SUFFIX, ".php" }, { GUARD_SIG_SUFFIX, ".asp" },
76 { GUARD_SIG_SUFFIX, ".aspx" }, { GUARD_SIG_SUFFIX, ".cgi" },
77 { GUARD_SIG_PREFIX, "/wp-" }, { GUARD_SIG_PREFIX, "/cgi-bin" },
78 { GUARD_SIG_PREFIX, "/vendor" },{ GUARD_SIG_PREFIX, "/.git" },
79 { GUARD_SIG_SUBSTR, "/.env" }, { GUARD_SIG_SUBSTR, "phpMyAdmin" },
80 { GUARD_SIG_SUBSTR, "/../" }, { GUARD_SIG_SUBSTR, "/.aws" },
81 { GUARD_SIG_SUBSTR, "/wp-config" },
82 };
83 size_t default_index;
84
85 for (default_index = 0;
86 default_index < sizeof(defaults) / sizeof(defaults[0]);
87 default_index++)
88 {
89 guard_ruleset_add_signature(rs, defaults[default_index].kind,
90 defaults[default_index].pat, strlen(defaults[default_index].pat));
91 }
92}

94/* ---- Add one valid namespace prefix ----
95 *
96 * WHAT: appends `pfx[0..len)` to the grammar prefix table. Returns 1 on
97 * success, 0 if the fixed table is full.
98 *
99 * WHY: operators narrow the guarded namespace per location (an ARC endpoint
100 * root, a WebDAV export root) beyond the profile defaults.
101 *
102 * HOW: 1. Refuse when n_prefixes is at GUARD_MAX_PREFIXES.
103 * 2. Store the borrowed pointer + length and bump the count.
104 */
105int
106guard_ruleset_add_prefix(guard_ruleset_t *rs, const char *pfx, size_t len)
107{
108 if (rs->n_prefixes >= GUARD_MAX_PREFIXES) {
109 return 0;
110 }
111 rs->prefixes[rs->n_prefixes] = pfx;
112 rs->prefix_len[rs->n_prefixes] = len;
113 rs->n_prefixes++;
114 return 1;
115}

117/* ---- Permit a fixed list of op-classes ----
118 *
119 * WHAT: sets op_allowed[op] = 1 for each entry of ops[0..count).
120 *
121 * WHY: profile grammars are op-lists; a tiny shared setter keeps each
122 * profile branch declarative.
123 *
124 * HOW: 1. Walk the list, flip each slot on.
125 */
126static void
127allow_ops(guard_ruleset_t *rs, const guard_op_class_t *ops, int count)
128{
129 int op_index;
130
131 for (op_index = 0; op_index < count; op_index++) {
132 rs->op_allowed[ops[op_index]] = 1;
133 }
134}

136/* ---- Load built-in grammar defaults for a service profile ----
137 *
138 * WHAT: configures rs for "arc" (ARC REST namespaces + job ops), "xrdhttp"
139 * (root-open export, data ops only), or "root" (root:// op-classes incl.
140 * handshake). Unknown profiles get permissive grammar (signatures still
141 * apply). Always enables grammar enforcement + both outcome flags except
142 * in the unknown-profile case, where grammar drops to advisory.
143 *
144 * WHY: sensible per-service defaults mean a bare `profile arc;` directive is
145 * already useful; operators refine with explicit prefix/method directives.
146 *
147 * HOW: 1. Turn on enforcement + notfound/authfail flagging.
148 * 2. Per profile: add namespace prefixes (arc only — xrdhttp/root serve
149 * the export root) and allow that service's op-classes.
150 * 3. Unknown profile: allow every op, drop to advisory grammar.
151 */
152void
153guard_ruleset_load_profile(guard_ruleset_t *rs, const char *profile)
154{
155 rs->enforce_grammar = 1;
156 rs->flag_notfound = 1;
157 rs->flag_authfail = 1;
158
159 if (strcmp(profile, "arc") == 0) {
160 static const guard_op_class_t arc_ops[] = {
161 GUARD_OP_READ, GUARD_OP_WRITE, GUARD_OP_LIST, GUARD_OP_DELETE,
162 GUARD_OP_JOBCTL, GUARD_OP_STAGE, GUARD_OP_INFO, GUARD_OP_DELEG };
163 guard_ruleset_add_prefix(rs, "/arex", 5);
164 guard_ruleset_add_prefix(rs, "/rest", 5);
165 guard_ruleset_add_prefix(rs, "/datadelivery", 13);
166 allow_ops(rs, arc_ops, (int) (sizeof(arc_ops) / sizeof(arc_ops[0])));
167 } else if (strcmp(profile, "xrdhttp") == 0) {
168 static const guard_op_class_t dav_ops[] = {
169 GUARD_OP_READ, GUARD_OP_WRITE, GUARD_OP_LIST, GUARD_OP_DELETE,
170 GUARD_OP_INFO };
171 /* XrdHttp/WebDAV serves the export root; operator narrows via
172 * brix_guard_valid_prefix. Default: root-open but grammar on ops. */
173 allow_ops(rs, dav_ops, (int) (sizeof(dav_ops) / sizeof(dav_ops[0])));
174 } else if (strcmp(profile, "root") == 0) {
175 static const guard_op_class_t root_ops[] = {
176 GUARD_OP_READ, GUARD_OP_WRITE, GUARD_OP_LIST, GUARD_OP_DELETE,
177 GUARD_OP_INFO, GUARD_OP_STAGE, GUARD_OP_HANDSHAKE };
178 allow_ops(rs, root_ops,
179 (int) (sizeof(root_ops) / sizeof(root_ops[0])));
180 } else {
181 /* unknown profile: permissive grammar, signatures still apply */
182 int op_index;
183 for (op_index = 0; op_index <= GUARD_OP_UNKNOWN; op_index++) {
184 rs->op_allowed[op_index] = 1;
185 }
186 rs->enforce_grammar = 0;
187 }
188}
