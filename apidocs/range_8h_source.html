20/*
21 * WHAT: Parse an RFC 7233 "bytes=" Range header value into brix_http_range_t.
22 *
23 * WHY: WebDAV GET and S3 GET handlers both need to honour HTTP Range headers for partial content
24 * responses. Delegates complex parsing logic to range_vector.c while providing a simple API
25 * that returns satisfiability status (416 when the requested range exceeds file size).
26 *
27 * HOW: Set defaults — present=0, full-file range. Reject NULL or short headers not starting with
28 * "bytes=". Delegate actual parsing to brix_http_parse_range_vector() in range_vector.c
29 * (single-range mode, max_ranges=1). If vector parser succeeds → copy start/end into out.
30 * If it fails → present=1, satisfiable=0. Caller checks these fields: 416 when
31 * present==1 && satisfiable==0.
32 *
33 * Input: hdr_val/hdr_len — raw Range header value (e.g. "bytes=0-499"). Pass NULL/0 if no
34 * Range header is present. file_size — st_size of the file for clamping/satisfiability.
35 */
36void brix_http_parse_range(const unsigned char *hdr_val, size_t hdr_len,
37 off_t file_size, brix_http_range_t *out);
