5/*
6 * WHAT: Configure the Prometheus metrics shared-memory zone and assign per-listener slots.
7 * WHY: All server blocks share a single atomic counters region so workers can increment
8 * counters lock-free via ngx_atomic_t fields. Each listener gets a deterministic slot
9 * number that becomes a stable Prometheus label source (low-cardinality invariant).
10 * HOW: Add shared memory zone "brix_metrics" sized via brix_shm_zone_size() so the table
11 * can be slab-allocated without clobbering the slab-pool header; register the shm_init
12 * callback; iterate cmcf->servers to assign slots 0..N to enabled listeners. Returns
13 * NGX_OK or NGX_ERROR on allocation failure.
14 */

16/*
17 * Create (or reuse) the module's metrics SHM zone. Idempotent per config

42ngx_int_t
43brix_configure_metrics(ngx_conf_t *cf, ngx_stream_core_main_conf_t *cmcf)
44{
45 ngx_stream_core_srv_conf_t **cscfp;
46 ngx_stream_brix_srv_conf_t *xcf;
47 ngx_uint_t i;
48 ngx_uint_t slot = 0;
49
50 /*
51 * One shared zone is used for all enabled server blocks. Each server is
52 * assigned a small integer slot; live connections cache that slot and
53 * update counters lock-free via atomics.
54 */
55
56 /*
57 * Size the zone so the metrics table can be allocated FROM the slab pool
58 * (brix_shm_table_alloc) without overwriting the ngx_slab_pool_t header.
59 * Laying the table directly over shm.addr would clobber the slab mutex that
60 * nginx's ngx_unlock_mutexes() force-unlocks on every child death, SIGSEGVing
61 * the master. The helper accounts for the table bytes plus slab overhead.
62 */
63 if (brix_metrics_ensure_zone(cf) != NGX_OK) {
64 return NGX_ERROR;
65 }
66
67 cscfp = cmcf->servers.elts;
68
69 /* Assign deterministic metrics slots to enabled listeners. */
70 for (i = 0; i < cmcf->servers.nelts; i++) {
71 xcf = ngx_stream_conf_get_module_srv_conf(cscfp[i],
72 ngx_stream_brix_module);
73 if (!xcf->common.enable || slot >= BRIX_METRICS_MAX_SERVERS) {
74 continue;
75 }
76
77 /* Slot numbers become stable label sources for the HTTP metrics exporter. */
78 xcf->metrics_slot = (ngx_int_t) slot++;
79 }
80
81 return NGX_OK;
82}

84/*
85 * WHAT: Initialize (or reuse) the shared memory zone containing Prometheus counters.
86 * WHY: nginx reloads preserve existing shared-memory mappings; without this logic every
87 * config reload would zero live counters, losing historical metrics data. Additionally
88 * the table MUST live in slab memory — not directly over shm.addr — so the
89 * ngx_slab_pool_t header that nginx's ngx_unlock_mutexes() relies on survives every
90 * child exit (otherwise the master SIGSEGVs whenever any worker dies).
91 * HOW: Delegate the fresh-alloc / reload / re-attach lifecycle to brix_shm_table_alloc,
92 * which allocates the ngx_brix_metrics_t from the slab pool, zeroes it on a brand-new
93 * allocation, and publishes it via shm_zone->data. The metrics table is lock-less
94 * (atomic counters only) so NULL is passed for the mutex argument. There are no
95 * fresh-only field inits beyond the zeroing the helper already performs.
96 */

98/*
99 * fnv1a64_file — FNV-1a 64-bit hash of a file's bytes (config fingerprint).
100 *
101 * Reads the file in fixed chunks so an arbitrarily large (included) config never
102 * needs to be held whole in memory. Returns 0 on any open/read failure — the
103 * caller treats 0 as "fingerprint unavailable", never as a real digest. Note
104 * this fingerprints the top-level config file only; `include`d files are not
105 * folded in (documented in reload-semantics.md), so config_generation, not the
106 * hash, is the authoritative "a reload happened" signal.
107 */
108static uint64_t
109fnv1a64_file(ngx_str_t *path, ngx_log_t *log)
110{
111 u_char buf[4096];
112 uint64_t h = BRIX_FNV1A64_OFFSET_BASIS;
113 ngx_fd_t fd;
114 ssize_t n;
115 size_t i;
116
117 if (path == NULL || path->len == 0) {
118 return 0;
119 }
120
121 fd = ngx_open_file(path->data, NGX_FILE_RDONLY, NGX_FILE_OPEN, 0);
122 if (fd == NGX_INVALID_FILE) {
123 ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
124 "brix: cannot read config \"%V\" for version hash",
125 path);
126 return 0;
127 }
128
129 for ( ;; ) {
130 n = ngx_read_fd(fd, buf, sizeof(buf));
131 if (n <= 0) {
132 break;
133 }
134 for (i = 0; i < (size_t) n; i++) {
135 h ^= (uint64_t) buf[i];
136 h *= BRIX_FNV1A64_PRIME;
137 }
138 }
139
140 (void) ngx_close_file(fd);
141 return h;
142}

145void
146brix_config_version_publish(ngx_cycle_t *cycle)
147{
148 ngx_brix_metrics_t *tbl;
149 ngx_atomic_uint_t gen;
150 uint64_t hash;
151
152 /*
153 * No metrics zone means no enabled stream server block — nothing to publish
154 * and nothing to probe. The zone's ->data is the slab-allocated table once
155 * the master has mapped it (before this init_module hook runs).
156 */
157 if (ngx_brix_shm_zone == NULL || ngx_brix_shm_zone->data == NULL) {
158 return;
159 }
160 tbl = ngx_brix_shm_zone->data;
161
162 hash = fnv1a64_file(&cycle->conf_file, cycle->log);
163
164 /*
165 * init_module runs exactly once per config load in the master, so a plain
166 * atomic increment counts loads precisely (the metrics zone re-attaches
167 * across reload, so the previous value is preserved). The hash is written
168 * by this single master before any worker forks off the new cycle.
169 */
170 gen = ngx_atomic_fetch_add(&tbl->config_generation, 1) + 1;
171 tbl->config_hash = hash;
172
173 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
174 "brix: config generation %ui live (pid %P, version %016xL)",
175 (ngx_uint_t) gen, ngx_pid, hash);
176}
