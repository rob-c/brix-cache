1/*
2 * tables.c — identity-table lookups + table lifetime (XrdAccAccess_Tables).
3 *
4 * WHAT: brix_acc_named_find() resolves a name (user/group/host/org/role/
5 * netgroup/template) to its capability list by exact match;
6 * brix_acc_domain_find() resolves a host to a domain capability list by
7 * SUFFIX match (a `.cern.ch` record matches any host ending in `.cern.ch`).
8 * brix_acc_tables_free() releases an entire table generation.
9 *
10 * WHY: faithful ports of XrdOucHash::Find (exact) and XrdAccCapName::Find
11 * (suffix). The category tables are modelled as singly-linked name lists
12 * rather than hashes — authdb files are small and authorization results are
13 * memoized by the existing auth cache, so O(rules) lookup is immaterial and
14 * keeps the build + atomic refresh swap trivial (one pool per generation).
15 *
16 * HOW: linear scan with ngx_strcmp for exact lookups; for domains, compare the
17 * record name against the tail of the host name of equal length.
18 */

22brix_acc_cap_t *
23brix_acc_named_find(brix_acc_named_t *list, const char *name)
24{
25 if (name == NULL) {
26 return NULL;
27 }
28 for (; list != NULL; list = list->next) {
29 if (ngx_strcmp(list->name, name) == 0) {
30 return list->caps;
31 }
32 }
33 return NULL;
34}

36brix_acc_cap_t *
37brix_acc_domain_find(brix_acc_named_t *dlist, const char *host)
38{
39 int hlen;
40
41 if (host == NULL) {
42 return NULL;
43 }
44 hlen = (int) ngx_strlen(host);
45
46 /* A domain record (e.g. ".cern.ch") matches when it is a suffix of the
47 * host of equal length (XrdAccCapName::Find). */
48 for (; dlist != NULL; dlist = dlist->next) {
49 if (dlist->nlen <= hlen
50 && ngx_strcmp(dlist->name, host + (hlen - dlist->nlen)) == 0)
51 {
52 return dlist->caps;
53 }
54 }
55 return NULL;
56}
