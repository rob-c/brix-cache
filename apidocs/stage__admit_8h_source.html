4/*
5 * stage_admit.h — two-tier write-back-staging backpressure decision.
6 *
7 * WHAT: The admission verdict for a new WRITE (root:// write-open, WebDAV/S3 PUT)
8 * based on how full the write-back staging filesystem is.
9 *
10 * WHY: Write-through mirrors each write into a durable staging copy before the
11 * FRM-journaled flush to the origin. If the origin is slow/down, staging
12 * grows; unbounded, it fills the disk. Backpressure sheds new writes — first
13 * gently (delay → client retries), then hard (reject) — so staging stays
14 * bounded. Reads never consume staging and are never throttled.
15 *
16 * HOW: The pure band logic (brix_wt_stage_decide) is kept header-only and
17 * nginx-free so the unit test links it directly (same split as
18 * cache_fs_sampler.h). The nginx-typed entry point brix_wt_stage_admit()
19 * — which samples the staging filesystem and applies this logic to a server
20 * conf — is declared in cache_internal.h.
21 */
22
23/* Soft-band retry hint handed to a delayed write: kXR_wait seconds (root://) and
24 * the HTTP Retry-After (WebDAV/S3). Short so a draining stage recovers promptly. */
25#define BRIX_WT_STAGE_WAIT_SECS 5

27/* ALLOW: accept the write. WAIT: soft band — delay (kXR_wait / 503 Retry-After),
28 * the client retries. REJECT: hard cap — refuse now (kXR_Overloaded / 429). */
29typedef enum {
30 BRIX_WT_ADMIT_ALLOW = 0,
31 BRIX_WT_ADMIT_WAIT,
32 BRIX_WT_ADMIT_REJECT

35/*
36 * brix_wt_stage_decide — the pure two-tier band over staging occupancy.
37 *
38 * high_ppm == 0 → backpressure disabled → ALLOW (any occupancy).
39 * occ >= high_ppm → REJECT (hard cap).
40 * low_ppm <= occ < high → WAIT (soft band; LOW boundary inclusive).
41 * occ < low_ppm → ALLOW.
42 *
43 * All values are filesystem occupancy in parts-per-million (0-1000000).
44 */
45static inline brix_wt_admit_t
46brix_wt_stage_decide(unsigned occ_ppm, unsigned low_ppm, unsigned high_ppm)
47{
48 if (high_ppm == 0) {
49 return BRIX_WT_ADMIT_ALLOW; /* disabled */
50 }
51 if (occ_ppm >= high_ppm) {
52 return BRIX_WT_ADMIT_REJECT;
53 }
54 if (occ_ppm >= low_ppm) {
55 return BRIX_WT_ADMIT_WAIT;
56 }
57 return BRIX_WT_ADMIT_ALLOW;
58}
