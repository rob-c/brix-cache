1/*
2 * process_timers.c — the self-arming per-worker maintenance timers.
3 *
4 * WHAT: Owns every maintenance-timer callback (CRL hot-reload, CMS
5 * pending-locate reaper, upload stage-out reaper, cache stale-dirty
6 * reaper, async stage-flush scheduler), their static ngx_event_t globals,
7 * and the three timer-arming entry points the init_process orchestrator
8 * calls. Each static timer event is co-located with BOTH its handler and
9 * its arming function so the global never crosses a file boundary.
10 * WHY: Split (phase-79 file-size cap) out of the former 925-line process.c.
11 * Grouping the timer machinery keeps the "arm -> fire -> re-arm" lifecycle
12 * of each maintenance timer reviewable in one place. brix_crl_reload_handler
13 * and brix_cache_reap_handler are non-static (declared in
14 * process_internal.h) because a server's init ladder in
15 * process_server_init.c arms those two per-server; the three arming
16 * functions are non-static because init_process (process.c) calls them.
17 * HOW: Every handler re-arms itself with ngx_add_timer unless ngx_exiting, and
18 * marks its timer cancelable so a draining worker is never held past its
19 * real work. No behaviour change from the split.
20 */

32/*
33 * nginx's timer-expiry debug log (ngx_event_expire_timers, compiled in on a
34 * --with-debug nginx and active when a server sets `error_log ... debug`) reads
35 * ngx_event_ident(ev->data) == ((ngx_connection_t *) ev->data)->fd for EVERY
36 * timer it fires. Our worker-init maintenance timers are connection-less, so a
37 * NULL ev->data made that log dereference NULL and SIGSEGV the worker — which
38 * nginx respawns, so it crash-loops (a flood of core dumps that spiked system
39 * load hard under the debug-logging test fleet). Point such timers' ev->data at
40 * this dummy connection (fd = -1) so the debug log has a valid fd to read; the
41 * timer handlers themselves do not use ev->data. Shared (non-static) so the
42 * other connection-less maintenance timers (uring-panic, accesslog flush) point
43 * at the same dummy — see extern declarations at their call sites.
44 */
45ngx_connection_t brix_maint_timer_conn = { .fd = (ngx_socket_t) -1 };

47/* Timer callback: rebuild the GSI X509_STORE from the configured CRL file at the
48 * reload interval, then re-arm the timer. */
49void
50brix_crl_reload_handler(ngx_event_t *ev)
51{
52 ngx_stream_brix_srv_conf_t *xcf = ev->data;
53 struct stat st;
54 int is_reg_file;
55
56 /*
57 * E5: skip the CRL re-parse + store rebuild when the CRL source is an
58 * unchanged regular file, so a large CRL on slow storage never stalls the
59 * event loop on every interval. A CApath DIRECTORY is always rebuilt — its
60 * mtime does not reflect content changes of the CRL files inside it.
61 */
62 is_reg_file = (xcf->crl.len > 0
63 && stat((const char *) xcf->crl.data, &st) == 0
64 && S_ISREG(st.st_mode));
65
66 if (is_reg_file && xcf->crl_mtime != 0 && st.st_mtime == xcf->crl_mtime) {
67 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, ev->log, 0,
68 "brix: CRL \"%s\" unchanged — skipping reload",
69 xcf->crl.data);
70 if (xcf->crl_reload > 0 && !ngx_exiting) {
71 ngx_add_timer(ev, (ngx_msec_t) xcf->crl_reload * 1000);
72 }
73 return;
74 }
75
76 ngx_log_error(NGX_LOG_INFO, ev->log, 0,
77 "brix: CRL reload timer fired, rebuilding store "
78 "from \"%s\"", xcf->crl.data);
79
80 /* NULL scope: the CRL hot-reload timer MUST rebuild from fresh CRLs on
81 * disk, never reuse the memoised startup store. */
82 if (brix_rebuild_gsi_store(xcf, ev->log, NULL) != NGX_OK) {
83 BRIX_DIAG_CRIT(ev->log, 0,
84 "brix: CRL reload failed for \"%s\" — keeping previous store",
85 "the CRL file/dir is unreadable, malformed, or mid-rewrite",
86 "check the path's permissions and that fetch-crl writes atomically; "
87 "the server keeps serving with the LAST-GOOD CRLs, so a newly "
88 "revoked cert may still be accepted until the next good reload",
89 xcf->crl.data);
90 } else if (is_reg_file) {
91 /* Record mtime only on a successful rebuild so a failed load is retried
92 * (not skipped) on the next interval. */
93 xcf->crl_mtime = st.st_mtime;
94 }
95
96 /* Re-arm the timer (suppressed once the worker is exiting). */
97 if (xcf->crl_reload > 0 && !ngx_exiting) {
98 ngx_add_timer(ev, (ngx_msec_t) xcf->crl_reload * 1000);
99 }
100}

126/*
127 * Upload stage-out reaper (worker 0). Finishes any cache->storage commit that
128 * was interrupted by a worker death: the first tick (soon after startup) recovers
129 * complete-but-uncommitted files left from a previous run; periodic ticks retry

153/*

155 * staging files dirty longer than brix_cache_dirty_max_age (the eviction guard
156 * protects them, so without this an abandoned flush leaks disk forever). Runs on
157 * a maintenance timer independent of occupancy. ev->data is the server conf.

161void
162brix_cache_reap_handler(ngx_event_t *ev)
163{
164 ngx_stream_brix_srv_conf_t *xcf = ev->data;
165 ngx_uint_t n = brix_cache_reap_dirty(xcf, ev->log);

167 if (n > 0) {
168 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
169 "brix: cache stale-dirty reaper removed %ui file(s)", n);
170 }
171 if (!ngx_exiting) {
172 ngx_add_timer(ev, BRIX_CACHE_REAP_INTERVAL_MS);

176/* phase-64 SP4: drain the deferred (async) stage-flush queue every second. Armed
177 * in every worker (the queue is per-worker); a no-op when empty. */
178#define BRIX_STAGE_SCHED_MS 1000
179static ngx_event_t brix_stage_sched_timer;
180
181static void
182brix_stage_sched_handler(ngx_event_t *ev)
183{
184 brix_stage_scheduler_tick();
185 /* Same per-worker cadence drains the durable async backend-op queue's TIME
186 * trigger (the SIZE trigger posts its own deferred drain event). The wait_ms
187 * cap is thus honoured at ~1s granularity — a coalescing backstop, not a
188 * precise deadline. */
189 brix_baq_tick();
190 if (!ngx_exiting) {
191 ngx_add_timer(ev, BRIX_STAGE_SCHED_MS);
192 }
193}

195/* ---- Arm the per-worker async stage-flush scheduler timer ----
196 *
197 * WHAT: Installs and starts the 1s brix_stage_sched_timer that drains this
198 * worker's deferred (async) stage-flush queue. No return value — timer arming
199 * cannot fail.
200 *
201 * WHY: The deferred-flush queue is per-worker and protocol-agnostic, so the
202 * timer must run in HTTP-only (WebDAV/S3) workers too; it is therefore armed
203 * BEFORE the stream-config early-return in init_process. The tick is a no-op
204 * when the queue is empty.
205 *
206 * HOW:
207 * 1. Point the static timer event at brix_stage_sched_handler with the
208 * cycle log.
209 * 2. Mark it cancelable so it never delays graceful shutdown.
210 * 3. Arm it at BRIX_STAGE_SCHED_MS.
211 */
212void
213brix_init_stage_sched_timer(ngx_cycle_t *cycle)
214{
215 brix_stage_sched_timer.handler = brix_stage_sched_handler;
216 brix_stage_sched_timer.data = &brix_maint_timer_conn;
217 brix_stage_sched_timer.log = cycle->log;
218 /* cancelable: a background maintenance timer must NOT keep a gracefully
219 * shutting-down worker alive until worker_shutdown_timeout — nginx's
220 * ngx_event_no_timers_left() ignores cancelable timers, so the worker exits
221 * as soon as its real work is done instead of lingering seconds on restart. */
222 brix_stage_sched_timer.cancelable = 1;
223 ngx_add_timer(&brix_stage_sched_timer, BRIX_STAGE_SCHED_MS);
224}

226/* ---- Arm the worker-0 pending-locate reaper timer (managers only) ----
227 *
228 * WHAT: Installs and starts the CMS pending-locate reaper timer when any
229 * server block is a manager and this is worker 0. No return value — timer
230 * arming cannot fail.
231 *
232 * WHY: A4 — abandoned in-flight locate slots must be reclaimed even when
233 * traffic ceases. The registry is a single process-global SHM table, so one
234 * worker suffices.
235 *
236 * HOW:
237 * 1. Skip unless a manager was seen and ngx_worker == 0.
238 * 2. Point the static timer event at brix_pending_reap_handler, mark
239 * cancelable, arm at BRIX_PENDING_REAP_INTERVAL_MS.
240 */
241void
242brix_init_pending_reap_timer(ngx_cycle_t *cycle, ngx_uint_t manager_seen)
243{
244 if (!manager_seen || ngx_worker != 0) {
245 return;
246 }
247 brix_pending_reap_timer.handler = brix_pending_reap_handler;
248 brix_pending_reap_timer.data = &brix_maint_timer_conn;
249 brix_pending_reap_timer.log = cycle->log;
250 brix_pending_reap_timer.cancelable = 1; /* don't delay graceful shutdown */
251 ngx_add_timer(&brix_pending_reap_timer,
252 BRIX_PENDING_REAP_INTERVAL_MS);
253}
