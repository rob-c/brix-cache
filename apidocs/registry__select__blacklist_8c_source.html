10/* WHAT
11 * Marks a registered server as temporarily unavailable for selection.
12 * Called from brix_cms_srv_close() when a data server's CMS connection
13 * drops. The server entry stays in the registry so its paths and metrics are
14 * preserved for the reconnect; brix_srv_select() and brix_srv_locate_all()
15 * both skip entries whose blacklisted_until is in the future.
16 *
17 * WHY
18 * A clean reconnect within the window re-registers and clears the flag,
19 * making the server immediately available again. A permanently dead server
20 * stays blacklisted until the window expires, at which point its stale metrics
21 * become visible — operators detect this via brix_cluster_server_last_seen_seconds.
22 *
23 * HOW
24 * Locks mutex → scans for host+port match → increments error_count →
25 * sets blacklisted_until = ngx_current_msec + duration_ms. Unlocks.
26 */
27void
28brix_srv_blacklist(const char *host, uint16_t port, ngx_msec_t duration_ms)
29{
30 brix_srv_table_t *tbl;
31 brix_srv_entry_t *e;
32 ngx_uint_t i;
33
34 tbl = srv_table();
35 if (tbl == NULL) {
36 return;
37 }
38
39 ngx_shmtx_lock(&brix_srv_mutex);
40
41 for (i = 0; i < tbl->capacity; i++) {
42 e = &tbl->slots[i];
43 if (!e->in_use || e->port != port
44 || ngx_strcmp(e->host, host) != 0)
45 {
46 continue;
47 }
48 e->error_count++;
49 e->blacklisted_until = ngx_current_msec + duration_ms;
50 break;
51 }
52
53 ngx_shmtx_unlock(&brix_srv_mutex);
54}
