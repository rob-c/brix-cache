26brix_fill_classify(ngx_int_t fill_rc, int err, brix_fill_retry_t *rs)
27{
28 if (fill_rc == NGX_OK || fill_rc == NGX_DECLINED) {

30 }
31 if (err == ENOENT || err == ENOTDIR || err == EACCES || err == EPERM) {
32 return BRIX_FILL_DEFINITIVE; /* 404/403: the origin's answer */
33 }
34 if (err == EBADMSG) {
35 /* digest MISMATCH: corruption is often path-local — try each
36 * remaining endpoint once, then give up definitively (502 — the
37 * data is proven bad, not "come back later"). */
38 if (rs->verify_budget > 0) {
39 rs->verify_budget--;
40 return BRIX_FILL_RETRY;
41 }
42 return BRIX_FILL_DEFINITIVE;
43 }
44 return BRIX_FILL_RETRY; /* refused/unreach/timeout/5xx */
45}

47int
48brix_fill_retry_wait(brix_fill_retry_t *rs)
49{
50 time_t now = time(NULL);
51 time_t window;
52 ngx_msec_t d;
53
54 window = (atomic_load_explicit(rs->waiters, memory_order_relaxed) > 0)
55 ? rs->client_hold : rs->max_life;
56 if (window <= 0 || now >= rs->start + window) {
57 return 0;
58 }
59 /* half-jitter: [backoff/2, backoff) — decorrelates a farm of fills */
60 d = rs->backoff_ms / 2
61 + (ngx_msec_t) (random() % (rs->backoff_ms / 2 + 1));
62 if (rs->backoff_ms < FILL_BACKOFF_CAP_MS) {
63 rs->backoff_ms *= 2;
64 }
65 usleep((useconds_t) d * 1000); /* fill worker thread: sleeping OK */
66 return 1;
67}
