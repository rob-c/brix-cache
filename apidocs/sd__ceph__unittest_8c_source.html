1/*
2 * sd_ceph_unittest.c — standalone unit test for the Ceph driver's pure
3 * LFN->object-key mapping (sd_ceph_normalize / sd_ceph_key / sd_ceph_ino).
4 *
5 * These are the security-critical, cluster-independent parts of the backend: the
6 * map must be injective (no two logical paths alias one object) and prefix-
7 * confined (no `..` escapes the export's key prefix). They need no librados and
8 * no nginx, so they are tested here with plain gcc and no running cluster.
9 *
10 * Build & run (BRIX_HAVE_CEPH intentionally OFF so only the pure helpers
11 * compile — no librados needed):
12 * cc -Wall -Wextra -I. sd_ceph_unittest.c sd_ceph.c sd_ceph_compat.c \
13 * -o /tmp/sd_ceph_ut && /tmp/sd_ceph_ut

61/* ---- Exercise sd_ceph_normalize canonicalization rules ----
62 *
63 * WHAT: Runs the accept-and-canonicalize table for sd_ceph_normalize; each row
64 * asserts a well-formed input maps to its expected canonical form. Bumps the
65 * shared failures counter (via check_norm_ok) on any mismatch.
66 *
67 * WHY: Canonicalization is the first half of the injective map contract. Grouping
68 * these rows in one helper keeps main's cyclomatic complexity under the cap while
69 * preserving every assertion and its order.
70 *
71 * HOW:
72 * 1. Assert slash collapsing, synthesized leading slash, and dot-drop rules.
73 * 2. Assert dot-dot popping (one component, to root, trailing) rules.
74 * 3. Assert trailing-slash stripping and the bare-root and empty-string cases.
75 */
76static void
77test_normalize_canonicalization(void)
78{
79 check_norm_ok("/a/b/c", "/a/b/c");
80 check_norm_ok("a/b/c", "/a/b/c"); /* leading slash synthesized */
81 check_norm_ok("/a//b///c", "/a/b/c"); /* repeated slashes collapse */
82 check_norm_ok("/a/./b/./c", "/a/b/c"); /* dot component dropped */
83 check_norm_ok("/a/b/../c", "/a/c"); /* dot-dot pops one component */
84 check_norm_ok("/a/b/../../c", "/c"); /* dot-dot pops to root */
85 check_norm_ok("/a/b/", "/a/b"); /* trailing slash stripped */
86 check_norm_ok("/", "/"); /* bare root */
87 check_norm_ok("", "/"); /* empty -> root */
88 check_norm_ok("/a/b/..", "/a"); /* trailing dot-dot */
89 check_norm_ok("/a/../b", "/b");
90}

92/* ---- Assert the key map is injective across aliasing inputs ----
93 *
94 * WHAT: Composes two object keys from inputs that differ only by redundant
95 * slashes and a dot component, then asserts they collapse to the identical key.
96 * Bumps the shared failures counter on mismatch.
97 *
98 * WHY: Injectivity is the security-critical half of the map contract: two logical
99 * paths that name the same file must never produce two distinct objects.
100 *
101 * HOW:
102 * 1. Build k1 from the redundant-slash, dot-bearing form.
103 * 2. Build k2 from the already-canonical form.
104 * 3. Fail if the two composed keys differ.
105 */
106static void
107test_key_alias_injectivity(void)
108{
109 char k1[256], k2[256];
110
111 sd_ceph_key("p", "/x//y/./z", k1, sizeof(k1));
112 sd_ceph_key("p", "/x/y/z", k2, sizeof(k2));
113 if (strcmp(k1, k2) != 0) {
114 fprintf(stderr, "FAIL alias: \"%s\" != \"%s\"\n", k1, k2);
115 failures++;
116 }
117}

119/* ---- Assert dot-dot escape attempts are rejected ----
120 *
121 * WHAT: Feeds sd_ceph_normalize a table of inputs whose dot-dot sequences would
122 * climb above the export root; each must be rejected. Bumps the shared failures
123 * counter (via check_norm_reject) on any accepted input.
124 *
125 * WHY: Prefix confinement is the other security-critical guarantee: no input may
126 * escape the export's key prefix, so every above-root climb must fail closed.
127 *
128 * HOW:
129 * 1. Assert rejection of dot-dot at or immediately below root.
130 * 2. Assert rejection of mid-path climbs that net above root.
131 * 3. Assert rejection of a relative traversal payload.
132 */
133static void
134test_normalize_escape_rejects(void)
135{
136 check_norm_reject("/..");
137 check_norm_reject("..");
138 check_norm_reject("/a/../..");
139 check_norm_reject("/a/../../b");
140 check_norm_reject("../etc/passwd");
141}

143/* ---- Exercise sd_ceph_key prefix composition ----
144 *
145 * WHAT: Runs the prefix-plus-normalized-path composition table; each row asserts
146 * a prefix and logical path yield the expected object key. Bumps the shared
147 * failures counter (via check_key) on any mismatch.
148 *
149 * WHY: Key composition is where the confined, canonical path is joined to the
150 * export prefix; the empty-prefix and trailing-slash-prefix edge cases must be
151 * pinned so a prefix change cannot silently alter object placement.
152 *
153 * HOW:
154 * 1. Assert the empty-prefix pass-through case.
155 * 2. Assert the plain-prefix and redundant-slash normalization cases.
156 * 3. Assert the trailing-slash prefix case preserves the literal double slash.
157 */
158static void
159test_key_composition(void)
160{
161 check_key("", "/data/f.root", "/data/f.root");
162 check_key("xrd", "/data/f.root", "xrd/data/f.root");
163 check_key("xrd", "data//f.root", "xrd/data/f.root");
164 check_key("pool42/", "/a/b", "pool42//a/b");
165}

167/* ---- Assert the inode hash is stable and collision-free on a spot check ----
168 *
169 * WHAT: Asserts sd_ceph_ino returns the same value for repeated calls on one oid
170 * and different values for two distinct oids. Bumps the shared failures counter
171 * on either violation.
172 *
173 * WHY: The synthetic inode number must be deterministic per object (so stat is
174 * stable across calls) yet distinguish distinct objects, which callers rely on
175 * for directory enumeration identity.

177 * HOW:

194/* ---- Assert XrdCeph striper-layout classification and pfn recovery ----
195 *
196 * WHAT: Exercises sd_ceph_oid_is_first_stripe, sd_ceph_oid_to_pfn, and
197 * sd_ceph_oid_is_stripe_data over first-stripe, data-stripe, flat-object, and
198 * malformed-suffix oids. Bumps the shared failures counter on any misclassification.
199 *
200 * WHY: Catalog enumeration must keep exactly the first stripe per file, skip data
201 * stripes, and treat flat (non-striped) objects as their own entry; getting this
202 * wrong duplicates or drops files during directory listing.
203 *
204 * HOW:
205 * 1. Assert first-stripe recognition accepts stripe .0 and rejects .1 and flat.
206 * 2. Assert pfn recovery strips the stripe suffix from a first-stripe oid.
207 * 3. Assert data-stripe recognition accepts hex data suffixes and rejects the
208 * first stripe, flat objects, non-hex, and short suffixes.
209 */
210static void
211test_striper_layout(void)
212{
213 char pfn[256];
214
215 if (!sd_ceph_oid_is_first_stripe("data/f.root.0000000000000000")
216 || sd_ceph_oid_is_first_stripe("data/f.root.0000000000000001")
217 || sd_ceph_oid_is_first_stripe("data/f.root")) {
218 fprintf(stderr, "FAIL is_first_stripe classification\n");
219 failures++;
220 }
221 if (sd_ceph_oid_to_pfn("data/f.root.0000000000000000", pfn, sizeof(pfn))
222 != 0 || strcmp(pfn, "data/f.root") != 0) {
223 fprintf(stderr, "FAIL oid_to_pfn -> \"%s\"\n", pfn);
224 failures++;
225 }
226 if (sd_ceph_oid_is_stripe_data("data/f.root.0000000000000000") /* first */
227 || !sd_ceph_oid_is_stripe_data("data/f.root.0000000000000001") /* data */
228 || !sd_ceph_oid_is_stripe_data("data/f.root.00000000000000ff") /* data */
229 || sd_ceph_oid_is_stripe_data("data/f.root") /* flat object */
230 || sd_ceph_oid_is_stripe_data("data/f.root.000000000000000g") /* non-hex */
231 || sd_ceph_oid_is_stripe_data("data/f.root.1")) { /* short suffix */
232 fprintf(stderr, "FAIL is_stripe_data classification\n");
233 failures++;
234 }
235}

237/* ---- Assert one-level child classification for the stripe-collapse listing ----
238 *
239 * WHAT: Exercises sd_ceph_path_child over root/deep listings: file children,
240 * synthetic-directory children, non-children (the dir itself, siblings whose
241 * name merely prefixes the dir, out-of-tree paths), and the unrepresentable-
242 * name skip. Bumps the shared failures counter on any misclassification.
243 *
244 * WHY: This is the collapse step of the phase-89 §B.1 directory listing: a
245 * wrong match either leaks entries across directories (a sibling whose name
246 * shares the dir as a string prefix) or drops/duplicates listing rows.
247 *
248 * HOW:
249 * 1. Assert root-listing file and dir children.
250 * 2. Assert deep-listing file and dir children.
251 * 3. Assert the non-child rejections (self, prefix-sibling, out-of-tree).
252 * 4. Assert the tiny-cap unrepresentable-name skip returns 0.
253 */
254static void
255test_path_child(void)
256{
257 char name[256];
258 char tiny[3];
259
260 if (sd_ceph_path_child("/", "/f.root", name, sizeof(name)) != 1
261 || strcmp(name, "f.root") != 0) {
262 fprintf(stderr, "FAIL path_child root file\n");
263 failures++;
264 }
265 if (sd_ceph_path_child("/", "/d/x/y", name, sizeof(name)) != 2
266 || strcmp(name, "d") != 0) {
267 fprintf(stderr, "FAIL path_child root dir\n");
268 failures++;
269 }
270 if (sd_ceph_path_child("/a/b", "/a/b/f", name, sizeof(name)) != 1
271 || strcmp(name, "f") != 0) {
272 fprintf(stderr, "FAIL path_child deep file\n");
273 failures++;
274 }
275 if (sd_ceph_path_child("/a/b", "/a/b/c/f", name, sizeof(name)) != 2
276 || strcmp(name, "c") != 0) {
277 fprintf(stderr, "FAIL path_child deep dir\n");
278 failures++;
279 }
280 if (sd_ceph_path_child("/a/b", "/a/b", name, sizeof(name)) != 0 /* self */
281 || sd_ceph_path_child("/a/b", "/a/bc/f", name, sizeof(name)) != 0 /* prefix sibling */
282 || sd_ceph_path_child("/a/b", "/a/x/f", name, sizeof(name)) != 0 /* out of tree */
283 || sd_ceph_path_child("/a/b", "/z", name, sizeof(name)) != 0) {
284 fprintf(stderr, "FAIL path_child non-child rejection\n");
285 failures++;

288 fprintf(stderr, "FAIL path_child unrepresentable-name skip\n");
289 failures++;

293/* ---- Run every pure-mapping check group and report the aggregate result ----
294 *
295 * WHAT: Invokes each check-group helper in order, then prints a pass line and
296 * returns 0 if the shared failures counter is zero, or a failure count and
297 * returns 1 otherwise.
298 *
299 * WHY: The pure LFN->object-key helpers are cluster-independent and security
300 * critical, so they are exercised here as a standalone program with no librados
301 * and no nginx; this orchestrator stays a flat sequence of named check groups.
302 *
303 * HOW:
304 * 1. Run canonicalization, alias-injectivity, and escape-rejection checks.
305 * 2. Run key composition, inode-hash, and striper-layout checks.
306 * 3. Report success or the failure count via the shared counter.
307 */

313 test_normalize_escape_rejects();
