1/*
2 * sd_ceph_unittest.c — standalone unit test for the Ceph driver's pure
3 * LFN->object-key mapping (sd_ceph_normalize / sd_ceph_key / sd_ceph_ino).
4 *
5 * These are the security-critical, cluster-independent parts of the backend: the
6 * map must be injective (no two logical paths alias one object) and prefix-
7 * confined (no `..` escapes the export's key prefix). They need no librados and
8 * no nginx, so they are tested here with plain gcc and no running cluster.
9 *
10 * Build & run (BRIX_HAVE_CEPH intentionally OFF so only the pure helpers
11 * compile — no librados needed):
12 * cc -Wall -Wextra -I. sd_ceph_unittest.c sd_ceph.c sd_ceph_compat.c \
13 * -o /tmp/sd_ceph_ut && /tmp/sd_ceph_ut
14 */

24check_norm_ok(const char *in, const char *want)
25{

30 fprintf(stderr, "FAIL normalize(\"%s\") -> rc=%d \"%s\" (want \"%s\")\n",

61int
62main(void)
63{
64 /* canonicalization: collapse slashes, drop ".", apply ".." */
65 check_norm_ok("/a/b/c", "/a/b/c");
66 check_norm_ok("a/b/c", "/a/b/c"); /* leading slash synthesized */
67 check_norm_ok("/a//b///c", "/a/b/c"); /* repeated slashes collapse */
68 check_norm_ok("/a/./b/./c", "/a/b/c"); /* "." dropped */
69 check_norm_ok("/a/b/../c", "/a/c"); /* ".." pops one component */
70 check_norm_ok("/a/b/../../c", "/c"); /* ".." pops to root */
71 check_norm_ok("/a/b/", "/a/b"); /* trailing slash stripped */
72 check_norm_ok("/", "/"); /* bare root */
73 check_norm_ok("", "/"); /* empty -> root */
74 check_norm_ok("/a/b/..", "/a"); /* trailing ".." */
75 check_norm_ok("/a/../b", "/b");
76
77 /* injectivity: distinct-looking inputs that must collapse to one key DO,
78 * and genuinely different paths stay different (spot check). */
79 {
80 char k1[256], k2[256];
81 sd_ceph_key("p", "/x//y/./z", k1, sizeof(k1));
82 sd_ceph_key("p", "/x/y/z", k2, sizeof(k2));
83 if (strcmp(k1, k2) != 0) {
84 fprintf(stderr, "FAIL alias: \"%s\" != \"%s\"\n", k1, k2);
85 failures++;
86 }
87 }
88
89 /* escape attempts: any ".." that would climb above the root is rejected. */
90 check_norm_reject("/..");
91 check_norm_reject("..");
92 check_norm_reject("/a/../..");
93 check_norm_reject("/a/../../b");
94 check_norm_reject("../etc/passwd");
95
96 /* key composition: prefix + normalized path */
97 check_key("", "/data/f.root", "/data/f.root");
98 check_key("xrd", "/data/f.root", "xrd/data/f.root");
99 check_key("xrd", "data//f.root", "xrd/data/f.root");
100 check_key("pool42/", "/a/b", "pool42//a/b");
101
102 /* inode hash: stable + distinct for distinct oids */
103 if (sd_ceph_ino("xrd/a") != sd_ceph_ino("xrd/a")) {
104 fprintf(stderr, "FAIL ino not stable\n");
105 failures++;
106 }
107 if (sd_ceph_ino("xrd/a") == sd_ceph_ino("xrd/b")) {
108 fprintf(stderr, "FAIL ino collision on distinct oids\n");
109 failures++;
110 }
111
112 /* XrdCeph striper layout: catalog enumeration keeps the first stripe (one per
113 * file), skips data stripes, and treats flat (non-striped) objects as their
114 * own entry. */
115 if (!sd_ceph_oid_is_first_stripe("data/f.root.0000000000000000")
116 || sd_ceph_oid_is_first_stripe("data/f.root.0000000000000001")
117 || sd_ceph_oid_is_first_stripe("data/f.root")) {
118 fprintf(stderr, "FAIL is_first_stripe classification\n");
119 failures++;
120 }
121 {
122 char pfn[256];
123 if (sd_ceph_oid_to_pfn("data/f.root.0000000000000000", pfn, sizeof(pfn))
124 != 0 || strcmp(pfn, "data/f.root") != 0) {
125 fprintf(stderr, "FAIL oid_to_pfn -> \"%s\"\n", pfn);
126 failures++;
127 }
128 }
129 if (sd_ceph_oid_is_stripe_data("data/f.root.0000000000000000") /* first */
130 || !sd_ceph_oid_is_stripe_data("data/f.root.0000000000000001") /* data */
131 || !sd_ceph_oid_is_stripe_data("data/f.root.00000000000000ff") /* data */
132 || sd_ceph_oid_is_stripe_data("data/f.root") /* flat object */
133 || sd_ceph_oid_is_stripe_data("data/f.root.000000000000000g") /* non-hex */
134 || sd_ceph_oid_is_stripe_data("data/f.root.1")) { /* short suffix */
135 fprintf(stderr, "FAIL is_stripe_data classification\n");
136 failures++;
137 }
138
139 if (failures == 0) {
140 printf("sd_ceph_unittest: all checks passed\n");
141 return 0;
142 }
143 fprintf(stderr, "sd_ceph_unittest: %d FAILURE(S)\n", failures);
144 return 1;
145}
