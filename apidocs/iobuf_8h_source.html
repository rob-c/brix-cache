1/*
2 * iobuf.h — per-handle read-ahead / write-back buffering engine for the FUSE
3 * drivers, parameterised by a backend pread/pwrite pair.
4 *
5 * WHAT: an brix_iobuf coalesces sequential reads into one larger server read and
6 * small contiguous writes into one larger server write (flushed on
7 * flush/fsync/release), exactly the state machine both FUSE drivers used to
8 * open-code. The actual server I/O is supplied as two function pointers over
9 * an opaque backend handle, so one engine serves every backend:
10 * - legacy: brix_file_read/_write over a pinned brix_conn+brix_file;
11 * - async : afh_pread (mfile/webfile) + brix_mfile_pwrite.
12 * WHY: the read/write/flush logic was duplicated byte-for-byte across xrootdfs.c
13 * and xrootdfs_legacy.c (~220 LoC); this is the one copy.
14 * HOW: NOT thread-safe — the caller (the FUSE handler) holds its per-handle mutex
15 * across the call, matching the historical behaviour. Read-ahead applies only
16 * to read-only handles; write-back only to writable ones; either is disabled
17 * by passing size 0.
18 */

27/* Backend I/O: pread returns bytes (>=0) or <0 on error; pwrite returns 0 / -1.
28 * Both set *st on failure. `be` is the opaque backend handle. */
29typedef ssize_t (*brix_iobuf_pread_fn)(void *be, int64_t off, void *buf,
30 size_t n, brix_status *st);
31typedef int (*brix_iobuf_pwrite_fn)(void *be, int64_t off, const void *buf,
32 size_t n, brix_status *st);
33
34typedef struct {
35 void *be; /* opaque backend handle */
36 brix_iobuf_pread_fn pread;
37 brix_iobuf_pwrite_fn pwrite;
38 int writable;
39 size_t ra_size; /* read-ahead window (0 disables) */
40 size_t wb_size; /* write-back window (0 disables) */
41
42 unsigned char *rbuf; /* read-ahead: holds [rbuf_off, +rbuf_len) */
43 off_t rbuf_off;
44 size_t rbuf_len;
45 off_t ra_next; /* next sequential offset (prefetch hint) */
46
47 unsigned char *wbuf; /* write-back: pending [wbuf_off, +wbuf_len) */
48 off_t wbuf_off;
49 size_t wbuf_len;
50} brix_iobuf;

63/* Write `n` bytes at `off` (coalescing into the write-back buffer). Returns 0 on
