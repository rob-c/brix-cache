23/*
24 * Resolve $brix_protocol for the current request: "webdav", "s3", "cvmfs",
25 * or "http". Precedence is webdav > s3 > cvmfs > plain http, decided by
26 * which sibling module is enabled in this request's location conf (WebDAV
27 * wins if several somehow apply). The labels are the central proto_list.h
28 * dash_names; "http" is the nothing-claimed fallback. Used in log_format /
29 * proxy decisions to label the served protocol.
30 */
31ngx_int_t
32brix_http_protocol_variable(ngx_http_request_t *r,
33 ngx_http_variable_value_t *v, uintptr_t data)
34{
35 ngx_http_brix_webdav_loc_conf_t *wdcf;
36 ngx_http_s3_loc_conf_t *scf;
37 ngx_http_brix_cvmfs_loc_conf_t *ccf;
38 const char *label;
39 size_t len;
40
41 (void) data;
42
43 label = "http";
44 len = sizeof("http") - 1;
45
46 wdcf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
47 scf = ngx_http_get_module_loc_conf(r, ngx_http_brix_s3_module);
48 ccf = ngx_http_get_module_loc_conf(r, ngx_http_brix_cvmfs_module);
49 if (wdcf != NULL && wdcf->common.enable) {
50 label = "webdav";
51 len = sizeof("webdav") - 1;
52 } else if (scf != NULL && scf->common.enable) {
53 label = "s3";
54 len = sizeof("s3") - 1;
55 } else if (ccf != NULL && ccf->cvmfs.enable) {
56 label = "cvmfs";
57 len = sizeof("cvmfs") - 1;
58 }
59
60 v->len = len;
61 v->valid = 1;
62 v->no_cacheable = 0;
63 v->not_found = 0;
64 v->data = (u_char *) label;
65
66 return NGX_OK;
67}

70/*
71 * delegated_cred_find_eec — locate the end-entity certificate of the
72 * TLS-verified peer: the leaf itself when the client authenticated with a
73 * bare cert, else the first non-proxy cert (brix_px_classify) in the
74 * presented chain (RFC 3820 proxies chain leaf → parent proxies → EEC).
75 * Returns a BORROWED reference (owned by `leaf`'s refcount or by the
76 * connection's chain) or NULL when every presented cert is a proxy.
77 */
78static X509 *
79delegated_cred_find_eec(SSL *sc, X509 *leaf)
80{
81 STACK_OF(X509) *chain;
82 int i, n;
83
84 if (brix_px_classify(leaf) == BRIX_PX_NONE) {
85 return leaf;
86 }
87
88 /* Server side: SSL_get_peer_cert_chain excludes the leaf. */
89 chain = SSL_get_peer_cert_chain(sc);
90 n = (chain != NULL) ? sk_X509_num(chain) : 0;
91
92 for (i = 0; i < n; i++) {
93 X509 *cert = sk_X509_value(chain, i);
94
95 if (brix_px_classify(cert) == BRIX_PX_NONE) {
96 return cert;
97 }
98 }
99
100 return NULL;
101}

104/*
105 * $brix_delegated_cred — the verified client's delegated-credential path.
106 *
107 * WHAT: resolves the per-user credential file stored by the delegation
108 * endpoint (delegation.c): EEC DN of the TLS-verified client chain →
109 * brix_sd_ucred_key ("x5h-" + SHA-256 prefix) → <storage_credential_dir>/
110 * <key>.pem, expiry-checked. Empty when anything is missing — FAIL CLOSED.
111 *
112 * WHY: replaces the hand-maintained `map $ssl_client_s_dn_legacy` block in
113 * credential-forwarding gateways (ARC-CE front proxy): the map duplicated
114 * the ucred key derivation per user and had to be regenerated + reloaded on
115 * every new user, and could not expiry-check the file it named. With this
116 * variable, `proxy_ssl_certificate $brix_delegated_cred` picks up a new
117 * user's credential the moment delegation stores it.
118 *
119 * HOW: identity comes from the TLS layer directly (SSL_get_verify_result
120 * must be X509_V_OK — same signal as $ssl_client_verify), NOT from brix
121 * request auth, so the variable works in plain proxy_pass locations with no
122 * brix handler. The EEC is found by skipping RFC 3820 proxy certs
123 * (delegation stores under the EEC DN — the delegation endpoint refuses
124 * proxy-authenticated uploads). Kind must be x509 (.pem): bearer/s3/ceph
125 * files cannot feed proxy_ssl_certificate. Expired → empty, with an info
126 * log naming the DN so operators can see who must re-delegate.
127 */
128ngx_int_t
129brix_http_delegated_cred_variable(ngx_http_request_t *r,
130 ngx_http_variable_value_t *v, uintptr_t data)
131{
132 ngx_http_brix_webdav_loc_conf_t *wdcf;
133 ngx_connection_t *c;
134 X509 *leaf, *eec;
135 ngx_str_t *dir;
136 brix_sd_ucred_t cred;
137 ngx_int_t rc;
138 size_t len;
139 u_char *p;
140 char dn[BRIX_UCRED_PRINC_MAX];
141 char key[BRIX_UCRED_KEY_MAX];
142 char dirbuf[BRIX_UCRED_PATH_MAX];
143
144 (void) data;
145
146 /* Fail-closed default: a valid, empty value ("" == no credential). */
147 v->len = 0;
148 v->valid = 1;
149 v->no_cacheable = 0;
150 v->not_found = 0;
151 v->data = (u_char *) "";
152
153 c = r->connection;
154
155 if (c->ssl == NULL
156 || SSL_get_verify_result(c->ssl->connection) != X509_V_OK)
157 {
158 return NGX_OK;
159 }
160
161 leaf = SSL_get_peer_certificate(c->ssl->connection);
162 if (leaf == NULL) {
163 return NGX_OK;
164 }
165
166 dn[0] = '\0';
167 eec = delegated_cred_find_eec(c->ssl->connection, leaf);
168 if (eec != NULL) {
169 brix_x509_oneline(X509_get_subject_name(eec), dn, sizeof(dn));
170 }
171 X509_free(leaf);
172
173 if (dn[0] == '\0') {
174 return NGX_OK;
175 }
176
177 wdcf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
178 dir = &wdcf->common.storage_credential_dir;
179
180 if (dir->len == 0 || dir->len >= sizeof(dirbuf)) {
181 return NGX_OK;
182 }
183 ngx_memcpy(dirbuf, dir->data, dir->len);
184 dirbuf[dir->len] = '\0';
185
186 if (brix_sd_ucred_key(dn, key, sizeof(key)) != NGX_OK) {
187 return NGX_OK;
188 }
189
190 /* resolve() leaves fields untouched on its not-found path — zero first
191 * so the expired-credential log below never reads garbage. */
192 ngx_memzero(&cred, sizeof(cred));
193
194 rc = brix_sd_ucred_resolve(dirbuf, key, &cred);
195
196 if (rc != NGX_OK || cred.is_bearer || cred.is_s3 || cred.is_ceph) {
197 if (rc == NGX_DECLINED && cred.expired) {
198 ngx_log_error(NGX_LOG_INFO, c->log, 0,
199 "brix_delegated_cred: stored credential for "
200 "\"%s\" has expired — re-delegation required", dn);
201 }
202 /* A bearer/s3/ceph credential may have resolved here (this variable
203 * only surfaces the x509 proxy PATH); erase any secret we decline. */
204 brix_sd_ucred_wipe(&cred);
205 return NGX_OK;
206 }
207
208 len = ngx_strlen(cred.path);
209 p = ngx_pnalloc(r->pool, len);
210 if (p == NULL) {
211 brix_sd_ucred_wipe(&cred);
212 return NGX_ERROR;
213 }
214 ngx_memcpy(p, cred.path, len);
215
216 v->len = len;
217 v->data = p;
218
219 brix_sd_ucred_wipe(&cred); /* x509 proxy PATH exported; erase residue */
220 return NGX_OK;
221}

224ngx_int_t
225brix_http_add_protocol_variables(ngx_conf_t *cf)
226{
227 ngx_http_variable_t *var;
228 ngx_str_t name = ngx_string("brix_protocol");
229 ngx_str_t cred_name = ngx_string("brix_delegated_cred");
230
231 var = ngx_http_add_variable(cf, &name, NGX_HTTP_VAR_NOCACHEABLE);
232 if (var == NULL) {
233 return NGX_ERROR;
234 }
235
236 var->get_handler = brix_http_protocol_variable;
237 var->data = 0;
238
239 /* Per-request cacheable: the TLS identity cannot change mid-request. */
240 var = ngx_http_add_variable(cf, &cred_name, 0);
241 if (var == NULL) {
242 return NGX_ERROR;
243 }
244
245 var->get_handler = brix_http_delegated_cred_variable;
246 var->data = 0;
247
248 return NGX_OK;
249}
