20/*
21 * Resolve $brix_protocol for the current request: "webdav", "s3", "cvmfs",
22 * or "http". Precedence is webdav > s3 > cvmfs > plain http, decided by
23 * which sibling module is enabled in this request's location conf (WebDAV
24 * wins if several somehow apply). The labels are the central proto_list.h
25 * dash_names; "http" is the nothing-claimed fallback. Used in log_format /
26 * proxy decisions to label the served protocol.
27 */
28ngx_int_t
29brix_http_protocol_variable(ngx_http_request_t *r,
30 ngx_http_variable_value_t *v, uintptr_t data)
31{
32 ngx_http_brix_webdav_loc_conf_t *wdcf;
33 ngx_http_s3_loc_conf_t *scf;
34 ngx_http_brix_cvmfs_loc_conf_t *ccf;
35 const char *label;
36 size_t len;
37
38 (void) data;
39
40 label = "http";
41 len = sizeof("http") - 1;
42
43 wdcf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
44 scf = ngx_http_get_module_loc_conf(r, ngx_http_brix_s3_module);
45 ccf = ngx_http_get_module_loc_conf(r, ngx_http_brix_cvmfs_module);
46 if (wdcf != NULL && wdcf->common.enable) {
47 label = "webdav";
48 len = sizeof("webdav") - 1;
49 } else if (scf != NULL && scf->common.enable) {
50 label = "s3";
51 len = sizeof("s3") - 1;
52 } else if (ccf != NULL && ccf->cvmfs.enable) {
53 label = "cvmfs";
54 len = sizeof("cvmfs") - 1;
55 }
56
57 v->len = len;
58 v->valid = 1;
59 v->no_cacheable = 0;
60 v->not_found = 0;
61 v->data = (u_char *) label;
62
63 return NGX_OK;
64}
