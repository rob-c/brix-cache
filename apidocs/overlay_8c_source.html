1/*
2 * overlay.c — writable union overlay core for brixMount cvmfs-rw.
3 *
4 * WHAT: implements the upper-tree primitives declared in overlay.h: the
5 * O_NOFOLLOW component walk, union classification, whiteout/opaque
6 * markers, mutations, atomic copy-up, readdir nameset, and CLI cores.
7 * WHY: one small, FUSE/CVMFS-free translation unit owns every union corner
8 * case so they are provable on plain tmp directories.
9 * HOW: ov_split() peels validated path components (".", "..", empty and
10 * oversized components are refused); ov_walk_parent() descends dir by
11 * dir with O_NOFOLLOW|O_DIRECTORY so planted symlinks dead-end instead
12 * of escaping; everything else composes those two.
13 */

29/* ---- path-component parsing --------------------------------------------- */
30
31/* Peel the leading component of *relp into name[]; advances *relp past the
32 * '/' (or leaves it NULL on the last component). Returns 1 = more components
33 * follow, 0 = this was the leaf, -EINVAL = empty/"."/".."/oversized. */
34static int ov_split(const char **relp, char *name, size_t cap) {
35 const char *rel = *relp;
36 const char *slash = strchr(rel, '/');
37 size_t n = slash ? (size_t) (slash - rel) : strlen(rel);
38
39 if (n == 0 || n >= cap) return -EINVAL;
40 memcpy(name, rel, n);
41 name[n] = '\0';
42 if (strcmp(name, ".") == 0 || strcmp(name, "..") == 0) return -EINVAL;
43 *relp = slash ? slash + 1 : NULL;
44 return slash ? 1 : 0;
45}

65/* ---- the O_NOFOLLOW walk ------------------------------------------------- */
66
67/* Open the parent directory of `rel`'s leaf, descending from upper_fd one
68 * component at a time with O_NOFOLLOW|O_DIRECTORY (symlinks dead-end). With
69 * `mk`, missing intermediate dirs are created (mode_fn(ud, prefix) or 0755).
70 * The leaf component is stored in leaf[]. Returns the parent dirfd (caller
71 * closes) or -errno. rel must be non-empty. */
72static int ov_walk_parent_mk(const brix_overlay *ov, const char *rel,
73 char *leaf, size_t leafcap, int mk,
74 mode_t (*mode_fn)(void *ud, const char *rel_dir),
75 void *ud) {
76 if (rel == NULL || rel[0] == '\0') return -EINVAL;
77
78 int cur = openat(ov->upper_fd, ".", O_RDONLY | O_DIRECTORY | O_CLOEXEC);
79 if (cur < 0) return -errno;
80
81 char prefix[4096]; /* walked prefix, for mode_fn */
82 size_t plen = 0;
83 const char *p = rel;
84 prefix[0] = '\0';
85
86 for (;;) {
87 int more = ov_split(&p, leaf, leafcap);
88 if (more < 0) { close(cur); return more; }
89 if (more == 0) return cur; /* cur = the leaf's parent */
90
91 size_t ll = strlen(leaf);
92 if (plen + ll + 2 < sizeof(prefix)) {
93 if (plen > 0) prefix[plen++] = '/';
94 memcpy(prefix + plen, leaf, ll + 1);
95 plen += ll;
96 }
97
98 int next = openat(cur, leaf, O_RDONLY | O_DIRECTORY | O_NOFOLLOW | O_CLOEXEC);
99 if (next < 0 && mk && errno == ENOENT) {
100 mode_t m = mode_fn ? mode_fn(ud, prefix) : 0755;
101 if (mkdirat(cur, leaf, m) != 0 && errno != EEXIST) {
102 int e = errno;
103 close(cur);
104 return -e;
105 }
106 next = openat(cur, leaf, O_RDONLY | O_DIRECTORY | O_NOFOLLOW | O_CLOEXEC);
107 }
108 if (next < 0) { int e = errno; close(cur); return -e; }
109 close(cur);
110 cur = next;
111 }
112}

114static int ov_walk_parent(const brix_overlay *ov, const char *rel,
115 char *leaf, size_t leafcap) {
116 return ov_walk_parent_mk(ov, rel, leaf, leafcap, 0, NULL, NULL);
117}

136/* ---- classification ------------------------------------------------------ */
137
138/* Walk `rel` from the upper root and decide how the union sees it: a whiteout
139 * on any component, an opaque ancestor, or a non-dir upper component shadowing
140 * the remainder all mask the lower layer; a leaf found in upper wins; anything
141 * else falls through to lower. */
142int brix_overlay_classify(const brix_overlay *ov, const char *rel,
143 struct stat *st, brix_ov_state *state) {
144 *state = BRIX_OV_NONE;
145 if (rel == NULL) return -EINVAL;
146
147 if (rel[0] == '\0') { /* the repo root is upper/ */
148 if (st != NULL && fstat(ov->upper_fd, st) != 0) return -errno;
149 *state = BRIX_OV_UPPER;
150 return 0;
151 }
152
153 int cur = openat(ov->upper_fd, ".", O_RDONLY | O_DIRECTORY | O_CLOEXEC);
154 if (cur < 0) return -errno;
155
156 int opaque = 0;
157 const char *p = rel;
158 char name[OV_NAME_MAX], wh[OV_NAME_MAX + sizeof(BRIX_OV_WH_PREFIX)];
159
160 for (;;) {
161 int more = ov_split(&p, name, sizeof(name));
162 if (more < 0) { close(cur); return more; }
163
164 ov_wh_name(name, wh, sizeof(wh));
165 if (ov_marker_at(cur, wh)) { /* whiteouted at this level */
166 close(cur);
167 *state = BRIX_OV_MASKED;
168 return 0;
169 }
170
171 struct stat cst;
172 if (fstatat(cur, name, &cst, AT_SYMLINK_NOFOLLOW) != 0) {
173 close(cur); /* not in upper from here on */
174 *state = opaque ? BRIX_OV_MASKED : BRIX_OV_NONE;
175 return 0;
176 }
177
178 if (more == 0) { /* leaf present in upper */
179 close(cur);
180 if (st != NULL) *st = cst;
181 *state = BRIX_OV_UPPER;
182 return 0;
183 }
184
185 if (!S_ISDIR(cst.st_mode)) { /* upper non-dir shadows all below */
186 close(cur);
187 *state = BRIX_OV_MASKED;
188 return 0;
189 }
190
191 int next = openat(cur, name, O_RDONLY | O_DIRECTORY | O_NOFOLLOW | O_CLOEXEC);
192 close(cur);
193 if (next < 0) return -errno;
194 cur = next;
195 if (ov_marker_at(cur, BRIX_OV_OPQ_NAME)) opaque = 1;
196 }
197}

238int brix_overlay_mkdirs(const brix_overlay *ov, const char *rel_dir,
239 mode_t (*mode_fn)(void *ud, const char *rel_dir),
240 void *ud) {
241 if (rel_dir[0] == '\0') return 0; /* the upper root exists */
242
243 char leaf[OV_NAME_MAX];
244 int parent = ov_walk_parent_mk(ov, rel_dir, leaf, sizeof(leaf), 1, mode_fn, ud);
245 if (parent < 0) return parent;
246
247 mode_t m = mode_fn ? mode_fn(ud, rel_dir) : 0755;
248 int rc = 0;
249 if (mkdirat(parent, leaf, m) != 0 && errno != EEXIST) rc = -errno;
250
251 struct stat st; /* an existing non-dir leaf is an error */
252 if (rc == 0 && fstatat(parent, leaf, &st, AT_SYMLINK_NOFOLLOW) == 0
253 && !S_ISDIR(st.st_mode))
254 rc = -ENOTDIR;
255 close(parent);
256 return rc;
257}

298/* Unlink every overlay marker inside dirfd; a real entry → -ENOTEMPTY. */
299static int ov_clear_markers(int dirfd) {
300 int lfd = openat(dirfd, ".", O_RDONLY | O_DIRECTORY | O_CLOEXEC);
301 if (lfd < 0) return -errno;
302 DIR *d = fdopendir(lfd); /* owns lfd from here */
303 if (d == NULL) { int e = errno; close(lfd); return -e; }
304
305 int rc = 0;
306 struct dirent *e;
307 while (rc == 0 && (e = readdir(d)) != NULL) {
308 if (strcmp(e->d_name, ".") == 0 || strcmp(e->d_name, "..") == 0) continue;
309 if (!brix_ov_name_reserved(e->d_name)) { rc = -ENOTEMPTY; break; }
310 if (unlinkat(dirfd, e->d_name, 0) != 0) { rc = -errno; break; }
311 }
312 closedir(d);
313 return rc;
314}

410/* ---- copy-up ------------------------------------------------------------- */
411
412#define OV_COPYUP_CHUNK (1024u * 1024u)
413
414/* Stream `size` lower bytes of `rel` into fd via the injected reader. A short
415 * read before `size` is a lower-layer lie → -EIO. */
416static int ov_copyup_stream(int fd, const char *rel, uint64_t size,
417 brix_ov_read_fn read_lower, void *ud) {
418 unsigned char *buf = malloc(OV_COPYUP_CHUNK);
419 if (buf == NULL) return -ENOMEM;
420
421 uint64_t off = 0;
422 int rc = 0;
423 while (rc == 0 && off < size) {
424 size_t want = size - off > OV_COPYUP_CHUNK ? OV_COPYUP_CHUNK
425 : (size_t) (size - off);
426 size_t got = 0;
427 rc = read_lower(ud, rel, off, want, buf, &got);
428 if (rc == 0 && got == 0) rc = -EIO; /* premature EOF */
429 for (size_t done = 0; rc == 0 && done < got; ) {
430 ssize_t w = pwrite(fd, buf + done, got - done, (off_t) (off + done)); /* vfs-seam-allow: copy-up write to upper writable overlay layer, not export data */
431 if (w < 0) rc = -errno;
432 else done += (size_t) w;
433 }
434 off += got;
435 }
436 free(buf);
437 return rc;
438}

440int brix_overlay_copyup(const brix_overlay *ov, const char *rel,
441 const struct stat *lower_st,
442 brix_ov_read_fn read_lower, void *ud) {
443 char leaf[OV_NAME_MAX], tmp[OV_NAME_MAX + sizeof(BRIX_OV_TMP_PREFIX)];
444 int parent = ov_walk_parent_mk(ov, rel, leaf, sizeof(leaf), 1, NULL, NULL);
445 if (parent < 0) return parent;
446
447 snprintf(tmp, sizeof(tmp), BRIX_OV_TMP_PREFIX "%s", leaf);
448 int fd = openat(parent, tmp, O_WRONLY | O_CREAT | O_TRUNC | O_NOFOLLOW | O_CLOEXEC,
449 lower_st->st_mode & 07777);
450 if (fd < 0) { int e = errno; close(parent); return -e; }
451
452 /* the openat mode is umask-filtered — restate the exact lower bits */
453 int rc = fchmod(fd, lower_st->st_mode & 07777) != 0 ? -errno : 0;
454 if (rc == 0)
455 rc = ov_copyup_stream(fd, rel, (uint64_t) lower_st->st_size, read_lower, ud);
456 if (rc == 0) {
457 struct timespec tv[2] = { { lower_st->st_mtime, 0 }, { lower_st->st_mtime, 0 } };
458 if (futimens(fd, tv) != 0) rc = -errno;
459 }
460 close(fd);
461
462 if (rc == 0 && renameat(parent, tmp, parent, leaf) != 0) rc = -errno;
463 if (rc != 0) unlinkat(parent, tmp, 0); /* leave no torn trace */
464 close(parent);
465
466 return rc == 0 ? brix_overlay_whiteout_clear(ov, rel) : rc;
467}

471static int ov_nameset_add(brix_ov_nameset *s, char flag, const char *name) {
472 size_t need = strlen(name) + 2; /* flag + name + NUL */
473 if (s->used + need > s->cap) {
474 size_t ncap = s->cap ? s->cap * 2 : 512;
475 while (ncap < s->used + need) ncap *= 2;
476 char *nb = realloc(s->buf, ncap);
477 if (nb == NULL) return -ENOMEM;
478 s->buf = nb;
479 s->cap = ncap;
480 }
481 s->buf[s->used] = flag;
482 memcpy(s->buf + s->used + 1, name, need - 1);
483 s->used += need;
484 s->count++;
485 return 0;
486}

488int brix_overlay_read_upper(const brix_overlay *ov, const char *rel,
489 brix_ov_nameset *set, int *opaque) {
490 memset(set, 0, sizeof(*set));
491 *opaque = 0;
492
493 int dir = ov_open_dir(ov, rel);
494 if (dir == -ENOENT || dir == -ENOTDIR) return 0; /* nothing in upper */
495 if (dir < 0) return dir;
496
497 DIR *d = fdopendir(dir); /* owns dir from here */
498 if (d == NULL) { int e = errno; close(dir); return -e; }
499
500 int rc = 0;
501 struct dirent *e;
502 while (rc == 0 && (e = readdir(d)) != NULL) {
503 const char *n = e->d_name;
504 if (strcmp(n, ".") == 0 || strcmp(n, "..") == 0) continue;
505 if (strcmp(n, BRIX_OV_OPQ_NAME) == 0) { *opaque = 1; continue; }
506 if (strncmp(n, BRIX_OV_TMP_PREFIX, sizeof(BRIX_OV_TMP_PREFIX) - 1) == 0) continue;
507 if (strncmp(n, BRIX_OV_WH_PREFIX, sizeof(BRIX_OV_WH_PREFIX) - 1) == 0)
508 rc = ov_nameset_add(set, 'w', n + sizeof(BRIX_OV_WH_PREFIX) - 1);
509 else
510 rc = ov_nameset_add(set, 'u', n);
511 }
512 closedir(d);
513 if (rc != 0) brix_ov_nameset_free(set);
514 return rc;
515}

548 * On success *upper_out (asprintf'd, caller frees) holds its path. */
549static int ov_cli_guard(const char *mountdir, char **upper_out) {
550 if (asprintf(upper_out, "%s/" BRIX_OV_DIRNAME "/" BRIX_OV_UPPER_DIRNAME,
551 mountdir) < 0)
552 return 1;
553 struct stat st;
554 if (lstat(*upper_out, &st) == 0 && S_ISDIR(st.st_mode)) return 0;
555 fprintf(stderr, "brixMount: no " BRIX_OV_DIRNAME
556 " overlay under %s (not a cvmfs-rw mountpoint?)\n", mountdir);
557 free(*upper_out);
558 *upper_out = NULL;
559 return 2;
560}

563 * magic xattr ("new"/"modified"); unmounted raw trees have none → "upper". */
564static void ov_cli_kind(const char *mountdir, const char *rel,
565 char *kind, size_t cap) {
566 char *p = NULL;

568 if (asprintf(&p, "%s/%s", mountdir, rel) >= 0) {
569 n = lgetxattr(p, "user.overlay", kind, cap - 1);

572 if (n > 0) kind[n] = '\0';
573 else snprintf(kind, cap, "upper");

576static int ov_cli_list_dir(const char *mountdir, const char *upper_root,
577 const char *rel, FILE *out) {
578 char *dirp = NULL;
579 if (asprintf(&dirp, "%s%s%s", upper_root, rel[0] ? "/" : "", rel) < 0) return 1;
580 DIR *d = opendir(dirp);
581 if (d == NULL) { free(dirp); return 1; }
582
583 int rc = 0;
584 struct dirent *e;
585 while (rc == 0 && (e = readdir(d)) != NULL) {
586 const char *n = e->d_name;
587 if (strcmp(n, ".") == 0 || strcmp(n, "..") == 0) continue;
588 if (strcmp(n, BRIX_OV_OPQ_NAME) == 0) continue;
589 if (strncmp(n, BRIX_OV_TMP_PREFIX, sizeof(BRIX_OV_TMP_PREFIX) - 1) == 0) continue;
590
591 if (strncmp(n, BRIX_OV_WH_PREFIX, sizeof(BRIX_OV_WH_PREFIX) - 1) == 0) {
592 fprintf(out, "deleted %s%s%s\n", rel, rel[0] ? "/" : "",
593 n + sizeof(BRIX_OV_WH_PREFIX) - 1);
594 continue;
595 }
596
597 char *crel = NULL, *full = NULL;
598 if (asprintf(&crel, "%s%s%s", rel, rel[0] ? "/" : "", n) < 0
599 || asprintf(&full, "%s/%s", dirp, n) < 0) {
600 free(crel);
601 rc = 1;
602 continue;
603 }
604 struct stat st;
605 if (lstat(full, &st) == 0 && S_ISDIR(st.st_mode)) {
606 fprintf(out, "dir %s\n", crel);
607 rc = ov_cli_list_dir(mountdir, upper_root, crel, out);
608 } else {
609 char kind[64];
610 ov_cli_kind(mountdir, crel, kind, sizeof(kind));
611 fprintf(out, "%s %s\n", kind, crel);
612 }
613 free(crel);
614 free(full);
615 }
616 closedir(d);
617 free(dirp);
618 return rc;
619}

630/* Recursively delete everything inside dirfd (never following symlinks). */
631static int ov_cli_reset_contents(int dirfd) {
632 int lfd = openat(dirfd, ".", O_RDONLY | O_DIRECTORY | O_CLOEXEC);
633 if (lfd < 0) return 1;
634 DIR *d = fdopendir(lfd); /* owns lfd */
635 if (d == NULL) { close(lfd); return 1; }
636
637 int rc = 0;
638 struct dirent *e;
639 while (rc == 0 && (e = readdir(d)) != NULL) {
640 const char *n = e->d_name;
641 if (strcmp(n, ".") == 0 || strcmp(n, "..") == 0) continue;
642 struct stat st;
643 if (fstatat(dirfd, n, &st, AT_SYMLINK_NOFOLLOW) != 0) { rc = 1; break; }
644 if (S_ISDIR(st.st_mode)) {
645 int sub = openat(dirfd, n, O_RDONLY | O_DIRECTORY | O_NOFOLLOW | O_CLOEXEC);
646 if (sub < 0) { rc = 1; break; }
647 rc = ov_cli_reset_contents(sub);
648 close(sub);
649 if (rc == 0 && unlinkat(dirfd, n, AT_REMOVEDIR) != 0) rc = 1;
650 } else if (unlinkat(dirfd, n, 0) != 0) {
651 rc = 1;
652 }
653 }
654 closedir(d);
655 return rc;
656}
