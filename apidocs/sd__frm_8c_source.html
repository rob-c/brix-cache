1/*
2 * sd_frm.c - the nearline (tape/MSS) backend FS driver (SP5). See header.
3 *
4 * The driver is a thin residency layer over a pluggable MSS adapter. A read of an
5 * offline object RECALLS it into the MSS online buffer (synchronously in this
6 * increment - the stub recall is a local copy), then serves from the buffer via a
7 * real fd; a staged write lands in the online buffer and is MIGRATED to tape on
8 * commit. The composing registry requires a cache tier in front (G8), so the
9 * recall is in practice the cache miss-fill sourced from tape.
10 *
11 * The built-in "stub" MSS adapter simulates tape with two local directories under
12 * a base: <base>/<key> is "on tape" (offline) and <base>/.online/<key> is the
13 * online buffer. It is what the SP5 tests drive; a real MSS (exec stagecmd, HPSS,
14 * CTA) is another adapter selected on the tape:// store-URL.
15 */

50/* Bounded synchronous recall: ensure `key` is online in the MSS buffer. Returns 0
51 * (online), or -1 (errno: ENOENT absent, EAGAIN still in-flight, EIO error). A
52 * genuinely slow MSS would return EAGAIN and the cache tier would park the open on
53 * the stage_engine waiter (the deferred async path); the stub completes at once. */
54static int
55frm_ensure_online(sd_frm_state *st, const char *key)
56{
57 off_t sz = 0;
58 time_t mt = 0;
59 int res = st->mss->residency(st->mss_ctx, key, &sz, &mt);
60
61 if (res == BRIX_RESIDENCY_ONLINE) {
62 return 0;
63 }
64 if (res == BRIX_RESIDENCY_ABSENT) {
65 errno = ENOENT;
66 return -1;
67 }
68 if (st->mss->recall_begin(st->mss_ctx, key) != 0) {
69 errno = EIO;
70 return -1;
71 }
72 /* One poll: a synchronous adapter (begin copied) is online now; an async MSS is
73 * still staging, so return EAGAIN and let the HTTP plane answer 202 (the open
74 * "parks" via client retry, §9.2). A later retry re-polls and completes. */
75 {
76 int p = st->mss->recall_poll(st->mss_ctx, key);
77
78 if (p == 1) {
79 return 0;
80 }
81 if (p < 0) {
82 errno = EIO;
83 return -1;
84 }
85 }
86 errno = EAGAIN;
87 return -1;
88}

90static brix_sd_obj_t *
91sd_frm_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
92 mode_t mode, int *err_out)
93{
94 sd_frm_state *st = SD_FRM_ST(inst);
95 brix_sd_obj_t *o;
96 int fd;
97
98 (void) mode;
99 if ((sd_flags & BRIX_SD_O_WRITE) != 0) {
100 /* writes go through the staged path (migrate); a direct write-open is not
101 * supported on a nearline backend. */
102 if (err_out) { *err_out = EROFS; }
103 return NULL;
104 }
105 if (frm_ensure_online(st, path) != 0) {
106 if (err_out) { *err_out = errno; }
107 return NULL;
108 }
109 fd = st->mss->open_online(st->mss_ctx, path);
110 if (fd < 0) {
111 if (err_out) { *err_out = errno ? errno : EIO; }
112 return NULL;
113 }
114 o = calloc(1, sizeof(*o));
115 if (o == NULL) {
116 (void) close(fd);
117 if (err_out) { *err_out = ENOMEM; }
118 return NULL;
119 }
120 o->driver = inst->driver;
121 o->inst = inst;
122 o->fd = fd;
123 o->heap_shell = 1;
124 {
125 struct stat sb;
126 if (fstat(fd, &sb) == 0) {
127 o->snap.size = sb.st_size;
128 o->snap.mtime = sb.st_mtime;
129 o->snap.mode = sb.st_mode;
130 o->snap.is_reg = 1;
131 }
132 }
133 return o;
134}

195/* Residency (the VFS residency seam, phase-64 §9) — classify `key` via the MSS
196 * adapter WITHOUT initiating a recall, mapping the adapter's residency model onto
197 * the SD residency enum the protocol handlers consume. ABSENT ⇒ LOST (errno ENOENT
198 * so the seam can surface a missing object); ONLINE/NEARLINE/OFFLINE pass through. */
199static ngx_int_t
200sd_frm_residency(brix_sd_instance_t *inst, const char *key,
201 brix_sd_residency_t *out)
202{
203 sd_frm_state *st = SD_FRM_ST(inst);
204 off_t sz = 0;
205 time_t mt = 0;
206 int res = st->mss->residency(st->mss_ctx, key, &sz, &mt);
207
208 switch (res) {
209 case BRIX_RESIDENCY_ONLINE: *out = BRIX_SD_RES_ONLINE; break;
210 case BRIX_RESIDENCY_NEARLINE: *out = BRIX_SD_RES_NEARLINE; break;
211 case BRIX_RESIDENCY_OFFLINE: *out = BRIX_SD_RES_OFFLINE; break;
212 default: errno = ENOENT; return NGX_ERROR; /* ABSENT */
213 }
214 return NGX_OK;
215}

217static ngx_int_t
218sd_frm_recall(brix_sd_instance_t *inst, const char *key, char reqid_out[40])
219{
220 sd_frm_state *st = SD_FRM_ST(inst);
221
222 if (reqid_out != NULL) {
223 reqid_out[0] = '\0'; /* synchronous recall: no parking handle */
224 }
225 if (frm_ensure_online(st, key) == 0) {
226 return NGX_OK; /* online now - the cache tier does a normal fill */
227 }
228 return (errno == EAGAIN) ? NGX_AGAIN : NGX_ERROR;
229}

243 if (fd < 0) {

254 return NULL;

258 ngx_cpystrn((u_char *) ss->key, (u_char *) final_path, sizeof(ss->key));
259 h->inst = inst;

261 return h;

272static ngx_int_t
273sd_frm_staged_commit(brix_sd_staged_t *st, int noreplace)
274{
275 sd_frm_staged_state *ss = st->state;
276 int rc;
277
278 (void) noreplace;
279 if (ss->fd >= 0) {
280 (void) close(ss->fd);
281 ss->fd = -1;
282 }
283 /* Publish: migrate the online-buffer object to tape. */
284 rc = ss->fst->mss->migrate(ss->fst->mss_ctx, ss->key);
285 free(ss);
286 free(st);
287 return (rc == 0) ? NGX_OK : NGX_ERROR;

321brix_sd_instance_t *
322brix_sd_frm_create(const char *adapter, const char *location, ngx_log_t *log)
323{
324 brix_sd_instance_t *inst;
325 sd_frm_state *st;
326
327 if (location == NULL || location[0] == '\0') {
328 errno = EINVAL;
329 return NULL;
330 }
331 inst = calloc(1, sizeof(*inst));
332 st = calloc(1, sizeof(*st));
333 if (inst == NULL || st == NULL) {
334 free(inst);
335 free(st);
336 errno = ENOMEM;
337 return NULL;
338 }
339 st->log = log;
340
341 /* "exec" drives a real HSM via $BRIX_FRM_STAGECMD (the classic FRM model). */
342 if (adapter != NULL && ngx_strcmp(adapter, "exec") == 0) {
343 const char *cmd = getenv("BRIX_FRM_STAGECMD");
344
345 if (cmd == NULL || cmd[0] == '\0') {
346 ngx_log_error(NGX_LOG_WARN, log, 0,
347 "xrootd frm: the \"exec\" MSS adapter needs $BRIX_FRM_STAGECMD "
348 "(the stage command); falling back to the built-in stub");
349 } else {
350 st->mss_ctx = brix_mss_exec_create(location, cmd, log);
351 if (st->mss_ctx == NULL) {
352 free(inst);
353 free(st);
354 errno = ENOMEM;
355 return NULL;
356 }
357 st->mss = &brix_mss_exec_adapter;
358 ngx_log_error(NGX_LOG_NOTICE, log, 0,
359 "xrootd frm: exec MSS adapter (stagecmd=%s, online buffer=%s)",
360 cmd, location);
361 }
362 }
363
364 /* Default / fallback: the built-in local-dir stub simulator. */
365 if (st->mss == NULL) {
366 if (adapter != NULL && adapter[0] != '\0'
367 && ngx_strcmp(adapter, "stub") != 0
368 && ngx_strcmp(adapter, "exec") != 0)
369 {
370 ngx_log_error(NGX_LOG_WARN, log, 0,
371 "xrootd frm: MSS adapter \"%s\" is not yet implemented "
372 "(phase-64 SP5: hpss/cta); using the built-in stub", adapter);
373 }
374 st->mss_ctx = brix_mss_stub_create(location, log);
375 if (st->mss_ctx == NULL) {
376 free(inst);
377 free(st);
378 errno = ENOMEM;
379 return NULL;
380 }
381 st->mss = &brix_mss_stub_adapter;
382 }
383
384 inst->driver = &brix_sd_frm_driver;
385 inst->log = log;
386 inst->pool = NULL;
387 inst->state = st;
388 return inst;
389}
