1/*
2 * methods_basic.c - OPTIONS and HEAD handlers.
3 *
4 * WHAT: Implements HTTP OPTIONS and HEAD responses for WebDAV resources. OPTIONS advertises supported DAV capabilities (version 1+2) and method Allow list derived from write permission configuration. HEAD returns resource metadata without body transfer. (PROPPATCH lives in the sibling methods_proppatch.c.)
5 *
6 * WHY: HTTP OPTIONS is required for pre-flight CORS validation (webdav_add_cors_headers integration) and DAV capability discovery. RFC 4918 §5.3 requires OPTIONS responses to include Allow header listing enabled methods. HEAD provides lightweight resource metadata access without body transfer overhead — essential for fd-cache optimization in GET operations (avoiding full file read).
7 *
8 * HOW: OPTIONS handler sets DAV: "1, 2" header, constructs Allow list from conf->common.allow_write (read-only vs write-enabled), pushes MS-Author-Via: "DAV" header for Microsoft client compatibility, sends via ngx_http_send_special with NGX_HTTP_LAST (no body). HEAD handler uses webdav_resolve_stat composition helper for path+stat, sets Content-Length based on file size (0 for directories), last_modified_time, allow_ranges flag (disabled for directories), Content-Type (httpd/unix-directory vs application/octet-stream), and optional ETag via webdav_add_etag.
9 */

29/*
30 * WHAT: Handle HTTP OPTIONS request — advertise DAV capabilities (version 1+2) and method Allow list for this location.
31 *
32 * WHY: RFC 4918 §5.3 requires OPTIONS responses to include DAV header listing supported versions and Allow header listing enabled methods. CORS pre-flight validation depends on OPTIONS response (webdav_add_cors_headers integration). MS-Author-Via: "DAV" header enables Microsoft Office client compatibility for WebDAV-based document storage workflows.
33 *
34 * HOW: Set status 200 OK with zero Content-Length, push DAV: "1, 2" header (DAV version 1 and extension), construct Allow list from conf->common.allow_write flag (read-only: OPTIONS+GET+HEAD+PROPFIND; write-enabled: adds PUT+DELETE+MKCOL+MOVE+COPY), push MS-Author-Via: "DAV" for Microsoft client compatibility. Send headers via ngx_http_send_header() then complete with ngx_http_send_special(r, NGX_HTTP_LAST) — no body required for OPTIONS response.
35 */

89/*
90 * WHAT: Handle HTTP HEAD request — return resource metadata headers without body transfer. Uses webdav_resolve_stat composition helper for path+stat lookup.
91 *
92 * WHY: HEAD provides lightweight resource metadata access without full file read overhead — essential for fd-cache optimization in GET operations (avoiding expensive stat+open syscall pairs when cache already holds valid fd). Also enables pre-flight content length validation before body transfer decisions. The send_body parameter controls whether to proceed with body generation after header response (used by PROPFIND/PROPPATCH integration paths).
93 *
94 * HOW: Resolve path + stat via webdav_resolve_stat composition helper, set Content-Length based on file type (0 for directories, st_size for files), last_modified_time from sb.st_mtime, allow_ranges flag (disabled for directories per RFC 7233 §14.1). Set Content-Type (httpd/unix-directory vs application/octet-stream). Add ETag via webdav_add_etag only for non-directories. Send headers via ngx_http_send_header() — if send_body=0 or r->header_only=1, complete with NGX_HTTP_LAST without body transfer; otherwise return NGX_OK allowing downstream body generation.
95 */
96/*
97 * WHAT: Set the resource-metadata response headers (status, content length,
98 * last-modified, range support, Content-Type) for a stat'd HEAD/GET target.
99 *
100 * WHY: The status/length/type decision differs only by file-vs-directory and is
101 * shared verbatim by the HEAD orchestrator; isolating it keeps the caller flat
102 * and puts the directory-vs-file branching in one nameable place.
103 *
104 * HOW: Directories advertise zero length, no ranges, and the WebDAV
105 * "httpd/unix-directory" Content-Type; regular files advertise their size,
106 * range support, and defer Content-Type to the nginx types{} block. Returns
107 * NGX_OK, or NGX_HTTP_INTERNAL_SERVER_ERROR on header-list allocation failure.
108 */
109static ngx_int_t
110webdav_head_emit_metadata(ngx_http_request_t *r, const struct stat *sb)
111{
112 ngx_table_elt_t *h;
113
114 r->headers_out.status = NGX_HTTP_OK;
115 r->headers_out.content_length_n = S_ISDIR(sb->st_mode) ? 0 : sb->st_size;
116 r->headers_out.last_modified_time = sb->st_mtime;
117 r->allow_ranges = S_ISDIR(sb->st_mode) ? 0 : 1;
118
119 if (S_ISDIR(sb->st_mode)) {
120 h = ngx_list_push(&r->headers_out.headers);
121 if (h == NULL) {
122 return NGX_HTTP_INTERNAL_SERVER_ERROR;
123 }
124 h->hash = 1;
125 ngx_str_set(&h->key, "Content-Type");
126 ngx_str_set(&h->value, "httpd/unix-directory");
127 return NGX_OK;
128 }
129
130 ngx_http_set_content_type(r); /* uses nginx types{} block */
131 return NGX_OK;
132}

134/*
135 * WHAT: Inject the RFC 3230 Digest: header for a HEAD request that carried a
136 * Want-Digest: (XrdClHttp checksum pre-flight), doing nothing otherwise.
137 *
138 * WHY: Extracting the checksum branch removes the deepest nesting from the HEAD
139 * handler. The file is opened only when a checksum was actually requested, via
140 * the same confined VFS read open GET uses, so the Digest reflects exactly the
141 * bytes GET would serve (including an in-export symlink target) and repeated
142 * HEADs for one file stay cheap through the xattr cache.
143 *
144 * HOW: Return early unless the request context carries a want_cksum. Build a
145 * metered, impersonation-aware VFS read context (mirroring GET), open the
146 * resolved path, add the checksum header from the fd, and close. Open failure
147 * is non-fatal (no Digest emitted). Returns nothing — Digest emission is
148 * best-effort per the RFC.
149 */
150static void
151webdav_head_emit_digest(ngx_http_request_t *r, const char *path,
152 const struct stat *sb)
153{
154 xrdhttp_req_ctx_t *ctx;
155 ngx_http_brix_webdav_loc_conf_t *conf;
156 ngx_http_brix_webdav_req_ctx_t *wctx;
157 brix_vfs_ctx_t vctx;
158 brix_vfs_file_t *fh;
159 int vfs_err = 0;
160 int is_tls = 0;
161
162 if (S_ISDIR(sb->st_mode)) {
163 return;
164 }
165
166 ctx = xrdhttp_get_ctx(r);
167 if (ctx == NULL || !ctx->want_cksum[0]) {
168 return;
169 }
170
171 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
172 wctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
173
174 /* Open the same way GET does (confined VFS read open) so the Digest
175 * reflects exactly the bytes a GET would serve — including an
176 * in-export symlink target. Metered + impersonation-aware. */
177#if (NGX_HTTP_SSL)
178 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
179#endif
180 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log,
181 BRIX_PROTO_WEBDAV, conf->common.root_canon,
182 conf->cache_root_canon, conf->common.allow_write, is_tls,
183 (wctx != NULL) ? wctx->identity : NULL, path);
184
185 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
186 if (fh != NULL) {
187 (void) xrdhttp_add_checksum_header(r, brix_vfs_file_fd(fh), sb);
188 brix_vfs_close(fh, r->connection->log);
189 }
190}

192ngx_int_t
193webdav_handle_head(ngx_http_request_t *r, int send_body)
194{
195 char path[WEBDAV_MAX_PATH];
196 struct stat sb;
197 ngx_int_t rc;
198
199 rc = webdav_resolve_stat(r, path, sizeof(path), &sb);
200 if (rc != NGX_OK) {
201 return rc;
202 }
203
204 rc = webdav_head_emit_metadata(r, &sb);
205 if (rc != NGX_OK) {
206 return rc;
207 }
208
209 if (!S_ISDIR(sb.st_mode)) {
210 rc = brix_http_add_etag_header(r, sb.st_mtime, sb.st_size,
211 BRIX_ETAG_WEAK, 1);
212 if (rc != NGX_OK) {
213 return rc;
214 }
215 }
216
217 /* Inject Digest: header for Want-Digest: requests (RFC 3230 / XrdClHttp).
218 * Opens the file only when a checksum was actually requested, uses xattr
219 * cache so repeated HEAD requests for the same file are cheap. */
220 webdav_head_emit_digest(r, path, &sb);
221
222 ngx_http_send_header(r);
223
224 if (!send_body || r->header_only) {
225 return ngx_http_send_special(r, NGX_HTTP_LAST);
226 }
227
228 return NGX_OK;
229}
