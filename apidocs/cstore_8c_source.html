1/*
2 * cstore.c - the cache's one storage adapter (section 6.1). See header.
3 *
4 * The fill and serve calls are thin forwards onto the cache-store driver's
5 * staged-write / open / pread slots; evict/scan/freespace walk the store's
6 * namespace; cinfo_load/store keep the per-object record. SP1 implements LOCAL
7 * mode (a posix cache store with
8 * byte-identical ".cinfo" sidecars and a per-worker write-through L1); XATTR and
9 * SIDECAR modes land in SP2 and return ENOSYS here until then.

23/* ---- LOCAL-mode path helper ----------------------------------------------- */
24
25/* Join the store's local root with an export-relative `key` into out[cap] (the
26 * on-disk path of the cached object). Returns 0, or -1 when it would overflow. */
27static int
28cstore_local_path(const brix_cstore_t *cs, const char *key, char *out,
29 size_t cap)
30{
31 size_t rlen = strlen(cs->local_root);
32 int n;
33
34 while (rlen > 1 && cs->local_root[rlen - 1] == '/') {
35 rlen--;
36 }
37 n = snprintf(out, cap, "%.*s%s%s", (int) rlen, cs->local_root,
38 (key[0] == '/') ? "" : "/", key);
39 return (n > 0 && (size_t) n < cap) ? 0 : -1;
40}

44ngx_int_t
45brix_cstore_init(brix_cstore_t *cs, brix_sd_instance_t *store,
46 const char *local_root, int meta_mode, size_t l1_entries, int batch_cinfo,
47 ngx_log_t *log)
48{
49 if (cs == NULL || store == NULL || store->driver == NULL) {
50 errno = EINVAL;
51 return NGX_ERROR;
52 }
53 memset(cs, 0, sizeof(*cs));
54 cs->store = store;
55 cs->batch_cinfo = batch_cinfo;
56 cs->log = log;
57
58 if (local_root != NULL && local_root[0] != '\0') {
59 snprintf(cs->local_root, sizeof(cs->local_root), "%s", local_root);
60 }
61
62 /* AUTO resolves from the store: a known local dir -> LOCAL; a store that can
63 * carry xattrs -> XATTR; else SIDECAR (section 6.3). */
64 if (meta_mode == BRIX_CMETA_AUTO) {
65 if (cs->local_root[0] != '\0') {
66 meta_mode = BRIX_CMETA_LOCAL;
67 } else if (brix_sd_caps(store) & BRIX_SD_CAP_XATTR) {
68 meta_mode = BRIX_CMETA_XATTR;
69 } else {
70 meta_mode = BRIX_CMETA_SIDECAR;
71 }
72 }
73 cs->meta_mode = meta_mode;
74
75 cs->l1 = brix_cinfo_l1_create(l1_entries, log);
76 if (cs->l1 == NULL) {
77 return NGX_ERROR; /* errno set by create */
78 }
79
80 if (meta_mode == BRIX_CMETA_SIDECAR && cs->store->driver->staged_open == NULL
81 && log != NULL)
82 {
83 ngx_log_error(NGX_LOG_WARN, log, 0,
84 "xrootd cstore: SIDECAR cinfo mode needs a writable store (staged_open) "
85 "to persist hit state; this store has none");
86 }
87 return NGX_OK;
88}

100/* ---- fill spine ----------------------------------------------------------- */
101
102/* Create the key's parent directories on the store (mkdir -p), so a staged write
103 * for a nested key (e.g. "/sub/file") does not fail on a missing parent. Each
104 * level is created through the store driver; an already-present level is ignored. */
105static void
106cstore_make_parents(brix_cstore_t *cs, const char *key)
107{
108 char path[PATH_MAX];
109 size_t i;
110 size_t n = strlen(key);
111
112 if (cs->store->driver->mkdir == NULL || n == 0 || n >= sizeof(path)) {
113 return;
114 }
115 memcpy(path, key, n + 1);
116 for (i = 1; i < n; i++) { /* skip the leading '/' */
117 if (path[i] == '/') {
118 path[i] = '\0';
119 (void) cs->store->driver->mkdir(cs->store, path, 0700); /* EEXIST ok; 0700: svc-owned cache tree, not world-listable */
120 path[i] = '/';
121 }
122 }
123}

188ssize_t
189brix_cstore_serve_pread(int cache_fd, const uint8_t *bitmap, uint64_t nblocks,
190 uint32_t block_size, off_t size, void *buf, size_t len, off_t off,
191 brix_cstore_fill_block_fn fill, void *ctx)
192{
193 brix_sd_obj_t o;
194 uint64_t first, last, blk;
195
196 if (block_size == 0) {
197 errno = EINVAL;
198 return -1;
199 }
200 if (off >= size) {
201 return 0;
202 }
203 if (off + (off_t) len > size) {
204 len = (size_t) (size - off);
205 }
206 if (len == 0) {
207 return 0;
208 }
209
210 /* Range-fill: any block the read touches that the present bitmap does not
211 * cover is filled through the decorator's callback (source -> cache store). */
212 first = (uint64_t) off / block_size;
213 last = (uint64_t) (off + (off_t) len - 1) / block_size;
214 for (blk = first; blk <= last; blk++) {
215 int present = (bitmap != NULL && blk < nblocks
216 && brix_cache_cinfo_block_present(bitmap, blk));
217 if (!present && fill != NULL && fill(ctx, blk) != 0) {
218 return -1;
219 }
220 }
221
222 /* Serve from the cache object through the POSIX store driver (raw byte I/O
223 * stays in the backend, not here). */
224 brix_sd_posix_wrap(&o, cache_fd);
225 return o.driver->pread(&o, buf, len, off);
226}

228int
229brix_cstore_partial_open(brix_cstore_t *cs, const char *key, mode_t mode,
230 off_t size, char *path_out, size_t path_cap)
231{
232 char path[PATH_MAX];
233 struct stat sb;
234 int fd;
235
236 if (cs == NULL || cs->meta_mode != BRIX_CMETA_LOCAL) {
237 errno = ENOTSUP; /* partial caching needs a local RW object */
238 return -1;
239 }
240 if (cstore_local_path(cs, key, path, sizeof(path)) != 0) {
241 errno = ENAMETOOLONG;
242 return -1;
243 }
244 cstore_make_parents(cs, key);
245 fd = open(path, O_RDWR | O_CREAT | O_CLOEXEC, mode ? mode : 0600); /* svc-owned cache file; 0600 default */
246 if (fd < 0) {
247 return -1;
248 }
249 /* Size the (sparse) object so per-block pwrites land at their file offsets. */
250 if (fstat(fd, &sb) == 0 && sb.st_size < size && ftruncate(fd, size) != 0) {
251 int e = errno;
252
253 (void) close(fd);
254 errno = e;
255 return -1;
256 }
257 if (path_out != NULL && path_cap > 0) {
258 snprintf(path_out, path_cap, "%s", path);
259 }
260 return fd;
261}

263/* ---- evict ---------------------------------------------------------------- */
264
265ngx_int_t
266brix_cstore_evict(brix_cstore_t *cs, const char *key)
267{
268 if (cs == NULL || cs->store == NULL || key == NULL) {
269 errno = EINVAL;
270 return NGX_ERROR;
271 }
272
273 brix_cinfo_l1_drop(cs->l1, key);
274
275 /* Drop the unified record (both carriers: xattr + sidecar). LOCAL stores
276 * also get the path-side unlink so a sidecar next to the local object is
277 * gone even when the store driver namespaces keys differently. */
278 (void) brix_xmeta_remove(cs->store, key);
279 if (cs->meta_mode == BRIX_CMETA_LOCAL) {
280 char path[PATH_MAX];
281 char cipath[PATH_MAX];
282
283 if (cstore_local_path(cs, key, path, sizeof(path)) == 0
284 && brix_cache_cinfo_path(cipath, sizeof(cipath), path) == 0)
285 {
286 (void) unlink(cipath); /* vfs-seam-allow: svc-owned cache tree */
287 }
288 }
289
290 if (cs->store->driver->unlink != NULL) {
291 (void) cs->store->driver->unlink(cs->store, key, 0); /* idempotent */
292 }
293 return NGX_OK;
294}

298ngx_int_t
299brix_cstore_cinfo_load(brix_cstore_t *cs, const char *key,
300 brix_cache_cinfo_t *ci)
301{
302 brix_xmeta_t xm;
303 ngx_int_t rc;
304
305 if (cs == NULL || key == NULL || ci == NULL) {
306 errno = EINVAL;
307 return NGX_ERROR;
308 }
309 if (brix_cinfo_l1_get(cs->l1, key, ci) == NGX_OK) {
310 return NGX_OK; /* warm: no store round-trip */
311 }
312 if (cs->meta_mode == BRIX_CMETA_LOCAL) {
313 /* LOCAL: read via the path carrier so the record is shared with the
314 * partial-fill/writethrough path-based writers (same file, same lock
315 * discipline). */
316 char path[PATH_MAX];
317 uint8_t *bitmap = NULL;
318 size_t blen = 0;
319
320 if (cstore_local_path(cs, key, path, sizeof(path)) != 0) {
321 errno = ENAMETOOLONG;
322 return NGX_ERROR;
323 }
324 rc = brix_cache_cinfo_load(path, ci, &bitmap, &blen);
325 free(bitmap); /* L1 caches only the header */
326 if (rc == NGX_OK) {
327 brix_cinfo_l1_put(cs->l1, key, ci);
328 }
329 return rc;
330 }
331
332 rc = brix_xmeta_load(cs->store, key, &xm);
333 if (rc != NGX_OK) {
334 return rc;
335 }
336 brix_cache_cinfo_from_xmeta(&xm, ci);
337 brix_xmeta_free(&xm);
338 brix_cinfo_l1_put(cs->l1, key, ci);
339 return NGX_OK;
340}

342ngx_int_t
343brix_cstore_cinfo_store(brix_cstore_t *cs, const char *key,
344 const brix_cache_cinfo_t *ci)
345{
346 brix_xmeta_t xm;
347 ngx_int_t rc;
348
349 if (cs == NULL || key == NULL || ci == NULL) {
350 errno = EINVAL;
351 return NGX_ERROR;
352 }
353
354 if (cs->meta_mode == BRIX_CMETA_LOCAL) {
355 /* LOCAL: preserve an existing present-bitmap (partial fills advance it
356 * through the path-based writers under the per-file lock); a header
357 * write must not clobber it. */
358 char path[PATH_MAX];
359 brix_cache_cinfo_t old;
360 uint8_t *bitmap = NULL;
361 size_t existing_len = 0;
362 size_t need_len;
363
364 if (cstore_local_path(cs, key, path, sizeof(path)) != 0) {
365 errno = ENAMETOOLONG;
366 return NGX_ERROR;
367 }
368 need_len = brix_cache_cinfo_bitmap_len(ci->nblocks);
369 if (brix_cache_cinfo_load(path, &old, &bitmap, &existing_len)
370 != NGX_OK
371 || existing_len != need_len)
372 {
373 free(bitmap);
374 bitmap = NULL;
375 }
376 rc = brix_cache_cinfo_store(path, ci, bitmap,
377 bitmap != NULL ? need_len : 0);
378 free(bitmap);
379 if (rc == NGX_OK) {
380 brix_cinfo_l1_put(cs->l1, key, ci); /* write-through (§6.4) */
381 }
382 return rc;
383 }
384
385 if (brix_cache_cinfo_to_xmeta(ci, NULL, 0, &xm) != NGX_OK) {
386 return NGX_ERROR;
387 }
388 rc = brix_xmeta_save(cs->store, key, &xm);
389 brix_xmeta_free(&xm);
390 if (rc == NGX_OK) {
391 brix_cinfo_l1_put(cs->l1, key, ci); /* write-through (§6.4) */
392 }
393 return rc;
394}
