5/*
6 * evict_policy.c — cache eviction policy driver (two-pass LRU by occupancy).
7 *
8 * WHAT: Implements the eviction decision loop that keeps cache filesystem
9 * occupancy below cache_eviction_threshold. The public entry point
10 * brix_cache_evict_if_needed() and the per-file actor
11 * brix_cache_evict_one() live here; candidate collection, sorting, and
12 * free-space measurement are delegated to evict_candidates.c.
13 *
14 * WHY: A read-through cache fills indefinitely as clients request new files.
15 * Without bounded occupancy the cache disk fills up and fills start
16 * failing. This file is the policy half of that pressure relief — what to
17 * remove and when — separated from the mechanics of scanning the tree.
18 *
19 * HOW: Invoked from the cache-fill thread (thread.c) after each completed
20 * download. It double-checks occupancy (cheaply, then again after taking
21 * brix_cache_try_evict_lock to avoid worker races), collects the
22 * candidate set with brix_cache_collect_dir(), sorts oldest-first via
23 * brix_cache_candidate_cmp(), then evicts in two passes:
24 * Pass 1 — files larger than cache_max_file_size (skipped if 0).
25 * Pass 2 — everything else, oldest first.
26 * Each pass stops as soon as occupancy drops below the threshold. The
27 * protect_path (the file currently being filled) is never evicted.
28 * Runs in a thread-pool thread, not the event loop, so direct filesystem
29 * syscalls are permitted; metric updates use atomic adds.
30 */

32/*
33 * brix_cache_evict_one — remove a single cache file and account for it.
34 *
35 * Unlinks list->elts[idx].path (and its sidecar meta file via
36 * brix_cache_meta_path), marks the slot evicted, and accumulates the
37 * evicted_files/evicted_bytes counters. In manager_mode it also unregisters
38 * the freed path from the server registry so the manager stops advertising it.
39 * Refreshes *usage with brix_cache_fs_usage() so the caller's threshold loop
40 * sees the new occupancy. A missing file (ENOENT) is treated as already gone
41 * (NGX_OK); other unlink errors bump cache_eviction_errors_total but still
42 * return NGX_OK so eviction continues. Returns NGX_ERROR only if the post-unlink
43 * usage re-measurement fails.

134/*
135 * brix_cache_evict_if_needed — evict cached files when the filesystem
136 * occupancy exceeds the configured threshold.
137 *
138 * Called from the cache-fill thread after each completed download. The
139 * function is a no-op if the cache is not enabled or the threshold is 0.
140 *
141 * Eviction strategy (two-pass LRU):
142 * Pass 1 (optional): evict files larger than cache_max_file_size, oldest
143 * first. Large cold files free the most space fastest.
144 * Pass 2: evict any remaining files oldest-atime-first until the filesystem
145 * occupancy drops below the threshold.
146 *
147 * protect_path is the file currently being filled (if any) — it is not
148 * a candidate for eviction even if it is the oldest file.
149 *
150 * The occupancy_ppm threshold is checked twice: once before acquiring the
151 * lock and once after, to avoid unnecessary lock contention when multiple
152 * workers race to evict.
153 *
154 * Thread safety: this function runs in a thread-pool thread, not the nginx
155 * event loop. All filesystem operations are safe to call there. The
156 * metrics increments use ngx_atomic_fetch_add() which is thread-safe.
157 */
158/*
159 * brix_cache_purge_to_target — the eviction engine, decoupled from any fill
160 * task so both the on-fill safety net and the background watermark reaper can
161 * drive it. Collects the candidate set under cache_root, sorts oldest-first, and
162 * evicts in two passes (large files first, then plain LRU) until occupancy drops
163 * to `target_ppm` or the candidate set is exhausted. `ctx` (per-connection
164 * metrics) and `c` (the manager socket for path-unregister) are OPTIONAL — the
165 * timer passes NULL for both; the macro/branches are NULL-safe. The caller owns
166 * the cross-worker eviction lock and the threshold pre-check. Writes the evicted
167 * file/byte counts (when the out-pointers are non-NULL) and returns NGX_OK, or
168 * NGX_ERROR if a usage re-measurement failed mid-purge (partial work is kept).
169 */
170ngx_int_t
171brix_cache_purge_to_target(ngx_stream_brix_srv_conf_t *conf,
172 brix_ctx_t *ctx, ngx_connection_t *c, const char *protect_path,
173 ngx_uint_t target_ppm, ngx_log_t *log, ngx_uint_t *evicted_files_out,
174 uint64_t *evicted_bytes_out)
175{
176 brix_cache_fs_usage_t usage;
177 brix_cache_evict_list_t list;
178 struct stat root_st;
179 size_t i;
180 ngx_uint_t evicted_files = 0;
181 uint64_t evicted_bytes = 0;
182 ngx_int_t evict_rc = NGX_OK;
183
184 if (evicted_files_out != NULL) { *evicted_files_out = 0; }
185 if (evicted_bytes_out != NULL) { *evicted_bytes_out = 0; }
186
187 /* §14a: measure + device-scope the PHYSICAL cache root (tier-aware). A pure
188 * tier cache advertises cache_root "/", whose device differs from the store
189 * dir — using it for root_dev would make the same-device candidate filter drop
190 * every object. */
191 const char *phys_root = brix_cache_state_root(conf);
192
193 if (phys_root == NULL) {
194 brix_cache_metric_add(ctx, cache_eviction_errors_total, 1);
195 return NGX_ERROR;
196 }
197 if (brix_cache_usage_measure(brix_cache_storage_cstore(conf),
198 phys_root, &usage) != NGX_OK)
199 {
200 brix_cache_metric_add(ctx, cache_eviction_errors_total, 1);
201 return NGX_ERROR;
202 }
203 if (usage.occupancy_ppm <= target_ppm) {
204 return NGX_OK; /* already at/below target */
205 }
206 if (stat(phys_root, &root_st) != 0) {
207 brix_cache_metric_add(ctx, cache_eviction_errors_total, 1);
208 return NGX_ERROR;
209 }
210
211 ngx_memzero(&list, sizeof(list));
212 list.root_dev = root_st.st_dev;
213 list.protect_path = protect_path;
214 list.inst = brix_cache_storage(conf); /* store instance (removal) */
215 list.cstore = brix_cache_storage_cstore(conf); /* enumerate via the adapter */
216 /* §14a: walk the PHYSICAL cache root (tier-aware — the posix cache_store dir
217 * for the tier grammar, else cache_state_root/cache_root), so a pure tier cache
218 * (advertised cache_root "/") still evicts from the real store directory. */
219 list.cache_root = phys_root;
220 list.state_root = phys_root;
221
222 if (list.inst == NULL || list.cstore == NULL
223 || brix_cache_collect_dir(&list, "/", log) != NGX_OK)
224 {
225 brix_cache_metric_add(ctx, cache_eviction_errors_total, 1);
226 }
227
228 if (list.nelts > 1) {
229 qsort(list.elts, list.nelts, sizeof(list.elts[0]),
230 brix_cache_candidate_cmp);
231 }
232
233 /*
234 * Two-pass eviction:
235 * Pass 1 — files above cache_max_file_size first (oldest first): large
236 * cold files free the most space fastest.
237 * Pass 2 — remaining files oldest-first until occupancy ≤ target_ppm.
238 * Pass 1 is skipped when cache_max_file_size is 0 (no size limit), so pass 2
239 * degrades to plain single-pass LRU.
240 */
241 if (conf->cache_max_file_size > 0) {
242 for (i = 0; i < list.nelts && usage.occupancy_ppm > target_ppm
243 && evict_rc == NGX_OK; i++)
244 {
245 if (list.elts[i].size <= conf->cache_max_file_size) {
246 continue;
247 }
248 evict_rc = brix_cache_evict_one(conf, ctx, c, &list, i, log,
249 &usage, &evicted_files, &evicted_bytes);
250 }
251 }
252
253 for (i = 0; i < list.nelts && usage.occupancy_ppm > target_ppm
254 && evict_rc == NGX_OK; i++)
255 {
256 if (list.evicted[i]) {
257 continue;
258 }
259 evict_rc = brix_cache_evict_one(conf, ctx, c, &list, i, log,
260 &usage, &evicted_files, &evicted_bytes);
261 }
262
263 brix_cache_free_candidates(&list);
264
265 if (evicted_files_out != NULL) { *evicted_files_out = evicted_files; }
266 if (evicted_bytes_out != NULL) { *evicted_bytes_out = evicted_bytes; }
267 return evict_rc;
268}

270/*
271 * brix_cache_evict_if_needed — on-fill safety net: when cache_root occupancy
272 * exceeds cache_eviction_threshold, take the cross-worker lock and purge back to
273 * the threshold. Runs in the fill worker after each download; the proactive
274 * watermark reaper (reap_watermark.c) handles the quiet-but-full case.
275 */
276void
277brix_cache_evict_if_needed(brix_cache_fill_t *t, const char *protect_path,
278 ngx_log_t *log)
279{
280 brix_cache_fs_usage_t usage;
281 char lock_path[PATH_MAX];
282 ngx_uint_t threshold;
283 ngx_uint_t evicted_files = 0;
284 uint64_t evicted_bytes = 0;
285
286 if (t == NULL || t->conf == NULL || !t->conf->cache) {
287 return;
288 }
289
290 threshold = t->conf->cache_eviction_threshold;
291 if (threshold == 0 || threshold >= 1000000) {
292 return;
293 }
294
295 if (brix_cache_usage_measure(brix_cache_storage_cstore(t->conf),
296 (char *) t->conf->cache_root.data, &usage) != NGX_OK)
297 {
298 brix_cache_metric_add(t->ctx, cache_eviction_errors_total, 1);
299 return;
300 }
301 if (usage.occupancy_ppm <= threshold) {
302 return; /* cheap pre-lock check */
303 }
304
305 ngx_memzero(lock_path, sizeof(lock_path));
306 if (brix_cache_try_evict_lock(t->conf, lock_path, sizeof(lock_path),
307 log) != NGX_OK)
308 {
309 return;
310 }
311
312 if (brix_cache_purge_to_target(t->conf, t->ctx, t->c, protect_path,
313 threshold, log, &evicted_files, &evicted_bytes) == NGX_OK
314 && evicted_files > 0)
315 {
316 brix_cache_metric_add(t->ctx, cache_evictions_total, evicted_files);
317 brix_cache_metric_add(t->ctx, cache_evicted_bytes_total,
318 evicted_bytes);
319 ngx_log_error(NGX_LOG_NOTICE, log, 0,
320 "brix: cache eviction (on-fill) removed %ui files, "
321 "%uL bytes, threshold=0.%06ui",
322 evicted_files, (uint64_t) evicted_bytes, threshold);
323 }
324
325 brix_cache_evict_unlock(lock_path);
326}
