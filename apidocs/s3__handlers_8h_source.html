41/*
42 * Recursively scan dir_path (filesystem), appending object/CommonPrefix entries
43 * into the growable `entries` array (elements are s3_entry_t) until it reaches
44 * max_entries. key_prefix is the key path accumulated so far; filter_prefix/
45 * delimiter apply the ListObjects prefix/delimiter semantics (NULL/"" = none).
46 *
47 * phase-45 W1: classification uses readdir d_type (an lstat fallback only on
48 * DT_UNKNOWN), and NO size/mtime/etag stat is done here — those are filled
49 * lazily by s3_entry_fill_stat() for the emitted page only. Key strings are
50 * pooled from entries->pool at their true length. Directory sentinels are
51 * omitted; symlinks are never listed or traversed. Returns entries->nelts.
52 */
53int s3_walk(ngx_log_t *log, const char *root, const char *dir_path,
54 const char *key_prefix, const char *filter_prefix, const char *delimiter,
55 ngx_array_t *entries, int max_entries);
56/* qsort(3) comparator: lexicographic strcmp on s3_entry_t.key (a char *). */
57int entry_cmp(const void *a, const void *b);
58/*
59 * phase-45 W1: lazily fill size/mtime/etag for one emitted OBJECT entry by
60 * lstat'ing (confined) root + "/" + e->key. No-op for CommonPrefixes. Returns
61 * NGX_OK (filled) or NGX_DECLINED (entry vanished or is no longer a regular
62 * file — the caller skips it, matching the eager walker's stat-failure skip).
63 */
64ngx_int_t s3_entry_fill_stat(ngx_pool_t *pool, ngx_log_t *log,
65 const char *root, s3_entry_t *e);
