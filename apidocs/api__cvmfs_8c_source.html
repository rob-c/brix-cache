1/*
2 * api_cvmfs.c - cvmfs:// site-cache section of the dashboard JSON API.
3 *
4 * WHAT: serialise the phase-68 cvmfs SHM metrics block (shm->cvmfs) as the
5 * "cvmfs" snapshot section and the /xrootd/api/v1/cvmfs endpoint:
6 * aggregate counters (request mix by class, LAN-out vs WAN-in bytes,
7 * fills/failures/verify/negative-hits/failovers) plus the bounded
8 * per-repository and per-upstream slot tables.
9 * WHY: a CVMFS site cache is judged on its hit ratio and its WAN-saved
10 * factor; the Prometheus exporter (metrics/cvmfs.c) has the numbers but
11 * the dashboard showed only a single bytes-out total. This file gives
12 * the dashboard UI the same view, one JSON fetch away.
13 * HOW: read lock-free from the metrics SHM zone exactly like
14 * dashboard_fill_cache; slot tables follow the exporter's rules -
15 * only READY slots, and a claim-race duplicate name is emitted once
16 * (lowest index wins). Repo fqrns and upstream hosts arrive from the
17 * wire/config, so the anonymous view redacts them (cluster-host
18 * pattern) while keeping the counters.
19 */
20#include "dashboard_api_internal.h"

31/* 1 iff an EARLIER READY repo slot carries the same name (claim-race
32 * duplicate - mirror of the exporter's lowest-index convergence rule). */
33static int
34cvmfs_json_repo_is_dup(const ngx_brix_cvmfs_repo_metrics_t *repos,
35 ngx_uint_t i)
36{
37 ngx_uint_t j;
38
39 for (j = 0; j < i; j++) {
40 if (repos[j].state == BRIX_CVMFS_REPO_READY
41 && strcmp(repos[j].name, repos[i].name) == 0)
42 {
43 return 1;
44 }
45 }
46 return 0;
47}

68/*
69 * Build the {"cas":n,"manifest":n,"geo":n,"reject":n} request-mix object from
70 * one per-class counter array. Returns NULL on OOM (json_object_set_new in
71 * the caller absorbs a NULL child).
72 */
73static json_t *
74cvmfs_json_requests(const ngx_atomic_t counts[BRIX_CVMFS_CLASS_COUNT])
75{
76 json_t *obj = json_object();
77 ngx_uint_t cls;
78
79 if (!obj) { return NULL; }
80 for (cls = 0; cls < BRIX_CVMFS_CLASS_COUNT; cls++) {
81 json_object_set_new(obj, cvmfs_json_class_names[cls],
82 json_integer((json_int_t) counts[cls]));
83 }
84 return obj;
85}

157/*
158 * dashboard_fill_cvmfs - adds the cvmfs aggregate counters plus the "repos"
159 * and "upstreams" arrays directly to `target`. Called with target=root for
160 * v1/cvmfs (flat shape) and with target=sub-object for the snapshot (nested
161 * under "cvmfs"). "enabled" is inferred from work done (any request seen),
162 * matching the write_through inference in dashboard_fill_cache - the cvmfs
163 * module's conf lives in another module and is not visible here.
164 */
165void
166dashboard_fill_cvmfs(json_t *target, ngx_uint_t redact)
167{
168 ngx_brix_metrics_t *met;
169 ngx_brix_cvmfs_metrics_t *c;
170 json_t *repos, *ups;
171 ngx_uint_t i;
172 uint64_t requests = 0;
173
174 if (ngx_brix_shm_zone == NULL
175 || ngx_brix_shm_zone->data == NULL
176 || ngx_brix_shm_zone->data == (void *) 1)
177 {
178 json_object_set_new(target, "enabled", json_false());
179 json_object_set_new(target, "repos", json_array());
180 json_object_set_new(target, "upstreams", json_array());
181 return;
182 }
183
184 met = ngx_brix_shm_zone->data;
185 c = &met->cvmfs;
186
187 json_object_set_new(target, "requests", cvmfs_json_requests(c->requests_total));
188 for (i = 0; i < BRIX_CVMFS_CLASS_COUNT; i++) {
189 requests += (uint64_t) c->requests_total[i];
190 }
191 json_object_set_new(target, "enabled", requests ? json_true() : json_false());
192
193 json_object_set_new(target, "negative_hits_total",
194 json_integer((json_int_t) c->negative_hits_total));
195 json_object_set_new(target, "fills_total",
196 json_integer((json_int_t) c->fills_total));
197 json_object_set_new(target, "fill_failures_total",
198 json_integer((json_int_t) c->fill_failures_total));
199 json_object_set_new(target, "verify_failures_total",
200 json_integer((json_int_t) c->verify_failures_total));
201 json_object_set_new(target, "origin_failovers_total",
202 json_integer((json_int_t) c->origin_failovers_total));
203 json_object_set_new(target, "secure_requests_total",
204 json_integer((json_int_t) c->secure_requests_total));
205
206 /* LAN out split by disposition + WAN in: hit ratio and WAN-saved factor
207 * are one division away for the UI. */
208 json_object_set_new(target, "bytes_served_hit_total",
209 json_integer((json_int_t) c->bytes_served_hit_total));
210 json_object_set_new(target, "bytes_served_fill_total",
211 json_integer((json_int_t) c->bytes_served_fill_total));
212 json_object_set_new(target, "origin_bytes_total",
213 json_integer((json_int_t) c->origin_bytes_total));
214
215 repos = json_array();
216 if (repos) {
217 for (i = 0; i < BRIX_CVMFS_REPO_SLOTS; i++) {
218 const ngx_brix_cvmfs_repo_metrics_t *rm = &c->repos[i];
219
220 if (rm->state != BRIX_CVMFS_REPO_READY
221 || cvmfs_json_repo_is_dup(c->repos, i))
222 {
223 continue;
224 }
225 json_array_append_new(repos, cvmfs_json_repo(rm, redact));
226 }
227 }
228 json_object_set_new(target, "repos", repos);
229
230 ups = json_array();
231 if (ups) {
232 for (i = 0; i < BRIX_CVMFS_UPSTREAM_SLOTS; i++) {
233 const ngx_brix_cvmfs_upstream_metrics_t *u = &c->upstreams[i];
234
235 if (u->state != BRIX_CVMFS_REPO_READY
236 || cvmfs_json_upstream_is_dup(c->upstreams, i))
237 {
238 continue;
239 }
240 json_array_append_new(ups, cvmfs_json_upstream(u, redact));
241 }
242 }
243 json_object_set_new(target, "upstreams", ups);
244}

247/*
248 * Top-level builder for GET /xrootd/api/v1/cvmfs - the flat single-section
249 * document (schema envelope + anonymous flag + the cvmfs section fields).
250 */
251json_t *
252dashboard_build_v1_cvmfs(int64_t now_ms,
253 const ngx_http_brix_dashboard_loc_conf_t *conf, ngx_uint_t redact)
254{
255 json_t *root = dashboard_new_v1_root(now_ms, conf);
256
257 if (!root) { return NULL; }
258 json_object_set_new(root, "anonymous", redact ? json_true() : json_false());
259 dashboard_fill_cvmfs(root, redact);
260 return root;
261}
