1/*
2 * stream_family.c — per-server-slot Prometheus family emitters (split from
3 * stream.c for file-size hygiene; zero behaviour change).
4 *
5 * WHAT: The descriptor-table infrastructure (srv_family_desc_t + the shared
6 * slot-scan emitter) and every per-server stream-layer metric family it
7 * drives: connections, transfer-heap budget, payload/wire/frame bytes,
8 * fault-timeout reaps, io_uring backend, path-depth violations, SSI, and
9 * the config-generation reload gauge.
10 * WHY: Each per-server family repeats the identical slot-scan/label/emit loop;
11 * keeping the descriptor tables plus the one emitter here keeps the
12 * exposition format frozen in one place and the top-level scrape driver
13 * (stream.c) a flat call sequence.
14 * HOW: brix_export_prometheus_metrics() in stream.c calls these helpers in a
15 * frozen order; each reads its counters lock-free via
16 * ngx_atomic_fetch_add(..., 0) for an eventually-consistent snapshot.
17 */
18
19#include "metrics_internal.h"
20#include "stream_internal.h"

22/*
23 * srv_family_desc_t — one per-server-slot Prometheus family.
24 *
25 * WHAT: Describes a family emitted once per in-use server slot with
26 * {port,auth} labels: its verbatim HELP/TYPE preamble, its sample-line
27 * metric name, and where its counter lives in the per-server struct.
28 * WHY: Every stream-layer per-server family repeats the identical
29 * slot-scan/label/emit loop; descriptor tables plus one emitter keep the
30 * exporter a flat call sequence with the output format frozen in one
31 * place.
32 * HOW: `header` is printed verbatim (it may include a DEPRECATED notice
33 * before the HELP line); `field_off` is the offsetof() of the family's
34 * ngx_atomic_t counter inside ngx_brix_srv_metrics_t.
35 */
36typedef struct {
37 const char *header; /* verbatim "# HELP…\n# TYPE…\n" preamble */
38 const char *name; /* metric name on each sample line */
39 size_t field_off; /* offsetof(ngx_brix_srv_metrics_t, <counter>) */
40} srv_family_desc_t;
41
42/* Standard counter/gauge HELP+TYPE preambles (compile-time concatenation —
43 * byte-identical to the historical hand-written banners). */
44#define SRV_COUNTER_HDR(name, help) \
45 "# HELP " name " " help "\n# TYPE " name " counter\n"
46#define SRV_GAUGE_HDR(name, help) \
47 "# HELP " name " " help "\n# TYPE " name " gauge\n"
48
49/* Descriptor-table entry: preamble, sample-line name, counter field. */
50#define SRV_FAMILY(hdr, metric, field) \
51 { hdr, metric, offsetof(ngx_brix_srv_metrics_t, field) }
52
53/*
54 * metrics_emit_srv_family() — emit one per-server family.
55 *
56 * WHAT: Prints the family's HELP/TYPE preamble, then one sample line per
57 * in-use server slot, labelled {port,auth}.
58 * WHY: Single home for the slot-scan/label/emit loop shared by every
59 * per-server counter and gauge in this exporter.
60 * HOW: Locates the descriptor's ngx_atomic_t via `field_off` and reads it
61 * with an atomic fetch — an eventually-consistent snapshot, no lock.
62 */
63static void
64metrics_emit_srv_family(metrics_writer_t *mw, ngx_brix_metrics_t *shm,
65 const srv_family_desc_t *d)
66{
67 ngx_brix_srv_metrics_t *srv;
68 ngx_atomic_t *field;
69 ngx_uint_t i;
70 char port_str[16];
71
72 mw_printf(mw, "%s", d->header);
73
74 for (i = 0; i < BRIX_METRICS_MAX_SERVERS; i++) {
75 srv = &shm->servers[i];
76 if (!srv->in_use) { continue; }
77 ngx_snprintf((u_char *) port_str, sizeof(port_str), "%ui%Z", srv->port);
78 field = (ngx_atomic_t *) ((char *) srv + d->field_off);
79 mw_printf(mw,
80 "%s{port=\"%s\",auth=\"%s\"} %lu\n",
81 d->name, port_str, srv->auth,
82 (unsigned long) ngx_atomic_fetch_add(field, 0));
83 }
84}

86/*
87 * metrics_emit_srv_families() — emit a descriptor table in order.
88 *
89 * WHAT: Runs metrics_emit_srv_family() over each entry of a family table.
90 * WHY: Lets each metric-family helper below be a table plus one call.
91 * HOW: Straight iteration; table order == exposition order (frozen).
92 */
93static void
94metrics_emit_srv_families(metrics_writer_t *mw, ngx_brix_metrics_t *shm,
95 const srv_family_desc_t *tab, ngx_uint_t n)
96{
97 ngx_uint_t k;
98
99 for (k = 0; k < n; k++) {
100 metrics_emit_srv_family(mw, shm, &tab[k]);
101 }
102}
