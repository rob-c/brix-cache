24/* brix_cache_hex_ieq — case-insensitive equality of two hex digests * Origins vary in hex case (XRootD lowercases; some HTTP Digest headers upper).
25 * Returns 1 when equal ignoring case, 0 otherwise. Lengths must match too. */
26static int
27brix_cache_hex_ieq(const char *a, const char *b)
28{
29 size_t la = strlen(a);
30 size_t lb = strlen(b);
31
32 if (la != lb) {
33 return 0;
34 }
35 return strncasecmp(a, b, la) == 0;
36}

39/* brix_cache_verify_no_origin — policy verdict when the origin gave no digest.
40 *
41 * WHAT: Returns BRIX_CACHE_VERIFY_ERROR (after recording a kXR_ChkSumErr on t)
42 * under REQUIRE mode, otherwise BRIX_CACHE_VERIFY_UNVERIFIED. Security-neutral:
43 * no bytes are hashed here — this is the "nothing to compare against" branch.
44 *
45 * WHY: Isolates the missing-origin-digest policy fork so the orchestrator stays
46 * a flat sequence; the REQUIRE-vs-permissive decision lives in one place.
47 *
48 * HOW:
49 * 1. REQUIRE mode → set the error on t and return ERROR.
50 * 2. Any other mode → return UNVERIFIED (commit without a checksum).
51 */
52static brix_cache_verify_result_e
53brix_cache_verify_no_origin(brix_cache_fill_t *t, brix_cache_verify_mode_e mode)
54{
55 if (mode == BRIX_CACHE_VERIFY_REQUIRE) {
56 brix_cache_set_error(t, kXR_ChkSumErr, 0,
57 "cache verify: origin provided no checksum (require mode)");
58 return BRIX_CACHE_VERIFY_ERROR;
59 }
60 return BRIX_CACHE_VERIFY_UNVERIFIED;
61}

64/* brix_cache_verify_unsupported — policy verdict for an unimplemented algorithm.
65 *
66 * WHAT: Called when brix_checksum_hex_name_fd returned NGX_DECLINED (we do not
67 * implement the origin's algorithm). Returns BRIX_CACHE_VERIFY_ERROR (recording
68 * a kXR_ChkSumErr) under REQUIRE mode, else logs an INFO note and returns
69 * BRIX_CACHE_VERIFY_UNVERIFIED. No comparison is performed — verification was
70 * impossible, never bypassed.
71 *
72 * WHY: Keeps the "cannot verify this algorithm" fork out of the orchestrator so
73 * its control flow (and thus its complexity) is dominated by the real verify path.
74 *
75 * HOW:
76 * 1. REQUIRE mode → format the unsupported-algorithm message, set it on t,
77 * return ERROR.
78 * 2. Otherwise → if a log is present, emit the "committing unverified" INFO
79 * line, then return UNVERIFIED.
80 */
81static brix_cache_verify_result_e
82brix_cache_verify_unsupported(brix_cache_fill_t *t,
83 const brix_cache_digest_t *origin, brix_cache_verify_mode_e mode,
84 ngx_log_t *log, const char *part_path)
85{
86 if (mode == BRIX_CACHE_VERIFY_REQUIRE) {
87 char msg[96];
88 (void) snprintf(msg, sizeof(msg),
89 "cache verify: unsupported origin checksum algorithm \"%.15s\"",
90 origin->alg);
91 brix_cache_set_error(t, kXR_ChkSumErr, 0, msg);
92 return BRIX_CACHE_VERIFY_ERROR;
93 }
94 if (log != NULL) {
95 ngx_log_error(NGX_LOG_INFO, log, 0,
96 "brix: cache verify: origin checksum \"%s\" not supported "
97 "locally; committing \"%s\" unverified",
98 origin->alg, part_path);
99 }
100 return BRIX_CACHE_VERIFY_UNVERIFIED;
101}

104/* brix_cache_verify_report_mismatch — record and log a digest mismatch.
105 *
106 * WHAT: Formats the "<alg> mismatch (origin=… computed=…)" message, sets it as
107 * a kXR_ChkSumErr on t, and (when a log is present) emits the ERR "discarding
108 * fill" line. Side-effect only; the caller returns BRIX_CACHE_VERIFY_MISMATCH.
109 *
110 * WHY: The mismatch branch is the security-critical failure path; concentrating
111 * its exact error text and logging here keeps that wording byte-for-byte stable
112 * and out of the orchestrator's branch budget.
113 *
114 * HOW:
115 * 1. snprintf the mismatch detail using alg_used (already resolved to the
116 * normalized name or the origin's advertised name) and both digests.
117 * 2. brix_cache_set_error on t with kXR_ChkSumErr.
118 * 3. If log is non-NULL, log the failure at ERR level.
119 */
120static void
121brix_cache_verify_report_mismatch(brix_cache_fill_t *t, ngx_log_t *log,
122 const char *part_path, const char *alg_used,
123 const brix_cache_digest_t *origin, const char *hex)
124{
125 char msg[256];
126
127 (void) snprintf(msg, sizeof(msg),
128 "cache verify: %s mismatch (origin=%.64s computed=%.64s)",
129 alg_used, origin->hex, hex);
130 brix_cache_set_error(t, kXR_ChkSumErr, 0, msg);
131 if (log != NULL) {
132 ngx_log_error(NGX_LOG_ERR, log, 0,
133 "brix: cache verify FAILED for \"%s\": %s — discarding fill",
134 part_path, msg);
135 }
136}

139/* brix_cache_verify_emit — copy the verified algorithm and digest to outputs.
140 *
141 * WHAT: When out_alg / out_hex are non-NULL, writes the truncated, NUL-terminated
142 * algorithm name (<=15 chars) and hex digest (<=128 chars) into them. No-op for
143 * any output pointer that is NULL.
144 *
145 * WHY: Only reached on the verified-match path; extracting it keeps the two
146 * optional-output branches out of the orchestrator and mirrors the exact
147 * truncation limits the callers of brix_cache_verify_part rely on.
148 *
149 * HOW:
150 * 1. If out_alg is set, memcpy min(strlen(alg_used),15) bytes and NUL-terminate.
151 * 2. If out_hex is set, memcpy min(strlen(hex),128) bytes and NUL-terminate.
152 */
153static void
154brix_cache_verify_emit(char *out_alg, char *out_hex, const char *alg_used,
155 const char *hex)
156{
157 if (out_alg != NULL) {
158 size_t n = ngx_min(strlen(alg_used), 15u);
159 ngx_memcpy(out_alg, alg_used, n);
160 out_alg[n] = '\0';
161 }
162 if (out_hex != NULL) {
163 size_t n = ngx_min(strlen(hex), 128u);
164 ngx_memcpy(out_hex, hex, n);
165 out_hex[n] = '\0';
166 }
167}

170brix_cache_verify_result_e
171brix_cache_verify_part(brix_cache_fill_t *t, const char *part_path,
172 const brix_cache_digest_t *origin, brix_cache_verify_mode_e mode,
173 char *out_alg, char *out_hex)
174{
175 ngx_log_t *log = (t->c != NULL) ? t->c->log : NULL;
176 const char *alg_used;
177 char hex[129];
178 char norm[32];
179 ngx_int_t rc;
180 int fd;
181
182 if (mode == BRIX_CACHE_VERIFY_OFF) {
183 return BRIX_CACHE_VERIFY_UNVERIFIED;
184 }
185
186 /* No origin digest available. */
187 if (origin == NULL || origin->alg[0] == '\0' || origin->hex[0] == '\0') {
188 return brix_cache_verify_no_origin(t, mode);
189 }
190
191 fd = open(part_path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC | O_NOCTTY);
192 if (fd < 0) {
193 brix_cache_set_syserror(t, kXR_IOError,
194 "cache verify: part file open failed");
195 return BRIX_CACHE_VERIFY_ERROR;
196 }
197
198 /* Compute the SAME algorithm the origin advertised over the staged bytes. */
199 rc = brix_checksum_hex_name_fd(origin->alg, fd, part_path, log,
200 hex, sizeof(hex), norm, sizeof(norm));
201 close(fd);
202
203 if (rc == NGX_DECLINED) {
204 /* We do not implement the origin's algorithm — cannot verify. */
205 return brix_cache_verify_unsupported(t, origin, mode, log, part_path);
206 }
207
208 if (rc != NGX_OK) {
209 brix_cache_set_syserror(t, kXR_IOError,
210 "cache verify: local checksum computation failed");
211 return BRIX_CACHE_VERIFY_ERROR;
212 }
213
214 alg_used = norm[0] ? norm : origin->alg;
215
216 if (!brix_cache_hex_ieq(hex, origin->hex)) {
217 brix_cache_verify_report_mismatch(t, log, part_path, alg_used,
218 origin, hex);
219 return BRIX_CACHE_VERIFY_MISMATCH;
220 }
221
222 brix_cache_verify_emit(out_alg, out_hex, alg_used, hex);
223 return BRIX_CACHE_VERIFY_VERIFIED;
224}

227/* Phase-68 CVMFS-CAS self-verification — see verify.h. The expected digest is
228 * the CAS hex embedded in the key itself (raw served bytes = sha1(name)). */
229brix_cache_verify_result_e
230brix_cache_verify_cvmfs_cas(const char *part_path, const char *key,
231 ngx_log_t *log, char *out_alg, char *out_hex)
232{
233 cvmfs_url_info_t info;
234 char hex[129];
235 char norm[32];
236 ngx_int_t rc;
237 int fd;
238
239 if (cvmfs_classify_url(key, strlen(key), &info) != 0
240 || info.cls != CVMFS_URL_CAS)
241 {
242 return BRIX_CACHE_VERIFY_UNVERIFIED; /* not content-addressed */
243 }
244 if (info.cas_hex_len != 40) {
245 /* Only the sha1 (40-hex) convention is verifiable today; longer
246 * hashes would need the repo's hash-algorithm advertisement. */
247 return BRIX_CACHE_VERIFY_UNVERIFIED;
248 }
249
250 fd = open(part_path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC | O_NOCTTY); /* vfs-seam-allow: cache-store staging file, svc-owned domain */
251 if (fd < 0) {
252 return BRIX_CACHE_VERIFY_ERROR;
253 }
254 rc = brix_checksum_hex_name_fd("sha1", fd, part_path, log,
255 hex, sizeof(hex), norm, sizeof(norm));
256 close(fd); /* vfs-seam-allow: cache-store staging file, svc-owned domain */
257 if (rc != NGX_OK) {
258 return BRIX_CACHE_VERIFY_ERROR;
259 }
260
261 if (!brix_cache_hex_ieq(hex, info.cas_hex)) {
262 if (log != NULL) {
263 ngx_log_error(NGX_LOG_ERR, log, 0,
264 "brix: cvmfs-cas verify FAILED for \"%s\" "
265 "(name=%.40s computed=%.40s)\n"
266 " cause: origin transfer failed CAS verification — network "
267 "corruption between cache and Stratum-1\n"
268 " fix: check WAN path / middleboxes; object was "
269 "quarantined, client will retry",
270 key, info.cas_hex, hex);
271 }
272 return BRIX_CACHE_VERIFY_MISMATCH;
273 }
274
275 if (out_alg != NULL) {
276 ngx_memcpy(out_alg, "sha1", sizeof("sha1"));
277 }
278 if (out_hex != NULL) {
279 ngx_memcpy(out_hex, hex, 41);
280 }
281 return BRIX_CACHE_VERIFY_VERIFIED;
282}

286brix_cache_quarantine_part(const char *part_path,
287 const char *quarantine_dir, ngx_log_t *log)
288{
289 char dst[PATH_MAX];
290 const char *base;
291 int n;
292
293 if (quarantine_dir == NULL || quarantine_dir[0] == '\0') {
294 unlink(part_path); /* vfs-seam-allow: cache-store staging file, svc-owned domain */

296 }
297 base = strrchr(part_path, '/');
298 base = (base != NULL) ? base + 1 : part_path;
299 n = snprintf(dst, sizeof(dst), "%s/%s.%ld", quarantine_dir, base,
300 (long) time(NULL));
301 if (n < 0 || (size_t) n >= sizeof(dst)
302 || rename(part_path, dst) != 0) /* vfs-seam-allow: cache-store staging file, svc-owned domain */
303 {
304 unlink(part_path); /* vfs-seam-allow: cache-store staging file, svc-owned domain */
305 return;
306 }
307 if (log != NULL) {
308 ngx_log_error(NGX_LOG_WARN, log, 0,
309 "brix: cvmfs-cas: corrupt fill quarantined at \"%s\"", dst);
