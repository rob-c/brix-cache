2 * aws_chunked.c — AWS streaming (aws-chunked) request-body decoder (phase-43 W0).

4 * See aws_chunked.h for the wire format and rationale. This is a streaming
5 * state machine: it pulls the request-body chain buffer-by-buffer (spooled
6 * buffers are read in 64 KiB windows) so a large upload never lands fully in
7 * RAM, and pwrites only decoded payload bytes to the destination fd.

73/* Constant-time equality for two equal-length hex strings. */
74static int
75s3_ct_eq(const char *a, const char *b, size_t n)
76{
77 unsigned char d = 0;
78 size_t i;
79 for (i = 0; i < n; i++) {
80 d |= (unsigned char) (a[i] ^ b[i]);
81 }
82 return d == 0;
83}

85/* Extract the 64-hex chunk-signature from an accumulated chunk-extension
86 * ("chunk-signature=<64hex>"). Returns 1 + writes out[65] on success. */
87static int
88s3_chunk_ext_signature(const char *ext, size_t len, char out[65])
89{
90 static const char key[] = "chunk-signature=";
91 const size_t keyn = sizeof(key) - 1;
92 size_t i, j;
93
94 for (i = 0; i + keyn <= len; i++) {
95 if (ngx_strncasecmp((u_char *) ext + i, (u_char *) key, keyn) != 0) {
96 continue;
97 }
98 if (len - i - keyn < 64) {
99 return 0;
100 }
101 for (j = 0; j < 64; j++) {
102 char ch = ext[i + keyn + j];
103 int hex = (ch >= '0' && ch <= '9') || (ch >= 'a' && ch <= 'f')
104 || (ch >= 'A' && ch <= 'F');
105 if (!hex) {
106 return 0;
107 }
108 out[j] = ch;
109 }
110 out[64] = '\0';
111 return 1;
112 }
113 return 0;
114}

116/*
117 * Verify the just-completed chunk's signature. `c->sha` is finalised over the
118 * chunk's payload, the AWS4-HMAC-SHA256-PAYLOAD string-to-sign is built with the
119 * rolling previous signature, and the HMAC is constant-time compared with the
120 * client-provided chunk-signature. Returns 0 on match (and advances prev_sig);
121 * on mismatch/error sets ST_ERROR with the right status and returns -1.
122 */
123static int
124s3_chunk_verify(s3_chunk_ctx_t *c)
125{
126 char provided[65];
127 u_char chunk_hash[32];
128 char chunk_hash_hex[65];
129 u_char mac[32];
130 char computed[65];
131 u_char sts[64 + sizeof(c->vctx->amz_date) + sizeof(c->vctx->scope) + 256];
132 u_char *p;
133
134 if (!s3_chunk_ext_signature(c->ext, c->ext_len, provided)) {
135 s3_chunk_fail(c, NGX_HTTP_FORBIDDEN);
136 return -1;
137 }
138 if (!brix_sha256_stream_final(c->sha, chunk_hash)) {
139 s3_chunk_fail(c, NGX_HTTP_INTERNAL_SERVER_ERROR);
140 return -1;
141 }
142 brix_hex_encode(chunk_hash, 32, chunk_hash_hex);
143
144 p = ngx_snprintf(sts, sizeof(sts),
145 "AWS4-HMAC-SHA256-PAYLOAD\n%s\n%s\n%s\n%s\n%s",
146 c->vctx->amz_date, c->vctx->scope, c->prev_sig,
147 S3_EMPTY_SHA256_HEX, chunk_hash_hex);
148
149 if (!brix_hmac_sha256(c->vctx->signing_key, 32, sts,
150 (size_t) (p - sts), mac)) {
151 s3_chunk_fail(c, NGX_HTTP_INTERNAL_SERVER_ERROR);
152 return -1;
153 }
154 brix_hex_encode(mac, 32, computed);
155
156 if (!s3_ct_eq(computed, provided, 64)) {
157 s3_chunk_fail(c, NGX_HTTP_FORBIDDEN); /* SignatureDoesNotMatch */
158 return -1;
159 }
160
161 ngx_memcpy(c->prev_sig, provided, 65); /* roll forward */
162 return 0;
163}

165static int
166s3_hexdigit(unsigned char c, unsigned *out)
167{
168 if (c >= '0' && c <= '9') { *out = c - '0'; return 1; }
169 if (c >= 'a' && c <= 'f') { *out = c - 'a' + 10; return 1; }
170 if (c >= 'A' && c <= 'F') { *out = c - 'A' + 10; return 1; }
171 return 0;
172}

181/* Write decoded payload, enforcing the declared decoded length as an upper
182 * bound so a malformed stream cannot overrun. */
183static int
184s3_chunk_emit(s3_chunk_ctx_t *c, const u_char *p, size_t n)
185{
186 if (c->total_decoded + n > c->expected) {
187 s3_chunk_fail(c, NGX_HTTP_BAD_REQUEST); /* more data than declared */
188 return -1;
189 }
190 /* Write the decoded chunk through the storage seam (handles EINTR/short
191 * writes); a failure fails the whole upload. */
192 if (brix_vfs_pwrite_full(c->fd, p, n, c->write_off) != NGX_OK) {
193 s3_chunk_fail(c, NGX_HTTP_INTERNAL_SERVER_ERROR);
194 return -1;
195 }
196 c->write_off += (off_t) n;
197 c->total_decoded += (uint64_t) n;
198 return 0;
199}

201/* Parse a completed trailer line "name:value" and capture an x-amz-checksum-*. */
202static void
203s3_chunk_trailer_line(s3_chunk_ctx_t *c)
204{
205 static const char pfx[] = "x-amz-checksum-";
206 const size_t pfxn = sizeof(pfx) - 1;
207 char *colon;
208 size_t namelen, i;
209
210 if (c->trailer == NULL || c->line_len <= pfxn) {
211 return;
212 }
213 if (ngx_strncasecmp((u_char *) c->line, (u_char *) pfx, pfxn) != 0) {
214 return;
215 }
216 colon = memchr(c->line, ':', c->line_len);
217 if (colon == NULL) {
218 return;
219 }
220 namelen = (size_t) (colon - c->line);
221 if (namelen <= pfxn || (namelen - pfxn) >= sizeof(c->trailer->algo_token)) {
222 return;
223 }
224
225 for (i = 0; i < namelen - pfxn; i++) {
226 c->trailer->algo_token[i] = (char) ngx_tolower(c->line[pfxn + i]);
227 }
228 c->trailer->algo_token[namelen - pfxn] = '\0';
229
230 {
231 size_t vlen = c->line_len - namelen - 1; /* after ':' */
232 char *v = colon + 1;
233 while (vlen > 0 && (*v == ' ' || *v == '\t')) { v++; vlen--; }
234 if (vlen >= sizeof(c->trailer->value)) {
235 vlen = sizeof(c->trailer->value) - 1;
236 }
237 ngx_memcpy(c->trailer->value, v, vlen);
238 c->trailer->value[vlen] = '\0';
239 }
240}

242/* Feed one contiguous span through the state machine. Returns -1 on error. */
243static int
244s3_chunk_feed(s3_chunk_ctx_t *c, const u_char *p, const u_char *end)
245{
246 while (p < end && c->state != ST_DONE && c->state != ST_ERROR) {
247 u_char ch = *p;
248
249 switch (c->state) {
250
251 case ST_SIZE: {
252 unsigned v;
253 if (s3_hexdigit(ch, &v)) {
254 if (c->cur_size > (UINT64_MAX >> 4)) {
255 s3_chunk_fail(c, NGX_HTTP_BAD_REQUEST);
256 return -1;
257 }
258 c->cur_size = (c->cur_size << 4) | v;
259 c->size_digits = 1;
260 p++;
261 } else if (ch == ';') {
262 c->state = ST_EXT; p++;
263 } else if (ch == '\r') {
264 if (!c->size_digits) { s3_chunk_fail(c, NGX_HTTP_BAD_REQUEST); return -1; }
265 c->state = ST_SIZE_LF; p++;
266 } else {
267 s3_chunk_fail(c, NGX_HTTP_BAD_REQUEST); return -1;
268 }
269 break;
270 }
271
272 case ST_EXT:
273 /* Capture the chunk-signature extension (W6a verify) or, when
274 * verification is off, skip it to CR (legacy behaviour). */
275 if (ch == '\r') {
276 c->state = ST_SIZE_LF;
277 } else if (c->verify && c->ext_len < sizeof(c->ext)) {
278 c->ext[c->ext_len++] = (char) ch;
279 }
280 p++;
281 break;
282
283 case ST_SIZE_LF:
284 if (ch != '\n') { s3_chunk_fail(c, NGX_HTTP_BAD_REQUEST); return -1; }
285 p++;
286 c->chunk_remaining = c->cur_size;
287 if (c->cur_size == 0) {
288 /* Terminating 0-chunk: its signature covers the empty payload. */
289 if (c->verify && s3_chunk_verify(c) != 0) { return -1; }
290 c->state = ST_TRAILER; /* last chunk */
291 c->line_len = 0;
292 } else {
293 c->state = ST_DATA;
294 }
295 c->cur_size = 0;
296 c->size_digits = 0;
297 break;
298
299 case ST_DATA: {
300 size_t avail = (size_t) (end - p);
301 size_t take = (c->chunk_remaining < avail) ? (size_t) c->chunk_remaining
302 : avail;
303 if (take > 0) {
304 if (c->verify
305 && !brix_sha256_stream_update(c->sha, p, take))
306 {
307 s3_chunk_fail(c, NGX_HTTP_INTERNAL_SERVER_ERROR);
308 return -1;
309 }
310 if (s3_chunk_emit(c, p, take) != 0) { return -1; }
311 p += take;
312 c->chunk_remaining -= take;
313 }
314 if (c->chunk_remaining == 0) {
315 /* All payload of this chunk is hashed — verify its signature. */
316 if (c->verify && s3_chunk_verify(c) != 0) { return -1; }
317 c->state = ST_DATA_CR;
318 }
319 break;
320 }
321
322 case ST_DATA_CR:
323 if (ch != '\r') { s3_chunk_fail(c, NGX_HTTP_BAD_REQUEST); return -1; }
324 c->state = ST_DATA_LF; p++;
325 break;
326
327 case ST_DATA_LF:
328 if (ch != '\n') { s3_chunk_fail(c, NGX_HTTP_BAD_REQUEST); return -1; }
329 c->state = ST_SIZE; c->ext_len = 0; p++; /* reset ext for next hdr */
330 break;
331
332 case ST_TRAILER:
333 if (ch == '\n') {
334 /* Strip a trailing CR, then process the completed line. */
335 if (c->line_len > 0 && c->line[c->line_len - 1] == '\r') {
336 c->line_len--;
337 }
338 if (c->line_len == 0) {
339 c->state = ST_DONE; /* blank line terminates */
340 } else {
341 s3_chunk_trailer_line(c);
342 c->line_len = 0;
343 }
344 p++;
345 } else {
346 if (c->line_len < sizeof(c->line)) {
347 c->line[c->line_len++] = (char) ch;
348 }
349 p++; /* overlong trailer lines are truncated, not fatal */
350 }
351 break;
352
353 default:
354 p++;
355 break;
356 }
357 }
358 return (c->state == ST_ERROR) ? -1 : 0;
359}

361int
362s3_body_is_aws_chunked(ngx_http_request_t *r)
363{
364 ngx_table_elt_t *h;
365
366 h = brix_http_find_header(r, "x-amz-content-sha256",
367 sizeof("x-amz-content-sha256") - 1);
368 if (h != NULL && h->value.len >= sizeof("STREAMING-") - 1
369 && ngx_strncmp(h->value.data, (u_char *) "STREAMING-",
370 sizeof("STREAMING-") - 1) == 0)
371 {
372 return 1;
373 }
374
375 h = brix_http_find_header(r, "Content-Encoding",
376 sizeof("Content-Encoding") - 1);
377 if (h != NULL && h->value.len >= sizeof("aws-chunked") - 1) {
378 /* aws-chunked may appear alone or combined (e.g. "aws-chunked,gzip"). */
379 size_t i, n = h->value.len - (sizeof("aws-chunked") - 1);
380 for (i = 0; i <= n; i++) {
381 if (ngx_strncasecmp(h->value.data + i, (u_char *) "aws-chunked",
382 sizeof("aws-chunked") - 1) == 0)
383 {
384 return 1;
385 }
386 }
387 }
388 return 0;
389}

391int
392s3_aws_chunked_has_inner_coding(ngx_http_request_t *r)
393{
394 ngx_table_elt_t *h = brix_http_find_header(r, "Content-Encoding",
395 sizeof("Content-Encoding") - 1);
396 const u_char *p, *end;
397
398 if (h == NULL || h->value.len == 0) {
399 return 0;
400 }
401 p = h->value.data;
402 end = p + h->value.len;
403
404 /* Walk comma-separated content-codings; any token that is neither
405 * "aws-chunked" nor "identity" (nor empty) is an inner content-coding the
406 * streaming path cannot decode. */
407 while (p < end) {
408 const u_char *tok, *tend;
409 size_t len;
410
411 while (p < end && (*p == ' ' || *p == ',' || *p == '\t')) { p++; }
412 tok = p;
413 while (p < end && *p != ',') { p++; }
414 tend = p;
415 while (tend > tok && (tend[-1] == ' ' || tend[-1] == '\t')) { tend--; }
416 len = (size_t) (tend - tok);
417 if (len == 0) {
418 continue;
419 }
420 if (len == sizeof("aws-chunked") - 1
421 && ngx_strncasecmp((u_char *) tok, (u_char *) "aws-chunked", len) == 0) {
422 continue;
423 }
424 if (len == sizeof("identity") - 1
425 && ngx_strncasecmp((u_char *) tok, (u_char *) "identity", len) == 0) {
426 continue;
427 }
428 return 1; /* a real inner content-coding (gzip/zstd/...) is present */
429 }
430 return 0;
431}

433/* Feed one request-body buffer (memory or spooled) into the decoder. */
434static int
435s3_chunk_feed_buf(s3_chunk_ctx_t *c, ngx_buf_t *b, u_char *window)
436{
437 if (ngx_buf_in_memory(b)) {
438 return s3_chunk_feed(c, b->pos, b->last);
439 }
440
441 if (b->in_file) {
442 off_t off = b->file_pos;
443
444 while (off < b->file_last && c->state != ST_DONE
445 && c->state != ST_ERROR)
446 {
447 off_t want = b->file_last - off;
448 size_t got = 0;
449 if (want > S3_CHUNK_READ_WINDOW) {
450 want = S3_CHUNK_READ_WINDOW;
451 }
452 /* One window through the storage seam (EINTR/short-read handled). */
453 if (brix_vfs_pread_full(b->file->fd, window, (size_t) want, off,
454 &got) != NGX_OK)
455 {
456 s3_chunk_fail(c, NGX_HTTP_INTERNAL_SERVER_ERROR);
457 return -1;
458 }
459 if (got == 0) {
460 break;
461 }
462 if (s3_chunk_feed(c, window, window + got) != 0) {
463 return -1;
464 }
465 off += (off_t) got;
466 }
467 }
468 return (c->state == ST_ERROR) ? -1 : 0;
469}

471ngx_int_t
472s3_aws_chunked_decode_to_fd(ngx_http_request_t *r, ngx_fd_t dst_fd,
473 const char *log_path, uint64_t decoded_len_expected,
474 s3_chunk_trailer_t *trailer, int *http_status_out, u_char *window,
475 const s3_chunk_verify_t *verify)
476{
477 s3_chunk_ctx_t c;
478 ngx_chain_t *cl;
479 ngx_int_t rc = NGX_OK;
480
481 ngx_memzero(&c, sizeof(c));
482 c.state = ST_SIZE;
483 c.fd = dst_fd;
484 c.log_path = log_path;
485 c.log = r->connection->log;
486 c.expected = decoded_len_expected;
487 c.trailer = trailer;
488 if (trailer != NULL) {
489 trailer->algo_token[0] = '\0';
490 trailer->value[0] = '\0';
491 }
492
493 /* W6a: arm per-chunk verification; the seed signature is the first chunk's
494 * previous-signature. The streaming SHA-256 handle hashes each chunk's
495 * payload (works on the worker thread — per-ctx, no shared state). */
496 if (verify != NULL && verify->enabled) {
497 c.verify = 1;
498 c.vctx = verify;
499 ngx_cpystrn((u_char *) c.prev_sig, (u_char *) verify->seed_signature,
500 sizeof(c.prev_sig));
501 c.sha = brix_sha256_stream_new();
502 if (c.sha == NULL) {
503 *http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR;
504 return NGX_ERROR;
505 }
506 }
507
508 if (r->request_body == NULL || r->request_body->bufs == NULL) {
509 /* An empty streamed body is only valid when 0 bytes were declared. */
510 if (decoded_len_expected != 0) {
511 *http_status_out = NGX_HTTP_BAD_REQUEST;
512 rc = NGX_ERROR;
513 }
514 brix_sha256_stream_free(c.sha); /* NULL-safe when verify off */
515 return rc;
516 }
517
518 /* The caller may pass a pre-allocated scratch window (the offload path does,
519 * so the worker thread never touches r->pool); otherwise allocate one. */
520 if (window == NULL) {
521 window = ngx_palloc(r->pool, S3_CHUNK_READ_WINDOW);
522 if (window == NULL) {
523 *http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR;
524 brix_sha256_stream_free(c.sha);
525 return NGX_ERROR;
526 }
527 }
528
529 for (cl = r->request_body->bufs; cl != NULL && rc == NGX_OK; cl = cl->next) {
530 if (s3_chunk_feed_buf(&c, cl->buf, window) != 0) {
531 *http_status_out = c.http_status;
532 rc = NGX_ERROR;
533 }
534 }
535
536 /* The stream must have reached the terminating blank line and produced
537 * exactly the declared number of decoded bytes. */
538 if (rc == NGX_OK
539 && (c.state != ST_DONE || c.total_decoded != decoded_len_expected))
540 {
541 *http_status_out = NGX_HTTP_BAD_REQUEST;
542 rc = NGX_ERROR;
543 }
544
545 brix_sha256_stream_free(c.sha);
546 return rc;
547}
