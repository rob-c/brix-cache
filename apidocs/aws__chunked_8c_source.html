1/*
2 * aws_chunked.c — AWS streaming (aws-chunked) request-body decode orchestrator
3 * (phase-43 W0; phase-79 file-size split).
4 *
5 * See aws_chunked.h for the wire format and rationale. This file owns the
6 * decode orchestration: it sets up the framing context, drives the nginx
7 * request-body chain buffer-by-buffer (spooled buffers are read in 64 KiB
8 * windows) so a large upload never lands fully in RAM, and enforces the
9 * declared decoded length and terminated-stream postcondition. The byte-level
10 * framing state machine and per-chunk signature verification live in the
11 * sibling aws_chunked_parse.c (via s3_chunk_feed / s3_chunk_fail); the
12 * header-classification predicates live in aws_chunked_encoding.c.
13 */

27/* Feed one request-body buffer (memory or spooled) into the decoder. */
28static int
29s3_chunk_feed_buf(s3_chunk_ctx_t *c, ngx_buf_t *b, u_char *window)
30{
31 if (ngx_buf_in_memory(b)) {
32 return s3_chunk_feed(c, b->pos, b->last);
33 }
34
35 if (b->in_file) {
36 off_t off = b->file_pos;
37
38 while (off < b->file_last && c->state != ST_DONE
39 && c->state != ST_ERROR)
40 {
41 off_t want = b->file_last - off;
42 size_t got = 0;
43 if (want > S3_CHUNK_READ_WINDOW) {
44 want = S3_CHUNK_READ_WINDOW;
45 }
46 /* One window through the storage seam (EINTR/short-read handled). */
47 if (brix_vfs_pread_full(b->file->fd, window, (size_t) want, off,
48 &got) != NGX_OK)
49 {
50 s3_chunk_fail(c, NGX_HTTP_INTERNAL_SERVER_ERROR);
51 return -1;
52 }
53 if (got == 0) {
54 break;
55 }
56 if (s3_chunk_feed(c, window, window + got) != 0) {
57 return -1;
58 }
59 off += (off_t) got;
60 }
61 }
62 return (c->state == ST_ERROR) ? -1 : 0;
63}

81/*
82 * WHAT: initialise the decoder context and, when requested, arm verification.
83 * WHY: separates one-time setup (state seed, trailer clear, SHA handle) from the
84 * feed loop, keeping the orchestrator flat.
85 * HOW: zero-inits c, seeds the framing state, clears the trailer out-params, and
86 * — when verify is enabled — copies the seed signature and allocates the
87 * streaming SHA-256 handle. Returns NGX_OK, or NGX_ERROR (with
88 * *http_status_out set) if the SHA handle cannot be allocated.
89 */
90static ngx_int_t
91s3_chunk_dec_init(const s3_chunk_dec_t *d, s3_chunk_ctx_t *c)
92{
93 ngx_memzero(c, sizeof(*c));
94 c->state = ST_SIZE;
95 c->fd = d->dst_fd;
96 c->log_path = d->log_path;
97 c->log = d->r->connection->log;
98 c->expected = d->decoded_len_expected;
99 c->trailer = d->trailer;
100 if (d->trailer != NULL) {
101 d->trailer->algo_token[0] = '\0';
102 d->trailer->value[0] = '\0';
103 }
104
105 /* W6a: arm per-chunk verification; the seed signature is the first chunk's
106 * previous-signature. The streaming SHA-256 handle hashes each chunk's
107 * payload (works on the worker thread — per-ctx, no shared state). */
108 if (d->verify != NULL && d->verify->enabled) {
109 c->verify = 1;
110 c->vctx = d->verify;
111 ngx_cpystrn((u_char *) c->prev_sig, (u_char *) d->verify->seed_signature,
112 sizeof(c->prev_sig));
113 c->sha = brix_sha256_stream_new();
114 if (c->sha == NULL) {
115 *d->http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR;
116 return NGX_ERROR;
117 }
118 }
119 return NGX_OK;
120}

122/*
123 * WHAT: ensure a scratch read-window exists for the spooled-buffer path.
124 * WHY: the offload path pre-allocates one so the worker thread never touches
125 * r->pool; the inline path allocates lazily here.
126 * HOW: returns the caller-supplied window if non-NULL, otherwise allocates one
127 * from r->pool; on allocation failure sets *http_status_out and returns
128 * NULL.
129 */
130static u_char *
131s3_chunk_dec_window(const s3_chunk_dec_t *d)
132{
133 u_char *window;
134
135 if (d->window != NULL) {
136 return d->window;
137 }
138 window = ngx_palloc(d->r->pool, S3_CHUNK_READ_WINDOW);
139 if (window == NULL) {
140 *d->http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR;
141 }
142 return window;
143}

145/*
146 * WHAT: drive every request-body buffer through the state machine.
147 * WHY: isolates the buffer-chain drain (and its per-buffer error surfacing) from
148 * setup/finalisation.
149 * HOW: feeds each chain buffer through s3_chunk_feed_buf; on the first failure
150 * publishes c->http_status and returns NGX_ERROR, else NGX_OK.
151 */
152static ngx_int_t
153s3_chunk_dec_drain(const s3_chunk_dec_t *d, s3_chunk_ctx_t *c, u_char *window)
154{
155 ngx_chain_t *cl;
156
157 for (cl = d->r->request_body->bufs; cl != NULL; cl = cl->next) {
158 if (s3_chunk_feed_buf(c, cl->buf, window) != 0) {
159 *d->http_status_out = c->http_status;
160 return NGX_ERROR;
161 }
162 }
163 return NGX_OK;
164}

166/*
167 * WHAT: run the decode once the context is armed (body present).
168 * WHY: the non-empty-body case — window acquisition, drain, and the
169 * terminated/length postcondition — as one linear helper.
170 * HOW: acquires the window, drains the buffer chain, then requires the stream to
171 * have reached ST_DONE with exactly the declared byte count. Returns
172 * NGX_OK or NGX_ERROR with *http_status_out set.
173 */
174static ngx_int_t
175s3_chunk_dec_run(const s3_chunk_dec_t *d, s3_chunk_ctx_t *c)
176{
177 u_char *window = s3_chunk_dec_window(d);
178 ngx_int_t rc;
179
180 if (window == NULL) {
181 return NGX_ERROR;
182 }
183
184 rc = s3_chunk_dec_drain(d, c, window);
185 if (rc != NGX_OK) {
186 return rc;
187 }
188
189 /* The stream must have reached the terminating blank line and produced
190 * exactly the declared number of decoded bytes. */
191 if (c->state != ST_DONE || c->total_decoded != d->decoded_len_expected) {
192 *d->http_status_out = NGX_HTTP_BAD_REQUEST;
193 return NGX_ERROR;
194 }
195 return NGX_OK;
196}

198ngx_int_t
199s3_aws_chunked_decode_to_fd(ngx_http_request_t *r, ngx_fd_t dst_fd,
200 const char *log_path, uint64_t decoded_len_expected,
201 s3_chunk_trailer_t *trailer, int *http_status_out, u_char *window,
202 const s3_chunk_verify_t *verify)
203{
204 s3_chunk_dec_t d = {
205 .r = r,
206 .dst_fd = dst_fd,
207 .log_path = log_path,
208 .decoded_len_expected = decoded_len_expected,
209 .trailer = trailer,
210 .http_status_out = http_status_out,
211 .window = window,
212 .verify = verify,
213 };
214 s3_chunk_ctx_t c;
215 ngx_int_t rc;
216
217 if (s3_chunk_dec_init(&d, &c) != NGX_OK) {
218 return NGX_ERROR;
219 }
220
221 if (r->request_body == NULL || r->request_body->bufs == NULL) {
222 /* An empty streamed body is only valid when 0 bytes were declared. */
223 rc = NGX_OK;
224 if (decoded_len_expected != 0) {
225 *http_status_out = NGX_HTTP_BAD_REQUEST;
226 rc = NGX_ERROR;
227 }
228 } else {
229 rc = s3_chunk_dec_run(&d, &c);
230 }
231
232 brix_sha256_stream_free(c.sha); /* NULL-safe when verify off */
233 return rc;
234}
