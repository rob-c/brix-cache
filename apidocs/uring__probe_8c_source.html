6/* File: src/core/aio/uring_probe.c — io_uring detection, gating, and the §32
7 * startup fail-fast validator (phase-79 split from uring.c).
8 * WHAT: Owns the "should this build/host/worker use io_uring at all" decisions:
9 * the memoized runtime opcode probe, the per-block config scan folded into
10 * one per-worker verdict, and the config-time hard-fail for `brix_io_uring
11 * on`. The ring singleton and its bring-up live in uring.c; the CQE
12 * reaper lives in uring_reap.c.
13 *
14 * WHY: The backend must be invisible unless explicitly built (pkg-config
15 * liburing) and runtime-available (opcode probe). `auto`/`off` always
16 * start and silently fall back; `on` is a hard requirement that fails

57/*
58 * brix_uring_runtime_available — authoritative, memoized opcode probe.
59 *
60 * Stands up a throwaway 8-entry ring, asks io_uring_get_probe whether every
61 * required server data opcode (READ/WRITE/READV/WRITEV/FSYNC) is supported,
62 * tears it down, and caches the verdict for the lifetime of the process. Never
63 * parses a kernel version: containers/seccomp routinely block io_uring_setup
64 * even on new kernels, so only an actual setup+probe is trustworthy.
65 *
66 * Returns 1 iff a ring can be created and all required opcodes are supported.
67 */
68ngx_int_t
69brix_uring_runtime_available(void)
70{
71 static const int required[] = {
72 IORING_OP_READ, IORING_OP_WRITE, IORING_OP_READV,
73 IORING_OP_WRITEV, IORING_OP_FSYNC
74 };
75 static ngx_int_t cached = -1; /* -1 = not yet probed */
76 struct io_uring ring;
77 struct io_uring_probe *probe;
78 ngx_int_t ok = 1;
79 ngx_uint_t i;
80
81 if (cached != -1) {
82 return cached;
83 }
84
85 if (io_uring_queue_init(8, &ring, 0) < 0) {
86 cached = 0; /* seccomp / no syscall / old kernel */
87 return 0;
88 }
89
90 probe = io_uring_get_probe_ring(&ring);
91 if (probe == NULL) {
92 io_uring_queue_exit(&ring);
93 cached = 0;
94 return 0;
95 }
96
97 for (i = 0; i < sizeof(required) / sizeof(required[0]); i++) {
98 if (!io_uring_opcode_supported(probe, required[i])) {
99 ok = 0;
100 break;
101 }
102 }
103
104 io_uring_free_probe(probe);
105 io_uring_queue_exit(&ring);
106
107 cached = ok;
108 return cached;
109}

111/*
112 * uring_scan_conf — fold every enabled server block's io_uring settings.
113 *
114 * WHAT: Scans the stream server blocks and merges their io_uring demand into
115 * *scan: whether any block wants the ring, whether any requires it
116 * (`on`), the max queue depth, the restriction preference, and the
117 * first configured panic-file path.
118 * WHY: The ring is a per-worker singleton, so per-block settings must be
119 * folded into one verdict before bring-up.
120 * HOW: Linear scan skipping disabled/off blocks; max for depth, AND for
121 * restrictions, first-wins for the panic file.
122 *
123 * Returns 1 if at least one enabled block wants io_uring, 0 otherwise.
124 */
125static ngx_int_t
126uring_scan_conf(ngx_cycle_t *cycle, brix_uring_scan_t *scan)
127{
128 ngx_stream_core_main_conf_t *cmcf;
129 ngx_stream_core_srv_conf_t **cscfp;
130 ngx_stream_brix_srv_conf_t *xcf;
131 ngx_uint_t i;
132 ngx_int_t want = 0;
133
134 cmcf = ngx_stream_cycle_get_module_main_conf(cycle, ngx_stream_core_module);
135 if (cmcf == NULL) {
136 return 0;
137 }
138
139 cscfp = cmcf->servers.elts;
140 for (i = 0; i < cmcf->servers.nelts; i++) {
141 xcf = ngx_stream_conf_get_module_srv_conf(cscfp[i],
142 ngx_stream_brix_module);
143 if (!xcf->common.enable || xcf->io_uring == BRIX_IO_URING_OFF) {
144 continue;
145 }
146 want = 1;
147 if (xcf->io_uring == BRIX_IO_URING_ON) {
148 scan->mode_on = 1;
149 }
150 if (xcf->io_uring_queue_depth > scan->depth) {
151 scan->depth = xcf->io_uring_queue_depth;
152 }
153 if (!xcf->io_uring_restrict) {
154 scan->want_restrict = 0;
155 }
156 if (scan->panic_file.len == 0 && xcf->io_uring_panic_file.len > 0) {
157 scan->panic_file = xcf->io_uring_panic_file;
158 }
159 }
160
161 return want;
162}

164/*
165 * uring_probe_features — decide whether this worker should bring a ring up.
166 *
167 * WHAT: Folds the configuration (uring_scan_conf) and the kernel capability
168 * probe into one go/no-go verdict, clamping the queue depth.
169 * WHY: Both "nobody wants io_uring" and "auto blocks on a host without it"
170 * are silent thread-pool outcomes that must be decided BEFORE any ring
171 * resource exists — keeping the post-creation fallback decision point
172 * (brix_uring_init_fail) single.
173 * HOW: Early-returns NGX_DECLINED for either skip reason (logging the
174 * `auto`-unavailable NOTICE), otherwise clamps depth into [8,4096] and
175 * returns NGX_OK to proceed with bring-up. Non-static: also called from
176 * brix_uring_init_worker() in uring.c.
177 */
178ngx_int_t
179uring_probe_features(ngx_cycle_t *cycle, brix_uring_scan_t *scan)
180{
181 scan->mode_on = 0;
182 scan->want_restrict = 1;
183 scan->depth = BRIX_IO_URING_QUEUE_DEPTH;
184 ngx_str_null(&scan->panic_file);
185
186 if (!uring_scan_conf(cycle, scan)) {
187 return NGX_DECLINED; /* every enabled block has io_uring off */
188 }
189
190 /* AUTO blocks need the per-process probe (seccomp-accurate); ON blocks
191 * already passed it at config time but re-checking is cheap + memoized. */
192 if (!scan->mode_on && !brix_uring_runtime_available()) {
193 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
194 "brix: io_uring unavailable on this host; using the thread pool");
195 return NGX_DECLINED;
196 }
197
198 if (scan->depth < 8) { scan->depth = 8; }
199 if (scan->depth > 4096) { scan->depth = 4096; }
200
201 return NGX_OK;
202}

214/*
215 * brix_uring_validate_conf — §32 startup fail-fast (ADR-16).
216 *
217 * If any enabled server block requests `brix_io_uring on`, io_uring MUST be
218 * provided or startup fails (NGX_ERROR -> nginx -t exits non-zero, master
219 * refuses to start). Two independent gates:
220 * (1) compile-time: a stub build can never satisfy `on` — caught with no
221 * kernel needed, so CI / `nginx -t` flags it anywhere;
222 * (2) runtime: a liburing build whose probe fails (old kernel / seccomp).
223 * `off` and `auto` always pass (auto degrades silently per worker).
224 *
225 * Returns NGX_OK to allow startup, NGX_ERROR to abort it.
226 */
227ngx_int_t
228brix_uring_validate_conf(ngx_conf_t *cf)
229{
230 if (!brix_uring_any_block_on(cf)) {
231 return NGX_OK;
232 }
233
234#if !(BRIX_HAVE_LIBURING)
235 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
236 "\"brix_io_uring on\" requires a build with liburing, but this binary "
237 "was compiled WITHOUT it. Rebuild with BRIX_ENABLE_IO_URING=1 and "
238 "liburing-devel installed, or set \"brix_io_uring auto\" to allow "
239 "silent fallback to the thread pool.");
240 return NGX_ERROR;
241#else
242 if (!brix_uring_runtime_available()) {
243 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
244 "\"brix_io_uring on\" requested but io_uring is unavailable on "
245 "this host (io_uring_setup/opcode probe failed). This is typically a "
246 "seccomp policy (Docker/containerd default profiles block io_uring) "
247 "or a kernel older than %d.%d. Set \"brix_io_uring auto\" to fall "
248 "back to the thread pool, or enable io_uring at the host/container "
249 "level.",
250 BRIX_IO_URING_MIN_KERNEL_MAJOR, BRIX_IO_URING_MIN_KERNEL_MINOR);
251 return NGX_ERROR;
252 }
253 return NGX_OK;
254#endif
255}
