48 if (strncmp(name, XFS_CKS_XATTR_PFX, sizeof(XFS_CKS_XATTR_PFX) - 1) == 0) {
49 const char *algo = name + sizeof(XFS_CKS_XATTR_PFX) - 1;
50 char hex[160];

64 return -ERANGE;

66 memcpy(value, hex, vlen);

68 }
