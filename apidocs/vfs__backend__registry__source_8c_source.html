51/*
52 * Per-tier-kind source builders.
53 *
54 * WHAT: one static builder per storage-backend kind. Each takes the resolved
55 * export entry + a log, constructs the RAW driver instance for that kind
56 * (no decorators, no memoization), logs the ready/failed line, and returns
57 * the instance or NULL.
58 * WHY: brix_vfs_backend_build_source was a 36-CCN if-ladder over the backend
59 * name; splitting each arm into a named builder selected by a descriptor
60 * table flattens it to a single table scan (§8 table-driven dispatch).
61 * HOW: the builders share the brix_vbr_source_fn signature so a static const
62 * descriptor array can map backend name → builder. The default POSIX kind
63 * (name "" or "posix") and the final pblock fallback are handled by the
64 * orchestrator, not the table.
65 *
66 * All log strings, config-error text, and construction order are byte-frozen
67 * from the pre-split ladder.
68 */
69typedef brix_sd_instance_t *(*brix_vbr_source_fn)(brix_vfs_backend_entry_t *e,
70 ngx_log_t *log);

211/* HTTP(S) source backend (read-only): the shared S3/libcurl transport does the
212 * HEAD/Range-GET; sd_http carries no libcurl itself. */
213static brix_sd_instance_t *
214brix_vbr_build_http(brix_vfs_backend_entry_t *e, ngx_log_t *log)
215{
216 brix_sd_http_cfg_t cfg;
217 brix_sd_http_ep_cfg_t extra[7];
218 int i;
219 brix_sd_instance_t *inst;
220
221 ngx_memzero(&cfg, sizeof(cfg));
222 cfg.host = e->origin_host;
223 cfg.port = e->origin_port;
224 cfg.tls = e->origin_tls;
225 cfg.base_path = e->origin_path;
226 cfg.transport = &brix_s3_origin_curl_transport;
227 cfg.timeout_ms = BRIX_SD_HTTP_DEFAULT_TIMEOUT_MS;
228 cfg.bearer_token = (e->origin_token[0] != '\0') ? e->origin_token : NULL;
229 /* §14/C-3: verify the https origin against the operator-configured CA
230 * (file or hashed dir); "" ⇒ system bundle (public-CA origin). */
231 cfg.ca_path = (e->origin_ca_dir[0] != '\0') ? e->origin_ca_dir : NULL;
232 cfg.put_checksum = e->origin_put_checksum; /* #12 origin-enforced PUT integrity */
233 cfg.failover_note = brix_vfs_http_failover_note; /* T16 */
234 cfg.health_note = brix_vfs_http_health_note;
235 /* phase-68 T11: the remaining pipe-separated failover origins */
236 for (i = 0; i < e->n_http_extra && i < 7; i++) {
237 extra[i].host = e->http_extra[i].host;
238 extra[i].port = e->http_extra[i].port;
239 extra[i].tls = e->http_extra[i].tls;
240 extra[i].base_path = e->http_extra[i].base;
241 }
242 cfg.extra = extra;
243 cfg.n_extra = (e->n_http_extra < 7) ? e->n_http_extra : 7;
244
245 inst = brix_sd_http_create(&cfg, log);
246 if (inst == NULL) {
247 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
248 "brix: http backend init failed for export \"%s\"",
249 e->root_canon);
250 } else {
251 if (e->has_http_ranks) {
252 sd_http_set_ranks(inst, e->http_ranks, 8); /* T19 geo/static */
253 }
254 ngx_log_error(NGX_LOG_NOTICE, log, 0,
255 "brix: http storage backend ready at \"%s\"", e->root_canon);
256 }
257 return inst;
258}

260/* Read-only S3 source backend (phase-64): the export's bytes live in a remote
261 * bucket, served over the shared libcurl S3 transport (signed Range GET). The
262 * remote driver is CAP_RANGE_READ only, so this primary is read-only — exactly
263 * like an http:// primary. origin_path carries the bucket. */

361/* Build the entry's RAW source driver instance (no decorators, no memoization). The
362 * instance is malloc/pool-owned, worker-safe; SQLite connections are per-worker so
363 * this is only ever called after fork. Returns NULL (logged) on init failure.
364 *
365 * The default POSIX source (backend "" or "posix") is dispatched first; any other
366 * named backend is looked up in brix_vbr_source_table; an unrecognised name falls
367 * through to the pblock fallback (or NULL when SQLite is unavailable). */
368brix_sd_instance_t *
369brix_vfs_backend_build_source(brix_vfs_backend_entry_t *e, ngx_log_t *log)
370{
371 size_t k;
372
373 if (e->backend[0] == '\0' || ngx_strcmp(e->backend, "posix") == 0) {
374 return brix_vbr_build_posix(e, log);
375 }
376
377 for (k = 0; k < sizeof(brix_vbr_source_table)
378 / sizeof(brix_vbr_source_table[0]); k++)
379 {
380 if (ngx_strcmp(e->backend, brix_vbr_source_table[k].name) == 0) {
381 return brix_vbr_source_table[k].build(e, log);
382 }
383 }
384
385#if BRIX_HAVE_SQLITE
386 return brix_vbr_build_pblock(e, log);
387#else
388 (void) log;
389 return NULL;
390#endif
391}
