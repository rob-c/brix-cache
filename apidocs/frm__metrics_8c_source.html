1/*
2 * frm/metrics.c — Prometheus exporter for the FRM tape-stage subsystem.
3 *
4 * WHAT: Emits the brix_frm_* metric families from the shared metrics block:
5 * stage requests/dedup/reject counters, success and fail-by-reason counters,
6 * the in-flight gauge, the evict/migrate/purge/cmsd-have/async counters, and a
7 * coarse seconds-scale stage-latency histogram.
8 *
9 * WHY: Tape staging is the one FRM operation an operator must watch (recall rate,
10 * failure rate, queue saturation, recall latency). These counters are the
11 * observability surface promised by Phase 1's Definition of Done. Like every
12 * other exporter the labels are strictly low-cardinality (a fixed fail reason);
13 * never a path/DN/reqid (INVARIANT #8).
14 *
15 * HOW: Counters are written lock-free by the stream-side FRM engine (queue.c /
16 * stage.c) via the BRIX_FRM_METRIC_* macros into shm->frm; this file only
17 * READS them (ngx_atomic_fetch_add(..., 0)) at scrape time, so the snapshot is
18 * eventually consistent. The histogram is stored non-cumulative in SHM and
19 * cumulated into Prometheus `le` buckets here, exactly like unified.c. This unit
20 * is compiled into the HTTP metrics module (it runs at /metrics request time),
21 * so it must NOT call stream-module symbols (frm_index_count etc.) — every
22 * value comes from the shared SHM block alone.
23 */
24

42void

44{
45 unsigned long long value;
46 ngx_uint_t b;
47
48 mw_emit_scalar(mw, "brix_frm_requests_total",
49 "Tape stage requests admitted to the FRM durable queue.",
50 &shm->frm.requests_total);
51 mw_emit_scalar(mw, "brix_frm_dedup_hits_total",
52 "Stage opens collapsed onto an already in-flight recall.",
53 &shm->frm.dedup_hits_total);
54 mw_emit_scalar(mw, "brix_frm_reject_inflight_total",
55 "Stage requests refused because the queue was at max_inflight.",
56 &shm->frm.reject_inflight_total);
57 mw_emit_scalar(mw, "brix_frm_stage_success_total",
58 "Recalls that completed and brought the file online.",
59 &shm->frm.stage_success_total);
60
61 mw_emit_labeled(mw, "brix_frm_stage_fail_total",
62 "Recalls that failed, by coarse reason.",
63 "reason", brix_frm_fail_names, BRIX_FRM_NFAIL,
64 shm->frm.stage_fail_total);
65
66 mw_emit_scalar(mw, "brix_frm_evict_total",
67 "kXR_evict / Tape-REST release marks applied.",
68 &shm->frm.evict_total);
69 mw_emit_scalar(mw, "brix_frm_waitresp_total",
70 "Async stalled opens parked with kXR_waitresp.",
71 &shm->frm.waitresp_total);
72 mw_emit_scalar(mw, "brix_frm_asynresp_total",
73 "Async stage completions delivered via kXR_attn(asynresp).",
74 &shm->frm.asynresp_total);
75 mw_emit_scalar(mw, "brix_frm_cmsd_have_total",
76 "Now-resident paths registered with the manager (cmsd Have).",
77 &shm->frm.cmsd_have_total);
78 mw_emit_scalar(mw, "brix_frm_migrate_total",
79 "Category-2 migrate-out attempts (scaffolding).",
80 &shm->frm.migrate_total);
81 mw_emit_scalar(mw, "brix_frm_purge_total",
82 "Category-2 purge decisions logged (scaffolding).",
83 &shm->frm.purge_total);
84
85 /* in_flight is a GAUGE — mw_emit_scalar declares TYPE counter, so emit the
86 * gauge banner + value by hand (the stream.c pattern). */
87 mw_printf(mw,
88 "# HELP brix_frm_in_flight Stage requests currently QUEUED or STAGING.\n"
89 "# TYPE brix_frm_in_flight gauge\n"
90 "brix_frm_in_flight %lu\n",
91 (unsigned long) ngx_atomic_fetch_add(&shm->frm.in_flight, 0));
92
93 /*
94 * Stage-latency histogram (seconds). The write side increments only the one
95 * bucket each recall lands in; Prometheus `le` buckets are cumulative, so the
96 * running sum is reported per bucket and the +Inf bucket equals the count.
97 */
98 mw_printf(mw,
99 "# HELP brix_frm_stage_latency_seconds Tape recall latency in seconds.\n"
100 "# TYPE brix_frm_stage_latency_seconds histogram\n");
101 value = 0;
102 for (b = 0; b < BRIX_FRM_LATENCY_BUCKETS - 1; b++) {
103 value += (unsigned long long)
104 ngx_atomic_fetch_add(&shm->frm.stage_latency_bucket[b], 0);
105 mw_printf(mw,
106 "brix_frm_stage_latency_seconds_bucket{le=\"%lu\"} %llu\n",
107 brix_frm_latency_bounds[b], value);
108 }
109 value += (unsigned long long) ngx_atomic_fetch_add(
110 &shm->frm.stage_latency_bucket[BRIX_FRM_LATENCY_BUCKETS - 1], 0);
111 mw_printf(mw,
112 "brix_frm_stage_latency_seconds_bucket{le=\"+Inf\"} %llu\n", value);
113 mw_printf(mw,
114 "brix_frm_stage_latency_seconds_sum %llu\n"
115 "brix_frm_stage_latency_seconds_count %llu\n",
116 (unsigned long long) ngx_atomic_fetch_add(
117 &shm->frm.stage_latency_sum_sec, 0),
118 (unsigned long long) ngx_atomic_fetch_add(
119 &shm->frm.stage_latency_count, 0));
120}
