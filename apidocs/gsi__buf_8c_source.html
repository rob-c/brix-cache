12 const uint8_t *end = buf + len;
13 size_t name_len;
14
15 if (len < 8) {
16 return -1;
17 }
18 name_len = strnlen((const char *) cur, len) + 1; /* protocol name + NUL */
19 if (name_len >= len) {
20 return -1;
21 }
22 cur += name_len;
23 if (cur + 4 > end) { /* step */
24 return -1;
25 }
26 cur += 4;
27
28 while (cur + 8 <= end) {
29 uint32_t btype, blen;
30 memcpy(&btype, cur, 4);
31 memcpy(&blen, cur + 4, 4);
32 btype = ntohl(btype);
33 blen = ntohl(blen);
34 cur += 8;
35 if (btype == (uint32_t) kXRS_none) {
36 break;
37 }
38 if ((size_t) (end - cur) < blen) {
39 return -1;
40 }
41 if (btype == type) {
42 *out = cur;
43 *outlen = blen;
44 return 0;
45 }
46 cur += blen;
47 }
48 return -1;
49}

71void
72brix_gbuf_raw(brix_gbuf *g, const void *data, size_t n)
73{
74 if (g->err) {
75 return;
76 }
77 if (g->len + n > g->cap) {
78 size_t ncap = g->cap ? g->cap : 256;
79 uint8_t *np;
80 while (g->len + n > ncap) {
81 ncap *= 2;
82 }
83 np = (uint8_t *) realloc(g->p, ncap);
84 if (np == NULL) {
85 g->err = 1;
86 return;
87 }
88 g->p = np;
89 g->cap = ncap;
90 }
91 memcpy(g->p + g->len, data, n);
92 g->len += n;
93}
