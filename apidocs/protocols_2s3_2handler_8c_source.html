1/*
2 * handler.c — S3 content handler: URI parsing, SigV4 auth gate, and method dispatch.
3 *
4 * WHAT: The nginx HTTP content handler for all S3 REST API operations. Every S3 request
5 * enters here — it parses the path-style URI (/<bucket>/<key>), verifies the AWS
6 * SigV4 signature, resolves the key to a filesystem path, and dispatches to the
7 * appropriate method handler (GetObject, PutObject, DeleteObject, ListObjectsV2, etc.).
8 *
9 * WHY: S3 clients use path-style URIs that must be parsed into bucket + key before any
10 * filesystem operations. The SigV4 auth gate rejects unsigned requests early without
11 * parsing overhead. Write operations check cf->common.allow_write before body read to reject
12 * writes on read-only endpoints without consuming client bandwidth.
13 *
14 * HOW:
15 * 1. Check cf->common.enable → NGX_DECLINED if disabled
16 * 2. Determine method_slot (list vs object) + track bytes_rx metric per IP version
17 * 3. Verify SigV4 signature — fail fast with XML error on invalid
18 * 4. Parse URI into bucket+key via s3_parse_uri()
19 * 5. Dispatch list requests (ListObjectsV2, ListMultipartUploads, ListParts)
20 * 6. Check special empty-key flags (uploads → InitiateMPU, delete → DeleteObjects)
21 * 7. Reject bare GET /<bucket>/? (empty key without flag) as InvalidURI
22 * 8. Resolve key to fs_path via s3_resolve_key() — AccessDenied on escape
23 * 9. Dispatch by HTTP method: GET/HEAD/PUT/DELETE/POST → specific handler
24 * 10. Unknown methods → 405 Method Not Allowed
25 *
26 * Pool allocation: path_copy uses ngx_pnalloc(r->pool, PATH_MAX) to survive until the
27 * async PUT/MPU-complete callback fires after body reading completes.
28 *
29 * Phase-79 file-size split: the URI parser and the post-auth dispatch tree moved to
30 * the sibling handler_dispatch.c (URI parse, token-scope gate, bucket/empty-key/list
31 * routing) and handler_object_route.c (object-key method routing). This file retains
32 * the entry handler, the SigV4/XrdAcc auth gate, request classification, OPTIONS/CORS,
33 * and the delegation/pmark hooks. Cross-file symbols are declared in
34 * s3_handler_internal.h.
35 */

56brix_acc_op_t
57s3_method_aop(ngx_http_request_t *r)
58{
59 switch (r->method) {
60 case NGX_HTTP_GET: return BRIX_AOP_READ; /* GetObject / ListObjects */
61 case NGX_HTTP_HEAD: return BRIX_AOP_STAT;
62 case NGX_HTTP_PUT: return BRIX_AOP_CREATE;
63 case NGX_HTTP_POST: return BRIX_AOP_CREATE; /* multipart upload */
64 case NGX_HTTP_DELETE: return BRIX_AOP_DELETE;
65 default: return BRIX_AOP_STAT;
66 }
67}

69/*
70 * s3_acc_check — XrdAcc tier for S3 (when `brix_authdb_format xrdacc`).
71 * Returns NGX_OK (allow / not selected) or NGX_HTTP_FORBIDDEN (deny).
72 */
73static ngx_int_t
74s3_acc_check(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
75 brix_identity_t *id)
76{
77 const char *name = "", *vorg = "", *role = "", *grp = "";
78 char host[64], path[1024];
79 size_t n;
80 ngx_int_t rc;
81
82 if (cf->acc.format != BRIX_AUTHDB_FORMAT_XRDACC) {
83 return NGX_OK;
84 }
85 if (id != NULL) {
86 name = brix_identity_dn_cstr(id); /* S3 access key (or subject) */
87 vorg = brix_identity_acc_vorg_cstr(id);
88 role = brix_identity_acc_role_cstr(id);
89 grp = brix_identity_acc_group_cstr(id);
90 }
91 n = ngx_min(r->connection->addr_text.len, sizeof(host) - 1);
92 ngx_memcpy(host, r->connection->addr_text.data, n);
93 host[n] = '\0';
94
95 /* Opt-in reverse DNS for `h <host>`/`h .domain` rules (per request). */
96 if (cf->acc.resolve_hosts) {
97 char hbuf[256];
98 const char *h = brix_acc_resolve_peer(r->connection->sockaddr,
99 r->connection->socklen,
100 hbuf, sizeof(hbuf));
101 if (h != NULL) {
102 n = ngx_min(ngx_strlen(h), sizeof(host) - 1);
103 ngx_memcpy(host, h, n);
104 host[n] = '\0';
105 }
106 }
107
108 n = ngx_min(r->uri.len, sizeof(path) - 1);
109 ngx_memcpy(path, r->uri.data, n);
110 path[n] = '\0';
111
112 rc = brix_acc_http_authorize(r->pool, r->connection->log,
113 &cf->acc, name, host, vorg, role, grp,
114 s3_method_aop(r), path);
115 return (rc == NGX_ERROR) ? NGX_HTTP_FORBIDDEN : NGX_OK;
116}

118/*
119 *
120 * WHAT: Detects whether the HTTP request is a ListObjectsV2 operation by searching for the
121 * "list-type=2" query parameter in the request args. Returns 1 if found, 0 otherwise.
122 * Only checks GET requests with non-empty query arguments — other methods cannot be lists.
123 *
124 * WHY: ListObjectsV2 has fundamentally different filesystem behavior from GetObject (it returns XML
125 * listing of objects rather than serving a single file). This detection allows the main handler
126 * to dispatch to separate metrics slots and distinct handler functions for list vs object operations. */
127/*
128 * HOW: Checks that r->method is GET and reads the exact list-type query
129 * parameter with the shared query parser. Returns 1 only when list-type=2.
130 */
131static int
132s3_is_list_request(ngx_http_request_t *r)
133{
134 char list_type[8];
135
136 if (r->method != NGX_HTTP_GET || r->args.len == 0) {
137 return 0;
138 }
139
140 return brix_http_query_get(r->args, "list-type", list_type,
141 sizeof(list_type), 0) > 0
142 && ngx_strcmp(list_type, "2") == 0;
143}

145static ngx_flag_t
146s3_is_post_object_form(ngx_http_request_t *r)
147{
148 ngx_table_elt_t *ct;
149
150 if (r->method != NGX_HTTP_POST || r->args.len != 0) {
151 return 0;
152 }
153
154 ct = brix_http_find_header(r, "Content-Type",
155 sizeof("Content-Type") - 1);
156 if (ct == NULL || ct->value.len < sizeof("multipart/form-data") - 1) {
157 return 0;
158 }
159
160 return ngx_strncasecmp(ct->value.data, (u_char *) "multipart/form-data",
161 sizeof("multipart/form-data") - 1) == 0;
162}

177static ngx_int_t
178s3_add_preflight_headers(ngx_http_request_t *r, const ngx_str_t *allow)
179{
180 ngx_table_elt_t *origin;
181 ngx_table_elt_t *req_headers;
182
183 origin = brix_http_find_header(r, "Origin", sizeof("Origin") - 1);
184 if (origin == NULL) {
185 return NGX_OK;
186 }
187
188 if (brix_http_set_header(r, "Access-Control-Allow-Origin", "*", NULL)
189 != NGX_OK
190 || brix_http_set_header(r, "Vary", "Origin", NULL) != NGX_OK
191 || brix_http_set_header_str(r, "Access-Control-Allow-Methods",
192 allow, 0, NULL) != NGX_OK)
193 {
194 return NGX_ERROR;
195 }
196
197 req_headers = brix_http_find_header(
198 r, "Access-Control-Request-Headers",
199 sizeof("Access-Control-Request-Headers") - 1);
200 if (req_headers != NULL
201 && !brix_http_str_has_ctl(req_headers->value.data,
202 req_headers->value.len))
203 {
204 if (brix_http_set_header_str(r, "Access-Control-Allow-Headers",
205 &req_headers->value, 0, NULL) != NGX_OK)
206 {
207 return NGX_ERROR;
208 }
209 } else if (brix_http_set_header(r, "Access-Control-Allow-Headers",
210 "Authorization, Content-Type, "
211 "Content-Length, Content-MD5, "
212 "x-amz-content-sha256, x-amz-date, "
213 "x-amz-security-token, x-amz-copy-source, "
214 "x-amz-checksum-crc64nvme, "
215 "x-amz-checksum-algorithm, "
216 "x-amz-sdk-checksum-algorithm, "
217 "Range",
218 NULL) != NGX_OK)
219 {
220 return NGX_ERROR;
221 }
222
223 return brix_http_set_header(r, "Access-Control-Max-Age", "86400", NULL);
224}

226static ngx_int_t
227s3_handle_options(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf)
228{
229 ngx_table_elt_t *h;
230 ngx_str_t allow;
231
232 if (brix_http_operation_allow_header(r->pool,
233 brix_s3_operations, brix_s3_operations_count,
234 s3_allow_flags(cf), &allow) != NGX_OK)
235 {
236 return NGX_HTTP_INTERNAL_SERVER_ERROR;
237 }
238
239 h = ngx_list_push(&r->headers_out.headers);
240 if (h == NULL) {
241 return NGX_HTTP_INTERNAL_SERVER_ERROR;
242 }
243 h->hash = 1;
244 ngx_str_set(&h->key, "Allow");
245 h->value = allow;
246
247 if (s3_add_preflight_headers(r, &allow) != NGX_OK) {
248 return NGX_HTTP_INTERNAL_SERVER_ERROR;
249 }
250
251 r->headers_out.status = NGX_HTTP_OK;
252 r->headers_out.content_length_n = 0;
253
254 ngx_http_send_header(r);
255 return ngx_http_send_special(r, NGX_HTTP_LAST);
256}

331/*
332 * ngx_http_s3_handler - nginx HTTP content handler for the S3 API.
333 *
334 * Implements the AWS S3 path-style API subset needed by XrdClS3 and other
335 * HEP data transfer tools. Dispatch order:
336 *
337 * 1. Parse the URI into bucket + key using s3_parse_uri().
338 * 2. Verify AWS SigV4 signature (s3_verify_sigv4).
339 * 3. ListObjectsV2 (GET /<bucket>/?list-type=2) -> s3_handle_list().
340 * 4. ListMultipartUploads (GET /<bucket>/?uploads) -> s3_handle_list_multipart_uploads().
341 * 5. ListParts (GET /<bucket>/<key>?uploadId=<id>) -> s3_handle_list_parts().
342 * 6. GetObject (GET /<bucket>/<key>) -> s3_handle_get().
343 * 7. HeadObject (HEAD /<bucket>/<key>) -> s3_handle_head().
344 * 8. PutObject / UploadPart (PUT /<bucket>/<key>) -> s3_put_body_handler().
345 * 9. DeleteObject / Abort (DELETE /<bucket>/<key>) -> s3_handle_delete().
346 * 10. CompleteMultipartUpload (POST /<bucket>/<key>?uploadId=<id>) -> s3_multipart_complete_body_handler().
347 * 11. InitiateMultipartUpload (POST /<bucket>/<key>?uploads) -> s3_handle_multipart_initiate().
348 *
349 * The fs_path (filesystem path within conf->common.root_canon) is resolved by
350 * s3_resolve_key() and stored in r->pool for the async PUT callback. The URI
351 * parse and post-auth method dispatch live in handler_dispatch.c and
352 * handler_object_route.c; this handler runs the auth gate and hands off to
353 * s3_dispatch_after_auth().
354 *
355 * Pool allocation: path_copy uses ngx_pnalloc(r->pool, PATH_MAX) so it
356 * survives until the PUT body callback fires after body reading completes.
357 */
358/*
359 * WHY: The dispatch order is critical — list requests are checked before object access to avoid treating empty-key GETs as invalid URIs. SigV4 verification happens early so unsigned requests fail fast without parsing overhead. Write operations (PUT/DELETE/POST) check cf->common.allow_write before body read to reject writes on read-only endpoints without consuming client bandwidth.
360 *
361 * HOW: Gets location config, returns NGX_DECLINED if disabled. Determines method_slot (list vs object). Tracks per-IP-version bytes_rx metric. Verifies SigV4 signature — fails with appropriate XML error if invalid. Parses URI into bucket+key via s3_parse_uri() — rejects mismatched bucket (-1) or malformed URI (0). Checks list-type=2, uploads flag, delete flag before empty-key rejection. Resolves key to fs_path via s3_resolve_key(). Dispatches by HTTP method: GET→list_parts/get; HEAD→head; PUT→upload_part_copy/copy_object/put_body; DELETE→multipart_abort/delete; POST→multipart_initiate/complete. Unknown methods → 405.
362 */
363ngx_int_t
364ngx_http_s3_handler(ngx_http_request_t *r)
365{
366 ngx_http_s3_loc_conf_t *cf;
367 ngx_int_t rc;
368 int is_list_request;
369 ngx_flag_t is_post_object_form;
370 ngx_uint_t method_slot;
371 ngx_http_s3_req_ctx_t *s3ctx;
372
373 cf = ngx_http_get_module_loc_conf(r, ngx_http_brix_s3_module);
374 if (!cf->common.enable) {
375 return NGX_DECLINED;
376 }
377
378 /* AGPL-3.0 sec.13: offer remote users the source (X-Source header). */
379 brix_http_source_offer(r);
380
381 BRIX_PCALLOC_OR_RETURN(s3ctx, r->pool, sizeof(*s3ctx), NGX_HTTP_INTERNAL_SERVER_ERROR);
382 s3ctx->identity = brix_identity_alloc(r->pool);
383 if (s3ctx->identity == NULL) {
384 return NGX_HTTP_INTERNAL_SERVER_ERROR;
385 }
386 ngx_http_set_ctx(r, s3ctx, ngx_http_brix_s3_module);
387
388 /*
389 * ListObjectsV2 is an HTTP GET on the wire, but it has very different
390 * filesystem behavior from GetObject, so it gets a separate metrics slot.
391 */
392 is_list_request = s3_is_list_request(r);
393 is_post_object_form = s3_is_post_object_form(r);
394 method_slot = is_list_request ? BRIX_S3_METHOD_LIST
395 : s3_metrics_method_slot(r);
396 s3_metrics_request_method(method_slot);
397
398
399 if (r->method == NGX_HTTP_OPTIONS) {
400 return s3_metrics_return_method(r, method_slot,
401 s3_handle_options(r, cf));
402 }
403
404 if (!is_post_object_form) {
405 rc = s3_verify_sigv4(r, cf, s3ctx->identity);
406 if (rc != NGX_OK) {
407 return s3_metrics_return_method(r, method_slot, rc);
408 }
409 }
410
411 /* XrdAcc engine (when brix_authdb_format xrdacc) */ rc = s3_acc_check(r, cf, s3ctx->identity);
412 if (rc != NGX_OK) {
413 return s3_metrics_return_method(r, method_slot, rc);
414 }
415
416 /*
417 * An auth gate that rejected the request by emitting its OWN XML error body
418 * (e.g. s3_verify_sigv4's "Missing/Malformed Authorization") returns NGX_OK
419 * but leaves r->header_sent set. Continuing to dispatch would let GetObject
420 * call ngx_http_send_header a second time ("header already sent" alert) and
421 * overwrite the sent 4xx with headers_out.status = 200 — hiding the auth
422 * failure from the LOG phase (metrics AND the bad-actor guard). Stop here so
423 * the already-sent status stands.
424 */
425 if (r->header_sent) {
426 return s3_metrics_return_method(r, method_slot, NGX_OK);
427 }
428
429 /* Phase-70 §5.1: capture an optional user-supplied x509 delegation proxy
430 * (X-Brix-Delegate-Proxy). No-op unless this export delegates to a backend;
431 * a rejected header (403) short-circuits dispatch. */
432 rc = s3_capture_delegate_proxy(r, s3ctx, cf);
433 if (rc != NGX_OK) {
434 return s3_metrics_return_method(r, method_slot, rc);
435 }
436
437 s3_sess_begin_request(r, method_slot);
438 s3_sess_attempt_request(r, method_slot);
439
440 /*
441 * SciTags packet marking (phase-34). S3 has no TPC, so only plain
442 * GET/PUT are marked, and only when brix_pmark_http_plain is on.
443 * Begun here post-SigV4; ended via a request-pool cleanup.
444 */
445 s3_pmark_begin_if_enabled(r, s3ctx, cf);
446
447 /*
448 * Phase 40: bracket the whole post-auth dispatch with the impersonation
449 * principal (now that auth has populated s3ctx->identity). This covers the
450 * SYNCHRONOUS ops (GET/HEAD/DELETE/list/multipart-initiate) — which open /
451 * stat / unlink as the mapped user, so e.g. a GetObject cannot read a file
452 * the mapped user has no UNIX permission for (a cross-tenant leak when run as
453 * the worker). The async body handlers (PUT/POST/multipart-complete/
454 * delete-objects) return NGX_DONE and re-establish the principal in their own
455 * callbacks, so clearing it here on return is correct. No-op unless map mode.
456 */
457 brix_imp_request_begin(s3ctx->identity);
458 rc = s3_dispatch_after_auth(r, cf, s3ctx, method_slot,
459 is_list_request, is_post_object_form);
460 brix_imp_request_end();
461 return rc;
462}
