1/*
2 * handler.c — S3 content handler: URI parsing, SigV4 auth gate, and method dispatch.
3 *
4 * WHAT: The nginx HTTP content handler for all S3 REST API operations. Every S3 request
5 * enters here — it parses the path-style URI (/<bucket>/<key>), verifies the AWS
6 * SigV4 signature, resolves the key to a filesystem path, and dispatches to the
7 * appropriate method handler (GetObject, PutObject, DeleteObject, ListObjectsV2, etc.).
8 *
9 * WHY: S3 clients use path-style URIs that must be parsed into bucket + key before any
10 * filesystem operations. The SigV4 auth gate rejects unsigned requests early without
11 * parsing overhead. Write operations check cf->common.allow_write before body read to reject
12 * writes on read-only endpoints without consuming client bandwidth.
13 *
14 * HOW:
15 * 1. Check cf->common.enable → NGX_DECLINED if disabled
16 * 2. Determine method_slot (list vs object) + track bytes_rx metric per IP version
17 * 3. Verify SigV4 signature — fail fast with XML error on invalid
18 * 4. Parse URI into bucket+key via s3_parse_uri()
19 * 5. Dispatch list requests (ListObjectsV2, ListMultipartUploads, ListParts)
20 * 6. Check special empty-key flags (uploads → InitiateMPU, delete → DeleteObjects)
21 * 7. Reject bare GET /<bucket>/? (empty key without flag) as InvalidURI
22 * 8. Resolve key to fs_path via s3_resolve_key() — AccessDenied on escape
23 * 9. Dispatch by HTTP method: GET/HEAD/PUT/DELETE/POST → specific handler
24 * 10. Unknown methods → 405 Method Not Allowed
25 *
26 * Pool allocation: path_copy uses ngx_pnalloc(r->pool, PATH_MAX) to survive until the
27 * async PUT/MPU-complete callback fires after body reading completes.
28 */

45/* Map an S3 request method to the XrdAcc operation it requires. */
46static brix_acc_op_t
47s3_method_aop(ngx_http_request_t *r)
48{
49 switch (r->method) {
50 case NGX_HTTP_GET: return BRIX_AOP_READ; /* GetObject / ListObjects */
51 case NGX_HTTP_HEAD: return BRIX_AOP_STAT;
52 case NGX_HTTP_PUT: return BRIX_AOP_CREATE;
53 case NGX_HTTP_POST: return BRIX_AOP_CREATE; /* multipart upload */
54 case NGX_HTTP_DELETE: return BRIX_AOP_DELETE;
55 default: return BRIX_AOP_STAT;
56 }
57}

59/*
60 * s3_acc_check — XrdAcc tier for S3 (when `brix_authdb_format xrdacc`).
61 * Returns NGX_OK (allow / not selected) or NGX_HTTP_FORBIDDEN (deny).
62 */
63static ngx_int_t
64s3_acc_check(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
65 brix_identity_t *id)
66{
67 const char *name = "", *vorg = "", *role = "", *grp = "";
68 char host[64], path[1024];
69 size_t n;
70 ngx_int_t rc;
71
72 if (cf->acc.format != BRIX_AUTHDB_FORMAT_XRDACC) {
73 return NGX_OK;
74 }
75 if (id != NULL) {
76 name = brix_identity_dn_cstr(id); /* S3 access key (or subject) */
77 vorg = brix_identity_acc_vorg_cstr(id);
78 role = brix_identity_acc_role_cstr(id);
79 grp = brix_identity_acc_group_cstr(id);
80 }
81 n = ngx_min(r->connection->addr_text.len, sizeof(host) - 1);
82 ngx_memcpy(host, r->connection->addr_text.data, n);
83 host[n] = '\0';
84
85 /* Opt-in reverse DNS for `h <host>`/`h .domain` rules (per request). */
86 if (cf->acc.resolve_hosts) {
87 char hbuf[256];
88 const char *h = brix_acc_resolve_peer(r->connection->sockaddr,
89 r->connection->socklen,
90 hbuf, sizeof(hbuf));
91 if (h != NULL) {
92 n = ngx_min(ngx_strlen(h), sizeof(host) - 1);
93 ngx_memcpy(host, h, n);
94 host[n] = '\0';
95 }
96 }
97
98 n = ngx_min(r->uri.len, sizeof(path) - 1);
99 ngx_memcpy(path, r->uri.data, n);
100 path[n] = '\0';
101
102 rc = brix_acc_http_authorize(r->pool, r->connection->log,
103 &cf->acc, name, host, vorg, role, grp,
104 s3_method_aop(r), path);
105 return (rc == NGX_ERROR) ? NGX_HTTP_FORBIDDEN : NGX_OK;
106}

108/*
109 *
110 * WHAT: Parses the S3 request URI into bucket name and object key components. Supports
111 * path-style addressing (/<bucket>/<key>) and list requests (/<bucket>/?list-type=2).
112 * Performs URL decoding on the object key portion to handle encoded characters in paths.
113 * Returns 1 for success, 0 for malformed URI, -1 if bucket name doesn't match configured value.
114 *
115 * WHY: S3 clients use path-style URIs that must be parsed into bucket + key before any filesystem
116 * operations can proceed. The parser handles both object access (GET/PUT) and list requests
117 * uniformly — distinguishing between them happens later via s3_is_list_request(). */
118/*
119 * HOW: Advances past the leading '/' in the URI, then checks if the remaining prefix matches conf->bucket.len bytes. If bucket match succeeds, advances past '/' to extract the key portion which is URL-decoded via s3_urldecode(). Returns -1 on bucket mismatch (NoSuchBucket), 0 on malformed URI (InvalidURI), 1 on success.
120 */
121/*
122 * Parse the request URI into bucket + key.
123 *
124 * Path-style: /<bucket>/<key>
125 * List: /<bucket>/?list-type=2[&prefix=...&delimiter=...&...]
126 *
127 * Returns:
128 * 1 - success
129 * 0 - malformed URI
130 * -1 - bucket name mismatch
131 * */
132
133static int
134s3_parse_uri(ngx_http_request_t *r,
135 ngx_http_s3_loc_conf_t *cf,
136 u_char *key_out, size_t key_sz)
137{
138 const u_char *uri = r->uri.data;
139 size_t ulen = r->uri.len;
140 size_t blen = cf->bucket.len;
141
142 if (ulen == 0 || uri[0] != '/') {
143 return 0;
144 }
145
146 uri++;
147 ulen--;
148
149 if (blen > 0) {
150 if (ulen < blen || ngx_strncmp(uri, cf->bucket.data, blen) != 0) {
151 return -1;
152 }
153 uri += blen;
154 ulen -= blen;
155
156 if (ulen == 0) {
157 /* /bucket with no trailing slash — valid, empty key (e.g. ListObjects) */
158 key_out[0] = '\0';
159 return 1;
160 }
161 if (uri[0] != '/') {
162 return 0;
163 }
164 uri++;
165 ulen--;
166 }
167
168 if (ulen == 0) {
169 key_out[0] = '\0';
170 return 1;
171 }
172
173 if (brix_http_urldecode(uri, ulen, (char *) key_out, key_sz,
174 BRIX_URLDECODE_PLUS_TO_SPACE |
175 BRIX_URLDECODE_REJECT_NUL) != BRIX_URLDECODE_OK)
176 {
177 return 0;
178 }
179
180 return 1;
181}

183/*
184 *
185 * WHAT: Detects whether the HTTP request is a ListObjectsV2 operation by searching for the
186 * "list-type=2" query parameter in the request args. Returns 1 if found, 0 otherwise.
187 * Only checks GET requests with non-empty query arguments — other methods cannot be lists.
188 *
189 * WHY: ListObjectsV2 has fundamentally different filesystem behavior from GetObject (it returns XML
190 * listing of objects rather than serving a single file). This detection allows the main handler
191 * to dispatch to separate metrics slots and distinct handler functions for list vs object operations. */
192/*
193 * HOW: Checks that r->method is GET and reads the exact list-type query
194 * parameter with the shared query parser. Returns 1 only when list-type=2.
195 */
196static int
197s3_is_list_request(ngx_http_request_t *r)
198{
199 char list_type[8];
200
201 if (r->method != NGX_HTTP_GET || r->args.len == 0) {
202 return 0;
203 }
204
205 return brix_http_query_get(r->args, "list-type", list_type,
206 sizeof(list_type), 0) > 0
207 && ngx_strcmp(list_type, "2") == 0;
208}

210static ngx_flag_t
211s3_is_post_object_form(ngx_http_request_t *r)
212{
213 ngx_table_elt_t *ct;
214
215 if (r->method != NGX_HTTP_POST || r->args.len != 0) {
216 return 0;
217 }
218
219 ct = brix_http_find_header(r, "Content-Type",
220 sizeof("Content-Type") - 1);
221 if (ct == NULL || ct->value.len < sizeof("multipart/form-data") - 1) {
222 return 0;
223 }
224
225 return ngx_strncasecmp(ct->value.data, (u_char *) "multipart/form-data",
226 sizeof("multipart/form-data") - 1) == 0;
227}

242static ngx_int_t
243s3_add_preflight_headers(ngx_http_request_t *r, const ngx_str_t *allow)
244{
245 ngx_table_elt_t *origin;
246 ngx_table_elt_t *req_headers;
247
248 origin = brix_http_find_header(r, "Origin", sizeof("Origin") - 1);
249 if (origin == NULL) {
250 return NGX_OK;
251 }
252
253 if (brix_http_set_header(r, "Access-Control-Allow-Origin", "*", NULL)
254 != NGX_OK
255 || brix_http_set_header(r, "Vary", "Origin", NULL) != NGX_OK
256 || brix_http_set_header_str(r, "Access-Control-Allow-Methods",
257 allow, 0, NULL) != NGX_OK)
258 {
259 return NGX_ERROR;
260 }
261
262 req_headers = brix_http_find_header(
263 r, "Access-Control-Request-Headers",
264 sizeof("Access-Control-Request-Headers") - 1);
265 if (req_headers != NULL
266 && !brix_http_str_has_ctl(req_headers->value.data,
267 req_headers->value.len))
268 {
269 if (brix_http_set_header_str(r, "Access-Control-Allow-Headers",
270 &req_headers->value, 0, NULL) != NGX_OK)
271 {
272 return NGX_ERROR;
273 }
274 } else if (brix_http_set_header(r, "Access-Control-Allow-Headers",
275 "Authorization, Content-Type, "
276 "Content-Length, Content-MD5, "
277 "x-amz-content-sha256, x-amz-date, "
278 "x-amz-security-token, x-amz-copy-source, "
279 "x-amz-checksum-crc64nvme, "
280 "x-amz-checksum-algorithm, "
281 "x-amz-sdk-checksum-algorithm, "
282 "Range",
283 NULL) != NGX_OK)
284 {
285 return NGX_ERROR;
286 }
287
288 return brix_http_set_header(r, "Access-Control-Max-Age", "86400", NULL);
289}

291static ngx_int_t
292s3_handle_options(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf)
293{
294 ngx_table_elt_t *h;
295 ngx_str_t allow;
296
297 if (brix_http_operation_allow_header(r->pool,
298 brix_s3_operations, brix_s3_operations_count,
299 s3_allow_flags(cf), &allow) != NGX_OK)
300 {
301 return NGX_HTTP_INTERNAL_SERVER_ERROR;
302 }
303
304 h = ngx_list_push(&r->headers_out.headers);
305 if (h == NULL) {
306 return NGX_HTTP_INTERNAL_SERVER_ERROR;
307 }
308 h->hash = 1;
309 ngx_str_set(&h->key, "Allow");
310 h->value = allow;
311
312 if (s3_add_preflight_headers(r, &allow) != NGX_OK) {
313 return NGX_HTTP_INTERNAL_SERVER_ERROR;
314 }
315
316 r->headers_out.status = NGX_HTTP_OK;
317 r->headers_out.content_length_n = 0;
318
319 ngx_http_send_header(r);
320 return ngx_http_send_special(r, NGX_HTTP_LAST);
321}

327/*
328 * ngx_http_s3_handler - nginx HTTP content handler for the S3 API.
329 *
330 * Implements the AWS S3 path-style API subset needed by XrdClS3 and other
331 * HEP data transfer tools. Dispatch order:
332 *
333 * 1. Parse the URI into bucket + key using s3_parse_uri().
334 * 2. Verify AWS SigV4 signature (s3_verify_sigv4).
335 * 3. ListObjectsV2 (GET /<bucket>/?list-type=2) -> s3_handle_list().
336 * 4. ListMultipartUploads (GET /<bucket>/?uploads) -> s3_handle_list_multipart_uploads().
337 * 5. ListParts (GET /<bucket>/<key>?uploadId=<id>) -> s3_handle_list_parts().
338 * 6. GetObject (GET /<bucket>/<key>) -> s3_handle_get().
339 * 7. HeadObject (HEAD /<bucket>/<key>) -> s3_handle_head().
340 * 8. PutObject / UploadPart (PUT /<bucket>/<key>) -> s3_put_body_handler().
341 * 9. DeleteObject / Abort (DELETE /<bucket>/<key>) -> s3_handle_delete().
342 * 10. CompleteMultipartUpload (POST /<bucket>/<key>?uploadId=<id>) -> s3_multipart_complete_body_handler().
343 * 11. InitiateMultipartUpload (POST /<bucket>/<key>?uploads) -> s3_handle_multipart_initiate().
344 *
345 * The fs_path (filesystem path within conf->common.root_canon) is resolved by
346 * s3_resolve_key() and stored in r->pool for the async PUT callback.
347 *
348 * Pool allocation: path_copy uses ngx_pnalloc(r->pool, PATH_MAX) so it
349 * survives until the PUT body callback fires after body reading completes.
350 */
351/*
352 * WHY: The dispatch order is critical — list requests are checked before object access to avoid treating empty-key GETs as invalid URIs. SigV4 verification happens early so unsigned requests fail fast without parsing overhead. Write operations (PUT/DELETE/POST) check cf->common.allow_write before body read to reject writes on read-only endpoints without consuming client bandwidth.
353 *
354 * HOW: Gets location config, returns NGX_DECLINED if disabled. Determines method_slot (list vs object). Tracks per-IP-version bytes_rx metric. Verifies SigV4 signature — fails with appropriate XML error if invalid. Parses URI into bucket+key via s3_parse_uri() — rejects mismatched bucket (-1) or malformed URI (0). Checks list-type=2, uploads flag, delete flag before empty-key rejection. Resolves key to fs_path via s3_resolve_key(). Dispatches by HTTP method: GET→list_parts/get; HEAD→head; PUT→upload_part_copy/copy_object/put_body; DELETE→multipart_abort/delete; POST→multipart_initiate/complete. Unknown methods → 405.
355 */
356static ngx_int_t s3_dispatch_after_auth(ngx_http_request_t *r,
357 ngx_http_s3_loc_conf_t *cf, ngx_http_s3_req_ctx_t *s3ctx,
358 ngx_uint_t method_slot, int is_list_request,
359 ngx_flag_t is_post_object_form);
360
361ngx_int_t
362ngx_http_s3_handler(ngx_http_request_t *r)
363{
364 ngx_http_s3_loc_conf_t *cf;
365 ngx_int_t rc;
366 int is_list_request;
367 ngx_flag_t is_post_object_form;
368 ngx_uint_t method_slot;
369 ngx_http_s3_req_ctx_t *s3ctx;
370
371 cf = ngx_http_get_module_loc_conf(r, ngx_http_brix_s3_module);
372 if (!cf->common.enable) {
373 return NGX_DECLINED;
374 }
375
376 /* AGPL-3.0 sec.13: offer remote users the source (X-Source header). */
377 brix_http_source_offer(r);
378
379 BRIX_PCALLOC_OR_RETURN(s3ctx, r->pool, sizeof(*s3ctx), NGX_HTTP_INTERNAL_SERVER_ERROR);
380 s3ctx->identity = brix_identity_alloc(r->pool);
381 if (s3ctx->identity == NULL) {
382 return NGX_HTTP_INTERNAL_SERVER_ERROR;
383 }
384 ngx_http_set_ctx(r, s3ctx, ngx_http_brix_s3_module);
385
386 /*
387 * ListObjectsV2 is an HTTP GET on the wire, but it has very different
388 * filesystem behavior from GetObject, so it gets a separate metrics slot.
389 */
390 is_list_request = s3_is_list_request(r);
391 is_post_object_form = s3_is_post_object_form(r);
392 method_slot = is_list_request ? BRIX_S3_METHOD_LIST
393 : s3_metrics_method_slot(r);
394 s3_metrics_request_method(method_slot);
395
396
397 if (r->method == NGX_HTTP_OPTIONS) {
398 return s3_metrics_return_method(r, method_slot,
399 s3_handle_options(r, cf));
400 }
401
402 if (!is_post_object_form) {
403 rc = s3_verify_sigv4(r, cf, s3ctx->identity);
404 if (rc != NGX_OK) {
405 return s3_metrics_return_method(r, method_slot, rc);
406 }
407 }
408
409 /* XrdAcc engine (when brix_authdb_format xrdacc) */ rc = s3_acc_check(r, cf, s3ctx->identity);
410 if (rc != NGX_OK) {
411 return s3_metrics_return_method(r, method_slot, rc);
412 }
413
414 /*
415 * An auth gate that rejected the request by emitting its OWN XML error body
416 * (e.g. s3_verify_sigv4's "Missing/Malformed Authorization") returns NGX_OK
417 * but leaves r->header_sent set. Continuing to dispatch would let GetObject
418 * call ngx_http_send_header a second time ("header already sent" alert) and
419 * overwrite the sent 4xx with headers_out.status = 200 — hiding the auth
420 * failure from the LOG phase (metrics AND the bad-actor guard). Stop here so
421 * the already-sent status stands.
422 */
423 if (r->header_sent) {
424 return s3_metrics_return_method(r, method_slot, NGX_OK);
425 }
426
427 /*
428 * SciTags packet marking (phase-34). S3 has no TPC, so only plain
429 * GET/PUT are marked, and only when brix_pmark_http_plain is on.
430 * Begun here post-SigV4; ended via a request-pool cleanup.
431 */
432 if (cf->common.pmark.enable && cf->common.pmark.http_plain
433 && (r->method == NGX_HTTP_GET || r->method == NGX_HTTP_PUT))
434 {
435 u_char pth[2048], cgi[512];
436 ngx_cpystrn(pth, r->uri.data, ngx_min(r->uri.len + 1, sizeof(pth)));
437 if (r->args.len) {
438 ngx_cpystrn(cgi, r->args.data, ngx_min(r->args.len + 1, sizeof(cgi)));
439 } else {
440 cgi[0] = '\0';
441 }
442 brix_pmark_http_mark(&cf->common.pmark, r->pool, r->connection,
443 (r->method == NGX_HTTP_PUT),
444 brix_identity_vo_csv_cstr(s3ctx->identity),
445 brix_identity_dn_cstr(s3ctx->identity),
446 (const char *) pth, (const char *) cgi);
447 }
448
449 /*
450 * Phase 40: bracket the whole post-auth dispatch with the impersonation
451 * principal (now that auth has populated s3ctx->identity). This covers the
452 * SYNCHRONOUS ops (GET/HEAD/DELETE/list/multipart-initiate) — which open /
453 * stat / unlink as the mapped user, so e.g. a GetObject cannot read a file
454 * the mapped user has no UNIX permission for (a cross-tenant leak when run as
455 * the worker). The async body handlers (PUT/POST/multipart-complete/
456 * delete-objects) return NGX_DONE and re-establish the principal in their own
457 * callbacks, so clearing it here on return is correct. No-op unless map mode.
458 */
459 brix_imp_request_begin(s3ctx->identity);
460 rc = s3_dispatch_after_auth(r, cf, s3ctx, method_slot,
461 is_list_request, is_post_object_form);
462 brix_imp_request_end();
463 return rc;
464}

466/*
467 * Post-auth S3 op dispatch (parse URI -> route by method). Runs inside the
468 * caller's impersonation bracket; see ngx_http_s3_handler.
469 */
470static ngx_int_t
471s3_dispatch_after_auth(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
472 ngx_http_s3_req_ctx_t *s3ctx, ngx_uint_t method_slot,
473 int is_list_request, ngx_flag_t is_post_object_form)
474{
475 u_char key[S3_MAX_KEY];
476 char fs_path[PATH_MAX];
477 ngx_int_t rc;
478
479 {
480 int parse_rc = s3_parse_uri(r, cf, key, sizeof(key));
481 if (parse_rc == -1) {
482 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INVALID_URI]);
483 return s3_metrics_return_method(
484 r, method_slot,
485 s3_send_xml_error(r, NGX_HTTP_NOT_FOUND,
486 "NoSuchBucket",
487 "The specified bucket does not exist."));
488 }
489 if (parse_rc == 0) {
490 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INVALID_URI]);
491 return s3_metrics_return_method(
492 r, method_slot,
493 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
494 "InvalidURI",
495 "Couldn't parse the specified URI."));
496 }
497 }
498
499 /*
500 * WLCG bearer-token scope enforcement.
501 *
502 * When a request was authenticated via a bearer token, verify that the
503 * token's scope covers the requested path and operation BEFORE dispatching
504 * to any sub-handler. Non-token requests (SigV4, anonymous) are allowed
505 * unconditionally by brix_identity_check_token_scope (scopes only apply to
506 * token auth). Empty keys (bucket-level ops, list) map to logical path "/".
507 */
508 if (s3ctx->identity != NULL
509 && (s3ctx->identity->auth_method & BRIX_AUTHN_TOKEN))
510 {
511 brix_acc_op_t aop = s3_method_aop(r);
512 int need_write = (aop == BRIX_AOP_CREATE
513 || aop == BRIX_AOP_DELETE);
514 char logical[PATH_MAX];
515 u_char *end;
516
517 end = ngx_snprintf((u_char *) logical, sizeof(logical),
518 "/%s%Z", (const char *) key);
519 (void) end;
520
521 if (brix_identity_check_token_scope(s3ctx->identity, logical, need_write)
522 != NGX_OK)
523 {
524 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_ACCESS_DENIED]);
525 return s3_metrics_return_method(r, method_slot,
526 s3_send_xml_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
527 "token scope does not cover this object"));
528 }
529 }
530
531 if (is_list_request) {
532 return s3_metrics_return_method(r, method_slot, s3_handle_list(r, cf));
533 }
534
535 /*
536 * ListMultipartUploads: GET /<bucket>/?uploads (empty key, flag present)
537 * Must be checked before the empty-key rejection below.
538 */
539 if (r->method == NGX_HTTP_GET
540 && key[0] == '\0'
541 && s3_has_query_flag(r, "uploads"))
542 {
543 return s3_metrics_return_method(r, method_slot,
544 s3_handle_list_multipart_uploads(r, cf));
545 }
546
547 /*
548 * DeleteObjects: POST /<bucket>/?delete (empty key, flag present)
549 * Must be checked before the empty-key rejection below.
550 */
551 if (r->method == NGX_HTTP_POST
552 && key[0] == '\0'
553 && s3_has_query_flag(r, "delete"))
554 {
555 if (!cf->common.allow_write) {
556 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_WRITE_DISABLED]);
557 return s3_metrics_return_method(
558 r, method_slot,
559 s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
560 "AccessDenied",
561 "Write access is disabled."));
562 }
563 r->request_body_in_single_buf = 1;
564 rc = brix_http_read_body(r, s3_delete_objects_body_handler);
565 if (rc != NGX_DONE) {
566 return s3_metrics_return_method(r, method_slot, rc);
567 }
568 return NGX_DONE;
569 }

571 /*
572 * POST Object: browser form upload to /<bucket>/ with multipart/form-data.
573 * The authentication material, when configured, is carried in the form's
574 * policy/signature fields rather than in the HTTP Authorization header.
575 */
576 if (r->method == NGX_HTTP_POST && key[0] == '\0' && is_post_object_form) {
577 if (!cf->common.allow_write) {
578 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_WRITE_DISABLED]);
579 return s3_metrics_return_method(
580 r, method_slot,
581 s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
582 "AccessDenied",
583 "Write access is disabled."));
584 }
585
586 r->request_body_in_single_buf = 1;
587 rc = brix_http_read_body(r, s3_post_object_body_handler);
588 if (rc != NGX_DONE) {
589 return s3_metrics_return_method(r, method_slot, rc);
590 }
591 return NGX_DONE;
592 }
593
594 /*
595 * HeadBucket: HEAD /<bucket> (empty key). SDK clients probe this at session
596 * start; answer 200 + region rather than the empty-key InvalidURI below.
597 */
598 if (r->method == NGX_HTTP_HEAD && key[0] == '\0') {
599 return s3_metrics_return_method(r, method_slot,
600 s3_handle_head_bucket(r, cf));
601 }
602
603 /*
604 * Bucket-level GET (empty key). ListObjectsV2 (list-type=2) and
605 * ListMultipartUploads (?uploads) were already routed above; here we route
606 * the GetBucketLocation subresource probe and otherwise fall back to the
607 * original ListObjects (V1), which older tooling and Rucio still use.
608 */
609 if (r->method == NGX_HTTP_GET && key[0] == '\0') {
610 if (s3_has_query_flag(r, "location")) {
611 return s3_metrics_return_method(
612 r, method_slot, s3_handle_get_bucket_location(r, cf));
613 }
614 if (s3_has_query_flag(r, "versioning")) {
615 return s3_metrics_return_method(
616 r, method_slot, s3_handle_get_bucket_versioning(r));
617 }
618 if (s3_has_query_flag(r, "acl")) {
619 return s3_metrics_return_method(
620 r, method_slot, s3_handle_get_acl(r, cf));
621 }
622 if (s3_has_query_flag(r, "cors")) {
623 return s3_metrics_return_method(
624 r, method_slot, s3_handle_get_cors(r));
625 }
626 return s3_metrics_return_method(r, method_slot,
627 s3_handle_list_v1(r, cf));
628 }
629
630 if (key[0] == '\0') {
631 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INVALID_URI]);
632 return s3_metrics_return_method(
633 r, method_slot,
634 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
635 "InvalidURI", "Missing object key."));
636 }
637
638 if (!s3_resolve_key(cf->common.root_canon,
639 (const char *) key,
640 fs_path, sizeof(fs_path))) {
641 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_ACCESS_DENIED]);
642 return s3_metrics_return_method(
643 r, method_slot,
644 s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
645 "AccessDenied", "Access Denied."));
646 }
647 ngx_cpystrn((u_char *) s3ctx->fs_path, (u_char *) fs_path,
648 sizeof(s3ctx->fs_path));
649
650 if (r->method == NGX_HTTP_GET) {
651 char list_parts_upload_id[128];
652
653 if (s3_has_query_flag(r, "tagging")) {
654 return s3_metrics_return_method(r, method_slot,
655 s3_handle_get_object_tagging(r, fs_path, cf));
656 }
657 if (s3_has_query_flag(r, "acl")) {
658 return s3_metrics_return_method(r, method_slot,
659 s3_handle_get_acl(r, cf));
660 }
661 if (s3_get_query_param(r, "uploadId",
662 list_parts_upload_id,
663 sizeof(list_parts_upload_id)))
664 {
665 return s3_metrics_return_method(
666 r, method_slot,
667 s3_handle_list_parts(r, fs_path, cf, (const char *) key));
668 }
669 return s3_metrics_return_method(r, method_slot,
670 s3_handle_get(r, fs_path, cf));
671 }
672
673 if (r->method == NGX_HTTP_HEAD) {
674 return s3_metrics_return_method(r, method_slot,
675 s3_handle_head(r, fs_path, cf));
676 }
677
678 if (r->method == NGX_HTTP_PUT) {
679 if (!cf->common.allow_write) {
680 BRIX_S3_METRIC_INC(
681 events_total[BRIX_S3_EVENT_WRITE_DISABLED]);
682 return s3_metrics_return_method(
683 r, method_slot,
684 s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
685 "AccessDenied",
686 "Write access is disabled."));
687 }
688
689 /* PutObjectTagging: PUT /key?tagging (XML body). */
690 if (s3_has_query_flag(r, "tagging")) {
691 r->request_body_in_single_buf = 1;
692 rc = brix_http_read_body(r, s3_put_object_tagging_body_handler);
693 if (rc != NGX_DONE) {
694 return s3_metrics_return_method(r, method_slot, rc);
695 }
696 return NGX_DONE;
697 }
698 /* Per-object ACL writes are not supported (authz is XrdAcc/tokens). */
699 if (s3_has_query_flag(r, "acl")) {
700 return s3_metrics_return_method(r, method_slot,
701 s3_send_xml_error(r, NGX_HTTP_NOT_IMPLEMENTED, "NotImplemented",
702 "Per-object ACLs are not supported by this gateway."));
703 }
704
705 /* Check for multipart UploadPart: PUT /key?partNumber=N&uploadId=ID */
706 char upload_id[25];
707 char part_num_str[8];
708 if (s3_get_query_param(r, "uploadId", upload_id, sizeof(upload_id)) &&
709 s3_get_query_param(r, "partNumber", part_num_str, sizeof(part_num_str)))
710 {
711 char *endptr;
712 long part_num = strtol(part_num_str, &endptr, 10);
713 if (*endptr != '\0' || part_num < 1 || part_num > 10000) {
714 return s3_metrics_return_method(r, method_slot,
715 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
716 "InvalidArgument",
717 "Part number must be an integer"
718 " between 1 and 10000."));
719 }
720 if (!s3_has_query_flag(r, "uploads")) {
721 const char *c;
722 for (c = upload_id; *c != '\0'; c++) {
723 if (!((*c >= '0' && *c <= '9')
724 || (*c >= 'a' && *c <= 'f')))
725 {
726 return s3_metrics_return_method(r, method_slot,
727 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
728 "InvalidArgument",
729 "The uploadId is invalid."));
730 }
731 }
732 }
733
734 char mpu_dir[PATH_MAX];
735 u_char *p;
736
737 /* UploadPartCopy: the handler re-derives the staging dir from the
738 * DESTINATION KEY fs_path (s3_get_mpu_dir), so it must run BEFORE we
739 * overwrite fs_path with the part-file path below. Calling it after
740 * the overwrite fed it the part path and made s3_get_mpu_dir compute
741 * the wrong dir — UploadPartCopy always returned NoSuchUpload. */
742 if (brix_http_find_header(r, "x-amz-copy-source",
743 sizeof("x-amz-copy-source") - 1)
744 != NULL)
745 {
746 return s3_metrics_return_method(r, method_slot,
747 s3_handle_upload_part_copy(r, fs_path, cf,
748 upload_id, (int) part_num));
749 }
750
751 s3_get_mpu_dir(fs_path, upload_id, mpu_dir, sizeof(mpu_dir));
752 /*
753 * nginx's ngx_snprintf (ngx_sprintf) does NOT support the 'l'
754 * length modifier. Use %z (ssize_t/size_t) instead; on 64-bit
755 * Linux long and ssize_t are the same width.
756 */
757 p = ngx_snprintf((u_char *)fs_path, PATH_MAX - 1,
758 "%s/part.%z", mpu_dir, (size_t) part_num);
759 *p = '\0';
760 /* Sync ctx so the async put body handler sees the part path. */
761 ngx_cpystrn((u_char *) s3ctx->fs_path, (u_char *) fs_path,
762 sizeof(s3ctx->fs_path));
763 }
764
765 /* CopyObject: PUT /bucket/key + x-amz-copy-source (no uploadId) */
766 if (brix_http_find_header(r, "x-amz-copy-source",
767 sizeof("x-amz-copy-source") - 1) != NULL)
768 {
769 return s3_metrics_return_method(r, method_slot,
770 s3_handle_copy_object(r, fs_path,
771 cf));
772 }
773
774 /*
775 * Conditional PutObject (If-None-Match: * create-if-absent / If-Match
776 * overwrite-if-match), evaluated before the body is read. Skipped for
777 * UploadPart (conditional writes apply to whole objects only).
778 */
779 {
780 char part_probe[8];
781 if (!s3_get_query_param(r, "uploadId", part_probe,
782 sizeof(part_probe)))
783 {
784 ngx_int_t prc = s3_put_precondition(r, cf->common.root_canon,
785 fs_path);
786 if (prc != NGX_DECLINED) {
787 return s3_metrics_return_method(r, method_slot, prc);
788 }
789 /* W6b: If-None-Match:* asks for atomic create-if-absent — the
790 * commit (any of the 3 PUT body paths) uses an exclusive rename
791 * and maps a concurrent winner (EEXIST) to 412. */
792 s3ctx->exclusive_create = s3_put_is_exclusive_create(r) ? 1 : 0;
793 }
794 }
795
796 r->request_body_in_single_buf = 1;
797
798 rc = brix_http_read_body(r, s3_put_body_handler);
799 if (rc != NGX_DONE) {
800 return s3_metrics_return_method(r, method_slot, rc);
801 }
802 return NGX_DONE;
803 }
804
805 if (r->method == NGX_HTTP_DELETE) {
806 if (!cf->common.allow_write) {
807 BRIX_S3_METRIC_INC(
808 events_total[BRIX_S3_EVENT_WRITE_DISABLED]);
809 return s3_metrics_return_method(
810 r, method_slot,
811 s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
812 "AccessDenied",
813 "Write access is disabled."));
814 }
815
816 if (s3_has_query_flag(r, "tagging")) {
817 return s3_metrics_return_method(r, method_slot,
818 s3_handle_delete_object_tagging(r, fs_path, cf));
819 }
820
821 char upload_id[128];
822 if (s3_get_query_param(r, "uploadId", upload_id, sizeof(upload_id))) {
823 return s3_metrics_return_method(r, method_slot,
824 s3_handle_multipart_abort(r, fs_path, cf, upload_id));
825 }
826
827 return s3_metrics_return_method(r, method_slot,
828 s3_handle_delete(r, fs_path, cf));
829 }
830
831 if (r->method == NGX_HTTP_POST) {
832 if (!cf->common.allow_write) {
833 BRIX_S3_METRIC_INC(
834 events_total[BRIX_S3_EVENT_WRITE_DISABLED]);
835 return s3_metrics_return_method(
836 r, method_slot,
837 s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
838 "AccessDenied",
839 "Write access is disabled."));
840 }
841
842 char upload_id[128];
843 if (s3_has_query_flag(r, "uploads")) {
844 return s3_metrics_return_method(r, method_slot,
845 s3_handle_multipart_initiate(r, fs_path, cf, (const char *)key));
846 } else if (s3_get_query_param(r, "uploadId", upload_id, sizeof(upload_id))) {
847 r->request_body_in_single_buf = 1;
848
849 rc = brix_http_read_body(r, s3_multipart_complete_body_handler);
850 if (rc != NGX_DONE) {
851 return s3_metrics_return_method(r, method_slot, rc);
852 }
853 return NGX_DONE;
854 }
855 }
856
857 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_METHOD_NOT_ALLOWED]);
858 return s3_metrics_return_method(r, method_slot, NGX_HTTP_NOT_ALLOWED);
859}
