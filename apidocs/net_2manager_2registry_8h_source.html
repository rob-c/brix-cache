4/*
5 * manager/registry.h — shared-memory server registry for cluster/redirector mode.
6 *
7 * Data servers that connect via the CMS management protocol call
8 * brix_srv_register() on login and brix_srv_update_load() on each
9 * heartbeat. brix_srv_select() picks the best server for a given path
10 * and is called from the kXR_locate and kXR_open handlers when
11 * brix_manager_mode is on.
12 *
13 * The table lives in a dedicated ngx_shm_zone_t so all worker processes share
14 * one consistent view. Concurrent access is serialised by a single
15 * ngx_shmtx_t spinlock embedded at the start of the shared region.
16 *
17 * Capacity: BRIX_SRV_REGISTRY_SLOTS entries. When the table is full new
18 * registrations are silently dropped — existing servers continue to be used.
19 * BRIX_SRV_MAX_PATHS bounds the colon-delimited path list stored per entry.
20 */

157/*
158 * Phase 22 — active health checks.
159 *
160 * brix_srv_hc_claim(): under the registry spinlock, find the first in-use
161 * slot whose hc_next_check is due and that no other worker is probing. On
162 * success sets hc_in_progress=1, advances hc_next_check by interval_ms,
163 * copies host/port to the out params, and returns 1. Returns 0 if nothing
164 * is due — guaranteeing exactly one worker probes each server per interval.
165 * When nothing is due, *next_due_ms (if non-NULL) is set to the delay until
166 * the soonest server becomes due (clamped to interval_ms when none/empty), so
167 * the caller can sleep to that deadline instead of polling at a fixed floor.
168 *
169 * brix_srv_hc_pass(): probe succeeded — clears hc_fail_count, sets
170 * hc_last_ok, clears hc_in_progress, and clears a blacklist only if it was
171 * set by health checking (hc_fail_count was > 0), never a CMS-disconnect one.
172 *
173 * brix_srv_hc_fail(): probe failed — increments hc_fail_count, clears
174 * hc_in_progress, and blacklists the server for blacklist_ms once
175 * hc_fail_count reaches threshold. Returns 1 if it newly blacklisted.
176 */
177int brix_srv_hc_claim(char *host_out, size_t host_size,

222/*
223 * Like brix_srv_select(), but when no live (non-blacklisted) server matches it
224 * falls back to a currently-blacklisted one as a LAST RESORT. A CMS heartbeat
225 * drop blacklists a data server for 30 s even though its data plane is almost
226 * always still serving; the kXR_open / kXR_stat handlers use this so a transient
227 * heartbeat blip under load redirects to the (live) data node instead of a false
228 * kXR_NotFound. kXR_locate keeps the strict brix_srv_select() — it reports
229 * only live servers, so a genuinely dead node is still answered "not found"
