13/* --- confined recursive walk (src/fs/vfs/vfs_walk.c) ---------------------------
14 * A thread-safe, NON-allocating, NON-metered confined traversal for bulk/off-
15 * thread consumers (checksum-scan, recursive copy/remove) that cannot use the
16 * pool-allocating, metered handle API. The traversal (open/opendir/fstatat) runs
17 * inside the VFS over the rootfd-relative confinement (openat2 RESOLVE_BENEATH),
18 * so callers never touch a confined-helper directly; per regular file it invokes
19 * a callback. Symlinks/specials are never followed or reported. */

22 ngx_uint_t max_files; /* cap on regular files visited; 0 = unlimited */
23 unsigned open_files:1;/* open each regular file O_RDONLY and pass fd to cb */
24} brix_vfs_walk_opts_t;

190/* --- unified verified write session (src/fs/vfs/vfs_writer.c) ------------------
191 * ONE write entry point for every backend: a protocol write path (GridFTP STOR,
192 * and any future caller) opens a session, feeds extents, and commits — the
193 * session picks the mechanics from the backend's capabilities and, when asked,
194 * folds a self-computed CRC that is verified against a read-back on commit.
195 *
196 * random-write backend (POSIX default export, pblock): an in-place
197 * O_WRITE|O_CREATE handle written with brix_vfs_file_pwrite at arbitrary
198 * offsets — REST/APPE and out-of-order (MODE E) extents are supported.
199 * staged-only backend (S3/object store, no CAP_RANDOM_WRITE): an atomic
200 * staged upload (temp → commit); extents MUST arrive sequentially from
201 * offset 0 (an object store cannot patch bytes in place), so a non-sequential
202 * write is refused.
203 *
204 * `verify` asserts the caller is writing a whole object from offset 0 (a fresh
205 * STOR, not a REST/APPE partial); when set, every extent is CRC-32'd and, on
206 * commit, the persisted object is re-read through its driver and compared. A
207 * mismatch/short object unlinks the file and fails the commit. An empty object
208 * (nothing written) is a complete [0,0) write and verifies trivially. */
209#ifndef BRIX_VFS_WRITER_T_DECLARED
210#define BRIX_VFS_WRITER_T_DECLARED
211typedef struct brix_vfs_writer_s brix_vfs_writer_t;
212#endif

214/* Open a write session on the resolved `ctx`. The writer self-contains a deep
215 * copy of `ctx` (and the resolved-path / root_canon buffers it points at), so the
216 * caller's ctx may be a per-request stack frame that dies before commit.
217 * `flags` honours BRIX_VFS_O_TRUNC on the random-write path (the staged path is
218 * always a fresh object) and BRIX_VFS_O_ATOMIC (force the staged temp+publish path
219 * even for a random-write backend, so a failed write never leaves a partial at the
220 * final path — the WebDAV/S3 PUT contract). Returns NULL with *err_out (errno) set
221 * on failure. */
222brix_vfs_writer_t *brix_vfs_writer_open(brix_vfs_ctx_t *ctx, unsigned flags,
223 int verify, int *err_out);
224/* Write `len` bytes from `buf` at object offset `off`. NGX_OK / NGX_ERROR (errno
225 * set; EINVAL on a non-sequential extent to a staged-only backend). */
226ngx_int_t brix_vfs_writer_write(brix_vfs_writer_t *w, const void *buf,
227 size_t len, off_t off);
228/* Ingest `len` bytes from an already-open source fd (`src_fd` at `src_off`) onto
229 * the object at `dst_off` — the fd-to-fd twin of brix_vfs_writer_write, for a
230 * caller that has the body spooled to a temp fd (WebDAV/S3 PUT) rather than in a
231 * memory buffer. A single-fd, sendfile-capable random backend with verify OFF is
232 * moved kernel-side (copy_file_range, zero-copy); a verifying, staged/object, or
233 * block backend is read into a bounce buffer and pushed through the normal write
234 * engine so the CRC accumulator sees every byte and block/staged routing holds.
235 * NGX_OK / NGX_ERROR (errno set). */
236ngx_int_t brix_vfs_writer_write_fd(brix_vfs_writer_t *w, int src_fd,
237 off_t src_off, size_t len, off_t dst_off);
238/* Next offset the session expects: the sequential append cursor on the staged/
239 * object path (a write at any other offset is refused EINVAL), or the high-water
240 * byte count on the random path. -1 when w is NULL. Lets a caller distinguish an
241 * out-of-order write (its own error mapping) from a backend I/O failure. */
242off_t brix_vfs_writer_expected_off(const brix_vfs_writer_t *w);
243/* Finalize: fsync/close (random) or atomically publish (staged), then — when the
244 * session was opened with verify — re-read and CRC-compare the object, unlinking
245 * it on mismatch. NGX_OK on success, NGX_ERROR otherwise. Consumes the session:
246 * after commit the only valid follow-up is brix_vfs_writer_abort (a NULL-safe
247 * no-op once finished). */
248ngx_int_t brix_vfs_writer_commit(brix_vfs_writer_t *w);
249/* Like brix_vfs_writer_commit, but `excl` publishes with RENAME_NOREPLACE on the
250 * staged path (S3 If-None-Match exclusive create): NGX_ERROR with errno==EEXIST if
251 * the final object already exists. `excl` is a no-op on the in-place random path. */
252ngx_int_t brix_vfs_writer_commit_ex(brix_vfs_writer_t *w, unsigned excl);
253/* Discard an un-committed session: close + remove any staged temp / created
254 * object. Idempotent and NULL-safe. */
255void brix_vfs_writer_abort(brix_vfs_writer_t *w);
256/* The writable kernel fd behind the session — the in-place handle fd (random path)
257 * or the staged temp fd (staged path). NGX_INVALID_FILE for a NULL writer or a
258 * driver-backed object with no kernel fd. Callers that must write bytes outside
259 * brix_vfs_writer_write (Content-Encoding decode, aws-chunked de-framing) use this;
260 * an NGX_INVALID_FILE result means "no fd — that path is unsupported here". */
261ngx_fd_t brix_vfs_writer_fd(const brix_vfs_writer_t *w);
262/* The underlying staged handle on the staged path (NULL on the in-place random

264 * streaming de-chunkers) borrow the session's temp while the writer owns its
265 * lifecycle. Do NOT commit/abort it directly; drive the session via the writer. */
266brix_vfs_staged_t *brix_vfs_writer_staged(const brix_vfs_writer_t *w);
