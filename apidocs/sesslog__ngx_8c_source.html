20 * caller may provide NULL or a much longer path.
21 * HOW: Treat NULL as "-" and count until NUL or max bytes.

24brix_sess_ngx_strnlen_or_dash(const char *s, size_t max)
25{
26 size_t n;

28 if (s == NULL) {

33 /* count only */
34 }
35
36 return n;

39/*
40 * WHAT: Initialize the fixed per-worker registry free list.
41 * WHY: Session begin/end must be O(1) on the event loop and must never allocate.
42 * HOW: Thread every BSS slot through registry_next once per worker.
43 */
44static void
45brix_sess_registry_init(void)
46{
47 int i;
48
49 if (brix_sess_registry_ready) {
50 return;
51 }
52
53 for (i = 0; i < BRIX_SESSLOG_REGISTRY_SLOTS - 1; i++) {
54 brix_sess_registry[i].registry_next = i + 1;
55 }
56 brix_sess_registry[BRIX_SESSLOG_REGISTRY_SLOTS - 1].registry_next = -1;
57 brix_sess_free_head = 0;
58 brix_sess_registry_ready = 1;
59}

61/*
62 * WHAT: Copy a possibly non-NUL peer/path/id string into fixed session storage.
63 * WHY: Teardown-synthesized RESULT and XFER lines need stable values after the
64 * caller's request buffer has gone away.
65 * HOW: Truncate with a terminal NUL; full untruncated values are still used by
66 * immediate ATTEMPT/RESULT formatting at the call site.
67 */
68static void
69brix_sess_copy_value(char *dst, size_t dst_size, const char *src,
70 size_t src_len)
71{
72 size_t n;
73
74 if (dst_size == 0) {
75 return;
76 }
77
78 if (src == NULL || src_len == 0) {
79 src = "-";
80 src_len = 1;
81 }
82
83 n = src_len;
84 if (n >= dst_size) {
85 n = dst_size - 1;
86 }
87 if (n > 0) {
88 ngx_memcpy(dst, src, n);

93/*
94 * WHAT: Adapter from the existing access-log sanitizer to sesslog's len-aware
95 * sanitizer signature.
96 * WHY: The line grammar requires the same escaping policy as access_log.c.
97 * HOW: Copy the bounded source into a temporary NUL-terminated buffer, then use

100static size_t

102 size_t src_len)
103{
104 char tmp[BRIX_SESSLOG_PATH_MAX + 4];

111 if (src == NULL) {
112 src = "-";
113 src_len = 1;

128/*
129 * WHAT: Mint a 64-bit lowercase hex session-log ID.
130 * WHY: Operators need a compact correlation key distinct from the wire-level
131 * XRootD session id.
132 * HOW: Prefer RAND_bytes(8); fall back to pid/random/monotonic counter if the
133 * entropy source is unavailable.
134 */
135static void
136brix_sess_mint_id(char id[BRIX_SESSLOG_ID_LEN + 1])
137{
138 uint64_t v = 0;
139
140 if (RAND_bytes((unsigned char *) &v, sizeof(v)) != 1) {
141 brix_sess_fallback_counter++;
142 v = ((uint64_t) ngx_pid << 48)
143 ^ ((uint64_t) ngx_random() << 16)
144 ^ brix_sess_fallback_counter;
145 }
146
147 snprintf(id, BRIX_SESSLOG_ID_LEN + 1, "%016" PRIx64, v);
148}

150/*
151 * WHAT: Emit one formatter-produced event through the existing batched writer.
152 * WHY: Sesslog lines must share the access log's timestamp and fd semantics.
153 * HOW: Prefix with brix_access_log_time_prefix() and call brix_alog_emit().
154 */
155static void
156brix_sess_emit(brix_sess_t *s, const char *event, size_t event_len)
157{
158 char line[BRIX_SESSLOG_LINE_MAX + 96];
159 size_t prefix_len;
160
161 if (s == NULL || !s->in_use || s->log_fd == NGX_INVALID_FILE
162 || event == NULL || event_len == 0)
163 {
164 return;
165 }
166
167 prefix_len = brix_access_log_time_prefix(line, sizeof(line));
168 if (prefix_len == 0 || prefix_len >= sizeof(line)) {
169 return;
170 }
171

173 event_len = sizeof(line) - prefix_len;
174 }

179/*
180 * WHAT: Claim one registry slot for a new live session.
181 * WHY: The event loop cannot allocate or lock in the accept hot path.
182 * HOW: Pop the free list; return NULL and warn once if the registry is full.
183 */
184static brix_sess_t *
185brix_sess_alloc(ngx_log_t *log)
186{
187 brix_sess_t *s;
188 int idx;
189 int next;
190
191 brix_sess_registry_init();
192 idx = brix_sess_free_head;
193 if (idx < 0) {
194 if (!brix_sess_full_warned) {
195 if (log != NULL) {
196 ngx_log_error(NGX_LOG_WARN, log, 0,
197 "sesslog: registry full (%d); further sessions "
198 "unlogged", BRIX_SESSLOG_REGISTRY_SLOTS);
199 }
200 brix_sess_full_warned = 1;
201 }
202 return NULL;
203 }
204
205 s = &brix_sess_registry[idx];
206 next = s->registry_next;
207 ngx_memzero(s, sizeof(*s));
208 s->registry_next = -1;
209 s->in_use = 1;
210 brix_sess_free_head = next;
211
212 return s;
213}

215/*
216 * WHAT: Release a completed session slot.
217 * WHY: Long-lived workers must recycle session state without heap churn.
218 * HOW: Clear ownership-visible state and push the slot back onto the free list.
219 */
220static void
221brix_sess_release(brix_sess_t *s)
222{
223 int idx;
224
225 if (s == NULL) {
226 return;
227 }
228
229 idx = (int) (s - brix_sess_registry);
230 if (idx < 0 || idx >= BRIX_SESSLOG_REGISTRY_SLOTS) {
231 return;
232 }
233
234 s->in_use = 0;
235 s->xfers = NULL;
236 s->pending_attempt = 0;
237 s->registry_next = brix_sess_free_head;
238 brix_sess_free_head = idx;
239}
