117/* Finalize session metrics: decrement connections_active and accumulate the
118 * lifecycle rx/tx byte totals (with per-IP-version and per-protocol breakdowns),
119 * which must be committed here before ctx is destroyed. */
120
121static void
122brix_disconnect_update_metrics(brix_ctx_t *ctx)
123{
124 if (ctx->metrics == NULL) {
125 return;
126 }
127
128 ngx_atomic_fetch_add(&ctx->metrics->connections_active,
129 (ngx_atomic_int_t) -1);
130 ngx_atomic_fetch_add(&ctx->metrics->bytes_rx_total,
131 (ngx_atomic_int_t) ctx->totals.bytes_written);
132 ngx_atomic_fetch_add(&ctx->metrics->bytes_tx_total,
133 (ngx_atomic_int_t) ctx->totals.bytes);
134
135 /* Per-IP-version byte accounting — mirrors the aggregate totals above. */
136 if (ctx->ip_version == AF_INET6) {
137 ngx_atomic_fetch_add(&ctx->metrics->bytes_rx_ipv6_total,
138 (ngx_atomic_int_t) ctx->totals.bytes_written);
139 ngx_atomic_fetch_add(&ctx->metrics->bytes_tx_ipv6_total,
140 (ngx_atomic_int_t) ctx->totals.bytes);
141 } else if (ctx->ip_version == AF_INET) {
142 ngx_atomic_fetch_add(&ctx->metrics->bytes_rx_ipv4_total,
143 (ngx_atomic_int_t) ctx->totals.bytes_written);
144 ngx_atomic_fetch_add(&ctx->metrics->bytes_tx_ipv4_total,
145 (ngx_atomic_int_t) ctx->totals.bytes);
146 }
147
148 /* Per-protocol byte accounting for native stream-layer traffic. */
149 if (ngx_strcmp(ctx->protocol_label, "root") == 0) {
150 ngx_atomic_fetch_add(&ctx->metrics->proto_root_bytes_rx_total,
151 (ngx_atomic_int_t) ctx->totals.bytes_written);
152 ngx_atomic_fetch_add(&ctx->metrics->proto_root_bytes_tx_total,
153 (ngx_atomic_int_t) ctx->totals.bytes);

157/* Dominant byte count for an open handle in disconnect logging: bytes_written when
158 * any write occurred (upload), else bytes_read — so an interrupted upload is not
159 * logged as a zero-byte read. */
160
161static size_t
162brix_disconnect_file_bytes(const brix_file_t *file)
163{
164 /*
165 * A handle is either read-heavy or write-heavy in the access log. Prefer
166 * written bytes when present so interrupted uploads are not reported as
167 * zero-byte reads.
168 */
169 if (file->bytes_written > 0) {
170 return file->bytes_written;
171 }
172
173 return file->bytes_read;
174}

202/* Emit a kXR_Cancelled access-log entry for every still-open handle when the
203 * connection drops (detail "interrupted X.XXMB/s" or "interrupted"), with duration
204 * measured from the original open_time via req_start reuse. */
205
206static void
207brix_disconnect_log_open_files(brix_ctx_t *ctx, ngx_connection_t *c,
208 ngx_msec_t now)
209{
210 int handle_index;
211
212 for (handle_index = 0; handle_index < BRIX_MAX_FILES; handle_index++) {
213 brix_file_t *file;
214 char detail[64];
215 size_t byte_total;
216 ngx_msec_t duration_ms;
217
218 file = &ctx->files[handle_index];
219 if (file->fd < 0) {
220 continue;
221 }
222
223 byte_total = brix_disconnect_file_bytes(file);
224 duration_ms = now - file->open_time;
225
226 if (byte_total > 0 && duration_ms > 0) {
227 double mb_per_second;
228
229 mb_per_second = (double) byte_total / (double) duration_ms
230 / 1000.0;
231 snprintf(detail, sizeof(detail), "interrupted %.2fMB/s",
232 mb_per_second);
233 } else {
234 snprintf(detail, sizeof(detail), "interrupted");
235 }
236
237 /*
238 * Reuse req_start so the common access logger reports per-handle
239 * duration from the original open time, not from disconnect time.
240 */
241 ctx->req_start = file->open_time;
242 brix_log_access(ctx, c, "CLOSE", file->path, detail, 0,
243 kXR_Cancelled, "connection lost", byte_total);
244 if (file->sess_xfer.active && byte_total > file->sess_xfer.bytes) {
245 brix_sess_xfer_add(&file->sess_xfer,
246 (uint64_t) (byte_total - file->sess_xfer.bytes));
247 }
248 brix_sess_xfer_end(ctx->sess, &file->sess_xfer,
249 (ngx_exiting || ngx_terminate)
250 ? BRIX_SESS_XFER_SHUTDOWN
251 : BRIX_SESS_XFER_ABORTED);
252 }
253}

255/* Format session-level throughput for the final access-log entry: separate rx/tx
256 * MB/s when both occurred, otherwise a single aggregate. */
257
258static void
259brix_disconnect_format_session_detail(brix_ctx_t *ctx, ngx_msec_t now,
260 char *detail, size_t detail_size, size_t *total_bytes)
261{
262 ngx_msec_t session_duration_ms;
263
264 *total_bytes = ctx->totals.bytes + ctx->totals.bytes_written;
265 session_duration_ms = now - ctx->totals.start;
266
267 if (*total_bytes == 0 || session_duration_ms == 0) {
268 snprintf(detail, detail_size, "-");
269 return;
270 }
271
272 if (ctx->totals.bytes_written > 0) {
273 double read_mbps;
274 double write_mbps;
275
276 read_mbps = (double) ctx->totals.bytes
277 / (double) session_duration_ms / 1000.0;
278 write_mbps = (double) ctx->totals.bytes_written
279 / (double) session_duration_ms / 1000.0;
280 snprintf(detail, detail_size, "rx=%.2fMB/s tx=%.2fMB/s",
281 read_mbps, write_mbps);
282 return;
283 }
284
285 snprintf(detail, detail_size, "%.2fMB/s",
286 (double) *total_bytes / (double) session_duration_ms / 1000.0);
287}

289/* brix_on_disconnect — entry point for an unexpected TCP close (not a normal
290 * kXR_close): three-phase cleanup — buffer/crypto release, metrics finalization,
291 * and cancelled access-log entries — releasing every resource (buffers, crypto
292 * objects, open handles, registry slots). */
293
294void
295brix_on_disconnect(brix_ctx_t *ctx, ngx_connection_t *c)
296{
297 char session_detail[128];
298 size_t session_total_bytes;
299 ngx_msec_t now;
300
301 now = ngx_current_msec;
302 ctx->destroyed = 1;
303
304 /* phase-59 W3a: release any throttle open-files slots still held by this
305 * connection (handles closed implicitly by disconnect, not kXR_close). */
306 if (ctx->throttle.open_held > 0) {
307 ngx_stream_brix_srv_conf_t *tconf = ngx_stream_get_module_srv_conf(
308 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
309 if (tconf->throttle.zone != NULL) {
310 const char *tuser = ctx->login.dn[0] ? ctx->login.dn : "anonymous";
311 while (ctx->throttle.open_held > 0) {
312 brix_throttle_open_dec(tconf->throttle.zone, tuser);
313 ctx->throttle.open_held--;
314 }
315 } else {
316 ctx->throttle.open_held = 0;
317 }
318 }
319
320 /*
321 * Phase 39: disarm any steady-state read/write deadline this connection armed.
322 * nginx's connection finalize also tears c->read/c->write timers down, but
323 * clearing them here keeps the armed-bit bookkeeping consistent and is
324 * defensive against an AIO completion racing teardown. Guarded on our armed
325 * bit so it never touches the CMS/FRM WAITING timers that share c->read.
326 */
327 if (ctx->deadline.read_armed && c->read->timer_set) {
328 ngx_del_timer(c->read);
329 }
330 ctx->deadline.read_armed = 0;
331 if (ctx->deadline.send_armed && c->write->timer_set) {
332 ngx_del_timer(c->write);
333 }
334 ctx->deadline.send_armed = 0;
335
336 /* SciTags packet marking (phase-34): cancel the echo timer (its ngx_event_t
337 * lives in this ctx, freed with the pool below), then emit the firefly "end"
338 * report while the socket fd is still open (TCP_INFO byte/rtt read here).
339 * No-op if the connection was never marked. */
340 if (ctx->pmark.echo_ev.timer_set) {
341 ngx_del_timer(&ctx->pmark.echo_ev);
342 }
343 if (ctx->pmark.flow != NULL) {
344 brix_pmark_flow_end(ctx->pmark.flow, c->log);
345 ctx->pmark.flow = NULL;
346 }
347
348 /* Phase 31 W4: return this connection's charged transfer-heap bytes to the
349 * SHM-global budget before its scratch buffers are freed below. */
350 brix_budget_release(ctx);
351
352 /* Phase 25 W7 (stream): release the per-connection concurrency slot reserved
353 * by the dispatch gate. The stream plane has no per-request LOG phase, so the
354 * in-flight slot is held for the connection's lifetime and freed exactly once
355 * here. No-op if no concurrency rule matched. */
356 brix_rl_release_ctx(ctx);
357
358 /* Phase 51 (E4): release this connection's in-flight GSI-handshake slot if it
359 * still holds one (handshake aborted before completion). Leak-proof: this
360 * funnel always runs on close, and the release is gated by ctx->login.gsi_counted
361 * so a completed handshake (already released) is a no-op. */
362 brix_gsi_inflight_release(ctx);
363
364 if (ctx->upstream != NULL) {
365 brix_upstream_cleanup(ctx->upstream);
366 }
367
368 if (ctx->proxy != NULL) {
369 brix_proxy_cleanup(ctx->proxy);
370 ctx->proxy = NULL;
371 }
372
373 brix_release_disconnect_owned_buffers(ctx);
374 brix_release_disconnect_crypto_state(ctx);
375 brix_disconnect_update_metrics(ctx);
376 brix_disconnect_log_open_files(ctx, c, now);
377 brix_sess_end(ctx->sess, brix_disconnect_sess_reason(ctx, c));
378 ctx->sess = NULL;
379
380 /* Free any transfer monitor slots for this session (handles kXR_close was never sent). */
381 if (ngx_brix_dashboard_shm_zone != NULL) {
382 brix_transfer_slot_free_all_for_session(
383 ngx_brix_dashboard_shm_zone->data, ctx->login.sessid);
384 }
385
386 if (!ctx->login.logged_in) {
387 /* Phase 33 C1: make this connection's buffered access-log lines durable
388 * (it may have logged errors before login). */
389 brix_access_log_flush();
390 return;
391 }
392
393 /*
394 * Bound sessions are represented in the shared registry by their parent
395 * session. Only unregister the session that owns the registry slot.
396 */
397 if (ctx->login.auth_done && !ctx->is_bound) {
398 brix_session_unregister(ctx->login.sessid);
399 }
400
401 brix_disconnect_format_session_detail(ctx, now, session_detail,
402 sizeof(session_detail),
403 &session_total_bytes);
404
405 ctx->req_start = ctx->totals.start;
406 brix_log_access(ctx, c, "DISCONNECT", "-", session_detail, 1, 0, NULL,
407 session_total_bytes);
408
409 /* Phase 33 C1: the connection is gone — flush its batched access-log lines
410 * (including the DISCONNECT record above) so they are durable now rather
411 * than waiting for the next buffer-full / timer tick. */
412 brix_access_log_flush();
413}

415/* brix_defer_teardown_if_writing — hold off teardown while pwrites run.
416 *
417 * Called at every data-plane finalize site (recv EOF/timeout/error, send
418 * error/timeout). If a pipelined kXR_write is still in flight (wr_inflight > 0)
419 * the connection MUST NOT be finalized yet — ctx and the open fds are still
420 * referenced by pwrites running in worker threads. Records the pending finalize
421 * (status), marks the connection destroyed so the recv loop stops and write
422 * completion callbacks skip their ack, disarms our deadline timers so no stale
423 * timer re-fires, and returns 1 (caller must return without tearing down). The
424 * last write completion (brix_write_aio_done) then runs the real teardown via
425 * brix_run_deferred_teardown(). Returns 0 when there are no in-flight writes,
426 * i.e. teardown may proceed normally.
427 *
428 * WHY: Pipelined writes break the old "exactly one AIO in flight, recv suspended"
429 * invariant that made teardown trivially safe. This is the single chokepoint
430 * that restores the guarantee: no finalize while any pwrite references ctx/fds. */
431ngx_flag_t
432brix_defer_teardown_if_writing(brix_ctx_t *ctx, ngx_connection_t *c,
433 ngx_int_t status)
434{
435 if (ctx->out.wr_inflight == 0) {
436 return 0;
437 }
438
439 if (!ctx->out.finalize_pending) {
440 ctx->out.finalize_pending = 1;
441 ctx->out.finalize_status = status;
442 ctx->destroyed = 1;
443
444 if (ctx->deadline.read_armed && c->read->timer_set) {
445 ngx_del_timer(c->read);
446 }
447 ctx->deadline.read_armed = 0;
448 if (ctx->deadline.send_armed && c->write->timer_set) {
449 ngx_del_timer(c->write);
450 }
451 ctx->deadline.send_armed = 0;
452 }
453
454 return 1;
455}

457/* brix_run_deferred_teardown — finalize once the last pwrite has landed.
458 *
459 * Invoked from brix_write_aio_done when wr_inflight reaches 0 and a
460 * teardown was deferred. Runs the full teardown that was held off — on_disconnect
461 * (metrics/log/registry, idempotent re: the already-set destroyed flag), then
462 * close_all_files (now safe: no pwrite references any fd), then finalize the
463 * stream session with the recorded status. After this returns the ctx/pool are
464 * gone, so the caller must touch nothing further. */
465void
466brix_run_deferred_teardown(brix_ctx_t *ctx, ngx_connection_t *c)
467{
468 ngx_stream_session_t *s = c->data;
469 ngx_int_t status = ctx->out.finalize_status;
470
471 ctx->out.finalize_pending = 0;
472 brix_on_disconnect(ctx, c);
473 brix_close_all_files(ctx);
474 ngx_stream_finalize_session(s, status);
475}
