25 * paths) and any kXR_prepare staging paths, on every disconnect/close path. */

28brix_release_disconnect_owned_buffers(brix_ctx_t *ctx)
29{

119/*
120 * WHAT: Release any throttle open-files slots still held by this connection at
121 * disconnect, then clear the held count. No-op when the connection holds none.
122 * WHY: Handles closed implicitly by a dropped connection (phase-59 W3a) — rather
123 * than an explicit kXR_close — would otherwise leak their per-user open-files
124 * reservations in the throttle zone; this is the single release point on that path.
125 * HOW: (1) return early when no slots are held; (2) look up the stream srv conf;
126 * (3) with a configured throttle zone, decrement per user until the held count
127 * reaches 0; (4) without a zone, just clear the held count.
128 */
129static void
130brix_disconnect_release_throttle_slots(brix_ctx_t *ctx, ngx_connection_t *c)
131{
132 if (ctx->throttle.open_held == 0) {
133 return;
134 }
135
136 ngx_stream_brix_srv_conf_t *tconf = ngx_stream_get_module_srv_conf(
137 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
138 if (tconf->throttle.zone == NULL) {
139 ctx->throttle.open_held = 0;
140 return;
141 }
142
143 const char *tuser = ctx->login.dn[0] ? ctx->login.dn : "anonymous";
144 while (ctx->throttle.open_held > 0) {
145 brix_throttle_open_dec(tconf->throttle.zone, tuser);
146 ctx->throttle.open_held--;
147 }
148}

150/*
151 * WHAT: Disarm this connection's steady-state read/write deadline timers and
152 * clear the armed bits. No-op for any timer that is not currently armed+set.
153 * WHY: nginx connection finalize also tears c->read/c->write timers down, but
154 * clearing them here keeps the armed-bit bookkeeping consistent and is defensive
155 * against an AIO completion racing teardown (phase 39). The armed-bit guard keeps
156 * this from touching the CMS/FRM WAITING timers that share c->read.
157 * HOW: (1) delete the read timer only when our read deadline is armed and set,
158 * then clear the armed bit; (2) do the same for the write/send deadline timer.
159 */
160static void
161brix_disconnect_disarm_deadline_timers(brix_ctx_t *ctx, ngx_connection_t *c)
162{
163 if (ctx->deadline.read_armed && c->read->timer_set) {
164 ngx_del_timer(c->read);
165 }
166 ctx->deadline.read_armed = 0;
167
168 if (ctx->deadline.send_armed && c->write->timer_set) {
169 ngx_del_timer(c->write);
170 }
171 ctx->deadline.send_armed = 0;
172}

174/*
175 * WHAT: End SciTags packet marking for this connection: cancel the echo timer and
176 * emit the firefly "end" report, clearing ctx->pmark.flow. No-op if never marked.
177 * WHY: The echo timer's ngx_event_t lives in this ctx (freed with the pool during
178 * teardown), so it must be cancelled first; the "end" report must be emitted while
179 * the socket fd is still open because it reads TCP_INFO byte/rtt counters here.
180 * HOW: (1) delete the echo timer when it is set; (2) when a flow exists, emit its
181 * end report and null the pointer so later teardown does not touch it again.
182 */
183static void
184brix_disconnect_end_pmark(brix_ctx_t *ctx, ngx_connection_t *c)
185{
186 if (ctx->pmark.echo_ev.timer_set) {
187 ngx_del_timer(&ctx->pmark.echo_ev);
188 }
189 if (ctx->pmark.flow != NULL) {
190 brix_pmark_flow_end(ctx->pmark.flow, c->log);
191 ctx->pmark.flow = NULL;
192 }
193}

195/* brix_on_disconnect — entry point for an unexpected TCP close (not a normal
196 * kXR_close): three-phase cleanup — buffer/crypto release, metrics finalization,
197 * and cancelled access-log entries — releasing every resource (buffers, crypto
198 * objects, open handles, registry slots). */
199
200void
201brix_on_disconnect(brix_ctx_t *ctx, ngx_connection_t *c)
202{
203 char session_detail[128];
204 size_t session_total_bytes;
205 ngx_msec_t now;
206
207 now = ngx_current_msec;
208 ctx->destroyed = 1;
209
210#if (BRIX_HAVE_LIBURING)
211 /* Sever any in-flight io_uring ops owned by this connection BEFORE its
212 * pool (which holds their task structs and the completion ngx_event_t
213 * inside each task) can be destroyed: a late CQE for a freed task must be
214 * dropped by the reaper, never posted — a freed event linked into
215 * ngx_posted_events corrupts the queue and crashes the worker. */
216 brix_uring_orphan_owner(c);
217#endif
218
219 /* phase-59 W3a: release any throttle open-files slots still held by this
220 * connection (handles closed implicitly by disconnect, not kXR_close). */
221 brix_disconnect_release_throttle_slots(ctx, c);
222
223 /* Phase 39: disarm any steady-state read/write deadline this connection armed
224 * (SciTags echo timer is handled next). */
225 brix_disconnect_disarm_deadline_timers(ctx, c);
226
227 /* SciTags packet marking (phase-34): end the flow while the socket is open. */
228 brix_disconnect_end_pmark(ctx, c);
229
230 /* Phase 31 W4: return this connection's charged transfer-heap bytes to the
231 * SHM-global budget before its scratch buffers are freed below. */
232 brix_budget_release(ctx);
233
234 /* Phase 25 W7 (stream): release the per-connection concurrency slot reserved
235 * by the dispatch gate. The stream plane has no per-request LOG phase, so the
236 * in-flight slot is held for the connection's lifetime and freed exactly once
237 * here. No-op if no concurrency rule matched. */
238 brix_rl_release_ctx(ctx);
239
240 /* Phase 51 (E4): release this connection's in-flight GSI-handshake slot if it
241 * still holds one (handshake aborted before completion). Leak-proof: this
242 * funnel always runs on close, and the release is gated by ctx->login.gsi_counted
243 * so a completed handshake (already released) is a no-op. */
244 brix_gsi_inflight_release(ctx);
245
246 if (ctx->upstream != NULL) {
247 brix_upstream_cleanup(ctx->upstream);
248 }
249
250 if (ctx->proxy != NULL) {
251 brix_proxy_cleanup(ctx->proxy);
252 ctx->proxy = NULL;
253 }
254
255 brix_release_disconnect_owned_buffers(ctx);
256 brix_release_disconnect_crypto_state(ctx);
257 brix_disconnect_update_metrics(ctx);
258 brix_disconnect_log_open_files(ctx, c, now);
259 brix_sess_end(ctx->sess, brix_disconnect_sess_reason(ctx, c));
260 ctx->sess = NULL;
261
262 /* Free any transfer monitor slots for this session (handles kXR_close was never sent). */
263 if (ngx_brix_dashboard_shm_zone != NULL) {
264 brix_transfer_slot_free_all_for_session(
265 ngx_brix_dashboard_shm_zone->data, ctx->login.sessid);
266 }
267
268 if (!ctx->login.logged_in) {
269 /* Phase 33 C1: make this connection's buffered access-log lines durable
270 * (it may have logged errors before login). */
271 brix_access_log_flush();
272 return;
273 }
274
275 /*
276 * Bound sessions are represented in the shared registry by their parent
277 * session. Only unregister the session that owns the registry slot.
278 */
279 if (ctx->login.auth_done && !ctx->is_bound) {
280 brix_session_unregister(ctx->login.sessid);
281 }
282
283 brix_disconnect_format_session_detail(ctx, now, session_detail,
284 sizeof(session_detail),
285 &session_total_bytes);
286
287 ctx->req_start = ctx->totals.start;
288 brix_log_access(ctx, c, "DISCONNECT", "-", session_detail, 1, 0, NULL,
289 session_total_bytes);
290
291 /* Phase 33 C1: the connection is gone — flush its batched access-log lines
292 * (including the DISCONNECT record above) so they are durable now rather
293 * than waiting for the next buffer-full / timer tick. */
294 brix_access_log_flush();
295}

297/* brix_defer_teardown_if_writing — hold off teardown while pwrites run.
298 *
299 * Called at every data-plane finalize site (recv EOF/timeout/error, send
300 * error/timeout). If a pipelined kXR_write is still in flight (wr_inflight > 0)
301 * the connection MUST NOT be finalized yet — ctx and the open fds are still
302 * referenced by pwrites running in worker threads. Records the pending finalize
303 * (status), marks the connection destroyed so the recv loop stops and write
304 * completion callbacks skip their ack, disarms our deadline timers so no stale
305 * timer re-fires, and returns 1 (caller must return without tearing down). The
306 * last write completion (brix_write_aio_done) then runs the real teardown via
307 * brix_run_deferred_teardown(). Returns 0 when there are no in-flight writes,
308 * i.e. teardown may proceed normally.
309 *
310 * WHY: Pipelined writes break the old "exactly one AIO in flight, recv suspended"
311 * invariant that made teardown trivially safe. This is the single chokepoint
312 * that restores the guarantee: no finalize while any pwrite references ctx/fds. */
313ngx_flag_t
314brix_defer_teardown_if_writing(brix_ctx_t *ctx, ngx_connection_t *c,
315 ngx_int_t status)
316{
317 if (ctx->out.wr_inflight == 0) {
318 return 0;
319 }
320
321 if (!ctx->out.finalize_pending) {
322 ctx->out.finalize_pending = 1;
323 ctx->out.finalize_status = status;
324 ctx->destroyed = 1;
325
326 brix_disconnect_disarm_deadline_timers(ctx, c);
327 }
328
329 return 1;
330}

332/* brix_run_deferred_teardown — finalize once the last pwrite has landed.
333 *
334 * Invoked from brix_write_aio_done when wr_inflight reaches 0 and a
335 * teardown was deferred. Runs the full teardown that was held off — on_disconnect
336 * (metrics/log/registry, idempotent re: the already-set destroyed flag), then
337 * close_all_files (now safe: no pwrite references any fd), then finalize the
338 * stream session with the recorded status. After this returns the ctx/pool are
339 * gone, so the caller must touch nothing further. */
340void
341brix_run_deferred_teardown(brix_ctx_t *ctx, ngx_connection_t *c)
342{
343 ngx_stream_session_t *s = c->data;
344 ngx_int_t status = ctx->out.finalize_status;
345
346 ctx->out.finalize_pending = 0;
347 brix_on_disconnect(ctx, c);
348 brix_close_all_files(ctx);
349 ngx_stream_finalize_session(s, status);
350}
