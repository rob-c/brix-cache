1/*
2 * handler_object_route.c — S3 object-key method routing.
3 *
4 * WHAT: Routes a resolved object request (/<bucket>/<key>) to the correct S3
5 * sub-handler by HTTP method — GetObject/ListParts/tagging/ACL (GET),
6 * HeadObject (HEAD), PutObject/UploadPart/UploadPartCopy/CopyObject/tagging
7 * (PUT), DeleteObject/AbortMultipartUpload/tagging (DELETE), and
8 * InitiateMultipartUpload/CompleteMultipartUpload (POST). Bucket-level and
9 * list routing live in the sibling handler_dispatch.c; the entry handler and
10 * auth gate live in handler.c.
11 *
12 * WHY: Split out of handler.c (976 lines) under the phase-79 500-line file-size
13 * cap. Object-key method routing — including the multipart upload-part
14 * preparation and PUT precondition checks — is a self-contained concern,
15 * distinct from URI-parse/bucket dispatch (handler_dispatch.c) and the
16 * entry/auth path (handler.c).
17 *
18 * HOW: s3_dispatch_object_method() branches on r->method to a per-method router.
19 * Write routers gate on cf->common.allow_write via the shared
20 * s3_reject_write_disabled() helper (in handler_dispatch.c, declared in
21 * s3_handler_internal.h) and stream request bodies through the shared
22 * s3_read_body_metric() helper. Each router returns the content-handler rc
23 * directly (responses are already framed by the sub-handlers).
24 */

44static int
45s3_upload_id_is_hex(const char *upload_id)
46{
47 const char *c;
48
49 for (c = upload_id; *c != '\0'; c++) {
50 if (!((*c >= '0' && *c <= '9') || (*c >= 'a' && *c <= 'f'))) {
51 return 0;
52 }
53 }
54 return 1;
55}

57static ngx_int_t
58s3_dispatch_object_get(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
59 ngx_uint_t method_slot, const u_char *key, char *fs_path)
60{
61 char upload_id[128];
62
63 if (s3_has_query_flag(r, "tagging")) {
64 return s3_metrics_return_method(r, method_slot,
65 s3_handle_get_object_tagging(r, fs_path, cf));
66 }
67 if (s3_has_query_flag(r, "acl")) {
68 return s3_metrics_return_method(r, method_slot,
69 s3_handle_get_acl(r, fs_path, cf));
70 }
71 if (s3_get_query_param(r, "uploadId", upload_id, sizeof(upload_id))) {
72 return s3_metrics_return_method(r, method_slot,
73 s3_handle_list_parts(r, fs_path, cf, (const char *) key));
74 }
75 return s3_metrics_return_method(r, method_slot,
76 s3_handle_get(r, fs_path, cf));
77}

79static ngx_int_t
80s3_prepare_upload_part(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
81 ngx_http_s3_req_ctx_t *s3ctx, ngx_uint_t method_slot, char *fs_path)
82{
83 char upload_id[25];
84 char part_num_str[8];
85 char *endptr;
86 long part_num;
87 char mpu_dir[PATH_MAX];
88 u_char *p;
89
90 if (!s3_get_query_param(r, "uploadId", upload_id, sizeof(upload_id))
91 || !s3_get_query_param(r, "partNumber", part_num_str,
92 sizeof(part_num_str)))
93 {
94 return NGX_DECLINED;
95 }
96 part_num = strtol(part_num_str, &endptr, 10);
97 if (*endptr != '\0' || part_num < 1 || part_num > 10000) {
98 return s3_metrics_return_method(r, method_slot,
99 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST, "InvalidArgument",
100 "Part number must be an integer between 1 and 10000."));
101 }
102 if (!s3_has_query_flag(r, "uploads") && !s3_upload_id_is_hex(upload_id)) {
103 return s3_metrics_return_method(r, method_slot,
104 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST, "InvalidArgument",
105 "The uploadId is invalid."));
106 }
107 if (brix_http_find_header(r, "x-amz-copy-source",
108 sizeof("x-amz-copy-source") - 1) != NULL)
109 {
110 return s3_metrics_return_method(r, method_slot,
111 s3_handle_upload_part_copy(r, fs_path, cf, upload_id, (int) part_num));
112 }
113
114 s3_get_mpu_dir(fs_path, upload_id, mpu_dir, sizeof(mpu_dir));
115 p = ngx_snprintf((u_char *) fs_path, PATH_MAX - 1, "%s/part.%z",
116 mpu_dir, (size_t) part_num);
117 *p = '\0';
118 ngx_cpystrn((u_char *) s3ctx->fs_path, (u_char *) fs_path,
119 sizeof(s3ctx->fs_path));
120 return NGX_OK;
121}

123static ngx_int_t
124s3_apply_put_precondition(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
125 ngx_http_s3_req_ctx_t *s3ctx, ngx_uint_t method_slot, const char *fs_path)
126{
127 char part_probe[8];
128 ngx_int_t rc;
129
130 /* NGX_DECLINED = "no response sent, proceed to the PUT body"; any other rc
131 * is a response already flushed (e.g. 412) that must gate dispatch — the
132 * send itself returns NGX_OK, so the caller must key on NGX_DECLINED, not
133 * NGX_OK (same sentinel convention as s3_dispatch_parse_uri). */
134 if (s3_get_query_param(r, "uploadId", part_probe, sizeof(part_probe))) {
135 return NGX_DECLINED;
136 }
137 rc = s3_put_precondition(r, cf->common.root_canon, fs_path);
138 if (rc != NGX_DECLINED) {
139 return s3_metrics_return_method(r, method_slot, rc);
140 }
141 s3ctx->exclusive_create = s3_put_is_exclusive_create(r) ? 1 : 0;
142 return NGX_DECLINED;
143}

145static ngx_int_t
146s3_dispatch_object_put(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
147 ngx_http_s3_req_ctx_t *s3ctx, ngx_uint_t method_slot, char *fs_path)
148{
149 ngx_int_t rc;
150
151 if (!cf->common.allow_write) {
152 return s3_reject_write_disabled(r, method_slot);
153 }
154 if (s3_has_query_flag(r, "tagging")) {
155 return s3_read_body_metric(r, method_slot,
156 s3_put_object_tagging_body_handler);
157 }
158 if (s3_has_query_flag(r, "acl")) {
159 return s3_metrics_return_method(r, method_slot,
160 s3_send_xml_error(r, NGX_HTTP_NOT_IMPLEMENTED, "NotImplemented",
161 "Per-object ACLs are not supported by this gateway."));
162 }
163
164 rc = s3_prepare_upload_part(r, cf, s3ctx, method_slot, fs_path);
165 if (rc == NGX_DONE || (rc != NGX_OK && rc != NGX_DECLINED)) {
166 return rc;
167 }
168 if (brix_http_find_header(r, "x-amz-copy-source",
169 sizeof("x-amz-copy-source") - 1) != NULL)
170 {
171 return s3_metrics_return_method(r, method_slot,
172 s3_handle_copy_object(r, fs_path, cf));
173 }
174 rc = s3_apply_put_precondition(r, cf, s3ctx, method_slot, fs_path);
175 if (rc != NGX_DECLINED) {
176 return rc; /* precondition response already sent (e.g. 412) */
177 }
178 return s3_read_body_metric(r, method_slot, s3_put_body_handler);
179}

202static ngx_int_t
203s3_dispatch_object_post(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
204 ngx_uint_t method_slot, const u_char *key, char *fs_path)
205{
206 char upload_id[128];
207
208 if (!cf->common.allow_write) {
209 return s3_reject_write_disabled(r, method_slot);
210 }
211 if (s3_has_query_flag(r, "uploads")) {
212 return s3_metrics_return_method(r, method_slot,
213 s3_handle_multipart_initiate(r, fs_path, cf, (const char *) key));
214 }
215 if (s3_get_query_param(r, "uploadId", upload_id, sizeof(upload_id))) {
216 return s3_read_body_metric(r, method_slot,
217 s3_multipart_complete_body_handler);
218 }
219 /* A bare POST to an object key (no ?uploads / ?uploadId, and not a
220 * POST-object form — that path routes on the empty key earlier) has no S3
221 * semantics: 405, not an unhandled NGX_DECLINED that would fall through. */
222 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_METHOD_NOT_ALLOWED]);
223 return s3_metrics_return_method(r, method_slot, NGX_HTTP_NOT_ALLOWED);
224}

226ngx_int_t
227s3_dispatch_object_method(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
228 ngx_http_s3_req_ctx_t *s3ctx, ngx_uint_t method_slot, const u_char *key,
229 char *fs_path)
230{
231 if (r->method == NGX_HTTP_GET) {
232 return s3_dispatch_object_get(r, cf, method_slot, key, fs_path);
233 }
234 if (r->method == NGX_HTTP_HEAD) {
235 return s3_metrics_return_method(r, method_slot,
236 s3_handle_head(r, fs_path, cf));
237 }
238 if (r->method == NGX_HTTP_PUT) {
239 return s3_dispatch_object_put(r, cf, s3ctx, method_slot, fs_path);
240 }
241 if (r->method == NGX_HTTP_DELETE) {
242 return s3_dispatch_object_delete(r, cf, method_slot, fs_path);
243 }
244 if (r->method == NGX_HTTP_POST) {
245 return s3_dispatch_object_post(r, cf, method_slot, key, fs_path);
246 }
247
248 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_METHOD_NOT_ALLOWED]);
249 return s3_metrics_return_method(r, method_slot, NGX_HTTP_NOT_ALLOWED);
250}
