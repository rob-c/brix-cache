7/*
8 * s3put_stream_aws_chunked — decode and finalize an aws-chunked body.
9 *
10 * WHAT: routes an aws-chunked PUT/UploadPart body through the streaming
11 * de-chunker, rejecting a combined inner content-coding first.
12 * WHY: aws-chunked bodies carry chunk framing that must be stripped; a combined
13 * "aws-chunked,gzip" would be stored still-compressed (silent corruption), so
14 * it is rejected with the same 400 the non-chunked codec path uses.
15 * HOW: always terminal — either aborts the write session and finalizes a codec
16 * error, or hands off to s3_put_streaming (which finalizes asynchronously).
17 */
18static void
19s3put_stream_aws_chunked(s3put_state_t *st)
20{
21 /* A combined Content-Encoding such as "aws-chunked,gzip" means the body
22 * was compressed THEN chunk-framed. The streaming de-chunker strips only
23 * the chunk envelope, so the inner-compressed payload would be stored
24 * still compressed (silent object corruption). Reject rather than store
25 * undecoded bytes — the same invariant the non-chunked path enforces with
26 * 400 below. Plain aws-chunked (no inner coding) continues normally. */
27 if (s3_aws_chunked_has_inner_coding(st->r)) {
28 brix_vfs_writer_abort(st->writer);
29 s3_put_finalize_codec_error(st->r, NGX_HTTP_BAD_REQUEST);
30 return;
31 }
32 s3_put_streaming(st->r, st->cf, st->writer, (const char *) st->fs_path,
33 st->body_mode);
34}

37/*
38 * s3put_select_codec — resolve any Content-Encoding into st->put_codec.
39 *
40 * WHAT: parses the request Content-Encoding header to a codec id used by the
41 * body-write phase.
42 * WHY: an unknown / not-compiled-in encoding must never be stored undecoded —
43 * it is a 400 client error, decided once here.
44 * HOW: returns NGX_OK (st->put_codec set, defaulting to IDENTITY); returns
45 * NGX_DONE after aborting the write session and finalizing a 400.
46 */
47static ngx_int_t
48s3put_select_codec(s3put_state_t *st)
49{
50 ngx_table_elt_t *ce = brix_http_find_header(
51 st->r, "Content-Encoding", sizeof("Content-Encoding") - 1);
52
53 if (ce != NULL && ce->value.len > 0) {
54 const brix_codec_desc_t *d = brix_codec_by_http_token(
55 (const char *) ce->value.data, ce->value.len);
56 if (d == NULL || !d->available) {
57 /* Unknown / not-compiled-in Content-Encoding: never store the
58 * undecoded bytes — reject as a client error (400), not 500. */
59 brix_vfs_writer_abort(st->writer);
60 s3_put_finalize_codec_error(st->r, NGX_HTTP_BAD_REQUEST);
61 return NGX_DONE;
62 }
63 st->put_codec = d->id;
64 }
65
66 return NGX_OK;
67}

70/*
71 * s3put_stream_offload — try to hand the plain body write to the thread pool.
72 *
73 * WHAT: for identity-encoded non-empty bodies with a thread pool configured,
74 * posts the write to the pool (phase-46 W1a) so large/spooled uploads do not
75 * block the event loop.
76 * WHY: keeps the async-offload task setup and its several failure finalizations
77 * out of the top-level flow, while preserving the exact eligibility condition.
78 * HOW: returns NGX_DONE when the offload path took ownership (posted, or already
79 * finalized on error); returns NGX_AGAIN when not eligible so the caller runs
80 * the synchronous write.
81 */
82static ngx_int_t
83s3put_stream_offload(s3put_state_t *st)
84{
85 ngx_thread_task_t *task;
86 s3_put_aio_t *t;
87
88 /*
89 * phase-46 W1a: offload the body write to the thread pool for ALL plain
90 * (identity-encoded) bodies — including spooled and mixed ones, not just
91 * in-memory. The worker fn (s3_put_aio_thread) already streams every buf,
92 * memory and file-backed, via brix_http_body_write_to_fd (kernel
93 * copy_file_range for the spooled nginx temp file), so large uploads — the
94 * ones nginx spools to disk — stop blocking the event loop. Content-Encoding
95 * decode (put_codec != IDENTITY) and aws-chunked (handled above) still run
96 * synchronously.
97 */
98 if (!(st->body_summary.bytes > 0
99 && st->put_codec == BRIX_CODEC_IDENTITY
100 && s3_thread_pool(st->cf) != NULL))
101 {
102 return NGX_AGAIN;
103 }
104
105 task = ngx_thread_task_alloc(st->r->pool, sizeof(s3_put_aio_t));
106 if (task == NULL) {
107 brix_vfs_writer_abort(st->writer);
108 s3_put_finalize_error(st->r);
109 return NGX_DONE;
110 }
111
112 t = task->ctx;
113 t->r = st->r;
114 t->writer = st->writer;
115 t->len = st->body_summary.bytes;
116 t->body_mode = st->body_mode;
117 t->body_bytes = st->body_bytes;
118 ngx_cpystrn((u_char *) t->final_path, (u_char *) st->fs_path,
119 sizeof(t->final_path));
120 ngx_cpystrn((u_char *) t->root_canon,
121 (u_char *) st->cf->common.root_canon, sizeof(t->root_canon));
122
123 /*
124 * Phase 31 W2: no full-body collection — the thread streams the
125 * in-memory body bufs straight to the staged fd (s3_put_aio_thread).
126 */
127
128 brix_task_bind(task, s3_put_aio_thread, s3_put_aio_done);
129
130 if (ngx_thread_task_post(st->cf->common.thread_pool, task) != NGX_OK) {
131 brix_vfs_writer_abort(st->writer);
132 s3_put_finalize_error(st->r);
133 return NGX_DONE;
134 }
135
136 st->r->main->count++;
137 return NGX_DONE;
138}

141/*
142 * s3put_stream_sync — write the body to the staged fd synchronously.
143 *
144 * WHAT: the inline body-write path — decodes a Content-Encoding body or streams
145 * an identity body straight to the staged fd.
146 * WHY: reached only when the async offload declined (encoded body, empty body,
147 * or no thread pool); maps a decode failure to the right S3 client error.
148 * HOW: returns NGX_OK on a successful write; returns NGX_DONE after aborting the
149 * write session and finalizing the mapped error (413/400 codec, else 500).
150 */
151static ngx_int_t
152s3put_stream_sync(s3put_state_t *st)
153{
154 ngx_int_t wrc;
155 ngx_int_t decode_status = 0;
156
157 if (st->put_codec != BRIX_CODEC_IDENTITY) {
158 ngx_fd_t wfd = brix_vfs_writer_fd(st->writer);
159
160 /* Codec decode writes through a raw fd; a driver-backed object session
161 * has none (NGX_INVALID_FILE) — reject with 501 rather than corrupt. */
162 if (wfd == NGX_INVALID_FILE) {
163 errno = ENOSYS;
164 wrc = NGX_ERROR;
165 decode_status = NGX_HTTP_NOT_IMPLEMENTED;
166 } else {
167 wrc = brix_http_body_decode_to_fd(st->r, wfd,
168 (const char *) st->fs_path,
169 st->put_codec,
170 BRIX_DECODE_MAX_OUTPUT,
171 &st->body_summary,
172 &decode_status);
173 }
174 } else {
175 wrc = brix_http_body_write_to_writer(st->r, st->writer);
176 }
177
178 if (wrc == NGX_OK) {
179 return NGX_OK;
180 }
181
182 brix_vfs_writer_abort(st->writer);
183 /* Map a decode failure to a clean S3 client error (413 bomb / 400
184 * malformed); a genuine I/O failure (decode_status 0 or 5xx) stays
185 * a 500. */
186 if (decode_status == NGX_HTTP_REQUEST_ENTITY_TOO_LARGE
187 || decode_status == NGX_HTTP_BAD_REQUEST
188 || decode_status == NGX_HTTP_UNSUPPORTED_MEDIA_TYPE)
189 {
190 s3_put_finalize_codec_error(st->r, decode_status
191 == NGX_HTTP_REQUEST_ENTITY_TOO_LARGE
192 ? NGX_HTTP_REQUEST_ENTITY_TOO_LARGE
193 : NGX_HTTP_BAD_REQUEST);
194 } else {
195 s3_put_finalize_error(st->r);
196 }
197 return NGX_DONE;
198}

201/*
202 * s3put_stream_body — summarize the body, emit the dashboard start, and write.
203 *
204 * WHAT: classifies the request body, records the dashboard/xfer start, then
205 * routes to the aws-chunked, async-offload, or synchronous write path.
206 * WHY: consolidates every body-handling branch behind one phase entry so the
207 * top-level flow reads as precondition → open → stream → commit.
208 * HOW: returns NGX_OK when the body was written synchronously and the caller
209 * should commit; returns NGX_DONE when a terminal path (chunked, offload, or
210 * an error) already finalized or will asynchronously finalize the request.
211 */
212ngx_int_t
213s3put_stream_body(s3put_state_t *st)
214{
215 char identity[128];
216 ngx_int_t rc;
217
218 if (brix_http_body_summary(st->r, &st->body_summary) != NGX_OK) {
219 brix_vfs_writer_abort(st->writer);
220 s3_put_finalize_error(st->r);
221 return NGX_DONE;
222 }
223
224 st->body_bytes = st->body_summary.bytes;
225 st->body_mode = s3_put_body_mode(&st->body_summary);
226 s3_dashboard_identity(st->r, st->cf, identity, sizeof(identity));
227 (void) brix_dashboard_http_start_identity(st->r, (const char *) st->fs_path,
228 identity, "", BRIX_XFER_PROTO_S3, BRIX_XFER_DIR_WRITE,
229 s3_dashboard_put_op(st->r), (int64_t) st->body_bytes);
230
231 /*
232 * AWS streaming upload (aws-chunked): the body carries chunk framing that
233 * must be decoded to the object bytes — storing it verbatim corrupts every
234 * object from a default modern SDK/CLI client. Routed through its own
235 * decode → commit → checksum → finalize path.
236 */
237 if (s3_body_is_aws_chunked(st->r)) {
238 s3put_stream_aws_chunked(st);
239 return NGX_DONE;
240 }
241
242 if (s3put_select_codec(st) == NGX_DONE) {
243 return NGX_DONE;
244 }
245
246 rc = s3put_stream_offload(st);
247 if (rc == NGX_DONE) {
248 return NGX_DONE;
249 }
250
251 /* Not eligible for offload — write synchronously (encoded / empty / no pool). */
252 return s3put_stream_sync(st);
253}

256/*
257 * s3put_emit_etag — set the ETag header the S3 API requires on PutObject 200.
258 *
259 * WHAT: probes the freshly committed object and emits an ETag derived from its
260 * final size and mtime.
261 * WHY: S3 clients rely on the ETag to confirm the stored object; a probe miss
262 * silently skips it (the object still committed).
263 * HOW: pure side-effecting emit — no return value; called after a successful
264 * commit.
265 */
266static void
267s3put_emit_etag(s3put_state_t *st)
268{
269 brix_vfs_ctx_t fctx;
270 brix_vfs_stat_t fst;
271 char etag_buf[48];
272
273 s3_build_vfs_ctx(st->r, (const char *) st->fs_path, st->cf, &fctx);
274 if (brix_vfs_probe(&fctx, 1 /* no-follow */, &fst) == NGX_OK) {
275 struct stat final_sb;
276
277 ngx_memzero(&final_sb, sizeof(final_sb));
278 final_sb.st_mtime = fst.mtime;
279 final_sb.st_size = fst.size;
280 s3_etag(&final_sb, etag_buf, sizeof(etag_buf));
281 (void) s3_set_header(st->r, "ETag", etag_buf);
282 }
283}

286/*
287 * s3put_commit_and_headers — atomically commit the object, then emit headers.
288 *
289 * WHAT: promotes the staged file to the final object, verifies the client
290 * checksum, applies tagging/user-metadata, and finalizes the 200.
291 * WHY: the post-write phase — every step here runs only after the body reached
292 * the staged fd, and each has its own finalize-and-return path.
293 * HOW: always terminal — returns nothing; finalizes the request on the commit
294 * error, checksum-mismatch, or success path.
295 */
296void
297s3put_commit_and_headers(s3put_state_t *st)
298{
299 if (s3_commit_put(st->r, st->r->connection->log,
300 st->cf->common.root_canon, st->writer,
301 (const char *) st->fs_path) != NGX_OK)
302 {
303 if (s3_put_commit_conflict(st->r)) {
304 return;
305 }
306 brix_log_safe_path(st->r->connection->log, NGX_LOG_ERR, ngx_errno,
307 "s3: staged commit to \"%s\" failed",
308 (const char *) st->fs_path);
309 s3_put_finalize_error(st->r);
310 return;
311 }
312
313 /* S3 API requires ETag on PutObject 200 response */
314 s3put_emit_etag(st);
315
316 /* Full-object checksum verify (client-supplied) + echo; failures remove the
317 * object and send the matching 400. */
318 if (s3_put_checksum_failed(st->r, (const char *) st->fs_path,
319 st->cf->common.root_canon))
320 {
321 return;
322 }
323
324 (void) s3_apply_put_tagging_header(st->r, (const char *) st->fs_path,
325 st->cf->common.root_canon);
326 (void) s3_apply_put_user_metadata(st->r, (const char *) st->fs_path,
327 st->cf->common.root_canon);
328 s3_put_finalize_ok(st->r, st->body_bytes, st->body_mode);
329}
