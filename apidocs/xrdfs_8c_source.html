55/* Poll a path's residency until it is online (kXR_offline clears) or `timeout_s`
56 * elapses. Returns 0 online, 1 still offline at timeout, -1 on a stat error. */
57int
58wait_online(brix_conn *c, const char *path, int timeout_s, brix_status *st)
59{
60 int waited = 0;
61 for (;;) {
62 brix_statinfo si;
63 struct timespec ts;
64 brix_status_clear(st);
65 if (brix_stat(c, path, &si, st) != 0) { return -1; }
66 if (!(si.flags & kXR_offline)) { return 0; }
67 if (waited >= timeout_s) { return 1; }
68 ts.tv_sec = 1; ts.tv_nsec = 0;
69 nanosleep(&ts, NULL);
70 waited++;
71 }
72}

87/*
88 * WHAT: emit the "unknown command '<name>'" error plus a TTY did-you-mean hint.
89 * WHY: spec WS-7 — unknown-subcommand sites must offer a did-you-mean hint for
90 * interactive users without polluting pipeline output (C3). Factored out
91 * of dispatch() so the router stays a flat table lookup.
92 * HOW: build a NULL-terminated name array from COMMANDS[] (bounded), then pass
93 * it to brix_suggest() (Damerau-Levenshtein distance ≤ 2 match).
94 */
95static void
96dispatch_unknown(const char *name)
97{
98 int ncmds;
99 const char *names[48 + 1]; /* COMMANDS has ≤ 47 entries */
100 const xrdfs_cmd *h;
101 const char *suggestion;
102
103 ncmds = 0;
104 for (h = COMMANDS; h->name != NULL && ncmds < 48; h++) {
105 names[ncmds++] = h->name;
106 }
107 names[ncmds] = NULL;
108
109 fprintf(stderr, "xrdfs: unknown command '%s'\n", name);
110 suggestion = brix_suggest(name, names);
111 if (suggestion != NULL) {
112 brix_cli_hint("hint: did you mean '%s'?\n", suggestion);
113 }
114}

117/*
118 * WHAT: handle the REPL-builtin verbs (exit/quit/pwd/cd/help) that are not in
119 * COMMANDS[]. Returns 1 when it owned the verb (all builtins yield shell
120 * code 0); returns 0 when tok[0] is not a builtin.
121 * WHY: splitting the builtin ladder out of dispatch() drops the router's
122 * branch count below the complexity gate while freezing behavior/output.
123 * HOW: string-compare ladder with early return; cd rebuilds cwd via build_path,
124 * exit/quit set *quit. State (cwd/quit) is passed in explicitly.
125 */
126static int
127dispatch_builtin(char *cwd, size_t cwdsz, int ntok, char **tok, int *quit)
128{
129 if (strcmp(tok[0], "exit") == 0 || strcmp(tok[0], "quit") == 0) {
130 if (quit != NULL) { *quit = 1; }
131 return 1;
132 }
133 if (strcmp(tok[0], "pwd") == 0) {
134 printf("%s\n", cwd);
135 return 1;
136 }
137 if (strcmp(tok[0], "cd") == 0) {
138 char next[XRDC_PATH_MAX];
139 build_path(cwd, ntok >= 2 ? tok[1] : "/", next, sizeof(next));
140 snprintf(cwd, cwdsz, "%s", next);
141 return 1;
142 }
143 if (strcmp(tok[0], "help") == 0) {
144 for (const xrdfs_cmd *h = COMMANDS; h->name != NULL; h++) {
145 printf(" %s\n", h->help);
146 }
147 printf(" cd <path> | pwd | help | exit\n");
148 return 1;
149 }
150 return 0;
151}

154/* Run one tokenized command. cwd/cwdsz hold the mutable working directory (for
155 * the REPL's cd/pwd). Returns a shell code; sets *quit when asked to leave. */
156int
157dispatch(brix_conn *c, char *cwd, size_t cwdsz, int ntok, char **tok, int *quit,
158 const char *prog)
159{
160 const xrdfs_cmd *cmd;
161
162 if (ntok == 0) {
163 return 0;
164 }
165 if (dispatch_builtin(cwd, cwdsz, ntok, tok, quit)) {
166 return 0; /* all REPL builtins yield shell code 0 */
167 }
168
169 cmd = find_command(tok[0]);
170 if (cmd == NULL) {
171 dispatch_unknown(tok[0]);
172 return 50;
173 }
174 /* <cmd> --help: print the one-line synopsis for this command. */
175 if (ntok >= 2 && strcmp(tok[1], "--help") == 0) {
176 printf("usage: %s <endpoint> %s\n", brix_prog_base(prog), cmd->help);
177 brix_usage_footer(stdout, prog);
178 return 0;
179 }
180 return cmd->fn(c, cwd, ntok, tok);
181}

186int
187tokenize(char *line, char **tok, int maxtok)
188{
189 int n = 0;
190 char *p = line;
191
192 while (*p != '\0' && n < maxtok) {
193 while (*p != '\0' && isspace((unsigned char) *p)) { p++; }
194 if (*p == '\0') { break; }
195 tok[n++] = p;
196 while (*p != '\0' && !isspace((unsigned char) *p)) { p++; }
197 if (*p != '\0') { *p++ = '\0'; }
198 }
199 return n;
200}

203int
204repl(brix_conn *c, const char *host, int port, const char *prog)
205{
206 char cwd[XRDC_PATH_MAX] = "/";
207 char *line = NULL;
208 size_t cap = 0;
209 int last = 0;
210
211 for (;;) {
212 char *tok[XRDFS_MAXTOK];
213 int ntok, quit = 0;
214 ssize_t r;
215
216 printf("[%s:%d] %s > ", host, port, cwd);
217 fflush(stdout);
218
219 r = getline(&line, &cap, stdin);
220 if (r < 0) {
221 printf("\n");
222 break; /* EOF */
223 }
224 ntok = tokenize(line, tok, XRDFS_MAXTOK);
225 last = dispatch(c, cwd, sizeof(cwd), ntok, tok, &quit, prog);
226 if (quit) {
227 break;
228 }
229 }
230 free(line);
231 return last;
232}

270/*
271 * WHAT: outcome of the leading-option scan: how far argv was consumed, the
272 * optional WebDAV bearer token, and whether an early-exit verb (--version/
273 * --help/-h) already handled everything (with its exit code).
274 * WHY: lets parse_global_opts() report all three back to main() without new
275 * globals, so main() stays a thin driver.
276 */
277typedef struct {
278 int argi; /* index of the first non-option argv token */
279 const char *web_token; /* --token / -T bearer (NULL = env fallback) */
280 int done; /* 1 = an early-exit verb fired; use exit_code */
281 int exit_code; /* process exit code when done == 1 */
282} xrdfs_optscan;

284/*
285 * WHAT: handle the early-exit option verbs (--version / --help / -h). Returns 1
286 * (and prints) when a==one of them; 0 otherwise.
287 * WHY: these three verbs terminate option parsing with side effects; pulling
288 * them out keeps parse_global_opts()'s branch count under the gate.
289 * HOW: --version/--help go to stdout (WS-2), -h keeps its stderr usage (C1).
290 */
291static int
292opt_early_exit(const char *a, const char *prog)
293{
294 if (strcmp(a, "--version") == 0) {
295 printf("%s (BriX-Cache client) %s\n", brix_prog_base(prog),
296 brix_client_version());
297 return 1;
298 }
299 if (strcmp(a, "--help") == 0) {
300 usage_fp(stdout, prog); /* --help → stdout (WS-2) */
301 return 1;
302 }
303 if (strcmp(a, "-h") == 0) {
304 usage(prog); /* -h keeps existing stderr behavior (C1) */
305 return 1;
306 }
307 return 0;
308}

311/*
312 * WHAT: apply one two-argument option (--auth / --token|-T / --capture) at
313 * sc->argi. Returns 1 when consumed (advancing argi by 2), 0 otherwise.
314 * WHY: the value-flags carry the argi+1<argc guard and are the compound arms
315 * of the ladder; isolating them keeps opt_apply_flag() under the gate.
316 * HOW: each arm checks the token name and that a value follows, exactly as the
317 * original inline loop did.
318 */
319static int
320opt_apply_valueflag(int argc, char **argv, brix_opts *opts, xrdfs_optscan *sc)
321{
322 char *a = argv[sc->argi];
323 if (strcmp(a, "--auth") == 0 && sc->argi + 1 < argc) {
324 opts->auth_force = argv[sc->argi + 1]; sc->argi += 2; return 1;
325 }
326 if ((strcmp(a, "--token") == 0 || strcmp(a, "-T") == 0)
327 && sc->argi + 1 < argc) {
328 sc->web_token = argv[sc->argi + 1]; sc->argi += 2; return 1; /* http(s)/WebDAV bearer */
329 }
330 if (strcmp(a, "--capture") == 0 && sc->argi + 1 < argc) {
331 opts->capture = argv[sc->argi + 1]; sc->argi += 2; return 1;
332 }
333 return 0;
334}

337/*
338 * WHAT: apply one leading -/-- option token at sc->argi to *opts, advancing
339 * sc->argi. Returns 1 when the token was a recognized option, 0 when it
340 * is not an option we consume (caller stops the scan).
341 * WHY: the flag ladder is the bulk of parse_global_opts()'s branches; keeping
342 * the flag arms in helpers holds each function under the gate.
343 * HOW: boolean flags handled inline here; two-arg flags delegate to
344 * opt_apply_valueflag(). Relative order matches the original inline loop.
345 */
346static int
347opt_apply_flag(int argc, char **argv, brix_opts *opts, xrdfs_optscan *sc)
348{
349 char *a = argv[sc->argi];
350 if (strcmp(a, "--no-cwd") == 0) { sc->argi++; return 1; }
351 if (strcmp(a, "--tls") == 0) { opts->want_tls = 1; sc->argi++; return 1; }
352 if (strcmp(a, "--notlsok") == 0) { opts->notlsok = 1; sc->argi++; return 1; }
353 if (strcmp(a, "--noverifyhost") == 0) { opts->verify_host = 0; sc->argi++; return 1; }
354 if (opt_apply_valueflag(argc, argv, opts, sc)) { return 1; }
355 if (strcmp(a, "--wire-trace") == 0) { opts->wire_trace = 1; sc->argi++; return 1; }
356 if (strncmp(a, "--wire-trace=", 13) == 0) { opts->wire_trace = atoi(a + 13); sc->argi++; return 1; }
357 if (strcmp(a, "--timing") == 0) { opts->timing = 1; sc->argi++; return 1; }
358 if (strcmp(a, "--redirect-trace") == 0) { opts->redir_trace = 1; sc->argi++; return 1; }
359 return 0;
360}

363/*
364 * WHAT: parse the leading -/-- options into *opts, advancing over argv.
365 * WHY: the option ladder is the bulk of main()'s branch count; isolating it
366 * drops main below the complexity gate with byte-identical behavior.
367 * HOW: per token, try opt_early_exit() (terminates via sc->done) then
368 * opt_apply_flag() (value flags); an unrecognized token stops the scan.
369 * Order matches the original inline loop exactly.
370 */
371static void
372parse_global_opts(int argc, char **argv, brix_opts *opts, xrdfs_optscan *sc)
373{
374 sc->argi = 1;
375 sc->web_token = NULL;
376 sc->done = 0;
377 sc->exit_code = 0;
378
379 while (sc->argi < argc && argv[sc->argi][0] == '-'
380 && strcmp(argv[sc->argi], "-") != 0) {
381 if (opt_apply_flag(argc, argv, opts, sc)) {
382 continue;
383 }
384 if (opt_early_exit(argv[sc->argi], argv[0])) {
385 sc->done = 1; sc->exit_code = 0;
386 return;
387 }
388 break;
389 }
390}

393/*
394 * WHAT: run an http(s)/WebDAV endpoint — no root:// session; serve read-only
395 * metadata (ls, stat) over WebDAV PROPFIND, mirroring official xrdfs.
396 * WHY: extracted from main() so each transport branch is its own unit; keeps
397 * the WebDAV error strings and exit codes exactly as before.
398 * HOW: parse the web URL, reject s3://, trim the export base, populate web_ctx,
399 * require a command, then hand off to web_dispatch(). endpoint/opts are
400 * passed in; args[] is the remaining argv slice.
401 */
402static int
403run_web_endpoint(const char *endpoint, const brix_opts *opts,
404 const char *web_token, int nargs, char **args)
405{
406 brix_weburl wu;
407 char base[XRDC_PATH_MAX];
408 size_t bl;
409 web_ctx w;
410
411 if (brix_weburl_parse(endpoint, &wu) != 0) {
412 fprintf(stderr, "xrdfs: bad web URL: %s\n", endpoint);
413 return 50;
414 }
415 if (wu.is_s3) {
416 fprintf(stderr, "xrdfs: s3:// endpoints are not supported "
417 "(use http/https/dav/davs)\n");
418 return 50;
419 }
420 snprintf(base, sizeof(base), "%s", wu.path); /* URL path = export base */
421 bl = strlen(base);
422 while (bl > 0 && base[bl - 1] == '/') {
423 base[--bl] = '\0';
424 }
425 w.u = &wu;
426 w.base = base;
427 w.bearer = web_token != NULL ? web_token : getenv("BEARER_TOKEN");
428 w.verify = opts->verify_host;
429 w.ca_dir = brix_resolve_ca_dir(NULL);
430 if (nargs <= 0) {
431 fprintf(stderr, "xrdfs: an http(s)/WebDAV endpoint needs a command "
432 "(e.g. ls); the interactive shell is root:// only\n");
433 return 50;
434 }
435 return web_dispatch(&w, nargs, args);
436}

439/*
440 * WHAT: run a root:// endpoint — resolve the URL, connect, then either drop
441 * into the interactive shell (no command) or dispatch one command.
442 * WHY: extracted from main() so the root:// path is a single unit and main is
443 * a thin driver; connect error strings + shell codes are unchanged.
444 * HOW: endpoint_to_url → brix_connect_resilient → repl()/dispatch() → close.
445 * nargs<=0 selects the REPL. opts is passed in; a private cwd is used.
446 */
447static int
448run_root_endpoint(const char *endpoint, brix_opts *opts, int nargs, char **args,
449 const char *prog)
450{
451 brix_url u;
452 brix_conn c;
453 brix_status st;
454 char cwd[XRDC_PATH_MAX] = "/";
455 int rc;
456
457 brix_status_clear(&st);
458 if (endpoint_to_url(endpoint, &u, &st) != 0) {
459 fprintf(stderr, "xrdfs: %s\n", st.msg);
460 return 50;
461 }
462 if (brix_connect_resilient(&c, &u, opts, &st) != 0) {
463 fprintf(stderr, "xrdfs: connect %s:%d: %s\n", u.host, u.port, st.msg);
464 return brix_shellcode(&st);
465 }
466
467 if (nargs <= 0) {
468 rc = repl(&c, u.host, u.port, prog); /* no command → interactive shell */
469 } else {
470 rc = dispatch(&c, cwd, sizeof(cwd), nargs, args, NULL, prog);
471 }
472
473 brix_close(&c);
474 return rc;
475}

478int
479main(int argc, char **argv)
480{
481 brix_opts opts;
482 xrdfs_optscan sc = {0};
483 const char *endpoint;
484
485 memset(&opts, 0, sizeof(opts));
486 opts.verify_host = 1;
487 brix_crypto_init(); /* arm libxrdproto SHA-256/HMAC for GSI + sigver */
488
489 parse_global_opts(argc, argv, &opts, &sc);
490 if (sc.done) {
491 return sc.exit_code;
492 }
493
494 if (sc.argi >= argc) { usage(argv[0]); return 50; }
495 endpoint = argv[sc.argi++];
496
497 /* http(s)/WebDAV endpoint: no root:// session (read-only metadata). */
498 if (brix_is_web_url(endpoint)) {
499 return run_web_endpoint(endpoint, &opts, sc.web_token,
500 argc - sc.argi, &argv[sc.argi]);
501 }
502
503 return run_root_endpoint(endpoint, &opts, argc - sc.argi, &argv[sc.argi],
504 argv[0]);
505}
