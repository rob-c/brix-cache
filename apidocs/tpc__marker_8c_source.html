1/*
2 * tpc_marker.c — 202 Accepted response with WLCG Performance-Marker streaming.
3 *
4 * When brix_webdav_tpc_marker_interval is non-zero, HTTP-TPC COPY requests
5 * receive a "202 Accepted" response immediately and stream Performance-Marker
6 * blocks in the chunked body while the curl transfer runs in a thread pool
7 * worker. The body ends with "success\r\n" or "failure\r\n" when the transfer
8 * completes. This matches the XrdHttp wire behaviour expected by FTS and RUCIO.
9 *
10 * For multi-stream pull (X-Number-Of-Streams: N), N parallel Range-based GET
11 * streams run concurrently via curl_multi. Each stream's write callback
12 * increments an atomic per-stream byte counter; the poll timer reads these and
13 * emits one Performance-Marker block per stripe per interval:
14 *
15 * Perf Marker\r\n
16 * Timestamp: <unix-epoch-seconds>\r\n
17 * Stripe Index: <0..N-1>\r\n
18 * Stripe Bytes Transferred: <bytes>\r\n
19 * Total Stripe Count: <N>\r\n
20 * End\r\n
21 *
22 * For single-stream pull, byte counts are read from stat(2) on the in-progress
23 * temp file. For push, interim markers report 0 bytes; the final marker carries
24 * the full file size.
25 *
26 * The poll timer fires every TPC_MARKER_POLL_MSEC and is re-armed until the
27 * thread marks progress->completed = 1. A pool cleanup callback aborts the
28 * transfer if the request is destroyed before completion.
29 */

50static void
51tpc_marker_send_one(ngx_http_request_t *r, time_t ts,
52 ngx_uint_t stripe_index, off_t bytes, ngx_uint_t total_stripes)
53{
54 char block[512];
55 int n;
56 ngx_buf_t *b;
57 ngx_chain_t out;
58
59 n = snprintf(block, sizeof(block),
60 "Perf Marker\r\n"
61 "Timestamp: %ld\r\n"
62 "Stripe Index: %u\r\n"
63 "Stripe Bytes Transferred: %lld\r\n"
64 "Total Stripe Count: %u\r\n"
65 "End\r\n",
66 (long) ts, (unsigned) stripe_index,
67 (long long) bytes, (unsigned) total_stripes);
68
69 if (n <= 0 || n >= (int) sizeof(block)) {
70 return;
71 }
72
73 b = ngx_create_temp_buf(r->pool, (size_t) n);
74 if (b == NULL) {
75 return;
76 }
77
78 ngx_memcpy(b->pos, block, (size_t) n);
79 b->last = b->pos + n;
80 out.buf = b;
81 out.next = NULL;
82 (void) ngx_http_output_filter(r, &out);
83}

85void
86tpc_marker_send_all(ngx_http_request_t *r, time_t ts,
87 tpc_marker_ctx_t *ctx)
88{
89 tpc_ms_progress_t *progress = ctx->progress;
90 ngx_uint_t n = progress->n_streams;
91 ngx_uint_t i;
92
93 if (n > 1) {
94 /* Emit one block per stripe with atomically-read per-stream counts. */
95 for (i = 0; i < n; i++) {
96 off_t stream_bytes = (off_t) progress->bytes_per_stream[i];
97 tpc_marker_send_one(r, ts, i, stream_bytes, n);
98 }
99 } else {
100 /* Single-stream: approximate via stat on the in-progress temp file. */
101 off_t bytes = 0;
102 if (ctx->is_pull && ctx->tmp_path[0] != '\0') {
103 struct stat sb;
104 if (stat(ctx->tmp_path, &sb) == 0) { /* vfs-seam-allow: TPC in-progress transfer temp (committed via rename) */
105 bytes = sb.st_size;
106 }
107 }
108 tpc_marker_send_one(r, ts, 0, bytes, 1);
109 }
110}

113/*
114 * tpc_marker_final_bytes — WHAT: compute the byte count for the final marker.
115 * WHY: the last Performance-Marker and the success metric both report the total
116 * transferred size, sourced differently for pull (stat the temp file) vs push
117 * (the pre-recorded local file size). HOW: pull with a live temp path stats it
118 * (0 on failure); otherwise pull is 0 and push is the recorded size. Pure query
119 * apart from the stat(2) probe — no state mutation.
120 */
121static off_t
122tpc_marker_final_bytes(tpc_marker_ctx_t *ctx)
123{
124 if (ctx->is_pull && ctx->tmp_path[0] != '\0') {
125 struct stat sb;
126 return (stat(ctx->tmp_path, &sb) == 0) ? sb.st_size : 0; /* vfs-seam-allow: TPC in-progress transfer temp */
127 }
128 return ctx->is_pull ? 0 : ctx->push_file_size;
129}

131/*
132 * tpc_marker_commit_pull — WHAT: atomically publish the pull temp file to its
133 * final path (link+unlink for no-overwrite, rename for overwrite). WHY: a
134 * successful pull stages into a temp file that must be committed under the
135 * caller-selected overwrite semantics; a commit failure downgrades the whole
136 * transfer to failed. HOW: pick link vs rename by ctx->overwrite; on error log,
137 * unlink the temp, bump the commit-error metric and return 1 (now failed); clear
138 * tmp_path in every branch so cleanup does not double-act. Returns the possibly
139 * updated failed flag.
140 */
141static int
142tpc_marker_commit_pull(ngx_http_request_t *r, tpc_marker_ctx_t *ctx)
143{
144 if (!ctx->overwrite) {
145 if (brix_link_confined_canon(r->connection->log, ctx->root_canon,
146 ctx->tmp_path, ctx->final_path) != 0)
147 {
148 ngx_log_error(NGX_LOG_ERR, r->connection->log, ngx_errno,
149 "brix_webdav: TPC marker link() failed");
150 brix_unlink_confined_canon(r->connection->log, ctx->root_canon,
151 ctx->tmp_path, 0);
152 ctx->tmp_path[0] = '\0';
153 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_COMMIT_ERROR]);
154 return 1;
155 }
156 brix_unlink_confined_canon(r->connection->log, ctx->root_canon,
157 ctx->tmp_path, 0);
158 ctx->tmp_path[0] = '\0';
159 return 0;
160 }
161
162 if (brix_rename_confined_canon(r->connection->log, ctx->root_canon,
163 ctx->tmp_path, ctx->final_path) != 0)
164 {
165 ngx_log_error(NGX_LOG_ERR, r->connection->log, ngx_errno,
166 "brix_webdav: TPC marker rename() failed");
167 brix_unlink_confined_canon(r->connection->log, ctx->root_canon,
168 ctx->tmp_path, 0);
169 ctx->tmp_path[0] = '\0';
170 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_COMMIT_ERROR]);
171 return 1;
172 }
173 ctx->tmp_path[0] = '\0';
174 return 0;
175}

177/*
178 * tpc_marker_settle_temp — WHAT: resolve the pull temp file's fate before the
179 * result is reported. WHY: a success must commit the staged temp to its final
180 * name; a failure must discard it — either way the outcome may flip failed.
181 * HOW: on success commit via tpc_marker_commit_pull; on failure unlink and clear
182 * the path. Non-pull or already-consumed temp is a no-op. Returns the possibly
183 * updated failed flag.
184 */
185static int
186tpc_marker_settle_temp(ngx_http_request_t *r, tpc_marker_ctx_t *ctx, int failed)
187{
188 if (!ctx->is_pull || ctx->tmp_path[0] == '\0') {
189 return failed;
190 }
191
192 if (!failed) {
193 return tpc_marker_commit_pull(r, ctx);
194 }
195
196 brix_unlink_confined_canon(r->connection->log, ctx->root_canon,
197 ctx->tmp_path, 0);
198 ctx->tmp_path[0] = '\0';
199 return failed;
200}

202/*
203 * tpc_marker_trailer_t — WHAT: the chunked-body trailer word and its length.
204 * WHY: lets tpc_marker_record_result return the "success\r\n"/"failure\r\n"
205 * selection by value instead of through a pair of out-params. HOW: filled by
206 * record_result, consumed by send_trailer.
207 */
208typedef struct {
209 const char *line;
210 size_t len;
211} tpc_marker_trailer_t;
212
213/*
214 * tpc_marker_record_result — WHAT: emit the terminal registry/metric/dashboard
215 * bookkeeping and return the trailer status word. WHY: after the temp is
216 * settled the transfer must deregister and record success/error counters plus
217 * the dashboard byte total, then the chunked body is closed with success/failure.
218 * HOW: always remove the registry entry; on failure bump error metrics and log a
219 * dashboard error; on success bump the direction-specific success metrics and add
220 * bytes; finish the dashboard. Returns the trailer word by value. Side effects
221 * live here (edge); no return-value control flow.
222 */
223static tpc_marker_trailer_t
224tpc_marker_record_result(ngx_http_request_t *r, tpc_marker_ctx_t *ctx,
225 int failed, off_t final_bytes)
226{
227 tpc_marker_trailer_t trailer = { NULL, 0 };
228
229 (void) brix_tpc_registry_remove(ctx->transfer_id, r->connection->log);
230
231 if (failed) {
232 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_ERROR]);
233 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV,
234 ctx->is_pull ? BRIX_TPC_DIR_PULL
235 : BRIX_TPC_DIR_PUSH,
236 BRIX_TPC_METRIC_ERROR, 0,
237 r->connection->log);
238 brix_dashboard_http_error(r, "webdav TPC failed");
239 trailer.line = "failure\r\n";
240 trailer.len = sizeof("failure\r\n") - 1;
241 brix_dashboard_http_finish(r);
242 return trailer;
243 }
244
245 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_SUCCESS]);
246 if (ctx->is_pull) {
247 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_PULL_SUCCESS]);
248 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV, BRIX_TPC_DIR_PULL,
249 BRIX_TPC_METRIC_SUCCESS,
250 (size_t) final_bytes, r->connection->log);
251 } else {
252 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_PUSH_SUCCESS]);
253 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV, BRIX_TPC_DIR_PUSH,
254 BRIX_TPC_METRIC_SUCCESS,
255 (size_t) final_bytes, r->connection->log);
256 }
257 if (final_bytes > 0) {
258 brix_dashboard_http_add(r, (ngx_atomic_int_t) final_bytes);
259 }
260 trailer.line = "success\r\n";
261 trailer.len = sizeof("success\r\n") - 1;
262 brix_dashboard_http_finish(r);
263 return trailer;
264}

266/*
267 * tpc_marker_send_trailer — WHAT: write the trailer status word and close the
268 * chunked response. WHY: the 202-streaming body ends with "success\r\n" or
269 * "failure\r\n" followed by the terminating chunk, then the request is finalized.
270 * HOW: build a temp buf with the status text (best-effort; skip on alloc
271 * failure), flush it, send the final chunk and finalize with NGX_DONE.
272 */
273static void
274tpc_marker_send_trailer(ngx_http_request_t *r, tpc_marker_trailer_t trailer)
275{
276 ngx_buf_t *b;
277 ngx_chain_t out;
278
279 b = ngx_create_temp_buf(r->pool, trailer.len);
280 if (b != NULL) {
281 ngx_memcpy(b->pos, trailer.line, trailer.len);
282 b->last = b->pos + trailer.len;
283 out.buf = b;
284 out.next = NULL;
285 (void) ngx_http_output_filter(r, &out);
286 }
287
288 (void) ngx_http_send_special(r, NGX_HTTP_LAST);
289 ngx_http_finalize_request(r, NGX_DONE);
290}

292/*
293 * tpc_marker_finish — WHAT: orchestrate transfer completion once the thread has
294 * signalled done. WHY: the last marker must report the final size, the pull temp
295 * must be committed or discarded, and the result recorded before the chunked body
296 * closes. HOW: a flat sequence — compute final bytes, emit progress + the final
297 * marker, settle the temp (may flip failed), record result counters, send the
298 * trailer. All logic lives in the named helpers.
299 */
300static void
301tpc_marker_finish(ngx_http_request_t *r, tpc_marker_ctx_t *ctx, int failed)
302{
303 time_t now = time(NULL);
304 off_t final_bytes = tpc_marker_final_bytes(ctx);
305 tpc_marker_trailer_t trailer;
306
307 (void) brix_tpc_progress_emit(ctx->transfer_id, final_bytes, final_bytes,
308 failed ? BRIX_TPC_STATE_ERROR
309 : BRIX_TPC_STATE_DONE,
310 r->connection->log);
311 tpc_marker_send_all(r, now, ctx);
312
313 failed = tpc_marker_settle_temp(r, ctx, failed);
314
315 trailer = tpc_marker_record_result(r, ctx, failed, final_bytes);
316 tpc_marker_send_trailer(r, trailer);
317}

320void
321tpc_marker_poll(ngx_event_t *ev)
322{
323 tpc_marker_ctx_t *ctx = ev->data;
324 ngx_http_request_t *r = ctx->r;
325 tpc_ms_progress_t *progress = ctx->progress;
326 time_t now;
327 int completed;
328
329 now = time(NULL);
330 completed = (int) progress->completed;
331
332 if (ctx->marker_interval_secs > 0
333 && (now - ctx->last_marker_time) >= (time_t) ctx->marker_interval_secs)
334 {
335 tpc_marker_send_all(r, now, ctx);
336 ctx->last_marker_time = now;
337 }
338
339 if (completed) {
340 int failed = (progress->result != NGX_OK
341 && progress->result != NGX_HTTP_CREATED
342 && progress->result != NGX_HTTP_NO_CONTENT);
343 tpc_marker_finish(r, ctx, failed);
344 return;
345 }
346
347 ngx_add_timer(ev, TPC_MARKER_POLL_MSEC);
348}

357{

361
362 if (ctx->timer.timer_set) {
363 ngx_del_timer(&ctx->timer);

368void
369tpc_marker_cleanup(void *data)
370{
371 tpc_marker_ctx_t *ctx = data;
372
373 /*
374 * Phase 39 (WS5): this pool cleanup fires when the request ends — including
375 * when the client disconnects mid-COPY. Signal the in-flight curl transfer
376 * (by id) to abort promptly via its progress callback, instead of leaving the
377 * thread-pool worker pulling into an abandoned destination until the
378 * low-speed/transfer-timeout bound or the registry reaper reclaims it. On a
379 * normal completion the transfer is already gone, so this is a no-op.
380 */
381 if (ctx->transfer_id != 0) {
382 (void) brix_tpc_registry_request_cancel(ctx->transfer_id);
383 }
384
385 if (ctx->timer.timer_set) {
386 ngx_del_timer(&ctx->timer);
387 }
388
389 if (ctx->is_pull && ctx->tmp_path[0] != '\0') {
390 (void) unlink(ctx->tmp_path); /* vfs-seam-allow: TPC in-progress transfer temp cleanup */
391 ctx->tmp_path[0] = '\0';
392 }
393}
