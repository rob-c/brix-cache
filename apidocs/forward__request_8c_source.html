4/*
5 * WHAT: Build and forward a client XRootD request to the upstream server,
6 * translating file handles (local → upstream), applying path rewriting
7 * when configured, capturing paths for audit logging, and saving copies
8 * for transparent kXR_wait retry.
9 *
10 * WHY: The transparent proxy must translate every file handle from the
11 * client's namespace to the upstream server's namespace so the backend
12 * can locate files using its own fh numbers. Path rewriting (strip prefix)
13 * allows nginx-xrootd to present a different root path than the backend.
14 * Audit logging captures which paths were mutated for compliance tracking.
15 * kXR_wait retry copies let the proxy silently re-issue requests when
16 * upstream servers return "busy, try later" without client involvement.
17 *
18 * HOW: brix_proxy_forward_request() allocates a request buffer, copies the
19 * client's header + payload, then dispatches through a reqid-based switch:
20 * kXR_open → pre-allocate fh slot + path strip + audit capture;
21 * read/write/close/sync/chkpoint → translate single fhandle + bound-secondary lazy-open check;
22 * stat/truncate/fattr → translate fhandle or capture path for audit;
23 * readv/writev → iterate payload segments translating each fhandle,
24 * bound-secondary collects ALL unresolved handles for sequential lazy-open;
25 * clone → translate src and dst handles;
26 * prepare → apply path strip rewrite;
27 * default (rm/rmdir/mkdir/chmod/mv) → path strip + audit capture.
28 * Finally saves a retry copy, queues to upstream write buffer, flushes,
29 * and arms the upstream read event loop.
30 */
31

41 uint16_t reqid = ctx->recv.cur_reqid;
42 /* body_len covers everything after the header: cur_dlen plus any
43 * streamed trailing body the recv framing appended (cur_body_extra) —
44 * kXR_writev segment data after the dlen-framed descriptor block, or the
45 * kXR_ckpXeq sub-request body after the dlen-framed embedded header.
46 * The header is forwarded verbatim (its dlen still frames only the
47 * dlen-counted part), so the upstream sees the exact stock wire layout. */
48 size_t body_len = (size_t) ctx->recv.cur_dlen + ctx->recv.cur_body_extra;
49 size_t total = XRD_REQUEST_HDR_LEN + body_len;

52 /* Allocate forwarded request buffer — freed by cleanup or after send */

56 "proxy: out of memory building request");

138 /* kXR_ckpXeq (opcode byte at header offset 19): the frame's payload
139 * is the embedded sub-request header (its streamed sub-body follows
140 * inside body_len, see chkpoint_xeq.c). The embedded request — and,
141 * for an embedded writev, every write_list descriptor — names the
142 * same local handle, so translate those too or the upstream will
143 * refuse the handle mismatch. */

146 uint16_t sub_reqid = (uint16_t) ((sub[2] << 8) | sub[3]);

207 * Payload is an array of readahead_list: { fhandle[4], rlen[4], offset[8] }
208 * Translate fhandle[0] in each entry.
209 *

211 * segments and open them sequentially before dispatching. xrdcp parallel
212 * streams may reference several handles in a single readv.
213 */

217 /* First pass: collect unique unresolved fhs */
218 int pending[BRIX_MAX_FILES];
219 int n_pending = 0;

221 int seen[BRIX_MAX_FILES];
222 int i;
223

225

227 while (pos + 16 <= ctx->recv.cur_dlen) {
228 int fh = (int)(unsigned char) payload[pos];
229 if (fh >= 0 && fh < BRIX_MAX_FILES

231 && !seen[fh])

233 seen[fh] = 1;
234 pending[n_pending++] = fh;
235 }
236 pos += 16;
237 }

240 /* Store the queue; lazy_open will consume entries one by one */
241 int first_fh = pending[0];

243 proxy->lazy_open_pending_fhs[i - 1] = pending[i];
244 }
245 proxy->lazy_open_pending_count = n_pending - 1;
246 return brix_proxy_lazy_open(proxy, ctx, c, first_fh, req, total);
247 }
248 }
249
250 {
251 size_t pos = 0;
252 while (pos + 16 <= ctx->recv.cur_dlen) {
253 if (proxy_translate_fh(proxy, payload + pos, 0) != 0) {
254 ngx_free(req);
255 return brix_send_error(ctx, c, kXR_InvalidRequest,
256 "proxy: invalid fhandle in readv");
257 }
258 pos += 16;
259 }
260 }
261 proxy->fwd_local_fh = -1;
262 break;
263 }

267 u_char *payload = req + XRD_REQUEST_HDR_LEN;

292 }

312 XRD_REQUEST_HDR_LEN, ctx->recv.cur_dlen,

316 if (proxy->conf != NULL

322 case kXR_rmdir:
323 case kXR_mkdir:
324 case kXR_chmod: {

384 proxy->wait_retry_count = 0;
385 }

388 * to the observation tap before it goes upstream. */

394 }

396 /* Queue the request to the upstream write buffer */
397 proxy->wbuf = req;
398 proxy->wbuf_len = total;
399 proxy->wbuf_pos = 0;
400 proxy->wbuf_owned = 1; /* Phase 39 (PXY-3): raw heap — free on send-complete */
401 proxy->state = XRD_PX_FORWARDING;
402
403 /* Reset response accumulator */
404 proxy->rhdr_pos = 0;
405 proxy->resp_dlen = 0;
406 proxy->resp_body = NULL;
407 proxy->resp_body_pos = 0;
408
409 /* Suspend client read loop while we wait for the upstream response */
410 ctx->state = XRD_ST_PROXY;
411
412 ngx_int_t rc = brix_proxy_flush(proxy);
413 if (rc == NGX_ERROR) {
414 brix_proxy_wbuf_release(proxy);
415 brix_proxy_abort(proxy, "proxy: upstream write error");
416 return NGX_ERROR;
417 }
418
419 if (proxy->wbuf_pos == proxy->wbuf_len) {
420 /* Fully sent immediately — free the request buffer and arm upstream read */
421 brix_proxy_wbuf_release(proxy);
422 if (ngx_handle_read_event(proxy->conn->read, 0) != NGX_OK) {
423 brix_proxy_abort(proxy, "proxy: read arm failed");
424 return NGX_ERROR;
425 }
426 } else if (proxy->conf != NULL && proxy->conf->proxy.write_timeout > 0) {
427 /* Phase 39 (PXY-2): the request is parked for the upstream write event —
428 * bound how long a slow/backpressured upstream may stall it (the write
429 * handler refreshes/clears the timer). Armed here too in case the write
430 * event never fires. */
431 ngx_add_timer(proxy->conn->write, proxy->conf->proxy.write_timeout);
432 }
433 /* else: write handler will complete the send and arm the read */
434
435 return NGX_OK;
436}
