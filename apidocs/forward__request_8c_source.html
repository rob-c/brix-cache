4/*
5 * WHAT: Build and forward a client XRootD request to the upstream server,
6 * translating file handles (local → upstream), applying path rewriting
7 * when configured, capturing paths for audit logging, and saving copies
8 * for transparent kXR_wait retry.
9 *
10 * WHY: The transparent proxy must translate every file handle from the
11 * client's namespace to the upstream server's namespace so the backend
12 * can locate files using its own fh numbers. Path rewriting (strip prefix)
13 * allows nginx-xrootd to present a different root path than the backend.
14 * Audit logging captures which paths were mutated for compliance tracking.
15 * kXR_wait retry copies let the proxy silently re-issue requests when
16 * upstream servers return "busy, try later" without client involvement.
17 *
18 * HOW: brix_proxy_forward_request() allocates a request buffer, copies the
19 * client's header + payload, then dispatches through a reqid-based switch:
20 * kXR_open → pre-allocate fh slot + path strip + audit capture;
21 * read/write/close/sync/chkpoint → translate single fhandle + bound-secondary lazy-open check;
22 * stat/truncate/fattr → translate fhandle or capture path for audit;
23 * readv/writev → iterate payload segments translating each fhandle,
24 * bound-secondary collects ALL unresolved handles for sequential lazy-open;
25 * clone → translate src and dst handles;
26 * prepare → apply path strip rewrite;
27 * default (rm/rmdir/mkdir/chmod/mv) → path strip + audit capture.
28 * Finally saves a retry copy, queues to upstream write buffer, flushes,
29 * and arms the upstream read event loop.
30 */
31
32/*
33 * WHAT: Handle kXR_open forwarding — pre-allocate a local fh slot, apply path
34 * rewriting, and capture the path for audit logging.
35 * WHY: The fh slot must be reserved before any error-free path so the open
36 * response can bind to it; path rewrite must precede audit capture.
37 * HOW: Allocates a local fh (255 sentinel = awaiting open response), rewrites
38 * the payload path if configured, then copies the (post-rewrite) path
39 * into the slot's audit buffer. Returns NGX_OK, or NGX_ABORT/NGX_ERROR
40 * via proxy_reject_request() (req freed, error sent to the client).
41 * *req and *total are updated in place.
42 */
43static ngx_int_t
44brix_proxy_forward_open(brix_proxy_ctx_t *proxy, brix_ctx_t *ctx,
45 ngx_connection_t *c, u_char **req, size_t *total)
46{
47 /* Pre-allocate a local file handle for when the response arrives */
48 int local_fh = brix_proxy_alloc_local_fh(proxy);
49 if (local_fh < 0) {
50 return proxy_reject_request(ctx, c, *req, kXR_IOError,
51 "proxy: no free file handles");
52 }
53 /* Mark slot as pending (non-free but no upstream handle yet) */
54 proxy->fh_map[local_fh].upstream_fh = 255; /* sentinel: awaiting open response */
55 proxy->fwd_local_fh = local_fh;
56
57 /* Apply path rewriting before capturing path for audit */
58 if (proxy->conf != NULL && proxy->conf->proxy.path_strip.len > 0
59 && ctx->recv.cur_dlen > 0)
60 {
61 *req = proxy_rewrite_path(c, proxy->conf, *req, *total,
62 XRD_REQUEST_HDR_LEN, ctx->recv.cur_dlen,
63 total);
64 /* Recalculate cur_dlen for audit capture below */
65 }
66
67 /* Capture path from payload for audit log */
68 if (proxy->conf != NULL
69 && proxy->conf->proxy.audit_log_fd != NGX_INVALID_FILE)
70 {
71 const u_char *path_start = *req + XRD_REQUEST_HDR_LEN;
72 size_t path_len = *total > XRD_REQUEST_HDR_LEN
73 ? *total - XRD_REQUEST_HDR_LEN : 0;
74 size_t plen = path_len < BRIX_PROXY_PATH_MAX - 1
75 ? path_len : BRIX_PROXY_PATH_MAX - 1;
76 ngx_memcpy(proxy->fh_map[local_fh].path, path_start, plen);
77 proxy->fh_map[local_fh].path[plen] = '\0';
78 }
79 return NGX_OK;
80}

82/*
83 * WHAT: Capture path/dest audit fields for a default path-based mutation op.
84 * WHY: Split out of brix_proxy_forward_pathop() so the rewrite path and the
85 * per-reqid audit-buffer copies each stay small and single-purpose.
86 * HOW: For rm/rmdir/mkdir/chmod copies the (post-rewrite) payload path; for
87 * kXR_mv splits src[arg1len] + ' ' + dst into fwd_path / fwd_path2.
88 */
89static void
90brix_proxy_pathop_audit(brix_proxy_ctx_t *proxy, uint16_t reqid,
91 u_char *req, size_t total)
92{
93 switch (reqid) {
94 case kXR_rm:
95 case kXR_rmdir:
96 case kXR_mkdir:
97 case kXR_chmod: {
98 size_t plen = (total - XRD_REQUEST_HDR_LEN) < BRIX_PROXY_PATH_MAX
99 ? total - XRD_REQUEST_HDR_LEN
100 : BRIX_PROXY_PATH_MAX - 1;
101 ngx_memcpy(proxy->fwd_path, req + XRD_REQUEST_HDR_LEN, plen);
102 proxy->fwd_path[plen] = '\0';
103 proxy->fwd_path2[0] = '\0';
104 proxy->fwd_path_audit = 1;
105 break;
106 }
107 case kXR_mv: {
108 /* payload: src[arg1len] + ' ' + dst */
109 uint16_t arg1len_be;
110 size_t src_len, dst_off, dst_len;
111 ngx_memcpy(&arg1len_be, req + 18, 2);
112 src_len = (size_t) ntohs(arg1len_be);
113 if (src_len >= BRIX_PROXY_PATH_MAX) {
114 src_len = BRIX_PROXY_PATH_MAX - 1;
115 }
116 ngx_memcpy(proxy->fwd_path, req + XRD_REQUEST_HDR_LEN, src_len);
117 proxy->fwd_path[src_len] = '\0';
118 dst_off = XRD_REQUEST_HDR_LEN + src_len + 1; /* skip space */
119 if (dst_off < total) {
120 dst_len = total - dst_off;
121 if (dst_len >= BRIX_PROXY_PATH_MAX) {
122 dst_len = BRIX_PROXY_PATH_MAX - 1;
123 }
124 ngx_memcpy(proxy->fwd_path2, req + dst_off, dst_len);
125 proxy->fwd_path2[dst_len] = '\0';
126 } else {
127 proxy->fwd_path2[0] = '\0';
128 }
129 proxy->fwd_path_audit = 1;
130 break;
131 }
132 default:
133 break;
134 }
135}

170/*
171 * WHAT: Save a copy of the fully-assembled request for transparent kXR_wait retry.
172 * WHY: If upstream returns "busy, try later", the proxy silently re-issues the
173 * exact same request without client involvement — but only for reasonably
174 * sized payloads (huge writes are not buffered for retry).
175 * HOW: For total < 128KB, frees any prior retry buffer (avoids leak/double-free),
176 * allocates and copies the request, and records length/fh/count.
177 */
178static void
179brix_proxy_save_wait_retry(brix_proxy_ctx_t *proxy, ngx_connection_t *c,
180 u_char *req, size_t total)
181{
182 /* Save a copy for transparent kXR_wait retry (if payload is not huge) */
183 if (total < 128 * 1024) {
184 if (proxy->wait_retry_req != NULL) {
185 ngx_free(proxy->wait_retry_req);
186 }
187 proxy->wait_retry_req = ngx_alloc(total, c->log);
188 if (proxy->wait_retry_req != NULL) {
189 ngx_memcpy(proxy->wait_retry_req, req, total);
190 proxy->wait_retry_req_len = total;
191 } else {
192 proxy->wait_retry_req_len = 0;
193 }
194 proxy->wait_retry_local_fh = proxy->fwd_local_fh;
195 proxy->wait_retry_count = 0;
196 }

199/*
200 * WHAT: Queue the request to the upstream write buffer, flush it, and arm the
201 * upstream read (or a write timeout if the send is parked).
202 * WHY: Completes forwarding — hands ownership of req to the write path and
203 * transitions the proxy/session state machines to await the response.
204 * HOW: Sets wbuf state, resets the response accumulator, suspends the client
205 * read loop, flushes; on full send arms the read event, otherwise arms a
206 * write timeout. Returns NGX_OK or NGX_ERROR (after abort).
207 */
208static ngx_int_t
209brix_proxy_queue_and_flush(brix_proxy_ctx_t *proxy, brix_ctx_t *ctx,
210 u_char *req, size_t total)
211{
212 /* Queue the request to the upstream write buffer */
213 proxy->wbuf = req;
214 proxy->wbuf_len = total;
215 proxy->wbuf_pos = 0;
216 proxy->wbuf_owned = 1; /* Phase 39 (PXY-3): raw heap — free on send-complete */
217 proxy->state = XRD_PX_FORWARDING;
218
219 /* Reset response accumulator */
220 proxy->rhdr_pos = 0;
221 proxy->resp_dlen = 0;
222 proxy->resp_body = NULL;
223 proxy->resp_body_pos = 0;
224
225 /* Suspend client read loop while we wait for the upstream response */
226 ctx->state = XRD_ST_PROXY;
227
228 ngx_int_t rc = brix_proxy_flush(proxy);
229 if (rc == NGX_ERROR) {
230 brix_proxy_wbuf_release(proxy);
231 brix_proxy_abort(proxy, "proxy: upstream write error");
232 return NGX_ERROR;
233 }
234
235 if (proxy->wbuf_pos == proxy->wbuf_len) {
236 /* Fully sent immediately — free the request buffer and arm upstream read */
237 brix_proxy_wbuf_release(proxy);
238 if (ngx_handle_read_event(proxy->conn->read, 0) != NGX_OK) {
239 brix_proxy_abort(proxy, "proxy: read arm failed");
240 return NGX_ERROR;
241 }
242 } else if (proxy->conf != NULL && proxy->conf->proxy.write_timeout > 0) {
243 /* Phase 39 (PXY-2): the request is parked for the upstream write event —
244 * bound how long a slow/backpressured upstream may stall it (the write
245 * handler refreshes/clears the timer). Armed here too in case the write
246 * event never fires. */
247 ngx_add_timer(proxy->conn->write, proxy->conf->proxy.write_timeout);
248 }
249 /* else: write handler will complete the send and arm the read */
250
251 return NGX_OK;
252}

379/*
380 * WHAT: Dispatch a forwarded request through its reqid-specific fh-translation /
381 * path-rewrite / audit-capture handler.
382 * WHY: Isolates the per-opcode switch from the surrounding build/queue logic so
383 * both stay under the readability gate; behavior is byte-identical to the
384 * inline switch it replaces.
385 * HOW: Tries the fh-op dispatcher first; unowned reqids fall through to prepare
386 * or the default path-op handler. Returns NGX_OK to continue, NGX_DONE when
387 * a lazy-open transferred req ownership (caller returns NGX_OK), NGX_ABORT
388 * when a kXR_error was queued to the client (req freed — caller must stop
389 * forwarding and return NGX_OK), or NGX_ERROR on a hard failure (req freed).
390 * *req and *total are updated in place by path-rewriting handlers.
391 */
392static ngx_int_t
393brix_proxy_translate_dispatch(brix_proxy_ctx_t *proxy, brix_ctx_t *ctx,
394 ngx_connection_t *c, uint16_t reqid,
395 u_char **req, size_t *total)
396{
397 /* file handle translation */
398 ngx_int_t rc = brix_proxy_translate_fh_op(proxy, ctx, c, reqid, req, total);
399 if (rc != NGX_DECLINED) {
400 return rc;
401 }
402

404 brix_proxy_forward_prepare(proxy, ctx, c, req, total);
405 return NGX_OK;
406 }

412/* public API: brix_proxy_forward_request() — build and send forwarded request * WHAT: Forward a client request to the upstream server with file handle translation,
413 * path rewriting, audit capture, and kXR_wait retry support. Returns NGX_OK or NGX_ERROR. */
414
415/* build and send the forwarded request */
416ngx_int_t
417brix_proxy_forward_request(brix_proxy_ctx_t *proxy,
418 brix_ctx_t *ctx,
419 ngx_connection_t *c)
420{
421 uint16_t reqid = ctx->recv.cur_reqid;
422 /* body_len covers everything after the header: cur_dlen plus any
423 * streamed trailing body the recv framing appended (cur_body_extra) —
424 * kXR_writev segment data after the dlen-framed descriptor block, or the
425 * kXR_ckpXeq sub-request body after the dlen-framed embedded header.
426 * The header is forwarded verbatim (its dlen still frames only the
427 * dlen-counted part), so the upstream sees the exact stock wire layout. */
428 size_t body_len = (size_t) ctx->recv.cur_dlen + ctx->recv.cur_body_extra;
429 size_t total = XRD_REQUEST_HDR_LEN + body_len;
430 u_char *req;
431
432 /* Allocate forwarded request buffer — freed by cleanup or after send */
433 req = ngx_alloc(total, c->log);
434 if (req == NULL) {
435 return brix_send_error(ctx, c, kXR_IOError,
436 "proxy: out of memory building request");
437 }
438
439 /* Copy full request header then payload (descriptors + any trailing data) */
440 ngx_memcpy(req, ctx->recv.hdr_buf, XRD_REQUEST_HDR_LEN);
441 if (body_len > 0 && ctx->recv.payload != NULL) {
442 ngx_memcpy(req + XRD_REQUEST_HDR_LEN, ctx->recv.payload, body_len);
443 }
444
445 /* Pre-set forwarding metadata */
446 proxy->fwd_reqid = reqid;
447 proxy->fwd_streamid[0] = ctx->recv.cur_streamid[0];
448 proxy->fwd_streamid[1] = ctx->recv.cur_streamid[1];
449 proxy->fwd_streaming = 0;
450 proxy->fwd_payload_len = ctx->recv.cur_dlen;
451
452 /* file handle translation + path rewrite + audit capture, per reqid */
453 ngx_int_t drc = brix_proxy_translate_dispatch(proxy, ctx, c, reqid,
454 &req, &total);
455 if (drc == NGX_DONE) {
456 return NGX_OK; /* lazy-open initiated; req ownership transferred */
457 }
458 if (drc == NGX_ABORT) {
459 /* Request rejected: req has been freed and a kXR_error queued to the
460 * client — the session continues, but req must NOT be touched again
461 * (save_wait_retry/tap/queue below would be a use-after-free). */
462 return NGX_OK;
463 }
464 if (drc != NGX_OK) {
465 return drc; /* hard failure; req freed */
466 }
467
468 /* fwd_payload_len may have changed if path was rewritten (total updated) */
469 proxy->fwd_payload_len = total > XRD_REQUEST_HDR_LEN
470 ? total - XRD_REQUEST_HDR_LEN : 0;
471
472 brix_proxy_save_wait_retry(proxy, c, req, total);
473
474 /* Tap: decode the fully-assembled (post-rewrite) request frame and emit it
475 * to the observation tap before it goes upstream. */
476 {
477 brix_tap_frame_t tf;
478 if (brix_tap_decode_request(req, total, &tf) > 0) {
479 brix_tap_emit(&proxy->tap, &tf, BRIX_TAP_C2U, NULL, 0);
480 }
481 }
482
483 return brix_proxy_queue_and_flush(proxy, ctx, req, total);
484}
