32static void
33brix_gsi_copy_parm(char *dst, size_t dst_cap, const char *value,
34 size_t value_len)
35{
36 size_t copy_len;
37
38 if (dst == NULL || dst_cap == 0) {
39 return;
40 }
41
42 copy_len = value_len < dst_cap - 1 ? value_len : dst_cap - 1;
43 memcpy(dst, value, copy_len);
44 dst[copy_len] = '\0';
45}

47static void
48brix_gsi_parse_short_parm(const char *field, size_t field_len,
49 uint32_t *version, char *crypto, size_t cryptosz)
50{
51 const char *value;
52 size_t value_len;
53
54 if (field_len <= 2 || field[1] != ':') {
55 return;
56 }
57
58 value = field + 2;
59 value_len = field_len - 2;
60 if (field[0] == 'v' && version != NULL) {
61 *version = (uint32_t) strtoul(value, NULL, 10);
62 return;
63 }
64 if (field[0] == 'c') {
65 brix_gsi_copy_parm(crypto, cryptosz, value, value_len);
66 }
67}

82/* Parse a gsi protocol parms string "v:10600,c:ssl,ca:HASH|HASH" into fields.
83 * Any out pointer may be NULL. `crypto`/`ca` are NUL-terminated, truncated. */
84void
85brix_gsi_parse_parms(const char *parms, uint32_t *version,
86 char *crypto, size_t cryptosz,
87 char *ca, size_t casz)
88{
89 const char *p;
90
91 if (version != NULL) { *version = 0; }
92 if (crypto != NULL && cryptosz > 0) { crypto[0] = '\0'; }
93 if (ca != NULL && casz > 0) { ca[0] = '\0'; }
94 if (parms == NULL) {
95 return;
96 }
97
98 for (p = parms; *p != '\0'; ) {
99 const char *comma = strchr(p, ',');
100 size_t flen = comma ? (size_t) (comma - p) : strlen(p);
101
102 brix_gsi_parse_short_parm(p, flen, version, crypto, cryptosz);
103 brix_gsi_parse_ca_parm(p, flen, ca, casz);
104 p += flen;
105 if (*p == ',') { p++; }
106 }
107}

110/*
111 * Build the standard XrdSecgsi round-1 certreq buffer. Outer (Step
112 * kXGC_certreq) carries kXRS_cryptomod, kXRS_version, kXRS_issuer_hash,
113 * kXRS_clnt_opts and a kXRS_main bucket whose data is a *nested* buffer
114 * (Step kXGC_certreq + kXRS_rtag + kXRS_none). Mirrors a stock client's first
115 * message; the server's XrdSutBuffer parser reads the certreq opcode from the
116 * main bucket (a bare top-level opcode is rejected with "main buffer missing").
117 * Returns a malloc'd buffer (*outlen set), or NULL.
118 */
119uint8_t *
120brix_gsi_build_certreq(const char *cryptomod, uint32_t version,
121 const char *issuer_hash, uint32_t clnt_opts,
122 const uint8_t *rtag, size_t rtaglen, size_t *outlen)
123{
124 brix_gbuf inner, outer;
125 uint32_t ver_be = htonl(version);
126 uint32_t opts_be = htonl(clnt_opts);
127 uint8_t *result = NULL;
128
129 /* Nested main: "gsi\0" + kXGC_certreq + rtag bucket + terminator. */
130 brix_gbuf_init(&inner);
131 brix_gbuf_start(&inner, (uint32_t) kXGC_certreq);
132 brix_gbuf_bucket(&inner, (uint32_t) kXRS_rtag, rtag, rtaglen);
133 brix_gbuf_end(&inner);
134 if (inner.err) {
135 brix_gbuf_free(&inner);
136 return NULL;
137 }
138
139 brix_gbuf_init(&outer);
140 brix_gbuf_start(&outer, (uint32_t) kXGC_certreq);
141 brix_gbuf_bucket(&outer, (uint32_t) kXRS_cryptomod,
142 cryptomod, strlen(cryptomod));
143 brix_gbuf_bucket(&outer, (uint32_t) kXRS_version, &ver_be, 4);
144 brix_gbuf_bucket(&outer, (uint32_t) kXRS_issuer_hash,
145 issuer_hash, strlen(issuer_hash));
146 brix_gbuf_bucket(&outer, (uint32_t) kXRS_clnt_opts, &opts_be, 4);
147 brix_gbuf_bucket(&outer, (uint32_t) kXRS_main, inner.p, inner.len);
148 brix_gbuf_end(&outer);
149
150 if (!outer.err) {
151 result = outer.p;
152 *outlen = outer.len;
153 outer.p = NULL; /* ownership → caller */
154 }
155 brix_gbuf_free(&inner);
156 brix_gbuf_free(&outer);
157 return result;
158}

161int
162brix_gsi_sigver_required(uint16_t op, int level)
163{
164 static const uint16_t exempt_ops[] = {
165 kXR_login, kXR_protocol, kXR_auth, kXR_endsess,
166 kXR_ping, kXR_sigver, kXR_bind
167 };
168 static const uint16_t level2_ops[] = {
169 kXR_open, kXR_write, kXR_pgwrite, kXR_writev, kXR_truncate,
170 kXR_mkdir, kXR_rm, kXR_rmdir, kXR_mv, kXR_chmod, kXR_fattr,
171 kXR_chkpoint, kXR_clone
172 };
173 size_t op_index;
174
175 if (level <= 1) {
176 return 0;
177 }
178
179 for (op_index = 0; op_index < sizeof(exempt_ops) / sizeof(exempt_ops[0]);
180 op_index++) {
181 if (op == exempt_ops[op_index]) {
182 return 0;
183 }
184 }
185
186 if (level == 2) {
187 for (op_index = 0; op_index < sizeof(level2_ops) / sizeof(level2_ops[0]);
188 op_index++) {
189 if (op == level2_ops[op_index]) {
190 return 1;
191 }
192 }
193 return 0;
194 }
195 return 1; /* level >= 3 */
196}

201void
202brix_gsi_sigver_seqno_be(uint64_t seq, uint8_t out[8])
203{
204 out[0] = (uint8_t) (seq >> 56);
205 out[1] = (uint8_t) (seq >> 48);
206 out[2] = (uint8_t) (seq >> 40);
207 out[3] = (uint8_t) (seq >> 32);
208 out[4] = (uint8_t) (seq >> 24);
209 out[5] = (uint8_t) (seq >> 16);
210 out[6] = (uint8_t) (seq >> 8);
211 out[7] = (uint8_t) seq;
212}

215int
216brix_gsi_sigver_hmac(const uint8_t key[32], uint64_t seqno,
217 const uint8_t hdr24[24], const uint8_t *payload,
218 size_t plen, int nodata, uint8_t mac_out[32])
219{
220 uint8_t seqbe[8];
221 uint8_t *msg;
222 size_t mlen;
223 int cover_payload, ok;
224
225 if (key == NULL || hdr24 == NULL || mac_out == NULL) {
226 return 0;
227 }
228 cover_payload = (!nodata && payload != NULL && plen > 0);
229 mlen = 8 + 24 + (cover_payload ? plen : 0);
230 msg = (uint8_t *) malloc(mlen);
231 if (msg == NULL) {
232 return 0;
233 }
234 brix_gsi_sigver_seqno_be(seqno, seqbe);
235 memcpy(msg, seqbe, 8);
236 memcpy(msg + 8, hdr24, 24);
237 if (cover_payload) {
238 memcpy(msg + 32, payload, plen);
239 }
240 ok = brix_hmac_sha256(key, 32, msg, mlen, mac_out);
241 free(msg);
242 return ok;
243}
