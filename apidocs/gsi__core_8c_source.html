26/* Parse a gsi protocol parms string "v:10600,c:ssl,ca:HASH|HASH" into fields.
27 * Any out pointer may be NULL. `crypto`/`ca` are NUL-terminated, truncated. */
28void
29brix_gsi_parse_parms(const char *parms, uint32_t *version,
30 char *crypto, size_t cryptosz,
31 char *ca, size_t casz)
32{
33 const char *p;
34
35 if (version != NULL) { *version = 0; }
36 if (crypto != NULL && cryptosz > 0) { crypto[0] = '\0'; }
37 if (ca != NULL && casz > 0) { ca[0] = '\0'; }
38 if (parms == NULL) {
39 return;
40 }
41
42 for (p = parms; *p != '\0'; ) {
43 const char *comma = strchr(p, ',');
44 size_t flen = comma ? (size_t) (comma - p) : strlen(p);
45
46 if (flen > 2 && p[1] == ':') {
47 const char *val = p + 2;
48 size_t vlen = flen - 2;
49 if (p[0] == 'v' && version != NULL) {
50 *version = (uint32_t) strtoul(val, NULL, 10);
51 } else if (p[0] == 'c' && crypto != NULL && cryptosz > 0) {
52 size_t n = vlen < cryptosz - 1 ? vlen : cryptosz - 1;
53 memcpy(crypto, val, n);
54 crypto[n] = '\0';
55 }
56 } else if (flen > 3 && p[0] == 'c' && p[1] == 'a' && p[2] == ':'
57 && ca != NULL && casz > 0) {
58 const char *val = p + 3;
59 size_t vlen = flen - 3;
60 size_t n = vlen < casz - 1 ? vlen : casz - 1;
61 memcpy(ca, val, n);
62 ca[n] = '\0';
63 }
64 p += flen;
65 if (*p == ',') { p++; }
66 }
67}

70/*
71 * Build the standard XrdSecgsi round-1 certreq buffer. Outer (Step
72 * kXGC_certreq) carries kXRS_cryptomod, kXRS_version, kXRS_issuer_hash,
73 * kXRS_clnt_opts and a kXRS_main bucket whose data is a *nested* buffer
74 * (Step kXGC_certreq + kXRS_rtag + kXRS_none). Mirrors a stock client's first
75 * message; the server's XrdSutBuffer parser reads the certreq opcode from the
76 * main bucket (a bare top-level opcode is rejected with "main buffer missing").
77 * Returns a malloc'd buffer (*outlen set), or NULL.
78 */
79uint8_t *
80brix_gsi_build_certreq(const char *cryptomod, uint32_t version,
81 const char *issuer_hash, uint32_t clnt_opts,
82 const uint8_t *rtag, size_t rtaglen, size_t *outlen)
83{
84 brix_gbuf inner, outer;
85 uint32_t ver_be = htonl(version);
86 uint32_t opts_be = htonl(clnt_opts);
87 uint8_t *result = NULL;
88
89 /* Nested main: "gsi\0" + kXGC_certreq + rtag bucket + terminator. */
90 brix_gbuf_init(&inner);
91 brix_gbuf_start(&inner, (uint32_t) kXGC_certreq);
92 brix_gbuf_bucket(&inner, (uint32_t) kXRS_rtag, rtag, rtaglen);
93 brix_gbuf_end(&inner);
94 if (inner.err) {
95 brix_gbuf_free(&inner);
96 return NULL;
97 }
98
99 brix_gbuf_init(&outer);
100 brix_gbuf_start(&outer, (uint32_t) kXGC_certreq);
101 brix_gbuf_bucket(&outer, (uint32_t) kXRS_cryptomod,
102 cryptomod, strlen(cryptomod));
103 brix_gbuf_bucket(&outer, (uint32_t) kXRS_version, &ver_be, 4);
104 brix_gbuf_bucket(&outer, (uint32_t) kXRS_issuer_hash,
105 issuer_hash, strlen(issuer_hash));
106 brix_gbuf_bucket(&outer, (uint32_t) kXRS_clnt_opts, &opts_be, 4);
107 brix_gbuf_bucket(&outer, (uint32_t) kXRS_main, inner.p, inner.len);
108 brix_gbuf_end(&outer);
109
110 if (!outer.err) {
111 result = outer.p;
112 *outlen = outer.len;
113 outer.p = NULL; /* ownership → caller */
114 }
115 brix_gbuf_free(&inner);
116 brix_gbuf_free(&outer);
117 return result;
118}

136/* Export an EVP_PKEY public part as PEM SubjectPublicKeyInfo
137 * (XrdCryptosslRSA::ExportPublic = PEM_write_bio_PUBKEY). malloc'd,
138 * NUL-terminated, *outlen = strlen. The signed-DH path sends this as kXRS_puk so
139 * the server can verify our signed DH before it has decrypted our cert chain. */
140char *
141gsi_cresp_export_pubkey_pem(EVP_PKEY *key, size_t *outlen)
142{
143 BIO *bio = BIO_new(BIO_s_mem());
144 char *pem = NULL, *out = NULL;
145 long len;
146
147 if (bio != NULL && PEM_write_bio_PUBKEY(bio, key) == 1) {
148 len = BIO_get_mem_data(bio, &pem);
149 out = malloc((size_t) len + 1);
150 if (out != NULL) {
151 memcpy(out, pem, (size_t) len);
152 out[len] = '\0';
153 *outlen = (size_t) len;
154 }
155 }
156 BIO_free(bio);
157 return out;
158}

161/* Choose the kXRS_md_alg digest to echo back: prefer "sha256" when the server
162 * offers it (it matches our SHA256(secret) key derivation), else the server's
163 * first ':'-separated token, else "sha256". dCache rejects the handshake if the
164 * reply names a digest it did not advertise. */
165size_t
166gsi_cresp_pick_md_alg(const uint8_t *sbody, uint32_t slen, char *out, size_t outcap)
167{
168 const uint8_t *list = NULL;
169 size_t listlen = 0, n = 0;
170
171 if (brix_gsi_find_bucket(sbody, slen, (uint32_t) kXRS_md_alg,
172 &list, &listlen) == 0 && listlen > 0) {
173 if (listlen >= 6) {
174 for (size_t i = 0; i + 6 <= listlen; i++) {
175 if (memcmp(list + i, "sha256", 6) == 0) {
176 memcpy(out, "sha256", 6);
177 out[6] = '\0';
178 return 6;
179 }
180 }
181 }
182 while (n < listlen && n + 1 < outcap && list[n] != ':') {
183 out[n] = (char) list[n];
184 n++;
185 }
186 }
187 if (n == 0) {
188 memcpy(out, "sha256", 6);
189 n = 6;
190 }
191 out[n] = '\0';
192 return n;
193}

223int
224brix_gsi_build_cert_response_ex(const uint8_t *sbody, uint32_t slen,
225 const uint8_t *proxy_pem, size_t proxy_pem_len,
226 EVP_PKEY *proxy_key,
227 uint8_t **payload, uint32_t *plen,
228 uint8_t *out_key, size_t *out_keylen,
229 char *out_cipher, size_t out_cipher_cap,
230 int *out_use_iv,
231 char *err, size_t errcap)
232{
233 const uint8_t *cipher = NULL, *puk = NULL, *sx509 = NULL, *xmain = NULL;
234 size_t cipherlen = 0, puklen = 0, sx509len = 0;
235 size_t xmainlen = 0, enclen = 0, cpublen = 0;
236 const uint8_t *peerpub = NULL;
237 size_t peerpublen = 0;
238 int signed_dh;
239 uint8_t aeskey[BRIX_GSI_MAX_KEY];
240 brix_gsi_cipher_t sesscipher;
241 char chosen_cipher[24];
242 char cipher_field[40];
243 int use_iv;
244 uint8_t newrtag[8];
245 char md_alg[32];
246 size_t md_alg_len;
247 gsi_cresp_ctx x;
248
249 memset(&x, 0, sizeof(x));
250 brix_gbuf_init(&x.inner);
251 brix_gbuf_init(&x.outer);
252
253 if (proxy_pem == NULL || proxy_key == NULL) {
254 return gsi_cresp_fail(&x, err, errcap, "gsi: missing proxy credential");
255 }
256
257 /* 1. Determine the DH variant + recover the server's DH public blob. */
258 signed_dh = brix_gsi_find_bucket(sbody, slen, (uint32_t) kXRS_cipher,
259 &cipher, &cipherlen) == 0;
260 if (signed_dh) {
261 if (brix_gsi_find_bucket(sbody, slen, (uint32_t) kXRS_x509,
262 &sx509, &sx509len) != 0
263 || (x.servpub = gsi_cresp_cert_pubkey(sx509, sx509len)) == NULL) {
264 return gsi_cresp_fail(&x, err, errcap,
265 "gsi: server certificate missing");
266 }
267 x.peerblob = malloc(cipherlen + 64);
268 if (x.peerblob == NULL) {
269 return gsi_cresp_fail(&x, err, errcap, "gsi: out of memory");
270 }
271 peerpublen = brix_gsi_rsa_decrypt_public(x.servpub, cipher, cipherlen,
272 x.peerblob, cipherlen + 64);
273 if (peerpublen == 0) {
274 return gsi_cresp_fail(&x, err, errcap,
275 "gsi: verifying signed server DH parameters");
276 }
277 peerpub = x.peerblob;
278 } else if (brix_gsi_find_bucket(sbody, slen, (uint32_t) kXRS_puk,
279 &puk, &puklen) == 0) {
280 peerpub = puk;
281 peerpublen = puklen;
282 } else {
283 return gsi_cresp_fail(&x, err, errcap, "gsi: server DH public missing");
284 }
285
286 /* Choose the session cipher from the server's kXRS_cipher_alg list, preferring
287 * aes-128-cbc (the universal XrdSecgsi default) whenever offered. */
288 {
289 const uint8_t *ca = NULL; size_t cal = 0;
290 char offered[128];
291
292 snprintf(chosen_cipher, sizeof(chosen_cipher), "aes-128-cbc");
293 if (brix_gsi_find_bucket(sbody, slen, (uint32_t) kXRS_cipher_alg,
294 &ca, &cal) == 0 && cal > 0
295 && cal < sizeof(offered)) {
296 memcpy(offered, ca, cal);
297 offered[cal] = '\0';
298 if (memmem(offered, cal, "aes-128-cbc", 11) == NULL) {
299 (void) brix_gsi_cipher_pick(offered, &sesscipher, chosen_cipher);
300 }
301 }
302 if (!brix_gsi_cipher_lookup(chosen_cipher, &sesscipher)) {
303 (void) brix_gsi_cipher_lookup("aes-128-cbc", &sesscipher);
304 snprintf(chosen_cipher, sizeof(chosen_cipher), "aes-128-cbc");
305 }
306 }
307
308 /* 2. Agree the session key (HasPad follows the variant). */
309 {
310 const uint8_t *cm = NULL; size_t cml = 0;
311 int peer_nopad = (brix_gsi_find_bucket(sbody, slen,
312 (uint32_t) kXRS_cryptomod, &cm, &cml) == 0
313 && cml >= 5
314 && memmem(cm, cml, "nopad", 5) != NULL);
315 int dh_pad = signed_dh && !peer_nopad;
316
317 x.peer = brix_gsi_cipher_parse_peer(peerpub, peerpublen);
318 x.mine = x.peer ? brix_gsi_cipher_keygen_from(x.peer) : NULL;
319 if (x.peer == NULL || x.mine == NULL
320 || !brix_gsi_cipher_session_key(x.mine, x.peer, dh_pad, aeskey,
321 sesscipher.key_len)) {
322 return gsi_cresp_fail(&x, err, errcap,
323 "gsi: session-key agreement failed");
324 }
325 }
326
327 /* 3. Sign the server's random tag with the proxy key (proof of possession). */
328 if (brix_gsi_find_bucket(sbody, slen, (uint32_t) kXRS_main,
329 &xmain, &xmainlen) == 0) {
330 const uint8_t *srtag = NULL;
331 size_t srtaglen = 0;
332
333 if (brix_gsi_find_bucket(xmain, xmainlen, (uint32_t) kXRS_rtag,
334 &srtag, &srtaglen) == 0) {
335 uint8_t sig[1024];
336 size_t siglen = brix_gsi_rsa_sign_raw(proxy_key, srtag, srtaglen,
337 sig);
338 if (siglen == 0 || siglen > sizeof(sig)
339 || (x.signed_rtag = malloc(siglen)) == NULL) {
340 OPENSSL_cleanse(aeskey, sizeof(aeskey));
341 return gsi_cresp_fail(&x, err, errcap,
342 "gsi: signing the server tag with the proxy key");
343 }
344 memcpy(x.signed_rtag, sig, siglen);
345 x.signed_rtag_len = siglen;
346 }
347 }
348
349 /* 4. Build + encrypt the response main: proxy chain + signed tag + new tag. */
350 if (!brix_gsi_rand(newrtag, sizeof(newrtag))) {
351 OPENSSL_cleanse(aeskey, sizeof(aeskey));
352 return gsi_cresp_fail(&x, err, errcap, "gsi: RNG failed");
353 }
354 brix_gbuf_start(&x.inner, (uint32_t) kXGC_cert);
355 brix_gbuf_bucket(&x.inner, (uint32_t) kXRS_x509, proxy_pem, proxy_pem_len);
356 if (x.signed_rtag != NULL) {
357 brix_gbuf_bucket(&x.inner, (uint32_t) kXRS_signed_rtag,
358 x.signed_rtag, x.signed_rtag_len);
359 }
360 brix_gbuf_bucket(&x.inner, (uint32_t) kXRS_rtag, newrtag, sizeof(newrtag));
361 brix_gbuf_end(&x.inner);
362 /* IV is prepended exactly for v>=10400 peers (signed_dh tracks the same
363 * condition); XRDC_GSI_USEIV overrides for interop debugging. */
364 use_iv = signed_dh;
365 {
366 const char *ivov = getenv("XRDC_GSI_USEIV");
367 if (ivov != NULL) { use_iv = atoi(ivov); }
368 }
369 if (!x.inner.err) {
370 x.enc = brix_gsi_cipher_encrypt(&sesscipher, aeskey, x.inner.p,
371 x.inner.len, use_iv, &enclen);
372 }
373 /* Hand the agreed session cipher to the caller before we wipe it — the
374 * client's X.509 delegation round (kXGC_sigpxy) reuses the same key/cipher. */
375 if (out_key != NULL && out_keylen != NULL) {
376 memcpy(out_key, aeskey, sesscipher.key_len);
377 *out_keylen = sesscipher.key_len;
378 }
379 if (out_cipher != NULL && out_cipher_cap > 0) {
380 snprintf(out_cipher, out_cipher_cap, "%s", chosen_cipher);
381 }
382 if (out_use_iv != NULL) {
383 *out_use_iv = use_iv;
384 }
385 OPENSSL_cleanse(aeskey, sizeof(aeskey));
386 if (x.inner.err || x.enc == NULL) {
387 return gsi_cresp_fail(&x, err, errcap, "gsi: main encrypt failed");
388 }
389
390 /* 5. Outer kXGC_cert: our DH public — kXRS_cipher (RSA-signed) for signed-DH,
391 * else a plain kXRS_puk. */
392 x.cpub = brix_gsi_cipher_public(x.mine, &cpublen);
393 if (x.cpub == NULL) {
394 return gsi_cresp_fail(&x, err, errcap, "gsi: cannot encode session public");
395 }
396 brix_gbuf_start(&x.outer, (uint32_t) kXGC_cert);
397 brix_gbuf_bucket(&x.outer, (uint32_t) kXRS_cryptomod, "ssl", 3);
398 if (signed_dh) {
399 size_t cap = cpublen + (size_t) EVP_PKEY_size(proxy_key) + 64;
400 x.signed_cpub = malloc(cap);
401 if (x.signed_cpub != NULL) {
402 x.signed_cpub_len = brix_gsi_rsa_encrypt_private(
403 proxy_key, (const uint8_t *) x.cpub, cpublen, x.signed_cpub, cap);
404 }
405 if (x.signed_cpub == NULL || x.signed_cpub_len == 0) {
406 return gsi_cresp_fail(&x, err, errcap, "gsi: signing session public");
407 }
408 brix_gbuf_bucket(&x.outer, (uint32_t) kXRS_cipher,
409 x.signed_cpub, x.signed_cpub_len);
410 {
411 size_t ppl = 0;
412 x.pubpem = gsi_cresp_export_pubkey_pem(proxy_key, &ppl);
413 if (x.pubpem == NULL) {
414 return gsi_cresp_fail(&x, err, errcap, "gsi: export proxy pubkey");
415 }
416 brix_gbuf_bucket(&x.outer, (uint32_t) kXRS_puk, x.pubpem, ppl);
417 }
418 } else {
419 brix_gbuf_bucket(&x.outer, (uint32_t) kXRS_puk, x.cpub, cpublen);
420 }
421 if (use_iv) {
422 snprintf(cipher_field, sizeof(cipher_field), "%s#%d",
423 chosen_cipher, sesscipher.iv_len);
424 } else {
425 snprintf(cipher_field, sizeof(cipher_field), "%s", chosen_cipher);
426 }
427 brix_gbuf_bucket(&x.outer, (uint32_t) kXRS_cipher_alg, cipher_field,
428 strlen(cipher_field));
429 md_alg_len = gsi_cresp_pick_md_alg(sbody, slen, md_alg, sizeof(md_alg));
430 brix_gbuf_bucket(&x.outer, (uint32_t) kXRS_md_alg, md_alg, md_alg_len);
431 brix_gbuf_bucket(&x.outer, (uint32_t) kXRS_main, x.enc, enclen);
432 brix_gbuf_end(&x.outer);
433 if (x.outer.err) {
434 return gsi_cresp_fail(&x, err, errcap, "gsi: out of memory");
435 }
436
437 *payload = x.outer.p;
438 *plen = (uint32_t) x.outer.len;
439 x.outer.p = NULL; /* ownership → caller */
440 (void) gsi_cresp_fail(&x, NULL, 0, NULL); /* free everything else, no err */
441 return 0;
442}

502 size_t plen, int nodata, uint8_t mac_out[32])
