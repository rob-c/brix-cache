8int64_t
9webdav_lock_parse_timeout(ngx_http_request_t *r,
10 ngx_http_brix_webdav_loc_conf_t *conf)
11{
12 ngx_table_elt_t *h;
13 u_char *p, *end;
14 ngx_uint_t timeout = 3600; /* default 1 hour if not bounded */
15
16 h = webdav_tpc_find_header(r, "Timeout", sizeof("Timeout") - 1);
17 if (h == NULL) {
18 timeout = 3600;
19 } else {
20 p = h->value.data;
21 end = p + h->value.len;
22
23 if (ngx_strncasecmp(p, (u_char *) "Second-", 7) == 0) {
24 p += 7;
25 timeout = ngx_atoi(p, end - p);
26 if (timeout == (ngx_uint_t) NGX_ERROR) {
27 timeout = 3600;
28 }
29 } else if (ngx_strncasecmp(p, (u_char *) "Infinite", 8) == 0) {
30 timeout = conf->lock_timeout;
31 }
32 }
33
34 /* Bound timeout to reasonable range [1, conf->lock_timeout]. */
35 if (timeout < 1) timeout = 1;
36 if (timeout > conf->lock_timeout) timeout = conf->lock_timeout;
37
38 /* Absolute WALL-CLOCK expiry (Unix seconds). Persisted in the lock xattr, so
39 * it must be reboot-stable — ngx_current_msec (monotonic) would not be. */
40 return (int64_t) ngx_time() + (int64_t) timeout;
41}

44webdav_lock_if_header_matches(ngx_http_request_t *r, const char *token)
45{
46 ngx_table_elt_t *h;
47
48 h = webdav_tpc_find_header(r, "If", sizeof("If") - 1);
49 if (h == NULL) {
50 /*
51 * Some clients incorrectly use Lock-Token for refreshes; accept it
52 * here to preserve compatibility with the previous handler behavior.
53 */
54 h = webdav_tpc_find_header(r, "Lock-Token", sizeof("Lock-Token") - 1);
55 if (h == NULL) return 0;
56 }
57
58 if (ngx_strstr(h->value.data, (u_char *) token) != NULL) {
59 return 1;
60 }
61
62 return 0;

66webdav_lock_parse_depth(ngx_http_request_t *r, int *depth_infinity)
67{
68 ngx_table_elt_t *depth_hdr;
69
70 *depth_infinity = 1;
71
72 depth_hdr = webdav_tpc_find_header(r, "Depth", sizeof("Depth") - 1);
73 if (depth_hdr == NULL) {
74 return NGX_OK;
75 }
76
77 if (depth_hdr->value.len == 1 && depth_hdr->value.data[0] == '0') {
78 *depth_infinity = 0;
79 return NGX_OK;

83 (u_char *) "infinity", 8) == 0)

85 return NGX_OK;
86 }
