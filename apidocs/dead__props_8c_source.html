23/*
24 * Phase 40: dead-property xattrs (user.nginx_xrootd.webdav.*) must be set / read /
25 * listed AS THE MAPPED USER under impersonation — else PROPPATCH setxattr on the
26 * user-owned resource fails EACCES from the worker. Derive the export root from
27 * the request and route every xattr op through the VFS xattr surface, which
28 * delegates to brix_*xattr_confined_canon (broker under map mode; raw
29 * path-based syscall otherwise) while adding OP_XATTR metering — confinement and
30 * errno behaviour are unchanged.
31 */

72/*
73 * Emit an empty self-closing element for one property name (used by PROPFIND
74 * propname, which lists names without values). The serialized form depends on
75 * the namespace: DAV: -> <D:name/>; no namespace -> <name/>; any other URI ->
76 * <X:name xmlns:X="..."/> with the (escaped) URI bound to a local "X" prefix.
77 * The local name is re-validated here so a stored-but-now-suspect key cannot
78 * inject markup; the namespace URI is XML-escaped before interpolation.
79 */
80ngx_int_t
81webdav_dead_prop_append_empty(ngx_http_request_t *r, const char *ns,
82 const char *local, ngx_chain_t **head, ngx_chain_t **tail)
83{
84 char *safe_ns;
85
86 if (!webdav_dead_prop_xml_name_ok(local)) {
87 return NGX_ERROR;
88 }
89
90 if (ns != NULL && strcmp(ns, "DAV:") == 0) {
91 return brix_http_chain_appendf(r->pool, head, tail,
92 "<D:%s/>", local) == NULL
93 ? NGX_ERROR : NGX_OK;
94 }
95
96 if (ns == NULL || ns[0] == '\0') {
97 return brix_http_chain_appendf(r->pool, head, tail,
98 "<%s/>", local) == NULL
99 ? NGX_ERROR : NGX_OK;
100 }
101
102 safe_ns = webdav_escape_xml_text(r->pool, ns);
103 if (safe_ns == NULL) {
104 return NGX_ERROR;
105 }
106
107 return brix_http_chain_appendf(r->pool, head, tail,
108 "<X:%s xmlns:X=\"%s\"/>",
109 local, safe_ns) == NULL
110 ? NGX_ERROR : NGX_OK;
111}

113/*
114 * WHAT: Bundle of the request + resource-path + property-identity fields that
115 * every dead-property xattr op needs (request `r`, export `path`,
116 * namespace URI `ns`, local element name `local`).
117 * WHY: The public set/append_value entry points carry these as separate
118 * parameters (their signatures are part of the module's header contract).
119 * Packing them once here lets the internal workers pass a single
120 * pointer, keeping each helper's parameter count within budget without
121 * adding globals or changing the public API.
122 * HOW: Plain aggregate populated by the thin public wrappers; read-only to the
123 * workers.
124 */
125typedef struct {
126 ngx_http_request_t *r;
127 const char *path;
128 const char *ns;
129 const char *local;
130} webdav_dead_prop_target_t;
131
132/*
133 * WHAT: Store `xml[xml_len]` as the xattr value for the target property.
134 * WHY: Core of webdav_dead_prop_set, factored out so the value bytes travel as
135 * one (buf,len) pair alongside the bundled target — five parameters total.
136 * HOW: Reject an oversized value or unencodable name up front (ENAMETOOLONG),
137 * build the confined VFS ctx, then setxattr. Behaviour is byte-for-byte
138 * identical to the pre-split path.
139 */
140static ngx_int_t
141webdav_dead_prop_set_target(const webdav_dead_prop_target_t *t,
142 const char *xml, size_t xml_len)
143{
144 char attr[WEBDAV_DEAD_PROP_NAME_MAX + 1];
145 brix_vfs_ctx_t vctx;
146
147 if (xml_len > WEBDAV_DEAD_PROP_VALUE_MAX
148 || webdav_dead_prop_attr_name(t->ns, t->local, attr, sizeof(attr))
149 != NGX_OK)
150 {
151 errno = ENAMETOOLONG;
152 return NGX_ERROR;
153 }
154
155 webdav_dead_prop_vfs_ctx_init(t->r, t->path, &vctx);
156
157 if (brix_vfs_setxattr(&vctx, attr, xml, xml_len, 0) != NGX_OK) {
158 ngx_log_error(NGX_LOG_WARN, t->r->connection->log, errno,
159 "brix_webdav: setxattr dead property failed");
160 return NGX_ERROR;
161 }
162
163 return NGX_OK;
164}

181/*
182 * PROPPATCH remove: delete the property's xattr. Removing a property that does
183 * not exist is success (idempotent) — ENODATA/ENOATTR are swallowed; any other
184 * errno is a real failure.
185 */
186ngx_int_t
187webdav_dead_prop_remove(ngx_http_request_t *r, const char *path,
188 const char *ns, const char *local)
189{
190 char attr[WEBDAV_DEAD_PROP_NAME_MAX + 1];
191 brix_vfs_ctx_t vctx;
192
193 if (webdav_dead_prop_attr_name(ns, local, attr, sizeof(attr)) != NGX_OK) {
194 errno = ENAMETOOLONG;
195 return NGX_ERROR;
196 }
197
198 webdav_dead_prop_vfs_ctx_init(r, path, &vctx);
199
200 /* "not present" == already removed, so treat ENODATA/ENOATTR as success. */
201 /* phase74-fp: ENOATTR is an alias of ENODATA on Linux but a distinct errno
202 * on BSD/macOS — both are checked deliberately for portability, so the
203 * "equivalent nested operands" finding is moot. */
204 /* NOLINTNEXTLINE(misc-redundant-expression) */
205 if (brix_vfs_removexattr(&vctx, attr) != NGX_OK
206 && errno != ENODATA && errno != ENOATTR)
207 {
208 ngx_log_error(NGX_LOG_WARN, r->connection->log, errno,
209 "brix_webdav: removexattr dead property failed");
210 return NGX_ERROR;
211 }
212
213 return NGX_OK;
214}

216/*
217 * PROPFIND (with values): if the named dead property exists, append its stored
218 * XML fragment to the response chain and set *found. A missing property is not
219 * an error: *found stays 0 and NGX_OK is returned so the caller can report it as
220 * 404 inside the multistatus. Returns NGX_ERROR only on a real fault.
221 *
222 * Two getxattr calls are intentional (size probe, then read). The first gets
223 * the length so we can size the alloc; the value is capped at VALUE_MAX so the
224 * probe cannot drive an unbounded allocation.
225 */
226/*
227 * WHAT: Read the target property's xattr value into a NUL-terminated,
228 * pool-allocated string; return it via *out_value / *out_len.
229 * WHY: The two-call (probe, then read) getxattr dance is the bulk of
230 * append_value; isolating it keeps the appender a short compose step and
231 * holds every helper at <=5 params.
232 * HOW: Encode the key, probe the length (missing/foreign -> NGX_DECLINED so the
233 * caller reports "not found" without an error), cap at VALUE_MAX, alloc,
234 * re-read. Return values and errno match the original inline logic.
235 */
236static ngx_int_t
237webdav_dead_prop_read_value(const webdav_dead_prop_target_t *t,
238 char **out_value, ssize_t *out_len)
239{
240 char attr[WEBDAV_DEAD_PROP_NAME_MAX + 1];
241 brix_vfs_ctx_t vctx;
242 char *value;
243 ssize_t len;
244
245 if (webdav_dead_prop_attr_name(t->ns, t->local, attr, sizeof(attr))
246 != NGX_OK)
247 {
248 return NGX_DECLINED;
249 }
250
251 webdav_dead_prop_vfs_ctx_init(t->r, t->path, &vctx);
252
253 /* Probe length first (buf=NULL, size=0). */
254 len = brix_vfs_getxattr(&vctx, attr, NULL, 0);
255 if (len < 0) {
256 if (errno == ENODATA || errno == ENOATTR) {
257 return NGX_DECLINED;
258 }
259 return NGX_ERROR;
260 }
261
262 if (len > WEBDAV_DEAD_PROP_VALUE_MAX) {
263 return NGX_ERROR;
264 }
265
266 BRIX_PNALLOC_OR_RETURN(value, t->r->pool, (size_t) len + 1, NGX_ERROR);
267
268 /* Second read fetches the actual bytes; a concurrent shrink is fine since
269 * we re-read the returned length, a concurrent grow is bounded by the buf. */
270 len = brix_vfs_getxattr(&vctx, attr, value, (size_t) len);
271 if (len < 0) {
272 return NGX_ERROR;
273 }
274 value[len] = '\0';
275
276 *out_value = value;
277 *out_len = len;
278 return NGX_OK;
279}

282webdav_dead_prop_append_value(ngx_http_request_t *r, const char *path,

309/*
310 * PROPFIND allprop / propname: enumerate every dead property on `path` and
311 * append each one to the response chain. listxattr returns all xattr keys as a
312 * single NUL-separated blob; we walk it, decode the dead-property keys (skipping
313 * any foreign xattr), and emit either just the name (names_only, for propname)
314 * or the name+value (for allprop). A node with no xattrs (len<=0) is normal.
315 */
316ngx_int_t
317webdav_dead_props_append_all(ngx_http_request_t *r, const char *path,
318 ngx_chain_t **head, ngx_chain_t **tail, ngx_flag_t names_only)
319{
320 brix_vfs_ctx_t vctx;
321 char *list;
322 ssize_t len;
323 char *p;
324
325 webdav_dead_prop_vfs_ctx_init(r, path, &vctx);
326
327 /* Probe the size of the NUL-separated key list. */
328 len = brix_vfs_listxattr(&vctx, NULL, 0);
329 if (len <= 0) {
330 return NGX_OK;
331 }
332
333 if (len > WEBDAV_DEAD_PROP_LIST_MAX) {
334 return NGX_OK;
335 }
336
337 BRIX_PNALLOC_OR_RETURN(list, r->pool, (size_t) len, NGX_ERROR);
338
339 len = brix_vfs_listxattr(&vctx, list, (size_t) len);
340 if (len <= 0) {
341 return NGX_OK;
342 }
343
344 /* Each key is NUL-terminated; advance past the terminator to the next. */
345 for (p = list; p < list + len; p += strlen(p) + 1) {
346 char *ns = NULL;
347 char *local = NULL;
348
349 /* Skip xattrs that are not our dead properties (other "user.*", etc.). */
350 if (webdav_dead_prop_decode_attr(r->pool, p, &ns, &local) != NGX_OK) {
351 continue;
352 }
353
354 if (names_only) {
355 if (webdav_dead_prop_append_empty(r, ns, local, head, tail)
356 != NGX_OK)
357 {
358 return NGX_ERROR;
359 }
360 } else {
361 ngx_flag_t found;
362 if (webdav_dead_prop_append_value(r, path, ns, local, head, tail,
363 &found) != NGX_OK)
364 {
365 return NGX_ERROR;
366 }
367 }
368 }
369
370 return NGX_OK;
371}
