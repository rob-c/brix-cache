3 *
4 * WHAT: Builds the 68-byte bootstrap payload (client hello + kXR_protocol

7 *
8 * WHY: Split out of connect_upstream.c as a self-contained concept; the login

24/*
25 * Builds the 68-byte bootstrap buffer:
26 * 20 XRootD client hello
27 * 24 kXR_protocol request
28 * 24 kXR_login request
29 *
30 * username: NUL-terminated string for kXR_login (max 8 chars).
31 * NULL or empty → default "xrd".
32 */
33size_t
34brix_proxy_build_bootstrap(u_char *buf, const char *username)
35{
36 u_char *cursor = buf;
37
38 /* Client hello: 12 zero bytes + protocol version + ROOTD_PQ selector */
39 ngx_memzero(cursor, 12);
40 cursor += 12;
41 {
42 uint32_t v = htonl(4);
43 ngx_memcpy(cursor, &v, 4); cursor += 4;
44 v = htonl(ROOTD_PQ);
45 ngx_memcpy(cursor, &v, 4); cursor += 4;
46 }
47
48 /* kXR_protocol */
49 {
50 ClientProtocolRequest *r = (ClientProtocolRequest *)(void *) cursor;
51 xrdw_protocol_req_t b = { .clientpv = kXR_PROTOCOLVERSION,
52 .expect = 0x03 };
53 ngx_memzero(r, sizeof(*r));
54 r->streamid[0] = 0; r->streamid[1] = 1;
55 r->requestid = htons(kXR_protocol);
56 xrdw_protocol_req_pack(&b, ((ClientRequestHdr *) (void *) cursor)->body);
57 cursor += sizeof(*r);
58 }
59
60 /* kXR_login */
61 {
62 ClientLoginRequest *r = (ClientLoginRequest *)(void *) cursor;
63 xrdw_login_req_t b = { .capver = kXR_ver005 };
64 ngx_atomic_uint_t seq = ngx_atomic_fetch_add(&proxy_upstream_seq, 1);
65
66 b.pid = (int32_t) ((ngx_pid << 16) ^ (seq & 0xFFFF));
67 if (username != NULL && username[0] != '\0') {
68 size_t ulen = ngx_strlen(username);
69 if (ulen > sizeof(b.username)) {
70 ulen = sizeof(b.username);
71 }
72 ngx_memcpy(b.username, username, ulen);
73 } else {
74 b.username[0] = 'x';
75 b.username[1] = 'r';
76 b.username[2] = 'd';
77 }
78
79 ngx_memzero(r, sizeof(*r));
80 r->streamid[0] = 0; r->streamid[1] = 1;
81 r->requestid = htons(kXR_login);
82 xrdw_login_req_pack(&b, ((ClientRequestHdr *) (void *) cursor)->body);
83 cursor += sizeof(*r);
84 }
85
86 return (size_t)(cursor - buf);
87}
