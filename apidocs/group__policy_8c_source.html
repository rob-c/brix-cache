1/*
2 * WHAT: This file implements parent directory group policy inheritance for XRootD mkdir operations. brix_finalize_group_rules() canonicalizes all group rule paths using brix_finalize_path_rules with sizeof(brix_group_rule_t) parameters; brix_parent_group_mode_bits() computes desired group permission bits from parent's st_mode (S_IRWXG for directories, S_IRGRP|S_IWGRP for files + inherited file execute bit); brix_apply_parent_group_policy_impl() performs full policy enforcement: finds matching group rule via longest-prefix match, extracts parent directory path via strrchr, stat(parent) and stat/fstat(child), computes desired_mode (child mode masked without group bits then ORed with parent-derived bits), applies S_ISGID from parent if child is a directory, chown/fchown child to parent's gid if different, chmod/fchmod child to desired_mode if differs. Public wrappers: brix_apply_parent_group_policy_fd() (fd-based), brix_apply_parent_group_policy_path() (path-based).
3 *
4 * WHY: Group policy inheritance ensures new files/directories created under a parent directory inherit the parent's group ownership and appropriate group permission bits — critical for shared HEP datasets where multiple users in the same Unix group need consistent access. S_ISGID propagation on directories ensures subdirectories also inherit the group; file execute bit inherited only if child already has it prevents over-permissive mode assignment on regular files. */

29static mode_t
30brix_parent_group_mode_bits(const struct stat *parent,
31 const struct stat *child)
32{
33 mode_t group_bits;
34
35 if (S_ISDIR(child->st_mode)) {
36 group_bits = parent->st_mode & S_IRWXG;
37 } else {
38 group_bits = parent->st_mode & (S_IRGRP | S_IWGRP);
39 if (child->st_mode & S_IXGRP) {
40 group_bits |= S_IXGRP;
41 }
42 }
43
44 return group_bits;
45}
46/* HOW: If S_ISDIR(child->st_mode): group_bits=parent->st_mode & S_IRWXG (all 3 group bits). Else: group_bits=parent->st_mode & (S_IRGRP | S_IWGRP) (read+write only for files). If child->st_mode & S_IXGRP → group_bits |= S_IXGRP (inherits file execute bit if child already has it). Returns computed group_bits. */

48static ngx_int_t
49brix_apply_parent_group_policy_impl(ngx_log_t *log, int fd,
50 const char *path, ngx_array_t *rules)
51{
52 const brix_group_rule_t *rule;
53 struct stat parent_st;
54 struct stat child_st;
55 char parent[PATH_MAX];
56 char *slash;
57 mode_t desired_mode;
58 int rc;
59
60 (void) log;
61
62 rule = brix_find_group_rule(path, rules);
63 if (rule == NULL) {
64 return NGX_DECLINED;
65 }
66
67 rc = snprintf(parent, sizeof(parent), "%s", path);
68 if (rc < 0 || (size_t) rc >= sizeof(parent)) {
69 errno = ENAMETOOLONG;
70 return NGX_ERROR;
71 }
72
73 /* Derive the parent directory by truncating at the final '/'. slash==parent
74 * means the only slash is at index 0 (a top-level entry like "/foo"), whose
75 * parent is "/" — declined here so policy never reattributes the filesystem
76 * root's gid/mode to a child. slash==NULL means a relative/malformed path. */
77 slash = strrchr(parent, '/');
78 if (slash == NULL || slash == parent) {
79 return NGX_DECLINED;
80 }
81
82 /* In-place truncate: turns "/data/atlas/file" into "/data/atlas". */
83 *slash = '\0';
84 if (stat(parent, &parent_st) != 0) {
85 return NGX_ERROR;
86 }
87
88 if (fd >= 0) {
89 if (fstat(fd, &child_st) != 0) {
90 return NGX_ERROR;
91 }
92 } else {
93 if (stat(path, &child_st) != 0) {
94 return NGX_ERROR;
95 }
96 }
97
98 /* Clear-then-set: strip the child's existing group bits and any stale setgid
99 * (~(S_IRWXG | S_ISGID)) so the parent-derived bits are authoritative rather
100 * than merged with whatever umask/create-mode left behind. Owner and other
101 * bits are preserved untouched. setgid is re-added below only when warranted,
102 * so it is never inherited onto plain files. */
103 desired_mode = (child_st.st_mode & ~(S_IRWXG | S_ISGID))
104 | brix_parent_group_mode_bits(&parent_st, &child_st);
105
106 /* Propagate setgid only to subdirectories of a setgid parent, mirroring the
107 * kernel's directory-inheritance semantics so deeper mkdir keeps the group. */
108 if (S_ISDIR(child_st.st_mode) && (parent_st.st_mode & S_ISGID)) {
109 desired_mode |= S_ISGID;
110 }
111
112 /* Apply gid first (uid kept via (uid_t)-1), THEN mode below. A chown can
113 * silently clear setuid/setgid on some kernels, so re-asserting desired_mode
114 * afterwards restores any S_ISGID this policy intends to keep. The gid is
115 * taken from the parent dir, never from the request, so the caller cannot
116 * choose an arbitrary group. */
117 if (child_st.st_gid != parent_st.st_gid) {
118 if (fd >= 0) {
119 if (fchown(fd, (uid_t) -1, parent_st.st_gid) != 0) {
120 return NGX_ERROR;
121 }
122 } else {
123 if (chown(path, (uid_t) -1, parent_st.st_gid) != 0) {
124 return NGX_ERROR;
125 }
126 }
127 }
128
129 /* Compare and apply only the low 12 permission/special bits (07777 =
130 * setuid|setgid|sticky + rwxrwxrwx); the file-type bits in st_mode are
131 * masked off so chmod is skipped entirely when nothing in those bits moved. */
132 if ((child_st.st_mode & 07777) != (desired_mode & 07777)) {
133 if (fd >= 0) {
134 if (fchmod(fd, desired_mode & 07777) != 0) {
135 return NGX_ERROR;
136 }
137 } else {
138 if (chmod(path, desired_mode & 07777) != 0) {
139 return NGX_ERROR;
140 }
141 }
142 }
143
144 return NGX_OK;
145}
146/* HOW: Finds rule=brix_find_group_rule(path, rules) — if NULL returns NGX_DECLINED (no matching rule). snprintf(parent,sizeof,parent,"%s",path) into parent buffer; rc<0 or rc>=sizeof → errno=ENAMETOOLONG+NGX_ERROR. strrchr(parent,'/') — if NULL || ==parent (root path) returns NGX_DECLINED. *slash='\0' truncates to parent dir path. stat(parent,&parent_st) — if !=0 returns NGX_ERROR. fd>=0: fstat(fd,&child_st); else: stat(path,&child_st) — both if !=0 return NGX_ERROR. desired_mode=(child_st.st_mode & ~(S_IRWXG|S_ISGID)) | brix_parent_group_mode_bits(&parent_st, &child_st). If S_ISDIR(child)&&parent has S_ISGID → desired_mode |= S_ISGID (propagate setgid to subdirs). If child_st.gid != parent_st.gid: fd>=0→fchown(fd,-1,parent_st.gid); else→chown(path,-1,parent_st.gid) — both if !=0 return NGX_ERROR. If (child&07777)!=(desired&07777): fd>=0→fchmod(fd,desired&07777); else→chmod(path,...) — both if !=0 return NGX_ERROR. Returns NGX_OK on success. */
