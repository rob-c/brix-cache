1/*
2 * WHAT: This file implements parent directory group policy inheritance for XRootD mkdir operations. brix_finalize_group_rules() canonicalizes all group rule paths using brix_finalize_path_rules with sizeof(brix_group_rule_t) parameters; brix_parent_group_mode_bits() computes desired group permission bits from parent's st_mode (S_IRWXG for directories, S_IRGRP|S_IWGRP for files + inherited file execute bit); brix_apply_parent_group_policy_impl() performs full policy enforcement: finds matching group rule via longest-prefix match, extracts parent directory path via strrchr, stat(parent) and stat/fstat(child), computes desired_mode (child mode masked without group bits then ORed with parent-derived bits), applies S_ISGID from parent if child is a directory, chown/fchown child to parent's gid if different, chmod/fchmod child to desired_mode if differs. Public wrappers: brix_apply_parent_group_policy_fd() (fd-based), brix_apply_parent_group_policy_path() (path-based).
3 *
4 * WHY: Group policy inheritance ensures new files/directories created under a parent directory inherit the parent's group ownership and appropriate group permission bits — critical for shared HEP datasets where multiple users in the same Unix group need consistent access. S_ISGID propagation on directories ensures subdirectories also inherit the group; file execute bit inherited only if child already has it prevents over-permissive mode assignment on regular files. */

29static mode_t
30brix_parent_group_mode_bits(const struct stat *parent,
31 const struct stat *child)
32{
33 mode_t group_bits;
34
35 if (S_ISDIR(child->st_mode)) {
36 group_bits = parent->st_mode & S_IRWXG;
37 } else {
38 group_bits = parent->st_mode & (S_IRGRP | S_IWGRP);
39 if (child->st_mode & S_IXGRP) {
40 group_bits |= S_IXGRP;
41 }
42 }
43
44 return group_bits;
45}
46/* HOW: If S_ISDIR(child->st_mode): group_bits=parent->st_mode & S_IRWXG (all 3 group bits). Else: group_bits=parent->st_mode & (S_IRGRP | S_IWGRP) (read+write only for files). If child->st_mode & S_IXGRP → group_bits |= S_IXGRP (inherits file execute bit if child already has it). Returns computed group_bits. */

48/*
49 * WHAT: brix_derive_parent_dir() copies path into the caller's parent buffer,
50 * then truncates it in place at the final '/' so it names the parent directory.
51 * Returns NGX_OK with parent populated, NGX_DECLINED when the path has no usable
52 * parent (relative/malformed, or a top-level entry whose parent is the FS root),
53 * or NGX_ERROR (with errno=ENAMETOOLONG) when path does not fit the buffer.
54 *
55 * WHY: Isolating the parent-derivation branches (overflow check plus the two
56 * strrchr sentinels) out of the policy driver keeps each unit simple while
57 * preserving the exact decline/error semantics the policy depends on: a
58 * top-level entry must be declined so policy never reattributes the filesystem
59 * root's gid/mode to a child.
60 *
61 * HOW: snprintf(parent,parent_size,"%s",path); rc<0 or rc>=parent_size →
62 * errno=ENAMETOOLONG + NGX_ERROR. strrchr(parent,'/') — if NULL (no slash) or
63 * ==parent (only slash at index 0, e.g. "/foo") → NGX_DECLINED. Otherwise
64 * *slash='\0' truncates ("/data/atlas/file" → "/data/atlas") and returns NGX_OK.
65 */
66static ngx_int_t
67brix_derive_parent_dir(const char *path, char *parent, size_t parent_size)
68{
69 char *slash;
70 int rc;
71
72 rc = snprintf(parent, parent_size, "%s", path);
73 if (rc < 0 || (size_t) rc >= parent_size) {
74 errno = ENAMETOOLONG;
75 return NGX_ERROR;
76 }
77
78 slash = strrchr(parent, '/');
79 if (slash == NULL || slash == parent) {
80 return NGX_DECLINED;
81 }
82
83 *slash = '\0';
84 return NGX_OK;
85}

114/*
115 * WHAT: brix_apply_child_gid() sets the child's group to gid (uid untouched via
116 * (uid_t)-1), using fchown(fd) when fd>=0 and chown(path) otherwise.
117 *
118 * WHY: The gid is taken from the parent dir, never from the request, so the
119 * caller cannot choose an arbitrary group. Factoring the fd/path split keeps the
120 * driver's ownership step to a single decision while preserving that guarantee.
121 *
122 * HOW: fd>=0 → fchown(fd,(uid_t)-1,gid); else → chown(path,(uid_t)-1,gid). Either
123 * returning non-zero → NGX_ERROR (errno from the syscall). Else NGX_OK.
124 */
125static ngx_int_t
126brix_apply_child_gid(int fd, const char *path, gid_t gid)
127{
128 if (fd >= 0) {
129 if (fchown(fd, (uid_t) -1, gid) != 0) {
130 return NGX_ERROR;
131 }
132 } else {
133 if (chown(path, (uid_t) -1, gid) != 0) {
134 return NGX_ERROR;
135 }
136 }
137
138 return NGX_OK;

141/*
142 * WHAT: brix_apply_child_mode() sets the child's low 12 permission/special bits
143 * to mode&07777, using fchmod(fd) when fd>=0 and chmod(path) otherwise.
144 *
145 * WHY: A chown can silently clear setuid/setgid on some kernels, so mode is
146 * re-asserted after ownership; keeping the fd/path split here mirrors the gid
147 * helper and leaves the driver a single mode-application decision.
148 *
149 * HOW: fd>=0 → fchmod(fd,mode&07777); else → chmod(path,mode&07777). Either
150 * returning non-zero → NGX_ERROR (errno from the syscall). Else NGX_OK.
151 */
152static ngx_int_t
153brix_apply_child_mode(int fd, const char *path, mode_t mode)
154{
155 if (fd >= 0) {
156 if (fchmod(fd, mode & 07777) != 0) {
157 return NGX_ERROR;
158 }
159 } else {
160 if (chmod(path, mode & 07777) != 0) {
161 return NGX_ERROR;
162 }
163 }
164
165 return NGX_OK;
166}

168/*
169 * WHAT: brix_desired_child_mode() computes the child's target mode from its
170 * current mode and the parent's mode: clears the child's group bits and any
171 * stale setgid, ORs in the parent-derived group bits, and re-adds S_ISGID only
172 * for a subdirectory of a setgid parent.
173 *
174 * WHY: Clear-then-set makes the parent-derived bits authoritative rather than
175 * merged with whatever umask/create-mode left behind, while owner/other bits are
176 * preserved untouched; setgid propagates only to subdirectories of a setgid
177 * parent, mirroring the kernel's directory-inheritance semantics so deeper mkdir
178 * keeps the group and plain files never inherit setgid.
179 *
180 * HOW: desired = (child->st_mode & ~(S_IRWXG|S_ISGID)) |
181 * brix_parent_group_mode_bits(parent,child). If S_ISDIR(child->st_mode) and
182 * (parent->st_mode & S_ISGID) → desired |= S_ISGID. Returns desired.
183 */
184static mode_t
185brix_desired_child_mode(const struct stat *parent, const struct stat *child)
186{
187 mode_t desired_mode;
188
189 desired_mode = (child->st_mode & ~(S_IRWXG | S_ISGID))
190 | brix_parent_group_mode_bits(parent, child);
191
192 if (S_ISDIR(child->st_mode) && (parent->st_mode & S_ISGID)) {
193 desired_mode |= S_ISGID;
194 }
195
196 return desired_mode;
197}

199/*
200 * WHAT: brix_apply_parent_group_policy_impl() enforces parent-directory group
201 * policy on the child named by path (statted via fd when fd>=0): it finds the
202 * matching group rule, derives and stats the parent directory, stats the child,
203 * computes the desired mode, then chowns the child's gid to the parent's and
204 * chmods it to the desired mode when either differs.
205 *
206 * WHY: This is the single policy driver behind both public wrappers. Ordering is
207 * load-bearing: gid is applied before mode so the mode re-assert restores any
208 * S_ISGID a chown may have cleared, and non-matching / rootless paths are
209 * declined so policy never touches objects it does not own.
210 *
211 * HOW: rule=brix_find_group_rule(path,rules) — NULL → NGX_DECLINED.
212 * brix_derive_parent_dir(path,parent,sizeof) — non-OK returned as-is
213 * (NGX_DECLINED for rootless/malformed, NGX_ERROR+ENAMETOOLONG for overflow).
214 * stat(parent,&parent_st)!=0 → NGX_ERROR. brix_stat_child(fd,path,&child_st)
215 * propagated. desired_mode=brix_desired_child_mode(&parent_st,&child_st). If
216 * child_st.st_gid!=parent_st.st_gid → brix_apply_child_gid(fd,path,
217 * parent_st.st_gid), propagate non-OK. If (child_st.st_mode & 07777) !=
218 * (desired_mode & 07777) → brix_apply_child_mode(fd,path,desired_mode),
219 * propagate non-OK. Returns NGX_OK on success.
220 */
221static ngx_int_t
222brix_apply_parent_group_policy_impl(ngx_log_t *log, int fd,
223 const char *path, ngx_array_t *rules)
224{
225 const brix_group_rule_t *rule;
226 struct stat parent_st;
227 struct stat child_st;
228 char parent[PATH_MAX];
229 mode_t desired_mode;
230 ngx_int_t rc;
231
232 (void) log;
233
234 rule = brix_find_group_rule(path, rules);
235 if (rule == NULL) {
236 return NGX_DECLINED;
237 }
238
239 rc = brix_derive_parent_dir(path, parent, sizeof(parent));
240 if (rc != NGX_OK) {
241 return rc;
242 }
243
244 if (stat(parent, &parent_st) != 0) {
245 return NGX_ERROR;
246 }
247
248 rc = brix_stat_child(fd, path, &child_st);
249 if (rc != NGX_OK) {
250 return rc;
251 }
252
253 desired_mode = brix_desired_child_mode(&parent_st, &child_st);
254
255 /* Apply gid first (uid kept via (uid_t)-1), THEN mode below. The gid is
256 * taken from the parent dir, never from the request, so the caller cannot
257 * choose an arbitrary group. */
258 if (child_st.st_gid != parent_st.st_gid) {
259 rc = brix_apply_child_gid(fd, path, parent_st.st_gid);
260 if (rc != NGX_OK) {
261 return rc;
262 }
263 }
264
265 /* Compare and apply only the low 12 permission/special bits (07777 =
266 * setuid|setgid|sticky + rwxrwxrwx); the file-type bits in st_mode are
267 * masked off so chmod is skipped entirely when nothing in those bits moved. */
268 if ((child_st.st_mode & 07777) != (desired_mode & 07777)) {
269 rc = brix_apply_child_mode(fd, path, desired_mode);
270 if (rc != NGX_OK) {
271 return rc;
272 }
273 }
274
275 return NGX_OK;
276}
