62/* Derive a 32-byte key from a keyfile: FNV-1a fold of the file bytes expanded
63 * with a small mix. NOT a KDF — a lab keyed transform, per the phase non-goals. */
64static int
65xform_load_key(const char *keyfile, unsigned char key[32])
66{
67 unsigned char buf[4096];
68 uint64_t h = 1469598103934665603ULL;
69 ssize_t n;
70 int fd, i, got = 0;
71
72 fd = open(keyfile, O_RDONLY);
73 if (fd < 0) {
74 return -1; /* errno = ENOENT/EACCES */
75 }
76 while ((n = read(fd, buf, sizeof(buf))) > 0) {
77 for (i = 0; i < n; i++) {
78 h ^= buf[i];
79 h *= 1099511628211ULL;
80 }
81 got += (int) n;
82 }
83 close(fd);
84 if (n < 0 || got == 0) {
85 errno = n < 0 ? errno : EINVAL; /* empty keyfile ⇒ EINVAL */
86 return -1;
87 }
88 for (i = 0; i < 32; i++) {
89 h ^= (uint64_t) (i + 1) * 0x9E3779B97F4A7C15ULL;
90 h *= 1099511628211ULL;
91 key[i] = (unsigned char) (h >> 29);
92 }
93 return 0;
94}

96int
97pblock_xform_config(pblock_xform_t *xf, const char *spec, size_t len)
98{
99 memset(xf, 0, sizeof(*xf));
100 if (spec == NULL || len == 0) {
101 return 0; /* kind NONE */
102 }
103 if (len == 4 && memcmp(spec, "zstd", 4) == 0) {
104#if BRIX_HAVE_ZSTD
105 xf->kind = PBLOCK_XFORM_ZSTD;
106 return 0;
107#else
108 errno = ENOTSUP; /* config error: no libzstd */
109 return -1;
110#endif
111 }
112 if (len > 6 && memcmp(spec, "crypt:", 6) == 0) {
113 char kf[4096];
114
115 if (len - 6 >= sizeof(kf)) {
116 errno = EINVAL;
117 return -1;
118 }
119 memcpy(kf, spec + 6, len - 6);
120 kf[len - 6] = '\0';
121 if (xform_load_key(kf, xf->key) != 0) {
122 return -1; /* errno from xform_load_key */
123 }
124 xf->kind = PBLOCK_XFORM_CRYPT;
125 return 0;
126 }
127 errno = EINVAL;
128 return -1;
129}

133/* Keyed XOR keystream over one block: seed an LCG from the export key folded with
134 * the block index, XOR len bytes in place. Reversible; NOT cryptographic. */
135static void
136xform_crypt_xor(const unsigned char key[32], int64_t idx, unsigned char *p,
137 size_t len)
138{
139 uint64_t seed = 1469598103934665603ULL ^ ((uint64_t) idx * 0x9E3779B97F4A7C15ULL);
140 size_t i;
141
142 for (i = 0; i < 32; i++) {
143 seed ^= key[i];
144 seed *= 1099511628211ULL;
145 }
146 for (i = 0; i < len; i++) {
147 seed = seed * 6364136223846793005ULL + 1442695040888963407ULL;
148 p[i] ^= (unsigned char) (seed >> 33);
149 }
150}

152/* Encode logical[0..llen) into out[0..outcap); returns phys_len or -1/errno. */
153static ssize_t
154xform_encode(const pblock_xform_t *xf, int64_t idx, const unsigned char *logical,
155 uint32_t llen, unsigned char *out, size_t outcap)
156{
157 if (xf->kind == PBLOCK_XFORM_CRYPT) {
158 if (llen > outcap) {
159 errno = ERANGE;
160 return -1;
161 }
162 memcpy(out, logical, llen);
163 xform_crypt_xor(xf->key, idx, out, llen);
164 return (ssize_t) llen;
165 }
166#if BRIX_HAVE_ZSTD
167 if (xf->kind == PBLOCK_XFORM_ZSTD) {
168 size_t z = ZSTD_compress(out, outcap, logical, llen, 3);
169
170 if (ZSTD_isError(z)) {
171 errno = EIO;
172 return -1;
173 }
174 return (ssize_t) z;
175 }
176#endif
177 errno = ENOTSUP;
178 return -1;
179}

181/* Decode phys[0..plen) (block idx) into out[0..llen); returns 0 or -1/errno. */
182static int
183xform_decode(const pblock_xform_t *xf, int64_t idx, const unsigned char *phys,
184 uint32_t plen, unsigned char *out, uint32_t llen)
185{
186 if (xf->kind == PBLOCK_XFORM_CRYPT) {
187 if (plen != llen) {
188 errno = EIO;
189 return -1;
190 }
191 memcpy(out, phys, llen);
192 xform_crypt_xor(xf->key, idx, out, llen);
193 return 0;
194 }
195#if BRIX_HAVE_ZSTD
196 if (xf->kind == PBLOCK_XFORM_ZSTD) {
197 size_t d = ZSTD_decompress(out, llen, phys, plen);
198
199 if (ZSTD_isError(d) || d != llen) {
200 errno = EIO;
201 return -1;
202 }
203 return 0;
204 }
205#endif
206 errno = ENOTSUP;
207 return -1;
208}

212static ssize_t
213read_full(int fd, void *buf, size_t len, off_t off)
214{
215 size_t done = 0;
216
217 while (done < len) {
218 ssize_t r = pread(fd, (char *) buf + done, len - done, off + (off_t) done);
219
220 if (r < 0) {
221 if (errno == EINTR) {
222 continue;
223 }
224 return -1;
225 }
226 if (r == 0) {
227 break;
228 }
229 done += (size_t) r;
230 }
231 return (ssize_t) done;
232}

234static int
235write_full(int fd, const void *buf, size_t len)
236{
237 size_t done = 0;
238
239 while (done < len) {
240 ssize_t w = write(fd, (const char *) buf + done, len - done);
241
242 if (w < 0) {
243 if (errno == EINTR) {
244 continue;
245 }
246 return -1;
247 }
248 done += (size_t) w;
249 }
250 return 0;
251}

253int
254pblock_xform_block_load(const pblock_xform_t *xf, int64_t idx, const char *path,
255 unsigned char *out, int64_t bs, uint32_t *llen_out)
256{
257 unsigned char hdr[PBLOCK_XFORM_HDR];
258 unsigned char *phys;
259 uint32_t llen, plen;
260 ssize_t r;
261 int fd, rc = 0;
262
263 memset(out, 0, (size_t) bs);
264 *llen_out = 0;
265
266 fd = open(path, O_RDONLY);
267 if (fd < 0) {
268 return errno == ENOENT ? 0 : -1; /* missing block = hole */
269 }
270 r = read_full(fd, hdr, sizeof(hdr), 0);
271 if (r == 0) { /* freshly-created empty block */
272 close(fd);
273 return 0;
274 }
275 if (r != (ssize_t) sizeof(hdr)) {
276 close(fd);
277 errno = EIO;
278 return -1;
279 }
280 llen = (uint32_t) hdr[0] | (uint32_t) hdr[1] << 8
281 | (uint32_t) hdr[2] << 16 | (uint32_t) hdr[3] << 24;
282 plen = (uint32_t) hdr[4] | (uint32_t) hdr[5] << 8
283 | (uint32_t) hdr[6] << 16 | (uint32_t) hdr[7] << 24;
284 if (llen > (uint32_t) bs || plen > (uint32_t) bs + PBLOCK_XFORM_HDR + 512) {
285 close(fd);
286 errno = EIO; /* corrupt header */
287 return -1;
288 }
289 phys = malloc(plen ? plen : 1);
290 if (phys == NULL) {
291 close(fd);
292 errno = ENOMEM;
293 return -1;
294 }
295 r = read_full(fd, phys, plen, PBLOCK_XFORM_HDR);
296 close(fd);
297 if (r != (ssize_t) plen) {
298 free(phys);
299 errno = EIO;
300 return -1;
301 }
302 if (xform_decode(xf, idx, phys, plen, out, llen) != 0) {
303 rc = -1; /* errno set by decode */
304 } else {
305 *llen_out = llen;
306 }
307 free(phys);
308 return rc;
309}

311int
312pblock_xform_block_store(const pblock_xform_t *xf, int64_t idx, const char *path,
313 const unsigned char *logical, uint32_t llen, int64_t bs)
314{
315 unsigned char hdr[PBLOCK_XFORM_HDR];
316 unsigned char *out;
317 size_t outcap;
318 ssize_t plen;
319 int fd;
320 (void) bs;
321
322#if BRIX_HAVE_ZSTD
323 outcap = xf->kind == PBLOCK_XFORM_ZSTD ? ZSTD_compressBound(llen) : llen;
324#else
325 outcap = llen;
326#endif
327 out = malloc(outcap ? outcap : 1);
328 if (out == NULL) {
329 errno = ENOMEM;
330 return -1;
331 }
332 plen = xform_encode(xf, idx, logical, llen, out, outcap);
333 if (plen < 0) {
334 free(out);
335 return -1;
336 }
337 hdr[0] = (unsigned char) (llen);
338 hdr[1] = (unsigned char) (llen >> 8);
339 hdr[2] = (unsigned char) (llen >> 16);
340 hdr[3] = (unsigned char) (llen >> 24);
341 hdr[4] = (unsigned char) ((uint32_t) plen);
342 hdr[5] = (unsigned char) ((uint32_t) plen >> 8);
343 hdr[6] = (unsigned char) ((uint32_t) plen >> 16);
344 hdr[7] = (unsigned char) ((uint32_t) plen >> 24);
345
346 /* In place (not temp+rename) so the block-0 fd the caller holds open stays
347 * valid; a transformed export is single-writer per block. */
348 fd = open(path, O_WRONLY | O_CREAT, 0600);
349 if (fd < 0) {
350 free(out);
351 return -1;
352 }
353 if (write_full(fd, hdr, sizeof(hdr)) != 0
354 || write_full(fd, out, (size_t) plen) != 0
355 || ftruncate(fd, (off_t) (PBLOCK_XFORM_HDR + plen)) != 0)
356 {
357 int e = errno;
358
359 close(fd);
360 free(out);
361 errno = e;
362 return -1;
363 }
364 close(fd);
365 free(out);
366 return 0;
367}
