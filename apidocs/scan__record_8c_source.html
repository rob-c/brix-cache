13/* Bound on a single escaped logical path rendered into a record. Export paths
14 * are far shorter; an over-long path yields -1 rather than a truncated line. */
15#define SCAN_ESC_MAX 2048

17int
18brix_scan_json_escape(const char *in, size_t len, char *out, size_t cap)
19{
20 size_t o = 0;
21 size_t i;
22 const char *esc;
23
24 for (i = 0; i < len; i++) {
25 unsigned char ch = (unsigned char) in[i];
26 char ubuf[7];
27
28 switch (ch) {
29 case '"': esc = "\\\""; break;
30 case '\\': esc = "\\\\"; break;
31 case '\b': esc = "\\b"; break;
32 case '\f': esc = "\\f"; break;
33 case '\n': esc = "\\n"; break;
34 case '\r': esc = "\\r"; break;
35 case '\t': esc = "\\t"; break;
36 default:
37 if (ch < 0x20) {
38 snprintf(ubuf, sizeof(ubuf), "\\u%04x", ch);
39 esc = ubuf;
40 } else {
41 /* room for the byte + NUL */
42 if (o + 1 >= cap) {
43 return -1;
44 }
45 out[o++] = (char) ch;
46 continue;
47 }
48 }
49 {
50 size_t elen = strlen(esc);
51 if (o + elen >= cap) {
52 return -1;
53 }
54 memcpy(out + o, esc, elen);
55 o += elen;
56 }
57 }
58 if (o >= cap) {
59 return -1;
60 }
61 out[o] = '\0';
62 return (int) o;
63}

65int
66brix_scan_record_file(char *buf, size_t cap, const char *path, int64_t size,
67 int64_t mtime, const char *alg, const char *stored,
68 const char *computed, const char *status)
69{
70 char esc[SCAN_ESC_MAX];
71 char stored_part[160];
72 char computed_part[160];
73 int n;
74
75 if (brix_scan_json_escape(path, strlen(path), esc, sizeof(esc)) < 0) {
76 return -1;
77 }
78 if (stored != NULL) {
79 snprintf(stored_part, sizeof(stored_part), "\"stored\":\"%s\"", stored);
80 } else {
81 snprintf(stored_part, sizeof(stored_part), "\"stored\":null");
82 }
83 if (computed != NULL) {
84 snprintf(computed_part, sizeof(computed_part),
85 ",\"computed\":\"%s\"", computed);
86 } else {
87 computed_part[0] = '\0';
88 }
89
90 n = snprintf(buf, cap,
91 "{\"t\":\"file\",\"path\":\"%s\",\"size\":%lld,\"mtime\":%lld,"
92 "\"alg\":\"%s\",%s%s,\"status\":\"%s\"}",
93 esc, (long long) size, (long long) mtime, alg,
94 stored_part, computed_part, status);
95 if (n < 0 || (size_t) n >= cap) {
96 return -1;
97 }
98 return n;
99}

139/* Render an optional string as a JSON value into `part` (cap bytes): a quoted,
140 * escaped string when `s` is non-NULL, else the literal `null`. Returns 0, or -1
141 * if escaping/quoting would overflow. Keeps the nullable key/path fields of the
142 * object/drift records single-sourced. */
143static int
144scan_str_or_null(const char *s, char *part, size_t cap)
145{
146 char esc[SCAN_ESC_MAX];
147 int n;
148
149 if (s == NULL) {
150 n = snprintf(part, cap, "null");
151 return (n < 0 || (size_t) n >= cap) ? -1 : 0;
152 }
153 if (brix_scan_json_escape(s, strlen(s), esc, sizeof(esc)) < 0) {
154 return -1;
155 }
156 n = snprintf(part, cap, "\"%s\"", esc);
157 return (n < 0 || (size_t) n >= cap) ? -1 : 0;
158}

164{
165 char keyesc[SCAN_ESC_MAX];

168
169 if (key == NULL) {
170 return -1; /* a catalog object always has a backend key */
171 }
172 if (brix_scan_json_escape(key, strlen(key), keyesc, sizeof(keyesc)) < 0) {
173 return -1;
174 }
175 if (scan_str_or_null(path, path_part, sizeof(path_part)) < 0) {
176 return -1;
177 }
178 n = snprintf(buf, cap,
179 "{\"t\":\"object\",\"key\":\"%s\",\"path\":%s,\"size\":%lld,"
180 "\"mtime\":%lld,\"orphan\":%s}",
181 keyesc, path_part, (long long) size, (long long) mtime,
