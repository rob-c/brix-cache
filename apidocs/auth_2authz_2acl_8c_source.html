20/* Return 1 if the client's colon/space-separated vo_list contains required_vo. */
21ngx_flag_t
22brix_vo_list_contains(const char *vo_list, const char *required_vo)
23{
24 const char *start;
25 const char *end;
26 size_t required_len;
27
28 if (required_vo == NULL || required_vo[0] == '\0') {
29 return 1;
30 }
31
32 if (vo_list == NULL || vo_list[0] == '\0') {
33 return 0;
34 }
35
36 required_len = strlen(required_vo);
37 start = vo_list;
38
39 while (*start != '\0') {
40 end = strchr(start, ',');
41
42 if (end == NULL) {
43 end = start + strlen(start);
44 }
45
46 if ((size_t) (end - start) == required_len
47 && ngx_strncmp(start, required_vo, required_len) == 0)
48 {
49 return 1;
50 }
51
52 start = (*end == '\0') ? end : end + 1;
53 }
54
55 return 0;
56}
