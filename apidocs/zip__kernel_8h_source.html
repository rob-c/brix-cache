1/*
2 * zip_kernel.h — shared ZIP central-directory parsing kernel.
3 *
4 * WHAT: The security-critical, read-only ZIP locator shared by the server's
5 * member-access path (src/zip/zip_dir.c) and the native client's archive
6 * reader/writer (client/lib/zip.c): little-endian field readers, a
7 * bounds-checked archive read, End-Of-Central-Directory + ZIP64 location,
8 * the CDFH ZIP64-extra-field decoder, and the Local-File-Header → first-
9 * data-byte resolver.
10 * WHY: Both consumers previously carried byte-for-byte copies of this parser.
11 * The archive is untrusted input and every offset/length must be validated
12 * against the archive size; keeping one audited, unit-tested copy removes
13 * the risk of the two diverging. The higher-level directory walk differs
14 * between the two (the client builds a full listing; the server resolves a
15 * single member, last-match-wins) so that loop stays with each consumer —
16 * only the genuinely-identical leaf parsers live here.
17 * HOW: Pure C — no nginx, no OpenSSL, no fixed I/O source. The archive is read
18 * solely through a caller-supplied `zip_pread_fn`, so the same code serves
19 * a local fd (server, via an SD-backed adapter) and a remote root:// handle
20 * (client). Every read is bounds-checked against the archive size before it
21 * is issued; a hostile or truncated archive yields a clean error code,
22 * never an out-of-bounds access.
23 */
24#ifndef BRIX_ZIP_KERNEL_H
25#define BRIX_ZIP_KERNEL_H

31/*
32 * Read exactly `len` bytes at archive offset `off` into `buf`. Returns the
33 * number of bytes read (a short read or -1 signals failure). `ctx` is the
34 * caller's opaque archive handle (an fd box, a remote-handle struct, ...).
35 */
36typedef ssize_t (*zip_pread_fn)(void *ctx, uint64_t off, void *buf, size_t len);

38/* Kernel result codes (consumers map these onto their own error enums). */
39#define ZIP_K_OK 0
40#define ZIP_K_ENOTZIP (-1) /* no End-Of-Central-Directory record found */
41#define ZIP_K_ECORRUPT (-2) /* malformed / failed bounds check / cap exceeded */
42#define ZIP_K_EIO (-3) /* the pread callback failed */

49/*
50 * Bounds-checked exact read: rejects any range not fully inside [0,
51 * archive_size) before calling `pread`. Returns ZIP_K_OK, ZIP_K_ECORRUPT (the
52 * range is out of bounds) or ZIP_K_EIO (the callback failed / short read).
53 */
54int zip_read_at(zip_pread_fn pread, void *ctx, uint64_t archive_size,
55 uint64_t off, void *buf, size_t len);

57/*
58 * Locate the central directory, following the ZIP64 records when any classic
59 * 32-bit field is saturated. `cd_max` / `max_entries` are anti-bomb caps (pass 0
60 * to disable a cap). On success fills *cd_off / *cd_size / *n_entries and
61 * guarantees [cd_off, cd_off+cd_size) lies within the archive. Returns ZIP_K_OK,
62 * ZIP_K_ENOTZIP (no EOCD), ZIP_K_ECORRUPT (malformed / cap exceeded / OOB) or
63 * ZIP_K_EIO.
64 */
65int zip_locate_cd(zip_pread_fn pread, void *ctx, uint64_t archive_size,
66 uint64_t cd_max, uint64_t max_entries,
67 uint64_t *cd_off, uint64_t *cd_size, uint64_t *n_entries);
68
69/*
70 * Apply a CDFH's ZIP64 extended-information extra field: override each of
71 * *uncomp / *comp / *lhdr_off, in ZIP spec order, but only for the fields whose
72 * classic value is saturated (0xFFFFFFFF). A malformed extra blob stops parsing
73 * (leaving the not-yet-overridden values unchanged). `extra` need not be NUL-
74 * terminated; `extra_len` bounds it.
75 */
76void zip_apply_zip64_extra(const uint8_t *extra, size_t extra_len,
77 uint64_t *uncomp, uint64_t *comp, uint64_t *lhdr_off);
78
79/*
80 * Resolve a member's first data byte by reading its Local File Header (whose own
81 * filename/extra lengths can differ from the CDFH's). Validates the LFH
82 * signature and that [data_off, data_off+comp_size) lies within the archive. On
83 * success stores the offset in *data_off. Returns ZIP_K_OK, ZIP_K_ECORRUPT (bad
84 * signature / out of bounds) or ZIP_K_EIO.
85 */
86int zip_resolve_data_off(zip_pread_fn pread, void *ctx, uint64_t archive_size,
87 uint64_t lhdr_off, uint64_t comp_size,
88 uint64_t *data_off);
