4/*
5 * af_policy.h — outbound address-family policy for cache/proxy origin connects.
6 *
7 * WHAT: a 3-value policy (auto/inet/inet6) plus a string parser. The enum values
8 * ARE the AF_* constants, so a policy assigns straight into a getaddrinfo
9 * hints.ai_family with no mapping table.
10 * WHY: a dual-stack/IPv6 cache node must be able to reach an IPv4-only or
11 * IPv6-only origin. Constraining the family of the OUTBOUND resolve is the whole
12 * mechanism — the listen side stays dual-stack via nginx `listen`.
13 * HOW: header-only, pure, no nginx deps, so it is shared by the directive handler,
14 * the resolver, and a standalone unit test.
15 */
16

31 if (len == 4 && memcmp(s, "auto", 4) == 0) { return BRIX_AF_AUTO; }
32 if (len == 4 && memcmp(s, "inet", 4) == 0) { return BRIX_AF_INET; }
