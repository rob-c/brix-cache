1/* cksum_manifest.c — parse/format the tree-audit manifest line format.
2 *
3 * WHAT: A single parser for the sha256sum-style manifest line format used by
4 * `xrdcksum tree` (writer) and `xrdcksum check` (reader):
5 * "<hex> <rel>\n" (two spaces — sha256sum -c compatible)
6 * WHY: A hostile or corrupt manifest must never direct reads or comparisons
7 * outside the audit root. Centralising the parse + safety gate keeps the
8 * security invariant in one place rather than duplicated across two
9 * subcommands.
10 * HOW: strstr for the mandatory two-space separator; isxdigit for every digit
11 * character; brix_rel_is_unsafe for the path guard (absolute / "..").
12 * Oversized fields are rejected before writing so callers need no truncation
13 * guard. */

15#include "brix_ops.h"
16#include <ctype.h>

19/*
20 * brix_ckmf_parse_line — parse one manifest line into hex + rel.
21 *
22 * WHAT: Splits "<hex> <rel>\n" at the two-space separator, strips the trailing
23 * newline/CRLF, and validates both halves before writing them.
24 * WHY: All callers must use this gate; the security contract is that no rel path
25 * returned here can ever escape the audit root directory.
26 * HOW: strstr locates the separator; isxdigit validates each hex character;
27 * brix_rel_is_unsafe rejects absolute paths and ".." components. Every
28 * error path returns -1 without modifying the output buffers.
29 *
30 * Returns 0 with hex[hexsz] and rel[relsz] filled (NUL-terminated), or -1 when:
31 * - line is NULL or empty
32 * - no " " (double-space) separator found, or hex part is empty
33 * - hex part contains a non-hex character
34 * - hex or rel part would not fit in the supplied buffer (overflow guard)
35 * - rel part is empty after stripping the newline
36 * - brix_rel_is_unsafe(rel) is true (absolute path or ".." escape)
37 */
38int
39brix_ckmf_parse_line(const char *line, char *hex, size_t hexsz,
40 char *rel, size_t relsz)
41{
42 const char *sep;
43 const char *r;
44 size_t hl, rl;
45
46 if (line == NULL) {
47 return -1;
48 }
49 sep = strstr(line, " ");
50 if (sep == NULL || sep == line) {
51 return -1;
52 }
53 hl = (size_t)(sep - line);
54 if (hl >= hexsz) {
55 return -1;
56 }
57 for (r = line; r < sep; r++) {
58 if (!isxdigit((unsigned char)*r)) {
59 return -1;
60 }
61 }
62 /* rel starts two characters after the separator; strip trailing newline. */
63 r = sep + 2;
64 rl = strlen(r);
65 while (rl > 0 && (r[rl - 1] == '\n' || r[rl - 1] == '\r')) {
66 rl--;
67 }
68 if (rl == 0 || rl >= relsz) {
69 return -1;
70 }
71 /* Guard: reject any path that escapes the audit root. */
72 /* Copy rel to a temporary NUL-terminated string for brix_rel_is_unsafe. */
73 memcpy(rel, r, rl);
74 rel[rl] = '\0';
75 if (brix_rel_is_unsafe(rel)) {
76 return -1;
77 }
78 memcpy(hex, line, hl);
79 hex[hl] = '\0';
80 return 0;
81}
