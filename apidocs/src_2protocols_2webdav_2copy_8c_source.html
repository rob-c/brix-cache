20/*
21 * Map a VFS single-file-copy errno to the same HTTP status the prior
22 * brix_ns_local_copy → status mapping produced. brix_vfs_copy() returns
23 * NGX_ERROR with errno set (the namespace sys_errno), so reconstruct the
24 * namespace status the same way namespace_ops' errno_to_ns_status does and feed
25 * it to brix_http_map_ns_status — guaranteeing byte-for-byte parity with the
26 * old code (including DENIED→403, TOO_LONG→414, NO_SPACE→507, CONFLICT→409,
27 * IO_ERROR→500). The two COPY-specific overrides (EXISTS→412, NOT_FOUND→409)
28 * are applied by the caller before this helper is reached.
29 */
30static ngx_int_t
31webdav_copy_errno_to_status(int err)
32{
33 brix_ns_status_t status;
34
35 switch (err) {
36 case 0: status = BRIX_NS_OK; break;
37 case ENOENT: status = BRIX_NS_NOT_FOUND; break;
38 case EACCES:
39 case EPERM:
40 case EXDEV:
41 case ELOOP: status = BRIX_NS_DENIED; break;
42 case EEXIST: status = BRIX_NS_EXISTS; break;
43 case ENOTEMPTY: status = BRIX_NS_NOT_EMPTY; break;
44 case ENAMETOOLONG: status = BRIX_NS_TOO_LONG; break;
45 case ENOSPC: status = BRIX_NS_NO_SPACE; break;
46 case EBUSY:
47 case EINVAL: status = BRIX_NS_CONFLICT; break;
48 default: status = BRIX_NS_IO_ERROR; break;
49 }
50
51 return brix_http_map_ns_status(status);
52}

67/*
68 * Copy a collection (directory) atomically.
69 * Strategy: build the whole copy in a sibling temp directory, then rename it
70 * into place — so a concurrent reader never sees a half-copied tree.
71 * 1. mkdir temp dir (ENOENT on the parent -> 409 Conflict);
72 * 2. copy dead props + fattrs, then (depth:infinity) recurse the contents;
73 * 3. if the destination already existed as a dir, remove it (rename can't
74 * replace a non-empty dir);
75 * 4. rename temp -> dst.
76 * Every failure after the temp dir exists removes it, so no orphan is left.
77 * Returns 201/204 on success or an HTTP error status. May run on a worker
78 * thread (see webdav_copy_collection_thread), so it uses only its parameters
79 * and thread-safe confined-FS helpers — no ngx_http_request_t access.
80 */
81static ngx_int_t
82webdav_copy_collection_execute(ngx_log_t *log, const char *root_canon,
83 const char *src_path, const char *dst_path, mode_t src_mode,
84 ngx_flag_t dst_existed, ngx_flag_t dst_was_dir,
85 ngx_flag_t depth_infinity)
86{
87 char tmp_path[WEBDAV_MAX_PATH];
88 ngx_int_t rc;
89
90 if (brix_make_tmp_path(dst_path, tmp_path, sizeof(tmp_path)) != NGX_OK) {
91 return NGX_HTTP_REQUEST_URI_TOO_LARGE;
92 }
93
94 if (brix_vfs_mkdir_path(log, root_canon, tmp_path, src_mode & 0777) != 0) {
95 if (errno == ENOENT) {
96 return NGX_HTTP_CONFLICT;
97 }
98 return NGX_HTTP_INTERNAL_SERVER_ERROR;
99 }
100
101 webdav_dead_props_copy(log, src_path, tmp_path);
102 brix_ns_copy_fattrs(log, src_path, tmp_path);
103
104 if (depth_infinity) {
105 rc = webdav_copy_dir_recursive(log, root_canon, src_path, tmp_path);
106 } else {
107 rc = NGX_OK;
108 }
109
110 if (rc != NGX_OK) {
111 (void) webdav_delete_path_recursive(log, root_canon, tmp_path);
112 return NGX_HTTP_INTERNAL_SERVER_ERROR;
113 }
114
115 if (dst_existed && dst_was_dir) {
116 if (webdav_delete_path_recursive(log, root_canon, dst_path) != NGX_OK) {
117 (void) webdav_delete_path_recursive(log, root_canon, tmp_path);
118 return NGX_HTTP_INTERNAL_SERVER_ERROR;
119 }
120 }
121
122 if (brix_rename_confined_canon(log, root_canon, tmp_path, dst_path) != 0) {
123 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
124 "brix_webdav COPY: dir rename failed: \"%s\"",
125 dst_path);
126 (void) webdav_delete_path_recursive(log, root_canon, tmp_path);
127 return NGX_HTTP_INTERNAL_SERVER_ERROR;
128 }
129
130 return dst_existed ? NGX_HTTP_NO_CONTENT : NGX_HTTP_CREATED;
131}

174/*
175 * Try to offload a collection COPY to the thread pool.
176 * Returns NGX_DONE if queued (request kept alive via r->main->count++),
177 * NGX_DECLINED if no thread pool is available (caller runs it synchronously),
178 * NGX_ERROR on allocation/post failure. Stack-buffer paths are copied into the
179 * task context so they remain valid after this returns.
180 */
181static ngx_int_t
182webdav_copy_collection_post_task(ngx_http_request_t *r,
183 ngx_http_brix_webdav_loc_conf_t *conf, const char *src_path,
184 const char *dst_path, mode_t src_mode, ngx_flag_t dst_existed,
185 ngx_flag_t dst_was_dir, ngx_flag_t depth_infinity)
186{
187 ngx_thread_task_t *task;
188 webdav_copy_collection_task_t *t;
189 ngx_thread_pool_t *pool;
190
191 /*
192 * Under impersonation the per-worker broker socket is a single fd shared by
193 * the event-loop thread; a thread-pool task issuing confined ops would use
194 * it concurrently and corrupt the request/reply framing, wedging the whole
195 * worker's broker channel. The thread also lacks the per-worker principal.
196 * Force the synchronous (NGX_DECLINED) path so the recursive copy runs on
197 * the event-loop thread where the principal is set and the broker socket
198 * has exactly one user.
199 */
200 if (brix_imp_enabled()) {
201 return NGX_DECLINED;
202 }
203
204 /* postconfig only visits server-level loc_conf; resolve lazily for nested
205 * location blocks where conf->common.thread_pool may still be NULL. */
206 pool = conf->common.thread_pool;
207 if (pool == NULL) {
208 static ngx_str_t default_name = ngx_string("default");
209 ngx_str_t *pname = conf->common.thread_pool_name.len > 0
210 ? &conf->common.thread_pool_name : &default_name;
211 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
212 if (pool != NULL) {
213 conf->common.thread_pool = pool;
214 }
215 }
216 if (pool == NULL) {
217 return NGX_DECLINED;
218 }
219
220 task = ngx_thread_task_alloc(r->pool, sizeof(webdav_copy_collection_task_t));
221 if (task == NULL) {
222 return NGX_ERROR;
223 }
224
225 t = task->ctx;
226 t->r = r;
227 t->log = r->connection->log;
228 t->src_mode = src_mode;
229 t->dst_existed = dst_existed;
230 t->dst_was_dir = dst_was_dir;
231 t->depth_infinity = depth_infinity;
232 t->http_status = NGX_HTTP_INTERNAL_SERVER_ERROR;
233
234 ngx_cpystrn((u_char *) t->root_canon, (u_char *) conf->common.root_canon,
235 sizeof(t->root_canon));
236 ngx_cpystrn((u_char *) t->src_path, (u_char *) src_path,
237 sizeof(t->src_path));
238 ngx_cpystrn((u_char *) t->dst_path, (u_char *) dst_path,
239 sizeof(t->dst_path));
240
241 brix_task_bind(task, webdav_copy_collection_thread, webdav_copy_collection_done);
242 task->event.log = r->connection->log;
243
244 if (ngx_thread_task_post(pool, task) != NGX_OK) {
245 return NGX_ERROR;
246 }
247
248 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
249 "brix_webdav: offloaded collection COPY to thread pool");
250
251 r->main->count++;
252 return NGX_DONE;
253}

255/*
256 * webdav_copy_probe — confined stat of `path` (follow semantics, matching the
257 * prior brix_lstat_confined_canon nofollow=0) through the VFS probe, projected
258 * into the struct stat fields the COPY handler needs (ino/dev for the self-copy
259 * guard, mode for is-dir, mtime/size for the conditional checks). Non-metered
260 * (the COPY op accounts for itself). Returns NGX_OK / NGX_DECLINED (errno kept).
261 */
262static ngx_int_t
263webdav_copy_probe(ngx_http_request_t *r, const char *path, struct stat *sb)
264{
265 ngx_http_brix_webdav_loc_conf_t *conf =
266 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
267 ngx_http_brix_webdav_req_ctx_t *rx =
268 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
269 brix_vfs_ctx_t vctx;
270 brix_vfs_stat_t vst;
271 int is_tls = 0;
272
273#if (NGX_HTTP_SSL)
274 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
275#endif
276
277 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
278 conf->common.root_canon, conf->cache_root_canon, conf->common.allow_write,
279 is_tls, (rx != NULL) ? rx->identity : NULL, path);
280
281 if (brix_vfs_probe(&vctx, 0 /* follow */, &vst) != NGX_OK) {
282 return NGX_DECLINED;
283 }
284
285 ngx_memzero(sb, sizeof(*sb));
286 sb->st_mode = (mode_t) vst.mode;
287 sb->st_size = vst.size;
288 sb->st_mtime = vst.mtime;
289 sb->st_ino = vst.ino;
290 sb->st_dev = vst.dev;
291 return NGX_OK;

294/*
295 *
296 * WHAT: Implements RFC 4918 §9.8 WebDAV COPY operation for server-side file/directory duplication within the same export root. Orchestrates the complete copy lifecycle: parses Destination/Overwrite/Depth headers, resolves both source and destination paths under root confinement, validates locks on destination, handles conditional checks (If-Match/If-None-Match), performs atomic copy via an intermediate staged temp path, then renames to final destination. Returns 201 Created when target didn't exist or 204 No Content when replacing existing resource; cleanup on failure ensures no orphaned temp files remain.
297 *
298 * WHY: WebDAV clients need an HTTP endpoint that can duplicate resources within the same filesystem namespace — unlike xrdcp native TPC which transfers between different servers, this COPY is local-only (same root). The tmp_path atomic rename strategy prevents partial copies from being visible to other clients; if copy fails the temp file is immediately deleted. Lock validation ensures no concurrent modifications interfere with the operation. Depth=0 means single-item copy; Depth=infinity recursively copies subdirectories (delegated to webdav_copy_dir_recursive in fs/copy_engine.c).
299 *
300 * HOW: Extracts and decodes Destination URI from request headers (strips scheme prefix to get filesystem path); resolves source via resolve_path() with root_canon confinement; resolves destination via webdav_resolve_destination_path(); checks both paths exist via stat(); validates locks on destination (including recursive child locks if depth=infinity); compares inode/dev numbers to reject self-copy attempt; applies Overwrite header semantics (F=don't overwrite, default=T=allow overwrite); runs copy_conditionals() for If-Match/If-None-Match; creates a staged temp file for file copies or a temp directory for collection copies; copies data via webdav_copy_file() or webdav_copy_dir_recursive(); commits temp→dst via confined rename. */
301ngx_int_t
302webdav_handle_copy(ngx_http_request_t *r)
303{
304 ngx_http_brix_webdav_loc_conf_t *conf;
305 ngx_table_elt_t *dest_hdr;
306 ngx_table_elt_t *depth_hdr;
307 char src_path[WEBDAV_MAX_PATH];
308 char dst_path[WEBDAV_MAX_PATH];
309 char dest_decoded[WEBDAV_MAX_PATH];
310 const u_char *dest_path_start;
311 size_t dest_path_len;
312 struct stat src_sb;
313 struct stat dst_sb;
314 ngx_int_t rc;
315 int overwrite;
316 int dst_existed;
317 int depth_infinity = 1;
318
319 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
320
321 dest_hdr = webdav_tpc_find_header(r, "Destination",
322 sizeof("Destination") - 1);
323 if (dest_hdr == NULL) {
324 return NGX_HTTP_BAD_REQUEST;
325 }
326
327 overwrite = !brix_http_overwrite_forbidden(r);
328
329 depth_hdr = webdav_tpc_find_header(r, "Depth", sizeof("Depth") - 1);
330 if (depth_hdr != NULL) {
331 if (depth_hdr->value.len == 1 && depth_hdr->value.data[0] == '0') {
332 depth_infinity = 0;
333 }
334 }
335
336 dest_path_start = dest_hdr->value.data;
337 dest_path_len = dest_hdr->value.len;
338 rc = webdav_destination_extract_path(dest_path_start, dest_path_len,
339 &dest_path_start, &dest_path_len);
340 if (rc != NGX_OK) {
341 return rc;
342 }
343
344 rc = webdav_urldecode(dest_path_start, dest_path_len,
345 dest_decoded, sizeof(dest_decoded));
346 if (rc != NGX_OK) {
347 return rc;
348 }
349
350 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
351 src_path, sizeof(src_path));
352 if (rc != NGX_OK) {
353 return rc;
354 }
355
356 if (webdav_copy_probe(r, src_path, &src_sb) != NGX_OK) {
357 return (errno == ENOENT) ? NGX_HTTP_NOT_FOUND
358 : NGX_HTTP_INTERNAL_SERVER_ERROR;
359 }
360
361 rc = webdav_resolve_destination_path(r->connection->log, "COPY",
362 conf->common.root_canon, dest_decoded,
363 dst_path, sizeof(dst_path));
364 if (rc != NGX_OK) {
365 return rc;
366 }
367
368 dst_existed = (webdav_copy_probe(r, dst_path, &dst_sb) == NGX_OK);
369
370 rc = webdav_check_locks_tree(r, dst_path);
371 if (rc != NGX_OK) {
372 return rc;
373 }
374
375 /* Reject copy-onto-self: same (dev, ino) means source and destination are
376 * the same file, which would corrupt/truncate it. RFC 4918 §9.8.5 -> 403. */
377 if (dst_existed
378 && src_sb.st_ino == dst_sb.st_ino
379 && src_sb.st_dev == dst_sb.st_dev)
380 {
381 return NGX_HTTP_FORBIDDEN;
382 }
383
384 if (dst_existed && !overwrite) {
385 return NGX_HTTP_PRECONDITION_FAILED;
386 }
387
388 rc = webdav_check_copy_conditionals(r, dst_path, dst_existed, &dst_sb);
389 if (rc != NGX_OK) {
390 return rc;
391 }
392
393 if (S_ISDIR(src_sb.st_mode)) {
394 /* Collection copy. For depth:infinity (potentially large), try to
395 * offload to a thread first: NGX_DONE means queued (we are finished
396 * here), NGX_ERROR is fatal, and NGX_DECLINED (no thread pool) falls
397 * through to the synchronous execute below. Depth:0 always runs
398 * synchronously since it only creates the top directory. */
399 if (depth_infinity) {
400 rc = webdav_copy_collection_post_task(r, conf, src_path, dst_path,
401 src_sb.st_mode, dst_existed,
402 dst_existed
403 && S_ISDIR(dst_sb.st_mode),
404 depth_infinity);
405 if (rc == NGX_DONE) {
406 return NGX_DONE;
407 }
408 if (rc == NGX_ERROR) {
409 return NGX_HTTP_INTERNAL_SERVER_ERROR;
410 }
411 /* rc == NGX_DECLINED: no thread pool, fall through to run inline. */
412 }
413
414 rc = webdav_copy_collection_execute(r->connection->log,
415 conf->common.root_canon, src_path,
416 dst_path, src_sb.st_mode,
417 dst_existed,
418 dst_existed
419 && S_ISDIR(dst_sb.st_mode),
420 depth_infinity);
421 if (rc != NGX_HTTP_CREATED && rc != NGX_HTTP_NO_CONTENT) {
422 return rc;
423 }
424 } else {
425 /* File COPY: route the data move through the metered VFS copy surface
426 * (delegates to the same namespace_ops local-copy service underneath).
427 * Pre-delete destination directory if overwrite is enabled; rename(2)
428 * cannot atomically replace a directory with a file. */
429 brix_vfs_copy_opts_t copy_opts;
430 brix_vfs_ctx_t vctx;
431
432 if (dst_existed && S_ISDIR(dst_sb.st_mode)) {
433 (void) webdav_delete_path_recursive(r->connection->log,
434 conf->common.root_canon, dst_path);
435 }
436
437 {
438 ngx_http_brix_webdav_req_ctx_t *wctx =
439 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
440 int is_tls = 0;
441#if (NGX_HTTP_SSL)
442 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
443#endif
444 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log,
445 BRIX_PROTO_WEBDAV, conf->common.root_canon,
446 conf->cache_root_canon, conf->common.allow_write, is_tls,
447 (wctx != NULL) ? wctx->identity : NULL, src_path);
448 }
449
450 ngx_memzero(&copy_opts, sizeof(copy_opts));
451 copy_opts.overwrite = overwrite ? 1 : 0;
452 copy_opts.preserve_xattrs = 1;
453 copy_opts.staged_commit = 1;
454
455 if (brix_vfs_copy(&vctx, dst_path, &copy_opts) != NGX_OK) {
456 /* COPY-specific RFC 4918 semantics that differ from the generic
457 * namespace→HTTP mapping: an existing dst with Overwrite:F is a
458 * precondition failure (412, not 409), and a missing destination
459 * parent is a Conflict (409, not 404). brix_vfs_copy reports the
460 * namespace failure via errno. */
461 int err = errno;
462 if (err == EEXIST) {
463 return NGX_HTTP_PRECONDITION_FAILED;
464 }
465 if (err == ENOENT) {
466 return NGX_HTTP_CONFLICT;
467 }
468 /* Everything else (DENIED→403, TOO_LONG→414, NO_SPACE→507,
469 * IO_ERROR→500) goes through the canonical mapper so a DAC denial
470 * — e.g. an impersonated cross-tenant copy into a 0700 dir — is a
471 * clean 403, not a blanket 500. */
472 return webdav_copy_errno_to_status(err);
473 }
474
475 webdav_dead_props_copy(r->connection->log, src_path, dst_path);
476 }
477
478 return webdav_send_no_body(r, dst_existed ? NGX_HTTP_NO_CONTENT
479 : NGX_HTTP_CREATED);
480}
