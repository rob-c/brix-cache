21/*
22 * Map a VFS single-file-copy errno to the same HTTP status the prior
23 * brix_ns_local_copy → status mapping produced. brix_vfs_copy() returns
24 * NGX_ERROR with errno set (the namespace sys_errno), so reconstruct the
25 * namespace status the same way namespace_ops' errno_to_ns_status does and feed
26 * it to brix_http_map_ns_status — guaranteeing byte-for-byte parity with the
27 * old code (including DENIED→403, TOO_LONG→414, NO_SPACE→507, CONFLICT→409,
28 * IO_ERROR→500). The two COPY-specific overrides (EXISTS→412, NOT_FOUND→409)
29 * are applied by the caller before this helper is reached.
30 */
31static ngx_int_t
32webdav_copy_errno_to_status(int err)
33{
34 brix_ns_status_t status;
35
36 switch (err) {
37 case 0: status = BRIX_NS_OK; break;
38 case ENOENT: status = BRIX_NS_NOT_FOUND; break;
39 case EACCES:
40 case EPERM:
41 case EXDEV:
42 case ELOOP: status = BRIX_NS_DENIED; break;
43 case EEXIST: status = BRIX_NS_EXISTS; break;
44 case ENOTEMPTY: status = BRIX_NS_NOT_EMPTY; break;
45 case ENAMETOOLONG: status = BRIX_NS_TOO_LONG; break;
46 case ENOSPC: status = BRIX_NS_NO_SPACE; break;
47#ifdef EDQUOT
48 case EDQUOT: status = BRIX_NS_NO_SPACE; break;
49#endif
50 case EBUSY:
51 case EINVAL: status = BRIX_NS_CONFLICT; break;
52 default: status = BRIX_NS_IO_ERROR; break;
53 }
54
55 return brix_http_map_ns_status(status);
56}

58/*
59 * webdav_copy_probe — confined stat of `path` (follow semantics, matching the
60 * prior brix_lstat_confined_canon nofollow=0) through the VFS probe, projected
61 * into the struct stat fields the COPY handler needs (ino/dev for the self-copy
62 * guard, mode for is-dir, mtime/size for the conditional checks). Non-metered
63 * (the COPY op accounts for itself). Returns NGX_OK / NGX_DECLINED (errno kept).
64 */
65static ngx_int_t
66webdav_copy_probe(ngx_http_request_t *r, const char *path, struct stat *sb)
67{
68 ngx_http_brix_webdav_loc_conf_t *conf =
69 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
70 ngx_http_brix_webdav_req_ctx_t *rx =
71 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
72 brix_vfs_ctx_t vctx;
73 brix_vfs_stat_t vst;
74 int is_tls = 0;
75
76#if (NGX_HTTP_SSL)
77 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
78#endif
79
80 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
81 conf->common.root_canon, conf->cache_root_canon, conf->common.allow_write,
82 is_tls, (rx != NULL) ? rx->identity : NULL, path);
83 /* Bind the export's per-user backend credential policy so this pre-copy
84 * probe (existence/self-copy/overwrite check) enforces deny for the
85 * REQUESTING USER on a remote-backed export, before any copy attempt. */
86 brix_vfs_ctx_bind_backend_cred(&vctx,
87 &conf->common.storage_credential_dir,
88 conf->common.storage_credential_fallback);
89 webdav_vfs_bind_deleg(r, conf, &vctx);
90
91 if (brix_vfs_probe(&vctx, 0 /* follow */, &vst) != NGX_OK) {
92 return NGX_DECLINED;
93 }
94
95 ngx_memzero(sb, sizeof(*sb));
96 sb->st_mode = (mode_t) vst.mode;
97 sb->st_size = vst.size;
98 sb->st_mtime = vst.mtime;
99 sb->st_ino = vst.ino;
100 sb->st_dev = vst.dev;
101 return NGX_OK;
102}

104/*
105 * WHAT: Parse the COPY request headers into req: decode the Destination URI into
106 * req->dest_decoded and set req->overwrite and req->depth_infinity.
107 *
108 * WHY: Header extraction/decoding is a self-contained first phase; isolating it
109 * keeps the handler a flat sequence of resolve → validate → execute.
110 *
111 * HOW: Requires a Destination header (400 if absent); Overwrite:F clears
112 * overwrite; Depth:"0" selects depth:0 (default is infinity). Extracts the path
113 * component of the Destination and URL-decodes it, propagating any helper error
114 * status. Returns NGX_OK on success.
115 */
116static ngx_int_t
117webdav_copy_parse_request(ngx_http_request_t *r, webdav_copy_req_t *req)
118{
119 ngx_table_elt_t *dest_hdr;
120 ngx_table_elt_t *depth_hdr;
121 const u_char *dest_path_start;
122 size_t dest_path_len;
123 ngx_int_t rc;
124
125 dest_hdr = webdav_tpc_find_header(r, "Destination",
126 sizeof("Destination") - 1);
127 if (dest_hdr == NULL) {
128 return NGX_HTTP_BAD_REQUEST;
129 }
130
131 req->overwrite = !brix_http_overwrite_forbidden(r);
132
133 depth_hdr = webdav_tpc_find_header(r, "Depth", sizeof("Depth") - 1);
134 if (depth_hdr != NULL
135 && depth_hdr->value.len == 1 && depth_hdr->value.data[0] == '0')
136 {
137 req->depth_infinity = 0;
138 }
139
140 dest_path_start = dest_hdr->value.data;
141 dest_path_len = dest_hdr->value.len;
142 rc = webdav_destination_extract_path(dest_path_start, dest_path_len,
143 &dest_path_start, &dest_path_len);
144 if (rc != NGX_OK) {
145 return rc;
146 }
147
148 return webdav_urldecode(dest_path_start, dest_path_len,
149 req->dest_decoded, sizeof(req->dest_decoded));
150}

152/*
153 * WHAT: Resolve and validate the source/destination pair: fill req->src_path,
154 * req->dst_path, their stat records, and req->dst_existed, then run the lock,
155 * self-copy, overwrite and conditional checks.
156 *
157 * WHY: This is the security/precondition gate between parsing and executing the
158 * copy — it enforces root confinement, INVARIANT 5's recursive child-lock check
159 * on the destination tree, the RFC 4918 §9.8.5 self-copy rejection, the
160 * Overwrite:F precondition, and If-Match/If-None-Match. Keeping it in one helper
161 * makes the deny paths auditable in isolation.
162 *
163 * HOW: Resolves the source under root_canon and probes it (ENOENT→404, else
164 * 500); resolves the destination and probes it (existence recorded, not
165 * required); runs webdav_check_locks_tree; rejects a same-(dev,ino) self-copy
166 * with 403; a pre-existing destination with Overwrite:F with 412; then the COPY
167 * conditionals. Returns NGX_OK when the copy may proceed, else the HTTP status.
168 */
169static ngx_int_t
170webdav_copy_resolve_pair(ngx_http_request_t *r,
171 ngx_http_brix_webdav_loc_conf_t *conf, webdav_copy_req_t *req)
172{
173 ngx_int_t rc;
174
175 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
176 req->src_path, sizeof(req->src_path));
177 if (rc != NGX_OK) {
178 return rc;
179 }
180
181 if (webdav_copy_probe(r, req->src_path, &req->src_sb) != NGX_OK) {
182 return (errno == ENOENT) ? NGX_HTTP_NOT_FOUND
183 : NGX_HTTP_INTERNAL_SERVER_ERROR;
184 }
185
186 rc = webdav_resolve_destination_path(r->connection->log, "COPY",
187 conf->common.root_canon,
188 req->dest_decoded, req->dst_path,
189 sizeof(req->dst_path),
190 conf->common.cache_store_endpoint);
191 if (rc != NGX_OK) {
192 return rc;
193 }
194
195 req->dst_existed = (webdav_copy_probe(r, req->dst_path, &req->dst_sb)
196 == NGX_OK);
197
198 rc = webdav_check_locks_tree(r, req->dst_path);
199 if (rc != NGX_OK) {
200 return rc;
201 }
202
203 /* Reject copy-onto-self: same (dev, ino) means source and destination are
204 * the same file, which would corrupt/truncate it. RFC 4918 §9.8.5 -> 403. */
205 if (req->dst_existed
206 && req->src_sb.st_ino == req->dst_sb.st_ino
207 && req->src_sb.st_dev == req->dst_sb.st_dev)
208 {
209 return NGX_HTTP_FORBIDDEN;
210 }
211
212 if (req->dst_existed && !req->overwrite) {
213 return NGX_HTTP_PRECONDITION_FAILED;
214 }
215
216 return webdav_check_copy_conditionals(r, req->dst_path, req->dst_existed,
217 &req->dst_sb);
218}

259/*
260 * WHAT: Execute a single-file COPY for a resolved request through the metered
261 * VFS copy surface.
262 *
263 * WHY: Isolates the file branch of the handler, including the pre-delete of a
264 * destination directory (rename(2) cannot atomically replace a directory with a
265 * file) and the COPY-specific errno→HTTP mapping.
266 *
267 * HOW: Pre-deletes a pre-existing destination directory; builds the VFS context;
268 * runs brix_vfs_copy with staged-commit + xattr preservation. On failure it maps
269 * EEXIST→412 and ENOENT→409 (RFC 4918 overrides), everything else through the
270 * canonical mapper. On success it clones dead props and returns NGX_OK.
271 */
272static ngx_int_t
273webdav_copy_do_file(ngx_http_request_t *r,
274 ngx_http_brix_webdav_loc_conf_t *conf, const webdav_copy_req_t *req)
275{
276 brix_vfs_copy_opts_t copy_opts;
277 brix_vfs_ctx_t vctx;
278
279 if (req->dst_existed && S_ISDIR(req->dst_sb.st_mode)) {
280 (void) webdav_delete_path_recursive(r->connection->log,
281 conf->common.root_canon,
282 req->dst_path);
283 }
284
285 webdav_copy_file_vfs_init(r, conf, req->src_path, &vctx);
286
287 ngx_memzero(&copy_opts, sizeof(copy_opts));
288 copy_opts.overwrite = req->overwrite ? 1 : 0;
289 copy_opts.preserve_xattrs = 1;
290 copy_opts.staged_commit = 1;
291
292 if (brix_vfs_copy(&vctx, req->dst_path, &copy_opts) != NGX_OK) {
293 /* COPY-specific RFC 4918 semantics that differ from the generic
294 * namespace→HTTP mapping: an existing dst with Overwrite:F is a
295 * precondition failure (412, not 409), and a missing destination
296 * parent is a Conflict (409, not 404). brix_vfs_copy reports the
297 * namespace failure via errno. */
298 int err = errno;
299 if (err == EEXIST) {
300 return NGX_HTTP_PRECONDITION_FAILED;
301 }
302 if (err == ENOENT) {
303 return NGX_HTTP_CONFLICT;
304 }
305 /* Everything else (DENIED→403, TOO_LONG→414, NO_SPACE→507,
306 * IO_ERROR→500) goes through the canonical mapper so a DAC denial
307 * — e.g. an impersonated cross-tenant copy into a 0700 dir — is a
308 * clean 403, not a blanket 500. */
309 return webdav_copy_errno_to_status(err);
310 }
311
312 webdav_dead_props_copy(r->connection->log, req->src_path, req->dst_path);
313 return NGX_OK;
314}

316/*
317 *
318 * WHAT: Implements RFC 4918 §9.8 WebDAV COPY operation for server-side file/directory duplication within the same export root. Orchestrates the complete copy lifecycle: parses Destination/Overwrite/Depth headers, resolves both source and destination paths under root confinement, validates locks on destination, handles conditional checks (If-Match/If-None-Match), performs atomic copy via an intermediate staged temp path, then renames to final destination. Returns 201 Created when target didn't exist or 204 No Content when replacing existing resource; cleanup on failure ensures no orphaned temp files remain.
319 *
320 * WHY: WebDAV clients need an HTTP endpoint that can duplicate resources within the same filesystem namespace — unlike xrdcp native TPC which transfers between different servers, this COPY is local-only (same root). The tmp_path atomic rename strategy prevents partial copies from being visible to other clients; if copy fails the temp file is immediately deleted. Lock validation ensures no concurrent modifications interfere with the operation. Depth=0 means single-item copy; Depth=infinity recursively copies subdirectories (delegated to webdav_copy_dir_recursive in fs/copy_engine.c).
321 *
322 * HOW: Extracts and decodes Destination URI from request headers (strips scheme prefix to get filesystem path); resolves source via resolve_path() with root_canon confinement; resolves destination via webdav_resolve_destination_path(); checks both paths exist via stat(); validates locks on destination (including recursive child locks if depth=infinity); compares inode/dev numbers to reject self-copy attempt; applies Overwrite header semantics (F=don't overwrite, default=T=allow overwrite); runs copy_conditionals() for If-Match/If-None-Match; creates a staged temp file for file copies or a temp directory for collection copies; copies data via webdav_copy_file() or webdav_copy_dir_recursive(); commits temp→dst via confined rename. */
323ngx_int_t
324webdav_handle_copy(ngx_http_request_t *r)
325{
326 ngx_http_brix_webdav_loc_conf_t *conf;
327 webdav_copy_req_t req;
328 ngx_int_t rc;
329
330 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
331
332 ngx_memzero(&req, sizeof(req));
333 req.depth_infinity = 1;
334
335 rc = webdav_copy_parse_request(r, &req);
336 if (rc != NGX_OK) {
337 return rc;
338 }
339
340 rc = webdav_copy_resolve_pair(r, conf, &req);
341 if (rc != NGX_OK) {
342 return rc;
343 }
344
345 if (S_ISDIR(req.src_sb.st_mode)) {
346 rc = webdav_copy_do_collection(r, conf, &req);
347 } else {
348 rc = webdav_copy_do_file(r, conf, &req);
349 }
350
351 /* NGX_DONE = collection copy offloaded to a thread (response sent later);
352 * any non-OK status is a failure to surface now. */
353 if (rc != NGX_OK) {
354 return rc;
355 }
356
357 return webdav_send_no_body(r, req.dst_existed ? NGX_HTTP_NO_CONTENT
358 : NGX_HTTP_CREATED);
359}
