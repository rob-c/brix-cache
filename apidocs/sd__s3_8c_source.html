1/*
2 * sd_s3.c — shared S3 storage driver, read path. See sd_s3.h.
3 *
4 * Ported from client/lib/vfs_s3*.c with two changes that make it shareable:
5 * 1. the HTTP transport is injected (sd_s3_transport.h) instead of calling the
6 * client's xrdc_http_* directly;
7 * 2. the error model is a neutral (rc, errbuf) pair instead of xrdc_status.
8 * SigV4 signing uses the same shared kernels (sigv4 / crypto / hex / uri) the
9 * server's verify path uses, so client-signs == server-verifies byte-for-byte.
10 *
11 * The SigV4 signing + error-mapping primitives live in sd_s3_sign.c; the
12 * single-PUT / multipart write path lives in sd_s3_write.c. Both share the
13 * sd_s3_file layout via sd_s3_internal.h.
14 */

34sd_s3_file *
35sd_s3_open_read(const sd_s3_open_params *p, char *errbuf, size_t errcap)
36{
37 sd_s3_file *f;
38
39 if (p == NULL || p->host == NULL || p->key == NULL || p->transport == NULL) {
40 sd_s3_set_err(errbuf, errcap, "s3 open: bad parameters");
41 return NULL;
42 }
43 f = calloc(1, sizeof(*f));
44 if (f == NULL) {
45 sd_s3_set_err(errbuf, errcap, "s3 open: out of memory");
46 return NULL;
47 }
48 snprintf(f->host, sizeof(f->host), "%s", p->host);
49 snprintf(f->key, sizeof(f->key), "%s", p->key);
50 snprintf(f->ak, sizeof(f->ak), "%s", p->ak ? p->ak : "");
51 snprintf(f->sk, sizeof(f->sk), "%s", p->sk ? p->sk : "");
52 snprintf(f->region, sizeof(f->region), "%s",
53 (p->region && p->region[0]) ? p->region : "us-east-1");
54 f->port = p->port;
55 f->tls = p->tls;
56 f->transport = p->transport;
57 f->tctx = p->tctx;
58 f->timeout_ms = (p->timeout_ms > 0) ? p->timeout_ms : 300000;
59 f->obj_size = -1;
60 return f;
61}

63int
64sd_s3_size(sd_s3_file *f, int64_t *out_size, char *errbuf, size_t errcap)
65{
66 char auth[SD_S3_AUTH_HDRS_CAP];
67 char cl[32];
68 brix_s3_resp_t resp;
69
70 if (f->obj_size >= 0) {
71 if (out_size != NULL) { *out_size = f->obj_size; }
72 return 0;
73 }
74 if (sd_s3_sign(f, "HEAD", "", auth, sizeof(auth)) != 0) {
75 sd_s3_set_err(errbuf, errcap, "s3 HEAD: SigV4 sign failed on %s", f->key);
76 return -1;
77 }
78 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "HEAD",
79 f->key, auth, NULL, 0, f->timeout_ms, &resp,
80 errbuf, errcap) != 0) {
81 return -1;
82 }
83 if (resp.status != 200) {
84 int rc = sd_s3_status_err(resp.status, "HEAD", f->key, errbuf, errcap);
85 f->transport->resp_free(&resp);
86 return rc;
87 }
88 if (f->transport->resp_header(&resp, "Content-Length", cl, sizeof(cl)) == 0) {
89 f->obj_size = (int64_t) strtoll(cl, NULL, 10);
90 } else {
91 f->obj_size = -1;
92 }
93 f->transport->resp_free(&resp);
94 if (out_size != NULL) { *out_size = f->obj_size; }
95 return 0;
96}

98ssize_t
99sd_s3_pread(sd_s3_file *f, void *buf, size_t n, off_t off,
100 char *errbuf, size_t errcap)
101{
102 char auth[SD_S3_AUTH_HDRS_CAP];
103 char combined[SD_S3_AUTH_HDRS_CAP + 80];
104 brix_s3_resp_t resp;
105 const void *body;
106 size_t body_len = 0, n_capped;
107 ssize_t copied;
108 int64_t end;
109 int cn;
110
111 if (n == 0) {
112 return 0;
113 }
114 n_capped = (n > (size_t) SD_S3_PREAD_MAX) ? (size_t) SD_S3_PREAD_MAX : n;
115 end = (int64_t) off + (int64_t) n_capped - 1;
116
117 if (sd_s3_sign(f, "GET", "", auth, sizeof(auth)) != 0) {
118 sd_s3_set_err(errbuf, errcap, "s3 GET: SigV4 sign failed on %s", f->key);
119 return -1;
120 }
121 cn = snprintf(combined, sizeof(combined), "Range: bytes=%lld-%lld\r\n%s",
122 (long long) off, (long long) end, auth);
123 if (cn < 0 || (size_t) cn >= sizeof(combined)) {
124 sd_s3_set_err(errbuf, errcap, "s3 GET: header too long");
125 return -1;
126 }
127 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "GET",
128 f->key, combined, NULL, 0, f->timeout_ms, &resp,
129 errbuf, errcap) != 0) {
130 return -1;
131 }
132 if (resp.status == 416) {
133 /* Range Not Satisfiable: the offset is at/after the object end - a
134 * positional read past EOF returns 0 (the pread/copy-loop EOF signal). */
135 f->transport->resp_free(&resp);
136 return 0;
137 }
138 if (resp.status != 206 && resp.status != 200) {
139 sd_s3_status_err(resp.status, "GET", f->key, errbuf, errcap);
140 f->transport->resp_free(&resp);
141 return -1;
142 }
143 body = f->transport->resp_body(&resp, &body_len);
144 if (body == NULL || body_len == 0) {
145 f->transport->resp_free(&resp);
146 return 0; /* EOF / empty range */
147 }
148 copied = (body_len < n_capped) ? (ssize_t) body_len : (ssize_t) n_capped;
149 memcpy(buf, body, (size_t) copied);
150 f->transport->resp_free(&resp);
151 return copied;
152}

154int
155sd_s3_delete(const sd_s3_open_params *p, char *errbuf, size_t errcap)
156{
157 sd_s3_file *f;
158 char auth[SD_S3_AUTH_HDRS_CAP];
159 brix_s3_resp_t resp;
160 int rc = 0;
161
162 f = sd_s3_open_read(p, errbuf, errcap); /* builds the handle; no I/O */
163 if (f == NULL) {
164 return -1;
165 }
166 if (sd_s3_sign(f, "DELETE", "", auth, sizeof(auth)) != 0) {
167 sd_s3_set_err(errbuf, errcap, "s3 DELETE: SigV4 sign failed on %s", f->key);
168 sd_s3_close(f);
169 return -1;
170 }
171 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "DELETE",
172 f->key, auth, NULL, 0, f->timeout_ms, &resp,
173 errbuf, errcap) != 0) {
174 sd_s3_close(f);
175 return -1;
176 }
177 /* S3 DELETE is idempotent: 204/200 succeed, 404 means already gone. */
178 if (resp.status != 204 && resp.status != 200 && resp.status != 404) {
179 rc = sd_s3_status_err(resp.status, "DELETE", f->key, errbuf, errcap);
180 }
181 f->transport->resp_free(&resp);
182 sd_s3_close(f);
183 return rc;
184}
