46/* sd_s3_utc_now — current UTC as SigV4 amzdate (YYYYMMDDTHHMMSSZ) + datestamp
47 * (YYYYMMDD), computed self-contained from the epoch (civil-from-days) so it
48 * needs neither <time.h> (shadowed here) nor the module-only compat/time.c. */
49void
50sd_s3_utc_now(char amzdate[20], char datestamp[12])
51{
52 struct timeval tv;
53 long long secs, days, rem, z, era, y;
54 unsigned doe, yoe, doy, mp, d, m;
55 int hh, mm, ss;
56
57 gettimeofday(&tv, NULL);
58 secs = (long long) tv.tv_sec;
59 days = secs / 86400;
60 rem = secs % 86400;
61 if (rem < 0) { rem += 86400; days -= 1; }
62 hh = (int) (rem / 3600);
63 mm = (int) ((rem % 3600) / 60);
64 ss = (int) (rem % 60);
65
66 /* Howard Hinnant's civil_from_days (days since 1970-01-01, UTC). */
67 z = days + 719468;
68 era = (z >= 0 ? z : z - 146096) / 146097;
69 doe = (unsigned) (z - era * 146097);
70 yoe = (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365;
71 y = (long long) yoe + era * 400;
72 doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
73 mp = (5 * doy + 2) / 153;
74 d = doy - (153 * mp + 2) / 5 + 1;
75 m = mp < 10 ? mp + 3 : mp - 9;
76 y += (m <= 2);
77
78 /* Bound the year to 4 digits (real epochs are well within 0000-9999) so the
79 * fixed amzdate[20]/datestamp[12] buffers are provably large enough. */
80 {
81 int yr = (int) (((y % 10000) + 10000) % 10000);
82 unsigned mo = m % 100, da = d % 100;
83 snprintf(datestamp, 12, "%04d%02u%02u", yr, mo, da);
84 snprintf(amzdate, 20, "%04d%02u%02uT%02u%02u%02uZ",
85 yr, mo, da, hh % 100, mm % 100, ss % 100);
86 }
87}

97/* sd_s3_sign — build the SigV4 x-amz-date / content-sha256 / Authorization header
98 * block for `method` on this object (payload = UNSIGNED-PAYLOAD). Ported from
99 * xrdc_s3_sign_v4_q; all kernels are shared. 0 / -1. */
100int
101sd_s3_sign(const sd_s3_file *f, const char *method, const char *canon_qs,
102 char *hdrs, size_t hdrsz)
103{
104 const char *payload_hex = "UNSIGNED-PAYLOAD";
105 char host[300];
106 char amzdate[20], datestamp[12];
107 char canon_hex[65], sighex[65];
108 char canon[8192], scope[160], sts[640], enc_uri[2048];
109 uint8_t k4[32], sig[32];
110 int cn;
111
112 if (canon_qs == NULL) { canon_qs = ""; }
113 brix_format_host_port(f->host, (uint16_t) f->port, host, sizeof(host));
114 sd_s3_utc_now(amzdate, datestamp);
115
116 if (brix_http_urlencode((const unsigned char *) f->key, strlen(f->key),
117 enc_uri, sizeof(enc_uri), "/") < 0) {
118 return -1;
119 }
120 cn = snprintf(canon, sizeof(canon),
121 "%s\n%s\n%s\nhost:%s\nx-amz-content-sha256:%s\nx-amz-date:%s\n\n"
122 "host;x-amz-content-sha256;x-amz-date\n%s",
123 method, enc_uri, canon_qs, host, payload_hex, amzdate, payload_hex);
124 if (cn < 0 || (size_t) cn >= sizeof(canon)) {
125 return -1;
126 }
127 sd_s3_sha256_hex(canon, strlen(canon), canon_hex);
128
129 cn = snprintf(scope, sizeof(scope), "%s/%s/s3/aws4_request",
130 datestamp, f->region);
131 if (cn < 0 || (size_t) cn >= sizeof(scope)) {
132 return -1;
133 }
134 cn = snprintf(sts, sizeof(sts), "AWS4-HMAC-SHA256\n%s\n%s\n%s",
135 amzdate, scope, canon_hex);
136 if (cn < 0 || (size_t) cn >= sizeof(sts)) {
137 return -1;
138 }
139
140 if (!brix_sigv4_signing_key((const uint8_t *) f->sk, strlen(f->sk),
141 datestamp, f->region, "s3", k4)
142 || !brix_hmac_sha256(k4, 32, (const uint8_t *) sts, strlen(sts), sig)) {
143 return -1;
144 }
145 brix_hex_encode(sig, 32, sighex);
146
147 cn = snprintf(hdrs, hdrsz,
148 "x-amz-date: %s\r\nx-amz-content-sha256: %s\r\n"
149 "Authorization: AWS4-HMAC-SHA256 Credential=%s/%s, "
150 "SignedHeaders=host;x-amz-content-sha256;x-amz-date, Signature=%s\r\n",
151 amzdate, payload_hex, f->ak, scope, sighex);
152 if (cn < 0 || (size_t) cn >= hdrsz) {
153 return -1;
154 }
155 return 0;
156}

158/* Map a non-2xx HTTP status to -1 with a diagnostic; returns -1 always. Sets
159 * errno to the matching POSIX class (EACCES for 401/403, ENOENT for 404, else
160 * EIO) so callers can recover the error category — the message channel alone
161 * loses it. The client S3 VFS maps this errno to XRDC_EAUTH/ENOENT; the server
162 * S3 backend maps it to kXR via the canonical errno↔kXR table. */
163int
164sd_s3_status_err(int status, const char *op, const char *key,
165 char *errbuf, size_t errcap)
166{
167 if (status == 401 || status == 403) {
168 sd_s3_set_err(errbuf, errcap,
169 "s3 %s: auth failed (HTTP %d) on %s — check AK/SK/region",
170 op, status, key);
171 errno = EACCES;
172 } else if (status == 404) {
173 sd_s3_set_err(errbuf, errcap, "s3 %s: not found (HTTP 404) on %s",
174 op, key);
175 errno = ENOENT;
176 } else {
177 sd_s3_set_err(errbuf, errcap, "s3 %s: HTTP %d on %s", op, status, key);
178 errno = EIO;
179 }
180 return -1;
181}

183sd_s3_file *
184sd_s3_open_read(const sd_s3_open_params *p, char *errbuf, size_t errcap)
185{
186 sd_s3_file *f;
187
188 if (p == NULL || p->host == NULL || p->key == NULL || p->transport == NULL) {
189 sd_s3_set_err(errbuf, errcap, "s3 open: bad parameters");
190 return NULL;
191 }
192 f = calloc(1, sizeof(*f));
193 if (f == NULL) {
194 sd_s3_set_err(errbuf, errcap, "s3 open: out of memory");
195 return NULL;
196 }
197 snprintf(f->host, sizeof(f->host), "%s", p->host);
198 snprintf(f->key, sizeof(f->key), "%s", p->key);
199 snprintf(f->ak, sizeof(f->ak), "%s", p->ak ? p->ak : "");
200 snprintf(f->sk, sizeof(f->sk), "%s", p->sk ? p->sk : "");
201 snprintf(f->region, sizeof(f->region), "%s",
202 (p->region && p->region[0]) ? p->region : "us-east-1");
203 f->port = p->port;
204 f->tls = p->tls;
205 f->transport = p->transport;
206 f->tctx = p->tctx;
207 f->timeout_ms = (p->timeout_ms > 0) ? p->timeout_ms : 300000;
208 f->obj_size = -1;
209 return f;

212int
213sd_s3_size(sd_s3_file *f, int64_t *out_size, char *errbuf, size_t errcap)
214{
215 char auth[SD_S3_AUTH_HDRS_CAP];
216 char cl[32];
217 brix_s3_resp_t resp;
218
219 if (f->obj_size >= 0) {
220 if (out_size != NULL) { *out_size = f->obj_size; }
221 return 0;
222 }
223 if (sd_s3_sign(f, "HEAD", "", auth, sizeof(auth)) != 0) {
224 sd_s3_set_err(errbuf, errcap, "s3 HEAD: SigV4 sign failed on %s", f->key);
225 return -1;
226 }
227 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "HEAD",
228 f->key, auth, NULL, 0, f->timeout_ms, &resp,
229 errbuf, errcap) != 0) {
230 return -1;
231 }
232 if (resp.status != 200) {
233 int rc = sd_s3_status_err(resp.status, "HEAD", f->key, errbuf, errcap);
234 f->transport->resp_free(&resp);
235 return rc;
236 }
237 if (f->transport->resp_header(&resp, "Content-Length", cl, sizeof(cl)) == 0) {
238 f->obj_size = (int64_t) strtoll(cl, NULL, 10);
239 } else {
240 f->obj_size = -1;
241 }
242 f->transport->resp_free(&resp);
243 if (out_size != NULL) { *out_size = f->obj_size; }
244 return 0;
245}

247ssize_t
248sd_s3_pread(sd_s3_file *f, void *buf, size_t n, off_t off,
249 char *errbuf, size_t errcap)
250{
251 char auth[SD_S3_AUTH_HDRS_CAP];
252 char combined[SD_S3_AUTH_HDRS_CAP + 80];
253 brix_s3_resp_t resp;
254 const void *body;
255 size_t body_len = 0, n_capped;
256 ssize_t copied;
257 int64_t end;
258 int cn;
259
260 if (n == 0) {
261 return 0;
262 }
263 n_capped = (n > (size_t) SD_S3_PREAD_MAX) ? (size_t) SD_S3_PREAD_MAX : n;
264 end = (int64_t) off + (int64_t) n_capped - 1;
265
266 if (sd_s3_sign(f, "GET", "", auth, sizeof(auth)) != 0) {
267 sd_s3_set_err(errbuf, errcap, "s3 GET: SigV4 sign failed on %s", f->key);
268 return -1;
269 }
270 cn = snprintf(combined, sizeof(combined), "Range: bytes=%lld-%lld\r\n%s",
271 (long long) off, (long long) end, auth);
272 if (cn < 0 || (size_t) cn >= sizeof(combined)) {
273 sd_s3_set_err(errbuf, errcap, "s3 GET: header too long");
274 return -1;
275 }
276 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "GET",
277 f->key, combined, NULL, 0, f->timeout_ms, &resp,
278 errbuf, errcap) != 0) {
279 return -1;
280 }
281 if (resp.status == 416) {
282 /* Range Not Satisfiable: the offset is at/after the object end - a
283 * positional read past EOF returns 0 (the pread/copy-loop EOF signal). */
284 f->transport->resp_free(&resp);
285 return 0;
286 }
287 if (resp.status != 206 && resp.status != 200) {
288 sd_s3_status_err(resp.status, "GET", f->key, errbuf, errcap);
289 f->transport->resp_free(&resp);
290 return -1;
291 }
292 body = f->transport->resp_body(&resp, &body_len);
293 if (body == NULL || body_len == 0) {
294 f->transport->resp_free(&resp);
295 return 0; /* EOF / empty range */
296 }
297 copied = (body_len < n_capped) ? (ssize_t) body_len : (ssize_t) n_capped;
298 memcpy(buf, body, (size_t) copied);
299 f->transport->resp_free(&resp);
300 return copied;
301}

303int
304sd_s3_delete(const sd_s3_open_params *p, char *errbuf, size_t errcap)
305{
306 sd_s3_file *f;
307 char auth[SD_S3_AUTH_HDRS_CAP];
308 brix_s3_resp_t resp;
309 int rc = 0;
310
311 f = sd_s3_open_read(p, errbuf, errcap); /* builds the handle; no I/O */
312 if (f == NULL) {
313 return -1;
314 }
315 if (sd_s3_sign(f, "DELETE", "", auth, sizeof(auth)) != 0) {
316 sd_s3_set_err(errbuf, errcap, "s3 DELETE: SigV4 sign failed on %s", f->key);
317 sd_s3_close(f);
318 return -1;
319 }
320 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "DELETE",
321 f->key, auth, NULL, 0, f->timeout_ms, &resp,
322 errbuf, errcap) != 0) {
323 sd_s3_close(f);
324 return -1;
325 }
326 /* S3 DELETE is idempotent: 204/200 succeed, 404 means already gone. */
327 if (resp.status != 204 && resp.status != 200 && resp.status != 404) {
328 rc = sd_s3_status_err(resp.status, "DELETE", f->key, errbuf, errcap);
329 }
330 f->transport->resp_free(&resp);
331 sd_s3_close(f);
332 return rc;
333}

337/* Extract <tag>…</tag> text from an XML body into out[outsz]. 0 / -1. */
338static int
339sd_s3_xml_tag(const char *xml, const char *tag, char *out, size_t outsz)
340{
341 char open_tag[64], close_tag[64];
342 const char *p, *e;
343 size_t n;
344
345 snprintf(open_tag, sizeof(open_tag), "<%s>", tag);
346 snprintf(close_tag, sizeof(close_tag), "</%s>", tag);
347 p = strstr(xml, open_tag);
348 if (p == NULL) { return -1; }
349 p += strlen(open_tag);
350 e = strstr(p, close_tag);
351 if (e == NULL) { return -1; }
352 n = (size_t) (e - p);
353 if (n == 0 || n >= outsz) { return -1; }
354 memcpy(out, p, n);
355 out[n] = '\0';
356 return 0;
357}

359static int
360sd_s3_etag_ensure_cap(sd_s3_file *f, int needed, char *errbuf, size_t errcap)
361{
362 int new_cap;
363 sd_s3_part_etag *ne;
364
365 if (f->etag_cap >= needed) { return 0; }
366 new_cap = (f->etag_cap == 0) ? SD_S3_ETAG_INIT_CAP : f->etag_cap * 2;
367 while (new_cap < needed) { new_cap *= 2; }
368 ne = realloc(f->etags, (size_t) new_cap * sizeof(*ne));
369 if (ne == NULL) {
370 sd_s3_set_err(errbuf, errcap, "s3 mpu: out of memory for ETag array");
371 return -1;
372 }
373 f->etags = ne;
374 f->etag_cap = new_cap;
375 return 0;
376}

378static int
379sd_s3_mpu_upload_part(sd_s3_file *f, int part_num, const void *data, size_t len,
380 char *errbuf, size_t errcap)
381{
382 char wire[SD_S3_KEY_MAX + 256], qs[256], auth[SD_S3_AUTH_HDRS_CAP];
383 char etag[SD_S3_ETAG_LEN];
384 brix_s3_resp_t resp;
385 int pn;
386
387 pn = snprintf(wire, sizeof(wire), "%s?partNumber=%d&uploadId=%s",
388 f->key, part_num, f->upload_id);
389 if (pn < 0 || (size_t) pn >= sizeof(wire)) {
390 sd_s3_set_err(errbuf, errcap, "s3 mpu: part path too long");
391 return -1;
392 }
393 pn = snprintf(qs, sizeof(qs), "partNumber=%d&uploadId=%s",
394 part_num, f->upload_id);
395 if (pn < 0 || (size_t) pn >= sizeof(qs)) {
396 sd_s3_set_err(errbuf, errcap, "s3 mpu: canon_qs too long");
397 return -1;
398 }
399 if (sd_s3_sign(f, "PUT", qs, auth, sizeof(auth)) != 0) {
400 sd_s3_set_err(errbuf, errcap, "s3 UploadPart: sign failed");
401 return -1;
402 }
403 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "PUT", wire,
404 auth, data, len, f->timeout_ms, &resp,
405 errbuf, errcap) != 0) {
406 return -1;
407 }
408 if (resp.status != 200) {
409 int rc = sd_s3_status_err(resp.status, "UploadPart", f->key, errbuf, errcap);
410 f->transport->resp_free(&resp);
411 return rc;
412 }
413 etag[0] = '\0';
414 f->transport->resp_header(&resp, "ETag", etag, sizeof(etag));
415 f->transport->resp_free(&resp);
416 if (etag[0] == '\0') {
417 sd_s3_set_err(errbuf, errcap, "s3 UploadPart: server returned no ETag");
418 return -1;
419 }
420 if (sd_s3_etag_ensure_cap(f, part_num, errbuf, errcap) != 0) {
421 return -1;
422 }
423 snprintf(f->etags[part_num - 1].val, SD_S3_ETAG_LEN, "%s", etag);
424 f->part_count = part_num;
425 return 0;
426}

440static int
441sd_s3_mpu_create(sd_s3_file *f, char *errbuf, size_t errcap)
442{
443 char wire[SD_S3_KEY_MAX + 16], auth[SD_S3_AUTH_HDRS_CAP];
444 brix_s3_resp_t resp;
445 const void *body;
446 size_t blen = 0;
447 int pn;
448
449 pn = snprintf(wire, sizeof(wire), "%s?uploads", f->key);
450 if (pn < 0 || (size_t) pn >= sizeof(wire)) {
451 sd_s3_set_err(errbuf, errcap, "s3 mpu: key path too long");
452 return -1;
453 }
454 if (sd_s3_sign(f, "POST", "uploads=", auth, sizeof(auth)) != 0) {
455 sd_s3_set_err(errbuf, errcap, "s3 CreateMPU: sign failed");
456 return -1;
457 }
458 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "POST", wire,
459 auth, NULL, 0, f->timeout_ms, &resp,
460 errbuf, errcap) != 0) {
461 return -1;
462 }
463 if (resp.status != 200) {
464 int rc = sd_s3_status_err(resp.status, "CreateMPU", f->key, errbuf, errcap);
465 int e = errno; /* preserve across resp_free cleanup */
466 f->transport->resp_free(&resp);
467 errno = e;
468 return rc;
469 }
470 body = f->transport->resp_body(&resp, &blen);
471 if (body == NULL || blen == 0
472 || sd_s3_xml_tag((const char *) body, "UploadId",
473 f->upload_id, sizeof(f->upload_id)) != 0) {
474 f->transport->resp_free(&resp);
475 sd_s3_set_err(errbuf, errcap, "s3 CreateMPU: no <UploadId> in response");
476 return -1;
477 }
478 f->transport->resp_free(&resp);
479 return 0;
480}

482static int
483sd_s3_mpu_complete(sd_s3_file *f, char *errbuf, size_t errcap)
484{
485 char wire[SD_S3_KEY_MAX + SD_S3_UPLOAD_ID_LEN + 16];
486 char qs[SD_S3_UPLOAD_ID_LEN + 16], auth[SD_S3_AUTH_HDRS_CAP];
487 brix_s3_resp_t resp;
488 char *xml;
489 size_t cap, len;
490 int i, pn;
491
492 pn = snprintf(wire, sizeof(wire), "%s?uploadId=%s", f->key, f->upload_id);
493 if (pn < 0 || (size_t) pn >= sizeof(wire)) {
494 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: path too long");
495 return -1;
496 }
497 pn = snprintf(qs, sizeof(qs), "uploadId=%s", f->upload_id);
498 if (pn < 0 || (size_t) pn >= sizeof(qs)) {
499 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: canon_qs too long");
500 return -1;
501 }
502 cap = 120 + (size_t) f->part_count * (60 + SD_S3_ETAG_LEN) + 1;
503 xml = malloc(cap);
504 if (xml == NULL) {
505 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: out of memory for XML");
506 return -1;
507 }
508 len = (size_t) snprintf(xml, cap,
509 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
510 "<CompleteMultipartUpload>\n");
511 for (i = 0; i < f->part_count; i++) {
512 int w = snprintf(xml + len, cap - len,
513 " <Part><PartNumber>%d</PartNumber><ETag>%s</ETag></Part>\n",
514 i + 1, f->etags[i].val);
515 if (w < 0 || (size_t) w >= cap - len) {
516 free(xml);
517 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: XML overflow");
518 return -1;
519 }
520 len += (size_t) w;
521 }
522 {
523 int t = snprintf(xml + len, cap - len, "</CompleteMultipartUpload>\n");
524 if (t < 0 || (size_t) t >= cap - len) {
525 free(xml);
526 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: XML overflow (tail)");
527 return -1;
528 }
529 len += (size_t) t;
530 }
531 if (sd_s3_sign(f, "POST", qs, auth, sizeof(auth)) != 0) {
532 free(xml);
533 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: sign failed");
534 return -1;
535 }
536 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "POST", wire,
537 auth, xml, len, f->timeout_ms, &resp,
538 errbuf, errcap) != 0) {
539 free(xml);
540 return -1;
541 }
542 free(xml);
543 if (resp.status != 200) {
544 int rc = sd_s3_status_err(resp.status, "CompleteMPU", f->key, errbuf, errcap);
545 f->transport->resp_free(&resp);
546 return rc;
547 }
548 f->transport->resp_free(&resp);
549 return 0;

552static void
553sd_s3_mpu_abort(sd_s3_file *f)
554{
555 char wire[SD_S3_KEY_MAX + SD_S3_UPLOAD_ID_LEN + 16];
556 char qs[SD_S3_UPLOAD_ID_LEN + 16], auth[SD_S3_AUTH_HDRS_CAP];
557 brix_s3_resp_t resp;
558 int pn;
559
560 if (f->upload_id[0] == '\0') { return; }
561 pn = snprintf(wire, sizeof(wire), "%s?uploadId=%s", f->key, f->upload_id);
562 if (pn < 0 || (size_t) pn >= sizeof(wire)) { return; }
563 pn = snprintf(qs, sizeof(qs), "uploadId=%s", f->upload_id);
564 if (pn < 0 || (size_t) pn >= sizeof(qs)) { return; }
565 if (sd_s3_sign(f, "DELETE", qs, auth, sizeof(auth)) != 0) { return; }
566 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "DELETE", wire,
567 auth, NULL, 0, f->timeout_ms, &resp,
568 NULL, 0) != 0) {
569 return; /* best-effort */
570 }
571 f->transport->resp_free(&resp);
572}

574static int
575sd_s3_check_sequential(int64_t off, int64_t expected, char *errbuf, size_t errcap)
576{
577 if (off != expected) {
578 sd_s3_set_err(errbuf, errcap,
579 "s3 requires sequential writes (got %lld, expected %lld)",
580 (long long) off, (long long) expected);
581 errno = EINVAL; /* usage error: maps to XRDC_EUSAGE / kXR_ArgInvalid */
582 return -1;
583 }
584 return 0;

587sd_s3_file *
588sd_s3_open_write(const sd_s3_open_params *p, int64_t expected_size,
589 int64_t part_size, char *errbuf, size_t errcap)
590{
591 sd_s3_file *f = sd_s3_open_read(p, errbuf, errcap);
592 if (f == NULL) {
593 return NULL;
594 }
595 f->is_write = 1;
596 f->part_size = (part_size > 0) ? part_size : (64LL * 1024 * 1024);
597
598 if (expected_size >= 0 && expected_size <= f->part_size) {
599 /* single buffered PUT */
600 size_t cap = (expected_size > 0) ? (size_t) expected_size
601 : (size_t) SD_S3_PUT_BUF_INIT;
602 if (cap < (size_t) SD_S3_PUT_BUF_INIT) { cap = SD_S3_PUT_BUF_INIT; }
603 f->put_buf = malloc(cap);
604 if (f->put_buf == NULL) {
605 sd_s3_set_err(errbuf, errcap, "s3 open write: out of memory (PUT)");
606 sd_s3_close(f);
607 return NULL;
608 }
609 f->put_cap = cap;
610 } else {
611 /* multipart upload */
612 if (sd_s3_mpu_create(f, errbuf, errcap) != 0) {
613 int e = errno; /* preserve the CreateMPU error class */
614 sd_s3_close(f);
615 errno = e;
616 return NULL;
617 }
618 f->part_buf = malloc((size_t) f->part_size);
619 if (f->part_buf == NULL) {
620 sd_s3_set_err(errbuf, errcap, "s3 open write: out of memory (part)");
621 sd_s3_mpu_abort(f);
622 sd_s3_close(f);
623 return NULL;
624 }
625 f->is_mpu = 1;
626 }
627 return f;
628}

630int
631sd_s3_pwrite(sd_s3_file *f, const void *buf, size_t n, off_t off,
632 char *errbuf, size_t errcap)
633{
634 if (!f->is_write) {
635 sd_s3_set_err(errbuf, errcap, "s3 pwrite: handle opened for read");
636 return -1;
637 }
638 if (n == 0) {
639 return 0;
640 }
641 if (f->is_mpu) {
642 const char *src = (const char *) buf;
643 size_t rem = n;
644 if (sd_s3_check_sequential((int64_t) off, f->mpu_write_off,
645 errbuf, errcap) != 0) {
646 return -1;
647 }
648 while (rem > 0) {
649 size_t space = (size_t) f->part_size - f->part_buf_len;
650 size_t tocopy = (rem < space) ? rem : space;
651 memcpy((char *) f->part_buf + f->part_buf_len, src, tocopy);
652 f->part_buf_len += tocopy;
653 src += tocopy;
654 rem -= tocopy;
655 if (f->part_buf_len == (size_t) f->part_size
656 && sd_s3_mpu_flush(f, errbuf, errcap) != 0) {
657 return -1;
658 }
659 }
660 f->mpu_write_off = (int64_t) off + (int64_t) n;
661 return 0;
662 }
663
664 /* single buffered PUT */
665 if (sd_s3_check_sequential((int64_t) off, f->put_write_off,
666 errbuf, errcap) != 0) {
667 return -1;
668 }
669 if (f->put_len + n > f->put_cap) {
670 size_t nc = f->put_cap ? f->put_cap * 2 : SD_S3_PUT_BUF_INIT;
671 void *nb;
672 while (nc < f->put_len + n) { nc *= 2; }
673 nb = realloc(f->put_buf, nc);
674 if (nb == NULL) {
675 sd_s3_set_err(errbuf, errcap, "s3 single-put: out of memory");
676 return -1;
677 }
678 f->put_buf = nb;
679 f->put_cap = nc;
680 }
681 memcpy((char *) f->put_buf + f->put_len, buf, n);
682 f->put_len += n;
683 f->put_write_off = (int64_t) off + (int64_t) n;
684 return 0;
685}

687int
688sd_s3_commit(sd_s3_file *f, char *errbuf, size_t errcap)
689{
690 if (!f->is_write) {
691 return 0; /* read handle */
692 }
693 if (f->is_mpu) {
694 if (sd_s3_mpu_flush(f, errbuf, errcap) != 0) {
695 return -1;
696 }
697 if (f->part_count == 0
698 && sd_s3_mpu_upload_part(f, 1, NULL, 0, errbuf, errcap) != 0) {
699 return -1; /* zero-byte object: one empty part */
700 }
701 return sd_s3_mpu_complete(f, errbuf, errcap);
702 }
703 /* single PUT of the whole buffer */
704 {
705 char auth[SD_S3_AUTH_HDRS_CAP];
706 brix_s3_resp_t resp;
707 if (sd_s3_sign(f, "PUT", "", auth, sizeof(auth)) != 0) {
708 sd_s3_set_err(errbuf, errcap, "s3 PUT: sign failed");
709 return -1;
710 }
711 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "PUT",
712 f->key, auth, f->put_buf, f->put_len,
713 f->timeout_ms, &resp, errbuf, errcap) != 0) {
714 return -1;
715 }
716 if (resp.status != 200) {
717 int rc = sd_s3_status_err(resp.status, "PUT", f->key, errbuf, errcap);
718 f->transport->resp_free(&resp);
719 return rc;
720 }
721 f->transport->resp_free(&resp);
722 }
723 return 0;
724}
