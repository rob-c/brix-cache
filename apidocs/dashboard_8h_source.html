4/*
5 * dashboard/dashboard.h — live transfer monitor: shared-memory types and public API.
6 *
7 * A self-contained admin dashboard that lets site operators watch every active
8 * transfer in real time: who, which file, which protocol, how many bytes, how fast.
9 *
10 * ARCHITECTURE:
11 * Stream workers update a shared-memory transfer table via lock-free atomics.
12 * An HTTP worker reads the same table and emits JSON at GET /xrootd/transfers.
13 * A small embedded HTML page polls that endpoint every two seconds and renders
14 * the result client-side, computing instantaneous rates from consecutive snapshots.
15 *
16 * CONCURRENCY:
17 * Slot allocation/free holds the table mutex briefly.
18 * Per-slot byte/timestamp updates use atomic operations — no lock held.
19 * The JSON exporter does a lock-free scan; it tolerates torn reads gracefully
20 * because the dashboard is display-only and eventual consistency is fine.
21 *
22 * SLOT LIFECYCLE:
23 * kXR_open → brix_transfer_slot_alloc() (store index in fh->dashboard_slot)
24 * kXR_read /
25 * kXR_write → brix_transfer_slot_update() (atomic byte/timestamp increment)
26 * kXR_close → brix_transfer_slot_free() (reset index to -1)
27 * disconnect → brix_transfer_slot_free_all_for_session() (catch dropped clients)
28 */

101/*
102 * One active-transfer record.
103 *
104 * Lives in shared memory. The lock in brix_transfer_table_t is held only
105 * during slot allocation — per-slot byte/timestamp updates are lock-free.
106 *
107 * Field notes:
108 * in_use — 0 = free slot; 1 = active transfer. Written under table lock.
109 * serial — monotonic ID so the JS dashboard can detect row churn without
110 * comparing every field; incremented each time a slot is reused.
111 * sessid — session ID copied from ctx->sessid; used for cleanup-by-sessid
112 * during disconnect (worker that opened the file may not be the
113 * worker that notices the drop).
114 * bytes — cumulative bytes transferred; updated atomically during I/O.
115 * start_ms — epoch milliseconds at transfer open; written once at alloc time.
116 * last_ms — epoch milliseconds of the most recent I/O; atomic word write.
117 */
118typedef struct {
119 ngx_atomic_t in_use; /* 0=free, 1=active; written under table lock */
120 uint32_t serial; /* monotonic ID — lets JS detect row churn */
121 u_char sessid[16]; /* session ID for cleanup-by-sessid on disconnect */
122 ngx_pid_t worker_pid;
123 char client_ip[BRIX_DASHBOARD_IP_LEN];
124 char identity[BRIX_DASHBOARD_IDENTITY_LEN]; /* DN or "anonymous" */
