1/*
2 * list_common.c — shared building blocks for the S3 ListObjects emitters.
3 *
4 * WHAT: The four steps that ListObjects V1 (list_objects_v1.c) and V2
5 * (list_objects_v2.c) perform identically — parse max-keys, acquire a sorted
6 * key listing (cache-or-walk), paginate it, emit the Contents/CommonPrefixes
7 * body, and finalize the response — plus they share the directory walker and
8 * comparator already in list_walk.c.
9 *
10 * WHY: V1 and V2 differ ONLY in their pagination parameter (`marker` vs
11 * `continuation-token`/`start-after`) and a few response element names
12 * (`<Marker>`/`<NextMarker>` vs `<KeyCount>`/`<NextContinuationToken>`).
13 * Everything else was copy-pasted between the two files; centralizing it here
14 * keeps the two dialects to just their genuine differences and removes the
15 * risk of the shared logic drifting apart.
16 *
17 * HOW: Each helper is a pure step with explicit inputs/outputs and no hidden
18 * state. s3_list_emit_entries() drives the flat-buffer XML_APPEND macros (which
19 * reference `xml`/`xml_len`/`xml_capacity` and return 500 on overflow), so it
20 * takes the buffer cursor by pointer and propagates that contract to its caller.
21 */

32/*
33 * s3_list_parse_max_keys — parse the `max-keys` query arg, clamped to
34 * (0, default_max). Returns default_max when the arg is absent or invalid, and
35 * applies a 1000 floor (matching AWS) when default_max is non-positive.
36 */
37int
38s3_list_parse_max_keys(ngx_http_request_t *r, int default_max)
39{
40 u_char max_keys_buf[32] = { 0 };
41 int max_keys = default_max;
42
43 if (brix_http_query_get(r->args, "max-keys",
44 (char *) max_keys_buf, sizeof(max_keys_buf),
45 S3_LIST_QUERY_FLAGS) > 0)
46 {
47 char *endp;
48 long mk;
49
50 errno = 0;
51 mk = strtol((const char *) max_keys_buf, &endp, 10);
52 if (errno == 0 && endp != (char *) max_keys_buf
53 && mk > 0 && mk < max_keys)
54 {
55 max_keys = (int) mk;
56 }
57 }
58 if (max_keys <= 0) {
59 max_keys = 1000;
60 }
61 return max_keys;
62}

64/*
65 * s3_list_collect_sorted — acquire the sorted (key + is_prefix) listing for the
66 * (root, prefix, delimiter) view: try the per-worker sorted-listing cache (W6c)
67 * and, on a miss, walk the bucket subtree (s3_walk), qsort by key (entry_cmp),
68 * and populate the cache. On return *items / *total describe a sorted array
69 * (pool- or cache-owned); size/mtime/ETag are filled lazily per emitted page by
70 * s3_entry_fill_stat(). Returns NGX_OK, or NGX_ERROR on allocation failure.
71 */
72ngx_int_t
73s3_list_collect_sorted(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
74 const char *prefix, const char *delimiter,
75 s3_entry_t **items, int *total)
76{
77 ngx_array_t *entries;
78 int cached = 0;
79 time_t dir_mtime = 0;
80
81 if (cf->list_cache) {
82 struct stat rst;
83
84 if (stat((const char *) cf->common.root.data, &rst) == 0) { /* vfs-seam-allow: export-root mtime for the list cache (the root itself, not a path beneath it) */
85 dir_mtime = rst.st_mtime;
86 }
87 cached = s3_list_cache_get(r, (const char *) cf->common.root.data,
88 prefix, delimiter,
89 dir_mtime, cf->list_cache_ttl,
90 items, total);
91 }
92
93 if (!cached) {
94 entries = ngx_array_create(r->pool, 256, sizeof(s3_entry_t));
95 if (entries == NULL) {
96 return NGX_ERROR;
97 }
98
99 s3_walk(r->connection->log,
100 (const char *) cf->common.root.data,
101 (const char *) cf->common.root.data,
102 "", prefix, delimiter, entries, S3_LIST_MAX_ENTRIES);
103
104 *total = (int) entries->nelts;
105 *items = entries->elts;
106
107 qsort(*items, (size_t) *total, sizeof(s3_entry_t), entry_cmp);
108
109 if (cf->list_cache) {
110 s3_list_cache_put(r->connection->log,
111 (const char *) cf->common.root.data,
112 prefix, delimiter, dir_mtime, *items, *total);
113 }
114 }
115 return NGX_OK;
116}

118/*
119 * s3_list_paginate — shared pagination math. Skip every entry whose key is <=
120 * `start_after` (the exclusive continuation marker; NULL/"" means start at the
121 * beginning) — a linear scan, acceptable since `total` is already capped by
122 * S3_LIST_MAX_ENTRIES — then take up to max_keys. Fills *start_idx / *end_idx
123 * and returns 1 when the listing is truncated (entries remain), 0 otherwise.
124 */
125int
126s3_list_paginate(const s3_entry_t *items, int total, const char *start_after,
127 int max_keys, int *start_idx, int *end_idx)
128{
129 int si = 0;
130 int ei;
131
132 if (start_after != NULL && start_after[0] != '\0') {
133 for (si = 0; si < total; si++) {
134 if (strcmp(items[si].key, start_after) > 0) {
135 break;
136 }
137 }
138 }
139
140 ei = si + max_keys;
141 if (ei > total) {
142 ei = total;
143 }
144
145 *start_idx = si;
146 *end_idx = ei;
147 return ei < total;
148}

150/*
151 * s3_list_emit_entries — append the Contents / CommonPrefixes body for the page
152 * slice [start_idx, end_idx) into the flat XML buffer. A CommonPrefix becomes a
153 * <CommonPrefixes>; an object is lazily stat'd (s3_entry_fill_stat) and skipped
154 * if it has vanished, else emitted as <Contents> (Key url-encoded when
155 * url_encode, with an <Owner> block when fetch_owner and an access key is set).
156 * The emitted-counts land in *contents_out / *prefixes_out. Returns NGX_OK, or
157 * NGX_HTTP_INTERNAL_SERVER_ERROR if the buffer overflows (caller propagates).
158 */
159ngx_int_t
160s3_list_emit_entries(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
161 s3_entry_t *items, int start_idx, int end_idx,
162 int url_encode, int fetch_owner,
163 u_char *xml, size_t *xml_len_io, size_t xml_capacity,
164 int *contents_out, int *prefixes_out)
165{
166 size_t xml_len = *xml_len_io;
167 int contents = 0;
168 int prefixes = 0;
169 char iso_buf[32];
170
171 for (int entry_index = start_idx; entry_index < end_idx; entry_index++) {
172 s3_entry_t *entry = &items[entry_index];
173
174 if (entry->is_prefix) {
175 prefixes++;
176 XML_APPEND("<CommonPrefixes>");
177 XML_APPEND_ELEM("Prefix", entry->key, strlen(entry->key));
178 XML_APPEND("</CommonPrefixes>");
179 continue;
180 }
181
182 char encoded_key[S3_MAX_KEY * 3 + 1];
183
184 /* phase-45 W1: stat is done HERE, only for the emitted page. If the
185 * object vanished or is no longer a regular file, skip it (matches the
186 * eager walker's stat-failure skip). */
187 if (s3_entry_fill_stat(r->pool, r->connection->log,
188 (const char *) cf->common.root.data,
189 entry) != NGX_OK)
190 {
191 continue;
192 }
193
194 contents++;
195 brix_format_iso8601(entry->mtime, iso_buf, sizeof(iso_buf));
196 XML_APPEND("<Contents>");
197 if (url_encode) {
198 brix_http_urlencode((const u_char *) entry->key,
199 strlen(entry->key),
200 encoded_key, sizeof(encoded_key), "");
201 XML_APPEND_ELEM("Key", encoded_key, strlen(encoded_key));
202 } else {
203 XML_APPEND_ELEM("Key", entry->key, strlen(entry->key));
204 }
205 XML_APPEND("<LastModified>%s</LastModified>", iso_buf);
206 XML_APPEND("<ETag>%s</ETag>", entry->etag);
207 XML_APPEND("<Size>%lld</Size>", (long long) entry->size);
208 XML_APPEND("<StorageClass>STANDARD</StorageClass>");
209 if (fetch_owner && cf->access_key.len > 0) {
210 XML_APPEND("<Owner>");
211 XML_APPEND_ELEM("ID", cf->access_key.data, cf->access_key.len);
212 XML_APPEND_ELEM("DisplayName", cf->access_key.data,
213 cf->access_key.len);
214 XML_APPEND("</Owner>");
215 }
216 XML_APPEND("</Contents>");
217 }
218
219 *xml_len_io = xml_len;
220 *contents_out = contents;
221 *prefixes_out = prefixes;
222 return NGX_OK;
223}

225/*
226 * s3_list_finalize — shared response tail: copy the built XML into a response
227 * buffer, record the list metrics (contents/common-prefixes/truncated/bytes),
228 * and send it as application/xml. Returns the send result, or 500 on
229 * buffer-allocation failure.
230 */
231ngx_int_t
232s3_list_finalize(ngx_http_request_t *r, const u_char *xml, size_t xml_len,
233 int contents, int prefixes, int truncated)
234{
235 ngx_buf_t *response_buf;
236
237 response_buf = ngx_create_temp_buf(r->pool, xml_len + 4);
238 if (response_buf == NULL) {
239 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
240 return NGX_HTTP_INTERNAL_SERVER_ERROR;
241 }
242 response_buf->last = ngx_cpymem(response_buf->last, xml, xml_len);
243 response_buf->last_buf = 1;
244
245 BRIX_S3_METRIC_ADD(list_contents_total, (size_t) contents);
246 BRIX_S3_METRIC_ADD(list_common_prefixes_total, (size_t) prefixes);
247 if (truncated) {
248 BRIX_S3_METRIC_INC(list_truncated_total);
249 }
250 BRIX_S3_METRIC_ADD(bytes_tx_total, xml_len);
251
252 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
253 (ngx_str_t) ngx_string("application/xml"), response_buf);
254}
