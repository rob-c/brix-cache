1/*
2 * client_ops.c — the impersonated filesystem op wrappers (phase 40).
3 *
4 * WHAT: The public brix_imp_* op wrappers (open/stat/mkdir/unlink/rmdir/rename/
5 * link/truncate/chmod/chown/setattr/symlink/readlink and the xattr family)
6 * that the confined beneath helpers call when `map` mode is active. Each
7 * wrapper builds a request frame and performs one synchronous request/reply
8 * round-trip with the broker via imp_exchange(); OPEN receives the resulting
9 * fd back over SCM_RIGHTS.
10 *
11 * WHY: Split out of client.c (which was over the file-size cap) so the
12 * connection/state singleton, transport primitives, and the imp_build_req +
13 * imp_exchange core stay in client.c while the op-wrapper surface lives here.
14 * The two cross-file primitives are declared in client_internal.h.
15 *
16 * HOW: imp_call_status runs the 0/-1 ops; imp_unpack_stat fills a caller
17 * struct stat; imp_get_or_list is the shared get/listxattr body. Broker
18 * status codes map to errno via imp_status_errno. No goto; pure helpers,
19 * side effects at the edges.
20 */

31/* Translate a positive IMP_STATUS_* into the conventional errno for callers. */
32static int
33imp_status_errno(int status)
34{
35 switch (status) {
36 case IMP_STATUS_DENY: return EACCES;
37 case IMP_STATUS_BADREQ: return EINVAL;
38 case IMP_STATUS_INTERNAL: return EIO;
39 case IMP_STATUS_PRIV_REFUSED: return EPERM; /* reserved-id refused at broker */
40 default: return EIO;
41 }
42}

44/*
45 * Run an op that returns 0/-1 (no fd). Maps the broker reply to errno: 0 ->
46 * success; status < 0 -> -status (the impersonated syscall's errno); status > 0
47 * -> the IMP_STATUS_* errno. Returns 0 or -1.
48 */
49static int
50imp_call_status(uint32_t op, const char *path, const char *path2,
51 uint32_t mode, int64_t length)
52{
53 imp_req_t req;
54 imp_rep_t rep;
55 int fd = -1;
56
57 imp_build_req(&req, op, path, path2, 0, mode, length);
58 if (imp_exchange(&req, &rep, &fd, NULL, 0, NULL, 0) != 0) {
59 return -1; /* errno=EIO set by imp_exchange */
60 }
61 if (fd >= 0) {
62 close(fd); /* a non-OPEN op should never send an fd */
63 }
64 if (rep.status == 0) {
65 return 0;
66 }
67 errno = (rep.status < 0) ? -rep.status : imp_status_errno(rep.status);
68 return -1;
69}

72int
73brix_imp_open(const char *reqpath, int flags, mode_t mode)
74{
75 imp_req_t req;
76 imp_rep_t rep;
77 int fd = -1;
78
79 imp_build_req(&req, IMP_OP_OPEN, reqpath, NULL,
80 (uint32_t) flags, (uint32_t) mode, 0);
81 if (imp_exchange(&req, &rep, &fd, NULL, 0, NULL, 0) != 0) {
82 return -1;
83 }
84 if (rep.status == 0 && (rep.rep_flags & IMP_REP_HAS_FD) && fd >= 0) {
85 return fd; /* caller owns the fd */
86 }
87 if (fd >= 0) {
88 close(fd); /* unexpected fd on a failed open */
89 }
90 errno = (rep.status < 0) ? -rep.status
91 : imp_status_errno(rep.status ? rep.status
92 : IMP_STATUS_INTERNAL);
93 return -1;
94}

98imp_unpack_stat(struct stat *st, const imp_stat_t *s)
99{
100 ngx_memzero(st, sizeof(*st));
101 st->st_ino = (ino_t) s->ino;
102 st->st_dev = (dev_t) s->dev;
103 st->st_size = (off_t) s->size;
104 st->st_blocks = (blkcnt_t) s->blocks;
105 st->st_mtime = (time_t) s->mtime;
106 st->st_ctime = (time_t) s->ctime;
107 st->st_atime = (time_t) s->atime;
108 st->st_mode = (mode_t) s->mode;
109 st->st_uid = (uid_t) s->uid;
110 st->st_gid = (gid_t) s->gid;
111 st->st_nlink = (nlink_t) s->nlink;
112 st->st_blksize = (blksize_t) s->blksize;

215 req.set_gid = (gid == (gid_t) -1) ? (uint32_t) -1 : (uint32_t) gid;
216
217 if (imp_exchange(&req, &rep, &fd, NULL, 0, NULL, 0) != 0) {
218 return -1;
219 }
220 if (fd >= 0) { close(fd); }
221 if (rep.status == 0) {

235ssize_t
236brix_imp_readlink(const char *reqpath, char *buf, size_t bufsz)
237{
238 imp_req_t req;
239 imp_rep_t rep;
240 int fd = -1;
241 char linkbuf[IMP_PATH_MAX];
242 size_t n;
243
244 imp_build_req(&req, IMP_OP_READLINK, reqpath, NULL, 0, 0, 0);
245 if (imp_exchange(&req, &rep, &fd, linkbuf, sizeof(linkbuf), NULL, 0) != 0) {
246 return -1;
247 }
248 if (fd >= 0) { close(fd); }
249 if (rep.status == 0 && (rep.rep_flags & IMP_REP_HAS_DATA)) {
250 n = rep.data_len;
251 if (n > bufsz) { n = bufsz; } /* truncate to caller's buffer (readlink) */
252 ngx_memcpy(buf, linkbuf, n);
253 return (ssize_t) n; /* readlinkat returns the byte count */
254 }
255 errno = (rep.status < 0) ? -rep.status
256 : imp_status_errno(rep.status ? rep.status
257 : IMP_STATUS_INTERNAL);
258 return -1;
259}

261/*
262 * imp_get_or_list — shared body for getxattr/listxattr (both return a
263 * variable-length payload via the reply-side trailing data). The broker fills
264 * its own IMP_XATTR_MAX buffer; we receive it into a process-private buffer and
265 * then honour the POSIX f*xattr contract against the caller's buffer: bufsz==0
266 * is a size probe (return the length, copy nothing); bufsz < length is ERANGE;
267 * otherwise copy and return the length. `xbuf` is static — the worker event
268 * loop is single-threaded and the broker round-trip is fully synchronous, so the
269 * call never overlaps itself; this keeps a 64 KiB buffer off the request stack.
270 */
271static ssize_t
272imp_get_or_list(uint32_t op, const char *reqpath, const char *name,
273 void *buf, size_t bufsz)
274{
275 static char xbuf[IMP_XATTR_MAX];
276 imp_req_t req;
277 imp_rep_t rep;
278 int fd = -1;
279 size_t n;
280
281 imp_build_req(&req, op, reqpath, name, 0, 0, 0);
282 if (imp_exchange(&req, &rep, &fd, xbuf, sizeof(xbuf), NULL, 0) != 0) {
283 return -1;
284 }
285 if (fd >= 0) { close(fd); }
286 if (rep.status != 0) {
287 errno = (rep.status < 0) ? -rep.status : imp_status_errno(rep.status);
288 return -1;
289 }
290 n = (rep.rep_flags & IMP_REP_HAS_DATA) ? rep.data_len : 0;
291 if (bufsz == 0) {
292 return (ssize_t) n; /* size probe */
293 }
294 if (n > bufsz) {
295 errno = ERANGE;
296 return -1;
297 }
298 ngx_memcpy(buf, xbuf, n);
299 return (ssize_t) n;
300}
