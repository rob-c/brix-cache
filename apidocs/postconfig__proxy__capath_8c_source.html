1/*
2 * postconfig_proxy_capath.c - brix_proxy_ssl_capath second-half location walk.
3 *
4 * WHAT: Post-merge half of the brix_proxy_ssl_capath directive — walks every
5 * finalised location of every server block and, where the directive is set,
6 * adds the hashed CA directory to that location's upstream (proxy_ssl) trust
7 * store. Extracted verbatim from postconfig.c so both translation units stay
8 * under the file-size guard.
9 *
10 * WHY: The upstream SSL_CTX does not exist until the stock proxy module's merge,
11 * so the directive's parse-time handler can only seed a single <hash>.N file
12 * through proxy_ssl_trusted_certificate; the whole hashed-directory lookup can
13 * only be attached here, after all merges, from the postconfiguration hook.
14 *
15 * HOW: webdav_postconf_setup_proxy_capath() (the sole entry point, declared in
16 * postconfig_internal.h) iterates cmcf->servers and drives the mutually
17 * recursive location/tree walkers below, which mirror proto_exclusive.c's walk
18 * of the finalised static-location trees / regex + named location arrays.
19 */
20

30/*
31 * webdav_postconf_proxy_capath_apply - add one location's hashed CA dir to
32 * its upstream (proxy_ssl) trust store.
33 *
34 * WHAT: For a single location conf array, when brix_proxy_ssl_capath is set,
35 * add the directory as an OpenSSL hashed lookup to the location's upstream
36 * SSL_CTX; NGX_OK when unset. Fatal (NGX_ERROR) when the location has no
37 * https proxy_pass — the directive would otherwise silently protect nothing.
38 *
39 * WHY: second half of brix_proxy_ssl_capath (see module_directives.c): the
40 * parse-time handler could only seed one <hash>.N file through the stock
41 * proxy_ssl_trusted_certificate slot, because the upstream SSL_CTX does not
42 * exist until the proxy module's merge. Here — after all merges — the ctx is
43 * live, so the whole directory becomes a verify-time hashed lookup: every CA
44 * in /etc/grid-security/certificates is trusted and IGTF package updates are
45 * picked up without a reload.
46 *
47 * HOW: reads the webdav loc-conf for the capath, then casts the proxy
48 * module's private loc-conf to its public first member
49 * (ngx_http_upstream_conf_t) to reach upstream->ssl->ctx. The ssl object is
50 * per-location here, not shared: injecting the trusted-certificate seed made
51 * this location "SSL-configured", so the proxy merge gave it its own ctx.
52 * X509_STORE_load_locations(store, NULL, dir) appends the hash-dir lookup
53 * (additive — the seeded file stays trusted). Conf tokens and
54 * ngx_conf_full_name results are NUL-terminated, so capath->data is a valid
55 * C string.
56 */
57static ngx_int_t
58webdav_postconf_proxy_capath_apply(ngx_conf_t *cf, void **loc_conf,
59 ngx_str_t *name)
60{
61 ngx_http_brix_webdav_loc_conf_t *wlcf;
62 ngx_http_upstream_conf_t *ucf;
63 X509_STORE *store;
64
65 wlcf = loc_conf[ngx_http_brix_webdav_module.ctx_index];
66 if (wlcf == NULL || wlcf->proxy_ssl_capath.len == 0) {
67 return NGX_OK;
68 }
69
70 ucf = loc_conf[ngx_http_proxy_module.ctx_index];
71
72 if (ucf == NULL || ucf->ssl == NULL || ucf->ssl->ctx == NULL) {
73 ngx_log_error(NGX_LOG_EMERG, cf->log, 0,
74 "brix_proxy_ssl_capath in location \"%V\" requires "
75 "\"proxy_pass https://...\" in the same location",
76 name);
77 return NGX_ERROR;
78 }
79
80 store = SSL_CTX_get_cert_store(ucf->ssl->ctx);
81 if (store == NULL
82 || X509_STORE_load_locations(store, NULL,
83 (const char *) wlcf->proxy_ssl_capath.data) != 1)
84 {
85 ngx_log_error(NGX_LOG_EMERG, cf->log, 0,
86 "brix_proxy_ssl_capath \"%V\": cannot add hashed "
87 "CA-directory lookup to the upstream trust store",
88 &wlcf->proxy_ssl_capath);
89 return NGX_ERROR;
90 }
91
92 ngx_log_error(NGX_LOG_INFO, cf->log, 0,
93 "brix_webdav: added hashed CA dir \"%V\" to the upstream "
94 "(proxy_ssl) trust store for location %V",
95 &wlcf->proxy_ssl_capath, name);
96 return NGX_OK;
97}

136 return NGX_ERROR;

138 if (node->inclusive != NULL
139 && webdav_postconf_proxy_capath_location(cf, node->inclusive)
140 != NGX_OK)
141 {
142 return NGX_ERROR;
143 }
144
145 if (webdav_postconf_proxy_capath_tree(cf, node->left) != NGX_OK
146 || webdav_postconf_proxy_capath_tree(cf, node->tree) != NGX_OK
147 || webdav_postconf_proxy_capath_tree(cf, node->right) != NGX_OK)
148 {
149 return NGX_ERROR;
150 }
151 return NGX_OK;

182/*
183 * webdav_postconf_setup_proxy_capath - walk every location of every server
184 * and apply the brix_proxy_ssl_capath second half where the directive is set.
185 * The directive is location-only (never merged), so the server-level conf
186 * array cannot carry it; only the finalised location structures are walked.
187 */
188ngx_int_t
189webdav_postconf_setup_proxy_capath(ngx_conf_t *cf,
190 ngx_http_core_main_conf_t *cmcf)
191{
192 ngx_http_core_srv_conf_t **cscfp = cmcf->servers.elts;
193 ngx_http_core_loc_conf_t *clcf;
194 ngx_uint_t s;
195
196 for (s = 0; s < cmcf->servers.nelts; s++) {
197 clcf = cscfp[s]->ctx->loc_conf[ngx_http_core_module.ctx_index];
198
199 if (webdav_postconf_proxy_capath_tree(cf, clcf->static_locations)
200 != NGX_OK)
201 {
202 return NGX_ERROR;
203 }
204
205#if (NGX_PCRE)
206 if (webdav_postconf_proxy_capath_loc_array(cf,
207 clcf->regex_locations) != NGX_OK)
208 {
209 return NGX_ERROR;
210 }
211#endif
212
213 if (webdav_postconf_proxy_capath_loc_array(cf,
214 cscfp[s]->named_locations) != NGX_OK)
215 {
216 return NGX_ERROR;
217 }
218 }
219
220 return NGX_OK;
221}
