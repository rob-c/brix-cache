1/*
2 * cvmfs_module_build.c — cvmfs:// enabled-branch export/backend/cache build.
3 *
4 * WHAT: The two functions that run only when a location has brix_cvmfs on:
5 * - brix_cvmfs_reject_unsupported — EMERG at config load for storage
6 * grammar cvmfs cannot honour (staging, CAS slicing, explicit writes).
7 * - cvmfs_merge_cache — the enabled-branch build: reject unsupported grammar,
8 * force read-only, rewrite the posix backend, anchor the export root
9 * (defaulting to "/"), open the confinement rootfd, register the
10 * composable http backend, stamp the cache TTL/deadline knobs, compose the
11 * cache/stage tiers, then apply T19 origin selection (geo ranks / rtt
12 * probe registration) and the coords-without-geo warning.
13 *
14 * WHY: Split out of module.c (file-size gate). This is the one part of the merge
15 * that mutates persistent export state and registers backends — a distinct
16 * concern from the pure config-field merge (cvmfs_module_merge.c). Isolating
17 * it lets the merge orchestrator gate it behind a single enable check while
18 * the build steps stay in their exact original order. Every EMERG/rejection
19 * line and the build-step order are byte-frozen against the pre-split module.c.
20 *
21 * HOW: Verbatim move. brix_cvmfs_reject_unsupported stays static (its only
22 * caller, cvmfs_merge_cache, lives here); cvmfs_merge_cache is exported via
23 * cvmfs_module_internal.h (the merge orchestrator in cvmfs_module_merge.c
24 * invokes it) and imports cvmfs_geo_rank_config from cvmfs_module_georank.c
25 * through the same header.
26 */

45/*
46 * brix_cvmfs_reject_unsupported() — EMERG at config load for storage grammar
47 * that cvmfs cannot honour.
48 *
49 * WHAT: Checks three merged common-preamble fields for features that are
50 * structurally incompatible with a read-only content-addressed site cache:
51 * (1) staging — brix_stage on / brix_stage_store, (2) object slicing —
52 * brix_cache_slice_size, and (3) explicit write permission — brix_allow_write.
53 *
54 * WHY: Without an explicit rejection at config-load time these directives
55 * silently pass nginx -t, misleading operators into believing their config
56 * is active when it is not. cvmfs is structurally read-only; writes and
57 * staging have no meaning (CAS objects are immutable whole objects), and
58 * slicing never applies because the cache fill and CAS verification run on
59 * the whole object. A loud nginx -t rejection surfaces operator mistakes
60 * before traffic is served.
61 *
62 * HOW: Called at the top of the cvmfs-enabled branch in merge_loc_conf,
63 * after brix_http_common_adopt() and ngx_http_brix_shared_merge() have
64 * resolved unified directive values into conf->common (so brix_stage,
65 * brix_allow_write, and brix_cache_slice_size are visible). Returns
66 * NGX_CONF_ERROR on any violation; NGX_CONF_OK otherwise. The caller
67 * must check the return and propagate NGX_CONF_ERROR to nginx.
68 */
69static char *
70brix_cvmfs_reject_unsupported(ngx_conf_t *cf,
71 ngx_http_brix_cvmfs_loc_conf_t *conf)
72{
73 /* Staging (brix_stage on / brix_stage_store) implies mutability: a cvmfs
74 * cache is a read-through CAS store; write-back staging makes no sense. */
75 if (conf->common.stage_enable == 1 || conf->common.stage_store.len > 0) {
76 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
77 "brix_stage/brix_stage_store: cvmfs is a read-only protocol; "
78 "staging is not supported");
79 return NGX_CONF_ERROR;
80 }
81
82 /* CAS object slicing: cvmfs objects are immutable and must be filled and
83 * verified as whole units; per-slice CAS verification is undefined. */
84 if (conf->common.cache_slice_size > 0) {
85 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
86 "brix_cache_slice_size: cvmfs CAS objects are immutable whole "
87 "objects; slicing is not supported");
88 return NGX_CONF_ERROR;
89 }
90
91 /* Explicit write permission: the hard force (conf->common.allow_write = 0)
92 * below already prevents accidental writes, but an explicit on is a
93 * config mistake that deserves a loud rejection rather than silent no-op. */
94 if (conf->common.allow_write == 1) {
95 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
96 "brix_allow_write on: cvmfs is a read-only protocol; "
97 "write permission cannot be granted");
98 return NGX_CONF_ERROR;
99 }
100
101 return NGX_CONF_OK;
102}

104/*
105 * cvmfs_merge_cache() — the cvmfs-enabled export/backend/cache build block.
106 *
107 * WHAT: When enable=1: reject unsupported storage grammar, force read-only,
108 * rewrite the posix backend, anchor the export root (defaulting to "/"),
109 * open the confinement rootfd, register the composable http backend, stamp
110 * the cache TTL/deadline knobs, compose the cache/stage tiers, then apply the
111 * T19 origin-selection (geo ranks / rtt probe registration) and the
112 * coords-without-geo warning.
113 *
114 * WHY: This is the one part of the merge that mutates persistent export state
115 * and registers backends; it only runs for cvmfs-enabled locations. Splitting
116 * it out lets the orchestrator gate it with a single early return while the
117 * build steps stay in their exact original order.
118 *
119 * HOW: Verbatim move of the `if (conf->cvmfs.enable)` block; returns
120 * NGX_CONF_ERROR on any step failure (identical EMERG/rejection lines), else
121 * NGX_CONF_OK. The caller only invokes it when enable is set.
122 */
123char *
124cvmfs_merge_cache(ngx_conf_t *cf, ngx_http_brix_cvmfs_loc_conf_t *conf)
125{
126 brix_export_root_opts_t root_opts;
127
128 /* Reject storage grammar cvmfs cannot honour before doing anything
129 * else: staging, slicing, and explicit writes make no sense for a
130 * read-only content-addressed site cache. */
131 if (brix_cvmfs_reject_unsupported(cf, conf) != NGX_CONF_OK) {
132 return NGX_CONF_ERROR;
133 }
134
135 /* CVMFS is a read-only protocol: no directive can enable writes. */
136 conf->common.allow_write = 0;
137
138 /* "posix:<path>" backend names the local export tree (composable
139 * brix_root replacement) — same rewrite every protocol applies. */
140 brix_storage_backend_posix_root(&conf->common);
141
142 /* Pure cache node (the normal CVMFS shape): no local export tree —
143 * anchor the namespace at "/" exactly like the stream plane's pure
144 * cache node; the location serves the "/" namespace, filling from
145 * the http backend into the cache store. */
146 if (conf->common.root.len == 0) {
147 ngx_str_set(&conf->common.root, "/");
148 }
149
150 root_opts.directive_name = "brix_cvmfs";
151 root_opts.allow_write = 0;
152 root_opts.required = 1;
153 root_opts.canon_size = sizeof(conf->common.root_canon);
154 if (brix_prepare_export_root(cf, &conf->common.root, &root_opts,
155 conf->common.root_canon) != NGX_CONF_OK)
156 {
157 return NGX_CONF_ERROR;
158 }
159
160 /* Persistent confinement rootfd (openat2 RESOLVE_BENEATH anchor). */
161 if (brix_http_open_rootfd(cf, &conf->common) != NGX_CONF_OK) {
162 return NGX_CONF_ERROR;
163 }
164
165 /* Register the composable storage backend (phase-63): the http(s)
166 * Stratum-1 origin URL routes every VFS op to sd_http. */
167 if (brix_vfs_backend_config_str(cf, conf->common.root_canon,
168 &conf->common.storage_backend, conf->common.pblock_block_size,
169 BRIX_AF_AUTO)
170 != NGX_OK)
171 {
172 return NGX_CONF_ERROR;
173 }
174
175 /* Verify-mismatch evidence lands in the protocol's quarantine dir;
176 * MANIFEST-class fills get the protocol's TTL stamped (T12). */
177 conf->common.cache_quarantine_dir = conf->cvmfs.quarantine_dir;
178 conf->common.cache_manifest_ttl = conf->cvmfs.manifest_ttl;
179 conf->common.cache_offline_ttl = conf->cvmfs.offline_ttl;
180 /* phase-85 F1: the master-key PATH rides the common preamble; the tier
181 * registration loads the PEM once at config time into the cache policy. */
182 conf->common.cache_cvmfs_master_key = conf->cvmfs.master_key;
183 /* T20 never-drop deadlines for the shared fill machinery */
184 conf->common.cache_client_hold = conf->cvmfs.client_hold;
185 conf->common.cache_fill_max_life = conf->cvmfs.fill_max_life;
186
187 /* Phase-64: compose the cache/stage tiers over the backend. */
188 if (brix_tier_register_stores(cf, &conf->common) != NGX_OK) {
189 return NGX_CONF_ERROR;
190 }
191
192 /* T19 origin selection: geo ranks compute once at config time;
193 * rtt registers the per-worker probe; static keeps configured
194 * order (all ranks 0 — the pick is order-stable on ties). */
195 if (conf->cvmfs.origin_select == BRIX_CVMFS_SELECT_GEO) {
196 if (cvmfs_geo_rank_config(cf, conf) != NGX_CONF_OK) {
197 return NGX_CONF_ERROR;
198 }
199 } else if (conf->cvmfs.origin_select == BRIX_CVMFS_SELECT_RTT) {
200 brix_cvmfs_rtt_register(conf->common.root_canon,
201 conf->cvmfs.rtt_interval,
202 &conf->common.thread_pool_name);
203 }
204
205 /* WARN (not NOTICE — config-parse NOTICE is dropped at cf->log ERR
206 * level) when coordinates were supplied but origin_select is not geo:
207 * the coordinates are silently ignored in all other modes and the
208 * operator may not realise their geo config has no effect. */
209 if (conf->cvmfs.origin_select != BRIX_CVMFS_SELECT_GEO
210 && conf->cvmfs.origin_coords != NULL)
211 {
212 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
213 "brix_cvmfs_origin_coords set but brix_cvmfs_origin_select "
214 "is not geo — coordinates are ignored");
215 }
216 return NGX_CONF_OK;
217}
