12/* File: recv.c — TCP read-event loop and request framing state machine
13 * WHAT: The core async recv loop that drives the XRootD protocol lifecycle on
14 * each TCP connection. It frames the 20-byte ClientInitHandShake, the 24-byte
15 * ClientRequestHdr, and payload bytes into a deterministic state machine —
16 * HANDSHAKE → REQ_HEADER → REQ_PAYLOAD → dispatch, with suspend states for
17 * SENDING/AIO/UPSTREAM/TLS. Security invariant: dlen must pass
18 * brix_max_payload_for_request() BEFORE any allocation.
19 *
20 * WHY: Every XRootD request flows through this framing layer before dispatch, so
21 * dispatch always receives a complete, validated request with streamid/reqid/
22 * dlen extracted from the wire header.
23 *
24 * HOW: This file holds only the event-loop skeleton (pre-loop teardown gate,
25 * handoff gate, and the for-loop that sequences read → process). The per-PDU
26 * framing phases live in recv_frame.c and return a brix_recv_step_t telling the
27 * loop whether to continue / return / break.
28 */
29
30/* Pre-loop teardown/deadline gate for the recv handler: handles a graceful-
31 * shutdown close (c->close) and the three read-timeout cases (WAITING_CMS,
32 * WAITING_FRM, and the steady-state read deadline). Returns 1 when it has
33 * finalized or parked the connection and the caller must return; 0 to proceed
34 * into the recv loop. Every original early-return path maps to `return 1`. */
35static int
36brix_recv_pre_loop(ngx_stream_session_t *s, ngx_connection_t *c,
37 brix_ctx_t *ctx, ngx_event_t *rev)
38{
39
40 /*
41 * Graceful shutdown signal: ngx_close_idle_connections() set c->close on a
42 * connection we had marked idle. Tear it down through the normal disconnect
43 * funnel — a clean FIN is the correct retry signal: the client's resilient
44 * layer treats it as a transport sever and reconnects to the new worker,
45 * resuming the transfer from its last offset. (kXR_wait would stall the
46 * client ≥1s on the dying worker; a self-redirect trips its loop guard.)
47 */
48 if (c->close) {
49 if (brix_defer_teardown_if_writing(ctx, c, NGX_STREAM_OK)) {
50 return 1;
51 }
52 brix_on_disconnect(ctx, c);
53 brix_close_all_files(ctx);
54 ngx_stream_finalize_session(s, NGX_STREAM_OK);
55 return 1;
56 }
57
58 if (rev->timedout) {
59 if (ctx->state == XRD_ST_WAITING_CMS) {
60 /* kYR_select did not arrive in time - tell client to retry. */
61 rev->timedout = 0;
62 brix_pending_remove(ctx->cms_wait_streamid, ngx_pid);
63 ctx->state = XRD_ST_REQ_HEADER;
64 brix_send_wait(ctx, c, 5);
65 brix_schedule_read_resume(c);
66 return 1;
67 }
68 if (ctx->state == XRD_ST_WAITING_FRM) {
69 /* The async recall took longer than stage_ttl — drop the parked
70 * waiter and ask the client to retry (it will re-poll residency:
71 * a hit if staged, or a fresh park otherwise). */
72 rev->timedout = 0;
73 brix_stage_waiter_drop_conn(c->fd, c->number);
74 ctx->state = XRD_ST_REQ_HEADER;
75 brix_send_wait(ctx, c, 5);
76 brix_schedule_read_resume(c);
77 return 1;
78 }
79 ngx_log_error(NGX_LOG_INFO, c->log, NGX_ETIMEDOUT,
80 "brix: client connection timed out");
81 /* Phase 39: our steady-state read deadline fired — it is the only c->read
82 * timer armed outside the WAITING_CMS/FRM states handled above. Attribute
83 * it (pre-auth handshake vs in-flight PDU) and tear down via the single
84 * disconnect funnel. */
85 ctx->deadline.read_armed = 0;
86 if (ctx->login.auth_done) {
87 BRIX_SRV_METRIC_INC(ctx, read_pdu_timeouts_total);
88 } else {
89 BRIX_SRV_METRIC_INC(ctx, handshake_timeouts_total);
90 }
91 if (brix_defer_teardown_if_writing(ctx, c, NGX_STREAM_OK)) {
92 return 1;
93 }
94 brix_on_disconnect(ctx, c);
95 brix_close_all_files(ctx);
96 ngx_stream_finalize_session(s, NGX_STREAM_OK);
97 return 1;
98 }
99
100
101 return 0;
102}

104/* Result of the recv-loop handoff gate (brix_recv_handoff_state). */
105enum {
106 BRIX_RECV_PROCEED = 0, /* not a handoff state — read/parse a PDU */
107 BRIX_RECV_RETURN, /* connection yielded — return from the handler */
108 BRIX_RECV_BREAK /* read-event re-arm failed — break the recv loop */
109};
110
111/* Connection-handoff gate at the top of the recv loop: when the connection is
112 * currently owned by another subsystem (SENDING / AIO / UPSTREAM / PROXY /
113 * WAITING_CMS|FRM / TLS_HANDSHAKE) the recv loop must yield rather than read
114 * more bytes. The four "yield until its event fires" states share one body
115 * (re-arm the read event, then return). Returns BRIX_RECV_RETURN to return
116 * from the handler, BRIX_RECV_BREAK to break the loop, or BRIX_RECV_PROCEED
117 * when no handoff applies and the caller should read/parse the next PDU. */
118static int
119brix_recv_handoff_state(brix_ctx_t *ctx, ngx_event_t *rev)
120{
121 if (ctx->state == XRD_ST_SENDING || ctx->state == XRD_ST_TLS_HANDSHAKE) {
122 return BRIX_RECV_RETURN;
123 }
124
125 if (ctx->state == XRD_ST_AIO
126 || ctx->state == XRD_ST_UPSTREAM
127 || ctx->state == XRD_ST_PROXY
128 || ctx->state == XRD_ST_WAITING_CMS
129 || ctx->state == XRD_ST_WAITING_FRM)
130 {
131 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
132 return BRIX_RECV_BREAK;
133 }
134 return BRIX_RECV_RETURN;
135 }
136
137 return BRIX_RECV_PROCEED;
138}

140/*
141 * WHAT: the core async recv loop that drives the XRootD protocol lifecycle on
142 * each TCP connection. Called by nginx whenever data is available or a timeout
143 * fires.
144 *
145 * HOW: after the pre-loop teardown/deadline gate, the for-loop repeatedly (1)
146 * runs a deferred non-pipelinable request once the queues drain, (2) yields on a
147 * handoff state, (3) reads the next PDU chunk, and (4) processes a completed PDU
148 * — each phase in recv_frame.c and reporting back a brix_recv_step_t. The read
149 * deadline is disarmed at the single point between read-complete and process, so
150 * the timer is never live across a sub-system handoff (the UAF rule).
151 */
152void
153ngx_stream_brix_recv(ngx_event_t *rev)
154{
155 ngx_connection_t *c;
156 ngx_stream_session_t *s;
157 ngx_stream_brix_srv_conf_t *conf;
158 brix_ctx_t *ctx;
159 size_t rx_pending;
160
161 c = rev->data;
162 s = c->data;
163 ctx = ngx_stream_get_module_ctx(s, ngx_stream_brix_module);
164 conf = ngx_stream_get_module_srv_conf(s, ngx_stream_brix_module);
165
166 /*
167 * Fast teardown: we are about to service this connection, so clear the idle
168 * marker for the duration of this handler run. It is re-set at the request
169 * boundary (brix_recv_header_prep) whenever we park waiting for the next
170 * request, so a graceful quit's ngx_close_idle_connections() can drop a
171 * parked keepalive at once instead of holding it until worker exit.
172 */
173 c->idle = 0;
174 if (brix_recv_pre_loop(s, c, ctx, rev)) {
175 return;
176 }
177
178 rx_pending = 0;
179
180 for (;;) {
181 brix_recv_step_t step;
182
183 /* Phase 29 drain barrier: run a fully-read non-pipelinable request once
184 * the response/ack queue and in-flight pwrites have drained. */
185 if (ctx->out.recv_deferred) {
186 step = brix_recv_run_deferred(s, c, conf, ctx, rev, &rx_pending);
187 if (step == BRIX_RECV_STEP_RETURN) {
188 return;
189 }
190 if (step == BRIX_RECV_STEP_BREAK) {
191 break;
192 }
193 continue;
194 }
195
196 {
197 int hr = brix_recv_handoff_state(ctx, rev);
198 if (hr == BRIX_RECV_RETURN) {
199 return;
200 }
201 if (hr == BRIX_RECV_BREAK) {
202 break; /* breaks the recv for-loop */
203 }
204 /* BRIX_RECV_PROCEED: not a handoff state — read/parse a PDU. */
205 }
206
207 step = brix_recv_read_frame(s, c, conf, ctx, rev, &rx_pending);
208 if (step == BRIX_RECV_STEP_RETURN) {
209 return;
210 }
211 if (step == BRIX_RECV_STEP_BREAK) {
212 break;
213 }
214 if (step == BRIX_RECV_STEP_CONTINUE) {
215 continue;
216 }
217
218 /* Phase 39: a full PDU unit just arrived, so the read obligation is
219 * satisfied — disarm the deadline before any dispatch, which may hand the
220 * connection off to AIO/SENDING/UPSTREAM/PROXY/WAITING_*. Idempotent on
221 * the healthy pipelined path where the timer was never armed. This is
222 * the single point guaranteeing the read timer is never live across a
223 * sub-system handoff (the UAF rule). */
224 brix_disarm_read_deadline(c, ctx);
225
226 step = brix_recv_process_frame(s, c, conf, ctx, rev, &rx_pending);
227 if (step == BRIX_RECV_STEP_RETURN) {
228 return;
229 }
230 if (step == BRIX_RECV_STEP_BREAK) {
231 break;
232 }
233 /* BRIX_RECV_STEP_CONTINUE / _NEXT: proceed to the next loop iteration. */
234 }
235
236 if (brix_defer_teardown_if_writing(ctx, c,
237 NGX_STREAM_INTERNAL_SERVER_ERROR)) {
238 return;
239 }
240 brix_on_disconnect(ctx, c);
241 brix_close_all_files(ctx);
242 ngx_stream_finalize_session(s, NGX_STREAM_INTERNAL_SERVER_ERROR);
243}
