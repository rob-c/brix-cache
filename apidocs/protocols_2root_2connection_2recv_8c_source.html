11/* File: recv.c — TCP read-event loop and request framing state machine
12 * WHAT: The core async recv loop that drives the XRootD protocol lifecycle on each TCP connection. Frames handshake (20-byte ClientInitHandShake), request headers (24-byte ClientRequestHdr), and payload bytes into a deterministic four-state machine — HANDSHAKE → REQ_HEADER → REQ_PAYLOAD → dispatch, with suspend states for SENDING/AIO/UPSTREAM/TLS. Security invariant: dlen must pass brix_max_payload_for_request() BEFORE any allocation. Timeout handling: CMS wait timeout sends retry response; other timeouts disconnect. Thread safety: single-owner per connection on nginx event thread — no locking required.
13 *
14 * WHY: Every XRootD request flows through recv.c's state machine before dispatch to opcode handlers. The recv loop ensures correct byte-level framing (handshake → header → payload) so dispatch receives a complete, validated request with streamid/reqid/dlen extracted from the wire header. Without this framing layer, dispatch would receive partial or misaligned data causing protocol errors.
15 *
16 * HOW: On each read-ready event, recv checks ctx->state to determine what bytes are expected (handshake=20, header=24, payload=dlen), calls c->recv(), accumulates into hdr_buf/payload_buf, dispatches when complete, then resets state for the next request. Suspend states (SENDING/AIO/UPSTREAM/TLS) return immediately without reading.
17 * */

19/* brix_max_payload_for_request — return per-opcode payload size limit
20 * WHAT: Security guard that defines the maximum allowed wire payload for each request opcode. Called BEFORE any allocation to prevent memory exhaustion attacks — clients sending dlen > this limit are disconnected immediately. Limits vary by opcode type: write/pgwrite/chkpoint -> BRIX_MAX_WRITE_PAYLOAD; readv -> segments x segsize; auth -> BRIX_MAX_AUTH_PAYLOAD (16KB); prepare -> BRIX_MAX_PREPARE_PAYLOAD; all others -> path + 64 bytes. This is the first line of defense against oversized payloads. */
21static uint32_t
22brix_max_payload_for_request(uint16_t reqid)
23{
24 if (reqid == kXR_pgwrite || reqid == kXR_write || reqid == kXR_writev
25 || reqid == kXR_chkpoint) {
26 return BRIX_MAX_WRITE_PAYLOAD;
27 }
28
29 if (reqid == kXR_readv) {
30 /* Each segment is BRIX_READV_SEGSIZE (16) bytes. */
31 return BRIX_READV_MAXSEGS * BRIX_READV_SEGSIZE;
32 }
33
34 if (reqid == kXR_auth) {
35 return BRIX_MAX_AUTH_PAYLOAD;
36 }
37
38 if (reqid == kXR_prepare) {
39 return BRIX_MAX_PREPARE_PAYLOAD;
40 }
41
42 /* All other requests carry only a path (BRIX_MAX_PATH) plus a small
43 * fixed-size body. The +64 covers opcode-specific extras (e.g. the
44 * kXR_login info field that follows the username in the payload). */
45 return BRIX_MAX_PATH + 64;
46}

48/* brix_ensure_payload_buffer: allocate/resize payload buffer for incoming request data
49 * WHAT: Ensures ctx->recv.payload_buf has sufficient capacity (dlen + 1 bytes) and points ctx->recv.payload to it. The extra byte is pre-zeroed as a C-string NUL terminator guarantee. Uses heap allocation (ngx_alloc), NOT pool-allocated, so the buffer can grow across multiple requests on the same connection without fragmenting the nginx pool. On resize: frees old buffer, allocates new one. Returns NGX_OK if capacity sufficient or successfully resized, NGX_ERROR on overflow check failure or allocation failure.
50 *
51 * WHY: Payload buffers must survive across multiple requests on a persistent TCP connection (pool-allocated buffers would fragment under repeated large allocations). Heap allocation (ngx_alloc) allows resizing without pool pressure and is freed explicitly when the connection closes via brix_on_disconnect(). The pre-zeroed NUL byte ensures ctx->recv.payload always has a valid C-string terminator even for zero-length payloads.
52 *
53 * HOW: First checks if existing buffer has sufficient capacity (ctx->recv.payload_buf_size >= need), reuses it and sets payload pointer. If insufficient, frees old buffer via ngx_free(), allocates new one via ngx_alloc(need), sets ctx->recv.payload_buf/payload_buf_size/payload, pre-zeroes byte at position dlen. Returns NGX_OK or NGX_ERROR. */
54static ngx_int_t
55brix_ensure_payload_buffer(brix_ctx_t *ctx, ngx_connection_t *c,
56 uint32_t dlen)
57{
58 u_char *buf;
59 size_t need;
60
61 if (dlen > (uint32_t) (SIZE_MAX - 1)) {
62 return NGX_ERROR;
63 }
64 need = (size_t) dlen + 1;
65
66 if (ctx->recv.payload_buf != NULL && ctx->recv.payload_buf_size >= need) {
67 ctx->recv.payload = ctx->recv.payload_buf;
68 ctx->recv.payload[dlen] = '\0';
69 return NGX_OK;
70 }
71
72 buf = ngx_alloc(need, c->log);
73 if (buf == NULL) {
74 return NGX_ERROR;
75 }
76
77 if (ctx->recv.payload_buf != NULL) {
78 ngx_free(ctx->recv.payload_buf);
79 }
80
81 ctx->recv.payload_buf = buf;
82 ctx->recv.payload_buf_size = need;
83 ctx->recv.payload = buf;
84 ctx->recv.payload[dlen] = '\0';
85
86 return NGX_OK;
87}

89/* brix_grow_payload_buffer — enlarge payload_buf PRESERVING received bytes
90 * WHAT: Grows the payload buffer to hold dlen (+1 NUL guard) bytes while keeping
91 * the payload_pos bytes already accumulated. Used mid-request by the kXR_writev
92 * body extension, which raises the expected body length AFTER the descriptor
93 * block has been received into the buffer.

104

134 /*
135 *
136 * WHAT: The core async recv loop that drives the XRootD protocol lifecycle on each TCP connection. Called by nginx whenever data is available or timeout fires.
137 *
138 * WHY: Every XRootD request requires byte-level framing before dispatch — handshake validates client identity, header extracts routing fields (streamid/reqid), payload accumulates opcode-specific data. Without this loop, dispatch would receive partial or misaligned wire data causing protocol errors. The suspend-state design prevents recv from reading while send/AIO/upstream subsystems own the connection.
139 *
140 * HOW: On each read-ready event, checks ctx->state to determine expected bytes (handshake=20, header=24, payload=dlen), calls c->recv(), accumulates into hdr_buf/payload_buf, dispatches when complete, resets state for next request. Suspend states return immediately without reading. Timeout handling: CMS wait timeout sends retry response; other timeouts disconnect.
141 * */
142/* Pre-loop teardown/deadline gate for the recv handler: handles a graceful-
143 * shutdown close (c->close) and the three read-timeout cases (WAITING_CMS,
144 * WAITING_FRM, and the steady-state read deadline). Returns 1 when it has
145 * finalized or parked the connection and the caller must return; 0 to proceed
146 * into the recv loop. Every original early-return path maps to `return 1`. */
147static int
148brix_recv_pre_loop(ngx_stream_session_t *s, ngx_connection_t *c,
149 brix_ctx_t *ctx, ngx_event_t *rev)
150{
151
152 /*
153 * Graceful shutdown signal: ngx_close_idle_connections() set c->close on a
154 * connection we had marked idle. Tear it down through the normal disconnect
155 * funnel — a clean FIN is the correct retry signal: the client's resilient
156 * layer treats it as a transport sever and reconnects to the new worker,
157 * resuming the transfer from its last offset. (kXR_wait would stall the
158 * client ≥1s on the dying worker; a self-redirect trips its loop guard.)
159 */
160 if (c->close) {
161 if (brix_defer_teardown_if_writing(ctx, c, NGX_STREAM_OK)) {
162 return 1;
163 }
164 brix_on_disconnect(ctx, c);
165 brix_close_all_files(ctx);
166 ngx_stream_finalize_session(s, NGX_STREAM_OK);
167 return 1;
168 }
169
170 if (rev->timedout) {
171 if (ctx->state == XRD_ST_WAITING_CMS) {
172 /* kYR_select did not arrive in time - tell client to retry. */
173 rev->timedout = 0;
174 brix_pending_remove(ctx->cms_wait_streamid, ngx_pid);
175 ctx->state = XRD_ST_REQ_HEADER;
176 brix_send_wait(ctx, c, 5);
177 brix_schedule_read_resume(c);
178 return 1;
179 }
180 if (ctx->state == XRD_ST_WAITING_FRM) {
181 /* The async recall took longer than stage_ttl — drop the parked
182 * waiter and ask the client to retry (it will re-poll residency:
183 * a hit if staged, or a fresh park otherwise). */
184 rev->timedout = 0;
185 brix_stage_waiter_drop_conn(c->fd, c->number);
186 ctx->state = XRD_ST_REQ_HEADER;
187 brix_send_wait(ctx, c, 5);
188 brix_schedule_read_resume(c);
189 return 1;
190 }
191 ngx_log_error(NGX_LOG_INFO, c->log, NGX_ETIMEDOUT,
192 "brix: client connection timed out");
193 /* Phase 39: our steady-state read deadline fired — it is the only c->read
194 * timer armed outside the WAITING_CMS/FRM states handled above. Attribute
195 * it (pre-auth handshake vs in-flight PDU) and tear down via the single
196 * disconnect funnel. */
197 ctx->deadline.read_armed = 0;
198 if (ctx->login.auth_done) {
199 BRIX_SRV_METRIC_INC(ctx, read_pdu_timeouts_total);
200 } else {
201 BRIX_SRV_METRIC_INC(ctx, handshake_timeouts_total);
202 }
203 if (brix_defer_teardown_if_writing(ctx, c, NGX_STREAM_OK)) {
204 return 1;
205 }
206 brix_on_disconnect(ctx, c);
207 brix_close_all_files(ctx);
208 ngx_stream_finalize_session(s, NGX_STREAM_OK);
209 return 1;
210 }
211
212
213 return 0;
214}

216/* Result of the recv-loop handoff gate (brix_recv_handoff_state). */
217enum {
218 BRIX_RECV_PROCEED = 0, /* not a handoff state — read/parse a PDU */
219 BRIX_RECV_RETURN, /* connection yielded — return from the handler */
220 BRIX_RECV_BREAK /* read-event re-arm failed — break the recv loop */
221};

223/* Connection-handoff gate at the top of the recv loop: when the connection is
224 * currently owned by another subsystem (SENDING / AIO / UPSTREAM / PROXY /
225 * WAITING_CMS|FRM / TLS_HANDSHAKE) the recv loop must yield rather than read
226 * more bytes. The four "yield until its event fires" states share one body
227 * (re-arm the read event, then return). Returns BRIX_RECV_RETURN to return
228 * from the handler, BRIX_RECV_BREAK to break the loop, or BRIX_RECV_PROCEED
229 * when no handoff applies and the caller should read/parse the next PDU. */
230static int
231brix_recv_handoff_state(brix_ctx_t *ctx, ngx_event_t *rev)
232{
233 if (ctx->state == XRD_ST_SENDING || ctx->state == XRD_ST_TLS_HANDSHAKE) {
234 return BRIX_RECV_RETURN;
235 }
236
237 if (ctx->state == XRD_ST_AIO
238 || ctx->state == XRD_ST_UPSTREAM
239 || ctx->state == XRD_ST_PROXY
240 || ctx->state == XRD_ST_WAITING_CMS
241 || ctx->state == XRD_ST_WAITING_FRM)
242 {
243 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
244 return BRIX_RECV_BREAK;
245 }
246 return BRIX_RECV_RETURN;
247 }
248
249 return BRIX_RECV_PROCEED;
250}

252void
253ngx_stream_brix_recv(ngx_event_t *rev)
254{
255 ngx_connection_t *c;
256 ngx_stream_session_t *s;
257 ngx_stream_brix_srv_conf_t *conf;
258 brix_ctx_t *ctx;
259 ssize_t n;
260 ngx_int_t rc;
261 size_t rx_pending;
262
263 c = rev->data;
264 s = c->data;
265 ctx = ngx_stream_get_module_ctx(s, ngx_stream_brix_module);
266 conf = ngx_stream_get_module_srv_conf(s, ngx_stream_brix_module);
267
268 /*
269 * Fast teardown: we are about to service this connection, so clear the
270 * idle marker for the duration of this handler run. It is re-set at the
271 * request boundary below whenever we park waiting for the next request, so
272 * that a graceful quit's ngx_close_idle_connections() can drop a parked
273 * keepalive connection at once instead of holding it until worker exit.
274 */
275 c->idle = 0;
276 if (brix_recv_pre_loop(s, c, ctx, rev)) {
277 return;
278 }
279
280 rx_pending = 0;
281
282 for (;;) {
283 u_char *dest;
284 size_t need;
285 size_t avail;
286
287 if (ctx->out.recv_deferred) {
288 /*
289 * Phase 29 drain barrier: a non-pipelinable request was fully read
290 * (header in cur_*, any payload in payload_buf) while pipelined reads
291 * OR writes were still in flight. It must run with the connection
292 * quiescent, so keep it parked until BOTH the ack/response queue
293 * (out_count) and the in-flight pwrites (wr_inflight) have drained — a
294 * write completion (schedule_read_resume) or the send-side drain
295 * re-enters recv to re-check. This is what lets a kXR_close safely
296 * follow a burst of pipelined writes: every pwrite has landed before
297 * the handle is retired.
298 */
299 if (ctx->out.count > 0 || ctx->out.wr_inflight > 0) {
300 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
301 break;
302 }
303 return;
304 }
305 ctx->out.recv_deferred = 0;
306 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, rx_pending);
307 rx_pending = 0;
308 rc = brix_dispatch(ctx, c, conf);
309 if (rc == NGX_ERROR) {
310 break;
311 }
312 if (ctx->state == XRD_ST_AIO) {
313 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
314 break;
315 }
316 return;
317 }
318 if (ctx->state != XRD_ST_SENDING) {
319 if (ctx->tls_pending) {
320 brix_start_tls(ctx, c, conf);
321 return;
322 }
323 ctx->state = XRD_ST_REQ_HEADER;
324 ctx->recv.hdr_pos = 0;
325 }
326 continue;
327 }
328
329 {
330 int hr = brix_recv_handoff_state(ctx, rev);
331 if (hr == BRIX_RECV_RETURN) {
332 return;
333 }
334 if (hr == BRIX_RECV_BREAK) {
335 break; /* breaks the recv for-loop */
336 }
337 /* BRIX_RECV_PROCEED: not a handoff state — read/parse a PDU. */
338 }
339
340 if (ctx->state == XRD_ST_HANDSHAKE) {
341 dest = ctx->recv.hdr_buf + ctx->recv.hdr_pos;
342 need = XRD_HANDSHAKE_LEN - ctx->recv.hdr_pos;
343
344 } else if (ctx->state == XRD_ST_REQ_HEADER) {
345 if (ctx->recv.hdr_pos == 0) {
346 /*
347 * Phase 29 pipelining backpressure: if pipeline_depth
348 * responses are already queued/draining (out_count) OR pwrites
349 * are still in flight (wr_inflight, write pipelining), stop
350 * reading new requests and suspend with state=SENDING. The write
351 * event re-enters recv once the output queue drains; a write
352 * completion queues its ack (which schedules that write event) and
353 * also nudges the read side. Bounding out_count + wr_inflight at
354 * ctx->out.pipeline_depth caps BOTH the in-flight pwrites and the
355 * queued acks, so the out_ring (pipeline_depth slots) can
356 * never overflow. (Both counters are 0 here for the common serial
357 * read/write case, so this is a no-op there.)
358 */
359 if (ctx->out.count + ctx->out.wr_inflight >= ctx->out.pipeline_depth) {
360 ctx->state = XRD_ST_SENDING;
361 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, rx_pending);
362 return;
363 }
364
365 /*
366 * Top of a fresh request with the queue empty — the previous
367 * response is fully sent, so no chain references the scratch
368 * buffers.
369 *
370 * Phase 31: trim any scratch a prior large read/pgwrite grew
371 * back to the streaming window so an idle session does not pin
372 * request-max heap. Gated on out_count==0 so a pipelined read
373 * still in flight (whose sendfile chain is file-backed, not
374 * scratch-backed) is never disturbed. The scratch buffers are
375 * RAW heap (ngx_alloc/ngx_free, see src/aio/buffers.c), which
376 * makes the trim safe.
377 */
378 if (ctx->out.count == 0 && ctx->out.wr_inflight == 0) {
379 brix_trim_scratch(ctx, c);
380
381 /* Phase 31 W4: reconcile this connection's transfer-heap
382 * footprint with the SHM-global budget after the trim. */
383 brix_budget_sync(ctx);
384
385 /*
386 * Fast teardown: the worker is draining and this connection
387 * is quiescent at a fresh request boundary (no queued
388 * response, no in-flight pwrite — so teardown is safe).
389 * Close it now rather than parking on the next request; the
390 * client reconnects to the new worker and resumes. This
391 * also catches a connection that finished its current op
392 * *after* the quit began (which ngx_close_idle_connections,
393 * a one-shot at quit, would have missed).
394 *
395 * BUT only when no file is open: a connection holding an open
396 * handle is mid-transfer (a streaming read parked between
397 * kXR_read chunks, or a cache slice-fill in progress). Forcing
398 * a reconnect there loses the in-flight fill and surfaces as a
399 * spurious kXR_NotFound to the client — especially through a
400 * proxy/cache tier, where the reconnect must re-open and resume
401 * mid-stream. Standard graceful-reload semantics: let the
402 * active transfer finish on the old worker (bounded by nginx's
403 * worker_shutdown_timeout backstop); tear down only once it is
404 * a true idle keepalive with no open handles.
405 */
406 if (ngx_exiting && !brix_ctx_has_open_file(ctx)) {
407 brix_on_disconnect(ctx, c);
408 brix_close_all_files(ctx);
409 ngx_stream_finalize_session(s, NGX_STREAM_OK);
410 return;
411 }
412
413 /*
414 * Mark idle so a graceful quit's ngx_close_idle_connections()
415 * drops this parked keepalive connection immediately instead
416 * of leaving it open until the worker finally exits. Cleared
417 * at the top of this handler the moment we service it again.
418 * An open handle means an active transfer, not a keepalive
419 * idle — leave it un-idle so the reload lets it complete.
420 */
421 c->idle = brix_ctx_has_open_file(ctx) ? 0 : 1;
422 }
423 }
424 dest = ctx->recv.hdr_buf + ctx->recv.hdr_pos;
425 need = XRD_REQUEST_HDR_LEN - ctx->recv.hdr_pos;
426
427 } else {
428 /* cur_body_extra is non-zero only for kXR_writev (segment data
429 * streams after the dlen-framed descriptor block) and for
430 * kXR_chkpoint/ckpXeq (the sub-request body streams after the
431 * dlen-framed embedded sub-header). */
432 dest = ctx->recv.payload + ctx->recv.payload_pos;
433 need = (size_t) ctx->recv.cur_dlen + ctx->recv.cur_body_extra
434 - ctx->recv.payload_pos;
435 }
436
437 if (need > 0) {
438 rev->available = -1;
439 n = c->recv(c, dest, need);
440
441 if (n == NGX_AGAIN) {
442 ngx_log_debug(NGX_LOG_DEBUG_STREAM, c->log, 0,
443 "brix: recv AGAIN st=%d hdr_pos=%uz avail=%d"
444 " ready=%d active=%d",
445 (int) ctx->state, ctx->recv.hdr_pos,
446 rev->available, (int) rev->ready,
447 (int) rev->active);
448 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
449 break;
450 }
451 /* Phase 39: genuine incompletion — we are suspending to wait for
452 * the rest of this PDU (or the unauth phase to finish). Arm the
453 * read deadline once; it is idempotent so repeated partial reads
454 * of the same PDU do NOT reset it (the deadline bounds time-to-
455 * complete, defeating a 1-byte-per-readWait slowloris). */
456 brix_arm_read_deadline(c, ctx);
457 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, rx_pending);
458 return;
459 }
460
461 if (n == NGX_ERROR || n == 0) {
462 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, c->log, 0,
463 "brix: client disconnected");
464 if (brix_defer_teardown_if_writing(ctx, c, NGX_STREAM_OK)) {
465 return;
466 }
467 brix_on_disconnect(ctx, c);
468 brix_close_all_files(ctx);
469 ngx_stream_finalize_session(s, NGX_STREAM_OK);
470 return;
471 }
472
473 avail = (size_t) n;
474 rx_pending += avail;
475
476 if (ctx->state == XRD_ST_HANDSHAKE) {
477 ctx->recv.hdr_pos += avail;
478
479 if (ctx->recv.hdr_pos >= 1 && ctx->recv.hdr_buf[0] != 0) {
480 /*
481 * Non-XRootD first byte (HTTP method letter / TLS 0x16; the
482 * XRootD client hello always begins with a zero streamid
483 * word). If brix_http_handoff is configured, splice this
484 * connection to the local HTTP/WebDAV listener so one
485 * registered port serves both protocols (see handoff.c);
486 * otherwise close as before.
487 */
488 if (conf->http_handoff_addr != NULL
489 && brix_http_handoff_start(s, c, conf,
490 ctx->recv.hdr_buf, ctx->recv.hdr_pos) == NGX_OK)
491 {
492 return; /* the relay owns the connection now */
493 }
494 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
495 "brix: non-XRootD client (first byte 0x%02xd)"
496 " closing immediately",
497 (unsigned) ctx->recv.hdr_buf[0]);
498 break;
499 }
500
501 } else if (ctx->state == XRD_ST_REQ_HEADER) {
502 ctx->recv.hdr_pos += avail;
503
504 } else {
505 ctx->recv.payload_pos += avail;
506 }
507
508 if (avail < need) {
509 continue;
510 }
511 }
512
513 /* Phase 39: a full PDU unit (handshake / header / payload) just arrived,
514 * so the current read obligation is satisfied — disarm the deadline before
515 * any dispatch, which may hand the connection off to AIO/SENDING/UPSTREAM/
516 * PROXY/WAITING_*. Idempotent: a no-op on the healthy pipelined path where
517 * the timer was never armed. This is the single point that guarantees the
518 * read timer is never live across a sub-system handoff (the UAF rule). */
519 brix_disarm_read_deadline(c, ctx);
520
521 if (ctx->state == XRD_ST_HANDSHAKE) {
522 rc = brix_process_handshake(ctx, c);
523 if (rc != NGX_OK) {
524 break;
525 }
526 ctx->state = XRD_ST_REQ_HEADER;
527 ctx->recv.hdr_pos = 0;
528
529 } else if (ctx->state == XRD_ST_REQ_HEADER) {
530 ClientRequestHdr *hdr = (ClientRequestHdr *) ctx->recv.hdr_buf;
531 uint32_t max_pl;
532
533 /*
534 * Decode the fixed 24-byte ClientRequestHdr now sitting in hdr_buf
535 * (wire layout, see XProtocol.hh: streamid[2] @0, requestid @2,
536 * body[16] @4, dlen @20). streamid is an opaque client tag echoed
537 * verbatim in the reply — copied byte-for-byte, never byte-swapped.
538 * requestid and dlen are big-endian on the wire, so ntohs/ntohl them
539 * to host order. cur_body is the raw 16-byte opcode argument block
540 * (its meaning is opcode-specific, e.g. fhandle+offset for kXR_read)
541 * and is handed to dispatch untouched for the handler to interpret.
542 */
543 ctx->recv.cur_streamid[0] = hdr->streamid[0];
544 ctx->recv.cur_streamid[1] = hdr->streamid[1];
545 ctx->recv.cur_reqid = ntohs(hdr->requestid);
546 ngx_memcpy(ctx->recv.cur_body, hdr->body, 16);
547 ctx->recv.cur_dlen = (uint32_t) ntohl(hdr->dlen);
548 ctx->recv.cur_body_extra = 0;
549 ctx->recv.cur_body_extended = 0;
550 BRIX_SRV_METRIC_INC(ctx, request_frames_total);
551 BRIX_SRV_METRIC_ADD(ctx, request_payload_bytes_total,
552 ctx->recv.cur_dlen);
553
554 ngx_log_debug(NGX_LOG_DEBUG_STREAM, c->log, 0,
555 "brix: req sid=[%02xd%02xd] reqid=%04xd dlen=%uz"
556 " avail=%d ready=%d",
557 (int) ctx->recv.cur_streamid[0],
558 (int) ctx->recv.cur_streamid[1],
559 (int) ctx->recv.cur_reqid, (size_t) ctx->recv.cur_dlen,
560 c->read->available, (int) c->read->ready);
561
562 /* dlen is untrusted client input — reject before any allocation. */
563 max_pl = brix_max_payload_for_request(ctx->recv.cur_reqid);
564 if (ctx->recv.cur_dlen > max_pl) {
565 ngx_log_error(NGX_LOG_WARN, c->log, 0,
566 "brix: payload too large (%uz), closing",
567 (size_t) ctx->recv.cur_dlen);
568 BRIX_SRV_METRIC_INC(ctx, oversized_payloads_total);
569 break;
570 }
571
572 if (ctx->recv.cur_dlen > 0) {
573 if (brix_ensure_payload_buffer(ctx, c, ctx->recv.cur_dlen)
574 != NGX_OK)
575 {
576 break;
577 }
578 ctx->recv.payload_pos = 0;
579 ctx->state = XRD_ST_REQ_PAYLOAD;
580 ctx->recv.hdr_pos = 0;
581 continue;
582 }
583
584 ctx->recv.payload = NULL;
585
586 /*
587 * Phase 29 drain barrier (extended for write pipelining): kXR_read
588 * and kXR_write both pipeline. Any other opcode arriving while reads
589 * or writes are still in flight must run with the connection quiescent
590 * (e.g. a kXR_close could free a handle an in-flight sendfile chain or
591 * pwrite still references). Defer it until both out_count and
592 * wr_inflight drain; the recv loop re-dispatches it then.
593 */
594 if ((ctx->out.count > 0 || ctx->out.wr_inflight > 0)
595 && ctx->recv.cur_reqid != kXR_read
596 && ctx->recv.cur_reqid != kXR_write)
597 {
598 ctx->out.recv_deferred = 1;
599 ctx->state = XRD_ST_SENDING;
600 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, rx_pending);
601 return;
602 }
603
604 /*
605 * Reset the pipelinable marker before dispatch; only the single-chunk
606 * sendfile read builder sets it back to 1. A read served from the
607 * memory/window path (TLS, non-regular) thus stays non-pipelinable
608 * (its header/data live in the shared scratch buffers).
609 */
610 ctx->out.resp_pipelinable = 0;
611
612 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, rx_pending);
613 rx_pending = 0;
614 rc = brix_dispatch(ctx, c, conf);
615 if (rc == NGX_ERROR) {
616 break;
617 }
618
619 if (ctx->state == XRD_ST_AIO) {
620 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
621 break;
622 }
623 return;
624 }
625
626 /*
627 * Phase 29 pipelining: a cleartext sendfile kXR_read parked its
628 * response (state SENDING, not a windowed read) and we are below the
629 * in-flight cap. Keep reading so the next read's sendfile span
630 * queues behind this one and the link stays full instead of idling
631 * while the prior response drains.
632 *
633 * All four conjuncts must hold. resp_pipelinable and !rd_win_active
634 * are NOT redundant: resp_pipelinable says the builder emitted a
635 * single self-contained sendfile span (header in its own slot, safe
636 * to queue another behind), while rd_win_active flags a *multi*-window
637 * read still streaming chunks out of the shared read_scratch — that
638 * one must stay serial because the next window would clobber the
639 * buffer mid-send. out_count < cap is the in-flight backpressure
640 * bound.
641 */
642 if (ctx->state == XRD_ST_SENDING
643 && ctx->recv.cur_reqid == kXR_read
644 && ctx->out.resp_pipelinable
645 && !ctx->rd.win_active
646 && ctx->out.count < ctx->out.pipeline_depth)
647 {
648 ctx->state = XRD_ST_REQ_HEADER;
649 ctx->recv.hdr_pos = 0;
650 }
651
652 if (ctx->state != XRD_ST_SENDING) {
653 if (ctx->tls_pending) {
654 brix_start_tls(ctx, c, conf);
655 return;
656 }
657
658 ctx->state = XRD_ST_REQ_HEADER;
659 ctx->recv.hdr_pos = 0;
660 }
661
662 } else {
663 /*
664 * kXR_writev stock wire framing: the header dlen covers ONLY the
665 * 16-byte write_list descriptor block; sum(wlen) bytes of segment
666 * data stream after the frame (XrdXrootdProtocol::do_WriteV +
667 * do_WriteVec — the reference client sends dlen = 16*N). The
668 * descriptor block just completed, so validate it and extend the
669 * read obligation by the trailing data length; descriptors + data
670 * then land contiguously in payload_buf before dispatch. A
671 * malformed block is deliberately NOT rejected here: dispatch must
672 * still run the login/auth/write gates first, and the handler then
673 * emits the stock-parity error and drops the link (once the
674 * descriptor framing is in doubt no resync is possible anyway).
675 * brix_writev_body_extra caps descriptors + data at
676 * BRIX_MAX_WRITE_PAYLOAD, so the grow below is bounded.
677 */
678 if (ctx->recv.cur_reqid == kXR_writev && !ctx->recv.cur_body_extended) {
679 uint32_t extra;
680
681 ctx->recv.cur_body_extended = 2;
682
683 if (brix_writev_body_extra(ctx->recv.payload, ctx->recv.cur_dlen,
684 &extra) == NGX_OK
685 && extra > 0)
686 {
687 if (brix_grow_payload_buffer(ctx, c,
688 ctx->recv.cur_dlen + extra) != NGX_OK)
689 {
690 break;
691 }
692 ctx->recv.cur_body_extra = extra;
693 continue; /* keep receiving the streamed segment data */
694 }
695 }
696
697 /*
698 * kXR_chkpoint/kXR_ckpXeq stock wire framing: the header dlen
699 * covers ONLY the embedded 24-byte sub-request header (stock
700 * do_ChkPntXeq rejects any other dlen); the sub-request body
701 * streams after the frame, exactly like the raw data of a plain
702 * write (XrdCl::MessageUtils marks ckpXeq write/pgwrite/writev
703 * as raw-body requests). The extension runs in up to two
704 * stages — embedded header -> sub_dlen bytes, then for an
705 * embedded kXR_writev the descriptor block -> sum(wlen) data —
706 * so cur_body_extended counts stages (2 = done). As with plain
707 * writev, a contract violation is NOT rejected here: dispatch
708 * must run the auth gates first, and the handler then emits the
709 * stock-parity error and drops the link (the un-extended
710 * trailing bytes make resync impossible). Both stages are
711 * bounded (BRIX_MAX_WRITE_PAYLOAD), so the grow is safe.
712 */
713 if (ctx->recv.cur_reqid == kXR_chkpoint
714 && ctx->recv.cur_body[15] == kXR_ckpXeq
715 && ctx->recv.cur_dlen == XRD_REQUEST_HDR_LEN
716 && ctx->recv.cur_body_extended < 2)
717 {
718 uint32_t extra = 0;
719 unsigned final = 1;
720
721 (void) brix_ckpxeq_body_extra(ctx->recv.payload,
722 ctx->recv.cur_dlen + ctx->recv.cur_body_extra,
723 &extra, &final);
724 ctx->recv.cur_body_extended = final ? 2 : 1;
725
726 if (extra > 0) {
727 if (brix_grow_payload_buffer(ctx, c,
728 ctx->recv.cur_dlen + ctx->recv.cur_body_extra + extra)
729 != NGX_OK)
730 {
731 break;
732 }
733 ctx->recv.cur_body_extra += extra;
734 continue; /* keep receiving the streamed sub-body */
735 }
736 }
737
738 /*
739 * Phase 29 drain barrier (extended for write pipelining): kXR_read
740 * and kXR_write both pipeline, so they are NOT deferred. Every other
741 * opcode (close, sync, pgwrite, …) must run with the connection
742 * quiescent, so defer it until BOTH the response/ack queue (out_count)
743 * and the in-flight pwrites (wr_inflight) have drained — e.g. a
744 * kXR_close must not retire a handle a pwrite is still writing.
745 */
746 if ((ctx->out.count > 0 || ctx->out.wr_inflight > 0)
747 && ctx->recv.cur_reqid != kXR_read
748 && ctx->recv.cur_reqid != kXR_write)
749 {
750 ctx->out.recv_deferred = 1;
751 ctx->state = XRD_ST_SENDING;
752 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, rx_pending);
753 return;
754 }
755
756 ctx->state = XRD_ST_REQ_HEADER;
757 ctx->recv.hdr_pos = 0;
758
759 /* Only the single-chunk sendfile read builder re-sets this. */
760 ctx->out.resp_pipelinable = 0;
761
762 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, rx_pending);
763 rx_pending = 0;
764 rc = brix_dispatch(ctx, c, conf);
765 if (rc == NGX_ERROR) {
766 break;
767 }
768
769 if (ctx->state == XRD_ST_AIO) {
770 /*
771 * Write pipelining: a plain kXR_write just posted its pwrite to
772 * the thread pool (wr_inflight bumped in brix_handle_write).
773 * Instead of suspending until it completes — which would serialize
774 * the next chunk's network receive behind this chunk's disk write
775 * — keep receiving. The backpressure boundary (out_count +
776 * wr_inflight >= pipeline_depth) caps the depth, and the
777 * completion callback queues the ack asynchronously and nudges the
778 * read side. Every other AIO op (read, pgwrite, …) still suspends.
779 */
780 if (ctx->recv.cur_reqid == kXR_write && ctx->out.wr_inflight > 0) {
781 ctx->state = XRD_ST_REQ_HEADER;
782 ctx->recv.hdr_pos = 0;
783 continue;
784 }
785 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
786 break;
787 }
788 return;
789 }
790
791 /*
792 * Phase 29 pipelining (payload-bearing kXR_read with a read-ahead
793 * list): identical to the dlen==0 read path — keep reading so the
794 * next read's sendfile span queues behind this one.
795 */
796 if (ctx->state == XRD_ST_SENDING
797 && ctx->recv.cur_reqid == kXR_read
798 && ctx->out.resp_pipelinable
799 && !ctx->rd.win_active
800 && ctx->out.count < ctx->out.pipeline_depth)
801 {
802 ctx->state = XRD_ST_REQ_HEADER;
803 ctx->recv.hdr_pos = 0;
804 continue;
805 }
806
807 if (ctx->state == XRD_ST_SENDING) {
808 return;
809 }
810 }
811 }
812
813 if (brix_defer_teardown_if_writing(ctx, c,
814 NGX_STREAM_INTERNAL_SERVER_ERROR)) {
815 return;
816 }
817 brix_on_disconnect(ctx, c);
818 brix_close_all_files(ctx);
819 ngx_stream_finalize_session(s, NGX_STREAM_INTERNAL_SERVER_ERROR);
820}
