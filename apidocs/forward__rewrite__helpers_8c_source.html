4/*
5 * WHAT: Path prefix rewriting and file handle translation helpers for the transparent XRootD proxy.
6 * WHY: The proxy translates paths between client-facing namespace and upstream filesystem namespace (e.g., /data → /mnt/storage).
7 * kXR_prepare payloads contain newline-separated path lists that need per-line rewriting. File handles are local to the client
8 * session but must be translated to upstream handle IDs before forwarding. INVARIANT: all rewrote buffers allocated via ngx_alloc;
9 * original request freed on success when reallocation occurred (caller responsibility tracked via total_out).
10 * HOW: proxy_rewrite_path uses three strategies — no-match pass-through, same-length in-place rewrite, longer-path reallocation with dlen update.
11 * proxy_rewrite_prepare_payload does two-pass approach (calculate length + check needs_rewrite first, then copy/rewrite). proxy_translate_fh
12 * looks up local handle in fh_map and replaces byte at offset; returns -1 for invalid/unmapped handles.
13 */

17/* proxy_rewrite_path — apply the strip/add prefix transform to a request payload
18 * carrying one filesystem path (transparent proxy: client /data → upstream
19 * /mnt/storage); same-length rewrites are in-place, longer results reallocate and
20 * update the dlen header. */
21u_char *
22proxy_rewrite_path(ngx_connection_t *c,
23 ngx_stream_brix_srv_conf_t *conf,
24 u_char *req, size_t total,
25 size_t path_off, size_t path_len,
26 size_t *total_out)
27{
28 const u_char *strip = conf->proxy.path_strip.data;
29 size_t slen = conf->proxy.path_strip.len;
30 const u_char *add = conf->proxy.path_add.data;
31 size_t alen = conf->proxy.path_add.len;
32 size_t new_path_len, new_total;
33 u_char *new_req;
34
35 if (slen == 0 || path_len < slen
36 || ngx_strncmp(req + path_off, strip, slen) != 0)
37 {
38 *total_out = total;
39 return req; /* no match — pass through unchanged */
40 }
41
42 new_path_len = alen + (path_len - slen);
43 new_total = total - path_len + new_path_len;
44
45 if (new_path_len == path_len) {
46 /* Same length: rewrite in-place */
47 ngx_memcpy(req + path_off, add, alen);
48 *total_out = total;
49 return req;
50 }
51
52 new_req = ngx_alloc(new_total, c->log);
53 if (new_req == NULL) {
54 *total_out = total;
55 return req; /* OOM: pass original unchanged */
56 }
57
58 /* Copy header portion before path, then new path, then anything after */
59 ngx_memcpy(new_req, req, path_off);
60 ngx_memcpy(new_req + path_off, add, alen);
61 ngx_memcpy(new_req + path_off + alen,
62 req + path_off + slen,
63 total - path_off - slen);
64
65 /* Update the 4-byte dlen field in the request header (bytes 20-23) */
66 {
67 uint32_t new_dlen = htonl((uint32_t) (new_total - XRD_REQUEST_HDR_LEN));
68 ngx_memcpy(new_req + 20, &new_dlen, 4);
69 }
70
71 *total_out = new_total;
72 return new_req;
73}
