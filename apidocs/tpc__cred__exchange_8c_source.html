35/**
36 * Perform an RFC 8693 token-exchange request.
37 *
38 * Sends a POST to the configured token endpoint with:
39 * grant_type=urn:ietf:params:oauth:grant-type:token-exchange
40 * subject_token=<JWT>
41 * resource=<source_url>
42 * audience=<source_url>
43 * scope=<scope>
44 *
45 * Uses a fork/exec'd curl subprocess (same pattern as tpc_curl.c).
46 */
47ngx_int_t
48tpc_cred_rfc8693_exchange(ngx_http_request_t *r,
49 const char *subject_token,
50 const char *source_url,
51 const char *scope,
52 const char *token_endpoint,
53 const char *client_id,
54 const char *client_secret,
55 ngx_str_t *token_out)
56{
57 char buf[TPC_CRED_MAX_TOKEN_LEN + 256];
58 char *curl_argv[16];
59 int argc = 0;
60 char body_buf[2048];
61 char body_file[NGX_MAX_PATH];
62 int body_fd;
63
64 /* Build the POST body. */
65 ngx_snprintf((u_char *) body_buf, sizeof(body_buf),
66 "grant_type=urn:ietf:params:oauth:grant-type:"
67 "token-exchange"
68 "&subject_token=%s"
69 "&resource=%s"
70 "&audience=%s"
71 "&scope=%s",
72 subject_token, source_url, source_url, scope);
73
74 /* Write body to a temp file (curl --data @file). */
75 ngx_snprintf((u_char *) body_file, sizeof(body_file),
76 "/tmp/tpc_cred_body_XXXXXX");
77 body_fd = mkstemp(body_file);
78 if (body_fd == -1) {
79 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
80 "tpc_cred(rfc8693): temp file creation failed");
81 return NGX_ERROR;
82 }
83
84 {
85 ssize_t nw = write(body_fd, body_buf, strlen(body_buf));
86 (void) nw;
87 }
88 close(body_fd);
89
90 /* Build curl argv. */
91 curl_argv[argc++] = (char *) "curl";
92 curl_argv[argc++] = (char *) "-s"; /* silent */
93 curl_argv[argc++] = (char *) "-S"; /* show errors */
94 curl_argv[argc++] = (char *) "-f"; /* fail on HTTP error */
95 curl_argv[argc++] = (char *) "-X";
96 curl_argv[argc++] = (char *) "POST";
97 curl_argv[argc++] = (char *) "-H";
98 curl_argv[argc++] = (char *) "Content-Type: application/x-www-form-urlencoded";
99 if (client_id && *client_id) {
100 /* Basic auth with client credentials. */
101 curl_argv[argc++] = (char *) "-u";
102 /* We'd need to base64 encode client_id:client_secret here.
103 * For simplicity, pass them separately to curl. */
104 curl_argv[argc++] = (char *) client_id;
105 curl_argv[argc++] = (char *) client_secret;
106 }
107 curl_argv[argc++] = (char *) "-d";
108 curl_argv[argc++] = (char *) body_file;
109 /* W3 — end-of-options terminator so the endpoint URL can never be parsed
110 * as a curl option, even if a misconfigured token_endpoint begins with '-'. */
111 curl_argv[argc++] = (char *) "--";
112 curl_argv[argc++] = (char *) token_endpoint;
113 curl_argv[argc++] = NULL;
114
115 /*
116 * Run curl synchronously and capture its stdout via the shared SIGCHLD-safe
117 * fork/exec helper (src/compat/subprocess.c) — it blocks SIGCHLD across the
118 * fork/waitpid internally so nginx's handler can't reap the child first. A
119 * non-zero rc = pipe/fork failure or signal-kill; a non-zero child exit
120 * (incl. curl -f on HTTP >= 400) is a credential-fetch failure.
121 */
122 {
123 int ec = -1;
124
125 if (brix_subprocess_capture(curl_argv, buf, sizeof(buf), NULL, &ec)
126 != 0)
127 {
128 ngx_log_error(NGX_LOG_ERR, r->connection->log, errno,
129 "tpc_cred(rfc8693): curl subprocess failed "
130 "(pipe/fork or signal)");
131 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
132 return NGX_ERROR;
133 }
134 if (ec != 0) {
135 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
136 "tpc_cred(rfc8693): curl exited %d: %s", ec, buf);
137 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
138 return NGX_ERROR;
139 }
140 }
141
142 /* Clean up temp file. */
143 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
144
145 /* Parse JSON response for access_token. */
146 return tpc_cred_parse_token_response(r, buf, token_out);
147}
