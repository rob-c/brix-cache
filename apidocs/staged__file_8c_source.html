20/*
21 * Confinement (Phase 8): the temp file and its final destination always live
22 * under root_canon (brix_make_tmp_path derives the temp name next to
23 * final_path, which the caller already confined to the export root). We open a
24 * kernel-confinement rootfd on root_canon and route the temp create / rename /
25 * unlink through the beneath API so the operation is bounded by the kernel
26 * regardless of how the caller derived the path. A path that does not strip
27 * cleanly under root_canon is refused (EXDEV) rather than touched raw.
28 */
29
30/*
31 * WHAT: Open a unique temporary file inside the confined root_canon for atomic write.
32 *
33 * WHY: S3 PUT, WebDAV PUT, and other operations need to write data safely without risking
34 * corruption of the final path if the process crashes mid-write. A temp file with O_EXCL
35 * guarantees atomicity — either the rename succeeds (final path appears) or it doesn't
36 * (temp file remains for cleanup).
37 *
38 * HOW: Generate a unique tmp_path via brix_make_tmp_path(). Open with
39 * open_flags | O_CREAT | O_EXCL inside root_canon via brix_open_confined_canon().
40 * Loop up to 'attempts' times (default 16) on EEXIST. On success: set staged->active=1,
41 * store fd and tmp_path, return NGX_OK. On non-EEXIST failure or exhaustion: errno set,
42 * return NGX_ERROR.
43 *
44 * Parameters:
45 * log — nginx log for error reporting
46 * req — request description: root_canon / final_path / open_flags / mode /
47 * attempts (see brix_staged_open_req_t)
48 * staged — output struct: fd, tmp_path, active flag populated on success
49 */
50ngx_int_t
51brix_staged_open(ngx_log_t *log, const brix_staged_open_req_t *req,
52 brix_staged_file_t *staged)
53{
54 ngx_uint_t attempts;
55 ngx_uint_t i;
56 int rootfd;
57
58 (void) log;
59
60 if (staged == NULL || req == NULL || req->final_path == NULL) {
61 errno = EINVAL;
62 return NGX_ERROR;
63 }
64
65 ngx_memzero(staged, sizeof(*staged));
66 staged->fd = NGX_INVALID_FILE;
67 /* SECURITY: the temp is created PRIVATE (0600) so another mapped uid on a
68 * shared filesystem cannot read an in-progress upload; the caller's intended
69 * final mode is restored at commit (staged_commit_internal). */
70 staged->final_mode = req->mode;
71
72 attempts = req->attempts;
73 if (attempts == 0) {
74 attempts = 16;
75 }
76
77 rootfd = brix_beneath_open_root(req->root_canon);
78 if (rootfd < 0) {
79 return NGX_ERROR;
80 }
81
82 for (i = 0; i < attempts; i++) {
83 const char *rel;
84
85 if (brix_make_tmp_path(req->final_path, staged->tmp_path,
86 sizeof(staged->tmp_path)) != NGX_OK)
87 {
88 errno = ENAMETOOLONG;
89 close(rootfd);
90 return NGX_ERROR;
91 }
92
93 rel = brix_beneath_strip_root(req->root_canon, staged->tmp_path);
94 if (rel == NULL) {
95 errno = EXDEV;
96 close(rootfd);
97 return NGX_ERROR;
98 }
99
100 staged->fd = brix_open_beneath(rootfd, rel,
101 req->open_flags | O_CREAT | O_EXCL,
102 0600);
103 if (staged->fd != NGX_INVALID_FILE) {
104 staged->active = 1;
105 close(rootfd);
106 return NGX_OK;
107 }
108
109 if (errno != EEXIST) {
110 close(rootfd);
111 return NGX_ERROR;
112 }
113 }
114
115 close(rootfd);
116 errno = EEXIST;
117 return NGX_ERROR;
118}

120/*
121 * WHAT: Open the DETERMINISTIC, identity-keyed upload-resume partial for a final
122 * path (confined inside root_canon), creating it if absent and PRESERVING
123 * any existing bytes (no O_TRUNC, no O_EXCL). Reports the current partial
124 * size in *cur_size so the caller / client can resume at that offset.
125 *
126 * WHY: WebDAV resumable PUT (Content-Range) needs a chunk to land at an absolute
127 * offset on a partial that survives across separate PUT requests and a
128 * server restart, then commit (rename) only when complete — the HTTP
129 * analogue of the root:// resume staging. Reuses the same name scheme
130 * (brix_make_resume_path) and confinement (brix_open_beneath) as the
131 * random staged_open so security and glob-clean are identical.
132 *
133 * Returns NGX_OK with staged->active=1, or NGX_ERROR (errno set).
134 *
135 * Parameters:
136 * log — nginx log for error reporting
137 * req — request description: root_canon / final_path / principal / stage_dir /
138 * mode (see brix_staged_open_req_t)
139 * staged — output struct populated on success
140 * cur_size — output: current partial size (resume offset), 0 if fresh
141 */
142ngx_int_t
143brix_staged_open_resume(ngx_log_t *log, const brix_staged_open_req_t *req,
144 brix_staged_file_t *staged, off_t *cur_size)
145{
146 const char *stage_dir;
147 int rootfd, fd;
148 const char *rel;
149 struct stat sb;
150
151 (void) log;
152
153 if (staged == NULL || req == NULL || req->final_path == NULL) {
154 errno = EINVAL;
155 return NGX_ERROR;
156 }
157 ngx_memzero(staged, sizeof(*staged));
158 staged->fd = NGX_INVALID_FILE;
159 /* SECURITY: the resume partial is created PRIVATE (0600) and stays private
160 * across requests/restarts (it persists between range chunks); the intended
161 * final mode is restored at commit. */
162 staged->final_mode = req->mode;
163 if (cur_size != NULL) {
164 *cur_size = 0;
165 }
166
167 if (brix_make_resume_path(req->final_path, req->principal, req->stage_dir,
168 staged->tmp_path, sizeof(staged->tmp_path))
169 != NGX_OK)
170 {
171 errno = ENAMETOOLONG;
172 return NGX_ERROR;
173 }
174
175 stage_dir = req->stage_dir;
176 if (stage_dir != NULL && stage_dir[0] != '\0') {
177 /* Partial lives on the configured fast device (outside root_canon). The
178 * basename is a server-generated hash inside the operator-trusted stage
179 * dir, so a direct O_NOFOLLOW open is safe; commit moves it to storage. */
180 fd = open(staged->tmp_path, O_RDWR | O_CREAT | O_NOFOLLOW | O_CLOEXEC,
181 0600);
182 if (fd == NGX_INVALID_FILE) {
183 return NGX_ERROR;
184 }
185 } else {
186 rootfd = brix_beneath_open_root(req->root_canon);
187 if (rootfd < 0) {
188 return NGX_ERROR;
189 }
190 rel = brix_beneath_strip_root(req->root_canon, staged->tmp_path);
191 if (rel == NULL) {
192 close(rootfd);
193 errno = EXDEV;
194 return NGX_ERROR;
195 }
196 /* O_CREAT but NOT O_EXCL / O_TRUNC: create-or-resume, preserving bytes. */
197 fd = brix_open_beneath(rootfd, rel, O_RDWR | O_CREAT, 0600);
198 close(rootfd);
199 if (fd == NGX_INVALID_FILE) {
200 return NGX_ERROR;
201 }
202 }
203
204 if (cur_size != NULL && fstat(fd, &sb) == 0) {
205 *cur_size = sb.st_size;
206 }
207 staged->fd = fd;
208 staged->active = 1;
209 return NGX_OK;
210}

212/*
213 * WHAT: Atomically rename the temp file to its final path and clean up.
214 *
215 * WHY: After all data has been written to the staged temp file, commit makes it visible at
216 * the target location. The rename is atomic on POSIX filesystems — readers see either
217 * the old file or the new one, never a partial write.
218 *
219 * HOW: Close the fd if still open (data should be flushed). Rename tmp_path → final_path via
220 * brix_rename_confined_canon(). On rename failure: unlink the temp file as cleanup. Set
221 * staged->active=0 and clear tmp_path buffer. Return NGX_OK on success, NGX_ERROR on fail.
222 *
223 * Parameters:
224 * log — nginx log for error/cleanup reporting
225 * root_canon — canonical root for confined rename
226 * staged — the staged file struct (must be active)
227 * final_path — destination path to rename into
228 */
229static ngx_int_t
230staged_commit_internal(ngx_log_t *log, const char *root_canon,
231 brix_staged_file_t *staged, const char *final_path, int exclusive)
232{
233 if (staged == NULL || !staged->active) {
234 errno = EINVAL;
235 return NGX_ERROR;
236 }
237
238 int rootfd;
239 int rc;
240 const char *tmp_rel, *final_rel;
241
242 /* Open the confinement root first so the fsync-failure cleanup path (C1) can
243 * unlink the temp without re-opening it. */
244 rootfd = brix_beneath_open_root(root_canon);
245 if (rootfd < 0) {
246 if (staged->fd != NGX_INVALID_FILE) {
247 ngx_close_file(staged->fd);
248 staged->fd = NGX_INVALID_FILE;
249 }
250 staged->active = 0;
251 return NGX_ERROR;
252 }
253 tmp_rel = brix_beneath_strip_root(root_canon, staged->tmp_path);
254 final_rel = brix_beneath_strip_root(root_canon, final_path);
255 if (tmp_rel == NULL || final_rel == NULL) {
256 if (staged->fd != NGX_INVALID_FILE) {
257 ngx_close_file(staged->fd);
258 staged->fd = NGX_INVALID_FILE;
259 }
260 close(rootfd);
261 staged->active = 0;
262 errno = EXDEV;
263 return NGX_ERROR;
264 }
265
266 /*
267 * Phase 51 (C1): flush the staged data to stable storage BEFORE the rename
268 * publishes it, so a crash / power loss / ENOSPC mid-write cannot expose a
269 * torn object. A failed fsync means the data is NOT durable — fail the
270 * commit (unlink the temp, leave the final path untouched) rather than
271 * publish possibly-incomplete data. (close() alone does not flush.)
272 */
273 if (staged->fd != NGX_INVALID_FILE) {
274 if (fsync(staged->fd) != 0) {
275 int e = errno;
276 ngx_log_error(NGX_LOG_ERR, log, e,
277 "brix: staged commit fsync failed — not publishing "
278 "\"%s\"", final_path);
279 ngx_close_file(staged->fd);
280 staged->fd = NGX_INVALID_FILE;
281 (void) brix_unlink_beneath(rootfd, tmp_rel, 0);
282 close(rootfd);
283 staged->active = 0;
284 errno = e;
285 return NGX_ERROR;
286 }
287 /* SECURITY: restore the caller's intended mode on the OPEN fd just before
288 * the rename publishes it. The temp was written 0600 (private); rename
289 * preserves the fd's mode, so the committed namespace object carries its
290 * client-intended bits (e.g. 0644) with no world-readable in-flight
291 * window. */
292 (void) fchmod(staged->fd, staged->final_mode);
293 ngx_close_file(staged->fd);
294 staged->fd = NGX_INVALID_FILE;
295 }
296
297 rc = exclusive ? brix_rename_beneath_excl(rootfd, tmp_rel, final_rel)
298 : brix_rename_beneath(rootfd, tmp_rel, final_rel);
299 if (rc != 0) {
300 int e = errno; /* preserve EEXIST for the caller */
301 (void) brix_unlink_beneath(rootfd, tmp_rel, 0);
302 close(rootfd);
303 staged->active = 0;
304 errno = e;
305 return NGX_ERROR;
306 }
307
308 /* C1: persist the directory entry so the rename itself survives a crash
309 * (best-effort — the data is already durable above). */
310 (void) fsync(rootfd);
311
312 close(rootfd);
313 staged->active = 0;
314 staged->tmp_path[0] = '\0';
315 return NGX_OK;
316}

332/*
333 * WHAT: Close the temp file fd and optionally unlink it from disk.
334 *
335 * WHY: On write failure or client disconnect, abort cleans up the staged temp file so it
336 * doesn't leak on disk. The caller decides whether to remove the temp file (remove_tmp=1)
337 * or leave it for later inspection (remove_tmp=0).
338 *
339 * HOW: Close fd if open and valid. If remove_tmp is set AND staged is active AND tmp_path is
340 * non-empty: unlink via brix_unlink_confined_canon(). Always set active=0 and clear
341 * tmp_path buffer.
342 *
343 * Parameters:
344 * log — nginx log for cleanup error reporting
345 * root_canon — canonical root for confined unlink
346 * staged — the staged file struct (NULL-safe)
347 * remove_tmp — 1 to delete temp file, 0 to leave it on disk
348 */
349void
350brix_staged_abort(ngx_log_t *log, const char *root_canon,
351 brix_staged_file_t *staged, ngx_flag_t remove_tmp)
352{
353 if (staged == NULL) {
354 return;
355 }
356
357 (void) log;
358
359 if (staged->fd != NGX_INVALID_FILE) {
360 ngx_close_file(staged->fd);
361 staged->fd = NGX_INVALID_FILE;
362 }
363
364 if (remove_tmp && staged->active && staged->tmp_path[0] != '\0') {
365 int rootfd = brix_beneath_open_root(root_canon);
366 const char *tmp_rel;
367
368 if (rootfd >= 0) {
369 tmp_rel = brix_beneath_strip_root(root_canon, staged->tmp_path);
370 if (tmp_rel != NULL) {
371 (void) brix_unlink_beneath(rootfd, tmp_rel, 0);
372 }
373 close(rootfd);
374 }
375 }
376
377 staged->active = 0;
378 staged->tmp_path[0] = '\0';
379}
