25/*
26 * Confinement (Phase 8): the temp file and its final destination always live
27 * under root_canon (brix_make_tmp_path derives the temp name next to
28 * final_path, which the caller already confined to the export root). We open a
29 * kernel-confinement rootfd on root_canon and route the temp create / rename /
30 * unlink through the beneath API so the operation is bounded by the kernel
31 * regardless of how the caller derived the path. A path that does not strip
32 * cleanly under root_canon is refused (EXDEV) rather than touched raw.
33 */
34
35/*
36 * WHAT: Open a unique temporary file inside the confined root_canon for atomic write.
37 *
38 * WHY: S3 PUT, WebDAV PUT, and other operations need to write data safely without risking
39 * corruption of the final path if the process crashes mid-write. A temp file with O_EXCL
40 * guarantees atomicity — either the rename succeeds (final path appears) or it doesn't
41 * (temp file remains for cleanup).
42 *
43 * HOW: Generate a unique tmp_path via brix_make_tmp_path(). Open with
44 * open_flags | O_CREAT | O_EXCL inside root_canon via brix_open_confined_canon().
45 * Loop up to 'attempts' times (default 16) on EEXIST. On success: set staged->active=1,
46 * store fd and tmp_path, return NGX_OK. On non-EEXIST failure or exhaustion: errno set,
47 * return NGX_ERROR.
48 *
49 * Parameters:
50 * log — nginx log for error reporting
51 * root_canon — canonical root directory the temp file must reside within
52 * final_path — destination path (used to derive tmp_path name prefix)
53 * open_flags — base open flags (O_WRONLY typically; O_EXCL is added internally)
54 * mode — file creation mode (e.g. 0644)
55 * attempts — max retry count on EEXIST (0 → defaults to 16)
56 * staged — output struct: fd, tmp_path, active flag populated on success
57 */
58ngx_int_t
59brix_staged_open(ngx_log_t *log, const char *root_canon,
60 const char *final_path, int open_flags, mode_t mode, ngx_uint_t attempts,
61 brix_staged_file_t *staged)
62{
63 ngx_uint_t i;
64 int rootfd;
65
66 (void) log;
67
68 if (staged == NULL || final_path == NULL) {
69 errno = EINVAL;
70 return NGX_ERROR;
71 }
72
73 ngx_memzero(staged, sizeof(*staged));
74 staged->fd = NGX_INVALID_FILE;
75 /* SECURITY: the temp is created PRIVATE (0600) so another mapped uid on a
76 * shared filesystem cannot read an in-progress upload; the caller's intended
77 * final mode is restored at commit (staged_commit_internal). */
78 staged->final_mode = mode;
79
80 if (attempts == 0) {
81 attempts = 16;
82 }
83
84 rootfd = brix_beneath_open_root(root_canon);
85 if (rootfd < 0) {
86 return NGX_ERROR;
87 }
88
89 for (i = 0; i < attempts; i++) {
90 const char *rel;
91
92 if (brix_make_tmp_path(final_path, staged->tmp_path,
93 sizeof(staged->tmp_path)) != NGX_OK)
94 {
95 errno = ENAMETOOLONG;
96 close(rootfd);
97 return NGX_ERROR;
98 }
99
100 rel = brix_beneath_strip_root(root_canon, staged->tmp_path);
101 if (rel == NULL) {
102 errno = EXDEV;
103 close(rootfd);
104 return NGX_ERROR;
105 }
106
107 staged->fd = brix_open_beneath(rootfd, rel,
108 open_flags | O_CREAT | O_EXCL, 0600);
109 if (staged->fd != NGX_INVALID_FILE) {
110 staged->active = 1;
111 close(rootfd);
112 return NGX_OK;
113 }
114
115 if (errno != EEXIST) {
116 close(rootfd);
117 return NGX_ERROR;
118 }
119 }
120
121 close(rootfd);
122 errno = EEXIST;
123 return NGX_ERROR;
124}

126/*
127 * WHAT: Open the DETERMINISTIC, identity-keyed upload-resume partial for a final
128 * path (confined inside root_canon), creating it if absent and PRESERVING
129 * any existing bytes (no O_TRUNC, no O_EXCL). Reports the current partial
130 * size in *cur_size so the caller / client can resume at that offset.
131 *
132 * WHY: WebDAV resumable PUT (Content-Range) needs a chunk to land at an absolute
133 * offset on a partial that survives across separate PUT requests and a
134 * server restart, then commit (rename) only when complete — the HTTP
135 * analogue of the root:// resume staging. Reuses the same name scheme
136 * (brix_make_resume_path) and confinement (brix_open_beneath) as the
137 * random staged_open so security and glob-clean are identical.
138 *
139 * Returns NGX_OK with staged->active=1, or NGX_ERROR (errno set).
140 */
141ngx_int_t
142brix_staged_open_resume(ngx_log_t *log, const char *root_canon,
143 const char *final_path, const char *principal, const char *stage_dir,
144 mode_t mode, brix_staged_file_t *staged, off_t *cur_size)
145{
146 int rootfd, fd;
147 const char *rel;
148 struct stat sb;
149
150 (void) log;
151
152 if (staged == NULL || final_path == NULL) {
153 errno = EINVAL;
154 return NGX_ERROR;
155 }
156 ngx_memzero(staged, sizeof(*staged));
157 staged->fd = NGX_INVALID_FILE;
158 /* SECURITY: the resume partial is created PRIVATE (0600) and stays private
159 * across requests/restarts (it persists between range chunks); the intended
160 * final mode is restored at commit. */
161 staged->final_mode = mode;
162 if (cur_size != NULL) {
163 *cur_size = 0;
164 }
165
166 if (brix_make_resume_path(final_path, principal, stage_dir,
167 staged->tmp_path, sizeof(staged->tmp_path))
168 != NGX_OK)
169 {
170 errno = ENAMETOOLONG;
171 return NGX_ERROR;
172 }
173
174 if (stage_dir != NULL && stage_dir[0] != '\0') {
175 /* Partial lives on the configured fast device (outside root_canon). The
176 * basename is a server-generated hash inside the operator-trusted stage
177 * dir, so a direct O_NOFOLLOW open is safe; commit moves it to storage. */
178 fd = open(staged->tmp_path, O_RDWR | O_CREAT | O_NOFOLLOW | O_CLOEXEC,
179 0600);
180 if (fd == NGX_INVALID_FILE) {
181 return NGX_ERROR;
182 }
183 } else {
184 rootfd = brix_beneath_open_root(root_canon);
185 if (rootfd < 0) {
186 return NGX_ERROR;
187 }
188 rel = brix_beneath_strip_root(root_canon, staged->tmp_path);
189 if (rel == NULL) {
190 close(rootfd);
191 errno = EXDEV;
192 return NGX_ERROR;
193 }
194 /* O_CREAT but NOT O_EXCL / O_TRUNC: create-or-resume, preserving bytes. */
195 fd = brix_open_beneath(rootfd, rel, O_RDWR | O_CREAT, 0600);
196 close(rootfd);
197 if (fd == NGX_INVALID_FILE) {
198 return NGX_ERROR;
199 }
200 }
201
202 if (cur_size != NULL && fstat(fd, &sb) == 0) {
203 *cur_size = sb.st_size;
204 }
205 staged->fd = fd;
206 staged->active = 1;
207 return NGX_OK;
208}

210/* VFS↔VFS (backend↔backend) byte move: copy the whole source object into the
211 * destination object by reading through the SOURCE object's storage driver and
212 * writing through the DESTINATION object's driver. The two objects may live on
213 * different backends — this is the cross-mount staging commit, so it is exactly
214 * the place where one side could later be a block/object (S3) store while the
215 * other stays POSIX.
216 *
217 * Today both ends are POSIX (brix_sd_posix_wrap over a kernel fd), so the only
218 * raw pread/pwrite happen INSIDE the POSIX backend (src/fs/backend/sd_posix.c) —
219 * never here. When a stage or final mount becomes a non-POSIX backend, only how
220 * the object is obtained changes (that driver's open() instead of a bare-fd
221 * wrap); this positional copy loop is unchanged. 0 / NGX_ERROR (errno set). */
222static ngx_int_t
223stage_move_objects(brix_sd_obj_t *src, brix_sd_obj_t *dst)
224{
225 /* The mover now lives in the transfer engine (src/fs/xfer/xfer_mover_pump.c)
226 * as the canonical BRIX_XFER_MOVE_PUMP strategy; this thin wrapper keeps the
227 * two staged-commit callsites unchanged. */
228 return brix_xfer_pump_objects(src, dst);
229}

231/* commit_be_logical — the export-root-relative ("/sub/file", or "/" for the root)
232 * key form a non-POSIX backend's namespace expects, matching
233 * brix_vfs_export_relative_root used by every other driver-backed op. */
234static const char *
235commit_be_logical(const char *abs, const char *root)
236{
237 size_t rl = (root != NULL) ? strlen(root) : 0;
238
239 if (rl > 0 && strncmp(abs, root, rl) == 0) {
240 if (abs[rl] == '/') { return abs + rl; } /* "/sub/file" */
241 if (abs[rl] == '\0') { return "/"; } /* the export root itself */
242 }
243 return abs;
244}

246/*
247 * commit_staged_to_backend — publish a POSIX-staged partial INTO a non-POSIX
248 * storage backend (e.g. pblock) by streaming it through that driver's staged
249 * write→commit state machine. This is the cross-FILESYSTEM atomic commit that
250 * rename(2) cannot do: the partial lives on an independent POSIX staging mount
251 * (brix_stage_dir) and the final export is a different, driver-owned namespace,
252 * so we read the partial through the (POSIX) source backend and write it through
253 * the destination driver's staged_write, then staged_commit (which publishes
254 * atomically — for pblock, a single catalog row insert pointing at the freshly
255 * written blocks). On success the POSIX partial is unlinked; on failure it is
256 * KEPT (a resume client can retry the close) and the driver's staged blob is
257 * aborted. Returns NGX_OK / NGX_ERROR (errno set).
258 */
259static ngx_int_t
260commit_staged_to_backend(ngx_fd_t fd, const char *stage_path,
261 const char *final_path, brix_sd_instance_t *dst, const char *root_canon,
262 ngx_log_t *log)
263{
264 const char *logical = commit_be_logical(final_path, root_canon);
265 brix_sd_obj_t src_obj;
266 brix_sd_staged_t *st;
267 struct stat sb;
268 mode_t mode;
269 int rfd, owned = 0, serr = 0;
270 off_t off = 0;
271
272 (void) log;
273
274 if (dst->driver->staged_open == NULL || dst->driver->staged_write == NULL
275 || dst->driver->staged_commit == NULL
276 || dst->driver->staged_abort == NULL)
277 {
278 errno = ENOTSUP;
279 return NGX_ERROR;
280 }
281
282 /* Source: the open staged partial if we still hold it, else open it O_RDONLY.
283 * Reads are positional (pread), so the fd's file position is irrelevant. */
284 if (fd != NGX_INVALID_FILE) {
285 rfd = fd;
286 } else {
287 rfd = open(stage_path, O_RDONLY | O_CLOEXEC);
288 if (rfd < 0) {
289 return NGX_ERROR;
290 }
291 owned = 1;
292 }
293 mode = (fstat(rfd, &sb) == 0) ? (sb.st_mode & 07777) : 0644;
294
295 st = dst->driver->staged_open(dst, logical, mode, &serr);
296 if (st == NULL) {
297 if (owned) { (void) close(rfd); }
298 errno = serr ? serr : EIO;
299 return NGX_ERROR;
300 }
301
302 brix_sd_posix_wrap(&src_obj, rfd); /* read the partial via the SD seam */
303 for ( ;; ) {
304 char buf[65536];
305 ssize_t r = src_obj.driver->pread(&src_obj, buf, sizeof(buf), off);
306 ssize_t w = 0;
307
308 if (r < 0) {
309 int e = errno;
310 dst->driver->staged_abort(st);
311 if (owned) { (void) close(rfd); }
312 errno = e;
313 return NGX_ERROR;
314 }
315 if (r == 0) {
316 break; /* EOF: whole partial consumed */
317 }
318 while (w < r) {
319 ssize_t k = dst->driver->staged_write(st, buf + w,
320 (size_t) (r - w), off + w);
321 if (k <= 0) {
322 int e = errno;
323 dst->driver->staged_abort(st);
324 if (owned) { (void) close(rfd); }
325 errno = e ? e : EIO;
326 return NGX_ERROR;
327 }
328 w += k;
329 }
330 off += r;
331 }
332
333 /* Atomic publish. On failure staged_commit leaves the handle valid, so abort
334 * to release it; KEEP the POSIX partial so a resume retry can republish. */
335 if (dst->driver->staged_commit(st, 0 /* replace allowed */) != NGX_OK) {
336 int e = errno;
337 dst->driver->staged_abort(st);
338 if (owned) { (void) close(rfd); }
339 errno = e ? e : EIO;
340 return NGX_ERROR;
341 }
342
343 if (owned) { (void) close(rfd); }
344 (void) unlink(stage_path); /* published → drop the partial */
345 return NGX_OK;
346}

348/*
349 * WHAT: Commit a staged file onto its final path — atomically and across
350 * filesystems. When the final export uses a NON-POSIX storage backend
351 * (e.g. pblock), the partial is uploaded INTO that backend via the driver's
352 * staged write→commit state machine (commit_staged_to_backend). Otherwise
353 * it tries rename(2) first (atomic, same-FS, the fast path); on
354 * EXDEV (the staged file is on a configured fast-cache device different from
355 * the storage, e.g. CEPHFS) it copies the data to a temp ON THE FINAL
356 * filesystem, fsync()s, atomically renames that temp onto the final path,
357 * and unlinks the staged copy. Either way a concurrent reader sees only the
358 * old object or the complete new one — never a partial.
359 *
360 * WHY: Upload staging on a fast device + final storage on a POSIX mount means the
361 * commit can be a cross-device move, which rename(2) cannot do. Centralising
362 * this lets both root:// (close) and WebDAV (PUT) commit identically.
363 *
364 * fd: the open staged-file descriptor (for the durability fsync before commit), or
365 * NGX_INVALID_FILE if already closed. Returns NGX_OK / NGX_ERROR (errno set).
366 */
367ngx_int_t
368brix_commit_staged(ngx_fd_t fd, const char *stage_path, const char *final_path,
369 mode_t final_mode, ngx_log_t *log)
370{
371 char tmp[PATH_MAX];
372 int rfd, dfd, e;
373 struct stat sb;
374
375 if (fd != NGX_INVALID_FILE) {
376 brix_sd_obj_t sobj;
377 brix_sd_posix_wrap(&sobj, fd); /* durability flush via the backend */
378 if (sobj.driver->fsync(&sobj) != NGX_OK) {
379 return NGX_ERROR; /* not durable — do not publish */
380 }
381 /* SECURITY: publish the caller's intended mode on the still-open fd
382 * before any commit path reads it. The temp may have been written 0600
383 * (private); each downstream path — driver upload (fstats stage_path),
384 * same-FS rename (preserves the fd mode), cross-device copy (fstat at
385 * :411) — then derives the final object's mode from this. final_mode==0
386 * means "leave the temp's current mode as-is" (e.g. root:// POSC temps,
387 * which already carry the client's create mode). */
388 if (final_mode != 0) {
389 (void) fchmod(fd, final_mode);
390 }
391 }
392
393 /* Non-POSIX final export (e.g. pblock): the staged POSIX partial lives on an
394 * independent staging mount and the final namespace is driver-owned, so
395 * rename(2) cannot publish it. Upload the partial INTO the backend via its
396 * staged write→commit state machine (a cross-filesystem atomic publish).
397 * POSIX exports (the common case) skip this and take the rename path below. */
398 {
399 const char *be_root = NULL;
400 brix_sd_instance_t *dst =
401 brix_vfs_backend_resolve_for_path(final_path, &be_root, log);
402
403 if (dst != NULL && dst->driver != NULL
404 && dst->driver != brix_sd_default_driver())
405 {
406 return commit_staged_to_backend(fd, stage_path, final_path, dst,
407 be_root, log);
408 }
409 }
410
411 if (rename(stage_path, final_path) == 0) {
412 return NGX_OK; /* same filesystem: atomic, zero-copy */
413 }
414 if (errno != EXDEV) {
415 return NGX_ERROR;
416 }
417
418 /* Cross-device commit: copy the staged data to a temp adjacent to (and on the
419 * same filesystem as) the final path, then atomically rename it into place. */
420 if (brix_make_tmp_path(final_path, tmp, sizeof(tmp)) != NGX_OK) {
421 errno = ENAMETOOLONG;
422 return NGX_ERROR;
423 }
424 rfd = open(stage_path, O_RDONLY | O_CLOEXEC);
425 if (rfd < 0) {
426 return NGX_ERROR;
427 }
428 /* Preserve the staged file's mode on the committed object. */
429 mode_t mode = (fstat(rfd, &sb) == 0) ? (sb.st_mode & 07777) : 0644;
430 dfd = open(tmp, O_WRONLY | O_CREAT | O_EXCL | O_NOFOLLOW | O_CLOEXEC, mode);
431 if (dfd < 0) {
432 e = errno; close(rfd); errno = e; return NGX_ERROR;
433 }
434 {
435 /* VFS↔VFS move: stage object (source backend) → temp object (final
436 * backend). Both POSIX today; the loop is driver-mediated so a non-POSIX
437 * mount on either side needs no change here. */
438 brix_sd_obj_t src_obj, dst_obj;
439 brix_sd_posix_wrap(&src_obj, rfd);
440 brix_sd_posix_wrap(&dst_obj, dfd);
441 if (stage_move_objects(&src_obj, &dst_obj) != NGX_OK
442 || dst_obj.driver->fsync(&dst_obj) != NGX_OK) {
443 e = errno; close(rfd); close(dfd); (void) unlink(tmp); errno = e;
444 return NGX_ERROR;
445 }
446 }
447 close(rfd);
448 if (close(dfd) != 0) {
449 e = errno; (void) unlink(tmp); errno = e; return NGX_ERROR;
450 }
451 if (rename(tmp, final_path) != 0) {
452 e = errno; (void) unlink(tmp); errno = e; return NGX_ERROR;
453 }
454 (void) unlink(stage_path); /* drop the cache copy; commit succeeded */
455 return NGX_OK;
456}

458/*
459 * When an upload completes, the (complete) staged file must be moved from the
460 * stage device to the final storage. With a synchronous commit the client waits
461 * for that move, but if the worker dies mid-move the COMPLETE file is left in the
462 * cache with nothing recording where it should go. A marker file
463 * "<stage_partial>.commit" (content = the final absolute path) is written +
464 * fsync'd just before the move and removed after it; if the move is interrupted
465 * the marker survives, and brix_stage_reap_dir() finishes the move on the next
466 * worker startup / periodic sweep — so complete-but-uncommitted files are tracked
467 * across restarts and always reach storage.
468 */
469#define BRIX_STAGE_COMMIT_SUFFIX ".commit"
470#define BRIX_STAGE_MAX_DIRS 32

472static char s_stage_dirs[BRIX_STAGE_MAX_DIRS][PATH_MAX];
473static ngx_uint_t s_stage_dir_count;
474
475void
476brix_stage_dir_register(const char *canon)
477{
478 ngx_uint_t i;
479
480 if (canon == NULL || canon[0] == '\0' || strlen(canon) >= PATH_MAX) {
481 return;
482 }
483 for (i = 0; i < s_stage_dir_count; i++) {
484 if (strcmp(s_stage_dirs[i], canon) == 0) {
485 return; /* already registered (dedup across server blocks) */
486 }
487 }
488 if (s_stage_dir_count >= BRIX_STAGE_MAX_DIRS) {
489 return;
490 }
491 ngx_memcpy(s_stage_dirs[s_stage_dir_count], canon, strlen(canon) + 1);
492 s_stage_dir_count++;
493}

540/* Finish every pending stage-out recorded in stage_dir. Returns the count of
541 * commits completed this pass. Idempotent and crash-safe: a marker whose partial
542 * is already gone (committed by a racing pass / a client retry) is just dropped. */
543ngx_uint_t
544brix_stage_reap_dir(const char *stage_dir, ngx_log_t *log)
545{
546 DIR *d;
547 struct dirent *de;
548 ngx_uint_t done = 0;
549 size_t slen = sizeof(BRIX_STAGE_COMMIT_SUFFIX) - 1;
550 /* Snapshot marker basenames first — we unlink while iterating, which would
551 * otherwise make readdir skip entries. */
552 char names[256][256];
553 ngx_uint_t ncount = 0, i;
554
555 if (stage_dir == NULL || stage_dir[0] == '\0') {
556 return 0;
557 }
558 d = opendir(stage_dir);
559 if (d == NULL) {
560 return 0;
561 }
562 while ((de = readdir(d)) != NULL && ncount < 256) {
563 size_t nlen = strlen(de->d_name);
564 if (nlen > slen && nlen < sizeof(names[0])
565 && strcmp(de->d_name + nlen - slen, BRIX_STAGE_COMMIT_SUFFIX) == 0)
566 {
567 ngx_memcpy(names[ncount], de->d_name, nlen + 1);
568 ncount++;
569 }
570 }
571 closedir(d);
572
573 for (i = 0; i < ncount; i++) {
574 char marker[PATH_MAX], partial[PATH_MAX], final[PATH_MAX];
575 size_t nlen = strlen(names[i]);
576 int mfd, n;
577 ssize_t r;
578 struct stat sb;
579
580 n = snprintf(marker, sizeof(marker), "%s/%s", stage_dir, names[i]);
581 if (n < 0 || (size_t) n >= sizeof(marker)) { continue; }
582 n = snprintf(partial, sizeof(partial), "%s/%.*s", stage_dir,
583 (int) (nlen - slen), names[i]);
584 if (n < 0 || (size_t) n >= sizeof(partial)) { continue; }
585
586 mfd = open(marker, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
587 if (mfd < 0) { continue; }
588 r = read(mfd, final, sizeof(final) - 1);
589 close(mfd);
590 if (r <= 0) { (void) unlink(marker); continue; }
591 final[r] = '\0';
592 while (r > 0 && (final[r - 1] == '\n' || final[r - 1] == '\r'
593 || final[r - 1] == ' ')) {
594 final[--r] = '\0';
595 }
596 if (final[0] != '/') { (void) unlink(marker); continue; } /* sanity */
597
598 if (stat(partial, &sb) != 0) {
599 /* Partial already gone (committed elsewhere) — drop the marker. */
600 (void) unlink(marker);
601 continue;
602 }
603 /* Crash-recovery: no open fd, so final_mode is unused (fchmod is
604 * fd-guarded); the recovered partial keeps its on-disk mode. */
605 if (brix_commit_staged(NGX_INVALID_FILE, partial, final, 0, log)
606 == NGX_OK)
607 {
608 (void) unlink(marker);
609 done++;
610 ngx_log_error(NGX_LOG_NOTICE, log, 0,
611 "brix: completed pending stage-out \"%s\" -> \"%s\"",
612 partial, final);
613 } else {
614 ngx_log_error(NGX_LOG_WARN, log, errno,
615 "brix: pending stage-out \"%s\" -> \"%s\" failed; will retry",
616 partial, final);
617 }
618 }
619 return done;
620}

632/*
633 * WHAT: Atomically rename the temp file to its final path and clean up.
634 *
635 * WHY: After all data has been written to the staged temp file, commit makes it visible at
636 * the target location. The rename is atomic on POSIX filesystems — readers see either
637 * the old file or the new one, never a partial write.
638 *
639 * HOW: Close the fd if still open (data should be flushed). Rename tmp_path → final_path via
640 * brix_rename_confined_canon(). On rename failure: unlink the temp file as cleanup. Set
641 * staged->active=0 and clear tmp_path buffer. Return NGX_OK on success, NGX_ERROR on fail.
642 *
643 * Parameters:
644 * log — nginx log for error/cleanup reporting
645 * root_canon — canonical root for confined rename
646 * staged — the staged file struct (must be active)
647 * final_path — destination path to rename into
648 */
649static ngx_int_t
650staged_commit_internal(ngx_log_t *log, const char *root_canon,
651 brix_staged_file_t *staged, const char *final_path, int exclusive)
652{
653 if (staged == NULL || !staged->active) {
654 errno = EINVAL;
655 return NGX_ERROR;
656 }
657
658 int rootfd;
659 int rc;
660 const char *tmp_rel, *final_rel;
661
662 /* Open the confinement root first so the fsync-failure cleanup path (C1) can
663 * unlink the temp without re-opening it. */
664 rootfd = brix_beneath_open_root(root_canon);
665 if (rootfd < 0) {
666 if (staged->fd != NGX_INVALID_FILE) {
667 ngx_close_file(staged->fd);
668 staged->fd = NGX_INVALID_FILE;
669 }
670 staged->active = 0;
671 return NGX_ERROR;
672 }
673 tmp_rel = brix_beneath_strip_root(root_canon, staged->tmp_path);
674 final_rel = brix_beneath_strip_root(root_canon, final_path);
675 if (tmp_rel == NULL || final_rel == NULL) {
676 if (staged->fd != NGX_INVALID_FILE) {
677 ngx_close_file(staged->fd);
678 staged->fd = NGX_INVALID_FILE;
679 }
680 close(rootfd);
681 staged->active = 0;
682 errno = EXDEV;
683 return NGX_ERROR;
684 }
685
686 /*
687 * Phase 51 (C1): flush the staged data to stable storage BEFORE the rename
688 * publishes it, so a crash / power loss / ENOSPC mid-write cannot expose a
689 * torn object. A failed fsync means the data is NOT durable — fail the
690 * commit (unlink the temp, leave the final path untouched) rather than
691 * publish possibly-incomplete data. (close() alone does not flush.)
692 */
693 if (staged->fd != NGX_INVALID_FILE) {
694 if (fsync(staged->fd) != 0) {
695 int e = errno;
696 ngx_log_error(NGX_LOG_ERR, log, e,
697 "brix: staged commit fsync failed — not publishing "
698 "\"%s\"", final_path);
699 ngx_close_file(staged->fd);
700 staged->fd = NGX_INVALID_FILE;
701 (void) brix_unlink_beneath(rootfd, tmp_rel, 0);
702 close(rootfd);
703 staged->active = 0;
704 errno = e;
705 return NGX_ERROR;
706 }
707 /* SECURITY: restore the caller's intended mode on the OPEN fd just before
708 * the rename publishes it. The temp was written 0600 (private); rename
709 * preserves the fd's mode, so the committed namespace object carries its
710 * client-intended bits (e.g. 0644) with no world-readable in-flight
711 * window. */
712 (void) fchmod(staged->fd, staged->final_mode);
713 ngx_close_file(staged->fd);
714 staged->fd = NGX_INVALID_FILE;
715 }
716
717 rc = exclusive ? brix_rename_beneath_excl(rootfd, tmp_rel, final_rel)
718 : brix_rename_beneath(rootfd, tmp_rel, final_rel);
719 if (rc != 0) {
720 int e = errno; /* preserve EEXIST for the caller */
721 (void) brix_unlink_beneath(rootfd, tmp_rel, 0);
722 close(rootfd);
723 staged->active = 0;
724 errno = e;
725 return NGX_ERROR;
726 }
727
728 /* C1: persist the directory entry so the rename itself survives a crash
729 * (best-effort — the data is already durable above). */
730 (void) fsync(rootfd);
731
732 close(rootfd);
733 staged->active = 0;
734 staged->tmp_path[0] = '\0';
735 return NGX_OK;
736}

752/*
753 * WHAT: Close the temp file fd and optionally unlink it from disk.
754 *
755 * WHY: On write failure or client disconnect, abort cleans up the staged temp file so it
756 * doesn't leak on disk. The caller decides whether to remove the temp file (remove_tmp=1)
757 * or leave it for later inspection (remove_tmp=0).
758 *
759 * HOW: Close fd if open and valid. If remove_tmp is set AND staged is active AND tmp_path is
760 * non-empty: unlink via brix_unlink_confined_canon(). Always set active=0 and clear
761 * tmp_path buffer.
762 *
763 * Parameters:
764 * log — nginx log for cleanup error reporting
765 * root_canon — canonical root for confined unlink
766 * staged — the staged file struct (NULL-safe)
767 * remove_tmp — 1 to delete temp file, 0 to leave it on disk
768 */
769void
770brix_staged_abort(ngx_log_t *log, const char *root_canon,
771 brix_staged_file_t *staged, ngx_flag_t remove_tmp)
772{
773 if (staged == NULL) {
774 return;
775 }
776
777 (void) log;
778
779 if (staged->fd != NGX_INVALID_FILE) {
780 ngx_close_file(staged->fd);
781 staged->fd = NGX_INVALID_FILE;
782 }
783
784 if (remove_tmp && staged->active && staged->tmp_path[0] != '\0') {
785 int rootfd = brix_beneath_open_root(root_canon);
786 const char *tmp_rel;
787
788 if (rootfd >= 0) {
789 tmp_rel = brix_beneath_strip_root(root_canon, staged->tmp_path);
790 if (tmp_rel != NULL) {
791 (void) brix_unlink_beneath(rootfd, tmp_rel, 0);
792 }
793 close(rootfd);
794 }
795 }
796
797 staged->active = 0;
798 staged->tmp_path[0] = '\0';
799}
