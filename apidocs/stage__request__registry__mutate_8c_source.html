1/*
2 * stage_request_registry_mutate.c — mutating client ops of the durable
3 * stage-request registry (split from stage_request_registry.c, phase-79).
4 *
5 * WHAT: The write side of the reqid-keyed durable request store — mint a reqid,
6 * admit a request, set its status, cancel/delete it, and reap expired
7 * records. Each op takes the whole-store lock, mutates a fixed slot with
8 * WAL ordering (record fdatasync then header fsync), and unlocks.
9 * WHY: The single 956-line registry owned three concerns; isolating the
10 * mutating ops here keeps every translation unit under the file-size cap
11 * and focused. The durable-store substrate (file I/O, locking, slot scan)
12 * stays in stage_request_registry.c and is reached only through the seam
13 * declared in stage_request_registry_internal.h — the lock/unlock
14 * boundaries and slot math are unchanged by the split.
15 * HOW: Every public op is a flat early-return sequence over the substrate
16 * helpers: srq_lock -> srq_hdr_read / srq_offset_by_reqid / srq_alloc_slot
17 * -> srq_rec_write + srq_hdr_write -> srq_unlock. The pure precondition and
18 * record-marshalling helpers for add() are kept local and side-effect-free.
19 */
20

29ngx_int_t
30brix_stage_request_reqid_generate(brix_stage_registry_t *reg,
31 char *reqid_out, size_t reqid_out_sz, ngx_log_t *log)
32{
33 srq_hdr_t hdr;
34 uint64_t seq;
35
36 if (reg == NULL || reg->fd < 0 || reqid_out == NULL
37 || reqid_out_sz < SRQ_REQID_LEN)
38 {
39 return NGX_ERROR;
40 }
41 if (srq_lock(reg, SRQ_LK_EXCL) != NGX_OK) {
42 return NGX_ERROR;
43 }
44 if (srq_hdr_read(reg, &hdr, log) != NGX_OK) {
45 srq_unlock(reg);
46 return NGX_ERROR;
47 }
48 seq = ++hdr.seq;
49 if (srq_hdr_write(reg, &hdr, log) != NGX_OK) {
50 srq_unlock(reg);
51 return NGX_ERROR;
52 }
53 srq_unlock(reg);
54 srq_reqid_format(reg->host, seq, reqid_out, reqid_out_sz);
55 return NGX_OK;
56}

61/* ---- Validate the add-request arguments before any lock is taken ----
62 *
63 * WHAT: Return NGX_OK when every argument brix_stage_request_add needs is
64 * usable, otherwise NGX_ERROR. Rejects null/closed registry, a missing view or
65 * lfn, a too-small reqid_out buffer, and an lfn wider than the on-disk field
66 * (the last logs the width violation).
67 *
68 * WHY: Keeps the mutating orchestrator flat and side-effect-honest by pulling
69 * the pure precondition checks out of the locked critical section — none of
70 * these touch the file, so they belong entirely before srq_lock.
71 *
72 * HOW:
73 * 1. Reject any null/closed handle or too-small caller buffer.
74 * 2. Reject an lfn that would not fit SRQ_LFN_LEN, logging the reason.
75 * 3. Return NGX_OK when all checks pass.
76 */
77static ngx_int_t
78srq_add_validate_args(brix_stage_registry_t *reg,
79 const brix_stage_request_view_t *view, const char *reqid_out,
80 size_t reqid_out_sz, ngx_log_t *log)
81{
82 if (reg == NULL || reg->fd < 0 || view == NULL || view->lfn == NULL
83 || reqid_out == NULL || reqid_out_sz < SRQ_REQID_LEN)
84 {
85 return NGX_ERROR;
86 }
87 if (ngx_strlen(view->lfn) >= SRQ_LFN_LEN) {
88 ngx_log_error(NGX_LOG_ERR, log, 0, "stage-req: lfn exceeds %d bytes",
89 SRQ_LFN_LEN);
90 return NGX_ERROR;
91 }
92 return NGX_OK;
93}

95/* ---- Populate a blank on-disk record from a caller's request view ----
96 *
97 * WHAT: Zero-fill *rec and fill every field for a freshly QUEUED request from
98 * *view, stamping the self-offset, timestamps, reqid, lfn and the optional
99 * requester_dn/user/cs_value strings.
100 *
101 * WHY: This is pure record marshalling with no I/O and no lock interaction, so
102 * isolating it keeps brix_stage_request_add a short, linear sequence and makes
103 * the field mapping independently reviewable.
104 *
105 * HOW:
106 * 1. Zero the record and set the fixed QUEUED-state scalar fields.
107 * 2. Format the reqid from the host + durable seq and copy the mandatory lfn.
108 * 3. Copy each optional string only when the view supplies it.
109 */
110static void
111srq_rec_populate_from_view(srq_rec_t *rec,
112 const brix_stage_request_view_t *view, const char *host, uint64_t seq,
113 int64_t off, time_t now)
114{
115 ngx_memzero(rec, sizeof(*rec));
116 rec->self = off;
117 rec->status = SRQ_ST_QUEUED;
118 rec->cs_type = (uint8_t) view->cs_type;
119 rec->tod_added = (int64_t) now;
120 rec->tod_status = (int64_t) now;
121 rec->tod_expire = view->tod_expire;
122 rec->opaque_off = -1;
123 rec->lfn_url_off = -1;
124 srq_reqid_format(host, seq, rec->reqid, sizeof(rec->reqid));
125 ngx_cpystrn((u_char *) rec->lfn, (u_char *) view->lfn, sizeof(rec->lfn));
126 if (view->requester_dn) {
127 ngx_cpystrn((u_char *) rec->requester_dn,
128 (u_char *) view->requester_dn, sizeof(rec->requester_dn));
129 }
130 if (view->user) {
131 ngx_cpystrn((u_char *) rec->user, (u_char *) view->user,
132 sizeof(rec->user));
133 }
134 if (view->cs_value) {
135 ngx_cpystrn((u_char *) rec->cs_value, (u_char *) view->cs_value,
136 sizeof(rec->cs_value));
137 }
138}

140ngx_int_t
141brix_stage_request_add(brix_stage_registry_t *reg,
142 const brix_stage_request_view_t *view, char *reqid_out,
143 size_t reqid_out_sz, ngx_log_t *log)
144{
145 srq_hdr_t hdr;
146 srq_rec_t rec;
147 int64_t off;
148 uint64_t seq;
149 time_t now;
150
151 if (srq_add_validate_args(reg, view, reqid_out, reqid_out_sz, log)
152 != NGX_OK)
153 {
154 return NGX_ERROR;
155 }
156 now = time(NULL);
157
158 if (srq_lock(reg, SRQ_LK_EXCL) != NGX_OK) {
159 return NGX_ERROR;
160 }
161 if (srq_hdr_read(reg, &hdr, log) != NGX_OK) {
162 srq_unlock(reg);
163 return NGX_ERROR;
164 }
165 off = srq_alloc_slot(reg, log);
166 if (off < 0) {
167 srq_unlock(reg);
168 return NGX_ERROR;
169 }
170 seq = ++hdr.seq;
171
172 srq_rec_populate_from_view(&rec, view, reg->host, seq, off, now);
173
174 if (srq_rec_write(reg, off, &rec, log) != NGX_OK
175 || srq_hdr_write(reg, &hdr, log) != NGX_OK)
176 {
177 srq_unlock(reg);
178 return NGX_ERROR;
179 }
180 srq_unlock(reg);
181
182 ngx_cpystrn((u_char *) reqid_out, (u_char *) rec.reqid, reqid_out_sz);
183 return NGX_OK;
184}

186ngx_int_t
187brix_stage_request_set_status(brix_stage_registry_t *reg,
188 const char *reqid, brix_stage_req_status_t status, ngx_log_t *log)
189{
190 srq_rec_t rec;
191 int64_t off;
192
193 if (reg == NULL || reg->fd < 0 || reqid == NULL) {
194 return NGX_ERROR;
195 }
196 if (srq_lock(reg, SRQ_LK_EXCL) != NGX_OK) {
197 return NGX_ERROR;
198 }
199 off = srq_offset_by_reqid(reg, reqid, log);
200 if (off < 0 || srq_rec_read(reg, off, &rec, log) != NGX_OK
201 || !srq_rec_valid(&rec, off) || rec.status == SRQ_ST_FREE)
202 {
203 srq_unlock(reg);
204 return NGX_DECLINED;
205 }
206 rec.status = srq_status_on_disk(status);
207 rec.tod_status = (int64_t) time(NULL);
208 if (srq_rec_write(reg, off, &rec, log) != NGX_OK) {
209 srq_unlock(reg);
210 return NGX_ERROR;
211 }
212 srq_unlock(reg);
213 return NGX_OK;
214}

256ngx_uint_t
257brix_stage_request_reap_expired(brix_stage_registry_t *reg, time_t now,
258 ngx_log_t *log)
259{
260 srq_rec_t rec;
261 int64_t off, size;
262 ngx_uint_t reaped = 0;
263
264 if (reg == NULL || reg->fd < 0) {
265 return 0;
266 }
267 if (srq_lock(reg, SRQ_LK_EXCL) != NGX_OK) {
268 return 0;
269 }
270 size = srq_file_size(reg);
271 for (off = SRQ_REC_OFF(0);
272 size > 0 && off + (int64_t) SRQ_REC_SIZE <= size;
273 off += (int64_t) SRQ_REC_SIZE)
274 {
275 if (srq_rec_read(reg, off, &rec, log) != NGX_OK) {
276 break;
277 }
278 if (!srq_rec_valid(&rec, off) || rec.status == SRQ_ST_FREE
279 || rec.tod_expire == 0 || rec.tod_expire > (int64_t) now)
280 {
281 continue;
282 }
283 ngx_memzero(&rec, sizeof(rec));
284 rec.status = SRQ_ST_FREE;
285 if (srq_rec_write(reg, off, &rec, log) == NGX_OK) {
286 reaped++;
287 }
288 }
289 srq_unlock(reg);
290 return reaped;
291}
