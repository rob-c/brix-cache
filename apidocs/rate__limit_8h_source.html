1/*
2 * rate_limit.h — cross-worker token-bucket rate limiter over the KV store.
3 *
4 * State (token count + last-refill timestamp) lives per identity in a KV
5 * zone, so the limit is enforced across all worker processes. The check is
6 * best-effort: get and set are two separate lock acquisitions, so a small
7 * over-admission is possible under high concurrency — acceptable for request
8 * throttling.
9 *
10 * key=dn limits per authenticated DN (stream/GSI, post-auth)
11 * key=ip limits per client IP (used before identity is known, e.g. HTTP)
12 */
13#ifndef BRIX_SHM_RATE_LIMIT_H
14#define BRIX_SHM_RATE_LIMIT_H

16#include "kv.h"

18/* Per-conf settings, filled by brix_rate_limit_directive(). */
19typedef struct {
20 brix_kv_t *kv; /* NULL = disabled */

22 ngx_uint_t burst; /* bucket capacity */
23 ngx_uint_t key_ip; /* 0 = key by DN, 1 = key by client IP */
24} brix_rate_limit_conf_t;
