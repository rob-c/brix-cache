7/*
8 *
9 * WHAT: Queries the size of an extended attribute's stored value (NULL buffer)
10 * through the VFS xattr seam — path mode (brix_vfs_getxattr on the vctx's
11 * resolved path) or open-handle mode (brix_vfs_fgetxattr on fd) selected
12 * by `path != NULL`. Returns ssize_t: positive = value length in bytes,
13 * -1 = error (errno set). First step before fattr_read_value() reads data.
14 *
15 * WHY: Two-phase read pattern — query size first then allocate buffer
16 * second avoids over-allocation and handles zero-length attributes
17 * correctly. Routing through the VFS keeps every xattr touch confined and
18 * observed (OP_XATTR metric) rather than hitting getxattr(2) directly. */
19static ssize_t
20fattr_read_value_size(brix_vfs_ctx_t *vctx, const char *path, int fd,
21 const char *xkey)
22{
23 return path != NULL ? brix_vfs_getxattr(vctx, xkey, NULL, 0)
24 : brix_vfs_fgetxattr(vctx, fd, xkey, NULL, 0);
25}

48 *
49 * WHAT: Returns the number of bytes an attribute's value contributes to the
50 * vvec (value vector) portion of the kXR_fattrGet response body. If
51 * attr->vlen > 0 returns that value as size_t; if vlen == 0 (error or
52 * zero-length attribute) returns 0. Used during response-size estimation

60/*
61 *
62 * WHAT: Iterates over requested attribute names from the nvec (name vector),
63 * queries each value size via getxattr/fgetxattr, allocates buffers for
64 * successful reads, then builds a vector status response containing
65 * error count, total attr count, name vector copy, and per-attribute
66 * value lengths + raw bytes. Sends the response as kXR_ok with the
67 * complete vvec payload. Supports both path-based and open-file-handle
68 * operations — dispatches to getxattr or fgetxattr based on whether
69 * path parameter is non-null. Thread safety: operates only on provided
70 * ctx, c, pool and local stack variables (attrs array).
71 *
72 * WHY: XRootD kXR_fattrGet returns a structured vector response where each
73 * attribute gets its own value_len field followed by raw bytes in the
74 * vvec. Per-attribute error recording allows partial success — clients
75 * can read some attributes even if others fail (e.g., permission denied
76 * on specific xkeys). Two-phase read (size query then buffer alloc +
77 * read) prevents over-allocation and handles zero-length attributes
78 * correctly. Value length capped at kXR_faMaxVlen to prevent oversized
79 * responses from a single attribute. */
