35ckp_recover_one(ngx_log_t *log, const char *root_canon,
36 const char *ckp_path)
37{
38 char orig_path[PATH_MAX];
39 size_t len;
40 int ckp_fd;
41 brix_staged_file_t staged;
42 struct stat st;
43
44 len = strlen(ckp_path);
45 if (len <= 4 || len >= sizeof(orig_path)) {
46 return NGX_ERROR;
47 }
48
49 ngx_memcpy(orig_path, ckp_path, len - 4);
50 orig_path[len - 4] = '\0';
51
52 ckp_fd = brix_vfs_open_fd(log, root_canon, ckp_path,
53 O_RDONLY | O_CLOEXEC | O_NOFOLLOW, 0);
54 if (ckp_fd < 0) {
55 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
56 "brix: checkpoint recovery cannot open \"%s\"",
57 ckp_path);
58 return NGX_ERROR;
59 }
60
61 if (fstat(ckp_fd, &st) != 0 || !S_ISREG(st.st_mode)) {
62 ngx_close_file(ckp_fd);
63 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
64 "brix: checkpoint recovery invalid snapshot "
65 "\"%s\"", ckp_path);
66 return NGX_ERROR;
67 }
68
69 {
70 brix_staged_open_req_t oreq = {
71 .root_canon = root_canon,
72 .final_path = orig_path,
73 .open_flags = O_WRONLY,
74 .mode = 0600,
75 .attempts = 16,
76 };
77 if (brix_staged_open(log, &oreq, &staged) != NGX_OK) {
78 ngx_close_file(ckp_fd);
79 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
80 "brix: checkpoint recovery cannot stage "
81 "\"%s\"", orig_path);
82 return NGX_ERROR;
83 }
84 }
85
86 if (st.st_size > 0
87 && brix_copy_range(log, ckp_fd, 0, staged.fd, 0,
88 (size_t) st.st_size, ckp_path,
89 staged.tmp_path) != NGX_OK)
90 {
91 brix_staged_abort(log, root_canon, &staged, 1);
92 ngx_close_file(ckp_fd);
93 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
94 "brix: checkpoint recovery copy failed for "
95 "\"%s\"", orig_path);
96 return NGX_ERROR;
97 }
98
99 {
100 brix_vfs_job_t job;
101
102 brix_vfs_job_sync_init(&job, staged.fd);
103 brix_vfs_io_execute(&job);
104 }
105
106 if (brix_staged_commit(log, root_canon, &staged, orig_path)
107 != NGX_OK)
108 {
109 ngx_close_file(ckp_fd);
110 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
111 "brix: checkpoint recovery commit failed for "
112 "\"%s\"", orig_path);
113 return NGX_ERROR;
114 }
115
116 ngx_close_file(ckp_fd);
117
118 if (brix_vfs_unlink_path(log, root_canon, ckp_path) != 0) {
119 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
120 "brix: checkpoint recovery cannot remove "
121 "\"%s\"", ckp_path);
122 return NGX_ERROR;
123 }
124
125 brix_log_safe_path(log, NGX_LOG_NOTICE, 0,
126 "brix: recovered abandoned checkpoint \"%s\"",
127 ckp_path);
128 return NGX_OK;
129}

131/* One opened recovery-scan directory: the stream plus its validated fd. */
132typedef struct {
133 DIR *dp; /* open stream — caller owns/closes it */
134 int scan_fd; /* dirfd(dp), validated >= 0 (safe for fstatat) */
135} ckp_scan_dir_t;
136
137/*
138 * Open one recovery-scan directory confined to the export root and hand back
139 * the DIR stream plus its validated directory fd (used for fstatat). Returns
140 * NGX_OK (caller owns out->dp), NGX_DECLINED for a skippable subdirectory
141 * (private/racing-removed — recovery must continue past it), or NGX_ERROR for
142 * a fatal condition (inaccessible export root or stream setup failure).
143 */
144static ngx_int_t
145ckp_recover_open_dir(ngx_log_t *log, const char *root_canon, const char *dir,
146 ngx_uint_t depth, ckp_scan_dir_t *out)
147{
148 DIR *dp;
149 int dfd, scan_fd;
150
151 dfd = brix_vfs_open_fd(log, root_canon, dir,
152 O_RDONLY | O_DIRECTORY | O_CLOEXEC
153 | O_NOFOLLOW, 0);
154 if (dfd < 0) {
155 /*
156 * A SUBDIRECTORY we cannot enter must NOT abort recovery: under per-request
157 * impersonation the export legitimately contains per-user PRIVATE dirs
158 * (e.g. 0700) the worker uid cannot read, and a dir can be removed mid-scan.
159 * Skip those (recovery only concerns this server's own .ckp temp files,
160 * which live in dirs the worker can reach). Only an inaccessible EXPORT
161 * ROOT (depth 0) or an unexpected errno is fatal.
162 */
163 if (depth > 0 && (ngx_errno == EACCES || ngx_errno == ENOENT
164 || ngx_errno == ENOTDIR || ngx_errno == ELOOP))
165 {
166 brix_log_safe_path(log, NGX_LOG_INFO, ngx_errno,
167 "brix: checkpoint recovery skipping "
168 "inaccessible dir \"%s\"", dir);
169 return NGX_DECLINED;
170 }
171 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
172 "brix: checkpoint recovery cannot scan \"%s\"",
173 dir);
174 return NGX_ERROR;
175 }
176
177 dp = fdopendir(dfd);
178 if (dp == NULL) {
179 ngx_close_file(dfd);
180 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
181 "brix: checkpoint recovery cannot scan \"%s\"",
182 dir);
183 return NGX_ERROR;
184 }
185
186 /* fdopendir owns dfd now; re-derive the scan fd once and refuse to walk on
187 * a failed dirfd() so fstatat never sees a negative directory fd. */
188 scan_fd = dirfd(dp);
189 if (scan_fd < 0) {
190 closedir(dp);
191 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
192 "brix: checkpoint recovery cannot scan \"%s\"",
193 dir);
194 return NGX_ERROR;
195 }
196
197 out->dp = dp;
198 out->scan_fd = scan_fd;
199 return NGX_OK;
200}

202static ngx_int_t
203ckp_recover_scan(ngx_log_t *log, const char *root_canon, const char *dir,
204 ngx_uint_t depth)
205{
206 ckp_scan_dir_t sd;
207 struct dirent *de;
208 ngx_int_t rc;
209
210 if (depth > 128) {
211 brix_log_safe_path(log, NGX_LOG_ERR, 0,
212 "brix: checkpoint recovery depth exceeded at "
213 "\"%s\"", dir);
214 return NGX_ERROR;
215 }
216
217 rc = ckp_recover_open_dir(log, root_canon, dir, depth, &sd);
218 if (rc == NGX_DECLINED) {
219 return NGX_OK; /* skippable subdirectory — logged by the helper */
220 }
221 if (rc != NGX_OK) {
222 return NGX_ERROR;
223 }
224
225 while ((de = readdir(sd.dp)) != NULL) {
226 char path[PATH_MAX];
227 size_t dlen, nlen;
228 struct stat st;
229
230 if (strcmp(de->d_name, ".") == 0 || strcmp(de->d_name, "..") == 0) {
231 continue;
232 }
233
234 dlen = strlen(dir);
235 nlen = strlen(de->d_name);
236 if (dlen + 1 + nlen >= sizeof(path)) {
237 closedir(sd.dp);
238 return NGX_ERROR;
239 }
240
241 ngx_memcpy(path, dir, dlen);
242 path[dlen] = '/';
243 ngx_memcpy(path + dlen + 1, de->d_name, nlen + 1);
244
245 if (fstatat(sd.scan_fd, de->d_name, &st, AT_SYMLINK_NOFOLLOW) != 0) {
246 /* A transiently-removed or inaccessible entry: skip it, don't abort
247 * the whole recovery (and thus the worker). */
248 brix_log_safe_path(log, NGX_LOG_INFO, ngx_errno,
249 "brix: checkpoint recovery skipping entry "
250 "\"%s\"", path);
251 continue;
252 }
253
254 if (S_ISDIR(st.st_mode)) {
255 if (ckp_recover_scan(log, root_canon, path, depth + 1) != NGX_OK) {
256 closedir(sd.dp);
257 return NGX_ERROR;
258 }
259 continue;
260 }
261
262 if (S_ISREG(st.st_mode) && ckp_name_has_suffix(de->d_name)) {
263 if (ckp_recover_one(log, root_canon, path) != NGX_OK) {
264 closedir(sd.dp);
265 return NGX_ERROR;
266 }
267 }
268 }
269
270 closedir(sd.dp);
271 return NGX_OK;
272}

274ngx_int_t
275brix_chkpoint_recover_root(ngx_log_t *log, const char *root_canon)
276{
277 char lock_path[PATH_MAX];
278 size_t root_len;
279 int lock_fd;
280 ngx_int_t rc;
281
282 if (root_canon == NULL || root_canon[0] == '\0') {
283 return NGX_OK;
284 }
285 /* A pure cache node (no brix_root) anchors at the "/" namespace: there is no
286 * local export tree, so no checkpoint journal to recover (and "/" is not a
287 * writable place to drop a recovery lock). Nothing to do. */
288 if (root_canon[0] == '/' && root_canon[1] == '\0') {
289 return NGX_OK;
290 }
291
292 root_len = strlen(root_canon);
293 if (root_len + sizeof("/.nginx-xrootd-ckp-recovery.lock")
294 > sizeof(lock_path))
295 {
296 return NGX_ERROR;
297 }
298
299 ngx_memcpy(lock_path, root_canon, root_len);
300 ngx_memcpy(lock_path + root_len, "/.nginx-xrootd-ckp-recovery.lock",
301 sizeof("/.nginx-xrootd-ckp-recovery.lock"));
302
303 lock_fd = open(lock_path, O_CREAT | O_RDWR | O_CLOEXEC | O_NOFOLLOW, 0600);
304 if (lock_fd < 0) {
305 /* An export this worker cannot write is not a recovery failure: a .ckp
306 * snapshot is only ever produced by a worker writing INTO this root, so
307 * a root that refuses our writes cannot hold a journal to roll back.
308 * Read-only exports (EROFS) and permission-restricted ones (EACCES /
309 * EPERM — e.g. a root-master fleet whose workers drop to an
310 * unprivileged user) are legitimate read-serving deployments, so skip
311 * recovery and carry on rather than failing worker init and
312 * crash-looping the server.
313 * Every other errno still fails loudly — it signals a genuinely broken
314 * export (ENOTDIR, ENAMETOOLONG, ELOOP from a hostile symlink, …). */
315 if (ngx_errno == EACCES || ngx_errno == EPERM || ngx_errno == EROFS) {
316 brix_log_safe_path(log, NGX_LOG_WARN, ngx_errno,
317 "brix: checkpoint recovery skipped — export "
318 "root is not writable by this worker "
319 "\"%s\"", lock_path);
320 return NGX_OK;
321 }
322 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
323 "brix: checkpoint recovery lock failed "
324 "\"%s\"", lock_path);
325 return NGX_ERROR;
326 }
327
328 if (flock(lock_fd, LOCK_EX) != 0) {
329 ngx_close_file(lock_fd);
330 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
331 "brix: checkpoint recovery cannot lock "
332 "\"%s\"", lock_path);
333 return NGX_ERROR;
334 }
335
336 rc = ckp_recover_scan(log, root_canon, root_canon, 0);
337
338 (void) flock(lock_fd, LOCK_UN);
339 ngx_close_file(lock_fd);
340
341 return rc;
342}
