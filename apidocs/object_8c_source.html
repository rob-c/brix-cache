13/* GetObject range/bytes metrics — shared by the inline serve and the off-loop
14 * serve completion (brix_http_serve_offload), so both report identically. */
15static void
16s3_serve_metrics(ngx_http_request_t *r,
17 const brix_http_serve_result_t *result)
18{
19 if (result->range_result == BRIX_SERVE_RANGE_UNSATISFIED) {
20 BRIX_S3_METRIC_INC(range_total[BRIX_S3_RANGE_UNSATISFIED]);
21 } else if (result->range_result == BRIX_SERVE_RANGE_PARTIAL) {
22 BRIX_S3_METRIC_INC(range_total[BRIX_S3_RANGE_PARTIAL]);
23 } else {
24 BRIX_S3_METRIC_INC(range_total[BRIX_S3_RANGE_FULL]);
25 }
26 if (result->bytes_sent > 0) {
27 BRIX_S3_METRIC_ADD(bytes_tx_total, (size_t) result->bytes_sent);
28 if (r->connection && r->connection->sockaddr
29 && r->connection->sockaddr->sa_family == AF_INET6) {
30 BRIX_S3_METRIC_ADD(bytes_tx_ipv6_total, (size_t) result->bytes_sent);
31 } else {
32 BRIX_S3_METRIC_ADD(bytes_tx_ipv4_total, (size_t) result->bytes_sent);
33 }
34 }
35}

82/* WHY: GET is the primary S3 data path — clients download object bytes via HTTP GET or byte-range requests. Range support (RFC 7233) enables resumable downloads and parallel chunked transfers, critical for large objects in HEP workflows where files often exceed gigabytes. The range-parse → headers → body-send pipeline is shared with WebDAV GET via brix_http_serve_file_ranged() (src/shared/file_serve.c); this handler keeps only the S3-specific concerns: NoSuchKey XML errors, identity resolution, and S3 range/bytes metrics. */
83
84/* HOW: Phase 1 — open the object through the VFS layer (brix_vfs_open, read-only, cache-aware). If the open fails: ENOENT/ENOTDIR → NoSuchKey 404 XML; other errno → brix_http_errno_to_status() with internal_error metric. Phase 2 — brix_vfs_file_stat(); a directory target → NoSuchKey 404 (S3 keys are objects, not directories). Phase 3 — resolve the display identity (token subject, else access key, else "anonymous"). Phase 4 — fill brix_http_serve_opts_t (xfer_proto=S3, op_name="GetObject", etag_flags=0) and delegate the entire range-parse/header/send pipeline to brix_http_serve_file_ranged(), which also takes ownership of the vfs handle. Phase 5 — from the returned result, increment the S3 range_total[FULL/PARTIAL/UNSATISFIED] counter and, on a non-zero body, bytes_tx_total plus the IPv4/IPv6 split. */
85/*
86 * s3_handle_get - serve a file as an S3 GetObject response.

88 * Supports byte-range requests (RFC 7233). After sending headers, checks

92 * Ownership: fd is opened here and closed either immediately on error or

135 /*
136 * Tape residency (phase-64 VFS seam): a GET of a nearline/offline object cannot
137 * be served from disk — S3/Glacier semantics require an explicit restore first
138 * (the WLCG Tape REST API), so report InvalidObjectState rather than faulting a
139 * recall. Checked BEFORE any open/fill so a tape-resident object never triggers

143 {

181 /* phase-64 SP2: offload a remote cache MISS fill to the thread pool (it would
182 * otherwise stall the worker inside brix_vfs_open's inline fill) and re-
183 * enter on completion. See webdav/get.c. NGX_DECLINED ⇒ open inline below. */
184 {
185 s3_get_reenter_t *rd = ngx_palloc(r->pool, sizeof(*rd));
186 char *fp;
187 size_t fplen;
188 ngx_int_t fr;
189
190 if (rd == NULL) {
191 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
192 return NGX_HTTP_INTERNAL_SERVER_ERROR;
193 }
194 fplen = ngx_strlen(fs_path);
195 fp = (char *) ngx_pnalloc(r->pool, fplen + 1);
196 if (fp == NULL) {
197 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
198 return NGX_HTTP_INTERNAL_SERVER_ERROR;

200 ngx_memcpy(fp, fs_path, fplen + 1);
201 rd->fs_path = fp;
202 rd->cf = cf;
203
204 fr = brix_http_cache_fill_if_needed(r, vctx.sd,
205 brix_vfs_export_relative(&vctx, fs_path), &cf->common,
206 s3_get_reenter, rd);
207 if (fr == NGX_DONE) {
208 return NGX_DONE;
209 }
210 if (fr == NGX_ERROR) {
211 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
212 return NGX_HTTP_INTERNAL_SERVER_ERROR;
213 }
214 }
215
216 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
217 if (fh == NULL) {
218 if (vfs_err == ENOENT || vfs_err == ENOTDIR) {
219 return s3_fail(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
220 "The specified key does not exist.",
221 BRIX_S3_EVENT_NO_SUCH_KEY);
222 }
223 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
224 return (ngx_int_t) brix_http_errno_to_status(vfs_err);
225 }
226
227 if (brix_vfs_file_stat(fh, &vst) != NGX_OK) {
228 brix_vfs_close(fh, r->connection->log);
229 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
230 return NGX_HTTP_INTERNAL_SERVER_ERROR;
231 }
232
233 if (vst.is_directory) {
234 brix_vfs_close(fh, r->connection->log);
235 return s3_fail(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
236 "The specified key does not exist.",
237 BRIX_S3_EVENT_NO_SUCH_KEY);
238 }

240 /*
241 * Conditional GET (If-Match / If-None-Match / If-(Un)Modified-Since). On a
242 * 304/412 short-circuit we must release the handle we opened above; on
243 * NGX_DECLINED we fall through to serve the object.
244 */
245 {
246 ngx_int_t crc = s3_handle_conditional(r, vst.mtime, vst.size);
247 if (crc != NGX_DECLINED) {
248 brix_vfs_close(fh, r->connection->log);
249 return crc;
250 }
251 }
252
253 s3ctx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
254 subject = s3ctx != NULL ? brix_identity_subject_cstr(s3ctx->identity)
255 : "";
256 if (subject[0] != '\0') {
257 ngx_cpystrn((u_char *) identity, (u_char *) subject,
258 sizeof(identity));
259 } else if (cf->access_key.len > 0 && cf->access_key.data != NULL) {
260 size_t n = cf->access_key.len < sizeof(identity) - 1
261 ? cf->access_key.len
262 : sizeof(identity) - 1;
263 ngx_memcpy(identity, cf->access_key.data, n);
264 identity[n] = '\0';
265 } else {
266 ngx_cpystrn((u_char *) identity, (u_char *) "anonymous",
267 sizeof(identity));

270 /*
271 * AWS full-object checksum echo (x-amz-checksum-crc64nvme). Cache-only: emit
272 * only when the value was stored at upload time, so the read path never pays
273 * a full-file recompute. Set before serve sends the response headers; the
274 * handle's fd stays valid until serve closes it.
275 */
276 {
277 ngx_fd_t cfd = brix_vfs_file_fd(fh);
278
279 if (cfd != NGX_INVALID_FILE) {
280 s3_echo_object_checksums(r, cfd, fs_path);
281 }
282 }
283
284 /* User metadata (x-amz-meta-*): echo the stored set before the response
285 * headers are sent (the handle stays open until serve sends them). */
286 s3_echo_user_metadata(r, fs_path);
287
288 {
289 brix_http_serve_opts_t opts;
290 brix_http_serve_result_t result;
291
292 ngx_memzero(&opts, sizeof(opts));
293 opts.xfer_proto = BRIX_XFER_PROTO_S3;
294 opts.op_name = "GetObject";
295 opts.identity = identity;
296 opts.etag_flags = 0;
297 opts.compress = cf->common.compress;
298 /* phase-43 W3: apply response-* query overrides just before send. */
299 opts.pre_header_send = s3_get_pre_header;
300
301 rc = brix_http_serve_file_ranged(r, fh, &vst, fs_path, &opts,
302 &result);
303
304 if (rc == NGX_HTTP_INTERNAL_SERVER_ERROR) {
305 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
306 }
307 s3_serve_metrics(r, &result);
308 }
309
310 return rc;
311}

410/*
411 * s3_handle_head_bucket — answer a HEAD on the bucket root.
412 *
413 * WHAT: AWS SDKs (boto3, rclone, s5cmd, mc) issue HEAD /<bucket> at session
414 * start to confirm the bucket exists and learn its region before any object
415 * operation. Returns 200 + x-amz-bucket-region when the configured export
416 * root is a directory, else 404 NoSuchBucket.
417 * WHY: Without this the empty-key guard answered 400 InvalidURI and an
418 * unmodified SDK client aborted the whole session.
419 * HOW: stat() the already-canonical, confinement-anchored export root (it is
420 * the bucket); no key is involved, so no per-request path resolution is
421 * needed. Header-only response.
422 */
423ngx_int_t
424s3_handle_head_bucket(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf)
425{
426 struct stat st;
427
428 if (cf->common.root_canon[0] == '\0'
429 || stat(cf->common.root_canon, &st) != 0 /* vfs-seam-allow: HeadBucket stats the export root itself (the confinement anchor), not a path beneath it */
430 || !S_ISDIR(st.st_mode))
431 {
432 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INVALID_URI]);
433 r->headers_out.status = NGX_HTTP_NOT_FOUND;
434 r->headers_out.content_length_n = 0;
435 ngx_http_send_header(r);
436 return ngx_http_send_special(r, NGX_HTTP_LAST);
437 }
438
439 if (cf->region.len > 0) {
440 char region[64];
441 size_t n = cf->region.len < sizeof(region) - 1
442 ? cf->region.len : sizeof(region) - 1;
443 ngx_memcpy(region, cf->region.data, n);
444 region[n] = '\0';
445 (void) s3_set_header(r, "x-amz-bucket-region", region);
446 }
447
448 r->headers_out.status = NGX_HTTP_OK;
449 r->headers_out.content_length_n = 0;
450 ngx_http_send_header(r);
451 return ngx_http_send_special(r, NGX_HTTP_LAST);
452}

454/*
455 * GET /bucket?location -> GetBucketLocation
456 * */

458/*
459 * s3_handle_get_bucket_location — answer GET /<bucket>?location.
460 *
461 * WHAT: Region-discovery probe. Many SDKs call this to decide which endpoint
462 * to sign against; a failure aborts the client. Emits the LocationConstraint
463 * document carrying the configured region (empty element for "us-east-1",
464 * matching AWS, since us-east-1 has no explicit constraint).
465 * WHY: Cheap probe-satisfier; pairs with HeadBucket so unmodified SDK clients
466 * complete their pre-flight.
467 * HOW: The region is config-supplied (trusted, no XML metacharacters), so it
468 * is appended directly into the element body.
469 */
470ngx_int_t
471s3_handle_get_bucket_location(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf)
472{
473 u_char *xml;
474 size_t xml_len = 0;
475 size_t xml_capacity = 256 + cf->region.len;
476 ngx_buf_t *response_buf;
477 int is_default;
478
479 is_default = (cf->region.len == 0
480 || (cf->region.len == sizeof("us-east-1") - 1
481 && ngx_strncmp(cf->region.data,
482 (u_char *) "us-east-1",
483 cf->region.len) == 0));
484
485 xml = ngx_palloc(r->pool, xml_capacity);
486 if (xml == NULL) {
487 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
488 return NGX_HTTP_INTERNAL_SERVER_ERROR;
489 }
490
491 XML_APPEND("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
492 if (is_default) {
493 XML_APPEND("<LocationConstraint "
494 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\"/>");
495 } else {
496 XML_APPEND("<LocationConstraint "
497 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">%.*s"
498 "</LocationConstraint>",
499 (int) cf->region.len, (const char *) cf->region.data);
500 }
501
502 response_buf = ngx_create_temp_buf(r->pool, xml_len + 4);
503 if (response_buf == NULL) {
504 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
505 return NGX_HTTP_INTERNAL_SERVER_ERROR;
506 }
507 response_buf->last = ngx_cpymem(response_buf->last, xml, xml_len);
508 response_buf->last_buf = 1;
509
510 BRIX_S3_METRIC_ADD(bytes_tx_total, xml_len);
511 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
512 (ngx_str_t) ngx_string("application/xml"), response_buf);
513}

515/*
516 * DELETE /bucket/key
517 * */
518
519ngx_int_t
520s3_handle_delete(ngx_http_request_t *r,
521 const char *fs_path,
522 ngx_http_s3_loc_conf_t *cf)
523{
524 brix_vfs_ctx_t vctx;
525 ngx_int_t rc;
526
527 /* Route DELETE through the metered VFS unlink. brix_vfs_unlink unlinks a
528 * file and rmdirs an (empty) directory — a non-empty dir surfaces as
529 * ENOTEMPTY (S3 BucketNotEmpty), exactly as the old require_empty_dir path.
530 * S3 DELETE is idempotent: a missing key (ENOENT) is still 204, and counts
531 * as DELETE_MISSING. errno is read only on the NGX_ERROR branch. */
532 s3_vfs_ctx(r, fs_path, cf, &vctx);
533 rc = brix_vfs_unlink(&vctx);
