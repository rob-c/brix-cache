16static void
17s3_serve_metrics(ngx_http_request_t *r,
18 const brix_http_serve_result_t *result)
19{
20 if (result->range_result == BRIX_SERVE_RANGE_UNSATISFIED) {
21 BRIX_S3_METRIC_INC(range_total[BRIX_S3_RANGE_UNSATISFIED]);
22 } else if (result->range_result == BRIX_SERVE_RANGE_PARTIAL) {
23 BRIX_S3_METRIC_INC(range_total[BRIX_S3_RANGE_PARTIAL]);

31 BRIX_S3_METRIC_ADD(bytes_tx_ipv6_total, (size_t) result->bytes_sent);
32 } else {

98 * WHAT: Serve a ZIP archive member when the export is configured for ZIP member
99 * access (xrdcl.unzip) — reads the requested member name from the request and
100 * streams that member out of the archive object.
101 * WHY: Phase-57 W2 folds ZIP member access into the S3 GET path; keeping it in a
102 * dedicated helper keeps the GET orchestrator flat. Auth/key resolution already
103 * happened in the dispatcher, so this only serves the member.
104 * HOW: Returns NGX_DECLINED when ZIP access is off or no member was requested so
105 * the caller falls through to the normal object serve. Otherwise it returns the
106 * terminal serve status (mapping a missing member to NoSuchKey 404) — the caller
107 * must return it verbatim.
108 */
109static ngx_int_t
110s3_get_serve_zip_member(ngx_http_request_t *r, const char *fs_path,
111 ngx_http_s3_loc_conf_t *cf)
112{
113 char member[PATH_MAX];
114 int zr;
115
116 if (!cf->zip_access) {
117 return NGX_DECLINED;
118 }
119
120 zr = brix_zip_http_member_arg(r, member, sizeof(member));
121 if (zr < 0) {
122 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST, "InvalidArgument",
123 "invalid xrdcl.unzip member");
124 }
125 if (zr == 0) {
126 return NGX_DECLINED;
127 }
128
129 {
130 ngx_int_t zs = brix_zip_http_serve(r, cf->common.root_canon,

142/*
143 * WHAT: Reject a GET of a tape-resident (nearline/offline) object before any
144 * open/fill is attempted.
145 * WHY: S3/Glacier semantics require an explicit restore (the WLCG Tape REST API)
146 * before a nearline object can be read; faulting a recall on a plain GET is
147 * wrong. An export with no nearline tier reports ONLINE, so a plain disk/object
148 * export is unaffected.
149 * HOW: Returns NGX_DECLINED when the object is online (or residency is unknown);

168/*
169 * WHAT: Run the off-event-loop remote serve for a socket-wire backend (a root://
170 * primary backend, or a cache_store/stage_store served from one).
171 * WHY: Such a backend cannot open/read on the event loop; the whole open+read
172 * must run off-loop, materialise, then sendfile (mirrors webdav/get.c).
173 * HOW: Returns NGX_DECLINED when this is not a socket serve (the caller opens
174 * inline). Returns NGX_DONE when the off-loop serve took ownership. Maps a
175 * credential-gate denial (EACCES/EPERM) to 403 AccessDenied and any other
176 * offload setup failure to 500 — both terminal statuses the caller returns.
177 */
178static ngx_int_t
179s3_get_serve_offload(ngx_http_request_t *r, const char *fs_path,
180 ngx_http_s3_loc_conf_t *cf, brix_vfs_ctx_t *vctx)
181{
182 brix_http_serve_opts_t sopts;
183 ngx_int_t sr;
184
185 ngx_memzero(&sopts, sizeof(sopts));
186 sopts.xfer_proto = BRIX_XFER_PROTO_S3;
187 sopts.op_name = "GetObject";
188 sopts.identity = "";
189 sopts.etag_flags = 0;
190 sopts.compress = cf->common.compress;
191
192 sr = brix_http_serve_offload_remote(r, vctx->sd,
193 brix_vfs_export_relative(vctx, fs_path), fs_path, &sopts,
194 &cf->common, vctx, s3_serve_metrics);
195 if (sr == NGX_DONE) {
196 return NGX_DONE;
197 }
198 if (sr == NGX_ERROR) {

200 * before any origin open was attempted (phase-2 follow-up);
201 * everything else is a genuine offload setup failure. */
202 if (errno == EACCES || errno == EPERM) {
203 return s3_fail(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
204 "Access to the requested object is denied.",
205 BRIX_S3_EVENT_ACCESS_DENIED);
206 }
207 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
208 return NGX_HTTP_INTERNAL_SERVER_ERROR;
209 }
210 return NGX_DECLINED;

213/*
214 * WHAT: Offload a remote cache-MISS fill to the thread pool, re-entering
215 * s3_handle_get on completion.
216 * WHY: An inline fill inside brix_vfs_open would otherwise stall the worker
217 * (mirrors webdav/get.c). The re-entry state carries the absolute fs_path + cf
218 * copied onto r->pool so they outlive the worker-thread fill.
219 * HOW: Returns NGX_DECLINED when no fill is needed (the caller opens inline).
220 * Returns NGX_DONE when a fill was scheduled. Allocation failure or a fill
221 * setup error maps to a terminal 500 the caller returns.
222 */
223static ngx_int_t
224s3_get_cache_fill(ngx_http_request_t *r, const char *fs_path,
225 ngx_http_s3_loc_conf_t *cf, brix_vfs_ctx_t *vctx)
226{
227 s3_get_reenter_t *rd;
228 char *fp;
229 size_t fplen;
230 ngx_int_t fr;
231
232 rd = ngx_palloc(r->pool, sizeof(*rd));
233 if (rd == NULL) {
234 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
235 return NGX_HTTP_INTERNAL_SERVER_ERROR;
236 }
237 fplen = ngx_strlen(fs_path);
238 fp = (char *) ngx_pnalloc(r->pool, fplen + 1);
239 if (fp == NULL) {
240 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
241 return NGX_HTTP_INTERNAL_SERVER_ERROR;

248 brix_vfs_export_relative(vctx, fs_path), &cf->common,
249 s3_get_reenter, rd);
250 if (fr == NGX_DONE) {
251 return NGX_DONE;
252 }
253 if (fr == NGX_ERROR) {
254 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
255 return NGX_HTTP_INTERNAL_SERVER_ERROR;
256 }
257 return NGX_DECLINED;
258}

260/*
261 * WHAT: Open the object inline through the VFS and stat it, rejecting a missing
262 * key or a directory target as NoSuchKey.
263 * WHY: S3 keys are objects, not directories; a GET must resolve to a regular
264 * file. This bundles the open/stat/type checks so the orchestrator reads as one
265 * resolve step, and returns the open handle + populated stat on success.
266 * HOW: On success sets *out_fh + *out_vst and returns NGX_DECLINED (fall through
267 * to serve). On any failure it closes any handle it opened, increments the right
268 * metric, and returns the terminal status the caller returns; *out_fh is left
269 * NULL in that case.
270 */
271static ngx_int_t
272s3_get_resolve(ngx_http_request_t *r, brix_vfs_ctx_t *vctx,
273 brix_vfs_file_t **out_fh, brix_vfs_stat_t *out_vst)
274{
275 brix_vfs_file_t *fh;
276 int vfs_err = 0;
277
278 *out_fh = NULL;
279
280 fh = brix_vfs_open(vctx, BRIX_VFS_O_READ, &vfs_err);
281 if (fh == NULL) {
282 if (vfs_err == ENOENT || vfs_err == ENOTDIR) {
283 return s3_fail(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
284 "The specified key does not exist.",
285 BRIX_S3_EVENT_NO_SUCH_KEY);
286 }
287 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
288 return (ngx_int_t) brix_http_errno_to_status(vfs_err);
289 }
290
291 if (brix_vfs_file_stat(fh, out_vst) != NGX_OK) {
292 brix_vfs_close(fh, r->connection->log);
293 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
294 return NGX_HTTP_INTERNAL_SERVER_ERROR;
295 }
296
297 if (out_vst->is_directory) {
298 brix_vfs_close(fh, r->connection->log);
299 return s3_fail(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
300 "The specified key does not exist.",
301 BRIX_S3_EVENT_NO_SUCH_KEY);
302 }
303
304 *out_fh = fh;
305 return NGX_DECLINED;
306}

406/* WHY: GET is the primary S3 data path — clients download object bytes via HTTP GET or byte-range requests. Range support (RFC 7233) enables resumable downloads and parallel chunked transfers, critical for large objects in HEP workflows where files often exceed gigabytes. The range-parse → headers → body-send pipeline is shared with WebDAV GET via brix_http_serve_file_ranged() (src/shared/file_serve.c); this handler keeps only the S3-specific concerns: NoSuchKey XML errors, identity resolution, and S3 range/bytes metrics. */

408/* HOW: Phase 1 — open the object through the VFS layer (brix_vfs_open, read-only, cache-aware). If the open fails: ENOENT/ENOTDIR → NoSuchKey 404 XML; other errno → brix_http_errno_to_status() with internal_error metric. Phase 2 — brix_vfs_file_stat(); a directory target → NoSuchKey 404 (S3 keys are objects, not directories). Phase 3 — resolve the display identity (token subject, else access key, else "anonymous"). Phase 4 — fill brix_http_serve_opts_t (xfer_proto=S3, op_name="GetObject", etag_flags=0) and delegate the entire range-parse/header/send pipeline to brix_http_serve_file_ranged(), which also takes ownership of the vfs handle. 409/*
410 * s3_handle_get - serve a file as an S3 GetObject response.
411 *
412 * Supports byte-range requests (RFC 7233). After sending headers, checks
413 * r->header_only so that HEAD requests (dispatched through s3_handle_head)
414 * do not accidentally trigger a body send.
415 *
416 * Ownership: fd is opened here and closed either immediately on error or
417 * via an ngx_pool_cleanup_file registered on r->pool.
418 *
419 * Pool allocation: r->pool lifetime (request scope).
420 */

433 rc = s3_get_serve_zip_member(r, fs_path, cf);

435 return rc;
436 }

438 s3_vfs_ctx(r, fs_path, cf, &vctx);

440 /* Tape residency (phase-64 VFS seam): reject a GET of a nearline/offline
441 * object before any open/fill. NGX_DECLINED ⇒ online; fall through. */
442 rc = s3_get_check_residency(r, &vctx);
443 if (rc != NGX_DECLINED) {
444 return rc;
445 }
446
447 /* phase-64 SP3: socket-wire backend serve off the event loop. NGX_DONE ⇒
448 * served; a terminal error status ⇒ return it; NGX_DECLINED ⇒ open inline. */
449 rc = s3_get_serve_offload(r, fs_path, cf, &vctx);
450 if (rc != NGX_DECLINED) {

455 * re-enter on completion. NGX_DECLINED ⇒ open inline below. */
456 rc = s3_get_cache_fill(r, fs_path, cf, &vctx);
457 if (rc != NGX_DECLINED) {
458 return rc;
459 }
460
461 /* Resolve: open + stat inline, rejecting a missing key / directory target.
462 * On success fh is an open handle owned by the serve step below. */
463 rc = s3_get_resolve(r, &vctx, &fh, &vst);
464 if (rc != NGX_DECLINED) {
465 return rc;
466 }

469 * Conditional GET (If-Match / If-None-Match / If-(Un)Modified-Since). On a
470 * 304/412 short-circuit we must release the handle we opened above; on
471 * NGX_DECLINED we fall through to serve the object.
472 */
473 {
474 /* phase78-fp: s3_get_resolve (above, NGX_DECLINED path) populated vst via brix_vfs_file_stat before returning */
475 ngx_int_t crc = s3_handle_conditional(r, vst.mtime, vst.size); /* NOLINT(clang-analyzer-core.CallAndMessage) */
476 if (crc != NGX_DECLINED) {
477 brix_vfs_close(fh, r->connection->log);
478 return crc;
479 }

484 /* Emit checksum + user-metadata headers before serve sends the response
485 * headers (the handle stays open until serve sends them). */

487
488 {
489 s3_get_serve_t sv;
490 ngx_memzero(&sv, sizeof(sv));
491 sv.fh = fh;
492 sv.vst = &vst;
493 sv.fs_path = fs_path;
494 sv.identity = identity;
495 sv.compress = cf->common.compress;
