1/*
2 * subprocess.h — capture a child process's stdout (shared).
3 *
4 * WHAT: fork/exec a command, drain its stdout into a caller buffer, reap it.
5 * WHY: the oidc-agent / RFC-8693 token fetch is done by fork/exec'ing
6 * `oidc-token` or `curl` and reading stdout — the SAME pipe→fork→dup2→
7 * drain→waitpid skeleton appears in the native client (credrefresh.c) and
8 * the module's TPC token paths (tpc/tpc_token.c, webdav/tpc_cred.c). Both
9 * run the fork SYNCHRONOUSLY (the module does it off the event loop), so —
10 * unusually for resilience code — the mechanism itself is genuinely shared.
11 * HOW: blocks SIGCHLD across fork+waitpid so a host SIGCHLD handler (nginx's)
12 * can't reap the child first; the child restores the mask before exec. The
13 * helper is status-agnostic — it returns the child's exit code + the raw
14 * bytes; each caller maps to its own error type and parses (JSON vs plain).
15 * ngx-free; libc only. (libxrdproto)
16 */

18#define BRIX_COMPAT_SUBPROCESS_H
