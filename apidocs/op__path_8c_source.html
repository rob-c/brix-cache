1/*
2 * op_path.c — unified path extractor + resolver for namespace operations.
3 *
4 * Centralises the repeated extract_path → depth_check → resolve sequence
5 * that was previously duplicated across 10+ handler files.
6 *
7 * Phase 8: resolution no longer calls realpath(3). Confinement is enforced by
8 * the kernel at the actual filesystem operation (openat2 RESOLVE_BENEATH via the
9 * beneath API), so this layer only has to (1) reject obviously-bad paths the same
10 * way the old brix_validate_components_cstr() did — length, depth, and the
11 * forbidden "."/".." components — and (2) reproduce the per-mode existence
12 * semantics the old resolve_path* variants provided (EXISTING needs the target,
13 * WRITE needs the parent directory) through the VFS seam (brix_vfs_probe, a
14 * non-observing confined existence/type check) rather than a realpath() that
15 * would have failed. `resolved` is filled with the lexical
16 * root_canon + reqpath join (brix_beneath_full_path); it is used downstream for
17 * ACL prefix matching and access logging, NOT as a confinement boundary — the
18 * boundary is RESOLVE_BENEATH at the op. A path that escapes the export root is
19 * rejected by the kernel (EXDEV) when the operation runs.
20 */

24#include "fs/path/path_internal.h"
25#include "fs/vfs/vfs.h" /* existence/type pre-gate via the VFS seam */

30/*
31 * Validate each path component the way the retired realpath resolver's
32 * brix_validate_components_cstr() did: reject "." and ".." segments outright.
33 * RESOLVE_BENEATH would block a "../" that escapes the root, but a within-root
34 * "/a/../b" was historically rejected here too — keep that behaviour so the
35 * resolver's contract is unchanged. Returns 1 if any component is forbidden.
36 */
37int
38brix_op_path_forbidden_component(const char *reqpath)
39{
40 const char *p = reqpath;
41 const char *seg;
42
43 while (*p != '\0') {
44 while (*p == '/') {
45 p++;
46 }
47 if (*p == '\0') {
48 break;
49 }
50 seg = p;
51 while (*p != '\0' && *p != '/') {
52 p++;
53 }
54 if (brix_path_component_forbidden(seg, (size_t) (p - seg))) {
55 return 1;
56 }
57 }
58 return 0;
59}

61/*
62 * Existence gate replacing the realpath()-failure path of the old resolvers.
63 * want_dir < 0 : no check (NOEXIST).
64 * want_dir = 0 : the target itself must exist (EXISTING).
65 * want_dir = 1 : the target's PARENT directory must exist (WRITE).
66 * Returns NGX_OK when the requirement is met, NGX_DECLINED otherwise.
67 */
68/* Build a probe VFS ctx for `reqpath` (joined beneath the export root) and run a
69 * non-observing confined existence/type check (no phantom OP_STAT metric). */
70static ngx_int_t
71op_path_probe(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
72 const char *reqpath, int nofollow, brix_vfs_stat_t *vst)
73{
74 brix_vfs_ctx_t vctx;
75 char full[PATH_MAX];
76
77 brix_beneath_full_path(conf->common.root_canon, reqpath,
78 full, sizeof(full));
79 brix_vfs_ctx_init(&vctx, NULL /* no alloc in a probe */, log,
80 BRIX_PROTO_ROOT, conf->common.root_canon, NULL,
81 conf->common.allow_write, 0 /* is_tls */, NULL, full);
82 return brix_vfs_probe(&vctx, nofollow, vst);
83}

85static ngx_int_t
86op_path_existence_gate(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
87 const char *reqpath, int want_dir)
88{
89 brix_vfs_stat_t vst;
90
91 if (want_dir < 0) {
92 return NGX_OK; /* NOEXIST: nothing to verify */
93 }
94
95 if (want_dir == 0) {
96 /* For a NON-POSIX backend the operation's own driver call validates the
97 * target and returns ENOENT→NotFound, so this gate's probe is a redundant
98 * catalog lookup — skip it (the driver is the single existence check). The
99 * default POSIX export keeps the probe below: its confined lstat is cheap
100 * and preserves the existing existence-before-auth ordering exactly. */
101 if (brix_vfs_backend_resolve(conf->common.root_canon, log) != NULL) {
102 return NGX_OK;
103 }
104 /* EXISTING: the target name must resolve, confined, to something present.
105 * nofollow (lstat semantics) so a symlink — including a dangling one —
106 * counts as present: rm/chmod/mv operate on the name itself, and rm of a
107 * symlink must succeed (it never dereferences the final component). This
108 * gate is only the ACL/logging existence check; the confined VFS ops
109 * remain the security boundary, so not following the final link here
110 * weakens nothing. */
111 return op_path_probe(conf, log, reqpath, 1 /* nofollow */, &vst) == NGX_OK
112 ? NGX_OK : NGX_DECLINED;
113 }
114
115 /* WRITE: the parent directory must already exist (target may not). Derive
116 * the parent by trimming the last '/'-separated component. */
117 {
118 char parent[BRIX_MAX_PATH + 1];
119 size_t len = ngx_strlen(reqpath);
120 const char *slash;
121
122 while (len > 1 && reqpath[len - 1] == '/') {
123 len--; /* ignore trailing slashes for the split */
124 }
125 slash = reqpath + len;
126 while (slash > reqpath && *(slash - 1) != '/') {
127 slash--;
128 }
129 /* slash now points just past the parent's trailing '/'. */
130 {
131 size_t plen = (size_t) (slash - reqpath);
132 if (plen <= 1) {
133 /* parent is the export root itself — always present. */
134 return NGX_OK;
135 }
136 /* `slash` points just PAST the parent's trailing '/', so plen still
137 * includes it. Drop it: a confined POSIX lstat tolerates a trailing
138 * slash on a directory ("/w0/"), but a driver whose namespace lives
139 * in a catalog (pblock) stores the key as "/w0" and a "/w0/" lookup
140 * misses — which silently broke every non-recursive nested mkdir on a
141 * non-POSIX backend. The parent path handed to the probe must be
142 * slash-free. */
143 plen--;
144 if (plen >= sizeof(parent)) {
145 return NGX_DECLINED;
146 }
147 ngx_memcpy(parent, reqpath, plen);
148 parent[plen] = '\0';
149 }
150 if (op_path_probe(conf, log, parent, 0 /* follow */, &vst) != NGX_OK
151 || !vst.is_directory)
152 {
153 return NGX_DECLINED;
154 }
155 return NGX_OK;
156 }
157}

159/*
160 * brix_path_resolve_beneath — validate reqpath and apply the per-mode
161 * existence gate without realpath(), filling `resolved` with the confined
162 * lexical join. Shared by brix_resolve_op_path() and direct multi-path
163 * callers (kXR_mv). reqpath must already be extracted from the wire.
164 *
165 * Returns:
166 * NGX_OK — valid and the existence requirement is met; resolved filled.
167 * NGX_DECLINED — path is well-formed but the existence gate failed
168 * (EXISTING: target missing; WRITE: parent dir missing) → 404.
169 * NGX_ERROR — malformed path (depth, "."/"..", WRITE trailing slash, or
170 * the join overflowed resolved_sz) → 4xx ArgInvalid.
171 */
172ngx_int_t
173brix_path_resolve_beneath(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
174 const char *reqpath, brix_path_mode_t mode,
175 char *resolved, size_t resolved_sz)
176{
177 int want_dir;
178 int strip_trailing_slash;
179 int ok;
180 char norm[BRIX_MAX_PATH + 1];
181
182 if (brix_count_path_depth(reqpath) != NGX_OK
183 || brix_op_path_forbidden_component(reqpath))
184 {
185 return NGX_ERROR;
186 }
187
188 switch (mode) {
189 case BRIX_PATH_EXISTING: want_dir = 0; strip_trailing_slash = 0; break;
190 case BRIX_PATH_WRITE: want_dir = 1; strip_trailing_slash = 1; break;
191 case BRIX_PATH_NOEXIST: want_dir = -1; strip_trailing_slash = 0; break;
192 case BRIX_PATH_EITHER: want_dir = 1; strip_trailing_slash = 0; break;
193 default: want_dir = 0; strip_trailing_slash = 0; break;
194 }
195
196 /* A write/create target with a trailing slash is NORMALIZED, not rejected,
197 * exactly as the reference's Squash collapses it: "mkdir /d/" -> /d and
198 * "open /f/" -> /f. (We previously returned ArgInvalid, so `mkdir /d/` —
199 * which stock accepts — failed.) Scoped to the create modes so a stat/cat of
200 * "/file/" keeps its existing file-vs-dir error behavior. */
201 if (strip_trailing_slash) {
202 size_t rl = ngx_strlen(reqpath);
203 if (rl > 1 && reqpath[rl - 1] == '/') {
204 if (rl >= sizeof(norm)) {
205 return NGX_ERROR;
206 }
207 ngx_memcpy(norm, reqpath, rl + 1);
208 while (rl > 1 && norm[rl - 1] == '/') {
209 norm[--rl] = '\0';
210 }
211 reqpath = norm;
212 }
213 }
214
215 ok = (op_path_existence_gate(conf, log, reqpath, want_dir) == NGX_OK);
216 if (!ok && mode == BRIX_PATH_EITHER) {
217 ok = (op_path_existence_gate(conf, log, reqpath, 0) == NGX_OK);
218 }
219 if (!ok) {
220 return NGX_DECLINED;
221 }
222
223 if (brix_beneath_full_path(conf->common.root_canon, reqpath,
224 resolved, resolved_sz) >= (int) resolved_sz)
225 {
226 return NGX_ERROR;
227 }
228
229 return NGX_OK;
230}

248ngx_int_t
249brix_resolve_op_path(brix_ctx_t *ctx, ngx_connection_t *c,
250 ngx_uint_t op_id, const char *op_name,
251 ngx_stream_brix_srv_conf_t *conf,
252 brix_path_mode_t mode,
253 char *reqpath, size_t reqpath_sz,
254 char *resolved, size_t resolved_sz)
255{
256 ngx_int_t rc;
257
258 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
259 brix_log_access(ctx, c, op_name, "-", "-",
260 0, kXR_ArgMissing, "no path given", 0);
261 BRIX_OP_ERR(ctx, op_id);
262 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgMissing,
263 "no path given");
264 return NGX_DONE;
265 }
266
267 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
268 reqpath, reqpath_sz, 1)) {
269 brix_log_access(ctx, c, op_name, "-", "-",
270 0, kXR_ArgInvalid, "invalid path payload", 0);
271 BRIX_OP_ERR(ctx, op_id);
272 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
273 "invalid path payload");
274 return NGX_DONE;
275 }
276
277 rc = brix_path_resolve_beneath(conf, c->log, reqpath, mode,
278 resolved, resolved_sz);
279 if (rc == NGX_ERROR) {
280 /*
281 * Restore the error-log diagnostic the retired realpath resolver
282 * (brix_validate_components_cstr) emitted: a rejected "."/".."
283 * traversal is recorded in the error log with control bytes escaped,
284 * so operators retain visibility into traversal attempts. Depth and
285 * join-overflow rejections fall through to the access log only.
286 */
287 if (brix_op_path_forbidden_component(reqpath)) {
288 brix_log_path_warning(c->log, "brix: path traversal attempt",
289 reqpath);
290 }
291 brix_log_access(ctx, c, op_name, reqpath, "-",
292 0, kXR_ArgInvalid, "invalid path", 0);
293 BRIX_OP_ERR(ctx, op_id);
294 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
295 "invalid path");
296 return NGX_DONE;
297 }
298 if (rc == NGX_DECLINED) {
299 brix_log_access(ctx, c, op_name, reqpath, "-",
300 0, kXR_NotFound, "no such file or directory", 0);
301 BRIX_OP_ERR(ctx, op_id);
302 ctx->write_rc = brix_send_error(ctx, c, kXR_NotFound,
303 "no such file or directory");
304 return NGX_DONE;
305 }
306
307 return NGX_OK;
308}
