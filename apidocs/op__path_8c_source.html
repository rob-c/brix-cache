1/*
2 * op_path.c — unified path extractor + resolver for namespace operations.
3 *
4 * Centralises the repeated extract_path → depth_check → resolve sequence
5 * that was previously duplicated across 10+ handler files.
6 *
7 * Phase 8: resolution no longer calls realpath(3). Confinement is enforced by
8 * the kernel at the actual filesystem operation (openat2 RESOLVE_BENEATH via the
9 * beneath API), so this layer only has to (1) reject obviously-bad paths the same
10 * way the old brix_validate_components_cstr() did — length, depth, and the
11 * forbidden "."/".." components — and (2) reproduce the per-mode existence
12 * semantics the old resolve_path* variants provided (EXISTING needs the target,
13 * WRITE needs the parent directory) through the VFS seam (brix_vfs_probe, a
14 * non-observing confined existence/type check) rather than a realpath() that
15 * would have failed. `resolved` is filled with the lexical
16 * root_canon + reqpath join (brix_beneath_full_path); it is used downstream for
17 * ACL prefix matching and access logging, NOT as a confinement boundary — the
18 * boundary is RESOLVE_BENEATH at the op. A path that escapes the export root is
19 * rejected by the kernel (EXDEV) when the operation runs.
20 */

30/*
31 * Validate each path component the way the retired realpath resolver's
32 * brix_validate_components_cstr() did: reject "." and ".." segments outright.
33 * RESOLVE_BENEATH would block a "../" that escapes the root, but a within-root
34 * "/a/../b" was historically rejected here too — keep that behaviour so the
35 * resolver's contract is unchanged. Returns 1 if any component is forbidden.
36 */
37int
38brix_op_path_forbidden_component(const char *reqpath)
39{
40 const char *p = reqpath;
41 const char *seg;
42
43 while (*p != '\0') {
44 while (*p == '/') {
45 p++;
46 }
47 if (*p == '\0') {
48 break;
49 }
50 seg = p;
51 while (*p != '\0' && *p != '/') {
52 p++;
53 }
54 if (brix_path_component_forbidden(seg, (size_t) (p - seg))) {
55 return 1;
56 }
57 }
58 return 0;
59}

61/*
62 * Existence gate replacing the realpath()-failure path of the old resolvers.
63 * want_dir < 0 : no check (NOEXIST).
64 * want_dir = 0 : the target itself must exist (EXISTING).
65 * want_dir = 1 : the target's PARENT directory must exist (WRITE).
66 * Returns NGX_OK when the requirement is met, NGX_DECLINED otherwise.
67 */
68/* Build a probe VFS ctx for `reqpath` (joined beneath the export root) and run a
69 * non-observing confined existence/type check (no phantom OP_STAT metric). */
70static ngx_int_t
71op_path_probe(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
72 const char *reqpath, int nofollow, brix_vfs_stat_t *vst)
73{
74 brix_vfs_ctx_t vctx;
75 char full[PATH_MAX];
76
77 brix_beneath_full_path(conf->common.root_canon, reqpath,
78 full, sizeof(full));
79 brix_vfs_ctx_init(&vctx, NULL /* no alloc in a probe */, log,
80 BRIX_PROTO_ROOT, conf->common.root_canon, NULL,
81 conf->common.allow_write, 0 /* is_tls */, NULL, full);
82 return brix_vfs_probe(&vctx, nofollow, vst);
83}

85static ngx_int_t
86op_path_existence_gate(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
87 const char *reqpath, int want_dir)
88{
89 brix_vfs_stat_t vst;
90
91 if (want_dir < 0) {
92 return NGX_OK; /* NOEXIST: nothing to verify */
93 }
94
95 if (want_dir == 0) {
96 /* For a NON-POSIX backend the operation's own driver call validates the
97 * target and returns ENOENT→NotFound, so this gate's probe is a redundant
98 * catalog lookup — skip it (the driver is the single existence check). The
99 * default POSIX export keeps the probe below: its confined lstat is cheap
100 * and preserves the existing existence-before-auth ordering exactly. */
101 if (brix_vfs_backend_resolve(conf->common.root_canon, log) != NULL) {
102 return NGX_OK;
103 }
104 /* EXISTING: the target name must resolve, confined, to something present.
105 * nofollow (lstat semantics) so a symlink — including a dangling one —
106 * counts as present: rm/chmod/mv operate on the name itself, and rm of a
107 * symlink must succeed (it never dereferences the final component). This
108 * gate is only the ACL/logging existence check; the confined VFS ops
109 * remain the security boundary, so not following the final link here
110 * weakens nothing. */
111 return op_path_probe(conf, log, reqpath, 1 /* nofollow */, &vst) == NGX_OK
112 ? NGX_OK : NGX_DECLINED;
113 }
114
115 /* WRITE: the parent directory must already exist (target may not). Derive
116 * the parent by trimming the last '/'-separated component. */
117 {
118 char parent[BRIX_MAX_PATH + 1];
119 size_t len = ngx_strlen(reqpath);
120 const char *slash;
121
122 while (len > 1 && reqpath[len - 1] == '/') {
123 len--; /* ignore trailing slashes for the split */
124 }
125 slash = reqpath + len;
126 while (slash > reqpath && *(slash - 1) != '/') {
127 slash--;
128 }
129 /* slash now points just past the parent's trailing '/'. */
130 {
131 size_t plen = (size_t) (slash - reqpath);
132 if (plen <= 1) {
133 /* parent is the export root itself — always present. */
134 return NGX_OK;
135 }
136 /* `slash` points just PAST the parent's trailing '/', so plen still
137 * includes it. Drop it: a confined POSIX lstat tolerates a trailing
138 * slash on a directory ("/w0/"), but a driver whose namespace lives
139 * in a catalog (pblock) stores the key as "/w0" and a "/w0/" lookup
140 * misses — which silently broke every non-recursive nested mkdir on a
141 * non-POSIX backend. The parent path handed to the probe must be
142 * slash-free. */
143 plen--;
144 if (plen >= sizeof(parent)) {
145 return NGX_DECLINED;
146 }
147 ngx_memcpy(parent, reqpath, plen);
148 parent[plen] = '\0';
149 }
150 if (op_path_probe(conf, log, parent, 0 /* follow */, &vst) != NGX_OK
151 || !vst.is_directory)
152 {
153 return NGX_DECLINED;
154 }
155 return NGX_OK;
156 }
157}

159/* ---- op_path_mode_flags — decode a path mode into gate parameters ----------
160 *
161 * WHAT: Map a brix_path_mode_t to the two scalars the resolver drives its
162 * existence check with: want_dir (see op_path_existence_gate) and
163 * strip_trailing_slash (whether a create target's trailing '/' is
164 * normalized away). Fills both out-params; no return value.
165 *
166 * WHY: Keeps the mode→behavior table as one small, side-effect-free lookup so
167 * the resolver stays a flat sequence and the per-mode policy lives in a
168 * single place instead of inline in a larger function.
169 *
170 * HOW: 1. EXISTING gates on the target itself, keeps trailing slash.
171 * 2. WRITE gates on the parent dir and strips a trailing slash.
172 * 3. NOEXIST skips the existence gate (want_dir < 0), keeps slash.
173 * 4. EITHER gates on the parent (with a target-retry upstream), keeps slash.
174 * 5. Any other value falls back to EXISTING semantics.
175 */
176static void
177op_path_mode_flags(brix_path_mode_t mode, int *want_dir,
178 int *strip_trailing_slash)
179{
180 switch (mode) {
181 case BRIX_PATH_EXISTING: *want_dir = 0; *strip_trailing_slash = 0; break;
182 case BRIX_PATH_WRITE: *want_dir = 1; *strip_trailing_slash = 1; break;
183 case BRIX_PATH_NOEXIST: *want_dir = -1; *strip_trailing_slash = 0; break;
184 case BRIX_PATH_EITHER: *want_dir = 1; *strip_trailing_slash = 0; break;
185 default: *want_dir = 0; *strip_trailing_slash = 0; break;
186 }
187}

189/* ---- op_path_normalize_trailing — collapse a create target's trailing '/' ---
190 *
191 * WHAT: For create modes only, rewrite "*reqpath" into the caller-provided
192 * "norm" buffer with any trailing '/' characters removed, and repoint
193 * "*reqpath" at "norm". Returns NGX_ERROR if the copy would overflow
194 * "norm", NGX_OK otherwise (including the common no-op cases).
195 *
196 * WHY: Reproduces the reference server's Squash of a trailing slash on a
197 * create target ("mkdir /d/" -> /d, "open /f/" -> /f) so stock-accepted
198 * requests are not rejected as ArgInvalid. Scoped to the create modes so a
199 * stat/cat of "/file/" keeps its existing file-vs-dir error behavior.
200 *
201 * HOW: 1. No-op when strip_trailing_slash is unset.
202 * 2. No-op when the path is "/" or does not end in '/'.
203 * 3. Reject when the path (plus its NUL) will not fit in norm_sz.
204 * 4. Copy into norm, drop trailing '/' down to a length of 1, and
205 * repoint *reqpath at the normalized buffer.
206 */
207static ngx_int_t
208op_path_normalize_trailing(int strip_trailing_slash, const char **reqpath,
209 char *norm, size_t norm_sz)
210{
211 size_t rl;
212
213 if (!strip_trailing_slash) {
214 return NGX_OK;
215 }
216
217 rl = ngx_strlen(*reqpath);
218 if (rl <= 1 || (*reqpath)[rl - 1] != '/') {
219 return NGX_OK;
220 }
221
222 if (rl >= norm_sz) {
223 return NGX_ERROR;
224 }
225
226 ngx_memcpy(norm, *reqpath, rl + 1);
227 while (rl > 1 && norm[rl - 1] == '/') {
228 norm[--rl] = '\0';
229 }
230 *reqpath = norm;
231 return NGX_OK;
232}

234/* ---- op_path_gate — run the per-mode existence gate with the EITHER retry ----
235 *
236 * WHAT: Apply op_path_existence_gate() with the mode's want_dir, and for
237 * BRIX_PATH_EITHER retry as a target-existence check when the first pass
238 * fails. Returns NGX_OK when the requirement is met, NGX_DECLINED when it
239 * is not (target/parent missing).
240 *
241 * WHY: EITHER means "target may already exist OR its parent must exist" (used
242 * where an op accepts both a present target and a create). Folding the
243 * fallback here keeps the resolver's control flow linear and single-purpose.
244 *
245 * HOW: 1. Run the gate with want_dir; return NGX_OK on success.
246 * 2. For EITHER, run the gate again with want_dir 0 (target must exist);
247 * return NGX_OK on success.
248 * 3. Otherwise return NGX_DECLINED.
249 */
250static ngx_int_t
251op_path_gate(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
252 const char *reqpath, brix_path_mode_t mode, int want_dir)
253{
254 if (op_path_existence_gate(conf, log, reqpath, want_dir) == NGX_OK) {
255 return NGX_OK;
256 }
257 if (mode == BRIX_PATH_EITHER
258 && op_path_existence_gate(conf, log, reqpath, 0) == NGX_OK)
259 {
260 return NGX_OK;
261 }
262 return NGX_DECLINED;
263}

265/*
266 * brix_path_resolve_beneath — validate reqpath and apply the per-mode
267 * existence gate without realpath(), filling `resolved` with the confined
268 * lexical join. Shared by brix_resolve_op_path() and direct multi-path
269 * callers (kXR_mv). reqpath must already be extracted from the wire.
270 *
271 * Returns:
272 * NGX_OK — valid and the existence requirement is met; resolved filled.
273 * NGX_DECLINED — path is well-formed but the existence gate failed
274 * (EXISTING: target missing; WRITE: parent dir missing) → 404.
275 * NGX_ERROR — malformed path (depth, "."/"..", WRITE trailing slash, or
276 * the join overflowed resolved_sz) → 4xx ArgInvalid.
277 */
278ngx_int_t
279brix_path_resolve_beneath(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
280 const char *reqpath, brix_path_mode_t mode,
281 char *resolved, size_t resolved_sz)
282{
283 int want_dir;
284 int strip_trailing_slash;
285 char norm[BRIX_MAX_PATH + 1];
286
287 if (brix_count_path_depth(reqpath) != NGX_OK
288 || brix_op_path_forbidden_component(reqpath))
289 {
290 return NGX_ERROR;
291 }
292
293 op_path_mode_flags(mode, &want_dir, &strip_trailing_slash);
294
295 /* Normalize a create target's trailing slash (see op_path_normalize_trailing).
296 * A copy that would overflow `norm` is the only failure and maps to ArgInvalid. */
297 if (op_path_normalize_trailing(strip_trailing_slash, &reqpath,
298 norm, sizeof(norm)) != NGX_OK)
299 {
300 return NGX_ERROR;
301 }
302
303 if (op_path_gate(conf, log, reqpath, mode, want_dir) != NGX_OK) {
304 return NGX_DECLINED;
305 }
306
307 if (brix_beneath_full_path(conf->common.root_canon, reqpath,
308 resolved, resolved_sz) >= (int) resolved_sz)
309 {
310 return NGX_ERROR;
311 }
312
313 return NGX_OK;
314}

332ngx_int_t
333brix_resolve_op_path(brix_ctx_t *ctx, ngx_connection_t *c,
334 ngx_uint_t op_id, const char *op_name,
335 ngx_stream_brix_srv_conf_t *conf,
336 brix_path_mode_t mode,
337 char *reqpath, size_t reqpath_sz,
338 char *resolved, size_t resolved_sz)
339{
340 ngx_int_t rc;
341
342 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
343 brix_log_access(ctx, c, op_name, "-", "-",
344 0, kXR_ArgMissing, "no path given", 0);
345 BRIX_OP_ERR(ctx, op_id);
346 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgMissing,
347 "no path given");
348 return NGX_DONE;
349 }
350
351 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
352 reqpath, reqpath_sz, 1)) {
353 brix_log_access(ctx, c, op_name, "-", "-",
354 0, kXR_ArgInvalid, "invalid path payload", 0);
355 BRIX_OP_ERR(ctx, op_id);
356 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
357 "invalid path payload");
358 return NGX_DONE;
359 }
360
361 rc = brix_path_resolve_beneath(conf, c->log, reqpath, mode,
362 resolved, resolved_sz);
363 if (rc == NGX_ERROR) {
364 /*
365 * Restore the error-log diagnostic the retired realpath resolver
366 * (brix_validate_components_cstr) emitted: a rejected "."/".."
367 * traversal is recorded in the error log with control bytes escaped,
368 * so operators retain visibility into traversal attempts. Depth and
369 * join-overflow rejections fall through to the access log only.
370 */
371 if (brix_op_path_forbidden_component(reqpath)) {
372 brix_log_path_warning(c->log, "brix: path traversal attempt",
373 reqpath);
374 }
375 brix_log_access(ctx, c, op_name, reqpath, "-",
376 0, kXR_ArgInvalid, "invalid path", 0);
377 BRIX_OP_ERR(ctx, op_id);
378 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
379 "invalid path");
380 return NGX_DONE;
381 }
382 if (rc == NGX_DECLINED) {
383 brix_log_access(ctx, c, op_name, reqpath, "-",
384 0, kXR_NotFound, "no such file or directory", 0);
385 BRIX_OP_ERR(ctx, op_id);
386 ctx->write_rc = brix_send_error(ctx, c, kXR_NotFound,
387 "no such file or directory");
388 return NGX_DONE;
389 }
390
391 return NGX_OK;
392}

394/* ---- brix_root_vfs_bind_deleg ----------------------------------------------
395 *
396 * WHAT: Bind the session's captured raw bearer JWT onto a cred-bound VFS ctx for
397 * backend PASSTHROUGH. See op_path.h for the full contract.
398 *
399 * WHY: Lets a remote-backed root:// export authenticate the backend leg AS the
400 * inbound user. Two forwardable credentials ride the GSI login: the raw
401 * bearer JWT (ctx->bearer_token) and — when the client opts in and the DN
402 * is proven (phase-70 §5.1, gsi_promote_fullproxy) — a full x509 proxy
403 * (ctx->deleg_proxy_pem, chain + private key) for backend PASSTHROUGH.
404 *
405 * HOW: No-op on the default SELECT export. Otherwise wraps whichever
406 * credential(s) the session captured as ngx_str_t (bytes owned by the
407 * session ctx, outliving the op) and hands them to brix_vfs_deleg_bind on
408 * the VFS ctx's pool. The full proxy is forwarded only in PASSTHROUGH
409 * mode — the only strategy that replays the user's own credential
410 * verbatim. If neither credential is present, nothing is bound. */
411void
412brix_root_vfs_bind_deleg(brix_ctx_t *ctx,
413 ngx_stream_brix_srv_conf_t *conf,
414 brix_vfs_ctx_t *vctx)
415{
416 ngx_str_t bearer;
417 ngx_str_t *bearer_arg = NULL;
418 ngx_str_t *proxy_arg = NULL;
419
420 if (ctx == NULL || conf == NULL || vctx == NULL
421 || conf->common.backend_delegation == BRIX_CRED_SELECT)
422 {
423 return;
424 }
425
426 bearer.data = (u_char *) ctx->bearer_token;
427 bearer.len = ngx_strlen(ctx->bearer_token);
428 if (bearer.len > 0) {
429 bearer_arg = &bearer;
430 }
431
432 /* A full proxy is a PASSTHROUGH-only credential: it is presented to the
433 * upstream unmodified, so it makes sense only when the resolved mode replays
434 * the user's own credential. */
435 if (ctx->deleg_proxy_pem.len > 0
436 && (enum brix_cred_mode) conf->common.backend_delegation
437 == BRIX_CRED_PASSTHROUGH)
438 {
439 proxy_arg = &ctx->deleg_proxy_pem;
440 }
441
442 if (bearer_arg == NULL && proxy_arg == NULL) {
443 return;
444 }
445
446 (void) brix_vfs_deleg_bind(vctx->pool, vctx,
447 (enum brix_cred_mode) conf->common.backend_delegation,
448 bearer_arg, proxy_arg);
449}
