7/*
8 * open_tpc.c — kXR_open TPC (third-party-copy) context detection, split from
9 * open_request.c. The XRootD TPC protocol embeds transfer-context parameters in
10 * the open path as CGI-style opaque strings:
11 *
12 * TPC destination (we pull FROM source): write open + tpc.src=root://host//path
13 * + tpc.key=<token>. Connect outbound to the source, stream the file
14 * locally, and return the fhandle only after the pull completes.
15 *
16 * TPC source (destination connects TO us): read open with tpc.key=<token>
17 * (+ optional tpc.dst=/tpc.org= for the two-step rendezvous). Register the
18 * first form and consume the second before serving bytes.
19 *
20 * Must act BEFORE the normal path-resolution/open logic, so it runs first from
21 * brix_handle_open. Bodies are moved verbatim; the early-returns are unchanged.
22 */
23

145/*
146 * TPC source role. XrdCl drives full native TPC as a two-step source
147 * rendezvous: first the initiating client opens the source with tpc.dst and the
148 * shared key, then the destination server opens the same source with tpc.org and
149 * that key. Register the first form and consume the second before serving
150 * bytes. NGX_DECLINED to serve the file normally; an error rc on a bad consume.
151 */
152static ngx_int_t
153tpc_handle_source(brix_ctx_t *ctx, ngx_connection_t *c,
154 ngx_stream_brix_srv_conf_t *conf, const brix_tpc_params_t *tpc)
155{
156 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, c->log, 0,
157 "brix: TPC source open key=%s dst=%s org=%s",
158 tpc->has_key ? tpc->key : "-",
159 tpc->has_dst ? tpc->dst : "-",
160 tpc->has_org ? tpc->org : "-");
161
162 if (tpc->has_key && tpc->key[0] != '\0' && tpc->has_dst
163 && !tpc->has_org) {
164 brix_tpc_key_register(tpc->key, conf->tpc_key_ttl_ms);
165 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
166 "brix: TPC source key=%s registered",
167 tpc->key);
168
169 } else if (tpc->has_key && tpc->key[0] != '\0'
170 && tpc->has_org) {
171 if (brix_tpc_key_consume(tpc->key)) {
172 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
173 "brix: TPC source key=%s consumed",
174 tpc->key);
175 } else {
176 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
177 ctx->recv.payload ? (char *) ctx->recv.payload : "-",
178 "tpc-source", kXR_NotAuthorized,
179 "TPC authorization missing or expired");
180 }
181 }
182
183 return NGX_DECLINED;
184}
