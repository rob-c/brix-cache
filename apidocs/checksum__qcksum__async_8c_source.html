1/*
2 * WHAT: Async kXR_Qcksum — compute a single-file checksum (adler32/crc32c/md5/sha1/sha256) off the event loop via nginx thread pool.
3 * The thread worker opens a confined fd, computes checksum via brix_checksum_hex_fd, writes "algo hex" result to t->resp.
4 * On failure sets error_code + error_msg. Never touches nginx event-loop state (no ctx, no c, no ngx_pool).
5 *
6 * WHY: Qcksum file checksums can be expensive for large files. Running on the event loop blocks all other connections while reading.
7 * The async path uses ngx_thread_pool_run to execute the open+checksum in a background worker thread, keeping nginx responsive.
8 * Results delivered via aio_done callback which restores request streamid, sends response to client, and resumes connection.
9 *
10 * HOW: brix_cksum_aio_thread() parses algo from t->algo string, computes checksum via hex_fd on pre-opened fd,
11 * formats result as "algo hex" into resp buffer. On parse failure sets kXR_ArgInvalid; on compute failure sets kXR_IOError.
12 * brix_cksum_aio_done() closes fd if path-based request (t->close_fd), restores streamid via aio_restore_request, sends ok+resp or error, frees resources, resumes client connection event loop via aio_resume().
13 */

32void
33brix_cksum_aio_thread(void *data, ngx_log_t *log)
34{

36 brix_integrity_info_t info;

38 char token[256];
39
40 ngx_memzero(&iopts, sizeof(iopts));
41 iopts.allow_xattr_cache = 1;
42 iopts.update_xattr_cache = 1;
43
44 if (brix_integrity_get_fd(log, t->fd, &t->obj, t->resolved, t->algo,
45 &iopts, &info) != NGX_OK)
46 {
47 t->error_code = kXR_IOError;
48 snprintf(t->error_msg, sizeof(t->error_msg),

54 snprintf(token, sizeof(token), "%s %s", info.alg_name, info.hex);

66/* public API: brix_cksum_aio_done() — async qcksum completion callback * WHAT: Main-thread event handler invoked when the thread worker completes. Closes t->fd if path-based request (t->close_fd), restores request streamid via aio_restore_request, sends error response or ok+checksum data to client, resumes client connection event loop via brix_aio_resume(). */

68void
69brix_cksum_aio_done(ngx_event_t *ev)
70{
71 ngx_thread_task_t *task = ev->data;
72 brix_cksum_aio_t *t = task->ctx;
73 brix_ctx_t *ctx = t->ctx;
74 ngx_connection_t *c = t->c;
75
76 if (t->close_fd) {
77 /* Path-based request: release the VFS handle we opened (it owns the fd).
78 * The handle is freed before the destroy check so it is never leaked even
79 * when the client disconnected while the AIO was in flight. */
80 if (t->fh != NULL) {
81 brix_vfs_close(t->fh, c->log);
82 t->fh = NULL;
83 } else if (t->fd >= 0) {
84 close(t->fd);
85 }
86 t->fd = -1;
87 }
88
89 if (!brix_aio_restore_request(ctx, t->streamid)) {
90 return;
91 }
92
93 if (t->error_code != 0) {
94 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSUM);
95 brix_send_error(ctx, c, (uint16_t) t->error_code, t->error_msg);
96 } else {
97 BRIX_OP_OK(ctx, BRIX_OP_QUERY_CKSUM);
98 brix_log_access(ctx, c, "QUERY", t->resolved, "cksum", 1, 0, NULL, 0);
99 brix_send_ok(ctx, c, t->resp, (uint32_t) (strlen(t->resp) + 1));
100 }
101
102 brix_aio_resume(c);
103}
