4/*
5 * stage_engine.h - the one async-staging engine (phase-64 P4/P5/P6, section 11).
6 *
7 * WHAT: A single durable-transfer front door, brix_stage_submit(), shared by
8 * every async data-movement path in the storage stack: a cache miss-fill of
9 * a nearline (tape) backend (RECALL), a write-stage flush / tape migrate
10 * (FLUSH), a client-body upload into the stage store (UPLOAD), and an S3
11 * multipart assembly (MULTIPART). One queue, one waiter, one byte mover, one
12 * audit ledger - for all four kinds.
13 *
14 * WHY: Before phase-64 the same "move bytes from A to B durably, possibly async,
15 * and wake a parked open when done" logic was re-implemented five times
16 * (the POSC upload stage, vfs_scratch, S3 multipart, writethrough flush, the
17 * FRM tape queue). P5 folds them into ONE engine; the src/dst become SD
18 * instances (any driver) instead of FRM's local-path/stagecmd model, so the
19 * same engine moves bytes between any two tiers. The sd_stage / sd_cache
20 * decorators and the (future) frm nearline driver are its only callers.
21 *
22 * HOW: brix_stage_submit(kind, src, src_key, dst, dst_key, opts) runs the
23 * generic promote loop - open `src` for read, staged_open `dst`, pread ->
24 * staged_write -> staged_commit - and books one brix_xfer_finish() audit
25 * line on the terminal state. A synchronous submit runs the mover inline and
26 * returns ""; an async submit parks the open on a durable request id.
27 *
28 * SP1 lands this seam with the inline mover fully working; the durable queue
29 * + waiter + restart-reconcile are EXTRACTED from src/frm/ in SP4 (section
30 * 13b) and attach behind this same surface without touching a caller. Until
31 * then an async submit degrades to an inline move (honest: no durability is
32 * claimed that is not yet implemented), and the scheduler/reconcile entry
33 * points are no-ops. See docs/refactor/phase-64-fully-tiered-composable-
34 * storage.md (section 11, Appendix H/I).
35 */

41/*
42 * Per-user owner identity embedded in the durable request record.
43 *
44 * WHAT: A wire-stable POD (no bitfields, no pointers, no padding hazards) that
45 * captures the credential key, canonical principal, credential directory,
46 * and deny-on-missing flag at submit time so a detached / async write-back
47 * flush — possibly after a crash and restart — can re-resolve and
48 * authenticate to the origin as the ORIGINAL user, rather than falling back
49 * silently to the service credential.
50 *
51 * WHY: The flush thread runs outside the request context. Without a persisted
52 * identity the engine has no choice but to flush as the service user, which
53 * can break per-user quota / audit / access-control on the backend.
54 * Persisting key + dir lets brix_sd_ucred_resolve() recheck expiry at flush
55 * time; setting deny=1 causes a hard EACCES / BRIX_XFER_DENIED when the
56 * per-user credential is absent or expired rather than silently promoting as
57 * the service identity.
58 *
59 * HOW: This struct is memcpy'd verbatim into brix_sreq_t (the on-disk journal
60 * record) as the FINAL member. NO bitfields, NO pointers, NO embedded
61 * structs with alignment holes — the layout must be byte-stable across
62 * builds. Field sizes match BRIX_UCRED_{KEY,PRINC,PATH}_MAX constants from
63 * ucred.h (128 / 512 / 1024) so there is no truncation risk at assignment.
64 */
65typedef struct {
66 char key[128]; /* filesystem-safe credential filename stem (ucred_key) */
67 char principal[512]; /* canonical principal string (DN or JWT sub) */
68 char dir[1024]; /* credential directory path (brix_sd_ucred_dir) */
69 uint8_t deny; /* 1 = EACCES on missing/expired cred; 0 = service-cred */
70} brix_stage_cred_t;

154/*
155 * Run a transfer INLINE with an explicit per-user credential.
156 *
157 * WHAT: Runs the mover inline (same as brix_stage_run_inline) but threads the
158 * optional `cred` through to the destination staged_open so the backend
159 * driver presents the per-user x509 proxy rather than the service credential.
160 *
161 * WHY: The staged_commit and reflush paths both need to carry the owner identity
162 * down to the byte mover; a separate entry point lets callers that already
163 * hold a resolved credential avoid a second key lookup while keeping the
164 * zero-cred path (brix_stage_run_inline) unchanged for callers that don't.
165 *
166 * HOW: Validates arguments, then calls stage_engine_run with the cred pointer.
167 * Returns NGX_OK on success, NGX_ERROR on argument error or transfer failure
168 * (errno set). A NULL cred is the service-credential path.

188/*
189 * Decode one durable request record from a raw buffer with size-tolerance.
190 *
191 * WHAT: Returns NGX_OK with *out filled when n == sizeof(brix_sreq_t) (current
192 * full record) or n == offsetof(brix_sreq_t, cred) (legacy pre-cred record,
193 * which yields a zeroed cred = service-credential flush). Returns NGX_ERROR
194 * for any other n (corrupt or from an incompatible future version).
195 *
196 * WHY: brix_sreq_t grew a trailing brix_stage_cred_t; journals written before
197 * the upgrade must replay without data loss. The zeroed-cred path preserves
198 * the pre-feature semantics: flush under the service identity.
199 *
200 * HOW: ngx_memzero + ngx_memcpy(out, buf, n) — the memzero ensures the cred
201 * fields are zero when a legacy-size record is decoded.
202 */

209/*
210 * Dead-letter cap constants for deny-mode flush retry bounding (Task 4).
211 *
212 * WHAT: After BRIX_STAGE_DENY_MAX_ATTEMPTS denied flushes, or when the record
213 * has been in the active journal for more than BRIX_STAGE_DENY_MAX_AGE_SEC
214 * seconds and a flush is still denied, the record is moved to the
215 * deadletter/ subdirectory and the scheduler / reconcile stop re-driving it.
216 *
217 * WHY: A BRIX_XFER_DENIED result means the per-user credential is permanently
218 * missing or expired in deny mode. Without a cap the scheduler tick and
219 * restart-reconcile retry the same record forever. Dead-lettering stops
220 * the loop while preserving the stage copy for operator recovery.
221 */
222#define BRIX_STAGE_DENY_MAX_ATTEMPTS 5
223#define BRIX_STAGE_DENY_MAX_AGE_SEC (24 * 3600)

225/*
226 * Evaluate the dead-letter cap for a permanently denied flush.
227 *
228 * WHAT: Increments rec->attempts, re-persists the updated record to the active
229 * journal slot (via journal_dir/<reqid>.req), then checks whether the
230 * attempt cap or age cap has been reached. On cap, moves the active
231 * record to <journal_dir>/deadletter/<reqid>.req, emits NGX_LOG_ERR, and
232 * returns 1 (stop re-driving). Returns 0 when below both caps.
233 *
234 * WHY: Shared by stage_complete (BRIX_XFER_DENIED terminal from the scheduler/
235 * thread path) and stage_reconcile_one (EACCES + deny cred from reflush),
236 * so the cap is enforced consistently, including across restarts.
237 *
238 * HOW: journal_dir is passed explicitly (rather than using the module-static
239 * stage_journal_dir) so the function is directly unit-testable. rec is
240 * modified in-place (attempts incremented). */
