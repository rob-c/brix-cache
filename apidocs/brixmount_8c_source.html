1/*
2 * brixmount.c — brixMount: one hardened front-end that mounts any supported
3 * backend by type.
4 *
5 * WHAT: `brixMount <type> <endpoint> <mountdir> [fuse-opts]` dispatches to the
6 * right FUSE driver:
7 * brixMount cvmfs atlas.cern.ch ~/myAtlasMountDir/ → CVMFS-brix
8 * brixMount eos root://eoslhcb.cern.ch ~/myEOSMountDir/ → XRootDFS-brix
9 * WHY: a single tool + one brand ("a hardened, iron-clad FUSE driver, battle-
10 * tested against bad/evil networks") over pluggable drivers that all share
11 * the same resilience core.
12 * HOW: a tiny table maps a type keyword to a driver entry function; dispatch
13 * rewrites argv into the driver's own `[<driver> <endpoint> <mount> …]`
14 * form and calls it. The dispatch core is pure and unit-tested with mock
15 * drivers; main() wires the real ones (xrootdfs is weak-linked so a
16 * cvmfs-only build still links).
17 */

52/* Route the overlay maintenance subcommands to the injected cores (injection
53 * keeps this dispatch TU link-free of the overlay library, and mock-testable).
54 * Returns -1 when argv[1] is not an overlay subcommand (fall through to mount
55 * dispatch), else the subcommand's exit code (2 = missing mountdir). */
56int brixmount_overlay_route(int argc, char **argv,
57 int (*list_fn)(const char *, FILE *),
58 int (*reset_fn)(const char *)) {
59 if (argc < 2) return -1;
60 int is_list = strcmp(argv[1], "--overlay-list") == 0;
61 int is_reset = strcmp(argv[1], "--overlay-reset") == 0;
62 if (!is_list && !is_reset) return -1;
63 if (argc < 3) {
64 fprintf(stderr, "usage: brixMount %s <mountdir>\n", argv[1]);
65 return 2;
66 }
67 return is_list ? list_fn(argv[2], stdout) : reset_fn(argv[2]);
68}

70/* Pure dispatch: select a driver by `argv[1]` type and invoke it with
71 * `[<brand> <endpoint> <mountdir> <fuse-opts…>]`. Returns the driver's exit code,
72 * or 2 on usage error. Exposed (non-static) for unit testing. */
73int brixmount_dispatch(int argc, char **argv, const brix_driver_t *drv, size_t ndrv) {
74 if (argc < 4) {
75 brixmount_usage(stderr, drv, ndrv);
76 return 2;
77 }
78 const char *type = argv[1];
79 const char *endpoint = argv[2];
80 const char *mountdir = argv[3];
81
82 const brix_driver_t *sel = NULL;
83 for (size_t i = 0; i < ndrv; i++)
84 if (strcmp(drv[i].type, type) == 0) { sel = &drv[i]; break; }
85
86 if (sel == NULL) {
87 /*
88 * WHAT: build a NULL-terminated name array from drv[] and emit a
89 * did-you-mean hint when brix_suggest() finds a close match.
90 * WHY: spec WS-7: every unknown-type/subcommand site must suggest a
91 * close match for interactive users (TTY-gated, C3 compliant).
92 * HOW: iterate drv[] up to ndrv entries, cap at 16 for safety; pass
93 * to brix_suggest() and emit via brix_cli_hint() if non-NULL.
94 */
95 const char *type_names[17]; /* at most 16 driver types + NULL */
96 size_t nt = 0;
97 size_t di;
98 const char *suggestion;
99
100 for (di = 0; di < ndrv && nt < 16; di++) {
101 type_names[nt++] = drv[di].type;
102 }
103 type_names[nt] = NULL;
104
105 fprintf(stderr, "brixMount: unknown type '%s'\n", type);
106 suggestion = brix_suggest(type, type_names);
107 if (suggestion != NULL) {
108 brix_cli_hint("hint: did you mean '%s'?\n", suggestion);
109 }
110 brixmount_usage(stderr, drv, ndrv);
111 return 2;
112 }
113 if (sel->fn == NULL) {
114 fprintf(stderr, "brixMount: type '%s' (%s) is not available in this build\n",
115 type, sel->brand);
116 return 2;
117 }
118
119 char *sub[BRIXMOUNT_MAX_ARGS];
120 int sc = 0;
121 sub[sc++] = (char *) sel->brand;
122 sub[sc++] = (char *) endpoint;
123 sub[sc++] = (char *) mountdir;
124 for (int i = 4; i < argc && sc < BRIXMOUNT_MAX_ARGS - 1; i++)
125 sub[sc++] = argv[i];
126 sub[sc] = NULL;
127
128 return sel->fn(sc, sub);
129}
