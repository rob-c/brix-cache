1/*
2 * classify_handler.c — ACCESS-phase pre-backend bounce.
3 *
4 * WHAT: runs guard_classify_pre on every guarded request before proxy_pass;
5 * bounces signature/grammar hits with the configured status so junk never
6 * reaches the ARC / XrdHttp backend.
7 * WHY: the ACCESS phase is the last hook before the content phase hands the
8 * request to the upstream — the one place a pre-backend verdict can act.
9 * HOW: build guard_request_t (guard_http_req.c), classify, and on BOUNCE
10 * record the reason in the request ctx (the LOG handler skips those),
11 * write the audit line immediately, and return 403/444.
12 */

16/* ---- ACCESS-phase handler: bounce junk pre-backend ----
17 *
18 * WHAT: returns NGX_DECLINED for clean/unguarded/internal requests
19 * (proxy_pass runs), or the configured bounce status (403/444) after
20 * auditing a signature or grammar hit.
21 *
22 * WHY: keeps scanner noise off the backend and gives fail2ban its
23 * pre-backend signals. 444 maps to nginx's close-without-response —
24 * ideal for scanners; 403 sends a normal error page.
25 *
26 * HOW: 1. Skip when the location has no guard or the request is internal.
27 * 2. Ensure the per-request ctx exists (LOG phase reads it).
28 * 3. Normalize into guard_request_t (stack scratch) + classify_pre.
29 * 4. On BOUNCE: record the reason, audit immediately (guaranteed even
30 * if the request never reaches the LOG phase), return the bounce
31 * status.
32 */

43 lcf = ngx_http_get_module_loc_conf(r, ngx_http_brix_guard_module);

49 if (ctx == NULL) {

51 if (ctx == NULL) {
52 return NGX_ERROR;
53 }
54 ngx_http_set_ctx(r, ctx, ngx_http_brix_guard_module);
55 }
56
57 ngx_http_brix_guard_build_request(r, lcf, &req,
58 pathbuf, sizeof(pathbuf), ipbuf, sizeof(ipbuf));
59
60 if (guard_classify_pre(&lcf->ruleset, &req, &why) == GUARD_BOUNCE) {
61 ctx->pre_reason = why;
62 ctx->bounced = 1;
63 req.status_code = (int) lcf->bounce_status;
64 ngx_http_brix_guard_write_audit(r, lcf, &req, why);
65 return lcf->bounce_status; /* 403 or 444 (NGX_HTTP_CLOSE) */
66 }
67
68 return NGX_DECLINED; /* clean -> proxy_pass runs */
69}
