147/*
148 * WHAT: Cache the network-fault deadlines and allocate the pipeline rings.
149 * WHY : Caching the merged deadlines keeps the hot recv/park paths off the
150 * srv_conf lookup; the rings size the in-flight window that absorbs wire
151 * jitter without stalling the recv->send loop.
152 * HOW : Copy the three deadlines, then pcalloc the out/read rings. pipeline_depth
153 * is set LAST so it stays 0 (teardown loops no-op) until both rings exist.
154 * Returns NGX_ERROR after finalizing on allocation failure, else NGX_OK.
155 */
156static ngx_int_t
157conn_setup_pipeline(ngx_stream_session_t *s, ngx_connection_t *c,
158 brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *mconf)
159{
160 ngx_uint_t depth = mconf->pipeline_depth;
161
162 /* Phase 39: cache the merged network-fault deadlines so the hot
163 * recv/park paths never do a srv_conf lookup. All default 0 = off. */
164 ctx->deadline.read_ms = mconf->read_timeout;
165 ctx->deadline.handshake_ms = mconf->handshake_timeout;
166 ctx->deadline.send_ms = mconf->send_timeout;
167
168 ctx->out.ring = ngx_pcalloc(c->pool, depth * sizeof(brix_resp_slot_t));
169 ctx->rd.pool = ngx_pcalloc(c->pool, depth * sizeof(brix_read_slot_t));
170 if (ctx->out.ring == NULL || ctx->rd.pool == NULL) {
171 ngx_stream_finalize_session(s, NGX_STREAM_INTERNAL_SERVER_ERROR);
172 return NGX_ERROR;
173 }
174 ctx->out.pipeline_depth = depth;
175 return NGX_OK;
176}

299/*
300 * WHAT: Bind the metrics row, publish cache labels, and apply the admission cap.
301 * WHY : Metrics must be attached and the pre-identity connection cap enforced
302 * BEFORE active++ so a refused connection never perturbs the gauge.
303 * HOW : Resolve the row (no-op if metrics disabled), write cache labels + one-time
304 * server labels, then enforce brix_max_connections; on refusal finalize the
305 * session and return NGX_DONE, else bump the counters and return NGX_OK.
306 */
307static ngx_int_t
308conn_register_metrics(ngx_stream_session_t *s, ngx_connection_t *c,
309 brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *mconf)
310{
311 ngx_brix_srv_metrics_t *srv = conn_metrics_row(mconf);
312
313 if (srv == NULL) {
314 return NGX_OK;
315 }
316
317 ctx->metrics = srv;
318
319 srv->cache_enabled = mconf->cache ? 1 : 0;
320 srv->cache_eviction_threshold = mconf->cache_eviction_threshold;
321 if (mconf->cache && mconf->cache_root.len < sizeof(srv->cache_root)) {
322 ngx_memcpy(srv->cache_root, mconf->cache_root.data,
323 mconf->cache_root.len);
324 srv->cache_root[mconf->cache_root.len] = '\0';
325 } else {
326 srv->cache_root[0] = '\0';
327 }
328
329 conn_metrics_init_labels(c, mconf, srv);
330
331 /*
332 * Phase 39 (WS9): pre-identity admission cap. Once the listener's
333 * active-connection gauge is at brix_max_connections, refuse with a
334 * plain TCP close — there is no streamid pre-login for a framed
335 * kXR_wait, and combined with brix_handshake_timeout this bounds a
336 * half-open / reconnect-storm flood. Checked BEFORE the active++ so
337 * a refused connection never perturbs the gauge. 0 = unlimited.
338 */
339 if (mconf->max_connections > 0
340 && (ngx_uint_t) ngx_atomic_fetch_add(&srv->connections_active, 0)
341 >= mconf->max_connections)
342 {
343 ngx_atomic_fetch_add(&srv->connections_rejected_total, 1);
344 ngx_log_error(NGX_LOG_INFO, c->log, 0,
345 "brix: connection refused — listener at "
346 "brix_max_connections (%ui)",
347 mconf->max_connections);
348 ngx_stream_finalize_session(s, NGX_STREAM_OK);
349 return NGX_DONE;
350 }
351
352 ngx_atomic_fetch_add(&srv->connections_total, 1);
353 ngx_atomic_fetch_add(&srv->connections_active, 1);
354 return NGX_OK;
355}

357/*
358 * WHAT: Apply all per-server configuration to the connection.
359 * WHY : Groups the srv_conf-derived setup (deadlines/rings, dead-peer sockopts,
360 * congestion control, metrics + admission) behind one flat sequence.
361 * HOW : Look up the srv_conf once, run the pipeline setup and TCP sockopts, then
362 * register metrics. Returns NGX_ERROR/NGX_DONE (session already finalized)
363 * if any step demands the caller return, else NGX_OK.
364 */
365static ngx_int_t
366conn_apply_srv_conf(ngx_stream_session_t *s, ngx_connection_t *c,
367 brix_ctx_t *ctx)
368{
369 ngx_stream_brix_srv_conf_t *mconf =
370 ngx_stream_get_module_srv_conf(s, ngx_stream_brix_module);
371 ngx_int_t rc;
372
373 rc = conn_setup_pipeline(s, c, ctx, mconf);
374 if (rc != NGX_OK) {
375 return rc;
376 }
377
378 /*
379 * Phase 39 (WS3): OS-level dead-peer reaping, applied once at accept on
380 * the control path. Both default off (leave the kernel defaults), so a
381 * stock deployment is byte-for-byte unchanged. setsockopt failures are
382 * deliberately non-fatal — a missing option must never abort a connection.
383 */
384 brix_apply_tcp_deadpeer_opts(c->fd, mconf->tcp_keepalive,
385 mconf->tcp_user_timeout);
386
387 /* Per-socket congestion control (e.g. "bbr"); empty = kernel default.
388 * Best-effort — a missing algorithm leaves the default, never aborts. */
389 brix_apply_tcp_congestion(c->fd, mconf->tcp_congestion);
390
391 return conn_register_metrics(s, c, ctx, mconf);
392}

467 * WHY : The front door: build the connection context, apply per-server config,
468 * branch to the transparent relay if configured, then start the protocol.
469 * HOW : Flat early-return sequence of focused helpers — any helper that has
