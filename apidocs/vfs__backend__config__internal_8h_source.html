27/*
28 * vfs_origin_parse_t — the shared output record for the http:// and s3://
29 * origin-segment parsers.
30 *
31 * WHAT: One caller-owned bundle carrying an origin segment's parse result —
32 * the host and base/bucket destination buffers (with their capacities),
33 * the resolved port and TLS flag, plus a `host_len` scratch slot the
34 * host/port splitter writes and the scheme stripper's `rest`/`rest_len`
35 * intermediates.
36 * WHY: The parse pipeline (strip scheme → split host:port → copy out) threaded
37 * the same host/port/tls/base four-tuple through every stage as separate
38 * by-value + out-pointer arguments, pushing each helper past the 5-param
39 * readability gate. Passing one pointer keeps the data flow explicit while
40 * collapsing the argument lists; behaviour is byte-for-byte identical
41 * (same buffers, same caps, same fields written).
42 * HOW: The caller stack-allocates and zero-inits one of these, sets `host`/
43 * `host_cap`/`base`/`base_cap` to its own char buffers, and passes its
44 * address down the pipeline. Each stage fills the fields it owns; the
45 * final stage copies the host and base into the caller's buffers.
46 */
47typedef struct {
48 char *host; /* caller buffer for the NUL-terminated host */
49 size_t host_cap; /* capacity of `host` (incl. the NUL) */
50 char *base; /* caller buffer for the base path (http) / bucket(s3)*/
51 size_t base_cap; /* capacity of `base` */
52 int port; /* resolved port */
53 int tls; /* 1 iff https (http parser); 0 for s3 */
54 size_t host_len; /* scratch: host length within the authority segment */
55 u_char *rest; /* scratch: segment remainder after the scheme */
56 size_t rest_len; /* scratch: length of `rest` */
57} vfs_origin_parse_t;
