2 * http_compress.c — outbound (GET) response compression (see http_compress.h).
3 *
4 * Negotiation picks the best AVAILABLE codec named in Accept-Encoding by a fixed
5 * preference (zstd > br > gzip > xz > deflate > bzip2 > lz4), skipping ranges, HEAD,

26/* Content types that are already compressed — never re-compress these. */
27static int
28content_type_incompressible(ngx_http_request_t *r)
29{
30 static const char *deny[] = {
31 "application/gzip", "application/zstd", "application/x-xz",
32 "application/x-bzip2", "application/zip", "application/x-7z-compressed",
33 "application/x-br", "application/x-brotli", /* already brotli-encoded */
34 "application/x-rar-compressed", "application/x-zstd",
35 "image/", "video/", "audio/", "application/octet-stream-compressed",
36 };
37 ngx_str_t *ct = &r->headers_out.content_type;
38 size_t i;
39
40 if (ct->len == 0 || ct->data == NULL) {
41 return 0; /* unknown type: allow compression */
42 }
43 for (i = 0; i < sizeof(deny) / sizeof(deny[0]); i++) {
44 size_t dl = ngx_strlen(deny[i]);
45 if (ct->len >= dl && ngx_strncasecmp(ct->data, (u_char *) deny[i], dl) == 0) {
46 return 1;
47 }
48 }
49 return 0;
50}

52/*
53 * WHAT: decide whether the ";q=..." parameter starting at `semi` explicitly
54 * disables the coding (an exactly-zero q-value).
55 * WHY: RFC 7231 5.3.1 lets a client veto a coding with "q=0" in any of its
56 * zero spellings ("q=0", "q=0.", "q=0.0", "q=0.00", "q=0.000"); a nonzero
57 * fractional digit (q=0.x) still accepts. Pulled out of the segment scan so
58 * the caller stays a flat name-match with no nested q-parser.
59 * HOW: find "q=0" within [semi, seg_end); a match is a real veto UNLESS it is
60 * followed by ".<digits>" containing a non-'0' digit (which makes q>0).
61 */
62static int
63accept_encoding_q_is_zero(const u_char *semi, const u_char *seg_end)
64{
65 const u_char *q = ngx_strlcasestrn((u_char *) semi, (u_char *) seg_end,
66 (u_char *) "q=0", 3 - 1);
67 const u_char *after;
68
69 if (q == NULL) {
70 return 0; /* no explicit q=0 => not vetoed */
71 }
72 after = q + 3; /* first char past "q=0" */
73 if (after < seg_end && *after == '.') {
74 const u_char *d = after + 1;
75 /* scan the fractional digits; any non-'0' => q>0 => accepted */
76 while (d < seg_end && *d >= '0' && *d <= '9') {
77 if (*d != '0') { return 0; }
78 d++;
79 }
80 }
81 return 1; /* q=0 / q=0. / q=0.0[00] => vetoed */
82}

84/*
85 * WHAT: test whether a single Accept-Encoding list member [seg, seg_end) names
86 * `token` (len `tlen`) as an acceptable coding.
87 * WHY: isolates the per-member name comparison + q-value veto from the list
88 * iteration so neither carries the other's branching.
89 * HOW: the coding name runs up to ';' (its q-value) or the member end, trimmed
90 * of trailing OWS; a case-insensitive length-exact match that is not
91 * vetoed by an explicit q=0 means the coding is acceptable.
92 */
93static int
94accept_encoding_segment_has(const u_char *seg, const u_char *seg_end,
95 const char *token, size_t tlen)
96{
97 const u_char *semi = ngx_strlchr((u_char *) seg, (u_char *) seg_end, ';');
98 const u_char *name_end = semi ? semi : seg_end;
99
100 while (name_end > seg && (name_end[-1] == ' ' || name_end[-1] == '\t')) {
101 name_end--;
102 }
103 if ((size_t) (name_end - seg) != tlen
104 || ngx_strncasecmp((u_char *) seg, (u_char *) token, tlen) != 0)
105 {
106 return 0;
107 }
108 if (semi != NULL && accept_encoding_q_is_zero(semi, seg_end)) {
109 return 0;
110 }
111 return 1;
112}

114/* Is `token` (len) present as an acceptable coding in the Accept-Encoding value
115 * (case-insensitive, ignoring an explicit ";q=0")? */
116static int
117accept_encoding_has(const u_char *ae, size_t ae_len, const char *token)
118{
119 size_t tlen = ngx_strlen(token);
120 const u_char *p = ae, *end = ae + ae_len;
121
122 while (p < end) {
123 const u_char *comma, *seg_end;
124
125 /* skip inter-member OWS — RFC 7230 permits SP and HTAB around list members */
126 while (p < end && (*p == ' ' || *p == '\t' || *p == ',')) { p++; }
127 comma = ngx_strlchr((u_char *) p, (u_char *) end, ',');
128 seg_end = comma ? comma : end;
129 if (accept_encoding_segment_has(p, seg_end, token, tlen)) {
130 return 1;
131 }
132 p = seg_end;
133 }
134 return 0;
135}

137brix_codec_id_t
138brix_http_compress_negotiate(ngx_http_request_t *r, off_t file_size,
139 ngx_flag_t is_range)
140{
141 ngx_table_elt_t *ae;
142 size_t i;
143
144 if (is_range || r->header_only || file_size < BRIX_COMPRESS_MIN_SIZE) {
145 return BRIX_CODEC_IDENTITY;
146 }
147 ae = r->headers_in.accept_encoding;
148 if (ae == NULL || ae->value.len == 0) {
149 return BRIX_CODEC_IDENTITY;
150 }
151 if (content_type_incompressible(r)) {
152 return BRIX_CODEC_IDENTITY;
153 }
154 for (i = 0; i < sizeof(brix_compress_pref) / sizeof(brix_compress_pref[0]);
155 i++)
156 {
157 brix_codec_id_t id = brix_compress_pref[i];
158 const brix_codec_desc_t *d = brix_codec_by_id(id);
159 if (d != NULL && d->available
160 && accept_encoding_has(ae->value.data, ae->value.len, d->http_token))
161 {
162 return id;
163 }
164 }
165 return BRIX_CODEC_IDENTITY;
166}

168/* Push one memory buffer of [data,len) to the output filter; last marks EOF. */
169static ngx_int_t
170compress_emit(ngx_http_request_t *r, const u_char *data, size_t len, int last)
171{
172 ngx_buf_t *b;
173 ngx_chain_t out;
174
175 b = ngx_create_temp_buf(r->pool, len > 0 ? len : 1);
176 if (b == NULL) {
177 return NGX_ERROR;
178 }
179 if (len > 0) {
180 ngx_memcpy(b->pos, data, len);
181 b->last = b->pos + len;
182 } else {
183 b->last = b->pos;
184 }
185 b->memory = 0;
186 b->temporary = 1;
187 b->flush = 1;
188 b->last_buf = last ? 1 : 0;
189 b->last_in_chain = 1;
190 out.buf = b;
191 out.next = NULL;
192 return ngx_http_output_filter(r, &out);
193}

195/*
196 * WHAT: file-local state for one compressed send — the fd/codec/buffers plus the
197 * running input offset and the caller's compressed-byte counter.
198 * WHY: lets the read/drain steps be split out as small helpers that take one
199 * `*st` instead of re-threading the eight send parameters (the extern
200 * signature stays frozen; this is the internal bundle only).
201 * HOW: zero-initialised by the orchestrator; `off` advances as chunks are read,
202 * `bytes_in_out` (nullable) accumulates emitted compressed bytes.
203 */
204typedef struct {
205 ngx_http_request_t *r;
206 ngx_fd_t fd;
207 off_t size;
208 off_t off;
209 brix_codec_stream_t *stream;
210 u_char *inbuf;
211 u_char *outbuf;
212 off_t *bytes_in_out;
213} compress_send_t;

215/*
216 * WHAT: emit the response headers for a compressed body — 200 + ETag/Last-Modified,
217 * forced chunked, Content-Encoding + Vary.
218 * WHY: keeps all header mutation (and its allocation failure paths) out of the
219 * streaming orchestrator so that reads as a flat init-then-stream sequence.
220 * HOW: set the file headers, drop Content-Length (unknown => chunked), then push
221 * the Content-Encoding (desc->http_token) and Vary list elements.
222 */
223static ngx_int_t
224compress_set_headers(ngx_http_request_t *r, const brix_codec_desc_t *desc,
225 off_t size, time_t mtime, unsigned etag_flags)
226{
227 ngx_table_elt_t *ce, *vary;
228
229 if (brix_http_set_file_headers(r, mtime, size, size, NULL, etag_flags,
230 0, 0, size > 0 ? size - 1 : 0) != NGX_OK)
231 {
232 return NGX_ERROR;
233 }
234 r->headers_out.content_length_n = -1; /* unknown -> chunked */
235 if (r->headers_out.content_length != NULL) {
236 r->headers_out.content_length->hash = 0;
237 r->headers_out.content_length = NULL;
238 }
239
240 ce = ngx_list_push(&r->headers_out.headers);
241 vary = ngx_list_push(&r->headers_out.headers);
242 if (ce == NULL || vary == NULL) {
243 return NGX_ERROR;
244 }
245 ce->hash = 1;
246 ngx_str_set(&ce->key, "Content-Encoding");
247 ce->value.data = (u_char *) desc->http_token;
248 ce->value.len = ngx_strlen(desc->http_token);
249 vary->hash = 1;
250 ngx_str_set(&vary->key, "Vary");
251 ngx_str_set(&vary->value, "Accept-Encoding");
252 return NGX_OK;
253}

255/*
256 * WHAT: drain the codec for the chunk currently in `st->inbuf` ([0,fill)), pushing
257 * every compressed output buffer to the filter; `finish` flushes at EOF.
258 * WHY: the inner compress/emit loop is the densest branching in the sender; as a
259 * named helper it is independently reviewable and the outer loop stays flat.
260 * HOW: step the codec until the chunk is consumed (or END at finish); emit each
261 * non-empty output; accumulate emitted bytes. Sets `*ended` on codec END.
262 * Returns NGX_OK on progress, NGX_ERROR on a codec/emit failure.
263 */
264static ngx_int_t
265compress_drain_chunk(compress_send_t *st, size_t fill, int finish, int *ended)
266{
267 size_t ip = 0;
268
269 for (;;) {
270 size_t op = 0;
271 brix_codec_rc_t crc;
272
273 crc = brix_codec_step(st->stream, st->inbuf, fill, &ip, st->outbuf,
274 BRIX_COMPRESS_CHUNK, &op, finish);
275 if (crc < 0) {
276 return NGX_ERROR;
277 }
278 if (st->bytes_in_out) { *st->bytes_in_out += (off_t) op; }
279 if (crc == BRIX_CODEC_END) {
280 *ended = 1;
281 return compress_emit(st->r, st->outbuf, op, 1);
282 }
283 if (op > 0 && compress_emit(st->r, st->outbuf, op, 0) != NGX_OK) {
284 return NGX_ERROR;
285 }
286 if (op == BRIX_COMPRESS_CHUNK) { continue; } /* drain more */
287 if (ip < fill) { continue; }
288 return NGX_OK; /* chunk consumed; read next */
289 }
290}

292/*
293 * WHAT: read one storage chunk into `st->inbuf` and drain it through the codec.
294 * WHY: one read+drain iteration as a helper turns the send loop into a flat
295 * `while (!done) { step; }` with no nested read/compress branching.
296 * HOW: pread a full chunk via the storage seam (EINTR/short-read handled by the
297 * primitive); a short fill means EOF; advance `off` and drain. Sets `*done`
298 * when the codec signals END. Returns NGX_OK / NGX_ERROR.
299 */
300static ngx_int_t
301compress_stream_step(compress_send_t *st, int *done)
302{
303 size_t got = 0;
304 size_t fill;
305 int finish;
306
307 if (brix_vfs_pread_full(st->fd, st->inbuf, BRIX_COMPRESS_CHUNK, st->off,
308 &got) != NGX_OK)
309 {
310 return NGX_ERROR;
311 }
312 fill = got;
313 if (got > 0) { st->off += (off_t) got; }
314 finish = (st->off >= st->size) || (got == 0);
315 return compress_drain_chunk(st, fill, finish, done);
316}

318/*
319 * WHAT: run the read -> codec -> output-filter stream until the codec ends or an
320 * error occurs (headers already sent, buffers/stream already open).
321 * WHY: separates the loop driver from the one-time setup so each is short.
322 * HOW: iterate compress_stream_step until it flags done or returns NGX_ERROR.
323 */
324static ngx_int_t
325compress_stream_body(compress_send_t *st)
326{
327 int done = 0;
328
329 while (!done) {
330 if (compress_stream_step(st, &done) != NGX_OK) {
331 return NGX_ERROR;
332 }
333 }
334 return NGX_OK;
335}

337ngx_int_t
338brix_http_send_file_compressed(ngx_http_request_t *r, ngx_fd_t send_fd,
339 const char *fs_path, off_t size, brix_codec_id_t codec, time_t mtime,
340 unsigned etag_flags, off_t *bytes_in_out)
341{
342 const brix_codec_desc_t *desc = brix_codec_by_id(codec);
343 compress_send_t st = {0};
344 ngx_int_t rc;
345
346 (void) fs_path;
347 if (desc == NULL || !desc->available) {
348 ngx_close_file(send_fd);
349 return NGX_HTTP_INTERNAL_SERVER_ERROR;
350 }
351
352 /* Headers: 200 + ETag/Last-Modified, then force chunked + Content-Encoding. */
353 if (compress_set_headers(r, desc, size, mtime, etag_flags) != NGX_OK) {
354 ngx_close_file(send_fd);
355 return NGX_HTTP_INTERNAL_SERVER_ERROR;
356 }
357
358 st.r = r;
359 st.fd = send_fd;
360 st.size = size;
361 st.bytes_in_out = bytes_in_out;
362 st.stream = brix_codec_open(codec, BRIX_CODEC_DIR_COMPRESS, -1, NULL);
363 st.inbuf = ngx_palloc(r->pool, BRIX_COMPRESS_CHUNK);
364 st.outbuf = ngx_palloc(r->pool, BRIX_COMPRESS_CHUNK);
365 if (st.stream == NULL || st.inbuf == NULL || st.outbuf == NULL) {
366 if (st.stream) { brix_codec_close(st.stream); }
367 ngx_close_file(send_fd);
368 return NGX_HTTP_INTERNAL_SERVER_ERROR;
369 }
370
371 if (ngx_http_send_header(r) == NGX_ERROR) {
372 brix_codec_close(st.stream);
373 ngx_close_file(send_fd);
374 return NGX_ERROR;
375 }
376
377 rc = compress_stream_body(&st);
378
379 brix_codec_close(st.stream);
380 ngx_close_file(send_fd);
381 return rc;
382}
