1/*
2 * http_compress.c — outbound (GET) response compression (see http_compress.h).
3 *
4 * Negotiation picks the best AVAILABLE codec named in Accept-Encoding by a fixed
5 * preference (zstd > br > gzip > xz > deflate > bzip2 > lz4), skipping ranges, HEAD,
6 * tiny files, and already-compressed content types. The compressed sender forces
7 * chunked output (compressed length is unknown up front) and streams the file
8 * through the shared codec abstraction into the nginx output filter.
9 */

26/* Content types that are already compressed — never re-compress these. */
27static int
28content_type_incompressible(ngx_http_request_t *r)
29{
30 static const char *deny[] = {
31 "application/gzip", "application/zstd", "application/x-xz",
32 "application/x-bzip2", "application/zip", "application/x-7z-compressed",
33 "application/x-br", "application/x-brotli", /* already brotli-encoded */
34 "application/x-rar-compressed", "application/x-zstd",
35 "image/", "video/", "audio/", "application/octet-stream-compressed",
36 };
37 ngx_str_t *ct = &r->headers_out.content_type;
38 size_t i;
39
40 if (ct->len == 0 || ct->data == NULL) {
41 return 0; /* unknown type: allow compression */
42 }
43 for (i = 0; i < sizeof(deny) / sizeof(deny[0]); i++) {
44 size_t dl = ngx_strlen(deny[i]);
45 if (ct->len >= dl && ngx_strncasecmp(ct->data, (u_char *) deny[i], dl) == 0) {
46 return 1;
47 }
48 }
49 return 0;
50}

52/* Is `token` (len) present as an acceptable coding in the Accept-Encoding value
53 * (case-insensitive, ignoring an explicit ";q=0")? */
54static int
55accept_encoding_has(const u_char *ae, size_t ae_len, const char *token)
56{
57 size_t tlen = ngx_strlen(token);
58 const u_char *p = ae, *end = ae + ae_len;
59
60 while (p < end) {
61 const u_char *comma, *semi, *tok_end;
62 size_t seg_len;
63
64 /* skip inter-member OWS — RFC 7230 permits SP and HTAB around list members */
65 while (p < end && (*p == ' ' || *p == '\t' || *p == ',')) { p++; }
66 comma = ngx_strlchr((u_char *) p, (u_char *) end, ',');
67 tok_end = comma ? comma : end;
68 seg_len = (size_t) (tok_end - p);
69 /* token name is up to ';' (q-value) or whitespace */
70 semi = ngx_strlchr((u_char *) p, (u_char *) tok_end, ';');
71 {
72 const u_char *name_end = semi ? semi : tok_end;
73 while (name_end > p && (name_end[-1] == ' ' || name_end[-1] == '\t')) {
74 name_end--;
75 }
76 if ((size_t) (name_end - p) == tlen
77 && ngx_strncasecmp((u_char *) p, (u_char *) token, tlen) == 0)
78 {
79 /* reject explicit q=0 in any of its exactly-zero spellings:
80 * "q=0", "q=0.", "q=0.0", "q=0.00", "q=0.000" (RFC 7231 5.3.1).
81 * Accept only a q-value with a NONZERO fractional digit (q=0.x). */
82 if (semi != NULL) {
83 const u_char *q = ngx_strlcasestrn((u_char *) semi,
84 (u_char *) tok_end, (u_char *) "q=0", 3 - 1);
85 if (q != NULL) {
86 const u_char *after = q + 3; /* first char past "q=0" */
87 int nonzero = 0;
88 if (after < tok_end && *after == '.') {
89 const u_char *d = after + 1;
90 /* scan the fractional digits; any non-'0' => q>0 */
91 while (d < tok_end && *d >= '0' && *d <= '9') {
92 if (*d != '0') { nonzero = 1; break; }
93 d++;
94 }
95 }
96 if (!nonzero) { /* q=0 / q=0. / q=0.0[00] => reject */
97 p = tok_end;
98 (void) seg_len;
99 continue;
100 }
101 }
102 }
103 return 1;
104 }
105 }
106 p = tok_end;
107 }
108 return 0;
109}

111brix_codec_id_t
112brix_http_compress_negotiate(ngx_http_request_t *r, off_t file_size,
113 ngx_flag_t is_range)
114{
115 ngx_table_elt_t *ae;
116 size_t i;
117
118 if (is_range || r->header_only || file_size < BRIX_COMPRESS_MIN_SIZE) {
119 return BRIX_CODEC_IDENTITY;
120 }
121 ae = r->headers_in.accept_encoding;
122 if (ae == NULL || ae->value.len == 0) {
123 return BRIX_CODEC_IDENTITY;
124 }
125 if (content_type_incompressible(r)) {
126 return BRIX_CODEC_IDENTITY;
127 }
128 for (i = 0; i < sizeof(brix_compress_pref) / sizeof(brix_compress_pref[0]);
129 i++)
130 {
131 brix_codec_id_t id = brix_compress_pref[i];
132 const brix_codec_desc_t *d = brix_codec_by_id(id);
133 if (d != NULL && d->available
134 && accept_encoding_has(ae->value.data, ae->value.len, d->http_token))
135 {
136 return id;
137 }
138 }
139 return BRIX_CODEC_IDENTITY;
140}

142/* Push one memory buffer of [data,len) to the output filter; last marks EOF. */
143static ngx_int_t
144compress_emit(ngx_http_request_t *r, const u_char *data, size_t len, int last)
145{
146 ngx_buf_t *b;
147 ngx_chain_t out;
148
149 b = ngx_create_temp_buf(r->pool, len > 0 ? len : 1);
150 if (b == NULL) {
151 return NGX_ERROR;
152 }
153 if (len > 0) {
154 ngx_memcpy(b->pos, data, len);
155 b->last = b->pos + len;
156 } else {
157 b->last = b->pos;
158 }
159 b->memory = 0;
160 b->temporary = 1;
161 b->flush = 1;
162 b->last_buf = last ? 1 : 0;
163 b->last_in_chain = 1;
164 out.buf = b;
165 out.next = NULL;
166 return ngx_http_output_filter(r, &out);
167}

169ngx_int_t
170brix_http_send_file_compressed(ngx_http_request_t *r, ngx_fd_t send_fd,
171 const char *fs_path, off_t size, brix_codec_id_t codec, time_t mtime,
172 unsigned etag_flags, off_t *bytes_in_out)
173{
174 const brix_codec_desc_t *desc = brix_codec_by_id(codec);
175 brix_codec_stream_t *s;
176 ngx_table_elt_t *ce, *vary;
177 u_char *inbuf, *outbuf;
178 off_t off = 0;
179 ngx_int_t rc;
180 int done = 0;
181
182 (void) fs_path;
183 if (desc == NULL || !desc->available) {
184 ngx_close_file(send_fd);
185 return NGX_HTTP_INTERNAL_SERVER_ERROR;
186 }
187
188 /* Headers: 200 + ETag/Last-Modified, then force chunked + Content-Encoding. */
189 if (brix_http_set_file_headers(r, mtime, size, size, NULL, etag_flags,
190 0, 0, size > 0 ? size - 1 : 0) != NGX_OK)
191 {
192 ngx_close_file(send_fd);
193 return NGX_HTTP_INTERNAL_SERVER_ERROR;
194 }
195 r->headers_out.content_length_n = -1; /* unknown -> chunked */
196 if (r->headers_out.content_length != NULL) {
197 r->headers_out.content_length->hash = 0;
198 r->headers_out.content_length = NULL;
199 }
200
201 ce = ngx_list_push(&r->headers_out.headers);
202 vary = ngx_list_push(&r->headers_out.headers);
203 if (ce == NULL || vary == NULL) {
204 ngx_close_file(send_fd);
205 return NGX_HTTP_INTERNAL_SERVER_ERROR;
206 }
207 ce->hash = 1;
208 ngx_str_set(&ce->key, "Content-Encoding");
209 ce->value.data = (u_char *) desc->http_token;
210 ce->value.len = ngx_strlen(desc->http_token);
211 vary->hash = 1;
212 ngx_str_set(&vary->key, "Vary");
213 ngx_str_set(&vary->value, "Accept-Encoding");
214
215 s = brix_codec_open(codec, BRIX_CODEC_DIR_COMPRESS, -1, NULL);
216 inbuf = ngx_palloc(r->pool, BRIX_COMPRESS_CHUNK);
217 outbuf = ngx_palloc(r->pool, BRIX_COMPRESS_CHUNK);
218 if (s == NULL || inbuf == NULL || outbuf == NULL) {
219 if (s) { brix_codec_close(s); }
220 ngx_close_file(send_fd);
221 return NGX_HTTP_INTERNAL_SERVER_ERROR;
222 }
223
224 if (ngx_http_send_header(r) == NGX_ERROR) {
225 brix_codec_close(s);
226 ngx_close_file(send_fd);
227 return NGX_ERROR;
228 }
229
230 rc = NGX_OK;
231 while (!done) {
232 size_t ip = 0, fill = 0;
233 size_t got = 0;
234 int finish;
235
236 /* One full chunk through the storage seam (EINTR/short-read handled by
237 * the primitive); a short fill means EOF and ends the stream. */
238 if (brix_vfs_pread_full(send_fd, inbuf, BRIX_COMPRESS_CHUNK, off,
239 &got) != NGX_OK)
240 {
241 rc = NGX_ERROR; break;
242 }
243 if (got > 0) { off += (off_t) got; fill = got; }
244 finish = (off >= size) || (got == 0);
245
246 for (;;) {
247 size_t op = 0;
248 brix_codec_rc_t crc;
249
250 crc = brix_codec_step(s, inbuf, fill, &ip, outbuf,
251 BRIX_COMPRESS_CHUNK, &op, finish);
252 if (crc < 0) { rc = NGX_ERROR; done = 1; break; }
253 if (bytes_in_out) { *bytes_in_out += (off_t) op; }
254 if (crc == BRIX_CODEC_END) {
255 if (compress_emit(r, outbuf, op, 1) != NGX_OK) { rc = NGX_ERROR; }
256 done = 1;
257 break;
258 }
259 if (op > 0) {
260 if (compress_emit(r, outbuf, op, 0) != NGX_OK) {
261 rc = NGX_ERROR; done = 1; break;
262 }
263 }
264 if (op == BRIX_COMPRESS_CHUNK) { continue; } /* drain more */
265 if (ip < fill) { continue; }
266 break; /* chunk consumed; read next */
267 }
268 }
269
270 brix_codec_close(s);
271 ngx_close_file(send_fd);
272 return rc;
273}
