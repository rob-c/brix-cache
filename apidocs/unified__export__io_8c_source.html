5/*
6 * unified_export_io.c — scrape-time exporter for the unified brix_io_* families
7 * plus the shared counter-reader and the legacy per-server stream fold.
8 *
9 * WHAT: Renders the three brix_io_* Prometheus families — io_bytes_{read,
10 * written}, io_ops_total, and the io_latency_usec histogram — and hosts
11 * the two helpers shared with the rest of the exporter: brix_metric_value
12 * (lock-free counter load) and the legacy-bridge functions that fold the
13 * predating per-server stream counters (servers[].op_ok/op_err/bytes_*,
14 * auth string) into the unified stream-protocol rows, including the
15 * export-time auth fold (brix_unified_legacy_auth) called from the auth
16 * exporter in unified_export.c.
17 * WHY: The exporter half of unified.c grew past the file-size budget; the io
18 * families plus the legacy-bridge machinery form one cohesive cluster,
19 * distinct from the cred/cache/auth/tpc families in unified_export.c.
20 * HOW: Each emitter reads its region of *shm with brix_metric_value, folds in
21 * the matching legacy stream totals for root:// (and the webdav/s3 per-
22 * server tx/rx totals for io_bytes), and prints HELP/TYPE + per-label
23 * lines. The latency histogram is stored NON-cumulative, so the series
24 * helper cumulates buckets as it emits them (Prometheus `le` semantics).
25 * Exposition bytes and emission order are frozen against the prior form.
26 */
27
28/* Lock-free read of an atomic counter (fetch-add of 0) as an unsigned long long. */
29unsigned long long
30brix_metric_value(ngx_atomic_t *counter)
31{
32 return (unsigned long long) ngx_atomic_fetch_add(counter, 0);
33}

35/*
36 * Legacy-bridge helpers (brix_unified_legacy_*): the stream protocol predates
37 * the unified counters and still records into the per-server servers[] slots
38 * (op_ok/op_err/bytes_rx/bytes_tx and the auth string). These functions sum the
39 * matching in-use servers[] slots so the exporter can fold legacy stream activity
40 * into the unified stream-protocol values, keeping /metrics output continuous.
41 * They are export-time read-only aggregations — never used on the record path.
42 */
43static unsigned long long
44brix_unified_legacy_stream_bytes(ngx_brix_metrics_t *shm,
45 unsigned int is_write)
46{
47 ngx_brix_srv_metrics_t *srv;
48 ngx_uint_t i;
49 unsigned long long total;
50
51 total = 0;
52 for (i = 0; i < BRIX_METRICS_MAX_SERVERS; i++) {
53 srv = &shm->servers[i];
54 if (!srv->in_use) {
55 continue;
56 }
57 total += is_write
58 ? brix_metric_value(&srv->bytes_rx_total)
59 : brix_metric_value(&srv->bytes_tx_total);
60 }
61
62 return total;
63}

90static unsigned long long
91brix_unified_legacy_stream_op(ngx_brix_metrics_t *shm,
92 brix_metric_op_t op, unsigned int ok)
93{
94 switch (op) {
95 case BRIX_METRIC_OP_READ:
96 return brix_unified_legacy_stream_op_slot(shm, BRIX_OP_READ, ok)
97 + brix_unified_legacy_stream_op_slot(shm, BRIX_OP_READV, ok)
98 + brix_unified_legacy_stream_op_slot(shm, BRIX_OP_PGREAD, ok);
99 case BRIX_METRIC_OP_WRITE:
100 return brix_unified_legacy_stream_op_slot(shm, BRIX_OP_WRITE, ok)
101 + brix_unified_legacy_stream_op_slot(shm, BRIX_OP_WRITEV, ok);
102 case BRIX_METRIC_OP_STAT:
103 return brix_unified_legacy_stream_op_slot(shm, BRIX_OP_STAT, ok)
104 + brix_unified_legacy_stream_op_slot(shm, BRIX_OP_STATX, ok);
105 case BRIX_METRIC_OP_DELETE:
106 return brix_unified_legacy_stream_op_slot(shm, BRIX_OP_RM, ok)
107 + brix_unified_legacy_stream_op_slot(shm, BRIX_OP_RMDIR, ok);
108 case BRIX_METRIC_OP_MKDIR:
109 return brix_unified_legacy_stream_op_slot(shm, BRIX_OP_MKDIR, ok);
110 case BRIX_METRIC_OP_RENAME:
111 return brix_unified_legacy_stream_op_slot(shm, BRIX_OP_MV, ok);
112 case BRIX_METRIC_OP_DIRLIST:
113 return brix_unified_legacy_stream_op_slot(shm, BRIX_OP_DIRLIST, ok);
114 default:
115 return 0;
116 }

144static unsigned long long
145brix_unified_legacy_stream_auth(ngx_brix_metrics_t *shm,
146 ngx_uint_t method, ngx_uint_t status)
147{
148 ngx_brix_srv_metrics_t *srv;
149 ngx_uint_t i;
150 unsigned long long total;
151
152 total = 0;
153 for (i = 0; i < BRIX_METRICS_MAX_SERVERS; i++) {

156 continue;
157 }
158 total += status == BRIX_METRIC_AUTH_OK
159 ? brix_metric_value(&srv->op_ok[BRIX_OP_AUTH])
160 : brix_metric_value(&srv->op_err[BRIX_OP_AUTH]);
161 }

163 return total;

179/*
180 * unified_emit_io_bytes — render the brix_io_bytes_read + brix_io_bytes_written
181 * families (per-protocol byte totals).
182 *
183 * WHAT: Emits both io_bytes_{read,written} HELP/TYPE + per-proto lines.
184 * WHY: The two directions share the same shape (unified total + a per-proto
185 * legacy/http fold), so one helper keeps their emission logic together.
186 * HOW: For each proto, read the unified counter then fold in the legacy stream
187 * bytes (root) or the webdav/s3 per-server tx/rx totals; note read folds
188 * the *tx* side and written folds the *rx* side (client-perspective naming).
189 */
190void
191unified_emit_io_bytes(metrics_writer_t *mw, ngx_brix_metrics_t *shm)
192{
193 ngx_uint_t proto;
194 unsigned long long value;
195
196 mw_printf(mw,
197 "# HELP brix_io_bytes_read Total bytes read from storage, by protocol.\n"
198 "# TYPE brix_io_bytes_read counter\n");
199 for (proto = 0; proto < BRIX_PROTO_COUNT; proto++) {
200 value = brix_metric_value(&shm->unified.io_bytes_read[proto]);
201 if (proto == BRIX_PROTO_ROOT) {
202 value += brix_unified_legacy_stream_bytes(shm, 0);
203 } else if (proto == BRIX_PROTO_WEBDAV) {
204 value += brix_metric_value(&shm->webdav.bytes_tx_total);
205 } else if (proto == BRIX_PROTO_S3) {
206 value += brix_metric_value(&shm->s3.bytes_tx_total);
207 }
208 mw_printf(mw, "brix_io_bytes_read{proto=\"%s\"} %llu\n",
209 brix_metric_proto_name((brix_proto_t) proto), value);
210 }
211
212 mw_printf(mw,
213 "# HELP brix_io_bytes_written Total bytes written to storage, by protocol.\n"
214 "# TYPE brix_io_bytes_written counter\n");
215 for (proto = 0; proto < BRIX_PROTO_COUNT; proto++) {
216 value = brix_metric_value(&shm->unified.io_bytes_written[proto]);
217 if (proto == BRIX_PROTO_ROOT) {
218 value += brix_unified_legacy_stream_bytes(shm, 1);
219 } else if (proto == BRIX_PROTO_WEBDAV) {
220 value += brix_metric_value(&shm->webdav.bytes_rx_total);
221 } else if (proto == BRIX_PROTO_S3) {
222 value += brix_metric_value(&shm->s3.bytes_rx_total);
223 }
224 mw_printf(mw, "brix_io_bytes_written{proto=\"%s\"} %llu\n",
225 brix_metric_proto_name((brix_proto_t) proto), value);
226 }
227}

229/*
230 * unified_emit_io_ops — render the brix_io_ops_total family
231 * (per-proto/op/status operation counters).
232 *
233 * WHAT: Emits the HELP/TYPE header + one line per (proto, op, status) triple.
234 * WHY: Isolates the triple-nested loop and the root:// legacy fold from the
235 * exporter orchestrator.
236 * HOW: For root:// the ok(err==NONE) and error(err==OTHER) legacy stream op
237 * totals are added in — matching the legacy ok/err split — before printing.
238 */
239void
240unified_emit_io_ops(metrics_writer_t *mw, ngx_brix_metrics_t *shm)
241{
242 ngx_uint_t proto, op, err;
243 unsigned long long value;
244
245 mw_printf(mw,
246 "# HELP brix_io_ops_total I/O operations completed, by protocol, operation, and status.\n"
247 "# TYPE brix_io_ops_total counter\n");
248 for (proto = 0; proto < BRIX_PROTO_COUNT; proto++) {
249 for (op = 0; op < BRIX_METRIC_OP_COUNT; op++) {
250 for (err = 0; err < BRIX_ERR_COUNT; err++) {
251 value = brix_metric_value(
252 &shm->unified.io_ops_total[proto][op][err]);
253 if (proto == BRIX_PROTO_ROOT) {
254 if (err == BRIX_ERR_NONE) {
255 value += brix_unified_legacy_stream_op(
256 shm, (brix_metric_op_t) op, 1);
257 } else if (err == BRIX_ERR_OTHER) {
258 value += brix_unified_legacy_stream_op(
259 shm, (brix_metric_op_t) op, 0);
260 }
261 }
262 mw_printf(mw,
263 "brix_io_ops_total"
264 "{proto=\"%s\",op=\"%s\",status=\"%s\"} %llu\n",
265 brix_metric_proto_name((brix_proto_t) proto),
266 brix_metric_op_name((brix_metric_op_t) op),
267 brix_metric_err_name((brix_err_class_t) err),
268 value);
269 }
270 }

274/*
275 * unified_emit_io_latency_series — render the bucket/sum/count lines for ONE
276 * (proto, op) latency series.
277 *
278 * WHAT: Emits the cumulative le="…" buckets, the le="+Inf" bucket, and the
279 * _sum/_count lines for a single protocol+operation.
280 * WHY: The per-series cumulation is the only stateful part of the histogram;
281 * confining it here keeps unified_emit_io_latency a flat double loop.
282 * HOW: Storage is NON-cumulative (each I/O bumps only its own bucket), so the
283 * running `value` accumulates lower buckets as they are emitted; the +Inf
284 * bucket then equals the total count. Byte-frozen against the prior inline
285 * form.
286 */
287static void
288unified_emit_io_latency_series(metrics_writer_t *mw, ngx_brix_metrics_t *shm,
289 ngx_uint_t proto, ngx_uint_t op)
290{
291 ngx_uint_t bucket;
292 unsigned long long value;
293
294 value = 0;
295 for (bucket = 0; bucket < BRIX_IO_LATENCY_BUCKETS - 1; bucket++) {
296 value += brix_metric_value(&shm->unified.io_latency_bucket
297 [proto][op][bucket]);
298 mw_printf(mw,
299 "brix_io_latency_usec_bucket"
300 "{proto=\"%s\",op=\"%s\",le=\"%llu\"} %llu\n",
301 brix_metric_proto_name((brix_proto_t) proto),
302 brix_metric_op_name((brix_metric_op_t) op),
303 (unsigned long long) brix_latency_bounds[bucket],
304 value);
305 }
306 value += brix_metric_value(&shm->unified.io_latency_bucket
307 [proto][op][BRIX_IO_LATENCY_BUCKETS - 1]);
308 mw_printf(mw,
309 "brix_io_latency_usec_bucket"
310 "{proto=\"%s\",op=\"%s\",le=\"+Inf\"} %llu\n",
311 brix_metric_proto_name((brix_proto_t) proto),
312 brix_metric_op_name((brix_metric_op_t) op),
313 value);
314 mw_printf(mw,
315 "brix_io_latency_usec_sum{proto=\"%s\",op=\"%s\"} %llu\n",
316 brix_metric_proto_name((brix_proto_t) proto),
317 brix_metric_op_name((brix_metric_op_t) op),
318 brix_metric_value(&shm->unified.io_latency_sum_usec[proto][op]));
319 mw_printf(mw,
320 "brix_io_latency_usec_count{proto=\"%s\",op=\"%s\"} %llu\n",
321 brix_metric_proto_name((brix_proto_t) proto),
322 brix_metric_op_name((brix_metric_op_t) op),
323 brix_metric_value(&shm->unified.io_latency_count[proto][op]));
324}
