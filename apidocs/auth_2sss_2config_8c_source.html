24/*
25 * WHAT: SSS keytab line parser — wraps the shared grammar and layers on the
26 * server-only authz options (ANYUSR/ALLUSR/ANYGRP/USRGRP/NOIPCK) derived
27 * from the user/group/name fields.
28 * HOW: The shared kernel tokenises and validates the line; this function maps
29 * its tri-state result onto NGX_OK / NGX_ERROR, copies the neutral entry
30 * into an brix_sss_key_t, computes the option flags, and pushes the key.
31 */
32static ngx_int_t
33brix_sss_parse_key_line(ngx_conf_t *cf, ngx_array_t *keys,
34 char *line, ngx_uint_t line_no)
35{
36 sss_keytab_entry_t entry;
37 brix_sss_key_t key, *dst;
38 size_t name_len;
39 int rc;
40
41 rc = sss_keytab_parse_line(line, &entry, (int64_t) ngx_time());
42 if (rc < 0) {
43 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
44 "brix_sss_keytab: invalid key entry on line %ui",
45 line_no);
46 return NGX_ERROR;
47 }
48 if (rc == 0) {
49 return NGX_OK; /* blank / comment / expired — nothing to load */
50 }
51
52 ngx_memzero(&key, sizeof(key));
53 key.id = entry.id;
54 key.exp = (time_t) entry.exp;
55 key.key_len = entry.key_len;
56 ngx_memcpy(key.key, entry.key, entry.key_len);
57 ngx_cpystrn((u_char *) key.user, (u_char *) entry.user, sizeof(key.user));
58 ngx_cpystrn((u_char *) key.group, (u_char *) entry.group, sizeof(key.group));
59 ngx_cpystrn((u_char *) key.name, (u_char *) entry.name, sizeof(key.name));
60
61 if (strcmp(key.user, "anybody") == 0) {
62 key.opts |= BRIX_SSS_OPT_ANYUSR;
63 } else if (strcmp(key.user, "allusers") == 0) {
64 key.opts |= BRIX_SSS_OPT_ALLUSR;
65 }
66
67 if (strcmp(key.group, "anygroup") == 0) {
68 key.opts |= BRIX_SSS_OPT_ANYGRP;
69 } else if (strcmp(key.group, "usrgroup") == 0) {
70 key.opts |= BRIX_SSS_OPT_USRGRP;
71 }
72
73 name_len = strlen(key.name);
74 if (name_len > 0 && key.name[name_len - 1] == '+') {
75 key.opts |= BRIX_SSS_OPT_NOIPCK;
76 }
77
78 dst = ngx_array_push(keys);
79 if (dst == NULL) {
80 return NGX_ERROR;
81 }
82 *dst = key;
83 return NGX_OK;
84}

118 /* Open with O_NOFOLLOW to reject symlinks before any permission check.
119 * Using fstat() on the resulting fd eliminates the stat()/open() TOCTOU
120 * race where an attacker could swap the keytab for a symlink between
121 * the permission check and the actual open. */
122 keytab_fd = open((const char *) path->data,
123 O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
124 if (keytab_fd < 0) {
125 BRIX_DIAG_CONF(NGX_LOG_EMERG, cf, ngx_errno,
126 "brix: cannot open SSS keytab \"%V\"",
127 "the path is wrong, the file is unreadable by the nginx user, or "
128 "it is a symlink (rejected for safety)",
129 "generate the keytab with xrdsssadmin and give the nginx user "
130 "read access to the real file; the OS reason is appended below",
131 path);
132 return NGX_ERROR;
133 }
134
135 if (fstat(keytab_fd, &st) != 0) {
136 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
137 "brix: cannot stat SSS keytab \"%V\"", path);
138 close(keytab_fd);
139 return NGX_ERROR;
140 }
141
142 if (sss_keytab_mode_ok((const char *) path->data, st.st_mode, 1) != 0) {
143 BRIX_DIAG_CONF(NGX_LOG_EMERG, cf, 0,
144 "brix: SSS keytab \"%V\" has unsafe permissions",
145 "the keytab is a shared secret but is group/world readable or not "
146 "owned correctly",
147 "chmod 0400 (or 0600) the keytab and ensure it is owned by the "
148 "nginx user; it must not be readable by anyone else",
149 path);
150 close(keytab_fd);
151 return NGX_ERROR;
152 }
153
154 fp = fdopen(keytab_fd, "r");
155 if (fp == NULL) {
156 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
157 "brix: cannot fdopen SSS keytab \"%V\"", path);
158 close(keytab_fd);
159 return NGX_ERROR;
160 }
161 /* O_CLOEXEC was set at open() — no separate fcntl needed */
162
163 keys = ngx_array_create(cf->pool, 4, sizeof(brix_sss_key_t));
164 if (keys == NULL) {
165 fclose(fp);
166 return NGX_ERROR;
167 }
168
169 line_no = 0;
170 while (fgets(line, sizeof(line), fp) != NULL) {
171 line_no++;
172 if (brix_sss_parse_key_line(cf, keys, line, line_no) != NGX_OK) {
173 fclose(fp);
174 return NGX_ERROR;
175 }
176 }
177
178 if (ferror(fp)) {
179 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
180 "brix: cannot read SSS keytab \"%V\"", path);
181 fclose(fp);
182 return NGX_ERROR;
183 }
184
185 fclose(fp);
186
187 if (keys->nelts == 0) {
188 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
189 "brix: SSS keytab \"%V\" has no usable keys", path);
190 return NGX_ERROR;
191 }
192
193 *out_keys = keys;
194 return NGX_OK;
195}

197/* Does this server need the SSS keytab loaded for UPSTREAM proxy auth? A proxy
198 * authenticates to an SSS upstream using conf->sss_keys even when its own client
199 * auth is not SSS (e.g. brix_auth none + brix_proxy_auth sss, or a
200 * per-upstream "sss" policy). Without this the keys are never parsed and the
201 * upstream SSS handshake silently fails NotAuthorized. */
202static int
203brix_sss_upstream_needed(ngx_stream_brix_srv_conf_t *xcf)
204{
205 ngx_uint_t i;
206
207 if (!xcf->proxy.enable) {
208 return 0;
209 }
210 if (xcf->proxy.auth == BRIX_PROXY_AUTH_SSS) {
211 return 1;
212 }
213 if (xcf->proxy.upstreams != NULL) {
214 brix_proxy_upstream_t *ups = xcf->proxy.upstreams->elts;
215 for (i = 0; i < xcf->proxy.upstreams->nelts; i++) {
216 if (ups[i].auth == (ngx_int_t) BRIX_PROXY_AUTH_SSS) {
217 return 1;
218 }
219 }
220 }
221 return 0;
222}

251 need_client ? (need_upstream ? "client+upstream"
252 : "client")
253 : "upstream");
254
255 return NGX_OK;
256}
