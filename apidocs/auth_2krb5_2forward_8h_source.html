6/*
7 * forward.h — krb5 GSSAPI credential forwarding to the backend/origin.
8 *
9 * WHAT: One-shot helper that, given a forwardable delegated GSS credential
10 * captured during inbound krb5 auth (a gss_cred_id_t obtained with
11 * GSS_C_DELEG_FLAG), initiates a fresh GSSAPI security context to the
12 * upstream/origin service principal so this node acts AS the inbound user.
13 *
14 * WHY: Phase-70 §5.7 — krb5 is only backend-usable by EXCHANGE (GSSAPI
15 * forwarding). When the client presents a forwardable ticket the node
16 * receives a delegated credential and can re-authenticate to the origin on
17 * the user's behalf, with no admin pre-provisioning.
18 *
19 * HOW: brix_krb5_deleg_to_origin() imports the origin service principal name,
20 * acquires nothing (it uses the supplied delegated cred as the initiator
21 * identity) and calls gss_init_sec_context() ONCE, returning the initial
22 * output token to send to the origin. This is deliberately a single
23 * init-context step: the multi-leg GSSAPI negotiation loop (feeding origin
24 * replies back through gss_init_sec_context until GSS_S_COMPLETE) belongs to
25 * the origin-auth caller, which owns the wire exchange and the context
26 * handle lifetime. Availability is reported by brix_krb5_forward_available()
27 * so callers can fall back to SELECT when forwarding is not compiled in or
28 * not supported.
29 */
30
31/*
32 * Return 1 if this build was compiled with krb5/GSSAPI support AND credential
33 * forwarding is available; 0 otherwise (caller falls back to SELECT).
34 */
35int brix_krb5_forward_available(void);
