1/*
2 * pblock_anomaly.c — F9 eventual-consistency anomaly emulation for pblock.
3 *
4 * WHAT: Implements pblock_anomaly.h: the `recent` event table, event
5 * recording at create/update boundaries, and the pure-read anomaly
6 * consultations (visibility lag, stale stat, list lag).
7 *
8 * HOW: Wall-clock milliseconds via CLOCK_REALTIME (simulated lag is a
9 * human-scale test knob, not a monotonic-critical deadline). Rules are
10 * read from ctl at each metadata boundary and clamped to a sane cap so
11 * a fat-fingered ctl row cannot make an export unusable for hours.
12 * Rows are only written while a matching rule is armed and expire
13 * logically — the read side never mutates. ngx-free (libc + sqlite3);
14 * BRIX_HAVE_SQLITE-gated.
15 */

40static int64_t
41anomaly_now_ms(void)
42{
43 struct timespec ts;
44
45 clock_gettime(CLOCK_REALTIME, &ts);
46 return (int64_t) ts.tv_sec * 1000 + ts.tv_nsec / 1000000L;
47}

49/* The armed window for `key` in ms, clamped to [0, AN_MS_CAP]. 0 = off. */
50static int64_t
51anomaly_rule(const pblock_state_t *st, const char *key)
52{
53 char val[64];
54 int64_t ms;
55
56 if (pblock_ctl_get(st->cat, key, val, sizeof(val)) != 1) {
57 return 0;
58 }
59 ms = strtoll(val, NULL, 10);
60 if (ms < 0) {
61 ms = 0;
62 }
63 if (ms > AN_MS_CAP) {
64 ms = AN_MS_CAP;
65 }
66 return ms;
67}

81void
82pblock_anomaly_created(pblock_state_t *st, const char *path)
83{
84 sqlite3_stmt *q;
85
86 if (anomaly_rule(st, "anomaly.visibility_ms") <= 0
87 && anomaly_rule(st, "anomaly.list_lag_ms") <= 0)
88 {
89 return;
90 }
91 /* A re-create (after unlink) starts a fresh history: the update snapshot
92 * of the previous incarnation must not leak into stale stats. */
93 q = cat_prepare(st->cat,
94 "INSERT INTO recent VALUES(?1, ?2, 0, 0, 0)"
95 " ON CONFLICT(path) DO UPDATE SET created_ms = excluded.created_ms,"
96 " updated_ms = 0, old_size = 0, old_mtime = 0;");
97 if (q == NULL) {
98 return;
99 }
100 sqlite3_bind_text(q, 1, path, -1, SQLITE_STATIC);
101 sqlite3_bind_int64(q, 2, anomaly_now_ms());
102 (void) sqlite3_step(q);
103 sqlite3_finalize(q);
104}

106void
107pblock_anomaly_updated(pblock_state_t *st, const char *path,
108 int64_t old_size, int64_t old_mtime)
109{
110 sqlite3_stmt *q;
111
112 if (anomaly_rule(st, "anomaly.stale_stat_ms") <= 0) {
113 return;
114 }
115 q = cat_prepare(st->cat,
116 "INSERT INTO recent VALUES(?1, 0, ?2, ?3, ?4)"
117 " ON CONFLICT(path) DO UPDATE SET updated_ms = excluded.updated_ms,"
118 " old_size = excluded.old_size, old_mtime = excluded.old_mtime;");
119 if (q == NULL) {
120 return;
121 }
122 sqlite3_bind_text(q, 1, path, -1, SQLITE_STATIC);
123 sqlite3_bind_int64(q, 2, anomaly_now_ms());
124 sqlite3_bind_int64(q, 3, old_size);
125 sqlite3_bind_int64(q, 4, old_mtime);
126 (void) sqlite3_step(q);
127 sqlite3_finalize(q);
128}

130/* The recorded event row, or created_ms = updated_ms = 0 when absent. */
131static void
132anomaly_row(const pblock_state_t *st, const char *path, int64_t *created_ms,
133 int64_t *updated_ms, int64_t *old_size, int64_t *old_mtime)
134{
135 sqlite3_stmt *q;
136
137 *created_ms = *updated_ms = 0;
138 q = cat_prepare(st->cat,
139 "SELECT created_ms, updated_ms, old_size, old_mtime"
140 " FROM recent WHERE path = ?1;");
141 if (q == NULL) {
142 return;
143 }
144 sqlite3_bind_text(q, 1, path, -1, SQLITE_STATIC);
145 if (sqlite3_step(q) == SQLITE_ROW) {
146 *created_ms = sqlite3_column_int64(q, 0);
147 *updated_ms = sqlite3_column_int64(q, 1);
148 if (old_size != NULL) {
149 *old_size = sqlite3_column_int64(q, 2);
150 }
151 if (old_mtime != NULL) {
152 *old_mtime = sqlite3_column_int64(q, 3);
153 }
154 }
155 sqlite3_finalize(q);
156}

158int
159pblock_anomaly_hidden(pblock_state_t *st, const char *path)
160{
161 int64_t rule, created_ms, updated_ms;
162
163 rule = anomaly_rule(st, "anomaly.visibility_ms");
164 if (rule <= 0) {
165 return 0;
166 }
167 anomaly_row(st, path, &created_ms, &updated_ms, NULL, NULL);
168 return created_ms > 0 && anomaly_now_ms() - created_ms < rule;
169}

171int
172pblock_anomaly_stale(pblock_state_t *st, const char *path,
173 int64_t *size_io, int64_t *mtime_io)
174{
175 int64_t rule, created_ms, updated_ms, old_size, old_mtime;
176
177 rule = anomaly_rule(st, "anomaly.stale_stat_ms");
178 if (rule <= 0) {
179 return 0;
180 }
181 anomaly_row(st, path, &created_ms, &updated_ms, &old_size, &old_mtime);
182 if (updated_ms <= 0 || anomaly_now_ms() - updated_ms >= rule) {
183 return 0;
184 }
185 *size_io = old_size;
186 *mtime_io = old_mtime;
187 return 1;
188}

194 char full[1088];

201 snprintf(full, sizeof(full), "%s/%s",
