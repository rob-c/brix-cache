15/* ---- Validate segments and populate the coalescer range array ----
16 *
17 * WHAT: Validates every segment's offset (non-negative, no read_length
18 * overflow) and fills the caller-owned `ranges` array with the
19 * brix_byte_range_t view the shared coalescer consumes. Returns NGX_OK, or
20 * NGX_ERROR with `error_message` set on the first offending segment.
21 *
22 * WHY: The per-segment offset/overflow checks and range construction are a
23 * single pure pass with no I/O; isolating them keeps brix_readv_read_segments
24 * below the complexity gate and makes the validation independently reviewable.
25 *
26 * HOW:
27 * 1. For each segment reject a negative offset (degenerate wire input).
28 * 2. Reject a read_length that would overflow off_t when added to offset.
29 * 3. Record fd/start/end for the coalescer; a zero-length segment gets a
30 * degenerate end (start-1) so it never coalesces.
31 */
32static ngx_int_t
33brix_readv_build_ranges(const brix_readv_seg_desc_t *segments,
34 size_t segment_count, brix_byte_range_t *ranges, char *error_message,
35 size_t error_message_len)
36{
37 size_t i;
38
39 for (i = 0; i < segment_count; i++) {
40 if (segments[i].offset < 0) {
41 snprintf(error_message, error_message_len,
42 "negative readv offset at seg %zu", i);
43 return NGX_ERROR;
44 }
45 if (segments[i].read_length > 0
46 && (off_t) segments[i].read_length
47 > NGX_MAX_OFF_T_VALUE - segments[i].offset)
48 {
49 snprintf(error_message, error_message_len,
50 "readv offset overflow at seg %zu", i);
51 return NGX_ERROR;
52 }
53 ranges[i].fd = segments[i].fd;
54 ranges[i].start = segments[i].offset;
55 ranges[i].end = (segments[i].read_length > 0)
56 ? segments[i].offset
57 + (off_t) segments[i].read_length - 1
58 : segments[i].offset - 1; /* degenerate, no coalescing */
59 ranges[i].handle = 0;
60 }
61
62 return NGX_OK;
63}

84/* ---- Read one coalesced run of contiguous same-fd segments ----
85 *
86 * WHAT: Determines the contiguous run starting at `segment_index`, issues one
87 * preadv() through the Storage Driver seam to fill each segment's buffer, and
88 * reports the run's byte total via *run_bytes_out and its length via
89 * *run_count_out. Returns NGX_OK, or NGX_ERROR with rctx->error_message set
90 * on an I/O error or a short (past-EOF) read.
91 *
92 * WHY: The coalesce-decision + iovec-assembly + single vectored read is the
93 * body of the segment loop; extracting it keeps the loop flat and lifts the
94 * EINTR / short-read policy into one named, testable step.
95 *
96 * HOW:
97 * 1. Write the actual read length back into the run's leading wire header.
98 * 2. Ask the shared coalescer how many segments form the next contiguous run.
99 * 3. Assemble the preadv iovec from the original descriptors, summing bytes.
100 * 4. Resolve the effective SD object (bound driver or POSIX fd wrap) and
101 * issue preadv, retrying on EINTR.
102 * 5. Map a negative result to an I/O error and a short read to a past-EOF
103 * error; otherwise publish the run count and byte total.
104 */
105static ngx_int_t
106brix_readv_read_one_run(const brix_readv_run_ctx_t *rctx,
107 size_t segment_index, ngx_uint_t *run_count_out, size_t *run_bytes_out)
108{
109 struct iovec iov[BRIX_READV_PREADV_MAXIOV];
110 brix_readv_seg_desc_t *segments = rctx->segments;
111 brix_readv_seg_desc_t *first_segment = &segments[segment_index];
112 ngx_uint_t run_count;
113 size_t run_bytes;
114 ngx_uint_t k;
115 ssize_t bytes_read;
116 uint32_t rlen_be;
117
118 /* Write actual read length back to the wire header. */
119 rlen_be = htonl(first_segment->read_length);
120 ngx_memcpy(first_segment->header_read_length_ptr, &rlen_be, 4);
121
122 /*
123 * Delegate the contiguous-run decision to the shared coalescer.
124 * The coalescer checks fd equality and byte adjacency using the
125 * ranges array built above.
126 */
127 run_count = brix_range_vector_next_coalesced_run(
128 rctx->ranges, (ngx_uint_t) rctx->segment_count,
129 (ngx_uint_t) segment_index,
130 (ngx_uint_t) BRIX_READV_PREADV_MAXIOV);
131
132 /* Assemble the preadv iovec from the original segment descriptors. */
133 run_bytes = 0;
134 for (k = 0; k < run_count; k++) {
135 iov[k].iov_base = segments[segment_index + k].payload_ptr;
136 iov[k].iov_len = (size_t) segments[segment_index + k].read_length;
137 run_bytes += (size_t) segments[segment_index + k].read_length;
138 }
139
140 {
141 brix_sd_obj_t scratch;
142 brix_sd_obj_t *obj;
143
144 /* Route the coalesced vectored read through the Storage Driver seam:
145 * the handle's bound driver object when set (block-striped/object
146 * backend), else a POSIX fd wrap (unchanged). The EINTR /
147 * coalescing policy stays here. */
148 obj = brix_vfs_effective_obj(&first_segment->obj,
149 first_segment->fd, &scratch);
150 do {
151 /* Compat seam: falls back to per-iovec pread when the driver has
152 * no native preadv slot (remote/object backends). */
153 bytes_read = brix_sd_obj_preadv(
154 obj, iov, (int) run_count, first_segment->offset);
155 } while (bytes_read < 0 && errno == EINTR);
156 }
157
158 if (bytes_read < 0) {
159 snprintf(rctx->error_message, rctx->error_message_len,
160 "readv I/O error at seg %zu: %s",
161 segment_index, strerror(errno));
162 return NGX_ERROR;
163 }
164
165 if ((size_t) bytes_read != run_bytes) {
166 snprintf(rctx->error_message, rctx->error_message_len,
167 "readv past EOF at seg %zu", segment_index);
168 return NGX_ERROR;
169 }
170
171 *run_count_out = run_count;
172 *run_bytes_out = run_bytes;
173 return NGX_OK;
174}
