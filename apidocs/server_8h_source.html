4/*
5 * cms/server.h — CMS server-side handler types and API.
6 *
7 * This module listens on the CMS management port (default 1213) and accepts
8 * incoming connections from XRootD data servers. On login it registers the
9 * data server in the shared server registry (src/manager/registry.h); on each
10 * heartbeat it refreshes load metrics; on disconnect it unregisters.

22/*
23 * SSS cluster-authentication handshake state (W1a). Mirrors the real cmsd
24 * kYR_xauth exchange: after the data node's LOGIN frame arrives we send our
25 * security parms (REQUESTED) and defer registration until the node returns a
26 * valid SSS credential (DONE). NONE means SSS auth is not required.
27 */
28typedef enum {

30 CMS_AUTH_REQUESTED, /* sss required; awaiting the node's LOGIN frame */
31 CMS_AUTH_CHALLENGED, /* parms sent, awaiting the node's kYR_xauth cred */
32 CMS_AUTH_DONE /* credential verified — registration permitted */
33} brix_cms_auth_state_t;

68 /* ---- Phase 50: accept-side network-fault resilience (the untrusted face) ----
69 * The manager arms no timer until AFTER a node logs in, so a peer that never
70 * completes LOGIN (or trickles a partial header) squats a ctx + fd forever,
71 * and a silently-dead post-login node is only reaped when a ping *send* fails
72 * (which a black-holed peer never does). These close both holes. Unset
73 * auto-derives a generous ON default; an explicit 0 disables. */
74 ngx_msec_t login_timeout; /* [brix_cms_server_login_timeout] deadline to
75 complete LOGIN (+sss xauth); unset => 10s. */
76 ngx_msec_t idle_timeout; /* [brix_cms_server_idle_timeout] post-login
77 inbound-silence deadline; unset =>
78 max(3*interval, 90s). 0 = off. */
79 ngx_int_t max_connections; /* [brix_cms_server_max_connections] per-worker
80 cap on accepted CMS connections; unset => 4096.
81 0 = unlimited. */
82 ngx_int_t max_connections_per_ip; /* [brix_cms_server_max_connections_per_ip]
83 A3: per-source-IP cap; unset => 256. 0 = off. */
84 ngx_flag_t tcp_keepalive; /* [brix_cms_server_tcp_keepalive on] */
85 ngx_msec_t tcp_user_timeout;/* [brix_cms_server_tcp_user_timeout] ms; unset
86 => idle-timeout backstop. 0 = off. */
87};

107/* Stream connection handler for an accepted CMS data-server connection.
108 * Allocates the per-conn ctx on c->pool, runs the W1b CIDR allowlist gate and
109 * arms the W1a sss handshake, installs the read/write handlers, then drives the
110 * first read inline. Finalizes the session (no ctx leak — pool-cleaned) on
111 * alloc failure or a denied peer; otherwise returns with the conn live. */
