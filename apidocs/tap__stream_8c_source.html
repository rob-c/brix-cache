1/*
2 * tap_stream.c — chunk-wise streaming frame decoder for the byte relay.
3 *
4 * The relay hands us bytes as they arrive off the wire (arbitrary chunk sizes).
5 * A two-phase state machine reassembles each frame: collect the fixed header
6 * (24B request / 8B response), decode it, then consume `dlen` payload bytes —
7 * capturing the first BRIX_TAP_PATH_CAP of them as the path for a path-bearing
8 * request. A path frame emits once its path bytes are in; every other frame emits
9 * the moment its header completes, so a large write/read payload is skipped, never
10 * buffered.
11 */
12

18void
19brix_tap_stream_init(brix_tap_stream_t *st, brix_tap_ctx_t *tap,
20 brix_tap_dir_t dir)
21{
22 memset(st, 0, sizeof(*st));
23 st->tap = tap;
24 st->dir = dir;
25 st->preamble_skip = (dir == BRIX_TAP_C2U) ? BRIX_TAP_C2U_PREAMBLE : 0;
26 st->hdr_need = (dir == BRIX_TAP_C2U) ? 24 : 8;
27}

29/* Header complete: decode it, set up payload accounting + path capture. */
30static void
31tap_stream_on_header(brix_tap_stream_t *st)
32{
33 if (st->dir == BRIX_TAP_C2U) {
34 brix_tap_decode_request(st->hdr, st->hdr_need, &st->cur);
35 } else {
36 brix_tap_decode_response(st->hdr, st->hdr_need, &st->cur);
37 }
38 st->in_payload = 1;
39 st->payload_left = st->cur.dlen;
40 st->emitted = 0;
41 st->path_got = 0;
42 st->path_cap = 0;
43
44 /* kXR_writev stock framing: dlen frames only the 16-byte write_list
45 * descriptors; sum(wlen) segment-data bytes stream after the frame. Sum
46 * the wlen fields as the descriptors pass through so the trailing data can
47 * be consumed afterwards, keeping the stream aligned on the next header.
48 * (A non-16-aligned dlen is malformed — the server drops that link — so no
49 * extension is attempted and plain dlen consumption is fine.) */
50 st->wv_active = (st->dir == BRIX_TAP_C2U
51 && st->cur.opcode == kXR_writev
52 && st->cur.dlen > 0
53 && st->cur.dlen % 16 == 0);
54 st->wv_desc_got = 0;
55 st->wv_extra = 0;
56
57 /* kXR_chkpoint/ckpXeq stock framing: dlen covers only the embedded
58 * 24-byte sub-request header; the sub-request body streams after the
59 * frame (see chkpoint_xeq.c). Capture the embedded header as it passes
60 * so the trailing byte count can be recovered once the frame completes.
61 * The ckpXeq opcode byte sits at header offset 19 (kXR_ckpXeq == 4). */
62 st->ckp_active = (st->dir == BRIX_TAP_C2U
63 && st->cur.opcode == kXR_chkpoint
64 && st->hdr[19] == 4
65 && st->cur.dlen == 24);
66 st->ckp_hdr_got = 0;
67
68 /* The path lives in the payload that follows the header (not in our 24B hdr
69 * buffer), so decide path-capture from the opcode and capture it ourselves. */
70 st->cur.path = NULL;
71 st->cur.path_len = 0;
72 if (st->dir == BRIX_TAP_C2U && st->cur.dlen > 0
73 && brix_tap_opcode_has_path(st->cur.opcode))
74 {
75 st->path_cap = (st->cur.dlen < BRIX_TAP_PATH_CAP)
76 ? st->cur.dlen : BRIX_TAP_PATH_CAP;
77 }
78
79 /* A kXR_error body is errnum[4 BE] + errmsg — capture just the errnum
80 * (reusing the path-capture machinery) so the frame emits with the error
81 * code (kXR_NotFound, …) the bad-actor guard classifies on. */
82 if (st->dir == BRIX_TAP_U2C && st->cur.status == kXR_error
83 && st->cur.dlen > 0)
84 {
85 st->path_cap = (st->cur.dlen < 4) ? st->cur.dlen : 4;
86 }
87
88 if (st->path_cap == 0) { /* nothing to wait for → emit now */
89 brix_tap_emit(st->tap, &st->cur, st->dir, NULL, 0);
90 st->emitted = 1;
91 if (st->payload_left == 0) { /* whole frame done */
92 st->in_payload = 0;
93 st->hdr_got = 0;
94 }
95 }
96}

98void
99brix_tap_stream_feed(brix_tap_stream_t *st, const uint8_t *buf, size_t len)
100{
101 while (len > 0) {
102 if (st->preamble_skip > 0) { /* skip the 20-byte handshake */
103 size_t take = (len < st->preamble_skip) ? len : st->preamble_skip;
104 st->preamble_skip -= take;
105 buf += take;
106 len -= take;
107 continue;
108 }
109 if (!st->in_payload) {
110 size_t need = st->hdr_need - st->hdr_got;
111 size_t take = (len < need) ? len : need;
112 memcpy(st->hdr + st->hdr_got, buf, take);
113 st->hdr_got += take;
114 buf += take;
115 len -= take;
116 if (st->hdr_got == st->hdr_need) {
117 tap_stream_on_header(st);
118 }
119 continue;
120 }
121
122 /* in payload: capture path bytes (if any left), then skip the rest */
123 {
124 size_t take = (len < st->payload_left)
125 ? len : (size_t) st->payload_left;
126 if (st->path_cap > st->path_got) {
127 size_t cap = st->path_cap - st->path_got;
128 if (cap > take) { cap = take; }
129 memcpy(st->pathbuf + st->path_got, buf, cap);
130 st->path_got += cap;
131 }
132
133 /* ckpXeq embedded-header pass: capture the 24-byte sub-request
134 * header (it is the entire dlen-framed payload). */
135 if (st->ckp_active && st->ckp_hdr_got < sizeof(st->ckp_hdr)) {
136 size_t cap = sizeof(st->ckp_hdr) - st->ckp_hdr_got;
137 if (cap > take) { cap = take; }
138 memcpy(st->ckp_hdr + st->ckp_hdr_got, buf, cap);
139 st->ckp_hdr_got += cap;
140 }
141
142 /* writev descriptor pass: fold each completed 16-byte descriptor's
143 * big-endian wlen (offset 4) into the trailing-data total. */
144 if (st->wv_active) {
145 size_t k;
146 for (k = 0; k < take; k++) {
147 st->wv_desc[st->wv_desc_got++] = buf[k];
148 if (st->wv_desc_got == sizeof(st->wv_desc)) {
149 st->wv_extra += ((uint64_t) st->wv_desc[4] << 24)
150 | ((uint64_t) st->wv_desc[5] << 16)
151 | ((uint64_t) st->wv_desc[6] << 8)
152 | (uint64_t) st->wv_desc[7];
153 st->wv_desc_got = 0;
154 }
155 }
156 }
157
158 buf += take;
159 len -= take;
160 st->payload_left -= take;
161
162 if (!st->emitted && st->path_got >= st->path_cap) {
163 if (st->dir == BRIX_TAP_U2C) {
164 /* captured bytes = the kXR_error errnum, not a path */
165 if (st->path_got >= 4) {
166 st->cur.errnum = ((uint32_t) st->pathbuf[0] << 24)
167 | ((uint32_t) st->pathbuf[1] << 16)
168 | ((uint32_t) st->pathbuf[2] << 8)
169 | (uint32_t) st->pathbuf[3];
170 }
171 } else {
172 st->cur.path = st->pathbuf;
173 st->cur.path_len = st->path_got;
174 }
175 brix_tap_emit(st->tap, &st->cur, st->dir, NULL, 0);
176 st->emitted = 1;
177 }
178 if (st->payload_left == 0) {
179 /* ckpXeq embedded header done: the sub-request body
180 * (sub_dlen bytes) streams next. Mirror the server framing
181 * (brix_ckpxeq_body_extra): only extend for the embeds and
182 * bounds the server itself accepts — anything else makes the
183 * server drop the link, which tears the relay down too.
184 * An embedded writev extends by its descriptor block and
185 * arms the wv_* machinery, which then extends again by
186 * sum(wlen) exactly like a standalone writev. */
187 if (st->ckp_active) {
188 uint16_t sub_reqid = ((uint16_t) st->ckp_hdr[2] << 8)
189 | (uint16_t) st->ckp_hdr[3];
190 uint64_t sub_dlen = ((uint64_t) st->ckp_hdr[20] << 24)
191 | ((uint64_t) st->ckp_hdr[21] << 16)
192 | ((uint64_t) st->ckp_hdr[22] << 8)
193 | (uint64_t) st->ckp_hdr[23];
194
195 st->ckp_active = 0;
196 if (sub_dlen > 0 && st->ckp_hdr_got == sizeof(st->ckp_hdr)) {
197 if ((sub_reqid == kXR_write || sub_reqid == kXR_pgwrite)
198 && sub_dlen <= 16u * 1024 * 1024) /* MAX_WRITE_PAYLOAD */
199 {
200 st->payload_left = sub_dlen;
201 continue;
202 }
203 if (sub_reqid == kXR_writev
204 && sub_dlen % 16 == 0
205 && sub_dlen <= 1024 * 16) /* MAXSEGS * SEGSIZE */
206 {
207 st->payload_left = sub_dlen;
208 st->wv_active = 1;
209 st->wv_desc_got = 0;
210 st->wv_extra = 0;
211 continue;
212 }
213 }
214 }
215
216 /* Descriptors done: the segment data (sum(wlen) bytes) streams
217 * next — extend consumption instead of closing the frame. */
218 if (st->wv_active) {
219 st->wv_active = 0;
220 if (st->wv_extra > 0) {
221 st->payload_left = st->wv_extra;
222 continue;
223 }
224 }
225 if (!st->emitted) { /* defensive: emit even if no path */
226 brix_tap_emit(st->tap, &st->cur, st->dir, NULL, 0);
227 }
228 st->in_payload = 0;
229 st->hdr_got = 0;
230 }
231 }
232 }
