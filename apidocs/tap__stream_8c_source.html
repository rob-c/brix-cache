1/*
2 * tap_stream.c — chunk-wise streaming frame decoder for the byte relay.
3 *
4 * The relay hands us bytes as they arrive off the wire (arbitrary chunk sizes).
5 * A two-phase state machine reassembles each frame: collect the fixed header
6 * (24B request / 8B response), decode it, then consume `dlen` payload bytes —
7 * capturing the first BRIX_TAP_PATH_CAP of them as the path for a path-bearing
8 * request. A path frame emits once its path bytes are in; every other frame emits
9 * the moment its header completes, so a large write/read payload is skipped, never
10 * buffered.
11 */
12
13#include "tap.h"
14#include "protocols/root/protocol/opcodes.h" /* kXR_writev, kXR_chkpoint */
15
16#include <string.h>

18void
19brix_tap_stream_init(brix_tap_stream_t *st, brix_tap_ctx_t *tap,
20 brix_tap_dir_t dir)
21{
22 memset(st, 0, sizeof(*st));
23 st->tap = tap;
24 st->dir = dir;
25 st->preamble_skip = (dir == BRIX_TAP_C2U) ? BRIX_TAP_C2U_PREAMBLE : 0;
26 st->hdr_need = (dir == BRIX_TAP_C2U) ? 24 : 8;
27}

29/*
30 * WHAT: decode the just-completed fixed header into st->cur and reset the
31 * per-frame payload accounting.
32 * WHY: every framing/path decision below reads a freshly-decoded st->cur; this
33 * is the one place the wire header is parsed, keeping request/response
34 * decode symmetric and the payload counters zeroed for the new frame.
35 * HOW: dispatch on direction to the request/response decoder, then clear the
36 * payload/path/framing counters (pure state init — no I/O).
37 */
38static void
39tap_header_decode(brix_tap_stream_t *st)
40{
41 if (st->dir == BRIX_TAP_C2U) {
42 brix_tap_decode_request(st->hdr, st->hdr_need, &st->cur);
43 } else {
44 brix_tap_decode_response(st->hdr, st->hdr_need, &st->cur);
45 }
46 st->in_payload = 1;
47 st->payload_left = st->cur.dlen;
48 st->emitted = 0;
49 st->path_got = 0;
50 st->path_cap = 0;
51}

53/*
54 * WHAT: classify the frame's stock-framing quirks (writev / chkpoint-ckpXeq) and
55 * arm the descriptor/embedded-header capture machinery for them.
56 * WHY: a few opcodes frame only a prefix in dlen and stream more data after; the
57 * decoder must recover the trailing byte count from bytes seen in the
58 * payload pass, so the arming decision is made once here at header time.
59 * HOW: set wv_active / ckp_active from the opcode + dlen shape and zero their
60 * accumulators (pure state — the payload pass does the byte capture).
61 */
62static void
63tap_header_classify_framing(brix_tap_stream_t *st)
64{
65 /* kXR_writev stock framing: dlen frames only the 16-byte write_list
66 * descriptors; sum(wlen) segment-data bytes stream after the frame. Sum
67 * the wlen fields as the descriptors pass through so the trailing data can
68 * be consumed afterwards, keeping the stream aligned on the next header.
69 * (A non-16-aligned dlen is malformed — the server drops that link — so no
70 * extension is attempted and plain dlen consumption is fine.) */
71 st->wv_active = (st->dir == BRIX_TAP_C2U
72 && st->cur.opcode == kXR_writev
73 && st->cur.dlen > 0
74 && st->cur.dlen % 16 == 0);
75 st->wv_desc_got = 0;
76 st->wv_extra = 0;
77
78 /* kXR_chkpoint/ckpXeq stock framing: dlen covers only the embedded
79 * 24-byte sub-request header; the sub-request body streams after the
80 * frame (see chkpoint_xeq.c). Capture the embedded header as it passes
81 * so the trailing byte count can be recovered once the frame completes.
82 * The ckpXeq opcode byte sits at header offset 19 (kXR_ckpXeq == 4). */
83 st->ckp_active = (st->dir == BRIX_TAP_C2U
84 && st->cur.opcode == kXR_chkpoint
85 && st->hdr[19] == 4
86 && st->cur.dlen == 24);
87 st->ckp_hdr_got = 0;
88}

90/*
91 * WHAT: decide how many leading payload bytes to capture (path for a path-bearing
92 * request, or errnum for a kXR_error response) into st->path_cap.
93 * WHY: the path/errnum lives in the payload after our header buffer, so the
94 * amount to buffer must be derived from the opcode/status before the
95 * payload pass and clamped to BRIX_TAP_PATH_CAP / 4.
96 * HOW: clear st->cur.path, then set path_cap from the request-path or the
97 * error-errnum rule (pure computation — no I/O).
98 */
99static void
100tap_header_set_capture(brix_tap_stream_t *st)
101{
102 /* The path lives in the payload that follows the header (not in our 24B hdr
103 * buffer), so decide path-capture from the opcode and capture it ourselves. */
104 st->cur.path = NULL;
105 st->cur.path_len = 0;
106 if (st->dir == BRIX_TAP_C2U && st->cur.dlen > 0
107 && brix_tap_opcode_has_path(st->cur.opcode))
108 {
109 st->path_cap = (st->cur.dlen < BRIX_TAP_PATH_CAP)
110 ? st->cur.dlen : BRIX_TAP_PATH_CAP;
111 }
112
113 /* A kXR_error body is errnum[4 BE] + errmsg — capture just the errnum
114 * (reusing the path-capture machinery) so the frame emits with the error
115 * code (kXR_NotFound, …) the bad-actor guard classifies on. */
116 if (st->dir == BRIX_TAP_U2C && st->cur.status == kXR_error
117 && st->cur.dlen > 0)
118 {
119 st->path_cap = (st->cur.dlen < 4) ? st->cur.dlen : 4;
120 }
121}
122
123/*
124 * WHAT: header complete — decode it, arm framing/capture, and emit immediately if
125 * nothing in the payload needs to be waited for.
126 * WHY: a frame with no path/errnum to capture emits the moment its header
127 * completes (large read/write payloads are skipped, never buffered); one
128 * place must sequence decode → classify → capture → early-emit.
129 * HOW: call the three pure setup helpers, then when path_cap == 0 emit now and,
130 * if the payload is empty too, close the frame back to header state.
131 */
132static void
133tap_stream_on_header(brix_tap_stream_t *st)
134{
135 tap_header_decode(st);
136 tap_header_classify_framing(st);
137 tap_header_set_capture(st);
138
139 if (st->path_cap == 0) { /* nothing to wait for → emit now */
140 brix_tap_emit(st->tap, &st->cur, st->dir, NULL, 0);
141 st->emitted = 1;
142 if (st->payload_left == 0) { /* whole frame done */
143 st->in_payload = 0;
144 st->hdr_got = 0;
145 }
146 }
147}

149/*
150 * WHAT: consume `st->preamble_skip` leading bytes of the current buffer chunk.
151 * WHY: a C2U stream opens with a 20-byte handshake preamble that is not a frame
152 * and must be swallowed before header reassembly begins.
153 * HOW: advance buf/len by min(len, preamble_skip) and shrink the counter
154 * (pure pointer/length arithmetic).
155 */
156static void
157tap_feed_preamble(brix_tap_stream_t *st, const uint8_t **buf, size_t *len)
158{
159 size_t take = (*len < st->preamble_skip) ? *len : st->preamble_skip;
160 st->preamble_skip -= take;
161 *buf += take;
162 *len -= take;
163}
164
165/*
166 * WHAT: accumulate header bytes from the buffer chunk; on completion decode the
167 * frame.
168 * WHY: the fixed header (24B/8B) arrives across arbitrary chunk boundaries and
169 * must be reassembled before it can be decoded.
170 * HOW: memcpy min(len, remaining-header) into st->hdr, advance buf/len, and
171 * when the header is full invoke tap_stream_on_header (its only side
172 * effect is the early-emit path).
173 */
174static void
175tap_feed_header(brix_tap_stream_t *st, const uint8_t **buf, size_t *len)
176{
177 size_t need = st->hdr_need - st->hdr_got;
178 size_t take = (*len < need) ? *len : need;
179 memcpy(st->hdr + st->hdr_got, *buf, take);
180 st->hdr_got += take;
181 *buf += take;
182 *len -= take;
183 if (st->hdr_got == st->hdr_need) {
184 tap_stream_on_header(st);
185 }
186}

188/*
189 * WHAT: from the next `take` payload bytes, capture the path prefix, the ckpXeq
190 * embedded header, and each writev descriptor's wlen (as applicable).
191 * WHY: the path/errnum, the ckpXeq sub-request header, and the writev
192 * trailing-data total all live in bytes that pass through the payload; they
193 * must be snapshot as they go by since the payload is otherwise skipped.
194 * HOW: copy the still-needed prefix into st->pathbuf and st->ckp_hdr, and for an
195 * active writev fold each completed 16-byte descriptor's big-endian wlen
196 * (offset 4) into st->wv_extra (pure capture — no I/O, no *buf advance).
197 */
198static void
199tap_payload_capture(brix_tap_stream_t *st, const uint8_t *buf, size_t take)
200{
201 if (st->path_cap > st->path_got) {
202 size_t cap = st->path_cap - st->path_got;
203 if (cap > take) { cap = take; }
204 memcpy(st->pathbuf + st->path_got, buf, cap);
205 st->path_got += cap;
206 }
207
208 /* ckpXeq embedded-header pass: capture the 24-byte sub-request
209 * header (it is the entire dlen-framed payload). */
210 if (st->ckp_active && st->ckp_hdr_got < sizeof(st->ckp_hdr)) {
211 size_t cap = sizeof(st->ckp_hdr) - st->ckp_hdr_got;
212 if (cap > take) { cap = take; }
213 memcpy(st->ckp_hdr + st->ckp_hdr_got, buf, cap);
214 st->ckp_hdr_got += cap;
215 }
216
217 /* writev descriptor pass: fold each completed 16-byte descriptor's
218 * big-endian wlen (offset 4) into the trailing-data total. */
219 if (st->wv_active) {
220 size_t k;
221 for (k = 0; k < take; k++) {
222 st->wv_desc[st->wv_desc_got++] = buf[k];
223 if (st->wv_desc_got == sizeof(st->wv_desc)) {
224 st->wv_extra += ((uint64_t) st->wv_desc[4] << 24)
225 | ((uint64_t) st->wv_desc[5] << 16)
226 | ((uint64_t) st->wv_desc[6] << 8)
227 | (uint64_t) st->wv_desc[7];
228 st->wv_desc_got = 0;
229 }
230 }
231 }
232}

234/*
235 * WHAT: once the awaited leading bytes are in, finalize path/errnum on st->cur
236 * and emit the frame (exactly once).
237 * WHY: a path-bearing request emits after its path bytes arrive, and a kXR_error
238 * response emits after its 4-byte errnum — both reuse the same capture
239 * buffer and must be interpreted per direction before the single emit.
240 * HOW: when not yet emitted and the capture is satisfied, decode the U2C errnum
241 * (BE) or attach the C2U path pointer/length, then emit and latch emitted.
242 */
243static void
244tap_payload_emit_if_ready(brix_tap_stream_t *st)
245{
246 if (st->emitted || st->path_got < st->path_cap) {
247 return;
248 }
249 if (st->dir == BRIX_TAP_U2C) {
250 /* captured bytes = the kXR_error errnum, not a path */
251 if (st->path_got >= 4) {
252 st->cur.errnum = ((uint32_t) st->pathbuf[0] << 24)
253 | ((uint32_t) st->pathbuf[1] << 16)
254 | ((uint32_t) st->pathbuf[2] << 8)
255 | (uint32_t) st->pathbuf[3];
256 }
257 } else {
258 st->cur.path = st->pathbuf;
259 st->cur.path_len = st->path_got;
260 }
261 brix_tap_emit(st->tap, &st->cur, st->dir, NULL, 0);
262 st->emitted = 1;
263}

265/*
266 * WHAT: at a completed ckpXeq frame, decode the embedded sub-request and, for the
267 * write/pgwrite/writev embeds the server accepts, extend payload_left to
268 * cover the sub-request body.
269 * WHY: ckpXeq frames only its 24-byte embedded header in dlen; the sub-request
270 * body streams after and must be consumed (mirroring brix_ckpxeq_body_extra)
271 * to keep the stream aligned, bounded to what the server itself accepts.
272 * HOW: read sub_reqid/sub_dlen from st->ckp_hdr; on a bounded write/pgwrite set
273 * payload_left, on a bounded writev also re-arm wv_*; return 1 if the frame
274 * was extended (caller must `continue`), 0 otherwise.
275 */
276static int
277tap_payload_extend_ckpxeq(brix_tap_stream_t *st)
278{
279 uint16_t sub_reqid;
280 uint64_t sub_dlen;
281
282 if (!st->ckp_active) {
283 return 0;
284 }
285 sub_reqid = ((uint16_t) st->ckp_hdr[2] << 8)
286 | (uint16_t) st->ckp_hdr[3];
287 sub_dlen = ((uint64_t) st->ckp_hdr[20] << 24)
288 | ((uint64_t) st->ckp_hdr[21] << 16)
289 | ((uint64_t) st->ckp_hdr[22] << 8)
290 | (uint64_t) st->ckp_hdr[23];
291
292 st->ckp_active = 0;
293 if (sub_dlen == 0 || st->ckp_hdr_got != sizeof(st->ckp_hdr)) {
294 return 0;
295 }
296 if ((sub_reqid == kXR_write || sub_reqid == kXR_pgwrite)
297 && sub_dlen <= 16u * 1024 * 1024) /* MAX_WRITE_PAYLOAD */
298 {
299 st->payload_left = sub_dlen;
300 return 1;
301 }
302 if (sub_reqid == kXR_writev
303 && sub_dlen % 16 == 0
304 && sub_dlen <= 1024 * 16) /* MAXSEGS * SEGSIZE */
305 {
306 st->payload_left = sub_dlen;
307 st->wv_active = 1;
308 st->wv_desc_got = 0;
309 st->wv_extra = 0;
310 return 1;
311 }
312 return 0;
313}

315/*
316 * WHAT: at a completed writev descriptor block, extend payload_left to cover the
317 * trailing segment data (sum(wlen) bytes).
318 * WHY: writev frames only its descriptors in dlen; the segment data streams
319 * after and must be consumed to realign on the next header.
320 * HOW: clear wv_active; if any wlen total was summed, set payload_left to it and
321 * return 1 (caller must `continue`), else return 0.
322 */
323static int
324tap_payload_extend_writev(brix_tap_stream_t *st)
325{
326 if (!st->wv_active) {
327 return 0;
328 }
329 st->wv_active = 0;
330 if (st->wv_extra > 0) {
331 st->payload_left = st->wv_extra;
332 return 1;
333 }
334 return 0;
335}

337/*
338 * WHAT: the current frame's payload is fully consumed — apply the stock-framing
339 * extensions, or close the frame back to header state.
340 * WHY: ckpXeq and writev extend the byte run past dlen; only when no extension
341 * applies is the frame truly done and (defensively) emitted.
342 * HOW: try the ckpXeq then writev extension (each returns 1 to keep consuming);
343 * otherwise emit if not already emitted and reset to header collection.
344 * Returns 1 if the loop should `continue`, 0 if the frame closed.
345 */
346static int
347tap_payload_on_frame_end(brix_tap_stream_t *st)
348{
349 /* ckpXeq embedded header done: the sub-request body (sub_dlen bytes)
350 * streams next. Mirror the server framing (brix_ckpxeq_body_extra): only
351 * extend for the embeds and bounds the server itself accepts — anything
352 * else makes the server drop the link, which tears the relay down too.
353 * An embedded writev extends by its descriptor block and arms the wv_*
354 * machinery, which then extends again by sum(wlen) like a standalone one. */
355 if (tap_payload_extend_ckpxeq(st)) {
356 return 1;
357 }
358
359 /* Descriptors done: the segment data (sum(wlen) bytes) streams next —
360 * extend consumption instead of closing the frame. */
361 if (tap_payload_extend_writev(st)) {
362 return 1;
363 }
364
365 if (!st->emitted) { /* defensive: emit even if no path */
366 brix_tap_emit(st->tap, &st->cur, st->dir, NULL, 0);
367 }
368 st->in_payload = 0;
369 st->hdr_got = 0;
370 return 0;
371}

373/*
374 * WHAT: consume the next `take` payload bytes of the current frame.
375 * WHY: payload bytes are captured (path/ckp/wv) and skipped, the frame emits
376 * once its awaited prefix is in, and the frame's end triggers the framing
377 * extensions — this is the per-chunk payload step of the state machine.
378 * HOW: capture, advance buf/len and payload_left by `take`, emit-if-ready,
379 * and on payload exhaustion run the frame-end logic; return that helper's
380 * continue signal (0 while the frame is still filling).
381 */
382static int
383tap_feed_payload(brix_tap_stream_t *st, const uint8_t **buf, size_t *len)
384{
385 size_t take = (*len < st->payload_left) ? *len : (size_t) st->payload_left;
386
387 tap_payload_capture(st, *buf, take);
388
389 *buf += take;
390 *len -= take;
391 st->payload_left -= take;
392
393 tap_payload_emit_if_ready(st);
394
395 if (st->payload_left == 0) {
396 return tap_payload_on_frame_end(st);
397 }
398 return 0;
399}

401void
402brix_tap_stream_feed(brix_tap_stream_t *st, const uint8_t *buf, size_t len)
403{
404 while (len > 0) {
405 if (st->preamble_skip > 0) { /* skip the 20-byte handshake */
406 tap_feed_preamble(st, &buf, &len);
407 continue;
408 }
409 if (!st->in_payload) {
410 tap_feed_header(st, &buf, &len);
411 continue;
412 }
413 /* in payload: capture path bytes (if any left), then skip the rest */
414 tap_feed_payload(st, &buf, &len);
415 }
