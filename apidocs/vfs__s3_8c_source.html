30/*
31 * s3_part_size_from_env — load the effective MPU part size.
32 *
33 * WHAT: returns S3_PART_MAX_DEFAULT unless S3_PART_MAX_OVERRIDE is set in the
34 * environment, in which case that value is used (minimum 1 byte).
35 * WHY: tests need to exercise the multipart code path with small objects; the
36 * env override lets a test set part_size=512 without recompiling.
37 * HOW: getenv + strtoll; fall back to default on absent/invalid values.
38 */
39int64_t
40s3_part_size_from_env(void)
41{
42 const char *e = getenv(S3_PART_MAX_ENV);
43 int64_t v;
44
45 if (e == NULL || e[0] == '\0') {
46 return S3_PART_MAX_DEFAULT;
47 }
48 v = strtoll(e, NULL, 10);
49 return (v > 0) ? v : S3_PART_MAX_DEFAULT;
50}

55{
56 switch (e) {

80/*
81 * s3_open_read — open a handle for reading.
82 *
83 * WHAT: sets the read-mode-specific field (lazy obj_size); common fields
84 * (host/port/tls/key_path/creds/ops/caps) are already set by s3_be_open.
85 * WHY: deferred HEAD avoids a round-trip for callers that only write or never
86 * call fstat.
87 * HOW: set obj_size = -1; base ops/caps are set by s3_be_open after dispatch.
88 */
89int
90s3_open_read(vfs_s3_file *sf, brix_status *st)
91{
92 char errbuf[256] = "";
93 sd_s3_open_params p;
94
95 sf->obj_size = -1; /* loaded lazily by s3_fstat → s3_load_size */
96
97 /* Read goes through the shared S3 driver (src/fs/backend/sd_s3.c) over the
98 * client's HTTP transport: HEAD-size + Range-GET live there, once. */
99 memset(&p, 0, sizeof(p));
100 p.host = sf->host;
101 p.port = sf->port;
102 p.tls = sf->tls;
103 p.key = sf->key_path;
104 p.ak = sf->ak;
105 p.sk = sf->sk;
106 p.region = sf->region;
107 p.transport = &brix_s3_http_transport;
108 p.tctx = NULL;
109 p.timeout_ms = S3_REQ_TIMEOUT_MS;
110
111 sf->sd = sd_s3_open_read(&p, errbuf, sizeof(errbuf));
112 if (sf->sd == NULL) {
113 brix_status_set(st, XRDC_EIO, 0, "s3 open read: %s", errbuf);
114 return -1;
115 }
116 return 0;
117}

122/*
123 * s3_be_open — allocate a vfs_s3_file and prepare the handle.
124 *
125 * WHAT: parses the URL; reads credentials; chooses single-PUT or MPU write mode;
126 * returns a ready handle in *out.
127 * WHY: single entry point for the S3 backend; hides URL/mode routing from the
128 * façade.
129 * HOW: brix_weburl_parse; s3_alloc_handle; route to s3_open_read /
130 * s3_open_write_single / s3_open_write_mpu based on flags + expected_size.
131 * On any failure, free sf and return -1.
132 */
133int
134s3_be_open(const brix_vfs_backend *be, const char *url, int flags,
135 const brix_vfs_open_opts *opts, brix_vfs_file **out,
136 brix_status *st)
137{
138 brix_weburl wu;
139 vfs_s3_file *sf;
140 int rc;
141
142 (void) be;
143
144 *out = NULL;
145
146 if (brix_weburl_parse(url, &wu) != 0 || !wu.is_s3) {
147 brix_status_set(st, XRDC_EUSAGE, 0,
148 "s3 backend: not an S3 URL: %s", url);
149 return -1;
150 }
151
152 sf = s3_alloc_handle();
153 if (sf == NULL) {
154 brix_status_set(st, XRDC_EIO, ENOMEM,
155 "s3 open: out of memory allocating handle");
156 return -1;
157 }
158 snprintf(sf->host, sizeof(sf->host), "%s", wu.host);
159 sf->port = wu.port;
160 sf->tls = wu.tls;
161 snprintf(sf->key_path, sizeof(sf->key_path), "%s", wu.path);
162 s3_creds_load(sf, opts);
163
164 if (flags & XRDC_VFS_WRITE) {
165 /* Write goes through the shared S3 driver (single-PUT vs MPU decided
166 * there from expected_size + part_size). */
167 char errbuf[256] = "";
168 sd_s3_open_params p;
169 int64_t exp = (opts != NULL) ? opts->expected_size : -1;
170
171 memset(&p, 0, sizeof(p));
172 p.host = sf->host;
173 p.port = sf->port;
174 p.tls = sf->tls;
175 p.key = sf->key_path;
176 p.ak = sf->ak;
177 p.sk = sf->sk;
178 p.region = sf->region;
179 p.transport = &brix_s3_http_transport;
180 p.tctx = NULL;
181 p.timeout_ms = S3_REQ_TIMEOUT_MS;
182
183 errno = 0; /* the driver sets errno at its error sites; see below */
184 sf->sd = sd_s3_open_write(&p, exp, s3_part_size_from_env(),
185 errbuf, sizeof(errbuf));
186 if (sf->sd == NULL) {
187 int e = errno;
188 brix_status_set(st, s3_brix_from_errno(e), e,
189 "s3 open write: %s", errbuf);
190 rc = -1;
191 } else {
192 sf->is_write = 1;
193 rc = 0;
194 }
195 } else {
196 rc = s3_open_read(sf, st);
197 }
198
199 if (rc != 0) {
200 free(sf);
201 return -1;
202 }
203
204 sf->base.ops = &s_s3_ops;
205 sf->base.caps = 0; /* S3: no RANDOM_WRITE, no TRUNCATE, no ATOMIC_TEMP */
206
207 *out = &sf->base;
208 return 0;
209}

212/*
213 * s3_be_stat — stat an S3 object URL without opening a handle.
214 *
215 * WHAT: allocates a temporary handle, issues HEAD, fills *out, frees the handle.
216 * WHY: allows existence/size checks without a full open (mirrors posix_be_stat).
217 * HOW: s3_alloc_handle; parse URL; s3_load_size; fill brix_vfs_stat; free.
218 * ENOENT-equivalent (404) → exists=0, return 0; other errors → -1.
219 */
220int
221s3_be_stat(const brix_vfs_backend *be, const char *url,
222 brix_vfs_stat *out, brix_status *st)
223{
224 brix_weburl wu;
225 vfs_s3_file *sf;
226 int rc;
227
228 (void) be;
229
230 if (brix_weburl_parse(url, &wu) != 0 || !wu.is_s3) {
231 brix_status_set(st, XRDC_EUSAGE, 0,
232 "s3 stat: not an S3 URL: %s", url);
233 return -1;
234 }
235
236 sf = s3_alloc_handle();
237 if (sf == NULL) {
238 brix_status_set(st, XRDC_EIO, ENOMEM,
239 "s3 stat: out of memory");
240 return -1;
241 }
242 snprintf(sf->host, sizeof(sf->host), "%s", wu.host);
243 sf->port = wu.port;
244 sf->tls = wu.tls;
245 snprintf(sf->key_path, sizeof(sf->key_path), "%s", wu.path);
246
247 /* Use no-op open_opts for stat (no cred store yet). */
248 {
249 brix_vfs_open_opts dummy_opts;
250 memset(&dummy_opts, 0, sizeof(dummy_opts));
251 dummy_opts.expected_size = -1;
252 s3_creds_load(sf, &dummy_opts);
253 }
254
255 /* Create the shared read driver handle (s3_load_size delegates to it). */
256 if (s3_open_read(sf, st) != 0) {
257 free(sf);
258 return -1;
259 }
260 rc = s3_load_size(sf, st);
261 if (rc != 0) {
262 /* 404 → exists=0 (not an error at the stat level) */
263 if (st->kxr == XRDC_ENOENT) {
264 out->exists = 0;
265 out->size = 0;
266 out->mtime = 0;
267 out->is_dir = 0;
268 sd_s3_close(sf->sd);
269 free(sf);
270 brix_status_clear(st);
271 return 0;
272 }
273 sd_s3_close(sf->sd);
274 free(sf);
275 return -1;
276 }
277
278 out->size = sf->obj_size;
279 out->mtime = 0;
280 out->is_dir = 0;
281 out->exists = 1;
282 sd_s3_close(sf->sd);
283 free(sf);
284 return 0;
285}
