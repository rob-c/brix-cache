1/*
2 * sd_s3_transport.h — pluggable HTTP transport for the shared S3 storage driver.
3 *
4 * WHAT: The narrow HTTP request/response interface the S3 driver (sd_s3.c) needs,
5 * so the S3 *protocol* logic (request building, SigV4, range GET, multipart
6 * upload, XML) can live once in src/fs/backend/ while the actual HTTP
7 * transport is injected per consumer.
8 * WHY: The S3 backend was client-only because it was welded to the client's
9 * hand-rolled HTTP/1.1 stack (xrdc_http_*). src/ cannot depend on
10 * client/lib, and the server has no HTTP-client substrate. Injecting the
11 * transport breaks that coupling: the userland client fills it with
12 * xrdc_http; a future server consumer (e.g. an S3 cache origin) fills it
13 * with its own. The driver code is shared either way.
14 * HOW: Pure C, ngx-free. One synchronous request op + response accessors. The
15 * response carries a transport-private `opaque` handle (the consumer's real
16 * response object); the driver only ever reaches it through these accessors
17 * and frees it via resp_free.
18 */
19#ifndef BRIX_SD_S3_TRANSPORT_H
20#define BRIX_SD_S3_TRANSPORT_H

23#include <sys/types.h> /* ssize_t */

25/* A transport-agnostic response handle. status is the HTTP status code; opaque
26 * is the transport's own response object, reached only via the accessors below. */
27typedef struct {
28 int status;
29 void *opaque;
30} brix_s3_resp_t;
31
32typedef struct brix_s3_transport_s {
33 /* Perform one HTTP request. `path_and_query` is the already-encoded
34 * "/key?canon-query" (or "/key"); `headers` is the pre-built header block
35 * (the SigV4 Authorization + x-amz-* lines). On success fills *resp and
36 * returns 0; on a transport-level failure returns -1 and writes a message
37 * into errbuf[errcap]. A non-2xx HTTP status is NOT a failure here — it is
38 * reported via resp->status for the driver to map. */
39 int (*request)(void *tctx,
40 const char *host, int port, int tls,
41 const char *method, const char *path_and_query,
42 const char *headers, const void *body, size_t body_len,
43 int timeout_ms, brix_s3_resp_t *resp,
44 char *errbuf, size_t errcap);
45
46 /* OPTIONAL — like `request`, plus a per-request TLS client certificate
47 * (mutual-TLS) so a caller can authenticate the upstream leg AS the end
48 * user (phase-70 §5.1 x509 pass-through/select over an https backend leg).
49 *
50 * `client_cert_pem` is a filesystem PATH to a PEM file holding the user's
51 * proxy — certificate chain AND private key in one file (the same on-disk
52 * form the GSI origin presenter consumes, see origin_auth.c). When non-NULL
53 * on a `tls` request the transport presents it as the client cert on the
54 * handshake; NULL means "no client cert" (identical to `request`).
55 *
56 * A transport that cannot present a client cert leaves this slot NULL; the
57 * sd_http driver treats a NULL slot as "GSI-over-https unsupported" and
58 * (in deny mode) refuses rather than silently falling back to anonymous. */
59 int (*request_cred)(void *tctx,
60 const char *host, int port, int tls,
61 const char *method, const char *path_and_query,
62 const char *headers, const void *body, size_t body_len,
63 int timeout_ms, const char *client_cert_pem,
64 brix_s3_resp_t *resp, char *errbuf, size_t errcap);
65
66 /* Copy response header `name` into out[outcap] (NUL-terminated). 0 if found,
67 * -1 if absent. */
68 int (*resp_header)(const brix_s3_resp_t *resp, const char *name,
69 char *out, size_t outcap);
70
71 /* OPTIONAL — the raw response header block ("Name: value" lines separated
72 * by CRLF/LF, NUL-terminated; valid until resp_free). Lets the driver
73 * ENUMERATE headers (generic x-amz-meta-* listxattr), which the by-name
74 * resp_header lookup cannot. A transport that does not retain the raw
75 * block leaves this NULL; the driver must then report enumeration as
76 * unsupported (ENOTSUP), never guess names. */
77 const char *(*resp_headers_raw)(const brix_s3_resp_t *resp);
78
79 /* The response body bytes + length (valid until resp_free). */
80 const void *(*resp_body)(const brix_s3_resp_t *resp, size_t *len);
81
82 /* Release the transport-private response. */
83 void (*resp_free)(brix_s3_resp_t *resp);
84} brix_s3_transport_t;
