79/*
80 * s3_put_streaming — decode an aws-chunked PUT/UploadPart body and finalize.
81 *
82 * Reads x-amz-decoded-content-length, then either offloads the chunk decode to
83 * the thread pool (so the blocking pread/pwrite state machine does not stall the
84 * event loop) or, when no pool is configured, decodes synchronously. Either way
85 * the commit/ETag/checksum/tagging/200 run on the event loop (s3_chunk_finalize).
86 */
87void
88s3_put_streaming(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
89 brix_vfs_staged_t *staged, const char *fs_path, ngx_uint_t body_mode)
90{
91 const char *root_canon = cf->common.root_canon;
92 ngx_table_elt_t *dcl;
93 uint64_t expected = 0;
94 ngx_thread_pool_t *pool;
95 size_t i;
96
97 dcl = brix_http_find_header(r, "x-amz-decoded-content-length",
98 sizeof("x-amz-decoded-content-length") - 1);
99 if (dcl == NULL || dcl->value.len == 0) {
100 brix_vfs_staged_abort(staged, 1);
101 s3_put_finalize_client_error(r, NGX_HTTP_BAD_REQUEST,
102 "MissingContentLength",
103 "Streaming upload is missing x-amz-decoded-content-length.");
104 return;
105 }
106 for (i = 0; i < dcl->value.len; i++) {
107 u_char d = dcl->value.data[i];
108 if (d < '0' || d > '9') {
109 brix_vfs_staged_abort(staged, 1);
110 s3_put_finalize_client_error(r, NGX_HTTP_BAD_REQUEST,
111 "InvalidArgument",
112 "x-amz-decoded-content-length is not a valid length.");
113 return;
114 }
115 expected = expected * 10 + (uint64_t) (d - '0');
116 }
117
118 pool = s3_thread_pool(cf);
119 if (pool != NULL) {
120 ngx_thread_task_t *task =
121 ngx_thread_task_alloc(r->pool, sizeof(s3_chunk_aio_t));
122 s3_chunk_aio_t *t;
123
124 if (task == NULL) {
125 brix_vfs_staged_abort(staged, 1);
126 s3_put_finalize_error(r);
127 return;
128 }
129 t = task->ctx; /* ngx_thread_task_alloc zeroes ctx */
130 t->r = r;
131 t->staged = staged;
132 t->expected = expected;
133 t->body_mode = body_mode;
134 s3_build_chunk_verify(r, cf, &t->verify); /* W6a */
135 ngx_cpystrn((u_char *) t->fs_path, (u_char *) fs_path,
136 sizeof(t->fs_path));
137 ngx_cpystrn((u_char *) t->root_canon, (u_char *) root_canon,
138 sizeof(t->root_canon));
139 brix_task_bind(task, s3_chunk_aio_thread, s3_chunk_aio_done);
140
141 if (ngx_thread_task_post(pool, task) != NGX_OK) {
142 brix_vfs_staged_abort(staged, 1);
143 s3_put_finalize_error(r);
144 return;
145 }
146 r->main->count++;
147 return;
148 }
149
150 /* Synchronous fallback (no thread pool): decode inline, then finalize. */
151 {
152 s3_chunk_trailer_t trailer;
153 s3_chunk_verify_t verify;
154 int status = NGX_HTTP_INTERNAL_SERVER_ERROR;
155
156 s3_build_chunk_verify(r, cf, &verify); /* W6a */
157 if (s3_aws_chunked_decode_to_fd(r, brix_vfs_staged_fd(staged), fs_path, expected,
158 &trailer, &status, NULL, &verify)
159 != NGX_OK)
160 {
161 s3_chunk_decode_failed(r, root_canon, staged, status);
162 return;
163 }
164 s3_chunk_finalize(r, root_canon, fs_path, staged, &trailer, expected,
165 body_mode);
166 }
167}

189/* Directory-sentinel fast path: if fs_path names an S3 directory sentinel
190 * (S3_DIR_SENTINEL), create the parent directory plus the zero-byte sentinel,
191 * finalize the request, and return 1 (the caller must return). Returns 0 when
192 * fs_path is a normal object, so the caller proceeds with the body write. */
193static int
194s3_put_try_sentinel(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
195 const u_char *fs_path)
196{
197 size_t plen = ngx_strlen(fs_path);
198 size_t slen = sizeof(S3_DIR_SENTINEL) - 1;
199 char parent[PATH_MAX];
200 char *slash;
201 int fd;
202 char identity[128];
203
204 if (!(plen >= slen
205 && ngx_strncmp(fs_path + plen - slen,
206 (u_char *) S3_DIR_SENTINEL, slen) == 0))
207 {
208 return 0;
209 }
210
211 /* Create the directory that the sentinel lives in. */
212 if (plen >= sizeof(parent)) {
213 s3_put_finalize_error(r);
214 return 1;
215 }
216 memcpy(parent, fs_path, plen + 1);
217 slash = strrchr(parent, '/');
218 if (slash != NULL) {
219 *slash = '\0';
220 }
221 {
222 brix_vfs_ctx_t pctx;
223
224 s3_build_vfs_ctx(r, parent, cf, &pctx);
225 if (brix_vfs_mkdir(&pctx, 0755, 0 /* no parents */) != NGX_OK
226 && errno != EEXIST)
227 {
228 int mk_errno = errno; /* capture before logging clobbers it */
229 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, mk_errno,
230 "s3: mkdir(\"%s\") failed", parent);
231 s3_put_finalize_fs_error(r, mk_errno);
232 return 1;
233 }
234 }
235
236 /* Write the zero-byte sentinel (confined create via the VFS seam). */
237 fd = brix_vfs_open_fd(r->connection->log, cf->common.root_canon,
238 (const char *) fs_path,
239 O_WRONLY | O_CREAT | O_TRUNC, 0644);
240 if (fd < 0) {
241 int open_errno = errno; /* capture before logging clobbers it */
242 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, open_errno,
243 "s3: open(\"%s\") for sentinel failed",
244 (const char *) fs_path);
245 s3_put_finalize_fs_error(r, open_errno);
246 return 1;
247 }
248 close(fd);
249
250 s3_dashboard_identity(r, cf, identity, sizeof(identity));
251 (void) brix_dashboard_http_start_identity(r, (const char *) fs_path,
252 identity, "", BRIX_XFER_PROTO_S3, BRIX_XFER_DIR_WRITE,
253 "PutObjectSentinel", 0);
254 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_DIR_SENTINEL]);
255 BRIX_S3_METRIC_INC(put_body_total[BRIX_S3_PUT_EMPTY]);
256 s3_put_finalize_empty_ok(r);
257 return 1;
258}

261void
262s3_put_body_inner(ngx_http_request_t *r)
263{
264 u_char *fs_path;
265 ngx_http_s3_loc_conf_t *cf;
266 ngx_http_s3_req_ctx_t *s3ctx;
267 brix_codec_id_t put_codec = BRIX_CODEC_IDENTITY;
268 size_t body_bytes = 0;
269 ngx_uint_t body_mode = BRIX_S3_PUT_EMPTY;
270 brix_vfs_staged_t *staged = NULL;
271 brix_vfs_ctx_t svctx;
272 int staged_err = 0;
273 brix_http_body_summary_t body_summary;
274 char identity[128];
275
276 cf = ngx_http_get_module_loc_conf(r, ngx_http_brix_s3_module);
277 s3ctx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
278 fs_path = s3ctx != NULL ? (u_char *) s3ctx->fs_path : NULL;
279
280 if (fs_path == NULL) {
281 s3_put_finalize_error(r);
282 return;
283 }
284
285 /* Directory-sentinel objects take a separate create-dir-and-return path. */
286 if (s3_put_try_sentinel(r, cf, fs_path)) {
287 return;
288 }
289
290 /* Create parent directories if needed using the shared confined mkdir helper. */
291 {
292 char parent[PATH_MAX];
293 char *last_slash;
294 size_t flen = strlen((const char *) fs_path);
295
296 if (flen < sizeof(parent)) {
297 memcpy(parent, fs_path, flen + 1);
298 last_slash = strrchr(parent, '/');
299 if (last_slash && last_slash != parent) {
300 brix_vfs_ctx_t pctx;
301 brix_vfs_stat_t pst;
302
303 *last_slash = '\0';
304
305 /*
306 * phase-46 W2a: fast-path the common "many objects into one
307 * prefix" pattern. A single confined probe that finds an
308 * existing directory replaces the per-path-component
309 * mkdirat(EEXIST) storm of the recursive mkdir. A symlink or a
310 * miss falls through to the confined vfs_mkdir, which re-enforces
311 * confinement.
312 */
313 s3_build_vfs_ctx(r, parent, cf, &pctx);
314 if (brix_vfs_probe(&pctx, 1 /* no-follow */, &pst) == NGX_OK
315 && pst.is_directory)
316 {
317 /* parent prefix already exists — nothing to create */
318 } else {
319 brix_vfs_ctx_t pvctx;
320
321 s3_build_vfs_ctx(r, parent, cf, &pvctx);
322 if (brix_vfs_mkdir(&pvctx, 0755, 1 /* parents */) != NGX_OK
323 && errno != EEXIST) {
324 int mk_errno = errno; /* capture before logging clobbers it */
325 brix_log_safe_path(r->connection->log, NGX_LOG_ERR,
326 mk_errno,
327 "s3: mkdirs_for(\"%s\") failed",
328 (const char *) fs_path);
329 /* DAC-denied create of a parent dir → 403, not 500. */
330 s3_put_finalize_fs_error(r, mk_errno);
331 return;
332 }
333 }
334 }
335 }
336 }
337
338 /* A transient stack ctx is fine: brix_vfs_staged_open deep-copies the ctx
339 * (and the strings it points at) onto r->pool, so the handle survives the async
340 * body-write completion (put_aio/put_chunk commit after this function returns). */
341 s3_build_vfs_ctx(r, (const char *) fs_path, cf, &svctx);
342 staged = brix_vfs_staged_open(&svctx, 0600, 16, &staged_err);
343 if (staged == NULL)
344 {
345 int open_errno = staged_err; /* the VFS open captured errno */
346 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, open_errno,
347 "s3: staged open for \"%s\" failed",
348 (const char *) fs_path);
349 /* A DAC-denied / confinement-blocked create is a 403, not a 500. */
350 s3_put_finalize_fs_error(r, open_errno);
351 return;
352 }
353
354 if (brix_http_body_summary(r, &body_summary) != NGX_OK) {
355 brix_vfs_staged_abort(staged, 1);
356 s3_put_finalize_error(r);
357 return;
358 }
359
360 body_bytes = body_summary.bytes;
361 body_mode = s3_put_body_mode(&body_summary);
362 s3_dashboard_identity(r, cf, identity, sizeof(identity));
363 (void) brix_dashboard_http_start_identity(r, (const char *) fs_path,
364 identity, "", BRIX_XFER_PROTO_S3, BRIX_XFER_DIR_WRITE,
365 s3_dashboard_put_op(r), (int64_t) body_bytes);
366
367 /*
368 * AWS streaming upload (aws-chunked): the body carries chunk framing that
369 * must be decoded to the object bytes — storing it verbatim corrupts every
370 * object from a default modern SDK/CLI client. Routed through its own
371 * decode → commit → checksum → finalize path.
372 */
373 if (s3_body_is_aws_chunked(r)) {
374 /* A combined Content-Encoding such as "aws-chunked,gzip" means the body
375 * was compressed THEN chunk-framed. The streaming de-chunker strips only
376 * the chunk envelope, so the inner-compressed payload would be stored
377 * still compressed (silent object corruption). Reject rather than store
378 * undecoded bytes — the same invariant the non-chunked path enforces with
379 * 400 below. Plain aws-chunked (no inner coding) continues normally. */
380 if (s3_aws_chunked_has_inner_coding(r)) {
381 brix_vfs_staged_abort(staged, 1);
382 s3_put_finalize_codec_error(r, NGX_HTTP_BAD_REQUEST);
383 return;
384 }
385 s3_put_streaming(r, cf, staged, (const char *) fs_path, body_mode);
386 return;
387 }
388
389 {
390 ngx_table_elt_t *ce = brix_http_find_header(
391 r, "Content-Encoding", sizeof("Content-Encoding") - 1);
392 if (ce != NULL && ce->value.len > 0) {
393 const brix_codec_desc_t *d = brix_codec_by_http_token(
394 (const char *) ce->value.data, ce->value.len);
395 if (d == NULL || !d->available) {
396 /* Unknown / not-compiled-in Content-Encoding: never store the
397 * undecoded bytes — reject as a client error (400), not 500. */
398 brix_vfs_staged_abort(staged, 1);
399 s3_put_finalize_codec_error(r, NGX_HTTP_BAD_REQUEST);
400 return;
401 }
402 put_codec = d->id;
403 }
404 }
405
406 /*
407 * phase-46 W1a: offload the body write to the thread pool for ALL plain
408 * (identity-encoded) bodies — including spooled and mixed ones, not just
409 * in-memory. The worker fn (s3_put_aio_thread) already streams every buf,
410 * memory and file-backed, via brix_http_body_write_to_fd (kernel
411 * copy_file_range for the spooled nginx temp file), so large uploads — the
412 * ones nginx spools to disk — stop blocking the event loop. Content-Encoding
413 * decode (put_codec != IDENTITY) and aws-chunked (handled above) still run
414 * synchronously.
415 */
416 if (body_summary.bytes > 0
417 && put_codec == BRIX_CODEC_IDENTITY && s3_thread_pool(cf) != NULL)
418 {
419 ngx_thread_task_t *task;
420 s3_put_aio_t *t;
421
422 task = ngx_thread_task_alloc(r->pool, sizeof(s3_put_aio_t));
423 if (task == NULL) {
424 brix_vfs_staged_abort(staged, 1);
425 s3_put_finalize_error(r);
426 return;
427 }
428
429 t = task->ctx;
430 t->r = r;
431 t->staged = staged;
432 t->len = body_summary.bytes;
433 t->body_mode = body_mode;
434 t->body_bytes = body_bytes;
435 ngx_cpystrn((u_char *) t->final_path, fs_path, sizeof(t->final_path));
436 ngx_cpystrn((u_char *) t->root_canon,
437 (u_char *) cf->common.root_canon, sizeof(t->root_canon));
438
439 /*
440 * Phase 31 W2: no full-body collection — the thread streams the
441 * in-memory body bufs straight to the staged fd (s3_put_aio_thread).
442 */
443
444 brix_task_bind(task, s3_put_aio_thread, s3_put_aio_done);
445
446 if (ngx_thread_task_post(cf->common.thread_pool, task) != NGX_OK) {
447 brix_vfs_staged_abort(staged, 1);
448 s3_put_finalize_error(r);
449 return;
450 }
451
452 r->main->count++;
453 return;
454 }
455
456 {
457 ngx_int_t wrc;
458 ngx_int_t decode_status = 0;
459
460 if (put_codec != BRIX_CODEC_IDENTITY) {
461 wrc = brix_http_body_decode_to_fd(r, brix_vfs_staged_fd(staged),
462 (const char *) fs_path,
463 put_codec,
464 BRIX_DECODE_MAX_OUTPUT,
465 &body_summary,
466 &decode_status);
467 } else {
468 wrc = brix_http_body_write_to_fd(r, brix_vfs_staged_fd(staged),
469 (const char *) fs_path,
470 &body_summary);
471 }
472 if (wrc != NGX_OK) {
473 brix_vfs_staged_abort(staged, 1);
474 /* Map a decode failure to a clean S3 client error (413 bomb / 400
475 * malformed); a genuine I/O failure (decode_status 0 or 5xx) stays
476 * a 500. */
477 if (decode_status == NGX_HTTP_REQUEST_ENTITY_TOO_LARGE
478 || decode_status == NGX_HTTP_BAD_REQUEST
479 || decode_status == NGX_HTTP_UNSUPPORTED_MEDIA_TYPE)
480 {
481 s3_put_finalize_codec_error(r, decode_status
482 == NGX_HTTP_REQUEST_ENTITY_TOO_LARGE
483 ? NGX_HTTP_REQUEST_ENTITY_TOO_LARGE
484 : NGX_HTTP_BAD_REQUEST);
485 } else {
486 s3_put_finalize_error(r);
487 }
488 return;
489 }
490 }
491
492 if (s3_commit_put(r, r->connection->log, cf->common.root_canon, staged,
493 (const char *) fs_path) != NGX_OK)
494 {
495 if (s3_put_commit_conflict(r)) {
496 return;
497 }
498 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, ngx_errno,
499 "s3: staged commit to \"%s\" failed",
500 (const char *) fs_path);
501 s3_put_finalize_error(r);
502 return;
503 }
504
505 /* S3 API requires ETag on PutObject 200 response */
506 {
507 brix_vfs_ctx_t fctx;
508 brix_vfs_stat_t fst;
509 char etag_buf[48];
510
511 s3_build_vfs_ctx(r, (const char *) fs_path, cf, &fctx);
512 if (brix_vfs_probe(&fctx, 1 /* no-follow */, &fst) == NGX_OK) {
513 struct stat final_sb;
514
515 ngx_memzero(&final_sb, sizeof(final_sb));
516 final_sb.st_mtime = fst.mtime;
517 final_sb.st_size = fst.size;
518 s3_etag(&final_sb, etag_buf, sizeof(etag_buf));
519 (void) s3_set_header(r, "ETag", etag_buf);
520 }
521 }
522
523 /* Full-object checksum verify (client-supplied) + echo; failures remove the
524 * object and send the matching 400. */
525 if (s3_put_checksum_failed(r, (const char *) fs_path,
526 cf->common.root_canon))
527 {
528 return;
529 }
530
531 (void) s3_apply_put_tagging_header(r, (const char *) fs_path,
532 cf->common.root_canon);
533 (void) s3_apply_put_user_metadata(r, (const char *) fs_path,
534 cf->common.root_canon);
535 s3_put_finalize_ok(r, body_bytes, body_mode);
536}
