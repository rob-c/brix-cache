1/*
2 * core/types/ctx_structs.h
3 *
4 * Per-connection context sub-struct helper types, grouped out of the main
5 * brix_ctx_t definition (context.h) so the connection state reads as a set of
6 * named concern groups rather than a flat wall of ~100 fields. Every field is
7 * reached as ctx-><group>.<field> (e.g. ctx->gsi.sess_key, ctx->recv.payload,
8 * ctx->out.pipeline_depth).
9 *
10 * NOT self-contained: included by context.h at the point right before the
11 * brix_ctx_t definition, AFTER the helper slot structs (brix_resp_slot_t,
12 * brix_read_slot_t) and after the translation unit has pulled in state.h,
13 * file.h, tunables.h, token/token.h, metrics/metrics.h, and nginx/OpenSSL
14 * headers — exactly where these fields previously lived inline. Do not include
15 * it directly — include context.h.
16 */
17

24/* XrdSecpwd (Phase 52 WS-B) multi-round handshake state. Round 1 derives the
25 * DH session key from the client's kXRS_puk; round 2 decrypts the credential
26 * with it, so the ephemeral DH keypair need not survive between rounds. */
27typedef struct {
28 uint8_t session_key[16]; /* aes-128 DH session key (round 1) */
29 unsigned round; /* rounds seen (1 = puk-exchange done, awaiting creds) */
30 char user[64]; /* username asserted in round 1, verified in round 2 */
31} brix_ctx_pwd_t;

50/* Phase 39 steady-state network-fault deadlines. *_armed track whether THIS
51 * module armed c->read / c->write's timer (so arm/disarm are idempotent);

53 * avoid a srv_conf lookup. 0 = the corresponding deadline is disabled. */

93/* kXR_sigver request-signing state (GSI sessions). The client wraps each
94 * request in a kXR_sigver envelope carrying an HMAC-SHA256 over the next
95 * request's header (and optionally payload); the replay guard requires seqno >
96 * last_seqno. signing_key = SHA-256(DH-shared-secret) set at kXGC_cert. */
97typedef struct {
98 u_char signing_key[32]; /* HMAC-SHA256 key (SHA-256 of DH secret) */
99 int signing_active; /* 1 = signing_key is valid and in use */
100 uint64_t last_seqno; /* highest seqno accepted so far */
101 int pending; /* 1 = next dispatch must verify the HMAC */
102 int verified; /* 1 = current request passed sigver verification */
103 uint16_t expectrid; /* the opcode the sigver envelope covers */
104 uint64_t seqno; /* seqno from the kXR_sigver frame */
105 int nodata; /* 1 = payload was excluded from the HMAC */
106 u_char hmac[32]; /* expected HMAC bytes to check against */
107 EVP_MAC *mac; /* cached OpenSSL HMAC provider handle */
108 EVP_MAC_CTX *mac_ctx; /* reusable HMAC context for signed reqs */
109} brix_ctx_sigver_t;

111/* GSI handshake + X.509 proxy delegation state. dh_key is the ephemeral DH key
112 * (kXGS_cert), freed after the shared secret is derived. signed_dh selects the
113 * signed vs unsigned DH wire form (phase-48). The deleg_* fields capture the
114 * client's proxy during login for a later TPC pull (phase-57 §F6). */
115typedef struct {
116 EVP_PKEY *dh_key; /* DH key (kXGS_cert), freed after secret derived */
117 int signed_dh; /* 0 = unsigned DH (default), 1 = signed DH (>=10400) */
118 char sess_cipher[24]; /* negotiated session cipher name */
119 u_char sess_key[32]; /* session AES key (DH-secret derived) */
120 int sess_keylen; /* valid bytes in sess_key (0 = unset) */
121 int sess_use_iv; /* 1 = IV-prepended main (signed-DH path) */
122 EVP_PKEY *deleg_reqkey; /* fresh proxy key (build_pxyreq), or NULL */
123 uint32_t clnt_opts; /* client's kXRS_clnt_opts (kOpts*) delegation mode */
124 int deleg_await; /* 1 = sent kXGS_pxyreq, awaiting kXGC_sigpxy */
125 u_char *deleg_chain_pem; /* client chain PEM (for assemble), heap */
126 size_t deleg_chain_len;
127 u_char *deleg_proxy_pem; /* captured delegated proxy credential (PEM) */
128 size_t deleg_proxy_len;
129 u_char deleg_client_rtag[64]; /* client's kXGC_cert random tag */
130 int deleg_client_rtag_len;
131 /* phase-70 §5.1: raw client-pushed FULL proxy PEM (kXRS_x509_fullproxy)
132 * captured from the decrypted kXGC_cert inner buffer, BEFORE DN validation.
133 * auth.c validates (chain+key parse, leaf DN == authenticated DN) then
134 * promotes the bytes to ctx->deleg_proxy_pem; heap-owned, freed at cleanup. */
135 u_char *client_fullproxy_pem;
136 size_t client_fullproxy_len;
137} brix_ctx_gsi_t;

139/* Per-connection rate-limit state (Phase 25/33). bw_* = the current request's
140 * bandwidth charge target; conc_* = a per-principal concurrency slot held for the
141 * connection's lifetime; key_cache = cached identity-stable rule keys. The
142 * *_rule pointers are brix_rl_rule_t* (void to keep ratelimit.h out of this
143 * widely-included header). */
144typedef struct {
145 void *bw_rule;
146 char bw_key[128];
147 void *conc_rule; /* NULL = no concurrency slot held */
148 char conc_key[128];
149 char key_cache[BRIX_RL_RULE_CACHE_MAX][128];
150 uint32_t key_cache_valid; /* bitmask: bit i ⇒ key_cache[i] holds rule i's key */
151} brix_ctx_rl_t;

153/* Session login + authenticated-identity state. Two-step XRootD login: kXR_login
154 * sets logged_in and issues sessid; kXR_auth sets auth_done and fills the identity
155 * strings (dn/primary_vo/vo_list/peer_ip). acc_host is the opt-in XrdAcc
156 * reverse-DNS cache; gsi_counted tracks an in-flight GSI handshake slot. */
157typedef struct {
158 u_char sessid[BRIX_SESSION_ID_LEN]; /* opaque ID we issued at login */
159 ngx_flag_t logged_in; /* set when kXR_login is accepted */
160 ngx_flag_t auth_done; /* set when authentication is complete */
161 char user[9]; /* fixed-width kXR_login username, NUL-terminated */
162 uint32_t pid; /* client pid from kXR_login, host byte order */
163 uint8_t auth_fail_count; /* failed kXR_auth attempts; capped */
164 size_t pool_bytes_used; /* cumulative ngx_palloc bytes; capped */
165 char dn[512]; /* GSI subject DN */
166 char primary_vo[128]; /* first VO from the VOMS attribute cert */
167 char vo_list[512]; /* space-separated list of all VOs */
168 char peer_ip[64]; /* remote peer address for authdb HOST ('p') rules */
169 const char *acc_host; /* XrdAcc reverse-DNS host cache (points into c->pool) */
170 unsigned acc_host_done:1;
171 unsigned gsi_counted:1; /* holds a GSI in-flight handshake slot (Phase 51 E4) */
172} brix_ctx_login_t;

174/* Request receive/framing state. Read in two stages: the fixed 24-byte header
175 * into hdr_buf, then (if dlen>0) the payload into the reusable payload_buf.
176 * cur_* are the parsed header fields; cur_body_extra/extended cover the trailing
177 * body streamed after the dlen-framed payload (kXR_writev / kXR_chkpoint). */
178typedef struct {
179 u_char hdr_buf[24]; /* raw bytes of the current request header */
180 size_t hdr_pos; /* header bytes received so far */
181 u_char cur_streamid[2]; /* echoed back unchanged in every response */
182 uint16_t cur_reqid; /* opcode, host byte order */
183 u_char cur_body[16]; /* request-specific parameter bytes (wire.h) */
184 uint32_t cur_dlen; /* payload length that follows the header */
185 uint32_t cur_body_extra; /* streamed data bytes beyond cur_dlen */
186 unsigned cur_body_extended:2;/* completed extension stages (2 = done) */
187 u_char *payload; /* current request payload, NULL if none */
188 size_t payload_pos; /* bytes accumulated so far */
189 u_char *payload_buf; /* reusable receive buffer */
190 size_t payload_buf_size; /* allocated size of payload_buf */
191} brix_ctx_recv_t;

193/* Output response ring + write-pipelining state (Phase 29/32). A response is
194 * built into ring[tail] and drained from ring[head]; count = slots in use; the
195 * ring holds pipeline_depth slots (all arithmetic modulo pipeline_depth).
196 * wr_inflight bounds in-flight plain-write pwrites; the recv loop keeps reading
197 * while count + wr_inflight < pipeline_depth. */
198typedef struct {
199 ngx_uint_t pipeline_depth; /* ring capacity = configured in-flight bound */
200 brix_resp_slot_t *ring; /* [pipeline_depth] response slots */
201 ngx_uint_t head; /* slot being drained to the socket */
202 ngx_uint_t tail; /* slot currently being built */
203 ngx_uint_t count; /* number of slots in use (responses queued) */
204 unsigned recv_deferred:1; /* drain barrier: parked non-pipelinable req awaits count==0 */
205 u_char deferred_streamid[2]; /* parked request's sid: a pipelined AIO ack completing
206 * during the park clobbers recv.cur_streamid, so the
207 * deferred dispatch must reinstall its own sid */
208 unsigned resp_pipelinable:1; /* current response is a single-chunk sendfile read */
209 ngx_uint_t wr_inflight; /* plain-write pwrites posted, not yet acked */
210 unsigned resp_async:1; /* ack drains without disturbing recv */
211 unsigned finalize_pending:1; /* deferred teardown while wr_inflight > 0 */
212 ngx_int_t finalize_status; /* ngx_stream status to finalize with */
213} brix_ctx_out_t;

215/* Read pipeline + reusable read/write scratch buffers. The *_scratch buffers
216 * are raw ngx_alloc kept for the session lifetime (grown on demand). The
217 * *_aio_task are reused per-opcode thread tasks for serial memory-backed reads.
218 * pool[pipeline_depth] backs concurrent single-shot memory reads (Phase 32 WS3);
219 * win_* is the windowed-read continuation for large TLS/non-regular reads. */
220typedef struct {
221 u_char *read_scratch; /* flat data block (read/pgread) */
222 size_t read_scratch_size;
223 u_char *read_hdr_scratch; /* per-chunk response headers (readv) */
224 size_t read_hdr_scratch_size;
225 u_char *write_scratch; /* pgwrite decode buffer */
226 size_t write_scratch_size;
227 u_char *cmp_scratch; /* inline read-compression codec output (Phase-42 W4) */
228 size_t cmp_scratch_size;
229 ngx_thread_task_t *read_aio_task;
230 ngx_thread_task_t *pgread_aio_task;
231 ngx_thread_task_t *readv_aio_task;
232 brix_read_slot_t *pool; /* [pipeline_depth] in-flight read buffers */
233 ngx_uint_t inflight; /* pool entries currently in use */
234 unsigned backpressured:1; /* recv stopped admitting reads (pool full) */
235 unsigned win_active:1; /* windowed memory-read in flight */
236 int win_idx;
237 int win_fd;
238 off_t win_offset; /* next file offset to read */
239 size_t win_remaining; /* bytes still to send */
240 u_char win_streamid[2];
241} brix_ctx_rd_t;
