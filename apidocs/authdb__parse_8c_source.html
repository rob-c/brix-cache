24/* Parse an XrdAcc privilege string (e.g. "rwld") into an BRIX_PRIV_* bitmask. */
25static uint32_t
26brix_parse_privs(const char *p, size_t len)
27{
28 uint32_t privs = 0;
29 size_t i;
30
31 /* Each privilege char maps to one or more BRIX_AUTH_* bits; OR-accumulate
32 * across the whole string. 'r' implies 'l' (you cannot read what you cannot
33 * look up), and 'a' (append) is folded into UPDATE since the FS-level write
34 * permission is identical. Unknown chars are silently ignored. */
35 for (i = 0; i < len; i++) {
36 switch (p[i]) {
37 case 'r': privs |= BRIX_AUTH_READ | BRIX_AUTH_LOOKUP; break;
38 case 'l': privs |= BRIX_AUTH_LOOKUP; break;
39 case 'w': privs |= BRIX_AUTH_UPDATE; break;
40 case 'a': privs |= BRIX_AUTH_UPDATE; break; /* append is update */
41 case 'd': privs |= BRIX_AUTH_DELETE; break;
42 case 'm': privs |= BRIX_AUTH_MKDIR; break;
43 case 'k': privs |= BRIX_AUTH_ADMIN; break;
44 default: break;
45 }
46 }
47
48 return privs;
49}

61/* WHAT: read a full authdb file into a heap buffer, enforcing the size policy.
62 * WHY: isolates the file/stat/read/limit I/O so the parser proper is pure over
63 * an in-memory buffer. On the empty-file fast path returns NGX_OK with
64 * *out_buf==NULL and *out_len==0 (caller has nothing to parse).
65 * HOW: open → fstat → size-guard (0 = ok/empty, >1 MiB = reject) → alloc+read.
66 * On any error logs via cf and closes the fd; the fd is always closed here.
67 * On success *out_buf is a heap block the caller must ngx_free(). */
68static ngx_int_t
69adb_read_file(ngx_conf_t *cf, ngx_str_t *filename, u_char **out_buf,
70 size_t *out_len)
71{
72 ngx_fd_t fd;
73 ngx_file_t file;
74 ngx_file_info_t fi;
75 u_char *buf;
76 ssize_t n;
77 size_t buf_size;
78
79 *out_buf = NULL;
80 *out_len = 0;
81
82 fd = ngx_open_file(filename->data, NGX_FILE_RDONLY, NGX_FILE_OPEN, 0);
83 if (fd == NGX_INVALID_FILE) {
84 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
85 ngx_open_file_n " \"%s\" failed", filename->data);
86 return NGX_ERROR;
87 }
88
89 ngx_memzero(&file, sizeof(ngx_file_t));
90 file.fd = fd;
91 file.name = *filename;
92 file.log = cf->log;
93
94 if (ngx_fd_info(fd, &fi) == NGX_FILE_ERROR) {
95 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
96 "fstat \"%s\" failed", filename->data);
97 ngx_close_file(fd);
98 return NGX_ERROR;
99 }
100
101 buf_size = (size_t) ngx_file_size(&fi);
102 if (buf_size == 0) {
103 ngx_close_file(fd);
104 return NGX_OK; /* empty file: nothing to parse */
105 }
106 if (buf_size > 1024 * 1024) {
107 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
108 "brix_authdb \"%s\" exceeds 1 MiB limit",
109 filename->data);
110 ngx_close_file(fd);
111 return NGX_ERROR;
112 }
113
114 buf = ngx_alloc(buf_size + 1, cf->log);
115 if (buf == NULL) {
116 ngx_close_file(fd);
117 return NGX_ERROR;
118 }
119
120 n = ngx_read_file(&file, buf, buf_size, 0);
121 if (n == NGX_ERROR) {
122 ngx_free(buf);
123 ngx_close_file(fd);
124 return NGX_ERROR;
125 }
126
127 ngx_close_file(fd);
128 *out_buf = buf;
129 *out_len = (size_t) n;
130 return NGX_OK;
131}

133/* WHAT: carve [line_start,line_end) for the next line from *cursor and advance
134 * *cursor past its EOL; return the leading-whitespace-trimmed start.
135 * WHY: isolates line-splitting + EOL handling from field tokenization.
136 * HOW: scan from *cursor to the first CR/LF for the line end, then consume the
137 * EOL (LF, CR, or CRLF — no double-count) so the next call resumes on the
138 * following line. *line_start_out is the first non-whitespace byte (may
139 * equal *line_end_out for a blank line). */
140static void
141adb_next_line(u_char **cursor, u_char *end, u_char **line_start_out,
142 u_char **line_end_out)
143{
144 u_char *p = *cursor;
145 u_char *line_start = p;
146
147 while (p < end && *p != '\n' && *p != '\r') {
148 p++;
149 }
150 *line_end_out = p;
151
152 if (p < end && *p == '\r') p++;
153 if (p < end && *p == '\n') p++;
154 *cursor = p; /* carries the loop forward */
155
156 while (line_start < *line_end_out && isspace(*line_start)) {
157 line_start++;
158 }
159 *line_start_out = line_start;
160}

162/* WHAT: scan one whitespace-delimited field from *p within [.,line_end); on
163 * return *field_p is the field start and *p is the field end (first
164 * trailing whitespace or line_end). Returns 1 if a following field can
165 * still exist (*p < line_end), 0 if the field ran to line_end.
166 * WHY: the four authdb fields share one (skip-ws, scan-to-ws, truncation-guard)
167 * pattern; expressing it once keeps the tokenizer flat and uniform.
168 * HOW: advance *p over leading whitespace to *field_p, then scan to the next
169 * whitespace. The boolean return lets the caller reject a line truncated
170 * before a still-required field (return 0) while allowing the final field
171 * to legitimately end at line_end. */
172static ngx_flag_t
173adb_scan_field(u_char **p, u_char *line_end, u_char **field_p)
174{
175 while (*p < line_end && isspace(**p)) (*p)++;
176 *field_p = *p;
177 while (*p < line_end && !isspace(**p)) (*p)++;
178 return *p < line_end;
179}

181/* WHAT: tokenize one authdb line into its four fields, advancing *cursor past
182 * the line (including its EOL) for the next call.
183 * WHY: splits the line-carving + field-scanning grammar out of the parse loop
184 * so the loop body reads as tokenize → append. Pure over the buffer.
185 * HOW: adb_next_line carves the trimmed line; a blank line or one starting '#'
186 * yields out->valid=0. Otherwise scan four space-delimited fields with
187 * adb_scan_field: a line truncated before path/privs (scan returns 0 on a
188 * non-final field) yields out->valid=0 (rejected). The last field (privs)
189 * may be empty. Field slices point into the caller's buffer. */
190static void
191adb_tokenize_line(u_char **cursor, u_char *end, adb_line_t *out)
192{
193 u_char *line_start;
194 u_char *line_end;
195 u_char *p;
196
197 ngx_memzero(out, sizeof(*out));
198 adb_next_line(cursor, end, &line_start, &line_end);
199
200 /* Skip comments and empty lines. */
201 if (line_start == line_end || *line_start == '#') {
202 return; /* out->valid stays 0 */
203 }
204
205 /* Format: [u|g|p|a] <id> <path> <privs>. Field 1 (type): only type_p[0] is
206 * read later; the rest of the token (if any) is scanned over but ignored.
207 * Fields 1-3 must be followed by another field, so a 0 return rejects the
208 * line. Field 4 (privs) is last, so its scan return is not required. */
209 p = line_start;
210 if (!adb_scan_field(&p, line_end, &out->type_p)) return;
211
212 if (!adb_scan_field(&p, line_end, &out->id_p)) return;
213 out->id_end = p;
214
215 if (!adb_scan_field(&p, line_end, &out->path_p)) return;
216 out->path_end = p;
217
218 (void) adb_scan_field(&p, line_end, &out->privs_p);
219 out->privs_end = p;
220
221 out->valid = 1;
222}

224/* WHAT: push one parsed line as a rule into `rules`, copying id/path into the
225 * config pool. Returns NGX_OK, or NGX_ERROR on allocation failure.
226 * WHY: isolates the array-push + pool ownership from the tokenizer so both are
227 * independently testable and the loop body stays flat.
228 * HOW: ngx_array_push, then set type from field-1's lead byte only, copy id and
229 * path into cf->pool (NUL-terminated — the source buffer is freed at
230 * function exit so rules must own their strings), parse privs, and zero
231 * resolved[] (filled later by brix_finalize_authdb_rules — deferred
232 * realpath). */
233static ngx_int_t
234adb_append(ngx_conf_t *cf, ngx_array_t *rules, const adb_line_t *line)
235{
236 brix_authdb_rule_t *rule = ngx_array_push(rules);
237
238 if (rule == NULL) {
239 return NGX_ERROR;
240 }
241
242 /* Rule type is the first byte of field 1 only (e.g. 'u','g','p','a'); a
243 * multi-char first token is effectively truncated to its lead char. */
244 rule->type = (brix_auth_type_t) line->type_p[0];
245
246 rule->id.len = line->id_end - line->id_p;
247 rule->id.data = ngx_palloc(cf->pool, rule->id.len + 1);
248 ngx_memcpy(rule->id.data, line->id_p, rule->id.len);
249 rule->id.data[rule->id.len] = '\0';
250
251 rule->path.len = line->path_end - line->path_p;
252 rule->path.data = ngx_palloc(cf->pool, rule->path.len + 1);
253 ngx_memcpy(rule->path.data, line->path_p, rule->path.len);
254 rule->path.data[rule->path.len] = '\0';
255
256 rule->privs = brix_parse_privs((const char *) line->privs_p,
257 line->privs_end - line->privs_p);
258
259 ngx_memzero(rule->resolved, sizeof(rule->resolved));
260 return NGX_OK;
261}

263/* Parse the authdb file into `rules`: one rule per line (path + identity matcher
264 * + privileges). Returns NGX_CONF_OK / NGX_CONF_ERROR. */
265ngx_int_t
266brix_parse_authdb(ngx_conf_t *cf, ngx_str_t *filename, ngx_array_t *rules)
267{
268 u_char *buf = NULL;
269 u_char *p;
270 u_char *end;
271 size_t buf_len = 0;
272 adb_line_t line;
273
274 (void) ngx_stream_conf_get_module_srv_conf(cf, ngx_stream_brix_module);
275
276 if (adb_read_file(cf, filename, &buf, &buf_len) != NGX_OK) {
277 return NGX_ERROR;
278 }
279 if (buf == NULL) {
280 return NGX_OK; /* empty file: no rules */
281 }
282
283 /* Line loop: p is the running cursor over the whole buffer. adb_tokenize_line
284 * carves one line and advances p past its EOL; a valid line is appended. */
285 p = buf;
286 end = buf + buf_len;
287 while (p < end) {
288 adb_tokenize_line(&p, end, &line);
289 if (!line.valid) {
290 continue;
291 }
292 if (adb_append(cf, rules, &line) != NGX_OK) {
293 ngx_free(buf);
294 return NGX_ERROR;
295 }
296 }
297
298 ngx_free(buf);
299 return NGX_OK;
300}
