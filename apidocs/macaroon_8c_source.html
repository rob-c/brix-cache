1/* Macaroon bundle validation — public entry points and third-party discharge orchestration.
2 *
3 * WHAT: Validates WLCG macaroon tokens (base64url-encoded binary bundles). brix_macaroon_validate_bundle() space-
4 * tokenizes a bundle (root + discharges), base64url-decodes the root, drives the HMAC chain parse
5 * (macaroon_parse_core, macaroon_parse.c) to extract claims and capture third-party caveats, then for each caveat
6 * locates the matching discharge, decrypts its vid (macaroon_decrypt_vid, macaroon_crypto.c), validates the
7 * discharge, and intersects its expiry/paths into the root claims. brix_token_is_macaroon() and
8 * brix_macaroon_secret_parse() are the routing/key-material helpers.
9 *
10 * WHY: WLCG uses macaroon tokens for delegated, caveatable authorization; third-party caveats enable cross-service
11 * delegation where a discharge macaroon proves the third party authorized access. This file owns the bundle-level
12 * orchestration; the HMAC chain (macaroon_parse.c), caveat interpretation (macaroon_caveats.c), and vid crypto
13 * (macaroon_crypto.c) were split out under the phase-79 file-size guard, sharing macaroon_internal.h.
14 *
15 * HOW: brix_token_is_macaroon() — no dots ⇒ macaroon. brix_macaroon_secret_parse() — hex→binary via nibble pairing.
16 * brix_macaroon_validate_bundle() — tokenize; decode+parse root with tp_arr capture; for each tp: find discharge by
17 * cid==identifier (macaroon_candidate_matches_cid uses brix_macaroon_packet_len), decrypt vid, parse discharge with
18 * recovered key, intersect claims. brix_macaroon_validate() is the single-root wrapper. */

40int
41brix_token_is_macaroon(const char *token, size_t token_len)
42/* WHAT: Quick heuristic to distinguish macaroon tokens from JWT tokens.
43 * WHY: The token layer needs to route authentication logic — JWTs use signature/algorithm verification while
44 * macaroons use HMAC chain reconstruction. This check avoids expensive parsing of non-macaroon tokens.
45 * HOW: Reject tokens shorter than 4 characters (both JWT and macaroon require minimum length); if memchr finds a '.' character,
46 * classify as JWT (return 0); otherwise classify as macaroon (return 1). Note: this is heuristic only — base64url-encoded content could theoretically contain dots. */
47{
48 /* JWTs always have dots. Macaroons in base64url do not. */
49 if (token_len < 4) return 0;
50 if (memchr(token, '.', token_len) == NULL) return 1;
51 return 0;
52}

54ssize_t
55brix_macaroon_secret_parse(const char *hex, size_t hex_len,
56 u_char *bin, size_t bin_max)
57/* WHAT: Convert a hex-encoded macaroon root secret string into binary bytes for HMAC computation.
58 * WHY: Macaroon secrets are stored as hex strings (e.g., in config files or environment variables); the HMAC chain
59 * requires raw binary key material. This helper performs safe hex-to-binary conversion with bounds checking.
60 * HOW: Validate hex_len is even and hex_len/2 ≤ bin_max; iterate each nibble pair calling brix_hex_from_char() on positions i*2 and i*2+1,
61 * reject if any nibble invalid (<0); combine nibbles via (v1<<4)|v2 into bin[i]; return hex_len/2 or -1 on failure. */
62{
63 size_t i;
64 int v1, v2;
65
66 if (hex_len % 2 != 0 || hex_len / 2 > bin_max) return -1;
67
68 for (i = 0; i < hex_len / 2; i++) {
69 v1 = brix_hex_from_char((unsigned char) hex[i * 2]);
70 v2 = brix_hex_from_char((unsigned char) hex[i * 2 + 1]);
71 if (v1 < 0 || v2 < 0) return -1;
72 bin[i] = (u_char) ((v1 << 4) | v2);
73 }
74
75 return (ssize_t) (hex_len / 2);
76}

78static int
79macaroon_bundle_tokenize(const char *token, size_t token_len,
80 const char **tokens, size_t *token_lens,
81 int *n_tokens)
82{
83 const char *cursor = token;
84 const char *end = token + token_len;
85 const char *token_start = cursor;
86
87 *n_tokens = 0;
88 for (;;) {
89 if (cursor >= end || *cursor == ' ') {
90 if (cursor > token_start
91 && *n_tokens <= BRIX_MACAROON_MAX_DISCHARGES) {
92 tokens[*n_tokens] = token_start;
93 token_lens[*n_tokens] = (size_t)(cursor - token_start);
94 (*n_tokens)++;
95 }
96 if (cursor >= end) {
97 break;
98 }
99 token_start = cursor + 1;
100 }
101 cursor++;
102 }
103 return (*n_tokens > 0) ? 0 : -1;
104}

106static int
107macaroon_candidate_matches_cid(const u_char *candidate, ssize_t candidate_len,
108 brix_macaroon_tp_t *tp)
109{
110 const u_char *packet = candidate;
111 const u_char *end = candidate + candidate_len;
112
113 while (packet + 4 <= end) {
114 int packet_len = brix_macaroon_packet_len(packet);
115 u_char *data;
116 size_t data_len;
117
118 if (packet_len < 4 || packet + packet_len > end) {
119 break;
120 }
121 data = (u_char *)(packet + 4);
122 data_len = (size_t)(packet_len - 4);
123 if (data_len > 0 && data[data_len - 1] == '\n') {
124 data_len--;
125 }
126
127 if (data_len >= 11 && memcmp(data, "identifier ", 11) == 0) {
128 const u_char *identifier = data + 11;
129 size_t identifier_len = data_len - 11;
130 return (identifier_len == tp->cid_len
131 && memcmp(identifier, tp->cid, identifier_len) == 0);
132 }
133 packet += packet_len;
134 }
135 return 0;
136}

138/*
139 * WHAT: A parsed space-separated macaroon bundle — parallel arrays of token
140 * pointers/lengths (tokens[0] = root, tokens[1..] = discharges) and the
141 * count.
142 * WHY: macaroon_find_discharge, macaroon_validate_one_discharge, and
143 * macaroon_validate_discharges all threaded tokens/token_lens/n_tokens as
144 * three separate parameters (6/7/7 total). Grouping them into one
145 * file-local descriptor keeps each helper ≤5 params; the discharge search
146 * walks the identical token set in the identical order, so which
147 * discharge matches which cid is unchanged.
148 * HOW: tokens[i]/token_lens[i] describe the i-th base64url token; index 0 is
149 * the root macaroon, indices 1..n_tokens-1 are candidate discharges.
150 */
151typedef struct {
152 const char **tokens;
153 size_t *token_lens;
154 int n_tokens;
155} macaroon_bundle_t;

157static ssize_t
158macaroon_find_discharge(const macaroon_bundle_t *bundle,
159 brix_macaroon_tp_t *tp, u_char *discharge_bin,
160 size_t discharge_cap)
161{
162 int token_index;
163
164 for (token_index = 1; token_index < bundle->n_tokens; token_index++) {
165 u_char candidate[MACAROON_MAX_BIN];
166 ssize_t candidate_len;
167
168 candidate_len = b64url_decode(bundle->tokens[token_index],
169 bundle->token_lens[token_index], candidate,
170 sizeof(candidate));
171 if (candidate_len < 4) {
172 continue;
173 }
174 if (macaroon_candidate_matches_cid(candidate, candidate_len, tp)) {
175 ngx_memcpy(discharge_bin, candidate, candidate_len);
176 return candidate_len <= (ssize_t) discharge_cap ? candidate_len : -1;
177 }
178 }
179 return -1;
180}

182static void
183macaroon_intersect_discharge_claims(ngx_log_t *log,
184 brix_token_claims_t *claims,
185 brix_token_claims_t *d_claims)
186{
187 int discharge_scope_index;
188 char discharge_paths[BRIX_MAX_TOKEN_SCOPES][BRIX_SCOPE_PATH_MAX];
189 int n_discharge_paths = 0;
190
191 if (d_claims->exp > 0
192 && (claims->exp == 0 || d_claims->exp < claims->exp)) {
193 claims->exp = d_claims->exp;
194 }
195
196 for (discharge_scope_index = 0;
197 discharge_scope_index < d_claims->scope_count
198 && n_discharge_paths < BRIX_MAX_TOKEN_SCOPES;
199 discharge_scope_index++) {
200 size_t path_len = strlen(
201 d_claims->scopes[discharge_scope_index].path);
202 if (path_len > 0 && path_len < BRIX_SCOPE_PATH_MAX) {
203 memcpy(discharge_paths[n_discharge_paths],
204 d_claims->scopes[discharge_scope_index].path,
205 path_len + 1);
206 n_discharge_paths++;
207 }
208 }
209
210 if (n_discharge_paths > 0) {
211 macaroon_apply_path_caveats(log, claims, discharge_paths,
212 n_discharge_paths);
213 }
214}

216static int
217macaroon_validate_one_discharge(ngx_log_t *log,
218 const macaroon_bundle_t *bundle,
219 brix_macaroon_tp_t *tp,
220 brix_token_claims_t *claims, int caveat_index)
221{
222 brix_token_claims_t d_claims;
223 u_char d_bin[MACAROON_MAX_BIN];
224 ssize_t d_bin_len;
225 u_char d_key[32];
226 macaroon_parse_input_t d_in;
227
228 d_bin_len = macaroon_find_discharge(bundle, tp, d_bin, sizeof(d_bin));
229 if (d_bin_len < 0) {
230 ngx_log_error(NGX_LOG_WARN, log, 0,
231 "brix_macaroon: no discharge Macaroon provided for "
232 "third-party caveat #%d", caveat_index);
233 return -1;
234 }
235
236 if (macaroon_decrypt_vid(tp->vid, tp->vid_len, tp->sig_before, d_key)
237 != 0) {
238 ngx_log_error(NGX_LOG_WARN, log, 0,
239 "brix_macaroon: cannot decrypt vid for caveat #%d "
240 "(unsupported vid format?)", caveat_index);
241 OPENSSL_cleanse(d_key, sizeof(d_key));
242 return -1;
243 }
244
245 ngx_memzero(&d_claims, sizeof(d_claims));
246 ngx_memzero(&d_in, sizeof(d_in));
247 d_in.log = log;
248 d_in.key = d_key;
249 d_in.key_len = 32;
250 d_in.claims = &d_claims;
251 /* tp_arr/n_tp/max_tp stay NULL/0: discharge parse, no nested capture */
252 if (macaroon_parse_core(&d_in, d_bin, (size_t)d_bin_len) != 0) {
253 OPENSSL_cleanse(d_key, sizeof(d_key));
254 ngx_log_error(NGX_LOG_WARN, log, 0,
255 "brix_macaroon: discharge Macaroon #%d is invalid",
256 caveat_index);
257 return -1;
258 }
259 OPENSSL_cleanse(d_key, sizeof(d_key));
260
261 ngx_log_error(NGX_LOG_DEBUG, log, 0,
262 "brix_macaroon: discharge #%d valid sub=\"%s\" scope=\"%s\"",
263 caveat_index, d_claims.sub, d_claims.scope_raw);
264 macaroon_intersect_discharge_claims(log, claims, &d_claims);
265 return 0;
266}

269macaroon_validate_discharges(ngx_log_t *log, const macaroon_bundle_t *bundle,
270 brix_macaroon_tp_t *tp_arr, int n_tp,
271 brix_token_claims_t *claims)
272{
273 int caveat_index;
274
275 if (n_tp == 0) {
276 return 0;
277 }
278 if (bundle->n_tokens < 2) {
279 ngx_log_error(NGX_LOG_WARN, log, 0,

282 return -1;
283 }
284
285 for (caveat_index = 0; caveat_index < n_tp; caveat_index++) {

290 }
291 }

295/*
296 * brix_macaroon_validate_bundle — validate a space-separated Macaroon bundle.
297 *
298 * Accepts: "<root_macaroon_b64url> [<discharge_b64url> ...]"
299 *
300 * For each third-party caveat (cid+vid pair) in the root Macaroon:
301 * 1. Find the discharge Macaroon whose identifier == cid.
302 * 2. Decrypt the vid with AES-256-CBC(key=sig_before_cid, IV=vid[0..15])
303 * to recover the 32-byte discharge Macaroon root key.
304 * 3. Validate the discharge Macaroon with that key.
305 * 4. Intersect discharge claims (path restrictions, expiry) with root claims.
306 *
307 * Returns 0 on success, -1 on any validation failure.
308 */
309int
310brix_macaroon_validate_bundle(ngx_log_t *log,
311 const char *token, size_t token_len,
312 const u_char *root_key, size_t root_key_len,
313 brix_token_claims_t *claims)
314/* WHAT: Validate a space-separated macaroon bundle containing one root macaroon and zero or more discharge macaroons.
315 * WHY: WLCG third-party caveats require the client to present corresponding discharge macaroons — each discharge proves
316 * that the third party authorized access for this specific request. Bundle validation ensures all discharges are valid,
317 * decryptable, and their scope/expiry constraints intersect correctly with root claims.
318 * HOW: Space-tokenize bundle into tokens[]/tlens[], base64url-decode root token → root_bin; zero-initialize claims and tp_arr;
319 * call macaroon_parse_core(log,root_bin,root_key,claims,tp_arr,&n_tp,max_tp) to reconstruct HMAC chain, extract caveats, record third-party (cid+vid);
320 * if n_tp>0: for each tp entry search discharge tokens[1..] for identifier matching tp->cid; b64url-decode matched discharge → d_bin;
321 * macaroon_decrypt_vid(tp->vid,tp->vid_len,tp->sig_before,d_key) to recover 32-byte discharge root key via AES-256-CBC;
322 * macaroon_parse_core(log,d_bin,d_key,32,&d_claims,NULL,NULL,0) to validate discharge (depth=1, no nested discharges); OPENSSL_cleanse d_key;
323 * intersect discharge expiry (earliest wins) and scope paths as additional path: caveats via macaroon_apply_path_caveats; return 0 success or -1 failure. */
324{
325 const char *tokens[BRIX_MACAROON_MAX_DISCHARGES + 1];
326 size_t tlens[BRIX_MACAROON_MAX_DISCHARGES + 1];
327 int n_tokens = 0;
328 u_char root_bin[MACAROON_MAX_BIN];
329 ssize_t root_bin_len;
330 brix_macaroon_tp_t tp_arr[BRIX_MACAROON_MAX_TP_CAVEATS];
331 int n_tp = 0;
332 macaroon_parse_input_t root_in;
333 macaroon_bundle_t bundle;
334
335 ngx_memzero(claims, sizeof(*claims));
336 ngx_memzero(tp_arr, sizeof(tp_arr));
337
338 if (macaroon_bundle_tokenize(token, token_len, tokens, tlens, &n_tokens)
339 != 0) {
340 ngx_log_error(NGX_LOG_WARN, log, 0,
341 "brix_macaroon: empty token bundle");
342 return -1;
343 }
344
345 /* Decode and validate the root Macaroon, collecting third-party caveats */
346 root_bin_len = b64url_decode(tokens[0], tlens[0],
347 root_bin, sizeof(root_bin));
348 if (root_bin_len < 0) {
349 ngx_log_error(NGX_LOG_WARN, log, 0,
350 "brix_macaroon: root token base64url decode failed");
351 return -1;
352 }
353
354 ngx_memzero(&root_in, sizeof(root_in));
355 root_in.log = log;
356 root_in.key = root_key;
357 root_in.key_len = root_key_len;
358 root_in.claims = claims;
359 root_in.tp_arr = tp_arr;
360 root_in.n_tp = &n_tp;
361 root_in.max_tp = BRIX_MACAROON_MAX_TP_CAVEATS;
362
363 if (macaroon_parse_core(&root_in, root_bin, (size_t)root_bin_len) != 0) {
364 return -1;
365 }
366
367 bundle.tokens = tokens;
368 bundle.token_lens = tlens;
369 bundle.n_tokens = n_tokens;
370
371 if (macaroon_validate_discharges(log, &bundle, tp_arr, n_tp, claims) != 0) {
372 return -1;
373 }
374
375 ngx_log_error(NGX_LOG_INFO, log, 0,
376 "brix_macaroon: valid Macaroon bundle sub=\"%s\" "
377 "iss=\"%s\" scope=\"%s\" exp=%L "
378 "(third-party: %d, discharges: %d)",
379 claims->sub, claims->iss, claims->scope_raw,
380 (long long)claims->exp, n_tp, n_tokens - 1);
381
382 return 0;
383}
