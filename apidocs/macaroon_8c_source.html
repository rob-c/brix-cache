1/* Macaroon token validation — HMAC-SHA256 signature chaining, WLCG caveat parsing, and third-party discharge verification.
2 *
3 * WHAT: Validates WLCG macaroon tokens (base64url-encoded binary bundles) by reconstructing the HMAC chain across
4 * packets (identifier, location, cid/vid caveats, signature), extracting scope/activity/path/expiry caveats into
5 * claims structure, and optionally validating discharge macaroons for each third-party caveat via AES-256-CBC vid decryption.
6 * Supports both single-root validation and space-separated multi-token bundles with up to 8 discharges per root.
7 *
8 * WHY: WLCG (Worldwide LHC Computing Grid) uses macaroon tokens for delegated, caveatable authorization. The HMAC chain
9 * ensures each caveat modifies the signature deterministically — any tampered caveat produces a mismatched final signature.
10 * Third-party caveats enable cross-service delegation where a discharge macaroon proves the third party authorized access.
11 *
12 * HOW: brix_token_is_macaroon() — check token has no dots (JWTs have 3 dot-separated parts, macaroons don't).
13 * brix_macaroon_secret_parse() — hex string to binary via nibble pairing. brix_macaroon_validate_bundle() — space-tokenize bundle,
14 * base64url-decode root, parse_core with tp_arr for third-party tracking, iterate discharges matching cid identifiers, decrypt vid via AES-256-CBC,
15 * validate each discharge, intersect expiry/paths into root claims. Static helpers: parse_packet_len (hex→uint32), parse_iso8601 (before: caveat time),
16 * macaroon_decrypt_vid (AES-256-CBC vid decryption), macaroon_rebuild_scope_raw (scope string reconstruction from parsed scopes),
17 * macaroon_apply_path_caveats (intersection logic for path caveats vs scope paths), macaroon_parse_core (HMAC chain reconstruction + caveat extraction). */

33static int
34parse_packet_len(const u_char *p)
35/* WHAT: Parse a 4-character hex-encoded packet length from macaroon binary data.
36 * WHY: Macaroon packets are prefixed with a hex-encoded 32-bit length field (8 hex chars → uint32). This helper converts the first 4 hex characters into an integer for bounds checking before reading packet data.
37 * HOW: Call brix_hex_from_char() on each of p[0..3], reject if any nibble invalid (<0); combine via bit shifts (v0<<12)|(v1<<8)|(v2<<4)|v3 to form uint32; return value or -1 on invalid nibble. */
38{
39 int v0, v1, v2, v3;
40 v0 = brix_hex_from_char(p[0]);
41 v1 = brix_hex_from_char(p[1]);
42 v2 = brix_hex_from_char(p[2]);
43 v3 = brix_hex_from_char(p[3]);
44 if (v0 < 0 || v1 < 0 || v2 < 0 || v3 < 0) return -1;
45 return (v0 << 12) | (v1 << 8) | (v2 << 4) | v3;
46}

48/* WHAT: Parse ISO8601 timestamp "YYYY-MM-DDTHH:MM:SSZ" into time_t for before: caveat expiry checking.
49 * WHY: Macaroon "before:" caveats specify absolute expiry times in a restricted ISO8601 format (no timezone offsets, no fractional seconds). This helper converts the string to epoch seconds for comparison with current time.
50 * HOW: Validate len≥20 and len<32; copy into buf[32] null-terminated; memset tm struct; sscanf(buf,"%d-%d-%dT%d:%d:%dZ") extracting year/month/day/hour/min/sec (expect 6 matches); adjust tm_year-=1900, tm_mon-=1, tm_isdst=-1; return timegm(&tm) or -1 on parse failure. */
51static time_t
52parse_iso8601(const char *s, size_t len)
53{
54 struct tm tm;
55 char buf[32];
56
57 if (len < 20 || len >= sizeof(buf)) return (time_t) -1;
58 memcpy(buf, s, len);
59 buf[len] = '\0';
60
61 memset(&tm, 0, sizeof(tm));
62 /* %Y-%m-%dT%H:%M:%SZ */
63 if (sscanf(buf, "%d-%d-%dT%d:%d:%dZ",
64 &tm.tm_year, &tm.tm_mon, &tm.tm_mday,
65 &tm.tm_hour, &tm.tm_min, &tm.tm_sec) != 6)
66 {
67 return (time_t) -1;
68 }
69
70 tm.tm_year -= 1900;
71 tm.tm_mon -= 1;
72 tm.tm_isdst = -1;
73
74 return timegm(&tm);
75}

77int
78brix_token_is_macaroon(const char *token, size_t token_len)
79/* WHAT: Quick heuristic to distinguish macaroon tokens from JWT tokens.
80 * WHY: The token layer needs to route authentication logic — JWTs use signature/algorithm verification while
81 * macaroons use HMAC chain reconstruction. This check avoids expensive parsing of non-macaroon tokens.
82 * HOW: Reject tokens shorter than 4 characters (both JWT and macaroon require minimum length); if memchr finds a '.' character,
83 * classify as JWT (return 0); otherwise classify as macaroon (return 1). Note: this is heuristic only — base64url-encoded content could theoretically contain dots. */
84{
85 /* JWTs always have dots. Macaroons in base64url do not. */
86 if (token_len < 4) return 0;
87 if (memchr(token, '.', token_len) == NULL) return 1;
88 return 0;
89}

91ssize_t
92brix_macaroon_secret_parse(const char *hex, size_t hex_len,
93 u_char *bin, size_t bin_max)
94/* WHAT: Convert a hex-encoded macaroon root secret string into binary bytes for HMAC computation.
95 * WHY: Macaroon secrets are stored as hex strings (e.g., in config files or environment variables); the HMAC chain
96 * requires raw binary key material. This helper performs safe hex-to-binary conversion with bounds checking.
97 * HOW: Validate hex_len is even and hex_len/2 ≤ bin_max; iterate each nibble pair calling brix_hex_from_char() on positions i*2 and i*2+1,
98 * reject if any nibble invalid (<0); combine nibbles via (v1<<4)|v2 into bin[i]; return hex_len/2 or -1 on failure. */
99{
100 size_t i;
101 int v1, v2;
102
103 if (hex_len % 2 != 0 || hex_len / 2 > bin_max) return -1;
104
105 for (i = 0; i < hex_len / 2; i++) {
106 v1 = brix_hex_from_char((unsigned char) hex[i * 2]);
107 v2 = brix_hex_from_char((unsigned char) hex[i * 2 + 1]);
108 if (v1 < 0 || v2 < 0) return -1;
109 bin[i] = (u_char) ((v1 << 4) | v2);
110 }
111
112 return (ssize_t) (hex_len / 2);
113}

123/*
124 * A third-party caveat (cid + vid pair) captured during root Macaroon parsing.
125 *
126 * sig_before is the HMAC sig value immediately before HMAC(sig, cid) was
127 * computed. At Macaroon creation time the discharge key was encrypted as:
128 * vid = [16-byte IV] || AES-256-CBC-encrypt(key=sig_before, IV, discharge_key)
129 * so sig_before is the AES decryption key needed to recover the discharge key.
130 */
131typedef struct {
132 u_char cid[BRIX_MACAROON_MAX_CID_LEN];
133 size_t cid_len;
134 u_char vid[BRIX_MACAROON_MAX_VID_LEN]; /* raw binary — NOT base64 */
135 size_t vid_len;
136 u_char sig_before[32]; /* HMAC sig before the cid update — AES-256 key */
137} brix_macaroon_tp_t;

139/* WHAT: Decrypt a third-party caveat vid blob to recover the discharge macaroon's 32-byte root key via AES-256-CBC.
140 * WHY: At macaroon creation time, the discharge key was encrypted as vid = [16-byte IV] || AES-256-CBC(sig_before_cid, discharge_key).
141 * sig_before_cid (the HMAC signature before cid update) serves as the AES decryption key. This function reverses that encryption so we can validate the discharge macaroon with its recovered root key.
142 * HOW: Validate vid_len≥32 (16-byte IV + minimum 16-byte ciphertext); EVP_CIPHER_CTX_new(); set_padding=0 to disable PKCS7 (discharge key is always 32 bytes = two AES blocks); EVP_DecryptInit_ex(ctx,EVP_aes_256_cbc,NULL,aes_key,vid) using vid[0..15] as IV; EVP_DecryptUpdate(plain,&olen,vid+16,vid_len-16); EVP_DecryptFinal_ex(plain+olen,&flen); verify olen+flen≥32; ngx_memcpy(discharge_key,plain,32); OPENSSL_cleanse plain; EVP_CIPHER_CTX_free(ctx); return 0 success or -1 failure. */
143/*
144 * Inner AES-256-CBC decrypt over an already-created ctx, writing the plaintext
145 * into the caller's `plain` scratch and, on success, copying the recovered
146 * 32-byte discharge key out. Returns 0 on success, -1 on any OpenSSL failure or
147 * a plaintext shorter than 32 bytes. The caller owns ctx and the cleanse/free of
148 * plain — keeping that cleanup at the edge lets this use flat early returns.
149 */
150static int
151macaroon_decrypt_vid_inner(EVP_CIPHER_CTX *ctx, u_char *plain,
152 const u_char *vid, size_t vid_len, const u_char *aes_key,
153 u_char *discharge_key)
154{
155 int olen = 0, flen = 0;
156
157 /*
158 * Disable PKCS7 padding: the discharge key is always 32 bytes (two AES
159 * blocks), so the ciphertext is always a multiple of the block size.
160 * This avoids ambiguity regardless of whether the issuer used padding.
161 */
162 EVP_CIPHER_CTX_set_padding(ctx, 0);
163
164 /* vid[0..15] = IV; vid[16..] = ciphertext; aes_key = 32-byte HMAC sig */
165 if (EVP_DecryptInit_ex(ctx, EVP_aes_256_cbc(), NULL, aes_key, vid) != 1) {
166 return -1;
167 }
168
169 if (EVP_DecryptUpdate(ctx, plain, &olen,
170 vid + 16, (int)(vid_len - 16)) != 1) {
171 return -1;
172 }
173
174 if (EVP_DecryptFinal_ex(ctx, plain + olen, &flen) != 1) {
175 return -1;
176 }
177
178 if (olen + flen < 32) {
179 return -1;
180 }
181
182 ngx_memcpy(discharge_key, plain, 32);
183 return 0;
184}

186static int
187macaroon_decrypt_vid(const u_char *vid, size_t vid_len,
188 const u_char *aes_key, u_char *discharge_key)
189{
190 EVP_CIPHER_CTX *ctx;
191 u_char plain[64];
192 int rc;
193
194 /* Need at least 16-byte IV + 16-byte ciphertext (one AES block) */
195 if (vid_len < 32) {
196 return -1;
197 }
198
199 ctx = EVP_CIPHER_CTX_new();
200 if (ctx == NULL) {
201 return -1;
202 }
203
204 rc = macaroon_decrypt_vid_inner(ctx, plain, vid, vid_len, aes_key,
205 discharge_key);
206
207 OPENSSL_cleanse(plain, sizeof(plain));
208 EVP_CIPHER_CTX_free(ctx);
209 return rc;
210}

212/* WHAT: Reconstruct scope_raw from individual scope entries and re-parse to refresh claims->scopes[].
213 * WHY: After path narrowing (macaroon_apply_path_caveats) modifies individual scope paths, the raw scope string becomes stale — it still contains old paths. This function rebuilds the canonical scope_raw string from the updated scopes[] array so downstream access checks use the correct narrowed paths.
214 * HOW: Iterate each scope entry's permission bits (read/write/create/modify); for each enabled bit append "perm_name:path" to scope_raw with space separator between entries; check bounds against sizeof(claims->scope_raw) — break if overflow; re-parse the rebuilt string via brix_token_parse_scopes() to refresh claims->scopes[]. */
215/* HOW: Clear scope_raw buffer. Iterate each scope entry's permission bits (read/write/create/modify).
216 * For each enabled bit, append "perm_name:path" to scope_raw with space separator between entries.
217 * Check bounds against sizeof(claims->scope_raw) — break if overflow.
218 * Re-parse the rebuilt string via brix_token_parse_scopes() to refresh claims->scopes[]. */
219static void
220macaroon_rebuild_scope_raw(brix_token_claims_t *claims)
221{
222 static const char *perm_names[] = {
223 "storage.read", "storage.write",
224 "storage.create", "storage.modify"
225 };
226 size_t off = 0;
227 int si;
228
229 claims->scope_raw[0] = '\0';
230 for (si = 0; si < claims->scope_count; si++) {
231 brix_token_scope_t *sc = &claims->scopes[si];
232 unsigned int bits[4] = {
233 sc->read, sc->write, sc->create, sc->modify
234 };
235 int bi;
236 for (bi = 0; bi < 4; bi++) {
237 size_t plen, pathlen;
238 if (!bits[bi]) {
239 continue;
240 }
241 plen = strlen(perm_names[bi]);
242 pathlen = strlen(sc->path);
243 if (off + plen + 1 + pathlen + 2 > sizeof(claims->scope_raw)) {
244 break;
245 }
246 if (off > 0) {
247 claims->scope_raw[off++] = ' ';
248 }
249 memcpy(claims->scope_raw + off, perm_names[bi], plen);
250 off += plen;
251 claims->scope_raw[off++] = ':';
252 memcpy(claims->scope_raw + off, sc->path, pathlen);
253 off += pathlen;
254 claims->scope_raw[off] = '\0';
255 }
256 }
257
258 claims->scope_count = brix_token_parse_scopes(
259 claims->scope_raw, claims->scopes, BRIX_MAX_TOKEN_SCOPES);
260}

268/* WHAT: Apply path: caveats to restrict scope paths via intersection logic — each caveat narrows allowed paths further.
269 * WHY: WLCG macaroon "path:" caveats enforce hierarchical path restrictions on top of already-granted scope permissions. The intersection ensures the final effective path is the most restrictive among all caveats and scopes, preventing over-authorization.
270 * HOW: For each caveat path cp: strip trailing slash from scope paths for comparison; case 1 (cp equal/deeper than sc->path): narrow scope to cp if different; case 2 (sc->path deeper than cp): keep sc->path already more restrictive; case 3 (disjoint paths): revoke all permissions (read/write/create/modify=0) on this scope entry; track narrowed flag; after processing all caveats, call macaroon_rebuild_scope_raw() if any narrowing occurred. */
271static void
272macaroon_apply_path_caveats(ngx_log_t *log,
273 brix_token_claims_t *claims,
274 char path_caveats[][BRIX_SCOPE_PATH_MAX], int n_path_caveats)
275{
276 int ci, si;
277 int narrowed = 0;
278
279 for (ci = 0; ci < n_path_caveats; ci++) {
280 const char *cp = path_caveats[ci];
281 size_t cplen = strlen(cp);
282
283 for (si = 0; si < claims->scope_count; si++) {
284 brix_token_scope_t *sc = &claims->scopes[si];
285 size_t slen = strlen(sc->path);
286 size_t slen_cmp = slen;
287
288 /* Strip trailing slash for comparison */
289 if (slen_cmp > 1 && sc->path[slen_cmp - 1] == '/') {
290 slen_cmp--;
291 }
292
293 /* Case 1: caveat path is equal to or deeper than scope path */
294 if (strncmp(cp, sc->path, slen_cmp) == 0
295 && (cp[slen_cmp] == '/' || cp[slen_cmp] == '\0'))
296 {
297 if (strcmp(sc->path, cp) != 0) {
298 ngx_log_error(NGX_LOG_INFO, log, 0,
299 "brix_macaroon: path: caveat \"%s\" narrows "
300 "scope path \"%s\" → \"%s\"", cp, sc->path, cp);
301 memcpy(sc->path, cp, cplen + 1);
302 narrowed = 1;
303 }
304 }
305 /* Case 2: scope path is already deeper than caveat path */
306 else if (strncmp(sc->path, cp, cplen) == 0
307 && (sc->path[cplen] == '/' || sc->path[cplen] == '\0'))
308 {
309 /* keep sc->path — already more restrictive */
310 }
311 /* Case 3: disjoint paths — revoke all permissions for this scope */
312 else {
313 if (sc->read || sc->write || sc->create || sc->modify) {
314 ngx_log_error(NGX_LOG_INFO, log, 0,
315 "brix_macaroon: path: caveat \"%s\" revokes "
316 "scope path \"%s\" (disjoint)", cp, sc->path);
317 narrowed = 1;
318 }
319 sc->read = sc->write = sc->create = sc->modify = 0;
320 }
321 }
322 }
323
324 if (narrowed) {
325 macaroon_rebuild_scope_raw(claims);
326 }
327}

329/* WHAT: Parse one macaroon binary, reconstruct HMAC-SHA256 signature chain across all packets, verify final signature, and extract WLCG caveats into claims.
330 * WHY: The macaroon security model requires each caveat to deterministically modify the HMAC chain — sig = HMAC(sig_prev, caveat_data). This ensures any tampered or reordered caveat produces a mismatched final signature. Extracting activity:/path:/before: caveats converts raw binary authorization into structured claims for access control decisions.
331 * HOW: Initialize sig=HMAC(key, identifier), scope_buf="", path_caveats[], last_cid/sig_before_cid state; loop packets (p+4≤end): parse_packet_len(p)→plen; data=p+4,dlen=plen-4; strip trailing newline if present; process packet types: "identifier " → HMAC(EVP_sha256,key,identifier)→sig, copy to claims->sub; "location " → copy to claims->iss; "cid " → save sig_before_cid, HMAC(sig,cid)→next_sig→sig, track last_cid for vid pairing; parse first-party caveats within cid data (activity:→scope mapping, before:→parse_iso8601→claims->exp min, path:→path_caveats array); "vid " → HMAC(sig,vid_data)→sig, record (cid+vid+sig_before) triple into tp_arr if available; "signature " → compare provided 32-byte sig against computed sig, reject mismatch; after loop: check found_sig and found_id, validate expiry (now>claims->exp), finalize scopes from scope_buf via brix_token_parse_scopes(), apply path caveats via macaroon_apply_path_caveats(); return 0 success or -1 failure. */
332static int
333macaroon_parse_core(ngx_log_t *log,
334 const u_char *bin, size_t bin_len,
335 const u_char *key, size_t key_len,
336 brix_token_claims_t *claims,
337 brix_macaroon_tp_t *tp_arr, int *n_tp, int max_tp)
338{
339 u_char sig[32];
340 const u_char *p, *end;
341 unsigned int sig_out_len;
342 int found_sig = 0;
343 int found_id = 0;
344 char scope_buf[1024];
345 size_t scope_off = 0;
346 char path_caveats[8][BRIX_SCOPE_PATH_MAX];
347 int n_path_caveats = 0;
348 /* State for cid+vid pairing within the packet loop */
349 u_char last_cid[BRIX_MACAROON_MAX_CID_LEN];
350 size_t last_cid_len = 0;
351 u_char sig_before_cid[32];
352 int have_last_cid = 0;
353
354 ngx_memzero(scope_buf, sizeof(scope_buf));
355
356 p = bin;
357 end = bin + bin_len;
358
359 while (p + 4 <= end) {
360 int plen = parse_packet_len(p);
361 u_char *data;
362 size_t dlen;
363
364 if (plen < 4 || p + plen > end) {
365 ngx_log_error(NGX_LOG_WARN, log, 0,
366 "brix_macaroon: malformed packet length: %d", plen);
367 return -1;
368 }
369
370 data = (u_char *)(p + 4);
371 dlen = (size_t)(plen - 4);
372
373 /* Strip trailing newline if present */
374 if (dlen > 0 && data[dlen - 1] == '\n') {
375 dlen--;
376 }
377
378 if (dlen >= 11 && memcmp(data, "identifier ", 11) == 0) {
379 const u_char *id = data + 11;
380 size_t idlen = dlen - 11;
381
382 HMAC(EVP_sha256(), key, key_len, id, idlen, sig, &sig_out_len);
383 ngx_memcpy(claims->sub, id,
384 idlen < sizeof(claims->sub) ? idlen : sizeof(claims->sub) - 1);
385 found_id = 1;
386 have_last_cid = 0;
387
388 } else if (dlen >= 9 && memcmp(data, "location ", 9) == 0) {
389 const u_char *loc = data + 9;
390 size_t loclen = dlen - 9;
391
392 ngx_memcpy(claims->iss, loc,
393 loclen < sizeof(claims->iss) ? loclen : sizeof(claims->iss) - 1);
394
395 } else if (dlen >= 4 && memcmp(data, "cid ", 4) == 0) {
396 const u_char *caveat = data + 4;
397 size_t clen = dlen - 4;
398 u_char next_sig[32];
399
400 if (!found_id) {
401 ngx_log_error(NGX_LOG_WARN, log, 0,
402 "brix_macaroon: caveat before identifier");
403 return -1;
404 }
405
406 /*
407 * Save the current sig before updating it.
408 * If the next packet is a vid, sig_before_cid is the AES-256-CBC
409 * key used to encrypt the discharge key in the vid blob.
410 */
411 ngx_memcpy(sig_before_cid, sig, 32);
412
413 /* Update HMAC chain: sig = HMAC(sig, caveat) */
414 HMAC(EVP_sha256(), sig, 32, caveat, clen, next_sig, &sig_out_len);
415 ngx_memcpy(sig, next_sig, 32);
416
417 /* Track this cid for a potential following vid packet */
418 last_cid_len = clen < sizeof(last_cid) ? clen : sizeof(last_cid) - 1;
419 ngx_memcpy(last_cid, caveat, last_cid_len);
420 have_last_cid = 1;
421
422 /* Parse WLCG first-party caveats */
423 if (clen >= 9 && memcmp(caveat, "activity:", 9) == 0) {
424 const char *act = (const char *)(caveat + 9);
425 size_t alen = clen - 9;
426 const char *scope = NULL;
427
428 if (alen == 8 && memcmp(act, "DOWNLOAD", 8) == 0) scope = "storage.read";
429 else if (alen == 4 && memcmp(act, "LIST", 4) == 0) scope = "storage.read";
430 else if (alen == 6 && memcmp(act, "UPLOAD", 6) == 0) scope = "storage.write";
431 else if (alen == 6 && memcmp(act, "DELETE", 6) == 0) scope = "storage.write";
432 else if (alen == 6 && memcmp(act, "MANAGE", 6) == 0) scope = "storage.modify";
433
434 if (scope) {
435 size_t slen = strlen(scope);
436 if (scope_off + slen + 2 < sizeof(scope_buf)) {
437 if (scope_off > 0) {
438 scope_buf[scope_off++] = ' ';
439 }
440 memcpy(scope_buf + scope_off, scope, slen);
441 scope_off += slen;
442 }
443 }
444
445 } else if (clen >= 7 && memcmp(caveat, "before:", 7) == 0) {
446 time_t exp = parse_iso8601((const char *)(caveat + 7), clen - 7);
447 if (exp != (time_t)-1) {
448 if (claims->exp == 0 || exp < claims->exp) {
449 claims->exp = exp;
450 }
451 }
452
453 } else if (clen >= 5 && memcmp(caveat, "path:", 5) == 0) {
454 const char *cp = (const char *)(caveat + 5);
455 size_t cplen = clen - 5;
456 if (n_path_caveats < 8 && cplen > 0
457 && cplen < BRIX_SCOPE_PATH_MAX)
458 {
459 memcpy(path_caveats[n_path_caveats], cp, cplen);
460 path_caveats[n_path_caveats][cplen] = '\0';
461 n_path_caveats++;
462 }
463 }
464
465 } else if (dlen >= 4 && memcmp(data, "vid ", 4) == 0) {
466 /*
467 * A vid packet pairs with the immediately preceding cid packet.
468 * Together they form a third-party caveat: the vid contains the
469 * discharge Macaroon's root key encrypted with AES-256-CBC using
470 * sig_before_cid as the AES key.
471 *
472 * HMAC chain update: sig = HMAC(sig_after_cid, vid_data)
473 */
474 const u_char *vid_data = data + 4;
475 size_t vid_len = dlen - 4;
476 u_char next_sig[32];
477
478 HMAC(EVP_sha256(), sig, 32, vid_data, vid_len, next_sig, &sig_out_len);
479 ngx_memcpy(sig, next_sig, 32);
480
481 /* Record the (cid, vid, sig_before) triple for discharge validation */
482 if (have_last_cid && tp_arr != NULL && n_tp != NULL
483 && *n_tp < max_tp)
484 {
485 brix_macaroon_tp_t *tp = &tp_arr[(*n_tp)++];
486 ngx_memcpy(tp->cid, last_cid, last_cid_len);
487 tp->cid_len = last_cid_len;
488 tp->vid_len = vid_len < sizeof(tp->vid)
489 ? vid_len : sizeof(tp->vid) - 1;
490 ngx_memcpy(tp->vid, vid_data, tp->vid_len);
491 ngx_memcpy(tp->sig_before, sig_before_cid, 32);
492 }
493
494 have_last_cid = 0; /* vid consumed the pending cid */
495
496 } else if (dlen >= 10 && memcmp(data, "signature ", 10) == 0) {
497 const u_char *provided_sig = data + 10;
498 /* Constant-time MAC compare: a timing-variable memcmp here is a
499 * byte-by-byte forgery oracle on a bearer token's signature. */
500 if (dlen < 10 + 32
501 || CRYPTO_memcmp(sig, provided_sig, 32) != 0) {
502 ngx_log_error(NGX_LOG_WARN, log, 0,
503 "brix_macaroon: signature mismatch");
504 return -1;
505 }
506 found_sig = 1;
507 }
508
509 p += plen;
510 }
511
512 if (!found_sig) {
513 ngx_log_error(NGX_LOG_WARN, log, 0,
514 "brix_macaroon: no valid signature found");
515 return -1;
516 }
517
518 /*
519 * Expiry handling (fail-closed). A macaroon is a bearer credential: a root
520 * with no before: caveat leaves claims->exp == 0 and would otherwise be valid
521 * forever, so a single leak would never lapse. We therefore REQUIRE an
522 * expiry on the root/standalone macaroon (tp_arr != NULL identifies that
523 * context; discharges are validated with tp_arr == NULL). dCache/WLCG
524 * macaroons always carry a before: caveat on the root, so this rejects only
525 * malformed or deliberately-unbounded tokens. Discharge macaroons may
526 * legitimately omit before: — their lifetime is governed by the root and
527 * intersected by the caller — so we only enforce "not already expired" for
528 * them. claims->exp is the earliest before: caveat seen in the packet loop
529 * above (macaroons only narrow, never widen).
530 */
531 {
532 time_t now = time(NULL);
533 if (tp_arr != NULL && claims->exp <= 0) {
534 ngx_log_error(NGX_LOG_WARN, log, 0,
535 "brix_macaroon: rejected — no before: (expiry) caveat; "
536 "non-expiring macaroons are not accepted");
537 return -1;
538 }
539 if (claims->exp > 0 && now > (time_t)claims->exp) {
540 ngx_log_error(NGX_LOG_WARN, log, 0,
541 "brix_macaroon: token expired at %L (now=%L)",
542 (long long)claims->exp, (long long)now);
543 return -1;
544 }
545 }
546
547 /* Finalize scopes from activity: caveats */
548 ngx_memcpy(claims->scope_raw, scope_buf, scope_off);
549 claims->scope_raw[scope_off] = '\0';
550 claims->scope_count = brix_token_parse_scopes(
551 claims->scope_raw, claims->scopes, BRIX_MAX_TOKEN_SCOPES);
552
553 /* Apply path: caveat restrictions */
554 if (n_path_caveats > 0) {
555 macaroon_apply_path_caveats(log, claims, path_caveats, n_path_caveats);
556 }
557
558 return 0;
559}

561/*
562 * brix_macaroon_validate_bundle — validate a space-separated Macaroon bundle.
563 *
564 * Accepts: "<root_macaroon_b64url> [<discharge_b64url> ...]"
565 *
566 * For each third-party caveat (cid+vid pair) in the root Macaroon:
567 * 1. Find the discharge Macaroon whose identifier == cid.
568 * 2. Decrypt the vid with AES-256-CBC(key=sig_before_cid, IV=vid[0..15])
569 * to recover the 32-byte discharge Macaroon root key.
570 * 3. Validate the discharge Macaroon with that key.
571 * 4. Intersect discharge claims (path restrictions, expiry) with root claims.
572 *
573 * Returns 0 on success, -1 on any validation failure.
574 */
575int
576brix_macaroon_validate_bundle(ngx_log_t *log,
577 const char *token, size_t token_len,
578 const u_char *root_key, size_t root_key_len,
579 brix_token_claims_t *claims)
580/* WHAT: Validate a space-separated macaroon bundle containing one root macaroon and zero or more discharge macaroons.
581 * WHY: WLCG third-party caveats require the client to present corresponding discharge macaroons — each discharge proves
582 * that the third party authorized access for this specific request. Bundle validation ensures all discharges are valid,
583 * decryptable, and their scope/expiry constraints intersect correctly with root claims.
584 * HOW: Space-tokenize bundle into tokens[]/tlens[], base64url-decode root token → root_bin; zero-initialize claims and tp_arr;
585 * call macaroon_parse_core(log,root_bin,root_key,claims,tp_arr,&n_tp,max_tp) to reconstruct HMAC chain, extract caveats, record third-party (cid+vid);
586 * if n_tp>0: for each tp entry search discharge tokens[1..] for identifier matching tp->cid; b64url-decode matched discharge → d_bin;
587 * macaroon_decrypt_vid(tp->vid,tp->vid_len,tp->sig_before,d_key) to recover 32-byte discharge root key via AES-256-CBC;
588 * macaroon_parse_core(log,d_bin,d_key,32,&d_claims,NULL,NULL,0) to validate discharge (depth=1, no nested discharges); OPENSSL_cleanse d_key;
589 * intersect discharge expiry (earliest wins) and scope paths as additional path: caveats via macaroon_apply_path_caveats; return 0 success or -1 failure. */
590{
591 const char *tokens[BRIX_MACAROON_MAX_DISCHARGES + 1];
592 size_t tlens[BRIX_MACAROON_MAX_DISCHARGES + 1];
593 int n_tokens = 0;
594 const char *p, *end, *tok_start;
595 u_char root_bin[MACAROON_MAX_BIN];
596 ssize_t root_bin_len;
597 brix_macaroon_tp_t tp_arr[BRIX_MACAROON_MAX_TP_CAVEATS];
598 int n_tp = 0;
599 int ti;
600
601 ngx_memzero(claims, sizeof(*claims));
602 ngx_memzero(tp_arr, sizeof(tp_arr));
603
604 /* Tokenize the bundle on space boundaries */
605 p = token;
606 end = token + token_len;
607 tok_start = p;
608 for (;;) {
609 if (p >= end || *p == ' ') {
610 if (p > tok_start
611 && n_tokens <= BRIX_MACAROON_MAX_DISCHARGES)
612 {
613 tokens[n_tokens] = tok_start;
614 tlens[n_tokens] = (size_t)(p - tok_start);
615 n_tokens++;
616 }
617 if (p >= end) {
618 break;
619 }
620 tok_start = p + 1;
621 }
622 p++;
623 }
624
625 if (n_tokens == 0) {
626 ngx_log_error(NGX_LOG_WARN, log, 0,
627 "brix_macaroon: empty token bundle");
628 return -1;
629 }
630
631 /* Decode and validate the root Macaroon, collecting third-party caveats */
632 root_bin_len = b64url_decode(tokens[0], tlens[0],
633 root_bin, sizeof(root_bin));
634 if (root_bin_len < 0) {
635 ngx_log_error(NGX_LOG_WARN, log, 0,
636 "brix_macaroon: root token base64url decode failed");
637 return -1;
638 }
639
640 if (macaroon_parse_core(log,
641 root_bin, (size_t)root_bin_len,
642 root_key, root_key_len,
643 claims,
644 tp_arr, &n_tp, BRIX_MACAROON_MAX_TP_CAVEATS) != 0)
645 {
646 return -1;
647 }
648
649 /* Validate discharge Macaroons for each third-party caveat */
650 if (n_tp > 0) {
651 if (n_tokens < 2) {
652 ngx_log_error(NGX_LOG_WARN, log, 0,
653 "brix_macaroon: root has %d third-party caveat(s) "
654 "but no discharge Macaroons were provided", n_tp);
655 return -1;
656 }
657
658 for (ti = 0; ti < n_tp; ti++) {
659 brix_macaroon_tp_t *tp = &tp_arr[ti];
660 brix_token_claims_t d_claims;
661 u_char d_bin[MACAROON_MAX_BIN];
662 ssize_t d_bin_len = -1;
663 u_char d_key[32];
664 int di;
665
666 /* Find the discharge Macaroon whose identifier matches tp->cid */
667 for (di = 1; di < n_tokens; di++) {
668 u_char cand[MACAROON_MAX_BIN];
669 ssize_t cand_len;
670 const u_char *pb, *pe;
671
672 cand_len = b64url_decode(tokens[di], tlens[di],
673 cand, sizeof(cand));
674 if (cand_len < 4) {
675 continue;
676 }
677
678 /* Scan for the identifier packet */
679 pb = cand;
680 pe = cand + cand_len;
681 while (pb + 4 <= pe) {
682 int cplen = parse_packet_len(pb);
683 u_char *cdata;
684 size_t cdlen;
685
686 if (cplen < 4 || pb + cplen > pe) {
687 break;
688 }
689 cdata = (u_char *)(pb + 4);
690 cdlen = (size_t)(cplen - 4);
691 if (cdlen > 0 && cdata[cdlen - 1] == '\n') {
692 cdlen--;
693 }
694
695 if (cdlen >= 11 && memcmp(cdata, "identifier ", 11) == 0) {
696 const u_char *id = cdata + 11;
697 size_t idlen = cdlen - 11;
698 if (idlen == tp->cid_len
699 && memcmp(id, tp->cid, idlen) == 0)
700 {
701 ngx_memcpy(d_bin, cand, cand_len);
702 d_bin_len = cand_len;
703 }
704 break;
705 }
706 pb += cplen;
707 }
708
709 if (d_bin_len >= 0) {
710 break; /* matched this discharge Macaroon */
711 }
712 }
713
714 if (d_bin_len < 0) {
715 ngx_log_error(NGX_LOG_WARN, log, 0,
716 "brix_macaroon: no discharge Macaroon provided for "
717 "third-party caveat #%d", ti);
718 return -1;
719 }
720
721 /* Decrypt the vid to recover the discharge Macaroon's root key */
722 if (macaroon_decrypt_vid(tp->vid, tp->vid_len,
723 tp->sig_before, d_key) != 0)
724 {
725 ngx_log_error(NGX_LOG_WARN, log, 0,
726 "brix_macaroon: cannot decrypt vid for caveat #%d "
727 "(unsupported vid format?)", ti);
728 OPENSSL_cleanse(d_key, sizeof(d_key));
729 return -1;
730 }
731
732 /* Validate the discharge Macaroon with the recovered key.
733 * Discharge Macaroons cannot themselves have third-party caveats
734 * (depth limit = 1) so we pass NULL for the tp_arr argument. */
735 ngx_memzero(&d_claims, sizeof(d_claims));
736 if (macaroon_parse_core(log,
737 d_bin, (size_t)d_bin_len,
738 d_key, 32,
739 &d_claims,
740 NULL, NULL, 0) != 0)
741 {
742 OPENSSL_cleanse(d_key, sizeof(d_key));
743 ngx_log_error(NGX_LOG_WARN, log, 0,
744 "brix_macaroon: discharge Macaroon #%d is invalid", ti);
745 return -1;
746 }
747 OPENSSL_cleanse(d_key, sizeof(d_key));
748
749 ngx_log_error(NGX_LOG_DEBUG, log, 0,
750 "brix_macaroon: discharge #%d valid sub=\"%s\" scope=\"%s\"",
751 ti, d_claims.sub, d_claims.scope_raw);
752
753 /* Intersect discharge expiry with root expiry (take earliest) */
754 if (d_claims.exp > 0) {
755 if (claims->exp == 0 || d_claims.exp < claims->exp) {
756 claims->exp = d_claims.exp;
757 }
758 }
759
760 /* Intersect discharge path/scope constraints into root claims */
761 if (d_claims.scope_count > 0) {
762 /* Treat discharge scope paths as additional path: caveats */
763 int dsi;
764 char dp_caveats[BRIX_MAX_TOKEN_SCOPES][BRIX_SCOPE_PATH_MAX];
765 int n_dp = 0;
766
767 for (dsi = 0; dsi < d_claims.scope_count
768 && n_dp < BRIX_MAX_TOKEN_SCOPES; dsi++) {
769 size_t plen = strlen(d_claims.scopes[dsi].path);
770 if (plen > 0 && plen < BRIX_SCOPE_PATH_MAX) {
771 memcpy(dp_caveats[n_dp], d_claims.scopes[dsi].path,
772 plen + 1);
773 n_dp++;
774 }
775 }
776
777 if (n_dp > 0) {
778 macaroon_apply_path_caveats(log, claims, dp_caveats, n_dp);
779 }
780 }
781 }
782 }
783
784 ngx_log_error(NGX_LOG_INFO, log, 0,
785 "brix_macaroon: valid Macaroon bundle sub=\"%s\" "
786 "iss=\"%s\" scope=\"%s\" exp=%L "
787 "(third-party: %d, discharges: %d)",
788 claims->sub, claims->iss, claims->scope_raw,
789 (long long)claims->exp, n_tp, n_tokens - 1);
790
791 return 0;
792}
