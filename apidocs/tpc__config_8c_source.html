1/* File: tpc_config.c — HTTP-TPC location defaults and inheritance
2 * WHAT: Configures HTTP third-party copy (TPC) settings for WebDAV locations. This file contains two lifecycle functions: create_loc_conf initializes TPC fields to NGX_CONF_UNSET sentinel values marking them as unset; merge_loc_conf inherits parent config values using ngx_conf_merge_* macros and applies smart defaults for curl path (/usr/bin/curl), timeout (0 = no limit), OAuth2/OIDC token delegation credentials (empty endpoint, storage.read default scope). Also implements cross-field inheritance: if TPC-specific cert/key paths are unset but general WebDAV cafile/cadir exist, copies those values to ensure TPC has certificate access even when not explicitly configured.
3 *
4 * WHY: HTTP-TPC requires curl as the transfer tool and TLS credentials for source authentication — these defaults must be inherited from parent/server configuration so individual locations can override only what they need. OAuth2/OIDC token delegation enables TPC to acquire temporary credentials for accessing remote endpoints without requiring client-side certificate exchange. The smart inheritance logic (cafile/cadir fallback, cert→key copy) reduces configuration burden by providing reasonable defaults when operators don't explicitly set every field. Thread safety: config setup runs once during nginx startup; no concurrent access after initialization. */

24ngx_http_brix_webdav_tpc_merge_loc_conf(
25 ngx_http_brix_webdav_loc_conf_t *conf,
26 ngx_http_brix_webdav_loc_conf_t *prev)
27{
28 ngx_conf_merge_value(conf->tpc, prev->tpc, 0);
29 /* SSRF policy: deny local, allow private by default (HEP federation nodes
30 * commonly reside on private networks, but loopback must stay blocked). */
31 ngx_conf_merge_value(conf->tpc_allow_local, prev->tpc_allow_local, 0);
32 ngx_conf_merge_value(conf->tpc_allow_private, prev->tpc_allow_private, 1);
33 ngx_conf_merge_str_value(conf->tpc_curl, prev->tpc_curl,
34 "/usr/bin/curl");
35 ngx_conf_merge_str_value(conf->tpc_cert, prev->tpc_cert, "");
36 ngx_conf_merge_str_value(conf->tpc_key, prev->tpc_key, "");
37 ngx_conf_merge_str_value(conf->tpc_cadir, prev->tpc_cadir, "");
38 ngx_conf_merge_str_value(conf->tpc_cafile, prev->tpc_cafile, "");
39 /* Phase 51 (B2): default the curl low-speed stall detector ON (abort a pull/
40 * push that averages < 1KB/s for 60s) so a stalled remote cannot hold a
41 * thread-pool worker indefinitely. This measures lack of PROGRESS, not
42 * duration, so a slow-but-advancing transfer is never clipped. The absolute
43 * total timeout stays opt-in (0 = unlimited) — operators who need a hard cap
44 * set brix_webdav_tpc_timeout. 0 on either low-speed knob disables it. */
45 ngx_conf_merge_uint_value(conf->tpc_timeout, prev->tpc_timeout, 0);
46 ngx_conf_merge_uint_value(conf->tpc_low_speed_bytes,
47 prev->tpc_low_speed_bytes, 1024);
48 ngx_conf_merge_uint_value(conf->tpc_low_speed_secs,
49 prev->tpc_low_speed_secs, 60);
50 ngx_conf_merge_uint_value(conf->tpc_marker_interval,
51 prev->tpc_marker_interval, 0);
52 ngx_conf_merge_uint_value(conf->tpc_max_streams, prev->tpc_max_streams, 1);
53
54 /* Per-user TPC credential forwarding defaults ON — a PULL presents the
55 * requesting user's delegated proxy / forwards their bearer to the source by
56 * default (the HTTP equivalent of the native root:// bearer passthrough).
57 * Opportunistic: absence of a per-user credential silently falls back to the
58 * service cert, so enabling it by default never introduces a new denial. */
59 ngx_conf_merge_value(conf->tpc_credential_forward,
60 prev->tpc_credential_forward, 1);
61
62 /* Merge OAuth2/OIDC token-delegation config. */
63 ngx_conf_merge_str_value(conf->tpc_cred.token_endpoint,
64 prev->tpc_cred.token_endpoint, "");
65 ngx_conf_merge_str_value(conf->tpc_cred.token_client_id,
66 prev->tpc_cred.token_client_id, "");
67 ngx_conf_merge_str_value(conf->tpc_cred.token_client_secret,
68 prev->tpc_cred.token_client_secret, "");
69 ngx_conf_merge_str_value(conf->tpc_cred.token_scope,
70 prev->tpc_cred.token_scope, "storage.read");
71
72 if (conf->tpc_cadir.len == 0 && conf->cadir.len > 0) {
73 conf->tpc_cadir = conf->cadir;
74 }
75 if (conf->tpc_cafile.len == 0 && conf->cafile.len > 0) {
76 conf->tpc_cafile = conf->cafile;
77 }
78 if (conf->tpc_key.len == 0 && conf->tpc_cert.len > 0) {
79 conf->tpc_key = conf->tpc_cert;
80 }
