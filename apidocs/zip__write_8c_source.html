9/* Append `len` bytes to the in-memory central directory buffer. */
10int
11cd_append(brix_zip_writer *w, const uint8_t *p, size_t len)
12{
13 if (w->cd_len + len > w->cd_cap) {
14 size_t ncap = w->cd_cap ? w->cd_cap * 2 : 4096;
15 uint8_t *np;
16 while (ncap < w->cd_len + len) {
17 ncap *= 2;
18 }
19 np = realloc(w->cd, ncap);
20 if (np == NULL) {
21 return XRDC_ZIP_ENOMEM;
22 }
23 w->cd = np;
24 w->cd_cap = ncap;
25 }
26 memcpy(w->cd + w->cd_len, p, len);
27 w->cd_len += len;
28 return XRDC_ZIP_OK;
29}

67/* Emit raw bytes through the sink, advancing the offset and latching errors. */
68int
69w_emit(brix_zip_writer *w, const void *p, size_t len)
70{
71 if (w->err != XRDC_ZIP_OK) {
72 return w->err;
73 }
74 if (w->wr(w->ctx, p, len) != 0) {
75 w->err = XRDC_ZIP_EIO;
76 return w->err;
77 }
78 w->offset += len;
79 return XRDC_ZIP_OK;
80}

83int
84brix_zip_writer_add_fd(brix_zip_writer *w, const char *name, int fd)
85{
86 struct stat stbuf;
87 uint64_t size, lfh_off;
88 uint32_t crc = (uint32_t) crc32(0L, Z_NULL, 0);
89 size_t namelen;
90 int zip64;
91 unsigned char buf[64 * 1024];
92 off_t pos;
93
94 if (w == NULL || name == NULL) {
95 return XRDC_ZIP_EBADF;
96 }
97 if (w->err != XRDC_ZIP_OK) {
98 return w->err;
99 }
100 namelen = strlen(name);
101 if (namelen == 0 || namelen > XRDC_ZIP_MAX_NAME) {
102 return XRDC_ZIP_EBADF;
103 }
104 if (fstat(fd, &stbuf) != 0) {
105 return XRDC_ZIP_EIO;
106 }
107 size = (uint64_t) stbuf.st_size;
108
109 /* Pass 1: CRC-32 over the whole file (STORE: comp == uncomp == size). */
110 for (pos = 0; (uint64_t) pos < size; ) {
111 ssize_t n = pread(fd, buf, sizeof(buf), pos); /* vfs-seam-allow: local zip-archive assembly, not export data */
112 if (n < 0) {
113 if (errno == EINTR) { continue; }
114 return XRDC_ZIP_EIO;
115 }
116 if (n == 0) {
117 break; /* file shrank under us */
118 }
119 crc = (uint32_t) crc32(crc, buf, (uInt) n);
120 pos += n;
121 }
122 size = (uint64_t) pos; /* the bytes we actually CRC'd/will write */
123
124 lfh_off = w->offset;
125 zip64 = (size > 0xfffffffeULL) || (lfh_off > 0xfffffffeULL);
126 if (zip64) {
127 w->any_zip64 = 1;
128 }
129
130 /* Local file header */ {
131 uint8_t lfh[30];
132 uint16_t extralen = zip64 ? 20 : 0; /* 4 hdr + 16 (uncomp,comp) */
133
134 put32(lfh + 0, SIG_LFH);
135 put16(lfh + 4, zip64 ? 45 : 20); /* version needed */
136 put16(lfh + 6, 0); /* flags */
137 put16(lfh + 8, 0); /* method = STORE */
138 put16(lfh + 10, 0); /* mod time */
139 put16(lfh + 12, 0); /* mod date */
140 put32(lfh + 14, crc);
141 put32(lfh + 18, zip64 ? ZIP_U32_MAX : (uint32_t) size); /* comp */
142 put32(lfh + 22, zip64 ? ZIP_U32_MAX : (uint32_t) size); /* uncomp */
143 put16(lfh + 26, (uint16_t) namelen);
144 put16(lfh + 28, extralen);
145 if (w_emit(w, lfh, sizeof(lfh)) != XRDC_ZIP_OK
146 || w_emit(w, name, namelen) != XRDC_ZIP_OK) {
147 return w->err;
148 }
149 if (zip64) {
150 uint8_t ex[20];
151 put16(ex + 0, 0x0001);
152 put16(ex + 2, 16);
153 put64(ex + 4, size); /* uncompressed */
154 put64(ex + 12, size); /* compressed */
155 if (w_emit(w, ex, sizeof(ex)) != XRDC_ZIP_OK) {
156 return w->err;
157 }
158 }
159 }
160
161 /* File data (STORE: verbatim) */ for (pos = 0; (uint64_t) pos < size; ) {
162 size_t want = (size - (uint64_t) pos < sizeof(buf))
163 ? (size_t) (size - (uint64_t) pos) : sizeof(buf);
164 ssize_t n = pread(fd, buf, want, pos); /* vfs-seam-allow: local zip-archive assembly, not export data */
165 if (n < 0) {
166 if (errno == EINTR) { continue; }
167 w->err = XRDC_ZIP_EIO;
168 return w->err;
169 }
170 if (n == 0) {
171 break;
172 }
173 if (w_emit(w, buf, (size_t) n) != XRDC_ZIP_OK) {
174 return w->err;
175 }
176 pos += n;
177 }
178
179 /* Central-directory file header (accumulated, written at finish) */ {
180 uint8_t cdfh[46];
181 uint16_t extralen = zip64 ? 28 : 0; /* 4 hdr + 24 (uncomp,comp,off) */
182
183 put32(cdfh + 0, SIG_CDFH);
184 put16(cdfh + 4, zip64 ? 45 : 20); /* version made by */
185 put16(cdfh + 6, zip64 ? 45 : 20); /* version needed */
186 put16(cdfh + 8, 0); /* flags */
187 put16(cdfh + 10, 0); /* method = STORE */
188 put16(cdfh + 12, 0); /* mod time */
189 put16(cdfh + 14, 0); /* mod date */
190 put32(cdfh + 16, crc);
191 put32(cdfh + 20, zip64 ? ZIP_U32_MAX : (uint32_t) size); /* comp */
192 put32(cdfh + 24, zip64 ? ZIP_U32_MAX : (uint32_t) size); /* uncomp */
193 put16(cdfh + 28, (uint16_t) namelen);
194 put16(cdfh + 30, extralen);
195 put16(cdfh + 32, 0); /* comment len */
196 put16(cdfh + 34, 0); /* disk number start */
197 put16(cdfh + 36, 0); /* internal attrs */
198 put32(cdfh + 38, 0); /* external attrs */
199 put32(cdfh + 42, zip64 ? ZIP_U32_MAX : (uint32_t) lfh_off);
200 if (cd_append(w, cdfh, sizeof(cdfh)) != XRDC_ZIP_OK
201 || cd_append(w, (const uint8_t *) name, namelen) != XRDC_ZIP_OK) {
202 w->err = XRDC_ZIP_ENOMEM;
203 return w->err;
204 }
205 if (zip64) {
206 uint8_t ex[28];
207 put16(ex + 0, 0x0001);
208 put16(ex + 2, 24);
209 put64(ex + 4, size); /* uncompressed */
210 put64(ex + 12, size); /* compressed */
211 put64(ex + 20, lfh_off); /* LFH offset */
212 if (cd_append(w, ex, sizeof(ex)) != XRDC_ZIP_OK) {
213 w->err = XRDC_ZIP_ENOMEM;
214 return w->err;
215 }
216 }
217 }
218
219 w->n++;
220 return XRDC_ZIP_OK;
221}

224int
225brix_zip_writer_finish(brix_zip_writer *w)
226{
227 uint64_t cd_off, cd_size;
228 int zip64_eocd;
229
230 if (w == NULL) {
231 return XRDC_ZIP_EBADF;
232 }
233 if (w->err != XRDC_ZIP_OK) {
234 return w->err;
235 }
236
237 cd_off = w->offset;
238 cd_size = w->cd_len;
239
240 if (w->cd_len > 0 && w_emit(w, w->cd, w->cd_len) != XRDC_ZIP_OK) {
241 return w->err;
242 }
243
244 zip64_eocd = w->any_zip64 || cd_off > 0xfffffffeULL
245 || cd_size > 0xfffffffeULL || w->n > 0xfffeu;
246
247 if (zip64_eocd) {
248 uint8_t z64[56];
249 uint8_t loc[20];
250 uint64_t z64_off = w->offset;
251
252 put32(z64 + 0, SIG_Z64EOCD_W);
253 put64(z64 + 4, 44); /* size of remaining record */
254 put16(z64 + 12, 45); /* version made by */
255 put16(z64 + 14, 45); /* version needed */
256 put32(z64 + 16, 0); /* this disk */
257 put32(z64 + 20, 0); /* disk with CD start */
258 put64(z64 + 24, w->n); /* entries this disk */
259 put64(z64 + 32, w->n); /* entries total */
260 put64(z64 + 40, cd_size);
261 put64(z64 + 48, cd_off);
262 if (w_emit(w, z64, sizeof(z64)) != XRDC_ZIP_OK) {
263 return w->err;
264 }
265
266 put32(loc + 0, SIG_Z64LOC_W);
267 put32(loc + 4, 0); /* disk with ZIP64 EOCD */
268 put64(loc + 8, z64_off);
269 put32(loc + 16, 1); /* total disks */
270 if (w_emit(w, loc, sizeof(loc)) != XRDC_ZIP_OK) {
271 return w->err;
272 }
273 }
274
275 {
276 uint8_t eocd[22];
277 uint16_t n16 = (w->n > 0xffffu) ? 0xffffu : (uint16_t) w->n;
278 uint32_t csz32 = (cd_size > 0xfffffffeULL) ? ZIP_U32_MAX : (uint32_t) cd_size;
279 uint32_t coff32 = (cd_off > 0xfffffffeULL) ? ZIP_U32_MAX : (uint32_t) cd_off;
280
281 put32(eocd + 0, SIG_EOCD);
282 put16(eocd + 4, 0); /* this disk */
283 put16(eocd + 6, 0); /* disk with CD start */
284 put16(eocd + 8, n16); /* entries this disk */
285 put16(eocd + 10, n16); /* entries total */
286 put32(eocd + 12, csz32);
287 put32(eocd + 16, coff32);
288 put16(eocd + 20, 0); /* comment len */
289 if (w_emit(w, eocd, sizeof(eocd)) != XRDC_ZIP_OK) {
290 return w->err;
291 }
292 }
293 return XRDC_ZIP_OK;
294}
