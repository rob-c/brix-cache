9/* Append `len` bytes to the in-memory central directory buffer. */
10int
11cd_append(brix_zip_writer *w, const uint8_t *p, size_t len)
12{
13 if (w->cd_len + len > w->cd_cap) {
14 size_t ncap = w->cd_cap ? w->cd_cap * 2 : 4096;
15 uint8_t *np;
16 while (ncap < w->cd_len + len) {
17 ncap *= 2;
18 }
19 np = realloc(w->cd, ncap);
20 if (np == NULL) {
21 return XRDC_ZIP_ENOMEM;
22 }
23 w->cd = np;
24 w->cd_cap = ncap;
25 }
26 memcpy(w->cd + w->cd_len, p, len);
27 w->cd_len += len;
28 return XRDC_ZIP_OK;
29}

82/*
83 * zw_entry - per-entry layout facts shared across the add-fd phases.
84 *
85 * WHAT: bundles the values computed once in the CRC pass (name, size, crc,
86 * local-header offset, zip64 flag) so each layout-emitting helper takes
87 * one struct instead of a long parameter list.
88 * WHY: keeps the helper signatures ≤5 params and makes the frozen ZIP byte
89 * layout the single source of truth passed down the phase sequence.
90 * HOW: filled by zw_crc_pass(), then read-only for the header/data/central-dir
91 * emitters — no phase mutates it after the CRC pass.
92 */
93typedef struct {
94 const char *name;
95 size_t namelen;
96 uint64_t size; /* bytes actually CRC'd and written (STORE) */
97 uint64_t lfh_off; /* offset of this entry's local file header */
98 uint32_t crc; /* CRC-32 of the file data */
99 int zip64; /* wide fields needed for size/offset */
100} zw_entry;

103/*
104 * zw_crc_pass - CRC-32 the source file and freeze the entry's layout facts.
105 *
106 * WHAT: reads the whole fd once, accumulates the CRC-32, records the number of
107 * bytes actually read (the file may shrink), and derives lfh_off/zip64.
108 * WHY: STORE means comp == uncomp == the bytes we CRC'd, so this single pass
109 * fixes every size/offset field the later phases emit.
110 * HOW: streaming pread into `buf`; EINTR retried, short read ends the pass.
111 * Latches w->any_zip64 exactly as the original inline code did.
112 * RETURNS: XRDC_ZIP_OK, or XRDC_ZIP_EIO on a read error.
113 */
114static int
115zw_crc_pass(brix_zip_writer *w, int fd, zw_entry *e,
116 unsigned char *buf, size_t bufsz)
117{
118 off_t pos;
119
120 e->crc = (uint32_t) crc32(0L, Z_NULL, 0);
121 for (pos = 0; (uint64_t) pos < e->size; ) {
122 ssize_t n = pread(fd, buf, bufsz, pos); /* vfs-seam-allow: local zip-archive assembly, not export data */
123 if (n < 0) {
124 if (errno == EINTR) { continue; }
125 return XRDC_ZIP_EIO;
126 }
127 if (n == 0) {
128 break; /* file shrank under us */
129 }
130 e->crc = (uint32_t) crc32(e->crc, buf, (uInt) n);
131 pos += n;
132 }
133 e->size = (uint64_t) pos; /* the bytes we actually CRC'd/will write */
134 e->lfh_off = w->offset;
135 e->zip64 = (e->size > 0xfffffffeULL) || (e->lfh_off > 0xfffffffeULL);
136 if (e->zip64) {
137 w->any_zip64 = 1;
138 }
139 return XRDC_ZIP_OK;
140}

143/*
144 * zw_write_local_header - emit the local file header, name and zip64 extra.
145 *
146 * WHAT: builds the 30-byte LFH (STORE method, zeroed dates), appends the name,
147 * and, when zip64, the 20-byte zip64 extra field (uncomp,comp).
148 * WHY: first bytes of every archive member; layout is frozen wire format.
149 * HOW: put16/put32/put64 into fixed-size stack buffers, then w_emit each; any
150 * emit error latches into w->err and short-circuits.
151 * RETURNS: XRDC_ZIP_OK, or the latched w->err on an emit failure.
152 */
153static int
154zw_write_local_header(brix_zip_writer *w, const zw_entry *e)
155{
156 uint8_t lfh[30];
157 uint16_t extralen = e->zip64 ? 20 : 0; /* 4 hdr + 16 (uncomp,comp) */
158
159 put32(lfh + 0, SIG_LFH);
160 put16(lfh + 4, e->zip64 ? 45 : 20); /* version needed */
161 put16(lfh + 6, 0); /* flags */
162 put16(lfh + 8, 0); /* method = STORE */
163 put16(lfh + 10, 0); /* mod time */
164 put16(lfh + 12, 0); /* mod date */
165 put32(lfh + 14, e->crc);
166 put32(lfh + 18, e->zip64 ? ZIP_U32_MAX : (uint32_t) e->size); /* comp */
167 put32(lfh + 22, e->zip64 ? ZIP_U32_MAX : (uint32_t) e->size); /* uncomp */
168 put16(lfh + 26, (uint16_t) e->namelen);
169 put16(lfh + 28, extralen);
170 if (w_emit(w, lfh, sizeof(lfh)) != XRDC_ZIP_OK
171 || w_emit(w, e->name, e->namelen) != XRDC_ZIP_OK) {
172 return w->err;
173 }
174 if (e->zip64) {
175 uint8_t ex[20];
176 put16(ex + 0, 0x0001);
177 put16(ex + 2, 16);
178 put64(ex + 4, e->size); /* uncompressed */
179 put64(ex + 12, e->size); /* compressed */
180 if (w_emit(w, ex, sizeof(ex)) != XRDC_ZIP_OK) {
181 return w->err;
182 }
183 }
184 return XRDC_ZIP_OK;
185}

188/*
189 * zw_stream_deflate - stream the file data verbatim (STORE, no compression).
190 *
191 * WHAT: copies exactly e->size bytes from fd to the sink in buffer-sized chunks.
192 * WHY: the archive stores files uncompressed, so the payload is the raw bytes
193 * already accounted for by the CRC pass — must match e->size exactly.
194 * HOW: streaming pread bounded by the remaining count; EINTR retried, a short
195 * read ends the copy; each chunk emitted via w_emit (latches w->err).
196 * RETURNS: XRDC_ZIP_OK, or the latched w->err on a read/emit failure.
197 */
198static int
199zw_stream_deflate(brix_zip_writer *w, int fd, const zw_entry *e,
200 unsigned char *buf, size_t bufsz)
201{
202 off_t pos;
203
204 for (pos = 0; (uint64_t) pos < e->size; ) {
205 size_t want = (e->size - (uint64_t) pos < bufsz)
206 ? (size_t) (e->size - (uint64_t) pos) : bufsz;
207 ssize_t n = pread(fd, buf, want, pos); /* vfs-seam-allow: local zip-archive assembly, not export data */
208 if (n < 0) {
209 if (errno == EINTR) { continue; }
210 w->err = XRDC_ZIP_EIO;
211 return w->err;
212 }
213 if (n == 0) {
214 break;
215 }
216 if (w_emit(w, buf, (size_t) n) != XRDC_ZIP_OK) {
217 return w->err;
218 }
219 pos += n;
220 }
221 return XRDC_ZIP_OK;
222}

225/*
226 * zw_update_central_dir - accumulate this entry's central-directory record.
227 *
228 * WHAT: builds the 46-byte CDFH, appends the name and (when zip64) the 28-byte
229 * zip64 extra (uncomp,comp,off) into the in-memory central directory.
230 * WHY: the central directory is written en bloc at finish; each add_fd must
231 * stage its record now with the frozen field layout.
232 * HOW: put16/put32/put64 into stack buffers, cd_append into w->cd; any append
233 * failure latches XRDC_ZIP_ENOMEM into w->err.
234 * RETURNS: XRDC_ZIP_OK, or the latched w->err on an allocation failure.
235 */
236static int
237zw_update_central_dir(brix_zip_writer *w, const zw_entry *e)
238{
239 uint8_t cdfh[46];
240 uint16_t extralen = e->zip64 ? 28 : 0; /* 4 hdr + 24 (uncomp,comp,off) */
241
242 put32(cdfh + 0, SIG_CDFH);
243 put16(cdfh + 4, e->zip64 ? 45 : 20); /* version made by */
244 put16(cdfh + 6, e->zip64 ? 45 : 20); /* version needed */
245 put16(cdfh + 8, 0); /* flags */
246 put16(cdfh + 10, 0); /* method = STORE */
247 put16(cdfh + 12, 0); /* mod time */
248 put16(cdfh + 14, 0); /* mod date */
249 put32(cdfh + 16, e->crc);
250 put32(cdfh + 20, e->zip64 ? ZIP_U32_MAX : (uint32_t) e->size); /* comp */
251 put32(cdfh + 24, e->zip64 ? ZIP_U32_MAX : (uint32_t) e->size); /* uncomp */
252 put16(cdfh + 28, (uint16_t) e->namelen);
253 put16(cdfh + 30, extralen);
254 put16(cdfh + 32, 0); /* comment len */
255 put16(cdfh + 34, 0); /* disk number start */
256 put16(cdfh + 36, 0); /* internal attrs */
257 put32(cdfh + 38, 0); /* external attrs */
258 put32(cdfh + 42, e->zip64 ? ZIP_U32_MAX : (uint32_t) e->lfh_off);
259 if (cd_append(w, cdfh, sizeof(cdfh)) != XRDC_ZIP_OK
260 || cd_append(w, (const uint8_t *) e->name, e->namelen) != XRDC_ZIP_OK) {
261 w->err = XRDC_ZIP_ENOMEM;
262 return w->err;
263 }
264 if (e->zip64) {
265 uint8_t ex[28];
266 put16(ex + 0, 0x0001);
267 put16(ex + 2, 24);
268 put64(ex + 4, e->size); /* uncompressed */
269 put64(ex + 12, e->size); /* compressed */
270 put64(ex + 20, e->lfh_off); /* LFH offset */
271 if (cd_append(w, ex, sizeof(ex)) != XRDC_ZIP_OK) {
272 w->err = XRDC_ZIP_ENOMEM;
273 return w->err;
274 }
275 }
276 return XRDC_ZIP_OK;
277}

280int
281brix_zip_writer_add_fd(brix_zip_writer *w, const char *name, int fd)
282{
283 struct stat stbuf;
284 zw_entry e = { 0 };
285 unsigned char buf[64 * 1024];
286 int rc;
287
288 if (w == NULL || name == NULL) {
289 return XRDC_ZIP_EBADF;
290 }
291 if (w->err != XRDC_ZIP_OK) {
292 return w->err;
293 }
294 e.name = name;
295 e.namelen = strlen(name);
296 if (e.namelen == 0 || e.namelen > XRDC_ZIP_MAX_NAME) {
297 return XRDC_ZIP_EBADF;
298 }
299 if (fstat(fd, &stbuf) != 0) {
300 return XRDC_ZIP_EIO;
301 }
302 e.size = (uint64_t) stbuf.st_size;
303
304 /* Pass 1: CRC-32 over the whole file (STORE: comp == uncomp == size). */
305 rc = zw_crc_pass(w, fd, &e, buf, sizeof(buf));
306 if (rc != XRDC_ZIP_OK) {
307 return rc;
308 }
309
310 if (zw_write_local_header(w, &e) != XRDC_ZIP_OK) {
311 return w->err;
312 }
313 if (zw_stream_deflate(w, fd, &e, buf, sizeof(buf)) != XRDC_ZIP_OK) {
314 return w->err;
315 }
316 if (zw_update_central_dir(w, &e) != XRDC_ZIP_OK) {
317 return w->err;
318 }
319
320 w->n++;
321 return XRDC_ZIP_OK;
322}

325int
326brix_zip_writer_finish(brix_zip_writer *w)
327{
328 uint64_t cd_off, cd_size;
329 int zip64_eocd;
330
331 if (w == NULL) {
332 return XRDC_ZIP_EBADF;
333 }
334 if (w->err != XRDC_ZIP_OK) {
335 return w->err;
336 }
337
338 cd_off = w->offset;
339 cd_size = w->cd_len;
340
341 if (w->cd_len > 0 && w_emit(w, w->cd, w->cd_len) != XRDC_ZIP_OK) {
342 return w->err;
343 }
344
345 zip64_eocd = w->any_zip64 || cd_off > 0xfffffffeULL
346 || cd_size > 0xfffffffeULL || w->n > 0xfffeu;
347
348 if (zip64_eocd) {
349 uint8_t z64[56];
350 uint8_t loc[20];
351 uint64_t z64_off = w->offset;
352
353 put32(z64 + 0, SIG_Z64EOCD_W);
354 put64(z64 + 4, 44); /* size of remaining record */
355 put16(z64 + 12, 45); /* version made by */
356 put16(z64 + 14, 45); /* version needed */
357 put32(z64 + 16, 0); /* this disk */
358 put32(z64 + 20, 0); /* disk with CD start */
359 put64(z64 + 24, w->n); /* entries this disk */
360 put64(z64 + 32, w->n); /* entries total */
361 put64(z64 + 40, cd_size);
362 put64(z64 + 48, cd_off);
363 if (w_emit(w, z64, sizeof(z64)) != XRDC_ZIP_OK) {
364 return w->err;
365 }
366
367 put32(loc + 0, SIG_Z64LOC_W);
368 put32(loc + 4, 0); /* disk with ZIP64 EOCD */
369 put64(loc + 8, z64_off);
370 put32(loc + 16, 1); /* total disks */
371 if (w_emit(w, loc, sizeof(loc)) != XRDC_ZIP_OK) {
372 return w->err;
373 }
374 }
375
376 {
377 uint8_t eocd[22];
378 uint16_t n16 = (w->n > 0xffffu) ? 0xffffu : (uint16_t) w->n;
379 uint32_t csz32 = (cd_size > 0xfffffffeULL) ? ZIP_U32_MAX : (uint32_t) cd_size;
380 uint32_t coff32 = (cd_off > 0xfffffffeULL) ? ZIP_U32_MAX : (uint32_t) cd_off;
381
382 put32(eocd + 0, SIG_EOCD);
383 put16(eocd + 4, 0); /* this disk */
384 put16(eocd + 6, 0); /* disk with CD start */
385 put16(eocd + 8, n16); /* entries this disk */
386 put16(eocd + 10, n16); /* entries total */
387 put32(eocd + 12, csz32);
388 put32(eocd + 16, coff32);
389 put16(eocd + 20, 0); /* comment len */
390 if (w_emit(w, eocd, sizeof(eocd)) != XRDC_ZIP_OK) {
391 return w->err;
392 }
393 }
394 return XRDC_ZIP_OK;
395}
