2 * tpc_cred_oidc.c — HTTP-TPC oidc-agent credential delegation
3 *
4 * The oidc-agent delegation mode: acquire an OAuth2/OIDC access token for a

40/*
41 * WHAT: Resolve the oidc-token binary to an absolute path — first honours an
42 * explicit override env var, then probes standard install locations.
43 * WHY: The fallback exec path must not be subject to $PATH substitution in the
44 * daemon's environment (a compromised PATH entry could shadow the real binary).
45 * HOW: secure_getenv("BRIX_OIDC_TOKEN_BIN") if access(X_OK) passes; else
46 * walk a fixed candidate list; return NULL if none executable.
47 */
48static const char *
49resolve_oidc_token_binary(void)
50{
51 static const char *const candidates[] = {
52 "/usr/bin/oidc-token", "/usr/local/bin/oidc-token", NULL
53 };
54 const char *const *p;
55 const char *override = secure_getenv("BRIX_OIDC_TOKEN_BIN");
56
57 if (override != NULL && access(override, X_OK) == 0) {
58 return override;
59 }
60 for (p = candidates; *p != NULL; p++) {
61 if (access(*p, X_OK) == 0) {
62 return *p;
63 }
64 }
65 return NULL;
66}

69/*
70 * WHAT: Resolve the oidc-agent UNIX-socket path from the environment.
71 * WHY: A per-user oidc-agent exports its control socket via $OIDC_SOCK; when
72 * unset we fall back to the conventional systemd user-runtime location so
73 * the helper still has a well-known path to try.
74 * HOW: Return $OIDC_SOCK verbatim when non-empty, else the fixed default; the
75 * result aliases process-owned storage, so the caller must not free it.
76 */
77static ngx_str_t
78oidc_resolve_socket_path(const char *sock_env)
79{
80 ngx_str_t sock_path;
81
82 if (sock_env != NULL && *sock_env != '\0') {
83 sock_path.data = (u_char *) sock_env;
84 sock_path.len = strlen(sock_env);
85 return sock_path;
86 }
87
88 /* Default oidc-agent socket location. */
89 sock_path.data = (u_char *) "/run/user/1000/oidc/oidc_agent.sock";
90 sock_path.len = strlen((char *) sock_path.data);
91 return sock_path;
92}

95/*
96 * WHAT: Choose the executable that fetches the token — the dedicated helper if
97 * installed, otherwise the `oidc-token` CLI on $PATH.
98 * WHY: Deployments that ship the packaged helper get the socket-IPC path; a
99 * plain oidc-agent install still works through the CLI fallback.
100 * HOW: Probe TPC_CRED_HELPER_PATH with access(X_OK); on failure return the
101 * literal "oidc-token" so the child branch can exec it via $PATH.
102 */
103static ngx_str_t
104oidc_resolve_helper_path(void)
105{
106 ngx_str_t helper_path;
107
108 helper_path.data = (u_char *) TPC_CRED_HELPER_PATH;
109 helper_path.len = strlen(TPC_CRED_HELPER_PATH);
110
111 if (access((char *) helper_path.data, X_OK) == 0) {
112 return helper_path;
113 }
114
115 /* Fall back to oidc-token CLI (must be in PATH). */
116 helper_path.data = (u_char *) "oidc-token";
117 helper_path.len = 10;
118 return helper_path;
119}

122/*
123 * WHAT: Derive the oidc-agent client name from a source URL's host, with an
124 * option-injection guard, into the caller-provided buffer.
125 * WHY: oidc-agent clients are conventionally named by host, but that host
126 * becomes the value of oidc-token's "-c" flag; a host beginning with '-'
127 * (impossible under RFC 952/1123) would be reparsed by getopt as a flag.
128 * HOW: Copy the URL authority up to the first '/', clamp to the buffer, then
129 * replace any leading-'-'/empty result with the safe literal "default".
130 */
131static void
132oidc_derive_client_name(const char *source_url, char *out, size_t out_sz)
133{
134 const char *host_start = strstr(source_url, "://");
135
136 if (host_start != NULL) {
137 const char *host_end;
138 size_t host_len;
139
140 host_start += 3;
141 host_end = strchr(host_start, '/');
142 if (host_end != NULL) {
143 host_len = (size_t) (host_end - host_start);
144 } else {
145 host_len = strlen(host_start);
146 }
147
148 if (host_len >= out_sz) {
149 host_len = out_sz - 1;
150 }
151 ngx_memcpy(out, host_start, host_len);
152 out[host_len] = '\0';
153 } else {
154 ngx_memcpy(out, "default", sizeof("default"));
155 }
156
157 /*
158 * W3 — option-injection guard. A legitimate hostname never begins with '-'
159 * (RFC 952/1123), so reject one that does and fall back to "default".
160 */
161 if (out[0] == '-' || out[0] == '\0') {
162 ngx_memcpy(out, "default", sizeof("default"));
163 }
164}

167/*
168 * WHAT: In the forked child, exec the `oidc-token` CLI fallback for a client.
169 * WHY: The CLI must run with an absolute binary path and a minimal controlled
170 * environment — a compromised $PATH or inherited env in the daemon could
171 * otherwise shadow the binary or influence its behaviour.
172 * HOW: Resolve the absolute oidc-token binary, build "-c <client>" argv and a
173 * PATH/HOME/OIDC_SOCK/XDG_RUNTIME_DIR envp, then execve; return on failure
174 * so the caller can _exit (this runs only in the child, never returns on
175 * success).
176 */
177static void
178oidc_child_exec_cli(const char *source_url, const char *sock_env)
179{
180 const char *fb_bin = resolve_oidc_token_binary();
181 char cli_client[64];
182 char cli_flag[3];
183
184 if (fb_bin == NULL) {
185 return;
186 }
187
188 oidc_derive_client_name(source_url, cli_client, sizeof(cli_client));
189
190 cli_flag[0] = '-';
191 cli_flag[1] = 'c';
192 cli_flag[2] = '\0';
193
194 {
195 const char *home_val = getenv("HOME");
196 const char *xdg_val = getenv("XDG_RUNTIME_DIR");
197 char home_buf[280], oidc_buf[280], xdg_buf[280];
198 char *fb_argv[4] = { (char *) fb_bin, cli_flag, cli_client, NULL };
199 char *fb_envp[5];
200 int ei = 0;
201
202 fb_envp[ei++] = "PATH=/usr/bin:/bin";
203 if (home_val != NULL) {
204 snprintf(home_buf, sizeof(home_buf), "HOME=%s", home_val);
205 fb_envp[ei++] = home_buf;
206 }
207 if (sock_env != NULL && *sock_env != '\0') {
208 snprintf(oidc_buf, sizeof(oidc_buf), "OIDC_SOCK=%s", sock_env);
209 fb_envp[ei++] = oidc_buf;
210 }
211 if (xdg_val != NULL) {
212 snprintf(xdg_buf, sizeof(xdg_buf), "XDG_RUNTIME_DIR=%s", xdg_val);
213 fb_envp[ei++] = xdg_buf;
214 }
215 fb_envp[ei] = NULL;
216 execve(fb_bin, fb_argv, fb_envp);
217 }
218}

221/*
222 * WHAT: The child half of the fork — wire stdout to the pipe and exec either the
223 * dedicated helper or the oidc-token CLI. Never returns.
224 * WHY: Isolating the child path keeps the parent orchestrator linear and makes
225 * the exec choices auditable in one place.
226 * HOW: dup2 the write end onto STDOUT, then dispatch on helper_path: CLI branch
227 * delegates to oidc_child_exec_cli, helper branch execs the packaged
228 * binary with "<sock> <url>" argv; _exit(127) if any exec fails.
229 */
230static void
231oidc_run_child(int pipefd[2], ngx_str_t helper_path, ngx_str_t sock_path,
232 const char *source_url, const char *sock_env)
233{
234 close(pipefd[0]);
235 dup2(pipefd[1], STDOUT_FILENO);
236 close(pipefd[1]);
237
238 if (ngx_strcmp((char *) helper_path.data, "oidc-token") == 0) {
239 oidc_child_exec_cli(source_url, sock_env);
240 } else {
241 /* Exec the dedicated helper: <helper> <sock-path> <source-url>. */
242 char helper_buf[512];
243 char *argv[4];
244
245 ngx_snprintf((u_char *) helper_buf, sizeof(helper_buf),
246 "%s", (char *) sock_path.data);
247 argv[0] = (char *) helper_path.data;
248 argv[1] = helper_buf; /* socket path */
249 argv[2] = (char *) source_url; /* source URL (for issuer) */
250 argv[3] = NULL;
251
252 execve(argv[0], argv, NULL);
253 }
254
255 _exit(127);
256}

259/*
260 * WHAT: Drain a pipe read end fully into buf, NUL-terminating it, and close it.
261 * WHY: The child's token can arrive in several read() chunks; a single owned
262 * loop keeps the parent orchestrator flat.
263 * HOW: Read until EOF/error or the buffer is one byte from full, NUL-terminate
264 * at the byte count, close read_fd, and return the count. buf_sz >= 1.
265 */
266static ssize_t
267oidc_drain_pipe(int read_fd, char *buf, size_t buf_sz)
268{
269 ssize_t nread = 0;
270
271 while (nread < (ssize_t) buf_sz - 1) {
272 ssize_t nr = read(read_fd, buf + nread,
273 buf_sz - 1 - (size_t) nread);
274 if (nr <= 0) {
275 break;
276 }
277 nread += nr;
278 }
279 buf[nread] = '\0';
280 close(read_fd);
281 return nread;
282}

285/*
286 * WHAT: Reap the child and restore the pre-fork signal mask, mapping its exit to
287 * NGX_OK / NGX_ERROR.
288 * WHY: SIGCHLD is blocked across the fork so nginx's handler cannot reap our
289 * PID first; the wait + mask restore must happen together, exactly once.
290 * HOW: waitpid, restore old_mask, then treat any non-zero / signalled exit as a
291 * fetch failure.
292 */
293static ngx_int_t
294oidc_reap_child(ngx_http_request_t *r, ngx_pid_t pid, const sigset_t *old_mask)
295{
296 int wstatus = 0;
297
298 waitpid(pid, &wstatus, 0);
299 sigprocmask(SIG_SETMASK, old_mask, NULL);
300
301 if (!WIFEXITED(wstatus) || WEXITSTATUS(wstatus) != 0) {
302 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
303 "tpc_cred(oidc): helper exited with status %d",
304 WIFEXITED(wstatus) ? WEXITSTATUS(wstatus) : -1);
305 return NGX_ERROR;
306 }
307 return NGX_OK;
308}

311/*
312 * WHAT: Turn the child's raw stdout into a token in *token_out.
313 * WHY: The helper may emit either a JSON access-token response or a bare token
314 * string; both must resolve to the same ngx_str_t token result.
315 * HOW: Strip trailing whitespace, then parse as JSON when it opens with '{',
316 * else pool-copy the trimmed bytes as the literal token.
317 */
318static ngx_int_t
319oidc_finalize_token(ngx_http_request_t *r, char *buf, ssize_t nread,
320 ngx_str_t *token_out)
321{
322 while (nread > 0 &&
323 (buf[nread - 1] == '\n' || buf[nread - 1] == '\r' ||
324 buf[nread - 1] == ' '))
325 {
326 nread--;
327 }
328 buf[nread] = '\0';
329
330 /* Try to parse as JSON first (helper may return JSON). */
331 if (buf[0] == '{') {
332 return tpc_cred_parse_token_response(r, buf, token_out);
333 }
334
335 /* Fallback: treat as raw token string. */
336 token_out->len = (size_t) nread;
337 token_out->data = ngx_pnalloc(r->pool, nread + 1);
338 if (token_out->data == NULL) {
339 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
340 "tpc_cred(oidc): pnalloc failed for token");
341 return NGX_ERROR;
342 }
343 memcpy(token_out->data, buf, nread);
344 token_out->data[nread] = '\0';
345
346 return NGX_OK;
347}

350/**
351 * Send a JSON request to the oidc-agent UNIX socket and read the token.
352 *
353 * Spawns a tiny fork/exec pipeline:
354 * nginx-worker → fork → exec "oidc-token -c <client>" or socket helper
355 *
356 * If the OIDC_SOCK environment variable is set, we connect directly to it.
357 * Otherwise we fall back to the default oidc-agent socket location.
358 *
359 * The helper process is: src/webdav/tpc_cred_oidc_helper (built separately)
360 * which handles the socket IPC. If that helper binary is not found we
361 * attempt a simpler approach using the `oidc-token` CLI if available.
362 */
363ngx_int_t
364tpc_cred_oidc_agent_fetch(ngx_http_request_t *r,
365 const char *source_url,
366 const char *scope,
367 ngx_str_t *token_out)
368{
369 ngx_pid_t pid;
370 int pipefd[2];
371 char buf[TPC_CRED_MAX_TOKEN_LEN + 64];
372 ssize_t nread = 0;
373 const char *sock_env = getenv("OIDC_SOCK");
374 ngx_str_t sock_path = oidc_resolve_socket_path(sock_env);
375 ngx_str_t helper_path = oidc_resolve_helper_path();
376 sigset_t old_mask, blk_mask;
377 ngx_int_t rc;
378
379 (void) scope;
380
381 /* Block SIGCHLD before fork so nginx's handler can't reap our child
382 * before we do. We restore the mask after waitpid(). */
383 sigemptyset(&blk_mask);
384 sigaddset(&blk_mask, SIGCHLD);
385 sigprocmask(SIG_BLOCK, &blk_mask, &old_mask);
386
387 /* Create a pipe for the child's stdout → parent's read end. */
388 if (pipe(pipefd) == -1) {
389 ngx_log_error(NGX_LOG_ERR, r->connection->log, errno,
390 "tpc_cred(oidc): pipe() failed");
391 sigprocmask(SIG_SETMASK, &old_mask, NULL);
392 return NGX_ERROR;
393 }
394
395 pid = fork();
396 if (pid == -1) {
397 ngx_log_error(NGX_LOG_ERR, r->connection->log, errno,
398 "tpc_cred(oidc): fork() failed");
399 close(pipefd[0]);
400 close(pipefd[1]);
401 sigprocmask(SIG_SETMASK, &old_mask, NULL);
402 return NGX_ERROR;
403 }
404
405 if (pid == 0) {
406 /* Child process — never returns. */
407 oidc_run_child(pipefd, helper_path, sock_path, source_url, sock_env);
408 }
409
410 /* Parent: read child's stdout, then reap it. */
411 close(pipefd[1]);
412
413 nread = oidc_drain_pipe(pipefd[0], buf, sizeof(buf));
414
415 rc = oidc_reap_child(r, pid, &old_mask);
416 if (rc != NGX_OK) {
417 return rc;
418 }
419
420 return oidc_finalize_token(r, buf, nread, token_out);
421}
