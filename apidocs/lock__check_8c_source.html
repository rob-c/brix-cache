37/*
38 * webdav_check_lock_at — evaluate the lock (if any) recorded on one ancestor
39 * path `check` while checking whether the target path is blocked.
40 *
41 * WHAT: Reads the lock xattr on `check`; if active and it covers the target and
42 * the client does not own it, reports the operation as blocked. Expired
43 * locks are cleaned up lazily (delete + lock-null reap) exactly as before.
44 * WHY: Isolating the per-level lock decision keeps the ascent loop in
45 * webdav_check_locks a simple, low-complexity path-walk.
46 * HOW: The ancestor `check`/`check_len` and the immutable target `path`/
47 * `path_len` travel together in a single walk struct. Returns NGX_OK
48 * (not blocked here), NGX_HTTP_LOCKED (an owning-mismatch lock covers the
49 * target), or NGX_HTTP_INTERNAL_SERVER_ERROR (xattr read failure).
50 * Coverage rule is unchanged: exact match, or a depth-infinity ancestor
51 * on a path boundary.
52 */
53/* One level of the ancestor walk: the (mutable) ancestor path being examined
54 * plus the (immutable) target path the whole walk is checking. */
55typedef struct {
56 const char *check;
57 size_t check_len;
58 const char *path;
59 size_t path_len;
60} webdav_lock_walk_t;
61
62static ngx_int_t
63webdav_check_lock_at(ngx_http_request_t *r, const webdav_lock_walk_t *w)
64{
65 webdav_lock_xattr_t e;
66 ngx_int_t rc;
67 int covers;
68
69 rc = webdav_lock_xattr_read(r, w->check, &e);
70 if (rc == NGX_ERROR) {
71 return NGX_HTTP_INTERNAL_SERVER_ERROR;
72 }
73 if (rc != NGX_OK) {
74 return NGX_OK;
75 }
76
77 if (e.expires <= (int64_t) ngx_time()) {
78 (void) webdav_lock_xattr_delete(r, w->check);
79 webdav_lock_reap_null(r, w->check, &e);
80 return NGX_OK;
81 }
82
83 /* Lock covers path if: exact match or depth-infinity ancestor. */
84 covers = (w->check_len == w->path_len)
85 || (e.depth_infinity
86 && w->check_len < w->path_len
87 && (w->check[w->check_len - 1] == '/'
88 || w->path[w->check_len] == '/'));
89 if (covers && !webdav_lock_if_header_matches(r, e.token)) {
90 return NGX_HTTP_LOCKED;
91 }
92 return NGX_OK;
93}

95/*
96 * webdav_lock_path_ascend — strip the last component from `check` in place.
97 *
98 * WHAT: Removes trailing slashes then the final path component of `check`.
99 * WHY: Advances the ancestor walk one level toward the export root.
100 * HOW: Returns 1 when a shorter parent path remains in `check`, 0 when the
101 * path can no longer be shortened (no interior slash) so the caller stops.
102 */
103static int
104webdav_lock_path_ascend(char *check, size_t check_len)
105{
106 char *slash = check + check_len - 1;
107
108 while (slash > check && *slash == '/') {
109 slash--;
110 }
111 while (slash > check && *slash != '/') {
112 slash--;
113 }
114 if (*slash != '/') {
115 return 0;
116 }
117 *slash = '\0';
118 return 1;
119}

121/*
122 * webdav_check_locks — walk from path up to export root, checking for active
123 * xattr locks at each level. O(path_depth) xattr reads.
124 *
125 * Returns NGX_HTTP_LOCKED if an active, client-unowned lock blocks this
126 * operation; NGX_OK otherwise.
127 */
128ngx_int_t
129webdav_check_locks(ngx_http_request_t *r, const char *path, int need_write)
130{
131 ngx_http_brix_webdav_loc_conf_t *conf;
132 char check[PATH_MAX];
133 size_t root_len;
134 webdav_lock_walk_t w;
135 ngx_int_t blocked;
136
137 (void) need_write;
138
139 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
140 root_len = strlen(conf->common.root_canon);
141
142 ngx_cpystrn((u_char *) check, (u_char *) path, sizeof(check));
143
144 w.check = check;
145 w.path = path;
146 w.path_len = strlen(path);
147
148 for (;;) {
149 w.check_len = strlen(check);
150
151 blocked = webdav_check_lock_at(r, &w);
152 if (blocked != NGX_OK) {
153 return blocked;
154 }
155
156 /* Stop at or above export root. */
157 if (w.check_len <= root_len) {
158 break;
159 }
160
161 if (!webdav_lock_path_ascend(check, w.check_len)) {
162 break;
163 }
164
165 if (strlen(check) < root_len) {
166 break;
167 }
168 }
169
170 return NGX_OK;
171}

173/*
174 * check_locks_descendants — scan direct and recursive children of a directory
175 * for unexpired xattr locks not covered by the client's If header.
176 * Does NOT walk upward; the caller is responsible for checking ancestors.
177 */
178static ngx_int_t
179check_locks_descendants(ngx_http_request_t *r, const char *dir)
180{
181 brix_vfs_ctx_t vctx;
182 brix_vfs_dir_t *dp;
183 char child[PATH_MAX];
184 size_t dir_len;
185 webdav_lock_xattr_t e;
186 ngx_int_t rc;
187
188 /* Confined NON-metered opendir (recursive lock scan; a planted in-export
189 * symlink cannot redirect it out of the export root). */
190 webdav_lock_vfs_ctx(r, dir, &vctx);
191 dp = brix_vfs_opendir_quiet(&vctx, NULL);
192 if (dp == NULL) {
193 return NGX_OK;
194 }
195
196 dir_len = strlen(dir);
197 rc = NGX_OK;
198
199 for ( ;; ) {
200 ngx_str_t name;
201 brix_vfs_dirent_kind_t dkind;
202 const char *dname;
203 int is_dir;
204 ngx_int_t xrc;
205
206 /* "."/".." filtered by readdir_kind; kind from d_type. */
207 if (brix_vfs_readdir_kind(dp, &name, &dkind) != NGX_OK) {
208 break; /* NGX_DONE (end) or error → stop */
209 }
210 dname = (const char *) name.data;
211
212 if (dir_len + 1 + name.len + 1 > sizeof(child)) {
213 continue; /* path too long — skip */
214 }
215 snprintf(child, sizeof(child), "%s/%s", dir, dname);
216
217 /* Check lock xattr directly on this child. */
218 xrc = webdav_lock_xattr_read(r, child, &e);
219 if (xrc == NGX_ERROR) {
220 rc = NGX_HTTP_INTERNAL_SERVER_ERROR;
221 break;
222 }
223
224 if (xrc == NGX_OK) {
225 if (e.expires > (int64_t) ngx_time()) {
226 if (!webdav_lock_if_header_matches(r, e.token)) {
227 rc = NGX_HTTP_LOCKED;
228 break;
229 }
230 } else {
231 (void) webdav_lock_xattr_delete(r, child);
232 webdav_lock_reap_null(r, child, &e);
233 }
234 }
235 /* xrc == NGX_DECLINED: no lock xattr on this child — OK */
236
237 /* Recurse into subdirectories. d_type gives the answer directly; on a
238 * DT_UNKNOWN filesystem fall back to a confined no-follow probe so a
239 * trailing symlink is never followed into recursion. */
240 is_dir = (dkind == BRIX_VFS_DT_DIR);
241 if (dkind == BRIX_VFS_DT_UNKNOWN) {
242 brix_vfs_ctx_t cctx;
243 brix_vfs_stat_t cvst;
244
245 webdav_lock_vfs_ctx(r, child, &cctx);
246 is_dir = (brix_vfs_probe(&cctx, 1 /* no-follow */, &cvst) == NGX_OK
247 && cvst.is_directory);
248 }
249 if (is_dir) {
250 rc = check_locks_descendants(r, child);
251 if (rc != NGX_OK) {
252 break;
253 }
254 }
255 }
256
257 brix_vfs_closedir(dp, r->connection->log);
258 return rc;
259}

261/*
262 * webdav_check_locks_tree — check locks on a path and all its descendants.
263 * Use this instead of webdav_check_locks() for DELETE/COPY/MOVE on
264 * collections, where a lock on any child must block the operation.
265 */
266ngx_int_t
267webdav_check_locks_tree(ngx_http_request_t *r, const char *path)
268{
269 brix_vfs_ctx_t vctx;
270 brix_vfs_stat_t vst;
271 ngx_int_t rc;
272
273 rc = webdav_check_locks(r, path, 1);
274 if (rc != NGX_OK) {
275 return rc;
276 }
277
278 /* Only recurse for a directory (confined no-follow probe). */
279 webdav_lock_vfs_ctx(r, path, &vctx);
280 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) == NGX_OK
281 && vst.is_directory)
282 {
283 rc = check_locks_descendants(r, path);
284 }
285
286 return rc;
287}
