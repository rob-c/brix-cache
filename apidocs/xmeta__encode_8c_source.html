1/*
2 * fs/meta/xmeta_encode.c — encode a per-file metadata record to its wire form.
3 *
4 * WHAT: brix_xmeta_encode() and its helpers: serialize a brix_xmeta_t into a
5 * malloc'd buffer — a byte-identical stock XrdPfc cinfo v4 prefix
6 * (version + Store POD + crc + bitmap + AStat[] + crc) followed by the
7 * "XCX1" extension header and the STATE / ORIGIN / DIGEST / BLOCKCRC TLV
8 * sections. WHY: split out of xmeta.c (942 lines) so the encode half and
9 * the decode half (xmeta_decode.c) each stay under the 500-line cap and
10 * can be reviewed as a matched pair. HOW: capacity is computed once by
11 * xmeta_encode_plan(), the buffer is malloc'd at exact size, then each
12 * region is appended by a cursor-tracked helper — every variable-length
13 * region is guarded by a crc32c, matching stock's Store/bitmap checksum
14 * convention.
15 */
16

40/* one TLV section: {u16 type, u16 reserved, u32 len, payload, u32 crc32c},
41 * crc computed over the 8-byte section header + payload */
42static void
43xmeta_put_section(uint8_t *buf, size_t *off, uint16_t type,
44 const void *payload, uint32_t plen)
45{
46 size_t hdr_off = *off;
47 uint16_t reserved = 0;
48 uint32_t crc;
49
50 xmeta_put(buf, off, &type, 2);
51 xmeta_put(buf, off, &reserved, 2);
52 xmeta_put(buf, off, &plen, 4);
53 xmeta_put(buf, off, payload, plen);
54 crc = brix_crc32c_value(buf + hdr_off, 8 + (size_t) plen);
55 xmeta_put(buf, off, &crc, 4);
56}

58/* encode layout plan: presence flags, payload lengths and total capacity,
59 * computed once before any bytes are written */
60typedef struct {
61 size_t blen; /* stock bitmap length in bytes */
62 unsigned have_origin; /* ORIGIN section present */
63 unsigned have_blockcrc; /* BLOCKCRC section present */
64 size_t origin_plen; /* ORIGIN payload length */
65 size_t blockcrc_plen; /* BLOCKCRC payload length (0 = absent) */
66 uint16_t nsec; /* extension section count */
67 size_t cap; /* total encoded record capacity */
68} xmeta_encode_plan_t;

70/* ---- Compute the encode layout for a record ----
71 *
72 * WHAT: Fills *plan with section presence flags, per-section payload lengths,
73 * the extension section count and the exact output buffer capacity for
74 * encoding m. Cannot fail.
75 *
76 * WHY: brix_xmeta_encode mallocs the output buffer once at exact capacity;
77 * keeping the capacity formula and the section presence decisions in one
78 * place guarantees the emit helpers below stay within the buffer.
79 *
80 * HOW: 1. Derive the bitmap byte length from nblocks.
81 * 2. ORIGIN is present iff any of etag/cks-alg/cks is non-empty.
82 * 3. BLOCKCRC is present iff tracked AND the table is allocated.
83 * 4. Count present sections, then sum stock prefix + extension header
84 * + 12-byte TLV overhead per section + payload lengths into cap.
85 */
86static void
87xmeta_encode_plan(const brix_xmeta_t *m, xmeta_encode_plan_t *plan)
88{
89 plan->blen = xmeta_bitmap_len(m->nblocks);
90 plan->have_origin = (m->etag_len > 0 || m->cks_len > 0
91 || m->cks_alg_len > 0);
92 plan->origin_plen = XMETA_ORIGIN_FIXED + m->etag_len
93 + m->cks_alg_len + m->cks_len;
94 plan->have_blockcrc = (m->have_blockcrc && m->blockcrc != NULL);
95 plan->blockcrc_plen = plan->have_blockcrc
96 ? 16 + (size_t) m->nblocks * 4 : 0;
97
98 plan->nsec = (uint16_t) ((m->have_state ? 1 : 0)
99 + (m->digests != NULL ? 1 : 0)
100 + (plan->have_blockcrc ? 1 : 0)
101 + (plan->have_origin ? 1 : 0));
102
103 plan->cap = 4 + sizeof(brix_xmeta_stock_store_t) + 4 /* version+store+crc */
104 + plan->blen
105 + (m->astat_count > 0 ? sizeof(m->astat) : 0) + 4
106 + 8 /* ext magic+ver+cnt */
107 + (m->have_state ? 12 + sizeof(xmeta_state_wire_t) : 0)
108 + (m->digests != NULL ? 12 + (size_t) m->digests_len : 0)
109 + (plan->blockcrc_plen ? 12 + plan->blockcrc_plen : 0)
110 + (plan->have_origin ? 12 + plan->origin_plen : 0);
111}

113/* ---- Emit the stock XrdPfc cinfo v4 prefix ----
114 *
115 * WHAT: Writes version + Store POD + crc, then bitmap + optional AStat
116 * records + their shared crc, advancing *off. Cannot fail (capacity
117 * was reserved by xmeta_encode_plan).
118 *
119 * WHY: The prefix must be byte-identical to what XrdPfc::Info::Write emits
120 * so stock tools can read the front of the record; isolating it keeps
121 * that compatibility contract in one reviewable block.
122 *
123 * HOW: 1. Build the Store POD from the in-memory record (astat_size is
124 * clamped to 0/1 — we keep at most one AStat).
125 * 2. Emit version, Store, crc32c(Store).
126 * 3. Emit the bitmap, then extend its crc over the AStat record if
127 * present, and emit the combined crc.
128 */
129static void
130xmeta_encode_stock_prefix(const brix_xmeta_t *m, uint8_t *buf, size_t *off,
131 size_t blen)
132{
133 brix_xmeta_stock_store_t store;
134 int32_t version = BRIX_XMETA_STOCK_VERSION;
135 uint32_t crc;
136
137 /* stock prefix — exactly what XrdPfc::Info::Write emits */
138 memset(&store, 0, sizeof(store));
139 store.buffer_size = m->buffer_size;
140 store.file_size = m->file_size;
141 store.creation_time = m->creation_time;
142 store.no_cksum_time = m->no_cksum_time;
143 store.access_cnt = m->access_cnt;
144 store.status_raw = m->status_raw;
145 store.astat_size = (m->astat_count > 0) ? 1 : 0;
146
147 xmeta_put(buf, off, &version, 4);
148 xmeta_put(buf, off, &store, sizeof(store));
149 crc = brix_crc32c_value(&store, sizeof(store));
150 xmeta_put(buf, off, &crc, 4);
151 xmeta_put(buf, off, m->bitmap, blen);
152 crc = brix_crc32c_value(m->bitmap, blen);
153 if (store.astat_size > 0) {
154 xmeta_put(buf, off, &m->astat, sizeof(m->astat));
155 crc = brix_crc32c_extend(crc, &m->astat, sizeof(m->astat));
156 }
157 xmeta_put(buf, off, &crc, 4);
158}

160/* ---- Emit the STATE extension section ----
161 *
162 * WHAT: Packs the write-back/lifecycle fields into the fixed 80-byte
163 * xmeta_state_wire_t and emits it as a STATE TLV section at *off.
164 *
165 * WHY: STATE is a persisted wire layout (size pinned by the static assert
166 * above); building it in a dedicated helper pairs it with
167 * xmeta_decode_state so the field order can be audited side by side.
168 *
169 * HOW: 1. Zero the wire struct, copy each field explicitly.
170 * 2. Emit via xmeta_put_section (TLV header + payload + crc32c).
171 */
172static void
173xmeta_encode_state_section(const brix_xmeta_t *m, uint8_t *buf, size_t *off)
174{
175 xmeta_state_wire_t state;
176
177 memset(&state, 0, sizeof(state));
178 state.origin_mtime = m->origin_mtime;
179 state.dirty_lo = m->dirty_lo;
180 state.dirty_hi = m->dirty_hi;
181 state.flush_gen = m->flush_gen;
182 state.dirty_since = m->dirty_since;
183 state.last_flush = m->last_flush;
184 state.bytes_flushed = m->bytes_flushed;
185 state.expires_at = m->expires_at;
186 state.filled_at = m->filled_at;
187 state.mode = m->mode;
188 state.state_flags = m->state_flags;
189 xmeta_put_section(buf, off, BRIX_XMETA_SEC_STATE,
190 &state, sizeof(state));
191}

193/* ---- Emit the ORIGIN extension section ----
194 *
195 * WHAT: Packs the origin validators (etag, checksum algorithm name, checksum
196 * hex) into the {u8 lens, strings...} ORIGIN payload and emits it as a
197 * TLV section at *off.
198 *
199 * WHY: ORIGIN is variable-length with three length-prefixed strings; the
200 * payload is staged in a stack buffer sized for the struct maxima so a
201 * single xmeta_put_section call covers header + payload + crc. Pairs
202 * with xmeta_decode_origin.
203 *
204 * HOW: 1. Write the 4-byte fixed header {etag_len, alg_len, cks_len, pad}.
205 * 2. Append the three strings back to back.
206 * 3. Emit via xmeta_put_section with the accumulated payload length.
207 */
208static void
209xmeta_encode_origin_section(const brix_xmeta_t *m, uint8_t *buf, size_t *off)
210{
211 uint8_t hdr4[XMETA_ORIGIN_FIXED] = {
212 m->etag_len, m->cks_alg_len, m->cks_len, 0
213 };
214 uint8_t payload[XMETA_ORIGIN_FIXED + sizeof(m->etag)
215 + sizeof(m->cks_alg) + sizeof(m->cks_hex)];
216 size_t poff = 0;
217
218 memcpy(payload, hdr4, XMETA_ORIGIN_FIXED); poff = XMETA_ORIGIN_FIXED;
219 memcpy(payload + poff, m->etag, m->etag_len); poff += m->etag_len;
220 memcpy(payload + poff, m->cks_alg, m->cks_alg_len); poff += m->cks_alg_len;
221 memcpy(payload + poff, m->cks_hex, m->cks_len); poff += m->cks_len;
222 xmeta_put_section(buf, off, BRIX_XMETA_SEC_ORIGIN,
223 payload, (uint32_t) poff);
224}

226/* ---- Emit the BLOCKCRC extension section ----
227 *
228 * WHAT: Emits the per-block crc32c table as a BLOCKCRC TLV section at *off.
229 * Only called when the plan marked the section present.
230 *
231 * WHY: The payload (16-byte header + nblocks*4 crc words) can be large, so
232 * it is built in place in the output buffer instead of staging a second
233 * nblocks*4 allocation like xmeta_put_section would require. Pairs with
234 * xmeta_decode_blockcrc.
235 *
236 * HOW: 1. Emit the TLV header (type, reserved, payload length) by hand.
237 * 2. Emit payload: u32 granule, u32 reserved, u64 nblocks, u32 crc[].
238 * 3. Compute crc32c over header+payload directly from the buffer and
239 * emit it (same coverage as xmeta_put_section).
240 */
241static void
242xmeta_encode_blockcrc_section(const brix_xmeta_t *m, uint8_t *buf, size_t *off)
243{
244 /* payload: u32 granule, u32 reserved, u64 nblocks, u32 crc[] — built
245 * in place to avoid a second nblocks*4 allocation */
246 size_t blockcrc_plen = 16 + (size_t) m->nblocks * 4;
247 size_t sec_hdr = *off;
248 uint16_t type = BRIX_XMETA_SEC_BLOCKCRC, rsv = 0;
249 uint32_t plen32 = (uint32_t) blockcrc_plen;
250 uint32_t granule = (uint32_t) m->buffer_size;
251 uint32_t zero = 0;
252 uint64_t nb = m->nblocks;
253 uint32_t crc;
254
255 xmeta_put(buf, off, &type, 2);
256 xmeta_put(buf, off, &rsv, 2);
257 xmeta_put(buf, off, &plen32, 4);
258 xmeta_put(buf, off, &granule, 4);
259 xmeta_put(buf, off, &zero, 4);
260 xmeta_put(buf, off, &nb, 8);
261 xmeta_put(buf, off, m->blockcrc, (size_t) m->nblocks * 4);
262 crc = brix_crc32c_value(buf + sec_hdr, 8 + blockcrc_plen);
263 xmeta_put(buf, off, &crc, 4);
264}

266int
267brix_xmeta_encode(const brix_xmeta_t *m, uint8_t **out, size_t *out_len)
268{
269 xmeta_encode_plan_t plan;
270 uint32_t ext_magic = BRIX_XMETA_EXT_MAGIC;
271 uint16_t ext_version = BRIX_XMETA_EXT_VERSION;
272 size_t off = 0;
273 uint8_t *buf;
274
275 if (m == NULL || out == NULL || out_len == NULL || m->buffer_size <= 0
276 || m->bitmap == NULL)
277 {
278 errno = EINVAL;
279 return BRIX_XMETA_ERR;
280 }
281
282 xmeta_encode_plan(m, &plan);
283 buf = malloc(plan.cap);
284 if (buf == NULL) {
285 errno = ENOMEM;
286 return BRIX_XMETA_ERR;
287 }
288
289 xmeta_encode_stock_prefix(m, buf, &off, plan.blen);
290
291 /* extension */
292 xmeta_put(buf, &off, &ext_magic, 4);
293 xmeta_put(buf, &off, &ext_version, 2);
294 xmeta_put(buf, &off, &plan.nsec, 2);
295
296 if (m->have_state) {
297 xmeta_encode_state_section(m, buf, &off);
298 }
299 if (plan.have_origin) {
300 xmeta_encode_origin_section(m, buf, &off);
301 }
302 if (m->digests != NULL) {
303 xmeta_put_section(buf, &off, BRIX_XMETA_SEC_DIGEST,
304 m->digests, m->digests_len);
305 }
306 if (plan.blockcrc_plen > 0) {
307 xmeta_encode_blockcrc_section(m, buf, &off);
308 }
309
310 *out = buf;
311 *out_len = off;
312 return BRIX_XMETA_OK;
313}
