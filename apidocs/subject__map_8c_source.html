30
31 if (path == NULL || path[0] == '\0' || subject == NULL) {
32 return -1;
33 }
34
35 f = fopen(path, "re"); /* 'e' = O_CLOEXEC */
36 if (f == NULL) {
37 return -1;
38 }
39
40 buf = malloc(BRIX_SUBJECT_MAP_MAX);
41 if (buf == NULL) {
42 fclose(f);
43 return -1;
44 }
45
46 got = fread(buf, 1, BRIX_SUBJECT_MAP_MAX - 1, f);
47 fclose(f);
48 buf[got] = '\0';
49
50 n = json_get_string(buf, got, subject, out, outsz);
51 free(buf);
52
53 return (n >= 0) ? 0 : -1;
54}
