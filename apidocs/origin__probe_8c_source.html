1/* origin_probe.c — per-worker RTT probe for brix_cvmfs_origin_select rtt.
2 *
3 * WHAT: a repeating per-worker timer posts a thread-pool task that measures
4 * TCP connect RTT to every configured origin endpoint; the event-loop
5 * completion folds samples into an EWMA, ranks endpoints, and pushes
6 * ranks into the sd_http driver.
7 * WHY: on a Tier-2 with erratic routing, configured order and geography
8 * both lie; measured connect RTT is what the fills will actually feel.
9 * HOW: measurement is blocking (getaddrinfo + nonblocking connect + poll)
10 * and therefore lives on a thread-pool worker, mirroring the cache
11 * fill I/O pattern; the event loop only re-arms the timer and writes
12 * ranks (relaxed atomics — see sd_http.h). Exports register at config
13 * time (merge runs in the master; the list survives the fork), and
14 * every worker arms its own timers at init_process — the same
15 * lifecycle the cache reaper uses.
16 */

45 const ngx_str_t *pool_name)
46{
47 ngx_uint_t i;
48 cvmfs_rtt_reg_t *reg = NULL;
49
50 for (i = 0; i < cvmfs_rtt_regs_n; i++) {
51 if (ngx_strcmp(cvmfs_rtt_regs[i].root, root_canon) == 0) {
52 reg = &cvmfs_rtt_regs[i]; /* reload: update in place */
53 break;
54 }
55 }
56 if (reg == NULL) {
57 if (cvmfs_rtt_regs_n >= CVMFS_PROBE_MAX_EXPORTS
58 || ngx_strlen(root_canon) >= sizeof(reg->root))
59 {
60 return;
61 }
62 reg = &cvmfs_rtt_regs[cvmfs_rtt_regs_n++];
63 }
64 ngx_cpystrn((u_char *) reg->root, (u_char *) root_canon,
65 sizeof(reg->root));
66 reg->interval = (interval > 0) ? interval : 60;
67 reg->pool[0] = '\0';
68 if (pool_name != NULL && pool_name->len > 0
69 && pool_name->len < sizeof(reg->pool))
70 {
71 ngx_memcpy(reg->pool, pool_name->data, pool_name->len);
72 reg->pool[pool_name->len] = '\0';
73 }
74}

95/* One nonblocking connect, timed. Returns RTT µs, or -1 on any failure
96 * (refused, unreachable, timeout, resolution failure). Shared with the
97 * on-demand geo-answer probe (geo_answer.c) — ONE connect-RTT implementation. */
98long
99brix_cvmfs_connect_rtt_us(const char *host, int port, int timeout_ms)
100{
101 struct addrinfo hints, *ai = NULL;
102 struct pollfd pfd;
103 struct timespec t0, t1;
104 char svc[8];
105 int fd, soerr = 0;
106 socklen_t slen = sizeof(soerr);
107 long us = -1;
108
109 ngx_memzero(&hints, sizeof(hints));
110 hints.ai_family = AF_UNSPEC;
111 hints.ai_socktype = SOCK_STREAM;
112 hints.ai_flags = AI_NUMERICSERV | AI_ADDRCONFIG;
113 snprintf(svc, sizeof(svc), "%d", port);
114 if (getaddrinfo(host, svc, &hints, &ai) != 0 || ai == NULL) {
115 return -1;
116 }
117 fd = socket(ai->ai_family, SOCK_STREAM | SOCK_NONBLOCK, 0); /* vfs-seam-allow: probe socket, non-export resource */
118 if (fd < 0) {
119 freeaddrinfo(ai);
120 return -1;
121 }
122 clock_gettime(CLOCK_MONOTONIC, &t0);
123 if (connect(fd, ai->ai_addr, ai->ai_addrlen) == 0) {
124 clock_gettime(CLOCK_MONOTONIC, &t1);
125 us = (t1.tv_sec - t0.tv_sec) * 1000000L
126 + (t1.tv_nsec - t0.tv_nsec) / 1000L;
127 } else if (errno == EINPROGRESS) {
128 pfd.fd = fd;
129 pfd.events = POLLOUT;
130 if (poll(&pfd, 1, timeout_ms) == 1
131 && getsockopt(fd, SOL_SOCKET, SO_ERROR, &soerr, &slen) == 0
132 && soerr == 0)
133 {
134 clock_gettime(CLOCK_MONOTONIC, &t1);
135 us = (t1.tv_sec - t0.tv_sec) * 1000000L
136 + (t1.tv_nsec - t0.tv_nsec) / 1000L;
137 }
138 }
139 close(fd); /* vfs-seam-allow: probe socket, non-export resource */
140 freeaddrinfo(ai);
141 return us;
142}

172/* Render the per-endpoint probe outcome as one bounded "; "-joined line:
173 * "host:port rank=N ewma=Xus rtt=Y" — the full evidence trail behind a
174 * ranking decision, so an operator can see WHY endpoint order changed. */
175static void
176cvmfs_probe_table(const cvmfs_probe_ctx_t *pc, const double *metric,
177 const int *ranks, char *buf, size_t cap)
178{
179 size_t off = 0;
180 char samp[32];
181 int i, w;
182
183 buf[0] = '\0';
184 for (i = 0; i < pc->n && off + 1 < cap; i++) {
185 if (pc->sample_us[i] < 0) {
186 snprintf(samp, sizeof(samp), "UNREACHABLE");
187 } else {
188 snprintf(samp, sizeof(samp), "%ldus", pc->sample_us[i]);
189 }
190 w = snprintf(buf + off, cap - off, "%s%s:%d rank=%d ewma=%.0fus rtt=%s",
191 (i > 0) ? "; " : "", pc->hosts[i], pc->ports[i],
192 ranks[i], metric[i], samp);
193 if (w < 0 || (size_t) w >= cap - off) {
194 break;
195 }
196 off += (size_t) w;
197 }
198}

200/* Log per-endpoint reachability TRANSITIONS (probe ok → fail and back) at
201 * WARN/NOTICE — steady state stays quiet, the flap is what matters. */
202static void
203cvmfs_probe_log_reachability(cvmfs_probe_ctx_t *pc, ngx_log_t *log)
204{
205 int i, failed;
206
207 for (i = 0; i < pc->n; i++) {
208 failed = (pc->sample_us[i] < 0);
209 if (failed == pc->probe_down[i]) {
210 continue;
211 }
212 pc->probe_down[i] = failed;
213 if (failed) {
214 ngx_log_error(NGX_LOG_WARN, log, 0,
215 "cvmfs rtt probe: origin %s:%d UNREACHABLE (connect probe "
216 "failed/timed out within %dms) - penalised in ranking at "
217 "%.0fus ewma until it answers again",
218 pc->hosts[i], pc->ports[i], CVMFS_PROBE_TIMEOUT_MS,
219 pc->ewma_us[i]);
220 } else {
221 ngx_log_error(NGX_LOG_NOTICE, log, 0,
222 "cvmfs rtt probe: origin %s:%d reachable again "
223 "(rtt %ldus, ewma %.0fus - preference recovers as the "
224 "ewma decays)",
225 pc->hosts[i], pc->ports[i], pc->sample_us[i],
226 pc->ewma_us[i]);
227 }
228 }
229}

241/* event-loop side: fold EWMA, rank, push, re-arm. Every rank ORDER change is
242 * logged with the full evidence table (NOTICE — this is the moment the next
243 * fill starts going somewhere else); steady-state ticks stay at INFO. */
244static void
245cvmfs_probe_done(ngx_event_t *ev)
246{
247 ngx_thread_task_t *task = ev->data;
248 cvmfs_probe_ctx_t *pc = task->ctx;
249 brix_sd_instance_t *inst;
250 double metric[SD_HTTP_EP_MAX] = { 0.0 };
251 char tbl[1024];
252 int ranks[SD_HTTP_EP_MAX] = { 0 };
253 int i, best;
254 int first_tick, changed, prev_best;
255
256 for (i = 0; i < pc->n; i++) {
257 double s = (pc->sample_us[i] < 0) ? (double) CVMFS_PROBE_FAIL_US
258 : (double) pc->sample_us[i];
259 pc->ewma_us[i] = (pc->ewma_us[i] == 0.0)
260 ? s : pc->ewma_us[i] * 0.75 + s * 0.25;
261 metric[i] = pc->ewma_us[i];
262 }
263 brix_cvmfs_rank_by_metric(metric, pc->n, ranks);
264
265 cvmfs_probe_log_reachability(pc, ev->log);
266
267 inst = cvmfs_unwrap_http(
268 brix_vfs_backend_resolve(pc->reg->root, ev->log));
269 if (inst != NULL) {
270 sd_http_set_ranks(inst, ranks, pc->n);
271 }
272
273 first_tick = (pc->prev_ranks[0] < 0);
274 changed = 0;
275 for (i = 0; i < pc->n && !first_tick; i++) {
276 if (ranks[i] != pc->prev_ranks[i]) {
277 changed = 1;
278 }
279 }
280 best = cvmfs_rank_zero_index(ranks, pc->n);
281 cvmfs_probe_table(pc, metric, ranks, tbl, sizeof(tbl));
282
283 if (first_tick) {
284 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
285 "cvmfs rtt initial ranking (export %s): preferred %s:%d - %s",
286 pc->reg->root, pc->hosts[best], pc->ports[best], tbl);
287 } else if (changed) {
288 prev_best = cvmfs_rank_zero_index(pc->prev_ranks, pc->n);
289 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
290 "cvmfs rtt ranking CHANGED (export %s): preferred %s:%d -> %s:%d "
291 "- %s",
292 pc->reg->root, pc->hosts[prev_best], pc->ports[prev_best],
293 pc->hosts[best], pc->ports[best], tbl);
294 } else {
295 ngx_log_error(NGX_LOG_INFO, ev->log, 0,
296 "cvmfs rtt ranks (export %s): preferred %s:%d - %s",
297 pc->reg->root, pc->hosts[best], pc->ports[best], tbl);
298 }
299 for (i = 0; i < pc->n; i++) {
300 pc->prev_ranks[i] = ranks[i];
301 }
302
303 if (!ngx_exiting) {
304 ngx_add_timer(&pc->timer, (ngx_msec_t) pc->reg->interval * 1000
305 + (ngx_msec_t) (ngx_random() % 1000));
306 }
307}

309/* timer side: post the probe task (skip the tick if no pool resolves) */
310static void
311cvmfs_probe_fire(ngx_event_t *ev)
312{
313 cvmfs_probe_ctx_t *pc = ev->data;
314 ngx_thread_pool_t *pool;
315 ngx_str_t pname;
316
317 if (pc->reg->pool[0] != '\0') {
318 pname.data = (u_char *) pc->reg->pool;
319 pname.len = ngx_strlen(pc->reg->pool);
320 } else {
321 ngx_str_set(&pname, "default");
322 }
323 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, &pname);
324 if (pool == NULL || ngx_thread_task_post(pool, pc->task) != NGX_OK) {
325 if (!ngx_exiting) {
326 ngx_add_timer(ev, (ngx_msec_t) pc->reg->interval * 1000);
327 }
328 }
329 /* re-armed by cvmfs_probe_done after the samples land */
330}

332ngx_int_t
333brix_cvmfs_rtt_init_worker(ngx_cycle_t *cycle)
334{
335 ngx_uint_t i;
336
337 for (i = 0; i < cvmfs_rtt_regs_n; i++) {
338 const cvmfs_rtt_reg_t *reg = &cvmfs_rtt_regs[i];
339 ngx_thread_task_t *task;
340 cvmfs_probe_ctx_t *pc;
341 const char *host;
342 int port, idx;
343
344 task = ngx_thread_task_alloc(cycle->pool, sizeof(cvmfs_probe_ctx_t));
345 if (task == NULL) {
346 return NGX_ERROR;
347 }
348 pc = task->ctx;
349 pc->task = task;
350 pc->reg = reg;
351 for (idx = 0; idx < SD_HTTP_EP_MAX; idx++) {
352 if (brix_vfs_backend_http_endpoint_at(reg->root, idx, &host,
353 &port) != 0)
354 {
355 break;
356 }
357 ngx_cpystrn((u_char *) pc->hosts[idx], (u_char *) host,
358 sizeof(pc->hosts[idx]));
359 pc->ports[idx] = port;
360 }
361 pc->n = idx;
362 if (pc->n == 0) {
363 continue; /* not an http backend (misconfig) */
364 }
365 for (idx = 0; idx < SD_HTTP_EP_MAX; idx++) {
366 pc->prev_ranks[idx] = -1; /* -1 = no ranking pushed yet */
367 }
368
369 brix_task_bind(task, cvmfs_probe_thread, cvmfs_probe_done);
370 task->event.log = cycle->log;
371
372 pc->timer.handler = cvmfs_probe_fire;
373 pc->timer.data = pc;
374 pc->timer.log = cycle->log;
375 /* first probe within 500 ms so ranks exist before the first fill */
376 ngx_add_timer(&pc->timer, (ngx_msec_t) (ngx_random() % 500) + 1);
377 }
378 return NGX_OK;
