5/*
6 * WHAT: Handle upstream read events — accumulate response header (8 bytes), optionally use Linux splice for zero-copy,
7 * then relay response body back to the client. Handles kXR_attn notifications, bootstrap responses, and forwarding mode.
8 * WHY: The proxy must collect complete XRootD wire frames before relaying them to clients. Each frame has an 8-byte header (status + dlen)
9 * followed by variable-length payload. nginx-xrootd uses edge-triggered epoll so it must drain all available data from each read event
10 * rather than returning early — otherwise buffered data would be unread until the next TCP segment arrives.
11 * HOW: Extract uconn and proxy ctx from rev->data; check client destruction/timeout; accumulate rhdr via recv() loop until XRD_RESPONSE_HDR_LEN;
12 * attempt Linux splice for zero-copy if FORWARDING state + resp_dlen fits in buffer; allocate resp_body via ngx_alloc; accumulate body data;
13 * handle kXR_attn relay (unsolicited notifications, special stream ID); process bootstrap or forwarding responses based on proxy->state.
14 */
15

17void
18brix_proxy_read_handler(ngx_event_t *rev)
19{
20 ngx_connection_t *uconn = rev->data;
21 brix_proxy_ctx_t *proxy = uconn->data;
22 brix_ctx_t *ctx;
23 ssize_t n;
24
25 /* Guard against use-after-free: if the client pool was freed while a
26 * reconnect was in progress, uconn->data may be NULL or stale. */
27 if (proxy == NULL) {
28 return;
29 }
30 ctx = proxy->client_ctx;
31
32 if (ctx == NULL || ctx->destroyed) {
33 brix_proxy_cleanup(proxy);
34 return;
35 }
36
37 if (rev->timedout) {
38 brix_proxy_abort(proxy, "proxy: upstream read timeout");
39 return;
40 }
41
42 for (;;) {
43 /*
44 * Teardown guard (UAF + busy-loop): a handler invoked in a previous
45 * iteration (handle_bootstrap, relay_to_client, …) may have called
46 * brix_proxy_abort(), which tears the proxy down and clears
47 * ctx->proxy *without* changing proxy->state. The per-state branches
48 * below key off proxy->state, so without this check the loop would
49 * `continue` straight back into the same handler and re-process the
50 * still-buffered upstream frame on a dead proxy forever (observed: a
51 * bad-credential upstream spinning a worker at 100% CPU, re-aborting
52 * ~500K times/sec and leaking until OOM). Once the proxy is no longer
53 * the connection's live proxy, stop immediately.
54 */
55 if (ctx->proxy != proxy) {
56 return;
57 }
58
59 /* accumulate response header (8 bytes) */ if (proxy->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
60 size_t need = XRD_RESPONSE_HDR_LEN - proxy->rhdr_pos;
61
62 n = uconn->recv(uconn, proxy->rhdr + proxy->rhdr_pos, need);
63 if (n == NGX_AGAIN) {
64 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
65 brix_proxy_abort(proxy, "proxy: read arm failed (hdr)");
66 return;
67 }
68 if (proxy->conf != NULL && proxy->conf->proxy.read_timeout > 0) {
69 ngx_add_timer(rev, proxy->conf->proxy.read_timeout);
70 }
71 return;
72 }
73 if (n <= 0) {
74 brix_proxy_abort(proxy, "proxy: upstream closed (hdr)");
75 return;
76 }
77
78 proxy->rhdr_pos += (size_t) n;
79 if (proxy->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
80 continue;
81 }
82
83 {
84 ServerResponseHdr *hdr = (ServerResponseHdr *)(void *) proxy->rhdr;
85 proxy->resp_status = ntohs(hdr->status);
86 proxy->resp_dlen = ntohl(hdr->dlen);
87 }
88
89 ngx_log_debug(NGX_LOG_DEBUG_STREAM, uconn->log, 0,
90 "xrootd proxy: upstream hdr status=%d dlen=%uz state=%d",
91 (int) proxy->resp_status,
92 (size_t) proxy->resp_dlen,
93 (int) proxy->state);
94
95 if (proxy->resp_dlen > 0) {
96#ifdef __linux__
97 /* Attempt zero-copy splice for plain-text read responses. */
98 if (proxy->state == XRD_PX_FORWARDING) {
99 ngx_int_t srt = brix_proxy_try_splice(proxy);
100 if (srt == NGX_OK) {
101 ngx_log_debug(NGX_LOG_DEBUG_STREAM, uconn->log, 0,
102 "xrootd proxy: splice started dlen=%uz",
103 (size_t) proxy->resp_dlen);
104 /* Splice started — pump drives I/O; don't buffer body. */
105 return;
106 }
107 if (srt == NGX_ERROR) {
108 /* PXY-6: try_splice aborted the session to avoid frame
109 * corruption — proxy is gone, do not touch it. */
110 return;
111 }
112 /* NGX_DECLINED — fall through to the buffered body path. */
113 }
114#endif
115 if (proxy->resp_dlen > BRIX_PROXY_MAX_BODY) {
116 brix_proxy_abort(proxy, "proxy: upstream response body too large");
117 return;
118 }
119 proxy->resp_body = ngx_alloc(proxy->resp_dlen + 1, uconn->log);
120 if (proxy->resp_body == NULL) {
121 brix_proxy_abort(proxy, "proxy: body alloc failed");
122 return;
123 }
124 ngx_log_debug(NGX_LOG_DEBUG_STREAM, uconn->log, 0,
125 "xrootd proxy: buffering body dlen=%uz",
126 (size_t) proxy->resp_dlen);
127 proxy->resp_body[proxy->resp_dlen] = '\0';
128 proxy->resp_body_pos = 0;
129 }
130 }
131
132 /* accumulate response body */#ifdef __linux__
133 /* If splice is active, upstream readable means more data to pump. */
134 if (proxy->splice_active) {
135 brix_proxy_splice_pump(proxy);
136 return;
137 }
138#endif
139 if (proxy->resp_dlen > 0 && proxy->resp_body_pos < proxy->resp_dlen) {
140 size_t need = proxy->resp_dlen - proxy->resp_body_pos;
141
142 n = uconn->recv(uconn,
143 proxy->resp_body + proxy->resp_body_pos, need);
144 if (n == NGX_AGAIN) {
145 ngx_log_debug(NGX_LOG_DEBUG_STREAM, uconn->log, 0,
146 "xrootd proxy: body AGAIN pos=%uz dlen=%uz",
147 (size_t) proxy->resp_body_pos,
148 (size_t) proxy->resp_dlen);
149 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
150 brix_proxy_abort(proxy, "proxy: read arm failed (body)");
151 return;
152 }
153 if (proxy->conf != NULL && proxy->conf->proxy.read_timeout > 0) {
154 ngx_add_timer(rev, proxy->conf->proxy.read_timeout);
155 }
156 return;
157 }
158 if (n <= 0) {
159 brix_proxy_abort(proxy, "proxy: upstream closed (body)");
160 return;
161 }
162
163 proxy->resp_body_pos += (size_t) n;
164 if (proxy->resp_body_pos < proxy->resp_dlen) {
165 continue;
166 }
167 }

169 /* kXR_status two-phase: expand body to include page data */ /*
170 * kXR_status (pgread/pgwrite) has hdr.dlen=24 (fixed header) but
171 * bdy.dlen more bytes of page data follow the standard body.
172 * After reading the 24-byte fixed body, re-allocate to include those
173 * extra bytes so the relay path sees a single contiguous buffer.
174 */
175 if (proxy->resp_status == kXR_status
176 && proxy->resp_dlen == 24
177 && proxy->resp_body_pos == 24)
178 {
179 uint32_t extra;
180 u_char *new_body;
181
182 ngx_memcpy(&extra, proxy->resp_body + 12, 4);
183 extra = ntohl(extra);
184
185 ngx_log_debug(NGX_LOG_DEBUG_STREAM, uconn->log, 0,
186 "xrootd proxy: kXR_status two-phase expand extra=%uz resptype=%d",
187 (size_t) extra,
188 (int)(unsigned char) proxy->resp_body[7]);
189
190 if (extra > BRIX_PROXY_MAX_BODY) {
191 brix_proxy_abort(proxy,
192 "proxy: kXR_status extra data too large");
193 return;
194 }
195 if (extra > 0) {
196 new_body = ngx_alloc(24 + extra + 1, uconn->log);
197 if (new_body == NULL) {
198 brix_proxy_abort(proxy,
199 "proxy: kXR_status extra body alloc failed");
200 return;
201 }
202 ngx_memcpy(new_body, proxy->resp_body, 24);
203 ngx_free(proxy->resp_body);
204 new_body[24 + extra] = '\0';
205 proxy->resp_body = new_body;
206 proxy->resp_dlen += extra;
207 /* resp_body_pos=24; body loop continues for extra bytes */
208 continue;
209 }
210 }
211
212 /* full response received */
213 /*
214 * Protocol Correctness: Handle kXR_attn (unsolicited or async notification).
215 * These frames can arrive while IDLE (unsolicited) or while FORWARDING
216 * (after a kXR_waitresp). We relay them using their own stream ID and
217 * do NOT satisfy the FORWARDING state.
218 */
219 if (proxy->resp_status == kXR_attn) {
220 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, proxy->client_conn->log, 0,
221 "xrootd proxy: relaying kXR_attn frame (dlen=%uz)",
222 (size_t) proxy->resp_dlen);
223
224 /* Relay directly to client; don't use proxy->fwd_streamid */
225 {
226 size_t total = XRD_RESPONSE_HDR_LEN + proxy->resp_dlen;
227 u_char *buf = ngx_palloc(proxy->client_conn->pool, total);
228 if (buf != NULL) {
229 ngx_memcpy(buf, proxy->rhdr, XRD_RESPONSE_HDR_LEN);
230 if (proxy->resp_dlen > 0) {
231 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, proxy->resp_body,
232 proxy->resp_dlen);
233 }
234 brix_queue_response(proxy->client_ctx, proxy->client_conn,
235 buf, total);
236 }
237 }
238
239 if (proxy->resp_body != NULL) {
240 ngx_free(proxy->resp_body);
241 proxy->resp_body = NULL;
242 }
243 proxy->rhdr_pos = 0;
244 proxy->resp_dlen = 0;
245 proxy->resp_body_pos = 0;
246 continue; /* Loop to check for the actual expected response */
247 }
248
249 if (proxy->state == XRD_PX_BOOTSTRAP) {
250 brix_proxy_handle_bootstrap(proxy);
251 /*
252 * Don't return here — the upstream may have already delivered
253 * subsequent bootstrap responses (protocol + login) in the same
254 * TCP segment. In edge-triggered epoll, returning now would leave
255 * that data unread with no further event to wake us up. Continue
256 * the loop; if there is no more data recv() will return NGX_AGAIN
257 * and we arm the read event there.
258 *
259 * brix_proxy_handle_bootstrap transitions state to XRD_PX_IDLE
260 * (and may call brix_proxy_flush / arm the read itself) when
261 * bootstrap completes, so check before looping.
262 */
263 if (proxy->state != XRD_PX_BOOTSTRAP) {
264 return; /* bootstrap done or aborted */
265 }
266 continue; /* read next bootstrap response from buffered data */
267 }
268
269 if (proxy->state == XRD_PX_FORWARDING) {
270 /* Under-draining-splice fallback: the body just accumulated is the
271 * RAW remainder of a spliced read (header already sent) — relay it
272 * verbatim and finish, never through the header-building relay. */
273 if (proxy->splice_fallback) {
274 brix_proxy_splice_fallback_finish(proxy);
275 return;
276 }
277 ngx_log_debug(NGX_LOG_DEBUG_STREAM, uconn->log, 0,
278 "xrootd proxy: relay_to_client status=%d dlen=%uz",
279 (int) proxy->resp_status,
280 (size_t) proxy->resp_dlen);
281 brix_proxy_relay_to_client(proxy);
282 /* relay_to_client resets rhdr_pos and resp_body for the
283 * next frame if status was kXR_oksofar; otherwise it
284 * resets ctx->state and returns, ending the loop. */
285 if (proxy->state == XRD_PX_FORWARDING) {
286 /* More kXR_oksofar frames expected — loop to read next */
287 continue;
288 }
289 /* Cancel any pending read timeout — response was fully relayed.
290 * Without this, the timer set during body accumulation (NGX_AGAIN)
291 * fires 60s later, triggering the read handler with a stale IDLE
292 * state and causing "unexpected state" abort + SIGSEGV. */
293 if (rev->timer_set) {
294 ngx_del_timer(rev);
295 }
296 return;
297 }
298
299 brix_proxy_abort(proxy, "proxy: unexpected state in read handler");
300 return;
301 }
