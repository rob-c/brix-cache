1/*
2 * tagging.c — S3 object tagging + canned bucket/object subresources (phase-43 W5).
3 *
4 * WHAT:
5 * - Object tagging: x-amz-tagging on PutObject, and GET/PUT/DELETE
6 * /<key>?tagging. The tag set is stored as a single "k1=v1&k2=v2"
7 * (URL-encoded) blob in a dedicated xattr beside the object — no new on-disk
8 * structures, and it is invisible to every other code path.
9 * - Canned probe-satisfiers: GetBucketVersioning (disabled),
10 * GetBucketAcl / GetObjectAcl (owner FULL_CONTROL), and
11 * GET ?cors (NoSuchCORSConfiguration). Mutating those subresources
12 * (PUT ?versioning / ?acl) is explicitly answered 501 NotImplemented rather
13 * than silently accepted (see the phase-43 non-goals).
14 *
15 * WHY: Rucio and data-management tooling label objects with tags, and many SDKs
16 * probe the canned subresources during setup; answering them well-formed (or
17 * with an honest error) keeps unmodified clients working.
18 */

70/* Read the stored tag blob into out (NUL-terminated). Returns length, 0 if
71 * none, -1 on error. */
72static ssize_t
73s3_tag_load(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
74 const char *fs_path, char *out, size_t outsz)
75{
76 brix_vfs_ctx_t vctx;
77 ssize_t n;
78
79 /*
80 * Impersonation: read the tag xattr AS THE MAPPED USER via the VFS xattr
81 * surface, which delegates to the brokered confined-xattr helper — NOT a
82 * worker-side fgetxattr on a broker-opened fd, which would evaluate the
83 * xattr-read DAC as the unprivileged worker (svc). Mirrors the WebDAV
84 * LOCK/PROPPATCH path (webdav/prop_xattr.c). Off impersonation the helper
85 * is a plain path-based getxattr.
86 */
87 s3_tag_vfs_ctx(r, fs_path, cf, &vctx);
88 n = brix_vfs_getxattr(&vctx, S3_TAG_XATTR, out, outsz - 1);
89 if (n < 0) {
90 /* phase74-fp: ENOTSUP == EOPNOTSUPP on Linux but they are distinct
91 * errnos on other POSIX systems — both are checked deliberately for
92 * portability, so the "equivalent nested operands" finding is moot. */
93 /* NOLINTNEXTLINE(misc-redundant-expression) */
94 if (errno == ENODATA || errno == ENOTSUP || errno == EOPNOTSUPP) {
95 return 0; /* object readable, just carries no tags */
96 }
97 return -1; /* denied / missing object → caller maps 403/404 */
98 }
99 out[n] = '\0';
100 return n;
101}

103/* Persist (or, when blob_len==0, remove) the tag blob. Returns NGX_OK/ERROR. */
104static ngx_int_t
105s3_tag_store(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
106 const char *fs_path, const char *blob, size_t blob_len, int remove_it)
107{
108 brix_vfs_ctx_t vctx;
109
110 /*
111 * Impersonation (the bug this fixes): the old path opened O_RDONLY via the
112 * broker (READ-checked) then did a worker-side fsetxattr/fremovexattr — so the
113 * xattr WRITE DAC was evaluated as the unprivileged worker (svc), NOT the
114 * mapped user. Effect: (a) the legitimate owner's PutObjectTagging failed
115 * (svc is "other" on the owner's file), and (b) on a permissive backing FS a
116 * requester who could merely READ another tenant's object could mutate/remove
117 * its tags. Route the MUTATION through the VFS xattr surface, which delegates
118 * to the brokered confined-xattr helper (broker setfsuid -> f{set,remove}xattr
119 * as the mapped user, user.* namespace filter), exactly like WebDAV
120 * LOCK/PROPPATCH. Off impersonation it is a plain path-based syscall
121 * (unchanged behaviour). Note: VFS set/remove xattr are intentionally NOT
122 * allow_write-gated, matching the prior direct-helper behaviour.
123 */
124 s3_tag_vfs_ctx(r, fs_path, cf, &vctx);
125
126 if (remove_it) {
127 if (brix_vfs_removexattr(&vctx, S3_TAG_XATTR) == NGX_OK) {
128 return NGX_OK;
129 }
130 return (errno == ENODATA) ? NGX_OK : NGX_ERROR;
131 }
132
133 return brix_vfs_setxattr(&vctx, S3_TAG_XATTR, blob, blob_len, 0) == NGX_OK
134 ? NGX_OK : NGX_ERROR;
135}

137/* validation */
138/* A tag blob is "k=v&k=v..."; reject control bytes (defends the GET XML emit and
139 * keeps the stored form a clean query string). */
140static int
141s3_tag_blob_valid(const char *s, size_t len)
142{
143 size_t i;
144 if (len > S3_TAG_MAX) {
145 return 0;
146 }
147 for (i = 0; i < len; i++) {
148 unsigned char c = (unsigned char) s[i];
149 if (c < 0x20 || c == 0x7f) {
150 return 0;
151 }
152 }
153 return 1;
154}

156/* GET /<key>?tagging */
157ngx_int_t
158s3_handle_get_object_tagging(ngx_http_request_t *r, const char *fs_path,
159 ngx_http_s3_loc_conf_t *cf)
160{
161 char blob[S3_TAG_MAX + 1];
162 ssize_t blen;
163 u_char *xml;
164 size_t xml_len = 0;
165 size_t xml_capacity = S3_TAG_XML_MAX;
166 ngx_buf_t *buf;
167 char *p, *amp;
168
169 blen = s3_tag_load(r, cf, fs_path, blob, sizeof(blob));
170 if (blen < 0) {
171 return s3_fail(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
172 "The specified key does not exist.",
173 BRIX_S3_EVENT_NO_SUCH_KEY);
174 }
175
176 xml = ngx_palloc(r->pool, xml_capacity);
177 if (xml == NULL) {
178 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
179 return NGX_HTTP_INTERNAL_SERVER_ERROR;
180 }
181
182 XML_APPEND("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
183 XML_APPEND("<Tagging xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">"
184 "<TagSet>");
185
186 /* Walk "k=v&k=v", URL-decoding each component, emit <Tag><Key/><Value/>. */
187 p = blob;
188 while (blen > 0 && p != NULL && *p != '\0') {
189 char kbuf[512], vbuf[512];
190 char *eq;
191 size_t klen, vlen;
192
193 amp = strchr(p, '&');
194 if (amp != NULL) {
195 *amp = '\0';
196 }
197 eq = strchr(p, '=');
198 if (eq != NULL) {
199 *eq = '\0';
200 klen = brix_http_urldecode((u_char *) p, ngx_strlen(p),
201 kbuf, sizeof(kbuf), BRIX_URLDECODE_PLUS_TO_SPACE)
202 == BRIX_URLDECODE_OK ? ngx_strlen(kbuf) : 0;
203 vlen = brix_http_urldecode((u_char *) (eq + 1),
204 ngx_strlen(eq + 1), vbuf, sizeof(vbuf),
205 BRIX_URLDECODE_PLUS_TO_SPACE)
206 == BRIX_URLDECODE_OK ? ngx_strlen(vbuf) : 0;
207 XML_APPEND("<Tag>");
208 XML_APPEND_ELEM("Key", kbuf, klen);
209 XML_APPEND_ELEM("Value", vbuf, vlen);
210 XML_APPEND("</Tag>");
211 }
212 if (amp == NULL) {
213 break;
214 }
215 p = amp + 1;
216 }
217
218 XML_APPEND("</TagSet></Tagging>");
219
220 buf = ngx_create_temp_buf(r->pool, xml_len + 4);
221 if (buf == NULL) {
222 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
223 return NGX_HTTP_INTERNAL_SERVER_ERROR;
224 }
225 buf->last = ngx_cpymem(buf->last, xml, xml_len);
226 buf->last_buf = 1;
227 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
228 (ngx_str_t) ngx_string("application/xml"), buf);
229}

271/*
272 * s3_xml_node_is — element-node name predicate.
273 *
274 * WHAT: true iff node is an XML_ELEMENT_NODE whose tag name equals name.
275 * WHY: the tagging parse repeats this "right kind of element?" guard at every
276 * nesting level (TagSet, Tag, Key/Value); folding it into one predicate keeps
277 * the walk loops flat and drops their branch count.
278 * HOW: guard the node type first, then a single xmlStrcmp on the name.
279 */
280static int
281s3_xml_node_is(const xmlNode *node, const char *name)
282{
283 return node->type == XML_ELEMENT_NODE
284 && xmlStrcmp(node->name, (const xmlChar *) name) == 0;
285}

287/*
288 * s3_tag_append_pair — append one URL-encoded "key=value" pair to out.
289 *
290 * WHAT: writes "&" (when out already holds a pair), then url-encoded key, "=",
291 * url-encoded value into out[*pos], advancing *pos; every write is bounded by
292 * outsz so a full buffer silently truncates (the caller rejects pos>=outsz).
293 * WHY: the byte-exact append/bounds arithmetic is the bulk of the parser's
294 * complexity; isolating it keeps the tree walk readable and the encoding
295 * behaviour identical to the original inline body.
296 * HOW: encode into a scratch buffer, memcpy the bounded span, mirror the exact
297 * pos<outsz / pos+n<outsz predicates the original used for each token.
298 */
299static void
300s3_tag_append_pair(char *out, size_t outsz, size_t *pos,
301 const xmlChar *key, const xmlChar *val)
302{
303 char enc[512];
304 size_t n;
305
306 if (*pos != 0 && *pos < outsz) {
307 out[(*pos)++] = '&';
308 }
309 brix_http_urlencode(key, ngx_strlen((char *) key), enc, sizeof(enc), "");
310 n = ngx_strlen(enc);
311 if (*pos + n < outsz) { ngx_memcpy(out + *pos, enc, n); *pos += n; }
312 if (*pos < outsz) { out[(*pos)++] = '='; }
313 brix_http_urlencode(val, ngx_strlen((char *) val), enc, sizeof(enc), "");
314 n = ngx_strlen(enc);
315 if (*pos + n < outsz) { ngx_memcpy(out + *pos, enc, n); *pos += n; }
316}

318/*
319 * s3_tag_emit_one — extract a single <Tag>'s Key/Value and append the pair.
320 *
321 * WHAT: scans the children of a <Tag> node for <Key> and <Value>, and when both
322 * are present appends the encoded pair to out via s3_tag_append_pair; frees the
323 * xmlNodeGetContent allocations before returning.
324 * WHY: pulling the inner Key/Value scan out of the tag loop removes one full
325 * nesting level (and its branches) from the top-level parser.
326 * HOW: single pass over children collecting key/val, then a bounded append.
327 */
328static void
329s3_tag_emit_one(const xmlNode *tag, char *out, size_t outsz, size_t *pos)
330{
331 const xmlChar *key = NULL, *val = NULL;
332 const xmlNode *kv;
333
334 for (kv = tag->children; kv != NULL; kv = kv->next) {
335 if (kv->type != XML_ELEMENT_NODE) {
336 continue;
337 }
338 if (xmlStrcmp(kv->name, (const xmlChar *) "Key") == 0) {
339 key = xmlNodeGetContent(kv);
340 } else if (xmlStrcmp(kv->name, (const xmlChar *) "Value") == 0) {
341 val = xmlNodeGetContent(kv);
342 }
343 }
344 if (key != NULL && val != NULL) {
345 s3_tag_append_pair(out, outsz, pos, key, val);
346 }
347 if (key != NULL) { xmlFree((void *) key); }
348 if (val != NULL) { xmlFree((void *) val); }
349}

351/*
352 * s3_tag_walk_tagsets — walk every <Tag> under every <TagSet> child of root.
353 *
354 * WHAT: iterates root's <TagSet> children, and within each its <Tag> children,
355 * emitting each into out; returns the final byte count via *pos.
356 * WHY: separates the pure tree traversal from doc lifetime + root validation,
357 * so neither function carries both concerns' branches.
358 * HOW: two flat loops gated by the s3_xml_node_is predicate.
359 */
360static void
361s3_tag_walk_tagsets(const xmlNode *root, char *out, size_t outsz, size_t *pos)
362{
363 const xmlNode *tagset, *tag;
364
365 for (tagset = root->children; tagset != NULL; tagset = tagset->next) {
366 if (!s3_xml_node_is(tagset, "TagSet")) {
367 continue;
368 }
369 for (tag = tagset->children; tag != NULL; tag = tag->next) {
370 if (!s3_xml_node_is(tag, "Tag")) {
371 continue;
372 }
373 s3_tag_emit_one(tag, out, outsz, pos);
374 }
375 }
376}

378/*
379 * s3_tag_blob_from_xml — parse a Tagging document into the stored blob form.
380 *
381 * WHAT: parses <Tagging><TagSet><Tag><Key>k</Key><Value>v</Value></Tag>...>
382 * into the stored "k=v&k=v" (URL-encoded) form via libxml2 with the same
383 * hardened posture (no-net, no-blanks) as the DeleteObjects parser.
384 * WHY: PUT ?tagging carries the tag set as XML; this converts it to the flat
385 * xattr blob without changing the on-disk form or the error contract.
386 * HOW: parse+root-validate, delegate the walk, then NUL-terminate — rejecting an
387 * overflowed buffer exactly as before (pos>=outsz → NGX_ERROR).
388 */
389static ngx_int_t
390s3_tag_blob_from_xml(ngx_http_request_t *r, const u_char *body, size_t len,
391 char *out, size_t outsz, size_t *out_len)
392{
393 xmlDocPtr doc;
394 xmlNodePtr root;
395 size_t pos = 0;
396 int options = XML_PARSE_NONET | XML_PARSE_NOBLANKS;
397
398 (void) r; /* frozen signature: request handle unused by the pure parse */
399
400 doc = xmlReadMemory((const char *) body, (int) len, "tagging.xml", NULL,
401 options);
402 if (doc == NULL) {
403 return NGX_ERROR;
404 }
405
406 root = xmlDocGetRootElement(doc);
407 if (root == NULL || xmlStrcmp(root->name, (const xmlChar *) "Tagging") != 0) {
408 xmlFreeDoc(doc);
409 return NGX_ERROR;
410 }
411
412 s3_tag_walk_tagsets(root, out, outsz, &pos);
413
414 xmlFreeDoc(doc);
415 if (pos >= outsz) {
416 return NGX_ERROR;
417 }
418 out[pos] = '\0';
419 *out_len = pos;
420 return NGX_OK;
421}

424s3_put_object_tagging_body_handler(ngx_http_request_t *r)
425{
426 ngx_http_s3_req_ctx_t *rx;
427 ngx_http_s3_loc_conf_t *cf;

431 size_t blob_len = 0;
432 ngx_int_t rc;
433

435 rx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
436 if (rx == NULL || rx->fs_path[0] == '\0') {
437 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);

468 r->headers_out.status = NGX_HTTP_OK;
469 r->headers_out.content_length_n = 0;
470 ngx_http_send_header(r);
471 ngx_http_finalize_request(r, ngx_http_send_special(r, NGX_HTTP_LAST));
472}

500ngx_int_t
501s3_handle_get_acl(ngx_http_request_t *r, const char *fs_path,
502 ngx_http_s3_loc_conf_t *cf)
503{
504 /* Canned owner-FULL_CONTROL ACL — the gateway authorizes via XrdAcc/tokens,
505 * not per-object S3 ACLs, but SDKs probe this and expect a well-formed doc. */
506 char owner[128];
507 size_t n;
508 u_char *xml;
509 size_t xml_len = 0;
510 size_t xml_capacity = 1024;
511 ngx_buf_t *buf;
512
513 /*
514 * A-3/T3: object ACL runs behind the object gate. SigV4 authenticates the
515 * requester; it does NOT prove the named object exists or is readable. A
516 * canned 200 for a GetObjectAcl on an absent (or directory) key leaks
517 * existence and mints a FULL_CONTROL grant for nothing — so, for an object
518 * ?acl (fs_path != NULL), stat the resolved key first (as the mapped user,
519 * via the same VFS surface the tag/HEAD paths use) and answer NoSuchKey when
520 * it is missing, before the canned document is produced. Bucket ?acl
521 * (fs_path == NULL) legitimately has no per-object target and is unchanged.
522 */
523 if (fs_path != NULL) {
524 brix_vfs_ctx_t vctx;
525 brix_vfs_stat_t vst;
526
527 s3_tag_vfs_ctx(r, fs_path, cf, &vctx);
528 if (brix_vfs_stat(&vctx, &vst) != NGX_OK || vst.is_directory) {
529 return s3_fail(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
530 "The specified key does not exist.",
531 BRIX_S3_EVENT_NO_SUCH_KEY);
532 }
533 }
534
535 if (cf->access_key.len > 0) {
536 n = cf->access_key.len < sizeof(owner) - 1
537 ? cf->access_key.len : sizeof(owner) - 1;
538 ngx_memcpy(owner, cf->access_key.data, n);
539 } else {
540 n = sizeof("anonymous") - 1;
541 ngx_memcpy(owner, "anonymous", n);
542 }
543 owner[n] = '\0';
544
545 BRIX_PALLOC_OR_RETURN(xml, r->pool, xml_capacity, NGX_HTTP_INTERNAL_SERVER_ERROR);
546 XML_APPEND("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
547 XML_APPEND("<AccessControlPolicy "
548 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\"><Owner>");
549 XML_APPEND_ELEM("ID", owner, n);
550 XML_APPEND_ELEM("DisplayName", owner, n);
551 XML_APPEND("</Owner><AccessControlList><Grant>"
552 "<Grantee xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" "
553 "xsi:type=\"CanonicalUser\">");
554 XML_APPEND_ELEM("ID", owner, n);
555 XML_APPEND_ELEM("DisplayName", owner, n);
556 XML_APPEND("</Grantee><Permission>FULL_CONTROL</Permission></Grant>"
557 "</AccessControlList></AccessControlPolicy>");
558
559 buf = ngx_create_temp_buf(r->pool, xml_len + 4);
560 if (buf == NULL) {
