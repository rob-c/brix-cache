2 * brix.h — internal API for the native XRootD root:// client library.
3 *
4 * WHAT: Connection/session + metadata/file ops over the XRootD binary protocol,
5 * built directly on the project's wire vocabulary (the src/protocols/root/protocol headers,
6 * shared via libxrdproto). This is the spine that xrdcp/xrdfs sit on.
7 * WHY: A pure-C, libXrdCl-free client (phase-37). Blocking sockets + poll(2)
8 * timeouts; one in-flight request per connection for now (the streamid
9 * counter is the seed for future parallel streams).
10 * HOW: Each request builds its packed ClientXxxRequest struct from wire.h, sets
11 * big-endian fields, and exchanges frames via frame.c. No ngx, no XrdCl.

13 * Clean-room: wire facts come only from the src/protocols/root/protocol headers (cross-checked

54/* Local error sentinels (negative so they never collide with kXR_* codes). */
55#define XRDC_ESOCK (-1) /* connect/socket/timeout failure */
56#define XRDC_EPROTO (-2) /* malformed/unexpected server frame */
57#define XRDC_EUSAGE (-3) /* CLI / argument error */
58#define XRDC_EAUTH (-4) /* server demanded auth we don't (yet) speak */
59#define XRDC_EINTEGRITY (-5)/* data corruption (CRC/checksum mismatch) — NOT
60 * retryable: a re-read yields the same bad bytes, so
61 * the resilient loop MUST fail fast, not spin. */
62#define XRDC_EUNSUPPORTED (-6) /* valid protocol feature this client build lacks;
63 * fatal, because reconnecting cannot add support. */
64#define XRDC_ERESOLVE (-7) /* permanent name-resolution failure (NXDOMAIN / no
65 * address) — NOT retryable: the name will not
66 * resolve on a retry, so the resilient loop must
67 * fail fast instead of burning its stall window.
68 * A *transient* resolver failure (EAI_AGAIN) keeps
69 * XRDC_ESOCK so it is still retried. */
70#define XRDC_EREDIRECT (-8) /* redirect loop / budget exhausted (self-redirect,
71 * bounce to an already-tried target, too many hops)
72 * — NOT retryable: re-issuing the op just walks into
73 * the same loop, so the resilient wrapper must fail
74 * fast rather than chase it for the whole window. */
75#define XRDC_EIO (-9) /* local filesystem I/O error (open/read/write/rename/fstat/truncate/alloc) — permanent, NOT retryable */
76#define XRDC_ENOENT (-10) /* object/path does not exist (HTTP 404 / ENOENT) — permanent, NOT retryable */
