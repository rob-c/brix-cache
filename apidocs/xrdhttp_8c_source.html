1/*
2 * xrdhttp.c — XrdHttp protocol extension support for the nginx-xrootd WebDAV module.
3 *
4 * WHAT: Implements XrdHttp parity features on top of the existing WebDAV handler:
5 * - Request parsing: X-Xrootd-Proto detection, ?xrd.* / ?tpc.* query params,
6 * X-Xrootd-Requuid / X-Xrootd-Tpc-Token header capture.
7 * - Response injection: X-Xrootd-Requuid echo, X-Xrootd-Status kXR error codes,
8 * X-Xrootd-Wait / X-Xrootd-Retry back-pressure headers.
9 * - Redirect dialect: X-Xrootd-Redir-Host/Port + tpc.key / xrd.opaque passthrough.
10 * - TPC shim: synthesise Source:/Destination: headers from ?tpc.src= / ?tpc.dst=.
11 * - Checksum: Digest: header computed on-demand for xrd.want.cksum requests.
12 * - HTTP→kXR status mapping for X-Xrootd-Status.
13 *
14 * WHY: XRootD-aware clients (xrdcp --prefer-xrdhttp, ROOT TFile, davix-get with XRD
15 * hints) negotiate the protocol version via X-Xrootd-Proto and rely on X-Xrootd-Status
16 * to distinguish between "file not found", "not authorised", "server busy" etc. without
17 * parsing HTTP status codes. Without this layer clients fall back to slow error-path
18 * retries or misinterpret error conditions.
19 *
20 * HOW: A single per-request context struct (xrdhttp_req_ctx_t) is allocated from r->pool
21 * by xrdhttp_get_ctx() the first time any XrdHttp function is called. Parsing functions
22 * walk r->headers_in and the raw r->args query string; they write NUL-terminated results
23 * into fixed-size fields (no dynamic allocation) and silently truncate over-length values
24 * so that untrusted client data cannot cause unbounded memory use. All values written to
25 * logs pass through brix_sanitize_log_string().
26 */

49/* context management */
50xrdhttp_req_ctx_t *
51xrdhttp_get_ctx(ngx_http_request_t *r)
52{
53 ngx_http_brix_webdav_req_ctx_t *webdav_ctx;
54
55 webdav_ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
56 if (webdav_ctx != NULL) {
57 /* xrdhttp is the first member — cast is well-defined by C11 §6.7.2.1p15. */
58 return &webdav_ctx->xrdhttp;
59 }
60
61 /* Allocate the full webdav context (xrdhttp fields zero-initialised by pcalloc).
62 * This path is hit on unauthenticated requests where auth_cert.c / auth_token.c
63 * never allocated the context themselves. */
64 BRIX_PCALLOC_OR_RETURN(webdav_ctx, r->pool, sizeof(ngx_http_brix_webdav_req_ctx_t), NULL);
65 ngx_http_set_ctx(r, webdav_ctx, ngx_http_brix_webdav_module);
66 return &webdav_ctx->xrdhttp;
67}

70/*
71 * Copy the value of query parameter <key> from the raw nginx args string into
72 * <dst> (max <dstsz> bytes including NUL terminator). Performs URL-decoding

81

88 | BRIX_HTTP_QUERY_TRUNCATE) > 0;

105/*
106 * Normalize an RFC 3230 algorithm token to the internal name used by
107 * brix_checksum_parse(). Extracts the first token from a comma-separated
108 * list (ignoring q-value suffixes), lowercases it, and strips hyphens so that
109 * "SHA-256" → "sha256", "SHA-1" → "sha1", "CRC32c" → "crc32c". The bare
110 * RFC 3230 name "SHA" (meaning SHA-1) is mapped to "sha1" as a special case.
111 */
112static void
113xrdhttp_normalize_rfc3230_algo(const u_char *value, size_t vlen,
114 char *dst, size_t dstsz)
115{
116 const u_char *p = value;
117 const u_char *end;
118 size_t out = 0;
119 size_t i;
120
121 /* First token only (up to ',' or ';'). */
122 for (end = p; (size_t)(end - p) < vlen && *end != ',' && *end != ';'; end++)
123 ;
124 vlen = (size_t)(end - p);
125
126 /* Trim leading/trailing whitespace. */
127 while (vlen > 0 && (*p == ' ' || *p == '\t')) { p++; vlen--; }
128 while (vlen > 0 && (p[vlen - 1] == ' ' || p[vlen - 1] == '\t')) { vlen--; }
129
130 /* Lowercase, strip hyphens: "sha-256" → "sha256". */
131 for (i = 0; i < vlen && out < dstsz - 1; i++) {
132 if (p[i] != '-') {
133 dst[out++] = (char) tolower((unsigned char) p[i]);
134 }
135 }
136 dst[out] = '\0';
137
138 /* RFC 3230 bare "sha" means SHA-1. */
139 if (strcmp(dst, "sha") == 0 && dstsz >= 5) {
140 ngx_memcpy(dst, "sha1", 5);
141 }
142}

144/* request parsing */
145void
146xrdhttp_parse_request(ngx_http_request_t *r)
147{
148 xrdhttp_req_ctx_t *ctx;
149 ngx_table_elt_t *h;
150
151 ctx = xrdhttp_get_ctx(r);
152 if (ctx == NULL) {
153 return;
154 }
155
156 /* Detect XrdHttp client via X-Xrootd-Proto header */ h = webdav_tpc_find_header(r, "X-Xrootd-Proto",
157 sizeof("X-Xrootd-Proto") - 1);
158 if (h != NULL) {
159 ctx->is_xrdhttp = 1;
160 ngx_cpystrn((u_char *) ctx->proto_version,
161 h->value.data,
162 ngx_min(h->value.len + 1,
163 (size_t) sizeof(ctx->proto_version)));
164
165 ngx_log_debug1(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
166 "xrdhttp: X-Xrootd-Proto=\"%s\"",
167 ctx->proto_version);
168 }
169
170 /* Want-Digest: adler32 (RFC 3230) enables the streaming body digest.
171 * Only meaningful for XrdHttp clients; the body filter folds the response
172 * through adler32 and emits a Digest trailer. Only adler32 streams here —
173 * other algorithms (crc32c, crc64, crc64nvme, md5, sha*) are computed from
174 * the fd via xrdhttp_add_checksum_header() because they are not folded
175 * incrementally over the response body in this filter. --- */
176 h = webdav_tpc_find_header(r, "Want-Digest", sizeof("Want-Digest") - 1);
177 if (ctx->is_xrdhttp && h != NULL && h->value.len > 0
178 && ngx_strcasestrn(h->value.data, "adler32",
179 sizeof("adler32") - 2) != NULL)
180 {
181 ctx->compute_digest = 1;
182 }
183
184 /* Capture X-Xrootd-Requuid (echo in every response) */ h = webdav_tpc_find_header(r, "X-Xrootd-Requuid",
185 sizeof("X-Xrootd-Requuid") - 1);
186 if (h != NULL && h->value.len > 0) {
187 size_t copy = ngx_min(h->value.len, (size_t) XRDHTTP_UUID_MAX - 1);
188 ngx_memcpy(ctx->requuid, h->value.data, copy);
189 ctx->requuid[copy] = '\0';
190 }
191
192 /* Capture X-Xrootd-Tpc-Token */ h = webdav_tpc_find_header(r, "X-Xrootd-Tpc-Token",
193 sizeof("X-Xrootd-Tpc-Token") - 1);
194 if (h != NULL && h->value.len > 0) {
195 size_t copy = ngx_min(h->value.len,
196 (size_t) XRDHTTP_TPC_KEY_MAX - 1);
197 ngx_memcpy(ctx->tpc_token, h->value.data, copy);
198 ctx->tpc_token[copy] = '\0';
199 }
200
201 /* Parse ?xrd.* and ?tpc.* query parameters */ if (r->args.len > 0) {
202 xrdhttp_args_get(&r->args, "xrd.clnt.uuid",
203 sizeof("xrd.clnt.uuid") - 1,
204 ctx->clnt_uuid, sizeof(ctx->clnt_uuid));
205
206 xrdhttp_args_get(&r->args, "xrd.clnt.app",
207 sizeof("xrd.clnt.app") - 1,
208 ctx->clnt_app, sizeof(ctx->clnt_app));
209
210 xrdhttp_args_get(&r->args, "xrd.want.cksum",
211 sizeof("xrd.want.cksum") - 1,
212 ctx->want_cksum, sizeof(ctx->want_cksum));
213
214 xrdhttp_args_get(&r->args, "xrd.opaque",
215 sizeof("xrd.opaque") - 1,
216 ctx->opaque, sizeof(ctx->opaque));
217
218 xrdhttp_args_get(&r->args, "tpc.src",
219 sizeof("tpc.src") - 1,
220 ctx->tpc_src, sizeof(ctx->tpc_src));
221
222 xrdhttp_args_get(&r->args, "tpc.dst",
223 sizeof("tpc.dst") - 1,
224 ctx->tpc_dst, sizeof(ctx->tpc_dst));
225
226 xrdhttp_args_get(&r->args, "tpc.key",
227 sizeof("tpc.key") - 1,
228 ctx->tpc_key, sizeof(ctx->tpc_key));
229 }
230
231 /* Want-Digest: (RFC 3230) — XrdClHttp sends this on HEAD to request checksums.
232 * Only checked when ?xrd.want.cksum= was not supplied (query param takes priority). */
233 if (!ctx->want_cksum[0]) {
234 h = webdav_tpc_find_header(r, "Want-Digest", sizeof("Want-Digest") - 1);
235 if (h != NULL && h->value.len > 0) {
236 xrdhttp_normalize_rfc3230_algo(h->value.data, h->value.len,
237 ctx->want_cksum, sizeof(ctx->want_cksum));
238 }
239 }
240
241 /* Log client identity at debug level (sanitised). */
242 if (ctx->clnt_app[0] || ctx->clnt_uuid[0]) {
243 char safe_app[sizeof(ctx->clnt_app) * 4];
244 char safe_uuid[sizeof(ctx->clnt_uuid) * 4];
245 brix_sanitize_log_string(ctx->clnt_app,
246 safe_app, sizeof(safe_app));
247 brix_sanitize_log_string(ctx->clnt_uuid,
248 safe_uuid, sizeof(safe_uuid));
249 ngx_log_debug2(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
250 "xrdhttp: client app=\"%s\" uuid=\"%s\"",
251 safe_app, safe_uuid);
252 }
253
254 if (ctx->want_cksum[0]) {
255 char safe_alg[sizeof(ctx->want_cksum) * 4];
256 brix_sanitize_log_string(ctx->want_cksum,
257 safe_alg, sizeof(safe_alg));
258 ngx_log_debug1(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
259 "xrdhttp: client requests checksum alg=\"%s\"",
260 safe_alg);
261 }
262}

264/* HTTP → kXR status mapping */
265int
266xrdhttp_http_to_xrd_status(ngx_int_t http_status)
267{
268 if (http_status >= 200 && http_status < 300) {
269 return kXR_ok;
270 }
271
272 switch (http_status) {
273 case NGX_HTTP_BAD_REQUEST: return kXR_ArgInvalid;
274 case NGX_HTTP_UNAUTHORIZED: return kXR_NotAuthorized;
275 case NGX_HTTP_FORBIDDEN: return kXR_NotAuthorized;
276 case NGX_HTTP_NOT_FOUND: return kXR_NotFound;
277 case NGX_HTTP_NOT_ALLOWED: return kXR_Unsupported;
278 case NGX_HTTP_CONFLICT: return kXR_FSError;
279 case NGX_HTTP_LENGTH_REQUIRED: return kXR_ArgMissing;
280 case 423 /* Locked */: return kXR_FileLocked;
281 case NGX_HTTP_REQUEST_URI_TOO_LARGE: return kXR_ArgTooLong;
282 case NGX_HTTP_RANGE_NOT_SATISFIABLE: return kXR_ArgInvalid;
283 case NGX_HTTP_INTERNAL_SERVER_ERROR: return kXR_ServerError;
284 case NGX_HTTP_NOT_IMPLEMENTED: return kXR_Unsupported;
285 case NGX_HTTP_BAD_GATEWAY: return kXR_FSError;
286 case NGX_HTTP_SERVICE_UNAVAILABLE: return kXR_TooManyErrs;
287 case NGX_HTTP_INSUFFICIENT_STORAGE: return kXR_NoSpace;
288 default:
289 if (http_status >= 400 && http_status < 500) {
290 return kXR_ArgInvalid;
291 }
292 if (http_status >= 500) {
293 return kXR_ServerError;
294 }
295 return kXR_ok;
296 }
297}

299ngx_int_t
300xrdhttp_add_response_headers(ngx_http_request_t *r, ngx_int_t http_status)
301{
302 xrdhttp_req_ctx_t *ctx;
303 int xrd_code;
304
305 ctx = (xrdhttp_req_ctx_t *)
306 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
307 if (ctx == NULL || !ctx->is_xrdhttp) {
308 return NGX_OK;
309 }
310
311 /* Guard against double-injection: filter + direct call in get.c. */
312 if (ctx->headers_injected) {
313 return NGX_OK;
314 }
315 ctx->headers_injected = 1;
316
317 /* X-Xrootd-Requuid — echo the request UUID back to the client. */
318 if (ctx->requuid[0]) {
319 if (brix_http_set_header(r, "X-Xrootd-Requuid", ctx->requuid,
320 NULL) != NGX_OK)
321 {
322 return NGX_ERROR;
323 }
324 }
325
326 /* X-Xrootd-Status — always emit per XrdHttp spec: 0 = success, non-zero = error. */
327 xrd_code = xrdhttp_http_to_xrd_status(http_status);
328 if (brix_http_set_header_num(r, "X-Xrootd-Status",
329 (long) xrd_code) != NGX_OK)
330 {
331 return NGX_ERROR;
332 }
333
334 /* X-Xrootd-Wait — back-pressure for async/tape-recall scenarios. */
335 if (ctx->wait_seconds > 0) {
336 if (brix_http_set_header_num(r, "X-Xrootd-Wait",
337 (long) ctx->wait_seconds) != NGX_OK)
338 {
339 return NGX_ERROR;
340 }
341 }
342
343 /* X-Xrootd-Retry — hint for client retry interval. */
344 if (ctx->retry_seconds > 0) {
345 if (brix_http_set_header_num(r, "X-Xrootd-Retry",
346 (long) ctx->retry_seconds) != NGX_OK)
347 {
348 return NGX_ERROR;
349 }
350 }
351
352 return NGX_OK;
353}

355/* redirect dialect */
356ngx_int_t
357xrdhttp_send_redirect(ngx_http_request_t *r,
358 const char *location_url,
359 const char *redir_host,
360 int redir_port)
361{
362 xrdhttp_req_ctx_t *ctx;
363 ngx_table_elt_t *location_hdr;
364 char loc_buf[XRDHTTP_TPC_URL_MAX + XRDHTTP_OPAQUE_MAX + 64];
365 int loc_len;
366 const char *sep;
367
368 ctx = (xrdhttp_req_ctx_t *)
369 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
370
371 /* Build Location URL: append tpc.key and xrd.opaque as query params. */
372 sep = (ngx_strchr(location_url, '?') != NULL) ? "&" : "?";
373
374 if (ctx != NULL && (ctx->tpc_key[0] || ctx->opaque[0])) {
375 if (ctx->tpc_key[0] && ctx->opaque[0]) {
376 loc_len = snprintf(loc_buf, sizeof(loc_buf),
377 "%s%stpc.key=%s&xrd.opaque=%s",
378 location_url, sep,
379 ctx->tpc_key, ctx->opaque);
380 } else if (ctx->tpc_key[0]) {
381 loc_len = snprintf(loc_buf, sizeof(loc_buf),
382 "%s%stpc.key=%s",
383 location_url, sep, ctx->tpc_key);
384 } else {
385 loc_len = snprintf(loc_buf, sizeof(loc_buf),
386 "%s%sxrd.opaque=%s",
387 location_url, sep, ctx->opaque);
388 }
389
390 if (loc_len < 0 || (size_t) loc_len >= sizeof(loc_buf)) {
391 /* Overflow: use the plain URL without appended params. */
392 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
393 "xrdhttp: redirect URL overflow, opaque dropped");
394 ngx_cpystrn((u_char *) loc_buf, (u_char *) location_url,
395 sizeof(loc_buf));
396 }
397 location_url = loc_buf;
398 }
399
400 /* Set Location header. */
401 location_hdr = ngx_list_push(&r->headers_out.headers);
402 if (location_hdr == NULL) {
403 return NGX_HTTP_INTERNAL_SERVER_ERROR;
404 }
405 location_hdr->hash = 1;
406 ngx_str_set(&location_hdr->key, "Location");
407 location_hdr->value.data = ngx_pstrdup(r->pool, &(ngx_str_t){
408 ngx_strlen(location_url), (u_char *) location_url
409 });
410 if (location_hdr->value.data == NULL) {
411 return NGX_HTTP_INTERNAL_SERVER_ERROR;
412 }
413 location_hdr->value.len = ngx_strlen(location_url);
414
415 /* X-Xrootd-Redir-Host / Port for hierarchical cluster topologies. */
416 if (redir_host != NULL && redir_host[0]) {
417 if (brix_http_set_header(r, "X-Xrootd-Redir-Host", redir_host,
418 NULL) != NGX_OK)
419 {
420 return NGX_HTTP_INTERNAL_SERVER_ERROR;
421 }
422 }
423 if (redir_port > 0) {
424 if (brix_http_set_header_num(r, "X-Xrootd-Redir-Port",
425 (long) redir_port) != NGX_OK)
426 {
427 return NGX_HTTP_INTERNAL_SERVER_ERROR;
428 }
429 }
430
431 /* Echo XrdHttp response headers (Requuid, Status=kXR_ok for redirect). */
432 if (xrdhttp_add_response_headers(r, NGX_HTTP_TEMPORARY_REDIRECT)
433 != NGX_OK)
434 {
435 return NGX_HTTP_INTERNAL_SERVER_ERROR;
436 }
437
438 r->headers_out.status = NGX_HTTP_TEMPORARY_REDIRECT;
439 r->headers_out.content_length_n = 0;
440
441 ngx_http_send_header(r);
442 return ngx_http_send_special(r, NGX_HTTP_LAST);
443}

473/* TPC header injection */
474ngx_int_t
475xrdhttp_inject_tpc_headers(ngx_http_request_t *r)
476{
477 xrdhttp_req_ctx_t *ctx;
478 ngx_table_elt_t *h;
479
480 ctx = (xrdhttp_req_ctx_t *)
481 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
482 if (ctx == NULL) {
483 return NGX_OK;
484 }
485
486 /*
487 * ?tpc.src= → inject Source: header (if not already present).
488 * This allows XrdHttp clients to signal TPC pull via URL instead of header.
489 */
490 if (ctx->tpc_src[0]
491 && webdav_tpc_find_header(r, "Source", sizeof("Source") - 1) == NULL)
492 {
493 brix_net_target_t target;
494 ngx_str_t url;
495 char err[256];
496
497 url.data = (u_char *) ctx->tpc_src;
498 url.len = ngx_strlen(ctx->tpc_src);
499
500 if (brix_net_target_parse(NULL, &url, &target, err, sizeof(err))
501 != NGX_OK)
502 {
503 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
504 "xrdhttp: invalid tpc.src URL: %s; ignored", err);
505 } else if (ngx_strncasecmp(url.data, (u_char *) "https://", 8) != 0) {
506 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
507 "xrdhttp: tpc.src must be https://; ignored");
508 } else {
509 if (brix_http_request_header_add(r, "Source", ctx->tpc_src,
510 NULL) != NGX_OK)
511 {
512 return NGX_ERROR;
513 }
514 }
515 }
516
517 /*
518 * ?tpc.dst= → inject Destination: header (if not already present).
519 */
520 if (ctx->tpc_dst[0]
521 && webdav_tpc_find_header(r, "Destination",
522 sizeof("Destination") - 1) == NULL)
523 {
524 brix_net_target_t target;
525 ngx_str_t url;
526 char err[256];
527
528 url.data = (u_char *) ctx->tpc_dst;
529 url.len = ngx_strlen(ctx->tpc_dst);
530
531 if (brix_net_target_parse(NULL, &url, &target, err, sizeof(err))
532 != NGX_OK)
533 {
534 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
535 "xrdhttp: invalid tpc.dst URL: %s; ignored", err);
536 } else if (ngx_strncasecmp(url.data, (u_char *) "https://", 8) != 0) {
537 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
538 "xrdhttp: tpc.dst must be https://; ignored");
539 } else {
540 if (brix_http_request_header_add(r, "Destination",
541 ctx->tpc_dst,
542 NULL) != NGX_OK)
543 {
544 return NGX_ERROR;
545 }
546 }
547 }
548
549 /*
550 * X-Xrootd-Tpc-Token → if Authorization: Bearer is absent, inject from
551 * the captured TPC token. This allows TPC delegation via this header.
552 */
553 if (ctx->tpc_token[0]
554 && r->headers_in.authorization == NULL)
555 {
556 char bearer_buf[sizeof("Bearer ") + XRDHTTP_TPC_KEY_MAX];
557 snprintf(bearer_buf, sizeof(bearer_buf), "Bearer %s", ctx->tpc_token);
558
559 if (brix_http_request_header_add(r, "Authorization", bearer_buf,
560 &h) != NGX_OK)
561 {
562 return NGX_ERROR;
563 }
564 /* Point nginx's fast-path field at this new entry. */
565 r->headers_in.authorization = h;
566 }
567
568 return NGX_OK;
569}

571/* checksum header injection */
572ngx_int_t
573xrdhttp_add_checksum_header(ngx_http_request_t *r,
574 ngx_fd_t fd,
575 const struct stat *sb)
576{
577 xrdhttp_req_ctx_t *ctx;
578 /* algo (32) + '=' + sha256 hex (64) + NUL = 98; round up generously */
579 char hdr_value[256];
580 char algo[32];
581 brix_checksum_alg_t alg;
582
583 ctx = (xrdhttp_req_ctx_t *)
584 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
585 if (ctx == NULL || !ctx->want_cksum[0] || fd == NGX_INVALID_FILE) {
586 return NGX_OK;
587 }
588
589 if (sb->st_size > (off_t) 2 * 1024 * 1024 * 1024LL) {
590 /* Skip checksum computation for files > 2 GiB to bound latency. */
591 ngx_log_debug0(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
592 "xrdhttp: skipping checksum for large file (> 2 GiB)");
593 return NGX_OK;
594 }
595
596 if (brix_checksum_parse(ctx->want_cksum, strlen(ctx->want_cksum),
597 &alg, algo, sizeof(algo)) == NGX_OK)
598 {
599 brix_integrity_info_t info;
600 brix_integrity_opts_t iopts;
601
602 ngx_memzero(&iopts, sizeof(iopts));
603 iopts.allow_xattr_cache = 1;
604 iopts.update_xattr_cache = 1;
605
606 if (brix_integrity_get_fd(r->connection->log, fd, NULL, "<xrdhttp>",
607 algo, &iopts, &info) != NGX_OK
608 || brix_integrity_format_http_digest(&info, hdr_value,
609 sizeof(hdr_value)) != NGX_OK)
610 {
611 return NGX_OK;
612 }
613 } else {
614 char safe_alg[sizeof(ctx->want_cksum) * 4];
615 brix_sanitize_log_string(ctx->want_cksum,
616 safe_alg, sizeof(safe_alg));
617 ngx_log_debug1(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
618 "xrdhttp: unsupported checksum algorithm \"%s\"",
619 safe_alg);
620 return NGX_OK;
621 }
622
623 return brix_http_set_header(r, "Digest", hdr_value, NULL);
624}

626/* streaming Digest body filter (Phase 21 Step B) */
627ngx_int_t
628xrdhttp_digest_body_filter(ngx_http_request_t *r, ngx_chain_t *in,
629 ngx_http_output_body_filter_pt next)
630{
631 xrdhttp_req_ctx_t *ctx;
632 ngx_chain_t *cl;
633 ngx_uint_t saw_last = 0;
634
635 ctx = (xrdhttp_req_ctx_t *)
636 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
637
638 if (ctx == NULL || !ctx->is_xrdhttp || !ctx->compute_digest) {
639 return next(r, in); /* pass-through */
640 }
641
642 if (ctx->adler == 0) {
643 ctx->adler = (uint32_t) adler32(0L, Z_NULL, 0); /* seed = 1 */
644 }
645
646 for (cl = in; cl != NULL; cl = cl->next) {
647 ngx_buf_t *b = cl->buf;
648
649 if (ngx_buf_in_memory(b) && b->last > b->pos) {
650 ctx->adler = (uint32_t) adler32((uLong) ctx->adler, b->pos,
651 (uInt) (b->last - b->pos));
652 }
653 if (b->last_buf || b->last_in_chain) {
654 saw_last = 1;
655 }
656 }
657
658 /* On the final buffer, queue the Digest as a response trailer. nginx only
659 * transmits trailers on chunked HTTP/1.1 and HTTP/2/3 responses; for a
660 * fixed Content-Length GET the fd-based xrdhttp_add_checksum_header() path
661 * supplies the Digest header instead, so this is purely additive. */
662 if (saw_last && !ctx->digest_emitted) {
663 ngx_table_elt_t *h = ngx_list_push(&r->headers_out.trailers);
664 if (h != NULL) {
665 u_char *v = ngx_pnalloc(r->pool, sizeof("adler32=") - 1 + 8 + 1);
666 if (v != NULL) {
667 size_t n = ngx_sprintf(v, "adler32=%08xD", ctx->adler) - v;
668 h->hash = 1;
669 ngx_str_set(&h->key, "Digest");
670 h->value.data = v;
671 h->value.len = n;
672 }
673 }
674 ctx->digest_emitted = 1;
675 }
676
677 return next(r, in);
678}
