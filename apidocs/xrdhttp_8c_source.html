1/*
2 * xrdhttp.c — XrdHttp protocol extension support for the nginx-xrootd WebDAV module.
3 *
4 * WHAT: Implements XrdHttp parity features on top of the existing WebDAV handler:
5 * - Request parsing: X-Xrootd-Proto detection, ?xrd.* / ?tpc.* query params,
6 * X-Xrootd-Requuid / X-Xrootd-Tpc-Token header capture.
7 * - Response injection: X-Xrootd-Requuid echo, X-Xrootd-Status kXR error codes,
8 * X-Xrootd-Wait / X-Xrootd-Retry back-pressure headers.
9 * - Redirect dialect: X-Xrootd-Redir-Host/Port + tpc.key / xrd.opaque passthrough.
10 * - TPC shim: synthesise Source:/Destination: headers from ?tpc.src= / ?tpc.dst=.
11 * - Checksum: Digest: header computed on-demand for xrd.want.cksum requests.
12 * - HTTP→kXR status mapping for X-Xrootd-Status.
13 *
14 * WHY: XRootD-aware clients (xrdcp --prefer-xrdhttp, ROOT TFile, davix-get with XRD
15 * hints) negotiate the protocol version via X-Xrootd-Proto and rely on X-Xrootd-Status
16 * to distinguish between "file not found", "not authorised", "server busy" etc. without
17 * parsing HTTP status codes. Without this layer clients fall back to slow error-path
18 * retries or misinterpret error conditions.
19 *
20 * HOW: A single per-request context struct (xrdhttp_req_ctx_t) is allocated from r->pool
21 * by xrdhttp_get_ctx() the first time any XrdHttp function is called. Parsing functions
22 * walk r->headers_in and the raw r->args query string; they write NUL-terminated results
23 * into fixed-size fields (no dynamic allocation) and silently truncate over-length values
24 * so that untrusted client data cannot cause unbounded memory use. All values written to
25 * logs pass through brix_sanitize_log_string().
26 */

49/* context management */
50xrdhttp_req_ctx_t *
51xrdhttp_get_ctx(ngx_http_request_t *r)
52{
53 ngx_http_brix_webdav_req_ctx_t *webdav_ctx;
54
55 webdav_ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
56 if (webdav_ctx != NULL) {
57 /* xrdhttp is the first member — cast is well-defined by C11 §6.7.2.1p15. */
58 return &webdav_ctx->xrdhttp;
59 }
60
61 /* Allocate the full webdav context (xrdhttp fields zero-initialised by pcalloc).
62 * This path is hit on unauthenticated requests where auth_cert.c / auth_token.c
63 * never allocated the context themselves. */
64 BRIX_PCALLOC_OR_RETURN(webdav_ctx, r->pool, sizeof(ngx_http_brix_webdav_req_ctx_t), NULL);
65 ngx_http_set_ctx(r, webdav_ctx, ngx_http_brix_webdav_module);
66 return &webdav_ctx->xrdhttp;
67}

91/*
92 * Normalize an RFC 3230 algorithm token to the internal name used by
93 * brix_checksum_parse(). Extracts the first token from a comma-separated
94 * list (ignoring q-value suffixes), lowercases it, and strips hyphens so that
95 * "SHA-256" → "sha256", "SHA-1" → "sha1", "CRC32c" → "crc32c". The bare
96 * RFC 3230 name "SHA" (meaning SHA-1) is mapped to "sha1" as a special case.
97 */
98static void
99xrdhttp_normalize_rfc3230_algo(const u_char *value, size_t vlen,
100 char *dst, size_t dstsz)
101{
102 const u_char *p = value;
103 const u_char *end;
104 size_t out = 0;
105 size_t i;
106
107 /* First token only (up to ',' or ';'). */
108 for (end = p; (size_t)(end - p) < vlen && *end != ',' && *end != ';'; end++)
109 ;
110 vlen = (size_t)(end - p);
111
112 /* Trim leading/trailing whitespace. */
113 while (vlen > 0 && (*p == ' ' || *p == '\t')) { p++; vlen--; }
114 while (vlen > 0 && (p[vlen - 1] == ' ' || p[vlen - 1] == '\t')) { vlen--; }
115
116 /* Lowercase, strip hyphens: "sha-256" → "sha256". */
117 for (i = 0; i < vlen && out < dstsz - 1; i++) {
118 if (p[i] != '-') {
119 dst[out++] = (char) tolower((unsigned char) p[i]);
120 }
121 }
122 dst[out] = '\0';
123
124 /* RFC 3230 bare "sha" means SHA-1. */
125 if (strcmp(dst, "sha") == 0 && dstsz >= 5) {
126 ngx_memcpy(dst, "sha1", 5);
127 }
128}

132/*
133 * WHAT: Copy a header value into a fixed-size ctx field, NUL-terminated and
134 * silently truncated to cap-1 bytes.
135 * WHY: Requuid / Tpc-Token capture shares identical bounded-copy logic;
136 * centralising it removes duplicated ngx_min/memcpy boilerplate.
137 * HOW: No-op when the header is absent or empty; otherwise copies at most
138 * cap-1 bytes and writes the terminator.
139 */
140static void
141xrdhttp_capture_header(ngx_http_request_t *r, const char *name,
142 size_t name_len, char *dst, size_t cap)
143{
144 ngx_table_elt_t *h = webdav_tpc_find_header(r, name, name_len);
145
146 if (h != NULL && h->value.len > 0) {
147 size_t copy = ngx_min(h->value.len, cap - 1);
148 ngx_memcpy(dst, h->value.data, copy);
149 dst[copy] = '\0';
150 }
151}

153/*
154 * WHAT: Detect the XrdHttp dialect and its digest/version hints from headers.
155 * WHY: X-Xrootd-Proto flags an XrdHttp client and gates the adler32 streaming
156 * digest (Want-Digest), which is only meaningful for such clients.
157 * HOW: Sets ctx->is_xrdhttp + proto_version from X-Xrootd-Proto, then arms
158 * ctx->compute_digest when an XrdHttp client asks for adler32.
159 */
160static void
161xrdhttp_parse_proto_headers(ngx_http_request_t *r, xrdhttp_req_ctx_t *ctx)
162{
163 ngx_table_elt_t *h;
164
165 h = webdav_tpc_find_header(r, "X-Xrootd-Proto",
166 sizeof("X-Xrootd-Proto") - 1);
167 if (h != NULL) {
168 ctx->is_xrdhttp = 1;
169 ngx_cpystrn((u_char *) ctx->proto_version,
170 h->value.data,
171 ngx_min(h->value.len + 1,
172 (size_t) sizeof(ctx->proto_version)));
173
174 ngx_log_debug1(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
175 "xrdhttp: X-Xrootd-Proto=\"%s\"",
176 ctx->proto_version);
177 }
178
179 /* Want-Digest: adler32 (RFC 3230) enables the streaming body digest.
180 * Only meaningful for XrdHttp clients; the body filter folds the response
181 * through adler32 and emits a Digest trailer. Only adler32 streams here.
182 * Other algorithms (crc32c, crc64, crc64nvme, md5, sha) are computed from
183 * the fd via xrdhttp_add_checksum_header() because they are not folded
184 * incrementally over the response body in this filter. */
185 h = webdav_tpc_find_header(r, "Want-Digest", sizeof("Want-Digest") - 1);
186 if (ctx->is_xrdhttp && h != NULL && h->value.len > 0
187 && ngx_strcasestrn(h->value.data, "adler32",
188 sizeof("adler32") - 2) != NULL)
189 {
190 ctx->compute_digest = 1;
191 }
192}

194/*
195 * WHAT: Copy the ?xrd.* / ?tpc.* query parameters into their ctx fields.
196 * WHY: XrdHttp clients can signal client identity, checksum wants, opaque
197 * data and TPC endpoints via the query string.

237/*
238 * WHAT: Emit sanitised debug lines for the parsed client identity + checksum.
239 * WHY: Untrusted client values must be escaped before logging; keeping the
240 * (debug-only) formatting out of the main path keeps it simple.
241 * HOW: Sanitises app/uuid and the requested checksum algorithm via
242 * brix_sanitize_log_string() before logging at debug level.
243 */
244static void
245xrdhttp_log_client_identity(ngx_http_request_t *r, xrdhttp_req_ctx_t *ctx)
246{
247 if (ctx->clnt_app[0] || ctx->clnt_uuid[0]) {
248 char safe_app[sizeof(ctx->clnt_app) * 4];
249 char safe_uuid[sizeof(ctx->clnt_uuid) * 4];
250 brix_sanitize_log_string(ctx->clnt_app,
251 safe_app, sizeof(safe_app));
252 brix_sanitize_log_string(ctx->clnt_uuid,
253 safe_uuid, sizeof(safe_uuid));
254 ngx_log_debug2(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
255 "xrdhttp: client app=\"%s\" uuid=\"%s\"",
256 safe_app, safe_uuid);
257 }
258
259 if (ctx->want_cksum[0]) {
260 char safe_alg[sizeof(ctx->want_cksum) * 4];
261 brix_sanitize_log_string(ctx->want_cksum,
262 safe_alg, sizeof(safe_alg));
263 ngx_log_debug1(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
264 "xrdhttp: client requests checksum alg=\"%s\"",
265 safe_alg);
266 }

269void
270xrdhttp_parse_request(ngx_http_request_t *r)
271{
272 xrdhttp_req_ctx_t *ctx;
273 ngx_table_elt_t *h;
274
275 ctx = xrdhttp_get_ctx(r);
276 if (ctx == NULL) {
277 return;
278 }
279
280 xrdhttp_parse_proto_headers(r, ctx);
281
282 /* Capture X-Xrootd-Requuid (echo in every response). */
283 xrdhttp_capture_header(r, "X-Xrootd-Requuid",
284 sizeof("X-Xrootd-Requuid") - 1,
285 ctx->requuid, XRDHTTP_UUID_MAX);
286
287 /* Capture X-Xrootd-Tpc-Token. */
288 xrdhttp_capture_header(r, "X-Xrootd-Tpc-Token",
289 sizeof("X-Xrootd-Tpc-Token") - 1,
290 ctx->tpc_token, XRDHTTP_TPC_KEY_MAX);
291
292 xrdhttp_parse_query_params(r, ctx);
293
294 /* Want-Digest (RFC 3230): XrdClHttp sends this on HEAD to request
295 * checksums. Only consulted when ?xrd.want.cksum= was not supplied
296 * (the query param takes priority). */
297 if (!ctx->want_cksum[0]) {
298 h = webdav_tpc_find_header(r, "Want-Digest", sizeof("Want-Digest") - 1);
299 if (h != NULL && h->value.len > 0) {
300 xrdhttp_normalize_rfc3230_algo(h->value.data, h->value.len,
301 ctx->want_cksum,
302 sizeof(ctx->want_cksum));
303 }
304 }
305
306 xrdhttp_log_client_identity(r, ctx);
307}
