1/*
2 * cli_opts.c — shared parser for the common connection/trace CLI flags.
3 *
4 * WHAT: brix_opts_init() zero-inits an brix_opts with the canonical defaults
5 * (verify_host on); brix_opts_parse_arg() recognises the connection flags
6 * every connecting tool accepts — --tls / --notlsok / --noverifyhost /
7 * --auth <p> / --wire-trace[=N] / --timing / --redirect-trace / --capture
8 * <p> — plus the resilience knobs --max-stall <ms> / --no-retry — and
9 * updates *o. Also handles two universal flags:
10 * --version prints "<argv0-basename> (BriX-Cache client) <version>"
11 * to stdout and exits 0 immediately.
12 * --help returns 2; the CALLER must print its usage to stdout and
13 * exit 0 (usage text is per-tool, so we cannot do it here).
14 * WHY: This exact else-if ladder was copy-pasted into ~8 front-ends (xrdcp,
15 * xrddiag, xrdmapc, xrdgsitest, both FUSE drivers, ...); centralising it
16 * guarantees one canonical spelling/semantics and lets each tool's parse
17 * loop fall through to its OWN flags.
18 * HOW: A tool's option loop calls brix_opts_parse_arg() first; on 1 it `continue`s,
19 * on 0 it handles the arg itself; on 2 it prints usage to stdout and returns 0.
20 * Value-taking flags advance *i past the value, mirroring the historical
21 * `argv[++i]` idiom exactly (behaviour-preserving — the fields set are
22 * identical to the inline ladders).
23 */

31/*
32 * opts_basename — return the basename of a file path without modifying the
33 * input string (POSIX basename() may modify its argument; we avoid that).
34 * WHY: --version uses argv[0] to print the personality name; a safe, in-place
35 * basename keeps the implementation self-contained.
36 */
37static const char *
38opts_basename(const char *path)
39{
40 const char *s = strrchr(path, '/');
41 return (s != NULL && s[1] != '\0') ? s + 1 : path;
42}

44void
45brix_opts_init(brix_opts *o)
46{
47 memset(o, 0, sizeof(*o));
48 o->verify_host = 1; /* default: check the server cert name */
49
50 /* Resilience is ON by default: max_stall_ms left 0 ⇒ the library uses
51 * XRDC_DEFAULT_MAX_STALL_MS. $XRDC_MAX_STALL_MS overrides — a positive value
52 * widens/narrows the window; 0 (or negative) disables it (fail fast). */
53 const char *e = getenv("XRDC_MAX_STALL_MS");
54 if (e != NULL && *e != '\0') {
55 int v = atoi(e);
56 if (v <= 0) {
57 o->no_retry = 1;
58 } else {
59 o->max_stall_ms = v;
60 }
61 }
62}

64int
65brix_opts_parse_arg(brix_opts *o, int argc, char **argv, int *i)
66{
67 const char *a = argv[*i];
68
69 /*
70 * Universal flags — handled for every tool that routes through this parser.
71 * --version exits inside the parser (acceptable per spec; it is a clean
72 * terminal action with no partial-state concern).
73 * --help returns 2 so the CALLER can print its own usage to stdout and exit 0
74 * (usage text is per-tool; we cannot print it here).
75 */
76 if (strcmp(a, "--version") == 0) {
77 printf("%s (BriX-Cache client) %s\n",
78 opts_basename(argv[0]), brix_client_version());
79 exit(0);
80 }
81 if (strcmp(a, "--help") == 0) {
82 return 2; /* caller: print usage to stdout, exit 0 */
83 }
84
85 if (strcmp(a, "--tls") == 0) { o->want_tls = 1; return 1; }
86 if (strcmp(a, "--notlsok") == 0) { o->notlsok = 1; return 1; }
87 if (strcmp(a, "--noverifyhost") == 0) { o->verify_host = 0; return 1; }
88 if (strcmp(a, "--timing") == 0) { o->timing = 1; return 1; }
89 if (strcmp(a, "--redirect-trace") == 0) { o->redir_trace = 1; return 1; }
90 if (strcmp(a, "--wire-trace") == 0) { o->wire_trace = 1; return 1; }
91 if (strncmp(a, "--wire-trace=", 13) == 0){ o->wire_trace = atoi(a + 13); return 1; }
92 if (strcmp(a, "--auth") == 0 && *i + 1 < argc) {
93 o->auth_force = argv[++(*i)];
94 return 1;
95 }
96 if (strcmp(a, "--capture") == 0 && *i + 1 < argc) {
97 o->capture = argv[++(*i)];
98 return 1;
99 }
100 if (strcmp(a, "--no-retry") == 0) { o->no_retry = 1; return 1; }
101 if (strcmp(a, "--max-stall") == 0 && *i + 1 < argc) {
102 int v = atoi(argv[++(*i)]);
103 if (v <= 0) {
104 o->no_retry = 1;
105 } else {
106 o->max_stall_ms = v;
107 o->no_retry = 0;
108 }
109 return 1;
110 }
111 return 0;
112}

115/*
116 * brix_cli_parse_io_uring — strict CLI parse for the --io-uring mode flag.
117 *
118 * WHAT: Map "on" / "off" / "auto" to XRDC_IO_URING_{ON,OFF,AUTO}; anything
119 * else (including NULL and empty string) returns -1.
120 * WHY: CLI flags must reject unknown values immediately with a usage error so
121 * the user gets clear feedback. The env-var path (XRDC_IO_URING) stays
122 * lenient — it silently falls back to AUTO because env vars are often set
123 * system-wide and must survive software upgrades without breaking callers.
124 * Only the explicit CLI flag is strict.
125 * HOW: Three strcmp checks; anything not matching returns -1 so the caller can
126 * print a tailored "invalid mode 'bogus' (use on|off|auto)" and exit 50.
127 */
128int
129brix_cli_parse_io_uring(const char *s)
130{
131 if (s == NULL || s[0] == '\0') { return -1; }
132 if (strcmp(s, "on") == 0) { return XRDC_IO_URING_ON; }
133 if (strcmp(s, "off") == 0) { return XRDC_IO_URING_OFF; }
134 if (strcmp(s, "auto") == 0) { return XRDC_IO_URING_AUTO; }
135 return -1;
136}
