1/*
2 * cli_opts.c — shared parser for the common connection/trace CLI flags.
3 *
4 * WHAT: brix_opts_init() zero-inits an brix_opts with the canonical defaults
5 * (verify_host on); brix_opts_parse_arg() recognises the connection flags
6 * every connecting tool accepts — --tls / --notlsok / --noverifyhost /
7 * --auth <p> / --wire-trace[=N] / --timing / --redirect-trace / --capture
8 * <p> — plus the resilience knobs --max-stall <ms> / --no-retry — and
9 * updates *o. Also handles two universal flags:
10 * --version prints "<argv0-basename> (BriX-Cache client) <version>"
11 * to stdout and exits 0 immediately.
12 * --help returns 2; the CALLER must print its usage to stdout and
13 * exit 0 (usage text is per-tool, so we cannot do it here).
14 * WHY: This exact else-if ladder was copy-pasted into ~8 front-ends (xrdcp,
15 * xrddiag, xrdmapc, xrdgsitest, both FUSE drivers, ...); centralising it
16 * guarantees one canonical spelling/semantics and lets each tool's parse
17 * loop fall through to its OWN flags.
18 * HOW: A tool's option loop calls brix_opts_parse_arg() first; on 1 it `continue`s,
19 * on 0 it handles the arg itself; on 2 it prints usage to stdout and returns 0.
20 * Value-taking flags advance *i past the value, mirroring the historical
21 * `argv[++i]` idiom exactly (behaviour-preserving — the fields set are
22 * identical to the inline ladders).
23 */

32/*
33 * opts_basename — return the basename of a file path without modifying the
34 * input string (POSIX basename() may modify its argument; we avoid that).
35 * WHY: --version uses argv[0] to print the personality name; a safe, in-place
36 * basename keeps the implementation self-contained.
37 */
38static const char *
39opts_basename(const char *path)
40{
41 const char *s = strrchr(path, '/');
42 return (s != NULL && s[1] != '\0') ? s + 1 : path;
43}

45void
46brix_opts_init(brix_opts *o)
47{
48 memset(o, 0, sizeof(*o));
49 o->verify_host = 1; /* default: check the server cert name */
50
51 /* Resilience is ON by default: max_stall_ms left 0 ⇒ the library uses
52 * XRDC_DEFAULT_MAX_STALL_MS. $XRDC_MAX_STALL_MS overrides — a positive value
53 * widens/narrows the window; 0 (or negative) disables it (fail fast). */
54 const char *e = getenv("XRDC_MAX_STALL_MS");
55 if (e != NULL && *e != '\0') {
56 int v = atoi(e);
57 if (v <= 0) {
58 o->no_retry = 1;
59 } else {
60 o->max_stall_ms = v;
61 }
62 }
63}

65/*
66 * opts_parse_universal — handle the two flags every tool honours the same way.
67 *
68 * WHAT: --version prints "<argv0-basename> (BriX-Cache client) <version>" to
69 * stdout and exits 0; --help returns 2 (the caller prints its own usage);
70 * any other argument returns 0 (not a universal flag).
71 * WHY: --version is a clean terminal action with no partial-state concern, so
72 * exiting inside the parser is safe. --help text is per-tool, so we defer
73 * to the caller rather than printing it here.
74 * HOW: Two strcmp checks; the split keeps the main dispatcher's complexity low.
75 */
76static int
77opts_parse_universal(const char *a, char **argv)
78{
79 if (strcmp(a, "--version") == 0) {
80 printf("%s (BriX-Cache client) %s\n",
81 opts_basename(argv[0]), brix_client_version());
82 exit(0);
83 }
84 if (strcmp(a, "--help") == 0) {
85 return 2; /* caller: print usage to stdout, exit 0 */
86 }
87 return 0;
88}

90/*
91 * Table of the value-less boolean toggles. Each row names the flag, the int
92 * field within brix_opts it sets (via offsetof, so one loop drives them all),
93 * and the value to store — 1 for the enabling flags, 0 for --noverifyhost which
94 * CLEARS the default-on verify_host. Order is irrelevant: every name is a
95 * distinct exact match.
96 */
97typedef struct {
98 const char *name;
99 size_t field_off;
100 int value;
101} opts_toggle_t;
102
103static const opts_toggle_t opts_toggles[] = {
104 { "--tls", offsetof(brix_opts, want_tls), 1 },
105 { "--notlsok", offsetof(brix_opts, notlsok), 1 },
106 { "--noverifyhost", offsetof(brix_opts, verify_host), 0 },
107 { "--timing", offsetof(brix_opts, timing), 1 },
108 { "--redirect-trace", offsetof(brix_opts, redir_trace), 1 },
109 { "--wire-trace", offsetof(brix_opts, wire_trace), 1 },
110 { "--no-retry", offsetof(brix_opts, no_retry), 1 },
111};

113/*
114 * opts_parse_toggle — match `a` against the value-less boolean flag table.
115 *
116 * WHAT: On an exact match, store the row's value into the named int field of *o
117 * and return 1; otherwise return 0 (not a toggle).
118 * WHY: Table dispatch replaces a seven-branch if-ladder with one loop, keeping
119 * both this helper and the main dispatcher well under the complexity cap.
120 * HOW: Linear scan; the matched field is addressed by byte offset (all listed
121 * fields are int), mirroring the historical `o->field = value` writes.
122 */
123static int
124opts_parse_toggle(brix_opts *o, const char *a)
125{
126 size_t k;
127 for (k = 0; k < sizeof(opts_toggles) / sizeof(opts_toggles[0]); k++) {
128 if (strcmp(a, opts_toggles[k].name) == 0) {
129 *(int *)((char *)o + opts_toggles[k].field_off) = opts_toggles[k].value;
130 return 1;
131 }
132 }
133 return 0;
134}

136/*
137 * opts_parse_valued — handle the flags that consume an argument (or a =value).
138 *
139 * WHAT: --wire-trace=N sets wire_trace from the inline value; --auth <p> /
140 * --capture <p> / --max-stall <ms> each consume the following argv slot
141 * (advancing *i) exactly as the historical `argv[++i]` idiom. Returns 1
142 * when a flag was consumed, 0 otherwise. --max-stall <=0 disables retry.
143 * WHY: Grouping the value-taking flags away from the boolean toggles keeps each
144 * helper single-purpose and under the complexity cap.
145 * HOW: The value-consuming flags guard on `*i + 1 < argc` so a trailing flag
146 * with no value falls through (returns 0) rather than reading past argv.
147 */
148static int
149opts_parse_valued(brix_opts *o, int argc, char **argv, int *i)
150{
151 const char *a = argv[*i];
152
153 if (strncmp(a, "--wire-trace=", 13) == 0) {
154 o->wire_trace = atoi(a + 13);
155 return 1;
156 }
157 if (strcmp(a, "--auth") == 0 && *i + 1 < argc) {
158 o->auth_force = argv[++(*i)];
159 return 1;
160 }
161 if (strcmp(a, "--capture") == 0 && *i + 1 < argc) {
162 o->capture = argv[++(*i)];
163 return 1;
164 }
165 if (strcmp(a, "--max-stall") == 0 && *i + 1 < argc) {
166 int v = atoi(argv[++(*i)]);
167 if (v <= 0) {
168 o->no_retry = 1;
169 } else {
170 o->max_stall_ms = v;
171 o->no_retry = 0;
172 }
173 return 1;
174 }
175 return 0;
176}

178int
179brix_opts_parse_arg(brix_opts *o, int argc, char **argv, int *i)
180{
181 const char *a = argv[*i];
182
183 /*
184 * Universal flags first (--version exits, --help returns 2), then the
185 * value-less boolean toggles, then the flags that consume a value. Each
186 * helper returns 1 when it consumed the arg; 0 means "not mine", so the
187 * whole function returns 0 and the caller handles the arg itself.
188 */
189 int r = opts_parse_universal(a, argv);
190 if (r != 0) {
191 return r;
192 }
193 if (opts_parse_toggle(o, a)) {
194 return 1;
195 }
196 if (opts_parse_valued(o, argc, argv, i)) {
197 return 1;
198 }
199 return 0;
200}

203/*
204 * brix_cli_parse_io_uring — strict CLI parse for the --io-uring mode flag.
205 *
206 * WHAT: Map "on" / "off" / "auto" to XRDC_IO_URING_{ON,OFF,AUTO}; anything
207 * else (including NULL and empty string) returns -1.
208 * WHY: CLI flags must reject unknown values immediately with a usage error so
209 * the user gets clear feedback. The env-var path (XRDC_IO_URING) stays
210 * lenient — it silently falls back to AUTO because env vars are often set
211 * system-wide and must survive software upgrades without breaking callers.
212 * Only the explicit CLI flag is strict.
213 * HOW: Three strcmp checks; anything not matching returns -1 so the caller can
214 * print a tailored "invalid mode 'bogus' (use on|off|auto)" and exit 50.
215 */
216int
217brix_cli_parse_io_uring(const char *s)
218{
219 if (s == NULL || s[0] == '\0') { return -1; }
220 if (strcmp(s, "on") == 0) { return XRDC_IO_URING_ON; }
221 if (strcmp(s, "off") == 0) { return XRDC_IO_URING_OFF; }
222 if (strcmp(s, "auto") == 0) { return XRDC_IO_URING_AUTO; }
223 return -1;
224}
