7/*
8 * http_get_ctx - one-shot HTTP/1.0 GET request/result bundle.
9 *
10 * WHAT: carries the full brix_http_get() call frame — request inputs (host/port/
11 * path/timeout) plus the caller's output slots (status/out-buffer/outlen) —
12 * as a single value so the send/read/extract helpers take one pointer instead
13 * of the original 9-parameter signature.
14 * WHY: collapses brix_http_get's param bloat (9p) and lets the linear GET flow
15 * decompose into single-responsibility helpers without threading eight
16 * arguments through each one; no field aliases the wire behavior.
17 * HOW: populated once at the top of brix_http_get from its public arguments and
18 * passed by pointer to each stage; helpers read inputs and write the caller's
19 * output slots through the stored pointers. Pure data — owns no resources.
20 */
21typedef struct {
22 const char *host;
23 int port;
24 const char *path;
25 int timeout_ms;
26 int *http_status; /* caller slot, may be NULL */
27 char *out; /* caller body buffer */
28 size_t outsz;
29 size_t *outlen; /* caller body-length slot, may be NULL */
30} http_get_ctx;

32/*
33 * http_get_send_request - format and blocking-write the GET request line.
34 *
35 * WHAT: builds "GET <path> HTTP/1.0" with Host + Connection: close and writes it
36 * in full to fd. HOW: snprintf into a stack buffer (bounded-length check), then
37 * a partial-write loop retrying on EINTR. WHY: isolates the request-emit side
38 * effect from the read/parse stages. 0 / -1 (st set; fd left open for caller).
39 */
40static int
41http_get_send_request(int fd, const http_get_ctx *c, brix_status *st)
42{
43 char req[1024];
44 char hp[300];
45 int rlen;
46 ssize_t w, off = 0;
47
48 brix_format_host_port(c->host, (uint16_t) c->port, hp, sizeof(hp));
49 rlen = snprintf(req, sizeof(req),
50 "GET %s HTTP/1.0\r\nHost: %s\r\nConnection: close\r\n\r\n",
51 c->path, hp);
52 if (rlen < 0 || (size_t) rlen >= sizeof(req)) {
53 brix_status_set(st, XRDC_EUSAGE, 0, "http_get: request too long");
54 return -1;
55 }
56 while (off < rlen) {
57 w = write(fd, req + off, (size_t) (rlen - off));
58 if (w < 0) {
59 if (errno == EINTR) { continue; }
60 brix_status_set(st, XRDC_ESOCK, errno, "http write: %s",
61 strerror(errno));
62 return -1;
63 }
64 off += w;
65 }
66 return 0;
67}

69/*
70 * http_get_read_response - read the whole response into resp[cap].
71 *
72 * WHAT: reads to EOF (HTTP/1.0 + Connection: close), bounded by poll(timeout_ms)
73 * and the cap ceiling, NUL-terminating at *total. HOW: poll-then-read loop,
74 * retrying poll/read on EINTR (and EAGAIN on read), stopping on EOF or when the
75 * buffer ceiling is hit. WHY: isolates the read side effect; leaves status/body
76 * parsing to the extract stage. 0 / -1 (st set; fd left open for caller).
77 */
78static int
79http_get_read_response(int fd, char *resp, size_t cap, int timeout_ms,
80 size_t *total, brix_status *st)
81{
82 *total = 0;
83 for (;;) {
84 struct pollfd pfd;
85 ssize_t r;
86 int pr;
87
88 if (*total >= cap - 1) {
89 break; /* hit the ceiling — keep what we have */
90 }
91 pfd.fd = fd;
92 pfd.events = POLLIN;
93 do {
94 pr = poll(&pfd, 1, timeout_ms);
95 } while (pr < 0 && errno == EINTR);
96 if (pr <= 0) {
97 brix_status_set(st, XRDC_ESOCK, pr == 0 ? ETIMEDOUT : errno,
98 "http read %s", pr == 0 ? "timed out" : "poll failed");
99 return -1;
100 }
101 r = read(fd, resp + *total, cap - 1 - *total);
102 if (r == 0) {
103 break; /* EOF — server closed, body complete */
104 }
105 if (r < 0) {
106 if (errno == EINTR || errno == EAGAIN) { continue; }
107 brix_status_set(st, XRDC_ESOCK, errno, "http read: %s", strerror(errno));
108 return -1;
109 }
110 *total += (size_t) r;
111 }
112 resp[*total] = '\0';
113 return 0;
114}

116/*
117 * http_get_extract - parse status line and copy the body into the caller slots.
118 *
119 * WHAT: reads the "HTTP/1.x NNN" status into *c->http_status (when requested) and
120 * copies the post-blank-line body into c->out, reporting its length via
121 * c->outlen. HOW: strchr for the status code; strstr for the header/body split;
122 * binary-safe memcpy (NOT snprintf — a NUL byte must not truncate the body),
123 * then NUL-terminate the truncation guard. WHY: pure parse/copy with no I/O,
124 * trivially testable; kept separate from the socket stages.
125 */
126static void
127http_get_extract(char *resp, size_t total, const http_get_ctx *c)
128{
129 char *eoh, *body;
130 size_t bodylen, n;
131
132 if (c->http_status != NULL && strncmp(resp, "HTTP/", 5) == 0) {
133 char *sp = strchr(resp, ' ');
134 if (sp != NULL) {
135 *c->http_status = atoi(sp + 1);
136 }
137 }
138
139 eoh = strstr(resp, "\r\n\r\n");
140 body = eoh ? eoh + 4 : resp;
141 bodylen = total - (size_t) (body - resp);
142 n = (bodylen < c->outsz - 1) ? bodylen : c->outsz - 1;
143 memcpy(c->out, body, n);
144 c->out[n] = '\0';
145 if (c->outlen != NULL) {
146 *c->outlen = n;
147 }
148}

150int
151brix_http_get(const char *host, int port, const char *path, int timeout_ms,
152 int *http_status, char *out, size_t outsz, size_t *outlen,
153 brix_status *st)
154{
155 http_get_ctx c = { host, port, path, timeout_ms,
156 http_status, out, outsz, outlen };
157 char *resp;
158 size_t total = 0;
159 int fd;
160
161 if (outsz == 0) {
162 brix_status_set(st, XRDC_EUSAGE, 0, "http_get: zero out buffer");
163 return -1;
164 }
165 if (http_status != NULL) {
166 *http_status = 0;
167 }
168
169 fd = brix_tcp_connect(host, port, timeout_ms, st);
170 if (fd < 0) {
171 return -1;
172 }
173
174 if (http_get_send_request(fd, &c, st) != 0) {
175 close(fd);
176 return -1;
177 }
178
179 resp = (char *) malloc(XRDC_HTTP_MAX);
180 if (resp == NULL) {
181 close(fd);
182 brix_status_set(st, XRDC_EPROTO, 0, "http_get: out of memory");
183 return -1;
184 }
185
186 if (http_get_read_response(fd, resp, XRDC_HTTP_MAX, timeout_ms,
187 &total, st) != 0) {
188 free(resp);
189 close(fd);
190 return -1;
191 }
192 close(fd);
193
194 http_get_extract(resp, total, &c);
195 free(resp);
196 return 0;
197}

212/* De-chunk a Transfer-Encoding: chunked body in place; returns new length. */
213size_t
214dechunk(char *b, size_t len)
215{
216 size_t in = 0, out = 0;
217 while (in < len) {
218 char *endp = NULL;
219 long csz = strtol(b + in, &endp, 16);
220 if (endp == NULL || csz < 0) { break; }
221 while (in < len && b[in] != '\n') { in++; } /* skip to end of size line */
222 if (in < len) { in++; } /* past the '\n' */
223 if (csz == 0) { break; } /* last chunk */
224 if (in + (size_t) csz > len) { csz = (long) (len - in); }
225 memmove(b + out, b + in, (size_t) csz);
226 out += (size_t) csz;
227 in += (size_t) csz;
228 while (in < len && (b[in] == '\r' || b[in] == '\n')) { in++; } /* trailing CRLF */
229 }
230 b[out] = '\0';
231 return out;
232}

255/* Parse the read buffer into resp (status line + header block + body). Pure. */
256void
257httpx_parse(char *buf, size_t total, brix_http_resp *resp)
258{
259 char *eoh;
260 if (strncmp(buf, "HTTP/", 5) == 0) {
261 char *sp = strchr(buf, ' ');
262 if (sp != NULL) {
263 char *r2 = strchr(sp + 1, ' ');
264 char *eol = strstr(buf, "\r\n");
265 resp->status = atoi(sp + 1);
266 if (r2 != NULL && eol != NULL && r2 < eol) {
267 size_t n = (size_t) (eol - (r2 + 1));
268 if (n >= sizeof(resp->reason)) { n = sizeof(resp->reason) - 1; }
269 memcpy(resp->reason, r2 + 1, n);
270 resp->reason[n] = '\0';
271 }
272 }
273 }
274 eoh = strstr(buf, "\r\n\r\n");
275 if (eoh != NULL) {
276 size_t hlen = (size_t) (eoh - buf);
277 size_t hn = (hlen < sizeof(resp->headers) - 1) ? hlen : sizeof(resp->headers) - 1;
278 char *bstart = eoh + 4;
279 size_t bodylen = total - (size_t) (bstart - buf);
280 memcpy(resp->headers, buf, hn);
281 resp->headers[hn] = '\0';
282 resp->body = (char *) malloc(bodylen + 1);
283 if (resp->body != NULL) {
284 memcpy(resp->body, bstart, bodylen);
285 resp->body[bodylen] = '\0';
286 resp->body_len = bodylen;
287 if (ci_contains(resp->headers, "Transfer-Encoding: chunked")) {
288 resp->body_len = dechunk(resp->body, resp->body_len);
289 }
290 }
291 } else {
292 size_t hn = (total < sizeof(resp->headers) - 1) ? total : sizeof(resp->headers) - 1;
293 memcpy(resp->headers, buf, hn);
294 resp->headers[hn] = '\0';
295 }
296}

326/* A byte source that first drains leftover bytes captured while reading the
327 * header block, then reads from the socket. Unifies "post-header body already in
328 * the header buffer" with "more body on the wire" for the framing decoders. */
329
330ssize_t
331bsrc_read(body_src *s, void *buf, size_t n, brix_status *st)
332{
333 if (s->lo_off < s->lo_len) {
334 size_t avail = s->lo_len - s->lo_off;
335 size_t cp = (avail < n) ? avail : n;
336 memcpy(buf, s->lo + s->lo_off, cp);
337 s->lo_off += cp;
338 return (ssize_t) cp;
339 }
340 return httpx_read_some(s->io, buf, n, s->timeout_ms, st);
341}

344/* Read one CRLF-terminated line (e.g. a chunk-size line) into out[outsz]; the
345 * trailing CRLF is stripped. 0 / -1. */
346int
347bsrc_getline(body_src *s, char *out, size_t outsz, brix_status *st)
348{
349 size_t n = 0;
350 for (;;) {
351 char ch;
352 ssize_t r = bsrc_read(s, &ch, 1, st);
353 if (r < 0) { return -1; }
354 if (r == 0) { break; }
355 if (ch == '\n') { break; }
356 if (ch == '\r') { continue; }
357 if (n + 1 < outsz) { out[n++] = ch; }
358 }
359 out[n] = '\0';
360 return 0;
361}
