14#define BRIX_STATX_MAX_PATHS 256
15/* kXR_statx returns exactly ONE flag byte per requested path — a packed byte
16 * array, no separators, no NUL (reference XrdXrootdXeq.cc:3194-3203):
17 * kXR_file(0) / kXR_isDir(2) / kXR_other(4) / kXR_offline(8). An inaccessible or
18 * missing path yields kXR_offline. (We previously emitted a full "id size flags
19 * mtime" text line per path — a kXR_stat body — which no standard statx parser
20 * could read.) */
21#define BRIX_STATX_BUF_MAX BRIX_STATX_MAX_PATHS
22
23/* Advance *cursor over one NUL-terminated path in the kXR_statx payload,
24 * returning the path (NULL at end). Bounds-checked against end. */
25static ngx_flag_t
26brix_statx_next_path(const u_char **cursor, const u_char *end,
27 char *path, size_t path_size)
28{
29 const u_char *path_start;
30 size_t path_len;
31
32 /* kXR_statx paths are NEWLINE-separated in the request (reference do_Statx
33 * tokenizes a '\n' list); the final path may lack a trailing newline. (We
34 * previously split on '\0', so a standard client's newline list was read as
35 * one giant path.) */
36 path_start = *cursor;
37 while (*cursor < end && **cursor != '\n') {
38 (*cursor)++;
39 }
40
41 path_len = (size_t) (*cursor - path_start);
42 if (*cursor < end) {
43 (*cursor)++; /* skip the '\n' separator */
44 }
45
46 if (path_len == 0 || path_len >= path_size) {
47 return 0;
48 }
49
50 ngx_memcpy(path, path_start, path_len);
51 path[path_len] = '\0';
52
53 return 1;
54}

56/* Handle kXR_statx — stat each NUL-separated path in the request and return one
57 * inline stat line (or flag byte) per path. */
58ngx_int_t
59brix_handle_statx(brix_ctx_t *ctx, ngx_connection_t *c,
60 ngx_stream_brix_srv_conf_t *conf)
61{
62 const u_char *cursor, *end;
63 u_char *rsp_buf, *rsp_ptr;
64 u_char *rsp_end;
65 char reqpath_buf[BRIX_MAX_PATH + 1];
66 char full_path[PATH_MAX];
67 struct stat st;
68 int n_paths = 0;
69
70 if (ctx->recv.cur_dlen == 0 || ctx->recv.payload == NULL) {
71 BRIX_OP_ERR(ctx, BRIX_OP_STATX);
72 return brix_send_error(ctx, c, kXR_ArgMissing, "no paths given");
73 }
74
75 BRIX_PALLOC_OR_RETURN(rsp_buf, c->pool, BRIX_STATX_BUF_MAX, NGX_ERROR);
76
77 rsp_ptr = rsp_buf;
78 rsp_end = rsp_buf + BRIX_STATX_BUF_MAX;
79 cursor = ctx->recv.payload;
80 end = ctx->recv.payload + ctx->recv.cur_dlen;
81
82 while (cursor < end && n_paths < BRIX_STATX_MAX_PATHS) {
83 if (!brix_statx_next_path(&cursor, end, reqpath_buf,
84 sizeof(reqpath_buf)))
85 {
86 continue;
87 }
88
89 n_paths++;
90
91 /* Internal artifacts (sidecars, upload temps) are invisible → report as
92 * absent, same as a stat miss. */
93 if (brix_is_internal_name(reqpath_buf)) {
94 BRIX_RETURN_ERR(ctx, c, BRIX_OP_STATX, "STATX", reqpath_buf, "-",
95 kXR_NotFound, "file not found");
96 }
97
98 /* Resolve and stat the path. */
99 brix_beneath_full_path(conf->common.root_canon, reqpath_buf,
100 full_path, sizeof(full_path));
101 /*
102 * W4 — apply the SAME authorization gate STAT uses (authdb + VO ACL +
103 * token scope), not just VO ACL + scope. Previously STATX skipped the
104 * authdb check, so an authdb-denied path could leak real metadata via
105 * the batched stat where the single STAT op would have refused it.
106 * A denial here falls through to the per-path "inaccessible" sentinel,
107 * preserving STATX's partial-result semantics.
108 */
109 if (rsp_ptr >= rsp_end) {
110 break;
111 }
112
113 /* The reference do_Statx returns an ERROR (fsError) on the FIRST path
114 * whose stat fails — it does NOT emit a per-path sentinel and continue.
115 * kXR_offline is reserved for a path that stat()s OK but reports mode==-1
116 * (a tape-staged file), handled via the FRM probe below. A missing or
117 * authz-denied path therefore terminates the batch with a kXR_error,
118 * matching XrdXrootdXeq.cc:do_Statx and every standard statx parser. */
119 if (brix_check_authdb(ctx, full_path, BRIX_AUTH_LOOKUP) != NGX_OK
120 || brix_check_vo_acl_identity(c->log, full_path, conf->vo_rules,
121 ctx->identity) != NGX_OK
122 || brix_check_token_scope(ctx, reqpath_buf, 0) != NGX_OK)
123 {
124 BRIX_RETURN_ERR(ctx, c, BRIX_OP_STATX, "STATX", reqpath_buf,
125 "-", kXR_NotAuthorized, "permission denied");
126 }
127 if (brix_stat_beneath(conf->rootfd, reqpath_buf, &st) != 0) {
128 /* Follow fallback for an in-export symlink with a host-ABSOLUTE
129 * target (RESOLVE_IN_ROOT chroots it to ENOENT; stock follows on the
130 * real fs). Match stock, confined via realpath within the export. */
131 int ok = 0;
132 if (errno == ENOENT && conf->common.root_canon[0] != '\0') {
133 char real[PATH_MAX];
134 size_t rl = ngx_strlen(conf->common.root_canon);
135 if (realpath(full_path, real) != NULL
136 && ngx_strncmp(real, conf->common.root_canon, rl) == 0
137 && (real[rl] == '/' || real[rl] == '\0'))
138 {
139 /* Canonical target confirmed within the export root; read
140 * its metadata through the VFS (chain already resolved). */
141 brix_vfs_ctx_t rvctx;
142 brix_vfs_stat_t rvst;
143
144 brix_vfs_ctx_init(&rvctx, c->pool, c->log,
145 BRIX_PROTO_ROOT, conf->common.root_canon, NULL,
146 conf->common.allow_write, 0 /* is_tls */, NULL, real);
147 if (brix_vfs_probe(&rvctx, 0 /* follow */, &rvst)
148 == NGX_OK)
149 {
150 brix_vfs_to_struct_stat(&rvst, &st);
151 ok = 1;
152 }
153 }
154 }
155 if (!ok) {
156 BRIX_RETURN_ERR(ctx, c, BRIX_OP_STATX, "STATX", reqpath_buf,
157 "-", brix_kxr_from_errno(errno),
158 strerror(errno));
159 }
160 }
161
162 {
163 /* One flag byte per path. The reference do_Statx emits ONLY
164 * kXR_isDir or kXR_file (plus kXR_offline for a tape-staged file via
165 * the FRM probe below) — there is NO kXR_other in statx, so a FIFO /
166 * socket / device stats as kXR_file(0), exactly as stock. (Plain
167 * kXR_stat does use kXR_other; statx deliberately does not.) */
168 u_char flag;
169
170 if (S_ISDIR(st.st_mode)) {
171 flag = (u_char) kXR_isDir;
172 } else {
173 flag = (u_char) kXR_file; /* 0 — incl. non-regular */
174 }
175 {
176 brix_vfs_ctx_t _rvc;
177 brix_sd_residency_t _res;
178
179 brix_vfs_ctx_init(&_rvc, c->pool, c->log, BRIX_PROTO_ROOT,
180 conf->common.root_canon, NULL, conf->common.allow_write,
181 0 /* is_tls */, NULL, full_path);
182 if (brix_vfs_residency(&_rvc, &_res, NULL) == NGX_OK
183 && (_res == BRIX_SD_RES_NEARLINE
184 || _res == BRIX_SD_RES_OFFLINE))
185 {
186 flag |= (u_char) kXR_offline;
187 }
188 }
189 *rsp_ptr++ = flag;
190 }
191 }
192
193 {
194 char detail[32];
195
196 snprintf(detail, sizeof(detail), "%d_paths", n_paths);
197 brix_log_access(ctx, c, "STATX", "-", detail, 1, 0, NULL, 0);
198 }
199 BRIX_OP_OK(ctx, BRIX_OP_STATX);
200
201 return brix_send_ok(ctx, c, rsp_buf,
202 (uint32_t)((size_t)(rsp_ptr - rsp_buf)));
203}
