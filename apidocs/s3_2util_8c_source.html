19/*
20 * Build a transient S3 VFS request descriptor for an already-resolved confined
21 * path. Shared by the PUT (put.c) and POST-object (post_object.c) write paths,
22 * which previously carried byte-identical private copies. Pulls pool/log, the
23 * TLS flag, and the authenticated identity from the request; roots and the
24 * write gate from the loc-conf. rootfd stays -1 (transient confined open).
25 */
26void
27s3_build_vfs_ctx(ngx_http_request_t *r, const char *fs_path,
28 ngx_http_s3_loc_conf_t *cf, brix_vfs_ctx_t *vctx)
29{
30 ngx_http_s3_req_ctx_t *s3ctx;
31 int is_tls = 0;
32
33 s3ctx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
34
35#if (NGX_HTTP_SSL)
36 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
37#endif
38
39 brix_vfs_ctx_init(vctx, r->pool, r->connection->log, BRIX_PROTO_S3,
40 cf->common.root_canon, cf->cache_root_canon, cf->common.allow_write,
41 is_tls, (s3ctx != NULL) ? s3ctx->identity : NULL, fs_path);
42 brix_vfs_ctx_bind_backend_cred(vctx,
43 &cf->common.storage_credential_dir,
44 cf->common.storage_credential_fallback);
45 /* Phase-2 T9: S3 access keys have no native x509 — this is the primary
46 * beneficiary of opt-in minting (see fs/backend/cred_mint.h). No-op
47 * unless brix_storage_credential_mint_ca is configured. */
48 brix_vfs_ctx_bind_backend_mint(vctx,
49 &cf->common.storage_credential_mint_ca_cert,
50 &cf->common.storage_credential_mint_ca_key,
51 cf->common.storage_credential_mint_ttl);
52 s3_vfs_bind_deleg(r, cf, vctx);
53}

55/* ---- s3_vfs_bind_deleg -----------------------------------------------------
56 *
57 * WHAT: Bind the request's captured forwardable credential (bearer JWT and/or
58 * user-supplied full x509 proxy PEM) onto a cred-bound VFS ctx, using the
59 * export's resolved delegation mode. See s3.h for the contract.
60 *
61 * WHY: Called at every S3 brix_vfs_ctx_bind_backend_cred site so a delegated
62 * export authenticates the backend leg AS the inbound user rather than the
63 * shared service credential. The bytes were captured once at the auth gate
64 * (bearer in s3_verify_bearer, proxy in handler.c's header capture) and
65 * stashed on the S3 req ctx; here they are handed to the VFS.
66 *
67 * HOW: Reads cf->common.backend_delegation as the mode and the req ctx's
68 * bearer_token / deleg_proxy_pem as the bytes; brix_vfs_deleg_bind is a
69 * no-op for SELECT mode or when nothing was captured. */
70void
71s3_vfs_bind_deleg(ngx_http_request_t *r,
72 ngx_http_s3_loc_conf_t *cf, brix_vfs_ctx_t *vctx)
73{
74 ngx_http_s3_req_ctx_t *s3ctx;
75
76 if (cf->common.backend_delegation == BRIX_CRED_SELECT) {
77 return;
78 }
79
80 s3ctx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
81
82 (void) brix_vfs_deleg_bind(r->pool, vctx,
83 (enum brix_cred_mode) cf->common.backend_delegation,
84 (s3ctx != NULL) ? &s3ctx->bearer_token : NULL,
85 (s3ctx != NULL) ? &s3ctx->deleg_proxy_pem : NULL);
86}

109/*
110 * s3_resolve_key — resolve an S3 object key to a filesystem path.
111 *
112 * Steps:
113 * 1. Strip leading slashes from the key (S3 URLs have /bucket/key).
114 * 2. Prepend exactly one slash so decoded_path starts with /.
115 * 3. Call brix_http_resolve_path() which canonicalizes, URL-decodes,
116 * and confines the path to root.
117 *
118 * Returns: 1 on success (path is confined), 0 if escape detected or
119 * buffer overflow.
120 */
121s3_resolve_key(const char *root, const char *key, char *out, size_t outsz)
122{
123 char key_path[PATH_MAX];
124 int n;
125
126 /* Strip leading slashes; prepend exactly one so decoded_path starts with /. */
127 while (*key == '/')
128 key++;
129
130 n = snprintf(key_path, sizeof(key_path), "/%s", key);
131 if (n < 0 || (size_t) n >= sizeof(key_path))
132 return 0;
133
134 return brix_http_resolve_path(root, key_path, out, outsz) == 0 ? 1 : 0;
135}

197/*
198 * s3_object_crc64nvme_b64 — compute (or cache-read) an object's CRC-64/NVME and
199 * base64-encode it in the AWS x-amz-checksum-crc64nvme wire form.
200 *
201 * WHAT: Produces the 12-char base64 of the 8 big-endian CRC-64/NVME bytes for the
202 * open fd into out (needs >= 13 bytes). Uses the xattr integrity cache, so
203 * the value computed at upload time is reused on later reads.
204 * WHY: AWS S3 clients (SDK/CLI default integrity) send and expect this checksum;
205 * it is base64-of-bytes, NOT hex like the root:///WebDAV digest forms.
206 * HOW: brix_integrity_get_fd("crc64nvme") returns the 16-hex value (from cache
207 * or freshly computed); strtoull → uint64 → 8 big-endian bytes →
208 * ngx_encode_base64. cache_only=1 declines (NGX_DECLINED) on a cache miss
209 * instead of paying a full-file read — used on the GET/HEAD echo path.
210 * Returns NGX_OK (out filled), NGX_DECLINED (cache_only miss), or NGX_ERROR.
211 */
212ngx_int_t
213s3_object_crc64nvme_b64(ngx_http_request_t *r, int fd, const char *path,
214 ngx_flag_t cache_only, char *out, size_t outsz)
215{
216 brix_integrity_info_t info;
217 brix_integrity_opts_t iopts;
218 uint64_t crc;
219 unsigned char be[8];
220 ngx_str_t src, dst;
221 ngx_int_t rc;
222 int i;
223
224 if (out == NULL || outsz < (size_t) (ngx_base64_encoded_length(8) + 1)) {
225 return NGX_ERROR;
226 }
227
228 ngx_memzero(&iopts, sizeof(iopts));
229 iopts.allow_xattr_cache = 1;
230 iopts.update_xattr_cache = cache_only ? 0 : 1;
231 iopts.require_regular_file = 1;
232 iopts.no_compute = cache_only ? 1 : 0;
233
234 rc = brix_integrity_get_fd(r->connection->log, fd, NULL, path, "crc64nvme",
235 &iopts, &info);
236 if (rc != NGX_OK) {
237 return rc; /* NGX_DECLINED (cache-only miss) or NGX_ERROR */
238 }
239
240 /* info.hex is the 16-char lowercase crc64nvme; convert to the raw u64 and
241 * emit the 8 big-endian bytes base64-encoded (the AWS wire format). */
242 crc = (uint64_t) strtoull(info.hex, NULL, 16);
243 for (i = 0; i < 8; i++) {
244 be[i] = (unsigned char) (crc >> (56 - 8 * i));
245 }
246 src.data = be;
247 src.len = sizeof(be);
248 dst.data = (u_char *) out;
249 ngx_encode_base64(&dst, &src);
250 out[dst.len] = '\0';
251 return NGX_OK;
252}
