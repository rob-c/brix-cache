65/*
66 * s3_resolve_key — resolve an S3 object key to a filesystem path.
67 *
68 * Steps:
69 * 1. Strip leading slashes from the key (S3 URLs have /bucket/key).
70 * 2. Prepend exactly one slash so decoded_path starts with /.
71 * 3. Call brix_http_resolve_path() which canonicalizes, URL-decodes,
72 * and confines the path to root.
73 *
74 * Returns: 1 on success (path is confined), 0 if escape detected or
75 * buffer overflow.
76 */
77s3_resolve_key(const char *root, const char *key, char *out, size_t outsz)
78{
79 char key_path[PATH_MAX];
80 int n;
81
82 /* Strip leading slashes; prepend exactly one so decoded_path starts with /. */
83 while (*key == '/')
84 key++;
85
86 n = snprintf(key_path, sizeof(key_path), "/%s", key);
87 if (n < 0 || (size_t) n >= sizeof(key_path))
88 return 0;
89
90 return brix_http_resolve_path(root, key_path, out, outsz) == 0 ? 1 : 0;
91}

150 * CRC-64/NVME object checksum (AWS x-amz-checksum-crc64nvme)
151 * */
152
153/*
154 * s3_object_crc64nvme_b64 — compute (or cache-read) an object's CRC-64/NVME and
155 * base64-encode it in the AWS x-amz-checksum-crc64nvme wire form.
156 *
157 * WHAT: Produces the 12-char base64 of the 8 big-endian CRC-64/NVME bytes for the
158 * open fd into out (needs >= 13 bytes). Uses the xattr integrity cache, so
159 * the value computed at upload time is reused on later reads.
160 * WHY: AWS S3 clients (SDK/CLI default integrity) send and expect this checksum;
161 * it is base64-of-bytes, NOT hex like the root:///WebDAV digest forms.
162 * HOW: brix_integrity_get_fd("crc64nvme") returns the 16-hex value (from cache
163 * or freshly computed); strtoull → uint64 → 8 big-endian bytes →
164 * ngx_encode_base64. cache_only=1 declines (NGX_DECLINED) on a cache miss
165 * instead of paying a full-file read — used on the GET/HEAD echo path.
166 * Returns NGX_OK (out filled), NGX_DECLINED (cache_only miss), or NGX_ERROR.
167 */
168ngx_int_t
169s3_object_crc64nvme_b64(ngx_http_request_t *r, int fd, const char *path,
170 ngx_flag_t cache_only, char *out, size_t outsz)
171{
172 brix_integrity_info_t info;
173 brix_integrity_opts_t iopts;
174 uint64_t crc;
175 unsigned char be[8];
176 ngx_str_t src, dst;
177 ngx_int_t rc;
178 int i;
179
180 if (out == NULL || outsz < (size_t) (ngx_base64_encoded_length(8) + 1)) {
181 return NGX_ERROR;
182 }
183
184 ngx_memzero(&iopts, sizeof(iopts));
185 iopts.allow_xattr_cache = 1;
186 iopts.update_xattr_cache = cache_only ? 0 : 1;
187 iopts.require_regular_file = 1;
188 iopts.no_compute = cache_only ? 1 : 0;
189
190 rc = brix_integrity_get_fd(r->connection->log, fd, NULL, path, "crc64nvme",
191 &iopts, &info);
192 if (rc != NGX_OK) {
193 return rc; /* NGX_DECLINED (cache-only miss) or NGX_ERROR */
194 }
195
196 /* info.hex is the 16-char lowercase crc64nvme; convert to the raw u64 and
197 * emit the 8 big-endian bytes base64-encoded (the AWS wire format). */
198 crc = (uint64_t) strtoull(info.hex, NULL, 16);
199 for (i = 0; i < 8; i++) {
200 be[i] = (unsigned char) (crc >> (56 - 8 * i));
201 }
202 src.data = be;
203 src.len = sizeof(be);
204 dst.data = (u_char *) out;
205 ngx_encode_base64(&dst, &src);
206 out[dst.len] = '\0';
207 return NGX_OK;
208}
