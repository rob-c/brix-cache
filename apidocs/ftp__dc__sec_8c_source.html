112void
113brix_ftp_dc_apply_policy(SSL *ssl)
114{
115 SSL_set_max_proto_version(ssl, TLS1_2_VERSION);
116 /* GridFTP stream mode signals end-of-data by closing the data connection,
117 * and globus does so without a TLS close_notify. Treat that abrupt close as
118 * a clean EOF (SSL_ERROR_ZERO_RETURN) instead of SSL_R_UNEXPECTED_EOF — a
119 * STOR receiver otherwise fails the transfer after the last block. */
120 SSL_set_options(ssl, SSL_OP_IGNORE_UNEXPECTED_EOF);
121 SSL_set_verify(ssl, SSL_VERIFY_PEER | SSL_VERIFY_FAIL_IF_NO_PEER_CERT,
122 brix_ftp_dc_verify_cb);
123}

126/* GSI identity match: does the peer's leaf DN name the same subject as the
127 * control-channel DN? In GSI a proxy's subject is exactly the delegator's
128 * subject plus one `/CN=<value>` component per delegation step, so the peer DN
129 * matches when it either equals the control DN (a plain client transfer, where
130 * the data peer presents the same proxy) or extends it by one or more trailing
131 * `/CN=` proxy components (a gsiftp<->gsiftp TPC leg, where the other server
132 * presents a proxy further delegated from the control identity). The chain has
133 * already been PKIX-verified as a well-formed RFC 3820 proxy chain, so any such
134 * extension is a genuine sub-proxy — the prefix cannot be forged without the
135 * control identity's key. Returns 1 on match, 0 otherwise. */
136static int
137brix_ftp_dc_dn_matches(const char *peer, const u_char *base, size_t blen)
138{
139 size_t plen = ngx_strlen(peer);
140 const char *p;
141
142 if (blen == 0 || plen < blen || ngx_strncmp(peer, base, blen) != 0) {
143 return 0;
144 }
145 if (plen == blen) {
146 return 1; /* exact identity */
147 }
148 if (peer[blen] != '/') {
149 return 0; /* prefix not at an RDN boundary */
150 }
151 /* Every trailing RDN past the control DN must be a `CN=` proxy component. */
152 for (p = peer + blen; *p == '/'; ) {
153 p++;
154 if (ngx_strncmp(p, "CN=", 3) != 0) {
155 return 0;
156 }
157 while (*p != '\0' && *p != '/') {
158 p++;
159 }
160 }
161 return *p == '\0';
162}

193 }
