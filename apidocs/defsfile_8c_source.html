1/*
2 * defsfile.c — load the SciTags experiment/activity registry (JSON).
3 *
4 * WHAT: Parse the scitags "defsfile" — the JSON registry that maps experiment
5 * and activity NAMES to their numeric ids — into an in-memory table the
6 * mapping layer (mapping.c) resolves config rules against. Mirrors XRootD's
7 * XrdNetPMarkCfg LoadJson (XrdNetPMarkCfg.cc:891-966).
8 *
9 * WHY: SciTags config is written in terms of names (experiment "atlas",
10 * activity "write"), but the wire/firefly carries numeric ids; the registry is
11 * the authoritative name→id source (published at scitags.org / by the WLCG).
12 *
13 * HOW: jansson (already linked, see config:70-77 and src/dashboard/api.c) parses
14 * the document {"experiments":[{"expName","expId","activities":[{"activityName",
15 * "activityId"}]}]}. Everything is copied into pool-allocated, NUL-terminated
16 * ngx_str_t so the jansson tree can be freed immediately. A missing file is
17 * NGX_DECLINED (not an error — only a problem if mappings reference names).
18 */
19

26/* Copy a jansson string into a pool-allocated, NUL-terminated ngx_str_t. */
27static ngx_int_t
28pmark_dup_json_str(ngx_pool_t *pool, json_t *js, ngx_str_t *out)
29{
30 const char *s;
31 size_t len;
32
33 if (!json_is_string(js)) {
34 return NGX_ERROR;
35 }
36 s = json_string_value(js);
37 len = ngx_strlen(s);
38
39 out->data = ngx_pnalloc(pool, len + 1);
40 if (out->data == NULL) {
41 return NGX_ERROR;
42 }
43 ngx_memcpy(out->data, s, len);
44 out->data[len] = '\0';
45 out->len = len;
46 return NGX_OK;
47}

50ngx_int_t
51brix_pmark_defsfile_load(const char *path, ngx_pool_t *pool,
52 ngx_array_t **out, ngx_log_t *log)
53{
54 json_t *root, *exps, *e;
55 json_error_t jerr;
56 ngx_array_t *defs;
57 size_t i;
58
59 *out = NULL;
60
61 if (path == NULL || *path == '\0') {
62 return NGX_DECLINED;
63 }
64 if (access(path, R_OK) != 0) {
65 return NGX_DECLINED; /* absent/unreadable → "no registry" */
66 }
67
68 root = json_load_file(path, 0, &jerr);
69 if (root == NULL) {
70 ngx_log_error(NGX_LOG_ERR, log, 0,
71 "pmark: cannot parse defsfile \"%s\": %s (line %d)",
72 path, jerr.text, jerr.line);
73 return NGX_ERROR;
74 }
75
76 exps = json_object_get(root, "experiments");
77 if (!json_is_array(exps)) {
78 json_decref(root);
79 ngx_log_error(NGX_LOG_ERR, log, 0,
80 "pmark: defsfile \"%s\" has no \"experiments\" array", path);
81 return NGX_ERROR;
82 }
83
84 defs = ngx_array_create(pool, json_array_size(exps),
85 sizeof(brix_pmark_exp_def_t));
86 if (defs == NULL) {
87 json_decref(root);
88 return NGX_ERROR;
89 }
90
91 json_array_foreach(exps, i, e) {
92 brix_pmark_exp_def_t *def;
93 json_t *acts, *a;
94 size_t j;
95
96 if (!json_is_object(e)) {
97 continue;
98 }
99
100 def = ngx_array_push(defs);
101 if (def == NULL) {
102 json_decref(root);
103 return NGX_ERROR;
104 }
105 ngx_memzero(def, sizeof(*def));
106
107 if (pmark_dup_json_str(pool, json_object_get(e, "expName"),
108 &def->exp_name) != NGX_OK)
109 {
110 def->exp_name.len = 0;
111 }
112 def->exp_id = (ngx_uint_t) json_integer_value(json_object_get(e, "expId"));
113
114 acts = json_object_get(e, "activities");
115 if (!json_is_array(acts)) {
116 continue;
117 }
118 def->activities = ngx_array_create(pool, json_array_size(acts),
119 sizeof(brix_pmark_named_t));
120 if (def->activities == NULL) {
121 json_decref(root);
122 return NGX_ERROR;
123 }
124
125 json_array_foreach(acts, j, a) {
126 brix_pmark_named_t *act;
127
128 if (!json_is_object(a)) {
129 continue;
130 }
131 act = ngx_array_push(def->activities);
132 if (act == NULL) {
133 json_decref(root);
134 return NGX_ERROR;
135 }
136 ngx_memzero(act, sizeof(*act));
137 if (pmark_dup_json_str(pool, json_object_get(a, "activityName"),
138 &act->name) != NGX_OK)
139 {
140 act->name.len = 0;
141 }
142 act->id = (ngx_uint_t)
143 json_integer_value(json_object_get(a, "activityId"));
144 }
145 }
146
147 json_decref(root);
148 *out = defs;
149 return NGX_OK;
150}

153ngx_uint_t
154brix_pmark_defs_exp_id(ngx_array_t *defs, ngx_str_t *name)
155{
156 brix_pmark_exp_def_t *d;
157 ngx_uint_t i;
158
159 if (defs == NULL || name == NULL) {
160 return 0;
161 }
162 d = defs->elts;
163 for (i = 0; i < defs->nelts; i++) {
164 if (d[i].exp_name.len == name->len
165 && ngx_strncmp(d[i].exp_name.data, name->data, name->len) == 0)
166 {
167 return d[i].exp_id;
168 }
169 }
170 return 0;
171}

175brix_pmark_defs_act_id(ngx_array_t *defs, ngx_uint_t exp_id, ngx_str_t *name)
176{
177 brix_pmark_exp_def_t *d;
178 ngx_uint_t i, j;
179
180 if (defs == NULL || name == NULL) {
181 return 0;
182 }
183 d = defs->elts;
184 for (i = 0; i < defs->nelts; i++) {
185 if (d[i].exp_id != exp_id || d[i].activities == NULL) {
186 continue;
187 }
188 {
189 brix_pmark_named_t *a = d[i].activities->elts;
190 for (j = 0; j < d[i].activities->nelts; j++) {
191 if (a[j].name.len == name->len
192 && ngx_strncmp(a[j].name.data, name->data, name->len) == 0)
193 {
194 return a[j].id;
195 }
196 }
197 }
198 }
199 return 0;
