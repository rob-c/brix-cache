1/*
2 * mapping.c — resolve (experiment, activity) for a transfer + first-use runtime.
3 *
4 * WHAT: Two things. (1) brix_pmark_runtime_ensure() lazily builds a pmark
5 * config's runtime data the first time a flow needs it (per worker): load the
6 * defsfile, resolve every map_experiment/map_activity NAME to its numeric id,
7 * and resolve every firefly_dest "host[:port]" string to a sockaddr. (2)
8 * brix_pmark_map_codes() picks the (experiment, activity) for one flow.
9 *
10 * WHY: This is the nginx analogue of XRootD's XrdNetPMarkCfg::getCodes
11 * (XrdNetPMarkCfg.cc:774-843). Resolution is deferred to first use so it works
12 * uniformly for the stream and HTTP modules without walking every server/
13 * location conf at startup; the result is cached on the conf (COW per worker).
14 *
15 * HOW: Code priority matches XRootD — client scitag.flow (override) → path glob →
16 * VO → default experiment; then activity user → role (parsed from the VOMS
17 * FQAN) → per-experiment default → 1. Everything is fail-open: an unmappable
18 * flow returns NGX_DECLINED and is simply not marked.
19 */

30/* Prefix/glob match: a trailing '*' is stripped, then `path` must start with the
31 * (remaining) pattern. An empty pattern never matches. */
32static int
33pmark_path_match(const char *path, ngx_str_t *pat)
34{
35 size_t n;
36
37 if (path == NULL || pat->len == 0) {
38 return 0;
39 }
40 n = pat->len;
41 if (pat->data[n - 1] == '*') {
42 n--;
43 }
44 if (n == 0) {
45 return 1; /* bare "*" matches everything */
46 }
47 return ngx_strncmp(path, pat->data, n) == 0;
48}

51/* CSV-token exact match: is `needle` one of the comma-separated tokens in csv? */
52static int
53pmark_csv_has(const char *csv, ngx_str_t *needle)
54{
55 const char *p, *start;
56 size_t tok;
57
58 if (csv == NULL || needle->len == 0) {
59 return 0;
60 }
61 p = start = csv;
62 for (;;) {
63 if (*p == ',' || *p == '\0') {
64 tok = (size_t) (p - start);
65 if (tok == needle->len
66 && ngx_strncmp(start, needle->data, tok) == 0)
67 {
68 return 1;
69 }
70 if (*p == '\0') {
71 return 0;
72 }
73 start = p + 1;
74 }
75 p++;
76 }
77}

80/* Extract the VOMS Role from an FQAN-bearing CSV (looks for "Role=<value>").
81 * Writes the value into out[] (NUL-terminated) and returns 1, or 0 if absent. */
82static int
83pmark_extract_role(const char *vo_csv, char *out, size_t outlen)
84{
85 const char *r, *e;
86 size_t n;
87
88 if (vo_csv == NULL) {
89 return 0;
90 }
91 r = strstr(vo_csv, "Role=");
92 if (r == NULL) {
93 return 0;
94 }
95 r += 5; /* past "Role=" */
96 for (e = r; *e && *e != '/' && *e != ','; e++) { /* value end */ }
97 n = (size_t) (e - r);
98 if (n == 0 || n >= outlen) {
99 return 0;
100 }
101 ngx_memcpy(out, r, n);
102 out[n] = '\0';
103 return 1;
104}

107/* Resolve a "host[:port]" string to a sockaddr (default port 10514). */
108static ngx_int_t
109pmark_resolve_dest(ngx_str_t *spec, brix_pmark_dest_t *dst, ngx_log_t *log)
110{
111 char host[256];
112 char port[16];
113 u_char *colon, *end;
114 size_t hlen;
115 struct addrinfo hints, *res;
116 int rc;
117
118 if (spec->len == 0 || spec->len >= sizeof(host)) {
119 return NGX_ERROR;
120 }
121 end = spec->data + spec->len;
122
123 /* Split host[:port]. For a bracketed IPv6 literal "[::1]:port" the colon
124 * after ']' is the separator; otherwise the single/last colon is. */
125 colon = ngx_strlchr(spec->data, end, ':');
126 if (spec->data[0] == '[') {
127 u_char *rb = ngx_strlchr(spec->data, end, ']');
128 colon = (rb && rb + 1 < end && rb[1] == ':') ? rb + 1 : NULL;
129 hlen = rb ? (size_t) (rb - spec->data - 1) : spec->len;
130 ngx_memcpy(host, spec->data + 1, hlen);
131 } else {
132 hlen = colon ? (size_t) (colon - spec->data) : spec->len;
133 ngx_memcpy(host, spec->data, hlen);
134 }
135 host[hlen] = '\0';
136
137 if (colon && colon + 1 < end) {
138 size_t plen = (size_t) (end - (colon + 1));
139 if (plen >= sizeof(port)) {
140 return NGX_ERROR;
141 }
142 ngx_memcpy(port, colon + 1, plen);
143 port[plen] = '\0';
144 } else {
145 ngx_snprintf((u_char *) port, sizeof(port), "%d%Z", BRIX_PMARK_FF_PORT);
146 }
147
148 ngx_memzero(&hints, sizeof(hints));
149 hints.ai_family = AF_UNSPEC;
150 hints.ai_socktype = SOCK_DGRAM;
151
152 rc = getaddrinfo(host, port, &hints, &res);
153 if (rc != 0 || res == NULL) {
154 ngx_log_error(NGX_LOG_WARN, log, 0,
155 "pmark: cannot resolve firefly dest \"%V\": %s",
156 spec, gai_strerror(rc));
157 return NGX_ERROR;
158 }
159
160 ngx_memzero(dst, sizeof(*dst));
161 ngx_memcpy(&dst->ss, res->ai_addr, res->ai_addrlen);
162 dst->len = res->ai_addrlen;
163 dst->family = res->ai_family;
164 freeaddrinfo(res);
165 return NGX_OK;
166}

170 * A load failure is fail-open: named mappings are disabled with a WARN and

172static ngx_array_t *

174{

192/* Resolve experiment rules: name → id (rules that don't resolve are dropped
193 * with a WARN). NGX_ERROR only on allocation failure. */
194static ngx_int_t
195pmark_resolve_exp_rules(brix_pmark_conf_t *pm, ngx_array_t *defs,
196 ngx_pool_t *pool, ngx_log_t *log)
197{
198 brix_pmark_exp_rule_t *src;
199 ngx_uint_t i;
200
201 if (pm->exp_rules == NULL || pm->exp_rules->nelts == 0) {
202 return NGX_OK;
203 }
204 src = pm->exp_rules->elts;
205 pm->exp_rules_r = ngx_array_create(pool, pm->exp_rules->nelts,
206 sizeof(brix_pmark_exp_rule_r_t));
207 if (pm->exp_rules_r == NULL) {
208 return NGX_ERROR;
209 }
210 for (i = 0; i < pm->exp_rules->nelts; i++) {
211 ngx_uint_t id = brix_pmark_defs_exp_id(defs, &src[i].exp_name);
212 brix_pmark_exp_rule_r_t *r;
213 if (id == 0) {
214 ngx_log_error(NGX_LOG_WARN, log, 0,
215 "pmark: experiment \"%V\" not in defsfile; rule ignored",
216 &src[i].exp_name);
217 continue;
218 }
219 r = ngx_array_push(pm->exp_rules_r);
220 if (r == NULL) {
221 return NGX_ERROR;
222 }
223 r->kind = src[i].kind;
224 r->match = src[i].match;
225 r->exp_id = id;
226 }
227 return NGX_OK;
228}

272/* Resolve firefly collector "host[:port]" specs to sockaddrs. Unresolvable
273 * specs are skipped (pmark_resolve_dest already WARNs); "origin" is handled
274 * per-flow against the client peer, not here. NGX_ERROR only on allocation
275 * failure. */
276static ngx_int_t
277pmark_resolve_dests(brix_pmark_conf_t *pm, ngx_pool_t *pool, ngx_log_t *log)
278{
279 ngx_str_t *spec;
280 ngx_uint_t i;
281
282 if (!pm->firefly || pm->firefly_dest == NULL
283 || pm->firefly_dest->nelts == 0)
284 {
285 return NGX_OK;
286 }
287 spec = pm->firefly_dest->elts;
288 pm->dest_sa = ngx_array_create(pool, pm->firefly_dest->nelts,
289 sizeof(brix_pmark_dest_t));
290 if (pm->dest_sa == NULL) {
291 return NGX_ERROR;
292 }
293 for (i = 0; i < pm->firefly_dest->nelts; i++) {
294 brix_pmark_dest_t d;
295 if (spec[i].len == 6
296 && ngx_strncmp(spec[i].data, "origin", 6) == 0)
297 {
298 continue;
299 }
300 if (pmark_resolve_dest(&spec[i], &d, log) == NGX_OK) {
301 brix_pmark_dest_t *dst = ngx_array_push(pm->dest_sa);
302 if (dst == NULL) {
303 return NGX_ERROR;
304 }
305 *dst = d;
306 }
307 }
308 return NGX_OK;
309}

312ngx_int_t
313brix_pmark_runtime_ensure(brix_pmark_conf_t *pm, ngx_pool_t *pool,
314 ngx_log_t *log)
315{
316 ngx_array_t *defs;
317
318 if (pm->rt_ready) {
319 return pm->rt_ok ? NGX_OK : NGX_DECLINED;
320 }
321 pm->rt_ready = 1;
322 pm->rt_ok = 0;
323
324 if (!pm->enable) {
325 return NGX_DECLINED;
326 }
327
328 defs = pmark_load_defs(pm, pool, log);
329
330 if (pmark_resolve_exp_rules(pm, defs, pool, log) != NGX_OK) {
331 return NGX_DECLINED;
332 }
333 if (pmark_resolve_act_rules(pm, defs, pool, log) != NGX_OK) {
334 return NGX_DECLINED;
335 }
336 if (pmark_resolve_dests(pm, pool, log) != NGX_OK) {
337 return NGX_DECLINED;
338 }
339
340 pm->rt_ok = 1;
341 return NGX_OK;
342}

356/* Pick the experiment code for one flow: path glob > VO > default (first
357 * matching rule of each kind wins). Returns 0 when nothing maps. */
358static ngx_uint_t
359pmark_map_experiment(brix_pmark_conf_t *pm, const char *vo_csv,
360 const char *path)
361{
362 ngx_uint_t e_path = 0, e_vo = 0, e_def = 0;
363 ngx_uint_t i;
364
365 if (pm->exp_rules_r == NULL) {
366 return 0;
367 }
368 brix_pmark_exp_rule_r_t *r = pm->exp_rules_r->elts;
369 for (i = 0; i < pm->exp_rules_r->nelts; i++) {
370 switch (r[i].kind) {
371 case BRIX_PMARK_EXP_PATH:
372 if (!e_path && pmark_path_match(path, &r[i].match)) {
373 e_path = r[i].exp_id;
374 }
375 break;
376 case BRIX_PMARK_EXP_VO:
377 if (!e_vo && pmark_csv_has(vo_csv, &r[i].match)) {
378 e_vo = r[i].exp_id;
379 }
380 break;
381 case BRIX_PMARK_EXP_DEFAULT:
382 if (!e_def) {
383 e_def = r[i].exp_id;
384 }
385 break;
386 }
387 }
388 return e_path ? e_path : (e_vo ? e_vo : e_def);
389}

392/* Pick the activity code for experiment `e`: user > VOMS role > per-experiment
393 * default (first matching rule of each kind wins). Returns 0 when no rule for
394 * this experiment maps (the caller applies the XRootD fallback). */
395static ngx_uint_t
396pmark_map_activity(brix_pmark_conf_t *pm, ngx_uint_t e, const char *vo_csv,
397 const char *user)
398{
399 ngx_uint_t a_user = 0, a_role = 0, a_def = 0;
400 ngx_uint_t i;
401 char role[128];
402
403 if (pm->act_rules_r == NULL) {
404 return 0;
405 }
406 role[0] = '\0'; /* empty → matches no rule */
407 (void) pmark_extract_role(vo_csv, role, sizeof(role));
408
409 brix_pmark_act_rule_r_t *r = pm->act_rules_r->elts;
410 for (i = 0; i < pm->act_rules_r->nelts; i++) {
411 if (r[i].exp_id != e) {
412 continue;
413 }
414 switch (r[i].kind) {
415 case BRIX_PMARK_ACTR_USER:
416 if (!a_user && pmark_str_eq(user, &r[i].match)) {
417 a_user = r[i].act_id;
418 }
419 break;
420 case BRIX_PMARK_ACTR_ROLE:
421 if (!a_role && pmark_str_eq(role, &r[i].match)) {
422 a_role = r[i].act_id;
423 }
424 break;
425 case BRIX_PMARK_ACTR_DEFAULT:
426 if (!a_def) {
427 a_def = r[i].act_id;
428 }
429 break;
430 }
431 }
432 return a_user ? a_user : (a_role ? a_role : a_def);
433}

436ngx_int_t
437brix_pmark_map_codes(brix_pmark_conf_t *pm,
438 const brix_pmark_flow_id_t *flow, ngx_uint_t *exp, ngx_uint_t *act)
439{
440 ngx_uint_t e, a;
441
442 *exp = 0;
443 *act = 0;
444
445 /* 1) client scitag.flow override (provides BOTH codes). */
446 if (pm->scitag_cgi && flow->cgi
447 && brix_pmark_parse_scitag(flow->cgi, exp, act) == NGX_OK)
448 {
449 return NGX_OK;
450 }
451
452 /* 2) experiment: path glob > VO > default. */
453 e = pmark_map_experiment(pm, flow->vo_csv, flow->path);
454 if (e == 0) {
455 return NGX_DECLINED; /* nothing maps → not marked */
456 }
457
458 /* 3) activity: user > role > per-experiment default; fallback 1. */
459 a = pmark_map_activity(pm, e, flow->vo_csv, flow->user);
460 if (a == 0) {
461 a = 1; /* XRootD default activity */
462 }
463
464 if (brix_pmark_codes_valid(e, a) != NGX_OK) {
465 return NGX_DECLINED;
466 }
467 *exp = e;
468 *act = a;
469 return NGX_OK;
470}
