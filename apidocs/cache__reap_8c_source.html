18/* A sidecar / slice file we drive off the data file rather than treat as one. */
19static int
20reap_is_sidecar(const char *name)
21{
22 const char *dot = strrchr(name, '.');
23
24 if (dot != NULL && (strcmp(dot, ".cinfo") == 0 || strcmp(dot, ".meta") == 0
25 || strcmp(dot, ".part") == 0 || strcmp(dot, ".lock") == 0))
26 {
27 return 1;
28 }
29 return strstr(name, ".__xrds") != NULL; /* slice files + slice meta */
30}

34reap_unlink_sidecars(const char *data_path)

37

39 (void) unlink(sc);
40 }
41 if (snprintf(sc, sizeof(sc), "%s.meta", data_path) < (int) sizeof(sc)) {
42 (void) unlink(sc);

66/* Recursively scan `dir` (same device as root), reaping aged write-back files:
67 * aged-dirty data (abandoned/incomplete → discarded) and finished write-back
68 * staging copies (completed → reclaimed). Each reaped file bumps the matching
69 * slot->cache_dirty_reaped[reason] (slot may be NULL). */
70static ngx_uint_t
71reap_dir(const char *dir, time_t cutoff, dev_t dev, ngx_log_t *log,
72 ngx_brix_srv_metrics_t *slot, brix_cstore_t *cstore,
73 const char *data_root)
74{
75 DIR *dp;
76 struct dirent *de;
77 ngx_uint_t n = 0;
78 char child[PATH_MAX];
79
80 dp = opendir(dir);
81 if (dp == NULL) {
82 return 0;
83 }
84
85 while ((de = readdir(dp)) != NULL) {
86 struct stat st;
87 brix_cache_cinfo_state_t cs;
88 brix_cache_reap_reason_t reason;
89
90 if (strcmp(de->d_name, ".") == 0 || strcmp(de->d_name, "..") == 0) {
91 continue;
92 }
93 if (reap_is_sidecar(de->d_name)) {
94 continue;
95 }
96 if (snprintf(child, sizeof(child), "%s/%s", dir, de->d_name)
97 >= (int) sizeof(child))
98 {
99 continue;
100 }
101 if (lstat(child, &st) != 0 || st.st_dev != dev) {
102 continue;
103 }
104 if (S_ISDIR(st.st_mode)) {
105 n += reap_dir(child, cutoff, dev, log, slot, cstore, data_root);
106 continue;
107 }
108 if (!S_ISREG(st.st_mode)) {
109 continue;
110 }
111 if (brix_cache_cinfo_state(child, &cs) != NGX_OK) {
112 continue; /* no record → keep (untracked file) */
113 }
114
115 /* Classify WHY this file is reapable (see brix_cache_reap_reason_t). */
116 if (cs.is_dirty) {
117 /* Un-flushed data: reap only once it has aged past the max-age. */
118 if (cs.dirty_since == 0 || (time_t) cs.dirty_since > cutoff) {
119 continue; /* dirty but not yet aged → keep */
120 }
121 reason = (cs.flush_gen > 0) ? BRIX_CACHE_REAP_INCOMPLETE
122 : BRIX_CACHE_REAP_ABANDONED;
123 } else if (cs.flush_gen > 0) {
124 /* Clean AND written back at least once: a finished write-back staging
125 * copy whose bytes are safely on the origin. Reclaim it once the last
126 * flush has aged out. A read-through fill (flush_gen==0, clean) has no
127 * write-back to reclaim and is left for occupancy-driven eviction. */
128 if (cs.last_flush == 0 || (time_t) cs.last_flush > cutoff) {
129 continue; /* completed but not yet aged → keep */
130 }
131 reason = BRIX_CACHE_REAP_COMPLETED;
132 } else {
133 continue; /* clean read-fill → keep (evictable) */
134 }
135
136 /* Remove the DATA through the cstore adapter (phase-64 P3/G5: the policy
137 * layer never unlinks via the store driver itself); cstore_evict also drops
138 * the object's cinfo + L1 entry. The .meta sidecar is the legacy stats plane
139 * the cstore does not own, dropped by reap_unlink_sidecars. For the default
140 * co-located cache the state root IS the data root, so the key is child
141 * minus data_root; a file outside the data root is state-only (raw unlink). */
142 if (cstore != NULL && data_root != NULL
143 && ngx_strncmp(child, data_root, ngx_strlen(data_root)) == 0)
144 {
145 (void) brix_cstore_evict(cstore, child + ngx_strlen(data_root));
146 } else {
147 (void) unlink(child);
148 }
149 reap_unlink_sidecars(child);
150 if (slot != NULL) {
151 (void) ngx_atomic_fetch_add(&slot->cache_dirty_reaped[reason], 1);
152 }
153 if (reason == BRIX_CACHE_REAP_COMPLETED) {
154 ngx_log_error(NGX_LOG_NOTICE, log, 0,
155 "brix: cache reaped completed write-back file "
156 "(flushed, reclaimed): \"%s\"", child);
157 } else {
158 ngx_log_error(NGX_LOG_WARN, log, 0,
159 "brix: cache reaped stale-dirty file (reason=%s, %uL "
160 "un-flushed bytes discarded): \"%s\"",
161 reason == BRIX_CACHE_REAP_INCOMPLETE ? "incomplete"
162 : "abandoned",
163 (unsigned long) (cs.dirty_hi - cs.dirty_lo), child);
164 }
165 n++;
166 }
167
168 closedir(dp);
169 return n;
170}

172ngx_uint_t
173brix_cache_reap_dirty(const ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log)
174{
175 const char *root;
176 struct stat rs;
177 time_t now, cutoff;
178
179 if (conf == NULL || conf->cache_dirty_max_age == 0) {
180 return 0;
181 }
182 root = brix_cache_state_root(conf);
183 if (root == NULL || stat(root, &rs) != 0) {
184 return 0;
185 }
186 now = time(NULL);
187 cutoff = now - conf->cache_dirty_max_age;
188 /* Data removal routes through the cstore adapter (P3/G5); data_root is
189 * cache_root (== the state root for the default co-located cache). The state
190 * tree is still walked raw above — it is the POSIX state plane (.cinfo state
191 * records), not the store's data objects, so it is not a store-driver touch. */
192 return reap_dir(root, cutoff, rs.st_dev, log, reap_metrics_slot(conf),
193 brix_cache_storage_cstore(conf),
194 (const char *) conf->cache_root.data);
195}
