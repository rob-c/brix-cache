18/* A sidecar / slice file we drive off the data file rather than treat as one. */
19static int
20reap_is_sidecar(const char *name)
21{
22 const char *dot = strrchr(name, '.');
23
24 if (dot != NULL && (strcmp(dot, ".cinfo") == 0 || strcmp(dot, ".meta") == 0
25 || strcmp(dot, ".part") == 0 || strcmp(dot, ".lock") == 0))
26 {
27 return 1;
28 }
29 return strstr(name, ".__xrds") != NULL; /* slice files + slice meta */
30}

32/* Remove a reaped data file's sidecars (best-effort). */
33static void
34reap_unlink_sidecars(const char *data_path)
35{
36 char sc[PATH_MAX];
37
38 if (brix_cache_cinfo_path(sc, sizeof(sc), data_path) == 0) {
39 (void) unlink(sc);
40 }
41 if (snprintf(sc, sizeof(sc), "%s.meta", data_path) < (int) sizeof(sc)) {
42 (void) unlink(sc);
43 }
44}

66/*
67 * WHAT: Immutable, per-walk context threaded through the recursive reaper.
68 * WHY: Collapses reap_dir's fixed argument set (cutoff/device/log/metrics/
69 * store/data-root) into a single pointer so the recursive call and every
70 * extracted helper carry one param instead of seven — keeping each
71 * function within the ≤5-param budget without introducing globals.
72 * HOW: Populated once by brix_cache_reap_dirty from the resolved config and
73 * root stat, then passed by const pointer through the whole walk. `dir`
74 * stays a separate argument because it changes per recursion level.
75 */
76typedef struct {
77 time_t cutoff; /* aged-out boundary (now - max_age) */
78 dev_t dev; /* root device — never cross it */
79 ngx_log_t *log; /* reap-event log sink */
80 ngx_brix_srv_metrics_t *slot; /* SHM metrics slot (may be NULL) */
81 brix_cstore_t *cstore; /* data-removal adapter (may be NULL)*/
82 const char *data_root; /* cstore key prefix (may be NULL) */
83} reap_ctx_t;

85/*
86 * WHAT: Decide whether a directory entry names a candidate regular file, and
87 * when so return its full path in `child` (a PATH_MAX buffer).
88 * WHY: Isolates the same-device, dot-skip, sidecar-skip, path-build and type
89 * gating from the classification/removal logic so each stays single-job.
90 * HOW: Skips "."/".."/sidecars, builds `dir/name` (skips on truncation),
91 * lstats it, and rejects anything off the root device. Directories are
92 * signalled to the caller (which recurses); non-regular files are
93 * skipped. Returns 1 when `child` describes a regular candidate,
94 * 0 to skip this entry, and sets *is_dir when a same-device subdir.
95 */
96static int
97reap_entry_candidate(const char *dir, const struct dirent *de,
98 const reap_ctx_t *rc, char *child, int *is_dir)
99{
100 struct stat st;
101
102 *is_dir = 0;
103
104 if (strcmp(de->d_name, ".") == 0 || strcmp(de->d_name, "..") == 0) {
105 return 0;
106 }
107 if (reap_is_sidecar(de->d_name)) {
108 return 0;
109 }
110 if (snprintf(child, PATH_MAX, "%s/%s", dir, de->d_name) >= PATH_MAX) {
111 return 0;
112 }
113 if (lstat(child, &st) != 0 || st.st_dev != rc->dev) {
114 return 0;
115 }
116 if (S_ISDIR(st.st_mode)) {
117 *is_dir = 1;
118 return 0;
119 }
120 return S_ISREG(st.st_mode) ? 1 : 0;
121}

123/*
124 * WHAT: Classify WHY a tracked regular file is reapable (or that it is not).
125 * WHY: Concentrates the aged-dirty vs finished-write-back decision — the
126 * behaviour-critical eviction policy — in one place off the walk loop.
127 * HOW: Reads the cinfo state; keeps untracked files. Dirty data reaps only
128 * once aged past cutoff (incomplete when a flush was started, abandoned
129 * otherwise). Clean-but-flushed staging copies reap once the last flush
130 * ages out (COMPLETED). Clean read-fills are left for occupancy-driven
131 * eviction. Returns 1 with *reason set when reapable, else 0.
132 */
133static int
134reap_classify(const char *child, const reap_ctx_t *rc,
135 brix_cache_cinfo_state_t *cs, brix_cache_reap_reason_t *reason)
136{
137 if (brix_cache_cinfo_state(child, cs) != NGX_OK) {
138 return 0; /* no record → keep (untracked file) */
139 }
140
141 if (cs->is_dirty) {
142 /* Un-flushed data: reap only once it has aged past the max-age. */
143 if (cs->dirty_since == 0 || (time_t) cs->dirty_since > rc->cutoff) {
144 return 0; /* dirty but not yet aged → keep */
145 }
146 *reason = (cs->flush_gen > 0) ? BRIX_CACHE_REAP_INCOMPLETE
147 : BRIX_CACHE_REAP_ABANDONED;
148 return 1;
149 }
150 if (cs->flush_gen > 0) {
151 /* Clean AND written back at least once: a finished write-back staging
152 * copy whose bytes are safely on the origin. Reclaim it once the last
153 * flush has aged out. A read-through fill (flush_gen==0, clean) has no
154 * write-back to reclaim and is left for occupancy-driven eviction. */
155 if (cs->last_flush == 0 || (time_t) cs->last_flush > rc->cutoff) {
156 return 0; /* completed but not yet aged → keep */
157 }
158 *reason = BRIX_CACHE_REAP_COMPLETED;
159 return 1;
160 }
161 return 0; /* clean read-fill → keep (evictable) */
162}

164/*
165 * WHAT: Remove one classified-reapable file (data + sidecars), bump the
166 * metric, and log the reap event.
167 * WHY: Keeps the destructive edge (unlink/evict + counter + log) in a single
168 * helper so the walk loop reads as pure control flow.
169 * HOW: Removes the DATA through the cstore adapter (phase-64 P3/G5: the policy
170 * layer never unlinks via the store driver itself); cstore_evict also
171 * drops the object's cinfo + L1 entry. The .meta sidecar is the legacy
172 * stats plane the cstore does not own, dropped by reap_unlink_sidecars.
173 * For the default co-located cache the state root IS the data root, so
174 * the key is child minus data_root; a file outside the data root is
175 * state-only (raw unlink). Then bumps slot->cache_dirty_reaped[reason]
176 * (slot may be NULL) and logs completed vs stale-dirty distinctly.
177 */
178static void
179reap_remove(const char *child, const reap_ctx_t *rc,
180 const brix_cache_cinfo_state_t *cs, brix_cache_reap_reason_t reason)
181{
182 if (rc->cstore != NULL && rc->data_root != NULL
183 && ngx_strncmp(child, rc->data_root, ngx_strlen(rc->data_root)) == 0)
184 {
185 (void) brix_cstore_evict(rc->cstore, child + ngx_strlen(rc->data_root));
186 } else {
187 (void) unlink(child);
188 }
189 reap_unlink_sidecars(child);
190
191 if (rc->slot != NULL) {
192 (void) ngx_atomic_fetch_add(&rc->slot->cache_dirty_reaped[reason], 1);
193 }
194 if (reason == BRIX_CACHE_REAP_COMPLETED) {
195 ngx_log_error(NGX_LOG_NOTICE, rc->log, 0,
196 "brix: cache reaped completed write-back file "
197 "(flushed, reclaimed): \"%s\"", child);
198 } else {
199 ngx_log_error(NGX_LOG_WARN, rc->log, 0,
200 "brix: cache reaped stale-dirty file (reason=%s, %uL "
201 "un-flushed bytes discarded): \"%s\"",
202 reason == BRIX_CACHE_REAP_INCOMPLETE ? "incomplete"
203 : "abandoned",
204 (unsigned long) (cs->dirty_hi - cs->dirty_lo), child);
205 }
206}

208/*
209 * WHAT: Recursively scan `dir` (same device as root), reaping aged write-back
210 * files: aged-dirty data (abandoned/incomplete → discarded) and finished
211 * write-back staging copies (completed → reclaimed).
212 * WHY: The dirty-reaper's tree walk; bounds abandoned staging the eviction
213 * guard would otherwise protect forever.
214 * HOW: Per entry: reap_entry_candidate gates to a same-device regular file
215 * (recursing into same-device subdirs), reap_classify decides if/why it
216 * is reapable, reap_remove does the destructive work. Each reaped file
217 * bumps rc->slot->cache_dirty_reaped[reason] (slot may be NULL). Returns
218 * the count reaped in this subtree.
219 */
220static ngx_uint_t
221reap_dir(const char *dir, const reap_ctx_t *rc)
222{
223 DIR *dp;
224 struct dirent *de;
225 ngx_uint_t n = 0;
226 char child[PATH_MAX];
227
228 dp = opendir(dir);
229 if (dp == NULL) {
230 return 0;
231 }
232
233 while ((de = readdir(dp)) != NULL) {
234 brix_cache_cinfo_state_t cs;
235 brix_cache_reap_reason_t reason = BRIX_CACHE_REAP_ABANDONED;
236 int is_dir = 0;
237
238 if (!reap_entry_candidate(dir, de, rc, child, &is_dir)) {
239 if (is_dir) {
240 n += reap_dir(child, rc);
241 }
242 continue;
243 }
244 if (!reap_classify(child, rc, &cs, &reason)) {
245 continue;
246 }
247 reap_remove(child, rc, &cs, reason);
248 n++;
249 }
250
251 closedir(dp);
252 return n;
253}

255ngx_uint_t
256brix_cache_reap_dirty(const ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log)
257{
258 const char *root;
259 struct stat rs;
260 reap_ctx_t rc;
261
262 if (conf == NULL || conf->cache_dirty_max_age == 0) {
263 return 0;
264 }
265 root = brix_cache_state_root(conf);
266 if (root == NULL || stat(root, &rs) != 0) {
267 return 0;
268 }
269
270 /* Data removal routes through the cstore adapter (P3/G5); data_root is
271 * cache_root (== the state root for the default co-located cache). The state
272 * tree is still walked raw above — it is the POSIX state plane (.cinfo state
273 * records), not the store's data objects, so it is not a store-driver touch. */
274 ngx_memzero(&rc, sizeof(rc));
275 rc.cutoff = time(NULL) - conf->cache_dirty_max_age;
276 rc.dev = rs.st_dev;
277 rc.log = log;
278 rc.slot = reap_metrics_slot(conf);
279 rc.cstore = brix_cache_storage_cstore(conf);
280 rc.data_root = (const char *) conf->cache_root.data;
281
282 return reap_dir(root, &rc);
283}
