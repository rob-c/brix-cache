1/*
2 * sd_s3.h — shared S3 object-store storage driver (read path; write/MPU follow).
3 *
4 * WHAT: The S3 storage logic — SigV4 signing (shared kernels), HEAD/Range-GET,
5 * and (later) single-PUT + multipart upload — living once in
6 * src/fs/backend/, used by the userland clients (and any future server
7 * consumer) through an injected HTTP transport (sd_s3_transport.h).
8 * WHY: S3 was previously a client-only backend welded to client/lib's HTTP
9 * stack. The transport injection breaks that coupling so the protocol logic
10 * is shared, ngx-free, in libxrdproto.
11 * HOW: Endpoint + credentials + transport are bound at open into an opaque
12 * handle; ops build signed requests and run them through the transport. The
13 * error model is a neutral (int rc, errbuf) pair — no xrdc_status / ngx.
14 */

86/* ---- object metadata (x-amz-meta-*) ----------------------------------- *
87 * S3 has no in-place metadata mutation: get is a HEAD that reads one user-meta
88 * header; set is a copy-onto-self with x-amz-metadata-directive: REPLACE. The
89 * advisory unix-attr blob (mode/uid/gid/mtime) rides in the reserved key
90 * x-amz-meta-<BRIX_META_ADVISORY_S3META>. Attribute names are lowercased to
91 * match the S3 user-metadata contract. */
92
93/* Destination buffer for sd_s3_get_meta: the value lands NUL-terminated in
94 * buf[0..cap). */
95typedef struct { char *buf; size_t cap; } sd_s3_meta_buf;
96
97/* Read user-metadata header x-amz-meta-<name> via a signed HEAD into out
98 * (NUL-terminated). Returns the value length (0 when the attribute is absent),
99 * or -1 with errbuf on a transport/HTTP error. */
100ssize_t sd_s3_get_meta(sd_s3_file *f, const char *name,
101 const sd_s3_meta_buf *out, char *errbuf, size_t errcap);
102
103/* Enumerate every user-metadata name on the object (signed HEAD + raw header
104 * walk) into buf as a listxattr(2)-style NUL-separated block of "user.<name>"
105 * entries; the reserved advisory key (BRIX_META_ADVISORY_S3META) is omitted —
106 * it surfaces as POSIX attrs, not as a user xattr. cap==0 probes the required
107 * size. Returns the byte count, or -1 with errbuf + errno (ERANGE when buf is
108 * too small; ENOTSUP when the transport cannot enumerate headers). */
109ssize_t sd_s3_list_meta(sd_s3_file *f, char *buf, size_t cap,
110 char *errbuf, size_t errcap);
111
112/* One x-amz-meta-<name>=value pair for sd_s3_set_meta. */
113typedef struct { const char *name; const char *value; } sd_s3_meta_kv;
114
115/* Replace the object's user metadata with kv[0..nkv) via a copy-onto-self (PUT
116 * key + x-amz-copy-source: key + x-amz-metadata-directive: REPLACE), each pair
117 * sent and SigV4-signed as x-amz-meta-<name>. 0 / -1 (errbuf). */
118int sd_s3_set_meta(const sd_s3_open_params *p, const sd_s3_meta_kv *kv,
119 size_t nkv, char *errbuf, size_t errcap);
120
121/* Advisory POSIX attrs carried in x-amz-meta-<S3META>. get returns 1 (present +
122 * decoded) / 0 (absent) / -1 (error); set replaces the user metadata with the
123 * single advisory blob. */
124int sd_s3_get_unixattr(sd_s3_file *f, brix_meta_advisory_t *out,
125 char *errbuf, size_t errcap);
126int sd_s3_set_unixattr(const sd_s3_open_params *p,
127 const brix_meta_advisory_t *a,
128 char *errbuf, size_t errcap);
