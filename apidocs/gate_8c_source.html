1/* gate.c — CVMFS protocol dispatch gate.
2 *
3 * WHAT: first step of the dedicated cvmfs:// content handler: restricts
4 * methods to GET/HEAD, classifies the URI, rejects non-CVMFS shapes,
5 * routes the geo API (and, until T12 lands manifest TTLs, the signed
6 * metadata) to the uncached passthrough, and lets CAS objects fall
7 * through (NGX_DECLINED) to the cache-tier serve path.
8 * WHY: a CVMFS cache must not be an open proxy or a generic HTTP endpoint;
9 * class routing here keeps every downstream layer (tier, admission,
10 * verify) free of CVMFS-specific branching.
11 * HOW: pure classifier (classify.c) + early returns; rejects emit one
12 * stable single-line WARN ("cvmfs-reject:") that the httpguard
13 * log-phase classifier and the fail2ban filter (T17) both key on,
14 * with the URI sanitized before logging.
15 */
16#include "cvmfs.h"
17#include "fs/path/path.h"

19/* --- negative cache (T13) --------------------------------------------------
20 * Per-worker fixed-size direct-mapped memo of recent 404s (the deliberate
21 * worker-local exception to the no-globals rule: each worker absorbing its
22 * own 404 storm is sufficient and avoids SHM). A slot collision simply
23 * overwrites (false eviction = one extra origin round-trip); false HITS are
24 * impossible short of a full 64-bit hash collision, and even that only
25 * mis-404s one object for negative_ttl seconds — acceptable for a cache
26 * whose entries are retried by design.
27 */
28#define CVMFS_NEG_SLOTS 512u /* power of two: mask, don't mod */
29
30typedef struct {
31 uint64_t path_hash; /* FNV-1a of the full URI, 0=empty */
32 time_t until;
33} cvmfs_neg_slot;
34
35static cvmfs_neg_slot cvmfs_neg[CVMFS_NEG_SLOTS];
36
37static uint64_t
38cvmfs_neg_hash(const ngx_str_t *uri)
39{
40 uint64_t h = 0xcbf29ce484222325ull;
41 size_t i;
42
43 for (i = 0; i < uri->len; i++) {
44 h = (h ^ uri->data[i]) * 0x100000001b3ull;
45 }
46 return (h != 0) ? h : 1; /* 0 is the empty-slot marker */
47}
48
49static int
50cvmfs_neg_check(const ngx_str_t *uri, time_t now)
51{
52 uint64_t h = cvmfs_neg_hash(uri);
53 cvmfs_neg_slot *s = &cvmfs_neg[h & (CVMFS_NEG_SLOTS - 1)];
54
55 return (s->path_hash == h && now < s->until);
56}
57
58static void
59cvmfs_neg_store(const ngx_str_t *uri, time_t now, time_t ttl)
60{
61 uint64_t h = cvmfs_neg_hash(uri);
62 cvmfs_neg_slot *s = &cvmfs_neg[h & (CVMFS_NEG_SLOTS - 1)];
63
64 s->path_hash = h;
65 s->until = now + ttl;
66}

81static ngx_int_t
82cvmfs_reject(ngx_http_request_t *r, ngx_uint_t status, const char *cause)
83{
84 char safe_uri[512];
85
86 brix_sanitize_log_string((const char *) r->uri.data, safe_uri,
87 sizeof(safe_uri));
88 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
89 "cvmfs-reject: method=%V uri=\"%s\" client=%V class=reject "
90 "cause=\"%s\" "
91 "fix=\"only /cvmfs/<repo>/{data/…,.cvmfspublished,.cvmfswhitelist,"
92 ".cvmfsreflog,api/v1.0/geo/…} are served\"",
93 &r->method_name, safe_uri, &r->connection->addr_text, cause);
94
95 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_REJECT]);
96 return (ngx_int_t) status;
97}

99ngx_int_t
100brix_cvmfs_gate(ngx_http_request_t *r, ngx_http_brix_cvmfs_loc_conf_t *lcf)
101{
102 ngx_http_brix_cvmfs_ctx_t *ctx =
103 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
104
105 /* Classify FIRST (a pure parse of the query-stripped path): every later
106 * reject — wrong method, malformed proxy target, authority not
107 * allowlisted — then carries the request's TRUE class in $cvmfs_class.
108 * (The allowlist 403 used to precede classification, so every rejected
109 * geo/manifest request was logged class=cas — the zero value.) */
110 if (cvmfs_classify_url((const char *) r->uri.data, r->uri.len, &ctx->url)
111 != 0)
112 {
113 return cvmfs_reject(r, NGX_HTTP_FORBIDDEN,
114 "path is not a CVMFS traffic shape");
115 }
116
117 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
118 return cvmfs_reject(r, NGX_HTTP_NOT_ALLOWED, "method not allowed");
119 }
120
121 /* proxy mode (T14): an absolute-form request line names its upstream —
122 * allowlist it, then serve against that upstream's per-worker backend
123 * (convention #2: the override rides the request ctx). */
124 if (r->host_start != NULL) {
125 ngx_str_t up_host;
126 in_port_t up_port;
127 ngx_int_t rc;
128
129 rc = brix_cvmfs_proxy_target(r, &lcf->cvmfs, &up_host, &up_port);
130 if (rc == NGX_HTTP_FORBIDDEN) {
131 return cvmfs_reject(r, NGX_HTTP_FORBIDDEN,
132 "upstream authority not allowlisted");
133 }
134 if (rc == NGX_HTTP_BAD_REQUEST) {
135 return cvmfs_reject(r, NGX_HTTP_BAD_REQUEST,
136 "malformed proxy target");
137 }
138 if (rc == NGX_OK && !lcf->cvmfs.unified_origin) {
139 ngx_uint_t status = NGX_HTTP_INTERNAL_SERVER_ERROR;
140
141 ctx->sd_override = brix_cvmfs_upstream_get(r, lcf, &up_host,
142 up_port,
143 &ctx->up_root,
144 &status);
145 if (ctx->sd_override == NULL) {
146 return (ngx_int_t) status;
147 }
148 }
149 /* rc == NGX_OK && unified_origin: the authority is allowlisted (checked
150 * above) but we do NOT bind a per-host backend — leaving sd_override /
151 * up_root NULL routes the request to the location's ONE configured
152 * multi-endpoint origin backend (brix_cvmfs_storage_backend, ranked
153 * failover + shared cache). Every Stratum-1 the client names is served
154 * from that same set, so a dead origin is hidden by internal failover:
155 * the client keeps getting 200 from the host it chose and never marks
156 * this proxy bad or wanders to another mirror.
157 * rc == NGX_DECLINED: origin-form on this listener — fall through. */
158 }
159
160 /* per-repository accounting (bounded slot table — metrics.h) */
161 ctx->repo = brix_cvmfs_repo_slot(ctx->url.repo, ctx->url.repo_len);
162
163 switch (ctx->url.cls) {
164 case CVMFS_URL_CAS:
165 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_CAS]);
166 if (ctx->repo != NULL) {
167 BRIX_ATOMIC_INC(&ctx->repo->requests_total[BRIX_CVMFS_CLASS_CAS]);
168 }
169 if (lcf->cvmfs.negative_ttl > 0
170 && cvmfs_neg_check(&r->uri, ngx_time()))
171 {
172 char neg_uri[512];
173
174 BRIX_CVMFS_METRIC_INC(negative_hits_total);
175 if (ctx->repo != NULL) {
176 BRIX_ATOMIC_INC(&ctx->repo->negative_hits_total);
177 }
178 ctx->cache_status = BRIX_CVMFS_CACHE_NEG;
179 /* One NOTICE per absorbed 404: a client hammering missing
180 * objects shows as a stream of these (bounded by its own
181 * request rate; the origin sees none of them). */
182 brix_sanitize_log_string((const char *) r->uri.data, neg_uri,
183 sizeof(neg_uri));
184 ngx_log_error(NGX_LOG_NOTICE, r->connection->log, 0,
185 "cvmfs-neg: event=absorbed-404 client=%V uri=\"%s\" "
186 "hint=\"repeated lines from one client = it is retrying a "
187 "missing object instead of backing off\"",
188 &r->connection->addr_text, neg_uri);
189 return NGX_HTTP_NOT_FOUND; /* absorbed 404 storm (T13) */
190 }
191 return NGX_DECLINED; /* tier serve path (handler.c) */
192 case CVMFS_URL_MANIFEST:
193 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_MANIFEST]);
194 if (ctx->repo != NULL) {
195 BRIX_ATOMIC_INC(
196 &ctx->repo->requests_total[BRIX_CVMFS_CLASS_MANIFEST]);
197 }
198 /* T12: signed metadata caches WITH a TTL — the fill stamps
199 * expires_at (= now + brix_cvmfs_manifest_ttl) in the cinfo, an
200 * expired entry refills, and a failed refill serves the stale copy
201 * within the bounded 10x-TTL stale-if-error window. */
202 return NGX_DECLINED;
203 case CVMFS_URL_GEO:
204 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_GEO]);
205 if (ctx->repo != NULL) {
206 BRIX_ATOMIC_INC(&ctx->repo->requests_total[BRIX_CVMFS_CLASS_GEO]);
207 }
208 /* Answer locally (RTT-ranked from this proxy's vantage) when enabled,
209 * bypassing a mis-ordering upstream GeoAPI; else relay verbatim. */
210 if (lcf->cvmfs.geo_answer == BRIX_CVMFS_GEO_RTT) {
211 return brix_cvmfs_geo_answer(r, lcf);
212 }
213 return brix_cvmfs_geo_passthrough(r, lcf);
214 case CVMFS_URL_REJECT:
215 default:
216 return cvmfs_reject(r, NGX_HTTP_FORBIDDEN,
217 "path is not a CVMFS traffic shape");
218 }
219}
