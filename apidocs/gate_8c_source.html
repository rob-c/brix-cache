1/* gate.c — CVMFS protocol dispatch gate.
2 *
3 * WHAT: first step of the dedicated cvmfs:// content handler: restricts
4 * methods to GET/HEAD, classifies the URI, rejects non-CVMFS shapes,
5 * routes the geo API (and, until T12 lands manifest TTLs, the signed
6 * metadata) to the uncached passthrough, and lets CAS objects fall
7 * through (NGX_DECLINED) to the cache-tier serve path.
8 * WHY: a CVMFS cache must not be an open proxy or a generic HTTP endpoint;
9 * class routing here keeps every downstream layer (tier, admission,
10 * verify) free of CVMFS-specific branching.
11 * HOW: pure classifier (classify.c) + early returns; rejects emit one
12 * stable single-line WARN ("cvmfs-reject:") that the httpguard
13 * log-phase classifier and the fail2ban filter (T17) both key on,
14 * with the URI sanitized before logging.
15 */

21/* --- negative cache (T13) --------------------------------------------------
22 * Per-worker fixed-size direct-mapped memo of recent 404s (the deliberate
23 * worker-local exception to the no-globals rule: each worker absorbing its
24 * own 404 storm is sufficient and avoids SHM). A slot collision simply
25 * overwrites (false eviction = one extra origin round-trip); false HITS are
26 * impossible short of a full 64-bit hash collision, and even that only
27 * mis-404s one object for negative_ttl seconds — acceptable for a cache
28 * whose entries are retried by design.
29 */
30#define CVMFS_NEG_SLOTS 512u /* power of two: mask, don't mod */
31
32typedef struct {
33 uint64_t path_hash; /* FNV-1a of the full URI, 0=empty */
34 time_t until;
35} cvmfs_neg_slot;
36
37static cvmfs_neg_slot cvmfs_neg[CVMFS_NEG_SLOTS];
38
39static uint64_t
40cvmfs_neg_hash(const ngx_str_t *uri)
41{
42 uint64_t h = BRIX_FNV1A64_OFFSET_BASIS;
43 size_t i;
44
45 for (i = 0; i < uri->len; i++) {
46 h = (h ^ uri->data[i]) * BRIX_FNV1A64_PRIME;
47 }
48 return (h != 0) ? h : 1; /* 0 is the empty-slot marker */
49}
50
51static int
52cvmfs_neg_check(const ngx_str_t *uri, time_t now)
53{
54 uint64_t h = cvmfs_neg_hash(uri);
55 cvmfs_neg_slot *s = &cvmfs_neg[h & (CVMFS_NEG_SLOTS - 1)];
56
57 return (s->path_hash == h && now < s->until);
58}
59
60static void
61cvmfs_neg_store(const ngx_str_t *uri, time_t now, time_t ttl)
62{
63 uint64_t h = cvmfs_neg_hash(uri);
64 cvmfs_neg_slot *s = &cvmfs_neg[h & (CVMFS_NEG_SLOTS - 1)];
65
66 s->path_hash = h;
67 s->until = now + ttl;
68}

81/* One stable, single-line, guard-parsable WARN per reject (convention #4).
82 * The URI is wire-supplied: sanitize before logging. */
83static ngx_int_t
84cvmfs_reject(ngx_http_request_t *r, ngx_uint_t status, const char *cause)
85{
86 char safe_uri[512];
87 char raw[512];
88 size_t n;
89
90 /* r->uri.data is NOT NUL-terminated (points into the request buffer);
91 * copy the exact uri span before sanitizing, or the sanitizer over-reads
92 * past uri.len (info-leak / log-injection). */
93 n = ngx_min(r->uri.len, sizeof(raw) - 1);
94 ngx_memcpy(raw, r->uri.data, n);
95 raw[n] = '\0';
96 brix_sanitize_log_string(raw, safe_uri, sizeof(safe_uri));
97 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
98 "cvmfs-reject: method=%V uri=\"%s\" client=%V class=reject "
99 "cause=\"%s\" "
100 "fix=\"only /cvmfs/<repo>/{data/…,.cvmfspublished,.cvmfswhitelist,"
101 ".cvmfsreflog,api/v1.0/geo/…} are served\"",
102 &r->method_name, safe_uri, &r->connection->addr_text, cause);
103
104 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_REJECT]);
105 return (ngx_int_t) status;
106}

108/* Emit one unified guard-core audit line (the fail2ban contract, proto=cvmfs)
109 * for this request. `raw` is the wire-supplied span to ride the path field
110 * (sanitized here); it rides alongside, not instead of, the human-readable
111 * cvmfs-reject WARN. */
112static void
113cvmfs_guard_emit(ngx_http_request_t *r, guard_reason_t reason,
114 ngx_uint_t status, const char *raw, size_t rawlen, int cred_present)
115{
116 guard_request_t req;
117 char ipbuf[64];
118 char rawbuf[256];
119 char san[256];
120 char line[512];
121 char ts[sizeof("YYYY-MM-DDThh:mm:ss+00:00")];
122 size_t n, ts_len;
123
124 n = ngx_min(r->connection->addr_text.len, sizeof(ipbuf) - 1);
125 ngx_memcpy(ipbuf, r->connection->addr_text.data, n);
126 ipbuf[n] = '\0';
127
128 n = ngx_min(rawlen, sizeof(rawbuf) - 1);
129 ngx_memcpy(rawbuf, raw, n);
130 rawbuf[n] = '\0';
131
132 req.ip = ipbuf;
133 req.proto = "cvmfs";
134 req.op = GUARD_OP_READ;
135 req.path = san;
136 req.path_len = brix_sanitize_log_string(rawbuf, san, sizeof(san));
137 req.cred_present = cred_present;
138 req.outcome = OUTCOME_PENDING;
139 req.status_code = (int) status;
140
141 ts_len = ngx_cached_http_log_iso8601.len;
142 if (ts_len >= sizeof(ts)) {
143 ts_len = sizeof(ts) - 1;
144 }
145 ngx_memcpy(ts, ngx_cached_http_log_iso8601.data, ts_len);
146 ts[ts_len] = '\0';
147
148 if (guard_audit_format(&req, reason, ts, line, sizeof(line)) > 0) {
149 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0, "%s", line);
150 }
151}

153/* signal=proxyabuse — a forward-proxy request tried to reach a non-allowlisted
154 * / wrong-scheme / malformed remote (the open-proxy / SSRF signal), banned by
155 * the [xrootd-guard-proxyabuse] jail. The attempted upstream authority
156 * (host[:port], straight off the parsed absolute-form request line) rides the
157 * path field so the operator sees which arbitrary remote resource the actor
158 * was reaching for. */
159static void
160cvmfs_guard_proxyabuse(ngx_http_request_t *r, ngx_uint_t status)
161{
162 char raw_auth[256];
163 const u_char *auth_end;
164 size_t alen;
165
166 /* host_start..host_end is the host; a ':' at host_end extends the span to
167 * uri_start to fold in the ":port". Absent host_start can't happen here
168 * (this only runs on an absolute-form target), but guard it anyway. */
169 alen = 0;
170 if (r->host_start != NULL && r->host_end != NULL
171 && r->host_end > r->host_start)
172 {
173 auth_end = r->host_end;
174 if (r->uri_start != NULL && r->uri_start > r->host_end
175 && *r->host_end == ':')
176 {
177 auth_end = r->uri_start;
178 }
179 alen = ngx_min((size_t) (auth_end - r->host_start),
180 sizeof(raw_auth) - 1);
181 ngx_memcpy(raw_auth, r->host_start, alen);
182 }
183
184 cvmfs_guard_emit(r, GUARD_R_PROXYABUSE, status, raw_auth, alen, 0);
185}

187/* Proxy mode (T14): an absolute-form request line names its upstream — allowlist
188 * it, then (non-unified origin) bind that upstream's per-worker backend on the
189 * request ctx. NGX_DECLINED to proceed; otherwise the response rc (reject/status). */
190static ngx_int_t
191cvmfs_gate_proxy_bind(ngx_http_request_t *r,
192 ngx_http_brix_cvmfs_loc_conf_t *lcf, ngx_http_brix_cvmfs_ctx_t *ctx)
193{
194 ngx_str_t up_host;
195 in_port_t up_port;
196 ngx_int_t rc;
197
198 rc = brix_cvmfs_proxy_target(r, &lcf->cvmfs, &up_host, &up_port);
199 if (rc == NGX_HTTP_FORBIDDEN) {
200 cvmfs_guard_proxyabuse(r, NGX_HTTP_FORBIDDEN);
201 return cvmfs_reject(r, NGX_HTTP_FORBIDDEN,
202 "upstream authority not allowlisted");
203 }
204 if (rc == NGX_HTTP_BAD_REQUEST) {
205 cvmfs_guard_proxyabuse(r, NGX_HTTP_BAD_REQUEST);
206 return cvmfs_reject(r, NGX_HTTP_BAD_REQUEST, "malformed proxy target");
207 }
208 if (rc == NGX_OK && !lcf->cvmfs.unified_origin) {
209 ngx_uint_t status = NGX_HTTP_INTERNAL_SERVER_ERROR;
210
211 ctx->sd_override = brix_cvmfs_upstream_get(r, lcf, &up_host, up_port,
212 &ctx->up_root, &status);
213 if (ctx->sd_override == NULL) {
214 return (ngx_int_t) status;
215 }
216 }
217 /* rc == NGX_OK && unified_origin: authority allowlisted (above) but no
218 * per-host backend bound — leaving sd_override/up_root NULL routes to the
219 * location's ONE multi-endpoint origin backend (ranked failover + shared
220 * cache), so a dead origin is hidden by internal failover and the client
221 * never marks this proxy bad. rc == NGX_DECLINED: origin-form — proceed. */
222 return NGX_DECLINED;
223}

225/* CVMFS_URL_CAS accounting + T13 negative-404 absorption. NGX_DECLINED for the
226 * tier serve path; NGX_HTTP_NOT_FOUND when an absorbed-404 storm is short-circuited. */
227static ngx_int_t
228cvmfs_gate_cas(ngx_http_request_t *r,
229 ngx_http_brix_cvmfs_loc_conf_t *lcf, ngx_http_brix_cvmfs_ctx_t *ctx)
230{
231 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_CAS]);
232 if (ctx->repo != NULL) {
233 BRIX_ATOMIC_INC(&ctx->repo->requests_total[BRIX_CVMFS_CLASS_CAS]);
234 }
235
236 if (lcf->cvmfs.negative_ttl > 0 && cvmfs_neg_check(&r->uri, ngx_time())) {
237 char neg_uri[512];
238 char raw[512];
239 size_t n;
240
241 BRIX_CVMFS_METRIC_INC(negative_hits_total);
242 if (ctx->repo != NULL) {
243 BRIX_ATOMIC_INC(&ctx->repo->negative_hits_total);
244 }
245 ctx->cache_status = BRIX_CVMFS_CACHE_NEG;
246 /* One NOTICE per absorbed 404: a client hammering missing objects shows
247 * as a stream of these (bounded by its own request rate). r->uri.data is
248 * NOT NUL-terminated — copy the exact span before sanitizing. */
249 n = ngx_min(r->uri.len, sizeof(raw) - 1);
250 ngx_memcpy(raw, r->uri.data, n);
251 raw[n] = '\0';
252 brix_sanitize_log_string(raw, neg_uri, sizeof(neg_uri));
253 ngx_log_error(NGX_LOG_NOTICE, r->connection->log, 0,
254 "cvmfs-neg: event=absorbed-404 client=%V uri=\"%s\" "
255 "hint=\"repeated lines from one client = it is retrying a "
256 "missing object instead of backing off\"",
257 &r->connection->addr_text, neg_uri);
258 return NGX_HTTP_NOT_FOUND; /* absorbed 404 storm (T13) */
259 }
260 return NGX_DECLINED; /* tier serve path (handler.c) */
261}

263ngx_int_t
264brix_cvmfs_gate(ngx_http_request_t *r, ngx_http_brix_cvmfs_loc_conf_t *lcf)
265{
266 ngx_http_brix_cvmfs_ctx_t *ctx =
267 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
268
269 /* Classify FIRST (a pure parse of the query-stripped path): every later
270 * reject — wrong method, malformed proxy target, authority not
271 * allowlisted — then carries the request's TRUE class in $cvmfs_class.
272 * (The allowlist 403 used to precede classification, so every rejected
273 * geo/manifest request was logged class=cas — the zero value.) */
274 if (cvmfs_classify_url((const char *) r->uri.data, r->uri.len, &ctx->url)
275 != 0)
276 {
277 return cvmfs_reject(r, NGX_HTTP_FORBIDDEN,
278 "path is not a CVMFS traffic shape");
279 }
280
281 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
282 return cvmfs_reject(r, NGX_HTTP_NOT_ALLOWED, "method not allowed");
283 }
284
285 /* proxy mode (T14): an absolute-form request line names its upstream —
286 * allowlist it, then serve against that upstream's per-worker backend
287 * (convention #2: the override rides the request ctx). */
288 if (r->host_start != NULL) {
289 ngx_int_t prc = cvmfs_gate_proxy_bind(r, lcf, ctx);
290 if (prc != NGX_DECLINED) {
291 return prc;
292 }
293 }
294
295 /* per-repository accounting (bounded slot table — metrics.h) */
296 ctx->repo = brix_cvmfs_repo_slot(ctx->url.repo, ctx->url.repo_len);
297
298 /* Token-gated repos (phase-85 F3) — evaluated BEFORE class routing so a
299 * gated repo's CAS, metadata, and geo traffic are all behind the gate. */
300 if (lcf->repo_authz != NULL) {
301 ngx_int_t arc = brix_cvmfs_repo_authz_eval(r, lcf);
302
303 if (arc == NGX_HTTP_UNAUTHORIZED) {
304 /* the guard-core authfail signal ([xrootd-guard-authfail] jail):
305 * unauthenticated probing of a private repo is the same actor
306 * shape as a credential brute-force elsewhere. */
307 cvmfs_guard_emit(r, GUARD_R_AUTHFAIL, NGX_HTTP_UNAUTHORIZED,
308 (const char *) r->uri.data, r->uri.len,
309 r->headers_in.authorization != NULL);
310 return cvmfs_reject(r, NGX_HTTP_UNAUTHORIZED,
311 "repo requires a valid read-scope bearer token");
312 }
313 if (arc == NGX_HTTP_BAD_REQUEST) {
314 return cvmfs_reject(r, NGX_HTTP_BAD_REQUEST,
315 "token-gated repo requires TLS");
316 }
317 if (arc != NGX_DECLINED) {
318 return arc; /* 414/500 plumbing failures */
319 }
320 }
321
322 switch (ctx->url.cls) {
323 case CVMFS_URL_CAS:
324 return cvmfs_gate_cas(r, lcf, ctx);
325 case CVMFS_URL_MANIFEST:
326 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_MANIFEST]);
327 if (ctx->repo != NULL) {
328 BRIX_ATOMIC_INC(
329 &ctx->repo->requests_total[BRIX_CVMFS_CLASS_MANIFEST]);
330 }
331 /* T12: signed metadata caches WITH a TTL — the fill stamps
332 * expires_at (= now + brix_cvmfs_manifest_ttl) in the cinfo, an
333 * expired entry refills, and a failed refill serves the stale copy
334 * within the bounded 10x-TTL stale-if-error window. */
335 return NGX_DECLINED;
336 case CVMFS_URL_GEO:
337 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_GEO]);
338 if (ctx->repo != NULL) {
339 BRIX_ATOMIC_INC(&ctx->repo->requests_total[BRIX_CVMFS_CLASS_GEO]);
340 }
341 /* Answer locally (RTT-ranked from this proxy's vantage) when enabled,
342 * bypassing a mis-ordering upstream GeoAPI; else relay verbatim. */
343 if (lcf->cvmfs.geo_answer == BRIX_CVMFS_GEO_RTT) {
344 return brix_cvmfs_geo_answer(r, lcf);
345 }
346 return brix_cvmfs_geo_passthrough(r, lcf);
347 case CVMFS_URL_REJECT:
348 default:
349 return cvmfs_reject(r, NGX_HTTP_FORBIDDEN,
350 "path is not a CVMFS traffic shape");
351 }
