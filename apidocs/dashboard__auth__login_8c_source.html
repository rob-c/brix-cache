83/* Helpers for sending HTML responses */
84/* Emit a complete text/html response from a static string. The buffer is
85 * memory-backed (b->memory) pointing directly at the const literal — safe
86 * because the data outlives the request and is never mutated. */
87static ngx_int_t
88send_html(ngx_http_request_t *r, ngx_int_t status,
89 const char *html, size_t html_len)
90{
91 ngx_buf_t *b;
92 ngx_chain_t out;
93
94 b = ngx_pcalloc(r->pool, sizeof(*b));
95 if (b == NULL) { return NGX_HTTP_INTERNAL_SERVER_ERROR; }
96
97 b->pos = b->last = (u_char *) html;
98 b->last += html_len;
99 b->memory = 1;
100 b->last_buf = 1;
101
102 r->headers_out.status = status;
103 r->headers_out.content_length_n = (off_t) html_len;
104 r->headers_out.content_type = (ngx_str_t) ngx_string("text/html; charset=utf-8");
105 r->headers_out.content_type_len = r->headers_out.content_type.len;
106
107 ngx_int_t rc = ngx_http_send_header(r);
108 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) { return rc; }
109
110 out.buf = b;
111 out.next = NULL;
112 return ngx_http_output_filter(r, &out);
113}

148/*
149 * WHAT: Reassemble the (possibly multi-buffer) buffered request body into one
150 * NUL-terminated allocation.
151 * WHY: dashboard_form_value() scans a contiguous byte range; nginx delivers
152 * the body as a buffer chain.
153 * HOW: Sum the chain, enforce the 1..4096-byte bound (NGX_DECLINED -> caller
154 * re-renders the form), allocate len+1 from r->pool, copy each buffer,
155 * NUL-terminate. NGX_ERROR on alloc failure.
156 */
157static ngx_int_t
158login_collect_body(ngx_http_request_t *r, u_char **body_out, size_t *total_out)
159{
160 ngx_chain_t *cl;
161 u_char *body_buf;
162 size_t body_len = 0;
163 size_t total = 0;
164 u_char *p;
165
166 for (cl = r->request_body->bufs; cl; cl = cl->next) {
167 body_len += ngx_buf_size(cl->buf);
168 }
169
170 if (body_len == 0 || body_len > 4096) {
171 return NGX_DECLINED;
172 }
173
174 body_buf = ngx_palloc(r->pool, body_len + 1);
175 if (body_buf == NULL) {
176 return NGX_ERROR;
177 }
178
179 p = body_buf;
180 for (cl = r->request_body->bufs; cl; cl = cl->next) {
181 size_t n = ngx_buf_size(cl->buf);
182 ngx_memcpy(p, cl->buf->pos, n);
183 p += n;
184 total += n;
185 }
186 body_buf[total] = '\0';
187
188 *body_out = body_buf;
189 *total_out = total;
190 return NGX_OK;
191}

193/*

198 * hash; the per-user hash becomes the cookie HMAC key.

261/*
262 * WHAT: Mint the session cookie value "<hmac>.<ts>[.<user>]" for verified
263 * credentials and push the Set-Cookie header.
264 * WHY: The cookie must be signed with exactly the key/message the verifier
265 * (dashboard_cookie_hmac) will recompute, and the success audit event is
266 * emitted here, between minting the value and pushing the header, to
267 * preserve the original event ordering.
268 * HOW: Timestamp = now; HMAC via dashboard_cookie_hmac(); assemble the
269 * Set-Cookie value with Path/HttpOnly/Secure/SameSite=Strict attributes.
270 * Returns NGX_OK or NGX_HTTP_INTERNAL_SERVER_ERROR (caller finalizes).
271 */
272static ngx_int_t
273login_issue_cookie(ngx_http_request_t *r,
274 const ngx_http_brix_dashboard_loc_conf_t *conf,
275 const login_creds_t *creds)
276{
277 char ts_str[TIMESTAMP_MAX + 1];
278 char cookie_hex[HMAC_HEX_LEN + 1];
279 char cookie_val[HMAC_HEX_LEN + TIMESTAMP_MAX + 260];
280 ngx_table_elt_t *set_cookie;
281 size_t cookie_full_len;
282 u_char *cookie_full;
283 time_t now;
284
285 now = time(NULL);
286 snprintf(ts_str, sizeof(ts_str), "%ld", (long) now);
287 if (dashboard_cookie_hmac(&creds->hmac_key, ts_str, strlen(ts_str),
288 dashboard_users_enabled(conf)
289 ? creds->username : NULL,
290 creds->username_len, cookie_hex) != NGX_OK)
291 {
292 return NGX_HTTP_INTERNAL_SERVER_ERROR;
293 }
294
295 if (dashboard_users_enabled(conf)) {
296 snprintf(cookie_val, sizeof(cookie_val), "%s.%s.%s", cookie_hex,
297 ts_str, creds->username);
298 } else {
299 snprintf(cookie_val, sizeof(cookie_val), "%s.%s", cookie_hex, ts_str);
300 }
301
302 brix_dashboard_event_add(BRIX_DASH_EVENT_AUTH, 0, NGX_HTTP_OK,
303 "dashboard login success",
304 dashboard_users_enabled(conf)
305 ? creds->username : NULL);
306
307 /* Set-Cookie: xrd_dashboard=<value>; Path=/xrootd; HttpOnly; Secure; SameSite=Strict */
308 set_cookie = ngx_list_push(&r->headers_out.headers);
309 if (set_cookie == NULL) {
310 return NGX_HTTP_INTERNAL_SERVER_ERROR;
311 }
312
313 cookie_full_len = sizeof("xrd_dashboard=; Path=; HttpOnly; Secure; SameSite=Strict") - 1
314 + strlen(cookie_val) + conf->cookie_path.len;
315 cookie_full = ngx_palloc(r->pool, cookie_full_len + 1);
316 if (cookie_full == NULL) {
317 return NGX_HTTP_INTERNAL_SERVER_ERROR;
318 }
319 snprintf((char *) cookie_full, cookie_full_len + 1,
320 "xrd_dashboard=%s; Path=%.*s; HttpOnly; Secure; SameSite=Strict",
321 cookie_val, (int) conf->cookie_path.len,
322 (char *) conf->cookie_path.data);
323
324 set_cookie->hash = 1;
325 ngx_str_set(&set_cookie->key, "Set-Cookie");
326 set_cookie->value.data = cookie_full;
327 set_cookie->value.len = cookie_full_len;
328
329 return NGX_OK;
330}

332/*
333 * WHAT: Install the post-login "Location: <cookie_path>/" redirect header.
334 * WHY: Use r->headers_out.location (nginx's dedicated field) +
335 * NGX_HTTP_MOVED_TEMPORARILY so that nginx's special-response handler
336 * sends a complete response including the body flush. Calling
337 * ngx_http_send_header() alone inside a body callback leaves headers in
338 * the write buffer unflushed because there is no trailing output chain.
339 * HOW: Push the header, build "<cookie_path>/" (NUL-terminated) in r->pool.
340 * Returns NGX_OK or NGX_HTTP_INTERNAL_SERVER_ERROR (caller finalizes).
341 */
342static ngx_int_t
343login_set_redirect(ngx_http_request_t *r,
344 const ngx_http_brix_dashboard_loc_conf_t *conf)
345{
346 u_char *loc;
347
348 r->headers_out.location = ngx_list_push(&r->headers_out.headers);
349 if (r->headers_out.location == NULL) {
350 return NGX_HTTP_INTERNAL_SERVER_ERROR;
351 }
352 r->headers_out.location->hash = 1;
353 ngx_str_set(&r->headers_out.location->key, "Location");
354
355 loc = ngx_pnalloc(r->pool, conf->cookie_path.len + 2);
356 if (loc == NULL) {
357 return NGX_HTTP_INTERNAL_SERVER_ERROR;
358 }
359 ngx_memcpy(loc, conf->cookie_path.data, conf->cookie_path.len);
360 loc[conf->cookie_path.len] = '/';
361 loc[conf->cookie_path.len + 1] = '\0';
362 r->headers_out.location->value.data = loc;
363 r->headers_out.location->value.len = conf->cookie_path.len + 1;
364
365 return NGX_OK;
366}

368/*
369 * WHAT: Body callback for a login POST — verifies credentials and, on success,
370 * issues the session cookie and redirects to the dashboard.
371 * WHY: This runs only after nginx has fully buffered the request body (set up
372 * by ngx_http_brix_dashboard_login_handler via
373 * ngx_http_read_client_request_body). It owns finalizing the request:
374 * every exit path calls ngx_http_finalize_request().
375 * HOW: Reassemble the (possibly multi-buffer) body (login_collect_body),
376 * extract + verify username/password (login_verify_credentials), then
377 * mint "<hmac>.<ts>[.<user>]" as Set-Cookie (login_issue_cookie) and
378 * redirect (login_set_redirect). Failures re-render the login form (200)
379 * so as not to reveal whether the username or password was wrong.
380 */
381static void
382login_post_body_handler(ngx_http_request_t *r)
383{
384 ngx_http_brix_dashboard_loc_conf_t *conf;
385 u_char *body_buf;
386 size_t total = 0;
387 ngx_int_t rc;
388 login_creds_t creds;
389
390 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
391
392 rc = login_collect_body(r, &body_buf, &total);
393 if (rc == NGX_DECLINED) {
394 login_fail_form(r);
395 return;
396 }
397 if (rc != NGX_OK) {
398 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
399 return;
400 }
401
402 ngx_memzero(&creds, sizeof(creds));
403 if (login_verify_credentials(r, conf, body_buf, total, &creds) != NGX_OK) {
404 login_fail_form(r);
405 return;
406 }
407
408 rc = login_issue_cookie(r, conf, &creds);
409 if (rc != NGX_OK) {
410 ngx_http_finalize_request(r, rc);
411 return;
412 }
413
414 rc = login_set_redirect(r, conf);
415 if (rc != NGX_OK) {
416 ngx_http_finalize_request(r, rc);
417 return;
418 }
419
420 ngx_http_finalize_request(r, NGX_HTTP_MOVED_TEMPORARILY);
421}
