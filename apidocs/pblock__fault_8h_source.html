1/*
2 * pblock_fault.h — Phase-83 fault injection (F1) + I/O shaping (F8) for pblock.
3 *
4 * WHAT: The instance-level lab rule cache and the per-open snapshot the hot byte
5 * path evaluates. Rules live as rows in the catalog ctl table
6 * (pblock_ctl.h); this module resolves them into a plain-struct snapshot at
7 * open (a metadata boundary) and evaluates that snapshot — a pure function
8 * over counters, no SQLite, no locks — on every pread/pwrite.
9 *
10 * Wave-A rule grammar (ctl value strings):
11 * fault.pread / fault.pwrite = "errno=EIO [after_bytes=N] | short=N"
12 * shape.read_bps / shape.write_bps = <bytes/sec> (nanosleep paced)
13 * shape.open_ms = <ms> (delay at open)
14 *
15 * WHY: The doc's hard bar is hot-path purity: with lab OFF the driver stores a
16 * NULL lab pointer and the byte path never calls in here. With lab ON, a
17 * set fault only affects handles opened AFTER it was set — the snapshot is
18 * taken once at open, so already-open handles are unaffected (documented
19 * consequence of the epoch model).
20 *
21 * HOW: ngx-free (libc + sqlite3 via pblock_ctl). The instance cache re-reads
22 * ctl only when MAX(epoch) advances; the per-open snapshot is a malloc'd
23 * value copy. Gated by BRIX_HAVE_SQLITE.
24 *
25 * Requires: sd_pblock_catalog.h (pblock_catalog) before inclusion.
26 */

36/* One resolved read-or-write rule. All-zero/-1 fields ⇒ inert. */
37typedef struct {

39 int64_t after_bytes; /* -1: inject immediately; else once cumulative >= */
40 int short_to; /* -1: none; else clamp the op to this many bytes */
41 int64_t rate_bps; /* 0: unlimited; else nanosleep-pace to bytes/sec */
42} pblock_rule_t;

69/* F7 crash points. If ls is non-NULL (lab on) and the ctl `crash.at` value
70 * equals `point`, the worker dies immediately with _exit(PBLOCK_CRASH_EXIT) to
71 * simulate a mid-operation power loss (nginx master respawns it). No-op when ls
72 * is NULL (gate off) or no crash point is armed. Compiled-in points are the
73 * named durability boundaries: after_block_write, before_catalog_update,
74 * after_catalog_update, mid_staged_commit, before_unlink_row. */
