1/*
2 * copy.c — S3 CopyObject handler.
3 *
4 * PUT /bucket/dest-key + x-amz-copy-source: /bucket/src-key
5 *
6 * Performs a server-side copy using copy_file_range (with read/write fallback),
7 * then returns a CopyObjectResult XML body containing the new object's ETag and
8 * LastModified timestamp. Source and destination are both confined to the
9 * configured root_canon to prevent path-traversal.
10 */
11/* WHY: S3 CopyObject enables zero-copy server-side duplication — the source and
12 * destination are both confined to root_canon, avoiding client download/upload
13 * round-trip. copy_file_range provides kernel-level zero-copy when filesystems
14 * support it; read/write fallback ensures portability across filesystem types.
15 * Staged commit (write to temp file then rename) prevents partial writes from
16 * being visible to concurrent readers. */
17

34/* Look up a request header and return its value as a NUL-terminated pool
35 * copy (NULL when absent or on alloc failure) — find + strdupz adapter. */
36static const char *
37s3_copy_find_header(ngx_http_request_t *r, const char *name, size_t nlen)
38{
39 ngx_table_elt_t *h;
40 ngx_str_t value;
41
42 h = brix_http_find_header(r, name, nlen);
43 if (h == NULL
44 || brix_pstrdupz(r->pool, &value, h->value.data, h->value.len)
45 != NGX_OK)
46 {
47 return NULL;
48 }
49 return (const char *) value.data;
50}

71ngx_int_t
72s3_handle_copy_object(ngx_http_request_t *r,
73 const char *dst_fs_path,
74 ngx_http_s3_loc_conf_t *cf)
75{
76 const char *copy_src_hdr;
77 const char *src_key;
78 char src_fs_path[PATH_MAX];
79 struct stat dst_sb;
80 brix_vfs_ctx_t vctx;
81 brix_vfs_copy_opts_t copy_opts;
82 char etag_buf[48];
83 char iso_buf[32];
84 char xml_buf[512];
85 size_t xml_len;
86 ngx_buf_t *b;
87
88 copy_src_hdr = s3_copy_find_header(r, "x-amz-copy-source",
89 sizeof("x-amz-copy-source") - 1);
90 if (copy_src_hdr == NULL) {
91 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
92 "InvalidArgument",
93 "Missing x-amz-copy-source header.");
94 }
95
96 /* Strip optional leading '/' then skip the bucket prefix component */
97 src_key = copy_src_hdr;
98 if (*src_key == '/') {
99 src_key++;
100 }
101 {
102 const char *slash = strchr(src_key, '/');
103 if (slash != NULL) {
104 src_key = slash + 1;
105 }
106 }
107 if (*src_key == '\0') {
108 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
109 "InvalidArgument",
110 "Invalid copy source key.");
111 }
112
113 /* Resolve source path — must stay within root_canon */
114 if (!s3_resolve_key(cf->common.root_canon, src_key, src_fs_path,
115 sizeof(src_fs_path)))
116 {
117 return s3_fail(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
118 "Copy source path is not accessible.",
119 BRIX_S3_EVENT_ACCESS_DENIED);
120 }
121
122 /*
123 * x-amz-metadata-directive (default COPY): REPLACE swaps the destination's
124 * user metadata for the request's x-amz-meta-*; COPY carries the source's
125 * across. A copy-onto-self with REPLACE is the metadata-only update SDKs (and
126 * our own sd_s3 set-meta) use — it must NOT rewrite the bytes.
127 */
128 {
129 const char *directive = s3_copy_find_header(r, "x-amz-metadata-directive",
130 sizeof("x-amz-metadata-directive") - 1);
131 int replace = (directive != NULL
132 && ngx_strcasecmp((u_char *) directive,
133 (u_char *) "REPLACE") == 0);
134 int is_self = (ngx_strcmp(src_fs_path, dst_fs_path) == 0);
135
136 if (!(is_self && replace)) {
137 /* Route the single-file copy through the metered VFS surface
138 * (OP_COPY). src = ctx path, dst passed explicitly; both confined. */
139 s3_copy_vfs_ctx(r, src_fs_path, cf, &vctx);
140 ngx_memzero(&copy_opts, sizeof(copy_opts));
141 copy_opts.overwrite = 1;
142 copy_opts.staged_commit = 1;
143
144 if (brix_vfs_copy(&vctx, dst_fs_path, &copy_opts) != NGX_OK) {
145 if (errno == ENOENT) {
146 return s3_fail(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
147 "The copy source does not exist.",
148 BRIX_S3_EVENT_NO_SUCH_KEY);
149 }
150 BRIX_S3_METRIC_INC(
151 events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
152 return NGX_HTTP_INTERNAL_SERVER_ERROR;
153 }
154 }
155
156 /* Propagate user metadata per the directive (after the bytes land). */
157 if (replace) {
158 (void) s3_apply_put_user_metadata(r, dst_fs_path,
159 cf->common.root_canon);
160 } else {
161 (void) s3_user_meta_copy(r, src_fs_path, dst_fs_path);
162 }
163 }
164
165 /* Confined no-follow probe of the freshly-copied destination for the ETag,
166 * through the VFS seam (non-metered — the CopyObject op already accounts for
167 * this via OP_COPY; a phantom OP_STAT would inflate the stat counter). */
168 {
169 brix_vfs_ctx_t dctx;
170 brix_vfs_stat_t dvst;
171
172 s3_copy_vfs_ctx(r, dst_fs_path, cf, &dctx);
173 if (brix_vfs_probe(&dctx, 1 /* no-follow */, &dvst) != NGX_OK) {
174 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
175 return NGX_HTTP_INTERNAL_SERVER_ERROR;
176 }
177 ngx_memzero(&dst_sb, sizeof(dst_sb));
178 dst_sb.st_size = dvst.size;
179 dst_sb.st_mtime = dvst.mtime;
180 }
181
182 s3_etag(&dst_sb, etag_buf, sizeof(etag_buf));
183 brix_format_iso8601(dst_sb.st_mtime, iso_buf, sizeof(iso_buf));
184
185 xml_len = (size_t) snprintf(xml_buf, sizeof(xml_buf),
186 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
187 "<CopyObjectResult "
188 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">"
189 "<ETag>%s</ETag>"
190 "<LastModified>%s</LastModified>"
191 "</CopyObjectResult>",
192 etag_buf, iso_buf);
193
194 b = ngx_create_temp_buf(r->pool, xml_len);
195 if (b == NULL) {
196 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
197 return NGX_HTTP_INTERNAL_SERVER_ERROR;
198 }
199
200 b->last = ngx_cpymem(b->pos, xml_buf, xml_len);
201 b->last_buf = 1;
202
203 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
204 (ngx_str_t) ngx_string("application/xml"), b);
205}
