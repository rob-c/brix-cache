12/*
13 * ftp_ev_cmd.c — namespace and metadata verbs that complete on the control
14 * channel (no data connection): CWD, SIZE, MKD/DELE/RMD, MDTM/MLST/STAT/CKSM,
15 * and the RNFR/RNTO rename pair.
16 *
17 * WHAT: each handler resolves its path argument under the export, performs one
18 * VFS operation, and queues a single reply.
19 *
20 * WHY: these are pure request→reply verbs — no socket I/O beyond the reply — so
21 * they carry over from the sync engine essentially unchanged, now writing through
22 * the event reply queue instead of a blocking send. Keeping them out of the
23 * dispatcher keeps that file focused on routing.
24 *
25 * HOW: brix_ftp_ev_resolve() confines the path (INVARIANT 4), brix_ftp_ev_vfs_ctx()
26 * builds the identity-bearing context, and the brix_vfs_* seam performs the
27 * operation; write verbs gate on conf->allow_write first (INVARIANT 3-adjacent).
28 */

31/* Format a VFS entry's RFC 3659 facts into `out`; returns the byte count. */
32static size_t
33ev_fmt_facts(char *out, size_t outsz, const brix_vfs_stat_t *st)
34{
35 struct tm tm;
36 char mtime[32];
37 time_t t = st->mtime;
38
39 mtime[0] = '\0';
40 if (gmtime_r(&t, &tm) != NULL) {
41 strftime(mtime, sizeof(mtime), "%Y%m%d%H%M%S", &tm);
42 }
43 return (size_t) snprintf(out, outsz,
44 "type=%s;size=%lld;modify=%s;perm=%s;",
45 st->is_directory ? "dir" : "file",
46 (long long) st->size, mtime,
47 st->is_directory ? "el" : "r");
48}

232/* Map a GridFTP CKSM algorithm token to a BRIX_CK_* kind, or -1 if unknown. */
233static int
234ev_cksm_kind(const char *algo)
235{
236 if (strcasecmp(algo, "ADLER32") == 0) { return BRIX_CK_ADLER32; }
237 if (strcasecmp(algo, "CRC32") == 0) { return BRIX_CK_CRC32; }
238 if (strcasecmp(algo, "CRC32C") == 0) { return BRIX_CK_CRC32C; }
239 if (strcasecmp(algo, "MD5") == 0) { return BRIX_CK_MD5; }
240 if (strcasecmp(algo, "SHA1") == 0) { return BRIX_CK_SHA1; }
241 if (strcasecmp(algo, "SHA256") == 0) { return BRIX_CK_SHA256; }
242 return -1;
243}

247brix_ftp_ev_cmd_cksm(ftp_ev_t *fc, const char *arg)
248{
249 char algo[32], abs[PATH_MAX];
250 long long offset, length;
251 int consumed = 0;
252 int kind;
253 off_t start, len;
254 const char *path;
255 brix_vfs_ctx_t vctx;
256 brix_vfs_file_t *fh;
257 brix_sd_obj_t obj;
258 int verr = 0;
259
260 if (sscanf(arg, "%31s %lld %lld %n", algo, &offset, &length, &consumed) < 3
261 || consumed == 0 || arg[consumed] == '\0')
262 {
263 return brix_ftp_ev_reply(fc,
264 "501 Usage: CKSM <algo> <offset> <length> <path>\r\n");
265 }
266 path = arg + consumed;
267
268 kind = ev_cksm_kind(algo);
269 if (kind < 0) {
270 return brix_ftp_ev_reply(fc, "504 Unsupported checksum algorithm %s\r\n",
271 algo);
272 }
273 if (offset < 0 || length < -1) {
274 return brix_ftp_ev_reply(fc, "501 Invalid CKSM range\r\n");
275 }
276 start = (off_t) offset;
277 len = (length < 0) ? -1 : (off_t) length;
278
279 if (brix_ftp_ev_resolve(fc, path, abs, sizeof(abs)) != 0) {
280 return brix_ftp_ev_reply(fc, "550 No such file\r\n");
281 }
282 brix_ftp_ev_vfs_ctx(fc, abs, &vctx);
283 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &verr);
284 if (fh == NULL) {
285 return brix_ftp_ev_reply(fc, "550 Cannot open file\r\n");
286 }
287 brix_vfs_file_sd_obj(fh, &obj);
288
289 if (kind == BRIX_CK_MD5 || kind == BRIX_CK_SHA1 || kind == BRIX_CK_SHA256) {
290 unsigned char raw[EVP_MAX_MD_SIZE];
291 u_char hex[2 * EVP_MAX_MD_SIZE + 1];
292 unsigned int rawlen = 0;
293 int ok = brix_cksum_digest_obj_range(kind, &obj, start, len,
294 raw, &rawlen);
295 brix_vfs_close(fh, fc->c->log);
296 if (ok != 0) {
297 return brix_ftp_ev_reply(fc, "550 Checksum failed\r\n");
298 }
299 ngx_hex_dump(hex, raw, rawlen);
300 hex[2 * rawlen] = '\0';
301 return brix_ftp_ev_reply(fc, "213 %s\r\n", hex);
302 } else {
303 uint32_t v = 0;
304 char hex[16];
305 int ok = brix_cksum_u32_obj_range(kind, &obj, start, len, &v);
306 brix_vfs_close(fh, fc->c->log);
307 if (ok != 0) {
308 return brix_ftp_ev_reply(fc, "550 Checksum failed\r\n");
309 }
310 snprintf(hex, sizeof(hex), "%08x", (unsigned) v);
311 return brix_ftp_ev_reply(fc, "213 %s\r\n", hex);
312 }
313}
