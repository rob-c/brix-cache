1/*
2 * sd_xroot_io.c — object open + data I/O + stat for the root:// origin driver.
3 *
4 * WHAT: The object-facing vtable slots — open/open_cred (connect the origin,
5 * kXR_open the file, build a memory-served object shell), close, pread
6 * (kXR_read range into a memory sink), preadv (per-segment pread),
7 * pwrite/ftruncate/fsync (the kXR_write/_truncate/_sync write data path),
8 * fstat (handle-snapshot), and stat/stat_cred (open-probe-then-teardown
9 * for the file size).
10 *
11 * WHY: Split out of sd_xroot.c (phase-79 file-size split): the per-object I/O
12 * path is one concept, distinct from the shared origin-open machinery +
13 * driver vtable + lifecycle (sd_xroot.c), the staged atomic-publish path
14 * (sd_xroot_staged.c), and the namespace + metadata ops (sd_xroot_ns.c).
15 *
16 * HOW: Every open (plain, cred, and the stat probes) funnels through
17 * sd_xroot_origin_open (sd_xroot.c) via sd_xroot_open_common, so the
18 * connect → bootstrap → kXR_open sequence and its error mapping live in
19 * one place; the object holds a live origin connection + fhandle and every
20 * read/write is memory-served (no kernel fd).
21 */

30/* sd_xroot_open_common — shared body for plain and cred-scoped opens.
31 *
32 * WHAT: Translate SD flags to want_write, call sd_xroot_origin_open with the
33 * supplied credential (NULL for the plain slot), build the SD object.
34 * WHY: Plain and cred paths differ only in the cred argument; a single body
35 * prevents divergence in error handling and object initialisation.
36 * A cred whose selected kind (bearer/S3) this driver cannot present —
37 * carrying neither an x509 proxy nor a bearer token — must not silently
38 * open on the service credential when the operator asked fallback_deny:
39 * that is exactly the leak the deny flag exists to prevent.
40 * HOW: want_write derived from sd_flags; a fallback_deny cred with no usable
41 * kind is refused (EACCES) before any origin session is opened;
42 * otherwise sd_xroot_origin_open owns the credential copy into the fill
43 * task; obj is heap-allocated (heap_shell=1). */
44static brix_sd_obj_t *
45sd_xroot_open_common(brix_sd_instance_t *inst, const char *path, int sd_flags,
46 mode_t mode, const brix_sd_cred_t *cred, int *err_out)
47{
48 sd_xroot_inst_state *is = inst->state;
49 sd_xroot_obj_state *st;
50 brix_sd_obj_t *obj;
51 off_t size = 0;
52 int want_write;
53
54 /* fallback_deny: cred kind unusable by this driver — refuse, don't fall
55 * back to service cred */
56 if (cred != NULL && cred->fallback_deny
57 && (cred->x509_proxy == NULL || cred->x509_proxy[0] == '\0')
58 && (cred->bearer == NULL || cred->bearer[0] == '\0'))
59 {
60 if (err_out) { *err_out = EACCES; }
61 errno = EACCES;
62 return NULL;
63 }
64
65 want_write = (sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE
66 | BRIX_SD_O_TRUNC | BRIX_SD_O_APPEND)) != 0;
67
68 sd_xroot_origin_open_req_t req = {
69 .conf = is->conf, .cred = cred, .path = path,
70 .want_write = want_write, .mode = mode,
71 .size_out = &size, .err_out = err_out,
72 };
73 st = sd_xroot_origin_open(&req);
74 if (st == NULL) {
75 return NULL;
76 }
77
78 obj = calloc(1, sizeof(*obj));
79 if (obj == NULL) {
80 sd_xroot_obj_teardown(st);
81 if (err_out) { *err_out = ENOMEM; }
82 return NULL;
83 }
84 obj->driver = inst->driver;
85 obj->inst = inst;
86 obj->fd = NGX_INVALID_FILE; /* no kernel fd — memory-served */
87 obj->state = st;
88 obj->heap_shell = 1;
89 obj->snap.size = size;
90 obj->snap.mode = S_IFREG | (want_write ? 0644 : 0444);
91 obj->snap.is_reg = 1;
92 return obj;
93}

95/* sd_xroot_open — vtable open slot: service credential / anonymous.
96 *
97 * WHAT: Plain open for callers that do not carry a per-user credential.
98 * WHY: Preserves the existing public vtable signature; passes cred=NULL so
99 * the bootstrap uses the static service credential (or anonymous).
100 * HOW: Delegates to sd_xroot_open_common with cred=NULL. */
101brix_sd_obj_t *
102sd_xroot_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
103 mode_t mode, int *err_out)
104{
105 return sd_xroot_open_common(inst, path, sd_flags, mode, NULL, err_out);
106}

133ssize_t
134sd_xroot_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
135{
136 sd_xroot_obj_state *st = obj->state;
137 brix_cache_sink_t sink;
138 brix_cache_read_range_t rng;
139
140 if (len == 0) {
141 return 0;
142 }
143 ngx_memzero(&sink, sizeof(sink));
144 sink.fd = -1;
145 sink.mem = buf;
146 sink.mem_cap = len;
147
148 ngx_memzero(&rng, sizeof(rng));
149 rng.read_off = (uint64_t) off;
150 rng.want = len;
151
152 if (brix_cache_origin_read_chunk(st->t, &st->oc, st->fhandle, &sink,
153 &rng) != 0)
154 {
155 errno = EIO;
156 return -1;
157 }
158 return (ssize_t) rng.got;
159}

161/* Write `len` bytes at `off` to the origin file via kXR_write. The origin file
162 * is offset-addressed, so sequential and random writes both map directly. Only
163 * valid on a write handle. Returns bytes written, or -1 with errno. */
164ssize_t
165sd_xroot_pwrite(brix_sd_obj_t *obj, const void *buf, size_t len, off_t off)
166{
167 sd_xroot_obj_state *st = obj->state;
168
169 if (st == NULL || !st->is_write) {
170 errno = EBADF;
171 return -1;
172 }
173 if (len == 0) {
174 return 0;
175 }
176 if (brix_cache_origin_write_chunk(st->t, &st->oc, st->fhandle,
177 (uint64_t) off, (const u_char *) buf, len) != 0)
178 {
179 errno = EIO;
180 return -1;
181 }
182 return (ssize_t) len;
183}

294/* Vectored read: serve each iov segment from the origin (contiguous from `off`).
295 * Reuses sd_xroot_pread per segment — short read on any segment ends the read. */
296ssize_t
297sd_xroot_preadv(brix_sd_obj_t *obj, const struct iovec *iov, int iovcnt,
298 off_t off)
299{
300 ssize_t total = 0;
301 int i;
302
303 for (i = 0; i < iovcnt; i++) {
304 ssize_t n = sd_xroot_pread(obj, iov[i].iov_base, iov[i].iov_len,
305 off + total);
306 if (n < 0) {
307 return (total > 0) ? total : -1;
308 }
309 total += n;
310 if ((size_t) n < iov[i].iov_len) {
311 break; /* short read = EOF */
312 }
313 }
314 return total;
315}
