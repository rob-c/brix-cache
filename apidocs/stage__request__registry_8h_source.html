1/*
2 * stage_request_registry.h — the durable tape/prepare REQUEST registry.
3 *
4 * WHAT: The reqid-keyed durable request store that backs kXR_prepare
5 * (src/query/prepare.c) and the WebDAV Tape REST API
6 * (src/webdav/tape_rest.c): admit a stage request, look it up, check its
7 * owner, list active requests + their files, cancel/delete, release a pin.
8 * This is the FRM-dissolution (§13b, phase-64 P6) re-home of the FRM queue
9 * (former src/frm/queue.c + reqfile.c + reqid.c + index.c) as a peer of the
10 * stage_engine: the request METADATA (owner/cs_type/lfn/status) the engine's
11 * transfer record does not carry, keyed by the same reqid the mover uses.
12 *
13 * WHY: The recall/stage TRANSFER is the stage_engine's job (submit -> mover ->
14 * journal, driven by the sd_frm backend's recall). But the client-facing
15 * request SEMANTICS — a stable reqid the client echoes to cancel, the
16 * requesting identity for owner-checked cancel, custodial-vs-staging class,
17 * the list/status views the Tape REST API returns — are a request registry,
18 * not a transfer queue. Splitting them lets src/frm/ dissolve while the wire
19 * contract (reqid format, owner auth, list/cancel/pin) is preserved exactly.
20 *
21 * HOW: A single durable file (crash-safe append + compaction, inherited from the
22 * FRM reqfile format) under the stage journal dir. reqid keeps the historic
23 * "<seq>.<pid>@<host>" form (BRIX_STAGE_REQID_LEN) so clients that echo a
24 * reqid keep working across the dissolution. All ops are worker-safe (a shm
25 * or file lock over the durable store); list uses an opaque cursor.
26 *
27 * See docs/superpowers/plans/2026-07-01-frm-dissolution.md (Task 4).
28 */

34/* reqid wire format preserved verbatim from the FRM queue ("<seq>.<pid>@<host>").

41/* Per-file checksum type for recall integrity verification (F5): the stage worker
42 * verifies the recalled file against the request's checksum and fails the recall on
43 * mismatch. Values match the former frm_cstype_t (WLCG checksumType names). */
44typedef enum {
45 BRIX_STAGE_CS_NONE = 0,
46 BRIX_STAGE_CS_SHA1 = 1,
47 BRIX_STAGE_CS_SHA2 = 2,
48 BRIX_STAGE_CS_SHA3 = 3,
49 BRIX_STAGE_CS_ADLER32 = 4,
50 BRIX_STAGE_CS_MD5 = 5,
51 BRIX_STAGE_CS_CRC32 = 6
52} brix_stage_cstype_t;

54/* Request lifecycle status (reported by kXR_prepare status + Tape REST). */
55typedef enum {
56 BRIX_STAGE_REQ_QUEUED = 0, /* admitted, recall not yet started */
57 BRIX_STAGE_REQ_ACTIVE, /* recall in flight (mover running) */
58 BRIX_STAGE_REQ_DONE, /* online, ready */
59 BRIX_STAGE_REQ_FAILED, /* recall failed (retryable/terminal per errno)*/
60 BRIX_STAGE_REQ_CANCELLED /* client/owner cancelled */
61} brix_stage_req_status_t;

65/* One request record as returned to the callers (a stable read view; the durable
66 * on-disk layout is private to the .c). */
67typedef struct {
68 char reqid[BRIX_STAGE_REQID_LEN];
69 char lfn[BRIX_STAGE_LFN_LEN]; /* logical path */
70 char requester_dn[BRIX_STAGE_DN_LEN];
71 brix_stage_cstype_t cs_type;
72 brix_stage_req_status_t status;
73 int64_t tod_added;
74 int64_t tod_expire; /* 0 = never (reaper) */
75} brix_stage_request_t;
76
77/* Caller-supplied view of a NEW request for brix_stage_request_add(). Only `lfn`
78 * is required; the rest default. Mirrors the former frm_req_view_t. */
79typedef struct {
80 const char *lfn; /* required — logical (client) path */
81 const char *requester_dn; /* GSI DN / token sub, NULL = anon */
82 const char *user; /* mapped user, NULL = none */
83 brix_stage_cstype_t cs_type; /* checksum alg, default NONE */
84 const char *cs_value; /* checksum value string, NULL = none */
85 int64_t tod_expire; /* 0 = never */
86} brix_stage_request_view_t;
