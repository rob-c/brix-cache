25/* Layer 3 — read-open existence + directory probe.
26 *
27 * A driver-backed export (block-striped / object store) keeps its namespace in
28 * the driver: the physical tree under conf->rootfd holds no data files, so a
29 * POSIX confined stat would spuriously report ENOENT for a file that the driver
30 * actually has. Probe the driver on the export-root-relative key when a driver
31 * is bound; otherwise fall back to the confined fd-relative POSIX stat (the
32 * unchanged default-export path). Returns 1 if the path exists (and sets
33 * *is_dir accordingly), 0 if absent.
34 *
35 * WHY (phase-70) `ctx`/`pool`: on a REMOTE driver-backed export (root://
36 * origin) the driver stat is not a local syscall — it opens the origin, which
37 * demands authentication. Without forwarding the session's delegated credential
38 * (bearer JWT / full proxy) this existence probe fails with "backend has NO
39 * credential" and reports the file absent, turning a legitimate authorized read
40 * into a spurious kXR_NotFound (the data open at brix_open_dispatch_open DOES
41 * forward the credential, so writes — which skip this read probe — succeeded).
42 * The probe binds the same deleg the data open uses and stats through
43 * brix_sd_stat_maybe_cred so the origin sees the inbound user. A NULL ctx/pool
44 * keeps the legacy service-credential behaviour (no passthrough configured). */
45static int
46brix_open_read_probe(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
47 ngx_connection_t *c, const char *clean_path, const char *full_path,
48 int *is_dir)
49{
50 ngx_log_t *log = c->log;
51 brix_sd_instance_t *sd =
52 brix_vfs_backend_resolve(conf->common.root_canon, log);
53
54 *is_dir = 0;
55
56 if (sd != NULL && sd->driver->stat != NULL) {
57 brix_sd_stat_t sst;
58 const char *key =
59 brix_vfs_export_relative_root(full_path, conf->common.root_canon);
60
61 /* Slow-tier composed-cache MISS (phase-64 SP2): a source stat here would
62 * be a blocking wire round-trip on the event loop. Report the path as
63 * provisionally existing (a regular file) and let the OFFLOADED fill
64 * resolve the truth — a missing origin object surfaces as kXR_NotFound
65 * from the parked open's done callback. A COMPLETE hit answers below via
66 * the decorator's cinfo-backed stat (no source touch). */
67 if (brix_sd_cache_fill_needs_offload(sd, key)) {
68 *is_dir = 0;
69 return 1;
70 }
71
72 /* Resolve the session's forwarded backend credential (deleg passthrough
73 * bearer / full proxy, or a per-user credential-dir selection) onto a
74 * transient VFS ctx and present it to the driver stat — the same
75 * credential the data-plane open forwards (open_resolved_file.c) — so a
76 * remote origin authorises this existence probe AS the inbound user. */
77 brix_vfs_ctx_t pvctx;
78 brix_sd_ucred_t ustore;
79 brix_sd_cred_t ucred;
80 int use_cred = 0;
81 int cerr = 0;
82
83 ngx_memzero(&ucred, sizeof(ucred));
84 brix_vfs_ctx_init(&pvctx, c->pool, log, BRIX_PROTO_ROOT,
85 conf->common.root_canon, NULL, conf->common.allow_write,
86 0 /* is_tls */, ctx != NULL ? ctx->identity : NULL, full_path);
87 brix_vfs_ctx_bind_backend_cred(&pvctx,
88 &conf->common.storage_credential_dir,
89 conf->common.storage_credential_fallback);
90 if (ctx != NULL) {
91 brix_root_vfs_bind_deleg(ctx, conf, &pvctx);
92 }
93 if (brix_vfs_backend_cred(&pvctx, &ustore, &ucred, &use_cred, &cerr)
94 != NGX_OK)
95 {
96 /* A deny-mode credential gate refused this principal — treat as a
97 * miss (never distinguish kXR_NotFound from kXR_NotAuthorized; the
98 * read auth gate above already authorised the logical path). */
99 return 0;
100 }
101
102 /* With a forwarded user cred, dispatch on the leaf instance so
103 * brix_sd_stat_maybe_cred finds the leaf driver's stat_cred slot
104 * (mirrors vfs_stat.c): the stage/cache decorators relay stat through
105 * the PLAIN slot only, so probing the TOP instance with a resolved
106 * user cred under fallback=deny hits the forwarder's
107 * cred-would-be-dropped refusal (EACCES, no wire I/O) and a
108 * stage-composed https/xroot origin spuriously reported every
109 * existing file as absent → kXR_NotFound on read-open. Without a
110 * cred the TOP instance must answer: a cache decorator's stat serves
111 * warm hits from cinfo even when the origin object is gone. */
112 if (brix_sd_stat_maybe_cred(use_cred ? brix_vfs_ns_leaf(sd) : sd,
113 key, &sst, use_cred ? &ucred : NULL) != NGX_OK) {
114 brix_sd_ucred_wipe(&ustore); /* secret consumed; erase (A-4/T4) */
115 return 0;
116 }
117 brix_sd_ucred_wipe(&ustore); /* secret consumed; erase (A-4/T4) */
118 *is_dir = sst.is_dir ? 1 : 0;
119 return 1;
120 }
121
122 {
123 struct stat est;
124
125 if (brix_stat_beneath(conf->rootfd, clean_path, &est) != 0) {
126 return 0;
127 }
128 *is_dir = S_ISDIR(est.st_mode) ? 1 : 0;
129 return 1;
130 }
131}

133/*
134 * Phase 35 residency gate for a read open of a nearline/offline file (backend
135 * model via the VFS seam, so a tape:// export classifies with no FRM xattr).
136 * Runs AFTER auth so an unauthorized caller never learns residency. NEARLINE
137 * recalls and parks the client (kXR_wait, or kXR_waitresp under async_recall);
138 * OFFLINE/LOST errors out; NGX_DECLINED when resident or FRM disabled.
139 */
140static ngx_int_t
141brix_open_residency_gate(brix_ctx_t *ctx, ngx_connection_t *c,
142 ngx_stream_brix_srv_conf_t *conf, const char *clean_path,
143 char *full_path, uint16_t options)
144{
145 brix_vfs_ctx_t _rvc;
146 brix_sd_residency_t _res;
147
148 if (!(conf->frm.enable && brix_stage_registry_singleton() != NULL)) {
149 return NGX_DECLINED;
150 }
151
152 brix_vfs_ctx_init(&_rvc, c->pool, c->log, BRIX_PROTO_ROOT,
153 conf->common.root_canon, NULL, conf->common.allow_write,
154 0 /* is_tls */, NULL, full_path);
155
156 if (brix_vfs_residency(&_rvc, &_res, NULL) != NGX_OK) {
157 return NGX_DECLINED;
158 }
159
160 if (_res == BRIX_SD_RES_OFFLINE || _res == BRIX_SD_RES_LOST) {
161 /* A failed/unretrievable recall — do not spin; surface an error so the
162 * client re-prepares or gives up. */
163 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
164 clean_path, "rd", kXR_FSError,
165 "file is offline (recall failed)");
166 }
167
168 if (_res == BRIX_SD_RES_NEARLINE) {
169 brix_stage_request_view_t _v;
170 char _rq[BRIX_STAGE_REQID_LEN];
171 ngx_memzero(&_v, sizeof(_v));
172 _v.lfn = full_path;
173 _v.requester_dn = (ctx->login.dn[0] != '\0') ? ctx->login.dn : NULL;
174 _v.tod_expire = (int64_t) ngx_time()
175 + (int64_t) (conf->frm.stage_ttl / 1000);
176 (void) brix_stage_request_add(
177 brix_stage_registry_singleton(),
178 &_v, _rq, sizeof(_rq), c->log);
179 /* recall driving (former frm_stage_kick) → engine step */
180
181 /* When async recall is on, park the open with kXR_waitresp and wake it in
182 * place via kXR_attn(asynresp) on completion. Falls back to the kXR_wait
183 * poll model if async is off or the waiter table is full. */
184 if (conf->frm.async_recall
185 && brix_stage_waiter_add(_rq, options,
186 ctx->recv.cur_streamid, c->fd, c->number,
187 ngx_pid, conf->frm.stage_ttl) == NGX_OK)
188 {
189 brix_log_access(ctx, c, "OPEN", clean_path,
190 "staging-async", 1, 0, NULL, 0);
191 (void) brix_send_waitresp(ctx, c);
192 ctx->state = XRD_ST_WAITING_FRM;
193 ngx_add_timer(c->read, conf->frm.stage_ttl);
194 return NGX_AGAIN;
195 }
196
197 brix_log_access(ctx, c, "OPEN", clean_path, "staging",
198 1, kXR_wait, NULL, 0);
199 return brix_send_wait(ctx, c, conf->frm.stage_wait);
200 }
201
202 return NGX_DECLINED;
203}

205/*
206 * Read-open resolution: cache-aware read returns directly; otherwise build the
207 * absolute path, run the auth gate BEFORE probing existence (no namespace
208 * existence oracle), reject a missing/dir target (or forward upstream), serve a
209 * ZIP member, and apply the residency gate. Fills full_path. NGX_DECLINED to
210 * proceed to brix_open_resolved_file; otherwise the response rc.
211 */
212ngx_int_t
213brix_open_read_resolve(brix_ctx_t *ctx, ngx_connection_t *c,
214 ngx_stream_brix_srv_conf_t *conf, const char *clean_path,
215 char *full_path, uint16_t options, uint16_t mode_bits)
216{
217 if (conf->cache) {
218 return brix_open_cached_read(ctx, c, conf, clean_path,
219 options, mode_bits);
220 }
221
222 /* Read opens: build the absolute path, then authorize BEFORE probing
223 * existence. A denied principal must never distinguish kXR_NotFound from
224 * kXR_NotAuthorized (a namespace existence oracle) and must not be forwarded
225 * upstream on a miss. The gate keys off (identity, logical path) with no
226 * dependency on the probe, so running it first is behavior-identical for
227 * every authorized principal (mirrors statx.c / locate.c ordering). */
228 brix_beneath_full_path(conf->common.root_canon, clean_path,
229 full_path, PATH_MAX);
230
231 if (brix_auth_gate(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
232 clean_path, full_path, conf,
233 BRIX_AUTH_READ, 0) != NGX_OK) {
234 return ctx->write_rc;
235 }
236
237 {
238 int _exists, _is_dir;
239
240 _exists = brix_open_read_probe(ctx, conf, c, clean_path,
241 full_path, &_is_dir);
242 if (!_exists) {
243 if (conf->upstream_host.len > 0) {
244 brix_log_access(ctx, c, "OPEN", clean_path,
245 "upstream", 1, 0, NULL, 0);
246 BRIX_OP_OK(ctx, BRIX_OP_OPEN_RD);
247 return brix_upstream_start(ctx, c, conf);
248 }
249 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
250 clean_path, "rd", kXR_NotFound,
251 "file not found");
252 }
253 if (_is_dir) {
254 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
255 clean_path, "rd", kXR_isDirectory,
256 "is a directory");
257 }
258 }
259
260 /* Phase-57 W2: ZIP member access. The archive existence check and READ auth
261 * above gate access to the member; serve the requested member instead of the
262 * whole file. (The read-through cache path returned earlier, so this is the
263 * direct-serve case.) */
264 if (conf->zip_access) {
265 char zipmember[PATH_MAX];
266 int zr = open_extract_zip_member(ctx, zipmember,
267 sizeof(zipmember));
268 if (zr < 0) {
269 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
270 clean_path, "zip", kXR_ArgInvalid,
271 "invalid zip member name");
272 }
273 if (zr > 0) {
274 return brix_zip_open_member(ctx, c, conf, clean_path,
275 full_path, zipmember, options);
276 }
277 }
278
279 return brix_open_residency_gate(ctx, c, conf, clean_path, full_path,
280 options);
281}

283/*
284 * Write-open resolution: build the absolute path, run the create/update auth
285 * gate, and (kXR_mkpath | kXR_async) make the parent directory chain confined
286 * beneath the export root. Fills full_path. NGX_DECLINED to proceed.
287 */
288ngx_int_t
289brix_open_write_resolve(brix_ctx_t *ctx, ngx_connection_t *c,
290 ngx_stream_brix_srv_conf_t *conf, const char *clean_path,
291 char *full_path, uint16_t options)
292{
293 brix_beneath_full_path(conf->common.root_canon, clean_path,
294 full_path, PATH_MAX);
295
296 /* XrdAcc distinguishes creating a NEW file (needs Insert) from updating an
297 * existing one; kXR_new is the create intent. */
298 if (brix_auth_gate_op(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
299 clean_path, full_path, conf,
300 BRIX_AUTH_UPDATE, 1,
301 (options & kXR_new) ? BRIX_AOP_CREATE
302 : BRIX_AOP_UPDATE) != NGX_OK) {
303 return ctx->write_rc;
304 }
305
306 /* Create the parent directory chain when the client set kXR_mkpath OR
307 * kXR_async — the EXACT condition the reference do_Open uses to add
308 * SFS_O_MKPTH. xrdcp sends kXR_async on every upload, so this is what makes
309 * uploads to a missing parent succeed; a create-open with NEITHER flag that
310 * names a missing parent must fail NotFound, matching stock. */
311 if (options & (kXR_mkpath | kXR_async)) {
312 char parent[PATH_MAX];
313 char *slash;
314 ngx_cpystrn((u_char *) parent, (u_char *) full_path, sizeof(parent));
315 slash = strrchr(parent, '/');
316 if (slash && slash > parent) {
317 *slash = '\0';
318 /* mode 0755 for new directories; propagate group policy */
319 brix_mkdir_recursive_policy(parent, 0755, c->log,
320 conf->group_rules);
321 }
322 }
323
324 return NGX_DECLINED;
325}
