6/* brix_ckpxeq_body_extra — trailing sub-body length for kXR_ckpXeq
7 * WHAT: Computes how many more bytes the client will stream after the ckpXeq
8 * frame, staged on `have` (body bytes received so far). have == 24: the
9 * embedded sub-header just landed — the sub-request's own dlen bytes follow
10 * (write/pgwrite data, or the writev descriptor block). have == 24 +
11 * sub_dlen: an embedded writev's descriptors landed — sum(wlen) data bytes
12 * follow, delegated to brix_writev_body_extra (the SAME contract as a
13 * standalone kXR_writev). *final = 0 tells the recv framing to call again
14 * after the current extension completes.
15 * WHY: Stock wire contract (do_ChkPntXeq + XrdCl MessageUtils): the outer
16 * chkpoint dlen frames ONLY the embedded 24-byte header; everything else is
17 * raw body streamed behind the frame. Shared by the recv framing so the
18 * whole sub-request lands contiguously in payload_buf before dispatch.
19 * HOW: Bound every stage by BRIX_MAX_WRITE_PAYLOAD (and the writev vector
20 * cap) and return NGX_DECLINED on any violation — the framing then does NOT
21 * extend, the auth gates still run, and ckp_xeq detects the un-read trailing
22 * bytes (count mismatch) and drops the link. */
23ngx_int_t
24brix_ckpxeq_body_extra(const u_char *body, uint32_t have,
25 uint32_t *extra, unsigned *final)
26{
27 uint16_t sub_reqid;
28 uint32_t sub_dlen;
29
30 *extra = 0;
31 *final = 1;
32
33 if (body == NULL || have < 24) {
34 return NGX_DECLINED;
35 }
36
37 sub_reqid = (uint16_t) ntohs(*(const uint16_t *) (body + 2));
38 sub_dlen = (uint32_t) ntohl(*(const uint32_t *) (body + 20));
39
40 if (have == 24) {
41 if (sub_dlen == 0) {
42 return NGX_OK; /* nothing streams (e.g. truncate) */
43 }
44
45 switch (sub_reqid) {
46
47 case kXR_write:
48 case kXR_pgwrite:
49 if (sub_dlen > BRIX_MAX_WRITE_PAYLOAD) {
50 return NGX_DECLINED;
51 }
52 *extra = sub_dlen;
53 return NGX_OK;
54
55 case kXR_writev:
56 /* Descriptor block first; the segment data extends in stage 2
57 * once the descriptors can be summed. */
58 if (sub_dlen % BRIX_WRITEV_SEGSIZE != 0
59 || sub_dlen > BRIX_WRITEV_MAXSEGS * BRIX_WRITEV_SEGSIZE)
60 {
61 return NGX_DECLINED;
62 }
63 *extra = sub_dlen;
64 *final = 0;
65 return NGX_OK;
66
67 default:
68 /* Invalid embed (chkpoint-in-chkpoint, truncate with data, or
69 * garbage) — no extension; ckp_xeq rejects and drops. */
70 return NGX_DECLINED;
71 }
72 }
73
74 /* Stage 2: the embedded writev descriptor block is in at body[24..). */
75 if (sub_reqid == kXR_writev && have == 24 + sub_dlen) {
76 return brix_writev_body_extra(body + 24, sub_dlen, extra);
77 }
78
79 return NGX_DECLINED;
80}

82/* WHAT: Unwraps the embedded sub-request carried by a kXR_ckpXeq frame and
83 * routes it to the matching write/pgwrite/truncate/writev handler for
84 * the checkpointed handle idx.
85 * WHY: kXR_ckpXeq tunnels an ordinary write op "inside" an active checkpoint
86 * so the op can be rolled back. Requiring an active checkpoint
87 * (ckp_path) guards against executing a tentative op with no rollback
88 * anchor. Stock wire contract (do_ChkPntXeq): the frame's dlen is
89 * EXACTLY 24 — the embedded ClientRequestHdr — and the sub-request body
90 * (sub_dlen bytes per the embedded header; for writev, descriptors then
91 * data, see ckp_xeq_writev) streams after the frame. The recv framing
92 * (brix_ckpxeq_body_extra) has already appended those streamed bytes
93 * to ctx->recv.payload (ctx->recv.cur_body_extra of them), so the sub-request is
94 * contiguous here.
95 * HOW: Stock-parity validation, then dispatch: 1) embedded streamid must
96 * match the outer header's. 2) dlen must be 24. 3) An embedded chkpoint,
97 * or a truncate carrying data, is invalid. All three reject + drop the
98 * link exactly like stock (do_ChkPntXeq returns -1) — for 1 and 3 the
99 * trailing bytes may not have been read, so no resync is possible.
100 * 4) Per-type byte-count cross-check against cur_body_extra (defensive:
101 * detects a declined extension) — also error + drop. 5) Only then the
102 * checkpoint-active check (error, link kept: the sub-body is safely
103 * buffered by that point), and the switch to the sub-handler. */
104/* WHAT: Runs every stock-parity wire-shape check a kXR_ckpXeq frame must pass
105 * before its embedded op is dispatched — streamid match, dlen == 24,
106 * no embedded chkpoint / data-bearing truncate, and the per-type
107 * trailing-byte cross-check.
108 * WHY: These four checks contribute the bulk of ckp_xeq's branching; hoisting
109 * them keeps the dispatcher small and the ordering (all wire-shape checks
110 * before the checkpoint-active check) explicit and unchanged. Each
111 * violation emits the SAME error + counts the SAME op-error as before and
112 * signals the caller to drop the link.
113 * HOW: On the first failing check, send the matching kXR_ArgInvalid and return
114 * NGX_ERROR; return NGX_OK when the frame is well-formed. sub_reqid /
115 * sub_dlen have already been decoded by the caller. The streamid check
116 * runs in ckp_xeq itself so its original before-dlen-check ordering (and
117 * NULL/short-payload guard) is preserved verbatim. */
118static ngx_int_t
119ckp_xeq_validate_frame(brix_ctx_t *ctx, ngx_connection_t *c,
120 uint16_t sub_reqid, uint32_t sub_dlen)
121{
122 /* Stock parity: a chkpoint may not embed a chkpoint, and an embedded
123 * truncate carries no data (its length rides in the offset field). */
124 if (sub_reqid == kXR_chkpoint
125 || (sub_reqid == kXR_truncate && sub_dlen != 0))
126 {
127 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
128 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
129 "chkpoint request is invalid");
130 return NGX_ERROR;
131 }
132
133 /* Defensive byte-count cross-check: the recv framing extended the body
134 * by exactly sub_dlen (write/pgwrite data; writev descriptors). A
135 * shortfall means the extension was declined (oversized sub_dlen or a
136 * malformed descriptor block) — the trailing bytes were never read, so
137 * no resync is possible: error + drop. Runs BEFORE the checkpoint-
138 * active check so a framing violation always takes the link-drop path. */
139 if (((sub_reqid == kXR_write || sub_reqid == kXR_pgwrite)
140 && ctx->recv.cur_body_extra != sub_dlen)
141 || (sub_reqid == kXR_writev && ctx->recv.cur_body_extra < sub_dlen))
142 {
143 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
144 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
145 "ckpXeq: sub-request length invalid");
146 return NGX_ERROR;
147 }
148
149 return NGX_OK;
150}

152/* WHAT: Routes a validated ckpXeq sub-request to the matching write-family
153 * handler for the checkpointed handle idx.
154 * WHY: Isolating the type switch from the wire-shape validation keeps each
155 * function single-purpose and under the CCN cap; behaviour (handler
156 * selection, default reject) is identical to the inline switch it replaces.
157 * HOW: Dispatch on sub_reqid; the default path logs, counts an op-error, and
158 * sends kXR_ArgInvalid while keeping the link (stock second-pass default). */
159static ngx_int_t
160ckp_xeq_dispatch(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
161 uint16_t sub_reqid, const ckp_write_desc_t *d)
162{
163 switch (sub_reqid) {
164
165 case kXR_write:
166 return ckp_xeq_write(ctx, c, idx, d);
167
168 case kXR_pgwrite:
169 return ckp_xeq_pgwrite(ctx, c, idx, d);
170
171 case kXR_truncate:
172 return ckp_xeq_truncate(ctx, c, idx, d->sub_hdr);
173
174 case kXR_writev:
175 return ckp_xeq_writev(ctx, c, idx, d,
176 ctx->recv.cur_body_extra - d->sub_dlen);
177
178 default:
179 /* Stock parity: unknown embedded op — error, keep the link (stock's
180 * second-pass default does not drop). */
181 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
182 "brix: ckpXeq unsupported sub-reqid=%d",
183 (int) sub_reqid);
184 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
185 return brix_send_error(ctx, c, kXR_ArgInvalid,
186 "chkpoint request is invalid");
187 }
188}

190ngx_int_t
191ckp_xeq(brix_ctx_t *ctx, ngx_connection_t *c, int idx)
192{
193 brix_file_t *f = &ctx->files[idx];
194 ckp_write_desc_t d;
195 uint16_t sub_reqid;
196
197 /* Stock parity: the embedded request must carry the outer streamid. Kept
198 * ahead of the dlen check (and behind its own NULL/short-payload guard) so
199 * a streamid mismatch reports before a length mismatch, exactly as before. */
200 if (ctx->recv.payload != NULL && ctx->recv.cur_dlen >= 2
201 && (ctx->recv.payload[0] != ctx->recv.cur_streamid[0]
202 || ctx->recv.payload[1] != ctx->recv.cur_streamid[1]))
203 {
204 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
205 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
206 "Request streamid mismatch");
207 return NGX_ERROR;
208 }
209
210 /* Stock parity: the frame is EXACTLY the embedded 24-byte header — the
211 * sub-request body streams after it. This also rejects the legacy
212 * private layout (header + body counted inside dlen) the way a stock
213 * server does. Guards the payload deref for every check below. */
214 if (ctx->recv.cur_dlen != 24 || ctx->recv.payload == NULL) {
215 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
216 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
217 "Request length invalid");
218 return NGX_ERROR;
219 }
220
221 /* Wire layout as buffered by the recv framing:
222 * [0 .. 24) embedded ClientRequestHdr
223 * [24 .. 24+cur_body_extra) streamed sub-request body
224 * requestid at header offset 2, dlen at offset 20, both big-endian. */
225 d.sub_hdr = ctx->recv.payload;
226 sub_reqid = (uint16_t) ntohs(*(const uint16_t *) (d.sub_hdr + 2));
227 d.sub_dlen = (uint32_t) ntohl(*(const uint32_t *) (d.sub_hdr + 20));
228 d.sub_payload = ctx->recv.payload + 24;
229
230 if (ckp_xeq_validate_frame(ctx, c, sub_reqid, d.sub_dlen) != NGX_OK) {
231 return NGX_ERROR;
232 }
233
234 /* A ckpXeq with no open checkpoint has nothing to make tentative — reject
235 * so a write can never masquerade as checkpointed when it is not. Runs
236 * AFTER the wire-shape checks above so a framing violation always takes
237 * the link-drop path even when no checkpoint is open. */
238 if (f->ckp_path == NULL) {
239 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CHKPOINT, "CHKPOINT", f->path, "xeq",
240 kXR_InvalidRequest, "no active checkpoint");
241 }
242
243 return ckp_xeq_dispatch(ctx, c, idx, sub_reqid, &d);
244}
