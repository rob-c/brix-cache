8/* WHAT: Writes one checkpoint payload range through the VFS I/O core and reports
9 * the byte count, errno, and short-write state to the caller.
10 * WHY: ckpXeq sub-ops mutate the exported file, so they must share the same
11 * raw-I/O chokepoint as normal write/pgwrite/writev handlers.
12 * HOW: Initialize a WRITE job, execute it inline, then normalize hard errors
13 * and partial writes into a single NGX_ERROR result. */
14
15static ngx_int_t
16ckp_xeq_vfs_write_full(ngx_fd_t fd, const u_char *data, size_t len,
17 off_t offset, ssize_t *written, int *io_errno, ngx_flag_t *short_io)
18{
19 brix_vfs_job_t job;
20
21 brix_vfs_job_write_init(&job, fd, offset, data, len);
22 brix_vfs_io_execute(&job);
23
24 if (written != NULL) {
25 *written = job.nio;
26 }
27 if (io_errno != NULL) {
28 *io_errno = 0;
29 }
30 if (short_io != NULL) {
31 *short_io = 0;
32 }
33
34 if (job.io_errno != 0 || job.nio < 0) {
35 if (io_errno != NULL) {
36 *io_errno = job.io_errno != 0 ? job.io_errno : EIO;
37 }
38 if (short_io != NULL && job.short_io) {
39 *short_io = 1;
40 }
41 return NGX_ERROR;
42 }
43
44 if ((size_t) job.nio < len) {
45 if (io_errno != NULL) {
46 *io_errno = EIO;
47 }
48 if (short_io != NULL) {
49 *short_io = 1;
50 }
51 return NGX_ERROR;
52 }
53
54 return NGX_OK;
55}

57/* WHAT: Performs a single VFS-core write to the checkpointed file at the specified offset. Validates handle matches idx, writes sub_dlen bytes,
58 * updates byte counters and sends success response. Handles zero-length payloads (no-op).
59 * WHY: ckpXeq write is the atomic transactional write operation — when executed under an active checkpoint, this write becomes "tentative" until commit/rollback.
60 * The handle mismatch check prevents cross-file corruption during checkpointed operations.
61 * HOW: 1) Validate fhandle[0] == idx. 2) Handle zero-length payload as no-op. 3) Run a VFS WRITE job. 4) Update counters + send_ok. */

63static ngx_int_t
64ckp_xeq_write(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
65 const u_char *sub_hdr, const u_char *sub_payload, uint32_t sub_dlen)
66{

68 int64_t offset;
69 ssize_t nw;

71 int err;
72 ngx_flag_t short_io;
73
74 xrdw_write_req_unpack(((ClientRequestHdr *) sub_hdr)->body, &wreq);
75

77 * was opened on (idx). Refusing a mismatch keeps a checkpointed op from
78 * silently touching a different file than the one being protected. */

81 "ckpXeq write: handle mismatch");
82 }
83

85 if (sub_dlen == 0) {

87 }

91 &short_io)
92 != NGX_OK)
93 {
94 const char *msg = short_io ? "short write (disk full?)"
95 : strerror(err);
96
97 snprintf(detail, sizeof(detail), "xeq_write %lld+%u",
98 (long long) offset, (unsigned) sub_dlen);
99 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CHKPOINT, "CHKPOINT",
100 ctx->files[idx].path, detail, kXR_IOError, msg);
101 }
102
103 snprintf(detail, sizeof(detail), "xeq_write %lld+%u",
104 (long long) offset, (unsigned) sub_dlen);
105
106 /* Integrity: the write went straight through the VFS I/O core, bypassing the
107 * normal write path's per-block CRC fold. Fold the bytes into the handle's
108 * csi engine so it marks the touched extent dirty — otherwise a read of the
109 * just-written region on this handle verifies the new data against the stale
110 * pre-write on-disk CRC and fails with EIO (kXR_IOError). */
111 if (ctx->files[idx].csi != NULL && nw > 0) {
112 (void) brix_csi_write_update((brix_csi_t *) ctx->files[idx].csi,
113 sub_payload, (off_t) offset, (size_t) nw);
114 }
115
116 ctx->files[idx].bytes_written += (size_t) nw;
117 ctx->totals.bytes_written += (size_t) nw;
118 return brix_send_ok(ctx, c, NULL, 0);

121/* WHAT: Performs a page-write under checkpoint protection, decoding the pgwrite payload (per-page CRC32c framing) into flat data,
122 * then writing in XRD_PGWRITE_PAGESZ-sized chunks. Validates checksum integrity before committing writes.
123 * WHY: kXR_pgwrite uses per-page CRC32c for data integrity verification — critical for large file transfers where partial corruption must be detected.
124 * Under ckpXeq, the pgwrite becomes tentative until commit; checksum mismatch rejection prevents corrupted data from entering the checkpoint.
125 * HOW: 1) Validate handle + offset + payload size (> XRD_PGWRITE_CKSZ). 2) Decode payload via brix_pgwrite_decode_payload (CRC32c verification). 3) Write in PAGE-sized chunks through the VFS core. 4) Send pgwrite_status with final offset. */

127static ngx_int_t
128ckp_xeq_pgwrite(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
129 const u_char *sub_hdr, const u_char *sub_payload, uint32_t sub_dlen)
130{
131 xrdw_pgwrite_req_t preq;
132 int64_t offset;
133 u_char *flat;
134 const u_char *src;
135 size_t rem, page_data;
136 size_t flat_sz;
137 int64_t bad_offset;
138 int64_t write_offset;
139 size_t total_written = 0;
140 char detail[64];
141 ngx_int_t rc;
142
143 xrdw_pgwrite_req_unpack(((ClientRequestHdr *) sub_hdr)->body, &preq);
144 if ((int)(unsigned char) preq.fhandle[0] != idx) {
145 return brix_send_error(ctx, c, kXR_InvalidRequest,
146 "ckpXeq pgwrite: handle mismatch");
147 }
148
149 /* Payload must hold at least one 4-byte CRC32c plus some data, so it has
150 * to exceed XRD_PGWRITE_CKSZ; a negative offset is never valid. */
151 offset = preq.offset;
152 if (offset < 0 || sub_dlen <= XRD_PGWRITE_CKSZ) {
153 snprintf(detail, sizeof(detail), "xeq_pgwrite %lld+%u",
154 (long long) offset, (unsigned) sub_dlen);
155 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CHKPOINT, "CHKPOINT",
156 ctx->files[idx].path, detail,
157 kXR_ArgInvalid, "invalid pgwrite payload");
158 }
159
160 /* Worst case the decoded (CRC-stripped) data is smaller than sub_dlen, so
161 * sub_dlen is a safe upper bound for the flat buffer. Freed on every exit
162 * path below (heap, not pool — must not leak on error returns). */
163 flat = ngx_alloc((size_t) sub_dlen, c->log);
164 if (flat == NULL) {
165 return NGX_ERROR;
166 }
167
168 /* Verify every per-page CRC32c and copy the page bodies into flat[].
169 * On checksum/format failure bad_offset is set to the offending file
170 * offset so the error/log line can pinpoint the bad page. flat_sz returns
171 * the contiguous byte count actually decoded. */

177 * any other non-OK is a malformed/short payload (-> kXR_ArgInvalid). */

179 : kXR_ArgInvalid;

182

184 (long long) bad_offset, (unsigned) sub_dlen);
185 brix_log_access(ctx, c, "CHKPOINT", ctx->files[idx].path, detail,
186 0, status, msg, 0);
187 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
188 ngx_free(flat);
189 return brix_send_error(ctx, c, status, msg);
190 }

192 /* Drain the verified data to disk one page at a time, advancing the file
193 * offset, source pointer, and remaining counter in lockstep. Writing in
194 * page units keeps offsets page-aligned and bounds the amount that can be
195 * lost/retried on a short write. */
196 write_offset = offset;

210 != NGX_OK)
211 {
212 const char *msg = short_io ? "short write (disk full?)"
213 : strerror(err);

216 (long long) offset, total_written);
217 brix_log_access(ctx, c, "CHKPOINT", ctx->files[idx].path, detail,
218 0, kXR_IOError, msg, 0);
219 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
220 ngx_free(flat);
221 return brix_send_error(ctx, c, kXR_IOError, msg);
222 }
223
224 total_written += (size_t) nw;
225 write_offset += (int64_t) nw;
226 src += page_data;
227 rem -= page_data;
228 }
229
230 /* Integrity: same as ckp_xeq_write — the pages went straight through the
231 * VFS I/O core, so fold the decoded plaintext into the csi engine to mark
232 * the written extent dirty. Without this a read-back on this handle verifies
233 * the new pages against the stale on-disk CRC and fails with EIO. */
234 if (ctx->files[idx].csi != NULL && flat_sz > 0) {
235 (void) brix_csi_write_update((brix_csi_t *) ctx->files[idx].csi,
236 flat, (off_t) offset, flat_sz);
237 }
238
239 ctx->files[idx].bytes_written += total_written;
240 ctx->totals.bytes_written += total_written;
241
242 snprintf(detail, sizeof(detail), "xeq_pgwrite %lld+%zu",
243 (long long) offset, total_written);
244 brix_log_access(ctx, c, "CHKPOINT", ctx->files[idx].path, detail,
245 1, kXR_ok, NULL, total_written);
246
247 ngx_free(flat);
248
249 return brix_send_pgwrite_status(ctx, c, write_offset);
250}

286/* WHAT: Executes an embedded kXR_writev under checkpoint protection: validates
287 * the descriptor block under the STOCK wire contract (sub_dlen frames
288 * ONLY the N*16-byte write_list descriptors; data_len bytes of segment
289 * data were streamed after them), requires every data-bearing segment to
290 * target the checkpointed handle idx, then writes each segment through
291 * the VFS core.
292 * WHY: kXR_ckpXeq tunnels the SAME wire layout as a standalone kXR_writev
293 * (see brix_handle_writev in writev.c): the embedded header's dlen
294 * counts descriptors only and the data rides behind — stock do_ChkPntXeq
295 * fetches the descriptor block, then hands off to do_WriteV, which
296 * streams the data. Multi-file vectors are refused (stock: "multi-file
297 * chkpoint writev not supported") because the rollback anchor covers
298 * exactly one file.
299 * HOW: 1) Validate sub_dlen non-zero, 16-aligned, <= BRIX_WRITEV_MAXSEGS
300 * segments — violations reject + drop the link like stock (once the
301 * descriptor framing is in doubt the trailing byte count is unknowable).
302 * 2) Sum wlen; require data-bearing segments to name idx and the sum to
303 * equal data_len (defensive: the recv framing extended by exactly that).
304 * 3) One VFS WRITE job per non-empty segment. */
305
306static ngx_int_t
307ckp_xeq_writev(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
308 const u_char *sub_payload, uint32_t sub_dlen, uint32_t data_len)
309{
310 write_list *wl;
311 size_t n_segs, i;
312 uint64_t total_wlen;
313 const u_char *data_ptr;
314 size_t bytes_written_total = 0;
315
316 /* Stock parity (do_WriteV, reached via do_ChkPntXeq): a sub_dlen that is
317 * zero or not a whole number of descriptors is invalid — error + drop. */
318 if (sub_dlen == 0 || sub_dlen % BRIX_WRITEV_SEGSIZE != 0) {
319 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
320 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
321 "Write vector is invalid");
322 return NGX_ERROR;
323 }
324
325 wl = (write_list *) sub_payload;
326 n_segs = sub_dlen / BRIX_WRITEV_SEGSIZE;
327
328 if (n_segs > BRIX_WRITEV_MAXSEGS) {
329 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
330 (void) brix_send_error(ctx, c, kXR_ArgTooLong,
331 "Write vector is too long");
332 return NGX_ERROR;
333 }
334
335 total_wlen = 0;
336 for (i = 0; i < n_segs; i++) {
337 uint32_t wlen = (uint32_t) ntohl((uint32_t) wl[i].wlen);
338
339 /* Every data-bearing segment must target the checkpointed handle so
340 * a checkpoint can never scatter writes across other files (stock
341 * ignores zero-length segments when it collects the vector, then
342 * refuses a multi-file result). */
343 if (wlen > 0 && (int)(unsigned char) wl[i].fhandle[0] != idx) {
344 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
345 (void) brix_send_error(ctx, c, kXR_Unsupported,
346 "multi-file chkpoint writev not supported");
347 return NGX_ERROR;
348 }
349 total_wlen += wlen;
350 }
351
352 /* Defensive: the recv framing extended the body by exactly sum(wlen); a
353 * mismatch means the extension never ran for this frame — the trailing
354 * data was never read, so the link cannot be resynchronised. */
355 if (total_wlen != (uint64_t) data_len) {
356 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
357 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
358 "ckpXeq writev: payload size mismatch");
359 return NGX_ERROR;
360 }
361
362 /* Data blocks begin right after the N descriptors. */
363 data_ptr = sub_payload + (size_t) sub_dlen;
364
365 for (i = 0; i < n_segs; i++) {
366 int64_t offset = (int64_t) be64toh((uint64_t) wl[i].offset);
367 uint32_t wlen = (uint32_t) ntohl((uint32_t) wl[i].wlen);
368 ssize_t nw;
369 int err;
370 ngx_flag_t short_io;
371
372 if (wlen == 0) {
373 continue;
374 }
375
376 if (ckp_xeq_vfs_write_full(ctx->files[idx].fd, data_ptr,
377 (size_t) wlen, (off_t) offset, &nw, &err,
378 &short_io)
379 != NGX_OK)
380 {
381 const char *msg = short_io ? "short write (disk full?)"
382 : strerror(err);
383
384 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CHKPOINT, "CHKPOINT",
385 ctx->files[idx].path, "xeq_writev",
386 kXR_IOError, msg);
387 }
388 ctx->files[idx].bytes_written += (size_t) nw;
389 ctx->totals.bytes_written += (size_t) nw;
390 bytes_written_total += (size_t) nw;
391 data_ptr += wlen;
392 }
393
394 brix_log_access(ctx, c, "CHKPOINT", ctx->files[idx].path,
395 "xeq_writev", 1, kXR_ok, NULL, bytes_written_total);
396 return brix_send_ok(ctx, c, NULL, 0);
397}

399/* brix_ckpxeq_body_extra — trailing sub-body length for kXR_ckpXeq
400 * WHAT: Computes how many more bytes the client will stream after the ckpXeq
401 * frame, staged on `have` (body bytes received so far). have == 24: the
402 * embedded sub-header just landed — the sub-request's own dlen bytes follow
403 * (write/pgwrite data, or the writev descriptor block). have == 24 +
404 * sub_dlen: an embedded writev's descriptors landed — sum(wlen) data bytes
405 * follow, delegated to brix_writev_body_extra (the SAME contract as a
406 * standalone kXR_writev). *final = 0 tells the recv framing to call again
407 * after the current extension completes.
408 * WHY: Stock wire contract (do_ChkPntXeq + XrdCl MessageUtils): the outer
409 * chkpoint dlen frames ONLY the embedded 24-byte header; everything else is
410 * raw body streamed behind the frame. Shared by the recv framing so the
411 * whole sub-request lands contiguously in payload_buf before dispatch.
412 * HOW: Bound every stage by BRIX_MAX_WRITE_PAYLOAD (and the writev vector
413 * cap) and return NGX_DECLINED on any violation — the framing then does NOT
414 * extend, the auth gates still run, and ckp_xeq detects the un-read trailing
415 * bytes (count mismatch) and drops the link. */
416ngx_int_t
417brix_ckpxeq_body_extra(const u_char *body, uint32_t have,
418 uint32_t *extra, unsigned *final)
419{
420 uint16_t sub_reqid;
421 uint32_t sub_dlen;
422
423 *extra = 0;
424 *final = 1;
425
426 if (body == NULL || have < 24) {
427 return NGX_DECLINED;
428 }
429
430 sub_reqid = (uint16_t) ntohs(*(const uint16_t *) (body + 2));
431 sub_dlen = (uint32_t) ntohl(*(const uint32_t *) (body + 20));
432
433 if (have == 24) {
434 if (sub_dlen == 0) {
435 return NGX_OK; /* nothing streams (e.g. truncate) */
436 }
437
438 switch (sub_reqid) {
439
440 case kXR_write:
441 case kXR_pgwrite:
442 if (sub_dlen > BRIX_MAX_WRITE_PAYLOAD) {
443 return NGX_DECLINED;
444 }
445 *extra = sub_dlen;
446 return NGX_OK;
447
448 case kXR_writev:
449 /* Descriptor block first; the segment data extends in stage 2
450 * once the descriptors can be summed. */
451 if (sub_dlen % BRIX_WRITEV_SEGSIZE != 0
452 || sub_dlen > BRIX_WRITEV_MAXSEGS * BRIX_WRITEV_SEGSIZE)
453 {
454 return NGX_DECLINED;
455 }
456 *extra = sub_dlen;
457 *final = 0;
458 return NGX_OK;
459
460 default:
461 /* Invalid embed (chkpoint-in-chkpoint, truncate with data, or
462 * garbage) — no extension; ckp_xeq rejects and drops. */
463 return NGX_DECLINED;
464 }
465 }
466
467 /* Stage 2: the embedded writev descriptor block is in at body[24..). */
468 if (sub_reqid == kXR_writev && have == 24 + sub_dlen) {
469 return brix_writev_body_extra(body + 24, sub_dlen, extra);
470 }
471
472 return NGX_DECLINED;
473}

475/* WHAT: Unwraps the embedded sub-request carried by a kXR_ckpXeq frame and
476 * routes it to the matching write/pgwrite/truncate/writev handler for
477 * the checkpointed handle idx.
478 * WHY: kXR_ckpXeq tunnels an ordinary write op "inside" an active checkpoint
479 * so the op can be rolled back. Requiring an active checkpoint
480 * (ckp_path) guards against executing a tentative op with no rollback
481 * anchor. Stock wire contract (do_ChkPntXeq): the frame's dlen is
482 * EXACTLY 24 — the embedded ClientRequestHdr — and the sub-request body
483 * (sub_dlen bytes per the embedded header; for writev, descriptors then
484 * data, see ckp_xeq_writev) streams after the frame. The recv framing
485 * (brix_ckpxeq_body_extra) has already appended those streamed bytes
486 * to ctx->recv.payload (ctx->recv.cur_body_extra of them), so the sub-request is
487 * contiguous here.
488 * HOW: Stock-parity validation, then dispatch: 1) embedded streamid must
489 * match the outer header's. 2) dlen must be 24. 3) An embedded chkpoint,
490 * or a truncate carrying data, is invalid. All three reject + drop the
491 * link exactly like stock (do_ChkPntXeq returns -1) — for 1 and 3 the
492 * trailing bytes may not have been read, so no resync is possible.
493 * 4) Per-type byte-count cross-check against cur_body_extra (defensive:
494 * detects a declined extension) — also error + drop. 5) Only then the
495 * checkpoint-active check (error, link kept: the sub-body is safely
496 * buffered by that point), and the switch to the sub-handler. */
497ngx_int_t
498ckp_xeq(brix_ctx_t *ctx, ngx_connection_t *c, int idx)
499{
500 brix_file_t *f = &ctx->files[idx];
501 const u_char *sub_hdr;
502 const u_char *sub_payload;
503 uint32_t sub_dlen;
504 uint16_t sub_reqid;
505
506 /* Stock parity: the embedded request must carry the outer streamid. */
507 if (ctx->recv.payload != NULL && ctx->recv.cur_dlen >= 2
508 && (ctx->recv.payload[0] != ctx->recv.cur_streamid[0]
509 || ctx->recv.payload[1] != ctx->recv.cur_streamid[1]))
510 {
511 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
512 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
513 "Request streamid mismatch");
514 return NGX_ERROR;
515 }
516
517 /* Stock parity: the frame is EXACTLY the embedded 24-byte header — the
518 * sub-request body streams after it. This also rejects the legacy
519 * private layout (header + body counted inside dlen) the way a stock
520 * server does. */
521 if (ctx->recv.cur_dlen != 24 || ctx->recv.payload == NULL) {
522 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
523 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
524 "Request length invalid");
525 return NGX_ERROR;
526 }
527
528 /* Wire layout as buffered by the recv framing:
529 * [0 .. 24) embedded ClientRequestHdr
530 * [24 .. 24+cur_body_extra) streamed sub-request body
531 * requestid at header offset 2, dlen at offset 20, both big-endian. */
532 sub_hdr = ctx->recv.payload;
533 sub_reqid = (uint16_t) ntohs(*(const uint16_t *) (sub_hdr + 2));
534 sub_dlen = (uint32_t) ntohl(*(const uint32_t *) (sub_hdr + 20));
535 sub_payload = ctx->recv.payload + 24;
536
537 /* Stock parity: a chkpoint may not embed a chkpoint, and an embedded
538 * truncate carries no data (its length rides in the offset field). */
539 if (sub_reqid == kXR_chkpoint
540 || (sub_reqid == kXR_truncate && sub_dlen != 0))
541 {
542 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
543 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
544 "chkpoint request is invalid");
545 return NGX_ERROR;
546 }
547
548 /* Defensive byte-count cross-check: the recv framing extended the body
549 * by exactly sub_dlen (write/pgwrite data; writev descriptors). A
550 * shortfall means the extension was declined (oversized sub_dlen or a
551 * malformed descriptor block) — the trailing bytes were never read, so
552 * no resync is possible: error + drop. Runs BEFORE the checkpoint-
553 * active check so a framing violation always takes the link-drop path. */
554 if (((sub_reqid == kXR_write || sub_reqid == kXR_pgwrite)
555 && ctx->recv.cur_body_extra != sub_dlen)
556 || (sub_reqid == kXR_writev && ctx->recv.cur_body_extra < sub_dlen))
557 {
558 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
559 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
560 "ckpXeq: sub-request length invalid");
561 return NGX_ERROR;
562 }
563
564 /* A ckpXeq with no open checkpoint has nothing to make tentative — reject
565 * so a write can never masquerade as checkpointed when it is not. Runs
566 * AFTER the wire-shape checks above so a framing violation always takes
567 * the link-drop path even when no checkpoint is open. */
568 if (f->ckp_path == NULL) {
569 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CHKPOINT, "CHKPOINT", f->path, "xeq",
570 kXR_InvalidRequest, "no active checkpoint");
571 }
572
573 switch (sub_reqid) {
574
575 case kXR_write:
576 return ckp_xeq_write(ctx, c, idx, sub_hdr, sub_payload, sub_dlen);
577
578 case kXR_pgwrite:
579 return ckp_xeq_pgwrite(ctx, c, idx, sub_hdr, sub_payload, sub_dlen);
580
581 case kXR_truncate:
582 return ckp_xeq_truncate(ctx, c, idx, sub_hdr);
583
584 case kXR_writev:
585 return ckp_xeq_writev(ctx, c, idx, sub_payload, sub_dlen,
586 ctx->recv.cur_body_extra - sub_dlen);
587
588 default:
589 /* Stock parity: unknown embedded op — error, keep the link (stock's
590 * second-pass default does not drop). */
591 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
592 "brix: ckpXeq unsupported sub-reqid=%d",
593 (int) sub_reqid);
594 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
595 return brix_send_error(ctx, c, kXR_ArgInvalid,
596 "chkpoint request is invalid");
597 }
598}
