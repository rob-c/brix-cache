19ssize_t
20json_get_string(const char *json, size_t json_len, const char *key,
21 char *out, size_t out_max)
22/* WHAT: Extract a single string value from JSON by key into caller-supplied output buffer with bounds checking.
23 * WHY: Used to extract JWT claims (sub, iss, aud), JWKS fields (kid, kty, n, e), and OIDC metadata strings
24 * without exposing callers to json_t type handling or reference counting.
25 * HOW: Validate out_max>0 and all pointers non-null; json_loadb() parse buffer into json_t root; check root is object;
26 * json_object_get(root,key); verify value is string; extract length via json_string_length(); copy min(length, out_max-1) bytes
27 * to out with null terminator; json_decref(root); return copy_len or -1 on any failure. */
28{
29 json_error_t err;
30 json_t *root;
31 json_t *value;
32 const char *str;
33 size_t copy_len;
34
35 if (out_max == 0 || json == NULL || key == NULL || out == NULL) {
36 return -1;
37 }
38
39 root = json_loadb(json, json_len, JSON_DECODE_INT_AS_REAL, &err);
40 if (root == NULL || !json_is_object(root)) {
41 if (root != NULL) {
42 json_decref(root);
43 }
44 return -1;
45 }
46
47 value = json_object_get(root, key);
48 if (!json_is_string(value)) {
49 json_decref(root);
50 return -1;
51 }
52
53 str = json_string_value(value);
54 copy_len = json_string_length(value);
55 if (copy_len >= out_max) {
56 copy_len = out_max - 1;
57 }
58
59 memcpy(out, str, copy_len);
60 out[copy_len] = '\0';
61 json_decref(root);
62
63 return (ssize_t) copy_len;
64}

67json_get_string_array(const char *json, size_t json_len, const char *key,
68 char out[][256], int max_items)
69/* WHAT: Extract an array of string values from JSON by key into caller-supplied fixed-size output array.
70 * WHY: Used to extract JWT claim arrays (aud), JWKS fields, and OIDC metadata lists where multiple string
71 * values need to be collected safely with per-item 256-byte truncation bounds.
72 * HOW: Validate max_items>0 and all pointers non-null; json_loadb() parse buffer into json_t root; check root is object;
73 * json_object_get(root,key); verify array type; json_array_foreach() iterate each item, skip non-strings; extract string length;
74 * copy min(length,255) bytes to out[count] with null terminator; increment count; break at max_items limit; json_decref(root); return count. */
75{
76 json_error_t err;
77 json_t *root;
78 json_t *array;
79 json_t *item;
80 size_t index;
81 int count;
82
83 if (max_items <= 0 || json == NULL || key == NULL || out == NULL) {
84 return 0;
85 }
86
87 root = json_loadb(json, json_len, JSON_DECODE_INT_AS_REAL, &err);
88 if (root == NULL || !json_is_object(root)) {
89 if (root != NULL) {
90 json_decref(root);
91 }
92 return 0;
93 }
94
95 array = json_object_get(root, key);
96 if (!json_is_array(array)) {
97 json_decref(root);
98 return 0;
99 }
100
101 count = 0;
102 json_array_foreach(array, index, item) {
103 const char *str;
104 size_t copy_len;
105
106 if (!json_is_string(item)) {
107 continue;
108 }
109
110 str = json_string_value(item);
111 copy_len = json_string_length(item);
112 if (copy_len > 255) {
113 copy_len = 255;
114 }
115
116 memcpy(out[count], str, copy_len);
117 out[count][copy_len] = '\0';
118 count++;
119 if (count >= max_items) {
120 break;
121 }
122 }
123
124 json_decref(root);
125 return count;
126}

128int
129json_get_int64(const char *json, size_t json_len, const char *key,
130 int64_t *out)
131/* WHAT: Extract a single numeric value from JSON by key into caller-supplied int64 output pointer.
132 * WHY: Used to extract JWT numeric claims (exp, iat, nbf). RFC 7519 §2 permits
133 * NumericDate to be fractional (float); jansson would fail to parse integers that
134 * overflow int64 (e.g. far-future exp = 99999999999999999999) unless given the
135 * JSON_DECODE_INT_AS_REAL flag, which causes all numbers to be represented as
136 * json_real so that over-range integers do not abort the document parse.
137 * HOW: Validate all pointers non-null; json_loadb() with JSON_DECODE_INT_AS_REAL
138 * so oversized integers are parsed as reals; json_object_get(root,key); accept
139 * json_integer (exact) OR json_real (truncate fractional, clamp huge positive
140 * doubles that would overflow int64_t on the cast); return 0 on success, -1 on
141 * any failure. */
142{
143 json_error_t err;
144 json_t *root;
145 json_t *value;
146 double d;
147
148 if (json == NULL || key == NULL || out == NULL) {
149 return -1;
150 }
151
152 /*
153 * JSON_DECODE_INT_AS_REAL: parse all JSON numbers as json_real so that
154 * over-range integers (> INT64_MAX) do not cause a document parse error.
155 * This allows "exp":99999999999999999999 (far-future timestamp) to be read
156 * and accepted rather than failing the whole payload decode.
157 */
158 root = json_loadb(json, json_len, JSON_DECODE_INT_AS_REAL, &err);
159 if (root == NULL || !json_is_object(root)) {
160 if (root != NULL) {
161 json_decref(root);
162 }
163 return -1;
164 }
165
166 value = json_object_get(root, key);
167 if (json_is_integer(value)) {
168 /* Exact integer path (kept for completeness; JSON_DECODE_INT_AS_REAL
169 * means this branch will not fire in practice, but is safe if the
170 * flag behaviour ever changes). */
171 *out = (int64_t) json_integer_value(value);
172 } else if (json_is_real(value)) {
173 d = json_real_value(value);
174 /*
175 * Truncate fractional seconds as RFC 7519 §2 requires.
176 * Guard against the cast UB that occurs when d > INT64_MAX: on x86-64
177 * cvttsd2si yields INT64_MIN for out-of-range inputs, which the
178 * caller's "exp > 0" check would then incorrectly treat as expired.
179 * Clamp to INT64_MAX so far-future tokens are accepted.
180 * 9223372036854775808.0 is 2^63, the first double above INT64_MAX.
181 */
182 if (d >= 9223372036854775808.0) {
183 *out = INT64_MAX;
184 } else {
185 *out = (int64_t) d;
186 }
187 } else {
188 json_decref(root);
189 return -1;
190 }
191 json_decref(root);
192 return 0;
193}

195int
196json_string_or_array_contains(const char *json, size_t json_len,
197 const char *key, const char *needle)
198/* WHAT: Membership test for a JWT claim that may be a string or an array of
199 * strings (RFC 7519 §4.1.3 "aud"). Returns 1 if json[key] is a string equal to
200 * needle, or an array containing a string equal to needle; 0 otherwise.
201 * WHY: json_get_string() rejects arrays, so using it for the audience check
202 * silently false-denies tokens whose aud is the (spec-legal, common) array form.
203 * HOW: json_loadb() parse; json_object_get(key); if string, strcmp; if array,
204 * json_array_foreach + strcmp each string item; json_decref; return match. */
205{
206 json_error_t err;
207 json_t *root;
208 json_t *value;
209 json_t *item;
210 size_t index;
211 int found = 0;
212
213 if (json == NULL || key == NULL || needle == NULL) {
214 return 0;
215 }
216
217 root = json_loadb(json, json_len, JSON_DECODE_INT_AS_REAL, &err);
218 if (root == NULL || !json_is_object(root)) {
219 if (root != NULL) {
220 json_decref(root);
221 }
222 return 0;
223 }
224
225 value = json_object_get(root, key);
226 if (json_is_string(value)) {
227 found = (strcmp(json_string_value(value), needle) == 0);
228 } else if (json_is_array(value)) {
229 json_array_foreach(value, index, item) {
230 if (json_is_string(item)
231 && strcmp(json_string_value(item), needle) == 0)
232 {
233 found = 1;
234 break;
235 }
236 }
237 }
238
239 json_decref(root);
240 return found;
241}

243int
244json_has_member(const char *json, size_t json_len, const char *key)
245/* WHAT: Returns 1 if the JSON object has a member named `key`, else 0.
246 * WHY: Used by the JWS `crit` check (RFC 7515 §4.1.11): any token whose
247 * header contains a `crit` member MUST be rejected by a processor that does
248 * not understand the listed extension parameters. This helper avoids exposing
249 * callers to jansson reference-counting or type-checking detail.
250 * HOW: json_loadb() parse buffer into json_t root; check root is object;
251 * json_object_get(root, key) != NULL → member present; json_decref(root);
252 * return 1 (found) or 0 (missing/parse failure). */
253{
254 json_error_t err;
255 json_t *root;
256 int found;
257
258 if (json == NULL || key == NULL) {
259 return 0;
260 }
261
262 root = json_loadb(json, json_len, JSON_DECODE_INT_AS_REAL, &err);
263 if (root == NULL || !json_is_object(root)) {
264 if (root != NULL) {
265 json_decref(root);
266 }
267 return 0;
268 }
269
270 found = (json_object_get(root, key) != NULL) ? 1 : 0;
271 json_decref(root);
272 return found;
273}
