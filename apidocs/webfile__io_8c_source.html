29int
30web_connect(brix_webfile *wf, brix_status *st)
31{
32 memset(&wf->io, 0, sizeof(wf->io));
33 wf->io.fd = brix_tcp_connect(wf->host, wf->port, wf->timeout_ms, st);
34 if (wf->io.fd < 0) {
35 return -1;
36 }
37 wf->io.timeout_ms = wf->timeout_ms;
38 if (wf->tls && brix_tls_client(&wf->io, wf->host, wf->verify, wf->verify,
39 wf->ca_dir[0] ? wf->ca_dir : NULL,
40 &wf->tls_ctx, st) != 0) {
41 close(wf->io.fd);
42 wf->io.fd = -1;
43 return -1;
44 }
45 wf->connected = 1;

50/* Read up to n bytes (branches on TLS). >0 bytes, 0 EOF, -1 error. */
51ssize_t
52web_read_some(brix_webfile *wf, void *buf, size_t n, brix_status *st)
53{
54 if (wf->io.ssl != NULL) {
55 size_t got = 0;
56 if (brix_tls_read_some(&wf->io, buf, n, &got, st) != 0) {
57 return -1;
58 }
59 return (ssize_t) got;
60 }
61 struct pollfd pfd;
62 ssize_t r;
63 int pr;
64 pfd.fd = wf->io.fd; pfd.events = POLLIN; pfd.revents = 0;
65 do { pr = poll(&pfd, 1, wf->io.timeout_ms); } while (pr < 0 && errno == EINTR);
66 if (pr <= 0) {
67 brix_status_set(st, XRDC_ESOCK, pr == 0 ? ETIMEDOUT : errno, "web read");
68 return -1;
69 }
70 do { r = read(wf->io.fd, buf, n); } while (r < 0 && errno == EINTR);
71 if (r < 0) {
72 brix_status_set(st, XRDC_ESOCK, errno, "web read: %s", strerror(errno));
73 return -1;
74 }
75 return r;
76}

80long long
81hdr_clen(const char *hdrs)
82{
83 const char *p = hdrs;
84 while ((p = strchr(p, '\n')) != NULL) {
85 p++;
86 if (strncasecmp(p, "Content-Length:", 15) == 0) {
87 return strtoll(p + 15, NULL, 10);
88 }
89 }
90 if (strncasecmp(hdrs, "Content-Length:", 15) == 0) {
91 return strtoll(hdrs + 15, NULL, 10);
92 }

97/* One ranged GET over the (already-connected) persistent socket. Fills up to len
98 * bytes at off into buf; returns bytes read (0 = EOF/416), or -1 (st set). On any
99 * transport/protocol fault returns -1 AND disconnects so the caller can retry. */
100ssize_t
101web_get_range(brix_webfile *wf, int64_t off, void *buf, size_t len,
102 brix_status *st)
103{
104 char req[3200];
105 char hbuf[WEB_HDR_MAX + 1];
106 size_t hlen = 0;
107 char *eoh = NULL;
108 int status = 0;
109 long long clen;
110 int rn = snprintf(req, sizeof(req),
111 "GET %s HTTP/1.1\r\nHost: %s\r\nUser-Agent: xrootdfs\r\n"
112 "Accept: */*\r\nConnection: keep-alive\r\n"
113 "Range: bytes=%lld-%lld\r\n%s\r\n",
114 wf->path[0] ? wf->path : "/", wf->hostport,
115 (long long) off, (long long) (off + (int64_t) len - 1),
116 wf->auth);
117 if (rn < 0 || (size_t) rn >= sizeof(req)) {
118 brix_status_set(st, XRDC_EUSAGE, 0, "web GET: request too long");
119 return -1;
120 }
121 if (brix_write_full(&wf->io, req, (size_t) rn, st) != 0) {
122 web_disconnect(wf);
123 return -1;
124 }
125
126 /* Read response headers (up to the CRLFCRLF), keeping any body overflow. */
127 for (;;) {
128 ssize_t r;
129 if (hlen >= WEB_HDR_MAX) {
130 web_disconnect(wf);
131 brix_status_set(st, XRDC_EPROTO, 0, "web GET: header too large");
132 return -1;
133 }
134 r = web_read_some(wf, hbuf + hlen, WEB_HDR_MAX - hlen, st);
135 if (r <= 0) {
136 web_disconnect(wf);
137 if (r == 0) {
138 brix_status_set(st, XRDC_ESOCK, 0, "web GET: peer closed");
139 }
140 return -1;
141 }
142 hlen += (size_t) r;
143 hbuf[hlen] = '\0';
144 eoh = strstr(hbuf, "\r\n\r\n");
145 if (eoh != NULL) {
146 break;
147 }
148 }
149 if (strncmp(hbuf, "HTTP/", 5) == 0) {
150 const char *sp = strchr(hbuf, ' ');
151 status = sp ? atoi(sp + 1) : 0;
152 }
153 if (status == 416) { /* range past EOF → no bytes */
154 return 0;
155 }
156 if (status != 206 && status != 200) {
157 web_disconnect(wf);
158 if (status == 404) {
159 brix_status_set(st, kXR_NotFound, 0, "not found");
160 } else if (status == 401 || status == 403) {
161 brix_status_set(st, kXR_NotAuthorized, 0, "HTTP %d", status);
162 } else {
163 brix_status_set(st, XRDC_EPROTO, 0, "web GET: HTTP %d", status);
164 }
165 return -1;
166 }
167
168 *eoh = '\0';
169 clen = hdr_clen(hbuf);
170 if (clen < 0) {
171 /* No Content-Length (e.g. chunked): we cannot keep the socket aligned;
172 * bail and let the caller fall back (rare for a ranged GET). */
173 web_disconnect(wf);
174 brix_status_set(st, XRDC_EPROTO, 0, "web GET: no Content-Length");
175 return -1;
176 }
177
178 /* body bytes already sitting in hbuf after the header terminator */
179 {
180 char *bstart = eoh + 4;
181 size_t have = hlen - (size_t) (bstart - hbuf);
182 size_t want = (clen < (long long) len) ? (size_t) clen : len;
183 size_t copied = (have < want) ? have : want;
184 size_t total_body = (size_t) clen; /* must fully consume */
185 size_t consumed;
186
187 memcpy(buf, bstart, copied);
188 consumed = have; /* bytes of body read so far */
189
190 /*
191 * Read the remainder of `want` INCREMENTALLY (not all-or-nothing): on a
192 * mid-body sever, return the bytes copied SO FAR so the caller advances
193 * `off` and resumes the rest with a fresh Range GET — forward progress
194 * under repeated severs, instead of discarding the whole range and
195 * re-reading it from the start (which never completes under heavy loss).
196 */
197 while (copied < want) {
198 ssize_t br = web_read_some(wf, (char *) buf + copied,
199 want - copied, st);
200 if (br <= 0) {
201 web_disconnect(wf);
202 if (copied > 0) {
203 return (ssize_t) copied; /* partial; caller resumes */
204 }
205 if (br == 0) {
206 brix_status_set(st, XRDC_ESOCK, 0,
207 "web GET: peer closed mid-body");
208 }
209 return -1;
210 }
211 consumed += (size_t) br;
212 copied += (size_t) br;
213 }
214 /* drain any body beyond what we wanted, to keep the connection aligned
215 * for the next keep-alive request (status 200 = server ignored Range) */
216 while (consumed < total_body) {
217 char sink[8192];
218 size_t chunk = total_body - consumed;
219 if (chunk > sizeof(sink)) {
220 chunk = sizeof(sink);
221 }
222 if (brix_read_full(&wf->io, sink, chunk, st) != 0) {
223 web_disconnect(wf);
224 return (ssize_t) want; /* have the wanted bytes; lost keep-alive */
225 }
226 consumed += chunk;
227 }
228 if (status == 200) {
229 /* Range ignored: keep-alive is fine (we consumed the whole body), but
230 * for a large file that is wasteful — leave the connection up; the
231 * caller still got the right bytes for this offset. */
232 }
233 return (ssize_t) want;
234 }

272 wf->io.fd = -1;

274 *si_out = si;

296 int v = atoi(e);
297 return (v > 0) ? v : 0;
298 }
299 { int xv; if (brix_xrdrc_default_ms("max_stall_ms", &xv)) { return xv; } }
300 return XRDC_DEFAULT_MAX_STALL_MS;

304ssize_t
305brix_webfile_pread(brix_webfile *wf, int64_t off, void *buf, size_t len,
306 brix_status *st)
307{
308 uint64_t deadline;
309 unsigned attempt = 0;
310 int window_ms;
311 size_t got = 0;
312
313 if (off >= wf->size) {
314 return 0; /* past EOF */
315 }
316 if ((int64_t) len > wf->size - off) {
317 len = (size_t) (wf->size - off); /* clamp tail read */
318 }
319 if (len == 0) {
320 return 0;
321 }
322
323 /*
324 * Deadline-bounded resume (mirrors the root:// resilience window). On a
325 * transport sever, reconnect and re-issue the Range GET from off+got —
326 * web_get_range resumes there and returns PARTIAL progress on a mid-body
327 * sever, so this loop accumulates `len` bytes across reconnects rather than
328 * failing on the first reset. The deadline measures time-since-progress (it
329 * resets on every byte read), so a steadily-advancing transfer never times
330 * out — only a true stall (no progress for the whole window) does. A non-
331 * retryable fault (404/403/…) or window<=0 ($XRDC_MAX_STALL_MS=0, fail-fast)
332 * returns immediately.
333 */
334 window_ms = webfile_window_ms();
335 deadline = brix_mono_ns() + (uint64_t) window_ms * 1000000ULL;
336
337 for (;;) {
338 if (wf->connected || web_connect(wf, st) == 0) {
339 ssize_t r = web_get_range(wf, off + (int64_t) got,
340 (char *) buf + got, len - got, st);
341 if (r > 0) {
342 got += (size_t) r;
343 if (got >= len) {
344 return (ssize_t) got; /* full read */
345 }
346 attempt = 0; /* progress: reset backoff */
347 if (window_ms > 0) {
348 deadline = brix_mono_ns()
349 + (uint64_t) window_ms * 1000000ULL;
350 }
351 continue; /* resume at off+got */
352 }
353 if (r == 0) {
354 return (ssize_t) got; /* genuine EOF */
355 }
356 /* r < 0: transport/protocol fault — fall through to the retry gate. */
357 }
358 if (window_ms <= 0 || !brix_status_retryable(st)
359 || brix_mono_ns() >= deadline) {
360 return (got > 0) ? (ssize_t) got : -1;
361 }
362 brix_backoff_sleep_fast(attempt++);
363 }
364}
