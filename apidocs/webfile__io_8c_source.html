13{

15 return;
16 }

19 wf->tls_ctx = NULL;

29int
30web_connect(brix_webfile *wf, brix_status *st)
31{
32 memset(&wf->io, 0, sizeof(wf->io));
33 wf->io.fd = brix_tcp_connect(wf->host, wf->port, wf->timeout_ms, st);
34 if (wf->io.fd < 0) {
35 return -1;
36 }
37 wf->io.timeout_ms = wf->timeout_ms;
38 if (wf->tls && brix_tls_client(&wf->io, wf->host, wf->verify, wf->verify,
39 wf->ca_dir[0] ? wf->ca_dir : NULL,
40 &wf->tls_ctx, st) != 0) {
41 close(wf->io.fd);
42 wf->io.fd = -1;
43 return -1;
44 }
45 wf->connected = 1;
46 return 0;
47}

50/* Read up to n bytes (branches on TLS). >0 bytes, 0 EOF, -1 error. */
51ssize_t
52web_read_some(brix_webfile *wf, void *buf, size_t n, brix_status *st)
53{
54 if (wf->io.ssl != NULL) {
55 size_t got = 0;
56 if (brix_tls_read_some(&wf->io, buf, n, &got, st) != 0) {
57 return -1;
58 }
59 return (ssize_t) got;
60 }
61 struct pollfd pfd;
62 ssize_t r;
63 int pr;
64 pfd.fd = wf->io.fd; pfd.events = POLLIN; pfd.revents = 0;
65 do { pr = poll(&pfd, 1, wf->io.timeout_ms); } while (pr < 0 && errno == EINTR);
66 if (pr <= 0) {
67 brix_status_set(st, XRDC_ESOCK, pr == 0 ? ETIMEDOUT : errno, "web read");
68 return -1;
69 }
70 do { r = read(wf->io.fd, buf, n); } while (r < 0 && errno == EINTR);
71 if (r < 0) {
72 brix_status_set(st, XRDC_ESOCK, errno, "web read: %s", strerror(errno));
73 return -1;
74 }
75 return r;
76}

79/* Case-insensitive header value lookup in a NUL-terminated header block. */
80long long
81hdr_clen(const char *hdrs)
82{
83 const char *p = hdrs;
84 while ((p = strchr(p, '\n')) != NULL) {
85 p++;
86 if (strncasecmp(p, "Content-Length:", 15) == 0) {
87 return strtoll(p + 15, NULL, 10);
88 }
89 }
90 if (strncasecmp(hdrs, "Content-Length:", 15) == 0) {
91 return strtoll(hdrs + 15, NULL, 10);
92 }
93 return -1;
94}

97/*
98 * Response-header parse result carried out of web_range_read_headers.
99 *
100 * WHAT: the status line code plus the location of the CRLFCRLF terminator and
101 * the total header+overflow bytes sitting in the caller's header buffer.
102 * WHY: splitting the header read out of web_get_range means the status code,
103 * the end-of-headers pointer, and the amount already buffered (some of
104 * which is body) must all flow back to the body-streaming step.
105 * HOW: populated only on a 206/200 success; on 416/error the header reader
106 * returns the disposition directly and this struct is not consumed.
107 */
108typedef struct {
109 char *hbuf; /* caller's header buffer (headers + any body overflow) */
110 char *eoh; /* points at the "\r\n\r\n" terminator inside hbuf */
111 size_t hlen; /* total bytes read into hbuf (headers + body overflow) */
112 long long clen; /* parsed Content-Length (>= 0 on the success path) */
113 int status; /* HTTP status code (206 or 200 on the success path) */
114} web_range_hdr_t;

117/*
118 * Build and send the ranged GET request over the persistent socket.
119 *
120 * WHAT: format the keep-alive "GET … Range: bytes=off-end" request and write it.
121 * WHY: isolates request framing (the wire-frozen request line + headers) from
122 * response handling so web_get_range reads as a linear send/read/stream.
123 * HOW: snprintf into a caller-owned buffer; on overflow set EUSAGE, on a write
124 * fault disconnect (so the caller reconnects and retries). Returns 0 on
125 * success, -1 with st set (and wf disconnected on transport fault).
126 */
127static int
128web_range_send_req(brix_webfile *wf, int64_t off, size_t len, brix_status *st)
129{
130 char req[3200];
131 int rn = snprintf(req, sizeof(req),
132 "GET %s HTTP/1.1\r\nHost: %s\r\nUser-Agent: xrootdfs\r\n"
133 "Accept: */*\r\nConnection: keep-alive\r\n"
134 "Range: bytes=%lld-%lld\r\n%s\r\n",
135 wf->path[0] ? wf->path : "/", wf->hostport,
136 (long long) off, (long long) (off + (int64_t) len - 1),
137 wf->auth);
138
139 if (rn < 0 || (size_t) rn >= sizeof(req)) {
140 brix_status_set(st, XRDC_EUSAGE, 0, "web GET: request too long");
141 return -1;
142 }
143 if (brix_write_full(&wf->io, req, (size_t) rn, st) != 0) {
144 web_disconnect(wf);
145 return -1;
146 }
147 return 0;
148}

151/*
152 * Read response headers and evaluate the status line.
153 *
154 * WHAT: read up to CRLFCRLF into hbuf, parse the HTTP status, and classify it.
155 * WHY: concentrates the header-buffer loop and the status→disposition mapping
156 * (416 = empty read, 404/401/403/other = error) in one place so the body
157 * stream step only ever runs on a 206/200 with a valid Content-Length.
158 * HOW: fills hbuf (NUL-terminated), locates the terminator, parses the code,
159 * then NUL-terminates the header block and confirms Content-Length. On a
160 * success (206/200) writes *out and returns 1; on the range-past-EOF 416
161 * returns 0 (no bytes); on any error returns -1 with st set. Every non-1
162 * path that leaves the socket unusable disconnects wf. On success *out
163 * carries hbuf, the terminator, the byte count, Content-Length, and status.
164 */
165static int
166web_range_read_headers(brix_webfile *wf, char *hbuf, brix_status *st,
167 web_range_hdr_t *out)
168{
169 size_t hlen = 0;
170 char *eoh = NULL;
171 int status = 0;
172 long long clen;
173
174 for (;;) {
175 ssize_t r;
176 if (hlen >= WEB_HDR_MAX) {
177 web_disconnect(wf);
178 brix_status_set(st, XRDC_EPROTO, 0, "web GET: header too large");
179 return -1;
180 }
181 r = web_read_some(wf, hbuf + hlen, WEB_HDR_MAX - hlen, st);
182 if (r <= 0) {
183 web_disconnect(wf);
184 if (r == 0) {
185 brix_status_set(st, XRDC_ESOCK, 0, "web GET: peer closed");
186 }
187 return -1;
188 }
189 hlen += (size_t) r;
190 hbuf[hlen] = '\0';
191 eoh = strstr(hbuf, "\r\n\r\n");
192 if (eoh != NULL) {
193 break;
194 }
195 }
196
197 if (strncmp(hbuf, "HTTP/", 5) == 0) {
198 const char *sp = strchr(hbuf, ' ');
199 status = sp ? atoi(sp + 1) : 0;
200 }
201 if (status == 416) { /* range past EOF → no bytes */
202 return 0;
203 }
204 if (status != 206 && status != 200) {
205 web_disconnect(wf);
206 if (status == 404) {
207 brix_status_set(st, kXR_NotFound, 0, "not found");
208 } else if (status == 401 || status == 403) {
209 brix_status_set(st, kXR_NotAuthorized, 0, "HTTP %d", status);
210 } else {
211 brix_status_set(st, XRDC_EPROTO, 0, "web GET: HTTP %d", status);
212 }
213 return -1;
214 }
215
216 *eoh = '\0';
217 clen = hdr_clen(hbuf);
218 if (clen < 0) {
219 /* No Content-Length (e.g. chunked): we cannot keep the socket aligned;
220 * bail and let the caller fall back (rare for a ranged GET). */
221 web_disconnect(wf);
222 brix_status_set(st, XRDC_EPROTO, 0, "web GET: no Content-Length");
223 return -1;
224 }
225
226 out->hbuf = hbuf;
227 out->eoh = eoh;
228 out->hlen = hlen;
229 out->clen = clen;
230 out->status = status;
231 return 1;
232}

235/*
236 * Copy the wanted range bytes out of the parsed response and drain the rest.
237 *
238 * WHAT: emit up to len bytes of body into buf, reading the remainder past the
239 * header overflow and draining any body beyond what was wanted.
240 * WHY: the body path has its own resume semantics (partial forward progress on
241 * a mid-body sever) and keep-alive alignment (full-body drain) that read
242 * cleanly as one step separate from header parsing.
243 * HOW: seeds `copied` from the overflow already in hbuf, then reads the rest of
244 * `want` INCREMENTALLY so a mid-body sever returns the bytes copied so far
245 * (caller resumes at off+copied); finally drains to Content-Length to keep
246 * the connection aligned. Returns bytes delivered, or -1 with st set.
247 */
248static ssize_t
249web_range_stream_body(brix_webfile *wf, void *buf, size_t len,
250 const web_range_hdr_t *hdr, brix_status *st)
251{
252 char *bstart = hdr->eoh + 4;
253 size_t have = hdr->hlen - (size_t) (bstart - hdr->hbuf);
254 size_t want = (hdr->clen < (long long) len) ? (size_t) hdr->clen : len;
255 size_t copied = (have < want) ? have : want;
256 size_t total_body = (size_t) hdr->clen; /* must fully consume */
257 size_t consumed;
258
259 memcpy(buf, bstart, copied);
260 consumed = have; /* bytes of body read so far */
261
262 /*
263 * Read the remainder of `want` INCREMENTALLY (not all-or-nothing): on a
264 * mid-body sever, return the bytes copied SO FAR so the caller advances
265 * `off` and resumes the rest with a fresh Range GET — forward progress
266 * under repeated severs, instead of discarding the whole range and
267 * re-reading it from the start (which never completes under heavy loss).
268 */
269 while (copied < want) {
270 ssize_t br = web_read_some(wf, (char *) buf + copied,
271 want - copied, st);
272 if (br <= 0) {
273 web_disconnect(wf);
274 if (copied > 0) {
275 return (ssize_t) copied; /* partial; caller resumes */
276 }
277 if (br == 0) {
278 brix_status_set(st, XRDC_ESOCK, 0,
279 "web GET: peer closed mid-body");
280 }
281 return -1;
282 }
283 consumed += (size_t) br;
284 copied += (size_t) br;
285 }
286 /* drain any body beyond what we wanted, to keep the connection aligned
287 * for the next keep-alive request (status 200 = server ignored Range) */
288 while (consumed < total_body) {
289 char sink[8192];
290 size_t chunk = total_body - consumed;
291 if (chunk > sizeof(sink)) {
292 chunk = sizeof(sink);
293 }
294 if (brix_read_full(&wf->io, sink, chunk, st) != 0) {
295 web_disconnect(wf);
296 return (ssize_t) want; /* have the wanted bytes; lost keep-alive */
297 }
298 consumed += chunk;
299 }
300 if (hdr->status == 200) {
301 /* Range ignored: keep-alive is fine (we consumed the whole body), but
302 * for a large file that is wasteful — leave the connection up; the
303 * caller still got the right bytes for this offset. */
304 }
305 return (ssize_t) want;
306}

309/* One ranged GET over the (already-connected) persistent socket. Fills up to len
310 * bytes at off into buf; returns bytes read (0 = EOF/416), or -1 (st set). On any
311 * transport/protocol fault returns -1 AND disconnects so the caller can retry. */
312ssize_t
313web_get_range(brix_webfile *wf, int64_t off, void *buf, size_t len,
314 brix_status *st)
315{
316 char hbuf[WEB_HDR_MAX + 1];
317 web_range_hdr_t hdr = {0};
318 int hr;
319
320 if (web_range_send_req(wf, off, len, st) != 0) {
321 return -1;
322 }
323
324 hr = web_range_read_headers(wf, hbuf, st, &hdr);
325 if (hr <= 0) {
326 return (ssize_t) hr; /* 0 = 416 (no bytes), -1 = error (st set) */
327 }
328
329 return web_range_stream_body(wf, buf, len, &hdr, st);
330}

384 * library default — the SAME window the root:// data path uses, so an HTTP
385 * download rides out packet loss the same way instead of dying on the first sever. */
386int
387webfile_window_ms(void)
388{
389 const char *e = getenv("XRDC_MAX_STALL_MS");
390 if (e != NULL && *e != '\0') {
391 int v = atoi(e);
392 return (v > 0) ? v : 0;
393 }
394 { int xv; if (brix_xrdrc_default_ms("max_stall_ms", &xv)) { return xv; } }
395 return XRDC_DEFAULT_MAX_STALL_MS;
396}

399ssize_t
400brix_webfile_pread(brix_webfile *wf, int64_t off, void *buf, size_t len,
401 brix_status *st)
402{
403 uint64_t deadline;
404 unsigned attempt = 0;
405 int window_ms;
406 size_t got = 0;
407
408 if (off >= wf->size) {
409 return 0; /* past EOF */
410 }
411 if ((int64_t) len > wf->size - off) {
412 len = (size_t) (wf->size - off); /* clamp tail read */
413 }
414 if (len == 0) {
415 return 0;
416 }
417
418 /*
419 * Deadline-bounded resume (mirrors the root:// resilience window). On a
420 * transport sever, reconnect and re-issue the Range GET from off+got —
421 * web_get_range resumes there and returns PARTIAL progress on a mid-body
422 * sever, so this loop accumulates `len` bytes across reconnects rather than
423 * failing on the first reset. The deadline measures time-since-progress (it
424 * resets on every byte read), so a steadily-advancing transfer never times
425 * out — only a true stall (no progress for the whole window) does. A non-
426 * retryable fault (404/403/…) or window<=0 ($XRDC_MAX_STALL_MS=0, fail-fast)
427 * returns immediately.
428 */
429 window_ms = webfile_window_ms();
430 deadline = brix_mono_ns() + (uint64_t) window_ms * 1000000ULL;
431
432 for (;;) {
433 if (wf->connected || web_connect(wf, st) == 0) {
434 ssize_t r = web_get_range(wf, off + (int64_t) got,
435 (char *) buf + got, len - got, st);
436 if (r > 0) {
437 got += (size_t) r;
438 if (got >= len) {
439 return (ssize_t) got; /* full read */
440 }
441 attempt = 0; /* progress: reset backoff */
442 if (window_ms > 0) {
443 deadline = brix_mono_ns()
444 + (uint64_t) window_ms * 1000000ULL;
445 }
446 continue; /* resume at off+got */
447 }
448 if (r == 0) {
449 return (ssize_t) got; /* genuine EOF */
450 }
451 /* r < 0: transport/protocol fault — fall through to the retry gate. */
452 }
453 if (window_ms <= 0 || !brix_status_retryable(st)
454 || brix_mono_ns() >= deadline) {
455 return (got > 0) ? (ssize_t) got : -1;
456 }
457 brix_backoff_sleep_fast(attempt++);
458 }
459}
