1/*
2 * cinfo_l1.c - the per-worker write-through cinfo cache (section 6.4). See header.
3 *
4 * A malloc-owned hash table (FNV-1a, chaining) over an intrusive MRU-at-head LRU
5 * list. get/put/drop are O(1) amortised; put past the bound evicts the LRU tail.
6 * No nginx pool so it is safe on the cache fill worker thread.
7 */
8#include "cinfo_l1.h"
9
10#include <errno.h>
11#include <pthread.h>
12#include <stdlib.h>
13#include <string.h>
14
15/* Default entry bound when the policy leaves l1_entries 0 (section 2.3). */
16#define CINFO_L1_DEFAULT_ENTRIES 4096
17
18typedef struct cinfo_l1_entry_s {
19 char *key; /* strdup'd object key (lookup) */
20 brix_cache_cinfo_t hdr; /* the cached header record */
21 struct cinfo_l1_entry_s *hnext; /* hash-bucket chain */
22 struct cinfo_l1_entry_s *lru_prev; /* LRU list (head = MRU, tail = LRU) */
23 struct cinfo_l1_entry_s *lru_next;
24} cinfo_l1_entry_t;
25
26typedef struct {
27 cinfo_l1_entry_t **buckets;
28 size_t nbuckets; /* power of two */
29 size_t mask; /* nbuckets - 1 */
30 cinfo_l1_entry_t *lru_head;
31 cinfo_l1_entry_t *lru_tail;
32 size_t count;
33 size_t max;
34 ngx_log_t *log;
35 /* The L1 is shared between the main event loop (get/drop on the open path)
36 * and cache-fill worker threads (put on fill completion). Every mutating
37 * accessor — including get, which reorders the LRU — must hold this. Without
38 * it the hash/LRU pointers race and corrupt, which manifested downstream as
39 * pool use-after-free / SIGSEGV under concurrent fills (found via TSan). */
40 pthread_mutex_t mtx;
41} cinfo_l1_impl_t;

45/* Smallest power of two >= n (>= 1). */
46static size_t
47cinfo_l1_pow2_ceil(size_t n)
48{
49 size_t p = 1;
50
51 while (p < n) {
52 p <<= 1;
53 }
54 return p;
55}

57/* FNV-1a over a NUL-terminated key. */
58static uint64_t
59cinfo_l1_hash(const char *key)
60{
61 uint64_t h = 1469598103934665603ULL; /* FNV offset basis */
62 const unsigned char *p = (const unsigned char *) key;
63
64 while (*p != '\0') {
65 h ^= (uint64_t) *p++;
66 h *= 1099511628211ULL; /* FNV prime */
67 }
68 return h;
69}

73static void
74cinfo_l1_lru_unlink(cinfo_l1_impl_t *t, cinfo_l1_entry_t *e)
75{
76 if (e->lru_prev != NULL) {
77 e->lru_prev->lru_next = e->lru_next;
78 } else {
79 t->lru_head = e->lru_next;
80 }
81 if (e->lru_next != NULL) {
82 e->lru_next->lru_prev = e->lru_prev;
83 } else {
84 t->lru_tail = e->lru_prev;
85 }
86 e->lru_prev = NULL;
87 e->lru_next = NULL;
88}

90static void
91cinfo_l1_lru_push_front(cinfo_l1_impl_t *t, cinfo_l1_entry_t *e)
92{
93 e->lru_prev = NULL;
94 e->lru_next = t->lru_head;
95 if (t->lru_head != NULL) {
96 t->lru_head->lru_prev = e;
97 }
98 t->lru_head = e;
99 if (t->lru_tail == NULL) {

101 }

104/* Promote `e` to MRU. */
105static void
106cinfo_l1_touch(cinfo_l1_impl_t *t, cinfo_l1_entry_t *e)
107{
108 if (t->lru_head == e) {
109 return;

117static cinfo_l1_entry_t *
118cinfo_l1_find(cinfo_l1_impl_t *t, const char *key, size_t bucket)
119{
120 cinfo_l1_entry_t *e;
121
122 for (e = t->buckets[bucket]; e != NULL; e = e->hnext) {
123 if (strcmp(e->key, key) == 0) {
124 return e;
125 }
126 }
127 return NULL;
128}

130/* Remove `e` from its hash chain (LRU unlink + free are the caller's job). */
131static void
132cinfo_l1_bucket_remove(cinfo_l1_impl_t *t, cinfo_l1_entry_t *e, size_t bucket)
133{
134 cinfo_l1_entry_t **pp = &t->buckets[bucket];
135
136 while (*pp != NULL) {
137 if (*pp == e) {
138 *pp = e->hnext;
139 return;
140 }
141 pp = &(*pp)->hnext;
142 }
143}

145static void
146cinfo_l1_evict_tail(cinfo_l1_impl_t *t)
147{
148 cinfo_l1_entry_t *victim = t->lru_tail;
149 size_t bucket;
150
151 if (victim == NULL) {
152 return;
153 }
154 bucket = (size_t) (cinfo_l1_hash(victim->key) & t->mask);
155 cinfo_l1_bucket_remove(t, victim, bucket);
156 cinfo_l1_lru_unlink(t, victim);
157 free(victim->key);
158 free(victim);
159 t->count--;
160}

164brix_cinfo_l1_t *
165brix_cinfo_l1_create(size_t max_entries, ngx_log_t *log)
166{
167 brix_cinfo_l1_t *l1;
168 cinfo_l1_impl_t *t;
169
170 if (max_entries == 0) {
171 max_entries = CINFO_L1_DEFAULT_ENTRIES;
172 }
173 l1 = calloc(1, sizeof(*l1));
174 t = calloc(1, sizeof(*t));
175 if (l1 == NULL || t == NULL) {
176 free(l1);
177 free(t);
178 errno = ENOMEM;
179 return NULL;
180 }
181 t->max = max_entries;
182 t->nbuckets = cinfo_l1_pow2_ceil(max_entries);
183 t->mask = t->nbuckets - 1;
184 t->log = log;
185 t->buckets = calloc(t->nbuckets, sizeof(*t->buckets));
186 if (t->buckets == NULL) {
187 free(t);
188 free(l1);
189 errno = ENOMEM;
190 return NULL;
191 }
192 if (pthread_mutex_init(&t->mtx, NULL) != 0) {
193 free(t->buckets);
194 free(t);
195 free(l1);
196 errno = ENOMEM;
197 return NULL;
198 }
199 l1->opaque = t;
200 return l1;
201}

206 cinfo_l1_impl_t *t;
207 cinfo_l1_entry_t *e;

214 e = t->lru_head;

217
218 free(e->key);
219 free(e);
220 e = next;
221 }

229brix_cinfo_l1_get(brix_cinfo_l1_t *l1, const char *key,

231{
232 cinfo_l1_impl_t *t;
233 cinfo_l1_entry_t *e;
234 size_t bucket;
235 ngx_int_t rc = NGX_DECLINED;
236
237 if (l1 == NULL || l1->opaque == NULL || key == NULL || out == NULL) {
238 return NGX_DECLINED;
239 }
240 t = l1->opaque;
241 bucket = (size_t) (cinfo_l1_hash(key) & t->mask);
242 pthread_mutex_lock(&t->mtx);
243 e = cinfo_l1_find(t, key, bucket);
244 if (e != NULL) {
245 *out = e->hdr;
246 cinfo_l1_touch(t, e); /* mutates the LRU — must hold the lock */
247 rc = NGX_OK;
248 }
249 pthread_mutex_unlock(&t->mtx);

253void
254brix_cinfo_l1_put(brix_cinfo_l1_t *l1, const char *key,
255 const brix_cache_cinfo_t *hdr)
256{
257 cinfo_l1_impl_t *t;
258 cinfo_l1_entry_t *e;
259 size_t bucket;
260
261 if (l1 == NULL || l1->opaque == NULL || key == NULL || hdr == NULL) {
262 return;
263 }
264 t = l1->opaque;
265 bucket = (size_t) (cinfo_l1_hash(key) & t->mask);
266
267 pthread_mutex_lock(&t->mtx);
268 e = cinfo_l1_find(t, key, bucket);
269 if (e != NULL) {
270 e->hdr = *hdr; /* write-through update */
271 cinfo_l1_touch(t, e);
272 pthread_mutex_unlock(&t->mtx);
273 return;
274 }
275
276 e = calloc(1, sizeof(*e));
277 if (e == NULL) {
278 pthread_mutex_unlock(&t->mtx);
279 return; /* L1 is best-effort - a miss just refetches */
280 }
281 e->key = strdup(key);
282 if (e->key == NULL) {
283 free(e);
284 pthread_mutex_unlock(&t->mtx);
285 return;
286 }
287 e->hdr = *hdr;
288 e->hnext = t->buckets[bucket];
289 t->buckets[bucket] = e;
290 cinfo_l1_lru_push_front(t, e);
291 t->count++;
292
293 if (t->count > t->max) {
294 cinfo_l1_evict_tail(t);
295 }
296 pthread_mutex_unlock(&t->mtx);
297}
