6/* ---- Parse ListObjectsV2 query parameters into caller buffers and flags ----
7 *
8 * WHAT: Reads the ListObjectsV2 query arguments (prefix, delimiter,
9 * continuation-token, fetch-owner, encoding-type, max-keys) from the request
10 * and fills the caller-provided prefix and delimiter buffers, the decoded
11 * start_after cursor, and the max_keys, fetch_owner and url_encode outputs.
12 * No return value: malformed inputs degrade gracefully (empty start_after,
13 * flags left disabled).
14 *
15 * WHY: Isolating parameter parsing keeps the response builder focused on
16 * pagination and XML emission, and confines the raw continuation-token buffer
17 * to the smallest scope that owns it. Pagination semantics are unchanged: a
18 * malformed token decodes to an empty start_after, which returns the listing
19 * from the beginning exactly as before.
20 *
21 * HOW:
22 * 1. Fetch each string argument via brix_http_query_get() into a bounded buffer.
23 * 2. Set fetch_owner and url_encode from their canonical "true" / "url" values.
24 * 3. Resolve max_keys through s3_list_parse_max_keys() (clamped to config).
25 * 4. If a continuation token is present, b64url-decode it into start_after;
26 * on decode failure leave start_after empty (return from the beginning).
27 */
28static void
29s3_list_decode_params(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
30 u_char *prefix_buf, size_t prefix_sz,
31 u_char *delimiter_buf, size_t delimiter_sz,
32 char *start_after, size_t start_after_sz,
33 int *max_keys, int *fetch_owner, int *url_encode)
34{
35 u_char token_buf[S3_MAX_PARAM] = { 0 }; /* ContinuationToken (b64url) */
36 u_char fetch_owner_buf[8] = { 0 };
37 u_char encoding_type_buf[8] = { 0 };
38
39 brix_http_query_get(r->args, "prefix", (char *) prefix_buf, prefix_sz, S3_LIST_QUERY_FLAGS);
40 brix_http_query_get(r->args, "delimiter", (char *) delimiter_buf, delimiter_sz, S3_LIST_QUERY_FLAGS);
41 brix_http_query_get(r->args, "continuation-token", (char *) token_buf, sizeof(token_buf), S3_LIST_QUERY_FLAGS);
42 brix_http_query_get(r->args, "fetch-owner", (char *) fetch_owner_buf, sizeof(fetch_owner_buf), S3_LIST_QUERY_FLAGS);
43 brix_http_query_get(r->args, "encoding-type", (char *) encoding_type_buf, sizeof(encoding_type_buf), S3_LIST_QUERY_FLAGS);
44
45 *fetch_owner = (ngx_strcasecmp(fetch_owner_buf, (u_char *) "true") == 0) ? 1 : 0;
46 *url_encode = (ngx_strcasecmp(encoding_type_buf, (u_char *) "url") == 0) ? 1 : 0;
47 *max_keys = s3_list_parse_max_keys(r, (int) cf->max_keys);
48
49 /* Decode continuation token: b64url-encoded last key from the previous page.
50 * On decode failure start_after stays empty and the listing starts over. */
51 if (token_buf[0] != '\0') {
52 ssize_t n = b64url_decode((const char *) token_buf,
53 strlen((const char *) token_buf),
54 (uint8_t *) start_after, start_after_sz - 1);
55 start_after[n >= 0 ? n : 0] = '\0';
56 }
57}

59/* ---- Emit the ListBucketResult opening element and pagination metadata ----
60 *
61 * WHAT: Appends the XML prologue, the ListBucketResult open tag and the
62 * listing-level elements (Name, Prefix, KeyCount, MaxKeys, optional Delimiter
63 * and EncodingType, IsTruncated, and NextContinuationToken when truncated)
64 * into the flat xml buffer. Returns NGX_OK, or NGX_HTTP_INTERNAL_SERVER_ERROR
65 * on buffer overflow (via the XML_APPEND macros). Advances *xml_len_io by the
66 * number of bytes written.
67 *
68 * WHY: Splitting the header out of the orchestrator keeps s3_handle_list within
69 * the complexity budget while leaving the byte-for-byte XML output unchanged —
70 * the element order, conditional Delimiter/EncodingType/NextContinuationToken
71 * emission, and the S3 namespace all match the previous inline sequence.
72 *
73 * HOW:
74 * 1. Adopt the caller's current length into a local xml_len the macros drive.
75 * 2. Append the prologue and ListBucketResult open tag with the S3 namespace.
76 * 3. Append Name, Prefix, KeyCount and MaxKeys.
77 * 4. Append Delimiter and EncodingType only when present/requested.
78 * 5. Append IsTruncated, then NextContinuationToken when truncated and set.
79 * 6. Publish the new length back through *xml_len_io and return NGX_OK.
80 */
81static ngx_int_t
82s3_list_emit_header(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
83 const u_char *prefix_buf, const u_char *delimiter_buf,
84 const char *next_token, int url_encode, int truncated,
85 int max_keys, int key_count,
86 u_char *xml, size_t *xml_len_io, size_t xml_capacity)
87{
88 size_t xml_len = *xml_len_io;
89
90 XML_APPEND("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
91 XML_APPEND("<ListBucketResult "
92 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">");
93 XML_APPEND_ELEM("Name", cf->bucket.data, cf->bucket.len);
94 XML_APPEND_ELEM("Prefix", prefix_buf, strlen((const char *) prefix_buf));
95 XML_APPEND("<KeyCount>%d</KeyCount>", key_count);
96 XML_APPEND("<MaxKeys>%d</MaxKeys>", max_keys);
97
98 if (delimiter_buf[0] != '\0') {
99 XML_APPEND_ELEM("Delimiter", delimiter_buf,
100 strlen((const char *) delimiter_buf));
101 }
102 if (url_encode) {
103 XML_APPEND("<EncodingType>url</EncodingType>");
104 }
105 XML_APPEND("<IsTruncated>%s</IsTruncated>", truncated ? "true" : "false");
106
107 if (truncated && next_token[0] != '\0') {
108 XML_APPEND_ELEM("NextContinuationToken", next_token,
109 strlen(next_token));
110 }
111
112 *xml_len_io = xml_len;
113 return NGX_OK;
114}

116/*
117 * s3_handle_list — ListObjectsV2 XML response builder
118 *
119 * WHAT: Builds the ListBucketResult XML for S3 ListObjectsV2 requests. Parses query
120 * parameters (prefix, delimiter, continuation-token, max-keys), walks the filesystem
121 * to collect matching entries, applies pagination via continuation tokens, and emits
122 * a compliant XML response with Contents (objects) and CommonPrefixes (directories).
123 *
124 * WHY: S3 clients paginate list results using continuation tokens — each token encodes
125 * the last key returned in base64url. The server must skip all entries <= that key on
126 * subsequent requests. Delimiter-based common prefixes group keys under shared prefixes
127 * into <CommonPrefixes> elements per the S3 spec. XML entities (quotes, ampersands) can
128 * expand up to 6× in worst case, so buffer sizing uses 6× as a safety multiplier.
129 *
130 * HOW:
131 * 1. Parse query parameters: prefix, delimiter, continuation-token, max-keys,
132 * fetch-owner, encoding-type — all via s3_get_arg() with bounded buffers
133 * 2. Decode continuation token (b64url → start_after) to find pagination offset
134 * 3. Walk filesystem via s3_walk() collecting entries filtered by prefix/delimiter
135 * 4. Sort entries lexicographically via qsort(entry_cmp)
136 * 5. Skip entries <= start_after to apply continuation token (linear scan on sorted)
137 * 6. Slice [start_idx, end_idx) capped at max_keys — set truncated flag
138 * 7. Encode next continuation token from last returned key if truncated
139 * 8. Estimate XML buffer capacity (512 header + per-key × 6× expansion)
140 * 9. Append XML: ListBucketResult root → Name, Prefix, KeyCount, MaxKeys,
141 * IsTruncated, NextContinuationToken → Contents/CommonPrefixes loop
142 * 10. Create response buffer, increment metrics, send via brix_http_send_xml_buffer()
143 *
144 * Pool allocation: entries array and xml buffer use ngx_palloc(r->pool) — lifetime
145 * survives until the HTTP response filter completes.
146 */
147
