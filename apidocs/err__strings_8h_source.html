1/*
2 * err_strings.h — canonical error message strings for common errnos.
3 *
4 * brix_kxr_err_string(err) returns the lowercase message string that
5 * conformance tests assert for each errno. strerror(EACCES) returns
6 * "Permission denied" (capital P) on Linux; centralising here prevents
7 * any future case or format drift across handlers.
8 */
9#ifndef BRIX_COMPAT_ERR_STRINGS_H
10#define BRIX_COMPAT_ERR_STRINGS_H
11#include <errno.h>
12#include <string.h>
13
14static inline const char *
15brix_kxr_err_string(int err)
16{
17 switch (err) {
18 case EACCES: /* fall through */
19 case EPERM: return "permission denied";
20 case ENOENT: return "no such file or directory";
21 case ENOTDIR:return "not a directory";
22 case EISDIR: return "is a directory";
23 case ENOSPC: return "no space left on device";
24 case EEXIST: return "already exists";
25 case ENOMEM: return "out of memory";
26 default: return strerror(err);
27 }
28}

30#endif /* BRIX_COMPAT_ERR_STRINGS_H */
