8/*
9 * WHAT: Sum the cluster-wide metric counters from the metrics SHM zone into a
10 * single stack struct for the caller to serialise.
11 * WHY: The SHM holds per-listener server slots plus per-protocol blocks; the
12 * dashboard "totals" view aggregates them. Errors are the sum of all
13 * per-op error counters (stream) and 4xx+5xx response counters (HTTP/S3).
14 * HOW: The "(void *) 1" sentinel is nginx's marker for a shm zone that has been
15 * declared but not yet initialised; treat it (and NULL) as "no data" and
16 * leave totals zeroed.
17 */
18void
19dashboard_collect_totals(brix_dashboard_totals_t *totals)
20{
21 ngx_brix_metrics_t *met;
22 ngx_uint_t srv, i, j;
23
24 ngx_memzero(totals, sizeof(*totals));
25
26 if (ngx_brix_shm_zone == NULL
27 || ngx_brix_shm_zone->data == NULL
28 || ngx_brix_shm_zone->data == (void *) 1)
29 {
30 return;
31 }
32
33 met = ngx_brix_shm_zone->data;
34 /* Per-listener slots: connection + byte counters, and stream op errors. */
35 for (srv = 0; srv < BRIX_METRICS_MAX_SERVERS; srv++) {
36 totals->conn_active += (uint64_t) met->servers[srv].connections_active;
37 totals->conn_total += (uint64_t) met->servers[srv].connections_total;
38 totals->bytes_rx += (uint64_t) met->servers[srv].bytes_rx_total;
39 totals->bytes_tx += (uint64_t) met->servers[srv].bytes_tx_total;
40 for (i = 0; i < BRIX_NOPS; i++) {
41 totals->stream_errors += (uint64_t) met->servers[srv].op_err[i];
42 }
43 }
44
45 /* Per-protocol sources stay explicit: each plane's SHM family is
46 * heterogeneous by design (proto_list.h checklist step 3). Indices
47 * come from the central list; JSON keys generate from it. */
48 totals->proto_rx[BRIX_XFER_PROTO_WEBDAV - 1] =
49 (uint64_t) met->webdav.bytes_rx_total;
50 totals->proto_tx[BRIX_XFER_PROTO_WEBDAV - 1] =
51 (uint64_t) met->webdav.bytes_tx_total;
52 totals->proto_rx[BRIX_XFER_PROTO_S3 - 1] =
53 (uint64_t) met->s3.bytes_rx_total;
54 totals->proto_tx[BRIX_XFER_PROTO_S3 - 1] =
55 (uint64_t) met->s3.bytes_tx_total;
56 /* cvmfs is a read-only site cache: clients never upload, so "in" is the
57 * WAN-side origin pull (Stratum-1 fills) and "out" is the LAN-side bytes
58 * served (cache hits + fresh fills). */
59 totals->proto_rx[BRIX_XFER_PROTO_CVMFS - 1] =
60 (uint64_t) met->cvmfs.origin_bytes_total;
61 totals->proto_tx[BRIX_XFER_PROTO_CVMFS - 1] =
62 (uint64_t) met->cvmfs.bytes_served_hit_total
63 + (uint64_t) met->cvmfs.bytes_served_fill_total;
64 totals->proto_errors[BRIX_XFER_PROTO_CVMFS - 1] =
65 (uint64_t) met->cvmfs.fill_failures_total
66 + (uint64_t) met->cvmfs.verify_failures_total;
67
68 /* HTTP error totals = count of all 4xx and 5xx responses, summed over every
69 * method. responses_total is indexed [method][status-class]. */
70 for (i = 0; i < BRIX_WEBDAV_NMETHODS; i++) {
71 totals->proto_errors[BRIX_XFER_PROTO_WEBDAV - 1] +=
72 (uint64_t) met->webdav.responses_total[i][BRIX_HTTP_STATUS_4XX]
73 + (uint64_t) met->webdav.responses_total[i][BRIX_HTTP_STATUS_5XX];
74 }
75
76 for (i = 0; i < BRIX_S3_NMETHODS; i++) {
77 for (j = BRIX_HTTP_STATUS_4XX; j <= BRIX_HTTP_STATUS_5XX; j++) {
78 totals->proto_errors[BRIX_XFER_PROTO_S3 - 1] +=
79 (uint64_t) met->s3.responses_total[i][j];
80 }
81 }
82}

85/*
86 * WHAT: Tally live transfers per protocol bucket (root/webdav/s3/tpc) from the
87 * dashboard transfer table, accumulating active counts and aggregate
88 * ingress/egress bandwidth.
89 * HOW: Each in-use slot is classified — TPC direction wins over protocol so a
90 * TPC transfer lands in the tpc bucket regardless of carrier protocol.
91 * Write transfers count toward ingress, all others toward egress.
92 */
93void
94dashboard_collect_protocols(brix_dashboard_protocols_t *out, int64_t now_ms)
95{
96 brix_transfer_table_t *tbl;
97 ngx_uint_t i;
98
99 ngx_memzero(out, sizeof(*out));
100
101 if (ngx_brix_dashboard_shm_zone == NULL
102 || ngx_brix_dashboard_shm_zone->data == NULL
103 || ngx_brix_dashboard_shm_zone->data == (void *) 1)
104 {
105 return;
106 }
107
108 tbl = ngx_brix_dashboard_shm_zone->data;
109 for (i = 0; i < BRIX_DASHBOARD_MAX_TRANSFERS; i++) {
110 brix_transfer_slot_t *slot = &tbl->slots[i];
111 brix_dashboard_proto_summary_t *summary;
112 uint64_t avg_bps;
113
114 if (slot->in_use == 0) { continue; }
115
116 if (slot->direction == BRIX_XFER_DIR_TPC) {
117 summary = &out->tpc;
118 } else if (slot->proto >= 1 && slot->proto <= BRIX_XFER_NPROTOS) {
119 summary = &out->per[slot->proto - 1];
120 } else {
121 summary = &out->per[BRIX_XFER_PROTO_ROOT - 1];
122 }
123
124 summary->active++;
125 avg_bps = dashboard_avg_bps((int64_t) slot->bytes, slot->start_ms, now_ms);
126 if (slot->direction == BRIX_XFER_DIR_WRITE) {
127 summary->ingress_bps += avg_bps;
128 } else {
129 summary->egress_bps += avg_bps;
130 }
131 }
132}

157json_t *
158dashboard_build_totals(const brix_dashboard_totals_t *totals)
159{
160 json_t *obj = json_object();
161 if (!obj) { return NULL; }
162 json_object_set_new(obj, "connections_active", json_integer((json_int_t) totals->conn_active));
163 json_object_set_new(obj, "connections_total", json_integer((json_int_t) totals->conn_total));
164 json_object_set_new(obj, "bytes_rx_total", json_integer((json_int_t) totals->bytes_rx));
165 json_object_set_new(obj, "bytes_tx_total", json_integer((json_int_t) totals->bytes_tx));
166 /* Per-HTTP-protocol keys generate from the central list (dash_name):
167 * "webdav_bytes_rx", "s3_errors_total", ... — byte-identical to the
168 * historic hand-written keys. The stream plane keeps its historic
169 * global keys (bytes_rx_total / stream_errors_total) above. */
170 {
171 static const char *names[BRIX_XFER_NPROTOS] = {
172#define X(ID, metric_label, dash_name, http_plane) dash_name,
173 BRIX_PROTO_LIST(X)
174#undef X
175 };
176 static const unsigned http_plane[BRIX_XFER_NPROTOS] = {
177#define X(ID, metric_label, dash_name, http_plane) http_plane,
178 BRIX_PROTO_LIST(X)
179#undef X
180 };
181 char key[64];
182 ngx_uint_t p;
183
184 for (p = 0; p < BRIX_XFER_NPROTOS; p++) {
185 if (!http_plane[p]) {
186 continue;
187 }
188 ngx_snprintf((u_char *) key, sizeof(key), "%s_bytes_rx%Z",
189 names[p]);
190 json_object_set_new(obj, key,
191 json_integer((json_int_t) totals->proto_rx[p]));
192 ngx_snprintf((u_char *) key, sizeof(key), "%s_bytes_tx%Z",
193 names[p]);
194 json_object_set_new(obj, key,
195 json_integer((json_int_t) totals->proto_tx[p]));
196 }
197 json_object_set_new(obj, "stream_errors_total",
198 json_integer((json_int_t) totals->stream_errors));
199 for (p = 0; p < BRIX_XFER_NPROTOS; p++) {
200 if (!http_plane[p]) {
201 continue;
202 }
203 ngx_snprintf((u_char *) key, sizeof(key), "%s_errors_total%Z",
204 names[p]);
205 json_object_set_new(obj, key,
206 json_integer((json_int_t) totals->proto_errors[p]));
207 }
208 }
209 return obj;
210}

230json_t *
231dashboard_build_protocols(int64_t now_ms,
232 const brix_dashboard_totals_t *totals)
233{
234 brix_dashboard_protocols_t ps;
235 json_t *obj, *tpc_obj;
236
237 dashboard_collect_protocols(&ps, now_ms);
238
239 obj = json_object();
240 if (!obj) { return NULL; }
241
242 /* One summary per protocol, keys from the central list. The stream
243 * plane pairs the GLOBAL byte counters (its historic shape); HTTP
244 * planes pair their per-proto totals. */
245 {
246 ngx_uint_t p;
247
248 for (p = 0; p < BRIX_XFER_NPROTOS; p++) {
249 uint64_t rx = (p == BRIX_XFER_PROTO_ROOT - 1)
250 ? totals->bytes_rx : totals->proto_rx[p];
251 uint64_t tx = (p == BRIX_XFER_PROTO_ROOT - 1)
252 ? totals->bytes_tx : totals->proto_tx[p];
253
254 json_object_set_new(obj, dashboard_proto_name((uint8_t)(p + 1)),
255 dashboard_build_proto_summary(&ps.per[p], rx, tx));
256 }
257 }
258
259 /* tpc has no cumulative byte counters — build inline */
260 tpc_obj = json_object();
261 if (tpc_obj) {
262 json_object_set_new(tpc_obj, "active", json_integer((json_int_t) ps.tpc.active));
263 json_object_set_new(tpc_obj, "ingress_bps", json_integer((json_int_t) ps.tpc.ingress_bps));
264 json_object_set_new(tpc_obj, "egress_bps", json_integer((json_int_t) ps.tpc.egress_bps));
265 }
266 json_object_set_new(obj, "tpc", tpc_obj);
267 return obj;
268}

271json_t *
272dashboard_build_events(ngx_pool_t *pool, ngx_uint_t redact)
273{
274 brix_dashboard_event_t *events;
275 ngx_uint_t n, i;
276 json_t *arr;
277
278 arr = json_array();
279 if (!arr) { return NULL; }
280
281 events = ngx_pcalloc(pool, sizeof(*events) * BRIX_DASHBOARD_MAX_EVENTS);
282 if (events == NULL) { return arr; }
283
284 n = brix_dashboard_events_snapshot(events, BRIX_DASHBOARD_MAX_EVENTS);
285 for (i = 0; i < n; i++) {
286 json_t *ev = json_object();
287 if (!ev) { continue; }
288 json_object_set_new(ev, "sequence", json_integer((json_int_t) events[i].sequence));
289 json_object_set_new(ev, "time_ms", json_integer((json_int_t) events[i].time_ms));
290 json_object_set_new(ev, "class", json_string(dashboard_event_class_name(events[i].class_id)));
291 json_object_set_new(ev, "protocol", json_string(dashboard_proto_name(events[i].proto)));
292 json_object_set_new(ev, "status", json_integer((json_int_t) events[i].status));
293 /* message/path_hint can embed paths/identities — empty for anonymous. */
294 json_object_set_new(ev, "message", json_string(redact ? "" : events[i].message));
295 json_object_set_new(ev, "path_hint", json_string(redact ? "" : events[i].path_hint));
296 json_array_append_new(arr, ev);
297 }
298 return arr;
299}

302/*
303 * dashboard_fill_history — adds "bucket_seconds" and "buckets" directly to
304 * `target`. Called with target=root for the v1/history endpoint (flat shape)
305 * and with target=sub-object for the snapshot endpoint (nested under "history").
306 */
307void
308dashboard_fill_history(json_t *target, ngx_pool_t *pool)
309{
310 brix_dashboard_history_bucket_t *buckets;
311 ngx_uint_t n, i;
312 json_t *arr;
313
314 json_object_set_new(target, "bucket_seconds",
315 json_integer(BRIX_DASHBOARD_HISTORY_INTERVAL_MS / 1000));
316
317 arr = json_array();
318 buckets = ngx_pcalloc(pool,
319 sizeof(*buckets) * BRIX_DASHBOARD_HISTORY_BUCKETS);
320 if (buckets == NULL || arr == NULL) {
321 json_object_set_new(target, "buckets", arr ? arr : json_array());
322 return;
323 }
324
325 n = brix_dashboard_history_snapshot(buckets,
326 BRIX_DASHBOARD_HISTORY_BUCKETS);
327 for (i = 0; i < n; i++) {
328 json_t *b = json_object();
329 if (!b) { continue; }
330 json_object_set_new(b, "time_ms", json_integer((json_int_t) buckets[i].bucket_start_ms));
331 {
332 char akey[48];
333 ngx_uint_t p;
334
335 for (p = 0; p < BRIX_XFER_NPROTOS; p++) {
336 ngx_snprintf((u_char *) akey, sizeof(akey), "active_%s%Z",
337 dashboard_proto_name((uint8_t)(p + 1)));
338 json_object_set_new(b, akey,
339 json_integer((json_int_t) buckets[i].active[p]));
340 }
341 }
342 json_object_set_new(b, "active_tpc", json_integer((json_int_t) buckets[i].active_tpc));
343 json_object_set_new(b, "bytes_rx", json_integer((json_int_t) buckets[i].bytes_rx));
344 json_object_set_new(b, "bytes_tx", json_integer((json_int_t) buckets[i].bytes_tx));
345 json_object_set_new(b, "errors", json_integer((json_int_t) buckets[i].errors));
346 json_object_set_new(b, "auth_failures", json_integer((json_int_t) buckets[i].auth_failures));
347 json_object_set_new(b, "write_stalls", json_integer((json_int_t) buckets[i].write_stalls));
348 json_object_set_new(b, "cache_occupancy_ppm", json_integer((json_int_t) buckets[i].cache_occupancy_ppm));
349 json_array_append_new(arr, b);
350 }
351 json_object_set_new(target, "buckets", arr);
352}

418 json_object_set_new(root, "history", history);
