7/*
8 * brix_cms_send_all — send a contiguous buffer over an nginx connection.
9 *
10 * WHAT: Loop c->send() until the entire buf is transmitted or an error occurs.
11 * WHY: nginx's c->send() may return NGX_AGAIN (partial write) or 0 on idle;
12 * the caller needs a single result indicating complete success, partial,
13 * or failure rather than polling the event loop manually.
14 * HOW: Accumulate sent bytes in a while-loop; each iteration advances buf+sent
15 * and reduces len-sent. Returns NGX_AGAIN when send returns 0 or NGX_AGAIN
16 * (caller must retry on next event), NGX_ERROR on NGX_ERROR, NGX_OK when
17 * sent == len.
18 */
19ngx_int_t
20brix_cms_send_all(ngx_connection_t *c, const u_char *buf, size_t len)
21{
22 size_t sent;
23
24 if (c == NULL) {
25 return NGX_ERROR;
26 }
27
28 sent = 0;
29 while (sent < len) {
30 ssize_t n;
31
32 n = c->send(c, (u_char *) buf + sent, len - sent);
33 if (n == NGX_AGAIN || n == 0) {
34 return NGX_AGAIN;
35 }
36
37 if (n == NGX_ERROR) {
38 return NGX_ERROR;
39 }
40
41 sent += (size_t) n;
42 }
43
44 return NGX_OK;
45}

47/*
48 * brix_cms_send_frame — build and transmit a CMS wire frame.
49 *
50 * WHAT: Construct an 8-byte big-endian header (streamid, code, modifier,
51 * payload_len) followed by the payload, then send both via
52 * brix_cms_send_all().
53 * WHY: The CMS manager protocol requires every message to carry a fixed-size
54 * header that encodes the target stream and frame type before the actual
55 * payload; this helper encapsulates header construction + sequential
56 * transmission so callers (send.c) emit frames with one function call.
57 * HOW: Validate c != NULL and payload_len <= 65535 (fits in uint16). Encode
58 * streamid via ngx_brix_cms_put32(), code/modifier as raw bytes,
59 * payload_len via ngx_brix_cms_put16(). Send header first; if
60 * payload_len > 0 send payload second. Returns NGX_OK on full success,
61 * NGX_ERROR on any brix_cms_send_all failure.
62 */
63ngx_int_t
64brix_cms_send_frame(ngx_connection_t *c, uint32_t streamid, u_char code,
65 u_char modifier, const u_char *payload, size_t payload_len)
66{
67 u_char hdr[NGX_BRIX_CMS_HDR_LEN];
68
69 if (c == NULL || payload_len > 65535) {
70 return NGX_ERROR;
71 }
72
73 ngx_brix_cms_put32(hdr, streamid);
74 hdr[4] = code;
75 hdr[5] = modifier;
76 ngx_brix_cms_put16(hdr + 6, (uint16_t) payload_len);
77
78 if (brix_cms_send_all(c, hdr, sizeof(hdr)) != NGX_OK) {
79 return NGX_ERROR;
80 }
81
82 if (payload_len > 0
83 && brix_cms_send_all(c, payload, payload_len) != NGX_OK)
84 {
85 return NGX_ERROR;
86 }
87
88 return NGX_OK;
89}
