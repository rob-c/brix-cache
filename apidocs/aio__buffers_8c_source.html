7int
8xbuf_reserve(xbuf *b, size_t need)
9{
10 if (b->len + need <= b->cap) {
11 return 0;
12 }
13 size_t ncap = (b->cap == 0) ? 4096 : b->cap;
14 while (ncap < b->len + need) {
15 ncap *= 2;
16 }
17 uint8_t *nb = (uint8_t *) realloc(b->buf, ncap);
18 if (nb == NULL) {
19 return -1;
20 }
21 b->buf = nb;
22 b->cap = ncap;
23 return 0;
24}

68/* request *//*
69 * One in-flight request. The finalized 24-byte header and the (owned) payload are
70 * retained so M2/M3 can re-issue the request after a reconnect. Reply bytes from

88areq_accumulate(brix_areq *r, const uint8_t *body, uint32_t n)
89{

91 return 0;

111/* Invoke the completion callback exactly once and free the request. On success the
112 * accumulated body ownership transfers to the callback; on failure body is NULL. */
113void
114areq_complete(brix_areq *r, int rc, uint16_t kxr, const brix_status *st)
115{
116 uint8_t *body = NULL;
117 uint32_t blen = 0;
118
119 if (rc == 0) {
120 body = r->acc;
121 blen = r->acc_len;
122 r->acc = NULL; /* ownership moves to the callback */
123 }
124 r->cb(r->ctx, rc, kxr, body, blen, st);
125 areq_free(r);

129/* reqmap *//*
130 * Open-addressing (linear-probe) map from streamid → in-flight request. The key
131 * space is dense and the live count is bounded by the pipeline depth, so this stays
132 * tiny and fast. Deleted slots are tombstoned and reclaimed on the next rehash.
133 */
134
135
136int
137reqmap_rehash(reqmap *m, uint32_t newcap)
138{
139 brix_areq **ns = (brix_areq **) calloc(newcap, sizeof(*ns));
140 if (ns == NULL) {
141 return -1;
142 }
143 for (uint32_t i = 0; i < m->cap; i++) {
144 brix_areq *r = m->slots[i];
145 if (r == NULL || r == REQMAP_TOMB) {
146 continue;
147 }
148 uint32_t idx = r->sid & (newcap - 1);
149 while (ns[idx] != NULL) {
150 idx = (idx + 1) & (newcap - 1);
151 }
152 ns[idx] = r;
153 }
154 free(m->slots);
155 m->slots = ns;
156 m->cap = newcap;
157 m->tomb = 0;
158 return 0;
159}

162int
163reqmap_put(reqmap *m, brix_areq *r)
164{
165 if (m->cap == 0) {
166 if (reqmap_rehash(m, 64) != 0) {
167 return -1;
168 }
169 }
170 if ((m->count + m->tomb + 1) * 4 >= m->cap * 3) {
171 uint32_t newcap = (m->count * 2 < m->cap) ? m->cap : m->cap * 2;
172 if (newcap < 64) {
173 newcap = 64;
174 }
175 if (reqmap_rehash(m, newcap) != 0) {
176 return -1;
177 }
178 }
179 uint32_t idx = r->sid & (m->cap - 1);
180 while (m->slots[idx] != NULL && m->slots[idx] != REQMAP_TOMB) {
181 idx = (idx + 1) & (m->cap - 1);
182 }
183 if (m->slots[idx] == REQMAP_TOMB) {
184 m->tomb--;
185 }
186 m->slots[idx] = r;
187 m->count++;
188 return 0;
189}
