1/*
2 * zip_kernel.c — shared ZIP central-directory parsing kernel. See zip_kernel.h.
3 *
4 * Pure C, callback-driven, fully bounds-checked. No nginx / OpenSSL / fixed I/O
5 * source so it can be unit-tested standalone and linked into both the module and
6 * the native client (libxrdproto). Every offset and length is validated against
7 * the archive size before use; a hostile or truncated archive yields a clean
8 * error, never an out-of-bounds read.
9 */

15/* ZIP record signatures (little-endian on disk). */
16#define ZIPK_EOCD_SIG 0x06054b50u /* End Of Central Directory */
17#define ZIPK_EOCD_BASE 22
18#define ZIPK_Z64LOC_SIG 0x07064b50u /* ZIP64 EOCD locator */
19#define ZIPK_Z64LOC_SIZE 20
20#define ZIPK_Z64EOCD_SIG 0x06064b50u /* ZIP64 EOCD */
21#define ZIPK_Z64EOCD_SZ 56
22#define ZIPK_LFH_SIG 0x04034b50u /* Local File Header */
23#define ZIPK_LFH_BASE 30
24
25#define ZIPK_U32_MAX 0xFFFFFFFFu
26#define ZIPK_U16_MAX 0xFFFFu
27
28#define ZIPK_MAX_COMMENT 65535 /* EOCD comment length cap (u16) */
29
30/* ---- little-endian field readers ---------------------------------------- */
31
32uint16_t zip_rd16le(const uint8_t *p)
33{
34 return (uint16_t) (p[0] | (p[1] << 8));
35}
36
37uint32_t zip_rd32le(const uint8_t *p)
38{
39 return (uint32_t) p[0] | ((uint32_t) p[1] << 8)
40 | ((uint32_t) p[2] << 16) | ((uint32_t) p[3] << 24);
41}
42
43uint64_t zip_rd64le(const uint8_t *p)
44{
45 return (uint64_t) zip_rd32le(p) | ((uint64_t) zip_rd32le(p + 4) << 32);
46}

48/* ---- bounds-checked archive read ---------------------------------------- */
49
50int
51zip_read_at(zip_pread_fn pread, void *ctx, uint64_t archive_size,
52 uint64_t off, void *buf, size_t len)
53{
54 ssize_t n;
55
56 if (len == 0) {
57 return ZIP_K_OK;
58 }
59 if (off > archive_size || (uint64_t) len > archive_size - off) {
60 return ZIP_K_ECORRUPT; /* out-of-bounds read attempt */
61 }
62 n = pread(ctx, off, buf, len);
63 if (n < 0 || (size_t) n != len) {
64 return ZIP_K_EIO;
65 }
66 return ZIP_K_OK;
67}

69/* ---- central-directory location (ZIP64-aware) --------------------------- */
70
71/*
72 * Promote the classic EOCD fields to their ZIP64 values when any of them is
73 * saturated. Returns ZIP_K_OK (promoted or no promotion needed) or
74 * ZIP_K_ECORRUPT (the ZIP64 locator / EOCD is malformed).
75 */
76static int
77zip_promote_zip64(zip_pread_fn pread, void *ctx, uint64_t archive_size,
78 uint64_t eocd_pos, uint64_t *cd_off, uint64_t *cd_size,
79 uint64_t *n_entries)
80{
81 uint8_t loc[ZIPK_Z64LOC_SIZE];
82 uint8_t z64[ZIPK_Z64EOCD_SZ];
83 uint64_t z64_off;
84 int rc;
85
86 if (*cd_off != ZIPK_U32_MAX && *cd_size != ZIPK_U32_MAX
87 && *n_entries != ZIPK_U16_MAX) {
88 return ZIP_K_OK; /* nothing saturated */
89 }
90 if (eocd_pos < ZIPK_Z64LOC_SIZE) {
91 return ZIP_K_ECORRUPT;
92 }
93
94 rc = zip_read_at(pread, ctx, archive_size, eocd_pos - ZIPK_Z64LOC_SIZE,
95 loc, sizeof(loc));
96 if (rc != ZIP_K_OK || zip_rd32le(loc) != ZIPK_Z64LOC_SIG) {
97 return ZIP_K_ECORRUPT;
98 }
99
100 z64_off = zip_rd64le(loc + 8);
101 rc = zip_read_at(pread, ctx, archive_size, z64_off, z64, sizeof(z64));
102 if (rc != ZIP_K_OK || zip_rd32le(z64) != ZIPK_Z64EOCD_SIG) {
103 return ZIP_K_ECORRUPT;
104 }
105
106 *n_entries = zip_rd64le(z64 + 32);
107 *cd_size = zip_rd64le(z64 + 40);
108 *cd_off = zip_rd64le(z64 + 48);
109 return ZIP_K_OK;
110}

112int
113zip_locate_cd(zip_pread_fn pread, void *ctx, uint64_t archive_size,
114 uint64_t cd_max, uint64_t max_entries,
115 uint64_t *cd_off, uint64_t *cd_size, uint64_t *n_entries)
116{
117 size_t tail_len;
118 uint64_t tail_off, eocd_pos = 0;
119 uint8_t *tail;
120 int found = 0, rc;
121
122 if (archive_size < ZIPK_EOCD_BASE) {
123 return ZIP_K_ENOTZIP;
124 }
125
126 /* The EOCD lives in the last 22 + comment(<=65535) bytes. */
127 tail_len = (size_t) ZIPK_EOCD_BASE + ZIPK_MAX_COMMENT;
128 if ((uint64_t) tail_len > archive_size) {
129 tail_len = (size_t) archive_size;
130 }
131 tail_off = archive_size - tail_len;
132
133 tail = malloc(tail_len);
134 if (tail == NULL) {
135 return ZIP_K_EIO;
136 }
137 rc = zip_read_at(pread, ctx, archive_size, tail_off, tail, tail_len);
138 if (rc != ZIP_K_OK) {
139 free(tail);
140 return rc;
141 }
142
143 /* Scan backward for the EOCD signature. */
144 for (size_t i = tail_len - ZIPK_EOCD_BASE + 1; i-- > 0; ) {
145 if (zip_rd32le(tail + i) == ZIPK_EOCD_SIG) {
146 eocd_pos = tail_off + i;
147 found = 1;
148 break;
149 }
150 }
151 if (!found) {
152 free(tail);
153 return ZIP_K_ENOTZIP;
154 }
155
156 {
157 const uint8_t *e = tail + (eocd_pos - tail_off);
158 *n_entries = zip_rd16le(e + 10);
159 *cd_size = zip_rd32le(e + 12);
160 *cd_off = zip_rd32le(e + 16);
161 }
162 free(tail);
163
164 rc = zip_promote_zip64(pread, ctx, archive_size, eocd_pos,
165 cd_off, cd_size, n_entries);
166 if (rc != ZIP_K_OK) {
167 return rc;
168 }
169
170 /* Anti-bomb caps + final bounds check against the archive. */
171 if ((cd_max != 0 && *cd_size > cd_max)
172 || (max_entries != 0 && *n_entries > max_entries)) {
173 return ZIP_K_ECORRUPT;
174 }
175 if (*cd_off > archive_size || *cd_size > archive_size - *cd_off) {
176 return ZIP_K_ECORRUPT;
177 }
178 return ZIP_K_OK;
179}

181/* ---- CDFH ZIP64 extra-field decode -------------------------------------- */
182
183void
184zip_apply_zip64_extra(const uint8_t *extra, size_t extra_len,
185 uint64_t *uncomp, uint64_t *comp, uint64_t *lhdr_off)
186{
187 size_t p = 0;
188
189 while (p + 4 <= extra_len) {
190 uint16_t id = zip_rd16le(extra + p);
191 uint16_t len = zip_rd16le(extra + p + 2);
192 if ((size_t) len > extra_len - p - 4) {
193 return; /* malformed extra → stop */
194 }
195 if (id == 0x0001) { /* ZIP64 extended info */
196 const uint8_t *d = extra + p + 4;
197 size_t q = 0;
198 if (*uncomp == ZIPK_U32_MAX && len - q >= 8) {
199 *uncomp = zip_rd64le(d + q); q += 8;
200 }
201 if (*comp == ZIPK_U32_MAX && len - q >= 8) {
202 *comp = zip_rd64le(d + q); q += 8;
203 }
204 if (*lhdr_off == ZIPK_U32_MAX && len - q >= 8) {
205 *lhdr_off = zip_rd64le(d + q); q += 8;
206 }
207 return;
208 }
209 p += 4 + len;
210 }
211}

215int
216zip_resolve_data_off(zip_pread_fn pread, void *ctx, uint64_t archive_size,
217 uint64_t lhdr_off, uint64_t comp_size, uint64_t *data_off)
218{
219 uint8_t lfh[ZIPK_LFH_BASE];
220 uint64_t off;
221 int rc;
222
223 rc = zip_read_at(pread, ctx, archive_size, lhdr_off, lfh, sizeof(lfh));
224 if (rc != ZIP_K_OK) {
225 return rc;
226 }
227 if (zip_rd32le(lfh) != ZIPK_LFH_SIG) {
228 return ZIP_K_ECORRUPT;
229 }
230
231 off = lhdr_off + ZIPK_LFH_BASE
232 + zip_rd16le(lfh + 26) /* LFH filename length */
233 + zip_rd16le(lfh + 28); /* LFH extra length */
234
235 if (off > archive_size || comp_size > archive_size - off) {
236 return ZIP_K_ECORRUPT;
237 }
238 *data_off = off;
239 return ZIP_K_OK;
240}
