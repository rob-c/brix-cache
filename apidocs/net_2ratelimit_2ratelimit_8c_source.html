1/*
2 * ratelimit.c — Phase 25 leaky-bucket core (request rate + bandwidth).
3 *
4 * The leaky-bucket formulation matches ngx_http_limit_req_module: a bucket
5 * fills with each charged unit and drains at the configured rate over elapsed
6 * wall-clock. req_excess is stored × 1000 so a 1-request charge is 1000 units
7 * and fractional drain over millisecond intervals stays integer. Bandwidth
8 * uses the same shape but charges raw byte counts (× 1) and drains at bw_rate
9 * bytes/s.
10 *
11 * Fail-open: any internal failure (zone not yet attached, slab exhausted after
12 * eviction) returns NGX_OK so rate limiting never denies otherwise-authorised
13 * access — availability beats strict enforcement.
14 */
15#include "ratelimit.h"

18ngx_int_t
19brix_rl_check(brix_rl_rule_t *rule, const char *key_str,
20 uint32_t *wait_seconds)
21{
22 brix_rl_zone_t *zone = rule->zone;
23 brix_rl_node_t *rln;
24 uint32_t hash;
25 size_t len;
26 ngx_msec_t now, elapsed;
27 ngx_uint_t excess, drain, burst_units;
28 ngx_int_t rc = NGX_OK;
29
30 if (rule->req_rate == 0 || zone == NULL || zone->sh == NULL) {
31 return NGX_OK;
32 }
33
34 len = ngx_strlen(key_str);
35 hash = brix_rl_hash(key_str, len);
36 now = ngx_current_msec;
37
38 ngx_shmtx_lock(&zone->shpool->mutex);
39
40 rln = brix_rl_lookup_locked(zone, hash, key_str, len);
41 if (rln == NULL) {
42 rln = brix_rl_create_locked(zone, hash, key_str, len);
43 if (rln == NULL) {
44 ngx_shmtx_unlock(&zone->shpool->mutex);
45 return NGX_OK; /* fail open */
46 }
47 rln->last = now;
48 }
49
50 elapsed = (now > rln->last) ? (now - rln->last) : 0;
51 drain = (ngx_uint_t) ((uint64_t) rule->req_rate * elapsed / 1000);
52 excess = (rln->req_excess > drain) ? (rln->req_excess - drain) : 0;
53
54 /* burst headroom in the same ×1000 unit as req_rate / excess. */
55 burst_units = rule->req_burst * BRIX_RL_REQ_SCALE;
56
57 if (excess + BRIX_RL_REQ_SCALE > burst_units) {
58 /* Throttled — seconds to drain back under the burst ceiling (ceil). */
59 ngx_uint_t over = excess + BRIX_RL_REQ_SCALE - burst_units;
60 *wait_seconds = (uint32_t) ((over + rule->req_rate - 1) / rule->req_rate);
61 if (*wait_seconds == 0) { *wait_seconds = 1; }
62 /*
63 * Persist the DRAINED excess (without the +1000 charge) and advance the
64 * clock together. Omitting this — writing rln->last = now but leaving
65 * rln->req_excess at its pre-drain value — discards the drain that
66 * accrued since the last request, so under sustained offered load the
67 * bucket never actually drains: every throttle resets the clock against
68 * a stale, pegged excess and the effective serve rate collapses far
69 * below the configured rate. Writing the drained value back is what
70 * lets the bucket drain at exactly req_rate between accepted requests.
71 */
72 rln->req_excess = excess;
73 rln->throttle_count++;
74 rln->last = now;
75 rc = NGX_AGAIN;
76 } else {
77 rln->req_excess = excess + BRIX_RL_REQ_SCALE;
78 rln->last = now;
79 rln->req_total++;
80 }
81
82 ngx_shmtx_unlock(&zone->shpool->mutex);
83 return rc;
84}

128void
129brix_rl_conc_release(brix_rl_rule_t *rule, const char *key_str)
130{
131 brix_rl_zone_t *zone = rule->zone;
132 brix_rl_node_t *rln;
133 uint32_t hash;
134 size_t len;
135
136 if (rule->req_conc == 0 || zone == NULL || zone->sh == NULL) {
137 return;
138 }
139
140 len = ngx_strlen(key_str);
141 hash = brix_rl_hash(key_str, len);
142
143 /*
144 * Fresh lookup under the lock — never dereference a cached node pointer,
145 * since the LRU reaper may have evicted (freed) the node between acquire
146 * and release. If the node is gone the slot accounting simply resets, which
147 * is a benign accuracy drift only possible under memory pressure.
148 */
149 ngx_shmtx_lock(&zone->shpool->mutex);
150 rln = brix_rl_lookup_locked(zone, hash, key_str, len);
151 if (rln != NULL && rln->in_flight > 0) {
152 rln->in_flight--;
153 }
154 ngx_shmtx_unlock(&zone->shpool->mutex);
155}

194 if (*wait_seconds == 0) { *wait_seconds = 1; }
195 rln->throttle_count++;
196 rc = NGX_AGAIN;
197 }
198
199 ngx_shmtx_unlock(&zone->shpool->mutex);

247/* dashboard snapshot */
248ngx_int_t
249brix_rl_snapshot(brix_rl_zone_t *zone, brix_rl_snapshot_entry_t *out,
250 ngx_uint_t max, ngx_uint_t *count)
251{
252 ngx_queue_t *q;
253 brix_rl_node_t *rln;
254 ngx_uint_t n = 0;
255
256 *count = 0;
257 if (zone == NULL || zone->sh == NULL) {
258 return NGX_OK;
259 }
260
261 ngx_shmtx_lock(&zone->shpool->mutex);
262
263 /* Walk the LRU queue (most-recent first) and copy out under the lock. */
264 for (q = ngx_queue_head(&zone->sh->queue);
265 q != ngx_queue_sentinel(&zone->sh->queue) && n < max;
266 q = ngx_queue_next(q))
267 {
268 rln = ngx_queue_data(q, brix_rl_node_t, queue);
269
270 ngx_memcpy(out[n].key_str, rln->key_str,
271 ngx_min(rln->len, BRIX_RL_KEY_LEN - 1));
272 out[n].key_str[ngx_min(rln->len, BRIX_RL_KEY_LEN - 1)] = '\0';
273 out[n].req_total = rln->req_total;
274 out[n].bytes_total = rln->bytes_total;
275 out[n].throttle_count = rln->throttle_count;
276 out[n].req_excess = rln->req_excess;
277 out[n].bw_excess = rln->bw_excess;
278 n++;
279 }
280
281 ngx_shmtx_unlock(&zone->shpool->mutex);
282
283 /* Sort by throttle_count descending (small N — insertion sort). */
284 {
285 ngx_uint_t i, j;
286 for (i = 1; i < n; i++) {
287 brix_rl_snapshot_entry_t tmp = out[i];
288 for (j = i; j > 0 && out[j - 1].throttle_count < tmp.throttle_count;
289 j--)
290 {
291 out[j] = out[j - 1];
292 }
293 out[j] = tmp;
294 }
295 }
296
297 *count = n;
298 return NGX_OK;
299}
