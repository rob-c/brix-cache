23/*
24 * brix_proxy_splice_done — called when all splice_total bytes have been
25 * moved from the upstream socket to the client socket. Mirrors the
26 * post-relay accounting that relay_to_client() does for reads.
27 */
28static void
29brix_proxy_splice_done(brix_proxy_ctx_t *proxy)
30{
31 brix_ctx_t *ctx = proxy->client_ctx;
32 ngx_connection_t *c = proxy->client_conn;
33 uint16_t status = proxy->resp_status;
34 size_t dlen = proxy->splice_total;
35 int lfh = proxy->fwd_local_fh;
36
37 /* Restore the normal client write handler (may have been changed for EAGAIN). */
38 c->write->handler = ngx_stream_brix_send;
39
40 /* Per-handle and aggregate byte metrics. */
41 if (lfh >= 0 && lfh < BRIX_MAX_FILES) {
42 proxy->fh_map[lfh].bytes_read += dlen;
43 }
44 BRIX_PROXY_METRIC_INC(ctx, reads_total);
45 BRIX_PROXY_METRIC_ADD(ctx, read_bytes_total, dlen);
46 BRIX_PROXY_UP_INC(proxy, reads_total);
47 BRIX_PROXY_UP_ADD(proxy, read_bytes_total, dlen);
48
49 /* Reset accumulator for the next response. */
50 proxy->splice_active = 0;
51 proxy->splice_fallback = 0;
52 proxy->splice_total = 0;
53 proxy->splice_upstream = 0;
54 proxy->splice_downstream = 0;
55 proxy->rhdr_pos = 0;
56 proxy->resp_dlen = 0;
57 proxy->resp_body = NULL;
58 proxy->resp_body_pos = 0;
59
60 if (status == kXR_oksofar) {
61 proxy->fwd_streaming = 1;
62 /*
63 * nginx uses edge-triggered epoll. The upstream socket may already
64 * have the trailing kXR_ok(0) frame buffered, but no new edge event
65 * will fire for it. Re-arm even when nginx still marks the event
66 * active: after a splice pump the event can be active-but-stale, which
67 * otherwise leaves the next oksofar/ok frame buffered forever.
68 * Then post the handler so it runs again regardless of whether new data
69 * has arrived.
70 */
71 {
72 ngx_event_t *urev = proxy->conn->read;
73 if (ngx_handle_read_event(urev, 0) != NGX_OK) {
74 brix_proxy_abort(proxy,
75 "proxy: read arm failed after splice oksofar");
76 return;
77 }
78 if (proxy->conf != NULL && proxy->conf->proxy.read_timeout > 0) {
79 ngx_add_timer(urev, proxy->conf->proxy.read_timeout);
80 }
81 if (!urev->posted) {
82 ngx_post_event(urev, &ngx_posted_events);
83 }
84 }
85 return;
86 }
87
88 /* Final response — hand control back to the client loop. */
89 proxy->state = XRD_PX_IDLE;
90 ctx->state = XRD_ST_REQ_HEADER;
91 brix_schedule_read_resume(c);
92}

94/*
95 * Pump-stage outcome — steers the pump loop after a fill/drain stage.
96 * BRIX_SPLICE_STEP_CONTINUE — stage made progress; re-run the loop body.
97 * BRIX_SPLICE_STEP_DONE — all splice_total bytes transferred; finish.
98 * BRIX_SPLICE_STEP_RETURN — stage armed an event or aborted; pump returns
99 * immediately without touching `proxy`.
100 */
101typedef enum {
102 BRIX_SPLICE_STEP_CONTINUE = 0,
103 BRIX_SPLICE_STEP_DONE,
104 BRIX_SPLICE_STEP_RETURN
105} brix_splice_step_e;
106
107/*
108 * brix_proxy_splice_fill — WHAT: move upstream-socket bytes into pipe[1].
109 * WHY: the pipe must hold data before it can be drained to the client; filling
110 * only when the pipe is empty keeps splice(upstream→pipe) EAGAIN meaningful
111 * (upstream readiness, never pipe-capacity). HOW: splice at most the residual
112 * body; on EAGAIN with body still outstanding hand off to the buffered relay,
113 * else arm upstream-read; on 0/error abort. Returns a pump-step outcome.
114 */
115static brix_splice_step_e
116brix_proxy_splice_fill(brix_proxy_ctx_t *proxy, ngx_connection_t *uconn)
117{
118 size_t in_pipe_now;
119 size_t want;
120 ssize_t r;
121
122 if (proxy->splice_upstream >= proxy->splice_total) {
123 return BRIX_SPLICE_STEP_CONTINUE;
124 }
125
126 in_pipe_now = proxy->splice_upstream - proxy->splice_downstream;
127 if (in_pipe_now != 0) {
128 /*
129 * Pipe still holds data (full) — attempting splice(upstream→pipe) would
130 * return EAGAIN for the wrong reason (pipe capacity, not upstream
131 * readiness) and arm the upstream-read event prematurely. Fall through
132 * and drain the pipe to the client first.
133 */
134 return BRIX_SPLICE_STEP_CONTINUE;
135 }
136
137 want = proxy->splice_total - proxy->splice_upstream;
138 r = splice(uconn->fd, NULL, proxy->splice_pipe[1], NULL, want,
139 SPLICE_F_MOVE | SPLICE_F_NONBLOCK);
140
141 if (r > 0) {
142 proxy->splice_upstream += (size_t) r;
143 return BRIX_SPLICE_STEP_CONTINUE;
144 }
145
146 if (r == 0) {
147 /* Upstream closed during splice (peer sent FIN). */
148 brix_proxy_abort(proxy, "proxy: upstream closed during splice");
149 return BRIX_SPLICE_STEP_RETURN;
150 }
151
152 if (ngx_errno != NGX_EAGAIN) {
153 brix_proxy_abort(proxy, "proxy: splice upstream→pipe failed");
154 return BRIX_SPLICE_STEP_RETURN;
155 }
156
157 /*
158 * The pump always drains everything currently in the upstream socket
159 * (looping pipe-fill/pipe-drain) before splice(upstream→pipe) reports
160 * EAGAIN. So reaching EAGAIN with body still outstanding means the
161 * remainder has not all arrived yet (or the kernel under-drains socket
162 * splice, as WSL2 does — moving a trickle per call and stalling a large
163 * read into a 60s client timeout). Either way the reliable, edge-efficient
164 * choice is to relay the remaining body via the buffered recv path, which
165 * drains the whole socket buffer per wakeup as data arrives. (When the
166 * entire body was already buffered, the pump finishes via splice_done and
167 * never reaches here — so the zero-copy fast path is preserved for the
168 * in-buffer case.)
169 */
170 if (proxy->splice_downstream < proxy->splice_total) {
171 brix_proxy_splice_to_buffered(proxy);
172 return BRIX_SPLICE_STEP_RETURN;
173 }
174
175 /* Body complete — nothing left; arm read and let the loop end. */
176 if (ngx_handle_read_event(uconn->read, 0) != NGX_OK) {
177 brix_proxy_abort(proxy, "proxy: splice read arm failed");
178 }
179 return BRIX_SPLICE_STEP_RETURN;
180}

182/*
183 * brix_proxy_splice_drain — WHAT: move pipe[0] bytes into the client socket.
184 * WHY: this is the client-facing half of the transfer; a full client send
185 * buffer must park on the write event, NOT arm upstream-read (that just spins
186 * on a full pipe and deadlocks). HOW: if the pipe is empty, either the body is
187 * done or we loop back to refill; on EAGAIN/0 arm the client-write handler; on
188 * error abort. Returns a pump-step outcome.
189 */
190static brix_splice_step_e
191brix_proxy_splice_drain(brix_proxy_ctx_t *proxy, ngx_connection_t *cconn)
192{
193 size_t in_pipe = proxy->splice_upstream - proxy->splice_downstream;
194 ssize_t r;
195
196 if (in_pipe == 0) {
197 if (proxy->splice_downstream >= proxy->splice_total) {
198 return BRIX_SPLICE_STEP_DONE;
199 }
200 return BRIX_SPLICE_STEP_CONTINUE;
201 }
202
203 r = splice(proxy->splice_pipe[0], NULL, cconn->fd, NULL, in_pipe,
204 SPLICE_F_MOVE | SPLICE_F_NONBLOCK);
205
206 if (r > 0) {
207 proxy->splice_downstream += (size_t) r;
208 if (proxy->splice_downstream >= proxy->splice_total) {
209 return BRIX_SPLICE_STEP_DONE;
210 }
211 return BRIX_SPLICE_STEP_CONTINUE;
212 }
213
214 if (r == 0 || ngx_errno == NGX_EAGAIN) {
215 /*
216 * splice(pipe→client) made no progress. Two very different causes:
217 *
218 * a) The client send buffer is genuinely full — the write event is not
219 * ready. Arm the client-write handler and wait for the drain edge.
220 * Do NOT arm upstream-read here: any unread upstream data sits in the
221 * kernel TCP receive buffer and is drained by splice_pump (via
222 * splice(upstream→pipe)) once the pipe has room — no epoll wakeup is
223 * needed. Arming upstream-read while the pipe is full just spins on a
224 * full pipe and deadlocks.
225 *
226 * b) The client socket IS writable (write->ready) yet splice still
227 * returned EAGAIN — the kernel's socket-splice under-drains (observed
228 * on WSL2: pipe→socket refuses a tiny residual even though the socket
229 * has room). No not-writable→writable transition will follow, so an
230 * armed edge-triggered write event would never fire and the pipe
231 * residual would stall into the read timeout. Drain the residual into
232 * userspace and relay it via the reliable buffered path instead.
233 */
234 if (cconn->write->ready) {
235 brix_proxy_splice_drain_to_buffered(proxy);
236 return BRIX_SPLICE_STEP_RETURN;
237 }
238 cconn->write->handler = brix_proxy_splice_wev;
239 if (ngx_handle_write_event(cconn->write, 0) != NGX_OK) {
240 brix_proxy_abort(proxy, "proxy: splice client write arm failed");
241 }
242 return BRIX_SPLICE_STEP_RETURN;
243 }
244
245 brix_proxy_abort(proxy, "proxy: splice pipe→client failed");
246 return BRIX_SPLICE_STEP_RETURN;
247}

249/*
250 * brix_proxy_splice_pump — drive splice transfers between upstream socket
251 * and client socket via the kernel pipe in proxy->splice_pipe[].
252 *
253 * Called from:
254 * brix_proxy_read_handler — upstream socket is readable
255 * brix_proxy_splice_wev — client socket is writable (drain pipe)
256 */
257void
258brix_proxy_splice_pump(brix_proxy_ctx_t *proxy)
259{
260 ngx_connection_t *uconn = proxy->conn;
261 ngx_connection_t *cconn = proxy->client_conn;
262
263 if (uconn == NULL || cconn == NULL) {
264 return;
265 }
266
267 for (;;) {
268 brix_splice_step_e step = brix_proxy_splice_fill(proxy, uconn);
269 if (step == BRIX_SPLICE_STEP_RETURN) {
270 return;
271 }
272 /* fill never signals DONE — only drain does. */
273
274 step = brix_proxy_splice_drain(proxy, cconn);
275 if (step == BRIX_SPLICE_STEP_RETURN) {
276 return;
277 }
278 if (step == BRIX_SPLICE_STEP_DONE) {
279 break;
280 }
281 }
282
283 brix_proxy_splice_done(proxy);
284}

287 * brix_proxy_splice_wev — client write event handler during splice.
288 * Replaces ngx_stream_brix_send temporarily while draining the pipe.
289 */
290static void
291brix_proxy_splice_wev(ngx_event_t *wev)
292{

294 brix_ctx_t *ctx = cconn->data;
295 brix_proxy_ctx_t *proxy;
296

298 return;
299 }

306 }

311/*
312 * brix_proxy_splice_to_buffered — switch an under-draining splice transfer to
313 * the reliable buffered recv relay for the REMAINDER of the current body.
314 *
315 * Called from the pump when splice(upstream→pipe) reports EAGAIN but a MSG_PEEK
316 * shows data is still queued (a kernel whose socket-splice under-drains). At this
317 * point the pipe is empty (splice_upstream == splice_downstream), the 8-byte
318 * response header and splice_downstream body bytes are already on the wire, so the
319 * remaining splice_total − splice_downstream bytes are accumulated via the normal
320 * body loop and relayed RAW (no second header) by brix_proxy_splice_fallback_finish.
321 */
322static void
323brix_proxy_splice_to_buffered(brix_proxy_ctx_t *proxy)
324{
325 size_t remaining = proxy->splice_total - proxy->splice_downstream;
326
327 /*
328 * Bound the buffer: the buffered path holds the remainder in one allocation,
329 * so cap it at the same ceiling the normal relay uses. An oversized remainder
330 * keeps splicing (arm + wait) — slow on a broken kernel, but never unbounded
331 * memory. Real reads are far below this; this is belt-and-braces.
332 */
333 if (remaining == 0 || remaining > BRIX_PROXY_MAX_BODY) {
334 if (ngx_handle_read_event(proxy->conn->read, 0) != NGX_OK) {
335 brix_proxy_abort(proxy, "proxy: splice read arm failed");
336 }
337 return;
338 }
339
340 proxy->resp_body = ngx_alloc(remaining + 1, proxy->client_conn->log);
341 if (proxy->resp_body == NULL) {
342 brix_proxy_abort(proxy, "proxy: splice fallback body alloc failed");
343 return;
344 }
345 proxy->resp_body[remaining] = '\0';
346 proxy->resp_dlen = (uint32_t) remaining; /* remaining body only */
347 proxy->resp_body_pos = 0;
348 proxy->splice_active = 0;
349 proxy->splice_fallback = 1;
350
351 /* Latch: this upstream's kernel under-drains socket-splice. Re-attempting the
352 * splice fast-path on every subsequent read just repeats the wasted trickle-
353 * splice + buffered-handoff dance (and its rare lost-wakeup stall into the
354 * proxy_read_timeout). Once latched, try_splice declines and every later read
355 * goes straight to the reliable buffered relay. Log only on the first latch. */
356 ngx_log_error(NGX_LOG_NOTICE, proxy->client_conn->log, 0,
357 "xrootd proxy: upstream splice under-draining (%uz/%uz body sent) — "
358 "relaying the remaining %uz bytes via the buffered path",
359 proxy->splice_downstream, proxy->splice_total, remaining);
360
361 /* Drive the body accumulation in the read handler: arm + post the read
362 * event (the queued data produced no fresh edge, so post explicitly). */
363 if (ngx_handle_read_event(proxy->conn->read, 0) != NGX_OK) {
364 brix_proxy_abort(proxy, "proxy: splice fallback read arm failed");
365 return;
366 }
367 if (!proxy->conn->read->posted) {
368 ngx_post_event(proxy->conn->read, &ngx_posted_events);
369 }
370}

372/*
373 * brix_proxy_splice_drain_to_buffered — the drain-side counterpart of
374 * brix_proxy_splice_to_buffered. Called when splice(pipe→client) reports EAGAIN
375 * while the client socket is writable — the kernel's pipe→socket splice
376 * under-drains (observed on WSL2: it refuses even a tiny residual with room in
377 * the socket). An armed edge-triggered write event would never fire (no
378 * not-writable→writable transition follows), stalling the pipe residual into the
379 * read timeout. Instead, read() the pipe residual (already fetched from the
380 * upstream socket) into the front of a heap buffer, recv any still-in-socket
381 * remainder via the normal body loop, and relay the whole tail RAW through
382 * brix_proxy_splice_fallback_finish — the same machinery the fill-side fallback
383 * uses. The 8-byte header and splice_downstream body bytes are already on the
384 * wire, so no second header is emitted.
385 */
386static void
387brix_proxy_splice_drain_to_buffered(brix_proxy_ctx_t *proxy)
388{
389 size_t in_pipe = proxy->splice_upstream - proxy->splice_downstream;
390 size_t remaining = proxy->splice_total - proxy->splice_downstream;
391 size_t pos = 0;
392
393 /* downstream == total is the drain DONE path; guard defensively. */
394 if (remaining == 0) {
395 brix_proxy_splice_done(proxy);
396 return;
397 }
398
399 /*
400 * An oversized remainder cannot be held in one buffered allocation. Fall
401 * back to the write-event wait (correct for genuine backpressure). The
402 * writable-but-under-draining stall is confined to small trailing residuals
403 * far below this ceiling, so this branch is belt-and-braces.
404 */
405 if (remaining > BRIX_PROXY_MAX_BODY) {
406 proxy->client_conn->write->handler = brix_proxy_splice_wev;
407 if (ngx_handle_write_event(proxy->client_conn->write, 0) != NGX_OK) {
408 brix_proxy_abort(proxy, "proxy: splice client write arm failed");
409 }
410 return;
411 }
412
413 proxy->resp_body = ngx_alloc(remaining + 1, proxy->client_conn->log);
414 if (proxy->resp_body == NULL) {
415 brix_proxy_abort(proxy, "proxy: splice drain fallback body alloc failed");
416 return;
417 }
418
419 /*
420 * Move the pipe residual — bytes already spliced out of the upstream socket —
421 * into the front of the buffer. The pipe is known to hold exactly in_pipe
422 * bytes, so a non-blocking read() returns them; loop only to absorb a short
423 * read or EINTR. Anything else breaks the pipe accounting invariant and
424 * would corrupt the frame, so abort rather than relay garbage.
425 */
426 while (pos < in_pipe) {
427 ssize_t got = read(proxy->splice_pipe[0], proxy->resp_body + pos,
428 in_pipe - pos);
429 if (got > 0) {
430 pos += (size_t) got;
431 continue;
432 }
433 if (got < 0 && ngx_errno == NGX_EINTR) {
434 continue;
435 }
436 ngx_free(proxy->resp_body);
437 proxy->resp_body = NULL;
438 brix_proxy_abort(proxy, "proxy: splice pipe drain read failed");
439 return;
440 }
441
442 proxy->resp_body[remaining] = '\0';
443 proxy->resp_dlen = (uint32_t) remaining;
444 proxy->resp_body_pos = pos; /* pipe residual already buffered */
445 proxy->splice_active = 0;
446 proxy->splice_fallback = 1;
447
448 /*
449 * Restore the normal client write handler: the buffered relay
450 * (fallback_finish → brix_queue_response_base) drives the client send through
451 * ngx_stream_brix_send, and it must own any genuine-backpressure write event.
452 */
453 proxy->client_conn->write->handler = ngx_stream_brix_send;
454
455 ngx_log_error(NGX_LOG_NOTICE, proxy->client_conn->log, 0,
456 "xrootd proxy: client splice under-draining (%uz/%uz body sent, socket "
457 "writable) — relaying the remaining %uz bytes via the buffered path",
458 proxy->splice_downstream, proxy->splice_total, remaining);
459
460 /*
461 * Recv any still-in-socket remainder in the read handler, then fallback_finish
462 * sends the whole tail. The buffered residual produced no fresh edge, so post
463 * the read event explicitly; when the socket remainder is zero the posted
464 * handler dispatches straight to fallback_finish.
465 */
466 if (ngx_handle_read_event(proxy->conn->read, 0) != NGX_OK) {
467 brix_proxy_abort(proxy, "proxy: splice drain fallback read arm failed");
468 return;
469 }
470 if (!proxy->conn->read->posted) {
471 ngx_post_event(proxy->conn->read, &ngx_posted_events);
472 }
473}

475/*
476 * brix_proxy_splice_fallback_finish — complete a splice→buffered fallback once
477 * the remaining body has been accumulated into resp_body by the read handler.
478 * Sends those bytes RAW to the client (the header is already on the wire), then
479 * runs the same post-transfer accounting/finish as a fully-spliced response.
480 */
481void
482brix_proxy_splice_fallback_finish(brix_proxy_ctx_t *proxy)
483{
484 brix_ctx_t *ctx = proxy->client_ctx;
485 u_char *body = proxy->resp_body;
486 size_t n = proxy->resp_dlen;
487
488 /* Hand ownership of the heap body to the send path (freed after it drains). */
489 proxy->resp_body = NULL;
490 proxy->splice_fallback = 0;
491
492 if (brix_queue_response_base(ctx, proxy->client_conn, body, n, body)
493 != NGX_OK)
494 {
495 brix_proxy_abort(proxy, "proxy: splice fallback relay failed");
496 return;
497 }
498
499 /* The whole body (header + spliced prefix + buffered remainder) is now on the
500 * wire — account and finish exactly as a fully-spliced response would. */
501 proxy->splice_downstream = proxy->splice_total;
502 brix_proxy_splice_done(proxy);
503}

505/*
506 * brix_proxy_splice_eligible — WHAT: gate the zero-copy splice fast-path for
507 * the current response. WHY: splice bypasses TLS and is safe only for a final
508 * fully-buffered read/pgread body; multi-frame oksofar streams need the buffered
509 * relay's ordinary read-loop/event ordering. HOW: reject TLS, wrong
510 * state/opcode/status, empty body, and (via FIONREAD) a body not yet fully in
511 * the upstream socket buffer. Returns 1 if eligible.
512 */
513static int
514brix_proxy_splice_eligible(brix_proxy_ctx_t *proxy)
515{
516 int avail = 0;
517
518#if (NGX_SSL)
519 /* Splice bypasses the TLS layer — only valid for plain-text connections. */
520 if (proxy->conn->ssl != NULL || proxy->client_conn->ssl != NULL) {
521 return 0;
522 }
523#endif
524
525 if (proxy->state != XRD_PX_FORWARDING) {
526 return 0;
527 }
528 if (proxy->fwd_reqid != kXR_read && proxy->fwd_reqid != kXR_pgread) {
529 return 0;
530 }
531 if (proxy->resp_status != kXR_ok) {
532 return 0;
533 }
534 if (proxy->resp_dlen == 0) {
535 return 0;
536 }
537
538 /*
539 * Only splice a body that has ALREADY fully arrived in the upstream socket
540 * buffer (the 8-byte header was just consumed, so FIONREAD now reports body
541 * bytes). Splicing a still-streaming body makes the pump hit a mid-transfer
542 * EAGAIN and hand the remainder to the buffered relay — a fragile transition
543 * whose epoll-ET re-arm is unreliable after splice() drains the socket
544 * (observed as rare multi-second lost-wakeup stalls into proxy_read_timeout
545 * over proxy→proxy hops). When the whole body is buffered, splice completes
546 * in one pump with no handoff; otherwise the caller's buffered relay reads the
547 * streamed body reliably (it drains the socket per wakeup and is plenty fast).
548 * This keeps zero-copy for the fully-buffered common case while removing the
549 * stall-prone streaming-splice handoff entirely.
550 */
551 if (ioctl(proxy->conn->fd, FIONREAD, &avail) != 0
552 || (size_t) avail < (size_t) proxy->resp_dlen) {
553 return 0;
554 }
555
556 return 1;
557}

559/*
560 * brix_proxy_splice_pipe_ensure — WHAT: lazily create the kernel pipe used as
561 * the splice conduit. WHY: the pipe is per-proxy and only needed on the splice
562 * path; a larger pipe cuts syscalls on big bodies. HOW: pipe2(O_NONBLOCK) once,
563 * best-effort F_SETPIPE_SZ(1 MiB). Returns NGX_OK, or NGX_DECLINED if the pipe
564 * could not be created (caller falls back to the buffered path).
565 */
566static ngx_int_t
567brix_proxy_splice_pipe_ensure(brix_proxy_ctx_t *proxy)
568{
569 if (proxy->splice_pipe[0] != -1) {
570 return NGX_OK;
571 }
572
573 if (pipe2(proxy->splice_pipe, O_NONBLOCK) < 0) {
574 ngx_log_error(NGX_LOG_WARN, proxy->client_conn->log, ngx_errno,
575 "xrootd proxy: pipe2 failed, using buffered path");
576 return NGX_DECLINED;
577 }
578
579 /*
580 * Enlarge the pipe to 1 MiB (default is 64 KiB). splice() pumps at most
581 * one pipe-buffer per syscall, so a larger pipe cuts the syscall count on
582 * big relayed bodies by ~16x. Best-effort: F_SETPIPE_SZ can fail if it
583 * exceeds /proc/sys/fs/pipe-max-size for an unprivileged process, in which
584 * case the kernel keeps the default size and splice still works correctly.
585 */
586 if (fcntl(proxy->splice_pipe[1], F_SETPIPE_SZ, 1 << 20) < 0) {

588 ngx_errno,

590 "using default pipe size", ngx_errno);
591 }

593 return NGX_OK;
594}

596/*
597 * brix_proxy_splice_send_hdr — WHAT: send the full 8-byte response header (with
598 * the client's stream ID) before splicing the body. WHY (Phase 39 / PXY-6):
599 * once ANY header byte is on the wire we must not fall back to the buffered
600 * relay — it rebuilds and re-sends the WHOLE header, duplicating bytes and
601 * corrupting the client frame stream. HOW: a bounded send loop —
602 * - nothing sent yet (NGX_AGAIN at off==0): decline → buffered path sends it;
603 * - partial: complete the (<=7-byte) remainder here (bounded, off caps at 8);
604 * - NGX_AGAIN mid-remainder or a socket error: abort cleanly (NGX_ERROR).
605 * Returns NGX_OK (fully sent), NGX_DECLINED (nothing sent), or NGX_ERROR
606 * (aborted — caller must return immediately without touching `proxy`).
607 */
608static ngx_int_t
609brix_proxy_splice_send_hdr(brix_proxy_ctx_t *proxy)
610{
611 u_char hdr[XRD_RESPONSE_HDR_LEN];
612 size_t off = 0;
613 ssize_t sent;
614
615 brix_build_resp_hdr(proxy->fwd_streamid,
616 proxy->resp_status,
617 proxy->resp_dlen,
618 (ServerResponseHdr *)(void *) hdr);
619
620 for ( ;; ) {
621 sent = proxy->client_conn->send(proxy->client_conn,
622 hdr + off,
623 XRD_RESPONSE_HDR_LEN - off);
624 if (sent > 0) {
625 off += (size_t) sent;
626 if (off == XRD_RESPONSE_HDR_LEN) {
627 return NGX_OK; /* fully sent — proceed to splice */
628 }
629 continue; /* partial — send the remainder */
630 }
631 if (sent == NGX_AGAIN && off == 0) {
632 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, proxy->client_conn->log, 0,
633 "xrootd proxy: splice header would block, "
634 "using buffered path");
635 return NGX_DECLINED;
636 }
637 /* NGX_AGAIN after a partial header, or a socket error: cannot fall
638 * back without duplicating on-wire bytes, and the splice path has no
639 * deferred-header state. Abort rather than corrupt. */
640 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, proxy->client_conn->log, 0,
641 "xrootd proxy: splice header send incomplete (%z), "
642 "aborting to avoid frame corruption", sent);
643 brix_proxy_abort(proxy, "proxy: splice header send incomplete");
644 return NGX_ERROR;
645 }
646}

648/*
649 * brix_proxy_try_splice — attempt to start a zero-copy splice for the
650 * current read response. Returns NGX_OK if splice was started (the caller
651 * must NOT allocate resp_body or loop for body data). Returns NGX_DECLINED
652 * if the conditions are not met and the caller should use the normal path.
653 * Returns NGX_ERROR if it had to abort the session (PXY-6 partial-header case);
654 * the proxy has been torn down and the caller MUST return immediately without
655 * touching `proxy`.
656 */
657ngx_int_t
658brix_proxy_try_splice(brix_proxy_ctx_t *proxy)
659{
660 ngx_int_t rc;
661
662 if (!brix_proxy_splice_eligible(proxy)) {
663 return NGX_DECLINED;
664 }
665
666 if (brix_proxy_splice_pipe_ensure(proxy) != NGX_OK) {
667 return NGX_DECLINED;
668 }
669
670 rc = brix_proxy_splice_send_hdr(proxy);
671 if (rc != NGX_OK) {
672 return rc; /* NGX_DECLINED or NGX_ERROR (aborted) */
673 }
674
675 proxy->splice_active = 1;
676 proxy->splice_total = proxy->resp_dlen;
677 proxy->splice_upstream = 0;
678 proxy->splice_downstream = 0;
679
680 brix_proxy_splice_pump(proxy);
681 return NGX_OK;
682}
