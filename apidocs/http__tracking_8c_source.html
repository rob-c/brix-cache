6/*
7 * dashboard/http_tracking.c — HTTP-request lifecycle binding for live transfer slots.
8 *
9 * WHAT: Implements the brix_dashboard_http_*() API declared in
10 * dashboard_tracking.h. Each tracked WebDAV/S3/TPC request gets one
11 * transfer slot in the dashboard SHM table; this file allocates that slot
12 * on start, streams byte/state/error updates into it during the request,
13 * reports redacted TPC remote endpoints, and releases it on finish. The
14 * per-request binding is stored as the request's module ctx
15 * (brix_dashboard_http_track_t) so any handler can reach the slot.
16 * WHY: WebDAV and S3 run in the HTTP module where the unit of work is an
17 * ngx_http_request_t, not a stream session; this layer adapts the
18 * protocol-agnostic transfer_table.c slot operations to that lifecycle and
19 * guarantees the slot is freed even on abnormal teardown via an
20 * ngx_pool_cleanup handler (dashboard_http_cleanup) tied to r->pool.
21 * HOW: brix_dashboard_http_start_identity() is the workhorse — it validates
22 * the SHM zone, reserves a slot with brix_transfer_slot_alloc_ex(),
23 * registers the pool cleanup, and parks the binding in module ctx; the
24 * _start() variant is a thin "anonymous" wrapper. Updates look up the
25 * binding via dashboard_http_track() and forward to the slot ops, all
26 * stamped with ngx_current_msec. dashboard_http_client() copies the
27 * non-NUL-terminated addr_text into a bounded buffer, and
28 * dashboard_redact_url() strips userinfo/query from TPC URLs before they
29 * are stored, so secrets never reach the dashboard.
30 */
31

57/*
58 * Copy the client address into a NUL-terminated, length-bounded buffer.
59 *
60 * r->connection->addr_text is an ngx_str_t whose .data is NOT NUL-terminated
61 * (only .len bytes are valid; the tail of the buffer is uninitialised). The
62 * dashboard slot copier uses ngx_cpystrn(), which scans for a NUL — handing it
63 * the raw .data reads past .len into uninitialised memory (Phase 27 / Valgrind
64 * finding) and can copy garbage. Always pass this bounded copy instead.
65 */
66static const char *
67dashboard_http_client(ngx_http_request_t *r, char *buf, size_t bufsz)
68{
69 size_t n;
70
71 if (r != NULL && r->connection != NULL
72 && r->connection->addr_text.data != NULL
73 && r->connection->addr_text.len > 0
74 && bufsz > 0)
75 {
76 n = ngx_min(r->connection->addr_text.len, bufsz - 1);
77 ngx_memcpy(buf, r->connection->addr_text.data, n);
78 buf[n] = '\0';
79 return buf;
80 }
81
82 return "-";
83}

85/*
86 * Begin tracking an HTTP request: reserve a transfer slot and bind it to r.
87 *
88 * Returns the slot index (>= 0) on success, or -1 if the dashboard SHM zone is
89 * absent/uninitialised or slot/cleanup allocation fails. Idempotent — if r is
90 * already bound to a slot the existing index is returned without re-allocating.
91 * The slot is automatically released when r->pool is destroyed (see
92 * dashboard_http_cleanup) even if brix_dashboard_http_finish() is never called.
93 */
94int
95brix_dashboard_http_start_identity(ngx_http_request_t *r,
96 const char *path, const char *identity, const char *vo,
97 uint8_t proto, uint8_t direction, const char *op,
98 int64_t expected_bytes)
99{
100 brix_dashboard_http_track_t *track;
101 ngx_pool_cleanup_t *cln;
102 u_char sessid[16];
103 char ipbuf[NGX_SOCKADDR_STRLEN + 1];
104 int slot;
105
106 if (r == NULL || ngx_brix_dashboard_shm_zone == NULL
107 || ngx_brix_dashboard_shm_zone->data == NULL
108 || ngx_brix_dashboard_shm_zone->data == (void *) 1)
109 {
110 return -1;
111 }
112
113 track = dashboard_http_track(r);
114 if (track != NULL && track->slot >= 0) {
115 return track->slot;
116 }
117
118 ngx_memzero(sessid, sizeof(sessid));
119 slot = brix_transfer_slot_alloc_ex(ngx_brix_dashboard_shm_zone->data,
120 sessid,
121 dashboard_http_client(r, ipbuf,
122 sizeof(ipbuf)),
123 identity ? identity : "anonymous",
124 vo ? vo : "", path, op, direction,
125 proto, expected_bytes,
126 (int64_t) ngx_current_msec);
127 if (slot < 0) {
128 return -1;
129 }
130
131 cln = ngx_pool_cleanup_add(r->pool, sizeof(*track));
132 if (cln == NULL) {
133 brix_transfer_slot_free(ngx_brix_dashboard_shm_zone->data, slot);
134 return -1;
135 }
136
137 track = cln->data;
138 track->slot = slot;
139 track->table = ngx_brix_dashboard_shm_zone->data;
140 track->r = r;
141 cln->handler = dashboard_http_cleanup;
142 ngx_http_set_ctx(r, track, ngx_http_brix_dashboard_module);
143
144 if (op != NULL) {
145 brix_transfer_slot_count_op(track->table, track->slot, op);
146 }
147
148 return slot;
149}

200/*
201 * Split a TPC remote URL into a display-safe "scheme://host[:port]" string and a
202 * basename-only path hint, dropping anything sensitive or noisy.
203 *
204 * userinfo ("user:pass@") is stripped from the authority, and the path is
205 * reduced to its final segment with the query ("?") and fragment ("#") removed,
206 * so credentials and full object paths never land in the dashboard event/slot
207 * record. Both outputs are NUL-terminated and bounded by hostsz / pathsz.
208 */
209static void
210dashboard_redact_url(const char *url, char *host, size_t hostsz,
211 char *path_hint, size_t pathsz)
212{
213 const char *scheme_end;
214 const char *authority;
215 const char *path;
216 const char *at;
217 const char *end;
218 const char *base;
219 size_t n;
220
221 if (hostsz > 0) {
222 host[0] = '\0';
223 }
224 if (pathsz > 0) {
225 path_hint[0] = '\0';
226 }
227
228 if (url == NULL) {
229 return;
230 }
231
232 scheme_end = strstr(url, "://");
233 authority = scheme_end != NULL ? scheme_end + 3 : url;
234 path = strchr(authority, '/');
235 end = path != NULL ? path : authority + ngx_strlen(authority);
236
237 at = memchr(authority, '@', (size_t) (end - authority));
238 if (at != NULL) {
239 authority = at + 1;
240 }
241
242 if (scheme_end != NULL) {
243 n = (size_t) (scheme_end - url + 3);
244 if (n >= hostsz) {
245 n = hostsz > 0 ? hostsz - 1 : 0;
246 }
247 if (n > 0) {
248 ngx_memcpy(host, url, n);
249 }
250 if (hostsz > 0) {
251 host[n] = '\0';
252 }
253 if (n + 1 < hostsz) {
254 size_t hn = (size_t) (end - authority);
255 if (hn > hostsz - n - 1) {
256 hn = hostsz - n - 1;
257 }
258 ngx_memcpy(host + n, authority, hn);
259 host[n + hn] = '\0';
260 }
261 } else if (hostsz > 0) {
262 n = (size_t) (end - authority);
263 if (n >= hostsz) {
264 n = hostsz - 1;
265 }
266 ngx_memcpy(host, authority, n);
267 host[n] = '\0';
268 }
269
270 if (path == NULL || pathsz == 0) {
271 return;
272 }
273
274 end = path;
275 while (*end != '\0' && *end != '?' && *end != '#') {
276 end++;
277 }
278 base = end;
279 while (base > path && *(base - 1) != '/') {
280 base--;
281 }
282
283 n = (size_t) (end - base);
284 if (n == 0) {
285 base = "/";
286 n = 1;
287 }
288 if (n >= pathsz) {
289 n = pathsz - 1;
290 }
291 ngx_memcpy(path_hint, base, n);
292 path_hint[n] = '\0';
293}
