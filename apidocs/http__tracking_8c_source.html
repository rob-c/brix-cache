6/*
7 * dashboard/http_tracking.c — HTTP-request lifecycle binding for live transfer slots.
8 *
9 * WHAT: Implements the brix_dashboard_http_*() API declared in
10 * dashboard_tracking.h. Each tracked WebDAV/S3/TPC request gets one
11 * transfer slot in the dashboard SHM table; this file allocates that slot
12 * on start, streams byte/state/error updates into it during the request,
13 * reports redacted TPC remote endpoints, and releases it on finish. The
14 * per-request binding is stored as the request's module ctx
15 * (brix_dashboard_http_track_t) so any handler can reach the slot.
16 * WHY: WebDAV and S3 run in the HTTP module where the unit of work is an
17 * ngx_http_request_t, not a stream session; this layer adapts the
18 * protocol-agnostic transfer_table.c slot operations to that lifecycle and
19 * guarantees the slot is freed even on abnormal teardown via an
20 * ngx_pool_cleanup handler (dashboard_http_cleanup) tied to r->pool.
21 * HOW: brix_dashboard_http_start_identity() is the workhorse — it validates
22 * the SHM zone, reserves a slot with brix_transfer_slot_alloc_ex(),
23 * registers the pool cleanup, and parks the binding in module ctx; the
24 * _start() variant is a thin "anonymous" wrapper. Updates look up the
25 * binding via dashboard_http_track() and forward to the slot ops, all
26 * stamped with ngx_current_msec. dashboard_http_client() copies the
27 * non-NUL-terminated addr_text into a bounded buffer, and
28 * dashboard_redact_url() strips userinfo/query from TPC URLs before they
29 * are stored, so secrets never reach the dashboard.
30 */

73/*
74 * Copy the client address into a NUL-terminated, length-bounded buffer.
75 *
76 * r->connection->addr_text is an ngx_str_t whose .data is NOT NUL-terminated
77 * (only .len bytes are valid; the tail of the buffer is uninitialised). The
78 * dashboard slot copier uses ngx_cpystrn(), which scans for a NUL — handing it
79 * the raw .data reads past .len into uninitialised memory (Phase 27 / Valgrind
80 * finding) and can copy garbage. Always pass this bounded copy instead.
81 */
82static const char *
83dashboard_http_client(ngx_http_request_t *r, char *buf, size_t bufsz)
84{
85 size_t n;
86
87 if (r != NULL && r->connection != NULL
88 && r->connection->addr_text.data != NULL
89 && r->connection->addr_text.len > 0
90 && bufsz > 0)
91 {
92 n = ngx_min(r->connection->addr_text.len, bufsz - 1);
93 ngx_memcpy(buf, r->connection->addr_text.data, n);
94 buf[n] = '\0';
95 return buf;
96 }
97
98 return "-";
99}

120/*
121 * Register the pool cleanup that binds a reserved slot to the request.
122 *
123 * WHAT: Allocates a brix_dashboard_http_track_t inside an r->pool cleanup,
124 * fills it in for the given slot, parks it as the module ctx, and returns
125 * the slot index; on cleanup-allocation failure it frees the slot and
126 * returns -1 so the caller reports "not tracked".
127 * WHY: The slot must be released even on abnormal teardown; tying the binding
128 * to the pool cleanup guarantees that, and isolating it keeps the start
129 * function's control flow flat.
130 * HOW: ngx_pool_cleanup_add() → populate track → ngx_http_set_ctx(); the
131 * cleanup handler (dashboard_http_cleanup) frees the slot on pool destroy.
132 */
133static int
134dashboard_http_bind_slot(ngx_http_request_t *r, int slot, const char *op)
135{
136 brix_dashboard_http_track_t *track;
137 ngx_pool_cleanup_t *cln;
138
139 cln = ngx_pool_cleanup_add(r->pool, sizeof(*track));
140 if (cln == NULL) {
141 brix_transfer_slot_free(ngx_brix_dashboard_shm_zone->data, slot);
142 return -1;
143 }
144
145 track = cln->data;
146 track->slot = slot;
147 track->table = ngx_brix_dashboard_shm_zone->data;
148 track->r = r;
149 cln->handler = dashboard_http_cleanup;
150 ngx_http_set_ctx(r, track, ngx_http_brix_dashboard_module);
151
152 if (op != NULL) {
153 brix_transfer_slot_count_op(track->table, track->slot, op);
154 }
155
156 return slot;
157}

192 * Returns the slot index (>= 0) on success, or -1 if the dashboard SHM zone is

282/*
283 * Parsed view of a URL split at scheme/authority/path boundaries. All members
284 * point into the original url buffer (no copies): url is the full string,
285 * scheme_end is the "://" position (or NULL when scheme-less), authority is the
286 * userinfo-stripped host start, end is one-past the authority, and path is the
287 * first '/' of the path component (or NULL).
288 */
289typedef struct {
290 const char *url;
291 const char *scheme_end;
292 const char *authority;
293 const char *end;
294 const char *path;
295} redact_parts_t;
296
297/*
298 * Locate the userinfo-stripped authority span of a URL.
299 *
300 * WHAT: Fills p with the scheme/authority/path boundaries of url: authority is
301 * set past any "scheme://" prefix and any "user:pass@" userinfo, path to
302 * the first '/' (or NULL), and end to one-past the authority byte.
303 * WHY: Isolating the parse keeps the host/path writers branch-flat and makes
304 * the userinfo-drop (credential redaction) a single, testable step.
305 * HOW: strstr("://") → advance past userinfo '@' within [authority,end); all
306 * pointers refer into the caller's url buffer (no copies).
307 */
308static void
309redact_split_authority(const char *url, redact_parts_t *p)
310{
311 const char *auth;
312 const char *at;
313
314 p->url = url;
315 p->scheme_end = strstr(url, "://");
316 auth = p->scheme_end != NULL ? p->scheme_end + 3 : url;
317 p->path = strchr(auth, '/');
318 p->end = p->path != NULL ? p->path : auth + ngx_strlen(auth);
319
320 at = memchr(auth, '@', (size_t) (p->end - auth));
321 if (at != NULL) {
322 auth = at + 1;
323 }
324
325 p->authority = auth;
326}

328/*
329 * Write "scheme://host" (userinfo-stripped) into a bounded host buffer.
330 *
331 * WHAT: Emits the "scheme://" prefix (bytes url..scheme_end+3) followed by the
332 * authority span [authority,end) into host, NUL-terminated and capped at
333 * hostsz. When p->scheme_end is NULL only the authority is written.
334 * WHY: The scheme-present and scheme-less cases share the same bounds logic;
335 * one writer keeps the truncation math (and the byte-identical output)
336 * in a single place.
337 * HOW: Copy the prefix first, then as much of the authority as remaining
338 * capacity allows; every branch NUL-terminates within [0,hostsz).
339 */
340static void
341redact_write_host(const redact_parts_t *p, char *host, size_t hostsz)
342{
343 size_t n;
344 size_t hn;
345
346 if (p->scheme_end != NULL) {
347 n = (size_t) (p->scheme_end - p->url + 3);
348 if (n >= hostsz) {
349 n = hostsz > 0 ? hostsz - 1 : 0;
350 }
351 if (n > 0) {
352 ngx_memcpy(host, p->url, n);
353 }
354 if (hostsz > 0) {
355 host[n] = '\0';
356 }
357 if (n + 1 < hostsz) {
358 hn = (size_t) (p->end - p->authority);
359 if (hn > hostsz - n - 1) {
360 hn = hostsz - n - 1;
361 }
362 ngx_memcpy(host + n, p->authority, hn);
363 host[n + hn] = '\0';
364 }
365 return;
366 }
367
368 if (hostsz > 0) {
369 n = (size_t) (p->end - p->authority);
370 if (n >= hostsz) {
371 n = hostsz - 1;
372 }
373 ngx_memcpy(host, p->authority, n);
374 host[n] = '\0';
375 }
376}

378/*
379 * Reduce a URL path to its basename, dropping query and fragment.
380 *
381 * WHAT: Writes the final path segment of [path, …) (stopping at '?' or '#')
382 * into path_hint, NUL-terminated and capped at pathsz; an empty basename
383 * (path ends in '/') collapses to "/".
384 * WHY: Full object paths and query strings can carry sensitive or
385 * high-cardinality data; keeping only the basename preserves a useful
386 * display hint without leaking them into the dashboard record.
387 * HOW: Scan to the query/fragment terminator, walk back to the last '/', then
388 * copy the [base,end) run under the size cap.
389 */
390static void
391redact_write_path_hint(const char *path, char *path_hint, size_t pathsz)
392{
393 const char *end;
394 const char *base;
395 size_t n;
396
397 end = path;
398 while (*end != '\0' && *end != '?' && *end != '#') {
399 end++;
400 }
401 base = end;
402 while (base > path && *(base - 1) != '/') {
403 base--;
404 }
405
406 n = (size_t) (end - base);
407 if (n == 0) {
408 base = "/";
409 n = 1;
410 }
411 if (n >= pathsz) {
412 n = pathsz - 1;
413 }
414 ngx_memcpy(path_hint, base, n);
415 path_hint[n] = '\0';
416}

418/*
419 * Split a TPC remote URL into a display-safe "scheme://host[:port]" string and a
420 * basename-only path hint, dropping anything sensitive or noisy.
421 *
422 * userinfo ("user:pass@") is stripped from the authority, and the path is
423 * reduced to its final segment with the query ("?") and fragment ("#") removed,
424 * so credentials and full object paths never land in the dashboard event/slot
425 * record. Both outputs are NUL-terminated and bounded by hostsz / pathsz.
426 */
427static void
428dashboard_redact_url(const char *url, char *host, size_t hostsz,
429 char *path_hint, size_t pathsz)
430{
431 redact_parts_t parts;
432
433 if (hostsz > 0) {
434 host[0] = '\0';
435 }
436 if (pathsz > 0) {
437 path_hint[0] = '\0';
438 }
439
440 if (url == NULL) {
441 return;
442 }
443
444 redact_split_authority(url, &parts);
445 redact_write_host(&parts, host, hostsz);
446
447 if (parts.path == NULL || pathsz == 0) {
448 return;
449 }
450
451 redact_write_path_hint(parts.path, path_hint, pathsz);
452}
