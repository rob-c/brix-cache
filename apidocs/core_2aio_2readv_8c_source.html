3/*
4 * kXR_readv async I/O — multi-segment parallel read for large files.
5 *
6 * This file implements the thread-pool offload for xrdcp's segmented read mode,
7 * where a single request is split into multiple segments and each segment is
8 * fetched by the VFS I/O core on a worker thread. The response is assembled as:
9 * [segment header][segment payload] × N_segments
10 *
11 * Two functions: the _thread function does the blocking I/O; the _done callback
12 * builds the response chain on the main event loop and resumes connection flow.
13 */
14

16/*
17 * brix_readv_aio_thread — thread-pool worker for kXR_readv.
18 *
19 * The main nginx thread has already allocated the response buffer and computed
20 * the segment layout (header + payload pointers per segment). This worker
21 * only executes the VFS READV job to fill each segment's payload region and records totals.
22 *
23 * The response layout is:
24 * For each segment i:
25 * [brix_readv_seg_header (BRIX_READV_SEGSIZE bytes)]
26 * [up to read_length bytes of file data at payload_ptr]
27 *
28 * t->response_bytes = n_segments * BRIX_READV_SEGSIZE + bytes_read_total.
29 * Segments that pread to fewer bytes than requested are zero-padded by the
30 * thread (the per-segment header's rlen is updated to reflect the actual bytes
31 * read by brix_readv_read_segments).
32 */
33void
34brix_readv_aio_thread(void *data, ngx_log_t *log)
35{
36 brix_readv_aio_t *t = data;
37 brix_vfs_job_t job;
38
39 t->bytes_read_total = 0;
40 t->io_error = 0;
41 t->response_bytes = 0;
42 t->err_msg[0] = '\0';
43
44 /*
45 * The main nginx thread already built the response layout. The worker
46 * only fills payload pointers and records totals; it must not touch the
47 * connection or request parser state.
48 */
49 ngx_memzero(&job, sizeof(job));
50 job.op = BRIX_VFS_IO_READV;
51 job.segs = t->segments;
52 job.nsegs = t->segment_count;
53 job.err_msg = t->err_msg;
54 job.err_msg_cap = sizeof(t->err_msg);
55
56 brix_vfs_io_execute(&job);
57
58 if (job.io_errno != 0) {
59 t->io_error = 1;
60 if (t->err_msg[0] == '\0') {
61 snprintf(t->err_msg, sizeof(t->err_msg), "readv I/O error");
62 }
63 return;
64 }
65
66 t->bytes_read_total = (size_t) job.nio;
67 t->response_bytes = job.out_size;
68}

70/*
71 * brix_readv_aio_done — main-thread response builder for kXR_readv AIO completion.
72 *
73 * WHAT: Reconstructs the XRootD response chain after the worker thread finishes reading
74 * all segments in a segmented read request. The response consists of multiple segment headers
75 * (each containing segment size, handle index, offset) followed by their corresponding payload data.
76 * Unlike standard kXR_read, each segment has its own header describing which file and offset
77 * it corresponds to — enabling clients to reassemble fragments from potentially different sources.
78 *
79 * WHY: kXR_readv enables parallel multi-segment reads for large files where segments can be fetched
80 * concurrently across different file descriptors or offsets. This callback must handle four distinct paths:
81 * connection teardown (free segments + buffer), I/O error (send IOError message), success (build chain,
82 * update counters), and allocation failure (release before resume). The segment array is dynamically allocated
83 * by the main thread and freed here after all processing completes.
84 *
85 * HOW: 1) Restore request context via brix_aio_restore_request(). 2) Free segments on failure path.
86 * 3) On I/O error: send kXR_IOError with t->err_msg, release buffers, resume.
87 * 4) On success: update session_bytes and per-handle bytes_read for each segment, build chunked chain via
88 * brix_build_chunked_chain(), queue response, free segments array, resume connection events.
89 */
90void
91brix_readv_aio_done(ngx_event_t *ev)
92{
93 ngx_thread_task_t *task = ev->data;
94 brix_readv_aio_t *t = task->ctx;
95 brix_ctx_t *ctx = t->ctx;
96 ngx_connection_t *c = t->c;
97 ngx_chain_t *rsp_chain;
98 size_t i;
99
100 if (!brix_aio_restore_request(ctx, t->streamid)) {
101 ngx_free(t->segments);
102 brix_release_read_buffer(ctx, c, t->response_buffer);
103 return;
104 }
105
106 if (t->io_error) {
107 ngx_free(t->segments);
108 brix_release_read_buffer(ctx, c, t->response_buffer);
109 BRIX_OP_ERR(ctx, BRIX_OP_READV);
110 brix_send_error(ctx, c, kXR_IOError, t->err_msg);
111 brix_aio_resume(c);
112 return;
113 }
114
115 BRIX_OP_OK(ctx, BRIX_OP_READV);
116 ctx->totals.bytes += t->bytes_read_total;
117 for (i = 0; i < t->segment_count; i++) {
118 ctx->files[t->segments[i].handle_index].bytes_read +=
119 t->segments[i].read_length;
120 }
121
122 rsp_chain = brix_build_chunked_chain(ctx, c,
123 t->response_buffer,
124 t->response_bytes);
125 if (rsp_chain == NULL) {
126 ngx_free(t->segments);
127 brix_release_read_buffer(ctx, c, t->response_buffer);
128 brix_aio_resume(c);
129 return;
130 }
131
132 ngx_free(t->segments);
133 brix_queue_response_chain(ctx, c, rsp_chain, t->response_buffer);
134 if (ctx->state != XRD_ST_SENDING) {
135 brix_release_read_buffer(ctx, c, t->response_buffer);
136 }
137 brix_aio_resume(c);
138}
