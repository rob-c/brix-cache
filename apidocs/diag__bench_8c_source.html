45/* §15.3: sweep read request sizes to expose the throughput knee. Reads the whole
46 * file at each size into a discard buffer (no local fd), timing each pass. */
47void
48bench_sweep(brix_conn *c, const char *target)
49{
50 static const size_t sizes[] = { 65536, 262144, 1048576, 4194304, 16777216 };
51 size_t i;
52
53 printf("Read-size sweep:\n");
54 printf(" %-10s %12s\n", "req-size", "MB/s");
55 for (i = 0; i < sizeof(sizes) / sizeof(sizes[0]); i++) {
56 brix_file f;
57 brix_status st;
58 uint8_t *buf;
59 int64_t off = 0;
60 uint64_t t0, t1;
61 double secs;
62
63 brix_status_clear(&st);
64 if (brix_file_open_read(c, target, &f, &st) != 0) {
65 printf(" %-10zu open: %s\n", sizes[i], st.msg);
66 continue;
67 }
68 buf = (uint8_t *) malloc(sizes[i]);
69 if (buf == NULL) {
70 brix_file_close(c, &f, &st);
71 continue;
72 }
73 t0 = brix_mono_ns();
74 for (;;) {
75 ssize_t r = brix_file_read(c, &f, off, buf, sizes[i], &st);
76 if (r <= 0) {
77 break;
78 }
79 off += r;
80 }
81 t1 = brix_mono_ns();
82 free(buf);
83 brix_file_close(c, &f, &st);
84 secs = (double) (t1 - t0) / 1e9;
85 if (secs <= 0.0) {
86 secs = 1e-9;
87 }
88 printf(" %-10zu %12.1f\n", sizes[i], (double) off / 1e6 / secs);
89 }
90}
