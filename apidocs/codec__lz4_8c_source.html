1/*
2 * codec_lz4.c — LZ4 backend for the codec abstraction (BRIX_CODEC_LZ4).
3 *
4 * WHAT: Streaming LZ4 (de)compression via liblz4's LZ4 Frame API (lz4frame.h),
5 * producing/consuming the self-describing .lz4 frame format.
6 * WHY: LZ4 is the fastest mainstream codec — useful when CPU, not bandwidth, is
7 * the bottleneck (fast links, cheap clients). It is a deliberate extension
8 * beyond upstream XRootD (which ships only DEFLATE), compile-gated on
9 * -DBRIX_HAVE_LZ4 exactly like zstd/xz/brotli/bzip2.
10 * HOW: Decompress maps directly onto LZ4F_decompress (it already streams to any
11 * output size). Compress is block-oriented (LZ4F_compressBegin/Update/End
12 * emit whole chunks that must fit the destination), so it uses a small
13 * internal STAGING buffer: each step feeds one input chunk through LZ4F into
14 * the stage, then drains the stage into the caller's out buffer — which lets
15 * the codec honour the step() contract for ANY out size (incl. 1-byte
16 * buffers) without assuming the caller pre-sized for compressBound.
17 */
18

27#define LZ4_CHUNK (64 * 1024) /* input fed to LZ4F_compressUpdate per refill */

29typedef struct {
30 int dir;
31 LZ4F_cctx *cctx; /* compress context */
32 LZ4F_dctx *dctx; /* decompress context */
33 LZ4F_preferences_t prefs;
34 int begun; /* compressBegin emitted */
35 int ended; /* compressEnd emitted */
36 uint8_t *stage; /* pending compressed output awaiting drain */
37 size_t stage_cap;
38 size_t stage_len; /* bytes valid in stage */
39 size_t stage_pos; /* bytes already drained to the caller */
40} lz4_state;

42static int
43lz4_init(void **state, brix_codec_id_t id, brix_codec_dir_t dir, int level)
44{
45 lz4_state *st = calloc(1, sizeof(*st));
46
47 (void) id;
48 if (st == NULL) {
49 return BRIX_CODEC_ERR_MEM;
50 }
51 st->dir = dir;
52 if (dir == BRIX_CODEC_DIR_DECOMPRESS) {
53 if (LZ4F_isError(LZ4F_createDecompressionContext(&st->dctx, LZ4F_VERSION))) {
54 free(st);
55 return BRIX_CODEC_ERR_MEM;
56 }
57 } else {
58 if (LZ4F_isError(LZ4F_createCompressionContext(&st->cctx, LZ4F_VERSION))) {
59 free(st);
60 return BRIX_CODEC_ERR_MEM;
61 }
62 st->prefs.compressionLevel = level;
63 /* One staging buffer big enough for the frame header + one chunk's
64 * compressBound (worst case), reused for every refill incl. the footer. */
65 st->stage_cap = LZ4F_compressBound(LZ4_CHUNK, &st->prefs)
66 + LZ4F_HEADER_SIZE_MAX + 64;
67 st->stage = malloc(st->stage_cap);
68 if (st->stage == NULL) {
69 LZ4F_freeCompressionContext(st->cctx);
70 free(st);
71 return BRIX_CODEC_ERR_MEM;
72 }
73 }
74 *state = st;
75 return 0;
76}

78/* Drain staged compressed bytes into the caller's out buffer; returns 1 if the
79 * stage still has bytes left (caller must call again with more out room). */
80static int
81lz4_drain(lz4_state *st, uint8_t *out, size_t out_size, size_t *out_pos)
82{
83 size_t avail = out_size - *out_pos;
84 size_t have = st->stage_len - st->stage_pos;
85 size_t n = (have < avail) ? have : avail;
86
87 if (n > 0) {
88 memcpy(out + *out_pos, st->stage + st->stage_pos, n);
89 *out_pos += n;
90 st->stage_pos += n;
91 }
92 if (st->stage_pos >= st->stage_len) {
93 st->stage_len = st->stage_pos = 0; /* fully drained */
94 return 0;
95 }
96 return 1;
97}

99/* ---- Decompress one step of an .lz4 frame into the caller's out buffer ----
100 *
101 * WHAT: Feeds in[*in_pos..in_len) through LZ4F_decompress into
102 * out[*out_pos..out_size), advancing both cursors by the amount
103 * consumed/produced. Returns BRIX_CODEC_END when the frame end marker is
104 * reached, BRIX_CODEC_OK when more input/output is expected, or
105 * BRIX_CODEC_ERR_DATA on malformed/hostile input.
106 *
107 * WHY: LZ4F_decompress already streams to any output size, so the decompress
108 * side of step() is a thin cursor-advancing wrapper. Splitting it out of
109 * lz4_step keeps the compress staging logic and the decompress logic each
110 * independently reviewable.
111 *
112 * HOW: 1. Compute remaining src/dst spans from the cursors.
113 * 2. Call LZ4F_decompress, which updates src_n/dst_n in place to the
114 * amounts actually consumed/produced.
115 * 3. Advance *in_pos and *out_pos by those amounts (unconditionally — the
116 * library reports partial progress even alongside an error hint).
117 * 4. Map the return: LZ4F_isError to ERR_DATA, a zero hint (frame end) to
118 * END, otherwise OK.
119 */
120static brix_codec_rc_t
121lz4_step_decompress(lz4_state *st, const uint8_t *in, size_t in_len,
122 size_t *in_pos, uint8_t *out, size_t out_size,
123 size_t *out_pos)
124{
125 size_t src_n = in_len - *in_pos;
126 size_t dst_n = out_size - *out_pos;
127 size_t hint = LZ4F_decompress(st->dctx, out + *out_pos, &dst_n,
128 in + *in_pos, &src_n, NULL);
129
130 *in_pos += src_n;
131 *out_pos += dst_n;
132 if (LZ4F_isError(hint)) {
133 return BRIX_CODEC_ERR_DATA;
134 }
135 /* hint == 0 means a frame end was reached. */
136 return (hint == 0) ? BRIX_CODEC_END : BRIX_CODEC_OK;
137}

139/* ---- Emit the frame header into the stage if not yet begun ----
140 *
141 * WHAT: On the first compress step, writes the .lz4 frame header (magic +
142 * frame descriptor) into st->stage via LZ4F_compressBegin and adds its
143 * byte count to *produced. Returns BRIX_CODEC_OK (including the no-op case
144 * where the header was already emitted) or BRIX_CODEC_ERR_INTERNAL if the
145 * backend reports an error.
146 *
147 * WHY: The header must be emitted exactly once, before any body chunk, and its
148 * length folds into the same staging cursor the body writes advance.
149 * Isolating it keeps lz4_emit_frame_piece a flat two-step sequence.
150 *
151 * HOW: 1. If st->begun is already set, return OK without touching the stage.
152 * 2. Otherwise call LZ4F_compressBegin over the full staging capacity.
153 * 3. On error return ERR_INTERNAL; on success add the byte count to
154 * *produced and set st->begun.
155 */
156static brix_codec_rc_t
157lz4_stage_header(lz4_state *st, size_t *produced)
158{
159 size_t r;
160
161 if (st->begun) {
162 return BRIX_CODEC_OK;
163 }
164 r = LZ4F_compressBegin(st->cctx, st->stage, st->stage_cap, &st->prefs);
165 if (LZ4F_isError(r)) {
166 return BRIX_CODEC_ERR_INTERNAL;
167 }
168 *produced += r;
169 st->begun = 1;
170 return BRIX_CODEC_OK;
171}

173/* ---- Emit one body chunk, or the frame footer, into the stage ----
174 *
175 * WHAT: After the header, appends compressed output to st->stage at offset
176 * *produced: one input chunk (up to LZ4_CHUNK) via LZ4F_compressUpdate
177 * while input remains, or — once input is exhausted and finish is set —
178 * the frame footer via LZ4F_compressEnd. Advances *in_pos and *produced by
179 * the amounts consumed/produced and sets st->ended on footer emission.
180 * Returns BRIX_CODEC_OK or BRIX_CODEC_ERR_INTERNAL.
181 *
182 * WHY: A single compress step either advances the body or finalises the frame,
183 * never both; keeping them one helper preserves the mutually-exclusive
184 * ordering (body drains fully before the footer) that the LZ4 frame format
185 * requires for byte-exact output.
186 *
187 * HOW: 1. If input remains, clamp the chunk to LZ4_CHUNK, compressUpdate into
188 * the stage past *produced (capacity st->stage_cap - *produced), and on
189 * success advance *in_pos and *produced; return.
190 * 2. Else if finish is requested and the footer has not been emitted,
191 * compressEnd into the stage, add its bytes to *produced, and mark
192 * st->ended.
193 * 3. Any backend error yields ERR_INTERNAL; otherwise OK.
194 */
195static brix_codec_rc_t
196lz4_stage_body(lz4_state *st, const uint8_t *in, size_t in_len, size_t *in_pos,
197 int finish, size_t *produced)
198{
199 size_t r;
200
201 if (*in_pos < in_len) {
202 size_t chunk = in_len - *in_pos;
203 if (chunk > LZ4_CHUNK) {
204 chunk = LZ4_CHUNK;
205 }
206 r = LZ4F_compressUpdate(st->cctx, st->stage + *produced,
207 st->stage_cap - *produced, in + *in_pos, chunk,
208 NULL);
209 if (LZ4F_isError(r)) {
210 return BRIX_CODEC_ERR_INTERNAL;
211 }
212 *in_pos += chunk;
213 *produced += r;
214 return BRIX_CODEC_OK;
215 }
216 if (finish && !st->ended) {
217 r = LZ4F_compressEnd(st->cctx, st->stage + *produced,
218 st->stage_cap - *produced, NULL);
219 if (LZ4F_isError(r)) {
220 return BRIX_CODEC_ERR_INTERNAL;
221 }
222 *produced += r;
223 st->ended = 1;
224 }
225 return BRIX_CODEC_OK;
226}

228/* ---- Stage the next frame piece (header + body/footer) for draining ----
229 *
230 * WHAT: Composes lz4_stage_header and lz4_stage_body into st->stage and
231 * publishes the result via st->stage_len / st->stage_pos so lz4_drain can
232 * copy it to the caller. Returns BRIX_CODEC_OK or a BRIX_CODEC_ERR_INTERNAL
233 * propagated from either sub-step.
234 *
235 * WHY: lz4_step calls this only when the stage is empty and the out buffer has
236 * room, so a single fixed-capacity staging buffer can back a step()
237 * contract that honours ANY caller out size. Centralising the stage-cursor
238 * reset here keeps the orchestrator free of frame-format bookkeeping.
239 *
240 * HOW: 1. Emit the header (no-op after the first call), propagating errors.
241 * 2. Emit one body chunk or the footer, propagating errors.
242 * 3. Publish the produced byte count as stage_len with stage_pos reset to
243 * 0 (produced may be 0 — e.g. header already sent, no input, not
244 * finishing — which stages nothing, exactly as before).
245 */
246static brix_codec_rc_t
247lz4_emit_frame_piece(lz4_state *st, const uint8_t *in, size_t in_len,
248 size_t *in_pos, int finish)
249{
250 size_t produced = 0;
251 brix_codec_rc_t rc;
252
253 rc = lz4_stage_header(st, &produced);
254 if (rc != BRIX_CODEC_OK) {
255 return rc;
256 }
257 rc = lz4_stage_body(st, in, in_len, in_pos, finish, &produced);
258 if (rc != BRIX_CODEC_OK) {
259 return rc;
260 }
261 st->stage_len = produced;
262 st->stage_pos = 0;
263 return BRIX_CODEC_OK;
264}

266/* ---- One streaming (de)compression step for the LZ4 backend ----
267 *
268 * WHAT: The brix_codec_backend_t step() entry point. For DECOMPRESS delegates to
269 * lz4_step_decompress. For COMPRESS drains any previously staged output,
270 * then (if the out buffer has room) stages and drains the next frame
271 * piece. Returns BRIX_CODEC_END when a frame is complete, BRIX_CODEC_OK
272 * when more calls are needed, or a BRIX_CODEC_ERR_* from a sub-step.
273 *
274 * WHY: LZ4F compress emits whole chunks that must fit their destination, so the
275 * backend interposes a staging buffer and honours the step() contract for
276 * any out size by producing at most one piece per call and draining it
277 * incrementally across calls.
278 *
279 * HOW: 1. Decompress direction: delegate wholesale.
280 * 2. Compress: if the stage holds undrained bytes, drain them; if the out
281 * buffer fills first, return OK (more staged).
282 * 3. If out has no room and the stage is empty, report END once everything
283 * is flushed, else OK.
284 * 4. Otherwise stage the next frame piece (propagating errors) and drain
285 * it into out.
286 * 5. Report END only when the footer has been emitted AND fully drained;
287 * otherwise OK.
288 */
289static brix_codec_rc_t
290lz4_step(void *state, const uint8_t *in, size_t in_len, size_t *in_pos,
291 uint8_t *out, size_t out_size, size_t *out_pos, int finish)
292{
293 lz4_state *st = state;
294 brix_codec_rc_t rc;
295
296 if (st->dir == BRIX_CODEC_DIR_DECOMPRESS) {
297 return lz4_step_decompress(st, in, in_len, in_pos, out, out_size,
298 out_pos);
299 }
300
301 /* compress: drain any output staged from a previous refill first. */
302 if (st->stage_len > st->stage_pos) {
303 if (lz4_drain(st, out, out_size, out_pos)) {
304 return BRIX_CODEC_OK; /* out full; more staged */
305 }
306 }
307 if (*out_pos >= out_size) {
308 /* No room to do useful work this call (and stage is empty). Report
309 * END only once everything is flushed; otherwise ask to be called again. */
310 return (st->ended) ? BRIX_CODEC_END : BRIX_CODEC_OK;
311 }
312
313 /* Stage empty + out has room: emit and drain the next frame piece. */
314 rc = lz4_emit_frame_piece(st, in, in_len, in_pos, finish);
315 if (rc != BRIX_CODEC_OK) {
316 return rc;
317 }
318 (void) lz4_drain(st, out, out_size, out_pos);
319
320 /* END only when finished AND the final bytes are fully drained. */
321 if (st->ended && st->stage_len == st->stage_pos) {
322 return BRIX_CODEC_END;
323 }
324 return BRIX_CODEC_OK;
325}

327static void
328lz4_end(void *state)
329{
330 lz4_state *st = state;
331
332 if (st == NULL) {
333 return;
334 }
335 if (st->cctx != NULL) { LZ4F_freeCompressionContext(st->cctx); }
336 if (st->dctx != NULL) { LZ4F_freeDecompressionContext(st->dctx); }
337 free(st->stage);
338 free(st);
339}
