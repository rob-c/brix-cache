1/*
2 * auth.c — XRootD `pwd` (XrdSecpwd) password authentication handler — WS-B.
3 *
4 * WHAT: Implements the kXR_auth handler for the XRootD `pwd` security protocol as
5 * a 2-round DH-bootstrapped exchange:
6 * round 1 client → kXRS_puk(client DH pub) + kXRS_user;
7 * server → kXR_authmore: kXRS_puk(server DH pub) + credsreq.
8 * round 2 client → kXRS_main = DH-session-encrypted { kXRS_creds };
9 * server → kXR_ok iff PBKDF2(password, salt) == stored hash.
10 *
11 * WHY: pwd is XRootD's legacy password scheme. It is opt-in (conf->auth ==
12 * BRIX_AUTH_PWD), requires brix_pwd_file (empty = deny all), and SHOULD
13 * run only under TLS (the password is recoverable by the server, and the
14 * credential is only DH-session-encrypted on the wire). Isolating it here
15 * keeps the password surface in one auditable place.
16 *
17 * HOW: Reuses the shared GSI DH + session-cipher primitives (src/gsi/gsi_core.c)
18 * for the key agreement and the kXRS_main encryption, and the XrdSut bucket
19 * kernels for the wire. The password never touches disk in cleartext — the
20 * check is PBKDF2-HMAC-SHA1 against brix_pwd_file (src/pwd/pwdfile.c).
21 *
22 * Wire reference: docs/refactor/phase-52-pwd-wire-spec.md. Interop scope: this
23 * exchanges DH publics in-band (no pre-shared srvpuk) and omits the mutual rtag
24 * verification, so it is verified our-client ↔ our-server; full stock-XrdSecpwd
25 * byte-interop (pre-shared srvpuk + 3-RT rtag) is a documented follow-on.
26 */

38/*
39 * pwd_status_word — marshal a pwdStatus_t {ctype,action,options} into the 4-byte
40 * big-endian word stock XrdSecpwd puts in kXRS_status (it htonl's the whole struct
41 * image; XrdSecProtocolpwd.cc:1127-1130). ctype/action are bytes 0/1, options is
42 * the low 2 bytes, then the 4-byte host image is htonl'd.
43 */
44static uint32_t
45pwd_status_word(uint8_t ctype, uint8_t action, uint16_t options)
46{
47 uint8_t img[4];
48
49 img[0] = ctype;
50 img[1] = action;
51 img[2] = (uint8_t) (options & 0xff);
52 img[3] = (uint8_t) (options >> 8);
53 return htonl(*(uint32_t *) img);
54}

56/*
57 * pwd_send_credsreq — round-1 reply: kXR_authmore carrying the server DH public
58 * (so the client can derive the same session key) and a credsreq status word.
59 */
60static ngx_int_t
61pwd_send_credsreq(brix_ctx_t *ctx, ngx_connection_t *c, EVP_PKEY *srv)
62{
63 brix_gbuf g;
64 char *pub;
65 size_t pub_len = 0;
66 uint32_t status;
67 u_char *buf;
68 size_t total;
69
70 pub = brix_gsi_cipher_public(srv, &pub_len);
71 if (pub == NULL) {
72 return brix_send_error(ctx, c, kXR_ServerError,
73 "pwd: cannot encode server key");
74 }
75
76 status = pwd_status_word(kpCT_normal, 0, 0);
77
78 brix_gbuf_init(&g);
79 brix_gbuf_raw(&g, "pwd", 4); /* protocol name + NUL */
80 brix_gbuf_u32(&g, kXPS_credsreq); /* step */
81 brix_gbuf_bucket(&g, kXRS_puk, pub, pub_len);
82 brix_gbuf_bucket(&g, kXRS_status, &status, sizeof(status));
83 brix_gbuf_end(&g); /* kXRS_none */
84 free(pub); /* malloc'd by cipher_public */
85
86 if (g.err) {
87 brix_gbuf_free(&g);
88 return brix_send_error(ctx, c, kXR_NoMemory, "pwd: out of memory");
89 }
90
91 total = XRD_RESPONSE_HDR_LEN + g.len;
92 buf = ngx_palloc(c->pool, total);
93 if (buf == NULL) {
94 brix_gbuf_free(&g);
95 return brix_send_error(ctx, c, kXR_NoMemory, "pwd: out of memory");
96 }
97 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_authmore,
98 (uint32_t) g.len, (ServerResponseHdr *) buf);
99 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, g.p, g.len);
100 brix_gbuf_free(&g);
101
102 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
103 "brix: pwd round 1 → credsreq (srv puk %uz B)", pub_len);
104 return brix_queue_response(ctx, c, buf, total);
105}

107/*
108 * pwd_round1 — establish the DH session key from the client's kXRS_puk, record the
109 * asserted user, and ask for the credential. The user is NOT looked up yet (that
110 * happens at round 2, with the password in hand, to avoid a user-enumeration
111 * oracle on the cheap first round).
112 */
113static ngx_int_t
114pwd_round1(brix_ctx_t *ctx, ngx_connection_t *c,
115 ngx_stream_brix_srv_conf_t *conf)
116{
117 const uint8_t *puk, *user;
118 size_t puk_len = 0, user_len = 0;
119 EVP_PKEY *peer, *srv;
120 int ok;
121
122 (void) conf;
123
124 if (brix_gsi_find_bucket(ctx->recv.payload, ctx->recv.cur_dlen,
125 (uint32_t) kXRS_puk, &puk, &puk_len) != 0
126 || brix_gsi_find_bucket(ctx->recv.payload, ctx->recv.cur_dlen,
127 (uint32_t) kXRS_user, &user, &user_len) != 0
128 || user_len == 0 || user_len >= sizeof(ctx->pwd.user))
129 {
130 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_PWD, 0);
131 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "pwd",
132 kXR_NotAuthorized, "malformed pwd credential");
133 }
134
135 peer = brix_gsi_cipher_parse_peer(puk, puk_len);
136 if (peer == NULL) {
137 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_PWD, 0);
138 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "pwd",
139 kXR_NotAuthorized, "pwd: bad client key");
140 }
141 srv = brix_gsi_cipher_keygen_from(peer);
142 if (srv == NULL) {
143 EVP_PKEY_free(peer);
144 return brix_send_error(ctx, c, kXR_ServerError, "pwd: keygen failed");
145 }
146 ok = brix_gsi_cipher_session_key(srv, peer, 0, ctx->pwd.session_key,
147 BRIX_PWD_SESSION_KEYLEN);
148 EVP_PKEY_free(peer);
149 if (!ok) {
150 EVP_PKEY_free(srv);
151 return brix_send_error(ctx, c, kXR_ServerError,
152 "pwd: key agreement failed");
153 }
154
155 ngx_memcpy(ctx->pwd.user, user, user_len);
156 ctx->pwd.user[user_len] = '\0';
157 ctx->pwd.round = 1;
158
159 {
160 ngx_int_t rc = pwd_send_credsreq(ctx, c, srv);
161 EVP_PKEY_free(srv);
162 return rc;
163 }
164}

166/*
167 * pwd_round2 — decrypt the credential with the DH session key and verify it
168 * (PBKDF2-HMAC-SHA1) against the stored hash for the round-1 user. On success the
169 * connection is authenticated AS that user.
170 */
171static ngx_int_t
172pwd_round2(brix_ctx_t *ctx, ngx_connection_t *c,
173 ngx_stream_brix_srv_conf_t *conf)
174{
175 const uint8_t *main_blob, *creds;
176 size_t main_len = 0, creds_len = 0, plain_len = 0;
177 uint8_t *plain;
178 brix_gsi_cipher_t cipher;
179 uint8_t salt[BRIX_PWD_MAX_SALT];
180 uint8_t hash[BRIX_PWD_HASH_LEN];
181 size_t saltlen = 0, hashlen = 0;
182 char pwdpath[1024];
183 int verified;
184
185 if (conf->pwd_file.len == 0 || conf->pwd_file.len >= sizeof(pwdpath)) {
186 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_PWD, 0);
187 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "pwd",
188 kXR_NotAuthorized, "pwd auth not configured");
189 }
190
191 if (brix_gsi_find_bucket(ctx->recv.payload, ctx->recv.cur_dlen,
192 (uint32_t) kXRS_main, &main_blob, &main_len) != 0
193 || !brix_gsi_cipher_lookup("aes-128-cbc", &cipher))
194 {
195 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_PWD, 0);
196 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "pwd",
197 kXR_NotAuthorized, "malformed pwd credential");
198 }
199
200 plain = brix_gsi_cipher_decrypt(&cipher, ctx->pwd.session_key,
201 main_blob, main_len, 0, &plain_len);
202 OPENSSL_cleanse(ctx->pwd.session_key, sizeof(ctx->pwd.session_key));
203 if (plain == NULL
204 || brix_gsi_find_bucket(plain, plain_len, (uint32_t) kXRS_creds,
205 &creds, &creds_len) != 0
206 || creds_len == 0)
207 {
208 if (plain != NULL) {
209 OPENSSL_cleanse(plain, plain_len);
210 free(plain);
211 }
212 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_PWD, 0);
213 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "pwd",
214 kXR_NotAuthorized, "pwd: credential decrypt failed");
215 }
216
217 ngx_memcpy(pwdpath, conf->pwd_file.data, conf->pwd_file.len);
218 pwdpath[conf->pwd_file.len] = '\0';
219
220 verified = 0;
221 if (brix_pwd_file_lookup(pwdpath, ctx->pwd.user, salt, &saltlen,
222 hash, &hashlen) == NGX_OK)
223 {
224 verified = brix_pwd_verify(creds, creds_len, salt, saltlen,
225 hash, hashlen);
226 }
227 OPENSSL_cleanse(plain, plain_len);
228 free(plain);
229 OPENSSL_cleanse(salt, sizeof(salt));
230 OPENSSL_cleanse(hash, sizeof(hash));
231
232 if (!verified) {
233 ngx_log_error(NGX_LOG_NOTICE, c->log, 0,
234 "brix: pwd auth denied for user (bad credential)");
235 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_PWD, 0);
236 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "pwd",
237 kXR_NotAuthorized, "invalid password");
238 }
239
240 ctx->login.auth_done = 1;
241 ctx->token.auth = 0;
242 ngx_cpystrn((u_char *) ctx->login.dn, (u_char *) ctx->pwd.user, sizeof(ctx->login.dn));
243 ctx->login.vo_list[0] = '\0';
244 ctx->login.primary_vo[0] = '\0';
245
246 if (ctx->identity != NULL
247 && brix_identity_set_dn(ctx->identity, c->pool, ctx->login.dn,
248 BRIX_AUTHN_PWD) != NGX_OK)
249 {
250 return brix_send_error(ctx, c, kXR_NoMemory,
251 "identity allocation failed");
252 }
253
254 brix_session_register(ctx->login.sessid, ctx->login.dn, ctx->login.vo_list, 0);
255
256 ngx_log_error(NGX_LOG_INFO, c->log, 0, "brix: pwd auth OK user=\"%s\"",
257 ctx->login.dn);
258 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_PWD, 1);
259 BRIX_RETURN_OK(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "pwd", 0);
260}

262ngx_int_t
263brix_handle_pwd_auth(brix_ctx_t *ctx, ngx_connection_t *c,
264 ngx_stream_brix_srv_conf_t *conf)
265{
266 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen < 8
267 || ngx_strncmp(ctx->recv.payload, "pwd", 4) != 0)
268 {
269 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_PWD, 0);
270 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "pwd",
271 kXR_NotAuthorized, "malformed pwd credential");
272 }
273
274 if (ctx->pwd.round == 0) {
275 return pwd_round1(ctx, c, conf);
276 }
277 return pwd_round2(ctx, c, conf);
