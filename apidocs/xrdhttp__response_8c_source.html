44/*
45 * WHAT: One row of the HTTP-status to kXR-error mapping table.
46 * WHY: A data table replaces a long switch ladder, keeping the lookup
47 * function trivially simple and the mapping auditable at a glance.
48 * HOW: http_status is the numeric HTTP code (nginx macro value); xrd_code
49 * is the corresponding kXR_* error constant.
50 */
51typedef struct {
52 ngx_int_t http_status;
53 int xrd_code;
54} xrdhttp_status_map_row_t;
55
56/* Exact HTTP -> kXR mappings. Codes not listed fall through to the
57 * class-based default (4xx -> kXR_ArgInvalid, 5xx -> kXR_ServerError). */
58static const xrdhttp_status_map_row_t xrdhttp_status_map[] = {
59 { NGX_HTTP_BAD_REQUEST, kXR_ArgInvalid },
60 { NGX_HTTP_UNAUTHORIZED, kXR_NotAuthorized },
61 { NGX_HTTP_FORBIDDEN, kXR_NotAuthorized },
62 { NGX_HTTP_NOT_FOUND, kXR_NotFound },
63 { NGX_HTTP_NOT_ALLOWED, kXR_Unsupported },
64 { NGX_HTTP_CONFLICT, kXR_FSError },
65 { NGX_HTTP_LENGTH_REQUIRED, kXR_ArgMissing },
66 { 423 /* Locked */, kXR_FileLocked },
67 { NGX_HTTP_REQUEST_URI_TOO_LARGE, kXR_ArgTooLong },
68 { NGX_HTTP_RANGE_NOT_SATISFIABLE, kXR_ArgInvalid },
69 { NGX_HTTP_INTERNAL_SERVER_ERROR, kXR_ServerError },
70 { NGX_HTTP_NOT_IMPLEMENTED, kXR_Unsupported },
71 { NGX_HTTP_BAD_GATEWAY, kXR_FSError },
72 { NGX_HTTP_SERVICE_UNAVAILABLE, kXR_TooManyErrs },
73 { NGX_HTTP_INSUFFICIENT_STORAGE, kXR_NoSpace },
74};

76/*
77 * WHAT: Map any non-2xx HTTP status to a kXR_* error code by class.
78 * WHY: HTTP codes without an exact table entry still need a sensible kXR
79 * error so clients never see kXR_ok on a failure.
80 * HOW: 4xx -> kXR_ArgInvalid, 5xx (and above) -> kXR_ServerError, and any
81 * remaining code (1xx/3xx) -> kXR_ok, matching the original ladder.
82 */
83static int
84xrdhttp_status_class_default(ngx_int_t http_status)
85{
86 if (http_status >= 400 && http_status < 500) {
87 return kXR_ArgInvalid;
88 }
89 if (http_status >= 500) {
90 return kXR_ServerError;
91 }
92 return kXR_ok;
93}

95int
96xrdhttp_http_to_xrd_status(ngx_int_t http_status)
97{
98 size_t i;
99
100 if (http_status >= 200 && http_status < 300) {
101 return kXR_ok;
102 }
103
104 for (i = 0; i < sizeof(xrdhttp_status_map) / sizeof(xrdhttp_status_map[0]);
105 i++)
106 {
107 if (xrdhttp_status_map[i].http_status == http_status) {
108 return xrdhttp_status_map[i].xrd_code;
109 }
110 }
111
112 return xrdhttp_status_class_default(http_status);
113}

115ngx_int_t
116xrdhttp_add_response_headers(ngx_http_request_t *r, ngx_int_t http_status)
117{
118 xrdhttp_req_ctx_t *ctx;
119 int xrd_code;
120
121 ctx = (xrdhttp_req_ctx_t *)
122 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
123 if (ctx == NULL || !ctx->is_xrdhttp) {
124 return NGX_OK;
125 }
126
127 /* Guard against double-injection: filter + direct call in get.c. */
128 if (ctx->headers_injected) {
129 return NGX_OK;
130 }
131 ctx->headers_injected = 1;
132
133 /* X-Xrootd-Requuid — echo the request UUID back to the client. */
134 if (ctx->requuid[0]) {
135 if (brix_http_set_header(r, "X-Xrootd-Requuid", ctx->requuid,
136 NULL) != NGX_OK)
137 {
138 return NGX_ERROR;
139 }
140 }
141
142 /* X-Xrootd-Status — always emit per XrdHttp spec: 0 = success, non-zero = error. */
143 xrd_code = xrdhttp_http_to_xrd_status(http_status);
144 if (brix_http_set_header_num(r, "X-Xrootd-Status",
145 (long) xrd_code) != NGX_OK)
146 {
147 return NGX_ERROR;
148 }
149
150 /* X-Xrootd-Wait — back-pressure for async/tape-recall scenarios. */
151 if (ctx->wait_seconds > 0) {
152 if (brix_http_set_header_num(r, "X-Xrootd-Wait",
153 (long) ctx->wait_seconds) != NGX_OK)
154 {
155 return NGX_ERROR;
156 }
157 }
158
159 /* X-Xrootd-Retry — hint for client retry interval. */
160 if (ctx->retry_seconds > 0) {
161 if (brix_http_set_header_num(r, "X-Xrootd-Retry",
162 (long) ctx->retry_seconds) != NGX_OK)
163 {
164 return NGX_ERROR;
165 }
166 }
167
168 return NGX_OK;
169}

171/* redirect dialect */

173/*
174 * WHAT: Produce the redirect Location URL, appending tpc.key / xrd.opaque
175 * query params from ctx when present.
176 * WHY: XrdHttp redirects must carry the TPC key and opaque data forward to
177 * the target node; this is the only place that formatting lives.
178 * HOW: Returns location_url unchanged when there is nothing to append or ctx
179 * is absent; otherwise formats into loc_buf and returns it. On overflow
180 * it warns and falls back to the plain (truncated) URL — identical to the
181 * original inline behaviour.
182 */
183static const char *
184xrdhttp_build_redirect_location(ngx_http_request_t *r,
185 const xrdhttp_req_ctx_t *ctx,
186 const char *location_url,
187 char *loc_buf, size_t loc_bufsz)
188{
189 const char *sep;
190 int loc_len;
191
192 if (ctx == NULL || !(ctx->tpc_key[0] || ctx->opaque[0])) {
193 return location_url;
194 }
195
196 sep = (ngx_strchr(location_url, '?') != NULL) ? "&" : "?";
197
198 if (ctx->tpc_key[0] && ctx->opaque[0]) {
199 loc_len = snprintf(loc_buf, loc_bufsz,
200 "%s%stpc.key=%s&xrd.opaque=%s",
201 location_url, sep, ctx->tpc_key, ctx->opaque);
202 } else if (ctx->tpc_key[0]) {
203 loc_len = snprintf(loc_buf, loc_bufsz,
204 "%s%stpc.key=%s",
205 location_url, sep, ctx->tpc_key);
206 } else {
207 loc_len = snprintf(loc_buf, loc_bufsz,
208 "%s%sxrd.opaque=%s",
209 location_url, sep, ctx->opaque);
210 }
211
212 if (loc_len < 0 || (size_t) loc_len >= loc_bufsz) {
213 /* Overflow: use the plain URL without appended params. */
214 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
215 "xrdhttp: redirect URL overflow, opaque dropped");
216 ngx_cpystrn((u_char *) loc_buf, (u_char *) location_url, loc_bufsz);
217 }
218
219 return loc_buf;
220}

222/*
223 * WHAT: Add the Location header for a redirect to location_url.
224 * WHY: Separating list-push + string-dup keeps the redirect entry point flat.
225 * HOW: Returns NGX_OK, or NGX_HTTP_INTERNAL_SERVER_ERROR on allocation
226 * failure (same sentinels the caller returns directly).
227 */
228static ngx_int_t
229xrdhttp_set_location_header(ngx_http_request_t *r, const char *location_url)
230{
231 ngx_table_elt_t *location_hdr = ngx_list_push(&r->headers_out.headers);
232
233 if (location_hdr == NULL) {
234 return NGX_HTTP_INTERNAL_SERVER_ERROR;
235 }
236 location_hdr->hash = 1;
237 ngx_str_set(&location_hdr->key, "Location");
238 location_hdr->value.data = ngx_pstrdup(r->pool, &(ngx_str_t){
239 ngx_strlen(location_url), (u_char *) location_url
240 });
241 if (location_hdr->value.data == NULL) {
242 return NGX_HTTP_INTERNAL_SERVER_ERROR;
243 }
244 location_hdr->value.len = ngx_strlen(location_url);
245
246 return NGX_OK;
247}

249/*
250 * WHAT: Emit the X-Xrootd-Redir-Host / X-Xrootd-Redir-Port headers.
251 * WHY: Hierarchical cluster topologies advertise the concrete target node
252 * alongside the Location URL.
253 * HOW: Each header is only added when its value is meaningful; returns
254 * NGX_OK or NGX_HTTP_INTERNAL_SERVER_ERROR on header-set failure.
255 */
256static ngx_int_t
257xrdhttp_set_redir_target_headers(ngx_http_request_t *r,
258 const char *redir_host, int redir_port)
259{
260 if (redir_host != NULL && redir_host[0]) {
261 if (brix_http_set_header(r, "X-Xrootd-Redir-Host", redir_host,
262 NULL) != NGX_OK)
263 {
264 return NGX_HTTP_INTERNAL_SERVER_ERROR;
265 }
266 }
267 if (redir_port > 0) {
268 if (brix_http_set_header_num(r, "X-Xrootd-Redir-Port",
269 (long) redir_port) != NGX_OK)
270 {
271 return NGX_HTTP_INTERNAL_SERVER_ERROR;
272 }
273 }
274
275 return NGX_OK;
276}

278ngx_int_t
279xrdhttp_send_redirect(ngx_http_request_t *r,
280 const char *location_url,
281 const char *redir_host,
282 int redir_port)
283{
284 xrdhttp_req_ctx_t *ctx;
285 char loc_buf[XRDHTTP_TPC_URL_MAX + XRDHTTP_OPAQUE_MAX + 64];
286 ngx_int_t rc;
287
288 ctx = (xrdhttp_req_ctx_t *)
289 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
290
291 location_url = xrdhttp_build_redirect_location(r, ctx, location_url,
292 loc_buf, sizeof(loc_buf));
293
294 rc = xrdhttp_set_location_header(r, location_url);
295 if (rc != NGX_OK) {
296 return rc;
297 }
298
299 rc = xrdhttp_set_redir_target_headers(r, redir_host, redir_port);
300 if (rc != NGX_OK) {
301 return rc;
302 }
303
304 /* Echo XrdHttp response headers (Requuid, Status=kXR_ok for redirect). */
305 if (xrdhttp_add_response_headers(r, NGX_HTTP_TEMPORARY_REDIRECT)
306 != NGX_OK)
307 {
308 return NGX_HTTP_INTERNAL_SERVER_ERROR;
309 }
310
311 r->headers_out.status = NGX_HTTP_TEMPORARY_REDIRECT;
312 r->headers_out.content_length_n = 0;
313
314 ngx_http_send_header(r);
315 return ngx_http_send_special(r, NGX_HTTP_LAST);
316}
