1/*
2 * brix_fault_tls.h — pure TLS record-layer surgery for brix-fault-proxy.
3 *
4 * A TLS/DTLS stream on the wire is a sequence of records, each a 5-byte header
5 * ([content-type][version-major][version-minor][length-hi][length-lo]) followed
6 * by `length` payload bytes. An on-path adversary who cannot decrypt the record
7 * bodies can still attack the RECORD LAYER: re-fragment records to stress a
8 * peer's reassembly, lie about a record's declared length (stall the parser),
9 * relabel a record's content-type (present application-data as handshake), flip a
10 * ciphertext byte so the MAC check fails, drop records of a chosen type, forge a
11 * plaintext alert record, or downgrade the record-layer version.
12 *
13 * These are the record-framing bugs a hardened TLS client/server must survive.
14 * Everything here is a PURE function over caller buffers — no globals, no I/O —
15 * so the relay snapshots its (locked) config and calls in from the hot path, and
16 * unit tests exercise the record math directly. Operates per read-buffer
17 * (best-effort across a record split by TCP segmentation, like the length-mangle
18 * lever): a record header straddling two reads is passed through untouched.
19 */

21#define BRIX_FAULT_TLS_H

25/* Per-direction TLS record-surgery config. Sentinel -1 (or 0 for frag/inflate/
26 * flip) disables an op; fp_tls_cfg_init() installs those defaults. */
27typedef struct {
28 int frag_max; /* >0: re-fragment each record to <= this payload size */
29 int set_type; /* >=0: overwrite every record's content-type byte */
30 int set_ver_major; /* >=0: overwrite the record version high byte */
31 int set_ver_minor; /* >=0: overwrite the record version low byte */
32 int inflate_len; /* !=0: add this to each record's declared length field */
33 int flip_payload; /* 1: XOR-flip the first payload byte of each record */
34 int drop_type; /* >=0: drop records whose content-type equals this */
35 int alert_level; /* >=0: prepend ONE forged alert record (consumed once) */
36 int alert_desc; /* alert description byte for the forged alert */
37} fp_tls_cfg;

39/* Tallies accumulated by fp_tls_rewrite() (added, not reset). */
40typedef struct {

44 unsigned long dropped; /* records dropped by drop-type */

46 unsigned long retyped; /* records whose content-type was rewritten */
