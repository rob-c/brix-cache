1/*
2 * ucred_internal.h — internal parser prototypes shared between ucred.c and
3 * ucred_parse.c.
4 *
5 * WHAT: Declares the four credential-file readers (PEM/token/S3/keyring) that
6 * live in ucred_parse.c and are called from brix_sd_ucred_resolve() in
7 * ucred.c. These are implementation details of the ucred helper — NOT
8 * part of the public ucred.h surface — so they live in a private header
9 * that only the two ucred translation units include.
10 *
11 * WHY: Splitting the file-format parsers out of ucred.c keeps each
12 * translation unit under the file-size guard while preserving the
13 * verbatim parser logic. Only the four reader entry points cross the
14 * split; the per-format line-scanning helpers remain static in
15 * ucred_parse.c.
16 *
17 * HOW: Prototypes only; include ucred.h for the ngx_int_t type and the
18 * BRIX_UCRED_* size bounds referenced by the callers.
19 */
20#ifndef BRIX_FS_BACKEND_UCRED_INTERNAL_H
21#define BRIX_FS_BACKEND_UCRED_INTERNAL_H
