1/*
2 * unified.h — cross-protocol metric vocabulary and record-side API.
3 *
4 * WHAT: Declares the enums that give stream, WebDAV, and S3 a single shared
5 * label vocabulary — brix_proto_t (stream/webdav/s3), brix_metric_op_t
6 * (read/write/stat/delete/mkdir/rename/dirlist/tpc), and brix_err_class_t
7 * (ok/not_found/forbidden/io_error/other) — plus the AUTH_*, TPC_*, and
8 * latency-bucket slot constants. It also declares the record-side helpers
9 * (brix_metric_op_done / _cache_result / _auth / _tpc) every protocol
10 * handler calls, and the name/classification helpers (proto/op/err/auth
11 * name lookups, errno→err and http-status→err mappers, auth-slot mapper).
12 * WHY: Per-protocol metric modules historically each defined their own labels,
13 * producing inconsistent dashboards. Funnelling all three protocols through
14 * one enum set keeps Prometheus labels low-cardinality (INVARIANT #8) and
15 * lets a single exporter (brix_export_unified_metrics in unified.c) emit
16 * directly comparable brix_io_* / brix_auth_total / brix_tpc_* series.
17 * HOW: Enum members are dense 0-based so they index fixed-size SHM counter arrays
18 * directly; the trailing *_COUNT member sizes those arrays. The implementation
19 * lives in unified.c; consumers include this header to record an event with a
20 * proto/op/err triple rather than touching SHM fields by hand.
21 */

99/*
100 * Classify a POSIX errno into an brix_err_class_t bucket: 0 -> NONE,

105/*
106 * Classify an HTTP status into an brix_err_class_t bucket: 2xx AND 3xx -> NONE,

109brix_err_class_t brix_metric_err_from_http_status(ngx_uint_t status);
110/*
111 * Reduce an identity auth_method bitmask to a single BRIX_METRIC_AUTH_* slot,

113 * BRIX_METRIC_AUTH_NONE when no known bit is set. Pure, no side effects.

115ngx_uint_t brix_metric_auth_slot(ngx_uint_t auth_method);

117/*
118 * Record one completed I/O op into the shared counters: bumps
119 * io_ops_total[proto][op][err], adds bytes to io_bytes_read/written (only for
120 * READ/WRITE ops; ignored otherwise), and files latency_usec (microseconds)
121 * into the single matching histogram bucket plus count and sum. Lock-free via
122 * atomics. Silently no-ops on an out-of-range proto/op/err triple or when the
123 * metrics SHM zone is not yet attached.
124 */
125void brix_metric_op_done(brix_proto_t proto, brix_metric_op_t op,
126 size_t bytes, ngx_msec_t latency_usec, brix_err_class_t err);
127/*
128 * brix_metric_backend_bytes — add a completed data op's byte count to the
129 * per-backend storage totals (io_bytes_{read,written}_backend). backend_name
130 * is the storage driver's census name (fs_list.h); NULL ⇒ "posix" (the
131 * default-instance convention everywhere in the VFS). Pure lock-free SHM
132 * atomics — safe from thread-pool workers (no pools, logs, or request state).
133 * No-ops on unknown names, non-READ/WRITE ops, zero bytes, or missing SHM.
134 */
135void brix_metric_backend_bytes(const char *backend_name,
136 brix_metric_op_t op, size_t bytes);
137/*
138 * Record a cache lookup outcome: bumps cache_hits or cache_misses[proto] by hit
139 * (treated as boolean) and adds bytes_evicted to cache_bytes_evicted[proto].
140 * Lock-free; no-ops on out-of-range proto or detached SHM.
141 */
142void brix_metric_cache_result(brix_proto_t proto, unsigned int hit,
143 size_t bytes_evicted);
144/*
145 * Watermark-driven LRU reaper telemetry (connection-less, process-wide):
146 * - cache_usage_ratio publishes cache_root occupancy (ppm) as a gauge each tick;
147 * - cache_watermark_purge accounts one purge run that reclaimed `files`/`bytes`.
148 * Lock-free; no-ops on detached SHM (and purge no-ops when files == 0).
149 */
150void brix_metric_cache_usage_ratio(ngx_uint_t occupancy_ppm);

152/*
153 * Write-back-staging backpressure telemetry: usage_ratio publishes staging

155 * hard-cap rejection, else a soft-band delay). Lock-free; no-op on detached SHM.
156 */
157void brix_metric_wt_stage_usage_ratio(ngx_uint_t occupancy_ppm);

159/*
160 * Record one auth attempt: maps auth_method (a bitmask) to its slot via
161 * brix_metric_auth_slot(), then bumps auth_total[proto][slot][ok|fail] keyed
162 * by success (boolean). Lock-free; no-ops on out-of-range proto or detached SHM.
163 */
164void brix_metric_auth(brix_proto_t proto, ngx_uint_t auth_method,
165 unsigned int success);
166/*
167 * Record a third-party-copy outcome: bumps tpc_transfers[proto][dir][err] where
168 * dir is push when is_push is nonzero else pull, and ONLY on err==BRIX_ERR_NONE

170 * proto/err or detached SHM.
171 */
172void brix_metric_tpc(brix_proto_t proto, unsigned int is_push,
173 size_t bytes, brix_err_class_t err);
