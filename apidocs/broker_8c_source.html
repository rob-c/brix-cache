20imp_peer_allowed(int conn_fd)
21{
22 struct ucred cred;

24

36/* Read exactly n bytes; returns 1 ok, 0 EOF, -1 error. */
37int
38imp_read_full(int fd, void *buf, size_t n)
39{
40 u_char *p = buf;
41 size_t got = 0;
42
43 while (got < n) {
44 ssize_t r = read(fd, p + got, n - got);
45 if (r == 0) { return 0; }
46 if (r < 0) {
47 if (errno == EINTR) { continue; }
48 return -1;
49 }
50 got += (size_t) r;
51 }
52 return 1;
53}

56/*
57 * Send the reply frame, optionally with `data_len` trailing bytes (READLINK
58 * target) in the same message, and attaching `fd` via SCM_RIGHTS when fd >= 0.
59 */
60int
61imp_send_reply(int conn_fd, const imp_rep_t *rep, int fd,
62 const void *data, size_t data_len)
63{
64 struct msghdr msg;
65 struct iovec iov[2];
66 union {
67 struct cmsghdr align;
68 char buf[CMSG_SPACE(sizeof(int))];
69 } cmsgbuf;
70 ssize_t n;
71
72 ngx_memzero(&msg, sizeof(msg));
73 iov[0].iov_base = (void *) rep;
74 iov[0].iov_len = sizeof(*rep);
75 msg.msg_iov = iov;
76 msg.msg_iovlen = 1;
77 if (data_len > 0) {
78 iov[1].iov_base = (void *) data;
79 iov[1].iov_len = data_len;
80 msg.msg_iovlen = 2;
81 }
82
83 if (fd >= 0) {
84 struct cmsghdr *c;
85 ngx_memzero(&cmsgbuf, sizeof(cmsgbuf));
86 msg.msg_control = cmsgbuf.buf;
87 msg.msg_controllen = sizeof(cmsgbuf.buf);
88 c = CMSG_FIRSTHDR(&msg);
89 c->cmsg_level = SOL_SOCKET;
90 c->cmsg_type = SCM_RIGHTS;
91 c->cmsg_len = CMSG_LEN(sizeof(int));
92 ngx_memcpy(CMSG_DATA(c), &fd, sizeof(int));
93 }
94
95 do {
96 n = sendmsg(conn_fd, &msg, MSG_NOSIGNAL);
97 } while (n < 0 && errno == EINTR);
98
99 return (n == (ssize_t) (sizeof(*rep) + data_len)) ? 0 : -1;
100}

104/*
105 * imp_recv_request — read and normalise one request frame off conn_fd.
106 * WHAT: reads the fixed frame, zeroes the reply, NUL-terminates the wire
107 * strings, and — for SETXATTR — drains the trailing inbound value payload.
108 * WHY: framing and payload draining are security-load-bearing (a leftover
109 * SETXATTR value would desync the SOCK_STREAM into the next request), so they
110 * are isolated as one indivisible receive stage with explicit outcomes.
111 * HOW: returns 1 = frame ready to dispatch, 0 = drop the connection, or 2 =
112 * the frame's version was rejected (BADREQ, keep-connection decided by caller).
113 * On a 0 return that still owes the peer a reply (bad length), the reply is
114 * sent here. `*in_len` receives the SETXATTR value length (0 for every other
115 * op). The version check precedes the SETXATTR drain, exactly as before.
116 */
117static int
118imp_recv_request(int conn_fd, imp_req_t *req, imp_rep_t *rep,
119 char *data_in, size_t *in_len)
120{
121 int rc;
122
123 *in_len = 0;
124
125 rc = imp_read_full(conn_fd, req, sizeof(*req));
126 if (rc <= 0) {
127 return 0; /* EOF or error -> drop the connection */
128 }
129
130 ngx_memzero(rep, sizeof(*rep));
131
132 /* Validate the frame: version, NUL-termination, op range. */
133 req->principal[IMP_PRINC_MAX - 1] = '\0';
134 req->path[IMP_PATH_MAX - 1] = '\0';
135 req->path2[IMP_PATH_MAX - 1] = '\0';
136 if (req->version != IMP_PROTO_VERSION) {
137 return 2; /* caller emits BADREQ (pre-drain) */
138 }
139
140 /*
141 * SETXATTR carries an inbound value payload after the fixed frame. It MUST
142 * be drained from the SOCK_STREAM before any deny/early-return path, or the
143 * leftover bytes would be mis-read as the next request frame (desync). An
144 * over-large declared length is rejected and the connection dropped (we
145 * cannot trust the stream position). Only SETXATTR consumes inbound data;
146 * every other op leaves req_data_len == 0.
147 */
148 if (req->op == IMP_OP_SETXATTR) {
149 if (req->req_data_len > IMP_XATTR_MAX) {
150 rep->status = IMP_STATUS_BADREQ;
151 (void) imp_send_reply(conn_fd, rep, -1, NULL, 0);
152 return 0; /* close: stream position untrustworthy */
153 }
154 *in_len = req->req_data_len;
155 if (*in_len > 0 && imp_read_full(conn_fd, data_in, *in_len) <= 0) {
156 return 0; /* EOF/error mid-payload -> drop */
157 }
158 }
159
160 return 1;
161}

164/*
165 * imp_fatal_priv_refused — handle the should-be-impossible RESERVED-credential
166 * refusal at the setfsuid edge. Does not return.
167 * WHAT: replies IMP_STATUS_PRIV_REFUSED, logs loudly to both nginx log and
168 * stderr, and terminates the broker process.
169 * WHY: reaching this means an upstream mapping-layer guard was bypassed; NO
170 * credential change was made, but continuing would risk a privileged file op,
171 * so the broker fails closed + fail-loud. Workers then fail closed on every
172 * subsequent impersonated op.
173 * HOW: extracted verbatim from imp_serve_one so the dispatch path stays under
174 * the complexity gate; identical reply, log text, and _exit as before.
175 */
176static void
177imp_fatal_priv_refused(int conn_fd, imp_rep_t *rep,
178 const brix_idmap_creds_t *creds, ngx_log_t *log)
179{
180 rep->status = IMP_STATUS_PRIV_REFUSED;
181 (void) imp_send_reply(conn_fd, rep, -1, NULL, 0);
182 if (log) ngx_log_error(NGX_LOG_CRIT, log, 0,
183 "impersonate broker: FATAL — refused RESERVED "
184 "credential at the setfsuid edge (uid=%d gid=%d, "
185 "floor=%d); terminating broker to guarantee no "
186 "privileged file op runs",
187 (int) creds->uid, (int) creds->gid,
188 BRIX_IMP_HARD_MIN_ID);
189 /* Also emit to stderr: this is fatal and log may be NULL (tests, or a
190 * mis-set error_log), and the reason must never be silently swallowed. */
191 fprintf(stderr, "impersonate broker: FATAL — refused RESERVED credential "
192 "(uid=%d gid=%d floor=%d); terminating\n",
193 (int) creds->uid, (int) creds->gid, BRIX_IMP_HARD_MIN_ID);
194 _exit(EXIT_FAILURE);
195}

198/*
199 * imp_dispatch_op — impersonate, run the fs op, and restore base creds.
200 * WHAT: becomes the mapped user, invokes imp_do_op via the caller-populated
201 * imp_op_ctx_t, then always restores (even on error). On a fatal reserved-id
202 * refusal it does not return (terminates the broker).
203 * WHY: the impersonate→op→restore trio is the security-critical core; keeping
204 * it in one helper guarantees imp_restore() pairs every successful imp_become()
205 * on every path. The OPEN result fd flows back through op_ctx->out_fd.
206 * HOW: `op_ctx` is fully filled by the caller except `rel` (imp_do_op derives
207 * it). Returns the op status (0 or -errno) to place in rep->status. A become
208 * failure short-circuits to -EPERM after restoring, matching prior behaviour.
209 */
210static int
211imp_dispatch_op(int conn_fd, imp_op_ctx_t *op_ctx,
212 const brix_idmap_creds_t *creds, ngx_log_t *log)
213{
214 int rc;
215
216 /* Impersonate -> op -> restore (always restore, even on error). */
217 rc = imp_become(creds);
218 if (rc == IMP_REFUSE_PRIV) {
219 /* does not return */
220 imp_fatal_priv_refused(conn_fd, op_ctx->rep, creds, log);
221 }
222 if (rc != 0) {
223 imp_restore();
224 if (log) ngx_log_error(NGX_LOG_ERR, log, 0,
225 "impersonate broker: could not become uid=%d",
226 (int) creds->uid);
227 return -EPERM;
228 }
229
230 op_ctx->rel = NULL; /* imp_do_op derives it from req.path */
231 rc = imp_do_op(op_ctx);
232 imp_restore();
233
234 return rc; /* 0 or -errno */
235}

238/* Serve one request on conn_fd. Returns 1 to keep the connection, 0 to close. */
239int
240imp_serve_one(int conn_fd, int rootfd, ngx_log_t *log)
241{
242 /*
243 * Outbound (READLINK/GETXATTR/LISTXATTR) and inbound (SETXATTR value)
244 * trailing-payload scratch. `static` rather than stack: the broker serve
245 * loop is strictly single-threaded and non-reentrant, so two 64 KiB frames
246 * on the stack per request would be wasteful; each request fully overwrites
247 * the bytes it sends (only rep.data_len of data_buf is ever transmitted), so
248 * nothing stale leaks across requests.
249 */
250 static char data_buf[IMP_XATTR_MAX]; /* outbound reply payload */
251 static char data_in[IMP_XATTR_MAX]; /* inbound SETXATTR value */
252 imp_req_t req;
253 imp_rep_t rep;
254 imp_op_ctx_t op_ctx;
255 brix_idmap_creds_t creds;
256 size_t in_len = 0;
257 int rc, fd = -1;
258
259 rc = imp_recv_request(conn_fd, &req, &rep, data_in, &in_len);
260 if (rc == 0) {
261 return 0; /* dropped (EOF/error/bad length) */
262 }
263 if (rc == 2) { /* version rejected, before any drain */
264 rep.status = IMP_STATUS_BADREQ;
265 return imp_send_reply(conn_fd, &rep, -1, NULL, 0) == 0 ? 1 : 0;
266 }
267
268 if (req.op == IMP_OP_PING) {
269 rep.status = IMP_STATUS_OK;
270 return imp_send_reply(conn_fd, &rep, -1, NULL, 0) == 0 ? 1 : 0;
271 }
272
273 /* Map the principal -> UNIX creds. Anything that is not a concrete OK/SQUASH
274 * mapping (deny uid 0 / below floor, or a hard resolver error) fails closed —
275 * the broker must never impersonate on an uncertain mapping. */
276 rc = brix_idmap_resolve(NULL, req.principal, &creds, log);
277 if (rc != BRIX_IDMAP_OK && rc != BRIX_IDMAP_SQUASH) {
278 rep.status = IMP_STATUS_DENY;
279 return imp_send_reply(conn_fd, &rep, -1, NULL, 0) == 0 ? 1 : 0;
280 }
281
282 op_ctx.rootfd = rootfd;
283 op_ctx.req = &req;
284 op_ctx.rep = &rep;
285 op_ctx.out_fd = &fd;
286 op_ctx.data_out = data_buf;
287 op_ctx.data_max = sizeof(data_buf);
288 op_ctx.data_in = data_in;
289 op_ctx.data_in_len = in_len;
290 op_ctx.rel = NULL; /* imp_dispatch_op/imp_do_op derive it */
291 rep.status = imp_dispatch_op(conn_fd, &op_ctx, &creds, log);
292
293 rc = imp_send_reply(conn_fd, &rep, fd, data_buf, rep.data_len);
294 if (fd >= 0) {
295 close(fd); /* the kernel duped it into the worker */
296 }
297 return rc == 0 ? 1 : 0;
298}

302/*
303 * imp_broker_init_creds — drop privileges and capture the base identity.
304 * WHAT: minimises capabilities (and optionally drops to a service account),
305 * then records imp_base_uid/gid/groups and imp_self_uid.
306 * WHY: privileges are dropped FIRST so the base/idle identity the broker
307 * restores to between ops, and imp_self_uid (the never-impersonate-to-self
308 * guard), reflect the final, minimal identity. Fails closed if caps cannot be
309 * dropped (DAC would otherwise not be enforced).
310 * HOW: returns 0 on success, -1 to abort broker startup. No behaviour change
311 * from the inline sequence it replaces.

313static int
314imp_broker_init_creds(ngx_log_t *log)
315{
316 int ng;
317
318 if (brix_imp_broker_drop_caps(log) != 0) {
319 return -1; /* fail closed: DAC would not be enforced */
320 }
321
322 imp_base_uid = geteuid();
323 imp_base_gid = getegid();
324 imp_self_uid = getuid();
325 ng = getgroups(BRIX_IDMAP_MAXGROUPS, imp_base_groups);
326 imp_base_ngroups = (ng > 0) ? ng : 0;
327
328 return 0;
329}

332/*
333 * imp_broker_accept — admit or reject a pending worker connection.
334 * WHAT: accepts on listen_fd, applies the SO_PEERCRED gate, and registers the
335 * fd into the pollfd array if there is room.
336 * WHY: peer-credential verification is security-load-bearing; over-capacity and
337 * disallowed peers must be closed, never queued. Isolating it keeps the run
338 * loop under the complexity gate with identical admission behaviour.
339 * HOW: takes the pollfd array and its live count by pointer; on a successful
340 * admit it appends a fresh slot (events POLLIN, revents cleared) and bumps
341 * *nfds. No return value — every outcome is fully handled here.
342 */
343static void
344imp_broker_accept(int listen_fd, struct pollfd *pfds, nfds_t *nfds,
345 ngx_log_t *log)
346{
347 int c = accept4(listen_fd, NULL, NULL, SOCK_CLOEXEC);
348
349 if (c < 0) {
350 return;
351 }
352 if (!imp_peer_allowed(c)) {
353 if (log) ngx_log_error(NGX_LOG_ERR, log, 0,
354 "impersonate broker: rejected peer "
355 "(SO_PEERCRED uid mismatch)");
356 close(c);
357 return;
358 }
359 if (*nfds > IMP_BROKER_MAXCONN) {
360 close(c); /* at capacity */
361 return;
362 }
363
364 pfds[*nfds].fd = c;
365 pfds[*nfds].events = POLLIN;
366 pfds[*nfds].revents = 0; /* MUST clear: the swap-remove
367 * compaction below may move this
368 * freshly-accepted slot into an
369 * active index and re-examine it in
370 * this same pass; a stale POLLHUP
371 * left in the slot would otherwise
372 * close the brand-new connection. */
373 (*nfds)++;
374}

377/*
378 * imp_broker_reap — serve ready worker connections and compact the array.
379 * WHAT: walks slots [1, *nfds), serves POLLIN slots via imp_serve_one, and
380 * swap-removes any slot that errored, hung up, or asked to close.
381 * WHY: keeps the connection lifetime (serve→keep/close→compact) in one place,
382 * preserving the exact swap-remove ordering the accept path relies on.
383 * HOW: takes the pollfd array and its live count by pointer; decrements *nfds
384 * as slots are removed. Behaviour is byte-for-byte the prior inner loop.
385 */
386static void
387imp_broker_reap(struct pollfd *pfds, nfds_t *nfds, int rootfd, ngx_log_t *log)
388{
389 nfds_t i;
390
391 for (i = 1; i < *nfds; ) {
392 short re = pfds[i].revents;
393 int keep = 1;
394
395 if (re & (POLLERR | POLLHUP | POLLNVAL)) {
396 keep = 0;
397 } else if (re & POLLIN) {
398 keep = imp_serve_one(pfds[i].fd, rootfd, log);
399 }
400
401 if (!keep) {
402 close(pfds[i].fd);
403 pfds[i] = pfds[*nfds - 1]; /* swap-remove */
404 (*nfds)--;
405 } else {
406 i++;
407 }
408 }
409}

412int
413brix_imp_broker_run(int listen_fd, int rootfd,
414 const volatile sig_atomic_t *stop, ngx_log_t *log)
415{
416 struct pollfd pfds[IMP_BROKER_MAXCONN + 1];
417 nfds_t nfds = 1;
418
419 if (imp_broker_init_creds(log) != 0) {
420 return -1; /* fail closed: DAC would not be enforced */
421 }
422
423 pfds[0].fd = listen_fd;
424 pfds[0].events = POLLIN;
425
426 for ( ;; ) {
427 int n;
428
429 if (stop != NULL && *stop) {
430 return 0;
431 }
432 n = poll(pfds, nfds, 1000);
433 if (n < 0) {
434 if (errno == EINTR) { continue; }
435 return -1;
436 }
437 if (n == 0) {
438 continue;
439 }
440
441 /* New worker connection. */
442 if (pfds[0].revents & POLLIN) {
443 imp_broker_accept(listen_fd, pfds, &nfds, log);
444 }
445
446 /* Serve / reap worker connections (compact the array in place). */
447 imp_broker_reap(pfds, &nfds, rootfd, log);
448 }
