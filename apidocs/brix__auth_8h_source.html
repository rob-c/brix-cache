41/* Phase 40 (c): client-side credential pre-flight / failure diagnostics.
42 * brix_cred_diagnose inspects whatever credential is present locally (bearer
43 * token via brix_token_discover, then GSI proxy at the default path) WITHOUT any
44 * network call or signature verification, and prints a specific, actionable hint
45 * to `out` (each line prefixed by `prefix`) when it finds a likely auth problem:
46 * an expired/near-expiry token or proxy, or — when want_write is set — a token
47 * that grants read scope only. Returns 1 if a likely-fatal local problem was
48 * found, else 0. So the user sees "token expired 3m ago" instantly instead of a
49 * bare "permission denied".
50 * brix_cred_hint_for_status calls it (with an indented hint prefix) only when
51 * `st` carries an auth/authz wire error (kXR_NotAuthorized / kXR_AuthFailed),
52 * for one-line use at an app's error-reporting site.
53 * The _url variant additionally fires the WS-7 doctor-referral hint with the
54 * given endpoint string (nullable; sanitized before printing); the plain
55 * variant delegates with url_str=NULL. */
56int brix_cred_diagnose(int want_write, const char *prefix, FILE *out);
57void brix_cred_hint_for_status(const brix_status *st, int want_write, FILE *out);
