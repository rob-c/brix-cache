1/*
2 * entity.c — expand an authenticated identity into attribute tuples.
3 *
4 * WHAT: brix_acc_entity_build() turns scalar name/host plus the comma-separated
5 * VO/role/group lists into an brix_acc_entity_t whose `tuples` array holds
6 * one (vorg, role, grup) combination per index — the form the access engine
7 * iterates (XrdAccEntity / XrdAccEntityInfo).
8 *
9 * WHY: VOMS / token identities carry parallel multi-valued attributes (e.g. two
10 * FQANs each with its own VO+role). XrdAcc pairs them POSITIONALLY — the i-th
11 * VO with the i-th role and i-th group — not as a cartesian product, so a
12 * `cms`/`production` credential does not accidentally gain `atlas`/`production`.
13 *
14 * HOW: split each list on commas, then build N = max(list lengths) tuples; a
15 * singleton list broadcasts to every tuple, a longer list contributes element
16 * i (or NULL past its end). At least one tuple is always produced so that
17 * user/host-only rules still evaluate for attribute-less identities.
18 */
19

23/*
24 * Split a comma-separated string into an array of tokens (pool-owned). Empty
25 * fields are preserved as NULL entries so that parallel attribute lists (VO,
26 * role, group derived from the same FQANs) stay index-aligned for the positional
27 * tuple build. Empty/NULL input yields an empty array.
28 */
29static ngx_array_t *
30acc_split_csv(ngx_pool_t *pool, const char *csv)
31{
32 ngx_array_t *out = ngx_array_create(pool, 4, sizeof(char *));
33 const char *p, *start;
34
35 if (out == NULL) {
36 return NULL;
37 }
38 if (csv == NULL || *csv == '\0') {
39 return out;
40 }
41
42 p = start = csv;
43 for (;;) {
44 if (*p == ',' || *p == '\0') {
45 const char *s = start;
46 size_t len = (size_t) (p - start);
47 char **slot;
48 /* trim surrounding spaces */
49 while (len > 0 && (*s == ' ' || *s == '\t')) { s++; len--; }
50 while (len > 0 && (s[len - 1] == ' ' || s[len - 1] == '\t')) { len--; }
51
52 slot = ngx_array_push(out);
53 if (slot == NULL) {
54 return NULL;
55 }
56 if (len == 0) {
57 *slot = NULL; /* empty field -> absent attribute */
58 } else {
59 char *tok = ngx_pnalloc(pool, len + 1);
60 if (tok == NULL) {
61 return NULL;
62 }
63 ngx_memcpy(tok, s, len);
64 tok[len] = '\0';
65 *slot = tok;
66 }
67 if (*p == '\0') {
68 break;
69 }
70 start = p + 1;
71 }
72 p++;
73 }
74 return out;
75}

77/* Positional pick: element i, broadcasting a singleton, NULL past the end. */
78static const char *
79acc_pick(ngx_array_t *a, ngx_uint_t i)
80{
81 char **e;
82
83 if (a == NULL || a->nelts == 0) {
84 return NULL;
85 }
86 e = a->elts;
87 if (a->nelts == 1) {
88 return e[0];
89 }
90 return (i < a->nelts) ? e[i] : NULL;
91}

93brix_acc_entity_t *
94brix_acc_entity_build(ngx_pool_t *pool, const char *name, const char *host,
95 int isuser, const char *vorg_csv, const char *role_csv,
96 const char *grp_csv)
97{
98 brix_acc_entity_t *ent;
99 ngx_array_t *vorgs, *roles, *grps;
100 ngx_uint_t n, i;
101
102 BRIX_PCALLOC_OR_RETURN(ent, pool, sizeof(*ent), NULL);
103 ent->pool = pool;
104 ent->name = (name != NULL && *name != '\0') ? name : "*";
105 ent->host = (host != NULL && *host != '\0') ? host : "?";
106 ent->isuser = isuser && name != NULL && *name != '\0';
107
108 vorgs = acc_split_csv(pool, vorg_csv);
109 roles = acc_split_csv(pool, role_csv);
110 grps = acc_split_csv(pool, grp_csv);
111 if (vorgs == NULL || roles == NULL || grps == NULL) {
112 return NULL;
113 }
114
115 n = vorgs->nelts;
116 if (roles->nelts > n) { n = roles->nelts; }
117 if (grps->nelts > n) { n = grps->nelts; }
118 if (n == 0) { n = 1; } /* always at least one (possibly empty) tuple */
119
120 ent->tuples = ngx_array_create(pool, n, sizeof(brix_acc_attr_t));
121 if (ent->tuples == NULL) {
122 return NULL;
123 }
124 for (i = 0; i < n; i++) {
125 brix_acc_attr_t *a = ngx_array_push(ent->tuples);
126 if (a == NULL) {
127 return NULL;
128 }
129 a->vorg = acc_pick(vorgs, i);
130 a->role = acc_pick(roles, i);
131 a->grup = acc_pick(grps, i);
132 }
133
134 return ent;
135}
