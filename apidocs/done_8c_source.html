1/* File: done.c — Main-thread completion callback for native TPC pull
2 * WHAT: Handles the thread-pool-to-event-loop handoff when a TPC pull operation completes. Restores the deferred request context, then dispatches based on reply_kind: SYNC mode sends kXR_open success response with fhandle+stat data (or error if pull failed), async mode sends open success response with file handle and optional stat block including tpc.key for xrdcp extraction. On connection close during in-flight pull, releases dst_fd, unlinks dst_path, frees fhandle slot.
3 *
4 * WHY: Native TPC pull runs blocking socket I/O inside a detached thread-pool worker; the nginx event loop must resume when the thread finishes because it deferred the kXR_open response. This callback bridges the thread→event-loop boundary — restoring streamid context, updating file metadata (tpc_done flag, bytes_written, cached stat info), and sending the wire response that the client is waiting for. Sync vs async reply_kind determines whether to build a full ServerOpenBody with stat data or just send an ok status code.
5 *
6 * HOW: ev->data = ngx_thread_task_t → extract t (brix_tpc_pull_t) from task→ctx → restore request context via brix_aio_restore_request(streamid) → if connection closed, close dst_fd + unlink dst_path + free fhandle slot → reply_kind == BRIX_TPC_REPLY_SYNC: build ServerOpenBody with fhandle[idx] + optional statbuf (inode/size/permissions/mtime + tpc.key) via snprintf, allocate response buffer with ngx_palloc, build wire header + body via brix_build_resp_hdr + ngx_memcpy, queue response via brix_queue_response → async reply_kind: send ok status code only → brix_aio_resume(c) in all paths. Returns void (callback).
7 * */
8

17tpc_sess_finish_pull(brix_tpc_pull_t *t, int ok, brix_sess_end_t why)
18{
19 char errscratch[BRIX_SESSLOG_ERR_MAX];
20 const char *err;
21
22 if (t == NULL) {
23 return;
24 }
25
26 err = ok ? NULL : brix_sesslog_err_from_kxr(
27 t->xrd_error ? t->xrd_error : kXR_ServerError,
28 errscratch, sizeof(errscratch));
29 brix_sess_xfer_add(&t->sess_xfer, t->bytes_written);
30 brix_sess_result(t->sess, ok, t->src_path, BRIX_SESS_MODE_READ, err);
31 brix_sess_xfer_end(t->sess, &t->sess_xfer,
32 ok ? BRIX_SESS_XFER_COMPLETE
33 : BRIX_SESS_XFER_ABORTED);
34 brix_sess_end(t->sess, why);
35 t->sess = NULL;
36}

38/* WHAT: Main-thread completion callback — restore deferred request, dispatch SYNC vs async reply, build/send kXR_open response or error. */
39void
40brix_tpc_pull_done(ngx_event_t *ev)
41{
42 ngx_thread_task_t *task;
43 brix_tpc_pull_t *t;
44 brix_ctx_t *ctx;
45 ngx_connection_t *c;
46 int idx;
47
48 /*
49 * Re-entry point from the thread pool: nginx delivers the completion as an
50 * event whose ->data is the original ngx_thread_task_t. The pull state
51 * (brix_tpc_pull_t) lives in task->ctx; we recover ctx/connection/handle
52 * from it. From here on we are back on the event thread — no blocking I/O.
53 */
54 task = ev->data;
55 t = task->ctx;
56 ctx = t->ctx;
57 c = t->c;
58 idx = t->fhandle_idx;
59
60 /*
61 * restore_request rebinds the deferred kXR_open streamid context. If it
62 * fails the client connection went away while the pull was running in the
63 * worker thread: there is nobody to reply to, so just reclaim everything
64 * (fd, partially-written file, fhandle slot, registry entry) and bail —
65 * no brix_aio_resume(), because there is no connection to resume.
66 */
67 if (!brix_aio_restore_request(ctx, t->streamid)) {
68 tpc_sess_finish_pull(t, t->result == NGX_OK,
69 t->result == NGX_OK ? BRIX_SESS_END_SERVER
70 : BRIX_SESS_END_ERROR);
71 /* Connection closed while pull was in-flight — release resources. */
72 (void) brix_tpc_registry_update(t->transfer_id,
73 (off_t) t->bytes_written,
74 BRIX_TPC_STATE_ERROR,
75 c != NULL ? c->log : NULL);
76 brix_tpc_metric_transfer(BRIX_TPC_PROTO_STREAM,
77 BRIX_TPC_DIR_PULL,
78 BRIX_TPC_METRIC_ERROR,
79 t->bytes_written,
80 c != NULL ? c->log : NULL);
81 (void) brix_tpc_registry_remove(t->transfer_id,
82 c != NULL ? c->log : NULL);
83 close(t->dst_fd);
84 (void) brix_vfs_unlink_path(c != NULL ? c->log : NULL,
85 t->conf->common.root_canon, t->dst_path);
86 if (idx >= 0 && idx < BRIX_MAX_FILES) {
87 ctx->files[idx].fd = -1;
88 ctx->files[idx].tpc_transfer_id = 0;
89 brix_free_fhandle(ctx, idx);
90 }
91 return;
92 }
93
94 /*
95 * SYNC reply path: the pull was triggered by a kXR_sync on an already-open
96 * destination handle (the client opened the file, then sync'd to start the
97 * copy). The waiting request is the sync, so we reply with ok/error for
98 * SYNC and keep the file handle open for the client to close itself.
99 */
100 if (t->reply_kind == BRIX_TPC_REPLY_SYNC) {
101 brix_file_t *file;
102
103 /* idx may be out of range if the handle was reaped; guard before use. */
104 file = (idx >= 0 && idx < BRIX_MAX_FILES) ? &ctx->files[idx] : NULL;
105
106 if (t->result != NGX_OK) {
107 int err = t->xrd_error ? t->xrd_error : kXR_ServerError;
108
109 tpc_sess_finish_pull(t, 0, BRIX_SESS_END_ERROR);
110
111 /* Failed copy: discard the half-written destination. If the slot is
112 * still valid, free it (which also closes the fd); otherwise close
113 * the raw fd directly. Either way unlink the partial file so a
114 * failed transfer never leaves a truncated object behind. */
115 if (file != NULL) {
116 file->fd = -1;
117 file->tpc_transfer_id = 0;
118 brix_free_fhandle(ctx, idx);
119 } else {
120 close(t->dst_fd);
121 }
122 (void) brix_vfs_unlink_path(c != NULL ? c->log : NULL,
123 t->conf->common.root_canon, t->dst_path);
124
125 (void) brix_tpc_registry_update(t->transfer_id,
126 (off_t) t->bytes_written,
127 BRIX_TPC_STATE_ERROR,
128 c->log);
129 brix_tpc_metric_transfer(BRIX_TPC_PROTO_STREAM,
130 BRIX_TPC_DIR_PULL,
131 BRIX_TPC_METRIC_ERROR,
132 t->bytes_written, c->log);
133 (void) brix_tpc_registry_remove(t->transfer_id, c->log);
134
135 brix_log_access(ctx, c, "TPC-PULL", t->dst_path, "error",
136 0, (uint16_t) err,
137 t->err_msg[0] ? t->err_msg : "TPC pull failed",
138 0);
139 BRIX_OP_ERR(ctx, BRIX_OP_SYNC);
140 brix_send_error(ctx, c, (uint16_t) err,
141 t->err_msg[0] ? t->err_msg : "TPC pull failed");
142 brix_aio_resume(c);
143 return;
144 }
145
146 /* Success: mark the handle done and refresh its cached metadata from
147 * the now-complete file so a subsequent stat/read on the same handle
148 * sees the real size/inode without another path syscall (INVARIANT 7). */
149 tpc_sess_finish_pull(t, 1, BRIX_SESS_END_SERVER);
150 if (file != NULL) {
151 struct stat st;
152
153 file->tpc_done = 1;
154 file->tpc_started = 0;
155 file->tpc_transfer_id = 0;
156 file->bytes_written += t->bytes_written;
157
158 if (fstat(file->fd, &st) == 0) {
159 file->cached_size = (off_t) st.st_size;
160 file->device = st.st_dev;
161 file->inode = st.st_ino;
162 file->is_regular = S_ISREG(st.st_mode) ? 1 : 0;
163 }
164 }
165
166 brix_log_access(ctx, c, "TPC-PULL", t->dst_path, "ok",
167 1, 0, NULL, t->bytes_written);
168 BRIX_OP_OK(ctx, BRIX_OP_SYNC);
169 (void) brix_tpc_registry_update(t->transfer_id,
170 (off_t) t->bytes_written,
171 BRIX_TPC_STATE_DONE, c->log);
172 brix_tpc_metric_transfer(BRIX_TPC_PROTO_STREAM,
173 BRIX_TPC_DIR_PULL,
174 BRIX_TPC_METRIC_SUCCESS,
175 t->bytes_written, c->log);
176 (void) brix_tpc_registry_remove(t->transfer_id, c->log);
177 brix_send_ok(ctx, c, NULL, 0);
178 brix_aio_resume(c);
179 return;
180 }
181
182 /*
183 * Async reply path (reply_kind != SYNC): the pull was kicked off directly
184 * from the kXR_open, so the deferred request awaiting us is the OPEN. We
185 * therefore reply with the kXR_open result and tear the handle down
186 * ourselves on failure (the client never got a usable fhandle).
187 */
188 if (t->result != NGX_OK) {
189 int err = t->xrd_error ? t->xrd_error : kXR_ServerError;
190
191 tpc_sess_finish_pull(t, 0, BRIX_SESS_END_ERROR);
192
193 /* Failed copy: close+unlink the partial destination and release the
194 * fhandle slot before replying with the error to the OPEN. */
195 close(t->dst_fd);
196 (void) brix_vfs_unlink_path(c != NULL ? c->log : NULL,
197 t->conf->common.root_canon, t->dst_path);
198 ctx->files[idx].fd = -1;
199 ctx->files[idx].tpc_transfer_id = 0;
200 brix_free_fhandle(ctx, idx);
201
202 (void) brix_tpc_registry_update(t->transfer_id,
203 (off_t) t->bytes_written,
204 BRIX_TPC_STATE_ERROR, c->log);
205 brix_tpc_metric_transfer(BRIX_TPC_PROTO_STREAM,
206 BRIX_TPC_DIR_PULL,
207 BRIX_TPC_METRIC_ERROR,
208 t->bytes_written, c->log);
209 (void) brix_tpc_registry_remove(t->transfer_id, c->log);
210
211 brix_log_access(ctx, c, "TPC-PULL", t->dst_path, "error",
212 0, (uint16_t) err,
213 t->err_msg[0] ? t->err_msg : "TPC pull failed", 0);
214 BRIX_OP_ERR(ctx, BRIX_OP_OPEN_WR);
215 brix_send_error(ctx, c, (uint16_t) err,
216 t->err_msg[0] ? t->err_msg : "TPC pull failed");
217 brix_aio_resume(c);
218 return;
219 }
220
221 tpc_sess_finish_pull(t, 1, BRIX_SESS_END_SERVER);
222 brix_log_access(ctx, c, "TPC-PULL", t->dst_path, "ok", 1, 0, NULL, 0);
223 BRIX_OP_OK(ctx, BRIX_OP_OPEN_WR);
224 if (idx >= 0 && idx < BRIX_MAX_FILES) {
225 ctx->files[idx].tpc_transfer_id = 0;
226 }
227 (void) brix_tpc_registry_update(t->transfer_id,
228 (off_t) t->bytes_written,
229 BRIX_TPC_STATE_DONE, c->log);
230 brix_tpc_metric_transfer(BRIX_TPC_PROTO_STREAM,
231 BRIX_TPC_DIR_PULL,
232 BRIX_TPC_METRIC_SUCCESS,
233 t->bytes_written, c->log);
234 (void) brix_tpc_registry_remove(t->transfer_id, c->log);
235
236 /*
237 * Success, async path: build the deferred kXR_open ok response.
238 * Body layout: fixed ServerOpenBody (carries the 1-byte fhandle index)
239 * optionally followed by a NUL-terminated ASCII stat/tpc.key trailer.
240 * `bodylen` is grown to cover the trailer before allocating; the wire
241 * header's dlen must equal bodylen (header length is added separately).
242 */
243 {
244 ServerOpenBody body_s;
245 struct stat st;
246 char statbuf[256];
247 size_t bodylen, total;
248 u_char *buf;
249 ngx_flag_t want_stat;
250
251 want_stat = (t->options & kXR_retstat) ? 1 : 0;
252 bodylen = sizeof(ServerOpenBody);
253 statbuf[0] = '\0';
254
255 if (fstat(t->dst_fd, &st) == 0) {
256 /* xrootd stat flags are derived from POSIX mode bits. */
257 int sf = 0;
258 if (st.st_mode & (S_IRUSR | S_IRGRP | S_IROTH)) { sf |= kXR_readable; }
259 if (st.st_mode & (S_IWUSR | S_IWGRP | S_IWOTH)) { sf |= kXR_writable; }
260
261 /*
262 * Three trailer shapes depending on what the client asked for and
263 * whether we have a tpc.key:
264 * retstat + key -> "ino size flags mtime\ntpc.key=KEY"
265 * retstat only -> "ino size flags mtime"
266 * key only -> "\ntpc.key=KEY" (so xrdcp can still scrape the
267 * key even though it didn't request a stat)
268 */
269 if (want_stat && t->tpc_key[0] != '\0') {
270 snprintf(statbuf, sizeof(statbuf), "%llu %lld %d %ld\ntpc.key=%s",
271 (unsigned long long) st.st_ino, (long long) st.st_size,
272 sf, (long) st.st_mtime, t->tpc_key);
273 } else if (want_stat) {
274 snprintf(statbuf, sizeof(statbuf), "%llu %lld %d %ld",
275 (unsigned long long) st.st_ino, (long long) st.st_size,
276 sf, (long) st.st_mtime);
277 } else if (t->tpc_key[0] != '\0') {
278 /* Include key even without kXR_retstat so xrdcp can extract it. */
279 snprintf(statbuf, sizeof(statbuf), "\ntpc.key=%s", t->tpc_key);
280 }
281
282 /* +1 for the NUL terminator that the trailer carries on the wire. */
283 if (statbuf[0] != '\0') {
284 bodylen += strlen(statbuf) + 1;
285 }
286 }
287
288 total = XRD_RESPONSE_HDR_LEN + bodylen;
289 buf = ngx_palloc(c->pool, total);
290 if (buf == NULL) {
291 brix_send_error(ctx, c, kXR_NoMemory, "alloc failed");
292 brix_aio_resume(c);
293 return;
294 }
295
296 /* Assemble buf in wire order: [response header][ServerOpenBody][trailer].
297 * The header is written first with dlen=bodylen; the body follows at
298 * the fixed header offset, and the optional trailer (with its NUL) is
299 * appended immediately after the body. */
300 brix_build_resp_hdr(t->streamid, kXR_ok, (uint32_t) bodylen,
301 (ServerResponseHdr *) buf);
302
303 ngx_memzero(&body_s, sizeof(body_s));
304 body_s.fhandle[0] = (u_char) idx;
305 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body_s, sizeof(body_s));
306
307 if (statbuf[0]) {
308 size_t slen = strlen(statbuf) + 1; /* include NUL terminator */
309 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + sizeof(ServerOpenBody),
310 statbuf, slen);
311 }
312
313 brix_queue_response(ctx, c, buf, total);
314 }
315
316 brix_aio_resume(c);
317}
