1/* File: done.c — Main-thread completion callback for native TPC pull
2 * WHAT: Handles the thread-pool-to-event-loop handoff when a TPC pull operation completes. Restores the deferred request context, then dispatches based on reply_kind: SYNC mode sends kXR_open success response with fhandle+stat data (or error if pull failed), async mode sends open success response with file handle and optional stat block including tpc.key for xrdcp extraction. On connection close during in-flight pull, releases dst_fd, unlinks dst_path, frees fhandle slot.
3 *
4 * WHY: Native TPC pull runs blocking socket I/O inside a detached thread-pool worker; the nginx event loop must resume when the thread finishes because it deferred the kXR_open response. This callback bridges the thread→event-loop boundary — restoring streamid context, updating file metadata (tpc_done flag, bytes_written, cached stat info), and sending the wire response that the client is waiting for. Sync vs async reply_kind determines whether to build a full ServerOpenBody with stat data or just send an ok status code.
5 *
6 * HOW: ev->data = ngx_thread_task_t → extract t (brix_tpc_pull_t) from task→ctx → restore request context via brix_aio_restore_request(streamid) → if connection closed, close dst_fd + unlink dst_path + free fhandle slot → reply_kind == BRIX_TPC_REPLY_SYNC: build ServerOpenBody with fhandle[idx] + optional statbuf (inode/size/permissions/mtime + tpc.key) via snprintf, allocate response buffer with ngx_palloc, build wire header + body via brix_build_resp_hdr + ngx_memcpy, queue response via brix_queue_response → async reply_kind: send ok status code only → brix_aio_resume(c) in all paths. Returns void (callback).
7 * */

38/* WHAT: Finish transfer accounting — registry state update, TPC metric, registry remove.
39 * WHY: Every completion path (orphaned, connection-gone, sync ok/fail, async ok/fail)
40 * performs exactly this triple with the same log target; centralizing it keeps the
41 * registry and metrics in lockstep across all outcomes.
42 * HOW: ok selects DONE+SUCCESS vs ERROR+ERROR; bytes_written is reported both to the
43 * registry (as offset) and to the metric, then the registry entry is removed. */
44static void

46{
47 (void) brix_tpc_registry_update(t->transfer_id,
48 (off_t) t->bytes_written,
49 ok ? BRIX_TPC_STATE_DONE
50 : BRIX_TPC_STATE_ERROR, log);
51 brix_tpc_metric_transfer(BRIX_TPC_PROTO_STREAM,
52 BRIX_TPC_DIR_PULL,
53 ok ? BRIX_TPC_METRIC_SUCCESS

59/* WHAT: Reclaim the destination of a failed/abandoned pull — close fd, unlink the
60 * partial file, release the fhandle slot when a live ctx still owns one.
61 * WHY: A failed transfer must never leave a truncated object or a leaked fd/fhandle
62 * behind; the same close→unlink→slot-free sequence is shared by the orphaned,
63 * connection-gone and async-failure paths.
64 * HOW: close(dst_fd) then confined unlink of dst_path under root_canon; if ctx is
65 * still present and idx is a valid slot, clear fd/transfer_id and free the
66 * fhandle (fd already -1, so the free cannot double-close). Pass ctx=NULL when
67 * there is no fhandle table to touch. */
68static void
69tpc_done_teardown_dst(brix_tpc_pull_t *t, brix_ctx_t *ctx, int idx,
70 ngx_log_t *log)
71{
72 close(t->dst_fd);
73 (void) brix_vfs_unlink_path(log, t->conf->common.root_canon,
74 t->dst_path);
75 if (ctx != NULL && idx >= 0 && idx < BRIX_MAX_FILES) {
76 ctx->files[idx].fd = -1;
77 ctx->files[idx].tpc_transfer_id = 0;
78 brix_free_fhandle(ctx, idx);
79 }
80}

82/* WHAT: SYNC-mode failure reply — discard the half-written destination, account the
83 * error, and answer the waiting kXR_sync with the pull's error code.
84 * WHY: In SYNC mode the client already holds the open handle; on failure we must
85 * free the slot (or close the raw fd if the slot was reaped) and unlink the
86 * partial file so a failed transfer never leaves a truncated object behind.
87 * HOW: slot-free XOR raw close depending on whether the handle survived, confined
88 * unlink, then registry/metric accounting and the error response + aio resume.
89 * c is never NULL here: brix_tpc_pull_done gates the no-connection case before
90 * dispatching to any reply helper. */
91static void
92tpc_done_sync_fail(brix_tpc_pull_t *t, brix_ctx_t *ctx, ngx_connection_t *c,
93 int idx)
94{
95 brix_file_t *file;
96 int err = t->xrd_error ? t->xrd_error : kXR_ServerError;
97
98 /* idx may be out of range if the handle was reaped; guard before use. */
99 file = (idx >= 0 && idx < BRIX_MAX_FILES) ? &ctx->files[idx] : NULL;
100
101 tpc_sess_finish_pull(t, 0, BRIX_SESS_END_ERROR);
102
103 /* Failed copy: discard the half-written destination. If the slot is
104 * still valid, free it (which also closes the fd); otherwise close
105 * the raw fd directly. Either way unlink the partial file so a
106 * failed transfer never leaves a truncated object behind. */
107 if (file != NULL) {
108 file->fd = -1;
109 file->tpc_transfer_id = 0;
110 brix_free_fhandle(ctx, idx);
111 } else {
112 close(t->dst_fd);
113 }
114 (void) brix_vfs_unlink_path(c->log, t->conf->common.root_canon, t->dst_path);
115
116 tpc_done_account(t, 0, c->log);
117
118 brix_log_access(ctx, c, "TPC-PULL", t->dst_path, "error",
119 0, (uint16_t) err,
120 t->err_msg[0] ? t->err_msg : "TPC pull failed",
121 0);
122 BRIX_OP_ERR(ctx, BRIX_OP_SYNC);
123 brix_send_error(ctx, c, (uint16_t) err,
124 t->err_msg[0] ? t->err_msg : "TPC pull failed");
125 brix_aio_resume(c);
126}

128/* WHAT: SYNC-mode reply — the waiting request is a kXR_sync on an already-open
129 * destination handle; answer it with ok/error and keep the handle open.
130 * WHY: The pull was triggered by a kXR_sync (the client opened the file, then
131 * sync'd to start the copy), so the client closes the handle itself; on
132 * success we refresh the handle's cached metadata from the completed file.
133 * HOW: failure delegates to tpc_done_sync_fail; success marks the handle done,
134 * accumulates bytes_written, fstat-refreshes cached size/inode so subsequent
135 * stat/read on the handle needs no path syscall (INVARIANT 7), then accounts
136 * the transfer and sends ok + aio resume. */
137static void
138tpc_done_reply_sync(brix_tpc_pull_t *t, brix_ctx_t *ctx, ngx_connection_t *c,
139 int idx)
140{
141 brix_file_t *file;
142
143 if (t->result != NGX_OK) {
144 tpc_done_sync_fail(t, ctx, c, idx);
145 return;
146 }
147
148 /* idx may be out of range if the handle was reaped; guard before use. */
149 file = (idx >= 0 && idx < BRIX_MAX_FILES) ? &ctx->files[idx] : NULL;
150
151 /* Success: mark the handle done and refresh its cached metadata from
152 * the now-complete file so a subsequent stat/read on the same handle
153 * sees the real size/inode without another path syscall (INVARIANT 7). */
154 tpc_sess_finish_pull(t, 1, BRIX_SESS_END_SERVER);
155 if (file != NULL) {
156 struct stat st;
157
158 file->tpc_done = 1;
159 file->tpc_started = 0;
160 file->tpc_transfer_id = 0;
161 file->bytes_written += t->bytes_written;
162
163 if (fstat(file->fd, &st) == 0) {
164 file->cached_size = (off_t) st.st_size;
165 file->device = st.st_dev;
166 file->inode = st.st_ino;
167 file->is_regular = S_ISREG(st.st_mode) ? 1 : 0;
168 }
169 }
170
171 brix_log_access(ctx, c, "TPC-PULL", t->dst_path, "ok",
172 1, 0, NULL, t->bytes_written);
173 BRIX_OP_OK(ctx, BRIX_OP_SYNC);
174 tpc_done_account(t, 1, c->log);
175 brix_send_ok(ctx, c, NULL, 0);
176 brix_aio_resume(c);
177}

179/* WHAT: Format the optional ASCII stat/tpc.key trailer for the kXR_open ok body.
180 * WHY: xrdcp scrapes tpc.key from the open response, and kXR_retstat clients need
181 * the "ino size flags mtime" stat line; the trailer shape depends on which of
182 * the two the client can consume.
183 * HOW: fstat(dst_fd) → derive xrootd readable/writable flags from POSIX mode bits →
184 * snprintf one of three trailer shapes:
185 * retstat + key -> "ino size flags mtime\ntpc.key=KEY"
186 * retstat only -> "ino size flags mtime"
187 * key only -> "\ntpc.key=KEY" (so xrdcp can still scrape the key even
188 * though it didn't request a stat)
189 * On fstat failure or neither-wanted, statbuf stays empty. */
190static void
191tpc_done_stat_trailer(brix_tpc_pull_t *t, char *statbuf, size_t buflen)
192{
193 struct stat st;
194 ngx_flag_t want_stat = (t->options & kXR_retstat) ? 1 : 0;
195 int sf = 0;
196
197 statbuf[0] = '\0';
198
199 if (fstat(t->dst_fd, &st) != 0) {
200 return;
201 }
202
203 /* xrootd stat flags are derived from POSIX mode bits. */
204 if (st.st_mode & (S_IRUSR | S_IRGRP | S_IROTH)) { sf |= kXR_readable; }
205 if (st.st_mode & (S_IWUSR | S_IWGRP | S_IWOTH)) { sf |= kXR_writable; }
206
207 if (want_stat && t->tpc_key[0] != '\0') {
208 snprintf(statbuf, buflen, "%llu %lld %d %ld\ntpc.key=%s",
209 (unsigned long long) st.st_ino, (long long) st.st_size,
210 sf, (long) st.st_mtime, t->tpc_key);
211 } else if (want_stat) {
212 snprintf(statbuf, buflen, "%llu %lld %d %ld",
213 (unsigned long long) st.st_ino, (long long) st.st_size,
214 sf, (long) st.st_mtime);
215 } else if (t->tpc_key[0] != '\0') {
216 /* Include key even without kXR_retstat so xrdcp can extract it. */
217 snprintf(statbuf, buflen, "\ntpc.key=%s", t->tpc_key);
218 }
219}

221/* WHAT: Build and queue the deferred kXR_open ok response for the async path.
222 * WHY: The client's OPEN was deferred while the pull ran in the worker thread;
223 * it now needs the ServerOpenBody carrying its fhandle index, plus the
224 * optional stat/tpc.key trailer so xrdcp can extract the key.
225 * HOW: Body layout: fixed ServerOpenBody (carries the 1-byte fhandle index)
226 * optionally followed by a NUL-terminated ASCII stat/tpc.key trailer.
227 * `bodylen` is grown to cover the trailer before allocating; the wire
228 * header's dlen must equal bodylen (header length is added separately).
229 * Assemble buf in wire order: [response header][ServerOpenBody][trailer] —
230 * header first with dlen=bodylen, body at the fixed header offset, optional
231 * trailer (with its NUL) appended immediately after the body — then queue
232 * it and resume the connection. Alloc failure replies kXR_NoMemory. */
233static void
234tpc_done_send_open_ok(brix_tpc_pull_t *t, brix_ctx_t *ctx,
235 ngx_connection_t *c, int idx)
236{
237 ServerOpenBody body_s;
238 char statbuf[256];
239 size_t bodylen, total;
240 u_char *buf;
241
242 tpc_done_stat_trailer(t, statbuf, sizeof(statbuf));
243
244 bodylen = sizeof(ServerOpenBody);
245 /* +1 for the NUL terminator that the trailer carries on the wire. */
246 if (statbuf[0] != '\0') {
247 bodylen += strlen(statbuf) + 1;
248 }
249
250 total = XRD_RESPONSE_HDR_LEN + bodylen;
251 buf = ngx_palloc(c->pool, total);
252 if (buf == NULL) {
253 brix_send_error(ctx, c, kXR_NoMemory, "alloc failed");
254 brix_aio_resume(c);
255 return;
256 }
257
258 brix_build_resp_hdr(t->streamid, kXR_ok, (uint32_t) bodylen,
259 (ServerResponseHdr *) buf);
260
261 ngx_memzero(&body_s, sizeof(body_s));
262 body_s.fhandle[0] = (u_char) idx;
263 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body_s, sizeof(body_s));
264
265 if (statbuf[0]) {
266 size_t slen = strlen(statbuf) + 1; /* include NUL terminator */
267 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + sizeof(ServerOpenBody),
268 statbuf, slen);
269 }
270
271 brix_queue_response(ctx, c, buf, total);
272 brix_aio_resume(c);
273}

275/* WHAT: Async-mode reply — the deferred request is the kXR_open itself; answer
276 * it with the pull result, tearing the handle down ourselves on failure.
277 * WHY: The pull was kicked off directly from the kXR_open, so on failure the
278 * client never got a usable fhandle — we must close+unlink the partial
279 * destination and release the slot before replying with the error.
280 * HOW: failure: session finish, destination teardown, accounting, access log,
281 * error response, aio resume. success: session finish, access log, clear
282 * the slot's transfer id, accounting, then build/queue the open ok
283 * response (with stat/tpc.key trailer) via tpc_done_send_open_ok. */
284static void
285tpc_done_reply_open(brix_tpc_pull_t *t, brix_ctx_t *ctx, ngx_connection_t *c,
286 int idx)
287{
288 if (t->result != NGX_OK) {
289 int err = t->xrd_error ? t->xrd_error : kXR_ServerError;
290
291 tpc_sess_finish_pull(t, 0, BRIX_SESS_END_ERROR);
292
293 /* Failed copy: close+unlink the partial destination and release the
294 * fhandle slot before replying with the error to the OPEN. (c is never
295 * NULL here — brix_tpc_pull_done gates the no-connection case.) */
296 tpc_done_teardown_dst(t, ctx, idx, c->log);
297
298 tpc_done_account(t, 0, c->log);
299
300 brix_log_access(ctx, c, "TPC-PULL", t->dst_path, "error",
301 0, (uint16_t) err,
302 t->err_msg[0] ? t->err_msg : "TPC pull failed", 0);
303 BRIX_OP_ERR(ctx, BRIX_OP_OPEN_WR);
304 brix_send_error(ctx, c, (uint16_t) err,
305 t->err_msg[0] ? t->err_msg : "TPC pull failed");
306 brix_aio_resume(c);
307 return;
308 }
309
310 tpc_sess_finish_pull(t, 1, BRIX_SESS_END_SERVER);
311 brix_log_access(ctx, c, "TPC-PULL", t->dst_path, "ok", 1, 0, NULL, 0);
312 BRIX_OP_OK(ctx, BRIX_OP_OPEN_WR);
313 if (idx >= 0 && idx < BRIX_MAX_FILES) {
314 ctx->files[idx].tpc_transfer_id = 0;
315 }
316 tpc_done_account(t, 1, c->log);
317
318 tpc_done_send_open_ok(t, ctx, c, idx);
319}

321/* WHAT: Main-thread completion callback — restore deferred request, dispatch SYNC vs async reply, build/send kXR_open response or error. */
322void
323brix_tpc_pull_done(ngx_event_t *ev)
324{
325 ngx_thread_task_t *task;
326 brix_tpc_pull_t *t;
327 brix_ctx_t *ctx;
328 ngx_connection_t *c;
329 int idx;
330
331 /*
332 * Re-entry point from the thread pool: nginx delivers the completion as an
333 * event whose ->data is the original ngx_thread_task_t. The pull state
334 * (brix_tpc_pull_t) lives in task->ctx; we recover ctx/connection/handle
335 * from it. From here on we are back on the event thread — no blocking I/O.
336 */
337 task = ev->data;
338 t = task->ctx;
339 ctx = t->ctx;
340 c = t->c;
341 idx = t->fhandle_idx;
342
343 /*
344 * Defensive: the pull state can outlive its session/connection (the worker
345 * thread completes after a teardown). With no ctx/connection there is
346 * nobody to reply to and no fhandle table to touch — finish the transfer
347 * bookkeeping against the cycle log and reclaim the destination.
348 */
349 if (ctx == NULL || c == NULL) {
350 ngx_log_t *log = (c != NULL) ? c->log : ngx_cycle->log;
351
352 tpc_sess_finish_pull(t, 0, BRIX_SESS_END_ERROR);
353 tpc_done_account(t, 0, log);
354 tpc_done_teardown_dst(t, NULL, idx, log);
355 return;
356 }
357
358 /*
359 * restore_request rebinds the deferred kXR_open streamid context. If it
360 * fails the client connection went away while the pull was running in the
361 * worker thread: there is nobody to reply to, so just reclaim everything
362 * (fd, partially-written file, fhandle slot, registry entry) and bail —
363 * no brix_aio_resume(), because there is no connection to resume.
364 */
365 if (!brix_aio_restore_request(ctx, t->streamid)) {
366 tpc_sess_finish_pull(t, t->result == NGX_OK,
367 t->result == NGX_OK ? BRIX_SESS_END_SERVER
368 : BRIX_SESS_END_ERROR);
369 /* Connection closed while pull was in-flight — release resources. */
370 tpc_done_account(t, 0, c->log);
371 tpc_done_teardown_dst(t, ctx, idx, c->log);
372 return;
373 }
374
375 /*
376 * SYNC reply path: the pull was triggered by a kXR_sync on an already-open
377 * destination handle (the client opened the file, then sync'd to start the
378 * copy). The waiting request is the sync, so we reply with ok/error for
379 * SYNC and keep the file handle open for the client to close itself.
380 *
381 * Async reply path (reply_kind != SYNC): the pull was kicked off directly
382 * from the kXR_open, so the deferred request awaiting us is the OPEN.
383 */
384 if (t->reply_kind == BRIX_TPC_REPLY_SYNC) {
385 tpc_done_reply_sync(t, ctx, c, idx);
386 } else {
387 tpc_done_reply_open(t, ctx, c, idx);
388 }
389}
