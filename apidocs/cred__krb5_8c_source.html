1/* client/lib/cred_krb5.c
2 *
3 * WHAT: Kerberos 5 credential handler for the unified credential store (cred.c).
4 * Implements brix_cred_krb5() returning the XRDC_CRED_KRB5 handler with
5 * available / acquire operations. Compile-gated on BRIX_HAVE_KRB5.
6 * WHY: ccache discovery was only available through sec_krb5.c; the credential
7 * store needs a standalone handler so auth pre-flight diagnostics and the
8 * auth handshake both share one ccache resolution path.
9 * HOW: Discovery precedence:
10 * cfg->ccache (CLI override, highest precedence)
11 * > $KRB5CCNAME (environment)
12 * > krb5_cc_default (library default)
13 * available() opens the ccache and verifies it has a client principal.
14 * acquire() fills out->path with "<type>:<name>" in a static buffer and
15 * best-effort reads the TGT endtime via krb5_cc_next_cred (krbtgt/... entry).
16 * refresh=NULL: TGT renewal (kinit) is the user's responsibility; this
17 * handler does NOT invoke kinit or perform any ticket refresh.
18 * When BRIX_HAVE_KRB5 is not defined at compile time, brix_cred_krb5()
19 * returns NULL so the store simply skips the KRB5 slot.
20 *
21 * Lifetime contract (cred.h): acquire() writes out->path as a pointer to
22 * a function-local static char[] that remains valid after acquire() returns.
23 * The store's slot_store_view strdup's out->path immediately afterwards —
24 * before any other acquire can overwrite the buffer.
25 *
26 * Mirrors brix_sec_krb5() in lib/sec/sec_krb5.c for all krb5 API calls.
27 * ngx-free. No goto. Functional/modular: one responsibility per function.
28 */

43/* ccache resolution */
44/*
45 * open_ccache — open the ccache selected by cfg->ccache > $KRB5CCNAME >
46 * krb5_cc_default and write it into *cc.
47 *
48 * WHAT: resolves the caller's preference into an open krb5_ccache handle.
49 * WHY: single resolution path reused by both available() and acquire() so the
50 * two cannot drift apart.
51 * HOW: three early-return branches; caller owns *cc on success (must close).
52 */
53static int
54open_ccache(krb5_context ctx, const brix_cred_config *cfg, krb5_ccache *cc)
55{
56 const char *env;
57
58 if (cfg != NULL && cfg->ccache != NULL && cfg->ccache[0] != '\0') {
59 return krb5_cc_resolve(ctx, cfg->ccache, cc) == 0 ? 0 : -1;
60 }
61 env = getenv("KRB5CCNAME");
62 if (env != NULL && env[0] != '\0') {
63 return krb5_cc_resolve(ctx, env, cc) == 0 ? 0 : -1;
64 }
65 return krb5_cc_default(ctx, cc) == 0 ? 0 : -1;
66}

68/* ccache name formatting */
69/*
70 * format_ccache_name — write the canonical "TYPE:name" ccache name into out[outsz].
71 *
72 * WHAT: combines krb5_cc_get_type and krb5_cc_get_name into one NUL-terminated
73 * string (e.g. "FILE:/tmp/krb5cc_1000").
74 * WHY: out->path should be the form users recognise from $KRB5CCNAME, not just
75 * the bare residual path.
76 * HOW: two-part snprintf; always NUL-terminated within outsz.
77 */
78static void
79format_ccache_name(krb5_context ctx, krb5_ccache cc, char *out, size_t outsz)
80{
81 const char *type = krb5_cc_get_type(ctx, cc);
82 const char *name = krb5_cc_get_name(ctx, cc);
83
84 if (type != NULL && type[0] != '\0' && name != NULL) {
85 snprintf(out, outsz, "%s:%s", type, name);
86 } else if (name != NULL) {
87 snprintf(out, outsz, "%s", name);
88 } else {
89 snprintf(out, outsz, "(unknown-ccache)");
90 }
91}

93/* TGT endtime extraction */
94/*
95 * extract_tgt_endtime — scan the ccache for the krbtgt/... credential and return
96 * its endtime, or 0 if not found / unreadable.
97 *
98 * WHAT: best-effort TGT expiry to populate the store's not_after field.
99 * WHY: lets the store's expiry gate warn before the TGT expires mid-session and
100 * trigger auto-refresh (kinit delegation) before a transfer stalls.
101 * HOW: krb5_cc_start_seq_get; iterate with krb5_cc_next_cred; check the first
102 * component of creds.server for "krbtgt"; capture endtime on match.
103 * All cred contents and the cursor are freed on every exit path.
104 */
105static int64_t
106extract_tgt_endtime(krb5_context ctx, krb5_ccache cc)
107{
108 krb5_cc_cursor cursor;
109 krb5_creds creds;
110 int64_t endtime = 0;
111 krb5_error_code krc;
112
113 if (krb5_cc_start_seq_get(ctx, cc, &cursor) != 0) {
114 return 0;
115 }
116
117 while ((krc = krb5_cc_next_cred(ctx, cc, &cursor, &creds)) == 0) {
118 krb5_data *comp = krb5_princ_component(ctx, creds.server, 0);
119 if (comp != NULL
120 && comp->length == 6 /* strlen("krbtgt") */
121 && memcmp(comp->data, "krbtgt", 6) == 0) {
122 endtime = (int64_t)creds.times.endtime;
123 krb5_free_cred_contents(ctx, &creds);
124 break;
125 }
126 krb5_free_cred_contents(ctx, &creds);
127 }
128
129 krb5_cc_end_seq_get(ctx, cc, &cursor);
130 return endtime;
131}

133/* handler callbacks */
134/*
135 * krb5_available — 1 if the resolved ccache has a usable client principal.
136 *
137 * WHAT: fast probe for auth pre-flight diagnostics and brix_cred_available().
138 * WHY: mirrors sec_krb5.c:krb5_have() semantics (open ccache + get_principal).
139 * HOW: init context; open_ccache; krb5_cc_get_principal; free all on every path.
140 */
141static int
142krb5_available(const brix_cred_config *cfg)
143{
144 krb5_context ctx;
145 krb5_ccache cc;
146 krb5_principal me = NULL;
147 int ok = 0;
148
149 if (krb5_init_context(&ctx) != 0) {
150 return 0;
151 }
152 if (open_ccache(ctx, cfg, &cc) == 0) {
153 if (krb5_cc_get_principal(ctx, cc, &me) == 0) {
154 ok = 1;
155 krb5_free_principal(ctx, me);
156 }
157 krb5_cc_close(ctx, cc);
158 }
159 krb5_free_context(ctx);
160 return ok;
161}

163/*
164 * krb5_acquire — open the ccache, verify it has a principal, fill *out.
165 *
166 * WHAT: the store's acquire call for XRDC_CRED_KRB5.
167 * WHY: single canonical acquire path for the Kerberos credential.
168 * HOW: 1) krb5_init_context (error → XRDC_EAUTH);
169 * 2) open_ccache (error → XRDC_EAUTH);
170 * 3) krb5_cc_get_principal to confirm a live principal (error → XRDC_EAUTH);
171 * 4) format ccache name into s_path;
172 * 5) extract_tgt_endtime (best-effort; 0 on failure);
173 * 6) free all krb5 objects;
174 * 7) fill *out; return 0.
175 *
176 * Lifetime of s_path: declared static so the buffer remains valid after this
177 * function returns. slot_store_view strdup's out->path before any other acquire
178 * can overwrite the buffer.
179 *
180 * Note: TGT renewal (kinit) is the user's responsibility. refresh=NULL is
181 * intentional — this handler never invokes kinit or requests a new TGT.
182 */
183static int
184krb5_acquire(const brix_cred_config *cfg, brix_cred_view *out,
185 int64_t *not_after, brix_status *st)
186{
187 /* static: must outlive this return so slot_store_view can strdup it */
188 static char s_path[XRDC_PATH_MAX];
189
190 krb5_context ctx;
191 krb5_ccache cc;
192 krb5_principal me = NULL;
193 int64_t endtime;
194
195 if (krb5_init_context(&ctx) != 0) {
196 brix_status_set(st, XRDC_EAUTH, 0, "krb5: cannot initialise context");
197 return -1;
198 }
199
200 if (open_ccache(ctx, cfg, &cc) != 0) {
201 krb5_free_context(ctx);
202 brix_status_set(st, XRDC_EAUTH, 0,
203 "krb5: no credential cache (run kinit)");
204 return -1;
205 }
206
207 if (krb5_cc_get_principal(ctx, cc, &me) != 0) {
208 krb5_cc_close(ctx, cc);
209 krb5_free_context(ctx);
210 brix_status_set(st, XRDC_EAUTH, 0,
211 "krb5: ccache has no principal (run kinit)");
212 return -1;
213 }
214 krb5_free_principal(ctx, me);
215
216 format_ccache_name(ctx, cc, s_path, sizeof(s_path));
217 endtime = extract_tgt_endtime(ctx, cc);
218
219 krb5_cc_close(ctx, cc);
220 krb5_free_context(ctx);
221
222 out->kind = XRDC_CRED_KRB5;
223 out->path = s_path; /* static buffer; store deep-copies via slot_store_view */
224 out->token = NULL;
225 out->s3_access = NULL;
226 out->s3_secret = NULL;
227
228 *not_after = endtime; /* 0 when TGT endtime could not be read (non-fatal) */
229 return 0;
230}
