1/*
2 * webdav/webdav_metrics.h
3 *
4 * WebDAV per-request metric helpers: classify the method slot, count arrivals,
5 * record response outcomes, and the status-only / finalise convenience tails.
6 * Split out of webdav.h so the metrics surface is grouped by concern and
7 * individually reviewable. Includes webdav.h for the shared request types.
8 */
9

12
13#include "webdav.h"
14
15/* Classify the request method into the WebDAV metric slot (the requests_total
16 * index); BRIX_WEBDAV_METHOD_OTHER for unrecognised verbs. */
17ngx_uint_t webdav_metrics_method(ngx_http_request_t *r);
18/* Count this request's arrival (requests_total[method]). */
19void webdav_metrics_request(ngx_http_request_t *r);
20/* Record the outcome (responses_total[method][status-class] + unified op),
21 * deriving the status from rc/headers_out. No-op when rc == NGX_DONE (the
22 * request already accounted for itself). */
23void webdav_metrics_response(ngx_http_request_t *r, ngx_int_t rc);
24/* webdav_metrics_response(r, rc) then return rc — convenience for handler tails. */
25ngx_int_t webdav_metrics_return(ngx_http_request_t *r, ngx_int_t rc);
26/* Emit a status-only (empty-body) response — set status + zero content length,
27 * send the header, and finalise via the send_special result (records response
28 * metrics). Finalises the request: must be the caller's last action on `r`.
29 * Set any extra headers (e.g. Location for 201) before calling. */
30void webdav_send_status_only(ngx_http_request_t *r, ngx_uint_t status);
31/* webdav_metrics_response(r, rc) then ngx_http_finalize_request(r, rc) — for
32 * async/self-finalising paths. */
33void webdav_metrics_finalize_request(ngx_http_request_t *r, ngx_int_t rc);
34
35#endif /* NGX_HTTP_BRIX_WEBDAV_METRICS_H */
