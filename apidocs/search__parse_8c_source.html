19/* Return the first direct child element of `parent` with the given local name,
20 * or NULL. NULL-safe in `parent` so callers can chain lookups without
21 * intermediate NULL checks. */
22static xmlNodePtr
23webdav_search_find_child(xmlNodePtr parent, const char *name)
24{
25 xmlNodePtr n;
26
27 for (n = parent != NULL ? parent->children : NULL; n != NULL; n = n->next) {
28 if (n->type == XML_ELEMENT_NODE
29 && xmlStrcmp(n->name, BAD_CAST name) == 0)
30 {
31 return n;
32 }
33 }
34
35 return NULL;
36}

38/*
39 * Depth-first search the subtree for the first DAV:literal element and copy its
40 * text content into q->literal (truncated to the fixed buffer). Stops at the
41 * first match: the supported query subset has only one literal term. q->literal
42 * being non-empty is the sentinel that unwinds the recursion early.
43 */
44static void
45webdav_search_find_literal(xmlNodePtr n, webdav_search_query_t *q)
46{
47 for (; n != NULL; n = n->next) {
48 if (n->type == XML_ELEMENT_NODE
49 && xmlStrcmp(n->name, BAD_CAST "literal") == 0)
50 {
51 xmlChar *text = xmlNodeGetContent(n);
52 if (text != NULL) {
53 ngx_cpystrn((u_char *) q->literal, text, sizeof(q->literal));
54 xmlFree(text);
55 }
56 return;
57 }
58
59 webdav_search_find_literal(n->children, q);
60 if (q->literal[0] != '\0') {
61 return;
62 }
63 }
64}

66/*
67 * Parse the SEARCH request body into a query descriptor.
68 * Expects DAV:searchrequest > DAV:basicsearch; extracts the scope depth
69 * (0 / 1 / infinity) and an optional DAV:contains literal. Returns
70 * NGX_HTTP_BAD_REQUEST for a missing/oversized/malformed body, NGX_OK otherwise.
71 * The libxml2 parse flags are the same hardened set as PROPFIND (no network,
72 * no external entities, no HUGE) to block entity-expansion DoS.
73 */
74ngx_int_t
75webdav_search_parse(ngx_http_request_t *r, webdav_search_query_t *q)
76{
77 u_char *body = NULL;
78 size_t body_len = 0;
79 ngx_int_t rc;
80 xmlDocPtr doc;
81 xmlNodePtr root, basic, from, scope, depth, where;
82 int opts = XML_PARSE_NONET | XML_PARSE_NOERROR | XML_PARSE_NOWARNING;
83#if defined(XML_PARSE_NO_XXE)
84 opts |= XML_PARSE_NO_XXE;
85#endif
86
87 q->depth = 0;
88 q->literal[0] = '\0';
89
90 rc = brix_http_body_read_all(r, WEBDAV_SEARCH_BODY_MAX, &body, &body_len);
91 if (rc != NGX_OK || body_len == 0) {
92 return NGX_HTTP_BAD_REQUEST;
93 }
94
95 doc = xmlReadMemory((const char *) body, (int) body_len,
96 "search.xml", NULL, opts);
97 if (doc == NULL) {
98 return NGX_HTTP_BAD_REQUEST;
99 }
100
101 root = xmlDocGetRootElement(doc);
102 basic = webdav_search_find_child(root, "basicsearch");
103 if (root == NULL || xmlStrcmp(root->name, BAD_CAST "searchrequest") != 0
104 || basic == NULL)
105 {
106 xmlFreeDoc(doc);
107 return NGX_HTTP_BAD_REQUEST;
108 }
109
110 from = webdav_search_find_child(basic, "from");
111 scope = webdav_search_find_child(from, "scope");
112 depth = webdav_search_find_child(scope, "depth");
113 if (depth != NULL) {
114 xmlChar *text = xmlNodeGetContent(depth);
115 if (text != NULL) {
116 if (xmlStrcmp(text, BAD_CAST "1") == 0) {
117 q->depth = 1;
118 } else if (xmlStrcmp(text, BAD_CAST "infinity") == 0) {
119 q->depth = -1;
120 }
121 xmlFree(text);
122 }
123 }
124
125 where = webdav_search_find_child(basic, "where");
126 if (where != NULL) {
127 webdav_search_find_literal(where->children, q);
128 }
129
130 xmlFreeDoc(doc);
131 return NGX_OK;
132}
