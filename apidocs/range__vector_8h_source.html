1/*
2 * range_vector.h — shared byte-range parsing, validation, and coalescing.
3 *
4 * Unifies range handling across:
5 * - XrdHttp multipart/byteranges GET (src/webdav/xrdhttp_multipart.c)
6 * - HTTP single-range responses (src/compat/range.c)
7 * - Native kXR_readv coalescing (src/read/readv.c)
8 *
9 * Serialisation (multipart MIME boundaries, wire headers, ngx_chain_t
10 * construction) stays protocol-specific; this layer handles only:
11 * 1. Parsing comma-separated HTTP byte ranges.
12 * 2. Summing and validating total response size.
13 * 3. Finding runs of contiguous ranges for I/O coalescing.
14 */
15

21/*
22 * brix_byte_range_t — a normalised, inclusive byte interval.
23 *
24 * start, end: byte offsets, both inclusive. Always start ≤ end after
25 * successful parse; end < file_size.
26 * fd: optional open file descriptor for I/O planning (-1 if unused)
27 * handle: optional native kXR_readv file handle (0 if unused)
28 */

31 off_t end;

33 ngx_uint_t handle;

36/*
37 * brix_range_vector_opts_t — caller-supplied policy for the parser.
38 *
39 * max_ranges: hard cap on number of entries stored (extra silently dropped)
40 * max_total_bytes: 0 → no limit; non-zero → reject vectors exceeding this
41 * allow_suffix: 1 → accept "-N" suffix ranges
42 * allow_open_ended: 1 → accept "A-" ranges (end mapped to file_size−1)
43 * drop_unsatisfiable: 1 → silently skip bad ranges; 0 → return NGX_ERROR
44 */
45typedef struct {
46 ngx_uint_t max_ranges;
47 off_t max_total_bytes;
48 ngx_flag_t allow_suffix;
49 ngx_flag_t allow_open_ended;
50 ngx_flag_t drop_unsatisfiable;
51} brix_range_vector_opts_t;

53/*
54 * brix_http_parse_range_vector — parse a HTTP byte-ranges header value.
55 *
56 * data/len: the header value WITHOUT the leading "bytes=" prefix.
57 * Callers must skip past "bytes=" before calling.
58 * file_size: the resource length; used to normalise suffix and open-ended
59 * ranges and to drop unsatisfiable ones.
60 * opts: parsing policy (see brix_range_vector_opts_t above).
61 * ranges: caller-supplied array of at least opts->max_ranges entries.
62 * nranges: set to the number of valid ranges stored.
63 *
64 * Returns NGX_OK on success (nranges may be 0 if all ranges were dropped).
65 * Returns NGX_ERROR if opts->drop_unsatisfiable is 0 and a bad range is seen.
66 */
67ngx_int_t brix_http_parse_range_vector(const u_char *data, size_t len,
68 off_t file_size, const brix_range_vector_opts_t *opts,
69 brix_byte_range_t *ranges, ngx_uint_t *nranges);
