4/*
5 * dashboard/dashboard_tracking.h — HTTP request → dashboard transfer-slot API.
6 *
7 * Declares the brix_dashboard_http_*() functions that WebDAV/S3/TPC handlers
8 * call to bind an ngx_http_request_t to a live transfer slot for the duration of
9 * a request: _start()/_start_identity() open and own a slot, _add()/_state()/
10 * _error()/_tpc_remote() update it, and _finish() releases it. Implemented by
11 * http_tracking.c for normal builds and by noop.c when the dashboard is disabled.
12 *
13 * This header is HTTP-only (it pulls in ngx_http.h) and must not be included
14 * from stream-module sources; the protocol-agnostic slot operations and SHM
15 * types live in dashboard.h.
16 */

24/*

35/*
36 * Reserve a dashboard transfer slot, bind it to r as module ctx, and return its
37 * index (>=0). All string args (path/identity/vo/op) are borrowed and copied
38 * into the slot; identity NULL is treated as "anonymous", vo NULL as "". Returns
39 * -1 if the dashboard SHM zone is absent/uninitialised or slot/cleanup alloc
40 * fails. Idempotent: re-binding an already-tracked r returns the existing slot.
41 * The slot is auto-freed when r->pool is destroyed even if _finish() is never
42 * called, so callers may ignore the return value. Stamps the slot start time
43 * with ngx_current_msec and snapshots the (NUL-bounded) client address.
44 */
45int brix_dashboard_http_start_identity(ngx_http_request_t *r,
46 const char *path, const char *identity, const char *vo,
47 uint8_t proto, uint8_t direction, const char *op,
48 int64_t expected_bytes);

63/*
