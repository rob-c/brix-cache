1/*
2 * cms/server_recv.c — CMS server-side read/write event loop + frame framing.
3 *
4 * WHAT: The nginx read/write event handlers for an accepted CMS data-server
5 * connection. brix_cms_srv_read accumulates bytes into ctx->inbuf a header
6 * then a dlen-sized payload at a time, hands each complete frame to the opcode
7 * router (cms_srv_process_frame), and enforces the read/idle deadlines and the
8 * per-wakeup fairness cap; brix_cms_srv_write is a write-timeout guard.
9 *
10 * WHY: This is the framing state machine — the "recv" core of the module. It
11 * was split (Phase-79 file-size split) from the connection teardown/audit
12 * (server_recv_lifecycle.c), the pure payload decoders (server_recv_parse.c),
13 * and the opcode handlers + dispatch table (server_recv_frame.c) so each
14 * concern is a focused, independently reviewable file under the size guideline.
15 * The cross-file entry points it calls are declared in server_recv_internal.h.
16 *
17 * HOW: cms_srv_read_accumulate performs one recv() step and grows in_need from
18 * the fixed header to the full frame; cms_srv_dispatch_frame routes a completed
19 * frame and re-arms the idle watchdog; brix_cms_srv_read drives both in a loop
20 * with a bounded per-wakeup yield. A handler that closes the connection leaves
21 * ctx->c NULL, which every step re-checks.
22 */
23

47/* brix_cms_srv_read — read event handler for connected data-server clients
48 * (server-side counterpart to recv.c): accumulate bytes to a complete header, read
49 * the dlen payload, and dispatch each frame (LOGIN→register, LOAD/AVAIL→update load,
50 * PONG→log, GONE→unregister) via cms_srv_process_frame(); disconnect on
51 * timeout/error or a frame over NGX_BRIX_CMS_MAX_FRAME. */
52
53/*
54 * cms_srv_read_timeout — the c->read timer fired. A1: distinguish the
55 * LOGIN-handshake deadline from the post-login idle watchdog (both fire the
56 * read handler via the same timer), count the right resilience metric, and
57 * close the connection.
58 */
59static void
60cms_srv_read_timeout(brix_cms_srv_ctx_t *ctx)
61{
62 if (ctx->logged_in) {
63 BRIX_RESIL_METRIC_INC(cms_idle_closes_total);
64 } else {
65 BRIX_RESIL_METRIC_INC(cms_login_timeouts_total);
66 }
67 cms_srv_fail_close(ctx, BRIX_SESS_END_TIMEOUT);
68}

70/*
71 * cms_srv_read_accumulate — one recv() step of the frame accumulator: pull
72 * bytes into ctx->inbuf, and once the fixed header is complete extend
73 * ctx->in_need to the full frame length (rejecting frames over
74 * NGX_BRIX_CMS_MAX_FRAME). Splitting accumulation from dispatch keeps the
75 * framing state machine in one place. Returns NGX_AGAIN (socket drained),
76 * NGX_DONE (progress made, frame still incomplete — call again), NGX_OK (a
77 * complete frame sits in ctx->inbuf), or NGX_ERROR (peer EOF / recv error /
78 * oversized frame — the connection has been closed).
79 */
80static ngx_int_t
81cms_srv_read_accumulate(brix_cms_srv_ctx_t *ctx, ngx_connection_t *c)
82{
83 ssize_t n;
84 uint16_t dlen;
85
86 n = c->recv(c, ctx->inbuf + ctx->in_pos,
87 ctx->in_need - ctx->in_pos);
88
89 if (n == NGX_AGAIN) {
90 return NGX_AGAIN;
91 }
92
93 if (n == NGX_ERROR || n == 0) {
94 cms_srv_fail_close(ctx, n == 0 ? BRIX_SESS_END_CLIENT
95 : BRIX_SESS_END_ERROR);
96 return NGX_ERROR;
97 }
98
99 ctx->in_pos += (size_t) n;
100
101 if (ctx->in_pos < ctx->in_need) {
102 return NGX_DONE;
103 }
104
105 /* Completed reading the header — extend to full frame if needed. */
106 if (ctx->in_need == NGX_BRIX_CMS_HDR_LEN) {
107 dlen = ngx_brix_cms_get16(ctx->inbuf + 6);
108
109 if ((size_t) dlen + NGX_BRIX_CMS_HDR_LEN
110 > NGX_BRIX_CMS_MAX_FRAME)
111 {
112 ngx_log_error(NGX_LOG_WARN, c->log, 0,
113 "brix: CMS server: frame too large (%ui) "
114 "from %s", (ngx_uint_t) dlen, ctx->host);
115 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
116 return NGX_ERROR;
117 }
118
119 ctx->in_need = NGX_BRIX_CMS_HDR_LEN + dlen;
120 if (ctx->in_pos < ctx->in_need) {
121 return NGX_DONE;
122 }
123 }
124
125 return NGX_OK;
126}

128/*
129 * cms_srv_dispatch_frame — hand the complete frame in ctx->inbuf to the
130 * opcode router, then reset the accumulator for the next header and re-arm
131 * the idle watchdog. WS3: a complete frame proves the node is alive — reset
132 * the post-login idle watchdog. Pre-login the absolute LOGIN deadline armed
133 * at accept is deliberately NOT reset here, so a slowloris that completes one
134 * frame cannot extend its handshake window. Returns NGX_ERROR if the frame
135 * handler closed the connection (ctx->c is NULL), NGX_OK otherwise.
136 */
137static ngx_int_t
138cms_srv_dispatch_frame(brix_cms_srv_ctx_t *ctx)
139{
140 u_char code;
141 uint16_t dlen;
142
143 code = ctx->inbuf[4];
144 dlen = ngx_brix_cms_get16(ctx->inbuf + 6);
145 cms_srv_process_frame(ctx, code, ngx_brix_cms_get32(ctx->inbuf),
146 ctx->inbuf + NGX_BRIX_CMS_HDR_LEN, dlen);
147
148 /* ctx->c may be NULL if the frame handler closed the connection. */
149 if (ctx->c == NULL) {
150 return NGX_ERROR;
151 }
152
153 ctx->in_pos = 0;
154 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
155
156 if (ctx->logged_in && ctx->idle_timeout_ms > 0) {
157 ngx_add_timer(ctx->c->read, ctx->idle_timeout_ms);
158 }
159
160 return NGX_OK;
161}

163/*
164 * cms_srv_read_yield — A2 fairness: after a bounded number of frames, re-arm
165 * the read event and resume via a posted read event, so a flooding data node
166 * cannot monopolise the event loop. Closes the connection if the event
167 * cannot be re-armed; either way the caller returns.
168 */
169static void
170cms_srv_read_yield(brix_cms_srv_ctx_t *ctx, ngx_connection_t *c)
171{
172 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
173 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
174 return;
175 }
176 BRIX_RESIL_METRIC_INC(cms_frame_yields_total);
177 ngx_post_event(c->read, &ngx_posted_events);
178}

180void
181brix_cms_srv_read(ngx_event_t *ev)
182{
183 ngx_connection_t *c;
184 brix_cms_srv_ctx_t *ctx;
185 ngx_int_t rc;
186 ngx_uint_t processed = 0;
187
188 c = ev->data;
189 ctx = c->data;
190
191 if (ev->timedout) {
192 cms_srv_read_timeout(ctx);
193 return;
194 }
195
196 for ( ;; ) {
197 rc = cms_srv_read_accumulate(ctx, c);
198
199 if (rc == NGX_AGAIN) {
200 break;
201 }
202 if (rc == NGX_ERROR) {
203 return;
204 }
205 if (rc == NGX_DONE) {
206 continue;
207 }
208
209 /* Full frame received — dispatch. */
210 if (cms_srv_dispatch_frame(ctx) != NGX_OK) {
211 return;
212 }
213
214 if (++processed >= NGX_BRIX_CMS_MAX_FRAMES_PER_WAKEUP) {
215 cms_srv_read_yield(ctx, c);
216 return;
217 }
218 }
219
220 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
221 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
222 }
223}
