9/* brix_cms_srv_close — tear down a CMS data-server connection: drop the ping
10 * timer, unregister host/port from the server registry if logged_in (so locate
11 * queries stop routing clients to a dead server), NULL ctx->c, and close. */
12
13void
14brix_cms_srv_close(brix_cms_srv_ctx_t *ctx)
15{
16 ngx_connection_t *c;
17
18 c = ctx->c;
19 if (c == NULL) {
20 return;
21 }
22
23 if (ctx->ping_timer.timer_set) {
24 ngx_del_timer(&ctx->ping_timer);
25 }
26
27 /* WS3: cancel the login/idle read deadline before tearing the socket down. */
28 if (c->read->timer_set) {
29 ngx_del_timer(c->read);
30 }
31
32 /* WS4 + A3: release the admission-cap slots (global + per-IP) exactly once. */
33 if (ctx->counted) {
34 brix_cms_srv_conn_dec();
35 brix_cms_srv_ip_dec(ctx->host);
36 ctx->counted = 0;
37 }
38
39 if (ctx->logged_in) {
40 /* Blacklist for 30 s so in-flight locate responses don't route to a
41 * server that just went away. brix_srv_register() clears the flag
42 * the moment the server successfully reconnects and re-heartbeats. */
43 brix_srv_blacklist(ctx->host, ctx->port, 30000);
44 BRIX_DIAG(NGX_LOG_NOTICE, c->log, 0,
45 "xrootd[cms]: data server %s:%d disconnected (blacklisted 30s)",
46 "the data server dropped its CMS connection — it crashed, was "
47 "restarted, or lost network to the manager",
48 "if it does not re-register within seconds, check that server's "
49 "health and connectivity; clients are routed away from it "
50 "meanwhile",
51 ctx->host, (int) ctx->port);
52 }
53
54 ctx->c = NULL;
55 ngx_close_connection(c);
56}

58/* TLV walk helpers */

60/*
61 * The LOGIN/LOAD/AVAIL payloads use a simple type-tagged encoding:
62 * CMS_PT_SHORT (0x80) + big-endian uint16
63 * CMS_PT_INT (0xa0) + big-endian uint32
64 *
65 * Walk *p forward by one tagged value, returning the decoded uint32.
66 * Advances *p past the tag+value bytes. Returns 0 and sets *p=end on error.
67 */
68static uint32_t
69tlv_read_next(const u_char **p, const u_char *end)
70{
71 if (*p >= end) {
72 *p = end;
73 return 0;
74 }
75
76 if (**p == CMS_PT_SHORT) {
77 if (end - *p < 3) { *p = end; return 0; }
78 uint32_t v = ngx_brix_cms_get16(*p + 1);
79 *p += 3;
80 return v;
81 }
82
83 if (**p == CMS_PT_INT) {
84 if (end - *p < 5) { *p = end; return 0; }
85 uint32_t v = ngx_brix_cms_get32(*p + 1);
86 *p += 5;
87 return v;
88 }
89
90 /* Unknown tag — treat as end of parseable data. */
91 *p = end;
92 return 0;
93}

95/*
96 * Read one XrdOucPup string from *p: a 2-byte big-endian length followed by
97 * <len> raw bytes (the length includes the trailing NUL). Sets *out / *out_len
98 * to the data span and advances *p. Returns 1 on success, 0 (and *p=end) on a
99 * short/overrun buffer. Strings carry NO type tag — the high bit of the first
100 * length byte is clear for any reasonable length, which is how the real Parser
101 * tells a string apart from a PT_short/PT_int scalar.
102 */
103static int
104cms_srv_read_string(const u_char **p, const u_char *end,
105 const u_char **out, size_t *out_len)
106{
107 uint16_t len;
108
109 if (end - *p < 2) { *p = end; return 0; }
110 len = ngx_brix_cms_get16(*p);
111 *p += 2;
112 if ((size_t) (end - *p) < len) { *p = end; return 0; }
113 *out = *p;
114 *out_len = len;
115 *p += len;
116 return 1;
117}

119/* LOGIN payload parser */
120
121/*
122 * Parse the CMS LOGIN frame payload in the real XrdCms CmsLoginData wire format
123 * (XrdOucPup), the same one a real cmsd and our own cms/send.c now emit:
124 *
125 * PT_SHORT Version PT_INT Mode PT_INT HoldTime
126 * PT_INT tSpace PT_INT fSpace ←free_mb PT_INT mSpace
127 * PT_SHORT fsNum PT_SHORT fsUtil ←util_pct PT_SHORT dPort ←port
128 * PT_SHORT sPort [Fence]
129 * string SID string Paths string ifList string envCGI
130 *
131 * The Paths string is a newline-separated list of "<type> <namespace-path>"
132 * entries (type 'r'/'w', e.g. "w /\nw /atlas"). We strip the type prefix and
133 * store the bare namespace paths colon-delimited ("/:/atlas") — the form the
134 * registry's srv_path_matches() expects.
135 */
136static int
137cms_srv_parse_login(brix_cms_srv_ctx_t *ctx,
138 const u_char *payload, size_t payload_len)
139{
140 const u_char *p = payload;
141 const u_char *end = payload + payload_len;
142 const u_char *sid;
143 const u_char *paths;
144 size_t sid_len;
145 size_t paths_len;
146 u_char *dst;
147 u_char *dst_end;
148 size_t i;
149
150 /* version */ (void) tlv_read_next(&p, end);
151 /* mode */ (void) tlv_read_next(&p, end);
152 /* holdtime */ (void) tlv_read_next(&p, end);
153 /* tSpace */ (void) tlv_read_next(&p, end);
154 ctx->free_mb = tlv_read_next(&p, end); /* fSpace */
155 /* mSpace */ (void) tlv_read_next(&p, end);
156 /* fsNum */ (void) tlv_read_next(&p, end);
157 ctx->util_pct = tlv_read_next(&p, end); /* fsUtil */
158 ctx->port = (uint16_t) tlv_read_next(&p, end); /* dPort */
159 /* sPort */ (void) tlv_read_next(&p, end);
160
161 /* SID (ignored) then Paths (extracted). ifList/envCGI ignored. */
162 (void) cms_srv_read_string(&p, end, &sid, &sid_len);
163 if (!cms_srv_read_string(&p, end, &paths, &paths_len)) {
164 paths = NULL;
165 paths_len = 0;
166 }
167
168 /*
169 * Convert "<type> <path>\n<type> <path>..." -> colon-delimited bare paths.
170 * Each newline segment is "<type> <path>"; the path follows the first
171 * space. A segment without a space is taken verbatim as the path.
172 */
173 dst = (u_char *) ctx->paths;
174 dst_end = dst + sizeof(ctx->paths) - 1;
175 i = 0;
176 while (i < paths_len && dst < dst_end) {
177 const u_char *tok;
178 size_t tok_len;
179 size_t seg_end;
180
181 while (i < paths_len
182 && (paths[i] == '\n' || paths[i] == ' '
183 || paths[i] == '\t' || paths[i] == '\0'))
184 {
185 i++;
186 }
187 if (i >= paths_len) {
188 break;
189 }
190
191 /* find end of this segment (newline / NUL terminated) */
192 seg_end = i;
193 while (seg_end < paths_len
194 && paths[seg_end] != '\n' && paths[seg_end] != '\0')
195 {
196 seg_end++;
197 }
198
199 /* split off the leading "<type> " prefix, if any */
200 tok = paths + i;
201 tok_len = seg_end - i;
202 {
203 size_t sp = 0;
204 while (sp < tok_len && tok[sp] != ' ') {
205 sp++;
206 }
207 if (sp < tok_len) { /* found a space → path follows it */
208 sp++;
209 while (sp < tok_len && tok[sp] == ' ') {
210 sp++;
211 }
212 tok += sp;
213 tok_len -= sp;
214 }
215 }
216 while (tok_len > 0
217 && (tok[tok_len - 1] == ' ' || tok[tok_len - 1] == '\t'))
218 {
219 tok_len--;
220 }
221
222 if (tok_len > 0) {
223 size_t cp;
224 if (dst != (u_char *) ctx->paths && dst < dst_end) {
225 *dst++ = ':';
226 }
227 cp = (size_t) (dst_end - dst);
228 if (cp > tok_len) {
229 cp = tok_len;
230 }
231 ngx_memcpy(dst, tok, cp);
232 dst += cp;
233 }
234
235 i = seg_end;
236 }
237 *dst = '\0';
238
239 /* Default XRootD port if the data server didn't advertise one. */
240 if (ctx->port == 0) {
241 ctx->port = BRIX_DEFAULT_PORT;
242 }
243
244 return 1;
245}

249/*
250 * LOAD payload (from cms/send.c):
251 * PT_SHORT count=6 (3 bytes)
252 * <6 raw bytes> (cpu load values, ignored)
253 * PT_INT free_mb (5 bytes) ← extracted
254 */
255static uint32_t
256cms_srv_parse_load_free_mb(const u_char *payload, size_t payload_len)
257{
258 const u_char *p = payload;
259 const u_char *end = payload + payload_len;
260
261 /* count field */
262 (void) tlv_read_next(&p, end);
263
264 /* skip 6 raw CPU bytes */
265 if (end - p >= 6) {
266 p += 6;
267 } else {
268 return 0;
269 }
270
271 return tlv_read_next(&p, end);
272}

312/*
313 * cms_srv_complete_login — register the data server and arm heartbeats.
314 *
315 * Called once the LOGIN frame has been parsed AND (when sss is configured) the
316 * kYR_xauth credential has been verified. Splitting this out lets the sss path
317 * defer registration until authentication succeeds without duplicating the
318 * register/log/timer sequence.
319 */
320static void
321cms_srv_complete_login(brix_cms_srv_ctx_t *ctx)
322{
323 brix_srv_register(ctx->host, ctx->port, ctx->paths,
324 ctx->free_mb, ctx->util_pct);
325 ctx->logged_in = 1;
326
327 /* Arm the periodic ping now that the data server is logged in. */
328 ctx->ping_timer.handler = brix_cms_srv_ping_timer;
329 ngx_add_timer(&ctx->ping_timer, ctx->interval_ms);
330
331 /*
332 * WS3: switch the read deadline from the LOGIN handshake bound to the
333 * post-login idle watchdog. Re-armed on every received frame in
334 * brix_cms_srv_read; if the node goes silent for idle_timeout_ms it is
335 * closed + unregistered + blacklisted (reaps a black-holed node the ping
336 * send cannot detect). 0 = disabled.
337 */
338 if (ctx->idle_timeout_ms > 0 && ctx->c != NULL) {
339 ngx_add_timer(ctx->c->read, ctx->idle_timeout_ms);
340 } else if (ctx->c != NULL && ctx->c->read->timer_set) {
341 ngx_del_timer(ctx->c->read); /* idle watchdog disabled — drop login timer */
342 }
343
344 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
345 "brix: CMS server: registered %s:%d paths=[%s] "
346 "free_mb=%uD util_pct=%uD",
347 ctx->host, (int) ctx->port, ctx->paths,
348 ctx->free_mb, ctx->util_pct);
349}

351static void
352cms_srv_process_frame(brix_cms_srv_ctx_t *ctx, u_char code, uint32_t streamid,
353 const u_char *payload, size_t payload_len)
354{
355 uint32_t free_mb, util_pct;
356
357 switch (code) {
358
359 case CMS_RR_LOGIN:
360 if (!cms_srv_parse_login(ctx, payload, payload_len)) {
361 ngx_log_error(NGX_LOG_WARN, ctx->c->log, 0,
362 "brix: CMS server: malformed LOGIN from %s",
363 ctx->host);
364 brix_cms_srv_close(ctx);
365 return;
366 }
367
368 /*
369 * W1a — if sss cluster auth is required, do NOT register yet. Mirror
370 * the real cmsd manager (XrdCmsLogin::Admit): after the LOGIN frame
371 * arrives, send our security parms and wait for the node's kYR_xauth
372 * credential before admitting it into the registry.
373 */
374 if (ctx->auth_state == CMS_AUTH_REQUESTED) {
375 static const u_char sss_parms[] = "&P=sss";
376 if (brix_cms_srv_send_xauth(ctx, sss_parms,
377 sizeof(sss_parms)) != NGX_OK)
378 {
379 brix_cms_srv_close(ctx);
380 return;
381 }
382 ctx->auth_state = CMS_AUTH_CHALLENGED;
383 break;
384 }
385
386 cms_srv_complete_login(ctx);
387 break;
388
389 case CMS_RR_XAUTH:
390 /* Only meaningful while we are waiting for a credential. */
391 if (ctx->auth_state != CMS_AUTH_CHALLENGED) {
392 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
393 "brix: CMS server: unexpected kYR_xauth from %s",
394 ctx->host);
395 brix_cms_srv_close(ctx);
396 return;
397 }
398 if (brix_cms_srv_verify_xauth(ctx, payload, payload_len) != NGX_OK) {
399 brix_cms_srv_close(ctx);
400 return;
401 }
402 ctx->auth_state = CMS_AUTH_DONE;
403 cms_srv_complete_login(ctx);
404 break;
405
406 case CMS_RR_LOAD:
407 if (!ctx->logged_in) { break; }
408 free_mb = cms_srv_parse_load_free_mb(payload, payload_len);
409 brix_srv_update_load(ctx->host, ctx->port, free_mb, ctx->util_pct);
410 ctx->free_mb = free_mb;
411 break;
412
413 case CMS_RR_AVAIL:
414 case CMS_RR_SPACE:
415 if (!ctx->logged_in) { break; }
416 cms_srv_parse_avail(payload, payload_len, &free_mb, &util_pct);
417 brix_srv_update_load(ctx->host, ctx->port, free_mb, util_pct);
418 ctx->free_mb = free_mb;
419 ctx->util_pct = util_pct;
420 break;
421
422 case CMS_RR_PONG:
423 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
424 "brix: CMS server: pong from %s", ctx->host);
425 break;
426
427 case CMS_RR_PING:
428 /*
429 * Symmetric liveness (do_Ping): a node (or peer) probing us gets a
430 * header-only kYR_pong. No auth/state needed — pure liveness.
431 */
432 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
433 "brix: CMS server: ping from %s -> pong", ctx->host);
434 if (brix_cms_srv_send_pong(ctx) != NGX_OK) {
435 brix_cms_srv_close(ctx);
436 return;
437 }
438 break;
439
440 case CMS_RR_DISC:
441 /*
442 * Graceful disconnect (do_Disc as manager): echo a kYR_disc back, then
443 * close — which unregisters the node from the registry.
444 */
445 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
446 "brix: CMS server: %s requested disconnect", ctx->host);
447 (void) brix_cms_srv_send_disc(ctx);
448 brix_cms_srv_close(ctx);
449 return;
450
451 case CMS_RR_UPDATE:
452 /*
453 * State-refresh request (do_Update): resend our state as a kYR_status
454 * frame so the peer keeps us active and eligible.
455 */
456 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
457 "brix: CMS server: update from %s -> status", ctx->host);
458 if (brix_cms_srv_send_status(ctx, CMS_ST_RESUME | CMS_ST_NOSTAGE)
459 != NGX_OK)
460 {
461 brix_cms_srv_close(ctx);
462 return;
463 }
464 break;
465
466 case CMS_RR_STATFS: {
467 /*
468 * Space query for a path (do_StatFS): answer kYR_data with aggregate
469 * cluster space for the servers that hold the path. We report a single
470 * tier (staging == writable) since we model one storage class.
471 */
472 brix_cms_rrdata_t d;
473 u_char out[64];
474 uint32_t free_mb = 0, util = 0, num;
475 int olen;
476
477 if (!ctx->logged_in
478 || brix_cms_rrdata_parse(CMS_RR_STATFS, payload, payload_len, &d)
479 != 0
480 || d.path == NULL)
481 {
482 break; /* malformed / pre-auth: ignore */
483 }
484
485 num = (uint32_t) brix_srv_count_matching((const char *) d.path);
486 brix_srv_aggregate_space(&free_mb, &util);
487 olen = brix_cms_statfs_encode(num, free_mb, util, num, free_mb, util,
488 out, sizeof(out));
489 if (olen > 0) {
490 (void) brix_cms_srv_send_data(ctx, streamid, out, (size_t) olen);
491 }
492 break;
493 }
494
495 case CMS_RR_GONE:
496 if (!ctx->logged_in) { break; }
497 if (payload_len > 0) {
498 /*
499 * A data server signals that it no longer holds a specific path.
500 * Remove that path token from its registry entry without touching
501 * the rest of the registration (the server still holds other paths).
502 */
503 char path[BRIX_SRV_MAX_PATHS];
504 size_t copy = payload_len < sizeof(path) - 1
505 ? payload_len : sizeof(path) - 1;
506 ngx_memcpy(path, payload, copy);
507 path[copy] = '\0';
508 brix_srv_unregister_path(ctx->host, ctx->port, path);
509 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
510 "brix: CMS server: kYR_gone path=%s from %s:%d",
511 path, ctx->host, (int) ctx->port);
512 }
513 break;
514
515 case CMS_RR_CNS:
516 /* §6 Composite Cluster Name Space: a data server reports a namespace
517 * mutation. Apply it into the manager inventory (collect mode only). */
518 if (!ctx->logged_in || !brix_cns_collecting()) {
519 break;
520 }
521 {
522 uint8_t op;
523 uint64_t size, mtime;
524 char path[BRIX_CNS_PATH_MAX + 1];
525 uint32_t server_id;
526 const u_char *h = (const u_char *) ctx->host;
527 size_t i;
528
529 if (brix_cns_event_decode(payload, payload_len, &op, &size, &mtime,
530 path, sizeof(path)) != NGX_OK)
531 {
532 ngx_log_error(NGX_LOG_WARN, ctx->c->log, 0,
533 "brix: CMS server: malformed CNS event from %s",
534 ctx->host);
535 break;
536 }
537 server_id = (uint32_t) ctx->port; /* host hash + port = origin id */
538 for (i = 0; h[i] != '\0'; i++) {
539 server_id = server_id * 33 + h[i];
540 }
541 (void) brix_cns_apply(op, path, size, mtime, server_id);
542 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
543 "brix: CNS event op=%ud path=%s from %s",
544 (unsigned) op, path, ctx->host);
545 }
546 break;
547
548 default: {
549 /*
550 * Consult the manager routing table so the log distinguishes a valid
551 * manager opcode we have not wired yet (e.g. usage/stats/statfs) from a
552 * truly unroutable code. Either way we drop it — matching cmsd's
553 * tolerance of frames it does not act on.
554 */
555 const brix_cms_route_t *r =
556 brix_cms_route_lookup(XRDCMS_ROLE_MANAGER, code);
557 if (r != NULL) {
558 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
559 "brix: CMS server: unhandled opcode '%s' from %s",
560 r->name, ctx->host);
561 } else {
562 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
563 "brix: CMS server: unknown rrCode=%ui from %s",
564 (ngx_uint_t) code, ctx->host);
565 }
566 break;
567 }
568 }
569}

589/* brix_cms_srv_read — read event handler for connected data-server clients
590 * (server-side counterpart to recv.c): accumulate bytes to a complete header, read
591 * the dlen payload, and dispatch each frame (LOGIN→register, LOAD/AVAIL→update load,
592 * PONG→log, GONE→unregister) via cms_srv_process_frame(); disconnect on
593 * timeout/error or a frame over NGX_BRIX_CMS_MAX_FRAME. */
594
595void
596brix_cms_srv_read(ngx_event_t *ev)
597{
598 ngx_connection_t *c;
599 brix_cms_srv_ctx_t *ctx;
600 ssize_t n;
601 uint16_t dlen;
602 u_char code;
603 ngx_uint_t processed = 0;
604
605 c = ev->data;
606 ctx = c->data;
607
608 if (ev->timedout) {
609 /* A1: distinguish the LOGIN-handshake deadline from the post-login idle
610 * watchdog (both fire this handler via the c->read timer). */
611 if (ctx->logged_in) {
612 BRIX_RESIL_METRIC_INC(cms_idle_closes_total);
613 } else {
614 BRIX_RESIL_METRIC_INC(cms_login_timeouts_total);
615 }
616 brix_cms_srv_close(ctx);
617 return;
618 }
619
620 for ( ;; ) {
621 n = c->recv(c, ctx->inbuf + ctx->in_pos,
622 ctx->in_need - ctx->in_pos);
623
624 if (n == NGX_AGAIN) {
625 break;
626 }
627
628 if (n == NGX_ERROR || n == 0) {
629 brix_cms_srv_close(ctx);
630 return;
631 }
632
633 ctx->in_pos += (size_t) n;
634
635 if (ctx->in_pos < ctx->in_need) {
636 continue;
637 }
638
639 /* Completed reading the header — extend to full frame if needed. */
640 if (ctx->in_need == NGX_BRIX_CMS_HDR_LEN) {
641 dlen = ngx_brix_cms_get16(ctx->inbuf + 6);
642
643 if ((size_t) dlen + NGX_BRIX_CMS_HDR_LEN
644 > NGX_BRIX_CMS_MAX_FRAME)
645 {
646 ngx_log_error(NGX_LOG_WARN, c->log, 0,
647 "brix: CMS server: frame too large (%ui) "
648 "from %s", (ngx_uint_t) dlen, ctx->host);
649 brix_cms_srv_close(ctx);
650 return;
651 }
652
653 ctx->in_need = NGX_BRIX_CMS_HDR_LEN + dlen;
654 if (ctx->in_pos < ctx->in_need) {
655 continue;
656 }
657 }
658
659 /* Full frame received — dispatch. */
660 code = ctx->inbuf[4];
661 dlen = ngx_brix_cms_get16(ctx->inbuf + 6);
662 cms_srv_process_frame(ctx, code, ngx_brix_cms_get32(ctx->inbuf),
663 ctx->inbuf + NGX_BRIX_CMS_HDR_LEN, dlen);
664
665 /* ctx->c may be NULL if the frame handler closed the connection. */
666 if (ctx->c == NULL) {
667 return;
668 }
669
670 ctx->in_pos = 0;
671 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
672
673 /*
674 * WS3: a complete frame proves the node is alive — reset the post-login
675 * idle watchdog. Pre-login the absolute LOGIN deadline armed at accept is
676 * deliberately NOT reset here, so a slowloris that completes one frame
677 * cannot extend its handshake window.
678 */
679 if (ctx->logged_in && ctx->idle_timeout_ms > 0) {
680 ngx_add_timer(ctx->c->read, ctx->idle_timeout_ms);
681 }
682
683 /* A2: fairness — yield after a bounded number of frames and resume via a
684 * posted read event, so a flooding data node cannot monopolise the loop. */
685 if (++processed >= NGX_BRIX_CMS_MAX_FRAMES_PER_WAKEUP) {
686 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
687 brix_cms_srv_close(ctx);
688 return;
689 }
690 BRIX_RESIL_METRIC_INC(cms_frame_yields_total);
691 ngx_post_event(c->read, &ngx_posted_events);
692 return;
693 }
694 }
695
696 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
697 brix_cms_srv_close(ctx);
698 }
