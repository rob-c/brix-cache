8/*
9 * s3_chunk_finalize — post-decode steps that must run on the event loop:
10 * commit the staged object, set the ETag, verify the (trailer or default)
11 * checksum, store tags, and send 200. Shared by the synchronous fallback and
12 * the offload completion; `staged` has been written but not yet committed.
13 */
14void
15s3_chunk_finalize(ngx_http_request_t *r, const char *root_canon,
16 const char *fs_path, brix_vfs_staged_t *staged,
17 const s3_chunk_trailer_t *trailer, uint64_t expected, ngx_uint_t body_mode)
18{
19 if (s3_commit_put(r, r->connection->log, root_canon, staged,
20 fs_path) != NGX_OK)
21 {
22 if (s3_put_commit_conflict(r)) {
23 return;
24 }
25 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, ngx_errno,
26 "s3: staged commit to \"%s\" failed", fs_path);
27 s3_put_finalize_error(r);
28 return;
29 }
30
31 {
32 brix_vfs_ctx_t fctx;
33 brix_vfs_stat_t fst;
34 char etag_buf[48];
35
36 brix_vfs_ctx_init(&fctx, r->pool, r->connection->log, BRIX_PROTO_S3,
37 root_canon, NULL, 0 /* allow_write */, 0 /* is_tls */, NULL, fs_path);
38 if (brix_vfs_probe(&fctx, 1 /* no-follow */, &fst) == NGX_OK) {
39 struct stat sb;
40
41 ngx_memzero(&sb, sizeof(sb));
42 sb.st_mtime = fst.mtime;
43 sb.st_size = fst.size;
44 s3_etag(&sb, etag_buf, sizeof(etag_buf));
45 (void) s3_set_header(r, "ETag", etag_buf);
46 }
47 }
48
49 if (trailer->algo_token[0] != '\0') {
50 switch (s3_put_trailer_checksum_apply(r, fs_path, root_canon,
51 trailer->algo_token,
52 trailer->value))
53 {
54 case S3_CKSUM_MISMATCH:
55 s3_put_finalize_bad_digest(r);
56 return;
57 case S3_CKSUM_CONFLICT:
58 s3_put_finalize_invalid_request(r);
59 return;
60 default:
61 break;
62 }
63 } else if (s3_put_checksum_failed(r, fs_path, root_canon)) {
64 return;
65 }
66
67 (void) s3_apply_put_tagging_header(r, fs_path, root_canon);
68 (void) s3_apply_put_user_metadata(r, fs_path, root_canon);
69 s3_put_finalize_ok(r, expected, body_mode);
70}

73/* Decode-failure path: abort the staged temp and send the mapped S3 error. */
74void
75s3_chunk_decode_failed(ngx_http_request_t *r, const char *root_canon,
76 brix_vfs_staged_t *staged, int http_status)
77{
78 brix_vfs_staged_abort(staged, 1);
79 if (http_status >= 500) {
80 s3_put_finalize_error(r);
81 } else if (http_status == NGX_HTTP_FORBIDDEN) {
82 /* W6a: a per-chunk signature mismatch — surface as SignatureDoesNotMatch. */
83 s3_put_finalize_client_error(r, NGX_HTTP_FORBIDDEN,
84 "SignatureDoesNotMatch",
85 "The chunk signature does not match the calculated signature.");
86 } else {
87 s3_put_finalize_client_error(r, NGX_HTTP_BAD_REQUEST, "InvalidRequest",
88 "The aws-chunked request body could not be decoded.");
89 }
90}

93/*
94 * phase-46 W1b: aws-chunked decode offload. The decode is a blocking
95 * pread/pwrite state machine over the (often spooled) body, so it runs on the
96 * thread pool; the completion commits/checksums/finalizes on the event loop.
97 * The worker uses the task's own `window` scratch — it never touches r->pool.
98 */
99
100void
101s3_chunk_aio_thread(void *data, ngx_log_t *log)
102{

108 t->expected, &t->trailer,
109 &t->http_status, t->window,
110 &t->verify);
111}

114/*
115 * s3_build_chunk_verify — W6a: assemble the per-chunk verification context from
116 * the location flag + the SigV4 material auth retained on the request ctx. Left
117 * disabled (zeroed) unless verification is configured AND a signed request was
118 * authenticated (an anonymous endpoint has no secret to verify against).
119 */
120void
121s3_build_chunk_verify(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
122 s3_chunk_verify_t *v)
123{
124 ngx_http_s3_req_ctx_t *rx;
125
126 ngx_memzero(v, sizeof(*v));
127 if (!cf->verify_chunk_signatures) {
128 return;
129 }
130 rx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
131 if (rx == NULL || !rx->have_sigv4) {
132 return;
133 }
134 v->enabled = 1;
135 ngx_memcpy(v->signing_key, rx->sigv4_signing_key, 32);
136 ngx_cpystrn((u_char *) v->seed_signature,
137 (u_char *) rx->sigv4_seed_signature, sizeof(v->seed_signature));
138 ngx_cpystrn((u_char *) v->amz_date,
139 (u_char *) rx->sigv4_amz_date, sizeof(v->amz_date));
140 ngx_cpystrn((u_char *) v->scope,
141 (u_char *) rx->sigv4_scope, sizeof(v->scope));
142}
