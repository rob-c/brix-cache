1/* client/lib/cred.h
2 *
3 * cred.h — common credential acquisition for the client tools.
4 *
5 * WHAT: one store that discovers, loads, caches, and refreshes each credential
6 * kind (X.509 proxy, bearer token, Kerberos TGT, SSS keytab, S3 keys) and
7 * serves BOTH the root:// auth handshake (the lib/sec modules) and the HTTP/S3
8 * transports (webfile.c, s3.c) from a single source of truth.
9 * WHY: discovery/refresh logic is duplicated across sec_gsi.c/sec_token.c/
10 * sec_sss.c/sec_krb5.c/credrefresh.c, and the HTTP path acquires bearers
11 * separately from the root:// token module. One store removes the drift
12 * and lets --auto-refresh cover every kind uniformly.
13 * HOW: per-kind handlers (cred_x509.c etc.) implement available/acquire/refresh;
14 * the store caches the result with an expiry and re-acquires when a caller
15 * asks for a credential within min_remaining_s of expiry. ngx-free.
16 */

32/* A read-only snapshot a consumer uses. Strings are owned by the store and stay
33 * valid until the next acquire of the SAME kind on the SAME store. */
34typedef struct {

39 const char *s3_secret; /* S3 secret key, or NULL */
40 int64_t not_after; /* unix expiry; 0 = unknown/none */
41} brix_cred_view;

72/* Per-kind handler contract (implemented by cred_x509.c, cred_bearer.c, …). */
73typedef struct {
74 brix_cred_kind kind;
75 int (*available)(const brix_cred_config *cfg);
76 /* acquire() fills *out and *not_after on success (returns 0) or sets *st
77 * and returns -1. LIFETIME CONTRACT (applies to ALL handlers — B3-B6):
78 * acquire() must write out->{path,token,s3_access,s3_secret} as pointers
79 * to storage that remains valid until acquire() RETURNS — the store copies
80 * (strdup) them immediately afterward. Do NOT point them at acquire()'s
81 * own stack locals; use static or otherwise non-stack storage. */
82 int (*acquire)(const brix_cred_config *cfg, brix_cred_view *out,
83 int64_t *not_after, brix_status *st);
84 int (*refresh)(const brix_cred_config *cfg, brix_status *st); /* NULL = no refresh */
85} brix_cred_handler;
