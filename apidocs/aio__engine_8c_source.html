8/* ============ Phase 44: pluggable loop I/O engine (epoll | io_uring) ============
9 *
10 * Two readiness engines share one interface. The epoll branch is the historical
11 * code, preserved verbatim. The io_uring branch (multishot IORING_OP_POLL_ADD,
12 * default OFF — gated by XRDC_IO_URING_LOOP=on + a runtime probe, best-effort
13 * fallback to epoll) is a drop-in readiness source: the loop still runs
14 * aconn_do_read/aconn_do_write unchanged, so TLS (which drives the fd through
15 * OpenSSL itself) is safe. Cross-thread wake is write(evfd) for BOTH engines —
16 * the io_uring engine arms a multishot poll on the same evfd.
17 *
18 * UAF safety: a poll's user_data carries (slot-generation<<32 | slot); the slot
19 * table maps back to the aconn and the reaper drops any CQE whose generation no
20 * longer matches (poll completing after its aconn was reconnected/freed) — the
21 * same discipline as the server ring. fd changes (reconnect) cancel the old
22 * poll and bump the slot generation before re-arming the new fd. */

40/* Claim a free poll slot for ac (loop-thread only; no lock). Returns 0 / -1. */
41int
42uring_slot_alloc(brix_loop *l, brix_aconn *ac)
43{
44 unsigned i;
45 for (i = 0; i < AIO_URING_SLOTS; i++) {
46 if (!l->uslots[i].in_use) {
47 l->uslots[i].in_use = 1;
48 l->uslots[i].ac = ac;
49 ac->uring_slot = (int) i;
50 return 0;
51 }
52 }
53 return -1; /* table full — pool has more conns than slots (shouldn't happen) */
54}

57/* Submit a multishot poll for ac with the current slot generation as user_data. */
58int
59uring_poll_submit(brix_loop *l, brix_aconn *ac, int want)
60{
61 struct io_uring_sqe *sqe = io_uring_get_sqe(&l->uring);
62 uint32_t slot = (uint32_t) ac->uring_slot;
63
64 if (sqe == NULL) {
65 return -1;
66 }
67 io_uring_prep_poll_multishot(sqe, ac->fd, uring_pollmask(want));
68 io_uring_sqe_set_data64(sqe, ((uint64_t) l->uslots[slot].gen << 32) | slot);
69 return io_uring_submit(&l->uring) < 0 ? -1 : 0;
70}

73/* Cancel ac's outstanding poll (by current user_data) and bump the slot gen so
74 * any late CQE for it is dropped. Keeps the slot allocated (caller re-arms) or
75 * frees it when freeing == 1. */
76void
77uring_poll_cancel(brix_loop *l, brix_aconn *ac, int freeing)
78{
79 uint32_t slot = (uint32_t) ac->uring_slot;
80 struct io_uring_sqe *sqe;
81
82 if (ac->uring_slot < 0) {
83 return;
84 }
85 sqe = io_uring_get_sqe(&l->uring);
86 if (sqe != NULL) {
87 io_uring_prep_poll_remove(sqe,
88 ((uint64_t) l->uslots[slot].gen << 32) | slot);
89 io_uring_sqe_set_data64(sqe, AIO_URING_IGNORE_UD);
90 (void) io_uring_submit(&l->uring);
91 }
92 l->uslots[slot].gen++; /* invalidate the old poll's user_data */
93 if (freeing) {
94 l->uslots[slot].in_use = 0;
95 l->uslots[slot].ac = NULL;
96 ac->uring_slot = -1;
97 }
98}

104/* Create the readiness set + the wake eventfd. evfd is used by both engines;
105 * epoll registers it in the set, io_uring arms a multishot poll on it. */
106int
107io_engine_setup(brix_loop *l, brix_status *st)
108{
109 l->evfd = eventfd(0, EFD_NONBLOCK | EFD_CLOEXEC);
110 if (l->evfd < 0) {
111 brix_status_set(st, XRDC_ESOCK, errno, "eventfd: %s", strerror(errno));
112 return -1;
113 }
114
115#if (BRIX_HAVE_LIBURING)
116 if (l->use_uring) {
117 struct io_uring_sqe *sqe;
118 if (io_uring_queue_init(256, &l->uring, 0) < 0) {
119 brix_status_set(st, XRDC_ESOCK, errno, "io_uring_queue_init");
120 return -1;
121 }
122 l->uring_ok = 1;
123 sqe = io_uring_get_sqe(&l->uring); /* multishot poll on the evfd */
124 if (sqe == NULL) {
125 brix_status_set(st, XRDC_ESOCK, 0, "io_uring evfd arm");
126 return -1;
127 }
128 io_uring_prep_poll_multishot(sqe, l->evfd, POLLIN);
129 io_uring_sqe_set_data64(sqe, AIO_URING_EVFD_UD);
130 if (io_uring_submit(&l->uring) < 0) {
131 brix_status_set(st, XRDC_ESOCK, errno, "io_uring evfd submit");
132 return -1;
133 }
134 return 0;
135 }
136#endif
137
138 l->epfd = epoll_create1(EPOLL_CLOEXEC);
139 if (l->epfd < 0) {
140 brix_status_set(st, XRDC_ESOCK, errno, "epoll_create1: %s",
141 strerror(errno));
142 return -1;
143 }
144 {
145 struct epoll_event ev;
146 memset(&ev, 0, sizeof(ev));
147 ev.events = EPOLLIN;
148 ev.data.ptr = l; /* loop pointer tags the eventfd */
149 if (epoll_ctl(l->epfd, EPOLL_CTL_ADD, l->evfd, &ev) != 0) {
150 brix_status_set(st, XRDC_ESOCK, errno, "epoll_ctl(evfd): %s",
151 strerror(errno));
152 return -1;
153 }
154 }
155 return 0;
156}

177/* Arm/modify interest for ac to `want`; sets ac->epoll_events on success.
178 * Returns 0 / -1 (caller keeps its own failure handling). */
179int
180io_engine_arm(brix_loop *l, brix_aconn *ac, int want)
181{
182#if (BRIX_HAVE_LIBURING)
183 if (l->use_uring) {
184 if (ac->uring_slot < 0) {
185 if (uring_slot_alloc(l, ac) != 0) { return -1; }
186 } else {
187 uring_poll_cancel(l, ac, 0); /* drop the old mask's poll, keep slot */
188 }
189 ac->fd_gen++;
190 if (uring_poll_submit(l, ac, want) != 0) { return -1; }
191 ac->epoll_events = want;
192 return 0;
193 }
194#endif
195 {
196 struct epoll_event ev;
197 int op = (ac->epoll_events == 0) ? EPOLL_CTL_ADD : EPOLL_CTL_MOD;
198 memset(&ev, 0, sizeof(ev));
199 ev.events = (uint32_t) want;
200 ev.data.ptr = ac;
201 if (epoll_ctl(l->epfd, op, ac->fd, &ev) != 0) {
202 return -1;
203 }
204 ac->epoll_events = want;
205 return 0;
206 }
207}

229/* ---- Translate one slot-poll CQE into a readiness event ----
230 *
231 * WHAT: Given a poll CQE whose user_data is `ud` (slot-generation in the high
232 * 32 bits, slot index in the low 32), validate it against the live slot table
233 * and, if current, fill *ev with the aconn pointer + EPOLL* mask and re-arm the
234 * multishot poll if it auto-disarmed. Returns 1 if *ev was written, 0 if the
235 * CQE was stale/cancelled and dropped. Does NOT mark the CQE seen (caller does).
236 *
237 * WHY: Preserves the UAF discipline shared with the server ring — a poll that
238 * completes after its aconn was reconnected or freed carries an outdated
239 * generation and must be dropped. Isolating the guard + translation keeps the
240 * wait loop within complexity limits without altering the drop semantics.
241 *
242 * HOW:
243 * 1. Split `ud` into slot index and generation.
244 * 2. Drop (return 0) unless the slot is in range, in use, generation-matched,
245 * still bound to an aconn, and the CQE result is non-negative.
246 * 3. Map poll bits to EPOLL* bits: POLLIN|POLLHUP|POLLERR -> EPOLLIN,
247 * POLLOUT -> EPOLLOUT; write aconn pointer + mask into *ev.
248 * 4. If the multishot auto-disarmed (no IORING_CQE_F_MORE), re-arm it.
249 * 5. Return 1.
250 */
251static int
252uring_translate_slot_cqe(brix_loop *l, struct io_uring_cqe *cqe,
253 struct epoll_event *ev, uint64_t ud)
254{
255 uint32_t slot = (uint32_t) (ud & 0xffffffffULL);
256 uint32_t gen = (uint32_t) (ud >> 32);
257 brix_aconn *ac;
258 uint32_t ev_mask = 0;
259
260 if (!(slot < AIO_URING_SLOTS && l->uslots[slot].in_use
261 && l->uslots[slot].gen == gen
262 && l->uslots[slot].ac != NULL && cqe->res >= 0)) {
263 return 0; /* stale/cancelled CQE for a recycled slot — dropped. */
264 }
265
266 ac = l->uslots[slot].ac;
267 if (cqe->res & (POLLIN | POLLHUP | POLLERR)) { ev_mask |= EPOLLIN; }
268 if (cqe->res & POLLOUT) { ev_mask |= EPOLLOUT; }
269 ev->data.ptr = ac;
270 ev->events = ev_mask;
271 /* Re-arm if the multishot auto-disarmed (no F_MORE). */
272 if (!(cqe->flags & IORING_CQE_F_MORE)) {
273 (void) uring_poll_submit(l, ac, ac->epoll_events);
274 }
275 return 1;
276}

279/* ---- Translate one CQE into at most one readiness event ----
280 *
281 * WHAT: Classify a single CQE by its user_data and, for the evfd wake and live
282 * slot polls, fill *ev with the corresponding readiness event. Returns 1 if
283 * *ev was written, 0 otherwise. Always marks the CQE seen before returning.
284 *
285 * WHY: Factors the three-way user_data classification (ignore / evfd / slot)
286 * out of the wait loop so each case is a single linear path; the loop then just
287 * accumulates the return count. The seen-marking is centralised here so every
288 * consumed CQE is retired exactly once regardless of classification.
289 *
290 * HOW:
291 * 1. Read the CQE's user_data.
292 * 2. AIO_URING_IGNORE_UD (a cancel echo): mark seen, return 0.
293 * 3. AIO_URING_EVFD_UD (cross-thread wake): write a loop-tagged EPOLLIN event,
294 * mark seen, return 1.
295 * 4. Otherwise treat it as a slot poll via uring_translate_slot_cqe, mark
296 * seen, and return its result.
297 */
298static int
299uring_translate_cqe(brix_loop *l, struct io_uring_cqe *cqe,
300 struct epoll_event *ev)
301{
302 uint64_t ud = io_uring_cqe_get_data64(cqe);
303 int produced;
304
305 if (ud == AIO_URING_IGNORE_UD) {
306 io_uring_cqe_seen(&l->uring, cqe);
307 return 0;
308 }
309 if (ud == AIO_URING_EVFD_UD) {
310 ev->data.ptr = l;
311 ev->events = EPOLLIN;
312 io_uring_cqe_seen(&l->uring, cqe);
313 return 1;
314 }
315
316 produced = uring_translate_slot_cqe(l, cqe, ev, ud);
317 io_uring_cqe_seen(&l->uring, cqe);
318 return produced;
319}

322/* ---- Drain ready io_uring poll CQEs into evs[] ----
323 *
324 * WHAT: Block up to `timeout_ms` for at least one CQE, then peek and translate
325 * up to `max` CQEs into evs[], returning the number of readiness events written
326 * (never negative — a timeout yields 0).
327 *
328 * WHY: Mirrors epoll_wait's contract for the io_uring engine so io_engine_wait
329 * can select an engine and return, staying a flat two-branch function.
330 *
331 * HOW:
332 * 1. Convert timeout_ms to a __kernel_timespec and wait for one CQE; on any
333 * wait error (timeout / -ETIME / -EINTR) return 0.
334 * 2. While there is room (n < max) and a CQE can be peeked, translate it and
335 * add its 0/1 event contribution to n.
336 * 3. Return n.
337 */
338static int
339uring_drain_cqes(brix_loop *l, struct epoll_event *evs, int max, int timeout_ms)
340{
341 struct io_uring_cqe *cqe;
342 struct __kernel_timespec ts;
343 int n = 0;
344
345 ts.tv_sec = timeout_ms / 1000;
346 ts.tv_nsec = (long) (timeout_ms % 1000) * 1000000L;
347
348 /* Block until at least one CQE or the timeout. */
349 if (io_uring_wait_cqe_timeout(&l->uring, &cqe, &ts) < 0) {
350 return 0; /* timeout / -ETIME / -EINTR: no events this tick */
351 }
352
353 while (n < max && io_uring_peek_cqe(&l->uring, &cqe) == 0) {
354 n += uring_translate_cqe(l, cqe, &evs[n]);
355 }
356 return n;
357}

362/* Fill evs[] with up to `max` readiness events (data.ptr + EPOLL* mask) and
363 * return the count, or -1 on a hard wait error. For io_uring, translate poll
364 * CQEs (generation-guarded) and re-arm any multishot that auto-disarmed. */
365int
366io_engine_wait(brix_loop *l, struct epoll_event *evs, int max, int timeout_ms)
367{
368#if (BRIX_HAVE_LIBURING)
369 if (l->use_uring) {
370 return uring_drain_cqes(l, evs, max, timeout_ms);
371 }
372#endif
373 return epoll_wait(l->epfd, evs, max, timeout_ms);
374}
