8/* ============ Phase 44: pluggable loop I/O engine (epoll | io_uring) ============
9 *
10 * Two readiness engines share one interface. The epoll branch is the historical
11 * code, preserved verbatim. The io_uring branch (multishot IORING_OP_POLL_ADD,
12 * default OFF — gated by XRDC_IO_URING_LOOP=on + a runtime probe, best-effort
13 * fallback to epoll) is a drop-in readiness source: the loop still runs
14 * aconn_do_read/aconn_do_write unchanged, so TLS (which drives the fd through
15 * OpenSSL itself) is safe. Cross-thread wake is write(evfd) for BOTH engines —
16 * the io_uring engine arms a multishot poll on the same evfd.
17 *
18 * UAF safety: a poll's user_data carries (slot-generation<<32 | slot); the slot
19 * table maps back to the aconn and the reaper drops any CQE whose generation no
20 * longer matches (poll completing after its aconn was reconnected/freed) — the
21 * same discipline as the server ring. fd changes (reconnect) cancel the old
22 * poll and bump the slot generation before re-arming the new fd. */
23

27/* epoll interest mask -> poll(2) mask for io_uring_prep_poll_*. */
28unsigned
29uring_pollmask(int want)
30{
31 unsigned m = 0;
32 if (want & EPOLLIN) { m |= POLLIN; }
33 if (want & EPOLLOUT) { m |= POLLOUT; }
34 return m;
35}

38/* Claim a free poll slot for ac (loop-thread only; no lock). Returns 0 / -1. */
39int
40uring_slot_alloc(brix_loop *l, brix_aconn *ac)
41{
42 unsigned i;
43 for (i = 0; i < AIO_URING_SLOTS; i++) {
44 if (!l->uslots[i].in_use) {
45 l->uslots[i].in_use = 1;
46 l->uslots[i].ac = ac;
47 ac->uring_slot = (int) i;
48 return 0;
49 }
50 }
51 return -1; /* table full — pool has more conns than slots (shouldn't happen) */
52}

55/* Submit a multishot poll for ac with the current slot generation as user_data. */
56int
57uring_poll_submit(brix_loop *l, brix_aconn *ac, int want)
58{
59 struct io_uring_sqe *sqe = io_uring_get_sqe(&l->uring);
60 uint32_t slot = (uint32_t) ac->uring_slot;
61
62 if (sqe == NULL) {
63 return -1;
64 }
65 io_uring_prep_poll_multishot(sqe, ac->fd, uring_pollmask(want));
66 io_uring_sqe_set_data64(sqe, ((uint64_t) l->uslots[slot].gen << 32) | slot);
67 return io_uring_submit(&l->uring) < 0 ? -1 : 0;
68}

71/* Cancel ac's outstanding poll (by current user_data) and bump the slot gen so
72 * any late CQE for it is dropped. Keeps the slot allocated (caller re-arms) or
73 * frees it when freeing == 1. */
74void
75uring_poll_cancel(brix_loop *l, brix_aconn *ac, int freeing)
76{
77 uint32_t slot = (uint32_t) ac->uring_slot;
78 struct io_uring_sqe *sqe;
79
80 if (ac->uring_slot < 0) {
81 return;
82 }
83 sqe = io_uring_get_sqe(&l->uring);
84 if (sqe != NULL) {
85 io_uring_prep_poll_remove(sqe,
86 ((uint64_t) l->uslots[slot].gen << 32) | slot);
87 io_uring_sqe_set_data64(sqe, AIO_URING_IGNORE_UD);
88 (void) io_uring_submit(&l->uring);
89 }
90 l->uslots[slot].gen++; /* invalidate the old poll's user_data */
91 if (freeing) {
92 l->uslots[slot].in_use = 0;
93 l->uslots[slot].ac = NULL;
94 ac->uring_slot = -1;
95 }
96}

100/* Create the readiness set + the wake eventfd. evfd is used by both engines;
101 * epoll registers it in the set, io_uring arms a multishot poll on it. */
102int
103io_engine_setup(brix_loop *l, brix_status *st)
104{
105 l->evfd = eventfd(0, EFD_NONBLOCK | EFD_CLOEXEC);
106 if (l->evfd < 0) {
107 brix_status_set(st, XRDC_ESOCK, errno, "eventfd: %s", strerror(errno));
108 return -1;
109 }
110
111#if (BRIX_HAVE_LIBURING)
112 if (l->use_uring) {
113 struct io_uring_sqe *sqe;
114 if (io_uring_queue_init(256, &l->uring, 0) < 0) {
115 brix_status_set(st, XRDC_ESOCK, errno, "io_uring_queue_init");
116 return -1;
117 }
118 l->uring_ok = 1;
119 sqe = io_uring_get_sqe(&l->uring); /* multishot poll on the evfd */
120 if (sqe == NULL) {
121 brix_status_set(st, XRDC_ESOCK, 0, "io_uring evfd arm");
122 return -1;
123 }
124 io_uring_prep_poll_multishot(sqe, l->evfd, POLLIN);
125 io_uring_sqe_set_data64(sqe, AIO_URING_EVFD_UD);
126 if (io_uring_submit(&l->uring) < 0) {
127 brix_status_set(st, XRDC_ESOCK, errno, "io_uring evfd submit");
128 return -1;
129 }
130 return 0;
131 }
132#endif
133
134 l->epfd = epoll_create1(EPOLL_CLOEXEC);
135 if (l->epfd < 0) {
136 brix_status_set(st, XRDC_ESOCK, errno, "epoll_create1: %s",
137 strerror(errno));
138 return -1;
139 }
140 {
141 struct epoll_event ev;
142 memset(&ev, 0, sizeof(ev));
143 ev.events = EPOLLIN;
144 ev.data.ptr = l; /* loop pointer tags the eventfd */
145 if (epoll_ctl(l->epfd, EPOLL_CTL_ADD, l->evfd, &ev) != 0) {
146 brix_status_set(st, XRDC_ESOCK, errno, "epoll_ctl(evfd): %s",
147 strerror(errno));
148 return -1;
149 }
150 }
151 return 0;
152}

173/* Arm/modify interest for ac to `want`; sets ac->epoll_events on success.
174 * Returns 0 / -1 (caller keeps its own failure handling). */
175int
176io_engine_arm(brix_loop *l, brix_aconn *ac, int want)
177{
178#if (BRIX_HAVE_LIBURING)
179 if (l->use_uring) {
180 if (ac->uring_slot < 0) {
181 if (uring_slot_alloc(l, ac) != 0) { return -1; }
182 } else {
183 uring_poll_cancel(l, ac, 0); /* drop the old mask's poll, keep slot */
184 }
185 ac->fd_gen++;
186 if (uring_poll_submit(l, ac, want) != 0) { return -1; }
187 ac->epoll_events = want;
188 return 0;
189 }
190#endif
191 {
192 struct epoll_event ev;
193 int op = (ac->epoll_events == 0) ? EPOLL_CTL_ADD : EPOLL_CTL_MOD;
194 memset(&ev, 0, sizeof(ev));
195 ev.events = (uint32_t) want;
196 ev.data.ptr = ac;
197 if (epoll_ctl(l->epfd, op, ac->fd, &ev) != 0) {
198 return -1;
199 }
200 ac->epoll_events = want;
201 return 0;
202 }
203}

211 uring_poll_cancel(l, ac, 1); /* cancel + free the slot */

223/* Fill evs[] with up to `max` readiness events (data.ptr + EPOLL* mask) and
224 * return the count, or -1 on a hard wait error. For io_uring, translate poll
225 * CQEs (generation-guarded) and re-arm any multishot that auto-disarmed. */
226int
227io_engine_wait(brix_loop *l, struct epoll_event *evs, int max, int timeout_ms)
228{
229#if (BRIX_HAVE_LIBURING)
230 if (l->use_uring) {
231 struct io_uring_cqe *cqe;
232 struct __kernel_timespec ts;
233 int n = 0;
234
235 ts.tv_sec = timeout_ms / 1000;
236 ts.tv_nsec = (long) (timeout_ms % 1000) * 1000000L;
237
238 /* Block until at least one CQE or the timeout. */
239 if (io_uring_wait_cqe_timeout(&l->uring, &cqe, &ts) < 0) {
240 return 0; /* timeout / -ETIME / -EINTR: no events this tick */
241 }
242
243 while (n < max && io_uring_peek_cqe(&l->uring, &cqe) == 0) {
244 uint64_t ud = io_uring_cqe_get_data64(cqe);
245
246 if (ud == AIO_URING_IGNORE_UD) {
247 io_uring_cqe_seen(&l->uring, cqe);
248 continue;
249 }
250 if (ud == AIO_URING_EVFD_UD) {
251 evs[n].data.ptr = l;
252 evs[n].events = EPOLLIN;
253 n++;
254 io_uring_cqe_seen(&l->uring, cqe);
255 continue;
256 }
257 {
258 uint32_t slot = (uint32_t) (ud & 0xffffffffULL);
259 uint32_t gen = (uint32_t) (ud >> 32);
260
261 if (slot < AIO_URING_SLOTS && l->uslots[slot].in_use
262 && l->uslots[slot].gen == gen
263 && l->uslots[slot].ac != NULL && cqe->res >= 0)
264 {
265 brix_aconn *ac = l->uslots[slot].ac;
266 uint32_t ev = 0;
267 if (cqe->res & (POLLIN | POLLHUP | POLLERR)) { ev |= EPOLLIN; }
268 if (cqe->res & POLLOUT) { ev |= EPOLLOUT; }
269 evs[n].data.ptr = ac;
270 evs[n].events = ev;
271 n++;
272 /* Re-arm if the multishot auto-disarmed (no F_MORE). */
273 if (!(cqe->flags & IORING_CQE_F_MORE)) {
274 (void) uring_poll_submit(l, ac, ac->epoll_events);
275 }
276 }
277 /* else: stale/cancelled CQE for a recycled slot — dropped. */
278 io_uring_cqe_seen(&l->uring, cqe);
279 }
280 }
281 return n;
282 }
283#endif
284 return epoll_wait(l->epfd, evs, max, timeout_ms);
285}
