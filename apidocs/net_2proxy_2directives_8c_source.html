1/*
2 * directives.c — nginx config directives for brix_proxy and brix_proxy_upstream.
3 *
4 * WHAT: Parses and validates nginx configuration directives for proxy mode operation.
5 * Provides four directive handlers: brix_conf_set_proxy_upstream (register upstream endpoints),
6 * brix_conf_set_proxy_auth (set global auth policy), brix_conf_set_proxy_login_user
7 * (control username sent to upstream), and brix_conf_set_proxy_path_rewrite (strip/add path prefix).
8 * Also includes static helper proxy_parse_host_port for parsing host:port address strings.
9 *
10 * WHY: Proxy mode requires configurable upstream endpoints with per-endpoint auth policies,
11 * username passthrough/fixed/anonymous modes, and optional path rewriting for namespace mapping.
12 * These directives populate conf->proxy.upstreams array, conf->proxy.auth policy enum,
13 * conf->proxy.login_user mode, and strip/add prefix strings — all consumed by proxy forward
14 * handlers (forward_request.c, forward_relay.c). Backward compat: first upstream occurrence sets
15 * legacy single-upstream fields (proxy_host/proxy_port) for prior configuration style.
16 *
17 * HOW: Each directive handler reads cf->args->elts for parsed argument values. proxy_parse_host_port()
18 * handles IPv6 literal [addr]:port and IPv4/hostname host:port formats — zeroing colon delimiter,
19 * extracting host string via ngx_pnalloc, parsing port via strtol with 1-65535 range validation.
20 * brix_conf_set_proxy_upstream() appends to conf->proxy.upstreams array (lazily created), sets
21 * host/port/auth=-1 (inherit global default), parses optional third arg for per-upstream auth override.
22 * brix_conf_set_proxy_auth() maps value string to BRIX_PROXY_AUTH_* enum, logs EMERG on invalid.
23 * brix_conf_set_proxy_login_user() handles anonymous/passthrough/fixed:<name> (1-8 chars) modes.
24 * brix_conf_set_proxy_path_rewrite() copies two argument values to strip/add fields directly.
25 */

34/*
35 * Parse "host[:port]" from addr_copy into *host_out / *port_out.
36 * addr_copy is NUL-terminated and may be modified (colon zeroing).
37 * Returns NGX_CONF_ERROR on parse failure.
38 */
39static char *
40proxy_parse_host_port(ngx_conf_t *cf, ngx_str_t *value,
41 char *addr_copy,
42 ngx_str_t *host_out, uint16_t *port_out)
43{
44 char *colon, *endp;
45 long pnum;
46
47 if (addr_copy[0] == '[') {
48 /* IPv6 literal: [addr]:port */
49 char *rb = strchr(addr_copy, ']');
50 size_t hostlen;
51
52 if (rb == NULL || *(rb + 1) != ':') {
53 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
54 "brix_proxy_upstream: invalid address \"%V\"", value);
55 return NGX_CONF_ERROR;
56 }
57 hostlen = (size_t)(rb - addr_copy - 1);
58 host_out->data = ngx_pnalloc(cf->pool, hostlen + 1);
59 if (host_out->data == NULL) {
60 return NGX_CONF_ERROR;
61 }
62 ngx_memcpy(host_out->data, addr_copy + 1, hostlen);
63 host_out->data[hostlen] = '\0';
64 host_out->len = hostlen;
65 pnum = strtol(rb + 2, &endp, 10);
66 } else {
67 /* IPv4 or hostname: host:port or host (default port) */
68 colon = strrchr(addr_copy, ':');
69 if (colon == NULL) {
70 host_out->data = (u_char *) addr_copy;
71 host_out->len = value->len;
72 *port_out = 1094;
73 return NGX_CONF_OK;
74 }
75 size_t hostlen = (size_t)(colon - addr_copy);
76 host_out->data = ngx_pnalloc(cf->pool, hostlen + 1);
77 if (host_out->data == NULL) {
78 return NGX_CONF_ERROR;
79 }
80 ngx_memcpy(host_out->data, addr_copy, hostlen);
81 host_out->data[hostlen] = '\0';
82 host_out->len = hostlen;
83 pnum = strtol(colon + 1, &endp, 10);
84 }
85
86 if (*endp != '\0' || pnum <= 0 || pnum > 65535) {
87 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
88 "brix_proxy_upstream: invalid port in \"%V\"", value);
89 return NGX_CONF_ERROR;
90 }
91 *port_out = (uint16_t) pnum;
92 return NGX_CONF_OK;
93}

95/* ---- Parse an optional per-upstream auth-policy argument into an entry ----
96 *
97 * WHAT: Maps auth_arg ("anonymous", "forward", "sss", or "sss:<keyname>") onto
98 * entry->auth (a BRIX_PROXY_AUTH_* value) and, for the keyed sss form,
99 * copies the key name into entry->sss_keyname. Returns NGX_CONF_OK on a
100 * recognised policy and NGX_CONF_ERROR (after logging EMERG) for an
101 * unknown policy or an over-long sss key name.
102 *
103 * WHY: brix_conf_set_proxy_upstream's optional third argument carries a small
104 * four-way policy grammar with its own nested validation. Confining it to a
105 * single helper keeps the directive handler a flat append-and-validate
106 * sequence and keeps this file's per-function cyclomatic complexity in
107 * bounds without changing any accepted/rejected input or field assignment.
108 *
109 * HOW: 1. Exact-length compare against "anonymous" (9) and "forward" (7).
110 * 2. Prefix-match "sss" (>= 3); set BRIX_PROXY_AUTH_SSS.
111 * 3. For the "sss:<keyname>" form (len > 4, data[3] == ':') validate the
112 * key length against BRIX_SSS_NAME_MAX, then copy and NUL-terminate it.
113 * 4. Any other value is an error: log EMERG and return NGX_CONF_ERROR.
114 */
115static char *
116proxy_parse_upstream_auth(ngx_conf_t *cf, ngx_str_t *auth_arg,
117 brix_proxy_upstream_t *entry)
118{
119 if (auth_arg->len == 9
120 && ngx_strncmp(auth_arg->data, "anonymous", 9) == 0)
121 {
122 entry->auth = BRIX_PROXY_AUTH_ANONYMOUS;
123 return NGX_CONF_OK;
124 }
125
126 if (auth_arg->len == 7
127 && ngx_strncmp(auth_arg->data, "forward", 7) == 0)
128 {
129 entry->auth = BRIX_PROXY_AUTH_FORWARD;
130 return NGX_CONF_OK;
131 }
132
133 if (auth_arg->len >= 3
134 && ngx_strncmp(auth_arg->data, "sss", 3) == 0)
135 {
136 entry->auth = BRIX_PROXY_AUTH_SSS;
137 /* Optional key name after colon: "sss" or "sss:<keyname>" */
138 if (auth_arg->len > 4 && auth_arg->data[3] == ':') {
139 size_t klen = auth_arg->len - 4;
140 if (klen >= BRIX_SSS_NAME_MAX) {
141 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
142 "brix_proxy_upstream: sss key name too long");
143 return NGX_CONF_ERROR;
144 }
145 ngx_memcpy(entry->sss_keyname, auth_arg->data + 4, klen);
146 entry->sss_keyname[klen] = '\0';
147 }
148 return NGX_CONF_OK;
149 }
150
151 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
152 "brix_proxy_upstream: invalid auth policy \"%V\"; "
153 "use anonymous, forward, sss, or sss:<keyname>", auth_arg);
154 return NGX_CONF_ERROR;
155}

157/*
158 * brix_proxy_upstream "host:port" [auth-policy]
159 *
160 * Appends to conf->proxy.upstreams (round-robin pool). The optional second
161 * argument overrides the global brix_proxy_auth policy for this specific
162 * upstream endpoint:
163 *
164 * anonymous — send no credentials to this upstream
165 * forward — replay the client's bearer token
166 * sss — build an SSS credential from conf->sss_keys[0]
167 * sss:<keyname> — build an SSS credential from the named key
168 *
169 * When the auth argument is omitted the global conf->proxy.auth applies.
170 * The directive may appear multiple times to register several backends.
171 */
172char *
173brix_conf_set_proxy_upstream(ngx_conf_t *cf, ngx_command_t *cmd, void *conf_ptr)
174{
175 ngx_stream_brix_srv_conf_t *conf = conf_ptr;
176 ngx_str_t *value;
177 char *addr_copy, *rc;
178 ngx_str_t host;
179 uint16_t port = 1094;
180 brix_proxy_upstream_t *entry;
181
182 (void) cmd;
183
184 value = cf->args->elts;
185
186 addr_copy = brix_pstrdup_z(cf->pool, &value[1]);
187 if (addr_copy == NULL) {
188 return NGX_CONF_ERROR;
189 }
190
191 host.data = NULL;
192 host.len = 0;
193
194 rc = proxy_parse_host_port(cf, &value[1], addr_copy, &host, &port);
195 if (rc != NGX_CONF_OK) {
196 return rc;
197 }
198
199 /* Append to the upstreams array (lazily created) */
200 if (conf->proxy.upstreams == NULL) {
201 conf->proxy.upstreams = ngx_array_create(cf->pool, 2,
202 sizeof(brix_proxy_upstream_t));
203 if (conf->proxy.upstreams == NULL) {
204 return NGX_CONF_ERROR;
205 }
206 }
207 entry = ngx_array_push(conf->proxy.upstreams);
208 if (entry == NULL) {
209 return NGX_CONF_ERROR;
210 }
211 entry->host = host;
212 entry->port = port;
213 entry->auth = -1; /* inherit global proxy_auth by default */
214 entry->sss_keyname[0] = '\0';
215
216 /* Parse optional per-upstream auth policy argument */
217 if (cf->args->nelts >= 3) {
218 rc = proxy_parse_upstream_auth(cf, &value[2], entry);
219 if (rc != NGX_CONF_OK) {
220 return rc;
221 }
222 }
223
224 /* Backward compat: first occurrence sets the legacy single-upstream fields */
225 if (conf->proxy.host.len == 0) {
226 conf->proxy.host = host;
227 conf->proxy.port = (ngx_int_t) port;
228 }
229
230 return NGX_CONF_OK;
231}

233/*
234 * brix_proxy_auth anonymous|forward|sss
235 *
236 * "anonymous" — upstream login uses no credentials (default).
237 * "forward" — forward the client's WLCG bearer token to upstream on kXR_authmore.
238 * "sss" — build an SSS credential from conf->sss_keys[0] when upstream
239 * requests SSS authentication via kXR_authmore.
240 */
241char *
242brix_conf_set_proxy_auth(ngx_conf_t *cf, ngx_command_t *cmd, void *conf_ptr)
243{
244 ngx_stream_brix_srv_conf_t *conf = conf_ptr;
245 ngx_str_t *value;
246
247 (void) cmd;
248
249 value = cf->args->elts;
250
251 if (ngx_strcmp(value[1].data, "anonymous") == 0) {
252 conf->proxy.auth = BRIX_PROXY_AUTH_ANONYMOUS;
253 } else if (ngx_strcmp(value[1].data, "forward") == 0) {
254 conf->proxy.auth = BRIX_PROXY_AUTH_FORWARD;
255 } else if (ngx_strcmp(value[1].data, "sss") == 0) {
256 conf->proxy.auth = BRIX_PROXY_AUTH_SSS;
257 } else if (ngx_strcmp(value[1].data, "gsi") == 0) {
258 conf->proxy.auth = BRIX_PROXY_AUTH_GSI;
259 } else {
260 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
261 "brix_proxy_auth: invalid value \"%V\"; "
262 "use anonymous, forward, sss, or gsi", &value[1]);
263 return NGX_CONF_ERROR;
264 }
265
266 return NGX_CONF_OK;
267}

269/*
270 * brix_proxy_login_user anonymous|passthrough|fixed:<name>
271 *
272 * Controls the username placed in the upstream kXR_login frame:
273 *
274 * anonymous — always send "xrd" (default, preserves prior behaviour)
275 * passthrough — copy the client's authenticated username (from kXR_login),
276 * truncated to 8 bytes if necessary. Falls back to "xrd" for
277 * anonymous client sessions.
278 * fixed:<name> — always send the literal <name> (1–8 characters)
279 */
280char *
281brix_conf_set_proxy_login_user(ngx_conf_t *cf, ngx_command_t *cmd,
282 void *conf_ptr)
283{
284 ngx_stream_brix_srv_conf_t *conf = conf_ptr;
285 ngx_str_t *value;
286
287 (void) cmd;
288
289 value = cf->args->elts;
290
291 if (ngx_strcmp(value[1].data, "anonymous") == 0) {
292 conf->proxy.login_user = BRIX_PROXY_LOGIN_ANONYMOUS;
293
294 } else if (ngx_strcmp(value[1].data, "passthrough") == 0) {
295 conf->proxy.login_user = BRIX_PROXY_LOGIN_PASSTHROUGH;
296
297 } else if (value[1].len >= 7
298 && ngx_strncmp(value[1].data, "fixed:", 6) == 0)
299 {
300 size_t nlen = value[1].len - 6;
301 if (nlen == 0 || nlen > 8) {
302 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
303 "brix_proxy_login_user: fixed name must be 1-8 characters");
304 return NGX_CONF_ERROR;
305 }
306 conf->proxy.login_user = BRIX_PROXY_LOGIN_FIXED;
307 ngx_memcpy(conf->proxy.login_user_name, value[1].data + 6, nlen);
308 conf->proxy.login_user_name[nlen] = '\0';
309
310 } else {
311 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
312 "brix_proxy_login_user: invalid value \"%V\"; "
313 "use anonymous, passthrough, or fixed:<name>", &value[1]);
314 return NGX_CONF_ERROR;
315 }
316
317 return NGX_CONF_OK;
318}
