1/*
2 * stage_engine_scheduler.c — the async front door + per-worker scheduler of the
3 * staging engine (split from stage_engine.c, phase-79).
4 *
5 * WHAT: Owns the submit front door (brix_stage_submit — run inline or defer onto
6 * the per-worker FIFO), the terminal post-processing of a completed mover
7 * (stage_complete — drop a flushed stage copy, dead-letter a permanent deny,
8 * keep a transient failure), the optional thread-pool offload of a flush,
9 * and the per-worker timer that drains the deferred queue
10 * (brix_stage_scheduler_tick).
11 *
12 * WHY: stage_engine.c owned three concerns and exceeded the file-size cap. The
13 * queue + scheduler is a self-contained concern: it builds the in-memory
14 * pending FIFO, persists it through the journal seam, and drains it through
15 * the generic mover seam. The FIFO ordering, the bounded per-tick budget,
16 * the bounded in-flight offload count, and the inline fallback are unchanged
17 * by the split — this is a re-home, not a behavior change.
18 *
19 * HOW: brix_stage_submit appends a stage_pending_t and journals it;
20 * brix_stage_scheduler_tick pops the head, offloads to the thread pool when
21 * one exists (bounded), else runs the mover inline, and calls stage_complete
22 * on the terminal result. All durable-record I/O is reached through
23 * stage_engine_internal.h; the byte mover is stage_engine_run.
24 */

42const char *
43brix_stage_submit(brix_stage_kind_t kind, brix_sd_instance_t *src,
44 const char *src_key, brix_sd_instance_t *dst, const char *dst_key,
45 const brix_stage_opts_t *opts)
46{
47 static const char ran_inline[] = "";
48 static char last_reqid[40]; /* event-loop single-threaded: stable enough */
49 stage_pending_t *p;
50
51 if (src == NULL || src_key == NULL || dst == NULL || dst_key == NULL
52 || src->driver == NULL || dst->driver == NULL)
53 {
54 return NULL;
55 }
56
57 /* Synchronous (the default): move now and return "" = done inline. */
58 if (opts == NULL || !opts->async) {
59 (void) stage_engine_run(kind, src, src_key, dst, dst_key,
60 (opts != NULL) ? opts->cred : NULL);
61 return ran_inline;
62 }
63
64 /* Asynchronous: defer to the scheduler (durable). The src/dst instances are
65 * the memoised per-worker tier instances, so holding them is safe. */
66 p = calloc(1, sizeof(*p));
67 if (p == NULL) {
68 (void) stage_engine_run(kind, src, src_key, dst, dst_key, opts->cred);
69 return ran_inline;
70 }
71 stage_reqid_mint(p->reqid);
72 p->kind = kind;
73 p->src = src;
74 p->dst = dst;
75 snprintf(p->src_key, sizeof(p->src_key), "%s", src_key);
76 snprintf(p->dst_key, sizeof(p->dst_key), "%s", dst_key);
77 if (opts->export_root != NULL) {
78 snprintf(p->export_root, sizeof(p->export_root), "%s", opts->export_root);
79 }
80 if (opts->cred != NULL) {
81 p->cred = *opts->cred; /* copy the owner identity into the pending item */
82 }
83
84 if (stage_pending_tail != NULL) {
85 stage_pending_tail->next = p;
86 } else {
87 stage_pending_head = p;
88 }
89 stage_pending_tail = p;
90
91 stage_journal_write(p);
92 snprintf(last_reqid, sizeof(last_reqid), "%s", p->reqid);
93 return last_reqid; /* non-empty = deferred; the caller may park on it */
94}

100/* Terminal post-processing for a completed mover (inline or thread): on success a
101 * FLUSH drops the now-redundant stage copy (the migrate semantic, §9.5) and the
102 * journal record is removed; on a permanent deny (BRIX_XFER_DENIED) the attempt
103 * counter is bumped and, when the cap is reached, the record is moved to the
104 * deadletter/ subdirectory (stage copy kept for operator recovery); on any other
105 * failure the record is KEPT (transient — the reconcile retries on restart or a
106 * later tick re-drives it). Shared by both the scheduler and thread paths.
107 *
108 * WHAT: Distinguishes three outcomes: OK (remove journal + drop stage copy),
109 * DENIED (call stage_deny_terminal to enforce the dead-letter cap), and
110 * all other errors (WARN + keep for retry).
111 *
112 * WHY: A BRIX_XFER_DENIED result is not transient: the per-user credential is
113 * permanently missing/expired in deny mode. Treating it like a transient
114 * error leaves the record looping forever. The dead-letter cap bounds
115 * that loop while preserving the data for operator recovery.
116 *
117 * HOW: On DENIED, re-reads the on-disk record to get the current attempts
118 * count (surviving across restarts), then delegates to stage_deny_terminal
119 * which bumps, persists, and conditionally moves to deadletter/. */
120static void
121stage_complete(brix_stage_kind_t kind, brix_sd_instance_t *src,
122 const char *src_key, const char *dst_key, const char *reqid,
123 brix_xfer_result_t res, int last_errno, ngx_log_t *log)
124{
125 if (res == BRIX_XFER_OK) {
126 if (kind == BRIX_STAGE_FLUSH && src->driver->unlink != NULL) {
127 (void) src->driver->unlink(src, src_key, 0);
128 }
129 stage_journal_remove(reqid);
130 return;
131 }
132
133 if (res == BRIX_XFER_DENIED && stage_journal_dir[0] != '\0') {
134 /* Permanent deny: re-read the on-disk record to get the current
135 * attempts count (may have been bumped by a prior drive after a
136 * restart), then apply the dead-letter cap. */
137 char path[1200];
138 char rbuf[sizeof(brix_sreq_t)];
139 brix_sreq_t rec;
140 int fd;
141 ssize_t n;
142
143 if ((size_t) snprintf(path, sizeof(path), "%s/%s.req",
144 stage_journal_dir, reqid) < sizeof(path))
145 {
146 fd = open(path, O_RDONLY | O_CLOEXEC);
147 if (fd >= 0) {
148 n = read(fd, rbuf, sizeof(rbuf));
149 (void) close(fd);
150 if (brix_sreq_decode(rbuf, (size_t) n, &rec) == NGX_OK) {
151 (void) stage_deny_terminal(stage_journal_dir, reqid, &rec,
152 log);
153 return;
154 }
155 }
156 }
157 /* Journal record unreadable: fall through to the WARN+keep path so
158 * the transient I/O error does not silently swallow the failure. */
159 }
160
161 /* Transient failure (dead/unreachable origin): mark the durable record
162 * FAILED so a crash-visible, replayable row survives — the restart reconcile
163 * re-drives it. Left in the ACTIVE journal (never dead-letter): unlike a
164 * permanent deny, a transient origin outage is expected to clear. */
165 stage_journal_mark_failed(stage_journal_dir, reqid, last_errno);
166
167 ngx_log_error(NGX_LOG_WARN, log, 0,
168 "xrootd stage: deferred %s of \"%s\" failed (reqid %s errno %d) - "
169 "record marked FAILED, kept for retry",
170 brix_stage_kind_str(kind), dst_key, reqid, last_errno);
171}

175/* In-flight offloaded movers (per-worker), bounded so a burst never floods the
176 * thread pool nor unbounds memory. The tick stops starting new ones at the cap and
177 * resumes next tick as they drain. */
178static ngx_uint_t stage_inflight;
179#define STAGE_MAX_INFLIGHT 8

181/* The off-loop mover task (lives on its own small pool, freed in the done event).
182 * `cred` carries the owner identity so the flush thread can re-resolve the per-user
183 * proxy at flush time via stage_engine_run's cred logic. */
184typedef struct {
185 ngx_pool_t *pool;
186 brix_stage_kind_t kind;
187 brix_sd_instance_t *src;
188 brix_sd_instance_t *dst;
189 brix_xfer_result_t res;
190 int last_errno; /* mover errno, carried to the done event */
191 ngx_log_t *log;
192 char reqid[40];
193 char src_key[1024];
194 char dst_key[1024];
195 brix_stage_cred_t cred; /* owner identity for per-user cred check */
196} stage_flush_task_t;

198static void
199stage_flush_thread(void *data, ngx_log_t *log)
200{
201 stage_flush_task_t *t = data;
202 const brix_stage_cred_t *credp = (t->cred.key[0] != '\0') ? &t->cred : NULL;
203
204 (void) log;
205 errno = 0;
206 t->res = stage_engine_run(t->kind, t->src, t->src_key, t->dst, t->dst_key,
207 credp);
208 t->last_errno = (t->res == BRIX_XFER_OK) ? 0 : errno;

211static void
212stage_flush_done(ngx_event_t *ev)
213{
214 ngx_thread_task_t *task = ev->data;
215 stage_flush_task_t *t = task->ctx;
216 ngx_pool_t *pool = t->pool;
217
218 if (stage_inflight > 0) {
219 stage_inflight--;
220 }
221 stage_complete(t->kind, t->src, t->src_key, t->dst_key, t->reqid, t->res,
222 t->last_errno, t->log);
223 ngx_destroy_pool(pool); /* frees the task + ctx */
224}

226/* Post `p`'s mover to `pool`. NGX_OK = posted (caller drops the pending item; the
227 * journal record persists until the thread completes); NGX_DECLINED = setup failed
228 * (caller runs it inline). */
229static ngx_int_t
230stage_flush_offload(const stage_pending_t *p, ngx_thread_pool_t *pool)
231{
232 ngx_log_t *log = (p->dst->log != NULL) ? p->dst->log : p->src->log;
233 ngx_pool_t *tp;
234 ngx_thread_task_t *task;
235 stage_flush_task_t *t;
236
237 if (log == NULL) {
238 log = ngx_cycle->log;
239 }
240 tp = ngx_create_pool(4096, log);
241 if (tp == NULL) {
242 return NGX_DECLINED;
243 }
244 task = ngx_thread_task_alloc(tp, sizeof(stage_flush_task_t));
245 if (task == NULL) {
246 ngx_destroy_pool(tp);
247 return NGX_DECLINED;
248 }
249 t = task->ctx;
250 t->pool = tp;
251 t->kind = p->kind;
252 t->src = p->src;
253 t->dst = p->dst;
254 t->res = BRIX_XFER_DST_ERR;
255 t->last_errno = EIO;
256 t->log = log;
257 t->cred = p->cred; /* copy owner identity for cred re-resolution in thread */
258 snprintf(t->reqid, sizeof(t->reqid), "%s", p->reqid);
259 snprintf(t->src_key, sizeof(t->src_key), "%s", p->src_key);
260 snprintf(t->dst_key, sizeof(t->dst_key), "%s", p->dst_key);
261
262 brix_task_bind(task, stage_flush_thread, stage_flush_done);
263 task->event.log = log;
264 if (ngx_thread_task_post(pool, task) != NGX_OK) {
265 ngx_destroy_pool(tp);
266 return NGX_DECLINED;
267 }
268 stage_inflight++;
269 return NGX_OK;
270}

284void
285brix_stage_scheduler_tick(void)
286{
287 int budget = STAGE_TICK_BUDGET;
288#if (NGX_THREADS)
289 ngx_thread_pool_t *pool = stage_thread_pool();
290#endif
291
292 while (stage_pending_head != NULL && budget-- > 0) {
293 stage_pending_t *p = stage_pending_head;
294 brix_xfer_result_t res;
295 int oerr;
296 ngx_log_t *log = (p->dst->log != NULL) ? p->dst->log
297 : p->src->log;
298
299#if (NGX_THREADS)
300 /* Run the mover OFF the event loop so a flush to a REMOTE backend (or from a
301 * remote stage store) never blocks/fails on the un-pumped loop. Bounded
302 * in-flight; the durable journal record is what survives a crash mid-flush
303 * (recovered by reconcile), so the on-disk record is dropped only in the
304 * completion. */
305 if (pool != NULL) {
306 if (stage_inflight >= STAGE_MAX_INFLIGHT) {
307 break; /* let in-flight drain; resume next tick */
308 }
309 if (stage_flush_offload(p, pool) == NGX_OK) {
310 stage_pending_head = p->next;
311 if (stage_pending_head == NULL) {
312 stage_pending_tail = NULL;
313 }
314 free(p);
315 continue;
316 }
317 /* offload setup failed - fall through to the inline path */
318 }
319#endif
320
321 stage_pending_head = p->next;
322 if (stage_pending_head == NULL) {
323 stage_pending_tail = NULL;
324 }
325 {
326 const brix_stage_cred_t *credp =
327 (p->cred.key[0] != '\0') ? &p->cred : NULL;
328 errno = 0;
329 res = stage_engine_run(p->kind, p->src, p->src_key, p->dst,
330 p->dst_key, credp);
331 oerr = (res == BRIX_XFER_OK) ? 0 : errno;
332 }
333 stage_complete(p->kind, p->src, p->src_key, p->dst_key, p->reqid, res,
334 oerr, log);
335 free(p);
336 }
337}
