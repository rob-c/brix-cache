1/*
2 * resilient.c — network resilience for the synchronous client tools.
3 *
4 * WHAT: gives the one-shot CLI tools the same recovery the xrootdfs FUSE driver
5 * has — reconnect + full re-auth + handle reopen + offset resume + bounded
6 * backoff — through two seams:
7 * - brix_with_resilience(): wrap any stateless op (stat/ls/query/mkdir/…)
8 * so a transport sever triggers reconnect + re-issue, gated by an
9 * idempotency class.
10 * - brix_rfile: a synchronous file handle that reopens and resumes a read
11 * or write at the same offset after a sever, with an adaptive read size.
12 *
13 * WHY: a single brix_connect() + one-shot brix_send/brix_recv fails the whole
14 * operation on the first dropped connection (rc=51). These helpers ride the
15 * loss out, matching the FUSE driver, while staying off (single attempt)
16 * when the window is 0 so --no-retry preserves the legacy fail-fast path.
17 *
18 * HOW: the read/write loops are lifted from the proven xrdcp pump (copy.c:
19 * pump_src_remote / pump_remote_reopen) and the async mfile_pwrite
20 * (aio_mgr.c); reconnect/re-auth is brix_reconnect (conn.c) to the home
21 * endpoint; retryability and backoff reuse brix_status_retryable /
22 * brix_backoff_sleep_fast. Raw ops and copy.c are left untouched, so there
23 * is no nested-retry surprise.
24 */

33/* Adaptive read size: start large for full throughput on a clean link, halve on
34 * each sever down to the floor so a lossy link converges on a request small
35 * enough to get through (mirrors copy.c's XRDC_RESILIENT_FLOOR). */
36#define XRDC_RFILE_CHUNK0 (4u * 1024u * 1024u)
37#define XRDC_RFILE_FLOOR (256u * 1024u)

39/* Resolve the resilience window (ms) from options: 0 if disabled, else the
40 * explicit value, else $XRDC_MAX_STALL_MS, else the built-in default. The env
41 * fallback keeps the knob uniform for tools that don't route options through
42 * cli_opts (e.g. the hand-rolled xrdfs parser). */
43static int
44opts_window_ms(const brix_opts *o)
45{
46 if (o != NULL) {
47 if (o->no_retry) {
48 return 0;
49 }
50 if (o->max_stall_ms > 0) {
51 return o->max_stall_ms;
52 }
53 }
54 const char *e = getenv("XRDC_MAX_STALL_MS");
55 if (e != NULL && *e != '\0') {
56 int v = atoi(e);
57 return (v > 0) ? v : 0;
58 }
59 { int xv; if (brix_xrdrc_default_ms("max_stall_ms", &xv)) { return xv; } }
60 return XRDC_DEFAULT_MAX_STALL_MS;
61}

69int
70brix_connect_resilient(brix_conn *c, const brix_url *u, const brix_opts *o,
71 brix_status *st)
72{
73 int window = opts_window_ms(o);
74 int rc = brix_connect(c, u, o, st);
75 uint64_t deadline;
76 unsigned attempt = 0;
77
78 if (rc == 0 || window <= 0) {
79 return rc;
80 }
81 deadline = brix_mono_ns() + (uint64_t) window * 1000000ULL;
82 for (;;) {
83 /* A connection that is actively REFUSED (nothing listening) is definitive,
84 * not a transient stall — fail fast rather than burn the whole window
85 * (matches copy.c's resilient_setup). A timeout/sever mid-handshake on a
86 * lossy link stays retryable so the multi-RTT GSI login can ride it out. */
87 if (st->kxr == XRDC_ESOCK && st->sys_errno == ECONNREFUSED) {
88 return rc;
89 }
90 if (!brix_status_retryable(st) || brix_mono_ns() >= deadline) {
91 return rc;
92 }
93 brix_backoff_sleep_fast(attempt++);
94 rc = brix_connect(c, u, o, st);
95 if (rc == 0) {
96 return 0;
97 }
98 }
99}

110int
111brix_with_resilience(brix_conn *c, int max_stall_ms, brix_op_class cls,
112 int benign_errno, brix_op_fn op, void *arg, brix_status *st)
113{
114 int rc = op(c, arg, st);
115 if (rc == 0) {
116 return 0;
117 }
118 /* Disabled, unsafe, or a definitive (non-transport) failure: surface as-is. */
119 if (max_stall_ms <= 0 || cls == XRDC_OP_UNSAFE || !brix_status_retryable(st)) {
120 return rc;
121 }
122
123 uint64_t deadline = brix_mono_ns() + (uint64_t) max_stall_ms * 1000000ULL;
124 unsigned attempt = 0;
125 for (;;) {
126 if (brix_mono_ns() >= deadline) {
127 return rc; /* rc and st hold the last failure */
128 }
129 brix_backoff_sleep_fast(attempt++);
130 if (brix_reconnect_home(c, st) != 0) {
131 if (!brix_status_retryable(st) || brix_mono_ns() >= deadline) {
132 return -1;
133 }
134 continue; /* link still flaky — keep trying to reconnect */
135 }
136
137 rc = op(c, arg, st);
138 if (rc == 0) {
139 return 0;
140 }
141 if (cls == XRDC_OP_MUTATION_NORMALIZE) {
142 /* Re-issue exactly ONCE for a mutation. If the first attempt had
143 * already been applied server-side (its reply was lost to the sever),
144 * the re-issue reports the benign "already in desired state" code —
145 * normalize that to success rather than a spurious EEXIST/ENOENT. */
146 if (benign_errno != 0 && brix_kxr_to_errno(st) == benign_errno) {
147 brix_status_clear(st);
148 return 0;
149 }
150 return rc;
151 }
152 if (!brix_status_retryable(st)) {
153 return rc; /* READONLY/IDEMPOTENT hit a hard error */
154 }
155 /* READONLY/IDEMPOTENT: loop until success or the deadline. */
156 }
157}

159/* Resilient single-frame roundtrip. The high-level metadata/fs ops (ops_fs.c
160 * fs_simple/fs_text, ops_meta.c stat) route through this, so every tool that
161 * calls brix_stat/mkdir/rm/query/... inherits reconnect+retry transparently.
162 * Re-sending the same hdr24/payload is safe: brix_send stamps a fresh streamid. */
163int
164brix_roundtrip_resilient(brix_conn *c, void *hdr24, const void *payload,
165 uint32_t plen, brix_op_class cls, int benign_errno,
166 uint16_t *status, uint8_t **body, uint32_t *blen,
167 brix_status *st)
168{
169 brix_payload pl = { payload, plen };
170 brix_resp_out out = { status, body, blen };
171 int window = brix_resilient_window_ms(c);
172 int rc = brix_roundtrip(c, hdr24, &pl, &out, st);
173 if (rc == 0) {
174 return 0;
175 }
176 if (window <= 0 || cls == XRDC_OP_UNSAFE || !brix_status_retryable(st)) {
177 return rc;
178 }
179
180 uint64_t deadline = brix_mono_ns() + (uint64_t) window * 1000000ULL;
181 unsigned attempt = 0;
182 for (;;) {
183 if (brix_mono_ns() >= deadline) {
184 return rc;
185 }
186 brix_backoff_sleep_fast(attempt++);
187 if (brix_reconnect_home(c, st) != 0) {
188 if (!brix_status_retryable(st) || brix_mono_ns() >= deadline) {
189 return -1;
190 }
191 continue;
192 }
193 rc = brix_roundtrip(c, hdr24, &pl, &out, st);
194 if (rc == 0) {
195 return 0;
196 }
197 if (cls == XRDC_OP_MUTATION_NORMALIZE) {
198 /* Re-issued ONCE after reconnect: if the original had already been
199 * applied (its reply lost to the sever), the benign "already in the
200 * desired state" code means success, not a spurious EEXIST/ENOENT. */
201 if (benign_errno != 0 && brix_kxr_to_errno(st) == benign_errno) {
202 brix_status_clear(st);
203 return 0;
204 }
205 return rc;
206 }
207 if (!brix_status_retryable(st)) {
208 return rc;
209 }
210 /* READONLY/IDEMPOTENT: keep retrying until success or the deadline. */
211 }
212}

214/* resilient file */
215/* (Re)open the handle on the CURRENT connection. Writes reopen in place via
216 * kXR_open_updt (no truncate) so already-written bytes survive a mid-write
217 * sever; reads use the opaque variant when an opaque suffix is present. */
218static int
219rfile_do_open(brix_rfile *rf, brix_status *st)
220{
221 if (rf->writable) {
222 return brix_file_open_update(rf->c, rf->path, rf->posc, &rf->f, st);
223 }
224 if (rf->opaque[0] != '\0') {
225 return brix_file_open_opaque(rf->c, rf->path, rf->opaque, 0, 0, 0,
226 &rf->f, st);
227 }
228 return brix_file_open_read(rf->c, rf->path, &rf->f, st);
229}

231/* Reconnect to home + reopen the handle, replacing the dead one. 0 / -1. */
232static int
233rfile_reopen(brix_rfile *rf, brix_status *st)
234{
235 if (brix_reconnect_home(rf->c, st) != 0) {
236 return -1;
237 }
238 return rfile_do_open(rf, st);

260int
261brix_rfile_open_read(brix_conn *c, const char *path, const char *opaque,
262 int pgrw, int max_stall_ms, brix_rfile *rf, brix_status *st)
263{
264 rfile_init(rf, c, path, pgrw, max_stall_ms);
265 rf->writable = 0;
266 if (opaque != NULL) {
267 snprintf(rf->opaque, sizeof(rf->opaque), "%s", opaque);
268 }
269
270 uint64_t deadline = brix_mono_ns() + (uint64_t) rf->max_stall_ms * 1000000ULL;
271 unsigned attempt = 0;
272 for (;;) {
273 if (rfile_do_open(rf, st) == 0) {
274 return 0;
275 }
276 if (!brix_status_retryable(st) || rfile_canceled(rf)
277 || rf->max_stall_ms <= 0 || brix_mono_ns() >= deadline) {
278 return -1;
279 }
280 brix_backoff_sleep_fast(attempt++);
281 (void) brix_reconnect_home(rf->c, st); /* best-effort; loop re-opens */
282 }

285int
286brix_rfile_open_write(brix_conn *c, const char *path, int force, int posc,
287 int pgrw, int max_stall_ms, brix_rfile *rf, brix_status *st)
288{
289 rfile_init(rf, c, path, pgrw, max_stall_ms);
290 rf->writable = 1;
291 rf->posc = posc;
292
293 /* First open creates/truncates per `force`; nothing is written yet, so
294 * retrying a create after a sever is safe. Subsequent reopens (rfile_reopen)
295 * switch to in-place update so resumed writes never re-truncate. */
296 uint64_t deadline = brix_mono_ns() + (uint64_t) rf->max_stall_ms * 1000000ULL;
297 unsigned attempt = 0;
298 for (;;) {
299 if (brix_file_open_write(rf->c, rf->path, force, posc, &rf->f, st) == 0) {
300 return 0;
301 }
302 if (!brix_status_retryable(st) || rfile_canceled(rf)
303 || rf->max_stall_ms <= 0 || brix_mono_ns() >= deadline) {
304 return -1;
305 }
306 brix_backoff_sleep_fast(attempt++);
307 (void) brix_reconnect_home(rf->c, st);
308 }

311ssize_t
312brix_rfile_pread(brix_rfile *rf, int64_t off, void *buf, size_t len,
313 brix_status *st)
314{
315 uint64_t deadline = brix_mono_ns() + (uint64_t) rf->max_stall_ms * 1000000ULL;
316 unsigned attempt = 0;
317 for (;;) {
318 if (rfile_canceled(rf)) {
319 brix_status_set(st, XRDC_EUSAGE, ECANCELED, "transfer canceled");
320 return -1;
321 }
322 size_t want = (len < rf->cur_chunk) ? len : rf->cur_chunk;
323 ssize_t n = rf->pgrw ? brix_file_pgread(rf->c, &rf->f, off, buf, want, st)
324 : brix_file_read(rf->c, &rf->f, off, buf, want, st);
325 if (n >= 0) {
326 return n;
327 }
328 if (!brix_status_retryable(st) || rf->max_stall_ms <= 0
329 || rfile_canceled(rf) || brix_mono_ns() >= deadline) {
330 return -1;
331 }
332 /* Transport fault: shrink the request, reconnect+reopen, retry at the
333 * same absolute offset (idempotent). */
334 if (rf->cur_chunk > XRDC_RFILE_FLOOR) {
335 rf->cur_chunk /= 2;
336 }
337 brix_backoff_sleep_fast(attempt++);
338 (void) rfile_reopen(rf, st);
339 }
340}

342int
343brix_rfile_pwrite(brix_rfile *rf, int64_t off, const void *buf, size_t len,
344 brix_status *st)
345{
346 uint64_t deadline = brix_mono_ns() + (uint64_t) rf->max_stall_ms * 1000000ULL;
347 unsigned attempt = 0;
348 for (;;) {
349 if (rfile_canceled(rf)) {
350 brix_status_set(st, XRDC_EUSAGE, ECANCELED, "transfer canceled");
351 return -1;
352 }
353 int rc = rf->pgrw ? brix_file_pgwrite(rf->c, &rf->f, off, buf, len, st)
354 : brix_file_write(rf->c, &rf->f, off, buf, len, st);
355 if (rc == 0) {
356 return 0;
357 }
358 if (!brix_status_retryable(st) || rf->max_stall_ms <= 0
359 || rfile_canceled(rf) || brix_mono_ns() >= deadline) {
360 return -1;
361 }
362 /* Re-issuing the same bytes at the same offset onto an in-place reopened
363 * handle is idempotent (overwrites identical content). */
364 brix_backoff_sleep_fast(attempt++);
365 (void) rfile_reopen(rf, st);
366 }
