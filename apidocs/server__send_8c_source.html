68/*
69 * brix_cms_srv_send_load — kYR_load reply to a node's kYR_usage query.
70 *
71 * WHAT: Sends a CMS_RR_LOAD frame echoing the request streamid, carrying the
72 * 6-byte load vector (cpu,net,xeq,mem,pag,dsk) and the aggregate free
73 * space in MB. WHY: Stock cmsd answers do_Usage with its current load
74 * report so a querying node/peer can weigh this manager; phase-89 W1
75 * brings the manager side to parity. HOW: Byte-exact with the node-side
76 * heartbeat in send.c — theLoad is a bare [2-byte len][6 bytes] Pup
77 * char-blob (NO scalar tag), dskFree a tagged int; total dlen = 13.
78 */
79ngx_int_t
80brix_cms_srv_send_load(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
81 const u_char load6[6], uint32_t free_mb)
82{
83 u_char payload[16];
84 u_char *cursor;
85
86 cursor = payload;
87 ngx_brix_cms_put16(cursor, 6); /* blob length: 6 load bytes */
88 cursor += 2;
89 ngx_memcpy(cursor, load6, 6);
90 cursor += 6;
91 cursor = ngx_brix_cms_put_int(cursor, free_mb);
92
93 return brix_cms_srv_send_frame(ctx, streamid, CMS_RR_LOAD, 0,
94 payload, (size_t) (cursor - payload));
95}
