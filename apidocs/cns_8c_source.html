15static void
16put_u64(uint8_t *p, uint64_t v)
17{
18 p[0] = (uint8_t) (v >> 56); p[1] = (uint8_t) (v >> 48);
19 p[2] = (uint8_t) (v >> 40); p[3] = (uint8_t) (v >> 32);
20 p[4] = (uint8_t) (v >> 24); p[5] = (uint8_t) (v >> 16);
21 p[6] = (uint8_t) (v >> 8); p[7] = (uint8_t) v;
22}
23
24static uint64_t
25get_u64(const uint8_t *p)
26{
27 return ((uint64_t) p[0] << 56) | ((uint64_t) p[1] << 48)
28 | ((uint64_t) p[2] << 40) | ((uint64_t) p[3] << 32)
29 | ((uint64_t) p[4] << 24) | ((uint64_t) p[5] << 16)
30 | ((uint64_t) p[6] << 8) | (uint64_t) p[7];
31}

33size_t
34brix_cns_event_encode(uint8_t op, const char *path, uint64_t size,
35 uint64_t mtime, uint8_t *buf, size_t bufsz)
36{
37 size_t plen = path ? strlen(path) : 0;
38
39 if (plen == 0 || plen > BRIX_CNS_PATH_MAX
40 || bufsz < BRIX_CNS_HDR_LEN + plen)
41 {
42 return 0;
43 }
44 buf[0] = op;
45 buf[1] = buf[2] = buf[3] = 0;
46 put_u64(buf + 4, size);
47 put_u64(buf + 12, mtime);
48 buf[20] = (uint8_t) (plen >> 8);
49 buf[21] = (uint8_t) plen;
50 memcpy(buf + BRIX_CNS_HDR_LEN, path, plen);
51 return BRIX_CNS_HDR_LEN + plen;
52}

54ngx_int_t
55brix_cns_event_decode(const uint8_t *buf, size_t len, uint8_t *op,
56 uint64_t *size, uint64_t *mtime, char *path,
57 size_t pathsz)
58{
59 size_t plen;
60
61 if (buf == NULL || len < BRIX_CNS_HDR_LEN) {
62 return NGX_ERROR;
63 }
64 plen = ((size_t) buf[20] << 8) | buf[21];
65 if (plen == 0 || plen > BRIX_CNS_PATH_MAX
66 || len < BRIX_CNS_HDR_LEN + plen || plen >= pathsz)
67 {
68 return NGX_ERROR;
69 }
70 *op = buf[0];
71 *size = get_u64(buf + 4);
72 *mtime = get_u64(buf + 12);
73 memcpy(path, buf + BRIX_CNS_HDR_LEN, plen);
74 path[plen] = '\0';
75 return NGX_OK;
76}

93static cns_entry_t *
94cns_find(const char *path)
95{
96 ngx_uint_t i;
97 if (s_inv == NULL) {
98 return NULL;
99 }
100 for (i = 0; i < CNS_CAP; i++) {
101 if (s_inv[i].used && strcmp(s_inv[i].path, path) == 0) {
102 return &s_inv[i];
103 }
104 }
105 return NULL;
106}

109cns_free_slot(void)
110{

112 for (i = 0; i < CNS_CAP; i++) {
113 if (!s_inv[i].used) {

120ngx_int_t
121brix_cns_apply(uint8_t op, const char *path, uint64_t size, uint64_t mtime,
122 uint32_t server_id)
123{
124 cns_entry_t *e;
125 size_t plen;
126
127 if (path == NULL) {
128 return NGX_ERROR;
129 }
130 plen = strlen(path);
131 if (plen == 0 || plen > BRIX_CNS_PATH_MAX) {
132 return NGX_ERROR;
133 }
134
135 if (op == BRIX_CNS_DEL || op == BRIX_CNS_RMDIR) {
136 e = cns_find(path);
137 if (e != NULL) {
138 e->used = 0;
139 if (s_count > 0) { s_count--; }
140 }
141 return NGX_OK;
142 }
143
144 if (op != BRIX_CNS_ADD && op != BRIX_CNS_MKDIR) {
145 return NGX_ERROR;
146 }
147
148 if (s_inv == NULL) {
149 s_inv = calloc(CNS_CAP, sizeof(*s_inv));
150 if (s_inv == NULL) {
151 return NGX_ERROR;
152 }
153 }
154
155 e = cns_find(path);
156 if (e == NULL) {
157 e = cns_free_slot();
158 if (e == NULL) {
159 return NGX_ERROR; /* inventory full (v1 fixed cap) */
160 }
161 memcpy(e->path, path, plen);
162 e->path[plen] = '\0';
163 e->used = 1;
164 s_count++;
165 }
166 e->size = size;
167 e->mtime = mtime;
168 e->server_id = server_id;
169 e->is_dir = (op == BRIX_CNS_MKDIR) ? 1 : 0;
170 return NGX_OK;
171}

173ngx_int_t
174brix_cns_stat(const char *path, struct stat *out)
175{
176 cns_entry_t *e;
177
178 if (path == NULL || out == NULL) {
179 return NGX_ERROR;
180 }
181 e = cns_find(path);
182 if (e == NULL) {
183 return NGX_DECLINED;
184 }
185 ngx_memzero(out, sizeof(*out));
186 out->st_size = (off_t) e->size;
187 out->st_mtime = (time_t) e->mtime;
188 out->st_mode = e->is_dir ? (S_IFDIR | 0755) : (S_IFREG | 0644);
189 out->st_nlink = 1;
190 return NGX_OK;
191}
