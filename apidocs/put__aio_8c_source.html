10/*
11 * s3_thread_pool — resolve (and cache) the async-I/O thread pool for this
12 * location.
13 *
14 * WHY: the S3 postconfiguration only resolves common.thread_pool on the
15 * *server-level* loc-conf, but `brix_s3 on` is normally inside a `location {}`
16 * block whose loc-conf never gets that pointer set — so the offload below would
17 * silently never engage. Mirror the WebDAV COPY/MOVE pattern
18 * (src/webdav/copy.c, move.c): resolve lazily at request time via ngx_cycle and
19 * cache the result into the loc-conf for subsequent requests.
20 */
21ngx_thread_pool_t *
22s3_thread_pool(ngx_http_s3_loc_conf_t *cf)
23{
24 static ngx_str_t default_name = ngx_string("default");
25 ngx_str_t *pname;
26 ngx_thread_pool_t *pool;
27
28 if (cf->common.thread_pool != NULL) {
29 return cf->common.thread_pool;
30 }
31 pname = cf->common.thread_pool_name.len > 0
32 ? &cf->common.thread_pool_name : &default_name;
33 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
34 if (pool != NULL) {
35 cf->common.thread_pool = pool; /* cache for subsequent requests */
36 }
37 return pool;
38}

41void
42s3_put_aio_thread(void *data, ngx_log_t *log)
43{
44 s3_put_aio_t *t = data;
45
46 (void) log;
47
48 t->io_errno = 0;
49
50 /*
51 * Phase 31 W2 / phase-46 W1a: stream every body buf straight to the staged
52 * temp fd — no full-body contiguous copy. Memory bufs go via pwrite(2);
53 * spooled bufs via kernel copy_file_range from the nginx temp file. Both are
54 * blocking syscalls, which is exactly why they run here on the thread pool
55 * rather than the event loop.
56 */
57 /* A driver-backed (remote-stage) staged target has no kernel fd — stream the
58 * body through the staged-write primitive; otherwise write straight to the temp
59 * fd (memory pwrite / spooled copy_file_range). Mirrors the WebDAV PUT path. */
60 if (brix_vfs_staged_is_driver(t->staged)
61 ? brix_http_body_write_to_staged(t->r, t->staged) != NGX_OK
62 : brix_http_body_write_to_fd(t->r, brix_vfs_staged_fd(t->staged),
63 t->final_path, NULL) != NGX_OK)
64 {
65 t->io_errno = errno;
66 t->nwritten = -1;
67 return;
68 }
69
70 t->nwritten = (ssize_t) t->len;
71}

74void
75s3_put_aio_done(ngx_event_t *ev)
76{
77 ngx_thread_task_t *task = ev->data;
78 s3_put_aio_t *t = task->ctx;
79 ngx_http_request_t *r = t->r;
80 ngx_log_t *log = r->connection->log;
81
82 if (t->nwritten < 0 || (size_t) t->nwritten < t->len) {
83 brix_log_safe_path(log, NGX_LOG_ERR,
84 (ngx_uint_t) t->io_errno,
85 "s3: async write() failed for: \"%s\"",
86 t->final_path);
87 brix_vfs_staged_abort(t->staged, 1);
88 s3_put_finalize_error(r);
89 return;
90 }
91
92 /* The VFS staged commit closes/promotes the temp itself — no manual close. */
93 if (s3_commit_put(r, log, t->root_canon, t->staged,
94 t->final_path) != NGX_OK)
95 {
96 if (s3_put_commit_conflict(r)) {
97 return;
98 }
99 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
100 "s3: async staged commit to \"%s\" failed",
101 t->final_path);
102 s3_put_finalize_error(r);
103 return;
104 }

106 /* S3 PutObject requires an ETag on the 200 response (the synchronous path
107 * sets it too — keep the offload path's response identical). */
108 {
109 brix_vfs_ctx_t fctx;
110 brix_vfs_stat_t fst;
111 char etag_buf[48];
112
113 brix_vfs_ctx_init(&fctx, r->pool, log, BRIX_PROTO_S3, t->root_canon,
114 NULL, 0 /* allow_write */, 0 /* is_tls */, NULL, t->final_path);
115 if (brix_vfs_probe(&fctx, 1 /* no-follow */, &fst) == NGX_OK) {
116 struct stat final_sb;
117
118 ngx_memzero(&final_sb, sizeof(final_sb));
119 final_sb.st_mtime = fst.mtime;
120 final_sb.st_size = fst.size;
121 s3_etag(&final_sb, etag_buf, sizeof(etag_buf));
122 (void) s3_set_header(r, "ETag", etag_buf);
123 }
124 }
125
126 /* Full-object checksum verify (client-supplied) + echo; failures remove the
127 * object and send the matching 400. */
128 if (s3_put_checksum_failed(r, t->final_path, t->root_canon)) {
129 return;
130 }
131
132 /* x-amz-tagging (best-effort): store the request's tag set on the object. */
133 (void) s3_apply_put_tagging_header(r, t->final_path, t->root_canon);
134 /* x-amz-meta-* (best-effort): store the request's user metadata set. */
135 (void) s3_apply_put_user_metadata(r, t->final_path, t->root_canon);
136
137 brix_dashboard_http_add(r, (ngx_atomic_int_t) t->body_bytes);
138 BRIX_S3_METRIC_ADD(bytes_rx_total, t->body_bytes);
139 if (r->connection && r->connection->sockaddr
140 && r->connection->sockaddr->sa_family == AF_INET6) {
141 BRIX_S3_METRIC_ADD(bytes_rx_ipv6_total, t->body_bytes);
142 } else {
143 BRIX_S3_METRIC_ADD(bytes_rx_ipv4_total, t->body_bytes);
144 }
145 BRIX_S3_METRIC_INC(put_body_total[t->body_mode]);
146 s3_put_finalize_empty_ok(r);
147}
