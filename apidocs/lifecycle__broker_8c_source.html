58/* Bind a 0600 listening AF_UNIX socket at `path`, owned by `wuid`. */
59static int
60imp_make_listen(const char *path, uid_t wuid, ngx_log_t *log)
61{
62 struct sockaddr_un addr;
63 int lfd;
64
65 if (ngx_strlen(path) >= sizeof(addr.sun_path)) {
66 ngx_log_error(NGX_LOG_EMERG, log, 0,
67 "impersonate: socket path too long \"%s\"", path);
68 return -1;
69 }
70 lfd = socket(AF_UNIX, SOCK_STREAM | SOCK_CLOEXEC, 0);
71 if (lfd < 0) {
72 return -1;
73 }
74 unlink(path);
75 ngx_memzero(&addr, sizeof(addr));
76 addr.sun_family = AF_UNIX;
77 ngx_memcpy(addr.sun_path, path, ngx_strlen(path) + 1);
78 if (bind(lfd, (struct sockaddr *) &addr, sizeof(addr)) != 0
79 || listen(lfd, 64) != 0)
80 {
81 BRIX_DIAG_EMERG(log, ngx_errno,
82 "impersonate: cannot bind broker socket \"%s\"",
83 "the directory is not writable by the master, or a stale socket "
84 "is held by another process",
85 "ensure the socket's parent directory exists and is writable; the "
86 "OS reason is appended below",
87 path);
88 close(lfd);
89 return -1;
90 }
91 chmod(path, 0600);
92 if (wuid != (uid_t) -1) {
93 if (chown(path, wuid, (gid_t) -1) != 0) {
94 ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
95 "impersonate: chown socket to uid %d failed",
96 (int) wuid);
97 }
98 }
99 return lfd;
100}

109/* Terminate a previously-spawned broker (across reload) if still alive. */
110static void
111imp_kill_stale_broker(ngx_log_t *log)
112{
113 char pf[256];
114 FILE *fp;
115 long pid = 0;
116
117 imp_pidfile(pf, sizeof(pf));
118 fp = fopen(pf, "re");
119 if (fp == NULL) {
120 return;
121 }
122 if (fscanf(fp, "%ld", &pid) == 1 && pid > 1 && kill((pid_t) pid, 0) == 0) {
123 kill((pid_t) pid, SIGTERM);
124 ngx_log_error(NGX_LOG_NOTICE, log, 0,
125 "impersonate: terminated stale broker pid %ld", pid);
126 }
127 (void) fclose(fp); /* phase74-fp: read-only pidfile stream, pid already parsed and acted on */
128 unlink(pf);
129}

131/* The broker child body (post double-fork): record pid, run the serve loop. */
132static void
133imp_broker_child(int lfd, int rootfd, ngx_log_t *log)
134{
135 char pf[256];
136 FILE *fp;
137
138 setsid();
139 imp_pidfile(pf, sizeof(pf));
140 fp = fopen(pf, "we");
141 if (fp != NULL) {
142 fprintf(fp, "%ld\n", (long) getpid());
143 (void) fclose(fp); /* phase74-fp: best-effort pidfile write by design — fopen failure is tolerated the same way */
144 }
145 /* broker_run drops caps to {SETUID,SETGID} then serves until killed. */
146 _exit(brix_imp_broker_run(lfd, rootfd, NULL, log) == 0 ? 0 : 1);
147}

149/*
150 * WHAT: Resolve the nginx worker uid the broker will be gated to.
151 * WHY: The broker socket is chowned to this uid and the broker refuses any
152 * caller uid but this one — defence in depth atop path confinement. A
153 * missing/unset `user` directive means "no gate" ((uid_t)-1).
154 * HOW: Read ngx_core_module's parsed conf; treat NGX_CONF_UNSET_UINT as none.
155 */
156static uid_t
157imp_worker_uid(ngx_cycle_t *cycle)
158{
159 ngx_core_conf_t *ccf;
160
161 ccf = (ngx_core_conf_t *) ngx_get_conf(cycle->conf_ctx, ngx_core_module);
162 if (ccf && ccf->user != (uid_t) NGX_CONF_UNSET_UINT) {
163 return ccf->user;
164 }
165 /* No `user` directive: nginx drops workers to its built-in default account
166 * ("nobody"). Resolve it so the broker's SO_PEERCRED gate stays ACTIVE — an
167 * unset `user` must NOT silently disable the gate (brix_imp_broker_allow_uid
168 * == 0 means "no gate", which would let any local process on the 0600 socket
169 * through). */
170 {
171 struct passwd *pw = getpwnam("nobody");
172 if (pw != NULL) {
173 return pw->pw_uid;
174 }
175 }
176 return (uid_t) -1;
177}

179/*
180 * WHAT: Resolve the optional non-root broker service account into the broker
181 * globals (brix_imp_broker_user_uid/gid).
182 * WHY: When configured, the broker drops its real uid/gid to this account
183 * (keeping only CAP_SETUID/CAP_SETGID) so nothing runs as root after
184 * startup. The account must be real, non-root, and differ from the
185 * worker uid so a compromise cannot escalate. Globals must be set BEFORE
186 * the fork so the broker inherits them.
187 * HOW: Default the globals to "none"; if a name is configured, look it up and
188 * validate it, returning NGX_ERROR (with a diagnostic) on any problem.
189 */
190static ngx_int_t
191imp_resolve_broker_user(ngx_cycle_t *cycle, uid_t wuid)
192{
193 char nm[256];
194 struct passwd *pw;
195
196 brix_imp_broker_user_uid = (uid_t) -1;
197 brix_imp_broker_user_gid = (gid_t) -1;
198 if (imp_settings.broker_user.len == 0) {
199 return NGX_OK;
200 }
201
202 ngx_snprintf((u_char *) nm, sizeof(nm), "%V%Z", &imp_settings.broker_user);
203 pw = getpwnam(nm);
204 if (pw == NULL) {
205 BRIX_DIAG_EMERG(cycle->log, 0,
206 "impersonate: broker user \"%s\" does not exist",
207 "brix_impersonation_broker_user names a local account that "
208 "is not present in /etc/passwd (or NSS)",
209 "create the dedicated service account first, or correct the "
210 "name in the directive",
211 nm);
212 return NGX_ERROR;
213 }
214 if (pw->pw_uid == 0 || (wuid != (uid_t) -1 && pw->pw_uid == wuid)) {
215 BRIX_DIAG_EMERG(cycle->log, 0,
216 "impersonate: broker user \"%s\" is not a safe choice",
217 "the broker account must NOT be root and must differ from the "
218 "nginx worker user, so a compromise cannot escalate",
219 "point brix_impersonation_broker_user at a dedicated, "
220 "unprivileged account used for nothing else",
221 nm);
222 return NGX_ERROR;
223 }
224 brix_imp_broker_user_uid = pw->pw_uid;
225 brix_imp_broker_user_gid = pw->pw_gid;
226 return NGX_OK;
227}

252/*
253 * WHAT: Bring up the broker's two file descriptors — the listening AF_UNIX
254 * socket (*lfd) and the O_PATH export-root handle (*rootfd).
255 * WHY: Both must exist before the fork; the broker inherits them and confines
256 * all impersonated I/O beneath the export root.
257 * HOW: Bind the 0600 socket (chowned to the worker uid), then open the root;
258 * on root-open failure close the socket so nothing leaks. fds->lfd /
259 * fds->rootfd are written only on full success.
260 */
261typedef struct {
262 int lfd;
263 int rootfd;
264} imp_broker_fds_t;

266static ngx_int_t
267imp_open_broker_fds(const char *sockbuf, const char *rootbuf, uid_t wuid,
268 ngx_log_t *log, imp_broker_fds_t *fds)
269{
270 int l, r;
271
272 l = imp_make_listen(sockbuf, wuid, log);
273 if (l < 0) {
274 return NGX_ERROR;
275 }
276 r = open(rootbuf, O_PATH | O_DIRECTORY | O_CLOEXEC);
277 if (r < 0) {
278 BRIX_DIAG_EMERG(log, ngx_errno,
279 "impersonate: cannot open export root \"%s\"",
280 "the export root directory is missing or unreadable by the master",
281 "create the directory and ensure the master can open it; the OS "
282 "reason is appended below",
283 rootbuf);
284 close(l);
285 return NGX_ERROR;
286 }
287 fds->lfd = l;
288 fds->rootfd = r;
289 return NGX_OK;
290}

292/*
293 * WHAT: Double-fork the privileged broker off the master, closing the passed
294 * fds in the master afterwards.
295 * WHY: Double-fork reparents the broker to init so nginx never reaps it
296 * (SHM-safe; the FRM pattern). SIGCHLD is blocked around the fork and
297 * the intermediate reaped so the master's signal handling is undisturbed.
298 * HOW: Block SIGCHLD, fork the intermediate (which forks the broker child and
299 * exits), close the master's fd copies, wait the intermediate, restore
300 * the signal mask. The master's fds are closed on the fork-failure path
301 * too. The broker child never returns from imp_broker_child.
302 */
303static ngx_int_t
304imp_spawn_broker(int lfd, int rootfd, ngx_log_t *log)
305{
306 sigset_t block, prev;
307 pid_t inter;
308
309 sigemptyset(&block);
310 sigaddset(&block, SIGCHLD);
311 sigprocmask(SIG_BLOCK, &block, &prev);
312
313 inter = fork();
314 if (inter < 0) {
315 sigprocmask(SIG_SETMASK, &prev, NULL);
316 close(lfd);
317 close(rootfd);
318 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno,
319 "impersonate: broker fork failed");
320 return NGX_ERROR;
321 }
322 if (inter == 0) {
323 if (fork() == 0) {
324 imp_broker_child(lfd, rootfd, log); /* never returns */
325 }
326 _exit(0); /* intermediate -> broker reparents */
327 }
328
329 close(lfd);
330 close(rootfd);
331 while (waitpid(inter, NULL, 0) < 0 && errno == EINTR) { }
332 sigprocmask(SIG_SETMASK, &prev, NULL);
333 return NGX_OK;
334}

336/*
337 * WHAT: Master-side guard — publish the config fingerprint every real load and
338 * decide whether the privileged-broker path should run at all.
339 * WHY: init_module is the module's single per-cycle master hook, so it owns
340 * the version publish (skipped under `nginx -t`, which must not bump the
341 * live generation counter). The broker is only built in `map` mode, and
342 * only when the master is genuinely root.
343 * HOW: Publish unless testing; then return NGX_DECLINED when the broker path
344 * must be skipped (non-map or `nginx -t`), NGX_OK to proceed, or
345 * NGX_ERROR if `map` was requested without a root master.
346 */
347static ngx_int_t
348imp_init_module_gate(ngx_cycle_t *cycle)
349{
350 if (!ngx_test_config) {
351 brix_config_version_publish(cycle);
352 }
353 if (imp_settings.mode != BRIX_IMP_MAP || ngx_test_config) {
354 return NGX_DECLINED;
355 }
356 if (geteuid() != 0) {
357 BRIX_DIAG_EMERG(cycle->log, 0,
358 "impersonate: map mode requires a root master process",
359 "brix_impersonation is set to 'map' but nginx was not started "
360 "as root, so it cannot set up the privileged uid-mapping broker",
361 "start nginx as root (workers still drop to the configured user), "
362 "or change brix_impersonation away from 'map'");
363 return NGX_ERROR;
364 }
365 return NGX_OK;
366}

368ngx_int_t
369brix_imp_init_module(ngx_cycle_t *cycle)
370{
371 char sockbuf[256], rootbuf[1024];
372 uid_t wuid;
373 imp_broker_fds_t fds = { -1, -1 };
374 ngx_int_t gate;
375
376 gate = imp_init_module_gate(cycle);
377 if (gate == NGX_DECLINED) {
378 return NGX_OK;
379 }
380 if (gate != NGX_OK) {
381 return NGX_ERROR;
382 }
383
384 ngx_snprintf((u_char *) sockbuf, sizeof(sockbuf), "%V%Z",
385 &imp_settings.socket);
386 ngx_snprintf((u_char *) rootbuf, sizeof(rootbuf), "%V%Z",
387 &imp_settings.export_root);
388
389 wuid = imp_worker_uid(cycle);
390
391 if (imp_resolve_broker_user(cycle, wuid) != NGX_OK) {
392 return NGX_ERROR;
393 }
394
395 imp_kill_stale_broker(cycle->log);
396
397 if (imp_load_idmap(cycle, wuid) != NGX_OK) {
398 return NGX_ERROR;
399 }
400
401 if (imp_open_broker_fds(sockbuf, rootbuf, wuid, cycle->log, &fds) != NGX_OK) {
402 return NGX_ERROR;
403 }
404
405 /* Gate the broker to the worker uid (defence in depth atop confinement). */
406 brix_imp_broker_allow_uid = (wuid != (uid_t) -1) ? wuid : 0;
407
408 if (imp_spawn_broker(fds.lfd, fds.rootfd, cycle->log) != NGX_OK) {
409 return NGX_ERROR;
410 }
411
412 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
413 "impersonate: started privileged broker on \"%s\" "
414 "(export root \"%s\", worker uid %d)",
415 sockbuf, rootbuf, (int) wuid);
416 return NGX_OK;
417}
