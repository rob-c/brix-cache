5 * brix_process_handshake
6 *
7 * Validates the 20-byte client handshake and sends the server reply.
8 *
9 * In XRootD v5 the client sends handshake + kXR_protocol together in a
10 * single 44-byte TCP segment. We send a proper ServerResponseHdr
11 * (streamid={0,0}, status=ok, dlen=8) followed by 8 bytes of protover+msgval.
12 */

32 if (ntohl(hs->fourth) != 4 || ntohl(hs->fifth) != ROOTD_PQ) {
