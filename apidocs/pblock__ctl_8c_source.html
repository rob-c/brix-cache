67/* Parse "caps=" value: a comma list of +name / -name (bare name ⇒ +name). */
68static int
69opts_parse_caps(const char *v, size_t vlen, pblock_opts_t *out)
70{
71 const char *p = v;
72 const char *end = v + vlen;
73
74 out->has_caps = 1;
75 while (p < end) {
76 const char *comma = memchr(p, ',', (size_t) (end - p));
77 const char *tok_end = comma ? comma : end;
78 int add = 1;
79 uint32_t bit;
80
81 if (p < tok_end && (*p == '+' || *p == '-')) {
82 add = (*p == '+');
83 p++;
84 }
85 bit = pblock_cap_bit(p, (size_t) (tok_end - p));
86 if (bit == 0) {
87 errno = EINVAL;
88 return -1;
89 }
90 if (add) {
91 out->caps_add |= bit;
92 } else {
93 out->caps_drop |= bit;
94 }
95 p = comma ? comma + 1 : end;
96 }
97 return 0;
98}

100/* pblock_parse_size — "10G"/"512m"/"100k"/"1234" → bytes/count (suffix k/m/g/t,
101 * case-insensitive). 0 on empty/garbage (⇒ the quota stays off). Non-static:
102 * pblock_quota.c reuses it for `quota.uid.<n>` ctl values. */
103int64_t
104pblock_parse_size(const char *val, size_t vlen)
105{
106 int64_t n = 0;
107 size_t i = 0;
108
109 while (i < vlen && val[i] >= '0' && val[i] <= '9') {
110 n = n * 10 + (val[i] - '0');
111 i++;
112 }
113 if (i == 0) {
114 return 0;
115 }
116 if (i < vlen) {
117 switch (val[i] | 0x20) {
118 case 'k': n *= 1024LL; break;
119 case 'm': n *= 1024LL * 1024; break;
120 case 'g': n *= 1024LL * 1024 * 1024; break;
121 case 't': n *= 1024LL * 1024 * 1024 * 1024; break;
122 default: break;
123 }
124 }
125 return n;
126}

128/* A boolean opts token is true when present with no value, or with a value whose
129 * first byte is not a falsey sentinel (0 / n / f — i.e. 0, no, false). The
130 * `vlen == 0` short-circuit keeps val[0] out of bounds when there is no value. */
131static int
132opts_truthy(const char *val, size_t vlen)
133{
134 return vlen == 0 || (val[0] != '0' && val[0] != 'n' && val[0] != 'f');
135}

137/* opts_apply_flag — the boolean feature toggles, table-driven: name → the int
138 * field it sets. Returns 1 iff KEY named one of them (and it was applied). */
139static int
140opts_apply_flag(pblock_opts_t *out, const char *key, size_t klen,
141 const char *val, size_t vlen)
142{
143 static const struct { const char *n; size_t off; } flags[] = {
144 { "lab", offsetof(pblock_opts_t, lab) }, /* master gate */
145 { "mem", offsetof(pblock_opts_t, mem) }, /* F16 */
146 { "audit", offsetof(pblock_opts_t, audit) }, /* F17 */
147 { "csi", offsetof(pblock_opts_t, csi) }, /* F3 */
148 { "nearline", offsetof(pblock_opts_t, nearline) }, /* F4 */
149 { "locks", offsetof(pblock_opts_t, locks) }, /* F15 */
150 { "dedup", offsetof(pblock_opts_t, dedup) }, /* F10 */
151 { "snap", offsetof(pblock_opts_t, snapshots) }, /* F6 */
152 { "trash", offsetof(pblock_opts_t, trash) }, /* F11 */
153 };
154 size_t i;
155
156 for (i = 0; i < sizeof(flags) / sizeof(flags[0]); i++) {
157 if (strlen(flags[i].n) == klen && memcmp(flags[i].n, key, klen) == 0) {
158 *(int *) ((char *) out + flags[i].off) = opts_truthy(val, vlen);
159 return 1;
160 }
161 }
162 return 0;
163}

165/* opts_apply_xform — F12/F13 transform spec ("crypt:<keyfile>"/"zstd"); copied
166 * into the fixed buffer, EINVAL if it would not fit. */
167static int
168opts_apply_xform(pblock_opts_t *out, const char *val, size_t vlen)
169{
170 if (vlen >= sizeof(out->xform)) {
171 errno = EINVAL;
172 return -1;
173 }
174 memcpy(out->xform, val, vlen);
175 out->xform[vlen] = '\0';
176 out->xform_len = vlen;
177 return 0;
178}

180/* opts_apply_scalar — the non-boolean keys (sizes, counts, caps, xform).
181 * Unknown keys are ignored for later-wave forward-compat. */
182static int
183opts_apply_scalar(pblock_opts_t *out, const char *key, size_t klen,
184 const char *val, size_t vlen)
185{
186 if (klen == 5 && memcmp(key, "quota", 5) == 0) {
187 out->quota_bytes = pblock_parse_size(val, vlen); /* F5 */
188 } else if (klen == 12 && memcmp(key, "quota_inodes", 12) == 0) {
189 out->quota_inodes = pblock_parse_size(val, vlen); /* F5 */
190 } else if (klen == 8 && memcmp(key, "versions", 8) == 0) {
191 out->versions = (int) pblock_parse_size(val, vlen); /* F11 */
192 } else if (klen == 9 && memcmp(key, "trash_ttl", 9) == 0) {
193 out->trash_ttl = pblock_parse_size(val, vlen); /* F11 */
194 } else if (klen == 4 && memcmp(key, "caps", 4) == 0) {
195 return opts_parse_caps(val, vlen, out); /* F2 */
196 } else if (klen == 5 && memcmp(key, "xform", 5) == 0) {
197 return opts_apply_xform(out, val, vlen); /* F12/F13 */
198 }
199 return 0;
200}

202int
203pblock_opts_parse(const char *query, pblock_opts_t *out)
204{
205 const char *p;
206
207 memset(out, 0, sizeof(*out));
208 if (query == NULL) {
209 return 0;
210 }
211 for (p = query; *p != '\0'; ) {
212 const char *amp = strchr(p, '&');
213 const char *kv_end = amp ? amp : p + strlen(p);
214 const char *eq = memchr(p, '=', (size_t) (kv_end - p));
215 const char *key = p;
216 size_t klen = eq ? (size_t) (eq - p) : (size_t) (kv_end - p);
217 const char *val = eq ? eq + 1 : kv_end;
218 size_t vlen = eq ? (size_t) (kv_end - (eq + 1)) : 0;
219
220 if (!opts_apply_flag(out, key, klen, val, vlen)
221 && opts_apply_scalar(out, key, klen, val, vlen) != 0)
222 {
223 return -1;
224 }
225 p = amp ? amp + 1 : kv_end;
226 }
227 return 0;
228}

238int
239pblock_opts_load_sidecar(const char *root, pblock_opts_t *out)
240{
241 char path[PATH_MAX];
242 char line[1024];
243 FILE *f;
244 char *nl;
245
246 memset(out, 0, sizeof(*out));
247 opts_sidecar_path(root, path, sizeof(path));
248 f = fopen(path, "re");
249 if (f == NULL) {
250 return (errno == ENOENT) ? 0 : -1; /* no sidecar ⇒ default (lab off) */
251 }
252 if (fgets(line, sizeof(line), f) == NULL) {
253 fclose(f);
254 return 0;
255 }
256 fclose(f);
257 nl = strchr(line, '\n');
258 if (nl != NULL) {
259 *nl = '\0';
260 }
261 return pblock_opts_parse(line, out);
262}

276int64_t
277pblock_ctl_epoch(pblock_catalog *cat)
278{
279 sqlite3_stmt *st;
280 int64_t epoch = 0;
281 int rc;
282
283 st = cat_prepare(cat, "SELECT MAX(epoch) FROM ctl;");
284 if (st == NULL) {
285 return -1;
286 }
287 rc = sqlite3_step(st);
288 if (rc == SQLITE_ROW && sqlite3_column_type(st, 0) != SQLITE_NULL) {
289 epoch = sqlite3_column_int64(st, 0);
290 }
291 sqlite3_finalize(st);
292 if (rc != SQLITE_ROW && rc != SQLITE_DONE) {
293 errno = EIO;
294 return -1;
295 }
296 return epoch;
297}

299int
300pblock_ctl_get(pblock_catalog *cat, const char *key, char *buf, size_t buflen)
301{
302 sqlite3_stmt *st;
303 int rc, found = 0;
304
305 if (buflen > 0) {
306 buf[0] = '\0';
307 }
308 st = cat_prepare(cat, "SELECT value FROM ctl WHERE key = ?1;");
309 if (st == NULL) {
310 return -1;
311 }
312 sqlite3_bind_text(st, 1, key, -1, SQLITE_STATIC);
313 rc = sqlite3_step(st);
314 if (rc == SQLITE_ROW) {
315 const unsigned char *v = sqlite3_column_text(st, 0);
316
317 if (v != NULL && buflen > 0) {
318 snprintf(buf, buflen, "%s", (const char *) v);
319 }
320 found = 1;
321 }
322 sqlite3_finalize(st);
323 if (rc != SQLITE_ROW && rc != SQLITE_DONE) {
324 errno = EIO;
325 return -1;
326 }
327 return found;
328}

361void
362pblock_audit_log(pblock_catalog *cat, const char *op, const char *path,
363 const char *aux, uint32_t uid, uint32_t gid, int result, int err)
364{
365 sqlite3_stmt *st;
366 int saved = errno; /* audit must not perturb the caller's errno */
367
368 st = cat_prepare(cat,
369 "INSERT INTO oplog(ts, op, path, aux, uid, gid, result, errno)"
370 " VALUES(?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8);");
371 if (st != NULL) {
372 sqlite3_bind_int64(st, 1, (sqlite3_int64) time(NULL));
373 sqlite3_bind_text(st, 2, op, -1, SQLITE_STATIC);
374 sqlite3_bind_text(st, 3, path ? path : "", -1, SQLITE_STATIC);
375 sqlite3_bind_text(st, 4, aux ? aux : "", -1, SQLITE_STATIC);
376 sqlite3_bind_int64(st, 5, (sqlite3_int64) uid);
377 sqlite3_bind_int64(st, 6, (sqlite3_int64) gid);
378 sqlite3_bind_int(st, 7, result);
379 sqlite3_bind_int(st, 8, err);
380 (void) sqlite3_step(st); /* best-effort: swallow any error */
381 sqlite3_finalize(st);
382 }
383 errno = saved;
384}
