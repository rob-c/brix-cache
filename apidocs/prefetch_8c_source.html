8/* Issue a POSIX_FADV_WILLNEED read-ahead hint for [offset, offset+length) on
9 * fd. Best-effort; a no-op on bad args or if fadvise is unavailable. */
10void
11brix_prefetch_fd_range(ngx_log_t *log, int fd, off_t offset, size_t length)
12{
13#if defined(POSIX_FADV_WILLNEED)
14 int rc;
15 if (fd < 0 || offset < 0 || length < BRIX_READ_PREFETCH_MIN) {
16 return;
17 }
18 rc = posix_fadvise(fd, offset, (off_t) length, POSIX_FADV_WILLNEED);
19 if (rc != 0) {
20 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, log, rc,
21 "brix: POSIX_FADV_WILLNEED ignored: %s",
22 strerror(rc));
23 }
24#else
25 (void) log;
26 (void) fd;
27 (void) offset;
28 (void) length;
29#endif
30}

32/* Sequential-aware prefetch for a read handle: detect sequential access from the
33 * handle's read_last_end / read-ahead window and issue a windowed WILLNEED hint
34 * ahead of the current read. No-op for random access. */
35void
36brix_prefetch_read_file(ngx_log_t *log, brix_file_t *file, off_t offset,
37 size_t length, off_t file_size)
38{
39 off_t end, hint_start, hint_end, low_water_end;
40 ngx_flag_t sequential;
41
42 if (file == NULL || file->fd < 0 || offset < 0
43 || length < BRIX_READ_PREFETCH_MIN)
44 {
45 return;
46 }
47
48 if ((off_t) length > NGX_MAX_OFF_T_VALUE - offset) {
49 return;
50 }
51
52 end = offset + (off_t) length;
53 if (file_size > 0 && end > file_size) {
54 end = file_size;
55 }
56 if (end <= offset) {
57 return;
58 }
59
60 sequential = (file->read_last_end < 0 || offset == file->read_last_end);
61 if (!sequential) {
62 file->read_ahead_end = 0;
63 }
64
65 if (sequential) {
66 if (end > NGX_MAX_OFF_T_VALUE - BRIX_READ_PREFETCH_LOW_WATER) {
67 low_water_end = NGX_MAX_OFF_T_VALUE;
68 } else {
69 low_water_end = end + BRIX_READ_PREFETCH_LOW_WATER;
70 }
71
72 if (file->read_ahead_end > low_water_end) {
73 file->read_last_end = end;
74 return;
75 }
76
77 hint_start = (file->read_ahead_end > offset)
78 ? file->read_ahead_end : offset;
79
80 if (end > NGX_MAX_OFF_T_VALUE - BRIX_READ_PREFETCH_WINDOW) {
81 hint_end = NGX_MAX_OFF_T_VALUE;
82 } else {
83 hint_end = end + BRIX_READ_PREFETCH_WINDOW;
84 }
85 if (file_size > 0 && hint_end > file_size) {
86 hint_end = file_size;
87 }
88
89 } else {
90 hint_start = offset;
91 hint_end = end;
92 }
93
94 if (hint_end > hint_start) {
95 brix_prefetch_fd_range(log, file->fd, hint_start,
96 (size_t) (hint_end - hint_start));
97 file->read_ahead_end = hint_end;
98 }
99
100 file->read_last_end = end;
101}

103/* Emit the pending coalesced prefetch range as a single WILLNEED hint on fd. */
104void
105brix_prefetch_flush(ngx_log_t *log, int fd, off_t range_start,
106 off_t range_end)
107{
108 if (fd >= 0 && range_end > range_start) {
109 brix_prefetch_fd_range(log, fd, range_start,
110 (size_t) (range_end - range_start));
111 }
112}

114/* Issue read-ahead hints for the segments of a kXR_readv request. */
115void
116brix_prefetch_readv_segments(brix_ctx_t *ctx, ngx_connection_t *c,
117 readahead_list *segments, size_t segment_count, size_t readv_seg_max)
118{
119 int merged_fd = -1;
120 off_t merged_start = 0;
121 off_t merged_end = 0;
122 size_t i;
123
124 for (i = 0; i < segment_count; i++) {
125 int handle_index;
126 int fd;
127 int64_t request_offset;
128 uint32_t request_length;
129 off_t segment_start;
130 off_t segment_end;
131
132 handle_index = (int) (unsigned char) segments[i].fhandle[0];
133 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES
134 || ctx->files[handle_index].fd < 0)
135 {
136 continue;
137 }
138
139 request_length = (uint32_t) ntohl((uint32_t) segments[i].rlen);
140 if ((size_t) request_length > readv_seg_max) {
141 request_length = (uint32_t) readv_seg_max;
142 }
143 if (request_length == 0) {
144 continue;
145 }
146
147 request_offset = (int64_t) be64toh((uint64_t) segments[i].offset);
148 if (request_offset < 0
149 || (off_t) request_length > NGX_MAX_OFF_T_VALUE - request_offset)
150 {
151 continue;
152 }
153
154 fd = ctx->files[handle_index].fd;
155 segment_start = (off_t) request_offset;
156 segment_end = segment_start + (off_t) request_length;
157 if (segment_end <= segment_start) {
158 continue;
159 }
160
161 /*
162 * readv often arrives as nearby ranges against the same file. Merge
163 * those ranges before issuing posix_fadvise() so one vector request
164 * does not turn into many tiny kernel hints.
165 */
166 if (merged_fd == fd
167 && segment_start >= merged_start
168 && segment_start <= merged_end + (off_t) BRIX_READV_PREFETCH_GAP
169 && segment_end - merged_start <= (off_t) BRIX_READV_PREFETCH_MAX)
170 {
171 if (segment_end > merged_end) {
172 merged_end = segment_end;
173 }
174 continue;
175 }
176
177 brix_prefetch_flush(c->log, merged_fd, merged_start, merged_end);
178 merged_fd = fd;
179 merged_start = segment_start;
180 merged_end = segment_end;
181 }
182
183 brix_prefetch_flush(c->log, merged_fd, merged_start, merged_end);
184}
