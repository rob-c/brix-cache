8/* Issue a POSIX_FADV_WILLNEED read-ahead hint for [offset, offset+length) on
9 * fd. Best-effort; a no-op on bad args or if fadvise is unavailable. */
10void
11brix_prefetch_fd_range(ngx_log_t *log, int fd, off_t offset, size_t length)
12{
13#if defined(POSIX_FADV_WILLNEED)
14 int rc;
15 if (fd < 0 || offset < 0 || length < BRIX_READ_PREFETCH_MIN) {
16 return;
17 }
18 rc = posix_fadvise(fd, offset, (off_t) length, POSIX_FADV_WILLNEED);
19 if (rc != 0) {
20 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, log, rc,
21 "brix: POSIX_FADV_WILLNEED ignored: %s",
22 strerror(rc));
23 }
24#else
25 (void) log;
26 (void) fd;
27 (void) offset;
28 (void) length;
29#endif
30}

32/* ---- Validate a read-file prefetch request and compute its clamped end ----
33 *
34 * WHAT: Applies the argument guards for brix_prefetch_read_file() and, when the
35 * request is prefetchable, writes the clamped end offset [offset, end) to
36 * *end_out and returns 1. Returns 0 (with *end_out untouched) when the handle
37 * is invalid, the read is too small, the offset+length would overflow off_t, or
38 * the range collapses to nothing after clamping to file_size.
39 *
40 * WHY: Isolating the guard/clamp arithmetic keeps the offset/length math (an
41 * overflow-safe hot-path calculation) in one small, reviewable place and lets
42 * the orchestrator read as a flat sequence of steps without a nested guard
43 * ladder.
44 *
45 * HOW:
46 * 1. Reject a null handle, closed fd, negative offset, or sub-threshold read.
47 * 2. Reject when offset + length would overflow NGX_MAX_OFF_T_VALUE.
48 * 3. Compute end = offset + length, clamping down to file_size when known.
49 * 4. Reject a range that clamped away to <= offset; otherwise publish end.
50 */
51static ngx_flag_t
52brix_prefetch_read_end(brix_file_t *file, off_t offset, size_t length,
53 off_t file_size, off_t *end_out)
54{
55 off_t end;
56
57 if (file == NULL || file->fd < 0 || offset < 0
58 || length < BRIX_READ_PREFETCH_MIN)
59 {
60 return 0;
61 }
62
63 if ((off_t) length > NGX_MAX_OFF_T_VALUE - offset) {
64 return 0;
65 }
66
67 end = offset + (off_t) length;
68 if (file_size > 0 && end > file_size) {
69 end = file_size;
70 }
71 if (end <= offset) {
72 return 0;
73 }
74
75 *end_out = end;
76 return 1;
77}

79/* ---- Compute the windowed WILLNEED hint range for a sequential read ----
80 *
81 * WHAT: For a read that continues the handle's sequential stream, writes the
82 * read-ahead hint range [*hint_start, *hint_end) that should be issued ahead of
83 * end. When the handle's existing read_ahead_end already extends past the
84 * low-water mark (end + LOW_WATER), no new hint is needed and both outputs are
85 * set equal (an empty range the caller skips).
86 *
87 * WHY: The sequential windowing — low-water suppression, start clamped to the
88 * already-prefetched frontier, and end extended by the read-ahead window and
89 * clamped to file_size — is the subtle part of the prefetch policy. Extracting
90 * it as a pure computation (state read from file, results returned by pointer,
91 * no I/O) keeps the arithmetic exact and independently reviewable.
92 *
93 * HOW:
94 * 1. Compute the low-water mark end + LOW_WATER (saturating at off_t max).
95 * 2. If read_ahead_end is already past low-water, emit an empty range (skip).
96 * 3. Start the hint at the further of read_ahead_end (the prefetched frontier)
97 * and offset.
98 * 4. End the hint at end + WINDOW (saturating), clamped down to file_size.
99 */
100static void
101brix_prefetch_seq_window(brix_file_t *file, off_t offset, off_t end,
102 off_t file_size, off_t *hint_start, off_t *hint_end)
103{
104 off_t low_water_end;
105
106 if (end > NGX_MAX_OFF_T_VALUE - BRIX_READ_PREFETCH_LOW_WATER) {
107 low_water_end = NGX_MAX_OFF_T_VALUE;
108 } else {
109 low_water_end = end + BRIX_READ_PREFETCH_LOW_WATER;
110 }
111
112 if (file->read_ahead_end > low_water_end) {
113 /* Already prefetched past the low-water mark; issue no new hint. */
114 *hint_start = 0;
115 *hint_end = 0;
116 return;
117 }
118
119 *hint_start = (file->read_ahead_end > offset)
120 ? file->read_ahead_end : offset;
121
122 if (end > NGX_MAX_OFF_T_VALUE - BRIX_READ_PREFETCH_WINDOW) {
123 *hint_end = NGX_MAX_OFF_T_VALUE;
124 } else {
125 *hint_end = end + BRIX_READ_PREFETCH_WINDOW;
126 }
127 if (file_size > 0 && *hint_end > file_size) {
128 *hint_end = file_size;
129 }
130}

132/* Sequential-aware prefetch for a read handle: detect sequential access from the
133 * handle's read_last_end / read-ahead window and issue a windowed WILLNEED hint
134 * ahead of the current read. No-op for random access. */
135void
136brix_prefetch_read_file(ngx_log_t *log, brix_file_t *file, off_t offset,
137 size_t length, off_t file_size)
138{
139 off_t end = 0, hint_start = 0, hint_end = 0;
140 ngx_flag_t sequential;
141
142 if (!brix_prefetch_read_end(file, offset, length, file_size, &end)) {
143 return;
144 }
145
146 sequential = (file->read_last_end < 0 || offset == file->read_last_end);
147 if (!sequential) {
148 /* Random access resets the read-ahead frontier and hints exactly the
149 * requested range. */
150 file->read_ahead_end = 0;
151 hint_start = offset;
152 hint_end = end;
153 } else {
154 brix_prefetch_seq_window(file, offset, end, file_size,
155 &hint_start, &hint_end);
156 }
157
158 if (hint_end > hint_start) {
159 brix_prefetch_fd_range(log, file->fd, hint_start,
160 (size_t) (hint_end - hint_start));
161 file->read_ahead_end = hint_end;
162 }
163
164 file->read_last_end = end;
165}

167/* Emit the pending coalesced prefetch range as a single WILLNEED hint on fd. */
168void
169brix_prefetch_flush(ngx_log_t *log, int fd, off_t range_start,
170 off_t range_end)
171{
172 if (fd >= 0 && range_end > range_start) {
173 brix_prefetch_fd_range(log, fd, range_start,
174 (size_t) (range_end - range_start));
175 }

178/* Issue read-ahead hints for the segments of a kXR_readv request. */
179void
180brix_prefetch_readv_segments(brix_ctx_t *ctx, ngx_connection_t *c,
181 readahead_list *segments, size_t segment_count, size_t readv_seg_max)
182{
183 int merged_fd = -1;
184 off_t merged_start = 0;
185 off_t merged_end = 0;
186 size_t i;
187
188 for (i = 0; i < segment_count; i++) {
189 int handle_index;
190 int fd;
191 int64_t request_offset;
192 uint32_t request_length;
193 off_t segment_start;
194 off_t segment_end;
195
196 handle_index = (int) (unsigned char) segments[i].fhandle[0];
197 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES
198 || ctx->files[handle_index].fd < 0)
199 {
200 continue;
201 }
202
203 request_length = (uint32_t) ntohl((uint32_t) segments[i].rlen);
204 if ((size_t) request_length > readv_seg_max) {
205 request_length = (uint32_t) readv_seg_max;
206 }
207 if (request_length == 0) {
208 continue;
209 }
210
211 request_offset = (int64_t) be64toh((uint64_t) segments[i].offset);
212 if (request_offset < 0
213 || (off_t) request_length > NGX_MAX_OFF_T_VALUE - request_offset)
214 {
215 continue;
216 }
217
218 fd = ctx->files[handle_index].fd;
219 segment_start = (off_t) request_offset;
220 segment_end = segment_start + (off_t) request_length;
221 if (segment_end <= segment_start) {
222 continue;
223 }
224
225 /*
226 * readv often arrives as nearby ranges against the same file. Merge
227 * those ranges before issuing posix_fadvise() so one vector request
228 * does not turn into many tiny kernel hints.
229 */
230 if (merged_fd == fd
231 && segment_start >= merged_start
232 && segment_start <= merged_end + (off_t) BRIX_READV_PREFETCH_GAP
233 && segment_end - merged_start <= (off_t) BRIX_READV_PREFETCH_MAX)
234 {
235 if (segment_end > merged_end) {
236 merged_end = segment_end;
237 }
238 continue;
239 }
240
241 brix_prefetch_flush(c->log, merged_fd, merged_start, merged_end);
242 merged_fd = fd;
243 merged_start = segment_start;
244 merged_end = segment_end;
245 }
246
247 brix_prefetch_flush(c->log, merged_fd, merged_start, merged_end);
248}
