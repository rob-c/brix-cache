83void
84brix_cms_reqid_map_put(const char *cms_reqid, const char *engine_reqid,
85 const char *notify, const char *prty)
86{
87 brix_cms_reqid_table_t *tbl;
88 brix_cms_reqid_entry_t *e;
89 uint32_t h;
90 ngx_uint_t i, idx, victim;
91 ngx_msec_t now = ngx_current_msec;
92
93 tbl = reqid_table();
94 if (tbl == NULL || cms_reqid == NULL || engine_reqid == NULL
95 || cms_reqid[0] == '\0'
96 || ngx_strlen(cms_reqid) >= BRIX_CMS_REQID_KEY_MAX
97 || ngx_strlen(engine_reqid) >= BRIX_CMS_REQID_ENGINE_MAX)
98 {
99 return;
100 }
101
102 h = reqid_hash(cms_reqid);
103 victim = BRIX_CMS_REQID_MAP_SLOTS; /* sentinel: none found yet */
104
105 ngx_shmtx_lock(&brix_cms_reqid_map_mutex);
106
107 for (i = 0; i < BRIX_CMS_REQID_MAP_SLOTS; i++) {
108 idx = (h + i) & (BRIX_CMS_REQID_MAP_SLOTS - 1);
109 e = &tbl->slots[idx];
110
111 if (e->in_use && !brix_shm_slot_expired(now, e->expires)
112 && (e->key_hash != h || ngx_strcmp(e->cms_reqid, cms_reqid) != 0))
113 {
114 continue; /* live entry for another reqid — keep probing */
115 }
116
117 victim = idx; /* free, expired, or same key: claim */
118 break;
119 }
120
121 if (victim == BRIX_CMS_REQID_MAP_SLOTS) {
122 /* Every slot live for other keys: overwrite the home slot —
123 * bounded eviction (reqid_map.h contract). */
124 victim = h & (BRIX_CMS_REQID_MAP_SLOTS - 1);
125 }
126
127 e = &tbl->slots[victim];
128 e->key_hash = h;
129 ngx_cpystrn((u_char *) e->cms_reqid, (u_char *) cms_reqid,
130 sizeof(e->cms_reqid));
131 ngx_cpystrn((u_char *) e->engine_reqid, (u_char *) engine_reqid,
132 sizeof(e->engine_reqid));
133 ngx_cpystrn((u_char *) e->notify,
134 (u_char *) (notify != NULL ? notify : ""),
135 sizeof(e->notify));
136 ngx_cpystrn((u_char *) e->prty,
137 (u_char *) (prty != NULL ? prty : ""),
138 sizeof(e->prty));
139 e->expires = now + BRIX_CMS_REQID_MAP_TTL_MS;
140 e->in_use = 1;
141
142 ngx_shmtx_unlock(&brix_cms_reqid_map_mutex);
143}

164

168 idx = (h + i) & (BRIX_CMS_REQID_MAP_SLOTS - 1);
169 e = &tbl->slots[idx];
170
171 if (!e->in_use) {
172 break; /* probe chain ends at the first never-used slot */
173 }
174 if (brix_shm_slot_expired(now, e->expires)) {
175 continue; /* stale — invisible, but keep probing past it */
176 }
177 if (e->key_hash == h && ngx_strcmp(e->cms_reqid, cms_reqid) == 0) {
178 ngx_cpystrn((u_char *) engine_reqid_out,
179 (u_char *) e->engine_reqid, engine_reqid_sz);
180 /* A prepdel consumes the mapping: expire the slot in place so
181 * the probe chain over it stays intact. */
182 e->expires = now;
183 hit = 1;
184 break;
185 }
186 }
187
188 ngx_shmtx_unlock(&brix_cms_reqid_map_mutex);
189 return hit;
