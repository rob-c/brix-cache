43/* WHAT: Build kXR_ok open response body (fhandle + optional statbuf from fstat) → brix_build_resp_hdr → brix_queue_response. Returns NGX_OK or NGX_ERROR on alloc failure. Caller: brix_tpc_prepare_pull (end of pull prep pipeline). */
44static ngx_int_t
45tpc_send_open_response(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
46 int fd, uint16_t options)
47{
48 ServerOpenBody body;
49 struct stat st;
50 char statbuf[256];
51 size_t bodylen;
52 size_t total;
53 u_char *buf;
54 ngx_flag_t want_stat;
55
56 want_stat = (options & kXR_retstat) ? 1 : 0;
57 statbuf[0] = '\0';
58 bodylen = sizeof(ServerOpenBody);
59
60 if (want_stat && fstat(fd, &st) == 0) {
61 int stat_flags = 0;
62
63 if (st.st_mode & (S_IRUSR | S_IRGRP | S_IROTH)) {
64 stat_flags |= kXR_readable;
65 }
66 if (st.st_mode & (S_IWUSR | S_IWGRP | S_IWOTH)) {
67 stat_flags |= kXR_writable;
68 }
69
70 snprintf(statbuf, sizeof(statbuf), "%llu %lld %d %ld",
71 (unsigned long long) st.st_ino,
72 (long long) st.st_size,
73 stat_flags,
74 (long) st.st_mtime);
75 bodylen += strlen(statbuf) + 1;
76 }
77
78 total = XRD_RESPONSE_HDR_LEN + bodylen;
79 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
80
81 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok, (uint32_t) bodylen,
82 (ServerResponseHdr *) buf);
83
84 ngx_memzero(&body, sizeof(body));
85 body.fhandle[0] = (u_char) idx;
86 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body, sizeof(body));
87
88 if (statbuf[0] != '\0') {
89 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + sizeof(ServerOpenBody),
90 statbuf, strlen(statbuf) + 1);
91 }
92
93 return brix_queue_response(ctx, c, buf, total);
94}

144/* WHAT: Translate the kXR_open options bitmask into POSIX open(2) flags for the
145 * TPC destination file.
146 * WHY: The wire-level create semantics differ from POSIX, so the mapping is
147 * explicit: kXR_new alone => create, fail if it already exists (O_EXCL);
148 * kXR_new + kXR_delete or kXR_delete alone => create-or-truncate; neither flag
149 * (the common "just receive the copy") => create-or-truncate as well. The
150 * O_EXCL is deliberately dropped whenever kXR_delete is present, since delete
151 * means "overwrite is intended". Always O_RDWR (we read back for stat) and
152 * O_NOCTTY (never acquire a controlling terminal). */
153static int
154tpc_destination_open_flags(uint16_t options)
155{
156 int oflags;
157
158 oflags = O_RDWR | O_NOCTTY;
159 if (options & kXR_new) {
160 oflags |= O_CREAT;
161 if (!(options & kXR_delete)) {
162 oflags |= O_EXCL; /* create-new only: refuse existing target */
163 }
164 }
165 if (options & kXR_delete) {
166 oflags |= O_CREAT | O_TRUNC; /* overwrite intended */
167 }
168 if (!(options & (kXR_new | kXR_delete))) {
169 oflags |= O_CREAT | O_TRUNC; /* default: create or replace */
170 }
171
172 return oflags;
173}

175/* WHAT: Validate the pull preconditions before any state is allocated — a
176 * configured thread pool, a non-empty TPC source host+path, and an SSRF-safe
177 * source address per the loopback/private allow flags. Sends the matching kXR
178 * error (and logs it) and returns non-NGX_OK on the first failure; returns
179 * NGX_OK when the request may proceed to fhandle allocation.
180 * WHY: prepare_pull's guard ladder is the security-load-bearing front door
181 * (SSRF gate included); isolating it keeps the orchestrator a flat sequence and
182 * the checks independently reviewable. Behaviour (order, codes, log strings) is

231/* WHAT: Open the TPC destination file relative to the export root, returning the
232 * new fd (or -1 with errno set). Strips the root_canon prefix off the absolute
233 * dst_path first so brix_vfs_open_fd_at() receives the LOGICAL export path.
234 * WHY: brix_vfs_open_fd_at() resolves relative to conf->rootfd (it strips the
235 * leading '/' via brix_beneath_rel), so it must be handed the logical path —
236 * passing the absolute root_canon-prefixed path doubles the root prefix and
237 * openat2() fails with ENOENT. dst_path itself must stay absolute for
238 * authz/logging/fhandle metadata, so the strip is confined here.
239 * HOW: if root_canon (len>1) is a prefix of dst_path at a '/' or end boundary,
240 * advance past it; open with tpc_destination_open_flags(options) + create_mode. */
241static int
242tpc_open_dst_logical(ngx_stream_brix_srv_conf_t *conf, const char *dst_path,
243 uint16_t options, mode_t create_mode)
244{
245 const char *dst_logical = dst_path;
246 size_t root_len = ngx_strlen(conf->common.root_canon);
247
248 if (root_len > 1
249 && ngx_strncmp(dst_path, conf->common.root_canon, root_len) == 0
250 && (dst_path[root_len] == '/' || dst_path[root_len] == '\0'))
251 {
252 dst_logical = dst_path + root_len;
253 }
254
255 return brix_vfs_open_fd_at(conf->rootfd, dst_logical,
256 tpc_destination_open_flags(options),
257 create_mode);
258}
