1/*
2 * locate.c — kXR_locate (3027) opcode handler: resolve a path to a serving endpoint.
3 *
4 * WHAT: Implements brix_handle_locate(), the protocol handler for kXR_locate.
5 * For a given client path it answers "which server should you talk to?" by
6 * returning either a kXR_redirect to another host (manager/CMS modes) or a
7 * kXR_ok carrying an "Sx<host>:<port>" location token (data-server mode),
8 * where the leading S marks a server and the access char ('w' if
9 * conf->common.allow_write, else 'r') advertises read/write capability.
10 * Also handles the "*" wildcard form (locate the local server itself).
11 *
12 * WHY: kXR_locate underpins XRootD's redirection/clustering model: clients query
13 * a manager to discover the data server holding a file before opening it.
14 * This handler centralises that discovery across the module's deployment
15 * modes (standalone data server, static manager map, dynamic registry,
16 * and CMS-backed cluster) so callers get a single consistent answer.
17 *
18 * HOW: Parse and confine the request path via brix_extract_path(), reject
19 * over-deep paths with brix_count_path_depth(). In manager_mode (non-
20 * wildcard) try, in order: the collapse-redir cache
21 * (brix_redir_cache_lookup), the live server registry (brix_srv_select,
22 * seeding the cache on hit), then an async kYR_locate to the CMS parent —
23 * which suspends the stream (XRD_ST_WAITING_CMS, pending registry +
24 * cms_locate_timeout timer) and returns NGX_AGAIN until the reply arrives.
25 * Falling through, consult the static manager_map (brix_find_manager_map),
26 * then in data-server mode stat the file beneath conf->rootfd
27 * (brix_stat_beneath) — redirecting to an upstream if configured and
28 * missing, else 404 — and enforce read access via brix_auth_gate before
29 * formatting the local "Sx..." location from c->local_sockaddr (IPv4/IPv6/
30 * fallback) and replying with brix_send_ok.
31 */

43ngx_int_t
44brix_handle_locate(brix_ctx_t *ctx, ngx_connection_t *c,
45 ngx_stream_brix_srv_conf_t *conf)
46{
47 xrdw_locate_req_t req;
48 char reqpath_buf[BRIX_MAX_PATH + 1];
49 struct sockaddr_in *sin;
50 char loc_buf[256];
51 int loc_len;
52 int is_wildcard;
53 uint16_t port;
54 char access_char;
55
56 /* options: kXR_prefname (0x0100) = prefer DNS names over IPs in response.
57 * We always store the server's registered hostname so this is the default.
58 * Parse the field so the compiler sees req as used. */
59 xrdw_locate_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
60 (void) req.options;
61
62 if (ctx->recv.cur_dlen == 0 || ctx->recv.payload == NULL) {
63 BRIX_OP_ERR(ctx, BRIX_OP_LOCATE);
64 return brix_send_error(ctx, c, kXR_ArgMissing, "no path given");
65 }
66
67 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
68 reqpath_buf, sizeof(reqpath_buf), 1))
69 {
70 BRIX_OP_ERR(ctx, BRIX_OP_LOCATE);
71 return brix_send_error(ctx, c, kXR_ArgInvalid,
72 "invalid path payload");
73 }
74
75 is_wildcard = (reqpath_buf[0] == '*' && reqpath_buf[1] == '\0');
76
77 /* Reject any ".." component (the reference does not normalize ".."); locate
78 * resolves through the kernel RESOLVE_BENEATH which would collapse it.
79 * The "*" wildcard locate carries no path to traverse. */
80 if (!is_wildcard
81 && brix_reject_dotdot_path(ctx, c, BRIX_OP_LOCATE, "LOCATE",
82 reqpath_buf)) {
83 return ctx->write_rc;
84 }
85
86 if (!is_wildcard && brix_count_path_depth(reqpath_buf) != NGX_OK) {
87 BRIX_OP_ERR(ctx, BRIX_OP_LOCATE);
88 return brix_send_error(ctx, c, kXR_ArgInvalid,
89 "path exceeds maximum depth");
90 }
91
92 if (conf->manager_mode && !is_wildcard) {
93
94 /* Collapse-redir cache: fast path — single recently-resolved server. */
95 if (conf->caps.collapse_redir) {
96 char redir_host[256];
97 uint16_t redir_port;
98 if (brix_redir_cache_lookup(reqpath_buf, redir_host,
99 sizeof(redir_host), &redir_port)) {
100 brix_log_access(ctx, c, "LOCATE", reqpath_buf,
101 "redir-cache", 1, 0, NULL, 0);
102 BRIX_OP_OK(ctx, BRIX_OP_LOCATE);
103 return brix_send_redirect(ctx, c, redir_host, redir_port);
104 }
105 }
106
107 /* Registry: redirect to the best available server for this path. */
108 {
109 char redir_host[256];
110 uint16_t redir_port;
111 if (brix_srv_select(reqpath_buf, 0, redir_host,
112 sizeof(redir_host), &redir_port)) {
113 if (conf->caps.collapse_redir) {
114 brix_redir_cache_insert(reqpath_buf, redir_host,
115 redir_port,
116 conf->caps.collapse_redir_ttl);
117 }
118 brix_log_access(ctx, c, "LOCATE", reqpath_buf,
119 "registry", 1, 0, NULL, 0);
120 BRIX_OP_OK(ctx, BRIX_OP_LOCATE);
121 return brix_send_redirect(ctx, c, redir_host, redir_port);
122 }
123 }
124
125 /* Registry miss — ask the CMS parent via kYR_locate. */
126 if (conf->cms.ctx != NULL) {
127 uint32_t streamid;
128
129 streamid = ngx_brix_cms_next_streamid(conf->cms.ctx);
130 if (brix_pending_insert(streamid, ngx_pid, c->fd,
131 c->number,
132 ctx->recv.cur_streamid,
133 conf->cms.locate_timeout) == NGX_OK)
134 {
135 ctx->cms_wait_streamid = streamid;
136 ctx->state = XRD_ST_WAITING_CMS;
137 ngx_add_timer(c->read, conf->cms.locate_timeout);
138 if (ngx_brix_cms_send_locate(conf->cms.ctx, streamid,
139 reqpath_buf) == NGX_OK)
140 {
141 return NGX_AGAIN;
142 }
143 ngx_del_timer(c->read);
144 ctx->state = XRD_ST_REQ_HEADER;
145 brix_pending_remove(streamid, ngx_pid);
146 }
147 /* Fall through to static-map / notFound if suspend fails. */
148 }
149 }
150
151 if (!is_wildcard && conf->manager_map != NULL) {
152 const brix_manager_map_t *m;
153
154 m = brix_find_manager_map(reqpath_buf, conf->manager_map);
155 if (m != NULL) {
156 brix_log_access(ctx, c, "LOCATE", reqpath_buf, "redirect",
157 1, 0, NULL, 0);
158 BRIX_OP_OK(ctx, BRIX_OP_LOCATE);
159 return brix_send_redirect(ctx, c, (const char *) m->host.data,
160 m->port);
161 }
162 }
163
164 if (!is_wildcard) {
165 struct stat _st;
166 char full_path[PATH_MAX];
167
168 /* SECURITY: authorize BEFORE probing existence so a denied principal
169 * gets an identical kXR_NotAuthorized whether or not the path exists —
170 * no namespace-existence oracle (mirrors the statx.c ordering). The gate
171 * keys off the identity + logical path, independent of on-disk existence. */
172 brix_beneath_full_path(conf->common.root_canon, reqpath_buf,
173 full_path, sizeof(full_path));
174 if (brix_auth_gate(ctx, c, BRIX_OP_LOCATE, "LOCATE",
175 reqpath_buf, full_path, conf,
176 BRIX_AUTH_READ, 0) != NGX_OK) {
177 return ctx->write_rc;
178 }
179
180 if (brix_stat_beneath(conf->rootfd, reqpath_buf, &_st) != 0) {
181 if (conf->upstream_host.len > 0) {
182 brix_log_access(ctx, c, "LOCATE", reqpath_buf,
183 "upstream", 1, 0, NULL, 0);
184 BRIX_OP_OK(ctx, BRIX_OP_LOCATE);
185 return brix_upstream_start(ctx, c, conf);
186 }
187
188 BRIX_RETURN_ERR(ctx, c, BRIX_OP_LOCATE, "LOCATE",
189 reqpath_buf, "-", kXR_NotFound, "file not found");
190 }
191 }
192
193 access_char = conf->common.allow_write ? 'w' : 'r';
194
195 if (c->local_sockaddr != NULL
196 && c->local_sockaddr->sa_family == AF_INET)
197 {
198 char ipbuf[INET_ADDRSTRLEN];
199 sin = (struct sockaddr_in *) c->local_sockaddr;
200 port = ntohs(sin->sin_port);
201 inet_ntop(AF_INET, &sin->sin_addr, ipbuf, sizeof(ipbuf));
202 loc_len = snprintf(loc_buf, sizeof(loc_buf), "S%c%s:%d",
203 access_char, ipbuf, (int) port);
204 } else if (c->local_sockaddr != NULL
205 && c->local_sockaddr->sa_family == AF_INET6)
206 {
207 char ipbuf[INET6_ADDRSTRLEN];
208 struct sockaddr_in6 *sin6;
209 sin6 = (struct sockaddr_in6 *) c->local_sockaddr;
210 port = ntohs(sin6->sin6_port);
211 inet_ntop(AF_INET6, &sin6->sin6_addr, ipbuf, sizeof(ipbuf));
212 loc_len = snprintf(loc_buf, sizeof(loc_buf), "S%c[%s]:%d",
213 access_char, ipbuf, (int) port);
214 } else {
215 loc_len = snprintf(loc_buf, sizeof(loc_buf), "S%clocalhost",
216 access_char);
217 }
218
219 brix_log_access(ctx, c, "LOCATE", reqpath_buf, loc_buf,
220 1, 0, NULL, 0);
221 BRIX_OP_OK(ctx, BRIX_OP_LOCATE);
222
223 return brix_send_ok(ctx, c, loc_buf, (uint32_t) (loc_len + 1));
224}
