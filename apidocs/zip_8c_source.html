1/*
2 * zip.c - (kept) routing + shared helpers
3 * Phase-38 split of zip.c; behavior-identical.
4 *
5 * The security-critical leaf parsers (LE field readers, EOCD/ZIP64 location,
6 * ZIP64-extra decode, local-header data-offset resolution) live in the shared,
7 * unit-tested zip_kernel.c (src/protocols/root/zip/) — the single copy linked into both the
8 * native client and the nginx module. The functions below are thin client-side
9 * adapters that keep the existing zip_internal.h signatures stable (so
10 * zip_write.c / copy_zip.c are untouched) and map the kernel's ZIP_K_* codes
11 * onto the client's XRDC_ZIP_* enum.

16/* Map a shared-kernel result code onto the client's XRDC_ZIP_* enum. */
17static int
18zip_map_kernel_rc(int krc)
19{
20 switch (krc) {
21 case ZIP_K_OK: return XRDC_ZIP_OK;
22 case ZIP_K_ENOTZIP: return XRDC_ZIP_ENOTZIP;
23 case ZIP_K_EIO: return XRDC_ZIP_EIO;
24 default: return XRDC_ZIP_EBADF; /* ECORRUPT / bounds */
25 }
26}

68int
69brix_zip_open(brix_zip_pread_fn pread, void *ctx, uint64_t archive_size,
70 brix_zip_dir *out)
71{
72 uint64_t cd_off, cd_size, n_entries;
73 uint8_t *cd;
74 size_t pos = 0, idx = 0;
75 int rc;
76
77 if (out == NULL || pread == NULL) {
78 return XRDC_ZIP_EBADF;
79 }
80 memset(out, 0, sizeof(*out));
81 out->archive_size = archive_size;
82
83 rc = find_eocd(pread, ctx, archive_size, &cd_off, &cd_size, &n_entries);
84 if (rc != XRDC_ZIP_OK) {
85 return rc;
86 }
87
88 cd = malloc(cd_size ? (size_t) cd_size : 1);
89 if (cd == NULL) {
90 return XRDC_ZIP_ENOMEM;
91 }
92 rc = read_exact(pread, ctx, cd_off, archive_size, cd, (size_t) cd_size);
93 if (rc != XRDC_ZIP_OK) {
94 free(cd);
95 return rc;
96 }
97
98 out->entries = calloc(n_entries ? (size_t) n_entries : 1,
99 sizeof(brix_zip_entry));
100 if (out->entries == NULL) {
101 free(cd);
102 return XRDC_ZIP_ENOMEM;
103 }
104
105 while (idx < n_entries && pos + 46 <= cd_size) {
106 const uint8_t *h = cd + pos;
107 uint16_t namelen, extralen, commentlen;
108 brix_zip_entry *ent;
109
110 if (le32(h) != SIG_CDFH) {
111 break;
112 }
113 namelen = le16(h + 28);
114 extralen = le16(h + 30);
115 commentlen = le16(h + 32);
116 if ((uint64_t) 46 + namelen + extralen + commentlen > cd_size - pos
117 || namelen > XRDC_ZIP_MAX_NAME) {
118 free(cd);
119 brix_zip_dir_free(out);
120 return XRDC_ZIP_EBADF;
121 }
122
123 ent = &out->entries[idx];
124 ent->method = le16(h + 10);
125 ent->crc32 = le32(h + 16);
126 ent->comp_size = le32(h + 20);
127 ent->uncomp_size = le32(h + 24);
128 ent->lfh_off = le32(h + 42);
129 ent->archive_size = archive_size;
130
131 ent->name = malloc((size_t) namelen + 1);
132 if (ent->name == NULL) {
133 free(cd);
134 brix_zip_dir_free(out);
135 return XRDC_ZIP_ENOMEM;
136 }
137 memcpy(ent->name, h + 46, namelen);
138 ent->name[namelen] = '\0';
139
140 apply_zip64_extra(ent, h + 46 + namelen, extralen);
141
142 /* Member data extent must fit inside the archive (best-effort: lfh + data). */
143 if (ent->lfh_off >= archive_size
144 || ent->comp_size > archive_size - ent->lfh_off) {
145 /* ent->name is allocated but this entry is not yet counted in out->n
146 * (bumped below), so brix_zip_dir_free would not reach it — free it
147 * here to avoid leaking the name on this reject path. */
148 free(ent->name);
149 ent->name = NULL;
150 free(cd);
151 brix_zip_dir_free(out);
152 return XRDC_ZIP_EBADF;
153 }
154
155 out->n = ++idx;
156 pos += (size_t) 46 + namelen + extralen + commentlen;
157 }
158
159 free(cd);
160 return XRDC_ZIP_OK;
161}

164const brix_zip_entry *
165brix_zip_find(const brix_zip_dir *d, const char *name)
166{
167 size_t i;
168
169 if (d == NULL || name == NULL) {
170 return NULL;
171 }
172 for (i = 0; i < d->n; i++) {
173 if (strcmp(d->entries[i].name, name) == 0) {
174 return &d->entries[i];
175 }
176 }
177 return NULL;
178}

193 d->entries = NULL;

209/* Feed produced plaintext to the sink while accumulating CRC + the size cap. */
210int
211sink_output(brix_zip_sink_fn sink, void *sink_ctx, const uint8_t *data, size_t len,
212 uLong *crc, uint64_t *produced, uint64_t uncomp_size)
213{
214 if (len == 0) {
215 return XRDC_ZIP_OK;
216 }
217 *produced += len;
218 if (*produced > uncomp_size || *produced > XRDC_ZIP_MAX_OUTPUT) {
219 return XRDC_ZIP_EBOMB; /* more output than declared = corrupt/bomb */
220 }
221 *crc = crc32(*crc, data, (uInt) len);
222 if (sink(sink_ctx, data, len) != 0) {
223 return XRDC_ZIP_EIO;
224 }
225 return XRDC_ZIP_OK;
226}

229int
230brix_zip_member_extract(brix_zip_pread_fn pread, void *ctx,
231 const brix_zip_entry *e,
232 brix_zip_sink_fn sink, void *sink_ctx)
233{
234 enum { CHUNK = 64 * 1024 };
235 uint64_t data_off, remaining, produced = 0;
236 uint8_t *inbuf, *outbuf = NULL;
237 uLong crc = crc32(0L, Z_NULL, 0);
238 z_stream zs;
239 int rc;
240
241 if (e->method != 0 && e->method != 8) {
242 return XRDC_ZIP_EMETHOD;
243 }
244 if (e->uncomp_size > XRDC_ZIP_MAX_OUTPUT) {
245 return XRDC_ZIP_EBOMB;
246 }
247 rc = member_data_offset(pread, ctx, e->archive_size, e, &data_off);
248 if (rc != XRDC_ZIP_OK) {
249 return rc;
250 }
251
252 inbuf = malloc(CHUNK);
253 if (inbuf == NULL) {
254 return XRDC_ZIP_ENOMEM;
255 }
256 if (e->method == 8) {
257 outbuf = malloc(CHUNK);
258 if (outbuf == NULL) { free(inbuf); return XRDC_ZIP_ENOMEM; }
259 memset(&zs, 0, sizeof(zs));
260 if (inflateInit2(&zs, -15) != Z_OK) { /* raw deflate, no header */
261 free(inbuf); free(outbuf);
262 return XRDC_ZIP_EINFLATE;
263 }
264 }
265
266 remaining = e->comp_size;
267 rc = XRDC_ZIP_OK;
268 while (remaining > 0) {
269 size_t want = (remaining < CHUNK) ? (size_t) remaining : CHUNK;
270
271 rc = read_exact(pread, ctx, data_off, e->archive_size, inbuf, want);
272 if (rc != XRDC_ZIP_OK) {
273 break;
274 }
275 data_off += want;
276 remaining -= want;
277
278 if (e->method == 0) { /* STORE: raw copy */
279 rc = sink_output(sink, sink_ctx, inbuf, want, &crc, &produced,
280 e->uncomp_size);
281 if (rc != XRDC_ZIP_OK) {
282 break;
283 }
284 continue;
285 }
286
287 /* DEFLATE: inflate this chunk, draining output. */
288 zs.next_in = inbuf;
289 zs.avail_in = (uInt) want;
290 do {
291 int zr;
292 zs.next_out = outbuf;
293 zs.avail_out = CHUNK;
294 zr = inflate(&zs, Z_NO_FLUSH);
295 if (zr != Z_OK && zr != Z_STREAM_END && zr != Z_BUF_ERROR) {
296 rc = XRDC_ZIP_EINFLATE;
297 break;
298 }
299 rc = sink_output(sink, sink_ctx, outbuf, CHUNK - zs.avail_out,
300 &crc, &produced, e->uncomp_size);
301 if (rc != XRDC_ZIP_OK) {
302 break;
303 }
304 if (zr == Z_STREAM_END) {
305 break;
306 }
307 } while (zs.avail_in > 0 || zs.avail_out == 0);
308 if (rc != XRDC_ZIP_OK) {
309 break;
310 }
311 }
312
313 if (e->method == 8) {
314 inflateEnd(&zs);
315 free(outbuf);
316 }
317 free(inbuf);
318
319 if (rc != XRDC_ZIP_OK) {
320 return rc;
321 }
322 if (produced != e->uncomp_size) {
323 return XRDC_ZIP_EBADF; /* short/over: truncated or corrupt */
324 }
325 if ((uint32_t) crc != e->crc32) {
326 return XRDC_ZIP_ECRC;
327 }
328 return XRDC_ZIP_OK;
329}
