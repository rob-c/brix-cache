16/* Map a shared-kernel result code onto the client's XRDC_ZIP_* enum. */
17static int
18zip_map_kernel_rc(int krc)
19{
20 switch (krc) {
21 case ZIP_K_OK: return XRDC_ZIP_OK;
22 case ZIP_K_ENOTZIP: return XRDC_ZIP_ENOTZIP;
23 case ZIP_K_EIO: return XRDC_ZIP_EIO;
24 default: return XRDC_ZIP_EBADF; /* ECORRUPT / bounds */
25 }
26}

69 * WHAT: The central directory as slurped from the archive — its malloc'd bytes,

81/*
82 * WHAT: Locate + read the whole central directory into a freshly malloc'd
83 * buffer, reporting its byte size and declared entry count in *cd.
84 * WHY: Splits the EOCD-locate + CD-slurp preamble out of brix_zip_open so the
85 * orchestrator reads as a flat sequence of named steps (coding-standards
86 * §8.5), keeping the CDFH parse loop as the only complex region.
87 * HOW: find_eocd() → malloc(size) → read_exact(). On any failure the buffer is
88 * freed here and cd->bytes is left NULL so the caller has nothing to
89 * release. Byte layout unchanged — same kernel calls, order, rc mapping.
90 */
91static int
92zip_read_central_dir(brix_zip_pread_fn pread, void *ctx, uint64_t archive_size,
93 zip_cd_t *cd)
94{
95 uint64_t cd_off = 0;
96 uint8_t *buf;
97 int rc;
98
99 cd->bytes = NULL;
100 cd->size = 0;
101 cd->n_entries = 0;
102
103 rc = find_eocd(pread, ctx, archive_size, &cd_off, &cd->size,
104 &cd->n_entries);
105 if (rc != XRDC_ZIP_OK) {
106 return rc;
107 }
108
109 buf = malloc(cd->size ? (size_t) cd->size : 1);
110 if (buf == NULL) {
111 return XRDC_ZIP_ENOMEM;
112 }
113 rc = read_exact(pread, ctx, cd_off, archive_size, buf, (size_t) cd->size);
114 if (rc != XRDC_ZIP_OK) {
115 free(buf);
116 return rc;
117 }
118 cd->bytes = buf;
119 return XRDC_ZIP_OK;
120}

123/*
124 * WHAT: Parse + validate ONE central-directory file header at cd[pos] into
125 * out->entries[idx], and advance *pos past this record on success.
126 * WHY: The per-entry CDFH decode (field reads, bounds/name-length checks,
127 * ZIP64-extra apply, in-archive extent check, name copy) is the sole
128 * complexity source in brix_zip_open; extracting it makes each check
129 * independently reviewable (§8.1) without changing any wire read.
130 * HOW: Reads the 46-byte fixed CDFH via le16/le32 at the frozen offsets, copies
131 * the name, applies the ZIP64 extra, then bounds-checks lfh + comp_size.
132 * On reject the freshly-allocated name (not yet counted in out->n) is freed
133 * here to avoid a leak, mirroring the original inline logic exactly.
134 * Returns XRDC_ZIP_OK to keep parsing, ZIP_CD_STOP to end the loop cleanly
135 * (non-CDFH signature), or a negative XRDC_ZIP_* error on a malformed entry.
136 */
137enum {
138 ZIP_CDFH_FIXED = 46, /* fixed-length prefix of a central-directory header */
139 ZIP_CD_STOP = 1 /* file-local: clean end of directory (positive, so it
140 * never collides with the 0/negative XRDC_ZIP_* set) */
141};
142
143static int
144zip_parse_cd_entry(const uint8_t *cd, uint64_t cd_size, size_t *pos,
145 uint64_t archive_size, brix_zip_entry *ent)
146{
147 const uint8_t *h = cd + *pos;
148 uint16_t namelen, extralen, commentlen;
149
150 if (le32(h) != SIG_CDFH) {
151 return ZIP_CD_STOP;
152 }
153 namelen = le16(h + 28);
154 extralen = le16(h + 30);
155 commentlen = le16(h + 32);
156 if ((uint64_t) ZIP_CDFH_FIXED + namelen + extralen + commentlen
157 > cd_size - *pos
158 || namelen > XRDC_ZIP_MAX_NAME) {
159 return XRDC_ZIP_EBADF;
160 }
161
162 ent->method = le16(h + 10);
163 ent->crc32 = le32(h + 16);
164 ent->comp_size = le32(h + 20);
165 ent->uncomp_size = le32(h + 24);
166 ent->lfh_off = le32(h + 42);
167 ent->archive_size = archive_size;
168
169 ent->name = malloc((size_t) namelen + 1);
170 if (ent->name == NULL) {
171 return XRDC_ZIP_ENOMEM;
172 }
173 memcpy(ent->name, h + ZIP_CDFH_FIXED, namelen);
174 ent->name[namelen] = '\0';
175
176 apply_zip64_extra(ent, h + ZIP_CDFH_FIXED + namelen, extralen);
177
178 /* Member data extent must fit inside the archive (best-effort: lfh + data). */
179 if (ent->lfh_off >= archive_size
180 || ent->comp_size > archive_size - ent->lfh_off) {
181 /* ent->name is allocated but this entry is not yet counted in out->n,
182 * so brix_zip_dir_free would not reach it — free it here to avoid
183 * leaking the name on this reject path. */
184 free(ent->name);
185 ent->name = NULL;
186 return XRDC_ZIP_EBADF;
187 }
188
189 *pos += (size_t) ZIP_CDFH_FIXED + namelen + extralen + commentlen;
190 return XRDC_ZIP_OK;
191}

194int
195brix_zip_open(brix_zip_pread_fn pread, void *ctx, uint64_t archive_size,
196 brix_zip_dir *out)
197{
198 zip_cd_t cd = {0};
199 size_t pos = 0, idx = 0;
200 int rc;
201
202 if (out == NULL || pread == NULL) {
203 return XRDC_ZIP_EBADF;
204 }
205 memset(out, 0, sizeof(*out));
206 out->archive_size = archive_size;
207
208 rc = zip_read_central_dir(pread, ctx, archive_size, &cd);
209 if (rc != XRDC_ZIP_OK) {
210 return rc;
211 }
212
213 out->entries = calloc(cd.n_entries ? (size_t) cd.n_entries : 1,
214 sizeof(brix_zip_entry));
215 if (out->entries == NULL) {
216 free(cd.bytes);
217 return XRDC_ZIP_ENOMEM;
218 }
219
220 while (idx < cd.n_entries && pos + ZIP_CDFH_FIXED <= cd.size) {
221 rc = zip_parse_cd_entry(cd.bytes, cd.size, &pos, archive_size,
222 &out->entries[idx]);
223 if (rc == ZIP_CD_STOP) {
224 break;
225 }
226 if (rc != XRDC_ZIP_OK) {
227 free(cd.bytes);
228 brix_zip_dir_free(out);
229 return rc;
230 }
231 out->n = ++idx;
232 }
233
234 free(cd.bytes);
235 return XRDC_ZIP_OK;
236}

284/* Feed produced plaintext to the sink while accumulating CRC + the size cap. */
285int
286sink_output(brix_zip_sink_fn sink, void *sink_ctx, const uint8_t *data, size_t len,
287 uLong *crc, uint64_t *produced, uint64_t uncomp_size)
288{
289 if (len == 0) {
290 return XRDC_ZIP_OK;
291 }
292 *produced += len;
293 if (*produced > uncomp_size || *produced > XRDC_ZIP_MAX_OUTPUT) {
294 return XRDC_ZIP_EBOMB; /* more output than declared = corrupt/bomb */
295 }
296 *crc = crc32(*crc, data, (uInt) len);
297 if (sink(sink_ctx, data, len) != 0) {
298 return XRDC_ZIP_EIO;
299 }
300 return XRDC_ZIP_OK;
301}

307/*
308 * WHAT: Running state for a single member extraction — the decode buffers,
309 * inflate stream (DEFLATE only), CRC accumulator, output-size counter and
310 * the archive read cursor.
311 * WHY: Bundling the ~8 locals that member_extract threads through its inner
312 * loops lets each step take one `ctx` pointer instead of a long parameter
313 * list (§8.2), keeping the extracted helpers well under the 5-param cap.
314 * HOW: `inflating` records whether `zs` was initialised so teardown only ends a
315 * stream it actually started. Purely a container — no behaviour of its own.
316 */
317typedef struct {
318 const brix_zip_entry *e;
319 brix_zip_pread_fn pread;
320 void *pread_ctx;
321 brix_zip_sink_fn sink;
322 void *sink_ctx;
323 uint8_t *inbuf;
324 uint8_t *outbuf;
325 z_stream zs;
326 int inflating;
327 uLong crc;
328 uint64_t produced;
329 uint64_t data_off;
330} zip_extract_t;

333/*
334 * WHAT: Allocate the decode buffers and, for DEFLATE members, initialise the
335 * raw-inflate stream inside `ex`.
336 * WHY: Isolates the resource-acquisition preamble so the orchestrator is a flat
337 * sequence and each early-return frees exactly what it owns (§4 Recipe 1),
338 * no goto.
339 * HOW: Always allocates `inbuf`; for method 8 also allocates `outbuf` and calls
340 * inflateInit2(-15) (raw deflate, no header). Frees partially-acquired
341 * buffers on failure and sets `ex->inflating` only once the stream is live.
342 */
343static int
344zip_extract_setup(zip_extract_t *ex)
345{
346 ex->inbuf = malloc(ZIP_EXTRACT_CHUNK);
347 if (ex->inbuf == NULL) {
348 return XRDC_ZIP_ENOMEM;
349 }
350 if (ex->e->method == 8) {
351 ex->outbuf = malloc(ZIP_EXTRACT_CHUNK);
352 if (ex->outbuf == NULL) {
353 free(ex->inbuf);
354 ex->inbuf = NULL;
355 return XRDC_ZIP_ENOMEM;
356 }
357 memset(&ex->zs, 0, sizeof(ex->zs));
358 if (inflateInit2(&ex->zs, -15) != Z_OK) { /* raw deflate, no header */
359 free(ex->inbuf);
360 free(ex->outbuf);
361 ex->inbuf = ex->outbuf = NULL;
362 return XRDC_ZIP_EINFLATE;
363 }
364 ex->inflating = 1;
365 }
366 return XRDC_ZIP_OK;
367}

370/*
371 * WHAT: Release everything zip_extract_setup acquired (inflate stream + buffers).
372 * WHY: One owner for teardown keeps the orchestrator's exits linear (§4).
373 * HOW: Ends the inflate stream only if it was started, then frees both buffers.
374 */
375static void
376zip_extract_teardown(zip_extract_t *ex)
377{
378 if (ex->inflating) {
379 inflateEnd(&ex->zs);
380 }
381 free(ex->outbuf);
382 free(ex->inbuf);
383}

386/*
387 * WHAT: Inflate one already-read compressed chunk of `len` bytes, draining all
388 * produced plaintext to the sink.
389 * WHY: The nested inflate/drain loop is the dominant complexity of member
390 * extraction; extracting it makes the outer read loop trivial (§8.5) and
391 * leaves this loop the single testable unit for the DEFLATE path.
392 * HOW: Feeds `ex->inbuf[0..len)` into the stream, then loops filling `outbuf`
393 * and sinking `CHUNK - avail_out` bytes until input is drained (or output
394 * stops filling). Treats Z_BUF_ERROR as benign (no progress) exactly as the
395 * original; any other non-OK/non-STREAM_END code is a corrupt stream.
396 */
397static int
398zip_extract_inflate_chunk(zip_extract_t *ex, size_t len)
399{
400 ex->zs.next_in = ex->inbuf;
401 ex->zs.avail_in = (uInt) len;
402 do {
403 int rc;
404 int zr;
405
406 ex->zs.next_out = ex->outbuf;
407 ex->zs.avail_out = ZIP_EXTRACT_CHUNK;
408 zr = inflate(&ex->zs, Z_NO_FLUSH);
409 if (zr != Z_OK && zr != Z_STREAM_END && zr != Z_BUF_ERROR) {
410 return XRDC_ZIP_EINFLATE;
411 }
412 rc = sink_output(ex->sink, ex->sink_ctx, ex->outbuf,
413 ZIP_EXTRACT_CHUNK - ex->zs.avail_out,
414 &ex->crc, &ex->produced, ex->e->uncomp_size);
415 if (rc != XRDC_ZIP_OK) {
416 return rc;
417 }
418 if (zr == Z_STREAM_END) {
419 return XRDC_ZIP_OK;
420 }
421 } while (ex->zs.avail_in > 0 || ex->zs.avail_out == 0);
422 return XRDC_ZIP_OK;
423}

426/*
427 * WHAT: Read the member's compressed extent chunk-by-chunk and route each chunk
428 * to the STORE copy or the DEFLATE inflater.
429 * WHY: Separates the archive-read driver from buffer setup and final verify so
430 * each is a single responsibility (§8.1); STORE vs DEFLATE dispatch is one
431 * branch instead of being tangled with allocation and teardown.
432 * HOW: Reads up to CHUNK bytes per iteration via read_exact (bounds-checked in
433 * the kernel), advancing the read cursor, until comp_size is consumed.
434 * Method 0 copies straight to the sink; method 8 hands the chunk to
435 * zip_extract_inflate_chunk. Byte order and CRC accumulation unchanged.
436 */
437static int
438zip_extract_stream(zip_extract_t *ex)
439{
440 uint64_t remaining = ex->e->comp_size;
441
442 while (remaining > 0) {
443 size_t want = (remaining < ZIP_EXTRACT_CHUNK)
444 ? (size_t) remaining : ZIP_EXTRACT_CHUNK;
445 int rc;
446
447 rc = read_exact(ex->pread, ex->pread_ctx, ex->data_off,
448 ex->e->archive_size, ex->inbuf, want);
449 if (rc != XRDC_ZIP_OK) {
450 return rc;
451 }
452 ex->data_off += want;
453 remaining -= want;
454
455 if (ex->e->method == 0) { /* STORE: raw copy */
456 rc = sink_output(ex->sink, ex->sink_ctx, ex->inbuf, want,
457 &ex->crc, &ex->produced, ex->e->uncomp_size);
458 } else { /* DEFLATE */
459 rc = zip_extract_inflate_chunk(ex, want);
460 }
461 if (rc != XRDC_ZIP_OK) {
462 return rc;
463 }
464 }
465 return XRDC_ZIP_OK;
466}

469/*
470 * WHAT: Confirm the extracted output matches the entry's declared size and CRC.
471 * WHY: Final integrity gate kept separate from the streaming loop (§8.3, pure).
472 * HOW: produced != uncomp_size ⇒ truncated/over-run (EBADF); CRC mismatch ⇒ ECRC.
473 */
474static int
475zip_extract_verify(const zip_extract_t *ex)
476{
477 if (ex->produced != ex->e->uncomp_size) {
478 return XRDC_ZIP_EBADF; /* short/over: truncated or corrupt */
479 }
480 if ((uint32_t) ex->crc != ex->e->crc32) {
481 return XRDC_ZIP_ECRC;
482 }
483 return XRDC_ZIP_OK;
484}

487int
488brix_zip_member_extract(brix_zip_pread_fn pread, void *ctx,
489 const brix_zip_entry *e,
490 brix_zip_sink_fn sink, void *sink_ctx)
491{
492 zip_extract_t ex = {0};
493 int rc;
494
495 if (e->method != 0 && e->method != 8) {
496 return XRDC_ZIP_EMETHOD;
497 }
498 if (e->uncomp_size > XRDC_ZIP_MAX_OUTPUT) {
499 return XRDC_ZIP_EBOMB;
500 }
501
502 ex.e = e;
503 ex.pread = pread;
504 ex.pread_ctx = ctx;
505 ex.sink = sink;
506 ex.sink_ctx = sink_ctx;
507 ex.crc = crc32(0L, Z_NULL, 0);
508
509 rc = member_data_offset(pread, ctx, e->archive_size, e, &ex.data_off);
510 if (rc != XRDC_ZIP_OK) {
511 return rc;
512 }
513
514 rc = zip_extract_setup(&ex);
515 if (rc != XRDC_ZIP_OK) {
516 return rc;
517 }
518
519 rc = zip_extract_stream(&ex);
520 zip_extract_teardown(&ex);
521 if (rc != XRDC_ZIP_OK) {
522 return rc;
523 }
524 return zip_extract_verify(&ex);
525}
