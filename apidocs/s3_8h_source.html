1/*
2 * s3.h — internal types for the nginx S3-compatible object storage module.
3 *
4 * Implements the subset of the AWS S3 REST API used by XrdClS3 (the XRootD
5 * S3 client plugin). Clients talk path-style URLs:
6 *
7 * GET /bucket/key → GetObject (file download)
8 * GET /bucket/?list-type=2 → ListObjectsV2 (directory listing)
9 * HEAD /bucket/key → HeadObject
10 * PUT /bucket/key → PutObject
11 * DELETE /bucket/key → DeleteObject
12 *
13 * Authentication is AWS Signature Version 4 (HMAC-SHA256), optional.
14 * If brix_s3_access_key is not configured, anonymous access is allowed.
15 *
16 * Directory sentinels: XrdClS3 marks directories by creating a zero-byte
17 * object named ".xrdcls3.dirsentinel". On a PUT of a sentinel the module
18 * creates the parent directory on disk. Sentinels are omitted from listings.
19 */
20/*
21 * ============================================================
22 * WHAT: Internal types and declarations for the S3-compatible module.
23 * ============================================================
24 *
25 * This header is the API boundary between module.c (the nginx module entry)
26 * point) and every S3 operation file (handler, get, put, list, multipart,
27 * copy, delete_objects). All internal sub-handlers, metrics functions,
28 * auth helpers, and utility declarations live here.
29 *
30 * The config struct ngx_http_s3_loc_conf_t carries per-location settings:
31 * enable flag, filesystem root, bucket name to strip, SigV4 credentials,
32 * allow_write gate, max_keys pagination limit, and the resolved canonical
33 * root path.
34 * ============================================================
35 */

179/*
180 * XML_APPEND / XML_APPEND_ELEM — flat-buffer XML building macros.
181 *
182 * Both reference local variables `xml` (u_char *), `xml_len` (size_t),
183 * and `xml_capacity` (size_t) that must be in scope at each call site.
184 * On overflow or encode error the macro returns NGX_HTTP_INTERNAL_SERVER_ERROR.
185 */
186#define XML_APPEND(fmt, ...) do { \
187 int _xml_n = snprintf((char *) xml + xml_len, xml_capacity - xml_len, \
188 fmt, ##__VA_ARGS__); \
189 if (_xml_n < 0 || (size_t) _xml_n >= xml_capacity - xml_len) { \
190 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]); \
191 return NGX_HTTP_INTERNAL_SERVER_ERROR; \
192 } \
193 xml_len += (size_t) _xml_n; \
194} while (0)
195
196#define XML_APPEND_ELEM(name, value, value_len) do { \
197 size_t _xml_written; \
198 if (brix_xml_write_text_element((name), \
199 (const unsigned char *)(value), (value_len), \
200 BRIX_XML_ESCAPE_APOS_ENTITY, \
201 xml + xml_len, xml_capacity - xml_len, \
202 &_xml_written) != 0) \
203 { \
204 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]); \
205 return NGX_HTTP_INTERNAL_SERVER_ERROR; \
206 } \
207 xml_len += _xml_written; \
208} while (0)
