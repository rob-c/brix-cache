4 * buffers_scratch.c — per-connection scratch- and read-buffer lifecycle for the
5 * synchronous read paths and nginx thread-pool AIO completions.
6 *
7 * Split from buffers.c (phase-79 file-size cap): this file owns the reusable
8 * heap-backed buffer machinery — the grow-on-demand pool scratch, the
9 * per-in-flight read pool acquire/release, and the between-request trim — while
10 * buffers.c / buffers_sendfile.c own the response-chain builders that consume
11 * those buffers. Keeping the buffer lifecycle separate from chain assembly keeps
12 * each file focused and under the 500-line cap.

15/*
16 * brix_get_pool_scratch — return a reusable scratch buffer from the
17 * connection pool, growing it only when the current allocation is too small.
18 *
19 * The caller owns one buffer slot (*slot / *slot_size) anchored in the
20 * connection pool. On first call *slot is NULL and ngx_palloc allocates it.
21 * On subsequent calls, if *slot_size >= need, the existing buffer is returned
22 * without a new allocation — critical for preventing unbounded pool growth
23 * during long xrdcp sessions.
24 *
25 * When the buffer must grow, the old one is freed with ngx_pfree (which is a
26 * no-op if the pool does not track the block, so it is always safe to call).
27 *
28 * NOTE: need==0 is clamped to 1 to satisfy ngx_palloc's precondition.
29 */
30u_char *
31brix_get_pool_scratch(ngx_pool_t *pool, u_char **slot, size_t *slot_size,
32 size_t need)
33{
34 u_char *p;
35
36 if (need == 0) {
37 need = 1;
38 }
39
40 if (*slot != NULL && *slot_size >= need) {
41 return *slot;
42 }
43
44 /*
45 * Phase 31: scratch buffers are raw heap allocations (ngx_alloc/ngx_free),
46 * NOT pool allocations. The earlier pool-backed version corrupted memory
47 * when brix_trim_scratch() freed and re-grew rd.read_scratch: ngx_pfree/
48 * ngx_palloc churn nginx's pool large-allocation list while stale pointers
49 * (the reused read_aio_task->databuf, read_fast_body_buf) still referenced
50 * the old block — a use-after-free triggered by a large kXR_read followed by
51 * a large kXR_readv. Raw heap allocation has no such pool lifecycle: the
52 * ctx owns the buffer and frees it explicitly on disconnect (mirroring how
53 * payload_buf is handled in src/connection/recv.c).
54 */
55 p = ngx_alloc(need, pool->log);
56 if (p == NULL) {
57 return NULL;
58 }
59
60 if (*slot != NULL) {
61 ngx_free(*slot);
62 }
63
64 *slot = p;
65 *slot_size = need;
66 return p;
67}

70/*
71 * brix_release_read_buffer — return a response data buffer to the pool,
72 * unless it is one of the reusable per-connection scratch slots.
73 *
74 * The scratch slots (rd.read_scratch / rd.read_hdr_scratch / rd.write_scratch) are
75 * long-lived raw heap allocations (see brix_get_pool_scratch) that must NOT
76 * be freed on every response — they are reused across requests and freed once
77 * at disconnect. Any OTHER buffer reaching here (e.g. a dirlist response) is a
78 * single-request ngx_palloc from c->pool and is returned with ngx_pfree.
79 *
80 * Called from brix_release_pending_buffer() in write_helpers.c.
81 */
82void
83brix_release_read_buffer(brix_ctx_t *ctx, ngx_connection_t *c, u_char *buf)
84{
85 ngx_uint_t i;
86
87 if (buf == NULL) {
88 return;
89 }
90
91 if (buf == ctx->rd.read_scratch || buf == ctx->rd.read_hdr_scratch
92 || buf == ctx->rd.write_scratch || buf == ctx->rd.cmp_scratch)
93 {
94 return;
95 }
96
97 /*
98 * Per-in-flight read-pool buffer (read pipelining): return the slot to the
99 * pool rather than freeing it — the buffer is reused for the next read and
100 * freed once at disconnect. Idempotent: a slot already marked free is a
101 * no-op, so the queued-then-error double-release path cannot underflow
102 * rd_inflight.
103 */
104 for (i = 0; i < ctx->out.pipeline_depth; i++) {
105 if (ctx->rd.pool[i].buf == buf) {
106 if (ctx->rd.pool[i].in_use) {
107 ctx->rd.pool[i].in_use = 0;
108 if (ctx->rd.inflight > 0) {
109 ctx->rd.inflight--;
110 }
111 }
112 return;
113 }
114 }
115
116 (void) ngx_pfree(c->pool, buf);
117}

119/*
120 * brix_acquire_read_buffer — borrow a per-in-flight data buffer from the
121 * connection's read pool (rd_pool), growing the chosen slot to >= need bytes.
122 *
123 * Unlike the single shared rd.read_scratch, each in-flight read gets its OWN buffer,
124 * so a memory-backed (userspace-TLS) read response can stay queued in the out_ring
125 * and drain while the recv loop already issues the NEXT read into a different
126 * buffer — i.e. memory reads can pipeline. The slot is returned to the pool by
127 * brix_release_read_buffer() when its out_ring response drains; the buffer is
128 * freed once at disconnect (raw heap, Phase-31 discipline).
129 *
130 * The recv loop bounds in-flight reads at pipeline_depth (== rd_pool slot count),
131 * so a free slot always exists on the hot path; NULL is returned only on OOM.
132 */
133u_char *
134brix_acquire_read_buffer(brix_ctx_t *ctx, ngx_connection_t *c, size_t need)
135{
136 ngx_uint_t i;
137
138 if (need == 0) {
139 need = 1;
140 }
141
142 for (i = 0; i < ctx->out.pipeline_depth; i++) {
143 if (ctx->rd.pool[i].in_use) {
144 continue;
145 }
146 if (ctx->rd.pool[i].size < need) {
147 u_char *p = ngx_alloc(need, c->log);
148 if (p == NULL) {
149 return NULL;
150 }
151 if (ctx->rd.pool[i].buf != NULL) {
152 ngx_free(ctx->rd.pool[i].buf);
153 }
154 ctx->rd.pool[i].buf = p;
155 ctx->rd.pool[i].size = need;
156 }
157 ctx->rd.pool[i].in_use = 1;
158 ctx->rd.inflight++;
159 return ctx->rd.pool[i].buf;
160 }
161
162 return NULL; /* pool exhausted — should not happen given the recv bound */
163}

165/*
166 * brix_trim_scratch — shrink the per-session transfer scratch buffers back to
167 * BRIX_READ_WINDOW once a large request has fully drained (Phase 31).
168 *
169 * rd.read_scratch and rd.write_scratch grow to the largest read / pgwrite the session
170 * has served and are then kept for reuse. Without trimming, a single 64 MiB
171 * read pins ~64 MiB of resident heap for the entire connection lifetime even
172 * while idle — the dominant memory-scaling term for a TLS gateway. This trims
173 * them back to the streaming window so the steady-state per-connection heap is
174 * ~window, not ~request-max.
175 *
176 * MUST be called only when the connection is between requests (state
177 * XRD_ST_REQ_HEADER with nothing buffered), so that no in-flight response chain
178 * still points into these buffers. The recv loop calls it at the top of a fresh
179 * request. Buffers at or below BRIX_SCRATCH_TRIM_THRESHOLD are left untouched
180 * (hysteresis avoids realloc thrash on sessions that oscillate near the window).
181 *
182 * rd.read_hdr_scratch (per-chunk wire headers) is tiny and never trimmed.
183 * payload_buf has detach semantics owned by the write path and is trimmed there.
184 */
185static void
186brix_trim_one(ngx_pool_t *pool, u_char **slot, size_t *slot_size)
187{
188 u_char *p;
189
190 if (*slot == NULL || *slot_size <= BRIX_SCRATCH_TRIM_THRESHOLD) {
191 return;
192 }
193
194 /* Raw heap free/alloc — see brix_get_pool_scratch for why these buffers
195 * are not pool-backed. This is what makes the trim safe to run. */
196 ngx_free(*slot);
197
198 p = ngx_alloc(BRIX_READ_WINDOW, pool->log);
199 if (p == NULL) {
200 /* Could not re-seat a warm buffer; drop it so the next request
201 * allocates fresh at exactly the size it needs. */
202 *slot = NULL;
203 *slot_size = 0;
204 return;
205 }
206
207 *slot = p;
208 *slot_size = BRIX_READ_WINDOW;
209}
