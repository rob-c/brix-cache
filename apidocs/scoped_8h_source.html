1/*
2 * scoped.h — Phase 27 W3: NULL-safe destroyers for external handles.
3 *
4 * WHAT: A small vocabulary of NULL-safe destroyers for OpenSSL handles (and a
5 * jansson refcount helper) so that long, branchy error paths driven by
6 * attacker-controlled input free *every* handle on *every* exit path without
7 * relying on a reviewer to spot a missed `*_free`.
8 *
9 * WHY: The TPC/GSI exchange and the EVP-using files build several handles over
10 * remote-controlled buckets. A single malformed bucket that takes an early
11 * exit before some `*_free` leaks an OpenSSL handle (and its heap) on every
12 * failed handshake — a slow OOM an external origin can drive. The structural
13 * fix (per docs/09-developer-guide/coding-standards.md §4, which forbids goto)
14 * is to confine the build to a worker that uses flat early returns and let the
15 * wrapper own one cleanup site that frees every handle unconditionally. NULL-
16 * safe destroyers make that one cleanup a flat, reviewable list.
17 *
18 * THE PATTERN (goto-free; see also CODE STYLE):
19 *
20 * static ngx_int_t build_inner(..., EVP_PKEY **pk_out, BIO **bio_out)
21 * {
22 * *pk_out = ...; if (*pk_out == NULL) return NGX_ERROR;
23 * *bio_out = ...; if (*bio_out == NULL) return NGX_ERROR;
24 * ... work ...
25 * return NGX_OK;
26 * }
27 *
28 * ngx_int_t build(...)
29 * {
30 * EVP_PKEY *pk = NULL;
31 * BIO *bio = NULL;
32 * ngx_int_t rc = build_inner(..., &pk, &bio);
33 *
34 * brix_evp_pkey_free(pk); // wrapper owns the single cleanup site;
35 * brix_bio_free(bio); // every destroyer tolerates NULL
36 * return rc;
37 * }
38 *
39 * Every destroyer below tolerates NULL (OpenSSL's own *_free already do, but a
40 * uniform set keeps the cleanup list flat and usable as cleanup callbacks).
41 *
42 * ---------------------------------------------------------------------------
43 * JANSSON OWNERSHIP CHEATSHEET (W3) — leaks and double-decrefs come from
44 * confusing borrowed vs owned references.
45 *
46 * OWNED (you must json_decref, unless you hand it to a stealing setter):
47 * json_object(), json_array(), json_string(), json_integer(), json_true(),
48 * json_loads()/json_loadb(), json_pack(), json_deep_copy(), json_incref().
49 * BORROWED (do NOT decref; valid only while the parent is alive):
50 * json_object_get(), json_array_get(), json_object_iter_value().
51 * STEALING (takes ownership of the value — do NOT decref it afterwards;
52 * still decref the CONTAINER once):
53 * json_object_set_new(), json_array_append_new(), json_array_insert_new().
54 * NON-STEALING (incref the value — you still own your reference):
55 * json_object_set(), json_array_append().
56 *
57 * Rule of thumb: if the name ends in `_new`, the call consumed your reference.
58 * Build with `*_set_new`/`*_append_new` + a single decref of the root.
59 * ---------------------------------------------------------------------------
60 */

74/* Free an EVP_PKEY (NULL-safe, no-op on NULL). Caller's pointer is not nulled;
75 * the slot stays dangling, so use only at a one-shot cleanup label. */
76static ngx_inline void brix_evp_pkey_free(EVP_PKEY *p)
77{ if (p) { EVP_PKEY_free(p); } }
