1/*
2 * stage_waiter.c — async stage/recall completion waiter table.
3 *
4 * See stage_waiter.h for the model. FRM-dissolution re-home of src/frm/waiter.c.
5 * The table is slab-allocated (brix_shm_table_alloc) so it does not clobber the
6 * slab-pool header — same contract as every other nginx-xrootd SHM zone (see
7 * src/compat/shm_slots.c). Delivery rebuilds the real open response by replaying
8 * brix_open_resolved_file() on the parked connection with ctx->prepare.stage_async_active
9 * set, so the open-OK body (fhandle) is produced by the normal machinery and
10 * emitted via kXR_attn(asynresp) on the saved streamid.
11 */

66static ngx_int_t
67stage_waiter_init_zone(ngx_shm_zone_t *shm_zone, void *data)
68{
69 stage_waiter_table_t *tbl;
70 ngx_flag_t fresh;
71 ngx_uint_t cap = stage_waiter_slots_req;
72
73 if (cap == 0) { cap = 256; }
74
75 tbl = brix_shm_table_alloc(shm_zone, data,
76 sizeof(stage_waiter_table_t)
77 + (size_t) (cap - 1) * sizeof(stage_waiter_t),
78 &stage_waiter_mtx, &fresh);
79 if (tbl == NULL) {
80 return NGX_ERROR;
81 }
82 if (fresh) {
83 tbl->capacity = cap;
84 /* helper already zeroed the table */
85 }
86 return NGX_OK;

91{

112ngx_int_t
113brix_stage_waiter_add(const char *reqid, uint16_t options,
114 const u_char client_streamid[2], int conn_fd,
115 ngx_atomic_uint_t conn_number, ngx_pid_t worker_pid,
116 ngx_msec_t timeout_ms)
117{
118 stage_waiter_table_t *tbl = stage_waiter_table();
119 ngx_uint_t i, free_slot;
120
121 if (tbl == NULL || reqid == NULL) {
122 return NGX_ERROR;
123 }
124 ngx_shmtx_lock(&stage_waiter_mtx);
125
126 /* phase79-fp: tbl NULL-checked at entry; analyzer drops the guard across ngx_shmtx_lock */
127 free_slot = tbl->capacity;
128 for (i = 0; i < tbl->capacity; i++) {
129 stage_waiter_t *s = &tbl->slots[i];
130 if (!s->in_use) {
131 brix_shm_remember_free_slot(&free_slot, tbl->capacity, i);
132 continue;
133 }
134 if (brix_shm_slot_expired(ngx_current_msec, s->expires)) {
135 s->in_use = 0;
136 brix_shm_remember_free_slot(&free_slot, tbl->capacity, i);
137 }
138 }
139 if (free_slot == tbl->capacity) {
140 ngx_shmtx_unlock(&stage_waiter_mtx);
141 return NGX_AGAIN; /* full → caller falls back to wait */
142 }
143
144 {
145 stage_waiter_t *s = &tbl->slots[free_slot];
146 ngx_memzero(s, sizeof(*s));
147 ngx_cpystrn((u_char *) s->reqid, (u_char *) reqid, sizeof(s->reqid));
148 s->options = options;
149 s->client_streamid[0] = client_streamid[0];
150 s->client_streamid[1] = client_streamid[1];
151 s->conn_fd = conn_fd;
152 s->conn_number = conn_number;
153 s->worker_pid = worker_pid;
154 s->expires = ngx_current_msec + timeout_ms;
155 s->ready = 0;
156 s->in_use = 1;
157 }
158 ngx_shmtx_unlock(&stage_waiter_mtx);
159 return NGX_OK;
160}

174 c = ngx_cycle->files[fd];

186/*
187 * Deliver one completed recall to a parked connection (a local copy of the slot
188 * row; the slot is already freed). Validates liveness, then either replays the
189 * open (code 0) or pushes an error — both wrapped in kXR_attn(asynresp) on the
190 * saved streamid — and resumes the connection.
191 */
192static void
193stage_waiter_deliver_one(const stage_waiter_t *w)
194{
195 ngx_connection_t *c;
196 ngx_stream_session_t *s;
197 brix_ctx_t *ctx;
198 ngx_stream_brix_srv_conf_t *conf;
199 brix_stage_registry_t *q = brix_stage_registry_singleton();
200 brix_stage_request_t rec;
201
202 c = stage_waiter_find_conn(w->conn_fd);
203 if (c == NULL || c->number != w->conn_number || c->destroyed) {
204 return; /* fd recycled / gone */
205 }
206 s = c->data;
207 if (s == NULL) {
208 return;
209 }
210 ctx = ngx_stream_get_module_ctx(s, ngx_stream_brix_module);
211 if (ctx == NULL || ctx->destroyed || ctx->state != XRD_ST_WAITING_FRM) {
212 return; /* not actually parked here */
213 }
214 conf = ngx_stream_get_module_srv_conf(s, ngx_stream_brix_module);
215
216 if (c->read->timer_set) {
217 ngx_del_timer(c->read);
218 }
219 ctx->state = XRD_ST_REQ_HEADER; /* un-suspend before we re-drive */
220
221 if (w->code == 0 && q != NULL
222 && brix_stage_request_get(q, w->reqid, &rec, c->log) == NGX_OK)
223 {
224 /* Replay the open: the file is now resident, so the normal machinery
225 * allocates the fhandle and builds the open-OK body; the stage_async flag
226 * makes the emit go out as kXR_attn(asynresp) on the saved streamid. */
227 ctx->prepare.stage_async_active = 1;
228 ctx->prepare.stage_async_streamid[0] = w->client_streamid[0];
229 ctx->prepare.stage_async_streamid[1] = w->client_streamid[1];
230 brix_open_request_t oreq = {
231 .resolved = rec.lfn,
232 .options = w->options,
233 .mode_bits = 0,
234 .is_write = 0,
235 .codec = 0,
236 };
237 (void) brix_open_resolved_file(ctx, c, conf, &oreq);
238 ctx->prepare.stage_async_active = 0;
239 } else {
240 /* Recall failed (or the record vanished): deliver a hard error so the
241 * client stops waiting. Body = errnum(4, big-endian) + message. */
242 u_char body[64];
243 const char *msg = "file is offline (recall failed)";
244 size_t mlen = ngx_strlen(msg);
245 uint32_t errnum = htonl((uint32_t) kXR_FSError);
246
247 if (mlen > sizeof(body) - 5) { mlen = sizeof(body) - 5; }
248 ngx_memcpy(body, &errnum, 4);
249 ngx_memcpy(body + 4, msg, mlen);
250 body[4 + mlen] = '\0';
251 (void) brix_send_attn_asynresp(ctx, c, w->client_streamid,
252 (uint16_t) kXR_error,
253 body, (uint32_t) (4 + mlen + 1));
254 }
255
256 /* (asynresp metric re-homes to the stage metrics in Task 5) */
257 brix_schedule_read_resume(c);
258}

260void
261brix_stage_waiter_deliver(const char *reqid, int code)
262{
263 stage_waiter_table_t *tbl = stage_waiter_table();
264 ngx_uint_t i;
265
266 if (tbl == NULL || reqid == NULL) {
267 return;
268 }
269 /* Mark every matching waiter ready; this worker's are then drained below,
270 * other workers' rows are drained by their own poll. */
271 ngx_shmtx_lock(&stage_waiter_mtx);
272 /* phase79-fp: tbl NULL-checked at entry; analyzer drops the guard across ngx_shmtx_lock */
273 for (i = 0; i < tbl->capacity; i++) {
274 stage_waiter_t *s = &tbl->slots[i];
275 if (s->in_use && ngx_strcmp(s->reqid, reqid) == 0) {
276 s->ready = 1;
277 s->code = code;
278 }
279 }
280 ngx_shmtx_unlock(&stage_waiter_mtx);
281
282 brix_stage_waiter_poll_local();
283}

285void
286brix_stage_waiter_poll_local(void)
287{
288 stage_waiter_table_t *tbl = stage_waiter_table();
289 stage_waiter_t batch[16];
290 ngx_uint_t i, n;
291
292 if (tbl == NULL) {
293 return;
294 }
295 /*
296 * Claim this worker's ready rows (and reap any expired) under the lock, then
297 * deliver them OUTSIDE the lock (delivery queues wire bytes / touches the
298 * event loop, which must never run under an SHM mutex). Bounded per tick;
299 * remaining rows are picked up on the next scheduler tick.
300 */
301 for ( ;; ) {
302 n = 0;
303 ngx_shmtx_lock(&stage_waiter_mtx);
304 for (i = 0; i < tbl->capacity && n < 16; i++) {
305 stage_waiter_t *s = &tbl->slots[i];
306 if (!s->in_use) {
307 continue;
308 }
309 if (s->ready && s->worker_pid == ngx_pid) {
310 batch[n++] = *s; /* copy out */
311 s->in_use = 0; /* claim it */
312 continue;
313 }
314 if (brix_shm_slot_expired(ngx_current_msec, s->expires)) {
315 s->in_use = 0; /* reap stale */
316 }
317 }
318 ngx_shmtx_unlock(&stage_waiter_mtx);
319
320 for (i = 0; i < n; i++) {
321 stage_waiter_deliver_one(&batch[i]);
322 }
323 if (n < 16) {
324 break; /* drained */
325 }
326 }
327}
