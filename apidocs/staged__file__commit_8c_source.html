1/*
2 * staged_file_commit.c — Cross-device / cross-backend staged-commit publish.
3 *
4 * WHAT: brix_commit_staged() and its helpers — publish a completed staged
5 * partial onto its final path when a plain rename(2) cannot: a
6 * cross-device fast-cache stage dir (EXDEV → copy + fsync + rename) or a
7 * non-POSIX final export (upload INTO the driver's staged write→commit
8 * state machine).
9 *
10 * WHY: Split from staged_file.c (phase-79 file-size split). The same-filesystem
11 * rename lifecycle (open/commit/abort) stays in staged_file.c; this file
12 * owns the heavier byte-moving publish paths that reach the storage-driver
13 * (SD) seam and the per-export backend registry. Isolating them keeps each
14 * file single-concept and under the size budget.
15 *
16 * HOW: brix_commit_staged() flushes+chmods the open fd (commit_flush_and_chmod),
17 * resolves the final export's backend, and dispatches: non-POSIX backend →
18 * commit_staged_to_backend (cstb_open_source + cstb_pump_and_commit);
19 * same-FS → rename(2); EXDEV → commit_cross_device. All byte I/O routes
20 * through the SD seam (brix_sd_posix_wrap / driver vtable), never raw
21 * pread/pwrite here. Zero behavior change from the pre-split code.
22 */

36/* VFS↔VFS (backend↔backend) byte move: copy the whole source object into the
37 * destination object by reading through the SOURCE object's storage driver and
38 * writing through the DESTINATION object's driver. The two objects may live on
39 * different backends — this is the cross-mount staging commit, so it is exactly
40 * the place where one side could later be a block/object (S3) store while the
41 * other stays POSIX.
42 *
43 * Today both ends are POSIX (brix_sd_posix_wrap over a kernel fd), so the only
44 * raw pread/pwrite happen INSIDE the POSIX backend (src/fs/backend/sd_posix.c) —
45 * never here. When a stage or final mount becomes a non-POSIX backend, only how
46 * the object is obtained changes (that driver's open() instead of a bare-fd
47 * wrap); this positional copy loop is unchanged. 0 / NGX_ERROR (errno set). */
48static ngx_int_t
49stage_move_objects(brix_sd_obj_t *src, brix_sd_obj_t *dst)
50{
51 /* The mover now lives in the transfer engine (src/fs/xfer/xfer_mover_pump.c)
52 * as the canonical BRIX_XFER_MOVE_PUMP strategy; this thin wrapper keeps the
53 * two staged-commit callsites unchanged. */
54 return brix_xfer_pump_objects(src, dst);
55}

57/* commit_be_logical — the export-root-relative ("/sub/file", or "/" for the root)
58 * key form a non-POSIX backend's namespace expects, matching
59 * brix_vfs_export_relative_root used by every other driver-backed op. */
60static const char *
61commit_be_logical(const char *abs, const char *root)
62{
63 size_t rl = (root != NULL) ? strlen(root) : 0;
64
65 if (rl > 0 && strncmp(abs, root, rl) == 0) {
66 if (abs[rl] == '/') { return abs + rl; } /* "/sub/file" */
67 if (abs[rl] == '\0') { return "/"; } /* the export root itself */
68 }
69 return abs;
70}

72/*
73 * WHAT: Resolve the read side of a backend commit — reuse the caller's still-open
74 * staged fd, or open the partial O_RDONLY, and report the mode to publish.
75 *
76 * WHY: commit_staged_to_backend can be reached with the partial fd still held
77 * (synchronous close) or already closed (crash-recovery reap), and the
78 * ownership decision governs whether the read fd must be closed on exit. A
79 * single resolver keeps that decision out of the copy loop.
80 *
81 * HOW: If fd is valid, borrow it (*owned=0); else open stage_path O_RDONLY
82 * (*owned=1). Derive the publish mode from fstat (default 0644). Returns the
83 * read fd, or NGX_INVALID_FILE with errno set on open failure.
84 *
85 * Parameters:
86 * fd — caller's staged fd, or NGX_INVALID_FILE
87 * stage_path — partial path to open when fd is closed
88 * owned — output: 1 if the returned fd must be closed by the caller
89 * mode — output: file mode to publish into the backend
90 */
91static ngx_fd_t
92cstb_open_source(ngx_fd_t fd, const char *stage_path, int *owned, mode_t *mode)
93{
94 struct stat sb;
95 ngx_fd_t rfd;
96
97 *owned = 0;
98 *mode = 0644;
99
100 if (fd != NGX_INVALID_FILE) {
101 rfd = fd;
102 } else {
103 rfd = open(stage_path, O_RDONLY | O_CLOEXEC);
104 if (rfd < 0) {
105 return NGX_INVALID_FILE;
106 }
107 *owned = 1;
108 }
109
110 if (fstat(rfd, &sb) == 0) {
111 *mode = sb.st_mode & 07777;
112 }
113 return rfd;
114}

116/*
117 * WHAT: Stream a POSIX-read partial (via the SD seam) into an open driver-staged
118 * blob, whole-object, and atomically publish it.
119 *
120 * WHY: The read→write pump is the body of the cross-backend commit; isolating it
121 * keeps commit_staged_to_backend a flat acquire/pump/cleanup orchestrator
122 * and confines the per-chunk error handling to one place.
123 *
124 * HOW: Wrap rfd as a POSIX source object, positional-read 64 KiB chunks, and
125 * staged_write each fully before advancing the offset. On any read/write
126 * error, or a failing staged_commit, abort the driver blob (releasing it)
127 * and return NGX_ERROR with the originating errno preserved. On success the
128 * blob is committed (partial NOT unlinked here — the caller owns that).
129 *
130 * Parameters:
131 * rfd — open read fd for the staged partial (positional reads)
132 * dst — destination backend instance (driver vtable)
133 * st — open driver-staged blob to fill and commit
134 */
135static ngx_int_t
136cstb_pump_and_commit(ngx_fd_t rfd, brix_sd_instance_t *dst,
137 brix_sd_staged_t *st)
138{
139 brix_sd_obj_t src_obj;
140 off_t off = 0;
141
142 brix_sd_posix_wrap(&src_obj, rfd); /* read the partial via the SD seam */
143 for ( ;; ) {
144 char buf[65536];
145 ssize_t r = src_obj.driver->pread(&src_obj, buf, sizeof(buf), off);
146 ssize_t w = 0;
147
148 if (r < 0) {
149 int e = errno;
150 dst->driver->staged_abort(st);
151 errno = e;
152 return NGX_ERROR;
153 }
154 if (r == 0) {
155 break; /* EOF: whole partial consumed */
156 }
157 while (w < r) {
158 ssize_t k = dst->driver->staged_write(st, buf + w,
159 (size_t) (r - w), off + w);
160 if (k <= 0) {
161 int e = errno;
162 dst->driver->staged_abort(st);
163 errno = e ? e : EIO;
164 return NGX_ERROR;
165 }
166 w += k;
167 }
168 off += r;
169 }
170
171 /* Atomic publish. On failure staged_commit leaves the handle valid, so abort
172 * to release it; KEEP the POSIX partial so a resume retry can republish. */
173 if (dst->driver->staged_commit(st, 0 /* replace allowed */) != NGX_OK) {
174 int e = errno;
175 dst->driver->staged_abort(st);
176 errno = e ? e : EIO;
177 return NGX_ERROR;
178 }
179 return NGX_OK;
180}

182/*
183 * commit_staged_to_backend — publish a POSIX-staged partial INTO a non-POSIX
184 * storage backend (e.g. pblock) by streaming it through that driver's staged
185 * write→commit state machine. This is the cross-FILESYSTEM atomic commit that
186 * rename(2) cannot do: the partial lives on an independent POSIX staging mount
187 * (brix_stage_dir) and the final export is a different, driver-owned namespace,
188 * so we read the partial through the (POSIX) source backend and write it through
189 * the destination driver's staged_write, then staged_commit (which publishes
190 * atomically — for pblock, a single catalog row insert pointing at the freshly
191 * written blocks). On success the POSIX partial is unlinked; on failure it is
192 * KEPT (a resume client can retry the close) and the driver's staged blob is
193 * aborted. Returns NGX_OK / NGX_ERROR (errno set).
194 */
195static ngx_int_t
196commit_staged_to_backend(ngx_fd_t fd, const char *stage_path,
197 const char *final_path, brix_sd_instance_t *dst, const char *root_canon,
198 ngx_log_t *log)
199{
200 const char *logical = commit_be_logical(final_path, root_canon);
201 brix_sd_staged_t *st;
202 mode_t mode = 0644;
203 int rfd, owned = 0, serr = 0;
204 ngx_int_t rc;
205
206 (void) log;
207
208 if (dst->driver->staged_open == NULL || dst->driver->staged_write == NULL
209 || dst->driver->staged_commit == NULL
210 || dst->driver->staged_abort == NULL)
211 {
212 errno = ENOTSUP;
213 return NGX_ERROR;
214 }
215
216 /* Source: the open staged partial if we still hold it, else open it O_RDONLY.
217 * Reads are positional (pread), so the fd's file position is irrelevant. */
218 rfd = cstb_open_source(fd, stage_path, &owned, &mode);
219 if (rfd == NGX_INVALID_FILE) {
220 return NGX_ERROR;
221 }
222
223 st = dst->driver->staged_open(dst, logical, mode, &serr);
224 if (st == NULL) {
225 if (owned) { (void) close(rfd); }
226 errno = serr ? serr : EIO;
227 return NGX_ERROR;
228 }
229
230 rc = cstb_pump_and_commit(rfd, dst, st); /* aborts st itself on failure */
231 if (owned) { (void) close(rfd); }
232 if (rc != NGX_OK) {
233 return NGX_ERROR; /* errno set by the pump */
234 }
235
236 (void) unlink(stage_path); /* published → drop the partial */
237 return NGX_OK;
238}

240/*
241 * WHAT: Make the still-open staged fd durable and publish the caller's intended
242 * mode on it, just before any commit path reads it.
243 *
244 * WHY: close() alone does not flush, and a crash / power loss / ENOSPC mid-write
245 * must not expose a torn object — so the data must be fsync'd BEFORE the
246 * commit publishes it. The temp is written 0600 (private); the final mode is
247 * restored here so every downstream commit path derives the object's mode
248 * from the fd.
249 *
250 * HOW: Wrap fd as a POSIX object and fsync via the backend; a failing fsync means
251 * "not durable" → NGX_ERROR (do not publish). Then fchmod to final_mode when
252 * non-zero (0 means "leave the temp's current mode", e.g. root:// POSC temps
253 * that already carry the client's create mode). No-op for a closed fd.
254 *
255 * Parameters:
256 * fd — open staged fd, or NGX_INVALID_FILE (crash-recovery reap)
257 * final_mode — mode to publish, or 0 to keep the temp's mode
258 */
259static ngx_int_t
260commit_flush_and_chmod(ngx_fd_t fd, mode_t final_mode)
261{
262 brix_sd_obj_t sobj;
263
264 if (fd == NGX_INVALID_FILE) {
265 return NGX_OK;
266 }
267
268 brix_sd_posix_wrap(&sobj, fd); /* durability flush via the backend */
269 if (sobj.driver->fsync(&sobj) != NGX_OK) {
270 return NGX_ERROR; /* not durable — do not publish */
271 }
272 /* SECURITY: publish the caller's intended mode on the still-open fd before any
273 * commit path reads it. */
274 if (final_mode != 0) {
275 (void) fchmod(fd, final_mode);
276 }
277 return NGX_OK;
278}

280/*
281 * WHAT: Cross-device commit — copy the staged partial to a temp ON THE FINAL
282 * filesystem, fsync it, atomically rename it onto the final path, and drop
283 * the staged copy.
284 *
285 * WHY: When the staged partial sits on a configured fast-cache device different
286 * from the storage (e.g. CEPHFS), rename(2) returns EXDEV and cannot publish
287 * it, so the bytes must be moved before the (now same-FS) atomic rename.
288 *
289 * HOW: Derive a temp adjacent to final_path, open the partial O_RDONLY, create
290 * the temp preserving the partial's mode, VFS↔VFS move + fsync, close, then
291 * rename temp→final and unlink the staged copy. Every error path unlinks the
292 * temp and restores the originating errno.
293 *
294 * Parameters:
295 * stage_path — the staged partial (cross-device source)
296 * final_path — destination path (drives the adjacent temp + final rename)
297 */
298static ngx_int_t
299commit_cross_device(const char *stage_path, const char *final_path)
300{
301 char tmp[PATH_MAX];
302 int rfd, dfd, e;
303 struct stat sb;
304 mode_t mode;
305
306 if (brix_make_tmp_path(final_path, tmp, sizeof(tmp)) != NGX_OK) {
307 errno = ENAMETOOLONG;
308 return NGX_ERROR;
309 }
310 rfd = open(stage_path, O_RDONLY | O_CLOEXEC);
311 if (rfd < 0) {
312 return NGX_ERROR;
313 }
314 /* Preserve the staged file's mode on the committed object. */
315 mode = (fstat(rfd, &sb) == 0) ? (sb.st_mode & 07777) : 0644;
316 dfd = open(tmp, O_WRONLY | O_CREAT | O_EXCL | O_NOFOLLOW | O_CLOEXEC, mode);
317 if (dfd < 0) {
318 e = errno; close(rfd); errno = e; return NGX_ERROR;
319 }
320 {
321 /* VFS↔VFS move: stage object (source backend) → temp object (final
322 * backend). Both POSIX today; the loop is driver-mediated so a non-POSIX
323 * mount on either side needs no change here. */
324 brix_sd_obj_t src_obj, dst_obj;
325 brix_sd_posix_wrap(&src_obj, rfd);
326 brix_sd_posix_wrap(&dst_obj, dfd);
327 if (stage_move_objects(&src_obj, &dst_obj) != NGX_OK
328 || dst_obj.driver->fsync(&dst_obj) != NGX_OK) {
329 e = errno; close(rfd); close(dfd); (void) unlink(tmp); errno = e;
330 return NGX_ERROR;
331 }
332 }
333 close(rfd);
334 if (close(dfd) != 0) {
335 e = errno; (void) unlink(tmp); errno = e; return NGX_ERROR;
336 }
337 if (rename(tmp, final_path) != 0) {
338 e = errno; (void) unlink(tmp); errno = e; return NGX_ERROR;
339 }
340 (void) unlink(stage_path); /* drop the cache copy; commit succeeded */
341 return NGX_OK;
342}

344/*
345 * WHAT: Commit a staged file onto its final path — atomically and across
346 * filesystems. When the final export uses a NON-POSIX storage backend
347 * (e.g. pblock), the partial is uploaded INTO that backend via the driver's
348 * staged write→commit state machine (commit_staged_to_backend). Otherwise
349 * it tries rename(2) first (atomic, same-FS, the fast path); on
350 * EXDEV (the staged file is on a configured fast-cache device different from
351 * the storage, e.g. CEPHFS) it copies the data to a temp ON THE FINAL
352 * filesystem, fsync()s, atomically renames that temp onto the final path,
353 * and unlinks the staged copy. Either way a concurrent reader sees only the
354 * old object or the complete new one — never a partial.
355 *
356 * WHY: Upload staging on a fast device + final storage on a POSIX mount means the
357 * commit can be a cross-device move, which rename(2) cannot do. Centralising
358 * this lets both root:// (close) and WebDAV (PUT) commit identically.
359 *
360 * fd: the open staged-file descriptor (for the durability fsync before commit), or
361 * NGX_INVALID_FILE if already closed. Returns NGX_OK / NGX_ERROR (errno set).
362 */
363ngx_int_t
364brix_commit_staged(ngx_fd_t fd, const char *stage_path, const char *final_path,
365 mode_t final_mode, ngx_log_t *log)
366{
367 const char *be_root = NULL;
368 brix_sd_instance_t *dst;
369
370 if (commit_flush_and_chmod(fd, final_mode) != NGX_OK) {
371 return NGX_ERROR;
372 }
373
374 /* Non-POSIX final export (e.g. pblock): the staged POSIX partial lives on an
375 * independent staging mount and the final namespace is driver-owned, so
376 * rename(2) cannot publish it. Upload the partial INTO the backend via its
377 * staged write→commit state machine (a cross-filesystem atomic publish).
378 * POSIX exports (the common case) skip this and take the rename path below. */
379 dst = brix_vfs_backend_resolve_for_path(final_path, &be_root, log);
380 if (dst != NULL && dst->driver != NULL
381 && dst->driver != brix_sd_default_driver())
382 {
383 return commit_staged_to_backend(fd, stage_path, final_path, dst,
384 be_root, log);
385 }
386
387 if (rename(stage_path, final_path) == 0) {
388 return NGX_OK; /* same filesystem: atomic, zero-copy */
389 }
390 if (errno != EXDEV) {
391 return NGX_ERROR;
392 }
393
394 /* Cross-device commit: copy the staged data to a temp adjacent to (and on the
395 * same filesystem as) the final path, then atomically rename it into place. */
396 return commit_cross_device(stage_path, final_path);
397}
