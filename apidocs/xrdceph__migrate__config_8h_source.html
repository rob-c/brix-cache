1/*
2 * xrdceph_migrate_config.h — site-profile config file for the C++
3 * XrdCeph<->CephFS migration tools (shared by xrdceph_striper_migrate.cpp
4 * and xrdceph_cephfs_to_striper.cpp; format identical to the Python tools'
5 * pymigrate.common.load_tool_config).
6 *
7 * WHAT: A tiny header-only parser for the flat `key = value` profile passed
8 * via --config (or $XRDCEPH_MIGRATE_CONF): '#' comments (inline
9 * allowed), blank lines ignored, whitespace trimmed, empty value =
10 * unset. The key set is CLOSED — an unknown key is a hard error,
11 * because a typo'd pool name in a tool that can delete data must fail
12 * loudly, not silently fall back to a default.
13 *
14 * WHY: Operators define pools + connection identity once per site instead
15 * of retyping them per invocation; it also makes the client id and
16 * CephFS fs name configurable (both were hardcoded before).
17 *
18 * HOW: Header-only so the tools' single-file `g++ tool.cpp -l...` builds
19 * keep working. Precedence is applied by the caller:
20 * explicit CLI > config file > built-in default (see
21 * xrdceph_migrate_cfg_resolve).
22 *
23 * Recognised keys:
24 * striper_pool meta_pool data_pool conf client fs_name dest_prefix strip
25 */

40inline std::string
41trim(const std::string &s)
42{
43 size_t b = s.find_first_not_of(" \t\r\n");
44 if (b == std::string::npos) { return ""; }
45 size_t e = s.find_last_not_of(" \t\r\n");
46 return s.substr(b, e - b + 1);
47}

49inline bool
50known_key(const std::string &k)
51{
52 static const char *keys[] = { "striper_pool", "meta_pool", "data_pool",
53 "conf", "client", "fs_name",
54 "dest_prefix", "strip" };
55 for (const char *key : keys) { if (k == key) { return true; } }

61/* Parse `path` into *cfg. Returns true on success; on failure prints a
62 * one-line reason (with file:line) to stderr and returns false. */
63inline bool
64xrdceph_migrate_cfg_load(const std::string &path, xrdceph_migrate_cfg *cfg)
65{
66 using xrdceph_migrate_cfg_detail::trim;
67 using xrdceph_migrate_cfg_detail::known_key;
68
69 std::ifstream f(path);
70 if (!f) {
71 fprintf(stderr, "--config: cannot open %s\n", path.c_str());
72 return false;
73 }
74 std::string raw;
75 int lineno = 0;
76 while (std::getline(f, raw)) {
77 lineno++;
78 std::string line = raw;
79 size_t hash = line.find('#');
80 if (hash != std::string::npos) { line.erase(hash); }
81 line = trim(line);
82 if (line.empty()) { continue; }
83
84 size_t eq = line.find('=');
85 if (eq == std::string::npos) {
86 fprintf(stderr, "--config: %s:%d: expected 'key = value', "
87 "got \"%s\"\n", path.c_str(), lineno, raw.c_str());
88 return false;
89 }
90 std::string key = trim(line.substr(0, eq));
91 std::string val = trim(line.substr(eq + 1));
92 if (!known_key(key)) {
93 fprintf(stderr, "--config: %s:%d: unknown config key '%s' "
94 "(known: striper_pool, meta_pool, data_pool, conf, "
95 "client, fs_name, dest_prefix, strip)\n",
96 path.c_str(), lineno, key.c_str());
97 return false;
98 }
99 if (!val.empty()) { cfg->kv[key] = val; }
100 }
101 return true;
102}

104/* Precedence: explicit CLI (non-empty) > config file > built-in default. */
105inline std::string
106xrdceph_migrate_cfg_resolve(const std::string &cli_value,
107 const xrdceph_migrate_cfg &cfg,
108 const std::string &key,
109 const std::string &dflt = "")
110{
111 if (!cli_value.empty()) { return cli_value; }
112 auto it = cfg.kv.find(key);
113 if (it != cfg.kv.end()) { return it->second; }
114 return dflt;
115}
