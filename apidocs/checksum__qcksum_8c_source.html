16/*
17 * WHAT: kXR_Qcksum — compute single-file checksum (adler32/crc32c/md5/sha1/sha256) by path or open file handle.
18 * Dispatches to path-based handler (full security chain + confined open) or handle-based handler (already-open fd).
19 * Supports async execution via thread pool when configured; falls back to synchronous event-loop computation otherwise.
20 *
21 * WHY: Qcksum is the most common XRootD query opcode — clients need file integrity verification before transfer, after staging,
22 * and for checksum-based deduplication. Multiple algorithm support (adler32/crc32c/md5/sha1/sha256) covers legacy HEP tools
23 * (xrdcp uses adler32) alongside modern requirements (SHA256 for cloud storage). Async execution prevents blocking on large files.
24 *
25 * HOW: brix_query_cksum() routes by payload prefix byte: non-zero → path-based handler (cksum_path); zero or empty → handle-based
26 * handler (cksum_handle). cksum_path parses algo:path from wire, resolves path, checks authdb/VO/token scope, opens confined fd,
27 * then either posts thread task or computes sync. cksum_handle validates fhandle index, optionally extracts algo prefix, delegates to
28 * build_checksum which dispatches via checksum_parse to specialized helper functions per algorithm. Both paths log access + increment metric.
29 */

120/*
121 * WHAT: Locate the "algo:path"/"algo path" prefix separator (':' or ' ') within the NUL-bounded
122 * leading run of the payload, reporting the algorithm length and the separator offset.
123 * WHY: The legacy non-standard wire form places the algorithm as a "<algo><sep><path>" prefix;
124 * it must be distinguished from a bare path before the path is extracted. Isolating the scan
125 * keeps qcksum_select_algo linear and free of the raw index loop.
126 * HOW: strnlen bounds the scan to the printable prefix, then a single forward pass returns the
127 * first ':'/' ' offset (also the algorithm length) via *alg_len; returns 1 iff a separator
128 * was found, 0 otherwise (out-params zeroed).
129 */
130static ngx_flag_t
131brix_qcksum_find_algo_sep(const u_char *payload, size_t payload_len,
132 size_t *sep_off, size_t *alg_len)
133{
134 size_t wire_len = strnlen((const char *) payload, payload_len);
135
136 *sep_off = 0;
137 *alg_len = 0;
138
139 for (size_t i = 0; i < wire_len; i++) {
140 if (payload[i] == ':' || payload[i] == ' ') {
141 *sep_off = i;
142 *alg_len = i;
143 return 1;
144 }
145 }
146 return 0;
147}

149/*
150 * WHAT: Parse the standard "?cks.type=<algo>" CGI field off the request path, overriding any
151 * legacy prefix algorithm. Writes the algorithm into *algo; sets *bad_algo=1 on an
152 * unrecognized value.
153 * WHY: Stock XRootD clients and EOS request the algorithm via a "?cks.type=<algo>" CGI (XrdCl
154 * appends exactly "?cks.type=<algo>"); honoring it makes `xrdfs ... cksum -a <algo>` and
155 * `xrdcp --cksum` select correctly here, not only via the non-standard prefix form.
156 * HOW: Find '?', then walk '&'-separated fields; for the "cks.type=" field trim the trailing wire
157 * NUL/CR/LF (last-field form has no '&', so the NUL lands inside the value and would corrupt
158 * "adler32\0" → unknown algorithm) before parsing. *bad_algo distinguishes "no cks.type"
159 * (leave algo untouched) from "cks.type present but unknown" (caller must reject).
160 */
161static void
162brix_qcksum_apply_cks_type_cgi(const u_char *path_payload,
163 size_t path_payload_len, char *algo, size_t algo_sz, ngx_flag_t *bad_algo)
164{
165 static const char key[] = "cks.type=";
166 const u_char *qmark;
167 const u_char *p;
168 const u_char *end;
169
170 *bad_algo = 0;
171
172 qmark = memchr(path_payload, '?', path_payload_len);
173 if (qmark == NULL) {
174 return;
175 }
176
177 p = qmark + 1;
178 end = path_payload + path_payload_len;
179 while (p < end) {
180 const u_char *amp = memchr(p, '&', (size_t) (end - p));
181 size_t flen = amp ? (size_t) (amp - p) : (size_t) (end - p);
182
183 while (flen > 0 && (p[flen - 1] == '\0'
184 || p[flen - 1] == '\r'
185 || p[flen - 1] == '\n')) {
186 flen--;
187 }
188 if (flen > sizeof(key) - 1
189 && ngx_strncmp(p, key, sizeof(key) - 1) == 0) {
190 if (!brix_query_parse_algorithm(p + sizeof(key) - 1,
191 flen - (sizeof(key) - 1), algo, algo_sz)) {
192 *bad_algo = 1;
193 }
194 return;
195 }
196 if (amp == NULL) {
197 return;
198 }
199 p = amp + 1;
200 }
201}

203/*
204 * WHAT: Select the checksum algorithm for a path-based Qcksum and split the payload into the
205 * path portion. On success returns NGX_OK with the path_payload / path_payload_len out-params
206 * pointing at the request path; on an unknown algorithm sends kXR_ArgInvalid, sets *out_rc to
207 * that send result, and returns NGX_DECLINED.
208 * WHY: Two independent algorithm sources feed one selection: the legacy "<algo><sep><path>"
209 * prefix and the standard "?cks.type=<algo>" CGI (which overrides the prefix). INVARIANT 9:
210 * crc64 (CRC-64/XZ) and crc64nvme (CRC-64/NVME) are DISTINCT names to brix_checksum_parse —
211 * this selector never conflates them; it only forwards the wire token to the parser.
212 * HOW: Detect the prefix separator (brix_qcksum_find_algo_sep) and, when present and non-empty,
213 * parse the prefix algorithm and advance the path payload past it. Then apply the CGI
214 * override (brix_qcksum_apply_cks_type_cgi), which wins. Any unknown token → error send.
215 */
216static ngx_int_t
217brix_qcksum_select_algo(brix_qcksum_req_t *rq, const u_char **path_payload,
218 size_t *path_payload_len, ngx_int_t *out_rc)
219{
220 const u_char *payload = rq->ctx->recv.payload;
221 size_t payload_len = (size_t) rq->ctx->recv.cur_dlen;
222 size_t sep_off = 0;
223 size_t alg_len = 0;
224 ngx_flag_t bad_algo = 0;
225
226 *path_payload = payload;
227 *path_payload_len = payload_len;
228
229 if (brix_qcksum_find_algo_sep(payload, payload_len, &sep_off, &alg_len)
230 && alg_len > 0 && alg_len + 1 < payload_len)
231 {
232 if (!brix_query_parse_algorithm(payload, alg_len, rq->algo,
233 sizeof(rq->algo))) {
234 BRIX_OP_ERR(rq->ctx, BRIX_OP_QUERY_CKSUM);
235 *out_rc = brix_send_error(rq->ctx, rq->c, kXR_ArgInvalid,
236 "unknown checksum algorithm");
237 return NGX_DECLINED;
238 }
239 *path_payload = payload + sep_off + 1;
240 *path_payload_len = payload_len - (alg_len + 1);
241 }
242
243 brix_qcksum_apply_cks_type_cgi(*path_payload, *path_payload_len,
244 rq->algo, sizeof(rq->algo), &bad_algo);
245 if (bad_algo) {
246 BRIX_OP_ERR(rq->ctx, BRIX_OP_QUERY_CKSUM);
247 *out_rc = brix_send_error(rq->ctx, rq->c, kXR_ArgInvalid,
248 "unknown checksum algorithm");
249 return NGX_DECLINED;
250 }
251 return NGX_OK;
252}

254/*
255 * WHAT: Manager / redirector bounce for a path-based Qcksum. Returns NGX_DECLINED when the caller
256 * should proceed with the local resolve; otherwise sets *out_rc to the terminal result
257 * (redirect sent, not-found sent, NGX_AGAIN for an async CMS locate) and returns NGX_OK.
258 * WHY: A pure redirector keeps no local copy, so a kXR_Qcksum — file metadata about one path —
259 * must be bounced to the data server that authoritatively holds it, exactly as kXR_stat and
260 * kXR_open do. Isolating the bounce keeps the orchestrator's happy path flat.
261 * HOW: tried/triedrc convergence → not-found once every holder was visited; registry hit →
262 * redirect; registry miss with a CMS parent → async kYR_locate (NGX_AGAIN); all misses →
263 * NGX_DECLINED so the caller falls through to the local resolve (404).
264 */
265static ngx_int_t
266brix_qcksum_manager_bounce(brix_qcksum_req_t *rq, ngx_int_t *out_rc)
267{
268 brix_ctx_t *ctx = rq->ctx;
269 ngx_connection_t *c = rq->c;
270 ngx_stream_brix_srv_conf_t *conf = rq->conf;
271 const char *pathbuf = rq->pathbuf;
272 char redir_host[256];
273 uint16_t redir_port;
274
275 if (!conf->manager_mode) {
276 return NGX_DECLINED;
277 }
278
279 /* tried/triedrc: converge to not-found once the client has visited
280 * every server holding this path (avoids the redirect-limit loop).
281 * Terminal outcomes are reported to the caller via *out_rc (mirroring
282 * BRIX_RETURN_ERR / BRIX_RETURN_REDIR, whose embedded `return` cannot be
283 * used from a helper that also signals "declined → local resolve"). */
284 if (brix_manager_tried_exhausted(ctx->recv.payload, ctx->recv.cur_dlen,
285 pathbuf)) {
286 brix_log_access(ctx, c, "QUERY", pathbuf, "cksum",
287 0, kXR_NotFound, "file not found on any data server",
288 0);
289 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSUM);
290 *out_rc = brix_send_error(ctx, c, kXR_NotFound,
291 "file not found on any data server");
292 return NGX_OK;
293 }
294
295 if (brix_srv_select(pathbuf, 0, redir_host,
296 sizeof(redir_host), &redir_port)) {
297 brix_log_access(ctx, c, "QUERY", pathbuf, "registry",
298 1, kXR_ok, NULL, 0);
299 BRIX_OP_OK(ctx, BRIX_OP_QUERY_CKSUM);
300 *out_rc = brix_send_redirect(ctx, c, redir_host, redir_port);
301 return NGX_OK;
302 }
303
304 /* Registry miss — ask the CMS parent via kYR_locate (async). */
305 if (conf->cms.ctx != NULL) {
306 uint32_t streamid;
307
308 streamid = ngx_brix_cms_next_streamid(conf->cms.ctx);
309 if (brix_pending_insert(streamid, ngx_pid, c->fd, c->number,
310 ctx->recv.cur_streamid,
311 conf->cms.locate_timeout) == NGX_OK)
312 {
313 ctx->cms_wait_streamid = streamid;
314 ctx->state = XRD_ST_WAITING_CMS;
315 ngx_add_timer(c->read, conf->cms.locate_timeout);
316 if (ngx_brix_cms_send_locate(conf->cms.ctx, streamid,
317 pathbuf) == NGX_OK)
318 {
319 *out_rc = NGX_AGAIN;
320 return NGX_OK;
321 }
322 ngx_del_timer(c->read);
323 ctx->state = XRD_ST_REQ_HEADER;
324 brix_pending_remove(streamid, ngx_pid);
325 }
326 }
327 /* Registry + CMS miss: fall through to the local resolve (404). */
328 return NGX_DECLINED;
329}

331/*
332 * WHAT: Resolve a path-based Qcksum from wire payload to an authorized, confinement-canonical
333 * full_path (and the request-relative pathbuf). Returns NGX_OK when the caller may open the
334 * file; otherwise sets *out_rc to the terminal result (algo error, redirect, not-found,
335 * denied) and returns NGX_DECLINED.
336 * WHY: Algorithm selection, path extraction, manager bounce, beneath-path canonicalization, and
337 * the read auth gate form the entire security chain that must run before any filesystem
338 * access. Composing them in one resolver keeps the open/compute stage strictly post-auth.
339 * HOW: select_algo → extract_path → manager_bounce → beneath_full_path → auth_gate, each with
340 * early-return on its terminal outcome. The phase74-fp NOLINT on the beneath call is
341 * preserved verbatim (pathbuf=request path, full_path=output buffer — not swapped).

343static ngx_int_t
344brix_qcksum_resolve(brix_qcksum_req_t *rq, ngx_int_t *out_rc)
345{
346 brix_ctx_t *ctx = rq->ctx;
347 ngx_connection_t *c = rq->c;
348 ngx_stream_brix_srv_conf_t *conf = rq->conf;
349 const u_char *path_payload = NULL;
350 size_t path_payload_len = 0;
351 ngx_int_t rc;
352
353 rc = brix_qcksum_select_algo(rq, &path_payload, &path_payload_len, out_rc);
354 if (rc != NGX_OK) {
355 return NGX_DECLINED;
356 }
357
358 if (!brix_extract_path(c->log, path_payload, path_payload_len,
359 rq->pathbuf, sizeof(rq->pathbuf), 1)) {
360 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSUM);

368 * file, and a pure redirector keeps no local copy. Bounce the query to the

370 * kXR_open do — so a client behind the redirector still gets a checksum.

389/*
390 * WHAT: Open the resolved path through the VFS seam for checksum computation, returning the open
391 * handle. On failure returns NULL and sets *out_rc to the terminal result (cache-origin
392 * redirect on a read-through miss, otherwise an errno-mapped error send).
393 * WHY: A checksum read is a full-content origin fetch, so the open mirrors the davs/S3 GET
394 * confinement + credential-bind path exactly (impersonation-aware, metered, opt-in mint).
395 * Keeping the cache-miss redirect beside the open localizes the "not yet pulled" decision.
396 * HOW: Init a VFS ctx bound to the storage credential dir + mint CA + delegation, open O_READ.
397 * On ENOENT with a configured cache origin → redirect the metadata query there (the origin
398 * holds the authoritative bytes); otherwise map errno → kXR and send.
399 */
400static brix_vfs_file_t *
401brix_qcksum_open(brix_qcksum_req_t *rq, ngx_int_t *out_rc)
402{
403 brix_ctx_t *ctx = rq->ctx;
404 ngx_connection_t *c = rq->c;
405 ngx_stream_brix_srv_conf_t *conf = rq->conf;
406 const char *pathbuf = rq->pathbuf;
407 const char *full_path = rq->full_path;
408 brix_vfs_ctx_t vctx;
409 brix_vfs_file_t *fh;
410 int vfs_err = 0;
411
412 /* Confined read-open through the VFS seam (impersonation-aware, metered).
413 * The fd backs the backend-agnostic checksum kernel; the handle is closed
414 * via brix_vfs_close on every exit (sync below, or the aio done cb). */
415 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
416 conf->common.root_canon, NULL, conf->common.allow_write,
417 0 /* is_tls */, ctx->identity, full_path);
418 brix_vfs_ctx_bind_backend_cred(&vctx,
419 &conf->common.storage_credential_dir,
420 conf->common.storage_credential_fallback);
421 /* Phase-3 T1: opt-in credential minting, mirroring the davs/S3 GET
422 * mint bind — a checksum read is a full-content origin fetch. */
423 brix_vfs_ctx_bind_backend_mint(&vctx,
424 &conf->common.storage_credential_mint_ca_cert,
425 &conf->common.storage_credential_mint_ca_key,
426 conf->common.storage_credential_mint_ttl);
427 brix_root_vfs_bind_deleg(ctx, conf, &vctx);
428 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
429 errno = vfs_err;
430
431 if (fh != NULL) {
432 return fh;
433 }
434
435 /*
436 * Read-through cache miss: the file has not been pulled from the origin
437 * yet (a cached hit is computed locally above). The origin holds the
438 * authoritative bytes — identical to anything we would later cache — so
439 * redirect the metadata query there rather than returning "not found".
440 */
441 if (errno == ENOENT && conf->cache_origin_host.len > 0) {
442 char origin_host[256];
443 size_t hlen = conf->cache_origin_host.len < sizeof(origin_host)
444 ? conf->cache_origin_host.len : sizeof(origin_host) - 1;
445
446 ngx_memcpy(origin_host, conf->cache_origin_host.data, hlen);
447 origin_host[hlen] = '\0';
448 /* Equivalent to BRIX_RETURN_REDIR but through the *out_rc out-param
449 * (the macro embeds its own `return`, which cannot leave a helper that
450 * signals failure by returning NULL). */
451 brix_log_access(ctx, c, "QUERY", pathbuf, "cache-origin",
452 1, kXR_ok, NULL, 0);
453 BRIX_OP_OK(ctx, BRIX_OP_QUERY_CKSUM);
454 *out_rc = brix_send_redirect(ctx, c, origin_host,
455 conf->cache_origin_port);
456 return NULL;
457 }
458 /* Equivalent to BRIX_RETURN_ERR via the *out_rc out-param. */
459 {
460 uint16_t code = brix_kxr_from_errno(errno);
461 brix_log_access(ctx, c, "QUERY", full_path, "cksum",
462 0, code, strerror(errno), 0);
463 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSUM);
464 *out_rc = brix_send_error(ctx, c, code, strerror(errno));
465 }
466 return NULL;
467}

469/*
470 * WHAT: Try to offload a whole-object checksum to the thread pool from an open VFS handle.
471 * Returns NGX_OK when the task was posted (*handled=1) OR the fall-through-to-sync case
472 * (*handled=0); returns a terminal rc with *handled=1 when the task could not be created or
473 * posted (the handle is closed on those failure exits).
474 * WHY: Async execution prevents the event loop blocking on large-file checksums. The path
475 * variant owns the handle (close_fd=1: the done cb releases the VFS handle), unlike the
476 * handle variant which computes on an already-open fd it does not own.
477 * HOW: Alloc the task, populate the brix_cksum_aio_t (fd/fh/obj/streamid/algo/resolved), bind
478 * the thread + done callbacks, then brix_aio_post_task. A full queue leaves *handled=0 so
479 * the caller reuses the still-open handle for a synchronous compute.
480 */
481static ngx_int_t
482brix_qcksum_try_async(brix_qcksum_req_t *rq, brix_vfs_file_t *fh, int fd,
483 ngx_flag_t *handled)
484{
485 brix_ctx_t *ctx = rq->ctx;
486 ngx_connection_t *c = rq->c;
487 ngx_stream_brix_srv_conf_t *conf = rq->conf;
488 brix_cksum_aio_t *t;
489 ngx_thread_task_t *task;
490 ngx_flag_t posted = 0;
491
492 *handled = 0;
493
494 if (conf->common.thread_pool == NULL) {
495 return NGX_OK;
496 }
497

499 if (task == NULL) {

540/*
541 * WHAT: Synchronously compute the checksum on an open VFS handle, close the handle, and on
542 * success emit the access log + OK reply. Returns the terminal result of the request.
543 * WHY: The sync path runs when no thread pool is configured or the pool queue was full; it
544 * shares the format/reply tail with the async done callback's success shape.
545 * HOW: Snapshot the sd_obj for the whole-object checksum, build the "algo hex" response, close
546 * the handle unconditionally, then map NGX_DONE (error already sent) / non-OK / OK → send_ok.
547 */
548static ngx_int_t
549brix_qcksum_compute_sync(brix_qcksum_req_t *rq, brix_vfs_file_t *fh, int fd)
550{
551 brix_ctx_t *ctx = rq->ctx;
552 ngx_connection_t *c = rq->c;
553 brix_sd_obj_t cobj;
554 char resp[256];
555 ngx_int_t rc;
556
557 brix_vfs_file_sd_obj(fh, &cobj);
558 rc = brix_query_build_checksum(ctx, c, fd, &cobj, rq->full_path, rq->algo,
559 resp, sizeof(resp));
560 brix_vfs_close(fh, c->log);
561 if (rc == NGX_DONE) {
562 return NGX_OK;
563 }
564 if (rc != NGX_OK) {
565 return rc;
566 }
567
568 brix_log_access(ctx, c, "QUERY", rq->full_path, "cksum", 1, 0, NULL, 0);
569 BRIX_OP_OK(ctx, BRIX_OP_QUERY_CKSUM);
570 return brix_send_ok(ctx, c, resp, (uint32_t) (strlen(resp) + 1));
571}

573/* WHAT: Parses a payload containing "algo:path", resolves the path through security checks (authdb, VO ACL, token scope),
574 * opens the file confined, and delegates checksum computation to brix_query_build_checksum(). Returns hex-formatted result.
575 * WHY: kXR_Qcksum can query either open-file handles or arbitrary paths; this handler implements the path-based variant with
576 * full security chain verification before accessing any filesystem resource.
577 * HOW: resolve (algo select + path extract + manager bounce + auth gate) → VFS open (or cache-origin redirect) →
578 * try async offload, else compute synchronously + reply. Each stage early-returns on its terminal outcome. */

580static ngx_int_t
581brix_query_cksum_path(brix_ctx_t *ctx, ngx_connection_t *c,
582 ngx_stream_brix_srv_conf_t *conf)
583{
584 brix_qcksum_req_t rq;
585 brix_vfs_file_t *fh;
586 int fd;

588 ngx_flag_t handled = 0;
589
590 ngx_memzero(&rq, sizeof(rq));
591 rq.ctx = ctx;
592 rq.c = c;
593 rq.conf = conf;
594 ngx_cpystrn((u_char *) rq.algo, (u_char *) BRIX_QCKSUM_DEFAULT_ALGO,
595 sizeof(rq.algo));

615/*
616 * WHAT: Extract the optional algorithm carried on a handle-based Qcksum payload (prefix byte 0,
617 * followed by a NUL-bounded algorithm name) into *algo. A missing/empty/unknown name leaves
618 * *algo at its default.
619 * WHY: Clients may pin the algorithm for a handle checksum by prefixing the fhandle payload with
620 * a zero byte + "<algo>". Unlike the path variant, an unknown name here is silently ignored
621 * (the original handler discarded the parse result) — the default algorithm still applies.
622 * HOW: Guard on payload[0]==0 with at least one following byte, strnlen the algorithm run, and
623 * forward it to brix_query_parse_algorithm (whose failure is intentionally ignored).
624 */
625static void
626brix_qcksum_handle_extract_algo(brix_ctx_t *ctx, char *algo, size_t algo_sz)
627{
628 const u_char *ap;
629 size_t alen;
630
631 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen <= 1
632 || ctx->recv.payload[0] != 0)
633 {
634 return;
635 }
636
637 ap = ctx->recv.payload + 1;
638 alen = strnlen((const char *) ap, (size_t) (ctx->recv.cur_dlen - 1));
639 if (alen > 0) {
640 (void) brix_query_parse_algorithm(ap, alen, algo, algo_sz);
641 }

644/*
645 * WHAT: Try to offload a whole-object checksum on an already-open fhandle fd to the thread pool.
646 * Returns NGX_OK with *handled=1 when posted, *handled=0 for the fall-through-to-sync case,
647 * or a terminal rc with *handled=1 on task-alloc/post failure.
648 * WHY: The handle variant computes on an fd it does NOT own (close_fd=0, no VFS handle to
649 * release) — so, unlike the path variant, its failure exits never close anything. Keeping
650 * the two async posters separate avoids a conditional-ownership branch that would re-inflate
651 * the complexity both splits are removing.
652 * HOW: Alloc the task, populate brix_cksum_aio_t from the files[] slot (fd + snapshotted sd_obj),
653 * bind the callbacks, and brix_aio_post_task; a full queue leaves *handled=0 for sync.

661 ngx_stream_brix_srv_conf_t *conf = rq->conf;
662 brix_cksum_aio_t *t;
663 ngx_thread_task_t *task;
664 ngx_flag_t posted = 0;
665
666 *handled = 0;
667
668 if (conf->common.thread_pool == NULL) {
669 return NGX_OK;
670 }
671
672 task = ngx_thread_task_alloc(c->pool, sizeof(brix_cksum_aio_t));
673 if (task == NULL) {
674 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSUM);
675 *handled = 1;
676 return brix_send_error(ctx, c, kXR_NoMemory, "out of memory");
677 }
678
679 t = task->ctx;
680 t->ctx = ctx;
681 t->c = c;
682 t->conf = conf;
683 t->fd = ctx->files[idx].fd;
684 t->close_fd = 0;
685 t->obj = ctx->files[idx].sd_obj; /* Layer 3: whole-object checksum */
686 ngx_memcpy(t->streamid, ctx->recv.cur_streamid, 2);
687 ngx_cpystrn((u_char *) t->algo, (u_char *) rq->algo, sizeof(t->algo));
688 ngx_cpystrn((u_char *) t->resolved, (u_char *) resolved,
689 sizeof(t->resolved));
690 t->error_code = 0;
691
692 brix_task_bind(task, brix_cksum_aio_thread, brix_cksum_aio_done);
693 task->ctx = t;
694
695 if (brix_aio_post_task(ctx, c, conf->common.thread_pool, task,
696 "cksum thread pool queue full, using sync",
697 &posted) != NGX_OK)
698 {
699 *handled = 1;
700 return NGX_ERROR;
701 }
702
703 if (posted) {
704 *handled = 1;
705 return NGX_OK;
706 }
707 return NGX_OK;
708}

710/* WHAT: Computes checksum on an already-open file using its stored fhandle index. Extracts optional algo from payload (prefix byte=0),
711 * validates the handle index, and delegates to brix_query_build_checksum(). Returns hex-formatted result.
712 * WHY: kXR_Qcksum can query either open-file handles or arbitrary paths; this handler implements the handle-based variant where
713 * the file is already opened (lower overhead than re-opening). Clients use handles for repeated checksum queries on same file.
714 * HOW: extract optional algo → validate fhandle index range + fd → try async offload, else build checksum synchronously + reply. */

717brix_query_cksum_handle(brix_ctx_t *ctx, ngx_connection_t *c,
718 ngx_stream_brix_srv_conf_t *conf, const xrdw_query_req_t *req)
719{
720 brix_qcksum_req_t rq;
721 char resolved[PATH_MAX];
722 char resp[256];
723 int idx;
724 ngx_int_t rc;
725 ngx_flag_t handled = 0;
726
727 ngx_memzero(&rq, sizeof(rq));
728 rq.ctx = ctx;
729 rq.c = c;
730 rq.conf = conf;

764 return brix_send_ok(ctx, c, resp, (uint32_t) (strlen(resp) + 1));
