12 * raw-I/O chokepoint as normal write/pgwrite/writev handlers.
13 * HOW: Initialize a WRITE job, execute it inline, then normalize hard errors
14 * and partial writes into a single NGX_ERROR result. */

16/* WHAT: Out-parameters of one checkpoint VFS write: bytes written, the errno on
17 * failure, and whether the failure was a short (partial) write.
18 * WHY: Bundling the three result slots collapses ckp_xeq_vfs_write_full's param
19 * list (which the callers all live in this file) below the ≤5 cap without
20 * changing any observed value — every field is written on every call.
21 * HOW: The helper zero-inits all three, then overwrites on the error paths. */

23 ssize_t written;
24 int io_errno;
25 ngx_flag_t short_io;
26} ckp_write_result_t;

58/* WHAT: Performs a single VFS-core write to the checkpointed file at the specified offset. Validates handle matches idx, writes sub_dlen bytes,
59 * updates byte counters and sends success response. Handles zero-length payloads (no-op).
60 * WHY: ckpXeq write is the atomic transactional write operation — when executed under an active checkpoint, this write becomes "tentative" until commit/rollback.
61 * The handle mismatch check prevents cross-file corruption during checkpointed operations.
62 * HOW: 1) Validate fhandle[0] == idx. 2) Handle zero-length payload as no-op. 3) Run a VFS WRITE job. 4) Update counters + send_ok. */

126/* WHAT: In/out state for the pgwrite page-drain loop: the decoded flat buffer
127 * and its byte count plus the starting file offset (in), and the final
128 * file offset and total bytes written (out).
129 * WHY: Passing one struct keeps ckp_xeq_pgwrite_drain at ≤5 params while the
130 * caller reads back both results after a successful drain.
131 * HOW: ckp_xeq_pgwrite seeds flat/flat_sz/offset; the drain fills end_offset
132 * and total on success. */
133typedef struct {
134 const u_char *flat;
135 size_t flat_sz;
136 int64_t offset;
137 int64_t end_offset;
138 size_t total;
139} ckp_pgwrite_drain_t;

141/* WHAT: Writes the verified pgwrite plaintext to disk one page at a time,
142 * advancing the file offset in lockstep, and reports the final offset and
143 * byte total (or emits the stock IO-error log + response on a failed page).
144 * WHY: Page-unit writes keep offsets page-aligned and bound what a short write
145 * can lose; hoisting the loop out of ckp_xeq_pgwrite keeps that function
146 * under the length/CCN caps with byte-identical behaviour.
147 * HOW: Loop over full-then-partial pages via ckp_xeq_vfs_write_full; on failure
148 * log + BRIX_OP_ERR + send kXR_IOError and return NGX_DONE (error already
149 * emitted — caller frees flat and returns brix_send_error's own NGX_OK);
150 * on success write st->end_offset / st->total and return NGX_OK. */
151static ngx_int_t
152ckp_xeq_pgwrite_drain(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
153 ckp_pgwrite_drain_t *st)
154{
155 const u_char *src = st->flat;
156 size_t rem = st->flat_sz;
157 int64_t write_offset = st->offset;
158 size_t total_written = 0;
159 size_t page_data;
160 char detail[64];
161
162 while (rem > 0) {
163 ckp_write_result_t res;
164
165 /* Final chunk may be a partial page. */
166 page_data = (rem >= XRD_PGWRITE_PAGESZ) ? XRD_PGWRITE_PAGESZ : rem;
167
168 if (ckp_xeq_vfs_write_full(ctx->files[idx].fd, src, page_data,
169 (off_t) write_offset, &res)
170 != NGX_OK)
171 {
172 const char *msg = res.short_io ? "short write (disk full?)"
173 : strerror(res.io_errno);
174
175 snprintf(detail, sizeof(detail), "xeq_pgwrite %lld+%zu",
176 (long long) st->offset, total_written);
177 brix_log_access(ctx, c, "CHKPOINT", ctx->files[idx].path, detail,
178 0, kXR_IOError, msg, 0);
179 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
180 (void) brix_send_error(ctx, c, kXR_IOError, msg);
181 return NGX_DONE;
182 }
183
184 total_written += (size_t) res.written;
185 write_offset += (int64_t) res.written;
186 src += page_data;
187 rem -= page_data;
188 }
189
190 st->end_offset = write_offset;
191 st->total = total_written;
192 return NGX_OK;
193}

223 }
224
225 /* Worst case the decoded (CRC-stripped) data is smaller than sub_dlen, so
226 * sub_dlen is a safe upper bound for the flat buffer. Freed on every exit
227 * path below (heap, not pool — must not leak on error returns). */

235 * offset so the error/log line can pinpoint the bad page. flat_sz returns

249 (long long) bad_offset, (unsigned) d->sub_dlen);

251 0, status, msg, 0);
252 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
253 ngx_free(flat);
254 return brix_send_error(ctx, c, status, msg);

326/* WHAT: Executes an embedded kXR_writev under checkpoint protection: validates
327 * the descriptor block under the STOCK wire contract (sub_dlen frames
328 * ONLY the N*16-byte write_list descriptors; data_len bytes of segment
329 * data were streamed after them), requires every data-bearing segment to
330 * target the checkpointed handle idx, then writes each segment through
331 * the VFS core.
332 * WHY: kXR_ckpXeq tunnels the SAME wire layout as a standalone kXR_writev
333 * (see brix_handle_writev in writev.c): the embedded header's dlen
334 * counts descriptors only and the data rides behind — stock do_ChkPntXeq
335 * fetches the descriptor block, then hands off to do_WriteV, which
336 * streams the data. Multi-file vectors are refused (stock: "multi-file
337 * chkpoint writev not supported") because the rollback anchor covers
338 * exactly one file.
339 * HOW: 1) Validate sub_dlen non-zero, 16-aligned, <= BRIX_WRITEV_MAXSEGS
340 * segments — violations reject + drop the link like stock (once the
341 * descriptor framing is in doubt the trailing byte count is unknowable).
342 * 2) Sum wlen; require data-bearing segments to name idx and the sum to
343 * equal data_len (defensive: the recv framing extended by exactly that).
344 * 3) One VFS WRITE job per non-empty segment. */
345
346ngx_int_t
347ckp_xeq_writev(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
348 const ckp_write_desc_t *d, uint32_t data_len)
349{

351 size_t n_segs, i;
352 uint64_t total_wlen;
353 const u_char *data_ptr;
354 size_t bytes_written_total = 0;
355
356 /* Stock parity (do_WriteV, reached via do_ChkPntXeq): a sub_dlen that is
357 * zero or not a whole number of descriptors is invalid — error + drop. */

359 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);

361 "Write vector is invalid");

363 }
364

366 n_segs = d->sub_dlen / BRIX_WRITEV_SEGSIZE;
367

369 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
370 (void) brix_send_error(ctx, c, kXR_ArgTooLong,
371 "Write vector is too long");
372 return NGX_ERROR;
373 }
374
375 total_wlen = 0;

398 "ckpXeq writev: payload size mismatch");

400 }
401

404

407 uint32_t wlen = (uint32_t) ntohl((uint32_t) wl[i].wlen);

409
410 if (wlen == 0) {
411 continue;
412 }
413
414 if (ckp_xeq_vfs_write_full(ctx->files[idx].fd, data_ptr,
415 (size_t) wlen, (off_t) offset, &res)

419 : strerror(res.io_errno);

422 ctx->files[idx].path, "xeq_writev",

424 }
425 ctx->files[idx].bytes_written += (size_t) res.written;
426 ctx->totals.bytes_written += (size_t) res.written;
427 bytes_written_total += (size_t) res.written;
428 data_ptr += wlen;
429 }

432 "xeq_writev", 1, kXR_ok, NULL, bytes_written_total);
