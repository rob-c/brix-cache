6/* File: src/aio/uring_admin.c — io_uring runtime kill switch (Phase 44)
7 * WHAT: The no-rebuild, no-reload kill switch for the optional io_uring backend
8 * (§8.1/§14.3): a single cross-worker ngx_atomic_t in shared memory that,
9 * when set, makes every worker's selector skip the io_uring tier on its
10 * next op (in-flight CQEs still drain). Three ways to flip it — the admin
11 * API (api_admin.c), a watched panic-file (per-worker poll timer here), or
12 * the runtime probe never bringing a ring up in the first place.
13 *
14 * WHY: io_uring has been a major kernel attack surface; operators need to
15 * neutralise it fleet-wide in seconds during an incident without a rebuild
16 * or reload. The flag lives in SHM so one admin call / one dropped file is
17 * observed by every worker on its next op. Build-independent (no liburing
18 * types) so the HTTP dashboard module can drive it.
19 *
20 * HOW: A tiny SHM zone holds one ngx_atomic_t. The hot path reads it lock-free
21 * (brix_uring_disabled in uring.h). The panic-file poll mirrors file
22 * existence into the flag. Everything degrades to a clean no-op when the
23 * zone was never registered (stub build / io_uring off everywhere). */
24
25/* Panic-file poll cadence (coarse: this is an incident switch, not a hot path). */
26#define BRIX_URING_PANIC_POLL_MS 2000

39/*
40 * brix_uring_ks_zone_init — SHM zone init callback. Slab-allocates the single
41 * ngx_atomic_t flag and stashes it in sp->data (so a reload re-attaches to the
42 * same flag). Mirrors the minimal single-object SHM pattern used elsewhere in
43 * the module.
44 */
45static ngx_int_t
46brix_uring_ks_zone_init(ngx_shm_zone_t *zone, void *data)
47{
48 ngx_slab_pool_t *sp;
49 ngx_atomic_t *flag;
50
51 if (data) { /* reload: inherit the previous flag */
52 zone->data = data;
53 return NGX_OK;
54 }
55
56 sp = (ngx_slab_pool_t *) zone->shm.addr;
57
58 if (zone->shm.exists) { /* re-attach to an existing segment */
59 zone->data = sp->data;
60 return NGX_OK;
61 }
62
63 flag = ngx_slab_alloc(sp, sizeof(ngx_atomic_t));
64 if (flag == NULL) {
65 return NGX_ERROR;
66 }
67 *flag = 0; /* enabled by default */
68 sp->data = (void *) flag;
69 zone->data = (void *) flag;
70 return NGX_OK;
71}

133/*
134 * brix_uring_panic_handler — poll timer: mirror the panic-file's existence
135 * into the SHM disable flag, then re-arm. A NOTICE is logged only on a state
136 * change so the flip is visible without log spam.
137 */
138static void
139brix_uring_panic_handler(ngx_event_t *ev)
140{
141 struct stat st;
142 ngx_uint_t exists;
143 ngx_int_t prev;
144
145 exists = (brix_uring_panic_path[0] != '\0'
146 && stat((char *) brix_uring_panic_path, &st) == 0) ? 1 : 0;
147
148 prev = brix_uring_killswitch_get();
149 if (prev != (ngx_int_t) exists) {
150 brix_uring_killswitch_set(exists);
151 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
152 "brix: io_uring %s via panic-file \"%s\"",
153 exists ? "DISABLED" : "re-enabled",
154 (char *) brix_uring_panic_path);
155 }
156
157 ngx_add_timer(ev, BRIX_URING_PANIC_POLL_MS);
158}

160ngx_int_t
161brix_uring_panicfile_arm(ngx_cycle_t *cycle, ngx_str_t *path)
162{
163 if (path == NULL || path->len == 0) {
164 return NGX_OK;
165 }
166 if (path->len >= sizeof(brix_uring_panic_path)) {
167 ngx_log_error(NGX_LOG_WARN, cycle->log, 0,
168 "brix: io_uring panic-file path too long; ignoring");
169 return NGX_OK;
170 }
171
172 ngx_memcpy(brix_uring_panic_path, path->data, path->len);
173 brix_uring_panic_path[path->len] = '\0';
174
175 /* Shared dummy connection (src/core/config/process.c): nginx's --with-debug
176 * timer-expiry log reads ngx_event_ident(ev->data)->fd, so a connection-less
177 * timer must not leave ev->data NULL or it SIGSEGVs the worker. */
178 extern ngx_connection_t brix_maint_timer_conn;
179 brix_uring_panic_timer.handler = brix_uring_panic_handler;
180 brix_uring_panic_timer.log = cycle->log;
181 brix_uring_panic_timer.data = &brix_maint_timer_conn;
182 brix_uring_panic_timer.cancelable = 1; /* don't hold up worker shutdown */
183
184 /* Fire once promptly so a file already present at start disables on boot. */
185 ngx_add_timer(&brix_uring_panic_timer, 1);
186 return NGX_OK;
187}
