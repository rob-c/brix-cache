12/* ---- range_vector_reject — map a rejected component to its return code. ----
13 *
14 * WHAT: Returns NGX_DECLINED when opts->drop_unsatisfiable is set (skip the
15 * component and continue), otherwise NGX_ERROR (abort the whole header).
16 *
17 * WHY: A malformed or unsatisfiable component is rejected identically from a
18 * dozen sites in the parser. Centralising the policy branch keeps every
19 * call site a single early-return and holds the accept/reject contract in
20 * exactly one place, so it cannot drift between the suffix and explicit
21 * paths.
22 *
23 * HOW:
24 * 1. Branch on opts->drop_unsatisfiable and return the matching code.
25 */
26static ngx_int_t
27range_vector_reject(const brix_range_vector_opts_t *opts)
28{
29 return opts->drop_unsatisfiable ? NGX_DECLINED : NGX_ERROR;
30}

32/* ---- range_vector_parse_suffix — parse a suffix component "-N". ----
33 *
34 * WHAT: Interprets "-N" as the last N bytes of the file and writes the
35 * normalised [*start_out, *last_out] window. Returns NGX_OK on success,
36 * or range_vector_reject(opts) when suffix ranges are disallowed or N is
37 * non-positive/unparseable.
38 *
39 * WHY: The suffix form has offset semantics distinct from an explicit range —
40 * N counts back from EOF — so isolating it keeps that arithmetic (and its
41 * clamp) out of the explicit path.
42 *
43 * HOW:
44 * 1. Reject when opts->allow_suffix is unset.
45 * 2. Parse N from just past the dash; reject when N <= 0 (also covers the
46 * ngx_atoof -1 parse-failure sentinel).
47 * 3. Clamp N to file_size so a suffix larger than the file spans it whole.
48 * 4. Emit last = file_size - 1 and start = file_size - N.
49 */
50static ngx_int_t
51range_vector_parse_suffix(const u_char *dash, const u_char *comma,
52 off_t file_size, const brix_range_vector_opts_t *opts,
53 off_t *start_out, off_t *last_out)
54{
55 off_t start;
56
57 if (!opts->allow_suffix) {
58 return range_vector_reject(opts);
59 }
60
61 start = (off_t) ngx_atoof((u_char *)(dash + 1), comma - (dash + 1));
62 if (start < 0) { /* covers the ngx_atoof -1 parse-failure sentinel */
63 return range_vector_reject(opts);
64 }
65
66 /* A well-formed zero-length suffix ("-0") is grammatically valid but
67 * unsatisfiable; let it flow through as start = file_size so the final
68 * bounds gate in range_vector_parse_one classifies it as unsatisfiable
69 * (not a malformed grammar error). */
70 if (start > file_size) {
71 start = file_size;
72 }
73
74 *last_out = file_size - 1;
75 *start_out = file_size - start;
76 return NGX_OK;
77}

79/* ---- range_vector_parse_explicit — parse an explicit component "A-" or "A-B". ----
80 *
81 * WHAT: Parses the start offset A and either an open-ended end (to EOF) or an
82 * explicit end B, writing [*start_out, *last_out]. Returns NGX_OK on
83 * success, or range_vector_reject(opts) when A or B is unparseable or the
84 * open-ended form is disallowed.
85 *
86 * WHY: Separated from the suffix path because start is always the literal A
87 * here (no EOF-relative arithmetic), and the open-ended vs. bounded end is
88 * a self-contained sub-decision.
89 *
90 * HOW:
91 * 1. Parse A from [p, dash); reject when negative (covers the ngx_atoof
92 * -1 parse-failure sentinel).
93 * 2. If the dash is the last byte ("A-"): reject when open-ended ranges
94 * are disallowed, else set last = file_size - 1.
95 * 3. Otherwise parse B; reject when negative, then clamp B to the last
96 * valid byte when it runs past EOF.
97 * 4. Emit start = A and last = the resolved end.
98 */
99static ngx_int_t
100range_vector_parse_explicit(const u_char *p, const u_char *dash,
101 const u_char *comma, off_t file_size,
102 const brix_range_vector_opts_t *opts,
103 off_t *start_out, off_t *last_out)
104{
105 off_t start, last;
106
107 start = (off_t) ngx_atoof((u_char *) p, dash - p);
108 if (start < 0) { /* covers the NGX_ERROR (-1) parse-failure sentinel */
109 return range_vector_reject(opts);
110 }
111
112 if (dash + 1 == comma) {
113 /* Open-ended range: "A-" means from A to EOF. */
114 if (!opts->allow_open_ended) {
115 return range_vector_reject(opts);
116 }
117 last = file_size - 1;
118 } else {
119 last = (off_t) ngx_atoof((u_char *)(dash + 1), comma - (dash + 1));
120 if (last < 0) { /* covers the NGX_ERROR (-1) parse-failure sentinel */
121 return range_vector_reject(opts);
122 }
123 if (last >= file_size) {
124 last = file_size - 1;
125 }
126 }
127
128 *start_out = start;
129 *last_out = last;
130 return NGX_OK;
131}

133/* ---- range_vector_parse_one — parse a single byte-range component [p, comma). ----
134 *
135 * WHAT: Splits the component by '-' (suffix "-N", open-ended "A-", or "A-B"),
136 * normalises the offsets against file_size, and on a satisfiable range
137 * fills *out. Returns NGX_OK when a range was stored, NGX_DECLINED to
138 * skip a malformed/unsatisfiable component (when opts->drop_unsatisfiable
139 * is set), or NGX_ERROR for the same conditions when dropping is disabled.
140 *
141 * WHY: One place resolves a raw component into a validated byte range so both
142 * the suffix and explicit forms share the same final satisfiability gate
143 * and the same reject policy.
144 *
145 * HOW:
146 * 1. Locate the '-' separator; reject when the component has none.
147 * 2. Dispatch to the suffix helper when the dash is leading, else the
148 * explicit helper; propagate any non-OK return unchanged (grammar
149 * error -> *fail = MALFORMED).
150 * 3. Reject a well-formed range that falls out of the file bounds
151 * (start >= file_size) as *fail = UNSATISFIABLE; a reversed span
152 * (start > last) is a malformed byte-range-spec (RFC 9110 §14.1.2).
153 * 4. Fill *out (fd/handle cleared) and return NGX_OK.
154 */
155static ngx_int_t
156range_vector_parse_one(const u_char *p, const u_char *comma, off_t file_size,
157 const brix_range_vector_opts_t *opts, brix_byte_range_t *out,
158 brix_range_fail_e *fail)
159{
160 const u_char *dash;
161 off_t start = 0, last = 0;
162 ngx_int_t rc;
163
164 *fail = BRIX_RANGE_FAIL_MALFORMED;
165
166 /* Find dash within this range component. */
167 dash = ngx_strlchr((u_char *) p, (u_char *) comma, '-');
168 if (dash == NULL) {
169 return range_vector_reject(opts);
170 }
171
172 if (p == dash) {
173 rc = range_vector_parse_suffix(dash, comma, file_size, opts,
174 &start, &last);
175 } else {
176 rc = range_vector_parse_explicit(p, dash, comma, file_size, opts,
177 &start, &last);
178 }
179 if (rc != NGX_OK) {
180 return rc;
181 }
182
183 /* A well-formed range that begins past EOF is unsatisfiable (416); a
184 * reversed span (start > last) is a malformed specifier (ignore -> 200). */
185 if (start >= file_size) {
186 *fail = BRIX_RANGE_FAIL_UNSATISFIABLE;
187 return range_vector_reject(opts);
188 }
189 if (start < 0 || start > last) {
190 return range_vector_reject(opts);
191 }
192
193 *fail = BRIX_RANGE_FAIL_NONE;
194 out->start = start;
195 out->end = last;
196 out->fd = -1;
197 out->handle = 0;
198 return NGX_OK;
199}

201/*
202 * brix_http_parse_range_vector — parse a HTTP byte-ranges header value.
203 *
204 * WHAT: Parses a comma-separated list of byte ranges (e.g. "0-499, -500, 9500-")
205 * and normalises them against file_size. Handles suffix and open-ended
206 * ranges according to opts policy.
207 * WHY: Both XrdHttp (multipart) and single-range HTTP responses need the same
208 * range validation and overflow detection. Centralising this avoids
209 * divergent behavior between protocol surfaces.
210 * HOW: Iterates through comma-separated tokens. For each token:
211 * 1. Splits by '-'.
212 * 2. Handles suffix ranges (no start digit).
213 * 3. Handles open-ended ranges (no end digit).
214 * 4. Normalises offsets against file_size.
215 * 5. Detects unsatisfiable ranges (start >= file_size or start > end).
216 */
217ngx_int_t
218brix_http_parse_range_vector(const u_char *data, size_t len,
219 off_t file_size, const brix_range_vector_opts_t *opts,
220 brix_byte_range_t *ranges, ngx_uint_t *nranges, brix_range_fail_e *fail)
221{
222 const u_char *p, *end, *comma;
223 ngx_uint_t n = 0;
224 brix_range_fail_e why = BRIX_RANGE_FAIL_NONE;
225
226 p = data;
227 end = data + len;
228
229 while (p < end && n < opts->max_ranges) {
230 ngx_int_t rc;
231
232 /* Skip leading whitespace. */
233 while (p < end && *p == ' ') p++;
234 if (p >= end) break;
235
236 /* Find next comma. */
237 comma = ngx_strlchr((u_char *) p, (u_char *) end, ',');
238 if (comma == NULL) {
239 comma = end;
240 }
241
242 rc = range_vector_parse_one(p, comma, file_size, opts, &ranges[n],
243 &why);
244 if (rc == NGX_ERROR) {
245 if (fail != NULL) {
246 *fail = why;
247 }
248 return NGX_ERROR;
249 }
250 if (rc == NGX_OK) {
251 n++;
252 }
253 /* NGX_DECLINED → unsatisfiable component dropped; just advance. */
254
255 p = comma + 1;
256 }
257
258 if (fail != NULL) {
259 *fail = BRIX_RANGE_FAIL_NONE;
260 }
261 *nranges = n;
262 return NGX_OK;
263}

265ngx_int_t
266brix_range_vector_validate_total(
267 const brix_byte_range_t *ranges, ngx_uint_t nranges,
268 off_t max_total_bytes, off_t *total_out)
269{
270 off_t total = 0;
271 ngx_uint_t i;
272
273 for (i = 0; i < nranges; i++) {
274 off_t len = ranges[i].end - ranges[i].start + 1;
275
276 /* Check for integer overflow in sum. */
277 if (len < 0 || (total > 0 && len > (off_t) (NGX_MAX_OFF_T_VALUE - total))) {
278 return NGX_ERROR;
279 }
280 total += len;
281 }
282
283 if (max_total_bytes > 0 && total > max_total_bytes) {
284 return NGX_ERROR;
285 }
286
287 if (total_out) {
288 *total_out = total;
289 }
290 return NGX_OK;
291}

293ngx_uint_t
294brix_range_vector_next_coalesced_run(
295 const brix_byte_range_t *ranges, ngx_uint_t nranges,
296 ngx_uint_t start_index, ngx_uint_t max_iov)
297{
298 ngx_uint_t i;
299 ngx_uint_t count = 1;
300
301 if (start_index >= nranges) return 0;
302 if (max_iov == 0) return 0;
303
304 for (i = start_index + 1; i < nranges && count < max_iov; i++) {
305 /* A run is contiguous if ranges[i].start follows ranges[i-1].end
306 * and both refer to the same open file (fd match). */
307 if (ranges[i].start == ranges[i-1].end + 1
308 && ranges[i].fd == ranges[i-1].fd)
309 {
310 count++;
311 } else {
312 break;
313 }
314 }
315
316 return count;
317}
