12/*
13 * range_vector_parse_one — parse a single byte-range component [p, comma).
14 *
15 * Splits the component by '-' (suffix "-N", open-ended "A-", or "A-B"),
16 * normalises the offsets against file_size, and on a satisfiable range fills
17 * *out. Returns NGX_OK when a range was stored, NGX_DECLINED to skip a
18 * malformed/unsatisfiable component (when opts->drop_unsatisfiable is set), or
19 * NGX_ERROR for the same conditions when dropping is disabled.
20 */
21static ngx_int_t
22range_vector_parse_one(const u_char *p, const u_char *comma, off_t file_size,
23 const brix_range_vector_opts_t *opts, brix_byte_range_t *out)
24{
25 const u_char *dash;
26 off_t start, last;
27
28 /* Find dash within this range component. */
29 dash = ngx_strlchr((u_char *) p, (u_char *) comma, '-');
30 if (dash == NULL) {
31 return opts->drop_unsatisfiable ? NGX_DECLINED : NGX_ERROR;
32 }
33
34 /* 1. Parse start offset (if present). */
35 if (p == dash) {
36 /* Suffix range: "-N" means the last N bytes. */
37 if (!opts->allow_suffix) {
38 return opts->drop_unsatisfiable ? NGX_DECLINED : NGX_ERROR;
39 }
40
41 start = (off_t) ngx_atoof((u_char *)(dash + 1), comma - (dash + 1));
42 if (start <= 0) {
43 return opts->drop_unsatisfiable ? NGX_DECLINED : NGX_ERROR;
44 }
45
46 if (start > file_size) {
47 start = file_size;
48 }
49 last = file_size - 1;
50 start = file_size - start;
51
52 } else {
53 start = (off_t) ngx_atoof((u_char *) p, dash - p);
54 if (start == (off_t) NGX_ERROR || start < 0) {
55 return opts->drop_unsatisfiable ? NGX_DECLINED : NGX_ERROR;
56 }
57
58 /* 2. Parse end offset (if present). */
59 if (dash + 1 == comma) {
60 /* Open-ended range: "A-" means from A to EOF. */
61 if (!opts->allow_open_ended) {
62 return opts->drop_unsatisfiable ? NGX_DECLINED : NGX_ERROR;
63 }
64 last = file_size - 1;
65 } else {
66 last = (off_t) ngx_atoof((u_char *)(dash + 1), comma - (dash + 1));
67 if (last == (off_t) NGX_ERROR || last < 0) {
68 return opts->drop_unsatisfiable ? NGX_DECLINED : NGX_ERROR;
69 }
70 if (last >= file_size) {
71 last = file_size - 1;
72 }
73 }
74 }
75
76 /* 3. Validate unsatisfiable ranges. */
77 if (start < 0 || start >= file_size || start > last) {
78 return opts->drop_unsatisfiable ? NGX_DECLINED : NGX_ERROR;
79 }
80
81 out->start = start;
82 out->end = last;
83 out->fd = -1;
84 out->handle = 0;
85 return NGX_OK;
86}

88/*
89 * brix_http_parse_range_vector — parse a HTTP byte-ranges header value.
90 *
91 * WHAT: Parses a comma-separated list of byte ranges (e.g. "0-499, -500, 9500-")
92 * and normalises them against file_size. Handles suffix and open-ended
93 * ranges according to opts policy.
94 * WHY: Both XrdHttp (multipart) and single-range HTTP responses need the same
95 * range validation and overflow detection. Centralising this avoids
96 * divergent behavior between protocol surfaces.
97 * HOW: Iterates through comma-separated tokens. For each token:
98 * 1. Splits by '-'.
99 * 2. Handles suffix ranges (no start digit).
100 * 3. Handles open-ended ranges (no end digit).
101 * 4. Normalises offsets against file_size.
102 * 5. Detects unsatisfiable ranges (start >= file_size or start > end).
103 */
104ngx_int_t
105brix_http_parse_range_vector(const u_char *data, size_t len,
106 off_t file_size, const brix_range_vector_opts_t *opts,
107 brix_byte_range_t *ranges, ngx_uint_t *nranges)
108{
109 const u_char *p, *end, *comma;
110 ngx_uint_t n = 0;
111
112 p = data;
113 end = data + len;
114
115 while (p < end && n < opts->max_ranges) {
116 ngx_int_t rc;
117
118 /* Skip leading whitespace. */
119 while (p < end && *p == ' ') p++;
120 if (p >= end) break;
121
122 /* Find next comma. */
123 comma = ngx_strlchr((u_char *) p, (u_char *) end, ',');
124 if (comma == NULL) {
125 comma = end;
126 }
127
128 rc = range_vector_parse_one(p, comma, file_size, opts, &ranges[n]);
129 if (rc == NGX_ERROR) {
130 return NGX_ERROR;
131 }
132 if (rc == NGX_OK) {
133 n++;
134 }
135 /* NGX_DECLINED → unsatisfiable component dropped; just advance. */
136
137 p = comma + 1;
138 }
139
140 *nranges = n;
141 return NGX_OK;
142}

144ngx_int_t
145brix_range_vector_validate_total(
146 const brix_byte_range_t *ranges, ngx_uint_t nranges,
147 off_t max_total_bytes, off_t *total_out)
148{
149 off_t total = 0;
150 ngx_uint_t i;
151
152 for (i = 0; i < nranges; i++) {
153 off_t len = ranges[i].end - ranges[i].start + 1;
154
155 /* Check for integer overflow in sum. */
156 if (len < 0 || (total > 0 && len > (off_t) (NGX_MAX_OFF_T_VALUE - total))) {
157 return NGX_ERROR;
158 }
159 total += len;
160 }
161
162 if (max_total_bytes > 0 && total > max_total_bytes) {
163 return NGX_ERROR;
164 }
165
166 if (total_out) {
167 *total_out = total;
168 }
169 return NGX_OK;
170}

172ngx_uint_t
173brix_range_vector_next_coalesced_run(
174 const brix_byte_range_t *ranges, ngx_uint_t nranges,
175 ngx_uint_t start_index, ngx_uint_t max_iov)
176{
177 ngx_uint_t i;
178 ngx_uint_t count = 1;
179
180 if (start_index >= nranges) return 0;
181 if (max_iov == 0) return 0;
182
183 for (i = start_index + 1; i < nranges && count < max_iov; i++) {
184 /* A run is contiguous if ranges[i].start follows ranges[i-1].end
185 * and both refer to the same open file (fd match). */
186 if (ranges[i].start == ranges[i-1].end + 1
187 && ranges[i].fd == ranges[i-1].fd)
188 {
189 count++;
190 } else {
191 break;
192 }
193 }
194
195 return count;
196}
