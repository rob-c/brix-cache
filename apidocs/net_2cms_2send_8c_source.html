25/* ngx_brix_cms_send_error — reply to a forwarded op that failed * WHAT: Sends a kYR_error reply frame: [4B big-endian ecode][text + NUL],
26 * echoing the request streamid. WHY: byte-exact with cmsd
27 * XrdCmsProtocol::Reply_Error — a data node answers a failed forwarded
28 * namespace op (Plane B) this way; success stays silent. HOW: pack the
29 * error code then the NUL-terminated text into a stack buffer and dispatch
30 * a CMS_RSP_ERROR frame. text is truncated to fit the frame cap. */
31ngx_int_t
32ngx_brix_cms_send_error(ngx_brix_cms_ctx_t *ctx, uint32_t streamid,
33 uint32_t ecode, const char *text)
34{
35 u_char buf[256];
36 size_t tlen;
37
38 tlen = (text != NULL) ? ngx_strlen(text) : 0;
39 if (tlen > sizeof(buf) - 5) { /* 4B ecode + text + NUL */
40 tlen = sizeof(buf) - 5;
41 }
42 ngx_brix_cms_put32(buf, ecode);
43 if (tlen > 0) {
44 ngx_memcpy(buf + 4, text, tlen);

47 return ngx_brix_cms_send_frame(ctx, streamid, CMS_RSP_ERROR, 0,

51/*
52 * ngx_brix_cms_send_login — initial CMS login frame with server capabilities.
53 *
54 * WHAT: Builds and sends the first CMS frame after establishing a TCP connection,
55 * reporting version, mode (client/manager), PID, filesystem space stats,
56 * listen port, exported paths, and reserved fields. WHY: The CMS manager
57 * needs this information to decide whether to route client requests here;
58 * without login the server is invisible in the cluster registry. HOW: Call
59 * stat_space() for total_gb/free_mb/util_pct (aggregate if manager mode),
60 * pack type/value payload fields in wire order via put_short/put_int, append
61 * exported paths from cms_export_paths(), send frame with CMS_RR_LOGIN code. */
62
63ngx_int_t
64ngx_brix_cms_send_login(ngx_brix_cms_ctx_t *ctx)
65{
66 u_char payload[1024];
67 u_char *payload_cursor;
68 u_char sid[256];
69 u_char pathbuf[640];
70 u_char hostbuf[200];
71 u_char *pp;
72 ngx_str_t paths;
73 size_t sid_len;
74 size_t path_len;
75 size_t i;
76 size_t seg_start;
77 uint32_t total_gb;
78 uint32_t free_mb;
79 uint32_t util_pct;
80 u_char ptype;
81
82 /*
83 * Build the exported-path list in the real XrdCms wire format: newline-
84 * separated "<type> <path>" entries, type 'w' if writable else 'r'
85 * (e.g. "all.export / r/w" -> "w /"). brix_cms_paths may carry several
86 * colon-separated namespace prefixes ("/data:/atlas").
87 */
88 paths = ngx_brix_cms_export_paths(ctx->conf);
89 ptype = ctx->conf->common.allow_write ? (u_char) 'w' : (u_char) 'r';
90 pp = pathbuf;
91 seg_start = 0;
92 /*
93 * Loop runs to i == paths.len (one past the last byte) so the final
94 * segment is flushed by the same branch that handles a ':' delimiter — no
95 * separate post-loop emit needed. The "+ 4" slack in the bound check
96 * reserves room for the per-segment framing this iteration may append:
97 * one '\n' separator + the type char + a space (3) plus a margin byte.
98 */
99 for (i = 0; i <= paths.len; i++) {
100 if (i == paths.len || paths.data[i] == ':') {
101 size_t seglen = i - seg_start;
102 if (seglen > 0
103 && (size_t) (pp - pathbuf) + seglen + 4 < sizeof(pathbuf))
104 {
105 if (pp != pathbuf) {
106 *pp++ = '\n';
107 }
108 *pp++ = ptype;
109 *pp++ = ' ';
110 ngx_memcpy(pp, paths.data + seg_start, seglen);
111 pp += seglen;
112 }
113 seg_start = i + 1;
114 }
115 }
116 path_len = (size_t) (pp - pathbuf);
117
118 /*
119 * Server identity "<host>:<dport>" — opaque to the manager but must be
120 * stable and reasonably unique for this node.
121 */
122 if (gethostname((char *) hostbuf, sizeof(hostbuf)) != 0) {
123 ngx_memcpy(hostbuf, "nginx", sizeof("nginx"));
124 }
125 hostbuf[sizeof(hostbuf) - 1] = '\0';
126 sid_len = (size_t) (ngx_snprintf(sid, sizeof(sid), "%s:%d", hostbuf,
127 (int) ctx->conf->listen_port) - sid);
128
129 total_gb = 0;
130 free_mb = 0;
131 util_pct = 0;
132 (void) ngx_brix_cms_stat_space(ctx->conf, &total_gb, &free_mb,
133 &util_pct);
134
135 if (ctx->conf->manager_mode) {
136 uint32_t agg_free = 0, agg_util = 0;
137 brix_srv_aggregate_space(&agg_free, &agg_util);
138 if (agg_free > 0 || agg_util > 0) {
139 free_mb = agg_free;
140 util_pct = agg_util;
141 }
142 }
143
144 /*
145 * kYR_login payload — XrdOucPup wire order (CmsLoginData):
146 * Version sh, Mode int, HoldTime int, tSpace int, fSpace int, mSpace int,
147 * fsNum sh, fsUtil sh, dPort sh, sPort sh, [Fence],
148 * SID str, Paths str, ifList str, envCGI str.
149 * Scalars carry a 0x80/0xa0 type tag; strings are bare [len][data+NUL].
150 * The frame header datalen (set by send_frame) is the total payload length.
151 */
152 payload_cursor = payload;
153 payload_cursor = ngx_brix_cms_put_short(payload_cursor,
154 CMS_LOGIN_VERSION);
155 payload_cursor = ngx_brix_cms_put_int(
156 payload_cursor,
157 CMS_LOGIN_MODE
158 | (ctx->conf->manager_mode ? CMS_LOGIN_MODE_MANAGER : 0));
159 payload_cursor = ngx_brix_cms_put_int(payload_cursor,
160 (uint32_t) getpid());
161 /* tSpace/fSpace: total disk (GB) and free (MB) reported to the manager. */
162 payload_cursor = ngx_brix_cms_put_int(payload_cursor, total_gb);
163 payload_cursor = ngx_brix_cms_put_int(payload_cursor, free_mb);
164 /*
165 * mSpace: minimum-free threshold (MB). Constant, not a live measurement —
166 * it is the policy floor below which the manager should stop selecting us.
167 */
168 payload_cursor = ngx_brix_cms_put_int(payload_cursor,
169 NGX_BRIX_CMS_MIN_FREE_MB);
170 /* fsNum: number of exported filesystems — we present a single namespace. */
171 payload_cursor = ngx_brix_cms_put_short(payload_cursor, 1);
172 payload_cursor = ngx_brix_cms_put_short(payload_cursor,
173 (uint16_t) util_pct);
174 /*
175 * dPort: the data port clients are redirected to (our listen port).
176 * sPort follows as 0 — we run no separate subscriber/admin port, so the
177 * field is present in wire order but advertised as unused.
178 */
179 payload_cursor = ngx_brix_cms_put_short(
180 payload_cursor, (uint16_t) ctx->conf->listen_port);
181 payload_cursor = ngx_brix_cms_put_short(payload_cursor, 0); /* sPort */
182
183 /*
184 * Trailing Pup strings in wire order: SID, Paths, then ifList and envCGI.
185 * The two NULL/0 puts emit empty (but present) ifList and envCGI strings —
186 * the manager still expects the length-prefixed slots, so they cannot be
187 * omitted even though we have nothing to advertise there.
188 */
189 payload_cursor = ngx_brix_cms_put_string(payload_cursor, sid, sid_len);
190 payload_cursor = ngx_brix_cms_put_string(payload_cursor, pathbuf,
191 path_len);
192 payload_cursor = ngx_brix_cms_put_string(payload_cursor, NULL, 0);
193 payload_cursor = ngx_brix_cms_put_string(payload_cursor, NULL, 0);
194
195 return ngx_brix_cms_send_frame(ctx, 0, CMS_RR_LOGIN, 0, payload,
196 (size_t) (payload_cursor - payload));
197}

215/*
216 * ngx_brix_cms_send_load — periodic load heartbeat report.
217 *
218 * WHAT: Builds and sends a CMS_LOAD frame reporting current free disk space in MB,
219 * optionally aggregated across all servers when in manager mode. WHY: The CMS
220 * manager uses load reports to distribute client requests proportionally to
221 * available capacity; stale or missing reports cause the manager to stop routing
222 * here. HOW: Call stat_space() for free_mb (aggregate via brix_srv_aggregate_space
223 * if manager mode), pack payload as put_short(6) + 6 zero bytes + put_int(free_mb),
224 * send with CMS_RR_LOAD code. */
225
226ngx_int_t
227ngx_brix_cms_send_load(ngx_brix_cms_ctx_t *ctx)
228{
229 u_char payload[32];
230 u_char *payload_cursor;
231 uint32_t free_mb;
232
233 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, ctx->cycle->log, 0,
234 "brix: send_load: conn=%p", ctx->connection);
235
236 free_mb = 0;
237 (void) ngx_brix_cms_stat_space(ctx->conf, NULL, &free_mb, NULL);
238
239 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, ctx->cycle->log, 0,
240 "brix: send_load: free_mb=%uD", free_mb);
241
242 if (ctx->conf->manager_mode) {
243 uint32_t agg_free = 0, agg_util = 0;
244 brix_srv_aggregate_space(&agg_free, &agg_util);
245 if (agg_free > 0) {
246 free_mb = agg_free;
247 }
248 }
249
250 /*
251 * kYR_load payload (CmsLoadRequest / lodArgs): theLoad is a Pup char-blob
252 * of 6 load bytes (cpu,net,xeq,mem,pag,dsk) — a bare [2-byte len][data] with
253 * NO scalar type tag — followed by dskFree as a tagged int. Zero load bytes
254 * report an idle node; the manager only uses them for balancing.
255 */
256 /*
257 * Hand-packed (put16 + manual cursor advance) rather than via put_string:
258 * put16 only writes the 2-byte length, so the cursor must be advanced by
259 * hand past both the length and the 6 data bytes before the tagged dskFree
260 * int is appended via put_int.
261 */
262 payload_cursor = payload;
263 ngx_brix_cms_put16(payload_cursor, 6); /* blob length: 6 load bytes */
264 payload_cursor += 2; /* skip the 2-byte length */
265 ngx_memzero(payload_cursor, 6); /* cpu,net,xeq,mem,pag,dsk = 0 */
266 payload_cursor += 6; /* advance past the 6 bytes */
267 payload_cursor = ngx_brix_cms_put_int(payload_cursor, free_mb);
268
269 return ngx_brix_cms_send_frame(ctx, 0, CMS_RR_LOAD, 0, payload,
270 (size_t) (payload_cursor - payload));
271}
