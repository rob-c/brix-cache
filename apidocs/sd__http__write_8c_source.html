1/*
2 * sd_http_write.c — write path for the HTTP-origin storage driver (SP3): the
3 * HTTP/WebDAV origin as a writable cache_store / stage_store.
4 *
5 * WHAT: The write-facing vtable slots — staged_open/staged_open_cred (buffer a
6 * whole object, capturing the per-user credential for the commit),
7 * staged_write (sequential append into the growable buffer), staged_commit
8 * (one PUT of the whole object — atomic from the reader's view),
9 * staged_abort, and unlink (a DELETE: eviction + post-flush stage cleanup).
10 *
11 * WHY: Split out of sd_http.c (phase-79 file-size split): the write path is one
12 * concept, distinct from the read/credential path (sd_http_read.c),
13 * selection/failover (sd_http_select.c), and the driver vtable/lifecycle
14 * (sd_http.c). Writes never fail over — a write to a non-primary origin
15 * would split-brain the store — so they always target endpoint 0 via
16 * sd_http_write_path (sd_http_select.c) rather than sd_http_request_fo.
17 *
18 * HOW: The per-user identity (bearer header / x509 client-cert path) is captured
19 * at staged_open and applied at commit — mirroring the read leg exactly,
20 * because an HTTP origin has no kernel fd / session to re-scope per user.
21 * cred_gate (sd_http_read.c) refuses a proxy-only cred the transport cannot
22 * present in deny mode, just like the read leg.
23 */

33/* #12 outbound integrity: base64(MD5(buf,len)) → out[25] ("" on failure). The
34 * classic RFC 1864 Content-MD5 the origin re-computes over the received body and
35 * rejects on mismatch (400/412) — the outbound analogue of the ingest
36 * s3_content_md5_verify gate. ngx-free (this backend layer builds into
37 * libxrdproto): OpenSSL EVP only, no ngx_* / ngx pool. */
38static void
39sd_http_content_md5(const void *buf, size_t len, char out[25])
40{
41 unsigned char md[EVP_MAX_MD_SIZE];
42 unsigned int mdlen = 0;
43
44 out[0] = '\0';
45 if (EVP_Digest(buf, len, md, &mdlen, EVP_md5(), NULL) != 1 || mdlen != 16) {
46 return;
47 }
48 EVP_EncodeBlock((unsigned char *) out, md, 16); /* 16 → 24 chars + NUL */
49}

51/* Per-staged-write state: HTTP has no streaming PUT through this transport, so the
52 * object is buffered and PUT whole at commit (a remote stage/cache store of typical
53 * file sizes; very large objects are a multipart follow-up).
54 *
55 * auth_hdr / cert_pem carry the per-open (per-user) credential captured at
56 * staged_open_cred so the commit PUT authenticates to the origin AS the requesting
57 * user rather than the static service credential (phase-70 §5.1, write leg). Both
58 * are COPIES — the cred fields are borrowed only for the staged_open() call. "" in
59 * either falls back to the instance static (is->auth_hdr) / no client cert. Exactly
60 * one kind is ever set (the VFS gate populates one of bearer / x509_proxy). */
61typedef struct {
62 char path[SD_HTTP_PATH_MAX];
63 u_char *buf;
64 size_t len;
65 size_t cap;
66 char auth_hdr[SD_HTTP_AUTH_MAX];
67 char cert_pem[SD_HTTP_PATH_MAX];
68} sd_http_staged_state;

70/* sd_http_staged_open_common — shared staged-open path for the plain and
71 * credential-scoped staged-open slots.
72 *
73 * WHAT: Allocates the staged buffer state, composes the write-target URL path
74 * and, when a `cred` is present, captures the per-user credential (bearer →
75 * Authorization header; x509 proxy → mutual-TLS client-cert PATH) into the
76 * staged state so the commit PUT presents THAT identity, not the static
77 * service credential.
78 * WHY: Phase-70 §5.1 write leg — an HTTP/WebDAV origin authenticates a PUT purely
79 * on the request credential, and this driver has no kernel fd / session to
80 * re-scope per user, so the per-user identity travels as staged state copied
81 * at open time (mirroring the read-leg sd_http_open_common exactly).
82 * HOW: cred==NULL (plain .staged_open) leaves auth_hdr AND cert_pem empty, so the
83 * commit falls back to the instance static header and no client cert. The
84 * same cred_gate the read path uses refuses a proxy-only cred in deny mode
85 * when the transport cannot mutual-TLS (request_cred==NULL) → EACCES. A
86 * bearer is snprintf'd into the staged auth_hdr; an x509 proxy path is copied
87 * into cert_pem only when the transport can present it (request_cred!=NULL).
88 * Both are COPIES — cred fields are borrowed only for this call. */
89static brix_sd_staged_t *
90sd_http_staged_open_common(brix_sd_instance_t *inst, const char *final_path,
91 mode_t mode, const brix_sd_cred_t *cred, int *err_out)
92{
93 sd_http_inst_state *is = inst->state;
94 sd_http_staged_state *ss;
95 brix_sd_staged_t *h;
96
97 (void) mode;
98
99 /* Same credential gate the read leg applies: a proxy-only cred that cannot be
100 * presented as a client cert must be refused in deny mode rather than served
101 * on the anonymous/service credential. */
102 if (sd_http_cred_gate(is, cred) != 0) {
103 if (err_out) { *err_out = errno; }
104 return NULL;
105 }
106
107 ss = calloc(1, sizeof(*ss));
108 h = calloc(1, sizeof(*h));
109 if (ss == NULL || h == NULL) {
110 free(ss);
111 free(h);
112 if (err_out) { *err_out = ENOMEM; }
113 return NULL;
114 }
115 sd_http_write_path(is, final_path, ss->path, sizeof(ss->path));
116
117 /* Capture the per-open credential for the commit PUT: bearer → header;
118 * x509 proxy → cert path (only when the transport can present it). Empty
119 * leaves the commit on the instance static / anonymous credential. */
120 if (cred != NULL && cred->bearer != NULL && cred->bearer[0] != '\0') {
121 snprintf(ss->auth_hdr, sizeof(ss->auth_hdr),
122 "Authorization: Bearer %s\r\n", cred->bearer);
123 }
124 if (cred != NULL && cred->x509_proxy != NULL && cred->x509_proxy[0] != '\0'
125 && is->transport->request_cred != NULL)
126 {
127 snprintf(ss->cert_pem, sizeof(ss->cert_pem), "%s", cred->x509_proxy);
128 }
129
130 h->inst = inst;
131 h->state = ss;
132 return h;
133}

160ssize_t
161sd_http_staged_write(brix_sd_staged_t *h, const void *buf, size_t len,
162 off_t off)
163{
164 sd_http_staged_state *ss = h->state;
165
166 /* Sequential append only (whole-object PUT has no random write). */
167 if ((size_t) off != ss->len) {
168 errno = ESPIPE;
169 return -1;
170 }
171 if (ss->len + len > ss->cap) {
172 size_t ncap = ss->cap ? ss->cap * 2 : (1u << 20);
173 u_char *nbuf;
174
175 while (ncap < ss->len + len) {
176 ncap *= 2;
177 }
178 nbuf = realloc(ss->buf, ncap);
179 if (nbuf == NULL) {
180 errno = ENOMEM;
181 return -1;
182 }
183 ss->buf = nbuf;
184 ss->cap = ncap;
185 }
186 ngx_memcpy(ss->buf + ss->len, buf, len);
187 ss->len += len;
188 return (ssize_t) len;
189}

191ngx_int_t
192sd_http_staged_commit(brix_sd_staged_t *h, int noreplace)
193{
194 sd_http_staged_state *ss = h->state;
195 sd_http_inst_state *is = h->inst->state;
196 brix_s3_resp_t resp;
197 char errbuf[256];
198 const char *auth_hdr;
199 const char *hdrs;
200 char hdr_block[SD_HTTP_AUTH_MAX + 64];
201 int rq;
202 ngx_int_t rc = NGX_OK;
203
204 (void) noreplace; /* HTTP PUT always replaces */
205
206 /* Per-user commit: a per-open bearer (staged_open_cred) wins over the
207 * instance's static bearer_token; "" (plain staged_open, or no usable cred)
208 * falls back. A per-open x509 proxy path (ss->cert_pem) is presented as the
209 * mutual-TLS client cert on the PUT via the transport's request_cred slot —
210 * exactly mirroring the read leg (sd_http_request_fo / sd_http_pread). The
211 * cred gate at staged_open already refused a proxy-only cred that cannot be
212 * presented in deny mode, so reaching here with a cert path guarantees a
213 * request_cred-capable transport. */
214 auth_hdr = ss->auth_hdr[0] ? ss->auth_hdr
215 : (is->auth_hdr[0] ? is->auth_hdr : NULL);
216
217 /* #12 outbound integrity: when put_checksum is on, prepend a Content-MD5 line
218 * (base64 MD5 of the whole staged object) to the header block so the origin
219 * re-computes it over the received body and rejects a wire-corrupted PUT with
220 * 400/412 rather than silently committing poison. The transport splits this
221 * CRLF block into request headers (s3o_build_slist). Off ⇒ the auth header
222 * passes through unchanged (byte-frozen prior behaviour). */
223 hdrs = auth_hdr;
224 if (is->put_checksum) {
225 char md5b64[25];
226
227 sd_http_content_md5(ss->buf, ss->len, md5b64);
228 if (md5b64[0] != '\0') {
229 snprintf(hdr_block, sizeof(hdr_block), "Content-MD5: %s\r\n%s",
230 md5b64, auth_hdr ? auth_hdr : "");
231 hdrs = hdr_block;
232 }
233 }
234
235 if (ss->cert_pem[0] != '\0' && is->transport->request_cred != NULL) {
236 rq = is->transport->request_cred(is->tctx, is->eps[0].host,
237 is->eps[0].port, is->eps[0].tls, "PUT",
238 ss->path, hdrs, ss->buf, ss->len,
239 is->timeout_ms, ss->cert_pem, &resp,
240 errbuf, sizeof(errbuf));
241 } else {
242 rq = is->transport->request(is->tctx, is->eps[0].host, is->eps[0].port,
243 is->eps[0].tls, "PUT",
244 ss->path, hdrs,
245 ss->buf, ss->len, is->timeout_ms, &resp,
246 errbuf, sizeof(errbuf));
247 }
248 if (rq != 0) {
249 free(ss->buf);
250 free(ss);
251 free(h);
252 errno = EIO;
253 return NGX_ERROR;
254 }
255 if (resp.status != 200 && resp.status != 201 && resp.status != 204) {
256 errno = (resp.status == 403 || resp.status == 401) ? EACCES : EIO;
257 rc = NGX_ERROR;
258 }
259 is->transport->resp_free(&resp);
260 free(ss->buf);
261 free(ss);
262 free(h);
263 return rc;
264}

289 errbuf, sizeof(errbuf)) != 0)
290 {
291 errno = EIO;
292 return NGX_ERROR;
293 }
294 /* Idempotent: 204/200 ok, 404 already gone. */
295 if (resp.status != 204 && resp.status != 200 && resp.status != 404) {
296 is->transport->resp_free(&resp);

298 return NGX_ERROR;
299 }
300 is->transport->resp_free(&resp);
301 return NGX_OK;
