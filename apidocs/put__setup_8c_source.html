66/* Add an "X-Upload-Offset: <off>" response header (the resumable-PUT progress
67 * marker the client reads to know where to continue). Best-effort. */
68static void
69webdav_set_upload_offset(ngx_http_request_t *r, off_t off)
70{
71 ngx_table_elt_t *h = ngx_list_push(&r->headers_out.headers);
72 if (h == NULL) {
73 return;
74 }
75 h->hash = 1;
76 ngx_str_set(&h->key, "X-Upload-Offset");
77 h->value.data = ngx_pnalloc(r->pool, NGX_OFF_T_LEN);
78 if (h->value.data == NULL) {
79 h->hash = 0;
80 return;
81 }
82 h->value.len = ngx_sprintf(h->value.data, "%O", off) - h->value.data;
83}

85/*
86 * Resumable Content-Range PUT (brix_webdav_upload_resume on). Writes this
87 * chunk to a persistent, identity-keyed partial at its absolute offset and
88 * commits (atomic rename) only when the upload is complete. Across separate PUT
89 * requests — and across an nginx restart — the partial survives, so the client
90 * resumes from X-Upload-Offset. Append-only: the chunk must start exactly at the
91 * current partial size, else 409 + X-Upload-Offset tells the client the truth.
92 */
93static void
94webdav_put_ranged_resume(ngx_http_request_t *r,
95 ngx_http_brix_webdav_loc_conf_t *conf, const char *path,
96 const brix_http_content_range_t *cr)
97{
98 brix_staged_file_t staged;
99 off_t cur = 0;
100 brix_http_body_summary_t bs;
101 ngx_http_brix_webdav_req_ctx_t *wctx =
102 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
103 const char *principal = (wctx != NULL && wctx->dn[0] != '\0')
104 ? wctx->dn : NULL;
105 const char *stage_dir = conf->upload_stage_dir_canon[0]
106 ? conf->upload_stage_dir_canon : NULL;
107
108 {
109 brix_staged_open_req_t staged_req = {
110 .root_canon = conf->common.root_canon,
111 .final_path = path,
112 .mode = NGX_FILE_DEFAULT_ACCESS,
113 .principal = principal,
114 .stage_dir = stage_dir,
115 };
116 if (brix_staged_open_resume(r->connection->log, &staged_req, &staged, &cur)
117 != NGX_OK)
118 {
119 ngx_int_t st = brix_http_map_errno(ngx_errno);
120 webdav_metrics_finalize_request(r,
121 st >= 500 ? NGX_HTTP_INTERNAL_SERVER_ERROR : st);
122 return;
123 }
124
125 /* Append-only contiguity: the chunk must begin at the current partial size.
126 * If it doesn't, report the real resume offset and let the client re-issue —
127 * keep the partial (close without unlink). */
128 if (cr->start != cur) {
129 brix_staged_abort(r->connection->log, conf->common.root_canon,
130 &staged, 0 /* keep */);
131 webdav_set_upload_offset(r, cur);
132 webdav_send_status_only(r, NGX_HTTP_CONFLICT);
133 return;
134 }
135
136 if (r->request_body != NULL) {
137 if (brix_http_body_write_to_fd_at(r, staged.fd, path, &bs, cr->start)
138 != NGX_OK)
139 {
140 /* keep the partial; the client can re-send this chunk */
141 brix_staged_abort(r->connection->log, conf->common.root_canon,
142 &staged, 0 /* keep */);
143 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
144 return;
145 }
146 BRIX_WEBDAV_METRIC_ADD(bytes_rx_total, bs.bytes);
147 cur = cr->end + 1;
148 /* Flush so the reported resume offset is durable across a restart. */
149 (void) fsync(staged.fd);
150 }
151
152 /* Final chunk (last byte of the declared total) → commit; else keep partial
153 * and report the next expected offset. */
154 if (cr->total >= 0 && cr->end + 1 >= cr->total) {
155 /* Commit the staged partial onto the destination — atomic rename on the
156 * same filesystem, or copy-then-rename when staging on a different device
157 * (brix_webdav_stage_dir). brix_commit_staged owns neither close nor
158 * the active flag, so finalize the staged struct ourselves. On a stage
159 * device, record a durable pending-commit marker first so an interrupted
160 * cross-device move is finished by the reaper across a restart. */
161 ngx_flag_t stage_tracked = (stage_dir != NULL);
162 ngx_int_t crc;
163 if (stage_tracked) {
164 (void) brix_stage_mark_pending(staged.tmp_path, path,
165 r->connection->log);
166 }
167 crc = brix_commit_staged(staged.fd, staged.tmp_path, path,
168 staged.final_mode, r->connection->log);
169 if (staged.fd != NGX_INVALID_FILE) {
170 ngx_close_file(staged.fd);
171 staged.fd = NGX_INVALID_FILE;
172 }
173 staged.active = 0;
174 if (crc != NGX_OK) {
175 /* Keep the marker + partial — the reaper / a client retry completes it. */
176 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
177 return;
178 }
179 if (stage_tracked) {
180 brix_stage_unmark_pending(staged.tmp_path);
181 }
182 webdav_put_persist_checksums(r, (const char *) path); /* §8.3 */
183 webdav_send_status_only(r, NGX_HTTP_CREATED);
184 return;
185 }
186
187 brix_staged_abort(r->connection->log, conf->common.root_canon,
188 &staged, 0 /* keep partial for the next chunk */);
189 webdav_set_upload_offset(r, cur);
190 webdav_send_status_only(r, NGX_HTTP_OK);
191 }
192}

194/*
195 * webdav_put_precheck — resolve the target and run pre-write gating.
196 *
197 * WHAT: Resolves the request path into `path`, dispatches a resumable
198 * Content-Range PUT (which fully handles its own response), probes the target
199 * for existence, and evaluates the ETag preconditions. Returns
200 * WEBDAV_PUT_CONTINUE to proceed to the staged-write path with `*created`
201 * set, or WEBDAV_PUT_DONE when the request has already been answered (path
202 * error → 409/4xx, resumable handled, or an ETag precondition failed).
203 *
204 * WHY: Consolidates every gate that must pass before a staged temp is opened,
205 * keeping the orchestrator a flat sequence and the "never open before we know
206 * we should" invariant in one place. Behavior is byte-identical to the prior
207 * inline block (same status codes, same order of checks).
208 *
209 * HOW:
210 * 1. Resolve the path; a missing intermediate collection is RFC 4918 §9.7.1
211 * 409 Conflict, any other resolve error propagates its status.
212 * 2. If resumable uploads are on and a Content-Range header is present, parse
213 * it (malformed → 400) and hand off to webdav_put_ranged_resume; DONE.
214 * 3. Probe the target through the metered VFS to populate a local stat for the
215 * ETag check and decide `*created` (0 = exists, 1 = new).
216 * 4. Run the weak-ETag precondition check; a failure sends its status. DONE.
217 */
218webdav_put_step_t
219webdav_put_precheck(ngx_http_request_t *r,
220 ngx_http_brix_webdav_loc_conf_t *conf, char *path, int *created)
221{
222 struct stat sb;
223 ngx_int_t rc;
224
225 *created = 0;
226 ngx_memzero(&sb, sizeof(sb));
227
228 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon, path,
229 WEBDAV_MAX_PATH);
230 if (rc != NGX_OK) {
231 /* RFC 4918 §9.7.1: missing intermediate collection → 409 Conflict */
232 if (rc == NGX_HTTP_NOT_FOUND) {
233 webdav_metrics_finalize_request(r, NGX_HTTP_CONFLICT);
234 return WEBDAV_PUT_DONE;
235 }
236 webdav_metrics_finalize_request(r, rc);
237 return WEBDAV_PUT_DONE;
238 }
239
240 /*
241 * Resumable upload: a Content-Range PUT places its chunk at an absolute
242 * offset on a persistent identity-keyed partial and commits only when the
243 * upload completes (brix_webdav_upload_resume on). Handled before the
244 * whole-body staged-write path below. A malformed Content-Range is 400.
245 */
246 if (conf->upload_resume) {
247 ngx_table_elt_t *crh = brix_http_find_header(
248 r, "Content-Range", sizeof("Content-Range") - 1);
249 if (crh != NULL && crh->value.len > 0) {
250 brix_http_content_range_t cr;
251 brix_http_parse_content_range(crh->value.data, crh->value.len, &cr);
252 if (!cr.present) {
253 webdav_metrics_finalize_request(r, NGX_HTTP_BAD_REQUEST);
254 return WEBDAV_PUT_DONE;
255 }
256 webdav_put_ranged_resume(r, conf, path, &cr);
257 return WEBDAV_PUT_DONE;
258 }
259 }
260
261 {
262 ngx_http_brix_webdav_req_ctx_t *rx =
263 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
264 brix_vfs_ctx_t pctx;
265 brix_vfs_stat_t pst;
266 int is_tls = 0;
267
268#if (NGX_HTTP_SSL)
269 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
270#endif
271 brix_vfs_ctx_init(&pctx, r->pool, r->connection->log,
272 BRIX_PROTO_WEBDAV, conf->common.root_canon, conf->cache_root_canon,
273 conf->common.allow_write, is_tls,
274 (rx != NULL) ? rx->identity : NULL, path);
275 brix_vfs_ctx_bind_backend_cred(&pctx,
276 &conf->common.storage_credential_dir,
277 conf->common.storage_credential_fallback);
278 webdav_vfs_bind_deleg(r, conf, &pctx);
279 if (brix_vfs_probe(&pctx, 0 /* follow */, &pst) == NGX_OK) {
280 *created = 0; /* target exists — sb populated for the ETag check */
281 sb.st_mtime = pst.mtime;
282 sb.st_size = pst.size;
283 sb.st_mode = (mode_t) pst.mode;
284 } else {
285 *created = 1; /* new file — sb unused (etag check sees !exists) */
286 }
287 }
288
289 rc = brix_http_check_etag_preconditions(
290 r, !*created, &sb, BRIX_ETAG_WEAK, BRIX_HTTP_COND_WEAK_EQUIV);
291 if (rc != NGX_OK) {
292 webdav_metrics_finalize_request(r, rc);
293 return WEBDAV_PUT_DONE;
294 }
295
296 return WEBDAV_PUT_CONTINUE;
297}

299/*
300 * webdav_put_open_target — open the unified write session for the final path.
301 *
302 * WHAT: Builds the staged-write VFS ctx and opens a brix_vfs_writer session on
303 * it with BRIX_VFS_O_ATOMIC (force the staged temp+publish path so a failed
304 * write never leaves a partial at the final path) and the export's
305 * verify-on-write setting. On success returns WEBDAV_PUT_CONTINUE with
306 * `*out_writer` set; on failure maps the captured errno to a clean 4xx (or 500
307 * for no-4xx-contract errnos), sends it, and returns WEBDAV_PUT_DONE.
308 *
309 * WHY: Keeps the "body is written to a staged temp and atomically renamed only
310 * on success" invariant (WS8/HTTP-4) and its errno→status mapping in one
311 * place, matching S3 PUT's s3_put_finalize_fs_error. Routing the write
312 * through the writer gives WebDAV the same common verified-write call every
313 * backend shares. Status codes unchanged.
314 *
315 * HOW:
316 * 1. Init the staged-write ctx via webdav_put_vfs_ctx_init.
317 * 2. brix_vfs_writer_open(O_ATOMIC, verify_write); a non-NULL result succeeds.
318 * 3. ENOENT/ENOTDIR (missing parent collection) → 409 (RFC 4918 §9.7.1).
319 * 4. Otherwise map errno via brix_http_map_errno, log the path, and finalize
320 * with the mapped 4xx or 500.
321 */
322webdav_put_step_t
323webdav_put_open_target(ngx_http_request_t *r,
324 ngx_http_brix_webdav_loc_conf_t *conf, const char *path,
325 brix_vfs_ctx_t *vctx, brix_vfs_writer_t **out_writer)
326{
327 brix_vfs_writer_t *writer;
328 int staged_err = 0;
329 ngx_int_t status;
330
331 *out_writer = NULL;
332
333 /*
334 * Open a write session that stages a temp beside the final path (O_ATOMIC,
335 * O_EXCL, unique name, up to 16 attempts) through the metered VFS surface.
336 * The body is written here and atomically renamed onto the final path only
337 * on success — never exposing a partial object. verify_write folds a
338 * read-back integrity check on commit when the export opts in.
339 */
340 webdav_put_vfs_ctx_init(r, conf, path, vctx);
341
342 writer = brix_vfs_writer_open(vctx, BRIX_VFS_O_ATOMIC,
343 conf->common.verify_write, &staged_err);
344 if (writer != NULL) {
345 *out_writer = writer;
346 return WEBDAV_PUT_CONTINUE;
347 }
348
349 /* Use the errno the VFS open captured (logging may clobber errno). */
350 {
351 int open_errno = staged_err;
352
353 if (open_errno == NGX_ENOENT || open_errno == NGX_ENOTDIR) {
354 /* RFC 4918 §9.7.1 — PUT to non-existent parent collection is 409 */
355 webdav_metrics_finalize_request(r, NGX_HTTP_CONFLICT);
356 return WEBDAV_PUT_DONE;
357 }
358 /*
359 * A DAC-denied staged-temp create (e.g. the accessor lacks write on the
360 * parent collection → EACCES/EPERM, EXDEV/ELOOP on a confinement-blocked
361 * traversal) is a forbidden, bounded request — surface it as a clean 4xx
362 * via the shared errno map (EACCES/EPERM → 403, ENOSPC → 507, …), not a
363 * blanket 500. Only an errno with no clean 4xx contract (e.g. EIO) keeps
364 * the internal-error path. This mirrors S3 PUT's s3_put_finalize_fs_error.
365 */
366 status = brix_http_map_errno(open_errno);
367 brix_log_safe_path(r->connection->log, NGX_LOG_ERR,
368 (ngx_uint_t) open_errno,
369 "brix_webdav: staged open for write failed for: "
370 "\"%s\"", path);
371 webdav_metrics_finalize_request(
372 r, status >= 500 ? NGX_HTTP_INTERNAL_SERVER_ERROR : status);
373 return WEBDAV_PUT_DONE;
374 }
375}
