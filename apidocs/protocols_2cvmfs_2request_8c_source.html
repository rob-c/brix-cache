1/* request.c — proxy-mode target extraction + allowlist.
2 *
3 * WHAT: for absolute-form request lines ("GET http://s1:8000/cvmfs/..."),
4 * extract the authority nginx already parsed (r->host_start/end,
5 * r->port_start/end) and check it against brix_cvmfs_upstream_allow.
6 * WHY: a CVMFS site proxy is a proxy for the site's Stratum-1s ONLY —
7 * without the allowlist it is an open HTTP proxy.
8 * HOW: no parsing of our own: nginx validates absolute-form targets during
9 * request-line parsing and r->uri already holds just the path, so the
10 * classifier and every layer below see exactly the reverse-mode shape.
11 */
12#include "cvmfs.h"

22 if (r->host_start == NULL) {
23 return NGX_DECLINED; /* origin-form: reverse mode */

25
26 /* cleartext only on cvmfs://: WLCG proxy traffic is plain HTTP; an
27 * https target means a misconfigured client, not a feature request.
28 * scvmfs:// (T22) lifts this on secure listeners: ctx->secure allows
29 * https authorities (and TLS upstream connects). */
30 {
31 ngx_http_brix_cvmfs_ctx_t *ctx =
32 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
33 size_t sl = (size_t) (r->schema_end - r->schema_start);
34 int is_http = (sl == 4
35 && ngx_strncasecmp(r->schema_start, (u_char *) "http", 4) == 0);
36 int is_https = (sl == 5
37 && ngx_strncasecmp(r->schema_start, (u_char *) "https", 5) == 0);
38
39 if (!is_http && !is_https) {
40 return NGX_HTTP_FORBIDDEN;
41 }
42 if (is_https && (ctx == NULL || !ctx->secure)) {
43 return NGX_HTTP_FORBIDDEN;
44 }
45 }
46
47 host->data = r->host_start;
48 host->len = (size_t) (r->host_end - r->host_start);
49
50 /* nginx (>= 1.11) keeps no port_start/port_end: for an absolute-form
51 * target the port digits sit in the raw request line between host_end
52 * (":") and uri_start ("/"), both of which the parser recorded. */
53 *port = 80;
54 if (r->uri_start != NULL && r->host_end != NULL
55 && r->host_end < r->uri_start && *r->host_end == ':')
56 {
57 p = ngx_atoi(r->host_end + 1,
58 (size_t) (r->uri_start - (r->host_end + 1)));
59 if (p < 1 || p > 65535) {
60 return NGX_HTTP_BAD_REQUEST;
61 }
62 *port = (in_port_t) p;
63 }
64
65 if (cc->upstream_allow == NULL || cc->upstream_allow->nelts == 0) {
66 /* allowlist unset = proxy mode off: absolute-form always refused */
67 return NGX_HTTP_FORBIDDEN;
68 }
69 allow = cc->upstream_allow->elts;
70 for (i = 0; i < cc->upstream_allow->nelts; i++) {
71 if (allow[i].len == host->len
72 && ngx_strncasecmp(allow[i].data, host->data, host->len) == 0)
73 {
74 return NGX_OK;
75 }
76 }
77 return NGX_HTTP_FORBIDDEN;
78}
