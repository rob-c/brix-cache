1/* request.c — proxy-mode target extraction + allowlist.
2 *
3 * WHAT: for absolute-form request lines ("GET http://s1:8000/cvmfs/..."),
4 * extract the authority nginx already parsed (r->host_start/end,
5 * r->port_start/end) and check it against brix_cvmfs_upstream_allow.
6 * WHY: a CVMFS site proxy is a proxy for the site's Stratum-1s ONLY —
7 * without the allowlist it is an open HTTP proxy.
8 * HOW: no parsing of our own: nginx validates absolute-form targets during
9 * request-line parsing and r->uri already holds just the path, so the
10 * classifier and every layer below see exactly the reverse-mode shape.
11 */
12#include "cvmfs.h"

14/* ---- Validate the absolute-form request scheme ----
15 *
16 * WHAT: returns NGX_OK when the absolute-form target scheme is acceptable on
17 * this listener, else NGX_HTTP_FORBIDDEN. http:// is always allowed;
18 * https:// is allowed only on a secure (scvmfs://) listener.
19 *
20 * WHY: WLCG proxy traffic is plain HTTP, so an https target on a cleartext
21 * cvmfs:// listener means a misconfigured client, not a feature request.
22 * The secure (T22) listener lifts this to permit https authorities (and
23 * TLS upstream connects), gated on ctx->secure.
24 *
25 * HOW: 1. Fetch the per-request cvmfs ctx (carries the secure flag).
26 * 2. Classify the parsed schema token as exactly "http" or "https".
27 * 3. Refuse anything that is neither.
28 * 4. Refuse https unless the ctx exists and is marked secure.
29 */
30static ngx_int_t
31brix_cvmfs_proxy_scheme_check(ngx_http_request_t *r)
32{
33 ngx_http_brix_cvmfs_ctx_t *ctx =
34 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
35 size_t sl = (size_t) (r->schema_end - r->schema_start);
36 int is_http = (sl == 4
37 && ngx_strncasecmp(r->schema_start, (u_char *) "http", 4) == 0);
38 int is_https = (sl == 5
39 && ngx_strncasecmp(r->schema_start, (u_char *) "https", 5) == 0);
40
41 if (!is_http && !is_https) {
42 return NGX_HTTP_FORBIDDEN;
43 }
44 if (is_https && (ctx == NULL || !ctx->secure)) {
45 return NGX_HTTP_FORBIDDEN;
46 }
47 return NGX_OK;
48}

50/* ---- Recover the target port from the raw request line ----
51 *
52 * WHAT: sets *port to the absolute-form target's port, defaulting to 80 when
53 * no ":port" is present. Returns NGX_OK, or NGX_HTTP_BAD_REQUEST when the
54 * port digits are out of the 1..65535 range.
55 *
56 * WHY: nginx (>= 1.11) keeps no port_start/port_end for absolute-form
57 * targets, so the port must be read back from the request line between
58 * host_end (":") and uri_start ("/") — both recorded by the parser.
59 *
60 * HOW: 1. Default the port to 80.
61 * 2. When a ':' separates host_end from uri_start, parse the digits.
62 * 3. Reject an out-of-range value; otherwise publish it.
63 */
64static ngx_int_t
65brix_cvmfs_proxy_parse_port(ngx_http_request_t *r, in_port_t *port)
66{
67 ngx_int_t p;
68
69 *port = 80;
70 if (r->uri_start != NULL && r->host_end != NULL
71 && r->host_end < r->uri_start && *r->host_end == ':')
72 {
73 p = ngx_atoi(r->host_end + 1,
74 (size_t) (r->uri_start - (r->host_end + 1)));
75 if (p < 1 || p > 65535) {
76 return NGX_HTTP_BAD_REQUEST;
77 }
78 *port = (in_port_t) p;
79 }
80 return NGX_OK;
81}

83/* ---- Match the target host against the Stratum-1 allowlist ----
84 *
85 * WHAT: returns NGX_OK when host is listed in cc->upstream_allow (case-
86 * insensitive), else NGX_HTTP_FORBIDDEN.
87 *
88 * WHY: the allowlist is the sole thing separating a CVMFS site proxy from an
89 * open HTTP proxy; an unset or empty allowlist means proxy mode is off,
90 * so every absolute-form target is refused.
91 *
92 * HOW: 1. Treat an unset/empty allowlist as proxy-off and refuse.
93 * 2. Scan the entries for a length-exact, case-insensitive host match.
94 * 3. Return NGX_OK on a hit, NGX_HTTP_FORBIDDEN otherwise.
95 */
96static ngx_int_t
97brix_cvmfs_proxy_host_allowed(const brix_cvmfs_conf_t *cc, const ngx_str_t *host)
98{
99 ngx_str_t *allow;
100 ngx_uint_t i;
101
102 if (cc->upstream_allow == NULL || cc->upstream_allow->nelts == 0) {
103 /* allowlist unset = proxy mode off: absolute-form always refused */
104 return NGX_HTTP_FORBIDDEN;
105 }
106 allow = cc->upstream_allow->elts;
107 for (i = 0; i < cc->upstream_allow->nelts; i++) {
108 if (allow[i].len == host->len
109 && ngx_strncasecmp(allow[i].data, host->data, host->len) == 0)
110 {
111 return NGX_OK;
112 }
113 }
114 return NGX_HTTP_FORBIDDEN;
115}

117/* ---- Extract and authorize an absolute-form proxy target ----
118 *
119 * WHAT: for an absolute-form request line, fills host/port with the target
120 * authority and returns NGX_OK when it is an allowed Stratum-1. Returns
121 * NGX_DECLINED for origin-form (reverse mode), or an HTTP error status
122 * (FORBIDDEN/BAD_REQUEST) when the scheme, port, or host is rejected.
123 *
124 * WHY: a CVMFS site proxy proxies the site's Stratum-1s ONLY; this gate keeps
125 * it from acting as an open HTTP proxy while leaving reverse-mode
126 * requests to fall through to the classifier.
127 *
128 * HOW: 1. Absent host_start means origin-form → NGX_DECLINED.
129 * 2. Enforce the scheme policy (http always, https only when secure).
130 * 3. Publish the host span nginx already parsed.
131 * 4. Recover the target port from the request line.
132 * 5. Authorize the host against the Stratum-1 allowlist.
133 */
134ngx_int_t
135brix_cvmfs_proxy_target(ngx_http_request_t *r, const brix_cvmfs_conf_t *cc,
136 ngx_str_t *host, in_port_t *port)
137{
138 ngx_int_t rc;
139
140 if (r->host_start == NULL) {
141 return NGX_DECLINED; /* origin-form: reverse mode */
142 }
143
144 rc = brix_cvmfs_proxy_scheme_check(r);
145 if (rc != NGX_OK) {
146 return rc;
147 }
148
149 host->data = r->host_start;
150 host->len = (size_t) (r->host_end - r->host_start);
151
152 rc = brix_cvmfs_proxy_parse_port(r, port);
153 if (rc != NGX_OK) {
154 return rc;
155 }
156
157 return brix_cvmfs_proxy_host_allowed(cc, host);
158}
