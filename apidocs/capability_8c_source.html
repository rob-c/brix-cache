1/*
2 * capability.c — path/template capability matching (XrdAccCapability).
3 *
4 * WHAT: brix_acc_cap_privs() walks one identity's capability list and, on the
5 * FIRST path whose prefix matches the request path, OR-s that capability's
6 * positive and negative privileges into the accumulator and returns 1.
7 * Template (`@=`) capabilities substitute a runtime value (the user name) into
8 * the path before matching; template-indirection capabilities recurse into a
9 * named template's own list.
10 *
11 * WHY: faithful port of XrdAccCapability::Privs()/Subcomp(). Two fidelity
12 * points vs the `native` engine: (1) matching is a RAW prefix (strncmp), so
13 * `/foo` matches `/foobar` — XrdAcc does not enforce component boundaries; and
14 * (2) within one identity's list the FIRST match wins, not the longest, so
15 * authdb authors order specific paths before general ones.
16 *
17 * HOW: for each node — if it is a template indirection, recurse; else if the
18 * request is at least as long as the prefix and matches (plain strncmp, or
19 * Subcomp when a substitution value is supplied), accumulate and stop.
20 */
21

24/*
25 * Subcomp — match a `@=` template path against (path, pathsub).
26 * Ports XrdAccCapability::Subcomp(): the fixed prefix before `@=` must match,
27 * the substitution value must appear at that point in the request path, and the
28 * fixed tail after `@=` must match what follows. Returns 1 on match.
29 */
30static int
31brix_acc_cap_subcomp(const brix_acc_cap_t *cap, const char *path, int plen,
32 const char *pathsub, int sublen)
33{
34 int ncmp;
35
36 /* Prefix before "@=" must match. */
37 if (ngx_strncmp(path, cap->path, cap->pins) != 0) {
38 return 0;
39 }
40
41 /* The substitution value must appear at the insertion point. */
42 if (ngx_strncmp(path + cap->pins, pathsub, sublen) != 0) {
43 return 0;
44 }
45
46 /* Enough room for the fixed tail? */
47 ncmp = cap->pins + sublen;
48 if ((plen - ncmp) < cap->prem) {
49 return 0;
50 }
51
52 /* Match the fixed tail after "@=" (cap->path[pins+2 ..]). */
53 if (cap->prem) {
54 return ngx_strncmp(cap->path + cap->pins + 2, path + ncmp,
55 (size_t) cap->prem) == 0;
56 }
57 return 1;
58}

60int
61brix_acc_cap_privs(brix_acc_cap_t *cap, brix_acc_priv_caps_t *out,
62 const char *path, int plen, const char *pathsub)
63{
64 int psl = (pathsub != NULL) ? (int) ngx_strlen(pathsub) : 0;
65
66 for (; cap != NULL; cap = cap->next) {
67
68 /* Template indirection: search the referenced template's list. */
69 if (cap->tmpl != NULL) {
70 if (brix_acc_cap_privs(cap->tmpl, out, path, plen, pathsub)) {
71 return 1;
72 }
73 continue;
74 }
75
76 if (plen < cap->plen) {
77 continue;
78 }
79
80 if ((pathsub == NULL && ngx_strncmp(path, cap->path, cap->plen) == 0)
81 || (pathsub != NULL
82 && brix_acc_cap_subcomp(cap, path, plen, pathsub, psl)))
83 {
84 out->pprivs |= cap->caps.pprivs;
85 out->nprivs |= cap->caps.nprivs;
86 return 1;
87 }
88 }
89
90 return 0;
91}
