119/*
120 * Try to acquire the cross-worker eviction lock (an O_CREAT|O_EXCL sentinel
121 * file under conf->cache_root). On success writes the sentinel path into
122 * lock_path (caller-owned buffer of lock_pathsz bytes); the caller must later
123 * release it with brix_cache_evict_unlock(). A sentinel older than
124 * conf->cache_lock_timeout seconds is treated as stale and reclaimed.
125 * Returns NGX_OK (lock held), NGX_DECLINED (another worker holds a fresh lock
126 * or a race lost the reclaim — skip eviction), or NGX_ERROR (path too long /
127 * unexpected OS error; errno set).
128 */
129ngx_int_t brix_cache_try_evict_lock(ngx_stream_brix_srv_conf_t *conf,
130 char *lock_path, size_t lock_pathsz, ngx_log_t *log);
131
132/*
