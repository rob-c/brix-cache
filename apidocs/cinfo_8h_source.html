4/*
5 * cinfo.h — block-present bitmap sidecar (the ".cinfo" record), Phase-58 §9.
6 *
7 * WHAT: A per-cached-file companion record, "<cachefile>.cinfo", carrying a
8 * fixed header (origin validity + cache access stats + optional origin
9 * digest) followed by a BLOCK-PRESENT BITMAP: one bit per fixed-size block
10 * (the slice granule), set when that block has been fetched from the
11 * origin and committed to the cache. This is the durable, inspectable
12 * record of *which parts of a file are cached* — XrdPfc's cinfo state
13 * vector, in this module's own versioned format.
14 *
15 * WHY: The slice cache stores each touched window as a separate file
16 * ("<cachefile>.__xrds_<k>_<idx>"); presence was therefore implicit (the
17 * file exists) and known only by scanning the directory. The .cinfo
18 * gives a single authoritative record so that: partial fills survive a
19 * restart (no re-fetch of windows already on disk), eviction can score by
20 * age/usage, and operators/tests can read exactly what a node holds. This
21 * header is *record-keeping*: it does not by itself change how reads are
22 * served (the per-slice files remain the served bytes).
23 *
24 * HOW: The header is a fixed-layout POD written/read verbatim (native byte
25 * order on the x86-64 target, exactly like meta.c); the bitmap of
26 * ceil(size/block_size) bits is appended immediately after it. A loader
27 * that finds the sidecar absent, short, or inconsistent returns
28 * NGX_DECLINED so the caller treats the file as un-recorded (safe: blocks
29 * simply look absent and would be refetched). Block bits are recorded by
30 * brix_cache_cinfo_record_block(), which serialises the
31 * read-modify-write with flock(2) so concurrent slice-fill workers never
32 * lose each other's bits.
33 */

59/*
60 * On-disk header. Field order is largest-alignment-first with an explicit
61 * `reserved` pad so the layout is stable across compilers; the record is zeroed
62 * before population so the trailing struct padding is deterministic on disk.
63 * The bitmap (brix_cache_cinfo_bitmap_len(nblocks) bytes) follows on disk
64 * immediately after sizeof(brix_cache_cinfo_t) bytes of header.
65 */
66typedef struct {
67 uint32_t magic; /* BRIX_CACHE_CINFO_MAGIC */
68 uint16_t version; /* BRIX_CACHE_CINFO_VERSION */
69 uint16_t flags; /* BRIX_CINFO_F_* */
70 uint32_t block_size; /* bytes per block (the slice granule); must be > 0 */
71 uint32_t mode; /* origin st_mode perm bits (0777); 0 = not recorded
72 * (pre-mode cinfos) — serve falls back to the store
73 * object's own bits. Was the 8-byte-align pad. */
74 uint64_t size; /* origin file size in bytes (validity) */
75 uint64_t mtime; /* origin mtime (validity) */
76 uint64_t nblocks; /* ceil(size/block_size) == bitmap bit count */
77 uint64_t access_count; /* cache hits served (stats parity with .meta) */
78 uint64_t bytes_served; /* cumulative bytes served from cache */
79 uint64_t last_access; /* unix secs of the most recent hit */
80 /* ---- write-back state (v3) — file-level, alongside the present bitmap ---- */
81 uint64_t dirty_lo; /* dirty byte-extent start (incl); lo==hi ⇒ clean */
82 uint64_t dirty_hi; /* dirty byte-extent end (excl) */
83 uint64_t dirty_since; /* unix secs the file first went dirty this episode */
84 uint64_t flush_gen; /* bumped on each successful write-back */
85 uint64_t last_flush; /* unix secs of the last successful write-back */
86 uint64_t bytes_flushed; /* cumulative mirrored bytes */
87 uint8_t etag_len;
88 char etag[BRIX_CACHE_META_ETAG_MAX]; /* origin etag, not NUL-term */
89 uint8_t cks_alg_len;
90 char cks_alg[16]; /* origin checksum algo name */
91 uint8_t cks_len;
92 char cks_hex[129]; /* origin checksum, hex */
93 /* ---- phase-68 trailer (appended LAST; version stays 3: readers accept
94 * files written without it — see cinfo.c's tolerant layout probe) ---- */
95 uint64_t expires_at; /* unix secs; entry stale when now >= expires_at
96 and F_EXPIRES is set. 0 + no flag = no TTL. */
97 uint64_t filled_at; /* unix secs the fill published this entry; keys
98 the bounded stale-if-error window (10x TTL). */
99} brix_cache_cinfo_t;
