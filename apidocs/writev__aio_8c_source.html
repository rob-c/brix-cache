1/*
2 * writev_aio.c — kXR_writev thread-pool offload path (split from writev.c,
3 * phase-79).
4 *
5 * WHAT: Holds writev_try_aio — the single function that, when a thread pool is
6 * configured, flattens the validated write_list descriptors into a
7 * self-contained decoded array, builds the writev AIO task, and posts it
8 * to a worker thread. The synchronous fallback, framing, handle admission,
9 * reply, and orchestrator stay in writev.c; the shared invariant context
10 * (writev_run_t) and this function's declaration live in writev_internal.h.
11 * WHY: writev.c was 551 lines and owned two distinct concerns — the
12 * event-loop-thread synchronous write and the thread-pool offload. The
13 * offload path is a cohesive, self-contained unit (big-endian decode +
14 * replay neutralisation + task build + buffer-ownership transfer), so
15 * lifting it into its own file holds each half under the 500-line cap and
16 * focused on one responsibility. Zero behaviour change: the code is moved
17 * verbatim and now reached through the shared header declaration.
18 * HOW: brix_handle_writev (writev.c) populates a writev_run_t and calls
19 * writev_try_aio here; on a successful post the done callback owns
20 * payload_buf and the reply, so the caller returns immediately.
21 */

28/* writev_try_aio — offload the whole vector to a worker thread if configured.
29 * WHAT: When a thread pool is configured, flattens the wire descriptors into a
30 * self-contained decoded array, builds the writev AIO task, and posts it.
31 * WHY: Keeps the per-segment big-endian decode + replay neutralisation on the
32 * event-loop thread (before the worker can race ctx) and hands buffer ownership
33 * to the task so the main thread can advance to the next request.
34 * HOW: data_ptr is the start of the contiguous per-segment data blocks. On a
35 * successful post it detaches payload_buf from ctx (the done callback frees it
36 * and replies) and reports posted via *posted (NGX_OK). It returns NGX_ERROR
37 * on allocation failure (caller drops the link). When no pool is configured or
38 * the post fails it leaves *posted 0 and returns NGX_OK so the caller falls
39 * through to the synchronous path with payload_buf still owned by ctx. */
40ngx_int_t
41writev_try_aio(const writev_run_t *run, int do_sync, ngx_flag_t *posted)
42{
43 brix_ctx_t *ctx = run->ctx;
44 ngx_connection_t *c = run->c;
45 const write_list *wl = run->wl;
46 size_t n_segs = run->n_segs;
47 ngx_stream_brix_srv_conf_t *conf;
48 ngx_thread_task_t *task;
49 brix_writev_aio_t *t;
50 brix_writev_seg_desc_t *seg_descs;
51 const u_char *dp = run->data_ptr;
52 size_t i;
53
54 *posted = 0;
55
56 conf = ngx_stream_get_module_srv_conf(
57 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
58
59 if (conf->common.thread_pool == NULL) {
60 return NGX_OK;
61 }
62
63 /* Flatten the wire descriptors into a self-contained array the worker
64 * thread can consume without touching ctx (which the main thread may mutate
65 * once we return). Decode all big-endian fields now, while we are still on
66 * the event-loop thread. */
67 seg_descs = ngx_palloc(c->pool,
68 n_segs * sizeof(brix_writev_seg_desc_t));
69 if (seg_descs == NULL) {
70 return NGX_ERROR;
71 }
72
73 for (i = 0; i < n_segs; i++) {
74 uint32_t wlen = (uint32_t) ntohl((uint32_t) wl[i].wlen);
75 /* phase77-fp: hidx (0..255) was already bounds-checked against
76 * BRIX_MAX_FILES for every segment by writev_validate_handles above,
77 * which the caller made all-or-nothing; the analyzer cannot correlate
78 * the two same-index loops. */
79 int hidx = (int)(unsigned char) wl[i].fhandle[0];
80 int64_t off = (int64_t) be64toh((uint64_t) wl[i].offset);
81
82 seg_descs[i].fd = ctx->files[hidx].fd; /* NOLINT(clang-analyzer-security.ArrayBound) */
83 seg_descs[i].obj = ctx->files[hidx].sd_obj; /* Layer 3 */
84 seg_descs[i].handle_idx = hidx;
85 seg_descs[i].offset = (off_t) off;
86 seg_descs[i].data = dp; /* points into payload_buf */
87 seg_descs[i].wlen = wlen;
88
89 /* write-recovery (kXR_recoverWrts): if this byte range was already
90 * durably written before the connection dropped, it is a client replay.
91 * Neutralise it to a no-op by zeroing wlen; the worker thread treats
92 * wlen == 0 as "skip" so the bytes are neither rewritten nor
93 * double-counted. */
94 if (ctx->files[hidx].wrts_enabled &&
95 brix_wrts_is_replay(&ctx->files[hidx], off, wlen))
96 {
97 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
98 "brix: writev AIO recovery replay skip"
99 " offset=%lld len=%u", (long long) off, wlen);
100 seg_descs[i].wlen = 0;
101 }
102
103 /* Advance over the (original, un-zeroed) data block so dp stays aligned
104 * with the wire layout regardless of replay skipping. */
105 dp += wlen;
106 }
107
108 task = ngx_thread_task_alloc(c->pool, sizeof(brix_writev_aio_t));
109 if (task == NULL) {
110 return NGX_ERROR;
111 }
112
113 t = task->ctx;
114 t->c = c;
115 t->ctx = ctx;
116 t->n_segs = n_segs;
117 t->segs = seg_descs;
118 t->payload_buf = ctx->recv.payload_buf; /* transfer ownership */
119 t->do_sync = do_sync;
120 t->bytes_total = 0;
121 t->io_error = 0;
122 t->streamid[0] = ctx->recv.cur_streamid[0];
123 t->streamid[1] = ctx->recv.cur_streamid[1];
124
125 brix_task_bind(task, brix_writev_write_aio_thread, brix_writev_write_aio_done);
126
127 (void) brix_aio_post_task(ctx, c, conf->common.thread_pool, task,
128 "brix: thread_task_post failed, falling back to sync writev",
129 posted);
130 /* *posted == 1: ownership of payload_buf now lives on the task and the done
131 * callback will free it and send the reply. Detach it from ctx so the main
132 * thread can start the next request without racing the worker or
133 * double-freeing the buffer. *posted == 0: posting failed; payload_buf is
134 * still ours and the caller falls through to the synchronous path using the
135 * original data_ptr. */
136 if (*posted) {
137 ctx->recv.payload = NULL;
138 ctx->recv.payload_buf = NULL;
139 ctx->recv.payload_buf_size = 0;
140 }
141
142 return NGX_OK;
143}
