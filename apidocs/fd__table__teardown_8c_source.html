27/* fhandle_unlink_staging — remove a handle-owned staging temp (checkpoint journal
28 * or POSC partial) through the VFS seam. `abs_path` is the absolute staging path;
29 * strip `root_canon` to the export-relative key and confined-unlink it (the
30 * teardown context has no request ctx, but the srv conf still gives root_canon +
31 * rootfd). Falls back to a raw unlink only when the path is not under the export
32 * root (no confinement is possible then). */
33static void
34fhandle_unlink_staging(const char *abs_path, const char *root_canon,
35 ngx_log_t *log)
36{
37 size_t root_len = (root_canon != NULL) ? ngx_strlen(root_canon) : 0;
38
39 if (root_len > 0
40 && ngx_strncmp((u_char *) abs_path, (u_char *) root_canon, root_len) == 0
41 && abs_path[root_len] == '/')
42 {
43 (void) brix_vfs_unlink_path(log, root_canon, abs_path + root_len);
44 return;
45 }
46 (void) unlink(abs_path); /* vfs-seam-allow: handle-owned staging temp not under the export root */
47}

49/* fhandle_release_descriptors — close the handle's byte-I/O descriptor(s) and the
50 * CSI engine. WHY: three mutually-exclusive descriptor owners must be torn down in
51 * one place so exactly one close runs: a phase-70 staged-write handle (abort — drop
52 * the temp unless already committed), a driver-backed object handle (its own close
53 * commits metadata and owns file->fd, so we must NOT also raw-close it), or a plain
54 * POSIX fd. HOW: staged first (independent of the fd owner), then the fd owner in
55 * strict driver-vs-raw priority, then the CSI flush+free. Fields are nulled as each
56 * resource is released so a later reset cannot double-free. */
57static void
58fhandle_release_descriptors(brix_file_t *file)
59{
60 /* Whole-object staged write adapter (phase-70): release the write session.
61 * brix_vfs_writer_abort is a no-op once the object was committed (kXR_sync/
62 * close ran); otherwise — a disconnect or an aborted upload — it drops the
63 * staged temp so no partial object is published. */
64 if (file->writer != NULL) {
65 brix_vfs_writer_abort(file->writer);
66 file->writer = NULL;
67 file->staged_committed = 0;
68 }
69
70 if (file->sd_obj.driver != NULL) {
71 /* Layer 3: a driver-backed handle owns its descriptor(s) and commits any
72 * pending metadata (e.g. catalog size/mtime) in its own close(); do NOT
73 * also raw-close file->fd, which is the driver's block-0 descriptor. */
74 (void) file->sd_obj.driver->close(&file->sd_obj);
75 file->sd_obj.driver = NULL;
76 file->sd_obj.inst = NULL;
77 file->sd_obj.state = NULL;
78 file->sd_obj.fd = -1;
79 } else if (file->fd >= 0) {
80 close(file->fd);
81 }
82
83 /* xmeta P3: flush + free the CSI engine (merges folded block CRCs
84 * into the file's metadata record). */
85 if (file->csi != NULL) {
86 brix_csi_close((brix_csi_t *) file->csi);
87 ngx_free(file->csi);
88 file->csi = NULL;
89 }
90}

92/* fhandle_release_staging_files — unlink + free the handle-owned staging temps and
93 * the heap path copy, and drop the TPC registry entry. WHY: checkpoint and POSC temps
94 * are rollback state owned by the handle; if still set at teardown the session ended
95 * without a clean kXR_close, so the partials must not be left on disk (except the
96 * deterministic identity-keyed resume partial, which is preserved for reconnect). HOW:
97 * confined-unlink each temp through the VFS seam, free every heap allocation, then
98 * remove any TPC transfer keyed on this handle. */
99static void
100fhandle_release_staging_files(brix_file_t *file, const char *root_canon,
101 ngx_log_t *tlog)
102{
103 /*
104 * Checkpoint files are temporary rollback state owned by the handle.
105 * Normal close/free removes any abandoned checkpoint path.
106 */
107 if (file->ckp_path != NULL) {
108 fhandle_unlink_staging(file->ckp_path, root_canon, tlog);
109 ngx_free(file->ckp_path);
110 }
111
112 /*
113 * POSC cleanup: if posc_final_path is still set here the session ended
114 * without a clean kXR_close (disconnect, error, or abandon). Unlink the
115 * temp staging file so partial writes are not left on disk.
116 *
117 * EXCEPTION — upload resume (is_resume): the partial is deterministic and
118 * identity-keyed, so it is PRESERVED on a non-clean close. That is the
119 * whole mechanism: the same client reconnecting re-opens it in place (no
120 * truncate) and resumes from its offset. Abandoned partials are reclaimed
121 * by an operator glob-clean ("*.xrdresume.*.part") rather than here.
122 */
123 if (file->posc_final_path != NULL) {
124 if (file->path != NULL && !file->is_resume) {
125 fhandle_unlink_staging(file->path, root_canon, tlog);
126 }
127 ngx_free(file->posc_final_path);
128 }
129
130 if (file->path != NULL) {
131 ngx_free(file->path);
132 }
133
134 if (file->tpc_transfer_id != 0) {
135 (void) brix_tpc_registry_remove(file->tpc_transfer_id, NULL);
136 }
137}

139/* fhandle_reset_slot — zero every field of a freed slot back to its unused state,
140 * releasing the ZIP inflate stream and SSI timers/registration along the way. WHY:
141 * a slot is reused for the next kXR_open, so no stale pointer, capability bit, or
142 * TPC/write-tracking counter may survive — a reused slot must be indistinguishable
143 * from a never-used one. HOW: run the two pool-owned sub-cleanups (zip, ssi) that
144 * must precede their pointer being dropped, then set every scalar/pointer field to
145 * its zero/null/sentinel value; must run AFTER descriptor + staging release so their
146 * niling is not clobbered. */
147static void
148fhandle_reset_slot(brix_file_t *file)
149{
150 /* ZIP member handle: release the deflate inflate stream (if any) and clear
151 * the zip_* state so a reused slot cannot be mistaken for a zip handle. */
152 brix_zip_handle_cleanup(file);
153 file->zip_mode = 0;
154 file->zip_method = 0;
155 file->zip_data_off = 0;
156 file->zip_comp_size = 0;
157 file->zip_uncomp_size = 0;
158 file->zip_crc32 = 0;
159 file->zip_logical_pos = 0;
160 file->zip_comp_pos = 0;
161
162 file->fd = -1;
163 ngx_memzero(&file->sd_obj, sizeof(file->sd_obj)); /* Layer 3: clear driver obj */
164 file->shared_handle_slot_hint = -1; /* Phase 33 C2: drop the cached SHM slot */
165 file->readable = 0;
166 file->writable = 0;
167 file->from_cache = 0;
168 file->bytes_read = 0;
169 file->bytes_written = 0;
170 file->open_time = 0;
171 file->path = NULL;
172 /* §7: cancel SSI async-deferral timers + unregister the session before the
173 * .ssi pointer is dropped (runs on close + disconnect; NULL-safe). */
174 brix_ssi_handle_cleanup(file->ssi);
175 file->ssi = NULL; /* §7: SSI state is connection-pool owned */
176 file->is_regular = 0;
177 file->device = 0;
178 file->inode = 0;
179 file->cached_size = 0;
180 file->read_last_end = -1;
181 file->read_ahead_end = 0;
182 file->ckp_path = NULL;
183 file->ckp_size = 0;
184 file->posc_final_path = NULL;
185 file->is_resume = 0;
186 file->tpc_destination = 0;
187 file->tpc_armed = 0;
188 file->tpc_started = 0;
189 file->tpc_done = 0;
190 file->tpc_key[0] = '\0';
191 file->tpc_org[0] = '\0';
192 file->tpc_src_host[0] = '\0';

211/* brix_free_fhandle — full teardown of one slot: close the fd, unpublish from
212 * shared memory (unbound only), unlink checkpoint/POSC staging files, free the heap
213 * path, and reset every field to its zero/null state. Called on session end, error
214 * recovery, and kXR_close completion. */
215
216void
217brix_free_fhandle(brix_ctx_t *ctx, int handle_index)
218{
219 brix_file_t *file;
220 ngx_stream_brix_srv_conf_t *conf;
221 const char *root_canon;
222 ngx_log_t *tlog;
223
224 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES) {
225 return;
226 }
227
228 file = &ctx->files[handle_index];
229
230 /* The teardown has no request ctx, but the srv conf still yields the export
231 * root_canon so handle-owned staging temps can be confined-unlinked. */
232 conf = ngx_stream_get_module_srv_conf(ctx->session, ngx_stream_brix_module);
233 root_canon = (conf != NULL) ? conf->common.root_canon : NULL;
234 tlog = (ctx->session != NULL && ctx->session->connection != NULL)
235 ? ctx->session->connection->log : NULL;
236
237 if (!ctx->is_bound && file->fd >= 0) {
238 brix_session_handle_unpublish(ctx->login.sessid, handle_index);
239 }
240
241 fhandle_release_descriptors(file);
242 fhandle_release_staging_files(file, root_canon, tlog);
243 brix_wt_mark_clean(ctx, handle_index);
244 fhandle_reset_slot(file);
245}

247/* brix_close_all_files — brix_free_fhandle every slot; used at session
248 * termination or error recovery when all handles must be released at once. */
249
250void
251brix_close_all_files(brix_ctx_t *ctx)
252{
253 int handle_index;
254
255 for (handle_index = 0; handle_index < BRIX_MAX_FILES; handle_index++) {
256 brix_free_fhandle(ctx, handle_index);
257 }
