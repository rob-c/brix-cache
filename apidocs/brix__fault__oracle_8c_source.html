38 pid_t pid = fork();
39 if (pid < 0) {
40 return -1;
41 }
42 if (pid == 0) {
43 setsid(); /* own process group for a clean timeout kill */
44 /* Close every inherited descriptor above stdio so the probe never holds a
45 * copy of the proxy's listen/control/relay sockets. Prefer close_range()
46 * (one syscall for the whole range) over a loop to RLIMIT_NOFILE, which on
47 * this box is ~500k and would cost half a million syscalls per fork. */
48#if defined(SYS_close_range)
49 if (syscall(SYS_close_range, 3, ~0U, 0) != 0)
50#endif
51 {
52 for (int fd = 3; fd < 4096; fd++) {
53 close(fd);
54 }
55 }
56 execl("/bin/sh", "sh", "-c", cmd, (char *) NULL);
57 _exit(127);
58 }
59 int waited = 0, status = 0;
60 for (;;) {
61 pid_t r = waitpid(pid, &status, WNOHANG);
62 if (r == pid) {
63 break;
64 }
65 if (r < 0) {
66 return -1;
67 }
68 if (timeout_ms > 0 && waited >= timeout_ms) {
69 kill(-pid, SIGKILL);
70 waitpid(pid, &status, 0);
71 return -1; /* timeout: inconclusive */
72 }
73 struct timespec ts = { 0, 10 * 1000 * 1000 }; /* 10 ms poll */
74 nanosleep(&ts, NULL);
75 waited += 10;
76 }
77 if (WIFEXITED(status)) {
78 return WEXITSTATUS(status);
