1/*
2 * vfs.h — public API for the unified VFS (POSIX-filesystem data plane).
3 *
4 * WHAT: The only header protocol op handlers include to touch the export root.
5 * Declares the open flags (BRIX_VFS_O_READ/WRITE/CREATE/EXCL/TRUNC/
6 * APPEND/MKDIRPATH/NOCACHE), the opaque handle types (brix_vfs_file_t,
7 * brix_vfs_dir_t), the per-operation request descriptor
8 * brix_vfs_ctx_t, the result/stat structs (brix_vfs_stat_t,
9 * brix_vfs_io_result_t), and every brix_vfs_* entry point —
10 * open/close, read/write, stat, opendir/readdir/closedir, and the
11 * namespace mutators unlink/rmdir/rename/mkdir plus truncate/sync.
12 *
13 * WHY: All four front ends (XRootD root://, WebDAV davs://, the S3 subset, and
14 * CMS data-server I/O) funnel through this one protocol-agnostic surface
15 * so confinement, metrics, access logging, page-CRC, and cache
16 * integration are implemented once and inherited for free. Handlers must
17 * never call open/pread/rename directly — they fill an brix_vfs_ctx_t
18 * and call here.
19 *
20 * HOW: A caller populates brix_vfs_ctx_t with the export root_canon (and the
21 * persistent per-worker rootfd), the already-resolved client path
22 * (brix_path_result_t, produced by ../path/), the caller identity,
23 * allow_write/is_tls/want_pgcrc/cache flags, and the metrics_proto, then
24 * invokes a single entry point. The handle accessors (brix_vfs_file_fd
25 * et al.) are the only way callers reach the underlying fd/size/mtime.
26 */
