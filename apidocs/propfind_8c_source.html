44 * Assemble the request body from r->request_body->bufs into a contiguous

49propfind_assemble_body(ngx_http_request_t *r, size_t *len)
50{
51 u_char *buf;

69/*
70 * WHAT: Parse a PROPFIND request body (XML) into a libxml2 document.
71 * WHY: Centralises the entity-expansion DoS guard so every parse of an
72 * attacker-supplied body uses the identical SAFE libxml2 option set.
73 * HOW: xmlReadMemory with the SAFE flags below; returns NULL on parse
74 * failure (the caller then defaults to allprop). W8/G1 — these flags
75 * are deliberately the SAFE set: XML_PARSE_NONET blocks network entity
76 * fetches and XML_PARSE_NO_XXE (where available) blocks external entity
77 * loading. Critically we do NOT set XML_PARSE_HUGE, so libxml2 keeps
78 * its default caps on entity-expansion depth/amplification (the "billion
79 * laughs" defense). Do not add XML_PARSE_HUGE here without a separate
80 * explicit size/време bound — it removes those caps and re-opens the DoS.
81 */
82static xmlDocPtr
83propfind_read_doc(const char *body, size_t body_len)
84{
85 int opts = XML_PARSE_NONET | XML_PARSE_NOERROR | XML_PARSE_NOWARNING;
86
87#if defined(XML_PARSE_NO_XXE)
88 opts |= XML_PARSE_NO_XXE;
89#endif
90
91 return xmlReadMemory(body, (int) body_len, "propfind.xml", NULL, opts);
92}

95/*
96 * WHAT: Record one unrecognised <prop> child as a pre-built reflected-XML
97 * empty-element for the 404 propstat block.
98 * WHY: The response writer must echo unknown properties verbatim (name +
99 * namespace) but the fixed-size propfind_req_t cannot grow without bound
100 * — a request naming thousands of bogus props must not exhaust memory.
101 * HOW: Silently cap capture at PF_UNKNOWN_MAX; store the raw namespace/local
102 * for bookkeeping, then pre-render the exact escaped XML element now so
103 * the writer only splices the stored string. DAV:-namespace props use
104 * the <D:name/> short form; any other namespace gets an inline xmlns
105 * declaration with the namespace href XML-escaped.
106 */
107static void
108propfind_capture_unknown(ngx_http_request_t *r, propfind_req_t *req,
109 xmlNodePtr prop)
110{
111 ngx_uint_t idx;
112 const char *local = (const char *) prop->name;
113 const char *ns_href = "";
114
115 if (req->unknown_count >= PF_UNKNOWN_MAX) {
116 return;
117 }
118
119 idx = req->unknown_count;
120
121 if (prop->ns != NULL && prop->ns->href != NULL) {
122 ns_href = (const char *) prop->ns->href;
123 }
124
125 snprintf(req->unknown[idx].ns, sizeof(req->unknown[idx].ns), "%s", ns_href);
126 snprintf(req->unknown[idx].local, sizeof(req->unknown[idx].local),
127 "%s", local);
128
129 if (prop->ns != NULL
130 && xmlStrcmp(prop->ns->href, BAD_CAST "DAV:") != 0)
131 {
132 char *safe_ns = webdav_escape_xml_text(r->pool, ns_href);
133 if (safe_ns == NULL) {
134 safe_ns = "";
135 }
136 /* Non-DAV: namespace — include xmlns declaration. */
137 snprintf(req->unknown[idx].xml, PF_UNKNOWN_XML_MAX,
138 "<ns%u:%s xmlns:ns%u=\"%s\"/>",
139 (unsigned) idx, local, (unsigned) idx, safe_ns);
140 } else {
141 snprintf(req->unknown[idx].xml, PF_UNKNOWN_XML_MAX, "<D:%s/>", local);
142 }
143
144 req->unknown_count++;
145}

148/*
149 * WHAT: Collect the named properties of a <prop> element into req.
150 * WHY: A PROPFIND <prop> selects an explicit property set; each child either
151 * maps to a known bit or must be reflected back as an unknown.
152 * HOW: Iterate element children — known names OR their bit into prop_mask,
153 * everything else is handed to propfind_capture_unknown. Non-element
154 * nodes (text/whitespace) are skipped.
155 */
156static void
157propfind_collect_props(ngx_http_request_t *r, propfind_req_t *req,
158 xmlNodePtr child)
159{
160 xmlNodePtr prop;
161
162 for (prop = child->children; prop != NULL; prop = prop->next) {
163 const char *local;
164 unsigned bit;
165
166 if (prop->type != XML_ELEMENT_NODE) {
167 continue;
168 }
169
170 local = (const char *) prop->name;
171 bit = propfind_name_to_bit(local);
172
173 if (bit) {
174 req->prop_mask |= bit;
175 continue;
176 }
177
178 propfind_capture_unknown(r, req, prop);
179 }
180}

183/*
184 * WHAT: Interpret the children of the <propfind> root element, setting the
185 * request type (allprop / propname / prop) and, for <prop>, its set.
186 * WHY: RFC 4918 §9.1 — exactly one of allprop/propname/prop selects the
187 * listing mode; the first recognised child wins.
188 * HOW: Scan element children; stop at the first allprop/propname/prop. For
189 * <prop>, delegate the property set to propfind_collect_props.
190 */
191static void
192propfind_walk_root(ngx_http_request_t *r, propfind_req_t *req, xmlNodePtr root)
193{
194 xmlNodePtr child;
195
196 for (child = root->children; child != NULL; child = child->next) {
197 if (child->type != XML_ELEMENT_NODE) {
198 continue;
199 }
200
201 if (xmlStrcmp(child->name, BAD_CAST "allprop") == 0) {
202 req->type = PROPFIND_ALLPROP;
203 break;
204 }
205
206 if (xmlStrcmp(child->name, BAD_CAST "propname") == 0) {
207 req->type = PROPFIND_PROPNAME;
208 break;
209 }
210
211 if (xmlStrcmp(child->name, BAD_CAST "prop") == 0) {
212 req->type = PROPFIND_PROP;
213 propfind_collect_props(r, req, child);
214 break;
215 }
216 }
217}

220/*
221 * Parse a PROPFIND request body into req.
222 * Initialises req to ALLPROP on entry so any early return is safe.
223 */
224void
225propfind_parse_request(ngx_http_request_t *r, propfind_req_t *req)
226{
227 const char *body;
228 size_t body_len;
229 xmlDocPtr doc;
230 xmlNodePtr root;
231
232 req->type = PROPFIND_ALLPROP;
233 req->prop_mask = 0;
234 req->unknown_count = 0;
235
236 body = propfind_assemble_body(r, &body_len);
237 if (body == NULL) {
238 return; /* no body → allprop per RFC 4918 §9.1 */
239 }
240
241 doc = propfind_read_doc(body, body_len);
242 if (doc == NULL) {
243 return;
244 }
245
246 root = xmlDocGetRootElement(doc);
247 if (root == NULL || xmlStrcmp(root->name, BAD_CAST "propfind") != 0) {
248 xmlFreeDoc(doc);
249 return;
250 }
251
252 propfind_walk_root(r, req, root);
253
254 xmlFreeDoc(doc);
255}

262/*
263 * Parse the Depth request header into an internal code:
264 * 0 -> "0" or absent (target only)
265 * 1 -> "1" (target + immediate children)
266 * -1 -> "infinity" (full recursive walk)
267 * Any unrecognised value defaults to 0, the safe/cheapest behaviour.
268 */
269int
270propfind_parse_depth(ngx_http_request_t *r)
271{
272 ngx_str_t val = brix_http_get_header(r, "Depth");
273 if (val.len == 0) return 0;
274 if (val.len == 1 && val.data[0] == '1') return 1;
275 if (val.len == 8 && ngx_strncasecmp(val.data, (u_char *) "infinity", 8) == 0) return -1;
276 return 0;
277}

284/*
285 * Body-ready callback: nginx invokes this once the (possibly chunked) request
286 * body is fully buffered. It runs propfind_do and finalizes the request with
287 * its return code — this is the async re-entry point after webdav_handle_propfind
288 * returned NGX_DONE to the request pipeline.
289 *
290 * Phase 40: the PROPFIND body is read asynchronously, so the outer dispatch
291 * wrapper (webdav_dispatch.c) has already cleared the impersonation principal by
292 * the time this callback runs. Re-establish it for the duration of the listing
293 * exactly as webdav_handle_put_body does — without it the directory-stat/opendir
294 * and the brix_dirlist_access_ok confidentiality check run as the unprivileged
295 * worker instead of the mapped user, which would both mis-own metadata and leak
296 * entries of a directory the mapped user cannot read. No-op unless map mode.
297 */
298void
299propfind_body_handler(ngx_http_request_t *r)
300{
301 ngx_http_brix_webdav_req_ctx_t *rx =
302 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
303 ngx_int_t rc;
304
305 brix_imp_request_begin(rx != NULL ? rx->identity : NULL);
306 rc = propfind_do(r);
307 brix_imp_request_end();
308
309 ngx_http_finalize_request(r, rc);
310}
