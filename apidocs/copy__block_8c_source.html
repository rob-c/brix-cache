9/*
10 * copy_remote_to_block — root:// → block:// single-file transfer.
11 *
12 * WHAT: connects to the remote XRootD source, opens the block:// destination
13 * directly through the VFS block backend (in-place write, no temp+rename),
14 * and streams all bytes via the existing download pump machinery.
15 * WHY: copy_download routes its destination through brix_url du->path (a stripped
16 * bare POSIX path), which selects the POSIX backend. A block:// URL must be
17 * passed as-is so vfs_url_to_scheme routes it to the block backend instead.
18 * HOW: parse src → resilient_setup → brix_vfs_open(dst_url, WRITE|FORCE) →
19 * download_stream_body → commit (fsync) on success, abort (no-op) on failure.
20 * FORCE is always set: block/device targets pre-exist by design.
21 */
22int
23copy_remote_to_block(const char *src_url, const char *dst_url,
24 const brix_copy_opts *o, const brix_opts *co,
25 brix_status *st)
26{
27 brix_url su;
28 brix_conn c;
29 brix_statinfo si;
30 brix_streamset ss;
31 brix_vfs_file *vf = NULL;
32 brix_vfs_open_opts vopts;
33 pump_local_t lc;
34 int stall;
35 int rc;
36
37 if (brix_url_parse(src_url, &su, st) != 0) {
38 return -1;
39 }
40
41 stall = copy_stall_ms(o, 60000);
42 ss.n = 0;
43 if (resilient_setup(&c, &su, co, &si, stall, st) != 0) {
44 return -1;
45 }
46 if (si.flags & kXR_isDir) {
47 brix_status_set(st, XRDC_EUSAGE, 0,
48 "block copy: remote source is a directory (use -r)");
49 brix_close(&c);
50 return -1;
51 }
52
53 vopts.io_uring = (o != NULL) ? o->io_uring : XRDC_IO_URING_AUTO;
54 vopts.expected_size = si.size;
55 vopts.cred = NULL;
56
57 if (brix_vfs_open(dst_url, XRDC_VFS_WRITE | XRDC_VFS_FORCE,
58 &vopts, &vf, st) != 0) {
59 brix_close(&c);
60 return -1;
61 }
62
63 lc.vf = vf;
64 {
65 download_body_ctx dj = { &c, &su, &si, o, &ss };
66 rc = download_stream_body(&dj, pump_sink_local_vfs, &lc, st);
67 }
68
69 if (rc == 0) {
70 rc = brix_vfs_commit(vf, st);
71 } else {
72 brix_vfs_abort(vf);
73 }
74 brix_vfs_close(vf);
75 brix_streams_close(&ss);
76 brix_close(&c);
77 return rc;
78}

81/*
82 * copy_block_to_remote — block:// → root:// single-file transfer.
83 *
84 * WHAT: opens the block:// source through the VFS block backend (READ) and
85 * uploads all bytes into the remote (root://) destination.
86 * WHY: copy_upload reads su->path as a bare path (POSIX backend). A block://
87 * source URL must be passed to brix_vfs_open so vfs_url_to_scheme routes
88 * it to the block backend; the stripped path alone selects the POSIX backend.
89 * HOW: brix_vfs_open(src_url, READ) → fstat for size →
90 * parse dst URL → upload_stream_body(pump_src_local_vfs) → close.
91 * Checksum verification on block sources is best-effort: cksum_verify
92 * opens the file by POSIX path; for a pure block:// URL it returns
93 * XRDC_CK_UNVERIFIED (a non-fatal warn), keeping the good upload intact.
94 */
95int
96copy_block_to_remote(const char *src_url, const char *dst_url,
97 const brix_copy_opts *o, const brix_opts *co,
98 brix_status *st)
99{
100 brix_url du;
101 brix_url fake_su;
102 brix_vfs_file *vf = NULL;
103 brix_vfs_open_opts vopts;
104 brix_vfs_stat vst;
105 brix_status tmp_st;
106 pump_local_t lc;
107 int64_t total = -1;
108 int rc;
109
110 if (brix_url_parse(dst_url, &du, st) != 0) {
111 return -1;
112 }
113
114 vopts.io_uring = (o != NULL) ? o->io_uring : XRDC_IO_URING_AUTO;
115 vopts.expected_size = -1;
116 vopts.cred = NULL;
117
118 if (brix_vfs_open(src_url, XRDC_VFS_READ, &vopts, &vf, st) != 0) {
119 return -1;
120 }
121
122 brix_status_clear(&tmp_st);
123 if (brix_vfs_fstat(vf, &vst, &tmp_st) == 0) {
124 total = vst.size;
125 }
126
127 /* Synthesise a source descriptor for upload_stream_body diagnostics.
128 * scheme=LOCAL tells it to use fake_su.path for optional cksum_verify;
129 * for a block:// URL that open() cannot resolve, cksum_verify returns
130 * XRDC_CK_UNVERIFIED (non-fatal warn) rather than a hard failure. */
131 memset(&fake_su, 0, sizeof(fake_su));
132 fake_su.scheme = XRDC_SCHEME_LOCAL;
133 snprintf(fake_su.path, sizeof(fake_su.path), "%s", src_url);
134
135 lc.vf = vf;
136 {
137 upload_body_ctx uj = { &fake_su, &du, o, co, total };
138 rc = upload_stream_body(&uj, pump_src_local_vfs, &lc, st);
139 }
140 brix_vfs_close(vf);
141 return rc;
142}

145/*
146 * copy_vfs_to_vfs — VFS-source → VFS-destination transfer (local↔block).
147 *
148 * WHAT: opens both src and dst through brix_vfs_open (which routes block://
149 * and /dev/ to the block backend; bare paths to the POSIX backend) and
150 * pumps bytes via transfer_pump. Covers local→block://, block://→local,
151 * and block://→block:// directions.
152 * WHY: when neither side is a root:// remote the generic copy machinery
153 * (copy_download / copy_upload) is unnecessary; two VFS opens + a pump
154 * are enough.
155 * HOW: open src READ → fstat → open dst WRITE|FORCE → pump → commit dst →
156 * close both. FORCE is always set on the destination: block targets
157 * pre-exist by design; POSIX destinations use atomic temp+rename whose
158 * overwrite semantics are controlled by the FORCE flag.
159 */
160int
161copy_vfs_to_vfs(const char *src_url, const char *dst_url,
162 const brix_copy_opts *o, brix_status *st)
163{
164 brix_vfs_file *src_vf = NULL;
165 brix_vfs_file *dst_vf = NULL;
166 brix_vfs_open_opts vopts;
167 brix_vfs_stat vst;
168 brix_status tmp_st;
169 pump_local_t src_lc, dst_lc;
170 int64_t total = -1;
171 int rc;
172
173 vopts.io_uring = (o != NULL) ? o->io_uring : XRDC_IO_URING_AUTO;
174 vopts.expected_size = -1;
175 vopts.cred = NULL;
176
177 if (brix_vfs_open(src_url, XRDC_VFS_READ, &vopts, &src_vf, st) != 0) {
178 return -1;
179 }
180
181 brix_status_clear(&tmp_st);
182 if (brix_vfs_fstat(src_vf, &vst, &tmp_st) == 0) {
183 total = vst.size;
184 }
185
186 vopts.expected_size = total;
187 if (brix_vfs_open(dst_url, XRDC_VFS_WRITE | XRDC_VFS_FORCE,
188 &vopts, &dst_vf, st) != 0) {
189 brix_vfs_close(src_vf);
190 return -1;
191 }
192
193 src_lc.vf = src_vf;
194 dst_lc.vf = dst_vf;
195 rc = transfer_pump(pump_src_local_vfs, &src_lc,
196 pump_sink_local_vfs, &dst_lc,
197 total, o, total, st);
198
199 if (rc == 0) {
200 rc = brix_vfs_commit(dst_vf, st);
201 } else {
202 brix_vfs_abort(dst_vf);
203 }
204 brix_vfs_close(dst_vf);
205 brix_vfs_close(src_vf);
206 return rc;
207}

210/*
211 * copy_block — dispatch for copies involving at least one block:// endpoint.
212 *
213 * WHAT: classifies the (src, dst) pair and routes to the right helper:
214 * root://→block:// → copy_remote_to_block
215 * block://→root:// → copy_block_to_remote
216 * local→block:// → copy_vfs_to_vfs (POSIX src + block dst)
217 * block://→local → copy_vfs_to_vfs (block src + POSIX dst)
218 * block://→block:// → copy_vfs_to_vfs (block src + block dst)
219 * WHY: brix_copy() intercepts block:// before brix_url_parse (which does not
220 * know the block:// scheme) and delegates here.
221 * HOW: classify src/dst by brix_is_block_url and brix_is_web_url; for the
222 * root:// directions use brix_url_parse to distinguish remote/local;
223 * recursion and zip are explicitly rejected (not supported for block).
224 */
225int
226copy_block(const char *src, const char *dst, const brix_copy_opts *o,
227 const brix_opts *co, brix_status *st)
228{
229 int src_block = brix_is_block_url(src);
230 int dst_block = brix_is_block_url(dst);
231 int src_remote = 0;
232 int dst_remote = 0;
233
234 if (o != NULL && o->recursive) {
235 brix_status_set(st, XRDC_EUSAGE, 0,
236 "recursive copy not supported for block:// endpoints");
237 return -1;
238 }
239
240 /* Classify non-block sides: parse with brix_url_parse to detect root://. */
241 if (!src_block) {
242 brix_url su;
243 brix_status tmp;
244 brix_status_clear(&tmp);
245 if (brix_url_parse(src, &su, &tmp) == 0) {
246 src_remote = (su.scheme == XRDC_SCHEME_ROOT
247 || su.scheme == XRDC_SCHEME_ROOTS);
248 }
249 }
250 if (!dst_block) {
251 brix_url du;
252 brix_status tmp;
253 brix_status_clear(&tmp);
254 if (brix_url_parse(dst, &du, &tmp) == 0) {
255 dst_remote = (du.scheme == XRDC_SCHEME_ROOT
256 || du.scheme == XRDC_SCHEME_ROOTS);
257 }
258 }
259
260 if (src_remote && dst_block) {
261 return copy_remote_to_block(src, dst, o, co, st);
262 }
263 if (src_block && dst_remote) {
264 return copy_block_to_remote(src, dst, o, co, st);
265 }
266 if (!src_remote && !dst_remote) {
267 /* both sides are local/block: pure VFS-to-VFS */
268 return copy_vfs_to_vfs(src, dst, o, st);
269 }
270
271 brix_status_set(st, XRDC_EUSAGE, 0,
272 "unsupported block:// copy direction "
273 "(src=%s dst=%s)", src, dst);
274 return -1;
275}
