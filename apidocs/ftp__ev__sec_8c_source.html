8/*
9 * ftp_ev_sec.c — RFC 2228 GSI security handshake on the control channel:
10 * AUTH GSSAPI, the ADAT token exchange, and the MIC/CONF/ENC protected-command
11 * unwrap.
12 *
13 * WHAT: stand up the mem-BIO GSSAPI engine on AUTH, feed each ADAT token through
14 * it until the handshake completes, then bind the verified proxy DN into a
15 * brix_identity_t and switch the control channel to GSS-wrapped replies.
16 *
17 * WHY: the security handshake is the one part of the protocol that is *already*
18 * event-compatible — brix_gssapi_srv_step() is token-driven, not socket-driven:
19 * each ADAT command line carries one handshake token and the reply carries the
20 * next, so no inline TLS I/O is needed on the control channel. That lets the
21 * event engine reach GSI parity here with no state-machine surgery; only the
22 * PROT P *data* channel needs event-driven TLS (P82.3).
23 *
24 * HOW: ported from the sync engine's ftp_cmd_auth/adat/protected + ftp_gss_finalize,
25 * writing through the event reply queue. The pre-auth handshake replies
26 * (334/335/235) go out cleartext because ->sec_active is still 0; finalize flips
27 * it so every subsequent reply is wrapped.
28 */
29

31/* After AUTH GSSAPI completes, bind the verified proxy DN into a brix_identity_t
32 * (so the VFS ctx and audit see a real GSI principal), capture any delegated
33 * credential, and activate the wrapped control channel. */
34static void
35ev_gss_finalize(ftp_ev_t *fc)
36{
37 ngx_str_t dn = { 0, NULL }, proxy = { 0, NULL };
38 brix_identity_t *id;
39
40 if (brix_gssapi_srv_peer(fc->gss, &dn, &proxy) == NGX_OK && dn.len > 0) {
41 id = brix_identity_alloc(fc->c->pool);
42 if (id != NULL) {
43 char dbuf[512];
44 size_t n = ngx_min(dn.len, sizeof(dbuf) - 1);
45 ngx_memcpy(dbuf, dn.data, n);
46 dbuf[n] = '\0';
47 if (brix_identity_set_dn(id, fc->c->pool, dbuf, BRIX_AUTHN_GSI)
48 == NGX_OK)
49 {
50 fc->identity = id;
51 }
52 }
53 fc->deleg_proxy = proxy;
54 fc->ctrl_dn.data = ngx_pnalloc(fc->c->pool, dn.len);
55 if (fc->ctrl_dn.data != NULL) {
56 ngx_memcpy(fc->ctrl_dn.data, dn.data, dn.len);
57 fc->ctrl_dn.len = dn.len;
58 }
59 /* the client's control leaf is the delegated proxy's direct issuer;
60 * needed to present a complete chain on a PROT P data channel. */
61 (void) brix_gssapi_srv_peer_cert_pem(fc->gss, &fc->ctrl_leaf_pem);
62 fc->authed = 1;
63 ngx_log_error(NGX_LOG_INFO, fc->c->log, 0,
64 "brix: GridFTP(ev) gsiftp authenticated dn=\"%V\" deleg=%uz",
65 &dn, proxy.len);
66 }
67 fc->wrap_code = "633"; /* default: private (ENC) replies */
68 fc->sec_active = 1;
69}

72/* AUTH GSSAPI: stand up the mem-BIO engine and invite ADAT tokens (334). */
73ngx_int_t
74brix_ftp_ev_cmd_auth(ftp_ev_t *fc, const char *arg)
75{
76 if (!fc->conf->gsi || fc->conf->tls_ctx == NULL
77 || fc->conf->tls_ctx->ctx == NULL)
78 {
79 return brix_ftp_ev_reply(fc, "534 Security mechanism not available\r\n");
80 }
81 if (strcasecmp(arg, "GSSAPI") != 0) {
82 return brix_ftp_ev_reply(fc, "504 Unknown security mechanism %s\r\n", arg);
83 }
84 if (fc->gss != NULL) {
85 return brix_ftp_ev_reply(fc,
86 "534 Security handshake already in progress\r\n");
87 }
88 fc->gss = brix_gssapi_srv_create(fc->c->pool, fc->c->log,
89 fc->conf->tls_ctx->ctx, fc->conf->ca_store, 1 /* accept delegation */);
90 if (fc->gss == NULL) {
91 return brix_ftp_ev_reply(fc, "431 GSSAPI context init failed\r\n");
92 }
93 return brix_ftp_ev_reply(fc,
94 "334 Using authentication type GSSAPI; ADAT must follow\r\n");
95}

98/* ADAT: feed one decoded token to the engine and reply 335 (continue) or 235
99 * (complete, then switch on the wrapped control channel). */
100ngx_int_t
101brix_ftp_ev_cmd_adat(ftp_ev_t *fc, const char *arg)
102{
103 ngx_str_t tok, out;
104 brix_gss_status_e st;
105
106 if (fc->gss == NULL) {
107 return brix_ftp_ev_reply(fc, "503 Send AUTH GSSAPI first\r\n");
108 }
109 if (brix_ftp_ev_b64_decode(fc->c->pool, arg, &tok) != NGX_OK) {
110 return brix_ftp_ev_reply(fc, "501 Malformed ADAT token\r\n");
111 }
112
113 st = brix_gssapi_srv_step(fc->gss, tok.data, tok.len, &out);
114 if (st == BRIX_GSS_FAILED) {
115 return brix_ftp_ev_reply(fc, "535 GSSAPI authentication failed\r\n");
116 }
117 if (st == BRIX_GSS_CONTINUE) {
118 return brix_ftp_ev_send_adat(fc, 335, &out);
119 }
120
121 /* COMPLETE: emit the final 235 cleartext, THEN activate wrapping. A final
122 * token is framed "235 ADAT=<b64>"; with none, send a bare completion line
123 * (an empty "ADAT=" would decode to a zero-length token and fail the client). */
124 if (out.len > 0) {
125 if (brix_ftp_ev_send_adat(fc, 235, &out) != NGX_OK) {
126 return NGX_ERROR;
127 }
128 } else if (brix_ftp_ev_reply(fc, "235 GSSAPI authentication succeeded\r\n")
129 != NGX_OK)
130 {
131 return NGX_ERROR;
132 }
133 ev_gss_finalize(fc);
134 return NGX_OK;
135}

138/* MIC/CONF/ENC: unwrap a protected command and re-dispatch its plaintext. The
139 * safety code chosen here also frames the reply(ies) that command emits. */
140ngx_int_t
141brix_ftp_ev_cmd_protected(ftp_ev_t *fc, const char *code, const char *arg)
142{
143 ngx_str_t tok, inner;
144 char *innerline;
145 size_t n;
146
147 if (fc->gss == NULL || !fc->sec_active) {
148 return brix_ftp_ev_reply(fc, "503 Not authenticated\r\n");
149 }
150 fc->wrap_code = code;
151 if (brix_ftp_ev_b64_decode(fc->c->pool, arg, &tok) != NGX_OK) {
152 return brix_ftp_ev_reply(fc, "501 Malformed protected token\r\n");
153 }
154 if (brix_gssapi_unwrap(fc->gss, tok.data, tok.len, &inner) != NGX_OK
155 || inner.len == 0)
156 {
157 return brix_ftp_ev_reply(fc, "535 Cannot decode protected command\r\n");
158 }
159 innerline = ngx_pnalloc(fc->c->pool, BRIX_FTP_EV_CMD_MAX);
160 if (innerline == NULL) {
161 return NGX_ERROR;
162 }
163 n = ngx_min(inner.len, (size_t) BRIX_FTP_EV_CMD_MAX - 1);
164 ngx_memcpy(innerline, inner.data, n);
165 while (n > 0 && (innerline[n - 1] == '\r' || innerline[n - 1] == '\n')) {
166 n--;
167 }
168 innerline[n] = '\0';
169 return brix_ftp_ev_dispatch(fc, innerline);
170}
