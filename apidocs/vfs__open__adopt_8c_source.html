1/*
2 * vfs_open_adopt.c — VFS ctx priming, handle adoption, and shared path helpers.
3 *
4 * WHAT: Hosts the handle-construction half of the VFS open unit:
5 * brix_vfs_ctx_init() (prime a per-request ctx), brix_vfs_fill_stat()
6 * (struct stat -> brix_vfs_stat_t), brix_vfs_copy_path() (pool-dup a C
7 * string), brix_vfs_adopt_fd()/brix_vfs_adopt_obj() (wrap an already-open
8 * fd / a backend object in a brix_vfs_file_t), and
9 * brix_vfs_export_relative()/_root() (export-root-relative path form).
10 *
11 * WHY: These are the pieces the open orchestrator (vfs_open.c) and the cache
12 * layer both build handles with. Splitting them out of vfs_open.c keeps
13 * both files under the size cap while leaving the open cascade in one place.
14 * Every symbol here is already public (declared in vfs_internal.h / vfs.h)
15 * because external units (the cache layer, vfs_staged.c) also call them —
16 * so nothing new crosses a boundary; this is purely a file-size split.
17 *
18 * HOW: brix_vfs_adopt_fd()/_obj() capture open-time metadata through the
19 * backend's fstat slot (never a direct fstat(2)) into a freshly pcalloc'd
20 * handle, dup the path, and record from_cache/is_tls/stat_current.
21 * brix_vfs_export_relative_root() strips an absolute export root from a
22 * confined path so an inst-keyed storage driver keys its namespace on the
23 * logical path. See vfs_open.c for the open cascade that consumes these.
24 */
25#include "vfs_internal.h"
26#include "vfs_backend_registry.h"

28/* Populate a per-request brix_vfs_ctx_t with the fields the HTTP front ends
29 * (WebDAV, S3) set identically: a transient (rootfd = -1) confined open of an
30 * already-resolved canonical path. Zeroes the ctx first, then fills pool/log,
31 * the metrics proto, the export + cache roots (deriving cache_enabled), the
32 * write gate, the TLS flag, the identity, and the resolved path (is_confined).
33 * Callers may still adjust individual fields afterwards (e.g. cache_writethrough
34 * config). Kept HTTP-agnostic on purpose so the header stays stream-includable —
35 * callers pass pool/log/is_tls extracted from their own request object. */
36void
37brix_vfs_ctx_init(brix_vfs_ctx_t *vctx, ngx_pool_t *pool, ngx_log_t *log,
38 brix_proto_t proto, const char *root_canon, const char *cache_root_canon,
39 int allow_write, int is_tls, brix_identity_t *identity,
40 const char *resolved_path)
41{
42 if (vctx == NULL) {
43 return;
44 }
45
46 ngx_memzero(vctx, sizeof(*vctx));
47 vctx->rootfd = -1;
48 vctx->pool = pool;
49 vctx->log = log;
50 vctx->metrics_proto = proto;
51 vctx->root_canon = root_canon;
52 /* Resolve the export's selected storage backend (NULL ⇒ default POSIX) so
53 * every VFS op on this ctx routes through the driver without each handler
54 * threading the instance. Per-worker, lazily created on first use. */
55 vctx->sd = brix_vfs_backend_resolve(root_canon, log);
56 vctx->cache_root_canon = cache_root_canon;
57 vctx->cache_enabled =
58 (cache_root_canon != NULL && cache_root_canon[0] != '\0') ? 1 : 0;
59 vctx->allow_write = allow_write ? 1 : 0;
60 vctx->is_tls = is_tls ? 1 : 0;
61 vctx->identity = identity;
62 if (resolved_path != NULL) {
63 vctx->resolved.resolved.data = (u_char *) resolved_path;
64 vctx->resolved.resolved.len = ngx_strlen(resolved_path);
65 vctx->resolved.is_confined = 1;
66 }
67}

69/* Copy a struct stat into the protocol-neutral brix_vfs_stat_t (zeroes the
70 * output first; sets is_directory/is_regular from the mode). No-op on NULL. */
71void
72brix_vfs_fill_stat(const struct stat *st, brix_vfs_stat_t *out)
73{
74 if (st == NULL || out == NULL) {
75 return;
76 }
77
78 ngx_memzero(out, sizeof(*out));
79 out->size = st->st_size;
80 out->mtime = st->st_mtime;
81 out->ctime = st->st_ctime;
82 out->atime = st->st_atime;
83 out->mode = (ngx_uint_t) st->st_mode;
84 out->ino = st->st_ino;
85 out->dev = st->st_dev;
86 out->uid = st->st_uid;
87 out->gid = st->st_gid;
88 out->blocks = st->st_blocks;
89 out->is_directory = S_ISDIR(st->st_mode) ? 1 : 0;
90 out->is_regular = S_ISREG(st->st_mode) ? 1 : 0;
91}

93char *
94brix_vfs_copy_path(ngx_pool_t *pool, const char *path)
95{
96 size_t len;
97 char *copy;
98
99 if (pool == NULL || path == NULL) {
100 errno = EINVAL;
101 return NULL;
102 }
103
104 len = strlen(path);
105 copy = ngx_pnalloc(pool, len + 1);
106 if (copy == NULL) {
107 errno = ENOMEM;
108 return NULL;
109 }
110
111 ngx_memcpy(copy, path, len);
112 copy[len] = '\0';
113 return copy;
114}

116/* Wrap an already-open fd in a freshly pcalloc'd brix_vfs_file_t: fstat the
117 * fd into the cached metadata, dup the path, and record attrs.from_cache/is_tls.
118 * attrs.writable gates the stat_current fast path (set only for read-only
119 * handles). Used by both brix_vfs_open() and the cache layer's open path. */
120ngx_int_t
121brix_vfs_adopt_fd(brix_vfs_ctx_t *ctx, const char *path, ngx_fd_t fd,
122 brix_vfs_adopt_attrs_t attrs, brix_vfs_file_t **out)
123{
124 brix_sd_stat_t st;
125 brix_sd_obj_t obj;
126 brix_vfs_file_t *fh;
127
128 if (out == NULL) {
129 errno = EINVAL;
130 return NGX_ERROR;
131 }
132 *out = NULL;
133
134 if (ctx == NULL || path == NULL || fd == NGX_INVALID_FILE) {
135 errno = EINVAL;
136 return NGX_ERROR;
137 }
138
139 /* Capture the open-time metadata through the backend's fstat slot rather
140 * than a direct fstat(2): the VFS records the cached size/mtime/etc. but
141 * leaves the syscall to the storage driver. */
142 brix_vfs_ctx_sd_obj(ctx, fd, &obj);
143 if (obj.driver->fstat == NULL || obj.driver->fstat(&obj, &st) != NGX_OK) {
144 return NGX_ERROR;
145 }
146
147 fh = ngx_pcalloc(ctx->pool, sizeof(*fh));
148 if (fh == NULL) {
149 errno = ENOMEM;
150 return NGX_ERROR;
151 }
152
153 fh->path = brix_vfs_copy_path(ctx->pool, path);
154 if (fh->path == NULL) {
155 return NGX_ERROR;
156 }
157
158 fh->obj = obj; /* backend object built above (driver + inst + fd) */
159 fh->pool = ctx->pool;
160 fh->log = ctx->log;
161 fh->ctx = ctx;
162 fh->size = st.size;
163 fh->mtime = st.mtime;
164 fh->ctime = st.ctime;
165 fh->ino = st.ino;
166 fh->mode = st.mode;
167 fh->from_cache = attrs.from_cache ? 1 : 0;
168 fh->is_tls = ctx->is_tls;
169 /* Trust the open-time metadata for stat() only on a read-only handle: the
170 * file cannot change through it, so the fstat above stays authoritative for
171 * the handle's lifetime. A writable handle leaves this 0 so brix_vfs_file_stat()
172 * always issues a live fstat (its bytes/mtime/size move as it is written). */
173 fh->stat_current = attrs.writable ? 0 : 1;
174 fh->memfd = NGX_INVALID_FILE;
175
176 *out = fh;
177 return NGX_OK;
178}

180/* brix_vfs_adopt_obj — build a handle from an object the backend's open slot
181 * already produced, PRESERVING its per-open state (e.g. an object backend's
182 * block map / metadata), rather than rebuilding from a bare fd (which would drop
183 * that state). Captures open-time metadata via the object's own fstat slot, then
184 * moves the object into the handle by value. A heap-allocated shell (driver set
185 * obj->heap_shell, e.g. pblock's malloc'd object) is freed once copied; the
186 * per-open `state` lives on and is released by driver->close at handle close.
187 * Used for the non-POSIX storage drivers; POSIX keeps the adopt_fd path. Also
188 * the cache hit-serve path (src/cache/open.c) adopts a cache-storage driver
189 * object — hence public (declared in vfs.h). */

194/* brix_vfs_export_relative — the export-root-relative ("logical") form of a
195 * confined path, which is what an inst-keyed storage-driver op expects (per the
196 * SD seam contract). Some callers (WebDAV/S3) resolve to an absolute path under
197 * root_canon; strip that prefix so a non-POSIX backend keys its namespace on the
198 * logical path. Returns `path` unchanged when it is not under root_canon.
199 * Declared in vfs_internal.h; shared with vfs_staged.c. */
200/* Path-based form: strip `root_canon` (an absolute export root) from `path`. */
201const char *
202brix_vfs_export_relative_root(const char *path, const char *root_canon)
203{
204 size_t rlen;
205
206 if (root_canon == NULL || path == NULL) {
207 return path;
208 }
209 rlen = ngx_strlen(root_canon);
210 if (rlen == 0 || ngx_strncmp(path, root_canon, rlen) != 0) {
211 return path;
212 }
213 if (path[rlen] == '/') {
214 return path + rlen; /* "/sub/file" */
215 }
216 if (path[rlen] == '\0') {
217 return "/"; /* the export root itself */
218 }
219 return path; /* a sibling that merely shares the prefix */
220}

223brix_vfs_export_relative(const brix_vfs_ctx_t *ctx, const char *path)
224{
225 return brix_vfs_export_relative_root(path, ctx->root_canon);

228ngx_int_t
229brix_vfs_adopt_obj(brix_vfs_ctx_t *ctx, const char *path,
230 brix_sd_obj_t *o, unsigned writable, brix_vfs_file_t **out)
231{
232 brix_sd_stat_t st;
233 brix_vfs_file_t *fh;
234
235 *out = NULL;
236 if (ctx == NULL || path == NULL || o == NULL) {
237 errno = EINVAL;
238 return NGX_ERROR;
239 }
240 if (o->driver->fstat == NULL || o->driver->fstat(o, &st) != NGX_OK) {
241 return NGX_ERROR;
242 }
243
244 fh = ngx_pcalloc(ctx->pool, sizeof(*fh));
245 if (fh == NULL) {
246 errno = ENOMEM;
247 return NGX_ERROR;
248 }
249 fh->path = brix_vfs_copy_path(ctx->pool, path);
250 if (fh->path == NULL) {
251 return NGX_ERROR;
252 }
253
254 fh->obj = *o; /* driver + instance + fd + per-open state */
255 if (o->heap_shell) {
256 free(o); /* shell copied into fh->obj; release it */
257 }
258 fh->obj.heap_shell = 0; /* fh->obj is embedded, not a heap shell */
259
260 fh->pool = ctx->pool;
261 fh->log = ctx->log;
262 fh->ctx = ctx;
263 fh->size = st.size;
264 fh->mtime = st.mtime;
265 fh->ctime = st.ctime;
266 fh->ino = st.ino;
267 fh->mode = st.mode;
268 fh->from_cache = 0;
269 fh->is_tls = ctx->is_tls;
270 fh->stat_current = writable ? 0 : 1;
271 fh->memfd = NGX_INVALID_FILE;
272
273 *out = fh;
274 return NGX_OK;
275}
