16/* brix_wt_default_decide — the policy decision at kXR_open write intent (from

22 void *user_data)
23{
24 brix_wt_decision_cfg_t *cfg;
25 brix_cache_admit_cfg_t a;
26 off_t size = 0;
27 int is_new;
28
29 if (user_data == NULL || path == NULL) {
30 return BRIX_WT_DECISION_DENY; /* fail-closed */
31 }
32 cfg = (brix_wt_decision_cfg_t *) user_data;
33
34 /* Resolve the size for the cap: a kXR_new create has no size yet; an existing
35 * regular file contributes its st_size; anything else skips the cap. stat()
36 * is a local syscall, not a network round-trip. */
37 is_new = (options & kXR_new) != 0;
38 if (!is_new && cfg->max_write_through_bytes > 0) {
39 struct stat st;
40 if (stat(path, &st) == 0 && S_ISREG(st.st_mode)) {
41 size = st.st_size;
42 } else {
43 is_new = 1; /* not a regular existing file */
44 }
45 }
46
47 a.deny_prefixes = cfg->deny_prefixes;
48 a.allow_prefixes = cfg->allow_prefixes;
49 a.size_limit = cfg->max_write_through_bytes;
50 a.include_regex = cfg->include_regex_set ? &cfg->include_regex : NULL;
51
52 if (brix_cache_admit(&a, path, size, is_new) == BRIX_CACHE_DECLINE) {
53 return BRIX_WT_DECISION_DENY;
54 }
55 return BRIX_WT_DECISION_ALLOW_ASYNC;
56}
