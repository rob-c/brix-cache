1/*
2 * mpxstats.c — aggregate + pretty-print an XRootD server's summary statistics.
3 *
4 * WHAT: `mpxstats-brix [host[:port]] [--metrics-port N]` pulls the server's Prometheus
5 * /metrics and prints a compact per-metric summary (series count + summed
6 * value), collapsing the label dimensions. With no host (or "-") it reads a
7 * metrics blob from stdin instead.
8 * WHY: The stock mpxstats relays the xrootd summary-stats stream; this is the
9 * parse-only, libXrdCl-free equivalent over the observability plane this
10 * project already exposes — handy for a one-screen server health glance.
11 * HOW: Reuse brix_http_get (the same cleartext pull `xrddiag status` uses) or
12 * read stdin; parse "name{labels} value" lines (skip # comments), fold by
13 * base metric name into a small table, print it sorted-as-seen.
14 *
15 * Clean-room: parse-only; no protocol core, no XrdCl.
16 */

37/* Fold one "name{labels} value" sample into the table by base metric name. */
38static void
39ingest_line(const char *line)
40{
41 char name[160];
42 const char *p, *val;
43 size_t nl;
44 int i;
45 double v;
46
47 if (line[0] == '#' || line[0] == '\0') {
48 return;
49 }
50 /* base name = up to '{' or the first space */
51 p = line;
52 while (*p != '\0' && *p != '{' && !isspace((unsigned char) *p)) {
53 p++;
54 }
55 nl = (size_t) (p - line);
56 if (nl == 0 || nl >= sizeof(name)) {
57 return;
58 }
59 memcpy(name, line, nl);
60 name[nl] = '\0';
61
62 /* value = the last whitespace-separated token on the line */
63 val = strrchr(line, ' ');
64 if (val == NULL) {
65 return;
66 }
67 v = strtod(val + 1, NULL);
68
69 for (i = 0; i < g_n; i++) {
70 if (strcmp(g_tab[i].name, name) == 0) {
71 g_tab[i].sum += v;
72 g_tab[i].series++;
73 return;
74 }
75 }
76 if (g_n < MPX_MAX_METRICS) {
77 snprintf(g_tab[g_n].name, sizeof(g_tab[g_n].name), "%s", name);
78 g_tab[g_n].sum = v;
79 g_tab[g_n].series = 1;
80 g_n++;
81 }
82}

84static void
85ingest_buffer(char *buf)
86{
87 char *line, *save;
88 for (line = strtok_r(buf, "\n", &save); line != NULL;
89 line = strtok_r(NULL, "\n", &save)) {
90 ingest_line(line);
91 }
92}

94static void
95report(const char *source)
96{
97 int i;
98 long total = 0;
99 printf("mpxstats-brix: %s\n", source);
100 printf("%-48s %8s %14s\n", "metric", "series", "sum");
101 for (i = 0; i < g_n; i++) {
102 printf("%-48s %8ld %14.0f\n", g_tab[i].name, g_tab[i].series, g_tab[i].sum);
103 total += g_tab[i].series;
104 }
105 printf("(%d metric name(s), %ld series)\n", g_n, total);
106}

108/*
109 * usage_fp — print mpxstats usage to the given stream.
110 * WHY: --help goes to stdout (WS-2); -h keeps the legacy stderr path (C1).
111 * Both now include the footer. Returns rc so callers can write
112 * `return usage_fp(stderr, prog, 0)`.

114static int
115usage_fp(FILE *out, const char *prog, int rc)
116{
117 prog = brix_prog_base(prog); /* display the invoked name, not a path */
118 fprintf(out,
119 "usage: %s [host | -] [--metrics-port N]\n"
120 " no host (or '-') reads a /metrics blob from stdin\n",
121 prog);
122 brix_usage_footer(out, prog);
123 return rc;
124}

127int
128brix_mpxstats_main(int argc, char **argv)
129{
130 const char *host = NULL;
131 int metrics_port = 9100;
132 int i;
133
134 /* --help / --version before main loop. */
135 if (argc >= 2) {
136 if (strcmp(argv[1], "--version") == 0) {
137 printf("%s (BriX-Cache client) %s\n", brix_prog_base(argv[0]),
138 brix_client_version());
139 return 0;
140 }
141 if (strcmp(argv[1], "--help") == 0) {
142 return usage_fp(stdout, argv[0], 0);
143 }
144 }
145
146 for (i = 1; i < argc; i++) {
147 const char *a = argv[i];
148 if (strcmp(a, "--metrics-port") == 0 && i + 1 < argc) {
149 metrics_port = atoi(argv[++i]);
150 } else if (strcmp(a, "-h") == 0) {
151 /* -h keeps the legacy stderr path (C1); footer now included. */
152 return usage_fp(stderr, argv[0], 0);
153 } else if (strcmp(a, "--help") == 0) {
154 /* Recognise --help at any position (not just argv[1]). */
155 return usage_fp(stdout, argv[0], 0);
156 } else if (host == NULL) {
157 host = a;
158 }
159 }
160
161 if (host == NULL || strcmp(host, "-") == 0) {
162 /* stdin */
163 char *buf;
164 size_t cap = 1u << 20, len = 0, r;
165 buf = (char *) malloc(cap);
166 if (buf == NULL) {
167 fprintf(stderr, "mpxstats-brix: out of memory\n");
168 return 51;
169 }
170 while ((r = fread(buf + len, 1, cap - 1 - len, stdin)) > 0) {
171 len += r;
172 if (len >= cap - 1) {
173 break;
174 }
175 }
176 buf[len] = '\0';
177 ingest_buffer(buf);
178 free(buf);
179 report("(stdin)");
180 return 0;
181 }
182
183 {
184 /* strip a host:port → host + port override */
185 char h[256];
186 char *colon;
187 char *body;
188 brix_status st;
189 int http = 0;
190 snprintf(h, sizeof(h), "%s", host);
191 colon = strrchr(h, ':');
192 if (colon != NULL && strchr(h, ':') == colon) { /* host:port (not v6) */
193 *colon = '\0';

195 }
196 body = (char *) malloc(1u << 20);
197 if (body == NULL) {
198 fprintf(stderr, "mpxstats-brix: out of memory\n");
199 return 51;
200 }
201 brix_status_clear(&st);
202 if (brix_http_get(h, metrics_port, "/metrics", 5000, &http, body,
203 1u << 20, NULL, &st) != 0) {
204 fprintf(stderr, "mpxstats-brix: GET %s:%d/metrics: %s\n", h, metrics_port,
205 st.msg);
206 free(body);
207 return 51;
208 }
209 ingest_buffer(body);
210 free(body);
211 {
212 char src[300];
213 snprintf(src, sizeof(src), "%s:%d (HTTP %d)", h, metrics_port, http);

215 }
