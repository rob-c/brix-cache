39int
40brix_xmeta_path_load(const char *path, brix_xmeta_t *xm)
41{
42 char sc[PATH_MAX];
43 uint8_t *buf;
44 ssize_t n;
45 int fd, drc;
46
47 if (path == NULL || xm == NULL) {
48 errno = EINVAL;
49 return BRIX_XMETA_ERR;
50 }
51
52 buf = malloc(BRIX_XMETA_PATH_XATTR_MAX);
53 if (buf == NULL) {
54 errno = ENOMEM;
55 return BRIX_XMETA_ERR;
56 }
57 n = getxattr(path, BRIX_XMETA_PATH_XATTR, buf,
58 BRIX_XMETA_PATH_XATTR_MAX);
59 if (n > 0) {
60 drc = brix_xmeta_decode(buf, (size_t) n, xm);
61 free(buf);
62 return drc;
63 }
64 free(buf);
65
66 if (brix_xmeta_sidecar_path(sc, sizeof(sc), path) != 0) {
67 errno = ENAMETOOLONG;
68 return BRIX_XMETA_ERR;
69 }
70 fd = open(sc, O_RDONLY | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW);
71 if (fd < 0) {
72 return (errno == ENOENT) ? BRIX_XMETA_FOREIGN : BRIX_XMETA_ERR;
73 }
74 {
75 struct stat stb;
76 uint8_t *fbuf;
77 size_t flen, got = 0;
78
79 if (fstat(fd, &stb) != 0 || stb.st_size <= 0
80 || (uint64_t) stb.st_size > BRIX_XMETA_MAX_SECTION)
81 {
82 close(fd);
83 return BRIX_XMETA_FOREIGN;
84 }
85 flen = (size_t) stb.st_size;
86 fbuf = malloc(flen);
87 if (fbuf == NULL) {
88 close(fd);
89 errno = ENOMEM;
90 return BRIX_XMETA_ERR;
91 }
92 while (got < flen) {
93 ssize_t r = pread(fd, fbuf + got, flen - got, (off_t) got);
94
95 if (r < 0 && errno == EINTR) {
96 continue;
97 }
98 if (r <= 0) {
99 break;
100 }
101 got += (size_t) r;
102 }
103 close(fd);
104 if (got != flen) {
105 free(fbuf);
106 return BRIX_XMETA_FOREIGN;
107 }
108 drc = brix_xmeta_decode(fbuf, flen, xm);
109 free(fbuf);
110 }
111 return drc;
112}

114int
115brix_xmeta_path_save(const char *path, const brix_xmeta_t *xm)
116{
117 char sc[PATH_MAX];
118 uint8_t *buf = NULL;
119 size_t len = 0, put = 0;
120 int fd;
121
122 if (path == NULL || xm == NULL) {
123 errno = EINVAL;
124 return BRIX_XMETA_ERR;
125 }
126 if (brix_xmeta_encode(xm, &buf, &len) != BRIX_XMETA_OK) {
127 return BRIX_XMETA_ERR;
128 }
129 if (brix_xmeta_sidecar_path(sc, sizeof(sc), path) != 0) {
130 free(buf);
131 errno = ENAMETOOLONG;
132 return BRIX_XMETA_ERR;
133 }
134
135 if (len <= BRIX_XMETA_PATH_XATTR_MAX
136 && setxattr(path, BRIX_XMETA_PATH_XATTR, buf, len, 0) == 0)
137 {
138 free(buf);
139 (void) unlink(sc); /* one carrier: drop a stale sidecar */
140 return BRIX_XMETA_OK;
141 }
142 if (len <= BRIX_XMETA_PATH_XATTR_MAX && !xmeta_path_xattr_unfit(errno)) {
143 free(buf);
144 return BRIX_XMETA_ERR;
145 }
146
147 /* SECURITY: the metadata sidecar (cache .cinfo residency bitmap / CSI record)
148 * is created and read AS THE WORKER (server-managed sidecar, see meta.c
149 * vfs-seam-allow; CSI verify is a backend op). 0600 (not 0644) so it cannot
150 * leak cache residency / integrity metadata to a mapped low-priv uid on a
151 * shared filesystem. Safe: with impersonation=map the master is root and
152 * reads any-owner sidecar; without impersonation the worker owns and reads it. */
153 fd = open(sc, O_WRONLY | O_CREAT | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW,
154 0600);
155 if (fd < 0) {
156 free(buf);
157 return BRIX_XMETA_ERR;
158 }
159 while (put < len) {
160 ssize_t w = pwrite(fd, buf + put, len - put, (off_t) put);
161
162 if (w < 0 && errno == EINTR) {
163 continue;
164 }
165 if (w <= 0) {
166 free(buf);
167 close(fd);
168 return BRIX_XMETA_ERR;
169 }
170 put += (size_t) w;
171 }
172 free(buf);
173 if (ftruncate(fd, (off_t) len) != 0 || fdatasync(fd) != 0) {
174 close(fd);
175 return BRIX_XMETA_ERR;
176 }
177 if (close(fd) != 0) {
178 return BRIX_XMETA_ERR;
179 }
180 (void) removexattr(path, BRIX_XMETA_PATH_XATTR); /* one carrier */
181 return BRIX_XMETA_OK;
182}

185brix_xmeta_path_remove(const char *path)
186{
187 char sc[PATH_MAX];
188
189 if (path == NULL) {
190 return BRIX_XMETA_OK;
191 }
192 (void) removexattr(path, BRIX_XMETA_PATH_XATTR);
193 if (brix_xmeta_sidecar_path(sc, sizeof(sc), path) == 0) {
194 (void) unlink(sc);
195 }
196 return BRIX_XMETA_OK;

199int
200brix_xmeta_path_lock(const char *path)
201{
202 char sc[PATH_MAX];
203 int fd;
204
205 fd = open(path, O_RDONLY | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW);
206 if (fd < 0) {
207 if (errno != ENOENT
208 || brix_xmeta_sidecar_path(sc, sizeof(sc), path) != 0)
209 {
210 return -1;
211 }
212 fd = open(sc, O_RDWR | O_CREAT | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW,
213 0600); /* worker-owned lock sidecar; 0600 (see path_save) */
214 if (fd < 0) {
215 return -1;
216 }
217 }
218 while (flock(fd, LOCK_EX) != 0) {
219 if (errno != EINTR) {
220 close(fd);
221 return -1;
222 }
223 }
224 return fd;
225}
