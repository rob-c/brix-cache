42/* ---- Try to load the record from the file's xattr carrier ----
43 *
44 * WHAT: Attempts to read and decode the metadata record from the preferred
45 * xattr carrier. Sets *done=1 and returns the decode result (or a hard error
46 * with errno set) when the outcome is definitive; sets *done=0 and returns
47 * BRIX_XMETA_OK when the xattr holds nothing, signalling the caller to fall
48 * back to the stock-readable sidecar.
49 *
50 * WHY: The xattr path is the fast, common case and is self-contained (alloc,
51 * read, decode, free). Isolating it keeps the loader orchestrator flat and
52 * lets the sidecar fallback be expressed as a single continue-or-return.
53 *
54 * HOW:
55 * 1. Allocate the max-sized xattr scratch buffer; malloc failure is a
56 * definitive ENOMEM error.
57 * 2. getxattr into it. A positive length is a present record: decode, free,
58 * and return the decode result as definitive.
59 * 3. Any non-positive length means no record in the xattr: free, clear *done,
60 * and return OK so the caller tries the sidecar.
61 */
62static int
63xmeta_path_load_from_xattr(const char *path, brix_xmeta_t *xm, int *done)
64{
65 uint8_t *buf;
66 ssize_t n;
67 int drc;
68
69 *done = 1;
70 buf = malloc(BRIX_XMETA_PATH_XATTR_MAX);
71 if (buf == NULL) {
72 errno = ENOMEM;
73 return BRIX_XMETA_ERR;
74 }
75 n = getxattr(path, BRIX_XMETA_PATH_XATTR, buf,
76 BRIX_XMETA_PATH_XATTR_MAX);
77 if (n > 0) {
78 drc = brix_xmeta_decode(buf, (size_t) n, xm);
79 free(buf);
80 return drc;
81 }
82 free(buf);
83 *done = 0; /* nothing in the xattr — try the sidecar */
84 return BRIX_XMETA_OK;
85}

87/* ---- Read and decode the record from an open sidecar fd ----
88 *
89 * WHAT: Reads the full contents of an already-open sidecar file and decodes it
90 * into *xm. Returns the decode result on success, BRIX_XMETA_FOREIGN when the
91 * sidecar is empty/oversized/short (not a valid record), or BRIX_XMETA_ERR
92 * (errno set) on allocation failure. Does NOT close fd — the caller owns it.
93 *
94 * WHY: The sidecar read is the bounded, syscall-heavy branch of the loader
95 * (stat, size-guard, alloc, interrupt-safe pread loop, decode). Pulling it
96 * into one helper keeps ownership of fd with the opener and the orchestrator
97 * free of the read machinery.
98 *
99 * HOW:
100 * 1. fstat and reject empty or over-BRIX_XMETA_MAX_SECTION files as FOREIGN.
101 * 2. Allocate a buffer sized to the file; malloc failure is an ENOMEM error.
102 * 3. pread until full, retrying on EINTR and stopping on EOF/error.
103 * 4. A short read is a truncated/foreign sidecar; otherwise decode and return.
104 */
105static int
106xmeta_path_read_sidecar_fd(int fd, brix_xmeta_t *xm)
107{
108 struct stat stb;
109 uint8_t *fbuf;
110 size_t flen, got = 0;
111 int drc;
112
113 if (fstat(fd, &stb) != 0 || stb.st_size <= 0
114 || (uint64_t) stb.st_size > BRIX_XMETA_MAX_SECTION)
115 {
116 return BRIX_XMETA_FOREIGN;
117 }
118 flen = (size_t) stb.st_size;
119 fbuf = malloc(flen);
120 if (fbuf == NULL) {
121 errno = ENOMEM;
122 return BRIX_XMETA_ERR;
123 }
124 while (got < flen) {
125 ssize_t r = pread(fd, fbuf + got, flen - got, (off_t) got);
126
127 if (r < 0 && errno == EINTR) {
128 continue;
129 }
130 if (r <= 0) {
131 break;
132 }
133 got += (size_t) r;
134 }
135 if (got != flen) {
136 free(fbuf);
137 return BRIX_XMETA_FOREIGN;
138 }
139 drc = brix_xmeta_decode(fbuf, flen, xm);
140 free(fbuf);
141 return drc;
142}

144/* ---- Load the unified metadata record for a path (xattr, then sidecar) ----
145 *
146 * WHAT: Loads the record into *xm, preferring the xattr carrier and falling
147 * back to the stock-readable sidecar. Returns BRIX_XMETA_OK, BRIX_XMETA_ERR
148 * (errno set), or BRIX_XMETA_FOREIGN (no valid record recorded).
149 *
150 * WHY: The two carriers must be tried in a fixed order with identical decode
151 * semantics; the orchestrator expresses that policy as a flat sequence while
152 * the per-carrier machinery lives in dedicated helpers.
153 *
154 * HOW:
155 * 1. Reject NULL arguments with EINVAL.
156 * 2. Try the xattr carrier; return its result if it was definitive.
157 * 3. Otherwise build the sidecar path and open it; a missing sidecar is
158 * FOREIGN, any other open error is a hard error.
159 * 4. Read and decode the sidecar, then close the fd and return the result.
160 */
161int
162brix_xmeta_path_load(const char *path, brix_xmeta_t *xm)
163{
164 char sc[PATH_MAX];
165 int fd, drc, done;
166
167 if (path == NULL || xm == NULL) {
168 errno = EINVAL;
169 return BRIX_XMETA_ERR;
170 }
171
172 drc = xmeta_path_load_from_xattr(path, xm, &done);
173 if (done) {
174 return drc;
175 }
176
177 if (brix_xmeta_sidecar_path(sc, sizeof(sc), path) != 0) {
178 errno = ENAMETOOLONG;
179 return BRIX_XMETA_ERR;
180 }
181 fd = open(sc, O_RDONLY | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW);
182 if (fd < 0) {
183 return (errno == ENOENT) ? BRIX_XMETA_FOREIGN : BRIX_XMETA_ERR;
184 }
185 drc = xmeta_path_read_sidecar_fd(fd, xm);
186 close(fd);
187 return drc;
188}

190/* ---- Try to store the record in the file's xattr carrier ----
191 *
192 * WHAT: Attempts setxattr of the encoded record onto the preferred xattr
193 * carrier. Sets *done=1 and returns BRIX_XMETA_OK (stored) or BRIX_XMETA_ERR
194 * (a hard setxattr failure, errno from setxattr) when the outcome is
195 * definitive; sets *done=0 and returns BRIX_XMETA_OK when the record cannot
196 * ride in this xattr (too big, or setxattr failed with an "unfit" errno),
197 * signalling the caller to write the sidecar instead.
198 *
199 * WHY: The xattr-vs-sidecar decision is a small policy — oversized records and
200 * filesystems that reject xattrs must fall back rather than fail. Isolating it
201 * keeps that classification in one place and the saver orchestrator flat.
202 *
203 * HOW:
204 * 1. Records over BRIX_XMETA_PATH_XATTR_MAX skip setxattr entirely (fall back).
205 * 2. A successful setxattr is a definitive store.
206 * 3. A setxattr failure that is not "unfit" is a definitive hard error.
207 * 4. Any remaining case (too big, or an unfit errno) clears *done for fallback.
208 */
209static int
210xmeta_path_save_to_xattr(const char *path, const uint8_t *buf, size_t len,
211 int *done)
212{
213 *done = 1;
214 if (len <= BRIX_XMETA_PATH_XATTR_MAX
215 && setxattr(path, BRIX_XMETA_PATH_XATTR, buf, len, 0) == 0)
216 {
217 return BRIX_XMETA_OK;
218 }
219 if (len <= BRIX_XMETA_PATH_XATTR_MAX && !xmeta_path_xattr_unfit(errno)) {
220 return BRIX_XMETA_ERR;
221 }
222 *done = 0; /* xattr unfit — caller writes the sidecar */
223 return BRIX_XMETA_OK;
224}

226/* ---- Write the encoded record to the stock-readable sidecar ----
227 *
228 * WHAT: Creates (or truncates) the sidecar at sc and writes len bytes from buf,
229 * then durably commits it. Returns BRIX_XMETA_OK on success or BRIX_XMETA_ERR
230 * on any open/write/truncate/sync/close failure. Does not free buf.
231 *
232 * WHY: The sidecar write is the bounded, syscall-heavy fallback path (create,
233 * interrupt-safe pwrite loop, truncate, fdatasync, close). Pulling it into one
234 * helper keeps the durability sequence together and the orchestrator flat.
235 *
236 * HOW:
237 * 1. Open the sidecar 0600 O_CREAT (see the SECURITY note below).
238 * 2. pwrite until the whole buffer is written, retrying on EINTR and failing
239 * on any short/zero write.
240 * 3. ftruncate to len (drop any prior tail), fdatasync, then close, mapping
241 * any failure to a hard error.
242 */
243static int
244xmeta_path_write_sidecar(const char *sc, const uint8_t *buf, size_t len)
245{
246 size_t put = 0;
247 int fd;
248
249 /* SECURITY: the metadata sidecar (cache .cinfo residency bitmap / CSI record)
250 * is created and read AS THE WORKER (server-managed sidecar, see meta.c
251 * vfs-seam-allow; CSI verify is a backend op). 0600 (not 0644) so it cannot
252 * leak cache residency / integrity metadata to a mapped low-priv uid on a
253 * shared filesystem. Safe: with impersonation=map the master is root and
254 * reads any-owner sidecar; without impersonation the worker owns and reads it. */
255 fd = open(sc, O_WRONLY | O_CREAT | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW,
256 0600);
257 if (fd < 0) {
258 return BRIX_XMETA_ERR;
259 }
260 while (put < len) {
261 ssize_t w = pwrite(fd, buf + put, len - put, (off_t) put);
262
263 if (w < 0 && errno == EINTR) {
264 continue;
265 }
266 if (w <= 0) {
267 close(fd);
268 return BRIX_XMETA_ERR;
269 }
270 put += (size_t) w;
271 }
272 if (ftruncate(fd, (off_t) len) != 0 || fdatasync(fd) != 0) {
273 close(fd);
274 return BRIX_XMETA_ERR;
275 }
276 if (close(fd) != 0) {
277 return BRIX_XMETA_ERR;
278 }
279 return BRIX_XMETA_OK;
280}

282/* ---- Save the unified metadata record for a path (xattr, else sidecar) ----
283 *
284 * WHAT: Encodes *xm and stores it in exactly one carrier — the xattr when it
285 * fits, otherwise the stock-readable sidecar — dropping the other carrier so
286 * only one copy survives. Returns BRIX_XMETA_OK or BRIX_XMETA_ERR (errno set).
287 *
288 * WHY: "One carrier" is the invariant: a successful xattr write removes any
289 * stale sidecar and a successful sidecar write removes the xattr. The
290 * orchestrator owns the encoded buffer and enforces that single-copy rule
291 * while the two carriers are handled by dedicated helpers.
292 *
293 * HOW:
294 * 1. Reject NULL arguments with EINVAL, then encode the record and build the
295 * sidecar path.
296 * 2. Try the xattr carrier; if definitive, free the buffer, unlink the stale
297 * sidecar on success, and return that result.
298 * 3. Otherwise write the sidecar, free the buffer, and on success remove the
299 * now-superseded xattr before returning OK.
300 */
301int
302brix_xmeta_path_save(const char *path, const brix_xmeta_t *xm)
303{
304 char sc[PATH_MAX];
305 uint8_t *buf = NULL;
306 size_t len = 0;
307 int rc, done;
308
309 if (path == NULL || xm == NULL) {
310 errno = EINVAL;
311 return BRIX_XMETA_ERR;
312 }
313 if (brix_xmeta_encode(xm, &buf, &len) != BRIX_XMETA_OK) {
314 return BRIX_XMETA_ERR;
315 }
316 if (brix_xmeta_sidecar_path(sc, sizeof(sc), path) != 0) {
317 free(buf);
318 errno = ENAMETOOLONG;
319 return BRIX_XMETA_ERR;
320 }
321
322 rc = xmeta_path_save_to_xattr(path, buf, len, &done);
323 if (done) {
324 free(buf);
325 if (rc == BRIX_XMETA_OK) {
326 (void) unlink(sc); /* one carrier: drop a stale sidecar */
327 }
328 return rc;
329 }
330
331 rc = xmeta_path_write_sidecar(sc, buf, len);
332 free(buf);
333 if (rc != BRIX_XMETA_OK) {
334 return rc;
335 }
336 (void) removexattr(path, BRIX_XMETA_PATH_XATTR); /* one carrier */
337 return BRIX_XMETA_OK;
338}

355int
356brix_xmeta_path_lock(const char *path)
357{
358 char sc[PATH_MAX];
359 int fd;
360
361 fd = open(path, O_RDONLY | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW);
362 if (fd < 0) {
363 if (errno != ENOENT
364 || brix_xmeta_sidecar_path(sc, sizeof(sc), path) != 0)
365 {
366 return -1;
367 }
368 fd = open(sc, O_RDWR | O_CREAT | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW,
369 0600); /* worker-owned lock sidecar; 0600 (see path_save) */
370 if (fd < 0) {
371 return -1;
372 }
373 }
374 while (flock(fd, LOCK_EX) != 0) {
375 if (errno != EINTR) {
376 close(fd);
377 return -1;
378 }
379 }
380 return fd;
381}
