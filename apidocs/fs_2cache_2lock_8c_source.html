13/* brix_cache_lock_is_stale — is the lock file at `lock_path` orphaned?
14 *
15 * WHY: the fill lock is an O_EXCL lock FILE, not an fcntl/flock the kernel
16 * releases on death. A worker SIGKILLed/crashed mid-fill (e.g. at reload's
17 * worker_shutdown_timeout) leaves it behind, and nothing else reclaims it
18 * (cache_reap skips *.lock; the eviction-sentinel reclaim is a different lock).
19 * The orphan survives reboots on disk, so without this check every later
20 * request for the entry polls to cache_lock_timeout and fails kXR_FileLocked
21 * forever, pinning a thread-pool thread for the whole timeout each time.
22 *
23 * HOW: the lock file records "pid=N". If that process is gone (kill(N,0)==ESRCH)
24 * the lock is definitively stale — after a reboot every pre-reboot pid is dead,
25 * so the very first request reclaims it. A live owner (kill==0) is a legitimate
26 * in-progress fill and is never reclaimed, however long it runs. Torn/legacy
27 * content with no parseable pid falls back to age: older than the fill timeout
28 * cannot belong to any still-waiting owner, so it is stale. EPERM (pid owned by
29 * another user) is treated as alive — conservative, never over-reclaims.
30 */
31static int
32brix_cache_lock_is_stale(const char *lock_path, time_t timeout)
33{
34 int fd;
35 ssize_t n;
36 char buf[64];
37 long pid = 0;
38 struct stat st;
39
40 fd = open(lock_path, O_RDONLY | O_NOCTTY | O_CLOEXEC);
41 if (fd < 0) {
42 return 0; /* vanished/unreadable — let caller retry */
43 }
44 n = read(fd, buf, sizeof(buf) - 1);
45 if (fstat(fd, &st) != 0) {
46 close(fd);
47 return 0;
48 }
49 close(fd);
50
51 if (n > 0) {
52 buf[n] = '\0';
53 if (sscanf(buf, "pid=%ld", &pid) == 1 && pid > 0) {
54 if (kill((pid_t) pid, 0) == 0) {
55 return 0; /* owner alive — legitimate fill */
56 }
57 return (errno == ESRCH); /* dead → stale; EPERM → assume alive */
58 }
59 }
60
61 /* No parseable owner (torn write / legacy lock): reclaim only once it is
62 * older than a full fill timeout, which no waiting owner can still hold. */
63 return (time(NULL) - st.st_mtime) > timeout;
64}

66/* brix_cache_try_lock — attempt exclusive lock on cache fill path
67 * WHAT: Creates a lock file (O_CREAT|O_EXCL) at t->lock_path to claim ownership of cache fill. Writes PID into the file for identification. Returns 1 if locked, 0 if another process holds it (EEXIST), -1 on any other error (including write failure or unlink cleanup). This is an atomic filesystem lock — no fcntl(2). */
68
69/* WHY: Atomic filesystem locking via O_CREAT|O_EXCL avoids the complexity of
70 * fcntl(2) advisory locks which require fd ownership and don't work across process
71 * boundaries. A lock file created with O_EXCL guarantees mutual exclusion at the
72 * kernel level — only one process can succeed in creating the file simultaneously.
73 * This approach is simpler, portable, and works correctly even when nginx workers
74 * are running under different UID/GID combinations (e.g., during privilege drops). */
75
76/* HOW: Three-step atomic acquisition. Step 1: open() with O_CREAT|O_EXCL — if another
77 * process already created the lock file, EEXIST is returned (another worker filling
78 * this cache entry). Step 2: write PID into lock file via snprintf+write — identifies
79 * the locking process for debugging. If write fails, close(fd) and unlink(lock_path)
80 * to clean up orphaned lock. Step 3: close(fd) without unlinking — the lock persists
81 * until the cache fill completes and cleanup removes it. Returns 1 (success), 0
82 * (EEXIST = someone else owns this), or -1 (any other error). */
83
84static int
85brix_cache_try_lock(brix_cache_fill_t *t)
86{
87 int fd, n;
88 char body[64];
89
90 fd = open(t->lock_path, O_CREAT | O_EXCL | O_WRONLY | O_NOCTTY | O_CLOEXEC, 0600);
91 if (fd < 0) {
92 return (errno == EEXIST) ? 0 : -1;
93 }
94
95 n = snprintf(body, sizeof(body), "pid=%ld\n", (long) getpid());
96 if (n > 0) {
97 ssize_t wr;
98
99 wr = write(fd, body, (size_t) n);
100 if (wr < 0) {
101 int err;
102
103 err = errno;
104 close(fd);
105 unlink(t->lock_path);
106 errno = err;
107 return -1;
108 }
109 }
110 close(fd);
111
112 return 1;
113}

115/* brix_cache_wait_or_lock — poll until cache file exists or claim fill ownership
116 * WHAT: Two-phase polling loop that either waits for the cache file to appear (another process already filling it) or claims ownership via lock. Phase 1: checks file_ready() — if file exists, returns 0 without locking. Phase 2: tries atomic lock — if locked, sets *owned=1 and fills. Polls at BRIX_CACHE_LOCK_POLL_USEC intervals until cache_lock_timeout expires (returns kXR_FileLocked). */
117
118/* WHY: Two-phase polling prevents race conditions between concurrent workers attempting
119 * to fill the same cache entry simultaneously. Without this loop, two workers might both
120 * try to open() the origin file and perform duplicate fills — wasting bandwidth and
121 * creating inconsistent cache state. The timeout ensures we don't hang indefinitely if
122 * a stale lock exists from a crashed process (the lock file cleanup in try_lock handles
123 * write failures, but orphaned locks can persist after SIGKILL). */
124
125/* HOW: Infinite loop with three checks per iteration. Check 1: file_ready() — the cache
126 * file already appeared (another worker completed fill first), return immediately without
127 * locking. Check 2: try_lock() — attempt O_CREAT|O_EXCL atomic lock; if success (*owned=1,
128 * caller will perform the origin fetch). Check 3: timeout comparison — time(NULL) - start >=
129 * conf->cache_lock_timeout (returns kXR_FileLocked with error message). Between checks:
130 * usleep(BRIX_CACHE_LOCK_POLL_USEC) provides backoff to reduce contention. Every failure
131 * path sets cache error via brix_cache_set_syserror() or brix_cache_set_error(). */
132
133int
134brix_cache_wait_or_lock(brix_cache_fill_t *t, int *owned)
135{
136 time_t start;
137
138 *owned = 0;
139 start = time(NULL);
140
141 for (;;) {
142 int ready;
143
144 ready = brix_cache_file_ready(t->cache_path);
145 if (ready == 1) {
146 return 0;
147 }
148 if (ready < 0) {
149 brix_cache_set_syserror(t, kXR_IOError,
150 "cache path is not usable");
151 return -1;
152 }
153
154 ready = brix_cache_try_lock(t);
155 if (ready == 1) {
156 *owned = 1;
157 return 0;
158 }
159 if (ready < 0) {
160 brix_cache_set_syserror(t, kXR_IOError,
161 "cache lock create failed");
162 return -1;
163 }
164
165 /* EEXIST: another worker claims the fill. If that owner is dead, its
166 * O_EXCL lock is orphaned and would strand this entry forever — reclaim
167 * it and retry immediately. A benign reclaim race at worst causes one
168 * duplicate origin fetch, which is safe: fills write a verified .part
169 * and atomically rename it into place (fetch.c). */
170 if (brix_cache_lock_is_stale(t->lock_path,
171 t->conf->cache_lock_timeout))
172 {
173 if (unlink(t->lock_path) == 0 && t->c != NULL) {
174 ngx_log_error(NGX_LOG_WARN, t->c->log, 0,
175 "brix: reclaimed stale cache-fill lock \"%s\" "
176 "(owner process gone)", t->lock_path);
177 }
178 continue; /* retry try_lock without sleeping */
179 }
180
181 if (time(NULL) - start >= t->conf->cache_lock_timeout) {
182 brix_cache_set_error(t, kXR_FileLocked, 0,
183 "timed out waiting for cache fill lock");
184 return -1;
185 }
186
187 usleep(BRIX_CACHE_LOCK_POLL_USEC);
188 }
189}
