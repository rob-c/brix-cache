1/*
2 * overlay.h — writable union overlay core for brixMount cvmfs-rw (pure C; no
3 * FUSE, no CVMFS, no ngx).
4 *
5 * WHAT: the union-layer primitives over an "upper" directory tree stored at
6 * <mnt>/.brixwrites/upper — classify a path (upper / masked / fall
7 * through to lower), whiteout + opaque markers, upper-tree mutations,
8 * atomic copy-up, merged-readdir support, and the --overlay-list /
9 * --overlay-reset CLI cores.
10 * WHY: full POSIX write semantics on top of a read-only CVMFS mount, with the
11 * local changes kept as a plain path-mirroring tree beside .brixcache —
12 * human-inspectable, diffable, rm -rf-resettable. Keeping the module
13 * FUSE/CVMFS-free makes every union corner unit-testable on tmp dirs.
14 * HOW: every upper access goes through a per-component O_NOFOLLOW walk rooted
15 * at the upper dirfd, so a symlink planted in the user-writable upper
16 * tree can never redirect the driver outside .brixwrites. The lower
17 * (read-only) layer is reached only via an injected read callback
18 * (copy-up); the FUSE driver composes overlay-first / cvmfs-fallback.
19 * Single-threaded by contract (the brixcvmfs FUSE loop runs -s).
20 * Functions return 0 / -errno (FUSE convention); fd-returning functions
21 * return fd >= 0 / -errno. Paths are repo-relative WITHOUT a leading
22 * slash ("" = the repo root), matching brixcvmfs's cat_path().
23 */

41/* union classification of one repo-relative path */
42typedef enum {
43 BRIX_OV_NONE = 0, /* not in upper, not masked → fall through to lower */
44 BRIX_OV_UPPER, /* present in upper (stat filled) */
45 BRIX_OV_MASKED, /* whiteout / opaque / shadowing upper file → the */
46 /* lower entry (if any) must read as ENOENT */
47} brix_ov_state;

157/* Walk <mountdir>/.brixwrites/upper and print one change per line to `out`:
158 * deleted <rel> whiteout marker

161 * upper <rel> files, when unmounted (no xattr available)
162 * Works identically on a live cvmfs-rw mount (via the /.brixwrites
163 * passthrough subtree) and on the raw on-disk tree. Returns an exit code:
