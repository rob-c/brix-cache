1/*
2 * streams.c — M8 parallel data streams (kXR_bind substreams).
3 *
4 * WHAT: Attach up to N-1 secondary TCP connections to a primary session via
5 * kXR_bind, and tear them down again.
6 * WHY: `xrdcp --streams N` opens extra bound channels — the parallel-transfer
7 * affordance of the protocol.
8 * HOW: Each secondary re-runs handshake + kXR_protocol [+ TLS] then sends
9 * kXR_bind{primary sessid} (brix_bind in conn.c). NOTE: this nginx data
10 * server accepts the bind and logs it, but kXR_read carries no pathid
11 * (wire_core_requests.h ClientReadRequest), so the server does not actually
12 * fan reads across substreams — the binds are established and the transfer
13 * still runs on the primary. Establishing them is exactly what the gate
14 * (tests/test_xrdcp_client_options.py) checks: BIND access-log entries +
15 * byte-exact round-trip. Best-effort: a secondary that won't bind is
16 * skipped, never failing the copy.
17 */

23int
24brix_streams_open(brix_streamset *ss, brix_conn *primary, int streams,
25 brix_status *st)
26{
27 int want, i;
28
29 memset(ss, 0, sizeof(*ss));
30 if (streams <= 1) {
31 return 0;
32 }
33 want = streams - 1;
34 if (want > XRDC_MAX_STREAMS - 1) {
35 want = XRDC_MAX_STREAMS - 1;
36 }
37
38 for (i = 0; i < want; i++) {
39 brix_status bst;
40 brix_status_clear(&bst);
41 if (brix_bind(&ss->sec[ss->n], primary, &bst) != 0) {
42 /* Best-effort: stop at the first failure, keep what bound. */
43 break;
44 }
45 ss->n++;
46 }
47 (void) st;
48 return ss->n;
49}

51void
52brix_streams_close(brix_streamset *ss)
53{
54 int i;
55 for (i = 0; i < ss->n; i++) {
56 /* A bound stream owns no session of its own — close it quietly, no
57 * kXR_endsess (that belongs to the primary). */
58 brix_tls_free(&ss->sec[i]);
59 if (ss->sec[i].io.fd >= 0) {
60 close(ss->sec[i].io.fd);
61 ss->sec[i].io.fd = -1;
62 }
63 }
64 ss->n = 0;
65}
