1/*
2 * sd_pblock_namespace.c — namespace, directory and xattr vtable slots for the
3 * pblock storage driver.
4 *
5 * WHAT: Implements the metadata-plane vtable slots of brix_sd_pblock_driver:
6 * stat, unlink, mkdir, setattr, rename and server_copy over the SQLite
7 * catalog; directory iteration (opendir/readdir/closedir); and the object
8 * xattr CRUD (get/list/set/remove). server_copy duplicates a file's blocks
9 * under a fresh blob id; rename is a pure catalog row move (blocks are
10 * id-addressed, so no bytes are touched).
11 *
12 * WHY: Split out of sd_pblock.c (phase-79) to keep every pblock file under the
13 * ~500-line, one-concept cap. These are the namespace mutations and reads,
14 * kept separate from the hot byte path (sd_pblock_io.c), the object
15 * lifecycle (sd_pblock.c) and the staged-publish path (sd_pblock_staged.c).
16 * The functions are non-static because the driver descriptor names them;
17 * sd_pblock_drop_dst is additionally reused by staged-commit's replace.
18 * Declarations live in sd_pblock_internal.h.
19 *
20 * HOW: Every operation reaches the namespace through the catalog API
21 * (sd_pblock_catalog.h) and touches block files only through the packed-block
22 * engine (pblock_store.h) — no raw byte syscalls here. ngx-free (libc +
23 * catalog + engine) so the module and standalone unit test compile it
24 * identically. Gated by BRIX_HAVE_SQLITE like the rest of the backend.
25 */

67ngx_int_t
68sd_pblock_stat(brix_sd_instance_t *inst, const char *path,
69 brix_sd_stat_t *out)
70{
71 pblock_state_t *st = inst->state;
72 pblock_meta meta;
73 int rc;
74
75 rc = pblock_catalog_lookup(st->cat, path, &meta);
76 if (rc < 0) {
77 return NGX_ERROR;
78 }
79 if (rc == 1) {
80 errno = ENOENT;
81 return NGX_ERROR;
82 }
83 /* F12/F13: an object recorded under a transform the current export is not
84 * configured for is undecodable — refuse it at the metadata boundary (EIO)
85 * so no read fast-path (sendfile/pread) ever serves its transformed bytes as
86 * another kind. Directories carry no transform (xform stays ""). */
87 if (!meta.is_dir
88 && pblock_xform_kind_from_name(meta.xform) != st->xform.kind)
89 {
90 errno = EIO;
91 return NGX_ERROR;
92 }
93 /* F9: a fresh creation is invisible to stat for the visibility window
94 * (S3 HEAD-after-PUT), and a fresh update may serve the pre-update row
95 * for the stale window (S3 HEAD-after-overwrite). */
96 if (st->lab != NULL) {
97 if (pblock_anomaly_hidden(st, path)) {
98 errno = ENOENT;
99 return NGX_ERROR;
100 }
101 (void) pblock_anomaly_stale(st, path, &meta.size, &meta.mtime);
102 }
103 pblock_fill_sd_stat(&meta, path, out);
104 return NGX_OK;
105}

107ngx_int_t
108sd_pblock_unlink(brix_sd_instance_t *inst, const char *path, int is_dir)
109{
110 pblock_state_t *st = inst->state;
111 pblock_meta meta;
112 int rc;
113
114 /* F6: rmdir on the reserved control namespace drops a snapshot — handled
115 * before any real catalog work (there is no such directory to remove). */
116 if (st->snap && is_dir && pblock_snap_ctl_path(path)) {
117 return pblock_snap_ctl_rmdir(st, path) == 0 ? NGX_OK : NGX_ERROR;
118 }
119
120 rc = pblock_catalog_lookup(st->cat, path, &meta);
121 if (rc < 0) {
122 return NGX_ERROR;
123 }
124 if (rc == 1) {
125 errno = ENOENT;
126 return NGX_ERROR;
127 }
128
129 /* F15: a live foreign lease must not be dissolvable by deleting the file
130 * under it — EBUSY (kXR_FileLocked / 423) until it is released/expires.
131 * The plain slot has no requester identity, so uid 0 (the service). */
132 if (st->locks && pblock_locks_ns_check(st, path, 0) != 0) {
133 return NGX_ERROR;
134 }
135
136 if (is_dir) {
137 int children;
138
139 if (!meta.is_dir) {
140 errno = ENOTDIR;
141 return NGX_ERROR;
142 }
143 children = pblock_catalog_child_count(st->cat, path);
144 if (children < 0) {
145 return NGX_ERROR;
146 }
147 if (children > 0) {
148 errno = ENOTEMPTY;
149 return NGX_ERROR;
150 }
151 } else if (meta.is_dir) {
152 errno = EISDIR;
153 return NGX_ERROR;
154 }
155
156 if (!meta.is_dir) {
157 /* F11: move the object into the trash BEFORE releasing it. trash_push
158 * holds its blob, so the release below only decrements (a copy-on-write
159 * transfer of the reference to the trash row). A failed push is
160 * fail-open — the unlink just frees the blob as usual, no trash entry. */
161 if (st->trash) {
162 (void) pblock_hist_trash_push(st, path, &meta);
163 }
164 /* F10-aware release: the last reference removes blocks + csi rows
165 * (byte-identical to the pre-F10 path with refs off); a still-shared
166 * blob just loses one reference. */
167 pblock_refs_release(st, meta.blob_id, meta.size, meta.block_size);
168 if (st->nearline) { /* F4: residency dies with the file */
169 pblock_nearline_drop(st, path);
170 }
171 if (st->lab != NULL) { /* F9: event history dies with it */
172 pblock_anomaly_drop(st, path);
173 }
174 if (st->locks) { /* F15: stale/own rows die with it */
175 pblock_locks_drop(st, path);
176 }
177 /* F7: blocks are gone but the row still points at them — a crash here
178 * leaves a dangling catalog row for pblock-fsck to flag and --gc. */
179 pblock_lab_crash(st->lab, "before_unlink_row");
180 }
181 {
182 ngx_int_t rc2 = pblock_catalog_remove(st->cat, path) == 0
183 ? NGX_OK : NGX_ERROR;
184
185 if (st->audit) { /* F17 */
186 pblock_audit_log(st->cat, is_dir ? "rmdir" : "unlink", path, "",
187 meta.uid, meta.gid, rc2 == NGX_OK ? 0 : -1,
188 rc2 == NGX_OK ? 0 : errno);
189 }
190 return rc2;
191 }
192}

194/* sd_pblock_mkdir_as — mkdir recording (uid, gid) as the new directory's
195 * owner. The plain slot passes 0/0 (service); mkdir_cred (sd_pblock_cred.c)
196 * passes the requester's resolved catalog ids. */
197ngx_int_t
198sd_pblock_mkdir_as(brix_sd_instance_t *inst, const char *path, mode_t mode,
199 uint32_t uid, uint32_t gid)
200{
201 pblock_state_t *st = inst->state;
202 pblock_meta meta;
203
204 /* F11: mkdir /.pblock/undelete/<path> pops <path> out of the trash — handled
205 * before any real catalog work, and before the F6 dispatch since it owns a
206 * distinct reserved sub-namespace. Service-only for the same reason as F6. */
207 if (st->trash && pblock_hist_ctl_mkdir_match(path)) {
208 return pblock_hist_ctl_mkdir(st, path) == 0 ? NGX_OK : NGX_ERROR;
209 }
210
211 /* F6: mkdir on the reserved control namespace takes/restores a snapshot —
212 * handled before any real catalog work. Reached only through the inner
213 * (service) mkdir path, so snapshot control is service-only. */
214 if (st->snap && pblock_snap_ctl_path(path)) {
215 return pblock_snap_ctl_mkdir(st, path) == 0 ? NGX_OK : NGX_ERROR;
216 }
217
218 if (pblock_quota_admit(st, uid, 0, 1) != 0) { /* F5 */
219 return NGX_ERROR;
220 }
221
222 memset(&meta, 0, sizeof(meta));
223 meta.is_dir = 1;
224 meta.mtime = meta.ctime = pblock_now();
225 meta.mode = S_IFDIR | (mode & 0777);
226 meta.uid = uid;
227 meta.gid = gid;
228 /* One INSERT — the PRIMARY KEY constraint is the existence check (EEXIST),
229 * so no separate lookup is needed. */
230 {
231 ngx_int_t rc2 = pblock_catalog_create(st->cat, path, &meta) == 0
232 ? NGX_OK : NGX_ERROR;
233
234 if (st->audit) { /* F17 */
235 pblock_audit_log(st->cat, "mkdir", path, "", uid, gid,
236 rc2 == NGX_OK ? 0 : -1, rc2 == NGX_OK ? 0 : errno);
237 }
238 return rc2;
239 }
240}

248/* sd_pblock_setattr — apply kXR_chmod (mode), chown (owner) and/or kXR_setattr
249 * (times) to an existing catalog row. The catalog records mode + owner (the
250 * catalog-internal synthetic uid/gid) + mtime + ctime; atime has no column and
251 * is accepted-and-ignored (object-store semantics) rather than failing — a
252 * setattr an export cannot represent should not break the op. ctime is bumped
253 * on any change, matching POSIX. This plain slot applies the caller's values
254 * verbatim (service semantics); ownership POLICY (who may chown) lives in the
255 * identity-aware setattr_cred slot. */
256ngx_int_t
257sd_pblock_setattr(brix_sd_instance_t *inst, const char *path,
258 const brix_sd_setattr_t *attr)
259{
260 pblock_state_t *st = inst->state;
261 int set_mtime = 0;
262 int64_t mtime = 0;
263 int64_t uid = -1, gid = -1;
264
265 if (attr->set_times && attr->mtime.tv_nsec != UTIME_OMIT) {
266 set_mtime = 1;
267 mtime = (attr->mtime.tv_nsec == UTIME_NOW)
268 ? pblock_now() : (int64_t) attr->mtime.tv_sec;
269 }
270 if (attr->set_owner) {
271 /* (uid_t)-1 / (gid_t)-1 mean "keep" (chown(2)); map to the catalog's
272 * signed -1 sentinel. */
273 uid = attr->uid == (uid_t) -1 ? -1 : (int64_t) attr->uid;
274 gid = attr->gid == (gid_t) -1 ? -1 : (int64_t) attr->gid;
275 }
276
277 /* One UPDATE — no read-modify-write. ENOENT (rc<0 with errno) when absent. */
278 return pblock_catalog_setattr(st->cat, path, attr->set_mode ? 1 : 0,
279 attr->mode & 0777, attr->set_owner ? 1 : 0, uid, gid,
280 set_mtime, mtime, pblock_now()) == 0
281 ? NGX_OK : NGX_ERROR;
282}

284/* sd_pblock_drop_dst — clear an existing rename/copy destination: reject a
285 * non-empty directory, remove a file's blocks, then drop the catalog row. */
286ngx_int_t
287sd_pblock_drop_dst(pblock_state_t *st, const char *dst,
288 const pblock_meta *dmeta)
289{
290 if (dmeta->is_dir) {
291 int children = pblock_catalog_child_count(st->cat, dst);
292
293 if (children < 0) {
294 return NGX_ERROR;
295 }
296 if (children > 0) {
297 errno = ENOTEMPTY;
298 return NGX_ERROR;
299 }
300 } else {
301 /* F10-aware release (see sd_pblock_unlink). */
302 pblock_refs_release(st, dmeta->blob_id, dmeta->size,
303 dmeta->block_size);
304 }
305 if (st->nearline) { /* F4: overwritten dst is replaced */
306 pblock_nearline_drop(st, dst);
307 }
308 if (st->lab != NULL) { /* F9: replaced dst's history too */
309 pblock_anomaly_drop(st, dst);
310 }
311 if (st->locks) { /* F15: replaced dst's rows too */
312 pblock_locks_drop(st, dst);
313 }
314 return pblock_catalog_remove(st->cat, dst) == 0 ? NGX_OK : NGX_ERROR;
315}

317ngx_int_t
318sd_pblock_rename(brix_sd_instance_t *inst, const char *src, const char *dst,
319 int noreplace)
320{
321 pblock_state_t *st = inst->state;
322 pblock_meta dmeta;
323 int rc;
324
325 rc = pblock_catalog_lookup(st->cat, src, NULL);
326 if (rc < 0) {
327 return NGX_ERROR;
328 }
329 if (rc == 1) {
330 errno = ENOENT;
331 return NGX_ERROR;
332 }
333
334 /* F15: renaming a leased src (or over a leased dst) is the classic lock
335 * bypass — refuse while any live foreign lease exists on either name. */
336 if (st->locks
337 && (pblock_locks_ns_check(st, src, 0) != 0
338 || pblock_locks_ns_check(st, dst, 0) != 0))
339 {
340 return NGX_ERROR;
341 }
342
343 rc = pblock_catalog_lookup(st->cat, dst, &dmeta);
344 if (rc < 0) {
345 return NGX_ERROR;
346 }
347 if (rc == 0) {
348 if (noreplace) {
349 errno = EEXIST;
350 return NGX_ERROR;
351 }
352 if (sd_pblock_drop_dst(st, dst, &dmeta) != NGX_OK) {
353 return NGX_ERROR;
354 }
355 }
356
357 /* Blocks are id-addressed, so moving the catalog row carries the content
358 * with it (and reparents a directory subtree) without touching any bytes. */
359 {
360 ngx_int_t rc2 = pblock_catalog_rename(st->cat, src, dst) == 0
361 ? NGX_OK : NGX_ERROR;
362
363 if (rc2 == NGX_OK && st->nearline) { /* F4: residency follows the path */
364 pblock_nearline_rename(st, src, dst);
365 }
366 if (rc2 == NGX_OK && st->lab != NULL) { /* F9: events follow the path */
367 pblock_anomaly_rename(st, src, dst);
368 }
369 if (rc2 == NGX_OK && st->locks) { /* F15: leases follow the path */
370 pblock_locks_rename(st, src, dst);
371 }
372 if (st->audit) { /* F17 */
373 pblock_audit_log(st->cat, "rename", src, dst, 0, 0,
374 rc2 == NGX_OK ? 0 : -1, rc2 == NGX_OK ? 0 : errno);
375 }
376 return rc2;
377 }

380/* sd_pblock_server_copy_as — server-side copy whose destination row is owned
381 * by (uid, gid): the copier, not the source's owner (POSIX cp semantics). The
382 * plain slot passes 0/0 (service); server_copy_cred passes the requester. */
383ngx_int_t
384sd_pblock_server_copy_as(brix_sd_instance_t *inst, const char *src,
385 const char *dst, off_t *bytes_out, uint32_t uid, uint32_t gid)
386{
387 pblock_state_t *st = inst->state;
388 pblock_meta smeta, dmeta;
389 int64_t last, i;
390 int rc;
391
392 rc = pblock_catalog_lookup(st->cat, src, &smeta);
393 if (rc < 0) {
394 return NGX_ERROR;
395 }
396 if (rc == 1) {
397 errno = ENOENT;
398 return NGX_ERROR;
399 }
400 if (smeta.is_dir) {
401 errno = EISDIR;
402 return NGX_ERROR;
403 }
404
405 if (st->quota) { /* F5 */
406 pblock_meta dexist;
407 int drc = pblock_catalog_lookup(st->cat, dst, &dexist);
408
409 if (pblock_quota_admit(st, uid,
410 (int64_t) smeta.size - (drc == 0 ? dexist.size : 0),
411 drc == 0 ? 0 : 1) != 0)
412 {
413 return NGX_ERROR;
414 }
415 }
416
417 if (st->refs) { /* F10: CoW copy */
418 /* O(metadata) copy: both rows share the source blob; the first write
419 * to either breaks the share at open. No bytes move. */
420 pblock_meta dexist;
421 int dhad = pblock_catalog_lookup(st->cat, dst, &dexist) == 0;
422
423 if (pblock_refs_bump(st, smeta.blob_id, smeta.size,
424 smeta.block_size) != 0)
425 {
426 return NGX_ERROR;
427 }
428 memset(&dmeta, 0, sizeof(dmeta));
429 memcpy(dmeta.blob_id, smeta.blob_id, sizeof(dmeta.blob_id));
430 dmeta.is_dir = 0;
431 dmeta.size = smeta.size;
432 dmeta.block_size = smeta.block_size;
433 dmeta.mtime = dmeta.ctime = pblock_now();
434 dmeta.mode = smeta.mode;
435 dmeta.uid = uid;
436 dmeta.gid = gid;
437 if (pblock_catalog_put(st->cat, dst, &dmeta) != 0) {
438 int err = errno;
439
440 pblock_refs_release(st, smeta.blob_id, smeta.size,
441 smeta.block_size);
442 errno = err;
443 return NGX_ERROR;
444 }
445 if (dhad && !dexist.is_dir) { /* the replaced dst's blob loses a ref */
446 pblock_refs_release(st, dexist.blob_id, dexist.size,
447 dexist.block_size);
448 }
449 if (st->lab != NULL) { /* F9 */
450 if (dhad) {
451 pblock_anomaly_updated(st, dst, dexist.size, dexist.mtime);
452 } else {
453 pblock_anomaly_created(st, dst);
454 }
455 }
456 if (bytes_out != NULL) {
457 *bytes_out = (off_t) smeta.size;
458 }
459 /* No csi flush: the shared blob's integrity rows already exist. */
460 if (st->audit) { /* F17 */
461 char aux[32];
462
463 snprintf(aux, sizeof(aux), "cow=1 w=%lld",
464 (long long) smeta.size);
465 pblock_audit_log(st->cat, "copy", dst, aux, uid, gid, 0, 0);
466 }
467 return NGX_OK;
468 }
469
470 memset(&dmeta, 0, sizeof(dmeta));
471 if (pblock_gen_blob_id(dmeta.blob_id) != 0
472 || pblock_ensure_obj_dir(st, dmeta.blob_id) != 0)
473 {
474 return NGX_ERROR;
475 }
476
477 last = pblock_last_block(smeta.size, smeta.block_size);
478 for (i = 0; i <= last; i++) {
479 char sp[PATH_MAX], dp[PATH_MAX];
480
481 if (pblock_block_path(st, smeta.blob_id, i, sp, sizeof(sp)) != 0
482 || pblock_block_path(st, dmeta.blob_id, i, dp, sizeof(dp)) != 0
483 || pblock_copy_one_block(sp, dp) < 0)
484 {
485 int err = errno;
486
487 pblock_remove_blocks(st, dmeta.blob_id, smeta.size,
488 smeta.block_size);
489 errno = err;
490 return NGX_ERROR;
491 }
492 }
493
494 dmeta.is_dir = 0;
495 dmeta.size = smeta.size;
496 dmeta.block_size = smeta.block_size;
497 dmeta.mtime = dmeta.ctime = pblock_now();
498 dmeta.mode = smeta.mode;
499 dmeta.uid = uid;
500 dmeta.gid = gid;
501
502 {
503 pblock_meta dexist;
504 int dhad = 0;
505
506 /* F9: is this copy a create or an overwrite of dst? The pre-put row
507 * is what a stale stat will serve. */
508 if (st->lab != NULL) {
509 dhad = pblock_catalog_lookup(st->cat, dst, &dexist) == 0;
510 }
511 if (pblock_catalog_put(st->cat, dst, &dmeta) != 0) {
512 int err = errno;
513
514 pblock_remove_blocks(st, dmeta.blob_id, smeta.size,
515 smeta.block_size);
516 errno = err;
517 return NGX_ERROR;
518 }
519 if (st->lab != NULL) {
520 if (dhad) {
521 pblock_anomaly_updated(st, dst, dexist.size, dexist.mtime);
522 } else {
523 pblock_anomaly_created(st, dst);
524 }
525 }
526 }
527 if (bytes_out != NULL) {
528 *bytes_out = (off_t) smeta.size;
529 }
530 if (st->csi) { /* F3: tag the copy */
531 (void) pblock_csi_flush(st, dmeta.blob_id, dmeta.size,
532 dmeta.block_size, 0, INT64_MAX);
533 }
534 if (st->audit) { /* F17 */
535 char aux[32];
536
537 snprintf(aux, sizeof(aux), "w=%lld", (long long) smeta.size);
538 pblock_audit_log(st->cat, "copy", dst, aux, uid, gid, 0, 0);
539 }
540 return NGX_OK;
541}

552brix_sd_dir_t *
553sd_pblock_opendir(brix_sd_instance_t *inst, const char *path, int *err_out)
554{
555 pblock_state_t *st = inst->state;
556 brix_sd_dir_t *dir;
557 pblock_dir_t *pd;
558 pblock_meta meta;
559 int rc;
560
561 rc = pblock_catalog_lookup(st->cat, path, &meta);
562 if (rc < 0) {
563 if (err_out != NULL) { *err_out = errno; }
564 return NULL;
565 }
566 if (rc == 1) {
567 if (err_out != NULL) { *err_out = ENOENT; }
568 return NULL;
569 }
570 if (!meta.is_dir) {
571 if (err_out != NULL) { *err_out = ENOTDIR; }
572 return NULL;
573 }
574
575 dir = calloc(1, sizeof(*dir));
576 pd = calloc(1, sizeof(*pd));
577 if (dir == NULL || pd == NULL) {
578 free(dir);
579 free(pd);
580 if (err_out != NULL) { *err_out = ENOMEM; }
581 return NULL;
582 }
583
584 pd->it = pblock_catalog_opendir(st->cat, path);
585 if (pd->it == NULL) {
586 if (err_out != NULL) { *err_out = errno; }
587 free(dir);
588 free(pd);
589 return NULL;
590 }
591 snprintf(pd->dir, sizeof(pd->dir), "%s", path);
592 dir->inst = inst;
593 dir->state = pd;
594 return dir;
595}

623ngx_int_t
624sd_pblock_closedir(brix_sd_dir_t *d)
625{
