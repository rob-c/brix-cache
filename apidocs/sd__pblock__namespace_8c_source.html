1/*
2 * sd_pblock_namespace.c — namespace, directory and xattr vtable slots for the
3 * pblock storage driver.
4 *
5 * WHAT: Implements the metadata-plane vtable slots of brix_sd_pblock_driver:
6 * stat, unlink, mkdir, setattr, rename and server_copy over the SQLite
7 * catalog; directory iteration (opendir/readdir/closedir); and the object
8 * xattr CRUD (get/list/set/remove). server_copy duplicates a file's blocks
9 * under a fresh blob id; rename is a pure catalog row move (blocks are
10 * id-addressed, so no bytes are touched).
11 *
12 * WHY: Split out of sd_pblock.c (phase-79) to keep every pblock file under the
13 * ~500-line, one-concept cap. These are the namespace mutations and reads,
14 * kept separate from the hot byte path (sd_pblock_io.c), the object
15 * lifecycle (sd_pblock.c) and the staged-publish path (sd_pblock_staged.c).
16 * The functions are non-static because the driver descriptor names them;
17 * sd_pblock_drop_dst is additionally reused by staged-commit's replace.
18 * Declarations live in sd_pblock_internal.h.
19 *
20 * HOW: Every operation reaches the namespace through the catalog API
21 * (sd_pblock_catalog.h) and touches block files only through the packed-block
22 * engine (pblock_store.h) — no raw byte syscalls here. ngx-free (libc +
23 * catalog + engine) so the module and standalone unit test compile it
24 * identically. Gated by BRIX_HAVE_SQLITE like the rest of the backend.
25 */

106 return NGX_ERROR;
107 }
108
109 if (!meta.is_dir) {
110 pblock_remove_blocks(st, meta.blob_id, meta.size, meta.block_size);
111 }
112 return pblock_catalog_remove(st->cat, path) == 0 ? NGX_OK : NGX_ERROR;

130/* sd_pblock_setattr — apply kXR_chmod (mode) and/or kXR_setattr (times) to an
131 * existing catalog row. The catalog records mode + mtime + ctime, so those are
132 * honoured; it has no owner (uid/gid) or atime columns, so set_owner and atime
133 * are accepted-and-ignored (object-store semantics) rather than failing — a chown
134 * an export cannot represent should not break the op. ctime is bumped on any
135 * change, matching POSIX. lookup→modify→put reuses the upsert path. */
136ngx_int_t
137sd_pblock_setattr(brix_sd_instance_t *inst, const char *path,
138 const brix_sd_setattr_t *attr)
139{
140 pblock_state_t *st = inst->state;
141 int set_mtime = 0;
142 int64_t mtime = 0;
143
144 if (attr->set_times && attr->mtime.tv_nsec != UTIME_OMIT) {
145 set_mtime = 1;
146 mtime = (attr->mtime.tv_nsec == UTIME_NOW)
147 ? pblock_now() : (int64_t) attr->mtime.tv_sec;
148 }
149
150 /* One UPDATE — no read-modify-write. ENOENT (rc<0 with errno) when absent. */
151 return pblock_catalog_setattr(st->cat, path, attr->set_mode ? 1 : 0,
152 attr->mode & 0777, set_mtime, mtime, pblock_now()) == 0
153 ? NGX_OK : NGX_ERROR;
154}

181 int noreplace)
182{

188 if (rc < 0) {

191 if (rc == 1) {
192 errno = ENOENT;
193 return NGX_ERROR;
194 }
195
196 rc = pblock_catalog_lookup(st->cat, dst, &dmeta);
197 if (rc < 0) {
198 return NGX_ERROR;
199 }
200 if (rc == 0) {
201 if (noreplace) {
202 errno = EEXIST;
203 return NGX_ERROR;
204 }
205 if (sd_pblock_drop_dst(st, dst, &dmeta) != NGX_OK) {
206 return NGX_ERROR;
207 }
208 }
209
210 /* Blocks are id-addressed, so moving the catalog row carries the content
211 * with it (and reparents a directory subtree) without touching any bytes. */
212 return pblock_catalog_rename(st->cat, src, dst) == 0 ? NGX_OK : NGX_ERROR;

215ngx_int_t
216sd_pblock_server_copy(brix_sd_instance_t *inst, const char *src,
217 const char *dst, off_t *bytes_out)
218{
219 pblock_state_t *st = inst->state;
220 pblock_meta smeta, dmeta;
221 int64_t last, i;
222 int rc;
223
224 rc = pblock_catalog_lookup(st->cat, src, &smeta);
225 if (rc < 0) {
226 return NGX_ERROR;
227 }
228 if (rc == 1) {
229 errno = ENOENT;
230 return NGX_ERROR;
231 }
232 if (smeta.is_dir) {
233 errno = EISDIR;
234 return NGX_ERROR;
235 }
236
237 memset(&dmeta, 0, sizeof(dmeta));
238 if (pblock_gen_blob_id(dmeta.blob_id) != 0
239 || pblock_ensure_obj_dir(st, dmeta.blob_id) != 0)
240 {
241 return NGX_ERROR;
242 }
243
244 last = pblock_last_block(smeta.size, smeta.block_size);
245 for (i = 0; i <= last; i++) {
246 char sp[PATH_MAX], dp[PATH_MAX];
247
248 if (pblock_block_path(st, smeta.blob_id, i, sp, sizeof(sp)) != 0
249 || pblock_block_path(st, dmeta.blob_id, i, dp, sizeof(dp)) != 0
250 || pblock_copy_one_block(sp, dp) < 0)
251 {
252 int err = errno;
253
254 pblock_remove_blocks(st, dmeta.blob_id, smeta.size,
255 smeta.block_size);
256 errno = err;
257 return NGX_ERROR;
258 }
259 }
260
261 dmeta.is_dir = 0;
262 dmeta.size = smeta.size;
263 dmeta.block_size = smeta.block_size;
264 dmeta.mtime = dmeta.ctime = pblock_now();
265 dmeta.mode = smeta.mode;
266
267 if (pblock_catalog_put(st->cat, dst, &dmeta) != 0) {
268 int err = errno;
269
270 pblock_remove_blocks(st, dmeta.blob_id, smeta.size, smeta.block_size);
271 errno = err;
272 return NGX_ERROR;
273 }
274 if (bytes_out != NULL) {
275 *bytes_out = (off_t) smeta.size;
276 }
277 return NGX_OK;
278}

282brix_sd_dir_t *
283sd_pblock_opendir(brix_sd_instance_t *inst, const char *path, int *err_out)
284{
285 pblock_state_t *st = inst->state;
286 brix_sd_dir_t *dir;
287 pblock_dir_t *pd;
288 pblock_meta meta;
289 int rc;
290
291 rc = pblock_catalog_lookup(st->cat, path, &meta);
292 if (rc < 0) {
293 if (err_out != NULL) { *err_out = errno; }
294 return NULL;
295 }
296 if (rc == 1) {
297 if (err_out != NULL) { *err_out = ENOENT; }
298 return NULL;
299 }
300 if (!meta.is_dir) {
301 if (err_out != NULL) { *err_out = ENOTDIR; }
302 return NULL;
303 }
304
305 dir = calloc(1, sizeof(*dir));
306 pd = calloc(1, sizeof(*pd));
307 if (dir == NULL || pd == NULL) {
308 free(dir);
309 free(pd);
310 if (err_out != NULL) { *err_out = ENOMEM; }
311 return NULL;
312 }
313
314 pd->it = pblock_catalog_opendir(st->cat, path);
315 if (pd->it == NULL) {
316 if (err_out != NULL) { *err_out = errno; }
317 free(dir);
318 free(pd);
319 return NULL;
320 }
321 dir->inst = inst;
322 dir->state = pd;
323 return dir;
324}
