26/*
27 * Load all PEM-encoded CRLs from a single file into the store.
28 * Returns the number of CRLs added, or -1 if the file cannot be opened.
29 */
30static int
31pki_load_crls_from_file(X509_STORE *store, const char *path, ngx_log_t *log)
32{
33 FILE *fp;
34 X509_CRL *crl;
35 int count;
36
37 fp = fopen(path, "r");
38 if (fp == NULL) {
39 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
40 "brix_pki: cannot open CRL file \"%s\"", path);
41 return -1;
42 }
43 fcntl(fileno(fp), F_SETFD, FD_CLOEXEC);
44
45 count = 0;
46 while ((crl = PEM_read_X509_CRL(fp, NULL, NULL, NULL)) != NULL) {
47 if (X509_STORE_add_crl(store, crl)) {
48 count++;
49 } else {
50 ngx_log_error(NGX_LOG_WARN, log, 0,
51 "brix_pki: failed to add CRL from \"%s\"", path);
52 }
53 X509_CRL_free(crl);
54 }
55
56 (void) fclose(fp); /* phase74-fp: read-only stream, close failure cannot lose data */
57 return count;
58}

60/*
61 * WHAT: decide whether a directory entry name is a CRL file we should load.
62 * WHY: grid CA CRL directories mix CRLs with hash-link and metadata files; we
63 * load only *.pem and the *.r0-*.r9 grid CRL naming convention. Isolating
64 * the predicate keeps the scan loop's branching (and CCN) contained here.
65 * HOW: suffix-test the name; return 1 on a match, 0 otherwise.
66 */
67static int
68pki_crl_name_matches(const char *name)
69{
70 size_t nlen = strlen(name);
71
72 if (nlen > 4 && strcmp(name + nlen - 4, ".pem") == 0) {
73 return 1;
74 }
75 if (nlen > 3 && name[nlen - 3] == '.'
76 && name[nlen - 2] == 'r'
77 && name[nlen - 1] >= '0' && name[nlen - 1] <= '9')
78 {
79 return 1;
80 }
81 return 0;
82}

84/*
85 * WHAT: load CRLs from one matched directory entry into the store.
86 * WHY: factors the per-entry path-join / regular-file guard / load out of the
87 * scan loop so the loop body is a single call and the loop's CCN drops.
88 * HOW: join dir+name, skip anything that is not a stat-able regular file, then
89 * delegate to pki_load_crls_from_file. Returns CRLs added (>=0); a
90 * non-regular or over-long entry contributes 0 (silently skipped, exactly
91 * as before).
92 */
93static int
94pki_load_crls_from_dirent(X509_STORE *store, const char *dir_path,
95 const char *name, ngx_log_t *log)
96{
97 char fpath[PATH_MAX];
98 struct stat st;
99 int n;
100
101 n = snprintf(fpath, sizeof(fpath), "%s/%s", dir_path, name);
102 if (n < 0 || (size_t) n >= sizeof(fpath)) {
103 return 0;
104 }
105
106 if (stat(fpath, &st) != 0 || !S_ISREG(st.st_mode)) {
107 return 0;
108 }
109
110 n = pki_load_crls_from_file(store, fpath, log);
111 return (n > 0) ? n : 0;
112}

114/*
115 * WHAT: scan a CRL directory, loading every matching *.pem / *.r[0-9] CRL.
116 * WHY: separates the directory-iteration stage from the file-vs-directory
117 * dispatch so each function keeps one job and a low CCN.
118 * HOW: opendir + readdir loop; delegate matching to pki_crl_name_matches and
119 * per-entry loading to pki_load_crls_from_dirent. Returns total CRLs
120 * added, or -1 if the directory cannot be opened.
121 */
122static int
123pki_load_crls_from_dir(X509_STORE *store, const char *path, ngx_log_t *log)
124{
125 DIR *dir;
126 struct dirent *ent;
127 int total;
128
129 dir = opendir(path);
130 if (dir == NULL) {
131 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
132 "brix_pki: cannot open CRL directory \"%s\"", path);
133 return -1;
134 }
135
136 total = 0;
137 while ((ent = readdir(dir)) != NULL) {
138 if (!pki_crl_name_matches(ent->d_name)) {
139 continue;
140 }
141 total += pki_load_crls_from_dirent(store, path, ent->d_name, log);
142 }
143
144 closedir(dir);
145 return total;
146}

148/*
149 * Load CRLs from a file or directory into the store.
150 * Returns total CRLs added, or -1 on infrastructure failure (stat/opendir).
151 * Directory scan matches *.pem and *.r0-*.r9 (grid CA CRL naming convention).
152 */
153static int
154pki_load_crls(X509_STORE *store, const char *path, ngx_log_t *log)
155{
156 struct stat st;
157
158 if (stat(path, &st) != 0) {
159 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
160 "brix_pki: cannot stat CRL path \"%s\"", path);
161 return -1;
162 }
163
164 if (S_ISREG(st.st_mode)) {
165 return pki_load_crls_from_file(store, path, log);
166 }
167
168 if (!S_ISDIR(st.st_mode)) {
169 ngx_log_error(NGX_LOG_ERR, log, 0,
170 "brix_pki: CRL path \"%s\" is neither a file nor a "
171 "directory", path);
172 return -1;
173 }
174
175 return pki_load_crls_from_dir(store, path, log);
176}

190X509_STORE *
191brix_build_ca_store(ngx_log_t *log,
192 const char *cadir,
193 const char *cafile,
194 const char *crl_path,
195 unsigned long extra_flags,
196 int *crl_count_out,
197 brix_sp_mode_t sp_mode,
198 int crl_mode)
199{
200 X509_STORE *store;
201
202 if (crl_count_out != NULL) {
203 *crl_count_out = 0;
204 }
205
206 store = X509_STORE_new();
207 if (store == NULL) {
208 ngx_log_error(NGX_LOG_ERR, log, 0,
209 "brix_pki: X509_STORE_new() failed");
210 return NULL;
211 }
212
213 if (cadir != NULL) {
214 if (!X509_STORE_load_path(store, cadir)) {
215 ngx_log_error(NGX_LOG_WARN, log, 0,
216 "brix_pki: failed to load CA directory \"%s\"",
217 cadir);
218 }
219 }
220
221 if (cafile != NULL) {
222 if (!X509_STORE_load_file(store, cafile)) {
223 ngx_log_error(NGX_LOG_WARN, log, 0,
224 "brix_pki: failed to load CA file \"%s\"", cafile);
225 }
226 }
227
228 if (cadir == NULL && cafile == NULL) {
229 X509_STORE_set_default_paths(store);
230 }
231
232 {
233 int crl_count = 0;
234
235 if (crl_path != NULL) {
236 crl_count = pki_load_crls(store, crl_path, log);
237 if (crl_count < 0) {
238 X509_STORE_free(store);
239 return NULL;
240 }
241 }
242
243 /*
244 * All flag/callback/signing_policy setup is centralised in
245 * brix_store_configure so the C conformance oracle configures a store
246 * identically to this production path.
247 */
248 if (brix_store_configure(store, cadir, extra_flags, crl_count,
249 sp_mode, crl_mode, log, brix_pki_sp_log) != 0)
250 {
251 X509_STORE_free(store);
252 return NULL;
253 }
254
255 if (crl_count_out != NULL) {
256 *crl_count_out = crl_count;
257 }
258 }
259
260 return store;
261}

292/*
293 * WHAT: probe the cache for an existing store matching @key.
294 * WHY: a hit skips the ~1s IGTF-CRL directory load; extracting the scan drops
295 * the branching (and CCN) out of the public entry point.
296 * HOW: linear scan of the fixed table. On a hit: up_ref the store (caller owns
297 * the returned ref exactly as in the miss path), publish crl_count, log,
298 * and return it. On a miss: record the first free slot via @free_slot_out
299 * and return NULL.
300 */
301static X509_STORE *
302brix_ca_store_cache_get(const char *key, ngx_log_t *log, const char *cadir,
303 const char *cafile, int *crl_count_out, int *free_slot_out)
304{
305 int i;
306
307 *free_slot_out = -1;
308
309 for (i = 0; i < BRIX_CA_STORE_CACHE_MAX; i++) {
310 brix_ca_store_cache_ent_t *e = &brix_ca_store_cache[i];
311
312 if (e->store != NULL && ngx_strcmp(e->key, key) == 0) {
313 if (crl_count_out != NULL) { *crl_count_out = e->crl_count; }
314 X509_STORE_up_ref(e->store);
315 /* WARN (not NOTICE): this memo is emitted at CONFIG-parse time, and
316 * nginx pins the config-parse log to WARN — a NOTICE here is silently
317 * dropped from `nginx -t` and from the startup diagnostics, so neither
318 * an operator nor a conformance test can confirm the sibling-block
319 * store reuse actually happened. brix already surfaces informational
320 * GSI config "NOTE:"s at WARN for exactly this reason (see
321 * postconfiguration.c). Low frequency (only when >1 block shares a CA). */
322 ngx_log_error(NGX_LOG_WARN, log, 0,
323 "brix_pki: NOTE: reusing the CA/CRL store already built for \"%s\" "
324 "(skipped a redundant CRL directory load)",
325 cadir ? cadir : (cafile ? cafile : "(system)"));
326 return e->store;
327 }
328 if (e->store == NULL && *free_slot_out < 0) { *free_slot_out = i; }
329 }
330
331 return NULL;
332}

334/*
335 * WHAT: install a freshly built store into a free cache slot.
336 * WHY: the cache holds one reference; publishing it here keeps the ref/copy
337 * bookkeeping beside the lookup it mirrors. A full table (slot < 0) is a
338 * no-op — the store is simply not memoised.
339 * HOW: up_ref for the cache's own ref, copy the key, record store + crl_count.
340 */
341static void
342brix_ca_store_cache_put(int slot, const char *key, X509_STORE *store,
343 int crl_count)
344{
345 brix_ca_store_cache_ent_t *e;
346
347 if (slot < 0) {
348 return;
349 }
350
351 e = &brix_ca_store_cache[slot];
352 X509_STORE_up_ref(store); /* the cache holds one ref */
353 ngx_cpystrn((u_char *) e->key, (u_char *) key, sizeof(e->key));
354 e->store = store;
355 e->crl_count = crl_count;
356}

358/* Keyed by INPUTS ONLY (not by scope/cycle) so a store built in the config-load
359 * process is reused by the daemonised master it forks into — collapsing the
360 * duplicate ~1s IGTF-CRL load a dynamic module otherwise pays twice at startup.
361 * `scope != NULL` enables caching; the per-worker CRL hot-reload timer passes
362 * NULL to force a fresh rebuild from the current CRLs on disk. A config change
363 * to the CA/CRL paths yields a different key (rebuilds); a reload that only
364 * rotates CRL *content* under the same dir is refreshed by that timer. */
365X509_STORE *
366brix_build_ca_store_cached(void *scope, ngx_log_t *log,
367 const char *cadir, const char *cafile, const char *crl_path,
368 unsigned long extra_flags, int *crl_count_out,
369 brix_sp_mode_t sp_mode, int crl_mode)
370{
371 char key[768];
372 int free_slot = -1;
373 X509_STORE *store;
374 int crl_count = 0;
375
376 if (scope == NULL) { /* caching disabled (CRL reload) */
377 return brix_build_ca_store(log, cadir, cafile, crl_path, extra_flags,
378 crl_count_out, sp_mode, crl_mode);
379 }
380
381 brix_ca_store_cache_key(key, sizeof(key), cadir, cafile, crl_path,
382 extra_flags, sp_mode, crl_mode);
383
384 store = brix_ca_store_cache_get(key, log, cadir, cafile, crl_count_out,
385 &free_slot);
386 if (store != NULL) {
387 return store;
388 }
389
390 store = brix_build_ca_store(log, cadir, cafile, crl_path, extra_flags,
391 &crl_count, sp_mode, crl_mode);
392 if (store == NULL) {
393 return NULL;
394 }
395 if (crl_count_out != NULL) { *crl_count_out = crl_count; }
396
397 brix_ca_store_cache_put(free_slot, key, store, crl_count);
398 return store;
399}
