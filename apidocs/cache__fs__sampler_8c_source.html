1/*
2* cache_fs_sampler.c — TTL-cached statvfs sampler for the cache fullness signals.
3 *
4 * WHAT: brix_cache_fs_usage_sampled() wraps brix_cache_fs_usage() (a raw
5 * statvfs) with a tiny per-worker, per-root cache so hot callers do not
6 * statvfs on every request.
7 *
8 * WHY: Two consumers need filesystem occupancy: the watermark reaper timer
9 * (cache_root, once per tick — cheap either way) and the write-back
10 * staging admission gate (stage_root, once per write-open — hot). A short
11 * TTL keeps the staging gate's per-open cost near zero while staying fresh
12 * enough to track fill/drain within a tick.
13 *
14 * HOW: A fixed slot table keyed by the root string. On a hit younger than ttl
15 * (brix_cache_sample_fresh) the cached snapshot is returned; otherwise a
16 * fresh statvfs refreshes the slot. Per-worker (event loop + the fill
17 * worker that calls eviction are serialized w.r.t. their own roots in
18 * practice); the table is advisory, so a benign race only costs an extra
19 * statvfs. No allocation.
20 */
21

38/* Find the slot for `root`, or the first empty slot to claim, or NULL when the
39 * table is full and `root` is absent (caller reuses slot 0 as an LRU-ish victim). */
40static fs_usage_slot_t *
41fs_usage_lookup(const char *root, fs_usage_slot_t **free_out)
42{
43 fs_usage_slot_t *empty = NULL;
44 ngx_uint_t i;
45
46 for (i = 0; i < BRIX_FS_USAGE_SLOTS; i++) {
47 if (fs_usage_cache[i].last_ms != 0
48 && strcmp(fs_usage_cache[i].root, root) == 0)
49 {
50 return &fs_usage_cache[i];
51 }
52 if (fs_usage_cache[i].last_ms == 0 && empty == NULL) {
53 empty = &fs_usage_cache[i];
54 }
55 }
56 *free_out = empty;
57 return NULL;
58}

60ngx_int_t
61brix_cache_fs_usage_sampled(const char *root, ngx_msec_t ttl_ms,
62 brix_cache_fs_usage_t *out)
63{
64 uint64_t now = (uint64_t) ngx_current_msec;
65 fs_usage_slot_t *slot;
66 fs_usage_slot_t *empty = NULL;
67 brix_cache_fs_usage_t fresh;
68
69 if (root == NULL || out == NULL) {
70 errno = EINVAL;
71 return NGX_ERROR;
72 }
73
74 slot = fs_usage_lookup(root, &empty);
75 if (slot != NULL
76 && brix_cache_sample_fresh(now, slot->last_ms, (uint64_t) ttl_ms))
77 {
78 *out = slot->usage;
79 return NGX_OK;
80 }
81
82 if (brix_cache_fs_usage(root, &fresh) != NGX_OK) {
83 return NGX_ERROR; /* leave any stale slot in place; caller decides */
84 }
85
86 if (slot == NULL) {
87 slot = (empty != NULL) ? empty : &fs_usage_cache[0]; /* victim when full */
88 ngx_snprintf((u_char *) slot->root, sizeof(slot->root) - 1, "%s%Z", root);
89 }
90 slot->usage = fresh;
91 slot->last_ms = (now == 0) ? 1 : now; /* never store 0 — that means "empty" */
92
93 *out = fresh;
94 return NGX_OK;
95}
