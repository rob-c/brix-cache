20/*
21 * 8192 direct-mapped slots (~128 KiB) is ample: the table only ever holds
22 * principals that are actively *missing* paths, and a throttle tolerates
23 * collisions gracefully. Fixed so the zone size is known without a directive.

52static ngx_int_t
53negcache_init_zone(ngx_shm_zone_t *shm_zone, void *data)
54{
55 negcache_table_t *tbl;
56 ngx_flag_t fresh;
57
58 tbl = brix_shm_table_alloc(shm_zone, data,
59 sizeof(negcache_table_t)
60 + (size_t) (NEGCACHE_SLOTS - 1)
61 * sizeof(brix_negcache_slot_t),
62 &negcache_mtx, &fresh);
63 if (tbl == NULL) {
64 return NGX_ERROR;
65 }
66 if (fresh) {
67 tbl->capacity = NEGCACHE_SLOTS;
68 /* helper already zeroed the table (key == 0 → all slots empty) */
69 }
70 return NGX_OK;
71}

74char *
75brix_negcache_set(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
76{
77 brix_negcache_conf_t *nc = (brix_negcache_conf_t *)
78 ((char *) conf + cmd->offset);
79 ngx_str_t *value = cf->args->elts;
80 ngx_int_t thr, win, back;
81
82 /* "off": explicit disable (threshold 0). */
83 if (cf->args->nelts == 2
84 && value[1].len == 3 && ngx_strncmp(value[1].data, "off", 3) == 0)
85 {
86 nc->threshold = 0;
87 nc->window_ms = 0;
88 nc->backoff_s = 0;
89 return NGX_CONF_OK;
90 }
91
92 if (cf->args->nelts != 4) {
93 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
94 "brix_negcache_backoff: expected \"off\" or"
95 " \"<threshold> <window_seconds> <wait_seconds>\"");
96 return NGX_CONF_ERROR;
97 }
98
99 thr = ngx_atoi(value[1].data, value[1].len);
100 win = ngx_atoi(value[2].data, value[2].len);
101 back = ngx_atoi(value[3].data, value[3].len);
102
103 if (thr == NGX_ERROR || thr < 1
104 || win == NGX_ERROR || win < 1 || win > 86400
105 || back == NGX_ERROR || back < 1 || back > 3600)
106 {
107 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
108 "brix_negcache_backoff: threshold >= 1, window_seconds 1..86400"
109 " and wait_seconds 1..3600 required");
110 return NGX_CONF_ERROR;
111 }
112
113 nc->threshold = (ngx_uint_t) thr;
114 nc->window_ms = (ngx_uint_t) win * 1000;
115 nc->backoff_s = (ngx_uint_t) back;
116 return NGX_CONF_OK;
117}

133 negcache_zone = ngx_shared_memory_add(cf, &zone_name, zone_size,

135 if (negcache_zone == NULL) {

138 negcache_zone->init = negcache_init_zone;
139 negcache_zone->data = (void *) 1;
140 return NGX_OK;

144/*
145 * FNV-1a32 over the principal string, seeded by a dimension tag so a token
146 * subject and a client IP that happen to share bytes never collide. The tag
147 * also keeps the three principal classes in disjoint hash sub-spaces.
148 */
149static uint32_t
150negcache_fnv(char tag, const u_char *s, size_t n)
151{
152 uint32_t h = 2166136261u;
153 size_t i;
154
155 h ^= (uint32_t) (u_char) tag;
156 h *= 16777619u;
157 for (i = 0; i < n; i++) {
158 h ^= (uint32_t) s[i];
159 h *= 16777619u;
160 }
161 return h;
162}

165unsigned
166brix_negcache_note_miss(brix_ctx_t *ctx, unsigned threshold,
167 unsigned window_ms, unsigned backoff_s)
168{
169 negcache_table_t *tbl = negcache_table();
170 brix_negcache_params_t p;
171 uint32_t h;
172 unsigned wait;
173
174 if (tbl == NULL || ctx == NULL) {
175 return 0; /* zone unattached → fail open */
176 }
177
178 /*
179 * Principal, most-specific first: an authenticated token subject, else a
180 * GSI subject DN, else the client IP so an unauthenticated bulk harvester
181 * is always bucketed (mirrors the rate-limiter's invariant-5 IP fallback).
182 */
183 if (ctx->identity != NULL && ctx->identity->subject.len > 0) {
184 h = negcache_fnv('s', ctx->identity->subject.data,
185 ctx->identity->subject.len);
186 } else if (ctx->login.dn[0] != '\0') {
187 h = negcache_fnv('d', (u_char *) ctx->login.dn,
188 ngx_strlen(ctx->login.dn));
189 } else {
190 h = negcache_fnv('p', (u_char *) ctx->login.peer_ip,
191 ngx_strlen(ctx->login.peer_ip));
192 }
193
194 p.threshold = threshold;
195 p.window_ms = window_ms;
196 p.backoff_s = backoff_s;
197
198 ngx_shmtx_lock(&negcache_mtx);
199 wait = brix_negcache_core_note(tbl->slots, (unsigned) tbl->capacity, h,
200 (uint64_t) ngx_current_msec, &p);
201 ngx_shmtx_unlock(&negcache_mtx);
202 return wait;
203}
