1/*
2 * xfer_ledger.c — the unified durable-transfer audit ledger.
3 *
4 * WHAT: brix_xfer_ledger_record() appends ONE consistent audit line for every
5 * terminal transfer (commit or abort), across all four kinds (stage, tape,
6 * write-through, TPC). One schema, one sink.
7 *
8 * WHY: The four paths each had their own audit surface and only one emitted an
9 * access-log line, so "what objects got published, by whom, by which path"
10 * meant reading four places. Metrics keep their existing per-subsystem
11 * names (no dashboard breakage — each caller still books its own); the
12 * ledger adds the single audit line that unifies the picture. See
13 * docs/superpowers/specs/2026-06-28-unified-durable-transfer-engine-design.md
14 * §6.
15 *
16 * HOW: A process-global append-only fd, lazy-opened on first record. O_APPEND
17 * makes concurrent writes from multiple workers atomic for the sub-PIPE_BUF
18 * lines we emit, so no lock is needed. The sink is $BRIX_XFER_AUDIT_LOG
19 * (explicit — never falls back), else <prefix>/logs/xfer_audit.log, else
20 * xfer_audit.log beside the cycle's error log (packaged deployments run
21 * with a prefix like /usr/share/nginx that has no logs/ directory, but the
22 * error-log directory always exists and is writable). Wire-sourced fields
23 * (path, principal) are escaped with the shared sanitizer. Auditing is
24 * best-effort: a sink that cannot be opened is warned once and skipped —
25 * it never fails a transfer. Early-return, no goto.
26 */

72/* Resolve the default audit-log path: <prefix>/logs/xfer_audit.log. */
73static void
74xfer_audit_resolve_path(char *out, size_t outsz)
75{
76 if (ngx_cycle != NULL && ngx_cycle->prefix.len > 0) {
77 ngx_snprintf((u_char *) out, outsz, "%V%s%Z", &ngx_cycle->prefix,
78 "logs/xfer_audit.log");
79 return;
80 }
81 ngx_cpystrn((u_char *) out, (u_char *) "logs/xfer_audit.log", outsz);
82}

84/* Build "xfer_audit.log" in the directory of the cycle's error log. Returns 1
85 * with *out set, 0 when the cycle logs to stderr (no file to sit beside) or the
86 * path would not fit. */
87static int
88xfer_audit_errorlog_sibling(char *out, size_t outsz)
89{
90 static const char leaf[] = "xfer_audit.log";
91 ngx_str_t *name;
92 u_char *slash;
93 size_t dirlen;
94
95 if (ngx_cycle == NULL || ngx_cycle->new_log.file == NULL) {
96 return 0;
97 }
98 name = &ngx_cycle->new_log.file->name;
99 if (name->len == 0 || name->data == NULL) {
100 return 0;
101 }
102
103 slash = NULL;
104 for (size_t i = 0; i < name->len; i++) {
105 if (name->data[i] == '/') {
106 slash = name->data + i;
107 }
108 }
109 if (slash == NULL) {
110 return 0;
111 }
112
113 dirlen = (size_t) (slash - name->data) + 1; /* keep the '/' */
114 if (dirlen + sizeof(leaf) > outsz) {
115 return 0;
116 }
117 ngx_memcpy(out, name->data, dirlen);
118 ngx_memcpy(out + dirlen, leaf, sizeof(leaf));
119 return 1;
120}

125 return open(path, O_WRONLY | O_CREAT | O_APPEND | O_CLOEXEC, 0640);

128/* Lazily open the append-only audit fd. $BRIX_XFER_AUDIT_LOG is an explicit
129 * operator choice and never falls back; the built-in default (<prefix>/logs/)
130 * falls back to the error-log directory so packaged deployments whose prefix
131 * has no logs/ directory still get auditing. -1 (one-shot warn) on failure. */
132static int
133xfer_audit_fd_get(ngx_log_t *log)
134{
135 char path[PATH_MAX];
136 char alt[PATH_MAX];
137 const char *env = getenv("BRIX_XFER_AUDIT_LOG");
138
139 if (xfer_audit_fd >= 0) {
140 return xfer_audit_fd;
141 }
142 if (xfer_audit_failed) {
143 return -1;
144 }
145
146 if (env != NULL && env[0] != '\0') {
147 ngx_cpystrn((u_char *) path, (u_char *) env, sizeof(path));
148 xfer_audit_fd = xfer_audit_try_open(path);
149 if (xfer_audit_fd < 0) {
150 xfer_audit_failed = 1;
151 ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
152 "xfer: cannot open audit log \"%s\" "
153 "($BRIX_XFER_AUDIT_LOG) — transfer auditing "
154 "disabled this worker", path);
155 }
156 return xfer_audit_fd;
157 }
158
159 xfer_audit_resolve_path(path, sizeof(path));
160 xfer_audit_fd = xfer_audit_try_open(path);
161 if (xfer_audit_fd >= 0) {
162 return xfer_audit_fd;
163 }
164
165 if (xfer_audit_errorlog_sibling(alt, sizeof(alt))) {
166 xfer_audit_fd = xfer_audit_try_open(alt);
167 if (xfer_audit_fd >= 0) {
168 ngx_log_error(NGX_LOG_NOTICE, log, 0,
169 "xfer: audit log at \"%s\" (default \"%s\" "
170 "unavailable)", alt, path);
171 return xfer_audit_fd;
172 }
173 }
174
175 xfer_audit_failed = 1;
176 ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
177 "xfer: cannot open audit log \"%s\" — transfer auditing "
178 "disabled this worker (set $BRIX_XFER_AUDIT_LOG to choose "
179 "a writable path)", path);
180 return -1;
181}

185void
186brix_xfer_ledger_record(const brix_xfer_audit_t *ev)
187{
188 char line[2048];
189 char timebuf[64];
190 char safe_path[1024];
191 char safe_principal[256];
192 struct tm tm;
193 ngx_time_t *tp;
194 int fd;
195 int n;
196
197 if (ev == NULL) {
198 return;
199 }
200
201 fd = xfer_audit_fd_get(ev->log);
202 if (fd < 0) {
203 return;
204 }
205
206 tp = ngx_timeofday();
207 ngx_libc_localtime(tp->sec, &tm);
208 if (strftime(timebuf, sizeof(timebuf), "%d/%b/%Y:%H:%M:%S %z", &tm) == 0) {
209 timebuf[0] = '-'; /* on failure timebuf is indeterminate */
210 timebuf[1] = '\0';
211 }
212
213 brix_sanitize_log_string(ev->path ? ev->path : "-", safe_path,
214 sizeof(safe_path));
215 brix_sanitize_log_string((ev->principal && ev->principal[0]) ? ev->principal
216 : "-",
217 safe_principal, sizeof(safe_principal));
218
219 n = snprintf(line, sizeof(line),
220 "%s kind=%s dir=%s result=%s bytes=%zu errno=%d "
221 "principal=%s path=\"%s\"\n",
222 timebuf, brix_xfer_kind_str(ev->kind),
223 ev->direction ? ev->direction : "-",
224 brix_xfer_result_str(ev->result), ev->bytes,
225 (ev->result == BRIX_XFER_OK) ? 0 : ev->sys_errno,
226 safe_principal, safe_path);
227 if (n <= 0) {
228 return;
229 }
230 if ((size_t) n > sizeof(line)) {
231 n = (int) sizeof(line);
232 }
233
234 /* O_APPEND: atomic for sub-PIPE_BUF lines, so safe across workers unlocked.
235 * Best-effort — a short/failed write loses one audit line, never a transfer. */
236 if (write(fd, line, (size_t) n) < 0) {
237 /* nothing we can do for an audit line; do not disturb the transfer */
238 }
239}
