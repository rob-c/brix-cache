1/*
2 * xfer_ledger.c — the unified durable-transfer audit ledger.
3 *
4 * WHAT: brix_xfer_ledger_record() appends ONE consistent audit line for every
5 * terminal transfer (commit or abort), across all four kinds (stage, tape,
6 * write-through, TPC). One schema, one sink.
7 *
8 * WHY: The four paths each had their own audit surface and only one emitted an
9 * access-log line, so "what objects got published, by whom, by which path"
10 * meant reading four places. Metrics keep their existing per-subsystem
11 * names (no dashboard breakage — each caller still books its own); the
12 * ledger adds the single audit line that unifies the picture. See
13 * docs/superpowers/specs/2026-06-28-unified-durable-transfer-engine-design.md
14 * §6.
15 *
16 * HOW: A process-global append-only fd, lazy-opened on first record. O_APPEND
17 * makes concurrent writes from multiple workers atomic for the sub-PIPE_BUF
18 * lines we emit, so no lock is needed. The sink is $BRIX_XFER_AUDIT_LOG,
19 * else <prefix>/logs/xfer_audit.log. Wire-sourced fields (path, principal)
20 * are escaped with the shared sanitizer. Auditing is best-effort: a sink
21 * that cannot be opened is warned once and skipped — it never fails a
22 * transfer. Early-return, no goto.
23 */

69/* Resolve the audit-log path: $BRIX_XFER_AUDIT_LOG, else <prefix>/logs/. */
70static void
71xfer_audit_resolve_path(char *out, size_t outsz)
72{
73 const char *env = getenv("BRIX_XFER_AUDIT_LOG");
74
75 if (env != NULL && env[0] != '\0') {
76 ngx_cpystrn((u_char *) out, (u_char *) env, outsz);
77 return;
78 }
79 if (ngx_cycle != NULL && ngx_cycle->prefix.len > 0) {
80 ngx_snprintf((u_char *) out, outsz, "%V%s%Z", &ngx_cycle->prefix,
81 "logs/xfer_audit.log");
82 return;
83 }
84 ngx_cpystrn((u_char *) out, (u_char *) "logs/xfer_audit.log", outsz);
85}

87/* Lazily open the append-only audit fd. -1 (and a one-shot warn) on failure. */
88static int
89xfer_audit_fd_get(ngx_log_t *log)
90{
91 char path[PATH_MAX];
92
93 if (xfer_audit_fd >= 0) {
94 return xfer_audit_fd;
95 }
96 if (xfer_audit_failed) {
97 return -1;
98 }
99
100 xfer_audit_resolve_path(path, sizeof(path));
101 xfer_audit_fd = open(path, O_WRONLY | O_CREAT | O_APPEND | O_CLOEXEC, 0640);
102 if (xfer_audit_fd < 0) {
103 xfer_audit_failed = 1;
104 ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
105 "xfer: cannot open audit log \"%s\" — transfer auditing "
106 "disabled this worker", path);
107 return -1;
108 }
109 return xfer_audit_fd;
110}

114void
115brix_xfer_ledger_record(const brix_xfer_audit_t *ev)
116{
117 char line[2048];
118 char timebuf[64];
119 char safe_path[1024];
120 char safe_principal[256];
121 struct tm tm;
122 ngx_time_t *tp;
123 int fd;
124 int n;
125
126 if (ev == NULL) {
127 return;
128 }
129
130 fd = xfer_audit_fd_get(ev->log);
131 if (fd < 0) {
132 return;
133 }
134
135 tp = ngx_timeofday();
136 ngx_libc_localtime(tp->sec, &tm);
137 strftime(timebuf, sizeof(timebuf), "%d/%b/%Y:%H:%M:%S %z", &tm);
138
139 brix_sanitize_log_string(ev->path ? ev->path : "-", safe_path,
140 sizeof(safe_path));
141 brix_sanitize_log_string((ev->principal && ev->principal[0]) ? ev->principal
142 : "-",
143 safe_principal, sizeof(safe_principal));
144
145 n = snprintf(line, sizeof(line),
146 "%s kind=%s dir=%s result=%s bytes=%zu errno=%d "
147 "principal=%s path=\"%s\"\n",
148 timebuf, brix_xfer_kind_str(ev->kind),
149 ev->direction ? ev->direction : "-",
150 brix_xfer_result_str(ev->result), ev->bytes,
151 (ev->result == BRIX_XFER_OK) ? 0 : ev->sys_errno,
152 safe_principal, safe_path);
153 if (n <= 0) {
154 return;
155 }
156 if ((size_t) n > sizeof(line)) {
157 n = (int) sizeof(line);
158 }
159
160 /* O_APPEND: atomic for sub-PIPE_BUF lines, so safe across workers unlocked.
161 * Best-effort — a short/failed write loses one audit line, never a transfer. */
162 if (write(fd, line, (size_t) n) < 0) {
163 /* nothing we can do for an audit line; do not disturb the transfer */
164 }
165}
