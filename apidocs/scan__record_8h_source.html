1/*
2 * scan_record.h — NDJSON record formatting for the storage-scan engine.
3 *
4 * WHAT: pure, ngx-free formatters that render one engine result as a single
5 * NDJSON line (file / cursor / summary). Strings are JSON-escaped.
6 * WHY: the engine streams one object per line to an admin; isolating the
7 * formatting keeps it allocation-free, protocol-agnostic, and unit-testable
8 * standalone (no nginx, no server) — see scan_unittest.c.
9 * HOW: each formatter writes a NUL-terminated line (no trailing newline; the
10 * caller appends '\n' when framing) into a caller buffer and returns the
11 * length written, or -1 if the buffer was too small (never truncates).
12 */
