2 * lock_discovery.c — WebDAV UNLOCK, startup lock sweep, and PROPFIND
3 * lock-discovery/supportedlock XML (RFC 4918 §9.11, §10.1, §15.8/§15.10).

5 * Split from lock.c (mechanical, zero behaviour change): the read/release side

38/*
39 * Startup lock sweep — remove every persisted lock xattr under the export
40 * root. Locks are stored as the WEBDAV_LOCK_XATTR_KEY xattr on each resource
41 * (prop_xattr.c), so they normally survive an nginx restart. When
42 * brix_webdav_lock_startup_sweep is on, this walk restores ephemeral
43 * RFC 4918 §10.1 semantics by clearing them once at startup.
44 *
45 * removexattr is idempotent: a node with no lock returns ENODATA/ENOATTR,
46 * which is not counted and not an error. The walk continues past per-node
47 * failures so one unreadable subtree cannot abort the sweep.
48 */
49/* Sweep state threaded to the walk callback so each node can build a transient,
50 * confined VFS ctx for its lock-xattr removal (the sweep runs at config-merge
51 * time, so it carries its own pool/log/root_canon rather than a request ctx). */
52typedef struct {
53 ngx_uint_t removed;
54 ngx_pool_t *pool;
55 ngx_log_t *log;
56 const char *root_canon;
57} webdav_lock_sweep_ctx_t;
58
59/* Remove the lock xattr on one absolute path through the VFS seam. removexattr
60 * via the VFS is idempotent for our purposes: ENODATA/ENOATTR (no lock) returns
61 * NGX_ERROR but is simply not counted, so a node without a lock is a no-op. */
62static void
63webdav_lock_sweep_remove(webdav_lock_sweep_ctx_t *sw, const char *path)
64{
65 brix_vfs_ctx_t vctx;
66
67 brix_vfs_ctx_init(&vctx, sw->pool, sw->log, BRIX_PROTO_WEBDAV,
68 sw->root_canon, NULL, 1 /* allow_write */, 0 /* is_tls */, NULL, path);
69
70 if (brix_vfs_removexattr(&vctx, WEBDAV_LOCK_XATTR_KEY) == NGX_OK) {
71 sw->removed++;
72 }
73}
74
75static ngx_int_t
76webdav_lock_sweep_cb(const brix_fs_walk_entry_t *entry, void *data)
77{
78 webdav_lock_sweep_remove(data, entry->path);
79 return NGX_OK; /* continue regardless of per-node result */
80}

82ngx_uint_t
83webdav_lock_startup_sweep(ngx_pool_t *pool, ngx_log_t *log,
84 const char *root_canon)
85{
86 brix_fs_walk_options_t opts;
87 webdav_lock_sweep_ctx_t sw;
88
89 if (root_canon == NULL || root_canon[0] == '\0') {
90 return 0;
91 }
92
93 ngx_memzero(&sw, sizeof(sw));
94 sw.pool = pool;
95 sw.log = log;
96 sw.root_canon = root_canon;
97
98 /* The tree walk visits children only; clear a lock on the root itself. */
99 webdav_lock_sweep_remove(&sw, root_canon);
100
101 ngx_memzero(&opts, sizeof(opts));
102 opts.include_files = 1; /* locks may sit on files ... */
103 opts.include_dirs = 1; /* ... and on collections (Depth: infinity) */
104
105 (void) brix_fs_walk(log, root_canon, &opts, webdav_lock_sweep_cb, &sw);
106 return sw.removed;
107}

109ngx_int_t
110webdav_handle_unlock(ngx_http_request_t *r)
111{
112 ngx_http_brix_webdav_loc_conf_t *conf;
113 char path[WEBDAV_MAX_PATH];
114 ngx_int_t rc;
115 ngx_table_elt_t *h;
116 webdav_lock_xattr_t e;
117 const u_char *lock_val;
118 size_t lock_len, token_len;
119
120 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
121
122 h = webdav_tpc_find_header(r, "Lock-Token", sizeof("Lock-Token") - 1);
123 if (h == NULL) {
124 return NGX_HTTP_BAD_REQUEST;
125 }
126
127 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
128 path, sizeof(path));
129 if (rc != NGX_OK) {
130 return rc;
131 }
132
133 rc = webdav_lock_xattr_read(r, path, &e);
134 if (rc == NGX_ERROR) {
135 return NGX_HTTP_INTERNAL_SERVER_ERROR;
136 }
137 if (rc == NGX_DECLINED || e.expires <= (int64_t) ngx_time()) {
138 /* No active lock on this path. */
139 if (rc == NGX_OK) {
140 (void) webdav_lock_xattr_delete(r, path);
141 webdav_lock_reap_null(r, path, &e);
142 }
143 return NGX_HTTP_CONFLICT;
144 }
145
146 /* Strip angle brackets from Lock-Token header: <opaquelocktoken:...> → token */
147 lock_val = h->value.data;
148 lock_len = (size_t) h->value.len;
149 if (lock_len > 0 && lock_val[0] == '<') {
150 lock_val++;
151 lock_len--;
152 if (lock_len > 0 && lock_val[lock_len - 1] == '>') {
153 lock_len--;
154 }
155 }
156
157 token_len = strlen(e.token);
158 if (lock_len != token_len
159 || CRYPTO_memcmp(lock_val, (u_char *) e.token, token_len) != 0)
160 {
161 return NGX_HTTP_CONFLICT; /* token mismatch — not our lock */
162 }
163
164 /*
165 * Remove the lock xattr and HONOUR the result. Under impersonation the
166 * removexattr runs (via the broker) as the requester's mapped user, so a
167 * NON-OWNER who presents a valid (e.g. stolen/leaked) lock token cannot
168 * actually clear the lock on another user's file — the broker returns EACCES.
169 * Returning 204 unconditionally would falsely report success for that denied
170 * removal; surface it as 403 instead so the lock state and the response agree.
171 * (ENODATA/ENOENT collapse to NGX_OK inside the helper — idempotent unlock.)
172 */
173 if (webdav_lock_xattr_delete(r, path) != NGX_OK) {
174 return NGX_HTTP_FORBIDDEN;
175 }
176
177 /* RFC 4918 §9.10.1: a lock-null resource that was never converted by a PUT/
178 * MKCOL disappears when its lock is removed. Reap the empty placeholder. */
179 webdav_lock_reap_null(r, path, &e);
180
181 return webdav_send_no_body(r, NGX_HTTP_NO_CONTENT);
182}

210/*
211 * webdav_lock_append_discovery — emit <D:lockdiscovery> XML for PROPFIND.
212 * Reads the xattr on path; emits an empty element if no active lock.
213 */
214ngx_int_t
215webdav_lock_append_discovery(ngx_http_request_t *r, const char *path,
216 ngx_chain_t **head, ngx_chain_t **tail)
217{
218 webdav_lock_xattr_t e;
219 ngx_int_t rc;
220 int64_t now;
221 ngx_uint_t remaining;
222 char *safe_owner;
223
224 if (brix_http_chain_appendf(r->pool, head, tail,
225 "<D:lockdiscovery>") == NULL)
226 {
227 return NGX_ERROR;
228 }
229
230 rc = webdav_lock_xattr_read(r, path, &e);
231 if (rc == NGX_ERROR) {
232 return NGX_ERROR;
233 }
234
235 now = (int64_t) ngx_time();
236
237 if (rc == NGX_OK && e.expires > now) {
238 remaining = (ngx_uint_t) (e.expires - now);
239 safe_owner = webdav_escape_xml_text(r->pool,
240 (e.owner[0] != '\0') ? e.owner : "anonymous");
241 if (safe_owner == NULL) {
242 safe_owner = "anonymous";
243 }
244
245 if (brix_http_chain_appendf(r->pool, head, tail,
246 "<D:activelock>"
247 "<D:locktype><D:write/></D:locktype>"
248 "<D:lockscope>%s</D:lockscope>"
249 "<D:depth>%s</D:depth>"
250 "<D:owner>%s</D:owner>"
251 "<D:timeout>Second-%lu</D:timeout>"
252 "<D:locktoken>"
253 "<D:href>%s</D:href>"
254 "</D:locktoken>"
255 "</D:activelock>",
256 e.exclusive ? "<D:exclusive/>" : "<D:shared/>",
257 e.depth_infinity ? "infinity" : "0",
258 safe_owner, (unsigned long) remaining, e.token) == NULL)
259 {
260 return NGX_ERROR;
261 }
262 } else if (rc == NGX_OK) {
263 /* Expired lock — clean up lazily (and reap a lock-null placeholder). */
264 (void) webdav_lock_xattr_delete(r, path);
265 webdav_lock_reap_null(r, path, &e);
266 }
267
268 if (brix_http_chain_appendf(r->pool, head, tail,
269 "</D:lockdiscovery>") == NULL)
270 {
271 return NGX_ERROR;
272 }
273
274 return NGX_OK;
275}
