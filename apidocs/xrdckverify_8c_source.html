1/*
2 * xrdckverify.c — verify a file on disk against its recorded checksum.
3 *
4 * WHAT: `xrdckverify [--cache|--storage|--auto] [--algo <name>] [-q] <file>`
5 * recomputes a local file's checksum and compares it to the value already
6 * recorded for that file — in a normal storage endpoint's
7 * "user.XrdCks.<alg>" xattr (or "<file>.cks" sidecar), or a proxy cache's
8 * "<file>.cinfo" / "<file>.meta" record.
9 * WHY: A single offline command to detect at-rest corruption of a cached or
10 * stored file, using the same checksum engine the server runs.
11 * HOW: Thin front-end over brix_cks_verify_file() (lib/cks_verify.c).
12 *
13 * exit: 0 ok · 1 mismatch (corruption) · 2 no recorded checksum / unsupported
14 * algorithm · 3 I/O or access error.
15 */

46/* ---- Handle the pre-parse informational flags (--version / --help) ----
47 *
48 * WHAT: Detects the two positional-first informational flags accepted before
49 * normal option parsing. If argv[1] is "--version" it prints the version
50 * banner; if it is "--help" or "-h" it prints usage to stdout. Returns 1
51 * when such a flag was handled (writing the process exit code into
52 * *exit_code), or 0 when no early flag was present and parsing continues.
53 *
54 * WHY: These two flags short-circuit before the option loop so that a bare
55 * "--version"/"--help" as the first argument never requires a file
56 * operand; keeping them in one helper leaves the orchestrator flat and
57 * drops branch count out of the main function.
58 *
59 * HOW: 1. Require at least one argument after argv[0]; otherwise return 0.
60 * 2. On "--version", print the banner and set exit_code 0, return 1.
61 * 3. On "--help"/"-h", print usage to stdout with exit_code 0, return 1.
62 * 4. Otherwise return 0 so the caller runs the full option loop.
63 */
64static int
65ckv_handle_early_flags(int argc, char **argv, int *exit_code)
66{
67 if (argc < 2) {
68 return 0;
69 }
70 if (strcmp(argv[1], "--version") == 0) {
71 printf("%s (BriX-Cache client) %s\n", argv[0], brix_client_version());
72 *exit_code = 0;
73 return 1;
74 }
75 if (strcmp(argv[1], "--help") == 0 || strcmp(argv[1], "-h") == 0) {
76 *exit_code = usage_fp(stdout, argv[0], 0);
77 return 1;
78 }
79 return 0;
80}

82/* ---- Parse the option/operand vector into the verify request fields ----
83 *
84 * WHAT: Walks argv filling *mode, *algo, *file and *quiet from the recognised
85 * options (--cache/--storage/--auto, --algo <name>, -q/--quiet,
86 * -h/--help) and the single file operand. Returns 1 when the caller must
87 * exit immediately with the code written into *exit_code (help request,
88 * bad option, missing --algo value, extra operand, or missing file), or
89 * 0 when parsing succeeded and *file is set.
90 *
91 * WHY: Isolating the whole option loop keeps every early-exit usage() path and
92 * the loop's branch fan-out out of the main function, so the orchestrator
93 * reads as a short linear sequence.
94 *
95 * HOW: 1. Iterate argv[1..argc); match each mode/quiet/help/algo option.
96 * 2. For --algo, consume the following token or fail with code 2.
97 * 3. Reject unknown "-"-prefixed options and a second file operand.
98 * 4. After the loop, require a file operand; missing → usage code 2.
99 * 5. Return 0 only on a clean parse with *file assigned.
100 */
101static int
102ckv_parse_args(int argc, char **argv, brix_ckv_mode *mode,
103 const char **algo, const char **file, int *quiet,
104 int *exit_code)
105{
106 int i;
107
108 for (i = 1; i < argc; i++) {
109 const char *a = argv[i];
110
111 if (strcmp(a, "--cache") == 0) {
112 *mode = XRDC_CKV_CACHE;
113 } else if (strcmp(a, "--storage") == 0) {
114 *mode = XRDC_CKV_STORAGE;
115 } else if (strcmp(a, "--auto") == 0) {
116 *mode = XRDC_CKV_AUTO;
117 } else if (strcmp(a, "-q") == 0 || strcmp(a, "--quiet") == 0) {
118 *quiet = 1;
119 } else if (strcmp(a, "-h") == 0 || strcmp(a, "--help") == 0) {
120 *exit_code = usage(argv[0], 0);
121 return 1;
122 } else if (strcmp(a, "--algo") == 0) {
123 if (++i >= argc) {
124 *exit_code = usage(argv[0], 2);
125 return 1;
126 }
127 *algo = argv[i];
128 } else if (a[0] == '-') {
129 fprintf(stderr, "%s: unknown option '%s'\n", argv[0], a);
130 *exit_code = usage(argv[0], 2);
131 return 1;
132 } else if (*file == NULL) {
133 *file = a;
134 } else {
135 fprintf(stderr, "%s: only one file may be given\n", argv[0]);
136 *exit_code = usage(argv[0], 2);
137 return 1;
138 }
139 }
140 if (*file == NULL) {
141 *exit_code = usage(argv[0], 2);
142 return 1;
143 }
144 return 0;
145}

147/* ---- Translate a verify result into stdout/stderr output and exit code ----
148 *
149 * WHAT: Renders the outcome of brix_cks_verify_file() and returns the process
150 * exit code: 0 (OK, one line unless quiet), 1 (MISMATCH, always printed),
151 * 2 (no recorded checksum / unsupported algorithm), or 3 (I/O or access
152 * error). Diagnostic text falls back to a fixed message when st has none.
153 *
154 * WHY: Keeping the result-to-output mapping in one pure-at-the-edges helper
155 * preserves the exact exit-code contract in a single reviewable place and
156 * removes the switch's branch weight from the main function.
157 *
158 * HOW: 1. Switch on the verify result code.
159 * 2. OK prints the recorded digest line unless quiet; return 0.
160 * 3. MISMATCH always prints recorded vs computed; return 1.
161 * 4. NO_RECORD/UNSUPPORTED report st.msg or a default; return 2.
162 * 5. Any other result reports st.msg or "error"; return 3.
163 */
164static int
165ckv_report_result(brix_ckv_result r, const char *file,
166 const brix_ckv_report *rep, const brix_status *st, int quiet)
167{
168 switch (r) {
169 case XRDC_CKV_OK:
170 if (!quiet) {
171 printf("OK %s %s %s (%s)\n", file, rep->algo, rep->recorded,
172 rep->source);
173 }
174 return 0;
175 case XRDC_CKV_MISMATCH:
176 printf("MISMATCH %s %s recorded=%s computed=%s (%s)\n",
177 file, rep->algo, rep->recorded, rep->computed, rep->source);
178 return 1;
179 case XRDC_CKV_NO_RECORD:
180 case XRDC_CKV_UNSUPPORTED:
181 fprintf(stderr, "%s: %s\n", file,
182 st->msg[0] ? st->msg : "no recorded checksum");
183 return 2;
184 default:
185 fprintf(stderr, "%s: %s\n", file, st->msg[0] ? st->msg : "error");
186 return 3;
187 }
188}

190/* ---- Real main; dispatched from the xrdcksum multi-call binary ----
191 *
192 * WHAT: Front-end for `xrdckverify`: parses options, runs the shared verify
193 * engine over one file, and returns its exit code (0 ok, 1 mismatch,
194 * 2 no recorded checksum / unsupported, 3 error).
195 *
196 * WHY: A thin orchestrator over brix_cks_verify_file() (lib/cks_verify.c) so
197 * the CLI owns only argument handling and output formatting.
198 *
199 * HOW: 1. Handle the pre-parse --version/--help flags, exiting if matched.
200 * 2. Parse the option/operand vector, exiting on any usage error.
201 * 3. Clear status and run brix_cks_verify_file() on the file.
202 * 4. Map the verify result to output and exit code.
203 */
204int
205brix_xrdckverify_main(int argc, char **argv)
206{
207 brix_ckv_mode mode = XRDC_CKV_AUTO;
208 const char *algo = NULL;
209 const char *file = NULL;
210 int quiet = 0;
211 int exit_code = 0;
212 brix_ckv_report rep;
213 brix_ckv_result r;
214 brix_status st;
215
216 if (ckv_handle_early_flags(argc, argv, &exit_code)) {
217 return exit_code;
218 }
219
220 if (ckv_parse_args(argc, argv, &mode, &algo, &file, &quiet, &exit_code)) {
221 return exit_code;
222 }
223
224 brix_status_clear(&st);
225 r = brix_cks_verify_file(file, algo, mode, &rep, &st);
226
227 return ckv_report_result(r, file, &rep, &st, quiet);
228}
