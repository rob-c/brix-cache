1/*
2 * delegation.c - authenticated proxy delegation endpoints: the simple
3 * proxy-UPLOAD form (phase-2 T8) plus the shared cert-chain / storage
4 * helpers used by every delegation entry point.
5 *
6 * This translation unit holds the Phase-2 T8 upload handler
7 * (webdav_delegation_handle) and the cross-file helpers declared in
8 * delegation_internal.h (chain parse/expiry/DN/trust checks, the atomic
9 * credential store, and the reject choke point). The Phase-3 T4 two-step
10 * GridSite flow lives in delegation_gridsite_req.c (getProxyReq) and
11 * delegation_gridsite_put.c (putProxy), and its per-worker pending-
12 * delegation store in delegation_store.c — see those files for the
13 * two-step WHAT/WHY/HOW.
14 *
15 * ============================ Phase-2 T8: upload ==========================
16 * WHAT: PUT/POST /.well-known/brix-delegation, body = the client's own
17 * RFC-3820 x.509 proxy chain (PEM). Requires GSI-cert-authenticated
18 * TLS (the request must already carry a verified client-certificate
19 * identity — see auth_cert.c), validates the uploaded proxy, and
20 * atomically stores it at <storage_credential_dir>/<key>.pem so
21 * Phase-1 credential selection (brix_sd_ucred_select) picks it up
22 * for that user's subsequent davs/S3 origin sessions.
23 *
24 * WHY: Phase-1 per-user backend credentials require a proxy to already
25 * exist under storage_credential_dir before a user's first origin
26 * request. Operationally that means out-of-band provisioning. This
27 * endpoint lets an already-authenticated client "capture" its own
28 * delegation over the wire (proxy-UPLOAD, not the full GridSite
29 * getProxyReq/putProxy two-step) so no pre-provisioning is required.
30 *
31 * HOW: 1. Gate on brix_delegation_endpoint + the well-known path.
32 * 2. Require ctx->verified && ctx->auth_source == "cert"/"nginx"/
33 * "manual" (GSI/client-cert path — never a bearer token) and a
34 * non-empty ctx->dn.
35 * 3. Read the whole body via brix_http_body_read_all (bounded, small).
36 * 4. Parse the PEM chain (PEM_read_bio_X509 loop, mirrors the pattern
37 * in src/auth/gsi/parse_x509.c); find the end-entity cert (the one
38 * brix_px_classify() reports as BRIX_PX_NONE — proxies in the
39 * uploaded chain are excluded by construction).
40 * 5. Check every cert in the chain (proxy and EEC) has a notAfter
41 * still in the future, and the EEC's subject DN (brix_x509_oneline,
42 * same normalisation as ctx->dn) equals the authenticated client's
43 * DN — a client may only upload a proxy for its own identity.
44 * 6. Derive the credential key (brix_sd_ucred_key) and atomically
45 * write the whole uploaded PEM to <cred_dir>/<key>.pem: open a
46 * temp file in the same directory (O_CREAT|O_EXCL|O_WRONLY, 0600),
47 * write, fsync, close, then rename() over the final path. This is
48 * a raw filesystem write to a service-owned config directory (not
49 * an export), so it is marked vfs-seam-allow rather than routed
50 * through the VFS, which confines to the export root only.
51 */

90/*
91 * delegation_find_eec - scan a parsed proxy chain for the end-entity
92 * certificate (the one cert that is NOT itself a proxy per
93 * brix_px_classify()). Returns a borrowed reference (still owned by
94 * `chain`) or NULL if every cert in the chain is a proxy (malformed
95 * upload — no EEC present).
96 */
97X509 *
98delegation_find_eec(STACK_OF(X509) *chain)
99{
100 int i, n;
101
102 n = sk_X509_num(chain);
103 for (i = 0; i < n; i++) {
104 X509 *cert = sk_X509_value(chain, i);
105
106 if (brix_px_classify(cert) == BRIX_PX_NONE) {
107 return cert;
108 }
109 }
110 return NULL;
111}

113/*
114 * delegation_parse_chain - parse a PEM-concatenated cert chain from a
115 * NUL-terminated buffer. Mirrors the BIO+loop pattern in
116 * src/auth/gsi/parse_x509.c (gsi_chain_from_plaintext). Returns a
117 * non-empty STACK_OF(X509) (caller must sk_X509_pop_free(chain, X509_free))
118 * or NULL on any parse failure.
119 */
120STACK_OF(X509) *
121delegation_parse_chain(const u_char *pem, size_t pem_len)
122{
123 BIO *bio;
124 STACK_OF(X509) *chain;
125 X509 *cert;
126
127 bio = BIO_new_mem_buf(pem, (int) pem_len);
128 if (bio == NULL) {
129 return NULL;
130 }
131
132 chain = sk_X509_new_null();
133 if (chain == NULL) {
134 BIO_free(bio);
135 return NULL;
136 }
137
138 while ((cert = PEM_read_bio_X509(bio, NULL, NULL, NULL)) != NULL) {
139 sk_X509_push(chain, cert);
140 }
141 BIO_free(bio);
142
143 if (sk_X509_num(chain) == 0) {
144 sk_X509_pop_free(chain, X509_free);
145 return NULL;
146 }
147 return chain;
148}

150/*
151 * delegation_chain_expired - true iff ANY certificate in the uploaded chain
152 * (proxy or EEC) has a notAfter that is not in the future. A delegation is
153 * only as strong as its weakest link: an expired proxy cert grants no
154 * authority even if the EEC underneath it is still valid, so the whole
155 * chain must check out.
156 */
157int
158delegation_chain_expired(STACK_OF(X509) *chain)
159{
160 int i, n = sk_X509_num(chain);
161
162 for (i = 0; i < n; i++) {
163 if (X509_cmp_current_time(X509_get0_notAfter(sk_X509_value(chain, i)))
164 <= 0)
165 {
166 return 1;
167 }
168 }
169 return 0;
170}

186/*
187 * delegation_chain_trusted - cryptographically verify that a delegated proxy
188 * chain actually chains to a CA in the frontend's trust store.
189 *
190 * WHAT: run the same PKIX + RFC-3820 proxy verification the WebDAV
191 * client-certificate auth path uses (brix_gsi_verify_chain against
192 * conf->ca_store), treating the topmost proxy (chain[0]) as the leaf
193 * and the whole parsed chain as the untrusted intermediate set.
194 *
195 * WHY: the DN-equality check (delegation_eec_dn_matches) alone compares the
196 * EEC's *self-asserted* subject string against the authenticated client
197 * DN — it proves nothing about issuance. A credential-writing endpoint
198 * must not store a proxy it cannot anchor to a trusted CA, otherwise the
199 * authenticated owner could plant a self-signed cert whose subject string
200 * is spoofed to equal their own DN. Running full chain verification
201 * first turns the subsequent DN string compare into an identity check on
202 * a cryptographically-anchored EEC rather than an attacker-controlled
203 * string. Mirrors auth_cert.c: a NULL ca_store means we cannot verify,
204 * so we refuse (caller maps to 403) rather than trust blindly.
205 *
206 * HOW: NULL store -> NGX_ERROR (refuse, logged). Otherwise leaf = chain[0]
207 * (assemble/upload order is proxy-first), untrusted = the full chain
208 * (OpenSSL tolerates the leaf appearing in the untrusted stack), and
209 * client_purpose = 0 so RFC-3820 proxy chains are accepted (the davs://
210 * GSI mode, not the TLS-leaf mode). Returns NGX_OK iff the chain
211 * verifies to the trust store; brix_gsi_verify_chain has already logged
212 * the specific reason on failure. Owns nothing.
213 */
214ngx_int_t
215delegation_chain_trusted(ngx_http_request_t *r,
216 const ngx_http_brix_webdav_loc_conf_t *conf, STACK_OF(X509) *chain)
217{
218 brix_gsi_verify_result_t verify_res;
219 X509 *leaf;
220
221 if (conf->ca_store == NULL) {
222 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
223 "brix_delegation: no trusted CA store configured — cannot verify"
224 " delegated proxy; refusing to store an unverifiable credential"
225 " (set brix_webdav_auth required with a CA on this location)");
226 return NGX_ERROR;
227 }
228
229 leaf = sk_X509_value(chain, 0);
230 if (leaf == NULL) {
231 return NGX_ERROR;
232 }
233
234 if (brix_gsi_verify_chain(r->connection->log, conf->ca_store, leaf, chain,
235 conf->verify_depth, &verify_res, 0 /* GSI: accept RFC-3820 proxy */)
236 != NGX_OK)
237 {
238 /* brix_gsi_verify_chain already logged the specific error. */
239 return NGX_ERROR;
240 }
241
242 return NGX_OK;
243}

245/*
246 * delegation_store_pem - atomically write `pem`/`pem_len` to
247 * <dir>/<key>.pem: temp file in the same directory (O_CREAT|O_EXCL|
248 * O_WRONLY, 0600), write, fsync, close, rename() over the final path.
249 * vfs-seam-allow: storage_credential_dir is service-owned server config,
250 * not export storage — the VFS confines to the export root only, the
251 * wrong root for this write.
252 *
253 * Returns NGX_OK on success, NGX_ERROR on any I/O failure (temp file is
254 * best-effort unlinked on the way out; caller maps failure to 507).
255 */
256ngx_int_t
257delegation_store_pem(ngx_log_t *log, const ngx_str_t *dir, const char *key,
258 const u_char *pem, size_t pem_len)
259{
260 char final_path[1024];
261 char tmp_path[1024];
262 int fd;
263 ssize_t written;
264
265 if (dir->len == 0
266 || dir->len + 1 + strlen(key) + sizeof(".pem") >= sizeof(final_path)
267 || dir->len + 1 + strlen(key) + sizeof(".pem.XXXXXX") >= sizeof(tmp_path))
268 {
269 ngx_log_error(NGX_LOG_ERR, log, 0,
270 "brix_delegation: credential dir/key too long");
271 return NGX_ERROR;
272 }
273
274 (void) ngx_snprintf((u_char *) final_path, sizeof(final_path) - 1,
275 "%V/%s.pem%Z", dir, key);
276 (void) snprintf(tmp_path, sizeof(tmp_path), "%.*s/.%s.pem.upload.%ld",
277 (int) dir->len, dir->data, key, (long) getpid());
278
279 fd = open(tmp_path, /* vfs-seam-allow: cred dir is svc-owned config, not an export */
280 O_WRONLY | O_CREAT | O_EXCL | O_NOFOLLOW, 0600);
281 if (fd < 0 && ngx_errno == NGX_ENOENT) {
282 /* Store dir vanished (tmpfs default is wiped by a reboot without a
283 * config reload, or an admin rm'd it) — recreate it 0700 as the
284 * worker uid and retry once before shouting. */
285 if (mkdir((const char *) dir->data, 0700) == 0 /* vfs-seam-allow: cred dir is svc-owned config, not an export */
286 || ngx_errno == NGX_EEXIST)
287 {
288 fd = open(tmp_path, /* vfs-seam-allow: cred dir is svc-owned config, not an export */
289 O_WRONLY | O_CREAT | O_EXCL | O_NOFOLLOW, 0600);
290 }
291 }
292 if (fd < 0) {
293 ngx_log_error(NGX_LOG_ALERT, log, ngx_errno,
294 "brix_delegation: open(\"%s\") temp cred file failed — "
295 "credential store \"%V\" is missing or not writable; "
296 "delegation will not work until "
297 "brix_storage_credential_dir is fixed",
298 tmp_path, dir);
299 return NGX_ERROR;
300 }
301
302 written = write(fd, pem, pem_len);
303 if (written < 0 || (size_t) written != pem_len || fsync(fd) != 0) {
304 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
305 "brix_delegation: write/fsync(\"%s\") failed", tmp_path);
306 close(fd);
307 unlink(tmp_path); /* vfs-seam-allow: cred dir is svc-owned config, not an export */
308 return NGX_ERROR;
309 }
310 close(fd);
311
312 if (rename(tmp_path, final_path) != 0) { /* vfs-seam-allow: cred dir is svc-owned config, not an export */
313 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
314 "brix_delegation: rename(\"%s\" -> \"%s\") failed",
315 tmp_path, final_path);
316 unlink(tmp_path); /* vfs-seam-allow: cred dir is svc-owned config, not an export */
317 return NGX_ERROR;
318 }
319 return NGX_OK;
320}

322/*
323 * delegation_reject - send a small plain-text error body and finalize the
324 * request. Single choke point so every failure branch below is one line.
325 */
326void
327delegation_reject(ngx_http_request_t *r, ngx_uint_t status, const char *msg)
328{
329 ngx_buf_t *b;
330 ngx_chain_t out;
331 size_t len = strlen(msg);
332 u_char *buf = ngx_pnalloc(r->pool, len);
333
334 if (buf == NULL) {
335 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
336 return;
337 }
338 ngx_memcpy(buf, msg, len);
339
340 b = ngx_pcalloc(r->pool, sizeof(*b));
341 if (b == NULL) {
342 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
343 return;
344 }
345 b->pos = buf;
346 b->last = buf + len;
347 b->memory = 1;
348 b->last_buf = 1;
349 out.buf = b;
350 out.next = NULL;
351
352 r->headers_out.status = status;
353 r->headers_out.content_length_n = (off_t) len;
354 r->headers_out.content_type.data = (u_char *) "text/plain";
355 r->headers_out.content_type.len = sizeof("text/plain") - 1;
356 r->headers_out.content_type_len = r->headers_out.content_type.len;
357
358 if (ngx_http_send_header(r) == NGX_ERROR || r->header_only) {
359 webdav_metrics_finalize_request(r, status);
360 return;
361 }
362 webdav_metrics_finalize_request(r, ngx_http_output_filter(r, &out));
363}

365/*
366 * delegation_validate_and_store - body is in hand; parse/validate/store.
367 * Split out of webdav_delegation_handle so each early-return failure path
368 * stays a single statement in the caller (no goto, no nested cleanup
369 * ladders — chain ownership lives entirely in this function).
370 */
371static void
372delegation_validate_and_store(ngx_http_request_t *r,
373 ngx_http_brix_webdav_loc_conf_t *conf,
374 const ngx_http_brix_webdav_req_ctx_t *ctx,
375 const u_char *body, size_t body_len)
376{
377 STACK_OF(X509) *chain;
378 X509 *eec;
379 char key[BRIX_UCRED_KEY_MAX];
380
381 chain = delegation_parse_chain(body, body_len);
382 if (chain == NULL) {
383 delegation_reject(r, NGX_HTTP_BAD_REQUEST,
384 "unparseable proxy PEM\n");
385 return;
386 }
387
388 eec = delegation_find_eec(chain);
389 if (eec == NULL) {
390 sk_X509_pop_free(chain, X509_free);
391 delegation_reject(r, NGX_HTTP_BAD_REQUEST,
392 "no end-entity certificate in chain\n");
393 return;
394 }
395
396 /* Expiry checked before identity: an expired-and-mismatched upload
397 * should not leak whether it also happens to name another identity. */
398 if (delegation_chain_expired(chain)) {
399 sk_X509_pop_free(chain, X509_free);
400 delegation_reject(r, NGX_HTTP_BAD_REQUEST, "proxy is expired\n");
401 return;
402 }
403
404 /* Chain-of-trust before identity: the EEC DN string compare below is only
405 * meaningful once the EEC is proven to chain to a trusted CA. A failure
406 * here (self-signed / wrong-CA / untrusted issuer) and a DN mismatch both
407 * report the same 403 so we never reveal which check the upload failed. */
408 if (delegation_chain_trusted(r, conf, chain) != NGX_OK) {
409 sk_X509_pop_free(chain, X509_free);
410 delegation_reject(r, NGX_HTTP_FORBIDDEN,
411 "proxy chain does not verify against a trusted"
412 " CA\n");
413 return;
414 }
415
416 if (!delegation_eec_dn_matches(eec, ctx->dn)) {
417 char dn_log[1024];
418
419 brix_sanitize_log_string(ctx->dn, dn_log, sizeof(dn_log));
420 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
421 "brix_delegation: DN mismatch — authenticated as \"%s\","
422 " uploaded proxy is for a different identity", dn_log);
423 sk_X509_pop_free(chain, X509_free);
424 delegation_reject(r, NGX_HTTP_FORBIDDEN,
425 "proxy identity does not match authenticated"
426 " client\n");
427 return;
428 }
429
430 sk_X509_pop_free(chain, X509_free);
431
432 if (brix_sd_ucred_key(ctx->dn, key, sizeof(key)) != NGX_OK) {
433 delegation_reject(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
434 "credential key derivation failed\n");
435 return;
436 }
437
438 if (delegation_store_pem(r->connection->log,
439 &conf->common.storage_credential_dir, key, body, body_len)
440 != NGX_OK)
441 {
442 delegation_reject(r, NGX_HTTP_INSUFFICIENT_STORAGE,
443 "failed to store credential\n");
444 return;
445 }
446
447 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
448 "brix_delegation: stored proxy for key=%s", key);
449 delegation_reject(r, NGX_HTTP_CREATED, "OK\n");
450}
