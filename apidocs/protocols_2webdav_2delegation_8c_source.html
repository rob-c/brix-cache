1/*
2 * delegation.c - authenticated proxy delegation endpoints: the simple
3 * proxy-UPLOAD form (phase-2 T8) and the standard GridSite two-step
4 * getProxyReq/putProxy REST flow (phase-3 T4).
5 *
6 * ============================ Phase-2 T8: upload ==========================
7 * WHAT: PUT/POST /.well-known/brix-delegation, body = the client's own
8 * RFC-3820 x.509 proxy chain (PEM). Requires GSI-cert-authenticated
9 * TLS (the request must already carry a verified client-certificate
10 * identity — see auth_cert.c), validates the uploaded proxy, and
11 * atomically stores it at <storage_credential_dir>/<key>.pem so
12 * Phase-1 credential selection (brix_sd_ucred_select) picks it up
13 * for that user's subsequent davs/S3 origin sessions.
14 *
15 * WHY: Phase-1 per-user backend credentials require a proxy to already
16 * exist under storage_credential_dir before a user's first origin
17 * request. Operationally that means out-of-band provisioning. This
18 * endpoint lets an already-authenticated client "capture" its own
19 * delegation over the wire (proxy-UPLOAD, not the full GridSite
20 * getProxyReq/putProxy two-step) so no pre-provisioning is required.
21 *
22 * HOW: 1. Gate on brix_delegation_endpoint + the well-known path.
23 * 2. Require ctx->verified && ctx->auth_source == "cert"/"nginx"/
24 * "manual" (GSI/client-cert path — never a bearer token) and a
25 * non-empty ctx->dn.
26 * 3. Read the whole body via brix_http_body_read_all (bounded, small).
27 * 4. Parse the PEM chain (PEM_read_bio_X509 loop, mirrors the pattern
28 * in src/auth/gsi/parse_x509.c); find the end-entity cert (the one
29 * brix_px_classify() reports as BRIX_PX_NONE — proxies in the
30 * uploaded chain are excluded by construction).
31 * 5. Check every cert in the chain (proxy and EEC) has a notAfter
32 * still in the future, and the EEC's subject DN (brix_x509_oneline,
33 * same normalisation as ctx->dn) equals the authenticated client's
34 * DN — a client may only upload a proxy for its own identity.
35 * 6. Derive the credential key (brix_sd_ucred_key) and atomically
36 * write the whole uploaded PEM to <cred_dir>/<key>.pem: open a
37 * temp file in the same directory (O_CREAT|O_EXCL|O_WRONLY, 0600),
38 * write, fsync, close, then rename() over the final path. This is
39 * a raw filesystem write to a service-owned config directory (not
40 * an export), so it is marked vfs-seam-allow rather than routed
41 * through the VFS, which confines to the export root only.
42 *
43 * ======================= Phase-3 T4: two-step GridSite ====================
44 * WHAT: The standard GridSite getProxyReq/putProxy handshake, adapted to
45 * plain HTTP: GET /.well-known/brix-delegation/request returns a
46 * fresh CSR + a delegation-id; the client signs the CSR with its own
47 * EEC key and PUTs the signed proxy back to
48 * /.well-known/brix-delegation/<id>, which assembles and stores the
49 * full delegated proxy exactly like the T8 upload path.
50 *
51 * WHY: T8's proxy-UPLOAD form requires the client to already hold a
52 * complete, signed proxy chain it is willing to hand over wholesale —
53 * fine for scripted/xrdcp-style clients but not how the reference
54 * GridSite delegation portType works, and not what some interop
55 * partners expect. The two-step form lets the SERVER generate the
56 * keypair (never touching disk, never leaving the server) so the
57 * client only ever signs a request — the standard, narrower trust
58 * model this specification exists to support alongside T8, not to
59 * replace it.
60 *
61 * HOW: 1. brix_deleg_store — a bounded (256-entry), per-worker, in-memory
62 * table of pending {id, fresh EVP_PKEY*, client_dn, expires_at}
63 * entries, TTL 600s, swept lazily on every get/put (no background
64 * thread — see brix_deleg_store_sweep). Per-worker, NOT SHM: the
65 * fresh private key must never leave the process that generated
66 * it, and this is a short interactive two-request handshake, not
67 * durable state — a multi-worker deployment needs sticky routing
68 * (an L7 proxy or client retry-on-same-worker) so both requests of
69 * one handshake land on the same worker. Documented on
70 * brix_deleg_store_t below; NOT a design flaw for THIS store, since
71 * SHM would additionally require serialising an EVP_PKEY private
72 * key into shared memory, which trades one small operational
73 * constraint (stickiness) for a much larger one (private key
74 * material outside process-local heap).
75 * 2. webdav_delegation_request_handle (GET .../request): reuse the
76 * SAME peer-cert access pattern as auth_cert.c's
77 * webdav_verify_proxy_cert() — SSL_get_peer_certificate(ssl) off
78 * r->connection->ssl->connection — PEM-export it as `parent_pem`,
79 * and call brix_gsi_build_pxyreq(parent_pem, ...) to get a fresh
80 * key + CSR (re-encoded request DER→PEM, mirroring
81 * src/auth/gsi/delegation.c's gsi_req_der_to_pem, since the CSR
82 * travels over HTTP as PEM here too — DER has no natural HTTP
83 * content-type and PEM is what a human/openssl workflow expects).
84 * Store the entry keyed by a CSPRNG id (RAND_bytes, 16 bytes / 32
85 * hex chars via brix_hex_encode); respond with the CSR PEM as the
86 * body and the id in X-Brix-Delegation-Id. UNLIKE the other two
87 * entry points, this one runs SYNCHRONOUSLY inside the content
88 * phase (a GET has no body to read asynchronously) and therefore
89 * returns an ngx_int_t instead of self-finalizing — see
90 * delegation.h's doc comment on this function for why
91 * self-finalizing here would double-finalize the request and
92 * crash the worker (caught by testing: the two-step e2e test
93 * initially crashed a worker with SIGSEGV in
94 * ngx_http_set_lingering_close via a double
95 * ngx_http_finalize_request, fixed by returning the status through
96 * webdav_metrics_return() like every other synchronous handler
97 * instead of calling webdav_metrics_finalize_request).
98 * 3. webdav_delegation_put_handle (PUT .../<id>): look up id (checked
99 * BEFORE sweeping the rest of the table, so an expired-but-present
100 * entry is reported as 410, distinct from a never-existed/already-
101 * consumed 404 — see brix_deleg_store_take); 404 if absent, 410 if
102 * expired, 403 (and DO NOT touch any file) if entry.client_dn !=
103 * ctx->dn. Otherwise split the
104 * uploaded body into its first PEM block (the signed proxy —
105 * brix_gsi_assemble_proxy's `proxy_pem` parses only the first
106 * block via PEM_read_bio_X509) and the remainder (the client's EEC
107 * chain — passed verbatim as `chain_pem`, appended after the proxy
108 * with no further parsing, exactly as brix_gsi_assemble_proxy's
109 * implementation does). Call brix_gsi_assemble_proxy(proxy_pem,
110 * reqkey, chain_pem, ...) — it verifies the signed proxy's pubkey
111 * matches the stored fresh key (this IS the delegation's proof of
112 * possession: only a client holding the matching CSR could produce
113 * a proxy whose key matches) and returns the assembled
114 * <proxy><chain><private key> PEM — the reqkey is serialized into
115 * the blob so the stored credential is complete and can
116 * authenticate downstream (proxy_ssl_certificate_key, TPC pull,
117 * origin auth all load key+chain from the one stored file).
118 * Validate every cert's notAfter is still in
119 * the future and the chain's end-entity DN matches ctx->dn (reuse
120 * delegation_chain_expired / delegation_eec_dn_matches, the SAME T8
121 * helpers — a two-step-assembled credential is validated exactly
122 * as strictly as an uploaded one). On success, atomically store
123 * via delegation_store_pem (the SAME T8 helper) and drop the store
124 * entry. One-shot policy (documented on brix_deleg_store_drop
125 * below): the entry is dropped and its EVP_PKEY freed on ANY
126 * terminal outcome of this call — success, DN mismatch, expired,
127 * malformed body, or assemble failure — never left pending for a
128 * retry. This is the simplest policy that guarantees no EVP_PKEY
129 * leak on any path and closes the id after first use either way,
130 * matching the brief's documented default.
131 */

178 * certificate (the one cert that is NOT itself a proxy per

180 * `chain`) or NULL if every cert in the chain is a proxy (malformed

185{

199/*
200 * delegation_parse_chain - parse a PEM-concatenated cert chain from a
201 * NUL-terminated buffer. Mirrors the BIO+loop pattern in
202 * src/auth/gsi/parse_x509.c (gsi_chain_from_plaintext). Returns a
203 * non-empty STACK_OF(X509) (caller must sk_X509_pop_free(chain, X509_free))
204 * or NULL on any parse failure.
205 */
206static STACK_OF(X509) *
207delegation_parse_chain(const u_char *pem, size_t pem_len)
208{
209 BIO *bio;
210 STACK_OF(X509) *chain;
211 X509 *cert;
212
213 bio = BIO_new_mem_buf(pem, (int) pem_len);
214 if (bio == NULL) {
215 return NULL;
216 }
217
218 chain = sk_X509_new_null();
219 if (chain == NULL) {
220 BIO_free(bio);
221 return NULL;
222 }
223
224 while ((cert = PEM_read_bio_X509(bio, NULL, NULL, NULL)) != NULL) {
225 sk_X509_push(chain, cert);
226 }
227 BIO_free(bio);
228
229 if (sk_X509_num(chain) == 0) {
230 sk_X509_pop_free(chain, X509_free);
231 return NULL;
232 }
233 return chain;
234}

236/*
237 * delegation_chain_expired - true iff ANY certificate in the uploaded chain
238 * (proxy or EEC) has a notAfter that is not in the future. A delegation is
239 * only as strong as its weakest link: an expired proxy cert grants no
240 * authority even if the EEC underneath it is still valid, so the whole
241 * chain must check out.
242 */
243static int
244delegation_chain_expired(STACK_OF(X509) *chain)
245{
246 int i, n = sk_X509_num(chain);
247
248 for (i = 0; i < n; i++) {
249 if (X509_cmp_current_time(X509_get0_notAfter(sk_X509_value(chain, i)))
250 <= 0)
251 {
252 return 1;
253 }
254 }
255 return 0;
256}

272/*
273 * delegation_chain_trusted - cryptographically verify that a delegated proxy
274 * chain actually chains to a CA in the frontend's trust store.
275 *
276 * WHAT: run the same PKIX + RFC-3820 proxy verification the WebDAV
277 * client-certificate auth path uses (brix_gsi_verify_chain against
278 * conf->ca_store), treating the topmost proxy (chain[0]) as the leaf
279 * and the whole parsed chain as the untrusted intermediate set.
280 *
281 * WHY: the DN-equality check (delegation_eec_dn_matches) alone compares the
282 * EEC's *self-asserted* subject string against the authenticated client
283 * DN — it proves nothing about issuance. A credential-writing endpoint
284 * must not store a proxy it cannot anchor to a trusted CA, otherwise the
285 * authenticated owner could plant a self-signed cert whose subject string
286 * is spoofed to equal their own DN. Running full chain verification
287 * first turns the subsequent DN string compare into an identity check on
288 * a cryptographically-anchored EEC rather than an attacker-controlled
289 * string. Mirrors auth_cert.c: a NULL ca_store means we cannot verify,
290 * so we refuse (caller maps to 403) rather than trust blindly.
291 *
292 * HOW: NULL store -> NGX_ERROR (refuse, logged). Otherwise leaf = chain[0]
293 * (assemble/upload order is proxy-first), untrusted = the full chain
294 * (OpenSSL tolerates the leaf appearing in the untrusted stack), and
295 * client_purpose = 0 so RFC-3820 proxy chains are accepted (the davs://
296 * GSI mode, not the TLS-leaf mode). Returns NGX_OK iff the chain
297 * verifies to the trust store; brix_gsi_verify_chain has already logged
298 * the specific reason on failure. Owns nothing.
299 */
300static ngx_int_t
301delegation_chain_trusted(ngx_http_request_t *r,
302 const ngx_http_brix_webdav_loc_conf_t *conf, STACK_OF(X509) *chain)
303{
304 brix_gsi_verify_result_t verify_res;
305 X509 *leaf;
306
307 if (conf->ca_store == NULL) {
308 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
309 "brix_delegation: no trusted CA store configured — cannot verify"
310 " delegated proxy; refusing to store an unverifiable credential"
311 " (set brix_webdav_auth required with a CA on this location)");
312 return NGX_ERROR;
313 }
314
315 leaf = sk_X509_value(chain, 0);
316 if (leaf == NULL) {
317 return NGX_ERROR;
318 }
319
320 if (brix_gsi_verify_chain(r->connection->log, conf->ca_store, leaf, chain,
321 conf->verify_depth, &verify_res, 0 /* GSI: accept RFC-3820 proxy */)
322 != NGX_OK)
323 {
324 /* brix_gsi_verify_chain already logged the specific error. */
325 return NGX_ERROR;
326 }
327
328 return NGX_OK;
329}

331/*
332 * delegation_store_pem - atomically write `pem`/`pem_len` to
333 * <dir>/<key>.pem: temp file in the same directory (O_CREAT|O_EXCL|
334 * O_WRONLY, 0600), write, fsync, close, rename() over the final path.
335 * vfs-seam-allow: storage_credential_dir is service-owned server config,
336 * not export storage — the VFS confines to the export root only, the
337 * wrong root for this write.
338 *
339 * Returns NGX_OK on success, NGX_ERROR on any I/O failure (temp file is
340 * best-effort unlinked on the way out; caller maps failure to 507).
341 */
342static ngx_int_t
343delegation_store_pem(ngx_log_t *log, const ngx_str_t *dir, const char *key,
344 const u_char *pem, size_t pem_len)
345{
346 char final_path[1024];
347 char tmp_path[1024];
348 int fd;
349 ssize_t written;
350
351 if (dir->len == 0
352 || dir->len + 1 + strlen(key) + sizeof(".pem") >= sizeof(final_path)
353 || dir->len + 1 + strlen(key) + sizeof(".pem.XXXXXX") >= sizeof(tmp_path))
354 {
355 ngx_log_error(NGX_LOG_ERR, log, 0,
356 "brix_delegation: credential dir/key too long");
357 return NGX_ERROR;
358 }
359
360 (void) ngx_snprintf((u_char *) final_path, sizeof(final_path) - 1,
361 "%V/%s.pem%Z", dir, key);
362 (void) snprintf(tmp_path, sizeof(tmp_path), "%.*s/.%s.pem.upload.%ld",
363 (int) dir->len, dir->data, key, (long) getpid());
364
365 fd = open(tmp_path, /* vfs-seam-allow: cred dir is svc-owned config, not an export */
366 O_WRONLY | O_CREAT | O_EXCL | O_NOFOLLOW, 0600);
367 if (fd < 0 && ngx_errno == NGX_ENOENT) {
368 /* Store dir vanished (tmpfs default is wiped by a reboot without a
369 * config reload, or an admin rm'd it) — recreate it 0700 as the
370 * worker uid and retry once before shouting. */
371 if (mkdir((const char *) dir->data, 0700) == 0 /* vfs-seam-allow: cred dir is svc-owned config, not an export */
372 || ngx_errno == NGX_EEXIST)
373 {
374 fd = open(tmp_path, /* vfs-seam-allow: cred dir is svc-owned config, not an export */
375 O_WRONLY | O_CREAT | O_EXCL | O_NOFOLLOW, 0600);
376 }
377 }
378 if (fd < 0) {
379 ngx_log_error(NGX_LOG_ALERT, log, ngx_errno,
380 "brix_delegation: open(\"%s\") temp cred file failed — "
381 "credential store \"%V\" is missing or not writable; "
382 "delegation will not work until "
383 "brix_storage_credential_dir is fixed",
384 tmp_path, dir);
385 return NGX_ERROR;
386 }
387
388 written = write(fd, pem, pem_len);
389 if (written < 0 || (size_t) written != pem_len || fsync(fd) != 0) {
390 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
391 "brix_delegation: write/fsync(\"%s\") failed", tmp_path);
392 close(fd);
393 unlink(tmp_path); /* vfs-seam-allow: cred dir is svc-owned config, not an export */
394 return NGX_ERROR;
395 }
396 close(fd);
397
398 if (rename(tmp_path, final_path) != 0) { /* vfs-seam-allow: cred dir is svc-owned config, not an export */
399 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
400 "brix_delegation: rename(\"%s\" -> \"%s\") failed",
401 tmp_path, final_path);
402 unlink(tmp_path); /* vfs-seam-allow: cred dir is svc-owned config, not an export */
403 return NGX_ERROR;
404 }
405 return NGX_OK;
406}

408/*
409 * delegation_reject - send a small plain-text error body and finalize the
410 * request. Single choke point so every failure branch below is one line.
411 */
412static void
413delegation_reject(ngx_http_request_t *r, ngx_uint_t status, const char *msg)
414{
415 ngx_buf_t *b;
416 ngx_chain_t out;
417 size_t len = strlen(msg);
418 u_char *buf = ngx_pnalloc(r->pool, len);
419
420 if (buf == NULL) {
421 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
422 return;
423 }
424 ngx_memcpy(buf, msg, len);
425
426 b = ngx_pcalloc(r->pool, sizeof(*b));
427 if (b == NULL) {
428 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
429 return;
430 }
431 b->pos = buf;
432 b->last = buf + len;
433 b->memory = 1;
434 b->last_buf = 1;
435 out.buf = b;
436 out.next = NULL;
437
438 r->headers_out.status = status;
439 r->headers_out.content_length_n = (off_t) len;
440 r->headers_out.content_type.data = (u_char *) "text/plain";
441 r->headers_out.content_type.len = sizeof("text/plain") - 1;
442 r->headers_out.content_type_len = r->headers_out.content_type.len;
443
444 if (ngx_http_send_header(r) == NGX_ERROR || r->header_only) {
445 webdav_metrics_finalize_request(r, status);
446 return;
447 }
448 webdav_metrics_finalize_request(r, ngx_http_output_filter(r, &out));
449}

451/* ========================================================================
452 * Phase-3 T4: pending-delegation store (per-worker, in-memory, TTL-bounded).
453 *
454 * WHAT: A fixed-capacity table of {id, fresh EVP_PKEY*, client_dn,
455 * expires_at} entries, one per in-flight getProxyReq/putProxy
456 * handshake. brix_deleg_store_put() inserts (evicting expired
457 * entries first, then refusing if still full); brix_deleg_store_take()
458 * looks up + REMOVES an entry by id (one-shot: the caller always owns
459 * the returned EVP_PKEY* and must free it), sweeping expired entries
460 * as a side effect of every call so no background thread is needed.
461 *
462 * WHY: Per-worker, NOT SHM. The fresh private key generated for a
463 * getProxyReq must NEVER leave the process that generated it (the
464 * brief's hardest requirement) — SHM would mean either serialising an
465 * EVP_PKEY into shared bytes (a private key exposed to every other
466 * worker process) or storing only a reference nginx cannot safely
467 * cross-process anyway. A module-static per-worker table keeps the
468 * key in this worker's heap for its ~600s lifetime and nowhere else.
469 * The operational cost: a getProxyReq and its matching putProxy MUST
470 * land on the same worker. For a single-worker deployment (typical
471 * test/small-site config) this is automatic. For multi-worker, the
472 * deployment needs sticky routing (e.g. an L7 proxy keyed on client
473 * IP/TLS session, or a client that simply retries putProxy against
474 * the same connection/worker it got the CSR from) — documented here
475 * and in the reference doc, not solved by this store.
476 *
477 * HOW: Fixed array (BRIX_DELEG_STORE_CAP slots), linear scan (the table is
478 * small and operations are rare — an O(n) scan is simpler and just as
479 * fast in practice as a hash table at this size). brix_deleg_store()
480 * lazily allocates the singleton on first use (module-static pointer,
481 * the ONE new global this feature introduces, exactly as scoped in
482 * the task brief) from ngx_cycle->pool so it lives for the worker's
483 * lifetime; nginx never destroys that pool mid-run.
484 * ======================================================================== */
485
486/* Delegation ids are CSPRNG bytes, hex-encoded; see brix_deleg_store_put. */
487#define BRIX_DELEG_ID_BYTES 16
488#define BRIX_DELEG_ID_HEXLEN (BRIX_DELEG_ID_BYTES * 2)
489
490/* Bounded capacity: a pending handshake is a brief interactive round-trip
491 * (client signs a CSR, usually within seconds); 256 concurrent in-flight
492 * handshakes per worker is generous headroom without unbounded growth. */
493#define BRIX_DELEG_STORE_CAP 256
494
495/* Pending-delegation lifetime: long enough for a human/script to sign and
496 * return a CSR, short enough that an abandoned handshake's key doesn't
497 * linger. Matches the brief's suggested 10 minutes. */
498#define BRIX_DELEG_TTL_SEC 600
499
500typedef struct {
501 char id[BRIX_DELEG_ID_HEXLEN + 1]; /* hex, NUL-terminated; empty = free slot */
502 EVP_PKEY *fresh_key; /* owned; never written to disk */
503 char client_dn[1024]; /* bound at getreq, rechecked at putProxy */
504 time_t created_at;
505 time_t expires_at;
506} brix_deleg_entry_t;
507
508typedef struct {
509 brix_deleg_entry_t slots[BRIX_DELEG_STORE_CAP];
510} brix_deleg_store_t;
511
512/* The one documented per-worker global this feature introduces (see the
513 * module doc-block above for why this cannot be SHM). Lazily allocated on
514 * first use from ngx_cycle->pool; never freed early (worker-lifetime). */
515static brix_deleg_store_t *brix_deleg_store_singleton = NULL;
516
517static brix_deleg_store_t *
518brix_deleg_store(void)
519{
520 if (brix_deleg_store_singleton == NULL) {
521 brix_deleg_store_singleton =
522 ngx_pcalloc(ngx_cycle->pool, sizeof(brix_deleg_store_t));
523 }
524 return brix_deleg_store_singleton;
525}

527/*
528 * brix_deleg_entry_free - release an entry's owned resources and mark the
529 * slot free. The ONLY place an EVP_PKEY is freed, so every store exit path
530 * (evict, sweep, take, drop) routes through this — the single choke point
531 * that makes "no EVP_PKEY leak on any path" auditable by inspection.
532 */
533static void
534brix_deleg_entry_free(brix_deleg_entry_t *e)
535{
536 if (e->fresh_key != NULL) {
537 EVP_PKEY_free(e->fresh_key);
538 e->fresh_key = NULL;
539 }
540 e->id[0] = '\0';
541 e->client_dn[0] = '\0';
542 e->created_at = 0;
543 e->expires_at = 0;
544}

546/*
547 * brix_deleg_store_sweep - free every expired entry. Called at the top of
548 * both brix_deleg_store_put (to make room) and brix_deleg_store_take (so a
549 * lookup never returns a stale entry) — this IS the store's TTL enforcement;
550 * there is no background timer.
551 */
552static void
553brix_deleg_store_sweep(brix_deleg_store_t *st, time_t now)
554{
555 int i;
556
557 for (i = 0; i < BRIX_DELEG_STORE_CAP; i++) {
558 if (st->slots[i].id[0] != '\0' && st->slots[i].expires_at <= now) {
559 brix_deleg_entry_free(&st->slots[i]);
560 }
561 }
562}

564/*
565 * brix_deleg_store_put - insert a new pending entry with a fresh CSPRNG id.
566 *
567 * WHAT: Sweeps expired entries, then scans for a free slot. On success,
568 * fills id_out (caller-supplied buffer, >= BRIX_DELEG_ID_HEXLEN+1)
569 * with the hex id, stores fresh_key (ownership transfers to the
570 * store — caller must NOT free it after a successful put) and
571 * client_dn, sets expires_at = now + BRIX_DELEG_TTL_SEC.
572 *
573 * WHY: CSPRNG (RAND_bytes, 16 bytes / 32 hex chars) makes the id
574 * unguessable — an attacker cannot brute-force or predict another
575 * client's pending delegation-id to attempt a cross-client putProxy
576 * (the DN check at take-time is the second, independent layer).
577 *
578 * HOW: 1. Sweep. 2. RAND_bytes + brix_hex_encode for the id (retry once on
579 * an astronomically unlikely id collision — a fresh RAND_bytes draw,
580 * not a fallback to a weaker source). 3. Linear scan for id[0]=='\0'.
581 * 4. All full after sweep -> NGX_DECLINED (caller maps to 503,
582 * documented eviction policy: evict-expired-first, else reject new
583 * getreq rather than evict a live pending handshake).
584 */
585static ngx_int_t
586brix_deleg_store_put(brix_deleg_store_t *st, EVP_PKEY *fresh_key,
587 const char *client_dn, char *id_out, size_t id_out_cap)
588{
589 unsigned char raw[BRIX_DELEG_ID_BYTES];
590 char hex[BRIX_DELEG_ID_HEXLEN + 1];
591 time_t now = time(NULL);
592 int i;
593
594 if (id_out_cap < sizeof(hex)) {
595 return NGX_ERROR;
596 }
597 brix_deleg_store_sweep(st, now);
598
599 if (RAND_bytes(raw, sizeof(raw)) != 1) {
600 return NGX_ERROR;
601 }
602 brix_hex_encode(raw, sizeof(raw), hex);
603
604 for (i = 0; i < BRIX_DELEG_STORE_CAP; i++) {
605 if (st->slots[i].id[0] == '\0') {
606 ngx_memcpy(st->slots[i].id, hex, sizeof(hex));
607 st->slots[i].fresh_key = fresh_key;
608 ngx_memcpy(st->slots[i].client_dn, client_dn,
609 ngx_min(strlen(client_dn) + 1,
610 sizeof(st->slots[i].client_dn) - 1));
611 st->slots[i].client_dn[sizeof(st->slots[i].client_dn) - 1] = '\0';
612 st->slots[i].created_at = now;
613 st->slots[i].expires_at = now + BRIX_DELEG_TTL_SEC;
614 ngx_memcpy(id_out, hex, sizeof(hex));
615 return NGX_OK;
616 }
617 }
618 return NGX_DECLINED; /* bounded store full even after sweep */
619}

621/* Outcome of brix_deleg_store_take, distinguishing every caller-relevant
622 * terminal state (dispatch.c / the putProxy handler map these to distinct
623 * HTTP statuses: 404 / 410 / 403 / found). */
624typedef enum {
625 BRIX_DELEG_TAKE_NOT_FOUND = 0,
626 BRIX_DELEG_TAKE_EXPIRED,
627 BRIX_DELEG_TAKE_DN_MISMATCH,
628 BRIX_DELEG_TAKE_OK
629} brix_deleg_take_t;
630
631/*
632 * brix_deleg_store_take - one-shot lookup-and-remove by id, DN-checked.
633 *
634 * WHAT: Scans for the id FIRST (before sweeping) so an expired-but-still-
635 * present entry can be reported as EXPIRED rather than being wiped by
636 * a blind sweep and misreported as NOT_FOUND — "never existed" /
637 * "already consumed" is a genuinely different, distinguishable
638 * outcome from "existed but timed out" for the caller's error
639 * message. On a live match: if want_dn does not match the entry's
640 * stored client_dn, frees the entry (one-shot: a cross-client attempt
641 * burns the id rather than leaving it available for a retry) and
642 * returns DN_MISMATCH WITHOUT transferring the key. On a DN match,
643 * transfers fresh_key ownership to *key_out (caller must
644 * EVP_PKEY_free it) and frees the slot's other fields, returning OK.
645 * Once the target id is resolved (found-and-handled, or absent),
646 * sweeps the REST of the table for any other expired entries — this
647 * is still the TTL enforcement's only trigger point (alongside
648 * brix_deleg_store_put), just ordered after the id-specific check so
649 * the two responsibilities (report THIS id's true state; reclaim
650 * OTHER entries' expired slots) don't fight over the same pass.
651 *
652 * WHY: This is the id-to-client binding re-check the brief requires:
653 * "must belong to THIS authenticated client DN — reject
654 * cross-client". One-shot on EVERY terminal outcome (not just
655 * success) is the documented policy — see the module doc-block —
656 * so a determined attacker gets exactly one guess per id and success
657 * or failure both close it out, with no leaked EVP_PKEY on any path
658 * (brix_deleg_entry_free is the sole free site).
659 *
660 * HOW: Linear scan for id match. Not found -> sweep the rest -> NOT_FOUND.
661 * Found but past expiry -> free + sweep the rest -> EXPIRED.
662 * DN mismatch -> free + sweep the rest -> DN_MISMATCH.
663 * Match -> transfer key, free the rest of the slot, sweep the
664 * remaining table, -> OK.
665 */
666static brix_deleg_take_t
667brix_deleg_store_take(brix_deleg_store_t *st, const char *id,
668 const char *want_dn, EVP_PKEY **key_out)
669{
670 time_t now = time(NULL);
671 int i;
672 brix_deleg_take_t result = BRIX_DELEG_TAKE_NOT_FOUND;
673
674 for (i = 0; i < BRIX_DELEG_STORE_CAP; i++) {
675 brix_deleg_entry_t *e = &st->slots[i];
676
677 if (e->id[0] == '\0' || strcmp(e->id, id) != 0) {
678 continue;
679 }
680 if (e->expires_at <= now) {
681 brix_deleg_entry_free(e);
682 result = BRIX_DELEG_TAKE_EXPIRED;
683 } else if (strcmp(e->client_dn, want_dn) != 0) {
684 brix_deleg_entry_free(e);
685 result = BRIX_DELEG_TAKE_DN_MISMATCH;
686 } else {
687 *key_out = e->fresh_key;
688 e->fresh_key = NULL; /* ownership -> caller; entry_free must not free it */
689 brix_deleg_entry_free(e);
690 result = BRIX_DELEG_TAKE_OK;
691 }
692 break;
693 }
694 brix_deleg_store_sweep(st, now);
695 return result;
696}

698/*
699 * delegation_get_peer_pem - PEM-export the TLS peer certificate off this
700 * request's SSL connection. Reuses the exact acquisition pattern
701 * auth_cert.c's webdav_verify_proxy_cert() uses (SSL_get_peer_certificate
702 * off r->connection->ssl->connection) — this endpoint runs strictly after
703 * that same verification has already populated ctx->verified/ctx->dn, so a
704 * peer cert is guaranteed present; this just re-fetches the X509* (the
705 * verified req_ctx stores only the derived DN string, not the cert object)
706 * and serialises it to PEM in memory for brix_gsi_build_pxyreq's
707 * `parent_pem` argument. Caller frees *pem via free() (malloc'd, matching
708 * the ngx-free proxy_req.c/gsi delegation.c convention for PEM buffers).
709 */
710static ngx_int_t
711delegation_get_peer_pem(ngx_http_request_t *r, u_char **pem, size_t *pem_len)
712{
713 SSL *ssl;
714 X509 *leaf;
715 BIO *bio;
716 char *data;
717 long n;
718
719 if (r->connection->ssl == NULL) {
720 return NGX_ERROR;
721 }
722 ssl = r->connection->ssl->connection;
723 leaf = SSL_get_peer_certificate(ssl);
724 if (leaf == NULL) {
725 return NGX_ERROR;
726 }
727
728 bio = BIO_new(BIO_s_mem());
729 if (bio == NULL || PEM_write_bio_X509(bio, leaf) != 1) {
730 BIO_free(bio);
731 X509_free(leaf);
732 return NGX_ERROR;
733 }
734 X509_free(leaf);
735
736 n = BIO_get_mem_data(bio, &data);
737 *pem = malloc((size_t) n + 1);
738 if (*pem == NULL) {
739 BIO_free(bio);
740 return NGX_ERROR;
741 }
742 memcpy(*pem, data, (size_t) n);
743 (*pem)[n] = '\0';
744 *pem_len = (size_t) n;
745 BIO_free(bio);
746 return NGX_OK;
747}

749/*
750 * delegation_req_der_to_pem - re-encode brix_gsi_build_pxyreq's DER
751 * X509_REQ output as PEM. The two-step flow travels the CSR over HTTP as
752 * a body a human or `openssl req` workflow can consume directly, and
753 * mirrors src/auth/gsi/delegation.c's gsi_req_der_to_pem (the wire-protocol
754 * delegation path re-encodes for the exact same reason — the crypto core
755 * emits DER, the transport wants PEM). Returns a malloc'd NUL-terminated
756 * PEM buffer (caller frees), or NULL on parse/encode failure.
757 */
758static u_char *
759delegation_req_der_to_pem(const uint8_t *der, size_t der_len, size_t *pem_len)
760{
761 const unsigned char *p = der;
762 X509_REQ *req = d2i_X509_REQ(NULL, &p, (long) der_len);
763 BIO *bio;
764 char *data;
765 long n;
766 u_char *out = NULL;
767
768 if (req == NULL) {
769 return NULL;
770 }
771 bio = BIO_new(BIO_s_mem());
772 if (bio != NULL && PEM_write_bio_X509_REQ(bio, req) == 1) {
773 n = BIO_get_mem_data(bio, &data);
774 out = malloc((size_t) n + 1);
775 if (out != NULL) {
776 memcpy(out, data, (size_t) n);
777 out[n] = '\0';
778 *pem_len = (size_t) n;
779 }
780 }
781 BIO_free(bio);
782 X509_REQ_free(req);
783 return out;
784}

786/*
787 * delegation_send_csr - build a 200 response with the CSR PEM as the body
788 * and the delegation-id in X-Brix-Delegation-Id. Content-Type: application/
789 * x-pem-file (RFC-ish convention for a bare PEM payload; chosen over
790 * text/plain so a client can distinguish "this response IS a PEM
791 * document" from the plain-text error bodies delegation_reject sends).
792 *
793 * Returns the ngx_int_t status the CALLER must return up through
794 * webdav_metrics_return() — this function does NOT call
795 * ngx_http_send_header/ngx_http_output_filter's result into
796 * webdav_metrics_finalize_request itself, unlike delegation_reject.
797 * webdav_delegation_request_handle runs synchronously inside the content
798 * phase (no async body read is in flight), so self-finalizing here would
799 * double-finalize the request once nginx's content phase ALSO finalizes
800 * whatever this handler returns — see delegation.h's doc comment on
801 * webdav_delegation_request_handle for why that crashes the worker.
802 */
803static ngx_int_t
804delegation_send_csr(ngx_http_request_t *r, const char *id,
805 const u_char *pem, size_t pem_len)
806{
807 ngx_buf_t *b;
808 ngx_chain_t out;
809 ngx_table_elt_t *h;
810 u_char *buf;
811 ngx_str_t id_str;
812 u_char *id_copy;
813
814 buf = ngx_pnalloc(r->pool, pem_len);
815 if (buf == NULL) {
816 return NGX_HTTP_INTERNAL_SERVER_ERROR;
817 }
818 ngx_memcpy(buf, pem, pem_len);
819
820 b = ngx_pcalloc(r->pool, sizeof(*b));
821 if (b == NULL) {
822 return NGX_HTTP_INTERNAL_SERVER_ERROR;
823 }
824 b->pos = buf;
825 b->last = buf + pem_len;
826 b->memory = 1;
827 b->last_buf = 1;
828 out.buf = b;
829 out.next = NULL;
830
831 id_str.len = strlen(id);
832 id_str.data = (u_char *) id;
833 id_copy = ngx_pstrdup(r->pool, &id_str);
834 if (id_copy == NULL) {
835 return NGX_HTTP_INTERNAL_SERVER_ERROR;
836 }
837
838 h = ngx_list_push(&r->headers_out.headers);
839 if (h == NULL) {
840 return NGX_HTTP_INTERNAL_SERVER_ERROR;
841 }
842 h->hash = 1;
843 ngx_str_set(&h->key, "X-Brix-Delegation-Id");
844 h->value.data = id_copy;
845 h->value.len = id_str.len;
846
847 r->headers_out.status = NGX_HTTP_OK;
848 r->headers_out.content_length_n = (off_t) pem_len;
849 r->headers_out.content_type.data = (u_char *) "application/x-pem-file";
850 r->headers_out.content_type.len = sizeof("application/x-pem-file") - 1;
851 r->headers_out.content_type_len = r->headers_out.content_type.len;
852
853 if (ngx_http_send_header(r) == NGX_ERROR || r->header_only) {
854 return NGX_HTTP_OK;
855 }
856 return ngx_http_output_filter(r, &out);
857}

902ngx_int_t
903webdav_delegation_request_handle(ngx_http_request_t *r)
904{
905 ngx_http_brix_webdav_req_ctx_t *ctx =
906 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
907 u_char *parent_pem = NULL;
908 size_t parent_pem_len = 0;
909 EVP_PKEY *newkey = NULL;
910 uint8_t *req_der = NULL;
911 size_t req_der_len = 0;
912 u_char *req_pem = NULL;
913 size_t req_pem_len = 0;
914 char err[160];
915 char id[BRIX_DELEG_ID_HEXLEN + 1];
916 ngx_int_t rc;
917
918 if (!delegation_client_authenticated(ctx)) {
919 return delegation_request_reject(r, NGX_HTTP_UNAUTHORIZED,
920 "GSI client-certificate authentication"
921 " required\n");
922 }
923
924 if (delegation_get_peer_pem(r, &parent_pem, &parent_pem_len) != NGX_OK) {
925 return delegation_request_reject(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
926 "cannot access client certificate\n");
927 }
928
929 err[0] = '\0';
930 {
931 const brix_gsi_blob_t parent_blob = { parent_pem, parent_pem_len };
932 const brix_gsi_err_t err_sink = { err, sizeof(err) };
933 brix_gsi_buf_t req_out = { NULL, 0 };
934
935 if (brix_gsi_build_pxyreq(&parent_blob, &newkey, &req_out,
936 &err_sink) != 0)
937 {
938 free(parent_pem);
939 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
940 "brix_delegation: build_pxyreq failed: %s", err);
941 return delegation_request_reject(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
942 "cannot build proxy request\n");
943 }
944 req_der = req_out.data;
945 req_der_len = req_out.len;
946 }
947 free(parent_pem);
948
949 req_pem = delegation_req_der_to_pem(req_der, req_der_len, &req_pem_len);
950 free(req_der);
951 if (req_pem == NULL) {
952 EVP_PKEY_free(newkey);
953 return delegation_request_reject(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
954 "cannot encode proxy request\n");
955 }
956
957 if (brix_deleg_store_put(brix_deleg_store(), newkey, ctx->dn,
958 id, sizeof(id)) != NGX_OK)
959 {
960 /* newkey ownership did NOT transfer on a failed put — free it here. */
961 EVP_PKEY_free(newkey);
962 free(req_pem);
963 return delegation_request_reject(r, NGX_HTTP_SERVICE_UNAVAILABLE,
964 "too many pending delegations, try again later\n");
965 }
966
967 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
968 "brix_delegation: issued getProxyReq id=%s", id);
969 rc = delegation_send_csr(r, id, req_pem, req_pem_len);
970 free(req_pem);
971 return rc;
972}

1031/*
1032 * delegation_split_first_pem_block - split a PEM buffer into its first
1033 * X509 certificate block (the signed proxy) and everything after it (the
1034 * issuer/EEC chain). brix_gsi_assemble_proxy's `proxy_pem` argument is
1035 * parsed with PEM_read_bio_X509, which reads only the FIRST PEM block off
1036 * the BIO — so the client-uploaded body (signed proxy followed by its EEC
1037 * chain, mirroring the wire-protocol delegation flow's signed_pem +
1038 * deleg_chain_pem split in src/auth/gsi/delegation.c) must be split at
1039 * that boundary before calling assemble. Re-parses via
1040 * delegation_parse_chain (same PEM-loop helper T8 uses) then re-serialises
1041 * cert[0] alone and certs[1..] concatenated — this also gives "unparseable"
1042 * and "no certs at all" rejection for free, reusing the T8 parse path
1043 * rather than a bespoke byte-offset scan.
1044 *
1045 * On success returns NGX_OK with proxy_pem / proxy_pem_len (cert[0] only)
1046 * and chain_pem / chain_pem_len (certs[1..], possibly zero-length if the
1047 * client sent only the proxy with no chain) both malloc'd (caller frees
1048 * both). NGX_ERROR on parse failure or an empty chain.
1049 */
1050static ngx_int_t
1051delegation_split_first_pem_block(const u_char *body, size_t body_len,
1052 u_char **proxy_pem, size_t *proxy_pem_len,
1053 u_char **chain_pem, size_t *chain_pem_len)
1054{
1055 STACK_OF(X509) *chain;
1056 X509 *first;
1057 size_t n;
1058
1059 chain = delegation_parse_chain(body, body_len);
1060 if (chain == NULL) {
1061 return NGX_ERROR;
1062 }
1063 n = (size_t) sk_X509_num(chain);
1064 first = sk_X509_value(chain, 0);
1065
1066 *proxy_pem = gsi_pem_export_one(first, proxy_pem_len);
1067 if (n > 1) {
1068 STACK_OF(X509) *rest = sk_X509_new_null();
1069 size_t i;
1070
1071 if (rest == NULL) {
1072 free(*proxy_pem);
1073 *proxy_pem = NULL;
1074 sk_X509_pop_free(chain, X509_free);
1075 return NGX_ERROR;
1076 }
1077 for (i = 1; i < n; i++) {
1078 sk_X509_push(rest, sk_X509_value(chain, (int) i));
1079 }
1080 *chain_pem = gsi_pem_export_chain(rest, chain_pem_len);
1081 sk_X509_free(rest); /* shallow: certs still owned by `chain` */
1082 } else {
1083 *chain_pem = malloc(1);
1084 if (*chain_pem != NULL) {
1085 (*chain_pem)[0] = '\0';
1086 }
1087 *chain_pem_len = 0;
1088 }
1089
1090 sk_X509_pop_free(chain, X509_free);
1091 if (*proxy_pem == NULL || *chain_pem == NULL) {
1092 free(*proxy_pem);
1093 free(*chain_pem);
1094 *proxy_pem = NULL;
1095 *chain_pem = NULL;

1101/*
1102 * delegation_extract_id - the trailing path segment of r->uri after
1103 * "/.well-known/brix-delegation/". Returns NGX_OK with id / id_len
1104 * pointing into r->uri.data (borrowed, valid for the request's lifetime —
1105 * no copy needed), or NGX_ERROR if the segment is empty or implausibly
1106 * long (> BRIX_DELEG_ID_HEXLEN*2, generous slack over the real 32-char
1107 * ids so a legitimate id is never rejected while an absurd one short-
1108 * circuits before the store's strcmp scan).
1109 */
1110static ngx_int_t
1111delegation_extract_id(ngx_http_request_t *r, const u_char **id,
1112 size_t *id_len)
1113{
1114 static const char prefix[] = "/.well-known/brix-delegation/";
1115 size_t plen = sizeof(prefix) - 1;
1116
1117 if (r->uri.len <= plen) {
1118 return NGX_ERROR;
1119 }
1120 /* The id is whatever follows the LAST occurrence of the prefix — mirror
1121 * dispatch.c's routing, which matches on this same fixed prefix. */
1122 {
1123 ngx_uint_t i;
1124 const u_char *found = NULL;
1125
1126 for (i = 0; i + plen <= r->uri.len; i++) {
1127 if (ngx_memcmp(r->uri.data + i, prefix, plen) == 0) {
1128 found = r->uri.data + i + plen;
1129 }
1130 }
1131 if (found == NULL) {
1132 return NGX_ERROR;
1133 }
1134 *id = found;
1135 *id_len = (size_t) (r->uri.data + r->uri.len - found);
1136 }
1137 if (*id_len == 0 || *id_len > (BRIX_DELEG_ID_HEXLEN * 2)) {
1138 return NGX_ERROR;
1139 }
1140 return NGX_OK;
1141}
