2 * fs_walk.c - shared directory traversal helpers.
3 *
4 * WHAT: Provides dot-entry detection, path joining, empty-directory check, recursive
5 * directory walk with callback, and confined tree removal. All functions share

25/*
26 * brix_fs_is_dot_entry - check if a directory entry name is "." or "..".
27 *
28 * WHAT: Returns NGX_TRUE if name is exactly "." (length 1) or ".." (length 2),
29 * NGX_FALSE otherwise. Handles NULL input safely.
30 *
31 * WHY: All directory traversal loops skip dot entries per POSIX convention. This
32 * shared check ensures consistency across dirlist, walk, and remove-tree paths.
33 *
34 * HOW: name[0] == '.' && (name[1]=='\0' || (name[1]=='.' && name[2]=='\0')).
35 */
36
37ngx_flag_t
38brix_fs_is_dot_entry(const char *name)
39{
40 return name != NULL
41 && name[0] == '.'
42 && (name[1] == '\0'
43 || (name[1] == '.' && name[2] == '\0'));
44}

46/*
47 * brix_fs_join_path - concatenate dir + '/' + name into out buffer.
48 *
49 * WHAT: Formats "dir/name" into out[0..outsz-1] via snprintf. Returns NGX_OK if
50 * the result fits, NGX_ERROR on NULL inputs or truncation.
51 *
52 * WHY: Directory traversal builds child paths by joining parent directory with entry
53 * name. This function handles buffer bounds checking uniformly.
54 *
55 * HOW: snprintf(out, outsz, "%s/%s", dir, name). Checks return value < outsz.
56 */
57
58ngx_int_t
59brix_fs_join_path(const char *dir, const char *name, char *out, size_t outsz)
60{
61 int n;
62
63 if (dir == NULL || name == NULL || out == NULL || outsz == 0) {
64 return NGX_ERROR;
65 }
66
67 n = snprintf(out, outsz, "%s/%s", dir, name);
68 return (n >= 0 && (size_t) n < outsz) ? NGX_OK : NGX_ERROR;
69}

71/*
72 * brix_fs_dir_is_empty - check whether a directory contains non-dot entries.
73 *
74 * WHAT: Opens path via opendir(), iterates readdir entries, skips dots, and sets
75 * *is_empty = 1 if no non-dot entries found (or empty dir), 0 otherwise.
76 *
77 * WHY: XRootD stat on collection returns whether it is empty. WebDAV MKCOL checks
78 * target directory emptiness. Single shared check avoids duplication.
79 *
80 * HOW: opendir → readdir loop, skip brix_fs_is_dot_entry(), set flag on first
81 * non-dot entry found. closedir always called. Returns NGX_OK or NGX_ERROR.
82 */
83
84ngx_int_t
85brix_fs_dir_is_empty(const char *path, ngx_flag_t *is_empty)
86{
87 DIR *dp;
88 struct dirent *de;
89
90 if (is_empty == NULL) {
91 return NGX_ERROR;
92 }
93
94 *is_empty = 1;
95 dp = opendir(path);
96 if (dp == NULL) {
97 return NGX_ERROR;
98 }
99
100 while ((de = readdir(dp)) != NULL) {
101 if (brix_fs_is_dot_entry(de->d_name)) {
102 continue;
103 }
104 *is_empty = 0;
105 break;
106 }
107
108 closedir(dp);
109 return NGX_OK;
110}

112/*
113 * brix_fs_walk_dir - recursive directory walk with callback, depth tracking.
114 *
115 * WHAT: Opens dir via opendir(), iterates entries (skip dots, skip hidden if opts),
116 * lstat each child, builds entry struct (path/name/st/depth), calls cb(entry,
117 * data). Recursively descends into subdirectories respecting max_depth and
118 * stay_on_device options.
119 *
120 * WHY: The recursive engine behind brix_fs_walk(). Used by dirlist handlers, PROPFIND
121 * collection walks, and tree removal. Options control filtering (hidden dirs,
122 * files vs dirs only, depth limit, cross-device skip).
123 *
124 * HOW: opendir → readdir loop per entry: join path via brix_fs_join_path(), lstat,
125 * apply opts filters, build entry struct, call cb(). If is_dir and within max_depth,
126 * recurse with depth+1. closedir on exit.
127 */
128
129static ngx_int_t
130brix_fs_walk_dir(ngx_log_t *log, const char *dir,
131 const brix_fs_walk_options_t *opts, brix_fs_walk_pt cb, void *data,
132 ngx_uint_t depth)
133{
134 DIR *dp;
135 struct dirent *de;
136 ngx_int_t rc;
137
138 dp = opendir(dir);
139 if (dp == NULL) {
140 ngx_log_error(NGX_LOG_WARN, log, errno,
141 "brix: cannot open directory \"%s\"", dir);
142 return NGX_ERROR;
143 }
144
145 rc = NGX_OK;
146
147 while ((de = readdir(dp)) != NULL) {
148 char child[PATH_MAX];
149 struct stat st;
150 brix_fs_walk_entry_t entry;
151 ngx_flag_t is_dir;
152
153 if (brix_fs_is_dot_entry(de->d_name)) {
154 continue;
155 }
156
157 if (opts != NULL && opts->skip_hidden && de->d_name[0] == '.') {
158 continue;
159 }
160
161 if (brix_fs_join_path(dir, de->d_name, child, sizeof(child))
162 != NGX_OK)
163 {
164 rc = NGX_ERROR;
165 continue;
166 }
167
168 if (lstat(child, &st) != 0) {
169 if (errno != ENOENT) {
170 ngx_log_error(NGX_LOG_WARN, log, errno,
171 "brix: lstat failed \"%s\"", child);
172 rc = NGX_ERROR;
173 }
174 continue;
175 }
176
177 if (opts != NULL && opts->stay_on_device && st.st_dev != opts->root_dev) {
178 continue;
179 }
180
181 is_dir = S_ISDIR(st.st_mode);
182 if (cb != NULL
183 && ((is_dir && (opts == NULL || opts->include_dirs))
184 || (!is_dir && (opts == NULL || opts->include_files))))
185 {
186 entry.path = child;
187 entry.name = de->d_name;
188 entry.st = &st;
189 entry.depth = depth;
190
191 if (cb(&entry, data) != NGX_OK) {
192 rc = NGX_ERROR;
193 break;
194 }
195 }
196
197 if (is_dir
198 && (opts == NULL || opts->max_depth == 0
199 || depth < opts->max_depth))
200 {
201 if (brix_fs_walk_dir(log, child, opts, cb, data, depth + 1)
202 != NGX_OK)
203 {
204 rc = NGX_ERROR;
205 }
206 }
207 }
208
209 closedir(dp);
210 return rc;
211}

216 * WHAT: Validates start_path, then delegates to brix_fs_walk_dir() starting at

237/*
238 * brix_fs_remove_tree_confined - recursively delete directory tree confined to root_canon.
239 *
240 * WHAT: Opens path via opendir(), iterates entries (skip dots), lstat each child,
241 * recurses into subdirectories, unlinks files via brix_unlink_confined_canon().
242 * After processing children, rmdir the directory itself. Returns NGX_OK on
243 * full removal or NGX_ERROR on any failure.
244 *
245 * WHY: XRootD DEL/MOVE/COPY on collections requires recursive child deletion. The
246 * confined check (root_canon) ensures no entry escapes the root path boundary,
247 * preventing ../ traversal attacks.
248 *
249 * HOW: opendir → readdir loop per entry: join path, lstat. If dir → recurse.
250 * If file → brix_unlink_confined_canon(root_canon, child, 0). After all
251 * children processed → brix_unlink_confined_canon(root_canon, path, 1) (rmdir).
252 */
253
254ngx_int_t
255brix_fs_remove_tree_confined(ngx_log_t *log, const char *root_canon,
256 const char *path)
257{
258 DIR *dp;
259 struct dirent *de;
260 ngx_int_t rc;
261 int rootfd;
262 const char *path_rel;
263
264 /*
265 * The opendir/readdir/lstat iteration below is NOT a beneath site: it walks
266 * by absolute child path built from single, kernel-supplied dirent names
267 * (no client `..`), and lstat does not follow the final component, so a
268 * dangling/outward symlink entry is seen as a symlink and unlinked as one
269 * (its target is never opened or followed). The actual mutations — every
270 * unlink and the final rmdir — DO go through the beneath API, which is where
271 * the export-root boundary is enforced. A readdir-relative
272 * openat(dirfd,...) walk would be a defence-in-depth refinement but is not a
273 * confinement gap here.
274 */
275 rootfd = brix_beneath_open_root(root_canon);
276 if (rootfd < 0) {
277 ngx_log_error(NGX_LOG_ERR, log, errno,
278 "brix: remove-tree cannot open root for \"%s\"", path);
279 return NGX_ERROR;
280 }
281 path_rel = brix_beneath_strip_root(root_canon, path);
282 if (path_rel == NULL) {
283 close(rootfd);
284 errno = EXDEV;
285 return NGX_ERROR;
286 }
287
288 /* Open the dir AS THE MAPPED USER under impersonation (broker fd + fdopendir):
289 * a multipart staging dir is owned 0700 by the mapped user, so a raw worker
290 * opendir() here fails EACCES and breaks AbortMultipartUpload / the cleanup.
291 * Off impersonation this is a plain opendir(). */
292 dp = brix_opendir_confined_canon(log, root_canon, path);
293 if (dp == NULL) {
294 close(rootfd);
295 if (errno == ENOENT) {
296 return NGX_OK;
297 }
298 ngx_log_error(NGX_LOG_ERR, log, errno,
299 "brix: remove-tree opendir failed \"%s\"", path);
300 return NGX_ERROR;
301 }
302
303 rc = NGX_OK;
304
305 while ((de = readdir(dp)) != NULL) {
306 char child[PATH_MAX];
307 struct stat st;
308 const char *child_rel;
309
310 if (brix_fs_is_dot_entry(de->d_name)) {
311 continue;
312 }
313
314 if (brix_fs_join_path(path, de->d_name, child, sizeof(child))
315 != NGX_OK)
316 {
317 rc = NGX_ERROR;
318 break;
319 }
320
321 child_rel = brix_beneath_strip_root(root_canon, child);
322 if (child_rel == NULL) {
323 errno = EXDEV;
324 rc = NGX_ERROR;
325 break;
326 }
327
328 /* lstat via the beneath helper so it runs AS THE MAPPED USER under
329 * impersonation (the raw lstat would fail EACCES inside a 0700 user-owned
330 * staging dir); off impersonation it is the same local lstat. */
331 if (brix_lstat_beneath(rootfd, child_rel, &st) != 0) {
332 if (errno == ENOENT) {
333 continue;
334 }
335 ngx_log_error(NGX_LOG_ERR, log, errno,
336 "brix: remove-tree lstat failed \"%s\"", child);
337 rc = NGX_ERROR;
338 break;
339 }
340
341 if (S_ISDIR(st.st_mode)) {
342 if (brix_fs_remove_tree_confined(log, root_canon, child)
343 != NGX_OK)
344 {
345 rc = NGX_ERROR;
346 break;
347 }
348 } else {
349 if (brix_unlink_beneath(rootfd, child_rel, 0) != 0) {
350 ngx_log_error(NGX_LOG_ERR, log, errno,
351 "brix: remove-tree unlink failed \"%s\"", child);
352 rc = NGX_ERROR;
353 break;
354 }
355 }
356 }
357
358 closedir(dp);
359
360 if (rc == NGX_OK
361 && brix_unlink_beneath(rootfd, path_rel, 1) != 0)
362 {
363 ngx_log_error(NGX_LOG_ERR, log, errno,
364 "brix: remove-tree rmdir failed \"%s\"", path);
365 rc = NGX_ERROR;
366 }
367
368 close(rootfd);
369 return rc;
370}
