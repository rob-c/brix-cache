2 * fs_walk.c - shared directory traversal helpers.
3 *
4 * WHAT: Provides dot-entry detection, path joining, empty-directory check, and
5 * recursive directory walk with callback. All functions share a common
6 * pattern of opendir/readdir/lstat with dot-filtering. Confined tree
7 * removal lives in the companion file fs_walk_remove.c.
8 *
9 * WHY: XRootD dirlist, WebDAV PROPFIND on collections, S3 ListObjects all need
10 * directory traversal. Single shared implementation avoids duplication

12 *
13 * HOW: opendir → readdir loop skipping "." and "..", lstat for each entry,

23/*
24 * brix_fs_is_dot_entry - check if a directory entry name is "." or "..".
25 *
26 * WHAT: Returns NGX_TRUE if name is exactly "." (length 1) or ".." (length 2),
27 * NGX_FALSE otherwise. Handles NULL input safely.
28 *
29 * WHY: All directory traversal loops skip dot entries per POSIX convention. This
30 * shared check ensures consistency across dirlist, walk, and remove-tree paths.
31 *
32 * HOW: name[0] == '.' && (name[1]=='\0' || (name[1]=='.' && name[2]=='\0')).
33 */
34
35ngx_flag_t
36brix_fs_is_dot_entry(const char *name)
37{
38 return name != NULL
39 && name[0] == '.'
40 && (name[1] == '\0'
41 || (name[1] == '.' && name[2] == '\0'));
42}

44/*
45 * brix_fs_join_path - concatenate dir + '/' + name into out buffer.
46 *
47 * WHAT: Formats "dir/name" into out[0..outsz-1] via snprintf. Returns NGX_OK if
48 * the result fits, NGX_ERROR on NULL inputs or truncation.
49 *
50 * WHY: Directory traversal builds child paths by joining parent directory with entry
51 * name. This function handles buffer bounds checking uniformly.
52 *
53 * HOW: snprintf(out, outsz, "%s/%s", dir, name). Checks return value < outsz.
54 */
55
56ngx_int_t
57brix_fs_join_path(const char *dir, const char *name, char *out, size_t outsz)
58{
59 int n;
60
61 if (dir == NULL || name == NULL || out == NULL || outsz == 0) {
62 return NGX_ERROR;
63 }
64
65 n = snprintf(out, outsz, "%s/%s", dir, name);
66 return (n >= 0 && (size_t) n < outsz) ? NGX_OK : NGX_ERROR;
67}

69/*
70 * brix_fs_dir_is_empty - check whether a directory contains non-dot entries.
71 *
72 * WHAT: Opens path via opendir(), iterates readdir entries, skips dots, and sets
73 * *is_empty = 1 if no non-dot entries found (or empty dir), 0 otherwise.
74 *
75 * WHY: XRootD stat on collection returns whether it is empty. WebDAV MKCOL checks
76 * target directory emptiness. Single shared check avoids duplication.
77 *
78 * HOW: opendir → readdir loop, skip brix_fs_is_dot_entry(), set flag on first
79 * non-dot entry found. closedir always called. Returns NGX_OK or NGX_ERROR.
80 */
81
82ngx_int_t
83brix_fs_dir_is_empty(const char *path, ngx_flag_t *is_empty)
84{
85 DIR *dp;
86 struct dirent *de;
87
88 if (is_empty == NULL) {
89 return NGX_ERROR;
90 }
91
92 *is_empty = 1;
93 dp = opendir(path);
94 if (dp == NULL) {
95 return NGX_ERROR;
96 }
97
98 while ((de = readdir(dp)) != NULL) {
99 if (brix_fs_is_dot_entry(de->d_name)) {
100 continue;
101 }
102 *is_empty = 0;
103 break;
104 }
105
106 closedir(dp);
107 return NGX_OK;
108}

110/*
111 * brix_fs_walk_action_e - per-entry disposition returned to the walk loop.
112 *
113 * WHAT: Encodes the three outcomes the walk loop must act on after processing a
114 * single directory entry: keep going cleanly, keep going but remember that
115 * an error occurred, or stop the loop and remember the error.
116 *
117 * WHY: Extracting the per-entry body into a helper (to cut cyclomatic complexity)
118 * means the helper cannot itself set the loop's rc or break; it returns an
119 * action code and the loop performs the control flow. This preserves the
120 * original behaviour exactly — a bad join/lstat marks rc but continues, a
121 * failed callback marks rc and breaks.
122 *
123 * HOW: The loop maps CONTINUE→next iteration, MARK_ERROR→rc=NGX_ERROR + continue,
124 * STOP→rc=NGX_ERROR + break.
125 */
126
127typedef enum {
128 BRIX_FS_WALK_CONTINUE = 0, /* skip this entry, no error */
129 BRIX_FS_WALK_MARK_ERROR, /* record error, keep walking */
130 BRIX_FS_WALK_STOP /* record error, stop the loop */
131} brix_fs_walk_action_e;

138/*
139 * brix_fs_walk_skip_name - decide whether an entry name is filtered out.
140 *
141 * WHAT: Returns true when the entry must be skipped before any syscall: it is a
142 * dot entry ("." / ".."), or it is a hidden entry (leading '.') and opts
143 * requests skip_hidden. Returns false otherwise.
144 *
145 * WHY: The walk loop applies the same name filter every iteration; isolating it
146 * keeps the per-entry handler small and keeps the dot/hidden rule in one
147 * place. Matches the original inline checks byte for byte.
148 *
149 * HOW: brix_fs_is_dot_entry() first, then (opts && opts->skip_hidden &&
150 * name[0] == '.'). Pure, no side effects.
151 */
152
153static ngx_flag_t
154brix_fs_walk_skip_name(const brix_fs_walk_options_t *opts, const char *name)
155{
156 if (brix_fs_is_dot_entry(name)) {
157 return 1;
158 }
159
160 return opts != NULL && opts->skip_hidden && name[0] == '.';
161}

163/*
164 * brix_fs_walk_cb_wants - decide whether the callback should fire for an entry.
165 *
166 * WHAT: Returns true when the entry's type matches the include filters: a
167 * directory when opts is NULL or include_dirs is set, a non-directory when
168 * opts is NULL or include_files is set.
169 *
170 * WHY: The include_files/include_dirs predicate is the densest boolean in the
171 * original loop. Naming it removes a compound condition from the handler and
172 * documents the intent (NULL opts means "emit everything").
173 *
174 * HOW: Branch on is_dir, then test the matching include flag (NULL opts ⇒ true).
175 * Equivalent to the original
176 * (is_dir && (opts==NULL||include_dirs)) || (!is_dir && (opts==NULL||include_files)).
177 */
178
179static ngx_flag_t
180brix_fs_walk_cb_wants(const brix_fs_walk_options_t *opts, ngx_flag_t is_dir)
181{
182 if (is_dir) {
183 return opts == NULL || opts->include_dirs;
184 }
185
186 return opts == NULL || opts->include_files;
187}

189/*
190 * brix_fs_walk_should_recurse - decide whether to descend into an entry.
191 *
192 * WHAT: Returns true only for directories that are still within the depth budget:
193 * opts is NULL, or max_depth is 0 (unlimited), or the current depth is below
194 * max_depth. Non-directories never recurse.
195 *
196 * WHY: Keeps the depth-limit rule out of the handler body and documents that a
197 * max_depth of 0 means unlimited. Matches the original recurse guard exactly.
198 *
199 * HOW: Early-return false for non-dirs, then evaluate the depth predicate.
200 */
201
202static ngx_flag_t
203brix_fs_walk_should_recurse(const brix_fs_walk_options_t *opts,
204 ngx_flag_t is_dir, ngx_uint_t depth)
205{
206 if (!is_dir) {
207 return 0;
208 }
209
210 return opts == NULL || opts->max_depth == 0 || depth < opts->max_depth;
211}

213/*
214 * brix_fs_walk_handle_entry - process one directory entry for the walk.
215 *
216 * WHAT: Applies the name filter, joins the child path, lstats it, applies the
217 * stay_on_device filter, fires the callback for matching entries, and
218 * recurses into subdirectories within the depth budget. Returns a
219 * brix_fs_walk_action_e telling the loop what to do next.
220 *
221 * WHY: This is the body extracted from brix_fs_walk_dir()'s readdir loop so the
222 * loop stays trivial and each concern (filter, syscall, callback, recursion)
223 * is a single early-return step. Behaviour is identical to the original:
224 * dot/hidden/cross-device/ENOENT entries are skipped silently; a truncated
225 * join or a non-ENOENT lstat marks an error and continues; a callback that
226 * rejects the entry stops the walk; a failed recursion marks an error and
227 * continues.
228 *
229 * HOW: Sequence of guarded early returns. child[PATH_MAX] and st live on this
230 * frame; entry borrows pointers to them and to de->d_name for the callback.
231 */
232
233static brix_fs_walk_action_e
234brix_fs_walk_handle_entry(ngx_log_t *log, const char *dir,
235 const brix_fs_walk_options_t *opts, brix_fs_walk_pt cb, void *data,
236 ngx_uint_t depth, const struct dirent *de)
237{
238 char child[PATH_MAX];
239 struct stat st;
240 brix_fs_walk_entry_t entry;
241 ngx_flag_t is_dir;
242
243 if (brix_fs_walk_skip_name(opts, de->d_name)) {
244 return BRIX_FS_WALK_CONTINUE;
245 }
246
247 if (brix_fs_join_path(dir, de->d_name, child, sizeof(child)) != NGX_OK) {
248 return BRIX_FS_WALK_MARK_ERROR;
249 }
250
251 if (lstat(child, &st) != 0) {
252 if (errno != ENOENT) {
253 ngx_log_error(NGX_LOG_WARN, log, errno,
254 "brix: lstat failed \"%s\"", child);
255 return BRIX_FS_WALK_MARK_ERROR;
256 }
257 return BRIX_FS_WALK_CONTINUE;
258 }
259
260 if (opts != NULL && opts->stay_on_device && st.st_dev != opts->root_dev) {
261 return BRIX_FS_WALK_CONTINUE;
262 }
263
264 is_dir = S_ISDIR(st.st_mode);
265
266 if (cb != NULL && brix_fs_walk_cb_wants(opts, is_dir)) {
267 entry.path = child;
268 entry.name = de->d_name;
269 entry.st = &st;
270 entry.depth = depth;
271
272 if (cb(&entry, data) != NGX_OK) {
273 return BRIX_FS_WALK_STOP;
274 }
275 }
276
277 if (brix_fs_walk_should_recurse(opts, is_dir, depth)
278 && brix_fs_walk_dir(log, child, opts, cb, data, depth + 1) != NGX_OK)
279 {
280 return BRIX_FS_WALK_MARK_ERROR;
281 }
282
283 return BRIX_FS_WALK_CONTINUE;
284}

286/*
287 * brix_fs_walk_dir - recursive directory walk with callback, depth tracking.
288 *
289 * WHAT: Opens dir via opendir(), iterates entries, and dispatches each to
290 * brix_fs_walk_handle_entry() which does the filtering, lstat, callback and
291 * recursion. Aggregates the per-entry action into the return code.
292 *
293 * WHY: The recursive engine behind brix_fs_walk(). Used by dirlist handlers,
294 * PROPFIND collection walks, and tree removal. Options control filtering
295 * (hidden dirs, files vs dirs only, depth limit, cross-device skip).
296 *
297 * HOW: opendir → readdir loop; the handler returns CONTINUE/MARK_ERROR/STOP and
298 * this function maps those to rc and loop control, then closedir on exit.
299 * Traversal order is readdir order, identical to the original.
300 */
301
302static ngx_int_t
303brix_fs_walk_dir(ngx_log_t *log, const char *dir,
304 const brix_fs_walk_options_t *opts, brix_fs_walk_pt cb, void *data,
305 ngx_uint_t depth)
306{
307 DIR *dp;
308 struct dirent *de;
309 ngx_int_t rc;
310
311 dp = opendir(dir);
312 if (dp == NULL) {
313 ngx_log_error(NGX_LOG_WARN, log, errno,
314 "brix: cannot open directory \"%s\"", dir);
315 return NGX_ERROR;
316 }
317
318 rc = NGX_OK;
319
320 while ((de = readdir(dp)) != NULL) {
321 brix_fs_walk_action_e action;
322
323 action = brix_fs_walk_handle_entry(log, dir, opts, cb, data, depth, de);
324
325 if (action == BRIX_FS_WALK_STOP) {
326 rc = NGX_ERROR;
327 break;
328 }
329
330 if (action == BRIX_FS_WALK_MARK_ERROR) {
331 rc = NGX_ERROR;
332 }
333 }
334
335 closedir(dp);
336 return rc;
337}
