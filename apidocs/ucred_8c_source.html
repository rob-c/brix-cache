1/*
2 * ucred.c — per-user backend-credential selection (phase-1 x509 + phase-2
3 * bearer/.token + phase-3 T3 S3/.s3 + ceph-peruser .keyring). See ucred.h.
4 *
5 * WHAT: Implements the four public functions declared in ucred.h: principal
6 * extraction, filesystem-safe key derivation, single-key resolve, and
7 * identity-to-credential selection with expiry checking.
8 *
9 * WHY: Backends acting on behalf of authenticated users need a per-user
10 * credential (x509 proxy, bearer token, S3 SigV4 triple, or CephX
11 * keyring). Centralising the key-derivation and validation logic
12 * here prevents every backend from reimplementing (and diverging on)
13 * the search and validation semantics.
14 *
15 * HOW: Static pure helpers (charset classifier, PEM expiry check, bearer/
16 * S3/keyring file parsers) keep side effects at the edges. All four
17 * public functions are small and single-purpose; none allocates heap
18 * memory.
19 */

32/* ---- Is `principal` usable verbatim as a credential filename stem? ----
33 * WHAT: 1 iff principal matches [A-Za-z0-9@._][A-Za-z0-9@._-]{0,63}.
34 * WHY: Human-manageable filenames for token subs / S3 access keys; DNs
35 * (which contain '/') always fall through to the hash form.
36 * HOW: 1. reject empty/oversized/leading '-'/leading '.'; 2. scan charset.
37 * Leading '.' is rejected for path-traversal / dotfile safety: "." and
38 * ".." would otherwise reach brix_sd_ucred_resolve as valid literals and
39 * produce paths like <dir>/../.pem — reads outside the credential dir. */
40/* ---- Is one character allowed in a verbatim credential filename stem? ----
41 * WHAT: 1 iff `c` is in [A-Za-z0-9@._-].
42 * WHY: Splitting the per-character charset test out of the scan loop keeps
43 * ucred_principal_fs_safe's cyclomatic complexity within the gate; the
44 * compound OR expression is the sole source of that function's branching.
45 * HOW: Range tests for the alnum classes plus the four literal specials. */
46static int
47ucred_fs_safe_char(char c)
48{
49 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
50 || (c >= '0' && c <= '9') || c == '@' || c == '.'
51 || c == '_' || c == '-';
52}
53
54static int
55ucred_principal_fs_safe(const char *principal)
56{
57 size_t i, len = strlen(principal);
58
59 if (len == 0 || len > 64 || principal[0] == '-' || principal[0] == '.') {
60 return 0;
61 }
62 for (i = 0; i < len; i++) {
63 if (!ucred_fs_safe_char(principal[i])) {
64 return 0;
65 }
66 }
67 return 1;
68}

70/* ---- Classify one candidate credential file ----
71 * WHAT: NGX_OK valid; NGX_DECLINED absent/unreadable/unparseable;
72 * NGX_DECLINED with *expired=1 when parseable but past notAfter.
73 * WHY: Expiry must be checked at the moment of use — a proxy can lapse
74 * between the request open and a deferred flush.
75 * HOW: 1. fopen; 2. PEM_read_X509 (first block = the proxy cert);
76 * 3. X509_cmp_current_time(notAfter) must be > 0. */
77static ngx_int_t
78ucred_check_pem(const char *path, int *expired)
79{
80 FILE *f;
81 X509 *cert;
82 int cmp;
83
84 *expired = 0;
85 f = fopen(path, "r");
86 if (f == NULL) {
87 return NGX_DECLINED;
88 }
89 cert = PEM_read_X509(f, NULL, NULL, NULL);
90 (void) fclose(f); /* read-only stream — nothing buffered to lose */
91 if (cert == NULL) {
92 return NGX_DECLINED; /* unparseable = treated missing */
93 }
94 cmp = X509_cmp_current_time(X509_get0_notAfter(cert));
95 X509_free(cert);
96 if (cmp <= 0) {
97 *expired = 1;
98 return NGX_DECLINED;
99 }
100 return NGX_OK;
101}

103/* ---- Read a bearer token from a .token file --------------------------------
104 * WHAT: Opens `path` (O_RDONLY|O_NOFOLLOW|O_CLOEXEC), reads at most cap-1
105 * bytes, trims at the first CR, LF, space, or NUL, and copies the
106 * result into out_bearer. Returns NGX_OK on success; NGX_DECLINED when
107 * the file is absent/unreadable or the trimmed value is empty.
108 * WHY: Bearer tokens have no X.509 notAfter field — the token's own exp claim
109 * is enforced by the origin server at authentication time. Expiry checking
110 * is therefore the origin's responsibility; we only validate that the file
111 * contains a non-empty string. O_NOFOLLOW prevents a symlink from
112 * escaping the credential directory.
113 * HOW: 1. open with O_RDONLY|O_NOFOLLOW|O_CLOEXEC; absent → DECLINED.
114 * 2. read up to cap-1 bytes into a stack buffer.
115 * 3. Trim at the first whitespace/NUL character.
116 * 4. Reject an empty result → DECLINED; copy into out_bearer → OK. */
117static ngx_int_t
118ucred_read_token(const char *path, char *out_bearer, size_t cap)
119{
120 char buf[BRIX_UCRED_BEARER_MAX];
121 ssize_t n;
122 size_t end;
123 int fd;
124
125 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
126 if (fd < 0) {
127 return NGX_DECLINED;
128 }
129 n = read(fd, buf, cap - 1);
130 close(fd);
131 if (n <= 0) {
132 return NGX_DECLINED;
133 }
134 buf[n] = '\0';
135
136 /* Trim at the first whitespace or embedded NUL — tokens must be a single
137 * word with no embedded spaces, CRs, or LFs. */
138 for (end = 0; end < (size_t) n; end++) {
139 unsigned char c = (unsigned char) buf[end];
140 if (c == '\r' || c == '\n' || c == ' ' || c == '\t' || c == '\0') {
141 break;
142 }
143 }
144 if (end == 0) {
145 OPENSSL_cleanse(buf, (size_t) n); /* scratch held a bearer token */
146 return NGX_DECLINED; /* empty after trimming */
147 }
148 memcpy(out_bearer, buf, end);
149 out_bearer[end] = '\0';
150 OPENSSL_cleanse(buf, (size_t) n); /* wipe the secret from the stack */
151 return NGX_OK;
152}

154/* ---- Trim one line from a read buffer at CR/LF, return its length --------
155 * WHAT: Scans buf[start..end) for the first CR or LF and returns the offset
156 * of that terminator (or `end` if none found before the buffer end).
157 * WHY: Shared by ucred_read_s3's 3-line parse; keeps the line-splitting
158 * logic in one place instead of duplicating a scan-for-newline loop
159 * three times.
160 * HOW: Linear scan; no allocation. */
161static size_t
162ucred_line_end(const char *buf, size_t start, size_t end)
163{
164 size_t i;
165
166 for (i = start; i < end; i++) {
167 if (buf[i] == '\r' || buf[i] == '\n') {
168 return i;
169 }
170 }
171 return end;
172}

174/* ---- Advance past exactly ONE line terminator (\r\n, \n, or \r) ----------
175 * WHAT: Given `pos` pointing at a CR/LF (as returned by ucred_line_end) or at
176 * `end` (no terminator found), returns the offset of the next line's
177 * first byte: pos+2 for "\r\n", pos+1 for a lone "\r" or "\n", or `end`
178 * unchanged when pos already equals end.
179 * WHY: A single-terminator skip (as opposed to skipping ALL consecutive
180 * CR/LF bytes) is required so an intentionally blank line — e.g. an
181 * empty secret-key field in a malformed .s3 file — is preserved as a
182 * zero-length field for the caller to reject, instead of being
183 * silently swallowed together with the next line's terminator.
184 * HOW: Bounds-checked lookahead for the 2-byte CRLF case; else advance 1. */
185static size_t
186ucred_skip_eol(const char *buf, size_t pos, size_t end)
187{
188 if (pos >= end) {
189 return end;
190 }
191 if (buf[pos] == '\r' && pos + 1 < end && buf[pos + 1] == '\n') {
192 return pos + 2;
193 }
194 return pos + 1;
195}

197/* ---- Read an S3 access-key/secret-key/region triple from a .s3 file ------
198 * WHAT: Opens `path` (O_RDONLY|O_NOFOLLOW|O_CLOEXEC), reads a bounded chunk,
199 * and parses up to 3 CR/LF-terminated lines: line 1 = access key id
200 * (required, non-empty), line 2 = secret key (required, non-empty),
201 * line 3 = region (optional; "us-east-1" when absent or empty).
202 * Fills out_ak/out_sk/out_region. Returns NGX_OK on a well-formed
203 * file, NGX_DECLINED when the file is absent/unreadable, either
204 * required line is missing/empty, or any field would overflow its
205 * output buffer.
206 * WHY: S3-backed exports need a per-user SigV4 credential (phase-3 T3),
207 * mirroring the .pem/.token precedence chain with a third, simpler
208 * file format (no ASN.1/JWT parsing — just delimited plaintext).
209 * O_NOFOLLOW prevents a symlink from escaping the credential
210 * directory, matching ucred_read_token's hardening.
211 * HOW: 1. open + bounded read into a stack buffer.
212 * 2. ucred_line_end three times to locate line boundaries.
213 * 3. Reject empty ak or sk → DECLINED.
214 * 4. memcpy each field with an explicit NUL; overflow → DECLINED.
215 * 5. Empty/absent region line → default "us-east-1". */
216static ngx_int_t
217ucred_read_s3(const char *path, char *out_ak, size_t ak_cap,
218 char *out_sk, size_t sk_cap, char *out_region, size_t region_cap)
219{
220 char buf[BRIX_UCRED_S3_AK_MAX + BRIX_UCRED_S3_SK_MAX
221 + BRIX_UCRED_S3_REGION_MAX + 8];
222 ssize_t n;
223 size_t ak_start, ak_end, sk_start, sk_end, rg_start, rg_end, rg_len;
224 int fd;
225
226 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
227 if (fd < 0) {
228 return NGX_DECLINED;
229 }
230 n = read(fd, buf, sizeof(buf) - 1);
231 close(fd);
232 if (n <= 0) {
233 return NGX_DECLINED;
234 }
235 buf[n] = '\0';
236
237 /* Line 1: access key id. */
238 ak_start = 0;
239 ak_end = ucred_line_end(buf, ak_start, (size_t) n);
240 if (ak_end == ak_start || (ak_end - ak_start) >= ak_cap) {
241 OPENSSL_cleanse(buf, sizeof(buf)); /* scratch held the secret key */
242 return NGX_DECLINED;
243 }
244
245 /* Line 2: secret key. Skip exactly the ONE line terminator after line 1
246 * (not all consecutive CR/LF bytes — a blank line must stay a zero-length
247 * field, not be merged into the following line). */
248 sk_start = ucred_skip_eol(buf, ak_end, (size_t) n);
249 sk_end = ucred_line_end(buf, sk_start, (size_t) n);
250 if (sk_end == sk_start || (sk_end - sk_start) >= sk_cap) {
251 OPENSSL_cleanse(buf, sizeof(buf));
252 return NGX_DECLINED;
253 }
254
255 /* Line 3: region (optional). */
256 rg_start = ucred_skip_eol(buf, sk_end, (size_t) n);
257 rg_end = ucred_line_end(buf, rg_start, (size_t) n);
258 rg_len = rg_end - rg_start;
259 if (rg_len >= region_cap) {
260 OPENSSL_cleanse(buf, sizeof(buf));
261 return NGX_DECLINED;
262 }
263
264 memcpy(out_ak, buf + ak_start, ak_end - ak_start);
265 out_ak[ak_end - ak_start] = '\0';
266 memcpy(out_sk, buf + sk_start, sk_end - sk_start);
267 out_sk[sk_end - sk_start] = '\0';
268 if (rg_len == 0) {
269 snprintf(out_region, region_cap, "us-east-1");
270 } else {
271 memcpy(out_region, buf + rg_start, rg_len);
272 out_region[rg_len] = '\0';
273 }
274 OPENSSL_cleanse(buf, sizeof(buf)); /* wipe the secret key from the stack */
275 return NGX_OK;
276}

278/* ---- Read a CephX user id from a `[client.NAME]` keyring section header --
279 * WHAT: Opens `path` (O_RDONLY|O_NOFOLLOW|O_CLOEXEC), reads a bounded chunk,
280 * and scans for the FIRST line of the form "[client.NAME]" (leading/
281 * trailing whitespace around the brackets tolerated). Fills out_path
282 * with `path` itself (the keyring PATH — librados reads the file, we
283 * never parse the key material) and out_user with the bare id NAME
284 * (no "client." prefix). Returns NGX_OK on a well-formed keyring,
285 * NGX_DECLINED when the file is absent/unreadable or no
286 * "[client.NAME]" section header is found.
287 * WHY: A `<key>.keyring` credential (ceph-peruser item) lets the driver
288 * authenticate to RADOS as a specific CephX user instead of the export's
289 * static service credential. librados's rados_create() wants the bare
290 * id (it re-adds "client." itself — see sd_ceph_user_id in sd_ceph.c),
291 * so the bare id is what this parser stores. O_NOFOLLOW prevents a
292 * symlink from escaping the credential directory, matching
293 * ucred_read_token/ucred_read_s3's hardening.
294 * HOW: 1. open + bounded read into a stack buffer.
295 * 2. Line-scan for "[client." ... "]"; extract the id between them.
296 * 3. Reject an empty id or overflow → DECLINED; else fill out_user and
297 * out_path (verbatim `path`) → OK. */
298/* ---- Advance `i` to the start of the next line (past the next '\n') -------
299 * WHAT: Returns the offset of the first byte after the next '\n' at or after
300 * `i`, or `end` when no '\n' remains.
301 * WHY: ucred_keyring_parse_line uses this to resume scanning after a line
302 * that is not a usable "[client.NAME]" header; factoring it out removes
303 * the three duplicated scan-to-newline loops that inflated the caller's
304 * cyclomatic complexity.
305 * HOW: Linear scan to the terminator; return terminator+1 (bounded by end). */
306static size_t
307ucred_keyring_next_line(const char *buf, size_t i, size_t end)
308{
309 while (i < end && buf[i] != '\n') {
310 i++;
311 }
312 if (i < end) {
313 i++; /* step past the '\n' */
314 }
315 return i;
316}

318/* ---- Parse one keyring line for a "[client.NAME]" section header ----------
319 * WHAT: Examines the line beginning at `line`. On a well-formed
320 * "[client.NAME]" header, sets out->id_start/out->id_len to the bare
321 * NAME span and returns NGX_OK. On any other line returns NGX_DECLINED
322 * and sets out->next to the start of the following line so the caller
323 * keeps scanning.
324 * WHY: Isolates the header-recognition branching (whitespace skip, bracket,
325 * "client." prefix, closing bracket) from ucred_read_keyring's loop,
326 * collapsing that loop to a single dispatch on this helper's result.
327 * HOW: 1. skip leading spaces/tabs; 2. require '['; 3. require "client.";
328 * 4. locate the closing ']' before EOL; on any miss, compute out->next
329 * via ucred_keyring_next_line and return NGX_DECLINED. */
330typedef struct {
331 size_t id_start; /* offset of the bare NAME span (valid iff NGX_OK) */
332 size_t id_len; /* length of the bare NAME span (valid iff NGX_OK) */
333 size_t next; /* start of the next line (valid iff NGX_DECLINED) */
334} ucred_keyring_line_t;
335
336static ngx_int_t
337ucred_keyring_parse_line(const char *buf, size_t line, size_t end,
338 ucred_keyring_line_t *out)
339{
340 size_t i = line;
341 size_t j, start, stop;
342
343 out->id_start = 0;
344 out->id_len = 0;
345 out->next = end;
346
347 while (i < end && (buf[i] == ' ' || buf[i] == '\t')) {
348 i++;
349 }
350 if (i >= end || buf[i] != '[') {
351 out->next = ucred_keyring_next_line(buf, i, end);
352 return NGX_DECLINED;
353 }
354 j = i + 1;
355 if (j + 7 > end || strncmp(buf + j, "client.", 7) != 0) {
356 out->next = ucred_keyring_next_line(buf, i, end);
357 return NGX_DECLINED;
358 }
359 start = j + 7;
360 stop = start;
361 while (stop < end && buf[stop] != ']' && buf[stop] != '\n') {
362 stop++;
363 }
364 if (stop >= end || buf[stop] != ']') {
365 out->next = ucred_keyring_next_line(buf, i, end);
366 return NGX_DECLINED;
367 }
368 out->id_start = start;
369 out->id_len = stop - start;
370 return NGX_OK;
371}

373static ngx_int_t
374ucred_read_keyring(const char *path, char *out_path, size_t path_cap,
375 char *out_user, size_t user_cap)
376{
377 char buf[4096];
378 ssize_t n;
379 size_t i, end;
380 int fd;
381
382 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
383 if (fd < 0) {
384 return NGX_DECLINED;
385 }
386 n = read(fd, buf, sizeof(buf) - 1);
387 close(fd);
388 if (n <= 0) {
389 return NGX_DECLINED;
390 }
391 buf[n] = '\0';
392 end = (size_t) n;
393
394 i = 0;
395 while (i < end) {
396 ucred_keyring_line_t line = { 0, 0, end };
397
398 if (ucred_keyring_parse_line(buf, i, end, &line) != NGX_OK) {
399 i = line.next;
400 continue;
401 }
402
403 if (line.id_len == 0 || line.id_len >= user_cap) {
404 OPENSSL_cleanse(buf, sizeof(buf)); /* scratch held CephX key material */
405 return NGX_DECLINED; /* empty or oversized id: malformed */
406 }
407 memcpy(out_user, buf + line.id_start, line.id_len);
408 out_user[line.id_len] = '\0';
409
410 if (snprintf(out_path, path_cap, "%s", path) >= (int) path_cap) {
411 OPENSSL_cleanse(buf, sizeof(buf));
412 return NGX_DECLINED;
413 }
414 OPENSSL_cleanse(buf, sizeof(buf)); /* wipe key material from the stack */
415 return NGX_OK;
416 }
417
418 OPENSSL_cleanse(buf, sizeof(buf));
419 return NGX_DECLINED; /* no "[client.NAME]" section header found */
420}

422/*
423 * brix_sd_ucred_principal — extract canonical principal string from identity.
424 *
425 * WHAT: Copies id->dn (if non-empty) else id->subject into buf as a
426 * NUL-terminated C string.
427 * WHY: DN is the richer identifier; subject (JWT sub / S3 key) is the
428 * fallback when no DN is present.
429 * HOW: Reject unauthenticated / both-empty / overflow; memcpy + NUL.
430 */
431ngx_int_t
432brix_sd_ucred_principal(const brix_identity_t *id, char *buf, size_t cap)
433{
434 ngx_str_t src;
435
436 if (id == NULL || !id->is_authenticated) {
437 return NGX_ERROR;
438 }
439 if (id->dn.len > 0) {
440 src = id->dn;
441 } else if (id->subject.len > 0) {
442 src = id->subject;
443 } else {
444 return NGX_ERROR;
445 }
446 if (src.len >= cap) {
447 return NGX_ERROR;
448 }
449 memcpy(buf, src.data, src.len);
450 buf[src.len] = '\0';
451 return NGX_OK;
452}

454/*
455 * brix_sd_ucred_key — derive a filesystem-safe filename stem from a principal.
456 *
457 * WHAT: Literal verbatim when fs-safe; "x5h-" + first 32 hex chars of
458 * SHA256(principal) otherwise.
459 * WHY: DNs contain '/' and other shell-hostile chars; the hash form gives
460 * a stable, collision-resistant, admin-provisionable name.
461 * HOW: Classify; literal snprintf or SHA256 + hex-encode first 16 bytes.
462 */
463ngx_int_t
464brix_sd_ucred_key(const char *principal, char *key, size_t cap)
465{
466 unsigned char digest[SHA256_DIGEST_LENGTH];
467 int n;
468
469 if (principal == NULL || principal[0] == '\0') {
470 return NGX_ERROR;
471 }
472 if (ucred_principal_fs_safe(principal)) {
473 n = snprintf(key, cap, "%s", principal);
474 if (n < 0 || (size_t) n >= cap) {
475 return NGX_ERROR;
476 }
477 return NGX_OK;
478 }
479 SHA256((const unsigned char *) principal, strlen(principal), digest);
480 n = snprintf(key, cap,
481 "x5h-%02x%02x%02x%02x%02x%02x%02x%02x"
482 "%02x%02x%02x%02x%02x%02x%02x%02x",
483 digest[0], digest[1], digest[2], digest[3],
484 digest[4], digest[5], digest[6], digest[7],
485 digest[8], digest[9], digest[10], digest[11],
486 digest[12], digest[13], digest[14], digest[15]);
487 if (n < 0 || (size_t) n >= cap) {
488 return NGX_ERROR;
489 }
490 return NGX_OK;
491}

493/*
494 * brix_sd_ucred_resolve — look up a credential file by its exact key.
495 *
496 * WHAT: Tries <dir>/<key>.pem (x509, expiry-checked) then, only when the .pem
497 * is ABSENT, <dir>/<key>.token (bearer, no expiry check). Fills *out on
498 * success (is_bearer=0 for x509, is_bearer=1 for bearer).
499 * WHY: Flush/write paths already know the key and need a fresh expiry check
500 * without re-running principal derivation. An expired .pem is a hard
501 * DECLINED — the .token file is NOT tried as a fallback (the operator
502 * must fix the proxy; silent promotion to bearer would change identity).
503 * HOW: 1. snprintf .pem path; overflow → NGX_ERROR.
504 * 2. ucred_check_pem: NGX_OK → x509 path, return OK.
505 * expired (expired=1) → set out->expired, return DECLINED immediately.
506 * absent (expired=0) → fall through to .token probe.
507 * 3. ucred_read_token on .token path: NGX_OK → bearer path, return OK;
508 * else → return DECLINED.
509 */
510ngx_int_t
511brix_sd_ucred_resolve(const char *dir, const char *key, brix_sd_ucred_t *out)
512{
513 char pem_path[BRIX_UCRED_PATH_MAX];
514 char tok_path[BRIX_UCRED_PATH_MAX];
515 char s3_path[BRIX_UCRED_PATH_MAX];
516 char keyring_path[BRIX_UCRED_PATH_MAX];
517 int n, expired;
518 ngx_int_t pem_rc;
519
520 n = snprintf(pem_path, sizeof(pem_path), "%s/%s.pem", dir, key);
521 if (n < 0 || (size_t) n >= sizeof(pem_path)) {
522 errno = ENAMETOOLONG;
523 return NGX_ERROR;
524 }
525
526 pem_rc = ucred_check_pem(pem_path, &expired);
527 if (pem_rc == NGX_OK) {
528 snprintf(out->path, sizeof(out->path), "%s", pem_path);
529 snprintf(out->key, sizeof(out->key), "%s", key);
530 out->expired = 0;
531 out->is_bearer = 0;
532 out->is_s3 = 0;
533 out->is_ceph = 0;
534 out->bearer[0] = '\0';
535 return NGX_OK;
536 }
537
538 /* Hard stop on an expired .pem: never fall through to .token or .s3.
539 * The operator must renew the proxy; silently promoting to bearer/s3
540 * would change the presented identity without the admin's intent. */
541 if (expired) {
542 out->expired = 1;
543 return NGX_DECLINED;
544 }
545
546 /* .pem is absent (not expired) — probe the .token file. */
547 n = snprintf(tok_path, sizeof(tok_path), "%s/%s.token", dir, key);
548 if (n < 0 || (size_t) n >= sizeof(tok_path)) {
549 errno = ENAMETOOLONG;
550 return NGX_ERROR;
551 }
552
553 if (ucred_read_token(tok_path, out->bearer, sizeof(out->bearer)) == NGX_OK) {
554 snprintf(out->path, sizeof(out->path), "%s", tok_path);
555 snprintf(out->key, sizeof(out->key), "%s", key);
556 out->expired = 0;
557 out->is_bearer = 1;
558 out->is_s3 = 0;
559 out->is_ceph = 0;
560 return NGX_OK;
561 }

580 }

604/*
605 * brix_sd_ucred_select — map an identity to its best available credential.
606 *
607 * WHAT: Tries literal-key then hash-key candidate; first valid+unexpired PEM
608 * wins (NGX_OK). On all-missed returns NGX_DECLINED with out->key set
609 * to the hash-form key so the caller can log the file to provision.
610 * WHY: Single entry point for all per-user credential lookups; hides the
611 * two-candidate search and expiry-OR logic from callers.
612 * HOW: Zero *out; derive principal; build up to 2 candidates; resolve each;
613 * return first OK or DECLINED with out->key = hash key.
614 */
615ngx_int_t
616brix_sd_ucred_select(const char *dir, const brix_identity_t *id,
617 brix_sd_ucred_t *out)
618{
619 char principal[BRIX_UCRED_PRINC_MAX];
620 char lit_key[BRIX_UCRED_KEY_MAX];
621 char hash_key[BRIX_UCRED_KEY_MAX];
622 int has_lit;
623 int any_expired;
624 ngx_int_t rc;
625
626 memset(out, 0, sizeof(*out));
627
628 if (brix_sd_ucred_principal(id, principal, sizeof(principal)) != NGX_OK) {
629 return NGX_DECLINED;
630 }
631 snprintf(out->principal, sizeof(out->principal), "%s", principal);
632
633 /* Derive the hash key unconditionally (always needed for the fallback). */
634 if (brix_sd_ucred_key(principal, hash_key, sizeof(hash_key)) != NGX_OK) {
635 return NGX_DECLINED;
636 }
637
638 /* Derive the literal key; flag whether it is usable.
639 * ucred_principal_fs_safe guarantees len <= 64 < BRIX_UCRED_KEY_MAX; use
640 * memcpy+NUL so the compiler sees a bounded copy (avoids -Wformat-truncation
641 * from comparing the declared buf sizes rather than the runtime constraint). */
642 has_lit = ucred_principal_fs_safe(principal);
643 if (has_lit) {
644 size_t plen = strlen(principal);
645 memcpy(lit_key, principal, plen);
646 lit_key[plen] = '\0';
647 }
648
649 any_expired = 0;
650
651 /* Try the literal candidate first (only when fs-safe). */
652 if (has_lit) {
653 rc = brix_sd_ucred_resolve(dir, lit_key, out);
654 if (rc == NGX_OK) {
655 /* out->principal already set above; no re-copy needed. */
656 return NGX_OK;
657 }
658 any_expired |= out->expired;
659 }
660
661 /* Try the hash candidate. */
662 rc = brix_sd_ucred_resolve(dir, hash_key, out);
663 if (rc == NGX_OK) {
664 /* out->principal already set above; no re-copy needed. */
665 return NGX_OK;
666 }
667 any_expired |= out->expired;
668
669 /* No valid credential found — return DECLINED with the hash key for
670 * logging so the operator knows which file to provision. */
671 snprintf(out->principal, sizeof(out->principal), "%s", principal);
672 snprintf(out->key, sizeof(out->key), "%s", hash_key);
673 out->expired = any_expired;
674 return NGX_DECLINED;
675}
