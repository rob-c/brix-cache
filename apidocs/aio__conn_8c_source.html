24/* Fail every in-flight request with `st`, leaving the map empty. */
25void
26aconn_drain_inflight(brix_aconn *ac, const brix_status *st)
27{
28 for (uint32_t i = 0; i < ac->inflight.cap; i++) {
29 brix_areq *r = ac->inflight.slots[i];
30 if (r == NULL || r == REQMAP_TOMB) {
31 continue;
32 }
33 ac->inflight.slots[i] = NULL;
34 areq_complete(r, -1, (uint16_t) (st->kxr & 0xffff), st);
35 }
36 ac->inflight.count = 0;
37 ac->inflight.tomb = 0;
38}

41/* reconnect *//*
42 * On a transport drop the connection does not fail its callers outright; it goes
43 * RECONNECTING: a worker thread re-establishes the session (off the loop thread so
44 * other connections keep flowing), retry-safe in-flight requests are parked, and
45 * once the new socket is back they are re-issued — so a stat/ping/dirlist that was
46 * mid-flight when the server bounced simply succeeds, transparently. Non-retry-safe
47 * (mutating or handle-bound) requests fail so the higher layer (M3) can reopen.
48 */
49
50/* Fail every parked request with `st`, emptying the pending list. */
51void
52aconn_pending_fail_all(brix_aconn *ac, const brix_status *st)
53{
54 brix_areq *p = ac->pending;
55 ac->pending = NULL;
56 while (p != NULL) {
57 brix_areq *nx = p->pend_next;
58 p->pend_next = NULL;
59 areq_complete(p, -1, (uint16_t) (st->kxr & 0xffff), st);
60 p = nx;
61 }
62}

65/* Enter RECONNECTING: pull the dead fd from epoll, partition in-flight requests
66 * (retry-safe → parked, others → failed), and spawn the reconnect worker. */
67void
68aconn_on_transport_error(brix_aconn *ac, const brix_status *st)
69{
70 if (ac->state != ACONN_ALIVE) {
71 return; /* already reconnecting or dead */
72 }
73 io_engine_del(ac->loop, ac); /* epoll DEL or io_uring poll cancel */
74 ac->dead = 1;
75
76 if (ac->max_stall_ms <= 0) { /* reconnect disabled — fail outright */
77 ac->state = ACONN_DEAD;
78 aconn_drain_inflight(ac, st);
79 aconn_pending_fail_all(ac, st);
80 return;
81 }
82
83 uint64_t now = brix_mono_ns();
84 ac->state = ACONN_RECONNECTING;
85 ac->reconnect_deadline_ns = now + (uint64_t) ac->max_stall_ms * 1000000ULL;
86 ac->ping_inflight = 0;
87 ac->tls_want_read_on_write = ac->tls_want_write_on_read = 0;
88
89 for (uint32_t i = 0; i < ac->inflight.cap; i++) {
90 brix_areq *r = ac->inflight.slots[i];
91 if (r == NULL || r == REQMAP_TOMB) {
92 continue;
93 }
94 ac->inflight.slots[i] = NULL;
95 if (r->retry_safe && r->retries_left > 0) {
96 r->retries_left--;
97 r->deadline_ns = ac->reconnect_deadline_ns; /* patience through reconnect */
98 r->pend_next = ac->pending;
99 ac->pending = r;
100 } else {
101 areq_complete(r, -1, (uint16_t) (st->kxr & 0xffff), st);
102 }
103 }
104 ac->inflight.count = 0;
105 ac->inflight.tomb = 0;
106 ac->wbuf.start = ac->wbuf.len = 0; /* stale: belonged to the dead socket */
107 ac->rbuf.start = ac->rbuf.len = 0;
108
109 if (pthread_create(&ac->rc_thread, NULL, rc_worker_main, ac) == 0) {
110 ac->rc_thread_live = 1;
111 } else {
112 ac->state = ACONN_DEAD;
113 aconn_pending_fail_all(ac, st);
114 }
115}

118/* Loop-thread side of a successful reconnect: adopt the new socket and re-issue
119 * every parked request. */
120void
121aconn_reconnect_succeeded(brix_aconn *ac)
122{
123 ac->fd = ac->conn->io.fd;
124 ac->ssl = ac->conn->io.ssl;
125
126 int fl = fcntl(ac->fd, F_GETFL, 0);
127 if (fl >= 0) {
128 fcntl(ac->fd, F_SETFL, fl | O_NONBLOCK);
129 }
130
131 if (io_engine_arm(ac->loop, ac, EPOLLIN) != 0) {
132 brix_status st;
133 brix_status_set(&st, XRDC_ESOCK, errno, "poll re-register failed");
134 ac->state = ACONN_DEAD;
135 aconn_pending_fail_all(ac, &st);
136 return;
137 }
138 ac->dead = 0;
139 ac->state = ACONN_ALIVE;
140 ac->last_activity_ns = brix_mono_ns();
141 ac->reconnect_deadline_ns = 0;
142
143 brix_areq *p = ac->pending;
144 ac->pending = NULL;
145 while (p != NULL) {
146 brix_areq *nx = p->pend_next;
147 p->pend_next = NULL;
148 aconn_issue_areq(ac, p);
149 if (ac->state != ACONN_ALIVE) {
150 /* a re-issue tripped another drop; the rest were re-parked or failed */
151 p = ac->pending;
152 ac->pending = NULL;
153 continue;
154 }
155 p = nx;
156 }
157}

160/* Detect a finished reconnect worker (called each tick) and apply its result. */
161void
162aconn_poll_reconnect(brix_aconn *ac)
163{
164 if (ac->state != ACONN_RECONNECTING) {
165 return;
166 }
167 if (!__atomic_load_n(&ac->rc_finished, __ATOMIC_ACQUIRE)) {
168 return;
169 }
170 if (ac->rc_thread_live) {
171 pthread_join(ac->rc_thread, NULL);
172 ac->rc_thread_live = 0;
173 }
174 ac->rc_finished = 0;
175 if (ac->rc_result == 0) {
176 aconn_reconnect_succeeded(ac);
177 } else {
178 ac->state = ACONN_DEAD;
179 aconn_pending_fail_all(ac, &ac->rc_st);
180 }
181}

198/* Send a kXR_ping if the link has been idle past the keepalive threshold. */
199void
200aconn_maybe_ping(brix_aconn *ac)
201{
202 if (ac->state != ACONN_ALIVE || ac->keepalive_ms <= 0 || ac->ping_inflight) {
203 return;
204 }
205 uint64_t now = brix_mono_ns();
206 if (now - ac->last_activity_ns < (uint64_t) ac->keepalive_ms * 1000000ULL) {
207 return;
208 }
209 brix_areq *r = (brix_areq *) calloc(1, sizeof(*r));
210 if (r == NULL) {
211 return;
212 }
213 uint16_t rid = htons(kXR_ping);
214 memcpy(r->hdr + 2, &rid, 2);
215 r->cb = aconn_ping_cb;
216 r->ctx = ac;
217 r->is_ping = 1;
218 r->retry_safe = 1;
219 r->retries_left = ac->def_retries;
220 uint64_t dl = aconn_rto_ns(ac) * 4;
221 if (dl < 2000000000ULL) { dl = 2000000000ULL; }
222 if (dl > 10000000000ULL) { dl = 10000000000ULL; }
223 r->deadline_ns = now + dl;
224 ac->ping_inflight = 1;
225 aconn_issue_areq(ac, r);
226}

229/* Unlink from the loop list, free buffers + struct (loop thread only). Joins a
230 * running reconnect worker first so conn is no longer touched by another thread. */
231void
232aconn_destroy(brix_aconn *ac)
233{
234 brix_loop *l = ac->loop;
235 brix_status st;
236 brix_status_set(&st, XRDC_ESOCK, 0, "connection closed");
237
238 if (ac->rc_thread_live) {
239 pthread_join(ac->rc_thread, NULL); /* bounded by the connect timeout */
240 ac->rc_thread_live = 0;
241 }
242 if (!ac->dead) {
243 io_engine_del(l, ac);
244 aconn_drain_inflight(ac, &st);
245 }
246 aconn_pending_fail_all(ac, &st); /* fail any parked requests */
247 brix_aconn **pp = &l->aconns;
248 while (*pp != NULL) {
249 if (*pp == ac) {
250 *pp = ac->next;
251 break;
252 }
253 pp = &(*pp)->next;
254 }
255 xbuf_free(&ac->wbuf);
256 xbuf_free(&ac->rbuf);
257 free(ac->inflight.slots);
258 free(ac);
259}

262/* Assign a fresh, currently-unused streamid (skips 0 and 0xffff). */
263uint16_t
264aconn_alloc_sid(brix_aconn *ac)
265{
266 for (int tries = 0; tries < 65536; tries++) {
267 uint16_t s = ac->next_sid++;
268 if (ac->next_sid == 0xffff) {
269 ac->next_sid = 1;
270 }
271 if (s == 0 || s == 0xffff) {
272 continue;
273 }
274 if (reqmap_get(&ac->inflight, s) == NULL) {
275 return s;
276 }
277 }
278 return 0; /* pipeline saturated (should never happen with bounded depth) */
279}

282/* Stamp a fresh streamid + dlen, queue the bytes on the wire, and register the
283 * request as in-flight. Used for a new submission and for re-issuing a parked
284 * request after a reconnect (the header's requestid + body are preserved; only
285 * streamid/dlen are re-stamped). */
286void
287aconn_issue_areq(brix_aconn *ac, brix_areq *r)
288{
289 uint16_t sid = aconn_alloc_sid(ac);
290 r->sid = sid;
291 r->hdr[0] = (uint8_t) (sid >> 8);
292 r->hdr[1] = (uint8_t) (sid & 0xff);
293 uint32_t be = htonl(r->plen);
294 memcpy(r->hdr + 20, &be, 4);
295 r->submit_ns = brix_mono_ns();
296
297 if (xbuf_append(&ac->wbuf, r->hdr, XRD_REQUEST_HDR_LEN) != 0 ||
298 (r->plen > 0 && xbuf_append(&ac->wbuf, r->payload, r->plen) != 0) ||
299 reqmap_put(&ac->inflight, r) != 0) {
300 brix_status st;
301 brix_status_set(&st, XRDC_EPROTO, 0, "out of memory (queue)");
302 areq_complete(r, -1, XRDC_EPROTO, &st);
303 return;
304 }
305
306 aconn_do_write(ac);
307 if (!ac->dead) {
308 aconn_update_epoll(ac);
309 }
310}

313/* Compute a request's hard deadline: the caller's explicit ms, else an adaptive
314 * value derived from the RTO and bounded by the reconnect patience budget. */
315uint64_t
316aconn_deadline_ns(brix_aconn *ac, int deadline_ms)
317{
318 uint64_t now = brix_mono_ns();
319 uint64_t ms;
320 if (deadline_ms > 0) {
321 ms = (uint64_t) deadline_ms;
322 } else {
323 uint64_t cap = (ac->max_stall_ms > 0) ? (uint64_t) ac->max_stall_ms : 60000;
324 ms = (aconn_rto_ns(ac) / 1000000ULL) * 20; /* ~20 RTOs of patience */
325 if (ms < 5000) {
326 ms = 5000;
327 }
328 if (ms > cap) {
329 ms = cap;
330 }
331 }
332 return now + ms * 1000000ULL;
333}

336/* Turn a SUBMIT command into a request. If the socket is mid-reconnect the request
337 * is parked (it has not been sent, so it is safe to issue fresh once the socket is
338 * back); otherwise it goes out immediately. Consumes cmd->payload ownership. */
339void
340aconn_submit_cmd(brix_aconn *ac, cmd *c)
341{
342 if (ac->state == ACONN_DEAD) {
343 brix_status st;
344 brix_status_set(&st, XRDC_ESOCK, 0, "connection is down");
345 c->cb(c->ctx, -1, XRDC_ESOCK, NULL, 0, &st);
346 free(c->payload);
347 return;
348 }
349
350 brix_areq *r = (brix_areq *) calloc(1, sizeof(*r));
351 if (r == NULL) {
352 brix_status st;
353 brix_status_set(&st, XRDC_EPROTO, 0, "out of memory (request)");
354 c->cb(c->ctx, -1, XRDC_EPROTO, NULL, 0, &st);
355 free(c->payload);
356 return;
357 }
358
359 memcpy(r->hdr, c->hdr, XRD_REQUEST_HDR_LEN);
360 r->payload = c->payload; /* move */
361 r->plen = c->plen;
362 r->cb = c->cb;
363 r->ctx = c->ctx;
364 r->retry_safe = c->retry_safe;
365 r->retries_left = (c->max_retries >= 0) ? c->max_retries : ac->def_retries;
366 r->deadline_ns = aconn_deadline_ns(ac, c->deadline_ms);
367
368 if (ac->state == ACONN_RECONNECTING) {
369 /* Give a parked request the full reconnect budget so it survives the drop. */
370 uint64_t stall = (ac->max_stall_ms > 0) ? (uint64_t) ac->max_stall_ms : 60000;
371 r->deadline_ns = brix_mono_ns() + stall * 1000000ULL;
372 r->pend_next = ac->pending;
373 ac->pending = r;
374 return;
375 }
376
377 aconn_issue_areq(ac, r);
378}
