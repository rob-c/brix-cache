8/* HEAD to get object size */
9/*
10 * s3_load_size — issue a SigV4-signed HEAD and populate sf->obj_size.
11 *
12 * WHAT: sends HEAD /key, parses the Content-Length response header, stores the
13 * result in sf->obj_size (or -1 if Content-Length is absent).
14 * WHY: fstat() on a read handle needs the object size; the HEAD is deferred
15 * to the first fstat call (lazy) to avoid a round-trip on write handles.
16 * HOW: s3_sign + brix_http_req; parse Content-Length with brix_http_header.
17 */
18int
19s3_load_size(vfs_s3_file *sf, brix_status *st)
20{
21 char errbuf[256] = "";
22
23 /* Delegate to the shared S3 driver (HEAD-size lives in src/fs/backend/sd_s3.c). */
24 if (sd_s3_size(sf->sd, &sf->obj_size, errbuf, sizeof(errbuf)) != 0) {
25 brix_status_set(st, XRDC_EIO, 0, "%s", errbuf);
26 return -1;
27 }
28 return 0;
29}

32/* vtable: pread */
33/*
34 * s3_pread — read n bytes at offset off from the S3 object into buf.
35 *
36 * WHAT: issues a SigV4-signed GET with a "Range: bytes=off-end\r\n" header;
37 * copies the 206 (or 200) response body into buf.
38 * WHY: VFS callers need seekable reads; S3 supports this via HTTP Range.
39 * HOW: cap n at S3_PREAD_MAX to stay within brix_http_req's 8 MiB body limit;
40 * build combined auth+range extra_headers string; issue GET; check status;
41 * memcpy body to buf; return bytes copied.
42 */
43ssize_t
44s3_pread(brix_vfs_file *f, int64_t off, void *buf, size_t n, brix_status *st)
45{
46 vfs_s3_file *sf = (vfs_s3_file *) f;
47 char errbuf[256] = "";
48 ssize_t r;
49
50 /* Delegate to the shared S3 driver (Range-GET lives in src/fs/backend/sd_s3.c). */
51 r = sd_s3_pread(sf->sd, buf, n, (off_t) off, errbuf, sizeof(errbuf));
52 if (r < 0) {
53 brix_status_set(st, XRDC_EIO, 0, "%s", errbuf);
54 }
55 return r;
56}

63/*
64 * s3_pwrite — write n bytes from buf at offset off.
65 *
66 * WHAT: dispatcher that routes to s3_pwrite_single or s3_pwrite_mpu based on
67 * the handle's write mode (sf->is_mpu).
68 * WHY: the vtable interface is uniform; mode dispatch is one place.
69 * HOW: cast f to vfs_s3_file; check is_write; delegate.
70 */
71int
72s3_pwrite(brix_vfs_file *f, int64_t off, const void *buf, size_t n,
73 brix_status *st)
74{
75 vfs_s3_file *sf = (vfs_s3_file *) f;
76 char errbuf[256] = "";
77
78 /* Delegate to the shared S3 driver (single-PUT buffer / MPU part flush).
79 * The driver sets errno at its error sites (EINVAL for a non-sequential
80 * write, EACCES for an auth failure on a part flush); reset it first so a
81 * stale value can't leak, then map it to the matching XRDC_* code. */
82 errno = 0;
83 if (sd_s3_pwrite(sf->sd, buf, n, (off_t) off, errbuf, sizeof(errbuf)) != 0) {
84 int e = errno;
85 brix_status_set(st, s3_brix_from_errno(e), e, "%s", errbuf);
86 return -1;
87 }
88 return 0;
89}

92/* vtable: fstat */
93/*
94 * s3_fstat — fill *out with size metadata for the open handle.
95 *
96 * WHAT: for READ handles, issues a lazy HEAD if size not yet loaded; for WRITE
97 * handles returns the accumulated byte count as the current size.
98 * WHY: copy.c may call fstat after open to get the object size for progress.
99 * HOW: READ: lazy s3_load_size; WRITE single-PUT: put_len; WRITE MPU:
100 * mpu_write_off (total bytes written so far).
101 */
102int
103s3_fstat(brix_vfs_file *f, brix_vfs_stat *out, brix_status *st)
104{
105 vfs_s3_file *sf = (vfs_s3_file *) f;
106
107 out->is_dir = 0;
108 out->mtime = 0;
109 out->exists = 1;
110
111 if (sf->is_write) {
112 out->size = sd_s3_write_size(sf->sd);
113 return 0;
114 }
115 /* READ handle: load size lazily on first fstat. */
116 if (sf->obj_size < 0) {
117 if (s3_load_size(sf, st) != 0) {
118 return -1;
119 }
120 }
121 out->size = sf->obj_size;
122 return 0;
123}
