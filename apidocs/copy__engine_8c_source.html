1/*
2 * WHAT: WebDAV local COPY helpers (not HTTP-TPC remote transfers).
3 * webdav_copy_file / webdav_copy_dir_recursive copy a confined file or
4 * directory tree on the local export, preserving metadata.
5 *
6 * WHY: WebDAV COPY/MOVE on local destinations needs a confined, metadata-
7 * preserving copy. The confined byte copy (copy_file_range + fallback), the
8 * confined open/opendir/mkdir, fattr preservation, and the recursion now
9 * live in the VFS (brix_vfs_copyfile / brix_vfs_copytree,
10 * src/fs/vfs/vfs_walk.c) so this protocol layer never reaches a confined helper
11 * directly — and the same engine is reusable by any future copy consumer.
12 *
13 * HOW: Both functions are thin wrappers that delegate to the VFS copy primitives
14 * with preserve_xattrs=1 (the generic user.* fattrs) plus a metadata
15 * callback (webdav_copy_meta_cb) that carries the WebDAV-specific dead-
16 * property xattrs — the one bit the protocol-agnostic VFS must not know.
17 */
18
19#include "copy_engine.h"
20#include "fs/vfs/vfs.h" /* brix_vfs_copyfile / brix_vfs_copytree */

22/*
23 * webdav_copy_meta_cb — per-entry metadata callback for the VFS copy primitives.
24 * The generic user.* fattrs are preserved by the VFS itself (preserve_xattrs=1);
25 * this carries the WebDAV-specific dead-property xattrs that the VFS layer must
26 * not know about. cookie is the request log.
27 */
28static ngx_int_t
29webdav_copy_meta_cb(void *cookie, const char *src, const char *dst, int is_dir)
30{
31 (void) is_dir;
32 webdav_dead_props_copy((ngx_log_t *) cookie, src, dst);
33 return NGX_OK;
34}

36/*
37 * WebDAV local COPY now delegates the confined copy + traversal to the VFS
38 * (brix_vfs_copyfile / brix_vfs_copytree, src/fs/vfs/vfs_walk.c) — which owns the
39 * copy_file_range/fallback, the confined open/opendir/mkdir, fattr preservation,
40 * and the recursion. These thin wrappers add only the WebDAV dead-property copy
41 * via the metadata callback, keeping the protocol-specific bit out of the VFS.
42 */
