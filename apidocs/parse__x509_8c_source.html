17 * connection so a later kXGS_pxyreq/kXGC_sigpxy delegation round (phase-57 §F6)

65/*
66 * gsi_capture_fullproxy — stash an OPTIONAL client-pushed full proxy PEM

99{

102 BIO *bio;
103 X509 *cert;
104 STACK_OF(X509) *chain;
105
106 if (gsi_find_bucket(plain, (size_t) plain_len, (uint32_t) kXRS_x509,
107 &x509_data, &x509_len) != 0) {
108 ngx_log_error(NGX_LOG_WARN, log, 0,
109 "brix: GSI kXGC_cert: kXRS_x509 not found "
110 "in decrypted inner buffer");
111 return NULL;
112 }
113
114 bio = BIO_new_mem_buf(x509_data, (int) x509_len);
115 chain = sk_X509_new_null();
116 if (!bio || !chain) {
117 BIO_free(bio);
118 sk_X509_free(chain);
119 return NULL;
120 }
121 while ((cert = PEM_read_bio_X509(bio, NULL, NULL, NULL)) != NULL) {
122 sk_X509_push(chain, cert);
123 }
124 BIO_free(bio);
125
126 if (sk_X509_num(chain) == 0) {
127 ngx_log_error(NGX_LOG_WARN, log, 0,
128 "brix: GSI kXGC_cert: kXRS_x509 contained no certs");
129 sk_X509_pop_free(chain, X509_free);
130 return NULL;
131 }
132 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, log, 0,
133 "brix: GSI parsed %d cert(s) from kXRS_x509 after decrypt",
134 sk_X509_num(chain));
135 return chain;
136}

138/*
139 * gsi_store_signing_key — WHAT: derive signing_key = SHA-256(secret) and, on
140 * success, install it into ctx->sigver and set signing_active. WHY: both the
141 * unsigned and signed-DH kXGC_cert paths derive the kXR_sigver HMAC key the
142 * SAME way (SHA-256 over the raw DH shared secret) — this is the single point of
143 * truth so the two paths stay byte-identical. HOW: pure digest over `secret`;
144 * only mutates ctx when all three EVP digest steps succeed (fail-quiet, matching
145 * prior behaviour where a digest failure simply left signing inactive). Returns
146 * 1 if signing was activated, 0 otherwise (caller may log on the 1 branch).
147 */
148int
149gsi_store_signing_key(brix_ctx_t *ctx, const unsigned char *secret,
150 size_t secret_len)
151{
152 EVP_MD_CTX *mdctx = EVP_MD_CTX_new();
153 unsigned int dlen = 32;
154 u_char digest[32];
155 int ok = 0;
156
157 if (mdctx
158 && EVP_DigestInit_ex(mdctx, EVP_sha256(), NULL) == 1
159 && EVP_DigestUpdate(mdctx, secret, secret_len) == 1
160 && EVP_DigestFinal_ex(mdctx, digest, &dlen) == 1)
161 {
162 ngx_memcpy(ctx->sigver.signing_key, digest, 32);
163 ctx->sigver.signing_active = 1;
164 ok = 1;
165 }
166 if (mdctx) {
167 EVP_MD_CTX_free(mdctx);
168 }
