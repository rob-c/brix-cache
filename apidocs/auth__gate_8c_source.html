3 *
4 * Called by every handler that performs a namespace or file operation on a

17/*
18 * brix_acc_gate_engine — tier-1 authorization via the faithful XrdAcc engine
19 * (when `brix_authdb_format xrdacc`). Maps the handler's native privilege
20 * level to an XrdAcc operation, builds the request entity from the resolved
21 * identity (DN/subject as the user name, the FQAN-derived VO/role/group views,
22 * and the peer address as the host), evaluates the per-worker tables, and
23 * audits the verdict. Fail-closed: a missing table set denies.
24 */
25static ngx_int_t
26brix_acc_gate_engine(brix_ctx_t *ctx, ngx_connection_t *c,
27 ngx_stream_brix_srv_conf_t *conf, const char *path,
28 const char *op_name, int auth_level, brix_acc_op_t op_in)
29{
30 brix_acc_entity_t *ent;
31 brix_acc_op_t op;
32 brix_acc_privs_t privs;
33 const char *name, *host;
34 const char *vorg = "", *role = "", *grp = "";
35
36 if (conf->acc.tables == NULL) {
37 return NGX_ERROR; /* xrdacc selected but authdb failed to load */
38 }
39 if (path == NULL) {
40 path = "/";
41 }
42
43 /* Use the caller's explicit operation when given (e.g. create vs update,
44 * stage); otherwise derive it from the native privilege level. */
45 if (op_in != BRIX_AOP_ANY) {
46 op = op_in;
47 }
48 else if (auth_level & BRIX_AUTH_DELETE) { op = BRIX_AOP_DELETE; }
49 else if (auth_level & BRIX_AUTH_MKDIR) { op = BRIX_AOP_MKDIR; }
50 else if (auth_level & BRIX_AUTH_ADMIN) { op = BRIX_AOP_CHMOD; }
51 else if (auth_level & BRIX_AUTH_UPDATE) { op = BRIX_AOP_UPDATE; }
52 else if (auth_level & BRIX_AUTH_READ) { op = BRIX_AOP_READ; }
53 else { op = BRIX_AOP_STAT; }
54
55 if (ctx->identity != NULL) {
56 name = brix_identity_dn_cstr(ctx->identity);
57 vorg = brix_identity_acc_vorg_cstr(ctx->identity);
58 role = brix_identity_acc_role_cstr(ctx->identity);
59 grp = brix_identity_acc_group_cstr(ctx->identity);
60 } else {
61 name = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "";
62 vorg = ctx->login.vo_list; /* best-effort when no structured identity */
63 }
64 host = (ctx->login.peer_ip[0] != '\0') ? ctx->login.peer_ip : "?";
65
66 /* Opt-in reverse DNS for `h <host>`/`h .domain` rules: resolve the peer to
67 * a hostname once per connection (cached on ctx), falling back to the IP
68 * when there is no PTR record. Off by default (XrdAccAccess::Resolve). */
69 if (conf->acc.resolve_hosts) {
70 if (!ctx->login.acc_host_done) {
71 char hbuf[256];
72 const char *h;
73
74 ctx->login.acc_host_done = 1;
75 h = brix_acc_resolve_peer(c->sockaddr, c->socklen,
76 hbuf, sizeof(hbuf));
77 if (h != NULL) {
78 size_t n = ngx_strlen(h);
79 char *dup = ngx_pnalloc(c->pool, n + 1);
80 if (dup != NULL) {
81 ngx_memcpy(dup, h, n + 1);
82 ctx->login.acc_host = dup;
83 }
84 }
85 }
86 if (ctx->login.acc_host != NULL) {
87 host = ctx->login.acc_host;
88 }
89 }
90
91 ent = brix_acc_entity_build(c->pool, name, host,
92 (name != NULL && name[0] != '\0'),
93 vorg, role, grp);
94 if (ent == NULL) {
95 return NGX_ERROR;
96 }
97
98 privs = brix_acc_access(conf->acc.tables, ent, path, op);
99
100 brix_acc_audit(c->log, conf->acc.audit, privs != BRIX_ACC_PRIV_NONE,
101 op_name, name, host, path);
102
103 return (privs != BRIX_ACC_PRIV_NONE) ? NGX_OK : NGX_ERROR;
104}

106/*
107 * brix_authz_check — decision-only, format-aware authorization for callers
108 * that format their own error response (TPC dest-open, prepare). Routes to the
109 * XrdAcc engine when `brix_authdb_format xrdacc`, else the native authdb —
110 * exactly matching the native call it replaces. No wire response, no metric.
111 *
112 * Like brix_auth_gate_op, the two engines key off different paths: XrdAcc
113 * authorizes the LOGICAL namespace path (reqpath) while the native authdb keys
114 * off the backing FS path (resolved). Callers pass both; native behavior is
115 * unchanged (it still sees `resolved`). Returns NGX_OK (allow) / NGX_ERROR.
116 */
117ngx_int_t
118brix_authz_check(brix_ctx_t *ctx, ngx_connection_t *c,
119 ngx_stream_brix_srv_conf_t *conf, const char *reqpath,
120 const char *resolved, const char *op_name, int auth_level,
121 brix_acc_op_t aop)
122{
123 if (conf->acc.format == BRIX_AUTHDB_FORMAT_XRDACC) {
124 return brix_acc_gate_engine(ctx, c, conf,
125 (reqpath != NULL) ? reqpath : resolved,
126 op_name, auth_level, aop);
127 }
128 return brix_check_authdb(ctx, resolved, auth_level);
129}

131/*
132 * Build the 32-byte auth-cache key from every input that determines the
133 * verdict: the required authdb level and write flag, the XrdAcc operation, the
134 * resolved and request paths, the peer host, the DN, the VO list, and the raw
135 * token-scope claim. Two requests therefore share a cache entry only when all
136 * of those match — so a cached grant can never be replayed for a different
137 * token, path, operation, host, or access level.
138 *
139 * `aop` and `host` are the XrdAcc engine's extra inputs (create vs update vs
140 * stage decide differently for the same auth_level; `h`/domain rules key off
141 * the peer). Native callers pass AOP_ANY + "" so their keys are unchanged.
142 * The peer IP keys the host because IP->hostname is deterministic per
143 * connection, so it stands in for the resolved name without an early lookup.
144 * Returns 1 on success, 0 if the components don't fit the staging buffer.
145 */
146static int
147brix_auth_gate_cache_key(u_char key[32], int auth_level, int need_write,
148 brix_acc_op_t aop, const char *host, const char *reqpath,
149 const char *resolved, brix_ctx_t *ctx)
150{
151 u_char buf[3 + 64 + 1 + PATH_MAX + PATH_MAX + 512 + 512 + 1024 + 8];
152 size_t n = 0;
153 const char *dn = ctx->login.dn;
154 const char *vo = ctx->login.vo_list;
155 const char *scope = "";
156 size_t scope_len = 0;
157 size_t hl, rl, ql, dl, vl;
158
159 if (ctx->identity != NULL && ctx->identity->scope_raw.len > 0) {
160 scope = (const char *) ctx->identity->scope_raw.data;
161 scope_len = ctx->identity->scope_raw.len;
162 }
163
164 hl = host ? ngx_strlen(host) : 0;
165 rl = resolved ? ngx_strlen(resolved) : 0;
166 ql = reqpath ? ngx_strlen(reqpath) : 0;
167 dl = dn ? ngx_strlen(dn) : 0;
168 vl = vo ? ngx_strlen(vo) : 0;
169
170 if (3 + hl + 1 + rl + 1 + ql + 1 + dl + 1 + vl + 1 + scope_len > sizeof(buf)) {
171 return 0;
172 }
173
174 buf[n++] = (u_char) auth_level;
175 buf[n++] = (u_char) need_write;
176 buf[n++] = (u_char) aop;
177 if (hl) { ngx_memcpy(buf + n, host, hl); n += hl; }
178 buf[n++] = '\0';
179 if (rl) { ngx_memcpy(buf + n, resolved, rl); n += rl; }
180 buf[n++] = '\0';
181 if (ql) { ngx_memcpy(buf + n, reqpath, ql); n += ql; }
182 buf[n++] = '\0';
183 if (dl) { ngx_memcpy(buf + n, dn, dl); n += dl; }
184 buf[n++] = '\0';
185 if (vl) { ngx_memcpy(buf + n, vo, vl); n += vl; }
186 buf[n++] = '\0';
187 if (scope_len) { ngx_memcpy(buf + n, scope, scope_len); n += scope_len; }
188
189 return brix_sha256(buf, n, key);
190}

192/* Record a grant/deny verdict in the auth-result cache (no-op if disabled). */
193static void
194brix_auth_gate_cache_put(ngx_stream_brix_srv_conf_t *conf,
195 const u_char key[32], int have_key, int allowed, int auth_level)
196{
197 brix_auth_cache_val_t cv;
198
199 if (conf->auth_cache.kv == NULL || !have_key) {
200 return;
201 }
202 cv.allowed = (uint8_t) (allowed ? 1 : 0);
203 cv.auth_level = (uint8_t) auth_level;
204 cv.pad = 0;
205 (void) brix_kv_set(conf->auth_cache.kv, key, 32, &cv, sizeof(cv),
206 (ngx_msec_t) conf->auth_cache.ttl_secs * 1000);
207 /* E2: also populate the per-worker L1 so the next hit skips the SHM lock. */
208 brix_auth_l1_store(conf->auth_l1, key, &cv,
209 (ngx_msec_t) conf->auth_cache.ttl_secs * 1000);
210}

212ngx_int_t
213brix_auth_gate_op(brix_ctx_t *ctx, ngx_connection_t *c,
214 ngx_uint_t op_id, const char *op_name,
215 const char *reqpath, const char *resolved,
216 ngx_stream_brix_srv_conf_t *conf,
217 int auth_level, int need_write, brix_acc_op_t aop)
218{
219 u_char ac_key[32];
220 int ac_have_key = 0;
221 int is_xrdacc =
222 (conf->acc.format == BRIX_AUTHDB_FORMAT_XRDACC);
223 /* xrdacc verdicts also depend on the operation and peer host, so fold both
224 * into the key (native passes AOP_ANY + "" → its keys are unchanged). OS
225 * group membership — the one remaining xrdacc input — is bounded by the
226 * gidlifetime TTL, matching the auth-cache TTL, so a cached entry cannot
227 * outlive a group change by more than the configured window. */
228 brix_acc_op_t key_aop = is_xrdacc ? aop : BRIX_AOP_ANY;
229 const char *key_host = is_xrdacc ? ctx->login.peer_ip : "";
230
231 /* auth-result cache: fast path (E2: lockless L1, then SHM L2) */ if (conf->auth_cache.kv != NULL) {
232 ac_have_key = brix_auth_gate_cache_key(ac_key, auth_level,
233 need_write, key_aop, key_host,
234 reqpath, resolved, ctx);
235 if (ac_have_key) {
236 brix_auth_cache_val_t cv;
237 int got = 0;
238
239 /* Lazily create the per-worker L1 (COW-private per worker). */
240 if (conf->auth_l1 == NULL) {
241 conf->auth_l1 = brix_auth_l1_create(ngx_cycle->pool, 0);
242 }
243
244 if (brix_auth_l1_lookup(conf->auth_l1, ac_key, &cv)) {
245 got = 1; /* L1 hit — no SHM spinlock */
246 BRIX_RESIL_METRIC_INC(auth_l1_hits_total);
247 } else {
248 BRIX_RESIL_METRIC_INC(auth_l1_misses_total);
249 size_t cl = sizeof(cv);
250 if (brix_kv_get(conf->auth_cache.kv, ac_key, sizeof(ac_key),
251 &cv, &cl) == 1 && cl == sizeof(cv))
252 {
253 got = 1;
254 /* Promote the L2 hit into L1 for the next presentation. */
255 brix_auth_l1_store(conf->auth_l1, ac_key, &cv,
256 (ngx_msec_t) conf->auth_cache.ttl_secs * 1000);
257 }
258 }
259
260 if (got) {
261 if (!cv.allowed) {
262 brix_log_access(ctx, c, op_name, resolved, "-",
263 0, kXR_NotAuthorized,
264 "auth cache: denied", 0);
265 BRIX_OP_ERR(ctx, op_id);
266 ctx->write_rc = brix_send_error(ctx, c,
267 kXR_NotAuthorized, "not authorized");
268 return NGX_DONE;
269 }
270 return NGX_OK; /* cached grant */
271 }
272 }
273 }
274
275 if (is_xrdacc) {
276 /* XrdAcc authorizes the logical namespace path, not the backing FS path. */
277 if (brix_acc_gate_engine(ctx, c, conf,
278 (reqpath != NULL) ? reqpath : resolved,
279 op_name, auth_level, aop) != NGX_OK)
280 {
281 brix_log_access(ctx, c, op_name, resolved, "-",
282 0, kXR_NotAuthorized, "xrdacc denied", 0);
283 BRIX_OP_ERR(ctx, op_id);
284 brix_auth_gate_cache_put(conf, ac_key, ac_have_key, 0, auth_level);
285 ctx->write_rc = brix_send_error(ctx, c, kXR_NotAuthorized,
286 "not authorized");
287 return NGX_DONE;
288 }
289 } else if (brix_check_authdb(ctx, resolved, auth_level) != NGX_OK) {
290 brix_log_access(ctx, c, op_name, resolved, "-",
291 0, kXR_NotAuthorized, "authdb denied", 0);
292 BRIX_OP_ERR(ctx, op_id);
293 brix_auth_gate_cache_put(conf, ac_key, ac_have_key, 0, auth_level);
294 ctx->write_rc = brix_send_error(ctx, c, kXR_NotAuthorized,
295 "authdb denied");
296 return NGX_DONE;
297 }
298
299 if (brix_check_vo_acl_identity(c->log, resolved, conf->vo_rules,
300 ctx->identity) != NGX_OK) {
301 brix_log_access(ctx, c, op_name, resolved, "-",
302 0, kXR_NotAuthorized, "VO not authorized", 0);
303 BRIX_OP_ERR(ctx, op_id);
304 brix_auth_gate_cache_put(conf, ac_key, ac_have_key, 0, auth_level);
305 ctx->write_rc = brix_send_error(ctx, c, kXR_NotAuthorized,
306 "VO not authorized");
307 return NGX_DONE;
308 }
309
310 if (brix_check_token_scope(ctx, reqpath, need_write) != NGX_OK) {
311 brix_log_access(ctx, c, op_name, reqpath, "-",
312 0, kXR_NotAuthorized, "token scope denied", 0);
313 BRIX_OP_ERR(ctx, op_id);
314 brix_auth_gate_cache_put(conf, ac_key, ac_have_key, 0, auth_level);
315 ctx->write_rc = brix_send_error(ctx, c, kXR_NotAuthorized,
316 "token scope denied");
317 return NGX_DONE;
318 }
319
320 brix_auth_gate_cache_put(conf, ac_key, ac_have_key, 1, auth_level);
321 return NGX_OK;
322}
