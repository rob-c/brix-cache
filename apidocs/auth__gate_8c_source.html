2 * auth_gate.c — three-tier auth check for path-based operations.
3 *
4 * Called by every handler that performs a namespace or file operation on a
5 * canonically-resolved path. Checks authdb, VO ACL, and token scope in
6 * order; sends the kXR_NotAuthorized response and returns NGX_DONE on the
7 * first failure, storing the nginx return code in ctx->write_rc.
8 */

17/*
18 * auth_gate_ctx_t — file-local bundle of one authorization request's inputs.
19 *
20 * WHAT: the connection/session state (ctx, c, conf) plus the per-operation
21 * parameters (op_id/op_name, the logical reqpath and backing resolved path,
22 * the native auth_level + need_write flag, and the explicit XrdAcc op) that
23 * the engine, cache-key, and gate helpers all read.
24 * WHY: the auth-gate cluster was param-bloat — brix_acc_gate_engine took 7
25 * params, brix_auth_gate_cache_key 7, and the extern brix_auth_gate_op 10.
26 * Threading one const bundle through the STATIC helpers collapses that to a
27 * single pointer while keeping the frozen extern signatures untouched.
28 * HOW: brix_auth_gate_op builds it once (zero-initialised) from its frozen
29 * parameters and passes a const pointer down; helpers never mutate it — the
30 * only mutable per-request state (cache key bytes, verdict) lives in locals.
31 */
32typedef struct {
33 brix_ctx_t *ctx;
34 ngx_connection_t *c;
35 ngx_stream_brix_srv_conf_t *conf;
36 ngx_uint_t op_id;
37 const char *op_name;
38 const char *reqpath; /* logical namespace path (xrdacc) */
39 const char *resolved; /* backing FS path (native authdb) */
40 int auth_level;
41 int need_write;
42 brix_acc_op_t aop; /* explicit xrdacc op, or ANY */
43} auth_gate_ctx_t;

45/*
46 * brix_acc_gate_select_op — map the request to an XrdAcc operation.
47 *
48 * WHAT: returns the caller's explicit operation when one was given, else
49 * derives it from the native privilege level (delete > mkdir > admin >
50 * update > read > stat).
51 * WHY: isolates the branch ladder so the engine body stays flat; the
52 * precedence order is verdict-load-bearing and must not change.
53 * HOW: pure function of op_in/auth_level — no side effects.
54 */
55static brix_acc_op_t
56brix_acc_gate_select_op(brix_acc_op_t op_in, int auth_level)
57{
58 if (op_in != BRIX_AOP_ANY) { return op_in; }
59 if (auth_level & BRIX_AUTH_DELETE) { return BRIX_AOP_DELETE; }
60 if (auth_level & BRIX_AUTH_MKDIR) { return BRIX_AOP_MKDIR; }
61 if (auth_level & BRIX_AUTH_ADMIN) { return BRIX_AOP_CHMOD; }
62 if (auth_level & BRIX_AUTH_UPDATE) { return BRIX_AOP_UPDATE; }
63 if (auth_level & BRIX_AUTH_READ) { return BRIX_AOP_READ; }
64 return BRIX_AOP_STAT;
65}

67/*
68 * brix_acc_gate_identity — resolve the XrdAcc entity's name and VO views.
69 *
70 * WHAT: fills *name / *vorg / *role / *grp from the structured identity when
71 * present, else from the best-effort login fields (DN + raw VO list).
72 * WHY: extracts the identity-vs-login fallback out of the engine body; the
73 * *role / *grp defaults ("") set by the caller survive the no-identity path.
74 * HOW: reads ctx only; writes through the four out-params.
75 */
76static void
77brix_acc_gate_identity(brix_ctx_t *ctx, const char **name,
78 const char **vorg, const char **role, const char **grp)
79{
80 if (ctx->identity != NULL) {
81 *name = brix_identity_dn_cstr(ctx->identity);
82 *vorg = brix_identity_acc_vorg_cstr(ctx->identity);
83 *role = brix_identity_acc_role_cstr(ctx->identity);
84 *grp = brix_identity_acc_group_cstr(ctx->identity);
85 } else {
86 *name = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "";
87 *vorg = ctx->login.vo_list; /* best-effort when no structured identity */
88 }
89}

91/*
92 * brix_acc_gate_host — pick the host token for `h <host>`/`h .domain` rules.
93 *
94 * WHAT: returns the peer IP by default; when reverse-DNS resolution is
95 * enabled (XrdAccAccess::Resolve), resolves the peer to a hostname once per
96 * connection (cached on ctx) and returns that, falling back to the IP.
97 * WHY: the opt-in PTR lookup was the engine's deepest nesting; hoisting it
98 * keeps the per-connection caching side effect at the edge.
99 * HOW: mutates only ctx->login.acc_host* (the connection-scoped cache); the
100 * verdict-relevant return value is a borrowed string owned by ctx or "?".
101 */
102static const char *
103brix_acc_gate_host(brix_ctx_t *ctx, ngx_connection_t *c,
104 ngx_stream_brix_srv_conf_t *conf)
105{
106 const char *host = (ctx->login.peer_ip[0] != '\0') ? ctx->login.peer_ip : "?";
107
108 if (!conf->acc.resolve_hosts) {
109 return host;
110 }
111
112 if (!ctx->login.acc_host_done) {
113 char hbuf[256];
114 const char *h;
115
116 ctx->login.acc_host_done = 1;
117 h = brix_acc_resolve_peer(c->sockaddr, c->socklen, hbuf, sizeof(hbuf));
118 if (h != NULL) {
119 size_t n = ngx_strlen(h);
120 char *dup = ngx_pnalloc(c->pool, n + 1);
121 if (dup != NULL) {
122 ngx_memcpy(dup, h, n + 1);
123 ctx->login.acc_host = dup;
124 }
125 }
126 }
127
128 return (ctx->login.acc_host != NULL) ? ctx->login.acc_host : host;
129}

131/*
132 * brix_acc_gate_engine — tier-1 authorization via the faithful XrdAcc engine
133 * (when `brix_authdb_format xrdacc`). Maps the handler's native privilege
134 * level to an XrdAcc operation, builds the request entity from the resolved
135 * identity (DN/subject as the user name, the FQAN-derived VO/role/group views,
136 * and the peer address as the host), evaluates the per-worker tables, and
137 * audits the verdict. Fail-closed: a missing table set denies.
138 */
139static ngx_int_t
140brix_acc_gate_engine(const auth_gate_ctx_t *g, const char *path)
141{
142 brix_ctx_t *ctx = g->ctx;
143 ngx_connection_t *c = g->c;
144 ngx_stream_brix_srv_conf_t *conf = g->conf;
145 brix_acc_entity_t *ent;
146 brix_acc_op_t op;
147 brix_acc_privs_t privs;
148 const char *name, *host;
149 const char *vorg, *role = "", *grp = "";
150
151 if (conf->acc.tables == NULL) {
152 return NGX_ERROR; /* xrdacc selected but authdb failed to load */
153 }
154 if (path == NULL) {
155 path = "/";
156 }
157
158 /* Use the caller's explicit operation when given (e.g. create vs update,
159 * stage); otherwise derive it from the native privilege level. */
160 op = brix_acc_gate_select_op(g->aop, g->auth_level);
161
162 brix_acc_gate_identity(ctx, &name, &vorg, &role, &grp);
163 host = brix_acc_gate_host(ctx, c, conf);
164
165 ent = brix_acc_entity_build(c->pool, name, host,
166 (name != NULL && name[0] != '\0'),
167 vorg, role, grp);
168 if (ent == NULL) {
169 return NGX_ERROR;
170 }
171
172 privs = brix_acc_access(conf->acc.tables, ent, path, op);
173
174 brix_acc_audit(c->log, conf->acc.audit, privs != BRIX_ACC_PRIV_NONE,
175 g->op_name, name, host, path);
176
177 return (privs != BRIX_ACC_PRIV_NONE) ? NGX_OK : NGX_ERROR;
178}

180/*
181 * brix_authz_check — decision-only, format-aware authorization for callers
182 * that format their own error response (TPC dest-open, prepare). Routes to the
183 * XrdAcc engine when `brix_authdb_format xrdacc`, else the native authdb —
184 * exactly matching the native call it replaces. No wire response, no metric.
185 *
186 * Like brix_auth_gate_op, the two engines key off different paths: XrdAcc
187 * authorizes the LOGICAL namespace path (reqpath) while the native authdb keys
188 * off the backing FS path (resolved). Callers pass both; native behavior is
189 * unchanged (it still sees `resolved`). Returns NGX_OK (allow) / NGX_ERROR.
190 */
191ngx_int_t
192brix_authz_check(brix_ctx_t *ctx, ngx_connection_t *c,
193 ngx_stream_brix_srv_conf_t *conf, const char *reqpath,
194 const char *resolved, const char *op_name, int auth_level,
195 brix_acc_op_t aop)
196{
197 if (conf->acc.format == BRIX_AUTHDB_FORMAT_XRDACC) {
198 auth_gate_ctx_t g = {0};
199
200 g.ctx = ctx;
201 g.c = c;
202 g.conf = conf;
203 g.op_name = op_name;
204 g.reqpath = reqpath;
205 g.resolved = resolved;
206 g.auth_level = auth_level;
207 g.aop = aop;
208
209 return brix_acc_gate_engine(&g, (reqpath != NULL) ? reqpath : resolved);
210 }
211 return brix_check_authdb(ctx, resolved, auth_level);
212}

214/*
215 * Build the 32-byte auth-cache key from every input that determines the
216 * verdict: the required authdb level and write flag, the XrdAcc operation, the
217 * resolved and request paths, the peer host, the DN, the VO list, and the raw
218 * token-scope claim. Two requests therefore share a cache entry only when all
219 * of those match — so a cached grant can never be replayed for a different
220 * token, path, operation, host, or access level.
221 *
222 * `key_aop` and `host` are the XrdAcc engine's extra inputs (create vs update
223 * vs stage decide differently for the same auth_level; `h`/domain rules key off
224 * the peer). Native callers fold in AOP_ANY + "" so their keys are unchanged.
225 * The peer IP keys the host because IP->hostname is deterministic per
226 * connection, so it stands in for the resolved name without an early lookup.
227 * Returns 1 on success, 0 if the components don't fit the staging buffer.
228 */
229static int
230brix_auth_gate_cache_key(u_char key[32], const auth_gate_ctx_t *g,
231 brix_acc_op_t key_aop, const char *host)
232{
233 brix_ctx_t *ctx = g->ctx;
234 u_char buf[3 + 64 + 1 + PATH_MAX + PATH_MAX + 512 + 512 + 1024 + 8];
235 size_t n = 0;
236 const char *dn = ctx->login.dn;
237 const char *vo = ctx->login.vo_list;
238 const char *scope = "";
239 size_t scope_len = 0;
240 size_t hl, rl, ql, dl, vl;
241
242 if (ctx->identity != NULL && ctx->identity->scope_raw.len > 0) {
243 scope = (const char *) ctx->identity->scope_raw.data;
244 scope_len = ctx->identity->scope_raw.len;
245 }
246
247 hl = host ? ngx_strlen(host) : 0;
248 rl = g->resolved ? ngx_strlen(g->resolved) : 0;
249 ql = g->reqpath ? ngx_strlen(g->reqpath) : 0;
250 dl = dn ? ngx_strlen(dn) : 0;
251 vl = vo ? ngx_strlen(vo) : 0;
252
253 if (3 + hl + 1 + rl + 1 + ql + 1 + dl + 1 + vl + 1 + scope_len > sizeof(buf)) {
254 return 0;
255 }
256
257 buf[n++] = (u_char) g->auth_level;
258 buf[n++] = (u_char) g->need_write;
259 buf[n++] = (u_char) key_aop;
260 if (hl) { ngx_memcpy(buf + n, host, hl); n += hl; }
261 buf[n++] = '\0';
262 if (rl) { ngx_memcpy(buf + n, g->resolved, rl); n += rl; }
263 buf[n++] = '\0';
264 if (ql) { ngx_memcpy(buf + n, g->reqpath, ql); n += ql; }
265 buf[n++] = '\0';
266 if (dl) { ngx_memcpy(buf + n, dn, dl); n += dl; }
267 buf[n++] = '\0';
268 if (vl) { ngx_memcpy(buf + n, vo, vl); n += vl; }
269 buf[n++] = '\0';
270 if (scope_len) { ngx_memcpy(buf + n, scope, scope_len); n += scope_len; }
271
272 return brix_sha256(buf, n, key);
273}

275/* Record a grant/deny verdict in the auth-result cache (no-op if disabled). */
276static void
277brix_auth_gate_cache_put(ngx_stream_brix_srv_conf_t *conf,
278 const u_char key[32], int have_key, int allowed, int auth_level)
279{
280 brix_auth_cache_val_t cv;
281
282 if (conf->auth_cache.kv == NULL || !have_key) {
283 return;
284 }
285 cv.allowed = (uint8_t) (allowed ? 1 : 0);
286 cv.auth_level = (uint8_t) auth_level;
287 cv.pad = 0;
288 (void) brix_kv_set(conf->auth_cache.kv, key, 32, &cv, sizeof(cv),
289 (ngx_msec_t) conf->auth_cache.ttl_secs * 1000);
290 /* E2: also populate the per-worker L1 so the next hit skips the SHM lock. */
291 brix_auth_l1_store(conf->auth_l1, key, &cv,
292 (ngx_msec_t) conf->auth_cache.ttl_secs * 1000);
293}

295/*
296 * brix_auth_gate_deny — emit a uniform tier-deny outcome.
297 *
298 * WHAT: logs the access denial, bumps the op error metric, records the deny
299 * verdict in the auth cache, sends the kXR_NotAuthorized wire response with
300 * `msg`, stashes the return code in ctx->write_rc, and returns NGX_DONE.
301 * WHY: every failing tier ran this identical five-step sequence; collapsing
302 * it removes four near-duplicate blocks and keeps the deny path (and its wire
303 * bytes / log strings) byte-identical across tiers — the security-load-bearing
304 * default-deny contract.
305 * HOW: `log_path` differs per tier (resolved vs reqpath), so the caller passes
306 * it explicitly; ac_key/ac_have_key thread the already-computed cache key.
307 */
308static ngx_int_t
309brix_auth_gate_deny(const auth_gate_ctx_t *g, const u_char ac_key[32],
310 int ac_have_key, const char *log_path, const char *msg)
311{
312 brix_ctx_t *ctx = g->ctx;
313
314 brix_log_access(ctx, g->c, g->op_name, log_path, "-",
315 0, kXR_NotAuthorized, msg, 0);
316 BRIX_OP_ERR(ctx, g->op_id);
317 brix_auth_gate_cache_put(g->conf, ac_key, ac_have_key, 0, g->auth_level);
318 ctx->write_rc = brix_send_error(ctx, g->c, kXR_NotAuthorized, msg);
319 return NGX_DONE;
320}

322/*
323 * brix_auth_gate_cache_probe — auth-result cache fast path (L1 then SHM L2).
324 *
325 * WHAT: computes the verdict key (folding the xrdacc-only op+host inputs) and,
326 * on a hit, sets *out_verdict to the cached allow/deny; always sets *out_key
327 * / *out_have_key so a subsequent miss can store the fresh verdict under the
328 * same key. Returns 1 when a cached verdict was found, 0 otherwise.
329 * WHY: extracts the two-level cache lookup so brix_auth_gate_op reads as a
330 * flat probe → evaluate → cache sequence; the E2 L1/L2 promotion and metric
331 * counters stay verbatim.
332 * HOW: lazily creates the per-worker L1; on an L2 hit it promotes into L1.
333 * *out_have_key stays 0 when caching is disabled or the key overflows.
334 */
335static int
336brix_auth_gate_cache_probe(const auth_gate_ctx_t *g, u_char out_key[32],
337 int *out_have_key, int *out_verdict)
338{
339 ngx_stream_brix_srv_conf_t *conf = g->conf;
340 int is_xrdacc = (conf->acc.format == BRIX_AUTHDB_FORMAT_XRDACC);
341 /* xrdacc verdicts also depend on the operation and peer host, so fold both
342 * into the key (native passes AOP_ANY + "" -> its keys are unchanged). OS
343 * group membership — the one remaining xrdacc input — is bounded by the
344 * gidlifetime TTL, matching the auth-cache TTL, so a cached entry cannot
345 * outlive a group change by more than the configured window. */
346 brix_acc_op_t key_aop = is_xrdacc ? g->aop : BRIX_AOP_ANY;
347 const char *key_host = is_xrdacc ? g->ctx->login.peer_ip : "";
348 brix_auth_cache_val_t cv;
349
350 *out_have_key = 0;
351
352 if (conf->auth_cache.kv == NULL) {
353 return 0;
354 }
355
356 *out_have_key = brix_auth_gate_cache_key(out_key, g, key_aop, key_host);
357 if (!*out_have_key) {
358 return 0;
359 }
360
361 /* Lazily create the per-worker L1 (COW-private per worker). */
362 if (conf->auth_l1 == NULL) {
363 conf->auth_l1 = brix_auth_l1_create(ngx_cycle->pool, 0);
364 }
365
366 if (brix_auth_l1_lookup(conf->auth_l1, out_key, &cv)) {
367 BRIX_RESIL_METRIC_INC(auth_l1_hits_total); /* L1 hit — no SHM spinlock */
368 } else {
369 size_t cl = sizeof(cv);
370 BRIX_RESIL_METRIC_INC(auth_l1_misses_total);
371 if (brix_kv_get(conf->auth_cache.kv, out_key, 32, &cv, &cl) != 1
372 || cl != sizeof(cv))
373 {
374 return 0;
375 }
376 /* Promote the L2 hit into L1 for the next presentation. */
377 brix_auth_l1_store(conf->auth_l1, out_key, &cv,
378 (ngx_msec_t) conf->auth_cache.ttl_secs * 1000);
379 }
380
381 *out_verdict = cv.allowed ? 1 : 0;
382 return 1;
383}

385/*
386 * brix_auth_gate_evaluate — run the three-tier check (authdb/xrdacc → VO ACL →
387 * token scope) and emit the first denial.
388 *
389 * WHAT: on the first failing tier returns NGX_DONE (after brix_auth_gate_deny
390 * has responded + stashed ctx->write_rc); returns NGX_OK when all tiers pass.
391 * WHY: separates the slow-path policy evaluation from the cache plumbing so
392 * neither exceeds the complexity gate; tier order and messages are frozen.
393 * HOW: threads the pre-computed cache key so each deny records the verdict.
394 */
395static ngx_int_t
396brix_auth_gate_evaluate(const auth_gate_ctx_t *g, const u_char ac_key[32],
397 int ac_have_key)
398{
399 brix_ctx_t *ctx = g->ctx;
400
401 if (g->conf->acc.format == BRIX_AUTHDB_FORMAT_XRDACC) {
402 /* XrdAcc authorizes the logical namespace path, not the backing FS path. */
403 if (brix_acc_gate_engine(g, (g->reqpath != NULL) ? g->reqpath
404 : g->resolved) != NGX_OK)
405 {
406 return brix_auth_gate_deny(g, ac_key, ac_have_key,
407 g->resolved, "xrdacc denied");
408 }
409 } else if (brix_check_authdb(ctx, g->resolved, g->auth_level) != NGX_OK) {
410 return brix_auth_gate_deny(g, ac_key, ac_have_key,
411 g->resolved, "authdb denied");
412 }
413
414 if (brix_check_vo_acl_identity(g->c->log, g->resolved, g->conf->vo_rules,
415 ctx->identity) != NGX_OK) {
416 return brix_auth_gate_deny(g, ac_key, ac_have_key,
417 g->resolved, "VO not authorized");
418 }
419
420 if (brix_check_token_scope(ctx, g->reqpath, g->need_write) != NGX_OK) {
421 return brix_auth_gate_deny(g, ac_key, ac_have_key,
422 g->reqpath, "token scope denied");
423 }
424
425 return NGX_OK;
426}

428ngx_int_t
429brix_auth_gate_op(brix_ctx_t *ctx, ngx_connection_t *c,
430 ngx_uint_t op_id, const char *op_name,
431 const char *reqpath, const char *resolved,
432 ngx_stream_brix_srv_conf_t *conf,
433 int auth_level, int need_write, brix_acc_op_t aop)
434{
435 auth_gate_ctx_t g = {0};
436 u_char ac_key[32];
437 int ac_have_key = 0;
438 int verdict = 0;
439 ngx_int_t rc;
440
441 g.ctx = ctx;
442 g.c = c;
443 g.conf = conf;
444 g.op_id = op_id;
445 g.op_name = op_name;
446 g.reqpath = reqpath;
447 g.resolved = resolved;
448 g.auth_level = auth_level;
449 g.need_write = need_write;
450 g.aop = aop;
451
452 /* auth-result cache: fast path (E2: lockless L1, then SHM L2). */
453 if (brix_auth_gate_cache_probe(&g, ac_key, &ac_have_key, &verdict)) {
454 if (!verdict) {
455 brix_log_access(ctx, c, op_name, resolved, "-",
456 0, kXR_NotAuthorized,
457 "auth cache: denied", 0);
458 BRIX_OP_ERR(ctx, op_id);
459 ctx->write_rc = brix_send_error(ctx, c,
460 kXR_NotAuthorized, "not authorized");
461 return NGX_DONE;
462 }
463 return NGX_OK; /* cached grant */
464 }
465
466 rc = brix_auth_gate_evaluate(&g, ac_key, ac_have_key);
467 if (rc != NGX_OK) {
468 return rc;
469 }
470
471 brix_auth_gate_cache_put(conf, ac_key, ac_have_key, 1, auth_level);
472 return NGX_OK;
473}
