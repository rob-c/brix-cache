1/*
2 * vfs_io_core.c — worker-safe VFS disk I/O execution core.
3 *
4 * WHAT: Implements brix_vfs_io_execute(), the POD-only execution surface for
5 * blocking read/write/readv/writev/pgread/sync/truncate work. It mutates
6 * only the job's OUT fields and caller-owned buffers; it never touches
7 * nginx pools, connection state, metrics, access logs, or cache metadata.
8 * Deliberate exception to "no metrics": brix_vfs_io_execute feeds the
9 * per-backend SHM byte totals via brix_metric_backend_bytes — pure
10 * lock-free atomics, POD-safe from thread-pool workers.
11 *
12 * WHY: Stream AIO workers historically duplicated raw syscalls outside the VFS,
13 * which let confinement, CRC, durability, truncation, short-I/O, and error
14 * behavior drift. This core gives worker and inline fallback paths one
15 * shared syscall/CRC body while preserving the event-loop-only public VFS
16 * contract.
17 *
18 * HOW: brix_vfs_io_execute() clears OUT fields, validates the descriptor, and
19 * dispatches to one small static helper per operation. Helpers use only
20 * thread-safe primitives, raw syscalls, and caller-provided POD buffers.
21 */

75/* brix_vfs_io_write_counted — pwrite all `len` bytes (EINTR-retried) via the
76 * Storage Driver seam, preserving the short-I/O fact the public full-write helper
77 * hides: a 0-byte write sets *short_io + EIO and a hard error returns NGX_ERROR,
78 * both with bytes-so-far in *written. The short-I/O accounting policy stays here
79 * in the VFS; the raw syscall lives in the backend. */
80static ngx_int_t
81brix_vfs_io_write_counted(brix_sd_obj_t *job_obj, ngx_fd_t fd,
82 const u_char *buf, size_t len, off_t offset, ssize_t *written,
83 unsigned *short_io)
84{
85 brix_sd_obj_t scratch;
86 brix_sd_obj_t *obj;
87 size_t w = 0;
88 int sh = 0;
89 int rc;
90
91 if (written == NULL || short_io == NULL) {
92 errno = EINVAL;
93 return NGX_ERROR;
94 }
95
96 /* The full-write loop + short-I/O accounting live in the shared `vfs` core
97 * (xvfs_pwrite_full, src/fs/core/vfs_core.c — one copy shared with the
98 * clients); the raw syscall stays in the backend driver. This wrapper bridges
99 * the server's ssize_t/unsigned OUT params. */
100 obj = brix_vfs_effective_obj(job_obj, fd, &scratch);
101 rc = xvfs_pwrite_full(obj, buf, len, offset, &w, &sh);
102 *written = (ssize_t) w;
103 *short_io = (unsigned) sh;
104 return (rc == 0) ? NGX_OK : NGX_ERROR;
105}

107/* brix_vfs_io_execute_read — EOF-safe pread of job->length into job->buf,
108 * filling nio/out_size and the optional CRC32c. (phase-59) When a CSI is present
109 * the bytes are verified against the stored page CRCs, failing the read with EIO
110 * + csi_mismatch on a mismatch rather than serving corrupt data. */
111static void
112brix_vfs_io_execute_read(brix_vfs_job_t *job)
113{
114 size_t nread;
115
116 if ((job->fd == NGX_INVALID_FILE && job->obj.driver == NULL)
117 || job->offset < 0
118 || (job->length > 0 && job->buf == NULL)
119 || job->buf_cap < job->length)
120 {
121 job->nio = -1;
122 job->io_errno = EINVAL;
123 return;
124 }
125
126 nread = 0;
127 {
128 brix_sd_obj_t scratch;
129 brix_sd_obj_t *o = brix_vfs_effective_obj(&job->obj, job->fd, &scratch);
130
131 if (xvfs_pread_full(o, job->buf, job->length, job->offset, &nread) != 0)
132 {
133 job->nio = -1;
134 job->out_size = nread;
135 job->io_errno = errno;
136 return;
137 }
138 }
139
140 job->nio = (ssize_t) nread;
141 job->out_size = nread;
142 if (job->want_pgcrc && nread > 0) {
143 job->crc32c = brix_crc32c_value(job->buf, nread);
144 }
145
146 /* phase-59 W2: verify the bytes just read against the stored page CRCs.
147 * A mismatch fails the read with EIO + the csi_mismatch flag so the handler
148 * maps it to kXR_ChkSumErr instead of serving corrupt data. */
149 if (job->csi != NULL && nread > 0) {
150 int v = brix_csi_verify_read((brix_csi_t *) job->csi, job->buf,
151 job->offset, nread);
152 if (v == BRIX_CSI_MISMATCH) {
153 job->nio = -1;
154 job->io_errno = EIO;
155 job->csi_mismatch = 1;
156 }
157 }
158}

160/* brix_vfs_io_execute_write — counted pwrite of job->buf at job->offset,
161 * recording hard errors vs short writes in the job OUT fields. (phase-59) Retags
162 * aligned page CRCs for the written range (fail-open; non-fatal to the write). */
163static void
164brix_vfs_io_execute_write(brix_vfs_job_t *job)
165{
166 ssize_t written;
167 unsigned short_io;
168
169 if ((job->fd == NGX_INVALID_FILE && job->obj.driver == NULL)
170 || job->offset < 0
171 || (job->length > 0 && job->buf == NULL))
172 {
173 job->nio = -1;
174 job->io_errno = EINVAL;
175 return;
176 }
177
178 if (job->length == 0) {
179 job->nio = 0;
180 return;
181 }
182
183 if (brix_vfs_io_write_counted(&job->obj, job->fd, job->buf, job->length,
184 job->offset, &written, &short_io)
185 != NGX_OK)
186 {
187 job->nio = short_io ? written : -1;
188 job->out_size = written > 0 ? (size_t) written : 0;
189 job->short_io = short_io ? 1 : 0;
190 job->io_errno = errno;
191 return;
192 }
193
194 job->nio = written;
195 job->out_size = (size_t) written;
196
197 /* xmeta P3: fold the written blocks' CRCs into the handle-local table
198 * (pure in-memory; the record is merged once at close). Failure is
199 * non-fatal to the data write — those blocks simply stay unverified. */
200 if (job->csi != NULL && written > 0) {
201 (void) brix_csi_write_update((brix_csi_t *) job->csi, job->buf,
202 job->offset, (size_t) written);
203 }
204}

206/* brix_vfs_io_execute_pgread — read straight into the final pgread wire buffer
207 * with per-page CRC32c words written in place (brix_pgread_read_encode_inplace),
208 * giving pgread the same worker contract as a plain read. */
209static void
210brix_vfs_io_execute_pgread(brix_vfs_job_t *job)
211{
212 brix_pgread_io_t io = { .nowait = 0 /* blocking */, .nread = 0,
213 .io_errno = 0 };
214
215 if ((job->fd == NGX_INVALID_FILE && job->obj.driver == NULL)
216 || job->offset < 0
217 || (job->length > 0 && job->buf == NULL)
218 || job->buf_cap < job->length)
219 {
220 job->nio = -1;
221 job->io_errno = EINVAL;
222 return;
223 }
224
225 {
226 brix_sd_obj_t scratch;
227 brix_sd_obj_t *obj = brix_vfs_effective_obj(&job->obj, job->fd,
228 &scratch);
229 job->out_size = brix_pgread_read_encode_inplace(obj, job->offset,
230 job->length, job->buf,
231 &io);
232 }
233 job->nio = io.nread;
234 if (io.nread < 0) {
235 job->io_errno = io.io_errno;
236 }
237}

239/* brix_vfs_io_execute_readv — run the pre-built, pre-validated coalesced readv
240 * plan over job->segs (brix_readv_read_segments); out_size = per-segment headers
241 * + payload bytes on success. */
242static void
243brix_vfs_io_execute_readv(brix_vfs_job_t *job)
244{
245 size_t bytes_read_total;
246
247 if (job->segs == NULL || job->nsegs == 0) {
248 job->nio = -1;
249 job->io_errno = EINVAL;
250 brix_vfs_io_set_error_message(job, "readv segment count out of range");
251 return;
252 }
253
254 bytes_read_total = 0;
255 if (brix_readv_read_segments((brix_readv_seg_desc_t *) job->segs,
256 job->nsegs, &bytes_read_total,
257 job->err_msg, job->err_msg_cap)
258 != NGX_OK)
259 {
260 job->nio = -1;
261 job->io_errno = EIO;
262 if (job->err_msg != NULL && job->err_msg_cap > 0
263 && job->err_msg[0] == '\0')
264 {
265 brix_vfs_io_set_error_message(job, "readv I/O error");
266 }
267 return;
268 }
269
270 job->nio = (ssize_t) bytes_read_total;
271 job->out_size = job->nsegs * BRIX_READV_SEGSIZE + bytes_read_total;
272}

274/* brix_vfs_io_execute_writev — counted pwrite of each writev segment (first
275 * error wins, with a segment-tagged message), accumulating out_size, then a
276 * best-effort fsync of the non-empty segment fds when job->do_sync. */
277static void
278brix_vfs_io_execute_writev(brix_vfs_job_t *job)
279{
280 brix_vfs_writev_seg_t *segments;
281 size_t segment_index;
282
283 if (job->segs == NULL || job->nsegs == 0) {
284 job->nio = -1;
285 job->io_errno = EINVAL;
286 brix_vfs_io_set_error_message(job, "writev segment count out of range");
287 return;
288 }
289
290 segments = job->segs;
291 for (segment_index = 0; segment_index < job->nsegs; segment_index++) {
292 brix_vfs_writev_seg_t *segment;
293 ssize_t written;
294 unsigned short_io;
295
296 segment = &segments[segment_index];
297 if (segment->wlen == 0) {
298 continue;
299 }
300
301 if (brix_vfs_io_write_counted(&segment->obj, segment->fd,
302 segment->data,
303 (size_t) segment->wlen,
304 segment->offset, &written, &short_io)
305 != NGX_OK)
306 {
307 job->nio = short_io ? written : -1;
308 job->io_errno = errno;
309 job->short_io = short_io ? 1 : 0;
310 if (short_io) {
311 if (job->err_msg != NULL && job->err_msg_cap > 0) {
312 snprintf(job->err_msg, job->err_msg_cap,
313 "writev short write at seg %d",
314 (int) segment_index);
315 }
316 } else {
317 char prefix[64];
318
319 snprintf(prefix, sizeof(prefix),
320 "writev I/O error at seg %d", (int) segment_index);
321 brix_vfs_io_set_errno_message(job, prefix, job->io_errno);
322 }
323 return;
324 }
325
326 job->out_size += (size_t) written;
327 }
328
329 if (job->do_sync) {
330 for (segment_index = 0; segment_index < job->nsegs; segment_index++) {
331 if (segments[segment_index].wlen > 0) {
332 (void) fsync(segments[segment_index].fd);
333 }
334 }
335 }
336
337 job->nio = (ssize_t) job->out_size;
338}

340/* brix_vfs_io_execute_sync — fsync one open fd via the Storage Driver seam so
341 * durability routes through the same chokepoint as writes; nio=0 on success,
342 * -1/io_errno on failure. */
343static void
344brix_vfs_io_execute_sync(brix_vfs_job_t *job)
345{
346 brix_sd_obj_t obj;
347
348 if (job->fd == NGX_INVALID_FILE && job->obj.driver == NULL) {
349 job->nio = -1;
350 job->io_errno = EINVAL;
351 return;
352 }
353
354 /* Durability via the shared `vfs` core (xvfs_fsync → backend driver). */
355 if (xvfs_fsync(brix_vfs_effective_obj(&job->obj, job->fd, &obj)) != 0) {
356 job->nio = -1;
357 job->io_errno = errno;
358 return;
359 }
360
361 job->nio = 0;
362}

408/* brix_vfs_io_account — post-op per-backend byte attribution: map the job's
409 * op to a read/write direction and add job->nio bytes to the backend totals.
410 * The job's obj names the bound driver; NULL ⇒ default POSIX. READV totals
411 * attribute to the job's primary obj (segments could in principle span
412 * handles — a bounded, deliberate approximation). Non-data ops no-op. */
413static void
414brix_vfs_io_account(const brix_vfs_job_t *job)
415{
416 brix_metric_op_t dir;
417
418 if (job->nio <= 0) {
419 return;
420 }
421
422 switch (job->op) {
423 case BRIX_VFS_IO_READ:
424 case BRIX_VFS_IO_PGREAD:
425 case BRIX_VFS_IO_READV:
426 dir = BRIX_METRIC_OP_READ;
427 break;
428 case BRIX_VFS_IO_WRITE:
429 case BRIX_VFS_IO_WRITEV:
430 dir = BRIX_METRIC_OP_WRITE;
431 break;
432 default:
433 return;
434 }
435
436 brix_metric_backend_bytes(
437 job->obj.driver != NULL ? job->obj.driver->name : "posix",
438 dir, (size_t) job->nio);
439}

441/* brix_vfs_io_execute — the single VFS-owned raw-I/O chokepoint: clear the OUT
442 * fields, dispatch *job to one static executor per op (EINVAL on a NULL or
443 * unknown job), then attribute the moved bytes to the job's backend. Callable
444 * from stream AIO workers and inline fallbacks without touching
445 * event-loop-only state. */
446void
447brix_vfs_io_execute(brix_vfs_job_t *job)
448{
449 if (job == NULL) {
450 return;
451 }
452
453 brix_vfs_io_reset_outputs(job);
454
455 switch (job->op) {
456 case BRIX_VFS_IO_READ:
457 brix_vfs_io_execute_read(job);
458 break;
459 case BRIX_VFS_IO_WRITE:
460 brix_vfs_io_execute_write(job);
461 break;
462 case BRIX_VFS_IO_PGREAD:
463 brix_vfs_io_execute_pgread(job);
464 break;
465 case BRIX_VFS_IO_READV:
466 brix_vfs_io_execute_readv(job);
467 break;
468 case BRIX_VFS_IO_WRITEV:
469 brix_vfs_io_execute_writev(job);
470 break;
471 case BRIX_VFS_IO_SYNC:
472 brix_vfs_io_execute_sync(job);
473 break;
474 case BRIX_VFS_IO_TRUNCATE:
475 brix_vfs_io_execute_truncate(job);
476 break;
477 case BRIX_VFS_IO_OPENDIR:
478 brix_vfs_io_execute_opendir(job);
479 break;
480 default:
481 job->nio = -1;
482 job->io_errno = EINVAL;
483 return;
484 }
485
486 brix_vfs_io_account(job);
487}
