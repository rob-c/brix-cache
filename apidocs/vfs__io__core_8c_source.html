1/*
2 * vfs_io_core.c — worker-safe VFS disk I/O execution core.
3 *
4 * WHAT: Implements brix_vfs_io_execute(), the POD-only execution surface for
5 * blocking read/write/readv/writev/pgread/sync/truncate work. It mutates
6 * only the job's OUT fields and caller-owned buffers; it never touches
7 * nginx pools, connection state, metrics, access logs, or cache metadata.
8 * Deliberate exception to "no metrics": brix_vfs_io_execute feeds the
9 * per-backend SHM byte totals via brix_metric_backend_bytes — pure
10 * lock-free atomics, POD-safe from thread-pool workers.
11 *
12 * WHY: Stream AIO workers historically duplicated raw syscalls outside the VFS,
13 * which let confinement, CRC, durability, truncation, short-I/O, and error
14 * behavior drift. This core gives worker and inline fallback paths one
15 * shared syscall/CRC body while preserving the event-loop-only public VFS
16 * contract.
17 *
18 * HOW: brix_vfs_io_execute() clears OUT fields, validates the descriptor, and
19 * dispatches to one small static helper per operation. Helpers use only
20 * thread-safe primitives, raw syscalls, and caller-provided POD buffers.
21 */

76/* brix_vfs_io_write_counted — pwrite all `len` bytes (EINTR-retried) via the
77 * Storage Driver seam, preserving the short-I/O fact the public full-write helper
78 * hides: a 0-byte write sets *short_io + EIO and a hard error returns NGX_ERROR,
79 * both with bytes-so-far in *written. The short-I/O accounting policy stays here
80 * in the VFS; the raw syscall lives in the backend. */
81static ngx_int_t
82brix_vfs_io_write_counted(brix_sd_obj_t *job_obj, ngx_fd_t fd,
83 const u_char *buf, size_t len, off_t offset, ssize_t *written,
84 unsigned *short_io)
85{
86 brix_sd_obj_t scratch;
87 brix_sd_obj_t *obj;
88 size_t w = 0;
89 int sh = 0;
90 int rc;
91
92 if (written == NULL || short_io == NULL) {
93 errno = EINVAL;
94 return NGX_ERROR;
95 }
96
97 /* The full-write loop + short-I/O accounting live in the shared `vfs` core
98 * (xvfs_pwrite_full, src/fs/core/vfs_core.c — one copy shared with the
99 * clients); the raw syscall stays in the backend driver. This wrapper bridges
100 * the server's ssize_t/unsigned OUT params. */
101 obj = brix_vfs_effective_obj(job_obj, fd, &scratch);
102 rc = xvfs_pwrite_full(obj, buf, len, offset, &w, &sh);
103 *written = (ssize_t) w;
104 *short_io = (unsigned) sh;
105 return (rc == 0) ? NGX_OK : NGX_ERROR;
106}

108/* brix_vfs_io_execute_read — EOF-safe pread of job->length into job->buf,
109 * filling nio/out_size and the optional CRC32c. (phase-59) When a CSI is present
110 * the bytes are verified against the stored page CRCs, failing the read with EIO
111 * + csi_mismatch on a mismatch rather than serving corrupt data. */
112static void
113brix_vfs_io_execute_read(brix_vfs_job_t *job)
114{
115 size_t nread;
116
117 if ((job->fd == NGX_INVALID_FILE && job->obj.driver == NULL)
118 || job->offset < 0
119 || (job->length > 0 && job->buf == NULL)
120 || job->buf_cap < job->length)
121 {
122 job->nio = -1;
123 job->io_errno = EINVAL;
124 return;
125 }
126
127 nread = 0;
128 {
129 brix_sd_obj_t scratch;
130 brix_sd_obj_t *o = brix_vfs_effective_obj(&job->obj, job->fd, &scratch);
131
132 if (xvfs_pread_full(o, job->buf, job->length, job->offset, &nread) != 0)
133 {
134 job->nio = -1;
135 job->out_size = nread;
136 job->io_errno = errno;
137 return;
138 }
139 }
140
141 job->nio = (ssize_t) nread;
142 job->out_size = nread;
143 if (job->want_pgcrc && nread > 0) {
144 job->crc32c = brix_crc32c_value(job->buf, nread);
145 }
146
147 /* phase-59 W2: verify the bytes just read against the stored page CRCs.
148 * A mismatch fails the read with EIO + the csi_mismatch flag so the handler
149 * maps it to kXR_ChkSumErr instead of serving corrupt data. */
150 if (job->csi != NULL && nread > 0) {
151 int v = brix_csi_verify_read((brix_csi_t *) job->csi, job->buf,
152 job->offset, nread);
153 if (v == BRIX_CSI_MISMATCH) {
154 job->nio = -1;
155 job->io_errno = EIO;
156 job->csi_mismatch = 1;
157 }
158 }
159}

161/* brix_vfs_io_execute_write — counted pwrite of job->buf at job->offset,
162 * recording hard errors vs short writes in the job OUT fields. (phase-59) Retags
163 * aligned page CRCs for the written range (fail-open; non-fatal to the write). */
164static void
165brix_vfs_io_execute_write(brix_vfs_job_t *job)
166{
167 ssize_t written;
168 unsigned short_io;
169
170 if ((job->fd == NGX_INVALID_FILE && job->obj.driver == NULL)
171 || job->offset < 0
172 || (job->length > 0 && job->buf == NULL))
173 {
174 job->nio = -1;
175 job->io_errno = EINVAL;
176 return;
177 }
178
179 if (job->length == 0) {
180 job->nio = 0;
181 return;
182 }
183
184 if (brix_vfs_io_write_counted(&job->obj, job->fd, job->buf, job->length,
185 job->offset, &written, &short_io)
186 != NGX_OK)
187 {
188 job->nio = short_io ? written : -1;
189 job->out_size = written > 0 ? (size_t) written : 0;
190 job->short_io = short_io ? 1 : 0;
191 job->io_errno = errno;
192 return;
193 }
194
195 job->nio = written;
196 job->out_size = (size_t) written;
197
198 /* xmeta P3: fold the written blocks' CRCs into the handle-local table
199 * (pure in-memory; the record is merged once at close). Failure is
200 * non-fatal to the data write — those blocks simply stay unverified. */
201 if (job->csi != NULL && written > 0) {
202 (void) brix_csi_write_update((brix_csi_t *) job->csi, job->buf,
203 job->offset, (size_t) written);
204 }
205}

207/* brix_vfs_io_execute_pgread — read straight into the final pgread wire buffer
208 * with per-page CRC32c words written in place (brix_pgread_read_encode_inplace),
209 * giving pgread the same worker contract as a plain read. */
210static void
211brix_vfs_io_execute_pgread(brix_vfs_job_t *job)
212{
213 brix_pgread_io_t io = { .nowait = 0 /* blocking */, .nread = 0,
214 .io_errno = 0 };
215
216 if ((job->fd == NGX_INVALID_FILE && job->obj.driver == NULL)
217 || job->offset < 0
218 || (job->length > 0 && job->buf == NULL)
219 || job->buf_cap < job->length)
220 {
221 job->nio = -1;
222 job->io_errno = EINVAL;
223 return;
224 }
225
226 {
227 brix_sd_obj_t scratch;
228 brix_sd_obj_t *obj = brix_vfs_effective_obj(&job->obj, job->fd,
229 &scratch);
230 job->out_size = brix_pgread_read_encode_inplace(obj, job->offset,
231 job->length, job->buf,
232 &io);
233 }
234 job->nio = io.nread;
235 if (io.nread < 0) {
236 job->io_errno = io.io_errno;
237 }
238}

240/* brix_vfs_io_execute_readv — run the pre-built, pre-validated coalesced readv
241 * plan over job->segs (brix_readv_read_segments); out_size = per-segment headers
242 * + payload bytes on success. */
243static void
244brix_vfs_io_execute_readv(brix_vfs_job_t *job)
245{
246 size_t bytes_read_total;
247
248 if (job->segs == NULL || job->nsegs == 0) {
249 job->nio = -1;
250 job->io_errno = EINVAL;
251 brix_vfs_io_set_error_message(job, "readv segment count out of range");
252 return;
253 }
254
255 bytes_read_total = 0;
256 if (brix_readv_read_segments((brix_readv_seg_desc_t *) job->segs,
257 job->nsegs, &bytes_read_total,
258 job->err_msg, job->err_msg_cap)
259 != NGX_OK)
260 {
261 job->nio = -1;
262 job->io_errno = EIO;
263 if (job->err_msg != NULL && job->err_msg_cap > 0
264 && job->err_msg[0] == '\0')
265 {
266 brix_vfs_io_set_error_message(job, "readv I/O error");
267 }
268 return;
269 }
270
271 job->nio = (ssize_t) bytes_read_total;
272 job->out_size = job->nsegs * BRIX_READV_SEGSIZE + bytes_read_total;
273}

275/* brix_vfs_io_execute_writev — counted pwrite of each writev segment (first
276 * error wins, with a segment-tagged message), accumulating out_size, then a
277 * best-effort fsync of the non-empty segment fds when job->do_sync. */
278static void
279brix_vfs_io_execute_writev(brix_vfs_job_t *job)
280{
281 brix_vfs_writev_seg_t *segments;
282 size_t segment_index;
283
284 if (job->segs == NULL || job->nsegs == 0) {
285 job->nio = -1;
286 job->io_errno = EINVAL;
287 brix_vfs_io_set_error_message(job, "writev segment count out of range");
288 return;
289 }
290
291 segments = job->segs;
292 for (segment_index = 0; segment_index < job->nsegs; segment_index++) {
293 brix_vfs_writev_seg_t *segment;
294 ssize_t written;
295 unsigned short_io;
296
297 segment = &segments[segment_index];
298 if (segment->wlen == 0) {
299 continue;
300 }
301
302 if (brix_vfs_io_write_counted(&segment->obj, segment->fd,
303 segment->data,
304 (size_t) segment->wlen,
305 segment->offset, &written, &short_io)
306 != NGX_OK)
307 {
308 job->nio = short_io ? written : -1;
309 job->io_errno = errno;
310 job->short_io = short_io ? 1 : 0;
311 if (short_io) {
312 if (job->err_msg != NULL && job->err_msg_cap > 0) {
313 snprintf(job->err_msg, job->err_msg_cap,
314 "writev short write at seg %d",
315 (int) segment_index);
316 }
317 } else {
318 char prefix[64];
319
320 snprintf(prefix, sizeof(prefix),
321 "writev I/O error at seg %d", (int) segment_index);
322 brix_vfs_io_set_errno_message(job, prefix, job->io_errno);
323 }
324 return;
325 }
326
327 job->out_size += (size_t) written;
328 }
329
330 if (job->do_sync) {
331 for (segment_index = 0; segment_index < job->nsegs; segment_index++) {
332 if (segments[segment_index].wlen > 0) {
333 (void) fsync(segments[segment_index].fd);
334 }
335 }
336 }
337
338 job->nio = (ssize_t) job->out_size;
339}

341/* brix_vfs_io_execute_sync — fsync one open fd via the Storage Driver seam so
342 * durability routes through the same chokepoint as writes; nio=0 on success,
343 * -1/io_errno on failure. */
344static void
345brix_vfs_io_execute_sync(brix_vfs_job_t *job)
346{
347 brix_sd_obj_t obj;
348
349 if (job->fd == NGX_INVALID_FILE && job->obj.driver == NULL) {
350 job->nio = -1;
351 job->io_errno = EINVAL;
352 return;
353 }
354
355 /* Durability via the shared `vfs` core (xvfs_fsync → backend driver). */
356 if (xvfs_fsync(brix_vfs_effective_obj(&job->obj, job->fd, &obj)) != 0) {
357 job->nio = -1;
358 job->io_errno = errno;
359 return;
360 }
361
362 job->nio = 0;
363}

365/* brix_vfs_io_execute_truncate — ftruncate one open fd to job->offset via the
366 * Storage Driver seam (truncation mutates file data, so it belongs on the VFS
367 * syscall chokepoint); nio=0 on success, -1/io_errno on failure. */
368static void
369brix_vfs_io_execute_truncate(brix_vfs_job_t *job)
370{
371 brix_sd_obj_t obj;

387 return;
388 }

390 job->nio = 0;

406/* brix_vfs_io_dirlist_name_unsafe — 1 if a dir entry must be skipped: "."/".."
407 * , the ".nginx-xrootd" control-file prefix, or any control/DEL byte (the
408 * newline-delimited kXR_dirlist body would otherwise be corrupted/polluted). */
409static ngx_flag_t
410brix_vfs_io_dirlist_name_unsafe(const char *name)
411{
412 const u_char *p;
413
414 if (name[0] == '.' && (name[1] == '\0'
415 || (name[1] == '.' && name[2] == '\0')))
416 {
417 return 1;
418 }
419
420 if (ngx_strncmp(name, ".nginx-xrootd",
421 sizeof(".nginx-xrootd") - 1) == 0)
422 {
423 return 1;
424 }
425
426 for (p = (const u_char *) name; *p != '\0'; p++) {
427 if (*p < 0x20 || *p == 0x7f) {
428 return 1;
429 }
430 }
431
432 return 0;
433}

446/* brix_vfs_io_dirlist_need_new_chunk — if the next entry won't fit the current
447 * chunk, flush a kXR_oksofar chunk and reset the data cursor; fails E2BIG when one
448 * more full chunk would exceed job->buf_cap (preserving the old fixed-buffer bound). */
449static ngx_int_t
450brix_vfs_io_dirlist_need_new_chunk(brix_vfs_job_t *job, u_char *out,
451 size_t *base, u_char **cdata, size_t *cdpos, size_t need)
452{
453 if (*cdpos + need <= BRIX_VFS_DIRLIST_CHUNK_CAP) {
454 return NGX_OK;
455 }
456
457 brix_vfs_io_dirlist_flush_chunk(job, out, base, *cdpos);
458 if (*base + XRD_RESPONSE_HDR_LEN + BRIX_VFS_DIRLIST_CHUNK_CAP
459 > job->buf_cap)
460 {
461 char message[96];
462
463 snprintf(message, sizeof(message),
464 "listing too large for AIO buffer (%zu bytes)",
465 job->buf_cap);
466 brix_vfs_io_dirlist_fail(job, E2BIG, message);
467 return NGX_ERROR;
468 }
469
470 *cdata = out + *base + XRD_RESPONSE_HDR_LEN;
471 *cdpos = 0;
472 return NGX_OK;
473}

475/* brix_vfs_io_dirlist_stat_entry — for job->want_stat, fstatat the entry
476 * (no-follow) and format the stat body plus optional checksum token, adding the
477 * extra wire bytes to *need; NGX_DECLINED skips a vanished/unstattable entry. */
478static ngx_int_t
479brix_vfs_io_dirlist_stat_entry(brix_vfs_job_t *job, int dfd,
480 const char *name, char *statbuf, size_t statbuf_cap,
481 char *cksum_token, size_t cksum_token_cap, size_t *need)
482{
483 struct stat entry_st;
484
485 statbuf[0] = '\0';
486 cksum_token[0] = '\0';
487
488 if (!job->want_stat) {
489 return NGX_OK;
490 }
491
492 /* dfd < 0 (a failed dirfd upstream) and any fstatat failure — vanished
493 * entry or otherwise — take the same path: skip the entry. */
494 if (dfd < 0 || fstatat(dfd, name, &entry_st, AT_SYMLINK_NOFOLLOW) != 0) {
495 return NGX_DECLINED;
496 }
497
498 if (job->want_cksum) {
499 char entry_path[PATH_MAX];
500 const char *algo;
501 int n;
502
503 brix_dirlist_make_dcksm_stat_body(&entry_st, statbuf, statbuf_cap);
504 algo = job->cksum_algo != NULL ? job->cksum_algo : "unknown";
505 n = snprintf(entry_path, sizeof(entry_path), "%s/%s",
506 job->path != NULL ? job->path : "", name);
507 if (n < 0 || (size_t) n >= sizeof(entry_path)) {
508 snprintf(cksum_token, cksum_token_cap, "%s:none", algo);
509 } else {
510 brix_dirlist_checksum_token(job->log, dfd, name, entry_path,
511 &entry_st, algo, cksum_token,
512 cksum_token_cap);
513 }
514 *need += strlen(cksum_token) + sizeof(" [ ]") - 1;
515
516 } else {
517 brix_make_stat_body(&entry_st, 0, 0, statbuf, statbuf_cap);
518 }
519
520 *need += strlen(statbuf) + 1;
521 return NGX_OK;
522}

558/* brix_vfs_io_execute_opendir — build the complete kXR_dirlist flat response in
559 * job->buf by fdopendir'ing a dup of the confined rootfd (so the scan stays
560 * anchored to the directory approved on the event loop, not reopened by path):
561 * optional dstat lead-in, then filtered entries with fd-relative fstat/checksum,
562 * then the final kXR_ok header. */
563static void
564brix_vfs_io_execute_opendir(brix_vfs_job_t *job)
565{
566 DIR *dp;
567 int scanfd;
568 u_char *out;
569 u_char *cdata;
570 size_t base;
571 size_t cdpos;
572
573 if (job->rootfd < 0 || job->buf == NULL) {
574 brix_vfs_io_dirlist_fail(job, EINVAL, "invalid opendir job");
575 return;
576 }
577
578 if (job->buf_cap < (size_t) (XRD_RESPONSE_HDR_LEN
579 + BRIX_VFS_DIRLIST_CHUNK_CAP))
580 {
581 brix_vfs_io_dirlist_fail(job, ENOMEM, "response buffer too small");
582 return;
583 }
584
585 scanfd = dup(job->rootfd);
586 if (scanfd < 0) {
587 brix_vfs_io_dirlist_fail(job, errno, NULL);
588 return;
589 }
590
591 dp = fdopendir(scanfd);
592 if (dp == NULL) {
593 int err;
594
595 err = errno;
596 close(scanfd);
597 brix_vfs_io_dirlist_fail(job, err, NULL);
598 return;
599 }
600
601 out = job->buf;
602 base = 0;
603 cdpos = 0;
604 cdata = out + XRD_RESPONSE_HDR_LEN;
605
606 if (job->want_stat) {
607 ngx_memcpy(cdata, BRIX_DSTAT_LEADIN, BRIX_DSTAT_LEADIN_LEN);
608 cdpos = BRIX_DSTAT_LEADIN_LEN;
609 }
610
611 for (;;) {
612 struct dirent *de;
613 const char *name;
614 size_t nlen;
615 size_t need;
616 char statbuf[256];
617 char cksum_token[EVP_MAX_MD_SIZE * 2 + 64];
618
619 errno = 0;
620 de = readdir(dp);
621 if (de == NULL) {
622 if (errno != 0) {
623 int err = errno;
624
625 closedir(dp);
626 brix_vfs_io_dirlist_fail(job, err, NULL);
627 return;
628 }
629 break;
630 }
631
632 name = de->d_name;
633 if (brix_vfs_io_dirlist_name_unsafe(name)) {
634 continue;
635 }
636
637 nlen = strlen(name);
638 need = nlen + 1;
639 if (brix_vfs_io_dirlist_stat_entry(job, dirfd(dp), name,
640 statbuf, sizeof(statbuf),
641 cksum_token, sizeof(cksum_token),
642 &need)
643 == NGX_DECLINED)
644 {
645 continue;
646 }
647
648 if (brix_vfs_io_dirlist_need_new_chunk(job, out, &base, &cdata,
649 &cdpos, need)
650 != NGX_OK)
651 {
652 closedir(dp);
653 return;
654 }
655
656 brix_vfs_io_dirlist_emit_entry(job, cdata, &cdpos, name, nlen,
657 statbuf, cksum_token);
658 }
659
660 closedir(dp);
661
662 if (cdpos > 0) {
663 cdata[cdpos - 1] = '\0';
664 }
665 brix_build_resp_hdr(job->streamid, kXR_ok, (uint32_t) cdpos,
666 (ServerResponseHdr *) (out + base));
667 job->out_size = base + XRD_RESPONSE_HDR_LEN + cdpos;
668 job->nio = 0;
669}

671/* brix_vfs_io_reset_outputs — clear every job OUT field (and NUL the caller's
672 * error buffer) before execution, since nginx thread tasks are reused across
673 * requests and stale errors or byte counts must not escape. */
674static void

676{
677 job->nio = 0;
678 job->out_size = 0;
679 job->crc32c = 0;
680 job->io_errno = 0;
681 job->short_io = 0;
682 if (job->err_msg != NULL && job->err_msg_cap > 0) {

687/* brix_vfs_io_account — post-op per-backend byte attribution: map the job's
688 * op to a read/write direction and add job->nio bytes to the backend totals.
689 * The job's obj names the bound driver; NULL ⇒ default POSIX. READV totals
690 * attribute to the job's primary obj (segments could in principle span
691 * handles — a bounded, deliberate approximation). Non-data ops no-op. */
692static void
693brix_vfs_io_account(const brix_vfs_job_t *job)
694{
695 brix_metric_op_t dir;
696
697 if (job->nio <= 0) {
698 return;
699 }
700
701 switch (job->op) {
702 case BRIX_VFS_IO_READ:
703 case BRIX_VFS_IO_PGREAD:
704 case BRIX_VFS_IO_READV:
705 dir = BRIX_METRIC_OP_READ;
706 break;
707 case BRIX_VFS_IO_WRITE:
708 case BRIX_VFS_IO_WRITEV:
709 dir = BRIX_METRIC_OP_WRITE;
710 break;
711 default:
712 return;
713 }
714
715 brix_metric_backend_bytes(
716 job->obj.driver != NULL ? job->obj.driver->name : "posix",
717 dir, (size_t) job->nio);
718}
719
720/* brix_vfs_io_execute — the single VFS-owned raw-I/O chokepoint: clear the OUT
721 * fields, dispatch *job to one static executor per op (EINVAL on a NULL or
722 * unknown job), then attribute the moved bytes to the job's backend. Callable
723 * from stream AIO workers and inline fallbacks without touching
724 * event-loop-only state. */
725void
726brix_vfs_io_execute(brix_vfs_job_t *job)
727{
728 if (job == NULL) {
729 return;
730 }
731
732 brix_vfs_io_reset_outputs(job);
733
734 switch (job->op) {
735 case BRIX_VFS_IO_READ:
736 brix_vfs_io_execute_read(job);
737 break;
738 case BRIX_VFS_IO_WRITE:
739 brix_vfs_io_execute_write(job);
740 break;
741 case BRIX_VFS_IO_PGREAD:
742 brix_vfs_io_execute_pgread(job);
743 break;
744 case BRIX_VFS_IO_READV:
745 brix_vfs_io_execute_readv(job);
746 break;
747 case BRIX_VFS_IO_WRITEV:
748 brix_vfs_io_execute_writev(job);
749 break;
750 case BRIX_VFS_IO_SYNC:
751 brix_vfs_io_execute_sync(job);
752 break;
753 case BRIX_VFS_IO_TRUNCATE:
754 brix_vfs_io_execute_truncate(job);
755 break;
756 case BRIX_VFS_IO_OPENDIR:
757 brix_vfs_io_execute_opendir(job);
758 break;
759 default:
760 job->nio = -1;
761 job->io_errno = EINVAL;
762 return;
763 }
764
765 brix_vfs_io_account(job);
766}
