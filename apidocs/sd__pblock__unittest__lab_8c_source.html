70/* F1: a fault.pread rule injects EIO on read for handles opened after it is set;
71 * a read on a handle opened with no rule succeeds (success + error). */
72void
73test_lab_fault_inject(void)
74{
75 char root[] = "/tmp/pb_lab.XXXXXX";
76 brix_sd_instance_t inst = {0};
77 brix_sd_pblock_conf_t conf = {0}; /* zero enforce_unprivileged: never drop in the unit test */
78 char buf[32];
79 int err = 0;
80 brix_sd_obj_t *o;
81
82 CHECK(mkdtemp(root) != NULL, "mkdtemp");
83 lab_write_sidecar(root, "lab=1");
84 conf.root = root;
85 conf.busy_timeout_ms = 2000;
86 conf.block_size = 0;
87 inst.driver = D;
88 inst.caps = D->caps;
89 CHECK(D->init(&inst, &conf) == NGX_OK, "lab init");
90
91 CHECK(write_file(&inst, "/f", "hello", 5) == 0, "seed write");
92
93 /* SUCCESS: no read fault yet → clean read. */
94 o = D->open(&inst, "/f", BRIX_SD_O_READ, 0, &err);
95 CHECK(o != NULL, "open read (clean)");
96 if (o != NULL) {
97 CHECK(D->pread(o, buf, sizeof(buf), 0) == 5, "clean pread");
98 pb_close(o);
99 }
100
101 /* ERROR: set a read fault, reopen → the new handle's snapshot injects EIO. */
102 lab_ctl_set(root, "fault.pread", "errno=EIO", 1);
103 o = D->open(&inst, "/f", BRIX_SD_O_READ, 0, &err);
104 CHECK(o != NULL, "open read (faulted)");
105 if (o != NULL) {
106 errno = 0;
107 CHECK(D->pread(o, buf, sizeof(buf), 0) == -1 && errno == EIO,
108 "faulted pread → EIO");
109 pb_close(o);
110 }
111 D->cleanup(&inst);
112}

171/* F14: catalog enumeration is a flat scan reporting every stored file (never a
172 * directory) at any depth in one pass — three independent oracles (enumerate
173 * count, enumerate size-sum, a recursive namespace walk) must agree. The scan is
174 * UNCONFINED: a nested object comes back by its full path with no subtree gate,
175 * which is precisely why enumeration is a service-plane inventory verb — there is
176 * no cred-scoped variant, so it is unreachable from user protocols. An aborting
177 * callback stops the walk early yet the slot still returns NGX_OK. */
178typedef struct {
179 int files; /* non-directory rows reported */
180 int64_t bytes; /* summed size (valid only when want_stat) */
181 int saw_nested; /* the nested /d/c path was reported */
182 int stop_after; /* 0 = run to completion, else abort after N */
183} enum_probe_t;
184
185static int
186enum_probe_cb(void *ctx, const brix_sd_catalog_ent_t *ent)
187{
188 enum_probe_t *p = ctx;
189
190 if (ent->key == NULL) {
191 return 0;
192 }
193 p->files++;
194 if (ent->have_stat) {
195 p->bytes += (int64_t) ent->size;
196 }
197 if (ent->path != NULL && strcmp(ent->path, "/d/c") == 0) {
198 p->saw_nested = 1;
199 }
200 return (p->stop_after && p->files >= p->stop_after) ? 1 : 0;
201}
202
203/* Independent oracle: recursively walk the namespace, counting regular files and
204 * summing their stat sizes. opendir/readdir/stat only — no catalog SELECT. */
205static void
206enum_walk(brix_sd_instance_t *inst, const char *dir, int *files, int64_t *bytes)
207{
208 brix_sd_dir_t *d;
209 brix_sd_dirent_t de;
210 int err = 0;
211
212 d = D->opendir(inst, dir, &err);
213 if (d == NULL) {
214 return;
215 }
216 while (D->readdir(d, &de) == NGX_OK) {
217 char child[512];
218 brix_sd_stat_t st;
219
220 (void) snprintf(child, sizeof(child), "%s%s%s", dir,
221 (strcmp(dir, "/") == 0) ? "" : "/", de.name);
222 if (D->stat(inst, child, &st) != NGX_OK) {
223 continue;
224 }
225 if (st.is_dir) {
226 enum_walk(inst, child, files, bytes);
227 } else {
228 (*files)++;
229 *bytes += (int64_t) st.size;
230 }
231 }
232 D->closedir(d);
233}

235void
236test_lab_enumerate(void)
237{
238 char root[] = "/tmp/pb_enum.XXXXXX";
239 brix_sd_instance_t inst = {0};
240 brix_sd_pblock_conf_t conf = {0};
241 enum_probe_t p;
242 int walk_files = 0;
243 int64_t walk_bytes = 0;
244
245 CHECK(mkdtemp(root) != NULL, "mkdtemp");
246 conf.root = root;
247 conf.busy_timeout_ms = 2000;
248 conf.block_size = 0;
249 inst.driver = D;
250 inst.caps = D->caps;
251 CHECK(D->init(&inst, &conf) == NGX_OK, "enum init");
252
253 /* three files (distinct sizes 1/3/5 = 9 bytes) + one directory. */
254 CHECK(write_file(&inst, "/a", "x", 1) == 0, "a");
255 CHECK(write_file(&inst, "/b", "yyy", 3) == 0, "b");
256 CHECK(D->mkdir(&inst, "/d", 0755) == NGX_OK, "mkdir d");
257 CHECK(write_file(&inst, "/d/c", "zzzzz", 5) == 0, "c");
258
259 /* SUCCESS + THREE-ORACLE AGREEMENT: enumerate (want_stat) reports 3 files
260 * summing to 9 bytes, and a recursive namespace walk reports the same — the
261 * /d directory is excluded, the nested /d/c is present (unconfined scan). */
262 memset(&p, 0, sizeof(p));
263 CHECK(D->enumerate(&inst, 1, enum_probe_cb, &p) == NGX_OK, "enumerate ok");
264 CHECK(p.files == 3, "enumerate counted %d files (want 3)", p.files);
265 CHECK(p.bytes == 9, "enumerate size-sum %lld (want 9)", (long long) p.bytes);
266 CHECK(p.saw_nested, "nested /d/c absent — scan must be unconfined");
267
268 enum_walk(&inst, "/", &walk_files, &walk_bytes);
269 CHECK(walk_files == p.files && walk_bytes == p.bytes,
270 "oracle drift: walk=%d/%lld enumerate=%d/%lld",
271 walk_files, (long long) walk_bytes, p.files, (long long) p.bytes);
272
273 /* EARLY-ABORT: cb aborts after one row; slot still NGX_OK. */
274 memset(&p, 0, sizeof(p));
275 p.stop_after = 1;
276 CHECK(D->enumerate(&inst, 0, enum_probe_cb, &p) == NGX_OK, "enumerate abort ok");
277 CHECK(p.files == 1, "abort stopped after %d (want 1)", p.files);
278
279 /* SECURITY-NEG: the enumerate verb is service-plane only — there is no
280 * cred-scoped enumerate slot in the driver vtable, so an identity/user
281 * protocol path can never reach the unconfined catalog scan. */
282 CHECK(D->enumerate != NULL, "enumerate advertised");
283 CHECK((D->caps & BRIX_SD_CAP_CATALOG) != 0, "CAP_CATALOG advertised");
284
285 D->cleanup(&inst);
286}
