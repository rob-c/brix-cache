79/* ------------------------------------------------------------------ */
80/* Vendor extension opcodes (nginx-xrootd local — NOT standard XRootD) */
81/* ------------------------------------------------------------------ */
82/*
83 * These opcodes close POSIX gaps the base XRootD protocol has no wire op for
84 * (set-mtime / chown / symlink / hard-link), so a FUSE mount can honour
85 * `cp -p`, `touch -d`, and `ln`/`ln -s`. They are deliberately placed well above
86 * the standard range (max real opcode is kXR_clone=3032) to avoid any future
87 * collision, and are CAPABILITY-NEGOTIATED: the server advertises support via
88 * kXR_Qconfig "xrdfs.ext" and the native client only emits them when advertised,
89 * so a stock XRootD server never receives one. The client-side per-opcode RTT
90 * table (frame.c) bounds-checks reqid-kXR_1stRequest < XRDC_NOP, so these high
91 * ids simply skip RTT accounting — no out-of-bounds access.
92 */
93#define kXR_setattr 3500 /* set times (utimens) and/or owner (chown) on a path */
94#define kXR_symlink 3501 /* create a symbolic link */
95#define kXR_readlink 3502 /* read a symbolic link's target */
96#define kXR_link 3503 /* create a hard link */

98/* ------------------------------------------------------------------ */
99/* Response status codes (ServerResponseHdr.status) */
100/* ------------------------------------------------------------------ */
101/*
102 * kXR_ok and kXR_oksofar are success paths; everything above 4000 carries
103 * additional data in the response body. kXR_oksofar means "there is more
104 * data coming for this request — keep listening for the next response frame
105 * with the same streamid".
106 */
107#define kXR_ok 0 /* request succeeded; body (if any) is the result */
108#define kXR_oksofar 4000 /* partial result — more response frames follow */
109#define kXR_attn 4001 /* unsolicited server push notification */
110#define kXR_authmore 4002 /* authentication needs another round-trip;

112#define kXR_error 4003 /* request failed; body = errnum[4] + errmsg (NUL-terminated) */
113#define kXR_redirect 4004 /* client should retry at another server;

115#define kXR_wait 4005 /* try again after N seconds (body carries N as uint32) */
116#define kXR_waitresp 4006 /* async result is coming in a future kXR_attn frame */

118/* ------------------------------------------------------------------ */
119/* kXR_attn action codes (body field actnum, big-endian) */
120/* ------------------------------------------------------------------ */
121/*
122 * These are carried in the 4-byte actnum field at the start of a kXR_attn
123 * body. All values 5000-5007 are deprecated ("No longer supported" in the
124 * v5 spec); only kXR_asyncms and kXR_asynresp are still active.
125 *
126 * kXR_asyncms — server-push text notification (unsolicited).
127 * Body layout: actnum[4] + reserved[4] +
128 * ServerResponseHdr[8] + message[dlen].
129 * Outer kXR_attn streamid is {0,0}.
130 *
131 * kXR_asynresp — deferred response to a kXR_waitresp-acknowledged request.
132 * Same body layout; inner ServerResponseHdr carries the
133 * original request's streamid and actual status.
134 * Outer kXR_attn streamid mirrors the deferred streamid.
135 */
136#define kXR_asyncms 5002 /* active: unsolicited server notification message */
137#define kXR_asynresp 5008 /* active: deferred response completion */
138#define kXR_status 4007 /* extended status with CRC32c integrity check;
139 used for kXR_pgwrite and kXR_pgread responses */
