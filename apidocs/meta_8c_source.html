39/*
40 * brix_cache_meta_read — load the sidecar meta for cache_path into *meta.
41 *
42 * Derives the sidecar path, opens it O_RDONLY|O_NOFOLLOW, and reads the full
43 * fixed-size record. Returns NGX_OK on success; NGX_DECLINED when no sidecar
44 * exists (ENOENT), the file is truncated, or etag_len is out of range (treat as
45 * cache miss); NGX_ERROR on any other I/O error.
46 */
47/* Legacy fixed-size ".meta" sidecar reader (pre-xmeta-migration caches). The
48 * on-disk layout is exactly the brix_cache_meta_t legacy base — origin mtime,
49 * size, and a bounded etag — so a cache written before the unified metadata
50 * record still validates. NGX_OK on a good record; NGX_DECLINED (cache miss)
51 * when the sidecar is absent, truncated, or its etag length is out of range. */
52static ngx_int_t
53cache_meta_read_legacy(const char *cache_path, brix_cache_meta_t *meta)
54{
55 char path[PATH_MAX];
56 int fd;
57 ssize_t n;
58
59 if (brix_cache_meta_path(path, sizeof(path), cache_path) != 0) {
60 return NGX_DECLINED;
61 }
62 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC); /* vfs-seam-allow: server-managed cache-root sidecar, opened as worker */
63 if (fd < 0) {
64 return NGX_DECLINED; /* ENOENT / no legacy sidecar -> cache miss */
65 }
66 ngx_memzero(meta, sizeof(*meta));
67 n = read(fd, meta, BRIX_CACHE_META_BASE_SIZE);
68 (void) close(fd);
69 if (n != (ssize_t) BRIX_CACHE_META_BASE_SIZE
70 || meta->etag_len > BRIX_CACHE_META_ETAG_MAX)
71 {
72 return NGX_DECLINED; /* truncated / corrupt -> treat as miss */
73 }
74 meta->version = 0; /* legacy base: no versioned tail */
75 return NGX_OK;
76}

78ngx_int_t
79brix_cache_meta_read(ngx_log_t *log, const char *cache_path,
80 brix_cache_meta_t *meta)
81{
82 brix_xmeta_t xm;
83 int rc;
84
85 (void) log;
86
87 if (cache_path == NULL || meta == NULL) {
88 errno = EINVAL;
89 return NGX_ERROR;
90 }
91 rc = brix_xmeta_path_load(cache_path, &xm);
92 if (rc != BRIX_XMETA_OK) {
93 /* No unified brix record (FOREIGN covers both "absent" and a foreign/
94 * stock record): fall back to a legacy ".meta" sidecar so caches written
95 * before the xmeta migration still validate. NGX_ERROR only on a genuine
96 * I/O fault loading the unified record. */
97 if (rc == BRIX_XMETA_FOREIGN) {
98 return cache_meta_read_legacy(cache_path, meta);
99 }
100 return NGX_ERROR;
101 }
102
103 ngx_memzero(meta, sizeof(*meta));
104 meta->mtime = xm.origin_mtime;
105 meta->size = (uint64_t) xm.file_size;
106 meta->version = BRIX_CACHE_META_VERSION;
107 meta->access_count = xm.access_cnt;
108 meta->last_access = (uint64_t) xm.astat.detach_time;
109 meta->bytes_served = (uint64_t) xm.astat.bytes_hit;
110 meta->etag_len = (xm.etag_len <= sizeof(meta->etag)) ? xm.etag_len : 0;
111 ngx_memcpy(meta->etag, xm.etag, meta->etag_len);
112 meta->cks_alg_len = (xm.cks_alg_len <= sizeof(meta->cks_alg))
113 ? xm.cks_alg_len : 0;
114 ngx_memcpy(meta->cks_alg, xm.cks_alg, meta->cks_alg_len);
115 meta->cks_len = (xm.cks_len < sizeof(meta->cks_hex)) ? xm.cks_len : 0;
116 ngx_memcpy(meta->cks_hex, xm.cks_hex, meta->cks_len);
117 meta->cks_hex[meta->cks_len] = 0;
118 brix_xmeta_free(&xm);
119 return NGX_OK;
120}

122/* Apply *meta's fields onto a loaded (or fresh) record, preserving whatever

126{
127 xm->origin_mtime = meta->mtime;
128 xm->access_cnt = meta->access_count;

135 xm->astat.bytes_hit = (int64_t) meta->bytes_served;
136 }
137 xm->etag_len = (meta->etag_len <= sizeof(xm->etag)) ? meta->etag_len : 0;

151{

154
155 (void) log;
156
157 if (cache_path == NULL) {
158 errno = EINVAL;
159 return NGX_ERROR;
160 }
161 lockfd = brix_xmeta_path_lock(cache_path);
162 if (lockfd < 0) {
163 return NGX_ERROR;

165 rc = brix_xmeta_path_load(cache_path, &xm);
166 if (rc != BRIX_XMETA_OK) {

177 xm.astat.bytes_hit += (int64_t) nbytes;
178
179 rc = (brix_xmeta_path_save(cache_path, &xm) == BRIX_XMETA_OK)
180 ? NGX_OK : NGX_ERROR;
181 brix_xmeta_free(&xm);
182 brix_xmeta_path_unlock(lockfd);
