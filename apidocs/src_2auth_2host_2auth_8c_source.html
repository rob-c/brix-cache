8/*
9 * auth.c — XRootD `host` (host-based) authentication handler — Phase 52 WS-C.
10 *
11 * WHAT: Implements the kXR_auth handler for the XRootD `host` security protocol.
12 * The client merely SELECTS `host`; it asserts no identity. The server
13 * reverse-resolves the peer's socket address to a hostname and checks it
14 * against the configured allowlist (brix_host_allow: exact hostnames or
15 * ".suffix" domain suffixes). On a match the peer is authenticated AS that
16 * hostname (ctx->login.dn).
17 *
18 * WHY: `host` is XRootD's weakest, oldest scheme — a hostname (or DNS, lacking
19 * DNSSEC) is spoofable, so it is for trusted closed networks only. It is
20 * therefore fail-closed: it must be explicitly selected (conf->auth ==
21 * BRIX_AUTH_HOST), an allowlist MUST be configured (empty = deny all),
22 * and the identity comes from the SOCKET reverse-DNS, never from any
23 * client-asserted name. Isolating it here keeps the spoofable surface in
24 * one auditable place.
25 *
26 * HOW: Downstream of kXR_login, inside the kXR_auth dispatcher (../gsi/auth.c)
27 * once it matched the "host" credtype. Reverse-resolve via
28 * brix_acc_resolve_peer() (the same getnameinfo path used by XrdAcc host
29 * rules, now circuit-breaker-bounded), match host_match(), set the identity
30 * + register + metrics, return kXR_ok — else kXR_NotAuthorized.
31 */
32
33/*
34 * Match a reverse-resolved peer hostname against one allowlist pattern.
35 * - a leading '.' pattern (".cern.ch") matches any host ending in it
36 * (case-insensitive), i.e. a domain suffix;
37 * - otherwise an exact (case-insensitive) hostname match.
38 */
39static ngx_flag_t
40brix_host_pattern_match(const char *host, const ngx_str_t *pat)
41{
42 size_t hlen = ngx_strlen(host);
43
44 if (pat->len == 0) {
45 return 0;
46 }
47 if (pat->data[0] == '.') {
48 /* domain suffix: host must be longer than the suffix and end with it */
49 if (hlen <= pat->len) {
50 return 0;
51 }
52 return ngx_strncasecmp((u_char *) host + (hlen - pat->len),
53 pat->data, pat->len) == 0;
54 }
55 return hlen == pat->len
56 && ngx_strncasecmp((u_char *) host, pat->data, pat->len) == 0;
57}

60brix_host_allowed(ngx_stream_brix_srv_conf_t *conf, const char *host)
61{
62 ngx_str_t *pats;
63 ngx_uint_t i;
64
65 if (conf->host_allow == NULL || conf->host_allow->nelts == 0) {
66 return 0; /* empty allowlist = deny all */
67 }
68 pats = conf->host_allow->elts;
69 for (i = 0; i < conf->host_allow->nelts; i++) {
70 if (brix_host_pattern_match(host, &pats[i])) {
71 return 1;
72 }
73 }
74 return 0;
75}
