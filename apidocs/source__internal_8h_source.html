4/*
5 * source_internal.h — cross-file seam for the TPC remote-source pull, split
6 * (phase-79 file-size burndown) from a single oversized source.c into three
7 * cohesive units with zero behaviour change:
8 *
9 * source.c — the public driver tpc_pull_from_source() (open→stream→close)
10 * source_open.c — Phase 1: build/send kXR_open, resolve the async reply,
11 * extract the origin fhandle (tpc_open_source)
12 * source_stream.c — Phase 2/3: kXR_read stream loop + fsync, and the
13 * best-effort remote close (tpc_stream_to_dst,
14 * tpc_close_source)
15 *
16 * Only the three phase entry points cross a file boundary; every framing/parse
17 * helper stays file-static in its own unit. Declared here so the driver and the
18 * two phase units share one contract.
19 */
