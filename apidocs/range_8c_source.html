1/*
2 * range.c — HTTP Range header parser shared by WebDAV and S3.
3 *
4 * WHAT: Parses an RFC 7233 "bytes=" Range header into a single byte-range
5 * (start, end). Used by the WebDAV GET handler for partial content
6 * responses and the S3 GET handler for Range requests.
7 *
8 * WHY: Both protocols need to handle three range forms — bytes=start-end,
9 * bytes=-suffix (last N bytes), bytes=start- (from start to EOF) — plus
10 * unsatisfiable ranges when the requested range exceeds file size. Centralising
11 * parsing here avoids duplication between WebDAV and S3 code.
12 *
13 * HOW: Defaults to full-file range. Checks header starts with "bytes=" prefix.
14 * Delegates to brix_http_parse_range_vector() for actual parsing (single-range mode).
15 * If vector parser returns NGX_OK → extract single range into out. If it fails →
16 * mark present=1, satisfiable=0 (unsatisfiable range). Returns no status code —
17 * caller checks out->present and out->satisfiable fields.
18 */

20/*
21 * WHAT: Parses an RFC 7233 "bytes=" Range header into a single byte-range.
22 *
23 * HOW: Set defaults to full-file (start=0, end=file_size-1). Reject headers that are NULL,
24 * shorter than 7 bytes, or don't start with "bytes=". Zero opts struct — max_ranges=1,
25 * allow_suffix=1, allow_open_ended=1. Call brix_http_parse_range_vector() on the
26 * suffix after "bytes=" to parse the range spec. If vector parser succeeds and nranges==1,
27 * copy range.start/range.end into out->start/out->end and set present=1. If vector parser
28 * fails or returns multiple ranges (not expected with max_ranges=1), set present=1,
29 * satisfiable=0.
30 *
31 * WHY: WebDAV GET and S3 GET handlers both need to honour HTTP Range headers for partial content.
32 * Delegates complex parsing logic to range_vector.c while providing a simple single-range API.
33 */
34

39void
40brix_http_parse_range(const unsigned char *hdr_val, size_t hdr_len,
41 off_t file_size, brix_http_range_t *out)
42{
43 brix_byte_range_t range;
44 brix_range_vector_opts_t opts;
45 ngx_uint_t nranges;
46
47 /* Defaults: full file, no range. */
48 out->start = 0;
49 out->end = (file_size > 0) ? file_size - 1 : 0;
50 out->present = 0;
51 out->satisfiable = 1;
52
53 if (hdr_val == NULL || hdr_len < 7 || memcmp(hdr_val, "bytes=", 6) != 0) {
54 return;
55 }
56
57 ngx_memzero(&opts, sizeof(opts));
58 opts.max_ranges = 1;
59 opts.allow_suffix = 1;
60 opts.allow_open_ended = 1;
61 opts.drop_unsatisfiable = 0;
62
63 if (brix_http_parse_range_vector(hdr_val + 6, hdr_len - 6,
64 file_size, &opts,
65 &range, &nranges) != NGX_OK)
66 {
67 out->present = 1;
68 out->satisfiable = 0;
69 return;
70 }
71
72 if (nranges == 1) {
73 out->present = 1;
74 out->start = range.start;
75 out->end = range.end;
76 }
77}

79/* Parse one run of decimal digits at *p (bounded by end) into *val; advance *p.
80 * Returns 1 if at least one digit was consumed (no overflow), else 0. */
81static int
82cr_parse_u64(const unsigned char **p, const unsigned char *end, off_t *val)
83{
84 const unsigned char *s = *p;
85 unsigned long long v = 0;
86
87 if (s >= end || *s < '0' || *s > '9') {
88 return 0;
89 }
90 for (; s < end && *s >= '0' && *s <= '9'; s++) {
91 if (v > (0x7fffffffffffffffULL - 9) / 10) {
92 return 0; /* overflow guard */
93 }
94 v = v * 10 + (unsigned long long) (*s - '0');
95 }
96 *val = (off_t) v;
97 *p = s;
98 return 1;
99}

101/* ---- Skip the optional "bytes " unit token and leading spaces ----
102 *
103 * WHAT: Advances *cursor past an optional "bytes " unit prefix (RFC 7233
104 * Content-Range grammar) followed by any run of single spaces, up to end.
105 *
106 * WHY: The unit token is optional in the values we accept, and callers must land
107 * on the first digit of the range regardless of whether it was present; this
108 * keeps that positioning quirk in one named step instead of inline in the
109 * orchestrator.
110 *
111 * HOW: 1. If at least 6 bytes remain and they equal "bytes ", advance 6.
112 * 2. Skip any following spaces one at a time until end or a non-space byte.
113 */
114static void
115cr_skip_unit_prefix(const unsigned char **cursor, const unsigned char *end)
116{
117 if ((size_t) (end - *cursor) >= 6 && memcmp(*cursor, "bytes ", 6) == 0) {
118 *cursor += 6;
119 }
120 while (*cursor < end && **cursor == ' ') {
121 (*cursor)++;
122 }
123}

125/* ---- Parse the "start-end" byte span into out->start / out->end ----
126 *
127 * WHAT: Parses "<start>-<end>" at *cursor, writing out->start and out->end and
128 * advancing *cursor. Returns 1 on a well-formed span, 0 otherwise.
129 *
130 * WHY: The start/end pair is the core of a Content-Range value; isolating its
131 * three-token grammar (number, literal '-', number) keeps the total-length
132 * and validation steps independent and separately testable.
133 *
134 * HOW: 1. Parse the start decimal; fail if absent.
135 * 2. Require a literal '-' separator; fail otherwise, then consume it.
136 * 3. Parse the end decimal; fail if absent.
137 */
138static int
139cr_parse_span(const unsigned char **cursor, const unsigned char *end,
140 brix_http_content_range_t *out)
141{
142 if (!cr_parse_u64(cursor, end, &out->start)) {
143 return 0;
144 }
145 if (*cursor >= end || **cursor != '-') {
146 return 0;
147 }
148 (*cursor)++;
149 if (!cr_parse_u64(cursor, end, &out->end)) {
150 return 0;
151 }
152 return 1;
153}

155/* ---- Parse the "/total" length component into out->total ----
156 *
157 * WHAT: Parses the "/<total>" or "/" + "*" form at *cursor, writing out->total
158 * (-1 for the "*" unknown-length form) and advancing *cursor. Returns 1 on success,
159 * 0 on a missing '/' or malformed total.
160 *
161 * WHY: RFC 7233 allows the instance length to be an explicit number or "*"
162 * (unknown); expressing that choice as a single helper keeps the "*" → -1
163 * sentinel convention in one place shared with the default init value.
164 *
165 * HOW: 1. Require a literal '/' separator; fail otherwise, then consume it.
166 * 2. If the next byte is '*', record total = -1, consume it, succeed.
167 * 3. Otherwise parse a decimal total into out->total.
168 */
169static int
170cr_parse_total(const unsigned char **cursor, const unsigned char *end,
171 brix_http_content_range_t *out)
172{
173 if (*cursor >= end || **cursor != '/') {
174 return 0;
175 }
176 (*cursor)++;
177 if (*cursor < end && **cursor == '*') {
178 out->total = -1;
179 (*cursor)++;
180 return 1;
181 }
182 return cr_parse_u64(cursor, end, &out->total);
183}

185/* ---- Confirm only tolerated trailing whitespace remains ----
186 *
187 * WHAT: Returns 1 if the bytes from cursor to end are entirely spaces/tabs (or
188 * empty), 0 if any other byte is present.
189 *
190 * WHY: A Content-Range value may carry trailing whitespace, but any other stray
191 * byte means the header is malformed and must be rejected; this makes the
192 * "consumed everything meaningful" check explicit at the end of the parse.
193 *
194 * HOW: 1. Skip a run of ' ' or '\t' bytes.
195 * 2. Report whether the cursor reached end.
196 */
197static int
198cr_only_trailing_ws(const unsigned char *cursor, const unsigned char *end)
199{
200 while (cursor < end && (*cursor == ' ' || *cursor == '\t')) {
201 cursor++;
202 }
203 return cursor == end;
204}

206/* ---- Parse an RFC 7233 Content-Range header into a byte span + total ----
207 *
208 * WHAT: Parses "[bytes ]start-end/total" (total may be "*") from hdr_val,
209 * populating out. Sets out->present=1 only on a fully well-formed value;
210 * leaves present=0 (and out unusable) on any malformation. No return value.
211 *
212 * WHY: WebDAV/S3 upload and TPC paths need the server-reported byte span and
213 * instance length; a single tolerant-but-strict parser keeps acceptance and
214 * rejection identical across those callers.
215 *
216 * HOW: 1. Initialise defaults (empty span, total=-1, present=0); bail on empty input.
217 * 2. Skip the optional unit prefix and leading spaces.
218 * 3. Parse the start-end span, then the /total component.
219 * 4. Reject an end-before-start span (malformed ordering).
220 * 5. Require nothing but tolerated trailing whitespace, then mark present.
221 */
222void
223brix_http_parse_content_range(const unsigned char *hdr_val, size_t hdr_len,
224 brix_http_content_range_t *out)
225{
226 const unsigned char *cursor, *end;
227
228 out->start = out->end = 0;
229 out->total = -1;
230 out->present = 0;
231
232 if (hdr_val == NULL || hdr_len == 0) {
233 return;
234 }
235 cursor = hdr_val;
236 end = hdr_val + hdr_len;
237
238 cr_skip_unit_prefix(&cursor, end);
239
240 if (!cr_parse_span(&cursor, end, out)) {
241 return;
242 }
243 if (!cr_parse_total(&cursor, end, out)) {
244 return;
245 }
246 if (out->end < out->start) {
247 return; /* malformed: end before start */
248 }
249 if (!cr_only_trailing_ws(cursor, end)) {
250 return;
251 }
252 out->present = 1;
253}
