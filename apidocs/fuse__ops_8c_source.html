26/* Retry budget exhausted? Deadline-bounded when `deadline` is set (ride the loss
27 * out for the patience window); else the legacy count bound (stop once `attempt`
28 * has reached `max`, i.e. max+1 total attempts — max==0 is a single attempt). */
29static int
30fuse_run_done(uint64_t deadline, unsigned attempt, unsigned max)
31{
32 if (deadline != 0) {
33 return brix_mono_ns() >= deadline;
34 }
35 return attempt >= max;
36}

38int
39brix_fuse_run(brix_pool *pool, int max_retries, int max_stall_ms,
40 int benign_errno, brix_fuse_op_fn op, void *ctx, brix_status *st)
41{
42 uint64_t deadline = (max_stall_ms > 0)
43 ? brix_mono_ns() + (uint64_t) max_stall_ms * 1000000ULL
44 : 0;
45 unsigned max = max_retries > 0 ? (unsigned) max_retries : 0;
46 unsigned attempt;
47
48 for (attempt = 0; ; attempt++) {
49 /* Exponential backoff + jitter BEFORE each retry (never the first), so a
50 * transient fault on a flaky link is ridden out without a reconnect
51 * storm and concurrent FUSE threads do not re-hammer in lockstep. */
52 if (attempt > 0) {
53 brix_backoff_sleep_fast(attempt - 1);
54 }
55
56 brix_conn *c = brix_pool_checkout(pool, st);
57 if (c == NULL) {
58 if (brix_status_retryable(st)
59 && !fuse_run_done(deadline, attempt, max)) {
60 continue;
61 }
62 return brix_fuse_errno(st);
63 }
64
65 int rc = op(c, ctx, st);
66 brix_pool_checkin(pool, c, rc == 0 ? 1 : brix_fuse_conn_healthy(st));
67 if (rc == 0) {
68 return 0;
69 }
70 /* Idempotency normalization for a re-issued mutation: once we have
71 * retried (attempt > 0), the first attempt may already have applied the
72 * change and had its reply lost to the sever. A benign "already in the
73 * desired state" code (EEXIST for mkdir/symlink/link, ENOENT for
74 * rm/rmdir/mv) then means success, not a spurious error. */
75 if (attempt > 0 && benign_errno != 0
76 && brix_kxr_to_errno(st) == benign_errno) {
77 brix_status_clear(st);
78 return 0;
79 }
80 if (!brix_status_retryable(st)
81 || fuse_run_done(deadline, attempt, max)) {
82 return brix_fuse_errno(st);
83 }
84 /* transient → backoff, then retry on a freshly (re)connected slot */
85 }
86}
