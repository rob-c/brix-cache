12/*
13 * WHAT: Emit owner property (authenticated DN or "anonymous"). Returns NGX_OK or NGX_ERROR.
14 *
15 * WHY: RFC 3744 D:owner identifies the resource owner. We use the authenticated
16 * client DN from the request context, or "anonymous" if unavailable.
17 *
18 * HOW: Check ctx for non-empty DN, XML-escape it, emit as <D:owner><D:href>.
19 */
20static ngx_int_t
21propfind_emit_owner(ngx_http_request_t *r, ngx_chain_t **head,
22 ngx_chain_t **tail, ngx_http_brix_webdav_req_ctx_t *ctx)
23{
24 ngx_pool_t *pool = r->pool;
25 const char *owner = (ctx != NULL && ctx->dn[0] != '\0')
26 ? ctx->dn : "anonymous";
27 char *safe_owner = webdav_escape_xml_text(pool, owner);
28
29 if (safe_owner == NULL
30 || brix_http_chain_appendf(pool, head, tail,
31 "<D:owner><D:href>%s</D:href></D:owner>",
32 safe_owner) == NULL)
33 {
34 return NGX_ERROR;
35 }
36
37 return NGX_OK;
38}

40/*
41 * WHAT: Emit current-user-privilege-set ACL property. Returns NGX_OK or NGX_ERROR.
42 *
43 * WHY: RFC 3744 requires reporting the effective privileges for the current user.
44 * We synthesize read-only or read-write privileges based on allow_write.
45 *
46 * HOW: Emit <D:current-user-privilege-set> with <D:read/>. If allow_write is
47 * enabled, add write/write-content/write-properties/bind/unbind.
48 */
49static ngx_int_t
50propfind_emit_current_privilege(ngx_http_request_t *r, ngx_chain_t **head,
51 ngx_chain_t **tail,
52 ngx_http_brix_webdav_loc_conf_t *conf)
53{
54 ngx_pool_t *pool = r->pool;
55
56 if (brix_http_chain_appendf(pool, head, tail,
57 "<D:current-user-privilege-set>"
58 "<D:privilege><D:read/></D:privilege>") == NULL)
59 {
60 return NGX_ERROR;
61 }
62
63 if (conf->common.allow_write
64 && brix_http_chain_appendf(pool, head, tail,
65 "<D:privilege><D:write/></D:privilege>"
66 "<D:privilege><D:write-content/></D:privilege>"
67 "<D:privilege><D:write-properties/></D:privilege>"
68 "<D:privilege><D:bind/></D:privilege>"
69 "<D:privilege><D:unbind/></D:privilege>") == NULL)
70 {
71 return NGX_ERROR;
72 }
73
74 if (brix_http_chain_appendf(pool, head, tail,
75 "</D:current-user-privilege-set>") == NULL)
76 {
77 return NGX_ERROR;
78 }
79
80 return NGX_OK;

129/*
130 * WHAT: Emit ACL element with all/read/write privileges. Returns NGX_OK or NGX_ERROR.
131 *
132 * WHY: RFC 3744 D:acl property describes the ACL entries for the resource.
133 * We advertise a simple all-principal grant with read and (optionally) write.
134 *
135 * HOW: Emit <D:acl><D:ace> with all principal, read privilege, and (if allow_write)
136 * write privilege.

138static ngx_int_t

145 "<D:acl><D:ace><D:principal><D:all/></D:principal>"
146 "<D:grant><D:privilege><D:read/></D:privilege>") == NULL)
147 {
148 return NGX_ERROR;
149 }
150

152 && brix_http_chain_appendf(pool, head, tail,

159 "</D:grant><D:protected/></D:ace></D:acl>") == NULL)

162 }

167/*
168 * Append the RFC 3744 ACL-related DAV: properties selected by `mask`.
169 * These are synthesized from the request's authenticated identity and the
170 * location's allow_write flag rather than from a real ACL store: owner is the
171 * client DN (or "anonymous"), and the privilege/ACL sets advertise read plus
172 * (only when writes are enabled) the write family. Each clause is gated on its
173 * own mask bit so PROP requests get exactly the properties they asked for.
174 */
175ngx_int_t
176propfind_append_acl_properties(ngx_http_request_t *r, ngx_chain_t **head,
177 ngx_chain_t **tail, unsigned mask)
178{
179 ngx_http_brix_webdav_loc_conf_t *conf;
180 ngx_http_brix_webdav_req_ctx_t *ctx;
181 ngx_pool_t *pool = r->pool;
182
183 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
184 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
185
186 if (mask & PF_OWNER) {
187 if (propfind_emit_owner(r, head, tail, ctx) != NGX_OK) {
188 return NGX_ERROR;
189 }
190 }
191
192 if ((mask & PF_GROUP)
193 && brix_http_chain_appendf(pool, head, tail, "<D:group/>") == NULL)
194 {
195 return NGX_ERROR;
196 }
197
198 if (mask & PF_CURRENT_PRIVILEGE) {
199 if (propfind_emit_current_privilege(r, head, tail, conf) != NGX_OK) {
200 return NGX_ERROR;
201 }
202 }
203
204 if (mask & PF_SUPPORTED_PRIVILEGE) {
205 if (propfind_emit_supported_privilege(r, head, tail) != NGX_OK) {
206 return NGX_ERROR;
207 }
208 }
209
210 if (mask & PF_ACL) {
211 if (propfind_emit_acl(r, head, tail, conf) != NGX_OK) {
212 return NGX_ERROR;
213 }
214 }
215
216 if ((mask & PF_ACL_RESTRICTIONS)
217 && brix_http_chain_appendf(pool, head, tail,
218 "<D:acl-restrictions><D:grant-only/><D:no-invert/>"
219 "</D:acl-restrictions>") == NULL)
220 {
221 return NGX_ERROR;
222 }
223
224 if ((mask & PF_PRINCIPAL_SET)
225 && brix_http_chain_appendf(pool, head, tail,
226 "<D:principal-collection-set/>") == NULL)
227 {
228 return NGX_ERROR;
229 }
230
231 return NGX_OK;
232}

235/*
236 * WHAT: Emit RFC 4331 quota properties for directories. Returns NGX_OK or NGX_ERROR.
237 *
238 * WHY: Quota properties describe a COLLECTION's storage quota. Emitting them on
239 * regular files breaks clients (gfal2/davix maps quota-used-bytes onto file
240 * size, so a file appears ~TB instead of its actual size).
241 *
242 * HOW: Gate on S_ISDIR, query filesystem usage via brix_fs_usage_stat(), emit
243 * quota-available-bytes and/or quota-used-bytes based on mask bits.
244 */
245static ngx_int_t
246propfind_emit_quota(ngx_http_request_t *r, ngx_chain_t **head,
247 ngx_chain_t **tail, unsigned mask, const char *path,
248 struct stat *sb)
249{
250 ngx_pool_t *pool = r->pool;
251
252 /*
253 * RFC 4331 quota properties describe a COLLECTION's storage quota, not a
254 * file's. Emitting them on a regular file is non-conformant and actively
255 * breaks clients: gfal2/davix maps <D:quota-used-bytes> onto st_size, so a
256 * file's reported size becomes the filesystem's used bytes (~TB) instead of
257 * getcontentlength. Stock XrdHttp never emits quota per-file. Gate to
258 * directories so files carry only getcontentlength as their size.
259 */
260 if ((mask & (PF_QUOTA_AVAILABLE | PF_QUOTA_USED)) && S_ISDIR(sb->st_mode)) {
261 brix_fs_usage_t fsu;
262 if (brix_fs_usage_stat(path, &fsu) == NGX_OK) {
263 if ((mask & PF_QUOTA_AVAILABLE)
264 && brix_http_chain_appendf(pool, head, tail,
265 "<D:quota-available-bytes>"
266 "%llu"
267 "</D:quota-available-bytes>",
268 (unsigned long long) fsu.available_bytes) == NULL)
269 return NGX_ERROR;
270 if ((mask & PF_QUOTA_USED)
271 && brix_http_chain_appendf(pool, head, tail,
272 "<D:quota-used-bytes>"
273 "%llu"
274 "</D:quota-used-bytes>",
275 (unsigned long long) fsu.used_bytes) == NULL)
276 return NGX_ERROR;
277 }
278 }
279
280 return NGX_OK;
281}

283/*
284 * WHAT: Emit displayname property (last path segment of href). Returns NGX_OK or NGX_ERROR.
285 *
286 * WHY: DAV:displayname is a human-readable resource name, typically the last
287 * segment of the URL path.
288 *
289 * HOW: Scan backwards from end of href to find the byte after the final '/'.
290 * XML-escape the name and emit as <D:displayname>.
291 */
292static ngx_int_t
293propfind_emit_displayname(ngx_http_request_t *r, ngx_chain_t **head,
294 ngx_chain_t **tail, const char *href)
295{
296 ngx_pool_t *pool = r->pool;
297 const char *name = href + strlen(href);
298 char *safe_name;
299
300 /* Display name is the last path segment: scan back from end of href to
301 * the byte after the final '/'. */
302 while (name > href && *(name - 1) != '/') {
303 name--;
304 }
305 safe_name = webdav_escape_xml_text(pool, name);
306 if (safe_name != NULL
307 && brix_http_chain_appendf(pool, head, tail,
308 "<D:displayname>%s</D:displayname>", safe_name) == NULL)
309 {
310 return NGX_ERROR;
311 }
312
313 return NGX_OK;
314}

316/*
317 * WHAT: Emit xrd:locality (tape residency) for regular files. Returns NGX_OK or NGX_ERROR.
318 *
319 * WHY: xrd:locality reports WLCG tape residency status (ONLINE, NEARLINE, etc.).
320 * Only emitted when explicitly requested (not in PF_ALL) and only for files.
321 *
322 * HOW: Query VFS residency via brix_vfs_residency(), map backend residency state
323 * (ONLINE/NEARLINE/OFFLINE/LOST) to WLCG locality vocab, emit as <xrd:locality>.
324 */
325static ngx_int_t
326propfind_emit_locality(ngx_http_request_t *r, ngx_chain_t **head,
327 ngx_chain_t **tail, const char *path, struct stat *sb)
328{
329 ngx_http_brix_webdav_loc_conf_t *conf;
330 ngx_pool_t *pool = r->pool;
331 brix_vfs_ctx_t vctx;
332 brix_sd_residency_t res;
333 int nearline = 0;
334 const char *loc = "ONLINE";
335
336 /*
337 * xrd:locality — tape residency (phase-64 VFS seam). Emitted only when the
338 * client explicitly names the prop (PF_LOCALITY is not in PF_ALL) and only for
339 * regular files. Residency comes from the storage backend's model via
340 * brix_vfs_residency (no FRM xattr): a plain disk/object export classifies
341 * ONLINE, so it needs no nearline tier; on a nearline (tape) export an online
342 * object is ONLINE_AND_NEARLINE (resident AND on the backend) and a
343 * nearline/offline object is NEARLINE. Values follow the WLCG locality vocab.
344 */
345 if (S_ISDIR(sb->st_mode)) {
346 return NGX_OK; /* Only emit for regular files. */
347 }
348
349 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
350 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log,
351 BRIX_PROTO_WEBDAV, conf->common.root_canon, conf->cache_root_canon,
352 conf->common.allow_write, 0 /* is_tls */, NULL, path);
353
354 if (brix_vfs_residency(&vctx, &res, &nearline) == NGX_OK && nearline) {
355 switch (res) {
356 case BRIX_SD_RES_ONLINE:
357 loc = "ONLINE_AND_NEARLINE";
358 break;
359 case BRIX_SD_RES_NEARLINE:
360 case BRIX_SD_RES_OFFLINE:
361 loc = "NEARLINE";
362 break;
363 case BRIX_SD_RES_LOST:
364 loc = "LOST";
365 break;
366 }
367 }
368
369 if (brix_http_chain_appendf(pool, head, tail,
370 "<xrd:locality>%s</xrd:locality>", loc) == NULL)
371 return NGX_ERROR;
372
373 return NGX_OK;
374}

377/*
378 * WHAT: Emit basic metadata properties (resourcetype, contentlength, lastmodified,
379 * etag, creationdate). Returns NGX_OK or NGX_ERROR.
380 *
381 * WHY: These five properties are the core DAV metadata set, each derived directly
382 * from stat(2) buffer fields.
383 *
384 * HOW: Check each mask bit, emit the corresponding property from sb fields.
385 * resourcetype is <D:collection/> for dirs, empty for files. contentlength
386 * is 0 for dirs, st_size for files. etag/creationdate use helper formatters.
387 */
388static ngx_int_t
389propfind_emit_basic_metadata(ngx_http_request_t *r, ngx_chain_t **head,
390 ngx_chain_t **tail, unsigned mask,
391 struct stat *sb, char *date_buf)
392{
393 ngx_pool_t *pool = r->pool;
394
395 if (mask & PF_RESOURCETYPE) {
396 if (S_ISDIR(sb->st_mode)) {
397 if (brix_http_chain_appendf(pool, head, tail,
398 "<D:resourcetype>"
399 "<D:collection/>"
400 "</D:resourcetype>") == NULL)
401 return NGX_ERROR;
402 } else if (brix_http_chain_appendf(pool, head, tail,
403 "<D:resourcetype/>") == NULL) {
404 return NGX_ERROR;
405 }
406 }
407
408 if (mask & PF_CONTENTLENGTH) {
409 if (brix_http_chain_appendf(pool, head, tail,
410 "<D:getcontentlength>%lld</D:getcontentlength>",
411 S_ISDIR(sb->st_mode) ? 0LL : (long long) sb->st_size) == NULL)
412 return NGX_ERROR;
413 }
414
415 if (mask & PF_LASTMODIFIED) {
416 if (brix_http_chain_appendf(pool, head, tail,
417 "<D:getlastmodified>%s</D:getlastmodified>",
418 date_buf) == NULL)
419 return NGX_ERROR;
420 }
421
422 if (mask & PF_ETAG) {
423 char etag_buf[64];
424 brix_http_etag_str(etag_buf, sizeof(etag_buf), sb->st_mtime,
425 sb->st_size, BRIX_ETAG_WEAK);
426 if (brix_http_chain_appendf(pool, head, tail,
427 "<D:getetag>%s</D:getetag>", etag_buf) == NULL)
428 return NGX_ERROR;
429 }
430
431 if (mask & PF_CREATIONDATE) {
432 char cdate_buf[32];
433
434 brix_format_iso8601(sb->st_ctime, cdate_buf, sizeof(cdate_buf));
435 if (brix_http_chain_appendf(pool, head, tail,
436 "<D:creationdate>%s</D:creationdate>", cdate_buf) == NULL)
437 return NGX_ERROR;
438 }
439
440 return NGX_OK;
441}

444/*
445 * WHAT: Emit standard DAV properties filtered by mask. Returns NGX_OK or NGX_ERROR.
446 *
447 * WHY: PROPFIND responses carry different sets of properties depending on the
448 * request type (ALLPROP vs explicit PROP) and the requested prop_mask.
449 * Each standard property (resourcetype, contentlength, etag, etc.) has
450 * its own emission logic and FS dependency.
451 *
452 * HOW: Iterate through the 11 standard property types (resourcetype through
453 * locality), checking the mask bit for each. Emit the property's XML
454 * element with its value derived from sb (stat buffer), fs usage, or VFS
455 * residency. The mask ladder is kept inline (not table-driven) because
456 * each property has unique FS logic and different data sources.
457 */
458static ngx_int_t
459propfind_emit_standard_props(ngx_http_request_t *r, ngx_chain_t **head,
460 ngx_chain_t **tail, unsigned mask,
461 const char *href, const char *path,
462 struct stat *sb, char *date_buf)
463{
464 ngx_pool_t *pool = r->pool;
465
466 /* Emit basic metadata properties (resourcetype, contentlength, etag, etc.). */
467 if (propfind_emit_basic_metadata(r, head, tail, mask, sb, date_buf)
468 != NGX_OK)
469 {
470 return NGX_ERROR;
471 }
472
473 if (mask & PF_DISPLAYNAME) {
474 if (propfind_emit_displayname(r, head, tail, href) != NGX_OK) {
475 return NGX_ERROR;
476 }
477 }
478
479 if (mask & (PF_QUOTA_AVAILABLE | PF_QUOTA_USED)) {
480 if (propfind_emit_quota(r, head, tail, mask, path, sb) != NGX_OK) {
481 return NGX_ERROR;
482 }
483 }
484
485 if (mask & PF_SUPPORTED_REPORT) {
486 if (brix_http_chain_appendf(pool, head, tail,
487 "<D:supported-report-set/>") == NULL)
488 return NGX_ERROR;
489 }
490
491 if (mask & PF_SUPPORTEDLOCK) {
492 /* Lock helpers are best-effort here: a failure to emit the optional
493 * supportedlock/lockdiscovery block must not abort the whole response. */
494 (void) webdav_lock_append_supported(r, head, tail);
495 }
496
497 if (mask & PF_LOCKDISCOVERY) {
498 (void) webdav_lock_append_discovery(r, path, head, tail);
499 }
500
501 if (mask & PF_CONTENTTYPE) {
502 const char *ct = S_ISDIR(sb->st_mode)
503 ? "httpd/unix-directory"
504 : "application/octet-stream";
505 if (brix_http_chain_appendf(pool, head, tail,
506 "<D:getcontenttype>%s</D:getcontenttype>", ct) == NULL)
507 return NGX_ERROR;
508 }
509
510 if (mask & PF_LOCALITY) {
511 if (propfind_emit_locality(r, head, tail, path, sb) != NGX_OK) {
512 return NGX_ERROR;
513 }
514 }
515
516 return NGX_OK;
517}

519/*
520 * WHAT: Resolve and emit dead (client-defined) properties. Returns NGX_OK or NGX_ERROR.
521 *
522 * WHY: ALLPROP dumps all stored dead properties; explicit PROP requests resolve
523 * only the unknown (non-DAV:) properties against the xattr store, tracking
524 * hits in unknown_found[] so misses can be reported in a 404 propstat.
525 *
526 * HOW: For ALLPROP, call webdav_dead_props_append_all() to dump everything.
527 * For PROP, iterate req->unknown[], skip DAV: namespace unknowns (they're
528 * genuinely unsupported), and call webdav_dead_prop_append_value() for
529 * each non-DAV unknown, recording resolution success in unknown_found[].
530 */
531static ngx_int_t
532propfind_dead_props_resolve_and_emit(ngx_http_request_t *r, ngx_chain_t **head,
533 ngx_chain_t **tail, const char *path,
534 const propfind_req_t *req,
535 ngx_flag_t unknown_found[PF_UNKNOWN_MAX])
536{
537 ngx_uint_t i;
538
539 /* Dead (client-defined) properties: ALLPROP dumps them all; an explicit
540 * PROP request instead tries to resolve each unrecognised name against the
541 * dead-property store, recording hits in unknown_found[] so the misses can
542 * go into a 404 propstat afterwards. DAV:-namespace unknowns are never
543 * looked up here — a DAV: name we don't know is genuinely unsupported. */
544 if (req->type == PROPFIND_ALLPROP) {
545 if (webdav_dead_props_append_all(r, path, head, tail, 0) != NGX_OK) {
546 return NGX_ERROR;
547 }
548 } else if (req->type == PROPFIND_PROP) {
549 for (i = 0; i < req->unknown_count; i++) {
550 if (strcmp(req->unknown[i].ns, "DAV:") != 0) {
551 if (webdav_dead_prop_append_value(r, path,
552 req->unknown[i].ns, req->unknown[i].local,
553 head, tail, &unknown_found[i]) != NGX_OK)
554 {
555 return NGX_ERROR;
556 }
557 }
558 }
559 }
560
561 return NGX_OK;
562}

564/*
565 * WHAT: Emit 404 propstat for missing properties in explicit PROP requests.
566 * Returns NGX_OK or NGX_ERROR.
567 *
568 * WHY: RFC 4918 requires PROPFIND to report which explicitly-requested
569 * properties were not found via a 404 propstat. Emitting an empty
570 * 404 propstat is invalid; we only emit it when at least one requested
571 * property is genuinely missing.
572 *
573 * HOW: Iterate unknown_found[] to detect if any requested properties were
574 * not resolved. If any are missing, emit <D:propstat><D:prop>, iterate
575 * again to emit each missing property's pre-built XML element, then
576 * close with 404 status.
577 */
578static ngx_int_t
579propfind_emit_404_propstat(ngx_http_request_t *r, ngx_chain_t **head,
580 ngx_chain_t **tail, const propfind_req_t *req,
581 const ngx_flag_t unknown_found[PF_UNKNOWN_MAX])
582{
583 ngx_pool_t *pool = r->pool;
584 ngx_uint_t i;
585 ngx_flag_t any_missing = 0;
586
587 /* For explicit PROP requests: emit 404 propstat for unknown properties.
588 * Skip the whole block if every named unknown turned out to exist as a dead
589 * property (all resolved above) — emitting an empty 404 propstat is invalid. */
590 if (req->type != PROPFIND_PROP || req->unknown_count == 0) {
591 return NGX_OK;
592 }
593
594 for (i = 0; i < req->unknown_count; i++) {
595 if (!unknown_found[i]) {
596 any_missing = 1;
597 break;
598 }
599 }
600
601 /* Emit the 404 propstat only when at least one named property is
602 * genuinely missing — an empty 404 propstat would be invalid, so when
603 * nothing is missing we skip straight to closing the response. */
604 if (!any_missing) {
605 return NGX_OK;
606 }
607
608 if (brix_http_chain_appendf(pool, head, tail,
609 "<D:propstat><D:prop>") == NULL)
610 return NGX_ERROR;
611
612 for (i = 0; i < req->unknown_count; i++) {
613 if (unknown_found[i]) {
614 continue;
615 }
616 if (brix_http_chain_appendf(pool, head, tail,
617 "%s", req->unknown[i].xml) == NULL)
618 return NGX_ERROR;
619 }
620
621 if (brix_http_chain_appendf(pool, head, tail,
622 "</D:prop>"
623 "<D:status>HTTP/1.1 404 Not Found</D:status>"
624 "</D:propstat>") == NULL)
625 return NGX_ERROR;
626
627 return NGX_OK;

631/*
632 * Emit a D:response XML element for one resource.
633 *
634 * Behaviour depends on req->type:
635 * ALLPROP — all known properties with values
636 * PROPNAME — all property names as empty elements, no values, no
637 * expensive calls (skips statvfs and lock scan)
638 * PROP — only req->prop_mask properties in 200 propstat; any
639 * unrecognised property in req->unknown[] in 404 propstat
640 */
641ngx_int_t
642propfind_entry(ngx_http_request_t *r, ngx_chain_t **head, ngx_chain_t **tail,
643 const char *href, const char *path, struct stat *sb,
644 const propfind_req_t *req)
645{
646 char date_buf[30];
647 char *safe_href;
648 ngx_pool_t *pool = r->pool;
649 /* For an explicit PROP request only the requested bits are emitted; ALLPROP
650 * (and PROPNAME, handled separately below) emits the full PF_ALL set. */
651 unsigned mask = (req->type == PROPFIND_PROP) ? req->prop_mask : PF_ALL;
652 /* Tracks which of the request's unknown props were actually resolved as
653 * dead properties, so the rest can be reported in the 404 propstat. */
654 ngx_flag_t unknown_found[PF_UNKNOWN_MAX];
655
656 /* ngx_http_time writes an RFC 1123 date with no NUL; terminate it here. */
657 *ngx_http_time((u_char *) date_buf, sb->st_mtime) = '\0';
658 ngx_memzero(unknown_found, sizeof(unknown_found));
659
660 safe_href = webdav_escape_xml_text(pool, href);
661 if (safe_href == NULL) {
662 return NGX_ERROR;
663 }
664
665 if (brix_http_chain_appendf(pool, head, tail,
666 "<D:response>"
667 "<D:href>%s</D:href>"
668 "<D:propstat>"
669 "<D:prop>", safe_href) == NULL)
670 {
671 return NGX_ERROR;
672 }
673
674 /* PROPNAME: emit all names as empty elements, then close. */
675 if (req->type == PROPFIND_PROPNAME) {
676 if (brix_http_chain_appendf(pool, head, tail,
677 "<D:resourcetype/>"
678 "<D:getcontentlength/>"
679 "<D:getlastmodified/>"
680 "<D:getetag/>"
681 "<D:creationdate/>"
682 "<D:displayname/>"
683 "<D:quota-available-bytes/>"
684 "<D:quota-used-bytes/>"
685 "<D:supported-report-set/>"
686 "<D:supportedlock/>"
687 "<D:lockdiscovery/>"
688 "<D:getcontenttype/>"
689 "<D:owner/>"
690 "<D:group/>"
691 "<D:current-user-privilege-set/>"
692 "<D:supported-privilege-set/>"
693 "<D:acl/>"
694 "<D:acl-restrictions/>"
695 "<D:principal-collection-set/>"
696 "<xrd:locality/>") == NULL
697 || webdav_dead_props_append_all(r, path, head, tail, 1)
698 != NGX_OK
699 || brix_http_chain_appendf(pool, head, tail,
700 "</D:prop>"
701 "<D:status>HTTP/1.1 200 OK</D:status>"
702 "</D:propstat>"
703 "</D:response>") == NULL)
704 {
705 return NGX_ERROR;
706 }
707 return NGX_OK;
708 }
709
710 /* Emit standard DAV properties. */
711 if (propfind_emit_standard_props(r, head, tail, mask, href, path, sb,
712 date_buf) != NGX_OK)
713 {
714 return NGX_ERROR;
715 }
716
717 /* Emit ACL-related properties. */
718 if (propfind_append_acl_properties(r, head, tail, mask) != NGX_OK) {
719 return NGX_ERROR;
720 }
721
722 /* Resolve and emit dead (client-defined) properties. */
723 if (propfind_dead_props_resolve_and_emit(r, head, tail, path, req,
724 unknown_found) != NGX_OK)
725 {
726 return NGX_ERROR;
727 }
728
729 if (brix_http_chain_appendf(pool, head, tail,
730 "</D:prop>"
731 "<D:status>HTTP/1.1 200 OK</D:status>"
732 "</D:propstat>") == NULL)
733 {
734 return NGX_ERROR;
735 }
736
737 /* Emit 404 propstat for missing properties in explicit PROP requests. */
738 if (propfind_emit_404_propstat(r, head, tail, req, unknown_found)
739 != NGX_OK)
740 {
741 return NGX_ERROR;
742 }
743
744 if (brix_http_chain_appendf(pool, head, tail, "</D:response>") == NULL) {
745 return NGX_ERROR;
746 }
747
748 return NGX_OK;
749}
