12/*

14 * These are synthesized from the request's authenticated identity and the
15 * location's allow_write flag rather than from a real ACL store: owner is the
16 * client DN (or "anonymous"), and the privilege/ACL sets advertise read plus
17 * (only when writes are enabled) the write family. Each clause is gated on its
18 * own mask bit so PROP requests get exactly the properties they asked for.

146/*
147 * Emit a D:response XML element for one resource.
148 *
149 * Behaviour depends on req->type:
150 * ALLPROP — all known properties with values
151 * PROPNAME — all property names as empty elements, no values, no
152 * expensive calls (skips statvfs and lock scan)
153 * PROP — only req->prop_mask properties in 200 propstat; any
154 * unrecognised property in req->unknown[] in 404 propstat
155 */
156ngx_int_t
157propfind_entry(ngx_http_request_t *r, ngx_chain_t **head, ngx_chain_t **tail,
158 const char *href, const char *path, struct stat *sb,
159 const propfind_req_t *req)
160{
161 char date_buf[30];
162 char *safe_href;
163 ngx_pool_t *pool = r->pool;
164 /* For an explicit PROP request only the requested bits are emitted; ALLPROP
165 * (and PROPNAME, handled separately below) emits the full PF_ALL set. */
166 unsigned mask = (req->type == PROPFIND_PROP) ? req->prop_mask : PF_ALL;
167 /* Tracks which of the request's unknown props were actually resolved as
168 * dead properties, so the rest can be reported in the 404 propstat. */
169 ngx_flag_t unknown_found[PF_UNKNOWN_MAX];
170 ngx_uint_t i;
171
172 /* ngx_http_time writes an RFC 1123 date with no NUL; terminate it here. */
173 *ngx_http_time((u_char *) date_buf, sb->st_mtime) = '\0';
174 ngx_memzero(unknown_found, sizeof(unknown_found));
175
176 safe_href = webdav_escape_xml_text(pool, href);
177 if (safe_href == NULL) {
178 return NGX_ERROR;
179 }
180
181 if (brix_http_chain_appendf(pool, head, tail,
182 "<D:response>"
183 "<D:href>%s</D:href>"
184 "<D:propstat>"
185 "<D:prop>", safe_href) == NULL)
186 {
187 return NGX_ERROR;
188 }
189
190 /* PROPNAME: emit all names as empty elements, then close. */ if (req->type == PROPFIND_PROPNAME) {
191 if (brix_http_chain_appendf(pool, head, tail,
192 "<D:resourcetype/>"
193 "<D:getcontentlength/>"
194 "<D:getlastmodified/>"
195 "<D:getetag/>"
196 "<D:creationdate/>"
197 "<D:displayname/>"
198 "<D:quota-available-bytes/>"
199 "<D:quota-used-bytes/>"
200 "<D:supported-report-set/>"
201 "<D:supportedlock/>"
202 "<D:lockdiscovery/>"
203 "<D:getcontenttype/>"
204 "<D:owner/>"
205 "<D:group/>"
206 "<D:current-user-privilege-set/>"
207 "<D:supported-privilege-set/>"
208 "<D:acl/>"
209 "<D:acl-restrictions/>"
210 "<D:principal-collection-set/>"
211 "<xrd:locality/>") == NULL
212 || webdav_dead_props_append_all(r, path, head, tail, 1)
213 != NGX_OK
214 || brix_http_chain_appendf(pool, head, tail,
215 "</D:prop>"
216 "<D:status>HTTP/1.1 200 OK</D:status>"
217 "</D:propstat>"
218 "</D:response>") == NULL)
219 {
220 return NGX_ERROR;
221 }
222 return NGX_OK;
223 }
224
225 /* ALLPROP / PROP: emit property values filtered by mask. */
226 if (mask & PF_RESOURCETYPE) {
227 if (S_ISDIR(sb->st_mode)) {
228 if (brix_http_chain_appendf(pool, head, tail,
229 "<D:resourcetype>"
230 "<D:collection/>"
231 "</D:resourcetype>") == NULL)
232 return NGX_ERROR;
233 } else if (brix_http_chain_appendf(pool, head, tail,
234 "<D:resourcetype/>") == NULL) {
235 return NGX_ERROR;
236 }
237 }
238
239 if (mask & PF_CONTENTLENGTH) {
240 if (brix_http_chain_appendf(pool, head, tail,
241 "<D:getcontentlength>%lld</D:getcontentlength>",
242 S_ISDIR(sb->st_mode) ? 0LL : (long long) sb->st_size) == NULL)
243 return NGX_ERROR;
244 }
245
246 if (mask & PF_LASTMODIFIED) {
247 if (brix_http_chain_appendf(pool, head, tail,
248 "<D:getlastmodified>%s</D:getlastmodified>",
249 date_buf) == NULL)
250 return NGX_ERROR;
251 }
252
253 if (mask & PF_ETAG) {
254 char etag_buf[64];
255 brix_http_etag_str(etag_buf, sizeof(etag_buf), sb->st_mtime,
256 sb->st_size, BRIX_ETAG_WEAK);
257 if (brix_http_chain_appendf(pool, head, tail,
258 "<D:getetag>%s</D:getetag>", etag_buf) == NULL)
259 return NGX_ERROR;
260 }
261
262 if (mask & PF_CREATIONDATE) {
263 char cdate_buf[32];
264
265 brix_format_iso8601(sb->st_ctime, cdate_buf, sizeof(cdate_buf));
266 if (brix_http_chain_appendf(pool, head, tail,
267 "<D:creationdate>%s</D:creationdate>", cdate_buf) == NULL)
268 return NGX_ERROR;
269 }
270
271 if (mask & PF_DISPLAYNAME) {
272 const char *name = href + strlen(href);
273 char *safe_name;
274
275 /* Display name is the last path segment: scan back from end of href to
276 * the byte after the final '/'. */
277 while (name > href && *(name - 1) != '/') {
278 name--;
279 }
280 safe_name = webdav_escape_xml_text(pool, name);
281 if (safe_name != NULL
282 && brix_http_chain_appendf(pool, head, tail,
283 "<D:displayname>%s</D:displayname>", safe_name) == NULL)
284 {
285 return NGX_ERROR;
286 }
287 }

289 /*
290 * RFC 4331 quota properties describe a COLLECTION's storage quota, not a
291 * file's. Emitting them on a regular file is non-conformant and actively
292 * breaks clients: gfal2/davix maps <D:quota-used-bytes> onto st_size, so a
293 * file's reported size becomes the filesystem's used bytes (~TB) instead of
294 * getcontentlength. Stock XrdHttp never emits quota per-file. Gate to
295 * directories so files carry only getcontentlength as their size.
296 */
297 if ((mask & (PF_QUOTA_AVAILABLE | PF_QUOTA_USED)) && S_ISDIR(sb->st_mode)) {
298 brix_fs_usage_t fsu;
299 if (brix_fs_usage_stat(path, &fsu) == NGX_OK) {
300 if ((mask & PF_QUOTA_AVAILABLE)
301 && brix_http_chain_appendf(pool, head, tail,
302 "<D:quota-available-bytes>"
303 "%llu"
304 "</D:quota-available-bytes>",
305 (unsigned long long) fsu.available_bytes) == NULL)
306 return NGX_ERROR;
307 if ((mask & PF_QUOTA_USED)
308 && brix_http_chain_appendf(pool, head, tail,
309 "<D:quota-used-bytes>"
310 "%llu"
311 "</D:quota-used-bytes>",
312 (unsigned long long) fsu.used_bytes) == NULL)
313 return NGX_ERROR;
314 }
315 }
316
317 if (mask & PF_SUPPORTED_REPORT) {
318 if (brix_http_chain_appendf(pool, head, tail,
319 "<D:supported-report-set/>") == NULL)
320 return NGX_ERROR;
321 }
322
323 if (mask & PF_SUPPORTEDLOCK) {
324 /* Lock helpers are best-effort here: a failure to emit the optional
325 * supportedlock/lockdiscovery block must not abort the whole response. */
326 (void) webdav_lock_append_supported(r, head, tail);
327 }
328
329 if (mask & PF_LOCKDISCOVERY) {
330 (void) webdav_lock_append_discovery(r, path, head, tail);
331 }
332
333 if (mask & PF_CONTENTTYPE) {
334 const char *ct = S_ISDIR(sb->st_mode)
335 ? "httpd/unix-directory"
336 : "application/octet-stream";
337 if (brix_http_chain_appendf(pool, head, tail,
338 "<D:getcontenttype>%s</D:getcontenttype>", ct) == NULL)
339 return NGX_ERROR;
340 }

342 /*
343 * xrd:locality — tape residency (phase-64 VFS seam). Emitted only when the
344 * client explicitly names the prop (PF_LOCALITY is not in PF_ALL) and only for
345 * regular files. Residency comes from the storage backend's model via
346 * brix_vfs_residency (no FRM xattr): a plain disk/object export classifies
347 * ONLINE, so it needs no nearline tier; on a nearline (tape) export an online
348 * object is ONLINE_AND_NEARLINE (resident AND on the backend) and a
349 * nearline/offline object is NEARLINE. Values follow the WLCG locality vocab.
350 */
351 if ((mask & PF_LOCALITY) && !S_ISDIR(sb->st_mode)) {
352 ngx_http_brix_webdav_loc_conf_t *conf =
353 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
354 brix_vfs_ctx_t vctx;
355 brix_sd_residency_t res;
356 int nearline = 0;
357 const char *loc = "ONLINE";
358
359 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log,
360 BRIX_PROTO_WEBDAV, conf->common.root_canon, conf->cache_root_canon,
361 conf->common.allow_write, 0 /* is_tls */, NULL, path);
362 if (brix_vfs_residency(&vctx, &res, &nearline) == NGX_OK && nearline) {
363 switch (res) {
364 case BRIX_SD_RES_ONLINE:
365 loc = "ONLINE_AND_NEARLINE";
366 break;
367 case BRIX_SD_RES_NEARLINE:
368 case BRIX_SD_RES_OFFLINE:
369 loc = "NEARLINE";
370 break;
371 case BRIX_SD_RES_LOST:
372 loc = "LOST";
373 break;
374 }
375 }
376 if (brix_http_chain_appendf(pool, head, tail,
377 "<xrd:locality>%s</xrd:locality>", loc) == NULL)
378 return NGX_ERROR;
379 }
380
381 if (propfind_append_acl_properties(r, head, tail, mask) != NGX_OK) {
382 return NGX_ERROR;
383 }

385 /* Dead (client-defined) properties: ALLPROP dumps them all; an explicit
386 * PROP request instead tries to resolve each unrecognised name against the
387 * dead-property store, recording hits in unknown_found[] so the misses can
388 * go into a 404 propstat afterwards. DAV:-namespace unknowns are never
389 * looked up here — a DAV: name we don't know is genuinely unsupported. */
390 if (req->type == PROPFIND_ALLPROP) {
391 if (webdav_dead_props_append_all(r, path, head, tail, 0) != NGX_OK) {
392 return NGX_ERROR;
393 }
394 } else if (req->type == PROPFIND_PROP) {
395 for (i = 0; i < req->unknown_count; i++) {
396 if (strcmp(req->unknown[i].ns, "DAV:") != 0) {
397 if (webdav_dead_prop_append_value(r, path,
398 req->unknown[i].ns, req->unknown[i].local,
399 head, tail, &unknown_found[i]) != NGX_OK)
400 {
401 return NGX_ERROR;
402 }
403 }
404 }
405 }
406
407 if (brix_http_chain_appendf(pool, head, tail,
408 "</D:prop>"
409 "<D:status>HTTP/1.1 200 OK</D:status>"
410 "</D:propstat>") == NULL)
411 return NGX_ERROR;
412
413 /* For explicit PROP requests: emit 404 propstat for unknown properties.
414 * Skip the whole block if every named unknown turned out to exist as a dead
415 * property (all resolved above) — emitting an empty 404 propstat is invalid. */
416 if (req->type == PROPFIND_PROP && req->unknown_count > 0) {
417 ngx_flag_t any_missing = 0;
418
419 for (i = 0; i < req->unknown_count; i++) {
420 if (!unknown_found[i]) {
421 any_missing = 1;
422 break;
423 }
424 }
425
426 /* Emit the 404 propstat only when at least one named property is
427 * genuinely missing — an empty 404 propstat would be invalid, so when
428 * nothing is missing we skip straight to closing the response. */
429 if (any_missing) {
430 if (brix_http_chain_appendf(pool, head, tail,
431 "<D:propstat><D:prop>") == NULL)
432 return NGX_ERROR;
433
434 for (i = 0; i < req->unknown_count; i++) {
435 if (unknown_found[i]) {
436 continue;
437 }
438 if (brix_http_chain_appendf(pool, head, tail,
439 "%s", req->unknown[i].xml) == NULL)
440 return NGX_ERROR;
441 }
442
443 if (brix_http_chain_appendf(pool, head, tail,
444 "</D:prop>"
445 "<D:status>HTTP/1.1 404 Not Found</D:status>"
446 "</D:propstat>") == NULL)
447 return NGX_ERROR;
448 }
449 }
450
451 if (brix_http_chain_appendf(pool, head, tail, "</D:response>") == NULL) {
452 return NGX_ERROR;
453 }
454
455 return NGX_OK;
