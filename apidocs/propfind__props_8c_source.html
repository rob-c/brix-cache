12/*
13 * WHAT: Emit RFC 4331 quota properties for directories. Returns NGX_OK or NGX_ERROR.
14 *
15 * WHY: Quota properties describe a COLLECTION's storage quota. Emitting them on
16 * regular files breaks clients (gfal2/davix maps quota-used-bytes onto file
17 * size, so a file appears ~TB instead of its actual size).
18 *
19 * HOW: Gate on S_ISDIR, query filesystem usage via brix_fs_usage_stat(), emit
20 * quota-available-bytes and/or quota-used-bytes based on mask bits.
21 */
22static ngx_int_t
23propfind_emit_quota(ngx_http_request_t *r, ngx_chain_t **head,
24 ngx_chain_t **tail, unsigned mask, const char *path,
25 struct stat *sb)
26{
27 ngx_pool_t *pool = r->pool;
28
29 /*
30 * RFC 4331 quota properties describe a COLLECTION's storage quota, not a
31 * file's. Emitting them on a regular file is non-conformant and actively
32 * breaks clients: gfal2/davix maps <D:quota-used-bytes> onto st_size, so a
33 * file's reported size becomes the filesystem's used bytes (~TB) instead of
34 * getcontentlength. Stock XrdHttp never emits quota per-file. Gate to
35 * directories so files carry only getcontentlength as their size.
36 */
37 if ((mask & (PF_QUOTA_AVAILABLE | PF_QUOTA_USED)) && S_ISDIR(sb->st_mode)) {
38 brix_fs_usage_t fsu;
39 if (brix_fs_usage_stat(path, &fsu) == NGX_OK) {
40 if ((mask & PF_QUOTA_AVAILABLE)
41 && brix_http_chain_appendf(pool, head, tail,
42 "<D:quota-available-bytes>"
43 "%llu"
44 "</D:quota-available-bytes>",
45 (unsigned long long) fsu.available_bytes) == NULL)
46 return NGX_ERROR;
47 if ((mask & PF_QUOTA_USED)
48 && brix_http_chain_appendf(pool, head, tail,
49 "<D:quota-used-bytes>"
50 "%llu"
51 "</D:quota-used-bytes>",
52 (unsigned long long) fsu.used_bytes) == NULL)
53 return NGX_ERROR;
54 }
55 }
56
57 return NGX_OK;
58}

60/*
61 * WHAT: Emit displayname property (last path segment of href). Returns NGX_OK or NGX_ERROR.
62 *
63 * WHY: DAV:displayname is a human-readable resource name, typically the last
64 * segment of the URL path.
65 *
66 * HOW: Scan backwards from end of href to find the byte after the final '/'.
67 * XML-escape the name and emit as <D:displayname>.
68 */
69static ngx_int_t
70propfind_emit_displayname(ngx_http_request_t *r, ngx_chain_t **head,
71 ngx_chain_t **tail, const char *href)
72{
73 ngx_pool_t *pool = r->pool;
74 const char *name = href + strlen(href);
75 char *safe_name;
76
77 /* Display name is the last path segment: scan back from end of href to
78 * the byte after the final '/'. */
79 while (name > href && *(name - 1) != '/') {
80 name--;
81 }
82 safe_name = webdav_escape_xml_text(pool, name);
83 if (safe_name != NULL
84 && brix_http_chain_appendf(pool, head, tail,
85 "<D:displayname>%s</D:displayname>", safe_name) == NULL)
86 {
87 return NGX_ERROR;
88 }
89
90 return NGX_OK;
91}

93/*
94 * WHAT: Emit xrd:locality (tape residency) for regular files. Returns NGX_OK or NGX_ERROR.
95 *
96 * WHY: xrd:locality reports WLCG tape residency status (ONLINE, NEARLINE, etc.).
97 * Only emitted when explicitly requested (not in PF_ALL) and only for files.
98 *
99 * HOW: Query VFS residency via brix_vfs_residency(), map backend residency state
100 * (ONLINE/NEARLINE/OFFLINE/LOST) to WLCG locality vocab, emit as <xrd:locality>.
101 */
102static ngx_int_t
103propfind_emit_locality(ngx_http_request_t *r, ngx_chain_t **head,
104 ngx_chain_t **tail, const char *path, struct stat *sb)
105{
106 ngx_http_brix_webdav_loc_conf_t *conf;
107 ngx_pool_t *pool = r->pool;
108 brix_vfs_ctx_t vctx;
109 brix_sd_residency_t res;
110 int nearline = 0;
111 const char *loc = "ONLINE";
112
113 /*
114 * xrd:locality — tape residency (phase-64 VFS seam). Emitted only when the
115 * client explicitly names the prop (PF_LOCALITY is not in PF_ALL) and only for
116 * regular files. Residency comes from the storage backend's model via
117 * brix_vfs_residency (no FRM xattr): a plain disk/object export classifies
118 * ONLINE, so it needs no nearline tier; on a nearline (tape) export an online
119 * object is ONLINE_AND_NEARLINE (resident AND on the backend) and a
120 * nearline/offline object is NEARLINE. Values follow the WLCG locality vocab.
121 */
122 if (S_ISDIR(sb->st_mode)) {
123 return NGX_OK; /* Only emit for regular files. */
124 }
125
126 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
127 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log,
128 BRIX_PROTO_WEBDAV, conf->common.root_canon, conf->cache_root_canon,
129 conf->common.allow_write, 0 /* is_tls */, NULL, path);
130
131 if (brix_vfs_residency(&vctx, &res, &nearline) == NGX_OK && nearline) {
132 switch (res) {
133 case BRIX_SD_RES_ONLINE:
134 loc = "ONLINE_AND_NEARLINE";
135 break;
136 case BRIX_SD_RES_NEARLINE:
137 case BRIX_SD_RES_OFFLINE:
138 loc = "NEARLINE";
139 break;
140 case BRIX_SD_RES_LOST:
141 loc = "LOST";
142 break;
143 }
144 }
145
146 if (brix_http_chain_appendf(pool, head, tail,
147 "<xrd:locality>%s</xrd:locality>", loc) == NULL)
148 return NGX_ERROR;
149
150 return NGX_OK;
151}

154/*
155 * WHAT: Emit basic metadata properties (resourcetype, contentlength, lastmodified,
156 * etag, creationdate). Returns NGX_OK or NGX_ERROR.
157 *
158 * WHY: These five properties are the core DAV metadata set, each derived directly
159 * from stat(2) buffer fields.
160 *
161 * HOW: Check each mask bit, emit the corresponding property from sb fields.
162 * resourcetype is <D:collection/> for dirs, empty for files. contentlength
163 * is 0 for dirs, st_size for files. etag/creationdate use helper formatters.
164 */
165static ngx_int_t
166propfind_emit_basic_metadata(ngx_http_request_t *r, ngx_chain_t **head,
167 ngx_chain_t **tail, unsigned mask,
168 struct stat *sb, char *date_buf)
169{
170 ngx_pool_t *pool = r->pool;
171
172 if (mask & PF_RESOURCETYPE) {
173 if (S_ISDIR(sb->st_mode)) {
174 if (brix_http_chain_appendf(pool, head, tail,
175 "<D:resourcetype>"
176 "<D:collection/>"
177 "</D:resourcetype>") == NULL)
178 return NGX_ERROR;
179 } else if (brix_http_chain_appendf(pool, head, tail,
180 "<D:resourcetype/>") == NULL) {
181 return NGX_ERROR;
182 }
183 }
184
185 if (mask & PF_CONTENTLENGTH) {
186 if (brix_http_chain_appendf(pool, head, tail,
187 "<D:getcontentlength>%lld</D:getcontentlength>",
188 S_ISDIR(sb->st_mode) ? 0LL : (long long) sb->st_size) == NULL)
189 return NGX_ERROR;
190 }
191
192 if (mask & PF_LASTMODIFIED) {
193 if (brix_http_chain_appendf(pool, head, tail,
194 "<D:getlastmodified>%s</D:getlastmodified>",
195 date_buf) == NULL)
196 return NGX_ERROR;
197 }
198
199 if (mask & PF_ETAG) {
200 char etag_buf[64];
201 brix_http_etag_str(etag_buf, sizeof(etag_buf), sb->st_mtime,
202 sb->st_size, BRIX_ETAG_WEAK);
203 if (brix_http_chain_appendf(pool, head, tail,
204 "<D:getetag>%s</D:getetag>", etag_buf) == NULL)
205 return NGX_ERROR;
206 }
207
208 if (mask & PF_CREATIONDATE) {
209 char cdate_buf[32];
210
211 brix_format_iso8601(sb->st_ctime, cdate_buf, sizeof(cdate_buf));
212 if (brix_http_chain_appendf(pool, head, tail,
213 "<D:creationdate>%s</D:creationdate>", cdate_buf) == NULL)
214 return NGX_ERROR;
215 }
216
217 return NGX_OK;
218}

221/*
222 * WHAT: Emit standard DAV properties filtered by mask. Returns NGX_OK or NGX_ERROR.
223 *
224 * WHY: PROPFIND responses carry different sets of properties depending on the
225 * request type (ALLPROP vs explicit PROP) and the requested prop_mask.
226 * Each standard property (resourcetype, contentlength, etag, etc.) has
227 * its own emission logic and FS dependency.
228 *
229 * HOW: Iterate through the 11 standard property types (resourcetype through
230 * locality), checking the mask bit for each. Emit the property's XML
231 * element with its value derived from sb (stat buffer), fs usage, or VFS
232 * residency. The mask ladder is kept inline (not table-driven) because
233 * each property has unique FS logic and different data sources.
234 */
235static ngx_int_t
236propfind_emit_standard_props(ngx_http_request_t *r, ngx_chain_t **head,
237 ngx_chain_t **tail, unsigned mask,
238 const char *href, const char *path,
239 struct stat *sb, char *date_buf)
240{
241 ngx_pool_t *pool = r->pool;
242
243 /* Emit basic metadata properties (resourcetype, contentlength, etag, etc.). */
244 if (propfind_emit_basic_metadata(r, head, tail, mask, sb, date_buf)
245 != NGX_OK)
246 {
247 return NGX_ERROR;
248 }
249
250 if (mask & PF_DISPLAYNAME) {
251 if (propfind_emit_displayname(r, head, tail, href) != NGX_OK) {
252 return NGX_ERROR;
253 }
254 }
255
256 if (mask & (PF_QUOTA_AVAILABLE | PF_QUOTA_USED)) {
257 if (propfind_emit_quota(r, head, tail, mask, path, sb) != NGX_OK) {
258 return NGX_ERROR;
259 }
260 }
261
262 if (mask & PF_SUPPORTED_REPORT) {
263 if (brix_http_chain_appendf(pool, head, tail,
264 "<D:supported-report-set/>") == NULL)
265 return NGX_ERROR;
266 }
267
268 if (mask & PF_SUPPORTEDLOCK) {
269 /* Lock helpers are best-effort here: a failure to emit the optional
270 * supportedlock/lockdiscovery block must not abort the whole response. */
271 (void) webdav_lock_append_supported(r, head, tail);
272 }
273
274 if (mask & PF_LOCKDISCOVERY) {
275 (void) webdav_lock_append_discovery(r, path, head, tail);
276 }
277
278 if (mask & PF_CONTENTTYPE) {
279 const char *ct = S_ISDIR(sb->st_mode)
280 ? "httpd/unix-directory"
281 : "application/octet-stream";
282 if (brix_http_chain_appendf(pool, head, tail,
283 "<D:getcontenttype>%s</D:getcontenttype>", ct) == NULL)
284 return NGX_ERROR;
285 }
286
287 if (mask & PF_LOCALITY) {
288 if (propfind_emit_locality(r, head, tail, path, sb) != NGX_OK) {
289 return NGX_ERROR;
290 }
291 }
292
293 return NGX_OK;
294}

296/*
297 * WHAT: Resolve and emit dead (client-defined) properties. Returns NGX_OK or NGX_ERROR.
298 *
299 * WHY: ALLPROP dumps all stored dead properties; explicit PROP requests resolve
300 * only the unknown (non-DAV:) properties against the xattr store, tracking
301 * hits in unknown_found[] so misses can be reported in a 404 propstat.
302 *
303 * HOW: For ALLPROP, call webdav_dead_props_append_all() to dump everything.
304 * For PROP, iterate req->unknown[], skip DAV: namespace unknowns (they're
305 * genuinely unsupported), and call webdav_dead_prop_append_value() for
306 * each non-DAV unknown, recording resolution success in unknown_found[].
307 */
308static ngx_int_t
309propfind_dead_props_resolve_and_emit(ngx_http_request_t *r, ngx_chain_t **head,
310 ngx_chain_t **tail, const char *path,
311 const propfind_req_t *req,
312 ngx_flag_t unknown_found[PF_UNKNOWN_MAX])
313{
314 ngx_uint_t i;
315
316 /* Dead (client-defined) properties: ALLPROP dumps them all; an explicit
317 * PROP request instead tries to resolve each unrecognised name against the
318 * dead-property store, recording hits in unknown_found[] so the misses can
319 * go into a 404 propstat afterwards. DAV:-namespace unknowns are never
320 * looked up here — a DAV: name we don't know is genuinely unsupported. */
321 if (req->type == PROPFIND_ALLPROP) {
322 if (webdav_dead_props_append_all(r, path, head, tail, 0) != NGX_OK) {
323 return NGX_ERROR;
324 }
325 } else if (req->type == PROPFIND_PROP) {
326 for (i = 0; i < req->unknown_count; i++) {
327 if (strcmp(req->unknown[i].ns, "DAV:") != 0) {
328 if (webdav_dead_prop_append_value(r, path,
329 req->unknown[i].ns, req->unknown[i].local,
330 head, tail, &unknown_found[i]) != NGX_OK)
331 {
332 return NGX_ERROR;
333 }
334 }
335 }
336 }
337
338 return NGX_OK;
339}

341/*
342 * WHAT: Emit 404 propstat for missing properties in explicit PROP requests.
343 * Returns NGX_OK or NGX_ERROR.
344 *
345 * WHY: RFC 4918 requires PROPFIND to report which explicitly-requested
346 * properties were not found via a 404 propstat. Emitting an empty
347 * 404 propstat is invalid; we only emit it when at least one requested
348 * property is genuinely missing.
349 *
350 * HOW: Iterate unknown_found[] to detect if any requested properties were
351 * not resolved. If any are missing, emit <D:propstat><D:prop>, iterate
352 * again to emit each missing property's pre-built XML element, then
353 * close with 404 status.
354 */
355static ngx_int_t
356propfind_emit_404_propstat(ngx_http_request_t *r, ngx_chain_t **head,
357 ngx_chain_t **tail, const propfind_req_t *req,
358 const ngx_flag_t unknown_found[PF_UNKNOWN_MAX])
359{
360 ngx_pool_t *pool = r->pool;
361 ngx_uint_t i;
362 ngx_flag_t any_missing = 0;
363
364 /* For explicit PROP requests: emit 404 propstat for unknown properties.
365 * Skip the whole block if every named unknown turned out to exist as a dead
366 * property (all resolved above) — emitting an empty 404 propstat is invalid. */
367 if (req->type != PROPFIND_PROP || req->unknown_count == 0) {
368 return NGX_OK;
369 }
370
371 for (i = 0; i < req->unknown_count; i++) {
372 if (!unknown_found[i]) {
373 any_missing = 1;
374 break;
375 }
376 }
377
378 /* Emit the 404 propstat only when at least one named property is
379 * genuinely missing — an empty 404 propstat would be invalid, so when
380 * nothing is missing we skip straight to closing the response. */
381 if (!any_missing) {
382 return NGX_OK;
383 }
384
385 if (brix_http_chain_appendf(pool, head, tail,
386 "<D:propstat><D:prop>") == NULL)
387 return NGX_ERROR;
388
389 for (i = 0; i < req->unknown_count; i++) {
390 if (unknown_found[i]) {
391 continue;
392 }
393 if (brix_http_chain_appendf(pool, head, tail,
394 "%s", req->unknown[i].xml) == NULL)
395 return NGX_ERROR;
396 }
397
398 if (brix_http_chain_appendf(pool, head, tail,
399 "</D:prop>"
400 "<D:status>HTTP/1.1 404 Not Found</D:status>"
401 "</D:propstat>") == NULL)
402 return NGX_ERROR;
403
404 return NGX_OK;
405}

408/*
409 * Emit a D:response XML element for one resource.
410 *
411 * Behaviour depends on req->type:
412 * ALLPROP — all known properties with values
413 * PROPNAME — all property names as empty elements, no values, no
414 * expensive calls (skips statvfs and lock scan)
415 * PROP — only req->prop_mask properties in 200 propstat; any
416 * unrecognised property in req->unknown[] in 404 propstat
417 */
418ngx_int_t
419propfind_entry(ngx_http_request_t *r, ngx_chain_t **head, ngx_chain_t **tail,
420 const char *href, const char *path, struct stat *sb,
421 const propfind_req_t *req)
422{
423 char date_buf[30];
424 char *safe_href;
425 ngx_pool_t *pool = r->pool;
426 /* For an explicit PROP request only the requested bits are emitted; ALLPROP
427 * (and PROPNAME, handled separately below) emits the full PF_ALL set. */
428 unsigned mask = (req->type == PROPFIND_PROP) ? req->prop_mask : PF_ALL;
429 /* Tracks which of the request's unknown props were actually resolved as
430 * dead properties, so the rest can be reported in the 404 propstat. */
431 ngx_flag_t unknown_found[PF_UNKNOWN_MAX];
432
433 /* ngx_http_time writes an RFC 1123 date with no NUL; terminate it here. */
434 *ngx_http_time((u_char *) date_buf, sb->st_mtime) = '\0';
435 ngx_memzero(unknown_found, sizeof(unknown_found));
436
437 safe_href = webdav_escape_xml_text(pool, href);
438 if (safe_href == NULL) {
439 return NGX_ERROR;
440 }
441
442 if (brix_http_chain_appendf(pool, head, tail,
443 "<D:response>"
444 "<D:href>%s</D:href>"
445 "<D:propstat>"
446 "<D:prop>", safe_href) == NULL)
447 {
448 return NGX_ERROR;
449 }
450
451 /* PROPNAME: emit all names as empty elements, then close. */
452 if (req->type == PROPFIND_PROPNAME) {
453 if (brix_http_chain_appendf(pool, head, tail,
454 "<D:resourcetype/>"
455 "<D:getcontentlength/>"
456 "<D:getlastmodified/>"
457 "<D:getetag/>"
458 "<D:creationdate/>"
459 "<D:displayname/>"
460 "<D:quota-available-bytes/>"
461 "<D:quota-used-bytes/>"
462 "<D:supported-report-set/>"
463 "<D:supportedlock/>"
464 "<D:lockdiscovery/>"
465 "<D:getcontenttype/>"
466 "<D:owner/>"
467 "<D:group/>"
468 "<D:current-user-privilege-set/>"
469 "<D:supported-privilege-set/>"
470 "<D:acl/>"
471 "<D:acl-restrictions/>"
472 "<D:principal-collection-set/>"
473 "<xrd:locality/>") == NULL
474 || webdav_dead_props_append_all(r, path, head, tail, 1)
475 != NGX_OK
476 || brix_http_chain_appendf(pool, head, tail,
477 "</D:prop>"
478 "<D:status>HTTP/1.1 200 OK</D:status>"
479 "</D:propstat>"
480 "</D:response>") == NULL)
481 {
482 return NGX_ERROR;
483 }
484 return NGX_OK;
485 }
486
487 /* Emit standard DAV properties. */
488 if (propfind_emit_standard_props(r, head, tail, mask, href, path, sb,
489 date_buf) != NGX_OK)
490 {
491 return NGX_ERROR;
492 }
493
494 /* Emit ACL-related properties. */
495 if (propfind_append_acl_properties(r, head, tail, mask) != NGX_OK) {
496 return NGX_ERROR;
497 }
498
499 /* Resolve and emit dead (client-defined) properties. */
500 if (propfind_dead_props_resolve_and_emit(r, head, tail, path, req,
501 unknown_found) != NGX_OK)
502 {
503 return NGX_ERROR;
504 }
505
506 if (brix_http_chain_appendf(pool, head, tail,
507 "</D:prop>"
508 "<D:status>HTTP/1.1 200 OK</D:status>"
509 "</D:propstat>") == NULL)
510 {
511 return NGX_ERROR;
512 }
513
514 /* Emit 404 propstat for missing properties in explicit PROP requests. */
515 if (propfind_emit_404_propstat(r, head, tail, req, unknown_found)
516 != NGX_OK)
517 {
518 return NGX_ERROR;
519 }
520
521 if (brix_http_chain_appendf(pool, head, tail, "</D:response>") == NULL) {
522 return NGX_ERROR;
523 }
524
525 return NGX_OK;
526}
