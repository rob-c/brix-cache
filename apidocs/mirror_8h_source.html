1/*
2 * mirror.h — Phase 24 traffic-mirroring shared config layer.
3 *
4 * WHAT: Protocol-agnostic configuration types shared by the HTTP/WebDAV mirror
5 * (src/mirror/http_mirror.c) and the XRootD stream mirror
6 * (src/mirror/stream_mirror.c). A mirror sends a fire-and-forget copy of a
7 * qualifying READ request to one or more shadow backends after the primary
8 * request has already been answered; the shadow response is read, its status
9 * compared against the primary, and any divergence counted. The client never
10 * sees the shadow response and is never delayed by it.
11 *
12 * WHY: Both surfaces need the same knobs — target list, sampling rate, a
13 * method/opcode filter, auth-stripping, divergence logging, and a timeout — so
14 * the operator-facing directives (brix_mirror_sample, _strip_auth, etc.) are
15 * defined once and reused. This header carries no HTTP- or stream-specific
16 * types so it can be included from either module without pulling in ngx_http.
17 *
18 * HOW: brix_mirror_conf_t is embedded in both the WebDAV location conf and
19 * the stream server conf. Targets are resolved at configuration time into
20 * brix_mirror_target_t (host/port/ssl + a pre-resolved sockaddr). Sampling
21 * is a per-request PRNG draw — sufficient for HEP traffic volumes, no reservoir
22 * sampler needed.
23 */

59/* ---- XRootD opcode bitmask (brix_mirror_opcodes) ----
60 * Only stateless, path-bearing read opcodes are meaningful to replay against a
61 * shadow: a kXR_read carries a file handle issued by the PRIMARY's kXR_open and
62 * is meaningless on a server that never saw that open. The default set is the
63 * stateless ones; read/readv are selectable but best-effort. */
64#define BRIX_MIRROR_OP_STAT (1u << 0)
65#define BRIX_MIRROR_OP_LOCATE (1u << 1)
66#define BRIX_MIRROR_OP_OPEN (1u << 2) /* read opens only */
67#define BRIX_MIRROR_OP_READ (1u << 3)
68#define BRIX_MIRROR_OP_READV (1u << 4)
69#define BRIX_MIRROR_OP_DIRLIST (1u << 5)
70#define BRIX_MIRROR_OP_STATX (1u << 6)
71#define BRIX_MIRROR_OP_QUERY (1u << 7)
72/* Write opcodes (Phase 24 write mirroring). MKDIR/RM/RMDIR/MV/TRUNCATE/CHMOD are
73 * self-contained, path-based metadata mutations replayable by the stateless
74 * one-shot mirror (W1). WRITE covers the stateful open(write)->write->close data
75 * session driven by the dedicated write-mirror (W3). All are gated by
76 * brix_mirror_writes and are deliberately EXCLUDED from OP_DEFAULT/OP_ALL so a
77 * write op must be BOTH explicitly listed in brix_mirror_opcodes AND have
78 * brix_mirror_writes on (two independent guards). */

145/*
146 * Per-request sampling decision. ngx_random() is seeded per worker; a plain
147 * modulo draw is adequate for traffic sampling (no cryptographic uniformity
148 * needed). pct==100 short-circuits to always-on so a fully-sampled config
149 * never calls the PRNG.
150 */
151static ngx_inline ngx_uint_t
152brix_mirror_should_sample(ngx_uint_t sample_pct)
153{
154 if (sample_pct >= 100) { return 1; }
155 if (sample_pct == 0) { return 0; }
156 return (ngx_uint_t) (ngx_random() % 100) < sample_pct;
157}
158
159/* Map an HTTP status (or kXR-derived pseudo-status) to its class digit 1..5. */
160static ngx_inline ngx_uint_t
161brix_mirror_status_class(ngx_uint_t status)
162{
163 return status / 100;
164}
