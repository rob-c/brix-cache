4/*
5 * vfs_io_core.h — thread-safe VFS I/O execution descriptors.
6 *
7 * WHAT: Declares the POD job descriptor and small segment descriptor types used
8 * by worker-thread and inline-fallback disk I/O. brix_vfs_io_execute()
9 * consumes one initialized job, performs the requested blocking syscall /
10 * CRC operation, and writes only the job OUT fields plus caller-owned
11 * buffers.
12 *
13 * WHY: Public VFS entry points are event-loop-only because they allocate from
14 * nginx pools, emit metrics/access logs, and touch cache state. This header
15 * exposes the narrower worker-safe execution core so stream AIO workers and
16 * synchronous fallbacks can share one raw-I/O implementation without
17 * pulling loop-only VFS internals into a thread.
18 *
19 * HOW: Callers fill immutable IN fields on brix_vfs_job_t, optionally pass
20 * readv/writev segment arrays and an error-message buffer, then call
21 * brix_vfs_io_execute(). The function zeroes OUT fields first, dispatches
22 * by op, captures errno into io_errno, and returns via the job.
23 *
24 * Requires: ngx_config.h / ngx_core.h types are included here explicitly.
25 */
26

103/* ---- brix_vfs_job_read_init — initialize a thread-safe READ job ----
104 *
105 * WHAT: Zeroes *job and fills a READ request over fd/offset/length into dst.
106 * dst_cap is retained for defensive validation. Returns nothing.
107 *
108 * WHY: nginx thread tasks are reused; a dedicated initializer prevents stale
109 * OUT fields from a previous operation from leaking into a new worker run.
110 *
111 * HOW: ngx_memzero() the descriptor, set op/fd/offset/length/buf/buf_cap, and
112 * leave all OUT fields clear for brix_vfs_io_execute().
113 */
114static ngx_inline void
115brix_vfs_job_read_init(brix_vfs_job_t *job, ngx_fd_t fd, off_t offset,
116 size_t length, u_char *dst, size_t dst_cap, unsigned want_pgcrc)
117{
118 ngx_memzero(job, sizeof(*job));
119 job->op = BRIX_VFS_IO_READ;
120 job->fd = fd;
121 job->offset = offset;
122 job->length = length;
123 job->buf = dst;
124 job->buf_cap = dst_cap;
125 job->want_pgcrc = want_pgcrc ? 1 : 0;
126}

128/* The effective storage object for a data op: the handle's bound driver object
129 * when set (a non-POSIX backend), else a POSIX wrap of the bare fd (the unchanged
130 * pre-Layer-3 path). `src` is the slot/job/segment obj (may be NULL/zeroed). */
131static ngx_inline brix_sd_obj_t *
132brix_vfs_effective_obj(brix_sd_obj_t *src, ngx_fd_t fd,
133 brix_sd_obj_t *scratch)
134{
135 if (src != NULL && src->driver != NULL) {
136 return src;
137 }
138 brix_sd_posix_wrap(scratch, fd);
139 return scratch;
140}

142/* Bind the handle's storage object to a job so the executor dispatches data I/O
143 * through its driver (block-striped/object backend). A NULL or default-POSIX obj
144 * is ignored — the job keeps using its bare fd (the unchanged POSIX path). Call
145 * after the op-specific init. */
146static ngx_inline void
147brix_vfs_job_set_obj(brix_vfs_job_t *job, const brix_sd_obj_t *obj)
148{
149 if (obj != NULL && obj->driver != NULL) {
150 job->obj = *obj;
151 }
152}

154/* ---- brix_vfs_job_write_init — initialize a thread-safe WRITE job ----
155 *
156 * WHAT: Zeroes *job and fills a WRITE request over fd/offset from src[0..len).
157 *
158 * WHY: Provides a small, explicit adapter between protocol task structs and the
159 * VFS execution core without letting protocol state cross into the worker.
160 *
161 * HOW: ngx_memzero() the descriptor, set op/fd/offset/buf/length, and leave OUT
162 * fields clear for execution.
163 */
164static ngx_inline void
165brix_vfs_job_write_init(brix_vfs_job_t *job, ngx_fd_t fd, off_t offset,
166 const u_char *src, size_t len)
167{
168 ngx_memzero(job, sizeof(*job));
169 job->op = BRIX_VFS_IO_WRITE;
170 job->fd = fd;
171 job->offset = offset;
172 job->buf = (u_char *) src;
173 job->length = len;
174 job->buf_cap = len;

177/* ---- brix_vfs_job_sync_init — initialize a thread-safe SYNC job ----
178 *
179 * WHAT: Zeroes *job and fills a SYNC request for one open storage fd.
180 *
181 * WHY: kXR_sync and writev doSync must not bypass the VFS raw-I/O choke point
182 * when flushing main-storage handles.
183 *
184 * HOW: Store fd with op=SYNC; brix_vfs_io_execute() performs the durable
185 * flush and reports errno in the job OUT fields.
186 */
187static ngx_inline void
188brix_vfs_job_sync_init(brix_vfs_job_t *job, ngx_fd_t fd)
189{
190 ngx_memzero(job, sizeof(*job));
191 job->op = BRIX_VFS_IO_SYNC;
192 job->fd = fd;
193}

195/* ---- brix_vfs_job_truncate_init — initialize a thread-safe TRUNCATE job ----

210 job->fd = fd;
211 job->offset = length;
212}

214/* ---- brix_vfs_job_opendir_init — initialize a thread-safe OPENDIR job ----

216 * WHAT: Zeroes *job and fills an OPENDIR request over a confined directory fd.

228 ngx_flag_t want_stat, ngx_flag_t want_cksum, const char *path,
229 const char *cksum_algo, ngx_log_t *log, char *err_msg, size_t err_msg_cap)
230{
231 ngx_memzero(job, sizeof(*job));
232 job->op = BRIX_VFS_IO_OPENDIR;
233 job->rootfd = rootfd;
234 job->buf = response;
235 job->buf_cap = response_cap;
236 job->want_stat = want_stat ? 1 : 0;
237 job->want_cksum = want_cksum ? 1 : 0;
238 job->streamid[0] = streamid[0];
239 job->streamid[1] = streamid[1];
