1/*
2 * cms/recv_frame.c — opcode frame handlers + dispatch for the CMS client-side
3 * (manager-connection) receive path.
4 *
5 * WHAT: The per-opcode handlers a connected CMS node runs on frames from its
6 * manager — PING→PONG, SPACE→AVAIL, STATUS→suspend/resume, SELECT/TRY→redirect
7 * a waiting client, STATE→kYR_have existence probe, forwarded namespace ops,
8 * UPDATE, DISC — plus the pending-client redirect/wake plumbing
9 * (cms_wake_pending_session / cms_find_client_connection), the frame dispatch
10 * table, and the router ngx_brix_cms_process_frame().
11 *
12 * WHY: Split (Phase-79 file-size split) from recv.c so the opcode-handler set
13 * and its dispatch table live in one focused, independently reviewable file
14 * under the size guideline, separate from the read/framing event loop (recv.c)
15 * and the Plane-B storage-mutation executor (recv_forward.c). The one entry
16 * point the read loop calls (ngx_brix_cms_process_frame) is declared in
17 * recv_internal.h; the forwarded-op executor it invokes (cms_node_exec_forward)
18 * is declared there too.
19 *
20 * HOW: cms_frame_* handlers share the cms_frame_handler_pt signature and derive
21 * their payload view from ctx; cms_frame_table maps opcode → handler;
22 * ngx_brix_cms_process_frame decodes the header's streamid/rrCode and dispatches
23 * through the table, silently ignoring unknown opcodes.
24 */

41/* cms_wake_pending_session — parse the first host:port from a kYR_select/kYR_try
42 * payload and wake the suspended XRootD client waiting on its locate: look up the
43 * pending entry by streamid+pid, resolve its saved fd to the live connection (same
44 * worker, per-worker design), set XRD_ST_REQ_HEADER, brix_send_redirect to the
45 * resolved server, and resume reading. */
46
47static ngx_int_t
48cms_wake_pending_session(ngx_brix_cms_ctx_t *cms_ctx, uint32_t streamid,
49 const char *host, uint16_t port)
50{
51 brix_pending_locate_t *pending;
52 ngx_connection_t *client_conn;
53 ngx_stream_session_t *session;
54 brix_ctx_t *xrd_ctx;
55 int conn_fd;
56 ngx_atomic_uint_t conn_number;
57 u_char client_streamid[2];
58
59 pending = brix_pending_lookup(streamid, ngx_pid);
60 if (pending == NULL) {
61 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, cms_ctx->cycle->log, 0,
62 "brix: CMS wake: streamid=%uD not found in pending table",
63 streamid);
64 return NGX_OK; /* session timed out and was already removed */
65 }
66
67 conn_fd = pending->conn_fd;
68 conn_number = pending->conn_number;
69 client_streamid[0] = pending->client_streamid[0];
70 client_streamid[1] = pending->client_streamid[1];
71 brix_pending_unlock();
72
73 brix_pending_remove(streamid, ngx_pid);
74
75 client_conn = cms_find_client_connection(conn_fd);
76 if (client_conn == NULL || client_conn->number != conn_number) {
77 return NGX_OK; /* fd was recycled after the client disconnected */
78 }
79
80 session = client_conn->data;
81 if (session == NULL) {
82 return NGX_OK;
83 }
84
85 xrd_ctx = ngx_stream_get_module_ctx(session, ngx_stream_brix_module);
86 if (xrd_ctx == NULL || xrd_ctx->state != XRD_ST_WAITING_CMS) {
87 return NGX_OK;
88 }
89
90 /*
91 * WS6: the redirect host comes straight from the manager's kYR_select /
92 * kYR_try payload and is copied verbatim into the "Shost:port" redirect the
93 * client parses. A compromised/hostile manager could inject control bytes or
94 * an alternate scheme here, so validate it with the same character allowlist
95 * the registry uses as its store choke point (brix_net_host_chars_valid).
96 * On reject, drop the redirect and leave the client in XRD_ST_WAITING_CMS to
97 * hit its own cms_locate_timeout — we never emit a poisoned host.
98 */
99 if (host == NULL
100 || !brix_net_host_chars_valid(host, ngx_strlen(host)))
101 {
102 char safe[256];
103 brix_sanitize_log_string(host, safe, sizeof(safe));
104 ngx_log_error(NGX_LOG_WARN, cms_ctx->cycle->log, 0,
105 "brix: CMS select: rejected redirect to invalid host "
106 "\"%s\" for fd=%d", safe, conn_fd);
107 return NGX_OK;
108 }
109
110 ngx_log_error(NGX_LOG_INFO, cms_ctx->cycle->log, 0,
111 "brix: CMS select: redirecting client fd=%d to %s:%u",
112 conn_fd, host, (unsigned) port);
113
114 ngx_del_timer(client_conn->read);
115 xrd_ctx->state = XRD_ST_REQ_HEADER;
116 xrd_ctx->recv.cur_streamid[0] = client_streamid[0];
117 xrd_ctx->recv.cur_streamid[1] = client_streamid[1];
118 if (brix_send_redirect(xrd_ctx, client_conn, host, port) == NGX_ERROR) {
119 ngx_log_error(NGX_LOG_ERR, cms_ctx->cycle->log, 0,
120 "brix: CMS select: failed to queue redirect for fd=%d",
121 conn_fd);
122 return NGX_ERROR;
123 }
124 brix_schedule_read_resume(client_conn);
125 return NGX_OK;
126}

128static ngx_connection_t *
129cms_find_client_connection(int fd)
130{
131 ngx_uint_t i;
132 ngx_connection_t *c;
133
134 if (fd < 0) {
135 return NULL;
136 }
137
138 if (ngx_cycle->files != NULL && (ngx_uint_t) fd < ngx_cycle->files_n) {
139 c = ngx_cycle->files[fd];
140 if (c != NULL && c->fd == fd) {
141 return c;
142 }
143 return NULL;
144 }
145
146 for (i = 0; i < ngx_cycle->connection_n; i++) {
147 c = &ngx_cycle->connections[i];
148 if (c->fd == fd) {
149 return c;
150 }
151 }
152
153 return NULL;
154}

172/* cms_frame_status — kYR_status: the manager flips our login gate. The modifier
173 * byte (offset 5) selects suspend (pause new logins) or resume; any other
174 * modifier is a no-op, matching stock cmsd. */
175static ngx_int_t
176cms_frame_status(ngx_brix_cms_ctx_t *ctx, uint32_t streamid, u_char code)
177{
178 u_char mod = ctx->inbuf[5];
179 if (mod & CMS_ST_SUSPEND) {
180 ctx->conf->cms.suspended = 1;
181 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
182 "brix: CMS suspend received — new logins paused");
183 } else if (mod & CMS_ST_RESUME) {
184 ctx->conf->cms.suspended = 0;
185 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
186 "brix: CMS resume received — accepting logins");
187 } else {
188 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
189 "brix: CMS status modifier=0x%02xi (no action)",
190 (ngx_uint_t) mod);
191 }
192 return NGX_OK;
193}

195/* cms_frame_redirect — kYR_select / kYR_try: the manager resolved a pending
196 * kYR_locate and names a server. Both payloads carry a NUL-terminated hostname
197 * + 2-byte big-endian port (kYR_try is an ordered list of such entries — use
198 * only the first; the client retries remaining entries itself), so one handler
199 * serves both opcodes. Truncated payloads are silently ignored. */
200static ngx_int_t
201cms_frame_redirect(ngx_brix_cms_ctx_t *ctx, uint32_t streamid, u_char code)
202{
203 const u_char *payload = ctx->inbuf + NGX_BRIX_CMS_HDR_LEN;
204 size_t payload_len = ctx->in_need - NGX_BRIX_CMS_HDR_LEN;
205 char host[256];
206 size_t host_len;
207 uint16_t port;
208
209 if (payload_len < 3) {
210 /* need at least one host byte, a NUL, and two port bytes */
211 return NGX_OK;
212 }
213
214 ngx_cpystrn((u_char *) host, (u_char *) payload, sizeof(host));
215 host_len = ngx_strlen(host);
216
217 if (host_len + 3 > payload_len) {
218 /* port bytes would fall outside the received payload */
219 return NGX_OK;
220 }
221
222 port = ngx_brix_cms_get16(payload + host_len + 1);
223 return cms_wake_pending_session(ctx, streamid, host, port);
224}

226/* cms_state_extract_path — pure validation of a kYR_state payload: bound the
227 * NUL-terminated namespace path, require an absolute path that fits the buffer,
228 * and reject any ".." traversal before the registry/filesystem is touched
229 * (cheap defence-in-depth ahead of the kernel-confined probe). Copies the
230 * NUL-terminated path into pathz and returns its length via *pl_out; returns
231 * NGX_OK or NGX_ERROR (caller stays silent, matching real cmsd). */
232static ngx_int_t
233cms_state_extract_path(const u_char *payload, size_t plen, char *pathz,
234 size_t pathz_size, size_t *pl_out)
235{
236 size_t pl;
237 size_t k;
238
239 /* bounded length of the NUL-terminated path */
240 for (pl = 0; pl < plen && payload[pl] != '\0'; pl++) { /* void */ }
241 if (pl == 0 || payload[0] != '/' || pl >= pathz_size) {
242 return NGX_ERROR;
243 }
244
245 /* reject path traversal before touching the registry/filesystem */
246 for (k = 0; k + 1 < pl; k++) {
247 if (payload[k] == '.' && payload[k + 1] == '.') {
248 return NGX_ERROR;
249 }
250 }
251
252 ngx_memcpy(pathz, payload, pl);
253 pathz[pl] = '\0';
254 *pl_out = pl;
255 return NGX_OK;
256}

258/* cms_frame_state — kYR_state (raw): the manager asks "do you hold <path>?" as
259 * part of on-demand selection. The payload is the raw NUL-terminated namespace
260 * path (no Pup framing). We answer kYR_have (echoing streamid = path hash) if
261 * we can serve the path, else stay silent so the manager won't select us —
262 * matching real cmsd.
263 *
264 * Two ways to "have" a path:
265 * - manager_mode (a sub-manager registered UP to a meta-manager): forward the
266 * query to our own server registry — if any registered leaf data node
267 * exports a prefix covering the path, we have it (the client will be
268 * redirected to us and we then redirect down to the leaf). This is what
269 * makes a multi-tier meta->nginx->leaf mesh resolve.
270 * - data node: the file exists on our local export filesystem. */
271static ngx_int_t
272cms_frame_state(ngx_brix_cms_ctx_t *ctx, uint32_t streamid, u_char code)
273{
274 const u_char *payload = ctx->inbuf + NGX_BRIX_CMS_HDR_LEN;
275 size_t plen = ctx->in_need - NGX_BRIX_CMS_HDR_LEN;
276 char pathz[1024];
277 size_t pl;
278 struct stat st;
279
280 if (cms_state_extract_path(payload, plen, pathz, sizeof(pathz), &pl)
281 != NGX_OK)
282 {
283 return NGX_OK;
284 }
285
286 if (ctx->conf->manager_mode) {
287 char host[256];
288 uint16_t dport;
289 if (brix_srv_select(pathz, 0, host, sizeof(host), &dport)) {
290 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
291 "brix: CMS state(mgr): registry serves "
292 "\"%*s\", replying kYR_have", pl, payload);
293 return ngx_brix_cms_send_have(ctx, streamid, pathz, pl);
294 }
295 return NGX_OK;
296 }
297
298 /*
299 * Kernel-confined existence probe. A malicious manager can ask
300 * "do you hold <path>?" for ANY path; the raw stat() this replaced
301 * followed symlinks, so a symlink planted under the export root
302 * (e.g. /link -> /etc) would make us answer kYR_have for a file
303 * OUTSIDE the root — a cross-root information leak and a
304 * cluster-poisoning vector. brix_stat_beneath() resolves the
305 * path under the persistent export rootfd with openat2
306 * RESOLVE_BENEATH, so any symlink or ".." that escapes the root is
307 * rejected by the kernel and we correctly stay silent. (The ".."
308 * pre-check in cms_state_extract_path remains as cheap
309 * defence-in-depth.) A node with no local export root (rootfd < 0)
310 * never holds files locally.
311 */
312 if (ctx->conf->rootfd >= 0
313 && brix_stat_beneath(ctx->conf->rootfd, pathz, &st) == 0)
314 {
315 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
316 "brix: CMS state: have \"%*s\", "
317 "replying kYR_have", pl, payload);
318 return ngx_brix_cms_send_have(ctx, streamid, pathz, pl);
319 }
320
321 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
322 "brix: CMS state: do not have \"%*s\"",
323 pl, payload);
324 return NGX_OK;
325}

392/* ngx_brix_cms_process_frame — decode a complete CMS frame's streamid + rrCode
393 * (first 4 bytes + offset 4) and dispatch by opcode through cms_frame_table:
394 * PING→PONG, SPACE→AVAIL, STATUS→suspend/resume conf flags, SELECT/TRY→client
395 * redirect, STATE→kYR_have probe, forwarded namespace ops, UPDATE, DISC.
396 * Unknown opcodes are silently ignored (debug log). */
397
398ngx_int_t
399ngx_brix_cms_process_frame(ngx_brix_cms_ctx_t *ctx)
400{
401 uint32_t streamid;
402 u_char code;
403 ngx_uint_t i;
404 const brix_cms_route_t *r;
405
406 streamid = ngx_brix_cms_get32(ctx->inbuf);
407 code = ctx->inbuf[4];
408
409 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
410 "brix: CMS process frame code=%ui streamid=%uD",
411 (ngx_uint_t) code, streamid);
412
413 for (i = 0; i < sizeof(cms_frame_table) / sizeof(cms_frame_table[0]); i++) {
414 if (cms_frame_table[i].code == code) {
415 return cms_frame_table[i].handler(ctx, streamid, code);
416 }
417 }
418
419 r = brix_cms_route_lookup(XRDCMS_ROLE_NODE, code);
420 if (r != NULL) {
421 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
422 "brix: CMS node: unhandled opcode '%s'", r->name);
423 } else {
424 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
425 "brix: ignoring CMS rrCode=%ui", (ngx_uint_t) code);
426 }
427 return NGX_OK;
428}
