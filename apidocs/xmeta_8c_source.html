1/*
2 * fs/meta/xmeta.c — unified per-file metadata record codec (xmeta P1).
3 *
4 * WHAT: Encode/decode the one-record-per-file metadata blob: a byte-identical
5 * stock XrdPfc cinfo v4 prefix followed by "XCX1" TLV extension
6 * sections (STATE / DIGEST / BLOCKCRC). WHY: one form of metadata on
7 * disk, readable by stock tools up front, extended at the end (spec
8 * 2026-07-02-xmeta-unified-metadata-design.md). HOW: pure C, ngx-free,
9 * malloc-based; every variable-length region is guarded by a crc32c,
10 * matching stock's own Store/bitmap checksum convention.
11 */
12

30static uint64_t
31xmeta_nblocks(int64_t file_size, int64_t buffer_size)
32{
33 if (file_size <= 0) {
34 return 0;
35 }
36 return ((uint64_t) file_size + (uint64_t) buffer_size - 1)
37 / (uint64_t) buffer_size;
38}
39
40static size_t
41xmeta_bitmap_len(uint64_t nblocks)
42{
43 return (size_t) ((nblocks + 7) / 8);
44}

46/* append raw bytes into a cursor-tracked buffer (capacity computed upfront) */
47static void
48xmeta_put(uint8_t *buf, size_t *off, const void *src, size_t len)
49{
50 memcpy(buf + *off, src, len);
51 *off += len;
52}

54/* bounded sequential read from the decode buffer; 0 = ok, -1 = short */
55static int
56xmeta_get(const uint8_t *buf, size_t len, size_t *off, void *dst, size_t n)
57{
58 if (len - *off < n) {
59 return -1;
60 }
61 memcpy(dst, buf + *off, n);
62 *off += n;
63 return 0;
64}

68int
69brix_xmeta_init(brix_xmeta_t *m, int64_t file_size, int64_t buffer_size)
70{
71 uint64_t nb;
72
73 if (m == NULL || buffer_size <= 0 || file_size < 0) {
74 errno = EINVAL;
75 return BRIX_XMETA_ERR;
76 }
77 nb = xmeta_nblocks(file_size, buffer_size);
78 if (nb > BRIX_XMETA_MAX_BLOCKS) {
79 errno = EFBIG;
80 return BRIX_XMETA_ERR;
81 }
82
83 memset(m, 0, sizeof(*m));
84 m->buffer_size = buffer_size;
85 m->file_size = file_size;
86 m->creation_time = (int64_t) time(NULL);
87 m->nblocks = nb;
88 m->have_state = 1;
89
90 m->bitmap = calloc(1, xmeta_bitmap_len(nb) ? xmeta_bitmap_len(nb) : 1);
91 m->blockcrc = calloc(nb ? (size_t) nb : 1, sizeof(uint32_t));
92 if (m->bitmap == NULL || m->blockcrc == NULL) {
93 brix_xmeta_free(m);
94 errno = ENOMEM;
95 return BRIX_XMETA_ERR;
96 }
97 m->have_blockcrc = 1;
98 return BRIX_XMETA_OK;
99}

113/* ---- bitmap ops (stock cfiBIT order: bit i = byte i/8, 1 << i%8) --------- */

115void
116brix_xmeta_block_set(brix_xmeta_t *m, uint64_t i)
117{
118 if (m->bitmap != NULL && i < m->nblocks) {
119 m->bitmap[i / 8] |= (uint8_t) (1u << (i % 8));
120 }
121}

123int
124brix_xmeta_block_test(const brix_xmeta_t *m, uint64_t i)
125{
126 if (m->bitmap == NULL || i >= m->nblocks) {
127 return 0;
128 }
129 return (m->bitmap[i / 8] & (uint8_t) (1u << (i % 8))) != 0;
130}

132int
133brix_xmeta_complete(const brix_xmeta_t *m)
134{
135 uint64_t i;
136
137 for (i = 0; i < m->nblocks; i++) {
138 if (!brix_xmeta_block_test(m, i)) {
139 return 0;
140 }
141 }
142 return 1;
143}

147int
148brix_xmeta_digest_add(brix_xmeta_t *m, uint16_t alg, const void *val,
149 uint16_t len)
150{
151 size_t need = 4 + (size_t) len;
152 uint8_t *grown;
153
154 if (m == NULL || (val == NULL && len != 0)) {
155 errno = EINVAL;
156 return BRIX_XMETA_ERR;
157 }
158 grown = realloc(m->digests, m->digests_len + need);
159 if (grown == NULL) {
160 errno = ENOMEM;
161 return BRIX_XMETA_ERR;
162 }
163 m->digests = grown;
164 memcpy(m->digests + m->digests_len, &alg, 2);
165 memcpy(m->digests + m->digests_len + 2, &len, 2);
166 if (len > 0) {
167 memcpy(m->digests + m->digests_len + 4, val, len);
168 }
169 m->digests_len += (uint32_t) need;
170 return BRIX_XMETA_OK;
171}

173int
174brix_xmeta_digest_set(brix_xmeta_t *m, uint16_t alg, const void *val,
175 uint16_t len)
176{
177 uint8_t *kept = NULL;
178 uint32_t kept_len = 0;
179 size_t off = 0;
180
181 if (m == NULL) {
182 errno = EINVAL;
183 return BRIX_XMETA_ERR;
184 }
185 if (m->digests != NULL) {
186 kept = malloc(m->digests_len ? m->digests_len : 1);
187 if (kept == NULL) {
188 errno = ENOMEM;
189 return BRIX_XMETA_ERR;
190 }
191 while (off + 4 <= m->digests_len) {
192 uint16_t a, l;
193
194 memcpy(&a, m->digests + off, 2);
195 memcpy(&l, m->digests + off + 2, 2);
196 if (off + 4 + l > m->digests_len) {
197 break; /* malformed tail: drop it */
198 }
199 if (a != alg) {
200 memcpy(kept + kept_len, m->digests + off, 4 + (size_t) l);
201 kept_len += 4 + (uint32_t) l;
202 }
203 off += 4 + (size_t) l;
204 }
205 free(m->digests);
206 m->digests = kept;
207 m->digests_len = kept_len;
208 if (kept_len == 0) {
209 free(m->digests);
210 m->digests = NULL;
211 m->digests_len = 0;
212 }
213 }
214 return brix_xmeta_digest_add(m, alg, val, len);
215}

217int
218brix_xmeta_digest_get(const brix_xmeta_t *m, uint32_t idx, uint16_t *alg,
219 const uint8_t **val, uint16_t *len)
220{
221 size_t off = 0;
222 uint32_t i = 0;
223
224 if (m == NULL || m->digests == NULL) {
225 return BRIX_XMETA_FOREIGN;
226 }
227 while (off + 4 <= m->digests_len) {
228 uint16_t a, l;
229
230 memcpy(&a, m->digests + off, 2);
231 memcpy(&l, m->digests + off + 2, 2);
232 if (off + 4 + l > m->digests_len) {
233 return BRIX_XMETA_FOREIGN; /* malformed payload */
234 }
235 if (i == idx) {
236 *alg = a;
237 *len = l;
238 *val = m->digests + off + 4;
239 return BRIX_XMETA_OK;
240 }
241 off += 4 + (size_t) l;
242 i++;
243 }
244 return BRIX_XMETA_FOREIGN; /* past the end */
245}

247/* ---- STATE section wire layout (80 bytes) --------------------------------- */
248
249typedef struct {
250 uint64_t origin_mtime;
251 uint64_t dirty_lo;
252 uint64_t dirty_hi;
253 uint64_t flush_gen;
254 uint64_t dirty_since;
255 uint64_t last_flush;
256 uint64_t bytes_flushed;
257 uint64_t expires_at;
258 uint64_t filled_at;
259 uint32_t mode;
260 uint32_t state_flags;
261} xmeta_state_wire_t;
262
263typedef char xmeta_state_size_check[
264 sizeof(xmeta_state_wire_t) == 80 ? 1 : -1];
265
266/* ORIGIN section: {u8 etag_len, u8 alg_len, u8 cks_len, u8 pad, strings...} */
267#define XMETA_ORIGIN_FIXED 4
268
269/* ---- encode ---------------------------------------------------------------- */
270
271/* one TLV section: {u16 type, u16 reserved, u32 len, payload, u32 crc32c},
272 * crc computed over the 8-byte section header + payload */
273static void
274xmeta_put_section(uint8_t *buf, size_t *off, uint16_t type,
275 const void *payload, uint32_t plen)
276{
277 size_t hdr_off = *off;
278 uint16_t reserved = 0;
279 uint32_t crc;
280
281 xmeta_put(buf, off, &type, 2);
282 xmeta_put(buf, off, &reserved, 2);
283 xmeta_put(buf, off, &plen, 4);
284 xmeta_put(buf, off, payload, plen);
285 crc = brix_crc32c_value(buf + hdr_off, 8 + (size_t) plen);
286 xmeta_put(buf, off, &crc, 4);
287}

289int
290brix_xmeta_encode(const brix_xmeta_t *m, uint8_t **out, size_t *out_len)
291{
292 brix_xmeta_stock_store_t store;
293 xmeta_state_wire_t state;
294 int32_t version = BRIX_XMETA_STOCK_VERSION;
295 uint32_t ext_magic = BRIX_XMETA_EXT_MAGIC;
296 uint16_t ext_version = BRIX_XMETA_EXT_VERSION;
297 uint16_t nsec;
298 size_t blen, cap, off = 0;
299 size_t blockcrc_plen = 0;
300 uint32_t crc;
301 uint8_t *buf;
302
303 if (m == NULL || out == NULL || out_len == NULL || m->buffer_size <= 0
304 || m->bitmap == NULL)
305 {
306 errno = EINVAL;
307 return BRIX_XMETA_ERR;
308 }
309
310 blen = xmeta_bitmap_len(m->nblocks);
311 unsigned have_origin = (m->etag_len > 0 || m->cks_len > 0
312 || m->cks_alg_len > 0);
313 size_t origin_plen = XMETA_ORIGIN_FIXED + m->etag_len
314 + m->cks_alg_len + m->cks_len;
315
316 nsec = (uint16_t) ((m->have_state ? 1 : 0)
317 + (m->digests != NULL ? 1 : 0)
318 + (m->have_blockcrc && m->blockcrc != NULL ? 1 : 0)
319 + (have_origin ? 1 : 0));
320 if (m->have_blockcrc && m->blockcrc != NULL) {
321 blockcrc_plen = 16 + (size_t) m->nblocks * 4;
322 }
323
324 cap = 4 + sizeof(store) + 4 /* version+store+crc */
325 + blen + (m->astat_count > 0 ? sizeof(m->astat) : 0) + 4
326 + 8 /* ext magic+ver+cnt */
327 + (m->have_state ? 12 + sizeof(state) : 0)
328 + (m->digests != NULL ? 12 + (size_t) m->digests_len : 0)
329 + (blockcrc_plen ? 12 + blockcrc_plen : 0)
330 + (have_origin ? 12 + origin_plen : 0);
331
332 buf = malloc(cap);
333 if (buf == NULL) {
334 errno = ENOMEM;
335 return BRIX_XMETA_ERR;
336 }
337
338 /* stock prefix — exactly what XrdPfc::Info::Write emits */
339 memset(&store, 0, sizeof(store));
340 store.buffer_size = m->buffer_size;
341 store.file_size = m->file_size;
342 store.creation_time = m->creation_time;
343 store.no_cksum_time = m->no_cksum_time;
344 store.access_cnt = m->access_cnt;
345 store.status_raw = m->status_raw;
346 store.astat_size = (m->astat_count > 0) ? 1 : 0;
347
348 xmeta_put(buf, &off, &version, 4);
349 xmeta_put(buf, &off, &store, sizeof(store));
350 crc = brix_crc32c_value(&store, sizeof(store));
351 xmeta_put(buf, &off, &crc, 4);
352 xmeta_put(buf, &off, m->bitmap, blen);
353 crc = brix_crc32c_value(m->bitmap, blen);
354 if (store.astat_size > 0) {
355 xmeta_put(buf, &off, &m->astat, sizeof(m->astat));
356 crc = brix_crc32c_extend(crc, &m->astat, sizeof(m->astat));
357 }
358 xmeta_put(buf, &off, &crc, 4);
359
360 /* extension */
361 xmeta_put(buf, &off, &ext_magic, 4);
362 xmeta_put(buf, &off, &ext_version, 2);
363 xmeta_put(buf, &off, &nsec, 2);
364
365 if (m->have_state) {
366 memset(&state, 0, sizeof(state));
367 state.origin_mtime = m->origin_mtime;
368 state.dirty_lo = m->dirty_lo;
369 state.dirty_hi = m->dirty_hi;
370 state.flush_gen = m->flush_gen;
371 state.dirty_since = m->dirty_since;
372 state.last_flush = m->last_flush;
373 state.bytes_flushed = m->bytes_flushed;
374 state.expires_at = m->expires_at;
375 state.filled_at = m->filled_at;
376 state.mode = m->mode;
377 state.state_flags = m->state_flags;
378 xmeta_put_section(buf, &off, BRIX_XMETA_SEC_STATE,
379 &state, sizeof(state));
380 }
381 if (have_origin) {
382 uint8_t hdr4[XMETA_ORIGIN_FIXED] = {
383 m->etag_len, m->cks_alg_len, m->cks_len, 0
384 };
385 uint8_t payload[XMETA_ORIGIN_FIXED + sizeof(m->etag)
386 + sizeof(m->cks_alg) + sizeof(m->cks_hex)];
387 size_t poff = 0;
388
389 memcpy(payload, hdr4, XMETA_ORIGIN_FIXED); poff = XMETA_ORIGIN_FIXED;
390 memcpy(payload + poff, m->etag, m->etag_len); poff += m->etag_len;
391 memcpy(payload + poff, m->cks_alg, m->cks_alg_len); poff += m->cks_alg_len;
392 memcpy(payload + poff, m->cks_hex, m->cks_len); poff += m->cks_len;
393 xmeta_put_section(buf, &off, BRIX_XMETA_SEC_ORIGIN,
394 payload, (uint32_t) poff);
395 }
396 if (m->digests != NULL) {
397 xmeta_put_section(buf, &off, BRIX_XMETA_SEC_DIGEST,
398 m->digests, m->digests_len);
399 }
400 if (blockcrc_plen > 0) {
401 /* payload: u32 granule, u32 reserved, u64 nblocks, u32 crc[] — built
402 * in place to avoid a second nblocks*4 allocation */
403 size_t sec_hdr = off;
404 uint16_t type = BRIX_XMETA_SEC_BLOCKCRC, rsv = 0;
405 uint32_t plen32 = (uint32_t) blockcrc_plen;
406 uint32_t granule = (uint32_t) m->buffer_size;
407 uint32_t zero = 0;
408 uint64_t nb = m->nblocks;
409
410 xmeta_put(buf, &off, &type, 2);
411 xmeta_put(buf, &off, &rsv, 2);
412 xmeta_put(buf, &off, &plen32, 4);
413 xmeta_put(buf, &off, &granule, 4);
414 xmeta_put(buf, &off, &zero, 4);
415 xmeta_put(buf, &off, &nb, 8);
416 xmeta_put(buf, &off, m->blockcrc, (size_t) m->nblocks * 4);
417 crc = brix_crc32c_value(buf + sec_hdr, 8 + blockcrc_plen);
418 xmeta_put(buf, &off, &crc, 4);
419 }
420
421 *out = buf;
422 *out_len = off;
423 return BRIX_XMETA_OK;
424}

426/* ---- decode ---------------------------------------------------------------- */

428static int
429xmeta_decode_section(brix_xmeta_t *m, uint16_t type, const uint8_t *p,
430 uint32_t plen)
431{
432 switch (type) {
433
434 case BRIX_XMETA_SEC_STATE: {
435 xmeta_state_wire_t state;
436
437 if (plen < sizeof(state)) {
438 return BRIX_XMETA_ERR;
439 }
440 memcpy(&state, p, sizeof(state));
441 m->have_state = 1;
442 m->origin_mtime = state.origin_mtime;
443 m->dirty_lo = state.dirty_lo;
444 m->dirty_hi = state.dirty_hi;
445 m->flush_gen = state.flush_gen;
446 m->dirty_since = state.dirty_since;
447 m->last_flush = state.last_flush;
448 m->bytes_flushed = state.bytes_flushed;
449 m->expires_at = state.expires_at;
450 m->filled_at = state.filled_at;
451 m->mode = state.mode;
452 m->state_flags = state.state_flags;
453 return BRIX_XMETA_OK;
454 }
455
456 case BRIX_XMETA_SEC_ORIGIN: {
457 uint8_t el, al, cl;
458
459 if (plen < XMETA_ORIGIN_FIXED) {
460 return BRIX_XMETA_ERR;
461 }
462 el = p[0]; al = p[1]; cl = p[2];
463 if (el > sizeof(m->etag) || al > sizeof(m->cks_alg)
464 || cl > sizeof(m->cks_hex)
465 || plen < (uint32_t) XMETA_ORIGIN_FIXED + el + al + cl)
466 {
467 return BRIX_XMETA_ERR;
468 }
469 m->etag_len = el;
470 memcpy(m->etag, p + XMETA_ORIGIN_FIXED, el);
471 m->cks_alg_len = al;
472 memcpy(m->cks_alg, p + XMETA_ORIGIN_FIXED + el, al);
473 m->cks_len = cl;
474 memcpy(m->cks_hex, p + XMETA_ORIGIN_FIXED + el + al, cl);
475 return BRIX_XMETA_OK;
476 }
477
478 case BRIX_XMETA_SEC_DIGEST: {
479 uint8_t *copy = malloc(plen ? plen : 1);
480
481 if (copy == NULL) {
482 errno = ENOMEM;
483 return BRIX_XMETA_ERR;
484 }
485 memcpy(copy, p, plen);
486 free(m->digests);
487 m->digests = copy;
488 m->digests_len = plen;
489 return BRIX_XMETA_OK;
490 }
491
492 case BRIX_XMETA_SEC_BLOCKCRC: {
493 uint64_t nb;
494
495 if (plen < 16) {
496 return BRIX_XMETA_ERR;
497 }
498 memcpy(&nb, p + 8, 8);
499 if (nb != m->nblocks || plen != 16 + nb * 4) {
500 return BRIX_XMETA_ERR; /* table doesn't match the file */
501 }
502 free(m->blockcrc);
503 m->blockcrc = malloc(nb ? (size_t) nb * 4 : 4);
504 if (m->blockcrc == NULL) {
505 errno = ENOMEM;
506 return BRIX_XMETA_ERR;
507 }
508 memcpy(m->blockcrc, p + 16, (size_t) nb * 4);
509 m->have_blockcrc = 1;
510 return BRIX_XMETA_OK;
511 }
512
513 default:
514 return BRIX_XMETA_OK; /* unknown type: skipped (fwd compat) */
515 }
516}

518int
519brix_xmeta_decode(const uint8_t *buf, size_t len, brix_xmeta_t *m)
520{
521 brix_xmeta_stock_store_t store;
522 int32_t version;
523 uint32_t crc, want, ext_magic;
524 uint16_t ext_version, nsec, i;
525 size_t off = 0, blen;
526
527 if (buf == NULL || m == NULL) {
528 errno = EINVAL;
529 return BRIX_XMETA_ERR;
530 }
531 memset(m, 0, sizeof(*m));
532
533 /* stock prefix */
534 if (xmeta_get(buf, len, &off, &version, 4) != 0
535 || version != BRIX_XMETA_STOCK_VERSION)
536 {
537 return BRIX_XMETA_FOREIGN;
538 }
539 if (xmeta_get(buf, len, &off, &store, sizeof(store)) != 0
540 || xmeta_get(buf, len, &off, &crc, 4) != 0)
541 {
542 return BRIX_XMETA_FOREIGN;
543 }
544 if (crc != brix_crc32c_value(&store, sizeof(store))) {
545 errno = EBADMSG;
546 return BRIX_XMETA_ERR;
547 }
548 if (store.buffer_size <= 0 || store.file_size < 0
549 || store.astat_size < 0)
550 {
551 return BRIX_XMETA_FOREIGN;
552 }
553
554 m->buffer_size = store.buffer_size;
555 m->file_size = store.file_size;
556 m->creation_time = store.creation_time;
557 m->no_cksum_time = store.no_cksum_time;
558 m->access_cnt = store.access_cnt;
559 m->status_raw = store.status_raw;
560 m->astat_count = store.astat_size;
561 m->nblocks = xmeta_nblocks(store.file_size, store.buffer_size);
562 if (m->nblocks > BRIX_XMETA_MAX_BLOCKS) {
563 return BRIX_XMETA_FOREIGN;
564 }
565
566 blen = xmeta_bitmap_len(m->nblocks);
567 m->bitmap = malloc(blen ? blen : 1);
568 if (m->bitmap == NULL) {
569 errno = ENOMEM;
570 return BRIX_XMETA_ERR;
571 }
572 if (xmeta_get(buf, len, &off, m->bitmap, blen) != 0) {
573 brix_xmeta_free(m);
574 return BRIX_XMETA_FOREIGN;
575 }
576 want = brix_crc32c_value(m->bitmap, blen);
577
578 /* AStat records: keep the first, checksum them all */
579 for (i = 0; (int32_t) i < store.astat_size; i++) {
580 brix_xmeta_astat_t a;
581
582 if (xmeta_get(buf, len, &off, &a, sizeof(a)) != 0) {
583 brix_xmeta_free(m);
584 return BRIX_XMETA_FOREIGN;
585 }
586 if (i == 0) {
587 m->astat = a;
588 }
589 want = brix_crc32c_extend(want, &a, sizeof(a));
590 }
591 if (xmeta_get(buf, len, &off, &crc, 4) != 0) {
592 brix_xmeta_free(m);
593 return BRIX_XMETA_FOREIGN;
594 }
595 if (crc != want) {
596 brix_xmeta_free(m);
597 errno = EBADMSG;
598 return BRIX_XMETA_ERR;
599 }
600
601 /* extension (optional: a pure stock cinfo ends here) */
602 if (off == len) {
603 return BRIX_XMETA_OK;
604 }
605 if (xmeta_get(buf, len, &off, &ext_magic, 4) != 0
606 || ext_magic != BRIX_XMETA_EXT_MAGIC
607 || xmeta_get(buf, len, &off, &ext_version, 2) != 0
608 || xmeta_get(buf, len, &off, &nsec, 2) != 0)
609 {
610 return BRIX_XMETA_OK; /* foreign trailer: stock part stands */
611 }
612
613 for (i = 0; i < nsec; i++) {
614 uint16_t type, rsv;
615 uint32_t plen;
616 size_t sec_hdr = off;
617
618 if (xmeta_get(buf, len, &off, &type, 2) != 0
619 || xmeta_get(buf, len, &off, &rsv, 2) != 0
620 || xmeta_get(buf, len, &off, &plen, 4) != 0
621 || plen > BRIX_XMETA_MAX_SECTION
622 || len - off < (size_t) plen + 4)
623 {
624 brix_xmeta_free(m);
625 errno = EBADMSG;
