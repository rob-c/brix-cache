2 * codec_bzip2.c — bzip2 backend for the codec abstraction (BRIX_CODEC_BZIP2).
3 *
4 * WHAT: Streaming bzip2 (de)compression via libbz2's low-level bz_stream API.
5 * WHY: bzip2 is a legacy-but-common codec; optional (compile-gated on

32static int
33bzip2_init(void **state, brix_codec_id_t id, brix_codec_dir_t dir, int level)
34{
35 bz_state *st = calloc(1, sizeof(*st));
36 int r;
37
38 (void) id;
39 if (st == NULL) {
40 return BRIX_CODEC_ERR_MEM;
41 }
42 st->dir = dir;
43 if (dir == BRIX_CODEC_DIR_DECOMPRESS) {
44 r = BZ2_bzDecompressInit(&st->strm, 0, 0);
45 } else {
46 r = BZ2_bzCompressInit(&st->strm, level, 0, 0);
47 }
48 if (r != BZ_OK) {
49 free(st);
50 return (r == BZ_MEM_ERROR) ? BRIX_CODEC_ERR_MEM : BRIX_CODEC_ERR_PARAM;
51 }
52 st->inited = 1;
53 *state = st;
54 return 0;
55}

57static brix_codec_rc_t
58bzip2_step(void *state, const uint8_t *in, size_t in_len, size_t *in_pos,
59 uint8_t *out, size_t out_size, size_t *out_pos, int finish)
60{
61 bz_state *st = state;
62 unsigned in_avail0, out_avail0;
63 int r;
64
65 st->strm.next_in = (char *) (uintptr_t) (in + *in_pos);
66 st->strm.avail_in = bz_clamp(in_len - *in_pos);
67 st->strm.next_out = (char *) (out + *out_pos);
68 st->strm.avail_out = bz_clamp(out_size - *out_pos);
69 in_avail0 = st->strm.avail_in;
70 out_avail0 = st->strm.avail_out;
71
72 if (st->dir == BRIX_CODEC_DIR_DECOMPRESS) {
73 r = BZ2_bzDecompress(&st->strm);
74 } else {
75 r = BZ2_bzCompress(&st->strm, finish ? BZ_FINISH : BZ_RUN);
76 }
77
78 *in_pos += (size_t) (in_avail0 - st->strm.avail_in);
79 *out_pos += (size_t) (out_avail0 - st->strm.avail_out);
80
81 if (r == BZ_STREAM_END) {
82 return BRIX_CODEC_END;
83 }
84 if (r == BZ_OK || r == BZ_RUN_OK || r == BZ_FINISH_OK || r == BZ_FLUSH_OK) {
85 return BRIX_CODEC_OK;
86 }
87 if (r == BZ_MEM_ERROR) {
88 return BRIX_CODEC_ERR_MEM;
89 }
90 return BRIX_CODEC_ERR_DATA;
91}

93static void
94bzip2_end(void *state)
95{
96 bz_state *st = state;
97
98 if (st == NULL) {
99 return;
100 }
101 if (st->inited) {
102 if (st->dir == BRIX_CODEC_DIR_DECOMPRESS) {
103 BZ2_bzDecompressEnd(&st->strm);
104 } else {
105 BZ2_bzCompressEnd(&st->strm);
106 }
107 }
108 free(st);
109}
