1/* File: delegation.c — GSI X.509 proxy delegation, inbound capture (§F6)
2 * WHAT: The destination's GSI-server side of proxy delegation. After a verified
3 * kXGC_cert, brix_gsi_begin_delegation() sends a kXGS_pxyreq (a proxy-cert
4 * request, encrypted under the session cipher); brix_gsi_handle_sigpxy()
5 * consumes the client's kXGC_sigpxy (the signed proxy) and assembles the
6 * delegated credential onto the connection.
7 *
8 * WHY: So a TPC pull can authenticate to the source AS THE USER (the delegated
9 * proxy) rather than as the gateway. Gated on brix_tpc_delegate (default off).
10 *
11 * HOW: build_pxyreq(leaf) → nested XrdSutBuffer {kXGS_pxyreq + kXRS_x509_req} →
12 * encrypt with the persisted session cipher → outer {kXRS_main + kXRS_cipher_alg}
13 * → kXR_authmore. On kXGC_sigpxy: find kXRS_main → decrypt → find kXRS_x509 →
14 * assemble_proxy(signed, saved reqkey, saved chain). No goto: bundled NULL-safe
15 * cleanup; ownership-transferred fields NULLed before cleanup. */

33/* Serialise an X509 (or the whole chain when chain!=NULL) to a malloc'd PEM. */
34static u_char *
35gsi_pem_export(X509 *one, STACK_OF(X509) *chain, size_t *len)
36{
37 BIO *b = BIO_new(BIO_s_mem());
38 char *d;
39 long n;
40 u_char *out = NULL;
41 int ok = 0;
42
43 if (b != NULL) {
44 if (chain != NULL) {
45 int i;
46 ok = 1;
47 for (i = 0; i < sk_X509_num(chain); i++) {
48 if (PEM_write_bio_X509(b, sk_X509_value(chain, i)) != 1) {
49 ok = 0;
50 }
51 }
52 } else {
53 ok = (PEM_write_bio_X509(b, one) == 1);
54 }
55 }
56 if (ok) {
57 n = BIO_get_mem_data(b, &d);
58 out = malloc((size_t) n + 1);
59 if (out != NULL) {
60 memcpy(out, d, (size_t) n);
61 out[n] = '\0';
62 *len = (size_t) n;
63 }
64 }
65 BIO_free(b);
66 return out;
67}

69/*
70 * Re-encode a DER X509_REQ as PEM. The stock XrdSecgsi client parses the
71 * kXRS_x509_req bucket with PEM_read_bio_X509_REQ (XrdCryptosslX509Req), so the
72 * proxy request MUST travel as PEM on the wire; brix_gsi_build_pxyreq emits
73 * DER (consumed as DER by brix_gsi_sign_pxyreq and its unit tests), so the
74 * conversion happens here at the wire edge rather than in the crypto core.
75 * A DER request sent verbatim makes the client reject it with
76 * "could not resolve proxy request" and decline to delegate.
77 */
78static u_char *
79gsi_req_der_to_pem(const u_char *der, size_t der_len, size_t *pem_len)
80{
81 const unsigned char *p = der;
82 X509_REQ *req = d2i_X509_REQ(NULL, &p, (long) der_len);
83 BIO *b;
84 char *d;
85 long n;
86 u_char *out = NULL;
87
88 if (req == NULL) {
89 return NULL;
90 }
91 b = BIO_new(BIO_s_mem());
92 if (b != NULL && PEM_write_bio_X509_REQ(b, req) == 1) {
93 n = BIO_get_mem_data(b, &d);
94 out = malloc((size_t) n + 1);
95 if (out != NULL) {
96 memcpy(out, d, (size_t) n);
97 out[n] = '\0';
98 *pem_len = (size_t) n;
99 }
100 }
101 BIO_free(b);
102 X509_REQ_free(req);
103 return out;
104}

106/* Resolve the persisted session cipher; -1 if none was captured. */
107static int
108gsi_session_cipher(brix_ctx_t *ctx, brix_gsi_cipher_t *cipher)
109{
110 if (ctx->gsi.sess_keylen <= 0) {
111 return -1;
112 }
113 if (!brix_gsi_cipher_lookup(ctx->gsi.sess_cipher, cipher)
114 && !brix_gsi_cipher_lookup("aes-128-cbc", cipher)) {
115 return -1;
116 }

182ngx_int_t
183brix_gsi_begin_delegation(brix_ctx_t *ctx, ngx_connection_t *c,
184 ngx_stream_brix_srv_conf_t *conf,
185 X509 *leaf, STACK_OF(X509) *chain)
186{
187 bdg_ctx b;
188 brix_gsi_cipher_t cipher;
189 char err[160];
190 size_t leaf_len = 0;
191 size_t calg_len, body_len, total;
192 u_char *buf, *p;
193 uint8_t signed_rtag[1024];
194 size_t signed_rtag_len = 0;
195 uint8_t new_rtag[20];
196
197 memset(&b, 0, sizeof(b));
198 brix_gbuf_init(&b.inner);
199 brix_gbuf_init(&b.outer);
200
201 if (gsi_session_cipher(ctx, &cipher) != 0) {
202 ngx_log_error(NGX_LOG_WARN, c->log, 0,
203 "brix: GSI delegation: no session cipher captured");
204 return bdg_fail(&b);
205 }
206
207 b.leaf_pem = gsi_pem_export(leaf, NULL, &leaf_len);
208 b.chain_pem = gsi_pem_export(NULL, chain, &b.chain_len);
209 if (b.leaf_pem == NULL || b.chain_pem == NULL) {
210 return bdg_fail(&b);
211 }
212
213 /* Build the proxy request for the client's leaf (verified RFC-3820 crypto). */
214 err[0] = '\0';
215 {
216 const brix_gsi_blob_t leaf_blob = { b.leaf_pem, leaf_len };
217 const brix_gsi_err_t err_sink = { err, sizeof(err) };
218 brix_gsi_buf_t req_out = { NULL, 0 };
219
220 if (brix_gsi_build_pxyreq(&leaf_blob, &b.reqkey, &req_out,
221 &err_sink) != 0) {
222 ngx_log_error(NGX_LOG_WARN, c->log, 0,
223 "brix: GSI delegation: build pxyreq failed: %s", err);
224 return bdg_fail(&b);
225 }
226 b.req_der = req_out.data;
227 b.req_len = req_out.len;
228 }
229
230 /* The stock client parses kXRS_x509_req as PEM — re-encode the DER request. */
231 b.req_pem = gsi_req_der_to_pem(b.req_der, b.req_len, &b.req_pem_len);
232 if (b.req_pem == NULL) {
233 ngx_log_error(NGX_LOG_WARN, c->log, 0,
234 "brix: GSI delegation: cannot PEM-encode proxy request");
235 return bdg_fail(&b);
236 }
237
238 /*
239 * Rtag proof-chain: RSA-sign (EncryptPrivate, our cert key) the client's
240 * kXGC_cert random tag → kXRS_signed_rtag; the client's CheckRtag verifies it
241 * with our public key and rejects the round otherwise ("random tag missing").
242 * Add a fresh kXRS_rtag challenge for the client to sign in kXGC_sigpxy.
243 */
244 if (ctx->gsi.deleg_client_rtag_len > 0 && conf->gsi_key != NULL) {
245 signed_rtag_len = brix_gsi_rsa_encrypt_private(
246 conf->gsi_key, ctx->gsi.deleg_client_rtag,
247 (size_t) ctx->gsi.deleg_client_rtag_len,
248 signed_rtag, sizeof(signed_rtag));
249 }
250 if (signed_rtag_len == 0 || !brix_gsi_rand(new_rtag, sizeof(new_rtag))) {
251 ngx_log_error(NGX_LOG_WARN, c->log, 0,
252 "brix: GSI delegation: cannot sign rtag (proof-of-possession)");
253 return bdg_fail(&b);
254 }
255
256 /* Nested main = {kXGS_pxyreq + signed_rtag + rtag + kXRS_x509_req + none}. */
257 brix_gbuf_start(&b.inner, (uint32_t) kXGS_pxyreq);
258 brix_gbuf_bucket(&b.inner, (uint32_t) kXRS_signed_rtag,
259 signed_rtag, signed_rtag_len);
260 brix_gbuf_bucket(&b.inner, (uint32_t) kXRS_rtag, new_rtag, sizeof(new_rtag));
261 brix_gbuf_bucket(&b.inner, (uint32_t) kXRS_x509_req, b.req_pem, b.req_pem_len);
262 brix_gbuf_end(&b.inner);
263 if (b.inner.err) {
264 return bdg_fail(&b);
265 }
266 b.enc = brix_gsi_cipher_encrypt(&cipher, ctx->gsi.sess_key, b.inner.p,
267 b.inner.len, ctx->gsi.sess_use_iv,
268 &b.enc_len);
269 if (b.enc == NULL) {
270 ngx_log_error(NGX_LOG_WARN, c->log, 0,
271 "brix: GSI delegation: encrypt pxyreq main failed");
272 return bdg_fail(&b);
273 }
274
275 /* Outer kXGS_pxyreq = {kXRS_main(enc) + kXRS_cipher_alg + none}. */
276 calg_len = ngx_strlen(ctx->gsi.sess_cipher);
277 brix_gbuf_start(&b.outer, (uint32_t) kXGS_pxyreq);
278 brix_gbuf_bucket(&b.outer, (uint32_t) kXRS_main, b.enc, b.enc_len);
279 brix_gbuf_bucket(&b.outer, (uint32_t) kXRS_cipher_alg,
280 ctx->gsi.sess_cipher, calg_len);
281 brix_gbuf_end(&b.outer);
282 if (b.outer.err) {
283 return bdg_fail(&b);
284 }
285
286 /* Frame as kXR_authmore + the gsi payload, into a pool buffer for queueing. */
287 body_len = b.outer.len;
288 total = XRD_RESPONSE_HDR_LEN + body_len;
289 buf = ngx_palloc(c->pool, total);
290 if (buf == NULL) {
291 return bdg_fail(&b);
292 }
293 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_authmore,
294 (uint32_t) body_len, (ServerResponseHdr *) buf);
295 p = buf + XRD_RESPONSE_HDR_LEN;
296 ngx_memcpy(p, b.outer.p, body_len);
297
298 /* Hand the request key + client chain to the connection for kXGC_sigpxy. */
299 ctx->gsi.deleg_reqkey = b.reqkey; b.reqkey = NULL;
300 ctx->gsi.deleg_chain_pem = b.chain_pem; b.chain_pem = NULL;
301 ctx->gsi.deleg_chain_len = b.chain_len;
302 ctx->gsi.deleg_await = 1;
303
304 (void) bdg_fail(&b); /* free the rest (req_der, enc, gbufs, leaf_pem) */
305
306 ngx_log_error(NGX_LOG_INFO, c->log, 0,
307 "brix: GSI delegation: sent kXGS_pxyreq (awaiting signed proxy)");
308 return brix_queue_response(ctx, c, buf, total);
309}

311ngx_int_t
312brix_gsi_handle_sigpxy(brix_ctx_t *ctx, ngx_connection_t *c)
313{
314 const u_char *payload = ctx->recv.payload;
315 size_t plen = ctx->recv.cur_dlen;
316 brix_gsi_cipher_t cipher;
317 const uint8_t *enc = NULL, *signed_pem = NULL;
318 size_t enc_len = 0, signed_len = 0;
319 uint8_t *plain = NULL;
320 size_t plain_len = 0;
321 char err[160];
322 ngx_int_t rc = NGX_ERROR;
323
324 if (!ctx->gsi.deleg_await || ctx->gsi.deleg_reqkey == NULL
325 || ctx->gsi.deleg_chain_pem == NULL
326 || gsi_session_cipher(ctx, &cipher) != 0) {
327 ngx_log_error(NGX_LOG_WARN, c->log, 0,
328 "brix: GSI kXGC_sigpxy: not awaiting delegation");
329 return NGX_ERROR;
330 }
331
332 if (brix_gsi_find_bucket(payload, plen, (uint32_t) kXRS_main,
333 &enc, &enc_len) != 0) {
334 ngx_log_error(NGX_LOG_WARN, c->log, 0,
335 "brix: GSI kXGC_sigpxy: kXRS_main missing");
336 return NGX_ERROR;
337 }
338 plain = brix_gsi_cipher_decrypt(&cipher, ctx->gsi.sess_key, enc, enc_len,
339 ctx->gsi.sess_use_iv, &plain_len);
340 if (plain == NULL) {
341 ngx_log_error(NGX_LOG_WARN, c->log, 0,
342 "brix: GSI kXGC_sigpxy: main decrypt failed");
343 return NGX_ERROR;
344 }
345
346 if (brix_gsi_find_bucket(plain, plain_len, (uint32_t) kXRS_x509,
347 &signed_pem, &signed_len) != 0) {
348 ngx_log_error(NGX_LOG_WARN, c->log, 0,
349 "brix: GSI kXGC_sigpxy: signed proxy (kXRS_x509) missing "
350 "(client declined to delegate?)");
351 free(plain);
352 return NGX_ERROR;
353 }
354
355 /*
356 * Assemble the full pull credential = proxy cert + issuer chain + its
357 * private key (our request key), PEM, so the TPC pull can authenticate to
358 * the source AS THE USER (loaded in src/tpc/gsi/gsi_outbound_certreq.c).
359 * assemble_proxy verifies the signed proxy's key matches the request key
360 * and serializes the key into the blob itself.
361 */
362 err[0] = '\0';
363 {
364 const brix_gsi_blob_t signed_blob = { signed_pem, signed_len };
365 const brix_gsi_blob_t chain_blob = { ctx->gsi.deleg_chain_pem,
366 ctx->gsi.deleg_chain_len };
367 const brix_gsi_err_t err_sink = { err, sizeof(err) };
368 brix_gsi_buf_t cred_out = { NULL, 0 };
369
370 if (brix_gsi_assemble_proxy(&signed_blob, ctx->gsi.deleg_reqkey,
371 &chain_blob, &cred_out,
372 &err_sink) != 0) {
373 ngx_log_error(NGX_LOG_WARN, c->log, 0,
374 "brix: GSI kXGC_sigpxy: assemble proxy failed: %s", err);
375 free(plain);
376 return NGX_ERROR;
377 }
378 free(ctx->gsi.deleg_proxy_pem);
379 ctx->gsi.deleg_proxy_pem = cred_out.data;
380 ctx->gsi.deleg_proxy_len = cred_out.len;
381 }
382 free(plain);
383
384 ngx_log_error(NGX_LOG_INFO, c->log, 0,
385 "brix: GSI delegation: captured delegated proxy (%uz bytes) "
386 "dn=\"%s\"", ctx->gsi.deleg_proxy_len, ctx->login.dn);
387
388 /* The request key is consumed; the captured proxy credential remains for the
389 * TPC pull. Clear the await flag + cleanse the session key. */
390 EVP_PKEY_free(ctx->gsi.deleg_reqkey);
391 ctx->gsi.deleg_reqkey = NULL;
392 ctx->gsi.deleg_await = 0;
393 OPENSSL_cleanse(ctx->gsi.sess_key, sizeof(ctx->gsi.sess_key));
394 ctx->gsi.sess_keylen = 0;
395 rc = NGX_OK;
396 return rc;
397}
