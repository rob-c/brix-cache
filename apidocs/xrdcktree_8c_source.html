1/* xrdcktree.c — recursive checksum manifests and manifest verification.
2 *
3 * WHAT: `xrdcksum tree <root> [--algo NAME] [-o FILE]` produces a sha256sum-style
4 * manifest of all regular files under a local directory or a root:// tree.
5 * `xrdcksum check <manifest> <root>` verifies every recorded digest against
6 * the same-side data — local files with brix_cksum_fd, remote files with one
7 * reused connection and brix_query_cksum.
8 * WHY: End-to-end audit: a site can generate a manifest during ingest and replay
9 * check periodically to catch at-rest corruption or unauthorised modification,
10 * both locally and over root://.
11 * HOW: Local tree: POSIX opendir/readdir recursion (lstat; skip symlinks + dots;
12 * overflow-checked path joins; mirrors copy_tree_upload discipline). Remote
13 * tree: brix_tree_walk visitor (dirs skipped; path stripped to rel by removing
14 * the root's path prefix + one slash). Manifest format "<hex> <rel>\n"
15 * (two spaces) is sha256sum -c compatible; brix_ckmf_parse_line is the sole
16 * gate on the check path so hostile manifests cannot escape the root. */

52/* path_join — safely concatenate a directory and filename.
53 *
54 * WHAT: Join dir + "/" + name into out[outsz]; return -1 on overflow, 0 on
55 * success. Avoids double-slash if dir already ends with /.
56 * WHY: Walking trees requires joining paths safely without stack overflow;
57 * mirrors copy_tree_upload discipline to stay overflow-consistent.
58 * HOW: Check dir's trailing slash; call snprintf with the appropriate format;
59 * validate output against outsz. */
60static int
61path_join(const char *dir, const char *name, char *out, size_t outsz)
62{
63 size_t dl = strlen(dir);
64 int n;
65
66 if (dl > 0 && dir[dl - 1] == '/') {
67 n = snprintf(out, outsz, "%s%s", dir, name);
68 } else {
69 n = snprintf(out, outsz, "%s/%s", dir, name);
70 }
71 return (n > 0 && (size_t) n < outsz) ? 0 : -1;
72}

74/* rel_join — safely concatenate a relative path prefix and name.
75 *
76 * WHAT: Join prefix + "/" + name into out[outsz]; prefix may be empty string.
77 * Return -1 on overflow, 0 on success.
78 * WHY: Building relative paths during tree walks (from tree root); handles
79 * the empty-prefix edge case (top-level files have rel="filename").
80 * HOW: Omit "/" separator when prefix is empty; otherwise snprintf with both;
81 * validate output size. */
82static int
83rel_join(const char *prefix, const char *name, char *out, size_t outsz)
84{
85 int n;
86
87 if (prefix[0] == '\0') {
88 n = snprintf(out, outsz, "%s", name);
89 } else {
90 n = snprintf(out, outsz, "%s/%s", prefix, name);
91 }
92 return (n > 0 && (size_t) n < outsz) ? 0 : -1;
93}

95/* rel_has_newline — detect a manifest-forgeable name.
96 *
97 * WHAT: Return 1 if `rel` contains a '\n' or '\r', 0 otherwise.
98 * WHY: The manifest format is one "<hex> <rel>\n" record per line, parsed
99 * line-by-line on the check side. A file whose name embeds a newline (or
100 * CR) would forge an extra record — smuggling a second "<hex> <path>"
101 * into the manifest and corrupting the whole verification. Such names are
102 * skipped-and-warned (counted as an error) rather than written.
103 * HOW: Linear scan for the two line-terminator bytes. */
104static int
105rel_has_newline(const char *rel)
106{
107 return strchr(rel, '\n') != NULL || strchr(rel, '\r') != NULL;
108}

112/* cktree_file_fn — per-regular-file callback for cktree_walk.
113 *
114 * WHAT: Invoked for every regular file found during the walk with the full
115 * local path and the root-relative path.
116 * WHY: Separates the traversal mechanics (one shared walker) from what is
117 * done per file; per-file failures are the callback's business (count
118 * and return 0 to keep walking).
119 * HOW: Return 0 to continue the walk; nonzero aborts it structurally. */
120typedef int (*cktree_file_fn)(const char *lpath, const char *rel, void *u);

122/* entry_is_dot — recognise the "." and ".." directory entries.
123 *
124 * WHAT: Return 1 if `name` is exactly "." or "..", 0 otherwise.
125 * WHY: Every readdir loop must skip these two to avoid infinite recursion;
126 * naming the test keeps the walker's loop readable.
127 * HOW: Byte comparison without strcmp calls (hot loop). */
128static int
129entry_is_dot(const char *name)
130{
131 return name[0] == '.'
132 && (name[1] == '\0' || (name[1] == '.' && name[2] == '\0'));
133}

135/* cktree_walk — recursive POSIX opendir/readdir walk with per-file callback.
136 *
137 * WHAT: Visit every regular file under `lpath`, invoking cb(full, rel, u)
138 * for each one.
139 * WHY: Mirrors copy_tree_upload discipline: lstat (not stat), skip symlinks,
140 * skip dot-entries, overflow-check every path join before use. Shared
141 * walker so callers only supply per-file behaviour.
142 * HOW: opendir → readdir loop → lstat → classify → recurse or callback.
143 * Returns 0 to continue, -1 on an unrecoverable structural error (path
144 * overflow or opendir failure on the root). Structural errors are
145 * counted in *errors; per-file errors are the callback's business. */
146static int
147cktree_walk(const char *lpath, const char *rel,
148 cktree_file_fn cb, void *u, int *errors)
149{
150 DIR *d;
151 struct dirent *de;
152
153 d = opendir(lpath);
154 if (d == NULL) {
155 fprintf(stderr, "xrdcksum tree: opendir %s: %s\n",
156 lpath, strerror(errno));
157 (*errors)++;
158 /* Root opendir failure is fatal to the subtree but callers continue. */
159 return -1;
160 }
161 while ((de = readdir(d)) != NULL) {
162 char relc[XRDC_PATH_MAX];
163 char lc[XRDC_PATH_MAX];
164 struct stat sb;
165
166 if (entry_is_dot(de->d_name)) {
167 continue;
168 }
169 if (path_join(lpath, de->d_name, lc, sizeof(lc)) != 0
170 || rel_join(rel, de->d_name, relc, sizeof(relc)) != 0) {
171 fprintf(stderr,
172 "xrdcksum tree: path too long under %s\n", lpath);
173 (*errors)++; /* a skipped file is an error, not clean output */
174 closedir(d);
175 return -1;
176 }
177 /* lstat so a symlink inside the tree is never followed outward. */
178 if (lstat(lc, &sb) != 0) {
179 continue; /* vanished between readdir and lstat — skip silently */
180 }
181 if (S_ISLNK(sb.st_mode)) {
182 continue; /* skip symlinks (loop-safety; mirrors -r copy default) */
183 }
184 if (S_ISDIR(sb.st_mode)) {
185 if (cktree_walk(lc, relc, cb, u, errors) != 0) {
186 closedir(d);
187 return -1;
188 }
189 } else if (S_ISREG(sb.st_mode)) {
190 if (cb(lc, relc, u) != 0) {
191 closedir(d);
192 return -1;
193 }
194 }
195 /* other file types (block/char/fifo/socket) are silently skipped */
196 }
197 closedir(d);
198 return 0;
199}

211/* tree_emit_file — cktree_walk callback: checksum one file, emit its record.
212 *
213 * WHAT: Compute the checksum of the regular file at `lpath` and write
214 * "<hex> <rel>\n" to ctx->out.
215 * WHY: The per-file half of `xrdcksum tree`'s local walk — the walker owns
216 * traversal, this owns digest + manifest emission.
217 * HOW: open O_RDONLY → brix_cksum_fd → refuse newline/CR-bearing names
218 * (manifest-forgery guard) → fprintf. Per-file errors go to stderr,
219 * bump ctx->errors, and return 0 so the walk continues. */
220static int
221tree_emit_file(const char *lpath, const char *rel, void *u)
222{
223 tree_ctx *ctx = (tree_ctx *) u;
224 char hex[TREE_HEX_MAX];
225 brix_status st;
226 int fd;
227
228 brix_status_clear(&st);
229 fd = open(lpath, O_RDONLY);
230 if (fd < 0) {
231 fprintf(stderr, "xrdcksum tree: open %s: %s\n",
232 lpath, strerror(errno));
233 (*ctx->errors)++;
234 return 0;
235 }
236 if (brix_cksum_fd(fd, ctx->algo, hex, sizeof(hex), &st) != 0) {
237 fprintf(stderr, "xrdcksum tree: %s: %s\n", lpath,
238 st.msg[0] ? st.msg : "checksum error");
239 close(fd);
240 (*ctx->errors)++;
241 return 0;
242 }
243 close(fd);
244 /* Refuse to emit a record for a newline/CR-bearing name: it would
245 * forge an extra manifest line. Skip-and-warn (counted). */
246 if (rel_has_newline(rel)) {
247 fprintf(stderr, "xrdcksum tree: %s: name contains newline/CR, "
248 "skipping\n", lpath);
249 (*ctx->errors)++;
250 return 0;
251 }
252 fprintf(ctx->out, "%s %s\n", hex, rel);
253 return 0;
254}

268/* remote_tree_visitor — brix_walk_fn callback for remote tree walks.
269 *
270 * WHAT: Called by brix_tree_walk for every entry under the root path.
271 * WHY: Skip directories (no checksum); query the server for each file.
272 * HOW: Strip the root path prefix (+ one slash) from the full path to get rel;
273 * call brix_query_cksum; emit "<hex> <rel>\n". Per-file errors: stderr,
274 * errors counter, continue (return 0 so the walk proceeds). */
275static int
276remote_tree_visitor(const char *path, const brix_dirent *e, int depth,
277 void *u)
278{
279 remote_tree_ctx *ctx = (remote_tree_ctx *) u;
280 const char *rel;
281 char hex[TREE_HEX_MAX];
282 brix_status st;
283
284 (void) depth;
285
286 /* Skip directories — the walker descends automatically. */
287 if (e->have_stat && (e->st.flags & kXR_isDir)) {
288 return 0;
289 }
290 /* Derive rel from the full path by stripping the root prefix + one slash.
291 * root_path "/foo/bar" → paths are "/foo/bar/sub/f.root" → rel = "sub/f.root".
292 * Guard: if the path does not start with root_path, skip (should not happen). */
293 if (strncmp(path, ctx->root_path, ctx->root_path_len) != 0
294 || path[ctx->root_path_len] != '/') {
295 fprintf(stderr,
296 "xrdcksum tree: unexpected path %s outside root %s\n",
297 path, ctx->root_path);
298 (*ctx->errors)++;
299 return 0;
300 }
301 rel = path + ctx->root_path_len + 1;
302
303 /* A newline/CR in the derived rel would forge an extra manifest line — skip
304 * it and count the error rather than corrupt the manifest. */
305 if (rel_has_newline(rel)) {
306 fprintf(stderr, "xrdcksum tree: %s: name contains newline/CR, skipping\n",
307 path);
308 (*ctx->errors)++;
309 return 0;
310 }
311
312 brix_status_clear(&st);
313 if (brix_query_cksum(ctx->c, path, ctx->algo_name,
314 hex, sizeof(hex), &st) != 0) {
315 fprintf(stderr, "xrdcksum tree: %s: %s\n", path,
316 st.msg[0] ? st.msg : "query_cksum error");
317 (*ctx->errors)++;
318 return 0;
319 }
320 fprintf(ctx->out, "%s %s\n", hex, rel);
321 return 0;
322}

348/* cktree_parse_args — decode the `xrdcksum tree` command line.
349 *
350 * WHAT: Fill *o from argv; on any usage error (or -h) set *rc to the exit
351 * code and return -1; return 0 to proceed.
352 * WHY: Keeps the option ladder out of the main flow so the main reads as
353 * parse → plan → run → report.
354 * HOW: Linear argv scan; --algo/-o consume the next argument; first
355 * non-option is the root; anything else is a usage error. */
356static int
357cktree_parse_args(int argc, char **argv, cktree_opts *o, int *rc)
358{
359 int i;
360
361 o->root = NULL;
362 o->algo_str = "adler32";
363 o->outpath = NULL;
364
365 for (i = 1; i < argc; i++) {
366 const char *a = argv[i];
367
368 if (strcmp(a, "--algo") == 0) {
369 if (++i >= argc) {
370 *rc = usage_tree(argv[0], XRDC_EXIT_USAGE);
371 return -1;
372 }
373 o->algo_str = argv[i];
374 } else if (strcmp(a, "-o") == 0) {
375 if (++i >= argc) {
376 *rc = usage_tree(argv[0], XRDC_EXIT_USAGE);
377 return -1;
378 }
379 o->outpath = argv[i];
380 } else if (strcmp(a, "-h") == 0 || strcmp(a, "--help") == 0) {
381 *rc = usage_tree(argv[0], 0);
382 return -1;
383 } else if (a[0] == '-') {
384 fprintf(stderr, "%s: unknown option '%s'\n", argv[0], a);
385 *rc = usage_tree(argv[0], XRDC_EXIT_USAGE);
386 return -1;
387 } else if (o->root == NULL) {
388 o->root = a;
389 } else {
390 fprintf(stderr, "%s: unexpected argument '%s'\n", argv[0], a);
391 *rc = usage_tree(argv[0], XRDC_EXIT_USAGE);
392 return -1;
393 }
394 }
395 if (o->root == NULL) {
396 *rc = usage_tree(argv[0], XRDC_EXIT_USAGE);
397 return -1;
398 }
399 return 0;
400}

402/* cktree_run_local — generate the manifest for a local directory tree.
403 *
404 * WHAT: Walk o->root with cktree_walk/tree_emit_file, writing records to out.
405 * WHY: The local half of `xrdcksum tree`, isolated so the main only routes.
406 * HOW: Build the tree_ctx and run the walk; a structural failure (root
407 * opendir / path overflow) must land in the exit code — a partial
408 * manifest is not clean — so ensure at least one counted error. */
409static void
410cktree_run_local(const cktree_opts *o, brix_cksum_algo algo,
411 FILE *out, int *errors)
412{
413 tree_ctx ctx;
414
415 ctx.out = out;
416 ctx.algo_name = o->algo_str;
417 ctx.algo = algo;
418 ctx.errors = errors;
419 if (cktree_walk(o->root, "", tree_emit_file, &ctx, errors) != 0
420 && *errors == 0) {
421 (*errors)++;
422 }
423}

425/* cktree_run_remote — generate the manifest for a root:// tree.
426 *
427 * WHAT: Parse the URL, connect once, and drive brix_tree_walk with
428 * remote_tree_visitor. Returns 0 when the run completed (per-file and
429 * walk errors are counted in *errors), or a nonzero exit code when the
430 * URL/connect failed before any work started.
431 * WHY: The remote half of `xrdcksum tree`; keeps connection lifecycle and
432 * prefix computation out of the main.
433 * HOW: brix_url_parse + scheme check → brix_connect → strip trailing slashes
434 * from the root path so `tree root://h//dir/` yields the same rel
435 * prefixes as `.../dir` (a bare "/" export root collapses to a
436 * zero-length prefix, keeping that case working) → walk → close. */
437static int
438cktree_run_remote(const char *prog, const cktree_opts *o,
439 FILE *out, int *errors)
440{
441 brix_url u;
442 brix_conn c;
443 remote_tree_ctx ctx;
444 brix_status st;
445 size_t rlen;
446 int wrc;
447
448 brix_status_clear(&st);
449 if (brix_url_parse(o->root, &u, &st) != 0
450 || (u.scheme != XRDC_SCHEME_ROOT
451 && u.scheme != XRDC_SCHEME_ROOTS)) {
452 fprintf(stderr, "%s: %s\n", prog,
453 st.msg[0] ? st.msg : "bad URL");
454 return XRDC_EXIT_USAGE;
455 }
456 if (brix_connect(&c, &u, NULL, &st) != 0) {
457 fprintf(stderr, "%s: connect: %s\n", prog, st.msg);
458 return XRDC_EXIT_USAGE;
459 }
460 ctx.c = &c;
461 ctx.out = out;
462 ctx.algo_name = o->algo_str;
463 ctx.root_path = u.path;
464 rlen = strlen(u.path);
465 while (rlen > 0 && u.path[rlen - 1] == '/') {
466 rlen--;
467 }
468 ctx.root_path_len = rlen;
469 ctx.errors = errors;
470
471 wrc = brix_tree_walk(&c, u.path, remote_tree_visitor, &ctx, &st);
472 if (wrc < 0) {
473 fprintf(stderr, "%s: walk error: %s\n", prog,
474 st.msg[0] ? st.msg : "tree walk failed");
475 (*errors)++;
476 }
477 brix_close(&c);
478 return 0;
479}

481/*
482 * brix_xrdcktree_main — xrdcksum tree: recursive checksum manifest generator.
483 *
484 * WHAT: Walk a local directory tree or a root:// tree, compute each regular
485 * file's checksum, and emit one "<hex> <rel>\n" line per file.
486 * WHY: Produces audit manifests compatible with sha256sum -c for ingestion
487 * verification and at-rest integrity checks.
488 * HOW: Parse args (cktree_parse_args); detect local vs remote by "://";
489 * local → cktree_run_local; remote → cktree_run_remote. -o opens the
490 * output file; stdout is the default. Exit 0 clean, 2 if any errors
491 * occurred.
492 */
493int
494brix_xrdcktree_main(int argc, char **argv)
495{
496 cktree_opts o;
497 brix_cksum_algo algo;
498 FILE *out = stdout;
499 int errors = 0;
500 int rc = 0;
501
502 if (cktree_parse_args(argc, argv, &o, &rc) != 0) {
503 return rc;
504 }
505 if (brix_cksum_algo_parse(o.algo_str, &algo) != 0) {
506 fprintf(stderr, "%s: unknown algorithm '%s'\n", argv[0], o.algo_str);
507 return XRDC_EXIT_USAGE;
508 }
509 if (o.outpath != NULL) {
510 out = fopen(o.outpath, "w");
511 if (out == NULL) {
512 fprintf(stderr, "%s: cannot open '%s': %s\n",
513 argv[0], o.outpath, strerror(errno));
514 return XRDC_EXIT_USAGE;
515 }
516 }
517
518 if (!is_root_url(o.root)) {
519 cktree_run_local(&o, algo, out, &errors);
520 } else {
521 rc = cktree_run_remote(argv[0], &o, out, &errors);
522 if (rc != 0) {
523 if (out != stdout) {
524 fclose(out);
525 }
526 return rc;
527 }
528 }
529
530 if (out != stdout) {
531 fclose(out);
532 }
533 return errors > 0 ? 2 : 0;
534}

570/* ckcheck_parse_args — decode the `xrdcksum check` command line.
571 *
572 * WHAT: Validate argc and parse the optional --algo flag; on any usage error
573 * (or -h) set *rc to the exit code and return -1; return 0 to proceed
574 * (manifest = argv[1], root = argv[2]).
575 * WHY: Keeps the option ladder out of the main flow.
576 * HOW: argv[1]/argv[2] are positional; scan from argv[3]; --algo consumes
577 * and validates the next argument via brix_cksum_algo_parse. */
578static int
579ckcheck_parse_args(int argc, char **argv, const char **algo_str,
580 brix_cksum_algo *algo, int *rc)
581{
582 int i;
583
584 *algo_str = NULL;
585 if (argc < 3) {
586 *rc = usage_check(argv[0], XRDC_EXIT_USAGE);
587 return -1;
588 }
589 for (i = 3; i < argc; i++) {
590 if (strcmp(argv[i], "--algo") == 0) {
591 if (++i >= argc) {
592 *rc = usage_check(argv[0], XRDC_EXIT_USAGE);
593 return -1;
594 }
595 *algo_str = argv[i];
596 if (brix_cksum_algo_parse(*algo_str, algo) != 0) {
597 fprintf(stderr, "%s: unknown algorithm '%s'\n",
598 argv[0], *algo_str);
599 *rc = XRDC_EXIT_USAGE;
600 return -1;
601 }
602 } else if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--help") == 0) {
603 *rc = usage_check(argv[0], 0);
604 return -1;

659/* ckcheck_line_algo — resolve the digest algorithm for one manifest line.
660 *
661 * WHAT: Set *out_algo/*out_name for the line: --algo when given, otherwise
662 * inferred from the hex digest length. Return 0 on success, -1 after
663 * printing the unrecognised-length error (caller counts it).
664 * WHY: Manifests do not record the algorithm; the historical inference is
665 * part of the CLI contract and must stay EXACT: adler32 = 8 hex chars,
666 * crc64 (CRC-64/XZ) = 16, md5 = 32. crc32c and zcrc32 also produce
667 * 8 hex chars and are indistinguishable by length — they require
668 * --algo. CRITICAL (INVARIANT 9): crc64 and crc64nvme are DIFFERENT
669 * polynomials; 16 hex maps to crc64 ONLY, and crc64nvme manifests must
670 * use --algo. Never merge or "normalize" the two.
671 * HOW: --algo short-circuit, else switch on strlen(hex). */
672static int
673ckcheck_line_algo(const ckcheck_ctx *cx, const char *hex, const char *rel,
674 brix_cksum_algo *out_algo, const char **out_name)
675{
676 size_t hexlen;
677
678 if (cx->algo_str != NULL) {
679 *out_algo = cx->algo;
680 *out_name = cx->algo_str;
681 return 0;
682 }
683 hexlen = strlen(hex);
684 switch (hexlen) {
685 case 8: *out_algo = XRDC_CK_ADLER32; *out_name = "adler32"; return 0;
686 case 16: *out_algo = XRDC_CK_CRC64; *out_name = "crc64"; return 0;
687 case 32: *out_algo = XRDC_CK_MD5; *out_name = "md5"; return 0;
688 default:
689 fprintf(stderr, "%s: unrecognised hex length %zu for '%s'\n",
690 cx->prog, hexlen, rel);
691 return -1;
692 }
693}

695/* ckcheck_digest_local — compute the actual digest of a local file.
696 *
697 * WHAT: Open cx->root/rel and compute its checksum into got (TREE_HEX_MAX
698 * bytes). Return 0 on success, -1 after printing the error.
699 * WHY: The local half of per-line verification; every failure is one
700 * stderr line and one counted error at the call site.
701 * HOW: Overflow-checked path_join → open O_RDONLY → brix_cksum_fd. */
702static int
703ckcheck_digest_local(const ckcheck_ctx *cx, const char *rel,
704 brix_cksum_algo algo, char *got)
705{
706 char lpath[XRDC_PATH_MAX];
707 brix_status st;
708 int fd;
709
710 if (path_join(cx->root, rel, lpath, sizeof(lpath)) != 0) {
711 fprintf(stderr, "%s: path too long: %s/%s\n",
712 cx->prog, cx->root, rel);
713 return -1;
714 }
715 fd = open(lpath, O_RDONLY);
716 if (fd < 0) {
717 fprintf(stderr, "%s: open %s: %s\n",
718 cx->prog, lpath, strerror(errno));
719 return -1;
720 }
721 brix_status_clear(&st);
722 if (brix_cksum_fd(fd, algo, got, TREE_HEX_MAX, &st) != 0) {
723 fprintf(stderr, "%s: %s: %s\n", cx->prog, lpath,
724 st.msg[0] ? st.msg : "checksum error");
725 close(fd);
726 return -1;
727 }
728 close(fd);
729 return 0;
730}

732/* ckcheck_digest_remote — query the server for a remote file's digest.
733 *
734 * WHAT: Build root_path + "/" + rel and ask the server (over the reused
735 * connection) for its digest into got (TREE_HEX_MAX bytes). Return 0
736 * on success, -1 after printing the error.
737 * WHY: The remote half of per-line verification; the server computes, we
738 * compare.
739 * HOW: Overflow-checked snprintf → brix_query_cksum with the per-line
740 * algorithm name. */
741static int
742ckcheck_digest_remote(const ckcheck_ctx *cx, const char *rel,
743 const char *algo_name, char *got)
744{
745 char rpath[XRDC_PATH_MAX];
746 brix_status st;
747 int n;
748
749 n = snprintf(rpath, sizeof(rpath), "%s/%s", cx->root_path, rel);
750 if (n <= 0 || (size_t) n >= sizeof(rpath)) {
751 fprintf(stderr, "%s: path too long: %s/%s\n",
752 cx->prog, cx->root_path, rel);
753 return -1;
754 }
755 brix_status_clear(&st);
756 if (brix_query_cksum(cx->c, rpath, algo_name,
757 got, TREE_HEX_MAX, &st) != 0) {
758 fprintf(stderr, "%s: query_cksum %s: %s\n", cx->prog, rpath,
759 st.msg[0] ? st.msg : "query error");
760 return -1;
761 }
762 return 0;
763}

765/* ckcheck_process_line — verify one manifest line and print its verdict.
766 *
767 * WHAT: Parse "<hex> <rel>" via brix_ckmf_parse_line, resolve the line's
768 * algorithm, obtain the actual digest (local compute or remote query),
769 * and print "OK <rel>" or "FAILED <rel>".
770 * WHY: One line = one independent verification; every failure mode counts
771 * an error and moves on so a single bad record never aborts the run.
772 * HOW: brix_ckmf_parse_line is the sole gate on the check path (hostile
773 * manifests cannot escape the root) → ckcheck_line_algo →
774 * ckcheck_digest_local/_remote → case-insensitive compare. */
775static void
776ckcheck_process_line(const ckcheck_ctx *cx, char *line)
777{
778 char hex[TREE_HEX_MAX];
779 char rel[XRDC_PATH_MAX];
780 char got[TREE_HEX_MAX];
781 brix_cksum_algo line_algo;
782 const char *line_algo_name;
783 int drc;
784
785 if (brix_ckmf_parse_line(line, hex, sizeof(hex),
786 rel, sizeof(rel)) != 0) {
787 ckcheck_report_malformed(cx->prog, line);
788 (*cx->errors)++;
789 return;
790 }
791 if (ckcheck_line_algo(cx, hex, rel, &line_algo, &line_algo_name) != 0) {
792 (*cx->errors)++;
793 return;
794 }
795
796 got[0] = '\0';
797 if (!cx->is_remote) {
798 drc = ckcheck_digest_local(cx, rel, line_algo, got);
799 } else {
800 drc = ckcheck_digest_remote(cx, rel, line_algo_name, got);
801 }
802 if (drc != 0) {
803 (*cx->errors)++;
804 return;
805 }
806
807 if (strcasecmp(hex, got) == 0) {
808 printf("OK %s\n", rel);
809 } else {
810 printf("FAILED %s\n", rel);
811 (*cx->mismatches)++;
812 }
813}

815/*
816 * brix_xrdckcheck_main — xrdcksum check: manifest verification.
817 *
818 * WHAT: Parse a manifest produced by `xrdcksum tree` and verify each recorded
819 * digest against the actual file content, local or remote.
820 * WHY: Periodic at-rest integrity checks without regenerating the full manifest;
821 * a single run covers the whole tree with one output line per file.
822 * HOW: Parse args (ckcheck_parse_args); open the manifest; for a remote root
823 * open one connection reused across all files (ckcheck_connect); verify
824 * every line with ckcheck_process_line (which owns algorithm inference —
825 * adler32=8 hex, crc64=16, md5=32 — and the OK/FAILED verdicts).
826 * Exit 0 all-OK, 1 any mismatch, 2 on any parse/I/O error.
827 */
828int
829brix_xrdckcheck_main(int argc, char **argv)
830{
831 const char *algo_str = NULL;
832 brix_cksum_algo algo = XRDC_CK_ADLER32; /* valid iff algo_str set */
833 FILE *mf;
834 char line[XRDC_PATH_MAX + TREE_HEX_MAX + 8];
835 int mismatches = 0;
836 int errors = 0;
837 int rc = 0;
838 brix_url u;
839 brix_conn c;
840 ckcheck_ctx cx;
841
842 if (ckcheck_parse_args(argc, argv, &algo_str, &algo, &rc) != 0) {
843 return rc;
844 }
845
846 mf = fopen(argv[1], "r");
847 if (mf == NULL) {
848 fprintf(stderr, "%s: cannot open manifest '%s': %s\n",
849 argv[0], argv[1], strerror(errno));
850 return 2;
851 }
852
853 cx.prog = argv[0];
854 cx.root = argv[2];
855 cx.root_path = NULL;
856 cx.c = NULL;
857 cx.is_remote = is_root_url(argv[2]);
858 cx.algo_str = algo_str;
859 cx.algo = algo;
860 cx.errors = &errors;
861 cx.mismatches = &mismatches;
862
863 if (cx.is_remote) {
864 if (ckcheck_connect(argv[0], argv[2], &u, &c) != 0) {
865 fclose(mf);
866 return XRDC_EXIT_USAGE;
867 }
868 cx.root_path = u.path;
869 cx.c = &c;
870 }
871
872 while (fgets(line, sizeof(line), mf) != NULL) {
873 ckcheck_process_line(&cx, line);
874 }
875
876 fclose(mf);
877 if (cx.c != NULL) {
878 brix_close(&c);
879 }
880
881 if (errors > 0) {
882 return 2;
883 }
884 return mismatches > 0 ? 1 : 0;
885}
