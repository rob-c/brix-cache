1/*
2 * tpc_cred.c — HTTP-TPC credential delegation implementation
3 *
4 * Acquires OAuth2/OIDC access tokens for third-party-copy pull transfers.
5 * Two delegation modes:
6 *
7 * oidc-agent — UNIX-socket JSON IPC to a local oidc-agent daemon
8 * token-exchange — RFC 8693 token-exchange request to an external OAuth2
9 * token endpoint
10 *
11 * Both paths are non-blocking from the nginx-worker perspective: external
12 * processes are fork/exec'd and their output is read synchronously in a
13 * small helper that exits once the token is returned.
14 */

66/* Path to the dedicated oidc-agent helper binary. */
67#define TPC_CRED_HELPER_PATH "/usr/local/sbin/nginx-xrootd-tpc-cred"

69/*
70 * WHAT: Resolve the oidc-token binary to an absolute path — first honours an
71 * explicit override env var, then probes standard install locations.
72 * WHY: The fallback exec path must not be subject to $PATH substitution in the
73 * daemon's environment (a compromised PATH entry could shadow the real binary).
74 * HOW: secure_getenv("BRIX_OIDC_TOKEN_BIN") if access(X_OK) passes; else
75 * walk a fixed candidate list; return NULL if none executable.
76 */
77static const char *
78resolve_oidc_token_binary(void)
79{
80 static const char *const candidates[] = {
81 "/usr/bin/oidc-token", "/usr/local/bin/oidc-token", NULL
82 };
83 const char *const *p;
84 const char *override = secure_getenv("BRIX_OIDC_TOKEN_BIN");
85
86 if (override != NULL && access(override, X_OK) == 0) {
87 return override;
88 }
89 for (p = candidates; *p != NULL; p++) {
90 if (access(*p, X_OK) == 0) {
91 return *p;
92 }
93 }
94 return NULL;
95}

98/**
99 * Send a JSON request to the oidc-agent UNIX socket and read the token.
100 *
101 * Spawns a tiny fork/exec pipeline:
102 * nginx-worker → fork → exec "oidc-token -c <client>" or socket helper
103 *
104 * If the OIDC_SOCK environment variable is set, we connect directly to it.
105 * Otherwise we fall back to the default oidc-agent socket location.
106 *
107 * The helper process is: src/webdav/tpc_cred_oidc_helper (built separately)
108 * which handles the socket IPC. If that helper binary is not found we
109 * attempt a simpler approach using the `oidc-token` CLI if available.
110 */
111static ngx_int_t
112tpc_cred_oidc_agent_fetch(ngx_http_request_t *r,
113 const char *source_url,
114 const char *scope,
115 ngx_str_t *token_out)
116{
117 ngx_pid_t pid;
118 int pipefd[2];
119 char buf[TPC_CRED_MAX_TOKEN_LEN + 64];
120 ssize_t nread;
121 const char *sock_env;
122 ngx_str_t sock_path;
123 ngx_str_t helper_path;
124 char *argv[4];
125 char helper_buf[512];
126
127 /* Determine the OIDC socket path. */
128 sock_env = getenv("OIDC_SOCK");
129 if (sock_env && *sock_env) {
130 sock_path.data = (u_char *) sock_env;
131 sock_path.len = strlen(sock_env);
132 } else {
133 /* Default oidc-agent socket location. */
134 sock_path.data = (u_char *) "/run/user/1000/oidc/oidc_agent.sock";
135 sock_path.len = strlen((char *) sock_path.data);
136 }
137
138 /* Try the dedicated helper binary first. */
139 helper_path.data = (u_char *) TPC_CRED_HELPER_PATH;
140 helper_path.len = strlen(TPC_CRED_HELPER_PATH);
141
142 if (access((char *) helper_path.data, X_OK) == 0) {
143 /* Helper exists — use it. */
144 } else {
145 /* Fall back to oidc-token CLI (must be in PATH). */
146 helper_path.data = (u_char *) "oidc-token";
147 helper_path.len = 10;
148 }
149
150 /* Block SIGCHLD before fork so nginx's handler can't reap our child
151 * before we do. We restore the mask after waitpid(). */
152 sigset_t _old_mask_oidc, _blk_mask_oidc;
153 sigemptyset(&_blk_mask_oidc);
154 sigaddset(&_blk_mask_oidc, SIGCHLD);
155 sigprocmask(SIG_BLOCK, &_blk_mask_oidc, &_old_mask_oidc);
156
157 /* Create a pipe for the child's stdout → parent's read end. */
158 if (pipe(pipefd) == -1) {
159 ngx_log_error(NGX_LOG_ERR, r->connection->log, errno,
160 "tpc_cred(oidc): pipe() failed");
161 sigprocmask(SIG_SETMASK, &_old_mask_oidc, NULL);
162 return NGX_ERROR;
163 }
164
165 pid = fork();
166 if (pid == -1) {
167 ngx_log_error(NGX_LOG_ERR, r->connection->log, errno,
168 "tpc_cred(oidc): fork() failed");
169 close(pipefd[0]);
170 close(pipefd[1]);
171 sigprocmask(SIG_SETMASK, &_old_mask_oidc, NULL);
172 return NGX_ERROR;
173 }
174
175 if (pid == 0) {
176 /* Child process. */
177 close(pipefd[0]);
178 dup2(pipefd[1], STDOUT_FILENO);
179 close(pipefd[1]);
180
181 /* Re-exec: either the helper binary or oidc-token CLI. */
182 if (ngx_strcmp((char *) helper_path.data, "oidc-token") == 0) {
183 /* Use oidc-token CLI: oidc-token -c <client> */
184 /* For simplicity, use the source URL host as the client name.
185 * oidc-agent clients are typically named by host. */
186 const char *host_start = strstr((char *) source_url, "://");
187 char cli_argv[3][64];
188
189 if (host_start) {
190 host_start += 3;
191 const char *host_end = strchr(host_start, '/');
192 size_t host_len;
193 if (host_end)
194 host_len = (size_t)(host_end - host_start);
195 else
196 host_len = strlen(host_start);
197
198 if (host_len >= sizeof(cli_argv[0]))
199 host_len = sizeof(cli_argv[0]) - 1;
200 ngx_memcpy(cli_argv[0], host_start, host_len);
201 cli_argv[0][host_len] = '\0';
202 } else {
203 ngx_memcpy(cli_argv[0], "default", sizeof("default"));
204 }
205
206 /*
207 * W3 — option-injection guard. cli_argv[0] is the client-derived
208 * host passed as the value of oidc-token's "-c" flag. oidc-token
209 * parses options with getopt, so a host like "-x" or "--config"
210 * would be misinterpreted as a flag. A legitimate hostname never
211 * begins with '-' (RFC 952/1123), so reject one that does and fall
212 * back to the safe literal "default".
213 */
214 if (cli_argv[0][0] == '-' || cli_argv[0][0] == '\0') {
215 ngx_memcpy(cli_argv[0], "default", sizeof("default"));
216 }
217
218 cli_argv[1][0] = '-';
219 cli_argv[1][1] = 'c';
220 cli_argv[1][2] = '\0';
221
222 /* Resolve to an absolute path — avoid $PATH substitution risk in
223 * the daemon's potentially untrusted environment. Build a minimal
224 * controlled envp: PATH + HOME + OIDC_SOCK + XDG_RUNTIME_DIR only. */
225 {
226 const char *fb_bin = resolve_oidc_token_binary();
227 if (fb_bin != NULL) {
228 const char *home_val = getenv("HOME");
229 const char *xdg_val = getenv("XDG_RUNTIME_DIR");
230 const char *oidc_val = sock_env; /* already resolved above */
231 char home_buf[280], oidc_buf[280], xdg_buf[280];
232 char *fb_argv[4] = {
233 (char *) fb_bin, cli_argv[1], cli_argv[0], NULL
234 };
235 char *fb_envp[5];
236 int ei = 0;
237
238 fb_envp[ei++] = "PATH=/usr/bin:/bin";
239 if (home_val != NULL) {
240 snprintf(home_buf, sizeof(home_buf), "HOME=%s", home_val);
241 fb_envp[ei++] = home_buf;
242 }
243 if (oidc_val != NULL && *oidc_val != '\0') {
244 snprintf(oidc_buf, sizeof(oidc_buf),
245 "OIDC_SOCK=%s", oidc_val);
246 fb_envp[ei++] = oidc_buf;
247 }
248 if (xdg_val != NULL) {
249 snprintf(xdg_buf, sizeof(xdg_buf),
250 "XDG_RUNTIME_DIR=%s", xdg_val);
251 fb_envp[ei++] = xdg_buf;
252 }
253 fb_envp[ei] = NULL;
254 execve(fb_bin, fb_argv, fb_envp);
255 }
256 }
257 _exit(127);
258 } else {
259 /* Exec the dedicated helper. */
260 /* Build a null-terminated argv array. */
261 ngx_snprintf((u_char *) helper_buf, sizeof(helper_buf),
262 "%s", (char *) sock_path.data);
263 argv[0] = (char *) helper_path.data;
264 argv[1] = helper_buf; /* socket path */
265 argv[2] = (char *) source_url; /* source URL (for issuer) */
266 argv[3] = NULL;
267
268 execve(argv[0], argv, NULL);
269 _exit(127);
270 }
271 }
272
273 /* Parent: read child's stdout. */
274 close(pipefd[1]);
275
276 nread = 0;
277 while (nread < (ssize_t) sizeof(buf) - 1) {
278 ssize_t nr = read(pipefd[0], buf + nread, sizeof(buf) - 1 - (size_t) nread);
279 if (nr <= 0)
280 break;
281 nread += nr;
282 }
283 buf[nread] = '\0';
284 close(pipefd[0]);
285
286 /* Wait for child. SIGCHLD is still blocked so nginx's signal handler
287 * cannot reap this PID out from under us. Restore mask afterwards. */
288 {
289 int wstatus;
290 waitpid(pid, &wstatus, 0);
291 sigprocmask(SIG_SETMASK, &_old_mask_oidc, NULL);
292 if (!WIFEXITED(wstatus) || WEXITSTATUS(wstatus) != 0) {
293 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
294 "tpc_cred(oidc): helper exited with status %d",
295 WIFEXITED(wstatus) ? WEXITSTATUS(wstatus) : -1);
296 return NGX_ERROR;
297 }
298 }
299
300 /* Strip trailing whitespace/newlines. */
301 while (nread > 0 &&
302 (buf[nread - 1] == '\n' || buf[nread - 1] == '\r' ||
303 buf[nread - 1] == ' '))
304 nread--;
305 buf[nread] = '\0';
306
307 /* Try to parse as JSON first (helper may return JSON). */
308 if (buf[0] == '{') {
309 return tpc_cred_parse_token_response(r, buf, token_out);
310 }
311
312 /* Fallback: treat as raw token string. */
313 token_out->len = (size_t) nread;
314 token_out->data = ngx_pnalloc(r->pool, nread + 1);
315 if (token_out->data == NULL) {
316 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
317 "tpc_cred(oidc): pnalloc failed for token");
318 return NGX_ERROR;
319 }
320 memcpy(token_out->data, buf, nread);
321 token_out->data[nread] = '\0';
322
323 return NGX_OK;
324}

327/**
328 * Perform an RFC 8693 token-exchange request.
329 *
330 * Sends a POST to the configured token endpoint with:
331 * grant_type=urn:ietf:params:oauth:grant-type:token-exchange
332 * subject_token=<JWT>
333 * resource=<source_url>
334 * audience=<source_url>
335 * scope=<scope>
336 *
337 * Uses a fork/exec'd curl subprocess (same pattern as tpc_curl.c).
338 */
339static ngx_int_t
340tpc_cred_rfc8693_exchange(ngx_http_request_t *r,
341 const char *subject_token,
342 const char *source_url,
343 const char *scope,
344 const char *token_endpoint,
345 const char *client_id,
346 const char *client_secret,
347 ngx_str_t *token_out)
348{
349 char buf[TPC_CRED_MAX_TOKEN_LEN + 256];
350 char *curl_argv[16];
351 int argc = 0;
352 char body_buf[2048];
353 char body_file[NGX_MAX_PATH];
354 int body_fd;
355
356 /* Build the POST body. */
357 ngx_snprintf((u_char *) body_buf, sizeof(body_buf),
358 "grant_type=urn:ietf:params:oauth:grant-type:"
359 "token-exchange"
360 "&subject_token=%s"
361 "&resource=%s"
362 "&audience=%s"
363 "&scope=%s",
364 subject_token, source_url, source_url, scope);
365
366 /* Write body to a temp file (curl --data @file). */
367 ngx_snprintf((u_char *) body_file, sizeof(body_file),
368 "/tmp/tpc_cred_body_XXXXXX");
369 body_fd = mkstemp(body_file);
370 if (body_fd == -1) {
371 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
372 "tpc_cred(rfc8693): temp file creation failed");
373 return NGX_ERROR;
374 }
375
376 {
377 ssize_t nw = write(body_fd, body_buf, strlen(body_buf));
378 (void) nw;
379 }
380 close(body_fd);
381
382 /* Build curl argv. */
383 curl_argv[argc++] = (char *) "curl";
384 curl_argv[argc++] = (char *) "-s"; /* silent */
385 curl_argv[argc++] = (char *) "-S"; /* show errors */
386 curl_argv[argc++] = (char *) "-f"; /* fail on HTTP error */
387 curl_argv[argc++] = (char *) "-X";
388 curl_argv[argc++] = (char *) "POST";
389 curl_argv[argc++] = (char *) "-H";
390 curl_argv[argc++] = (char *) "Content-Type: application/x-www-form-urlencoded";
391 if (client_id && *client_id) {
392 /* Basic auth with client credentials. */
393 curl_argv[argc++] = (char *) "-u";
394 /* We'd need to base64 encode client_id:client_secret here.
395 * For simplicity, pass them separately to curl. */
396 curl_argv[argc++] = (char *) client_id;
397 curl_argv[argc++] = (char *) client_secret;
398 }
399 curl_argv[argc++] = (char *) "-d";
400 curl_argv[argc++] = (char *) body_file;
401 /* W3 — end-of-options terminator so the endpoint URL can never be parsed
402 * as a curl option, even if a misconfigured token_endpoint begins with '-'. */
403 curl_argv[argc++] = (char *) "--";
404 curl_argv[argc++] = (char *) token_endpoint;
405 curl_argv[argc++] = NULL;
406
407 /*
408 * Run curl synchronously and capture its stdout via the shared SIGCHLD-safe
409 * fork/exec helper (src/compat/subprocess.c) — it blocks SIGCHLD across the
410 * fork/waitpid internally so nginx's handler can't reap the child first. A
411 * non-zero rc = pipe/fork failure or signal-kill; a non-zero child exit
412 * (incl. curl -f on HTTP >= 400) is a credential-fetch failure.
413 */
414 {
415 int ec = -1;
416
417 if (brix_subprocess_capture(curl_argv, buf, sizeof(buf), NULL, &ec)
418 != 0)
419 {
420 ngx_log_error(NGX_LOG_ERR, r->connection->log, errno,
421 "tpc_cred(rfc8693): curl subprocess failed "
422 "(pipe/fork or signal)");
423 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
424 return NGX_ERROR;
425 }
426 if (ec != 0) {
427 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
428 "tpc_cred(rfc8693): curl exited %d: %s", ec, buf);
429 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
430 return NGX_ERROR;
431 }
432 }
433
434 /* Clean up temp file. */
435 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
436
437 /* Parse JSON response for access_token. */
438 return tpc_cred_parse_token_response(r, buf, token_out);
439}

516 (const char *) wconf->tpc_cred.token_endpoint.data,
517 wconf->tpc_cred.token_client_id.len > 0 ?

519 wconf->tpc_cred.token_client_secret.len > 0 ?
520 (const char *) wconf->tpc_cred.token_client_secret.data : NULL,
521 token_out);
522 if (rc == NGX_OK) {
523 rc = webdav_tpc_cred_validate_token(r, token_out);
524 }
525 if (rc == NGX_OK) {
526 webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NSUCCESS);
527 } else {
528 webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NERROR);
529 }
530 return rc;
531
532 default:
533 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
534 "tpc_cred: unknown delegation mode %d", (int) mode);
535 webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NUNKNOWN_MODE);
536 return NGX_ERROR;
537 }
