1/*
2 * tpc_cred.c — HTTP-TPC credential delegation implementation
3 *
4 * Acquires OAuth2/OIDC access tokens for third-party-copy pull transfers.
5 * Two delegation modes:
6 *
7 * oidc-agent — UNIX-socket JSON IPC to a local oidc-agent daemon
8 * token-exchange — RFC 8693 token-exchange request to an external OAuth2
9 * token endpoint
10 *
11 * Both paths are non-blocking from the nginx-worker perspective: external
12 * processes are fork/exec'd and their output is read synchronously in a
13 * small helper that exits once the token is returned.
14 *
15 * Per-mode fetch implementations live in siblings tpc_cred_oidc.c (oidc-agent)
16 * and tpc_cred_exchange.c (RFC 8693); this file keeps the credential parse/
17 * validate, mode dispatch, and metric machinery.
18 */

52webdav_tpc_cred_validate_token(ngx_http_request_t *r, ngx_str_t *token)
53{
54 brix_tpc_credential_t cred;
55
56 if (token == NULL || token->data == NULL || token->len == 0) {
57 return NGX_ERROR;
58 }
59
60 if (brix_tpc_credential_parse(token, BRIX_TPC_CREDENTIAL_TOKEN,
61 &cred, r->pool, r->connection->log)
62 != NGX_OK)
63 {
64 return NGX_ERROR;
65 }
66
67 return brix_tpc_credential_validate(&cred, r->connection->log);
68}

71brix_tpc_cred_mode_e
72webdav_tpc_cred_parse_mode(const char *value, size_t len)
73{
74 if (len == 4 && ngx_strncmp((u_char *) value,
75 (u_char *) "none", 4) == 0)
76 return BRIX_TPC_CRED_NONE;
77
78 if (len == 10 && ngx_strncmp((u_char *) value,
79 (u_char *) "oidc-agent", 10) == 0)
80 return BRIX_TPC_CRED_OIDC_AGENT;
81
82 if (len == 14 && ngx_strncmp((u_char *) value,
83 (u_char *) "token-exchange", 14) == 0)
84 return BRIX_TPC_CRED_TOKEN_EXCHANGE;
85
86 return BRIX_TPC_CRED_UNKNOWN;
87}

89/*
90 * WHAT: Immutable bundle of the inputs a single credential-obtain attempt needs.
91 * WHY: It threads the request/URL/subject-token/scope plus the resolved output
92 * slot through the per-mode helpers without re-plumbing the frozen
93 * 6-parameter public entry point's argument list at each step.
94 * HOW: Populated once in webdav_tpc_cred_obtain_token and passed by const
95 * pointer; token_out is the caller-owned result slot (mutated in place).
96 */
97typedef struct {
98 ngx_http_request_t *r;
99 const char *source_url;
100 const char *subject_token;
101 const char *scope;
102 ngx_str_t *token_out;
103} tpc_cred_request_t;

106/*
107 * WHAT: Record the success/error metric for a completed obtain attempt.
108 * WHY: Both mode paths share the same "validate on OK, then count" tail, so it
109 * lives once to keep the metric labels identical across modes.
110 * HOW: On NGX_OK re-validate the fetched token, then bump NSUCCESS/NERROR by
111 * the (possibly downgraded) result and return it.
112 */
113static ngx_int_t
114tpc_cred_finish(ngx_http_request_t *r, ngx_str_t *token_out, ngx_int_t rc)
115{
116 if (rc == NGX_OK) {
117 rc = webdav_tpc_cred_validate_token(r, token_out);
118 }
119 if (rc == NGX_OK) {
120 webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NSUCCESS);
121 } else {
122 webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NERROR);
123 }
124 return rc;
125}

128/*
129 * WHAT: Run the oidc-agent delegation mode for one request.
130 * WHY: Isolates the OIDC branch so the public dispatcher stays a flat switch.
131 * HOW: Fetch via the oidc-agent pipeline, then fold into the shared
132 * validate-and-count tail.
133 */
134static ngx_int_t
135tpc_cred_run_oidc_agent(const tpc_cred_request_t *req)
136{
137 ngx_int_t rc = tpc_cred_oidc_agent_fetch(req->r, req->source_url,
138 req->scope, req->token_out);
139 return tpc_cred_finish(req->r, req->token_out, rc);
140}

143/*
144 * WHAT: Run the RFC 8693 token-exchange delegation mode for one request.
145 * WHY: Isolates the exchange branch — config presence, subject-token presence,
146 * then the curl exchange — from the dispatcher.
147 * HOW: Fail-closed (NERROR + diagnostic) when the token endpoint or subject
148 * token is missing; otherwise exchange and fold into the shared tail.
149 */
150static ngx_int_t
151tpc_cred_run_token_exchange(const tpc_cred_request_t *req,
152 ngx_http_brix_webdav_loc_conf_t *wconf)
153{
154 ngx_http_request_t *r = req->r;
155 ngx_int_t rc;
156
157 if (wconf->tpc_cred.token_endpoint.len == 0
158 || wconf->tpc_cred.token_endpoint.data == NULL) {
159 BRIX_DIAG_ERR(r->connection->log, 0,
160 "tpc_cred: token-exchange is selected but no token endpoint "
161 "is configured",
162 "third-party-copy credential mode is token-exchange, but "
163 "brix_webdav_tpc_token_endpoint is unset",
164 "set the OAuth token endpoint for your IdP, or switch the TPC "
165 "credential mode away from token-exchange");
166 webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NERROR);
167 return NGX_ERROR;
168 }
169 if (req->subject_token == NULL || *req->subject_token == '\0') {
170 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
171 "tpc_cred: no subject token for token-exchange");
172 webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NERROR);
173 return NGX_ERROR;
174 }
175
176 rc = tpc_cred_rfc8693_exchange(
177 r, req->subject_token, req->source_url, req->scope,
178 (const char *) wconf->tpc_cred.token_endpoint.data,
179 wconf->tpc_cred.token_client_id.len > 0 ?
180 (const char *) wconf->tpc_cred.token_client_id.data : NULL,
181 wconf->tpc_cred.token_client_secret.len > 0 ?
182 (const char *) wconf->tpc_cred.token_client_secret.data : NULL,
183 req->token_out);
184
185 return tpc_cred_finish(r, req->token_out, rc);
186}

189ngx_int_t
190webdav_tpc_cred_obtain_token(ngx_http_request_t *r,
191 brix_tpc_cred_mode_e mode,
192 const char *source_url,
193 const char *subject_token,
194 const char *scope,
195 ngx_str_t *token_out)
196{
197 ngx_http_brix_webdav_loc_conf_t *wconf;
198 tpc_cred_request_t req = {
199 r, source_url, subject_token, scope, token_out
200 };
201
202 wconf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
203 if (wconf == NULL) {
204 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
205 "tpc_cred: no WebDAV location config");
206 return NGX_ERROR;
207 }
208
209 /* Increment started counter. */
210 (void) webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NSTARTED);
211
212 switch (mode) {
213 case BRIX_TPC_CRED_OIDC_AGENT:
214 return tpc_cred_run_oidc_agent(&req);
215
216 case BRIX_TPC_CRED_TOKEN_EXCHANGE:
217 return tpc_cred_run_token_exchange(&req, wconf);
218
219 default:
220 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
221 "tpc_cred: unknown delegation mode %d", (int) mode);
222 webdav_tpc_cred_metric_increment(r, BRIX_TPC_CRED_NUNKNOWN_MODE);
223 return NGX_ERROR;
224 }
225}
