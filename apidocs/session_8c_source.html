15/*
16 * Per-worker monotonic session sequence. SSI sessions are bound to one TCP
17 * connection on one worker, so a per-worker counter is sufficient as the
18 * delivery-guard key: a stale async completion that names a recycled conn_id is
19 * rejected because the generation it captured no longer matches the live slot.

46brix_ssi_req_t *
47brix_ssi_session_req(brix_ssi_session_t *s, uint32_t req_id, int create)
48{
49 int i, free_slot = -1;
50 int cap = BRIX_SSI_MAX_INFLIGHT;
51
52 if (s == NULL) {
53 return NULL;
54 }
55 /* Honor a runtime concurrency cap (never above the fixed table size). */
56 if (s->max_inflight > 0 && s->max_inflight < cap) {
57 cap = s->max_inflight;
58 }
59 for (i = 0; i < cap; i++) {
60 if (s->rr[i].in_use && s->rr[i].req_id == req_id) {
61 return &s->rr[i];
62 }
63 if (!s->rr[i].in_use && free_slot < 0) {
64 free_slot = i;
65 }
66 }
67 if (!create || free_slot < 0) {
68 return NULL;
69 }
70 memset(&s->rr[free_slot], 0, sizeof(s->rr[free_slot]));
71 s->rr[free_slot].in_use = 1;
72 s->rr[free_slot].req_id = req_id;
73 s->rr[free_slot].pool = s->pool;
74 s->rr[free_slot].resp.pool = s->pool; /* growable response buffer */
75 s->rr[free_slot].response_max = s->response_max;
76 return &s->rr[free_slot];
77}
