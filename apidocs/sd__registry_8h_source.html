73/* driver_conf for brix_sd_pblock_driver.init(): the export's storage root (the
74 * blob folder `data/` and `catalog.db` are created beneath it), the SQLite busy
75 * timeout used for cross-worker write contention, and the default object block
76 * size (bytes) for NEW files — 0 selects PBLOCK_DEFAULT_BLOCK_SIZE (64 MiB). The
77 * block size is recorded per file at creation, so retuning it only affects files
78 * written afterwards.
79 *
80 * `enforce_unprivileged` + `unpriv_user`: pblock writes blob files and the SQLite
81 * catalog as the worker's own uid (it has no impersonation broker and never
82 * chowns). Worker-time production builds set `enforce_unprivileged` so a worker
83 * that is (mis)configured to run as root is permanently dropped to an
84 * unprivileged account — `unpriv_user` if set, else "nobody" — BEFORE any blob/
85 * dir/DB is created, so pblock on-disk data can never be root-owned. The normal
86 * way to choose the account is the nginx `user <acct>;` directive (the worker
87 * then already runs unprivileged and the drop is a no-op); `unpriv_user` is the
88 * caller-supplied fallback account for a root worker (NULL ⇒ "nobody"). The flag
89 * is left 0 by the standalone unit test (and any master/config-time probe), which
90 * therefore never drops. */

92 const char *root;
93 int busy_timeout_ms;
94 int64_t block_size;
95 const char *unpriv_user; /* root-worker fallback; NULL/"" ⇒ "nobody" */
96 unsigned enforce_unprivileged:1; /* worker build ⇒ drop off root pre-write */
97} brix_sd_pblock_conf_t;

107/* ---- brix_sd_posix_wrap — bind a bare fd to a stack POSIX SD object ------
108 *
109 * WHAT: Zero-initializes *obj and points it at the built-in POSIX driver over a
110 * plain fd, for dispatching a single fd op through the driver vtable.
111 *
112 * WHY: The VFS's fd-keyed I/O loops (pread_full/pwrite_full/write_counted) own
113 * the EINTR/short-I/O policy but must route the raw syscall through the
114 * Storage Driver seam. The POSIX raw slots touch only obj->fd, so a
115 * zero-initialized stack object with driver+fd set is sufficient and keeps
116 * the hot path allocation-free.
117 *
118 * HOW: ngx_memzero, set the driver pointer and fd.
119 */
120static ngx_inline void
121brix_sd_posix_wrap(brix_sd_obj_t *obj, ngx_fd_t fd)
122{
123 ngx_memzero(obj, sizeof(*obj));
124 obj->driver = &brix_sd_posix_driver;
125 obj->fd = fd;
126}

128/* ---- brix_sd_obj_preadv — vectored read with per-driver fallback ---------
129 *
130 * WHAT: Issues a positioned vectored read on `obj`, using the driver's native
131 * preadv slot when present, else emulating it with one driver->pread per
132 * iovec. Returns total bytes read (short = EOF), or -1 with errno set
133 * (ENOTSUP when the driver has no byte-read slot at all).
134 *
135 * WHY: preadv/preadv2 are OPTIONAL vtable slots — remote/object drivers
136 * (sd_remote) implement only pread, and the read fan-out paths (pgread,
137 * kXR_readv batches) previously called obj->driver->preadv unguarded,
138 * which is a NULL call (SIGSEGV) on those backends. One shared fallback
139 * keeps the emulation policy (stop on short read) in a single place.
140 *
141 * HOW: Native slot when non-NULL; otherwise fill each iovec with a pread
142 * loop — a short-but-nonzero pread is NOT EOF (drivers may cap a single
143 * request), only a 0-byte pread is, so keep reading until the iovec is
144 * full or the driver reports EOF/error. A -1 mid-loop after bytes were
145 * already delivered reports the bytes (POSIX readv semantics); the
146 * caller's next call surfaces the error.
147 */
148static ngx_inline ssize_t
149brix_sd_obj_preadv(brix_sd_obj_t *obj, const struct iovec *iov, int iovcnt,
150 off_t off)
151{
152 ssize_t total = 0;
153 int i;
154 int eof = 0;
155
156 if (obj->driver->preadv != NULL) {
157 return obj->driver->preadv(obj, iov, iovcnt, off);
158 }
159
160 if (obj->driver->pread == NULL) {
161 errno = ENOTSUP;
162 return -1;
163 }
164
165 for (i = 0; i < iovcnt && !eof; i++) {
166 size_t filled = 0;
167
168 while (filled < iov[i].iov_len) {
169 ssize_t n = obj->driver->pread(obj,
170 (char *) iov[i].iov_base + filled,
171 iov[i].iov_len - filled,
172 off + total);
173 if (n < 0) {
174 if (total > 0) {
175 return total;
176 }
177 return -1;
178 }
179 if (n == 0) {
180 eof = 1; /* only a 0-byte pread means EOF */
181 break;
182 }
183 filled += (size_t) n;
184 total += n;
185 }
186 }
187
188 return total;
189}
