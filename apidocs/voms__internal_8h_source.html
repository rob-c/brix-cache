8/*
9 * ABI-compatible struct definitions for voms-2.1.3 / EL9.
10 *
11 * We dlopen() libvoms at runtime (to avoid a hard dependency) and call its
12 * VOMS_Retrieve() / VOMS_Destroy() functions through function pointers.
13 * These struct layouts match the libvoms ABI; only fields we dereference are
14 * given meaningful types. The rest are opaque placeholders so the byte offsets
15 * stay correct across minor library versions.
16 *
17 * DO NOT change field order or add/remove fields — that would break the dlopen
18 * ABI contract.
19 */
20
21/*
22 * One VOMS attribute item: a group/role/capability triple from one AC.
23 * group: VOMS group name, e.g. "/cms/Production"
24 * role: role name, e.g. "pilot" (may be NULL for group-only membership)
25 * cap: capability string (rarely used; usually NULL)
26 */
27struct voms_data_item {
28 char *group;
29 char *role;
30 char *cap;
31};

33/*
34 * One VOMS attribute certificate (AC) entry: metadata about one VO's AC
35 * embedded in the X.509 proxy.
36 * voname: the VO name, e.g. "cms"
37 * fqan: NULL-terminated array of fully-qualified attribute names, e.g.
38 * ["/cms/Production/Role=pilot/Capability=NULL", NULL]
39 * std: NULL-terminated array of voms_data_item pointers (structured fqan)
40 */
41struct voms_entry {

51 char *date2; /* AC validity end */
