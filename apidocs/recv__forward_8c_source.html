1/*
2 * cms/recv_forward.c — Plane-B forwarded-namespace-op executor for the CMS
3 * client-side (manager-connection) receive path.
4 *
5 * WHAT: Applies a manager-forwarded namespace mutation (kYR_chmod/mkdir/mkpath/
6 * mv/rm/rmdir/trunc) to our local storage. cms_node_exec_forward() decodes the
7 * request and routes to one of two storage legs: a non-POSIX export (pblock/
8 * object) mutates through its backend driver's namespace slots
9 * (cms_forward_via_driver); the default POSIX export takes the kernel-confined
10 * *_beneath path (cms_forward_via_posix). Replies silent-on-success /
11 * kYR_error-on-failure, byte-exact with stock cmsd.
12 *
13 * WHY: Split (Phase-79 file-size split) from recv.c so the storage-mutation
14 * machinery — driver slot plumbing and kernel-confined POSIX apply — lives in
15 * its own focused, independently reviewable file under the size guideline. The
16 * one entry point the opcode dispatcher calls (cms_node_exec_forward) is
17 * declared in recv_internal.h; everything else here is a file-local helper.
18 *
19 * HOW: cms_drv_* wrap individual driver slots (unlink/setattr/ftruncate);
20 * cms_node_exec_driver / cms_posix_apply are the pure per-leg executors;
21 * cms_forward_via_driver / cms_forward_via_posix add the shared logging + wire
22 * reply tail; cms_node_exec_forward is the decode-then-route orchestrator. The
23 * driver namespace and the POSIX *_beneath helpers are both inherently
24 * export-confined, so a hostile manager cannot escape the export root.
25 */

38/*
39 * cms_drv_unlink — remove a file (isdir=0) or directory (isdir=1) through a
40 * backend driver's unlink slot. Shared by the RM and RMDIR forwarded-op cases so
41 * the ENOSYS-on-missing-slot check lives once. Returns 0 / -1+errno.
42 */
43static int
44cms_drv_unlink(brix_sd_instance_t *sd, const char *path, int isdir)
45{
46 if (sd->driver->unlink == NULL) { errno = ENOSYS; return -1; }
47 return sd->driver->unlink(sd, path, isdir) == NGX_OK ? 0 : -1;
48}

50/*
51 * cms_drv_chmod — apply a forwarded chmod through the driver's setattr slot.
52 * A driver with no mutable metadata (setattr slot absent) treats chmod as a
53 * successful no-op, matching how object catalogs ignore POSIX modes. Returns
54 * 0 / -1+errno.
55 */
56static int
57cms_drv_chmod(brix_sd_instance_t *sd, const brix_cms_node_plan_t *plan)
58{
59 brix_sd_setattr_t attr;
60
61 if (sd->driver->setattr == NULL) { return 0; } /* no mutable metadata — no-op */
62 ngx_memzero(&attr, sizeof(attr));
63 attr.set_mode = 1;
64 attr.mode = plan->mode;
65 return sd->driver->setattr(sd, plan->path, &attr) == NGX_OK ? 0 : -1;
66}

68/*
69 * cms_drv_trunc — truncate an object through the driver's open/ftruncate/close
70 * slots. Owns the driver object handle linearly (open → truncate → close) and
71 * preserves the truncate errno across the close so the caller reports the real
72 * failure cause. Returns 0 / -1+errno.
73 */
74static int
75cms_drv_trunc(brix_sd_instance_t *sd, const brix_cms_node_plan_t *plan)
76{
77 const brix_sd_driver_t *drv = sd->driver;
78 int err = 0;
79 int rc;
80 int saved;
81 brix_sd_obj_t *o;
82
83 if (drv->open == NULL || drv->ftruncate == NULL) { errno = ENOSYS; return -1; }
84 o = drv->open(sd, plan->path, BRIX_SD_O_WRITE, 0, &err);
85 if (o == NULL) { errno = err ? err : EIO; return -1; }
86 rc = drv->ftruncate(o, (off_t) plan->size) == NGX_OK ? 0 : -1;
87 saved = errno;
88 if (drv->close != NULL) { drv->close(o); }
89 errno = saved;
90 return rc;
91}

93/*
94 * cms_node_exec_driver — apply a manager-forwarded namespace op through a
95 * NON-default backend driver's slots, so a pblock/object data node mutates its
96 * catalog instead of the real filesystem (the confined *_beneath helpers the
97 * POSIX path uses only touch the real FS). plan->path/path2 are export-relative
98 * (leading slash), the format the driver slots expect. Returns 0 / -1+errno; sets
99 * *handled=0 only for an action the driver cannot express. The driver namespace
100 * is inherently export-confined, so a hostile manager still cannot escape it.
101 */
102static int
103cms_node_exec_driver(brix_sd_instance_t *sd, const char *root_canon,
104 const brix_cms_node_plan_t *plan, ngx_log_t *log, int *handled)
105{
106 const brix_sd_driver_t *drv = sd->driver;
107
108 *handled = 1;
109
110 switch (plan->action) {
111 case XRDCMS_NACT_MKDIR:
112 if (drv->mkdir == NULL) { errno = ENOSYS; return -1; }
113 return drv->mkdir(sd, plan->path, plan->mode) == NGX_OK ? 0 : -1;
114
115 case XRDCMS_NACT_MKPATH:
116 /* create the whole path + missing parents in the driver namespace. */
117 return brix_vfs_backend_mkpath(root_canon, plan->path, plan->mode, log);
118
119 case XRDCMS_NACT_RMDIR:
120 return cms_drv_unlink(sd, plan->path, 1);
121
122 case XRDCMS_NACT_RM:
123 return cms_drv_unlink(sd, plan->path, 0);
124
125 case XRDCMS_NACT_MV:
126 if (drv->rename == NULL) { errno = ENOSYS; return -1; }
127 return drv->rename(sd, plan->path, plan->path2, 0) == NGX_OK ? 0 : -1;
128
129 case XRDCMS_NACT_CHMOD:
130 return cms_drv_chmod(sd, plan);
131
132 case XRDCMS_NACT_TRUNC:
133 return cms_drv_trunc(sd, plan);
134
135 default:
136 *handled = 0;
137 return -1;
138 }
139}

141/* cms_forward_via_driver — driver-backend leg of a forwarded namespace op: run
142 * the planned mutation through the non-default backend's namespace slots and
143 * reply like stock cmsd (silent on success, kYR_error + strerror on failure,
144 * "unsupported operation" for an action the driver cannot express). Split from
145 * cms_node_exec_forward so each storage leg owns its own logging/reply tail. */
146static ngx_int_t
147cms_forward_via_driver(ngx_brix_cms_ctx_t *ctx, brix_sd_instance_t *sd,
148 u_char code, uint32_t streamid, const brix_cms_node_plan_t *plan)
149{
150 const char *root_canon = ctx->conf->common.root_canon;
151 int handled;
152 int rc;
153
154 rc = cms_node_exec_driver(sd, root_canon, plan, ctx->cycle->log,
155 &handled);
156 if (!handled) {
157 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
158 "unsupported operation");
159 }
160 if (rc != 0) {
161 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
162 "brix: CMS node: forwarded op code=%ui path=%s failed: %s",
163 (ngx_uint_t) code, plan->path, strerror(errno));
164 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
165 strerror(errno));
166 }
167 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
168 "brix: CMS node: forwarded op code=%ui path=%s OK (driver)",
169 (ngx_uint_t) code, plan->path);
170 return NGX_OK;
171}

173/* cms_posix_apply — POSIX-export leg of a forwarded namespace op: apply the
174 * planned mutation to the local filesystem UNDER KERNEL CONFINEMENT
175 * (src/fs/path/beneath.h confined helpers / VFS *_at helpers, openat2
176 * RESOLVE_BENEATH under the persistent export rootfd). Pure execution — the
177 * caller owns logging and the wire reply. Returns 0 / -1+errno; sets
178 * *handled=0 only for an action the plan cannot express. */
179static int
180cms_posix_apply(ngx_brix_cms_ctx_t *ctx, const brix_cms_node_plan_t *plan,
181 int *handled)
182{
183 int rootfd = ctx->conf->rootfd;
184 const char *root_canon = ctx->conf->common.root_canon;
185
186 *handled = 1;
187
188 switch (plan->action) {
189 case XRDCMS_NACT_MKDIR:
190 return brix_mkdir_beneath(rootfd, plan->path, plan->mode);
191
192 case XRDCMS_NACT_MKPATH: {
193 char full[PATH_MAX];
194 brix_beneath_full_path(root_canon, plan->path, full, sizeof(full));
195 return brix_mkdir_recursive_beneath(ctx->cycle->log, rootfd, root_canon,
196 full, plan->mode, NULL);
197 }
198 case XRDCMS_NACT_RMDIR:
199 return brix_vfs_unlink_at(rootfd, plan->path, 1);
200
201 case XRDCMS_NACT_RM:
202 return brix_vfs_unlink_at(rootfd, plan->path, 0);
203
204 case XRDCMS_NACT_MV:
205 return brix_rename_beneath(rootfd, plan->path, plan->path2);
206
207 case XRDCMS_NACT_CHMOD: {
208 int rc;
209 int fd = brix_vfs_open_fd_at(rootfd, plan->path, O_RDONLY, 0);
210 if (fd < 0) { return -1; }
211 rc = fchmod(fd, plan->mode); /* vfs-seam-allow: metadata on a VFS-opened confined fd */
212 close(fd); /* vfs-seam-allow: metadata on a VFS-opened confined fd */
213 return rc;
214 }
215 case XRDCMS_NACT_TRUNC: {
216 int rc;
217 int fd = brix_vfs_open_fd_at(rootfd, plan->path, O_WRONLY, 0);
218 if (fd < 0) { return -1; }
219 rc = ftruncate(fd, (off_t) plan->size); /* vfs-seam-allow: metadata on a VFS-opened confined fd */
220 close(fd); /* vfs-seam-allow: metadata on a VFS-opened confined fd */
221 return rc;
222 }
223 default:
224 *handled = 0;
225 return -1;
226 }
227}

229/* cms_forward_via_posix — POSIX-export reply wrapper for a forwarded namespace
230 * op: run cms_posix_apply and reply like stock cmsd — silent on success (as
231 * cmsd Execute() does on a NULL return from a non-forwarding leaf node),
232 * kYR_error (kYR_EINVAL + strerror) on failure, "unsupported operation" for an
233 * inexpressible action. A hostile manager cannot make the node mutate outside
234 * its export root — an escape fails EXDEV and becomes kYR_error. */
235static ngx_int_t
236cms_forward_via_posix(ngx_brix_cms_ctx_t *ctx, u_char code, uint32_t streamid,
237 const brix_cms_node_plan_t *plan)
238{
239 int handled;
240 int rc;
241
242 rc = cms_posix_apply(ctx, plan, &handled);
243 if (!handled) {
244 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
245 "unsupported operation");
246 }
247
248 if (rc != 0) {
249 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
250 "brix: CMS node: forwarded op code=%ui path=%s failed: %s",
251 (ngx_uint_t) code, plan->path, strerror(errno));
252 /* byte-exact: ecode is always kYR_EINVAL; text carries strerror. */
253 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
254 strerror(errno));
255 }
256
257 /* Success: stay silent — exactly as cmsd Execute() does on a NULL return
258 * from a non-forwarding leaf node. */
259 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
260 "brix: CMS node: forwarded op code=%ui path=%s ok",
261 (ngx_uint_t) code, plan->path);
262 return NGX_OK;
263}

265/* cms_node_exec_forward — execute a manager-forwarded namespace op (Plane B): decode
266 * a kYR_chmod/mkdir/mkpath/mv/rm/rmdir/trunc (the shared request-marshal
267 * prologue: rrdata_parse → pure node_plan), then route to the storage leg —
268 * a non-POSIX export (pblock/object) mutates through its driver's namespace
269 * slots (cms_forward_via_driver); the default POSIX export takes the kernel-
270 * confined *_beneath path (cms_forward_via_posix). */
271ngx_int_t
272cms_node_exec_forward(ngx_brix_cms_ctx_t *ctx, u_char code, uint32_t streamid,
273 const u_char *payload, size_t plen)
274{
275 brix_cms_rrdata_t d;
276 brix_cms_node_plan_t plan;
277 brix_sd_instance_t *sd;
278
279 if (brix_cms_rrdata_parse(code, payload, plen, &d) != 0
280 || brix_cms_node_plan(code, &d, &plan) != 0)
281 {
282 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
283 "badly formed request");
284 }
285
286 /* A manager-only node (no local export) cannot satisfy a mutation. */
287 if (ctx->conf->rootfd < 0) {
288 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
289 "no local storage");
290 }
291
292 /* A non-POSIX export (pblock/object) routes the mutation through its driver's
293 * namespace slots; the default POSIX export keeps the confined *_beneath path
294 * unchanged. */
295 sd = brix_vfs_backend_resolve(ctx->conf->common.root_canon,
296 ctx->cycle->log);
297 if (sd != NULL && sd->driver != brix_sd_default_driver()) {
298 return cms_forward_via_driver(ctx, sd, code, streamid, &plan);
299 }
300
301 return cms_forward_via_posix(ctx, code, streamid, &plan);
302}
