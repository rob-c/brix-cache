1/*
2 * cli_conn.c — shared connect-and-report scaffold for the front-end tools.
3 *
4 * WHAT: brix_cli_connect() does the endpoint-parse → connect ladder every
5 * single-shot tool repeats, printing the canonical "prog: <msg>" (parse) /
6 * "prog: connect: <msg>" (connect) on failure and returning a process exit
7 * code (0 = connected). brix_report_err() is the per-operation failure
8 * idiom — "tool: op path: msg" + a credential hint + the mapped shell code —
9 * that xrdfs repeats ~35×.
10 * WHY: Collapses the ~12 inline connect ladders in xrddiag and the bodies of the
11 * single-shot tools, and the xrdfs per-handler error block, to one call each
12 * — behaviour-preserving (identical messages, hints and exit codes).
13 * HOW: Mirrors the historical inline code exactly so adoption is a pure
14 * de-duplication: endpoint_parse failure → XRDC_EXIT_USAGE; connect failure
15 * → brix_shellcode(st); both after the same stderr line.
16 * brix_report_err accepts a nullable brix_url* to enable WS-3 (double-slash
17 * hint) and WS-7 (doctor referral) on the same error line (spec P3).
18 */

20#include "cli/cli_hint.h" /* brix_hint_url_double_slash, brix_hint_doctor_referral */

24int
25brix_cli_connect(const char *endpoint, const brix_opts *o,
26 brix_conn *c, const char *prog, brix_status *st)
27{
28 brix_url u;
29
30 brix_status_clear(st);
31 if (brix_endpoint_parse(endpoint, &u, st) != 0) {
32 fprintf(stderr, "%s: %s\n", prog, st->msg);
33 return XRDC_EXIT_USAGE;
34 }
35 if (brix_connect_resilient(c, &u, o, st) != 0) {
36 fprintf(stderr, "%s: connect: %s\n", prog, st->msg);
37 return brix_shellcode(st);
38 }
39 return 0;
40}

42/*
43 * WHAT: build a "host[:port]" endpoint string from a parsed brix_url for the
44 * WS-7 doctor-referral hint ("xrddiag check <endpoint>").
45 * WHY: brix_report_err receives the parsed URL, not the original string;
46 * host[:port] is exactly what xrddiag check accepts as an endpoint.
47 * HOW: snprintf into the caller's buffer; returns the buffer on success or
48 * NULL when there is no host to report (the hint then falls back to a
49 * generic "<endpoint>" placeholder). Sanitization (control-byte
50 * escaping) happens inside brix_hint_doctor_referral, not here.
51 */
52static const char *
53report_err_endpoint(const brix_url *url, char *buf, size_t bufsz)
54{
55 if (url == NULL || url->host[0] == '\0') {
56 return NULL;
57 }
58 if (url->port > 0) {
59 snprintf(buf, bufsz, "%s:%d", url->host, url->port);
60 } else {
61 snprintf(buf, bufsz, "%s", url->host);
62 }
63 return buf;
64}

66int
67brix_report_err(FILE *out, const char *tool, const char *op, const char *path,
68 const brix_status *st, int want_write, const brix_url *url)
69{
70 /*
71 * WHAT: print the "tool: op path: msg" line, then credential hint and
72 * the WS-3 / WS-7 canned hints where applicable.
73 * WHY: single shared error-reporter across xrdfs, xrd, xrdcp front-ends
74 * so WS-3/WS-7 hints are consistent and easy to audit.
75 * HOW: url is nullable. The doctor referral (WS-7) is owned by
76 * brix_cred_hint_for_status_url — one canonical call chain — and
77 * here receives the host[:port] endpoint built from the parsed URL
78 * so the interactive hint names a concrete target. The double-slash
79 * hint (WS-3) fires only for not-found-class failures on a URL whose
80 * single_slash_path bit is set (both gates live inside the helper).
81 */
82 char epbuf[300]; /* host(256) + ':' + port digits + NUL */
83 const char *endpoint;
84
85 fprintf(out, "%s: %s %s: %s\n", tool, op, path, st->msg);
86 endpoint = report_err_endpoint(url, epbuf, sizeof(epbuf));
87 brix_cred_hint_for_status_url(st, want_write, out, endpoint); /* Phase 40 (c) + WS-7 */
88 brix_hint_url_double_slash(st, url); /* spec WS-3 */
89 return brix_shellcode(st);
90}
