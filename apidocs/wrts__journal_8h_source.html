1/*
2 * src/write/wrts_journal.h — kXR_recoverWrts per-handle write-recovery journal.
3 *
4 * WHAT: A fixed-size ring buffer (BRIX_WRTS_JOURNAL_SLOTS entries) embedded
5 * in brix_file_t that records the offset+length of every pwrite() that
6 * has been committed to disk. When a client reconnects and replays an
7 * in-flight write, brix_wrts_is_replay() detects the overlap and the
8 * write handler short-circuits the pwrite() — preventing double-writes
9 * that would corrupt the file.
10 *
11 * WHY: Advertising kXR_recoverWrts in kXR_protocol tells XrdCl that it may
12 * replay in-flight writes after a TCP disconnect. Without this journal
13 * the server cannot distinguish a fresh write from a recovery replay,
14 * so advertising the flag without the journal would silently double-write.
15 *
16 * HOW: Four public functions:
17 * brix_wrts_open() — initialise journal fields on kXR_open (writable)
18 * brix_wrts_record() — append a committed write to the ring
19 * brix_wrts_is_replay() — return 1 if (offset,len) is already journalled
20 * brix_wrts_flush() — clear the ring on kXR_sync / kXR_close
21 *
22 * The ring is O(BRIX_WRTS_JOURNAL_SLOTS) to scan; 64 slots covers normal
23 * HEP streaming workloads without heap allocation.
24 */
25

46/*
47 * brix_wrts_is_replay — return 1 if the incoming (offset, length) range is
48 * fully covered by any existing journal entry.
49 *
50 * A "covered" entry satisfies:
51 * entry.offset <= offset AND
52 * entry.offset + entry.length >= offset + length
53 *
54 * Returns 0 if the journal is empty, wrts_enabled is 0, or no entry covers
55 * the incoming range.
56 */
57int brix_wrts_is_replay(const brix_file_t *f,

60/*
61 * brix_wrts_flush — clear the journal.
62 *
63 * Called on kXR_sync (writes committed to stable storage — new generation
64 * starts) and on kXR_close (handle going away). After flush, any subsequent
65 * write with the same offset is treated as a fresh write, not a replay.
66 */
67void brix_wrts_flush(brix_file_t *f);
