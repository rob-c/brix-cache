21/*
22 * stat_manager_route — redirector-side answers for a path-mode stat.
23 *
24 * A redirector must answer stat without local storage — stock and go-hep
25 * clients stat a path before they open it. In precedence order: a static
26 * manager_map prefix redirects outright (mirrors the open/locate paths); in
27 * manager mode the §6 CNS inventory can answer size/mtime directly (a true
28 * global-namespace stat), tried/triedrc exhaustion answers not-found to stop
29 * redirect loops, the registry redirects to a live data server (tolerating a
30 * just-dropped heartbeat, like open), and a registry miss is forwarded to
31 * the CMS parent, parking the session (NGX_AGAIN). Returns 0 when the
32 * request was answered (*rc holds the value to return, possibly NGX_AGAIN)
33 * and 1 when the caller should continue with a local stat.
34 */
35int
36stat_manager_route(brix_ctx_t *ctx, ngx_connection_t *c,
37 ngx_stream_brix_srv_conf_t *conf, const char *reqpath, ngx_int_t *rc)
38{
39 /* Static manager_map: an explicit prefix→backend redirect (mirrors the
40 * open/locate paths) so a static-map redirector also serves stat — stock
41 * and go-hep clients stat a path before they open it, and without this a
42 * map-only redirector answered stat locally (IOError, no root). */
43 if (conf->manager_map != NULL) {
44 const brix_manager_map_t *m =
45 brix_find_manager_map(reqpath, conf->manager_map);
46 if (m != NULL) {
47 brix_log_access(ctx, c, "STAT", reqpath, "manager_map",
48 1, kXR_ok, NULL, 0);
49 BRIX_OP_OK(ctx, BRIX_OP_STAT);
50 *rc = brix_send_redirect(ctx, c, (const char *) m->host.data,
51 m->port);
52 return 0;
53 }
54 }
55
56 /* Manager mode: redirect to a registered data server. */
57 if (!conf->manager_mode) {
58 return 1;
59 }
60

62 * stat directly (size/mtime) instead of redirecting — a true global
63 * namespace stat at the redirector. */
64 if (conf->cns_mode == BRIX_CNS_COLLECT) {
65 struct stat cst;
66 if (brix_cns_stat(reqpath, &cst) == NGX_OK) {
67 char cbody[256];

69 brix_log_access(ctx, c, "STAT", reqpath, "cns",
70 1, 0, NULL, 0);
71 BRIX_OP_OK(ctx, BRIX_OP_STAT);
72 *rc = brix_send_ok(ctx, c, cbody,
73 (uint32_t) (strlen(cbody) + 1));
74 return 0;
75 }
76 }
77
78 /* tried/triedrc: if the client has already visited every server
79 * that holds this path and they returned enoent, stop redirecting
80 * and answer not-found — otherwise the client redirect-loops. */
81 if (brix_manager_tried_exhausted(ctx->recv.payload, ctx->recv.cur_dlen,
82 reqpath)) {
83 BRIX_BAIL_ERR(ctx, c, BRIX_OP_STAT, "STAT", reqpath, "-",
84 kXR_NotFound, "file not found on any data server", rc);
85 }
86
87 /* Like open: tolerate a server whose CMS heartbeat just dropped (it
88 * is almost certainly still serving) rather than a false NotFound. */
89 {

92
93 if (brix_srv_select_or_blacklisted(reqpath, 0, redir_host,
94 sizeof(redir_host), &redir_port)) {
95 brix_log_access(ctx, c, "STAT", reqpath, "registry",
96 1, kXR_ok, NULL, 0);
97 BRIX_OP_OK(ctx, BRIX_OP_STAT);
98 *rc = brix_send_redirect(ctx, c, redir_host, redir_port);
99 return 0;
100 }
101 }
102
103 /* Registry miss — ask CMS parent if configured. */
104 if (conf->cms.ctx != NULL) {
105 uint32_t streamid;
106
107 streamid = ngx_brix_cms_next_streamid(conf->cms.ctx);
108 if (brix_pending_insert(streamid, ngx_pid, c->fd,
109 c->number,
110 ctx->recv.cur_streamid,
111 conf->cms.locate_timeout) == NGX_OK)
112 {
113 ctx->cms_wait_streamid = streamid;
114 ctx->state = XRD_ST_WAITING_CMS;
115 ngx_add_timer(c->read, conf->cms.locate_timeout);
116 if (ngx_brix_cms_send_locate(conf->cms.ctx, streamid,
117 reqpath) == NGX_OK)
118 {
119 *rc = NGX_AGAIN;
120 return 0;
121 }
122 ngx_del_timer(c->read);
123 ctx->state = XRD_ST_REQ_HEADER;
124 brix_pending_remove(streamid, ngx_pid);
125 }
126 }
127
128 return 1;
129}
