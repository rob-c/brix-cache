9/*
10 * webdav_tpc_curl_ctx_t — file-local working bundle for a single-stream TPC
11 * transfer.
12 *
13 * WHAT: carries the immutable request parameters (log/conf/headers/direction/
14 * paths/id, snapshotted from the caller's webdav_tpc_req_t) plus the
15 * mutable libcurl resources (curl handle, header and resolve slists, the
16 * local FILE*) that the setup helpers below acquire and thread among
17 * themselves.
18 * WHY: webdav_tpc_run_curl_core's body decomposes into setup/perform/finish
19 * steps that must share the same growing resource set — a struct passes
20 * that state explicitly (no globals) instead of a long helper arg list.
21 * It stays file-local because it owns live curl resources, not a public
22 * request contract (that is webdav_tpc_req_t in webdav_tpc.h).
23 * HOW: the core builds one on the stack, hands its address to each helper, and
24 * always tears the resources down through webdav_tpc_curl_finish().
25 */
26typedef struct {
27 ngx_log_t *log;
28 ngx_http_brix_webdav_loc_conf_t *conf;
29 ngx_array_t *transfer_headers;
30 int is_push;
31 const char *file_path;
32 const char *url;
33 const char *log_tag;
34 uint64_t transfer_id;
35 const char *user_cert; /* per-user pull-leg cert (or NULL) */
36 const char *user_key; /* per-user pull-leg key (or NULL) */
37
38 CURL *curl;
39 struct curl_slist *hdrs;
40 struct curl_slist *resolve;
41 FILE *fp;
42} webdav_tpc_curl_ctx_t;

45/*
46 * tpc_core_apply_tls_creds — set the client cert/key and CA options on the easy
47 * handle, preferring the requesting user's delegated proxy over the static
48 * service cert.
49 *
50 * WHAT: emits CURLOPT_SSLCERT/SSLKEY from ctx->user_cert/_key when present (a
51 * per-user delegated proxy the resolver supplied), else from conf->tpc_cert/
52 * _key (the service identity); CA options always come from conf.
53 * WHY: a TPC PULL must present the END USER's x509 to the source, not the
54 * destination's service cert. When no per-user proxy was resolved
55 * (non-delegated setups), behaviour is byte-identical to before. The
56 * bearer path is untouched (it is an Authorization header).
57 * HOW: user_cert/_key win their respective slots and each falls back
58 * independently to the matching conf field only when NULL; the two CA
59 * setopts are emitted only when their conf field is non-empty, as before.
60 */
61static void
62tpc_core_apply_tls_creds(webdav_tpc_curl_ctx_t *ctx)
63{
64 ngx_http_brix_webdav_loc_conf_t *conf = ctx->conf;
65
66 if (ctx->user_cert != NULL) {
67 curl_easy_setopt(ctx->curl, CURLOPT_SSLCERT, ctx->user_cert);
68 } else if (conf->tpc_cert.len > 0) {
69 curl_easy_setopt(ctx->curl, CURLOPT_SSLCERT,
70 (const char *) conf->tpc_cert.data);
71 }
72 if (ctx->user_key != NULL) {
73 curl_easy_setopt(ctx->curl, CURLOPT_SSLKEY, ctx->user_key);
74 } else if (conf->tpc_key.len > 0) {
75 curl_easy_setopt(ctx->curl, CURLOPT_SSLKEY,
76 (const char *) conf->tpc_key.data);
77 }
78 if (conf->tpc_cafile.len > 0) {
79 curl_easy_setopt(ctx->curl, CURLOPT_CAINFO,
80 (const char *) conf->tpc_cafile.data);
81 }
82 if (conf->tpc_cadir.len > 0) {
83 curl_easy_setopt(ctx->curl, CURLOPT_CAPATH,
84 (const char *) conf->tpc_cadir.data);
85 }
86}

89/*
90 * tpc_core_append_headers — build the CURLOPT_HTTPHEADER slist from the
91 * transfer_headers array.
92 *
93 * WHAT: appends each header string (Source/Credential and friends) to
94 * ctx->hdrs, then installs the list on the handle.
95 * WHY: the append loop carries an OOM early-exit; keeping it in its own helper
96 * lets the core return through finish() without a nested loop+branch.
97 * HOW: NGX_OK when the list is installed (or there are no headers to add);
98 * NGX_ERROR on curl_slist_append() OOM (ctx->hdrs holds the partial list
99 * for the caller to free via finish()).
100 */
101static ngx_int_t
102tpc_core_append_headers(webdav_tpc_curl_ctx_t *ctx)
103{
104 ngx_str_t *headers;
105 ngx_uint_t i;
106
107 if (ctx->transfer_headers == NULL || ctx->transfer_headers->nelts == 0) {
108 return NGX_OK;
109 }
110
111 headers = ctx->transfer_headers->elts;
112 for (i = 0; i < ctx->transfer_headers->nelts; i++) {
113 struct curl_slist *next;
114 next = curl_slist_append(ctx->hdrs, (const char *) headers[i].data);
115 if (next == NULL) {
116 ngx_log_error(NGX_LOG_ERR, ctx->log, 0,
117 "brix_webdav: curl_slist_append() OOM for TPC %s",
118 ctx->log_tag);
119 return NGX_ERROR;
120 }
121 ctx->hdrs = next;
122 }
123 curl_easy_setopt(ctx->curl, CURLOPT_HTTPHEADER, ctx->hdrs);
124 return NGX_OK;
125}

128/*
129 * tpc_core_open_local — open the local file endpoint through the confined VFS
130 * resolver and wire it to the handle.
131 *
132 * WHAT: push → read-only open of the export path + UPLOAD/READDATA (and
133 * INFILESIZE from fstat); pull → create/truncate the staged temp +
134 * WRITEDATA.
135 * WHY: the two directions share the confined-open + fdopen + null-check shape;
136 * the confinement (openat2 RESOLVE_BENEATH + O_NOFOLLOW) is the symlink-
137 * escape guard and must stay in one reviewable place.
138 * HOW: NGX_OK with ctx->fp set on success; NGX_ERROR (ctx->fp left NULL) when
139 * the confined open or fdopen fails. Behavior is byte-identical to the
140 * former inline branches.
141 */
142static ngx_int_t
143tpc_core_open_local(webdav_tpc_curl_ctx_t *ctx)
144{
145 ngx_http_brix_webdav_loc_conf_t *conf = ctx->conf;
146 int fd;
147
148 if (ctx->is_push) {
149 struct stat st;
150
151 fd = brix_vfs_open_fd(ctx->log, conf->common.root_canon, ctx->file_path,
152 O_RDONLY | O_CLOEXEC | O_NOFOLLOW, 0);
153 if (fd >= 0) {
154 ctx->fp = fdopen(fd, "rb");
155 if (ctx->fp == NULL) {
156 (void) close(fd);
157 }
158 }
159 if (ctx->fp == NULL) {
160 ngx_log_error(NGX_LOG_ERR, ctx->log, ngx_errno,
161 "brix_webdav: TPC push open(\"%s\") failed",
162 ctx->file_path);
163 return NGX_ERROR;
164 }
165 if (fstat(fileno(ctx->fp), &st) == 0) {
166 curl_easy_setopt(ctx->curl, CURLOPT_INFILESIZE_LARGE,
167 (curl_off_t) st.st_size);
168 }
169 curl_easy_setopt(ctx->curl, CURLOPT_UPLOAD, 1L);
170 curl_easy_setopt(ctx->curl, CURLOPT_READDATA, ctx->fp);
171 return NGX_OK;
172 }
173
174 fd = brix_vfs_open_fd(ctx->log, conf->common.root_canon, ctx->file_path,
175 O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC | O_NOFOLLOW,
176 0600);
177 if (fd >= 0) {
178 ctx->fp = fdopen(fd, "wb");
179 if (ctx->fp == NULL) {
180 (void) close(fd);
181 }
182 }
183 if (ctx->fp == NULL) {
184 ngx_log_error(NGX_LOG_ERR, ctx->log, ngx_errno,
185 "brix_webdav: TPC pull open(\"%s\") failed",
186 ctx->file_path);
187 return NGX_ERROR;
188 }
189 curl_easy_setopt(ctx->curl, CURLOPT_WRITEDATA, ctx->fp);
190 return NGX_OK;
191}

194/*
195 * tpc_core_setup — initialise the easy handle and apply every transfer option
196 * up to (but not including) curl_easy_perform().
197 *
198 * WHAT: curl_easy_init + fixed options + errbuf + optional progress cb +
199 * HTTPS-only pin + URL + TLS/rebind pin (tpc_curl_secure) + timeout/stall
200 * bounds + client creds + request headers + local-file open.
201 * WHY: concentrates the whole configure phase so the core reads as
202 * setup → perform → finish; each early failure returns a distinct rc that
203 * the core forwards through finish().
204 * HOW: errbuf and progress are owned by the caller (they must outlive
205 * curl_easy_perform), so they are passed in. Returns NGX_OK on a fully
206 * configured handle, or the mapped HTTP error status on failure.
207 */
208static ngx_int_t
209tpc_core_setup(webdav_tpc_curl_ctx_t *ctx, char *errbuf
210#ifdef CURLOPT_XFERINFOFUNCTION
211 , webdav_tpc_curl_progress_t *progress
212#endif
213 )
214{
215 ctx->curl = curl_easy_init();
216 if (ctx->curl == NULL) {
217 ngx_log_error(NGX_LOG_ERR, ctx->log, 0,
218 "brix_webdav: curl_easy_init() failed for TPC %s",
219 ctx->log_tag);
220 return NGX_HTTP_INTERNAL_SERVER_ERROR;
221 }
222
223 errbuf[0] = '\0';
224 curl_easy_setopt(ctx->curl, CURLOPT_ERRORBUFFER, errbuf);
225 curl_easy_setopt(ctx->curl, CURLOPT_NOSIGNAL, 1L);
226 curl_easy_setopt(ctx->curl, CURLOPT_FAILONERROR, 1L);
227
228#ifdef CURLOPT_XFERINFOFUNCTION
229 if (ctx->transfer_id != 0) {
230 ngx_memzero(progress, sizeof(*progress));
231 progress->transfer_id = ctx->transfer_id;
232 progress->log = ctx->log;
233 progress->is_push = ctx->is_push;
234 curl_easy_setopt(ctx->curl, CURLOPT_NOPROGRESS, 0L);
235 curl_easy_setopt(ctx->curl, CURLOPT_XFERINFOFUNCTION,
236 webdav_tpc_curl_progress);
237 curl_easy_setopt(ctx->curl, CURLOPT_XFERINFODATA, progress);
238 }
239#endif
240
241 /* Restrict to HTTPS only — equivalent to curl --proto =https */
242#ifdef CURLOPT_PROTOCOLS_STR
243 curl_easy_setopt(ctx->curl, CURLOPT_PROTOCOLS_STR, "https");
244#else
245 curl_easy_setopt(ctx->curl, CURLOPT_PROTOCOLS, (long) CURLPROTO_HTTPS);
246#endif
247
248 curl_easy_setopt(ctx->curl, CURLOPT_URL, ctx->url);
249
250 /* TLS verification + DNS-rebind pin (W2). A prohibited/rebind target or
251 * resolve failure aborts the transfer with 403 rather than connecting. */
252 if (tpc_curl_secure(ctx->curl, ctx->conf, ctx->url, ctx->log,
253 &ctx->resolve) < 0) {
254 return NGX_HTTP_FORBIDDEN;
255 }
256
257 if (ctx->conf->tpc_timeout > 0) {
258 curl_easy_setopt(ctx->curl, CURLOPT_TIMEOUT,
259 (long) ctx->conf->tpc_timeout);
260 }
261 tpc_curl_apply_stall_bounds(ctx->curl, ctx->conf); /* Phase 39 (WS4) */
262
263 tpc_core_apply_tls_creds(ctx);
264
265 if (tpc_core_append_headers(ctx) != NGX_OK) {
266 return NGX_HTTP_INTERNAL_SERVER_ERROR;
267 }
268
269 /* Open the local endpoint through the confined resolver (openat2
270 * RESOLVE_BENEATH + O_NOFOLLOW): the push side reads an export path supplied
271 * by the client, so a raw fopen() would follow a planted in-export symlink
272 * out of the export root. The pull side targets the staged temp; confining
273 * it too costs nothing and removes the absolute-path re-open TOCTOU. */
274 if (tpc_core_open_local(ctx) != NGX_OK) {
275 return NGX_HTTP_INTERNAL_SERVER_ERROR;
276 }
277
278 return NGX_OK;
279}

282/*
283 * tpc_core_perform — run curl_easy_perform and map its result to an rc.
284 *
285 * WHAT: executes the configured transfer and returns NGX_OK on CURLE_OK,
286 * NGX_HTTP_BAD_GATEWAY otherwise (logging errbuf/strerror).
287 * WHY: keeps the perform + result-mapping out of the orchestrator body.
288 * HOW: errbuf is the same buffer installed in setup; unchanged behavior.
289 */
290static ngx_int_t
291tpc_core_perform(webdav_tpc_curl_ctx_t *ctx, const char *errbuf)
292{
293 CURLcode res = curl_easy_perform(ctx->curl);
294
295 if (res == CURLE_OK) {
296 return NGX_OK;
297 }
298 ngx_log_error(NGX_LOG_WARN, ctx->log, 0,
299 "brix_webdav: HTTP-TPC %s failed: %s",
300 ctx->log_tag, errbuf[0] ? errbuf : curl_easy_strerror(res));
301 return NGX_HTTP_BAD_GATEWAY;
302}

305/*
306 * webdav_tpc_run_curl_core — perform a single HTTP transfer using libcurl.
307 *
308 * @is_push: 0 = pull (GET remote → write to file_path)
309 * 1 = push (read from file_path → PUT to url)
310 * @file_path: local file to write (pull) or read (push)
311 * @url: remote HTTPS URL to fetch from (pull) or PUT to (push)
312 * @log_tag: short label for log messages, e.g. "pull" or "push"
313 *
314 * Reads as setup → perform → finish: tpc_core_setup() configures the handle
315 * (returning the mapped HTTP error on any failure), tpc_core_perform() runs the
316 * transfer, and webdav_tpc_curl_finish() tears down every resource on all exits.
317 * The 8-param signature is frozen by tpc_curl_internal.h; state is threaded
318 * through a file-local webdav_tpc_curl_ctx_t instead.
319 */
320ngx_int_t
321webdav_tpc_run_curl_core(ngx_log_t *log,
322 ngx_http_brix_webdav_loc_conf_t *conf,
323 ngx_array_t *transfer_headers,
324 int is_push,
325 const char *file_path,
326 const char *url,
327 const char *log_tag,
328 uint64_t transfer_id,
329 const char *user_cert,
330 const char *user_key)
331{
332 webdav_tpc_curl_ctx_t ctx;
333 char errbuf[CURL_ERROR_SIZE];
334 ngx_int_t rc;
335#ifdef CURLOPT_XFERINFOFUNCTION
336 webdav_tpc_curl_progress_t progress;
337#endif
338#ifdef WEBDAV_TPC_PMARK_SOCKCB
339 webdav_tpc_pmark_rec_t pmrec; /* outlives curl_easy_perform() below */
340#endif
341
342 ngx_memzero(&ctx, sizeof(ctx));
343 ctx.log = log;
344 ctx.conf = conf;
345 ctx.transfer_headers = transfer_headers;
346 ctx.is_push = is_push;
347 ctx.file_path = file_path;
348 ctx.url = url;
349 ctx.log_tag = log_tag;
350 ctx.transfer_id = transfer_id;
351 ctx.user_cert = user_cert;
352 ctx.user_key = user_key;
353
354 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_STARTED]);
355
356 rc = tpc_core_setup(&ctx, errbuf
357#ifdef CURLOPT_XFERINFOFUNCTION
358 , &progress
359#endif
360 );
361 if (rc != NGX_OK) {
362 return webdav_tpc_curl_finish(rc, ctx.curl, ctx.hdrs, ctx.resolve,
363 ctx.fp);
364 }
365
366#ifdef WEBDAV_TPC_PMARK_SOCKCB
367 /* SciTags: mark the outbound data socket (flow label at open, firefly at
368 * close). pmrec lives until curl_easy_perform() returns. */
369 webdav_tpc_pmark_attach(ctx.curl, &pmrec, conf, is_push, file_path, log);
370#endif
371
372 rc = tpc_core_perform(&ctx, errbuf);
373
374 return webdav_tpc_curl_finish(rc, ctx.curl, ctx.hdrs, ctx.resolve, ctx.fp);
375}

407/*
408 * webdav_tpc_run_curl_multi_finish — tear down a curl_multi run and record the
409 * success/error metric, returning rc unchanged.
410 *
411 * Called at every exit of webdav_tpc_run_curl_pull_multi (success and each early
412 * failure). The easy[], hdrs[] and resolve[] arrays start zeroed, so iterating
413 * all n_streams slots frees exactly the handles that were set up before the exit
414 * — letting the driver exit with a plain `return` instead of a shared
415 * goto/label. cm is created before the first failure site, so it is always live.
416 */
417ngx_int_t
418webdav_tpc_run_curl_multi_finish(ngx_int_t rc, CURLM *cm, CURL **easy,
419 struct curl_slist **hdrs, struct curl_slist **resolve,
420 ngx_uint_t n_streams, int fd)
421{
422 ngx_uint_t i;
423
424 for (i = 0; i < n_streams; i++) {
425 if (easy[i] != NULL) {
426 curl_multi_remove_handle(cm, easy[i]);
427 curl_easy_cleanup(easy[i]);
428 }
429 if (hdrs[i] != NULL) {
430 curl_slist_free_all(hdrs[i]);
431 }
432 if (resolve[i] != NULL) {
433 curl_slist_free_all(resolve[i]);
434 }
435 }
436 curl_multi_cleanup(cm);
437 close(fd);
438
439 if (rc == NGX_OK) {
440 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_SUCCESS]);
441 } else {
442 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_ERROR]);
443 }
444 return rc;
445}

448/*
449 * tpc_ms_prepare_output — create and pre-size the multi-stream assembly temp.
450 *
451 * WHAT: opens tmp_path O_CREAT|O_TRUNC and ftruncates it to total_size so every
452 * stream can pwrite its disjoint range in place.
453 * WHY: factors the fd lifecycle (open + failure-close on ftruncate) out of the
454 * driver so the driver never owns a half-set-up fd.
455 * HOW: returns the ready fd (>= 0) on success; -1 on open or ftruncate
456 * failure (any opened fd is closed before returning).
457 */
458static int
459tpc_ms_prepare_output(ngx_log_t *log, const char *tmp_path, off_t total_size)
460{
461 int fd;
462 brix_sd_obj_t obj;
463
464 /* Pre-create the output file at full size so all streams can pwrite. */
465 fd = open(tmp_path, O_WRONLY | O_CREAT | O_TRUNC | O_NOFOLLOW, 0600); /* vfs-seam-allow: TPC multi-stream assembly temp (committed via rename); O_NOFOLLOW matches the sibling staging opens */
466 if (fd < 0) {
467 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
468 "brix_webdav: multi-stream: open(\"%s\") failed",
469 tmp_path);
470 return -1;
471 }
472
473 brix_sd_posix_wrap(&obj, fd);
474 if (brix_sd_posix_driver.ftruncate(&obj, total_size) != NGX_OK) {
475 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
476 "brix_webdav: multi-stream: ftruncate failed");
477 close(fd);
478 return -1;
479 }
480 return fd;
481}

488/*
489 * tpc_ms_setup_stream — configure one range-covering easy handle and add it to
490 * the multi handle.
491 *
492 * WHAT: computes stream i's byte range, initialises its write context, creates
493 * and configures the easy handle (conf/egress + Range + write cb + pmark),
494 * and registers it with ctx->cm.
495 * WHY: isolates the per-stream construction (with its two distinct early-fail
496 * codes) so the driver loop stays flat.
497 * HOW: writes the handle/slists into ctx->easy[i]/hdrs[i]/resolve[i] and the
498 * write context into ctx->write_ctx[i]. Returns NGX_OK, or the mapped
499 * HTTP status (INTERNAL on curl_easy_init OOM, FORBIDDEN on egress/OOM) —
500 * the caller tears down all slots via webdav_tpc_run_curl_multi_finish().
501 */
502static ngx_int_t
503tpc_ms_setup_stream(webdav_tpc_ms_ctx_t *ctx, ngx_uint_t i)
504{
505 off_t chunk = ctx->total_size / (off_t) ctx->n_streams;
506 off_t start = (off_t) i * chunk;
507 off_t end = (i == ctx->n_streams - 1)
508 ? ctx->total_size - 1 : start + chunk - 1;
509 char range_buf[64];
510
511 ctx->write_ctx[i].fd = ctx->fd;
512 ctx->write_ctx[i].cur_offset = start;
513 ctx->write_ctx[i].stream_idx = i;
514 ctx->write_ctx[i].progress = ctx->progress;
515
516 ctx->easy[i] = curl_easy_init();
517 if (ctx->easy[i] == NULL) {
518 return NGX_HTTP_INTERNAL_SERVER_ERROR;
519 }
520
521 if (tpc_curl_apply_conf(ctx->easy[i], ctx->conf, ctx->source_url,
522 ctx->transfer_headers, ctx->log,
523 ctx->user_cert, ctx->user_key,
524 &ctx->hdrs[i], &ctx->resolve[i]) < 0) {
525 /* Egress check failed (prohibited/rebind) or OOM — abort. */
526 return NGX_HTTP_FORBIDDEN;
527 }
528
529 snprintf(range_buf, sizeof(range_buf), "%lld-%lld",
530 (long long) start, (long long) end);
531 curl_easy_setopt(ctx->easy[i], CURLOPT_RANGE, range_buf);
532 curl_easy_setopt(ctx->easy[i], CURLOPT_WRITEFUNCTION, ms_write_cb);
533 curl_easy_setopt(ctx->easy[i], CURLOPT_WRITEDATA, &ctx->write_ctx[i]);
534
535#ifdef WEBDAV_TPC_PMARK_SOCKCB
536 /* SciTags: mark each parallel pull stream's outbound socket. pmrec[]
537 * lives until the curl_multi loop + cleanup below. */
538 webdav_tpc_pmark_attach(ctx->easy[i], &ctx->pmrec[i], ctx->conf, 0 /*pull*/,
539 ctx->tmp_path, ctx->log);
540#endif
541
542 curl_multi_add_handle(ctx->cm, ctx->easy[i]);
543 return NGX_OK;
544}

547/*
548 * tpc_ms_drive — pump all handles on cm to completion.
549 *
550 * WHAT: runs the curl_multi_perform / curl_multi_wait loop until no handle is
551 * still running.
552 * WHY: separates the event pump (with its wait-error early exit) from result
553 * harvesting.
554 * HOW: NGX_OK when the loop drains normally; NGX_HTTP_BAD_GATEWAY on a
555 * curl_multi_wait error.
556 */
557static ngx_int_t
558tpc_ms_drive(ngx_log_t *log, CURLM *cm)
559{
560 int still_running;
561
562 curl_multi_perform(cm, &still_running);
563 while (still_running) {
564 int numfds;
565 CURLMcode mc = curl_multi_wait(cm, NULL, 0, 1000, &numfds);
566 if (mc != CURLM_OK) {
567 ngx_log_error(NGX_LOG_ERR, log, 0,
568 "brix_webdav: curl_multi_wait error: %s",
569 curl_multi_strerror(mc));
570 return NGX_HTTP_BAD_GATEWAY;
571 }
572 curl_multi_perform(cm, &still_running);
573 }
574 return NGX_OK;
575}

578/*
579 * tpc_ms_harvest — drain completion messages; any stream error fails the run.
580 *
581 * WHAT: reads CURLMSG_DONE messages and maps the first per-stream failure to
582 * NGX_HTTP_BAD_GATEWAY (all messages are still drained/logged).
583 * WHY: keeps result inspection out of the driver body.
584 * HOW: NGX_OK when every stream reported CURLE_OK; NGX_HTTP_BAD_GATEWAY
585 * otherwise.
586 */
587static ngx_int_t
588tpc_ms_harvest(ngx_log_t *log, CURLM *cm)
589{
590 CURLMsg *msg;
591 int msgs_left;
592 ngx_int_t rc = NGX_OK;
593
594 while ((msg = curl_multi_info_read(cm, &msgs_left)) != NULL) {
595 if (msg->msg == CURLMSG_DONE && msg->data.result != CURLE_OK) {
596 ngx_log_error(NGX_LOG_WARN, log, 0,
597 "brix_webdav: HTTP-TPC multi-stream failed: %s",
598 curl_easy_strerror(msg->data.result));
599 rc = NGX_HTTP_BAD_GATEWAY;
600 }
601 }
602 return rc;
603}

606/*
607 * webdav_tpc_run_curl_pull_multi — parallel Range-based GET using curl_multi.
608 *
609 * Issues a HEAD request to learn Content-Length. On success, divides the file
610 * into n_streams equal-sized byte ranges and runs them concurrently via
611 * curl_multi_perform. Each stream writes to the output file via pwrite at its
612 * own offset so the file is assembled in-place without post-merge overhead.
613 *
614 * Falls back to single-stream when Content-Length is unknown (chunked source or
615 * HEAD failure) or when n_streams <= 1.
616 *
617 * progress may be NULL when called from the non-marker thread path. When
618 * non-NULL, each stream's write callback increments bytes_per_stream[i]
619 * atomically so the event-loop poll timer can report per-stream progress.
620 *
621 * Body reads as: fall back → prepare output → set up N streams → drive →
622 * harvest → finish. Every exit past cm-init routes through
623 * webdav_tpc_run_curl_multi_finish() so exactly the set-up handles are freed.
624 * The 8-param signature is frozen (out-of-file callers in tpc_marker.c /
625 * tpc_thread.c) and reported as a residual.
626 */
627ngx_int_t
628webdav_tpc_run_curl_pull_multi(ngx_log_t *log,
629 ngx_http_brix_webdav_loc_conf_t *conf,
630 const char *source_url, const char *tmp_path,
631 ngx_array_t *transfer_headers, ngx_uint_t n_streams,
632 uint64_t transfer_id, tpc_ms_progress_t *progress,
633 const char *user_cert, const char *user_key)
634{
635 off_t total_size;
636 CURLM *cm = NULL;
637 CURL *easy[BRIX_TPC_MAX_STREAMS];
638 struct curl_slist *hdrs[BRIX_TPC_MAX_STREAMS];
639 struct curl_slist *resolve[BRIX_TPC_MAX_STREAMS];
640 ms_stream_ctx_t write_ctx[BRIX_TPC_MAX_STREAMS];
641#ifdef WEBDAV_TPC_PMARK_SOCKCB
642 webdav_tpc_pmark_rec_t pmrec[BRIX_TPC_MAX_STREAMS];
643#endif
644 webdav_tpc_ms_ctx_t msctx;
645 int fd = -1;
646 ngx_uint_t i;
647 ngx_int_t rc = NGX_OK;
648
649 if (n_streams <= 1) {
650 return webdav_tpc_run_curl_pull(log, conf, source_url, tmp_path,
651 transfer_headers, transfer_id,
652 user_cert, user_key);
653 }
654 if (n_streams > BRIX_TPC_MAX_STREAMS) {
655 n_streams = BRIX_TPC_MAX_STREAMS;
656 }
657
658 /* HEAD to learn file size so we can split into ranges. */
659 total_size = tpc_curl_head_size(log, conf, source_url, transfer_headers,
660 user_cert, user_key);
661 if (total_size <= 0) {
662 ngx_log_error(NGX_LOG_INFO, log, 0,
663 "brix_webdav: multi-stream: unknown Content-Length,"
664 " falling back to single stream");
665 return webdav_tpc_run_curl_pull(log, conf, source_url, tmp_path,
666 transfer_headers, transfer_id,
667 user_cert, user_key);
668 }
669
670 if (progress != NULL) {
671 progress->total_size = total_size;
672 }
673
674 fd = tpc_ms_prepare_output(log, tmp_path, total_size);
675 if (fd < 0) {
676 return NGX_HTTP_INTERNAL_SERVER_ERROR;
677 }
678
679 cm = curl_multi_init();
680 if (cm == NULL) {
681 close(fd);
682 return NGX_HTTP_INTERNAL_SERVER_ERROR;
683 }
684
685 ngx_memzero(easy, sizeof(easy));
686 ngx_memzero(hdrs, sizeof(hdrs));
687 ngx_memzero(resolve, sizeof(resolve));
688
689 ngx_memzero(&msctx, sizeof(msctx));
690 msctx.log = log;
691 msctx.conf = conf;
692 msctx.source_url = source_url;
693 msctx.tmp_path = tmp_path;
694 msctx.transfer_headers = transfer_headers;
695 msctx.user_cert = user_cert;
696 msctx.user_key = user_key;
697 msctx.n_streams = n_streams;
698 msctx.total_size = total_size;
699 msctx.fd = fd;
700 msctx.cm = cm;
701 msctx.easy = easy;
702 msctx.hdrs = hdrs;
703 msctx.resolve = resolve;
704 msctx.write_ctx = write_ctx;
705 msctx.progress = progress;
706#ifdef WEBDAV_TPC_PMARK_SOCKCB
707 msctx.pmrec = pmrec;
708#endif
709
710 /* Set up N easy handles, each covering a disjoint byte range. */
711 for (i = 0; i < n_streams; i++) {
712 rc = tpc_ms_setup_stream(&msctx, i);
713 if (rc != NGX_OK) {
714 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs,
715 resolve, n_streams, fd);
716 }
717 }
718
719 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_STARTED]);
720
721 rc = tpc_ms_drive(log, cm);
722 if (rc != NGX_OK) {
723 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs,
724 resolve, n_streams, fd);
725 }
726
727 rc = tpc_ms_harvest(log, cm);
728
729 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs, resolve,
730 n_streams, fd);
731}
