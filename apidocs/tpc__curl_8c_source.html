9/*
10 * webdav_tpc_run_curl_core — perform a single HTTP transfer using libcurl.
11 *
12 * @is_push: 0 = pull (GET remote → write to file_path)
13 * 1 = push (read from file_path → PUT to url)
14 * @file_path: local file to write (pull) or read (push)
15 * @url: remote HTTPS URL to fetch from (pull) or PUT to (push)
16 * @log_tag: short label for log messages, e.g. "pull" or "push"
17 */
18ngx_int_t
19webdav_tpc_run_curl_core(ngx_log_t *log,
20 ngx_http_brix_webdav_loc_conf_t *conf,
21 ngx_array_t *transfer_headers,
22 int is_push,
23 const char *file_path,
24 const char *url,
25 const char *log_tag,
26 uint64_t transfer_id)
27{
28 CURL *curl = NULL;
29 struct curl_slist *hdrs = NULL;
30 struct curl_slist *resolve = NULL;
31 CURLcode res;
32 FILE *fp = NULL;
33 char errbuf[CURL_ERROR_SIZE];
34 ngx_int_t rc = NGX_HTTP_INTERNAL_SERVER_ERROR;
35 ngx_uint_t i;
36 ngx_str_t *headers;
37#ifdef CURLOPT_XFERINFOFUNCTION
38 webdav_tpc_curl_progress_t progress;
39#endif
40#ifdef WEBDAV_TPC_PMARK_SOCKCB
41 webdav_tpc_pmark_rec_t pmrec; /* outlives curl_easy_perform() below */
42#endif
43
44 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_STARTED]);
45
46 curl = curl_easy_init();
47 if (curl == NULL) {
48 ngx_log_error(NGX_LOG_ERR, log, 0,
49 "brix_webdav: curl_easy_init() failed for TPC %s",
50 log_tag);
51 return webdav_tpc_curl_finish(rc, curl, hdrs, resolve, fp);
52 }
53
54 errbuf[0] = '\0';
55 curl_easy_setopt(curl, CURLOPT_ERRORBUFFER, errbuf);
56 curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
57 curl_easy_setopt(curl, CURLOPT_FAILONERROR, 1L);
58
59#ifdef CURLOPT_XFERINFOFUNCTION
60 if (transfer_id != 0) {
61 ngx_memzero(&progress, sizeof(progress));
62 progress.transfer_id = transfer_id;
63 progress.log = log;
64 progress.is_push = is_push;
65 curl_easy_setopt(curl, CURLOPT_NOPROGRESS, 0L);
66 curl_easy_setopt(curl, CURLOPT_XFERINFOFUNCTION,
67 webdav_tpc_curl_progress);
68 curl_easy_setopt(curl, CURLOPT_XFERINFODATA, &progress);
69 }
70#endif
71
72 /* Restrict to HTTPS only — equivalent to curl --proto =https */
73#ifdef CURLOPT_PROTOCOLS_STR
74 curl_easy_setopt(curl, CURLOPT_PROTOCOLS_STR, "https");
75#else
76 curl_easy_setopt(curl, CURLOPT_PROTOCOLS, (long) CURLPROTO_HTTPS);
77#endif

79 curl_easy_setopt(curl, CURLOPT_URL, url);
80
81 /* TLS verification + DNS-rebind pin (W2). A prohibited/rebind target or
82 * resolve failure aborts the transfer with 403 rather than connecting. */
83 if (tpc_curl_secure(curl, conf, url, log, &resolve) < 0) {
84 rc = NGX_HTTP_FORBIDDEN;
85 return webdav_tpc_curl_finish(rc, curl, hdrs, resolve, fp);
86 }
87
88 if (conf->tpc_timeout > 0) {
89 curl_easy_setopt(curl, CURLOPT_TIMEOUT, (long) conf->tpc_timeout);
90 }
91 tpc_curl_apply_stall_bounds(curl, conf); /* Phase 39 (WS4) */
92
93 if (conf->tpc_cert.len > 0) {
94 curl_easy_setopt(curl, CURLOPT_SSLCERT,
95 (const char *) conf->tpc_cert.data);
96 }
97 if (conf->tpc_key.len > 0) {
98 curl_easy_setopt(curl, CURLOPT_SSLKEY,
99 (const char *) conf->tpc_key.data);
100 }
101 if (conf->tpc_cafile.len > 0) {

103 (const char *) conf->tpc_cafile.data);
104 }
105 if (conf->tpc_cadir.len > 0) {
106 curl_easy_setopt(curl, CURLOPT_CAPATH,
107 (const char *) conf->tpc_cadir.data);
108 }
109
110 if (transfer_headers != NULL && transfer_headers->nelts > 0) {
111 headers = transfer_headers->elts;
112 for (i = 0; i < transfer_headers->nelts; i++) {
113 struct curl_slist *next;
114 next = curl_slist_append(hdrs, (const char *) headers[i].data);
115 if (next == NULL) {
116 ngx_log_error(NGX_LOG_ERR, log, 0,
117 "brix_webdav: curl_slist_append() OOM for TPC %s",
118 log_tag);
119 return webdav_tpc_curl_finish(rc, curl, hdrs, resolve, fp);

121 hdrs = next;
122 }
123 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, hdrs);

126 /* Open the local endpoint through the confined resolver (openat2
127 * RESOLVE_BENEATH + O_NOFOLLOW): the push side reads an export path supplied
128 * by the client, so a raw fopen() would follow a planted in-export symlink
129 * out of the export root. The pull side targets the staged temp; confining
130 * it too costs nothing and removes the absolute-path re-open TOCTOU. */
131 if (is_push) {
132 struct stat st;
133 int fd;
134
135 fd = brix_vfs_open_fd(log, conf->common.root_canon, file_path,
136 O_RDONLY | O_CLOEXEC | O_NOFOLLOW, 0);
137 if (fd >= 0) {
138 fp = fdopen(fd, "rb");
139 if (fp == NULL) {
140 (void) close(fd);
141 }
142 }
143 if (fp == NULL) {
144 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
145 "brix_webdav: TPC push open(\"%s\") failed",
146 file_path);
147 return webdav_tpc_curl_finish(rc, curl, hdrs, resolve, fp);
148 }
149 if (fstat(fileno(fp), &st) == 0) {
150 curl_easy_setopt(curl, CURLOPT_INFILESIZE_LARGE,
151 (curl_off_t) st.st_size);
152 }
153 curl_easy_setopt(curl, CURLOPT_UPLOAD, 1L);
154 curl_easy_setopt(curl, CURLOPT_READDATA, fp);
155 } else {
156 int fd;
157
158 fd = brix_vfs_open_fd(log, conf->common.root_canon, file_path,
159 O_WRONLY | O_CREAT | O_TRUNC
160 | O_CLOEXEC | O_NOFOLLOW,
161 0600);
162 if (fd >= 0) {
163 fp = fdopen(fd, "wb");
164 if (fp == NULL) {
165 (void) close(fd);
166 }
167 }
168 if (fp == NULL) {
169 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
170 "brix_webdav: TPC pull open(\"%s\") failed",
171 file_path);
172 return webdav_tpc_curl_finish(rc, curl, hdrs, resolve, fp);
173 }
174 curl_easy_setopt(curl, CURLOPT_WRITEDATA, fp);
175 }
176
177#ifdef WEBDAV_TPC_PMARK_SOCKCB
178 /* SciTags: mark the outbound data socket (flow label at open, firefly at
179 * close). pmrec lives until curl_easy_perform() returns. */
180 webdav_tpc_pmark_attach(curl, &pmrec, conf, is_push, file_path, log);
181#endif
182
183 res = curl_easy_perform(curl);
184 if (res == CURLE_OK) {
185 rc = NGX_OK;
186 } else {
187 ngx_log_error(NGX_LOG_WARN, log, 0,
188 "brix_webdav: HTTP-TPC %s failed: %s",
189 log_tag, errbuf[0] ? errbuf : curl_easy_strerror(res));
190 rc = NGX_HTTP_BAD_GATEWAY;
191 }
192
193 return webdav_tpc_curl_finish(rc, curl, hdrs, resolve, fp);
194}

223/*
224 * webdav_tpc_run_curl_multi_finish — tear down a curl_multi run and record the
225 * success/error metric, returning rc unchanged.
226 *
227 * Called at every exit of webdav_tpc_run_curl_pull_multi (success and each early
228 * failure). The easy[], hdrs[] and resolve[] arrays start zeroed, so iterating
229 * all n_streams slots frees exactly the handles that were set up before the exit
230 * — letting the driver exit with a plain `return` instead of a shared
231 * goto/label. cm is created before the first failure site, so it is always live.
232 */
233ngx_int_t
234webdav_tpc_run_curl_multi_finish(ngx_int_t rc, CURLM *cm, CURL **easy,
235 struct curl_slist **hdrs, struct curl_slist **resolve,
236 ngx_uint_t n_streams, int fd)
237{
238 ngx_uint_t i;
239
240 for (i = 0; i < n_streams; i++) {
241 if (easy[i] != NULL) {
242 curl_multi_remove_handle(cm, easy[i]);
243 curl_easy_cleanup(easy[i]);
244 }
245 if (hdrs[i] != NULL) {
246 curl_slist_free_all(hdrs[i]);
247 }
248 if (resolve[i] != NULL) {
249 curl_slist_free_all(resolve[i]);
250 }
251 }
252 curl_multi_cleanup(cm);
253 close(fd);
254
255 if (rc == NGX_OK) {
256 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_SUCCESS]);
257 } else {
258 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_ERROR]);
259 }
260 return rc;
261}

264/*
265 * webdav_tpc_run_curl_pull_multi — parallel Range-based GET using curl_multi.
266 *
267 * Issues a HEAD request to learn Content-Length. On success, divides the file
268 * into n_streams equal-sized byte ranges and runs them concurrently via
269 * curl_multi_perform. Each stream writes to the output file via pwrite at its
270 * own offset so the file is assembled in-place without post-merge overhead.
271 *
272 * Falls back to single-stream when Content-Length is unknown (chunked source or
273 * HEAD failure) or when n_streams <= 1.
274 *
275 * progress may be NULL when called from the non-marker thread path. When
276 * non-NULL, each stream's write callback increments bytes_per_stream[i]
277 * atomically so the event-loop poll timer can report per-stream progress.
278 */
279ngx_int_t
280webdav_tpc_run_curl_pull_multi(ngx_log_t *log,
281 ngx_http_brix_webdav_loc_conf_t *conf,
282 const char *source_url, const char *tmp_path,
283 ngx_array_t *transfer_headers, ngx_uint_t n_streams,
284 uint64_t transfer_id, tpc_ms_progress_t *progress)
285{
286 off_t total_size;
287 CURLM *cm = NULL;
288 CURL *easy[BRIX_TPC_MAX_STREAMS];
289 struct curl_slist *hdrs[BRIX_TPC_MAX_STREAMS];
290 struct curl_slist *resolve[BRIX_TPC_MAX_STREAMS];
291 ms_stream_ctx_t write_ctx[BRIX_TPC_MAX_STREAMS];
292#ifdef WEBDAV_TPC_PMARK_SOCKCB
293 webdav_tpc_pmark_rec_t pmrec[BRIX_TPC_MAX_STREAMS];
294#endif
295 int fd = -1;
296 ngx_uint_t i;
297 int still_running;
298 ngx_int_t rc = NGX_OK;
299
300 if (n_streams <= 1) {
301 return webdav_tpc_run_curl_pull(log, conf, source_url, tmp_path,
302 transfer_headers, transfer_id);
303 }
304 if (n_streams > BRIX_TPC_MAX_STREAMS) {
305 n_streams = BRIX_TPC_MAX_STREAMS;
306 }
307
308 /* HEAD to learn file size so we can split into ranges. */
309 total_size = tpc_curl_head_size(log, conf, source_url, transfer_headers);
310 if (total_size <= 0) {
311 ngx_log_error(NGX_LOG_INFO, log, 0,
312 "brix_webdav: multi-stream: unknown Content-Length,"
313 " falling back to single stream");
314 return webdav_tpc_run_curl_pull(log, conf, source_url, tmp_path,
315 transfer_headers, transfer_id);
316 }
317
318 if (progress != NULL) {
319 progress->total_size = total_size;
320 }
321
322 /* Pre-create the output file at full size so all streams can pwrite. */
323 fd = open(tmp_path, O_WRONLY | O_CREAT | O_TRUNC, 0600); /* vfs-seam-allow: TPC multi-stream assembly temp (committed via rename) */
324 if (fd < 0) {
325 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
326 "brix_webdav: multi-stream: open(\"%s\") failed",
327 tmp_path);
328 return NGX_HTTP_INTERNAL_SERVER_ERROR;
329 }
330 {
331 brix_sd_obj_t obj;
332 brix_sd_posix_wrap(&obj, fd);
333 if (brix_sd_posix_driver.ftruncate(&obj, total_size) != NGX_OK) {
334 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
335 "brix_webdav: multi-stream: ftruncate failed");
336 close(fd);
337 return NGX_HTTP_INTERNAL_SERVER_ERROR;
338 }
339 }
340
341 cm = curl_multi_init();
342 if (cm == NULL) {
343 close(fd);
344 return NGX_HTTP_INTERNAL_SERVER_ERROR;
345 }
346
347 ngx_memzero(easy, sizeof(easy));
348 ngx_memzero(hdrs, sizeof(hdrs));
349 ngx_memzero(resolve, sizeof(resolve));
350
351 /* Set up N easy handles, each covering a disjoint byte range. */
352 for (i = 0; i < n_streams; i++) {
353 off_t chunk = total_size / (off_t) n_streams;
354 off_t start = (off_t) i * chunk;
355 off_t end = (i == n_streams - 1) ? total_size - 1 : start + chunk - 1;
356 char range_buf[64];
357
358 write_ctx[i].fd = fd;
359 write_ctx[i].cur_offset = start;
360 write_ctx[i].stream_idx = i;
361 write_ctx[i].progress = progress;
362
363 easy[i] = curl_easy_init();
364 if (easy[i] == NULL) {
365 rc = NGX_HTTP_INTERNAL_SERVER_ERROR;
366 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs,
367 resolve, n_streams, fd);
368 }
369
370 if (tpc_curl_apply_conf(easy[i], conf, source_url, transfer_headers,
371 log, &hdrs[i], &resolve[i]) < 0) {
372 /* Egress check failed (prohibited/rebind) or OOM — abort. */
373 rc = NGX_HTTP_FORBIDDEN;
374 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs,
375 resolve, n_streams, fd);
376 }
377
378 snprintf(range_buf, sizeof(range_buf), "%lld-%lld",
379 (long long) start, (long long) end);
380 curl_easy_setopt(easy[i], CURLOPT_RANGE, range_buf);
381 curl_easy_setopt(easy[i], CURLOPT_WRITEFUNCTION, ms_write_cb);
382 curl_easy_setopt(easy[i], CURLOPT_WRITEDATA, &write_ctx[i]);
383
384#ifdef WEBDAV_TPC_PMARK_SOCKCB
385 /* SciTags: mark each parallel pull stream's outbound socket. pmrec[]
386 * lives until the curl_multi loop + cleanup below. */
387 webdav_tpc_pmark_attach(easy[i], &pmrec[i], conf, 0 /*pull*/,
388 tmp_path, log);
389#endif
390
391 curl_multi_add_handle(cm, easy[i]);
392 }
393
394 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_STARTED]);
395
396 /* Drive all handles to completion. */
397 curl_multi_perform(cm, &still_running);
398 while (still_running) {
399 int numfds;
400 CURLMcode mc = curl_multi_wait(cm, NULL, 0, 1000, &numfds);
401 if (mc != CURLM_OK) {
402 ngx_log_error(NGX_LOG_ERR, log, 0,
403 "brix_webdav: curl_multi_wait error: %s",
404 curl_multi_strerror(mc));
405 rc = NGX_HTTP_BAD_GATEWAY;
406 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs,
407 resolve, n_streams, fd);
408 }
409 curl_multi_perform(cm, &still_running);
410 }
411
412 /* Harvest results — any per-stream failure fails the whole transfer. */
413 {
414 CURLMsg *msg;
415 int msgs_left;
416 while ((msg = curl_multi_info_read(cm, &msgs_left)) != NULL) {
417 if (msg->msg == CURLMSG_DONE
418 && msg->data.result != CURLE_OK)
419 {
420 ngx_log_error(NGX_LOG_WARN, log, 0,
421 "brix_webdav: HTTP-TPC multi-stream failed: %s",
422 curl_easy_strerror(msg->data.result));
423 rc = NGX_HTTP_BAD_GATEWAY;
424 }
425 }
426 }
427
428 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs, resolve,
429 n_streams, fd);
430}
