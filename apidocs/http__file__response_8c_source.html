4 * WebDAV and S3 both serve local files through nginx's file-backed buffers.

18/*
19 * brix_http_chain_append_file_range - append one file-backed ngx_buf_t to a chain.
20 *
21 * Allocates ngx_buf_t + ngx_file_t from r->pool, sets in_file / file_pos / file_last
22 * for [start, end] (inclusive), links it onto the chain tail, and registers a pool
23 * cleanup to close fd if close_fd=1.
24 */
25
26ngx_int_t
27brix_http_chain_append_file_range(ngx_http_request_t *r,
28 ngx_chain_t **tail, ngx_fd_t fd, const char *path,
29 off_t start, off_t end, ngx_flag_t close_fd)
30{
31 ngx_buf_t *b;
32 ngx_chain_t *link;
33 ngx_pool_cleanup_t *cln;
34 ngx_pool_cleanup_file_t *clnf;
35 size_t path_len;
36
37 BRIX_PCALLOC_OR_RETURN(b, r->pool, sizeof(ngx_buf_t), NGX_ERROR);
38 b->file = ngx_pcalloc(r->pool, sizeof(ngx_file_t));
39 if (b->file == NULL) {
40 return NGX_ERROR;
41 }
42
43 path_len = ngx_strlen(path);
44 b->file->name.data = ngx_pnalloc(r->pool, path_len + 1);
45 if (b->file->name.data == NULL) {
46 return NGX_ERROR;
47 }
48 ngx_cpystrn(b->file->name.data, (u_char *) path, path_len + 1);
49 b->file->name.len = path_len;
50 b->file->fd = fd;
51 b->file->log = r->connection->log;
52
53 b->file_pos = start;
54 b->file_last = end + 1;
55 b->in_file = 1;
56
57 if (close_fd) {
58 cln = ngx_pool_cleanup_add(r->pool, sizeof(ngx_pool_cleanup_file_t));
59 if (cln == NULL) {
60 return NGX_ERROR;
61 }
62 cln->handler = ngx_pool_cleanup_file;
63 clnf = cln->data;
64 clnf->fd = fd;
65 clnf->name = b->file->name.data;
66 clnf->log = r->pool->log;
67 }
68
69 link = ngx_alloc_chain_link(r->pool);
70 if (link == NULL) {
71 return NGX_ERROR;
72 }
73 link->buf = b;
74 link->next = NULL;
75
76 if (*tail != NULL) {
77 (*tail)->next = link;
78 }
79 *tail = link;
80
81 return NGX_OK;
82}

84/*
85 * brix_http_add_etag_header - generate ETag from mtime/size and set as response header.
86 *
87 * WHAT: Calls brix_http_etag_str() to format ETag string, then brix_http_set_header()
88 * to insert "ETag" into r->headers_out. Optionally registers h as
89 * r->headers_out.etag for later conditional-request comparison (If-None-Match).
90 *
91 * WHY: RFC 7232 §2.3 requires servers to emit ETags on responses with payload.
92 * WebDAV PROPFIND and S3 HEAD/GET both need ETag headers. Registering
93 * r->headers_out.etag enables downstream conditional-request checks.
94 *
95 * HOW: etag_str(mtime, size, flags) → set_header("ETag", etag_buf). If register_not_modified,
96 * sets r->headers_out.etag = h for later ngx_http_check_if_none_match() use.
97 */
98
99ngx_int_t
100brix_http_add_etag_header(ngx_http_request_t *r, time_t mtime, off_t size,
101 unsigned etag_flags, ngx_flag_t register_not_modified)
102{
103 char etag_buf[64];
104 ngx_table_elt_t *h;
105 ngx_int_t rc;
106
107 brix_http_etag_str(etag_buf, sizeof(etag_buf), mtime, size, etag_flags);
108
109 rc = brix_http_set_header(r, "ETag", etag_buf, &h);
110 if (rc != NGX_OK) {
111 return rc;
112 }
113
114 if (register_not_modified) {
115 r->headers_out.etag = h;
116 }
117
118 return NGX_OK;
119}

121/*
122 * brix_http_add_content_range_header - set Content-Range header for partial responses.
123 *
124 * WHAT: Formats "bytes start-end/size" into the Content-Range response header via
125 * snprintf and brix_http_set_header().
126 *
127 * WHY: HTTP Range requests (RFC 7233 §4.2) require servers to return 206 Partial
128 * Content with a Content-Range header specifying served byte range and total size.
129 * WebDAV GET/HEAD with Range and S3 GET with Range both use this.
130 *
131 * HOW: snprintf("bytes %lld-%lld/%lld") → set_header("Content-Range", cr_buf).
132 */
133
134ngx_int_t
135brix_http_add_content_range_header(ngx_http_request_t *r,
136 off_t start, off_t end, off_t size)
137{
138 char cr_buf[80];
139
140 snprintf(cr_buf, sizeof(cr_buf), "bytes %lld-%lld/%lld",
141 (long long) start, (long long) end, (long long) size);
142
143 return brix_http_set_header(r, "Content-Range", cr_buf, NULL);
144}

146/*
147 * brix_http_set_file_headers - set standard file-serving response headers.
148 */
149
150ngx_int_t
151brix_http_set_file_headers(ngx_http_request_t *r,
152 time_t mtime, off_t total_size, off_t send_len,
153 const char *content_type, unsigned etag_flags,
154 int has_range, off_t range_start, off_t range_end)
155{
156 r->headers_out.status = has_range ? NGX_HTTP_PARTIAL_CONTENT
157 : NGX_HTTP_OK;
158 r->headers_out.content_length_n = send_len;
159 r->headers_out.last_modified_time = mtime;
160
161 if (content_type != NULL) {
162 if (brix_http_set_header(r, "Content-Type", content_type, NULL)
163 != NGX_OK)
164 {
165 return NGX_ERROR;
166 }
167 } else {
168 /* Use nginx types{} hash to map file extension → MIME type.
169 * Falls back to application/octet-stream if no match. */
170 ngx_http_set_content_type(r);
171 }
172
173 /* register_not_modified=1: sets r->headers_out.etag so nginx's built-in
174 * not-modified filter can evaluate If-Match / If-None-Match conditionals. */
175 if (brix_http_add_etag_header(r, mtime, total_size, etag_flags, 1)
176 != NGX_OK)
177 {
178 return NGX_ERROR;
179 }
180
181 if (has_range) {
182 if (brix_http_add_content_range_header(r, range_start, range_end,
183 total_size) != NGX_OK)
184 {
185 return NGX_ERROR;
186 }
187 }
188
189 return NGX_OK;
190}

192/*
193 * brix_http_send_file_range - build file-backed ngx_buf_t for byte range and dispatch.
194 *
195 * WHAT: Allocates ngx_buf_t + ngx_file_t from r->pool, sets in_file/last_buf,
196 * file_pos=file_last=start+len. Optionally adds pool cleanup (ngx_pool_cleanup_file)
197 * to auto-close fd after response sent. Sends header via ngx_http_send_header(),
198 * then filters chain through nginx output.
199 *
200 * WHY: WebDAV GET with Range and S3 GET need zero-copy sendfile from disk. This
201 * function handles buffer setup, cleanup registration, and response dispatch
202 * in one call — avoiding duplicated alloc/filter code across modules.
203 *
204 * HOW: ngx_pcalloc buf+file → set fields (fd, name, pos/last) → if close_fd:
205 * ngx_pool_cleanup_add(ngx_pool_cleanup_file). ngx_http_send_header() →
206 * ngx_http_output_filter(r, &out) or ngx_http_send_special(r, NGX_HTTP_LAST).
207 */
208
209ngx_int_t
210brix_http_send_file_range(ngx_http_request_t *r, ngx_fd_t fd,
211 const char *path, off_t start, off_t len, ngx_flag_t close_fd)
212{
213 ngx_buf_t *b;
214 ngx_chain_t out;
215 ngx_pool_cleanup_t *cln;
216 ngx_pool_cleanup_file_t *clnf;
217 ngx_int_t rc;
218
219 b = NULL;
220 clnf = NULL;
221 ngx_memzero(&out, sizeof(out));
222
223 if (len > 0) {
224 b = ngx_pcalloc(r->pool, sizeof(ngx_buf_t));
225 if (b == NULL) {
226 if (close_fd) {
227 ngx_close_file(fd);
228 }
229 return NGX_HTTP_INTERNAL_SERVER_ERROR;
230 }
231
232 b->file = ngx_pcalloc(r->pool, sizeof(ngx_file_t));
233 if (b->file == NULL) {
234 if (close_fd) {
235 ngx_close_file(fd);
236 }
237 return NGX_HTTP_INTERNAL_SERVER_ERROR;
238 }
239
240 b->file->name.len = ngx_strlen(path);
241 b->file->name.data = ngx_pnalloc(r->pool, b->file->name.len + 1);
242 if (b->file->name.data == NULL) {
243 if (close_fd) {
244 ngx_close_file(fd);
245 }
246 return NGX_HTTP_INTERNAL_SERVER_ERROR;
247 }
248 ngx_cpystrn(b->file->name.data, (u_char *) path,
249 b->file->name.len + 1);
250
251 b->in_file = 1;
252 b->last_buf = 1;
253 b->last_in_chain = 1;
254 b->file->fd = fd;
255 b->file->log = r->connection->log;
256 b->file_pos = start;
257 b->file_last = start + len;
258
259 if (close_fd) {
260 cln = ngx_pool_cleanup_add(r->pool,
261 sizeof(ngx_pool_cleanup_file_t));
262 if (cln == NULL) {
263 ngx_close_file(fd);
264 return NGX_HTTP_INTERNAL_SERVER_ERROR;
265 }
266
267 cln->handler = ngx_pool_cleanup_file;
268 clnf = cln->data;
269 clnf->fd = fd;
270 clnf->name = b->file->name.data;
271 clnf->log = r->pool->log;
272 }
273
274 out.buf = b;
275 out.next = NULL;
276 }
277
278 rc = ngx_http_send_header(r);
279 if (rc == NGX_ERROR || r->header_only) {
280 if (close_fd) {
281 ngx_close_file(fd);
282 if (clnf != NULL) {
283 clnf->fd = NGX_INVALID_FILE;
284 }
285 }
286 return rc;
287 }
288
289 if (len == 0) {
290 if (close_fd) {
291 ngx_close_file(fd);
292 }
293 return ngx_http_send_special(r, NGX_HTTP_LAST);
294 }
295
296 return ngx_http_output_filter(r, &out);
297}
