1/*
2 * stream_mirror_config.c — Phase 24 XRootD stream mirror: configuration-time
3 * directive setters (see stream_mirror.h).
4 *
5 * WHAT: Owns the three brix_stream_mirror_* directive setters and their private
6 * parse helpers: brix_stream_mirror_url (resolve + append a shadow target),
7 * and the opcode allowlist / exclude setters plus the opcode-name -> bit
8 * table they share.
9 * WHY: split out (phase-79 file-size cap) from the request-path code in
10 * stream_mirror.c / stream_mirror_launch.c so each file stays under the
11 * 500-line cap. This file runs ONLY at configuration time (getaddrinfo,
12 * token parsing) and shares nothing with the event-loop engine beyond the
13 * public conf struct — so it needs neither stream_mirror_internal.h nor the
14 * shadow-connection primitives.
15 * HOW: the setters are declared in stream_mirror.h and registered in the stream
16 * module's directive table; each populates fields on conf->mirror. Opcode
17 * tokens are mapped through a NULL-terminated name/bit table, keeping the
18 * parse loop a flat data walk.

24/*
25 * brix_stream_mirror_url host:port — append one shadow target, resolved at
26 * configuration time so request handlers never call getaddrinfo on the event
27 * loop. Up to BRIX_MIRROR_MAX_TARGETS may be configured.
28 */
29char *
30brix_stream_mirror_set_url(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
31{
32 ngx_stream_brix_srv_conf_t *xcf = conf;
33 ngx_str_t *value = cf->args->elts;
34 brix_mirror_target_t *t;
35 ngx_url_t u;
36 u_char *colon;
37 ngx_str_t hostport = value[1];
38
39 (void) cmd;
40
41 if (xcf->mirror.targets == NULL) {
42 xcf->mirror.targets = ngx_array_create(cf->pool,
43 BRIX_MIRROR_MAX_TARGETS, sizeof(brix_mirror_target_t));
44 if (xcf->mirror.targets == NULL) {
45 return NGX_CONF_ERROR;
46 }
47 }
48 if (xcf->mirror.targets->nelts >= BRIX_MIRROR_MAX_TARGETS) {
49 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
50 "brix_stream_mirror_url: at most %d targets supported",
51 BRIX_MIRROR_MAX_TARGETS);
52 return NGX_CONF_ERROR;
53 }
54
55 if (hostport.len == 0
56 || ngx_strlchr(hostport.data, hostport.data + hostport.len, ':')
57 == NULL)
58 {
59 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
60 "brix_stream_mirror_url: expected host:port, got \"%V\"",
61 &hostport);
62 return NGX_CONF_ERROR;
63 }
64
65 ngx_memzero(&u, sizeof(ngx_url_t));
66 u.url = hostport;
67 u.default_port = 1094;
68 if (ngx_parse_url(cf->pool, &u) != NGX_OK || u.naddrs == 0) {
69 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
70 "brix_stream_mirror_url: cannot resolve \"%V\"%s%s", &hostport,
71 u.err ? ": " : "", u.err ? u.err : "");
72 return NGX_CONF_ERROR;
73 }
74
75 t = ngx_array_push(xcf->mirror.targets);
76 if (t == NULL) {
77 return NGX_CONF_ERROR;
78 }
79 ngx_memzero(t, sizeof(*t));
80 t->url = hostport;
81 t->host = u.host;
82 t->port = u.port;
83 if (u.addrs[0].socklen > sizeof(t->sockaddr)) {
84 return NGX_CONF_ERROR;
85 }
86 ngx_memcpy(&t->sockaddr, u.addrs[0].sockaddr, u.addrs[0].socklen);
87 t->socklen = u.addrs[0].socklen;
88
89 /* keep the host:port colon split out of the host label for logging */
90 colon = ngx_strlchr(hostport.data, hostport.data + hostport.len, ':');
91 if (colon != NULL && t->host.len == 0) {
92 t->host.data = hostport.data;
93 t->host.len = (size_t) (colon - hostport.data);
94 }
95
96 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
97 "brix: stream mirror target %V", &hostport);
98 return NGX_CONF_OK;
99}

101/*
102 * Opcode-name → bitmask table.
103 *
104 * WHAT: static lookup pairing each accepted brix_mirror_opcodes token with the
105 * BRIX_MIRROR_OP_* bit(s) it selects. "all" expands to the full set.
106 * WHY: a data table replaces a 16-arm if/else ladder, keeping the parse loop
107 * branch-flat and making the accepted vocabulary a single edit point.
108 * HOW: terminated by a NULL name; brix_mirror_opcode_name_bit() scans it.
109 */
110typedef struct {
111 const char *name;
112 ngx_uint_t bit;
113} brix_mirror_opcode_name_t;
114
115static const brix_mirror_opcode_name_t brix_mirror_opcode_names[] = {
116 { "all", BRIX_MIRROR_OP_ALL },
117 { "stat", BRIX_MIRROR_OP_STAT },
118 { "locate", BRIX_MIRROR_OP_LOCATE },
119 { "open", BRIX_MIRROR_OP_OPEN },
120 { "read", BRIX_MIRROR_OP_READ },
121 { "readv", BRIX_MIRROR_OP_READV },
122 { "dirlist", BRIX_MIRROR_OP_DIRLIST },
123 { "statx", BRIX_MIRROR_OP_STATX },
124 { "query", BRIX_MIRROR_OP_QUERY },

126 { "mkdir", BRIX_MIRROR_OP_MKDIR },
127 { "rm", BRIX_MIRROR_OP_RM },

129 { "mv", BRIX_MIRROR_OP_MV },
130 { "truncate", BRIX_MIRROR_OP_TRUNCATE },
131 { "chmod", BRIX_MIRROR_OP_CHMOD },
132 { "write", BRIX_MIRROR_OP_WRITE },
133 { NULL, 0 },
134};

136/*
137 * Map one opcode-name token to its bitmask.
138 *
139 * WHAT: return the BRIX_MIRROR_OP_* bit(s) for @name, or 0 if unrecognised.
140 * WHY: isolates the table scan so the parse loop stays a flat data walk.
141 * HOW: linear scan of brix_mirror_opcode_names (small, config-time only).
142 */
143static ngx_uint_t
144brix_mirror_opcode_name_bit(const u_char *name)
145{
146 const brix_mirror_opcode_name_t *e;
147
148 for (e = brix_mirror_opcode_names; e->name != NULL; e++) {
149 if (ngx_strcmp(name, e->name) == 0) {
150 return e->bit;
151 }
152 }
153 return 0;
154}

156/*
157 * Parse opcode name args (cf->args[1..]) into a bitmask. "all" expands to
158 * BRIX_MIRROR_OP_ALL. Shared by the allowlist and exclude setters.
159 */
160static char *
161brix_mirror_parse_opcode_args(ngx_conf_t *cf, const char *directive,
162 ngx_uint_t *mask_out)
163{
164 ngx_str_t *value = cf->args->elts;
165 ngx_uint_t i, mask = 0;
166
167 for (i = 1; i < cf->args->nelts; i++) {
168 ngx_str_t *v = &value[i];
169 ngx_uint_t bit = brix_mirror_opcode_name_bit(v->data);
170
171 if (bit == 0) {
172 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
173 "%s: unknown opcode \"%V\" (expected one of"
174 " all stat locate open read readv dirlist statx query"
175 " mkdir rm rmdir mv truncate chmod write)",
176 directive, v);
177 return NGX_CONF_ERROR;
178 }
179 mask |= bit;
180 }
181 *mask_out = mask;
182 return NGX_CONF_OK;
183}
