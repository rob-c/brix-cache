4/*
5 * WHAT: Prometheus HTTP exporter for read-through cache metrics — filesystem occupancy, eviction counters,
6 * and configured thresholds per server listener.
7 * WHY: The brix_cache_root provides XCache-style direct-mode fills from anonymous root:// origins. Cache
8 * behavior (eviction on high occupancy, bytes reclaimed, eviction errors) needs observable Prometheus
9 * gauges and counters so operators can tune cache_eviction_threshold_ratio and monitor fill efficiency.
10 * HOW: brix_cache_statvfs() calls statvfs(2) on the configured root directory to compute total/used/available
11 * bytes and occupancy_ppm (parts per million). brix_export_stream_cache_metrics() calls one static
12 * per-family emit helper in sequence; each helper writes the family's HELP/TYPE block then fans a
13 * labelled row (or rows) across the SHM server slots. The exporter itself is a flat call sequence.
14 */
15
16/*
17 * WHAT: Snapshot the filesystem occupancy of one cache root directory.
18 * WHY: Cache gauges (occupancy ratio, bytes by state) must reflect live disk
19 * usage at scrape time, so we re-stat per scrape rather than caching.
20 * HOW: Delegates to the shared brix_fs_usage_stat() (statvfs(2) wrapper) and
21 * unpacks its struct into the caller's out-params. occupancy_ppm is
22 * parts-per-million (used/total * 1e6) computed once by the helper so the
23 * caller need not redo the division. Returns NGX_ERROR if the root cannot
24 * be statted (e.g. unmounted/missing) — callers skip that server's row.
25 */
26static ngx_int_t
27brix_cache_statvfs(const char *root, uint64_t *total, uint64_t *used,
28 uint64_t *available, ngx_uint_t *occupancy_ppm)
29{
30 brix_fs_usage_t fsu;
31
32 if (brix_fs_usage_stat(root, &fsu) != NGX_OK) {
33 return NGX_ERROR;
34 }
35
36 *total = fsu.total_bytes;
37 *available = fsu.available_bytes;
38 *used = fsu.occupancy_bytes;
39 *occupancy_ppm = fsu.occupancy_ppm;
40
41 return NGX_OK;
42}

44/*
45 * WHAT: Render one server slot's numeric port into a NUL-terminated C string.
46 * WHY: Every labelled row uses the "%s" port label, but ngx_snprintf does not
47 * NUL-terminate; the "%Z" verb appends the trailing '\0' so the buffer is a
48 * valid C string for "%s". Centralised so all families render identically.
49 * HOW: Writes into the caller's fixed buffer (port_str[16], matching u16 port
50 * widths). Pure formatter with no side effects beyond filling the buffer.
51 */
52static void
53stream_cache_port_str(char *buf, size_t buf_size, ngx_uint_t port)
54{
55 ngx_snprintf((u_char *) buf, buf_size, "%ui%Z", port);
56}

58/*
59 * WHAT: Emit the occupancy-ratio gauge family (used/total, ppm scaled to 0..1).
60 * WHY: Prometheus groups all rows of a family under one HELP/TYPE block, so this
61 * helper owns both the header and the per-server fan-out for this family.
62 * HOW: Header first; then one row per cache-enabled active server that can be
63 * statted (missing/unmounted roots are skipped). ppm/1e6 gives the 0..1
64 * ratio convention Prometheus expects for a gauge.
65 */
66static void
67stream_cache_emit_occupancy_ratio(metrics_writer_t *mw,
68 ngx_brix_metrics_t *shm)
69{
70 ngx_brix_srv_metrics_t *srv;
71 ngx_uint_t i;
72 char port_str[16];
73
74 mw_printf(mw,
75 "# HELP brix_cache_occupancy_ratio "
76 "Filesystem occupancy ratio for brix_cache_export.\n"
77 "# TYPE brix_cache_occupancy_ratio gauge\n");
78 for (i = 0; i < BRIX_METRICS_MAX_SERVERS; i++) {
79 uint64_t total = 0, used = 0, available = 0;
80 ngx_uint_t occupancy_ppm = 0;
81
82 srv = &shm->servers[i];
83 if (!srv->in_use || !srv->cache_enabled) { continue; }
84 if (brix_cache_statvfs(srv->cache_root, &total, &used,
85 &available, &occupancy_ppm) != NGX_OK)
86 {
87 continue;
88 }
89 stream_cache_port_str(port_str, sizeof(port_str), srv->port);
90 /* ppm (parts-per-million) -> ratio for Prometheus's 0..1 gauge convention. */
91 mw_printf(mw,
92 "brix_cache_occupancy_ratio{port=\"%s\",auth=\"%s\"} %0.6f\n",
93 port_str, srv->auth, (double) occupancy_ppm / 1000000.0);
94 }
95}

171/*
172 * WHAT: Emit one single-row cache-eviction counter family (cache_enabled gate).
173 * WHY: The three eviction counters (files/bytes/errors) share an identical shape
174 * — header + one atomically-read counter row per cache-enabled server — so a
175 * single parametrised helper avoids three near-duplicate loops.
176 * HOW: Emits the caller-supplied HELP/TYPE header block, then one row per
177 * cache-enabled active server reading *counter via ngx_atomic_fetch_add(&x,0)
178 * (adding zero is the lock-free atomic-load idiom, not a mutation). counter
179 * is a pointer to the srv-relative offset resolved by the caller per slot.

189

193
194 srv = &shm->servers[i];

237/*
