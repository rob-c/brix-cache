4/*
5 * WHAT: Prometheus HTTP exporter for read-through cache metrics — filesystem occupancy, eviction counters,
6 * and configured thresholds per server listener.
7 * WHY: The brix_cache_root provides XCache-style direct-mode fills from anonymous root:// origins. Cache
8 * behavior (eviction on high occupancy, bytes reclaimed, eviction errors) needs observable Prometheus
9 * gauges and counters so operators can tune cache_eviction_threshold_ratio and monitor fill efficiency.
10 * HOW: brix_cache_statvfs() calls statvfs(2) on the configured root directory to compute total/used/available
11 * bytes and occupancy_ppm (parts per million). brix_export_stream_cache_metrics() iterates server slots,
12 * skips inactive/cache-disabled entries, emits occupancy_ratio gauge, eviction_threshold gauge, cache_bytes
13 * gauge with state labels (total/used/available), evictions_total counter, evicted_bytes_total counter,
14 * and eviction_errors_total counter — all with port/auth label pairs.
15 */
16
17/*
18 * WHAT: Snapshot the filesystem occupancy of one cache root directory.
19 * WHY: Cache gauges (occupancy ratio, bytes by state) must reflect live disk
20 * usage at scrape time, so we re-stat per scrape rather than caching.
21 * HOW: Delegates to the shared brix_fs_usage_stat() (statvfs(2) wrapper) and
22 * unpacks its struct into the caller's out-params. occupancy_ppm is
23 * parts-per-million (used/total * 1e6) computed once by the helper so the
24 * caller need not redo the division. Returns NGX_ERROR if the root cannot
25 * be statted (e.g. unmounted/missing) — callers skip that server's row.
26 */
27static ngx_int_t
28brix_cache_statvfs(const char *root, uint64_t *total, uint64_t *used,
29 uint64_t *available, ngx_uint_t *occupancy_ppm)
30{
31 brix_fs_usage_t fsu;
32
33 if (brix_fs_usage_stat(root, &fsu) != NGX_OK) {
34 return NGX_ERROR;
35 }
36
37 *total = fsu.total_bytes;
38 *available = fsu.available_bytes;
39 *used = fsu.occupancy_bytes;
40 *occupancy_ppm = fsu.occupancy_ppm;
41
42 return NGX_OK;
43}

45/*
46 * WHAT: Emit the full read-through-cache + write-through metric families to the
47 * Prometheus writer, one labelled row per active server listener.
48 * WHY: Each server block can have its own cache root, threshold, and counters,
49 * so every gauge/counter is fanned out across the SHM server slots and
50 * labelled by {port,auth} to stay low-cardinality (no paths as labels).
51 * HOW: Prometheus format requires all rows of a family to be grouped under a
52 * single HELP/TYPE block, hence the deliberate "one loop per family"
53 * shape rather than one loop emitting all metrics per server. Within each
54 * loop:
55 * - skip slots that are not in_use (free) and, for cache families, not
56 * cache_enabled — write-through families gate on in_use only;
57 * - render the numeric port into a NUL-terminated C string (see the
58 * %Z note below) for use with the "%s" label format;
59 * - counters are read with ngx_atomic_fetch_add(&x, 0): adding zero is
60 * the lock-free read idiom — it is NOT a mutation, just an atomic load.
61 * The two statvfs-backed families re-stat per server each time (cheap,
62 * and keeps the gauge fresh); intervening families need no syscall.
63 */
