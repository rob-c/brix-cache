1/*
2 * stream_wmirror_replay.c — XRootD stream data-write mirror: detached-replay
3 * lifecycle and socket I/O (see stream_wmirror_internal.h).
4 *
5 * WHAT: The connection-lifetime half of the detached replay. Opens the shadow
6 * socket, wires it into nginx's event loop, drains write buffers, reads
7 * response frames, bounds the whole exchange with a deadline timer, and
8 * tears everything down (freeing the transferred heap buffers) exactly
9 * once. wmir_launch seeds a replay from a fully-buffered file.
10 *
11 * WHY: Split out of stream_wmirror.c under the phase-79 500-line cap. The
12 * socket setup, event handlers, flush, teardown and launch form one
13 * cohesive concern — the replay's lifetime — distinct from the shadow-side
14 * protocol decisions (stream_wmirror_state.c) and the client-side
15 * accumulation (stream_wmirror.c).
16 *
17 * HOW: Each read event pulls one complete frame (wmir_recv_frame) and hands it
18 * to the state machine (wmir_dispatch, stream_wmirror_state.c); the state
19 * machine calls back into wmir_flush to queue frames and wmir_finish to
20 * tear down. The shadow MUST be an isolated namespace — replaying writes
21 * onto the primary's backing store would corrupt it.
22 */

60/* Read event entry point: pull one complete frame, then hand it to the state
61 * machine. NGX_AGAIN re-arms and yields; the deadline timer (wmir_timeout_
62 * handler) is the only thing that bounds a stalled shadow. */
63static void
64wmir_read_handler(ngx_event_t *rev)
65{
66 ngx_connection_t *c = rev->data;
67 brix_wmirror_replay_t *r = c->data;
68 ngx_int_t rc;
69
70 if (rev->timedout) { wmir_finish(r, 0); return; }
71
72 rc = wmir_recv_frame(r);
73 if (rc == NGX_AGAIN) {
74 if (ngx_handle_read_event(rev, 0) != NGX_OK) { wmir_finish(r, 0); }
75 return;
76 }
77 if (rc == NGX_ERROR) { wmir_finish(r, 0); return; }
78 wmir_dispatch(r);
79}

81/*
82 * Write event entry point. Two jobs: (1) finalize an in-flight non-blocking
83 * connect() — the first writable event after EINPROGRESS means connect resolved,
84 * so SO_ERROR is checked to distinguish success from a deferred failure; (2)
85 * resume draining the current wbuf after an earlier NGX_AGAIN. This is the only
86 * place the `connecting` flag is cleared.
87 */
88static void
89wmir_write_handler(ngx_event_t *wev)
90{
91 ngx_connection_t *c = wev->data;
92 brix_wmirror_replay_t *r = c->data;
93
94 if (wev->timedout) { wmir_finish(r, 0); return; }
95
96 if (r->connecting) {
97 int err = 0;
98 socklen_t len = sizeof(err);
99 /* SO_ERROR carries the async connect() result; nonzero (or getsockopt
100 * failure) means the connection never came up. */
101 if (getsockopt(c->fd, SOL_SOCKET, SO_ERROR, (char *) &err, &len) == -1
102 || err != 0)
103 {
104 wmir_finish(r, 0);
105 return;
106 }
107 r->connecting = 0;
108 }
109
110 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
111}

113/* Whole-exchange deadline fired: the shadow stalled at some phase — tear down

115static void
116wmir_timeout_handler(ngx_event_t *ev)
117{

121/*
122 * Single teardown path for the replay (success and every failure route here).
123 *
124 * WHAT: cancel the deadline timer, close the shadow connection, record the
125 * outcome metric, and free everything. OWNERSHIP/ORDER (important): r->data and
126 * r->open_frame are heap buffers (ngx_alloc) transferred in at launch and must
127 * be ngx_free()d explicitly; r itself was pcalloc'd FROM r->pool, so destroying
128 * the pool last frees r along with all its pool-allocated buffers. `pool` is
129 * cached first precisely because r becomes invalid after ngx_destroy_pool().
130 */
131void
132wmir_finish(brix_wmirror_replay_t *r, int ok)
133{
134 ngx_pool_t *pool = r->pool;
135
136 if (r->tev.timer_set) { ngx_del_timer(&r->tev); }
137 if (r->conn != NULL) { ngx_close_connection(r->conn); r->conn = NULL; }
138
139 if (ok) {
140 BRIX_WMIR_METRIC_INC(mirror_stream_total);
141 } else {
142 BRIX_WMIR_METRIC_INC(mirror_stream_errors_total);
143 if (r->log_diverge) {
144 ngx_log_error(NGX_LOG_NOTICE, r->log, 0,
145 "xrootd write-mirror: shadow write replay failed/diverged");
146 }
147 }
148 if (r->data != NULL) { ngx_free(r->data); }
149 if (r->open_frame != NULL) { ngx_free(r->open_frame); }
150 ngx_destroy_pool(pool); /* frees r itself (last use above) */
151}

153/*
154 * Open the shadow socket, register it with nginx's event loop, queue the
155 * pipelined bootstrap, and kick off the (possibly async) connect.
156 *
157 * HOW: a non-blocking socket is wrapped in an ngx_connection_t whose read/write
158 * handlers drive the state machine; the bootstrap (handshake + protocol + login)
159 * is built into wbuf up front so it is sent the moment the socket is writable.
160 * connect() either completes immediately (rc==0 -> flush now) or returns
161 * EINPROGRESS (-> set `connecting`, finish in the write handler). Any setup
162 * failure routes through wmir_finish so the pool is always reclaimed.
163 */
164static void
165wmir_start(brix_wmirror_replay_t *r, ngx_msec_t timeout_ms)
166{
167 ngx_connection_t *c;
168 ngx_socket_t fd;
169 size_t bslen;
170 int rc;
171
172 fd = ngx_socket(r->sockaddr.ss_family, SOCK_STREAM, 0);
173 if (fd == (ngx_socket_t) -1 || ngx_nonblocking(fd) == -1) {
174 if (fd != (ngx_socket_t) -1) { ngx_close_socket(fd); }
175 wmir_finish(r, 0);
176 return;
177 }
178 c = ngx_get_connection(fd, r->log);
179 if (c == NULL) { ngx_close_socket(fd); wmir_finish(r, 0); return; }
180 c->pool = r->pool;
181 c->data = r;
182 c->recv = ngx_recv;
183 c->send = ngx_send;
184 c->read->handler = wmir_read_handler;
185 c->write->handler = wmir_write_handler;
186 c->read->log = c->write->log = r->log;
187 r->conn = c;
188
189 /* Bootstrap is three frames concatenated into one write so they pipeline:
190 * handshake, then ClientProtocolRequest, then ClientLoginRequest. */
191 bslen = XRD_HANDSHAKE_LEN + sizeof(ClientProtocolRequest)
192 + sizeof(ClientLoginRequest);
193 r->wbuf = ngx_palloc(r->pool, bslen);
194 if (r->wbuf == NULL) { wmir_finish(r, 0); return; }
195 brix_upstream_build_bootstrap(r->wbuf);
196 r->wbuf_len = bslen;
197 r->wbuf_pos = 0;
198 r->phase = WMIR_HANDSHAKE;
199
200 r->tev.handler = wmir_timeout_handler;
201 r->tev.data = r;
202 r->tev.log = r->log;
203 ngx_add_timer(&r->tev,
204 timeout_ms ? timeout_ms : BRIX_MIRROR_DEFAULT_TIMEOUT_MS);
205
206 rc = connect(fd, (struct sockaddr *) &r->sockaddr, r->socklen);
207 /* Only a hard, non-EINPROGRESS error is fatal here; EINPROGRESS is the
208 * normal async case and is resolved later via SO_ERROR. */
209 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
210 wmir_finish(r, 0);
211 return;
212 }
213 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
214 wmir_finish(r, 0);
215 return;
216 }
217 if (rc == 0) {
218 r->connecting = 0; /* connected synchronously — send now */
219 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
220 } else {
221 r->connecting = 1; /* completes in the write handler */
222 }
223}

225/* Launch a detached replay for a fully-buffered file. Ownership of the data
226 * buffer transfers to the replay (freed in wmir_finish). */
227void
228wmir_launch(ngx_stream_brix_srv_conf_t *conf, brix_wmirror_file_t *f)
229{
230 brix_mirror_target_t *t;
231 ngx_pool_t *pool;
232 brix_wmirror_replay_t *r;
233 u_char *frame;
234
235 if (conf->mirror.targets == NULL || conf->mirror.targets->nelts == 0) {
236 return;
237 }
238 t = (brix_mirror_target_t *) conf->mirror.targets->elts; /* first target */
239 if (t->socklen == 0) { return; }
240
241 pool = ngx_create_pool(2048, ngx_cycle->log);
242 if (pool == NULL) { return; }
243 r = ngx_pcalloc(pool, sizeof(*r));
244 if (r == NULL) { ngx_destroy_pool(pool); return; }
245 r->pool = pool;
246 r->log = ngx_cycle->log; /* outlives the client connection */
247 r->log_diverge = conf->mirror.log_diverge ? 1 : 0;
248 r->port = t->port;
249 r->socklen = t->socklen;
250 ngx_memcpy(&r->sockaddr, &t->sockaddr, t->socklen);
251 ngx_cpystrn((u_char *) r->host,
252 t->host.data ? t->host.data : (u_char *) "?", sizeof(r->host));
253
254 /* Rebuild the open frame as a single heap buffer: the captured 24-byte open
255 * header followed by its path/cgi payload. This is ngx_alloc (not pool) so
256 * it survives independently and is ngx_free()d in wmir_finish. */
257 r->open_frame_len = (size_t) 24 + f->open_dlen;
258 frame = ngx_alloc(r->open_frame_len, ngx_cycle->log);
259 if (frame == NULL) { ngx_destroy_pool(pool); return; }
260 ngx_memcpy(frame, f->open_hdr, 24);
261 if (f->open_dlen && f->open_payload != NULL) {
262 ngx_memcpy(frame + 24, f->open_payload, f->open_dlen);
263 }
264 r->open_frame = frame;
265 /* Move the accumulated file bytes to the replay rather than copy: f->data is
266 * nulled so wmir_file_reset (called by the caller after this) won't free it;
267 * the replay now owns and frees it. */
268 r->data = f->data; /* transfer ownership */
269 r->data_len = f->data_len;
270 f->data = NULL; /* the replay frees it now */
271
272 wmir_start(r, conf->mirror.timeout_ms);
273}
