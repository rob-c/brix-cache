39static ngx_int_t
40xmeta_sidecar_write(brix_sd_instance_t *store, const char *key,
41 const uint8_t *buf, size_t len)
42{
43 char ck[PATH_MAX];
44 brix_sd_staged_t *st;
45 int err = 0;
46
47 if (store->driver->staged_open == NULL
48 || store->driver->staged_write == NULL
49 || store->driver->staged_commit == NULL)
50 {
51 errno = ENOTSUP;
52 return NGX_ERROR;
53 }
54 if (xmeta_sidecar_key(key, ck, sizeof(ck)) != 0) {
55 errno = ENAMETOOLONG;
56 return NGX_ERROR;
57 }
58 /* SECURITY: the "<key>.cinfo" sidecar leaks cache residency (block-present
59 * bitmap), size and mtime. 0600 (not 0644) so a mapped low-priv uid cannot
60 * read another user's cache metadata from the svc-owned store. */
61 st = store->driver->staged_open(store, ck, 0600, &err);
62 if (st == NULL) {
63 errno = err ? err : EIO;
64 return NGX_ERROR;
65 }
66 if (store->driver->staged_write(st, buf, len, 0) != (ssize_t) len) {
67 int e = errno;
68
69 if (store->driver->staged_abort != NULL) {
70 store->driver->staged_abort(st);
71 }
72 errno = e ? e : EIO;
73 return NGX_ERROR;
74 }
75 if (store->driver->staged_commit(st, 0) != NGX_OK) {
76 return NGX_ERROR; /* errno from the driver */
77 }
78 return NGX_OK;
79}

81/* ---- Close an open sidecar object, preserving errno ----
82 *
83 * WHAT: Releases the driver object and, when the driver returned a heap-shell
84 * wrapper, frees that too. errno is saved and restored so a preceding read
85 * failure's errno survives this best-effort cleanup.
86 *
87 * WHY: The read loop sets the caller-visible errno before tearing the object
88 * down (see the original inline ordering), and close/free may clobber errno.
89 * Centralising the close keeps that ordering guarantee in one place and drops
90 * the thrice-repeated close+heap_shell idiom.
91 *
92 * HOW:
93 * 1. Snapshot errno.
94 * 2. Call the driver close, then free the heap shell if present.
95 * 3. Restore the snapshot.
96 */
97static void
98xmeta_obj_close(brix_sd_instance_t *store, brix_sd_obj_t *obj)
99{
100 int saved = errno;
101
102 store->driver->close(obj);
103 if (obj->heap_shell) { free(obj); }
104 errno = saved;
105}

107/* ---- Ensure the read buffer has room for one more pread chunk ----
108 *
109 * WHAT: When the buffer is full (got == *capp), doubles the capacity (seeding
110 * an empty buffer at 64 KiB) via realloc. Returns 0 on success (buffer may have
111 * moved); -1 on allocation failure, in which case the old buffer is freed and
112 * *bufp is set to NULL so the caller must not free it again.
113 *
114 * WHY: Isolates the grow-on-demand branch out of the read loop so the loop body
115 * stays flat, and makes the "free + NULL on failure" ownership rule explicit.
116 *
117 * HOW:
118 * 1. If not yet full, return 0 unchanged.
119 * 2. Compute the next capacity and realloc.
120 * 3. On failure free the old block, NULL the handle, return -1.
121 * 4. On success publish the grown block and capacity, return 0.
122 */
123static int
124xmeta_read_buf_ensure(uint8_t **bufp, size_t *capp, size_t got)
125{
126 uint8_t *grown;
127 size_t cap = *capp;
128
129 if (got != cap) {
130 return 0;
131 }
132 cap = cap ? cap * 2 : 64 * 1024;
133 grown = realloc(*bufp, cap);
134 if (grown == NULL) {
135 free(*bufp);
136 *bufp = NULL;
137 return -1;
138 }
139 *bufp = grown;
140 *capp = cap;
141 return 0;
142}

144/* ---- Slurp a whole open object into a fresh malloc buffer ----
145 *
146 * WHAT: Reads the object to EOF into a growing malloc buffer. On a non-empty
147 * object sets out and out_len and returns NGX_OK; on a zero-length object returns
148 * NGX_DECLINED; on error returns NGX_ERROR with errno set (ENOMEM for growth
149 * failure, else the pread errno or EIO). Owns nothing it returns on the error
150 * and declined paths (buffer freed).
151 *
152 * WHY: Pulls the buffer-management loop out of xmeta_sidecar_read so that
153 * function reduces to open + slurp + close, each independently reviewable. The
154 * object itself is left open for the caller to close.
155 *
156 * HOW:
157 * 1. Loop: grow the buffer when full, then pread at the current offset.
158 * 2. On a short/negative read, free and return NGX_ERROR with errno.
159 * 3. On EOF (n == 0) break.
160 * 4. Empty object -> free and NGX_DECLINED; otherwise publish and NGX_OK.
161 */
162static ngx_int_t
163xmeta_obj_slurp(brix_sd_instance_t *store, brix_sd_obj_t *obj,
164 uint8_t **out, size_t *out_len)
165{
166 uint8_t *buf = NULL;
167 size_t cap = 0, got = 0;
168
169 for ( ;; ) {
170 ssize_t n;
171
172 if (xmeta_read_buf_ensure(&buf, &cap, got) != 0) {
173 errno = ENOMEM;
174 return NGX_ERROR;
175 }
176 n = store->driver->pread(obj, buf + got, cap - got, (off_t) got);
177 if (n < 0) {
178 int e = errno;
179
180 free(buf);
181 errno = e ? e : EIO;
182 return NGX_ERROR;
183 }
184 if (n == 0) {
185 break;
186 }
187 got += (size_t) n;
188 }
189 if (got == 0) {
190 free(buf);
191 return NGX_DECLINED;
192 }
193 *out = buf;
194 *out_len = got;
195 return NGX_OK;
196}

198/* Read the whole sidecar object into a malloc'd buffer. NGX_OK/DECLINED/ERROR. */
199static ngx_int_t
200xmeta_sidecar_read(brix_sd_instance_t *store, const char *key,
201 uint8_t **out, size_t *out_len)
202{
203 char ck[PATH_MAX];
204 brix_sd_obj_t *obj;
205 ngx_int_t rc;
206 int err = 0;
207
208 if (store->driver->open == NULL || store->driver->pread == NULL) {
209 errno = ENOTSUP;
210 return NGX_ERROR;
211 }
212 if (xmeta_sidecar_key(key, ck, sizeof(ck)) != 0) {
213 errno = ENAMETOOLONG;
214 return NGX_ERROR;
215 }
216 obj = store->driver->open(store, ck, BRIX_SD_O_READ, 0, &err);
217 if (obj == NULL) {
218 return (err == ENOENT) ? NGX_DECLINED : NGX_ERROR;
219 }
220
221 rc = xmeta_obj_slurp(store, obj, out, out_len);
222 xmeta_obj_close(store, obj);
223 return rc;
224}

228ngx_int_t
229brix_xmeta_save(brix_sd_instance_t *store, const char *key,
230 const brix_xmeta_t *m)
231{
232 uint8_t *buf = NULL;
233 size_t len = 0;
234 ngx_int_t rc;
235
236 if (store == NULL || store->driver == NULL || key == NULL || m == NULL) {
237 errno = EINVAL;
238 return NGX_ERROR;
239 }
240 if (brix_xmeta_encode(m, &buf, &len) != BRIX_XMETA_OK) {
241 return NGX_ERROR;
242 }
243
244 if (store->driver->setxattr != NULL && len <= BRIX_XMETA_XATTR_MAX) {
245 rc = store->driver->setxattr(store, key, BRIX_XMETA_XATTR_NAME,
246 buf, len, 0);
247 if (rc == NGX_OK) {
248 free(buf);
249 return NGX_OK;
250 }
251 if (!xmeta_xattr_unfit(errno)) {
252 int e = errno;
253
254 free(buf);
255 errno = e;
256 return NGX_ERROR;
257 }
258 /* doesn't fit / not supported here: ride the sidecar instead */
259 }
260
261 rc = xmeta_sidecar_write(store, key, buf, len);
262 free(buf);
263 if (rc != NGX_OK) {
264 return NGX_ERROR;
265 }
266 /* exactly one carrier: drop any (stale or just-outgrown) xattr copy */
267 if (store->driver->removexattr != NULL) {
268 (void) store->driver->removexattr(store, key,
269 BRIX_XMETA_XATTR_NAME);
270 }
271 return NGX_OK;
272}

274/* ---- Map a brix_xmeta_decode() result to the load API's ngx_int_t ----
275 *
276 * WHAT: Translates the decode status: BRIX_XMETA_OK -> NGX_OK, BRIX_XMETA_FOREIGN
277 * -> NGX_DECLINED (a valid record owned by someone else), anything else ->
278 * NGX_ERROR.
279 *
280 * WHY: The identical three-way mapping is applied to both the xattr and sidecar
281 * decode results; factoring it keeps the two carriers in lock-step and out of the
282 * caller's branch count.
283 *
284 * HOW:
285 * 1. OK -> NGX_OK.
286 * 2. FOREIGN -> NGX_DECLINED, else NGX_ERROR.
287 */
288static ngx_int_t
289xmeta_decode_rc(int drc)
290{
291 if (drc == BRIX_XMETA_OK) {
292 return NGX_OK;
293 }
294 return (drc == BRIX_XMETA_FOREIGN) ? NGX_DECLINED : NGX_ERROR;
295}

297/* ---- Attempt the xattr carrier for a load ----
298 *
299 * WHAT: When the driver exposes getxattr, reads the metadata xattr into a
300 * fixed-size buffer and decodes it. Returns NGX_OK/NGX_DECLINED/NGX_ERROR when
301 * the xattr settled the outcome, or NGX_AGAIN when the record is simply absent
302 * / the xattr is unsupported here and the caller should fall through to the
303 * sidecar. NGX_ERROR carries errno (ENOMEM on alloc failure, else the getxattr
304 * errno for a genuine failure).
305 *
306 * WHY: The xattr-first branch is the bulk of brix_xmeta_load's complexity;
307 * isolating it lets the public function read as "try xattr, else sidecar". The
308 * NGX_AGAIN sentinel is internal only (the public API never returns it) and
309 * distinguishes "not here, keep looking" from a hard decode/IO result.
310 *
311 * HOW:
312 * 1. No getxattr -> NGX_AGAIN (fall through).
313 * 2. Alloc the max-size buffer; ENOMEM -> NGX_ERROR.
314 * 3. getxattr; on a positive length decode and map the result.
315 * 4. On a real error (not absent/unsupported) -> NGX_ERROR.
316 * 5. Otherwise (absent / unsupported) -> NGX_AGAIN.
317 */
318static ngx_int_t
319xmeta_load_from_xattr(brix_sd_instance_t *store, const char *key,
320 brix_xmeta_t *m)
321{
322 uint8_t *buf;
323 ssize_t n;
324 int drc;
325
326 if (store->driver->getxattr == NULL) {
327 return NGX_AGAIN;
328 }
329 buf = malloc(BRIX_XMETA_XATTR_MAX);
330 if (buf == NULL) {
331 errno = ENOMEM;
332 return NGX_ERROR;
333 }
334 n = store->driver->getxattr(store, key, BRIX_XMETA_XATTR_NAME,
335 buf, BRIX_XMETA_XATTR_MAX);
336 if (n > 0) {
337 drc = brix_xmeta_decode(buf, (size_t) n, m);
338 free(buf);
339 return xmeta_decode_rc(drc);
340 }
341 free(buf);
342 if (n < 0 && errno != ENODATA && errno != ENOENT
343#ifdef ENOATTR
344 && errno != ENOATTR
345#endif
346 && !xmeta_xattr_unfit(errno))
347 {
348 return NGX_ERROR;
349 }
350 return NGX_AGAIN; /* absent / unsupported: try the sidecar */
351}

354brix_xmeta_load(brix_sd_instance_t *store, const char *key,
355 brix_xmeta_t *m)
356{
357 uint8_t *buf;
358 size_t len = 0;
359 ngx_int_t rc;
360 int drc;
361
362 if (store == NULL || store->driver == NULL || key == NULL || m == NULL) {
363 errno = EINVAL;
364 return NGX_ERROR;
365 }
366
367 /* xattr carrier first */
368 rc = xmeta_load_from_xattr(store, key, m);
369 if (rc != NGX_AGAIN) {
370 return rc;
371 }
372
373 rc = xmeta_sidecar_read(store, key, &buf, &len);
374 if (rc != NGX_OK) {
375 return rc;
376 }
377 drc = brix_xmeta_decode(buf, len, m);
378 free(buf);
379 return xmeta_decode_rc(drc);
