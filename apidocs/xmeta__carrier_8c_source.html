14/* Compose the sidecar object key "<key>.cinfo" into out[cap]. 0 / -1. */

17{
18 int n = snprintf(out, cap, "%s%s", key, BRIX_XMETA_SIDECAR_SUFFIX);

36static ngx_int_t
37xmeta_sidecar_write(brix_sd_instance_t *store, const char *key,
38 const uint8_t *buf, size_t len)
39{
40 char ck[PATH_MAX];
41 brix_sd_staged_t *st;
42 int err = 0;
43
44 if (store->driver->staged_open == NULL
45 || store->driver->staged_write == NULL
46 || store->driver->staged_commit == NULL)
47 {
48 errno = ENOTSUP;
49 return NGX_ERROR;
50 }
51 if (xmeta_sidecar_key(key, ck, sizeof(ck)) != 0) {
52 errno = ENAMETOOLONG;
53 return NGX_ERROR;
54 }
55 /* SECURITY: the "<key>.cinfo" sidecar leaks cache residency (block-present
56 * bitmap), size and mtime. 0600 (not 0644) so a mapped low-priv uid cannot
57 * read another user's cache metadata from the svc-owned store. */
58 st = store->driver->staged_open(store, ck, 0600, &err);
59 if (st == NULL) {
60 errno = err ? err : EIO;
61 return NGX_ERROR;
62 }
63 if (store->driver->staged_write(st, buf, len, 0) != (ssize_t) len) {
64 int e = errno;
65
66 if (store->driver->staged_abort != NULL) {
67 store->driver->staged_abort(st);
68 }
69 errno = e ? e : EIO;
70 return NGX_ERROR;
71 }
72 if (store->driver->staged_commit(st, 0) != NGX_OK) {
73 return NGX_ERROR; /* errno from the driver */
74 }
75 return NGX_OK;
76}

78/* Read the whole sidecar object into a malloc'd buffer. NGX_OK/DECLINED/ERROR. */
79static ngx_int_t
80xmeta_sidecar_read(brix_sd_instance_t *store, const char *key,
81 uint8_t **out, size_t *out_len)
82{
83 char ck[PATH_MAX];
84 brix_sd_obj_t *obj;
85 uint8_t *buf = NULL;
86 size_t cap = 0, got = 0;
87 int err = 0;
88
89 if (store->driver->open == NULL || store->driver->pread == NULL) {
90 errno = ENOTSUP;
91 return NGX_ERROR;
92 }
93 if (xmeta_sidecar_key(key, ck, sizeof(ck)) != 0) {
94 errno = ENAMETOOLONG;
95 return NGX_ERROR;
96 }
97 obj = store->driver->open(store, ck, BRIX_SD_O_READ, 0, &err);
98 if (obj == NULL) {
99 return (err == ENOENT) ? NGX_DECLINED : NGX_ERROR;
100 }
101
102 for ( ;; ) {
103 ssize_t n;
104
105 if (got == cap) {
106 uint8_t *grown;
107
108 cap = cap ? cap * 2 : 64 * 1024;
109 grown = realloc(buf, cap);
110 if (grown == NULL) {
111 free(buf);
112 store->driver->close(obj);
113 if (obj->heap_shell) { free(obj); }
114 errno = ENOMEM;
115 return NGX_ERROR;
116 }
117 buf = grown;
118 }
119 n = store->driver->pread(obj, buf + got, cap - got, (off_t) got);
120 if (n < 0) {
121 int e = errno;
122
123 free(buf);
124 store->driver->close(obj);
125 if (obj->heap_shell) { free(obj); }
126 errno = e ? e : EIO;
127 return NGX_ERROR;
128 }
129 if (n == 0) {
130 break;
131 }
132 got += (size_t) n;
133 }
134 store->driver->close(obj);
135 if (obj->heap_shell) { free(obj); }
136
137 if (got == 0) {
138 free(buf);
139 return NGX_DECLINED;
140 }
141 *out = buf;
142 *out_len = got;
143 return NGX_OK;
144}

148ngx_int_t
149brix_xmeta_save(brix_sd_instance_t *store, const char *key,
150 const brix_xmeta_t *m)
151{
152 uint8_t *buf = NULL;
153 size_t len = 0;
154 ngx_int_t rc;
155
156 if (store == NULL || store->driver == NULL || key == NULL || m == NULL) {
157 errno = EINVAL;
158 return NGX_ERROR;
159 }
160 if (brix_xmeta_encode(m, &buf, &len) != BRIX_XMETA_OK) {
161 return NGX_ERROR;
162 }
163
164 if (store->driver->setxattr != NULL && len <= BRIX_XMETA_XATTR_MAX) {
165 rc = store->driver->setxattr(store, key, BRIX_XMETA_XATTR_NAME,
166 buf, len, 0);
167 if (rc == NGX_OK) {
168 free(buf);
169 return NGX_OK;
170 }
171 if (!xmeta_xattr_unfit(errno)) {
172 int e = errno;
173
174 free(buf);
175 errno = e;
176 return NGX_ERROR;
177 }
178 /* doesn't fit / not supported here: ride the sidecar instead */
179 }
180
181 rc = xmeta_sidecar_write(store, key, buf, len);
182 free(buf);
183 if (rc != NGX_OK) {
184 return NGX_ERROR;
185 }
186 /* exactly one carrier: drop any (stale or just-outgrown) xattr copy */
187 if (store->driver->removexattr != NULL) {
188 (void) store->driver->removexattr(store, key,
189 BRIX_XMETA_XATTR_NAME);
190 }
191 return NGX_OK;
192}

195brix_xmeta_load(brix_sd_instance_t *store, const char *key,
196 brix_xmeta_t *m)
197{
198 uint8_t *buf;
199 ssize_t n;
200 size_t len = 0;
201 ngx_int_t rc;
202 int drc;
203
204 if (store == NULL || store->driver == NULL || key == NULL || m == NULL) {
205 errno = EINVAL;
206 return NGX_ERROR;
207 }
208
209 /* xattr carrier first */
210 if (store->driver->getxattr != NULL) {
211 buf = malloc(BRIX_XMETA_XATTR_MAX);
212 if (buf == NULL) {
213 errno = ENOMEM;
214 return NGX_ERROR;
215 }
216 n = store->driver->getxattr(store, key, BRIX_XMETA_XATTR_NAME,
217 buf, BRIX_XMETA_XATTR_MAX);
218 if (n > 0) {
219 drc = brix_xmeta_decode(buf, (size_t) n, m);
220 free(buf);
221 if (drc == BRIX_XMETA_OK) {
222 return NGX_OK;
223 }
224 return (drc == BRIX_XMETA_FOREIGN) ? NGX_DECLINED : NGX_ERROR;
225 }
226 free(buf);
227 if (n < 0 && errno != ENODATA && errno != ENOENT
228#ifdef ENOATTR
229 && errno != ENOATTR
230#endif
231 && !xmeta_xattr_unfit(errno))
232 {
233 return NGX_ERROR;
234 }
235 /* absent / unsupported: try the sidecar */
236 }
237
238 rc = xmeta_sidecar_read(store, key, &buf, &len);
239 if (rc != NGX_OK) {
240 return rc;
241 }
242 drc = brix_xmeta_decode(buf, len, m);
243 free(buf);
244 if (drc == BRIX_XMETA_OK) {
245 return NGX_OK;
246 }
247 return (drc == BRIX_XMETA_FOREIGN) ? NGX_DECLINED : NGX_ERROR;
248}
