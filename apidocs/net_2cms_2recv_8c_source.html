19/*
20 * cms_node_exec_driver — apply a manager-forwarded namespace op through a
21 * NON-default backend driver's slots, so a pblock/object data node mutates its
22 * catalog instead of the real filesystem (the confined *_beneath helpers the
23 * POSIX path uses only touch the real FS). plan->path/path2 are export-relative
24 * (leading slash), the format the driver slots expect. Returns 0 / -1+errno; sets
25 * *handled=0 only for an action the driver cannot express. The driver namespace
26 * is inherently export-confined, so a hostile manager still cannot escape it.
27 */
28static int
29cms_node_exec_driver(brix_sd_instance_t *sd, const char *root_canon,
30 const brix_cms_node_plan_t *plan, ngx_log_t *log, int *handled)
31{
32 const brix_sd_driver_t *drv = sd->driver;
33
34 *handled = 1;
35
36 switch (plan->action) {
37 case XRDCMS_NACT_MKDIR:
38 if (drv->mkdir == NULL) { errno = ENOSYS; return -1; }
39 return drv->mkdir(sd, plan->path, plan->mode) == NGX_OK ? 0 : -1;
40
41 case XRDCMS_NACT_MKPATH:
42 /* create the whole path + missing parents in the driver namespace. */
43 return brix_vfs_backend_mkpath(root_canon, plan->path, plan->mode, log);
44
45 case XRDCMS_NACT_RMDIR:
46 if (drv->unlink == NULL) { errno = ENOSYS; return -1; }
47 return drv->unlink(sd, plan->path, 1) == NGX_OK ? 0 : -1;
48
49 case XRDCMS_NACT_RM:
50 if (drv->unlink == NULL) { errno = ENOSYS; return -1; }
51 return drv->unlink(sd, plan->path, 0) == NGX_OK ? 0 : -1;
52
53 case XRDCMS_NACT_MV:
54 if (drv->rename == NULL) { errno = ENOSYS; return -1; }
55 return drv->rename(sd, plan->path, plan->path2, 0) == NGX_OK ? 0 : -1;
56
57 case XRDCMS_NACT_CHMOD: {
58 brix_sd_setattr_t attr;
59 if (drv->setattr == NULL) { return 0; } /* no mutable metadata — no-op */
60 ngx_memzero(&attr, sizeof(attr));
61 attr.set_mode = 1;
62 attr.mode = plan->mode;
63 return drv->setattr(sd, plan->path, &attr) == NGX_OK ? 0 : -1;
64 }
65
66 case XRDCMS_NACT_TRUNC: {
67 int err = 0;
68 int rc;
69 int saved;
70 brix_sd_obj_t *o;
71
72 if (drv->open == NULL || drv->ftruncate == NULL) { errno = ENOSYS; return -1; }
73 o = drv->open(sd, plan->path, BRIX_SD_O_WRITE, 0, &err);
74 if (o == NULL) { errno = err ? err : EIO; return -1; }
75 rc = drv->ftruncate(o, (off_t) plan->size) == NGX_OK ? 0 : -1;
76 saved = errno;
77 if (drv->close != NULL) { drv->close(o); }
78 errno = saved;
79 return rc;
80 }
81
82 default:
83 *handled = 0;
84 return -1;
85 }
86}

88/* cms_node_exec_forward — execute a manager-forwarded namespace op (Plane B): decode
89 * a kYR_chmod/mkdir/mkpath/mv/rm/rmdir/trunc and apply it to local storage UNDER
90 * KERNEL CONFINEMENT (rrdata_parse → pure node_plan → src/path/beneath.h confined
91 * helpers, openat2 RESOLVE_BENEATH), replying like stock cmsd: silent on success,
92 * kYR_error (kYR_EINVAL + strerror) on failure. A hostile manager cannot make the
93 * node mutate outside its export root — an escape fails EXDEV and becomes kYR_error. */
94static ngx_int_t
95cms_node_exec_forward(ngx_brix_cms_ctx_t *ctx, u_char code, uint32_t streamid,
96 const u_char *payload, size_t plen)
97{
98 brix_cms_rrdata_t d;
99 brix_cms_node_plan_t plan;
100 int rootfd = ctx->conf->rootfd;
101 const char *root_canon = ctx->conf->common.root_canon;
102 int rc = 0;
103
104 if (brix_cms_rrdata_parse(code, payload, plen, &d) != 0
105 || brix_cms_node_plan(code, &d, &plan) != 0)
106 {
107 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
108 "badly formed request");
109 }
110
111 /* A manager-only node (no local export) cannot satisfy a mutation. */
112 if (rootfd < 0) {
113 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
114 "no local storage");
115 }
116
117 /* A non-POSIX export (pblock/object) routes the mutation through its driver's
118 * namespace slots; the default POSIX export keeps the confined *_beneath path
119 * below unchanged. */
120 {
121 brix_sd_instance_t *sd =
122 brix_vfs_backend_resolve(root_canon, ctx->cycle->log);
123
124 if (sd != NULL && sd->driver != brix_sd_default_driver()) {
125 int handled;
126 rc = cms_node_exec_driver(sd, root_canon, &plan, ctx->cycle->log,
127 &handled);
128 if (!handled) {
129 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
130 "unsupported operation");
131 }
132 if (rc != 0) {
133 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
134 "brix: CMS node: forwarded op code=%ui path=%s failed: %s",
135 (ngx_uint_t) code, plan.path, strerror(errno));
136 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
137 strerror(errno));
138 }
139 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
140 "brix: CMS node: forwarded op code=%ui path=%s OK (driver)",
141 (ngx_uint_t) code, plan.path);
142 return NGX_OK;
143 }
144 }
145
146 switch (plan.action) {
147 case XRDCMS_NACT_MKDIR:
148 rc = brix_mkdir_beneath(rootfd, plan.path, plan.mode);
149 break;
150 case XRDCMS_NACT_MKPATH: {
151 char full[PATH_MAX];
152 brix_beneath_full_path(root_canon, plan.path, full, sizeof(full));
153 rc = brix_mkdir_recursive_beneath(ctx->cycle->log, rootfd, root_canon,
154 full, plan.mode, NULL);
155 break;
156 }
157 case XRDCMS_NACT_RMDIR:
158 rc = brix_vfs_unlink_at(rootfd, plan.path, 1);
159 break;
160 case XRDCMS_NACT_RM:
161 rc = brix_vfs_unlink_at(rootfd, plan.path, 0);
162 break;
163 case XRDCMS_NACT_MV:
164 rc = brix_rename_beneath(rootfd, plan.path, plan.path2);
165 break;
166 case XRDCMS_NACT_CHMOD: {
167 int fd = brix_vfs_open_fd_at(rootfd, plan.path, O_RDONLY, 0);
168 if (fd < 0) { rc = -1; }
169 else { rc = fchmod(fd, plan.mode); close(fd); } /* vfs-seam-allow: metadata on a VFS-opened confined fd */
170 break;
171 }
172 case XRDCMS_NACT_TRUNC: {
173 int fd = brix_vfs_open_fd_at(rootfd, plan.path, O_WRONLY, 0);
174 if (fd < 0) { rc = -1; }
175 else { rc = ftruncate(fd, (off_t) plan.size); close(fd); } /* vfs-seam-allow: metadata on a VFS-opened confined fd */
176 break;
177 }
178 default:
179 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
180 "unsupported operation");
181 }
182
183 if (rc != 0) {
184 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
185 "brix: CMS node: forwarded op code=%ui path=%s failed: %s",
186 (ngx_uint_t) code, plan.path, strerror(errno));
187 /* byte-exact: ecode is always kYR_EINVAL; text carries strerror. */
188 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
189 strerror(errno));
190 }
191
192 /* Success: stay silent — exactly as cmsd Execute() does on a NULL return
193 * from a non-forwarding leaf node. */
194 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
195 "brix: CMS node: forwarded op code=%ui path=%s ok",
196 (ngx_uint_t) code, plan.path);
197 return NGX_OK;
198}

200/* cms_wake_pending_session — parse the first host:port from a kYR_select/kYR_try
201 * payload and wake the suspended XRootD client waiting on its locate: look up the
202 * pending entry by streamid+pid, resolve its saved fd to the live connection (same
203 * worker, per-worker design), set XRD_ST_REQ_HEADER, brix_send_redirect to the
204 * resolved server, and resume reading. */
205
206static ngx_int_t
207cms_wake_pending_session(ngx_brix_cms_ctx_t *cms_ctx, uint32_t streamid,
208 const char *host, uint16_t port)
209{
210 brix_pending_locate_t *pending;
211 ngx_connection_t *client_conn;
212 ngx_stream_session_t *session;
213 brix_ctx_t *xrd_ctx;
214 int conn_fd;
215 ngx_atomic_uint_t conn_number;
216 u_char client_streamid[2];
217
218 pending = brix_pending_lookup(streamid, ngx_pid);
219 if (pending == NULL) {
220 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, cms_ctx->cycle->log, 0,
221 "brix: CMS wake: streamid=%uD not found in pending table",
222 streamid);
223 return NGX_OK; /* session timed out and was already removed */
224 }
225
226 conn_fd = pending->conn_fd;
227 conn_number = pending->conn_number;
228 client_streamid[0] = pending->client_streamid[0];
229 client_streamid[1] = pending->client_streamid[1];
230 brix_pending_unlock();
231
232 brix_pending_remove(streamid, ngx_pid);
233
234 client_conn = cms_find_client_connection(conn_fd);
235 if (client_conn == NULL || client_conn->number != conn_number) {
236 return NGX_OK; /* fd was recycled after the client disconnected */
237 }
238
239 session = client_conn->data;
240 if (session == NULL) {
241 return NGX_OK;
242 }
243
244 xrd_ctx = ngx_stream_get_module_ctx(session, ngx_stream_brix_module);
245 if (xrd_ctx == NULL || xrd_ctx->state != XRD_ST_WAITING_CMS) {
246 return NGX_OK;
247 }
248
249 /*
250 * WS6: the redirect host comes straight from the manager's kYR_select /
251 * kYR_try payload and is copied verbatim into the "Shost:port" redirect the
252 * client parses. A compromised/hostile manager could inject control bytes or
253 * an alternate scheme here, so validate it with the same character allowlist
254 * the registry uses as its store choke point (brix_net_host_chars_valid).
255 * On reject, drop the redirect and leave the client in XRD_ST_WAITING_CMS to
256 * hit its own cms_locate_timeout — we never emit a poisoned host.
257 */
258 if (host == NULL
259 || !brix_net_host_chars_valid(host, ngx_strlen(host)))
260 {
261 char safe[256];
262 brix_sanitize_log_string(host, safe, sizeof(safe));
263 ngx_log_error(NGX_LOG_WARN, cms_ctx->cycle->log, 0,
264 "brix: CMS select: rejected redirect to invalid host "
265 "\"%s\" for fd=%d", safe, conn_fd);
266 return NGX_OK;
267 }
268
269 ngx_log_error(NGX_LOG_INFO, cms_ctx->cycle->log, 0,
270 "brix: CMS select: redirecting client fd=%d to %s:%u",
271 conn_fd, host, (unsigned) port);
272
273 ngx_del_timer(client_conn->read);
274 xrd_ctx->state = XRD_ST_REQ_HEADER;
275 xrd_ctx->recv.cur_streamid[0] = client_streamid[0];
276 xrd_ctx->recv.cur_streamid[1] = client_streamid[1];
277 if (brix_send_redirect(xrd_ctx, client_conn, host, port) == NGX_ERROR) {
278 ngx_log_error(NGX_LOG_ERR, cms_ctx->cycle->log, 0,
279 "brix: CMS select: failed to queue redirect for fd=%d",
280 conn_fd);
281 return NGX_ERROR;
282 }
283 brix_schedule_read_resume(client_conn);
284 return NGX_OK;
285}

287static ngx_connection_t *
288cms_find_client_connection(int fd)
289{
290 ngx_uint_t i;
291 ngx_connection_t *c;
292
293 if (fd < 0) {
294 return NULL;
295 }
296
297 if (ngx_cycle->files != NULL && (ngx_uint_t) fd < ngx_cycle->files_n) {
298 c = ngx_cycle->files[fd];
299 if (c != NULL && c->fd == fd) {
300 return c;
301 }
302 return NULL;
303 }
304

308 return c;
309 }
310 }
311
312 return NULL;

315/* ngx_brix_cms_process_frame — decode a complete CMS frame's streamid + rrCode
316 * (first 4 bytes + offset 4) and dispatch by opcode: PING→PONG, SPACE→AVAIL,
317 * STATUS→suspend/resume conf flags, SELECT/TRY→client redirect. Unknown opcodes are
318 * silently ignored (debug log). Handlers are inline to keep recv.c self-contained. */
319
320static ngx_int_t
321ngx_brix_cms_process_frame(ngx_brix_cms_ctx_t *ctx)
322{
323 uint32_t streamid;
324 u_char code;
325
326 streamid = ngx_brix_cms_get32(ctx->inbuf);
327 code = ctx->inbuf[4];
328
329 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
330 "brix: CMS process frame code=%ui streamid=%uD",
331 (ngx_uint_t) code, streamid);
332
333 switch (code) {
334 case CMS_RR_PING:
335 return ngx_brix_cms_send_pong(ctx, streamid);
336
337 case CMS_RR_SPACE:
338 return ngx_brix_cms_send_avail(ctx, streamid);
339
340 case CMS_RR_STATUS: {
341 u_char mod = ctx->inbuf[5];
342 if (mod & CMS_ST_SUSPEND) {
343 ctx->conf->cms.suspended = 1;
344 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
345 "brix: CMS suspend received — new logins paused");
346 } else if (mod & CMS_ST_RESUME) {
347 ctx->conf->cms.suspended = 0;
348 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
349 "brix: CMS resume received — accepting logins");
350 } else {
351 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
352 "brix: CMS status modifier=0x%02xi (no action)",
353 (ngx_uint_t) mod);
354 }
355 return NGX_OK;
356 }
357
358 case CMS_RR_SELECT: {
359 /*
360 * kYR_select payload: NUL-terminated hostname + 2-byte big-endian port.
361 * The manager has resolved the kYR_locate and named a specific server.
362 */
363 const u_char *payload = ctx->inbuf + NGX_BRIX_CMS_HDR_LEN;
364 size_t payload_len = ctx->in_need - NGX_BRIX_CMS_HDR_LEN;
365 char host[256];
366 size_t host_len;
367 uint16_t port;
368
369 if (payload_len < 3) {
370 /* need at least one host byte, a NUL, and two port bytes */
371 return NGX_OK;
372 }
373
374 ngx_cpystrn((u_char *) host, (u_char *) payload, sizeof(host));
375 host_len = ngx_strlen(host);
376
377 if (host_len + 3 > payload_len) {
378 /* port bytes would fall outside the received payload */
379 return NGX_OK;
380 }
381
382 port = ngx_brix_cms_get16(payload + host_len + 1);
383 return cms_wake_pending_session(ctx, streamid, host, port);
384 }
385
386 case CMS_RR_TRY: {
387 /*
388 * kYR_try: the manager offers an ordered list of alternatives.
389 * Each entry is a NUL-terminated hostname followed by a 2-byte
390 * big-endian port. Use only the first entry; the client will
391 * retry remaining entries if it cannot reach this one.
392 */
393 const u_char *payload = ctx->inbuf + NGX_BRIX_CMS_HDR_LEN;
394 size_t payload_len = ctx->in_need - NGX_BRIX_CMS_HDR_LEN;
395 char host[256];
396 size_t host_len;
397 uint16_t port;
398
399 if (payload_len < 3) {
400 return NGX_OK;
401 }
402
403 ngx_cpystrn((u_char *) host, (u_char *) payload, sizeof(host));
404 host_len = ngx_strlen(host);
405
406 if (host_len + 3 > payload_len) {
407 return NGX_OK;
408 }
409
410 port = ngx_brix_cms_get16(payload + host_len + 1);
411 return cms_wake_pending_session(ctx, streamid, host, port);
412 }
413
414 case CMS_RR_STATE: {
415 /*
416 * kYR_state (raw): the manager asks "do you hold <path>?" as part of
417 * on-demand selection. The payload is the raw NUL-terminated namespace
418 * path (no Pup framing). We answer kYR_have (echoing streamid = path
419 * hash) if we can serve the path, else stay silent so the manager won't
420 * select us — matching real cmsd.
421 *
422 * Two ways to "have" a path:
423 * - manager_mode (a sub-manager registered UP to a meta-manager):
424 * forward the query to our own server registry — if any registered
425 * leaf data node exports a prefix covering the path, we have it (the
426 * client will be redirected to us and we then redirect down to the
427 * leaf). This is what makes a multi-tier meta->nginx->leaf mesh
428 * resolve.
429 * - data node: the file exists on our local export filesystem.
430 */
431 const u_char *payload = ctx->inbuf + NGX_BRIX_CMS_HDR_LEN;
432 size_t plen = ctx->in_need - NGX_BRIX_CMS_HDR_LEN;
433 char pathz[1024];
434 size_t pl;
435
436 /* bounded length of the NUL-terminated path */
437 for (pl = 0; pl < plen && payload[pl] != '\0'; pl++) { /* void */ }
438 if (pl == 0 || payload[0] != '/' || pl >= sizeof(pathz)) {
439 return NGX_OK;
440 }
441
442 /* reject path traversal before touching the registry/filesystem */
443 {
444 size_t k;
445 for (k = 0; k + 1 < pl; k++) {
446 if (payload[k] == '.' && payload[k + 1] == '.') {
447 return NGX_OK;
448 }
449 }
450 }
451
452 ngx_memcpy(pathz, payload, pl);
453 pathz[pl] = '\0';
454
455 if (ctx->conf->manager_mode) {
456 char host[256];
457 uint16_t dport;
458 if (brix_srv_select(pathz, 0, host, sizeof(host), &dport)) {
459 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
460 "brix: CMS state(mgr): registry serves "
461 "\"%*s\", replying kYR_have", pl, payload);
462 return ngx_brix_cms_send_have(ctx, streamid, pathz, pl);
463 }
464 return NGX_OK;
465 }
466
467 {
468 struct stat st;
469
470 /*
471 * Kernel-confined existence probe. A malicious manager can ask
472 * "do you hold <path>?" for ANY path; the raw stat() this replaced
473 * followed symlinks, so a symlink planted under the export root
474 * (e.g. /link -> /etc) would make us answer kYR_have for a file
475 * OUTSIDE the root — a cross-root information leak and a
476 * cluster-poisoning vector. brix_stat_beneath() resolves the
477 * path under the persistent export rootfd with openat2
478 * RESOLVE_BENEATH, so any symlink or ".." that escapes the root is
479 * rejected by the kernel and we correctly stay silent. (The ".."
480 * pre-check above remains as cheap defence-in-depth.) A node with
481 * no local export root (rootfd < 0) never holds files locally.
482 */
483 if (ctx->conf->rootfd >= 0
484 && brix_stat_beneath(ctx->conf->rootfd, pathz, &st) == 0)
485 {
486 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
487 "brix: CMS state: have \"%*s\", "
488 "replying kYR_have", pl, payload);
489 return ngx_brix_cms_send_have(ctx, streamid, pathz, pl);
490 }
491 }
492
493 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
494 "brix: CMS state: do not have \"%*s\"",
495 pl, payload);
496 return NGX_OK;
497 }
498
499 case CMS_RR_CHMOD:
500 case CMS_RR_MKDIR:
501 case CMS_RR_MKPATH:
502 case CMS_RR_MV:
503 case CMS_RR_RM:
504 case CMS_RR_RMDIR:
505 case CMS_RR_TRUNC: {
506 /*
507 * Plane B: a manager-forwarded namespace mutation. Execute it under
508 * kernel confinement and reply silent-on-success / kYR_error-on-failure.
509 */
510 const u_char *payload = ctx->inbuf + NGX_BRIX_CMS_HDR_LEN;
511 size_t plen = ctx->in_need - NGX_BRIX_CMS_HDR_LEN;
512 return cms_node_exec_forward(ctx, code, streamid, payload, plen);
513 }
514
515 case CMS_RR_UPDATE:
516 /* Manager asks us to resend state (do_Update -> sendState). */
517 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
518 "brix: CMS node: update -> status");
519 return ngx_brix_cms_send_status(ctx);
520
521 case CMS_RR_DISC:
522 /* Manager requested disconnect (do_Disc on a node simply closes). */
523 ngx_log_error(NGX_LOG_NOTICE, ctx->cycle->log, 0,
524 "brix: CMS node: manager requested disconnect");
525 ngx_brix_cms_disconnect(ctx);
526 ngx_brix_cms_schedule_retry(ctx);
527 return NGX_OK;
528
529 default: {
530 const brix_cms_route_t *r =
531 brix_cms_route_lookup(XRDCMS_ROLE_NODE, code);
532 if (r != NULL) {
533 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
534 "brix: CMS node: unhandled opcode '%s'", r->name);
535 } else {
536 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
537 "brix: ignoring CMS rrCode=%ui", (ngx_uint_t) code);
538 }
539 return NGX_OK;
540 }
541 }
542}

544/* ngx_brix_cms_read_handler — read event handler for the manager connection:
545 * accumulate bytes to a complete header, read the dlen-sized payload, and dispatch
546 * each frame via ngx_brix_cms_process_frame(); disconnect and retry on
547 * timeout/error. */
548
549void
550ngx_brix_cms_read_handler(ngx_event_t *ev)
551{
552 ngx_connection_t *c;
553 ngx_brix_cms_ctx_t *ctx;
554 ssize_t n;
555 uint16_t dlen;
556 ngx_uint_t processed = 0;
557
558 c = ev->data;
559 ctx = c->data;
560
561 if (ctx == NULL || ctx->connection != c) {
562 return;
563 }
564
565 ngx_log_debug3(NGX_LOG_DEBUG_EVENT, ev->log, 0,
566 "brix: CMS read handler timedout=%d in_pos=%uz in_need=%uz",
567 (int) ev->timedout, ctx->in_pos, ctx->in_need);
568
569 if (ev->timedout) {
570 /*
571 * WS1: the manager went silent past cms_read_timeout (black-holed /
572 * half-open). Tear down and reconnect with backoff so we fail over
573 * instead of heartbeating into a dead socket forever.
574 */
575 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
576 "brix: CMS manager silent past read timeout — "
577 "reconnecting");
578 BRIX_RESIL_METRIC_INC(cms_read_timeouts_total);
579 ngx_brix_cms_disconnect(ctx);
580 ngx_brix_cms_schedule_retry(ctx);
581 return;
582 }
583
584 for ( ;; ) {
585 n = c->recv(c, ctx->inbuf + ctx->in_pos,
586 ctx->in_need - ctx->in_pos);
587
588 if (n == NGX_AGAIN) {
589 break;
590 }
591
592 if (n == NGX_ERROR || n == 0) {
593 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, ev->log, 0,
594 "brix: CMS recv EOF/error, disconnecting");
595 ngx_brix_cms_disconnect(ctx);
596 ngx_brix_cms_schedule_retry(ctx);
597 return;
598 }
599
600 ctx->in_pos += (size_t) n;
601
602 if (ctx->in_pos < ctx->in_need) {
603 continue;
604 }
605
606 if (ctx->in_need == NGX_BRIX_CMS_HDR_LEN) {
607 dlen = ngx_brix_cms_get16(ctx->inbuf + 6);
608
609 if ((size_t) dlen + NGX_BRIX_CMS_HDR_LEN
610 > NGX_BRIX_CMS_MAX_FRAME)
611 {
612 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
613 "brix: CMS frame too large: %ui",
614 (ngx_uint_t) dlen);
615 ngx_brix_cms_disconnect(ctx);
616 ngx_brix_cms_schedule_retry(ctx);
617 return;
618 }
619
620 ctx->in_need = NGX_BRIX_CMS_HDR_LEN + dlen;
621 if (ctx->in_pos < ctx->in_need) {
622 continue;
623 }
624 }
625
626 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, ev->log, 0,
627 "brix: CMS process_frame code=%ui",
628 (ngx_uint_t) ctx->inbuf[4]);
629
630 if (ngx_brix_cms_process_frame(ctx) != NGX_OK) {
631 ngx_brix_cms_disconnect(ctx);
632 ngx_brix_cms_schedule_retry(ctx);
633 return;
634 }
635
636 ctx->in_pos = 0;
637 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
638
639 /* WS1: a frame from the manager proves it is alive — reset the silence
640 * deadline so a responsive manager is never reconnected. */
641 ngx_brix_cms_arm_read_deadline(ctx);
642
643 /* A2: fairness — after a bounded number of frames, yield the worker to
644 * other connections and resume via a posted read event, so a flooding
645 * manager cannot monopolise the event loop. */
646 if (++processed >= NGX_BRIX_CMS_MAX_FRAMES_PER_WAKEUP) {
647 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
648 ngx_brix_cms_disconnect(ctx);
649 ngx_brix_cms_schedule_retry(ctx);
650 return;
651 }
652 BRIX_RESIL_METRIC_INC(cms_frame_yields_total);
653 ngx_post_event(c->read, &ngx_posted_events);
654 return;
655 }
656 }
657
658 if (ctx->connection != NULL
659 && ngx_handle_read_event(c->read, 0) != NGX_OK)
660 {
661 ngx_brix_cms_disconnect(ctx);
662 ngx_brix_cms_schedule_retry(ctx);
663 }
664}
