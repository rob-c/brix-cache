8/* WHAT: stat [-j] <path> — print file metadata in human or JSON format.
9 * WHY: -j enables machine-readable output for scripting and pipeline use.
10 * HOW: flags are parsed first; the path argument is whatever is left.
11 * On error no output goes to stdout so partial JSON is never emitted. */
12int
13do_stat(brix_conn *c, const char *cwd, int argc, char **argv)
14{
15 brix_status st;
16 brix_statinfo si;
17 char path[XRDC_PATH_MAX];
18 const char *arg = NULL;
19 int json = 0, i;
20
21 for (i = 1; i < argc; i++) {
22 if (strcmp(argv[i], "-j") == 0 || strcmp(argv[i], "--json") == 0) {
23 json = 1;
24 } else {
25 arg = argv[i];
26 }
27 }
28 if (arg == NULL) { fprintf(stderr, "usage: stat [-j] <path>\n"); return 50; }
29 build_path(cwd, arg, path, sizeof(path));
30
31 brix_status_clear(&st);
32 if (brix_stat(c, path, &si, &st) != 0) {
33 fprintf(stderr, "xrdfs: stat %s: %s\n", path, st.msg);
34 xrdfs_op_hints(&st, 0, c); /* WS-3/WS-7 */
35 return brix_shellcode(&st);
36 }
37 if (json) { json_statinfo(path, &si); } else { print_statinfo(path, &si); }
38 return 0;
39}

42/* ---- Print one ls directory entry line ----
43 *
44 * WHAT: Emits a single directory entry to stdout — the long "drw SIZE
45 * prefixNAME" form when want_long is set and the entry carries stat data,
46 * otherwise the plain "prefixNAME" form. Returns nothing.
47 *
48 * WHY: The per-entry formatting is the branchiest part of listing a directory;
49 * pulling it out keeps ls_print_dir's control flow (dirlist, loop,
50 * recurse) flat and under the complexity cap.
51 *
52 * HOW: 1. When want_long and the entry has stat data, derive the type and rwx
53 * glyphs from the flags and format the size through fmt_size.
54 * 2. Otherwise print just the prefixed name.
55 */
56static void
57ls_print_entry(const brix_dirent *ent, const char *prefix, int want_long,
58 int human)
59{
60 if (want_long && ent->have_stat) {
61 int f = ent->st.flags;
62 char td = (f & kXR_isDir) ? 'd' : '-';
63 char r = (f & kXR_readable) ? 'r' : '-';
64 char w = (f & kXR_writable) ? 'w' : '-';
65 char szs[32];
66 fmt_size(ent->st.size, szs, sizeof(szs), human);
67 printf("%c%c%c %12s %s%s\n", td, r, w, szs, prefix, ent->name);
68 } else {
69 printf("%s%s\n", prefix, ent->name);
70 }
71}

74/* ---- Recurse into each subdirectory (ls -R) ----
75 *
76 * WHAT: For every child of an already-listed directory that is a real
77 * subdirectory, prints a "fullpath:" section header and lists it
78 * recursively. Returns 0 on success, -1 if any recursive listing fails.
79 *
80 * WHY: Isolating the descent loop from the entry-printing loop keeps each loop
81 * single-purpose and holds ls_print_dir under the complexity cap.
82 *
83 * HOW: 1. Skip "." / ".." and any entry not flagged kXR_isDir with stat data
84 * (directory symlinks report kXR_other and are listed, not descended).
85 * 2. Join the parent path and child name; skip entries that overflow.
86 * 3. Print the section header and recurse, propagating a failure as -1.
87 */
88static int
89ls_recurse_subdirs(brix_conn *c, const char *path, const brix_dirent *ents,
90 size_t n, int want_long, int human, brix_status *st)
91{
92 size_t k;
93
94 for (k = 0; k < n; k++) {
95 char full[XRDC_PATH_MAX];
96 if (is_dot(ents[k].name)
97 || !(ents[k].have_stat && (ents[k].st.flags & kXR_isDir))) {
98 continue;
99 }
100 if (join_path(path, ents[k].name, full, sizeof(full)) != 0) {
101 continue;
102 }
103 printf("\n%s:\n", full);
104 if (ls_print_dir(c, full, want_long, 1, human, st) != 0) {
105 return -1;
106 }
107 }
108 return 0;
109}

112/* Print one directory's entries; if recursive, descend into each subdir under a
113 * "fullpath:" section header (classic ls -R). 0 / -1. */
114
115int
116ls_print_dir(brix_conn *c, const char *path, int want_long, int recursive,
117 int human, brix_status *st)
118{
119 brix_dirent *ents = NULL;
120 size_t n = 0, k;
121 char prefix[XRDC_PATH_MAX + 2];
122 size_t plen;
123 const char *sep;
124
125 if (brix_dirlist(c, path, (want_long || recursive), &ents, &n, st) != 0) {
126 return -1;
127 }
128 plen = strlen(path);
129 sep = (plen > 0 && path[plen - 1] == '/') ? "" : "/";
130 snprintf(prefix, sizeof(prefix), "%s%s", path, sep);
131
132 for (k = 0; k < n; k++) {
133 ls_print_entry(&ents[k], prefix, want_long, human);
134 }
135 if (recursive
136 && ls_recurse_subdirs(c, path, ents, n, want_long, human, st) != 0) {
137 free(ents);
138 return -1;
139 }
140 free(ents);
141 return 0;
142}

145/* WHAT: flat JSON array of one directory's entries for ls -j.
146 * WHY: brix_dirlist returns all entries at once, so no partial output can
147 * reach stdout on error — the array is complete or nothing is emitted.
148 * HOW: forces want_stat=1; entries with no stat data emit -1 sentinel values.
149 * Name strings go through brix_json_kv_str so embedded quotes and control
150 * bytes are safely escaped (security requirement). */
151static int
152ls_json_dir(brix_conn *c, const char *path, brix_status *st)
153{
154 brix_dirent *ents = NULL;
155 size_t n = 0, k;
156 char prefix[XRDC_PATH_MAX + 2];
157 size_t plen;
158 const char *sep;
159 int first = 1;
160
161 if (brix_dirlist(c, path, 1 /*want_stat*/, &ents, &n, st) != 0) {
162 return -1;
163 }
164 plen = strlen(path);
165 sep = (plen > 0 && path[plen - 1] == '/') ? "" : "/";
166 snprintf(prefix, sizeof(prefix), "%s%s", path, sep);
167
168 fputc('[', stdout);
169 for (k = 0; k < n; k++) {
170 char fullname[XRDC_PATH_MAX];
171 int is_dir = ents[k].have_stat && (ents[k].st.flags & kXR_isDir);
172
173 if ((size_t) snprintf(fullname, sizeof(fullname), "%s%s",
174 prefix, ents[k].name) >= sizeof(fullname)) {
175 continue; /* skip overlong paths rather than truncating silently */
176 }
177 if (!first) { fputc(',', stdout); }
178 first = 0;
179
180 fputc('{', stdout);
181 brix_json_kv_str(stdout, "name", fullname, 1);
182 brix_json_kv_ll(stdout, "size",
183 ents[k].have_stat ? (long long) ents[k].st.size : -1LL, 1);
184 brix_json_kv_ll(stdout, "mtime",
185 ents[k].have_stat ? (long long) ents[k].st.mtime : -1LL, 1);
186 brix_json_kv_bool(stdout, "is_dir", is_dir, 0);
187 fputc('}', stdout);
188 }
189 fputs("]\n", stdout);
190 free(ents);
191 return 0;
192}

195/* WHAT: ls [-l] [-R] [-h] [-j] [path] — list directory entries.
196 * WHY: -j enables machine-readable JSON output for scripting.
197 * HOW: -j dispatches to ls_json_dir (flat array, safe escaping);
198 * all other flag combinations go through the existing ls_print_dir path. */
199int
200do_ls(brix_conn *c, const char *cwd, int argc, char **argv)
201{
202 brix_status st;
203 char path[XRDC_PATH_MAX];
204 const char *arg = NULL;
205 int want_long = 0, recursive = 0, human = 0, json = 0, i;
206
207 for (i = 1; i < argc; i++) {
208 if (strcmp(argv[i], "-l") == 0) { want_long = 1; }
209 else if (strcmp(argv[i], "-R") == 0) { recursive = 1; }
210 else if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--human") == 0) { human = 1; }
211 else if (strcmp(argv[i], "-j") == 0 || strcmp(argv[i], "--json") == 0) {
212 json = 1;
213 } else if (strcmp(argv[i], "-lR") == 0 || strcmp(argv[i], "-Rl") == 0) {
214 want_long = 1; recursive = 1;
215 } else { arg = argv[i]; }
216 }
217 build_path(cwd, arg != NULL ? arg : ".", path, sizeof(path));
218
219 brix_status_clear(&st);
220 if (json) {
221 if (ls_json_dir(c, path, &st) != 0) {
222 fprintf(stderr, "xrdfs: ls %s: %s\n", path, st.msg);

224 return brix_shellcode(&st);
225 }
226 return 0;
227 }
228 if (ls_print_dir(c, path, want_long, recursive, human, &st) != 0) {
229 fprintf(stderr, "xrdfs: ls %s: %s\n", path, st.msg);
230 xrdfs_op_hints(&st, 0, c); /* WS-3/WS-7 */
231 return brix_shellcode(&st);
232 }
233 return 0;

273/* WHAT: rm [-r] [-v] <path> — delete a file, or a whole tree with -r.
274 * WHY: rmdir only takes empty dirs; users cleaning a tree need one command.
275 * HOW: -r stats the target; directories go through brix_rmtree (post-order,
276 * depth-capped). The resolved export root "/" is always refused.
277 * brix_rmtree probes each kXR_isDir entry with brix_lstat before
278 * descending: directory symlinks (kXR_other) are unlinked, not descended,
279 * matching POSIX `rm -r` semantics. */
280int
281do_rm(brix_conn *c, const char *cwd, int argc, char **argv)
282{
283 brix_status st;
284 char path[XRDC_PATH_MAX];
285 int recursive = 0, verbose = 0, i;
286 const char *arg = NULL;
287
288 for (i = 1; i < argc; i++) {
289 if (strcmp(argv[i], "-r") == 0 || strcmp(argv[i], "-R") == 0) {
290 recursive = 1;
291 } else if (strcmp(argv[i], "-v") == 0 || strcmp(argv[i], "--verbose") == 0) {
292 verbose = 1;
293 } else {
294 arg = argv[i];
295 }
296 }
297 if (arg == NULL) {
298 fprintf(stderr, "usage: rm [-r] [-v, --verbose] <path>\n");
299 return 50;
300 }
301 build_path(cwd, arg, path, sizeof(path));
302 brix_status_clear(&st);
303 if (recursive) {
304 brix_statinfo si;
305 if (strcmp(path, "/") == 0) {
306 fprintf(stderr, "xrdfs: rm -r: refusing to delete the export root\n");
307 return 50;
308 }
309 if (brix_stat(c, path, &si, &st) != 0) {
310 fprintf(stderr, "xrdfs: rm %s: %s\n", path, st.msg);
311 brix_cred_hint_for_status(&st, 1, stderr); /* Phase 40 (c) */
312 return brix_shellcode(&st);
313 }
314 if (si.flags & kXR_isDir) {
315 if (brix_rmtree(c, path, 0, verbose ? rm_report : NULL, NULL,
316 &st) != 0) {
317 fprintf(stderr, "xrdfs: rm -r %s: %s\n", path, st.msg);
318 xrdfs_op_hints(&st, 1, c); /* WS-3/WS-7 */
319 return brix_shellcode(&st);
320 }
321 return 0;
322 }
323 /* -r on a plain file falls through to the single unlink */
324 }
325 if (brix_rm(c, path, &st) != 0) {
326 fprintf(stderr, "xrdfs: rm %s: %s\n", path, st.msg);
327 xrdfs_op_hints(&st, 1, c); /* WS-3/WS-7 */
328 return brix_shellcode(&st);
329 }
330 return 0;
331}

371/* chmod [-R] <path> <mode> — keeps the historical xrdfs arg order (path first).
372 * <mode> is the stock 9-char symbolic form ("rwxr-xr-x") or an octal absolute
373 * mode ("755"). -R recurses into directories, applying the same mode to every
374 * entry. */
375int
376do_chmod(brix_conn *c, const char *cwd, int argc, char **argv)
377{
378 brix_status st;
379 char path[XRDC_PATH_MAX];
380 const char *patharg = NULL, *modearg = NULL;
381 int recursive = 0, mode, i;
382
383 for (i = 1; i < argc; i++) {
384 if (strcmp(argv[i], "-R") == 0) { recursive = 1; }
385 else if (patharg == NULL) { patharg = argv[i]; }
386 else if (modearg == NULL) { modearg = argv[i]; }
387 }
388 if (patharg == NULL || modearg == NULL) {
389 fprintf(stderr, "usage: chmod [-R] <path> <rwxr-xr-x | octal-mode>\n");
390 return 50;
391 }
392 build_path(cwd, patharg, path, sizeof(path));
393 mode = parse_chmod_mode(modearg);
394 if (mode < 0) {
395 fprintf(stderr, "xrdfs: chmod: invalid mode '%s' "
396 "(expected rwxr-xr-x or octal)\n", modearg);
397 return 50;
398 }
399
400 brix_status_clear(&st);
401 if (brix_chmod(c, path, mode, &st) != 0) {
402 fprintf(stderr, "xrdfs: chmod %s: %s\n", path, st.msg);
403 xrdfs_op_hints(&st, 1, c); /* WS-3/WS-7 */
404 return brix_shellcode(&st);
405 }
406 if (recursive) {
407 int failures = 0;
408 brix_status wst;
409 brix_status_clear(&wst);
410 if (chmod_recursive(c, path, mode, &failures, &wst) != 0) {
411 fprintf(stderr, "xrdfs: chmod -R %s: %s\n", path, wst.msg);
412 return brix_shellcode(&wst);
413 }
414 if (failures > 0) { return 1; } /* per-entry errors already reported */
415 }
416 return 0;
417}

475/* df [-h] [path] — friendly disk-space report over kXR_Qspace (the server's oss.*
476 * capacity record). Default path "/". -h humanizes the byte columns. Falls back to
477 * printing the raw reply verbatim when the shape is unrecognized (never crashes).
478 * Cluster-wide aggregation (per-holder rows) is out of scope here — use `xrdmapc`. */
479int
480do_df(brix_conn *c, const char *cwd, int argc, char **argv)
481{
482 brix_status st;
483 char path[XRDC_PATH_MAX], reply[4096];
484 int64_t total, avail, used, largest;
485 int human = 0, i;
486 const char *arg = NULL;
487
488 for (i = 1; i < argc; i++) {
489 if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--human") == 0) { human = 1; }
490 else { arg = argv[i]; }
491 }
492 build_path(cwd, arg != NULL ? arg : "/", path, sizeof(path));
493
494 brix_status_clear(&st);
495 if (brix_query(c, kXR_Qspace, path, reply, sizeof(reply), &st) != 0) {
496 fprintf(stderr, "xrdfs: df %s: %s\n", path, st.msg);
497 return brix_shellcode(&st);
498 }
499 if (df_parse_space(reply, &total, &avail, &used, &largest) != 0) {
500 printf("%s\n", reply); /* unknown shape: honest raw passthrough */
501 return 0;
502 }
503 {
504 char ts[32], us[32], as[32], ls[32], pctbuf[8];
505 fmt_size(total >= 0 ? total : 0, ts, sizeof(ts), human);
506 fmt_size(used >= 0 ? used : 0, us, sizeof(us), human);
507 fmt_size(avail >= 0 ? avail : 0, as, sizeof(as), human);
508 fmt_size(largest >= 0 ? largest : 0, ls, sizeof(ls), human);
509 if (total > 0 && used >= 0) {
510 snprintf(pctbuf, sizeof(pctbuf), "%d%%", (int) ((used * 100) / total));
511 } else {
512 snprintf(pctbuf, sizeof(pctbuf), "-");
513 }
514 printf("%-12s %-12s %-12s %-6s %-12s %s\n",
515 "Size", "Used", "Avail", "Use%", "Largest", "Path");
516 printf("%-12s %-12s %-12s %-6s %-12s %s\n", ts, us, as, pctbuf, ls, path);
517 }
518 return 0;
519}

522/* ---- Parse touch's argument vector ----
523 *
524 * WHAT: Scans argv for -c/-a/-m/-t|--timestamp flags and the single path
525 * operand, writing each into the caller's out-params. Returns 0 on
526 * success, or 50 (after printing a diagnostic) on a malformed
527 * -t/--timestamp value.
528 *
529 * WHY: The flag scan — especially the -t branch that consumes the next argv
530 * slot and parses a timestamp — is what pushes do_touch over the
531 * complexity cap; isolating it keeps the command body linear.
532 *
533 * HOW: 1. Walk argv setting the boolean out-param for each recognised option.
534 * 2. For -t/--timestamp with a following argument, parse it into *tspec
535 * via touch_parse_time; on failure print the usage hint and return 50.
536 * 3. Treat any other token (including a bare -t with no following value,
537 * matching the original loop) as the path operand, last one winning.
538 */
539static int
540touch_parse_argv(int argc, char **argv, int *no_create, int *do_atime,
541 int *do_mtime, int *have_t, struct timespec *tspec,
542 const char **arg)
543{
544 int i;
545
546 for (i = 1; i < argc; i++) {
547 if (strcmp(argv[i], "-c") == 0) { *no_create = 1; }
548 else if (strcmp(argv[i], "-a") == 0) { *do_atime = 1; }
549 else if (strcmp(argv[i], "-m") == 0) { *do_mtime = 1; }
550 else if ((strcmp(argv[i], "-t") == 0 || strcmp(argv[i], "--timestamp") == 0)
551 && i + 1 < argc) {
552 if (touch_parse_time(argv[++i], tspec) != 0) {
553 fprintf(stderr, "xrdfs: touch: bad -t/--timestamp value '%s' "
554 "(want [[CC]YY]MMDDhhmm[.ss])\n", argv[i]);
555 return 50;
556 }
557 *have_t = 1;
558 } else { *arg = argv[i]; }
559 }
560 return 0;
561}

564/* ---- Populate the utimensat times[2] pair for touch ----
565 *
566 * WHAT: Fills times[0] (atime) and times[1] (mtime) with UTIME_NOW, an explicit
567 * parsed timestamp, or UTIME_OMIT according to which fields touch selected.
568 *
569 * WHY: Encapsulates the per-field UTIME_NOW / OMIT / explicit decision so
570 * do_touch reads as a flat sequence and stays under the complexity cap.
571 *
572 * HOW: 1. Zero tv_sec for both slots; set tv_nsec to UTIME_NOW when the field
573 * is selected without an explicit -t, UTIME_OMIT when unselected, and 0
574 * (a real seconds value follows) when an explicit -t was given.
575 * 2. When have_t is set, copy tspec into each selected slot.
576 */
577static void
578touch_fill_times(struct timespec times[2], int do_atime, int do_mtime,
579 int have_t, const struct timespec *tspec)
580{
581 times[0].tv_sec = times[1].tv_sec = 0;

591/* ---- Ensure the touch target exists before stamping ----
592 *
593 * WHAT: With no_create set, reports whether the file is absent; otherwise
594 * creates the file if it does not exist. Returns 1 when do_touch should
595 * stop early with success (-c on an absent file), 0 to proceed to setattr.
596 *
597 * WHY: POSIX touch treats -c on a missing file as a silent no-op and otherwise
598 * creates the file before stamping; isolating that keeps do_touch's error
599 * handling focused on the setattr call that actually reports failures.
600 *
601 * HOW: 1. Under -c, stat the path; an absent file returns 1 (no-op), an
602 * existing file returns 0 to continue to the stamp.
603 * 2. Otherwise open write with force=0 (kXR_new); on success close it. An
604 * already-existing file makes the open fail, which is ignored — the
605 * file is present and the caller's setattr still runs. Returns 0.
606 */
607static int
608touch_ensure_file(brix_conn *c, const char *path, int no_create)
609{
610 brix_status cs;
611
612 if (no_create) {
613 brix_statinfo si;
614 brix_status_clear(&cs);
615 if (brix_stat(c, path, &si, &cs) != 0) { return 1; }
616 return 0;
617 }
618 /* Create-if-absent: force=0 ⇒ kXR_new (fails if it already exists, which we
619 * ignore — the file is there and the setattr below still runs). */
620 {
621 brix_file f;
622 brix_status_clear(&cs);
623 if (brix_file_open_write(c, path, 0 /*force=new*/, 0 /*posc*/, &f, &cs) == 0) {
624 brix_file_close(c, &f, &cs);
625 }
626 }
627 return 0;
628}

631/* touch [-c] [-a] [-m] [-t STAMP] <path> — create the file if absent (unless -c) and
632 * set its access/modification times (default: both to now). -a/-m restrict to
633 * atime/mtime; -t sets an explicit [[CC]YY]MMDDhhmm[.ss] time. NEVER changes
634 * ownership: brix_setattr is always called with set_owner = 0. */
635int
636do_touch(brix_conn *c, const char *cwd, int argc, char **argv)
637{
638 brix_status st;
639 char path[XRDC_PATH_MAX];
640 struct timespec times[2], tspec;
641 const char *arg = NULL;
642 int no_create = 0, do_atime = 0, do_mtime = 0, have_t = 0, prc;
643
644 tspec.tv_sec = 0; /* unused unless have_t; init to silence -Wmaybe-uninit */
645 tspec.tv_nsec = 0;
646 prc = touch_parse_argv(argc, argv, &no_create, &do_atime, &do_mtime,
647 &have_t, &tspec, &arg);
648 if (prc != 0) {
649 return prc;
650 }
651 if (arg == NULL) {
652 fprintf(stderr, "usage: touch [-c] [-a] [-m] [-t/--timestamp STAMP] <path>\n");
653 return 50;
654 }

656 build_path(cwd, arg, path, sizeof(path));
657
658 /* atime = slot 0, mtime = slot 1; per-field UTIME_OMIT when not selected. */
659 touch_fill_times(times, do_atime, do_mtime, have_t, &tspec);
660
661 if (touch_ensure_file(c, path, no_create)) {

676/* ln [-s] [-f] <target> <linkpath> — create a hard link (default) or a symbolic link
677 * (-s), GNU arg order (target first). -f removes an existing linkpath first
678 * (best-effort, non-atomic). For -s the target is stored verbatim (link content, not

777/* xattr ls|get|set|rm — extended attributes via kXR_fattr (client/lib/fattr.c).
778 * xattr ls <path> list attribute names
779 * xattr get <path> <name> print one value
780 * xattr set <path> <name> <value> set/replace a value
781 * xattr rm <path> <name> delete an attribute
782 * `xattr <path>` with no subcommand is treated as `ls`. */
783int
784xattr_ls(brix_conn *c, const char *path)
785{

787 char names[8192];

789
790 brix_status_clear(&st);

803 }
