4 * WHAT: Declares the capability bitmap, the opaque driver/instance/object/dir/

13 * call "move these bytes / mutate this name" against a driver it selected
14 * at config time, while keeping all policy (confinement re-check, metrics,

22 * returned object handle from any dispatch tier. Phase 55.A ships this

78/* ---- capability bitmap ----------------------------------------------------
79 * A driver advertises what it can do; the VFS consults this to shape behaviour
80 * (e.g. only emit a sendfile buffer when CAP_SENDFILE). Absences are honest:
81 * the VFS degrades or rejects rather than emulating a missing primitive. */
82typedef enum {
83 BRIX_SD_CAP_FD = 1u << 0, /* exposes a real kernel fd */
84 /* CAP_SENDFILE implies CAP_FD: brix_sd_fd(obj) is a real seekable kernel
85 * fd valid as the source of sendfile(2) and of an nginx file-backed
86 * (b->in_file) buffer for any byte range. A backend without it MUST be
87 * served memory-backed; the VFS read path enforces that fallback. */
88 BRIX_SD_CAP_SENDFILE = 1u << 1,
89 BRIX_SD_CAP_RANDOM_WRITE = 1u << 2, /* pwrite at arbitrary offset */
90 BRIX_SD_CAP_RANGE_READ = 1u << 3, /* pread at arbitrary offset */
91 BRIX_SD_CAP_TRUNCATE = 1u << 4, /* ftruncate */
92 BRIX_SD_CAP_SERVER_COPY = 1u << 5, /* native copy (copy_file_range/COPY)*/
93 BRIX_SD_CAP_XATTR = 1u << 6, /* user.* xattrs / object metadata */
94 BRIX_SD_CAP_HARD_RENAME = 1u << 7, /* atomic rename (else copy+delete) */
95 BRIX_SD_CAP_DIRS = 1u << 8, /* real directories (else key-prefix)*/
96 BRIX_SD_CAP_APPEND = 1u << 9, /* O_APPEND semantics */
97 BRIX_SD_CAP_IOURING = 1u << 10, /* fd is io_uring-submittable */
98 BRIX_SD_CAP_FSCS = 1u << 11, /* filesystem page checksums (CSI) */
99 /* The backend is NEARLINE (tape/MSS): an object may be offline, so a read can
100 * fault a slow async recall instead of returning bytes. A nearline backend
101 * advertises this AND implements the recall slot below; the composing registry
102 * then REQUIRES a cache tier (the recall target, phase-64 P4/§9.4) in front of
103 * it. Drivers that always serve online leave this 0 (the common case). */
104 BRIX_SD_CAP_NEARLINE = 1u << 12, /* tape/MSS: reads may recall (§9) */
105 BRIX_SD_CAP_CATALOG = 1u << 13, /* native object-catalog enumeration */
106 /* phase-71 capability-uniformity: split implicit read-only/writable and
107 * memory-serve assumptions out of VFS backend-identity branches into caps. */
108 BRIX_SD_CAP_DIRS_WRITE = 1u << 14, /* mutable catalog: mkdir/rmdir/rename */
109 BRIX_SD_CAP_XATTR_WRITE = 1u << 15, /* set/remove xattr (read = CAP_XATTR) */
110 BRIX_SD_CAP_MEMFILE = 1u << 16 /* serve bytes memory-backed w/o CAP_FD */
111} brix_sd_cap_t;

145/* Per-open user credential passed from the protocol handler to the storage
146 * driver so a remote backend can authenticate AS the client user (Phase 1:
147 * x509 proxy; Phase 2 T2: WLCG bearer token). Extended with the fields needed
148 * to re-resolve the credential for async/deferred flushes.
149 *
150 * WHAT: Borrowed pointers valid for the duration of the open() / staged_open()
151 * call. Drivers that defer the open (thread-pool) MUST copy the strings
152 * internally before returning — the caller's buffers may be freed once
153 * the vtable function returns.
154 *
155 * WHY: A per-open cred lets the VFS pass identity down to the driver without
156 * threading it through every intermediate layer; the driver is the only
157 * entity that knows how to present it to a specific remote protocol.
158 * The extra fields (key, cred_dir, fallback_deny) let decorator layers
159 * (sd_stage, sd_cache) embed a re-resolvable identity into their durable
160 * state so an async flush — possibly after a crash and restart — can
161 * re-authenticate as the original user rather than the service account.
162 *
163 * HOW: The gate fills exactly ONE credential kind: {x509_proxy}, {bearer},
164 * {s3_ak + s3_sk (+ s3_region)}, or {ceph_keyring + ceph_user}
165 * depending on the kind selected by ucred_select; the other kinds'
166 * fields are NULL. Drivers check the kind they support (sd_xroot:
167 * x509_proxy then bearer; sd_remote/S3: s3_ak; sd_ceph: ceph_keyring) —
168 * only one kind is ever set for a given open.
169 * A NULL cred or a driver with no open_cred slot falls back to the plain
170 * open slot (service credential / anonymous).
171 * sd_xroot reads x509_proxy OR bearer + principal; sd_remote reads
172 * s3_ak/s3_sk/s3_region (phase-3 T3) to re-init its SigV4 signer per
173 * open instead of the export's static access_key/secret_key/region;
174 * sd_ceph reads ceph_keyring/ceph_user (ceph-peruser item) to open a
175 * per-user librados connection instead of the export's static
176 * user/keyring.
177 * The extra fields (key, cred_dir, fallback_deny) are consumed by
178 * sd_stage / sd_cache and are not required by sd_xroot, sd_remote, or
179 * sd_ceph. */

181/* How the per-open credential in brix_sd_cred_t was obtained — the strategy the
182 * VFS gate resolved for the backend leg (phase-70 §4). SELECT (the default, 0)
183 * is the pre-phase-70 directory-lookup behaviour, so every existing caller that
184 * leaves the struct zeroed keeps the same meaning. The other modes are set by
185 * the delegation gate (vfs_deleg.c) when the front door captured a forwardable
186 * credential:
187 * PASSTHROUGH — replay the exact credential the user presented (bearer bytes;
188 * a user-supplied full x509 proxy incl. private key);
189 * EXCHANGE — trade the inbound credential for a backend-valid one (RFC 8693
190 * token-exchange; S3 STS; GSSAPI krb5 forwarding);
191 * DELEGATE — obtain a fresh short-lived proxy via a GridSite handshake;
192 * MINT — mint a fresh short-lived proxy from a local CA;
193 * AUTO — dispatch by id->auth_method (§2 matrix).
194 * The field is advisory metadata for audit/metrics and for the async re-acquire
195 * record; the cred's byte/path fields still say WHICH credential to present. */

197 BRIX_CRED_SELECT = 0,
198 BRIX_CRED_PASSTHROUGH,
199 BRIX_CRED_EXCHANGE,
200 BRIX_CRED_DELEGATE,
201 BRIX_CRED_MINT,
202 BRIX_CRED_AUTO
203};

548/* Release a driver object obtained from driver->open() by a caller that holds it
549 * by POINTER (not the VFS, which adopts the object by value and frees the shell
550 * itself in vfs_open.c): close it via its own vtable, then free a heap-allocated
551 * shell (heap_shell=1, e.g. POSIX's malloc'd obj — allocated off inst->pool so a
552 * cache-fill thread never touches the thread-unsafe ngx_cycle->pool). NULL-safe;

561 o->driver->close(o);

583 * WHAT: Route through the cred-scoped slot when a per-user credential is
584 * present AND the driver implements it; else the plain slot.
585 * WHY: One forwarding rule shared by the VFS and every decorator, so no
586 * tier can accidentally drop the credential on the floor.
587 * HOW: cred+slot → *_cred; otherwise the legacy slot — UNLESS the caller is
588 * in DENY mode (cred->fallback_deny) and the plain slot would actually
589 * run (driver-><op> != NULL): that combination means a per-user op
590 * would silently execute on the shared service credential in a mode
591 * that explicitly forbids the fallback, so refuse instead (EACCES).
592 * Defensive hardening (phase-3 T1): today inert (no driver has the
593 * cred-slot-missing/plain-slot-present shape), but prevents a future
594 * driver from silently leaking onto the service credential. Does NOT
595 * change allow-mode (fallback_deny==0 keeps falling back) or the
596 * legitimate no-op cases (a NULL plain slot with no *_cred slot either
597 * means "this driver has nothing to do here", not a leak). */

599brix_sd_open_maybe_cred(brix_sd_instance_t *inst, const char *path,

601{
602 if (cred != NULL && inst->driver->open_cred != NULL) {
603 return inst->driver->open_cred(inst, path, sd_flags, mode, cred,
604 err_out);
605 }
606 if (cred != NULL && cred->fallback_deny
607 && inst->driver->open_cred == NULL && inst->driver->open != NULL)
608 {
609 if (err_out != NULL) {
610 *err_out = EACCES;
611 }
612 errno = EACCES;
613 return NULL;
614 }
615 if (inst->driver->open == NULL) {
616 if (err_out != NULL) {
617 *err_out = ENOSYS;
618 }
619 errno = ENOSYS;
620 return NULL;
621 }
622 return inst->driver->open(inst, path, sd_flags, mode, err_out);
623}

996/* ---- brix_sd_obj_preadv — vectored read with per-driver fallback ---------
997 *
998 * WHAT: Issues a positioned vectored read on `obj`, using the driver's native
999 * preadv slot when present, else emulating it with one driver->pread per
1000 * iovec. Returns total bytes read (short = EOF), or -1 with errno set
1001 * (ENOTSUP when the driver has no byte-read slot at all).
1002 *
1003 * WHY: preadv/preadv2 are OPTIONAL vtable slots — remote/object drivers
1004 * (sd_remote) implement only pread, and the read fan-out paths (pgread,
1005 * kXR_readv batches) previously called obj->driver->preadv unguarded,
1006 * which is a NULL call (SIGSEGV) on those backends. One shared fallback
1007 * keeps the emulation policy (stop on short read) in a single place.
1008 *
1009 * HOW: Native slot when non-NULL; otherwise fill each iovec with a pread
1010 * loop — a short-but-nonzero pread is NOT EOF (drivers may cap a single
1011 * request), only a 0-byte pread is, so keep reading until the iovec is
1012 * full or the driver reports EOF/error. A -1 mid-loop after bytes were
1013 * already delivered reports the bytes (POSIX readv semantics); the
1014 * caller's next call surfaces the error.
1015 */
1016static ngx_inline ssize_t
1017brix_sd_obj_preadv(brix_sd_obj_t *obj, const struct iovec *iov, int iovcnt,
1018 off_t off)
1019{
1020 ssize_t total = 0;
1021 int i;
1022 int eof = 0;
1023
1024 if (obj->driver->preadv != NULL) {
1025 return obj->driver->preadv(obj, iov, iovcnt, off);
1026 }
1027
1028 if (obj->driver->pread == NULL) {
1029 errno = ENOTSUP;
1030 return -1;
1031 }
1032
1033 for (i = 0; i < iovcnt && !eof; i++) {
1034 size_t filled = 0;
1035
1036 while (filled < iov[i].iov_len) {
1037 ssize_t n = obj->driver->pread(obj,
1038 (char *) iov[i].iov_base + filled,
1039 iov[i].iov_len - filled,
1040 off + total);
1041 if (n < 0) {
1042 if (total > 0) {
1043 return total;
1044 }
1045 return -1;
1046 }
1047 if (n == 0) {
1048 eof = 1; /* only a 0-byte pread means EOF */
1049 break;
1050 }
1051 filled += (size_t) n;
1052 total += n;
1053 }
1054 }
1055
1056 return total;
1057}
