1/*
2 * sd.h — Storage Driver (SD) interface: the pluggable layer below the VFS.
3 *
4 * WHAT: Declares the capability bitmap, the opaque driver/instance/object/dir/
5 * staged handle types, the POD stat/dirent descriptors, the driver vtable
6 * (brix_sd_driver_s), the small capability-gated accessor helpers, and
7 * the registry API that turns a backend name into a bound per-export
8 * instance. POSIX is the default driver (sd_posix.c); block/object drivers
9 * (phases 55.D/E) register the same way.
10 *
11 * WHY: The VFS (src/fs/) is the protocol-agnostic data plane, but it is still
12 * hard-wired to POSIX syscalls. This header is the seam that lets the VFS
13 * call "move these bytes / mutate this name" against a driver it selected
14 * at config time, while keeping all policy (confinement re-check, metrics,
15 * access log, cache, buffer shaping) above the seam. See
16 * docs/refactor/phase-55-storage-backend-abstraction.md.
17 *
18 * HOW: A driver is a static const brix_sd_driver_t with a caps bitmap and a
19 * flat table of function pointers. The registry (sd_registry.c) builds an
20 * brix_sd_instance_t per export by name; the VFS opens objects on the
21 * instance and runs the worker-safe raw ops (pread/pwrite/...) on the
22 * returned object handle from any dispatch tier. Phase 55.A ships this
23 * header + the POSIX driver + the registry, registered in the build but
24 * not yet wired into any VFS callsite (that is 55.B+).
25 */

76/* ---- capability bitmap ----------------------------------------------------
77 * A driver advertises what it can do; the VFS consults this to shape behaviour
78 * (e.g. only emit a sendfile buffer when CAP_SENDFILE). Absences are honest:
79 * the VFS degrades or rejects rather than emulating a missing primitive. */
80typedef enum {
81 BRIX_SD_CAP_FD = 1u << 0, /* exposes a real kernel fd */
82 /* CAP_SENDFILE implies CAP_FD: brix_sd_fd(obj) is a real seekable kernel
83 * fd valid as the source of sendfile(2) and of an nginx file-backed
84 * (b->in_file) buffer for any byte range. A backend without it MUST be
85 * served memory-backed; the VFS read path enforces that fallback. */
86 BRIX_SD_CAP_SENDFILE = 1u << 1,
87 BRIX_SD_CAP_RANDOM_WRITE = 1u << 2, /* pwrite at arbitrary offset */
88 BRIX_SD_CAP_RANGE_READ = 1u << 3, /* pread at arbitrary offset */
89 BRIX_SD_CAP_TRUNCATE = 1u << 4, /* ftruncate */
90 BRIX_SD_CAP_SERVER_COPY = 1u << 5, /* native copy (copy_file_range/COPY)*/
91 BRIX_SD_CAP_XATTR = 1u << 6, /* user.* xattrs / object metadata */
92 BRIX_SD_CAP_HARD_RENAME = 1u << 7, /* atomic rename (else copy+delete) */
93 BRIX_SD_CAP_DIRS = 1u << 8, /* real directories (else key-prefix)*/
94 BRIX_SD_CAP_APPEND = 1u << 9, /* O_APPEND semantics */
95 BRIX_SD_CAP_IOURING = 1u << 10, /* fd is io_uring-submittable */
96 BRIX_SD_CAP_FSCS = 1u << 11, /* filesystem page checksums (CSI) */
97 /* The backend is NEARLINE (tape/MSS): an object may be offline, so a read can
98 * fault a slow async recall instead of returning bytes. A nearline backend
99 * advertises this AND implements the recall slot below; the composing registry
100 * then REQUIRES a cache tier (the recall target, phase-64 P4/§9.4) in front of
101 * it. Drivers that always serve online leave this 0 (the common case). */
102 BRIX_SD_CAP_NEARLINE = 1u << 12, /* tape/MSS: reads may recall (§9) */
103 BRIX_SD_CAP_CATALOG = 1u << 13 /* native object-catalog enumeration */
104} brix_sd_cap_t;

124/* Residency of a nearline (tape/MSS) object — the online/offline model the VFS
125 * residency seam (brix_vfs_residency) exposes to protocol handlers so they can
126 * advertise tape state (the HTTP Tape REST API, S3 InvalidObjectState /
127 * x-amz-storage-class, root:// stat's nearline flag) WITHOUT forcing a recall. A
128 * non-nearline driver has no residency slot; the seam reports ONLINE for it (a
129 * plain disk/object export is always resident). */

134 BRIX_SD_RES_LOST = 3 /* the object is gone */

342/* Release a driver object obtained from driver->open() by a caller that holds it
343 * by POINTER (not the VFS, which adopts the object by value and frees the shell
344 * itself in vfs_open.c): close it via its own vtable, then free a heap-allocated
345 * shell (heap_shell=1, e.g. POSIX's malloc'd obj — allocated off inst->pool so a
346 * cache-fill thread never touches the thread-unsafe ngx_cycle->pool). NULL-safe;
347 * a pool-allocated shell (heap_shell=0) is just closed. */
348static inline void
349brix_sd_obj_release(brix_sd_obj_t *o)
350{
351 if (o == NULL) {
352 return;
353 }
354 if (o->driver != NULL && o->driver->close != NULL) {
355 o->driver->close(o);
356 }
357 if (o->heap_shell) {
358 free(o);
359 }
360}

443/* ---- brix_sd_posix_wrap — bind a bare fd to a stack POSIX SD object ------
444 *
445 * WHAT: Zero-initializes *obj and points it at the built-in POSIX driver over a
446 * plain fd, for dispatching a single fd op through the driver vtable.
447 *
448 * WHY: The VFS's fd-keyed I/O loops (pread_full/pwrite_full/write_counted) own
449 * the EINTR/short-I/O policy but must route the raw syscall through the
450 * Storage Driver seam. The POSIX raw slots touch only obj->fd, so a
451 * zero-initialized stack object with driver+fd set is sufficient and keeps
452 * the hot path allocation-free.
453 *
454 * HOW: ngx_memzero, set the driver pointer and fd.
455 */
