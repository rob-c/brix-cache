1/*
2 * guard_http_req.c — ngx_http_request_t -> guard_request_t normalization.
3 *
4 * WHAT: the builder both phase handlers share (method -> op-class, credential
5 * detection, sanitized path/ip) and the audit-log file writer.
6 * WHY: the guard core is protocol-agnostic; this file is the single place
7 * HTTP request shape is interpreted, so ACCESS and LOG phases can never
8 * disagree about what was requested.
9 * HOW: NUL-terminate r->uri into stack scratch, sanitize it via the shared
10 * brix_sanitize_log_string helper; copy addr_text (not NUL-terminated by
11 * nginx) likewise; detect credentials from the verified client cert or an
12 * Authorization header; timestamp + append via one ngx_write_fd.
13 */

17/* ---- HTTP method -> guard op-class ----
18 *
19 * WHAT: maps the nginx method flag to the guard's protocol-agnostic op-class
20 * (profile-independent baseline); anything unlisted is GUARD_OP_UNKNOWN.
21 *
22 * WHY: grammar rules are written against op-classes, not HTTP verbs, so the
23 * same ruleset semantics serve ARC, WebDAV, and root://.
24 *
25 * HOW: 1. switch on r->method's flag value.
26 */
27static guard_op_class_t
28method_to_op(ngx_uint_t method)
29{
30 switch (method) {
31 case NGX_HTTP_GET:
32 case NGX_HTTP_HEAD: return GUARD_OP_READ;
33 case NGX_HTTP_PUT:
34 case NGX_HTTP_POST: return GUARD_OP_WRITE;
35 case NGX_HTTP_DELETE: return GUARD_OP_DELETE;
36 case NGX_HTTP_PROPFIND: return GUARD_OP_LIST;
37 case NGX_HTTP_OPTIONS: return GUARD_OP_INFO;
38 default: return GUARD_OP_UNKNOWN;
39 }
40}

42/* ---- Detect a presented credential ----
43 *
44 * WHAT: returns 1 if the client presented a verified TLS client certificate
45 * or a non-empty Authorization header, else 0.
46 *
47 * WHY: cred_present separates anonymous scanner noise from failing-but-
48 * authenticated users in the audit trail (and future signal tuning).
49 *
50 * HOW: 1. Under SSL, ask nginx for the client-verify result and accept
51 * exactly "SUCCESS" (the ngx_ssl_get_client_verify token — covers
52 * ssl_verify_client optional/optional_no_ca setups).
53 * 2. Otherwise any non-empty Authorization header counts (bearer,
54 * basic — scheme is the backend's concern).
55 */
56static int
57cred_present(ngx_http_request_t *r)
58{
59#if (NGX_HTTP_SSL)
60 if (r->connection->ssl) {
61 ngx_str_t verify;
62
63 if (ngx_ssl_get_client_verify(r->connection, r->pool, &verify)
64 == NGX_OK
65 && verify.len == sizeof("SUCCESS") - 1
66 && ngx_strncmp(verify.data, "SUCCESS", verify.len) == 0)
67 {
68 return 1;
69 }
70 }
71#endif
72 return r->headers_in.authorization != NULL
73 && r->headers_in.authorization->value.len > 0;
74}

76/* ---- NUL-terminate an ngx_str_t into a fixed buffer ----
77 *
78 * WHAT: copies up to bufsz-1 bytes of src into buf and NUL-terminates;
79 * returns the copied length.
80 *
81 * WHY: both r->uri and addr_text are counted strings while the pure-C guard
82 * core takes C strings; truncation (never overflow) is the failure mode.
83 *
84 * HOW: 1. Clamp, ngx_memcpy, terminate.
85 */
86static size_t
87copy_str_z(const ngx_str_t *src, char *buf, size_t bufsz)
88{
89 size_t copy_len = src->len < bufsz - 1 ? src->len : bufsz - 1;
90
91 ngx_memcpy(buf, src->data, copy_len);
92 buf[copy_len] = '\0';
93 return copy_len;
94}

96/* ---- Build guard_request_t from the HTTP request ----
97 *
98 * WHAT: fills *out with ip/proto/op/path/cred_present; outcome starts
99 * PENDING with status 0 (the LOG handler overwrites both). pathbuf/ipbuf
100 * are caller-owned scratch that must outlive the classify call.
101 *
102 * WHY: single normalization point for both phase handlers, and the one place
103 * the wire-derived URI is sanitized (INVARIANT: every logged wire string
104 * goes through brix_sanitize_log_string).
105 *
106 * HOW: 1. NUL-terminate r->uri into local scratch (ngx_str_t is counted).
107 * 2. Sanitize into pathbuf — control/quote/backslash/non-ASCII bytes
108 * become \xNN so the audit line can never be split or forged.
109 * 3. Copy addr_text into ipbuf (nginx does not NUL-terminate it).
110 * 4. proto = configured profile token; "http" when no profile is set
111 * (the audit line's proto= field must never be empty for fail2ban).
112 */
113void
114ngx_http_brix_guard_build_request(ngx_http_request_t *r,
115 ngx_http_brix_guard_loc_conf_t *lcf, guard_request_t *out,
116 char *pathbuf, size_t pathbuf_sz, char *ipbuf, size_t ipbuf_sz)
117{
118 char uri_z[BRIX_GUARD_PATH_BUF];
119
120 copy_str_z(&r->uri, uri_z, sizeof(uri_z));
121 out->path_len = brix_sanitize_log_string(uri_z, pathbuf, pathbuf_sz);
122 out->path = pathbuf;
123
124 copy_str_z(&r->connection->addr_text, ipbuf, ipbuf_sz);
125 out->ip = ipbuf;
126
127 out->proto = lcf->profile.len ? (const char *) lcf->profile.data : "http";
128 out->op = method_to_op(r->method);
129 out->cred_present = cred_present(r);
130 out->outcome = OUTCOME_PENDING;
131 out->status_code = 0;
132}

134/* ---- Append one audit line to the configured log ----
135 *
136 * WHAT: formats req/reason via guard_audit_format and appends the line to
137 * the location's audit log; silently a no-op when no log is configured or
138 * the line does not fit.
139 *
140 * WHY: one atomic write per signal is the fail2ban contract — the file is
141 * opened O_APPEND by ngx_conf_open_file, and a single sub-PIPE_BUF
142 * write(2) never interleaves across workers.
143 *
144 * HOW: 1. Bail without a usable log fd.
145 * 2. Timestamp from nginx's cached ISO-8601 clock (no syscall,
146 * refreshed by the event loop; adapters own the clock per the
147 * guard-core contract).
148 * 3. guard_audit_format into a stack line; append '\n'; one
149 * ngx_write_fd. A failed write is reported once per request on the
150 * connection log (fail2ban silently losing lines would be worse).
151 */
152void
153ngx_http_brix_guard_write_audit(ngx_http_request_t *r,
154 ngx_http_brix_guard_loc_conf_t *lcf, const guard_request_t *req,
155 guard_reason_t reason)
156{
157 char line[BRIX_GUARD_PATH_BUF + 256];
158 char ts[sizeof("YYYY-MM-DDThh:mm:ss+00:00")];
159 size_t ts_len;
160 size_t line_len;
161
162 if (lcf->audit_log == NULL || lcf->audit_log->fd == NGX_INVALID_FILE) {
163 return;
164 }
165
166 ts_len = ngx_cached_http_log_iso8601.len;
167 if (ts_len >= sizeof(ts)) {
168 ts_len = sizeof(ts) - 1;
169 }
170 ngx_memcpy(ts, ngx_cached_http_log_iso8601.data, ts_len);
171 ts[ts_len] = '\0';
172
173 line_len = guard_audit_format(req, reason, ts, line, sizeof(line) - 1);
174 if (line_len == 0) {
175 return;
176 }
177 line[line_len++] = '\n';
178
179 if (ngx_write_fd(lcf->audit_log->fd, line, line_len)
180 != (ssize_t) line_len)
181 {
182 ngx_log_error(NGX_LOG_ALERT, r->connection->log, ngx_errno,
183 "brix_guard: audit log write failed");
184 }
185}
