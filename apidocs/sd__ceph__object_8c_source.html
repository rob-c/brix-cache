2 * sd_ceph_object.c — Ceph/RADOS driver: object lifecycle + object metadata.
3 *
4 * WHAT: The namespace/metadata half of the driver split out of sd_ceph.c
5 * (file-size guard): the open path (existence probe, create/excl/trunc
6 * intent, handle build) shared by the plain and cred-scoped opens
7 * (sd_ceph_open_on_ioctx), the vtable open/close/stat/unlink slots, and
8 * the four xattr slots (get/list/set/removexattr) that let a RADOS object
9 * carry the cinfo/checksum-at-rest records.

14 * under the source-size guard while preserving byte-for-byte behaviour.

16 * HOW: The private struct definitions and the cross-TU helper/op declarations

36/* sd_ceph_probe_t — result of probing an object's existence at open time:
37 * whether it exists (`present`), its size/mtime if so, and whether it is a
38 * stock-XrdCeph striper object (so pread reassembles via libradosstriper). */
39typedef struct {
40 int present;
41 int striped;
42 uint64_t size;
43 time_t mtime;
44} sd_ceph_probe_t;

46/* sd_ceph_probe_object — establish whether the object already exists.
47 *
48 * WHAT: Fills `pr` from a stat probe. On a read open it first consults the
49 * stock-XrdCeph striper view (marking `striped` when a striped object of
50 * that name exists); otherwise, or on a miss, it falls back to a flat
51 * rados_stat. Returns 0 on success (object present or cleanly absent) or
52 * a negative errno for a hard stat error the caller must surface.
53 *
54 * WHY: The existence probe is the branchiest part of the open path; isolating
55 * it lets the orchestrator apply create/excl/trunc intent as a flat
56 * sequence instead of interleaving I/O with policy.
57 *
58 * HOW: 1. Zero the result. 2. For a non-write open, try the shared striper
59 * stat; a hit sets present+striped. 3. Otherwise flat rados_stat:
60 * 0 → present; -ENOENT → cleanly absent; any other negative → error. */
61static int
62sd_ceph_probe_object(const sd_ceph_open_req_t *req, const char *oid,
63 sd_ceph_probe_t *pr)
64{
65 int rc = -ENOENT;
66
67 ngx_memzero(pr, sizeof(*pr));
68
69#if defined(BRIX_HAVE_RADOSSTRIPER)
70 /* On a read open, prefer the stock-XrdCeph striper view: if a striped object
71 * exists for this name, mark it so pread reassembles via libradosstriper.
72 * The striper handle is bound to the EXPORT's service ioctx (lazily created
73 * on `st`, shared across users) — a per-user cred-scoped open still reads
74 * striped objects through it; only the raw rados_read/write/stat/trunc calls
75 * use the per-user `ioctx`. */
76 if (!(req->sd_flags & BRIX_SD_O_WRITE)) {
77 rados_striper_t s = sd_ceph_striper(req->st);
78 if (s != NULL
79 && sd_ceph_striper_stat(s, oid, &pr->size, &pr->mtime) == 0)
80 {
81 pr->present = 1;
82 pr->striped = 1;
83 return 0;
84 }
85 }
86#endif
87 rc = rados_stat(req->ioctx, oid, &pr->size, &pr->mtime);
88 if (rc == 0) {
89 pr->present = 1;
90 return 0;
91 }
92 if (rc == -ENOENT) {
93 return 0; /* cleanly absent */
94 }
95 return rc; /* hard stat error */
96}

98/* sd_ceph_apply_intent — enforce create/excl/trunc open flags against the
99 * probe result. Returns 0 if the open may proceed, or a positive errno the
100 * caller surfaces (ENOENT for absent-without-create, EEXIST for excl-collision,
101 * or a truncate failure). On a successful truncate the probe size is reset. */
102static int
103sd_ceph_apply_intent(const sd_ceph_open_req_t *req, const char *oid,
104 sd_ceph_probe_t *pr)
105{
106 if (!pr->present && !(req->sd_flags & BRIX_SD_O_CREATE)) {
107 return ENOENT;
108 }
109 if (pr->present && (req->sd_flags & BRIX_SD_O_EXCL)) {
110 return EEXIST;
111 }
112 if (req->sd_flags & BRIX_SD_O_TRUNC) {
113 if (sd_ceph_set_errno(rados_trunc(req->ioctx, oid, 0))) {
114 return errno;
115 }
116 pr->size = 0;
117 }
118 return 0;
119}

121/* sd_ceph_obj_build — allocate and populate the open object + its driver state
122 * from a completed probe. Pins a cred-scoped connection (req->pin_conn) so the
123 * cred-conn LRU cannot free its ioctx while the handle is open (the UAF fix).
124 * Returns the handle, or NULL with *err_out on allocation failure. */
125static brix_sd_obj_t *
126sd_ceph_obj_build(const sd_ceph_open_req_t *req, const char *oid,
127 const sd_ceph_probe_t *pr, int *err_out)
128{
129 brix_sd_obj_t *obj;
130 sd_ceph_obj_state_t *os;
131
132 obj = ngx_pcalloc(req->inst->pool, sizeof(*obj));
133 os = ngx_pcalloc(req->inst->pool, sizeof(*os));
134 if (obj == NULL || os == NULL) {
135 if (err_out != NULL) { *err_out = ENOMEM; }
136 return NULL;
137 }
138
139 memcpy(os->oid, oid, strlen(oid) + 1);
140 os->size = pr->present ? pr->size : 0;
141 os->striped = pr->striped;
142 os->ioctx = req->ioctx;
143 os->conn = req->pin_conn;
144 os->for_write = (req->sd_flags & BRIX_SD_O_WRITE) ? 1 : 0;
145 obj->driver = req->inst->driver;
146 obj->inst = req->inst;
147 obj->fd = NGX_INVALID_FILE;
148 obj->state = os;
149
150 if (req->pin_conn != NULL) {
151 sd_ceph_conn_pin(req->pin_conn); /* UAF fix: keep alive until close() */
152 }
153
154 /* Populate the metadata snapshot the caller copies into its handle and uses
155 * to build the open reply (open_resolved_file.c). RADOS has no fd/inode, so
156 * synthesize a stable inode from the object id and present a regular file. */
157 obj->snap.size = (off_t) os->size;
158 obj->snap.mtime = pr->mtime;
159 obj->snap.ctime = pr->mtime;
160 obj->snap.mode = S_IFREG | 0644;
161 obj->snap.ino = (ino_t) sd_ceph_ino(os->oid);
162 obj->snap.is_reg = 1;
163 return obj;
164}

166/* sd_ceph_open_on_ioctx — shared body for plain and cred-scoped opens: resolve
167 * the LFN to an object id against the SUPPLIED ioctx, honour create/excl/trunc
168 * intent against a stat probe, and return a handle carrying the id + cached
169 * size. There is no fd: obj->fd stays NGX_INVALID_FILE (the VFS serves such
170 * handles memory-backed).
171 *
172 * WHAT: Resolves `req->path` to an object id, probes its existence, applies the
173 * open intent, and builds the handle. Returns the object, or NULL with
174 * *err_out set on any failure.
175 * WHY: A cred-scoped open (sd_ceph_open_cred) must issue every rados_* /
176 * striper call against the PER-USER connection's ioctx, not the
177 * export's static service ioctx — otherwise the "per-user" open would
178 * silently authenticate as the service account, defeating the whole
179 * point of the ceph-peruser credential kind. Factoring the object-open
180 * body into one function (fed a bundled request) keeps the plain and
181 * cred-scoped paths from diverging in error handling or metadata.
182 * HOW: 1. Map the logical path to an object id (sd_ceph_key). 2. Probe
183 * existence (sd_ceph_probe_object). 3. Enforce create/excl/trunc
184 * (sd_ceph_apply_intent). 4. Allocate + populate the handle, pinning
185 * req->pin_conn for a cred-scoped open (sd_ceph_obj_build). On any
186 * failure path nothing is pinned (the caller still owns releasing
187 * req->pin_conn itself, e.g. sd_ceph_open_cred's pin bookkeeping). */
188brix_sd_obj_t *
189sd_ceph_open_on_ioctx(const sd_ceph_open_req_t *req, int *err_out)
190{
191 char oid[1024];
192 sd_ceph_probe_t pr = {0};
193 int rc;
194 int intent_err;
195
196 (void) req->mode;
197
198 if (sd_ceph_key(req->st->key_prefix, req->path, oid, sizeof(oid)) != 0) {
199 if (err_out != NULL) { *err_out = errno; }
200 return NULL;
201 }
202
203 rc = sd_ceph_probe_object(req, oid, &pr);
204 if (rc < 0) {
205 if (err_out != NULL) { *err_out = -rc; }
206 return NULL;
207 }
208
209 intent_err = sd_ceph_apply_intent(req, oid, &pr);
210 if (intent_err != 0) {
211 if (err_out != NULL) { *err_out = intent_err; }
212 return NULL;
213 }
214
215 return sd_ceph_obj_build(req, oid, &pr, err_out);
216}

219 *
220 * WHAT: Plain open for callers that do not carry a per-user credential.
221 * WHY: Preserves the existing public vtable signature; uses the export's
222 * static service ioctx (inst->state->ioctx).
223 * HOW: Delegates to sd_ceph_open_on_ioctx with the instance's own ioctx. */

242/* sd_ceph_close — release this handle's pin (if any) on its cred-scoped
243 * connection. A plain (service-credential) open has os->conn == NULL — the
244 * export's own ioctx is instance-lived and is torn down only by
245 * sd_ceph_cleanup, so there is nothing to release. A cred-scoped open's
246 * os->conn is non-NULL: sd_ceph_conn_unpin drops the pin taken at open, and
247 * if this was the connection's last pin AND it had already been evicted
248 * from the cred-conn cache table (doomed) while still in use, the deferred
249 * rados_ioctx_destroy/rados_shutdown happens right here — this is the fix
250 * for the historical UAF (the connection used to be destroyed by the LRU
251 * evictor regardless of whether a handle was still reading/writing through
252 * it). Idempotent-safe: the VFS never calls close twice on the same handle,
253 * but sd_ceph_conn_unpin tolerates refs already at 0. */
254ngx_int_t
255sd_ceph_close(brix_sd_obj_t *obj)
256{
257 sd_ceph_obj_state_t *os = obj->state;
258
259 if (os->conn != NULL) {
260 sd_ceph_conn_unpin(os->conn);
261 os->conn = NULL;
262 }
263 return NGX_OK;
264}

268/* sd_ceph_stat — rados_stat on the object id for a logical path. The export
269 * root ("/") is the one always-present SYNTHETIC directory (phase-89 ADR-1:
270 * directories are prefixes, not objects) and is answered without touching the
271 * cluster; deeper synthetic directories are served by opendir only — a
272 * per-stat child probe would turn every stat-miss into a pool scan. */
273ngx_int_t
274sd_ceph_stat(brix_sd_instance_t *inst, const char *path,
275 brix_sd_stat_t *out)
276{
277 sd_ceph_state_t *st = inst->state;
278 char oid[1024];
279 char norm[1024];
280 uint64_t size = 0;
281 time_t mtime = 0;
282
283 if (sd_ceph_normalize(path, norm, sizeof(norm)) == 0
284 && norm[0] == '/' && norm[1] == '\0')
285 {
286 ngx_memzero(out, sizeof(*out));
287 out->mode = S_IFDIR | 0755;
288 out->is_dir = 1;
289 out->ino = (ino_t) sd_ceph_ino(
290 st->key_prefix != NULL ? st->key_prefix : "/");
291 return NGX_OK;
292 }
293
294 if (sd_ceph_key(st->key_prefix, path, oid, sizeof(oid)) != 0) {
295 return NGX_ERROR;
296 }
297 if (sd_ceph_set_errno(rados_stat(st->ioctx, oid, &size, &mtime))) {
298 return NGX_ERROR;
299 }
300 sd_ceph_fill_stat(oid, size, mtime, out);
301 return NGX_OK;
302}

326/* sd_ceph_rmdir_synthetic — directory removal on a flat namespace (phase-89
327 * ADR-1): a synthetic directory holds no object of its own, so "removing" an
328 * empty one succeeds without touching the cluster, a non-empty one is
329 * ENOTEMPTY, and the export root is never removable (EBUSY). */
330static ngx_int_t
331sd_ceph_rmdir_synthetic(brix_sd_instance_t *inst, const char *path)
332{
333 char norm[1024];
334 sd_ceph_child_probe_t c;
335
336 if (sd_ceph_normalize(path, norm, sizeof(norm)) != 0) {
337 return NGX_ERROR;
338 }
339 if (norm[0] == '/' && norm[1] == '\0') {
340 errno = EBUSY;
341 return NGX_ERROR;
342 }
343
344 c.dir = norm;
345 c.found = 0;
346 if (sd_ceph_enumerate(inst, 0, sd_ceph_child_probe_cb, &c) != NGX_OK) {
347 return NGX_ERROR;
348 }
349 if (c.found) {
350 errno = ENOTEMPTY;
351 return NGX_ERROR;
352 }
353 return NGX_OK;
354}

402/* sd_ceph_path_probe_t / sd_ceph_probe_oid — existence/layout probe for a
403 * bare oid (no open request): striper view first (a striped object must be
404 * copied through the striper to reassemble), then flat. 0 on success (present
405 * or cleanly absent), negative errno on a hard stat error. */
406typedef struct {
407 int present;
408 int striped;
409 uint64_t size;
410} sd_ceph_path_probe_t;
411
412static int
413sd_ceph_probe_oid(sd_ceph_state_t *st, const char *oid,
414 sd_ceph_path_probe_t *pr)
415{
416 uint64_t size = 0;
417 time_t mtime = 0;
418 int rc;
419
420 ngx_memzero(pr, sizeof(*pr));
421
422#if defined(BRIX_HAVE_RADOSSTRIPER)
423 {
424 rados_striper_t s = sd_ceph_striper(st);
425
426 if (s != NULL && sd_ceph_striper_stat(s, oid, &size, &mtime) == 0) {
427 pr->present = 1;
428 pr->striped = 1;
429 pr->size = size;
430 return 0;
431 }
432 }
433#endif
434 rc = rados_stat(st->ioctx, oid, &size, &mtime);
435 if (rc == 0) {
436 pr->present = 1;
437 pr->size = size;
438 return 0;
439 }
440 return (rc == -ENOENT) ? 0 : rc;
441}

459/* sd_ceph_copy_bytes — chunked byte copy src → dst. A striped source is read
460 * AND written through the striper so the destination keeps the stock-XrdCeph
461 * layout; a flat source copies flat. An empty source is created flat (the
462 * caller forces striped=0 for size 0 — a zero-length striper write would not
463 * materialize the first stripe). 0 or negative errno. */
464static int
465sd_ceph_copy_bytes(sd_ceph_state_t *st, const char *src, int striped,
466 uint64_t size, const char *dst)
467{
468 char *buf;
469 uint64_t off = 0;
470
471 if (size == 0) {
472 return rados_write_full(st->ioctx, dst, "", 0);
473 }
474
475 buf = malloc(SD_CEPH_COPY_CHUNK);
476 if (buf == NULL) {
477 return -ENOMEM;
478 }
479
480 while (off < size) {
481 size_t want = (size - off < SD_CEPH_COPY_CHUNK)
482 ? (size_t) (size - off) : SD_CEPH_COPY_CHUNK;
483 ssize_t n;
484 ssize_t wrc;
485
486#if defined(BRIX_HAVE_RADOSSTRIPER)
487 if (striped) {
488 rados_striper_t s = sd_ceph_striper(st);
489
490 if (s == NULL) {
491 free(buf);
492 return -EIO;
493 }
494 n = sd_ceph_striper_read(s, src, buf, want, off);
495 if (n > 0) {
496 wrc = sd_ceph_striper_write(s, dst, buf, (size_t) n, off);
497 } else {
498 wrc = 0;
499 }
500 } else
501#endif
502 {
503 n = rados_read(st->ioctx, src, buf, want, off);
504 wrc = (n > 0) ? rados_write(st->ioctx, dst, buf, (size_t) n, off)
505 : 0;
506 }
507
508 if (n < 0) {
509 free(buf);
510 return (int) n;
511 }
512 if (n == 0) {
513 free(buf);
514 return -EIO; /* source shrank mid-copy */
515 }
516 if (wrc < 0) {
517 free(buf);
518 return (int) wrc;
519 }
520 off += (uint64_t) n;
521 }
522
523 free(buf);
524 return 0;
525}

528/* sd_ceph_copy_xattrs_striped — the striped-source leg of the xattr copy:
529 * list the striper object's names, then get/set each one. Striper-internal
530 * "striper.*" layout attrs are never copied — the destination's own write path
531 * stamps its layout. 0 or negative errno. */
532static int
533sd_ceph_copy_xattrs_striped(sd_ceph_state_t *st, const char *src,
534 const char *dst, int dst_striped)
535{
536 rados_striper_t s = sd_ceph_striper(st);
537 char *names;
538 char *val;
539 ssize_t total;
540 ssize_t used = 0;
541 int rc = 0;
542
543 if (s == NULL) {
544 return -EIO;
545 }
546 names = malloc(SD_CEPH_XATTR_MAX);
547 val = malloc(SD_CEPH_XATTR_MAX);
548 if (names == NULL || val == NULL) {
549 free(names);
550 free(val);
551 return -ENOMEM;
552 }
553 total = sd_ceph_striper_listxattr(s, src, names, SD_CEPH_XATTR_MAX);
554 if (total < 0) {
555 free(names);

557 return (int) total;
558 }
559 while (rc == 0 && used < total) {
560 const char *nm = names + used;
561 ssize_t vlen;
562
563 used += (ssize_t) strlen(nm) + 1;
564 if (strncmp(nm, "striper.", 8) == 0) {
565 continue; /* layout attrs: never copied */
566 }
567 vlen = sd_ceph_striper_getxattr(s, src, nm, val,
568 SD_CEPH_XATTR_MAX);
569 if (vlen < 0) {

571 break;
572 }
573 rc = dst_striped
574 ? sd_ceph_striper_setxattr(s, dst, nm, val, (size_t) vlen)
575 : rados_setxattr(st->ioctx, dst, nm, val, (size_t) vlen);
576 }
577 free(names);
578 free(val);
579 return rc;
580}

583/* sd_ceph_copy_xattrs_plain — the flat-object leg of the xattr copy: walk the
584 * source's xattr iterator and stamp each pair onto the destination (striper
585 * or flat, per dst_striped). 0 or negative errno. */
586static int
587sd_ceph_copy_xattrs_plain(sd_ceph_state_t *st, const char *src,
588 const char *dst, int dst_striped)
589{
590 rados_xattrs_iter_t it;
591 int rc;
592
593 rc = rados_getxattrs(st->ioctx, src, &it);
594 if (rc < 0) {
595 return rc;
596 }
597 for (;;) {
598 const char *nm = NULL;
599 const char *vv = NULL;
600 size_t vlen = 0;
601
602 rc = rados_getxattrs_next(it, &nm, &vv, &vlen);
603 if (rc < 0 || nm == NULL) {
604 break;
605 }
606#if defined(BRIX_HAVE_RADOSSTRIPER)
607 if (dst_striped) {
608 rados_striper_t s = sd_ceph_striper(st);
609
610 rc = (s != NULL)
611 ? sd_ceph_striper_setxattr(s, dst, nm, vv, vlen) : -EIO;
612 } else
613#endif
614 {
615 rc = rados_setxattr(st->ioctx, dst, nm, vv, vlen);
616 }
617 if (rc < 0) {
618 break;
619 }
620 }
621 rados_getxattrs_end(it);

623 (void) dst_striped;
624#endif
625 return (rc < 0) ? rc : 0;
626}

698/* sd_ceph_rename_copy_verify — land the destination's bytes AND xattrs, then
699 * size-verify it against the source probe. 0 or negative errno (the caller
700 * cleans up the partial destination on failure). */
701static int
702sd_ceph_rename_copy_verify(sd_ceph_state_t *st, const char *so,
703 const char *dm, const sd_ceph_path_probe_t *sp, int dst_striped)
704{
705 int rc;
706
707 rc = sd_ceph_copy_bytes(st, so, dst_striped, sp->size, dm);
708 if (rc == 0) {
709 rc = sd_ceph_copy_xattrs(st, so, sp->striped, dm, dst_striped);
710 }
711 if (rc == 0) {
712 sd_ceph_path_probe_t vp;
713
714 rc = sd_ceph_probe_oid(st, dm, &vp);
715 if (rc == 0 && (!vp.present || vp.size != sp->size)) {
716 rc = -EIO; /* verify: dst must hold src's bytes */
717 }
718 }
719 return rc;
720}

722/* sd_ceph_rename — copy + delete on the flat namespace (phase-89 §B.2).
723 * NON-ATOMIC by design (the driver honestly does not advertise
724 * CAP_HARD_RENAME — same posture as the other object backends): the source is
725 * removed only after the destination's bytes AND xattrs landed and a size
726 * verify passed, so a mid-copy failure leaves the source intact (the partial
727 * destination is cleaned up best-effort). `noreplace` is honoured by a probe
728 * (racy by nature on an object store — documented). Directory rename is
729 * refused (EISDIR): a flat namespace has no atomic prefix move, and a
730 * recursive key rewrite is out of scope (ADR-5). */
731ngx_int_t
732sd_ceph_rename(brix_sd_instance_t *inst, const char *src, const char *dst,
733 int noreplace)
734{
735 sd_ceph_state_t *st = inst->state;
736 char so[1024];
737 char dm[1024];
738 sd_ceph_path_probe_t sp;
739 int dst_striped;
740 int rc;
741
742 if (sd_ceph_key(st->key_prefix, src, so, sizeof(so)) != 0
743 || sd_ceph_key(st->key_prefix, dst, dm, sizeof(dm)) != 0)
744 {
745 return NGX_ERROR;
746 }
747 if (strcmp(so, dm) == 0) {
748 return NGX_OK; /* same object — nothing to do */
749 }
750
751 rc = sd_ceph_probe_oid(st, so, &sp);
752 if (rc < 0) {
753 errno = -rc;
754 return NGX_ERROR;
755 }
756 if (!sp.present) {
757 return sd_ceph_rename_missing_src(inst, src);
758 }
759
760 if (sd_ceph_rename_prep_dst(st, dm, noreplace) != NGX_OK) {
761 return NGX_ERROR;
762 }
763
764 dst_striped = (sp.striped && sp.size > 0);
765 rc = sd_ceph_rename_copy_verify(st, so, dm, &sp, dst_striped);
766 if (rc < 0) {
767 (void) sd_ceph_remove_oid(st, dm, dst_striped); /* best-effort cleanup */
768 errno = -rc;
769 return NGX_ERROR;
770 }
771
772 rc = sd_ceph_remove_oid(st, so, sp.striped);
773 if (rc < 0) {
774 errno = -rc; /* dst intact; src survived too */
775 return NGX_ERROR;
776 }
777 return NGX_OK;
778}

780/* xattr / object metadata — RADOS objects carry their own xattrs, so the
781 * checksum-at-rest seam (user.XrdCks.*) and protocol GETFATTR/SETFATTR work on a
782 * Ceph export exactly as on POSIX. All four key the object id off the logical
783 * path; the object must already exist (set/get/list/remove on a missing oid
784 * return -ENOENT via librados). */
785
786ssize_t
787sd_ceph_getxattr(brix_sd_instance_t *inst, const char *path,
788 const char *name, void *buf, size_t cap)
789{
790 sd_ceph_state_t *st = inst->state;
791 char oid[1024];
792 char tmp[SD_CEPH_XATTR_MAX];
793 int n;
794
795 if (sd_ceph_key(st->key_prefix, path, oid, sizeof(oid)) != 0) {
796 return -1;
797 }
798 n = rados_getxattr(st->ioctx, oid, name, tmp, sizeof(tmp));
799 if (n < 0) {
800 errno = -n;
801 return -1;
802 }
803 if (cap == 0) {
804 return n; /* size probe (getxattr(2) convention) */
805 }
806 if ((size_t) n > cap) {
807 errno = ERANGE;
808 return -1;
809 }
810 memcpy(buf, tmp, (size_t) n);
811 return n;
812}

814ssize_t
815sd_ceph_listxattr(brix_sd_instance_t *inst, const char *path,
816 void *buf, size_t cap)
817{
818 sd_ceph_state_t *st = inst->state;
819 char oid[1024];
820 rados_xattrs_iter_t it;
821 char *out = buf;
822 size_t total = 0;
823
824 if (sd_ceph_key(st->key_prefix, path, oid, sizeof(oid)) != 0) {
825 return -1;
826 }
827 if (sd_ceph_set_errno(rados_getxattrs(st->ioctx, oid, &it))) {
828 return -1;
829 }
830 for (;;) {
831 const char *nm = NULL;
832 const char *val = NULL;
833 size_t vlen = 0;
834 size_t nlen;
835
836 if (sd_ceph_set_errno(rados_getxattrs_next(it, &nm, &val, &vlen))) {
837 rados_getxattrs_end(it);
838 return -1;
839 }
840 if (nm == NULL) {
841 break; /* end of iteration */
842 }
843 nlen = strlen(nm) + 1; /* listxattr(2): names are NUL-separated */
844 if (cap != 0) {
845 if (total + nlen > cap) {
846 rados_getxattrs_end(it);
847 errno = ERANGE;
848 return -1;
849 }
850 memcpy(out + total, nm, nlen);
851 }
852 total += nlen;
853 }
854 rados_getxattrs_end(it);
855 return (ssize_t) total;
856}
