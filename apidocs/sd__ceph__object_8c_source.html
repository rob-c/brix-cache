2 * sd_ceph_object.c — Ceph/RADOS driver: object lifecycle + object metadata.
3 *
4 * WHAT: The namespace/metadata half of the driver split out of sd_ceph.c
5 * (file-size guard): the open path (existence probe, create/excl/trunc
6 * intent, handle build) shared by the plain and cred-scoped opens
7 * (sd_ceph_open_on_ioctx), the vtable open/close/stat/unlink slots, and
8 * the four xattr slots (get/list/set/removexattr) that let a RADOS object
9 * carry the cinfo/checksum-at-rest records.
10 *
11 * WHY: These map a logical path to an object id and apply open policy against a
12 * stat probe — a distinct concern from the raw byte ops (sd_ceph_io.c) and
13 * the credential machinery (sd_ceph_cred.c); isolating it keeps every file
14 * under the source-size guard while preserving byte-for-byte behaviour.
15 *
16 * HOW: The private struct definitions and the cross-TU helper/op declarations
17 * come from sd_ceph_internal.h. The whole body is gated on BRIX_HAVE_CEPH,
18 * exactly like the driver it was split from.
19 */

36/* sd_ceph_probe_t — result of probing an object's existence at open time:
37 * whether it exists (`present`), its size/mtime if so, and whether it is a
38 * stock-XrdCeph striper object (so pread reassembles via libradosstriper). */
39typedef struct {
40 int present;
41 int striped;
42 uint64_t size;
43 time_t mtime;
44} sd_ceph_probe_t;

46/* sd_ceph_probe_object — establish whether the object already exists.
47 *
48 * WHAT: Fills `pr` from a stat probe. On a read open it first consults the
49 * stock-XrdCeph striper view (marking `striped` when a striped object of
50 * that name exists); otherwise, or on a miss, it falls back to a flat
51 * rados_stat. Returns 0 on success (object present or cleanly absent) or
52 * a negative errno for a hard stat error the caller must surface.
53 *
54 * WHY: The existence probe is the branchiest part of the open path; isolating
55 * it lets the orchestrator apply create/excl/trunc intent as a flat
56 * sequence instead of interleaving I/O with policy.
57 *
58 * HOW: 1. Zero the result. 2. For a non-write open, try the shared striper
59 * stat; a hit sets present+striped. 3. Otherwise flat rados_stat:
60 * 0 → present; -ENOENT → cleanly absent; any other negative → error. */
61static int
62sd_ceph_probe_object(const sd_ceph_open_req_t *req, const char *oid,
63 sd_ceph_probe_t *pr)
64{
65 int rc = -ENOENT;
66
67 ngx_memzero(pr, sizeof(*pr));
68
69#if defined(BRIX_HAVE_RADOSSTRIPER)
70 /* On a read open, prefer the stock-XrdCeph striper view: if a striped object
71 * exists for this name, mark it so pread reassembles via libradosstriper.
72 * The striper handle is bound to the EXPORT's service ioctx (lazily created
73 * on `st`, shared across users) — a per-user cred-scoped open still reads
74 * striped objects through it; only the raw rados_read/write/stat/trunc calls
75 * use the per-user `ioctx`. */
76 if (!(req->sd_flags & BRIX_SD_O_WRITE)) {
77 rados_striper_t s = sd_ceph_striper(req->st);
78 if (s != NULL
79 && sd_ceph_striper_stat(s, oid, &pr->size, &pr->mtime) == 0)
80 {
81 pr->present = 1;
82 pr->striped = 1;
83 return 0;
84 }
85 }
86#endif
87 rc = rados_stat(req->ioctx, oid, &pr->size, &pr->mtime);
88 if (rc == 0) {
89 pr->present = 1;
90 return 0;
91 }
92 if (rc == -ENOENT) {
93 return 0; /* cleanly absent */
94 }
95 return rc; /* hard stat error */
96}

98/* sd_ceph_apply_intent — enforce create/excl/trunc open flags against the
99 * probe result. Returns 0 if the open may proceed, or a positive errno the
100 * caller surfaces (ENOENT for absent-without-create, EEXIST for excl-collision,
101 * or a truncate failure). On a successful truncate the probe size is reset. */
102static int
103sd_ceph_apply_intent(const sd_ceph_open_req_t *req, const char *oid,
104 sd_ceph_probe_t *pr)
105{
106 if (!pr->present && !(req->sd_flags & BRIX_SD_O_CREATE)) {
107 return ENOENT;
108 }
109 if (pr->present && (req->sd_flags & BRIX_SD_O_EXCL)) {
110 return EEXIST;
111 }
112 if (req->sd_flags & BRIX_SD_O_TRUNC) {
113 if (sd_ceph_set_errno(rados_trunc(req->ioctx, oid, 0))) {
114 return errno;
115 }
116 pr->size = 0;
117 }
118 return 0;
119}

121/* sd_ceph_obj_build — allocate and populate the open object + its driver state
122 * from a completed probe. Pins a cred-scoped connection (req->pin_conn) so the
123 * cred-conn LRU cannot free its ioctx while the handle is open (the UAF fix).
124 * Returns the handle, or NULL with *err_out on allocation failure. */
125static brix_sd_obj_t *
126sd_ceph_obj_build(const sd_ceph_open_req_t *req, const char *oid,
127 const sd_ceph_probe_t *pr, int *err_out)
128{
129 brix_sd_obj_t *obj;
130 sd_ceph_obj_state_t *os;
131
132 obj = ngx_pcalloc(req->inst->pool, sizeof(*obj));
133 os = ngx_pcalloc(req->inst->pool, sizeof(*os));
134 if (obj == NULL || os == NULL) {
135 if (err_out != NULL) { *err_out = ENOMEM; }
136 return NULL;
137 }
138
139 memcpy(os->oid, oid, strlen(oid) + 1);
140 os->size = pr->present ? pr->size : 0;
141 os->striped = pr->striped;
142 os->ioctx = req->ioctx;
143 os->conn = req->pin_conn;
144 os->for_write = (req->sd_flags & BRIX_SD_O_WRITE) ? 1 : 0;
145 obj->driver = req->inst->driver;
146 obj->inst = req->inst;
147 obj->fd = NGX_INVALID_FILE;
148 obj->state = os;
149
150 if (req->pin_conn != NULL) {
151 sd_ceph_conn_pin(req->pin_conn); /* UAF fix: keep alive until close() */
152 }
153
154 /* Populate the metadata snapshot the caller copies into its handle and uses
155 * to build the open reply (open_resolved_file.c). RADOS has no fd/inode, so
156 * synthesize a stable inode from the object id and present a regular file. */
157 obj->snap.size = (off_t) os->size;
158 obj->snap.mtime = pr->mtime;
159 obj->snap.ctime = pr->mtime;
160 obj->snap.mode = S_IFREG | 0644;
161 obj->snap.ino = (ino_t) sd_ceph_ino(os->oid);
162 obj->snap.is_reg = 1;
163 return obj;
164}

166/* sd_ceph_open_on_ioctx — shared body for plain and cred-scoped opens: resolve
167 * the LFN to an object id against the SUPPLIED ioctx, honour create/excl/trunc
168 * intent against a stat probe, and return a handle carrying the id + cached
169 * size. There is no fd: obj->fd stays NGX_INVALID_FILE (the VFS serves such
170 * handles memory-backed).
171 *
172 * WHAT: Resolves `req->path` to an object id, probes its existence, applies the
173 * open intent, and builds the handle. Returns the object, or NULL with
174 * *err_out set on any failure.
175 * WHY: A cred-scoped open (sd_ceph_open_cred) must issue every rados_* /
176 * striper call against the PER-USER connection's ioctx, not the
177 * export's static service ioctx — otherwise the "per-user" open would
178 * silently authenticate as the service account, defeating the whole
179 * point of the ceph-peruser credential kind. Factoring the object-open
180 * body into one function (fed a bundled request) keeps the plain and
181 * cred-scoped paths from diverging in error handling or metadata.
182 * HOW: 1. Map the logical path to an object id (sd_ceph_key). 2. Probe
183 * existence (sd_ceph_probe_object). 3. Enforce create/excl/trunc
184 * (sd_ceph_apply_intent). 4. Allocate + populate the handle, pinning
185 * req->pin_conn for a cred-scoped open (sd_ceph_obj_build). On any
186 * failure path nothing is pinned (the caller still owns releasing
187 * req->pin_conn itself, e.g. sd_ceph_open_cred's pin bookkeeping). */
188brix_sd_obj_t *
189sd_ceph_open_on_ioctx(const sd_ceph_open_req_t *req, int *err_out)
190{
191 char oid[1024];
192 sd_ceph_probe_t pr = {0};
193 int rc;
194 int intent_err;
195
196 (void) req->mode;
197
198 if (sd_ceph_key(req->st->key_prefix, req->path, oid, sizeof(oid)) != 0) {
199 if (err_out != NULL) { *err_out = errno; }
200 return NULL;
201 }
202
203 rc = sd_ceph_probe_object(req, oid, &pr);
204 if (rc < 0) {
205 if (err_out != NULL) { *err_out = -rc; }
206 return NULL;
207 }
208
209 intent_err = sd_ceph_apply_intent(req, oid, &pr);
210 if (intent_err != 0) {
211 if (err_out != NULL) { *err_out = intent_err; }
212 return NULL;
213 }
214
215 return sd_ceph_obj_build(req, oid, &pr, err_out);
216}

218/* sd_ceph_open — vtable open slot: service credential.
219 *
220 * WHAT: Plain open for callers that do not carry a per-user credential.
221 * WHY: Preserves the existing public vtable signature; uses the export's
222 * static service ioctx (inst->state->ioctx).
223 * HOW: Delegates to sd_ceph_open_on_ioctx with the instance's own ioctx. */
224brix_sd_obj_t *
225sd_ceph_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
226 mode_t mode, int *err_out)
227{
228 sd_ceph_state_t *st = inst->state;
229 sd_ceph_open_req_t req = {
230 .inst = inst,
231 .st = st,
232 .ioctx = st->ioctx,
233 .pin_conn = NULL,
234 .path = path,
235 .sd_flags = sd_flags,
236 .mode = mode,
237 };
238
239 return sd_ceph_open_on_ioctx(&req, err_out);
240}

242/* sd_ceph_close — release this handle's pin (if any) on its cred-scoped
243 * connection. A plain (service-credential) open has os->conn == NULL — the
244 * export's own ioctx is instance-lived and is torn down only by
245 * sd_ceph_cleanup, so there is nothing to release. A cred-scoped open's
246 * os->conn is non-NULL: sd_ceph_conn_unpin drops the pin taken at open, and
247 * if this was the connection's last pin AND it had already been evicted
248 * from the cred-conn cache table (doomed) while still in use, the deferred
249 * rados_ioctx_destroy/rados_shutdown happens right here — this is the fix
250 * for the historical UAF (the connection used to be destroyed by the LRU
251 * evictor regardless of whether a handle was still reading/writing through
252 * it). Idempotent-safe: the VFS never calls close twice on the same handle,
253 * but sd_ceph_conn_unpin tolerates refs already at 0. */
254ngx_int_t
255sd_ceph_close(brix_sd_obj_t *obj)
256{
257 sd_ceph_obj_state_t *os = obj->state;
258
259 if (os->conn != NULL) {
260 sd_ceph_conn_unpin(os->conn);
261 os->conn = NULL;
262 }
263 return NGX_OK;
264}

307/* xattr / object metadata — RADOS objects carry their own xattrs, so the
308 * checksum-at-rest seam (user.XrdCks.*) and protocol GETFATTR/SETFATTR work on a
309 * Ceph export exactly as on POSIX. All four key the object id off the logical
310 * path; the object must already exist (set/get/list/remove on a missing oid
311 * return -ENOENT via librados). */
312
313ssize_t
314sd_ceph_getxattr(brix_sd_instance_t *inst, const char *path,
315 const char *name, void *buf, size_t cap)
316{
317 sd_ceph_state_t *st = inst->state;
318 char oid[1024];
319 char tmp[SD_CEPH_XATTR_MAX];
320 int n;
321
322 if (sd_ceph_key(st->key_prefix, path, oid, sizeof(oid)) != 0) {
323 return -1;
324 }
325 n = rados_getxattr(st->ioctx, oid, name, tmp, sizeof(tmp));
326 if (n < 0) {
327 errno = -n;
328 return -1;
329 }
330 if (cap == 0) {
331 return n; /* size probe (getxattr(2) convention) */
332 }
333 if ((size_t) n > cap) {
334 errno = ERANGE;
335 return -1;
336 }
337 memcpy(buf, tmp, (size_t) n);
338 return n;
339}

341ssize_t
342sd_ceph_listxattr(brix_sd_instance_t *inst, const char *path,
343 void *buf, size_t cap)
344{
345 sd_ceph_state_t *st = inst->state;
346 char oid[1024];
347 rados_xattrs_iter_t it;
348 char *out = buf;
349 size_t total = 0;
350
351 if (sd_ceph_key(st->key_prefix, path, oid, sizeof(oid)) != 0) {
352 return -1;
353 }
354 if (sd_ceph_set_errno(rados_getxattrs(st->ioctx, oid, &it))) {
355 return -1;
356 }
357 for (;;) {
358 const char *nm = NULL;
359 const char *val = NULL;
360 size_t vlen = 0;
361 size_t nlen;
362
363 if (sd_ceph_set_errno(rados_getxattrs_next(it, &nm, &val, &vlen))) {
364 rados_getxattrs_end(it);
365 return -1;
366 }
367 if (nm == NULL) {
368 break; /* end of iteration */
369 }
370 nlen = strlen(nm) + 1; /* listxattr(2): names are NUL-separated */
371 if (cap != 0) {
372 if (total + nlen > cap) {
373 rados_getxattrs_end(it);
374 errno = ERANGE;
375 return -1;
376 }
377 memcpy(out + total, nm, nlen);
378 }
379 total += nlen;
380 }
381 rados_getxattrs_end(it);
382 return (ssize_t) total;
383}
