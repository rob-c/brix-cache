43/* 1 iff serving `inst` reads from a socket-wire backend (one that cannot pump its
44 * blocking socket on the un-pumped event loop). A cache serves from its STORE; a
45 * stage decorator serves from its SOURCE; a bare driver answers by name. Today the
46 * only such driver is "xroot"; in-process (rados) and curl (s3/http) block-but-
47 * complete on-loop and are served inline. */
48static int
49serve_is_remote_socket(const brix_sd_instance_t *inst)
50{
51 if (inst == NULL) {
52 return 0;
53 }
54 if (brix_sd_cache_instance_is(inst)) {
55 return serve_is_remote_socket(brix_sd_cache_store_instance(inst));
56 }
57 if (brix_sd_stage_instance_is(inst)) {
58 return serve_is_remote_socket(brix_sd_stage_source_instance(inst));
59 }
60 return (inst->driver != NULL && inst->driver->name != NULL
61 && ngx_strcmp(inst->driver->name, "xroot") == 0) ? 1 : 0;
62}

64/* Open an anonymous, auto-cleaned temp file for the materialised object. Prefers
65 * O_TMPFILE (no name, removed on last close); falls back to mkstemp+unlink. A
66 * transient worker-owned scratch buffer, not export storage. */
67static int
68serve_offload_tmp_open(ngx_log_t *log)
69{
70 const char *dir = getenv("TMPDIR");
71 int fd;
72
73 if (dir == NULL || dir[0] == '\0') {
74 dir = "/tmp";
75 }
76 fd = open(dir, O_TMPFILE | O_RDWR | O_CLOEXEC, 0600); /* vfs-seam-allow: transient serve scratch (not export storage) */
77 if (fd >= 0) {
78 return fd;
79 }
80 {
81 char tmpl[PATH_MAX];
82 int n = ngx_snprintf((u_char *) tmpl, sizeof(tmpl) - 1,
83 "%s/xrd-serve.XXXXXX", dir) - (u_char *) tmpl;
84
85 tmpl[n] = '\0';
86 fd = mkstemp(tmpl); /* vfs-seam-allow: transient serve scratch (not export storage) */
87 if (fd >= 0) {
88 (void) unlink(tmpl); /* vfs-seam-allow: transient serve scratch (not export storage) */
89 return fd;
90 }
91 }
92 ngx_log_error(NGX_LOG_ERR, log, errno,
93 "serve offload: cannot open a temp scratch file under \"%s\"", dir);
94 return -1;
95}

97/* Worker thread: open the object through the composed driver (eager connect, cinfo
98 * load, miss fill all run here) and copy it into the temp fd. Pure blocking I/O off
99 * the event loop - the socket wire client opens + reads here safely. */
100static void
101serve_offload_thread(void *data, ngx_log_t *log)
102{
103 serve_offload_ctx *t = data;
104 brix_sd_obj_t *obj;
105 brix_sd_obj_t dst; /* worker-owned scratch, driver-routed */
106 brix_sd_stat_t snap;
107 u_char *buf;
108 off_t off = 0;
109 int err = 0;
110
111 (void) log;
112 if (t->inst->driver->open == NULL || t->inst->driver->pread == NULL) {
113 t->mret = ENOSYS;
114 return;
115 }
116 obj = t->inst->driver->open(t->inst, t->key, BRIX_SD_O_READ, 0, &err);
117 if (obj == NULL) {
118 t->mret = err ? err : EIO;
119 return;
120 }
121
122 snap = obj->snap;
123 if (obj->driver->fstat != NULL) {
124 (void) obj->driver->fstat(obj, &snap);
125 }
126 t->mtime = snap.mtime;
127
128 buf = malloc(BRIX_SERVE_OFFLOAD_CHUNK);
129 if (buf == NULL) {
130 obj->driver->close(obj);
131 if (obj->heap_shell) { free(obj); }
132 t->mret = ENOMEM;
133 return;
134 }
135 /* Materialise the whole object into the scratch fd through the driver seam:
136 * read from the (possibly remote/object) source obj, write to the POSIX-
137 * wrapped temp. xvfs_drain owns the chunked pread->pwrite + EINTR loop. */
138 brix_sd_posix_wrap(&dst, t->tmp_fd);
139 t->mret = (xvfs_drain(obj, &dst, buf, BRIX_SERVE_OFFLOAD_CHUNK, &off) == 0)
140 ? 0 : (errno ? errno : EIO);
141 free(buf);
142 obj->driver->close(obj);
143 if (obj->heap_shell) { free(obj); }
144 if (t->mret == 0) {
145 t->size = off;
146 }
147}

149/* Event loop: adopt the temp fd as a POSIX handle and serve it through the shared
150 * sendfile pipeline, run the protocol metrics callback, then finalise. The single
151 * ngx_http_finalize_request balances the r->main->count++ taken at post. */
152static void
153serve_offload_done(ngx_event_t *ev)
154{
155 ngx_thread_task_t *task = ev->data;
156 serve_offload_ctx *t = task->ctx;
157 ngx_http_request_t *r = t->r;
158 ngx_connection_t *c = r->connection;
159 brix_vfs_ctx_t tvctx;
160 brix_vfs_file_t *tfh = NULL;
161 brix_vfs_stat_t vst;
162 brix_http_serve_opts_t opts;
163 brix_http_serve_result_t result;
164 ngx_int_t rc;
165
166 if (t->mret != 0) {
167 if (t->tmp_fd >= 0) {
168 (void) close(t->tmp_fd);
169 }
170 ngx_log_error(NGX_LOG_ERR, c->log, t->mret,
171 "serve offload: materialise failed for \"%s\" - returning %d",
172 t->fs_path, (t->mret == ENOENT) ? 404 : 502);
173 ngx_http_finalize_request(r,
174 (t->mret == ENOENT) ? NGX_HTTP_NOT_FOUND : NGX_HTTP_BAD_GATEWAY);
175 ngx_http_run_posted_requests(c);
176 return;
177 }
178
179 /* Wrap the temp fd in a POSIX VFS handle (sd == NULL -> default driver). */
180 ngx_memzero(&tvctx, sizeof(tvctx));
181 tvctx.rootfd = -1;
182 tvctx.pool = r->pool;
183 tvctx.log = t->log;
184 tvctx.sd = NULL;
185 tvctx.is_tls = t->is_tls;
186 if (brix_vfs_adopt_fd(&tvctx, t->fs_path, t->tmp_fd, 0, 0, &tfh) != NGX_OK) {
187 (void) close(t->tmp_fd);
188 ngx_log_error(NGX_LOG_ERR, c->log, ngx_errno,
189 "serve offload: temp adopt failed for \"%s\"", t->fs_path);
190 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
191 ngx_http_run_posted_requests(c);
192 return;
193 }
194
195 ngx_memzero(&vst, sizeof(vst));
196 vst.size = t->size;
197 vst.mtime = t->mtime;
198 vst.mode = S_IFREG | 0644;
199 vst.is_regular = 1;
200
201 ngx_memzero(&opts, sizeof(opts));
202 opts.xfer_proto = t->xfer_proto;
203 opts.op_name = t->op_name;
204 opts.identity = t->identity;
205 opts.etag_flags = t->etag_flags;
206 opts.compress = t->compress;
207 /* The protocol pre-header hook references request-stack state that did not
208 * survive the offload, so it is not run: a materialised serve omits the
209 * checksum / xrdhttp / response-override headers (the bytes/range are exact). */
210
211 rc = brix_http_serve_file_ranged(r, tfh, &vst, t->fs_path, &opts, &result);
212 if (t->metrics_cb != NULL) {
213 t->metrics_cb(r, &result);
214 }
215 ngx_http_finalize_request(r, rc);
216 ngx_http_run_posted_requests(c);
217}

219/* Lazily resolve the export's async thread pool (the webdav/copy.c idiom). */
220static ngx_thread_pool_t *
221serve_offload_pool(ngx_http_brix_shared_conf_t *common)
222{
223 ngx_thread_pool_t *pool = common->thread_pool;
224
225 if (pool == NULL) {
226 static ngx_str_t default_name = ngx_string("default");
227 ngx_str_t *pname = common->thread_pool_name.len > 0
228 ? &common->thread_pool_name : &default_name;
229
230 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
231 if (pool != NULL) {
232 common->thread_pool = pool;
233 }
234 }
235 return pool;
236}

238ngx_int_t
239brix_http_serve_offload_remote(ngx_http_request_t *r,
240 brix_sd_instance_t *inst, const char *key, const char *fs_path,
241 const brix_http_serve_opts_t *opts,
242 ngx_http_brix_shared_conf_t *common,
243 brix_http_serve_metrics_pt metrics_cb)
244{
245 ngx_thread_task_t *task;
246 serve_offload_ctx *t;
247 ngx_thread_pool_t *pool;
248 int tmp_fd;
249
250 if (r == NULL || inst == NULL || key == NULL || fs_path == NULL
251 || opts == NULL || common == NULL || r->header_only
252 || !serve_is_remote_socket(inst))
253 {
254 return NGX_DECLINED; /* serve inline (local / in-process / HEAD) */
255 }
256
257 pool = serve_offload_pool(common);
258 if (pool == NULL) {
259 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
260 "serve offload: \"%s\" reads from a remote socket backend but no "
261 "thread pool is configured - serving inline (may stall)", fs_path);
262 return NGX_DECLINED;
263 }
264
265 tmp_fd = serve_offload_tmp_open(r->connection->log);
266 if (tmp_fd < 0) {
267 return NGX_ERROR;
268 }
269
270 task = ngx_thread_task_alloc(r->pool, sizeof(serve_offload_ctx));
271 if (task == NULL) {
272 (void) close(tmp_fd);
273 return NGX_ERROR;
274 }
275 t = task->ctx;
276 t->r = r;
277 t->inst = inst;
278 t->metrics_cb = metrics_cb;
279 t->log = r->connection->log;
280 t->etag_flags = opts->etag_flags;
281 t->xfer_proto = opts->xfer_proto;
282 t->tmp_fd = tmp_fd;
283 t->mret = EIO;
284 t->compress = opts->compress ? 1 : 0;
285 t->is_tls = 0;
286#if (NGX_HTTP_SSL)
287 t->is_tls = (r->connection->ssl != NULL) ? 1 : 0;
288#endif
289 ngx_cpystrn((u_char *) t->key, (u_char *) key, sizeof(t->key));
290 ngx_cpystrn((u_char *) t->fs_path, (u_char *) fs_path, sizeof(t->fs_path));
291 ngx_cpystrn((u_char *) t->op_name,
292 (u_char *) (opts->op_name ? opts->op_name : ""),
293 sizeof(t->op_name));
294 ngx_cpystrn((u_char *) t->identity,
295 (u_char *) (opts->identity ? opts->identity : ""),
296 sizeof(t->identity));
297
298 brix_task_bind(task, serve_offload_thread, serve_offload_done);
299 task->event.log = r->connection->log;
300
301 if (ngx_thread_task_post(pool, task) != NGX_OK) {
302 (void) close(tmp_fd);
303 return NGX_ERROR;
304 }
305
306 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
307 "serve offload: materialising remote \"%s\" on the thread pool", fs_path);
308 r->main->count++;
309 return NGX_DONE;
310}
