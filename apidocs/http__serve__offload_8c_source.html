29 brix_http_serve_metrics_pt metrics_cb;
30 ngx_log_t *log;

32 time_t mtime; /* captured object mtime */
33 int etag_flags;
34 int xfer_proto;
35 int tmp_fd; /* O_TMPFILE / unlinked scratch */
36 int mret; /* 0 ok, else errno */
37 unsigned compress:1;
38 unsigned is_tls:1;
39 char key[PATH_MAX]; /* export-relative open key */
40 char fs_path[PATH_MAX]; /* logical path for headers */
41 char op_name[24];

43 /* Per-user backend credential (phase-2 follow-up), detached-copied at
44 * submit (see brix_stage_cred_t in fs/xfer/stage_engine.h for the same
45 * pattern/rationale): the worker thread outlives the request pool
46 * boundary that owns the borrowed brix_sd_cred_t strings, so the raw
47 * pointers cannot cross into the thread — only these fixed buffers can.
48 * use_cred=0 means the gate resolved a service-credential fallback (or
49 * the ctx had no cred policy bound at all); the thread then opens with a
50 * NULL cred, exactly as brix_sd_open_maybe_cred's NULL-cred contract. */
51 unsigned use_cred:1;
52 char cred_x509_proxy[BRIX_UCRED_PATH_MAX];
53 char cred_bearer[BRIX_UCRED_BEARER_MAX];
54 char cred_key[BRIX_UCRED_KEY_MAX];

56 char cred_dir[BRIX_UCRED_PATH_MAX];

58} serve_offload_ctx;

81/* 1 iff serving `inst` reads from a socket-wire backend (one that cannot pump its
82 * blocking socket on the un-pumped event loop). A cache serves from its STORE; a
83 * stage decorator serves from its SOURCE; a bare driver answers by name. Today the
84 * only such driver is "xroot"; in-process (rados) and curl (s3/http) block-but-
85 * complete on-loop and are served inline. */
86static int
87serve_is_remote_socket(const brix_sd_instance_t *inst)
88{
89 if (inst == NULL) {
90 return 0;
91 }
92 if (brix_sd_cache_instance_is(inst)) {
93 return serve_is_remote_socket(brix_sd_cache_store_instance(inst));
94 }
95 if (brix_sd_stage_instance_is(inst)) {
96 return serve_is_remote_socket(brix_sd_stage_source_instance(inst));
97 }
98 return (inst->driver != NULL && inst->driver->name != NULL
99 && ngx_strcmp(inst->driver->name, "xroot") == 0) ? 1 : 0;
100}

102/* Open an anonymous, auto-cleaned temp file for the materialised object. Prefers
103 * O_TMPFILE (no name, removed on last close); falls back to mkstemp+unlink. A
104 * transient worker-owned scratch buffer, not export storage. */
105static int
106serve_offload_tmp_open(ngx_log_t *log)
107{
108 const char *dir = getenv("TMPDIR");
109 int fd;
110
111 if (dir == NULL || dir[0] == '\0') {
112 dir = "/tmp";
113 }
114 fd = open(dir, O_TMPFILE | O_RDWR | O_CLOEXEC, 0600); /* vfs-seam-allow: transient serve scratch (not export storage) */
115 if (fd >= 0) {
116 return fd;
117 }
118 {
119 char tmpl[PATH_MAX];
120 int n = ngx_snprintf((u_char *) tmpl, sizeof(tmpl) - 1,
121 "%s/xrd-serve.XXXXXX", dir) - (u_char *) tmpl;
122
123 tmpl[n] = '\0';
124 fd = mkstemp(tmpl); /* vfs-seam-allow: transient serve scratch (not export storage) */
125 if (fd >= 0) {
126 (void) unlink(tmpl); /* vfs-seam-allow: transient serve scratch (not export storage) */
127 return fd;
128 }
129 }
130 ngx_log_error(NGX_LOG_ERR, log, errno,
131 "serve offload: cannot open a temp scratch file under \"%s\"", dir);
132 return -1;
133}

135/* Rebuild a stack brix_sd_cred_t from the task ctx's detached-copied buffers.
136 *
137 * WHAT: Reconstitute the per-user credential the event loop resolved, mapping
138 * each empty fixed buffer back to a NULL field.
139 * WHY: The request pool that owned the original borrowed brix_sd_cred_t
140 * strings does not outlive this worker thread call, so the credential
141 * must travel as fixed-buffer copies and be rebuilt here (see the
142 * serve_offload_ctx struct comment + http_serve_offload.h PHASE-2).
143 * HOW: Zero the out cred, then point each field at its buffer iff non-empty.
144 */
145static void
146serve_offload_rebuild_cred(const serve_offload_ctx *t, brix_sd_cred_t *cred)
147{
148 ngx_memzero(cred, sizeof(*cred));
149 cred->x509_proxy = (t->cred_x509_proxy[0] != '\0')
150 ? t->cred_x509_proxy : NULL;
151 cred->bearer = (t->cred_bearer[0] != '\0') ? t->cred_bearer : NULL;
152 cred->key = (t->cred_key[0] != '\0') ? t->cred_key : NULL;
153 cred->principal = (t->cred_principal[0] != '\0')
154 ? t->cred_principal : NULL;
155 cred->cred_dir = (t->cred_dir[0] != '\0') ? t->cred_dir : NULL;
156 cred->fallback_deny = t->cred_fallback_deny;
157}

159/* Open the composed instance for the task's key with the per-user credential.
160 *
161 * WHAT: Open through brix_sd_open_maybe_cred so a remote-backed export sees the
162 * REQUESTING USER's credential (or a NULL cred when use_cred=0), exactly
163 * as the caller's own inline brix_vfs_open would have.
164 * WHY: use_cred=0 (no policy bound, or the gate resolved a service-credential
165 * fallback) opens with a NULL cred, unchanged from before this fix.
166 * HOW: Rebuild the stack cred iff use_cred, then delegate to the open verb;
167 * returns the opened obj or NULL (with *err set to the open errno).
168 */
169static brix_sd_obj_t *
170serve_offload_thread_open(const serve_offload_ctx *t, int *err)
171{
172 if (t->use_cred) {
173 brix_sd_cred_t cred;
174
175 serve_offload_rebuild_cred(t, &cred);
176 return brix_sd_open_maybe_cred(t->inst, t->key, BRIX_SD_O_READ, 0,
177 &cred, err);
178 }
179 return brix_sd_open_maybe_cred(t->inst, t->key, BRIX_SD_O_READ, 0,
180 NULL, err);
181}

183/* Materialise the opened object into the scratch temp fd through the driver seam.
184 *
185 * WHAT: Capture the object mtime, then copy every byte into the POSIX-wrapped
186 * temp fd; sets t->mtime / t->size / t->mret.
187 * WHY: Keeps the byte-pump (buffer alloc, fstat, drain) out of the thread
188 * orchestrator so the open and copy phases read as one line each.
189 * HOW: fstat for a fresh mtime snapshot, malloc the chunk, xvfs_drain owns the
190 * chunked pread->pwrite + EINTR loop; the caller still owns obj close.
191 */
192static void
193serve_offload_thread_materialise(serve_offload_ctx *t, brix_sd_obj_t *obj)
194{
195 brix_sd_obj_t dst; /* worker-owned scratch, driver-routed */
196 brix_sd_stat_t snap;
197 u_char *buf;
198 off_t off = 0;
199
200 snap = obj->snap;
201 if (obj->driver->fstat != NULL) {
202 (void) obj->driver->fstat(obj, &snap);
203 }
204 t->mtime = snap.mtime;
205
206 buf = malloc(BRIX_SERVE_OFFLOAD_CHUNK);
207 if (buf == NULL) {
208 t->mret = ENOMEM;
209 return;
210 }
211 /* Read from the (possibly remote/object) source obj, write to the POSIX-
212 * wrapped temp. xvfs_drain owns the chunked pread->pwrite + EINTR loop. */
213 brix_sd_posix_wrap(&dst, t->tmp_fd);
214 t->mret = (xvfs_drain(obj, &dst, buf, BRIX_SERVE_OFFLOAD_CHUNK, &off) == 0)
215 ? 0 : (errno ? errno : EIO);
216 free(buf);
217 if (t->mret == 0) {
218 t->size = off;
219 }
220}

222/* Worker thread: open the object through the composed driver (eager connect, cinfo
223 * load, miss fill all run here) and copy it into the temp fd. Pure blocking I/O off
224 * the event loop - the socket wire client opens + reads here safely. */
225static void
226serve_offload_thread(void *data, ngx_log_t *log)
227{
228 serve_offload_ctx *t = data;
229 brix_sd_obj_t *obj;
230 int err = 0;
231
232 (void) log;
233 if (t->inst->driver->open == NULL || t->inst->driver->pread == NULL) {
234 t->mret = ENOSYS;
235 return;
236 }
237
238 obj = serve_offload_thread_open(t, &err);
239 if (obj == NULL) {
240 t->mret = err ? err : EIO;
241 return;
242 }
243
244 serve_offload_thread_materialise(t, obj);
245
246 obj->driver->close(obj);
247 if (obj->heap_shell) { free(obj); }
248}

250/* Event loop: adopt the temp fd as a POSIX handle and serve it through the shared
251 * sendfile pipeline, run the protocol metrics callback, then finalise. The single
252 * ngx_http_finalize_request balances the r->main->count++ taken at post. */
253static void
254serve_offload_done(ngx_event_t *ev)
255{
256 ngx_thread_task_t *task = ev->data;
257 serve_offload_ctx *t = task->ctx;
258 ngx_http_request_t *r = t->r;
259 ngx_connection_t *c = r->connection;
260 brix_vfs_ctx_t tvctx;
261 brix_vfs_file_t *tfh = NULL;
262 brix_vfs_stat_t vst;
263 brix_http_serve_opts_t opts;
264 brix_http_serve_result_t result;
265 brix_vfs_adopt_attrs_t tattrs;
266 ngx_int_t rc;
267
268 if (t->mret != 0) {
269 if (t->tmp_fd >= 0) {
270 (void) close(t->tmp_fd);
271 }
272 ngx_log_error(NGX_LOG_ERR, c->log, t->mret,
273 "serve offload: materialise failed for \"%s\" - returning %d",
274 t->fs_path, (t->mret == ENOENT) ? 404 : 502);
275 ngx_http_finalize_request(r,
276 (t->mret == ENOENT) ? NGX_HTTP_NOT_FOUND : NGX_HTTP_BAD_GATEWAY);
277 ngx_http_run_posted_requests(c);
278 return;
279 }
280
281 /* Wrap the temp fd in a POSIX VFS handle (sd == NULL -> default driver). */
282 ngx_memzero(&tvctx, sizeof(tvctx));
283 tvctx.rootfd = -1;
284 tvctx.pool = r->pool;
285 tvctx.log = t->log;
286 tvctx.sd = NULL;
287 tvctx.is_tls = t->is_tls;
288 ngx_memzero(&tattrs, sizeof(tattrs)); /* from_cache=0, writable=0 */
289 if (brix_vfs_adopt_fd(&tvctx, t->fs_path, t->tmp_fd, tattrs, &tfh)
290 != NGX_OK)
291 {
292 (void) close(t->tmp_fd);
293 ngx_log_error(NGX_LOG_ERR, c->log, ngx_errno,
294 "serve offload: temp adopt failed for \"%s\"", t->fs_path);
295 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
296 ngx_http_run_posted_requests(c);
297 return;
298 }
299
300 ngx_memzero(&vst, sizeof(vst));
301 vst.size = t->size;
302 vst.mtime = t->mtime;
303 vst.mode = S_IFREG | 0644;
304 vst.is_regular = 1;
305
306 ngx_memzero(&opts, sizeof(opts));
307 opts.xfer_proto = t->xfer_proto;
308 opts.op_name = t->op_name;
309 opts.identity = t->identity;
310 opts.etag_flags = t->etag_flags;
311 opts.compress = t->compress;
312 /* The protocol pre-header hook references request-stack state that did not
313 * survive the offload, so it is not run: a materialised serve omits the
314 * checksum / xrdhttp / response-override headers (the bytes/range are exact). */
315
316 rc = brix_http_serve_file_ranged(r, tfh, &vst, t->fs_path, &opts, &result);
317 if (t->metrics_cb != NULL) {
318 t->metrics_cb(r, &result);
319 }
320 ngx_http_finalize_request(r, rc);
321 ngx_http_run_posted_requests(c);
322}

343/* Run the per-user backend credential gate on the event loop.
344 *
345 * WHAT: Resolve the requesting user's backend credential (or a service-cred
346 * fallback) for this request, writing the borrowed cred into *ucred and
347 * the use-cred flag into *use_cred.
348 * WHY: Deny mode (no cred / not authorized, fallback forbidden) must refuse
349 * the whole GET with errno EACCES BEFORE any origin connection is
350 * attempted, exactly as the inline brix_vfs_open path already does. A
351 * NULL vctx (no bound ctx / no cred policy) skips the gate — use_cred
352 * stays 0, unchanged pre-fix behaviour.
353 * HOW: Zero the out cred; when vctx is set, delegate to brix_vfs_backend_cred
354 * and on deny log + set errno EACCES. Returns NGX_OK on grant/skip,
355 * NGX_ERROR on deny. `out->ustore` is the fixed backing storage the gate
356 * fills; `out->cred` borrows into it for the caller's stack lifetime.
357 */
358static ngx_int_t
359serve_offload_cred_gate(ngx_http_request_t *r, brix_vfs_ctx_t *vctx,
360 const char *fs_path, serve_offload_cred_t *out)
361{
362 int gate_err = 0;
363
364 ngx_memzero(&out->cred, sizeof(out->cred));
365 out->use_cred = 0;
366
367 if (vctx == NULL) {
368 return NGX_OK;
369 }
370 if (brix_vfs_backend_cred(vctx, &out->ustore, &out->cred, &out->use_cred,
371 &gate_err) == NGX_OK)
372 {
373 return NGX_OK;
374 }
375 ngx_log_error(NGX_LOG_ERR, r->connection->log,
376 gate_err ? gate_err : EACCES,
377 "serve offload: credential denied for \"%s\" - refusing "
378 "before origin open", fs_path);
379 errno = gate_err ? gate_err : EACCES;
380 return NGX_ERROR;
381}

383/* Detached-copy the resolved credential's strings into the task ctx.
384 *
385 * WHAT: Copy each non-NULL borrowed credential string into its fixed task
386 * buffer (empty buffer == NULL field), and set use_cred / fallback_deny.
387 * WHY: The borrowed brix_sd_cred_t pointers are only valid for the duration of
388 * the submit call; the worker thread outlives it, so only fixed buffers
389 * (rebuilt by serve_offload_rebuild_cred) may cross the thread boundary.
390 * HOW: Pre-clear every buffer, then ngx_cpystrn each present field; a NULL/0
391 * cred (use_cred=0) leaves all buffers empty — the thread opens NULL-cred.
392 */
393static void
394serve_offload_copy_cred(serve_offload_ctx *t, const serve_offload_cred_t *rc)
395{
396 const brix_sd_cred_t *ucred = &rc->cred;
397
398 t->use_cred = rc->use_cred ? 1 : 0;
399 t->cred_x509_proxy[0] = '\0';
400 t->cred_bearer[0] = '\0';
401 t->cred_key[0] = '\0';
402 t->cred_principal[0] = '\0';
403 t->cred_dir[0] = '\0';
404 t->cred_fallback_deny = 0;
405
406 if (!rc->use_cred) {
407 return;
408 }
409 if (ucred->x509_proxy != NULL) {
410 ngx_cpystrn((u_char *) t->cred_x509_proxy,
411 (u_char *) ucred->x509_proxy, sizeof(t->cred_x509_proxy));
412 }
413 if (ucred->bearer != NULL) {
414 ngx_cpystrn((u_char *) t->cred_bearer, (u_char *) ucred->bearer,
415 sizeof(t->cred_bearer));
416 }
417 if (ucred->key != NULL) {
418 ngx_cpystrn((u_char *) t->cred_key, (u_char *) ucred->key,
419 sizeof(t->cred_key));
420 }
421 if (ucred->principal != NULL) {
422 ngx_cpystrn((u_char *) t->cred_principal, (u_char *) ucred->principal,
423 sizeof(t->cred_principal));
424 }
425 if (ucred->cred_dir != NULL) {
426 ngx_cpystrn((u_char *) t->cred_dir, (u_char *) ucred->cred_dir,
427 sizeof(t->cred_dir));
428 }
429 t->cred_fallback_deny = ucred->fallback_deny;
430}

432/* Populate the per-serve task ctx from the request, opts, and resolved cred.
433 *
434 * WHAT: Fill every field of the freshly-allocated serve_offload_ctx: request
435 * handles, serve opts, the claimed temp fd, the detached credential, the
436 * TLS flag, and the copied path/name strings.
437 * WHY: Keeps the (long, mechanical) field-by-field population out of the
438 * orchestrator so it reads as gate -> alloc -> fill -> post.
439 * HOW: Direct assignment for scalars, serve_offload_copy_cred for the cred, and
440 * ngx_cpystrn for the fixed string buffers (never capture c->log lifetime
441 * state — t->log is the connection log pointer, valid for the request).
442 */
443static void
444serve_offload_fill_ctx(serve_offload_ctx *t, ngx_http_request_t *r,
445 const serve_offload_args_t *a, int tmp_fd, const serve_offload_cred_t *rc)
446{
447 const brix_http_serve_opts_t *opts = a->opts;
448
449 t->r = r;
450 t->inst = a->inst;
451 t->metrics_cb = a->metrics_cb;
452 t->log = r->connection->log;
453 t->etag_flags = opts->etag_flags;
454 t->xfer_proto = opts->xfer_proto;
455 t->tmp_fd = tmp_fd;
456 t->mret = EIO;
457 t->compress = opts->compress ? 1 : 0;
458
459 serve_offload_copy_cred(t, rc);
460
461 t->is_tls = 0;
462#if (NGX_HTTP_SSL)
463 t->is_tls = (r->connection->ssl != NULL) ? 1 : 0;
464#endif
465 ngx_cpystrn((u_char *) t->key, (u_char *) a->key, sizeof(t->key));
466 ngx_cpystrn((u_char *) t->fs_path, (u_char *) a->fs_path,
467 sizeof(t->fs_path));
468 ngx_cpystrn((u_char *) t->op_name,
469 (u_char *) (opts->op_name ? opts->op_name : ""),
470 sizeof(t->op_name));
471 ngx_cpystrn((u_char *) t->identity,
472 (u_char *) (opts->identity ? opts->identity : ""),
473 sizeof(t->identity));
474}

476ngx_int_t
477brix_http_serve_offload_remote(ngx_http_request_t *r,
478 brix_sd_instance_t *inst, const char *key, const char *fs_path,
479 const brix_http_serve_opts_t *opts,
480 ngx_http_brix_shared_conf_t *common,
481 brix_vfs_ctx_t *vctx,
482 brix_http_serve_metrics_pt metrics_cb)
483{
484 ngx_thread_task_t *task;
485 serve_offload_ctx *t;
486 ngx_thread_pool_t *pool;
487 int tmp_fd;
488 serve_offload_cred_t rc;
489 serve_offload_args_t args;
490
491 if (r == NULL || inst == NULL || key == NULL || fs_path == NULL
492 || opts == NULL || common == NULL || r->header_only
493 || !serve_is_remote_socket(inst))
494 {
495 return NGX_DECLINED; /* serve inline (local / in-process / HEAD) */
496 }
497
498 pool = serve_offload_pool(common);
499 if (pool == NULL) {
500 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
501 "serve offload: \"%s\" reads from a remote socket backend but no "
502 "thread pool is configured - serving inline (may stall)", fs_path);
503 return NGX_DECLINED;
504 }
505
506 if (serve_offload_cred_gate(r, vctx, fs_path, &rc) != NGX_OK) {
507 return NGX_ERROR;
508 }
509
510 tmp_fd = serve_offload_tmp_open(r->connection->log);
511 if (tmp_fd < 0) {
512 return NGX_ERROR;
513 }
514
515 task = ngx_thread_task_alloc(r->pool, sizeof(serve_offload_ctx));
516 if (task == NULL) {
517 (void) close(tmp_fd);
518 return NGX_ERROR;
519 }
520 t = task->ctx;
521 args.inst = inst;
522 args.key = key;
523 args.fs_path = fs_path;
524 args.opts = opts;
525 args.metrics_cb = metrics_cb;
526 serve_offload_fill_ctx(t, r, &args, tmp_fd, &rc);
527 /* The resolved secret is now detached-copied into the task ctx buffers;
528 * erase the gate's stack copy so it does not outlive the submit (A-4/T4). */
529 brix_sd_ucred_wipe(&rc.ustore);
530
531 brix_task_bind(task, serve_offload_thread, serve_offload_done);
532 task->event.log = r->connection->log;
533
534 if (ngx_thread_task_post(pool, task) != NGX_OK) {
535 (void) close(tmp_fd);
536 return NGX_ERROR;
537 }
538
539 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
540 "serve offload: materialising remote \"%s\" on the thread pool", fs_path);
541 r->main->count++;
542 return NGX_DONE;
543}
