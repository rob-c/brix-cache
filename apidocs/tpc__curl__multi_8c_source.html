8/*
9 * webdav_tpc_run_curl_multi_finish — tear down a curl_multi run and record the
10 * success/error metric, returning rc unchanged.
11 *
12 * Called at every exit of webdav_tpc_run_curl_pull_multi (success and each early
13 * failure). The easy[], hdrs[] and resolve[] arrays start zeroed, so iterating
14 * all n_streams slots frees exactly the handles that were set up before the exit
15 * — letting the driver exit with a plain `return` instead of a shared
16 * goto/label. cm is created before the first failure site, so it is always live.
17 */
18ngx_int_t
19webdav_tpc_run_curl_multi_finish(ngx_int_t rc, CURLM *cm, CURL **easy,
20 struct curl_slist **hdrs, struct curl_slist **resolve,
21 ngx_uint_t n_streams, int fd)
22{
23 ngx_uint_t i;
24
25 for (i = 0; i < n_streams; i++) {
26 if (easy[i] != NULL) {
27 curl_multi_remove_handle(cm, easy[i]);
28 curl_easy_cleanup(easy[i]);
29 }
30 if (hdrs[i] != NULL) {
31 curl_slist_free_all(hdrs[i]);
32 }
33 if (resolve[i] != NULL) {
34 curl_slist_free_all(resolve[i]);
35 }
36 }
37 curl_multi_cleanup(cm);
38 close(fd);
39
40 if (rc == NGX_OK) {
41 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_SUCCESS]);
42 } else {
43 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_ERROR]);
44 }
45 return rc;
46}

49/*
50 * tpc_ms_prepare_output — create and pre-size the multi-stream assembly temp.
51 *
52 * WHAT: opens tmp_path O_CREAT|O_TRUNC and ftruncates it to total_size so every
53 * stream can pwrite its disjoint range in place.
54 * WHY: factors the fd lifecycle (open + failure-close on ftruncate) out of the
55 * driver so the driver never owns a half-set-up fd.
56 * HOW: returns the ready fd (>= 0) on success; -1 on open or ftruncate
57 * failure (any opened fd is closed before returning).
58 */
59static int
60tpc_ms_prepare_output(ngx_log_t *log, const char *tmp_path, off_t total_size)
61{
62 int fd;
63 brix_sd_obj_t obj;
64
65 /* Pre-create the output file at full size so all streams can pwrite. */
66 fd = open(tmp_path, O_WRONLY | O_CREAT | O_TRUNC | O_NOFOLLOW, 0600); /* vfs-seam-allow: TPC multi-stream assembly temp (committed via rename); O_NOFOLLOW matches the sibling staging opens */
67 if (fd < 0) {
68 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
69 "brix_webdav: multi-stream: open(\"%s\") failed",
70 tmp_path);
71 return -1;
72 }
73
74 brix_sd_posix_wrap(&obj, fd);
75 if (brix_sd_posix_driver.ftruncate(&obj, total_size) != NGX_OK) {
76 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
77 "brix_webdav: multi-stream: ftruncate failed");
78 close(fd);
79 return -1;
80 }
81 return fd;
82}

89/*
90 * tpc_ms_setup_stream — configure one range-covering easy handle and add it to
91 * the multi handle.
92 *
93 * WHAT: computes stream i's byte range, initialises its write context, creates
94 * and configures the easy handle (conf/egress + Range + write cb + pmark),
95 * and registers it with ctx->cm.
96 * WHY: isolates the per-stream construction (with its two distinct early-fail
97 * codes) so the driver loop stays flat.
98 * HOW: writes the handle/slists into ctx->easy[i]/hdrs[i]/resolve[i] and the
99 * write context into ctx->write_ctx[i]. Returns NGX_OK, or the mapped
100 * HTTP status (INTERNAL on curl_easy_init OOM, FORBIDDEN on egress/OOM) —
101 * the caller tears down all slots via webdav_tpc_run_curl_multi_finish().
102 */
103static ngx_int_t
104tpc_ms_setup_stream(webdav_tpc_ms_ctx_t *ctx, ngx_uint_t i)
105{
106 off_t chunk = ctx->total_size / (off_t) ctx->n_streams;
107 off_t start = (off_t) i * chunk;
108 off_t end = (i == ctx->n_streams - 1)
109 ? ctx->total_size - 1 : start + chunk - 1;
110 char range_buf[64];
111
112 ctx->write_ctx[i].fd = ctx->fd;
113 ctx->write_ctx[i].cur_offset = start;
114 ctx->write_ctx[i].stream_idx = i;
115 ctx->write_ctx[i].progress = ctx->progress;
116
117 ctx->easy[i] = curl_easy_init();
118 if (ctx->easy[i] == NULL) {
119 return NGX_HTTP_INTERNAL_SERVER_ERROR;
120 }
121
122 if (tpc_curl_apply_conf(ctx->easy[i], ctx->conf, ctx->source_url,
123 ctx->transfer_headers, ctx->log,
124 ctx->user_cert, ctx->user_key,
125 &ctx->hdrs[i], &ctx->resolve[i]) < 0) {
126 /* Egress check failed (prohibited/rebind) or OOM — abort. */
127 return NGX_HTTP_FORBIDDEN;
128 }
129
130 snprintf(range_buf, sizeof(range_buf), "%lld-%lld",
131 (long long) start, (long long) end);
132 curl_easy_setopt(ctx->easy[i], CURLOPT_RANGE, range_buf);
133 curl_easy_setopt(ctx->easy[i], CURLOPT_WRITEFUNCTION, ms_write_cb);
134 curl_easy_setopt(ctx->easy[i], CURLOPT_WRITEDATA, &ctx->write_ctx[i]);
135
136#ifdef WEBDAV_TPC_PMARK_SOCKCB
137 /* SciTags: mark each parallel pull stream's outbound socket. pmrec[]
138 * lives until the curl_multi loop + cleanup below. */
139 webdav_tpc_pmark_attach(ctx->easy[i], &ctx->pmrec[i], ctx->conf, 0 /*pull*/,
140 ctx->tmp_path, ctx->log);
141#endif
142
143 curl_multi_add_handle(ctx->cm, ctx->easy[i]);
144 return NGX_OK;
145}

148/*
149 * tpc_ms_drive — pump all handles on cm to completion.
150 *
151 * WHAT: runs the curl_multi_perform / curl_multi_wait loop until no handle is
152 * still running.
153 * WHY: separates the event pump (with its wait-error early exit) from result
154 * harvesting.
155 * HOW: NGX_OK when the loop drains normally; NGX_HTTP_BAD_GATEWAY on a
156 * curl_multi_wait error.
157 */
158static ngx_int_t
159tpc_ms_drive(ngx_log_t *log, CURLM *cm)
160{
161 int still_running;
162
163 curl_multi_perform(cm, &still_running);
164 while (still_running) {
165 int numfds;
166 CURLMcode mc = curl_multi_wait(cm, NULL, 0, 1000, &numfds);
167 if (mc != CURLM_OK) {
168 ngx_log_error(NGX_LOG_ERR, log, 0,
169 "brix_webdav: curl_multi_wait error: %s",
170 curl_multi_strerror(mc));
171 return NGX_HTTP_BAD_GATEWAY;
172 }
173 curl_multi_perform(cm, &still_running);
174 }
175 return NGX_OK;
176}

179/*
180 * tpc_ms_harvest — drain completion messages; any stream error fails the run.
181 *
182 * WHAT: reads CURLMSG_DONE messages and maps the first per-stream failure to
183 * NGX_HTTP_BAD_GATEWAY (all messages are still drained/logged).
184 * WHY: keeps result inspection out of the driver body.
185 * HOW: NGX_OK when every stream reported CURLE_OK; NGX_HTTP_BAD_GATEWAY
186 * otherwise.
187 */
188static ngx_int_t
189tpc_ms_harvest(ngx_log_t *log, CURLM *cm)
190{
191 CURLMsg *msg;
192 int msgs_left;
193 ngx_int_t rc = NGX_OK;
194
195 while ((msg = curl_multi_info_read(cm, &msgs_left)) != NULL) {
196 if (msg->msg == CURLMSG_DONE && msg->data.result != CURLE_OK) {
197 ngx_log_error(NGX_LOG_WARN, log, 0,
198 "brix_webdav: HTTP-TPC multi-stream failed: %s",
199 curl_easy_strerror(msg->data.result));
200 rc = NGX_HTTP_BAD_GATEWAY;
201 }
202 }
203 return rc;
204}

207/*
208 * webdav_tpc_run_curl_pull_multi — parallel Range-based GET using curl_multi.
209 *
210 * Issues a HEAD request to learn Content-Length. On success, divides the file
211 * into n_streams equal-sized byte ranges and runs them concurrently via
212 * curl_multi_perform. Each stream writes to the output file via pwrite at its
213 * own offset so the file is assembled in-place without post-merge overhead.
214 *
215 * Falls back to single-stream when Content-Length is unknown (chunked source or
216 * HEAD failure) or when n_streams <= 1.
217 *
218 * progress may be NULL when called from the non-marker thread path. When
219 * non-NULL, each stream's write callback increments bytes_per_stream[i]
220 * atomically so the event-loop poll timer can report per-stream progress.
221 *
222 * Body reads as: fall back → prepare output → set up N streams → drive →
223 * harvest → finish. Every exit past cm-init routes through
224 * webdav_tpc_run_curl_multi_finish() so exactly the set-up handles are freed.
225 * The 8-param signature is frozen (out-of-file callers in tpc_marker.c /
226 * tpc_thread.c) and reported as a residual.
227 */
228ngx_int_t
229webdav_tpc_run_curl_pull_multi(ngx_log_t *log,
230 ngx_http_brix_webdav_loc_conf_t *conf,
231 const char *source_url, const char *tmp_path,
232 ngx_array_t *transfer_headers, ngx_uint_t n_streams,
233 uint64_t transfer_id, tpc_ms_progress_t *progress,
234 const char *user_cert, const char *user_key)
235{
236 off_t total_size;
237 CURLM *cm = NULL;
238 CURL *easy[BRIX_TPC_MAX_STREAMS];
239 struct curl_slist *hdrs[BRIX_TPC_MAX_STREAMS];
240 struct curl_slist *resolve[BRIX_TPC_MAX_STREAMS];
241 ms_stream_ctx_t write_ctx[BRIX_TPC_MAX_STREAMS];
242#ifdef WEBDAV_TPC_PMARK_SOCKCB
243 webdav_tpc_pmark_rec_t pmrec[BRIX_TPC_MAX_STREAMS];
244#endif
245 webdav_tpc_ms_ctx_t msctx;
246 int fd = -1;
247 ngx_uint_t i;
248 ngx_int_t rc = NGX_OK;
249
250 if (n_streams <= 1) {
251 return webdav_tpc_run_curl_pull(log, conf, source_url, tmp_path,
252 transfer_headers, transfer_id,
253 user_cert, user_key);
254 }
255 if (n_streams > BRIX_TPC_MAX_STREAMS) {
256 n_streams = BRIX_TPC_MAX_STREAMS;
257 }
258
259 /* HEAD to learn file size so we can split into ranges. */
260 total_size = tpc_curl_head_size(log, conf, source_url, transfer_headers,
261 user_cert, user_key);
262 if (total_size <= 0) {
263 ngx_log_error(NGX_LOG_INFO, log, 0,
264 "brix_webdav: multi-stream: unknown Content-Length,"
265 " falling back to single stream");
266 return webdav_tpc_run_curl_pull(log, conf, source_url, tmp_path,
267 transfer_headers, transfer_id,
268 user_cert, user_key);
269 }
270
271 if (progress != NULL) {
272 progress->total_size = total_size;
273 }
274
275 fd = tpc_ms_prepare_output(log, tmp_path, total_size);
276 if (fd < 0) {
277 return NGX_HTTP_INTERNAL_SERVER_ERROR;
278 }
279
280 cm = curl_multi_init();
281 if (cm == NULL) {
282 close(fd);
283 return NGX_HTTP_INTERNAL_SERVER_ERROR;
284 }
285
286 ngx_memzero(easy, sizeof(easy));
287 ngx_memzero(hdrs, sizeof(hdrs));
288 ngx_memzero(resolve, sizeof(resolve));
289
290 ngx_memzero(&msctx, sizeof(msctx));
291 msctx.log = log;
292 msctx.conf = conf;
293 msctx.source_url = source_url;
294 msctx.tmp_path = tmp_path;
295 msctx.transfer_headers = transfer_headers;
296 msctx.user_cert = user_cert;
297 msctx.user_key = user_key;
298 msctx.n_streams = n_streams;
299 msctx.total_size = total_size;
300 msctx.fd = fd;
301 msctx.cm = cm;
302 msctx.easy = easy;
303 msctx.hdrs = hdrs;
304 msctx.resolve = resolve;
305 msctx.write_ctx = write_ctx;
306 msctx.progress = progress;
307#ifdef WEBDAV_TPC_PMARK_SOCKCB
308 msctx.pmrec = pmrec;
309#endif
310
311 /* Set up N easy handles, each covering a disjoint byte range. */
312 for (i = 0; i < n_streams; i++) {
313 rc = tpc_ms_setup_stream(&msctx, i);
314 if (rc != NGX_OK) {
315 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs,
316 resolve, n_streams, fd);
317 }
318 }
319
320 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_STARTED]);
321
322 rc = tpc_ms_drive(log, cm);
323 if (rc != NGX_OK) {
324 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs,
325 resolve, n_streams, fd);
326 }
327
328 rc = tpc_ms_harvest(log, cm);
329
330 return webdav_tpc_run_curl_multi_finish(rc, cm, easy, hdrs, resolve,
331 n_streams, fd);
332}
