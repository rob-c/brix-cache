2 * WHAT: Three nginx event-loop callbacks managing outbound upstream redirector connection lifecycle. brix_upstream_wait_timer_handler(ev) is the kXR_wait expiry timer callback — checks ctx validity/cleaned flag, logs info-level "upstream kXR_wait expired; retrying", calls brix_upstream_send_request() to resend saved client request, aborts on failure. brix_upstream_write_handler(wev) handles non-blocking upstream write: checks ctx validity, aborts on timeout ("connect/write timeout"), if state=XRD_UP_CONNECTING performs SO_ERROR getsockopt check (abort on TCP connect failure, otherwise transitions to XRD_UP_BOOTSTRAP with bs_phase=BS_HANDSHAKE and resets response accumulator), if wbuf_pos < wbuf_len calls brix_upstream_flush() for partial write drain (abort on error), after flush completes arms read event via ngx_handle_read_event(uconn->read, 0). brix_upstream_read_handler(rev) implements response accumulation loop: checks ctx validity/timeout, accumulates XRD_RESPONSE_HDR_LEN bytes parsing ServerResponseHdr to extract resp_status/resps_dlen (abort on NGX_AGAIN with re-arm, abort on n<=0), allocates resp_body from uconn->pool for dlen>0 bytes with size cap MAX_PATH+256 (abort on oversized or pool failure), accumulates body bytes until full (NGX_AGAIN re-arms read event, n<=0 closes connection), dispatches based on state: XRD_UP_BOOTSTRAP→handle_bootstrap_response(), XRD_UP_REQUEST/XRD_UP_ASYNC→forward_response(), default abort ("unexpected state"). 14/*
15 * WHAT: Timer callback fired when a server-issued kXR_wait delay expires.
16 * WHY: When the backend answers an opcode with kXR_wait it is asking us to retry the
17 * same request after N seconds; this timer is the retry trigger.
18 * HOW: Validate the client session still exists, then resend the saved request via
19 * send_request(). On failure (with the upstream conn still live) abort the proxy.
20 */
21void
22brix_upstream_wait_timer_handler(ngx_event_t *ev)
23{
24 brix_upstream_t *up = ev->data;
25 brix_ctx_t *ctx = up->client_ctx;
26
27 /* Client session may have gone away while we were waiting — drop the upstream. */
28 if (ctx == NULL || ctx->destroyed) {
29 brix_upstream_cleanup(up);
30 return;
31 }
32
33 ngx_log_error(NGX_LOG_INFO, up->client_conn->log, 0,
34 "brix: upstream kXR_wait expired; retrying");
35
36 if (brix_upstream_send_request(up) != NGX_OK && up->conn != NULL) {
37 brix_upstream_abort(up, "upstream retry failed");
38 }
39}

41/*
42 * WHAT: Writable-event callback for the upstream socket: completes the async TCP connect
43 * and drains the pending write buffer (bootstrap bytes or a replayed request).
44 * WHY: nginx has no separate connect-completion callback for non-blocking sockets — the
45 * socket simply becomes writable. So the first write event doubles as the signal
46 * that connect() finished, which we confirm via SO_ERROR.
47 * HOW: While in CONNECTING, check SO_ERROR (abort on TCP failure) and transition to
48 * BOOTSTRAP/HANDSHAKE, zeroing the response accumulator. Then if wbuf still has
49 * unsent bytes, flush() and return (stay armed for the next writable event).
50 * Only once the buffer is fully drained do we arm the read side for the reply.
51 */
52void
53brix_upstream_write_handler(ngx_event_t *wev)
54{
55 ngx_connection_t *uconn = wev->data;
56 brix_upstream_t *up = uconn->data;
57 brix_ctx_t *ctx = up->client_ctx;
58
59 /* Don't act on a socket whose client session has already been torn down. */
60 if (ctx == NULL || ctx->destroyed) {
61 brix_upstream_cleanup(up);
62 return;
63 }
64
65 if (wev->timedout) {
66 brix_upstream_abort(up, "upstream connect/write timeout");
67 return;
68 }
69
70 if (up->state == XRD_UP_CONNECTING) {
71 int err = 0;
72 socklen_t len = sizeof(err);
73
74 /* SO_ERROR holds the pending connect() result; a getsockopt failure or a
75 * non-zero err means the async connect did not succeed. */
76 if (getsockopt(uconn->fd, SOL_SOCKET, SO_ERROR,
77 (char *) &err, &len) == -1 || err)
78 {
79 BRIX_DIAG_ERR(up->client_conn->log,
80 err ? err : ngx_socket_errno,
81 "brix: cannot connect to upstream data server",
82 "the upstream is down, unreachable, or refusing connections "
83 "(wrong host/port, or a firewall)",
84 "confirm the upstream xrootd/data server is up and reachable "
85 "from this host; the OS reason is appended below");
86 brix_upstream_abort(up, "upstream TCP connect failed");
87 return;
88 }
89
90 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, up->client_conn->log, 0,
91 "brix: upstream TCP connected");
92
93 /* Connect succeeded — begin bootstrap and reset the read accumulator so the
94 * first reply (the handshake response) parses from a clean slate. */
95 up->state = XRD_UP_BOOTSTRAP;
96 up->bs_phase = XRD_UP_BS_HANDSHAKE;
97 up->rhdr_pos = 0;
98 up->resp_dlen = 0;
99 up->resp_body = NULL;
100 up->resp_body_pos = 0;
101 }
102
103 /* Partial-write drain: flush() advances wbuf_pos. If bytes remain we return and
104 * wait for the next writable event rather than arming the read side prematurely. */
105 if (up->wbuf_pos < up->wbuf_len) {
106 ngx_int_t rc = brix_upstream_flush(up);
107
108 if (rc == NGX_ERROR) {
109 brix_upstream_abort(up, "upstream write error");
110 }
111 return;
112 }
113
114 /* Buffer fully sent — now wait for the upstream's reply. */
115 if (ngx_handle_read_event(uconn->read, 0) != NGX_OK) {
116 brix_upstream_abort(up, "upstream read arm failed in write handler");
117 }
118}

120/*
121 * WHAT: Perform a single non-blocking recv into buf[*pos..total), advancing *pos on
122 * success, and translate the three possible outcomes into a caller-return signal.
123 * WHY: Both accumulation stages of the read handler (header and body) share identical
124 * recv/NGX_AGAIN/peer-closed handling — only the destination buffer and the abort
125 * messages differ. Centralising it keeps the offsets (rhdr_pos/resp_body_pos)
126 * resumable across re-entries and removes a duplicated branch ladder.
127 * HOW: recv `total - *pos` bytes at buf + *pos. On NGX_AGAIN re-arm the read event
128 * (abort with again_msg if arming fails) and return NGX_DONE — the caller must
129 * return so re-entry resumes at the same offset. On <=0 abort with closed_msg and
130 * return NGX_DONE. On a real read advance *pos and return NGX_OK; the caller then
131 * decides (via the offset) whether the stage is complete or must loop for more.
132 */
133static ngx_int_t
134brix_upstream_recv_chunk(brix_upstream_t *up, ngx_connection_t *uconn,
135 ngx_event_t *rev, u_char *buf, size_t *pos, size_t total,
136 const char *again_msg, const char *closed_msg)
137{
138 size_t need = total - *pos;
139 ssize_t n = uconn->recv(uconn, buf + *pos, need);
140
141 if (n == NGX_AGAIN) {
142 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
143 brix_upstream_abort(up, again_msg);
144 }
145 return NGX_DONE;
146 }
147 if (n <= 0) {
148 brix_upstream_abort(up, closed_msg);
149 return NGX_DONE;
150 }
151
152 *pos += (size_t) n;
153 return NGX_OK;
154}

156/*
157 * WHAT: Decode the just-completed response header and, when a body follows, allocate a
158 * size-capped body buffer for stage 2.
159 * WHY: The 8-byte ServerResponseHdr carries the wire status and the body length that
160 * governs stage-2 accumulation. Untrusted `dlen` must be bounded before it drives
161 * an allocation, and the buffer needs a spare NUL so downstream parsers can treat
162 * the body as a C string.
163 * HOW: Read status (ntohs) and dlen (ntohl) from the header in host order. With dlen==0
164 * there is no body — return NGX_OK immediately. Otherwise reject a dlen above
165 * BRIX_MAX_PATH+256 (abort, NGX_ERROR), then ngx_palloc dlen+1 from the connection
166 * pool (abort on failure), NUL-terminate at [dlen], and reset resp_body_pos.
167 */
168static ngx_int_t
169brix_upstream_parse_header_and_alloc(brix_upstream_t *up,
170 ngx_connection_t *uconn)
171{
172 ServerResponseHdr *hdr = (ServerResponseHdr *) (void *) up->rhdr;
173
174 up->resp_status = ntohs(hdr->status);
175 up->resp_dlen = ntohl(hdr->dlen);
176
177 if (up->resp_dlen == 0) {
178 return NGX_OK;
179 }
180
181 /* Cap untrusted body size to bound allocation — the largest legitimate
182 * upstream payload here is a path plus protocol slack. */
183 if (up->resp_dlen > BRIX_MAX_PATH + 256) {
184 brix_upstream_abort(up, "upstream response body too large");
185 return NGX_ERROR;
186 }
187
188 /* +1 byte for a NUL terminator so the body can be treated as a C string by
189 * downstream parsers; allocated from the connection pool. */
190 up->resp_body = ngx_palloc(uconn->pool, up->resp_dlen + 1);
191 if (up->resp_body == NULL) {
192 brix_upstream_abort(up, "upstream pool alloc failed");
193 return NGX_ERROR;
194 }
195 up->resp_body[up->resp_dlen] = '\0';
196 up->resp_body_pos = 0;
197 return NGX_OK;
198}

200/*
201 * WHAT: Hand a fully-buffered header+body response to the correct consumer for the
202 * current connection state.
203 * WHY: Bootstrap and live-request phases consume responses differently: bootstrap replies
204 * drive the login FSM, while request/async replies are relayed verbatim. Any other
205 * state reaching a complete frame is a logic error worth aborting on.
206 * HOW: BOOTSTRAP → handle_bootstrap_response(); REQUEST/ASYNC → forward_response();
207 * anything else → abort. The handler returns unconditionally after this call.
208 */
209static void
210brix_upstream_dispatch_response(brix_upstream_t *up)
211{
212 /* Bootstrap replies (handshake/protocol/TLS/login) drive the login FSM, which
213 * advances bs_phase and, when DONE, replays the saved client request. */
214 if (up->state == XRD_UP_BOOTSTRAP) {
215 brix_upstream_handle_bootstrap_response(up);
216 return;
217 }
218
219 /* Live request/async replies are relayed verbatim to the client connection. */
220 if (up->state == XRD_UP_REQUEST || up->state == XRD_UP_ASYNC) {
221 brix_upstream_forward_response(up);
222 return;
223 }
224
225 /* No other state should ever reach the read handler with a complete frame. */
226 brix_upstream_abort(up, "upstream: unexpected state in read handler");
227}

229/*
230 * WHAT: Readable-event callback that accumulates one complete upstream response
231 * (fixed-size header + variable body) and dispatches it by connection state.
232 * WHY: XRootD replies are framed as an 8-byte ServerResponseHdr followed by `dlen`
233 * body bytes. Reads can fragment arbitrarily, so we must persist partial progress
234 * (rhdr_pos / resp_body_pos) across re-entries to this handler. A single response
235 * is consumed per invocation, after which we hand off and return.
236 * HOW: Two-stage accumulation inside one loop. Stage 1 fills rhdr to XRD_RESPONSE_HDR_LEN
237 * via recv_chunk() then parses status/dlen and allocates the body. Stage 2 fills the
238 * body via recv_chunk(). recv_chunk() returns NGX_DONE on NGX_AGAIN (re-armed) or
239 * peer-close (aborted) — in both cases we return so re-entry resumes at the same
240 * offset; on NGX_OK a short read loops to pull the remainder. Once both stages are
241 * complete, dispatch_response() routes the frame by connection state.
242 */
243void
244brix_upstream_read_handler(ngx_event_t *rev)
245{
246 ngx_connection_t *uconn = rev->data;
247 brix_upstream_t *up = uconn->data;
248 brix_ctx_t *ctx = up->client_ctx;
249
250 /* Stale event after the client session was destroyed: just clean up. */
251 if (ctx == NULL || ctx->destroyed) {
252 brix_upstream_cleanup(up);
253 return;
254 }
255
256 if (rev->timedout) {
257 brix_upstream_abort(up, "upstream read timeout");
258 return;
259 }
260
261 for (;;) {
262 /* Stage 1: accumulate the fixed-length response header. */
263 if (up->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
264 if (brix_upstream_recv_chunk(up, uconn, rev, up->rhdr,
265 &up->rhdr_pos, XRD_RESPONSE_HDR_LEN,
266 "upstream read arm failed (hdr)",
267 "upstream connection closed") != NGX_OK)
268 {
269 return;
270 }
271 if (up->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
272 /* Short read — loop to pull the rest of the header. */
273 continue;
274 }
275 if (brix_upstream_parse_header_and_alloc(up, uconn) != NGX_OK) {
276 return;
277 }
278 }
279
280 /* Stage 2: accumulate the variable-length body (skipped when dlen==0). */
281 if (up->resp_body_pos < up->resp_dlen) {
282 if (brix_upstream_recv_chunk(up, uconn, rev, up->resp_body,
283 &up->resp_body_pos, up->resp_dlen,
284 "upstream read arm failed (body)",
285 "upstream connection closed (body)") != NGX_OK)
286 {
287 return;
288 }
289 if (up->resp_body_pos < up->resp_dlen) {
290 /* More body to come — loop. */
291 continue;
292 }
293 }
294
295 /* Dispatch: a full header+body is now buffered. */
296 brix_upstream_dispatch_response(up);
297 return;
298 }
299}
