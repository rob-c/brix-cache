1/* File: events.c — Upstream connection non-blocking read/write event handlers and wait retry timer
2 * WHAT: Three nginx event-loop callbacks managing outbound upstream redirector connection lifecycle. brix_upstream_wait_timer_handler(ev) is the kXR_wait expiry timer callback — checks ctx validity/cleaned flag, logs info-level "upstream kXR_wait expired; retrying", calls brix_upstream_send_request() to resend saved client request, aborts on failure. brix_upstream_write_handler(wev) handles non-blocking upstream write: checks ctx validity, aborts on timeout ("connect/write timeout"), if state=XRD_UP_CONNECTING performs SO_ERROR getsockopt check (abort on TCP connect failure, otherwise transitions to XRD_UP_BOOTSTRAP with bs_phase=BS_HANDSHAKE and resets response accumulator), if wbuf_pos < wbuf_len calls brix_upstream_flush() for partial write drain (abort on error), after flush completes arms read event via ngx_handle_read_event(uconn->read, 0). brix_upstream_read_handler(rev) implements response accumulation loop: checks ctx validity/timeout, accumulates XRD_RESPONSE_HDR_LEN bytes parsing ServerResponseHdr to extract resp_status/resps_dlen (abort on NGX_AGAIN with re-arm, abort on n<=0), allocates resp_body from uconn->pool for dlen>0 bytes with size cap MAX_PATH+256 (abort on oversized or pool failure), accumulates body bytes until full (NGX_AGAIN re-arms read event, n<=0 closes connection), dispatches based on state: XRD_UP_BOOTSTRAP→handle_bootstrap_response(), XRD_UP_REQUEST/XRD_UP_ASYNC→forward_response(), default abort ("unexpected state"). All handlers clean up via brix_upstream_cleanup() when ctx destroyed. */

14/*
15 * WHAT: Timer callback fired when a server-issued kXR_wait delay expires.
16 * WHY: When the backend answers an opcode with kXR_wait it is asking us to retry the
17 * same request after N seconds; this timer is the retry trigger.
18 * HOW: Validate the client session still exists, then resend the saved request via
19 * send_request(). On failure (with the upstream conn still live) abort the proxy.
20 */
21void
22brix_upstream_wait_timer_handler(ngx_event_t *ev)
23{
24 brix_upstream_t *up = ev->data;
25 brix_ctx_t *ctx = up->client_ctx;
26
27 /* Client session may have gone away while we were waiting — drop the upstream. */
28 if (ctx == NULL || ctx->destroyed) {
29 brix_upstream_cleanup(up);
30 return;
31 }
32
33 ngx_log_error(NGX_LOG_INFO, up->client_conn->log, 0,
34 "brix: upstream kXR_wait expired; retrying");
35
36 if (brix_upstream_send_request(up) != NGX_OK && up->conn != NULL) {
37 brix_upstream_abort(up, "upstream retry failed");
38 }
39}

41/*
42 * WHAT: Writable-event callback for the upstream socket: completes the async TCP connect
43 * and drains the pending write buffer (bootstrap bytes or a replayed request).
44 * WHY: nginx has no separate connect-completion callback for non-blocking sockets — the
45 * socket simply becomes writable. So the first write event doubles as the signal
46 * that connect() finished, which we confirm via SO_ERROR.
47 * HOW: While in CONNECTING, check SO_ERROR (abort on TCP failure) and transition to
48 * BOOTSTRAP/HANDSHAKE, zeroing the response accumulator. Then if wbuf still has
49 * unsent bytes, flush() and return (stay armed for the next writable event).
50 * Only once the buffer is fully drained do we arm the read side for the reply.
51 */
52void
53brix_upstream_write_handler(ngx_event_t *wev)
54{
55 ngx_connection_t *uconn = wev->data;
56 brix_upstream_t *up = uconn->data;
57 brix_ctx_t *ctx = up->client_ctx;
58
59 /* Don't act on a socket whose client session has already been torn down. */
60 if (ctx == NULL || ctx->destroyed) {
61 brix_upstream_cleanup(up);
62 return;
63 }
64
65 if (wev->timedout) {
66 brix_upstream_abort(up, "upstream connect/write timeout");
67 return;
68 }
69
70 if (up->state == XRD_UP_CONNECTING) {
71 int err = 0;
72 socklen_t len = sizeof(err);
73
74 /* SO_ERROR holds the pending connect() result; a getsockopt failure or a
75 * non-zero err means the async connect did not succeed. */
76 if (getsockopt(uconn->fd, SOL_SOCKET, SO_ERROR,
77 (char *) &err, &len) == -1 || err)
78 {
79 BRIX_DIAG_ERR(up->client_conn->log,
80 err ? err : ngx_socket_errno,
81 "brix: cannot connect to upstream data server",
82 "the upstream is down, unreachable, or refusing connections "
83 "(wrong host/port, or a firewall)",
84 "confirm the upstream xrootd/data server is up and reachable "
85 "from this host; the OS reason is appended below");
86 brix_upstream_abort(up, "upstream TCP connect failed");
87 return;
88 }
89
90 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, up->client_conn->log, 0,
91 "brix: upstream TCP connected");
92
93 /* Connect succeeded — begin bootstrap and reset the read accumulator so the
94 * first reply (the handshake response) parses from a clean slate. */
95 up->state = XRD_UP_BOOTSTRAP;
96 up->bs_phase = XRD_UP_BS_HANDSHAKE;
97 up->rhdr_pos = 0;
98 up->resp_dlen = 0;
99 up->resp_body = NULL;
100 up->resp_body_pos = 0;
101 }
102
103 /* Partial-write drain: flush() advances wbuf_pos. If bytes remain we return and
104 * wait for the next writable event rather than arming the read side prematurely. */
105 if (up->wbuf_pos < up->wbuf_len) {
106 ngx_int_t rc = brix_upstream_flush(up);
107
108 if (rc == NGX_ERROR) {
109 brix_upstream_abort(up, "upstream write error");
110 }
111 return;
112 }
113
114 /* Buffer fully sent — now wait for the upstream's reply. */
115 if (ngx_handle_read_event(uconn->read, 0) != NGX_OK) {
116 brix_upstream_abort(up, "upstream read arm failed in write handler");
117 }
118}

120/*
121 * WHAT: Readable-event callback that accumulates one complete upstream response
122 * (fixed-size header + variable body) and dispatches it by connection state.
123 * WHY: XRootD replies are framed as an 8-byte ServerResponseHdr followed by `dlen`
124 * body bytes. Reads can fragment arbitrarily, so we must persist partial progress
125 * (rhdr_pos / resp_body_pos) across re-entries to this handler. A single response
126 * is consumed per invocation, after which we hand off and return.
127 * HOW: Two-stage accumulation inside one loop. Stage 1 fills rhdr to XRD_RESPONSE_HDR_LEN
128 * then parses status/dlen and (if dlen>0) allocates a capped body buffer. Stage 2
129 * fills the body. Each recv may yield NGX_AGAIN (re-arm and return — re-entry resumes
130 * at the same offset) or <=0 (peer closed — abort). Once both stages are complete,
131 * dispatch: BOOTSTRAP responses drive the login state machine; REQUEST/ASYNC
132 * responses are forwarded to the client; any other state is a logic error.
133 */
134void
135brix_upstream_read_handler(ngx_event_t *rev)
136{
137 ngx_connection_t *uconn = rev->data;
138 brix_upstream_t *up = uconn->data;
139 brix_ctx_t *ctx = up->client_ctx;
140 ssize_t n;
141
142 /* Stale event after the client session was destroyed: just clean up. */
143 if (ctx == NULL || ctx->destroyed) {
144 brix_upstream_cleanup(up);
145 return;
146 }
147
148 if (rev->timedout) {
149 brix_upstream_abort(up, "upstream read timeout");
150 return;
151 }
152
153 for (;;) {
154 /* Stage 1: accumulate the fixed-length response header */ if (up->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
155 size_t need = XRD_RESPONSE_HDR_LEN - up->rhdr_pos;
156
157 n = uconn->recv(uconn, up->rhdr + up->rhdr_pos, need);
158 if (n == NGX_AGAIN) {
159 /* Socket drained mid-header; re-arm and resume here next event. */
160 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
161 brix_upstream_abort(up, "upstream read arm failed (hdr)");
162 }
163 return;
164 }
165 if (n <= 0) {
166 brix_upstream_abort(up, "upstream connection closed");
167 return;
168 }
169
170 up->rhdr_pos += (size_t) n;
171 if (up->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
172 /* Short read — loop to pull the rest of the header. */
173 continue;
174 }
175
176 /* Header complete: decode wire fields (network byte order) into host
177 * order. status is uint16 at the start of the header; dlen is the uint32
178 * body length that determines how many more bytes to read in stage 2. */
179 {
180 ServerResponseHdr *hdr;
181
182 hdr = (ServerResponseHdr *) (void *) up->rhdr;
183 up->resp_status = ntohs(hdr->status);
184 up->resp_dlen = ntohl(hdr->dlen);
185 }
186
187 if (up->resp_dlen > 0) {
188 /* Cap untrusted body size to bound allocation — the largest legitimate
189 * upstream payload here is a path plus protocol slack. */
190 if (up->resp_dlen > BRIX_MAX_PATH + 256) {
191 brix_upstream_abort(up,
192 "upstream response body too large");
193 return;
194 }
195 /* +1 byte for a NUL terminator so the body can be treated as a C
196 * string by downstream parsers; allocated from the connection pool. */
197 up->resp_body = ngx_palloc(uconn->pool, up->resp_dlen + 1);
198 if (up->resp_body == NULL) {
199 brix_upstream_abort(up, "upstream pool alloc failed");
200 return;
201 }
202 up->resp_body[up->resp_dlen] = '\0';
203 up->resp_body_pos = 0;
204 }
205 }
206
207 /* Stage 2: accumulate the variable-length body (skipped when dlen==0) */ if (up->resp_body_pos < up->resp_dlen) {
208 size_t need = up->resp_dlen - up->resp_body_pos;
209
210 n = uconn->recv(uconn, up->resp_body + up->resp_body_pos, need);
211 if (n == NGX_AGAIN) {
212 /* Body fragmented across reads; re-arm and resume at this offset. */
213 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
214 brix_upstream_abort(up,
215 "upstream read arm failed (body)");
216 }
217 return;
218 }
219 if (n <= 0) {
220 brix_upstream_abort(up, "upstream connection closed (body)");
221 return;
222 }
223
224 up->resp_body_pos += (size_t) n;
225 if (up->resp_body_pos < up->resp_dlen) {
226 /* More body to come — loop. */
227 continue;
228 }
229 }
230
231 /* Dispatch: a full header+body is now buffered */
232 /* Bootstrap replies (handshake/protocol/TLS/login) drive the login FSM, which
233 * advances bs_phase and, when DONE, replays the saved client request. */
234 if (up->state == XRD_UP_BOOTSTRAP) {
235 brix_upstream_handle_bootstrap_response(up);
236 return;
237 }
238
239 /* Live request/async replies are relayed verbatim to the client connection. */
240 if (up->state == XRD_UP_REQUEST || up->state == XRD_UP_ASYNC) {
241 brix_upstream_forward_response(up);
242 return;
243 }
244
245 /* No other state should ever reach the read handler with a complete frame. */
246 brix_upstream_abort(up, "upstream: unexpected state in read handler");
247 return;
248 }
249}
