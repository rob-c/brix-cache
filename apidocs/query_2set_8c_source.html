5/*
6 * kXR_set (3018) — client sends advisory configuration hints to the server.
7 *
8 * WHAT: Handles the SET opcode where a client provides optional metadata about
9 * its identity or session parameters. The modifier byte determines which
10 * hint type is being sent; unrecognized modifiers are accepted with kXR_ok.
11 *
12 * WHY: CMS grid infrastructure sends "cms.space" appid reports to advertise
13 * disk capacity at storage nodes. Session TTL hints (clttl) let clients
14 * suggest keep-alive duration. The server accepts all SET requests per spec,
15 * enabling monitoring and operational visibility without strict enforcement.
16 *
17 * HOW: Parse modifier byte from the request header → switch on known values
18 * (appid, clttl) → handle payload for recognized types → log details at
19 * debug/info level → always return kXR_ok regardless of modifier value.
20 */

22/*
23 * CMS space report constants.
24 *
25 * The "cms.space" appid payload format is: "cms.space <total_bytes> <free_bytes>"

27 * to advertise storage node capacity in kXR_set(appid) requests.

30#define CMS_SPACE_PREFIX_LEN (sizeof(CMS_SPACE_PREFIX) - 1)

32/*
33 * brix_set_handle_cms_space — parse and log CMS space-availability report.
34 *
35 * WHAT: Extracts total and free byte counts from a "cms.space" payload string,
36 * then logs the capacity breakdown at INFO level for operational visibility.
37 *
38 * WHY: The CMS grid sends kXR_set(appid) with "cms.space <total> <free>" payloads
39 * to advertise storage node disk availability. This function parses those
40 * reports so operators can monitor capacity via nginx access logs.
41 *
42 * HOW: Skip the "cms.space" prefix, parse two ASCII decimal unsigned integers
43 * separated by a space using strtoull, validate separator presence between
44 * them, compute used = total - free, log at INFO level. Returns silently on
45 * malformed payload (warn logged).
46 */
47static void
48brix_set_handle_cms_space(ngx_connection_t *c, const char *payload,
49 size_t payload_len)
50{
51 /* Expected format: "cms.space <total_bytes> <free_bytes>" */
52 const char *p;
53 char *end;
54 unsigned long long total_bytes, free_bytes;
55
56 p = payload + CMS_SPACE_PREFIX_LEN;
57 while (p < payload + payload_len && *p == ' ') {
58 p++;
59 }
60
61 total_bytes = strtoull(p, &end, 10);
62 if (end == p || *end != ' ') {
63 ngx_log_error(NGX_LOG_WARN, c->log, 0,
64 "brix: kXR_set cms.space: malformed payload");
65 return;
66 }
67 p = end;
68 while (*p == ' ') {
69 p++;
70 }
71 free_bytes = strtoull(p, &end, 10);
72 if (end == p) {
73 ngx_log_error(NGX_LOG_WARN, c->log, 0,
74 "brix: kXR_set cms.space: missing free_bytes");
75 return;
76 }
77
78 ngx_log_error(NGX_LOG_INFO, c->log, 0,
79 "brix: cms.space report: total=%lluB free=%lluB "
80 "used=%lluB",
81 total_bytes, free_bytes,
82 (total_bytes > free_bytes) ? total_bytes - free_bytes : 0ULL);
83}

85/*
86 * brix_handle_set — handle kXR_set (3018) opcode.
87 *
88 * WHAT: Dispatches the SET opcode based on its modifier byte, handling known
89 * hint types (appid with CMS space reports, clttl session TTL) and
90 * returning kXR_ok for all modifiers including unrecognized ones.
91 *
92 * WHY: The XRootD protocol allows clients to send advisory hints about their
93 * identity or session parameters. The server accepts all SET requests per
94 * spec — it does not enforce TTL or validate appid format strictly, but
95 * logs recognized types for operational visibility (CMS space reports at
96 * INFO level, other modifiers at debug level).
97 *
98 * HOW: Extract modifier byte from request header → switch on known values
99 * (kXR_set_appid → parse CMS space payload; kXR_set_clttl → log silently)
100 * → default → accept with kXR_ok. Always return kXR_ok regardless of
101 * modifier value. Payload snippet logged at debug level for all types.
102 *
103 * Parameters:
104 * ctx — xrootd connection context containing parsed request header and payload
105 * c — nginx connection for logging
106 */
