3/*
4 * auth.c — authorization gate for third-party-copy initiation.
5 *
6 * WHAT: Implements brix_tpc_check_authz(), which decides whether a given
7 * identity may launch a TPC for a (src_path, dst_path) pair: read access
8 * to the source, write access to the destination.
9 *
10 * WHY: A TPC moves data on the user's behalf between two endpoints, so the
11 * initiating identity's token scope must actually permit the read and the
12 * write. This check runs before the transport begins so an unauthorised
13 * copy is rejected up front rather than failing mid-stream. S3 SigV4
14 * identities are blocked outright because their request-signed model does
15 * not carry a delegable scope suitable for third-party transfers.
16 *
17 * HOW: The static brix_tpc_check_scope_path() copies each ngx_str_t path into
18 * a NUL-terminated PATH_MAX buffer (rejecting over-long paths) and defers
19 * to brix_identity_check_token_scope() with the required read/write bit.
20 * A NULL/empty path is treated as "no constraint" (NGX_OK). The public
21 * entry point first rejects BRIX_AUTHN_S3KEY identities, then checks the
22 * source for read and the destination for write, returning NGX_OK only when
23 * both pass and NGX_DECLINED otherwise.
24 */
25

30 const ngx_str_t *path, int need_write, ngx_log_t *log)
31{
32 char local_path[PATH_MAX];
33
34 if (path == NULL || path->data == NULL || path->len == 0) {
35 return NGX_OK;
36 }
37
38 if (path->len >= sizeof(local_path)) {
39 ngx_log_error(NGX_LOG_WARN, log, 0,
40 "brix_tpc: authorization path is too long");
41 return NGX_DECLINED;
42 }
43
44 ngx_memcpy(local_path, path->data, path->len);
45 local_path[path->len] = '\0';
46
47 if (brix_identity_check_token_scope(identity, local_path, need_write)
48 != NGX_OK)
49 {
50 ngx_log_error(NGX_LOG_WARN, log, 0,
51 "brix_tpc: token scope denied %s access to \"%s\"",
52 need_write ? "write" : "read", local_path);
53 return NGX_DECLINED;
54 }
55
56 return NGX_OK;
57}

59/*
60 * Authorize a TPC initiated by `identity`: require read scope on src_path and
61 * write scope on dst_path. S3 SigV4 identities are refused outright. Either path
62 * may be NULL/empty to skip its check. Returns NGX_OK if permitted, else
63 * NGX_DECLINED.
64 */
65ngx_int_t
66brix_tpc_check_authz(const brix_identity_t *identity,
67 const ngx_str_t *src_path, const ngx_str_t *dst_path, ngx_log_t *log)
68{
69 if (identity != NULL && (identity->auth_method & BRIX_AUTHN_S3KEY)) {
70 ngx_log_error(NGX_LOG_WARN, log, 0,
71 "brix_tpc: S3 SigV4 identities cannot initiate TPC");
72 return NGX_DECLINED;
73 }
74
75 if (brix_tpc_check_scope_path(identity, src_path, 0, log) != NGX_OK) {
76 return NGX_DECLINED;
77 }
78
79 if (brix_tpc_check_scope_path(identity, dst_path, 1, log) != NGX_OK) {
80 return NGX_DECLINED;
81 }
82
83 return NGX_OK;
84}
