10/* Recursively walk the remote tree under `path` (depth-first, pre-order), invoking
11 * `visit` for every entry. Directories are descended up to XRDFS_MAXDEPTH. 0 / -1. */
12int
13walk_dir(brix_conn *c, const char *path, int depth, xrdfs_visit visit, void *u,
14 brix_status *st)
15{
16 brix_dirent *ents = NULL;
17 size_t n = 0, i;
18
19 if (brix_dirlist(c, path, 1 /*want_stat*/, &ents, &n, st) != 0) {
20 return -1;
21 }
22 for (i = 0; i < n; i++) {
23 char full[XRDC_PATH_MAX];
24 if (is_dot(ents[i].name) || join_path(path, ents[i].name, full, sizeof(full)) != 0) {
25 continue;
26 }
27 if (visit(full, &ents[i], depth, u) != 0) {
28 free(ents);
29 return 1;
30 }
31 if (ents[i].have_stat && (ents[i].st.flags & kXR_isDir)
32 && depth + 1 < XRDFS_MAXDEPTH) {
33 int rc = walk_dir(c, full, depth + 1, visit, u, st);
34 if (rc != 0) {
35 free(ents);
36 return rc;
37 }
38 }
39 }
40 free(ents);
41 return 0;
42}

47/* walk_dir visitor: chmod each entry, counting (and reporting) per-entry failures
48 * without aborting the walk. */
49int
50chmod_visit(const char *full, const brix_dirent *e, int depth, void *u)
51{
52 chmod_walk *w = (chmod_walk *) u;
53 brix_status st;
54 (void) e; (void) depth;

56 if (brix_chmod(w->c, full, w->mode, &st) != 0) {
57 fprintf(stderr, "xrdfs: chmod %s: %s\n", full, st.msg);
58 w->failures++;
59 }
60 return 0; /* keep walking */

64/* Recursively chmod every entry under `path` (the top path is chmod'd by the caller).
65 * *failures accumulates per-entry errors; returns 0 / -1 (walk-level error, st set). */
66int
67chmod_recursive(brix_conn *c, const char *path, int mode, int *failures,
68 brix_status *st)
69{
70 chmod_walk w;
71 int rc;
72 w.c = c; w.mode = mode; w.failures = 0;
73 rc = walk_dir(c, path, 0, chmod_visit, &w, st);
74 *failures = w.failures;
75 return (rc < 0) ? -1 : 0;
76}

81/* WHAT: print one du result in human-readable or JSON format.
82 * WHY: deduplicates the output logic shared by the multi-path and
83 * default-path branches of do_du.
84 * HOW: JSON emits a single-line object WITHOUT a trailing newline; the caller
85 * (do_du) owns array framing and item separators so the combined output
86 * is always a top-level JSON array (single path → array of 1). Human
87 * mode delegates to fmt_size and includes its own newline. */
88static void
89du_print(const char *path, const du_acc *a, int human, int json)
90{
91 if (json) {
92 fputc('{', stdout);
93 brix_json_kv_str(stdout, "path", path, 1);
94 brix_json_kv_ll(stdout, "bytes", (long long) a->bytes, 1);
95 brix_json_kv_ll(stdout, "files", (long long) a->files, 1);
96 brix_json_kv_ll(stdout, "dirs", (long long) a->dirs, 0);
97 fputc('}', stdout); /* caller handles array separators + trailing newline */
98 } else {
99 char sz[32];
100 fmt_size(a->bytes, sz, sizeof(sz), human);
101 printf("%-10s %s (%ld files, %ld dirs)\n", sz, path, a->files, a->dirs);
102 }
103}

122/* WHAT: du [-h] [-j] <path>... — recursive total size + file/dir counts.
123 * WHY: -j enables JSON output for scripting; -h humanizes byte counts.
124 * JSON output is always a top-level array so parsers can handle both
125 * single-path (array of 1) and multi-path invocations uniformly.
126 * HOW: flags are scanned first; if json, print "[" before any items.
127 * Each positional arg drives a walk_dir pass; du_print emits the object
128 * body (no trailing newline in JSON mode); do_du emits the comma-newline
129 * separator between items and the closing "]\n" at the end. */
130int
131do_du(brix_conn *c, const char *cwd, int argc, char **argv)
132{
133 int human = 0, json = 0, i, rc = 0, any = 0;
134 int jfirst = 1; /* tracks first JSON item for array framing */
135
136 for (i = 1; i < argc; i++) {
137 if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--human") == 0) { human = 1; }
138 else if (strcmp(argv[i], "-j") == 0 || strcmp(argv[i], "--json") == 0) { json = 1; }
139 }
140 if (json) { fputs("[\n", stdout); }
141 for (i = 1; i < argc; i++) {
142 char path[XRDC_PATH_MAX];
143 brix_status st;
144 du_acc a = { 0, 0, 0 };
145 if (argv[i][0] == '-') { continue; }
146 any = 1;
147 build_path(cwd, argv[i], path, sizeof(path));
148 brix_status_clear(&st);
149 if (walk_dir(c, path, 0, du_visit, &a, &st) != 0) {
150 fprintf(stderr, "xrdfs: du %s: %s\n", path, st.msg);
151 rc = brix_shellcode(&st);
152 continue;
153 }
154 if (json && !jfirst) { fputs(",\n", stdout); }
155 du_print(path, &a, human, json);
156 if (json) { jfirst = 0; }
157 }
158 if (!any) {
159 char path[XRDC_PATH_MAX];
160 brix_status st;
161 du_acc a = { 0, 0, 0 };
162 build_path(cwd, ".", path, sizeof(path));
163 brix_status_clear(&st);
164 if (walk_dir(c, path, 0, du_visit, &a, &st) != 0) {
165 fprintf(stderr, "xrdfs: du %s: %s\n", path, st.msg);
166 if (json) { fputs("\n]\n", stdout); }
167 return brix_shellcode(&st);
168 }
169 if (json && !jfirst) { fputs(",\n", stdout); }
170 du_print(path, &a, human, json);
171 if (json) { jfirst = 0; }
172 }
173 if (json) { fputs("\n]\n", stdout); }
174 return rc;
175}

181find_visit(const char *full, const brix_dirent *e, int depth, void *u)
182{
183 const find_pred *p = (const find_pred *) u;
184 int is_dir = e->have_stat && (e->st.flags & kXR_isDir);
185 (void) depth;
186
187 if (p->type == 1 && is_dir) { return 0; }
188 if (p->type == 2 && !is_dir) { return 0; }
189 if (p->name_glob != NULL && fnmatch(p->name_glob, e->name, 0) != 0) { return 0; }
190 if (p->size_sign != 0 && e->have_stat) {
191 if (p->size_sign < 0 && !(e->st.size < p->size_val)) { return 0; }
192 if (p->size_sign > 0 && !(e->st.size > p->size_val)) { return 0; }
193 }
194 printf("%s\n", full);
195 return 0;
196}

208
209 for (i = 1; i < argc; i++) {

236int
237tree_recurse(brix_conn *c, const char *path, const char *prefix, int depth,
238 tree_opts *o, brix_status *st)
239{
240 brix_dirent *ents = NULL;
241 size_t n = 0, i;
242 int *keep, nk = 0, j;
243
244 if (o->maxdepth >= 0 && depth > o->maxdepth) { return 0; }
245 if (brix_dirlist(c, path, 1, &ents, &n, st) != 0) { return -1; }
246
247 keep = (int *) malloc((n ? n : 1) * sizeof(int));
248 if (keep == NULL) {
249 free(ents);
250 brix_status_set(st, XRDC_EPROTO, 0, "tree: out of memory");
251 return -1;
252 }
253 for (i = 0; i < n; i++) {
254 int is_dir = ents[i].have_stat && (ents[i].st.flags & kXR_isDir);
255 if (is_dot(ents[i].name)) { continue; }
256 if (o->dirs_only && !is_dir) { continue; }
257 keep[nk++] = (int) i;
258 }
259 for (j = 0; j < nk; j++) {
260 brix_dirent *e = &ents[keep[j]];
261 int last = (j == nk - 1);
262 int is_dir = e->have_stat && (e->st.flags & kXR_isDir);
263 char full[XRDC_PATH_MAX];
264 char child_prefix[512];
265
266 printf("%s%s%s\n", prefix, last ? "\\-- " : "|-- ", e->name);
267 if (is_dir) { o->ndirs++; } else { o->nfiles++; }
268 if (is_dir && join_path(path, e->name, full, sizeof(full)) == 0
269 && depth + 1 < XRDFS_MAXDEPTH) {
270 snprintf(child_prefix, sizeof(child_prefix), "%s%s",
271 prefix, last ? " " : "| ");
272 if (tree_recurse(c, full, child_prefix, depth + 1, o, st) != 0) {
273 free(keep);
274 free(ents);
275 return -1;
276 }
277 }
278 }
279 free(keep);
280 free(ents);
281 return 0;
282}
