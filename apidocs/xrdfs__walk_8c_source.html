10/* Recursively walk the remote tree under `path` (depth-first, pre-order), invoking
11 * `visit` for every entry. Directories are descended up to XRDFS_MAXDEPTH. 0 / -1. */
12int
13walk_dir(brix_conn *c, const char *path, int depth, xrdfs_visit visit, void *u,
14 brix_status *st)
15{
16 brix_dirent *ents = NULL;
17 size_t n = 0, i;
18
19 if (brix_dirlist(c, path, 1 /*want_stat*/, &ents, &n, st) != 0) {
20 return -1;
21 }
22 for (i = 0; i < n; i++) {
23 char full[XRDC_PATH_MAX];
24 if (is_dot(ents[i].name) || join_path(path, ents[i].name, full, sizeof(full)) != 0) {
25 continue;
26 }
27 if (visit(full, &ents[i], depth, u) != 0) {
28 free(ents);
29 return 1;
30 }
31 if (ents[i].have_stat && (ents[i].st.flags & kXR_isDir)
32 && depth + 1 < XRDFS_MAXDEPTH) {
33 int rc = walk_dir(c, full, depth + 1, visit, u, st);
34 if (rc != 0) {
35 free(ents);
36 return rc;
37 }
38 }
39 }
40 free(ents);
41 return 0;
42}

50chmod_visit(const char *full, const brix_dirent *e, int depth, void *u)

64/* Recursively chmod every entry under `path` (the top path is chmod'd by the caller).
65 * *failures accumulates per-entry errors; returns 0 / -1 (walk-level error, st set). */
66int
67chmod_recursive(brix_conn *c, const char *path, int mode, int *failures,
68 brix_status *st)
69{
70 chmod_walk w;
71 int rc;
72 w.c = c; w.mode = mode; w.failures = 0;
73 rc = walk_dir(c, path, 0, chmod_visit, &w, st);
74 *failures = w.failures;
75 return (rc < 0) ? -1 : 0;
76}

81/* WHAT: print one du result in human-readable or JSON format.
82 * WHY: deduplicates the output logic shared by the multi-path and
83 * default-path branches of do_du.
84 * HOW: JSON emits a single-line object WITHOUT a trailing newline; the caller
85 * (do_du) owns array framing and item separators so the combined output
86 * is always a top-level JSON array (single path → array of 1). Human
87 * mode delegates to fmt_size and includes its own newline. */
88static void
89du_print(const char *path, const du_acc *a, int human, int json)
90{
91 if (json) {
92 fputc('{', stdout);
93 brix_json_kv_str(stdout, "path", path, 1);
94 brix_json_kv_ll(stdout, "bytes", (long long) a->bytes, 1);
95 brix_json_kv_ll(stdout, "files", (long long) a->files, 1);
96 brix_json_kv_ll(stdout, "dirs", (long long) a->dirs, 0);
97 fputc('}', stdout); /* caller handles array separators + trailing newline */
98 } else {
99 char sz[32];
100 fmt_size(a->bytes, sz, sizeof(sz), human);
101 printf("%-10s %s (%ld files, %ld dirs)\n", sz, path, a->files, a->dirs);
102 }
103}

122/* WHAT: scan do_du's argv for the -h/--human and -j/--json flags.
123 * WHY: hoisting the option scan out of do_du keeps the command handler under
124 * the complexity cap; the scan is a pure classification with no I/O.

139/* WHAT: run du over one already-built path — walk, accumulate, print.
140 * WHY: do_du's positional-arg loop and its default-"." fallback share
141 * identical walk + error-report + JSON-separator + print logic; hoisting
142 * it removes the duplication and keeps do_du below the complexity cap.
143 * HOW: walk_dir accumulates into a fresh du_acc. On a walk-level error it
144 * reports to stderr and returns the nonzero shell code WITHOUT printing
145 * an item (the caller decides whether to continue or abort). On success
146 * it emits the JSON item separator when this is not the first item
147 * (*jfirst == 0), prints the object/line via du_print, clears *jfirst,
148 * and returns 0. *jfirst is only touched in JSON mode. */
149static int
150du_one_path(brix_conn *c, const char *path, int human, int json, int *jfirst)
151{
152 brix_status st;
153 du_acc a = { 0, 0, 0 };
154
155 brix_status_clear(&st);
156 if (walk_dir(c, path, 0, du_visit, &a, &st) != 0) {
157 fprintf(stderr, "xrdfs: du %s: %s\n", path, st.msg);
158 return brix_shellcode(&st);
159 }
160 if (json && !*jfirst) { fputs(",\n", stdout); }
161 du_print(path, &a, human, json);
162 if (json) { *jfirst = 0; }
163 return 0;
164}

167/* WHAT: du [-h] [-j] <path>... — recursive total size + file/dir counts.
168 * WHY: -j enables JSON output for scripting; -h humanizes byte counts.
169 * JSON output is always a top-level array so parsers can handle both
170 * single-path (array of 1) and multi-path invocations uniformly.
171 * HOW: flags are scanned first; if json, print "[" before any items.
172 * Each positional arg drives a du_one_path pass; du_print emits the
173 * object body (no trailing newline in JSON mode); do_du emits the
174 * comma-newline separator between items and the closing "]\n" at the
175 * end. A walk error on a positional path records the shell code (last
176 * error wins) and continues; a walk error on the default "." path
177 * closes the array and returns immediately, matching the original. */
178int
179do_du(brix_conn *c, const char *cwd, int argc, char **argv)
180{
181 int human = 0, json = 0, i, rc = 0, any = 0;
182 int jfirst = 1; /* tracks first JSON item for array framing */
183
184 du_parse_flags(argc, argv, &human, &json);
185 if (json) { fputs("[\n", stdout); }
186 for (i = 1; i < argc; i++) {
187 char path[XRDC_PATH_MAX];
188 int r;
189 if (argv[i][0] == '-') { continue; }
190 any = 1;
191 build_path(cwd, argv[i], path, sizeof(path));
192 r = du_one_path(c, path, human, json, &jfirst);
193 if (r != 0) { rc = r; }
194 }
195 if (!any) {
196 char path[XRDC_PATH_MAX];
197 int r;
198 build_path(cwd, ".", path, sizeof(path));
199 r = du_one_path(c, path, human, json, &jfirst);
200 if (r != 0) {
201 if (json) { fputs("\n]\n", stdout); }
202 return r;
203 }
204 }
205 if (json) { fputs("\n]\n", stdout); }
206 return rc;
207}

212int
213find_visit(const char *full, const brix_dirent *e, int depth, void *u)
214{
215 const find_pred *p = (const find_pred *) u;
216 int is_dir = e->have_stat && (e->st.flags & kXR_isDir);
217 (void) depth;
218
219 if (p->type == 1 && is_dir) { return 0; }
220 if (p->type == 2 && !is_dir) { return 0; }
221 if (p->name_glob != NULL && fnmatch(p->name_glob, e->name, 0) != 0) { return 0; }
222 if (p->size_sign != 0 && e->have_stat) {
223 if (p->size_sign < 0 && !(e->st.size < p->size_val)) { return 0; }
224 if (p->size_sign > 0 && !(e->st.size > p->size_val)) { return 0; }
225 }
226 printf("%s\n", full);
227 return 0;
228}

231/* WHAT: parse a find -size argument (+N / -N / N) into sign + magnitude.
232 * WHY: isolating the sign/offset arithmetic keeps find_parse_args below the
233 * complexity cap; the +/- handling is otherwise a dense knot of nested
234 * ternaries.
235 * HOW: size_sign is +1 for "+N" (strictly greater), -1 for "-N" (strictly
236 * less), and +1 for a bare "N" (matching the original default). The
237 * magnitude is the decimal value after any leading sign character. */
238static void
239find_parse_size(const char *s, find_pred *p)
240{
241 p->size_sign = (s[0] == '+') ? 1 : (s[0] == '-') ? -1 : 1;
242 p->size_val = (int64_t) strtoll((s[0] == '+' || s[0] == '-') ? s + 1 : s,
243 NULL, 10);
244}

247/* WHAT: parse find's argv into a find_pred and a starting path token.
248 * WHY: hoisting the option scan out of do_find keeps the command handler
249 * under the complexity cap; the scan is pure classification, no I/O.
250 * HOW: recognizes -name GLOB, -type f|d, and -size +N|-N (each requiring a
251 * following argument, consumed via ++i); -type maps 'd'->2, 'f'->1,
252 * anything else ->0; the first non-flag token becomes *start (a later
253 * positional overrides an earlier one, as in the original). */
254static void
255find_parse_args(int argc, char **argv, find_pred *p, const char **start)
256{
257 int i;
258 for (i = 1; i < argc; i++) {
259 if (strcmp(argv[i], "-name") == 0 && i + 1 < argc) {
260 p->name_glob = argv[++i];
261 } else if (strcmp(argv[i], "-type") == 0 && i + 1 < argc) {
262 const char *t = argv[++i];
263 p->type = (t[0] == 'd') ? 2 : (t[0] == 'f') ? 1 : 0;
264 } else if (strcmp(argv[i], "-size") == 0 && i + 1 < argc) {
265 find_parse_size(argv[++i], p);
266 } else if (argv[i][0] != '-') {
267 *start = argv[i];
268 }
269 }
270}

295/* WHAT: select, in directory order, the dirlist entries tree will render.
296 * WHY: separating the filter pass keeps tree_recurse under the complexity
297 * cap and isolates the dot / dirs-only skip rules from the render loop.
298 * HOW: writes the indices of kept entries into keep[] (which the caller sizes
299 * to n) in the original listing order, skipping "." / ".." and — when
300 * o->dirs_only is set — every non-directory. Returns the number of
301 * indices written. Pure: touches only keep[] and its return value. */
302static int
303tree_collect(const brix_dirent *ents, size_t n, const tree_opts *o, int *keep)
304{
305 size_t i;
306 int nk = 0;
307
308 for (i = 0; i < n; i++) {
309 int is_dir = ents[i].have_stat && (ents[i].st.flags & kXR_isDir);
310 if (is_dot(ents[i].name)) { continue; }
311 if (o->dirs_only && !is_dir) { continue; }
312 keep[nk++] = (int) i;
313 }
314 return nk;
315}

318/* WHAT: print one tree entry and, when it is a descendable directory, recurse.
319 * WHY: pulling the render-loop body out of tree_recurse keeps the caller
320 * below the complexity cap while preserving the exact glyphs, counters,
321 * depth limit, and error propagation.
322 * HOW: prints the branch glyph (the corner "\\-- " for the last sibling,
323 * the tee "|-- " otherwise) then the entry name; bumps o->ndirs or
324 * o->nfiles. For a directory whose path joins cleanly and whose child
325 * depth stays under XRDFS_MAXDEPTH, it builds the continuation prefix
326 * (blanks after the last sibling, a pipe otherwise) and recurses.
327 * Returns 0 on success or -1 on a walk-level error with st already set
328 * by the recursive call. */
329static int
330tree_emit(brix_conn *c, const char *path, const char *prefix,
331 brix_dirent *e, int last, int depth, tree_opts *o, brix_status *st)
332{
333 int is_dir = e->have_stat && (e->st.flags & kXR_isDir);
334 char full[XRDC_PATH_MAX];
335 char child_prefix[512];
336
337 printf("%s%s%s\n", prefix, last ? "\\-- " : "|-- ", e->name);
338 if (is_dir) { o->ndirs++; } else { o->nfiles++; }
339 if (is_dir && join_path(path, e->name, full, sizeof(full)) == 0
340 && depth + 1 < XRDFS_MAXDEPTH) {
341 snprintf(child_prefix, sizeof(child_prefix), "%s%s",
342 prefix, last ? " " : "| ");
343 if (tree_recurse(c, full, child_prefix, depth + 1, o, st) != 0) {
344 return -1;
345 }
346 }
347 return 0;
348}

351int
352tree_recurse(brix_conn *c, const char *path, const char *prefix, int depth,
353 tree_opts *o, brix_status *st)
354{
355 brix_dirent *ents = NULL;
356 size_t n = 0;
357 int *keep, nk, j;
358
359 if (o->maxdepth >= 0 && depth > o->maxdepth) { return 0; }
360 if (brix_dirlist(c, path, 1, &ents, &n, st) != 0) { return -1; }
361
362 keep = (int *) malloc((n ? n : 1) * sizeof(int));
363 if (keep == NULL) {
364 free(ents);
365 brix_status_set(st, XRDC_EPROTO, 0, "tree: out of memory");
366 return -1;
367 }
368 nk = tree_collect(ents, n, o, keep);
369 for (j = 0; j < nk; j++) {
370 if (tree_emit(c, path, prefix, &ents[keep[j]], j == nk - 1,
371 depth, o, st) != 0) {
372 free(keep);
373 free(ents);
374 return -1;
375 }
376 }
377 free(keep);
378 free(ents);
379 return 0;
380}
