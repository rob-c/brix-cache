5/*
6 * WHAT: Upstream connection lifecycle helpers — write buffer flush, error abort with reconnect budget, and full resource cleanup.
7 * WHY: The transparent XRootD proxy must manage upstream TCP connections carefully: flushing buffered requests atomically to the socket;
8 * handling errors gracefully (reconnect if idle + no open files + budget available); releasing all resources on session teardown including
9 * file handle audit, splice pipe closure, connection pool return. INVARIANT: reconnect attempts tracked in proxy->reconnect_left per-connection;
10 * only triggers when XRD_PX_IDLE state and all upstream handles are closed (not 255=pending).
11 * HOW: flush() loops until wbuf_pos reaches wbuf_len; returns NGX_AGAIN/NGX_ERROR for partial send. abort() logs error, checks reconnect conditions,
12 * attempts brix_proxy_connect() if eligible, falls through to hard abort on failure or ineligible state. cleanup() audits abandoned handles via proxy_write_audit,
13 * frees resp_body/saved_req/wait_retry_req, deletes timers, closes splice pipe and upstream connection (or returns to pool if idle).
14 */

16/* brix_proxy_flush — drain all buffered request data to the upstream TCP socket
17 * via the uconn->send loop (the proxy buffers client requests before writing). */
18
19ngx_int_t
20brix_proxy_flush(brix_proxy_ctx_t *proxy)
21{
22 ngx_connection_t *uconn = proxy->conn;
23 ssize_t n;
24
25 while (proxy->wbuf_pos < proxy->wbuf_len) {
26 n = uconn->send(uconn,
27 proxy->wbuf + proxy->wbuf_pos,
28 proxy->wbuf_len - proxy->wbuf_pos);
29 if (n == NGX_AGAIN) {
30 return NGX_AGAIN;
31 }
32 if (n < 0) {
33 return NGX_ERROR;
34 }
35 proxy->wbuf_pos += (size_t) n;
36 }
37
38 return NGX_OK;
39}

41/* brix_proxy_abort — handle an upstream error: log, mark the upstream failed
42 * (brix_proxy_up_mark_failed), and reconnect only when idle (no active transfers,
43 * all handles closed) with budget left, so recovery never interrupts a transfer. */
44
45void
46brix_proxy_abort(brix_proxy_ctx_t *proxy, const char *reason)
47{
48 brix_ctx_t *ctx = proxy->client_ctx;
49 ngx_connection_t *c = proxy->client_conn;
50 int i;
51
52 if (ctx == NULL || ctx->destroyed) {
53 brix_proxy_cleanup(proxy);
54 return;
55 }
56
57 /*
58 * Reconnect if the upstream dropped while idle and no files are open.
59 * The reconnect budget is per-connection and resets when a new upstream
60 * bootstrap completes successfully.
61 */
62 if (proxy->state == XRD_PX_IDLE
63 && proxy->conf != NULL
64 && proxy->reconnect_left > 0)
65 {
66 /* Only reconnect when all upstream file handles are already closed */
67 int has_open = 0;
68 for (i = 0; i < BRIX_MAX_FILES; i++) {
69 if (proxy->fh_map[i].upstream_fh != BRIX_PROXY_FH_FREE
70 && proxy->fh_map[i].upstream_fh != 255)
71 {
72 has_open = 1;
73 break;
74 }
75 }
76
77 if (!has_open) {
78 proxy->reconnect_left--;
79 BRIX_PROXY_METRIC_INC(ctx, reconnects_total);
80 BRIX_PROXY_UP_INC(proxy, reconnects_total);
81 ngx_log_error(NGX_LOG_WARN, c->log, 0,
82 "xrootd proxy: upstream dropped while idle, "
83 "reconnecting (%d attempt(s) left): %s",
84 proxy->reconnect_left, reason);
85
86 /* Close the stale upstream connection but keep the proxy ctx */
87 if (proxy->conn != NULL) {
88 ngx_close_connection(proxy->conn);
89 proxy->conn = NULL;
90 }
91 if (proxy->resp_body != NULL) {
92 ngx_free(proxy->resp_body);
93 proxy->resp_body = NULL;
94 }
95 if (proxy->saved_req != NULL) {
96 ngx_free(proxy->saved_req);
97 proxy->saved_req = NULL;
98 }
99
100 /* Reset upstream state for a fresh bootstrap */
101 proxy->state = XRD_PX_CONNECTING;
102 proxy->bs_phase = XRD_PX_BS_HANDSHAKE;
103 proxy->rhdr_pos = 0;
104 proxy->resp_dlen = 0;
105 proxy->resp_body_pos = 0;
106 proxy->fwd_local_fh = -1;
107 proxy->fwd_streaming = 0;
108 proxy->fwd_is_lazy_open = 0;
109 proxy->lazy_open_pending_count = 0;
110
111 if (brix_proxy_connect(proxy, c, proxy->conf) == NGX_OK) {
112 /* Client stays in XRD_ST_REQ_HEADER — reconnect in progress */
113 return;
114 }
115
116 /* connect() failed — fall through to hard abort */
117 ngx_log_error(NGX_LOG_ERR, c->log, 0,
118 "xrootd proxy: reconnect failed, aborting session");
119 }
120 }
121
122 ngx_log_error(NGX_LOG_ERR, c->log, 0,
123 "xrootd proxy: upstream error: %s", reason);
124
125 brix_proxy_up_mark_failed(proxy);
126
127 /* Count this failure against the per-connection budget so a permanently
128 * failing upstream cannot drive an unbounded reconnect loop (the dispatch
129 * path stops spawning new proxies once BRIX_PROXY_MAX_CONN_FAILS is hit).
130 * A successful bootstrap resets the counter. */
131 ctx->proxy_fail_count++;
132
133 ctx->state = XRD_ST_REQ_HEADER;
134 brix_proxy_cleanup(proxy);
135 ctx->proxy = NULL;
136
137 brix_send_error(ctx, c, kXR_IOError, reason);
138 brix_schedule_read_resume(c);
139}

141/* brix_proxy_cleanup_audit_handles — account for upstream file handles still open
142 * at teardown by auditing each one and marking its slot free.
143 *
144 * WHAT: For every upstream fh_map slot that is neither FREE nor 255 (pending open),
145 * runs proxy_write_audit(), bumps the abandoned-handle metrics, and resets the
146 * slot to BRIX_PROXY_FH_FREE. No-op when the client context is already gone.
147 *
148 * WHY: A session may be torn down (error/abort) while the client left files open on
149 * the upstream. Those handles must be audited (so any pending write state is
150 * flushed/observed) and counted so the abandoned-handle metric stays accurate;
151 * leaving slots non-FREE would also confuse the idle/pool-return decision later.
152 *
153 * HOW: 1. Bail if client_ctx is NULL (nothing to audit against).
154 * 2. Scan all BRIX_MAX_FILES slots.
155 * 3. Skip FREE and 255 (pending) slots; for the rest audit, increment both the
156 * per-ctx and per-upstream abandoned counters, then free the slot.
157 */
158static void
159brix_proxy_cleanup_audit_handles(brix_proxy_ctx_t *proxy)
160{
161 int i;
162
163 if (proxy->client_ctx == NULL) {
164 return;
165 }
166
167 for (i = 0; i < BRIX_MAX_FILES; i++) {
168 if (proxy->fh_map[i].upstream_fh != BRIX_PROXY_FH_FREE
169 && proxy->fh_map[i].upstream_fh != 255 /* pending open */)
170 {
171 proxy_write_audit(proxy, i);
172 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
173 abandoned_handles_total);
174 BRIX_PROXY_UP_INC(proxy, abandoned_handles_total);
175 proxy->fh_map[i].upstream_fh = BRIX_PROXY_FH_FREE;
176 }
177 }
178}

180/* brix_proxy_cleanup_free_buffers — release the heap-owned request/response buffers.
181 *
182 * WHAT: Frees resp_body and saved_req (each NULL-safe, then nulled) and releases the
183 * in-flight write buffer via brix_proxy_wbuf_release().
184 *
185 * WHY: These buffers are heap-allocated per upstream exchange and must be freed on
186 * teardown to avoid leaks; the wbuf is owned-gated so a pool-allocated bootstrap
187 * frame is merely detached rather than double-freed.
188 *
189 * HOW: 1. Free resp_body if set, null it.
190 * 2. Release the pending write buffer (Phase 39 PXY-3 — NULL-safe, owned-gated).
191 * 3. Free saved_req if set, null it.
192 */
193static void
194brix_proxy_cleanup_free_buffers(brix_proxy_ctx_t *proxy)
195{
196 if (proxy->resp_body != NULL) {
197 ngx_free(proxy->resp_body);
198 proxy->resp_body = NULL;
199 }
200
201 /* Phase 39 (PXY-3): free a heap-owned in-flight write buffer that was still
202 * pending at teardown/abort (e.g. an aborted forwarded request). NULL-safe
203 * and owned-gated, so a pool-allocated bootstrap frame is merely detached. */
204 brix_proxy_wbuf_release(proxy);
205
206 if (proxy->saved_req != NULL) {
207 ngx_free(proxy->saved_req);
208 proxy->saved_req = NULL;

276/* brix_proxy_cleanup_conn — pool-return or close the upstream connection safely.
277 *
278 * WHAT: If an upstream connection exists, returns it to the pool when idle (no open
279 * handles, XRD_PX_IDLE, pooling allowed); otherwise (or if pool-return did not
280 * consume it) nulls conn->data and closes the connection, clearing proxy->conn.
281 *
282 * WHY: An idle, cleanly-bootstrapped connection is reusable and should be recycled;
283 * any other connection must be closed. Nulling conn->data before close makes any
284 * pending upstream-fd event handler observe NULL and bail, preventing a
285 * use-after-free once the client pool is freed.
286 *
287 * HOW: 1. Return immediately if proxy->conn is NULL.
288 * 2. Scan slots to decide has_open (any slot not FREE).
289 * 3. If !has_open && state==XRD_PX_IDLE && !no_pool, brix_proxy_pool_put().
290 * 4. If proxy->conn is still set, null its data pointer, close it, and null conn.
291 */
292static void
293brix_proxy_cleanup_conn(brix_proxy_ctx_t *proxy)
294{
295 int has_open = 0;
296 int i;
297
298 if (proxy->conn == NULL) {
299 return;
300 }
301
302 /*
303 * If the connection is idle (no open files) and the session was
304 * successfully bootstrapped, return it to the pool.
305 */
306 for (i = 0; i < BRIX_MAX_FILES; i++) {
307 if (proxy->fh_map[i].upstream_fh != BRIX_PROXY_FH_FREE) {
308 has_open = 1;
309 break;
310 }
311 }
312
313 if (!has_open && proxy->state == XRD_PX_IDLE && !proxy->no_pool) {
314 brix_proxy_pool_put(proxy);
315 }
316
317 if (proxy->conn != NULL) {
318 /* Null the data pointer before closing so any pending event
319 * handlers on the upstream fd see NULL and bail out safely,
320 * preventing use-after-free when the client pool is freed. */
321 proxy->conn->data = NULL;
322 ngx_close_connection(proxy->conn);
323 proxy->conn = NULL;
324 }
325}
