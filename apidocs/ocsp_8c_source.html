1/*
2 * OCSP client — public revocation-check and TLS-staple entry points.
3 *
4 * WHAT: Provides the two public functions:
5 * brix_ocsp_check_cert() — query OCSP responder for a client certificate.
6 * brix_ocsp_staple_fetch() — fetch OCSP staple for the server certificate.
7 * Each extracts the responder URLs from the certificate's Authority Information
8 * Access extension, builds the OCSP cert id, drives the per-URL query loop, and
9 * interprets / caches the result.
10 *
11 * WHY: X.509 chain verification alone does not catch revoked certificates.
12 * OCSP gives real-time revocation status without the overhead of fetching full
13 * CRLs, which can be hundreds of megabytes for large CAs. This file owns the
14 * public policy: which URLs to try, how soft_fail maps network errors / UNKNOWN,
15 * that REVOKED is never overridden, and how a GOOD staple is DER-cached on xcf.
16 *
17 * HOW: The URL/BIO transport (parse + connect + handshake) lives in
18 * ocsp_transport.c and the OCSP request/response crypto (nonce, signature
19 * verify, status mapping) in ocsp_request.c — both split out under the phase-79
20 * file-size guard and shared through ocsp_internal.h. This file calls
21 * do_ocsp_request() + check_ocsp_response() in a loop over the AIA URLs and
22 * applies the soft_fail / staple-caching policy on top. The synchronous
23 * blocking model is acceptable because the auth path already performs blocking
24 * crypto work.
25 */

59/*
60 * brix_ocsp_check_cert — query the OCSP responder for leaf's revocation.
61 *
62 * Extracts the OCSP URL from the leaf's AIA extension, builds and sends an
63 * OCSP request, and interprets the response.
64 *
65 * Returns 0 if the certificate is GOOD (or soft_fail allows the outcome).
66 * Returns -1 if the certificate is REVOKED or the check definitively fails.
67 */
68/*
69 * ocsp_check_urls — query each responder URL until a definitive answer.
70 *
71 * WHAT: Iterates the AIA responder URLs, sending an OCSP request for `id` and
72 * interpreting the reply, and returns the check result: 0 (GOOD), -1 (REVOKED
73 * or hard fail), or the soft_fail default for network errors / UNKNOWN.
74 * WHY: Factors the multi-URL loop out of brix_ocsp_check_cert() so the public
75 * function is validate-inputs → build-id → loop → free, keeping each part
76 * small; behaviour (GOOD/REVOKED break semantics, soft_fail fallback) is
77 * preserved exactly.
78 * HOW: GOOD returns 0 immediately; REVOKED returns -1 immediately (never
79 * overridden); UNKNOWN and network errors leave the running soft_fail default,
80 * which is returned once the URLs are exhausted. The trust store is NULL by
81 * design (chain already verified before this call — see the note below).
82 */
83static int
84ocsp_check_urls(ngx_log_t *log, STACK_OF(OPENSSL_STRING) *ocsp_urls,
85 const ocsp_query_t *q, int soft_fail)
86{
87 int result = soft_fail ? 0 : -1; /* default on error / UNKNOWN */
88 int n_urls = sk_OPENSSL_STRING_num(ocsp_urls);
89 int i;
90
91 for (i = 0; i < n_urls; i++) {
92 const char *url = sk_OPENSSL_STRING_value(ocsp_urls, i);
93 OCSP_RESPONSE *resp;
94 int status;
95
96 ngx_log_error(NGX_LOG_DEBUG, log, 0,
97 "brix_ocsp: querying responder \"%s\"", url);
98
99 resp = do_ocsp_request(log, url, q->leaf, q->issuer, q->id);
100 if (resp == NULL) {
101 /* Network error — try next URL */
102 continue;
103 }
104
105 /* NOTE: We pass NULL for the trust store so OpenSSL uses
106 * the default verification. A production deployment should
107 * pass xcf->gsi_store here. For the auth path this is
108 * acceptable because the chain is already verified before
109 * brix_ocsp_check_cert() is called. */
110 status = check_ocsp_response(log, resp, NULL, q->id, NULL);
111 OCSP_RESPONSE_free(resp);
112
113 if (status == 0) {
114 ngx_log_error(NGX_LOG_DEBUG, log, 0,
115 "brix_ocsp: certificate is GOOD (\"%s\")", url);
116 return 0; /* GOOD */
117 }
118 if (status == -1) {
119 ngx_log_error(NGX_LOG_WARN, log, 0,
120 "brix_ocsp: certificate is REVOKED (\"%s\")", url);
121 return -1; /* REVOKED — never override */
122 }
123 /* UNKNOWN — apply soft_fail policy and keep trying other URLs */
124 result = soft_fail ? 0 : -1;
125 }
126
127 return result;
128}

130/* HOW: Validates leaf and issuer are non-NULL (returns soft_fail result if either is missing). Extracts OCSP URLs from the certificate's AIA extension via X509_get1_ocsp(). Builds the OCSP certificate ID (SHA-1 hash of issuer fields) using OCSP_cert_to_id(). Delegates the per-URL query loop to ocsp_check_urls() (GOOD→0, REVOKED→-1 never overridden, UNKNOWN/network-error→soft_fail default). Frees ID and URL stack on exit. */
131int
132brix_ocsp_check_cert(ngx_log_t *log, X509 *leaf, X509 *issuer, int soft_fail)
133{
134 STACK_OF(OPENSSL_STRING) *ocsp_urls = NULL;
135 OCSP_CERTID *id = NULL;
136 int result;
137
138 if (leaf == NULL) {
139 return soft_fail ? 0 : -1;
140 }
141
142 if (issuer == NULL) {
143 ngx_log_error(NGX_LOG_WARN, log, 0,
144 "brix_ocsp: no issuer certificate for OCSP check; "
145 "treating as soft_fail=%d", soft_fail);
146 return soft_fail ? 0 : -1;
147 }
148
149 /* Extract OCSP URLs from the Authority Information Access extension */
150 ocsp_urls = X509_get1_ocsp(leaf);
151 if (ocsp_urls == NULL || sk_OPENSSL_STRING_num(ocsp_urls) == 0) {
152 ngx_log_error(NGX_LOG_DEBUG, log, 0,
153 "brix_ocsp: no OCSP URL in certificate AIA extension");
154 if (ocsp_urls) {
155 X509_email_free(ocsp_urls);
156 }
157 return soft_fail ? 0 : -1;
158 }
159
160 /* Build the OCSP certificate ID (SHA-1 hash of issuer fields) */
161 id = OCSP_cert_to_id(NULL, leaf, issuer);
162 if (id == NULL) {
163 ngx_log_error(NGX_LOG_WARN, log, 0,
164 "brix_ocsp: OCSP_cert_to_id() failed");
165 X509_email_free(ocsp_urls);
166 return soft_fail ? 0 : -1;
167 }
168
169 {
170 ocsp_query_t q = { leaf, issuer, id };
171 result = ocsp_check_urls(log, ocsp_urls, &q, soft_fail);
172 }
173
174 OCSP_CERTID_free(id);
175 X509_email_free(ocsp_urls);
176 return result;
177}

179/*
180 * brix_ocsp_staple_fetch — fetch and cache an OCSP staple.
181 *
182 * Queries the OCSP responder for xcf->gsi_cert and stores the raw DER
183 * response bytes in xcf->ocsp.staple_data / ocsp.staple_len.
184 *
185 * Memory is allocated with ngx_alloc() and freed on the next reload.
186 */
187/*
188 * ocsp_find_issuer — locate leaf's issuer certificate in the CA trust store.
189 *
190 * WHAT: Returns the issuer of `leaf` found in `store` (caller owns the ref via
191 * X509_free), or NULL if the store is absent or the issuer is not found.
192 * WHY: The staple's OCSP request needs the issuer; isolating the store-context
193 * dance keeps brix_ocsp_staple_fetch() readable and its cleanup unambiguous.
194 * HOW: X509_STORE_CTX_get1_issuer() bumps the returned cert's refcount, so the
195 * caller must X509_free() it — identical ownership to the original inline code.
196 */
197static X509 *
198ocsp_find_issuer(X509_STORE *store, X509 *leaf)
199{
200 X509 *issuer = NULL;
201 X509_STORE_CTX *ctx;
202
203 if (store == NULL) {
204 return NULL;
205 }
206 ctx = X509_STORE_CTX_new();
207 if (ctx == NULL) {
208 return NULL;
209 }
210 if (X509_STORE_CTX_init(ctx, store, leaf, NULL) == 1) {
211 /* Retrieve the issuer from the store (refcounted) */
212 X509_STORE_CTX_get1_issuer(&issuer, ctx, leaf);
213 }
214 X509_STORE_CTX_free(ctx);
215 return issuer;
216}

218/*
219 * ocsp_store_staple — DER-encode a GOOD response and cache it on xcf.
220 *
221 * WHAT: Encodes `resp` to DER, copies it into nginx-managed memory, replaces
222 * any previously cached staple, and records buf/len on xcf->ocsp. Returns
223 * NGX_OK on success, NGX_ERROR if encoding or allocation fails.
224 * WHY: Concentrates the encode → copy → swap sequence (and its OPENSSL_free /
225 * ngx_free ordering) so the fetch loop stays a thin driver.
226 * HOW: i2d_OCSP_RESPONSE() allocates `der` via OPENSSL_free-owned memory; it is
227 * always released here. The old staple is freed only after the new buffer is
228 * in hand, matching the original reload path exactly.
229 */
230static ngx_int_t
231ocsp_store_staple(ngx_log_t *log, ngx_stream_brix_srv_conf_t *xcf,
232 OCSP_RESPONSE *resp, const char *url)
233{
234 unsigned char *der = NULL;
235 int der_len;
236 u_char *buf;
237
238 /* Encode response to DER for the TLS extension */
239 der_len = i2d_OCSP_RESPONSE(resp, &der);
240 if (der_len <= 0 || der == NULL) {
241 ngx_log_error(NGX_LOG_WARN, log, 0,
242 "brix_ocsp: DER encoding of staple failed");
243 return NGX_ERROR;
244 }
245
246 /* Copy into nginx-managed memory so the TLS callback can use it */
247 buf = ngx_alloc((size_t)der_len, log);
248 if (buf == NULL) {
249 OPENSSL_free(der);
250 return NGX_ERROR;
251 }
252
253 ngx_memcpy(buf, der, (size_t)der_len);
254 OPENSSL_free(der);
255
256 /* Free old staple if present (reload path) */
257 if (xcf->ocsp.staple_data != NULL) {
258 ngx_free(xcf->ocsp.staple_data);
259 }
260 xcf->ocsp.staple_data = buf;
261 xcf->ocsp.staple_len = (size_t)der_len;
262
263 ngx_log_error(NGX_LOG_INFO, log, 0,
264 "brix_ocsp: staple cached (%d bytes) from \"%s\"",
265 der_len, url);
266 return NGX_OK;
267}

269/*
270 * ocsp_fetch_staple_urls — query responders until one yields a cacheable staple.
271 *
272 * WHAT: Iterates the AIA URLs; for each, fetches the OCSP response, requires a
273 * GOOD verified status (against xcf->gsi_store), and on success caches it via
274 * ocsp_store_staple(). Returns NGX_OK once a staple is cached, else NGX_ERROR.
275 * WHY: Splits the per-URL loop out of brix_ocsp_staple_fetch() so the public
276 * function is validate → find-issuer → build-id → loop → free.
277 * HOW: Each response is freed on every path (skip / cache). Only GOOD (status
278 * 0) responses are cached, preserving the original "not GOOD → skip" policy.
279 */
280static ngx_int_t
281ocsp_fetch_staple_urls(ngx_log_t *log, ngx_stream_brix_srv_conf_t *xcf,
282 STACK_OF(OPENSSL_STRING) *ocsp_urls, const ocsp_query_t *q)
283{
284 int n_urls = sk_OPENSSL_STRING_num(ocsp_urls);
285 int i;
286
287 for (i = 0; i < n_urls; i++) {
288 const char *url = sk_OPENSSL_STRING_value(ocsp_urls, i);
289 OCSP_RESPONSE *resp;
290 int status;
291
292 ngx_log_error(NGX_LOG_DEBUG, log, 0,
293 "brix_ocsp: fetching staple from \"%s\"", url);
294
295 resp = do_ocsp_request(log, url, q->leaf, q->issuer, q->id);
296 if (resp == NULL) {
297 continue;
298 }
299
300 /* Verify the response is GOOD before caching */
301 status = check_ocsp_response(log, resp, xcf->gsi_store, q->id, NULL);
302 if (status != 0) {
303 ngx_log_error(NGX_LOG_WARN, log, 0,
304 "brix_ocsp: staple response not GOOD (status=%d) "
305 "from \"%s\" — skipping", status, url);
306 OCSP_RESPONSE_free(resp);
307 continue;
308 }
309
310 if (ocsp_store_staple(log, xcf, resp, url) != NGX_OK) {
311 OCSP_RESPONSE_free(resp);
312 continue;
313 }
314 OCSP_RESPONSE_free(resp);
315 return NGX_OK;
316 }
317
318 return NGX_ERROR;
319}

321/* HOW: Validates server certificate is non-NULL. Extracts OCSP URLs from the AIA extension. Locates the issuer certificate via ocsp_find_issuer() (X509_STORE_CTX against gsi_store). Builds the OCSP certificate ID via OCSP_cert_to_id(). Delegates the per-URL fetch/verify/cache loop to ocsp_fetch_staple_urls() (only GOOD responses cached, DER-encoded into nginx memory, old staple freed on the reload path). Returns NGX_OK once a staple is cached, NGX_ERROR otherwise. Frees ID, issuer, and URL stack on exit. */
322ngx_int_t
323brix_ocsp_staple_fetch(ngx_log_t *log, ngx_stream_brix_srv_conf_t *xcf)
324{
325 X509 *leaf = xcf->gsi_cert;
326 X509 *issuer = NULL;
327 STACK_OF(OPENSSL_STRING) *ocsp_urls = NULL;
328 OCSP_CERTID *id = NULL;
329 ngx_int_t rc;
330
331 if (leaf == NULL) {
332 ngx_log_error(NGX_LOG_WARN, log, 0,
333 "brix_ocsp: no server certificate; cannot fetch staple");
334 return NGX_ERROR;
335 }
336
337 ocsp_urls = X509_get1_ocsp(leaf);
338 if (ocsp_urls == NULL || sk_OPENSSL_STRING_num(ocsp_urls) == 0) {
339 ngx_log_error(NGX_LOG_DEBUG, log, 0,
340 "brix_ocsp: no OCSP URL in server certificate AIA");
341 if (ocsp_urls) {
342 X509_email_free(ocsp_urls);
343 }
344 return NGX_ERROR;
345 }
346
347 /*
348 * For the staple we need the issuer certificate. Attempt to find it
349 * in the configured CA trust store (gsi_store). If not available,
350 * we cannot produce a staple — log and return.
351 */
352 issuer = ocsp_find_issuer(xcf->gsi_store, leaf);
353 if (issuer == NULL) {
354 ngx_log_error(NGX_LOG_WARN, log, 0,
355 "brix_ocsp: cannot locate issuer certificate for staple");
356 X509_email_free(ocsp_urls);
357 return NGX_ERROR;
358 }
359
360 id = OCSP_cert_to_id(NULL, leaf, issuer);
361 if (id == NULL) {
362 X509_free(issuer);
363 X509_email_free(ocsp_urls);
364 return NGX_ERROR;
365 }
366
367 {
368 ocsp_query_t q = { leaf, issuer, id };
369 rc = ocsp_fetch_staple_urls(log, xcf, ocsp_urls, &q);
370 }
371
372 OCSP_CERTID_free(id);
373 X509_free(issuer);
374 X509_email_free(ocsp_urls);
375 return rc;
376}
