1/*
2 * cms/server_recv_lifecycle.c — CMS data-server connection teardown + audit.
3 *
4 * WHAT: Owns the lifecycle side of an accepted CMS data-server connection: the
5 * session-end-hint bookkeeping, the registration/auth audit-log helpers, the
6 * canonical connection teardown (brix_cms_srv_close), and the shared fatal
7 * epilogue (cms_srv_fail_close) every error path funnels through.
8 *
9 * WHY: Split out of the former monolithic server_recv.c (Phase-79 file-size
10 * split). Grouping "how a connection ends and what we record about it" in one
11 * file keeps the parse, dispatch, and event-loop concerns each single-purpose.
12 * The functions called from the frame handlers and the read loop are declared
13 * in server_recv_internal.h and are therefore non-static here.
14 *
15 * HOW: brix_cms_srv_close drops timers, releases admission-cap slots, blacklists
16 * a departing logged-in node, ends the audit session, and closes the socket;
17 * cms_srv_fail_close records an end reason first. The audit helpers translate
18 * connection state into brix_sess_* calls.
19 */

37static brix_sess_end_t
38cms_srv_end_reason(brix_cms_srv_ctx_t *ctx, ngx_connection_t *c)
39{
40 if (ngx_exiting || ngx_terminate) {
41 return BRIX_SESS_END_SHUTDOWN;
42 }
43
44 if (ctx != NULL && ctx->sess_end_hint_set) {
45 return ctx->sess_end_hint;
46 }
47
48 if (c != NULL) {
49 if ((c->read != NULL && c->read->timedout)
50 || (c->write != NULL && c->write->timedout))
51 {
52 return BRIX_SESS_END_TIMEOUT;
53 }
54 if (c->error) {
55 return BRIX_SESS_END_ERROR;
56 }
57 }
58
59 return BRIX_SESS_END_CLIENT;
60}

62static const char *
63cms_srv_target_path(brix_cms_srv_ctx_t *ctx, char *dst, size_t dst_size)
64{
65 u_char *p;
66 u_char *end;
67
68 if (ctx == NULL || dst == NULL || dst_size == 0) {
69 return "-";
70 }
71
72 end = (u_char *) dst + dst_size;
73 p = ngx_snprintf((u_char *) dst, dst_size, "%s:%d",
74 ctx->host, (int) ctx->port);
75 if (p < end) {
76 *p = '\0';
77 } else {
78 dst[dst_size - 1] = '\0';
79 }
80
81 return dst;
82}

94void
95cms_srv_log_registration(brix_cms_srv_ctx_t *ctx)
96{
97 char target[BRIX_SESSLOG_PATH_MAX];
98 const char *path;
99
100 if (ctx == NULL || ctx->sess_attempt_logged) {
101 return;
102 }
103
104 path = cms_srv_target_path(ctx, target, sizeof(target));
105 brix_sess_auth_once(ctx->sess, BRIX_SESS_AM_HOST, ctx->host, "-");
106 brix_sess_attempt(ctx->sess, path, BRIX_SESS_MODE_META);
107 ctx->sess_attempt_logged = 1;
108 brix_sess_result(ctx->sess, 1, path, BRIX_SESS_MODE_META, NULL);
109}

111/* brix_cms_srv_close — tear down a CMS data-server connection: drop the ping
112 * timer, unregister host/port from the server registry if logged_in (so locate
113 * queries stop routing clients to a dead server), NULL ctx->c, and close. */
114
115void
116brix_cms_srv_close(brix_cms_srv_ctx_t *ctx)
117{
118 ngx_connection_t *c;
119
120 c = ctx->c;
121 if (c == NULL) {
122 return;
123 }
124
125 if (ctx->ping_timer.timer_set) {
126 ngx_del_timer(&ctx->ping_timer);
127 }
128
129 /* WS3: cancel the login/idle read deadline before tearing the socket down. */
130 if (c->read->timer_set) {
131 ngx_del_timer(c->read);
132 }
133
134 /* W3: leave the per-worker fan-out set (idempotent via in_node_list). */
135 brix_cms_srv_node_del(ctx);
136
137 /* WS4 + A3: release the admission-cap slots (global + per-IP) exactly once. */
138 if (ctx->counted) {
139 brix_cms_srv_conn_dec();
140 brix_cms_srv_ip_dec(ctx->host);
141 ctx->counted = 0;
142 }
143
144 if (ctx->logged_in) {
145 /* Blacklist for 30 s so in-flight locate responses don't route to a
146 * server that just went away. brix_srv_register() clears the flag
147 * the moment the server successfully reconnects and re-heartbeats. */
148 brix_srv_blacklist(ctx->host, ctx->port,
149 NGX_BRIX_CMS_SRV_DROP_BLACKLIST_MS);
150 BRIX_DIAG(NGX_LOG_NOTICE, c->log, 0,
151 "xrootd[cms]: data server %s:%d disconnected (blacklisted 30s)",
152 "the data server dropped its CMS connection — it crashed, was "
153 "restarted, or lost network to the manager",
154 "if it does not re-register within seconds, check that server's "
155 "health and connectivity; clients are routed away from it "
156 "meanwhile",
157 ctx->host, (int) ctx->port);
158 }
159
160 brix_sess_end(ctx->sess, cms_srv_end_reason(ctx, c));
161 ctx->sess = NULL;
162
163 ctx->c = NULL;
164 ngx_close_connection(c);
165}

167/*
168 * cms_srv_fail_close — record why the session ended, then tear it down.
169 * Shared epilogue for every fatal per-frame/per-read error so each handler
170 * stays single-purpose; after this returns ctx->c is NULL and the caller must
171 * not touch the connection again.
172 */
173void
174cms_srv_fail_close(brix_cms_srv_ctx_t *ctx, brix_sess_end_t why)
175{
176 cms_srv_set_end_hint(ctx, why);
177 brix_cms_srv_close(ctx);
178}
