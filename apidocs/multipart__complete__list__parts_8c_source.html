1/*
2 * multipart_complete_list_parts.c — S3 ListParts handler + part-entry helpers.
3 *
4 * WHAT: This fragment implements the GET /bucket/key?uploadId=<id> → ListParts operation,
5 * which enumerates uploaded parts stored in a staging directory for an active multipart upload.
6 * It contains three components:
7 * - mpu_part_entry_t typedef: holds part_num (int), size (off_t), and mtime (time_t) — the
8 * metadata collected per-part during directory enumeration.
9 * - mpu_part_entry_cmp(): qsort comparator that sorts entries in ascending part-number order,
10 * ensuring the XML response lists parts numerically as required by S3 spec.
11 * - s3_handle_list_parts(): the main handler — validates uploadId, probes staging directory,
12 * enumerates "part.<N>" files (stat size/mtime), sorts by number, applies pagination
13 * (part-number-marker + max-parts), and emits a ListPartsResult XML document with each
14 * part's PartNumber, LastModified (ISO 8601), ETag ("mtime-size"), and Size.
15 * ETags are computed as "\"mtime-size\"" — the staging-file convention for multipart parts.
16 *
17 * WHY: Split from multipart.c to keep ListParts logic self-contained alongside its sibling fragments
18 * (multipart_complete_body.c, multipart_complete_upload_part_copy.c). The typedef + comparator pair
19 * is reused by CompleteMultipartUpload's own part-verification scan, keeping the sort logic shared.
20 */

79/*
80 * lp_parse_query — read ListParts pagination parameters off the request.
81 *
82 * WHAT: Parses the optional "part-number-marker" and "max-parts" query params into
83 * *marker (default 0) and *max_parts (default MPU_MAX_PART_NUMBER), applying the same
84 * bounds the inline handler used (marker in [0, MPU_MAX_PART_NUMBER]; max-parts in
85 * (0, MPU_MAX_PART_NUMBER)). Out-of-range or unparsable values leave the default.
86 * WHY: Isolates pure query parsing from the handler orchestration so the pagination
87 * window computation reads as a flat sequence (functional/modular §8).
88 * HOW: Two independent strtol scans over stack buffers; no I/O, no side effects beyond
89 * writing the two out-params (zero-initialised by the caller-supplied defaults here).
90 */
91static void
92lp_parse_query(ngx_http_request_t *r, int *marker, int *max_parts)
93{
94 char marker_str[16];
95 char max_parts_str[16];
96
97 *marker = 0;
98 marker_str[0] = '\0';
99 if (s3_get_query_param(r, "part-number-marker",
100 marker_str, sizeof(marker_str)))
101 {
102 char *endp;
103 long mn = strtol(marker_str, &endp, 10);
104 if (endp != marker_str && mn >= 0 && mn <= MPU_MAX_PART_NUMBER) {
105 *marker = (int) mn;
106 }
107 }
108
109 *max_parts = MPU_MAX_PART_NUMBER;
110 max_parts_str[0] = '\0';
111 if (s3_get_query_param(r, "max-parts",
112 max_parts_str, sizeof(max_parts_str)))
113 {
114 char *endp;
115 long mp = strtol(max_parts_str, &endp, 10);
116 if (endp != max_parts_str && mp > 0 && mp < MPU_MAX_PART_NUMBER) {
117 *max_parts = (int) mp;
118 }
119 }
120}

122/*
123 * lp_scan_staged_parts — enumerate "part.<N>" staging files into a sorted array.
124 *
125 * WHAT: Opens the multipart staging directory (mpu_dir) via the VFS seam, iterates its
126 * entries, matches "part.<N>" with N in [1, MPU_MAX_PART_NUMBER], probes each for
127 * size/mtime, records up to MPU_MAX_PART_NUMBER entries into parts[], and sorts them by
128 * ascending part number. Returns the count found, or -1 on opendir failure.
129 * WHY: Extracts the directory-scan phase (the bulk of the handler's branching) so the
130 * orchestrator only decides the pagination window and rendering (§8).
131 * HOW: brix_vfs_opendir_quiet + brix_vfs_readdir_kind loop under the caller's vctx
132 * (impersonation-aware); per-entry confined no-follow brix_vfs_probe for stat; qsort at
133 * the end. Side effects (opendir/closedir) stay inside this edge helper. The EACCES /
134 * impersonation confinement rationale is unchanged from the inline version.
135 */
136static int
137lp_scan_staged_parts(ngx_http_request_t *r,
138 brix_vfs_ctx_t *vctx,
139 const char *mpu_dir,
140 ngx_http_s3_loc_conf_t *cf,
141 mpu_part_entry_t *parts)
142{
143 brix_vfs_dir_t *dp;
144 int nparts = 0;
145
146 /* As the mapped user under impersonation (broker fd + fdopendir): the staging
147 * dir is owned 0700 by the mapped user, so a raw worker opendir() fails EACCES
148 * (ListParts 500). Off impersonation this is a plain opendir(). Non-metered
149 * (the ListParts op accounts for the scan). */
150 dp = brix_vfs_opendir_quiet(vctx, NULL);
151 if (dp == NULL) {
152 return -1;
153 }
154
155 for ( ;; ) {
156 ngx_str_t name;
157 const char *dname;
158 char part_path[PATH_MAX];
159 char *endptr;
160 long pn;
161 brix_vfs_ctx_t pctx;
162 brix_vfs_stat_t pst;
163
164 if (nparts >= MPU_MAX_PART_NUMBER) {
165 break;
166 }
167 if (brix_vfs_readdir_kind(dp, &name, NULL) != NGX_OK) {
168 break; /* NGX_DONE (end) or error → stop */
169 }
170 dname = (const char *) name.data;
171
172 /* Match "part.<N>" — must start with "part." */
173 if (ngx_strncmp(dname, "part.", 5) != 0) {
174 continue;
175 }
176
177 /* Parse the part number after "part." */
178 pn = strtol(dname + 5, &endptr, 10);
179 if (*endptr != '\0' || pn < 1 || pn > MPU_MAX_PART_NUMBER) {
180 continue;
181 }
182
183 /* part_path is mpu_dir + "/" + name (readdir result); confinement is
184 * inherited from mpu_dir (see confinement comment above). */
185 if ((size_t) snprintf(part_path, sizeof(part_path), "%s/%s",
186 mpu_dir, dname) >= sizeof(part_path)) {
187 continue;
188 }
189
190 /* Confined no-follow probe for size/mtime (non-metered). */
191 s3_build_vfs_ctx(r, part_path, cf, &pctx);
192 if (brix_vfs_probe(&pctx, 1 /* no-follow */, &pst) != NGX_OK
193 || !pst.is_regular) {
194 continue;
195 }
196
197 parts[nparts].part_num = (int) pn;
198 parts[nparts].size = pst.size;
199 parts[nparts].mtime = pst.mtime;
200 nparts++;
201 }
202 brix_vfs_closedir(dp, r->connection->log);
203
204 /* Sort by part number */
205 if (nparts > 1) {
206 qsort(parts, (size_t) nparts, sizeof(mpu_part_entry_t),
207 mpu_part_entry_cmp);
208 }
209
210 return nparts;
211}

213/*
214 * lp_render_xml — build and send the ListPartsResult XML document.
215 *
216 * WHAT: Allocates a right-sized XML buffer, emits the ListPartsResult header (Bucket/Key/
217 * UploadId/StorageClass/PartNumberMarker/MaxParts/IsTruncated, plus NextPartNumberMarker
218 * when truncated) and one <Part> element per entry in parts[start_idx, end_idx), then
219 * sends it as application/xml with HTTP 200. Returns the send result, or an internal
220 * error status on allocation failure.
221 * WHY: Isolates the response-framing phase so the handler is a flat orchestration; the XML
222 * byte layout is frozen (identical XML_APPEND sequence and capacity formula as inline).
223 * HOW: Same xml_capacity formula, same XML_APPEND macro calls, ETag "\"mtime-size\"" and
224 * ISO-8601 LastModified per entry; buffer copied into an ngx_create_temp_buf and handed
225 * to brix_http_send_xml_buffer. Side effects (alloc, send, metric on failure) at the edge.
226 */
227static ngx_int_t
228lp_render_xml(ngx_http_request_t *r,
229 ngx_http_s3_loc_conf_t *cf,
230 const lp_page_t *page)
231{
232 size_t xml_capacity;
233 u_char *xml;
234 size_t xml_len = 0;
235 ngx_buf_t *b;
236 char iso_buf[32];
237 char etag_buf[48];
238 const char *key_str = page->key_str;
239 const char *upload_id = page->upload_id;
240 const mpu_part_entry_t *parts = page->parts;
241 int start_idx = page->start_idx;
242 int end_idx = page->end_idx;
243 int part_number_marker = page->part_number_marker;
244 int max_parts = page->max_parts;
245 int truncated = (end_idx < page->nparts);
246 int i;
247
248 /* Allocate XML buffer: header + per-part entries */
249 xml_capacity = 512
250 + (size_t) cf->bucket.len * 6 + 32
251 + strlen(key_str) * 6 + 32
252 + strlen(upload_id) * 6 + 32
253 + (size_t) (end_idx - start_idx) * 256
254 + (truncated ? 64 : 0);
255
256 xml = ngx_palloc(r->pool, xml_capacity);
257 if (xml == NULL) {
258 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
259 return NGX_HTTP_INTERNAL_SERVER_ERROR;
260 }
261
262 XML_APPEND("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
263 XML_APPEND("<ListPartsResult"
264 " xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">\n");
265 XML_APPEND(" ");
266 XML_APPEND_ELEM("Bucket", cf->bucket.data, cf->bucket.len);
267 XML_APPEND("\n ");
268 XML_APPEND_ELEM("Key", key_str, strlen(key_str));
269 XML_APPEND("\n ");
270 XML_APPEND_ELEM("UploadId", upload_id, strlen(upload_id));
271 XML_APPEND("\n");
272 XML_APPEND(" <StorageClass>STANDARD</StorageClass>\n");
273 XML_APPEND(" <PartNumberMarker>%d</PartNumberMarker>\n",
274 part_number_marker);
275 XML_APPEND(" <MaxParts>%d</MaxParts>\n", max_parts);
276 XML_APPEND(" <IsTruncated>%s</IsTruncated>\n",
277 truncated ? "true" : "false");
278
279 if (truncated) {
280 XML_APPEND(" <NextPartNumberMarker>%d</NextPartNumberMarker>\n",
281 parts[end_idx - 1].part_num);
282 }
283
284 for (i = start_idx; i < end_idx; i++) {
285 brix_format_iso8601(parts[i].mtime, iso_buf, sizeof(iso_buf));
286 snprintf(etag_buf, sizeof(etag_buf),
287 "\"%ld-%lld\"",
288 (long) parts[i].mtime,
289 (long long) parts[i].size);
290 XML_APPEND(" <Part>\n");
291 XML_APPEND(" <PartNumber>%d</PartNumber>\n", parts[i].part_num);
292 XML_APPEND(" <LastModified>%s</LastModified>\n", iso_buf);
293 XML_APPEND(" <ETag>%s</ETag>\n", etag_buf);
294 XML_APPEND(" <Size>%lld</Size>\n", (long long) parts[i].size);
295 XML_APPEND(" </Part>\n");
296 }
297
298 XML_APPEND("</ListPartsResult>\n");
299
300 b = ngx_create_temp_buf(r->pool, xml_len);
301 if (b == NULL) {
302 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
303 return NGX_HTTP_INTERNAL_SERVER_ERROR;
304 }
305 ngx_memcpy(b->pos, xml, xml_len);
306 b->last = b->pos + xml_len;
307 b->last_buf = 1;
308
309 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
310 (ngx_str_t) ngx_string("application/xml"), b);
311}

313ngx_int_t
314s3_handle_list_parts(ngx_http_request_t *r,
315 const char *fs_path,
316 ngx_http_s3_loc_conf_t *cf,
317 const char *key_str)
318{
319 char upload_id[128];
320 char mpu_dir[PATH_MAX];
321 brix_vfs_ctx_t vctx;
322 brix_vfs_stat_t vst;
323 mpu_part_entry_t *parts;
324 lp_page_t page = { NULL, NULL, NULL, 0, 0, 0, 0, 0 };
325 int nparts = 0;
326 int part_number_marker = 0;
327 int max_parts = 0;
328 int start_idx = 0;
329 int end_idx = 0;
330
331 /* Parse uploadId */
332 if (!s3_get_query_param(r, "uploadId", upload_id, sizeof(upload_id))) {
333 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
334 "InvalidArgument", "Missing uploadId.");
335 }
336
337 if (!mpu_validate_upload_id(upload_id)) {
338 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
339 "InvalidArgument", "The uploadId is invalid.");
340 }
341
342 s3_get_mpu_dir(fs_path, upload_id, mpu_dir, sizeof(mpu_dir));
343
344 /*
345 * Confinement + impersonation: mpu_dir comes from s3_get_mpu_dir() (validated
346 * upload_id + s3_resolve_key()-confined fs_path). Under MAP mode the staging
347 * dir / its parent are owned 0700 by the mapped user, so a bare worker lstat
348 * EACCESes on the parent search bit and ListParts wrongly 404s for the owner.
349 * Route through the VFS probe (broker stat as the mapped user under
350 * impersonation; plain no-follow lstat off impersonation).
351 */
352 s3_build_vfs_ctx(r, mpu_dir, cf, &vctx);
353 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) != NGX_OK
354 || !vst.is_directory)
355 {
356 return s3_send_xml_error(r, NGX_HTTP_NOT_FOUND,
357 "NoSuchUpload",
358 "The specified upload does not exist.");
359 }
360
361 /* Enumerate part.* files — at most MPU_MAX_PART_NUMBER entries */
362 parts = ngx_palloc(r->pool,
363 sizeof(mpu_part_entry_t) * (MPU_MAX_PART_NUMBER + 1));
364 if (parts == NULL) {
365 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
366 return NGX_HTTP_INTERNAL_SERVER_ERROR;
367 }
368
369 nparts = lp_scan_staged_parts(r, &vctx, mpu_dir, cf, parts);
370 if (nparts < 0) {
371 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
372 return NGX_HTTP_INTERNAL_SERVER_ERROR;
373 }
374
375 /* Parse pagination parameters */
376 lp_parse_query(r, &part_number_marker, &max_parts);
377
378 /* Apply part-number-marker: skip parts whose number <= marker */
379 start_idx = 0;
380 if (part_number_marker > 0) {
381 for (start_idx = 0; start_idx < nparts; start_idx++) {
382 if (parts[start_idx].part_num > part_number_marker) {
383 break;
384 }
385 }
386 }
387
388 end_idx = start_idx + max_parts;
389 if (end_idx > nparts) {
390 end_idx = nparts;
391 }
392
393 page.key_str = key_str;
394 page.upload_id = upload_id;
395 page.parts = parts;
396 page.start_idx = start_idx;
397 page.end_idx = end_idx;
398 page.nparts = nparts;
399 page.part_number_marker = part_number_marker;
400 page.max_parts = max_parts;
401
402 return lp_render_xml(r, cf, &page);
403}
