1/*
2 * acc.h — nginx-facing umbrella header for the `xrdacc` authorization engine.
3 *
4 * WHAT: the public surface of src/acc/ — a faithful re-implementation of
5 * XRootD's XrdAcc authorization framework, selectable at runtime via
6 * `brix_authdb_format xrdacc;`. This header re-exports the pure privilege
7 * algebra (privs.h) and will declare the entity, tables, and Access() engine
8 * as the milestones land.
9 *
10 * WHY: the module's original authdb (src/path/authdb.c, the `native` engine) is
11 * a 6-bit, single-rule, root://-only ACL. XrdAcc is a 9-bit additive model
12 * with negative privileges, templates, exclusive/compound rules, OS/NIS group
13 * resolution, hot-reload, and audit, applied uniformly to every protocol.
14 * `src/acc/` ports that engine alongside `native` (the directive picks which).
15 *
16 * HOW: leaf algebra in privs.h (no nginx deps, unit-testable); the nginx-coupled
17 * engine (entity/tables/access/groups/audit/config/refresh) builds on top and
18 * is reached through brix_acc_access(tables, entity, path, op) from the
19 * root://, WebDAV, and S3 authorization call sites.
20 */
21

35/*
36 * brix_acc_cap_t — one path/template capability (XrdAccCapability).
37 *
38 * A capability either references a named template (`tmpl` set, path fields
39 * unused) or carries a concrete path prefix + the privilege caps it grants.
40 * Capabilities chain via `next` to form one identity's ordered cap list; XrdAcc
41 * applies the FIRST prefix match in the list (not longest), so order matters.
42 * `pins`/`prem` describe a `@=` template-substitution point inside `path`.
43 */
44typedef struct brix_acc_cap_s {
45 struct brix_acc_cap_s *next; /* next capability in this list */
46 struct brix_acc_cap_s *tmpl; /* template indirection (a T-list caplist) */
47 char *path; /* prefix path, NUL-terminated (pool-owned) */
48 int plen; /* strlen(path) */
49 int pins; /* index of "@=" in path, 0 if none */
50 int prem; /* bytes after "@=" */
51 brix_acc_priv_caps_t caps;
52} brix_acc_cap_t;
53
54/*
55 * brix_acc_named_t — a name -> capability-list binding. One per entry in the
56 * user/group/host/netgroup/org/role/template categories (XrdAcc's per-category
57 * hashes). Also used for the domain (.suffix) list (name = ".cern.ch").
58 */
59typedef struct brix_acc_named_s {
60 struct brix_acc_named_s *next;
61 char *name;
62 int nlen;
63 brix_acc_cap_t *caps;
64} brix_acc_named_t;
65
66/*
67 * brix_acc_idrule_t — a compound-identity rule (from a `=` definition plus an
68 * `x` exclusive or `s` inclusive reference). Selectors are AND-ed; a NULL
69 * selector is unconstrained. `host` beginning with '.' is a domain suffix.
70 * `rule` >= 0 orders exclusive rules; -1 marks inclusive.
71 */
72typedef struct brix_acc_idrule_s {
73 struct brix_acc_idrule_s *next;
74 char *name;
75 char *user;
76 char *host;
77 char *org;
78 char *role;
79 char *grp;
80 int hlen; /* strlen(host) for domain suffix match */
81 int rule; /* exclusive order, or -1 = inclusive */
82 brix_acc_cap_t *caps;
83} brix_acc_idrule_t;
84
85/*
86 * brix_acc_tables_t — the full table set built from one authdb file. All
87 * nodes are allocated from `pool`, so a refresh frees the whole generation with
88 * one ngx_destroy_pool() and swaps in a new pointer atomically.
89 */
90typedef struct brix_acc_tables_s {
91 ngx_pool_t *pool; /* owns every allocation below */
92 brix_acc_named_t *g_list; /* groups (g) */
93 brix_acc_named_t *h_list; /* hosts exact (h <name>) */
94 brix_acc_named_t *n_list; /* netgroups (n) */
95 brix_acc_named_t *o_list; /* orgs (o) */
96 brix_acc_named_t *r_list; /* roles (r) */
97 brix_acc_named_t *t_list; /* templates (t) */
98 brix_acc_named_t *u_list; /* users (u <name>) */
99 brix_acc_named_t *d_list; /* domains (h .suffix) */
100 brix_acc_cap_t *z_list; /* default (u *) */
101 brix_acc_cap_t *x_list; /* fungible (u =) */
102 brix_acc_idrule_t *id_defs; /* all `=` definitions, by name */
103 brix_acc_idrule_t *sx_list; /* exclusive rules (x), ordered */
104 brix_acc_idrule_t *sy_list; /* inclusive rules (s) */
105 ngx_uint_t rule_count; /* parsed identity/path rules (diagnostics) */
106 time_t mtime; /* source mtime, for hot-reload */
107 /* Parse-time inputs (legacy XrdAcc tunables; transient, consulted only
108 * while this generation is being built — see brix_acc_authfile_parse). */
109 char parse_spacechar; /* spacechar: 0=off else ->' ' in ids */
110 ngx_flag_t parse_uridecode; /* encoding pct path: URI-decode paths */
111} brix_acc_tables_t;

137/*
138 * OS/NIS group hooks — installed by groups.c (M4). When unset, `n` (netgroup)

185/*
186 * brix_acc_http_t — the XrdAcc engine settings + per-worker state shared by
187 * the WebDAV and S3 HTTP loc-confs. Both embed it as a member named `acc`, so
188 * the shared directive setters, the lazy table build, the hot-reload timer and
189 * the authorize helper all operate on ONE type regardless of which HTTP module
190 * owns the location. The stream side keeps its flat srv-conf fields; this is
191 * the HTTP analogue of that group.
192 *
193 * The `tables`/`timer` state is per-worker: nginx forks after config load, so
194 * the first write in each worker (lazy build, timer arm) COW-privatises the
195 * page — every worker keeps its own tables and its own refresh timer with no
196 * locking, exactly like the stream srv conf.
197 */
198typedef struct {
199 ngx_uint_t format; /* [brix_authdb_format] native|xrdacc */
200 ngx_uint_t audit; /* [brix_authdb_audit] */
201 ngx_str_t authdb; /* [brix_authdb <path>] */
202 ngx_int_t refresh; /* [brix_authdb_refresh] secs; 0=off */
203 ngx_int_t gidlifetime; /* [brix_acc_gidlifetime] */
204 ngx_flag_t pgo; /* [brix_acc_pgo] primary group only */
205 ngx_str_t nisdomain; /* [brix_acc_nisdomain] */
206 ngx_flag_t resolve_hosts; /* [brix_acc_resolve_hosts] reverse DNS */
207 ngx_str_t spacechar; /* [brix_acc_spacechar] legacy id space char */
208 ngx_flag_t encoding; /* [brix_acc_encoding] legacy URI-decode paths */
209 ngx_str_t gidretran; /* [brix_acc_gidretran] legacy shared-gid skip */
210 struct brix_acc_tables_s *tables; /* per-worker, built lazily */
211 ngx_event_t timer; /* per-worker refresh timer (embedded) */
212 unsigned timer_armed:1; /* this worker has armed `timer` */
213} brix_acc_http_t;
