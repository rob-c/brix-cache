31/*
32 * chk_acc — accumulators shared across the do_check probe helpers.
33 *
34 * WHAT: the JSON-mode result state that each probe writes its outcome into.
35 * WHY: the probes were extracted from one large function; passing this struct
36 * by pointer keeps their data flow explicit with no new globals.
37 * HOW: zero-init in do_check (bools default false, counts 0); each probe sets
38 * its own fields and bumps `fails` on a failure — same semantics as the
39 * original inline locals.

53} chk_acc;

56/* (1) auth-as-advertised: the driver's chosen protocol must be anon (no &P=)
57 * or appear in the server's advertised security list. */
58static void
59chk_probe_auth(const diag_args *a, const brix_conn *c, chk_acc *acc)
60{
61 int ok;
62 if (c->sec_list[0] == '\0') {
63 ok = (c->diag.chosen_auth == NULL);
64 } else {
65 ok = (c->diag.chosen_auth != NULL) &&
66 (strstr(c->sec_list, c->diag.chosen_auth) != NULL);
67 }
68 acc->auth_ok = ok;
69 if (!ok) { acc->fails++; }
70 if (c->diag.chosen_auth != NULL) {
71 snprintf(acc->auth_proto, sizeof(acc->auth_proto), "%s", c->diag.chosen_auth);
72 }
73 if (!a->json) {
74 if (c->sec_list[0] == '\0') {
75 probe("auth-as-advertised", ok,
76 c->diag.chosen_auth == NULL ? "anonymous (no &P= offered)"
77 : "client used %s but server offered none",
78 c->diag.chosen_auth ? c->diag.chosen_auth : "");
79 } else {
80 probe("auth-as-advertised", ok, "chose %s from \"%s\"",
81 c->diag.chosen_auth ? c->diag.chosen_auth : "(none)", c->sec_list);
82 }
83 }
84}

87/* (2) no-silent-TLS-downgrade: gotoTLS advertised ⇒ session must be TLS. */
88static void
89chk_probe_tls(const diag_args *a, brix_conn *c, chk_acc *acc)
90{
91 unsigned f = (unsigned) c->server_flags;
92 const char *v = NULL, *cf = NULL;
93 int tls_active = brix_tls_info(c, &v, &cf);
94 acc->tls_active = tls_active;
95 acc->tls_required = (f & kXR_gotoTLS) ? 1 : 0;
96 if (v != NULL) {
97 snprintf(acc->tls_ver, sizeof(acc->tls_ver), "%s", v);
98 }
99 /* count as a failure only when the server demanded TLS but we got cleartext */
100 if (acc->tls_required && !tls_active) { acc->fails++; }
101 if (!a->json) {
102 if (f & kXR_gotoTLS) {
103 probe("no-tls-downgrade", tls_active,
104 tls_active ? "gotoTLS honored (%s)" : "gotoTLS advertised but cleartext!",
105 v ? v : "");
106 } else {
107 note("no-tls-downgrade", tls_active ? "TLS active" : "cleartext (gotoTLS not required)");
108 }
109 }

132/* (4) dirlist works + (7) dirlist-dstat == stat for the first entry. */
133static void
134chk_probe_dirlist(const diag_args *a, brix_conn *c, chk_acc *acc)
135{
136 brix_dirent *ents = NULL;
137 size_t n = 0;
138 brix_status dst;
139 brix_status_clear(&dst);
140 if (brix_dirlist(c, "/", 1, &ents, &n, &dst) != 0) {
141 acc->dirlist_ok = 0;
142 acc->fails++;
143 if (!a->json) { probe("dirlist", 0, "%s", dst.msg); }
144 return;
145 }
146 acc->dirlist_ok = 1;
147 acc->dirlist_count = (long long) n;
148 if (!a->json) { probe("dirlist", 1, "%zu entries under /", n); }
149 /* find a regular file entry with a stat and cross-check it */
150 for (size_t i = 0; i < n; i++) {
151 if (ents[i].have_stat && !(ents[i].st.flags & kXR_isDir)) {
152 char p[XRDC_PATH_MAX];
153 brix_statinfo s2;
154 brix_status s2st;
155 snprintf(p, sizeof(p), "/%s", ents[i].name);
156 brix_status_clear(&s2st);
157 if (brix_stat(c, p, &s2, &s2st) == 0) {
158 int ok = (s2.size == ents[i].st.size);
159 acc->dstat_tested = 1;
160 acc->dstat_ok = ok;
161 if (!ok) { acc->fails++; }
162 if (!a->json) {
163 probe("dstat==stat", ok,
164 "%s size dstat=%lld stat=%lld", ents[i].name,
165 (long long) ents[i].st.size, (long long) s2.size);
166 }
167 }
168 break;
169 }
170 }
171 free(ents);
172}

175/* (5) checksum-works: server digest == local digest of the downloaded bytes. */
176static void
177chk_probe_checksum(const diag_args *a, brix_conn *c, const char *target, chk_acc *acc)
178{
179 char srv[160], loc[160];
180 brix_status qst, lst;
181 int tmpfd;
182 char tmpl[] = "/tmp/xrddiag.XXXXXX";
183
184 brix_status_clear(&qst);
185 if (brix_query_cksum(c, target, "adler32", srv, sizeof(srv), &qst) != 0) {
186 if (!a->json) {
187 note("checksum-works", "server has no adler32 (%s)", qst.msg);
188 }
189 return;
190 }
191 tmpfd = mkstemp(tmpl);
192 if (tmpfd < 0) {
193 if (!a->json) { note("checksum-works", "mkstemp failed"); }
194 return;
195 }
196 {
197 int64_t got = 0;
198 brix_status_clear(&lst);
199 if (download_to_fd(c, target, tmpfd, &got, &lst) == 0 &&
200 brix_cksum_fd(tmpfd, XRDC_CK_ADLER32, loc, sizeof(loc), &lst) == 0) {
201 int ok = (strcmp(srv, loc) == 0);
202 acc->cksum_tested = 1;
203 acc->cksum_ok = ok;
204 if (!ok) { acc->fails++; }
205 if (!a->json) {
206 probe("checksum-works", ok,
207 "%s server=%s local=%s", target, srv, loc);
208 }
209 } else {
210 acc->cksum_tested = 1;
211 acc->cksum_ok = 0;
212 acc->fails++;
213 if (!a->json) { probe("checksum-works", 0, "%s", lst.msg); }
214 }
215 close(tmpfd);
216 unlink(tmpl);
217 }
218}

251/* (8) POSC-atomicity: a non-finalized POSC upload must leave NO file. Open a
252 * SECOND connection, posc-open + partial write, then ABANDON it (close the
253 * socket without kXR_close) and confirm the path is absent on the main conn. */
254static void
255chk_probe_posc(const diag_args *a, brix_conn *c, const brix_url *u, chk_acc *acc)
256{
257 brix_conn pc;
258 brix_status pst;
259 brix_file pf;
260 char ppath[64];
261 snprintf(ppath, sizeof(ppath), "/_xrddiag_posc_%d.tmp", (int) getpid());
262 brix_status_clear(&pst);
263 if (brix_connect(&pc, u, &a->conn, &pst) != 0) {
264 if (!a->json) {
265 note("posc-atomicity", "skipped — 2nd connect: %s", pst.msg);
266 }
267 return;
268 }
269 if (brix_file_open_write(&pc, ppath, 1, 1, &pf, &pst) != 0) {
270 if (!a->json) {
271 note("posc-atomicity", "skipped — posc open: %s (read-only export?)",
272 pst.msg);
273 }
274 brix_close(&pc);
275 return;
276 }
277 {
278 brix_statinfo si;
279 brix_status s2;
280 int visible;
281 (void) brix_file_write(&pc, &pf, 0, "partial", 7, &pst);
282 /* ABANDON: drop the socket with no kXR_close → server discards POSC. */
283 if (pc.io.fd >= 0) { close(pc.io.fd); pc.io.fd = -1; }
284 brix_close(&pc);
285 brix_status_clear(&s2);
286 visible = (brix_stat(c, ppath, &si, &s2) == 0);
287 acc->posc_tested = 1;
288 acc->posc_ok = !visible;
289 if (visible) { acc->fails++; }
290 if (!a->json) {
291 probe("posc-atomicity", !visible,
292 visible ? "PARTIAL FILE VISIBLE after abandoned upload!"
293 : "abandoned upload left no file (%s)",
294 visible ? "" : brix_kxr_name(s2.kxr));
295 }
296 if (visible) { brix_rm(c, ppath, &s2); } /* clean up the leak */
297 }
298}

301/* (9) handle-limits: opening files past the server cap must fail GRACEFULLY
302 * (a clean kXR_* error, not a crash/hang) and the connection survive. */
303static void
304chk_probe_handle_limits(const diag_args *a, brix_conn *c, const char *target, chk_acc *acc)
305{
306 brix_file fhs[64];
307 brix_status hst;
308 int opened = 0, i, graceful;
309 for (i = 0; i < 64; i++) {
310 brix_status_clear(&hst);
311 if (brix_file_open_read(c, target, &fhs[opened], &hst) != 0) {
312 break;
313 }
314 opened++;
315 }
316 graceful = (opened < 64); /* hit a cap with a clean error */
317 {
318 brix_statinfo si;
319 brix_status s2;
320 int alive;
321 brix_status_clear(&s2);
322 alive = (brix_stat(c, "/", &si, &s2) == 0) || (s2.kxr > 0);
323 if (graceful) {
324 acc->hlim_tested = 1;
325 acc->hlim_ok = alive;
326 if (!alive) { acc->fails++; }
327 if (!a->json) {
328 probe("handle-limits", alive, "capped at %d open (%s), conn alive",
329 opened, brix_kxr_name(hst.kxr));
330 }
331 } else {
332 if (!a->json) {
333 note("handle-limits", "no cap hit (opened %d), conn %s", opened,
334 alive ? "alive" : "DEAD");
335 }
336 }
337 }
338 for (i = 0; i < opened; i++) {
339 brix_status cs;
340 brix_status_clear(&cs);
341 brix_file_close(c, &fhs[i], &cs);
342 }
343}

391/*
392 * do_check — protocol-correctness and security probes against a live endpoint.
393 *
394 * WHAT: connects, runs a battery of functional and security probes, then
395 * outputs either a human-readable report or (--json) a single JSON object.
396 * WHY: JSON mode lets CI/CD pipelines and monitoring scripts consume results
397 * without parsing prose; the machine-readable path mirrors the human one 1:1.
398 * HOW: each probe runs regardless of output mode; outcomes accumulate in `acc`.
399 * Human path (probe/note/printf) is guarded by !a->json throughout.
400 * JSON is emitted once at the very end — never partial, never on early error.
401 */
402int
403do_check(const diag_args *a)
404{
405 brix_url u;
406 brix_conn c;
407 brix_status st;
408 brix_statinfo sti;
409 char target[XRDC_PATH_MAX];
410 int have_file;
411 /* JSON-mode accumulators — zero-init: bools default false, counts 0 */
412 chk_acc acc = { 0 };
413 snprintf(acc.auth_proto, sizeof(acc.auth_proto), "%s", "anon");
414
415 brix_status_clear(&st);
416 if (brix_endpoint_parse(a->url, &u, &st) != 0) {
417 fprintf(stderr, "xrddiag: %s\n", st.msg);
418 return 50;
419 }
420 if (brix_connect(&c, &u, &a->conn, &st) != 0) {
421 fprintf(stderr, "xrddiag: connect %s:%d: %s\n", u.host, u.port, st.msg);
422 return brix_shellcode(&st);
423 }
424
425 if (!a->json) {
426 printf("Session facts:\n");
427 brix_explain_conn(&c, &a->conn, stdout);
428 printf("Probes:\n");
429 }
430
431 chk_probe_auth(a, &c, &acc);
432 chk_probe_tls(a, &c, &acc);
433 chk_probe_confinement(a, &c, &acc);
434 chk_probe_dirlist(a, &c, &acc);
435
436 /* Resolve a file for the integrity probes (skip cleanly if none). */
437 have_file = (resolve_target(&c, &u, target, sizeof(target), &sti, &st) == 0);
438 if (!have_file && !a->json) {
439 note("checksum/pgread", "skipped — %s", st.msg);
440 }
441
442 if (have_file) {
443 chk_probe_checksum(a, &c, target, &acc);
444 chk_probe_pgread(a, &c, target, &acc);
445 chk_probe_posc(a, &c, &u, &acc);
446 chk_probe_handle_limits(a, &c, target, &acc);
447 }
448
449 if (!a->json) {
450 chk_probe_cred_validity();
451 }
452
453 brix_close(&c);
454
455 if (a->json) {
456 return chk_emit_json(a, &acc);
457 }
458
459 printf("Result: %d failure(s)\n", g_fails);
460 return g_fails ? 1 : 0;
461}

563/*
564 * checksum integrity: compare the server's advertised checksum against one recomputed
565 * locally from the bytes we downloaded. A disagreement means a stale checksum DB or
566 * on-disk corruption — a class of bug only a cross-check can surface.
567 */
568void
569dx_probe_checksum(brix_conn *c, const char *target, doctor_ep *e)
570{
571 char srv[160], loc[160];
572 brix_status st;
573 int fd;
574 char tmpl[] = "/tmp/xrddiag-dx.XXXXXX";
575 int64_t got = 0;
576
577 /* Request adler32 (the client can recompute it) and compare the bare hex,
578 * mirroring the proven `check` checksum-works probe. */
579 brix_status_clear(&st);
580 if (brix_query_cksum(c, target, "adler32", srv, sizeof(srv), &st) != 0) {
581 if (st.kxr == kXR_Unsupported) {
582 dx_record_status(e, "checksum", &st);
583 }
584 return; /* server simply doesn't expose a checksum — not a problem */
585 }
586 fd = mkstemp(tmpl);
587 if (fd < 0) {
588 return;
589 }
590 brix_status_clear(&st);
591 if (download_to_fd(c, target, fd, &got, &st) == 0
592 && brix_cksum_fd(fd, XRDC_CK_ADLER32, loc, sizeof(loc), &st) == 0) {
593 if (strcmp(srv, loc) == 0) {
594 dx_record(e, &(dx_note){ "checksum", DX_OK, 0, "server checksum matches read data", "" });
595 } else {
596 dx_record(e, &(dx_note){ "checksum", DX_FAIL, 0,
597 "server checksum disagrees with the bytes read (stale checksum DB or data corruption)",

606/*
607 * write path (GATED): create a unique temp dir, write+read-back a small object, verify
608 * byte-exactness, then clean up. The failure code pins the cause precisely
609 * (kXR_fsReadOnly = read-only export, kXR_NotAuthorized = no write scope, quota/space).
610 * Always reverses its own mutations; bounded; never touches user data paths.
611 */
612void
613dx_probe_write(brix_conn *c, doctor_ep *e)
614{
615 char dir[96], path[160];
616 brix_status st;
617 brix_file f;
618 const char payload[] = "xrddiag-remote-doctor-write-probe\n";
619 int wrote = 0;
620
621 /* pid + monotonic clock makes the temp namespace collision-proof across runs
622 * and pid reuse, so the probe never reuses or fights a pre-existing directory. */
623 snprintf(dir, sizeof(dir), "/.xrddiag-dx-%ld-%llx", (long) getpid(),
624 (unsigned long long) brix_mono_ns());
625 snprintf(path, sizeof(path), "%s/probe.tmp", dir);
626
627 brix_status_clear(&st);
628 if (brix_mkdir(c, dir, 0700, 0, &st) != 0) {
629 dx_record_status(e, "write", &st);
630 return;
631 }
632 brix_status_clear(&st);
633 if (brix_file_open_write(c, path, 1 /*force*/, 0 /*posc*/, &f, &st) != 0) {
634 dx_record_status(e, "write", &st);
635 brix_rmdir(c, dir, &st);
636 return;
637 }
638 brix_status_clear(&st);
639 if (brix_file_write(c, &f, 0, payload, sizeof(payload) - 1, &st) == 0) {
640 wrote = 1;
641 }
642 {
643 /* a close failure means the server never durably committed the write. */
644 brix_status cst;
645 brix_status_clear(&cst);
646 if (brix_file_close(c, &f, &cst) != 0 && wrote) {
647 wrote = 0;
648 st = cst;
649 }
650 }
651
652 if (!wrote) {
653 dx_record_status(e, "write", &st);
654 } else {
655 /* read back + verify byte-exactness */
656 brix_file rf;
657 uint8_t rb[64];
658 ssize_t rn = -1;
659 brix_status rst;
660 brix_status_clear(&rst);
661 if (brix_file_open_read(c, path, &rf, &rst) == 0) {
662 rn = brix_file_read(c, &rf, 0, rb, sizeof(rb), &rst);
663 brix_file_close(c, &rf, &rst);
664 }
665 if (rn == (ssize_t) (sizeof(payload) - 1)
666 && memcmp(rb, payload, sizeof(payload) - 1) == 0) {
667 dx_record(e, &(dx_note){ "write", DX_OK, 0, "write path healthy (write/read-back verified)", "" });
668 } else {
669 dx_record(e, &(dx_note){ "write", DX_FAIL, 0,
670 "write succeeded but read-back did not match (durability/consistency fault)",
671 "check the server write-through/cache flush path and storage backend" });
672 }
673 }
674 /* always clean up our mutations; warn (no silent residue) if the dir survives. */
675 brix_status_clear(&st);
676 brix_rm(c, path, &st);
677 brix_status_clear(&st);
678 if (brix_rmdir(c, dir, &st) != 0) {
679 dx_record(e, &(dx_note){ "write", DX_WARN, st.kxr,
680 "write-probe test directory could not be removed",
681 "remove the leftover write-probe directory under the export root" });
682 }
683}

716dx_authz_anon(const diag_args *a, const brix_url *u, const char *target,

726 dx_record(e, &(dx_note){ "authz-anon", DX_WARN, st.kxr,

728 "check reachability and retry when the server is up" });

830 * in a unique temp namespace. A correct server denies (kXR_NotAuthorized). A write
831 * that SUCCEEDS means token write-scope is not enforced — privilege escalation.
832 * Always reverses any mutation.

835dx_authz_scope(const diag_args *a, const brix_url *u, const char *tok, doctor_ep *e)
836{

838 brix_status st;

840

865 } else if (st.kxr == kXR_NotAuthorized || st.kxr == kXR_AuthFailed
866 || st.kxr == XRDC_EAUTH) {

868 "write (mkdir) correctly denied for a read-only token (scope/ACL enforced)", "" });
869 } else if (st.kxr == kXR_fsReadOnly) {
870 dx_record(e, &(dx_note){ "authz-scope", DX_WARN, st.kxr,
871 "export is read-only (allow_write off) — cannot isolate token write-scope enforcement",
872 "re-run against a read-write export to test token write-scope" });
873 } else {
874 dx_record(e, &(dx_note){ "authz-scope", DX_WARN, st.kxr,
875 "write probe failed for an unexpected reason; scope enforcement unclear",
