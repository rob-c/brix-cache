1/*
2 * http_headers.c - shared HTTP request/response header helpers.
3 *
4 * WHAT: Finds, reads, sets, and compares HTTP headers in nginx requests (headers_in)
5 * and responses (headers_out). Supports case-insensitive name lookup,
6 * control-character validation, whitespace-trimmed value comparison,
7 * and safe header insertion with pool allocation.
8 *
9 * WHY: WebDAV and S3 modules both read request headers (If-None-Match, Range,
10 * Overwrite) and write response headers (ETag, Content-Range). Shared
11 * implementation prevents duplicated list traversal, alloc patterns,
12 * and comparison logic across the two HTTP protocols.
13 *
14 * HOW: find_header walks r->headers_in.headers.part chain with ngx_strncasecmp;
15 * set_header pushes to r->headers_out.headers via ngx_list_push; value_equals
16 * trims whitespace then case-insensitive compares. ctl check rejects bytes <0x20/0x7F.
17*/

25/*
26 * brix_http_find_header - search headers_in list by name (case-insensitive).
27 *
28 * WHAT: Iterates r->headers_in.headers.part → next parts, comparing each header
29 * key against name using ngx_strncasecmp. Returns pointer to matching
30 * ngx_table_elt_t or NULL if not found.
31 *
32 * WHY: HTTP headers are stored in nginx's linked-list structure (ngx_list_part_t).
33 * Both WebDAV and S3 need to look up specific headers (If-None-Match,
34 * Range, Overwrite) without knowing which part of the list they reside in.
35 *
36 * HOW: Walk r->headers_in.headers.part → next until end. Compare key.len == name_len
37 * && ngx_strncasecmp(key.data, name, name_len). Return match or NULL.
38 */
39
40ngx_table_elt_t *
41brix_http_find_header(ngx_http_request_t *r, const char *name,
42 size_t name_len)
43{
44 ngx_list_part_t *part;
45 ngx_table_elt_t *hdr;
46 ngx_uint_t i;
47
48 if (r == NULL || name == NULL) {
49 return NULL;
50 }
51
52 part = &r->headers_in.headers.part;
53 hdr = part->elts;
54
55 for (;;) {
56 for (i = 0; i < part->nelts; i++) {
57 if (hdr[i].key.len == name_len
58 && ngx_strncasecmp(hdr[i].key.data,
59 (u_char *) name, name_len) == 0)
60 {
61 return &hdr[i];
62 }
63 }
64
65 if (part->next == NULL) {
66 break;
67 }
68
69 part = part->next;
70 hdr = part->elts;
71 }
72
73 return NULL;
74}

103/*
104 * brix_http_extract_bearer - parse "Authorization: Bearer <token>".
105 *
106 * WHAT: Returns a no-copy token slice when the auth scheme is Bearer,
107 * matching the scheme case-insensitively and trimming optional
108 * whitespace around the token value.
109 *
110 * WHY: WebDAV token auth and HTTP-TPC credential delegation both need this
111 * parsing. Keeping it here prevents auth-scheme case bugs from
112 * reappearing in individual protocol handlers.
113 */
114ngx_int_t
115brix_http_extract_bearer(const ngx_str_t *auth_header, ngx_str_t *token_out)
116{
117 static const u_char bearer[] = "Bearer";
118 u_char *p;
119 u_char *end;
120 size_t bearer_len = sizeof(bearer) - 1;
121
122 if (token_out == NULL) {
123 return NGX_ERROR;
124 }
125
126 token_out->data = NULL;
127 token_out->len = 0;
128
129 if (auth_header == NULL || auth_header->data == NULL) {
130 return NGX_DECLINED;
131 }
132
133 p = auth_header->data;
134 end = auth_header->data + auth_header->len;
135
136 while (p < end && (*p == ' ' || *p == '\t')) {
137 p++;
138 }
139
140 if ((size_t) (end - p) < bearer_len
141 || ngx_strncasecmp(p, (u_char *) bearer, bearer_len) != 0)
142 {
143 return NGX_DECLINED;
144 }
145
146 p += bearer_len;
147 if (p >= end || (*p != ' ' && *p != '\t')) {
148 return NGX_DECLINED;
149 }
150
151 while (p < end && (*p == ' ' || *p == '\t')) {
152 p++;
153 }
154 while (end > p && (end[-1] == ' ' || end[-1] == '\t')) {
155 end--;
156 }
157
158 if (p >= end) {
159 return NGX_ERROR;
160 }
161
162 token_out->data = p;
163 token_out->len = (size_t) (end - p);
164 return NGX_OK;
165}

167/*
168 * brix_http_str_has_ctl - check for HTTP control characters in string data.
169 *
170 * WHAT: Scans data[0..len] for bytes < 0x20 (space) or == 0x7F (DEL). Returns
171 * 1 if any found, 0 if clean. NULL data returns 1 (unsafe).
172 *
173 * WHY: RFC 7230 §3 restricts header values to visible characters (0x21-0x7E)
174 * plus HTAB (0x09). Control bytes in path strings or header values can
175 * cause log corruption, injection, or parsing ambiguity. This check
176 * validates user-supplied strings before use.
177 *
178 * HOW: for(i=0..len): if data[i]<0x20 || data[i]==0x7F → return 1. Else return 0.
179 */
180
181ngx_int_t
182brix_http_str_has_ctl(const u_char *data, size_t len)
183{
184 size_t i;
185
186 if (data == NULL) {
187 return 1;
188 }
189
190 for (i = 0; i < len; i++) {
191 if (data[i] < 0x20 || data[i] == 0x7f) {
192 return 1;
193 }
194 }
195
196 return 0;
197}

199/*
200 * brix_http_header_value_equals - case-insensitive literal comparison with whitespace trim.
201 *
202 * WHAT: Strips leading/trailing space and tab from value->data, then compares
203 * the trimmed length and bytes against literal using ngx_strncasecmp.
204 * Returns 1 (true) or 0 (false).
205 *
206 * WHY: HTTP header values may contain optional whitespace per RFC 7230 §3.2.
207 * Comparisons for "F" (Overwrite), "T", etc. must ignore leading/trailing
208 * whitespace while still enforcing exact length match after trimming.
209 *
210 * HOW: trim start past ' '/'\t'; trim end back past ' '/'\t'. Compare len==strlen(literal)
211 * && ngx_strncasecmp(trimmed, literal, literal_len). Return result.
212 */
213
214ngx_int_t
215brix_http_header_value_equals(const ngx_str_t *value, const char *literal)
216{
217 u_char *start;
218 u_char *end;
219 size_t len;
220 size_t literal_len;
221
222 if (value == NULL || literal == NULL || value->data == NULL) {
223 return 0;
224 }
225
226 start = value->data;
227 end = value->data + value->len;
228
229 while (start < end && (*start == ' ' || *start == '\t')) {
230 start++;
231 }
232 while (end > start && (end[-1] == ' ' || end[-1] == '\t')) {
233 end--;
234 }
235
236 len = (size_t) (end - start);
237 literal_len = strlen(literal);
238
239 return len == literal_len
240 && ngx_strncasecmp(start, (u_char *) literal, literal_len) == 0;
241}

272/*
273 * brix_http_set_header_str - insert header into headers_out with optional value copy and pointer return.
274 *
275 * WHAT: Pushes a new ngx_table_elt_t onto r->headers_out.headers list. Sets
276 * key from const char (hash=1), value either copied via ngx_pstrdup or
277 * set directly from value struct. Optionally returns h pointer via out.
278 *
279 * WHY: WebDAV and S3 both need to construct response headers dynamically (ETag,
280 * Content-Range, CORS). This function handles pool allocation, key/value
281 * setup, and optional caller-pointer return in one call.
282 *
283 * HOW: ngx_list_push(&r->headers_out.headers) → set hash=1, key from strlen(key),
284 * value: if copy_value → ngx_pstrdup(r->pool, value); else h->value=*value.
285 * If out != NULL → *out = h. Return NGX_OK or NGX_HTTP_INTERNAL_SERVER_ERROR.
286 */
287
288ngx_int_t
289brix_http_set_header_str(ngx_http_request_t *r, const char *key,
290 const ngx_str_t *value, ngx_flag_t copy_value, ngx_table_elt_t **out)
291{
292 ngx_table_elt_t *h;
293
294 if (out != NULL) {
295 *out = NULL;
296 }
297
298 if (r == NULL || key == NULL || value == NULL) {
299 return NGX_HTTP_INTERNAL_SERVER_ERROR;
300 }
301
302 h = ngx_list_push(&r->headers_out.headers);
303 if (h == NULL) {
304 return NGX_HTTP_INTERNAL_SERVER_ERROR;
305 }
306
307 h->hash = 1;
308 h->key.len = ngx_strlen(key);
309 h->key.data = (u_char *) key;
310
311 if (copy_value) {
312 h->value.data = ngx_pstrdup(r->pool, (ngx_str_t *) value);
313 if (h->value.data == NULL) {
314 return NGX_HTTP_INTERNAL_SERVER_ERROR;
315 }
316 h->value.len = value->len;
317
318 } else {
319 h->value = *value;
320 }
321
322 if (out != NULL) {
323 *out = h;
324 }
325
326 return NGX_OK;
327}

428static int
429brix_hex_nibble(u_char c)
430{
431 if (c >= '0' && c <= '9') { return c - '0'; }
432 if (c >= 'a' && c <= 'f') { return c - 'a' + 10; }
433 if (c >= 'A' && c <= 'F') { return c - 'A' + 10; }
434 return -1;
435}

437size_t
438brix_urldecode_inplace(char *str)
439{
440 char *rp = str;
441 char *wp = str;
442 int hi, lo;
443
444 while (*rp) {
445 if (*rp == '%' && rp[1] && rp[2]
446 && (hi = brix_hex_nibble((u_char) rp[1])) >= 0
447 && (lo = brix_hex_nibble((u_char) rp[2])) >= 0)
448 {
449 *wp++ = (char) ((hi << 4) | lo);
450 rp += 3;
451 } else if (*rp == '+') {
452 *wp++ = ' ';
453 rp++;
454 } else {
455 *wp++ = *rp++;
456 }
457 }
458 *wp = '\0';
459 return (size_t) (wp - str);
460}

482void
483brix_http_redact_query_token(ngx_str_t *query)
484{
485 static const char *const keys[] = { "authz=", "access_token=" };
486 size_t k;
487
488 if (query == NULL || query->data == NULL || query->len == 0) {
489 return;
490 }
491
492 /* Length-preserving: overwrite each token value with 'x' in place. The log
493 * sources (r->args / r->unparsed_uri / r->request_line) alias overlapping
494 * regions of the same request buffer, so we must NOT memmove/shrink — that
495 * would desync their independent ->len fields. Overwriting same-length is
496 * idempotent and safe to apply to all three. */
497 for (k = 0; k < sizeof(keys) / sizeof(keys[0]); k++) {
498 u_char *p = query->data;
499 u_char *end = query->data + query->len;
500 size_t klen = ngx_strlen(keys[k]);
501
502 while (p < end
503 && (p = ngx_strlcasestrn(p, end, (u_char *) keys[k], klen - 1))
504 != NULL)
505 {
506 u_char *v = p + klen;
507 /* A query arg value ends at '&'; in r->request_line it also ends at the
508 * SP before the HTTP version (and CR/LF/HTAB) — stop there so we never
509 * clobber " HTTP/1.1". */
510 while (v < end && *v != '&' && *v != ' ' && *v != '\t'
511 && *v != '\r' && *v != '\n') {
512 *v++ = 'x';
513 }
514 p = v;
515 }
516 }
517}
