7/*
8 * open_manager.c — kXR_open manager-mode redirection, split from open_request.c.
9 * A manager (redirector) never serves bytes: it points the client at a data
10 * server via the dynamic registry / CMS parent (with the tried-exhausted and
11 * collapse-redir caches short-circuiting the common cases), or via the static
12 * manager_map prefix table. Bodies are moved verbatim; the early-returns are
13 * unchanged. Returns NGX_DECLINED to resolve locally instead.
14 */
15/*
16 * Registry miss — ask the CMS parent via kYR_locate. Parks the open in
17 * XRD_ST_WAITING_CMS and returns NGX_AGAIN when the locate was sent; NGX_DECLINED
18 * to fall through to the static map / local resolve / error.
19 */
20static ngx_int_t
21brix_open_cms_locate(brix_ctx_t *ctx, ngx_connection_t *c,
22 ngx_stream_brix_srv_conf_t *conf, const char *clean_path)
23{
24 uint32_t streamid;
25
26 if (conf->cms.ctx == NULL) {
27 return NGX_DECLINED;
28 }
29
30 streamid = ngx_brix_cms_next_streamid(conf->cms.ctx);
31 if (brix_pending_insert(streamid, ngx_pid, c->fd, c->number,
32 ctx->recv.cur_streamid,
33 conf->cms.locate_timeout) == NGX_OK)
34 {
35 ctx->cms_wait_streamid = streamid;
36 ctx->state = XRD_ST_WAITING_CMS;
37 ngx_add_timer(c->read, conf->cms.locate_timeout);
38 if (ngx_brix_cms_send_locate(conf->cms.ctx, streamid,
39 clean_path) == NGX_OK)
40 {
41 return NGX_AGAIN;
42 }
43 ngx_del_timer(c->read);
44 ctx->state = XRD_ST_REQ_HEADER;
45 brix_pending_remove(streamid, ngx_pid);
46 }
47 return NGX_DECLINED;
48}

51brix_open_manager_redirect(brix_ctx_t *ctx, ngx_connection_t *c,
52 ngx_stream_brix_srv_conf_t *conf, int is_write, const char *clean_path)
53{
54 int op = is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD;
55
56 /* Dynamic manager mode: redirect to best registered server. */
57 if (conf->manager_mode) {
58 char redir_host[256];
59 uint16_t redir_port;
60
61 /* tried/triedrc: a read whose client has already visited every server
62 * holding this path (all returned enoent) must get not-found, not yet
63 * another redirect — otherwise it loops to the client redirect limit.
64 * Writes are excluded: they create the file on the selected server. */
65 if (!is_write
66 && brix_manager_tried_exhausted(ctx->recv.payload, ctx->recv.cur_dlen,
67 clean_path)) {
68 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN", clean_path,
69 "rd", kXR_NotFound,
70 "file not found on any data server");
71 }
72
73 /* Collapse-redir cache: serve reads from cache to skip CMS. */
74 if (!is_write && conf->caps.collapse_redir
75 && brix_redir_cache_lookup(clean_path, redir_host,
76 sizeof(redir_host), &redir_port)) {
77 BRIX_RETURN_REDIR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
78 clean_path, "redir-cache",
79 redir_host, redir_port);
80 }
81
82 /* Open may redirect to a server whose CMS heartbeat just dropped (a
83 * transient blip under load blacklists it for 30 s though its data plane
84 * is still serving) — better than a false NotFound for a file that
85 * exists. kXR_locate stays strict. A truly dead target just makes the
86 * client's connect fail and the tried/triedrc retry converges to
87 * NotFound. */
88 if (brix_srv_select_or_blacklisted(clean_path, is_write, redir_host,
89 sizeof(redir_host), &redir_port)) {
90 if (!is_write && conf->caps.collapse_redir) {
91 brix_redir_cache_insert(clean_path, redir_host, redir_port,
92 conf->caps.collapse_redir_ttl);
93 }
94 BRIX_RETURN_REDIR(ctx, c, op,
95 "OPEN", clean_path, "registry",
96 redir_host, redir_port);
97 }
98
99 /* Registry miss — ask the CMS parent via kYR_locate. */
100 {
101 ngx_int_t crc = brix_open_cms_locate(ctx, c, conf, clean_path);
102 if (crc != NGX_DECLINED) {
103 return crc; /* parked on the CMS locate (NGX_AGAIN) */
104 }
105 /* NGX_DECLINED: fall through to static-map / local resolve. */
106 }
107 }
108
109 /* Manager-mode mapping: redirect opens for configured prefixes. */
110 if (conf->manager_map != NULL) {
111 const brix_manager_map_t *m = brix_find_manager_map(clean_path,
112 conf->manager_map);
113 if (m != NULL) {
114 BRIX_RETURN_REDIR(ctx, c,
115 op,
116 "OPEN", clean_path, "redirect",
117 (const char *) m->host.data, m->port);
118 }
119 }
120
121 return NGX_DECLINED;
122}
