30/* Pup wire builders (mirror src/cms/wire.c put_string / put_int) */
31/* Append a Pup string: [2B BE len incl NUL][bytes][NUL]. Empty -> [00 00]. */
32static unsigned char *
33put_str(unsigned char *p, const char *s)
34{
35 size_t n = (s == NULL) ? 0 : strlen(s);
36 if (n == 0) { *p++ = 0; *p++ = 0; return p; }
37 unsigned wlen = (unsigned) (n + 1);
38 *p++ = (unsigned char) (wlen >> 8);
39 *p++ = (unsigned char) wlen;
40 memcpy(p, s, n); p += n;
41 *p++ = '\0';
42 return p;
43}
44
45/* Append a Pup int: tag 0xa0 + 4B BE. */
46static unsigned char *
47put_int(unsigned char *p, uint32_t v)
48{
49 *p++ = 0xa0;
50 *p++ = (unsigned char) (v >> 24);
51 *p++ = (unsigned char) (v >> 16);
52 *p++ = (unsigned char) (v >> 8);
53 *p++ = (unsigned char) v;
54 return p;
55}

58span_eq(const unsigned char *p, size_t len, const char *s)
59{
60 return p != NULL && len == strlen(s) && memcmp(p, s, len) == 0;

64static void
65test_mkdir(void)
66{
67 /* fwdArgA: ident, mode, path, [fence], opaque(absent) */
68 unsigned char buf[256], *p = buf;
69 p = put_str(p, "alice.0:1@host");
70 p = put_str(p, "755");
71 p = put_str(p, "/atlas/new");
72 brix_cms_rrdata_t d;
73 int rc = brix_cms_rrdata_parse(K_MKDIR, buf, (size_t)(p - buf), &d);
74 CHECK(rc == 0);
75 CHECK(span_eq(d.ident, d.ident_len, "alice.0:1@host"));
76 CHECK(span_eq(d.mode, d.mode_len, "755"));
77 CHECK(span_eq(d.path, d.path_len, "/atlas/new"));
78 CHECK(d.opaque == NULL);
79 /* the path span is NUL-terminated in place (wire NUL retained) */
80 CHECK(d.path != NULL && d.path[d.path_len] == '\0');
81}
