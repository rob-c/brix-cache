1/*
2 * tmp_path.h — Uniform temporary-file path construction for atomic writes.
3 *
4 * Single function: brix_make_tmp_path() generates unique temp paths in the format
5 * <base>.xrd-tmp.<pid>.<random> used by S3 PUT, WebDAV COPY, and WebDAV TPC pull. The .xrd-tmp.
6 * prefix enables operators to glob-clean orphaned temp files across all subsystems with one
7 * pattern: find /export -name "*.xrd-tmp.*" -mtime +1 -delete.
8 */

16/*
17 * brix_make_tmp_path — build a unique temporary path adjacent to base_path.
18 *
19 * Writes "<base_path>.xrd-tmp.<pid>.<random>" into out[out_sz].
20 * Using a uniform suffix across all protocols means stale temp files from
21 * any subsystem (WebDAV COPY, WebDAV TPC pull, S3 PUT) are recognisable
22 * and can be cleaned by a single glob pattern ("*.xrd-tmp.*").
23 *
24 * Returns NGX_OK on success, NGX_ERROR if out_sz is too small.
25 */
26ngx_int_t brix_make_tmp_path(const char *base_path, char *out, size_t out_sz);

28/*
29 * brix_make_resume_path — build the DETERMINISTIC upload-resume staging path.
30 *
31 * With stage_dir empty/NULL: writes
32 * "<base_path>.xrdresume.<hex16(SHA-256(principal "\0" base_path))>.part"
33 * (adjacent to the destination → atomic rename commit). With stage_dir set:
34 * "<stage_dir>/<hex16(...)>.xrdresume.part"
35 * (the partial lives on a fast cache device; commit moves it to storage).
36 * Either way the name is a pure function of (principal, base_path), so a
37 * reconnecting client's re-open of the same final path by the same identity
38 * lands on the same staging file and resumes from its offset. principal ""/NULL
39 * => "anonymous". Stale partials glob-clean with "*.xrdresume*.part".
40 *
41 * Returns NGX_OK, or NGX_ERROR on hash failure / truncation.
42 */
43ngx_int_t brix_make_resume_path(const char *base_path, const char *principal,
44 const char *stage_dir, char *out,
45 size_t out_sz);
