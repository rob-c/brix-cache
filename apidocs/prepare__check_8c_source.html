21static ngx_flag_t
22brix_prepare_has_forbidden_component(const char *path)
23{
24 const char *p = path;
25
26 while (*p != '\0') {
27 const char *seg;
28 size_t len;
29
30 while (*p == '/') {
31 p++;
32 }
33
34 seg = p;
35 while (*p != '\0' && *p != '/') {
36 p++;
37 }
38
39 len = (size_t) (p - seg);
40 if ((len == 1 && seg[0] == '.')
41 || (len == 2 && seg[0] == '.' && seg[1] == '.'))
42 {
43 return 1;
44 }
45 }
46
47 return 0;
48}

64/*
65 * WHAT: run the three prepare authorization tiers (authdb VO/ACL, VO identity
66 * ACL, token scope) on one resolved path.
67 * WHY: the existing-file and noerrs-absent branches of check_path must apply
68 * the SAME gate on the SAME paths (verdict parity between "exists" and
69 * "absent") — factoring it here removes the duplication AND guarantees the
70 * two branches can never drift apart. Authorization is a property of the
71 * identity + logical path, not of on-disk existence.
72 * HOW: each tier that denies sends its specific error via check_fail and its rc
73 * (NGX_DONE/error) is returned; NGX_OK only when all three pass.
74 */
75static ngx_int_t
76prepare_path_authz(brix_ctx_t *ctx, ngx_connection_t *c,
77 ngx_stream_brix_srv_conf_t *conf, const char *pathbuf, const char *full_path)
78{
79 if (brix_authz_check(ctx, c, conf, pathbuf, full_path, "PREPARE",

81 return brix_prepare_check_fail(ctx, c, full_path, kXR_NotAuthorized,

83 }

96/*
97 * Map a confined-stat failure (non-noerrs, or non-ENOENT errno) to the wire
98 * error: ENOENT/ENOTDIR → kXR_NotFound, EACCES/EPERM → kXR_NotAuthorized, any
99 * other errno → kXR_IOError. Returns the check_fail rc (NGX_DONE/error).
100 */
101static ngx_int_t
102prepare_stat_error(brix_ctx_t *ctx, ngx_connection_t *c,
103 const char *pathbuf, const char *full_path)
104{
105 if (errno == ENOENT || errno == ENOTDIR) {
106 return brix_prepare_check_fail(ctx, c, pathbuf, kXR_NotFound,
107 "file not found");
108 }
109 if (errno == EACCES || errno == EPERM) {
110 return brix_prepare_check_fail(ctx, c, full_path, kXR_NotAuthorized,
111 "not authorized");
112 }
113 return brix_prepare_check_fail(ctx, c, full_path, kXR_IOError,
114 "prepare stat failed");
115}

149 /*
150 * CONTRACT: the same "file is absent" condition (ENOENT/ENOTDIR from the
151 * confined stat) has two outcomes selected by the kXR_noerrs flag:
152 * - noerrs set → not an error. The path is counted in *missing and the
153 * request still succeeds, so a client can prepare/stage files that do
154 * not exist on disk yet (tape nearline recall, not-yet-cached objects).
155 * - noerrs clear → kXR_NotFound, failing the request on the first miss.
156 * EACCES/EPERM and any other errno always fail regardless of noerrs.
157 */

159 if ((errno == ENOENT || errno == ENOTDIR) && noerrs) {
160 (*missing)++;

162 * PATH, not of on-disk existence. A prepare/stage of a not-yet-
163 * materialised object (tape nearline recall, not-yet-cached) must
164 * still prove the caller may READ/STAGE this namespace path —
165 * otherwise an unauthorized principal drives recalls or enumerates
166 * the namespace via prepare, and later serves the recalled bytes from
167 * the shared cache. Run the SAME three tiers, on the SAME paths, as
168 * the existing-file branch below (verdict parity between "exists" and
169 * "absent"); only then supply the staging path. */

172 full_path);
173 if (arc != NGX_OK) {
174 return arc;
175 }
176 }
177 /* For staging: supply absolute path even if file doesn't exist yet
178 * (tape nearline / not-yet-created). */
179 if (out_resolved != NULL) {
180 ngx_cpystrn((u_char *) out_resolved, (u_char *) full_path,
181 PATH_MAX);
182 }
183 return NGX_OK;
184 }
185 return prepare_stat_error(ctx, c, pathbuf, full_path);
186 }
187
188 {
189 ngx_int_t arc = prepare_path_authz(ctx, c, conf, pathbuf, full_path);
190 if (arc != NGX_OK) {
191 return arc;
192 }
193 }
194
195 /* Copy the absolute export path for the staging command; only authorized
196 * paths reach here so the staging hook can trust the value. */
197 if (out_resolved != NULL) {
198 ngx_cpystrn((u_char *) out_resolved, (u_char *) full_path, PATH_MAX);
199 }
200
201 if (S_ISDIR(st.st_mode)) {
202 if (noerrs) {
203 (*missing)++;
204 return NGX_OK;
205 }
206 return brix_prepare_check_fail(ctx, c, pathbuf, kXR_isDirectory,
207 "prepare target is a directory");
208 }
