1/*
2 * open_flags.h — kXR_open option-bit semantics (single source of truth).
3 *
4 * WHAT: the create/exclusive/truncate/append/rdwr meaning of the kXR_open
5 * `options` field, expressed once and shared by every encoder and the one
6 * decoder:
7 * - brix_open_options_build: intent -> kXR_open options (the request)
8 * - brix_open_options_to_posix: options -> POSIX open(2) flags (the server)
9 * - brix_open_options_is_write: options -> is this a write-mode open?
10 * - brix_open_force_for_open/create: POSIX flags -> the `force` tri-state
11 * WHY: this flag contract was hand-coded in FOUR places — the server's
12 * options->O_* decoder (src/read/open_resolved_file.c), the xrdcp/xrdfs
13 * options builder (client/lib/ops_file.c), and BOTH FUSE drivers' POSIX->force
14 * mapping (client/apps/xrootdfs.c, xrootdfs_legacy.c). The two halves are
15 * inverse operations on the SAME spec: if "kXR_new without kXR_delete" drifts
16 * between the client's intent and the server's O_EXCL derivation, the result
17 * is a spurious EEXIST or a SILENT OVERWRITE — a data-integrity bug no test
18 * catches, because the halves never referenced a shared definition.
19 * HOW: header-only static inlines — pure flag math, no ngx/alloc/OpenSSL — so the
20 * same code compiles into the nginx module and the ngx-free client. Server
21 * policy that is NOT part of the flag spec (POSC staging, directory checks,
22 * access-control) stays at the call sites.
23 *
24 * The `force` tri-state (shared by xrdcp and both FUSE drivers):
25 * 1 = truncate/overwrite (kXR_delete) 0 = create-new, fail if exists (kXR_new)
26 * 2 = update in place (kXR_open_updt only, neither create nor truncate)
27 *
28 * Clean-room: bit meanings from src/protocol/flags.h (vs XProtocol kXR_open*).
29 */

39/* The option bits that make an open a write (any one present => write mode). */
40#define BRIX_OPEN_WRITE_BITS \
41 (kXR_new | kXR_delete | kXR_open_updt | kXR_open_wrto | kXR_open_apnd)

43/* Is this kXR_open options word a write-mode open? */
44static inline int
45brix_open_options_is_write(uint16_t options)
46{
47 return (options & BRIX_OPEN_WRITE_BITS) ? 1 : 0;

50/*
51 * Build the kXR_open `options` field from a caller's write intent.
52 * write : 0 = read-only (kXR_open_read), nonzero = write
53 * force : the create disposition tri-state described in the file header
54 * posc : add kXR_posc (persist-on-successful-close) when nonzero
55 * mkpath: add kXR_mkpath (create parent dirs) when nonzero
56 * A write open always uses kXR_open_updt (read+write) so the server opens O_RDWR
57 * and partial writes can read-modify-write existing content.
58 */
59static inline uint16_t
60brix_open_options_build(int write, int force, int posc, int mkpath)
61{
62 uint16_t options;
63
64 if (!write) {
65 return (uint16_t) kXR_open_read;
66 }
67
68 options = (uint16_t) kXR_open_updt;
69 if (mkpath) {
70 options |= (uint16_t) kXR_mkpath;
71 }
72 if (force == 1) {
73 options |= (uint16_t) kXR_delete; /* truncate/overwrite */
74 } else if (force == 0) {
75 options |= (uint16_t) kXR_new; /* create-new, fail if exists */
76 }
77 /* force == 2: update in place — neither kXR_delete nor kXR_new. */
78 if (posc) {
79 options |= (uint16_t) kXR_posc;
80 }
81 return options;
82}

84/*
85 * Derive POSIX open(2) flags (and whether the fd is readable) from a kXR_open
86 * `options` word — the exact inverse of brix_open_options_build and the single
87 * definition of each open bit's POSIX meaning:
88 * read -> O_RDONLY (readable)
89 * kXR_open_updt -> O_RDWR (readable)
90 * kXR_open_apnd -> O_WRONLY | O_APPEND (write-only)
91 * else (wrto/...) -> O_WRONLY (write-only)
92 * kXR_new -> O_CREAT (+ O_EXCL unless kXR_delete is also set)
93 * kXR_delete -> O_CREAT | O_TRUNC
94 * O_NOCTTY is always added. `is_write` is the caller's already-computed write-mode
95 * flag (see brix_open_options_is_write). Either out-pointer may be NULL.
96 */
97static inline void
98brix_open_options_to_posix(uint16_t options, int is_write,
99 int *oflags, int *is_readable)
100{
101 int f, rd;
102
103 if (!is_write) {
104 f = O_RDONLY;
105 rd = 1;
106 } else {
107 if (options & kXR_open_updt) {
108 f = O_RDWR;
109 rd = 1;
110 } else if (options & kXR_open_apnd) {
111 f = O_WRONLY | O_APPEND;
112 rd = 0;
113 } else {
114 f = O_WRONLY;
115 rd = 0;
116 }
117
118 if (options & kXR_new) {
119 f |= O_CREAT;
120 if (!(options & kXR_delete)) {
121 f |= O_EXCL;
122 }
123 }
124 if (options & kXR_delete) {
125 f |= O_CREAT | O_TRUNC;
126 }
127 }
128
129 f |= O_NOCTTY;
130
131 if (oflags != NULL) {
132 *oflags = f;
133 }
134 if (is_readable != NULL) {
135 *is_readable = rd;
136 }
137}

139/*
140 * Map the POSIX open(2) flags a FUSE open()/create() receives onto the `force`
141 * tri-state, so both xrootdfs drivers derive it identically. For a writable
142 * open(): O_TRUNC means overwrite (1), otherwise update in place (2). For
143 * create(): O_EXCL means create-new (0), otherwise truncate-create (1).
144 */
145static inline int
146brix_open_force_for_open(int o_flags)
147{
148 return (o_flags & O_TRUNC) ? 1 : 2;
149}
150
151static inline int
152brix_open_force_for_create(int o_flags)
153{
154 return (o_flags & O_EXCL) ? 0 : 1;
155}
