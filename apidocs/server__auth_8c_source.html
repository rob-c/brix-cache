5/*
6 * server_auth.c — CMS registration authentication (Phase 28 W1).
7 *
8 * WHAT: Authentication/authorisation gates applied to a data node before it is
9 * allowed into the manager's server registry (and therefore before the
10 * manager will redirect clients to it). Three layered, individually
11 * vanilla-XRootD-compatible controls:
12 * W1a — real cmsd sss credential check (kYR_xauth handshake), reusing
13 * the shared SSS verifier in src/sss.
14 * W1b — accept-time CIDR allowlist of permitted data-node IPs.
15 * W1c — host-character validation at registration store time
16 * (brix_net_host_chars_valid, called from registry.c).
17 *
18 * WHY: Without these, any peer that can reach the CMS port can self-report an
19 * arbitrary host:port:paths and have the manager redirect clients to it
20 * (redirect poisoning — the critical finding). The controls are
21 * opt-in/compatible: a vanilla cmsd from a trusted subnet (W1b) carrying
22 * a valid cluster sss credential (W1a) is admitted unchanged.
23 */
24
25/* Credential freshness window for the cluster sss handshake. The credential
26 * is generated fresh by the data node at each login, so this only needs to
27 * exceed connect latency + clock skew; a generous hour avoids false rejects. */
28#define CMS_SSS_LIFETIME 3600
