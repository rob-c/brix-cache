15/*
16 * brix_pgread_run_t - per-request state threaded through the pgread steps.
17 *
18 * WHAT: The decoded request (handle index, fd, offset, capped length), the
19 * shared scratch buffer, and the produced output {out_buf, out_size,
20 * flat_buf} — filled by exactly one producer path (warm hit, AIO
21 * offload, or sync fallback).
22 *
23 * WHY: Makes the phase-72.A invariant structural: out_buf/flat_buf/out_size
24 * start NULL/0 (the handler zeroes the struct) and every producer sets
25 * all three through this one struct, so the pre-framing out_buf==NULL
26 * guard catches any path that failed to produce output.
27 */
28typedef struct {
29 int idx; /* file-handle table index */
30 int fd; /* resolved file descriptor */
31 int64_t offset; /* requested file offset */
32 size_t rlen; /* capped request length; sync path: bytes read */
33 u_char *scratch; /* gapped wire buffer (read_scratch slot) */
34 u_char *out_buf; /* encoded output start (NULL until produced) */
35 u_char *flat_buf; /* buffer to release after send (NULL until set) */
36 size_t out_size; /* encoded output bytes (0 until produced) */
37} brix_pgread_run_t;

39/*
40 * brix_pgread_parse_validate - decode the request and run all early checks.
41 *
42 * WHAT: Unpacks the kXR_pgread request into `run`, validates the handle,
43 * rejects negative offset/length, answers a zero-length read with an
44 * empty kXR_status frame, caps rlen, and resolves the fd. Returns 1 to
45 * continue; 0 when the request was fully handled (*rc holds the
46 * handler's return value — error sent, empty response queued, or
47 * NGX_ERROR).
48 *
49 * WHY: All reject/short-circuit paths in one early-return helper keeps the
50 * handler a flat orchestrator (coding-standards §8).
51 *
52 * HOW: Mirrors brix_validate_read_handle's continue-flag + *rc convention
53 * so the caller propagates the exact wire response codes unchanged.
54 */
55static ngx_int_t
56brix_pgread_parse_validate(brix_ctx_t *ctx, ngx_connection_t *c,
57 brix_pgread_run_t *run, ngx_int_t *rc)
58{
59 xrdw_pgread_req_t req;
60 ServerStatusResponse_pgRead *hdr_buf;
61
62 xrdw_pgread_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
63 run->idx = (int) (unsigned char) req.fhandle[0];
64 run->offset = req.offset;
65 run->rlen = (size_t) (uint32_t) req.rlen;
66
67 if (!brix_validate_read_handle(ctx, c, run->idx, "PGREAD",
68 BRIX_OP_PGREAD, rc)) {
69 return 0;
70 }
71
72 if (run->offset < 0) {
73 BRIX_OP_ERR(ctx, BRIX_OP_PGREAD);
74 *rc = brix_send_error(ctx, c, kXR_IOError,
75 "negative read offset");
76 return 0;
77 }
78
79 /* The wire rlen is a signed 32-bit field; a negative request length is
80 * invalid. Read unsigned it would turn -1 into ~4 GiB (then capped),
81 * silently succeeding where the reference rejects with kXR_ArgInvalid. */
82 if (req.rlen < 0) {
83 BRIX_OP_ERR(ctx, BRIX_OP_PGREAD);
84 *rc = brix_send_error(ctx, c, kXR_ArgInvalid,
85 "negative read length");
86 return 0;
87 }
88
89 if (run->rlen == 0) {
90 hdr_buf = ngx_palloc(c->pool, sizeof(*hdr_buf));
91 if (hdr_buf == NULL) {
92 *rc = NGX_ERROR;
93 return 0;
94 }
95 brix_build_pgread_status(ctx, run->offset, 0, hdr_buf);
96 BRIX_OP_OK(ctx, BRIX_OP_PGREAD);
97 *rc = brix_queue_response(ctx, c, (u_char *) hdr_buf,
98 sizeof(*hdr_buf));
99 return 0;
100 }
101
102 if (run->rlen > BRIX_READ_REQUEST_MAX) {
103 run->rlen = BRIX_READ_REQUEST_MAX;
104 }
105
106 run->fd = ctx->files[run->idx].fd;
107 return 1;
108}

110/*
111 * brix_pgread_alloc_scratch - size and fetch the gapped wire buffer.
112 *
113 * WHAT: Computes the worst-case page count for the request and returns the
114 * per-connection read scratch slot grown to hold the interleaved
115 * [CRC32c(4)][data] wire output (NULL on allocation failure).
116 *
117 * WHY: The buffer-size math is pure and self-contained; isolating it keeps
118 * the page-count subtlety (alignment split) documented in one place.
119 *
120 * HOW: File-offset alignment can split an otherwise single-page read across
121 * two pages (short first fragment + remainder), so the page count is
122 * derived from the in-page offset, not just rlen — otherwise the
123 * scratch/out region would be one CRC short.
124 */
125static u_char *
126brix_pgread_alloc_scratch(brix_ctx_t *ctx, ngx_connection_t *c,
127 brix_pgread_run_t *run)
128{
129 size_t n_pages_max;
130 size_t scratch_size;
131
132 n_pages_max = ((size_t) (run->offset & (kXR_pgPageSZ - 1)) + run->rlen
133 + kXR_pgPageSZ - 1) / kXR_pgPageSZ;
134 if (n_pages_max == 0) {
135 n_pages_max = 1;
136 }
137 /*
138 * Single buffer holding the final interleaved [CRC32c(4)][data] wire
139 * output (data is read straight into it — no separate flat copy region),
140 * so it needs only the data bytes plus one 4-byte CRC per page.
141 */
142 scratch_size = run->rlen + n_pages_max * BRIX_PG_CKSZ;
143
144 return BRIX_GET_SCRATCH(ctx, c, rd.read_scratch, rd.read_scratch_size,
145 scratch_size);
146}

148/*
149 * brix_pgread_try_warm - inline warm-cache fast path.
150 *
151 * WHAT: Attempts the whole read + in-place CRC on the event loop via
152 * preadv2(RWF_NOWAIT). On a full hit fills run->{out_buf, out_size,
153 * flat_buf}, charges the backend byte metric, and returns 1; on any
154 * miss returns 0 with the run output untouched.
155 *
156 * WHY: When the whole range is already page-cache resident, reading + CRCing
157 * inline skips the thread-pool handoff entirely — the handoff latency,
158 * not the copy, is the single-stream (n=1) cost. A miss (not resident /
159 * EOF / error) falls through to the blocking offload, which re-reads the
160 * full range. Only attempted with a pool configured (else the blocking
161 * path runs inline anyway) and for a regular file (RWF_NOWAIT is
162 * meaningful against the page cache). Mirrors the kXR_read Phase-32 probe.
163 *
164 * HOW: Delegates to brix_pgread_read_encode_inplace with nowait=1 against
165 * the handle's effective storage object; a hit means errno stayed 0 and
166 * every requested byte was delivered.
167 */
168static ngx_flag_t
169brix_pgread_try_warm(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *rconf,
170 brix_pgread_run_t *run)
171{
172 brix_pgread_io_t warm_io = { .nowait = 1, .nread = 0, .io_errno = 0 };
173 size_t warm_osz;
174 brix_sd_obj_t warm_scratch;
175 brix_sd_obj_t *warm_obj;
176
177 if (rconf->common.thread_pool == NULL
178 || !ctx->files[run->idx].is_regular)
179 {
180 return 0;
181 }
182
183 warm_obj = brix_vfs_effective_obj(&ctx->files[run->idx].sd_obj, run->fd,
184 &warm_scratch);
185
186 /* The RWF_NOWAIT probe needs the driver's native preadv2; drivers without
187 * one (remote/object backends) have no page cache to probe — treat as a
188 * miss so the read offloads to the blocking path. */
189 if (warm_obj->driver->preadv2 == NULL) {
190 return 0;
191 }
192
193 warm_osz = brix_pgread_read_encode_inplace(warm_obj, (off_t) run->offset,
194 run->rlen, run->scratch,
195 &warm_io);
196 if (warm_io.io_errno != 0 || warm_io.nread != (ssize_t) run->rlen) {
197 return 0;
198 }
199
200 run->out_size = warm_osz;
201 run->flat_buf = run->scratch;
202 run->out_buf = run->scratch; /* rlen already == bytes encoded */
203
204 /* The warm fast path bypasses brix_vfs_io_execute (where the
205 * cold pgread paths attribute), so charge the per-backend read
206 * total here for the file bytes just read. */
207 brix_metric_backend_bytes(
208 ctx->files[run->idx].sd_obj.driver != NULL
209 ? ctx->files[run->idx].sd_obj.driver->name : "posix",
210 BRIX_METRIC_OP_READ, (size_t) warm_io.nread);
211
212 return 1;
213}

215/*
216 * brix_pgread_post_aio - offload the read to the thread pool.
217 *
218 * WHAT: Fills the connection's reusable pgread AIO task from `run` and posts
219 * it to the configured thread pool. Returns NGX_OK when posted (the
220 * completion handler sends the response), NGX_DECLINED when the post
221 * failed (caller falls back to the sync path), NGX_ERROR on
222 * allocation failure.
223 *
224 * WHY: The task-population boilerplate is one nameable step of the handler;
225 * extracting it keeps the orchestrator flat (coding-standards §8).
226 *
227 * HOW: Reuses ctx->rd.pgread_aio_task across requests (allocating it once
228 * from the connection pool), binds the pgread worker/done pair, and
229 * lets brix_aio_post_task log the fallback warning on post failure.
230 */
231static ngx_int_t
232brix_pgread_post_aio(brix_ctx_t *ctx, ngx_connection_t *c,
233 ngx_stream_brix_srv_conf_t *rconf, brix_pgread_run_t *run)
234{
235 ngx_thread_task_t *task;
236 brix_pgread_aio_t *t;
237 ngx_flag_t posted;
238
239 task = ctx->rd.pgread_aio_task;
240 if (task == NULL) {
241 task = ngx_thread_task_alloc(c->pool,
242 sizeof(brix_pgread_aio_t));
243 if (task == NULL) {
244 return NGX_ERROR;
245 }
246 ctx->rd.pgread_aio_task = task;
247 } else {
248 task->next = NULL;
249 task->event.complete = 0;
250 }
251
252 t = task->ctx;
253 t->c = c;
254 t->ctx = ctx;
255 t->fd = run->fd;
256 t->handle_idx = run->idx;
257 t->offset = (off_t) run->offset;
258 t->rlen = run->rlen;
259 t->scratch = run->scratch;
260 t->out_size = 0;
261 t->streamid[0] = ctx->recv.cur_streamid[0];
262 t->streamid[1] = ctx->recv.cur_streamid[1];
263 t->obj = ctx->files[run->idx].sd_obj; /* Layer 3: driver obj (or zeroed) */
264
265 brix_task_bind(task, brix_pgread_aio_thread, brix_pgread_aio_done);
266
267 (void) brix_aio_post_task(ctx, c, rconf->common.thread_pool, task,
268 "brix: thread_task_post failed, sync pgread fallback",
269 &posted);
270
271 return posted ? NGX_OK : NGX_DECLINED;
272}

274/*
275 * brix_pgread_sync_fill - blocking fallback read into the wire buffer.
276 *
277 * WHAT: Runs the pgread VFS job inline (read directly into the gapped wire
278 * buffer, CRC each page in place — no flat-buffer copy). On success
279 * fills run->{out_buf, out_size, flat_buf}, updates run->rlen to the
280 * bytes actually read (accounting), and returns 0; on I/O error
281 * returns the job's errno for the caller's error triplet.
282 *
283 * WHY: Same code path as the AIO worker, kept as a pure produce-or-errno
284 * step so the handler owns the wire error response (side effects at
285 * the edges, coding-standards §8).
286 *
287 * HOW: Skipped by the caller when the warm-cache fast path already produced
288 * the encoding; runs when no thread pool is configured or the post
289 * failed.
290 */
291static int
292brix_pgread_sync_fill(brix_ctx_t *ctx, brix_pgread_run_t *run)
293{
294 brix_vfs_job_t job;
295
296 brix_vfs_job_read_init(&job, run->fd, (off_t) run->offset, run->rlen,
297 run->scratch, run->rlen, 0);
298 job.op = BRIX_VFS_IO_PGREAD;
299 brix_vfs_job_set_obj(&job, &ctx->files[run->idx].sd_obj);
300 brix_vfs_io_execute(&job);
301
302 if (job.io_errno != 0) {
303 return job.io_errno;
304 }
305
306 run->out_size = job.out_size;
307 run->flat_buf = run->scratch;
308 run->out_buf = run->scratch; /* output starts at offset 0 now */
309 run->rlen = (size_t) job.nio; /* actual bytes read (accounting) */
310 return 0;
311}

313/*
314 * brix_pgread_send_response - frame, account, log, and queue the reply.
315 *
316 * WHAT: Builds the kXR_status(4007) response chain over the produced output,
317 * charges the byte accounting / bandwidth limiter, writes the access
318 * log line, and queues the chain. Returns the queue rc (NGX_ERROR on
319 * framing failure), releasing the read buffer on any non-send outcome.
320 *
321 * WHY: Response assembly is the handler's final nameable step; the order of
322 * framing → accounting → log → metric → queue is frozen (byte-identical
323 * wire output and log lines).
324 *
325 * HOW: run->rlen here is the actual byte count (the sync path overwrote it
326 * with job.nio; the warm path read exactly rlen). The buffer is kept
327 * only while the send is in flight (state XRD_ST_SENDING).
328 */
329static ngx_int_t
330brix_pgread_send_response(brix_ctx_t *ctx, ngx_connection_t *c,
331 ngx_stream_brix_srv_conf_t *rconf, brix_pgread_run_t *run)
332{
333 ngx_chain_t *rsp_chain;
334 char detail[64];
335 ngx_int_t rc;
336
337 rsp_chain = brix_build_pgread_chain(ctx, c, run->offset, run->out_buf,
338 (uint32_t) run->out_size);
339 if (rsp_chain == NULL) {
340 brix_release_read_buffer(ctx, c, run->flat_buf);
341 return NGX_ERROR;
342 }
343
344 ctx->files[run->idx].bytes_read += run->rlen;
345 ctx->totals.bytes += run->rlen;
346 brix_rl_charge_ctx(ctx, run->rlen); /* Phase 25 bandwidth */
347
348 if (rconf->access_log_fd != NGX_INVALID_FILE) {
349 snprintf(detail, sizeof(detail), "%lld+%zu",
350 (long long) run->offset, run->rlen);
351 brix_log_access(ctx, c, "PGREAD", ctx->files[run->idx].path,
352 detail, 1, 0, NULL, run->rlen);
353 }
354 BRIX_OP_OK(ctx, BRIX_OP_PGREAD);
355
356 rc = brix_queue_response_chain(ctx, c, rsp_chain, run->flat_buf);
357 if (rc != NGX_OK || ctx->state != XRD_ST_SENDING) {
358 brix_release_read_buffer(ctx, c, run->flat_buf);
359 }
360 return rc;
361}

363/*
364 * brix_handle_pgread - kXR_pgread orchestrator.
365 *
366 * WHAT: Decodes and validates the request, sizes the wire buffer, then tries
367 * the producer paths in order — warm-cache inline, thread-pool offload,
368 * sync fallback — and frames whatever output was produced.
369 *
370 * WHY: Flat sequence of named steps per coding-standards §8; the complexity
371 * lives in the helpers above.
372 *
373 * HOW: run.{out_buf, flat_buf, out_size} start NULL/0 (phase-72.A) and are
374 * set only by a producer; the out_buf==NULL guard before framing makes
375 * the exactly-one-producer invariant enforceable.
376 */
377ngx_int_t
378brix_handle_pgread(brix_ctx_t *ctx, ngx_connection_t *c)
379{
380 /* phase-42 W4 invariant: pgread is ALWAYS plaintext — it never consults
381 * ctx->files[idx].read_codec. Inline read compression is a kXR_read-only
382 * handle property; pgread's kXR_status(4007) framing + per-page CRC32c must
383 * stay byte-for-byte intact, so compression is deliberately not applied here. */
384 brix_pgread_run_t run;
385 ngx_stream_brix_srv_conf_t *rconf;
386 ngx_int_t rc;
387 ngx_flag_t warm_hit;
388
389 ngx_memzero(&run, sizeof(run)); /* out_buf/flat_buf NULL, out_size 0 */
390
391 if (!brix_pgread_parse_validate(ctx, c, &run, &rc)) {
392 return rc;
393 }
394
395 rconf = ngx_stream_get_module_srv_conf(
396 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
397
398 run.scratch = brix_pgread_alloc_scratch(ctx, c, &run);
399 if (run.scratch == NULL) {
400 return NGX_ERROR;
401 }
402
403 warm_hit = brix_pgread_try_warm(ctx, rconf, &run);
404
405 if (!warm_hit && rconf->common.thread_pool != NULL) {
406 rc = brix_pgread_post_aio(ctx, c, rconf, &run);
407 if (rc == NGX_ERROR) {
408 return NGX_ERROR;
409 }
410 if (rc == NGX_OK) {
411 return NGX_OK; /* posted; the AIO done handler responds */
412 }
413 /* NGX_DECLINED: post failed — fall through to the sync path. */
414 }
415
416 if (!warm_hit) {
417 int io_errno = brix_pgread_sync_fill(ctx, &run);
418
419 if (io_errno != 0) {
420 BRIX_RETURN_ERR(ctx, c, BRIX_OP_PGREAD, "PGREAD",
421 ctx->files[run.idx].path, "-",
422 kXR_IOError, strerror(io_errno));
423 }
424 }
425
426 /* Invariant: exactly one of the producer paths (warm hit or sync
427 * fallback) must have filled the output; the AIO path returned above. */
428 if (run.out_buf == NULL) {
429 BRIX_RETURN_ERR(ctx, c, BRIX_OP_PGREAD, "PGREAD",
430 ctx->files[run.idx].path, "-",
431 kXR_ServerError, "pgread: no output produced");
432 }
433
434 return brix_pgread_send_response(ctx, c, rconf, &run);
435}
