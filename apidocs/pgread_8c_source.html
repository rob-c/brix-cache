15/* CRC32c word size per page unit ([CRC32c(4)][data]); == kXR_pgUnitSZ - page. */
16#define BRIX_PG_CKSZ ((size_t) (kXR_pgUnitSZ - kXR_pgPageSZ))
17/* preadv scatter cap per syscall — mirrors kXR_readv (BRIX_READV_PREADV_MAXIOV). */

20/*
21 * brix_pgread_batch_t - one preadv scatter batch of gapped wire pages.
22 *
23 * WHAT: The per-batch page layout for the zero-copy paged read: the iovecs a
24 * single preadv/preadv2 call scatters into, plus each page's data
25 * pointer (just past its 4-byte CRC gap) and length, the page count,
26 * and the total file bytes the batch covers.
27 *
28 * WHY: Groups the parallel arrays the layout / read / CRC steps of
29 * brix_pgread_read_encode_inplace share, so each step can be a small
30 * single-purpose helper instead of one monolithic loop body.
31 *
32 * HOW: Filled by brix_pgread_layout_batch, read into by the driver's
33 * preadv/preadv2, checksummed by brix_pgread_crc_batch.
34 */
35typedef struct {
36 struct iovec iov[BRIX_PGREAD_MAXIOV]; /* scatter targets (page data) */
37 u_char *data[BRIX_PGREAD_MAXIOV]; /* page data start (after CRC) */
38 size_t dlen[BRIX_PGREAD_MAXIOV]; /* page data length */
39 int k; /* pages in this batch */
40 size_t bytes; /* total file bytes laid out */
41} brix_pgread_batch_t;
42
43/*
44 * brix_pgread_layout_batch - lay out one batch of gapped wire pages.
45 *
46 * WHAT: Fills `b` with up to BRIX_PGREAD_MAXIOV pages covering the next
47 * `remaining` file bytes of an rlen-byte read starting at `offset`,
48 * placing each page's data just after its 4-byte CRC gap in the wire
49 * buffer at cursor `o`. Returns the advanced wire-buffer cursor.
50 *
51 * WHY: Pure gap-layout math, separated from the read syscall and the CRC
52 * pass so each stays independently reviewable (phase-72 decomposition).
53 *
54 * HOW: Page lengths use the in-page offset so the first fragment shortens on
55 * an unaligned read, exactly as xrdp_pg_encode does; the caller derives
56 * the batch's file offset from (rlen - remaining) the same way.
57 */
58static u_char *
59brix_pgread_layout_batch(off_t offset, size_t rlen, size_t remaining,
60 u_char *o, brix_pgread_batch_t *b)
61{
62 b->k = 0;
63 b->bytes = 0;
64
65 while (b->k < BRIX_PGREAD_MAXIOV && remaining > 0) {
66 off_t cur = offset + (off_t) (rlen - remaining);
67 size_t in_off = (size_t) (cur & (off_t) (kXR_pgPageSZ - 1));
68 size_t len = (size_t) kXR_pgPageSZ - in_off;
69
70 if (len > remaining) {
71 len = remaining;
72 }
73 b->data[b->k] = o + BRIX_PG_CKSZ;
74 b->dlen[b->k] = len;
75 b->iov[b->k].iov_base = b->data[b->k];
76 b->iov[b->k].iov_len = len;
77 o += BRIX_PG_CKSZ + len;
78 b->bytes += len;
79 remaining -= len;
80 b->k++;
81 }
82
83 return o;
84}

86/*
87 * brix_pgread_crc_batch - checksum the bytes one batch actually delivered.
88 *
89 * WHAT: Fuses the per-page CRC32c over exactly `got` bytes of batch `b`,
90 * writing each 4-byte big-endian checksum into the gap preceding its
91 * page data and adding the encoded bytes to *out_size. Returns 1 when
92 * a short page ends the file (EOF), 0 otherwise.
93 *
94 * WHY: The CRC pass over delivered bytes is pure computation over the batch
95 * layout; splitting it from the read loop keeps the encode function
96 * under the complexity gate without touching the wire bytes.
97 *
98 * HOW: Walks the batch pages until the delivered bytes run out; a page that
99 * comes up short means EOF, so stop — no more data follows.
100 */
101static int
102brix_pgread_crc_batch(brix_pgread_batch_t *b, size_t got, size_t *out_size)
103{
104 int i;
105
106 for (i = 0; i < b->k; i++) {
107 size_t al = (got < b->dlen[i]) ? got : b->dlen[i];
108 uint32_t crc_be;
109
110 if (al == 0) {
111 break;
112 }
113 crc_be = htonl(brix_crc32c_value(b->data[i], al));
114 memcpy(b->data[i] - BRIX_PG_CKSZ, &crc_be, BRIX_PG_CKSZ);
115 *out_size += BRIX_PG_CKSZ + al;
116 got -= al;
117 if (al < b->dlen[i]) {
118 return 1; /* short page => short read; no more data follows */
119 }
120 }
121
122 return 0;
123}

125/*
126 * brix_pgread_read_encode_inplace - zero-copy paged read + in-place CRC.
127 *
128 * WHAT: Reads up to `rlen` bytes from `fd` at `offset` DIRECTLY into the final
129 * kXR page-mode wire buffer `out` (laid out as [CRC32c(4)][data] per page,
130 * file-offset aligned) and computes each page's CRC32c in place, writing
131 * the 4-byte big-endian checksum into the gap that precedes its data.
132 * Returns the encoded byte count; sets io->nread to bytes read (-1 on I/O
133 * error, with io->io_errno = errno). io->nowait selects the read mode.
134 *
135 * WHY: The previous path pread() the data into a flat buffer and then ran
136 * xrdp_pg_encode to COPY it into the interleaved wire buffer while
137 * checksumming. Flame-graph profiling showed that copy (a full extra pass
138 * over every byte — the dst-write memory stream) dominating read CPU, and a
139 * copy is memory-bandwidth-bound so the 3-way CRC barely helps it. Reading
140 * straight into the gapped wire buffer removes the copy entirely; the CRC
141 * then runs read-only (brix_crc32c_value, the latency-hiding 3-way path)
142 * over the data already in place. This mirrors the zero-copy preadv-into-
143 * final-buffer pattern kXR_readv already uses. Output is byte-identical to
144 * xrdp_pg_encode (Invariant #1): same page splitting, same CRC, same layout.
145 *
146 * HOW: Lay out one batch of <= BRIX_PGREAD_MAXIOV pages (data after each 4-byte
147 * CRC gap), preadv the batch's contiguous file region into those gapped
148 * positions, then fuse the per-page CRC over exactly the bytes the batch
149 * delivered — a short page or short batch means EOF, so stop. Page lengths
150 * use the in-page offset so the first fragment shortens on an unaligned
151 * read, exactly as xrdp_pg_encode does.
152 */
153size_t
154brix_pgread_read_encode_inplace(brix_sd_obj_t *obj, off_t offset,
155 size_t rlen, u_char *out, brix_pgread_io_t *io)
156{
157 u_char *o = out; /* write cursor in the gapped wire buffer */
158 size_t remaining = rlen; /* file bytes not yet laid out into a batch */
159 size_t out_size = 0; /* encoded bytes produced so far */
160 ssize_t total = 0; /* file bytes actually read */
161 int eof = 0;
162
163 io->nread = 0;
164 io->io_errno = 0;
165
166 /* The batched vectored read goes through the handle's storage driver (POSIX
167 * or block-striped); the page layout / in-place CRC policy stays here. */
168
169 while (remaining > 0 && !eof) {
170 brix_pgread_batch_t batch;
171 off_t batch_off = offset + (off_t) (rlen - remaining);
172 ssize_t n;
173
174 /* Lay out a batch of pages: data lands after each 4-byte CRC gap. */
175 o = brix_pgread_layout_batch(offset, rlen, remaining, o, &batch);
176 remaining -= batch.bytes;
177
178#if defined(RWF_NOWAIT)
179 if (io->nowait) {
180 /* Warm-cache probe: read only what is already resident. Any short
181 * batch or EAGAIN means "not fully in page cache" — abort the whole
182 * inline attempt so the caller offloads a blocking read (which also
183 * re-detects true EOF correctly). A real error likewise aborts; the
184 * blocking re-read surfaces it. Earlier full batches' work is
185 * discarded by that re-read. */
186 n = obj->driver->preadv2(obj, batch.iov, batch.k, batch_off,
187 RWF_NOWAIT);
188 if (n < 0 || (size_t) n < batch.bytes) {
189 io->nread = 0;
190 io->io_errno = (n < 0) ? errno : EAGAIN;
191 return 0;
192 }
193 } else
194#endif
195 {
196 /* Compat seam: drivers without a native preadv slot (remote/object
197 * backends) fall back to per-iovec pread inside the helper. */
198 n = brix_sd_obj_preadv(obj, batch.iov, batch.k, batch_off);
199 if (n < 0) {
200 io->nread = -1;
201 io->io_errno = errno;
202 return 0;
203 }
204 }
205 total += n;
206
207 /* Fuse the per-page CRC over exactly the bytes this batch delivered. */
208 if (brix_pgread_crc_batch(&batch, (size_t) n, &out_size)) {
209 eof = 1;
210 }
211
212 if ((size_t) n < batch.bytes) {
213 eof = 1; /* short batch overall => EOF */
214 }
215 }
216
217 io->nread = total;
218 return out_size;
219}

259/*
260 * brix_pgread_parse_validate - decode the request and run all early checks.
261 *
262 * WHAT: Unpacks the kXR_pgread request into `run`, validates the handle,
263 * rejects negative offset/length, answers a zero-length read with an
264 * empty kXR_status frame, caps rlen, and resolves the fd. Returns 1 to
265 * continue; 0 when the request was fully handled (*rc holds the
266 * handler's return value — error sent, empty response queued, or
267 * NGX_ERROR).
268 *
269 * WHY: All reject/short-circuit paths in one early-return helper keeps the
270 * handler a flat orchestrator (coding-standards §8).
271 *
272 * HOW: Mirrors brix_validate_read_handle's continue-flag + *rc convention
273 * so the caller propagates the exact wire response codes unchanged.
274 */
275static ngx_int_t
276brix_pgread_parse_validate(brix_ctx_t *ctx, ngx_connection_t *c,
277 brix_pgread_run_t *run, ngx_int_t *rc)
278{
279 xrdw_pgread_req_t req;
280 ServerStatusResponse_pgRead *hdr_buf;
281
282 xrdw_pgread_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
283 run->idx = (int) (unsigned char) req.fhandle[0];
284 run->offset = req.offset;
285 run->rlen = (size_t) (uint32_t) req.rlen;
286
287 if (!brix_validate_read_handle(ctx, c, run->idx, "PGREAD",
288 BRIX_OP_PGREAD, rc)) {
289 return 0;
290 }
291
292 if (run->offset < 0) {
293 BRIX_OP_ERR(ctx, BRIX_OP_PGREAD);
294 *rc = brix_send_error(ctx, c, kXR_IOError,
295 "negative read offset");
296 return 0;
297 }
298
299 /* The wire rlen is a signed 32-bit field; a negative request length is
300 * invalid. Read unsigned it would turn -1 into ~4 GiB (then capped),
301 * silently succeeding where the reference rejects with kXR_ArgInvalid. */
302 if (req.rlen < 0) {
303 BRIX_OP_ERR(ctx, BRIX_OP_PGREAD);
304 *rc = brix_send_error(ctx, c, kXR_ArgInvalid,
305 "negative read length");
306 return 0;
307 }
308
309 if (run->rlen == 0) {
310 hdr_buf = ngx_palloc(c->pool, sizeof(*hdr_buf));
311 if (hdr_buf == NULL) {
312 *rc = NGX_ERROR;
313 return 0;
314 }
315 brix_build_pgread_status(ctx, run->offset, 0, hdr_buf);
316 BRIX_OP_OK(ctx, BRIX_OP_PGREAD);
317 *rc = brix_queue_response(ctx, c, (u_char *) hdr_buf,
318 sizeof(*hdr_buf));
319 return 0;
320 }
321
322 if (run->rlen > BRIX_READ_REQUEST_MAX) {
323 run->rlen = BRIX_READ_REQUEST_MAX;
324 }
325
326 run->fd = ctx->files[run->idx].fd;
327 return 1;
328}

330/*
331 * brix_pgread_alloc_scratch - size and fetch the gapped wire buffer.
332 *
333 * WHAT: Computes the worst-case page count for the request and returns the
334 * per-connection read scratch slot grown to hold the interleaved
335 * [CRC32c(4)][data] wire output (NULL on allocation failure).
336 *
337 * WHY: The buffer-size math is pure and self-contained; isolating it keeps
338 * the page-count subtlety (alignment split) documented in one place.
339 *
340 * HOW: File-offset alignment can split an otherwise single-page read across
341 * two pages (short first fragment + remainder), so the page count is
342 * derived from the in-page offset, not just rlen — otherwise the
343 * scratch/out region would be one CRC short.
344 */
345static u_char *
346brix_pgread_alloc_scratch(brix_ctx_t *ctx, ngx_connection_t *c,
347 brix_pgread_run_t *run)
348{
349 size_t n_pages_max;
350 size_t scratch_size;
351
352 n_pages_max = ((size_t) (run->offset & (kXR_pgPageSZ - 1)) + run->rlen
353 + kXR_pgPageSZ - 1) / kXR_pgPageSZ;
354 if (n_pages_max == 0) {
355 n_pages_max = 1;
356 }
357 /*
358 * Single buffer holding the final interleaved [CRC32c(4)][data] wire
359 * output (data is read straight into it — no separate flat copy region),
360 * so it needs only the data bytes plus one 4-byte CRC per page.
361 */
362 scratch_size = run->rlen + n_pages_max * BRIX_PG_CKSZ;
363
364 return BRIX_GET_SCRATCH(ctx, c, rd.read_scratch, rd.read_scratch_size,
365 scratch_size);
366}

368/*
369 * brix_pgread_try_warm - inline warm-cache fast path.
370 *
371 * WHAT: Attempts the whole read + in-place CRC on the event loop via
372 * preadv2(RWF_NOWAIT). On a full hit fills run->{out_buf, out_size,
373 * flat_buf}, charges the backend byte metric, and returns 1; on any
374 * miss returns 0 with the run output untouched.
375 *
376 * WHY: When the whole range is already page-cache resident, reading + CRCing
377 * inline skips the thread-pool handoff entirely — the handoff latency,
378 * not the copy, is the single-stream (n=1) cost. A miss (not resident /
379 * EOF / error) falls through to the blocking offload, which re-reads the
380 * full range. Only attempted with a pool configured (else the blocking
381 * path runs inline anyway) and for a regular file (RWF_NOWAIT is
382 * meaningful against the page cache). Mirrors the kXR_read Phase-32 probe.
383 *
384 * HOW: Delegates to brix_pgread_read_encode_inplace with nowait=1 against
385 * the handle's effective storage object; a hit means errno stayed 0 and
386 * every requested byte was delivered.
387 */
388static ngx_flag_t
389brix_pgread_try_warm(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *rconf,
390 brix_pgread_run_t *run)
391{
392 brix_pgread_io_t warm_io = { .nowait = 1, .nread = 0, .io_errno = 0 };
393 size_t warm_osz;
394 brix_sd_obj_t warm_scratch;
395 brix_sd_obj_t *warm_obj;
396
397 if (rconf->common.thread_pool == NULL
398 || !ctx->files[run->idx].is_regular)
399 {
400 return 0;
401 }
402
403 warm_obj = brix_vfs_effective_obj(&ctx->files[run->idx].sd_obj, run->fd,
404 &warm_scratch);
405
406 /* The RWF_NOWAIT probe needs the driver's native preadv2; drivers without
407 * one (remote/object backends) have no page cache to probe — treat as a
408 * miss so the read offloads to the blocking path. */
409 if (warm_obj->driver->preadv2 == NULL) {
410 return 0;
411 }
412
413 warm_osz = brix_pgread_read_encode_inplace(warm_obj, (off_t) run->offset,
414 run->rlen, run->scratch,
415 &warm_io);
416 if (warm_io.io_errno != 0 || warm_io.nread != (ssize_t) run->rlen) {
417 return 0;
418 }
419
420 run->out_size = warm_osz;
421 run->flat_buf = run->scratch;
422 run->out_buf = run->scratch; /* rlen already == bytes encoded */
423
424 /* The warm fast path bypasses brix_vfs_io_execute (where the
425 * cold pgread paths attribute), so charge the per-backend read
426 * total here for the file bytes just read. */
427 brix_metric_backend_bytes(
428 ctx->files[run->idx].sd_obj.driver != NULL
429 ? ctx->files[run->idx].sd_obj.driver->name : "posix",
430 BRIX_METRIC_OP_READ, (size_t) warm_io.nread);
431
432 return 1;
433}

435/*
436 * brix_pgread_post_aio - offload the read to the thread pool.
437 *
438 * WHAT: Fills the connection's reusable pgread AIO task from `run` and posts
439 * it to the configured thread pool. Returns NGX_OK when posted (the
440 * completion handler sends the response), NGX_DECLINED when the post
441 * failed (caller falls back to the sync path), NGX_ERROR on
442 * allocation failure.
443 *
444 * WHY: The task-population boilerplate is one nameable step of the handler;
445 * extracting it keeps the orchestrator flat (coding-standards §8).
446 *
447 * HOW: Reuses ctx->rd.pgread_aio_task across requests (allocating it once
448 * from the connection pool), binds the pgread worker/done pair, and
449 * lets brix_aio_post_task log the fallback warning on post failure.
450 */
451static ngx_int_t
452brix_pgread_post_aio(brix_ctx_t *ctx, ngx_connection_t *c,
453 ngx_stream_brix_srv_conf_t *rconf, brix_pgread_run_t *run)
454{
455 ngx_thread_task_t *task;
456 brix_pgread_aio_t *t;
457 ngx_flag_t posted;
458
459 task = ctx->rd.pgread_aio_task;
460 if (task == NULL) {
461 task = ngx_thread_task_alloc(c->pool,
462 sizeof(brix_pgread_aio_t));
463 if (task == NULL) {
464 return NGX_ERROR;
465 }
466 ctx->rd.pgread_aio_task = task;
467 } else {
468 task->next = NULL;
469 task->event.complete = 0;
470 }
471
472 t = task->ctx;
473 t->c = c;
474 t->ctx = ctx;
475 t->fd = run->fd;
476 t->handle_idx = run->idx;
477 t->offset = (off_t) run->offset;
478 t->rlen = run->rlen;
479 t->scratch = run->scratch;
480 t->out_size = 0;
481 t->streamid[0] = ctx->recv.cur_streamid[0];
482 t->streamid[1] = ctx->recv.cur_streamid[1];
483 t->obj = ctx->files[run->idx].sd_obj; /* Layer 3: driver obj (or zeroed) */
484
485 brix_task_bind(task, brix_pgread_aio_thread, brix_pgread_aio_done);
486
487 (void) brix_aio_post_task(ctx, c, rconf->common.thread_pool, task,
488 "brix: thread_task_post failed, sync pgread fallback",
489 &posted);
490
491 return posted ? NGX_OK : NGX_DECLINED;
492}

494/*
495 * brix_pgread_sync_fill - blocking fallback read into the wire buffer.
496 *
497 * WHAT: Runs the pgread VFS job inline (read directly into the gapped wire
498 * buffer, CRC each page in place — no flat-buffer copy). On success
499 * fills run->{out_buf, out_size, flat_buf}, updates run->rlen to the
500 * bytes actually read (accounting), and returns 0; on I/O error
501 * returns the job's errno for the caller's error triplet.
502 *
503 * WHY: Same code path as the AIO worker, kept as a pure produce-or-errno
504 * step so the handler owns the wire error response (side effects at
505 * the edges, coding-standards §8).
506 *
507 * HOW: Skipped by the caller when the warm-cache fast path already produced
508 * the encoding; runs when no thread pool is configured or the post
509 * failed.
510 */
511static int
512brix_pgread_sync_fill(brix_ctx_t *ctx, brix_pgread_run_t *run)
513{
514 brix_vfs_job_t job;
515
516 brix_vfs_job_read_init(&job, run->fd, (off_t) run->offset, run->rlen,
517 run->scratch, run->rlen, 0);
518 job.op = BRIX_VFS_IO_PGREAD;
519 brix_vfs_job_set_obj(&job, &ctx->files[run->idx].sd_obj);
520 brix_vfs_io_execute(&job);
521
522 if (job.io_errno != 0) {
523 return job.io_errno;
524 }
525
526 run->out_size = job.out_size;
527 run->flat_buf = run->scratch;
528 run->out_buf = run->scratch; /* output starts at offset 0 now */
529 run->rlen = (size_t) job.nio; /* actual bytes read (accounting) */
530 return 0;
531}

533/*
534 * brix_pgread_send_response - frame, account, log, and queue the reply.
535 *
536 * WHAT: Builds the kXR_status(4007) response chain over the produced output,
537 * charges the byte accounting / bandwidth limiter, writes the access
538 * log line, and queues the chain. Returns the queue rc (NGX_ERROR on
539 * framing failure), releasing the read buffer on any non-send outcome.
540 *
541 * WHY: Response assembly is the handler's final nameable step; the order of
542 * framing → accounting → log → metric → queue is frozen (byte-identical
543 * wire output and log lines).
544 *
545 * HOW: run->rlen here is the actual byte count (the sync path overwrote it
546 * with job.nio; the warm path read exactly rlen). The buffer is kept
547 * only while the send is in flight (state XRD_ST_SENDING).
548 */
549static ngx_int_t
550brix_pgread_send_response(brix_ctx_t *ctx, ngx_connection_t *c,
551 ngx_stream_brix_srv_conf_t *rconf, brix_pgread_run_t *run)
552{
553 ngx_chain_t *rsp_chain;
554 char detail[64];
555 ngx_int_t rc;
556
557 rsp_chain = brix_build_pgread_chain(ctx, c, run->offset, run->out_buf,
558 (uint32_t) run->out_size);
559 if (rsp_chain == NULL) {
560 brix_release_read_buffer(ctx, c, run->flat_buf);
561 return NGX_ERROR;
562 }
563
564 ctx->files[run->idx].bytes_read += run->rlen;
565 ctx->totals.bytes += run->rlen;
566 brix_rl_charge_ctx(ctx, run->rlen); /* Phase 25 bandwidth */
567
568 if (rconf->access_log_fd != NGX_INVALID_FILE) {

583/*
584 * brix_handle_pgread - kXR_pgread orchestrator.
585 *
586 * WHAT: Decodes and validates the request, sizes the wire buffer, then tries
587 * the producer paths in order — warm-cache inline, thread-pool offload,
588 * sync fallback — and frames whatever output was produced.
589 *
590 * WHY: Flat sequence of named steps per coding-standards §8; the complexity
591 * lives in the helpers above.
592 *
593 * HOW: run.{out_buf, flat_buf, out_size} start NULL/0 (phase-72.A) and are
594 * set only by a producer; the out_buf==NULL guard before framing makes
595 * the exactly-one-producer invariant enforceable.
596 */
597ngx_int_t
598brix_handle_pgread(brix_ctx_t *ctx, ngx_connection_t *c)
599{
600 /* phase-42 W4 invariant: pgread is ALWAYS plaintext — it never consults
601 * ctx->files[idx].read_codec. Inline read compression is a kXR_read-only
602 * handle property; pgread's kXR_status(4007) framing + per-page CRC32c must
603 * stay byte-for-byte intact, so compression is deliberately not applied here. */
604 brix_pgread_run_t run;
605 ngx_stream_brix_srv_conf_t *rconf;
606 ngx_int_t rc;
607 ngx_flag_t warm_hit;
608
609 ngx_memzero(&run, sizeof(run)); /* out_buf/flat_buf NULL, out_size 0 */
610
611 if (!brix_pgread_parse_validate(ctx, c, &run, &rc)) {
612 return rc;
613 }
614
615 rconf = ngx_stream_get_module_srv_conf(
616 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
617
618 run.scratch = brix_pgread_alloc_scratch(ctx, c, &run);
619 if (run.scratch == NULL) {
620 return NGX_ERROR;
621 }
622
623 warm_hit = brix_pgread_try_warm(ctx, rconf, &run);
624
625 if (!warm_hit && rconf->common.thread_pool != NULL) {
626 rc = brix_pgread_post_aio(ctx, c, rconf, &run);
627 if (rc == NGX_ERROR) {
628 return NGX_ERROR;
629 }
630 if (rc == NGX_OK) {
631 return NGX_OK; /* posted; the AIO done handler responds */
632 }
633 /* NGX_DECLINED: post failed — fall through to the sync path. */
634 }
635
636 if (!warm_hit) {
637 int io_errno = brix_pgread_sync_fill(ctx, &run);
638
639 if (io_errno != 0) {
640 BRIX_RETURN_ERR(ctx, c, BRIX_OP_PGREAD, "PGREAD",
641 ctx->files[run.idx].path, "-",
642 kXR_IOError, strerror(io_errno));
643 }
644 }
645
646 /* Invariant: exactly one of the producer paths (warm hit or sync
647 * fallback) must have filled the output; the AIO path returned above. */
648 if (run.out_buf == NULL) {
649 BRIX_RETURN_ERR(ctx, c, BRIX_OP_PGREAD, "PGREAD",
650 ctx->files[run.idx].path, "-",
651 kXR_ServerError, "pgread: no output produced");
652 }
653
654 return brix_pgread_send_response(ctx, c, rconf, &run);
655}
