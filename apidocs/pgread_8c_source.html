20/*
21 * brix_pgread_read_encode_inplace - zero-copy paged read + in-place CRC.
22 *
23 * WHAT: Reads up to `rlen` bytes from `fd` at `offset` DIRECTLY into the final
24 * kXR page-mode wire buffer `out` (laid out as [CRC32c(4)][data] per page,
25 * file-offset aligned) and computes each page's CRC32c in place, writing
26 * the 4-byte big-endian checksum into the gap that precedes its data.
27 * Returns the encoded byte count; sets *nread_out to bytes read (-1 on I/O
28 * error, with *io_errno_out = errno).
29 *
30 * WHY: The previous path pread() the data into a flat buffer and then ran
31 * xrdp_pg_encode to COPY it into the interleaved wire buffer while
32 * checksumming. Flame-graph profiling showed that copy (a full extra pass
33 * over every byte — the dst-write memory stream) dominating read CPU, and a
34 * copy is memory-bandwidth-bound so the 3-way CRC barely helps it. Reading
35 * straight into the gapped wire buffer removes the copy entirely; the CRC
36 * then runs read-only (brix_crc32c_value, the latency-hiding 3-way path)
37 * over the data already in place. This mirrors the zero-copy preadv-into-
38 * final-buffer pattern kXR_readv already uses. Output is byte-identical to
39 * xrdp_pg_encode (Invariant #1): same page splitting, same CRC, same layout.
40 *
41 * HOW: Lay out one batch of <= BRIX_PGREAD_MAXIOV pages (data after each 4-byte
42 * CRC gap), preadv the batch's contiguous file region into those gapped
43 * positions, then fuse the per-page CRC over exactly the bytes the batch
44 * delivered — a short page or short batch means EOF, so stop. Page lengths
45 * use the in-page offset so the first fragment shortens on an unaligned
46 * read, exactly as xrdp_pg_encode does.
47 */
48size_t
49brix_pgread_read_encode_inplace(brix_sd_obj_t *obj, off_t offset,
50 size_t rlen, u_char *out, ssize_t *nread_out, int *io_errno_out, int nowait)
51{
52 u_char *o = out; /* write cursor in the gapped wire buffer */
53 size_t remaining = rlen; /* file bytes not yet laid out into a batch */
54 size_t out_size = 0; /* encoded bytes produced so far */
55 ssize_t total = 0; /* file bytes actually read */
56 int eof = 0;
57
58 *nread_out = 0;
59 *io_errno_out = 0;
60
61 /* The batched vectored read goes through the handle's storage driver (POSIX
62 * or block-striped); the page layout / in-place CRC policy stays here. */
63
64 while (remaining > 0 && !eof) {
65 struct iovec iov[BRIX_PGREAD_MAXIOV];
66 u_char *data[BRIX_PGREAD_MAXIOV];
67 size_t dlen[BRIX_PGREAD_MAXIOV];
68 off_t batch_off = offset + (off_t) (rlen - remaining);
69 size_t batch_bytes = 0;
70 ssize_t n;
71 size_t got;
72 int k = 0;
73 int i;
74
75 /* Lay out a batch of pages: data lands after each 4-byte CRC gap. */
76 while (k < BRIX_PGREAD_MAXIOV && remaining > 0) {
77 off_t cur = offset + (off_t) (rlen - remaining);
78 size_t in_off = (size_t) (cur & (off_t) (kXR_pgPageSZ - 1));
79 size_t len = (size_t) kXR_pgPageSZ - in_off;
80
81 if (len > remaining) {
82 len = remaining;
83 }
84 data[k] = o + BRIX_PG_CKSZ;
85 dlen[k] = len;
86 iov[k].iov_base = data[k];
87 iov[k].iov_len = len;
88 o += BRIX_PG_CKSZ + len;
89 batch_bytes += len;
90 remaining -= len;
91 k++;
92 }
93
94#if defined(RWF_NOWAIT)
95 if (nowait) {
96 /* Warm-cache probe: read only what is already resident. Any short
97 * batch or EAGAIN means "not fully in page cache" — abort the whole
98 * inline attempt so the caller offloads a blocking read (which also
99 * re-detects true EOF correctly). A real error likewise aborts; the
100 * blocking re-read surfaces it. Earlier full batches' work is
101 * discarded by that re-read. */
102 n = obj->driver->preadv2(obj, iov, k, batch_off,
103 RWF_NOWAIT);
104 if (n < 0 || (size_t) n < batch_bytes) {
105 *nread_out = 0;
106 *io_errno_out = (n < 0) ? errno : EAGAIN;
107 return 0;
108 }
109 } else
110#endif
111 {
112 n = obj->driver->preadv(obj, iov, k, batch_off);
113 if (n < 0) {
114 *nread_out = -1;
115 *io_errno_out = errno;
116 return 0;
117 }
118 }
119 total += n;
120
121 /* Fuse the per-page CRC over exactly the bytes this batch delivered. */
122 got = (size_t) n;
123 for (i = 0; i < k; i++) {
124 size_t al = (got < dlen[i]) ? got : dlen[i];
125 uint32_t crc_be;
126
127 if (al == 0) {
128 break;
129 }
130 crc_be = htonl(brix_crc32c_value(data[i], al));
131 memcpy(data[i] - BRIX_PG_CKSZ, &crc_be, BRIX_PG_CKSZ);
132 out_size += BRIX_PG_CKSZ + al;
133 got -= al;
134 if (al < dlen[i]) {
135 eof = 1; /* short page => short read; no more data follows */
136 break;
137 }
138 }
139
140 if ((size_t) n < batch_bytes) {
141 eof = 1; /* short batch overall => EOF */
142 }
143 }
144
145 *nread_out = total;
146 return out_size;
147}

163ngx_int_t
164brix_handle_pgread(brix_ctx_t *ctx, ngx_connection_t *c)
165{
166 /* phase-42 W4 invariant: pgread is ALWAYS plaintext — it never consults
167 * ctx->files[idx].read_codec. Inline read compression is a kXR_read-only
168 * handle property; pgread's kXR_status(4007) framing + per-page CRC32c must
169 * stay byte-for-byte intact, so compression is deliberately not applied here. */
170 xrdw_pgread_req_t req;
171 int idx;
172 int64_t offset;
173 size_t rlen;
174 int fd;
175 u_char *flat_buf;
176 u_char *out_buf;
177 size_t out_size;
178 ServerStatusResponse_pgRead *hdr_buf;
179 ngx_chain_t *rsp_chain;
180 ngx_stream_brix_srv_conf_t *rconf;
181 char detail[64];
182 ngx_int_t validate_rc;
183
184 xrdw_pgread_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
185 idx = (int) (unsigned char) req.fhandle[0];
186 offset = req.offset;
187 rlen = (size_t) (uint32_t) req.rlen;
188
189 if (!brix_validate_read_handle(ctx, c, idx, "PGREAD",
190 BRIX_OP_PGREAD, &validate_rc)) {
191 return validate_rc;
192 }
193
194 if (offset < 0) {
195 BRIX_OP_ERR(ctx, BRIX_OP_PGREAD);
196 return brix_send_error(ctx, c, kXR_IOError,
197 "negative read offset");
198 }
199
200 /* The wire rlen is a signed 32-bit field; a negative request length is
201 * invalid. Read unsigned it would turn -1 into ~4 GiB (then capped),
202 * silently succeeding where the reference rejects with kXR_ArgInvalid. */
203 if (req.rlen < 0) {
204 BRIX_OP_ERR(ctx, BRIX_OP_PGREAD);
205 return brix_send_error(ctx, c, kXR_ArgInvalid,
206 "negative read length");
207 }
208
209 if (rlen == 0) {
210 hdr_buf = ngx_palloc(c->pool, sizeof(*hdr_buf));
211 if (hdr_buf == NULL) {
212 return NGX_ERROR;
213 }
214 brix_build_pgread_status(ctx, offset, 0, hdr_buf);
215 BRIX_OP_OK(ctx, BRIX_OP_PGREAD);
216 return brix_queue_response(ctx, c, (u_char *) hdr_buf,
217 sizeof(*hdr_buf));
218 }
219
220 if (rlen > BRIX_READ_REQUEST_MAX) {
221 rlen = BRIX_READ_REQUEST_MAX;
222 }
223
224 fd = ctx->files[idx].fd;
225
226 {
227 size_t n_pages_max;
228 size_t scratch_size;
229 u_char *scratch;
230 ngx_flag_t warm_hit = 0;
231
232 rconf = ngx_stream_get_module_srv_conf(
233 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
234
235 /*
236 * File-offset alignment can split an otherwise single-page read across
237 * two pages (short first fragment + remainder), so the page count is
238 * derived from the in-page offset, not just rlen — otherwise the
239 * scratch/out region would be one CRC short.
240 */
241 n_pages_max = ((size_t) (offset & (kXR_pgPageSZ - 1)) + rlen
242 + kXR_pgPageSZ - 1) / kXR_pgPageSZ;
243 if (n_pages_max == 0) {
244 n_pages_max = 1;
245 }
246 /*
247 * Single buffer holding the final interleaved [CRC32c(4)][data] wire
248 * output (data is read straight into it — no separate flat copy region),
249 * so it needs only the data bytes plus one 4-byte CRC per page.
250 */
251 scratch_size = rlen + n_pages_max * BRIX_PG_CKSZ;
252
253 scratch = BRIX_GET_SCRATCH(ctx, c, rd.read_scratch, rd.read_scratch_size,
254 scratch_size);
255 if (scratch == NULL) {
256 return NGX_ERROR;
257 }
258
259 /*
260 * Warm-cache fast path: when the whole range is already page-cache
261 * resident, read + CRC it inline on the event loop via preadv2
262 * (RWF_NOWAIT) and skip the thread-pool handoff entirely — the handoff
263 * latency, not the copy, is the single-stream (n=1) cost. A miss
264 * (not resident / EOF / error) sets no warm_hit and falls through to the
265 * blocking offload below, which re-reads the full range. Only attempted
266 * with a pool configured (else the blocking path runs inline anyway) and
267 * for a regular file (RWF_NOWAIT is meaningful against the page cache).
268 * Mirrors the kXR_read Phase-32 probe.
269 */
270 if (rconf->common.thread_pool != NULL && ctx->files[idx].is_regular) {
271 ssize_t warm_nread = 0;
272 int warm_errno = 0;
273 size_t warm_osz;
274 brix_sd_obj_t warm_scratch;
275 brix_sd_obj_t *warm_obj;
276
277 warm_obj = brix_vfs_effective_obj(&ctx->files[idx].sd_obj, fd,
278 &warm_scratch);
279 warm_osz = brix_pgread_read_encode_inplace(warm_obj, (off_t) offset,
280 rlen, scratch,
281 &warm_nread,
282 &warm_errno, 1);
283 if (warm_errno == 0 && warm_nread == (ssize_t) rlen) {
284 out_size = warm_osz;
285 flat_buf = scratch;
286 out_buf = scratch;
287 warm_hit = 1; /* rlen already == bytes encoded */
288
289 /* The warm fast path bypasses brix_vfs_io_execute (where the
290 * cold pgread paths attribute), so charge the per-backend read
291 * total here for the file bytes just read. */
292 brix_metric_backend_bytes(
293 ctx->files[idx].sd_obj.driver != NULL
294 ? ctx->files[idx].sd_obj.driver->name : "posix",
295 BRIX_METRIC_OP_READ, (size_t) warm_nread);
296 }
297 }
298
299 if (!warm_hit && rconf->common.thread_pool != NULL) {
300 ngx_thread_task_t *task;
301 brix_pgread_aio_t *t;
302 ngx_flag_t posted;
303
304 task = ctx->rd.pgread_aio_task;
305 if (task == NULL) {
306 task = ngx_thread_task_alloc(c->pool,
307 sizeof(brix_pgread_aio_t));
308 if (task == NULL) {
309 return NGX_ERROR;
310 }
311 ctx->rd.pgread_aio_task = task;
312 } else {
313 task->next = NULL;
314 task->event.complete = 0;
315 }
316
317 t = task->ctx;
318 t->c = c;
319 t->ctx = ctx;
320 t->fd = fd;
321 t->handle_idx = idx;
322 t->offset = (off_t) offset;
323 t->rlen = rlen;
324 t->scratch = scratch;
325 t->out_size = 0;
326 t->streamid[0] = ctx->recv.cur_streamid[0];
327 t->streamid[1] = ctx->recv.cur_streamid[1];
328 t->obj = ctx->files[idx].sd_obj; /* Layer 3: driver obj (or zeroed) */
329
330 brix_task_bind(task, brix_pgread_aio_thread, brix_pgread_aio_done);
331
332 (void) brix_aio_post_task(ctx, c, rconf->common.thread_pool, task,
333 "brix: thread_task_post failed, sync pgread fallback",
334 &posted);
335 if (posted) {
336 return NGX_OK;
337 }
338 }
339
340 /*
341 * Sync fallback: read directly into the gapped wire buffer and CRC each
342 * page in place (no flat-buffer copy). Same code path as the AIO worker.
343 * Skipped when the warm-cache fast path already produced the encoding.
344 */
345 if (!warm_hit) {
346 brix_vfs_job_t job;
347
348 brix_vfs_job_read_init(&job, fd, (off_t) offset, rlen,
349 scratch, rlen, 0);
350 job.op = BRIX_VFS_IO_PGREAD;
351 brix_vfs_job_set_obj(&job, &ctx->files[idx].sd_obj);
352 brix_vfs_io_execute(&job);
353
354 if (job.io_errno != 0) {
355 BRIX_RETURN_ERR(ctx, c, BRIX_OP_PGREAD, "PGREAD",
356 ctx->files[idx].path, "-",
357 kXR_IOError, strerror(job.io_errno));
358 }
359
360 out_size = job.out_size;
361 flat_buf = scratch;
362 out_buf = scratch; /* output starts at offset 0 now */
363 rlen = (size_t) job.nio; /* actual bytes read (accounting) */
364 }
365 }
366
367 rsp_chain = brix_build_pgread_chain(ctx, c, offset, out_buf,
368 (uint32_t) out_size);
369 if (rsp_chain == NULL) {
370 brix_release_read_buffer(ctx, c, flat_buf);
371 return NGX_ERROR;
372 }
373
374 ctx->files[idx].bytes_read += rlen;
375 ctx->totals.bytes += rlen;
376 brix_rl_charge_ctx(ctx, rlen); /* Phase 25 bandwidth */
377
378 if (rconf->access_log_fd != NGX_INVALID_FILE) {
379 snprintf(detail, sizeof(detail), "%lld+%zu", (long long) offset, rlen);
380 brix_log_access(ctx, c, "PGREAD", ctx->files[idx].path,
381 detail, 1, 0, NULL, rlen);
382 }
383 BRIX_OP_OK(ctx, BRIX_OP_PGREAD);
384
385 {
386 ngx_int_t rc = brix_queue_response_chain(ctx, c, rsp_chain, flat_buf);
387
388 if (rc != NGX_OK || ctx->state != XRD_ST_SENDING) {
389 brix_release_read_buffer(ctx, c, flat_buf);
390 }
391 return rc;
392 }
