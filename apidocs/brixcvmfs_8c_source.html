1/*
2 * brixcvmfs.c — CVMFS-brix: a hardened, read-only CVMFS FUSE driver.
3 *
4 * WHAT: mounts a real CVMFS repository (e.g. atlas.cern.ch) as a read-only FUSE
5 * filesystem, driven entirely by the shared ngx-free CVMFS core
6 * (shared/cvmfs) — trust chain, SQLite catalogs, content-addressed fetch
7 * with replica+proxy failover, hash-verified retry, and offline cache mode.
8 * WHY: a single small binary that is "battle-tested against bad/evil networks":
9 * every object is verified by content hash, so any mirror/proxy failure or
10 * tampered reply is retried elsewhere and never trusted unverified.
11 * HOW: FUSE 3.1 high-level ops translate straight into cvmfs_client_* calls; a
12 * libcurl transport implements the injected fetch seam (per-request connect
13 * and low-speed timeouts = the adaptive-timeout pillar), while failover and
14 * verification are owned by the shared core. Read-only: every mutating op is
15 * refused with -EROFS.
16 */

69/* Extract scheme + host + port from a URL for env-proxy resolution. */
70static void url_host(const char *url, char *sch, size_t sl, char *host, size_t hl, int *port) {
71 sch[0] = host[0] = '\0'; *port = 0;
72 const char *p = url;
73 const char *sep = strstr(url, "://");
74 if (sep) { size_t n = (size_t)(sep - url); if (n < sl) { memcpy(sch, url, n); sch[n] = '\0'; } p = sep + 3; }
75 size_t n = 0;
76 while (p[n] && p[n] != ':' && p[n] != '/' && n < hl - 1) { host[n] = p[n]; n++; }
77 host[n] = '\0';
78 *port = (p[n] == ':') ? atoi(p + n + 1) : (strcmp(sch, "https") == 0 ? 443 : 80);
79}

81/* Persistent easy handles, pooled through the generic brix_cpool (the SAME
82 * slot/mutex/condvar engine the xrootdfs root:// and WebDAV-metadata paths use —
83 * uniformity across the two FUSE drivers, phase-86). libcurl's connection cache
84 * lives on the handle, so keeping handles across fetches is what keeps
85 * origin/proxy connections alive (keepalive reuse = fewer handshakes for a DPI
86 * middlebox to interfere with). checkout() hands a caller exclusive ownership of
87 * one handle for the duration of a transfer (libcurl easy handles are NOT
88 * concurrency-safe), so the foreground FUSE loop and the background prefetch
89 * worker each borrow their own handle without racing. Every per-request option
90 * is re-set on each call below, so no stale request state survives reuse; `-o
91 * fresh` still forces a new connection per request. */
92#define BRIX_CURL_POOL_SLOTS 4 /* foreground (-s serialized) + prefetch + slack */

113/* One GET of `url`, RESUMING from `*got` bytes already in `out` (HTTP Range).
114 * Appends new bytes and updates *got. Returns the CURLcode. If a resume request
115 * comes back 200 (server ignored Range), the freshly re-sent from-0 bytes are
116 * slid to the front so the buffer stays a valid prefix. */
117static CURLcode http_get_range(CURL **slot, const char *proxy, const char *url,
118 unsigned char *out, size_t outcap, size_t *got) {
119 if (*slot == NULL) *slot = curl_easy_init();
120 CURL *c = *slot;
121 if (c == NULL) return CURLE_FAILED_INIT;
122
123 size_t start = *got;
124 curl_sink_t sink = { out, outcap, start }; /* append after what we have */
125 curl_easy_setopt(c, CURLOPT_URL, url);
126 curl_easy_setopt(c, CURLOPT_WRITEFUNCTION, curl_write_cb);
127 curl_easy_setopt(c, CURLOPT_WRITEDATA, &sink);
128 curl_easy_setopt(c, CURLOPT_CONNECTTIMEOUT, g_tcfg.connect_timeout_s);
129 curl_easy_setopt(c, CURLOPT_LOW_SPEED_LIMIT, g_tcfg.low_speed_bytes);
130 curl_easy_setopt(c, CURLOPT_LOW_SPEED_TIME, g_tcfg.low_speed_time_s);
131 curl_easy_setopt(c, CURLOPT_FOLLOWLOCATION, 1L);
132 curl_easy_setopt(c, CURLOPT_FAILONERROR, 1L); /* HTTP >=400 → error */
133 curl_easy_setopt(c, CURLOPT_TCP_KEEPALIVE, 1L);
134 /* Reused handle: set these unconditionally so a prior request's resume
135 * offset / freshness flags can never leak into this one. */
136 curl_easy_setopt(c, CURLOPT_RESUME_FROM_LARGE, (curl_off_t) start);
137 curl_easy_setopt(c, CURLOPT_FRESH_CONNECT, g_tcfg.fresh_connect ? 1L : 0L);
138 curl_easy_setopt(c, CURLOPT_FORBID_REUSE, g_tcfg.fresh_connect ? 1L : 0L);
139 /* Proxy precedence: env proxy wins; else CVMFS-config proxy; else force direct. */
140 char sch[8], thost[256]; int tport;
141 url_host(url, sch, sizeof(sch), thost, sizeof(thost), &tport);
142 brix_proxy_t px;
143 if (brix_proxy_resolve(sch, thost, tport, &px)) {
144 curl_easy_setopt(c, CURLOPT_PROXY, px.url);
145 brix_proxy_report(&px, thost, tport);
146 } else if (proxy != NULL && strcmp(proxy, "DIRECT") != 0) {
147 curl_easy_setopt(c, CURLOPT_PROXY, proxy);
148 } else {
149 curl_easy_setopt(c, CURLOPT_PROXY, "");
150 }
151
152 CURLcode rc = curl_easy_perform(c);
153 long code = 0;
154 curl_easy_getinfo(c, CURLINFO_RESPONSE_CODE, &code);
155 if (start > 0 && code != 206 && sink.len > start) {
156 /* Range not honoured (200): the bytes appended after `start` are actually
157 * a fresh stream from offset 0 — slide them to the front. */
158 size_t fresh = sink.len - start;
159 memmove(out, out + start, fresh);
160 *got = fresh;
161 } else {
162 *got = sink.len;
163 }
164 return rc;
165}

174/* DPI/loss-hardened transport: RANGE-RESUME on a severed transfer (so a large
175 * object survives per-chunk connection loss — the key to 10%+ loss), optional
176 * TLS-first, optional fresh-connection. As long as each attempt makes progress
177 * (more bytes received) we keep resuming up to a generous cap; only attempts that
178 * make NO progress count against the retry budget. Across-mirror failover +
179 * hash-verify are owned by the fetch layer. */
180static int brixcvmfs_transport(const char *proxy, const char *host, const char *rel,
181 unsigned char *out, size_t outcap, size_t *outlen, void *ud) {
182 (void) ud; /* handles now come from g_curl_pool */
183 brix_status st; brix_status_clear(&st);
184 CURL **slot = brix_cpool_checkout(g_curl_pool, &st); /* CURL** == the classic slot */
185 if (slot == NULL) return -1;
186
187 char httpurl[1024];
188 snprintf(httpurl, sizeof(httpurl), "%s/%s", host, rel);
189
190 int budget = g_tcfg.max_retries > 0 ? g_tcfg.max_retries : 6;
191 int hard_cap = 64; /* absolute ceiling on resume attempts */
192 int stalls = 0;
193 size_t got = 0, last = 0;
194 int ret = -1;
195
196 for (int i = 0; i < hard_cap; i++) {
197 char httpsbuf[1024];
198 int use_https = (i == 0 && g_tcfg.prefer_tls
199 && to_https(httpurl, httpsbuf, sizeof(httpsbuf)));
200 const char *url = use_https ? httpsbuf : httpurl;
201
202 CURLcode rc = http_get_range(slot, proxy, url, out, outcap, &got);
203 if (rc == CURLE_OK) { *outlen = got; ret = 0; break; }
204
205 /* hard 4xx over plain http = mirror lacks the object → fail over. A 4xx on
206 * the https probe just means no TLS — fall through to http. */
207 if (rc == CURLE_HTTP_RETURNED_ERROR && !use_https) break;
208
209 /* Range-blind server: libcurl aborts a resumed request answered 200 with
210 * CURLE_RANGE_ERROR before any body byte reaches the sink, so resume can
211 * never progress there — throw away the partial prefix and restart the
212 * whole object from 0 (the fetch layer's hash check keeps this safe). */
213 if (rc == CURLE_RANGE_ERROR) { got = 0; last = 0; }
214
215 if (got > last) { last = got; stalls = 0; } /* progress → keep resuming */
216 else if (++stalls > budget) break; /* no progress → give up route */
217
218 long ms = 200L << (stalls < 6 ? stalls : 6); /* backoff on stalls only */
219 struct timespec ts = { ms / 1000, (ms % 1000) * 1000000L };
220 nanosleep(&ts, NULL);
221 }
222 /* A libcurl easy handle stays healthy across transfers (it owns its own
223 * connection cache and re-establishes internally), so always check it back
224 * in reusable — never health-drop. */
225 brix_cpool_checkin(g_curl_pool, slot, 1);
226 return ret;
227}

266/* ---- predictive prefetch (phase-85 F4) ---------------------------------- *
267 * Opt-in subtree readahead: the first readdir of a directory queues it for a
268 * background worker that walks the catalog subtree (cvmfs_walk_subtree) and
269 * pre-pulls every referenced CAS object into the shared cache, so subsequent
270 * opens are cache hits. The worker owns its OWN failover state, scratch and
271 * CAS-store handle (puts are O_EXCL+rename atomic, so sharing the cache
272 * DIRECTORY with the foreground is safe); its libcurl handle is borrowed from
273 * the shared g_curl_pool per fetch (exclusive checkout, so it never races the
274 * FUSE loop's handle). Bounded by a byte budget;
275 * prefetch errors are always swallowed — they can never fail a foreground op. */
276
277#define BRIX_PF_QCAP 32
278#define BRIX_PF_SEENCAP 512
279#define BRIX_PF_OBJCAP (32u * 1024u * 1024u)
280
281typedef struct {
282 int enabled;
283 int depth; /* nested-catalog descent budget */
284 long budget; /* bytes; <= 0 = unbounded */
285 long spent;
286 int capped; /* budget hit: audit once, drain queue */
287 char tmp_dir[512];
288 char cache_dir[512];
289 int cache_dirfd; /* dup'd overlay dirfd, or -1 */
290 long quota;
291 pthread_t tid;
292 pthread_mutex_t mu;
293 pthread_cond_t cv;
294 struct { char path[1024]; cvmfs_hash_t root; } q[BRIX_PF_QCAP];
295 int qh, qn;
296 uint64_t seen[BRIX_PF_SEENCAP]; /* FNV-1a of queued dir paths */
297 int nseen;
298 cvmfs_failover_t fo; /* worker-owned failover state */
299 cvmfs_failover_t fo0; /* pristine snapshot (blacklist reset) */
300} brix_prefetch_t;

302static brix_prefetch_t g_pf = { .cache_dirfd = -1 };

304static uint64_t pf_fnv1a(const char *s) {
305 uint64_t h = 1469598103934665603ull;
306 for (; *s; s++) { h ^= (unsigned char) *s; h *= 1099511628211ull; }
307 return h;
308}

312 unsigned char *out;

315static int pf_visit(const cvmfs_walk_item_t *it, void *ud) {
316 pf_walk_ud_t *p = ud;
317 if (it->kind == CVMFS_WALK_CATALOG) return 0; /* the walk itself cached it */
318 if (g_pf.budget > 0 && g_pf.spent >= g_pf.budget) {
319 if (!g_pf.capped) {
320 g_pf.capped = 1;
321 fprintf(stderr, "brixcvmfs: audit signal=prefetchcap repo=%s "
322 "budget=%ld spent=%ld (prefetch stopped, foreground unaffected)\n",
323 g_cl->config.name, g_pf.budget, g_pf.spent);
324 }
325 return 1;
326 }
327 size_t n = 0;
328 if (cvmfs_fetch_object(p->fx, &it->hash, it->suffix,
329 p->out, BRIX_PF_OBJCAP, &n, mono_now()) == 0)
330 g_pf.spent += (long) n;
331 else
332 g_pf.fo = g_pf.fo0; /* one bad object blacklists its route — restore the
333 * pristine snapshot so it can't shadow the sweep */
334 return 0; /* fetch errors never stop the sweep */
335}

337static void *pf_main(void *arg) {
338 (void) arg;
339 brix_cas_store_t cache;
340 int crc = g_pf.cache_dirfd >= 0
341 ? brix_cas_init_at(&cache, g_pf.cache_dirfd, g_pf.quota)
342 : brix_cas_init(&cache, g_pf.cache_dir, g_pf.quota);
343 unsigned char *scratch = malloc(BRIX_PF_OBJCAP);
344 unsigned char *out = malloc(BRIX_PF_OBJCAP);
345 if (crc != 0 || scratch == NULL || out == NULL) {
346 free(scratch); free(out);
347 return NULL; /* prefetch unavailable, mount unaffected */
348 }
349
350 cvmfs_fetch_ctx_t fx;
351 memset(&fx, 0, sizeof(fx));
352 fx.fo = &g_pf.fo;
353 fx.cache = &cache;
354 fx.transport = brixcvmfs_transport;
355 fx.transport_ud = NULL; /* handle borrowed from g_curl_pool per fetch */
356 fx.store_form = CVMFS_STORE_COMPRESSED;
357 fx.scratch = scratch;
358 fx.scratch_cap = BRIX_PF_OBJCAP;
359
360 for (;;) {
361 pthread_mutex_lock(&g_pf.mu);
362 while (g_pf.qn == 0) pthread_cond_wait(&g_pf.cv, &g_pf.mu);
363 char path[1024];
364 cvmfs_hash_t root = g_pf.q[g_pf.qh].root;
365 snprintf(path, sizeof(path), "%s", g_pf.q[g_pf.qh].path);
366 g_pf.qh = (g_pf.qh + 1) % BRIX_PF_QCAP;
367 g_pf.qn--;
368 pthread_mutex_unlock(&g_pf.mu);
369
370 if (g_pf.capped) continue; /* drain silently once the budget is hit */
371 pf_walk_ud_t ud = { &fx, out };
372 cvmfs_walk_subtree(&fx, &root, g_pf.tmp_dir, path, g_pf.depth,
373 pf_visit, &ud, mono_now());
374 }
375 return NULL;
376}

378/* FUSE-thread side: queue `path` once. Drops silently when the queue is full
379 * or the path was already prefetched — advisory readahead, never a failure. */
380static void pf_enqueue(const char *path) {
381 if (!g_pf.enabled) return;
382 uint64_t h = pf_fnv1a(path);
383 for (int i = 0; i < g_pf.nseen; i++)
384 if (g_pf.seen[i] == h) return;
385 if (g_pf.nseen < BRIX_PF_SEENCAP) g_pf.seen[g_pf.nseen++] = h;
386
387 pthread_mutex_lock(&g_pf.mu);
388 if (g_pf.qn < BRIX_PF_QCAP) {
389 int slot = (g_pf.qh + g_pf.qn) % BRIX_PF_QCAP;
390 snprintf(g_pf.q[slot].path, sizeof(g_pf.q[slot].path), "%s", path);
391 g_pf.q[slot].root = g_cl->pin_set ? g_cl->pin_root
392 : g_cl->manifest.root_catalog;
393 g_pf.qn++;
394 pthread_cond_signal(&g_pf.cv);
395 }
396 pthread_mutex_unlock(&g_pf.mu);
397}

399/* Called once between mount and fuse_main (still single-threaded): snapshot the
400 * failover set and start the worker. Failure to start just disables prefetch. */
401static void pf_start(int depth, long budget, const char *tmp_dir,
402 const char *cache_dir, int cache_dirfd, long quota) {
403 g_pf.depth = depth;
404 g_pf.budget = budget;
405 snprintf(g_pf.tmp_dir, sizeof(g_pf.tmp_dir), "%s", tmp_dir);
406 snprintf(g_pf.cache_dir, sizeof(g_pf.cache_dir), "%s", cache_dir);
407 g_pf.cache_dirfd = cache_dirfd >= 0 ? dup(cache_dirfd) : -1;
408 g_pf.quota = quota;
409 g_pf.fo = g_pf.fo0 = g_cl->fo; /* snapshot; worker state evolves alone */
410 pthread_mutex_init(&g_pf.mu, NULL);
411 pthread_cond_init(&g_pf.cv, NULL);
412 if (pthread_create(&g_pf.tid, NULL, pf_main, NULL) == 0) {
413 pthread_detach(g_pf.tid);
414 g_pf.enabled = 1;
415 }
416}

434 st->st_uid = e.uid ? e.uid : getuid();
435 st->st_gid = e.gid ? e.gid : getgid();

510/* access(2): the fs is read-only, so any W_OK is refused up front; X_OK is
511 * honored against the node's mode bits (a 0644 file has no execute bit). R_OK /
512 * F_OK on a world-readable RO tree always succeed. Mirrors official cvmfs. */
513int brixcvmfs_op_access(const char *path, int mask) {
514 cvmfs_dirent_t e;
515 int rc = cvmfs_client_resolve(g_cl, cat_path(path), &e, mono_now());
516 if (rc < 0) return -EIO;
517 if (rc == 0) return -ENOENT;
518 if (mask & W_OK) return -EROFS; /* read-only filesystem */
519 if ((mask & X_OK) && !(e.mode & (S_IXUSR | S_IXGRP | S_IXOTH))) return -EACCES;
520 return 0;
521}

576static unsigned char *read_file(const char *path, size_t *len) {
577 FILE *f = fopen(path, "rb");
578 if (f == NULL) return NULL;
579 fseek(f, 0, SEEK_END);
580 long n = ftell(f);
581 fseek(f, 0, SEEK_SET);
582 unsigned char *b = malloc(n > 0 ? (size_t) n : 1);
583 if (b && fread(b, 1, (size_t) n, f) != (size_t) n) { free(b); b = NULL; }
584 fclose(f);
585 if (b) *len = (size_t) n;
586 return b;
587}

589/* Load the repo master key(s). If `path` is a directory (the stock
590 * /etc/cvmfs/keys/<domain>/ layout), concatenate every *.pub in it — CVMFS
591 * rotates master keys and the whitelist is signed by one of them. */
592static unsigned char *load_master_key(const char *path, size_t *len) {
593 struct stat st;
594 if (stat(path, &st) != 0) return NULL;
595 if (!S_ISDIR(st.st_mode)) return read_file(path, len);
596
597 DIR *d = opendir(path);
598 if (d == NULL) return NULL;
599 unsigned char *buf = NULL; size_t cap = 0, used = 0;
600 struct dirent *e;
601 while ((e = readdir(d)) != NULL) {
602 const char *dot = strrchr(e->d_name, '.');
603 if (dot == NULL || strcmp(dot, ".pub") != 0) continue;
604 char full[1024];
605 snprintf(full, sizeof(full), "%s/%s", path, e->d_name);
606 size_t kl = 0;
607 unsigned char *k = read_file(full, &kl);
608 if (k == NULL) continue;
609 if (used + kl + 1 > cap) { cap = (used + kl + 1) * 2;
610 unsigned char *nb = realloc(buf, cap); if (nb == NULL) { free(k); continue; } buf = nb; }
611 memcpy(buf + used, k, kl); used += kl;
612 buf[used++] = '\n';
613 free(k);
614 }
615 closedir(d);
616 if (buf) *len = used;
617 return buf;
618}

636static void opts_o_list(char *list, brix_opts_t *o) {
637 char *save = NULL;
638 for (char *t = strtok_r(list, ",", &save); t; t = strtok_r(NULL, ",", &save)) {
639 if (strcmp(t, "clever") == 0) o->clever = 1;
640 else if (strcmp(t, "noclever") == 0) o->clever = 0;
641 else if (strcmp(t, "fresh") == 0) o->fresh = 1;
642 else if (strcmp(t, "tls") == 0) o->tls = 1;
643 else if (strncmp(t, "quota=", 6) == 0) o->quota_mb = atoi(t + 6);
644 else if (strncmp(t, "retries=", 8) == 0) o->retries = atoi(t + 8);
645 else if (strncmp(t, "prefetch=", 9) == 0) o->prefetch = atoi(t + 9);
646 else if (strncmp(t, "prefetch_budget=", 16) == 0) o->prefetch_budget = atol(t + 16);
647 else if (strncmp(t, "pin=", 4) == 0)
648 snprintf(o->pin, sizeof(o->pin), "%s", t + 4);
649 else if (strncmp(t, "cache=", 6) == 0)
650 snprintf(o->cache_dir, sizeof(o->cache_dir), "%s", t + 6);
651 else if (strncmp(t, "writes=", 7) == 0)
652 snprintf(o->writes_dir, sizeof(o->writes_dir), "%s", t + 7);
653 else { /* forward to libfuse */
654 size_t cur = strlen(o->fuse_extra);
655 snprintf(o->fuse_extra + cur, sizeof(o->fuse_extra) - cur,
656 "%s%s", cur ? "," : "", t);
657 }
658 }
659}

661static void parse_opts(int argc, char **argv, int start, brix_opts_t *o) {
662 memset(o, 0, sizeof(*o));
663 o->clever = 1; o->retries = -1; o->prefetch = -1;
664 char obuf[512];
665 for (int i = start; i < argc; i++) {

671 opts_o_list(obuf, o);
672 } else if (o->nflags < 16) {
673 o->flags[o->nflags++] = argv[i]; /* -f / -d / -h / ... */
674 }
675 }

678/* Populate the client failover set for `repo`. $BRIXCVMFS_SERVER pins a single
679 * DIRECT stratum-1; otherwise apply the loaded config cascade, and if that yields
680 * no hosts, synthesise the conventional cvmfs-stratum-one.<domain> fallback. */
681static void brixcvmfs_build_failover(cvmfs_client_t *cl, const cvmfs_conf_t *cf,
682 const char *repo) {
683 cvmfs_failover_init(&cl->fo, 60);
684 const char *env_server = getenv("BRIXCVMFS_SERVER");

686 cvmfs_failover_add_proxy(&cl->fo, "DIRECT", 0);
687 cvmfs_failover_add_host(&cl->fo, env_server);
688 return;
689 }
690 int hosts = cvmfs_conf_apply(cf, repo, &cl->config, &cl->fo);
691 if (hosts == 0) {
692 char s1[400];
693 snprintf(s1, sizeof(s1), "http://cvmfs-stratum-one.%s/cvmfs/%s",
694 strchr(repo, '.') + 1, repo);
695 cvmfs_failover_add_host(&cl->fo, s1);
696 }

712/* Resolve the effective cache quota in bytes: a positive `quota_override` wins;
713 * otherwise fall back to CVMFS_QUOTA_LIMIT (MB) from the conf cascade; else 0. */
714static long brixcvmfs_resolve_quota(const cvmfs_conf_t *cf, long quota_override) {
715 if (quota_override > 0) return quota_override;
716 const char *q = cvmfs_conf_get(cf, "CVMFS_QUOTA_LIMIT"); /* MB */
717 if (q && atol(q) > 0) return atol(q) * 1024L * 1024L;
718 return 0;

721/* Load the repo master key: $BRIXCVMFS_PUBKEY overrides the config default path.
722 * A config-derived CVMFS_KEYS_DIR/<domain>.pub that does not exist falls back to
723 * the key-chain DIRECTORY itself (stock layouts rotate keys as e.g.
724 * keys/cern.ch/cern-it4.cern.ch.pub with no cern.ch.pub; load_master_key
725 * concatenates every *.pub and the verifier tries each). Returns the allocated
726 * key blob (caller frees) + `*klen`, or NULL on read failure (message printed). */
727static unsigned char *brixcvmfs_load_repo_key(const cvmfs_client_t *cl, size_t *klen) {
728 const char *env_key = getenv("BRIXCVMFS_PUBKEY");
729 const char *keypath = env_key ? env_key : cl->config.master_pub_path;
730 unsigned char *master = load_master_key(keypath, klen);
731 if (master == NULL && env_key == NULL) {
732 char chain[512];

734 char *slash = strrchr(chain, '/');
735 if (slash != NULL && slash != chain) {
736 *slash = '\0';
737 master = load_master_key(chain, klen);
738 }
739 }
740 if (master == NULL)
741 fprintf(stderr, "brixcvmfs: cannot read master key %s\n", keypath);
742 return master;
743}

772/* Build failover + config and verify-mount the repo trust chain. Cache backing:
773 * `cache_dirfd >= 0` = overlay dirfd mode; else `cache_dir_override` (or the
774 * default/env path). quota/retries fall back to the config cascade when the
775 * override is unset (<=0 quota, <0 retries). Returns a mounted client or NULL. */
776static cvmfs_client_t *brixcvmfs_open(const char *repo, const char *cache_dir_override,
777 int cache_dirfd, long quota_override, int retries_override,
778 const char *pin_opt) {
779 cvmfs_client_t *cl = calloc(1, sizeof(*cl));
780 if (cl == NULL) { fprintf(stderr, "brixcvmfs: out of memory\n"); return NULL; }
781
782 if (cvmfs_repo_config_defaults(repo, &cl->config) != 0) {
783 fprintf(stderr, "brixcvmfs: '%s' is not a fully-qualified repo name\n", repo);
784 free(cl); return NULL;
785 }
786
787 const char *pin = (pin_opt != NULL && pin_opt[0]) ? pin_opt : getenv("BRIXCVMFS_PIN");
788 if (pin != NULL && pin[0] && cvmfs_client_pin_root(cl, pin) != 0) {
789 fprintf(stderr, "brixcvmfs: bad pin '%s' (want a root-catalog hash)\n", pin);
790 free(cl); return NULL;
791 }
792
793 cvmfs_conf_t cf;
794 cvmfs_conf_init(&cf);
795 cvmfs_conf_load_cascade(&cf, getenv("BRIXCVMFS_ETC"), repo); /* for quota/retries */
796
797 brixcvmfs_build_failover(cl, &cf, repo);
798 brixcvmfs_build_transport_cfg(cl, &cf, repo, retries_override);
799 long quota = brixcvmfs_resolve_quota(&cf, quota_override);
800
801 size_t klen = 0;
802 unsigned char *master = brixcvmfs_load_repo_key(cl, &klen);
803 if (master == NULL) { free(cl); return NULL; }
804
805 char tmp_dir[512];
806 brixcvmfs_prepare_tmp_dir(repo, tmp_dir, sizeof(tmp_dir));
807
808 char cache_dir[512] = "";
809 brixcvmfs_prepare_cache_dir(repo, cache_dir_override, cache_dirfd,
810 cache_dir, sizeof(cache_dir));
811
812 curl_global_init(CURL_GLOBAL_DEFAULT);
813 /* Bring the shared curl-handle pool up before the mount fetches the root
814 * catalog (eager slot-0 connect surfaces a curl_easy_init failure here). */
815 brix_status pst; brix_status_clear(&pst);
816 g_curl_pool = brix_cpool_create(&CURL_VT, NULL, BRIX_CURL_POOL_SLOTS, &pst);
817 if (g_curl_pool == NULL) {
818 fprintf(stderr, "brixcvmfs: curl handle pool init failed: %s\n", pst.msg);
819 curl_global_cleanup();
820 free(cl); return NULL;
821 }
822 int mrc = cvmfs_client_mount(cl, repo, master, klen,
823 cache_dirfd < 0 ? cache_dir : NULL, tmp_dir,
824 quota, cache_dirfd, brixcvmfs_transport, NULL, mono_now());
825 free(master);
826 if (mrc != 0) {
827 fprintf(stderr, "brixcvmfs: mount of %s failed (trust/catalog error %d)\n", repo, mrc);
828 transport_cleanup();
829 curl_global_cleanup();
830 free(cl); return NULL;
831 }
832 return cl;
833}

876/* --prewarm (phase-85 F5): walk the WHOLE snapshot (the pin when
877 * $BRIXCVMFS_PIN is set, else the current root) and pull every referenced CAS
878 * object into the local cache, so a shared cache dir is fully warm before a
879 * job wave. No mount, single-threaded: reuses the client's own fetch ctx.
880 * Exit 0 = every object landed; a fetch error or a tampered catalog ⇒ 1. */
881typedef struct {
882 cvmfs_fetch_ctx_t *fx;
883 unsigned char *out;
884 cvmfs_failover_t fo0; /* pristine snapshot (blacklist reset, cf. F4) */
885 long objs, bytes, errs;
886} prewarm_ud_t;
887
888static int prewarm_visit(const cvmfs_walk_item_t *it, void *ud) {
889 prewarm_ud_t *p = ud;
890 if (it->kind == CVMFS_WALK_CATALOG) return 0; /* the walk itself caches it */
891 size_t n = 0;
892 if (cvmfs_fetch_object(p->fx, &it->hash, it->suffix,
893 p->out, BRIX_PF_OBJCAP, &n, mono_now()) == 0) {
894 p->objs++;
895 p->bytes += (long) n;
896 } else {
897 p->errs++;
898 *p->fx->fo = p->fo0; /* one bad object blacklists its route — restore
899 * so it can't shadow the rest of the sweep */
900 }
901 return 0;
902}
903
904static int brixcvmfs_prewarm(const char *repo) {
905 cvmfs_client_t *cl = brixcvmfs_open(repo, NULL, -1, 0, -1, NULL);
906 if (cl == NULL) return 1;
907
908 prewarm_ud_t ud = { &cl->fetch, malloc(BRIX_PF_OBJCAP), cl->fo, 0, 0, 0 };
909 int rc = -1;
910 if (ud.out != NULL) {
911 const cvmfs_hash_t *root = cl->pin_set ? &cl->pin_root
912 : &cl->manifest.root_catalog;
913 rc = cvmfs_walk_catalog(&cl->fetch, root, cl->catalog_tmp,
914 INT_MAX, prewarm_visit, &ud, mono_now());
915 char hex[48];
916 cvmfs_hash_to_hex(root, 0, hex, sizeof(hex));
917 printf("CVMFS-brix prewarm: %s\n", repo);
918 printf(" root catalog ... %s%s\n", hex, cl->pin_set ? " (pinned)" : "");
919 printf(" objects ........ %ld fetched (%ld bytes)\n", ud.objs, ud.bytes);
920 printf(" errors ......... %ld%s\n", ud.errs,
921 rc != 0 ? " (walk aborted: catalog fetch/verify failure)" : "");
922 printf("%s\n", rc == 0 && ud.errs == 0 ? "WARM" : "INCOMPLETE");
923 } else {
924 fprintf(stderr, "brixcvmfs: out of memory\n");
925 }
926 free(ud.out);
927
928 cvmfs_client_umount(cl);
929 transport_cleanup();
930 curl_global_cleanup();
931 free(cl);
932 return rc == 0 && ud.errs == 0 ? 0 : 1;
933}

935/* Resolve the clever-overlay cache: unless an explicit cache (-o cache= or
936 * $BRIXCVMFS_CACHE) opts out, create <mnt>/.brixcache and open a dirfd on it
937 * BEFORE the FUSE mount hides it. Sets `*cache_override` (the explicit-cache
938 * path, else NULL) and returns the overlay dirfd, or -1 (explicit/fallback). */
939static int brixcvmfs_open_clever_cache(const char *mnt, const brix_opts_t *o,
940 const char **cache_override) {
941 *cache_override = NULL;
942 int clever = o->clever;
943 if (o->cache_dir[0]) { *cache_override = o->cache_dir; clever = 0; }
944 else if (getenv("BRIXCVMFS_CACHE")) { clever = 0; }
945 if (!clever) return -1;
946

958/* Bind the cvmfs-rw writable overlay (<mnt>/.brixwrites or -o writes=) BEFORE

960 * weak so a ro-only link (test builds) still works. No-op unless brixcvmfs_rw.
961 * Returns 0 on success (or when not requested), else a process exit code. */

963 if (!brixcvmfs_rw) return 0;
964 if (brixcvmfs_setup_rw == NULL || &brixcvmfs_rw_ops == NULL) {
965 fprintf(stderr, "brixcvmfs: rw overlay driver not linked in this build\n");

972/* Hand the mountpoint (+ passthrough flags/opts) to libfuse. Force
973 * single-threaded (-s): the client shares one SQLite catalog handle + lock-free
974 * failover state, so serialised FUSE dispatch is the correct, race-free choice
975 * (reads are cache-served, so throughput is unaffected). Returns fuse_main's rc. */
976static int brixcvmfs_run_fuse(char *arg0, const char *mnt, const brix_opts_t *o) {
977 char *fargv[24]; int fargc = 0;
978 fargv[fargc++] = arg0;
979 fargv[fargc++] = (char *) mnt;
980 fargv[fargc++] = (char *) "-s";
981 /* Flag the kernel mount read-only (matches official cvmfs2) — the pure-ro
982 * build only. The --rw overlay build is genuinely writable, so must not. */
983 if (!brixcvmfs_rw) { fargv[fargc++] = (char *) "-o"; fargv[fargc++] = (char *) "ro"; }
984 for (int i = 0; i < o->nflags && fargc < 20; i++) fargv[fargc++] = o->flags[i];
985 if (o->fuse_extra[0]) { fargv[fargc++] = (char *) "-o"; fargv[fargc++] = (char *) o->fuse_extra; }
986
987 return fuse_main(fargc, fargv,
988 brixcvmfs_rw ? &brixcvmfs_rw_ops : &brixcvmfs_ops, NULL);
989}

991/* Full mount bring-up: clever-cache dirfd, rw overlay, verify-mount the repo,
992 * run FUSE, then tear everything down. `cache_dirfd` ownership is local to this
993 * function. Returns the process exit code. */
994static int brixcvmfs_mount_run(char *arg0, const char *repo, const char *mnt,
995 const brix_opts_t *o) {
996 g_tcfg.fresh_connect = o->fresh;
997 g_tcfg.prefer_tls = o->tls;
998 long quota = o->quota_mb > 0 ? (long) o->quota_mb * 1024L * 1024L : 0;
999
1000 const char *cache_override = NULL;
1001 int cache_dirfd = brixcvmfs_open_clever_cache(mnt, o, &cache_override);
1002
1003 int rw_rc = brixcvmfs_setup_rw_overlay(mnt, o);
1004 if (rw_rc != 0) { if (cache_dirfd >= 0) close(cache_dirfd); return rw_rc; }
1005
1006 cvmfs_client_t *cl = brixcvmfs_open(repo, cache_override, cache_dirfd, quota, o->retries,
1007 o->pin);
1008 if (cl == NULL) { if (cache_dirfd >= 0) close(cache_dirfd); return 1; }
1009 g_cl = cl;
1010
1011 /* F4 predictive prefetch: -o prefetch=<depth> or $BRIXCVMFS_PREFETCH. */
1012 const char *env_pf = getenv("BRIXCVMFS_PREFETCH");
1013 int pf_depth = o->prefetch >= 0 ? o->prefetch : (env_pf ? atoi(env_pf) : -1);
1014 if (pf_depth >= 0) {
1015 const char *env_pb = getenv("BRIXCVMFS_PREFETCH_BUDGET");
1016 long pf_budget = o->prefetch_budget > 0 ? o->prefetch_budget
1017 : (env_pb ? atol(env_pb) : 0);
1018 char pf_cache[512] = "";
1019 brixcvmfs_prepare_cache_dir(repo, cache_override, cache_dirfd,
1020 pf_cache, sizeof(pf_cache));
1021 pf_start(pf_depth, pf_budget, cl->catalog_tmp, pf_cache, cache_dirfd, quota);
1022 }
1023
1024 int rc = brixcvmfs_run_fuse(arg0, mnt, o);
1025
1026 if (brixcvmfs_rw && brixcvmfs_teardown_rw != NULL) brixcvmfs_teardown_rw();
1027 cvmfs_client_umount(cl);
1028 transport_cleanup();
1029 curl_global_cleanup();
1030 if (cache_dirfd >= 0) close(cache_dirfd);
1031 free(cl);
1032 return rc;
1033}
