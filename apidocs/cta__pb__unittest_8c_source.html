23/* Build a cta.xrd.Request{ notification{ wf{event,instance{name}}, cli{user{u,g}},
24 * file{lpath,archive_file_id} } } into out; returns total length. */
25static size_t
26build_request(uint64_t event, const char *inst, const char *user,
27 const char *grp, const char *lpath, uint64_t afid,
28 unsigned char *out, size_t cap)
29{
30 unsigned char svc[64], wf[128], rid[64], cli[128], file[1100], notif[1400];
31 pb_writer ws = { svc, 0, sizeof(svc) };
32 pb_writer ww = { wf, 0, sizeof(wf) };
33 pb_writer wr = { rid, 0, sizeof(rid) };
34 pb_writer wc = { cli, 0, sizeof(cli) };
35 pb_writer wf2= { file, 0, sizeof(file) };
36 pb_writer wn = { notif,0, sizeof(notif) };
37 pb_writer wo = { out, 0, cap };
38
39 /* Service{ name=1 } */
40 pb_write_string(&ws, 1, inst);
41 /* Workflow{ event=1 (varint), instance=5 (Service) } */
42 pb_write_varint_field(&ww, 1, event);
43 pb_write_len_delim(&ww, 5, svc, ws.len);
44 /* RequesterId{ username=1, groupname=2 } */
45 pb_write_string(&wr, 1, user);
46 pb_write_string(&wr, 2, grp);
47 /* Client{ user=1 (RequesterId) } */
48 pb_write_len_delim(&wc, 1, rid, wr.len);
49 /* Metadata{ lpath=11, archive_file_id=15 (varint) } */
50 pb_write_string(&wf2, 11, lpath);
51 pb_write_varint_field(&wf2, 15, afid);
52 /* Notification{ wf=1, cli=2, file=4 } */
53 pb_write_len_delim(&wn, 1, wf, ww.len);
54 pb_write_len_delim(&wn, 2, cli, wc.len);
55 pb_write_len_delim(&wn, 4, file, wf2.len);
56 /* Request{ notification=1 } */
57 pb_write_len_delim(&wo, 1, notif, wn.len);
58 return wo.len;
59}

97static void test_unknown_field_skipped(void)
98{
99 /* a top-level varint field 99 (unknown) before the notification → skipped */
100 unsigned char buf[2048], full[2100];
101 cta_request_t req;
102 size_t n = build_request(4, "e", "u", "g", "/f", 1, buf, sizeof(buf));
103 pb_writer w = { full, 0, sizeof(full) };
104 pb_write_varint_field(&w, 99, 12345); /* unknown field */
105 memcpy(full + w.len, buf, n); /* then the real request */
106 CHECK(cta_pb_decode_request(full, w.len + n, &req) == 0);
107 CHECK(req.op == CTA_OP_ARCHIVE);
108 CHECK(strcmp(req.path, "/f") == 0);
109}

120static void test_encode_response(void)
121{
122 unsigned char out[256];
123 size_t n = 0;
124 pb_reader r;
125 uint32_t f; int wt;
126 int saw_type = 0, saw_msg = 0;
127
128 CHECK(cta_pb_encode_response(CTA_RSP_SUCCESS, "ok", 0, out, sizeof(out), &n) == 0);
129 CHECK(n > 0);
130 /* decode back: type field 1 (varint) == 1, message_txt field 3 == "ok" */
131 r.p = out; r.end = out + n;
132 while (r.p < r.end) {
133 CHECK(pb_read_tag(&r, &f, &wt) == 0);
134 if (f == 1 && wt == PB_WT_VARINT) {
135 uint64_t v; CHECK(pb_read_varint(&r, &v) == 0);
136 CHECK(v == CTA_RSP_SUCCESS); saw_type = 1;
137 } else if (f == 3 && wt == PB_WT_LEN) {
138 const unsigned char *d; size_t L;
139 CHECK(pb_read_len_delim(&r, &d, &L) == 0);
140 CHECK(L == 2 && memcmp(d, "ok", 2) == 0); saw_msg = 1;
141 } else {
142 CHECK(pb_skip_field(&r, wt) == 0);
143 }
144 }
145 CHECK(saw_type && saw_msg);
146}
