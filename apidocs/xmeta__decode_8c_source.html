1/*
2 * fs/meta/xmeta_decode.c — decode a per-file metadata record from its wire form.
3 *
4 * WHAT: brix_xmeta_decode() and its helpers: parse a wire buffer into a
5 * brix_xmeta_t — the byte-identical stock XrdPfc cinfo v4 prefix
6 * (version + Store POD + crc + bitmap + AStat[] + crc) followed by the
7 * optional "XCX1" extension header and its STATE / ORIGIN / DIGEST /
8 * BLOCKCRC TLV sections. WHY: split out of xmeta.c (942 lines) so the
9 * decode half and the encode half (xmeta_encode.c) each stay under the
10 * 500-line cap and can be reviewed as a matched pair. HOW: every
11 * variable-length region is crc-verified before use; a wrong version or
12 * insane sizes yield BRIX_XMETA_FOREIGN ("not our record"), while a crc
13 * mismatch on a recognized layout yields BRIX_XMETA_ERR ("torn write").
14 * Unknown extension section types are skipped for forward compat.
15 */

27/* bounded sequential read from the decode buffer; 0 = ok, -1 = short */
28static int
29xmeta_get(const uint8_t *buf, size_t len, size_t *off, void *dst, size_t n)
30{
31 if (len - *off < n) {
32 return -1;
33 }
34 memcpy(dst, buf + *off, n);
35 *off += n;
36 return 0;
37}

39/* ---- Decode a STATE section payload ----
40 *
41 * WHAT: Unpacks the fixed 80-byte xmeta_state_wire_t from p into m's
42 * write-back/lifecycle fields. Returns BRIX_XMETA_OK, or
43 * BRIX_XMETA_ERR if the payload is shorter than the wire struct.
44 *
45 * WHY: Pairs with xmeta_encode_state_section — the two field-copy lists must
46 * mirror each other exactly for the persisted format to round-trip.
47 *
48 * HOW: 1. Reject payloads shorter than the wire struct (crc already checked
49 * by the section loop; this guards the struct read itself).
50 * 2. Copy the struct out and assign each field to m; set have_state.
51 */
52static int
53xmeta_decode_state(brix_xmeta_t *m, const uint8_t *p, uint32_t plen)
54{
55 xmeta_state_wire_t state;
56
57 if (plen < sizeof(state)) {
58 return BRIX_XMETA_ERR;
59 }
60 memcpy(&state, p, sizeof(state));
61 m->have_state = 1;
62 m->origin_mtime = state.origin_mtime;
63 m->dirty_lo = state.dirty_lo;
64 m->dirty_hi = state.dirty_hi;
65 m->flush_gen = state.flush_gen;
66 m->dirty_since = state.dirty_since;
67 m->last_flush = state.last_flush;
68 m->bytes_flushed = state.bytes_flushed;
69 m->expires_at = state.expires_at;
70 m->filled_at = state.filled_at;
71 m->mode = state.mode;
72 m->state_flags = state.state_flags;
73 return BRIX_XMETA_OK;
74}

76/* ---- Decode an ORIGIN section payload ----
77 *
78 * WHAT: Unpacks the origin validators (etag, checksum algorithm name,
79 * checksum hex) from the {u8 lens, strings...} payload into m. Returns
80 * BRIX_XMETA_OK, or BRIX_XMETA_ERR on a short payload or a length
81 * byte exceeding the in-memory field capacity.
82 *
83 * WHY: Pairs with xmeta_encode_origin_section. The length bytes come from
84 * disk, so each is bounds-checked against its destination buffer before
85 * any copy — a corrupt record must never overflow m's fixed arrays.
86 *
87 * HOW: 1. Require the 4-byte fixed header.
88 * 2. Validate all three lengths against the struct field sizes AND the
89 * payload length before copying anything.
90 * 3. Copy the three strings back to back into m.
91 */
92static int
93xmeta_decode_origin(brix_xmeta_t *m, const uint8_t *p, uint32_t plen)
94{
95 uint8_t el, al, cl;
96
97 if (plen < XMETA_ORIGIN_FIXED) {
98 return BRIX_XMETA_ERR;
99 }
100 el = p[0]; al = p[1]; cl = p[2];
101 if (el > sizeof(m->etag) || al > sizeof(m->cks_alg)
102 || cl > sizeof(m->cks_hex)
103 || plen < (uint32_t) XMETA_ORIGIN_FIXED + el + al + cl)
104 {
105 return BRIX_XMETA_ERR;
106 }
107 m->etag_len = el;
108 memcpy(m->etag, p + XMETA_ORIGIN_FIXED, el);
109 m->cks_alg_len = al;
110 memcpy(m->cks_alg, p + XMETA_ORIGIN_FIXED + el, al);
111 m->cks_len = cl;
112 memcpy(m->cks_hex, p + XMETA_ORIGIN_FIXED + el + al, cl);
113 return BRIX_XMETA_OK;
114}

116/* ---- Decode a DIGEST section payload ----
117 *
118 * WHAT: Takes an owned copy of the raw digest TLV list into m->digests.
119 * Returns BRIX_XMETA_OK, or BRIX_XMETA_ERR (ENOMEM) on allocation
120 * failure.
121 *
122 * WHY: The digest list is stored opaquely and parsed lazily by
123 * brix_xmeta_digest_get, which re-validates entry framing on every
124 * access — so the decode side only needs to own the bytes.
125 *
126 * HOW: 1. Allocate plen bytes (1 minimum so an empty list stays non-NULL).
127 * 2. Copy the payload, replace any previous list.
128 */
129static int
130xmeta_decode_digest(brix_xmeta_t *m, const uint8_t *p, uint32_t plen)
131{
132 uint8_t *copy = malloc(plen ? plen : 1);
133
134 if (copy == NULL) {
135 errno = ENOMEM;
136 return BRIX_XMETA_ERR;
137 }
138 memcpy(copy, p, plen);
139 free(m->digests);
140 m->digests = copy;
141 m->digests_len = plen;
142 return BRIX_XMETA_OK;
143}

145/* ---- Decode a BLOCKCRC section payload ----
146 *
147 * WHAT: Loads the per-block crc32c table from the {u32 granule, u32 rsv,
148 * u64 nblocks, u32 crc[]} payload into m->blockcrc. Returns
149 * BRIX_XMETA_OK, or BRIX_XMETA_ERR on a short payload, a block count
150 * that does not match the file, or allocation failure (ENOMEM).
151 *
152 * WHY: Pairs with xmeta_encode_blockcrc_section. The table is only usable if
153 * it describes exactly this file's block count, so a mismatch is
154 * treated as corruption, not skipped.
155 *
156 * HOW: 1. Require the 16-byte payload header.
157 * 2. Cross-check the embedded nblocks against m->nblocks and the
158 * payload length (also bounds the crc[] read).
159 * 3. Allocate and copy the table, replacing any previous one.
160 */
161static int
162xmeta_decode_blockcrc(brix_xmeta_t *m, const uint8_t *p, uint32_t plen)
163{
164 uint64_t nb;
165
166 if (plen < 16) {
167 return BRIX_XMETA_ERR;
168 }
169 memcpy(&nb, p + 8, 8);
170 if (nb != m->nblocks || plen != 16 + nb * 4) {
171 return BRIX_XMETA_ERR; /* table doesn't match the file */
172 }
173 free(m->blockcrc);
174 m->blockcrc = malloc(nb ? (size_t) nb * 4 : 4);
175 if (m->blockcrc == NULL) {
176 errno = ENOMEM;
177 return BRIX_XMETA_ERR;
178 }
179 memcpy(m->blockcrc, p + 16, (size_t) nb * 4);
180 m->have_blockcrc = 1;
181 return BRIX_XMETA_OK;
182}

184/* ---- Dispatch one extension section to its codec pair's decoder ----
185 *
186 * WHAT: Routes a crc-verified section payload to the per-type decoder.
187 * Returns the decoder's result; unknown types return BRIX_XMETA_OK.
188 *
189 * WHY: Unknown section types are skipped rather than rejected so newer
190 * writers stay readable by older code (forward compatibility).
191 *
192 * HOW: switch on the section type; one case per known section.
193 */
194static int
195xmeta_decode_section(brix_xmeta_t *m, uint16_t type, const uint8_t *p,
196 uint32_t plen)
197{
198 switch (type) {
199
200 case BRIX_XMETA_SEC_STATE:
201 return xmeta_decode_state(m, p, plen);
202
203 case BRIX_XMETA_SEC_ORIGIN:
204 return xmeta_decode_origin(m, p, plen);
205
206 case BRIX_XMETA_SEC_DIGEST:
207 return xmeta_decode_digest(m, p, plen);
208
209 case BRIX_XMETA_SEC_BLOCKCRC:
210 return xmeta_decode_blockcrc(m, p, plen);
211
212 default:
213 return BRIX_XMETA_OK; /* unknown type: skipped (fwd compat) */
214 }
215}

217/* ---- Decode the stock Store POD and populate the fixed fields ----
218 *
219 * WHAT: Reads version + Store + crc from the front of the record, validates
220 * them, and fills m's fixed fields (sizes, times, counters, nblocks).
221 * Returns BRIX_XMETA_OK; BRIX_XMETA_FOREIGN for a record we don't
222 * recognize as ours; BRIX_XMETA_ERR (EBADMSG) for a crc mismatch.
223 * Allocates nothing, so failure needs no cleanup.
224 *
225 * WHY: The FOREIGN/ERR distinction is load-bearing: a wrong version or
226 * insane sizes mean "not our record — leave it alone", while a crc
227 * mismatch on a recognized layout means "our record, torn".
228 *
229 * HOW: 1. Read + check the stock version (FOREIGN on mismatch/short).
230 * 2. Read Store + crc; verify crc32c(Store) (ERR on mismatch).
231 * 3. Sanity-check sizes and astat count (FOREIGN if insane).
232 * 4. Copy the fields into m; derive nblocks and bound it.
233 */
234static int
235xmeta_decode_stock_store(const uint8_t *buf, size_t len, size_t *off,
236 brix_xmeta_t *m)
237{
238 brix_xmeta_stock_store_t store;
239 int32_t version;
240 uint32_t crc;
241
242 if (xmeta_get(buf, len, off, &version, 4) != 0
243 || version != BRIX_XMETA_STOCK_VERSION)
244 {
245 return BRIX_XMETA_FOREIGN;
246 }
247 if (xmeta_get(buf, len, off, &store, sizeof(store)) != 0
248 || xmeta_get(buf, len, off, &crc, 4) != 0)
249 {
250 return BRIX_XMETA_FOREIGN;
251 }
252 if (crc != brix_crc32c_value(&store, sizeof(store))) {
253 errno = EBADMSG;
254 return BRIX_XMETA_ERR;
255 }
256 if (store.buffer_size <= 0 || store.file_size < 0
257 || store.astat_size < 0)
258 {
259 return BRIX_XMETA_FOREIGN;
260 }
261
262 m->buffer_size = store.buffer_size;
263 m->file_size = store.file_size;
264 m->creation_time = store.creation_time;
265 m->no_cksum_time = store.no_cksum_time;
266 m->access_cnt = store.access_cnt;
267 m->status_raw = store.status_raw;
268 m->astat_count = store.astat_size;
269 m->nblocks = xmeta_nblocks(store.file_size, store.buffer_size);
270 if (m->nblocks > BRIX_XMETA_MAX_BLOCKS) {
271 return BRIX_XMETA_FOREIGN;
272 }
273 return BRIX_XMETA_OK;
274}

276/* ---- Decode the stock bitmap + AStat records and verify their crc ----
277 *
278 * WHAT: Allocates and reads the present-block bitmap, consumes the AStat
279 * records (keeping the first), and verifies the shared trailing crc.
280 * Returns BRIX_XMETA_OK; BRIX_XMETA_FOREIGN on truncation;
281 * BRIX_XMETA_ERR (EBADMSG/ENOMEM) on crc mismatch or allocation
282 * failure. Frees m itself on any failure after the bitmap exists.
283 *
284 * WHY: Stock cinfo covers bitmap and AStat records with ONE crc32c, so they
285 * must be consumed and checksummed together; splitting them apart would
286 * break the coverage. Requires m->astat_count/nblocks from
287 * xmeta_decode_stock_store.
288 *
289 * HOW: 1. Allocate the bitmap (1 byte minimum for zero-block files) and
290 * read it; start the running crc over it.
291 * 2. Read each AStat record, keep the first, extend the crc over all.
292 * 3. Read the stored crc and compare against the running value.
293 */
294static int
295xmeta_decode_bitmap_astat(const uint8_t *buf, size_t len, size_t *off,
296 brix_xmeta_t *m)
297{
298 uint32_t crc, want;
299 uint16_t i;
300 size_t blen;
301
302 blen = xmeta_bitmap_len(m->nblocks);
303 m->bitmap = malloc(blen ? blen : 1);
304 if (m->bitmap == NULL) {
305 errno = ENOMEM;
306 return BRIX_XMETA_ERR;
307 }
308 if (xmeta_get(buf, len, off, m->bitmap, blen) != 0) {
309 brix_xmeta_free(m);
310 return BRIX_XMETA_FOREIGN;
311 }
312 want = brix_crc32c_value(m->bitmap, blen);
313
314 /* AStat records: keep the first, checksum them all */
315 for (i = 0; (int32_t) i < m->astat_count; i++) {
316 brix_xmeta_astat_t a;
317
318 if (xmeta_get(buf, len, off, &a, sizeof(a)) != 0) {
319 brix_xmeta_free(m);
320 return BRIX_XMETA_FOREIGN;
321 }
322 if (i == 0) {
323 m->astat = a;
324 }
325 want = brix_crc32c_extend(want, &a, sizeof(a));
326 }
327 if (xmeta_get(buf, len, off, &crc, 4) != 0) {
328 brix_xmeta_free(m);
329 return BRIX_XMETA_FOREIGN;
330 }
331 if (crc != want) {
332 brix_xmeta_free(m);
333 errno = EBADMSG;
334 return BRIX_XMETA_ERR;
335 }
336 return BRIX_XMETA_OK;
337}

339/* ---- Decode the optional XCX1 extension trailer ----
340 *
341 * WHAT: Consumes the extension header and its TLV sections, dispatching each
342 * crc-verified payload through xmeta_decode_section. Returns
343 * BRIX_XMETA_OK — including when there is no trailer or the trailer is
344 * foreign (the stock part stands) — or BRIX_XMETA_ERR (EBADMSG) for a
345 * truncated/corrupt section, freeing m before returning.
346 *
347 * WHY: The extension is optional by design: a pure stock cinfo ends at the
348 * prefix, and an unrecognized trailer (e.g. stock's own additions) must
349 * not invalidate the stock data we already decoded. But once the XCX1
350 * magic matched, a bad section means a torn write of OUR record — that
351 * is corruption, not foreignness.
352 *
353 * HOW: 1. off == len → no trailer, done.
354 * 2. Read magic/version/count; any mismatch → foreign trailer, done OK.
355 * 3. Per section: read the 8-byte TLV header, bound the payload length,
356 * verify crc32c(header+payload), then dispatch by type.
357 */
358static int
359xmeta_decode_ext_sections(const uint8_t *buf, size_t len, size_t *off,
360 brix_xmeta_t *m)
361{
362 uint32_t crc, ext_magic;
363 uint16_t ext_version, nsec, i;
364
365 /* extension (optional: a pure stock cinfo ends here) */
366 if (*off == len) {
367 return BRIX_XMETA_OK;
368 }
369 if (xmeta_get(buf, len, off, &ext_magic, 4) != 0
370 || ext_magic != BRIX_XMETA_EXT_MAGIC
371 || xmeta_get(buf, len, off, &ext_version, 2) != 0
372 || xmeta_get(buf, len, off, &nsec, 2) != 0)
373 {
374 return BRIX_XMETA_OK; /* foreign trailer: stock part stands */
375 }
376
377 for (i = 0; i < nsec; i++) {
378 uint16_t type, rsv;
379 uint32_t plen;
380 size_t sec_hdr = *off;
381
382 if (xmeta_get(buf, len, off, &type, 2) != 0
383 || xmeta_get(buf, len, off, &rsv, 2) != 0
384 || xmeta_get(buf, len, off, &plen, 4) != 0
385 || plen > BRIX_XMETA_MAX_SECTION
386 || len - *off < (size_t) plen + 4)
387 {
388 brix_xmeta_free(m);
389 errno = EBADMSG;
390 return BRIX_XMETA_ERR; /* truncated section = torn record */
391 }
392 memcpy(&crc, buf + *off + plen, 4);
393 if (crc != brix_crc32c_value(buf + sec_hdr, 8 + (size_t) plen)) {
394 brix_xmeta_free(m);
395 errno = EBADMSG;
396 return BRIX_XMETA_ERR;
397 }
398 if (xmeta_decode_section(m, type, buf + *off, plen)
399 != BRIX_XMETA_OK)
400 {
401 brix_xmeta_free(m);
402 errno = EBADMSG;
403 return BRIX_XMETA_ERR;
404 }
405 *off += (size_t) plen + 4;
406 }
407
408 return BRIX_XMETA_OK;
409}

411int
412brix_xmeta_decode(const uint8_t *buf, size_t len, brix_xmeta_t *m)
413{
414 size_t off = 0;
415 int rc;
416
417 if (buf == NULL || m == NULL) {
418 errno = EINVAL;
419 return BRIX_XMETA_ERR;
420 }
421 memset(m, 0, sizeof(*m));
422
423 /* stock prefix */
424 rc = xmeta_decode_stock_store(buf, len, &off, m);
425 if (rc != BRIX_XMETA_OK) {
426 return rc;
427 }
428 rc = xmeta_decode_bitmap_astat(buf, len, &off, m);
429 if (rc != BRIX_XMETA_OK) {
430 return rc;
431 }
432 return xmeta_decode_ext_sections(buf, len, &off, m);
433}
