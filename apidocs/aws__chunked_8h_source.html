1/*
2 * aws_chunked.h — AWS streaming (aws-chunked) request-body decoder (phase-43 W0).
3 *
4 * AWS SDKs upload with an application-layer chunk framing carried INSIDE the HTTP
5 * body (signalled by x-amz-content-sha256: STREAMING-*), distinct from HTTP
6 * Transfer-Encoding: chunked (which nginx strips). Each chunk is:
7 *
8 * <hex-size>[;chunk-signature=<64hex>]\r\n <size bytes payload> \r\n
9 *
10 * terminated by a 0-size chunk optionally followed by trailer headers (e.g.
11 * x-amz-checksum-crc32:<base64>) and a blank line. Without decoding, this
12 * framing would be stored verbatim — corrupting every object from a default
13 * modern aws-cli/boto3 client. This module decodes the payload to a file
14 * descriptor, enforces x-amz-decoded-content-length, and captures a trailer
15 * checksum for verification.
16 */
17

31/*
32 * W6a: per-chunk SigV4 signature-verification context. When `enabled`, the
33 * decoder verifies each chunk's `chunk-signature=` against
34 * HMAC(signing_key, "AWS4-HMAC-SHA256-PAYLOAD\n<amz_date>\n<scope>\n
35 * <prev-sig>\n<sha256("")>\n<sha256(chunk-data)>")
36 * rolling prev-sig from the seed (request Authorization) signature, and rejects
37 * a mismatch with 403. All fields come from the SigV4 auth already performed.
38 */
39typedef struct {
40 int enabled;
41 u_char signing_key[32];
42 char seed_signature[65]; /* hex, the request's Authorization signature */
43 char amz_date[32]; /* YYYYMMDDTHHMMSSZ */
44 char scope[96]; /* YYYYMMDD/region/s3/aws4_request */
45} s3_chunk_verify_t;

54/*
55 * s3_aws_chunked_has_inner_coding — 1 if the request's Content-Encoding names a
56 * content-coding OTHER than aws-chunked/identity (e.g. "aws-chunked,gzip").
57 *
58 * The streaming de-chunker strips only the chunk framing, so an inner-compressed
59 * payload would be committed still compressed. Callers reject such uploads (400)
60 * rather than store undecoded bytes — the same invariant the non-chunked PUT path
61 * enforces. Returns 0 for plain aws-chunked (no inner coding).
62 */
63int s3_aws_chunked_has_inner_coding(ngx_http_request_t *r);

66 * offload path pre-allocates one so the worker thread never touches r->pool). */
67#define S3_CHUNK_READ_WINDOW (64 * 1024)

69/*
70 * s3_aws_chunked_decode_to_fd — decode the framed request body to dst_fd.
71 *
72 * Iterates the nginx request-body chain (memory + spooled buffers), runs the
73 * chunk state machine, and pwrites only payload bytes to dst_fd. Enforces that
74 * the total decoded length equals decoded_len_expected. Any trailer checksum
75 * (x-amz-checksum-*) is recorded in *trailer.
76 *
77 * `window` is a caller-provided S3_CHUNK_READ_WINDOW-byte scratch buffer; pass
78 * NULL to have it allocated from r->pool (only safe on the event-loop thread).
79 *
80 * `verify` (may be NULL or .enabled=0) requests per-chunk SigV4 verification; a
81 * signature mismatch fails the decode with *http_status_out = 403.
82 *
83 * Returns NGX_OK on success; on failure returns NGX_ERROR and sets
84 * *http_status_out to the HTTP status to surface (400 malformed/short, 403
85 * signature mismatch, 500 I/O).
86 */
87ngx_int_t s3_aws_chunked_decode_to_fd(ngx_http_request_t *r, ngx_fd_t dst_fd,
88 const char *log_path, uint64_t decoded_len_expected,

90 const s3_chunk_verify_t *verify);
