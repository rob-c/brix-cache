35static EVP_PKEY *genkey(void)
36{
37 EVP_PKEY *k = NULL;

42 }
43 EVP_PKEY_CTX_free(c);

47static uint8_t *pem_of(X509 *x, size_t *len)
48{
49 BIO *b = BIO_new(BIO_s_mem());
50 char *d;
51 long n;
52 uint8_t *out;
53 PEM_write_bio_X509(b, x);
54 n = BIO_get_mem_data(b, &d);
55 out = malloc((size_t) n + 1);
56 memcpy(out, d, (size_t) n);
57 out[n] = '\0';
58 *len = (size_t) n;
59 BIO_free(b);
60 return out;
61}

63/* Make a cert (self-signed if ca==NULL, else signed by ca/cakey) with keyUsage,
64 * and CA:TRUE if is_ca. Returns the X509 + sets *key. */
65static X509 *mkcert(const char *cn, int is_ca, X509 *ca, EVP_PKEY *cakey,
66 EVP_PKEY **key)
67{
68 EVP_PKEY *k = genkey();
69 X509 *x = X509_new();
70 X509_NAME *n;
71 X509_EXTENSION *e;
72
73 *key = k;
74 X509_set_version(x, 2);
75 ASN1_INTEGER_set(X509_get_serialNumber(x), (long) (rand() % 1000000 + 2));
76 X509_gmtime_adj(X509_getm_notBefore(x), 0);
77 X509_gmtime_adj(X509_getm_notAfter(x), 3600);
78 X509_set_pubkey(x, k);
79 n = X509_get_subject_name(x);
80 X509_NAME_add_entry_by_txt(n, "O", MBSTRING_ASC, (const unsigned char *) "T", -1, -1, 0);
81 X509_NAME_add_entry_by_txt(n, "CN", MBSTRING_ASC, (const unsigned char *) cn, -1, -1, 0);
82 X509_set_issuer_name(x, ca ? X509_get_subject_name(ca) : n);
83 if (is_ca) {
84 e = X509V3_EXT_conf_nid(NULL, NULL, NID_basic_constraints, "critical,CA:TRUE");
85 if (e) { X509_add_ext(x, e, -1); X509_EXTENSION_free(e); }
86 e = X509V3_EXT_conf_nid(NULL, NULL, NID_key_usage, "critical,keyCertSign,cRLSign");
87 if (e) { X509_add_ext(x, e, -1); X509_EXTENSION_free(e); }
88 } else {
89 e = X509V3_EXT_conf_nid(NULL, NULL, NID_key_usage,
90 "critical,digitalSignature,keyEncipherment");
91 if (e) { X509_add_ext(x, e, -1); X509_EXTENSION_free(e); }
92 }
93 X509_sign(x, ca ? cakey : k, EVP_sha256());
94 return x;
95}

97/* Verify `leaf` chains to `ca` through the `inter` certs, allowing proxies. */
98static int verify_proxy(X509 *leaf, X509 *ca, X509 **inter, int ninter)
99{
100 X509_STORE *store = X509_STORE_new();
101 STACK_OF(X509) *unt = sk_X509_new_null();
102 X509_STORE_CTX *ctx = X509_STORE_CTX_new();
103 int rc, i;
104
105 X509_STORE_add_cert(store, ca);
106 X509_STORE_set_flags(store, X509_V_FLAG_ALLOW_PROXY_CERTS);
107 for (i = 0; i < ninter; i++) sk_X509_push(unt, inter[i]);
108 X509_STORE_CTX_init(ctx, store, leaf, unt);
109 rc = X509_verify_cert(ctx);
110 if (rc != 1) {
111 printf(" verify err: %s\n",
112 X509_verify_cert_error_string(X509_STORE_CTX_get_error(ctx)));
113 }
114 X509_STORE_CTX_free(ctx);
115 sk_X509_free(unt);
116 X509_STORE_free(store);
117 return rc == 1;
118}

190/* WHAT: stage 2 — issue/sign the proxy from the request (brix_gsi_sign_pxyreq).
191 * WHY: the issued proxy must chain to the EEC and carry a critical PCI.
192 * HOW: sign, parse the PEM, and check issuer/key/pubkey/PCI/subject shape. */
193static void test_sign_pxyreq(suite_t *s)
194{
195 printf("[sign_pxyreq]\n");
196 const brix_gsi_blob_t eec_blob = { s->eec_pem, s->eec_len };
197 const brix_gsi_blob_t req_blob = { s->reqder, s->reqlen };
198 brix_gsi_buf_t pxy_out = { NULL, 0 };
199 int rc = brix_gsi_sign_pxyreq(&eec_blob, s->eeckey, &req_blob, &pxy_out,
200 s->esink);
201 s->pxy_pem = pxy_out.data; s->pxy_len = pxy_out.len;
202 CHECK(rc == 0, "sign_pxyreq issues a proxy");
203 if (rc) printf(" err: %s\n", s->err);
204 s->pxy = s->pxy_pem ? x509_from_pem(s->pxy_pem, s->pxy_len) : NULL;
205 CHECK(s->pxy != NULL, "issued proxy PEM parses");
206 if (s->pxy) {
207 CHECK(X509_NAME_cmp(X509_get_issuer_name(s->pxy),
208 X509_get_subject_name(s->eec)) == 0,
209 "proxy issuer == EEC subject");
210 CHECK(X509_verify(s->pxy, s->eeckey) == 1, "proxy signed by EEC key");
211 EVP_PKEY *ppub = X509_get_pubkey(s->pxy);
212 CHECK(ppub && EVP_PKEY_eq(ppub, s->reqkey) == 1, "proxy pubkey == request key");
213 EVP_PKEY_free(ppub);
214 int crit_pci = 0, i;
215 for (i = 0; i < X509_get_ext_count(s->pxy); i++) {
216 X509_EXTENSION *e = X509_get_ext(s->pxy, i);
217 char o[128];
218 OBJ_obj2txt(o, sizeof(o), X509_EXTENSION_get_object(e), 1);
219 if (strcmp(o, "1.3.6.1.5.5.7.1.14") == 0)
220 crit_pci = X509_EXTENSION_get_critical(e);
221 }
222 CHECK(crit_pci, "proxy has a critical proxyCertInfo");
223 /* subject is the EEC subject + one extra CN RDN */
224 CHECK(X509_NAME_entry_count(X509_get_subject_name(s->pxy))
225 == X509_NAME_entry_count(X509_get_subject_name(s->eec)) + 1,
226 "proxy subject = EEC + 1 RDN");
227 }

230/* WHAT: stage 3 — assemble the delegated credential (brix_gsi_assemble_proxy).
231 * WHY: the credential must be proxy||chain||key — a complete credential the
232 * holder can authenticate with — and reject a mismatched key.
233 * HOW: assemble with the matching key, count the certs, extract the embedded
234 * private key and match it to reqkey, then retry with a wrong key. */
235static void test_assemble_proxy(suite_t *s)
236{
237 printf("[assemble_proxy]\n");
238 size_t cred_len = 0;
239 const brix_gsi_blob_t eec_blob = { s->eec_pem, s->eec_len };
240 const brix_gsi_blob_t pxy_blob = { s->pxy_pem, s->pxy_len };
241 brix_gsi_buf_t cred_out = { NULL, 0 };
242 int rc = brix_gsi_assemble_proxy(&pxy_blob, s->reqkey, &eec_blob, &cred_out,
243 s->esink);
244 s->cred = cred_out.data; cred_len = cred_out.len;
245 CHECK(rc == 0, "assemble_proxy succeeds with matching key");
246 CHECK(s->cred && cred_len > s->pxy_len + s->eec_len,
247 "credential = proxy + chain + key");
248 /* count certs in the assembled credential (cert-reader must stop cleanly
249 * at the trailing key block) */
250 if (s->cred) {
251 BIO *b = BIO_new_mem_buf(s->cred, (int) cred_len);
252 int cnt = 0; X509 *t;
253 while ((t = PEM_read_bio_X509(b, NULL, NULL, NULL))) { cnt++; X509_free(t); }
254 BIO_free(b);
255 CHECK(cnt == 2, "assembled credential holds 2 certs (proxy+EEC)");
256 }
257 /* the embedded private key must be present and be the request key */
258 if (s->cred) {
259 BIO *b = BIO_new_mem_buf(s->cred, (int) cred_len);
260 EVP_PKEY *ck = b ? PEM_read_bio_PrivateKey(b, NULL, NULL, NULL) : NULL;
261 CHECK(ck != NULL, "assembled credential holds a private key");
262 CHECK(ck != NULL && EVP_PKEY_eq(ck, s->reqkey) == 1,
263 "embedded private key == request key");
264 EVP_PKEY_free(ck);
265 BIO_free(b);
266 }
267
268 /* key mismatch must fail */
269 EVP_PKEY *wrong = genkey();
270 brix_gsi_buf_t bad_out = { NULL, 0 };
271 rc = brix_gsi_assemble_proxy(&pxy_blob, wrong, &eec_blob, &bad_out, s->esink);
272 CHECK(rc == -1 && bad_out.data == NULL, "assemble rejects a mismatched key");
273 EVP_PKEY_free(wrong);
274}

287/* WHAT: stage 5 — delegate a proxy of a proxy (two-level chain).
288 * WHY: nested delegation and path length must still verify to the CA.
289 * HOW: request+sign a level-2 proxy off level-1, check issuer and chain. */

291{
292 printf("[two-level delegation]\n");
293 const brix_gsi_blob_t pxy_blob = { s->pxy_pem, s->pxy_len };
294 EVP_PKEY *rk2 = NULL; uint8_t *rd2 = NULL; size_t rl2 = 0;
295 brix_gsi_buf_t rd2_out = { NULL, 0 };
296 int r = brix_gsi_build_pxyreq(&pxy_blob, &rk2, &rd2_out, s->esink);
297 rd2 = rd2_out.data; rl2 = rd2_out.len;
298 CHECK(r == 0, "build_pxyreq from a proxy (level 2)");
299 uint8_t *pxy2_pem = NULL; size_t pxy2_len = 0;
300 const brix_gsi_blob_t rd2_blob = { rd2, rl2 };
301 brix_gsi_buf_t pxy2_out = { NULL, 0 };
302 r = brix_gsi_sign_pxyreq(&pxy_blob, s->reqkey, &rd2_blob, &pxy2_out,
303 s->esink);
304 pxy2_pem = pxy2_out.data; pxy2_len = pxy2_out.len;
305 CHECK(r == 0, "sign level-2 request with the level-1 proxy");
306 if (r) printf(" err: %s\n", s->err);
307 X509 *pxy2 = pxy2_pem ? x509_from_pem(pxy2_pem, pxy2_len) : NULL;
308 CHECK(pxy2 != NULL, "level-2 proxy parses");
309 if (pxy2) {
310 CHECK(X509_NAME_cmp(X509_get_issuer_name(pxy2),
311 X509_get_subject_name(s->pxy)) == 0,
312 "level-2 issuer == level-1 proxy subject");
313 X509 *inter2[2] = { s->pxy, s->eec };
314 CHECK(verify_proxy(pxy2, s->ca, inter2, 2),
315 "proxy2 -> proxy1 -> EEC -> CA verifies");
316 X509_free(pxy2);
317 }
318 EVP_PKEY_free(rk2); free(rd2); free(pxy2_pem);
319}
