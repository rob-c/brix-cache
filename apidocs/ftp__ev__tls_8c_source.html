5/*
6 * ftp_ev_tls.c — non-blocking PROT P data-channel TLS for the event engine.
7 *
8 * WHAT: brings a GSI-protected (DCAU A / PROT P) data connection up as a TLS
9 * session under the event loop, then hands the encrypted socket to the transfer
10 * pump. The cert logic (present the delegated user credential, verify the peer
11 * proxy chain, pin its DN to the control identity) is the shared ftp_dc_sec core;
12 * this file only drives the *handshake* non-blocking.
13 *
14 * WHY: the sync engine blocks the worker inside SSL_accept()/SSL_connect(). Here
15 * the handshake runs through nginx's own SSL connection layer (ngx_ssl_handshake),
16 * which arms the data connection's read/write events on WANT_READ/WANT_WRITE and
17 * calls back when the handshake settles — so the worker keeps serving other
18 * sessions. Once the handshake completes nginx swaps the connection's recv/send
19 * vtable to ngx_ssl_recv/ngx_ssl_write, so the RETR/STOR/LIST pumps in
20 * ftp_ev_xfer.c move TLS records with no change: c->send/c->recv already speak TLS.
21 *
22 * HOW: the handshake driver is factored into two reusable primitives —
23 * brix_ftp_ev_tls_begin() (create pool + SSL + cred/policy, arm the handshake with
24 * a caller-supplied completion handler) and brix_ftp_ev_tls_verify() (the
25 * post-handshake identity gate). The single data connection wraps them behind
26 * brix_ftp_ev_dc_start_tls()/ev_dc_tls_done(); each MODE E child stream
27 * (ftp_ev_mode_e.c) reuses the same primitives with its own completion handler, so
28 * N parallel PROT P streams all handshake through one implementation.
29 */

63/* Drive the data-channel TLS handshake non-blocking. Gives the wrapped socket a
64 * per-connection pool (nginx SSL allocates its state on c->pool; a wrapped fd has
65 * none), creates the SSL from the gateway ctx, installs the delegated credential
66 * and TLS/proxy policy, sets `done` as the ssl->handler, and starts the handshake.
67 * The pool is returned via *poolp for the caller to release on teardown. */
68ngx_int_t
69brix_ftp_ev_tls_begin(ftp_ev_t *fc, ngx_connection_t *c, ngx_pool_t **poolp,
70 int tls_client, void (*done)(ngx_connection_t *))
71{
72 brix_ftp_dc_sec_t sec;
73 ngx_uint_t flags;
74 ngx_int_t rc;
75
76 if (fc->conf->tls_ctx == NULL) {
77 ngx_log_error(NGX_LOG_ERR, fc->c->log, 0,
78 "brix: GridFTP(ev) PROT P requested but no server TLS "
79 "context configured");
80 return NGX_ERROR;
81 }
82
83 *poolp = ngx_create_pool(1024, fc->c->log);
84 if (*poolp == NULL) {
85 return NGX_ERROR;
86 }
87 c->pool = *poolp;
88
89 /* The TLS role follows which side opened the socket: an active-mode connect
90 * (we dialled out) is the TLS client; a passive accept (the peer dialled in,
91 * incl. every MODE E child stream) is the TLS server. */
92 flags = tls_client ? NGX_SSL_CLIENT : 0;
93
94 if (ngx_ssl_create_connection(fc->conf->tls_ctx, c, flags) != NGX_OK) {
95 ngx_log_error(NGX_LOG_ERR, fc->c->log, 0,
96 "brix: GridFTP(ev) data-channel ngx_ssl_create_connection "
97 "failed");
98 return NGX_ERROR;
99 }
100
101 ev_dc_sec_of(fc, &sec);
102 if (brix_ftp_dc_load_deleg(&sec, c->ssl->connection) != NGX_OK) {
103 return NGX_ERROR;
104 }
105 brix_ftp_dc_apply_policy(c->ssl->connection);
106
107 c->ssl->handler = done;
108
109 rc = ngx_ssl_handshake(c);
110 if (rc == NGX_AGAIN) {
111 return NGX_AGAIN; /* ssl->handler fires on completion */
112 }
113 if (rc == NGX_OK) {
114 done(c); /* completed inline (loopback) */
115 return NGX_OK;
116 }
117 return NGX_ERROR;
118}

121/* nginx ssl->handler for the single data connection: enforce the GSI DN pin, then
122 * start the transfer — or fail closed. */
123static void
124ev_dc_tls_done(ngx_connection_t *c)
125{

136}
