11/* Remove the source file at `url` after a successful transfer (local => unlink;
12 * root:// => connect + brix_rm). Called only when the copy (including any
13 * --cksum/--verify check) reported success.
14 *
15 * WHAT: best-effort post-transfer source deletion ("move" semantics).
16 * WHY: a plain unlink/rm is the correct model — the copy is complete and the
17 * source is now redundant. Web sources (davs/http/s3) are rejected at the
18 * validation stage in xrdcp.c before we reach here, so XRDC_SCHEME_LOCAL
19 * and XRDC_SCHEME_ROOT/ROOTS cover every reachable source.
20 * HOW: parse the URL to discriminate local from root://; for local: unlink the
21 * path directly; for root://: open a fresh connection and issue brix_rm.
22 * Returns -1 on any failure (the caller logs a warning and continues — the
23 * transfer itself succeeded). */
24static int
25remove_source_entry(const char *url, const brix_opts *co)
26{
27 brix_url u;
28 brix_status st;
29
30 brix_status_clear(&st);
31 if (brix_url_parse(url, &u, &st) != 0) {
32 return -1;
33 }
34 if (u.scheme == XRDC_SCHEME_LOCAL) {
35 return unlink(u.path);
36 }
37 if (u.scheme == XRDC_SCHEME_ROOT || u.scheme == XRDC_SCHEME_ROOTS) {
38 brix_conn c;
39 int rc;
40 if (brix_connect(&c, &u, co, &st) != 0) { return -1; }
41 rc = brix_rm(&c, u.path, &st);
42 brix_close(&c);
43 return rc;
44 }
45 return -1;
46}

49/* Copy one src->dst, retrying up to `retries` times with capped exponential backoff. */
50int
51copy_one_with_retry(const char *src, const char *dst, const brix_copy_opts *o,
52 const brix_opts *co, int retries, brix_status *st)
53{
54 int attempt = 0;
55 /* web->web has no direct wire path — stage through a local temp. Each relay
56 * leg is web<->local (never web->web), so it re-enters here on the normal
57 * path with no recursion. */
58 if (both_web(src, dst)) {
59 return relay_web_to_web(src, dst, o, co, retries, st);
60 }
61 for (;;) {
62 brix_status_clear(st);
63 if (brix_copy(src, dst, o, co, st) == 0) {
64 return 0;
65 }
66 if (attempt >= retries) {
67 return -1;
68 }
69 {
70 int backoff = 1 << (attempt < 5 ? attempt : 5); /* 1,2,4,8,16,32 -> cap */
71 unsigned half_ms, wait_ms;
72 struct timespec ts;
73 if (backoff > 30) { backoff = 30; }
74 /* Phase 40 (a): "equal jitter" — wait half the backoff plus a random
75 * slice of the other half, so many clients/jobs retrying in lockstep
76 * spread their attempts instead of stampeding the server together. */
77 half_ms = (unsigned) backoff * 500u; /* backoff/2 in ms */
78 wait_ms = half_ms + brix_jitter_ms(half_ms + 1u);
79 if (!o->silent) {
80 fprintf(stderr, "xrdcp: %s failed (%s); retry %d/%d in %.1fs\n",
81 src, st->msg, attempt + 1, retries, wait_ms / 1000.0);
82 }
83 ts.tv_sec = wait_ms / 1000u;
84 ts.tv_nsec = (long) (wait_ms % 1000u) * 1000000L;
85 (void) nanosleep(&ts, NULL);
86 }
87 attempt++;
88 }
89}

92/* Size + mtime of a regular file at `url` (root:// or local). 0 with *size and
93 * *mtime set if it exists as a regular file; -1 otherwise (missing, a directory,
94 * web, or error). The -1-on-web behavior keeps --sync always copying web
95 * targets: an undeterminable comparison must copy, never skip. */
96int
97entry_meta(const char *url, const brix_opts *co, long long *size, long long *mtime)
98{
99 brix_url u;
100 brix_status st;
101 if (brix_is_web_url(url)) {
102 return -1; /* no cheap stat for web; --sync always copies web targets */
103 }
104 brix_status_clear(&st);
105 if (brix_url_parse(url, &u, &st) != 0) {
106 return -1;
107 }
108 if (u.scheme == XRDC_SCHEME_LOCAL) {
109 struct stat sb;
110 if (stat(u.path, &sb) != 0 || !S_ISREG(sb.st_mode)) { return -1; }
111 *size = (long long) sb.st_size;
112 *mtime = (long long) sb.st_mtime;
113 return 0;
114 }
115 if (u.scheme == XRDC_SCHEME_ROOT || u.scheme == XRDC_SCHEME_ROOTS) {
116 brix_conn c;
117 brix_statinfo si;
118 int ok;
119 if (brix_connect(&c, &u, co, &st) != 0) { return -1; }
120 ok = (brix_stat(&c, u.path, &si, &st) == 0 && !(si.flags & kXR_isDir));
121 if (ok) {
122 *size = (long long) si.size;
123 *mtime = (long long) si.mtime;
124 }
125 brix_close(&c);
126 return ok ? 0 : -1;
127 }
128 return -1;
129}

132/* Digest of a file at `url` (local: computed via brix_cksum_fd; root://: server
133 * kXR_Qcksum). Writes the lowercase hex digest into hex[hexsz]. 0 / -1.
134 *
135 * WHY a fresh connection for root://: this runs in the single/batch copy paths
136 * where no walker connection exists; the recursive walkers use their own helper
137 * that reuses the open conn instead. */
138static int
139entry_cksum(const char *url, const brix_opts *co, const char *algo,
140 char *hex, size_t hexsz)
141{
142 brix_url u;
143 brix_status st;
144
145 brix_status_clear(&st);
146 if (brix_is_web_url(url) || brix_url_parse(url, &u, &st) != 0) {
147 return -1;
148 }
149 if (u.scheme == XRDC_SCHEME_LOCAL) {
150 brix_cksum_algo a;
151 int fd, rc;
152 if (brix_cksum_algo_parse(algo, &a) != 0) { return -1; }
153 fd = open(u.path, O_RDONLY);
154 if (fd < 0) { return -1; }
155 rc = brix_cksum_fd(fd, a, hex, hexsz, &st);
156 close(fd);
157 return rc;
158 }
159 if (u.scheme == XRDC_SCHEME_ROOT || u.scheme == XRDC_SCHEME_ROOTS) {
160 brix_conn c;
161 int rc;
162 if (brix_connect(&c, &u, co, &st) != 0) { return -1; }
163 rc = brix_query_cksum(&c, u.path, algo, hex, hexsz, &st);
164 brix_close(&c);
165 return rc;
166 }
167 return -1;
168}

171/* 1 = both ends have the same <algo> digest; 0 = differ or undeterminable
172 * (undeterminable must copy — a sync that silently skips on error is data loss). */
173static int
174sync_cksum_equal(const char *src, const char *dst, const char *algo,
175 const brix_opts *co)
176{
177 char shex[132], dhex[132];
178
179 if (entry_cksum(src, co, algo, shex, sizeof(shex)) != 0
180 || entry_cksum(dst, co, algo, dhex, sizeof(dhex)) != 0) {
181 return 0;
182 }
183 return strcasecmp(shex, dhex) == 0;
184}

187/* ---- Filter guard for transfer_one (returns 1 = filtered/skip, 0 = proceed) ----
188 *
189 * WHAT: decide whether `src` is excluded by the name filters. Returns 1 when the
190 * item is filtered (the caller treats that like a skip), 0 to proceed.
191 * Caches the source classification in *src_meta / *src_sz / *src_mt for the
192 * recursive path so the sync block never re-stats the same source.
193 *
194 * WHY: the filter is applied to the BASENAME, covering the single-file and
195 * batch-copy paths. When o->recursive is set the walkers apply the filter
196 * per-file internally, so a directory source must NOT be filtered here
197 * (that would wrongly test the top-level dir name against, e.g., *.log). A
198 * plain regular file supplied with -r never reaches a walker, so it must
199 * still be filtered here; entry_meta() returns 0 only for regular files,
200 * which distinguishes that case precisely.
201 *
202 * HOW: 1. recursive: classify src via entry_meta (cached in *src_meta); filter
203 * only when it is a regular file (src_meta == 0) whose basename fails
204 * the match.
205 * 2. non-recursive: always apply the basename filter.
206 * 3. return 1 when filtered, 0 otherwise. */
207static int
208transfer_filter_check(const char *src, const char *base, const brix_copy_opts *o,
209 const brix_opts *co, int *src_meta,
210 long long *src_sz, long long *src_mt)
211{
212 if (o->recursive) {
213 *src_meta = entry_meta(src, co, src_sz, src_mt);
214 if (*src_meta == 0 && !brix_copy_filter_match(o, base)) {
215 return 1; /* filtered - like a skip */
216 }
217 return 0;
218 }
219 if (!brix_copy_filter_match(o, base)) {
220 return 1; /* filtered - like a skip */
221 }
222 return 0;
223}

226/* ---- Sync up-to-date check for transfer_one (returns 1 = skip, 0 = copy) ----
227 *
228 * WHAT: decide whether src and dst are already in sync and the copy can be
229 * skipped. Returns 1 to skip (up-to-date), 0 to fall through to the copy.
230 * Fetches src meta on demand (caching it in *src_meta / *src_sz / *src_mt)
231 * when the filter guard did not already classify it.
232 *
233 * WHY: a sync must skip ONLY when both endpoints stat as regular files AND the
234 * configured comparison (size / mtime / cksum via o->sync_cmp) agrees. Any
235 * undeterminable side (web, missing, error) must fall through to the copy -
236 * the data-loss rule: a sync that silently skips on error loses updates.
237 *
238 * HOW: 1. lazily populate src meta if still unfetched (src_meta == -2).
239 * 2. require src regular, dst statable regular, and brix_sync_should_skip.
240 * 3. for the cksum comparison, additionally require both digests equal.
241 * 4. return 1 only when every condition holds; otherwise 0. */
242static int
243transfer_sync_check(const char *src, const char *dst, const brix_copy_opts *o,
244 const brix_opts *co, int *src_meta,
245 long long *src_sz, long long *src_mt)
246{
247 long long dsz = 0, dmt = 0;
248
249 /* Fetch src meta on demand for the non-recursive path (not yet classified by
250 * the filter guard); reuse the cached result for the recursive path. Any
251 * failure leaves src_meta != 0 -> falls through to the copy. */
252 if (*src_meta == -2) {
253 *src_meta = entry_meta(src, co, src_sz, src_mt);
254 }
255 if (*src_meta == 0
256 && entry_meta(dst, co, &dsz, &dmt) == 0
257 && brix_sync_should_skip(o->sync_cmp, *src_sz, *src_mt, dsz, dmt)) {
258 if (o->sync_cmp != XRDC_SYNC_CKSUM
259 || sync_cksum_equal(src, dst,
260 o->sync_cksum_algo ? o->sync_cksum_algo : "adler32", co)) {
261 return 1; /* up-to-date - skip */
262 }
263 }
264 return 0;
265}

268/* Transfer src -> dst honoring filters, --sync and --dry-run.
269 * Returns 0 (copied), 1 (skipped), or -1 (failed, st set).
270 *
271 * Filter is applied to the BASENAME of src here, covering the single-file and
272 * batch-copy paths. When o->recursive is set the walkers (copy_tree_download,
273 * copy_tree_upload) apply the filter per-file internally, so we skip the check
274 * for directory-source callers (applying it there would incorrectly test the
275 * top-level source directory name, e.g. "mydir", against *.log).
276 * A plain regular file supplied with -r — e.g. `xrdcp -r --exclude '*.log'
277 * a.log remote/` — does NOT go through a walker, so it must still be filtered
278 * here. entry_meta() returns 0 only for regular files (local or root://); -1
279 * for directories, web, and errors. That lets us apply the check precisely
280 * for the plain-file-with-r case without re-filtering recursive walker outputs.
281 *
282 * Sync: skip only when both endpoints stat as regular files AND the configured
283 * comparison (size / mtime / cksum via o->sync_cmp) says they match. Any
284 * undeterminable side (web, missing, error) falls through to the copy — the
285 * data-loss rule: a sync that silently skips on error loses updates.
286 *
287 * Meta-cache: for the recursive path the filter guard calls entry_meta(src)
288 * to classify the source type; the result is stored in src_meta / src_sz /
289 * src_mt so the sync block never issues a second stat on the same source.
290 * For the non-recursive path, src's meta is deferred until the sync block
291 * needs it, avoiding an extra stat on every single-file copy. */
292int
293transfer_one(const char *src, const char *dst, const brix_copy_opts *o,
294 const brix_opts *co, int retries, int sync_mode, brix_status *st)
295{
296 char base[XRDC_NAME_MAX];
297 long long src_sz = 0, src_mt = 0;
298 int src_meta = -2; /* -2 = not yet fetched; -1 = not a reg file; 0 = reg file */
299
300 path_basename(src, base, sizeof(base));
301 /* Filter guard: classifies the source (recursive path) into src_meta so the
302 * sync block below can reuse it without a second round-trip to the server. */
303 if (transfer_filter_check(src, base, o, co, &src_meta, &src_sz, &src_mt)) {
304 return 1; /* filtered — like a skip */
305 }
306 if ((sync_mode || o->sync)
307 && transfer_sync_check(src, dst, o, co, &src_meta, &src_sz, &src_mt)) {
308 return 1; /* up-to-date — skip */
309 }
310 /* When --delete is active the walker must run even in dry-run mode: it
311 * needs to list both sides to report which extras would be removed. Skip
312 * the early-out so copy_one_with_retry drives the full recursive walk;
313 * the walker prints its own per-entry "[dry-run] copy/delete" messages and
314 * guards every mutation with !o->dry_run internally.
315 * For all other dry-run cases, print one top-level line and return. */
316 if (o->dry_run && !o->sync_delete) {
317 printf("[dry-run] copy %s -> %s%s\n", src, dst,
318 o->remove_source ? " (then remove source)" : "");
319 return 0;
320 }
321 if (copy_one_with_retry(src, dst, o, co, retries, st) != 0) {
322 return -1;
323 }
324 /* For non-recursive copies, remove the source; recursive removal is already
325 * handled per-file and per-directory inside copy_recursive. */
326 if (o->remove_source && !o->recursive && remove_source_entry(src, co) != 0) {
327 fprintf(stderr, "xrdcp: warning: transferred but could not remove "
328 "source %s\n", src);
329 }
330 return 0;
331}

334/* Relay a web->web copy (e.g. davs://a/f -> s3://b/k) by staging through a private
335 * local temp file: download src into it, then upload it to dst. The wire has no
336 * direct web->web op and brix_http_upload needs a seekable/sized body, so a temp
337 * is the only correct path. The temp is created 0600 via mkstemp in $TMPDIR and
338 * unlinked on every return. Note the download leg rewrites the temp via its own
339 * temp+rename (which lands 0644), so we re-tighten it to 0600 before the upload
340 * leg, keeping the staged bytes private during the (longer) upload window. Each
341 * leg is web<->local, so cancellation is only as prompt as a single web transfer
342 * (a timeout/EINTR boundary), not instantaneous. */
343int
344relay_web_to_web(const char *src, const char *dst, const brix_copy_opts *o,
345 const brix_opts *co, int retries, brix_status *st)
346{
347 const char *tmpdir = getenv("TMPDIR");
348 char tmpl[XRDC_PATH_MAX];
349 brix_copy_opts leg;
350 int fd, rc;
351
352 if (tmpdir == NULL || tmpdir[0] == '\0') { tmpdir = "/tmp"; }
353 if ((size_t) snprintf(tmpl, sizeof(tmpl), "%s/xrdcp-w2w-XXXXXX", tmpdir)
354 >= sizeof(tmpl)) {
355 brix_status_set(st, XRDC_EUSAGE, 0, "web->web: temp path too long");
356 return -1;
357 }
358 fd = mkstemp(tmpl);
359 if (fd < 0) {
360 brix_status_set(st, XRDC_ESOCK, errno,
361 "web->web: mkstemp in %s: %s", tmpdir, strerror(errno));
362 return -1;
363 }
364 close(fd); /* the download leg reopens by path */
365
366 if (!o->silent) {
367 fprintf(stderr, "xrdcp: %s -> %s (web->web via local temp)\n", src, dst);
368 }
369 /* Leg 1: download src -> our private temp. Force-overwrite the empty mkstemp
370 * file (it is ours) and never recurse. */
371 leg = *o;
372 leg.force = 1;
373 leg.recursive = 0;
374 rc = copy_one_with_retry(src, tmpl, &leg, co, retries, st);
375 if (rc == 0) {
376 /* The download's temp+rename left the staged file group/other-readable;
377 * re-tighten before the upload so the bytes aren't world-readable in a
378 * shared /tmp for the whole upload. */
379 (void) chmod(tmpl, S_IRUSR | S_IWUSR);
380 /* Leg 2: upload temp -> dst, honouring the user's real force/posc/creds. */
381 leg = *o;
382 leg.recursive = 0;
383 rc = copy_one_with_retry(tmpl, dst, &leg, co, retries, st);
384 }
385 (void) unlink(tmpl);
386 return rc;
387}

411/* Shared state for the parallel batch worker pool. Each brix_copy() is fully
412 * independent (its own connection + fds), so workers only contend on this counter
413 * block + stderr, guarded by one mutex. The journal (b->jrn) is NULL when
414 * journalling is disabled; its write path is internally serialized, so workers
415 * call brix_journal_mark without holding b->lock. */
416
417/* WHAT: consume batch items from b->items until exhausted; skip journal-completed
418 * sources, call batch_copy_one for the rest, and mark successes.
419 * WHY: each worker is independent (no shared connection/fd), so concurrency is
420 * safe. Journal has() reads an immutable sorted array (thread-safe); mark()
421 * is internally mutex-protected.
422 * HOW: claim item by incrementing b->next under b->lock (only the counter claim
423 * is locked — the copy itself runs unlocked), then lock again to update
424 * counters and print. Journal mark runs after releasing b->lock (different
425 * mutex; avoids unnecessary contention). */
426void *
427batch_worker(void *arg)
428{
429 batch_ctx *b = (batch_ctx *) arg;
430 for (;;) {
431 size_t idx;
432 char dpath[XRDC_PATH_MAX];
433 brix_status st;
434 int rc;
435
436 pthread_mutex_lock(&b->lock);
437 idx = b->next++;
438 pthread_mutex_unlock(&b->lock);
439 if (idx >= b->n) {
440 break;
441 }
442 /* journal skip: done[] is immutable after load — bsearch is thread-safe */
443 if (b->jrn != NULL && brix_journal_has(b->jrn, b->items[idx])) {
444 pthread_mutex_lock(&b->lock);
445 b->skip++;
446 if (!b->o->silent) {
447 fprintf(stderr, "[%zu/%zu] %s (already transferred)\n",
448 b->ok + b->skip + b->fail, b->n, b->items[idx]);
449 }
450 pthread_mutex_unlock(&b->lock);
451 continue;
452 }
453 brix_status_clear(&st);
454 rc = batch_copy_one(b->items[idx], b->dst, b->o, b->co, b->retries,
455 b->sync_mode, dpath, sizeof(dpath), &st);
456 pthread_mutex_lock(&b->lock);
457 if (rc == 0) {
458 b->ok++;
459 if (!b->o->silent) {
460 fprintf(stderr, "[%zu/%zu] %s -> %s\n", b->ok + b->skip + b->fail,
461 b->n, b->items[idx], dpath);
462 }
463 } else if (rc == 1) {
464 b->skip++;
465 if (!b->o->silent) {
466 fprintf(stderr, "[%zu/%zu] %s (up-to-date)\n",
467 b->ok + b->skip + b->fail, b->n, b->items[idx]);
468 }
469 } else {
470 b->fail++;
471 fprintf(stderr, "xrdcp: %s: %s\n", b->items[idx], st.msg);
472 }
473 pthread_mutex_unlock(&b->lock);
474 /* mark after releasing b->lock; brix_journal_mark is internally serialized.
475 * dry-run completes nothing — the actual bytes were not moved. */
476 if (rc == 0 && b->jrn != NULL && !b->o->dry_run) {
477 (void) brix_journal_mark(b->jrn, b->items[idx]);
478 }
479 }
480 return NULL;
481}

484/* WHAT: run the batch with `jobs` concurrent workers (jobs>=2).
485 * WHY: concurrency amortizes per-connection and per-file latency over many items.
486 * HOW: initialize a batch_ctx with shared state, spawn worker threads (fall back
487 * to single-threaded drain on malloc/pthread failure), join, return counts. */
488void
489batch_parallel(char **items, size_t n, const char *dst, const brix_copy_opts *o,
490 const brix_opts *co, int retries, int sync_mode, int jobs,
491 brix_journal *jrn, size_t *ok, size_t *skip, size_t *fail)
492{
493 batch_ctx b;
494 pthread_t *th;
495 int j, spawned = 0;
496
497 memset(&b, 0, sizeof(b));
498 b.items = items; b.n = n; b.dst = dst; b.o = o; b.co = co; b.retries = retries;
499 b.sync_mode = sync_mode; b.jrn = jrn;
500 pthread_mutex_init(&b.lock, NULL);
501 th = (pthread_t *) malloc((size_t) jobs * sizeof(pthread_t));
502 if (th == NULL) {
503 /* Fall back to single-threaded drain on allocation failure. */
504 batch_worker(&b);
505 } else {
506 for (j = 0; j < jobs; j++) {
507 if (pthread_create(&th[j], NULL, batch_worker, &b) == 0) {
508 spawned++;
509 }
510 }
511 if (spawned == 0) {
512 batch_worker(&b); /* no thread started — drain inline */
513 }
514 for (j = 0; j < spawned; j++) {
515 pthread_join(th[j], NULL);
516 }
517 free(th);
518 }
519 pthread_mutex_destroy(&b.lock);
520 *ok = b.ok;
521 *skip = b.skip;
522 *fail = b.fail;
523}
