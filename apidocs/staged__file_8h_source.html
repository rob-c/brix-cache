1/*
2 * staged_file.h — Atomic temp-file write lifecycle.
3 *
4 * Provides brix_staged_file_t (a struct tracking fd, tmp_path, active flag) and three
5 * operations: open → commit/abort. Used by S3 PUT, WebDAV PUT for crash-safe writes:
6 * 1) Open a unique temp file with O_EXCL inside root_canon
7 * 2) Write data to the fd
8 * 3) Commit (rename temp → final path) or abort (close + unlink)
9 */

20/*
21 * brix_staged_file_t — Tracks the lifecycle of a staged (temp) file.
22 *
23 * Fields:
24 * fd — open file descriptor for the temp file (NGX_INVALID_FILE when closed)
25 * active — 1 while the temp file is open and data can be written; 0 after commit/abort
26 * tmp_path — filesystem path of the temp file (used by commit to rename, abort to unlink)
27 */
28typedef struct {
29 ngx_fd_t fd;
30 ngx_flag_t active;
31 char tmp_path[PATH_MAX];
32 mode_t final_mode; /* intended mode to publish at commit; the temp is
33 * kept 0600 during the write so a peer mapped uid
34 * cannot read an in-progress upload. */
35} brix_staged_file_t;
