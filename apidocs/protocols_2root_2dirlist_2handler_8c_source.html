25/*
26 * brix_dirlist_parse_request — decode the ClientDirlistRequest into walk
27 * state: option flags, checksum algorithm, and the validated request path.
28 *
29 * Rejects a missing path, an unsupported dcksm algorithm, an unparseable
30 * path payload, and any ".." component (the reference does not normalize
31 * ".."; dirlist resolves through the kernel RESOLVE_BENEATH which would
32 * collapse it).
33 *
34 * Returns 1 to continue; 0 when a response was already sent (*rc set).
35 */
36static int
37brix_dirlist_parse_request(brix_ctx_t *ctx, ngx_connection_t *c,
38 brix_dirlist_walk_t *walk, ngx_int_t *rc)
39{
40 xrdw_dirlist_req_t req;
41 char bad_algo[32];
42
43 xrdw_dirlist_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body,
44 &req);
45 walk->options = req.options;
46 walk->want_cksum = (walk->options & kXR_dcksm) ? 1 : 0;
47 walk->want_stat = (walk->options & (kXR_dstat | kXR_dcksm)) ? 1 : 0;
48
49 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
50 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", "-", "-",
51 kXR_ArgMissing, "no path given", rc);
52 }
53
54 if (walk->want_cksum
55 && brix_dirlist_checksum_algorithm(ctx->recv.payload,
56 ctx->recv.cur_dlen,
57 walk->cksum_algo,
58 sizeof(walk->cksum_algo),
59 bad_algo, sizeof(bad_algo))
60 != NGX_OK)
61 {
62 char errmsg[128];
63
64 snprintf(errmsg, sizeof(errmsg), "%s checksum not supported.",
65 bad_algo[0] ? bad_algo : "requested");
66 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", "-", "dcksm",
67 kXR_ServerError, errmsg, rc);
68 }
69
70 if (!walk->want_cksum) {
71 ngx_cpystrn((u_char *) walk->cksum_algo, (u_char *) "adler32",
72 sizeof(walk->cksum_algo));
73 }
74
75 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
76 walk->reqpath, sizeof(walk->reqpath), 1)) {
77 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", "-", "-",
78 kXR_ArgInvalid, "invalid path payload", rc);
79 }
80
81 if (brix_reject_dotdot_path(ctx, c, BRIX_OP_DIRLIST, "DIRLIST",
82 walk->reqpath)) {
83 *rc = ctx->write_rc;
84 return 0;
85 }
86
87 return 1;
88}

90/*
91 * brix_dirlist_check_redirect — apply the two cluster redirect modes before
92 * touching local storage.
93 *
94 * Static manager_map: explicit prefix→backend redirect (mirrors open/stat),
95 * so a static-map redirector serves dirlist too (go-hep ls = stat + dirlist).
96 * Manager mode: redirect dirlist to a registered data server, or fail with
97 * kXR_Overloaded when none is available.
98 *
99 * Returns 1 to continue serving locally; 0 when a response was already sent
100 * (*rc set).
101 */
102static int
103brix_dirlist_check_redirect(brix_ctx_t *ctx, ngx_connection_t *c,
104 ngx_stream_brix_srv_conf_t *conf, brix_dirlist_walk_t *walk,
105 ngx_int_t *rc)
106{
107 if (conf->manager_map != NULL) {
108 const brix_manager_map_t *m =
109 brix_find_manager_map(walk->reqpath, conf->manager_map);
110 if (m != NULL) {
111 brix_log_access(ctx, c, "DIRLIST", walk->reqpath, "manager_map",
112 1, kXR_ok, NULL, 0);
113 BRIX_OP_OK(ctx, BRIX_OP_DIRLIST);
114 *rc = brix_send_redirect(ctx, c, (const char *) m->host.data,
115 m->port);
116 return 0;
117 }
118 }
119
120 if (conf->manager_mode) {
121 char redir_host[256];
122 uint16_t redir_port;
123
124 if (brix_srv_select(walk->reqpath, 0, redir_host,
125 sizeof(redir_host), &redir_port)) {
126 brix_log_access(ctx, c, "DIRLIST", walk->reqpath, "registry",
127 1, kXR_ok, NULL, 0);
128 BRIX_OP_OK(ctx, BRIX_OP_DIRLIST);
129 *rc = brix_send_redirect(ctx, c, redir_host, redir_port);
130 return 0;
131 }
132 BRIX_OP_ERR(ctx, BRIX_OP_DIRLIST);
133 *rc = brix_send_error(ctx, c, kXR_Overloaded,
134 "no data server available");
135 return 0;
136 }
137
138 return 1;
139}

141/*
142 * brix_dirlist_open_dir — confine the request path, run the authorization
143 * gate, and open the directory through the VFS seam.
144 *
145 * Synchronous dirlist via the VFS seam: brix_vfs_opendir is
146 * impersonation-aware (broker fdopendir as the mapped user) and
147 * brix_vfs_readdir yields each name with an optional no-follow lstat. (The
148 * thread-pool AIO variant in aio/dirlist.c is currently disabled — it could
149 * complete without delivering a response frame, wedging xrdfs probes — so the
150 * request path stays synchronous.)
151 *
152 * Returns 1 with walk->dh open; 0 when a response was already sent (*rc set).
153 */
154static int
155brix_dirlist_open_dir(brix_ctx_t *ctx, ngx_connection_t *c,
156 ngx_stream_brix_srv_conf_t *conf, brix_dirlist_walk_t *walk,
157 ngx_int_t *rc)
158{
159 brix_vfs_ctx_t vctx;
160 int err = 0;
161
162 brix_beneath_full_path(conf->common.root_canon, walk->reqpath,
163 walk->full_path, sizeof(walk->full_path));
164
165 if (brix_auth_gate(ctx, c, BRIX_OP_DIRLIST, "DIRLIST",
166 walk->reqpath, walk->full_path, conf,
167 BRIX_AUTH_LOOKUP, 0) != NGX_OK) {
168 *rc = ctx->write_rc;
169 return 0;
170 }
171
172 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
173 conf->common.root_canon, NULL, 0 /* allow_write */, 0 /* is_tls */,
174 ctx->identity, walk->full_path);
175 brix_vfs_ctx_bind_backend_cred(&vctx,
176 &conf->common.storage_credential_dir,
177 conf->common.storage_credential_fallback);
178 brix_root_vfs_bind_deleg(ctx, conf, &vctx);
179 walk->dh = brix_vfs_opendir(&vctx, &err);
180 if (walk->dh == NULL) {
181 if (err == ENOTDIR) {
182 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", walk->reqpath,
183 "-", kXR_NotFile, "path is not a directory", rc);
184 }
185 if (err == ENOENT) {
186 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", walk->reqpath,
187 "-", kXR_NotFound, "directory not found", rc);
188 }
189 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", walk->reqpath,
190 "-", kXR_IOError, strerror(err), rc);
191 }
192
193 return 1;
194}

196/*
197 * brix_dirlist_alloc_chunk — allocate the header + 64KB chunk accumulator
198 * and seed the optional dstat lead-in sentinel.
199 *
200 * Guards against pool exhaustion from a flood of dirlist calls before
201 * charging the connection pool accounting; closes the directory handle on
202 * any failure so the caller can return immediately.
203 *
204 * Returns 1 with walk->chunk ready; 0 when a response was already sent or
205 * allocation failed (*rc set).
206 */
207static int
208brix_dirlist_alloc_chunk(brix_ctx_t *ctx, ngx_connection_t *c,
209 brix_dirlist_walk_t *walk, ngx_int_t *rc)
210{
211 if (ctx->login.pool_bytes_used + XRD_RESPONSE_HDR_LEN + walk->chunk_cap
212 > BRIX_MAX_CONN_POOL_BYTES)
213 {
214 brix_vfs_closedir(walk->dh, c->log);
215 ngx_log_error(NGX_LOG_WARN, c->log, 0,
216 "brix: dirlist pool limit reached (%uz bytes), "
217 "closing connection", ctx->login.pool_bytes_used);
218 *rc = brix_send_error(ctx, c, kXR_NoMemory,
219 "connection pool limit exceeded");
220 return 0;
221 }
222 walk->chunk = ngx_palloc(c->pool, XRD_RESPONSE_HDR_LEN + walk->chunk_cap);
223 if (walk->chunk == NULL) {
224 brix_vfs_closedir(walk->dh, c->log);
225 *rc = NGX_ERROR;
226 return 0;
227 }
228 ctx->login.pool_bytes_used += XRD_RESPONSE_HDR_LEN + walk->chunk_cap;
229
230 if (walk->want_stat) {
231 ngx_memcpy(walk->chunk + XRD_RESPONSE_HDR_LEN,
232 BRIX_DSTAT_LEADIN, BRIX_DSTAT_LEADIN_LEN);
233 walk->chunk_pos = BRIX_DSTAT_LEADIN_LEN;
234 }
235
236 return 1;
237}

239/*
240 * brix_handle_dirlist — handle kXR_dirlist: enumerate a directory and send
241 * the entries as a multi-frame kXR_oksofar ... kXR_ok response.
242 *
243 * Wire format (ClientDirlistRequest):
244 * options: bitfield controlling response content:
245 * kXR_dstat (0x01) — include per-entry stat info
246 * kXR_dcksm (0x02) — include per-entry checksum (implies kXR_dstat)
247 * payload: NUL-terminated path, optionally followed by CGI "?cks.type=algo"
248 *
249 * Response format: a flat text block with one entry per line, each terminated
250 * by '\n'. If kXR_dstat is set, each entry is followed by a stat body
251 * (stat-line format: "f|d|p flags mtime size"). Entries with control
252 * characters in their names are silently skipped to prevent wire corruption.
253 *
254 * A 65536-byte chunk buffer is accumulated and flushed as kXR_oksofar frames;
255 * the final flush uses kXR_ok to signal end-of-listing.
256 *
257 * Pipeline: parse request → cluster redirect check → authz + VFS opendir →
258 * chunk allocation → chunked entry streaming (helpers above, state in
259 * brix_dirlist_walk_t).
260 */
261/* §14 (phase-64): the fork/exec `xrdfs <origin> ls` namespace forward for a
262 * legacy cache_origin GSI cache is RETIRED with the cache_origin config model —
263 * a tier cache lists through the composed backend driver's opendir/readdir. */
264
265ngx_int_t
266brix_handle_dirlist(brix_ctx_t *ctx, ngx_connection_t *c,
267 ngx_stream_brix_srv_conf_t *conf)
268{
269 brix_dirlist_walk_t walk;
270 ngx_int_t rc;
271
272 ngx_memzero(&walk, sizeof(walk));
273 walk.chunk_cap = 65536;
274
275 if (!brix_dirlist_parse_request(ctx, c, &walk, &rc)) {
276 return rc;
277 }
278
279 if (!brix_dirlist_check_redirect(ctx, c, conf, &walk, &rc)) {
280 return rc;
281 }
282
283 if (!brix_dirlist_open_dir(ctx, c, conf, &walk, &rc)) {
284 return rc;
285 }
286
287 if (!brix_dirlist_alloc_chunk(ctx, c, &walk, &rc)) {
288 return rc;
289 }
290
291 return brix_dirlist_stream_entries(ctx, c, &walk);
292}
