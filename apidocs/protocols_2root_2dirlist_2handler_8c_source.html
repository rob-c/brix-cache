2 * Directory listing handler — implements the kXR_dirlist operation.
3 * Clients request a directory listing and receive entries as newline-delimited
4 * text, optionally with per-entry stat information and checksum tokens.
5 * The response is sent in 64KB chunks using kXR_oksofar continuation frames,
6 * ending with a single kXR_ok frame to signal completion.
7 */

24/*
25 * brix_dirlist_walk_t — per-request dirlist state threaded through the
26 * decode → redirect → open → stream pipeline below.
27 *
28 * Groups the decoded request options, resolved paths, the open VFS directory
29 * handle, and the 64KB chunk accumulator so each pipeline stage receives one
30 * explicit state pointer instead of a parameter fan-out.
31 */
32typedef struct {
33 u_char options; /* raw kXR_dirlist options */
34 ngx_flag_t want_stat; /* kXR_dstat or kXR_dcksm */
35 ngx_flag_t want_cksum; /* kXR_dcksm */
36 char cksum_algo[32]; /* negotiated cksum algo */
37 char reqpath[BRIX_MAX_PATH + 1]; /* client-supplied path */
38 char full_path[PATH_MAX]; /* confined absolute path */
39 brix_vfs_dir_t *dh; /* open directory handle */
40 u_char *chunk; /* hdr + data accumulator */
41 size_t chunk_cap; /* data capacity (64KB) */
42 size_t chunk_pos; /* bytes accumulated */
43} brix_dirlist_walk_t;

45/*
46 * brix_dirlist_entry_fmt_t — formatted per-entry output produced by
47 * brix_dirlist_entry_meta and consumed by brix_dirlist_append_entry.
48 *
49 * Carries the rendered stat line, the optional checksum token, whether stat
50 * data is valid for this entry, and the total wire bytes the entry needs so
51 * the streaming loop can decide when to flush a kXR_oksofar chunk.
52 */
53typedef struct {
54 char statbuf[256]; /* rendered stat line */
55 char cksum_token[EVP_MAX_MD_SIZE * 2 + 64]; /* "algo:hex" token */
56 ngx_flag_t have_stat; /* statbuf is valid */
57 size_t need; /* wire bytes required */
58} brix_dirlist_entry_fmt_t;

60/*
61 * brix_dirlist_name_is_unsafe — detect control characters in a directory
62 * entry name that would corrupt the newline-delimited kXR_dirlist wire format.
63 *
64 * The XRootD dirlist response separates entries with '\n'. A filename
65 * containing '\n' (or any other byte < 0x20) would be misinterpreted as a
66 * record separator by the client. Such filenames are skipped silently.
67 *
68 * Returns 1 (unsafe, skip this entry), 0 (safe to include).
69 */
70static ngx_flag_t
71brix_dirlist_name_is_unsafe(const char *name)
72{
73 const u_char *p;
74
75 for (p = (const u_char *) name; *p != '\0'; p++) {
76 if (*p < 0x20 || *p == 0x7f) {
77 return 1;
78 }
79 }
80
81 return 0;
82}

84/*
85 * brix_dirlist_parse_request — decode the ClientDirlistRequest into walk
86 * state: option flags, checksum algorithm, and the validated request path.
87 *
88 * Rejects a missing path, an unsupported dcksm algorithm, an unparseable
89 * path payload, and any ".." component (the reference does not normalize
90 * ".."; dirlist resolves through the kernel RESOLVE_BENEATH which would
91 * collapse it).
92 *
93 * Returns 1 to continue; 0 when a response was already sent (*rc set).
94 */
95static int
96brix_dirlist_parse_request(brix_ctx_t *ctx, ngx_connection_t *c,
97 brix_dirlist_walk_t *walk, ngx_int_t *rc)
98{
99 xrdw_dirlist_req_t req;
100 char bad_algo[32];
101
102 xrdw_dirlist_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body,
103 &req);
104 walk->options = req.options;
105 walk->want_cksum = (walk->options & kXR_dcksm) ? 1 : 0;
106 walk->want_stat = (walk->options & (kXR_dstat | kXR_dcksm)) ? 1 : 0;
107
108 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
109 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", "-", "-",
110 kXR_ArgMissing, "no path given", rc);
111 }
112
113 if (walk->want_cksum
114 && brix_dirlist_checksum_algorithm(ctx->recv.payload,
115 ctx->recv.cur_dlen,
116 walk->cksum_algo,
117 sizeof(walk->cksum_algo),
118 bad_algo, sizeof(bad_algo))
119 != NGX_OK)
120 {
121 char errmsg[128];
122
123 snprintf(errmsg, sizeof(errmsg), "%s checksum not supported.",
124 bad_algo[0] ? bad_algo : "requested");
125 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", "-", "dcksm",
126 kXR_ServerError, errmsg, rc);
127 }
128
129 if (!walk->want_cksum) {
130 ngx_cpystrn((u_char *) walk->cksum_algo, (u_char *) "adler32",
131 sizeof(walk->cksum_algo));
132 }
133
134 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
135 walk->reqpath, sizeof(walk->reqpath), 1)) {
136 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", "-", "-",
137 kXR_ArgInvalid, "invalid path payload", rc);
138 }
139
140 if (brix_reject_dotdot_path(ctx, c, BRIX_OP_DIRLIST, "DIRLIST",
141 walk->reqpath)) {
142 *rc = ctx->write_rc;
143 return 0;
144 }
145
146 return 1;
147}

149/*
150 * brix_dirlist_check_redirect — apply the two cluster redirect modes before
151 * touching local storage.
152 *
153 * Static manager_map: explicit prefix→backend redirect (mirrors open/stat),
154 * so a static-map redirector serves dirlist too (go-hep ls = stat + dirlist).
155 * Manager mode: redirect dirlist to a registered data server, or fail with
156 * kXR_Overloaded when none is available.
157 *
158 * Returns 1 to continue serving locally; 0 when a response was already sent
159 * (*rc set).
160 */
161static int
162brix_dirlist_check_redirect(brix_ctx_t *ctx, ngx_connection_t *c,
163 ngx_stream_brix_srv_conf_t *conf, brix_dirlist_walk_t *walk,
164 ngx_int_t *rc)
165{
166 if (conf->manager_map != NULL) {
167 const brix_manager_map_t *m =
168 brix_find_manager_map(walk->reqpath, conf->manager_map);
169 if (m != NULL) {
170 brix_log_access(ctx, c, "DIRLIST", walk->reqpath, "manager_map",
171 1, kXR_ok, NULL, 0);
172 BRIX_OP_OK(ctx, BRIX_OP_DIRLIST);
173 *rc = brix_send_redirect(ctx, c, (const char *) m->host.data,
174 m->port);
175 return 0;
176 }
177 }
178
179 if (conf->manager_mode) {
180 char redir_host[256];
181 uint16_t redir_port;
182
183 if (brix_srv_select(walk->reqpath, 0, redir_host,
184 sizeof(redir_host), &redir_port)) {
185 brix_log_access(ctx, c, "DIRLIST", walk->reqpath, "registry",
186 1, kXR_ok, NULL, 0);
187 BRIX_OP_OK(ctx, BRIX_OP_DIRLIST);
188 *rc = brix_send_redirect(ctx, c, redir_host, redir_port);
189 return 0;
190 }
191 BRIX_OP_ERR(ctx, BRIX_OP_DIRLIST);
192 *rc = brix_send_error(ctx, c, kXR_Overloaded,
193 "no data server available");
194 return 0;
195 }
196
197 return 1;
198}

200/*
201 * brix_dirlist_open_dir — confine the request path, run the authorization
202 * gate, and open the directory through the VFS seam.
203 *
204 * Synchronous dirlist via the VFS seam: brix_vfs_opendir is
205 * impersonation-aware (broker fdopendir as the mapped user) and
206 * brix_vfs_readdir yields each name with an optional no-follow lstat. (The
207 * thread-pool AIO variant in aio/dirlist.c is currently disabled — it could
208 * complete without delivering a response frame, wedging xrdfs probes — so the
209 * request path stays synchronous.)
210 *
211 * Returns 1 with walk->dh open; 0 when a response was already sent (*rc set).
212 */
213static int
214brix_dirlist_open_dir(brix_ctx_t *ctx, ngx_connection_t *c,
215 ngx_stream_brix_srv_conf_t *conf, brix_dirlist_walk_t *walk,
216 ngx_int_t *rc)
217{
218 brix_vfs_ctx_t vctx;
219 int err = 0;
220
221 brix_beneath_full_path(conf->common.root_canon, walk->reqpath,
222 walk->full_path, sizeof(walk->full_path));
223
224 if (brix_auth_gate(ctx, c, BRIX_OP_DIRLIST, "DIRLIST",
225 walk->reqpath, walk->full_path, conf,
226 BRIX_AUTH_LOOKUP, 0) != NGX_OK) {
227 *rc = ctx->write_rc;
228 return 0;
229 }
230
231 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
232 conf->common.root_canon, NULL, 0 /* allow_write */, 0 /* is_tls */,
233 ctx->identity, walk->full_path);
234 brix_vfs_ctx_bind_backend_cred(&vctx,
235 &conf->common.storage_credential_dir,
236 conf->common.storage_credential_fallback);
237 brix_root_vfs_bind_deleg(ctx, conf, &vctx);
238 walk->dh = brix_vfs_opendir(&vctx, &err);
239 if (walk->dh == NULL) {
240 if (err == ENOTDIR) {
241 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", walk->reqpath,
242 "-", kXR_NotFile, "path is not a directory", rc);
243 }
244 if (err == ENOENT) {
245 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", walk->reqpath,
246 "-", kXR_NotFound, "directory not found", rc);
247 }
248 BRIX_BAIL_ERR(ctx, c, BRIX_OP_DIRLIST, "DIRLIST", walk->reqpath,
249 "-", kXR_IOError, strerror(err), rc);
250 }
251
252 return 1;
253}

255/*
256 * brix_dirlist_alloc_chunk — allocate the header + 64KB chunk accumulator
257 * and seed the optional dstat lead-in sentinel.
258 *
259 * Guards against pool exhaustion from a flood of dirlist calls before
260 * charging the connection pool accounting; closes the directory handle on
261 * any failure so the caller can return immediately.
262 *
263 * Returns 1 with walk->chunk ready; 0 when a response was already sent or
264 * allocation failed (*rc set).
265 */
266static int
267brix_dirlist_alloc_chunk(brix_ctx_t *ctx, ngx_connection_t *c,
268 brix_dirlist_walk_t *walk, ngx_int_t *rc)
269{
270 if (ctx->login.pool_bytes_used + XRD_RESPONSE_HDR_LEN + walk->chunk_cap
271 > BRIX_MAX_CONN_POOL_BYTES)
272 {
273 brix_vfs_closedir(walk->dh, c->log);
274 ngx_log_error(NGX_LOG_WARN, c->log, 0,
275 "brix: dirlist pool limit reached (%uz bytes), "
276 "closing connection", ctx->login.pool_bytes_used);
277 *rc = brix_send_error(ctx, c, kXR_NoMemory,
278 "connection pool limit exceeded");
279 return 0;
280 }
281 walk->chunk = ngx_palloc(c->pool, XRD_RESPONSE_HDR_LEN + walk->chunk_cap);
282 if (walk->chunk == NULL) {
283 brix_vfs_closedir(walk->dh, c->log);
284 *rc = NGX_ERROR;
285 return 0;
286 }
287 ctx->login.pool_bytes_used += XRD_RESPONSE_HDR_LEN + walk->chunk_cap;
288
289 if (walk->want_stat) {
290 ngx_memcpy(walk->chunk + XRD_RESPONSE_HDR_LEN,
291 BRIX_DSTAT_LEADIN, BRIX_DSTAT_LEADIN_LEN);
292 walk->chunk_pos = BRIX_DSTAT_LEADIN_LEN;
293 }
294
295 return 1;
296}

298/*
299 * brix_dirlist_entry_skip — decide whether a directory entry must be
300 * excluded from the listing.
301 *
302 * Skips "." and "..", this gateway's internal control artifacts (e.g. the
303 * checkpoint recovery lock — a stock XRootD export shows no such files;
304 * mirror that), internal metadata/staging artifacts (cache sidecars, stage
305 * markers, in-flight upload temps — never enumerable to a client), and names
306 * with control bytes that would corrupt the wire format (logged at WARN).
307 *
308 * Returns 1 (skip this entry), 0 (include it).
309 */
310static ngx_flag_t
311brix_dirlist_entry_skip(ngx_log_t *log, const char *name)
312{
313 char safe_name[256];
314
315 if (name[0] == '.' && (name[1] == '\0' ||
316 (name[1] == '.' && name[2] == '\0')))
317 {
318 return 1;
319 }
320
321 if (ngx_strncmp(name, ".nginx-xrootd",
322 sizeof(".nginx-xrootd") - 1) == 0)
323 {
324 return 1;
325 }
326
327 if (brix_is_internal_name(name)) {
328 return 1;
329 }
330
331 if (brix_dirlist_name_is_unsafe(name)) {
332 brix_sanitize_log_string(name, safe_name, sizeof(safe_name));
333 ngx_log_error(NGX_LOG_WARN, log, 0,
334 "brix: dirlist skipping entry with control bytes \"%s\"",
335 safe_name);
336 return 1;
337 }
338
339 return 0;
340}

342/*
343 * brix_dirlist_entry_meta — render the optional stat line and checksum
344 * token for one entry, accumulating the wire bytes it needs into fmt->need.
345 *
346 * readdir already did the no-follow lstat (and skipped any entry whose stat
347 * failed), so vst is valid whenever want_stat is set. The dcksm variant
348 * uses the dcksm stat-body format; the checksum token falls back to
349 * "algo:none" when the entry path would overflow PATH_MAX.
350 *
351 * fmt->need must be pre-seeded with the bare name length + '\n'.
352 */
353static void
354brix_dirlist_entry_meta(ngx_log_t *log, brix_dirlist_walk_t *walk,
355 const char *name, const brix_vfs_stat_t *vst,
356 brix_dirlist_entry_fmt_t *fmt)
357{
358 struct stat entry_st;
359 char entry_path[PATH_MAX];
360
361 fmt->cksum_token[0] = '\0';
362 fmt->have_stat = 0;
363
364 if (walk->want_stat) {
365 ngx_memzero(&entry_st, sizeof(entry_st));
366 entry_st.st_mode = (mode_t) vst->mode;
367 entry_st.st_size = vst->size;
368 entry_st.st_mtime = vst->mtime;
369 entry_st.st_ctime = vst->ctime;
370 entry_st.st_ino = vst->ino;
371 entry_st.st_dev = vst->dev;
372 entry_st.st_uid = vst->uid;
373 entry_st.st_gid = vst->gid;
374 entry_st.st_blocks = vst->blocks;
375
376 fmt->have_stat = 1;
377 if (walk->want_cksum) {
378 brix_dirlist_make_dcksm_stat_body(&entry_st, fmt->statbuf,
379 sizeof(fmt->statbuf));
380 } else {
381 brix_make_stat_body(&entry_st, 0, 0, fmt->statbuf,
382 sizeof(fmt->statbuf));
383 }
384 fmt->need += strlen(fmt->statbuf) + 1;
385 }
386
387 if (walk->want_cksum && fmt->have_stat) {
388 int n;
389
390 n = snprintf(entry_path, sizeof(entry_path), "%s/%s",
391 walk->full_path, name);
392 if (n < 0 || (size_t) n >= sizeof(entry_path)) {
393 snprintf(fmt->cksum_token, sizeof(fmt->cksum_token),
394 "%s:none", walk->cksum_algo);
395 } else {
396 brix_dirlist_checksum_token(log,
397 brix_vfs_dir_fd(walk->dh), name,
398 entry_path, &entry_st,
399 walk->cksum_algo, fmt->cksum_token,
400 sizeof(fmt->cksum_token));
401 }
402 fmt->need += strlen(fmt->cksum_token) + sizeof(" [ ]") - 1;
403 }
404}

406/*
407 * brix_dirlist_flush_chunk — frame the accumulated chunk as a kXR_oksofar
408 * continuation and queue it, resetting the accumulator for the next chunk.
409 *
410 * Returns the brix_queue_response result (NGX_OK on success).
411 */
412static ngx_int_t
413brix_dirlist_flush_chunk(brix_ctx_t *ctx, ngx_connection_t *c,
414 brix_dirlist_walk_t *walk)
415{
416 ngx_int_t rc;
417
418 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_oksofar,
419 (uint32_t) walk->chunk_pos,
420 (ServerResponseHdr *) walk->chunk);
421
422 rc = brix_queue_response(ctx, c, walk->chunk,
423 XRD_RESPONSE_HDR_LEN + walk->chunk_pos);
424 if (rc != NGX_OK) {
425 return rc;
426 }
427
428 walk->chunk_pos = 0;
429 return NGX_OK;
430}

432/*
433 * brix_dirlist_append_entry — copy one formatted entry into the chunk
434 * accumulator: name '\n' [statline [" [ token ]"] '\n'].
435 *
436 * The caller has already guaranteed fmt->need bytes of headroom by flushing
437 * a full chunk beforehand, so the copies cannot overflow chunk_cap.
438 */
439static void
440brix_dirlist_append_entry(brix_dirlist_walk_t *walk, const char *name,
441 size_t nlen, const brix_dirlist_entry_fmt_t *fmt)
442{
443 u_char *data = walk->chunk + XRD_RESPONSE_HDR_LEN;
444
445 ngx_memcpy(data + walk->chunk_pos, name, nlen);
446 walk->chunk_pos += nlen;
447 data[walk->chunk_pos++] = '\n';
448
449 if (walk->want_stat && fmt->have_stat) {
450 size_t slen;
451
452 slen = strlen(fmt->statbuf);
453 ngx_memcpy(data + walk->chunk_pos, fmt->statbuf, slen);
454 walk->chunk_pos += slen;
455
456 if (walk->want_cksum) {
457 int n;
458
459 n = snprintf((char *) data + walk->chunk_pos,
460 walk->chunk_cap - walk->chunk_pos,
461 " [ %s ]", fmt->cksum_token);
462 if (n > 0) {
463 walk->chunk_pos += (size_t) n;
464 }
465 }
466
467 data[walk->chunk_pos++] = '\n';
468 }
469}

471/*
472 * brix_dirlist_send_final — frame the last accumulated chunk as kXR_ok to
473 * signal end-of-listing, log the access, and bump the success metric.
474 *
475 * The trailing '\n' of the final entry is replaced by a NUL terminator (an
476 * empty listing sends a zero-length body).
477 *
478 * Returns the brix_queue_response result.
479 */
480static ngx_int_t
481brix_dirlist_send_final(brix_ctx_t *ctx, ngx_connection_t *c,
482 brix_dirlist_walk_t *walk)
483{
484 u_char *data = walk->chunk + XRD_RESPONSE_HDR_LEN;
485 size_t final_len;
486
487 if (walk->chunk_pos == 0) {
488 final_len = 0;
489 } else {
490 data[walk->chunk_pos - 1] = '\0';
491 final_len = walk->chunk_pos;
492 }
493
494 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
495 (uint32_t) final_len,
496 (ServerResponseHdr *) walk->chunk);
497
498 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
499 "brix: kXR_dirlist final chunk %uz bytes",
500 walk->chunk_pos);
501
502 brix_log_access(ctx, c, "DIRLIST", walk->reqpath,
503 walk->want_cksum ? "dcksm"
504 : (walk->want_stat ? "stat" : "-"),
505 1, 0, NULL, 0);
506 BRIX_OP_OK(ctx, BRIX_OP_DIRLIST);
507
508 return brix_queue_response(ctx, c, walk->chunk,
509 XRD_RESPONSE_HDR_LEN + final_len);
510}

512/*
513 * brix_dirlist_stream_entries — the chunked streaming loop: enumerate the
514 * open directory, format each visible entry, flush full 64KB chunks as
515 * kXR_oksofar frames, and finish with the kXR_ok final frame.
516 *
517 * Owns walk->dh from here on: the handle is closed on every exit path
518 * (queue failure mid-stream or normal end-of-directory).
519 *
520 * Returns the final queue result (NGX_OK on success).
521 */
522static ngx_int_t
523brix_dirlist_stream_entries(brix_ctx_t *ctx, ngx_connection_t *c,
524 brix_dirlist_walk_t *walk)
525{
526 ngx_str_t entry_name;
527 brix_vfs_stat_t entry_vst;
528 ngx_int_t rc;
529
530 while (brix_vfs_readdir(walk->dh, &entry_name,
531 walk->want_stat ? &entry_vst : NULL) == NGX_OK) {
532 const char *name = (char *) entry_name.data;
533 size_t nlen = entry_name.len;
534 brix_dirlist_entry_fmt_t fmt;
535
536 if (brix_dirlist_entry_skip(c->log, name)) {
537 continue;
538 }
539
540 fmt.need = nlen + 1;
541 brix_dirlist_entry_meta(c->log, walk, name, &entry_vst, &fmt);
542
543 if (walk->chunk_pos + fmt.need > walk->chunk_cap) {
544 rc = brix_dirlist_flush_chunk(ctx, c, walk);
545 if (rc != NGX_OK) {
546 brix_vfs_closedir(walk->dh, c->log);
547 return rc;
548 }
549 }
550
551 brix_dirlist_append_entry(walk, name, nlen, &fmt);
552 }
553
554 brix_vfs_closedir(walk->dh, c->log);
555
556 return brix_dirlist_send_final(ctx, c, walk);
557}

559/*
560 * brix_handle_dirlist — handle kXR_dirlist: enumerate a directory and send
561 * the entries as a multi-frame kXR_oksofar ... kXR_ok response.
562 *
563 * Wire format (ClientDirlistRequest):
564 * options: bitfield controlling response content:
565 * kXR_dstat (0x01) — include per-entry stat info
566 * kXR_dcksm (0x02) — include per-entry checksum (implies kXR_dstat)
567 * payload: NUL-terminated path, optionally followed by CGI "?cks.type=algo"
568 *
569 * Response format: a flat text block with one entry per line, each terminated
570 * by '\n'. If kXR_dstat is set, each entry is followed by a stat body
571 * (stat-line format: "f|d|p flags mtime size"). Entries with control
572 * characters in their names are silently skipped to prevent wire corruption.
573 *
574 * A 65536-byte chunk buffer is accumulated and flushed as kXR_oksofar frames;
575 * the final flush uses kXR_ok to signal end-of-listing.
576 *
577 * Pipeline: parse request → cluster redirect check → authz + VFS opendir →
578 * chunk allocation → chunked entry streaming (helpers above, state in
579 * brix_dirlist_walk_t).
580 */
581/* §14 (phase-64): the fork/exec `xrdfs <origin> ls` namespace forward for a
582 * legacy cache_origin GSI cache is RETIRED with the cache_origin config model —
583 * a tier cache lists through the composed backend driver's opendir/readdir. */
584
585ngx_int_t
586brix_handle_dirlist(brix_ctx_t *ctx, ngx_connection_t *c,
587 ngx_stream_brix_srv_conf_t *conf)
588{
589 brix_dirlist_walk_t walk;
590 ngx_int_t rc;
591
592 ngx_memzero(&walk, sizeof(walk));
593 walk.chunk_cap = 65536;
594
595 if (!brix_dirlist_parse_request(ctx, c, &walk, &rc)) {
596 return rc;
597 }
598
599 if (!brix_dirlist_check_redirect(ctx, c, conf, &walk, &rc)) {
600 return rc;
601 }
602
603 if (!brix_dirlist_open_dir(ctx, c, conf, &walk, &rc)) {
604 return rc;
605 }
