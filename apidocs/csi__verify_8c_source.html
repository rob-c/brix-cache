1/*
2 * fs/backend/csi_verify.c — CSI block-granule verify / fold / flush on the
3 * unified metadata record (xmeta P3).
4 *
5 * WHAT: Read-verify fully-spanned blocks against the record's BLOCKCRC table;
6 * fold write CRCs into the handle-local table as bytes stream through;
7 * merge everything into the record once at flush. WHY: at-rest
8 * corruption detection with ONE metadata record per file and zero
9 * record I/O on the write hot path. HOW: the record's own buffer_size
10 * is the verify granule; a slot of 0 means "not computed" and never
11 * fails a read; flush recomputes bounded unaligned edges from disk and
12 * zeroes anything it cannot vouch for (fail-open on coverage,
13 * fail-closed on mismatch).
14 */
15

28/* ---- read verify ----------------------------------------------------------- */
29
30/* Verify the bytes in [off, off+len) against the block CRCs in *xm. Pure: no
31 * I/O, no allocation — the record is supplied by the caller (cached snapshot on
32 * the read hot path, or a one-off load on the write read-after-write path). */
33static int
34csi_verify_against(const brix_csi_t *c, const brix_xmeta_t *xm,
35 const unsigned char *buf, off_t off, size_t len)
36{
37 int64_t g;
38 uint64_t b, b0;
39
40 if (!xm->have_blockcrc || xm->blockcrc == NULL || xm->buffer_size <= 0) {
41 return BRIX_CSI_NOTAGS;
42 }
43
44 /* Verify every block the buffer FULLY covers (the last file block is
45 * short: it ends at file_size). Slot 0 = not computed -> skipped. */
46 g = xm->buffer_size;
47 b0 = ((uint64_t) off + (uint64_t) g - 1) / (uint64_t) g;
48 for (b = b0; b < xm->nblocks; b++) {
49 int64_t bstart = (int64_t) b * g;
50 int64_t bend = bstart + g;
51
52 if (bend > xm->file_size) {
53 bend = xm->file_size;
54 }
55 if (bend > off + (int64_t) len) {
56 break; /* not fully covered by this read */
57 }
58 /* A block written through THIS handle since open has a stale on-disk CRC
59 * (the record is only recomputed at flush/close). Skip it: the bytes we
60 * serve are the ones just written, so there is no disk-corruption risk to
61 * catch here, and verifying against the pre-write CRC would spuriously
62 * fail every read-after-write on an integrity-tracked file (e.g. kXR_write
63 * / kXR_clone / a ckpXeq write followed by a read on the same handle). */
64 if (c->dirty && bstart < c->dirty_hi && bend > c->dirty_lo) {
65 continue;
66 }
67 if (xm->blockcrc[b] == BRIX_XMETA_CRC_UNSET) {
68 continue;
69 }
70 if (brix_crc32c_value(buf + (bstart - off),
71 (size_t) (bend - bstart)) != xm->blockcrc[b])
72 {
73 return BRIX_CSI_MISMATCH;
74 }
75 }
76 return BRIX_CSI_OK;
77}

79int
80brix_csi_verify_read(brix_csi_t *c, const unsigned char *buf, off_t off,
81 size_t len)
82{
83 brix_xmeta_t xm;
84 int rc;
85
86 if (c == NULL || buf == NULL || off < 0) {
87 errno = EINVAL;
88 return BRIX_CSI_ERR;
89 }
90 if (len == 0 || c->trust_fs) {
91 return BRIX_CSI_OK; /* nothing to check / self-checksumming fs */
92 }
93
94 /* Read hot path: verify against the snapshot taken once at open. No
95 * getxattr / parse / malloc per read — the record cannot change under a
96 * read handle, so the at-open copy is authoritative for at-rest checking. */
97 if (c->record != NULL) {
98 return csi_verify_against(c, c->record, buf, off, len);
99 }
100
101 /* Fallback: a handle with no cached record (a write handle serving a
102 * read-after-write) loads the record for this call. Rare and not the
103 * throughput hot path. */
104 switch (brix_xmeta_path_load(c->path, &xm)) {
105 case BRIX_XMETA_OK:
106 break;
107 case BRIX_XMETA_FOREIGN:
108 return BRIX_CSI_NOTAGS;
109 default:
110 return BRIX_CSI_ERR;
111 }
112 rc = csi_verify_against(c, &xm, buf, off, len);
113 brix_xmeta_free(&xm);
114 return rc;
115}

117/* ---- write fold ------------------------------------------------------------- */

119/* Grow the handle-local table to hold block index b. 0 / -1 (cap/OOM). */
120static int
121csi_local_reserve(brix_csi_t *c, uint64_t b)
122{
123 uint64_t want;
124 uint32_t *grown;
125
126 if (b >= BRIX_CSI_LOCAL_MAX) {
127 c->overflow = 1;
128 return -1;
129 }
130 if (b < c->local_n) {
131 return 0;
132 }
133 want = c->local_n ? c->local_n : 64;
134 while (want <= b) {
135 want *= 2;
136 }
137 if (want > BRIX_CSI_LOCAL_MAX) {
138 want = BRIX_CSI_LOCAL_MAX;
139 }
140 grown = realloc(c->local, (size_t) want * sizeof(uint32_t));
141 if (grown == NULL) {
142 c->overflow = 1;
143 return -1;
144 }
145 memset(grown + c->local_n, 0,
146 (size_t) (want - c->local_n) * sizeof(uint32_t));
147 c->local = grown;
148 c->local_n = want;
149 return 0;
150}

152int
153brix_csi_write_update(brix_csi_t *c, const unsigned char *buf, off_t off,
154 size_t len)
155{
156 uint64_t b, b0;
157 int64_t g;
158
159 if (c == NULL || off < 0 || (buf == NULL && len != 0)) {
160 errno = EINVAL;
161 return BRIX_CSI_ERR;
162 }
163 if (len == 0) {
164 return BRIX_CSI_OK;
165 }
166
167 if (!c->dirty) {
168 c->dirty = 1;
169 c->dirty_lo = off;
170 c->dirty_hi = off + (int64_t) len;
171 } else {
172 if (off < c->dirty_lo) {
173 c->dirty_lo = off;
174 }
175 if (off + (int64_t) len > c->dirty_hi) {
176 c->dirty_hi = off + (int64_t) len;
177 }
178 }
179
180 /* Fold the CRCs of fully covered blocks (at the CONFIGURED granule; if
181 * the record turns out to use a different one, flush drops the fold and
182 * falls back to its bounded recompute). */
183 g = (int64_t) c->granule;
184 b0 = ((uint64_t) off + (uint64_t) g - 1) / (uint64_t) g;
185 for (b = b0; ; b++) {
186 int64_t bstart = (int64_t) b * g;
187
188 if (bstart + g > off + (int64_t) len) {
189 break;
190 }
191 if (csi_local_reserve(c, b) != 0) {
192 break; /* cap/OOM: flush covers what it can */
193 }
194 c->local[b] = brix_crc32c_value(buf + (bstart - off), (size_t) g);
195 }
196 return BRIX_CSI_OK;
197}

245int
246brix_csi_flush(brix_csi_t *c)
247{
248 brix_xmeta_t xm;
249 struct stat st;
250 int64_t g;
251 uint64_t b, b_lo, b_hi;
252 int fd, lockfd, budget = BRIX_CSI_FLUSH_READ;
253 int fold_ok, any_unset = 0, rc;
254 unsigned char *blockbuf = NULL;
255
256 if (c == NULL || !c->writable) {
257 errno = EINVAL;
258 return BRIX_CSI_ERR;
259 }
260 if (!c->dirty) {
261 return BRIX_CSI_OK;
262 }
263
264 /* Flush reads the file back, so it opens its own O_RDONLY fd — the
265 * handle's data fd may already be closed or write-only. */
266 fd = open(c->path, O_RDONLY | O_NOCTTY | O_CLOEXEC | O_NOFOLLOW);
267 if (fd < 0) {
268 return BRIX_CSI_ERR;
269 }
270 if (fstat(fd, &st) != 0 || !S_ISREG(st.st_mode)) {
271 close(fd);
272 return BRIX_CSI_ERR;
273 }
274
275 lockfd = brix_xmeta_path_lock(c->path);
276 if (lockfd < 0) {
277 close(fd);
278 return BRIX_CSI_ERR;
279 }
280
281 /* Load / create / re-geometry the record. */
282 rc = brix_xmeta_path_load(c->path, &xm);
283 if (rc == BRIX_XMETA_ERR
284 || (rc == BRIX_XMETA_FOREIGN
285 && brix_xmeta_init(&xm, st.st_size,
286 (int64_t) c->granule) != BRIX_XMETA_OK))
287 {
288 brix_xmeta_path_unlock(lockfd);
289 close(fd);
290 return BRIX_CSI_ERR;
291 }
292 if (rc == BRIX_XMETA_FOREIGN) {
293 /* Fresh record for a file this handle wrote: fully present. */
294 xm.origin_mtime = (uint64_t) st.st_mtime;
295 if (xm.nblocks > 0) {
296 memset(xm.bitmap, 0xFF, (size_t) ((xm.nblocks + 7) / 8));
297 }
298 } else if (xm.file_size != st.st_size || xm.buffer_size <= 0) {
299 if (csi_record_regeom(&xm, st.st_size,
300 xm.buffer_size > 0 ? xm.buffer_size
301 : (int64_t) c->granule) != 0)
302 {
303 brix_xmeta_free(&xm);
304 brix_xmeta_path_unlock(lockfd);
305 close(fd);
306 return BRIX_CSI_ERR;
307 }
308 xm.origin_mtime = (uint64_t) st.st_mtime;
309 }
310
311 if (xm.blockcrc == NULL) {
312 xm.blockcrc = calloc(xm.nblocks ? (size_t) xm.nblocks : 1,
313 sizeof(uint32_t));
314 if (xm.blockcrc == NULL) {
315 brix_xmeta_free(&xm);
316 brix_xmeta_path_unlock(lockfd);
317 close(fd);
318 errno = ENOMEM;
319 return BRIX_CSI_ERR;
320 }
321 }
322 xm.have_blockcrc = 1;
323
324 g = xm.buffer_size;
325 fold_ok = (g == (int64_t) c->granule && !c->overflow);
326
327 /* Every block the write extent touched gets a fresh truth: the folded
328 * CRC when we have one, a bounded recompute from disk, or UNSET. */
329 b_lo = (uint64_t) (c->dirty_lo / g);
330 b_hi = (uint64_t) ((c->dirty_hi + g - 1) / g);
331 if (b_hi > xm.nblocks) {
332 b_hi = xm.nblocks;
333 }
334 for (b = b_lo; b < b_hi; b++) {
335 if (fold_ok && b < c->local_n
336 && c->local[b] != BRIX_XMETA_CRC_UNSET)
337 {
338 xm.blockcrc[b] = c->local[b];
339 continue;
340 }
341 if (budget > 0) {
342 int64_t bstart = (int64_t) b * g;
343 int64_t bend = bstart + g;
344 size_t got = 0;
345 int short_read = 0;
346
347 if (bend > xm.file_size) {
348 bend = xm.file_size;
349 }
350 if (blockbuf == NULL) {
351 blockbuf = malloc((size_t) g);
352 }
353 if (blockbuf != NULL) {
354 while (got < (size_t) (bend - bstart)) {
355 ssize_t n = pread(fd, blockbuf + got,
356 (size_t) (bend - bstart) - got,
357 bstart + (off_t) got);
358
359 if (n < 0 && errno == EINTR) {
360 continue;
361 }
362 if (n <= 0) {
363 short_read = 1;
364 break;
365 }
366 got += (size_t) n;
367 }
368 budget--;
369 if (!short_read) {
370 xm.blockcrc[b] = brix_crc32c_value(blockbuf, got);
371 continue;
372 }
373 }
374 }
375 xm.blockcrc[b] = BRIX_XMETA_CRC_UNSET; /* cannot vouch for it */
376 }
377 free(blockbuf);
378
379 for (b = 0; b < xm.nblocks; b++) {
380 if (xm.blockcrc[b] == BRIX_XMETA_CRC_UNSET) {
381 any_unset = 1;
382 break;
383 }
384 }
385 if (any_unset) {
386 if (xm.no_cksum_time == 0) {
387 xm.no_cksum_time = (int64_t) time(NULL);
388 }
389 } else {
390 xm.no_cksum_time = 0;
391 }
392
393 rc = (brix_xmeta_path_save(c->path, &xm) == BRIX_XMETA_OK)
394 ? BRIX_CSI_OK : BRIX_CSI_ERR;
395 brix_xmeta_free(&xm);
396 brix_xmeta_path_unlock(lockfd);
397 close(fd);
398 if (rc == BRIX_CSI_OK) {
399 c->dirty = 0;
400 }
401 return rc;
402}
