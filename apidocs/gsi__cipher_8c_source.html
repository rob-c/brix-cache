26char *
27brix_gsi_cipher_public(EVP_PKEY *dh, size_t *outlen)
28{
29 BIGNUM *pub = NULL;
30 char *phex = NULL;
31 BIO *bio = NULL;
32 char *pem = NULL;
33 long pemlen;
34 char *out = NULL;
35 size_t total;
36
37 if (!EVP_PKEY_get_bn_param(dh, OSSL_PKEY_PARAM_PUB_KEY, &pub)) {
38 return NULL;
39 }
40 phex = BN_bn2hex(pub); /* UPPERCASE hex, as XrdCrypto */
41 BN_free(pub);
42 bio = BIO_new(BIO_s_mem());
43
44 if (phex != NULL && bio != NULL
45 && PEM_write_bio_Parameters(bio, dh) == 1) {
46 pemlen = BIO_get_mem_data(bio, &pem); /* PEM params incl END+\n */
47 total = (size_t) pemlen + 10 + strlen(phex) + 10;
48 out = (char *) malloc(total);
49 if (out != NULL) {
50 size_t cur = 0;
51 /* phase74-fp: out is a length-tracked binary blob (*outlen); both
52 * consumers (pwd credsreq, gsi kXGC_cert) bucket it by length and
53 * never treat it as a NUL-terminated string. */
54 memcpy(out + cur, pem, (size_t) pemlen); cur += (size_t) pemlen; // NOLINT(bugprone-not-null-terminated-result)
55 memcpy(out + cur, "---BPUB---", 10); cur += 10; // NOLINT(bugprone-not-null-terminated-result)
56 memcpy(out + cur, phex, strlen(phex)); cur += strlen(phex); // NOLINT(bugprone-not-null-terminated-result)
57 memcpy(out + cur, "---EPUB---", 10); cur += 10; // NOLINT(bugprone-not-null-terminated-result)
58 *outlen = cur;
59 }
60 }
61
62 if (phex != NULL) { OPENSSL_free(phex); }
63 BIO_free(bio);
64 return out;
65}

68EVP_PKEY *
69brix_gsi_cipher_parse_peer(const uint8_t *buf, size_t len)
70{
71 const char *pb = (const char *) memmem(buf, len, "---BPUB---", 10);
72 const char *pe = (const char *) memmem(buf, len, "---EPUB--", 9);
73 BIGNUM *bnpub = NULL;
74 EVP_PKEY *peer = NULL;
75 char *hex;
76 size_t hlen, pemlen;
77
78 if (pb == NULL || pe == NULL || pe <= pb + 10) {
79 return NULL;
80 }
81 pemlen = (size_t) ((const uint8_t *) pb - buf); /* PEM params length */
82 hlen = (size_t) (pe - (pb + 10));
83 hex = (char *) malloc(hlen + 1);
84 if (hex == NULL) {
85 return NULL;
86 }
87 memcpy(hex, pb + 10, hlen);
88 hex[hlen] = '\0';
89 BN_hex2bn(&bnpub, hex);
90 free(hex);
91 if (bnpub == NULL) {
92 return NULL;
93 }
94
95 {
96 BIO *bio = BIO_new_mem_buf(buf, (int) pemlen);
97 EVP_PKEY *params = bio ? PEM_read_bio_Parameters(bio, NULL) : NULL;
98 OSSL_PARAM *pp = NULL, *pk = NULL, *merged = NULL;
99 OSSL_PARAM_BLD *bld = OSSL_PARAM_BLD_new();
100 EVP_PKEY_CTX *ctx = NULL;
101
102 if (params != NULL && bld != NULL
103 && EVP_PKEY_todata(params, EVP_PKEY_KEY_PARAMETERS, &pp) == 1
104 && OSSL_PARAM_BLD_push_BN(bld, OSSL_PKEY_PARAM_PUB_KEY, bnpub) == 1
105 && (pk = OSSL_PARAM_BLD_to_param(bld)) != NULL
106 && (merged = OSSL_PARAM_merge(pp, pk)) != NULL) {
107 ctx = EVP_PKEY_CTX_new_id(EVP_PKEY_DH, NULL);
108 if (ctx != NULL && EVP_PKEY_fromdata_init(ctx) == 1) {
109 if (EVP_PKEY_fromdata(ctx, &peer, EVP_PKEY_PUBLIC_KEY,
110 merged) != 1) {
111 peer = NULL;
112 }
113 }
114 }
115 EVP_PKEY_CTX_free(ctx);
116 OSSL_PARAM_free(merged);
117 OSSL_PARAM_free(pk);
118 OSSL_PARAM_free(pp);
119 OSSL_PARAM_BLD_free(bld);
120 EVP_PKEY_free(params);
121 BIO_free(bio);
122 }
123 BN_free(bnpub);
124 return peer;
125}

157int
158brix_gsi_cipher_lookup(const char *name, brix_gsi_cipher_t *out)
159{
160 const EVP_CIPHER *evp;
161 int i, allowed = 0;
162
163 if (name == NULL || out == NULL) {
164 return 0;
165 }
166 for (i = 0; gsi_cipher_allow[i] != NULL; i++) {
167 if (strcmp(name, gsi_cipher_allow[i]) == 0) { allowed = 1; break; }
168 }
169 if (!allowed) {
170 return 0; /* not a negotiable GSI cipher */
171 }
172#if OPENSSL_VERSION_NUMBER >= 0x30000000L
173 gsi_load_legacy_once();
174#endif
175 evp = EVP_get_cipherbyname(name);
176 if (evp == NULL) {
177 return 0; /* provider not loaded (e.g. legacy bf/3des) */
178 }
179 out->evp = evp;
180 out->key_len = EVP_CIPHER_key_length(evp);
181 out->iv_len = EVP_CIPHER_iv_length(evp);
182 if (out->key_len <= 0 || out->key_len > BRIX_GSI_MAX_KEY
183 || out->iv_len < 0 || out->iv_len > BRIX_GSI_MAX_IV) {
184 return 0;
185 }
186 return 1;
187}

191brix_gsi_cipher_pick(const char *offered, brix_gsi_cipher_t *out,
192 char chosen[24])
193{
194 const char *p = offered;
195
196 if (offered == NULL || out == NULL) {
197 return 0;
198 }
199 while (*p) {
200 char name[24];
201 size_t n = 0;
202 const char *start = p;
203
204 while (*p && *p != ':') { p++; }
205 n = (size_t) (p - start);
206 if (n > 0 && n < sizeof(name)) {
207 memcpy(name, start, n);
208 name[n] = '\0';
209 if (brix_gsi_cipher_lookup(name, out)) {
210 if (chosen != NULL) {
211 memcpy(chosen, name, n + 1);
212 }
213 return 1;
214 }
215 }
216 if (*p == ':') { p++; }
217 }
218 return 0;

222int
223brix_gsi_cipher_session_key(EVP_PKEY *mine, EVP_PKEY *peer, int padded,
224 uint8_t *key, int key_len)
225{
226 EVP_PKEY_CTX *ctx = EVP_PKEY_CTX_new(mine, NULL);
227 uint8_t *secret;
228 size_t slen = 0;
229 int ok = 0;
230
231 /* `padded` MUST match the peer's XrdSecgsi HasPad: pre-DHsigned (<10400)
232 * peers use dh_pad=0 (leading zeros stripped), newer ones dh_pad=1. The
233 * session key is the first key_len bytes of the resulting secret. */
234 if (key_len <= 0 || key_len > BRIX_GSI_MAX_KEY
235 || ctx == NULL
236 || EVP_PKEY_derive_init(ctx) != 1
237 || EVP_PKEY_CTX_set_dh_pad(ctx, padded ? 1 : 0) != 1
238 || EVP_PKEY_derive_set_peer(ctx, peer) != 1
239 || EVP_PKEY_derive(ctx, NULL, &slen) != 1
240 || slen < (size_t) key_len) {
241 EVP_PKEY_CTX_free(ctx);
242 return 0;
243 }
244 secret = (uint8_t *) malloc(slen);
245 if (secret != NULL && EVP_PKEY_derive(ctx, secret, &slen) == 1) {
246 memcpy(key, secret, (size_t) key_len);
247 ok = 1;
248 }
249 if (secret != NULL) {
250 OPENSSL_cleanse(secret, slen);
251 free(secret);
252 }
253 EVP_PKEY_CTX_free(ctx);
254 return ok;
255}

258uint8_t *
259brix_gsi_cipher_encrypt(const brix_gsi_cipher_t *c, const uint8_t *key,
260 const uint8_t *in, size_t inlen, int use_iv,
261 size_t *outlen)
262{
263 XRD_AUTO(EVP_CIPHER_CTX) *ctx = EVP_CIPHER_CTX_new();
264 uint8_t iv[BRIX_GSI_MAX_IV];
265 size_t ivl = (size_t) c->iv_len;
266 size_t off = use_iv ? ivl : 0; /* IV prepended only when use_iv */
267 uint8_t *out;
268 int l1 = 0, l2 = 0;
269
270 /* ctx is scope-owned (XRD_AUTO): freed automatically on every return below.
271 * use_iv: a fresh random IV is prepended (XrdSecgsi >=DHsigned). Otherwise
272 * a zero IV is used and nothing is prepended (pre-DHsigned peers). */
273 if (ctx == NULL) {
274 return NULL;
275 }
276 if (use_iv) {
277 if (RAND_bytes(iv, c->iv_len) != 1) {
278 return NULL;
279 }
280 } else {
281 memset(iv, 0, ivl);
282 }
283 out = (uint8_t *) malloc(off + inlen + (size_t) EVP_CIPHER_block_size(c->evp));
284 if (out == NULL) {
285 return NULL;
286 }
287 if (use_iv) {
288 memcpy(out, iv, ivl);
289 }
290 if (EVP_EncryptInit_ex(ctx, c->evp, NULL, key, iv) == 1
291 && EVP_EncryptUpdate(ctx, out + off, &l1, in, (int) inlen) == 1
292 && EVP_EncryptFinal_ex(ctx, out + off + l1, &l2) == 1) {
293 *outlen = off + (size_t) (l1 + l2);
294 } else {
295 free(out); /* out is caller-owned on success — freed only here */
296 out = NULL;
297 }
298 return out;
299}

302uint8_t *
303brix_gsi_cipher_decrypt(const brix_gsi_cipher_t *c, const uint8_t *key,
304 const uint8_t *in, size_t inlen, int use_iv,
305 size_t *outlen)
306{
307 XRD_AUTO(EVP_CIPHER_CTX) *ctx = EVP_CIPHER_CTX_new();
308 uint8_t iv[BRIX_GSI_MAX_IV];
309 size_t ivl = (size_t) c->iv_len;
310 size_t off = use_iv ? ivl : 0;
311 uint8_t *out;
312 int l1 = 0, l2 = 0;
313
314 /* ctx is scope-owned (XRD_AUTO): freed automatically on every return below. */
315 if (ctx == NULL || inlen < off) {
316 return NULL;
317 }
318 if (use_iv) {
319 memcpy(iv, in, ivl); /* IV is the first iv_len bytes */
320 } else {
321 memset(iv, 0, ivl);
322 }
323 out = (uint8_t *) malloc(inlen - off + (size_t) EVP_CIPHER_block_size(c->evp));
324 if (out == NULL) {
325 return NULL;
326 }
327 if (EVP_DecryptInit_ex(ctx, c->evp, NULL, key, iv) == 1
328 && EVP_DecryptUpdate(ctx, out, &l1, in + off, (int) (inlen - off)) == 1
329 && EVP_DecryptFinal_ex(ctx, out + l1, &l2) == 1) {
330 *outlen = (size_t) l1 + (size_t) l2; /* widen BEFORE adding */
331 } else {
332 free(out); /* out is caller-owned on success — freed only here */
333 out = NULL;
334 }
335 return out;
