4/* ---- File: src/aio/uring.h — optional io_uring disk-I/O backend (Phase 44) ----
5 *
6 * WHAT: Declares the per-worker io_uring singleton, its completion-slot table,
7 * the op-kind enum, and the public API (runtime probe, startup fail-fast
8 * validation, per-worker lifecycle, the submit/reaper entry points, and
9 * the runtime kill-switch hooks). The backend is a THIRD AIO dispatch
10 * tier below the existing thread pool: the cascade becomes
11 * io_uring -> thread pool -> inline sync, selected per server block by
12 * `brix_io_uring on|off|auto`.
13 *
14 * WHY: The server module never touches a raw socket (nginx core owns egress),
15 * so server-side io_uring is strictly a disk-I/O backend interposed on the
16 * single dispatch seam `brix_aio_post_task()` (src/aio/resume.c). It is
17 * OFF BY DEFAULT, double-gated (pkg-config liburing at build +
18 * authoritative opcode probe at runtime), and degrades transparently — but
19 * `on` is a hard requirement that fails startup when it cannot be provided
20 * (§32 fail-fast). See docs/refactor/phase-44-io-uring-backend.md.
21 *
22 * HOW: Everything liburing-specific is guarded by BRIX_HAVE_LIBURING. When
23 * the macro is undefined the whole translation unit compiles to inert
24 * stubs: the worker accessor returns NULL, the probe returns 0, and the
25 * validator hard-fails any `brix_io_uring on` block at config time with
26 * no kernel needed (so `nginx -t` flags a stub build anywhere). Assumes
27 * ngx_brix_module.h (nginx + config + tunables) is already included. */
28

32/*
33 * Completion-slot table entry (UAF-safe task mapping).
34 *
35 * Never put a raw task pointer in a CQE's user_data: a late completion for a
36 * connection that has since been torn down would dereference freed memory.
37 * Instead user_data = ((uint64_t)generation << 32) | slot_index; the reaper
38 * validates the generation and drops a stale CQE. This guards the ring slot;
39 * the done-callback's own ctx->destroyed check remains the authoritative guard
40 * for the connection (both layers are kept).
41 */
42typedef struct {
43 void *task; /* ngx_thread_task_t carrying the *_aio_t */
44 ngx_event_handler_pt done_fn; /* brix_*_aio_done */
45 uint32_t generation; /* bumped on free -> detects a stale CQE */
46 uint8_t op_kind; /* brix_uring_op_e; selects OUT xlation */
47 uint8_t in_use; /* 1 = claimed & submitted, 0 = free */
48} brix_uring_slot_t;
49
50/*
51 * Per-worker io_uring singleton. One ring per worker process (same lifetime as
52 * every other per-worker async resource), reached only through the file-static
53 * accessor brix_uring_worker() — no new exported global. Only the event
54 * thread mutates the ring (submit on dispatch, reap on the eventfd handler).
55 */
56typedef struct {
57 struct io_uring ring; /* liburing SQ/CQ rings */
58 int eventfd; /* CQE notifier (register_eventfd) */
59 ngx_connection_t *evc; /* fake connection wrapping the eventfd */
60 uint32_t queue_depth; /* SQ entries; also slot-table length */
61 uint32_t inflight; /* SQEs submitted, CQEs not yet reaped */
62 brix_uring_slot_t *slots; /* completion-slot table, queue_depth */
63 ngx_atomic_t *disabled_flag; /* SHM hot kill-switch (read each submit)*/
64 unsigned ring_active:1; /* io_uring_queue_init succeeded */
65 unsigned enabled:1; /* ring up & probe passed this worker */
66 unsigned restrict_ops:1;/* io_uring_register_restrictions applied*/
67 ngx_log_t *log; /* worker cycle log (for the reaper) */
68} brix_uring_t;
