1/*
2 * postconfig.c - content handler registration and HTTP SSL/thread setup.
3 *
4 * WHAT: Post-configuration lifecycle hook executed after nginx main configuration parsing — registers WebDAV content handler in NGX_HTTP_CONTENT_PHASE, enables GSI proxy certificate support on SSL contexts where proxy_certs=on, initializes SSL fd-cache index for connection reuse optimization, and allocates thread pool references for async file I/O operations (PUT/GET).
5 *
6 * WHY: nginx requires all modules to register their handlers during postconfiguration phase before accepting any traffic. WebDAV content handler must be registered in NGX_HTTP_CONTENT_PHASE so nginx can dispatch HTTP requests to ngx_http_brix_webdav_handler() after authentication completes. GSI/x509 proxy certificate authentication requires X509_V_FLAG_ALLOW_PROXY_CERTS flag on SSL context — without this flag, proxy certificates issued by CA intermediates are rejected as invalid (AGENTS.md INVARIANT: GSI auth uses proxy cert chains). Thread pool allocation enables async file I/O for blocking operations (PUT body write, GET range read) per AGENTS.md FAQ: "Async I/O? Event-loop only. No wait/sleep/read. Use ngx_thread_pool_run."
7 *
8 * HOW: Called once during nginx startup after all configuration directives parsed. First initializes SSL fd-cache index via webdav_auth_init_ssl_indices() for connection reuse optimization (already documented in fd_cache.c). Second, retrieves NGX_HTTP_CORE_MAIN_CONF and pushes ngx_http_brix_webdav_handler into content phase handler array — this is the entry point for all WebDAV HTTP requests after authentication routing. Third, iterates CMCF servers to find SSL contexts with proxy_certs=on flag, then sets X509_V_FLAG_ALLOW_PROXY_CERTS via SSL_CTX_get0_param + X509_VERIFY_PARAM_set_flags enabling GSI proxy cert chains. Fourth reinitializes fd-cache SSL index for connection tracking (per fd_cache.c). Finally iterates servers again under NGX_THREADS macro to locate thread pool via ngx_thread_pool_get() — uses wdcf->common.thread_pool_name if configured, otherwise defaults to "default" pool name defined in nginx.conf directive `thread_pool default threads=4 max_queue=65536`. Logs NOTICE-level message when pool not found (async I/O disabled) or successfully allocated.
9 */

20{
21 ngx_http_handler_pt *h;
22 ngx_http_core_main_conf_t *cmcf;
23 ngx_http_core_srv_conf_t **cscfp;
24 ngx_http_ssl_srv_conf_t *sslcf;
25 ngx_http_brix_webdav_loc_conf_t *wdcf;
26 ngx_uint_t s;
27 X509_VERIFY_PARAM *param;
28
29 if (webdav_auth_init_ssl_indices(cf->log) != NGX_OK) {
30 return NGX_ERROR;
31 }
32
33 cmcf = ngx_http_conf_get_module_main_conf(cf, ngx_http_core_module);
34
35 /* Access phase: auth, CORS, write guards, and token-scope checks. */
36 h = ngx_array_push(&cmcf->phases[NGX_HTTP_ACCESS_PHASE].handlers);
37 if (h == NULL) {
38 return NGX_ERROR;
39 }
40 *h = ngx_http_brix_webdav_access_handler;
41
42 /* Phase 21 Step C: OIDC introspection runs as a second access-phase
43 * handler (after the main auth handler), so its subrequest suspend/resume
44 * re-entry replays only the introspection check — not the whole auth gate. */
45 h = ngx_array_push(&cmcf->phases[NGX_HTTP_ACCESS_PHASE].handlers);
46 if (h == NULL) {
47 return NGX_ERROR;
48 }
49 *h = webdav_introspect_access_handler;
50
51 /* Phase 25: advanced rate limiting runs as a third access-phase handler
52 * (after auth, so the identity is populated) and chains a body filter for
53 * bandwidth accounting. */
54 h = ngx_array_push(&cmcf->phases[NGX_HTTP_ACCESS_PHASE].handlers);
55 if (h == NULL) {
56 return NGX_ERROR;
57 }
58 *h = brix_rl_http_access_handler;
59
60 /* Bandwidth is charged in the log phase from the known response size. */
61 h = ngx_array_push(&cmcf->phases[NGX_HTTP_LOG_PHASE].handlers);
62 if (h == NULL) {
63 return NGX_ERROR;
64 }
65 *h = brix_rl_http_log_handler;
66
67 /* Phase 24: traffic mirror. The precontent handler does both jobs: on the
68 * main request it fires the background shadow subrequests, and on each
69 * mirror subrequest it takes the request over and proxies it to the shadow.
70 * Doing the takeover in the precontent phase avoids any dependence on
71 * content-phase handler ordering. */
72 h = ngx_array_push(&cmcf->phases[NGX_HTTP_PRECONTENT_PHASE].handlers);
73 if (h == NULL) {
74 return NGX_ERROR;
75 }
76 *h = brix_http_mirror_precontent_handler;
77
78 /* Content phase: native WebDAV method routing for all implemented methods. */
79 h = ngx_array_push(&cmcf->phases[NGX_HTTP_CONTENT_PHASE].handlers);
80 if (h == NULL) {
81 return NGX_ERROR;
82 }
83 *h = ngx_http_brix_webdav_handler;
84
85 /* Log phase: stamp the primary's final status for the divergence compare. */
86 h = ngx_array_push(&cmcf->phases[NGX_HTTP_LOG_PHASE].handlers);
87 if (h == NULL) {
88 return NGX_ERROR;
89 }
90 *h = brix_http_mirror_log_handler;
91
92 cscfp = cmcf->servers.elts;
93 for (s = 0; s < cmcf->servers.nelts; s++) {
94 ngx_http_conf_ctx_t *ctx = cscfp[s]->ctx;
95
96 wdcf = ctx->loc_conf[ngx_http_brix_webdav_module.ctx_index];
97 if (wdcf == NULL) {
98 continue;
99 }
100
101 sslcf = ctx->srv_conf[ngx_http_ssl_module.ctx_index];
102 if (sslcf == NULL || sslcf->ssl.ctx == NULL) {
103 continue;
104 }
105
106 if (wdcf->proxy_certs) {
107 param = SSL_CTX_get0_param(sslcf->ssl.ctx);
108 if (param) {
109 X509_VERIFY_PARAM_set_flags(param, X509_V_FLAG_ALLOW_PROXY_CERTS);
110 ngx_log_error(NGX_LOG_INFO, cf->log, 0,
111 "brix_webdav: enabled X509_V_FLAG_ALLOW_PROXY_CERTS"
112 " on SSL context for server %V",
113 &cscfp[s]->server_name);
114 }
115 }
116
117 /* kTLS: opt every https server's TLS context into kernel-TLS so HTTPS
118 * GET sendfiles over kTLS and PUT decrypts in-kernel (unless brix_ktls
119 * off on that server). The server-level webdav loc-conf carries the flag
120 * (default on) for EVERY server — WebDAV and S3 alike, since brix_webdav/
121 * brix_s3 typically sit in a location block while ssl + brix_ktls are
122 * server-level. Transparent no-op for non-offloadable ciphers. */
123 if (wdcf->common.ktls) {
124 brix_http_ktls_enable_ctx(sslcf->ssl.ctx, cf->log,
125 &cscfp[s]->server_name);
126 }
127 }
128
129 {
130 static ngx_str_t default_pool_name = ngx_string("default");
131
132 for (s = 0; s < cmcf->servers.nelts; s++) {
133 ngx_http_conf_ctx_t *ctx = cscfp[s]->ctx;
134 ngx_str_t *pool_name;
135
136 wdcf = ctx->loc_conf[ngx_http_brix_webdav_module.ctx_index];
137 if (wdcf == NULL || !wdcf->common.enable) {
138 continue;
139 }
140
141 pool_name = (wdcf->common.thread_pool_name.len > 0)
142 ? &wdcf->common.thread_pool_name
143 : &default_pool_name;
144
145 wdcf->common.thread_pool = ngx_thread_pool_get(cf->cycle, pool_name);
146 if (wdcf->common.thread_pool == NULL) {
147 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
148 "brix_webdav: thread pool \"%V\" not found - "
149 "async file I/O disabled (add a thread_pool directive)",
150 pool_name);
151 } else {
152 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
153 "brix_webdav: using thread pool \"%V\" for async file I/O",
154 pool_name);
155 }
156 }
157 }
158
159 if (brix_tpc_registry_configure(cf) != NGX_OK) {
160 return NGX_ERROR;
161 }
162
163 /* All module merges are done by the time any postconfiguration runs, so
164 * every protocol's enable flag is final here: reject a config that mixes
165 * brix protocols within one location or under one listen port. */
166 if (brix_http_proto_exclusive_check(cf) != NGX_OK) {
167 return NGX_ERROR;
168 }
169
170 return NGX_OK;
171}
