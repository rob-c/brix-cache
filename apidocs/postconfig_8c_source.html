1/*
2 * postconfig.c - content handler registration and HTTP SSL/thread setup.
3 *
4 * WHAT: Post-configuration lifecycle hook executed after nginx main configuration parsing — registers WebDAV content handler in NGX_HTTP_CONTENT_PHASE, enables GSI proxy certificate support on SSL contexts where proxy_certs=on, initializes SSL fd-cache index for connection reuse optimization, and allocates thread pool references for async file I/O operations (PUT/GET).
5 *
6 * WHY: nginx requires all modules to register their handlers during postconfiguration phase before accepting any traffic. WebDAV content handler must be registered in NGX_HTTP_CONTENT_PHASE so nginx can dispatch HTTP requests to ngx_http_brix_webdav_handler() after authentication completes. GSI/x509 proxy certificate authentication requires X509_V_FLAG_ALLOW_PROXY_CERTS flag on SSL context — without this flag, proxy certificates issued by CA intermediates are rejected as invalid (AGENTS.md INVARIANT: GSI auth uses proxy cert chains). Thread pool allocation enables async file I/O for blocking operations (PUT body write, GET range read) per AGENTS.md FAQ: "Async I/O? Event-loop only. No wait/sleep/read. Use ngx_thread_pool_run."
7 *
8 * HOW: Called once during nginx startup after all configuration directives parsed. First initializes SSL fd-cache index via webdav_auth_init_ssl_indices() for connection reuse optimization (already documented in fd_cache.c). Second, retrieves NGX_HTTP_CORE_MAIN_CONF and pushes ngx_http_brix_webdav_handler into content phase handler array — this is the entry point for all WebDAV HTTP requests after authentication routing. Third, iterates CMCF servers to find SSL contexts with proxy_certs=on flag, then sets X509_V_FLAG_ALLOW_PROXY_CERTS via SSL_CTX_get0_param + X509_VERIFY_PARAM_set_flags enabling GSI proxy cert chains. Fourth reinitializes fd-cache SSL index for connection tracking (per fd_cache.c). Finally iterates servers again under NGX_THREADS macro to locate thread pool via ngx_thread_pool_get() — uses wdcf->common.thread_pool_name if configured, otherwise defaults to "default" pool name defined in nginx.conf directive `thread_pool default threads=4 max_queue=65536`. Logs NOTICE-level message when pool not found (async I/O disabled) or successfully allocated.
9 */

20/*
21 * webdav_postconf_push_handler - append one phase handler to a phase array.
22 *
23 * WHAT: Push a single WebDAV request handler onto the handler array of a given
24 * nginx HTTP phase, returning NGX_OK on success or NGX_ERROR if the array
25 * allocation fails.
26 *
27 * WHY: The postconfiguration hook installs seven handlers across five phases;
28 * each install is the identical push-then-null-check dance. Folding it into one
29 * helper removes six repeated branches from the top-level function and keeps the
30 * registration order visible as a flat, ordered list of calls at the call site.
31 *
32 * HOW: Calls ngx_array_push() on the target phase's handler array; on a NULL
33 * return (out-of-memory) returns NGX_ERROR without touching the slot, otherwise
34 * assigns the handler pointer and returns NGX_OK.
35 */
36static ngx_int_t
37webdav_postconf_push_handler(ngx_http_core_main_conf_t *cmcf,
38 ngx_uint_t phase, ngx_http_handler_pt handler)
39{
40 ngx_http_handler_pt *h;
41
42 h = ngx_array_push(&cmcf->phases[phase].handlers);
43 if (h == NULL) {
44 return NGX_ERROR;
45 }
46
47 *h = handler;
48 return NGX_OK;
49}

51/*
52 * webdav_postconf_install_handlers - register all WebDAV phase handlers.
53 *
54 * WHAT: Install, in a fixed order, the access-phase auth/introspect/rate-limit
55 * handlers, the log-phase bandwidth and mirror handlers, the precontent-phase
56 * traffic-mirror handler, and the content-phase WebDAV method router; returns
57 * NGX_ERROR if any array push fails.
58 *
59 * WHY: nginx dispatches requests through phase handlers in registration order,
60 * so the exact sequence here is behavior-load-bearing (auth before rate-limit,
61 * mirror takeover in precontent, method routing in content). Grouping the pushes
62 * in one helper preserves that order as an auditable list while keeping the
63 * top-level hook short.
64 *
65 * HOW: Calls webdav_postconf_push_handler() once per (phase, handler) pair in
66 * the original order; the first failure short-circuits with NGX_ERROR via
67 * early-return, otherwise returns NGX_OK.
68 */
69static ngx_int_t
70webdav_postconf_install_handlers(ngx_http_core_main_conf_t *cmcf)
71{
72 /* Access phase: auth, CORS, write guards, and token-scope checks. */
73 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_ACCESS_PHASE,
74 ngx_http_brix_webdav_access_handler)
75 != NGX_OK)
76 {
77 return NGX_ERROR;
78 }
79
80 /* Phase 21 Step C: OIDC introspection runs as a second access-phase
81 * handler (after the main auth handler), so its subrequest suspend/resume
82 * re-entry replays only the introspection check — not the whole auth gate. */
83 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_ACCESS_PHASE,
84 webdav_introspect_access_handler)
85 != NGX_OK)
86 {
87 return NGX_ERROR;
88 }
89
90 /* Phase 25: advanced rate limiting runs as a third access-phase handler
91 * (after auth, so the identity is populated) and chains a body filter for
92 * bandwidth accounting. */
93 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_ACCESS_PHASE,
94 brix_rl_http_access_handler)
95 != NGX_OK)
96 {
97 return NGX_ERROR;
98 }
99
100 /* Bandwidth is charged in the log phase from the known response size. */
101 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_LOG_PHASE,
102 brix_rl_http_log_handler)
103 != NGX_OK)
104 {
105 return NGX_ERROR;
106 }
107
108 /* Phase 24: traffic mirror. The precontent handler does both jobs: on the
109 * main request it fires the background shadow subrequests, and on each
110 * mirror subrequest it takes the request over and proxies it to the shadow.
111 * Doing the takeover in the precontent phase avoids any dependence on
112 * content-phase handler ordering. */
113 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_PRECONTENT_PHASE,
114 brix_http_mirror_precontent_handler)
115 != NGX_OK)
116 {
117 return NGX_ERROR;
118 }
119
120 /* Content phase: native WebDAV method routing for all implemented methods. */
121 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_CONTENT_PHASE,
122 ngx_http_brix_webdav_handler)
123 != NGX_OK)
124 {
125 return NGX_ERROR;
126 }
127
128 /* Log phase: stamp the primary's final status for the divergence compare. */
129 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_LOG_PHASE,
130 brix_http_mirror_log_handler)
131 != NGX_OK)
132 {
133 return NGX_ERROR;
134 }
135
136 return NGX_OK;
137}

139/*
140 * webdav_postconf_load_client_capath - add a hashed CA dir to one server's
141 * TLS client-verify trust store.
142 *
143 * WHAT: Load the brix_ssl_client_capath directory into the SSL_CTX certificate
144 * store of one server block as an OpenSSL hashed-directory lookup; returns
145 * NGX_OK, or NGX_ERROR (a fatal config error) when the path is not an
146 * accessible directory or the store cannot be extended.
147 *
148 * WHY: Grid hosts ship their trust anchors as an OpenSSL hashed directory
149 * (/etc/grid-security/certificates, the IGTF layout) and no bundle file, but
150 * stock nginx's ssl_client_certificate/ssl_trusted_certificate are file-only —
151 * so a fail-closed `ssl_verify_client on` front leg could not consume the dir
152 * at all. A hashed lookup also resolves issuers at verify time, so CA-package
153 * updates are picked up without a reload. Misconfiguration is fatal (not a
154 * warning) because this directive IS the trust perimeter: a typo that silently
155 * loaded nothing would reject every client with an unhelpful handshake error.
156 *
157 * HOW: 1) ngx_file_info() the path (conf tokens are NUL-terminated) and demand
158 * an existing directory; 2) fetch the server's verify store with
159 * SSL_CTX_get_cert_store(); 3) X509_STORE_load_locations(store, NULL, dir)
160 * appends the hash-dir lookup (additive — certs from ssl_client_certificate
161 * stay trusted); 4) log one INFO line naming the server and directory.
162 */
163static ngx_int_t
164webdav_postconf_load_client_capath(ngx_conf_t *cf,
165 ngx_http_core_srv_conf_t *cscf,
166 ngx_http_ssl_srv_conf_t *sslcf,
167 ngx_str_t *capath)
168{
169 ngx_file_info_t fi;
170 X509_STORE *store;
171
172 if (ngx_file_info(capath->data, &fi) != 0) {
173 ngx_log_error(NGX_LOG_EMERG, cf->log, ngx_errno,
174 "brix_ssl_client_capath \"%V\" is not accessible",
175 capath);
176 return NGX_ERROR;
177 }
178
179 if (!ngx_is_dir(&fi)) {
180 ngx_log_error(NGX_LOG_EMERG, cf->log, 0,
181 "brix_ssl_client_capath \"%V\" is not a directory "
182 "(for a bundle file use ssl_client_certificate)",
183 capath);
184 return NGX_ERROR;
185 }
186
187 store = SSL_CTX_get_cert_store(sslcf->ssl.ctx);
188 if (store == NULL
189 || X509_STORE_load_locations(store, NULL,
190 (const char *) capath->data) != 1)
191 {
192 ngx_log_error(NGX_LOG_EMERG, cf->log, 0,
193 "brix_ssl_client_capath \"%V\": cannot add hashed "
194 "CA-directory lookup to the TLS trust store", capath);
195 return NGX_ERROR;
196 }
197
198 ngx_log_error(NGX_LOG_INFO, cf->log, 0,
199 "brix_webdav: added hashed CA dir \"%V\" to the client-"
200 "verify trust store for server %V",
201 capath, &cscf->server_name);
202 return NGX_OK;
203}

205/*
206 * webdav_postconf_setup_ssl_ctx - apply GSI-proxy and kTLS flags to one server.
207 *
208 * WHAT: For a single server block, enable X509_V_FLAG_ALLOW_PROXY_CERTS on its
209 * SSL verify params when proxy_certs is on, add the brix_ssl_client_capath
210 * hashed CA directory to its verify store when set (returning NGX_ERROR when
211 * that directory is unusable — a fatal config error), and opt its TLS context
212 * into kernel-TLS when brix_ktls is on; returns NGX_OK for servers without
213 * WebDAV config or without an SSL context (no-op).
214 *
215 * WHY: GSI proxy-cert chains are rejected by OpenSSL unless the proxy-certs flag
216 * is set (AGENTS.md INVARIANT), and kTLS offload must be armed per-server while
217 * the ssl + brix_ktls directives live at server level. Isolating the per-server
218 * body lets the caller iterate cleanly and keeps this security-load-bearing flag
219 * manipulation in one auditable place.
220 *
221 * HOW: Reads the WebDAV loc-conf and SSL srv-conf from the server's context and
222 * early-returns if either is absent. When proxy_certs is set it fetches the
223 * verify params via SSL_CTX_get0_param() and sets the proxy-cert flag (logging
224 * INFO); when ssl_client_capath is set it delegates to
225 * webdav_postconf_load_client_capath() and propagates its failure; when
226 * common.ktls is set it calls brix_http_ktls_enable_ctx(). Registration
227 * effects are identical whether or not this is split out.
228 */
229static ngx_int_t
230webdav_postconf_setup_ssl_ctx(ngx_conf_t *cf, ngx_http_core_srv_conf_t *cscf)
231{
232 ngx_http_conf_ctx_t *ctx = cscf->ctx;
233 ngx_http_brix_webdav_loc_conf_t *wdcf;
234 ngx_http_ssl_srv_conf_t *sslcf;
235 X509_VERIFY_PARAM *param;
236
237 wdcf = ctx->loc_conf[ngx_http_brix_webdav_module.ctx_index];
238 if (wdcf == NULL) {
239 return NGX_OK;
240 }
241
242 sslcf = ctx->srv_conf[ngx_http_ssl_module.ctx_index];
243 if (sslcf == NULL || sslcf->ssl.ctx == NULL) {
244 return NGX_OK;
245 }
246
247 if (wdcf->proxy_certs) {
248 param = SSL_CTX_get0_param(sslcf->ssl.ctx);
249 if (param) {
250 X509_VERIFY_PARAM_set_flags(param, X509_V_FLAG_ALLOW_PROXY_CERTS);
251 ngx_log_error(NGX_LOG_INFO, cf->log, 0,
252 "brix_webdav: enabled X509_V_FLAG_ALLOW_PROXY_CERTS"
253 " on SSL context for server %V",
254 &cscf->server_name);
255 }
256 }
257
258 if (wdcf->ssl_client_capath.len > 0
259 && webdav_postconf_load_client_capath(cf, cscf, sslcf,
260 &wdcf->ssl_client_capath)
261 != NGX_OK)
262 {
263 return NGX_ERROR;
264 }
265
266 /* kTLS: opt every https server's TLS context into kernel-TLS so HTTPS
267 * GET sendfiles over kTLS and PUT decrypts in-kernel (unless brix_ktls
268 * off on that server). The server-level webdav loc-conf carries the flag
269 * (default on) for EVERY server — WebDAV and S3 alike, since brix_webdav/
270 * brix_s3 typically sit in a location block while ssl + brix_ktls are
271 * server-level. Transparent no-op for non-offloadable ciphers. */
272 if (wdcf->common.ktls) {
273 brix_http_ktls_enable_ctx(sslcf->ssl.ctx, cf->log,
274 &cscf->server_name);
275 }
276
277 return NGX_OK;
278}

280/*
281 * webdav_postconf_setup_thread_pool - resolve the async-I/O pool for one server.
282 *
283 * WHAT: For a WebDAV-enabled server block, look up the configured thread pool
284 * (falling back to the "default" pool) and store it on the common conf, logging
285 * a NOTICE for both the found and the not-found cases; a no-op for servers that
286 * lack a WebDAV conf or have WebDAV disabled.
287 *
288 * WHY: Async file I/O for PUT/GET runs on an nginx thread pool resolved once at
289 * config time. The pool name is per-server (explicit or "default"), so the
290 * lookup must run per server block; keeping it in a helper mirrors the SSL-setup
291 * split and keeps the loop body in the caller trivial.
292 *
293 * HOW: Reads the WebDAV loc-conf from the server context, early-returns if it is
294 * absent or common.enable is unset, selects the configured pool name or the
295 * shared "default", and calls ngx_thread_pool_get(); the returned pointer (or
296 * NULL) is written to common.thread_pool with the matching NOTICE log.
297 */
298static void
299webdav_postconf_setup_thread_pool(ngx_conf_t *cf,
300 ngx_http_core_srv_conf_t *cscf)
301{
302 static ngx_str_t default_pool_name = ngx_string("default");
303
304 ngx_http_conf_ctx_t *ctx = cscf->ctx;
305 ngx_http_brix_webdav_loc_conf_t *wdcf;
306 ngx_str_t *pool_name;
307
308 wdcf = ctx->loc_conf[ngx_http_brix_webdav_module.ctx_index];
309 if (wdcf == NULL || !wdcf->common.enable) {
310 return;
311 }
312
313 pool_name = (wdcf->common.thread_pool_name.len > 0)
314 ? &wdcf->common.thread_pool_name
315 : &default_pool_name;
316
317 wdcf->common.thread_pool = ngx_thread_pool_get(cf->cycle, pool_name);
318 if (wdcf->common.thread_pool == NULL) {
319 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
320 "brix_webdav: thread pool \"%V\" not found - "
321 "async file I/O disabled (add a thread_pool directive)",
322 pool_name);
323 } else {
324 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
325 "brix_webdav: using thread pool \"%V\" for async file I/O",
326 pool_name);
327 }
328}

330/*
331 * webdav_postconf_setup_servers - run per-server SSL and thread-pool setup.
332 *
333 * WHAT: Iterate every server block twice — first applying GSI-proxy/kTLS flags
334 * to each SSL context, then resolving each WebDAV server's async-I/O thread pool.
335 *
336 * WHY: The original hook did both passes over cmcf->servers inline; preserving
337 * two separate passes (rather than fusing them) keeps behavior identical — the
338 * SSL flags are all set before any thread-pool lookup, matching the prior order
339 * of side effects and log output.
340 *
341 * HOW: Fetches the servers array once, loops applying webdav_postconf_setup_ssl_ctx()
342 * to each element (propagating NGX_ERROR — an unusable brix_ssl_client_capath
343 * is a fatal config error), then loops again applying
344 * webdav_postconf_setup_thread_pool(), which cannot fail.
345 */
346static ngx_int_t
347webdav_postconf_setup_servers(ngx_conf_t *cf, ngx_http_core_main_conf_t *cmcf)
348{
349 ngx_http_core_srv_conf_t **cscfp = cmcf->servers.elts;
350 ngx_uint_t s;
351
352 for (s = 0; s < cmcf->servers.nelts; s++) {
353 if (webdav_postconf_setup_ssl_ctx(cf, cscfp[s]) != NGX_OK) {
354 return NGX_ERROR;
355 }
356 }
357
358 for (s = 0; s < cmcf->servers.nelts; s++) {
359 webdav_postconf_setup_thread_pool(cf, cscfp[s]);
360 }
361
362 return NGX_OK;
363}
