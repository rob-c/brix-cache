1/*
2 * postconfig.c - content handler registration and HTTP SSL/thread setup.
3 *
4 * WHAT: Post-configuration lifecycle hook executed after nginx main configuration parsing — registers WebDAV content handler in NGX_HTTP_CONTENT_PHASE, enables GSI proxy certificate support on SSL contexts where proxy_certs=on, initializes SSL fd-cache index for connection reuse optimization, and allocates thread pool references for async file I/O operations (PUT/GET).
5 *
6 * WHY: nginx requires all modules to register their handlers during postconfiguration phase before accepting any traffic. WebDAV content handler must be registered in NGX_HTTP_CONTENT_PHASE so nginx can dispatch HTTP requests to ngx_http_brix_webdav_handler() after authentication completes. GSI/x509 proxy certificate authentication requires X509_V_FLAG_ALLOW_PROXY_CERTS flag on SSL context — without this flag, proxy certificates issued by CA intermediates are rejected as invalid (AGENTS.md INVARIANT: GSI auth uses proxy cert chains). Thread pool allocation enables async file I/O for blocking operations (PUT body write, GET range read) per AGENTS.md FAQ: "Async I/O? Event-loop only. No wait/sleep/read. Use ngx_thread_pool_run."
7 *
8 * HOW: Called once during nginx startup after all configuration directives parsed. First initializes SSL fd-cache index via webdav_auth_init_ssl_indices() for connection reuse optimization (already documented in fd_cache.c). Second, retrieves NGX_HTTP_CORE_MAIN_CONF and pushes ngx_http_brix_webdav_handler into content phase handler array — this is the entry point for all WebDAV HTTP requests after authentication routing. Third, iterates CMCF servers to find SSL contexts with proxy_certs=on flag, then sets X509_V_FLAG_ALLOW_PROXY_CERTS via SSL_CTX_get0_param + X509_VERIFY_PARAM_set_flags enabling GSI proxy cert chains. Fourth reinitializes fd-cache SSL index for connection tracking (per fd_cache.c). Finally iterates servers again under NGX_THREADS macro to locate thread pool via ngx_thread_pool_get() — uses wdcf->common.thread_pool_name if configured, otherwise defaults to "default" pool name defined in nginx.conf directive `thread_pool default threads=4 max_queue=65536`. Logs NOTICE-level message when pool not found (async I/O disabled) or successfully allocated.
9 */

19/*
20 * webdav_postconf_push_handler - append one phase handler to a phase array.
21 *
22 * WHAT: Push a single WebDAV request handler onto the handler array of a given
23 * nginx HTTP phase, returning NGX_OK on success or NGX_ERROR if the array
24 * allocation fails.
25 *
26 * WHY: The postconfiguration hook installs seven handlers across five phases;
27 * each install is the identical push-then-null-check dance. Folding it into one
28 * helper removes six repeated branches from the top-level function and keeps the
29 * registration order visible as a flat, ordered list of calls at the call site.
30 *
31 * HOW: Calls ngx_array_push() on the target phase's handler array; on a NULL
32 * return (out-of-memory) returns NGX_ERROR without touching the slot, otherwise
33 * assigns the handler pointer and returns NGX_OK.
34 */
35static ngx_int_t
36webdav_postconf_push_handler(ngx_http_core_main_conf_t *cmcf,
37 ngx_uint_t phase, ngx_http_handler_pt handler)
38{
39 ngx_http_handler_pt *h;
40
41 h = ngx_array_push(&cmcf->phases[phase].handlers);
42 if (h == NULL) {
43 return NGX_ERROR;
44 }
45
46 *h = handler;
47 return NGX_OK;
48}

50/*
51 * webdav_postconf_install_handlers - register all WebDAV phase handlers.
52 *
53 * WHAT: Install, in a fixed order, the access-phase auth/introspect/rate-limit
54 * handlers, the log-phase bandwidth and mirror handlers, the precontent-phase
55 * traffic-mirror handler, and the content-phase WebDAV method router; returns
56 * NGX_ERROR if any array push fails.
57 *
58 * WHY: nginx dispatches requests through phase handlers in registration order,
59 * so the exact sequence here is behavior-load-bearing (auth before rate-limit,
60 * mirror takeover in precontent, method routing in content). Grouping the pushes
61 * in one helper preserves that order as an auditable list while keeping the
62 * top-level hook short.
63 *
64 * HOW: Calls webdav_postconf_push_handler() once per (phase, handler) pair in
65 * the original order; the first failure short-circuits with NGX_ERROR via
66 * early-return, otherwise returns NGX_OK.
67 */
68static ngx_int_t
69webdav_postconf_install_handlers(ngx_http_core_main_conf_t *cmcf)
70{
71 /* Access phase: auth, CORS, write guards, and token-scope checks. */
72 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_ACCESS_PHASE,
73 ngx_http_brix_webdav_access_handler)
74 != NGX_OK)
75 {
76 return NGX_ERROR;
77 }
78
79 /* Phase 21 Step C: OIDC introspection runs as a second access-phase
80 * handler (after the main auth handler), so its subrequest suspend/resume
81 * re-entry replays only the introspection check — not the whole auth gate. */
82 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_ACCESS_PHASE,
83 webdav_introspect_access_handler)
84 != NGX_OK)
85 {
86 return NGX_ERROR;
87 }
88
89 /* Phase 25: advanced rate limiting runs as a third access-phase handler
90 * (after auth, so the identity is populated) and chains a body filter for
91 * bandwidth accounting. */
92 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_ACCESS_PHASE,
93 brix_rl_http_access_handler)
94 != NGX_OK)
95 {
96 return NGX_ERROR;
97 }
98
99 /* Bandwidth is charged in the log phase from the known response size. */
100 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_LOG_PHASE,
101 brix_rl_http_log_handler)
102 != NGX_OK)
103 {
104 return NGX_ERROR;
105 }
106
107 /* Phase 24: traffic mirror. The precontent handler does both jobs: on the
108 * main request it fires the background shadow subrequests, and on each
109 * mirror subrequest it takes the request over and proxies it to the shadow.
110 * Doing the takeover in the precontent phase avoids any dependence on
111 * content-phase handler ordering. */
112 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_PRECONTENT_PHASE,
113 brix_http_mirror_precontent_handler)
114 != NGX_OK)
115 {
116 return NGX_ERROR;
117 }
118
119 /* Content phase: native WebDAV method routing for all implemented methods. */
120 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_CONTENT_PHASE,
121 ngx_http_brix_webdav_handler)
122 != NGX_OK)
123 {
124 return NGX_ERROR;
125 }
126
127 /* Log phase: stamp the primary's final status for the divergence compare. */
128 if (webdav_postconf_push_handler(cmcf, NGX_HTTP_LOG_PHASE,
129 brix_http_mirror_log_handler)
130 != NGX_OK)
131 {
132 return NGX_ERROR;
133 }
134
135 return NGX_OK;
136}

138/*
139 * webdav_postconf_load_client_capath - add a hashed CA dir to one server's
140 * TLS client-verify trust store.
141 *
142 * WHAT: Load the brix_ssl_client_capath directory into the SSL_CTX certificate
143 * store of one server block as an OpenSSL hashed-directory lookup; returns
144 * NGX_OK, or NGX_ERROR (a fatal config error) when the path is not an
145 * accessible directory or the store cannot be extended.
146 *
147 * WHY: Grid hosts ship their trust anchors as an OpenSSL hashed directory
148 * (/etc/grid-security/certificates, the IGTF layout) and no bundle file, but
149 * stock nginx's ssl_client_certificate/ssl_trusted_certificate are file-only —
150 * so a fail-closed `ssl_verify_client on` front leg could not consume the dir
151 * at all. A hashed lookup also resolves issuers at verify time, so CA-package
152 * updates are picked up without a reload. Misconfiguration is fatal (not a
153 * warning) because this directive IS the trust perimeter: a typo that silently
154 * loaded nothing would reject every client with an unhelpful handshake error.
155 *
156 * HOW: 1) ngx_file_info() the path (conf tokens are NUL-terminated) and demand
157 * an existing directory; 2) fetch the server's verify store with
158 * SSL_CTX_get_cert_store(); 3) X509_STORE_load_locations(store, NULL, dir)
159 * appends the hash-dir lookup (additive — certs from ssl_client_certificate
160 * stay trusted); 4) log one INFO line naming the server and directory.
161 */
162static ngx_int_t
163webdav_postconf_load_client_capath(ngx_conf_t *cf,
164 ngx_http_core_srv_conf_t *cscf,
165 ngx_http_ssl_srv_conf_t *sslcf,
166 ngx_str_t *capath)
167{
168 ngx_file_info_t fi;
169 X509_STORE *store;
170
171 if (ngx_file_info(capath->data, &fi) != 0) {
172 ngx_log_error(NGX_LOG_EMERG, cf->log, ngx_errno,
173 "brix_ssl_client_capath \"%V\" is not accessible",
174 capath);
175 return NGX_ERROR;
176 }
177
178 if (!ngx_is_dir(&fi)) {
179 ngx_log_error(NGX_LOG_EMERG, cf->log, 0,
180 "brix_ssl_client_capath \"%V\" is not a directory "
181 "(for a bundle file use ssl_client_certificate)",
182 capath);
183 return NGX_ERROR;
184 }
185
186 store = SSL_CTX_get_cert_store(sslcf->ssl.ctx);
187 if (store == NULL
188 || X509_STORE_load_locations(store, NULL,
189 (const char *) capath->data) != 1)
190 {
191 ngx_log_error(NGX_LOG_EMERG, cf->log, 0,
192 "brix_ssl_client_capath \"%V\": cannot add hashed "
193 "CA-directory lookup to the TLS trust store", capath);
194 return NGX_ERROR;
195 }
196
197 ngx_log_error(NGX_LOG_INFO, cf->log, 0,
198 "brix_webdav: added hashed CA dir \"%V\" to the client-"
199 "verify trust store for server %V",
200 capath, &cscf->server_name);
201 return NGX_OK;
202}

210/*
211 * webdav_postconf_proxy_capath_apply - add one location's hashed CA dir to
212 * its upstream (proxy_ssl) trust store.
213 *
214 * WHAT: For a single location conf array, when brix_proxy_ssl_capath is set,
215 * add the directory as an OpenSSL hashed lookup to the location's upstream
216 * SSL_CTX; NGX_OK when unset. Fatal (NGX_ERROR) when the location has no
217 * https proxy_pass — the directive would otherwise silently protect nothing.
218 *
219 * WHY: second half of brix_proxy_ssl_capath (see module_directives.c): the
220 * parse-time handler could only seed one <hash>.N file through the stock
221 * proxy_ssl_trusted_certificate slot, because the upstream SSL_CTX does not
222 * exist until the proxy module's merge. Here — after all merges — the ctx is
223 * live, so the whole directory becomes a verify-time hashed lookup: every CA
224 * in /etc/grid-security/certificates is trusted and IGTF package updates are
225 * picked up without a reload.
226 *
227 * HOW: reads the webdav loc-conf for the capath, then casts the proxy
228 * module's private loc-conf to its public first member
229 * (ngx_http_upstream_conf_t) to reach upstream->ssl->ctx. The ssl object is
230 * per-location here, not shared: injecting the trusted-certificate seed made
231 * this location "SSL-configured", so the proxy merge gave it its own ctx.
232 * X509_STORE_load_locations(store, NULL, dir) appends the hash-dir lookup
233 * (additive — the seeded file stays trusted). Conf tokens and
234 * ngx_conf_full_name results are NUL-terminated, so capath->data is a valid
235 * C string.
236 */
237static ngx_int_t
238webdav_postconf_proxy_capath_apply(ngx_conf_t *cf, void **loc_conf,
239 ngx_str_t *name)
240{
241 ngx_http_brix_webdav_loc_conf_t *wlcf;
242 ngx_http_upstream_conf_t *ucf;
243 X509_STORE *store;
244
245 wlcf = loc_conf[ngx_http_brix_webdav_module.ctx_index];
246 if (wlcf == NULL || wlcf->proxy_ssl_capath.len == 0) {
247 return NGX_OK;
248 }
249
250 ucf = loc_conf[ngx_http_proxy_module.ctx_index];
251
252 if (ucf == NULL || ucf->ssl == NULL || ucf->ssl->ctx == NULL) {
253 ngx_log_error(NGX_LOG_EMERG, cf->log, 0,
254 "brix_proxy_ssl_capath in location \"%V\" requires "
255 "\"proxy_pass https://...\" in the same location",
256 name);
257 return NGX_ERROR;
258 }
259
260 store = SSL_CTX_get_cert_store(ucf->ssl->ctx);
261 if (store == NULL
262 || X509_STORE_load_locations(store, NULL,
263 (const char *) wlcf->proxy_ssl_capath.data) != 1)
264 {
265 ngx_log_error(NGX_LOG_EMERG, cf->log, 0,
266 "brix_proxy_ssl_capath \"%V\": cannot add hashed "
267 "CA-directory lookup to the upstream trust store",
268 &wlcf->proxy_ssl_capath);
269 return NGX_ERROR;
270 }
271
272 ngx_log_error(NGX_LOG_INFO, cf->log, 0,
273 "brix_webdav: added hashed CA dir \"%V\" to the upstream "
274 "(proxy_ssl) trust store for location %V",
275 &wlcf->proxy_ssl_capath, name);
276 return NGX_OK;
277}

404/*
405 * webdav_postconf_setup_ssl_ctx - apply GSI-proxy and kTLS flags to one server.
406 *
407 * WHAT: For a single server block, enable X509_V_FLAG_ALLOW_PROXY_CERTS on its
408 * SSL verify params when proxy_certs is on, add the brix_ssl_client_capath
409 * hashed CA directory to its verify store when set (returning NGX_ERROR when
410 * that directory is unusable — a fatal config error), and opt its TLS context
411 * into kernel-TLS when brix_ktls is on; returns NGX_OK for servers without
412 * WebDAV config or without an SSL context (no-op).
413 *
414 * WHY: GSI proxy-cert chains are rejected by OpenSSL unless the proxy-certs flag
415 * is set (AGENTS.md INVARIANT), and kTLS offload must be armed per-server while
416 * the ssl + brix_ktls directives live at server level. Isolating the per-server
417 * body lets the caller iterate cleanly and keeps this security-load-bearing flag
418 * manipulation in one auditable place.
419 *
420 * HOW: Reads the WebDAV loc-conf and SSL srv-conf from the server's context and
421 * early-returns if either is absent. When proxy_certs is set it fetches the
422 * verify params via SSL_CTX_get0_param() and sets the proxy-cert flag (logging
423 * INFO); when ssl_client_capath is set it delegates to
424 * webdav_postconf_load_client_capath() and propagates its failure; when
425 * common.ktls is set it calls brix_http_ktls_enable_ctx(). Registration
426 * effects are identical whether or not this is split out.
427 */
428static ngx_int_t
429webdav_postconf_setup_ssl_ctx(ngx_conf_t *cf, ngx_http_core_srv_conf_t *cscf)
430{
431 ngx_http_conf_ctx_t *ctx = cscf->ctx;
432 ngx_http_brix_webdav_loc_conf_t *wdcf;
433 ngx_http_ssl_srv_conf_t *sslcf;
434 X509_VERIFY_PARAM *param;
435
436 wdcf = ctx->loc_conf[ngx_http_brix_webdav_module.ctx_index];
437 if (wdcf == NULL) {
438 return NGX_OK;
439 }
440
441 sslcf = ctx->srv_conf[ngx_http_ssl_module.ctx_index];
442 if (sslcf == NULL || sslcf->ssl.ctx == NULL) {
443 return NGX_OK;
444 }
445
446 if (wdcf->proxy_certs) {
447 param = SSL_CTX_get0_param(sslcf->ssl.ctx);
448 if (param) {
449 X509_VERIFY_PARAM_set_flags(param, X509_V_FLAG_ALLOW_PROXY_CERTS);
450 ngx_log_error(NGX_LOG_INFO, cf->log, 0,
451 "brix_webdav: enabled X509_V_FLAG_ALLOW_PROXY_CERTS"
452 " on SSL context for server %V",
453 &cscf->server_name);
454 }
455 }
456
457 if (wdcf->ssl_client_capath.len > 0
458 && webdav_postconf_load_client_capath(cf, cscf, sslcf,
459 &wdcf->ssl_client_capath)
460 != NGX_OK)
461 {
462 return NGX_ERROR;
463 }
464
465 /* kTLS: opt every https server's TLS context into kernel-TLS so HTTPS
466 * GET sendfiles over kTLS and PUT decrypts in-kernel (unless brix_ktls
467 * off on that server). The server-level webdav loc-conf carries the flag
468 * (default on) for EVERY server — WebDAV and S3 alike, since brix_webdav/
469 * brix_s3 typically sit in a location block while ssl + brix_ktls are
470 * server-level. Transparent no-op for non-offloadable ciphers. */
471 if (wdcf->common.ktls) {
472 brix_http_ktls_enable_ctx(sslcf->ssl.ctx, cf->log,
473 &cscf->server_name);
474 }
475
476 return NGX_OK;
477}

479/*
480 * webdav_postconf_setup_thread_pool - resolve the async-I/O pool for one server.
481 *
482 * WHAT: For a WebDAV-enabled server block, look up the configured thread pool
483 * (falling back to the "default" pool) and store it on the common conf, logging
484 * a NOTICE for both the found and the not-found cases; a no-op for servers that
485 * lack a WebDAV conf or have WebDAV disabled.
486 *
487 * WHY: Async file I/O for PUT/GET runs on an nginx thread pool resolved once at
488 * config time. The pool name is per-server (explicit or "default"), so the
489 * lookup must run per server block; keeping it in a helper mirrors the SSL-setup
490 * split and keeps the loop body in the caller trivial.
491 *
492 * HOW: Reads the WebDAV loc-conf from the server context, early-returns if it is
493 * absent or common.enable is unset, selects the configured pool name or the
494 * shared "default", and calls ngx_thread_pool_get(); the returned pointer (or
495 * NULL) is written to common.thread_pool with the matching NOTICE log.
496 */
497static void
498webdav_postconf_setup_thread_pool(ngx_conf_t *cf,
499 ngx_http_core_srv_conf_t *cscf)
500{
501 static ngx_str_t default_pool_name = ngx_string("default");
502
503 ngx_http_conf_ctx_t *ctx = cscf->ctx;
504 ngx_http_brix_webdav_loc_conf_t *wdcf;
505 ngx_str_t *pool_name;
506
507 wdcf = ctx->loc_conf[ngx_http_brix_webdav_module.ctx_index];
508 if (wdcf == NULL || !wdcf->common.enable) {
509 return;
510 }
511
512 pool_name = (wdcf->common.thread_pool_name.len > 0)
513 ? &wdcf->common.thread_pool_name
514 : &default_pool_name;
515
516 wdcf->common.thread_pool = ngx_thread_pool_get(cf->cycle, pool_name);
517 if (wdcf->common.thread_pool == NULL) {
518 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
519 "brix_webdav: thread pool \"%V\" not found - "
520 "async file I/O disabled (add a thread_pool directive)",
521 pool_name);
522 } else {
523 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
524 "brix_webdav: using thread pool \"%V\" for async file I/O",
525 pool_name);
526 }
527}

529/*
530 * webdav_postconf_setup_servers - run per-server SSL and thread-pool setup.
531 *
532 * WHAT: Iterate every server block twice — first applying GSI-proxy/kTLS flags
533 * to each SSL context, then resolving each WebDAV server's async-I/O thread pool.
534 *
535 * WHY: The original hook did both passes over cmcf->servers inline; preserving
536 * two separate passes (rather than fusing them) keeps behavior identical — the
537 * SSL flags are all set before any thread-pool lookup, matching the prior order
538 * of side effects and log output.
539 *
540 * HOW: Fetches the servers array once, loops applying webdav_postconf_setup_ssl_ctx()
541 * to each element (propagating NGX_ERROR — an unusable brix_ssl_client_capath
542 * is a fatal config error), then loops again applying
543 * webdav_postconf_setup_thread_pool(), which cannot fail.
544 */
545static ngx_int_t
546webdav_postconf_setup_servers(ngx_conf_t *cf, ngx_http_core_main_conf_t *cmcf)
547{
548 ngx_http_core_srv_conf_t **cscfp = cmcf->servers.elts;
549 ngx_uint_t s;
550
551 for (s = 0; s < cmcf->servers.nelts; s++) {
552 if (webdav_postconf_setup_ssl_ctx(cf, cscfp[s]) != NGX_OK) {
