7/*
8 * async_job.h — minimal async job lifecycle helpers.
9 *
10 * Several features (WebDAV HTTP-TPC, S3 PUT, native read/write AIO, Qckscan)
11 * run work outside the main request path. They all repeat the same lifecycle
12 * mechanics: allocate per-job state, register cleanup, finalize metrics exactly
13 * once, and abort staged temp files on failure.
14 *
15 * This is deliberately small. It provides:
16 * brix_async_job_t — per-job lifecycle state
17 * brix_async_job_init() — zero and link the job to its owner
18 * brix_async_job_set_cleanup() — register a cleanup callback
19 * brix_async_job_cleanup_once() — idempotent cleanup (safe to call twice)
20 *
21 * Each protocol surface wraps the job in its own struct; this helper only
22 * manages the lifecycle, not the I/O or the wire response.
23 */
24
25typedef void (*brix_job_cleanup_pt)(void *data);
