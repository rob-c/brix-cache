2 * cephfs_layout.c — typed decoders for CephFS on-RADOS metadata (see header).
3 *
4 * Field orders and version guards mirror Ceph v18.2.4 verbatim:
5 * - inode_t::decode src/include/cephfs/types.h
6 * - InodeStoreBase::decode_bare src/mds/CInode.cc
7 * - CDir::_load_dentry src/mds/CDir.cc (the dentry value wrapper)

9 * The byte layout is cross-checked against captured reef fixtures
10 * (tests/ceph/fixtures/reef-18.2.4). Every decode is bounds-safe via cephfs_denc:
11 * we test the cursor's sticky error once at the end rather than per field.

40/* Decode inode_t up to mtime (everything the driver needs), honouring the
41 * version guards from inode_t::decode, then frame-skip the remaining fields.
42 * v>=4 : dir_layout (8 bytes) present
43 * v>=5 : truncate_pending (u32) present
44 * Fields after mtime (atime, dirstat/rstat, client_ranges, quota, btime, fscrypt,
45 * ...) are not needed here and are skipped by jumping to the frame end. */
46int
47cephfs_decode_inode(cephfs_denc_t *d, cephfs_inode_t *out)
48{
49 cephfs_denc_frame_t f;
50 uint8_t v;
51
52 v = cephfs_denc_start(d, &f);
53 out->struct_v = v;
54
55 out->ino = cephfs_denc_u64(d);
56 (void) cephfs_denc_u32(d); /* rdev */
57 decode_utime(d, &out->ctime_sec, &out->ctime_nsec);/* ctime */
58 out->mode = cephfs_denc_u32(d);
59 out->uid = cephfs_denc_u32(d);
60 out->gid = cephfs_denc_u32(d);
61 out->nlink = cephfs_denc_u32(d);
62 (void) cephfs_denc_u8(d); /* anchored (bool) */
63
64 if (v >= 4) {
65 cephfs_denc_skip(d, 8); /* dir_layout */
66 }
67
68 if (cephfs_decode_file_layout(d, &out->layout) != 0) {
69 return -1;
70 }
71
72 out->size = cephfs_denc_u64(d);
73 (void) cephfs_denc_u32(d); /* truncate_seq */
74 (void) cephfs_denc_u64(d); /* truncate_size */
75 (void) cephfs_denc_u64(d); /* truncate_from */
76 if (v >= 5) {
77 (void) cephfs_denc_u32(d); /* truncate_pending */
78 }
79 decode_utime(d, &out->mtime_sec, &out->mtime_nsec); /* mtime */
80
81 cephfs_denc_finish(d, &f); /* skip the rest */
82 return cephfs_denc_ok(d) ? 0 : -1;
83}

85/* frag_t is encoded as a single u32 `_enc` = (bits << 24) | value. */
86static uint32_t frag_bits (uint32_t enc) { return enc >> 24; }
87static uint32_t frag_value(uint32_t enc) { return enc & 0x00ffffffu; }
88static uint32_t
89frag_make_child(uint32_t enc, uint32_t nbits, uint32_t i)
90{
91 uint32_t cb = frag_bits(enc) + nbits;
92 uint32_t cv = ((frag_value(enc) << nbits) | i) & 0x00ffffffu;
93 return (cb << 24) | cv;
94}

96/* one split-map entry: frag `enc` splits into 2^nway children (nway<=0 = leaf). */
97typedef struct { uint32_t enc; int32_t nway; } cephfs_frag_split_t;
98
99/* ---- Read the fragtree split map from the cursor ----
100 *
101 * WHAT: Decodes the leading u32 count followed by that many (enc:u32, nway:u32)
102 * split-map entries into splits[], writing the entry count to *nsplit_out.
103 * Returns 0 on success, -1 if the cursor faulted or the count is implausible.
104 *
105 * WHY: The split map is a fixed wire prefix of the CephFS fragtree encoding; it
106 * must be read byte-exactly (count, then count pairs) before the tree walk.
107 * An implausibly large count is a corrupt/hostile buffer and is rejected here
108 * (sticky-faulting the cursor) so the walk never indexes past splits[].
109 *
110 * HOW:
111 * 1. Read the u32 entry count; bail if the cursor is already faulted.
112 * 2. Reject counts above CEPHFS_MAX_FRAGS (fault the cursor, return -1).
113 * 3. Read each (enc, nway) pair verbatim into splits[].
114 * 4. Re-check the cursor once; publish the count and return 0 on success.
115 */
116static int
117cephfs_read_frag_splits(cephfs_denc_t *d, cephfs_frag_split_t *splits,
118 uint32_t *nsplit_out)
119{
120 uint32_t nsplit, i;
121
122 nsplit = cephfs_denc_u32(d);
123 if (!cephfs_denc_ok(d)) { return -1; }
124 if (nsplit > CEPHFS_MAX_FRAGS) { d->err = 1; return -1; } /* implausible */
125 for (i = 0; i < nsplit; i++) {
126 splits[i].enc = cephfs_denc_u32(d);
127 splits[i].nway = (int32_t) cephfs_denc_u32(d);
128 }
129 if (!cephfs_denc_ok(d)) { return -1; }
130 *nsplit_out = nsplit;
131 return 0;
132}

134/* ---- Look up how many bits a frag node splits by ----
135 *
136 * WHAT: Scans the split map for an entry whose `enc` equals f and returns its
137 * nway; returns -1 when f is not a split node (i.e. it is a leaf).
138 *
139 * WHY: The tree walk must decide, per popped frag, whether it is an interior
140 * node (present in the split map) or a leaf. Isolating the linear scan keeps
141 * the walk loop flat and the map representation private to this file.
142 *
143 * HOW:
144 * 1. Linearly compare each split entry's enc against f.
145 * 2. Return the matching entry's nway on the first hit; -1 if none matches.
146 */
147static int32_t
148cephfs_frag_lookup_nway(const cephfs_frag_split_t *splits, uint32_t nsplit,
149 uint32_t f)
150{
151 uint32_t i;
152
153 for (i = 0; i < nsplit; i++) {
154 if (splits[i].enc == f) { return splits[i].nway; }
155 }
156 return -1;
157}

159/* ---- Push the children of a split frag onto the walk stack ----
160 *
161 * WHAT: Expands an interior frag f that splits by nb bits into its 2^nb child
162 * frags, pushing each onto stack[] (bounded by CEPHFS_MAX_FRAGS). Sets *trunc
163 * to 1 if any child could not be pushed because the stack was full.
164 *
165 * WHY: Keeping fan-out expansion in one helper caps the branching that would
166 * otherwise dominate the walk loop's complexity, and centralises the two
167 * safety caps (fan-out and stack depth) that stop a hostile tree from
168 * exploding memory or overrunning the fixed stack.
169 *
170 * HOW:
171 * 1. Cap nb at 8 so fan-out never exceeds 256 children.
172 * 2. Compute child frags via frag_make_child (byte-exact frag arithmetic).
173 * 3. Push each child while the stack has room; otherwise mark truncation.
174 */
175static void
176cephfs_frag_push_children(uint32_t f, int32_t nb, uint32_t *stack,
177 uint32_t *sp, int *trunc)
178{
179 uint32_t children, c;
180
181 if (nb > 8) { nb = 8; } /* cap fan-out (256) */
182 children = 1u << nb;
183 for (c = 0; c < children; c++) {
184 if (*sp < CEPHFS_MAX_FRAGS) {
185 stack[(*sp)++] = frag_make_child(f, (uint32_t) nb, c);
186 } else {
187 *trunc = 1;
188 }
189 }
190}

192/* ---- Decode a CephFS fragtree into its leaf frags ----
193 *
194 * WHAT: Reads the split map, then walks the fragtree from the root frag,
195 * collecting up to `max` leaf frags into leaves[] and writing the emitted
196 * count to *count. *truncated (if non-NULL) is set when leaves or the walk
197 * stack overflowed. Returns 0 on success, -1 on a cursor fault.
198 *
199 * WHY: The driver needs the set of directory fragment leaves to locate dir
200 * objects; the encoding is a split map plus an implicit tree, so we
201 * materialise leaves by an explicit (non-recursive) walk that stays bounded
202 * under corrupt input.
203 *
204 * HOW:
205 * 1. Read the split map via cephfs_read_frag_splits.
206 * 2. Seed the stack with the root frag (enc 0).
207 * 3. Pop a frag; a node absent from the split map (nway<=0) is a leaf —
208 * record it (or flag truncation past `max`); otherwise expand its children.
209 * 4. Clamp *count to `max`, publish truncation, and report the cursor state.
210 */
211int
212cephfs_decode_fragtree(cephfs_denc_t *d, uint32_t *leaves, uint32_t max,
213 uint32_t *count, int *truncated)
214{
215 cephfs_frag_split_t splits[CEPHFS_MAX_FRAGS];
216 uint32_t stack[CEPHFS_MAX_FRAGS];
217 uint32_t nsplit, sp = 0, nleaf = 0;
218 int trunc = 0;
219
220 if (truncated != NULL) { *truncated = 0; }
221
222 if (cephfs_read_frag_splits(d, splits, &nsplit) != 0) { return -1; }
223
224 /* walk the tree from the root frag (enc 0); a node not in the split map is a
225 * leaf. Explicit stack — no recursion. */
226 stack[sp++] = 0;
227 while (sp > 0) {
228 uint32_t f = stack[--sp];
229 int32_t nb = cephfs_frag_lookup_nway(splits, nsplit, f);
230
231 if (nb <= 0) { /* leaf */
232 if (nleaf < max) { leaves[nleaf] = f; } else { trunc = 1; }
233 nleaf++;
234 } else {
235 cephfs_frag_push_children(f, nb, stack, &sp, &trunc);
236 }
237 }
238
239 *count = (nleaf <= max) ? nleaf : max;
240 if (truncated != NULL) { *truncated = trunc; }
241 return cephfs_denc_ok(d) ? 0 : -1;
242}

276/* Decode a primary inode body (InodeStore decode_bare prefix): the framed inode;
277 * the symlink target if it is a symlink; the directory fragment tree; and the
278 * xattr map. Fields after the xattrs (snap_blob, old_inodes, oldest_snap,
279 * damage_flags) are not needed and are frame-skipped by the caller (for 'i', via
280 * the InodeStore frame). */
281static int
282decode_primary_body(cephfs_denc_t *d, cephfs_dentry_t *out)
283{
284 out->kind = CEPHFS_DENTRY_PRIMARY;
285 out->symlink = NULL;
286 out->symlink_len = 0;
287 out->nfrags = 0;
288 out->frags_truncated = 0;
289 out->nxattrs = 0;
290 out->xattrs_truncated = 0;
291
292 if (cephfs_decode_inode(d, &out->inode) != 0) {
293 return -1;
294 }
295 if (cephfs_mode_is_link(out->inode.mode)) {
296 cephfs_denc_str(d, &out->symlink, &out->symlink_len);
297 }
298 cephfs_decode_fragtree(d, out->frag_enc, CEPHFS_MAX_FRAGS,
299 &out->nfrags, &out->frags_truncated);
300 cephfs_decode_xattrs(d, out->xattrs, CEPHFS_MAX_XATTRS,
301 &out->nxattrs, &out->xattrs_truncated);
302 return cephfs_denc_ok(d) ? 0 : -1;
303}

305int
306cephfs_decode_dentry(const void *buf, size_t len, cephfs_dentry_t *out)
307{
308 cephfs_denc_t d;
309 uint8_t type;
310
311 cephfs_denc_init(&d, buf, len);
312
313 (void) cephfs_denc_u64(&d); /* snapid_t first */
314 type = cephfs_denc_u8(&d);
315 if (!cephfs_denc_ok(&d)) {
316 return -1;
317 }
318
319 /* remote dentry (hardlink): inodeno_t ino, then d_type; 'L' also carries an
320 * alternate_name we don't need. Only ino + d_type are stored here — resolving
321 * the inode itself is the caller's job. */
322 if (type == 'L' || type == 'l') {
323 out->kind = CEPHFS_DENTRY_REMOTE;
324 out->remote_ino = cephfs_denc_u64(&d);
325 out->remote_d_type = cephfs_denc_u8(&d);
326 return cephfs_denc_ok(&d) ? 0 : -1;
327 }
328
329 /* primary dentry. 'i' (old) wraps the InodeStore in a small frame that also
330 * carries an alternate_name; 'I' (new) stores the InodeStore bare. */
331 if (type == 'i') {
332 cephfs_denc_frame_t wrap, store;
333 int rc;
334
335 cephfs_denc_start(&d, &wrap); /* dentry wrapper (v>=2) */
336 if (wrap.struct_v >= 2) {
337 cephfs_denc_str(&d, NULL, NULL); /* alternate_name (discard) */
338 }
339 cephfs_denc_start(&d, &store); /* InodeStore frame */
340 rc = decode_primary_body(&d, out);
341 cephfs_denc_finish(&d, &store); /* skip fragtree/xattrs/... */
342 cephfs_denc_finish(&d, &wrap);
343 if (rc != 0 || !cephfs_denc_ok(&d)) {
344 return -1;
345 }
346 return 0;
347 }
348
349 if (type == 'I') {
350 /* InodeStore stored bare (no surrounding frame) */
351 return decode_primary_body(&d, out);
352 }
353
354 /* unknown dentry marker */
355 return -1;
356}
