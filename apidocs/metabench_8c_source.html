10/* WHAT: choose D (dirs) and F (files/dir) so per-worker CREATE op count is close
11 * to `ops_per_worker`. WHY: hit ~1000 total with a small balanced tree.
12 * HOW: CREATE ops/worker = 2 + 2*D + 3*D*F; fix F=4, solve D, clamp >=1. */
13void metabench_plan_init(metabench_plan *p, int workers, int ops_per_worker,
14 int p99_ceil_ms)
15{
16 int F = 4;
17 int D = (ops_per_worker - 2) / (2 + 3 * F); /* 2 + 2D + 3DF = budget */
18 if (D < 1) {
19 D = 1;
20 }
21 p->workers = workers < 1 ? 1 : workers;
22 p->files_per_dir = F;
23 p->dirs_per_worker = D;
24 p->p99_ceil_ms = p99_ceil_ms;
25}

27/* counted CREATE ops, all workers. */
28long metabench_storm_ops(const metabench_plan *p)
29{
30 long per = 2L + 2L * p->dirs_per_worker
31 + 3L * p->dirs_per_worker * p->files_per_dir;
32 return per * p->workers;
33}

35/* WHAT: emit the exact namespace a CREATE run leaves, sorted by path.
36 * HOW: /w<id> (0755), /w<id>/d<d> (0700), /w<id>/d<d>/f<f> (0640). Built in
37 * natural order then qsort'd, so callers can diff against a sorted listing. */
38static int entry_cmp(const void *a, const void *b)
39{
40 return strcmp(((const metabench_entry *) a)->path,
41 ((const metabench_entry *) b)->path);
42}

44size_t metabench_expected(const metabench_plan *p, metabench_entry *out,
45 size_t cap)
46{
47 size_t n = 0;
48
49 for (int w = 0; w < p->workers; w++) {
50 if (n < cap) {
51 snprintf(out[n].path, sizeof(out[n].path), "/w%d", w);
52 out[n].mode = 0755;
53 out[n].is_dir = 1;
54 }
55 n++;
56 for (int d = 0; d < p->dirs_per_worker; d++) {
57 if (n < cap) {
58 snprintf(out[n].path, sizeof(out[n].path), "/w%d/d%d", w, d);
59 out[n].mode = 0700;
60 out[n].is_dir = 1;
61 }
62 n++;
63 for (int f = 0; f < p->files_per_dir; f++) {
64 if (n < cap) {
65 snprintf(out[n].path, sizeof(out[n].path),
66 "/w%d/d%d/f%d", w, d, f);
67 out[n].mode = 0640;
68 out[n].is_dir = 0;
69 }
70 n++;
71 }
72 }
73 }
74
75 size_t sortable = n < cap ? n : cap;
76 qsort(out, sortable, sizeof(*out), entry_cmp);
77 return n;
78}

81static int dbl_cmp(const void *a, const void *b)
82{
83 double x = *(const double *) a, y = *(const double *) b;
84 return (x > y) - (x < y);
85}

87double metabench_percentile(const double *ms, size_t n, double pct)
88{
89 if (n == 0) {
90 return 0.0;
91 }
92
93 double *c = malloc(n * sizeof(*c));
94 if (c == NULL) {
95 return 0.0;
96 }
97 memcpy(c, ms, n * sizeof(*c));
98 qsort(c, n, sizeof(*c), dbl_cmp);
99
100 double rank = (pct / 100.0) * (double) (n - 1);
101 size_t lo = (size_t) rank;
102 double frac = rank - (double) lo;
103 double v = (lo + 1 < n) ? c[lo] + frac * (c[lo + 1] - c[lo]) : c[lo];
104
105 free(c);
106 return v;
107}
