1/*
2 * checksum.c — local streaming checksums + the server-side kXR_Qcksum query.
3 *
4 * WHAT: Compute adler32 / crc32c / md5 over a local file descriptor, and ask the
5 * server for a file's checksum (kXR_query, infotype kXR_Qcksum). Both yield
6 * a lowercase hex digest so xrdcp --cksum can compare local vs source.
7 * WHY: --cksum[:source] needs (a) a digest of the bytes we transferred and (b)
8 * the server's digest of the same file, then asserts they agree.
9 * HOW: The fd compute is delegated to the SHARED kernel brix_cksum_*_fd
10 * (src/core/compat/checksum_core.c, linked via libxrdproto) — the exact same
11 * adler32/crc32c/md5 code the nginx module runs, so client and server agree
12 * by construction (single source). This file only maps the client's algo
13 * enum, hex-encodes, and drives the Qcksum wire query (payload "<algo>
14 * <path>" → reply "<algo> <hexdigest>").
15 *
16 * wire: XProtocol.hh kXR_query infotype kXR_Qcksum — body "<algo> <path>" → "<algo> <hex>".
17 */

28int
29brix_cksum_algo_parse(const char *name, brix_cksum_algo *out)
30{
31 if (name == NULL) {
32 return -1;
33 }
34 if (strcmp(name, "adler32") == 0) { *out = XRDC_CK_ADLER32; return 0; }
35 if (strcmp(name, "crc32c") == 0) { *out = XRDC_CK_CRC32C; return 0; }
36 if (strcmp(name, "md5") == 0) { *out = XRDC_CK_MD5; return 0; }
37 if (strcmp(name, "crc64") == 0) { *out = XRDC_CK_CRC64; return 0; }
38 if (strcmp(name, "crc64xz") == 0) { *out = XRDC_CK_CRC64; return 0; }
39 if (strcmp(name, "crc64nvme") == 0) { *out = XRDC_CK_CRC64NVME; return 0; }
40 if (strcmp(name, "zcrc32") == 0) { *out = XRDC_CK_ZCRC32; return 0; }
41 return -1;
42}

44int
45brix_cksum_fd(int fd, brix_cksum_algo algo, char *hex, size_t hexsz,
46 brix_status *st)
47{
48 /* Delegate the compute to the shared (ngx-free) kernel — the same code the
49 * nginx module uses (src/core/compat/checksum_core.c via libxrdproto). The kernel
50 * preads from offset 0; callers pass freshly-opened regular-file fds. */
51 if (algo == XRDC_CK_MD5) {
52 unsigned char dg[64]; /* EVP_MAX_MD_SIZE */
53 unsigned int dn = 0;
54 if (brix_cksum_digest_fd(BRIX_CK_MD5, fd, dg, &dn) != 0) {
55 brix_status_set(st, XRDC_ESOCK, errno, "md5: %s", strerror(errno));
56 return -1;
57 }
58 if (hexsz < (size_t) dn * 2 + 1) {
59 brix_status_set(st, XRDC_EUSAGE, 0, "hex buffer too small");
60 return -1;
61 }
62 brix_hex_encode(dg, dn, hex);
63 return 0;
64 }
65
66 if (algo == XRDC_CK_CRC64 || algo == XRDC_CK_CRC64NVME) {
67 int kind = (algo == XRDC_CK_CRC64) ? BRIX_CK_CRC64
68 : BRIX_CK_CRC64NVME;
69 uint64_t value;
70 if (brix_cksum_u64_fd(kind, fd, &value) != 0) {
71 brix_status_set(st, XRDC_ESOCK, errno, "checksum read: %s",
72 strerror(errno));
73 return -1;
74 }
75 if (hexsz < 17) { /* 16 hex digits + NUL */
76 brix_status_set(st, XRDC_EUSAGE, 0, "hex buffer too small");
77 return -1;
78 }
79 snprintf(hex, hexsz, "%016llx", (unsigned long long) value);
80 return 0;
81 }
82
83 {
84 int kind = (algo == XRDC_CK_ADLER32) ? BRIX_CK_ADLER32
85 : (algo == XRDC_CK_ZCRC32) ? BRIX_CK_ZCRC32
86 : BRIX_CK_CRC32C;
87 uint32_t value;
88 if (brix_cksum_u32_fd(kind, fd, &value) != 0) {
89 brix_status_set(st, XRDC_ESOCK, errno, "checksum read: %s",
90 strerror(errno));
91 return -1;
92 }
93 if (hexsz < 9) {
94 brix_status_set(st, XRDC_EUSAGE, 0, "hex buffer too small");
95 return -1;
96 }
97 snprintf(hex, hexsz, "%08x", value); /* 8 zero-padded hex digits */
98 return 0;
99 }
100}

102int
103brix_query_cksum(brix_conn *c, const char *path, const char *algo_name,
104 char *hex, size_t hexsz, brix_status *st)
105{
106 ClientQueryRequest req;
107 uint16_t status;
108 uint8_t *body = NULL;
109 uint32_t blen = 0;
110 char payload[XRDC_PATH_MAX + 32];
111 int plen;
112 const char *sp;
113
114 /* kXR_Qcksum wire format is the PATH, with the algorithm requested via the
115 * standard "?cks.type=<algo>" CGI — NOT a leading "<algo> <path>" (which a
116 * standard server, e.g. EOS, treats as a bogus relative path and rejects:
117 * "Check summing relative path 'adler32 /…' is disallowed"). Our own server
118 * accepts a bare path too, so this is correct for both. */
119 if (algo_name != NULL && algo_name[0] != '\0') {
120 plen = snprintf(payload, sizeof(payload), "%s%ccks.type=%s", path,
121 (strchr(path, '?') != NULL) ? '&' : '?', algo_name);
122 } else {
123 plen = snprintf(payload, sizeof(payload), "%s", path);
124 }
125 if (plen < 0 || (size_t) plen >= sizeof(payload)) {
126 brix_status_set(st, XRDC_EUSAGE, 0, "checksum query path too long");
127 return -1;
128 }
129
130 memset(&req, 0, sizeof(req));
131 req.requestid = htons(kXR_query);
132 {
133 xrdw_query_req_t b = { .infotype = kXR_Qcksum };
134 xrdw_query_req_pack(&b, ((ClientRequestHdr *) &req)->body);
135 }
136 /* fhandle stays zero: this is a path-based (not open-handle) query. */
137
138 /* Read-only/idempotent: a re-query after a sever yields the same digest. */
139 if (brix_roundtrip_resilient(c, &req, payload, (uint32_t) plen,
140 XRDC_OP_READONLY, 0,
141 &status, &body, &blen, st) != 0) {
142 return -1;
143 }
144
145 /* Reply: "<algo> <hexdigest>" — take the token after the (last) space. */
146 {
147 char tmp[256];
148 uint32_t n = (blen < sizeof(tmp) - 1) ? blen : (uint32_t) (sizeof(tmp) - 1);
149 memcpy(tmp, body, n);
150 tmp[n] = '\0';
151 free(body);
152
153 sp = strrchr(tmp, ' ');
154 sp = (sp != NULL) ? sp + 1 : tmp; /* tolerate a digest-only reply */
155 /* trim trailing whitespace/newline */
156 {
157 size_t L = strlen(sp);
158 while (L > 0 && (sp[L - 1] == '\n' || sp[L - 1] == '\r'
159 || sp[L - 1] == ' ')) {
160 ((char *) sp)[--L] = '\0';
161 }
162 if (L == 0 || L + 1 > hexsz) {
163 brix_status_set(st, XRDC_EPROTO, 0,
164 "unparseable checksum reply: \"%s\"", tmp);
165 return -1;
166 }
167 }
168 snprintf(hex, hexsz, "%s", sp);
169 }
170 return 0;
171}
