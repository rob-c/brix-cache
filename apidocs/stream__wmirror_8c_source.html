1/*
2 * stream_wmirror.c — Phase 24 W3: XRootD stream DATA-write mirroring.
3 *
4 * See stream_wmirror.h. Two halves:
5 *
6 * 1. Per-connection accumulation. A write-open's bytes are gathered into a
7 * bounded per-file buffer hanging off the client connection context
8 * (brix_ctx_t.wmirror). Only sequential kXR_write data is captured; a
9 * kXR_pgwrite (CRC-interleaved payload), a non-sequential offset, or a buffer
10 * that exceeds the per-file / per-connection cap aborts that file's mirror
11 * (counted) — never blocking the client.
12 *
13 * 2. Detached replay. On kXR_close, a complete sequential file is handed to a
14 * self-contained async shadow client (its own cycle-pool context, the same
15 * fire-and-forget lifetime as the read mirror, src/mirror/stream_mirror.c)
16 * that bootstraps a fresh shadow session and performs open(create) -> write
17 * -> close. The shadow-socket framing (flush / recv_frame) is shared with
18 * the read mirror via src/mirror/stream_mirror_io.c; the read+write handlers
19 * and start/finish lifecycle deliberately mirror stream_mirror.c.
20 *
21 * The shadow MUST be an isolated namespace — replaying writes onto the primary's
22 * backing store would corrupt it (see brix_mirror_writes).
23 */

64 * Replay state machine. The first three are the shared bootstrap (identical to
65 * the read mirror); the last three are this mirror's stateful create sequence.

139/*
140 * Replay the client's original open frame (header[24] + path/cgi payload) to the
141 * shadow. WHAT: re-sends the captured open verbatim so the shadow recreates the
142 * same path under its own namespace. HOW: copy into a fresh pool buffer (the
143 * saved open_frame is reused per launch and must stay intact), overwrite the
144 * streamid to 0x0002 (marks this as a mirror request, matching the read mirror),
145 * then flush. The original open's create/truncate flags are preserved as-is.
146 */
147static void
148wmir_send_open(brix_wmirror_replay_t *r)
149{
150 u_char *p = ngx_palloc(r->pool, r->open_frame_len);
151
152 if (p == NULL) { wmir_finish(r, 0); return; }
153 ngx_memcpy(p, r->open_frame, r->open_frame_len);
154 p[0] = 0; p[1] = 2; /* streamid 0x0002 */
155 r->wbuf = p; r->wbuf_len = r->open_frame_len; r->wbuf_pos = 0;
156 wmir_reset_frame(r);
157 r->phase = WMIR_OPEN;
158 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
159}

161/*
162 * Send the whole accumulated file as a SINGLE kXR_write to the shadow.
163 *
164 * WHAT/WHY: the accumulator only ever captured sequential writes starting at
165 * offset 0 (wmir_observe enforces contiguity), so the entire file collapses into
166 * one write at offset 0 — we do NOT need to replay the client's original chunk
167 * boundaries. HOW: hand-build the 24-byte ClientWriteRequest header:
168 * [0..1] streamid = 0x0002 (mirror marker)
169 * [2..3] requestid = kXR_write, big-endian
170 * [4..7] fhandle = the shadow's handle from the open response
171 * [8..15] offset = 0 (whole file from start; off_be is already zero)
172 * [16..19] reserved (left zero by ngx_memzero)
173 * [20..23] dlen = data_len, big-endian
174 * followed by data_len payload bytes. All multi-byte fields are network order.
175 */
176static void
177wmir_send_write(brix_wmirror_replay_t *r)
178{
179 size_t total = (size_t) 24 + r->data_len;
180 u_char *p = ngx_palloc(r->pool, total);
181 uint32_t dlen_be;
182 uint64_t off_be = 0;
183
184 if (p == NULL) { wmir_finish(r, 0); return; }
185 ngx_memzero(p, 24);
186 p[0] = 0; p[1] = 2;
187 p[2] = (u_char) (kXR_write >> 8); p[3] = (u_char) (kXR_write & 0xff);
188 ngx_memcpy(p + 4, r->shadow_fhandle, 4);
189 ngx_memcpy(p + 8, &off_be, 8); /* whole file at offset 0 */
190 dlen_be = htonl((uint32_t) r->data_len);
191 ngx_memcpy(p + 20, &dlen_be, 4);
192 if (r->data_len) { ngx_memcpy(p + 24, r->data, r->data_len); }
193 r->wbuf = p; r->wbuf_len = total; r->wbuf_pos = 0;
194 wmir_reset_frame(r);
195 r->phase = WMIR_WRITE;
196 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
197}

199/*
200 * Send kXR_close for the shadow handle — finalizes the replayed file. Same
201 * 24-byte header layout as the write: streamid 0x0002, requestid kXR_close at
202 * [2..3], the shadow fhandle at [4..7]; all other fields stay zero.
203 */
204static void
205wmir_send_close(brix_wmirror_replay_t *r)
206{
207 u_char *p = ngx_palloc(r->pool, 24);
208
209 if (p == NULL) { wmir_finish(r, 0); return; }
210 ngx_memzero(p, 24);
211 p[0] = 0; p[1] = 2;
212 p[2] = (u_char) (kXR_close >> 8); p[3] = (u_char) (kXR_close & 0xff);
213 ngx_memcpy(p + 4, r->shadow_fhandle, 4);
214 r->wbuf = p; r->wbuf_len = 24; r->wbuf_pos = 0;
215 wmir_reset_frame(r);
216 r->phase = WMIR_CLOSE;
217 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }

239/* Advance one phase; bootstrap phases re-post the read so pipelined frames are
240 * processed; request phases send the next frame (which arms the read itself). */
241static void
242wmir_dispatch(brix_wmirror_replay_t *r)
243{
244 switch (r->phase) {
245
246 case WMIR_HANDSHAKE:
247 if (r->resp_status != kXR_ok) { wmir_finish(r, 0); return; }
248 r->phase = WMIR_PROTOCOL;
249 break;
250
251 case WMIR_PROTOCOL:
252 if (r->resp_status != kXR_ok) { wmir_finish(r, 0); return; }
253 /* A shadow that demands TLS cannot continue the cleartext replay. The
254 * gotoTLS flag lives in the protocol response body at byte offset 4
255 * (a 4-byte big-endian flags word). */
256 if (r->resp_dlen >= 8 && r->resp_body != NULL) {
257 uint32_t flags_be;
258 ngx_memcpy(&flags_be, r->resp_body + 4, sizeof(flags_be));
259 if (ntohl(flags_be) & kXR_gotoTLS) { wmir_finish(r, 0); return; }
260 }
261 r->phase = WMIR_LOGIN;
262 break;
263
264 case WMIR_LOGIN:
265 /* authmore => the shadow wants credentials we have no way to supply on
266 * this anonymous replay; treat as a non-fatal stop (counted as error). */
267 if (r->resp_status == kXR_authmore) { wmir_finish(r, 0); return; }
268 if (r->resp_status != kXR_ok) { wmir_finish(r, 0); return; }
269 wmir_send_open(r); /* resets accumulator + sends + arms read */
270 return;
271
272 case WMIR_OPEN:
273 /* The shadow's open response body begins with its 4-byte file handle;
274 * everything downstream (write/close) addresses this handle, so a short
275 * or missing body means we cannot continue. */
276 if (!wmir_status_ok(r->resp_status) || r->resp_dlen < 4
277 || r->resp_body == NULL)
278 {
279 wmir_finish(r, 0); return; /* shadow open failed/diverged */
280 }
281 ngx_memcpy(r->shadow_fhandle, r->resp_body, 4);
282 wmir_send_write(r);
283 return;
284
285 case WMIR_WRITE:
286 if (!wmir_status_ok(r->resp_status)) { wmir_finish(r, 0); return; }
287 wmir_send_close(r);
288 return;
289
290 case WMIR_CLOSE:
291 /* Terminal: success iff the close was accepted — this is the only path
292 * that reports the replay as fully succeeded (ok=1). */
293 wmir_finish(r, wmir_status_ok(r->resp_status) ? 1 : 0);
294 return;
295 }
296
297 /* Only the bootstrap arms (HANDSHAKE/PROTOCOL) fall through to here — they
298 * advanced r->phase via `break` instead of sending a frame. The request
299 * arms (LOGIN/OPEN/WRITE/CLOSE) all `return` after queuing their next frame,
300 * whose wmir_flush() arms the read itself. Here we instead re-post the read
301 * manually: the pipelined bootstrap responses may already be sitting in the
302 * socket buffer, so re-posting drains them within the same loop cycle rather
303 * than waiting for a fresh readable event. */
304 wmir_reset_frame(r);
305 if (ngx_handle_read_event(r->conn->read, 0) != NGX_OK) {
306 wmir_finish(r, 0);
307 return;
308 }
309 ngx_post_event(r->conn->read, &ngx_posted_events);
310}

312/* Read event entry point: pull one complete frame, then hand it to the state
313 * machine. NGX_AGAIN re-arms and yields; the deadline timer (wmir_timeout_
314 * handler) is the only thing that bounds a stalled shadow. */
315static void
316wmir_read_handler(ngx_event_t *rev)
317{
318 ngx_connection_t *c = rev->data;
319 brix_wmirror_replay_t *r = c->data;
320 ngx_int_t rc;
321
322 if (rev->timedout) { wmir_finish(r, 0); return; }
323
324 rc = wmir_recv_frame(r);
325 if (rc == NGX_AGAIN) {
326 if (ngx_handle_read_event(rev, 0) != NGX_OK) { wmir_finish(r, 0); }
327 return;
328 }
329 if (rc == NGX_ERROR) { wmir_finish(r, 0); return; }
330 wmir_dispatch(r);
331}

333/*
334 * Write event entry point. Two jobs: (1) finalize an in-flight non-blocking
335 * connect() — the first writable event after EINPROGRESS means connect resolved,
336 * so SO_ERROR is checked to distinguish success from a deferred failure; (2)
337 * resume draining the current wbuf after an earlier NGX_AGAIN. This is the only
338 * place the `connecting` flag is cleared.
339 */
340static void
341wmir_write_handler(ngx_event_t *wev)
342{
343 ngx_connection_t *c = wev->data;
344 brix_wmirror_replay_t *r = c->data;
345
346 if (wev->timedout) { wmir_finish(r, 0); return; }
347
348 if (r->connecting) {
349 int err = 0;
350 socklen_t len = sizeof(err);
351 /* SO_ERROR carries the async connect() result; nonzero (or getsockopt
352 * failure) means the connection never came up. */
353 if (getsockopt(c->fd, SOL_SOCKET, SO_ERROR, (char *) &err, &len) == -1
354 || err != 0)
355 {
356 wmir_finish(r, 0);
357 return;
358 }
359 r->connecting = 0;
360 }
361
362 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
363}

373/*
374 * Single teardown path for the replay (success and every failure route here).
375 *
376 * WHAT: cancel the deadline timer, close the shadow connection, record the
377 * outcome metric, and free everything. OWNERSHIP/ORDER (important): r->data and
378 * r->open_frame are heap buffers (ngx_alloc) transferred in at launch and must
379 * be ngx_free()d explicitly; r itself was pcalloc'd FROM r->pool, so destroying
380 * the pool last frees r along with all its pool-allocated buffers. `pool` is
381 * cached first precisely because r becomes invalid after ngx_destroy_pool().
382 */
383static void
384wmir_finish(brix_wmirror_replay_t *r, int ok)
385{
386 ngx_pool_t *pool = r->pool;
387
388 if (r->tev.timer_set) { ngx_del_timer(&r->tev); }
389 if (r->conn != NULL) { ngx_close_connection(r->conn); r->conn = NULL; }
390
391 if (ok) {
392 BRIX_WMIR_METRIC_INC(mirror_stream_total);
393 } else {
394 BRIX_WMIR_METRIC_INC(mirror_stream_errors_total);
395 if (r->log_diverge) {
396 ngx_log_error(NGX_LOG_NOTICE, r->log, 0,
397 "xrootd write-mirror: shadow write replay failed/diverged");
398 }
399 }
400 if (r->data != NULL) { ngx_free(r->data); }
401 if (r->open_frame != NULL) { ngx_free(r->open_frame); }
402 ngx_destroy_pool(pool); /* frees r itself (last use above) */
403}

405/*
406 * Open the shadow socket, register it with nginx's event loop, queue the
407 * pipelined bootstrap, and kick off the (possibly async) connect.
408 *
409 * HOW: a non-blocking socket is wrapped in an ngx_connection_t whose read/write
410 * handlers drive the state machine; the bootstrap (handshake + protocol + login)
411 * is built into wbuf up front so it is sent the moment the socket is writable.
412 * connect() either completes immediately (rc==0 -> flush now) or returns
413 * EINPROGRESS (-> set `connecting`, finish in the write handler). Any setup
414 * failure routes through wmir_finish so the pool is always reclaimed.
415 */
416static void
417wmir_start(brix_wmirror_replay_t *r, ngx_msec_t timeout_ms)
418{
419 ngx_connection_t *c;
420 ngx_socket_t fd;
421 size_t bslen;
422 int rc;
423
424 fd = ngx_socket(r->sockaddr.ss_family, SOCK_STREAM, 0);
425 if (fd == (ngx_socket_t) -1 || ngx_nonblocking(fd) == -1) {
426 if (fd != (ngx_socket_t) -1) { ngx_close_socket(fd); }
427 wmir_finish(r, 0);
428 return;
429 }
430 c = ngx_get_connection(fd, r->log);
431 if (c == NULL) { ngx_close_socket(fd); wmir_finish(r, 0); return; }
432 c->pool = r->pool;
433 c->data = r;
434 c->recv = ngx_recv;
435 c->send = ngx_send;
436 c->read->handler = wmir_read_handler;
437 c->write->handler = wmir_write_handler;
438 c->read->log = c->write->log = r->log;
439 r->conn = c;
440
441 /* Bootstrap is three frames concatenated into one write so they pipeline:
442 * handshake, then ClientProtocolRequest, then ClientLoginRequest. */
443 bslen = XRD_HANDSHAKE_LEN + sizeof(ClientProtocolRequest)
444 + sizeof(ClientLoginRequest);
445 r->wbuf = ngx_palloc(r->pool, bslen);
446 if (r->wbuf == NULL) { wmir_finish(r, 0); return; }
447 brix_upstream_build_bootstrap(r->wbuf);
448 r->wbuf_len = bslen;
449 r->wbuf_pos = 0;
450 r->phase = WMIR_HANDSHAKE;
451
452 r->tev.handler = wmir_timeout_handler;
453 r->tev.data = r;
454 r->tev.log = r->log;
455 ngx_add_timer(&r->tev, timeout_ms ? timeout_ms : 5000);
456
457 rc = connect(fd, (struct sockaddr *) &r->sockaddr, r->socklen);
458 /* Only a hard, non-EINPROGRESS error is fatal here; EINPROGRESS is the
459 * normal async case and is resolved later via SO_ERROR. */
460 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
461 wmir_finish(r, 0);
462 return;
463 }
464 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
465 wmir_finish(r, 0);
466 return;
467 }
468 if (rc == 0) {
469 r->connecting = 0; /* connected synchronously — send now */
470 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
471 } else {
472 r->connecting = 1; /* completes in the write handler */
473 }
474}

476/* Launch a detached replay for a fully-buffered file. Ownership of the data
477 * buffer transfers to the replay (freed in wmir_finish). */
478static void
479wmir_launch(ngx_stream_brix_srv_conf_t *conf, brix_wmirror_file_t *f)
480{
481 brix_mirror_target_t *t;
482 ngx_pool_t *pool;
483 brix_wmirror_replay_t *r;
484 u_char *frame;
485
486 if (conf->mirror.targets == NULL || conf->mirror.targets->nelts == 0) {
487 return;
488 }
489 t = (brix_mirror_target_t *) conf->mirror.targets->elts; /* first target */
490 if (t->socklen == 0) { return; }
491
492 pool = ngx_create_pool(2048, ngx_cycle->log);
493 if (pool == NULL) { return; }
494 r = ngx_pcalloc(pool, sizeof(*r));
495 if (r == NULL) { ngx_destroy_pool(pool); return; }
496 r->pool = pool;
497 r->log = ngx_cycle->log; /* outlives the client connection */
498 r->log_diverge = conf->mirror.log_diverge ? 1 : 0;
499 r->port = t->port;
500 r->socklen = t->socklen;
501 ngx_memcpy(&r->sockaddr, &t->sockaddr, t->socklen);
502 ngx_cpystrn((u_char *) r->host,
503 t->host.data ? t->host.data : (u_char *) "?", sizeof(r->host));
504
505 /* Rebuild the open frame as a single heap buffer: the captured 24-byte open
506 * header followed by its path/cgi payload. This is ngx_alloc (not pool) so
507 * it survives independently and is ngx_free()d in wmir_finish. */
508 r->open_frame_len = (size_t) 24 + f->open_dlen;
509 frame = ngx_alloc(r->open_frame_len, ngx_cycle->log);
510 if (frame == NULL) { ngx_destroy_pool(pool); return; }
511 ngx_memcpy(frame, f->open_hdr, 24);
512 if (f->open_dlen && f->open_payload != NULL) {
513 ngx_memcpy(frame + 24, f->open_payload, f->open_dlen);
514 }
515 r->open_frame = frame;
516 /* Move the accumulated file bytes to the replay rather than copy: f->data is
517 * nulled so wmir_file_reset (called by the caller after this) won't free it;
518 * the replay now owns and frees it. */
519 r->data = f->data; /* transfer ownership */
520 r->data_len = f->data_len;
521 f->data = NULL; /* the replay frees it now */
522
523 wmir_start(r, conf->mirror.timeout_ms);
524}

527/*
528 * Free a per-file accumulator slot and return it to the empty state. WHAT:
529 * releases the two heap buffers (accumulated data + open payload) and zeroes the
530 * struct. WHY the guarded subtract: the connection-wide byte counter must be
531 * decremented by exactly this file's contribution; the `<=` guard defends
532 * against any accounting drift so total_buffered can never underflow. Safe to
533 * call on an already-empty slot (NULL buffers, data_len 0) and after launch has
534 * stolen f->data (set to NULL).
535 */
536static void
537wmir_file_reset(brix_wmirror_conn_t *wm, brix_wmirror_file_t *f)
538{
539 if (f->data != NULL) { ngx_free(f->data); }
540 if (f->open_payload != NULL) { ngx_free(f->open_payload); }
541 if (wm != NULL && f->data_len <= wm->total_buffered) {
542 wm->total_buffered -= f->data_len;
543 }
544 ngx_memzero(f, sizeof(*f));
545}

547/*
548 * Is data-write mirroring active for this server? Returns 1 only if ALL gates
549 * pass: mirroring enabled with a target, the explicit mirror_writes opt-in is on
550 * (writes are off by default — replaying them needs an isolated namespace), and
551 * the OP_WRITE bit is in the allowlist and not in the exclude mask. Every entry
552 * point re-checks this so config can disable the feature mid-connection.
553 */
554static int
555wmir_gate(ngx_stream_brix_srv_conf_t *conf)
556{
557 if (!conf->mirror.enabled || conf->mirror.targets == NULL) { return 0; }
558 if (!conf->mirror.mirror_writes) { return 0; }
559 if ((conf->mirror.opcode_mask & BRIX_MIRROR_OP_WRITE) == 0) { return 0; }
560 if ((conf->mirror.opcode_exclude_mask & BRIX_MIRROR_OP_WRITE) != 0) {
561 return 0;
562 }
563 return 1;
564}

567/*
568 * Hook: a write-open succeeded on the primary — begin accumulating this file.
569 *
570 * WHAT: snapshot the client's open request (header + path/cgi payload) into the
571 * per-connection slot keyed by the primary's file handle index, so kXR_close can
572 * later replay it to the shadow. Called only for write opens; read opens are
573 * ignored. HOW: lazily allocate the per-connection accumulator from the client
574 * connection pool on first use; reset any prior open occupying this slot (handle
575 * indices are reused across opens). The payload is copied (own heap buffer)
576 * because ctx->recv.payload is transient and reused by later requests; an oversize
577 * open payload or OOM marks the slot aborted so it is never replayed.
578 */
579void
580brix_stream_wmirror_on_open(brix_ctx_t *ctx, ngx_connection_t *c,
581 ngx_stream_brix_srv_conf_t *conf, int client_idx, int is_write)
582{
583 brix_wmirror_conn_t *wm;
584 brix_wmirror_file_t *f;
585
586 if (!is_write || !wmir_gate(conf)) { return; }
587 if (client_idx < 0 || client_idx >= BRIX_MAX_FILES) { return; }
588
589 wm = ctx->wmirror;
590 if (wm == NULL) {
591 wm = ngx_pcalloc(c->pool, sizeof(*wm));
592 if (wm == NULL) { return; }
593 ctx->wmirror = wm;
594 }
595
596 f = &wm->files[client_idx];
597 wmir_file_reset(wm, f); /* drop any prior open on this slot */
598
599 ngx_memcpy(f->open_hdr, ctx->recv.hdr_buf, 24);
600 if (ctx->recv.cur_dlen > 0 && ctx->recv.payload != NULL
601 && ctx->recv.cur_dlen <= BRIX_WMIRROR_FILE_CAP)
602 {
603 f->open_payload = ngx_alloc(ctx->recv.cur_dlen, ngx_cycle->log);
604 if (f->open_payload == NULL) { f->aborted = 1; }
605 else {
606 ngx_memcpy(f->open_payload, ctx->recv.payload, ctx->recv.cur_dlen);
607 f->open_dlen = ctx->recv.cur_dlen;
608 }
609 }
610 f->active = 1;
611 f->next_off = 0;
612}

614/*
615 * Hook: observe each write/pgwrite/close on an accumulating file.
616 *
617 * WHAT: drives the per-file accumulator forward as the client streams writes,
618 * and fires the detached replay on a clean close. The slot is located by the
619 * file-handle index carried in the request header (byte 4, the first byte of the
620 * 4-byte fhandle the open hook keyed on). This is best-effort: any condition we
621 * can't faithfully replay just marks the slot aborted and is silently dropped —
622 * the client is never blocked or affected.
623 *
624 * Abort triggers (file will NOT be mirrored): kXR_pgwrite (CRC-interleaved, not
625 * plain bytes), a non-sequential offset, exceeding the per-file or per-connection
626 * cap, an OOM on grow, or the primary itself failing the write.
627 */
628void
629brix_stream_wmirror_observe(brix_ctx_t *ctx, ngx_connection_t *c,
630 ngx_stream_brix_srv_conf_t *conf, ngx_int_t primary_rc)
631{
632 brix_wmirror_conn_t *wm;
633 brix_wmirror_file_t *f;
634 int idx;
635
636 (void) c;
637 wm = ctx->wmirror;
638 if (wm == NULL || !wmir_gate(conf)) { return; }
639
640 idx = (int) (unsigned char) ctx->recv.hdr_buf[4]; /* fhandle byte 0 */
641 if (idx < 0 || idx >= BRIX_MAX_FILES) { return; }
642 f = &wm->files[idx];
643 if (!f->active) { return; }
644
645 switch (ctx->recv.cur_reqid) {
646
647 case kXR_pgwrite:
648 f->aborted = 1; /* CRC-interleaved payload — not a plain write */
649 return;
650
651 case kXR_write: {
652 uint64_t off_be;
653 off_t off;
654 size_t len = ctx->recv.cur_dlen;
655 u_char *ndata;
656
657 if (f->aborted) { return; }
658 if (primary_rc == NGX_ERROR) { f->aborted = 1; return; }
659
660 /* Write offset is an 8-byte big-endian field at header byte 8. Only a
661 * strictly contiguous stream (off == expected next offset) can collapse
662 * into the single offset-0 replay write; a gap/seek aborts the file. */
663 ngx_memcpy(&off_be, ctx->recv.hdr_buf + 8, 8);
664 off = (off_t) be64toh(off_be);
665 if (off != f->next_off) { f->aborted = 1; return; } /* non-sequential */
666 if (len == 0) { return; }
667 /* Bounded buffering: enforce both the per-file and the connection-wide
668 * caps before growing, so a large or many-file upload can't balloon
669 * memory. Hitting a cap is counted (dropped metric), not an error. */
670 if (f->data_len + len > BRIX_WMIRROR_FILE_CAP
671 || wm->total_buffered + len > BRIX_WMIRROR_CONN_CAP)
672 {
673 f->aborted = 1;
674 BRIX_WMIR_METRIC_INC(mirror_stream_dropped_total);
675 return;
676 }
677 /* Grow-by-copy append: allocate old+new, copy the existing prefix, free
678 * the old buffer, then append the new bytes. (Not ngx_realloc so the
679 * old contents are explicitly preserved across the move.) */
680 ndata = ngx_alloc(f->data_len + len, ngx_cycle->log);
681 if (ndata == NULL) { f->aborted = 1; return; }
682 if (f->data_len) {
683 ngx_memcpy(ndata, f->data, f->data_len);
684 ngx_free(f->data);
685 }
686 ngx_memcpy(ndata + f->data_len, ctx->recv.payload, len);
687 f->data = ndata;
688 f->data_len += len;
689 f->next_off += (off_t) len; /* advance expected contiguous offset */
690 wm->total_buffered += len;
691 return;
692 }
693
694 case kXR_close:
695 f->active = 0;
696 /* Replay only a clean, non-empty, non-aborted file whose primary close
697 * also succeeded. wmir_launch steals f->data, so the unconditional
698 * wmir_file_reset that follows safely clears the (now-nulled) slot. */
699 if (!f->aborted && f->data_len > 0 && primary_rc != NGX_ERROR) {
700 wmir_launch(conf, f); /* transfers f->data ownership */
701 }
702 wmir_file_reset(wm, f);
703 return;
704
705 default:
706 return;
707 }
708}

710/*
711 * Hook: client connection is going away — free any still-open accumulators.
712 *
713 * WHY this matters: f->data / f->open_payload are heap buffers (ngx_alloc), NOT
714 * pool memory, so they would leak if the connection dies mid-upload (no close
715 * ever observed). Resetting every slot frees them. The accumulator struct (wm)
716 * itself lives in the connection pool and is reclaimed automatically, so we only
717 * drop the pointer. Any replay already launched is fully detached and unaffected
718 * (it owns its own copy of the data).
719 */
720void
721brix_stream_wmirror_cleanup(brix_ctx_t *ctx)
722{
723 brix_wmirror_conn_t *wm = ctx->wmirror;
724 int i;
725
726 if (wm == NULL) { return; }
727 for (i = 0; i < BRIX_MAX_FILES; i++) {
728 wmir_file_reset(wm, &wm->files[i]);
729 }
730 ctx->wmirror = NULL; /* wm itself is connection-pool memory */
731}
