1/*
2 * stream_wmirror.c — Phase 24 W3: XRootD stream DATA-write mirroring —
3 * per-connection accumulation and the public on_open / observe / cleanup hooks.
4 *
5 * See stream_wmirror.h. This file owns the first of the mirror's two halves:
6 *
7 * Per-connection accumulation. A write-open's bytes are gathered into a
8 * bounded per-file buffer hanging off the client connection context
9 * (brix_ctx_t.wmirror). Only sequential kXR_write data is captured; a
10 * kXR_pgwrite (CRC-interleaved payload), a non-sequential offset, or a buffer
11 * that exceeds the per-file / per-connection cap aborts that file's mirror
12 * (counted) — never blocking the client. On kXR_close a complete sequential
13 * file is handed to wmir_launch, which starts a self-contained detached shadow
14 * replay (open(create) -> write -> close).
15 *
16 * The two other halves of the mirror live in siblings after the phase-79
17 * file-size split (see stream_wmirror_internal.h):
18 * stream_wmirror_replay.c — detached-replay lifecycle (socket, event handlers,
19 * flush, teardown, wmir_launch)
20 * stream_wmirror_state.c — replay protocol state machine (frame build/send +
21 * per-phase step decisions + dispatch)
22 *
23 * The shadow MUST be an isolated namespace — replaying writes onto the primary's
24 * backing store would corrupt it (see brix_mirror_writes).
25 */

43/*
44 * Free a per-file accumulator slot and return it to the empty state. WHAT:
45 * releases the two heap buffers (accumulated data + open payload) and zeroes the
46 * struct. WHY the guarded subtract: the connection-wide byte counter must be
47 * decremented by exactly this file's contribution; the `<=` guard defends
48 * against any accounting drift so total_buffered can never underflow. Safe to
49 * call on an already-empty slot (NULL buffers, data_len 0) and after launch has
50 * stolen f->data (set to NULL).
51 */
52static void
53wmir_file_reset(brix_wmirror_conn_t *wm, brix_wmirror_file_t *f)
54{
55 if (f->data != NULL) { ngx_free(f->data); }
56 if (f->open_payload != NULL) { ngx_free(f->open_payload); }
57 if (wm != NULL && f->data_len <= wm->total_buffered) {
58 wm->total_buffered -= f->data_len;
59 }
60 ngx_memzero(f, sizeof(*f));
61}

63/*
64 * Is data-write mirroring active for this server? Returns 1 only if ALL gates
65 * pass: mirroring enabled with a target, the explicit mirror_writes opt-in is on
66 * (writes are off by default — replaying them needs an isolated namespace), and
67 * the OP_WRITE bit is in the allowlist and not in the exclude mask. Every entry
68 * point re-checks this so config can disable the feature mid-connection.
69 */
70static int
71wmir_gate(ngx_stream_brix_srv_conf_t *conf)
72{
73 if (!conf->mirror.enabled || conf->mirror.targets == NULL) { return 0; }
74 if (!conf->mirror.mirror_writes) { return 0; }
75 if ((conf->mirror.opcode_mask & BRIX_MIRROR_OP_WRITE) == 0) { return 0; }
76 if ((conf->mirror.opcode_exclude_mask & BRIX_MIRROR_OP_WRITE) != 0) {
77 return 0;
78 }
79 return 1;
80}

83/*
84 * Hook: a write-open succeeded on the primary — begin accumulating this file.
85 *
86 * WHAT: snapshot the client's open request (header + path/cgi payload) into the
87 * per-connection slot keyed by the primary's file handle index, so kXR_close can
88 * later replay it to the shadow. Called only for write opens; read opens are
89 * ignored. HOW: lazily allocate the per-connection accumulator from the client
90 * connection pool on first use; reset any prior open occupying this slot (handle
91 * indices are reused across opens). The payload is copied (own heap buffer)
92 * because ctx->recv.payload is transient and reused by later requests; an oversize
93 * open payload or OOM marks the slot aborted so it is never replayed.
94 */
95void
96brix_stream_wmirror_on_open(brix_ctx_t *ctx, ngx_connection_t *c,
97 ngx_stream_brix_srv_conf_t *conf, int client_idx, int is_write)
98{
99 brix_wmirror_conn_t *wm;
100 brix_wmirror_file_t *f;
101
102 if (!is_write || !wmir_gate(conf)) { return; }
103 if (client_idx < 0 || client_idx >= BRIX_MAX_FILES) { return; }
104
105 wm = ctx->wmirror;
106 if (wm == NULL) {
107 wm = ngx_pcalloc(c->pool, sizeof(*wm));
108 if (wm == NULL) { return; }
109 ctx->wmirror = wm;
110 }
111
112 f = &wm->files[client_idx];
113 wmir_file_reset(wm, f); /* drop any prior open on this slot */
114
115 ngx_memcpy(f->open_hdr, ctx->recv.hdr_buf, 24);
116 if (ctx->recv.cur_dlen > 0 && ctx->recv.payload != NULL
117 && ctx->recv.cur_dlen <= BRIX_WMIRROR_FILE_CAP)
118 {
119 f->open_payload = ngx_alloc(ctx->recv.cur_dlen, ngx_cycle->log);
120 if (f->open_payload == NULL) { f->aborted = 1; }
121 else {
122 ngx_memcpy(f->open_payload, ctx->recv.payload, ctx->recv.cur_dlen);
123 f->open_dlen = ctx->recv.cur_dlen;
124 }
125 }
126 f->active = 1;
127 f->next_off = 0;
128}

130/*
131 * Hook: observe each write/pgwrite/close on an accumulating file.
132 *
133 * WHAT: drives the per-file accumulator forward as the client streams writes,
134 * and fires the detached replay on a clean close. The slot is located by the
135 * file-handle index carried in the request header (byte 4, the first byte of the
136 * 4-byte fhandle the open hook keyed on). This is best-effort: any condition we
137 * can't faithfully replay just marks the slot aborted and is silently dropped —
138 * the client is never blocked or affected.
139 *
140 * Abort triggers (file will NOT be mirrored): kXR_pgwrite (CRC-interleaved, not
141 * plain bytes), a non-sequential offset, exceeding the per-file or per-connection
142 * cap, an OOM on grow, or the primary itself failing the write.
143 */
144/*
145 * Append one sequential kXR_write's payload to a file accumulator.
146 *
147 * WHAT: validate contiguity and the caps, then grow-by-copy the captured bytes.
148 * WHY split out: keeps the observe dispatcher a thin switch while this holds the
149 * whole plain-write accumulation policy in one focused place. HOW: the write
150 * offset is an 8-byte big-endian field at header byte 8 — only a strictly
151 * contiguous stream (off == expected next offset) can collapse into the single
152 * offset-0 replay write, so a gap/seek aborts. Enforce the per-file and
153 * connection-wide caps before growing (a cap hit is counted, not an error), then
154 * allocate old+new, copy the existing prefix, free the old buffer, and append the
155 * new bytes (not ngx_realloc, so old contents are explicitly preserved). Any
156 * failure marks the slot aborted; primary_rc==NGX_ERROR aborts up front.
157 */
158static void
159wmir_accumulate_write(brix_wmirror_conn_t *wm, brix_wmirror_file_t *f,
160 brix_ctx_t *ctx, ngx_int_t primary_rc)
161{
162 uint64_t off_be;
163 off_t off;
164 size_t len = ctx->recv.cur_dlen;
165 u_char *ndata;
166
167 if (f->aborted) { return; }
168 if (primary_rc == NGX_ERROR) { f->aborted = 1; return; }
169
170 ngx_memcpy(&off_be, ctx->recv.hdr_buf + 8, 8);
171 off = (off_t) be64toh(off_be);
172 if (off != f->next_off) { f->aborted = 1; return; } /* non-sequential */
173 if (len == 0) { return; }
174
175 if (f->data_len + len > BRIX_WMIRROR_FILE_CAP
176 || wm->total_buffered + len > BRIX_WMIRROR_CONN_CAP)
177 {
178 f->aborted = 1;
179 BRIX_WMIR_METRIC_INC(mirror_stream_dropped_total);
180 return;
181 }
182
183 ndata = ngx_alloc(f->data_len + len, ngx_cycle->log);
184 if (ndata == NULL) { f->aborted = 1; return; }
185 if (f->data_len) {
186 ngx_memcpy(ndata, f->data, f->data_len);
187 ngx_free(f->data);
188 }
189 ngx_memcpy(ndata + f->data_len, ctx->recv.payload, len);
190 f->data = ndata;
191 f->data_len += len;
192 f->next_off += (off_t) len; /* advance expected contiguous offset */
193 wm->total_buffered += len;
194}

196/*
197 * kXR_close handler: fire the detached replay for a clean file, then reset.
198 * WHAT: replay only a non-empty, non-aborted file whose primary close also
199 * succeeded. HOW: wmir_launch steals f->data, so the unconditional
200 * wmir_file_reset that follows safely clears the (now-nulled) slot.
201 */
202static void
203wmir_observe_close(ngx_stream_brix_srv_conf_t *conf, brix_wmirror_conn_t *wm,
204 brix_wmirror_file_t *f, ngx_int_t primary_rc)
205{
206 f->active = 0;
207 if (!f->aborted && f->data_len > 0 && primary_rc != NGX_ERROR) {
208 wmir_launch(conf, f); /* transfers f->data ownership */
209 }
210 wmir_file_reset(wm, f);
211}

213void
214brix_stream_wmirror_observe(brix_ctx_t *ctx, ngx_connection_t *c,
215 ngx_stream_brix_srv_conf_t *conf, ngx_int_t primary_rc)
216{
217 brix_wmirror_conn_t *wm;
218 brix_wmirror_file_t *f;
219 int idx;
220
221 (void) c;
222 wm = ctx->wmirror;
223 if (wm == NULL || !wmir_gate(conf)) { return; }
224
225 idx = (int) (unsigned char) ctx->recv.hdr_buf[4]; /* fhandle byte 0 */
226 if (idx < 0 || idx >= BRIX_MAX_FILES) { return; }
227 f = &wm->files[idx];
228 if (!f->active) { return; }
229
230 switch (ctx->recv.cur_reqid) {
231
232 case kXR_pgwrite:
233 f->aborted = 1; /* CRC-interleaved payload — not a plain write */
234 return;
235
236 case kXR_write:
237 wmir_accumulate_write(wm, f, ctx, primary_rc);
238 return;
239
240 case kXR_close:
241 wmir_observe_close(conf, wm, f, primary_rc);
242 return;
243
244 default:
245 return;
246 }
247}

249/*
250 * Hook: client connection is going away — free any still-open accumulators.
251 *
252 * WHY this matters: f->data / f->open_payload are heap buffers (ngx_alloc), NOT
253 * pool memory, so they would leak if the connection dies mid-upload (no close
254 * ever observed). Resetting every slot frees them. The accumulator struct (wm)
255 * itself lives in the connection pool and is reclaimed automatically, so we only
256 * drop the pointer. Any replay already launched is fully detached and unaffected
257 * (it owns its own copy of the data).
258 */
259void
260brix_stream_wmirror_cleanup(brix_ctx_t *ctx)
261{
262 brix_wmirror_conn_t *wm = ctx->wmirror;
263 int i;
264
265 if (wm == NULL) { return; }
266 for (i = 0; i < BRIX_MAX_FILES; i++) {
267 wmir_file_reset(wm, &wm->files[i]);
268 }
269 ctx->wmirror = NULL; /* wm itself is connection-pool memory */
270}
