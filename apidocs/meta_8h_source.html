4/*
5 * meta.h — on-disk cache-validity metadata sidecar interface.
6 *
7 * WHAT: Defines brix_cache_meta_t (origin mtime, size, and a bounded etag)
8 * and the read/write/derive API implemented in meta.c, plus the
9 * BRIX_CACHE_META_ETAG_MAX bound on the inline etag field.
10 *
11 * WHY: Each cached file gets a companion sidecar so the cache can prove a
12 * local copy still matches the origin before serving it. Centralising the
13 * record layout here keeps the writer (meta.c), the validator (open.c),
14 * eviction (evict_policy.c), and the slice cache (slice.h) byte-compatible.
15 *
16 * HOW: The record is a fixed-size POD struct read/written verbatim, so its
17 * layout is the on-disk format — fields must not be reordered or resized
18 * without invalidating existing sidecars. brix_cache_meta_path() (in
19 * paths.c) maps a cache file path to its sidecar path; the remaining
20 * helpers operate on that path.
21 */
22

31/*
32 * §9 .cinfo upgrade: the record grows a versioned tail (access stats + origin
33 * checksum) AFTER the legacy {mtime,size,etag_len,etag} base. Readers accept a
34 * legacy-sized sidecar (the tail reads short and stays zeroed → version 0); a
35 * grown record is detected by version >= 1. The base layout is unchanged so old
36 * code paths and on-disk validity checks are byte-compatible.
37 */
38#define BRIX_CACHE_META_VERSION 1

40typedef struct {
41 /* ---- legacy base (do not reorder/resize) ---- */
42 uint64_t mtime;
43 uint64_t size;
