4/*
5 * cstore.h - the cache's one storage adapter (phase-64 section 6.1).
6 *
7 * WHAT: The ONLY code that touches a cache-store driver. It turns a generic SD
8 * instance (the cache_store tier - any driver: posix, pblock, a roots://
9 * cache server, s3, ...) into the small set of operations the read cache
10 * needs: open a fill sink, write/commit/abort it, open a served object,
11 * pread it, evict it, load/store its cinfo record, scan the store, and
12 * report free space. Every cache policy module (admit, evict, reap, verify)
13 * drives the store through THIS adapter and never through a driver directly
14 * (P3, review gate G5).
15 *
16 * WHY: Before phase-64 the cache hard-coded a local POSIX tree (.cinfo sidecars,
17 * raw unlink, statvfs) and a per-scheme origin fetch. Routing every store
18 * touch through one adapter makes the cache driver-agnostic: the same
19 * eviction/fill/serve logic runs whether the store is a local SSD or a
20 * remote cache server, with the cinfo record living LOCAL (sidecar files) or
21 * as a store xattr (SP2) - a transport choice, not a format change.
22 *
23 * HOW: the fill and serve calls forward to store->driver staged_x/open/pread/
24 * close; evict to unlink; scan to opendir/readdir/stat; freespace to statvfs
25 * (LOCAL)
26 * or the driver (SP2). cinfo_load/store keep the record: LOCAL writes the
27 * byte-identical "<obj>.cinfo" sidecar through the existing cinfo.c codec and
28 * a per-worker write-through L1 (cinfo_l1.h); XATTR/SIDECAR land in SP2. SP1
29 * implements LOCAL mode fully (a posix cache store is byte-identical to the
30 * pre-phase-64 tree). See docs/refactor/phase-64-fully-tiered-composable-
31 * storage.md (section 6, Appendix C/I/K).
32 */

42/* cinfo/meta encoding mode (section 6.3). AUTO resolves from the store's caps at
43 * init: a store with a known local POSIX dir -> LOCAL (sidecar files next to the
44 * object, byte-identical to pre-phase-64); any store advertising CAP_XATTR ->
45 * XATTR (the cinfo blob lives in user.xrd.cinfo on the store, SP2); otherwise ->
46 * SIDECAR (a co-located <key>.xrdcinfo object, SP2). */
47#define BRIX_CMETA_AUTO 0
48#define BRIX_CMETA_LOCAL 1
49#define BRIX_CMETA_XATTR 2
50#define BRIX_CMETA_SIDECAR 3
