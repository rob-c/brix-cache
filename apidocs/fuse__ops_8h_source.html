1/*
2 * fuse_ops.h — pooled metadata-op runner + op thunks shared by the FUSE drivers.
3 *
4 * WHAT: A small library of "run one metadata op on a connection from the pool"
5 * primitives. brix_fuse_run() checks a conn out of the pool, invokes an
6 * op thunk, checks it back in (marking the slot healthy/unhealthy from the
7 * fault class), and maps the result to a FUSE-style negative errno. The
8 * op thunks (brix_fuse_op_*) each wrap one brix_* metadata call with a
9 * typed ctx struct so they share the one runner.
10 * WHY: Both FUSE drivers — xrootdfs.c (resilient/async) and xrootdfs_legacy.c
11 * (simple/synchronous) — performed the identical checkout → op → checkin →
12 * errno-map dance for every metadata op, differing only in whether they
13 * retried transient faults. Folding that into one parameterised runner
14 * (max_retries==0 == the legacy single-shot path, >0 == the async retry+
15 * backoff path) removes the duplication without changing either driver's
16 * behaviour, and gives both the same canonical op thunks.
17 * HOW: No fuse3 dependency — this operates purely on brix_conn/brix_pool, so it
18 * lives in the general client lib. A driver builds a typed ctx, then calls
19 * brix_fuse_run(pool, retries, brix_fuse_op_X, &ctx, &st); on non-zero it
20 * gets a ready-to-return negative errno.
21 */

42/*
43 * Run `op` on a connection borrowed from `pool`.
44 * Retry budget (mirrors the data-plane mfile path so metadata is as resilient
45 * as read/write on a lossy link):
46 * max_stall_ms > 0 → DEADLINE-bounded: retry TRANSIENT faults (socket/protocol)
47 * with exponential backoff + jitter, reconnecting the dropped
48 * slot, until the op succeeds or the patience window elapses
49 * (NOT a fixed count — a flapping link is ridden out as long
50 * as progress is possible). max_retries is ignored.
51 * max_stall_ms == 0 → legacy COUNT bound: max_retries == 0 is a single attempt
52 * (the simple driver), max_retries > 0 retries that many times.
53 *
54 * benign_errno (idempotency normalization for re-issued MUTATIONS): when > 0 and
55 * a RETRY (not the first attempt) fails with this POSIX errno, treat it as
56 * success — the first attempt's change was already applied and its reply was lost
57 * to the sever (EEXIST for mkdir/symlink/link, ENOENT for rm/rmdir/mv). Pass 0
58 * for read-only / naturally-idempotent ops (chmod/truncate/setattr).
59 *
60 * Fatal faults (NotFound, NotAuthorized, …) return immediately. Returns 0 on
61 * success or a negative errno ready to hand back to FUSE.
62 */
63int brix_fuse_run(brix_pool *pool, int max_retries, int max_stall_ms,
64 int benign_errno, brix_fuse_op_fn op, void *ctx,
65 brix_status *st);
