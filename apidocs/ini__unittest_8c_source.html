34/* collector: record every (section,key,value) the parser emits */struct kv {
35 char section[64];
36 char key[64];
37 char value[256];
38};
39
40struct collector {
41 struct kv kv[64];
42 int n;
43};
44
45static int
46collect_cb(void *u, const char *section, const char *key, const char *value)
47{
48 struct collector *c = u;
49 if (c->n >= 64) {
50 return -1;
51 }
52 snprintf(c->kv[c->n].section, sizeof(c->kv[c->n].section), "%s", section);
53 snprintf(c->kv[c->n].key, sizeof(c->kv[c->n].key), "%s", key);
54 snprintf(c->kv[c->n].value, sizeof(c->kv[c->n].value), "%s", value);
55 c->n++;
56 return 0;
57}

59static const struct kv *
60find_kv(const struct collector *c, const char *section, const char *key)
61{
62 int i;
63 for (i = 0; i < c->n; i++) {
64 if (strcmp(c->kv[i].section, section) == 0
65 && strcmp(c->kv[i].key, key) == 0)
66 {
67 return &c->kv[i];
68 }
69 }
70 return NULL;
71}

84/* test: the real upstream scitokens.cfg fixture */static void
85test_stock_fixture(void)
86{
87 struct collector c = {0};
88 char err[256] = "";
89 const char *path = "tests/fixtures/scitokens.cfg";
90
91 /* allow running from repo root or src/token */
92 if (access(path, R_OK) != 0) {
93 path = "../../tests/fixtures/scitokens.cfg";
94 }
95 int rc = brix_ini_parse_file(path, collect_cb, &c, err, sizeof(err));
96 CHECK(rc == 0, "stock fixture parses");
97 if (rc != 0) { fprintf(stderr, " err: %s\n", err); return; }
98
99 const struct kv *k;
100 k = find_kv(&c, "Issuer OSG-Connect", "issuer");
101 CHECK(k && strcmp(k->value, "https://scitokens.org/osg-connect") == 0,
102 "OSG-Connect issuer URL");
103 k = find_kv(&c, "Issuer OSG-Connect", "base_path");
104 CHECK(k && strcmp(k->value, "/stash") == 0, "OSG-Connect base_path");
105 k = find_kv(&c, "Issuer OSG-Connect", "map_subject");
106 CHECK(k && strcmp(k->value, "True") == 0, "OSG-Connect map_subject");
107 k = find_kv(&c, "Issuer CMS", "issuer");
108 CHECK(k && strcmp(k->value, "https://scitokens.org/cms") == 0,
109 "CMS issuer URL");
110 k = find_kv(&c, "Issuer CMS", "map_subject");
111 CHECK(k && strcmp(k->value, "False") == 0, "CMS map_subject");
112}

115test_trim_and_comments(void)
116{

131 CHECK(rc == 0, "trim/comment fixture parses");

134 CHECK(k != NULL, "global audience present");
135 CHECK(k && strcmp(k->value, "https://a , https://b") == 0,
136 "value trimmed, inline comment stripped");
137 k = find_kv(&c, "Issuer X", "issuer");
138 CHECK(k && strcmp(k->value, "https://x") == 0, "no-space key=value");
139 unlink(path);
140}
