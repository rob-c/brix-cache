14{
15 brix_sd_obj_t obj;
16
17 /* Positional write-full through the Storage Driver seam (syscall stays in
18 * the backend); the caller passes the destination offset. */
19 brix_sd_posix_wrap(&obj, fd);
20 while (len > 0) {
21 ssize_t nwritten;
22
23 nwritten = obj.driver->pwrite(&obj, buf, len, offset);
24 if (nwritten < 0) {
25 if (errno == EINTR) {
26 continue;
27 }
28 return NGX_ERROR;
29 }
30
31 if (nwritten == 0) {
32 errno = EIO;
33 return NGX_ERROR;
34 }
35
36 buf += (size_t) nwritten;
37 len -= (size_t) nwritten;
38 offset += nwritten;
39 }
40
41 return NGX_OK;
42}

44/**
45 * WHAT: Copy a spooled temp file body to the final destination using zero-copy
46 * copy_file_range() on Linux, falling back to pread+write on other platforms.
47 *
48 * Transfers bytes from buf->file (the spooled PUT temp file) to dst_fd (the final
49 * WebDAV resource) in chunks. On Linux with SYS_copy_file_range support this uses
50 * kernel-level zero-copy transfer; otherwise falls back to userspace pread+write loop.
51 * Returns NGX_OK on complete copy or NGX_ERROR on any failure. The scratch buffer is
52 * allocated lazily if needed for the fallback path.
53 *
54 * WHY: Spooled PUT writes (where the request body is stored in a temp file first) need
55 * to be moved to their final destination after authentication and validation succeed.
56 * copy_file_range() on Linux provides zero-copy kernel-level transfer without userspace
57 * buffering — this is significantly faster for large files (>1MB) and avoids allocating
58 * temporary buffers. The fallback path uses pread(2) + webdav_write_full(2) which works
59 * on all platforms but has higher CPU overhead due to copying through user-space buffers.
60 * This function handles multiple error conditions: EXDEV (cross-device copy not supported),
61 * EOPNOTSUPP/ENOSYS (syscall unavailable), and retries only EINTR.
62 *
63 * HOW: Four-phase algorithm. Phase 1: validate source buf has valid file pointer and fd,
64 * compute remaining bytes from buf->file_last - buf->file_pos. Phase 2 (Linux only): loop
65 * using copy_file_range() syscall in WEBDAV_PUT_COPY_CHUNK-sized chunks — successful copies
66 * decrement remaining, EINTR retries, non-recoverable errors return immediately, EXDEV/
67 * ENOSYS/EOPNOTSUPP/EINVAL/EPERM break out to fallback. Phase 3 (fallback): allocate scratch
68 * buffer via ngx_palloc(r->pool) if not already done, loop using pread + webdav_write_full.
69 * Phase 4: return NGX_OK if remaining == 0 after either phase. Logging uses
70 * brix_log_safe_path() for safe path formatting in error messages.
71 */
