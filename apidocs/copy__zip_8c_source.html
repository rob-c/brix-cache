19int
20unzip_sink_write(void *sc, const uint8_t *d, size_t l)
21{
22 unzip_sink_ctx *s = sc;
23 size_t off = 0;
24 while (off < l) {
25 ssize_t n = write(s->fd, d + off, l - off);
26 if (n < 0) {
27 if (errno == EINTR) { continue; }
28 return -1;
29 }
30 off += (size_t) n;
31 }
32 return 0;

36/* ---- Open the local destination sink for a ZIP-member extraction ----
37 *
38 * WHAT: Resolves `du` to a writable fd. For a stdio destination returns
39 * STDOUT_FILENO with *use_tmp cleared; for a file destination opens a temp file
40 * (path written to `tmp`) and sets *use_tmp so the caller renames on success.
41 * Returns the fd, or -1 with `st` populated (destination exists without -f, or
42 * temp open failed).
43 *
44 * WHY: The destination-setup branch (stdout vs. force-check vs. temp-open) is a
45 * self-contained decision that inflated copy_unzip past the complexity cap;
46 * isolating it keeps the orchestrator a flat sequence and this policy testable.
47 *
48 * HOW:
49 * 1. Clear *use_tmp up front (stdout and error paths leave no temp to clean).
50 * 2. For a stdio scheme, return STDOUT_FILENO.
51 * 3. Otherwise, unless -f (force) is set, refuse an existing destination.
52 * 4. Open the download temp; on failure return -1.
53 * 5. Mark *use_tmp and return the temp fd.
54 */
55static int
56copy_unzip_open_dest(const brix_url *du, const brix_copy_opts *o,
57 char *tmp, size_t tmp_sz, int *use_tmp, brix_status *st)
58{
59 int outfd;
60
61 *use_tmp = 0;
62 if (du->scheme == XRDC_SCHEME_STDIO) {
63 return STDOUT_FILENO;
64 }
65 if (!o->force && access(du->path, F_OK) == 0) {
66 brix_status_set(st, XRDC_EUSAGE, 0,
67 "destination exists (use -f): %s", du->path);
68 return -1;
69 }
70 outfd = open_download_temp(du->path, tmp, tmp_sz, st);
71 if (outfd < 0) {
72 return -1;
73 }
74 *use_tmp = 1;
75 return outfd;
76}

79/* ---- Locate and inflate one member of the remote ZIP archive ----
80 *
81 * WHAT: Opens the ZIP central directory of the archive of `size` bytes, finds
82 * `member`, and inflates it into `outfd` via the streaming sink. Returns 0 on
83 * success, or -1 with `st` populated (bad archive, member absent, or extract
84 * failure).
85 *
86 * WHY: The parse/find/extract block owns three failure branches that pushed
87 * copy_unzip over the complexity cap; keeping it separate confines the
88 * brix_zip_dir lifetime (freed on every exit) to one small function.
89 *
90 * HOW:
91 * 1. Open the directory; on error report "ZIP open failed" and return -1.
92 * 2. Find the member; if absent, free the dir and report "member not found".
93 * 3. Extract the member into `outfd` through unzip_sink_write.
94 * 4. Free the directory unconditionally.
95 * 5. Map a non-OK extract result to -1, else return 0.
96 */
97static int
98copy_unzip_extract_member(zip_remote_ctx *zc, uint64_t size,
99 const char *archive_path, const char *member,
100 int outfd, brix_status *st)
101{
102 brix_zip_dir dir;
103 const brix_zip_entry *e;
104 int zr;
105
106 zr = brix_zip_open(zip_remote_pread, zc, size, &dir);
107 if (zr != XRDC_ZIP_OK) {
108 brix_status_set(st, XRDC_EUSAGE, 0,
109 "ZIP open failed (%d) for %s", zr, archive_path);
110 return -1;
111 }
112 e = brix_zip_find(&dir, member);
113 if (e == NULL) {
114 brix_status_set(st, XRDC_EUSAGE, 0, "ZIP member not found: %s", member);
115 brix_zip_dir_free(&dir);
116 return -1;
117 }
118 {
119 unzip_sink_ctx sink = { outfd };
120 zr = brix_zip_member_extract(zip_remote_pread, zc, e,
121 unzip_sink_write, &sink);
122 }
123 brix_zip_dir_free(&dir);
124 if (zr != XRDC_ZIP_OK) {
125 brix_status_set(st, XRDC_EUSAGE, 0,
126 "ZIP member extract failed (%d): %s", zr, member);
127 return -1;
128 }
129 return 0;
130}

133/* ---- Commit or discard the temp destination after extraction ----
134 *
135 * WHAT: When a temp file was used, renames it over `du->path` on success or
136 * unlinks it on failure. Returns the final result code (`rc` unchanged, or -1 if
137 * the rename failed with `st` populated). A no-op for the stdout path.
138 *
139 * WHY: The commit step is the mirror of copy_unzip_open_dest and keeps the
140 * atomic temp-then-rename policy in one place, so the orchestrator's cleanup
141 * stays a single call.
142 *
143 * HOW:
144 * 1. If no temp was used, return `rc` untouched.
145 * 2. On a failed extraction, unlink the temp and return `rc`.
146 * 3. On success, rename the temp over the destination; if that fails, unlink
147 * and return -1, otherwise return `rc`.
148 */
149static int
150copy_unzip_finalize(int rc, int use_tmp, const char *tmp,
151 const brix_url *du, brix_status *st)
152{
153 if (!use_tmp) {
154 return rc;
155 }
156 if (rc != 0) {
157 unlink(tmp);
158 return rc;
159 }
160 if (rename(tmp, du->path) != 0) {
161 brix_status_set(st, XRDC_EUSAGE, errno, "rename to %s", du->path);
162 unlink(tmp);
163 return -1;
164 }
165 return rc;
166}

169/* Extract `member` from the remote ZIP archive at `archive_path` into the local
170 * destination du. The server is untouched (serves raw archive bytes); the client
171 * parses the directory and inflates the member locally (zlib-only). */
172int
173copy_unzip(const brix_url *su, const char *archive_path, const char *member,
174 const brix_url *du, const brix_copy_opts *o, const brix_opts *co,
175 brix_status *st)
176{
177 brix_conn c;
178 brix_statinfo si;
179 brix_file f;
180 zip_remote_ctx zc;
181 int outfd, use_tmp = 0, rc;
182 int to_stdout = (du->scheme == XRDC_SCHEME_STDIO);
183 char tmp[XRDC_PATH_MAX];
184
185 if (brix_connect(&c, su, co, st) != 0) {
186 return -1;
187 }
188 if (brix_stat(&c, archive_path, &si, st) != 0) {
189 brix_close(&c);
190 return -1;
191 }
192 if (brix_file_open_read(&c, archive_path, &f, st) != 0) {
193 brix_close(&c);
194 return -1;
195 }
196
197 outfd = copy_unzip_open_dest(du, o, tmp, sizeof(tmp), &use_tmp, st);
198 if (outfd < 0) {
199 brix_file_close(&c, &f, st);
200 brix_close(&c);
201 return -1;
202 }
203
204 zc.c = &c; zc.f = &f; zc.st = st;
205 rc = copy_unzip_extract_member(&zc, (uint64_t) si.size, archive_path,
206 member, outfd, st);
207
208 brix_file_close(&c, &f, st);
209 if (outfd >= 0 && !to_stdout) {
210 close(outfd);
211 }
212 rc = copy_unzip_finalize(rc, use_tmp, tmp, du, st);
213 brix_close(&c);
214 return rc;
215}

218/* If `src` carries the opaque key xrdcl.unzip=<member>, copy `member` into out
219 * (caller buffer) and the archive path (opaque stripped) into arch; return 1.
220 * Otherwise return 0. */
221int
222unzip_member_from_src(const char *src, const brix_url *su,
223 char *member, size_t member_sz, char *arch, size_t arch_sz)
224{
225 const char *q = strstr(src, "xrdcl.unzip=");
226 const char *v, *end;
227 size_t n, an;
228
229 if (q == NULL) {
230 return 0;
231 }
232 v = q + (sizeof("xrdcl.unzip=") - 1);
233 end = v;
234 while (*end != '\0' && *end != '&') { end++; }
235 n = (size_t) (end - v);
236 if (n == 0 || n >= member_sz) {
237 return 0;
238 }
239 memcpy(member, v, n);
240 member[n] = '\0';
241
242 /* archive path = su->path with any trailing "?opaque" removed. */
243 an = strlen(su->path);
244 {
245 const char *qm = strchr(su->path, '?');
246 if (qm != NULL) {
247 an = (size_t) (qm - su->path);
248 }
249 }
250 if (an >= arch_sz) {
251 return 0;
252 }
253 memcpy(arch, su->path, an);
254 arch[an] = '\0';
255 return 1;
256}

300/* Read the existing archive's EOCD + raw central directory for append. Refuses a
301 * ZIP64 archive (append-in-place would need 64-bit CD rewrite). Returns 0 with a
302 * malloc'd *seed_cd (caller frees), or -1 on error. */
303int
304zip_read_seed(brix_zip_pread_fn pr, void *ctx, uint64_t size, uint64_t *base,
305 uint8_t **seed_cd, size_t *seed_len, size_t *seed_n, brix_status *st)
306{
307 uint64_t cd_off, cd_size, n;
308 int z64;
309 uint8_t *buf;
310
311 if (brix_zip_read_eocd(pr, ctx, size, &cd_off, &cd_size, &n, &z64)
312 != XRDC_ZIP_OK)
313 {
314 brix_status_set(st, XRDC_EUSAGE, 0,
315 "--zip-append: destination is not a valid ZIP archive");
316 return -1;
317 }
318 if (z64) {
319 brix_status_set(st, XRDC_EUSAGE, 0,
320 "--zip-append: ZIP64 archives are not supported for append");
321 return -1;
322 }
323 buf = malloc(cd_size ? (size_t) cd_size : 1);
324 if (buf == NULL) {
325 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
326 return -1;

329 free(buf);
330 brix_status_set(st, XRDC_ESOCK, 0, "--zip-append: cannot read central directory");
331 return -1;
332 }
333 *base = cd_off;
334 *seed_cd = buf;
335 *seed_len = (size_t) cd_size;

359int
360copy_zip_store_local(const char *member, int srcfd, const brix_url *du,
361 int append, brix_status *st)
362{
363 int flags = append ? (O_RDWR | O_CREAT) : (O_WRONLY | O_CREAT | O_TRUNC);
364 int dfd = open(du->path, flags, 0644);
365 uint64_t base = 0;
366 uint8_t *seed = NULL;
367 size_t seed_len = 0, seed_n = 0;
368 zipw_local_sink sink;
369 brix_zip_writer *w;
370 int rc;
371
372 if (dfd < 0) {
373 brix_status_set(st, XRDC_EUSAGE, errno, "open %s: %s",
374 du->path, strerror(errno));
375 return -1;
376 }
377 if (append) {
378 struct stat sb;
379 if (fstat(dfd, &sb) == 0 && sb.st_size > 0) {
380 if (zip_read_seed(zipw_local_pread, &dfd, (uint64_t) sb.st_size,
381 &base, &seed, &seed_len, &seed_n, st) != 0) {
382 close(dfd);
383 return -1;
384 }
385 }
386 }
387 sink.fd = dfd;
388 sink.off = base;
389 brix_zip_seed zs = { seed, seed_len, seed_n };
390 w = seed ? brix_zip_writer_new_append(zipw_local_write, &sink, base, &zs)
391 : brix_zip_writer_new(zipw_local_write, &sink);
392 rc = zip_emit_member(w, member, srcfd, st);
393 brix_zip_writer_free(w);
394 free(seed);
395 /* Append always grows the archive (new member + larger CD), and create uses
396 * O_TRUNC, so the written length is authoritative — no tail to trim. */
397 close(dfd);
398 return rc;
399}

435 brix_close(&c);
436 return -1;
437 }
438

441 sink.off = base;
442 sink.st = st;
