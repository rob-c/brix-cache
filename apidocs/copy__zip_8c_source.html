20unzip_sink_write(void *sc, const uint8_t *d, size_t l)
21{

23 size_t off = 0;

36/* Extract `member` from the remote ZIP archive at `archive_path` into the local
37 * destination du. The server is untouched (serves raw archive bytes); the client
38 * parses the directory and inflates the member locally (zlib-only). */
39int
40copy_unzip(const brix_url *su, const char *archive_path, const char *member,
41 const brix_url *du, const brix_copy_opts *o, const brix_opts *co,
42 brix_status *st)
43{
44 brix_conn c;
45 brix_statinfo si;
46 brix_file f;
47 brix_zip_dir dir;
48 zip_remote_ctx zc;
49 const brix_zip_entry *e;
50 int outfd = -1, to_stdout = (du->scheme == XRDC_SCHEME_STDIO);
51 int use_tmp = 0, rc = -1, zr;
52 char tmp[XRDC_PATH_MAX];
53
54 if (brix_connect(&c, su, co, st) != 0) {
55 return -1;
56 }
57 if (brix_stat(&c, archive_path, &si, st) != 0) {
58 brix_close(&c);
59 return -1;
60 }
61 if (brix_file_open_read(&c, archive_path, &f, st) != 0) {
62 brix_close(&c);
63 return -1;
64 }
65
66 if (to_stdout) {
67 outfd = STDOUT_FILENO;
68 } else {
69 if (!o->force && access(du->path, F_OK) == 0) {
70 brix_status_set(st, XRDC_EUSAGE, 0,
71 "destination exists (use -f): %s", du->path);
72 brix_file_close(&c, &f, st); brix_close(&c); return -1;
73 }
74 outfd = open_download_temp(du->path, tmp, sizeof(tmp), st);
75 if (outfd < 0) {
76 brix_file_close(&c, &f, st); brix_close(&c); return -1;
77 }
78 use_tmp = 1;
79 }
80
81 zc.c = &c; zc.f = &f; zc.st = st;
82 zr = brix_zip_open(zip_remote_pread, &zc, (uint64_t) si.size, &dir);
83 if (zr != XRDC_ZIP_OK) {
84 brix_status_set(st, XRDC_EUSAGE, 0,
85 "ZIP open failed (%d) for %s", zr, archive_path);
86 } else {
87 e = brix_zip_find(&dir, member);
88 if (e == NULL) {
89 brix_status_set(st, XRDC_EUSAGE, 0,
90 "ZIP member not found: %s", member);
91 } else {
92 unzip_sink_ctx sink = { outfd };
93 zr = brix_zip_member_extract(zip_remote_pread, &zc, e,
94 unzip_sink_write, &sink);
95 if (zr == XRDC_ZIP_OK) {
96 rc = 0;
97 } else {
98 brix_status_set(st, XRDC_EUSAGE, 0,
99 "ZIP member extract failed (%d): %s", zr, member);
100 }
101 }
102 brix_zip_dir_free(&dir);
103 }
104
105 brix_file_close(&c, &f, st);
106 if (outfd >= 0 && !to_stdout) {
107 close(outfd);
108 }
109 if (use_tmp) {
110 if (rc == 0) {
111 if (rename(tmp, du->path) != 0) {
112 brix_status_set(st, XRDC_EUSAGE, errno, "rename to %s", du->path);
113 unlink(tmp);
114 rc = -1;
115 }
116 } else {
117 unlink(tmp);
118 }
119 }
120 brix_close(&c);
121 return rc;
122}

125/* If `src` carries the opaque key xrdcl.unzip=<member>, copy `member` into out
126 * (caller buffer) and the archive path (opaque stripped) into arch; return 1.
127 * Otherwise return 0. */
128int
129unzip_member_from_src(const char *src, const brix_url *su,
130 char *member, size_t member_sz, char *arch, size_t arch_sz)
131{
132 const char *q = strstr(src, "xrdcl.unzip=");
133 const char *v, *end;
134 size_t n, an;
135
136 if (q == NULL) {
137 return 0;
138 }
139 v = q + (sizeof("xrdcl.unzip=") - 1);
140 end = v;
141 while (*end != '\0' && *end != '&') { end++; }
142 n = (size_t) (end - v);
143 if (n == 0 || n >= member_sz) {
144 return 0;
145 }
146 memcpy(member, v, n);
147 member[n] = '\0';
148
149 /* archive path = su->path with any trailing "?opaque" removed. */
150 an = strlen(su->path);
151 {
152 const char *qm = strchr(su->path, '?');
153 if (qm != NULL) {
154 an = (size_t) (qm - su->path);
155 }
156 }
157 if (an >= arch_sz) {
158 return 0;
159 }
160 memcpy(arch, su->path, an);
161 arch[an] = '\0';
162 return 1;
163}

207/* Read the existing archive's EOCD + raw central directory for append. Refuses a
208 * ZIP64 archive (append-in-place would need 64-bit CD rewrite). Returns 0 with a
209 * malloc'd *seed_cd (caller frees), or -1 on error. */
210int
211zip_read_seed(brix_zip_pread_fn pr, void *ctx, uint64_t size, uint64_t *base,
212 uint8_t **seed_cd, size_t *seed_len, size_t *seed_n, brix_status *st)
213{
214 uint64_t cd_off, cd_size, n;
215 int z64;
216 uint8_t *buf;
217
218 if (brix_zip_read_eocd(pr, ctx, size, &cd_off, &cd_size, &n, &z64)
219 != XRDC_ZIP_OK)
220 {
221 brix_status_set(st, XRDC_EUSAGE, 0,
222 "--zip-append: destination is not a valid ZIP archive");
223 return -1;
224 }
225 if (z64) {
226 brix_status_set(st, XRDC_EUSAGE, 0,
227 "--zip-append: ZIP64 archives are not supported for append");
228 return -1;
229 }
230 buf = malloc(cd_size ? (size_t) cd_size : 1);
231 if (buf == NULL) {
232 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
233 return -1;
234 }
235 if (pr(ctx, cd_off, buf, (size_t) cd_size) != (ssize_t) cd_size) {
236 free(buf);
237 brix_status_set(st, XRDC_ESOCK, 0, "--zip-append: cannot read central directory");
238 return -1;
239 }
240 *base = cd_off;
241 *seed_cd = buf;
242 *seed_len = (size_t) cd_size;
243 *seed_n = (size_t) n;
244 return 0;
245}

266int
267copy_zip_store_local(const char *member, int srcfd, const brix_url *du,
268 int append, brix_status *st)
269{
270 int flags = append ? (O_RDWR | O_CREAT) : (O_WRONLY | O_CREAT | O_TRUNC);
271 int dfd = open(du->path, flags, 0644);
272 uint64_t base = 0;
273 uint8_t *seed = NULL;
274 size_t seed_len = 0, seed_n = 0;
275 zipw_local_sink sink;
276 brix_zip_writer *w;
277 int rc;
278
279 if (dfd < 0) {
280 brix_status_set(st, XRDC_EUSAGE, errno, "open %s: %s",
281 du->path, strerror(errno));
282 return -1;
283 }
284 if (append) {
285 struct stat sb;
286 if (fstat(dfd, &sb) == 0 && sb.st_size > 0) {
287 if (zip_read_seed(zipw_local_pread, &dfd, (uint64_t) sb.st_size,
288 &base, &seed, &seed_len, &seed_n, st) != 0) {
289 close(dfd);
290 return -1;
291 }
292 }
293 }
294 sink.fd = dfd;
295 sink.off = base;
296 w = seed ? brix_zip_writer_new_append(zipw_local_write, &sink, base,
297 seed, seed_len, seed_n)
298 : brix_zip_writer_new(zipw_local_write, &sink);
299 rc = zip_emit_member(w, member, srcfd, st);
300 brix_zip_writer_free(w);
301 free(seed);
302 /* Append always grows the archive (new member + larger CD), and create uses
303 * O_TRUNC, so the written length is authoritative — no tail to trim. */
304 close(dfd);
305 return rc;
306}
