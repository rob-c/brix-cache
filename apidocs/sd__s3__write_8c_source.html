36/* sd_s3_ck_crc32 — render the AWS x-amz-checksum-crc32 value for a body: base64
37 * of the 4 big-endian bytes of CRC-32/IEEE (the algorithm AWS/MinIO/radosgw
38 * validate for that header). out must hold >= 9 bytes (8 chars + NUL). #12. */
39static void
40sd_s3_ck_crc32(const void *buf, size_t len, char out[9])
41{
42 uint32_t crc = brix_crc32_ieee((const uint8_t *) buf, len);
43 uint8_t be[4] = { (uint8_t) (crc >> 24), (uint8_t) (crc >> 16),
44 (uint8_t) (crc >> 8), (uint8_t) crc };
45 /* EVP_EncodeBlock writes exactly 8 base64 chars + NUL for a 4-byte input. */
46 EVP_EncodeBlock((unsigned char *) out, be, 4);
47}

49/* Extract <tag>…</tag> text from an XML body into out[outsz]. 0 / -1. */
50static int
51sd_s3_xml_tag(const char *xml, const char *tag, char *out, size_t outsz)
52{
53 char open_tag[64], close_tag[64];
54 const char *p, *e;
55 size_t n;
56
57 snprintf(open_tag, sizeof(open_tag), "<%s>", tag);
58 snprintf(close_tag, sizeof(close_tag), "</%s>", tag);
59 p = strstr(xml, open_tag);
60 if (p == NULL) { return -1; }
61 p += strlen(open_tag);
62 e = strstr(p, close_tag);
63 if (e == NULL) { return -1; }
64 n = (size_t) (e - p);
65 if (n == 0 || n >= outsz) { return -1; }
66 memcpy(out, p, n);
67 out[n] = '\0';
68 return 0;

90static int
91sd_s3_mpu_upload_part(sd_s3_file *f, int part_num, const void *data, size_t len,
92 char *errbuf, size_t errcap)
93{
94 char wire[SD_S3_KEY_MAX + 256], qs[256], auth[SD_S3_AUTH_HDRS_CAP];
95 char etag[SD_S3_ETAG_LEN];
96 brix_s3_resp_t resp;
97 int pn;
98
99 pn = snprintf(wire, sizeof(wire), "%s?partNumber=%d&uploadId=%s",
100 f->key, part_num, f->upload_id);
101 if (pn < 0 || (size_t) pn >= sizeof(wire)) {
102 sd_s3_set_err(errbuf, errcap, "s3 mpu: part path too long");
103 return -1;
104 }
105 pn = snprintf(qs, sizeof(qs), "partNumber=%d&uploadId=%s",
106 part_num, f->upload_id);
107 if (pn < 0 || (size_t) pn >= sizeof(qs)) {
108 sd_s3_set_err(errbuf, errcap, "s3 mpu: canon_qs too long");
109 return -1;
110 }
111 {
112 char ck[9];
113 int sr;
114 /* #12: per-part origin-enforced integrity — each UploadPart carries its
115 * own signed x-amz-checksum-crc32 (S3 validates parts individually). */
116 if (f->put_checksum) {
117 sd_s3_ck_crc32(data, len, ck);
118 sr = sd_s3_sign_ex(f, "PUT", qs, "x-amz-checksum-crc32", ck,
119 auth, sizeof(auth));
120 } else {
121 sr = sd_s3_sign(f, "PUT", qs, auth, sizeof(auth));
122 }
123 if (sr != 0) {
124 sd_s3_set_err(errbuf, errcap, "s3 UploadPart: sign failed");
125 return -1;
126 }
127 }
128 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "PUT", wire,
129 auth, data, len, f->timeout_ms, &resp,
130 errbuf, errcap) != 0) {
131 return -1;
132 }
133 if (resp.status != 200) {
134 int rc = sd_s3_status_err(resp.status, "UploadPart", f->key, errbuf, errcap);
135 f->transport->resp_free(&resp);
136 return rc;
137 }
138 etag[0] = '\0';
139 f->transport->resp_header(&resp, "ETag", etag, sizeof(etag));
140 f->transport->resp_free(&resp);
141 if (etag[0] == '\0') {
142 sd_s3_set_err(errbuf, errcap, "s3 UploadPart: server returned no ETag");
143 return -1;
144 }
145 if (sd_s3_etag_ensure_cap(f, part_num, errbuf, errcap) != 0) {
146 return -1;
147 }
148 snprintf(f->etags[part_num - 1].val, SD_S3_ETAG_LEN, "%s", etag);
149 f->part_count = part_num;
150 return 0;
151}

165static int
166sd_s3_mpu_create(sd_s3_file *f, char *errbuf, size_t errcap)
167{
168 char wire[SD_S3_KEY_MAX + 16], auth[SD_S3_AUTH_HDRS_CAP];
169 brix_s3_resp_t resp;
170 const void *body;
171 size_t blen = 0;
172 int pn;
173
174 pn = snprintf(wire, sizeof(wire), "%s?uploads", f->key);
175 if (pn < 0 || (size_t) pn >= sizeof(wire)) {
176 sd_s3_set_err(errbuf, errcap, "s3 mpu: key path too long");
177 return -1;
178 }
179 if (sd_s3_sign(f, "POST", "uploads=", auth, sizeof(auth)) != 0) {
180 sd_s3_set_err(errbuf, errcap, "s3 CreateMPU: sign failed");
181 return -1;
182 }
183 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "POST", wire,
184 auth, NULL, 0, f->timeout_ms, &resp,
185 errbuf, errcap) != 0) {
186 return -1;
187 }
188 if (resp.status != 200) {
189 int rc = sd_s3_status_err(resp.status, "CreateMPU", f->key, errbuf, errcap);
190 int e = errno; /* preserve across resp_free cleanup */
191 f->transport->resp_free(&resp);
192 errno = e;
193 return rc;
194 }
195 body = f->transport->resp_body(&resp, &blen);
196 if (body == NULL || blen == 0
197 || sd_s3_xml_tag((const char *) body, "UploadId",
198 f->upload_id, sizeof(f->upload_id)) != 0) {
199 f->transport->resp_free(&resp);
200 sd_s3_set_err(errbuf, errcap, "s3 CreateMPU: no <UploadId> in response");
201 return -1;
202 }
203 f->transport->resp_free(&resp);
204 return 0;
205}

207static int
208sd_s3_mpu_complete(sd_s3_file *f, char *errbuf, size_t errcap)
209{
210 char wire[SD_S3_KEY_MAX + SD_S3_UPLOAD_ID_LEN + 16];
211 char qs[SD_S3_UPLOAD_ID_LEN + 16], auth[SD_S3_AUTH_HDRS_CAP];
212 brix_s3_resp_t resp;
213 char *xml;
214 size_t cap, len;
215 int i, pn;
216
217 pn = snprintf(wire, sizeof(wire), "%s?uploadId=%s", f->key, f->upload_id);
218 if (pn < 0 || (size_t) pn >= sizeof(wire)) {
219 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: path too long");
220 return -1;
221 }
222 pn = snprintf(qs, sizeof(qs), "uploadId=%s", f->upload_id);
223 if (pn < 0 || (size_t) pn >= sizeof(qs)) {
224 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: canon_qs too long");
225 return -1;
226 }
227 cap = 120 + (size_t) f->part_count * (60 + SD_S3_ETAG_LEN) + 1;
228 xml = malloc(cap);
229 if (xml == NULL) {
230 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: out of memory for XML");
231 return -1;
232 }
233 len = (size_t) snprintf(xml, cap,
234 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
235 "<CompleteMultipartUpload>\n");
236 for (i = 0; i < f->part_count; i++) {
237 int w = snprintf(xml + len, cap - len,
238 " <Part><PartNumber>%d</PartNumber><ETag>%s</ETag></Part>\n",
239 i + 1, f->etags[i].val);
240 if (w < 0 || (size_t) w >= cap - len) {
241 free(xml);
242 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: XML overflow");
243 return -1;
244 }
245 len += (size_t) w;
246 }
247 {
248 int t = snprintf(xml + len, cap - len, "</CompleteMultipartUpload>\n");
249 if (t < 0 || (size_t) t >= cap - len) {
250 free(xml);
251 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: XML overflow (tail)");
252 return -1;
253 }
254 len += (size_t) t;
255 }
256 if (sd_s3_sign(f, "POST", qs, auth, sizeof(auth)) != 0) {
257 free(xml);
258 sd_s3_set_err(errbuf, errcap, "s3 CompleteMPU: sign failed");
259 return -1;
260 }
261 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "POST", wire,
262 auth, xml, len, f->timeout_ms, &resp,
263 errbuf, errcap) != 0) {
264 free(xml);
265 return -1;
266 }
267 free(xml);
268 if (resp.status != 200) {
269 int rc = sd_s3_status_err(resp.status, "CompleteMPU", f->key, errbuf, errcap);
270 f->transport->resp_free(&resp);
271 return rc;
272 }
273 f->transport->resp_free(&resp);
274 return 0;
275}

312sd_s3_file *
313sd_s3_open_write(const sd_s3_open_params *p, int64_t expected_size,
314 int64_t part_size, char *errbuf, size_t errcap)
315{
316 sd_s3_file *f = sd_s3_open_read(p, errbuf, errcap);
317 if (f == NULL) {
318 return NULL;
319 }
320 f->is_write = 1;
321 f->put_checksum = p->put_checksum; /* #12: origin-enforced body integrity */
322 f->part_size = (part_size > 0) ? part_size : (64LL * 1024 * 1024);
323
324 if (expected_size <= f->part_size) {
325 /* Single buffered PUT — including UNKNOWN size (P80.2): start in the
326 * PUT buffer and upgrade to MPU mid-stream only if the volume actually
327 * exceeds part_size (sd_s3_put_upgrade_mpu). Small objects of unknown
328 * size cost one PUT instead of Create+UploadPart+Complete, and a
329 * known-size PUT that overruns its announcement upgrades instead of
330 * growing the buffer without bound. */
331 size_t cap = (expected_size > 0) ? (size_t) expected_size
332 : (size_t) SD_S3_PUT_BUF_INIT;
333 if (cap < (size_t) SD_S3_PUT_BUF_INIT) { cap = SD_S3_PUT_BUF_INIT; }
334 f->put_buf = malloc(cap);
335 if (f->put_buf == NULL) {
336 sd_s3_set_err(errbuf, errcap, "s3 open write: out of memory (PUT)");
337 sd_s3_close(f);
338 return NULL;
339 }
340 f->put_cap = cap;
341 f->lazy_mpu = 1;
342 } else {
343 /* announced size larger than one part: multipart upload from the start */
344 if (sd_s3_mpu_create(f, errbuf, errcap) != 0) {
345 int e = errno; /* preserve the CreateMPU error class */
346 sd_s3_close(f);
347 errno = e;
348 return NULL;
349 }
350 f->part_buf = malloc((size_t) f->part_size);
351 if (f->part_buf == NULL) {
352 sd_s3_set_err(errbuf, errcap, "s3 open write: out of memory (part)");
353 sd_s3_mpu_abort(f);
354 sd_s3_close(f);
355 return NULL;
356 }
357 f->is_mpu = 1;
358 }
359 return f;
360}

362/* Upgrade a lazy PUT-buffered write to a multipart upload mid-stream (P80.2):
363 * create the MPU, move the buffered bytes into the part buffer (put_len is
364 * <= part_size by the caller's trigger condition, so they fit in one part),
365 * and continue accumulating through the normal MPU path. On failure the PUT
366 * buffer is intact and the handle unchanged, so the caller's error return
367 * leaves a consistent (abortable) handle. */
368static int
369sd_s3_put_upgrade_mpu(sd_s3_file *f, char *errbuf, size_t errcap)
370{
371 void *part;
372
373 if (sd_s3_mpu_create(f, errbuf, errcap) != 0) {
374 return -1;
375 }
376 part = malloc((size_t) f->part_size);
377 if (part == NULL) {
378 sd_s3_set_err(errbuf, errcap, "s3 mpu upgrade: out of memory (part)");
379 sd_s3_mpu_abort(f);
380 return -1;
381 }
382 memcpy(part, f->put_buf, f->put_len);
383 f->part_buf = part;
384 f->part_buf_len = f->put_len;
385 f->mpu_write_off = f->put_write_off;
386 free(f->put_buf);
387 f->put_buf = NULL;
388 f->put_len = 0;
389 f->put_cap = 0;
390 f->is_mpu = 1;
391 return 0;
392}

394int
395sd_s3_pwrite(sd_s3_file *f, const void *buf, size_t n, off_t off,
396 char *errbuf, size_t errcap)
397{
398 if (!f->is_write) {
399 sd_s3_set_err(errbuf, errcap, "s3 pwrite: handle opened for read");
400 return -1;
401 }
402 if (n == 0) {
403 return 0;
404 }
405 if (!f->is_mpu && f->lazy_mpu
406 && f->put_len + n > (size_t) f->part_size
407 && sd_s3_put_upgrade_mpu(f, errbuf, errcap) != 0)
408 {
409 return -1;
410 }
411 if (f->is_mpu) {
412 const char *src = (const char *) buf;
413 size_t rem = n;
414 if (sd_s3_check_sequential((int64_t) off, f->mpu_write_off,
415 errbuf, errcap) != 0) {
416 return -1;
417 }
418 while (rem > 0) {
419 size_t space = (size_t) f->part_size - f->part_buf_len;
420 size_t tocopy = (rem < space) ? rem : space;
421 memcpy((char *) f->part_buf + f->part_buf_len, src, tocopy);
422 f->part_buf_len += tocopy;
423 src += tocopy;
424 rem -= tocopy;
425 if (f->part_buf_len == (size_t) f->part_size
426 && sd_s3_mpu_flush(f, errbuf, errcap) != 0) {
427 return -1;
428 }
429 }
430 f->mpu_write_off = (int64_t) off + (int64_t) n;
431 return 0;
432 }
433
434 /* single buffered PUT */
435 if (sd_s3_check_sequential((int64_t) off, f->put_write_off,
436 errbuf, errcap) != 0) {
437 return -1;
438 }
439 if (f->put_len + n > f->put_cap) {
440 size_t nc = f->put_cap ? f->put_cap * 2 : SD_S3_PUT_BUF_INIT;
441 void *nb;
442 while (nc < f->put_len + n) { nc *= 2; }
443 nb = realloc(f->put_buf, nc);
444 if (nb == NULL) {
445 sd_s3_set_err(errbuf, errcap, "s3 single-put: out of memory");
446 return -1;
447 }
448 f->put_buf = nb;
449 f->put_cap = nc;
450 }
451 memcpy((char *) f->put_buf + f->put_len, buf, n);
452 f->put_len += n;
453 f->put_write_off = (int64_t) off + (int64_t) n;
454 return 0;
455}

457int
458sd_s3_commit(sd_s3_file *f, char *errbuf, size_t errcap)
459{
460 if (!f->is_write) {
461 return 0; /* read handle */
462 }
463 if (f->is_mpu) {
464 if (sd_s3_mpu_flush(f, errbuf, errcap) != 0) {
465 return -1;
466 }
467 if (f->part_count == 0
468 && sd_s3_mpu_upload_part(f, 1, NULL, 0, errbuf, errcap) != 0) {
469 return -1; /* zero-byte object: one empty part */
470 }
471 return sd_s3_mpu_complete(f, errbuf, errcap);
472 }
473 /* single PUT of the whole buffer */
474 {
475 char auth[SD_S3_AUTH_HDRS_CAP];
476 char ck[9];
477 brix_s3_resp_t resp;
478 int sr;
479 /* #12: attach a signed x-amz-checksum-crc32 so the origin validates the
480 * body and rejects a wire-corrupted PUT with 400 BadDigest. */
481 if (f->put_checksum) {
482 sd_s3_ck_crc32(f->put_buf, f->put_len, ck);
483 sr = sd_s3_sign_ex(f, "PUT", "", "x-amz-checksum-crc32", ck,
484 auth, sizeof(auth));
485 } else {
486 sr = sd_s3_sign(f, "PUT", "", auth, sizeof(auth));
487 }
488 if (sr != 0) {
489 sd_s3_set_err(errbuf, errcap, "s3 PUT: sign failed");
490 return -1;
491 }
492 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "PUT",
493 f->key, auth, f->put_buf, f->put_len,
494 f->timeout_ms, &resp, errbuf, errcap) != 0) {
495 return -1;
496 }
497 if (resp.status != 200) {
498 int rc = sd_s3_status_err(resp.status, "PUT", f->key, errbuf, errcap);
499 f->transport->resp_free(&resp);
500 return rc;
501 }
502 f->transport->resp_free(&resp);
503 }
504 return 0;
505}
