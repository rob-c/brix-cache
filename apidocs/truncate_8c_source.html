12/*
13 * brix_handle_truncate â set the length of a file to the given offset.
14 *
15 * Wire format (ClientTruncateRequest):
16 * fhandle[4]: open handle (only meaningful when dlen == 0)
17 * offset: big-endian int64 â the new file length in bytes
18 * dlen: 0 means handle-based truncate; >0 means path-based truncate
19 *
20 * Handle-based path: uses the fd already open on the slot.
21 * Path-based path: resolves the path, opens it O_WRONLY, calls ftruncate,
22 * then closes the temporary fd.
23 *
24 * Token scope: write scope is required for both paths.
25 */

28 ngx_stream_brix_srv_conf_t *conf)

36

38

42 char reqpath[BRIX_MAX_PATH + 1];

73 if (fh == NULL) {

77 BRIX_RETURN_ERR(ctx, c, BRIX_OP_TRUNCATE, "TRUNCATE",

83 int err = errno;
84

87 resolved, detail,
88 brix_kxr_from_errno(err), strerror(err));
89 }
90 brix_vfs_close(fh, c->log);
91 }
92 brix_log_access(ctx, c, "TRUNCATE", resolved, detail,
93 1, 0, NULL, 0);
94 } else {
95 /* Handle-based truncate bypasses path resolution and uses the already-open fd. */
96 int idx = (int)(unsigned char) req.fhandle[0];
97 ngx_int_t validate_rc;
98 brix_vfs_job_t job;
99
100 if (!brix_validate_file_handle(ctx, c, idx, "TRUNCATE",
101 BRIX_OP_TRUNCATE, &validate_rc)) {
102 return validate_rc;
103 }
104 brix_vfs_job_truncate_init(&job, ctx->files[idx].fd,
105 (off_t) length);
106 brix_vfs_job_set_obj(&job, &ctx->files[idx].sd_obj);
107 brix_vfs_io_execute(&job);
108 if (job.io_errno != 0) {
109 BRIX_RETURN_ERR(ctx, c, BRIX_OP_TRUNCATE, "TRUNCATE",
110 ctx->files[idx].path, detail,
111 kXR_IOError, strerror(job.io_errno));
112 }
113 if (ctx->files[idx].wt_enabled) {
114 brix_wt_mark_dirty(ctx, idx,
115 (length > 0) ? (off_t) length - 1 : 0,
116 0);
117 }
118 brix_log_access(ctx, c, "TRUNCATE", ctx->files[idx].path, detail,
119 1, 0, NULL, 0);
120 }
121
122 BRIX_OP_OK(ctx, BRIX_OP_TRUNCATE);
123 return brix_send_ok(ctx, c, NULL, 0);
124}
