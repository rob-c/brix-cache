57/* ---- sd_remote_open_impl (shared by the plain and cred-scoped open slots) --
58 *
59 * WHAT: Opens `path` for read against the S3 origin, using `ak`/`sk`/`region`
60 * as the SigV4 credential instead of the instance's static
61 * cfg->access_key/secret_key/region when they are non-NULL.
62 *
63 * WHY: Phase-3 T3 per-user S3 credentials: a request whose identity resolves
64 * to a `<key>.s3` file must sign against the ORIGIN with that user's
65 * keys, not the export's shared service credential. Sharing one open
66 * body between sd_remote_open and sd_remote_open_cred keeps the object-
67 * construction logic (size HEAD, obj/state alloc, snap fill) written
68 * exactly once.
69 *
70 * HOW: Builds the object path, then an sd_s3_open_params using the override
71 * ak/sk/region when given (falling back to the instance's static
72 * triple field-by-field when a pointer is NULL — region alone can be
73 * overridden without ak/sk, though callers always pass all three or
74 * none), and proceeds identically to the prior sd_remote_open body. */
75static brix_sd_obj_t *
76sd_remote_open_impl(brix_sd_instance_t *inst, const char *path, int sd_flags,
77 const char *ak, const char *sk, const char *region, int *err_out)
78{
79 const brix_sd_remote_cfg_t *cfg = inst->state;
80 sd_s3_open_params p;
81 char objpath[768];
82 char errbuf[256];
83 sd_s3_file *s3;
84 sd_remote_obj_state *st;
85 brix_sd_obj_t *obj;
86 int64_t size = 0;
87
88 /* Read-only origin: refuse any write/create/trunc intent up front. */
89 if (sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC
90 | BRIX_SD_O_APPEND)) {
91 if (err_out) { *err_out = EROFS; }
92 return NULL;
93 }
94
95 sd_remote_s3_key(cfg, path, objpath, sizeof(objpath));
96 sd_remote_s3_params(cfg, objpath, &p);
97 if (ak != NULL) { p.ak = ak; }
98 if (sk != NULL) { p.sk = sk; }
99 if (region != NULL) { p.region = region; }
100
101 s3 = sd_s3_open_read(&p, errbuf, sizeof(errbuf));
102 if (s3 == NULL) {
103 if (err_out) { *err_out = ENOMEM; } /* open_read only fails on bad args/OOM */
104 return NULL;
105 }
106 if (sd_s3_size(s3, &size, errbuf, sizeof(errbuf)) != 0) {
107 int e = errno; /* HEAD set errno (ENOENT on 404) */
108
109 sd_s3_close(s3);
110 if (err_out) { *err_out = e ? e : EIO; }
111 return NULL;
112 }
113
114 st = calloc(1, sizeof(*st));
115 obj = calloc(1, sizeof(*obj));
116 if (st == NULL || obj == NULL) {
117 free(st);
118 free(obj);
119 sd_s3_close(s3);
120 if (err_out) { *err_out = ENOMEM; }
121 return NULL;
122 }
123 st->s3 = s3;
124
125 obj->driver = inst->driver;
126 obj->inst = inst;
127 obj->fd = NGX_INVALID_FILE; /* no kernel fd — memory-served */
128 obj->state = st;
129 obj->heap_shell = 1; /* malloc'd shell; caller may free */
130 obj->snap.size = (off_t) size;
131 obj->snap.mode = S_IFREG | 0444;
132 obj->snap.is_reg = 1;
133
134 return obj;
135}

146/* ---- sd_remote_open_cred — per-user SigV4 credential override (phase-3 T3) -
147 *
148 * WHAT: Like sd_remote_open, but when cred->s3_ak (and s3_sk) are set, signs
149 * the request with the caller's per-user access key / secret key /
150 * region instead of the instance's static service credential.
151 *
152 * WHY: brix_sd_open_maybe_cred (sd.h) routes through this slot whenever the
153 * VFS credential gate (vfs_cred.c) resolved a `<key>.s3` file for the
154 * requesting identity — the origin must see THAT user's SigV4
155 * signature, not the export's shared key.
156 *
157 * HOW: A NULL cred, or a cred with s3_ak/s3_sk unset (a non-S3 credential
158 * kind — x509 or bearer — was selected instead, which this backend
159 * cannot use), falls back to the static instance credential via
160 * sd_remote_open_impl(..., NULL, NULL, NULL, ...) — identical to the
161 * plain open. EXCEPT when the operator set fallback_deny: an x509/
162 * bearer-kind cred (missing ak or sk) reaching an S3-only backend under
163 * deny mode must be refused rather than silently signed with the
164 * export's shared service credential — that is exactly the fallback
165 * the operator forbade. */
166static brix_sd_obj_t *
167sd_remote_open_cred(brix_sd_instance_t *inst, const char *path, int sd_flags,
168 mode_t mode, const brix_sd_cred_t *cred, int *err_out)
169{
170 (void) mode;
171
172 if (cred != NULL && cred->s3_ak != NULL && cred->s3_sk != NULL) {
173 return sd_remote_open_impl(inst, path, sd_flags,
174 cred->s3_ak, cred->s3_sk, cred->s3_region, err_out);
175 }
176
177 /* fallback_deny: cred kind unusable by this driver — refuse, don't fall
178 * back to service cred */
179 if (cred != NULL && cred->fallback_deny
180 && (cred->s3_ak == NULL || cred->s3_ak[0] == '\0'
181 || cred->s3_sk == NULL || cred->s3_sk[0] == '\0'))
182 {
183 if (err_out) { *err_out = EACCES; }
184 errno = EACCES;
185 return NULL;
186 }
187
188 return sd_remote_open_impl(inst, path, sd_flags, NULL, NULL, NULL,
189 err_out);
190}

266static brix_sd_staged_t *
267sd_remote_staged_open(brix_sd_instance_t *inst, const char *final_path,
268 mode_t mode, int *err_out)
269{
270 const brix_sd_remote_cfg_t *cfg = inst->state;
271 sd_s3_open_params p;
272 char objpath[768];
273 char errbuf[256];
274 sd_s3_file *s3;
275 sd_remote_staged_state *ss;
276 brix_sd_staged_t *h;
277
278 (void) mode;
279 sd_remote_s3_key(cfg, final_path, objpath, sizeof(objpath));
280 sd_remote_s3_params(cfg, objpath, &p);
281
282 /* Unknown final size -> multipart upload (handles any object size). */
283 s3 = sd_s3_open_write(&p, -1, SD_REMOTE_PART_SIZE, errbuf, sizeof(errbuf));
284 if (s3 == NULL) {
285 if (err_out) { *err_out = EIO; }
286 return NULL;
287 }
288 ss = calloc(1, sizeof(*ss));
289 h = calloc(1, sizeof(*h));
290 if (ss == NULL || h == NULL) {
291 free(ss);
292 free(h);
293 sd_s3_abort(s3);
294 sd_s3_close(s3);
295 if (err_out) { *err_out = ENOMEM; }
296 return NULL;
297 }
298 ss->s3 = s3;
299 h->inst = inst;
300 h->state = ss;
301 return h;
302}
