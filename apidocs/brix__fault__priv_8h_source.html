1/*
2 * brix_fault_priv.h — privileged ("root-ful") fault subsystem for
3 * brix-fault-proxy.
4 *
5 * The base proxy (brix_fault_proxy.c) is deliberately root-FREE: every lever it
6 * offers lives at or above the TCP byte stream it relays. Some network faults,
7 * however, live BELOW TCP and cannot be honestly emulated in userland — genuine
8 * IP packet loss / reordering / duplication / single-bit corruption (which force
9 * real retransmits or checksum drops), a next-hop MTU black hole, or an off-path
10 * attacker forging an ICMP "fragmentation needed" or a mid-stream RST. Those need
11 * CAP_NET_ADMIN / CAP_NET_RAW.
12 *
13 * This module adds those levers as an OPT-IN, root-gated complement, driven by the
14 * same control port via a `priv <verb> ...` command family. It mutates HOST-GLOBAL
15 * network state (a `tc netem` qdisc, an `nft` table, a NIC's MTU) and therefore
16 * registers a teardown that MUST run on every exit path, restoring the host.
17 *
18 * SECURITY: arming requires BOTH euid 0 AND an explicit --privileged opt-in — root
19 * alone does not arm it. Every shell-out is execvp() with an explicit argv (never
20 * a shell), and interface names / numeric operands are validated before use.
21 */
22#ifndef BRIX_FAULT_PRIV_H
23#define BRIX_FAULT_PRIV_H

26#include <sys/socket.h>

28/* Arm the privileged subsystem. `iface` (may be NULL) is the NIC that netem /
29 * mtu levers act on; `listen_port` and the `target_ports[nports]` are used to
30 * scope the nft cut rules to this proxy's own traffic. On failure returns -1 and
31 * points *err at a static human-readable reason (not root, --privileged missing,
32 * bad iface). */
33int fp_priv_enable(const char *iface, int listen_port,

41 * reply line into reply/rsz. Always returns 1 (the command was claimed). */
42int fp_priv_command(char *args, char *reply, size_t rsz);

47/* Remove ALL installed host state: delete the netem qdisc, delete the nft table,

50void fp_priv_teardown(void);
