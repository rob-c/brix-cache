1/*
2 * stage_engine.c - the one async-staging engine (phase-64 section 11). See header.
3 *
4 * SP1 lands the seam: the generic promote-loop mover (open src for read,
5 * staged_open dst, pread -> staged_write -> staged_commit) plus the inline submit
6 * front door and the unified audit line. The durable queue + waiter + restart
7 * reconcile are extracted from src/frm/ in SP4 (section 13b) and attach behind
8 * brix_stage_submit() without touching a caller; until then async degrades to an
9 * honest inline move and the scheduler/reconcile hooks are no-ops.
10 */

27/* Move granule: one 1 MiB driver-mediated pread/staged_write per turn (the same
28 * window the legacy sd_stage promote and the xfer pump use). */
29#define STAGE_ENGINE_CHUNK (1u << 20)

31/* ---- SP4: the durable async queue -----------------------------------------
32 * An async submit is DEFERRED rather than run inline: the request is appended to
33 * a per-worker in-memory pending list (holding the live src/dst instances, which
34 * are the memoised per-worker tier instances - they outlive the request) and,
35 * when a journal dir is configured, persisted as a small record for crash
36 * visibility/recovery. brix_stage_scheduler_tick() (a per-worker timer) drains
37 * the list, runs each mover, and drops the stage copy of a completed FLUSH. This
38 * generalises the FRM queue model to SD instances (section 11); the full physical
39 * extraction of src/frm/ is the remaining SP4/SP5 migration. */
40

47 char dst_key[1024];
48 char export_root[1024]; /* anchor for restart reconcile */
49 struct stage_pending_s *next;
50} stage_pending_t;

75/* Persist (best-effort) a QUEUED request record so a crash leaves a recoverable
76 * row; removed on completion. Skipped when no journal dir is configured. */
77static void
78stage_journal_write(const stage_pending_t *p)
79{
80 brix_sreq_t rec;
81 char path[1200];
82 int fd;
83
84 if (stage_journal_dir[0] == '\0') {
85 return;
86 }
87 if ((size_t) snprintf(path, sizeof(path), "%s/%s.req",
88 stage_journal_dir, p->reqid) >= sizeof(path))
89 {
90 return;
91 }
92 ngx_memzero(&rec, sizeof(rec));
93 snprintf(rec.reqid, sizeof(rec.reqid), "%s", p->reqid);
94 rec.kind = p->kind;
95 rec.state = BRIX_SREQ_QUEUED;
96 snprintf(rec.src_driver, sizeof(rec.src_driver), "%s",
97 (p->src->driver && p->src->driver->name) ? p->src->driver->name : "");
98 snprintf(rec.src_key, sizeof(rec.src_key), "%s", p->src_key);
99 snprintf(rec.dst_driver, sizeof(rec.dst_driver), "%s",
100 (p->dst->driver && p->dst->driver->name) ? p->dst->driver->name : "");
101 snprintf(rec.dst_key, sizeof(rec.dst_key), "%s", p->dst_key);
102 snprintf(rec.export_root, sizeof(rec.export_root), "%s", p->export_root);
103 rec.enqueued_at = (int64_t) time(NULL);
104
105 fd = open(path, O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC, 0600);
106 if (fd < 0) {
107 return;
108 }
109 if (write(fd, &rec, sizeof(rec)) == (ssize_t) sizeof(rec)) {
110 (void) fsync(fd);
111 }
112 (void) close(fd);
113}

165/* ---- the generic promote-loop mover --------------------------------------- */
166
167/* Copy the whole object `src_key` on `src` into `dst_key` on `dst` by reading
168 * through the source driver and writing through the destination's staged sink,
169 * then committing it atomically. Both endpoints are SD instances, so this same
170 * loop moves bytes between ANY two tiers (posix stage -> remote backend, tape
171 * buffer -> posix cache, ...). On success *bytes_out carries the moved size and
172 * the staged handle is consumed by commit; on failure the staged temp is aborted
173 * and *err_out carries errno. Returns an brix_xfer_result_t terminal code. */
174static brix_xfer_result_t
175stage_engine_move(brix_sd_instance_t *src, const char *src_key,
176 brix_sd_instance_t *dst, const char *dst_key, off_t *bytes_out,
177 int *err_out)
178{
179 brix_sd_obj_t *so;
180 brix_sd_staged_t *ds;
181 u_char *buf;
182 off_t off = 0;
183 int oerr = 0;
184 mode_t mode;
185
186 if (src->driver->open == NULL || src->driver->pread == NULL) {
187 *err_out = ENOSYS;
188 return BRIX_XFER_SRC_ERR;
189 }
190 if (dst->driver->staged_open == NULL || dst->driver->staged_write == NULL
191 || dst->driver->staged_commit == NULL || dst->driver->staged_abort == NULL)
192 {
193 *err_out = ENOSYS;
194 return BRIX_XFER_DST_ERR;
195 }
196
197 so = src->driver->open(src, src_key, BRIX_SD_O_READ, 0, &oerr);
198 if (so == NULL) {
199 *err_out = oerr ? oerr : EIO;
200 ngx_log_error(NGX_LOG_ERR, src->log, *err_out,
201 "stage move: source open failed (%s key=\"%s\")",
202 src->driver->name, src_key);
203 return BRIX_XFER_SRC_ERR;
204 }
205
206 /* open() may not populate snap (the posix driver fstats lazily); fstat for an
207 * accurate mode so a flush preserves the source's permission bits. */
208 {
209 brix_sd_stat_t snap = so->snap;
210
211 if (src->driver->fstat != NULL) {
212 (void) src->driver->fstat(so, &snap);
213 }
214 mode = (mode_t) (snap.mode & 0777);
215 }
216 if (mode == 0) {
217 /* SECURITY: unknown source provenance → a PRIVATE tier artifact (0600),
218 * not world-readable. Real source modes flow through snap.mode above and
219 * are preserved; this is only the fallback when the driver reports no
220 * mode, consistent with the 0600/0700 physical-cache convention. */
221 mode = 0600;
222 }
223
224 ds = dst->driver->staged_open(dst, dst_key, mode, &oerr);
225 if (ds == NULL) {
226 brix_sd_obj_release(so);
227 *err_out = oerr ? oerr : EIO;
228 ngx_log_error(NGX_LOG_ERR, dst->log, *err_out,
229 "stage move: dest staged_open failed (%s key=\"%s\")",
230 dst->driver->name, dst_key);
231 return BRIX_XFER_DST_ERR;
232 }
233
234 buf = malloc(STAGE_ENGINE_CHUNK);
235 if (buf == NULL) {
236 dst->driver->staged_abort(ds);
237 brix_sd_obj_release(so);
238 *err_out = ENOMEM;
239 return BRIX_XFER_DST_ERR;
240 }
241
242 for ( ;; ) {
243 ssize_t r = src->driver->pread(so, buf, STAGE_ENGINE_CHUNK, off);
244
245 if (r < 0) {
246 if (errno == EINTR) {
247 continue;
248 }
249 oerr = errno ? errno : EIO;
250 ngx_log_error(NGX_LOG_ERR, src->log, oerr,
251 "stage move: source read failed at off=%O (%s key=\"%s\")",
252 off, src->driver->name, src_key);
253 free(buf);
254 dst->driver->staged_abort(ds);
255 brix_sd_obj_release(so);
256 *err_out = oerr;
257 return BRIX_XFER_SRC_ERR;
258 }
259 if (r == 0) {
260 break; /* EOF - the whole object is moved */
261 }
262 if (dst->driver->staged_write(ds, buf, (size_t) r, off) < 0) {
263 oerr = errno ? errno : EIO;
264 ngx_log_error(NGX_LOG_ERR, dst->log, oerr,
265 "stage move: dest write failed at off=%O (%s key=\"%s\")",
266 off, dst->driver->name, dst_key);
267 free(buf);
268 dst->driver->staged_abort(ds);
269 brix_sd_obj_release(so);
270 *err_out = oerr;
271 return BRIX_XFER_DST_ERR;
272 }
273 off += r;
274 }
275
276 free(buf);
277 brix_sd_obj_release(so);
278
279 if (dst->driver->staged_commit(ds, 0) != NGX_OK) {
280 oerr = errno ? errno : EIO;
281 ngx_log_error(NGX_LOG_ERR, dst->log, oerr,
282 "stage move: dest commit failed (%s key=\"%s\")",
283 dst->driver->name, dst_key);
284 dst->driver->staged_abort(ds); /* commit failed - drop the temp */
285 *err_out = oerr;
286 return BRIX_XFER_COMMIT_ERR;
287 }
288
289 *bytes_out = off;
290 return BRIX_XFER_OK;
291}

293/* Move the object inline and book one unified audit line. Shared by the async
294 * front door (brix_stage_submit) and the sync path (brix_stage_run_inline). */
295static brix_xfer_result_t
296stage_engine_run(brix_stage_kind_t kind, brix_sd_instance_t *src,
297 const char *src_key, brix_sd_instance_t *dst, const char *dst_key)
298{
299 brix_xfer_result_t res;
300 off_t bytes = 0;
301 int oerr = 0;
302 ngx_log_t *log = (dst->log != NULL) ? dst->log : src->log;
303
304 res = stage_engine_move(src, src_key, dst, dst_key, &bytes, &oerr);
305
306 /* One unified audit line per terminal transfer (transport-agnostic). */
307 brix_xfer_finish(stage_kind_to_xfer(kind), stage_kind_dir(kind), dst_key,
308 NULL, (size_t) bytes, res, (res == BRIX_XFER_OK) ? 0 : oerr, log);
309
310 if (res != BRIX_XFER_OK && oerr != 0) {
311 errno = oerr;
312 }
313 return res;
314}

316/* ---- the public front doors ----------------------------------------------- */
317
318const char *
319brix_stage_submit(brix_stage_kind_t kind, brix_sd_instance_t *src,
320 const char *src_key, brix_sd_instance_t *dst, const char *dst_key,
321 const brix_stage_opts_t *opts)
322{
323 static const char ran_inline[] = "";
324 static char last_reqid[40]; /* event-loop single-threaded: stable enough */
325 stage_pending_t *p;
326
327 if (src == NULL || src_key == NULL || dst == NULL || dst_key == NULL
328 || src->driver == NULL || dst->driver == NULL)
329 {
330 return NULL;
331 }
332
333 /* Synchronous (the default): move now and return "" = done inline. */
334 if (opts == NULL || !opts->async) {
335 (void) stage_engine_run(kind, src, src_key, dst, dst_key);
336 return ran_inline;
337 }
338
339 /* Asynchronous: defer to the scheduler (durable). The src/dst instances are
340 * the memoised per-worker tier instances, so holding them is safe. */
341 p = calloc(1, sizeof(*p));
342 if (p == NULL) {
343 (void) stage_engine_run(kind, src, src_key, dst, dst_key); /* never lose it */
344 return ran_inline;
345 }
346 stage_reqid_mint(p->reqid);
347 p->kind = kind;
348 p->src = src;
349 p->dst = dst;
350 snprintf(p->src_key, sizeof(p->src_key), "%s", src_key);
351 snprintf(p->dst_key, sizeof(p->dst_key), "%s", dst_key);
352 if (opts->export_root != NULL) {
353 snprintf(p->export_root, sizeof(p->export_root), "%s", opts->export_root);
354 }
355
356 if (stage_pending_tail != NULL) {
357 stage_pending_tail->next = p;
358 } else {
359 stage_pending_head = p;
360 }
361 stage_pending_tail = p;
362
363 stage_journal_write(p);
364 snprintf(last_reqid, sizeof(last_reqid), "%s", p->reqid);
365 return last_reqid; /* non-empty = deferred; the caller may park on it */
366}

368ngx_int_t
369brix_stage_run_inline(brix_stage_kind_t kind, brix_sd_instance_t *src,
370 const char *src_key, brix_sd_instance_t *dst, const char *dst_key)
371{
372 if (src == NULL || src_key == NULL || dst == NULL || dst_key == NULL
373 || src->driver == NULL || dst->driver == NULL)
374 {
375 errno = EINVAL;
376 return NGX_ERROR;

382/* Max transfers a single tick STARTS, so a backlog never monopolises the worker.
383 * The next tick continues the queue. */
384#define STAGE_TICK_BUDGET 32
385
386/* Terminal post-processing for a completed mover (inline or thread): on success a
387 * FLUSH drops the now-redundant stage copy (the migrate semantic, §9.5) and the
388 * journal record is removed; on failure the record is KEPT (the reconcile retries
389 * it on restart, or a later tick re-drives it). Shared by both paths. */
390static void
391stage_complete(brix_stage_kind_t kind, brix_sd_instance_t *src,
392 const char *src_key, const char *dst_key, const char *reqid,
393 brix_xfer_result_t res, ngx_log_t *log)
394{
395 if (res == BRIX_XFER_OK) {
396 if (kind == BRIX_STAGE_FLUSH && src->driver->unlink != NULL) {
397 (void) src->driver->unlink(src, src_key, 0);
398 }
399 stage_journal_remove(reqid);
400 } else {
401 ngx_log_error(NGX_LOG_WARN, log, 0,
402 "xrootd stage: deferred %s of \"%s\" failed (reqid %s) - record kept",
403 brix_stage_kind_str(kind), dst_key, reqid);
404 }
405}

409/* In-flight offloaded movers (per-worker), bounded so a burst never floods the
410 * thread pool nor unbounds memory. The tick stops starting new ones at the cap and
411 * resumes next tick as they drain. */
412static ngx_uint_t stage_inflight;
413#define STAGE_MAX_INFLIGHT 8
414
415/* The off-loop mover task (lives on its own small pool, freed in the done event). */
416typedef struct {
417 ngx_pool_t *pool;
418 brix_stage_kind_t kind;
419 brix_sd_instance_t *src;
420 brix_sd_instance_t *dst;
421 brix_xfer_result_t res;
422 ngx_log_t *log;
423 char reqid[40];
424 char src_key[1024];
425 char dst_key[1024];
426} stage_flush_task_t;

428static void
429stage_flush_thread(void *data, ngx_log_t *log)
430{
431 stage_flush_task_t *t = data;
432
433 (void) log;
434 t->res = stage_engine_run(t->kind, t->src, t->src_key, t->dst, t->dst_key);

437static void
438stage_flush_done(ngx_event_t *ev)
439{
440 ngx_thread_task_t *task = ev->data;
441 stage_flush_task_t *t = task->ctx;
442 ngx_pool_t *pool = t->pool;
443
444 if (stage_inflight > 0) {
445 stage_inflight--;
446 }
447 stage_complete(t->kind, t->src, t->src_key, t->dst_key, t->reqid, t->res,
448 t->log);
449 ngx_destroy_pool(pool); /* frees the task + ctx */
450}

452/* Post `p`'s mover to `pool`. NGX_OK = posted (caller drops the pending item; the
453 * journal record persists until the thread completes); NGX_DECLINED = setup failed
454 * (caller runs it inline). */
455static ngx_int_t
456stage_flush_offload(const stage_pending_t *p, ngx_thread_pool_t *pool)
457{
458 ngx_log_t *log = (p->dst->log != NULL) ? p->dst->log : p->src->log;
459 ngx_pool_t *tp;
460 ngx_thread_task_t *task;
461 stage_flush_task_t *t;
462
463 if (log == NULL) {
464 log = ngx_cycle->log;
465 }
466 tp = ngx_create_pool(4096, log);
467 if (tp == NULL) {
468 return NGX_DECLINED;
469 }
470 task = ngx_thread_task_alloc(tp, sizeof(stage_flush_task_t));
471 if (task == NULL) {
472 ngx_destroy_pool(tp);
473 return NGX_DECLINED;
474 }
475 t = task->ctx;
476 t->pool = tp;
477 t->kind = p->kind;
478 t->src = p->src;
479 t->dst = p->dst;
480 t->res = BRIX_XFER_DST_ERR;
481 t->log = log;
482 snprintf(t->reqid, sizeof(t->reqid), "%s", p->reqid);
483 snprintf(t->src_key, sizeof(t->src_key), "%s", p->src_key);
484 snprintf(t->dst_key, sizeof(t->dst_key), "%s", p->dst_key);
485
486 brix_task_bind(task, stage_flush_thread, stage_flush_done);
487 task->event.log = log;
488 if (ngx_thread_task_post(pool, task) != NGX_OK) {
489 ngx_destroy_pool(tp);
490 return NGX_DECLINED;
491 }
492 stage_inflight++;
493 return NGX_OK;
494}

508void
509brix_stage_scheduler_tick(void)
510{
511 int budget = STAGE_TICK_BUDGET;
512#if (NGX_THREADS)
513 ngx_thread_pool_t *pool = stage_thread_pool();
514#endif
515
516 while (stage_pending_head != NULL && budget-- > 0) {
517 stage_pending_t *p = stage_pending_head;
518 brix_xfer_result_t res;
519 ngx_log_t *log = (p->dst->log != NULL) ? p->dst->log
520 : p->src->log;
521
522#if (NGX_THREADS)
523 /* Run the mover OFF the event loop so a flush to a REMOTE backend (or from a
524 * remote stage store) never blocks/fails on the un-pumped loop. Bounded
525 * in-flight; the durable journal record is what survives a crash mid-flush
526 * (recovered by reconcile), so the on-disk record is dropped only in the
527 * completion. */
528 if (pool != NULL) {
529 if (stage_inflight >= STAGE_MAX_INFLIGHT) {
530 break; /* let in-flight drain; resume next tick */
531 }
532 if (stage_flush_offload(p, pool) == NGX_OK) {
533 stage_pending_head = p->next;
534 if (stage_pending_head == NULL) {
535 stage_pending_tail = NULL;
536 }
537 free(p);
538 continue;
539 }
540 /* offload setup failed - fall through to the inline path */
541 }
542#endif
543
544 stage_pending_head = p->next;
545 if (stage_pending_head == NULL) {
546 stage_pending_tail = NULL;
547 }
548 res = stage_engine_run(p->kind, p->src, p->src_key, p->dst, p->dst_key);
549 stage_complete(p->kind, p->src, p->src_key, p->dst_key, p->reqid, res,
550 log);
551 free(p);
552 }
553}

555/* Re-flush ONE persisted record. Only a staged-write FLUSH is replayable: its
556 * bytes are durable on the stage store and the export anchor lets us rebuild both
557 * tiers. An UPLOAD/MULTIPART (client body -> stage) cannot be replayed - the body
558 * is gone after a crash, so the client retries; such records are dropped, never
559 * silently re-driven wrong. Returns 1 replayed / 0 kept (retry) / -1 dropped. */
560static int
561stage_reconcile_one(const char *path, ngx_log_t *log)
562{
563 brix_sreq_t rec;
564 brix_sd_instance_t *inst;
565 int fd;
566 ssize_t n;
567
568 fd = open(path, O_RDONLY | O_CLOEXEC);
569 if (fd < 0) {
570 return 0;
571 }
572 n = read(fd, &rec, sizeof(rec));
573 (void) close(fd);
574 if (n != (ssize_t) sizeof(rec)) {
575 (void) unlink(path); /* corrupt/short record - drop */
576 return -1;
577 }
578 if (rec.kind != BRIX_STAGE_FLUSH || rec.export_root[0] == '\0') {
579 (void) unlink(path); /* not a recoverable staged write */
580 return -1;
581 }
582
583 /* Rebuild the export's composed stack and unwrap to its stage decorator. */
584 inst = brix_vfs_backend_resolve(rec.export_root, log);
585 if (brix_sd_cache_instance_is(inst)) {
586 inst = brix_sd_cache_source_instance(inst);
587 }
588 if (brix_sd_stage_instance_is(inst)
589 && brix_sd_stage_reflush(inst, rec.dst_key) == NGX_OK)
590 {
591 (void) unlink(path); /* re-flushed + stage copy dropped */
592 return 1;
593 }
594 /* Backend unreachable / no stage tier now: keep the record for a later retry. */
595 ngx_log_error(NGX_LOG_WARN, log, 0,
596 "xrootd stage: reconcile could not re-flush \"%s\" (export \"%s\") - "
597 "record kept for retry", rec.dst_key, rec.export_root);
598 return 0;
599}

603{
604 DIR *d;
605 struct dirent *de;
606 char names[1024][256];
607 ngx_uint_t ncount = 0, i, replayed = 0, kept = 0, dropped = 0;
608 ngx_log_t *log;
609
610 (void) queue;
611 if (stage_journal_dir[0] == '\0') {
612 return; /* no durable journal configured */
613 }
614 /* Reconcile REPORTS every replay/keep/drop decision through ngx_log_error,
615 * whose macro dereferences the log unconditionally — a NULL log here is a
616 * crash, not a quiet run. Without a cycle (pre-init or a bare harness)
617 * DEFER: return with the journal untouched so a properly-initialised boot
618 * replays it. The in-tree caller (worker-0 init_process) always has one. */
619 if (ngx_cycle == NULL) {
620 return;
621 }
622 log = ngx_cycle->log;
623 d = opendir(stage_journal_dir);
624 if (d == NULL) {
625 return;
626 }
627 /* Snapshot the *.req names first (we unlink while driving). */
628 while ((de = readdir(d)) != NULL && ncount < 1024) {
629 size_t nlen = strlen(de->d_name);
630
631 if (nlen > 4 && nlen < sizeof(names[0])
632 && strcmp(de->d_name + nlen - 4, ".req") == 0)
633 {
634 memcpy(names[ncount], de->d_name, nlen + 1);
635 ncount++;
636 }
637 }
638 closedir(d);
639
640 for (i = 0; i < ncount; i++) {
641 char path[1300];
642 int r;
643
644 if ((size_t) snprintf(path, sizeof(path), "%s/%s",
645 stage_journal_dir, names[i]) >= sizeof(path))
646 {
647 continue;
648 }
649 r = stage_reconcile_one(path, log);
650 if (r > 0) { replayed++; }
651 else if (r < 0) { dropped++; }
652 else { kept++; }
653 }
654
655 if (replayed > 0 || kept > 0 || dropped > 0) {
656 ngx_log_error(NGX_LOG_NOTICE, log, 0,
657 "xrootd stage: restart reconcile - %ui staged flush(es) replayed, "
658 "%ui kept for retry, %ui dropped", replayed, kept, dropped);
659 }
660}
