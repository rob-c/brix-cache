1/*
2 * stage_engine.c - the generic byte mover at the heart of the one async-staging
3 * engine (phase-64 section 11). See stage_engine.h.
4 *
5 * This translation unit owns the kind->ledger vocabulary and the generic
6 * promote-loop mover: open src for read, staged_open dst, pread -> staged_write
7 * -> staged_commit, with optional per-user credential re-resolution and one
8 * unified audit line per terminal transfer. It also carries the inline run
9 * front doors (brix_stage_run_inline[_cred]) and the size-tolerant journal-record
10 * decode.
11 *
12 * The durable journal + dead-letter substrate lives in stage_engine_journal.c;
13 * the async front door + per-worker scheduler in stage_engine_scheduler.c; and
14 * the restart reconcile in stage_engine_reconcile.c (phase-79 file-size split).
15 * The three halves reach the mover (stage_engine_run) and each other only

68/* The two endpoints of one move: a source instance+key read through the source
69 * driver and a destination instance+key written through the staged sink. Bundled
70 * into one file-local descriptor so the mover helpers thread a single argument
71 * instead of four positional ones (source/dest confusion is a data-flow hazard). */
72typedef struct {
73 brix_sd_instance_t *src;
74 const char *src_key;
75 brix_sd_instance_t *dst;
76 const char *dst_key;
77} stage_move_ep_t;
78
79/* Validate that both endpoints expose the driver ops the mover needs.
80 *
81 * WHAT: Returns BRIX_XFER_OK when the source can open+pread and the destination
82 * provides the full staged-write quartet (open/write/commit/abort);
83 * otherwise sets *err_out=ENOSYS and returns the endpoint-specific
84 * terminal code (SRC_ERR for a source gap, DST_ERR for a dest gap).
85 *
86 * WHY: A capability-typed driver may omit ops; the mover must fail loudly and
87 * early rather than NULL-deref a missing function pointer mid-copy.
88 *
89 * HOW: Two guard checks, each setting *err_out on the failing side. */
90static brix_xfer_result_t
91stage_move_caps_check(const stage_move_ep_t *ep, int *err_out)
92{
93 if (ep->src->driver->open == NULL || ep->src->driver->pread == NULL) {
94 *err_out = ENOSYS;
95 return BRIX_XFER_SRC_ERR;
96 }
97 if (ep->dst->driver->staged_open == NULL
98 || ep->dst->driver->staged_write == NULL
99 || ep->dst->driver->staged_commit == NULL
100 || ep->dst->driver->staged_abort == NULL)
101 {
102 *err_out = ENOSYS;
103 return BRIX_XFER_DST_ERR;
104 }
105 return BRIX_XFER_OK;
106}

108/* Resolve the source object's permission bits for a mode-preserving flush.
109 *
110 * WHAT: Returns the low 9 permission bits of the source, fstat-refreshed when
111 * the driver supports it, falling back to 0600 when the driver reports no
112 * mode.
113 *
114 * WHY: open() may not populate snap (the posix driver fstats lazily); an
115 * accurate mode lets a flush preserve the source's permission bits. A
116 * zero mode means unknown provenance → a PRIVATE tier artifact (0600),
117 * not world-readable, consistent with the 0600/0700 physical-cache
118 * convention.
119 *
120 * HOW: Copy snap, refresh via fstat if available, mask to 0777, default 0600. */
121static mode_t
122stage_move_source_mode(brix_sd_instance_t *src, brix_sd_obj_t *so)
123{
124 brix_sd_stat_t snap = so->snap;
125 mode_t mode;
126
127 if (src->driver->fstat != NULL) {
128 (void) src->driver->fstat(so, &snap);
129 }
130 mode = (mode_t) (snap.mode & 0777);
131 if (mode == 0) {
132 mode = 0600;
133 }
134 return mode;
135}

137/* Stream the whole source object into the open staged destination handle.
138 *
139 * WHAT: Runs the 1 MiB-granule pread -> staged_write copy loop until EOF. On
140 * success returns BRIX_XFER_OK with *bytes_out set to the total copied;
141 * on a read or write error sets *err_out=errno and returns the
142 * endpoint-specific terminal code. Neither `so` nor `ds` is closed here —
143 * the caller owns handle teardown.
144 *
145 * WHY: Isolating the loop keeps the branch-heavy error handling out of the
146 * orchestration function so each has a single job.
147 *
148 * HOW: Loop pread (retry on EINTR), staged_write at the running offset; break
149 * on EOF; on error early-return with the appropriate terminal code. */
150static brix_xfer_result_t
151stage_move_copy_loop(const stage_move_ep_t *ep, brix_sd_obj_t *so,
152 brix_sd_staged_t *ds, off_t *bytes_out, int *err_out)
153{
154 u_char *buf;
155 off_t off = 0;
156 int oerr;
157
158 buf = malloc(STAGE_ENGINE_CHUNK);
159 if (buf == NULL) {
160 *err_out = ENOMEM;
161 return BRIX_XFER_DST_ERR;
162 }
163
164 for ( ;; ) {
165 ssize_t r = ep->src->driver->pread(so, buf, STAGE_ENGINE_CHUNK, off);
166
167 if (r < 0) {
168 if (errno == EINTR) {
169 continue;
170 }
171 oerr = errno ? errno : EIO;
172 ngx_log_error(NGX_LOG_ERR, ep->src->log, oerr,
173 "stage move: source read failed at off=%O (%s key=\"%s\")",
174 off, ep->src->driver->name, ep->src_key);
175 free(buf);
176 *err_out = oerr;
177 return BRIX_XFER_SRC_ERR;
178 }
179 if (r == 0) {
180 break; /* EOF - the whole object is moved */
181 }
182 if (ep->dst->driver->staged_write(ds, buf, (size_t) r, off) < 0) {
183 oerr = errno ? errno : EIO;
184 ngx_log_error(NGX_LOG_ERR, ep->dst->log, oerr,
185 "stage move: dest write failed at off=%O (%s key=\"%s\")",
186 off, ep->dst->driver->name, ep->dst_key);
187 free(buf);
188 *err_out = oerr;
189 return BRIX_XFER_DST_ERR;
190 }
191 off += r;
192 }
193
194 free(buf);
195 *bytes_out = off;
196 return BRIX_XFER_OK;
197}

199/* Copy the whole object `src_key` on `src` into `dst_key` on `dst` by reading
200 * through the source driver and writing through the destination's staged sink,
201 * then committing it atomically. Both endpoints are SD instances, so this same
202 * loop moves bytes between ANY two tiers (posix stage -> remote backend, tape
203 * buffer -> posix cache, ...). On success *bytes_out carries the moved size and
204 * the staged handle is consumed by commit; on failure the staged temp is aborted
205 * and *err_out carries errno. `cred` is the per-user credential for the dst
206 * staged_open (NULL = service credential / driver default). Returns an
207 * brix_xfer_result_t terminal code. */
208static brix_xfer_result_t
209stage_engine_move(const stage_move_ep_t *ep, const brix_sd_cred_t *cred,
210 off_t *bytes_out, int *err_out)
211{
212 brix_sd_obj_t *so;
213 brix_sd_staged_t *ds;
214 brix_xfer_result_t res;
215 int oerr = 0;
216 mode_t mode;
217
218 res = stage_move_caps_check(ep, err_out);
219 if (res != BRIX_XFER_OK) {
220 return res;
221 }
222
223 so = ep->src->driver->open(ep->src, ep->src_key, BRIX_SD_O_READ, 0, &oerr);
224 if (so == NULL) {
225 *err_out = oerr ? oerr : EIO;
226 ngx_log_error(NGX_LOG_ERR, ep->src->log, *err_out,
227 "stage move: source open failed (%s key=\"%s\")",
228 ep->src->driver->name, ep->src_key);
229 return BRIX_XFER_SRC_ERR;
230 }
231
232 mode = stage_move_source_mode(ep->src, so);
233
234 /* Use the cred-aware staged_open for the destination so the backend driver
235 * presents the per-user x509 proxy when flushing to a remote origin. The
236 * source (stage store) is always local so no cred is needed there. */
237 ds = brix_sd_staged_open_maybe_cred(ep->dst, ep->dst_key, mode, cred, &oerr);
238 if (ds == NULL) {
239 brix_sd_obj_release(so);
240 *err_out = oerr ? oerr : EIO;
241 ngx_log_error(NGX_LOG_ERR, ep->dst->log, *err_out,
242 "stage move: dest staged_open failed (%s key=\"%s\")",
243 ep->dst->driver->name, ep->dst_key);
244 return BRIX_XFER_DST_ERR;
245 }
246
247 res = stage_move_copy_loop(ep, so, ds, bytes_out, err_out);
248 brix_sd_obj_release(so);
249 if (res != BRIX_XFER_OK) {
250 ep->dst->driver->staged_abort(ds); /* copy failed - drop the temp */
251 return res;
252 }
253
254 if (ep->dst->driver->staged_commit(ds, 0) != NGX_OK) {
255 oerr = errno ? errno : EIO;
256 ngx_log_error(NGX_LOG_ERR, ep->dst->log, oerr,
257 "stage move: dest commit failed (%s key=\"%s\")",
258 ep->dst->driver->name, ep->dst_key);
259 ep->dst->driver->staged_abort(ds); /* commit failed - drop the temp */
260 *err_out = oerr;
261 return BRIX_XFER_COMMIT_ERR;
262 }
263
264 return BRIX_XFER_OK;
265}

267/*
268 * Move the object inline, applying optional per-user credential re-resolution
269 * before touching the origin, and booking one unified audit line.
270 *
271 * WHAT: Re-resolves the per-user x509 proxy (if cred != NULL && cred->key[0])
272 * via brix_sd_ucred_resolve before calling the byte mover. On success
273 * the resolved path is threaded into brix_sd_cred_t and passed to
274 * stage_engine_move so the destination driver presents the user's identity.
275 * On failure with cred->deny set, returns BRIX_XFER_DENIED without moving
276 * data (hard EACCES, audit line emitted). On failure without deny, warns
277 * and falls back to the service credential.
278 *
279 * WHY: A detached async flush (possibly after a restart) must authenticate to
280 * the origin as the ORIGINAL user — not the service account — for per-user
281 * quota / audit / ACL enforcement on the backend. Deny mode makes the
282 * missing-credential case loud rather than silently promoting.
283 *
284 * HOW: 1. If cred is NULL or key[0]=='\0': move with service credential (NULL).
285 * 2. Call brix_sd_ucred_resolve(dir, key, &ru): on OK build brix_sd_cred_t
286 * (x509_proxy=ru.path, principal=cred->principal) and move with it.
287 * 3. On failure + deny=1: log ERR, book DENIED audit line, EACCES, return.
288 * 4. On failure + deny=0: log WARN, move with service cred (credp=NULL).
289 * 5. Pass the principal through to the audit line (was hard-coded NULL).
290 */
291brix_xfer_result_t
292stage_engine_run(brix_stage_kind_t kind, brix_sd_instance_t *src,
293 const char *src_key, brix_sd_instance_t *dst, const char *dst_key,
294 const brix_stage_cred_t *cred)
295{
296 brix_xfer_result_t res;
297 off_t bytes = 0;
298 int oerr = 0;
299 ngx_log_t *log = (dst->log != NULL) ? dst->log : src->log;
300 brix_sd_cred_t sdcred;
301 const brix_sd_cred_t *credp = NULL;
302 const char *principal = NULL;
303 /* Function-scoped so sdcred.x509_proxy (which borrows ru.path) stays valid
304 * through the stage_engine_move below, and so the resolved secret can be
305 * cleansed after the flush consumes it (A-4/T4). */
306 brix_sd_ucred_t ru;
307
308 ngx_memzero(&ru, sizeof(ru));
309
310 if (cred != NULL && cred->principal[0] != '\0') {
311 principal = cred->principal;
312 }
313
314 if (cred != NULL && cred->key[0] != '\0') {
315 if (brix_sd_ucred_resolve(cred->dir, cred->key, &ru) == NGX_OK) {
316 ngx_memzero(&sdcred, sizeof(sdcred));
317 sdcred.x509_proxy = ru.path;
318 sdcred.principal = cred->principal;
319 credp = &sdcred;
320 } else if (cred->deny) {
321 /* Hard deny: missing or expired per-user credential; refuse to flush
322 * under the service identity — the caller opted into strict mode. */
323 ngx_log_error(NGX_LOG_ERR, log, 0,
324 "xrootd stage: %s of \"%s\" DENIED - per-user credential "
325 "key=%s principal=\"%s\" %s (fallback=deny)",
326 brix_stage_kind_str(kind), dst_key, cred->key,
327 cred->principal[0] ? cred->principal : "-",
328 ru.expired ? "EXPIRED" : "missing");
329 brix_xfer_finish(stage_kind_to_xfer(kind), stage_kind_dir(kind),
330 dst_key, principal, 0, BRIX_XFER_DENIED, EACCES, log);
331 errno = EACCES;
332 return BRIX_XFER_DENIED;
333 } else {
334 /* Soft fallback: warn and continue under the service credential. */
335 ngx_log_error(NGX_LOG_WARN, log, 0,
336 "xrootd stage: per-user credential key=%s %s - flushing "
337 "\"%s\" with the service credential (fallback=allow)",
338 cred->key, ru.expired ? "EXPIRED" : "missing", dst_key);
339 }
340 }
341
342 {
343 stage_move_ep_t ep = { src, src_key, dst, dst_key };
344
345 res = stage_engine_move(&ep, credp, &bytes, &oerr);
346 }
347
348 /* The flush has consumed the per-user proxy; erase the resolved secret. */
349 brix_sd_ucred_wipe(&ru);
350
351 /* One unified audit line per terminal transfer (transport-agnostic). */
352 brix_xfer_finish(stage_kind_to_xfer(kind), stage_kind_dir(kind), dst_key,
353 principal, (size_t) bytes, res, (res == BRIX_XFER_OK) ? 0 : oerr, log);
354
355 if (res != BRIX_XFER_OK && oerr != 0) {
356 errno = oerr;
357 }
358 return res;
359}

361/* ---- brix_sreq_decode ---------------------------------------------------
362 *
363 * WHAT: NGX_OK with *out filled from a full-size OR legacy (pre-cred) record
364 * buffer; NGX_ERROR for any other size (corrupt).
365 *
366 * WHY: brix_sreq_t grew an appended cred; journals written before the upgrade
367 * must replay (with a zeroed cred -> service-credential flush, matching
368 * their pre-feature semantics).
369 *
370 * HOW: legacy size == offsetof(brix_sreq_t, cred) because the cred is the
371 * final member; memzero then copy exactly n bytes. */
372ngx_int_t
373brix_sreq_decode(const void *buf, size_t n, brix_sreq_t *out)
374{
375 if (n != sizeof(brix_sreq_t) && n != offsetof(brix_sreq_t, cred)) {
376 return NGX_ERROR;
377 }
378 ngx_memzero(out, sizeof(*out));
379 ngx_memcpy(out, buf, n);
380 return NGX_OK;
381}
