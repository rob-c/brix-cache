20/*
21 * Progress counters shared between the curl background thread and the
22 * nginx event-loop poll timer during 202-streaming TPC transfers.
23 * Allocated from r->pool before the thread is posted; valid until
24 * ngx_http_finalize_request is called.
25 */
26typedef struct {
27 volatile ngx_atomic_t bytes_per_stream[BRIX_TPC_MAX_STREAMS];
28 volatile ngx_atomic_t completed; /* 1 when thread is done */
29 ngx_int_t result; /* HTTP status; set before completed=1 */
30 ngx_uint_t n_streams;
31 off_t total_size; /* from HEAD; -1 if unknown */
32} tpc_ms_progress_t;
