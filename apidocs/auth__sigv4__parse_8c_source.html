29 * Format:
30 * AWS4-HMAC-SHA256 Credential=AKID/DATE/REGION/s3/aws4_request,
31 * SignedHeaders=host;x-amz-content-sha256;x-amz-date,
32 * Signature=<hex64>

35/*
36 * WHAT: Locate "<name>=" inside the Authorization header value [start,end).
37 * WHY: The header is a comma/space-separated list of Key=Value pairs
38 * (Credential=, SignedHeaders=, Signature=). We must match a *field
39 * boundary*, not an arbitrary substring — otherwise "Signature" could be
40 * falsely found inside another value. HOW: only accept a match when the
41 * byte before it is the start of the string, a space, or a comma (i.e.
42 * the start of a new pair), and the byte after the name is '='.
43 * Returns a pointer to the start of the name, or NULL if not present.
44 */
45static const char *
46sigv4_find_auth_pair(const char *start, const char *end, const char *name)
47{
48 const char *p;
49 size_t name_len;
50
51 name_len = strlen(name);
52
53 for (p = start; p < end; p++) {
54 /* Need room for "name" + at least the trailing '=' to follow. */
55 if ((size_t) (end - p) <= name_len) {
56 return NULL;
57 }
58
59 /* Enforce pair boundary: must be at list start or just after a
60 * separator, else this is a substring match inside another value. */
61 if (p != start && p[-1] != ' ' && p[-1] != ',') {
62 continue;
63 }
64
65 if (p[name_len] == '='
66 && ngx_strncmp((u_char *) p, (u_char *) name, name_len) == 0)
67 {
68 return p;
69 }
70 }
71
72 return NULL;
73}

75/*
76 * WHAT: NUL-terminated copy of [value_start,value_end) into dst.
77 * WHY: Components come straight off the wire; reject empty and over-long
78 * values up front so callers can treat dst as a trusted C string.
79 * Returns 1 on success, 0 if empty or it would not fit (with room for NUL).
80 */
81static int
82sigv4_copy_component(const char *value_start, const char *value_end,
83 char *dst, size_t dst_size)
84{
85 size_t value_len;
86
87 value_len = (size_t) (value_end - value_start);
88 if (value_len == 0 || value_len >= dst_size) {
89 return 0;
90 }
91
92 memcpy(dst, value_start, value_len);
93 dst[value_len] = '\0';
94 return 1;
95}

97/*
98 * WHAT: Find "<name>=" and copy its value (up to the next separator) into dst.
99 * HOW: value_start skips past the name and the '=' (hence name_len + 1); the
100 * value runs until the next ',', ' ', NUL, or end of the header.
101 * Returns 1 on success, 0 if the field is absent or the value does not fit.
102 */
103static int
104sigv4_extract_auth_value(const char *start, const char *end,
105 const char *name, char *dst, size_t dst_size)
106{
107 const char *field;
108 const char *value_start;
109 const char *value_end;
110 size_t name_len;
111
112 field = sigv4_find_auth_pair(start, end, name);
113 if (field == NULL) {
114 return 0;
115 }
116
117 name_len = strlen(name);
118 value_start = field + name_len + 1; /* +1 steps over the '=' */
119 value_end = value_start;
120
121 /* Value ends at the first pair separator (',' or ' '), NUL, or buffer end. */
122 while (value_end < end
123 && *value_end != ','
124 && *value_end != ' '
125 && *value_end != '\0')
126 {
127 value_end++;
128 }
129
130 return sigv4_copy_component(value_start, value_end, dst, dst_size);
131}

133/*
134 * WHAT: Split a Credential into its scope parts and store them in `out`.
135 * WHY: AWS SigV4 credential scope is exactly "AKID/DATE/REGION/s3/aws4_request".
136 * We only need the first three components; the trailing "s3/aws4_request"
137 * is implied and not validated here. HOW: find the first three '/' to
138 * bound AKID (before slash1), DATE (slash1..slash2), REGION (slash2..slash3).
139 * Anything past slash3 is the service/terminator and is ignored.
140 * Returns 1 if at least three slashes exist and every component copies; else 0.
141 */
142static int
143sigv4_parse_credential_scope(const char *credential,
144 sigv4_components_t *out)
145{
146 const char *credential_end;
147 const char *slash1;
148 const char *slash2;
149 const char *slash3;
150
151 credential_end = credential + strlen(credential);
152
153 /* Locate the first three '/' delimiters; each search starts just past the
154 * previous slash so the bounded lengths cannot read off the end. */
155 slash1 = memchr(credential, '/', (size_t) (credential_end - credential));
156 if (slash1 == NULL) {
157 return 0;
158 }
159
160 slash2 = memchr(slash1 + 1,
161 '/',
162 (size_t) (credential_end - slash1 - 1));
163 if (slash2 == NULL) {
164 return 0;
165 }
166
167 slash3 = memchr(slash2 + 1,
168 '/',
169 (size_t) (credential_end - slash2 - 1));
170 if (slash3 == NULL) {
171 return 0;
172 }
173
174 /* AKID = [start,slash1), DATE = (slash1,slash2), REGION = (slash2,slash3). */
175 if (!sigv4_copy_component(credential, slash1, out->akid,
176 sizeof(out->akid))
177 || !sigv4_copy_component(slash1 + 1, slash2, out->date,
178 sizeof(out->date))
179 || !sigv4_copy_component(slash2 + 1, slash3, out->region,
180 sizeof(out->region)))
181 {
182 return 0;
183 }
184
185 return 1;
186}

188/*
189 * WHAT: Fetch a query-string arg by name and URL-decode it into dst.
190 * WHY: Presigned-URL fields (X-Amz-Credential, X-Amz-Date, ...) arrive percent-
191 * encoded in the query string. We decode '+' as space and reject embedded
192 * NUL so callers receive a clean, trusted C string. HOW: ngx_http_arg
193 * gets the raw value; brix_http_urldecode does the bounded decode.
194 * Returns 1 on success, 0 if the arg is missing/empty or decode fails/overflows.
195 */
196static int
197sigv4_copy_arg_decoded(ngx_http_request_t *r, const char *name,
198 char *dst, size_t dst_size)
199{
200 ngx_str_t raw;
201
202 if (ngx_http_arg(r, (u_char *) name, ngx_strlen(name), &raw) != NGX_OK
203 || raw.len == 0)
204 {
205 return 0;
206 }
207
208 return brix_http_urldecode(raw.data, raw.len, dst, dst_size,
209 BRIX_URLDECODE_PLUS_TO_SPACE |
210 BRIX_URLDECODE_REJECT_NUL) == BRIX_URLDECODE_OK;
211}

213/*
214 * WHAT: Parse X-Amz-Expires (presigned URL validity, in seconds).
215 * WHY: AWS caps presigned-URL lifetime at 7 days; a zero or out-of-range value
216 * is rejected so we never honour an unbounded/garbage expiry window. HOW:
217 * strtoul with strict validation — require the whole string to be consumed
218 * (*end == '\0'), no errno, and 1 <= value <= 604800 (7*24*3600).
219 * Returns 1 and sets *out on success, 0 on any malformed/out-of-range input.
220 */
221static int
222sigv4_parse_expires(const char *value, ngx_uint_t *out)
223{
224 char *end;
225 unsigned long parsed;
226
227 errno = 0;
228 parsed = strtoul(value, &end, 10);
229 if (errno != 0 || end == value || *end != '\0'
230 || parsed == 0 || parsed > 604800)
231 {
232 return 0;
233 }
234
235 *out = (ngx_uint_t) parsed;
236 return 1;
237}

241{
242 char credential[256];
243 const char *start;
244 const char *end;
245
246 /* Require the exact scheme prefix "AWS4-HMAC-SHA256 " (17 bytes incl. the
247 * trailing space). Reject anything shorter or differently named before we
248 * attempt to scan key=value pairs. */
249 if (auth->len < 17
250 || ngx_strncasecmp(auth->data,
251 (u_char *) "AWS4-HMAC-SHA256 ", 17) != 0)
252 {
253 return 0;
254 }
255
256 /* Scan only the pair list that follows the scheme prefix. */
257 start = (char *) auth->data + 17;
258 end = (char *) auth->data + auth->len;
259

261 out->signed_hdrs,
262 sizeof(out->signed_hdrs))
263 || !sigv4_extract_auth_value(start, end, "Signature",
264 out->signature,
265 sizeof(out->signature))
266 || !sigv4_extract_auth_value(start, end, "Credential",
267 credential, sizeof(credential)))
268 {
269 return 0;
270 }
271

273 if (!sigv4_parse_credential_scope(credential, out)) {
274 return 0;
275 }

277 return 1;

280/*
281 * WHAT: Parse SigV4 components from a presigned-URL query string (no Auth header).
282 * WHY: Presigned requests carry the same Credential/SignedHeaders/Signature plus
283 * X-Amz-Date/X-Amz-Expires in the query instead of an Authorization header.
284 * Presence of X-Amz-Signature is what distinguishes a presigned request, so
285 * its absence is NGX_DECLINED ("not a presigned request"), while a present-
286 * but-malformed set of params is NGX_ERROR.
287 * HOW: Mark out->presigned, copy the (already-decoded by ngx) signature verbatim,
288 * then URL-decode the remaining fields; every required field must be present
289 * and the algorithm must be AWS4-HMAC-SHA256 — any failure is all-or-nothing.
290 * Returns NGX_OK on a complete valid set, NGX_DECLINED if not presigned at all,
291 * NGX_ERROR if presigned but malformed.
292 */
293int
294parse_presigned_authorization(ngx_http_request_t *r, sigv4_components_t *out)
295{
296 ngx_str_t sig_qs;
297 char algorithm[64];
298 char credential[256];
299 char expires[32];
300
301 /* X-Amz-Signature is the presence test: absent => this is not presigned. */
302 if (ngx_http_arg(r, (u_char *) "X-Amz-Signature", 15, &sig_qs)
303 != NGX_OK)
304 {
305 return NGX_DECLINED;
306 }
307
308 ngx_memzero(out, sizeof(*out));
309 out->presigned = 1;
310
311 if (sig_qs.len == 0
312 || brix_str_cbuf(out->signature, sizeof(out->signature), &sig_qs)
313 == NULL)
314 {
315 return NGX_ERROR;
316 }
317
318 /* All of these fields are mandatory; the short-circuit && treats any
319 * missing/over-long/badly-encoded field (or wrong algorithm) as a single
320 * malformed-request failure. */
321 if (!sigv4_copy_arg_decoded(r, "X-Amz-Algorithm",
322 algorithm, sizeof(algorithm))
323 || ngx_strcmp((u_char *) algorithm,
324 (u_char *) "AWS4-HMAC-SHA256") != 0
325 || !sigv4_copy_arg_decoded(r, "X-Amz-Credential",
326 credential, sizeof(credential))
327 || !sigv4_copy_arg_decoded(r, "X-Amz-Date",
328 out->amz_date, sizeof(out->amz_date))
329 || !sigv4_copy_arg_decoded(r, "X-Amz-Expires",
330 expires, sizeof(expires))
331 || !sigv4_copy_arg_decoded(r, "X-Amz-SignedHeaders",
332 out->signed_hdrs,
333 sizeof(out->signed_hdrs)))
334 {
335 return NGX_ERROR;
336 }
337
338 if (!sigv4_parse_credential_scope(credential, out)
339 || !sigv4_parse_expires(expires, &out->amz_expires))
340 {
341 return NGX_ERROR;
342 }
343
344 return NGX_OK;
345}
