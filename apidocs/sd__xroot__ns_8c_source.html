25/* The kXR_fattr protocol handler stores a user attribute "X" under the on-disk key
26 * "user.U.X" (BRIX_FATTR_XKEY_PFX, applied ABOVE the VFS). When we forward to
27 * ANOTHER xrootd server it re-applies the SAME mapping, so handing it the already-
28 * mapped key would double-prefix it on the origin ("user.U.user.U.X") and break
29 * direct-origin interop. So strip one "user.U." before forwarding get/set/remove,
30 * and re-add it on list — the origin then carries a single, standard "user.U.X".
31 * (Kept in sync with src/fattr/ngx_brix_fattr.h rather than #included, to avoid a
32 * backend→protocol-handler dependency. Names from other consumers — webdav locks/
33 * dead-props, s3 tags — have no "user.U." prefix and pass through unchanged.) */
34/* SD_XROOT_FATTR_PFX / _LEN live in sd_xroot_internal.h — shared with the
35 * credential-scoped wrappers (sd_xroot_ns_cred.c). */
36const char *
37sd_xroot_fattr_unmap(const char *name)
38{
39 if (strncmp(name, SD_XROOT_FATTR_PFX, SD_XROOT_FATTR_PFX_LEN) == 0) {
40 return name + SD_XROOT_FATTR_PFX_LEN;
41 }
42 return name;
43}

45/* WHAT: Report whether a fallback_deny credential must be refused outright.
46 * WHY: A fallback_deny cred whose selected kind (e.g. S3 ak/sk) this driver
47 * cannot present carries neither x509_proxy nor bearer. Falling back to
48 * the static service credential would silently serve the request under
49 * the wrong identity, so it must be refused before any session is opened.
50 * Every *_cred namespace wrapper (unlink_cred, rename_cred, opendir_cred,
51 * ...) reaches this via the shared sd_xroot_session choke point.
52 * HOW: True only when cred is non-NULL, has fallback_deny set, and carries
53 * neither a non-empty x509_proxy nor a non-empty bearer. NULL cred and
54 * creds with a presentable kind return false (allowed). */
55static int
56sd_xroot_cred_must_deny(const brix_sd_cred_t *cred)
57{
58 return cred != NULL && cred->fallback_deny
59 && (cred->x509_proxy == NULL || cred->x509_proxy[0] == '\0')
60 && (cred->bearer == NULL || cred->bearer[0] == '\0');
61}

63/* WHAT: Copy a per-user credential into a fill task so the origin bootstrap
64 * presents it at authentication instead of the static service cred.
65 * WHY: Namespace ops that carry a user credential must NOT fall back to the
66 * static service credential; this mirrors sd_xroot_origin_open.
67 * HOW: Exactly one of {x509_proxy, bearer} is non-NULL for a credential-scoped
68 * session; copy whichever is set, plus principal. Caller guarantees the
69 * task's cred_* fields are already zeroed (calloc), so a NULL cred is a
70 * no-op and the service-cred path is left unchanged. */
71static void
72sd_xroot_cred_copy(brix_cache_fill_t *t, const brix_sd_cred_t *cred)
73{
74 if (cred == NULL) {
75 return;
76 }
77 if (cred->x509_proxy != NULL && cred->x509_proxy[0] != '\0') {
78 ngx_cpystrn((u_char *) t->cred_x509_proxy, (u_char *) cred->x509_proxy,
79 sizeof(t->cred_x509_proxy));
80 }
81 if (cred->bearer != NULL && cred->bearer[0] != '\0') {
82 ngx_cpystrn((u_char *) t->cred_bearer, (u_char *) cred->bearer,
83 sizeof(t->cred_bearer));
84 }
85 if (cred->principal != NULL) {
86 ngx_cpystrn((u_char *) t->cred_principal,
87 (u_char *) cred->principal, sizeof(t->cred_principal));
88 }
89}

91/* Connect + bootstrap a fresh origin session (no file open) for a path-based op.
92 * When cred is non-NULL the bootstrap uses the per-user credential instead of
93 * the static service credential. Supports both x509 proxies (cred->x509_proxy)
94 * and WLCG bearer tokens (cred->bearer); they are mutually exclusive per the
95 * brix_sd_cred_t contract. NULL cred → service credential / anonymous.
96 * On success fills *oc + *t_out (caller closes oc + frees t); returns -1 with
97 * *err_out on failure.
98 *
99 * WHAT: Refuse an unpresentable fallback_deny cred, then allocate and wire a
100 * fill task, copying any per-user credential, and connect+bootstrap.
101 * WHY: This function opens its own origin session independent of
102 * sd_xroot_open_common, so the wrong-kind cred leak is reachable via
103 * every *_cred namespace wrapper unless checked at this shared point.
104 * HOW: sd_xroot_cred_must_deny gates unusable creds up front; calloc zeroes
105 * all cred_* fields so sd_xroot_cred_copy is a no-op for NULL cred.
106 * Prototype in sd_xroot_internal.h — shared with the credential-scoped
107 * wrappers (sd_xroot_ns_cred.c) and the directory-listing path
108 * (sd_xroot_ns_dir.c). */
109int
110sd_xroot_session(ngx_stream_brix_srv_conf_t *conf,
111 const brix_sd_cred_t *cred,
112 brix_cache_origin_conn_t *oc, brix_cache_fill_t **t_out, int *err_out)
113{
114 brix_cache_fill_t *t;
115
116 if (sd_xroot_cred_must_deny(cred)) {
117 if (err_out) { *err_out = EACCES; }
118 errno = EACCES;
119 return -1;
120 }
121
122 t = calloc(1, sizeof(*t));
123 if (t == NULL) {
124 if (err_out) { *err_out = ENOMEM; }
125 return -1;
126 }
127 oc->fd = -1;
128 t->conf = conf;
129
130 sd_xroot_cred_copy(t, cred);
131
132 if (brix_cache_origin_connect(t, oc) != 0
133 || brix_cache_origin_bootstrap(t, oc) != 0)
134 {
135 if (err_out) { *err_out = sd_xroot_errno(t); }
136 brix_cache_origin_close(oc);
137 free(t);
138 return -1;
139 }
140 *t_out = t;
141 return 0;
142}

164ssize_t
165sd_xroot_listxattr(brix_sd_instance_t *inst, const char *path,
166 void *buf, size_t cap)
167{
168 sd_xroot_inst_state *is = inst->state;
169 brix_cache_origin_conn_t oc;
170 brix_cache_fill_t *t;
171 char *raw;
172 const size_t rawcap = 65536;
173 ssize_t n;
174 size_t out = 0, i;
175 int e = 0;
176
177 raw = malloc(rawcap);
178 if (raw == NULL) { errno = ENOMEM; return -1; }
179 if (sd_xroot_session(is->conf, NULL, &oc, &t, &e) != 0) {
180 free(raw); errno = e; return -1;
181 }
182 /* The origin returns its user attrs as a NUL-separated list of CLIENT names
183 * (its own "user.U." stripped). Re-add the "user.U." prefix to each so the
184 * kXR_fattr list handler — which keeps "user.U.*" keys — recognizes them. */
185 n = brix_cache_origin_listfattr(t, &oc, path, raw, rawcap);
186 e = errno;
187 brix_cache_origin_close(&oc);
188 free(t);
189 if (n < 0) { free(raw); errno = e; return -1; }
190
191 for (i = 0; i < (size_t) n; ) {
192 size_t nl = strnlen(raw + i, (size_t) n - i);
193
194 if (nl == 0) { i += 1; continue; } /* skip an empty entry */
195 if (buf != NULL && cap > 0) {
196 if (out + SD_XROOT_FATTR_PFX_LEN + nl + 1 > cap) {
197 free(raw); errno = ERANGE; return -1;
198 }
199 ngx_memcpy((char *) buf + out, SD_XROOT_FATTR_PFX,
200 SD_XROOT_FATTR_PFX_LEN);
201 ngx_memcpy((char *) buf + out + SD_XROOT_FATTR_PFX_LEN, raw + i, nl);
202 ((char *) buf)[out + SD_XROOT_FATTR_PFX_LEN + nl] = '\0';
203 }
204 out += SD_XROOT_FATTR_PFX_LEN + nl + 1;
205 i += nl + 1;
206 }
207 free(raw);
208 return (ssize_t) out;
209}

292/* Copy src→dst byte stream on an open session (read each chunk from src_fh, write
293 * to dst_fh), then truncate+sync dst. Returns NGX_OK + *bytes_out, or NGX_ERROR.
294 * Prototype in sd_xroot_internal.h — shared with the credential-scoped
295 * server_copy_cred wrapper (sd_xroot_ns_cred.c). */
296ngx_int_t
297sd_xroot_copy_body(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
298 const u_char *src_fh, const u_char *dst_fh, off_t *bytes_out)
299{
300 const size_t cap = 1u << 20;
301 u_char *buf = malloc(cap);
302 off_t off = 0;
303
304 if (buf == NULL) { errno = ENOMEM; return NGX_ERROR; }
305
306 for (;;) {
307 brix_cache_sink_t sink;
308 brix_cache_read_range_t rng;
309
310 ngx_memzero(&sink, sizeof(sink));
311 sink.fd = -1;
312 sink.mem = buf;
313 sink.mem_cap = cap;
314
315 ngx_memzero(&rng, sizeof(rng));
316 rng.read_off = (uint64_t) off;
317 rng.want = cap;
318
319 if (brix_cache_origin_read_chunk(t, oc, src_fh, &sink, &rng) != 0) {
320 free(buf); errno = EIO; return NGX_ERROR;
321 }
322 if (rng.got == 0) {
323 break;
324 }
325 if (brix_cache_origin_write_chunk(t, oc, dst_fh, (uint64_t) off, buf,
326 rng.got) != 0)
327 {
328 free(buf); errno = EIO; return NGX_ERROR;
329 }
330 off += (off_t) rng.got;
331 if (rng.got < cap) {
332 break; /* short read = EOF */
333 }
334 }
335 free(buf);
336
337 if (brix_cache_origin_truncate(t, oc, dst_fh, (uint64_t) off) != 0
338 || brix_cache_origin_sync(t, oc, dst_fh) != 0)
339 {
340 errno = EIO;
341 return NGX_ERROR;
342 }
343 if (bytes_out) { *bytes_out = off; }
344 return NGX_OK;
345}

347/* Server-side copy: the gateway reads src and writes dst on the origin (no client

349ngx_int_t
350sd_xroot_server_copy(brix_sd_instance_t *inst, const char *src,

352{

354 brix_cache_origin_conn_t oc;
355 brix_cache_fill_t *t;

357 ngx_int_t rc;
358 int e = 0;
359
360 if (sd_xroot_session(is->conf, NULL, &oc, &t, &e) != 0) { errno = e; return NGX_ERROR; }
361

363 if (brix_cache_origin_open(t, &oc, src_fh) != 0) {

366 }
367 if (brix_cache_origin_open_write(t, &oc, dst, 0644, dst_fh) != 0) {

375 brix_cache_origin_close_file(&oc, dst_fh);

377 brix_cache_origin_close(&oc);
378 free(t);
379 errno = e;
380 return rc;
381}
