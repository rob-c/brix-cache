25/* The kXR_fattr protocol handler stores a user attribute "X" under the on-disk key
26 * "user.U.X" (BRIX_FATTR_XKEY_PFX, applied ABOVE the VFS). When we forward to
27 * ANOTHER xrootd server it re-applies the SAME mapping, so handing it the already-
28 * mapped key would double-prefix it on the origin ("user.U.user.U.X") and break
29 * direct-origin interop. So strip one "user.U." before forwarding get/set/remove,
30 * and re-add it on list — the origin then carries a single, standard "user.U.X".
31 * (Kept in sync with src/fattr/ngx_brix_fattr.h rather than #included, to avoid a
32 * backend→protocol-handler dependency. Names from other consumers — webdav locks/
33 * dead-props, s3 tags — have no "user.U." prefix and pass through unchanged.) */
34#define SD_XROOT_FATTR_PFX "user.U."
35#define SD_XROOT_FATTR_PFX_LEN 7
36
37static const char *
38sd_xroot_fattr_unmap(const char *name)
39{
40 if (strncmp(name, SD_XROOT_FATTR_PFX, SD_XROOT_FATTR_PFX_LEN) == 0) {
41 return name + SD_XROOT_FATTR_PFX_LEN;
42 }
43 return name;
44}

46/* WHAT: Report whether a fallback_deny credential must be refused outright.
47 * WHY: A fallback_deny cred whose selected kind (e.g. S3 ak/sk) this driver
48 * cannot present carries neither x509_proxy nor bearer. Falling back to
49 * the static service credential would silently serve the request under
50 * the wrong identity, so it must be refused before any session is opened.
51 * Every *_cred namespace wrapper (unlink_cred, rename_cred, opendir_cred,
52 * ...) reaches this via the shared sd_xroot_session choke point.
53 * HOW: True only when cred is non-NULL, has fallback_deny set, and carries
54 * neither a non-empty x509_proxy nor a non-empty bearer. NULL cred and
55 * creds with a presentable kind return false (allowed). */
56static int
57sd_xroot_cred_must_deny(const brix_sd_cred_t *cred)
58{
59 return cred != NULL && cred->fallback_deny
60 && (cred->x509_proxy == NULL || cred->x509_proxy[0] == '\0')
61 && (cred->bearer == NULL || cred->bearer[0] == '\0');
62}

64/* WHAT: Copy a per-user credential into a fill task so the origin bootstrap
65 * presents it at authentication instead of the static service cred.
66 * WHY: Namespace ops that carry a user credential must NOT fall back to the
67 * static service credential; this mirrors sd_xroot_origin_open.
68 * HOW: Exactly one of {x509_proxy, bearer} is non-NULL for a credential-scoped
69 * session; copy whichever is set, plus principal. Caller guarantees the
70 * task's cred_* fields are already zeroed (calloc), so a NULL cred is a
71 * no-op and the service-cred path is left unchanged. */
72static void
73sd_xroot_cred_copy(brix_cache_fill_t *t, const brix_sd_cred_t *cred)
74{
75 if (cred == NULL) {
76 return;
77 }
78 if (cred->x509_proxy != NULL && cred->x509_proxy[0] != '\0') {
79 ngx_cpystrn((u_char *) t->cred_x509_proxy, (u_char *) cred->x509_proxy,
80 sizeof(t->cred_x509_proxy));
81 }
82 if (cred->bearer != NULL && cred->bearer[0] != '\0') {
83 ngx_cpystrn((u_char *) t->cred_bearer, (u_char *) cred->bearer,
84 sizeof(t->cred_bearer));
85 }
86 if (cred->principal != NULL) {
87 ngx_cpystrn((u_char *) t->cred_principal,
88 (u_char *) cred->principal, sizeof(t->cred_principal));
89 }
90}

92/* Connect + bootstrap a fresh origin session (no file open) for a path-based op.
93 * When cred is non-NULL the bootstrap uses the per-user credential instead of
94 * the static service credential. Supports both x509 proxies (cred->x509_proxy)
95 * and WLCG bearer tokens (cred->bearer); they are mutually exclusive per the
96 * brix_sd_cred_t contract. NULL cred → service credential / anonymous.
97 * On success fills *oc + *t_out (caller closes oc + frees t); returns -1 with
98 * *err_out on failure.
99 *
100 * WHAT: Refuse an unpresentable fallback_deny cred, then allocate and wire a
101 * fill task, copying any per-user credential, and connect+bootstrap.
102 * WHY: This function opens its own origin session independent of
103 * sd_xroot_open_common, so the wrong-kind cred leak is reachable via
104 * every *_cred namespace wrapper unless checked at this shared point.
105 * HOW: sd_xroot_cred_must_deny gates unusable creds up front; calloc zeroes
106 * all cred_* fields so sd_xroot_cred_copy is a no-op for NULL cred. */
107static int
108sd_xroot_session(ngx_stream_brix_srv_conf_t *conf,
109 const brix_sd_cred_t *cred,
110 brix_cache_origin_conn_t *oc, brix_cache_fill_t **t_out, int *err_out)
111{
112 brix_cache_fill_t *t;
113
114 if (sd_xroot_cred_must_deny(cred)) {
115 if (err_out) { *err_out = EACCES; }
116 errno = EACCES;
117 return -1;
118 }
119
120 t = calloc(1, sizeof(*t));
121 if (t == NULL) {
122 if (err_out) { *err_out = ENOMEM; }
123 return -1;
124 }
125 oc->fd = -1;
126 t->conf = conf;
127
128 sd_xroot_cred_copy(t, cred);
129
130 if (brix_cache_origin_connect(t, oc) != 0
131 || brix_cache_origin_bootstrap(t, oc) != 0)
132 {
133 if (err_out) { *err_out = sd_xroot_errno(t); }
134 brix_cache_origin_close(oc);
135 free(t);
136 return -1;
137 }
138 *t_out = t;
139 return 0;
140}

162ssize_t
163sd_xroot_listxattr(brix_sd_instance_t *inst, const char *path,
164 void *buf, size_t cap)
165{
166 sd_xroot_inst_state *is = inst->state;
167 brix_cache_origin_conn_t oc;
168 brix_cache_fill_t *t;
169 char *raw;
170 const size_t rawcap = 65536;
171 ssize_t n;
172 size_t out = 0, i;
173 int e = 0;
174
175 raw = malloc(rawcap);
176 if (raw == NULL) { errno = ENOMEM; return -1; }
177 if (sd_xroot_session(is->conf, NULL, &oc, &t, &e) != 0) {
178 free(raw); errno = e; return -1;
179 }
180 /* The origin returns its user attrs as a NUL-separated list of CLIENT names
181 * (its own "user.U." stripped). Re-add the "user.U." prefix to each so the
182 * kXR_fattr list handler — which keeps "user.U.*" keys — recognizes them. */
183 n = brix_cache_origin_listfattr(t, &oc, path, raw, rawcap);
184 e = errno;
185 brix_cache_origin_close(&oc);
186 free(t);
187 if (n < 0) { free(raw); errno = e; return -1; }
188
189 for (i = 0; i < (size_t) n; ) {
190 size_t nl = strnlen(raw + i, (size_t) n - i);
191
192 if (nl == 0) { i += 1; continue; } /* skip an empty entry */
193 if (buf != NULL && cap > 0) {
194 if (out + SD_XROOT_FATTR_PFX_LEN + nl + 1 > cap) {
195 free(raw); errno = ERANGE; return -1;
196 }
197 ngx_memcpy((char *) buf + out, SD_XROOT_FATTR_PFX,
198 SD_XROOT_FATTR_PFX_LEN);
199 ngx_memcpy((char *) buf + out + SD_XROOT_FATTR_PFX_LEN, raw + i, nl);
200 ((char *) buf)[out + SD_XROOT_FATTR_PFX_LEN + nl] = '\0';
201 }
202 out += SD_XROOT_FATTR_PFX_LEN + nl + 1;
203 i += nl + 1;
204 }
205 free(raw);
206 return (ssize_t) out;
207}

290/* Copy src→dst byte stream on an open session (read each chunk from src_fh, write
291 * to dst_fh), then truncate+sync dst. Returns NGX_OK + *bytes_out, or NGX_ERROR. */
292static ngx_int_t
293sd_xroot_copy_body(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
294 const u_char *src_fh, const u_char *dst_fh, off_t *bytes_out)
295{
296 const size_t cap = 1u << 20;
297 u_char *buf = malloc(cap);
298 off_t off = 0;
299
300 if (buf == NULL) { errno = ENOMEM; return NGX_ERROR; }
301
302 for (;;) {
303 brix_cache_sink_t sink;
304 brix_cache_read_range_t rng;
305
306 ngx_memzero(&sink, sizeof(sink));
307 sink.fd = -1;
308 sink.mem = buf;
309 sink.mem_cap = cap;
310
311 ngx_memzero(&rng, sizeof(rng));
312 rng.read_off = (uint64_t) off;
313 rng.want = cap;
314
315 if (brix_cache_origin_read_chunk(t, oc, src_fh, &sink, &rng) != 0) {
316 free(buf); errno = EIO; return NGX_ERROR;
317 }
318 if (rng.got == 0) {
319 break;
320 }
321 if (brix_cache_origin_write_chunk(t, oc, dst_fh, (uint64_t) off, buf,
322 rng.got) != 0)
323 {
324 free(buf); errno = EIO; return NGX_ERROR;
325 }
326 off += (off_t) rng.got;
327 if (rng.got < cap) {
328 break; /* short read = EOF */
329 }
330 }
331 free(buf);
332
333 if (brix_cache_origin_truncate(t, oc, dst_fh, (uint64_t) off) != 0
334 || brix_cache_origin_sync(t, oc, dst_fh) != 0)
335 {
336 errno = EIO;
337 return NGX_ERROR;
338 }
339 if (bytes_out) { *bytes_out = off; }
340 return NGX_OK;
341}

343/* Server-side copy: the gateway reads src and writes dst on the origin (no client
344 * round-trip). Not zero-copy on the origin (no remote TPC) — a read+write relay. */
345ngx_int_t
346sd_xroot_server_copy(brix_sd_instance_t *inst, const char *src,
347 const char *dst, off_t *bytes_out)
348{
349 sd_xroot_inst_state *is = inst->state;
350 brix_cache_origin_conn_t oc;
351 brix_cache_fill_t *t;
352 u_char src_fh[XRD_FHANDLE_LEN], dst_fh[XRD_FHANDLE_LEN];
353 ngx_int_t rc;
354 int e = 0;
355
356 if (sd_xroot_session(is->conf, NULL, &oc, &t, &e) != 0) { errno = e; return NGX_ERROR; }
357
358 ngx_cpystrn((u_char *) t->clean_path, (u_char *) src, sizeof(t->clean_path));
359 if (brix_cache_origin_open(t, &oc, src_fh) != 0) {
360 e = sd_xroot_errno(t);
361 brix_cache_origin_close(&oc); free(t); errno = e; return NGX_ERROR;
362 }
363 if (brix_cache_origin_open_write(t, &oc, dst, 0644, dst_fh) != 0) {
364 e = sd_xroot_errno(t);
365 brix_cache_origin_close_file(&oc, src_fh);
366 brix_cache_origin_close(&oc); free(t); errno = e; return NGX_ERROR;
367 }
368
369 rc = sd_xroot_copy_body(t, &oc, src_fh, dst_fh, bytes_out);
370 e = errno;
371 brix_cache_origin_close_file(&oc, dst_fh);
372 brix_cache_origin_close_file(&oc, src_fh);
373 brix_cache_origin_close(&oc);
374 free(t);
375 errno = e;
376 return rc;
377}

379/* ---- credential-scoped namespace wrappers (Phase 2 Task 1) -----------------
380 *
381 * WHAT: One wrapper per supported ns op that opens a per-user origin session
382 * (via sd_xroot_session with a non-NULL cred) instead of the anonymous /
383 * service-credential session the plain slots use.
384 *
385 * WHY: A deny-mode request whose data-plane credential gate fires must also
386 * present the user proxy for every pre-flight probe and metadata mutation
387 * (unlink, rename, copy, xattr). Without these wrappers the probe stat
388 * still runs under the static service credential even when the protocol
389 * handler correctly denied the operation.
390 *
391 * HOW: Each wrapper delegates the operation body verbatim to the matching plain
392 * op helper but passes the cred to sd_xroot_session so the bootstrap uses
393 * the per-user proxy. The plain ops now call sd_xroot_session(…, NULL, …);
394 * these wrappers call sd_xroot_session(…, cred, …) — the only difference.
395 * stat_cred is implemented in sd_xroot.c (alongside sd_xroot_stat, which
396 * reuses sd_xroot_origin_open — a file-private type). */

584/* ---- directory listing (kXR_dirlist, fetch-all-then-iterate) --------------
585 *
586 * WHAT: opendir issues ONE kXR_dirlist against the origin, buffers every
587 * name into a brix_cache_dirlist_t held as the brix_sd_dir_t's driver-
588 * private state, then closes the origin session immediately (mirrors
589 * every other ns op in this file — no op here holds a live origin
590 * connection across separate VFS calls). readdir yields one buffered
591 * name per call; closedir frees the buffer.
592 * WHY: brix_sd_dir_t iteration (opendir/readdir/closedir) is 3 separate VFS
593 * calls with no guarantee they run back-to-back on the same thread, so
594 * a streamed/session-held approach would need to keep an origin TCP
595 * connection open across arbitrary VFS-call gaps — fragile and unlike
596 * every other sd_xroot op. Fetch-all-then-iterate trades a larger
597 * up-front read for a stateless, connection-free readdir/closedir.
598 * HOW: opendir_common does the session + dirlist fetch + dir handle alloc;
599 * opendir/opendir_cred differ only in which credential (NULL vs the
600 * caller's) sd_xroot_session presents at the origin. readdir walks the
601 * buffered brix_cache_dirlist_t by index; closedir frees it.
602 */
603
604/* Driver-private dir state: the fetched name buffer plus a read cursor. */
605typedef struct {
606 brix_cache_dirlist_t dl; /* heap-owned name array (see cache_internal.h) */
607 size_t next; /* index of the next name readdir will yield */
608} sd_xroot_dir_state;
609
610/* sd_xroot_opendir_common — shared body for opendir/opendir_cred: open a
611 * session under `cred` (NULL ⇒ service credential), fetch the whole listing,
612 * close the session, and wrap the result in a brix_sd_dir_t. Returns the dir
613 * handle, or NULL with *err_out set. */
614static brix_sd_dir_t *
615sd_xroot_opendir_common(brix_sd_instance_t *inst, const char *path,
616 const brix_sd_cred_t *cred, int *err_out)
617{
618 sd_xroot_inst_state *is = inst->state;
619 brix_cache_origin_conn_t oc;
620 brix_cache_fill_t *t;
621 sd_xroot_dir_state *ds;
622 brix_sd_dir_t *dir;
623 int e = 0;
624
625 if (sd_xroot_session(is->conf, cred, &oc, &t, &e) != 0) {
626 if (err_out) { *err_out = e; }
627 return NULL;
628 }
629
630 ds = calloc(1, sizeof(*ds));
631 if (ds == NULL) {
632 brix_cache_origin_close(&oc);
633 free(t);
634 if (err_out) { *err_out = ENOMEM; }
635 return NULL;
636 }
637 brix_cache_dirlist_init(&ds->dl);
638
639 if (brix_cache_origin_dirlist(t, &oc, path, &ds->dl) != 0) {
640 e = sd_xroot_errno(t);
641 brix_cache_dirlist_free(&ds->dl);
642 free(ds);
643 brix_cache_origin_close(&oc);
644 free(t);
645 if (err_out) { *err_out = e; }
646 return NULL;
647 }
648
649 /* The listing is fully buffered — the origin session is no longer
650 * needed for readdir/closedir (fetch-all-then-iterate, see above). */
651 brix_cache_origin_close(&oc);
652 free(t);
653
654 dir = calloc(1, sizeof(*dir));
655 if (dir == NULL) {
656 brix_cache_dirlist_free(&ds->dl);
657 free(ds);
658 if (err_out) { *err_out = ENOMEM; }
659 return NULL;
660 }
661 dir->inst = inst;
662 dir->state = ds;
663 return dir;
664}

682/* readdir — vtable readdir slot: yield the next buffered name, bound-copied
683 * into out->name (255 chars + NUL — brix_sd_dirent_t's fixed field). Returns
684 * NGX_OK (out filled), NGX_DONE (end of the buffered listing — matches the
685 * POSIX driver's readdir contract that brix_vfs_readdir relies on), or
686 * NGX_ERROR (malformed state; errno set). */
687ngx_int_t
688sd_xroot_readdir(brix_sd_dir_t *d, brix_sd_dirent_t *out)
689{
690 sd_xroot_dir_state *ds = d->state;
691
692 if (ds == NULL) {
693 errno = EINVAL;
694 return NGX_ERROR;
695 }
696 if (ds->next >= ds->dl.count) {
697 return NGX_DONE;
698 }
699 ngx_cpystrn((u_char *) out->name, (u_char *) ds->dl.names[ds->next],
700 sizeof(out->name));
701 ds->next++;
702 return NGX_OK;
703}
704
705/* closedir — vtable closedir slot: free the buffered name array, the dir
706 * state, and the brix_sd_dir_t handle itself. No origin session to close
707 * (opendir already released it). The VFS (brix_vfs_closedir, vfs_dir.c)
708 * calls driver->closedir(dh->sd_dir) but never frees dh->sd_dir itself, so
709 * — unlike sd_posix's pool-allocated dir handle, which the pool reclaims —
710 * this heap-allocated handle (calloc'd in sd_xroot_opendir_common) MUST be
711 * freed here or every dirlist leaks one brix_sd_dir_t. NULL-safe (both on
712 * `d` and on `d->state`) so a double-close is harmless. */
713ngx_int_t
714sd_xroot_closedir(brix_sd_dir_t *d)
715{
716 sd_xroot_dir_state *ds;
717
718 if (d == NULL) {
719 return NGX_OK;
720 }
721 ds = d->state;
722 if (ds != NULL) {
723 brix_cache_dirlist_free(&ds->dl);
724 free(ds);
725 d->state = NULL;
726 }
727 free(d);
728 return NGX_OK;
729}
