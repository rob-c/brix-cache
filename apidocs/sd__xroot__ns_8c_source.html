25/* The kXR_fattr protocol handler stores a user attribute "X" under the on-disk key
26 * "user.U.X" (BRIX_FATTR_XKEY_PFX, applied ABOVE the VFS). When we forward to
27 * ANOTHER xrootd server it re-applies the SAME mapping, so handing it the already-
28 * mapped key would double-prefix it on the origin ("user.U.user.U.X") and break
29 * direct-origin interop. So strip one "user.U." before forwarding get/set/remove,
30 * and re-add it on list — the origin then carries a single, standard "user.U.X".
31 * (Kept in sync with src/fattr/ngx_brix_fattr.h rather than #included, to avoid a
32 * backend→protocol-handler dependency. Names from other consumers — webdav locks/
33 * dead-props, s3 tags — have no "user.U." prefix and pass through unchanged.) */
34#define SD_XROOT_FATTR_PFX "user.U."
35#define SD_XROOT_FATTR_PFX_LEN 7
36
37static const char *
38sd_xroot_fattr_unmap(const char *name)
39{
40 if (strncmp(name, SD_XROOT_FATTR_PFX, SD_XROOT_FATTR_PFX_LEN) == 0) {
41 return name + SD_XROOT_FATTR_PFX_LEN;
42 }
43 return name;
44}

46/* Connect + bootstrap a fresh anonymous origin session (no file open) for a
47 * path-based op. On success fills *oc + *t_out (caller closes oc + frees t);
48 * returns -1 with *err_out on failure. */
49static int
50sd_xroot_session(ngx_stream_brix_srv_conf_t *conf,
51 brix_cache_origin_conn_t *oc, brix_cache_fill_t **t_out, int *err_out)
52{
53 brix_cache_fill_t *t = calloc(1, sizeof(*t));
54
55 if (t == NULL) {
56 if (err_out) { *err_out = ENOMEM; }
57 return -1;
58 }
59 oc->fd = -1;
60 t->conf = conf;
61 if (brix_cache_origin_connect(t, oc) != 0
62 || brix_cache_origin_bootstrap(t, oc) != 0)
63 {
64 if (err_out) { *err_out = sd_xroot_errno(t); }
65 brix_cache_origin_close(oc);
66 free(t);
67 return -1;
68 }
69 *t_out = t;
70 return 0;
71}

93ssize_t
94sd_xroot_listxattr(brix_sd_instance_t *inst, const char *path,
95 void *buf, size_t cap)
96{
97 sd_xroot_inst_state *is = inst->state;
98 brix_cache_origin_conn_t oc;
99 brix_cache_fill_t *t;
100 char *raw;
101 const size_t rawcap = 65536;
102 ssize_t n;
103 size_t out = 0, i;
104 int e = 0;
105
106 raw = malloc(rawcap);
107 if (raw == NULL) { errno = ENOMEM; return -1; }
108 if (sd_xroot_session(is->conf, &oc, &t, &e) != 0) {
109 free(raw); errno = e; return -1;
110 }
111 /* The origin returns its user attrs as a NUL-separated list of CLIENT names
112 * (its own "user.U." stripped). Re-add the "user.U." prefix to each so the
113 * kXR_fattr list handler — which keeps "user.U.*" keys — recognizes them. */
114 n = brix_cache_origin_listfattr(t, &oc, path, raw, rawcap);
115 e = errno;
116 brix_cache_origin_close(&oc);
117 free(t);
118 if (n < 0) { free(raw); errno = e; return -1; }
119
120 for (i = 0; i < (size_t) n; ) {
121 size_t nl = strnlen(raw + i, (size_t) n - i);
122
123 if (nl == 0) { i += 1; continue; } /* skip an empty entry */
124 if (buf != NULL && cap > 0) {
125 if (out + SD_XROOT_FATTR_PFX_LEN + nl + 1 > cap) {
126 free(raw); errno = ERANGE; return -1;
127 }
128 ngx_memcpy((char *) buf + out, SD_XROOT_FATTR_PFX,
129 SD_XROOT_FATTR_PFX_LEN);
130 ngx_memcpy((char *) buf + out + SD_XROOT_FATTR_PFX_LEN, raw + i, nl);
131 ((char *) buf)[out + SD_XROOT_FATTR_PFX_LEN + nl] = '\0';
132 }
133 out += SD_XROOT_FATTR_PFX_LEN + nl + 1;
134 i += nl + 1;
135 }
136 free(raw);
137 return (ssize_t) out;
138}

221/* Copy src→dst byte stream on an open session (read each chunk from src_fh, write
222 * to dst_fh), then truncate+sync dst. Returns NGX_OK + *bytes_out, or NGX_ERROR. */
223static ngx_int_t
224sd_xroot_copy_body(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
225 const u_char *src_fh, const u_char *dst_fh, off_t *bytes_out)
226{
227 const size_t cap = 1u << 20;
228 u_char *buf = malloc(cap);
229 off_t off = 0;
230
231 if (buf == NULL) { errno = ENOMEM; return NGX_ERROR; }
232
233 for (;;) {
234 brix_cache_sink_t sink;
235 size_t got = 0;
236
237 ngx_memzero(&sink, sizeof(sink));
238 sink.fd = -1;
239 sink.mem = buf;
240 sink.mem_cap = cap;
241 if (brix_cache_origin_read_chunk(t, oc, src_fh, &sink, (uint64_t) off,
242 0, cap, &got) != 0)
243 {
244 free(buf); errno = EIO; return NGX_ERROR;
245 }
246 if (got == 0) {
247 break;
248 }
249 if (brix_cache_origin_write_chunk(t, oc, dst_fh, (uint64_t) off, buf,
250 got) != 0)
251 {
252 free(buf); errno = EIO; return NGX_ERROR;
253 }
254 off += (off_t) got;
255 if (got < cap) {
256 break; /* short read = EOF */
257 }
258 }
259 free(buf);
260
261 if (brix_cache_origin_truncate(t, oc, dst_fh, (uint64_t) off) != 0
262 || brix_cache_origin_sync(t, oc, dst_fh) != 0)
263 {
264 errno = EIO;
265 return NGX_ERROR;
266 }
267 if (bytes_out) { *bytes_out = off; }
268 return NGX_OK;
269}

271/* Server-side copy: the gateway reads src and writes dst on the origin (no client
272 * round-trip). Not zero-copy on the origin (no remote TPC) — a read+write relay. */
273ngx_int_t
274sd_xroot_server_copy(brix_sd_instance_t *inst, const char *src,
275 const char *dst, off_t *bytes_out)
276{
277 sd_xroot_inst_state *is = inst->state;
278 brix_cache_origin_conn_t oc;
279 brix_cache_fill_t *t;
280 u_char src_fh[XRD_FHANDLE_LEN], dst_fh[XRD_FHANDLE_LEN];
281 ngx_int_t rc;
282 int e = 0;
283
284 if (sd_xroot_session(is->conf, &oc, &t, &e) != 0) { errno = e; return NGX_ERROR; }
285
286 ngx_cpystrn((u_char *) t->clean_path, (u_char *) src, sizeof(t->clean_path));
287 if (brix_cache_origin_open(t, &oc, src_fh) != 0) {
288 e = sd_xroot_errno(t);
289 brix_cache_origin_close(&oc); free(t); errno = e; return NGX_ERROR;
290 }
291 if (brix_cache_origin_open_write(t, &oc, dst, 0644, dst_fh) != 0) {
292 e = sd_xroot_errno(t);
293 brix_cache_origin_close_file(&oc, src_fh);
294 brix_cache_origin_close(&oc); free(t); errno = e; return NGX_ERROR;
295 }
296
297 rc = sd_xroot_copy_body(t, &oc, src_fh, dst_fh, bytes_out);
298 e = errno;
299 brix_cache_origin_close_file(&oc, dst_fh);
300 brix_cache_origin_close_file(&oc, src_fh);
301 brix_cache_origin_close(&oc);
302 free(t);
303 errno = e;
304 return rc;
305}
