1/*
2 * metrics.c - S3-compatible endpoint request/result accounting helpers.
3 *
4 * WHAT: Maps HTTP method codes to XRootD metric slot enums, increments request
5 * and response counters by method+status-class buckets, and converts handler
6 * return codes to HTTP status for metric recording. Provides six public functions:
7 * s3_metrics_method_slot (HTTP→metric enum), s3_metrics_request_method (request counter),
8 * s3_metrics_response_status (response counter by class), s3_metrics_response_method
9 * (handler RC → status → response counter), s3_metrics_return_method (counter + return),
10 * and s3_metrics_finalize_request_method (counter + ngx_http_finalize_request).
11 *
12 * WHY: S3-compatible endpoints require separate metric buckets from WebDAV/XRootD stream.
13 * Metrics track requests_total per-method and responses_total per-method×status-class.
14 * These helpers centralize method-to-slot mapping, status-class extraction via
15 * brix_http_status_class(), and handler-RC-to-HTTP-status conversion — ensuring all
16 * S3 handlers (get.c, put.c, list.c, multipart.c) use consistent metric accounting.
17 * NGX_DONE handling: async PUT body reads defer final response counting to callbacks.
18 *
19 * HOW: s3_metrics_method_slot() maps r->method enum values to BRIX_S3_METHOD_* constants —
20 * GET→GET, HEAD→HEAD, PUT→PUT, DELETE→DELETE, POST→POST, everything else→OTHER.
21 * s3_metrics_request_method() clamps method_slot to BRIX_S3_NMETHODS, increments
22 * requests_total[slot] via BRIX_S3_METRIC_INC(). s3_metrics_response_status() same clamp,
23 * extracts status_class from http_status via brix_http_status_class(), increments
24 * responses_total[slot][status_class]. s3_metrics_response_method() skips NGX_DONE (deferred),
25 * converts handler RC to HTTP status: ERROR→500, >=NGX_HTTP_SPECIAL_RESPONSE→RC value,
26 * r->headers_out.status non-zero→that value, else→200. Calls s3_metrics_response_status().
27 * s3_metrics_return_method() delegates response counting then returns original handler_rc.
28 * s3_metrics_finalize_request_method() same + calls ngx_http_finalize_request(r, handler_rc).
29 */

36static brix_metric_op_t
37s3_unified_op(ngx_uint_t method_slot)
38{
39 switch (method_slot) {
40 case BRIX_S3_METHOD_GET:
41 return BRIX_METRIC_OP_READ;
42 case BRIX_S3_METHOD_HEAD:
43 return BRIX_METRIC_OP_STAT;
44 case BRIX_S3_METHOD_PUT:
45 case BRIX_S3_METHOD_POST:
46 return BRIX_METRIC_OP_WRITE;
47 case BRIX_S3_METHOD_DELETE:
48 return BRIX_METRIC_OP_DELETE;
49 case BRIX_S3_METHOD_LIST:
50 return BRIX_METRIC_OP_DIRLIST;
51 default:
52 return BRIX_METRIC_OP_STAT;
53 }
54}

67void
68s3_metrics_request_method(ngx_uint_t method_slot)
69{
70 if (method_slot >= BRIX_S3_NMETHODS) {
71 method_slot = BRIX_S3_METHOD_OTHER;
72 }
73
74 BRIX_S3_METRIC_INC(requests_total[method_slot]);

77void
78s3_metrics_response_status(ngx_uint_t method_slot, ngx_uint_t http_status)
79{
80 ngx_uint_t status_class;
81
82 if (method_slot >= BRIX_S3_NMETHODS) {
83 method_slot = BRIX_S3_METHOD_OTHER;
84 }
85
86 status_class = brix_http_status_class(http_status);
87 BRIX_S3_METRIC_INC(responses_total[method_slot][status_class]);
88 brix_metric_op_done(BRIX_PROTO_S3, s3_unified_op(method_slot),
89 0, 0,
90 brix_metric_err_from_http_status(http_status));
91}

93void
94s3_metrics_response_method(ngx_http_request_t *r, ngx_uint_t method_slot,
95 ngx_int_t handler_rc)
96{
97 ngx_uint_t http_status;
98
99 /*
100 * NGX_DONE means nginx will finish the request asynchronously, e.g. after
101 * reading a PUT body. The callback accounts for the final response.
102 */
103 if (handler_rc == NGX_DONE) {
104 return;
105 }
106
107 http_status = brix_http_effective_status(r, handler_rc);
108 s3_metrics_response_status(method_slot, http_status);
109}
