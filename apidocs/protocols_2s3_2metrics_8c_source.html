1/*
2 * metrics.c - S3-compatible endpoint request/result accounting helpers.
3 *
4 * WHAT: Maps HTTP method codes to XRootD metric slot enums, increments request
5 * and response counters by method+status-class buckets, and converts handler
6 * return codes to HTTP status for metric recording. Provides six public functions:
7 * s3_metrics_method_slot (HTTP→metric enum), s3_metrics_request_method (request counter),
8 * s3_metrics_response_status (response counter by class), s3_metrics_response_method
9 * (handler RC → status → response counter), s3_metrics_return_method (counter + return),
10 * and s3_metrics_finalize_request_method (counter + ngx_http_finalize_request).
11 *
12 * WHY: S3-compatible endpoints require separate metric buckets from WebDAV/XRootD stream.
13 * Metrics track requests_total per-method and responses_total per-method×status-class.
14 * These helpers centralize method-to-slot mapping, status-class extraction via
15 * brix_http_status_class(), and handler-RC-to-HTTP-status conversion — ensuring all
16 * S3 handlers (get.c, put.c, list.c, multipart.c) use consistent metric accounting.
17 * NGX_DONE handling: async PUT body reads defer final response counting to callbacks.
18 *
19 * HOW: s3_metrics_method_slot() maps r->method enum values to BRIX_S3_METHOD_* constants —
20 * GET→GET, HEAD→HEAD, PUT→PUT, DELETE→DELETE, POST→POST, everything else→OTHER.
21 * s3_metrics_request_method() clamps method_slot to BRIX_S3_NMETHODS, increments
22 * requests_total[slot] via BRIX_S3_METRIC_INC(). s3_metrics_response_status() same clamp,
23 * extracts status_class from http_status via brix_http_status_class(), increments
24 * responses_total[slot][status_class]. s3_metrics_response_method() skips NGX_DONE (deferred),
25 * converts handler RC to HTTP status: ERROR→500, >=NGX_HTTP_SPECIAL_RESPONSE→RC value,
26 * r->headers_out.status non-zero→that value, else→200. Calls s3_metrics_response_status().
27 * s3_metrics_return_method() delegates response counting then returns original handler_rc.
28 * s3_metrics_finalize_request_method() same + calls ngx_http_finalize_request(r, handler_rc).
29 */

39{

132static int64_t
133s3_sess_xfer_expected(ngx_http_request_t *r, brix_sess_mode_t mode)
134{
135 if (r == NULL) {
136 return -1;
137 }
138
139 if (mode == BRIX_SESS_MODE_READ) {
140 return r->headers_out.content_length_n >= 0
141 ? (int64_t) r->headers_out.content_length_n : -1;
142 }
143
144 if (mode == BRIX_SESS_MODE_WRITE) {
145 return r->headers_in.content_length_n >= 0
146 ? (int64_t) r->headers_in.content_length_n : -1;
147 }
148
149 return -1;
150}

152static uint64_t
153s3_sess_xfer_bytes(ngx_http_request_t *r, brix_sess_mode_t mode,
154 int64_t expected)
155{
156 if (mode == BRIX_SESS_MODE_READ) {
157 return expected > 0 ? (uint64_t) expected : 0;
158 }
159
160 if (mode == BRIX_SESS_MODE_WRITE && r != NULL
161 && r->headers_in.content_length_n > 0)
162 {
163 return (uint64_t) r->headers_in.content_length_n;
164 }
165
166 return 0;
167}

169static void
170s3_sess_start_xfer(ngx_http_request_t *r, brix_sess_t *sess,
171 const char *path, ngx_uint_t method_slot)
172{
173 ngx_http_s3_req_ctx_t *ctx;
174
175 if (!s3_sess_method_has_xfer(method_slot)) {
176 return;
177 }
178
179 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
180 if (ctx == NULL || ctx->sess_xfer_started) {
181 return;
182 }
183
184 brix_sess_xfer_start(sess, &ctx->sess_xfer, path,
185 s3_sess_mode(method_slot), -1);
186 ctx->sess_xfer_started = ctx->sess_xfer.active ? 1 : 0;
187}

189static void
190s3_sess_finish_xfer(ngx_http_request_t *r, brix_sess_t *sess,
191 ngx_uint_t method_slot, ngx_uint_t http_status)
192{
193 ngx_http_s3_req_ctx_t *ctx;
194 brix_sess_mode_t mode;
195 int64_t expected;
196 uint64_t bytes;
197
198 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
199 if (ctx == NULL || !ctx->sess_xfer_started) {
200 return;
201 }
202
203 mode = s3_sess_mode(method_slot);
204 expected = s3_sess_xfer_expected(r, mode);
205 bytes = s3_sess_xfer_bytes(r, mode, expected);
206 ctx->sess_xfer.expected = expected;
207 if (bytes > ctx->sess_xfer.bytes) {
208 brix_sess_xfer_add(&ctx->sess_xfer, bytes - ctx->sess_xfer.bytes);
209 }
210
211 brix_sess_xfer_end(sess, &ctx->sess_xfer,
212 http_status < NGX_HTTP_BAD_REQUEST
213 ? BRIX_SESS_XFER_COMPLETE
214 : BRIX_SESS_XFER_ABORTED);
215 ctx->sess_xfer_started = 0;
216}

266void
267s3_metrics_response_status(ngx_uint_t method_slot, ngx_uint_t http_status)
268{
269 ngx_uint_t status_class;
270
271 if (method_slot >= BRIX_S3_NMETHODS) {
272 method_slot = BRIX_S3_METHOD_OTHER;
273 }
274
275 status_class = brix_http_status_class(http_status);
276 BRIX_S3_METRIC_INC(responses_total[method_slot][status_class]);
277 brix_metric_op_done(BRIX_PROTO_S3, s3_unified_op(method_slot),
278 0, 0,
279 brix_metric_err_from_http_status(http_status));
280}

282void
283s3_metrics_response_method(ngx_http_request_t *r, ngx_uint_t method_slot,
284 ngx_int_t handler_rc)
285{
286 ngx_uint_t http_status;
287 brix_sess_t *sess;
288 char path[BRIX_SESSLOG_PATH_MAX];
289 char errscratch[BRIX_SESSLOG_ERR_MAX];
290
291 /*
292 * NGX_DONE means nginx will finish the request asynchronously, e.g. after
293 * reading a PUT body. The callback accounts for the final response.
294 */
295 if (handler_rc == NGX_DONE) {
296 return;
297 }
298
299 http_status = brix_http_effective_status(r, handler_rc);
300 sess = s3_sess(r);
301 brix_sess_result(sess, http_status < NGX_HTTP_BAD_REQUEST,
302 brix_http_sess_uri(r, path, sizeof(path)),
303 s3_sess_mode(method_slot),
304 http_status < NGX_HTTP_BAD_REQUEST ? NULL
305 : brix_sesslog_err_from_http((int) http_status,
306 errscratch,
307 sizeof(errscratch)));
308 s3_sess_finish_xfer(r, sess, method_slot, http_status);
309 s3_metrics_response_status(method_slot, http_status);
310}
