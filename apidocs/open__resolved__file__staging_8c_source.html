28/*
29 * open_resolved_file_staging.c — pre-open staging preflight for kXR_open
30 * (phase-79 split from open_resolved_file.c).
31 *
32 * WHAT: The write-side staging gate that runs BEFORE any handle/temp is
33 * allocated: reject writes onto a symlink/directory, shed write-opens
34 * under write-back-staging backpressure, enforce kXR_new exclusive-create
35 * against the final path, build the POSC/resume temp path, decide
36 * in-place-update vs upload, and reject a read of a directory.
37 *
38 * WHY: These checks share the confined VFS probe and mutate the same staging
39 * decision fields (use_resume/stage/posc_temp_path); grouping them keeps
40 * the staging policy in one focused unit, separate from the open ops.
41 *
42 * HOW: brix_open_stage_preflight (the sole cross-file entry) composes the
43 * file-local checks in order; each returns NGX_DECLINED to continue or a
44 * sent-error rc to abort. Behaviour is byte-identical to the original.

47/* Confined existence/type probe of an absolute path beneath `root` via the VFS
48 * (no metric, no pool). Returns 1 with *vst filled when the path exists, else 0.
49 * `nofollow` selects lstat vs stat semantics. Used for the kXR_open pre-flight
50 * checks (directory reject, exclusive-create, resume-partial existence), each of
51 * which has its own confinement root — the export root for the final path, the
52 * upload stage dir for an external resume partial.
53 *
54 * WHY (phase-70): on a driver-backed (remote root://) export the probe's
55 * existence/type check is not a local stat — it opens the ORIGIN to stat, so it
56 * must present the same forwarded credential the real open does. Without the
57 * deleg the origin refuses ("backend has NO credential") and the probe reports
58 * the file absent, which turns a legitimate read into a spurious kXR_NotFound.
59 * The actual open path binds the deleg on its own vctx (brix_open_dispatch_open
60 * → brix_root_vfs_bind_deleg); the pre-flight probe must do the same. Pass a
61 * NULL ctx/conf for a probe of a purely worker-local path (an external upload
62 * stage-dir partial), which never routes to the origin. `pool` supplies the
63 * deleg bag's allocation (the request pool); NULL leaves the probe deleg-less. */
64static int
65brix_open_probe(ngx_log_t *log, const char *root, const char *abs,
66 int nofollow, brix_vfs_stat_t *vst,
67 brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf, ngx_pool_t *pool)
68{
69 brix_vfs_ctx_t vctx;
70
71 brix_vfs_ctx_init(&vctx, pool, log, BRIX_PROTO_ROOT, root, NULL,
72 1 /* allow_write */, 0 /* is_tls */, NULL, abs);
73 if (ctx != NULL && conf != NULL && pool != NULL) {
74 brix_root_vfs_bind_deleg(ctx, conf, &vctx);
75 }
76 return brix_vfs_probe(&vctx, nofollow, vst) == NGX_OK;
77}

79 /* A write open of an existing DIRECTORY must be rejected up front with
80 * kXR_isDirectory (stock parity: O_WRONLY on a directory fails EISDIR).
81 * Without this the staging path below would derive a ".part" FILE from the
82 * directory's name, create it, and wrongly report success — diverging from
83 * stock. The read side is rejected symmetrically just below. */
84static ngx_int_t
85brix_open_check_write_target(brix_ctx_t *ctx, ngx_connection_t *c,
86 ngx_stream_brix_srv_conf_t *conf, const char *resolved, ngx_flag_t is_write)
87{
88 if (is_write) {
89 brix_vfs_stat_t dst;
90 /* A final path that is itself a symlink must be rejected for write: we
91 * never write THROUGH an in-root link. The direct-open mapping enforces
92 * this with O_NOFOLLOW on the final component (ELOOP), but the staging
93 * path opens a randomly-named temp instead of the final and would commit
94 * over the link on rename — so guard it here. lstat (no-follow) reports
95 * the link as itself; resolution is already confined to the export, so
96 * this catches an in-export link with EITHER an in-root or outward target
97 * without following it. */
98 if (brix_open_probe(c->log, conf->common.root_canon, resolved, 1,
99 &dst, ctx, conf, c->pool)
100 && S_ISLNK((mode_t) dst.mode)) {
101 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
102 resolved, "wr", kXR_NotAuthorized,
103 "refusing to write through a symlink");
104 }
105 if (brix_open_probe(c->log, conf->common.root_canon, resolved, 0,
106 &dst, ctx, conf, c->pool)
107 && dst.is_directory) {
108 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
109 resolved, "wr", kXR_isDirectory,
110 "is a directory");
111 }
112 }
113 /* NGX_DECLINED = allow/continue. A rejection above sends its own error and
114 * returns that send's rc (NGX_OK on success), so the caller MUST test for
115 * NGX_DECLINED — not NGX_OK — to tell "continue" from "error already sent"
116 * (else it falls through and sends a second, conflicting reply). */
117 return NGX_DECLINED;
118}

120 /* Phase C: two-tier write-back-staging backpressure. When write-through
121 * staging is configured with watermarks, shed new write-opens while the
122 * staging filesystem is full — delay in the soft band (kXR_wait, the client
123 * retries), reject at the hard cap (kXR_Overloaded). Runs before any handle or
124 * staging-temp allocation, so a shed write consumes nothing. Reads never reach
125 * here. */
126static ngx_int_t /* the value to return from the caller, or NGX_DECLINED = allow/continue */
127brix_open_stage_backpressure(brix_ctx_t *ctx, ngx_connection_t *c,
128 ngx_stream_brix_srv_conf_t *conf, const char *resolved, ngx_flag_t is_write)
129{
130 if (is_write && conf->wt.enable) {
131 switch (brix_wt_stage_admit(conf)) {
132 case BRIX_WT_ADMIT_WAIT:
133 brix_metric_wt_stage_throttled(0 /* wait */);
134 brix_log_access(ctx, c, "OPEN", resolved, "wr-staging-wait",
135 0, 0, "write-back staging busy; retry", 0);
136 return brix_send_wait(ctx, c, BRIX_WT_STAGE_WAIT_SECS);
137 case BRIX_WT_ADMIT_REJECT:
138 brix_metric_wt_stage_throttled(1 /* reject */);
139 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN", resolved,
140 "wr", kXR_Overloaded,
141 "write-back staging area full");
142 case BRIX_WT_ADMIT_ALLOW:
143 default:
144 break;
145 }
146 }
147 return NGX_DECLINED;
148}

150 /* When staging is active the kXR_new exclusive-create check must run against
151 * the FINAL path, not the staging temp (which is what actually gets O_EXCL):
152 * staging never opens the final, so without this an exclusive create over an
153 * existing object would wrongly succeed and overwrite it on commit.
154 *
155 * This is an EXCLUSIVE-create check, so it only applies when kXR_new is set
156 * WITHOUT kXR_delete — exactly the O_EXCL condition in the direct-open
157 * mapping (open_flags.h: kXR_new adds O_EXCL "unless kXR_delete is also
158 * set"). With kXR_delete also present the client explicitly asked to
159 * truncate/overwrite any existing object (the delete intent wins over the
160 * new intent), so an existing final must NOT be rejected — staging will
161 * replace it on the commit rename. Without this guard a delete+new overwrite
162 * (e.g. xrdcp -f, or a kXR_recoverWrts reopen) was wrongly rejected with
163 * kXR_ItExists. */
164static ngx_int_t
165brix_open_check_exclusive_create(brix_open_args_t *a)
166{
167 brix_ctx_t *ctx = a->ctx;
168 ngx_connection_t *c = a->c;
169 ngx_stream_brix_srv_conf_t *conf = a->conf;
170 const char *resolved = a->resolved;
171
172 if (a->stage && (a->options & kXR_new) && !(a->options & kXR_delete)) {
173 brix_vfs_stat_t fst;
174 if (brix_open_probe(c->log, conf->common.root_canon, resolved, 0,
175 &fst, ctx, conf, c->pool)) {
176 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
177 resolved, "wr", kXR_ItExists,
178 "file already exists");
179 }
180 }
181 /* NGX_DECLINED = allow/continue; a kXR_ItExists rejection above already sent
182 * its reply and returned that send's rc — the caller tests for NGX_DECLINED,
183 * not NGX_OK, so a sent error stops the open instead of falling through to
184 * open the staging temp and send a second (fhandle) reply. */
185 return NGX_DECLINED;
186}

188/* Build the POSC/resume staging temp path into posc_temp_path (resume = a
189 * deterministic identity-keyed partial; POSC = a random ".posc.*" temp on the
190 * final path's fs). Returns NGX_OK, or an error response on a too-long path. */
191static ngx_int_t
192brix_open_build_stage_temp(brix_open_args_t *a)
193{
194 brix_ctx_t *ctx = a->ctx;
195 ngx_connection_t *c = a->c;
196 ngx_stream_brix_srv_conf_t *conf = a->conf;
197 const char *resolved = a->resolved;
198 char *posc_temp_path = a->posc_temp_path;
199 size_t posc_temp_path_sz = sizeof(a->posc_temp_path);
200
201 if (a->use_resume) {
202 /* Deterministic, identity-keyed: a reconnecting client re-opening the
203 * same final path lands on the SAME partial and resumes from its
204 * offset. Anonymous (empty dn) shares per-path (no per-user isolation
205 * on such an endpoint anyway). When brix_stage_dir is set the partial
206 * lives on that fast device and the close-time commit moves it to the
207 * destination (cross-device copy). */
208 const char *principal = ctx->login.dn[0] ? ctx->login.dn : NULL;
209 const char *stage_dir = conf->upload_stage_dir_canon[0]
210 ? conf->upload_stage_dir_canon : NULL;
211 if (brix_make_resume_path(resolved, principal, stage_dir,
212 posc_temp_path, posc_temp_path_sz)
213 != NGX_OK) {
214 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
215 resolved, "wr", kXR_ServerError,
216 "resume temp path too long");
217 }
218 } else if (a->use_posc) {
219 if (brix_make_tmp_path(resolved, posc_temp_path,
220 posc_temp_path_sz) != NGX_OK) {
221 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
222 resolved, "wr", kXR_ServerError,
223 "POSC temp path too long");
224 }
225 }
226 /* NGX_DECLINED = built OK / nothing to build (continue); a path-too-long
227 * rejection above already sent its reply — caller tests NGX_DECLINED, not
228 * NGX_OK, so a sent error is not mistaken for "continue". */
229 return NGX_DECLINED;
230}

232/* In-place update vs upload for resume staging: a pure update-in-place open
233 * (kXR_open_updt, no delete/new) with NO resume partial in flight is NOT an
234 * upload to stage — drop out of resume staging so the direct open mapping edits
235 * the committed file in place (or fails kXR_NotFound), matching stock xrootd.
236 * Clears *use_resume / recomputes *stage when so. */
237static void
238brix_open_resume_inplace_decide(brix_open_args_t *a)
239{
240 brix_ctx_t *ctx = a->ctx;
241 ngx_connection_t *c = a->c;
242 ngx_stream_brix_srv_conf_t *conf = a->conf;
243 const char *resolved = a->resolved;
244 const char *posc_temp_path = a->posc_temp_path;
245 uint16_t options = a->options;
246 ngx_flag_t use_posc = a->use_posc;
247 ngx_flag_t *use_resume = &a->use_resume;
248 ngx_flag_t *stage = &a->stage;
249
250 if (*use_resume && !(options & kXR_delete) && !(options & kXR_new)) {
251 brix_vfs_stat_t fst;
252 int have_partial;
253
254 /* A pure update-in-place open is NOT an upload to stage when no resume
255 * partial is in flight: drop out of resume staging and let the direct
256 * open mapping (O_RDWR, no O_CREAT) decide, matching stock xrootd:
257 * - final is a committed regular file -> edit it in place (preserve the
258 * bytes the client does not rewrite);
259 * - final does not exist -> fail kXR_NotFound, exactly as
260 * O_RDWR-without-O_CREAT would. Staging would otherwise CREATE the
261 * missing file and return kXR_ok, diverging from stock which derives
262 * no O_CREAT for kXR_open_updt alone (XrdXrootdXeq.cc:1524). */
263 /* The partial lives under the upload stage dir when one is configured
264 * (a separate, svc-owned storage domain), else next to the final under
265 * the export root. Probe the export-root partial through the VFS; the
266 * external stage-dir partial is checked as the worker (separate domain,
267 * same reasoning as the open below). */
268 if (conf->upload_stage_dir_canon[0] != '\0') {
269 struct stat sst; /* vfs-seam-allow: separate upload stage-dir domain */
270 have_partial = (stat(posc_temp_path, &sst) == 0); /* vfs-seam-allow: separate upload stage-dir domain */
271 } else {
272 brix_vfs_stat_t pst;
273 have_partial = brix_open_probe(c->log, conf->common.root_canon,
274 posc_temp_path, 0, &pst,
275 ctx, conf, c->pool);
276 }
277 int final_exists = brix_open_probe(c->log, conf->common.root_canon,
278 resolved, 0, &fst,
279 ctx, conf, c->pool);
280 if (!have_partial && (!final_exists || fst.is_regular)) {
281 *use_resume = 0;
282 *stage = use_posc;
283 }
284 }
285}

287/* Run the pre-open staging preflight: reject a write onto a symlink/dir, shed
288 * write-opens under staging backpressure, enforce kXR_new exclusive-create
289 * against the final path, build the POSC/resume temp path, drop resume staging
290 * for a pure in-place update with no partial, and reject a read of a directory.
291 * Mutates *use_resume / *stage / posc_temp_path. Returns NGX_DECLINED to proceed;
292 * on any rejection the error is already sent and its rc is returned. */
293ngx_int_t
294brix_open_stage_preflight(brix_open_args_t *a)
295{
296 brix_ctx_t *ctx = a->ctx;
297 ngx_connection_t *c = a->c;
298 ngx_stream_brix_srv_conf_t *conf = a->conf;
299 const char *resolved = a->resolved;
300 ngx_int_t rc;
301
302 /* Reject a write open onto a symlink or existing directory (split out). */
303 rc = brix_open_check_write_target(ctx, c, conf, resolved, a->is_write);
304 if (rc != NGX_DECLINED) {
305 return rc; /* rejected: error already sent, propagate its rc */
306 }
307
308 /* Two-tier write-back-staging backpressure: shed new write-opens when the
309 * staging fs is full (kXR_wait / kXR_Overloaded) (split out). */
310 rc = brix_open_stage_backpressure(ctx, c, conf, resolved, a->is_write);
311 if (rc != NGX_DECLINED) {
312 return rc;
313 }
314
315 /* kXR_new exclusive-create check against the FINAL path when staging: the
316 * staging temp is same-fs of the destination so rename(2) is atomic (split
317 * out). */
318 rc = brix_open_check_exclusive_create(a);
319 if (rc != NGX_DECLINED) {
320 return rc; /* kXR_ItExists already sent, propagate its rc */
321 }
322
323 /* Build the POSC/resume staging temp path (split out). */
324 rc = brix_open_build_stage_temp(a);
325 if (rc != NGX_DECLINED) {
326 return rc; /* path-too-long error already sent, propagate its rc */
327 }
328
329 /*
330 * In-place update vs upload: resume staging assumes a writable open is an
331 * UPLOAD — it starts from an empty partial and commits it over the final path
332 * with a rename. But a pure update-in-place open (kXR_open_updt, neither
333 * kXR_delete/truncate nor kXR_new/create) on an ALREADY-COMMITTED file is a
334 * read-modify-write that must preserve the bytes the client does not rewrite.
335 * Staged through an empty partial, those bytes are silently lost on the commit
336 * rename (e.g. a 100-byte write at offset 100 of a 200-byte file zero-fills
337 * [0,100)) — a data-integrity bug, and a divergence from stock xrootd which
338 * always edits such a file in place.
339 *
340 * Only stage such an open when a resume partial ALREADY exists — that is a
341 * genuine reconnect continuing an interrupted upload (the partial holds the
342 * bytes received so far). When no partial exists and the final file is a real
343 * committed regular file, fall through to opening it directly, in place.
344 */
345 brix_open_resume_inplace_decide(a);
346
347 if (!a->is_write) {
348 brix_vfs_stat_t rst;
349 if (brix_open_probe(c->log, conf->common.root_canon, resolved, 0,
350 &rst, ctx, conf, c->pool)
351 && rst.is_directory) {
352 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
353 resolved, "rd", kXR_isDirectory,
354 "is a directory");
355 }
356 }
357 return NGX_DECLINED;
358}
