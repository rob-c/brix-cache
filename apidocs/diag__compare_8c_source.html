49/*
50 * §15.6 cross-protocol consistency oracle: read the SAME object via root:// and
51 * cleartext WebDAV (HTTP GET) and assert size + MD5 agree. The capability no
52 * upstream client has — this project unifies the planes over one VFS, so a
53 * divergence here is a real cross-protocol bug. S3 (SigV4) and HTTPS-davs
54 * (TLS+chunked) planes are deferred — noted, not implemented.
55 */
56int
57do_compare_davs(const diag_args *a)
58{
59 brix_url ua;
60 brix_conn ca;
61 brix_status st;
62 char dhost[256];
63 int dport;
64 char root_md5[64], davs_md5[64];
65 char *body;
66 size_t blen = 0;
67 int http = 0, fd;
68 char tmpl[] = "/tmp/xrddiag-davs.XXXXXX";
69
70 brix_status_clear(&st);
71 if (brix_endpoint_parse(a->url, &ua, &st) != 0) {
72 fprintf(stderr, "xrddiag: %s\n", st.msg);
73 return 50;
74 }
75 if (ua.path[0] == '\0' || strcmp(ua.path, "/") == 0) {
76 fprintf(stderr, "xrddiag: compare --davs needs a file path in the URL\n");
77 return 50;
78 }
79 parse_http_hostport(a->davs, dhost, sizeof(dhost), &dport);
80 printf("Cross-protocol compare %s\n root:// %s:%d davs(http) %s:%d path %s\n",
81 a->url, ua.host, ua.port, dhost, dport, ua.path);
82
83 if (brix_connect(&ca, &ua, &a->conn, &st) != 0) {
84 fprintf(stderr, "xrddiag: connect %s:%d: %s\n", ua.host, ua.port, st.msg);
85 return brix_shellcode(&st);
86 }
87 if (remote_md5(&ca, ua.path, root_md5, sizeof(root_md5), &st) != 0) {
88 probe("root-read", 0, "%s", st.msg);
89 brix_close(&ca);
90 return 1;
91 }
92 brix_close(&ca);
93
94 /* WebDAV plane: cleartext HTTP GET of the same logical path (binary-safe). */
95 body = (char *) malloc(1u << 20);
96 if (body == NULL) {
97 fprintf(stderr, "xrddiag: out of memory\n");
98 return 51;
99 }
100 brix_status_clear(&st);
101 if (brix_http_get(dhost, dport, ua.path, 5000, &http, body, 1u << 20, &blen,
102 &st) != 0) {
103 probe("davs-http", 0, "GET %s:%d%s: %s", dhost, dport, ua.path, st.msg);
104 free(body);
105 return 1;
106 }
107 probe("davs-http", http == 200, "HTTP %d for %s", http, ua.path);
108 if (http != 200) {
109 free(body);
110 printf("Result: %d difference(s)\n", g_fails);
111 return 1;
112 }
113 fd = mkstemp(tmpl);
114 if (fd < 0 || (size_t) write(fd, body, blen) != blen
115 || brix_cksum_fd(fd, XRDC_CK_MD5, davs_md5, sizeof(davs_md5), &st) != 0) {
116 probe("davs-md5", 0, "local md5 failed");
117 if (fd >= 0) { close(fd); unlink(tmpl); }
118 free(body);
119 return 1;
120 }
121 close(fd);
122 unlink(tmpl);
123 free(body);
124
125 probe("davs-md5", strcmp(root_md5, davs_md5) == 0,
126 "root=%s davs=%s", root_md5, davs_md5);
127 note("s3 / https-davs", "deferred (needs SigV4 / HTTPS+chunked)");
128 printf("Result: %d difference(s)\n", g_fails);
129 return g_fails ? 1 : 0;
130}

133/*
134 * WHAT: Parse both compare URLs, validate the A-side path, and open both
135 * connections (`ca`, `cb`).
136 * WHY: Isolates the argument/URL validation + dual-connect prologue (with its
137 * exact error strings and exit codes) from the actual comparison so the
138 * orchestrator reads as a flat setup → compare → teardown.
139 * HOW: Reject a missing reference, a parse failure, or a bare/empty A path; then
140 * connect A and connect B (closing A if B fails). On success fills `ua`/`ub`
141 * and the open `ca`/`cb`. Returns 0 on success; on failure returns the exact
142 * exit code the inline code used (50 / brix_shellcode). No behavior change.
143 */
144static int
145compare_setup(const diag_args *a, brix_url *ua, brix_url *ub,
146 brix_conn *ca, brix_conn *cb)
147{
148 brix_status st;
149
150 if (a->ref_url == NULL) {
151 fprintf(stderr, "xrddiag: compare needs --vs-reference <url>\n");
152 return 50;
153 }
154 brix_status_clear(&st);
155 if (brix_endpoint_parse(a->url, ua, &st) != 0 ||
156 brix_endpoint_parse(a->ref_url, ub, &st) != 0) {
157 fprintf(stderr, "xrddiag: %s\n", st.msg);
158 return 50;
159 }
160 if (ua->path[0] == '\0' || strcmp(ua->path, "/") == 0) {
161 fprintf(stderr, "xrddiag: compare needs a file/dir path in the URL\n");
162 return 50;
163 }
164 if (brix_connect(ca, ua, &a->conn, &st) != 0) {
165 fprintf(stderr, "xrddiag: connect A %s:%d: %s\n", ua->host, ua->port, st.msg);
166 return brix_shellcode(&st);
167 }
168 if (brix_connect(cb, ub, &a->conn, &st) != 0) {
169 fprintf(stderr, "xrddiag: connect B %s:%d: %s\n", ub->host, ub->port, st.msg);
170 brix_close(ca);
171 return brix_shellcode(&st);
172 }
173 return 0;
174}

176/*
177 * WHAT: Compare two directories by their dirlist name-set (order-independent).
178 * WHY: Keeps the O(n*m) set-equality probe out of the orchestrator's compare
179 * block so the file-vs-dir branch stays a one-line dispatch.
180 * HOW: List both dirs; if either fails, emit a "dirlist-set" fail probe. Else
181 * equal iff counts match and every A name appears in B; emit the probe with
182 * the entry counts. Side effect: probe() output only. No behavior change.
183 */
184static void
185compare_dirs(brix_conn *ca, brix_conn *cb, const char *pa, const char *pb)
186{
187 brix_dirent *ea = NULL, *eb = NULL;
188 size_t na = 0, nb = 0;
189 int eq = 1;
190 brix_status sta, stb;
191
192 brix_status_clear(&sta);
193 brix_status_clear(&stb);
194 if (brix_dirlist(ca, pa, 0, &ea, &na, &sta) == 0 &&
195 brix_dirlist(cb, pb, 0, &eb, &nb, &stb) == 0) {
196 if (na != nb) {
197 eq = 0;
198 } else {
199 for (size_t i = 0; i < na && eq; i++) {
200 int hit = 0;
201 for (size_t j = 0; j < nb; j++) {
202 if (strcmp(ea[i].name, eb[j].name) == 0) { hit = 1; break; }
203 }
204 if (!hit) { eq = 0; }
205 }
206 }
207 probe("dirlist-set", eq, "A=%zu B=%zu entries", na, nb);
208 } else {
209 probe("dirlist-set", 0, "list failed");
210 }
211 free(ea);
212 free(eb);
213}

215/*
216 * WHAT: Compare two files by their remote MD5 digests.
217 * WHY: Confines the dual remote_md5 fetch + compare (and its failure-status
218 * formatting) to one helper mirroring compare_dirs.
219 * HOW: Fetch MD5 of both paths; on both-OK emit an "md5" probe of the string
220 * compare, else a fail probe carrying whichever side reported an error.
221 * Side effect: probe() output only. No behavior change.
222 */
223static void
224compare_files(brix_conn *ca, brix_conn *cb, const char *pa, const char *pb)
225{
226 char ha[64], hb[64];
227 brix_status ma, mb;
228
229 brix_status_clear(&ma);
230 brix_status_clear(&mb);
231 if (remote_md5(ca, pa, ha, sizeof(ha), &ma) == 0 &&
232 remote_md5(cb, pb, hb, sizeof(hb), &mb) == 0) {
233 probe("md5", strcmp(ha, hb) == 0, "A=%s B=%s", ha, hb);
234 } else {
235 probe("md5", 0, "A:%s B:%s",
236 ma.kxr ? ma.msg : "ok", mb.kxr ? mb.msg : "ok");
237 }
238}

240/*
241 * WHAT: Stat both paths, probe size equality, then dispatch to the dir- or
242 * file-comparison helper.
243 * WHY: Splits the comparison body from setup/teardown; returns whether any early
244 * exit (stat failure) occurred so the orchestrator can short-circuit.
245 * HOW: Stat A and B; on either failure emit a "stat" probe and return 1 (caller
246 * returns 1). Else probe "size", then compare_dirs / compare_files per
247 * kXR_isDir. Returns 0 to continue. No behavior change.
248 */
249static int
250compare_probe(brix_conn *ca, brix_conn *cb, const brix_url *ua, const brix_url *ub)
251{
252 brix_statinfo sa, sb;
253 brix_status sta, stb;
254 const char *pa = ua->path;
255 const char *pb = (ub->path[0] && strcmp(ub->path, "/")) ? ub->path : ua->path;
256 int oka, okb;
257
258 brix_status_clear(&sta);
259 brix_status_clear(&stb);
260 oka = brix_stat(ca, pa, &sa, &sta) == 0;
261 okb = brix_stat(cb, pb, &sb, &stb) == 0;
262 if (!oka || !okb) {
263 probe("stat", 0, "A:%s B:%s", oka ? "ok" : sta.msg, okb ? "ok" : stb.msg);
264 return 1;
265 }
266 probe("size", sa.size == sb.size, "A=%lld B=%lld",
267 (long long) sa.size, (long long) sb.size);
268
269 if (sa.flags & kXR_isDir) {
270 compare_dirs(ca, cb, pa, pb);
271 } else {
272 compare_files(ca, cb, pa, pb);
273 }
274 return 0;
275}

277int
278do_compare(const diag_args *a)
279{
280 brix_url ua = {0}, ub = {0};
281 brix_conn ca, cb;
282 int rc;
283
284 if (a->davs != NULL) {
285 return do_compare_davs(a);
286 }
287
288 rc = compare_setup(a, &ua, &ub, &ca, &cb);
289 if (rc != 0) {
290 return rc;
291 }
292
293 printf("Compare %s vs %s\n", a->url, a->ref_url);
294
295 if (compare_probe(&ca, &cb, &ua, &ub) != 0) {
296 brix_close(&ca);
297 brix_close(&cb);
298 return 1;
299 }
300
301 brix_close(&ca);
302 brix_close(&cb);
303 printf("Result: %d difference(s)\n", g_fails);
304 return g_fails ? 1 : 0;
305}
