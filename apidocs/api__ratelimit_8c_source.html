10dashboard_build_v1_ratelimit(int64_t now_ms,
11 const ngx_http_brix_dashboard_loc_conf_t *conf, ngx_uint_t redact)
12{
13 json_t *root = dashboard_new_v1_root(now_ms, conf);
14 json_t *zones_arr;
15 brix_rl_zone_t *zones[16];
16 ngx_uint_t nz, zi;
17
18 if (!root) { return NULL; }
19 json_object_set_new(root, "anonymous", redact ? json_true() : json_false());
20 zones_arr = json_array();
21 if (zones_arr == NULL) { json_decref(root); return NULL; }
22
23 nz = brix_rl_zones_all(zones, 16);
24 for (zi = 0; zi < nz; zi++) {
25 /* `snap` is static to keep this large (per-principal) buffer off the
26 * worker stack. Safe: nginx workers are single-threaded for request
27 * processing, so there is no concurrent reuse within this loop. */
28 static brix_rl_snapshot_entry_t snap[256];
29 ngx_uint_t count = 0, i;
30 json_t *zo = json_object();
31 json_t *parr = json_array();
32
33 if (zo == NULL || parr == NULL) {
34 if (zo) json_decref(zo);
35 if (parr) json_decref(parr);
36 continue;
37 }
38
39 brix_rl_snapshot(zones[zi], snap, 256, &count);
40
41 json_object_set_new(zo, "zone",
42 json_stringn((const char *) zones[zi]->name.data,
43 zones[zi]->name.len));
44 json_object_set_new(zo, "size_bytes",
45 json_integer((json_int_t) zones[zi]->size));
46 json_object_set_new(zo, "node_count", json_integer((json_int_t) count));
47
48 for (i = 0; i < count; i++) {
49 json_t *p = json_object();
50 if (p == NULL) { continue; }
51 /* key_str is the rate-limit principal (DN/VO/IP/path) — drop for anon. */
52 if (!redact) {
53 json_object_set_new(p, "key", json_string(snap[i].key_str));
54 }
55 json_object_set_new(p, "req_total",
56 json_integer((json_int_t) snap[i].req_total));
57 json_object_set_new(p, "bytes_total",
58 json_integer((json_int_t) snap[i].bytes_total));
59 json_object_set_new(p, "throttle_count",
60 json_integer((json_int_t) snap[i].throttle_count));
61 json_object_set_new(p, "req_excess",
62 json_integer((json_int_t) snap[i].req_excess));
63 json_object_set_new(p, "bw_excess",
64 json_integer((json_int_t) snap[i].bw_excess));
65 json_array_append_new(parr, p);
66 }
67 json_object_set_new(zo, "principals", parr);
68 json_array_append_new(zones_arr, zo);
69 }
70
71 json_object_set_new(root, "zones", zones_arr);
72 return root;
73}
