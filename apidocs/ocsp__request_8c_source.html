1/*
2 * OCSP request/response — build the query, POST it, and verify the reply.
3 *
4 * WHAT: The OCSP protocol crypto proper. do_ocsp_request() builds an
5 * OCSP_REQUEST (cert id + anti-replay nonce), opens/connects the responder BIO
6 * via the transport layer, bounds every network phase under
7 * BRIX_OCSP_TIMEOUT_SECS, POSTs the request, and returns the parsed
8 * OCSP_RESPONSE. check_ocsp_response() verifies that response — overall status,
9 * responder signature against a trust store, nonce match — and maps the
10 * certificate status to GOOD(0) / REVOKED(-1) / UNKNOWN(1).
11 *
12 * WHY: Split out of ocsp.c under the phase-79 file-size guard. This file is the
13 * security-critical heart of the revocation check: the nonce guards against
14 * replay, the signature verification proves the response came from a trusted
15 * responder, and the status mapping is the actual revocation decision. Keeping
16 * it separate from the URL/BIO transport (ocsp_transport.c) and the public entry
17 * points (ocsp.c) keeps each concern reviewable in isolation.
18 *
19 * HOW: OCSP_request_add0_id() takes ownership of `id` on success (the caller
20 * must not free it after a successful add). do_ocsp_request() owns the BIO
21 * teardown (ocsp_conn_free, cbio then ssl_ctx) and the request free on every
22 * path. check_ocsp_response() frees the BASICRESP on every path; a REVOKED
23 * status is never overridden. Shared structs/decls live in ocsp_internal.h.
24 */

33/*
34 * ocsp_build_request — allocate an OCSP request for cert id `id` with a nonce.
35 *
36 * WHAT: Returns a new OCSP_REQUEST carrying `id` and a fresh nonce, or NULL on
37 * allocation failure.
38 * WHY: Splits request construction out of the network path so do_ocsp_request()
39 * reads as build → connect → send.
40 * HOW: OCSP_request_add0_id() takes ownership of `id` on success (unchanged
41 * from the original — the caller must not free `id` after a successful add).
42 * The nonce (OCSP_request_add1_nonce) guards against replay. On the add0_id
43 * failure path the half-built request is freed here.
44 */
45static OCSP_REQUEST *
46ocsp_build_request(OCSP_CERTID *id)
47{
48 OCSP_REQUEST *req = OCSP_REQUEST_new();
49
50 if (req == NULL) {
51 return NULL;
52 }
53 if (OCSP_request_add0_id(req, id) == NULL) {
54 OCSP_REQUEST_free(req);
55 return NULL;
56 }
57 /* Add a nonce to prevent replay attacks */
58 OCSP_request_add1_nonce(req, NULL, -1);
59 return req;
60}

62/*
63 * do_ocsp_request — build request, POST to responder, return parsed response.
64 *
65 * The caller must call OCSP_RESPONSE_free() on the returned pointer.
66 * Returns NULL on any network or protocol error.
67 */
68/* HOW: Parses the OCSP URL via parse_ocsp_url() into an ocsp_url_t. Builds the OCSP_REQUEST (id + nonce) via ocsp_build_request(). Opens the responder BIO via ocsp_open_bio() (plain TCP or verifying TLS with SNI). Bounds the connect (ocsp_connect_bio) and, for HTTPS, the handshake+verify (ocsp_tls_handshake) under BRIX_OCSP_TIMEOUT_SECS. Sends the request and reads the reply through an OCSP_REQ_CTX whose response length is capped at OCSP_MAX_RESPONSE_BYTES (A-6/T2 — an untrusted responder must not stream an unbounded body), then tears down the connection (cbio then ssl_ctx) and the request. Returns NULL on any network/protocol failure or if the reply exceeds the cap. */
69OCSP_RESPONSE *
70do_ocsp_request(ngx_log_t *log, const char *url,
71 X509 *leaf, X509 *issuer, OCSP_CERTID *id, OCSP_REQUEST **req_out)
72{
73 ocsp_url_t u;
74 char hostport[320];
75 OCSP_REQUEST *req = NULL;
76 OCSP_RESPONSE *resp = NULL;
77 OCSP_REQ_CTX *rctx;
78 int rc;
79 ocsp_conn_t conn;
80
81 *req_out = NULL;
82
83 if (parse_ocsp_url(url, &u) != 0) {
84 ngx_log_error(NGX_LOG_WARN, log, 0,
85 "brix_ocsp: cannot parse URL \"%s\"", url);
86 return NULL;
87 }
88
89 req = ocsp_build_request(id);
90 if (req == NULL) {
91 return NULL;
92 }
93
94 /* Construct "host:port" for BIO */
95 snprintf(hostport, sizeof(hostport), "%s:%d", u.host, u.port);
96
97 if (ocsp_open_bio(log, &u, hostport, &conn) != 0) {
98 OCSP_REQUEST_free(req);
99 return NULL;
100 }
101
102 /* E1: bound the connect with a deadline (a black-holed responder must not
103 * freeze the worker for the kernel TCP timeout). */
104 if (ocsp_connect_bio(log, &conn, hostport) != 0) {
105 ocsp_conn_free(&conn);
106 OCSP_REQUEST_free(req);
107 return NULL;
108 }
109
110 /* E1: bound the TLS handshake + request/response read/write phases. */
111 ocsp_set_io_timeouts(conn.cbio, BRIX_OCSP_TIMEOUT_SECS);
112
113 if (u.use_ssl
114 && ocsp_tls_handshake(log, &conn, u.host, hostport) != 0)
115 {
116 ocsp_conn_free(&conn);
117 OCSP_REQUEST_free(req);
118 return NULL;
119 }
120
121 /* A2/T2: bound the response body. The one-shot sendreq_bio helper reads
122 * under only OpenSSL's internal default cap; drive the request context
123 * ourselves so we pin the ceiling at OCSP_MAX_RESPONSE_BYTES — an untrusted
124 * (and usually
125 * plaintext/MITM-able) responder must not stream an unbounded body into the
126 * worker. Blocking BIO + SO_RCVTIMEO bound the wait; the -1 retry arm
127 * covers a spurious BIO_should_retry. */
128 rctx = OCSP_sendreq_new(conn.cbio, u.path, req, -1);
129 if (rctx == NULL) {
130 ocsp_conn_free(&conn);
131 OCSP_REQUEST_free(req);
132 ngx_log_error(NGX_LOG_WARN, log, 0,
133 "brix_ocsp: cannot build request context for \"%s\"", url);
134 return NULL;
135 }
136 OCSP_set_max_response_length(rctx, OCSP_MAX_RESPONSE_BYTES);
137
138 do {
139 rc = OCSP_sendreq_nbio(&resp, rctx);
140 } while (rc == -1 && BIO_should_retry(conn.cbio));
141
142 OCSP_REQ_CTX_free(rctx);
143 ocsp_conn_free(&conn);
144
145 if (rc <= 0 || resp == NULL) {
146 /* rc == 0 also fires when the reply exceeded OCSP_MAX_RESPONSE_BYTES:
147 * OpenSSL aborts the read at the cap and returns failure. */
148 ngx_log_error(NGX_LOG_WARN, log, 0,
149 "brix_ocsp: no valid response from \"%s\" (capped at %d bytes)",
150 url, OCSP_MAX_RESPONSE_BYTES);
151 if (resp != NULL) {
152 OCSP_RESPONSE_free(resp);
153 resp = NULL;
154 }
155 OCSP_REQUEST_free(req);
156 return NULL;
157 }
158
159 /* Success: hand the request (with its nonce) to the caller for the nonce
160 * match; the caller owns it now and frees it after check_ocsp_response(). */
161 *req_out = req;
162 return resp;
163}

165/*
166 * check_ocsp_response — verify the OCSP response and return the cert status.
167 *
168 * Returns:
169 * 0 — certificate is GOOD
170 * -1 — certificate is REVOKED or response is invalid
171 * 1 — status is UNKNOWN
172 */
173/* HOW: Checks response status — returns -1 if not successful. Extracts the BASICRESP via OCSP_response_get1_basic(). Optionally verifies the response signature against a trust store (NULL means use OpenSSL defaults). If original request is available, checks nonce match — mismatch causes failure; missing nonce is a warning only. Finds certificate status via OCSP_resp_find_status() and maps to GOOD(0)/REVOKED(-1)/UNKNOWN(1) using a switch on the V_OCSP_CERTSTATUS_* enum. */
174int
175check_ocsp_response(ngx_log_t *log, OCSP_RESPONSE *resp,
176 X509_STORE *store, OCSP_CERTID *id, OCSP_REQUEST *req_for_nonce,
177 int require_nonce)
178{
179 OCSP_BASICRESP *bresp;
180 int status, reason;
181 ASN1_GENERALIZEDTIME *rev = NULL, *thisupd = NULL, *nextupd = NULL;
182 int rc = -1;
183
184 if (OCSP_response_status(resp) != OCSP_RESPONSE_STATUS_SUCCESSFUL) {
185 ngx_log_error(NGX_LOG_WARN, log, 0,
186 "brix_ocsp: OCSP response status = %d (not successful)",
187 OCSP_response_status(resp));
188 return -1;
189 }
190
191 bresp = OCSP_response_get1_basic(resp);
192 if (bresp == NULL) {
193 return -1;
194 }
195
196 /* Verify the response signature against the trust store */
197 if (store != NULL && OCSP_basic_verify(bresp, NULL, store, 0) <= 0) {
198 ngx_log_error(NGX_LOG_WARN, log, 0,
199 "brix_ocsp: OCSP response signature verification failed");
200 OCSP_BASICRESP_free(bresp);
201 return -1;
202 }
203
204 /* Check the nonce if we have the original request */
205 if (req_for_nonce != NULL) {
206 int nonce_rc = OCSP_check_nonce(req_for_nonce, bresp);
207 if (nonce_rc < 0) {
208 /* Nonce present in request but missing in response. A missing nonce
209 * lets an on-path attacker replay a captured (still-signed) GOOD
210 * response. Under brix_ocsp_require_nonce this is a hard failure
211 * (A-6 item 2); otherwise warn only — most CA responders serve
212 * pre-signed, nonce-less responses, so hard-fail is opt-in. */
213 if (require_nonce) {
214 ngx_log_error(NGX_LOG_WARN, log, 0,
215 "brix_ocsp: nonce missing in OCSP response and "
216 "brix_ocsp_require_nonce is on — denying (replay guard)");
217 OCSP_BASICRESP_free(bresp);
218 return -1;
219 }
220 ngx_log_error(NGX_LOG_WARN, log, 0,
221 "brix_ocsp: nonce missing in OCSP response");
222 } else if (nonce_rc == 0) {
223 ngx_log_error(NGX_LOG_WARN, log, 0,
224 "brix_ocsp: OCSP response nonce mismatch");
225 OCSP_BASICRESP_free(bresp);
226 return -1;
227 }
228 }
229
230 if (!OCSP_resp_find_status(bresp, id, &status, &reason,
231 &rev, &thisupd, &nextupd))
232 {
233 ngx_log_error(NGX_LOG_WARN, log, 0,
234 "brix_ocsp: certificate not found in OCSP response");
235 OCSP_BASICRESP_free(bresp);
236 return -1;
237 }
238
239 switch (status) {
240 case V_OCSP_CERTSTATUS_GOOD:
241 rc = 0;
242 break;
243 case V_OCSP_CERTSTATUS_REVOKED:
244 ngx_log_error(NGX_LOG_WARN, log, 0,
245 "brix_ocsp: certificate is REVOKED (reason=%d)", reason);
246 rc = -1;
247 break;
248 case V_OCSP_CERTSTATUS_UNKNOWN:
249 default:
250 ngx_log_error(NGX_LOG_WARN, log, 0,
251 "brix_ocsp: certificate status is UNKNOWN");
252 rc = 1; /* caller decides based on soft_fail */
253 break;
254 }
255
256 OCSP_BASICRESP_free(bresp);
257 return rc;
258}
