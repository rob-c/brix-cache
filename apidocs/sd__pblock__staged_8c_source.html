1/*
2 * sd_pblock_staged.c — staged atomic-publish vtable slots for the pblock driver.
3 *
4 * WHAT: Implements the staged-write path of brix_sd_pblock_driver:
5 * staged_open/write/commit/abort. Body bytes are written to a fresh blob id
6 * (no visible catalog row yet); commit publishes them atomically by
7 * inserting the final row that points at that blob (the staged blocks simply
8 * become the object — no copy or rename), while abort removes them.
9 *
10 * WHY: Split out of sd_pblock.c (phase-79) to keep every pblock file under the
11 * ~500-line, one-concept cap. Atomic publish is its own concern, distinct
12 * from the object lifecycle (sd_pblock.c), the hot byte path (sd_pblock_io.c)
13 * and the namespace ops (sd_pblock_namespace.c). The functions are
14 * non-static because the driver descriptor names them; commit reuses
15 * sd_pblock_drop_dst from the namespace file. Declarations live in
16 * sd_pblock_internal.h.
17 *
18 * HOW: staged_open gates the parent collection up front (POSIX parity with the
19 * POSIX driver's O_EXCL temp) before any blob is allocated, then reserves a
20 * blob id + object dir. staged_write appends through the packed-block engine
21 * with no persistent block-0 fd (blk0_fd = -1). commit does the replace/
22 * insert against the catalog and consumes the handle on success. ngx-free;
23 * gated by BRIX_HAVE_SQLITE like the rest of the backend.
24 */

74/* sd_pblock_staged_open_as — staged open whose eventual committed row is owned
75 * by (uid, gid). The plain slot passes 0/0 (service); staged_open_cred
76 * (sd_pblock_cred.c) passes the requester's resolved catalog ids. */
77brix_sd_staged_t *
78sd_pblock_staged_open_as(brix_sd_instance_t *inst, const char *final_path,
79 mode_t mode, uint32_t uid, uint32_t gid, int *err_out)
80{
81 pblock_state_t *st = inst->state;
82 brix_sd_staged_t *handle;
83 pblock_staged_t *ps;
84
85 /* POSIX parity with the posix driver's staged temp (O_EXCL in the final
86 * directory): a missing parent collection fails HERE — before any blob
87 * is allocated or a single body byte is accepted — not at commit. */
88 if (pblock_catalog_parent_ok(st->cat, final_path) != 0) {
89 if (err_out != NULL) { *err_out = errno; }
90 return NULL;
91 }
92
93 /* F5: refuse an inode-quota-busting PUT before any body byte is accepted.
94 * Byte admission happens at commit (the size is unknown here); an overwrite
95 * of an existing row adds no inode, so admit 0 in that case. */
96 if (st->quota
97 && pblock_quota_admit(st, uid, 0,
98 pblock_catalog_lookup(st->cat, final_path, NULL) == 0 ? 0 : 1)
99 != 0)
100 {
101 if (err_out != NULL) { *err_out = errno; }
102 return NULL;
103 }
104
105 handle = calloc(1, sizeof(*handle));
106 ps = calloc(1, sizeof(*ps));
107 if (handle == NULL || ps == NULL) {
108 free(handle);
109 free(ps);
110 if (err_out != NULL) { *err_out = ENOMEM; }
111 return NULL;
112 }
113
114 if (pblock_gen_blob_id(ps->blob_id) != 0
115 || pblock_ensure_obj_dir(st, ps->blob_id) != 0)
116 {
117 if (err_out != NULL) { *err_out = errno; }
118 free(handle);
119 free(ps);
120 return NULL;
121 }
122
123 ps->st = st;
124 ps->block_size = st->block_size;
125 ps->size = 0;
126 ps->mode = mode;
127 ps->uid = uid;
128 ps->gid = gid;
129 snprintf(ps->final_path, sizeof(ps->final_path), "%s", final_path);
130 if (st->refs) { /* F10 */
131 ps->wv = brix_wverify_begin();
132 }
133 if (st->snap) { /* F6: block restore */
134 __atomic_add_fetch(&st->open_files, 1, __ATOMIC_RELEASE);
135 }
136 handle->inst = inst;
137 handle->state = ps;
138 if (st->audit) { /* F17 */
139 pblock_audit_log(st->cat, "staged_open", final_path, "", uid, gid, 0, 0);
140 }
141 return handle;
142}

151ssize_t
152sd_pblock_staged_write(brix_sd_staged_t *st, const void *buf, size_t len,
153 off_t off)
154{
155 pblock_staged_t *ps = st->state;
156 ssize_t n;
157
158 n = pblock_write_blocks(ps->st, ps->blob_id, ps->block_size, -1, buf, len,
159 off);
160 if (n > 0) {
161 if ((int64_t) off + n > ps->size) {
162 ps->size = (int64_t) off + n;
163 }
164 if (ps->wv != NULL) { /* F10: grow the dedup CRC */
165 (void) brix_wverify_update(ps->wv, buf, off, (size_t) n);
166 }
167 }
168 return n;
169}

171/* sd_pblock_staged_commit — publish the staged blocks atomically by inserting
172 * the final catalog row pointing at the staged blob id (the blocks simply become
173 * the final object — no copy or rename). On success the handle is consumed; on
174 * failure it stays valid and the caller must staged_abort to release it. */
175ngx_int_t
176sd_pblock_staged_commit(brix_sd_staged_t *st, int noreplace)
177{
178 pblock_staged_t *ps = st->state;
179 pblock_state_t *pst = ps->st;
180 pblock_meta meta, dmeta;
181 int rc;
182
183 rc = pblock_catalog_lookup(pst->cat, ps->final_path, &dmeta);
184 if (rc < 0) {
185 return NGX_ERROR;
186 }
187
188 /* F15: publishing over a leased name is a write to it — refuse while a
189 * live foreign lease exists (EBUSY; the handle stays valid for retry or
190 * staged_abort, same contract as a refused quota). */
191 if (pst->locks
192 && pblock_locks_ns_check(pst, ps->final_path, ps->uid) != 0)
193 {
194 return NGX_ERROR;
195 }
196
197 /* F5: byte admission BEFORE the destructive drop_dst — a refused commit
198 * must leave the existing object (and the usage rollup) untouched so the
199 * caller's staged_abort releases only the staged blocks. */
200 if (pblock_quota_admit(pst, ps->uid,
201 ps->size - (rc == 0 ? dmeta.size : 0), rc == 0 ? 0 : 1) != 0)
202 {
203 return NGX_ERROR;
204 }
205
206 if (rc == 0) {
207 if (noreplace) {
208 errno = EEXIST;
209 return NGX_ERROR;
210 }
211 /* F11: capture the prior object as a new version BEFORE the destructive
212 * drop. version_push holds its blob, so drop_dst's release only
213 * decrements (a copy-on-write transfer of the reference to the version
214 * row). A failed push is fail-open — the overwrite just keeps no
215 * history. */
216 if (pst->versions > 0) {
217 (void) pblock_hist_version_push(pst, ps->final_path, &dmeta);
218 }
219 if (sd_pblock_drop_dst(pst, ps->final_path, &dmeta) != NGX_OK) {
220 return NGX_ERROR;
221 }
222 }
223
224 memset(&meta, 0, sizeof(meta));
225 meta.is_dir = 0;
226 snprintf(meta.blob_id, sizeof(meta.blob_id), "%s", ps->blob_id);
227 meta.size = ps->size;
228 meta.block_size = ps->block_size;
229 meta.mtime = meta.ctime = pblock_now();
230 meta.mode = S_IFREG | (ps->mode & 0777);
231 meta.uid = ps->uid;
232 meta.gid = ps->gid;
233 snprintf(meta.xform, sizeof(meta.xform), "%s", /* F12/F13: record kind */
234 pblock_xform_name(pst->xform.kind));
235
236 /* F7: a crash here leaves the staged blocks on disk with no catalog row —
237 * the canonical orphan-blob residue pblock-fsck must detect and --gc. */
238 pblock_lab_crash(pst->lab, "mid_staged_commit");
239
240 if (pblock_catalog_put(pst->cat, ps->final_path, &meta) != 0) {
241 return NGX_ERROR;
242 }
243
244 if (pst->lab != NULL) { /* F9 */
245 if (rc == 0) {
246 pblock_anomaly_updated(pst, ps->final_path, dmeta.size,
247 dmeta.mtime);
248 } else {
249 pblock_anomaly_created(pst, ps->final_path);
250 }
251 }
252
253 if (pst->csi) { /* F3: tag the blob */
254 (void) pblock_csi_flush(pst, ps->blob_id, ps->size, ps->block_size,
255 0, INT64_MAX);
256 }
257
258 if (ps->wv != NULL) { /* F10: dedup fold */
259 if (pst->refs) {
260 uint32_t crc = 0;
261 off_t total = 0;
262 int ok;
263
264 ok = brix_wverify_expected(ps->wv, &crc, &total) == 0
265 && (int64_t) total == ps->size;
266 (void) pblock_refs_dedup_publish(pst, ps->final_path, &meta,
267 crc, ok);
268 }
269 brix_wverify_free(ps->wv);
270 }
271
272 if (pst->audit) { /* F17 */
273 char aux[32];
274
275 snprintf(aux, sizeof(aux), "w=%lld", (long long) ps->size);
276 pblock_audit_log(pst->cat, "commit", ps->final_path, aux,
277 ps->uid, ps->gid, 0, 0);
278 }
279
280 if (pst->snap) { /* F6: released — no longer blocks restore */
281 __atomic_sub_fetch(&pst->open_files, 1, __ATOMIC_RELEASE);
282 }
283 free(ps);
284 free(st);
285 return NGX_OK;
286}

288void
289sd_pblock_staged_abort(brix_sd_staged_t *st)
290{
291 pblock_staged_t *ps = st->state;
292
293 if (ps != NULL) {
294 /* Staged blobs are never tracked pre-publish, so an unconditional
295 * remove is correct even with refs armed. */
296 pblock_remove_blocks(ps->st, ps->blob_id, ps->size, ps->block_size);
297 brix_wverify_free(ps->wv); /* F10 (NULL-safe) */
298 if (ps->st->snap) { /* F6: released */
299 __atomic_sub_fetch(&ps->st->open_files, 1, __ATOMIC_RELEASE);
300 }
301 free(ps);
302 }
303 free(st);
304}
