1/*
2 * cpool.h — a thread-safe pool of opaque connections (generic engine).
3 *
4 * WHAT: brix_cpool_create/checkout/checkin/destroy over a caller-defined
5 * connection type (vtable {conn_size, connect, close}). Generalises
6 * lib/net/pool.c (now a thin brix_conn adapter over this) so the binary
7 * root:// path and the HTTP keep-alive metadata path share ONE
8 * slot/mutex/condvar/health-drop implementation.
9 * WHY: a connection is one-op-in-flight and NOT thread-safe; a multi-threaded
10 * consumer (the FUSE driver) needs N independent connections. The pool is
11 * the concurrency primitive; the transport is a vtable parameter.
12 * HOW: one mutex guards slot bookkeeping only (never held across connect/op);
13 * a condvar wakes a waiter on checkin; the vtable connects/closes the
14 * opaque per-slot memory. No goto. Clean-room (libc + pthread only).
15 */
