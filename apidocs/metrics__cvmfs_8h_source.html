30/* ---- per-repository (fqrn) counters --------------------------------------
31 * A site serves O(20) repositories, so the fqrn is a usable label — but the
32 * name arrives FROM THE WIRE, so the label set must be BOUNDED or a scanner
33 * minting random repo names explodes both the SHM and the Prometheus series
34 * space. A fixed slot table holds the first BRIX_CVMFS_REPO_SLOTS-1
35 * distinct fqrns seen; everything past capacity folds into the reserved
36 * last slot, exported as repo="_other". Slots register lock-free
37 * (state: EMPTY -> CAS -> CLAIMED(write name) -> READY); a claim race can
38 * duplicate a name across two slots, so RESOLUTION always returns the
39 * LOWEST-index READY match and the exporter skips later duplicates — all
40 * increments and all output converge on one slot per name. */
41#define BRIX_CVMFS_REPO_SLOTS 32
42#define BRIX_CVMFS_REPO_NAME_MAX 64
43
44#define BRIX_CVMFS_REPO_EMPTY 0u
45#define BRIX_CVMFS_REPO_CLAIMED 1u
46#define BRIX_CVMFS_REPO_READY 2u
47
48typedef struct {
49 ngx_atomic_t state; /* EMPTY / CLAIMED / READY */
50 char name[BRIX_CVMFS_REPO_NAME_MAX]; /* fqrn, NUL-terminated */
51
52 ngx_atomic_t requests_total[BRIX_CVMFS_CLASS_COUNT]; /* by class */
53 ngx_atomic_t files_accessed_total; /* CAS objects served OK (hit+fill) */
54 ngx_atomic_t cache_hits_total; /* served from the local store */
55 ngx_atomic_t cache_misses_total; /* needed an origin fill */
56 ngx_atomic_t fills_total; /* fills that published */
57 ngx_atomic_t fill_failures_total; /* fills that failed definitively */
58 ngx_atomic_t verify_failures_total; /* CAS mismatches (quarantined) */
59 ngx_atomic_t negative_hits_total; /* 404s absorbed by the worker memo */
60 ngx_atomic_t bytes_served_hit_total; /* LAN out, from cache */
61 ngx_atomic_t bytes_served_fill_total; /* LAN out, via a fresh fill */
62 ngx_atomic_t origin_bytes_total; /* WAN in, pulled from Stratum-1s */
63} ngx_brix_cvmfs_repo_metrics_t;

65/* ---- per-upstream (Stratum-1) counters -----------------------------------
66 * "Which of my origins am I pulling from, and how much?" — the RAL-vs-CERN
67 * question. The upstream "host:port" arrives FROM THE WIRE / config, so the
68 * label set is BOUNDED exactly like the repo table (INVARIANT #8): a fixed
69 * slot table holds the first BRIX_CVMFS_UPSTREAM_SLOTS-1 distinct hosts;
70 * the reserved last slot is exported upstream="_other". Registration and
71 * convergence use the same lock-free EMPTY->CLAIMED->READY lowest-index
72 * scheme as the repo table. */
73#define BRIX_CVMFS_UPSTREAM_SLOTS 16 /* >= a site's Stratum-1 allowlist */
74#define BRIX_CVMFS_UPSTREAM_NAME_MAX 300 /* matches sd_http last_origin buf */
75
76/* Fill-duration histogram: cumulative-le upper bounds in MILLISECONDS. The

78 * implicit +Inf. They bracket a LAN-adjacent fill (ms) through a cold-WAN
79 * fill (seconds). */
