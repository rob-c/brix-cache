1/*
2 * sd_stage_write.c - the two interposed WRITE paths of the write-stage decorator
3 * (section 12.2). Split from sd_stage.c (phase-79) to hold each half under the
4 * file-size cap; see sd_stage.h for the decorator contract and sd_stage_internal.h
5 * for the shared seam.
6 *
7 * WHAT: Implements the only paths the stage decorator interposes on:
8 * - the write-BACK object (Option A): a random-access write open lands on
9 * the stage store as a normal writable object; pwrite buffers there and
10 * fsync/close flush the whole object to the backend; and
11 * - the staged-upload path (HTTP PUT): a staged slot lands on the stage
12 * store and is FLUSHed to the backend on commit.
13 * Both flush through the one staging engine (brix_stage_run_inline_cred /
14 * brix_stage_submit FLUSH). A posix stage store is byte-equivalent to
15 * phase-63's local-temp promote.
16 * WHY: Keeping both write paths together isolates the ONLY interposed logic from
17 * the decorator core (forwarders + dispatch + driver table + lifecycle in
18 * sd_stage.c), so each file owns one concept and stays under 500 lines. The
19 * write-back path lets root:// kXR_write reuse the SAME stage mechanism as
20 * the staged (HTTP PUT) path.
21 * HOW: The open dispatch in sd_stage.c calls sd_stage_open_writeback here for
22 * write opens; the driver table in sd_stage.c routes byte-I/O and staged
23 * ops to the methods below. Each captured cred records the owner identity at
24 * open time so a deferred or async flush authenticates as the original user.
25 */

45/* Write-BACK object state (Option A / §12.2): a random-access write open lands on
46 * the STAGE STORE as a normal writable object; pwrite buffers there, and fsync/close
47 * flush the whole object to the backend through the one staging engine. This lets
48 * root:// kXR_write (direct pwrite at arbitrary offsets) use the SAME stage mechanism
49 * as the staged (HTTP PUT) path, so writethrough_flush's bespoke loop is retired.
50 * `cred` records the owner identity at open time so the flush can authenticate as
51 * the original user even if the flush runs on a background thread or after a restart. */
52typedef struct {
53 sd_stage_inst_state *is; /* back-ref: source / store / policy */
54 char key[PATH_MAX]; /* export-relative object key */
55 brix_sd_obj_t store_obj; /* the writable stage-store object */
56 off_t high_water; /* max offset+len written (flush size) */
57 unsigned dirty:1; /* written since the last flush */
58 brix_stage_cred_t cred; /* per-user identity (zeroed = service) */
59} sd_stage_wb_state;

61/* Record the caller's per-user identity into a durable stage cred slot.
62 *
63 * WHAT: Copies key/principal/cred_dir/fallback_deny from an optional
64 * brix_sd_cred_t into `dst` (a brix_stage_cred_t embedded in the durable
65 * write-back or staged state). A NULL or empty-key cred leaves `dst`
66 * untouched (zeroed = service-credential path).
67 *
68 * WHY: Both the write-back open and the staged open must capture the owner
69 * identity NOW, while the request context is live, so a deferred or async
70 * flush can authenticate to the backend as the original user rather than
71 * the service account. Sharing one copier keeps the two capture sites
72 * byte-identical and keeps each caller's branch count in check.
73 *
74 * HOW: 1. Return immediately unless cred is non-NULL with a non-empty key.
75 * 2. ngx_cpystrn key; principal/dir default to "" when their source
76 * pointer is NULL. 3. Narrow fallback_deny into dst->deny. */
77static void
78sd_stage_record_cred(brix_stage_cred_t *dst, const brix_sd_cred_t *cred)
79{
80 if (cred == NULL || cred->key == NULL || cred->key[0] == '\0') {
81 return;
82 }
83 ngx_cpystrn((u_char *) dst->key, (u_char *) cred->key, sizeof(dst->key));
84 ngx_cpystrn((u_char *) dst->principal,
85 (u_char *) (cred->principal ? cred->principal : ""),
86 sizeof(dst->principal));
87 ngx_cpystrn((u_char *) dst->dir,
88 (u_char *) (cred->cred_dir ? cred->cred_dir : ""),
89 sizeof(dst->dir));
90 dst->deny = (uint8_t) cred->fallback_deny;
91}

93/* Open the write-BACK object on the stage store (a normal writable object).
94 *
95 * WHAT: Opens a writable object on the stage store and wires a sd_stage_wb_state
96 * that carries the owner identity for the eventual flush.
97 *
98 * WHY: The returned obj carries the stage driver so its pwrite/pread/fsync/close
99 * dispatch to the write-back methods below; the stage-store object is held
100 * BY VALUE so the handle is self-contained. Recording the caller's cred
101 * in wb->cred ensures sd_stage_wb_flush can authenticate to the backend
102 * as the original user rather than the service account.
103 *
104 * HOW: Opens the stage STORE (always local) with a plain open — the store is
105 * service-owned. Copies key/principal/cred_dir/fallback_deny from `cred`
106 * into wb->cred when cred is non-NULL and cred->key is non-empty;
107 * otherwise wb->cred stays zeroed (service-credential path). */
108brix_sd_obj_t *
109sd_stage_open_writeback(brix_sd_instance_t *inst, sd_stage_inst_state *is,
110 const char *path, int sd_flags, mode_t mode,
111 const brix_sd_cred_t *cred, int *err_out)
112{
113 brix_sd_obj_t *store_obj;
114 brix_sd_obj_t *obj;
115 sd_stage_wb_state *wb;
116 int e = 0;
117
118 if (is->store->driver->open == NULL || is->store->driver->pwrite == NULL) {
119 if (err_out != NULL) { *err_out = ENOSYS; }
120 return NULL;
121 }
122 /* The stage store is service-owned (local POSIX); always open it plain. */
123 store_obj = is->store->driver->open(is->store, path, sd_flags, mode, &e);
124 if (store_obj == NULL) {
125 if (err_out != NULL) { *err_out = e; }
126 return NULL;
127 }
128
129 obj = calloc(1, sizeof(*obj));
130 wb = calloc(1, sizeof(*wb));
131 if (obj == NULL || wb == NULL) {
132 if (store_obj->driver->close != NULL) { store_obj->driver->close(store_obj); }
133 if (store_obj->heap_shell) { free(store_obj); }
134 free(obj);
135 free(wb);
136 if (err_out != NULL) { *err_out = ENOMEM; }
137 return NULL;
138 }
139
140 wb->is = is;
141 wb->high_water = 0;
142 wb->dirty = 0;
143 ngx_cpystrn((u_char *) wb->key, (u_char *) path, sizeof(wb->key));
144 wb->store_obj = *store_obj; /* adopt by value */
145 if (store_obj->heap_shell) { free(store_obj); }
146 wb->store_obj.heap_shell = 0;
147
148 /* Record the owner identity for the flush; zeroed cred = service account. */
149 sd_stage_record_cred(&wb->cred, cred);
150
151 obj->driver = &brix_sd_stage_driver; /* → the write-back methods below */
152 obj->inst = inst;
153 obj->fd = wb->store_obj.fd; /* expose the stage fd (posix sendfile) */
154 obj->snap = wb->store_obj.snap;
155 obj->state = wb;
156 obj->heap_shell = 1; /* malloc'd shell; caller frees post-copy */
157 return obj;
158}

162/* Flush the buffered stage object to the backend through the one staging engine.
163 * Persists the stage buffer first (fsync), then FLUSHes store→source by key. On
164 * success clears the dirty flag; on failure keeps the stage copy for retry. */
165static ngx_int_t
166sd_stage_wb_flush(sd_stage_wb_state *wb)
167{
168 sd_stage_inst_state *is = wb->is;
169 ngx_int_t rc;
170
171 if (wb->store_obj.driver->fsync != NULL) {
172 (void) wb->store_obj.driver->fsync(&wb->store_obj);
173 }
174 if (!wb->dirty) {
175 return NGX_OK;
176 }
177 rc = brix_stage_run_inline_cred(BRIX_STAGE_FLUSH, is->store, wb->key,
178 is->source, wb->key,
179 (wb->cred.key[0] != '\0') ? &wb->cred : NULL);
180 if (rc == NGX_OK) {
181 wb->dirty = 0;
182 }
183 return rc;
184}

259/* Common staged_open body shared by sd_stage_staged_open and
260 * sd_stage_staged_open_cred.
261 *
262 * WHAT: Opens a staged upload slot on the STORE (local — always plain) and
263 * records the optional per-user cred in ss->cred for the commit-time flush.
264 *
265 * WHY: The store is service-owned so its staged_open is always plain; the
266 * credential is only needed at flush time (store→source) and must be
267 * captured now while the request context is still live.
268 *
269 * HOW: Allocate ss + h, wire them, copy cred when key is non-empty. */
270static brix_sd_staged_t *
271sd_stage_staged_open_inner(brix_sd_instance_t *inst, sd_stage_inst_state *is,
272 const char *final_path, mode_t mode,
273 const brix_sd_cred_t *cred, int *err_out)
274{
275 sd_stage_staged_state *ss;
276 brix_sd_staged_t *h;
277 brix_sd_staged_t *inner;
278 int err = 0;
279
280 if (is->store->driver->staged_open == NULL) {
281 if (err_out != NULL) { *err_out = ENOSYS; }
282 return NULL;
283 }
284 inner = is->store->driver->staged_open(is->store, final_path, mode, &err);
285 if (inner == NULL) {
286 if (err_out != NULL) { *err_out = err ? err : EIO; }
287 return NULL;
288 }
289 ss = calloc(1, sizeof(*ss));
290 h = calloc(1, sizeof(*h));
291 if (ss == NULL || h == NULL) {
292 if (is->store->driver->staged_abort != NULL) {
293 is->store->driver->staged_abort(inner);
294 }
295 free(ss);
296 free(h);
297 if (err_out != NULL) { *err_out = ENOMEM; }
298 return NULL;
299 }
300 ss->is = is;
301 ss->inner = inner;
302 ngx_cpystrn((u_char *) ss->key, (u_char *) final_path, sizeof(ss->key));
303
304 /* Record the owner identity for the flush; zeroed cred = service account. */
305 sd_stage_record_cred(&ss->cred, cred);
306
307 h->inst = inst;
308 h->state = ss;
309 return h;
310}

350/* Publish the buffered object on the stage store, then FLUSH it to the backend
351 * through the one staging engine. On a successful flush the stage buffer copy is
352 * dropped; on a failed flush it is KEPT (durability preserved for retry, section
353 * 16). Consumes the handle. */
354ngx_int_t
355sd_stage_staged_commit(brix_sd_staged_t *st, int noreplace)
356{
357 sd_stage_staged_state *ss = st->state;
358 sd_stage_inst_state *is = ss->is;
359 brix_sd_instance_t *store = is->store;
360 brix_sd_instance_t *source = is->source;
361 ngx_int_t rc;
362
363 /* 1. publish the buffered object on the stage store. */
364 if (store->driver->staged_commit(ss->inner, noreplace) != NGX_OK) {
365 int e = errno;
366 if (store->driver->staged_abort != NULL) {
367 store->driver->staged_abort(ss->inner);
368 }
369 free(ss);
370 free(st);
371 errno = e ? e : EIO;
372 return NGX_ERROR;
373 }
374 /* ss->inner is consumed by the commit above. */
375
376 /* 2a. ASYNC write-back (SP4): the object is durable on the stage store now, so
377 * the commit succeeds immediately; the scheduler flushes it to the backend and
378 * drops the stage copy on completion. The export anchor rides on the durable
379 * record so a restart-reconcile can rebuild both tiers and re-flush (§11.3).
380 * The owner cred is embedded in the opts so the scheduler can authenticate as
381 * the original user when it drains the queue (non-NULL only when a key was
382 * recorded at staged_open time). */
383 if (is->policy.flush_mode == BRIX_WT_MODE_ASYNC) {
384 brix_stage_opts_t o;
385
386 ngx_memzero(&o, sizeof(o));
387 o.async = 1;
388 o.export_root = (is->root_canon[0] != '\0') ? is->root_canon : NULL;
389 o.cred = (ss->cred.key[0] != '\0') ? &ss->cred : NULL;
390 /* phase74-fp: argument order verified against brix_stage_submit(kind,
391 * src, src_key, dst, dst_key, opts) — a FLUSH moves bytes FROM the
392 * stage `store` TO the backend instance (locally named `source`), so
393 * store is the src and source the dst; the name swap is deliberate. */
394 (void) brix_stage_submit(BRIX_STAGE_FLUSH, store, ss->key, source, /* NOLINT(readability-suspicious-call-argument) */
395 ss->key, &o);
396 free(ss);
397 free(st);
398 return NGX_OK;
399 }
400
401 /* 2b. SYNC write-back: flush inline and reflect the result, threading the
402 * owner cred so the backend driver uses the per-user proxy rather than the
403 * service credential. */
404 /* phase74-fp: same verified src/dst order as the async submit above —
405 * FLUSH reads from the stage store and writes to the backend `source`. */
406 rc = brix_stage_run_inline_cred(BRIX_STAGE_FLUSH, store, ss->key, source, /* NOLINT(readability-suspicious-call-argument) */
407 ss->key,
408 (ss->cred.key[0] != '\0') ? &ss->cred : NULL);
409
410 /* 3. on success drop the stage buffer copy; on failure keep it for retry. */
411 if (rc == NGX_OK && store->driver->unlink != NULL) {
412 (void) store->driver->unlink(store, ss->key, 0);
413 }
414
415 free(ss);
416 free(st);
417 return rc;
418}
