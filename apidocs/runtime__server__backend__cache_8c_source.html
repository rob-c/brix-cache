25/* Load the CVMFS repo master public key PEM named by `path` into the cache
26 * policy (phase-85 F1). Config-time, cf->pool-owned (cycle lifetime — workers
27 * inherit the pointer through the registered policy copy). The file may hold
28 * several concatenated PEM keys (CVMFS key rotation); content is validated by
29 * the OpenSSL PEM parser at verify time, here only shape-checked. [emerg] on
30 * any failure — a verifying proxy with an unloadable trust anchor must not
31 * start. */
32static ngx_int_t
33brix_tier_load_master_key(ngx_conf_t *cf, const ngx_str_t *path,
34 brix_cache_policy_t *pol)
35{
36 char pathz[1024];
37 u_char *buf;
38 ssize_t n;
39 off_t size;
40 ngx_fd_t fd;
41 ngx_file_info_t fi;
42
43 if (path->len >= sizeof(pathz)) {
44 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
45 "brix_cvmfs_verify_manifest: key path too long");
46 return NGX_ERROR;
47 }
48 ngx_cpystrn((u_char *) pathz, path->data, path->len + 1);
49
50 fd = open(pathz, O_RDONLY | O_CLOEXEC); /* vfs-seam-allow: config-domain trust-anchor PEM (not export storage) */
51 if (fd == -1) {
52 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
53 "brix_cvmfs_verify_manifest: cannot open \"%s\"", pathz);
54 return NGX_ERROR;
55 }
56 if (ngx_fd_info(fd, &fi) == -1
57 || (size = ngx_file_size(&fi)) <= 0 || size > 65536)
58 {
59 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
60 "brix_cvmfs_verify_manifest: \"%s\" is empty, unreadable "
61 "or larger than 64KB", pathz);
62 (void) close(fd);
63 return NGX_ERROR;
64 }
65 buf = ngx_palloc(cf->pool, (size_t) size + 1);
66 if (buf == NULL) {
67 (void) close(fd);
68 return NGX_ERROR;
69 }
70 n = read(fd, buf, (size_t) size); /* vfs-seam-allow: config-domain trust-anchor PEM (not export storage) */
71 (void) close(fd);
72 if (n != (ssize_t) size) {
73 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
74 "brix_cvmfs_verify_manifest: short read on \"%s\"", pathz);
75 return NGX_ERROR;
76 }
77 buf[size] = '\0';
78 if (ngx_strstr(buf, "BEGIN PUBLIC KEY") == NULL) {
79 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
80 "brix_cvmfs_verify_manifest: \"%s\" holds no PEM public key",
81 pathz);
82 return NGX_ERROR;
83 }
84 pol->cvmfs_master_pub = buf;
85 pol->cvmfs_master_pub_len = (size_t) size;
86 return NGX_OK;
87}

89/* Parse + validate the brix_cache_peers ring (phase-85 F8) and record it on
90 * the backend registry. Each member token is "host:port"; this node's own slot
91 * is "self=host:port" (the mesh needs every node to carry the IDENTICAL list so
92 * rendezvous ownership agrees, so self is marked, never omitted). Operator
93 * errors — malformed authority, no/duplicate self, fewer than 2 members, more
94 * than 16 — are [emerg], failing nginx -t. */
95static ngx_int_t
96brix_tier_register_cache_peers(ngx_conf_t *cf,
97 ngx_http_brix_shared_conf_t *common)
98{
99 char hosts[16][256];
100 int ports[16];
101 int self = -1;
102 ngx_str_t *tok = common->cache_peers->elts;
103 ngx_uint_t i;
104
105 if (common->cache_peers->nelts < 2) {
106 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
107 "brix_cache_peers: a mesh needs at least 2 ring members "
108 "(self=host:port plus one sibling)");
109 return NGX_ERROR;
110 }
111 if (common->cache_peers->nelts > 16) {
112 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
113 "brix_cache_peers: at most 16 ring members supported");
114 return NGX_ERROR;
115 }
116
117 for (i = 0; i < common->cache_peers->nelts; i++) {
118 char buf[300];
119 char *auth = buf;
120 char *colon;
121 long port;
122
123 if (tok[i].len >= sizeof(buf)) {
124 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
125 "brix_cache_peers: member \"%V\" is too long", &tok[i]);
126 return NGX_ERROR;
127 }
128 ngx_memcpy(buf, tok[i].data, tok[i].len);
129 buf[tok[i].len] = '\0';
130
131 if (ngx_strncmp(auth, "self=", sizeof("self=") - 1) == 0) {
132 if (self >= 0) {
133 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
134 "brix_cache_peers: more than one self= member");
135 return NGX_ERROR;
136 }
137 self = (int) i;
138 auth += sizeof("self=") - 1;
139 }
140 colon = strrchr(auth, ':');
141 if (colon == NULL || colon == auth || colon[1] == '\0') {
142 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
143 "brix_cache_peers: member \"%V\" is not host:port", &tok[i]);
144 return NGX_ERROR;
145 }
146 port = strtol(colon + 1, NULL, 10);
147 if (port < 1 || port > 65535) {
148 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
149 "brix_cache_peers: member \"%V\" has an invalid port",
150 &tok[i]);
151 return NGX_ERROR;
152 }
153 *colon = '\0';
154 if (ngx_strlen(auth) >= sizeof(hosts[0])) {
155 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
156 "brix_cache_peers: member \"%V\" host is too long", &tok[i]);
157 return NGX_ERROR;
158 }
159 ngx_cpystrn((u_char *) hosts[i], (u_char *) auth, sizeof(hosts[i]));
160 ports[i] = (int) port;
161 }
162
163 if (self < 0) {
164 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
165 "brix_cache_peers: mark this node's own ring slot with "
166 "self=host:port");
167 return NGX_ERROR;
168 }
169
170 brix_vfs_backend_config_cache_peers(common->root_canon,
171 (const char (*)[256]) hosts, ports,
172 (int) common->cache_peers->nelts, self);
173 return NGX_OK;
174}

176/* Fill the read-through cache policy from the merged srv-conf knobs. Split out
177 * of brix_tier_register_cache_store so each function's branching stays within
178 * the readability gate — pure knob-to-policy plumbing, no validation here. */
179static void
180brix_tier_fill_cache_policy(ngx_http_brix_shared_conf_t *common,
181 brix_cache_policy_t *polp)
182{
183 brix_cache_policy_t pol;
184
185 ngx_memzero(&pol, sizeof(pol));
186 pol.enabled = 1;
187 pol.max_file_size = common->cache_max_object;
188 /* Read-fill admission (bridged from the srv conf at finalisation): deny/allow
189 * prefixes + include regex gate the composable sd_cache fill for parity with
190 * write-through and the legacy cache_origin admit. NULL => no filter. */
191 pol.deny_prefixes = common->cache_deny_prefixes;
192 pol.allow_prefixes = common->cache_allow_prefixes;
193 pol.include_regex = common->cache_include_re;
194 /* Documented defaults (90/80) applied here: the merge keeps the pair UNSET
195 * so the stream reaper merge can detect an explicit setting. */
196 pol.evict_at = (common->cache_evict_at == NGX_CONF_UNSET_UINT)
197 ? 90 : common->cache_evict_at;
198 pol.evict_to = (common->cache_evict_to == NGX_CONF_UNSET_UINT)
199 ? 80 : common->cache_evict_to;
200 pol.meta_mode = (int) common->cache_meta_mode;
201 pol.batch_cinfo = (common->cache_batch_cinfo == 2)
202 ? -1 : (int) common->cache_batch_cinfo;
203 pol.l1_entries = common->cache_index_cache;
204 pol.slice_size = common->cache_slice_size;
205 /* phase-68 digest-verification mode — the posix-store constraint it
206 * carries is validated by the caller against the parsed store driver. */
207 pol.verify = (common->cache_verify_mode == NGX_CONF_UNSET_UINT)
208 ? BRIX_CACHE_VERIFY_OFF
209 : (brix_cache_verify_mode_e) common->cache_verify_mode;
210 pol.cvmfs_manifest_ttl = common->cache_manifest_ttl;
211 pol.cvmfs_offline_ttl = common->cache_offline_ttl;
212 if (common->cache_quarantine_dir.len > 0) {
213 ngx_cpystrn((u_char *) pol.quarantine_dir,
214 common->cache_quarantine_dir.data,
215 ngx_min(common->cache_quarantine_dir.len + 1,
216 sizeof(pol.quarantine_dir)));
217 }
218 *polp = pol;
219}

221/* Parse the cache_store URL and record its tier cfg + read-through policy on the
222 * backend registry. Split out of brix_tier_register_stores so each function's
223 * branching stays within the readability gate. Operator errors are [emerg]. */
224ngx_int_t
225brix_tier_register_cache_store(ngx_conf_t *cf,
226 ngx_http_brix_shared_conf_t *common)
227{
228 char err[256];
229 brix_tier_cfg_t cfg;
230 brix_cache_policy_t pol;
231 brix_tier_parse_t tp = { cf, &cfg, err, sizeof(err) };
232
233 if (brix_tier_parse_store(&tp, &common->cache_store,
234 common->cache_store_args, BRIX_TIER_CACHE) != NGX_OK)
235 {
236 return NGX_ERROR; /* [emerg] already logged */
237 }
238 brix_tier_fill_cache_policy(common, &pol);
239 /* phase-68: digest verification on fill (cvmfs-cas today). The verify
240 * runs on the staged temp BEFORE commit, which needs the store's
241 * staged_path — a local posix store; reject other stores loudly. */
242 if (pol.verify == BRIX_CACHE_VERIFY_CVMFS_CAS
243 && ngx_strcmp(cfg.driver, "posix") != 0)
244 {
245 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
246 "brix_cache_verify cvmfs-cas requires a local posix "
247 "cache store (got \"%s\")", cfg.driver);
248 return NGX_ERROR;
249 }
250 /* phase-85 F1: brix_cvmfs_verify_manifest — load the repo master public
251 * key(s) once at config time; the fill spine verifies every MANIFEST-class
252 * fill's signature chain against it before publish. Same posix-store
253 * constraint as cvmfs-cas (the verify reads the staged part path). */
254 if (common->cache_cvmfs_master_key.len > 0) {
255 if (ngx_strcmp(cfg.driver, "posix") != 0) {
256 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
257 "brix_cvmfs_verify_manifest requires a local posix "
258 "cache store (got \"%s\")", cfg.driver);
259 return NGX_ERROR;
260 }
261 if (brix_tier_load_master_key(cf, &common->cache_cvmfs_master_key,
262 &pol) != NGX_OK)
263 {
264 return NGX_ERROR; /* [emerg] already logged */
265 }
266 }
267 brix_vfs_backend_config_cache_store(common->root_canon, &cfg, &pol);
268
269 /* Phase-85 F7: the optional cold tier under the cache — its own store URL,
270 * governed by the hot cache's policy (no separate knobs). */
271 if (common->cache_cold_store.len > 0) {
272 brix_tier_cfg_t ccfg;
273 brix_tier_parse_t ctp = { cf, &ccfg, err, sizeof(err) };
274
275 if (brix_tier_parse_store(&ctp, &common->cache_cold_store,
276 common->cache_cold_store_args, BRIX_TIER_CACHE) != NGX_OK)
277 {
278 return NGX_ERROR; /* [emerg] already logged */
279 }
280 brix_vfs_backend_config_cache_cold_store(common->root_canon, &ccfg);
281 }
282
283 /* Phase-85 F8: the sibling-mesh ring under the cache tier. */
284 if (common->cache_peers != NULL
285 && brix_tier_register_cache_peers(cf, common) != NGX_OK)
286 {
287 return NGX_ERROR; /* [emerg] already logged */
288 }
289 return NGX_OK;
290}
