1/*
2 * sockstats.c — socket statistics + time/address formatting for firefly.
3 *
4 * WHAT: Three small helpers the firefly emitter needs that have no existing
5 * reuse point in the tree: (1) read a connected socket's transferred-byte and
6 * RTT counters from the kernel (TCP_INFO), (2) render "now" as the firefly
7 * microsecond UTC timestamp, and (3) render a socket end's IP + port as
8 * strings with the SciTags address-family indicator.
9 *
10 * WHY: XRootD's firefly carries usage.received/sent (from TCP_INFO), netlink.rtt,
11 * and flow-id.{src,dst}-ip/port. The codebase's brix_format_iso8601() only
12 * emits millisecond ".000Z" precision, and there is no sockaddr→"ip:port"
13 * helper, so firefly needs its own. All three are best-effort: on any failure
14 * they zero/blank the output and return NGX_DECLINED — packet marking never
15 * fails a transfer.
16 *
17 * HOW: TCP_INFO is read into a RAW buffer and the needed fields are pulled by
18 * their fixed kernel-ABI byte offsets. glibc's <netinet/tcp.h> struct tcp_info
19 * is a frozen subset that lacks tcpi_bytes_acked/received, and the kernel's
20 * <linux/tcp.h> struct conflicts with it, so neither struct can be used
21 * directly — the offset read is the portable approach (the same one `ss` uses).
22 * The kernel tcp_info layout is append-only and stable across versions.
23 */

37/* Fixed offsets into the kernel `struct tcp_info` (bytes). 8-byte u8 preamble,
38 * then 24 u32 fields (tcpi_rtt is the 16th, at 8 + 15*4 = 68), then 8-aligned
39 * u64 fields: pacing_rate(104), max_pacing_rate(112), bytes_acked(120),
40 * bytes_received(128). */
41#define PMARK_TCPI_OFF_RTT 68
42#define PMARK_TCPI_OFF_BYTES_ACKED 120
43#define PMARK_TCPI_OFF_BYTES_RECEIVED 128

47ngx_int_t
48brix_pmark_sockstats(int fd, brix_pmark_sockstats_t *st)
49{
50 ngx_memzero(st, sizeof(*st));
51
52#if defined(__linux__)
53 {
54 unsigned char buf[256];
55 socklen_t len = sizeof(buf);
56 uint32_t rtt;
57 uint64_t acked, recvd;
58
59 if (fd < 0
60 || getsockopt(fd, IPPROTO_TCP, TCP_INFO, buf, &len) != 0)
61 {
62 return NGX_DECLINED;
63 }
64
65 if (len >= PMARK_TCPI_OFF_RTT + sizeof(rtt)) {
66 ngx_memcpy(&rtt, buf + PMARK_TCPI_OFF_RTT, sizeof(rtt));
67 st->rtt_us = rtt;
68 }
69 if (len >= PMARK_TCPI_OFF_BYTES_RECEIVED + sizeof(recvd)) {
70 ngx_memcpy(&acked, buf + PMARK_TCPI_OFF_BYTES_ACKED, sizeof(acked));
71 ngx_memcpy(&recvd, buf + PMARK_TCPI_OFF_BYTES_RECEIVED,
72 sizeof(recvd));
73 st->bytes_sent = acked;
74 st->bytes_recv = recvd;
75 }
76 return NGX_OK;
77 }
78#else
79 (void) fd;
80 return NGX_DECLINED;
81#endif
82}

85void
86brix_pmark_iso8601_now(char *buf, size_t buflen)
87{
88 struct timeval tv;
89 struct tm tm;
90
91 if (buflen == 0) {
92 return;
93 }
94 buf[0] = '\0';
95
96 if (gettimeofday(&tv, NULL) != 0 || gmtime_r(&tv.tv_sec, &tm) == NULL) {
97 return;
98 }
99
100 /* yyyy-mm-ddThh:mm:ss.uuuuuu+00:00 — the scitags firefly format. */
101 (void) snprintf(buf, buflen,
102 "%04d-%02d-%02dT%02d:%02d:%02d.%06ld+00:00",
103 tm.tm_year + 1900, tm.tm_mon + 1, tm.tm_mday,
104 tm.tm_hour, tm.tm_min, tm.tm_sec, (long) tv.tv_usec);
105}

108ngx_int_t
109brix_pmark_endpoint(int fd, int which, char *ip, size_t iplen, int *port,
110 char *afi)
111{
112 struct sockaddr_storage ss;
113 socklen_t len = sizeof(ss);
114 int rc;
115
116 if (iplen) {
117 ip[0] = '\0';
118 }
119 *port = 0;
120 *afi = '4';
121
122 if (fd < 0) {
123 return NGX_DECLINED;
124 }
125
126 /* Zero-init so ss_family is well-defined even if the kernel writes fewer
127 * bytes than sizeof(ss) (analyzer-proven read of an indeterminate value
128 * otherwise). */
129 ngx_memzero(&ss, sizeof(ss));
130
131 rc = which == 0 ? getpeername(fd, (struct sockaddr *) &ss, &len)
132 : getsockname(fd, (struct sockaddr *) &ss, &len);
133 if (rc != 0) {
134 return NGX_DECLINED;
135 }
136
137 if (ss.ss_family == AF_INET) {
138 struct sockaddr_in *sin = (struct sockaddr_in *) &ss;
139 *port = ntohs(sin->sin_port);
140 *afi = '4';
141 return inet_ntop(AF_INET, &sin->sin_addr, ip, (socklen_t) iplen)
142 ? NGX_OK : NGX_DECLINED;
143 }
144
145 if (ss.ss_family == AF_INET6) {
146 struct sockaddr_in6 *sin6 = (struct sockaddr_in6 *) &ss;
147 *port = ntohs(sin6->sin6_port);
148
149 if (IN6_IS_ADDR_V4MAPPED(&sin6->sin6_addr)) {
150 /* ::ffff:a.b.c.d — report as plain IPv4 (matches XRootD afi). */
151 struct in_addr v4;
152 ngx_memcpy(&v4, sin6->sin6_addr.s6_addr + 12, sizeof(v4));
153 *afi = '4';
154 return inet_ntop(AF_INET, &v4, ip, (socklen_t) iplen)
155 ? NGX_OK : NGX_DECLINED;
156 }
157
158 *afi = '6';
159 return inet_ntop(AF_INET6, &sin6->sin6_addr, ip, (socklen_t) iplen)
160 ? NGX_OK : NGX_DECLINED;
161 }
162
163 return NGX_DECLINED;
164}
