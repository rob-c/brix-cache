7/*
8 * WHAT: Relay upstream XRootD responses to the client, handling special cases:
9 * bound-secondary lazy-open (synthetic kXR_open), kXR_wait retry,
10 * kXR_redirect follow-through, fhandle translation, path audit,
11 * and streaming kXR_oksofar / mid-stream kXR_wait.
12 *
13 * WHY: The transparent proxy must translate upstream file handles to local
14 * handles, absorb transient kXR_wait responses with timed retries,
15 * silently follow redirects (up to 3 hops), emit audit records for
16 * path operations, and support streaming multi-chunk reads — all
17 * while keeping the client unaware of upstream topology changes.

20 * from the upstream response. It handles each special case in sequence:
21 * lazy-open → wait-retry → redirect-follow → audit → fhandle-translation
22 * → read/write tracking → close audit → build/send relay buffer.

32/* kXR_redirect follow-through: when the upstream returns a redirect (and we are
33 * under the 3-hop limit), parse the "host:port" target, tear down the current
34 * upstream, and reconnect to it re-issuing the saved request. Returns 1 when
35 * the reconnect is in progress and the caller must return; 0 to fall through and
36 * relay the redirect to the client (not a redirect, malformed target, or the
37 * reconnect attempt failed). */
38static int
39brix_proxy_relay_try_redirect(brix_proxy_ctx_t *proxy, ngx_connection_t *c,
40 uint16_t status, u_char *body, uint32_t dlen)
41{
42 if (status == kXR_redirect && body != NULL && dlen > 0
43 && proxy->redirect_count < 3)
44 {
45 /* Payload is "host:port\0" */
46 char *target = (char *) body;
47 char *colon = ngx_strchr(target, ':');
48
49 if (colon != NULL) {
50 *colon = '\0';
51 if (proxy->redirect_host.data != NULL) {
52 ngx_free(proxy->redirect_host.data);
53 }
54 proxy->redirect_host.len = (size_t)(colon - target);
55 proxy->redirect_host.data = ngx_alloc(proxy->redirect_host.len + 1,
56 c->log);
57 if (proxy->redirect_host.data != NULL) {
58 (void) brix_cbuf_copy((char *) proxy->redirect_host.data,
59 proxy->redirect_host.len + 1, target, proxy->redirect_host.len);
60 {
61 char *endp;
62 long pval;
63 errno = 0;
64 pval = strtol(colon + 1, &endp, 10);
65 if (errno != 0 || endp == colon + 1 || pval < 1 || pval > 65535) {
66 ngx_log_error(NGX_LOG_WARN, c->log, 0,
67 "xrootd proxy: invalid redirect port, dropping");
68 } else {
69 proxy->redirect_port = (uint16_t) pval;
70
71 proxy->redirect_count++;
72 ngx_log_error(NGX_LOG_INFO, c->log, 0,
73 "xrootd proxy: following redirect %d to %s:%d",
74 proxy->redirect_count,
75 proxy->redirect_host.data,
76 (int) proxy->redirect_port);
77
78 /* Close current upstream, reconnect to redirected target */
79 if (proxy->conn != NULL) {
80 ngx_close_connection(proxy->conn);
81 proxy->conn = NULL;
82 }
83 if (proxy->resp_body != NULL) {
84 ngx_free(proxy->resp_body);
85 proxy->resp_body = NULL;
86 }
87 /* Reuse the wait-retry copy to re-issue the request */
88 proxy->saved_req = proxy->wait_retry_req;
89 proxy->saved_req_len = proxy->wait_retry_req_len;
90 proxy->saved_local_fh = proxy->wait_retry_local_fh;
91 proxy->wait_retry_req = NULL;
92 proxy->wait_retry_req_len = 0;
93
94 proxy->state = XRD_PX_CONNECTING;
95 proxy->bs_phase = XRD_PX_BS_HANDSHAKE;
96 proxy->rhdr_pos = 0;
97 proxy->resp_dlen = 0;
98 proxy->resp_body_pos = 0;
99
100 if (brix_proxy_connect(proxy, c, proxy->conf) == NGX_OK) {
101 return 1; /* reconnect in progress; dispatches saved_req */
102 }
103 /* reconnect failed — fall through to relay redirect */
104 ngx_log_error(NGX_LOG_ERR, c->log, 0,
105 "xrootd proxy: redirect follow failed, relaying");
106 }
107 }
108 }
109 }
110 }
111
112 return 0;
113}

261{
262 int local_fh = proxy->fwd_local_fh;
263
264 if ((status != kXR_ok && status != kXR_oksofar)
265 || local_fh < 0 || local_fh >= BRIX_MAX_FILES)
266 {
267 return;
268 }
269

272 case kXR_pgread:

275 BRIX_PROXY_METRIC_INC(ctx, reads_total);

277 BRIX_PROXY_UP_INC(proxy, reads_total);
278 BRIX_PROXY_UP_ADD(proxy, read_bytes_total, dlen);
279 break;
280 case kXR_write:
281 case kXR_pgwrite:
282 case kXR_writev:
283 proxy->fh_map[local_fh].bytes_written += proxy->fwd_payload_len;
284 BRIX_PROXY_METRIC_INC(ctx, writes_total);

286 BRIX_PROXY_UP_INC(proxy, writes_total);

313 uint16_t status, u_char *body, uint32_t dlen, size_t *total)
314{
315 u_char *buf;

375void
376brix_proxy_relay_to_client(brix_proxy_ctx_t *proxy)
377{
378 brix_ctx_t *ctx = proxy->client_ctx;
379 ngx_connection_t *c = proxy->client_conn;
380 uint16_t status = proxy->resp_status;
381 uint32_t dlen = proxy->resp_dlen;
382 u_char *body = proxy->resp_body;
383 size_t total;
384 u_char *buf;
385
386 /* Tap: emit the upstream response metadata (status/dlen) to the observation
387 * tap, keyed to the client's streamid. */
388 brix_proxy_relay_emit_tap(proxy, status, dlen);
389
390 /* lazy open (bound secondary): handle synthetic kXR_open response */
391 if (brix_proxy_relay_lazy_open(proxy)) {
392 return;
393 }
394
395 if (brix_proxy_relay_absorb_wait(proxy, ctx, c, status, body, dlen)) {
396 return;
397 }
398
399 /* kXR_wait exhausted retries — free retry buffer and relay the wait to client */
400 brix_proxy_relay_wait_exhausted(proxy, status);
401
402 /* kXR_redirect follow-through (transparently reconnect to the target). */
403 if (brix_proxy_relay_try_redirect(proxy, c, status, body, dlen)) {
404 return;
405 }
406
407 brix_proxy_relay_path_audit(proxy, ctx, status);
408
409 brix_proxy_relay_open_status(proxy, ctx, status, body, dlen);
410
411 brix_proxy_relay_io_metrics(proxy, ctx, status, dlen);
412
413 brix_proxy_relay_close_status(proxy, ctx, status);
414
415 buf = brix_proxy_build_relay_buffer(proxy, c, status, body, dlen, &total);
416 if (buf == NULL) {
417 return;
418 }
419
420 /*
421 * Save resptype from kXR_status body[7] before freeing.
422 * kXR_PartialResult (0x01) means upstream will send more kXR_status frames;
423 * we must stay FORWARDING rather than going IDLE after relaying this chunk.
424 */
425 u_char resptype = (status == kXR_status && body != NULL && dlen >= 8)
426 ? body[7] : 0;
427
428 /* Free the heap-allocated body now that we've copied it */
429 if (proxy->resp_body != NULL) {
430 ngx_free(proxy->resp_body);
431 proxy->resp_body = NULL;
432 }
433
434 ngx_log_debug(NGX_LOG_DEBUG_STREAM, c->log, 0,
435 "xrootd proxy: relay reqid=%d status=%d dlen=%uz resptype=%d",
436 (int) proxy->fwd_reqid, (int) status, (size_t) dlen,
437 (int) resptype);
438
439 if (brix_proxy_relay_streaming_frame(proxy, ctx, c, status, resptype, buf,
440 total))
441 {
442 return;
443 }
444
445 /* Final response — transition back to client loop.
446 * Cancel any pending kXR_wait retry timer: a final response arrived
447 * before the timer fired (spontaneous upstream send). Free the saved
448 * retry buffer too; wait_handler checks req==NULL as a no-op guard. */
449 brix_proxy_relay_finish_final(proxy, ctx, c, buf, total);
450}
