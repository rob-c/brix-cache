21/* Recover (mdLen, pfxLen, dbL) exactly as XrdSsiTaskReal::GetResp does. */
22static void
23getresp(const unsigned char *buf, size_t n,
24 unsigned *pfx, unsigned *md, long *db)
25{
26 *pfx = ((unsigned) buf[2] << 8) | buf[3]; /* ntohs(pfxLen) */
27 *md = ((unsigned) buf[4] << 24) | ((unsigned) buf[5] << 16)
28 | ((unsigned) buf[6] << 8) | buf[7]; /* ntohl(mdLen) */
29 *db = (long) n - (long) *md - (long) *pfx;
30}

32static void
33test_full_response_layout(void)
34{
35 unsigned char out[64];
36 size_t n = brix_ssi_reply_build(BRIX_SSI_ATTN_FULL,
37 (const unsigned char *) "m", 1,
38 (const unsigned char *) "hi", 2, out);
39 CHECK(n == 19); /* 16 prefix + 1 md + 2 data */
40 CHECK(brix_ssi_reply_len(1, 2) == 19);
41 CHECK(out[0] == ':'); /* fullResp tag */
42
43 unsigned pfx, md; long db;
44 getresp(out, n, &pfx, &md, &db);
45 CHECK(pfx == 16);
46 CHECK(md == 1);
47 CHECK(db == 2); /* client recovers data len */
48 /* metadata at +pfx, data at +md+pfx (GetResp) */
49 CHECK(memcmp(out + pfx, "m", 1) == 0);
50 CHECK(memcmp(out + md + pfx, "hi", 2) == 0);
51}
