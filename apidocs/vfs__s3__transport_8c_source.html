1/* client/lib/vfs_s3_transport.c
2 *
3 * WHAT: The client-side HTTP transport for the shared S3 driver
4 * (src/fs/backend/sd_s3.c). Implements brix_s3_transport_t over the
5 * client's HTTP/1.1 stack (brix_http_req/_header/_resp_free).
6 * WHY: sd_s3 holds the S3 protocol logic in src/ (shared, ngx-free) but needs
7 * an injected transport — this is the client's. A server consumer would
8 * provide its own (e.g. libcurl) without touching the driver.
9 * HOW: one stateless request op (tctx unused) + thin response accessors that
10 * wrap a heap brix_http_resp behind the transport's opaque handle.
11 */

17static int
18s3t_request(void *tctx, const char *host, int port, int tls,
19 const char *method, const char *path_and_query, const char *headers,
20 const void *body, size_t body_len, int timeout_ms,
21 brix_s3_resp_t *resp, char *errbuf, size_t errcap)
22{
23 brix_http_resp *r;
24 brix_status st;
25
26 (void) tctx;
27
28 r = calloc(1, sizeof(*r));
29 if (r == NULL) {
30 if (errbuf != NULL && errcap > 0) {
31 snprintf(errbuf, errcap, "s3 transport: out of memory");
32 }
33 return -1;
34 }
35 brix_status_clear(&st);
36 /* verify=0, ca_dir=NULL — matches the original vfs_s3 request policy. */
37 if (brix_http_req(host, port, tls, method, path_and_query, headers,
38 body, body_len, timeout_ms, 0, NULL, r, &st) != 0) {
39 if (errbuf != NULL && errcap > 0) {
40 snprintf(errbuf, errcap, "%s", st.msg);
41 }
42 free(r);
43 return -1;
44 }
45 resp->status = r->status;
46 resp->opaque = r;
47 return 0;
48}
