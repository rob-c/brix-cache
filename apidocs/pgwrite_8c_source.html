11/* pgwrite_retry_spans_multiple_pages()
12 * A kXR_pgRetry resend must correct exactly one page. Mirrors stock
13 * do_PgWIORetry: an unaligned offset may carry at most (bytes-to-boundary + one
14 * CRC); an aligned offset at most one page unit (kXR_pgUnitSZ = 4100). Returns
15 * 1 if the wire payload `dlen` is too large to be a single-page retry. */
16static int
17pgwrite_retry_spans_multiple_pages(int64_t offset, size_t dlen)
18{
19 size_t in_page = (size_t) (offset & (int64_t) (kXR_pgPageSZ - 1));
20
21 if (in_page != 0) {
22 size_t to_boundary = (size_t) kXR_pgPageSZ - in_page;
23 return dlen > to_boundary + XRD_PGWRITE_CKSZ;
24 }
25 return dlen > (size_t) kXR_pgUnitSZ;
26}

28/* Payload decoding helper
29 * Decodes a kXR_pgwrite payload into a flat pwrite() buffer.
30 *
31 * Payload layout is CRC first:
32 * [XRD_PGWRITE_CKSZ=4 bytes CRC32c][up to XRD_PGWRITE_PAGESZ=4096 bytes data]
33 * repeated for each page fragment. The first and last fragments may be shorter
34 * when the write offset is unaligned or the request ends mid-page.
35 *
36 * Returns NGX_OK on success, NGX_DECLINED on checksum mismatch, and NGX_ERROR
37 * for malformed framing.
38 *
39 * Uses brix_crc32c_copy() to fuse the CRC verification and data copy into a
40 * single pass over each page, avoiding the double-read that separate
41 * brix_crc32c() + ngx_memcpy() calls would incur.
42 */
43ngx_int_t
44brix_pgwrite_decode_payload(const u_char *payload, size_t payload_len,
45 int64_t offset, u_char *flat, size_t *flat_len, int64_t *bad_offset)
46{
47 ssize_t n;
48
49 if (flat_len == NULL || bad_offset == NULL || offset < 0) {
50 return NGX_ERROR;
51 }
52
53 *flat_len = 0;
54 *bad_offset = offset;
55
56 if (payload == NULL || flat == NULL || payload_len <= XRD_PGWRITE_CKSZ) {
57 return NGX_ERROR;
58 }
59
60 /* Shared single-pass decode (libxrdproto): verify each page's CRC32c while
61 * copying into the flat pwrite buffer — byte-identical to what the native
62 * client encodes. -1 = a page's CRC mismatched (→ kXR_ChkSumErr); -2 =
63 * malformed framing / offset overflow. dstcap = payload_len is a safe upper
64 * bound (decoded data is always < payload by at least one CRC). */
65 n = xrdp_pg_decode(payload, payload_len, offset, flat, payload_len,
66 bad_offset);
67 if (n == -1) {
68 return NGX_DECLINED;
69 }
70 if (n < 0) {
71 return NGX_ERROR;
72 }
73 *flat_len = (size_t) n;
74 return NGX_OK;
75}

77/* pgwrite handler section
78 * WHAT: Handle kXR_pgwrite, the page-mode write with a 4-byte CRC32c before
79 * each ≤4 KiB page. Implements the stock CSE (checksum-error) retransmit
80 * machine: verify EVERY page (xrdp_pg_decode_collect), write ALL pages —
81 * good and bad — to disk, and on any failure reply with a SUCCESS
82 * kXR_status frame carrying the pgWrCSE retransmit list
83 * (brix_send_pgwrite_cse) rather than a hard error. Each bad page is
84 * registered in the per-handle Fob (pgw_fob.c).
85 *
86 * WHY: Accept-then-correct lets a single corrupt page be re-sent without
87 * restarting a multi-GB transfer. The kXR_close gate (read/close.c)
88 * returns kXR_ChkSumErr while the Fob is non-empty, so a committed file
89 * never retains known-corrupt bytes despite the bad bytes being written.
90 *
91 * HOW: A kXR_pgRetry resend (reqflags) is single-page-bounded, must target a
92 * registered offset (else downgraded to a normal write), takes the
93 * synchronous path, and clears its Fob entry on a clean re-verify. Over
94 * kXR_pgMaxEpr errors in one request, or kXR_pgMaxEos outstanding per
95 * file, → kXR_TooManyErrs; malformed framing → kXR_ArgInvalid. A clean
96 * write replies via brix_send_pgwrite_status (info = request offset).
97 */
98ngx_int_t
99brix_handle_pgwrite(brix_ctx_t *ctx, ngx_connection_t *c)
100{
101 xrdw_pgwrite_req_t req;
102 ngx_stream_brix_srv_conf_t *rconf;
103 int idx;
104 int64_t offset;
105 size_t dlen = ctx->recv.cur_dlen;
106 u_char *payload = ctx->recv.payload;
107 int is_retry;
108 ngx_int_t rc;
109
110 xrdw_pgwrite_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
111 idx = (int)(unsigned char) req.fhandle[0];
112 offset = req.offset;
113 is_retry = (req.reqflags & kXR_pgRetry) != 0;
114 int64_t write_offset;
115 size_t total_written;
116 ssize_t nw;
117 char write_detail[64];
118
119 rconf = ngx_stream_get_module_srv_conf(
120 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
121
122 if (!brix_validate_write_handle(ctx, c, idx, "WRITE",
123 BRIX_OP_WRITE, &rc)) {
124 return rc;
125 }
126
127 if (offset < 0 || dlen <= XRD_PGWRITE_CKSZ) {
128 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
129 (long long) offset, dlen);
130 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
131 ctx->files[idx].path, write_detail,
132 kXR_ArgInvalid, "invalid pgwrite payload");
133 }
134
135 /* kXR_pgRetry correction request * A retry must correct exactly one registered page. Too large → reject.
136 * Not registered (stray/forged retry, or a resend during write recovery)
137 * → drop the retry flag and treat as a normal write so it cannot corrupt
138 * the Fob (matches stock do_PgWIORetry). */
139 if (is_retry) {
140 size_t page_dlen;
141
142 if (pgwrite_retry_spans_multiple_pages(offset, dlen)) {
143 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
144 (long long) offset, dlen);
145 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
146 ctx->files[idx].path, write_detail,
147 kXR_ArgInvalid,
148 "pgwrite retry of more than one page not allowed");
149 }
150 page_dlen = dlen - XRD_PGWRITE_CKSZ;
151 if (!brix_pgw_fob_has(&ctx->files[idx], offset,
152 (uint32_t) page_dlen)) {
153 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
154 "brix: pgwrite retry %zu@%lld not in error — normal write",
155 page_dlen, (long long) offset);
156 is_retry = 0;
157 }
158 }
159
160 {
161 /* Reusable per-session buffer: avoids a malloc/free per request. */
162 u_char *flat = BRIX_GET_SCRATCH(ctx, c, rd.write_scratch,
163 rd.write_scratch_size, dlen);
164 size_t flat_sz = 0;
165 /* CSE (checksum-error) retransmit: verify ALL pages, copy them all into
166 * flat (accept-then-correct, matching stock), and collect the failures.
167 * A non-empty list yields a SUCCESS kXR_status reply carrying the bad
168 * page offsets — NOT a hard error. */
169 xrdp_pg_bad_t bad_pages[kXR_pgMaxEpr];
170 size_t bad_count = 0;
171 ssize_t dn;
172
173 if (flat == NULL) { return NGX_ERROR; }
174
175 dn = xrdp_pg_decode_collect(payload, dlen, offset, flat, dlen,
176 bad_pages, kXR_pgMaxEpr, &bad_count);
177 if (dn == -3) {
178 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
179 (long long) offset, dlen);
180 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
181 ctx->files[idx].path, write_detail,
182 kXR_TooManyErrs,
183 "too many checksum errors in pgwrite request");
184 }
185 if (dn < 0) {
186 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
187 (long long) offset, dlen);
188 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
189 ctx->files[idx].path, write_detail,
190 kXR_ArgInvalid, "invalid pgwrite payload");
191 }
192 flat_sz = (size_t) dn;
193
194 /* kXR_recoverWrts replay detection (post-decode) * Only short-circuit a clean replay; a request with bad pages must take
195 * the normal accept-then-correct path so the CSE list is emitted. A
196 * kXR_pgRetry correction deliberately re-writes a range already in the
197 * journal, so it must NOT be mistaken for a replay (that would skip the
198 * Fob correction and leave the close gate stuck). */
199 if (!is_retry && bad_count == 0 && ctx->files[idx].wrts_enabled &&
200 brix_wrts_is_replay(&ctx->files[idx], offset, (uint32_t) flat_sz))
201 {
202 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
203 "brix: pgwrite recovery replay skip offset=%lld len=%zu",
204 (long long) offset, flat_sz);
205 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
206 /* info offset = request offset (stock parity), not offset+len. */
207 return brix_send_pgwrite_status(ctx, c, offset);
208 }
209
210 /* register newly-corrupt pages in the Fob (close-time gate) * The data is still written (accept-then-correct); the Fob is what
211 * blocks a clean close until every page is corrected. Per-file
212 * capacity overflow → kXR_TooManyErrs (matches stock addOffs). */
213 if (bad_count > 0) {
214 size_t bi;
215
216 brix_pgw_fob_open(&ctx->files[idx]);
217 for (bi = 0; bi < bad_count; bi++) {
218 if (!brix_pgw_fob_add(&ctx->files[idx], bad_pages[bi].off,
219 bad_pages[bi].dlen)) {
220 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
221 (long long) offset, dlen);
222 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
223 ctx->files[idx].path, write_detail,
224 kXR_TooManyErrs,
225 "too many uncorrected checksum errors in file");
226 }
227 }
228 }
229
230 /* Retries are single-page (≤ 4 KiB): take the synchronous path so the
231 * Fob correction (brix_pgw_fob_del) commits in-line right after the
232 * write, rather than threading retry state through the AIO callback. */
233 if (!is_retry) {
234 ngx_flag_t posted;
235
236 /* Pass NULL as payload_to_free: flat is rd.write_scratch (pool-managed,
237 * not heap-allocated) and must not be freed by the done handler.
238 * The CSE bad-page list (if any) rides along so the async completion
239 * sends the retransmit frame. */
240 rc = brix_try_post_write_aio(ctx, c, idx, (off_t) offset, flat,
241 flat_sz, offset, 1, NULL,
242 bad_count ? bad_pages : NULL, bad_count,
243 "brix: thread_task_post failed, falling back to sync pgwrite",
244 &posted);
245 if (rc != NGX_OK) {
246 return rc;
247 }
248 if (posted) {
249 /* Async completion sends the mandatory kXR_status pgwrite reply. */
250 return NGX_OK;
251 }
252 } /* end NGX_THREADS block */
253
254 /* Synchronous path: write the entire flat buffer in one syscall. */
255 write_offset = offset;
256 {
257 brix_vfs_job_t job;
258
259 brix_vfs_job_write_init(&job, ctx->files[idx].fd,
260 (off_t) write_offset, flat, flat_sz);
261 job.csi = ctx->files[idx].csi; /* phase-59 W2: update page tags */
262 brix_vfs_job_set_obj(&job, &ctx->files[idx].sd_obj);
263 brix_vfs_io_execute(&job);
264 nw = job.nio;
265 if (job.io_errno != 0) {
266 errno = job.io_errno;
267 }
268 }
269 if (nw < 0) {
270 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
271 (long long) offset, flat_sz);
272 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
273 ctx->files[idx].path, write_detail,
274 kXR_IOError, strerror(errno));
275 }
276 if ((size_t) nw < flat_sz) {
277 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
278 (long long) offset, (size_t) nw);
279 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
280 ctx->files[idx].path, write_detail,
281 kXR_IOError, "short write (disk full?)");
282 }
283
284 total_written = (size_t) nw;
285 write_offset += (int64_t) nw;
286
287 ctx->files[idx].bytes_written += total_written;
288 ctx->totals.bytes_written += total_written;
289 brix_rl_charge_ctx(ctx, total_written); /* Phase 25 bandwidth */
290
291 /* write-through dirty state tracking (mirrors XrdPfcFile::m_dirtyOffset)
292 * Same logic as in write.c: when wt_enabled = 1 this handle has a cached
293 * DECISION to propagate writes back to the origin at close time. We track
294 * cumulative bytes written and mark the dirty offset so write-back
295 * knows what data needs to be flushed during the close-flush phase.
296 */
297 if (ctx->files[idx].wt_enabled) {
298 brix_wt_mark_dirty(ctx, idx, offset + (int64_t) nw - 1,
299 total_written);
300 }
301
302 if (rconf->access_log_fd != NGX_INVALID_FILE) {
303 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
304 (long long) offset, total_written);
305 brix_log_access(ctx, c, "WRITE", ctx->files[idx].path,
306 write_detail, 1, 0, NULL, total_written);
307 }
308 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
309
310 /* A successful retry whose page now verifies clears it from the Fob.
311 * (A still-bad retry kept bad_count > 0 and was re-added above, so the
312 * close gate still holds.) */
313 if (is_retry && bad_count == 0) {
314 brix_pgw_fob_del(&ctx->files[idx], offset, (uint32_t) flat_sz);
315 }
316
317 /* Record the committed write in the recovery journal. */
318 if (ctx->files[idx].wrts_enabled) {
319 brix_wrts_record(&ctx->files[idx], offset, (uint32_t) nw);
320 }
321
322 /* The kXR_status "info" offset echoes the REQUEST offset (where the data
323 * was written), matching the reference do_pgWrite — NOT offset+len. A
324 * next-expected-offset here diverged from stock (test_conf_pgio).
325 * When pages failed CRC32c, the reply is a SUCCESS frame carrying the
326 * CSE retransmit list (accept-then-correct), not a hard error. */
327 (void) write_offset;
328 if (bad_count > 0) {
329 return brix_send_pgwrite_cse(ctx, c, offset,
330 bad_pages, bad_count);
331 }
332 return brix_send_pgwrite_status(ctx, c, offset);
333 }
