1/*
2 * pblock_quota.c — F5 usage rollup + quota admission for the pblock driver.
3 *
4 * WHAT: Implements pblock_quota.h: trigger-maintained `usage` rollup (see the
5 * header for why triggers — transactional by construction) plus the
6 * admission checks the driver calls at catalog boundaries.
7 *
8 * HOW: The three triggers mirror insert/delete/update of `objects` rows into
9 * per-scope UPSERTs. Directory rows count as inodes but contribute zero
10 * bytes. The rebuild at init recomputes the rollup from `objects` so
11 * enabling quota on a populated export starts honest. ngx-free
12 * (libc + sqlite3); gated by BRIX_HAVE_SQLITE.

34/* One UPSERT per scope; `B` is the byte contribution expression. */
35#define Q_UPS(scope_lit, id_expr, B) \
36 "INSERT INTO usage VALUES(" scope_lit "," id_expr "," B ",1)" \
37 " ON CONFLICT(scope,id) DO UPDATE SET" \
38 " bytes=bytes+excluded.bytes, inodes=inodes+excluded.inodes;"
39#define Q_DEC(scope_lit, id_expr, B) \
40 "UPDATE usage SET bytes=bytes-(" B "), inodes=inodes-1" \
41 " WHERE scope=" scope_lit " AND id=" id_expr ";"
42
43#define NEW_B "CASE WHEN NEW.is_dir=0 THEN NEW.size ELSE 0 END"
44#define OLD_B "CASE WHEN OLD.is_dir=0 THEN OLD.size ELSE 0 END"

46int
47pblock_quota_init(pblock_state_t *st)
48{
49 static const char *ddl[] = {
50 "CREATE TABLE IF NOT EXISTS usage("
51 " scope TEXT NOT NULL, id INTEGER NOT NULL,"
52 " bytes INTEGER NOT NULL, inodes INTEGER NOT NULL,"
53 " PRIMARY KEY(scope, id));",
54
55 "CREATE TRIGGER IF NOT EXISTS usage_ai AFTER INSERT ON objects BEGIN "
56 Q_UPS("'total'", "0", NEW_B)
57 Q_UPS("'uid'", "NEW.uid", NEW_B)
58 Q_UPS("'gid'", "NEW.gid", NEW_B)
59 "END;",
60
61 "CREATE TRIGGER IF NOT EXISTS usage_ad AFTER DELETE ON objects BEGIN "
62 Q_DEC("'total'", "0", OLD_B)
63 Q_DEC("'uid'", "OLD.uid", OLD_B)
64 Q_DEC("'gid'", "OLD.gid", OLD_B)
65 "END;",
66
67 "CREATE TRIGGER IF NOT EXISTS usage_au AFTER UPDATE OF size, uid, gid "
68 "ON objects BEGIN "
69 Q_DEC("'total'", "0", OLD_B)
70 Q_DEC("'uid'", "OLD.uid", OLD_B)
71 Q_DEC("'gid'", "OLD.gid", OLD_B)
72 Q_UPS("'total'", "0", NEW_B)
73 Q_UPS("'uid'", "NEW.uid", NEW_B)
74 Q_UPS("'gid'", "NEW.gid", NEW_B)
75 "END;",
76
77 /* rebuild: honest rollup even when quota is enabled on a populated
78 * export (or after a run without quota). */
79 "DELETE FROM usage;",
80 "INSERT INTO usage SELECT 'total', 0,"
81 " COALESCE(SUM(CASE WHEN is_dir=0 THEN size ELSE 0 END),0), COUNT(*)"
82 " FROM objects;",
83 "INSERT INTO usage SELECT 'uid', uid,"
84 " SUM(CASE WHEN is_dir=0 THEN size ELSE 0 END), COUNT(*)"
85 " FROM objects GROUP BY uid;",
86 "INSERT INTO usage SELECT 'gid', gid,"
87 " SUM(CASE WHEN is_dir=0 THEN size ELSE 0 END), COUNT(*)"
88 " FROM objects GROUP BY gid;",
89 };
90 size_t i;
91
92 if (cat_exec(st->cat, "BEGIN;") != 0) {
93 return -1;
94 }
95 for (i = 0; i < sizeof(ddl) / sizeof(ddl[0]); i++) {
96 if (cat_exec(st->cat, ddl[i]) != 0) {
97 (void) cat_exec(st->cat, "ROLLBACK;");
98 return -1;
99 }
100 }
101 return cat_exec(st->cat, "COMMIT;");
102}

104int
105pblock_quota_usage(const pblock_state_t *st, const char *scope, int64_t id,
106 int64_t *bytes, int64_t *inodes)
107{
108 sqlite3_stmt *q;
109
110 *bytes = 0;
111 *inodes = 0;
112 q = cat_prepare(st->cat,
113 "SELECT bytes, inodes FROM usage WHERE scope = ?1 AND id = ?2;");
114 if (q == NULL) {
115 return -1;
116 }
117 sqlite3_bind_text(q, 1, scope, -1, SQLITE_STATIC);
118 sqlite3_bind_int64(q, 2, id);
119 if (sqlite3_step(q) == SQLITE_ROW) {
120 *bytes = sqlite3_column_int64(q, 0);
121 *inodes = sqlite3_column_int64(q, 1);
122 }
123 sqlite3_finalize(q);
124 return 0;
125}

140int
141pblock_quota_admit(const pblock_state_t *st, uint32_t uid,
142 int64_t add_bytes, int64_t add_inodes)
143{
144 int64_t bytes, inodes, limit;
145
146 if (!st->quota) {
147 return 0;
148 }
149 if (pblock_quota_usage(st, "total", 0, &bytes, &inodes) != 0) {
150 return 0; /* fail-open on a rollup read error */
151 }
152 if ((st->quota_bytes > 0 && add_bytes > 0
153 && bytes + add_bytes > st->quota_bytes)
154 || (st->quota_inodes > 0 && add_inodes > 0
155 && inodes + add_inodes > st->quota_inodes))
156 {
157 errno = EDQUOT;
158 return -1;
159 }
160
161 limit = quota_uid_limit(st, uid);
162 if (limit > 0 && add_bytes > 0) {
163 if (pblock_quota_usage(st, "uid", (int64_t) uid, &bytes, &inodes) != 0) {
164 return 0;
165 }
166 if (bytes + add_bytes > limit) {
167 errno = EDQUOT;
168 return -1;
169 }
170 }
171 return 0;
172}

174int64_t
175pblock_quota_max_size(const pblock_state_t *st, uint32_t uid, int64_t cur_size)
176{
177 int64_t bytes, inodes, room = INT64_MAX, limit;
178
179 if (!st->quota) {
180 return INT64_MAX;
181 }
182 if (st->quota_bytes > 0
183 && pblock_quota_usage(st, "total", 0, &bytes, &inodes) == 0)
184 {
185 room = st->quota_bytes - bytes;
186 }
187 limit = quota_uid_limit(st, uid);
188 if (limit > 0
189 && pblock_quota_usage(st, "uid", (int64_t) uid, &bytes, &inodes) == 0
190 && limit - bytes < room)
191 {
192 room = limit - bytes;
193 }
194 if (room == INT64_MAX) {
195 return INT64_MAX; /* no byte limit in play */
196 }
197 if (room < 0) {
198 room = 0;
199 }
200 return cur_size + room;
201}
