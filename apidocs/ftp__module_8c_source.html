41/* brix_ftp_install_backend_credential — bind the named brix_credential block's
42 * upstream identity (SigV4 keys for an s3:// backend) to this export's registry
43 * entry, so per-request ctx resolution builds a signed backend instance. Mirrors
44 * the root/webdav wiring via the ONE shared credential→backend_cred mapper
45 * (P80.1). A no-op when no credential is named; a hard error on an unknown name
46 * or an unreadable token_file (a misconfigured upstream must fail the config, not
47 * serve unauthenticated). */
48static char *
49brix_ftp_install_backend_credential(ngx_conf_t *cf,
50 ngx_stream_brix_ftp_srv_conf_t *conf)
51{
52 char name[256];
53 char bearer[4096];
54 const brix_credential_t *cred;
55 brix_vfs_backend_cred_t bcred;
56
57 if (conf->storage_credential.len == 0) {
58 return NGX_CONF_OK;
59 }
60 ngx_cpystrn((u_char *) name, conf->storage_credential.data,
61 ngx_min(conf->storage_credential.len + 1, sizeof(name)));
62 cred = brix_credential_lookup(name);
63 if (cred == NULL) {
64 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
65 "brix_gridftp_storage_credential \"%s\" names no brix_credential "
66 "block", name);
67 return NGX_CONF_ERROR;
68 }
69 if (brix_credential_to_backend_cred(cred, bearer, sizeof(bearer),
70 &bcred, cf->log) != NGX_OK)
71 {
72 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
73 "brix_gridftp: cannot derive credential \"%s\" for export \"%s\"",
74 name, conf->root_canon);
75 return NGX_CONF_ERROR;
76 }
77 brix_vfs_backend_set_credential(conf->root_canon, &bcred);
78
79 /* An explicit `mode` on the block overrides the gateway's default per-request
80 * delegation mode (e.g. `mode select` pins the service credential and never
81 * forwards the client's proxy). */
82 if (cred->mode != NGX_CONF_UNSET) {
83 conf->deleg_mode = (enum brix_cred_mode) cred->mode;
84 }
85 return NGX_CONF_OK;
86}

89/* brix_ftp_merge_conf — parent→child merge: disabled and read-only by default;
90 * inherit the export root when the child omitted its own. */
91static char *
92brix_ftp_merge_conf(ngx_conf_t *cf, void *parent, void *child)
93{
94 ngx_stream_brix_ftp_srv_conf_t *prev = parent;
95 ngx_stream_brix_ftp_srv_conf_t *conf = child;
96
97 ngx_conf_merge_value(conf->enable, prev->enable, 0);
98 ngx_conf_merge_value(conf->allow_write, prev->allow_write, 0);
99 ngx_conf_merge_value(conf->verify_write, prev->verify_write, 0);
100 ngx_conf_merge_value(conf->gsi, prev->gsi, 0);
101 ngx_conf_merge_value(conf->pasv_port_lo, prev->pasv_port_lo, 0);
102 ngx_conf_merge_value(conf->pasv_port_hi, prev->pasv_port_hi, 0);
103 ngx_conf_merge_value(conf->require_allo_size, prev->require_allo_size, 0);
104 ngx_conf_merge_str_value(conf->export, prev->export, "");
105 ngx_conf_merge_str_value(conf->storage_backend, prev->storage_backend, "");
106 ngx_conf_merge_str_value(conf->storage_credential,
107 prev->storage_credential, "");
108 ngx_conf_merge_str_value(conf->certificate, prev->certificate, "");
109 ngx_conf_merge_str_value(conf->certificate_key, prev->certificate_key, "");
110 ngx_conf_merge_str_value(conf->trusted_ca, prev->trusted_ca, "");
111
112 if (conf->root_canon[0] == '\0' && prev->root_canon[0] != '\0') {
113 ngx_memcpy(conf->root_canon, prev->root_canon,
114 sizeof(conf->root_canon));
115 }
116
117 if (conf->enable && conf->root_canon[0] == '\0') {
118 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
119 "brix_gridftp is on but brix_gridftp_export is unset or "
120 "unresolvable");
121 return NGX_CONF_ERROR;
122 }
123
124 /* Register this export's storage backend so every per-request ftp_vfs_ctx()
125 * (which calls brix_vfs_ctx_init → brix_vfs_backend_resolve on root_canon)
126 * routes through the selected driver — "pblock" today, POSIX when unset.
127 * The gateway itself only ever touches storage through brix_vfs_*, so no
128 * data-path change is needed once the choice is on the registry. */
129 if (conf->enable && conf->root_canon[0] != '\0') {
130 char *rv;
131 if (brix_vfs_backend_config_str(cf, conf->root_canon,
132 &conf->storage_backend, 0, BRIX_AF_AUTO) != NGX_OK)
133 {
134 return NGX_CONF_ERROR;
135 }
136 /* Default: forward the client's delegated proxy (PASSTHROUGH). A named
137 * brix_credential block's `mode` may override this inside install below;
138 * the request-time bind (ftp_ev_path.c) additionally no-ops on backends
139 * that do not consume a proxy, so posix/pblock exports are unaffected. */
140 conf->deleg_mode = BRIX_CRED_PASSTHROUGH;
141 rv = brix_ftp_install_backend_credential(cf, conf);
142 if (rv != NGX_CONF_OK) {
143 return rv;
144 }
145 }
146
147 if (conf->tls_ctx == NULL && prev->tls_ctx != NULL) {
148 conf->tls_ctx = prev->tls_ctx; /* inherit built ctx */
149 conf->ca_store = prev->ca_store;
150 } else if (conf->enable && conf->gsi && conf->tls_ctx == NULL) {
151 char *rv = brix_ftp_build_gsi(cf, conf);
152 if (rv != NGX_CONF_OK) {
153 return rv;
154 }
155 }
156
157 return NGX_CONF_OK;
158}

199/* brix_ftp_set_export — store the raw export string and realpath(3) it into
200 * root_canon at config time so every per-request brix_http_resolve_path() has a
201 * canonical confinement root. A not-yet-existing tree fails the config (unlike
202 * a cache dir, an export must exist to serve). */
203static char *
204brix_ftp_set_export(ngx_conf_t *cf, ngx_command_t *cmd, void *conf_ptr)
205{
206 ngx_stream_brix_ftp_srv_conf_t *conf = conf_ptr;
207 ngx_str_t *value = cf->args->elts;
208 ngx_str_t dir = value[1];
209 char raw[PATH_MAX];
210
211 (void) cmd;
212
213 if (conf->export.len != 0) {
214 return "is duplicate";
215 }
216 if (ngx_conf_full_name(cf->cycle, &dir, 1) != NGX_OK) {
217 return NGX_CONF_ERROR;
218 }
219 if (dir.len >= sizeof(raw)) {
220 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
221 "brix_gridftp_export path too long: %V", &dir);
222 return NGX_CONF_ERROR;
223 }
224 ngx_memcpy(raw, dir.data, dir.len);
225 raw[dir.len] = '\0';
226
227 if (realpath(raw, conf->root_canon) == NULL) {
228 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
229 "brix_gridftp_export \"%s\" cannot be resolved (does it exist?)",
230 raw);
231 return NGX_CONF_ERROR;
232 }
233
234 conf->export = dir;
235
236 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
237 "brix: GridFTP gateway export=%s (canon)", conf->root_canon);
238
239 return NGX_CONF_OK;
240}

243/* brix_ftp_set_pasv_range — parse `brix_gridftp_pasv_port_range <lo> <hi>` into
244 * the inclusive passive-data-port window. Both must be valid TCP ports and
245 * lo <= hi; a well-formed but empty/inverted range is a config error rather than
246 * a silent fall-back to ephemeral, so a firewalled deployment cannot boot with a
247 * range that would still hand out un-openable ports. */
248static char *
249brix_ftp_set_pasv_range(ngx_conf_t *cf, ngx_command_t *cmd, void *conf_ptr)
250{
251 ngx_stream_brix_ftp_srv_conf_t *conf = conf_ptr;
252 ngx_str_t *value = cf->args->elts;
253 ngx_int_t lo, hi;
254
255 (void) cmd;
256
257 if (conf->pasv_port_lo != NGX_CONF_UNSET) {
258 return "is duplicate";
259 }
260 lo = ngx_atoi(value[1].data, value[1].len);
261 hi = ngx_atoi(value[2].data, value[2].len);
262 if (lo == NGX_ERROR || hi == NGX_ERROR
263 || lo < 1 || lo > 65535 || hi < 1 || hi > 65535)
264 {
265 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
266 "brix_gridftp_pasv_port_range: each bound must be a TCP port "
267 "1..65535 (got \"%V\" \"%V\")", &value[1], &value[2]);
268 return NGX_CONF_ERROR;
269 }
270 if (lo > hi) {
271 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
272 "brix_gridftp_pasv_port_range: low bound %i exceeds high bound %i",
273 lo, hi);
274 return NGX_CONF_ERROR;
275 }
276 conf->pasv_port_lo = lo;
277 conf->pasv_port_hi = hi;
278 return NGX_CONF_OK;
279}

394/* brix_ftp_init_process — per-worker credential replay for every enabled GridFTP
395 * export. brix_vfs_backend_set_credential runs at config parse in the master; the
396 * VFS backend registry is rebuilt per worker, so without this replay a forked
397 * worker holds the s3:// backend with an EMPTY credential and the first upstream
398 * PUT/GET fails "no credential set". Mirrors the core stream module's
399 * brix_init_server_backend_credential (process_server_init.c) for the gridftp
400 * module's own srv conf. A missing credential/backend is a legitimate no-op; a
401 * name that resolves to no block, or a credential that cannot be mapped, is
402 * logged (WARN) and the worker still comes up — the failure surfaces on first use,
403 * exactly as the config-time path already guaranteed the name exists. */
404static ngx_int_t
405brix_ftp_init_process(ngx_cycle_t *cycle)
406{
407 ngx_stream_core_main_conf_t *cmcf;
408 ngx_stream_core_srv_conf_t **cscfp;
409 ngx_stream_brix_ftp_srv_conf_t *conf;
410 ngx_uint_t i;
411
412 cmcf = ngx_stream_cycle_get_module_main_conf(cycle, ngx_stream_core_module);
413 if (cmcf == NULL) {
414 return NGX_OK;
415 }
416 cscfp = cmcf->servers.elts;
417
418 for (i = 0; i < cmcf->servers.nelts; i++) {
419 char name[256];
420 char bearer[4096];
421 const brix_credential_t *cred;
422 brix_vfs_backend_cred_t bcred;
423
424 conf = ngx_stream_conf_get_module_srv_conf(cscfp[i],
425 ngx_stream_brix_ftp_module);
426 if (conf == NULL || !conf->enable
427 || conf->storage_credential.len == 0
428 || conf->root_canon[0] == '\0')
429 {
430 continue;
431 }
432
433 ngx_cpystrn((u_char *) name, conf->storage_credential.data,
434 ngx_min(conf->storage_credential.len + 1, sizeof(name)));
435 cred = brix_credential_lookup(name);
436 if (cred == NULL) {
437 ngx_log_error(NGX_LOG_WARN, cycle->log, 0,
438 "brix_gridftp: worker credential replay: no brix_credential "
439 "\"%s\" for export \"%s\" — upstream auth WILL fail",
440 name, conf->root_canon);
441 continue;
442 }
443 if (brix_credential_to_backend_cred(cred, bearer, sizeof(bearer),
444 &bcred, cycle->log) != NGX_OK)
445 {
446 ngx_log_error(NGX_LOG_WARN, cycle->log, 0,
447 "brix_gridftp: worker credential replay: cannot derive "
448 "credential \"%s\" for export \"%s\" — upstream auth WILL fail",
449 name, conf->root_canon);
450 continue;
451 }
452 brix_vfs_backend_set_credential(conf->root_canon, &bcred);
453 }
454
455 return NGX_OK;
456}
