1/*
2 * sss_keytab_kernel.h — shared SSS keytab text grammar.
3 *
4 * WHAT: The pure-C keytab grammar shared by the server's keytab loader
5 * (src/sss/config.c) and the native client's keytab reader
6 * (client/lib/sss_keytab.c): the one-line field parser and the keytab-file
7 * permission check. The SSS crypto kernels (CRC-32/IEEE, Blowfish-CFB64)
8 * and the hex codec are already shared via libxrdproto; this completes the
9 * set by removing the last byte-for-byte copy — the line grammar itself.
10 * WHY: Both consumers must agree exactly on how a keytab line is tokenised and
11 * on what counts as a safe keytab file; two hand-maintained copies of a
12 * security-sensitive parser are a divergence risk. Keeping one audited copy
13 * means a key minted/written by the client is parsed by the server (and
14 * vice-versa) under identical rules.
15 * HOW: Plain C — no nginx, no OpenSSL, no I/O. The caller owns the file I/O and
16 * reads each line; the kernel parses one line into a neutral entry and the
17 * caller copies the fields into its own key struct (computing any
18 * server-only authz options at the edge). The parser fails closed on a
19 * malformed required field (bad version tag, non-hex key, non-numeric id /
20 * expiry, over-long field) so a tampered keytab is rejected, never
21 * silently truncated.
22 */

30/*
31 * Field caps for the neutral entry. These match both consumers' key structs
32 * (brix_sss_key_t in src/types/config.h and xrdc_sss_key in
33 * client/lib/sss_keytab.h); each consumer copies out with its own bounded copy,
34 * so a future size change there only ever truncates safely.

41/* One parsed keytab entry (the raw fields; no server-side authz options). */
42typedef struct {
43 int64_t id; /* N: wire key id (>= 0) */
44 int64_t exp; /* e: epoch expiry, 0 = never */
45 size_t key_len; /* decoded length of key[] */
46 uint8_t key[SSS_K_KEY_MAX]; /* k: raw key bytes */
47 char user[SSS_K_USER_MAX]; /* u: (default "nobody") */
48 char group[SSS_K_GROUP_MAX]; /* g: (default "nogroup") */
49 char name[SSS_K_NAME_MAX]; /* n: (default "nowhere") */
50} sss_keytab_entry_t;
51
52/*
53 * Parse one keytab line (mutated in place by strtok_r) into *out. `now` is the
54 * current epoch used for the expiry check. Returns:
55 * 1 — a usable key was parsed into *out;
56 * 0 — the line is blank / a comment / an expired key (skip it);
57 * -1 — a required field is malformed (caller should fail closed).
58 * Grammar: <0|1> [u:user] [g:group] [n:name] [N:id] [k:hexkey] [e:expiry]
59 * Required: a "0"/"1" version tag, N: (>= 0) and k: (non-empty). Unknown fields
60 * are ignored; an inline '#' field starts a comment.
61 */
62int sss_keytab_parse_line(char *line, sss_keytab_entry_t *out, int64_t now);
63
64/*
65 * Validate keytab file permissions (shared secret — must not be exposed).
66 * Owner read/write is always allowed; all group/other bits are otherwise
67 * rejected. When `allow_dotgrp` is non-zero, a path ending in ".grp" may
68 * additionally be group-readable (the server's distributed-keytab variant).
69 * Returns 0 if the mode is safe, -1 otherwise.
70 */
71int sss_keytab_mode_ok(const char *path, mode_t mode, int allow_dotgrp);
