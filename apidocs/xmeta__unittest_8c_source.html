1/*
2 * xmeta_unittest.c — standalone unit test for the xmeta record codec (P1).
3 *
4 * Compiles without nginx:
5 *
6 * gcc -Wall -Wextra -Werror -I src -o /tmp/xmeta_ut \
7 * src/fs/meta/xmeta_unittest.c src/fs/meta/xmeta.c \
8 * src/core/compat/crc32c.c && /tmp/xmeta_ut
9 *
10 * With an argument, additionally writes a populated record to that path so
11 * the caller can cross-check the stock prefix with xrdpfc_print (see
12 * tests/run_xmeta.sh). Exit 0 = all checks pass.
13 */

31build_sample(brix_xmeta_t *m)
32{
33 /* 2.5 blocks of 1MiB */
34 brix_xmeta_init(m, 2621440, 1048576);
35 m->origin_mtime = 1751400000;
36 m->mode = 0644;
37 m->dirty_lo = 100;
38 m->dirty_hi = 200;
39 m->flush_gen = 7;
40 m->access_cnt = 3;
41 m->astat_count = 1;
42 m->astat.attach_time = 1751400001;
43 m->astat.detach_time = 1751400002;
44 m->astat.bytes_hit = 12345;
45 brix_xmeta_block_set(m, 0);
46 brix_xmeta_block_set(m, 2);
47 m->blockcrc[0] = 0xAABBCCDD;
48 m->blockcrc[2] = 0x11223344;
49 m->last_flush = 1751400500;
50 m->bytes_flushed = 4096;
51 m->filled_at = 1751400600;
52 m->expires_at = 1751401500;
53 m->state_flags = BRIX_XMETA_F_VERIFIED | BRIX_XMETA_F_EXPIRES;
54 m->etag_len = 8; memcpy(m->etag, "\"abc123\"", 8);
55 m->cks_alg_len = 7; memcpy(m->cks_alg, "adler32", 7);
56 m->cks_len = 8; memcpy(m->cks_hex, "0badcafe", 8);
57 brix_xmeta_digest_add(m, BRIX_XMETA_ALG_CRC32C,
58 "\xde\xad\xbe\xef", 4);
59 brix_xmeta_digest_add(m, BRIX_XMETA_ALG_MD5,
60 "0123456789abcdef", 16);

63int
64main(int argc, char **argv)
65{
66 brix_xmeta_t m, d;
67 uint8_t *buf = NULL;
68 size_t blen = 0;
69
70 /* ---- round trip ---- */
71 build_sample(&m);
72 CHECK(m.nblocks == 3, "init: 2.5 blocks rounds up to 3");
73 CHECK(brix_xmeta_encode(&m, &buf, &blen) == BRIX_XMETA_OK,
74 "encode ok");
75 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_OK,
76 "decode ok");
77 CHECK(d.file_size == 2621440 && d.buffer_size == 1048576,
78 "stock store fields round-trip");
79 CHECK(d.astat_count == 1 && d.astat.bytes_hit == 12345,
80 "astat record round-trips");
81 CHECK(brix_xmeta_block_test(&d, 0) && !brix_xmeta_block_test(&d, 1)
82 && brix_xmeta_block_test(&d, 2), "bitmap bits round-trip");
83 CHECK(!brix_xmeta_complete(&d), "incomplete bitmap detected");
84 CHECK(d.have_state && d.origin_mtime == 1751400000 && d.mode == 0644
85 && d.dirty_lo == 100 && d.dirty_hi == 200 && d.flush_gen == 7
86 && d.last_flush == 1751400500 && d.bytes_flushed == 4096
87 && d.filled_at == 1751400600 && d.expires_at == 1751401500
88 && d.state_flags == (BRIX_XMETA_F_VERIFIED
89 | BRIX_XMETA_F_EXPIRES),
90 "STATE section round-trips (incl. cvmfs manifest TTL)");
91 CHECK(d.etag_len == 8 && memcmp(d.etag, "\"abc123\"", 8) == 0
92 && d.cks_alg_len == 7 && memcmp(d.cks_alg, "adler32", 7) == 0
93 && d.cks_len == 8 && memcmp(d.cks_hex, "0badcafe", 8) == 0,
94 "ORIGIN section round-trips");
95 CHECK(d.have_blockcrc && d.blockcrc[0] == 0xAABBCCDD
96 && d.blockcrc[1] == BRIX_XMETA_CRC_UNSET
97 && d.blockcrc[2] == 0x11223344, "BLOCKCRC table round-trips");
98 {
99 uint16_t alg = 0, dl = 0;
100 const uint8_t *val = NULL;
101
102 CHECK(brix_xmeta_digest_get(&d, 0, &alg, &val, &dl)
103 == BRIX_XMETA_OK
104 && alg == BRIX_XMETA_ALG_CRC32C && dl == 4
105 && memcmp(val, "\xde\xad\xbe\xef", 4) == 0,
106 "digest[0] round-trips");
107 CHECK(brix_xmeta_digest_get(&d, 1, &alg, &val, &dl)
108 == BRIX_XMETA_OK
109 && alg == BRIX_XMETA_ALG_MD5 && dl == 16,
110 "digest[1] round-trips");
111 CHECK(brix_xmeta_digest_get(&d, 2, &alg, &val, &dl)
112 == BRIX_XMETA_FOREIGN, "digest[2] is past the end");
113 }
114 brix_xmeta_free(&d);
115
116 /* ---- corruption: each guarded region fails decode ---- */
117 buf[6] ^= 0xFF; /* store POD byte */
118 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_ERR,
119 "store corruption detected");
120 buf[6] ^= 0xFF;
121 buf[57] ^= 0xFF; /* bitmap byte */
122 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_ERR,
123 "bitmap corruption detected");
124 buf[57] ^= 0xFF;
125 buf[blen - 6] ^= 0xFF; /* last section payload */
126 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_ERR,
127 "section corruption detected");
128 buf[blen - 6] ^= 0xFF;
129 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_OK,
130 "restored record decodes again");
131 brix_xmeta_free(&d);
132
133 /* ---- foreign inputs ---- */
134 {
135 int32_t v = 3;
136
137 CHECK(brix_xmeta_decode((uint8_t *) &v, 4, &d)
138 == BRIX_XMETA_FOREIGN, "wrong version is FOREIGN");
139 CHECK(brix_xmeta_decode(buf, 10, &d) == BRIX_XMETA_FOREIGN,
140 "short buffer is FOREIGN");
141 }
142
143 /* ---- unknown section type is skipped (fwd compat) ---- */
144 {
145 /* retag the DIGEST section (2nd ext section) as type 0x7777 and fix
146 * its crc: decoder must skip it and still deliver STATE+BLOCKCRC */
147 uint8_t *tmp = malloc(blen);
148 size_t sec;
149 uint32_t plen, crc;
150
151 memcpy(tmp, buf, blen);
152 /* walk the sections (ext header sits right after the stock trailer:
153 * 4+48+4 + bitmap(1) + astat(56) + 4 + 8) and find DIGEST by type */
154 sec = 4 + 48 + 4 + 1 + 56 + 4 + 8;
155 for ( ;; ) {
156 uint16_t t;
157
158 memcpy(&t, tmp + sec, 2);
159 if (t == BRIX_XMETA_SEC_DIGEST) {
160 break;
161 }
162 memcpy(&plen, tmp + sec + 4, 4);
163 sec += 8 + plen + 4;
164 }
165 tmp[sec] = 0x77; tmp[sec + 1] = 0x77;
166 memcpy(&plen, tmp + sec + 4, 4);
167 crc = brix_crc32c_value(tmp + sec, 8 + plen);
168 memcpy(tmp + sec + 8 + plen, &crc, 4);
169 CHECK(brix_xmeta_decode(tmp, blen, &d) == BRIX_XMETA_OK,
170 "unknown section type skipped");
171 CHECK(d.have_state && d.have_blockcrc && d.digests == NULL,
172 "known sections still delivered, unknown dropped");
173 brix_xmeta_free(&d);
174 free(tmp);
175 }
176
177 /* ---- stock-only record (no extension) decodes ---- */
178 {
179 brix_xmeta_t s;
180 uint8_t *sbuf = NULL;
181 size_t sblen = 0;
182
183 brix_xmeta_init(&s, 1048576, 1048576);
184 s.have_state = 0;
185 s.have_blockcrc = 0;
186 brix_xmeta_block_set(&s, 0);
187 CHECK(brix_xmeta_encode(&s, &sbuf, &sblen) == BRIX_XMETA_OK,
188 "stock-only encode ok");
189 brix_xmeta_free(&s);
190 CHECK(brix_xmeta_decode(sbuf, sblen, &d) == BRIX_XMETA_OK
191 && !d.have_state && d.blockcrc == NULL
192 && brix_xmeta_block_test(&d, 0),
193 "stock-only record decodes (extension optional)");
194 brix_xmeta_free(&d);
195 free(sbuf);
196 }
197
198 /* ---- optional: emit the sample for the xrdpfc_print cross-check ---- */
199 if (argc > 1 && g_failed == 0) {
200 FILE *f = fopen(argv[1], "wb");
201
202 if (f == NULL || fwrite(buf, 1, blen, f) != blen) {
203 fprintf(stderr, "cannot write %s\n", argv[1]);
204 g_failed++;
205 }
206 if (f != NULL) {
207 fclose(f);
208 }
209 }
210
211 free(buf);
212 brix_xmeta_free(&m);
213 printf("xmeta_unittest: %d checks, %d failed\n", g_checks, g_failed);
214 return g_failed == 0 ? 0 : 1;
215}
