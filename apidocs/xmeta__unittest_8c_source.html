1/*
2 * xmeta_unittest.c — standalone unit test for the xmeta record codec (P1).
3 *
4 * Compiles without nginx:
5 *
6 * gcc -Wall -Wextra -Werror -I src -o /tmp/xmeta_ut \
7 * src/fs/meta/xmeta_unittest.c src/fs/meta/xmeta.c \
8 * src/core/compat/crc32c.c && /tmp/xmeta_ut
9 *
10 * With an argument, additionally writes a populated record to that path so
11 * the caller can cross-check the stock prefix with xrdpfc_print (see
12 * tests/run_xmeta.sh). Exit 0 = all checks pass.
13 *
14 * STRUCTURE: each self-contained assertion group is one static test_*()
15 * function; a descriptor table (g_tests) drives main() so the file stays under
16 * the complexity gate without any change to the assertions or their order.
17 * Every test rebuilds the record state it needs from build_sample(), so the
18 * cases share no live state and run in table order.
19 */

36/*
37 * build_sample — WHAT: populate a fully-featured xmeta record (STATE, ORIGIN,
38 * BLOCKCRC and two digests over 2.5 blocks). WHY: every test starts from the
39 * same known record so the cases are self-contained. HOW: field-by-field, then
40 * two block-set / digest-add calls.
41 */
42static void
43build_sample(brix_xmeta_t *m)
44{
45 /* 2.5 blocks of 1MiB */
46 brix_xmeta_init(m, 2621440, 1048576);
47 m->origin_mtime = 1751400000;
48 m->mode = 0644;
49 m->dirty_lo = 100;
50 m->dirty_hi = 200;
51 m->flush_gen = 7;
52 m->access_cnt = 3;
53 m->astat_count = 1;
54 m->astat.attach_time = 1751400001;
55 m->astat.detach_time = 1751400002;
56 m->astat.bytes_hit = 12345;
57 brix_xmeta_block_set(m, 0);
58 brix_xmeta_block_set(m, 2);
59 m->blockcrc[0] = 0xAABBCCDD;
60 m->blockcrc[2] = 0x11223344;
61 m->last_flush = 1751400500;
62 m->bytes_flushed = 4096;
63 m->filled_at = 1751400600;
64 m->expires_at = 1751401500;
65 m->state_flags = BRIX_XMETA_F_VERIFIED | BRIX_XMETA_F_EXPIRES;
66 m->etag_len = 8; memcpy(m->etag, "\"abc123\"", 8);
67 m->cks_alg_len = 7; memcpy(m->cks_alg, "adler32", 7);
68 m->cks_len = 8; memcpy(m->cks_hex, "0badcafe", 8);
69 brix_xmeta_digest_add(m, BRIX_XMETA_ALG_CRC32C,
70 "\xde\xad\xbe\xef", 4);
71 brix_xmeta_digest_add(m, BRIX_XMETA_ALG_MD5,
72 "0123456789abcdef", 16);
73}

75/*
76 * encode_sample — WHAT: build the sample record and encode it into a freshly
77 * allocated buffer. WHY: the round-trip, corruption and unknown-section tests
78 * all need an on-wire buffer to poke at. HOW: build_sample() then encode; the
79 * source record is freed here, the caller owns *buf and frees it. Assertion-
80 * free on purpose so it can be called by every case without perturbing the
81 * check count/order; the init/encode CHECKs live once in test_round_trip.
82 */
83static void
84encode_sample(uint8_t **buf, size_t *blen)
85{
86 brix_xmeta_t m;
87
88 build_sample(&m);
89 brix_xmeta_encode(&m, buf, blen);
90 brix_xmeta_free(&m);
91}

110/*
111 * check_digests — WHAT: assert the two added digests round-trip and index 2 is
112 * past the end. WHY: isolates the nested out-param dance from test_round_trip.
113 * HOW: three brix_xmeta_digest_get() calls with a shared alg/val/dl scratch.
114 */
115static void
116check_digests(const brix_xmeta_t *d)
117{
118 uint16_t alg = 0, dl = 0;
119 const uint8_t *val = NULL;
120
121 CHECK(brix_xmeta_digest_get(d, 0, &alg, &val, &dl)
122 == BRIX_XMETA_OK
123 && alg == BRIX_XMETA_ALG_CRC32C && dl == 4
124 && memcmp(val, "\xde\xad\xbe\xef", 4) == 0,
125 "digest[0] round-trips");
126 CHECK(brix_xmeta_digest_get(d, 1, &alg, &val, &dl)
127 == BRIX_XMETA_OK
128 && alg == BRIX_XMETA_ALG_MD5 && dl == 16,
129 "digest[1] round-trips");
130 CHECK(brix_xmeta_digest_get(d, 2, &alg, &val, &dl)
131 == BRIX_XMETA_FOREIGN, "digest[2] is past the end");
132}

134/*
135 * test_round_trip — WHAT: encode the sample, decode it, and assert every
136 * section survives the round trip. WHY: the primary happy path of the codec.
137 * HOW: encode_sample() then decode + per-section CHECK helpers.
138 */
139static void
140test_round_trip(void)
141{
142 brix_xmeta_t m, d;
143 uint8_t *buf = NULL;
144 size_t blen = 0;
145
146 build_sample(&m);
147 CHECK(m.nblocks == 3, "init: 2.5 blocks rounds up to 3");
148 CHECK(brix_xmeta_encode(&m, &buf, &blen) == BRIX_XMETA_OK,
149 "encode ok");
150 brix_xmeta_free(&m);
151 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_OK,
152 "decode ok");
153 CHECK(d.file_size == 2621440 && d.buffer_size == 1048576,
154 "stock store fields round-trip");
155 CHECK(d.astat_count == 1 && d.astat.bytes_hit == 12345,
156 "astat record round-trips");
157 CHECK(brix_xmeta_block_test(&d, 0) && !brix_xmeta_block_test(&d, 1)
158 && brix_xmeta_block_test(&d, 2), "bitmap bits round-trip");
159 CHECK(!brix_xmeta_complete(&d), "incomplete bitmap detected");
160 check_state_section(&d);
161 CHECK(d.etag_len == 8 && memcmp(d.etag, "\"abc123\"", 8) == 0
162 && d.cks_alg_len == 7 && memcmp(d.cks_alg, "adler32", 7) == 0
163 && d.cks_len == 8 && memcmp(d.cks_hex, "0badcafe", 8) == 0,
164 "ORIGIN section round-trips");
165 CHECK(d.have_blockcrc && d.blockcrc[0] == 0xAABBCCDD
166 && d.blockcrc[1] == BRIX_XMETA_CRC_UNSET
167 && d.blockcrc[2] == 0x11223344, "BLOCKCRC table round-trips");
168 check_digests(&d);
169 brix_xmeta_free(&d);
170 free(buf);
171}

173/*
174 * test_corruption — WHAT: flip a byte in each guarded region and assert decode
175 * fails, then restore and assert it decodes again. WHY: proves every CRC guard
176 * covers its region. HOW: XOR a store/bitmap/section byte in turn, decode,
177 * XOR back.
178 */
179static void
180test_corruption(void)
181{
182 brix_xmeta_t d;
183 uint8_t *buf = NULL;
184 size_t blen = 0;
185
186 encode_sample(&buf, &blen);
187 buf[6] ^= 0xFF; /* store POD byte */
188 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_ERR,
189 "store corruption detected");
190 buf[6] ^= 0xFF;
191 buf[57] ^= 0xFF; /* bitmap byte */
192 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_ERR,
193 "bitmap corruption detected");
194 buf[57] ^= 0xFF;
195 buf[blen - 6] ^= 0xFF; /* last section payload */
196 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_ERR,
197 "section corruption detected");
198 buf[blen - 6] ^= 0xFF;
199 CHECK(brix_xmeta_decode(buf, blen, &d) == BRIX_XMETA_OK,
200 "restored record decodes again");
201 brix_xmeta_free(&d);
202 free(buf);
203}

226/*
227 * test_unknown_section — WHAT: retag the DIGEST section as an unknown type,
228 * fix its crc, and assert the decoder skips it while still delivering
229 * STATE+BLOCKCRC. WHY: forward compatibility with unknown section types.
230 * HOW: walk the ext sections to DIGEST, rewrite its type, recompute its crc,
231 * then decode a copy.
232 */
233static void
234test_unknown_section(void)
235{
236 brix_xmeta_t d;
237 uint8_t *buf = NULL;
238 size_t blen = 0;
239 uint8_t *tmp;
240 size_t sec;
241 uint32_t plen, crc;
242
243 encode_sample(&buf, &blen);
244 tmp = malloc(blen);
245 memcpy(tmp, buf, blen);
246 /* walk the sections (ext header sits right after the stock trailer:
247 * 4+48+4 + bitmap(1) + astat(56) + 4 + 8) and find DIGEST by type */
248 sec = 4 + 48 + 4 + 1 + 56 + 4 + 8;
249 for ( ;; ) {
250 uint16_t t;
251
252 memcpy(&t, tmp + sec, 2);
253 if (t == BRIX_XMETA_SEC_DIGEST) {
254 break;
255 }
256 memcpy(&plen, tmp + sec + 4, 4);
257 sec += 8 + plen + 4;
258 }
259 tmp[sec] = 0x77; tmp[sec + 1] = 0x77;
260 memcpy(&plen, tmp + sec + 4, 4);
261 crc = brix_crc32c_value(tmp + sec, 8 + plen);
262 memcpy(tmp + sec + 8 + plen, &crc, 4);
263 CHECK(brix_xmeta_decode(tmp, blen, &d) == BRIX_XMETA_OK,
264 "unknown section type skipped");
265 CHECK(d.have_state && d.have_blockcrc && d.digests == NULL,
266 "known sections still delivered, unknown dropped");
267 brix_xmeta_free(&d);
268 free(tmp);
269 free(buf);
270}

312/*
313 * emit_sample — WHAT: optionally write a populated sample record to a path for
314 * the xrdpfc_print stock cross-check. WHY: keeps the file-I/O side effect out
315 * of the assertion-only main() and behind the "all passed" guard. HOW: encode
316 * the sample, write it, free the buffer; bumps g_failed on write error.
317 */
318static void
319emit_sample(const char *path)
320{
321 uint8_t *buf = NULL;
322 size_t blen = 0;
323 FILE *f;
324
325 encode_sample(&buf, &blen);
326 f = fopen(path, "wb");
327 if (f == NULL || fwrite(buf, 1, blen, f) != blen) {
328 fprintf(stderr, "cannot write %s\n", path);
329 g_failed++;
330 }
331 if (f != NULL) {
332 fclose(f);
333 }
334 free(buf);
335}

337int
338main(int argc, char **argv)
339{
340 size_t i;
341
342 for (i = 0; i < sizeof(g_tests) / sizeof(g_tests[0]); i++) {
343 g_tests[i]();
344 }
345
346 /* ---- optional: emit the sample for the xrdpfc_print cross-check ---- */
347 if (argc > 1 && g_failed == 0) {
348 emit_sample(argv[1]);
349 }
350
351 printf("xmeta_unittest: %d checks, %d failed\n", g_checks, g_failed);
352 return g_failed == 0 ? 0 : 1;
353}
