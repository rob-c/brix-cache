10/* Append one entry's current state to the journal (tab-delimited record). */
11static void
12journal_append(brix_cta_queue_t *q, const cta_req_t *e)
13{
14 FILE *f = q->journal;
15
16 if (f == NULL) {
17 return;
18 }
19 fprintf(f, "%llu\t%d\t%d\t%s\t%s\n",
20 (unsigned long long) e->id, (int) e->req.op, (int) e->state,
21 e->owner, e->req.path);
22 fflush(f);
23}

25/* Legal transitions, table-driven. A request may advance SUBMITTED→QUEUED→ACTIVE
26 * →COMPLETE, fail from ACTIVE, or be cancelled from any non-terminal state. */
27static int
28transition_ok(cta_state_t from, cta_state_t to)
29{
30 switch (from) {
31 case CTA_ST_SUBMITTED:
32 return to == CTA_ST_QUEUED || to == CTA_ST_CANCELED;
33 case CTA_ST_QUEUED:
34 return to == CTA_ST_ACTIVE || to == CTA_ST_CANCELED ||
35 to == CTA_ST_FAILED;
36 case CTA_ST_ACTIVE:
37 return to == CTA_ST_COMPLETE || to == CTA_ST_FAILED ||
38 to == CTA_ST_CANCELED;
39 case CTA_ST_COMPLETE:
40 case CTA_ST_FAILED:
41 case CTA_ST_CANCELED:
42 return 0; /* terminal */
43 }
44 return 0;
45}

72/* Find an existing slot for id, or claim a free one (replay only). */
73static cta_req_t *
74replay_slot(brix_cta_queue_t *q, uint64_t id)
75{
76 int i, free_slot = -1;
77
78 for (i = 0; i < CTA_QUEUE_MAX; i++) {
79 if (q->slots[i].in_use && q->slots[i].id == id) {
80 return &q->slots[i];
81 }
82 if (!q->slots[i].in_use && free_slot < 0) {
83 free_slot = i;
84 }
85 }
86 if (free_slot < 0) {
87 return NULL;
88 }
89 memset(&q->slots[free_slot], 0, sizeof(q->slots[free_slot]));
90 q->slots[free_slot].in_use = 1;
91 q->slots[free_slot].id = id;
92 q->slots[free_slot].queue = q;
93 return &q->slots[free_slot];
94}

96int
97cta_queue_open_journal(brix_cta_queue_t *q, const char *path)
98{
99 FILE *f = fopen(path, "a+");
100 char line[1280];
101
102 if (f == NULL) {
103 return -1;
104 }
105 rewind(f);
106 while (fgets(line, sizeof(line), f) != NULL) {
107 unsigned long long id;
108 int op = 0, state = 0;
109 char owner[64] = "", pathbuf[1024] = "";
110 cta_req_t *e;
111 /* "id\top\tstate\towner\tpath" — owner/path are tab-free single tokens.
112 * owner/path may be empty (their conversions then fail harmlessly). */
113 if (sscanf(line, "%llu\t%d\t%d\t%63[^\t]\t%1023[^\n]",
114 &id, &op, &state, owner, pathbuf) < 3) {
115 continue; /* need at least id/op/state */
116 }
117 e = replay_slot(q, (uint64_t) id);
118 if (e == NULL) {
119 break;
120 }
121 e->req.op = (cta_op_t) op;
122 e->state = (cta_state_t) state;
123 snprintf(e->owner, sizeof(e->owner), "%s", owner);
124 snprintf(e->req.path, sizeof(e->req.path), "%s", pathbuf);
125 if ((uint64_t) id >= q->next_id) {
126 q->next_id = (uint64_t) id + 1;
127 }
128 }
129 q->journal = f; /* keep open for append */
130 return 0;
131}

133cta_req_t *
134cta_queue_submit(brix_cta_queue_t *q, const cta_request_t *r, const char *owner)
135{
136 int i;
137
138 for (i = 0; i < CTA_QUEUE_MAX; i++) {
139 if (!q->slots[i].in_use) {
140 cta_req_t *e = &q->slots[i];
141 memset(e, 0, sizeof(*e));
142 e->req = *r;
143 e->state = CTA_ST_SUBMITTED;
144 e->id = q->next_id++;
145 e->in_use = 1;
146 e->queue = q;
147 if (owner != NULL) {
148 size_t n = strlen(owner);
149 if (n >= sizeof(e->owner)) {
150 n = sizeof(e->owner) - 1;
151 }
152 memcpy(e->owner, owner, n);
153 e->owner[n] = '\0';
154 }
155 journal_append(q, e);
156 return e;
157 }
158 }
159 return NULL; /* full */
160}

175int
176cta_queue_transition(cta_req_t *e, cta_state_t to)
177{
178 if (!transition_ok(e->state, to)) {
179 return -1;
180 }
181 e->state = to;
182 if (e->queue != NULL) {
183 journal_append((brix_cta_queue_t *) e->queue, e);
184 }
185 return 0;
186}

188int
189cta_queue_cancel(brix_cta_queue_t *q, uint64_t id, const char *requester,
190 int is_admin)
191{
192 cta_req_t *e = cta_queue_find(q, id);
193
194 if (e == NULL) {
195 return CTA_QUEUE_ENOENT;
196 }
197 if (!is_admin && (requester == NULL || strcmp(requester, e->owner) != 0)) {
198 return CTA_QUEUE_EACCES;
199 }
200 return cta_queue_transition(e, CTA_ST_CANCELED);
201}
