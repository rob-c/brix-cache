20void
21brix_scan_tb_refill(brix_scan_tb_t *tb, uint64_t now_ns)
22{
23 double dt_s;
24
25 if (tb->rate_bps <= 0.0) {
26 return;
27 }
28 if (now_ns <= tb->last_ns) { /* clock went backwards / no time passed */
29 tb->last_ns = now_ns;
30 return;
31 }
32 dt_s = (double) (now_ns - tb->last_ns) / 1.0e9;
33 tb->tokens += tb->rate_bps * dt_s;
34 if (tb->tokens > tb->capacity) {
35 tb->tokens = tb->capacity;
36 }
37 tb->last_ns = now_ns;
38}

40uint64_t
41brix_scan_tb_wait_ns(brix_scan_tb_t *tb, double need, uint64_t now_ns)
42{
43 double deficit;
44
45 if (tb->rate_bps <= 0.0) {
46 return 0; /* unlimited */
47 }
48 brix_scan_tb_refill(tb, now_ns);
49 if (tb->tokens >= need) {
50 return 0;
51 }
52 deficit = need - tb->tokens;
53 return (uint64_t) ((deficit / tb->rate_bps) * 1.0e9);
54}

56void
57brix_scan_tb_consume(brix_scan_tb_t *tb, double need)

59 tb->tokens -= need;

78double
79brix_scan_adapt(int foreground_active, double latency_ewma_ms,
80 double nominal_latency_ms)
81{
82 double pressure_factor;
83 double latency_factor;
84 double m;
85
86 if (nominal_latency_ms <= 0.0) {
87 nominal_latency_ms = 1.0;
88 }
89 if (foreground_active < 0) {
90 foreground_active = 0;
91 }
92
93 /* each active foreground request damps the rate; latency above nominal
94 * shrinks it proportionally */
95 pressure_factor = 1.0 / (1.0 + 0.5 * (double) foreground_active);
96 latency_factor = (latency_ewma_ms <= nominal_latency_ms)
97 ? 1.0
98 : nominal_latency_ms / latency_ewma_ms;
99
100 m = pressure_factor * latency_factor;
101 if (m > 1.0) {
102 m = 1.0;
103 }
104 if (m < 0.1) {
105 m = 0.1;
106 }
107 return m;
108}
