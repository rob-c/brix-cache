39/* pblock_fnv — stable 64-bit hash of a logical path, used as a synthetic inode
40 * (the namespace lives in the catalog, not on a filesystem with real inodes). */
41uint64_t
42pblock_fnv(const char *s)
43{
44 uint64_t h = 1469598103934665603ULL; /* FNV-1a offset basis */
45
46 while (*s != '\0') {
47 h ^= (unsigned char) *s++;
48 h *= 1099511628211ULL; /* FNV prime */
49 }
50 return h;
51}

53/* pblock_mkdir_p — create `path` and any missing parents (mode 0700), tolerating
54 * components that already exist. Returns 0 or -1/errno. */
55int
56pblock_mkdir_p(const char *path)
57{
58 char tmp[PATH_MAX];
59 char *p;
60 size_t len;
61
62 len = (size_t) snprintf(tmp, sizeof(tmp), "%s", path);
63 if (len == 0 || len >= sizeof(tmp)) {
64 errno = len == 0 ? EINVAL : ENAMETOOLONG;
65 return -1;
66 }
67 if (tmp[len - 1] == '/') {
68 tmp[len - 1] = '\0';
69 }
70
71 for (p = tmp + 1; *p != '\0'; p++) {
72 if (*p == '/') {
73 *p = '\0';
74 if (mkdir(tmp, 0700) != 0 && errno != EEXIST) {
75 return -1;
76 }
77 *p = '/';
78 }
79 }
80 if (mkdir(tmp, 0700) != 0 && errno != EEXIST) {
81 return -1;
82 }
83 return 0;
84}

86/* pblock_gen_blob_id — fill out[] with 32 lowercase hex chars from 16 CSPRNG
87 * bytes. Returns 0 or -1/errno. */
88int
89pblock_gen_blob_id(char out[PBLOCK_BLOB_ID_CAP])
90{
91 static const char hex[] = "0123456789abcdef";
92 unsigned char raw[16];
93 size_t got = 0;
94 int i;
95
96 while (got < sizeof(raw)) {
97 ssize_t n = getrandom(raw + got, sizeof(raw) - got, 0);
98
99 if (n < 0) {
100 if (errno == EINTR) {
101 continue;
102 }
103 return -1;
104 }
105 got += (size_t) n;
106 }
107 for (i = 0; i < 16; i++) {
108 out[i * 2] = hex[raw[i] >> 4];
109 out[i * 2 + 1] = hex[raw[i] & 0x0f];
110 }
111 out[PBLOCK_BLOB_ID_HEX] = '\0';
112 return 0;
113}

181/* pblock_write_blocks — write `len` bytes of buf at file offset `off`, striped
182 * across the object's block files. blk0_fd (>=0) is reused for block 0; higher
183 * blocks (and block 0 when blk0_fd<0) are opened O_RDWR|O_CREAT transiently.
184 * Returns bytes written, or -1/errno when nothing was written. */
185/* pblock_xform_write — read-modify-write path for a transform-configured export.
186 * Every block (including block 0) is a headered file the block-0 fd never serves,
187 * so the persistent fd is ignored; each touched block is loaded whole, overlaid
188 * with the new bytes, and re-encoded. Returns bytes written or -1/errno. */
189static ssize_t
190pblock_xform_write(const pblock_state_t *st, const char *blob_id, int64_t bs,
191 const void *buf, size_t len, off_t off)
192{
193 const char *p = buf;
194 size_t done = 0;
195 unsigned char *scratch = malloc((size_t) bs);
196
197 if (scratch == NULL) {
198 errno = ENOMEM;
199 return -1;
200 }
201 while (done < len) {
202 int64_t idx = (off + (off_t) done) / bs;
203 int64_t boff = (off + (off_t) done) % bs;
204 size_t room = (size_t) (bs - boff);
205 size_t chunk = len - done < room ? len - done : room;
206 char bp[PATH_MAX];
207 uint32_t llen, nl;
208
209 if (pblock_block_path(st, blob_id, idx, bp, sizeof(bp)) != 0
210 || pblock_xform_block_load(&st->xform, idx, bp, scratch, bs, &llen)
211 != 0)
212 {
213 break;
214 }
215 memcpy(scratch + boff, p + done, chunk);
216 nl = (uint32_t) (boff + (int64_t) chunk);
217 if (llen > nl) {
218 nl = llen;
219 }
220 if (pblock_xform_block_store(&st->xform, idx, bp, scratch, nl, bs) != 0) {
221 break;
222 }
223 done += chunk;
224 }
225 free(scratch);
226 return done ? (ssize_t) done : -1;
227}

229ssize_t
230pblock_write_blocks(const pblock_state_t *st, const char *blob_id, int64_t bs,
231 int blk0_fd, const void *buf, size_t len, off_t off)
232{
233 const char *p = buf;
234 size_t done = 0;
235
236 if (pblock_xform_active(&st->xform)) {
237 return pblock_xform_write(st, blob_id, bs, buf, len, off);
238 }
239
240 while (done < len) {
241 int64_t idx = (off + (off_t) done) / bs;
242 int64_t boff = (off + (off_t) done) % bs;
243 size_t room = (size_t) (bs - boff);
244 size_t chunk = len - done < room ? len - done : room;
245 int fd, transient = 0;
246 size_t w = 0;
247
248 if (idx == 0 && blk0_fd >= 0) {
249 fd = blk0_fd;
250 } else {
251 char bp[PATH_MAX];
252
253 if (pblock_block_path(st, blob_id, idx, bp, sizeof(bp)) != 0) {
254 return done ? (ssize_t) done : -1;
255 }
256 fd = open(bp, O_RDWR | O_CREAT, 0600);
257 if (fd < 0) {
258 return done ? (ssize_t) done : -1;
259 }
260 transient = 1;
261 }
262
263 while (w < chunk) {
264 ssize_t k = pwrite(fd, p + done + w, chunk - w, boff + (off_t) w);
265
266 if (k < 0) {
267 if (errno == EINTR) {
268 continue;
269 }
270 if (transient) {
271 close(fd);
272 }
273 return (done + w) ? (ssize_t) (done + w) : -1;
274 }
275 w += (size_t) k;
276 }
277 if (transient) {
278 close(fd);
279 }
280 done += chunk;
281 }
282 return (ssize_t) done;
283}

285/* pblock_read_blocks — read `len` bytes at file offset `off` (the caller has
286 * already clamped len to the file size), striped across block files. blk0_fd
287 * (>=0) is reused for block 0. Missing blocks and short blocks read back as
288 * zeros (holes). Returns bytes produced, or -1/errno. */
289/* pblock_xform_read — decode path for a transform-configured export: load each
290 * touched block whole (holes and short blocks zero-fill) and copy out the wanted
291 * sub-range. Returns bytes produced or -1/errno. */
292static ssize_t
293pblock_xform_read(const pblock_state_t *st, const char *blob_id, int64_t bs,
294 void *buf, size_t len, off_t off)
295{
296 char *p = buf;
297 size_t done = 0;
298 unsigned char *scratch = malloc((size_t) bs);
299
300 if (scratch == NULL) {
301 errno = ENOMEM;
302 return -1;
303 }
304 while (done < len) {
305 int64_t idx = (off + (off_t) done) / bs;
306 int64_t boff = (off + (off_t) done) % bs;
307 size_t room = (size_t) (bs - boff);
308 size_t chunk = len - done < room ? len - done : room;
309 char bp[PATH_MAX];
310 uint32_t llen;
311
312 if (pblock_block_path(st, blob_id, idx, bp, sizeof(bp)) != 0
313 || pblock_xform_block_load(&st->xform, idx, bp, scratch, bs, &llen)
314 != 0)
315 {
316 break;
317 }

353 if (fd < 0) {
354 if (errno == ENOENT) {
355 memset(p + done, 0, chunk); /* whole-block hole */
356 done += chunk;
357 continue;
358 }
359 return done ? (ssize_t) done : -1;
360 }
361 transient = 1;
362 }
363
364 r = pread(fd, p + done, chunk, boff);
365 if (transient) {
366 close(fd);
367 }
368 if (r < 0) {
369 return done ? (ssize_t) done : -1;
370 }
371 if ((size_t) r < chunk) {
372 memset(p + done + (size_t) r, 0, chunk - (size_t) r); /* tail hole */
373 }
374 done += chunk;
375 }
376 return (ssize_t) done;

379/* pblock_remove_blocks — unlink every block file of an object and remove its
380 * per-object directory (and best-effort the now-possibly-empty fan-out dirs). */
381void
382pblock_remove_blocks(const pblock_state_t *st, const char *blob_id,
383 int64_t size, int64_t bs)
384{
385 int64_t last = pblock_last_block(size, bs);
386 int64_t i;
387 char path[PATH_MAX];
388
389 for (i = 0; i <= last; i++) {
390 if (pblock_block_path(st, blob_id, i, path, sizeof(path)) == 0) {
391 unlink(path);
392 }
393 }
394 if (pblock_obj_dir(st, blob_id, path, sizeof(path)) == 0) {
395 rmdir(path);
396 }
397 /* best-effort: drop the now-empty fan-out directories (skip on truncation) */
398 if (snprintf(path, sizeof(path), "%s/%c%c/%c%c", st->data_dir,
399 blob_id[0], blob_id[1], blob_id[2], blob_id[3])
400 < (int) sizeof(path))
401 {
402 rmdir(path);
403 }
404 if (snprintf(path, sizeof(path), "%s/%c%c", st->data_dir, blob_id[0],
405 blob_id[1]) < (int) sizeof(path))
406 {
407 rmdir(path);
408 }
409}

411/* pblock_copy_one_block — copy one source block file to a destination block path
412 * via a pread/pwrite loop; a missing source block (hole) is skipped. Returns the
413 * bytes copied (>=0) or -1/errno. */
414ssize_t
415pblock_copy_one_block(const char *src_path, const char *dst_path)
416{
417 char buf[65536];
418 off_t off = 0;
419 int sfd, dfd;
420 ssize_t copied = 0;
421
422 sfd = open(src_path, O_RDONLY);
423 if (sfd < 0) {
424 return errno == ENOENT ? 0 : -1; /* hole: nothing to copy */
425 }
426 dfd = open(dst_path, O_RDWR | O_CREAT | O_EXCL, 0600);
427 if (dfd < 0) {
428 close(sfd);
429 return -1;
430 }
431
432 for ( ;; ) {
433 ssize_t n = pread(sfd, buf, sizeof(buf), off);
434 ssize_t w = 0;
435
436 if (n < 0) {
437 close(sfd);
438 close(dfd);
439 return -1;
440 }
441 if (n == 0) {
442 break;
443 }
444 while (w < n) {
445 ssize_t k = pwrite(dfd, buf + w, (size_t) (n - w), off + w);
446
447 if (k < 0) {
448 close(sfd);
449 close(dfd);
450 return -1;
451 }
452 w += k;
453 }
454 off += n;
455 }
456 close(sfd);
457 close(dfd);
458 copied = off;
459 return copied;
460}
