39/* pblock_fnv — stable 64-bit hash of a logical path, used as a synthetic inode
40 * (the namespace lives in the catalog, not on a filesystem with real inodes). */
41uint64_t
42pblock_fnv(const char *s)
43{
44 uint64_t h = 1469598103934665603ULL; /* FNV-1a offset basis */
45
46 while (*s != '\0') {
47 h ^= (unsigned char) *s++;
48 h *= 1099511628211ULL; /* FNV prime */
49 }
50 return h;
51}

53/* pblock_mkdir_p — create `path` and any missing parents (mode 0700), tolerating
54 * components that already exist. Returns 0 or -1/errno. */
55int
56pblock_mkdir_p(const char *path)
57{
58 char tmp[PATH_MAX];
59 char *p;
60 size_t len;
61
62 len = (size_t) snprintf(tmp, sizeof(tmp), "%s", path);
63 if (len == 0 || len >= sizeof(tmp)) {
64 errno = len == 0 ? EINVAL : ENAMETOOLONG;
65 return -1;
66 }
67 if (tmp[len - 1] == '/') {
68 tmp[len - 1] = '\0';
69 }
70
71 for (p = tmp + 1; *p != '\0'; p++) {
72 if (*p == '/') {
73 *p = '\0';
74 if (mkdir(tmp, 0700) != 0 && errno != EEXIST) {
75 return -1;
76 }
77 *p = '/';
78 }
79 }
80 if (mkdir(tmp, 0700) != 0 && errno != EEXIST) {
81 return -1;
82 }
83 return 0;
84}

86/* pblock_gen_blob_id — fill out[] with 32 lowercase hex chars from 16 CSPRNG
87 * bytes. Returns 0 or -1/errno. */
88int
89pblock_gen_blob_id(char out[PBLOCK_BLOB_ID_CAP])
90{
91 static const char hex[] = "0123456789abcdef";
92 unsigned char raw[16];
93 size_t got = 0;
94 int i;
95
96 while (got < sizeof(raw)) {
97 ssize_t n = getrandom(raw + got, sizeof(raw) - got, 0);
98
99 if (n < 0) {
100 if (errno == EINTR) {
101 continue;
102 }
103 return -1;
104 }
105 got += (size_t) n;
106 }
107 for (i = 0; i < 16; i++) {
108 out[i * 2] = hex[raw[i] >> 4];
109 out[i * 2 + 1] = hex[raw[i] & 0x0f];
110 }
111 out[PBLOCK_BLOB_ID_HEX] = '\0';
112 return 0;
113}

179/* pblock_write_blocks — write `len` bytes of buf at file offset `off`, striped
180 * across the object's block files. blk0_fd (>=0) is reused for block 0; higher
181 * blocks (and block 0 when blk0_fd<0) are opened O_RDWR|O_CREAT transiently.
182 * Returns bytes written, or -1/errno when nothing was written. */
183ssize_t
184pblock_write_blocks(const pblock_state_t *st, const char *blob_id, int64_t bs,
185 int blk0_fd, const void *buf, size_t len, off_t off)
186{
187 const char *p = buf;
188 size_t done = 0;
189
190 while (done < len) {
191 int64_t idx = (off + (off_t) done) / bs;
192 int64_t boff = (off + (off_t) done) % bs;
193 size_t room = (size_t) (bs - boff);
194 size_t chunk = len - done < room ? len - done : room;
195 int fd, transient = 0;
196 size_t w = 0;
197
198 if (idx == 0 && blk0_fd >= 0) {
199 fd = blk0_fd;
200 } else {
201 char bp[PATH_MAX];
202
203 if (pblock_block_path(st, blob_id, idx, bp, sizeof(bp)) != 0) {
204 return done ? (ssize_t) done : -1;
205 }
206 fd = open(bp, O_RDWR | O_CREAT, 0600);
207 if (fd < 0) {
208 return done ? (ssize_t) done : -1;
209 }
210 transient = 1;
211 }
212
213 while (w < chunk) {
214 ssize_t k = pwrite(fd, p + done + w, chunk - w, boff + (off_t) w);
215
216 if (k < 0) {
217 if (errno == EINTR) {
218 continue;
219 }
220 if (transient) {
221 close(fd);
222 }
223 return (done + w) ? (ssize_t) (done + w) : -1;
224 }
225 w += (size_t) k;
226 }
227 if (transient) {
228 close(fd);
229 }
230 done += chunk;
231 }
232 return (ssize_t) done;
233}

235/* pblock_read_blocks — read `len` bytes at file offset `off` (the caller has
236 * already clamped len to the file size), striped across block files. blk0_fd
237 * (>=0) is reused for block 0. Missing blocks and short blocks read back as
238 * zeros (holes). Returns bytes produced, or -1/errno. */
239ssize_t
240pblock_read_blocks(const pblock_state_t *st, const char *blob_id, int64_t bs,
241 int blk0_fd, void *buf, size_t len, off_t off)
242{
243 char *p = buf;
244 size_t done = 0;
245
246 while (done < len) {
247 int64_t idx = (off + (off_t) done) / bs;
248 int64_t boff = (off + (off_t) done) % bs;
249 size_t room = (size_t) (bs - boff);
250 size_t chunk = len - done < room ? len - done : room;
251 int fd, transient = 0;
252 ssize_t r;
253
254 if (idx == 0 && blk0_fd >= 0) {
255 fd = blk0_fd;
256 } else {
257 char bp[PATH_MAX];
258
259 if (pblock_block_path(st, blob_id, idx, bp, sizeof(bp)) != 0) {
260 return done ? (ssize_t) done : -1;
261 }
262 fd = open(bp, O_RDONLY);
263 if (fd < 0) {
264 if (errno == ENOENT) {
265 memset(p + done, 0, chunk); /* whole-block hole */
266 done += chunk;
267 continue;
268 }
269 return done ? (ssize_t) done : -1;
270 }
271 transient = 1;
272 }
273
274 r = pread(fd, p + done, chunk, boff);
275 if (transient) {
276 close(fd);
277 }
278 if (r < 0) {
279 return done ? (ssize_t) done : -1;
280 }
281 if ((size_t) r < chunk) {
282 memset(p + done + (size_t) r, 0, chunk - (size_t) r); /* tail hole */
283 }
284 done += chunk;
285 }
286 return (ssize_t) done;
287}

289/* pblock_remove_blocks — unlink every block file of an object and remove its
290 * per-object directory (and best-effort the now-possibly-empty fan-out dirs). */
291void
292pblock_remove_blocks(const pblock_state_t *st, const char *blob_id,
293 int64_t size, int64_t bs)
294{
295 int64_t last = pblock_last_block(size, bs);
296 int64_t i;
297 char path[PATH_MAX];
298
299 for (i = 0; i <= last; i++) {
300 if (pblock_block_path(st, blob_id, i, path, sizeof(path)) == 0) {
301 unlink(path);
302 }
303 }
304 if (pblock_obj_dir(st, blob_id, path, sizeof(path)) == 0) {
305 rmdir(path);
306 }
307 /* best-effort: drop the now-empty fan-out directories (skip on truncation) */
308 if (snprintf(path, sizeof(path), "%s/%c%c/%c%c", st->data_dir,
309 blob_id[0], blob_id[1], blob_id[2], blob_id[3])
310 < (int) sizeof(path))
311 {
312 rmdir(path);
313 }
314 if (snprintf(path, sizeof(path), "%s/%c%c", st->data_dir, blob_id[0],
315 blob_id[1]) < (int) sizeof(path))
316 {
317 rmdir(path);
318 }
319}

321/* pblock_copy_one_block — copy one source block file to a destination block path
322 * via a pread/pwrite loop; a missing source block (hole) is skipped. Returns the
323 * bytes copied (>=0) or -1/errno. */
324ssize_t
325pblock_copy_one_block(const char *src_path, const char *dst_path)
326{
327 char buf[65536];
328 off_t off = 0;
329 int sfd, dfd;
330 ssize_t copied = 0;
331
332 sfd = open(src_path, O_RDONLY);
333 if (sfd < 0) {
334 return errno == ENOENT ? 0 : -1; /* hole: nothing to copy */
335 }
336 dfd = open(dst_path, O_RDWR | O_CREAT | O_EXCL, 0600);
337 if (dfd < 0) {
338 close(sfd);
339 return -1;
340 }
341
342 for ( ;; ) {
343 ssize_t n = pread(sfd, buf, sizeof(buf), off);
344 ssize_t w = 0;
345
346 if (n < 0) {
347 close(sfd);
348 close(dfd);
349 return -1;
350 }
351 if (n == 0) {
352 break;
353 }
354 while (w < n) {
355 ssize_t k = pwrite(dfd, buf + w, (size_t) (n - w), off + w);
356
357 if (k < 0) {
358 close(sfd);
359 close(dfd);
360 return -1;
361 }
362 w += k;
363 }
364 off += n;
365 }
366 close(sfd);
367 close(dfd);
368 copied = off;
369 return copied;
370}
