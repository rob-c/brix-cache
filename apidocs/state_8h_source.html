4/*
5 * Per-connection state machine enum and opaque forward declarations.
6 *
7 * Normal flow (no TLS, no upstream):
8 * HANDSHAKE → REQ_HEADER → REQ_PAYLOAD (if dlen > 0) → REQ_HEADER → …
9 *
10 * SENDING: entered when brix_queue_response_base() gets EAGAIN from
11 * c->send(). The remaining bytes stay in ctx->wbuf; the write event
12 * is armed. The read event is NOT active while SENDING.
13 *
14 * AIO: entered when a pread(2)/pwrite(2) is posted to the nginx thread
15 * pool. Both read and write events are disarmed. The completion
16 * callback re-arms the write event with the response already queued.
17 *
18 * TLS_HANDSHAKE: entered after sending kXR_haveTLS in a kXR_protocol
19 * response. Both events are re-armed by the TLS accept machinery.
20 *
21 * UPSTREAM: entered while waiting for the redirector to reply. The
22 * client-facing read event is disarmed; upstream events drive progress.
23 */
24typedef enum {
25 XRD_ST_HANDSHAKE, /* accumulating the 20-byte client hello */
26 XRD_ST_REQ_HEADER, /* accumulating a 24-byte request header */
27 XRD_ST_REQ_PAYLOAD, /* accumulating dlen bytes of payload */
28 XRD_ST_SENDING, /* draining a large pending write buffer */
29 XRD_ST_AIO, /* async file I/O posted to thread pool */
30 XRD_ST_TLS_HANDSHAKE, /* kXR_ableTLS: TLS accept in progress */
31 XRD_ST_UPSTREAM, /* upstream redirector query in progress */
32 XRD_ST_PROXY, /* proxy request forwarded, awaiting response */
33 XRD_ST_WAITING_CMS, /* kYR_locate sent to manager; awaiting kYR_select */
34 XRD_ST_WAITING_FRM, /* kXR_waitresp sent; awaiting async stage completion */
35} brix_state_t;
