2 * lifecycle_timing.c — implementation of the lifecycle phase stopwatch.
3 *
4 * See lifecycle_timing.h for the WHAT/WHY/HOW. This file is deliberately tiny and
5 * dependency-free beyond ngx_core: it is linked into the module so the permanent
6 * one-line phase summaries are always available, with no per-request cost.
7 */

9#include "lifecycle_timing.h"
10#include <time.h>

12/*
13 * brix_phase_now_ns — monotonic clock in nanoseconds.
14 *
15 * CLOCK_MONOTONIC (not _COARSE): vDSO-backed and ~ns-resolution, so a sub-ms phase
16 * is measured honestly instead of being rounded to a millisecond tick. On the
17 * (effectively impossible) clock failure we return 0; callers clamp negative
18 * deltas to 0, so a bad reading degrades to "0us", never to a garbage figure.
19 */
20uint64_t
21brix_phase_now_ns(void)
22{
23 struct timespec ts;
24
25 if (clock_gettime(CLOCK_MONOTONIC, &ts) != 0) {
26 return 0;
27 }
28
29 return (uint64_t) ts.tv_sec * 1000000000ull + (uint64_t) ts.tv_nsec;
30}

32/* Begin a run: snapshot the clock as both the absolute start and the first
33 * per-mark baseline, and empty the summary buffer. */
34void
35brix_phase_timer_start(brix_phase_timer_t *t)
36{
37 if (t == NULL) {
38 return;
39 }
40
41 t->start_ns = brix_phase_now_ns();
42 t->last_ns = t->start_ns;
43 t->pos = t->summary;
44 t->summary[0] = '\0';
45}

47/*
48 * us_since — microseconds elapsed from `from_ns` to now, clamped at 0 if the
49 * monotonic clock appears to have gone backwards (or was unreadable).
50 */
51static uint64_t
52us_since(uint64_t from_ns, uint64_t now_ns)
53{
54 if (now_ns <= from_ns) {
55 return 0;
56 }
57
58 return (now_ns - from_ns) / 1000ull;
59}

61/*
62 * brix_phase_mark — append "<name>=<delta>us " for the time since the previous
63 * mark, then advance the per-mark baseline. Writes are bounded by the buffer
64 * capacity; once full, further marks are dropped rather than overflowing (the
65 * total in _log still reflects real elapsed time, so no information is fabricated).
66 */
67void
68brix_phase_mark(brix_phase_timer_t *t, const char *name)
69{
70 uint64_t now_ns;
71 u_char *end;
72
73 if (t == NULL || name == NULL) {
74 return;
75 }
76
77 now_ns = brix_phase_now_ns();
78 end = t->summary + sizeof(t->summary) - 1;
79
80 /* ngx_snprintf never writes past `end` and returns the new cursor, but it does
81 * NOT null-terminate — so terminate explicitly (t->pos <= end leaves room at
82 * summary[cap-1]) to keep the buffer a valid C string for the "%s" in _log. */
83 t->pos = ngx_snprintf(t->pos, (size_t) (end - t->pos), "%s=%uLus ",
84 name, us_since(t->last_ns, now_ns));
85 *t->pos = '\0';
86
87 t->last_ns = now_ns;
88}

90/*
91 * brix_phase_timer_log — emit the accumulated per-phase summary plus a trailing
92 * total at NOTICE. NOTICE (not INFO) so it survives the default log level: an
93 * operator gets a boot-cost breakdown without raising verbosity, yet it is one
94 * line per lifecycle event so it never spams.
95 */
96void
97brix_phase_timer_log(brix_phase_timer_t *t, ngx_log_t *log,
98 const char *context)
99{
100 uint64_t total_us;
101
102 if (t == NULL || log == NULL) {
103 return;
104 }
105
106 total_us = us_since(t->start_ns, brix_phase_now_ns());
107
108 ngx_log_error(NGX_LOG_NOTICE, log, 0, "%s: %stotal=%uLus",
109 context != NULL ? context : "xrootd lifecycle",
110 t->summary, total_us);
111}
