1/*
2 * readv_seg.h — kXR_readv segment-header codec (single source of truth).
3 *
4 * WHAT: each kXR_readv request and response carries an array of fixed 16-byte
5 * segment headers, one per requested range:
6 * [ fhandle[4] ][ rlen[4 BE] ][ offset[8 BE] ] (BRIX_READV_SEGSIZE)
7 * This header owns that layout for everyone who assembles or reads it:
8 * - brix_readv_seg_pack: fields -> 16-byte header
9 * - brix_readv_seg_rlen: 16-byte header -> rlen (the field a reader needs)
10 * WHY: the layout — and the magic +4 / +8 / 16 offsets — were hand-spelled in
11 * BOTH directions across BOTH trees: the native client builds the request
12 * segments and reads the response segments (client/lib/ops_file.c), and the
13 * module builds the response segments (src/read/readv.c). pgio already
14 * single-sources the paged-I/O wire layout; readv was the remaining gap.
15 * HOW: header-only static inlines over the shared, unaligned-safe big-endian
16 * accessors in frame_hdr.h — no ngx, no allocation — so the same code
17 * compiles into the nginx module and the ngx-free client. The module's
18 * request *parse* uses the typed `readahead_list` struct directly (clean,
19 * not raw bytes) and is left as-is; this codec covers the raw-byte sites.
20 *
21 * Clean-room: layout from src/protocol readahead_list (vs XProtocol read_args).
22 */
23#ifndef BRIX_PROTOCOL_READV_SEG_H
24#define BRIX_PROTOCOL_READV_SEG_H

27#include <string.h>

32/*
33 * Pack one readv segment header into `out` (BRIX_READV_SEGSIZE bytes): the
34 * 4-byte file handle verbatim, then rlen and offset in big-endian. `offset` and
35 * `rlen` are host-order; the encode is the exact inverse of brix_readv_seg_rlen
36 * (and of the typed readahead_list the module parses on the request side).
37 */
38static inline void
39brix_readv_seg_pack(uint8_t out[BRIX_READV_SEGSIZE], const uint8_t fhandle[4],
40 uint32_t rlen, uint64_t offset)
41{
42 memcpy(out, fhandle, 4);
43 xrd_put_u32_be(out + 4, rlen);
44 xrd_put_u64_be(out + 8, offset);
45}
46
47/* Read the rlen field (host order) from a 16-byte segment header. */
48static inline uint32_t
49brix_readv_seg_rlen(const uint8_t in[BRIX_READV_SEGSIZE])
50{
51 return xrd_get_u32_be(in + 4);
52}
53
54#endif /* BRIX_PROTOCOL_READV_SEG_H */
