2 * health_check_probe.c — Phase 22 active stream health-check probe I/O state
3 * machine (see health_check.h / health_check_internal.h).
4 *
5 * This is the async half of the health check: once health_check.c has opened a

37/* write side */
38/*
39 * Drain hc->wbuf to the socket without blocking. Returns NGX_OK once the whole
40 * buffer is sent (and read events re-armed so the reply can arrive), NGX_AGAIN
41 * if the socket is full (write event re-armed; brix_hc_write_handler resumes
42 * here), or NGX_ERROR on a fatal send error.
43 */
44ngx_int_t
45brix_hc_flush(brix_hc_ctx_t *hc)
46{
47 ngx_connection_t *c = hc->conn;
48 ssize_t n;
49
50 /* Partial-write loop: send() may consume only part of the buffer, so track
51 * wbuf_pos and resume from there on the next writable event. */
52 while (hc->wbuf_pos < hc->wbuf_len) {
53 n = c->send(c, hc->wbuf + hc->wbuf_pos, hc->wbuf_len - hc->wbuf_pos);
54 if (n > 0) {
55 hc->wbuf_pos += (size_t) n;
56 continue;
57 }
58 if (n == NGX_AGAIN) {
59 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
60 return NGX_ERROR;
61 }
62 return NGX_AGAIN;
63 }
64 return NGX_ERROR;
65 }
66 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
67 return NGX_ERROR;
68 }
69 return NGX_OK;
70}

72/*
73 * Build and send the final liveness probe (kXR_ping or kXR_stat "/") after a
74 * successful login, transitioning to XRD_HC_PROBE. The reply's status is the
75 * pass/fail verdict, decided back in brix_hc_dispatch().
76 */
77static void
78brix_hc_send_probe(brix_hc_ctx_t *hc)
79{
80 if (hc->probe_type == BRIX_HC_TYPE_STAT) {
81 /* kXR_stat request followed by a 1-byte path body "/". Over-allocate
82 * by 1 so the path char sits immediately after the fixed header. */
83 size_t total = sizeof(ClientStatRequest) + 1; /* path "/" */
84 ClientStatRequest *r = ngx_palloc(hc->pool, total);
85 if (r == NULL) { brix_hc_finish(hc, 0); return; }
86 ngx_memzero(r, sizeof(*r));
87 /* streamid is a 2-byte client tag echoed in the reply; {0,2} marks this
88 * as a health request (see ping branch below). */
89 r->streamid[0] = 0;
90 r->streamid[1] = 2;
91 r->requestid = htons(kXR_stat); /* opcode, network order */
92 r->dlen = htonl((kXR_int32) 1); /* body length = 1 ("/") */
93 ((u_char *) r)[sizeof(*r)] = '/'; /* append path after header */
94 hc->wbuf = (u_char *) r;
95 hc->wbuf_len = total;
96 } else {
97 ClientPingRequest *r = ngx_palloc(hc->pool, sizeof(*r));
98 if (r == NULL) { brix_hc_finish(hc, 0); return; }
99 ngx_memzero(r, sizeof(*r));
100 r->streamid[0] = 0;
101 r->streamid[1] = 2; /* streamid 2 marks a health request */
102 r->requestid = htons(kXR_ping);
103 r->dlen = 0; /* ping carries no body */
104 hc->wbuf = (u_char *) r;
105 hc->wbuf_len = sizeof(*r);
106 }
107 /* Reset both write and read accumulators for this fresh request/response. */
108 hc->wbuf_pos = 0;
109 hc->rhdr_pos = 0;
110 hc->resp_dlen = 0;
111 hc->resp_body = NULL;
112 hc->resp_body_pos = 0;
113 hc->phase = XRD_HC_PROBE;
114
115 if (brix_hc_flush(hc) == NGX_ERROR) {
116 brix_hc_finish(hc, 0);
117 }
118}

120/*
121 * Writable-event handler. First writable event after a non-blocking connect()
122 * confirms (or rejects) the TCP connection via SO_ERROR; subsequent calls just
123 * resume a partially-flushed write buffer.
124 */
125void
126brix_hc_write_handler(ngx_event_t *wev)
127{
128 ngx_connection_t *c = wev->data;
129 brix_hc_ctx_t *hc = c->data;
130 ngx_int_t rc;
131
132 if (wev->timedout) {
133 brix_hc_finish(hc, 0);
134 return;
135 }
136
137 /* Non-blocking connect completes asynchronously: the first writable event
138 * means connect() finished — SO_ERROR carries success (0) or the failure
139 * code, since connect() itself returned EINPROGRESS earlier. */
140 if (hc->connecting) {
141 int err = 0;
142 socklen_t len = sizeof(err);
143 if (getsockopt(c->fd, SOL_SOCKET, SO_ERROR, (char *) &err, &len) == -1
144 || err != 0)
145 {
146 BRIX_DIAG_WARN(hc->log, err, /* may be 0 */
147 "brix: health check failed to connect to %s:%d",
148 "the cluster member is down or unreachable from the manager",
149 "check that member's xrootd/data service and the network path "
150 "to it; it is marked DOWN and clients are routed elsewhere "
151 "until it passes again",
152 hc->host, (int) hc->port);
153 brix_hc_finish(hc, 0);
154 return;
155 }
156 hc->connecting = 0;
157 }
158
159 rc = brix_hc_flush(hc);
160 if (rc == NGX_ERROR) {
161 brix_hc_finish(hc, 0);
162 }
163}

166/* read side */
167/* Accumulate one full ServerResponseHdr + body frame; return NGX_OK when a
168 * complete frame is ready, NGX_AGAIN to wait for more, NGX_ERROR on close. */
169static ngx_int_t
170brix_hc_recv_frame(brix_hc_ctx_t *hc)
171{
172 ngx_connection_t *c = hc->conn;
173 ssize_t n;
174
175 /* Stage 1: fill the fixed 8-byte header, possibly across several reads
176 * (rhdr_pos tracks how much has arrived). Only once it is complete do we
177 * decode status/dlen and learn the body length. */
178 if (hc->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
179 size_t need = XRD_RESPONSE_HDR_LEN - hc->rhdr_pos;
180 n = c->recv(c, hc->rhdr + hc->rhdr_pos, need);
181 if (n == NGX_AGAIN) { return NGX_AGAIN; }
182 if (n <= 0) { return NGX_ERROR; } /* 0 = peer closed */
183 hc->rhdr_pos += (size_t) n;
184 if (hc->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
185 return NGX_AGAIN;
186 }
187 /* Header complete: decode the wire fields (big-endian on the wire). */
188 {
189 ServerResponseHdr *h = (ServerResponseHdr *) (void *) hc->rhdr;
190 hc->resp_status = ntohs(h->status);
191 hc->resp_dlen = ntohl(h->dlen);
192 }
193 if (hc->resp_dlen > 0) {
194 /* Bound attacker/garbage-controlled body length: a probe reply is
195 * tiny, so cap the allocation rather than trust the wire dlen. */
196 if (hc->resp_dlen > 4096) { /* bound probe response bodies */
197 return NGX_ERROR;
198 }
199 hc->resp_body = ngx_palloc(c->pool, hc->resp_dlen);
200 if (hc->resp_body == NULL) { return NGX_ERROR; }
201 hc->resp_body_pos = 0;
202 }
203 }
204
205 /* Stage 2: fill the dlen-byte body (again possibly across several reads).
206 * Skipped entirely when dlen == 0. */
207 if (hc->resp_body_pos < hc->resp_dlen) {
208 size_t need = hc->resp_dlen - hc->resp_body_pos;
209 n = c->recv(c, hc->resp_body + hc->resp_body_pos, need);
210 if (n == NGX_AGAIN) { return NGX_AGAIN; }
211 if (n <= 0) { return NGX_ERROR; }
212 hc->resp_body_pos += (size_t) n;
213 if (hc->resp_body_pos < hc->resp_dlen) {
214 return NGX_AGAIN;
215 }
216 }
217
218 return NGX_OK;
219}

222/*
223 * Handshake-done callback for a Step-F TLS-upgraded probe. Mirrors the
224 * upstream brix_upstream_tls_handshake_done(): verify the handshake (and the
225 * peer, explicitly), restore the probe's event handlers, and resend a fresh
226 * kXR_login over the TLS channel — the probe then continues LOGIN -> PROBE
227 * exactly as on cleartext. Any failure is a probe failure: a server that
228 * demands TLS but cannot complete it cannot serve TLS clients.
229 */
230static void
231brix_hc_tls_handshake_done(ngx_connection_t *c)
232{
233 brix_hc_ctx_t *hc = c->data;
234 ClientLoginRequest *req;
235
236 if (!c->ssl->handshaked) {
237 ngx_log_error(NGX_LOG_WARN, hc->log, 0,
238 "brix: health check: %s:%d TLS handshake failed",
239 hc->host, (int) hc->port);
240 brix_hc_finish(hc, 0);
241 return;
242 }
243 if (SSL_get_verify_result(c->ssl->connection) != X509_V_OK) {
244 ngx_log_error(NGX_LOG_WARN, hc->log, 0,
245 "brix: health check: %s:%d TLS peer verification failed",
246 hc->host, (int) hc->port);
247 brix_hc_finish(hc, 0);
248 return;
249 }
250
251 hc->tls = 1;
252
253 /* Restore normal event handlers (the TLS handshake replaces them). */
254 c->read->handler = brix_hc_read_handler;
255 c->write->handler = brix_hc_write_handler;
256
257 /* Fresh kXR_login for the TLS channel (the plaintext one is discarded). */
258 req = ngx_palloc(hc->pool, sizeof(*req));
259 if (req == NULL) {
260 brix_hc_finish(hc, 0);
261 return;
262 }
263 brix_upstream_build_login(req);
264
265 hc->wbuf = (u_char *) req;
266 hc->wbuf_len = sizeof(*req);
267 hc->wbuf_pos = 0;
268 hc->phase = XRD_HC_LOGIN;
269
270 hc->rhdr_pos = 0;
271 hc->resp_dlen = 0;
272 hc->resp_body = NULL;
273 hc->resp_body_pos = 0;
274
275 if (brix_hc_flush(hc) == NGX_ERROR) {
276 brix_hc_finish(hc, 0);
277 }
278}

281/*
282 * XRD_HC_PROTOCOL step of brix_hc_dispatch(): consume the kXR_protocol reply
283 * and decide how the probe proceeds. Returns NGX_DONE when the step reached a
284 * verdict or issued its own follow-up send (the caller must return without
285 * touching hc — it may already be freed), or NGX_OK when the phase advanced to
286 * LOGIN and the caller should fall through to the accumulator reset.
287 */
288static ngx_int_t
289brix_hc_dispatch_protocol(brix_hc_ctx_t *hc)
290{
291 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return NGX_DONE; }
292
293 /* If the server demands TLS we don't probe deeper (no TLS in the
294 * probe path); it answered at the protocol level, so treat it as
295 * alive rather than blacklisting it.
296 *
297 * ServerResponseBody_Protocol layout: bytes 0-3 = pval (protocol
298 * version), bytes 4-7 = flags. Read the flags word at offset 4 and
299 * test the kXR_gotoTLS bit. Guard on dlen >= 8 so short/old replies
300 * that omit the flags simply fall through to the LOGIN phase. */
301 if (hc->resp_dlen >= 8) {
302 uint32_t flags_be;
303 ngx_memcpy(&flags_be, hc->resp_body + 4, sizeof(flags_be));
304 if (ntohl(flags_be) & kXR_gotoTLS) {
305#if (NGX_SSL)
306 /* Step F: with an outbound TLS ctx configured, probe DEEP —
307 * upgrade this connection and continue LOGIN -> PROBE over
308 * TLS, exactly as the upstream bootstrap does (the server
309 * discards the plaintext login pre-sent in the pipelined
310 * bootstrap). The probe deadline timer keeps running. */
311 if (hc->conf->upstream_tls
312 && hc->conf->upstream_tls_ctx != NULL)
313 {
314 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, hc->log, 0,
315 "brix: health check: %s:%d wants TLS; upgrading probe",
316 hc->host, (int) hc->port);
317 if (brix_outbound_start_tls(hc->conf->upstream_tls_ctx,
318 hc->conn, hc->host,
319 brix_hc_tls_handshake_done)
320 != NGX_OK)
321 {
322 brix_hc_finish(hc, 0);
323 }
324 return NGX_DONE;
325 }
326#endif
327 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, hc->log, 0,
328 "brix: health check: %s:%d wants TLS; "
329 "treating protocol-OK as alive", hc->host, (int) hc->port);
330 brix_hc_finish(hc, 1);
331 return NGX_DONE;
332 }
333 }
334 if (hc->tls_capable) {
335 /* No gotoTLS: send the deferred login now, in cleartext. (A
336 * TLS-capable probe never pipelined it — see brix_hc_start.) */
337 ClientLoginRequest *req = ngx_palloc(hc->pool, sizeof(*req));
338 if (req == NULL) { brix_hc_finish(hc, 0); return NGX_DONE; }
339 brix_upstream_build_login(req);
340 hc->wbuf = (u_char *) req;
341 hc->wbuf_len = sizeof(*req);
342 hc->wbuf_pos = 0;
343 hc->phase = XRD_HC_LOGIN;
344 hc->rhdr_pos = 0;
345 hc->resp_dlen = 0;
346 hc->resp_body = NULL;
347 hc->resp_body_pos = 0;
348 if (brix_hc_flush(hc) == NGX_ERROR) { brix_hc_finish(hc, 0); }
349 return NGX_DONE;
350 }
351 hc->phase = XRD_HC_LOGIN;
352 return NGX_OK;
353}

355/*
356 * Consume one complete response frame and advance the probe state machine.
357 * Several phases short-circuit to a verdict (alive/dead) without reaching the
358 * final PROBE phase — see the per-case comments. On a non-terminal transition
359 * it resets the accumulator and re-posts the read event so any already-buffered
360 * pipelined reply is processed in the same event cycle.
361 */
362
363static void
364brix_hc_dispatch(brix_hc_ctx_t *hc)
365{
366 switch (hc->phase) {
367
368 case XRD_HC_HANDSHAKE:
369 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return; }
370 hc->phase = XRD_HC_PROTOCOL;
371 break;
372
373 case XRD_HC_PROTOCOL:
374 if (brix_hc_dispatch_protocol(hc) == NGX_DONE) {
375 return;
376 }
377 break;
378
379 case XRD_HC_LOGIN:
380 /* authmore => server is alive but wants credentials; we don't carry
381 * any, so accept protocol liveness rather than fail. */
382 if (hc->resp_status == kXR_authmore) {
383 brix_hc_finish(hc, 1);
384 return;
385 }
386 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return; }
387 brix_hc_send_probe(hc); /* sets phase = PROBE, sends ping/stat */
388 return;
389
390 case XRD_HC_PROBE:
391 brix_hc_finish(hc, hc->resp_status == kXR_ok ? 1 : 0);
392 return;
393 }
394
395 /* Reset accumulator for the next frame and post a synthetic read so any
396 * pipelined bytes already in the socket buffer are processed this cycle. */
397 hc->rhdr_pos = 0;
398 hc->resp_dlen = 0;
399 hc->resp_body = NULL;
400 hc->resp_body_pos = 0;
401
402 if (ngx_handle_read_event(hc->conn->read, 0) != NGX_OK) {
403 brix_hc_finish(hc, 0);
404 return;
405 }
406 ngx_post_event(hc->conn->read, &ngx_posted_events);
407}

409/*
410 * Readable-event handler. Pulls exactly one complete frame per invocation and
411 * hands it to brix_hc_dispatch(), which re-arms (or posts) the read event for
412 * the next frame; this keeps each event-loop iteration bounded.
413 */
414void
415brix_hc_read_handler(ngx_event_t *rev)
416{
417 ngx_connection_t *c = rev->data;
418 brix_hc_ctx_t *hc = c->data;
419 ngx_int_t rc;
420
421 if (rev->timedout) {
422 brix_hc_finish(hc, 0);
423 return;
424 }
425
426 /* Loop is structural only: every branch returns. It exists so a future
427 * "read another frame inline" change has an obvious place to continue. */
428 for ( ;; ) {
429 rc = brix_hc_recv_frame(hc);
430 if (rc == NGX_AGAIN) {
431 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
432 brix_hc_finish(hc, 0);
433 }
434 return;
435 }
436 if (rc == NGX_ERROR) {
437 brix_hc_finish(hc, 0);
438 return;
439 }
440 brix_hc_dispatch(hc);
441 return; /* dispatch re-arms / posts as needed */
442 }
