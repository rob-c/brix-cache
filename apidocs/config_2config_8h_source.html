8/* ---- brix_path_kind_t — filesystem object kind for path validation ----
9 *
10 * WHAT: Enumerates the three kinds of filesystem objects that brix_validate_path checks:
11 * regular file, directory, or either. Used during postconfiguration to validate
12 * configured paths (brix_root, cache directories) against actual filesystem state.
13 *
14 * WHY: Postconfiguration validation catches misconfigured paths early — nginx -t fails
15 * with emerg errors before any traffic is accepted. Each kind maps to a specific
16 * stat check and errno translation so the error message accurately describes what
17 * was wrong (e.g., path exists but is a directory when a file was expected). */
18typedef enum {
19 BRIX_PATH_REGULAR_FILE, /* path must be an existing regular file */
20 BRIX_PATH_DIRECTORY, /* path must be an existing directory */
21 BRIX_PATH_FILE_OR_DIRECTORY /* path must exist; either kind is acceptable */
22} brix_path_kind_t;

56 *
57 * WHAT: Loads host certificate/key pair, reads CA directory for proxy cert verification,

64/* ---- brix_configure_tls — resolve certificate/key and prepare SSL_CTX ----
65 *
66 * WHAT: Loads brix_tls_certificate/brix_tls_certificate_key, creates and configures
67 * an SSL_CTX for TLS termination on root:// connections (kXR_wantTLS upgrade).
68 * Returns NGX_OK on success, NGX_ERROR on failure.
69 *
70 * WHY: In-protocol XRootD TLS upgrade requires a pre-built SSL_CTX that clients can

74/* ---- brix_configure_token_auth — load JWKS keys for bearer-token authentication ----
75 *
76 * WHAT: Loads JSON Web Key Set (JWKS) file from configured path, parses public keys for
77 * JWT signature verification. Returns NGX_OK on success, NGX_ERROR on failure.
78 *
79 * WHY: WLCG/SciToken bearer tokens are signed with RSA/ECDSA keys published in JWKS format.
80 * The server loads these keys once at startup to verify token signatures without
81 * network calls per request. Token scope enforcement is applied per-path after verification.
82 */

84/* ---- brix_token_jwks_schedule_refresh — start periodic JWKS refresh timer ----
85 *
86 * WHAT: Schedules a per-worker mtime-poll timer that watches the JWKS file for changes
87 * and reloads keys when modified. Called from init_process callback.
88 *
89 * WHY: JWKS keys rotate periodically (key rotation, new signing keys). The server must
90 * detect and load updated keys without restart — mtime polling avoids unnecessary
91 * HTTP fetches while catching file changes promptly. Each worker polls independently.
92 */

94/* ---- brix_configure_sss_auth — load SSS key file and validate format ----
95 *

98 *
99 * WHY: SSS provides HMAC-based authentication for internal cluster communication where
100 * GSI certs are unavailable. Key file must contain valid hex-encoded bytes with
