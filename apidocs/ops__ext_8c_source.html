1/*
2 * ops_ext.c — client wrappers for the nginx-xrootd vendor extension opcodes
3 * (kXR_setattr / kXR_symlink / kXR_readlink / kXR_link).
4 *
5 * WHAT: Path-based ops that close POSIX gaps the base XRootD protocol lacks, so a
6 * FUSE mount can honour `cp -p` / `touch -d` (setattr times), chown (setattr
7 * owner), and `ln`/`ln -s` (link/symlink). Plus brix_ext_probe(), which
8 * queries kXR_Qconfig "xrdfs.ext" so a caller only emits these opcodes
9 * against a server that advertises support.
10 * HOW: Each builds its 24-byte ClientXxxRequest (wire_vendor_ext.h) + payload and
11 * exchanges it via brix_roundtrip (redirect-aware). setattr's variable
12 * attribute block is encoded big-endian into the payload prefix.
13 *
14 * Clean-room: the shared wire structs + frame.c only. No XrdCl.
15 */

41int
42brix_setattr(brix_conn *c, const char *path, int set_times,
43 const struct timespec times[2], int set_owner,
44 uint32_t uid, uint32_t gid, brix_status *st)
45{
46 ClientSetattrRequest req;
47 uint8_t *payload;
48 size_t plen = strlen(path);
49 size_t total = BRIX_SETATTR_PREFIX_LEN + plen + 1;
50 uint8_t *p;
51 uint32_t flags = 0;
52 int rc;
53
54 if (set_times) { flags |= kXR_sa_times; }
55 if (set_owner) { flags |= kXR_sa_owner; }
56
57 payload = (uint8_t *) calloc(1, total);
58 if (payload == NULL) {
59 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
60 return -1;
61 }
62 p = payload;
63 {
64 /* Shared 44-byte attribute-prefix codec (libxrdproto) — the offset layout
65 * lives in one place so client-encode == server-decode. */
66 xrdp_setattr_t a;
67 a.flags = flags;
68 a.atime_sec = set_times ? (int64_t) times[0].tv_sec : 0;
69 a.atime_nsec = set_times ? (int64_t) times[0].tv_nsec : 0;
70 a.mtime_sec = set_times ? (int64_t) times[1].tv_sec : 0;
71 a.mtime_nsec = set_times ? (int64_t) times[1].tv_nsec : 0;
72 a.uid = set_owner ? (int32_t) uid : -1;
73 a.gid = set_owner ? (int32_t) gid : -1;
74 xrdp_setattr_prefix_pack(&a, p);
75 }
76 memcpy(p + BRIX_SETATTR_PREFIX_LEN, path, plen + 1);
77
78 memset(&req, 0, sizeof(req));
79 req.requestid = htons(kXR_setattr);
80 xrdw_empty_req_pack(((ClientRequestHdr *) &req)->body); /* 44B prefix is in the payload */
81
82 rc = ext_simple(c, &req, payload, (uint32_t) total, st);
83 free(payload);
84 return rc;
85}

88brix_symlink(brix_conn *c, const char *target, const char *linkpath,
89 brix_status *st)
90{
91 ClientSymlinkRequest req;
92 char *payload;
93 size_t tl = strlen(target), ll = strlen(linkpath);
94 size_t total = tl + 1 + ll;
95 int rc;
96
97 if (tl == 0 || tl > 0x7fff) {
98 brix_status_set(st, XRDC_EUSAGE, 0, "symlink target length out of range");
99 return -1;
100 }
101 payload = (char *) malloc(total);
102 if (payload == NULL) {
103 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
104 return -1;
105 }
106 memcpy(payload, target, tl);
107 payload[tl] = ' ';
108 memcpy(payload + tl + 1, linkpath, ll);
109
110 memset(&req, 0, sizeof(req));
111 req.requestid = htons(kXR_symlink);
112 {
113 xrdw_twopath_req_t b = { .arg1len = (int16_t) tl };
114 xrdw_twopath_req_pack(&b, ((ClientRequestHdr *) &req)->body);
115 }
116
117 rc = ext_simple(c, &req, payload, (uint32_t) total, st);
118 free(payload);
119 return rc;
120}

157ssize_t
158brix_readlink(brix_conn *c, const char *path, char *out, size_t outsz,
159 brix_status *st)
160{
161 ClientReadlinkRequest req;
162 uint16_t status;
163 uint8_t *body = NULL;
164 uint32_t blen = 0;
165 size_t n;
166
167 memset(&req, 0, sizeof(req));
168 req.requestid = htons(kXR_readlink);
169 xrdw_empty_req_pack(((ClientRequestHdr *) &req)->body);
170
171 if (brix_roundtrip(c, &req, path, (uint32_t) strlen(path),
172 &status, &body, &blen, st) != 0) {
173 return -1;
174 }
175 n = (blen < outsz - 1) ? blen : (outsz - 1);
176 memcpy(out, body, n);
177 out[n] = '\0';
178 free(body);
179 return (ssize_t) n;
180}

182int
183brix_ext_probe(brix_conn *c, int *has_setattr, int *has_symlink,
184 int *has_readlink, int *has_link, brix_status *st)
185{
186 char reply[256];
187
188 if (has_setattr) { *has_setattr = 0; }
189 if (has_symlink) { *has_symlink = 0; }
190 if (has_readlink) { *has_readlink = 0; }
191 if (has_link) { *has_link = 0; }
192
193 if (brix_query(c, kXR_Qconfig, "xrdfs.ext", reply, sizeof(reply), st) != 0) {
194 return -1;
195 }
196 /* reply is "xrdfs.ext=setattr,symlink,readlink,link" or "xrdfs.ext=0". */
197 if (has_setattr && strstr(reply, "setattr") != NULL) { *has_setattr = 1; }
198 if (has_symlink && strstr(reply, "symlink") != NULL) { *has_symlink = 1; }
199 if (has_readlink && strstr(reply, "readlink") != NULL) { *has_readlink = 1; }
200 if (has_link && strstr(reply, "link") != NULL) { *has_link = 1; }
201 return 0;
202}
