1/*
2 * xfer_mover_pump.c — the in-process byte mover for the transfer engine.
3 *
4 * WHAT: brix_xfer_pump_objects() — a positional read-through/write-through copy
5 * between two SD objects. This is the canonical VFS<->VFS (backend<->backend)
6 * move that the synchronous STAGE path and any same-process SD-to-SD commit
7 * use.
8 *
9 * WHY: The loop lived as a private `stage_move_objects` static inside
10 * compat/staged_file.c, reachable only by the staged-commit path. The
11 * transfer engine needs the same mover as a first-class, reusable strategy
12 * (BRIX_XFER_MOVE_PUMP), so it is lifted here verbatim — behaviour
13 * identical — and staged_file.c now calls it. One mover, one place.
14 *
15 * HOW: Read CHUNK bytes from the source driver, write them through the
16 * destination driver at the same offset, until EOF. EINTR is retried on
17 * both sides. No raw pread/pwrite here — the driver slots do the I/O inside
18 * the SD backend, preserving the zero-data-POSIX-outside-the-backend
19 * invariant. Early-return, no goto.
20 */

28brix_xfer_pump_objects(brix_sd_obj_t *src, brix_sd_obj_t *dst)
29{
30 static const size_t CHUNK = 256 * 1024;
31 char *buf;
32 off_t off = 0;
33
34 if (src == NULL || dst == NULL) {
35 errno = EINVAL;
36 return NGX_ERROR;
37 }
38
39 buf = malloc(CHUNK);
40 if (buf == NULL) {
41 errno = ENOMEM;
42 return NGX_ERROR;
43 }
44
45 for ( ;; ) {
46 ssize_t n = src->driver->pread(src, buf, CHUNK, off);
47 ssize_t w_done = 0;
48
49 if (n < 0) {
50 if (errno == EINTR) {
51 continue;
52 }
53 free(buf);
54 return NGX_ERROR;
55 }
56 if (n == 0) {
57 break;
58 }
59
60 while (w_done < n) {
61 ssize_t w = dst->driver->pwrite(dst, buf + w_done,
62 (size_t) (n - w_done), off + w_done);
63 if (w < 0) {
64 if (errno == EINTR) {
65 continue;
66 }
67 free(buf);
68 return NGX_ERROR;
69 }
70 w_done += w;
71 }
72 off += n;
73 }
74
75 free(buf);
76 return NGX_OK;
77}
