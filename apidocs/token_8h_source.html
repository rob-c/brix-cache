30/*

34 * 256 bytes is sufficient for all practical deployments and keeps the

123/*
124 * brix_token_validate_args_t — caller-supplied state for brix_token_validate().
125 *
126 * WHAT: Bundles the validator's inputs (log sink, raw token bytes, trusted
127 * JWKS keys, expected issuer/audience pins, macaroon secret, clock-skew
128 * window) and its single output (claims) into one struct.
129 * WHY: brix_token_validate() took 11 positional parameters; a named-field
130 * carrier keeps the extern surface at one parameter, makes every
131 * callsite self-documenting, and lets the internal RFC pipeline thread
132 * the same read-only pointer through its static helpers.
133 * HOW: Populated field-by-field at each callsite (zero-init unused optional
134 * fields) and passed read-only; the validator writes only through
135 * ->claims, the caller's output buffer.
136 */
137typedef struct {
138 ngx_log_t *log; /* error/audit log sink */
139 const char *token; /* raw bearer token bytes */
140 size_t token_len; /* length of token */
141 const brix_jwks_key_t *keys; /* trusted JWKS keys (or NULL) */
142 int key_count; /* entries in keys[] */
143 const char *expected_issuer; /* iss pin; NULL/"" = any */
144 const char *expected_audience; /* aud pin; NULL/"" = any */
145 const u_char *macaroon_secret; /* macaroon HMAC key or NULL */
146 size_t secret_len; /* length of macaroon_secret */
147 int clock_skew; /* exp/nbf tolerance (seconds) */
148 brix_token_claims_t *claims; /* OUT: verified claims */
149} brix_token_validate_args_t;
