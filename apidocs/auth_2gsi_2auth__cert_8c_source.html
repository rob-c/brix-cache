9/*
10 * gsi_promote_fullproxy — validate an OPTIONAL client-pushed full proxy
11 * (kXRS_x509_fullproxy, phase-70 §5.1) captured during kXGC_cert and, if it
12 * passes, promote its bytes to ctx->deleg_proxy_pem for backend PASSTHROUGH.
13 *
14 * WHY: A node can only replay x509 upstream if the user voluntarily hands over
15 * a full proxy (chain + private key). This is the wire-receive + identity
16 * gate for that opt-in: it MUST NOT let a session present a proxy for a
17 * different identity, and MUST NOT accept a key over cleartext.
18 *
19 * HOW: Runs only after the front door proved the GSI identity (ctx->login.dn
20 * is set). Rejects the login when the captured proxy fails any check;
21 * silently no-ops when nothing was captured (the common case).
22 * Checks: (1) transport is TLS (roots://) — a private key never crosses
23 * cleartext; (2) the PEM parses to a cert chain AND a private key;
24 * (3) the supplied leaf/EEC subject is a parent of the authenticated
25 * proxy DN (ctx->login.dn begins with it) — no identity swap. On success
26 * the raw bytes move to ctx->deleg_proxy_pem (borrowed by the binder).
27 * Returns NGX_OK to proceed, NGX_ERROR to reject the login.
28 */
29static ngx_int_t
30gsi_promote_fullproxy(brix_ctx_t *ctx, ngx_connection_t *c)
31{
32 BIO *bio;
33 X509 *leaf;
34 EVP_PKEY *key = NULL;
35 char *subj;
36 int dn_ok;
37
38 if (ctx->gsi.client_fullproxy_pem == NULL) {
39 return NGX_OK; /* nothing supplied — normal */
40 }
41
42 /* (1) TLS-only: a full proxy carries a private key; never over cleartext. */
43 if (c->ssl == NULL) {
44 ngx_log_error(NGX_LOG_WARN, c->log, 0,
45 "brix: GSI full-proxy passthrough rejected: "
46 "supplied over cleartext (roots:// required)");
47 return NGX_ERROR;
48 }
49
50 /* (2) Parse the chain leaf and the private key from the same PEM. */
51 bio = BIO_new_mem_buf(ctx->gsi.client_fullproxy_pem,
52 (int) ctx->gsi.client_fullproxy_len);
53 if (bio == NULL) {
54 return NGX_ERROR;
55 }
56 leaf = PEM_read_bio_X509(bio, NULL, NULL, NULL);
57 if (leaf != NULL) {
58 (void) BIO_reset(bio);
59 key = PEM_read_bio_PrivateKey(bio, NULL, NULL, NULL);
60 }
61 BIO_free(bio);
62 if (leaf == NULL || key == NULL) {
63 X509_free(leaf);
64 EVP_PKEY_free(key);
65 ngx_log_error(NGX_LOG_WARN, c->log, 0,
66 "brix: GSI full-proxy passthrough rejected: "
67 "PEM missing cert chain or private key");
68 return NGX_ERROR;
69 }
70 EVP_PKEY_free(key); /* only presence is checked here */
71
72 /* (3) Identity gate: the authenticated proxy DN must sit beneath the
73 * supplied leaf/EEC subject — the user may only pass through a proxy for
74 * their own identity, never elevate to another. */
75 subj = X509_NAME_oneline(X509_get_subject_name(leaf), NULL, 0);
76 dn_ok = (subj != NULL
77 && ngx_strncmp(ctx->login.dn, subj, ngx_strlen(subj)) == 0);
78 OPENSSL_free(subj);
79 X509_free(leaf);
80
81 if (!dn_ok) {
82 ngx_log_error(NGX_LOG_WARN, c->log, 0,
83 "brix: GSI full-proxy passthrough rejected: "
84 "supplied proxy identity does not match authenticated DN");
85 return NGX_ERROR;
86 }
87
88 /* Passed: hand the bytes to the session ctx for the root:// VFS binder.
89 * Copy into c->pool so the lifetime matches the rest of the session ctx,
90 * then release the heap capture. Never logged. */
91 ctx->deleg_proxy_pem.data = ngx_pnalloc(c->pool,
92 ctx->gsi.client_fullproxy_len);
93 if (ctx->deleg_proxy_pem.data == NULL) {
94 return NGX_ERROR;
95 }
96 ngx_memcpy(ctx->deleg_proxy_pem.data, ctx->gsi.client_fullproxy_pem,
97 ctx->gsi.client_fullproxy_len);
98 ctx->deleg_proxy_pem.len = ctx->gsi.client_fullproxy_len;
99
100 OPENSSL_cleanse(ctx->gsi.client_fullproxy_pem, ctx->gsi.client_fullproxy_len);
101 free(ctx->gsi.client_fullproxy_pem);
102 ctx->gsi.client_fullproxy_pem = NULL;
103 ctx->gsi.client_fullproxy_len = 0;
104
105 ngx_log_error(NGX_LOG_INFO, c->log, 0,
106 "brix: GSI full-proxy passthrough accepted for backend "
107 "delegation (identity verified)");
108 return NGX_OK;
109}

111/*
112 * gsi_cert_deny — shared deny tail for the kXGC_cert step.
113 *
114 * WHY: Every post-parse failure in the cert step must emit the same
115 * access-log line shape, bump the AUTH error metric, free the parsed
116 * chain, and answer kXR_NotAuthorized — one helper keeps the deny
117 * paths byte-identical.
118 *
119 * HOW: `log_msg` goes to the access log, `err_msg` to the wire error (they
120 * differ for the OCSP case). Frees `chain`; returns the send rc.
121 */
122static ngx_int_t
123gsi_cert_deny(brix_ctx_t *ctx, ngx_connection_t *c, STACK_OF(X509) *chain,
124 const char *log_msg, const char *err_msg)
125{
126 brix_log_access(ctx, c, "AUTH", "-", "gsi",
127 0, kXR_NotAuthorized, log_msg, 0);
128 BRIX_OP_ERR(ctx, BRIX_OP_AUTH);
129 sk_X509_pop_free(chain, X509_free);
130 return brix_send_error(ctx, c, kXR_NotAuthorized, err_msg);
131}

133/*
134 * gsi_cert_verify — verify the parsed client chain against the CA store.
135 *
136 * WHY: The chain-verify call needs the non-leaf certs split out as the
137 * untrusted stack; isolating the dup/verify/free sequence keeps the
138 * cert step linear.
139 *
140 * HOW: Builds the untrusted stack (chain minus leaf) when present, runs
141 * brix_gsi_verify_chain (RFC 3820 proxies accepted, DN/VOMS material
142 * returned in *verify_res), frees the dup, and returns the verify rc
143 * (brix_gsi_verify_chain already logged any specific error).
144 */
145static ngx_int_t
146gsi_cert_verify(ngx_connection_t *c, ngx_stream_brix_srv_conf_t *conf,
147 STACK_OF(X509) *chain, X509 *leaf,
148 brix_gsi_verify_result_t *verify_res)
149{
150 STACK_OF(X509) *untrusted = NULL;
151 ngx_int_t gsi_rc;
152
153 if (sk_X509_num(chain) > 1) {
154 untrusted = sk_X509_dup(chain);
155 sk_X509_delete(untrusted, 0);
156 }
157
158 gsi_rc = brix_gsi_verify_chain(c->log, conf->gsi_store,
159 leaf, untrusted, 0, verify_res,
160 0 /* GSI: accept RFC 3820 proxies */);
161
162 if (untrusted) {
163 sk_X509_free(untrusted);
164 }
165
166 return gsi_rc;
167}

169/*
170 * gsi_cert_capture_dn — copy the verified DN into the session identity.
171 *
172 * WHY: The DN is the session's authorization identity; an over-long DN is
173 * truncated (buffer-bounded) but operators must be warned because VO
174 * ACL rules may then fail to match.
175 *
176 * HOW: Warns when the verified DN exceeds ctx->login.dn, then bounded-copies
177 * it with ngx_cpystrn.
178 */
179static void
180gsi_cert_capture_dn(brix_ctx_t *ctx, ngx_connection_t *c,
181 const brix_gsi_verify_result_t *verify_res)
182{
183 if (strlen(verify_res->dn_buf) >= sizeof(ctx->login.dn)) {
184 ngx_log_error(NGX_LOG_WARN, c->log, 0,
185 "brix: GSI DN too long (%uz bytes), truncating to %uz; "
186 "VO ACL rules may not match correctly",
187 strlen(verify_res->dn_buf), sizeof(ctx->login.dn) - 1);
188 }
189 ngx_cpystrn((u_char *) ctx->login.dn,
190 (u_char *) verify_res->dn_buf,
191 sizeof(ctx->login.dn));
192}

232/*
233 * gsi_auth_step_cert — GSI step 2 (kXGC_cert): the client's proxy chain.
234 *
235 * WHY: Round 2 of the DH exchange carries the client's encrypted proxy
236 * chain — the security-load-bearing heart of GSI auth: parse, chain
237 * verify (+ optional OCSP), DN capture, optional full-proxy
238 * passthrough, VOMS extraction, then either begin §F6 delegation or
239 * complete the auth.
240 *
241 * HOW: Linear pipeline with early-return denies (gsi_cert_deny frees the
242 * chain on every failure). The ephemeral DH key is freed as soon as
243 * parse_x509 has consumed it. When tpc_delegate is on and a session
244 * cipher was persisted, begin_delegation sends kXGS_pxyreq
245 * (kXR_authmore) and auth completes when kXGC_sigpxy arrives;
246 * otherwise brix_gsi_complete_auth finalizes immediately.
247 */
248ngx_int_t
249gsi_auth_step_cert(brix_ctx_t *ctx, ngx_connection_t *c,
250 ngx_stream_brix_srv_conf_t *conf)
251{
252 STACK_OF(X509) *chain;
253 X509 *leaf;
254 brix_gsi_verify_result_t verify_res;
255
256 chain = brix_gsi_parse_x509(ctx, c);
257
258 if (ctx->gsi.dh_key) {
259 EVP_PKEY_free(ctx->gsi.dh_key);
260 ctx->gsi.dh_key = NULL;
261 }
262
263 if (chain == NULL) {
264 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "gsi",
265 kXR_NotAuthorized, "cannot parse GSI credential");
266 }
267
268 leaf = sk_X509_value(chain, 0);
269
270 if (gsi_cert_verify(c, conf, chain, leaf, &verify_res) != NGX_OK) {
271 /* brix_gsi_verify_chain already logged the specific error */
272 return gsi_cert_deny(ctx, c, chain,
273 "certificate verification failed",
274 "certificate verification failed");
275 }
276
277 /*
278 * OCSP revocation check (Feature 8e).
279 * The certificate chain is still valid at this point: leaf is chain[0],
280 * its issuer is chain[1] (may be NULL for single-cert chains).
281 */
282 if (conf->ocsp.enable) {
283 X509 *issuer = (sk_X509_num(chain) > 1)
284 ? sk_X509_value(chain, 1) : NULL;
285 if (brix_ocsp_check_cert(c->log, leaf, issuer,
286 (int)conf->ocsp.soft_fail,
287 (int)conf->ocsp.require_nonce) != 0)
288 {
289 return gsi_cert_deny(ctx, c, chain,
290 "OCSP check failed",
291 "OCSP certificate check failed");
292 }
293 }
294
295 gsi_cert_capture_dn(ctx, c, &verify_res);
296
297 /* phase-70 §5.1: validate + promote an OPTIONAL client-pushed full proxy
298 * now that the identity DN is known. A supplied-but-invalid proxy fails the
299 * login (never silently downgrades to service cred). */
300 if (gsi_promote_fullproxy(ctx, c) != NGX_OK) {
301 return gsi_cert_deny(ctx, c, chain,
302 "full-proxy passthrough rejected",
303 "full-proxy passthrough rejected");
304 }
305
306 gsi_cert_extract_voms(ctx, c, conf, leaf, chain);
307
308 /*
309 * §F6 X.509 proxy delegation: when enabled, run an extra handshake round to
310 * capture the client's delegated proxy BEFORE completing auth. The DN/VOMS are
311 * already extracted (above) and the chain/leaf are still valid here; the GSI
312 * session cipher was persisted by parse_x509. begin_delegation sends
313 * kXGS_pxyreq (kXR_authmore) and auth completes when kXGC_sigpxy arrives.
314 */
315 if (conf->tpc_delegate && !ctx->gsi.deleg_await && ctx->gsi.sess_keylen > 0) {
316 ngx_int_t drc = brix_gsi_begin_delegation(ctx, c, conf, leaf, chain);
317
318 sk_X509_pop_free(chain, X509_free);
319 if (drc != NGX_OK) {
320 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "gsi",
321 kXR_NotAuthorized, "GSI proxy delegation failed");
322 }
323 return NGX_OK; /* kXGS_pxyreq sent; auth completes on kXGC_sigpxy */
324 }
325
326 sk_X509_pop_free(chain, X509_free);
327 return brix_gsi_complete_auth(ctx, c, conf);
328}
