1/* client/lib/cred_bearer.c
2 *
3 * WHAT: Bearer-token credential handler for the unified credential store (cred.c).
4 * Implements brix_cred_bearer() returning the XRDC_CRED_BEARER handler with
5 * available / acquire / refresh operations.
6 * WHY: Bearer token discovery was only available through the sec_token.c auth
7 * module path; the unified store needs a standalone handler so the same token
8 * is shared between the root:// auth handshake, HTTP Authorization headers,
9 * and auto-refresh — all from a single source of truth.
10 * HOW: Discovery precedence:
11 * cfg->bearer_literal (verbatim — highest precedence)
12 * > cfg->bearer_path (read that file, symlink-safe via brix_open_credfile)
13 * > brix_token_discover() (env/default: $BEARER_TOKEN → $BEARER_TOKEN_FILE
14 * → $XDG_RUNTIME_DIR/bt_u<uid> → /tmp/bt_u<uid>)
15 * Lifetime contract (see cred.h acquire pointer doc): tokens can be arbitrarily
16 * large, so a process-lifetime `static char *g_tok` is kept; on each acquire:
17 * free(g_tok); g_tok = <new malloc'd token>; out->token = g_tok;
18 * The store calls strdup on out->token immediately after acquire() returns, so
19 * g_tok stays valid for that window. No stack pointers; no fixed static buffer.
20 * refresh() is a documented no-op for B4 — oidc-token refresh is deferred to
21 * task C2 where brix_cred_autorefresh is adapted to the (cfg, st) contract.
22 *
23 * ngx-free. No goto. Functional/modular: one responsibility per function.
24 */

38/* process-lifetime token buffer */
39/*
40 * g_tok — owns the most recently acquired token string.
41 *
42 * WHAT: single malloc'd buffer holding the current token; replaced on every
43 * acquire() call by free+assign.
44 * WHY: tokens are variable-length (JWTs can be several KB); a fixed static char[]
45 * would truncate or waste memory. The free/replace invariant keeps exactly one
46 * live allocation and satisfies the lifetime contract: g_tok is valid until the
47 * next acquire() call overwrites it, which is after the store has already
48 * strdup'd the value.
49 * HOW: module-scoped static pointer; set to NULL at process start (C default).
50 * Only bearer_acquire() touches it.
51 */
52static char *g_tok = NULL;

54/* file reading helper */
55/*
56 * slurp_credfile — read a credential file safely into a malloc'd string.
57 *
58 * WHAT: open `path` via brix_open_credfile (O_NOFOLLOW, owned-by-euid, non-secret)
59 * and read up to 1 MiB into a malloc'd NUL-terminated string, stripping
60 * trailing whitespace. Returns the string or NULL on any error.
61 * WHY: cfg->bearer_path points at a user-managed file; symlink safety and owner
62 * check match the behaviour of sec_token.c:slurp() for the file-path case.
63 * HOW: brix_open_credfile returns an fd (caller closes); fdopen + fread;
64 * bail if the file exceeds 1 MiB (not a JWT).
65 */
66static char *
67slurp_credfile(const char *path)
68{
69 int fd;
70 FILE *fp;
71 long sz;
72 size_t got;
73 char *buf;
74 size_t i;
75
76 fd = brix_open_credfile(path, 0, NULL);
77 if (fd < 0) {
78 return NULL;
79 }
80 fp = fdopen(fd, "rb");
81 if (fp == NULL) {
82 close(fd);
83 return NULL;
84 }
85 if (fseek(fp, 0, SEEK_END) != 0 || (sz = ftell(fp)) < 0) {
86 fclose(fp);
87 return NULL;
88 }
89 rewind(fp);
90 if (sz > (1 << 20)) { /* a JWT is never a megabyte */
91 fclose(fp);
92 return NULL;
93 }
94 buf = (char *)malloc((size_t)sz + 1);
95 if (buf == NULL) {
96 fclose(fp);
97 return NULL;
98 }
99 got = fread(buf, 1, (size_t)sz, fp);
100 fclose(fp);
101 buf[got] = '\0';
102
103 /* strip trailing whitespace in-place */
104 i = got;
105 while (i > 0 &&
106 (buf[i - 1] == '\n' || buf[i - 1] == '\r' ||
107 buf[i - 1] == ' ' || buf[i - 1] == '\t')) {
108 buf[--i] = '\0';
109 }
110 if (buf[0] == '\0') {
111 free(buf);
112 return NULL;
113 }
114 return buf;
115}

117/* token acquisition */
118/*
119 * obtain_token — apply discovery precedence and return a malloc'd token string.
120 *
121 * WHAT: cfg->bearer_literal > cfg->bearer_path > brix_token_discover().
122 * WHY: explicit CLI/programmatic overrides must win over environment variables so
123 * callers can specify a token for a particular operation.
124 * HOW: three early-return branches; each branch produces a malloc'd string (or NULL)
125 * so the caller owns the result uniformly.
126 */
127static char *
128obtain_token(const brix_cred_config *cfg)
129{
130 if (cfg != NULL && cfg->bearer_literal != NULL && cfg->bearer_literal[0] != '\0') {
131 return strdup(cfg->bearer_literal);
132 }
133 if (cfg != NULL && cfg->bearer_path != NULL && cfg->bearer_path[0] != '\0') {
134 return slurp_credfile(cfg->bearer_path);
135 }
136 return brix_token_discover();
137}

140/*
141 * bearer_available — 1 if a token is obtainable via the discovery precedence.
142 *
143 * WHAT: fast probe mirroring sec_token.c:token_have(); does NOT cache the result.
144 * WHY: used by auth pre-flight diagnostics and brix_cred_available(); must agree
145 * with what acquire() would do.
146 * HOW: obtain_token then free; 1 if non-NULL.
147 */
148static int
149bearer_available(const brix_cred_config *cfg)
150{
151 char *t = obtain_token(cfg);
152 if (t == NULL) {
153 return 0;
154 }
155 free(t);
156 return 1;
157}

159/*
160 * bearer_acquire — obtain the token, fill *out and *not_after.
161 *
162 * WHAT: the store's acquire call for XRDC_CRED_BEARER. Replaces g_tok with the
163 * new token so the pointer survives until the store strdup's it.
164 * WHY: g_tok is the owner of the most-recently-acquired token (free/replace
165 * invariant); out->token = g_tok satisfies the lifetime contract.
166 * HOW: 1) obtain_token() for discovery+precedence;
167 * 2) free(g_tok); g_tok = new token;
168 * 3) fill *out;
169 * 4) brix_token_meta_get for not_after (0 if not a JWT or exp absent);
170 * 5) return 0 on success, -1 + XRDC_EAUTH on no token.
171 */
172static int
173bearer_acquire(const brix_cred_config *cfg, brix_cred_view *out,
174 int64_t *not_after, brix_status *st)
175{
176 char *tok;
177 brix_token_meta meta;
178
179 tok = obtain_token(cfg);
180 if (tok == NULL) {
181 brix_status_set(st, XRDC_EAUTH, 0,
182 "no bearer token (set BEARER_TOKEN, BEARER_TOKEN_FILE, "
183 "or cfg->bearer_literal)");
184 return -1;
185 }
186
187 /* Replace the process-lifetime buffer; old pointer is freed first. */
188 free(g_tok);
189 g_tok = tok;
190
191 out->kind = XRDC_CRED_BEARER;
192 out->path = NULL;
193 out->token = g_tok; /* valid until next acquire(); store strdup's immediately */
194 out->s3_access = NULL;
195 out->s3_secret = NULL;
196 out->not_after = 0;
197
198 /* Best-effort expiry: parse the JWT exp claim; 0 is acceptable for non-JWTs. */
199 brix_token_meta_get(g_tok, &meta);
200 *not_after = (int64_t)meta.exp; /* 0 if absent or unparseable; still succeed */
201
202 return 0;
203}
