1/*
2 * sec_pwd.c — XRootD `pwd` (XrdSecpwd) auth module, client side — Phase 52 WS-B.
3 *
4 * WHAT: Drives the 2-round password exchange against our server dialect:
5 * round 1 client → kXRS_puk(client DH pub) + kXRS_user + kXRS_version;
6 * server → kXR_authmore: kXRS_puk(server DH pub) + credsreq.
7 * round 2 client → kXRS_main = DH-session-encrypted { kXRS_creds(password) };
8 * server → kXR_ok.
9 * WHY: Completes encryption-protocol parity (a client that can speak `pwd`).
10 * The password is sourced non-interactively (XRDC_PWD, or the stock
11 * XrdSecCREDS hex blob) and only DH-session-encrypted on the wire, so `pwd`
12 * is for trusted/closed networks under TLS only — preferred LAST.
13 * HOW: Reuses the shared gsi_core DH + session-cipher + bucket kernels (the same
14 * primitives the GSI module uses). Per-connection round-1 state (our DH key
15 * + the password) is kept in file-statics; the CLI runs one connection at a
16 * time (mirrors sec_gsi.c).
17 *
18 * Wire reference: docs/refactor/phase-52-pwd-wire-spec.md.
19 */

47static int
48pwd_have(brix_conn *c)
49{
50 /*
51 * WHAT: check whether a password credential is available.
52 * WHY: the auth dispatch calls this before attempting the pwd handshake;
53 * using brix_env_resolve keeps chain precedence consistent with the
54 * loader below.
55 * HOW: resolve the chain; any non-empty value counts as "have password".
56 */
57 const char *v;
58 (void) c;
59 v = brix_env_resolve(pwd_chain, NULL);
60 return v != NULL && v[0] != '\0';
61}

63/* Hex-decode in place into out (cap bytes); returns bytes or -1. */
64static int
65pwd_unhex(const char *hex, uint8_t *out, size_t cap)
66{
67 size_t n = strlen(hex), i;
68
69 if ((n % 2) != 0 || n / 2 > cap) {
70 return -1;
71 }
72 for (i = 0; i < n; i += 2) {
73 unsigned v;
74 if (sscanf(hex + i, "%2x", &v) != 1) {
75 return -1;
76 }
77 out[i / 2] = (uint8_t) v;
78 }
79 return (int) (n / 2);
80}

82/*
83 * Resolve the plaintext password into g_pwd_pass. Priority: XRDC_PWD
84 * (literal, highest precedence), then XrdSecCREDS hex blob
85 * ("&pwd""\0"<4-byte pfx><password>). Uses brix_env_resolve so legacy
86 * environments (XrdSecCREDS only) behave identically and a TTY note fires
87 * when both are set to different values.
88 * Returns 0 on success, -1 if none usable.
89 */
90static int
91pwd_load_password(void)
92{
93 const char *which = NULL;
94 const char *p = brix_env_resolve(pwd_chain, &which);
95 const char *creds;
96
97 /*
98 * WHAT: dispatch by which name won so literal vs hex-blob handling is
99 * correct regardless of which env var the user set.
100 * WHY: XRDC_PWD stores the plaintext literal; XrdSecCREDS stores a hex-
101 * encoded blob with a fixed header — they need different decoders.
102 * HOW: check which name matched; if canonical (XRDC_PWD) treat as literal;
103 * otherwise treat as XrdSecCREDS blob.
104 */
105 if (p != NULL && p[0] != '\0' && which != NULL
106 && strcmp(which, "XRDC_PWD") == 0) {
107 size_t l = strlen(p);
108 if (l > sizeof(g_pwd_pass)) {
109 l = sizeof(g_pwd_pass);
110 }
111 memcpy(g_pwd_pass, p, l);
112 g_pwd_pass_len = l;
113 return 0;
114 }
115
116 /* Reaching here means XRDC_PWD was not set (or empty); use the blob path
117 * with whatever value the resolver returned (XrdSecCREDS or nothing). */
118 creds = (p != NULL && p[0] != '\0') ? p : NULL;
119 if (creds != NULL) {
120 uint8_t raw[512];
121 int n = pwd_unhex(creds, raw, sizeof(raw));
122 uint8_t *pwd;
123 if (n > 9) {
124 /* layout: "&pwd""\0"<4-byte pfx><password...> */
125 pwd = (uint8_t *) memmem(raw, (size_t) n, "&pwd", 4);
126 if (pwd != NULL) {
127 size_t off = (size_t) (pwd - raw) + 5 + 4; /* &pwd\0 + pfx */
128 if (off < (size_t) n) {
129 size_t l = (size_t) n - off;
130 if (l > sizeof(g_pwd_pass)) {
131 l = sizeof(g_pwd_pass);
132 }
133 memcpy(g_pwd_pass, raw + off, l);
134 g_pwd_pass_len = l;
135 OPENSSL_cleanse(raw, sizeof(raw));
136 return 0;
137 }
138 }
139 }
140 OPENSSL_cleanse(raw, sizeof(raw));
141 }
142 return -1;
143}

145/* pwdStatus_t {ctype,action,options} → the htonl'd 4-byte word (see spec). */
146static uint32_t
147pwd_status_word(uint8_t ctype, uint16_t options)
148{
149 uint8_t img[4];
150 img[0] = ctype;
151 img[1] = 0;
152 img[2] = (uint8_t) (options & 0xff);
153 img[3] = (uint8_t) (options >> 8);
154 return htonl(*(uint32_t *) img);
155}

157static int
158pwd_first(brix_conn *c, const char *parms, uint8_t **payload, uint32_t *plen,
159 brix_status *st)
160{
161 brix_gbuf g;
162 char *pub;
163 size_t pub_len = 0;
164 const char *user;
165 uint32_t version, status;
166
167 (void) c;
168 (void) parms;
169
170 if (pwd_load_password() != 0) {
171 brix_status_set(st, XRDC_EAUTH, 0,
172 "pwd: no password (set XRDC_PWD or XrdSecCREDS)");
173 return -1;
174 }
175 user = getenv("XRDC_PWD_USER");
176 if (user == NULL || user[0] == '\0') {
177 user = getenv("USER");
178 }
179 if (user == NULL || user[0] == '\0') {
180 brix_status_set(st, XRDC_EAUTH, 0, "pwd: no username (set XRDC_PWD_USER)");
181 return -1;
182 }
183
184 if (g_pwd_key != NULL) {
185 EVP_PKEY_free(g_pwd_key);
186 g_pwd_key = NULL;
187 }
188 g_pwd_key = brix_gsi_cipher_keygen();
189 if (g_pwd_key == NULL) {
190 brix_status_set(st, XRDC_EAUTH, 0, "pwd: DH keygen failed");
191 return -1;
192 }
193 pub = brix_gsi_cipher_public(g_pwd_key, &pub_len);
194 if (pub == NULL) {
195 brix_status_set(st, XRDC_EAUTH, 0, "pwd: cannot encode key");
196 return -1;
197 }
198
199 version = htonl(PWD_VERSION);
200 status = pwd_status_word(0 /*kpCT_normal*/, kOptsClntTty);
201
202 brix_gbuf_init(&g);
203 brix_gbuf_raw(&g, "pwd", 4); /* protocol name + NUL */
204 brix_gbuf_u32(&g, kXPC_normal); /* step */
205 brix_gbuf_bucket(&g, 3000 /*kXRS_cryptomod*/, "ssl", 3);
206 brix_gbuf_bucket(&g, 3004 /*kXRS_puk*/, pub, pub_len);
207 brix_gbuf_bucket(&g, 3014 /*kXRS_version*/, &version, sizeof(version));
208 brix_gbuf_bucket(&g, 3008 /*kXRS_user*/, user, strlen(user));
209 brix_gbuf_bucket(&g, 3015 /*kXRS_status*/, &status, sizeof(status));
210 brix_gbuf_end(&g);
211 free(pub);
212
213 if (g.err) {
214 brix_gbuf_free(&g);
215 brix_status_set(st, XRDC_EAUTH, 0, "pwd: out of memory");
216 return -1;
217 }
218 *payload = g.p; /* ownership → caller (g.p is malloc'd) */
219 *plen = (uint32_t) g.len;
220 return 0;
221}

223static int
224pwd_more(brix_conn *c, const uint8_t *sbody, uint32_t slen, uint8_t **payload,
225 uint32_t *plen, brix_status *st)
226{
227 const uint8_t *spuk;
228 size_t spuk_len = 0, enc_len = 0;
229 EVP_PKEY *peer;
230 uint8_t key[PWD_KEYLEN];
231 brix_gsi_cipher_t cipher;
232 brix_gbuf inner, outer;
233 uint8_t *enc;
234 int ok;
235
236 (void) c;
237
238 if (g_pwd_key == NULL
239 || brix_gsi_find_bucket(sbody, slen, 3004 /*kXRS_puk*/,
240 &spuk, &spuk_len) != 0)
241 {
242 brix_status_set(st, XRDC_EAUTH, 0, "pwd: server key missing");
243 return -1;
244 }
245 peer = brix_gsi_cipher_parse_peer(spuk, spuk_len);
246 if (peer == NULL) {
247 brix_status_set(st, XRDC_EAUTH, 0, "pwd: bad server key");
248 return -1;
249 }
250 ok = brix_gsi_cipher_session_key(g_pwd_key, peer, 0, key, PWD_KEYLEN)
251 && brix_gsi_cipher_lookup("aes-128-cbc", &cipher);
252 EVP_PKEY_free(peer);
253 EVP_PKEY_free(g_pwd_key);
254 g_pwd_key = NULL;
255 if (!ok) {
256 OPENSSL_cleanse(key, sizeof(key));
257 brix_status_set(st, XRDC_EAUTH, 0, "pwd: key agreement failed");
258 return -1;
259 }
260
261 /* inner main buffer: "pwd\0" + step + kXRS_creds(password) + kXRS_none */
262 brix_gbuf_init(&inner);
263 brix_gbuf_raw(&inner, "pwd", 4);
264 brix_gbuf_u32(&inner, kXPC_creds);
265 brix_gbuf_bucket(&inner, 3010 /*kXRS_creds*/, g_pwd_pass, g_pwd_pass_len);
266 brix_gbuf_end(&inner);
267 OPENSSL_cleanse(g_pwd_pass, sizeof(g_pwd_pass));
268 g_pwd_pass_len = 0;
269 if (inner.err) {
270 brix_gbuf_free(&inner);
271 OPENSSL_cleanse(key, sizeof(key));
272 brix_status_set(st, XRDC_EAUTH, 0, "pwd: out of memory");
273 return -1;
274 }
275
276 enc = brix_gsi_cipher_encrypt(&cipher, key, inner.p, inner.len, 0, &enc_len);
277 OPENSSL_cleanse(key, sizeof(key));
278 OPENSSL_cleanse(inner.p, inner.len);
279 brix_gbuf_free(&inner);
280 if (enc == NULL) {
281 brix_status_set(st, XRDC_EAUTH, 0, "pwd: credential encrypt failed");
282 return -1;
283 }
284
285 /* outer: "pwd\0" + step + kXRS_main(enc) + kXRS_none */
286 brix_gbuf_init(&outer);
287 brix_gbuf_raw(&outer, "pwd", 4);
288 brix_gbuf_u32(&outer, kXPC_creds);
289 brix_gbuf_bucket(&outer, 3001 /*kXRS_main*/, enc, enc_len);
290 brix_gbuf_end(&outer);
291 free(enc);
292 if (outer.err) {
293 brix_gbuf_free(&outer);
294 brix_status_set(st, XRDC_EAUTH, 0, "pwd: out of memory");
295 return -1;
296 }
297
298 *payload = outer.p;
299 *plen = (uint32_t) outer.len;
300 return 0;
301}
