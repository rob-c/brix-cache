1/*
2 * http_mirror.c — Phase 24 HTTP/WebDAV traffic mirror (see http_mirror.h).
3 *
4 * WHAT: Replays qualifying read requests to one or more shadow HTTP(S) backends
5 * fire-and-forget, compares the shadow status to the primary, and counts
6 * divergence — without ever delaying or exposing the shadow path to the client.
7 *
8 * HOW: A PRECONTENT-phase handler on the MAIN request fires one background
9 * subrequest per shadow target (re-using the same location so the mirror config
10 * is in scope). Each subrequest is tagged is_mirror in its WebDAV request ctx;
11 * the WebDAV access/content handlers skip those, and a CONTENT-phase handler
12 * registered ahead of the WebDAV handler takes the subrequest over and proxies
13 * it to the shadow with credentials stripped (unless an explicit mirror token is
14 * configured). The shadow response is parsed with the same upstream-callback
15 * pattern as src/webdav/proxy_response.c but writes into a scratch status field
16 * in the request ctx; finalize compares status classes and updates metrics. A
17 * LOG-phase handler stamps the primary's final status for that comparison.
18 *
19 * Split (phase-79 file-size cap) into three focused files:
20 * http_mirror.c — this file: method classifiers + phase-handler /
21 * subrequest orchestration (precontent/log handlers)
22 * http_mirror_request.c — shadow HTTP request/response build + the nginx
23 * upstream callbacks it wires
24 * http_mirror_config.c — merge-time upstream setup + directive setters
25 * Cross-file symbols (the body-method predicate and the five upstream callbacks)
26 * plus the shared MIR_HTTP_INC macro live in http_mirror_internal.h.
27 */
28#include "http_mirror.h"
29#include "http_mirror_internal.h"
30#include "core/http/http_body.h"
31#include "core/http/http_headers.h"

34/* method filter */
35static ngx_uint_t
36brix_http_mirror_method_bit(ngx_http_request_t *r)
37{
38 switch (r->method) {
39 case NGX_HTTP_GET: return BRIX_MIRROR_M_GET;
40 case NGX_HTTP_HEAD: return BRIX_MIRROR_M_HEAD;
41 case NGX_HTTP_PROPFIND: return BRIX_MIRROR_M_PROPFIND;
42 case NGX_HTTP_OPTIONS: return BRIX_MIRROR_M_OPTIONS;
43 /* Write methods (Phase 24 write mirroring; gated by brix_mirror_writes). */
44 case NGX_HTTP_PUT: return BRIX_MIRROR_M_PUT;
45 case NGX_HTTP_DELETE: return BRIX_MIRROR_M_DELETE;
46 case NGX_HTTP_MKCOL: return BRIX_MIRROR_M_MKCOL;
47 case NGX_HTTP_MOVE: return BRIX_MIRROR_M_MOVE;
48 case NGX_HTTP_COPY: return BRIX_MIRROR_M_COPY;
49 default: return 0;
50 }
51}

63/* mirror subrequest execution (CONTENT phase) */
64static ngx_int_t
65brix_http_mirror_proxy(ngx_http_request_t *r,
66 ngx_http_brix_webdav_req_ctx_t *ctx)
67{
68 ngx_http_brix_webdav_loc_conf_t *conf;
69 ngx_http_upstream_t *u;
70 brix_mirror_target_t *t;
71 struct sockaddr *sa;
72
73 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
74 if (conf->mirror.targets == NULL
75 || ctx->mirror_target_idx >= conf->mirror.targets->nelts)
76 {
77 return NGX_HTTP_INTERNAL_SERVER_ERROR;
78 }
79 t = (brix_mirror_target_t *) conf->mirror.targets->elts
80 + ctx->mirror_target_idx;
81
82 if (ngx_http_upstream_create(r) != NGX_OK) {
83 return NGX_HTTP_INTERNAL_SERVER_ERROR;
84 }
85 u = r->upstream;
86
87 u->conf = &conf->mirror_upstream_conf;
88 u->buffering = 0;
89
90 u->create_request = mirror_create_request;
91 u->reinit_request = mirror_reinit_request;
92 u->process_header = mirror_process_status_line;
93 u->abort_request = mirror_abort_request;
94 u->finalize_request = mirror_finalize_request;
95
96 sa = ngx_palloc(r->pool, t->socklen);
97 u->resolved = ngx_pcalloc(r->pool, sizeof(ngx_http_upstream_resolved_t));
98 if (sa == NULL || u->resolved == NULL) {
99 return NGX_HTTP_INTERNAL_SERVER_ERROR;
100 }
101 ngx_memcpy(sa, &t->sockaddr, t->socklen);
102 u->resolved->sockaddr = sa;
103 u->resolved->socklen = t->socklen;
104 u->resolved->naddrs = 1;
105 u->resolved->host = t->host;
106 u->resolved->port = t->port;
107
108#if (NGX_HTTP_SSL)
109 if (t->ssl) {
110 u->ssl = 1;
111 }
112#endif
113
114 /* Drive the shadow connection. This is a background subrequest, so its
115 * count (taken by ngx_http_subrequest) keeps the request alive until the
116 * upstream finalizes; ngx_http_upstream_init runs the connect→send→read. */
117 ngx_http_upstream_init(r);
118 return NGX_DONE;
119}

121/* Fire one background subrequest per shadow target. ctx (main) is already set
122 * and ctx->mirror_fired marked by the caller. Each subrequest is tagged
123 * is_mirror so the precontent handler takes it over and proxies it. */
124static void
125mirror_fire_subrequests(ngx_http_request_t *r,
126 ngx_http_brix_webdav_loc_conf_t *conf)
127{
128 ngx_uint_t i;
129
130 for (i = 0; i < conf->mirror.targets->nelts; i++) {
131 ngx_http_brix_webdav_req_ctx_t *sctx;
132 ngx_http_request_t *sr;
133
134 sctx = ngx_pcalloc(r->pool, sizeof(*sctx));
135 if (sctx == NULL) { continue; }
136 sctx->is_mirror = 1;
137 sctx->mirror_target_idx = i;
138
139 if (ngx_http_subrequest(r, &r->uri, &r->args, &sr, NULL,
140 NGX_HTTP_SUBREQUEST_BACKGROUND) != NGX_OK)
141 {
142 continue;
143 }
144 ngx_http_set_ctx(sr, sctx, ngx_http_brix_webdav_module);
145 sr->method = r->method;
146 sr->method_name = r->method_name;
147 sr->header_only = 1; /* discard shadow response body */
148 }
149}

151/*
152 * Request-body completion handler for body-bearing mirrored methods (PUT).
153 * Modelled on stock ngx_http_mirror_module: read the body first (in PRECONTENT),
154 * then fire the shadow subrequests (which clone the now-available body) and resume
155 * the main request's phase processing so its own content handler runs normally.
156 */
157static void
158mirror_put_body_handler(ngx_http_request_t *r)
159{
160 ngx_http_brix_webdav_loc_conf_t *conf;
161
162 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
163
164 /* Keep the buffered body alive: the shadow subrequest sends a cloned chain
165 * over the same memory/temp-file while the primary PUT handler also reads it. */
166 r->preserve_body = 1;
167 mirror_fire_subrequests(r, conf);
168
169 r->write_event_handler = ngx_http_core_run_phases;
170 ngx_http_core_run_phases(r);
171}

173/* PRECONTENT handler * Two jobs: on the MAIN request, fire one background subrequest per shadow
174 * target; on a mirror SUBREQUEST, take it over and proxy it to the shadow.
175 * Doing the takeover here (before the content phase) avoids any dependence on
176 * content-phase handler ordering. */
177
178/*
179 * WHAT: true if the request carries an X-Xrootd-Mirror loop-guard header.
180 * WHY: a shadow pointed back at this server would otherwise mirror the replay
181 * again; mirror_create_request stamps this header, so its presence means "do not
182 * re-mirror".
183 * HOW: linear scan of the parsed request headers, case-insensitive key compare;
184 * pure query, no side effects.
185 */
186static int
187mirror_request_is_replay(ngx_http_request_t *r)
188{
189 ngx_list_part_t *part = &r->headers_in.headers.part;
190 ngx_table_elt_t *hdr = part->elts;
191 ngx_uint_t k;
192
193 for (k = 0; /* void */; k++) {
194 if (k >= part->nelts) {
195 if (part->next == NULL) { break; }
196 part = part->next;
197 hdr = part->elts;
198 k = 0;
199 }
200 if (hdr[k].key.len == sizeof("X-Xrootd-Mirror") - 1
201 && ngx_strncasecmp(hdr[k].key.data,
202 (u_char *) "X-Xrootd-Mirror",
203 sizeof("X-Xrootd-Mirror") - 1) == 0)
204 {
205 return 1;
206 }
207 }
208 return 0;
209}

211/*
212 * WHAT: decide whether the MAIN request qualifies to be mirrored this pass.
213 * WHY: consolidates every eligibility gate (enabled, already-fired, method mask,
214 * write-guard, loop-guard replay, sampling) so the handler is pure dispatch and
215 * each gate is reviewable in one place.
216 * HOW: returns 1 (fire) or 0 (decline). ctx may be NULL (never fired). The
217 * sampling gate has the side effect of counting a dropped request — kept here to
218 * preserve the original ordering and metric behaviour exactly.
219 */
220static int
221mirror_precontent_eligible(ngx_http_request_t *r,
222 ngx_http_brix_webdav_loc_conf_t *conf,
223 ngx_http_brix_webdav_req_ctx_t *ctx)
224{
225 ngx_uint_t mbit;
226
227 if (!conf->mirror.enabled || conf->mirror.targets == NULL) {
228 return 0;
229 }
230 if (ctx != NULL && ctx->mirror_fired) {
231 return 0; /* idempotent: fire once per request */
232 }
233
234 mbit = brix_http_mirror_method_bit(r);
235 if ((mbit & conf->mirror.method_mask) == 0) {
236 return 0;
237 }
238 /* Write methods only mirror when brix_mirror_writes is on — a second,
239 * independent guard beyond the method mask. The shadow must be an isolated
240 * namespace (replaying writes onto the primary's store would corrupt it). */
241 if ((mbit & BRIX_MIRROR_M_WRITE_ALL) && !conf->mirror.mirror_writes) {
242 return 0;
243 }
244 /* Loop guard: never mirror a request that is itself a mirror replay (set by
245 * mirror_create_request) — protects against a shadow pointed at this server. */
246 if (mirror_request_is_replay(r)) {
247 return 0;
248 }
249 if (!brix_mirror_should_sample(conf->mirror.sample_pct)) {
250 MIR_HTTP_INC(mirror_http_dropped_total);
251 return 0;
252 }
253 return 1;
254}

256ngx_int_t
257brix_http_mirror_precontent_handler(ngx_http_request_t *r)
258{
259 ngx_http_brix_webdav_loc_conf_t *conf;
260 ngx_http_brix_webdav_req_ctx_t *ctx;
261
262 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
263
264 if (r != r->main) {
265 if (ctx != NULL && ctx->is_mirror) {
266 return brix_http_mirror_proxy(r, ctx); /* take over the shadow req */
267 }
268 return NGX_DECLINED; /* some other subrequest — ignore */
269 }
270
271 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
272
273 if (!mirror_precontent_eligible(r, conf, ctx)) {
274 return NGX_DECLINED;
275 }
276
277 if (ctx == NULL) {
278 ctx = ngx_pcalloc(r->pool, sizeof(*ctx));
279 if (ctx == NULL) { return NGX_DECLINED; }
280 ngx_http_set_ctx(r, ctx, ngx_http_brix_webdav_module);
281 }
282 ctx->mirror_fired = 1;
283
284 /* Body-bearing methods (PUT): read the request body first so the shadow
285 * subrequest can forward a clone of it, then fire from the body handler and
286 * resume phase processing. NGX_DONE suspends the main request until the body
287 * is buffered — it is not exposed to the client and the client is not delayed
288 * beyond its own normal body read. */
289 if (brix_http_mirror_method_has_body(r->method)) {
290 ngx_int_t rc;
291
292 r->preserve_body = 1;
293 rc = ngx_http_read_client_request_body(r, mirror_put_body_handler);
294 if (rc >= NGX_HTTP_SPECIAL_RESPONSE) {
295 return rc;
296 }
297 return NGX_DONE;
298 }
299
300 /* Bodyless methods (GET/HEAD/PROPFIND/OPTIONS/DELETE/MKCOL/MOVE/COPY): fire
301 * immediately and let the content handler proceed. */
302 mirror_fire_subrequests(r, conf);
303 return NGX_DECLINED; /* main request continues to its content handler */
304}
