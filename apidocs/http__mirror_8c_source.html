1/*
2 * http_mirror.c — Phase 24 HTTP/WebDAV traffic mirror (see http_mirror.h).
3 *
4 * WHAT: Replays qualifying read requests to one or more shadow HTTP(S) backends
5 * fire-and-forget, compares the shadow status to the primary, and counts
6 * divergence — without ever delaying or exposing the shadow path to the client.
7 *
8 * HOW: A PRECONTENT-phase handler on the MAIN request fires one background
9 * subrequest per shadow target (re-using the same location so the mirror config
10 * is in scope). Each subrequest is tagged is_mirror in its WebDAV request ctx;
11 * the WebDAV access/content handlers skip those, and a CONTENT-phase handler
12 * registered ahead of the WebDAV handler takes the subrequest over and proxies
13 * it to the shadow with credentials stripped (unless an explicit mirror token is
14 * configured). The shadow response is parsed with the same upstream-callback
15 * pattern as src/webdav/proxy_response.c but writes into a scratch status field
16 * in the request ctx; finalize compares status classes and updates metrics. A
17 * LOG-phase handler stamps the primary's final status for that comparison.
18 */
19#include "http_mirror.h"
20#include "core/http/http_body.h"
21#include "core/http/http_headers.h"

58/*
59 * Rewrite a WebDAV Destination header URL so its host points at the shadow.
60 * "https://public.example.org/dav/foo" + url_base "http://shadow:8443"
61 * => "http://shadow:8443/dav/foo". Mirrors webdav_proxy_rewrite_destination()
62 * (which is static to the proxy module); a relative/odd value is passed through.
63 */
64static ngx_str_t
65brix_http_mirror_rewrite_dest(ngx_pool_t *pool, ngx_str_t *dest,
66 ngx_str_t *url_base)
67{
68 ngx_str_t result = *dest;
69 u_char *p, *end, *path_start;
70 size_t path_len;
71
72 if (dest->len < 7
73 || (ngx_strncasecmp(dest->data, (u_char *) "http://", 7) != 0
74 && ngx_strncasecmp(dest->data, (u_char *) "https://", 8) != 0))
75 {
76 return result; /* not absolute — forward unchanged */
77 }
78 p = ngx_strlchr(dest->data, dest->data + dest->len, '/');
79 if (p == NULL) { return result; }
80 p++; /* skip first '/' of "://" */
81 p = ngx_strlchr(p, dest->data + dest->len, '/');
82 if (p == NULL) { return *url_base; } /* scheme://host only */
83
84 path_start = p;
85 end = dest->data + dest->len;
86 path_len = (size_t) (end - path_start);
87
88 result.len = url_base->len + path_len;
89 result.data = ngx_pnalloc(pool, result.len);
90 if (result.data == NULL) { return *dest; }
91 p = ngx_copy(result.data, url_base->data, url_base->len);
92 (void) ngx_copy(p, path_start, path_len);
93 return result;
94}

96/*
97 * Clone a request-body buf chain into r->pool with INDEPENDENT buf structs so the
98 * shadow send (which advances buf->pos / file_pos) never disturbs the primary's
99 * own consumption of the same body. The underlying memory / file handle is shared
100 * read-only; only the small ngx_buf_t bookkeeping is duplicated.
101 */
102static ngx_chain_t *
103brix_http_mirror_clone_body(ngx_http_request_t *r,
104 ngx_http_request_t *parent, off_t *len_out)
105{
106 ngx_chain_t *in, *head = NULL, **tail = &head;
107 off_t total = 0;
108
109 *len_out = 0;
110 if (parent->request_body == NULL || parent->request_body->bufs == NULL) {
111 return NULL;
112 }
113 for (in = parent->request_body->bufs; in != NULL; in = in->next) {
114 ngx_buf_t *b;
115 ngx_chain_t *cl;
116
117 if (ngx_buf_size(in->buf) == 0) { continue; }
118 cl = ngx_alloc_chain_link(r->pool);
119 b = ngx_palloc(r->pool, sizeof(ngx_buf_t));
120 if (cl == NULL || b == NULL) { *len_out = -1; return NULL; }
121 *b = *in->buf; /* independent pos/last/file_pos/file_last */
122 b->last_buf = b->last_in_chain = 0;
123 cl->buf = b;
124 cl->next = NULL;
125 *tail = cl;
126 tail = &cl->next;
127 total += ngx_buf_size(in->buf);
128 }
129 *len_out = total;
130 return head;
131}

134/* upstream callbacks (shadow request/response) */
135static ngx_int_t
136mirror_create_request(ngx_http_request_t *r)
137{
138 ngx_http_brix_webdav_loc_conf_t *conf;
139 ngx_http_brix_webdav_req_ctx_t *ctx;
140 brix_mirror_target_t *t;
141 ngx_buf_t *b;
142 ngx_chain_t *cl;
143 size_t len;
144 u_char *p;
145 ngx_str_t host;
146 int inject_token;
147 ngx_uint_t method, has_body;
148 ngx_table_elt_t *depth_h = NULL, *over_h = NULL;
149 ngx_str_t dest; /* rewritten Destination */
150 ngx_chain_t *body = NULL;
151 off_t body_len = 0;
152 u_char clbuf[NGX_OFF_T_LEN];
153 size_t cl_digits = 0;
154
155 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
156 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
157 if (ctx == NULL || conf->mirror.targets == NULL
158 || ctx->mirror_target_idx >= conf->mirror.targets->nelts)
159 {
160 return NGX_ERROR;
161 }
162 t = (brix_mirror_target_t *) conf->mirror.targets->elts
163 + ctx->mirror_target_idx;
164 host = t->host;
165 method = r->method;
166 has_body = brix_http_mirror_method_has_body(method);
167 ngx_str_null(&dest);
168
169 /* Auth policy on the shadow request:
170 * token configured => inject "Authorization: Bearer <token>"
171 * strip_auth on => send no Authorization at all (default)
172 * strip_auth off => forward the client's Authorization header. */
173 inject_token = (conf->mirror.token.len > 0);
174
175 /* MOVE/COPY: rewrite the client's Destination to point at the shadow, and
176 * forward Depth/Overwrite verbatim so the shadow performs the same op. */
177 if (method == NGX_HTTP_MOVE || method == NGX_HTTP_COPY) {
178 ngx_table_elt_t *dest_h = brix_http_find_header(r,
179 "Destination", sizeof("Destination") - 1);
180 if (dest_h != NULL) {
181 dest = brix_http_mirror_rewrite_dest(r->pool, &dest_h->value,
182 &t->url_base);
183 }
184 depth_h = brix_http_find_header(r, "Depth",
185 sizeof("Depth") - 1);
186 over_h = brix_http_find_header(r, "Overwrite",
187 sizeof("Overwrite") - 1);
188 }
189
190 /* PUT: forward the (preserved) request body cloned from the parent request. */
191 if (has_body && r->parent != NULL) {
192 body = brix_http_mirror_clone_body(r, r->parent, &body_len);
193 if (body_len < 0) {
194 return NGX_ERROR; /* clone alloc failed — don't send a truncated PUT */
195 }
196 cl_digits = (size_t) (ngx_sprintf(clbuf, "%O", body_len) - clbuf);
197 }
198
199 /* "METHOD uri[?args] HTTP/1.1\r\nHost: <host>\r\nConnection: close\r\n" */
200 len = r->method_name.len + 1
201 + r->uri.len
202 + (r->args.len ? 1 + r->args.len : 0)
203 + sizeof(" HTTP/1.1" CRLF) - 1
204 + sizeof("Host: ") - 1 + host.len + sizeof(CRLF) - 1
205 + sizeof("Connection: close" CRLF) - 1
206 + sizeof("X-Xrootd-Mirror: 1" CRLF) - 1
207 + sizeof(CRLF) - 1;
208
209 if (inject_token) {
210 len += sizeof("Authorization: Bearer ") - 1
211 + conf->mirror.token.len + sizeof(CRLF) - 1;
212 } else if (!conf->mirror.strip_auth
213 && r->headers_in.authorization != NULL) {
214 len += sizeof("Authorization: ") - 1
215 + r->headers_in.authorization->value.len + sizeof(CRLF) - 1;
216 }
217 if (dest.len) {
218 len += sizeof("Destination: ") - 1 + dest.len + sizeof(CRLF) - 1;
219 }
220 if (depth_h != NULL) {
221 len += sizeof("Depth: ") - 1 + depth_h->value.len + sizeof(CRLF) - 1;
222 }
223 if (over_h != NULL) {
224 len += sizeof("Overwrite: ") - 1 + over_h->value.len + sizeof(CRLF) - 1;
225 }
226 if (has_body) {
227 len += sizeof("Content-Length: ") - 1 + cl_digits + sizeof(CRLF) - 1;
228 }
229
230 b = ngx_create_temp_buf(r->pool, len);
231 if (b == NULL) {
232 return NGX_ERROR;
233 }
234 p = b->last;
235
236 p = ngx_copy(p, r->method_name.data, r->method_name.len);
237 *p++ = ' ';
238 p = ngx_copy(p, r->uri.data, r->uri.len);
239 if (r->args.len) {
240 *p++ = '?';
241 p = ngx_copy(p, r->args.data, r->args.len);
242 }
243 p = ngx_copy(p, " HTTP/1.1" CRLF, sizeof(" HTTP/1.1" CRLF) - 1);
244
245 p = ngx_copy(p, "Host: ", sizeof("Host: ") - 1);
246 p = ngx_copy(p, host.data, host.len);
247 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
248
249 p = ngx_copy(p, "Connection: close" CRLF,
250 sizeof("Connection: close" CRLF) - 1);
251
252 /* Loop guard: a shadow that happens to be this same server will see this
253 * marker and decline to mirror again (brix_http_mirror_precontent_handler). */
254 p = ngx_copy(p, "X-Xrootd-Mirror: 1" CRLF,
255 sizeof("X-Xrootd-Mirror: 1" CRLF) - 1);
256
257 if (inject_token) {
258 p = ngx_copy(p, "Authorization: Bearer ",
259 sizeof("Authorization: Bearer ") - 1);
260 p = ngx_copy(p, conf->mirror.token.data, conf->mirror.token.len);
261 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
262 } else if (!conf->mirror.strip_auth
263 && r->headers_in.authorization != NULL) {
264 p = ngx_copy(p, "Authorization: ", sizeof("Authorization: ") - 1);
265 p = ngx_copy(p, r->headers_in.authorization->value.data,
266 r->headers_in.authorization->value.len);
267 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
268 }
269 if (dest.len) {
270 p = ngx_copy(p, "Destination: ", sizeof("Destination: ") - 1);
271 p = ngx_copy(p, dest.data, dest.len);
272 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
273 }
274 if (depth_h != NULL) {
275 p = ngx_copy(p, "Depth: ", sizeof("Depth: ") - 1);
276 p = ngx_copy(p, depth_h->value.data, depth_h->value.len);
277 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
278 }
279 if (over_h != NULL) {
280 p = ngx_copy(p, "Overwrite: ", sizeof("Overwrite: ") - 1);
281 p = ngx_copy(p, over_h->value.data, over_h->value.len);
282 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
283 }
284 if (has_body) {
285 p = ngx_copy(p, "Content-Length: ", sizeof("Content-Length: ") - 1);
286 p = ngx_copy(p, clbuf, cl_digits);
287 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
288 }
289
290 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
291 b->last = p;
292
293 cl = ngx_alloc_chain_link(r->pool);
294 if (cl == NULL) {
295 return NGX_ERROR;
296 }
297 cl->buf = b;
298 cl->next = body; /* PUT body (cloned) follows the header block */
299 r->upstream->request_bufs = cl;
300 return NGX_OK;
301}

369{
370 ngx_http_brix_webdav_loc_conf_t *conf;

385 }
386
387 if (parent != NULL && shadow_status != 0) {
388 ngx_http_brix_webdav_req_ctx_t *pctx =
389 ngx_http_get_module_ctx(parent, ngx_http_brix_webdav_module);
390
391 if (pctx != NULL && pctx->primary_status != 0
392 && brix_mirror_status_class(shadow_status)
393 != brix_mirror_status_class(pctx->primary_status))
394 {
395 MIR_HTTP_INC(mirror_http_divergence_total);
396 if (conf->mirror.log_diverge) {
397 ngx_log_error(NGX_LOG_NOTICE, r->connection->log, 0,
398 "xrootd mirror divergence: primary=%ui shadow=%ui uri=%V",
399 pctx->primary_status, shadow_status, &parent->uri);
400 }
401 }
402 }

406/* mirror subrequest execution (CONTENT phase) */
407static ngx_int_t
408brix_http_mirror_proxy(ngx_http_request_t *r,
409 ngx_http_brix_webdav_req_ctx_t *ctx)
410{
411 ngx_http_brix_webdav_loc_conf_t *conf;
412 ngx_http_upstream_t *u;
413 brix_mirror_target_t *t;
414 struct sockaddr *sa;
415
416 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
417 if (conf->mirror.targets == NULL
418 || ctx->mirror_target_idx >= conf->mirror.targets->nelts)
419 {
420 return NGX_HTTP_INTERNAL_SERVER_ERROR;
421 }
422 t = (brix_mirror_target_t *) conf->mirror.targets->elts
423 + ctx->mirror_target_idx;
424
425 if (ngx_http_upstream_create(r) != NGX_OK) {
426 return NGX_HTTP_INTERNAL_SERVER_ERROR;
427 }
428 u = r->upstream;
429
430 u->conf = &conf->mirror_upstream_conf;
431 u->buffering = 0;
432
433 u->create_request = mirror_create_request;
434 u->reinit_request = mirror_reinit_request;
435 u->process_header = mirror_process_status_line;
436 u->abort_request = mirror_abort_request;
437 u->finalize_request = mirror_finalize_request;
438
439 sa = ngx_palloc(r->pool, t->socklen);
440 u->resolved = ngx_pcalloc(r->pool, sizeof(ngx_http_upstream_resolved_t));
441 if (sa == NULL || u->resolved == NULL) {
442 return NGX_HTTP_INTERNAL_SERVER_ERROR;
443 }
444 ngx_memcpy(sa, &t->sockaddr, t->socklen);
445 u->resolved->sockaddr = sa;
446 u->resolved->socklen = t->socklen;
447 u->resolved->naddrs = 1;
448 u->resolved->host = t->host;
449 u->resolved->port = t->port;
450
451#if (NGX_HTTP_SSL)
452 if (t->ssl) {
453 u->ssl = 1;
454 }
455#endif
456
457 /* Drive the shadow connection. This is a background subrequest, so its
458 * count (taken by ngx_http_subrequest) keeps the request alive until the
459 * upstream finalizes; ngx_http_upstream_init runs the connect→send→read. */
460 ngx_http_upstream_init(r);
461 return NGX_DONE;
462}

464/* Fire one background subrequest per shadow target. ctx (main) is already set
465 * and ctx->mirror_fired marked by the caller. Each subrequest is tagged
466 * is_mirror so the precontent handler takes it over and proxies it. */
467static void
468mirror_fire_subrequests(ngx_http_request_t *r,
469 ngx_http_brix_webdav_loc_conf_t *conf)
470{
471 ngx_uint_t i;
472
473 for (i = 0; i < conf->mirror.targets->nelts; i++) {
474 ngx_http_brix_webdav_req_ctx_t *sctx;
475 ngx_http_request_t *sr;
476
477 sctx = ngx_pcalloc(r->pool, sizeof(*sctx));
478 if (sctx == NULL) { continue; }
479 sctx->is_mirror = 1;
480 sctx->mirror_target_idx = i;
481
482 if (ngx_http_subrequest(r, &r->uri, &r->args, &sr, NULL,
483 NGX_HTTP_SUBREQUEST_BACKGROUND) != NGX_OK)
484 {
485 continue;
486 }
487 ngx_http_set_ctx(sr, sctx, ngx_http_brix_webdav_module);
488 sr->method = r->method;
489 sr->method_name = r->method_name;
490 sr->header_only = 1; /* discard shadow response body */
491 }
492}

494/*
495 * Request-body completion handler for body-bearing mirrored methods (PUT).
496 * Modelled on stock ngx_http_mirror_module: read the body first (in PRECONTENT),
497 * then fire the shadow subrequests (which clone the now-available body) and resume
498 * the main request's phase processing so its own content handler runs normally.
499 */
500static void
501mirror_put_body_handler(ngx_http_request_t *r)
502{
503 ngx_http_brix_webdav_loc_conf_t *conf;
504
505 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
506
507 /* Keep the buffered body alive: the shadow subrequest sends a cloned chain
508 * over the same memory/temp-file while the primary PUT handler also reads it. */
509 r->preserve_body = 1;

512 r->write_event_handler = ngx_http_core_run_phases;

514}

516/* PRECONTENT handler * Two jobs: on the MAIN request, fire one background subrequest per shadow
517 * target; on a mirror SUBREQUEST, take it over and proxy it to the shadow.
518 * Doing the takeover here (before the content phase) avoids any dependence on
519 * content-phase handler ordering. */
520
521ngx_int_t
522brix_http_mirror_precontent_handler(ngx_http_request_t *r)
523{
524 ngx_http_brix_webdav_loc_conf_t *conf;
525 ngx_http_brix_webdav_req_ctx_t *ctx;
526 ngx_uint_t mbit;
527
528 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
529
530 if (r != r->main) {
531 if (ctx != NULL && ctx->is_mirror) {
532 return brix_http_mirror_proxy(r, ctx); /* take over the shadow req */
533 }
534 return NGX_DECLINED; /* some other subrequest — ignore */
535 }
536
537 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
538 if (!conf->mirror.enabled || conf->mirror.targets == NULL) {
539 return NGX_DECLINED;
540 }
541
542 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
543 if (ctx != NULL && ctx->mirror_fired) {
544 return NGX_DECLINED; /* idempotent: fire once per request */
545 }
546
547 mbit = brix_http_mirror_method_bit(r);
548 if ((mbit & conf->mirror.method_mask) == 0) {
549 return NGX_DECLINED;
550 }
551 /* Write methods only mirror when brix_mirror_writes is on — a second,
552 * independent guard beyond the method mask. The shadow must be an isolated
553 * namespace (replaying writes onto the primary's store would corrupt it). */
554 if ((mbit & BRIX_MIRROR_M_WRITE_ALL) && !conf->mirror.mirror_writes) {
555 return NGX_DECLINED;
556 }
557
558 /* Loop guard: never mirror a request that is itself a mirror replay (set by
559 * mirror_create_request) — protects against a shadow pointed at this server. */
560 {
561 ngx_list_part_t *part = &r->headers_in.headers.part;
562 ngx_table_elt_t *hdr = part->elts;
563 ngx_uint_t k;
564 for (k = 0; /* void */; k++) {
565 if (k >= part->nelts) {
566 if (part->next == NULL) { break; }
567 part = part->next;
568 hdr = part->elts;
569 k = 0;
570 }
571 if (hdr[k].key.len == sizeof("X-Xrootd-Mirror") - 1
572 && ngx_strncasecmp(hdr[k].key.data,
573 (u_char *) "X-Xrootd-Mirror",
574 sizeof("X-Xrootd-Mirror") - 1) == 0)
575 {
576 return NGX_DECLINED;
577 }
578 }
579 }
580 if (!brix_mirror_should_sample(conf->mirror.sample_pct)) {
581 MIR_HTTP_INC(mirror_http_dropped_total);
582 return NGX_DECLINED;
583 }
584
585 if (ctx == NULL) {
586 ctx = ngx_pcalloc(r->pool, sizeof(*ctx));
587 if (ctx == NULL) { return NGX_DECLINED; }
588 ngx_http_set_ctx(r, ctx, ngx_http_brix_webdav_module);
589 }
590 ctx->mirror_fired = 1;
591
592 /* Body-bearing methods (PUT): read the request body first so the shadow
593 * subrequest can forward a clone of it, then fire from the body handler and
594 * resume phase processing. NGX_DONE suspends the main request until the body
595 * is buffered — it is not exposed to the client and the client is not delayed
596 * beyond its own normal body read. */
597 if (brix_http_mirror_method_has_body(r->method)) {
598 ngx_int_t rc;
599
600 r->preserve_body = 1;
601 rc = ngx_http_read_client_request_body(r, mirror_put_body_handler);
602 if (rc >= NGX_HTTP_SPECIAL_RESPONSE) {
603 return rc;
604 }
605 return NGX_DONE;
606 }
607
608 /* Bodyless methods (GET/HEAD/PROPFIND/OPTIONS/DELETE/MKCOL/MOVE/COPY): fire
609 * immediately and let the content handler proceed. */
610 mirror_fire_subrequests(r, conf);
611 return NGX_DECLINED; /* main request continues to its content handler */
612}

694/* directive setters */
695char *
696brix_http_mirror_set_url(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
697{
698 ngx_http_brix_webdav_loc_conf_t *wlcf = conf;
699 ngx_str_t *value = cf->args->elts;
700 ngx_str_t url = value[1];
701 brix_mirror_target_t *t;
702 ngx_url_t u;
703 size_t scheme_len;
704 ngx_uint_t ssl;
705 in_port_t default_port;
706 u_char *p;
707
708 (void) cmd;
709
710 if (ngx_strncasecmp(url.data, (u_char *) "https://", 8) == 0) {
711 ssl = 1; scheme_len = 8; default_port = 443;
712 } else if (ngx_strncasecmp(url.data, (u_char *) "http://", 7) == 0) {
713 ssl = 0; scheme_len = 7; default_port = 80;
714 } else {
715 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
716 "brix_mirror_url: \"%V\" must start with http:// or https://",
717 &url);
718 return NGX_CONF_ERROR;
719 }
720
721 if (wlcf->mirror.targets == NULL) {
722 wlcf->mirror.targets = ngx_array_create(cf->pool,
723 BRIX_MIRROR_MAX_TARGETS, sizeof(brix_mirror_target_t));
724 if (wlcf->mirror.targets == NULL) { return NGX_CONF_ERROR; }
725 }
726 if (wlcf->mirror.targets->nelts >= BRIX_MIRROR_MAX_TARGETS) {
727 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
728 "brix_mirror_url: at most %d targets supported",
729 BRIX_MIRROR_MAX_TARGETS);
730 return NGX_CONF_ERROR;
731 }
732
733 ngx_memzero(&u, sizeof(ngx_url_t));
734 u.url.data = url.data + scheme_len;
735 u.url.len = url.len - scheme_len;
736 u.uri_part = 1;
737 u.default_port = default_port;
738 if (ngx_parse_url(cf->pool, &u) != NGX_OK || u.naddrs == 0) {
739 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
740 "brix_mirror_url: cannot resolve \"%V\"%s%s", &url,
741 u.err ? ": " : "", u.err ? u.err : "");
742 return NGX_CONF_ERROR;
743 }
744
745 t = ngx_array_push(wlcf->mirror.targets);
746 if (t == NULL) { return NGX_CONF_ERROR; }
747 ngx_memzero(t, sizeof(*t));
748 t->url = url;
749 t->ssl = ssl;
750 t->port = u.port;
751
752 /* Host: header value — "host" or "host:port" if non-default. */
753 if (u.port == default_port) {
754 t->host = u.host;
755 } else {
756 p = ngx_pnalloc(cf->pool, u.host.len + 1 + sizeof("65535") - 1);
757 if (p == NULL) { return NGX_CONF_ERROR; }
758 t->host.data = p;
759 t->host.len = ngx_sprintf(p, "%V:%d", &u.host, (int) u.port) - p;
760 }
761
762 /* url_base — "scheme://host[:port]" for logging. */
763 {
764 size_t base = scheme_len + t->host.len;
765 p = ngx_pnalloc(cf->pool, base);
766 if (p == NULL) { return NGX_CONF_ERROR; }
767 ngx_memcpy(p, url.data, scheme_len);
768 ngx_memcpy(p + scheme_len, t->host.data, t->host.len);
769 t->url_base.data = p;
770 t->url_base.len = base;
771 }
772
773 if (u.addrs[0].socklen > sizeof(t->sockaddr)) { return NGX_CONF_ERROR; }
774 ngx_memcpy(&t->sockaddr, u.addrs[0].sockaddr, u.addrs[0].socklen);
775 t->socklen = u.addrs[0].socklen;
776
777 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
778 "brix: WebDAV mirror target %V (ssl=%d)", &t->url_base, (int) ssl);
779 return NGX_CONF_OK;
780}

782char *
783brix_http_mirror_set_methods(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
784{

812 return NGX_CONF_OK;
813}
