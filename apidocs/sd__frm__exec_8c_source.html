23/* ===================== the "exec" MSS adapter (real HSM) =====================
24 * The classic FRM model: an operator-supplied stage command drives the real MSS
25 * (HPSS, CTA, dCache, an Enstore wrapper, ...). The local online buffer lives at
26 * <base>/.online/<key>; the recall/migrate/exists verbs shell out to:
27 * $BRIX_FRM_STAGECMD <verb> <key> <online-path>
28 * recall is expected to be ASYNC-SUBMIT (start the MSS recall and return promptly,
29 * not block until online); the driver then parks the open and polls until the
30 * online buffer appears. A `recall_poll` is the cheap local-buffer existence check,
31 * so no per-poll fork. */
32
33typedef struct {
34 char base[PATH_MAX]; /* local online-buffer root */
35 char stagecmd[PATH_MAX]; /* $BRIX_FRM_STAGECMD */
36 ngx_log_t *log;
37} exec_ctx_t;

39static int
40exec_online_path(const exec_ctx_t *c, const char *key, char *out, size_t cap)
41{
42 int n = snprintf(out, cap, "%s/.online/%s", c->base,
43 (key[0] == '/') ? key + 1 : key);
44
45 return (n > 0 && (size_t) n < cap) ? 0 : -1;
46}

48/* Run "<stagecmd> <verb> <key> <online>"; returns the child's exit code (0 ok), or
49 * -1 on spawn/wait failure. No shell - argv is passed directly (no injection). */
50static int
51exec_run(const exec_ctx_t *c, const char *verb, const char *key,
52 const char *online)
53{
54 char *argv[5];
55 pid_t pid;
56 int status;
57
58 argv[0] = (char *) c->stagecmd;
59 argv[1] = (char *) verb;
60 argv[2] = (char *) key;
61 argv[3] = (char *) online;
62 argv[4] = NULL;
63
64 if (posix_spawn(&pid, c->stagecmd, NULL, NULL, argv, environ) != 0) {
65 return -1;
66 }
67 while (waitpid(pid, &status, 0) < 0) {
68 if (errno != EINTR) {
69 return -1;
70 }
71 }
72 return WIFEXITED(status) ? WEXITSTATUS(status) : -1;
73}

75static int
76exec_residency(void *mss, const char *key, off_t *size_out, time_t *mtime_out)
77{
78 exec_ctx_t *c = mss;
79 char online[PATH_MAX];
80 struct stat sb;
81
82 if (exec_online_path(c, key, online, sizeof(online)) == 0
83 && stat(online, &sb) == 0)
84 {
85 if (size_out) { *size_out = sb.st_size; }
86 if (mtime_out) { *mtime_out = sb.st_mtime; }
87 return BRIX_RESIDENCY_ONLINE;
88 }
89 /* Ask the MSS: exit 0 = on tape (offline), non-zero = absent. The size is
90 * unknown until recalled; the cache fill restats the online buffer. */
91 if (exec_run(c, "exists", key, "") == 0) {
92 if (size_out) { *size_out = 0; }
93 if (mtime_out) { *mtime_out = time(NULL); }
94 return BRIX_RESIDENCY_OFFLINE;
95 }
96 return BRIX_RESIDENCY_ABSENT;
97}

99static int
100exec_recall_begin(void *mss, const char *key)
101{
102 exec_ctx_t *c = mss;
103 char online[PATH_MAX];
104
105 if (exec_online_path(c, key, online, sizeof(online)) != 0) {
106 return -1;
107 }
108 if (access(online, F_OK) == 0) {
109 return 0; /* already online */
110 }
111 frm_mkparents(online); /* the cmd writes the online buffer */
112 return (exec_run(c, "recall", key, online) == 0) ? 0 : -1;
113}
