55/* * Shared chunked-transfer core (B5).
56 *
57 * Every byte-pump in this file is the same loop — fill a CHUNK-sized buffer from
58 * a source, drain it to a sink, honour cooperative cancel + optional progress —
59 * differing only in (a) where bytes come from (remote pgread/read vs a local
60 * fd), (b) where they go (a local fd vs remote pgwrite/write), and (c) whether
61 * the length is known up front (download/r2r: stop at si->size, where a 0-read
62 * is a short-read ERROR) or EOF-driven (upload/recursive: a 0-read is the clean
63 * end). transfer_pump() owns the loop once; the source/sink are supplied as
64 * small read/write adapters, so each caller keeps its own open/close/teardown
65 * verbatim — only the malloc+loop+free block collapses here.
66 * ------------------------------------------------------------------------- */
67
68/* Fill up to cap bytes at file offset off. Returns bytes read (>0), 0 for "no
69 * more" (EOF for an EOF-driven pump; a short read for a sized pump), or -1 (st
70 * set). */
71/* Drain exactly n bytes at offset off. Returns 0 or -1 (st set). */
72
73/* Smallest adaptive read request on a high packet-loss link. Balances two costs:
74 * smaller = each read likelier to get through, but more reads ⇒ more (expensive,
75 * multi-RTT) reconnects after the severs that do happen. 256 KiB is the sweet
76 * spot measured against a fault proxy up to 15% per-segment loss. */
77
78/* Effective resilience (max-stall) window in ms for a copy, honouring the
79 * fail-fast knob: `no_retry` (set by --no-retry / --retry 0 / --max-stall 0)
80 * yields 0 — every bounded copy loop then has a zero-length deadline and fails
81 * on the first transport fault instead of spinning the default window against a
82 * dead endpoint. Otherwise an explicit positive max_stall_ms wins, else `dflt`.
83 * Centralises the choice the bounded loops below all make so `no_retry` can
84 * never again be silently dropped (the raw `max_stall_ms > 0 ? : DEFAULT`
85 * ternary could not distinguish "fail fast" from "use the default"). */
86int
87copy_stall_ms(const brix_copy_opts *o, int dflt)
88{
89 if (o != NULL && o->no_retry) {
90 return 0;
91 }
92 if (o != NULL && o->max_stall_ms > 0) {
93 return o->max_stall_ms;
94 }
95 return dflt;
96}

99/*
100 * Resilient session bring-up for a download: connect + stat, retried within a
101 * max_stall window on a transport fault (with fast backoff), so the MULTI-RTT
102 * setup phase (handshake + login + stat) survives a flaky/lossy link instead of
103 * failing the whole copy on the first sever — the same patience the read pump
104 * gets. On success `c` is connected and `*si` is filled. 0 / -1 (st set).
105 */
106int
107resilient_setup(brix_conn *c, const brix_url *su, const brix_opts *co,
108 brix_statinfo *si, int max_stall_ms, brix_status *st)
109{
110 uint64_t deadline = brix_mono_ns() + (uint64_t) max_stall_ms * 1000000ULL;
111 unsigned attempt = 0;
112 int up = 0;
113 for (;;) {
114 if (!up && brix_connect(c, su, co, st) == 0) {
115 up = 1;
116 }
117 if (up && brix_stat(c, su->path, si, st) == 0) {
118 return 0;
119 }
120 /* A connection that has NEVER established and is actively REFUSED is a
121 * definitive failure (nothing is listening), not a transient stall — fail
122 * fast instead of burning the whole max_stall window retrying it. Re-attempts
123 * for a momentarily-down/restarting endpoint are the OUTER --retry's job
124 * (it backs off with jitter between whole-copy attempts). A connect TIMEOUT
125 * / unreachable (lossy link) or any fault AFTER we were up stays retryable. */
126 if (!up && st->kxr == XRDC_ESOCK && st->sys_errno == ECONNREFUSED) {
127 return -1;
128 }
129 if (!brix_status_retryable(st) || brix_copy_quit_requested()
130 || brix_mono_ns() >= deadline) {
131 if (up) {
132 brix_close(c);
133 }
134 return -1;
135 }
136 if (up) { /* drop the suspect session before reconnecting */
137 brix_close(c);
138 up = 0;
139 }
140 brix_backoff_sleep_fast(attempt++);
141 }
142}

147/* Copy one remote file (open under conn c) to a fresh local file. */
148int
149copy_one_r2l(brix_conn *c, const char *rpath, const char *lpath,
150 int64_t expected_size, brix_status *st)
151{
152 brix_file f;
153 brix_vfs_file *vf = NULL;
154 brix_vfs_open_opts vopts;
155 pump_local_t lc;
156 pump_remote_t src = {0};
157 int rc;
158
159 if (brix_file_open_read(c, rpath, &f, st) != 0) {
160 return -1;
161 }
162
163 /* Atomic temp+rename + signal-abort via VFS, mirroring copy_download — so a
164 * failed or interrupted recursive transfer leaves no partial/truncated tree
165 * entry. FORCE is always set: recursive copies overwrite existing files. */
166 vopts.io_uring = XRDC_IO_URING_AUTO;
167 vopts.expected_size = expected_size;
168 vopts.cred = NULL;
169 if (brix_vfs_open(lpath, XRDC_VFS_WRITE | XRDC_VFS_FORCE,
170 &vopts, &vf, st) != 0) {
171 brix_file_close(c, &f, st);
172 return -1;
173 }
174
175 /* remote → local VFS (plain read, recursive path; not paged). */
176 src.c = c;
177 src.f = &f;
178 src.pgrw = 0;
179 lc.vf = vf;
180 rc = transfer_pump(pump_src_remote, &src, pump_sink_local_vfs, &lc,
181 expected_size, NULL, expected_size, st);
182
183 if (rc == 0) {
184 rc = brix_vfs_commit(vf, st);
185 } else {
186 brix_vfs_abort(vf);
187 }
188 brix_vfs_close(vf);
189 brix_file_close(c, &f, st);
190 return rc;
191}

194/* Copy one local file to a fresh remote file (open under conn c). */
195int
196copy_one_l2r(brix_conn *c, const char *lpath, const char *rpath,
197 const brix_copy_opts *o, brix_status *st)
198{
199 brix_file f;
200 brix_vfs_file *vf = NULL;
201 brix_vfs_open_opts vopts;
202 pump_local_t lc;
203 pump_remote_t sink = {0};
204 int rc;
205
206 vopts.io_uring = (o != NULL) ? o->io_uring : XRDC_IO_URING_AUTO;
207 vopts.expected_size = -1;
208 vopts.cred = NULL;
209 if (brix_vfs_open(lpath, XRDC_VFS_READ, &vopts, &vf, st) != 0) {
210 return -1;
211 }
212 if (brix_file_open_write(c, rpath, 1 /*force*/, o ? o->posc : 0, &f, st) != 0) {
213 brix_vfs_close(vf);
214 return -1;
215 }
216
217 /* local VFS (EOF-driven) → remote, plain write (recursive path). On failure
218 * the file is still closed (preserving the historical recursive teardown,
219 * which never relied on POSC-discard the way single-file upload does). */
220 sink.c = c;
221 sink.f = &f;
222 sink.pgrw = 0;
223 lc.vf = vf;
224 rc = transfer_pump(pump_src_local_vfs, &lc, pump_sink_remote, &sink,
225 -1, NULL, 0, st);
226
227 brix_vfs_close(vf);
228 if (rc != 0) {
229 brix_file_close(c, &f, st);
230 return -1;
231 }
232 return brix_file_close(c, &f, st);
233}

236int
237brix_copy(const char *src, const char *dst, const brix_copy_opts *o,
238 const brix_opts *co_in, brix_status *st)
239{
240 brix_url su, du;
241 int src_remote, dst_remote, src_local, dst_local;
242
243 /* copy.c manages its OWN reconnect/retry — resilient_setup() for the multi-RTT
244 * bring-up and the read pump (pump_src_remote) for mid-transfer severs. Disable
245 * the library's op-level baked resilience on the connections this path owns, so
246 * stat/dirlist/mkdir don't double-retry inside copy's already-bounded loops. */
247 brix_opts co_local;
248 if (co_in != NULL) {
249 co_local = *co_in;
250 } else {
251 memset(&co_local, 0, sizeof(co_local));
252 co_local.verify_host = 1;
253 }
254 co_local.no_retry = 1;
255 const brix_opts *co = &co_local;
256
257 /* Web schemes (davs/http(s)/s3) take the HTTP transfer path, never the root://
258 * session machinery. Check before brix_url_parse (which is root-only). */
259 if (brix_is_web_url(src) || brix_is_web_url(dst)) {
260 return copy_web(src, dst, o, co, st);
261 }
262
263 /* block:// (and /dev/) endpoints route through the VFS block backend.
264 * brix_url_parse does not know the block:// scheme and would reject it,
265 * so intercept here before the parse. */
266 if (brix_is_block_url(src) || brix_is_block_url(dst)) {
267 return copy_block(src, dst, o, co, st);
268 }
269
270 if (brix_url_parse(src, &su, st) != 0) {
271 return -1;
272 }
273 if (brix_url_parse(dst, &du, st) != 0) {
274 return -1;
275 }
276
277 src_remote = (su.scheme == XRDC_SCHEME_ROOT || su.scheme == XRDC_SCHEME_ROOTS);
278 dst_remote = (du.scheme == XRDC_SCHEME_ROOT || du.scheme == XRDC_SCHEME_ROOTS);
279 src_local = (su.scheme == XRDC_SCHEME_LOCAL || su.scheme == XRDC_SCHEME_STDIO);
280 dst_local = (du.scheme == XRDC_SCHEME_LOCAL || du.scheme == XRDC_SCHEME_STDIO);
281
282 if (o != NULL && o->recursive) {
283 if (src_remote && dst_local) {
284 return copy_recursive(&su, &du, 1, o, co, st); /* remote tree → local */
285 }
286 if (src_local && dst_remote) {
287 return copy_recursive(&su, &du, 0, o, co, st); /* local tree → remote */
288 }
289 brix_status_set(st, XRDC_EUSAGE, 0,
290 "recursive copy requires one remote and one local endpoint");
291 return -1;
292 }
293
294 /* phase-42 W3 write: --zip / --zip-append stores a LOCAL source as a STORE
295 * member of the destination ZIP archive (local or remote dst). */
296 if (o != NULL && (o->zip || o->zip_append)) {
297 if (!src_local) {
298 brix_status_set(st, XRDC_EUSAGE, 0,
299 "--zip requires a local-file source");
300 return -1;
301 }
302 return copy_zip_store(&su, &du, o, co, st);
303 }
304
305 if (src_remote && dst_local) {
306 char member[XRDC_PATH_MAX], arch[XRDC_PATH_MAX];
307 if (unzip_member_from_src(src, &su, member, sizeof(member),
308 arch, sizeof(arch))) {
309 return copy_unzip(&su, arch, member, &du, o, co, st); /* ?xrdcl.unzip= */
310 }
311 return copy_download(&su, &du, o, co, st); /* remote → local */
312 }
313 if (src_local && dst_remote) {
314 return copy_upload(&su, &du, o, co, st); /* local → remote */
315 }
316 if (src_remote && dst_remote) {
317 if (o->tpc_mode == XRDC_TPC_ONLY) {
318 return copy_tpc(&su, &du, o, co, st); /* TPC or hard fail */
319 }
320 if (o->tpc_mode == XRDC_TPC_FIRST
321 || o->tpc_mode == XRDC_TPC_DELEGATE) {
322 if (copy_tpc(&su, &du, o, co, st) == 0) {
323 return 0;
324 }
325 if (!o->silent) {
326 fprintf(stderr, "xrdcp: TPC failed (%s); falling back to "
327 "client-mediated copy\n", st->msg);
328 }
329 return copy_remote_to_remote(&su, &du, o, co, st);
330 }
331 return copy_remote_to_remote(&su, &du, o, co, st); /* default: client-mediated */
332 }
333
334 brix_status_set(st, XRDC_EUSAGE, 0,
335 "unsupported copy direction (local→local not supported)");
336 return -1;
337}
