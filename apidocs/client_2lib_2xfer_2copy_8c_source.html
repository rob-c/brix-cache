55/* * Shared chunked-transfer core (B5).
56 *
57 * Every byte-pump in this file is the same loop — fill a CHUNK-sized buffer from
58 * a source, drain it to a sink, honour cooperative cancel + optional progress —
59 * differing only in (a) where bytes come from (remote pgread/read vs a local
60 * fd), (b) where they go (a local fd vs remote pgwrite/write), and (c) whether
61 * the length is known up front (download/r2r: stop at si->size, where a 0-read
62 * is a short-read ERROR) or EOF-driven (upload/recursive: a 0-read is the clean
63 * end). transfer_pump() owns the loop once; the source/sink are supplied as
64 * small read/write adapters, so each caller keeps its own open/close/teardown
65 * verbatim — only the malloc+loop+free block collapses here.
66 * ------------------------------------------------------------------------- */
67
68/* Fill up to cap bytes at file offset off. Returns bytes read (>0), 0 for "no
69 * more" (EOF for an EOF-driven pump; a short read for a sized pump), or -1 (st
70 * set). */
71/* Drain exactly n bytes at offset off. Returns 0 or -1 (st set). */
72
73/* Smallest adaptive read request on a high packet-loss link. Balances two costs:
74 * smaller = each read likelier to get through, but more reads ⇒ more (expensive,
75 * multi-RTT) reconnects after the severs that do happen. 256 KiB is the sweet
76 * spot measured against a fault proxy up to 15% per-segment loss. */
77
78/* Effective resilience (max-stall) window in ms for a copy, honouring the
79 * fail-fast knob: `no_retry` (set by --no-retry / --retry 0 / --max-stall 0)
80 * yields 0 — every bounded copy loop then has a zero-length deadline and fails
81 * on the first transport fault instead of spinning the default window against a
82 * dead endpoint. Otherwise an explicit positive max_stall_ms wins, else `dflt`.
83 * Centralises the choice the bounded loops below all make so `no_retry` can
84 * never again be silently dropped (the raw `max_stall_ms > 0 ? : DEFAULT`
85 * ternary could not distinguish "fail fast" from "use the default"). */
86int
87copy_stall_ms(const brix_copy_opts *o, int dflt)
88{
89 if (o != NULL && o->no_retry) {
90 return 0;
91 }
92 if (o != NULL && o->max_stall_ms > 0) {
93 return o->max_stall_ms;
94 }
95 return dflt;
96}

99/*
100 * Resilient session bring-up for a download: connect + stat, retried within a
101 * max_stall window on a transport fault (with fast backoff), so the MULTI-RTT
102 * setup phase (handshake + login + stat) survives a flaky/lossy link instead of
103 * failing the whole copy on the first sever — the same patience the read pump
104 * gets. On success `c` is connected and `*si` is filled. 0 / -1 (st set).
105 */
106int
107resilient_setup(brix_conn *c, const brix_url *su, const brix_opts *co,
108 brix_statinfo *si, int max_stall_ms, brix_status *st)
109{
110 uint64_t deadline = brix_mono_ns() + (uint64_t) max_stall_ms * 1000000ULL;
111 unsigned attempt = 0;
112 int up = 0;
113 for (;;) {
114 if (!up && brix_connect(c, su, co, st) == 0) {
115 up = 1;
116 }
117 if (up && brix_stat(c, su->path, si, st) == 0) {
118 return 0;
119 }
120 /* A connection that has NEVER established and is actively REFUSED is a
121 * definitive failure (nothing is listening), not a transient stall — fail
122 * fast instead of burning the whole max_stall window retrying it. Re-attempts
123 * for a momentarily-down/restarting endpoint are the OUTER --retry's job
124 * (it backs off with jitter between whole-copy attempts). A connect TIMEOUT
125 * / unreachable (lossy link) or any fault AFTER we were up stays retryable. */
126 if (!up && st->kxr == XRDC_ESOCK && st->sys_errno == ECONNREFUSED) {
127 return -1;
128 }
129 if (!brix_status_retryable(st) || brix_copy_quit_requested()
130 || brix_mono_ns() >= deadline) {
131 if (up) {
132 brix_close(c);
133 }
134 return -1;
135 }
136 if (up) { /* drop the suspect session before reconnecting */
137 brix_close(c);
138 up = 0;
139 }
140 brix_backoff_sleep_fast(attempt++);
141 }
142}

147/* Copy one remote file (open under conn c) to a fresh local file. */
148int
149copy_one_r2l(brix_conn *c, const char *rpath, const char *lpath,
150 int64_t expected_size, brix_status *st)
151{
152 brix_file f;
153 brix_vfs_file *vf = NULL;
154 brix_vfs_open_opts vopts;
155 pump_local_t lc;
156 pump_remote_t src = {0};
157 int rc;
158
159 if (brix_file_open_read(c, rpath, &f, st) != 0) {
160 return -1;
161 }
162
163 /* Atomic temp+rename + signal-abort via VFS, mirroring copy_download — so a
164 * failed or interrupted recursive transfer leaves no partial/truncated tree
165 * entry. FORCE is always set: recursive copies overwrite existing files. */
166 vopts.io_uring = XRDC_IO_URING_AUTO;
167 vopts.expected_size = expected_size;
168 vopts.cred = NULL;
169 if (brix_vfs_open(lpath, XRDC_VFS_WRITE | XRDC_VFS_FORCE,
170 &vopts, &vf, st) != 0) {
171 brix_file_close(c, &f, st);
172 return -1;
173 }
174
175 /* remote → local VFS (plain read, recursive path; not paged). */
176 src.c = c;
177 src.f = &f;
178 src.pgrw = 0;
179 lc.vf = vf;
180 rc = transfer_pump(pump_src_remote, &src, pump_sink_local_vfs, &lc,
181 expected_size, NULL, expected_size, st);
182
183 if (rc == 0) {
184 rc = brix_vfs_commit(vf, st);
185 } else {
186 brix_vfs_abort(vf);
187 }
188 brix_vfs_close(vf);
189 brix_file_close(c, &f, st);
190 return rc;
191}

194/* Copy one local file to a fresh remote file (open under conn c). */
195int
196copy_one_l2r(brix_conn *c, const char *lpath, const char *rpath,
197 const brix_copy_opts *o, brix_status *st)
198{
199 brix_file f;
200 brix_vfs_file *vf = NULL;
201 brix_vfs_open_opts vopts;
202 pump_local_t lc;
203 pump_remote_t sink = {0};
204 int rc;
205
206 vopts.io_uring = (o != NULL) ? o->io_uring : XRDC_IO_URING_AUTO;
207 vopts.expected_size = -1;
208 vopts.cred = NULL;
209 if (brix_vfs_open(lpath, XRDC_VFS_READ, &vopts, &vf, st) != 0) {
210 return -1;
211 }
212 if (brix_file_open_write(c, rpath, 1 /*force*/, o ? o->posc : 0, &f, st) != 0) {
213 brix_vfs_close(vf);
214 return -1;
215 }
216
217 /* local VFS (EOF-driven) → remote, plain write (recursive path). On failure
218 * the file is still closed (preserving the historical recursive teardown,
219 * which never relied on POSC-discard the way single-file upload does). */
220 sink.c = c;
221 sink.f = &f;
222 sink.pgrw = 0;
223 lc.vf = vf;
224 rc = transfer_pump(pump_src_local_vfs, &lc, pump_sink_remote, &sink,
225 -1, NULL, 0, st);
226
227 brix_vfs_close(vf);
228 if (rc != 0) {
229 brix_file_close(c, &f, st);
230 return -1;
231 }
232 return brix_file_close(c, &f, st);
233}

236/*
237 * WHAT: One classified root:// copy route — both parsed endpoint URLs plus the
238 * four scheme predicates the direction dispatch keys on (src/dst each
239 * remote-vs-local).
240 * WHY: brix_copy formerly parsed the URLs and derived src_remote/dst_remote/
241 * src_local/dst_local inline, then ran a long direction ladder over them in
242 * the same function (CCN 34). Bundling the parse result lets the classify
243 * step and each dispatch step take one const struct, so the orchestrator
244 * stays a flat sequence and the ladder splits per direction without passing
245 * six loose locals around.
246 * HOW: brix_copy_classify() fills one of these (the two brix_url members are
247 * owned by the struct — copied by value from brix_url_parse; the four ints
248 * are pure booleans); the dispatch helpers read it by const pointer and
249 * never mutate it.
250 */
251typedef struct {
252 brix_url su; /* parsed source URL */
253 brix_url du; /* parsed destination URL */
254 int src_remote; /* source is root:///roots:// */
255 int dst_remote; /* dest is root:///roots:// */
256 int src_local; /* source is local/stdio */
257 int dst_local; /* dest is local/stdio */
258} copy_route_t;

261/*
262 * WHAT: Parse both root:// endpoints and classify each end as remote vs local,
263 * filling a copy_route_t.
264 * WHY: Isolates the parse+classify computation from the direction dispatch so
265 * the orchestrator reads linearly and the scheme predicates live in one
266 * named place (§8: pure helper, side effects — the two parses — at the
267 * edge of the classification, error surfaced via st).
268 * HOW: Zero-init the route, run brix_url_parse on src then dst (each may set st
269 * and fail), then derive the four scheme booleans exactly as the former
270 * inline code did. Returns 0 on success (route filled), -1 on a parse
271 * failure (st set by brix_url_parse).
272 */
273static int
274brix_copy_classify(const char *src, const char *dst, copy_route_t *r,
275 brix_status *st)
276{
277 memset(r, 0, sizeof(*r));
278
279 if (brix_url_parse(src, &r->su, st) != 0) {
280 return -1;
281 }
282 if (brix_url_parse(dst, &r->du, st) != 0) {
283 return -1;
284 }
285
286 r->src_remote = (r->su.scheme == XRDC_SCHEME_ROOT
287 || r->su.scheme == XRDC_SCHEME_ROOTS);
288 r->dst_remote = (r->du.scheme == XRDC_SCHEME_ROOT
289 || r->du.scheme == XRDC_SCHEME_ROOTS);
290 r->src_local = (r->su.scheme == XRDC_SCHEME_LOCAL
291 || r->su.scheme == XRDC_SCHEME_STDIO);
292 r->dst_local = (r->du.scheme == XRDC_SCHEME_LOCAL
293 || r->du.scheme == XRDC_SCHEME_STDIO);
294 return 0;
295}

298/*
299 * WHAT: Dispatch a remote-source → local-destination copy: an inline
300 * ?xrdcl.unzip= member extraction if the source names one, else a plain
301 * download.
302 * WHY: Lifts the two-way remote→local branch out of the direction ladder so
303 * brix_copy's dispatch stays flat (§5); behaviour is byte-identical to the
304 * former inline block.
305 * HOW: Try unzip_member_from_src on the raw src string + parsed URL; if it names
306 * an archive member, route through copy_unzip, otherwise copy_download.
307 */
308static int
309brix_copy_dispatch_r2l(const char *src, const copy_route_t *r,
310 const brix_copy_opts *o, const brix_opts *co,
311 brix_status *st)
312{
313 char member[XRDC_PATH_MAX], arch[XRDC_PATH_MAX];
314
315 if (unzip_member_from_src(src, &r->su, member, sizeof(member),
316 arch, sizeof(arch))) {
317 return copy_unzip(&r->su, arch, member, &r->du, o, co, st); /* ?xrdcl.unzip= */
318 }
319 return copy_download(&r->su, &r->du, o, co, st); /* remote → local */
320}

323/*
324 * WHAT: Dispatch a remote-source → remote-destination copy per the third-party-
325 * copy mode: TPC-only (hard fail if TPC fails), TPC-first/delegate (TPC
326 * then fall back to a client-mediated copy), or default client-mediated.
327 * WHY: The tpc_mode selection was the deepest arm of brix_copy's ladder (the
328 * CCN driver). Extracting it keeps the mode branching — including the
329 * fall-back diagnostic — in one named, testable place while the
330 * orchestrator stays flat.
331 * HOW: Branch on o->tpc_mode exactly as the former inline code: ONLY → copy_tpc;
332 * FIRST/DELEGATE → copy_tpc, on success return, else emit the same stderr
333 * fall-back note (unless silent) and copy_remote_to_remote; otherwise the
334 * default client-mediated copy_remote_to_remote.
335 */
336static int
337brix_copy_dispatch_r2r(const copy_route_t *r, const brix_copy_opts *o,
338 const brix_opts *co, brix_status *st)
339{
340 if (o->tpc_mode == XRDC_TPC_ONLY) {
341 return copy_tpc(&r->su, &r->du, o, co, st); /* TPC or hard fail */
342 }
343 if (o->tpc_mode == XRDC_TPC_FIRST
344 || o->tpc_mode == XRDC_TPC_DELEGATE) {
345 if (copy_tpc(&r->su, &r->du, o, co, st) == 0) {
346 return 0;
347 }
348 if (!o->silent) {
349 fprintf(stderr, "xrdcp: TPC failed (%s); falling back to "
350 "client-mediated copy\n", st->msg);
351 }
352 return copy_remote_to_remote(&r->su, &r->du, o, co, st);
353 }
354 return copy_remote_to_remote(&r->su, &r->du, o, co, st); /* default: client-mediated */
355}

358/*
359 * WHAT: Dispatch a recursive (-r) copy: one remote↔local direction to
360 * copy_recursive, or a usage error for any other endpoint pairing.
361 * WHY: The recursive arm carried three of brix_copy_route's decisions on its
362 * own; extracting it drops the router under the CCN gate and keeps the
363 * "one remote + one local" rule in one named place (§1).
364 * HOW: Build the copy_recurse_req for the download (remote→local) or upload
365 * (local→remote) direction exactly as the former inline code did; any other
366 * pairing (both-remote / both-local) is the same XRDC_EUSAGE error.
367 */
368static int
369brix_copy_dispatch_recursive(const copy_route_t *r, const brix_copy_opts *o,
370 const brix_opts *co, brix_status *st)
371{
372 if (r->src_remote && r->dst_local) {
373 copy_recurse_req rq = { &r->su, &r->du, 1, o, co };
374 return copy_recursive(&rq, st); /* remote tree → local */
375 }
376 if (r->src_local && r->dst_remote) {
377 copy_recurse_req rq = { &r->su, &r->du, 0, o, co };
378 return copy_recursive(&rq, st); /* local tree → remote */
379 }
380 brix_status_set(st, XRDC_EUSAGE, 0,
381 "recursive copy requires one remote and one local endpoint");
382 return -1;
383}

386/*
387 * WHAT: Route a classified root:// copy to the matching transfer path
388 * (recursive, --zip store, download, upload, or remote-to-remote).
389 * WHY: Holds the direction ladder itself, separated from URL classification and
390 * the co_local/web/block pre-steps in brix_copy, so each concern is one
391 * function (§1) and the ladder never re-derives the scheme predicates.
392 * HOW: Early-return through the same ordered checks the former inline code used
393 * — recursive first, then --zip store, then the three supported directions
394 * via the r2l/r2r dispatch helpers and copy_upload — ending in the
395 * unsupported-direction usage error.
396 */
397static int
398brix_copy_route(const char *src, const copy_route_t *r, const brix_copy_opts *o,
399 const brix_opts *co, brix_status *st)
400{
401 if (o != NULL && o->recursive) {
402 return brix_copy_dispatch_recursive(r, o, co, st);
403 }
404
405 /* phase-42 W3 write: --zip / --zip-append stores a LOCAL source as a STORE
406 * member of the destination ZIP archive (local or remote dst). */
407 if (o != NULL && (o->zip || o->zip_append)) {
408 if (!r->src_local) {
409 brix_status_set(st, XRDC_EUSAGE, 0,
410 "--zip requires a local-file source");
411 return -1;
412 }
413 return copy_zip_store(&r->su, &r->du, o, co, st);
414 }
415
416 if (r->src_remote && r->dst_local) {
417 return brix_copy_dispatch_r2l(src, r, o, co, st); /* remote → local */
418 }
419 if (r->src_local && r->dst_remote) {
420 return copy_upload(&r->su, &r->du, o, co, st); /* local → remote */
421 }
422 if (r->src_remote && r->dst_remote) {
423 return brix_copy_dispatch_r2r(r, o, co, st);
424 }
425
426 brix_status_set(st, XRDC_EUSAGE, 0,
427 "unsupported copy direction (local→local not supported)");
428 return -1;
429}

432int
433brix_copy(const char *src, const char *dst, const brix_copy_opts *o,
434 const brix_opts *co_in, brix_status *st)
435{
436 copy_route_t route;
437
438 /* copy.c manages its OWN reconnect/retry — resilient_setup() for the multi-RTT
439 * bring-up and the read pump (pump_src_remote) for mid-transfer severs. Disable
440 * the library's op-level baked resilience on the connections this path owns, so
441 * stat/dirlist/mkdir don't double-retry inside copy's already-bounded loops. */
442 brix_opts co_local;
443 if (co_in != NULL) {
444 co_local = *co_in;
445 } else {
446 memset(&co_local, 0, sizeof(co_local));
447 co_local.verify_host = 1;
448 }
449 co_local.no_retry = 1;
450 const brix_opts *co = &co_local;
451
452 /* Web schemes (davs/http(s)/s3) take the HTTP transfer path, never the root://
453 * session machinery. Check before brix_url_parse (which is root-only). */
454 if (brix_is_web_url(src) || brix_is_web_url(dst)) {
455 return copy_web(src, dst, o, co, st);
456 }
457
458 /* block:// (and /dev/) endpoints route through the VFS block backend.
459 * brix_url_parse does not know the block:// scheme and would reject it,
460 * so intercept here before the parse. */
461 if (brix_is_block_url(src) || brix_is_block_url(dst)) {
462 return copy_block(src, dst, o, co, st);
463 }
464
465 if (brix_copy_classify(src, dst, &route, st) != 0) {
466 return -1;
467 }
468 return brix_copy_route(src, &route, o, co, st);
469}
