7/*
8 * WHAT: Internal declarations for the transparent XRootD proxy module — state machine enums,
9 * per-connection context struct, file handle map entry, pooled connection metadata, upstream
10 * health status, constants, and internal function signatures organized by source file.
11 * This header bridges proxy.h (public API) with implementation files across connect.c, events.c,
12 * forward.c, pool.c, and their sub-fragments.
13 *
14 * WHY: The proxy operates a multi-phase upstream connection lifecycle (connecting → TLS handshake →
15 * bootstrap → idle → forwarding) tracked via brix_proxy_up_state_t + brix_proxy_bs_t enums.
16 * Each client-proxy session maintains a full context struct with write buffers, response accumulators,
17 * fh translation map, lazy-open queue, wait-retry state, redirect follow-through, and splice zero-copy
18 * plumbing. Pooled connections carry auth type/token hash/upstream index metadata for matching reuse.
19 * Health status per upstream enables fail detection and automatic skip of DOWN servers.
20 * Constants define pool size limits, keepalive intervals, retry budgets, and path length caps.
21 *
22 * HOW: Structs defined inline with field comments explaining each member's purpose. Enums map state phases
23 * to numeric values for efficient comparison in event handlers. Function declarations grouped by source
24 * file (connect.c, events.c, forward.c) with inline WHAT comments where logic is non-obvious. The opaque
25 * typedef brix_proxy_ctx_t references the full struct definition here; public API in proxy.h declares
26 * only the externally visible functions.
27 */

248/* Phase 39 (PXY-3): release the upstream write buffer once its send has fully
249 * completed. Frees it iff it is heap-owned (a forwarded/relayed request from
250 * ngx_alloc); pool-allocated bootstrap frames are merely detached (the pool owns
251 * them). Idempotent and NULL-safe. Call this on EVERY send-complete path so a
252 * deferred (backpressured / slow-consumer) request is not leaked. */
253static ngx_inline void
254brix_proxy_wbuf_release(brix_proxy_ctx_t *proxy)
255{
256 if (proxy->wbuf_owned && proxy->wbuf != NULL) {
257 ngx_free(proxy->wbuf);
258 }
259 proxy->wbuf = NULL;
260 proxy->wbuf_owned = 0;
261}
