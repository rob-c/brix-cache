50ngx_int_t
51brix_protocol_register(const brix_protocol_t *p)
52{
53 ngx_uint_t i;
54
55 if (p == NULL || p->name == NULL) {
56 return NGX_ERROR;
57 }
58 for (i = 0; i < brix_protocol_count; i++) {
59 if (ngx_strcmp(brix_protocols[i]->name, p->name) == 0) {
60 return NGX_OK; /* already registered — idempotent */
61 }
62 }
63 if (brix_protocol_count >= BRIX_PROTOCOL_MAX) {
64 return NGX_ERROR;
65 }
66 brix_protocols[brix_protocol_count++] = p;
67 return NGX_OK;
68}
