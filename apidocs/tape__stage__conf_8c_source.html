94/* Parse "0.95" / "95%" / "95" into parts-per-million, clamped to [0, 1e6]. */
95static ngx_uint_t
96frm_ratio_to_ppm(const ngx_str_t *s)
97{
98 char buf[32];
99 double v;
100 size_t n = s->len;
101
102 if (n == 0 || n >= sizeof(buf)) {
103 return 0;
104 }
105 ngx_memcpy(buf, s->data, n);
106 buf[n] = '\0';
107 v = strtod(buf, NULL);
108 if (buf[n - 1] == '%' || v > 1.0) { /* a percentage like "95" or "95%" */
109 v /= 100.0;
110 }
111 if (v < 0.0) { v = 0.0; }
112 if (v > 1.0) { v = 1.0; }
113 return (ngx_uint_t) (v * 1000000.0 + 0.5);
114}

116char *
117brix_frm_set_purge_watermark(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
118{
119 brix_frm_conf_t *frm = (brix_frm_conf_t *)
120 ((char *) conf + cmd->offset);
121 ngx_str_t *value = cf->args->elts;
122
123 frm->purge_hi_ppm = frm_ratio_to_ppm(&value[1]);
124 frm->purge_lo_ppm = frm_ratio_to_ppm(&value[2]);
125
126 if (frm->purge_lo_ppm > frm->purge_hi_ppm) {
127 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
128 "brix_frm_purge_watermark: low (%ui) must not exceed high (%ui)",
129 frm->purge_lo_ppm, frm->purge_hi_ppm);
130 return NGX_CONF_ERROR;
131 }
132 return NGX_CONF_OK;
133}
