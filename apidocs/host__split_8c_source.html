12/* ---- Validate and store a decimal port string ----
13 *
14 * WHAT: Parses `port_str` as a decimal port, storing it into `*port` and
15 * returning 0 on success; returns -1 when the value is out of the valid
16 * 1..65535 range (or non-numeric, which atoi maps to 0).
17 *
18 * WHY: The bracketed and plain paths both accept an explicit port with the
19 * identical accept/reject rule; centralising it keeps that boundary
20 * byte-for-byte consistent and out of the two split helpers.
21 *
22 * HOW:
23 * 1. Convert with atoi (non-numeric leading input yields 0).
24 * 2. Reject values <= 0 or > 65535.
25 * 3. Publish the accepted value to `*port` and return 0.
26 */
27static int
28brix_parse_port_str(const char *port_str, int *port)
29{
30 int p = atoi(port_str);
31 if (p <= 0 || p > 65535) {
32 return -1;
33 }
34 *port = p;
35 return 0;
36}

38/* ---- Copy a bounded host substring into the caller's buffer ----
39 *
40 * WHAT: Copies `hlen` bytes from `src` into `host` and NUL-terminates it,
41 * returning 0 on success; returns -1 when `hlen` is empty or would not
42 * fit (including the terminator) in `hsz`.
43 *
44 * WHY: Every parse path applies the same empty-and-overflow guard before the
45 * copy; sharing it preserves the exact length boundaries and avoids
46 * duplicating the memcpy/terminate idiom.
47 *
48 * HOW:
49 * 1. Reject an empty host or one whose length is >= the buffer size (leaving
50 * no room for the terminator).
51 * 2. memcpy the bytes and write the trailing NUL.
52 */
53static int
54brix_copy_host(char *host, size_t hsz, const char *src, size_t hlen)
55{
56 if (hlen == 0 || hlen >= hsz) {
57 return -1;
58 }
59 memcpy(host, src, hlen);
60 host[hlen] = '\0';
61 return 0;
62}

64/* ---- Parse a bracketed IPv6 authority "[addr]" or "[addr]:port" ----
65 *
66 * WHAT: Extracts the bracket-delimited literal into `host` and resolves the
67 * trailing port (explicit or `default_port`), returning 0 on success and
68 * -1 for an unterminated literal, an unusable host length, a bad port,
69 * or junk after the closing bracket.
70 *
71 * WHY: Brackets disambiguate the address colons from the port colon; isolating
72 * this case keeps that special IPv6 handling separate from the plain
73 * host:port rule.
74 *
75 * HOW:
76 * 1. Locate the closing ']'; a missing one is an unterminated literal.
77 * 2. Copy the inner literal (length excludes both brackets) into `host`.
78 * 3. If ':' follows ']', parse the explicit port; if the string ends there,
79 * use `default_port`; anything else after ']' is junk and rejected.
80 */
81static int
82brix_split_bracketed(const char *in, char *host, size_t hsz, int *port,
83 int default_port)
84{
85 const char *rb = strchr(in, ']');
86 if (rb == NULL) {
87 return -1; /* unterminated literal */
88 }
89 if (brix_copy_host(host, hsz, in + 1, (size_t) (rb - in - 1)) != 0) {
90 return -1;
91 }
92 if (rb[1] == ':') {
93 return brix_parse_port_str(rb + 2, port);
94 }
95 if (rb[1] == '\0') {
96 *port = default_port;
97 return 0;
98 }
99 return -1; /* junk after ']' */
100}

102/* ---- Parse a plain authority "host" or "host:port" ----
103 *
104 * WHAT: Splits `in` on the last ':' into `host` and a port (or uses
105 * `default_port` when no colon is present), returning 0 on success and
106 * -1 for an unusable host length or a bad port.
107 *
108 * WHY: Handles the non-bracketed case, where the rightmost colon separates the
109 * port so that bare hostnames and IPv4 literals parse correctly.
110 *
111 * HOW:
112 * 1. Find the last ':' with strrchr.
113 * 2. With a colon: copy the leading host and parse the trailing port.
114 * 3. Without a colon: copy the whole string as host and apply
115 * `default_port`.
116 */
117static int
118brix_split_plain(const char *in, char *host, size_t hsz, int *port,
119 int default_port)
120{
121 const char *colon = strrchr(in, ':');
122 if (colon != NULL) {
123 if (brix_copy_host(host, hsz, in, (size_t) (colon - in)) != 0) {
124 return -1;
125 }
126 return brix_parse_port_str(colon + 1, port);
127 }
128 if (brix_copy_host(host, hsz, in, strlen(in)) != 0) {
129 return -1;
130 }
131 *port = default_port;
132 return 0;
133}

135/* ---- Parse "host[:port]" / "[ipv6][:port]" into host and port ----
136 *
137 * WHAT: Validates the arguments then routes to the bracketed-IPv6 or plain
138 * authority parser, writing the host into `host`/`hsz` and the resolved
139 * port into `*port`; returns 0 on success and -1 on any malformed input.
140 *
141 * WHY: The single entry point the native client relies on for authority
142 * parsing; a leading '[' selects the IPv6-bracket grammar, everything else
143 * the plain grammar.
144 *
145 * HOW:
146 * 1. Reject NULL pointers and a zero-sized host buffer.
147 * 2. A leading '[' dispatches to the bracketed parser.
148 * 3. Otherwise dispatch to the plain host:port parser.
149 */
150int
151brix_split_host_port(const char *in, char *host, size_t hsz, int *port,
152 int default_port)
153{
154 if (in == NULL || host == NULL || hsz == 0 || port == NULL) {
155 return -1;
156 }
157
158 /* Bracketed IPv6 literal: "[::1]" or "[::1]:port" — the brackets disambiguate
159 * the address colons from the port colon, and are stripped from `host`. */
160 if (in[0] == '[') {
161 return brix_split_bracketed(in, host, hsz, port, default_port);
162 }
163
164 return brix_split_plain(in, host, hsz, port, default_port);
165}
