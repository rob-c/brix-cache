1/*
2 * uring_internal.h — declarations shared across the io_uring backend files
3 * after the phase-79 file-size split.
4 *
5 * WHAT: Cross-declares the two entry points and the one scan struct that cross
6 * file boundaries between uring.c (ring singleton + bring-up/lifecycle),
7 * uring_probe.c (gating + capability probe + startup validation), and
8 * uring_reap.c (completion reaper + completion-slot table).
9 * WHY: uring.c (1033 lines) split into four focused files under the 500-line
10 * cap: uring.c keeps the per-worker ring singleton and the init/exit
11 * lifecycle orchestrator; uring_bringup.c owns the ordered register-phase
12 * bring-up steps (create/eventfd/restrict/enable/self-tests/epoll-bridge/
13 * slot-table); uring_probe.c owns detection, per-block gating, and the
14 * §32 fail-fast validator; uring_reap.c owns the eventfd-driven CQE
15 * reaper and the UAF-safe slot table. Exactly the symbols that cross
16 * those boundaries become non-static and are declared here — nothing else.
17 * HOW: All three .c files include this header. Its contents are gated on
18 * BRIX_HAVE_LIBURING because the scan struct and both entry points exist
19 * only in a liburing build; a stub build never reaches them.
20 *
21 * Requires: core/ngx_brix_module.h and uring.h included before this header.
22 */

31/* Merged per-worker io_uring demand, folded across every enabled server block
32 * by uring_scan_conf() (uring_probe.c) and consumed by uring_probe_features()
33 * (uring_probe.c) and brix_uring_init_worker() (uring.c). */
34typedef struct {
35 ngx_uint_t mode_on; /* any block requires io_uring (mode `on`) */
36 ngx_uint_t want_restrict; /* 0 iff any wanting block disabled it */
37 ngx_int_t depth; /* max requested queue depth (pre-clamp) */
38 ngx_str_t panic_file; /* first configured kill-switch path */
39} brix_uring_scan_t;

41/* Defined in uring_probe.c. Folds the configuration and the kernel capability
42 * probe into one go/no-go verdict for this worker, clamping the queue depth.
43 * Returns NGX_OK to proceed with bring-up, NGX_DECLINED to skip it. */
44ngx_int_t uring_probe_features(ngx_cycle_t *cycle, brix_uring_scan_t *scan);
45
46/* Defined in uring_reap.c. The completion reaper: the read handler of the fake
47 * connection wrapping the ring's registered eventfd (installed by
48 * uring_install_eventfd). Drains the eventfd counter, then harvests every ready
49 * CQE. */
50void brix_uring_eventfd_handler(ngx_event_t *ev);

52/* Register-phase bring-up steps, defined in uring_bringup.c and driven in order
53 * by brix_uring_init_worker() (uring.c). Each mutates *u as its step lands and
54 * returns NULL on success or the failing step's name for brix_uring_init_fail. */
55const char *uring_setup_rings(brix_uring_t *u, ngx_uint_t want_restrict);
56const char *uring_selftest_nop(brix_uring_t *u);
57const char *uring_selftest_burst(brix_uring_t *u);

59const char *uring_register_buffers(brix_uring_t *u, ngx_cycle_t *cycle);
