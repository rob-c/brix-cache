36/*
37 * Deflate inflate state for a method-8 member (stored in fh->zip_inflate).
38 *
39 * ZIP members carry RAW deflate (RFC 1951, no zlib/gzip wrapper), so the stream
40 * is initialised with windowBits = -15 — codec_core's DEFLATE codec is zlib-
41 * wrapped (RFC 1950) and would not decode it. `cin` is a persistent input
42 * buffer so zlib's next_in stays valid across read calls (a stack buffer would
43 * dangle). The member's declared comp_size/uncomp_size bound input and output,
44 * so they ARE the decompression-bomb guard — we never feed past comp_size nor
45 * emit past uncomp_size.
46 */
47typedef struct {
48 z_stream zs;
49 int inited;
50 uint64_t in_fed; /* compressed bytes pread from the archive so far */
51 u_char cin[64 * 1024]; /* persistent input buffer (zlib next_in target) */
52} zip_infl_t;
53
54#define ZIP_INFL_SCRATCH (64 * 1024)

56static void
57zip_deflate_free(brix_file_t *fh)
58{
59 zip_infl_t *st = fh->zip_inflate;
60 if (st == NULL) {
61 return;
62 }
63 if (st->inited) {
64 inflateEnd(&st->zs);
65 }
66 free(st);
67 fh->zip_inflate = NULL;
68}

70/* (Re)start the inflate stream at the member's beginning. 0 / -1. */
71static int
72zip_deflate_reset(brix_file_t *fh)
73{
74 zip_infl_t *st;
75
76 zip_deflate_free(fh);
77 st = calloc(1, sizeof(*st));
78 if (st == NULL) {
79 return -1;
80 }
81 if (inflateInit2(&st->zs, -15) != Z_OK) { /* raw deflate */
82 free(st);
83 return -1;
84 }
85 st->inited = 1;
86 st->in_fed = 0;
87 fh->zip_inflate = st;
88 fh->zip_comp_pos = 0;
89 fh->zip_logical_pos = 0;
90 return 0;
91}

93/*
94 * Produce up to `want` uncompressed bytes into out (out == NULL → discard, for a
95 * forward skip). Returns bytes produced (may be < want at end-of-stream), or -1
96 * on a decode/IO error. Advances fh->zip_logical_pos and fh->zip_comp_pos.
97 */
98static ssize_t
99zip_deflate_pump(brix_file_t *fh, u_char *out, size_t want)
100{
101 zip_infl_t *st = fh->zip_inflate;
102 u_char scratch[ZIP_INFL_SCRATCH];
103 size_t produced = 0;
104
105 while (produced < want) {
106 size_t ocap;
107 u_char *o;
108 uInt in_before, out_before;
109 int zr;
110
111 /* Refill compressed input only once zlib has drained the buffer. */
112 if (st->zs.avail_in == 0 && st->in_fed < fh->zip_comp_size) {
113 uint64_t remain = fh->zip_comp_size - st->in_fed;
114 size_t cn = remain < sizeof(st->cin) ? (size_t) remain
115 : sizeof(st->cin);
116 brix_sd_obj_t obj;
117 brix_sd_posix_wrap(&obj, fh->fd);
118 ssize_t r = obj.driver->pread(&obj, st->cin, cn,
119 (off_t) (fh->zip_data_off + st->in_fed));
120 if (r <= 0) {
121 return -1;
122 }
123 st->zs.next_in = st->cin;
124 st->zs.avail_in = (uInt) r;
125 st->in_fed += (uint64_t) r;
126 fh->zip_comp_pos = st->in_fed;
127 }
128
129 o = out ? out + produced : scratch;
130 ocap = want - produced;
131 if (out == NULL && ocap > sizeof(scratch)) {
132 ocap = sizeof(scratch);
133 }
134
135 in_before = st->zs.avail_in;
136 out_before = (uInt) ocap;
137 st->zs.next_out = o;
138 st->zs.avail_out = (uInt) ocap;
139
140 zr = inflate(&st->zs, Z_NO_FLUSH);
141
142 produced += out_before - st->zs.avail_out;
143 fh->zip_logical_pos += out_before - st->zs.avail_out;
144
145 if (zr == Z_STREAM_END) {
146 break;
147 }
148 if (zr != Z_OK && zr != Z_BUF_ERROR) {
149 return -1; /* corrupt deflate stream */
150 }
151 /* No forward progress and no more input → truncated member; stop. */
152 if (st->zs.avail_in == in_before
153 && (out_before - st->zs.avail_out) == 0
154 && st->in_fed >= fh->zip_comp_size)
155 {
156 break;
157 }
158 }
159 return (ssize_t) produced;
160}

162/*
163 * Serve a deflate member read at `offset` for `want` bytes into out[want].
164 * Sequential reads reuse the stream; a backward seek re-inflates from the start;
165 * a forward seek inflates-and-discards up to the offset. Returns bytes produced
166 * or -1 on error.
167 */
168static ssize_t
169zip_deflate_read(brix_file_t *fh, uint64_t offset, u_char *out, size_t want)
170{
171 if (fh->zip_inflate == NULL || offset < fh->zip_logical_pos) {
172 if (zip_deflate_reset(fh) != 0) {
173 return -1;
174 }
175 }
176 while (offset > fh->zip_logical_pos) {
177 uint64_t skip = offset - fh->zip_logical_pos;
178 size_t chunk = skip < ZIP_INFL_SCRATCH ? (size_t) skip
179 : ZIP_INFL_SCRATCH;
180 ssize_t got = zip_deflate_pump(fh, NULL, chunk);
181 if (got <= 0) {
182 return -1; /* cannot reach the requested offset */
183 }
184 }
185 return zip_deflate_pump(fh, out, want);
186}

188/* Read exactly n bytes at off into buf; total read on success, -1 on error. */
189static ssize_t
190zip_pread_full(int fd, u_char *buf, size_t n, off_t off)
191{
192 size_t done = 0;
193 brix_sd_obj_t obj;
194 brix_sd_posix_wrap(&obj, fd);
195 while (done < n) {
196 ssize_t r = obj.driver->pread(&obj, buf + done, n - done,
197 off + (off_t) done);
198 if (r < 0) {
199 if (errno == EINTR) {
200 continue;
201 }
202 return -1;
203 }
204 if (r == 0) {
205 break; /* short file — return what we got */
206 }
207 done += (size_t) r;
208 }
209 return (ssize_t) done;
210}

212ngx_int_t
213brix_zip_open_member(brix_ctx_t *ctx, ngx_connection_t *c,
214 ngx_stream_brix_srv_conf_t *conf, const char *archive_logical,
215 const char *archive_full, const char *member, uint16_t options)
216{
217 int fd, idx, zrc;
218 struct stat ast;
219 brix_zip_member_t m;
220 size_t cd_max;
221
222 /* Open the archive read-only, confined beneath the export root (same
223 * RESOLVE_BENEATH path the normal read open uses). */
224 fd = brix_vfs_open_fd_at(conf->rootfd, archive_logical,
225 O_RDONLY | O_CLOEXEC, 0);
226 if (fd < 0) {
227 int err = errno;
228 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN", archive_full,
229 "zip",
230 (err == ENOENT || err == ENOTDIR) ? kXR_NotFound
231 : (err == EACCES) ? kXR_NotAuthorized : kXR_IOError,
232 "cannot open zip archive");
233 }
234 if (fstat(fd, &ast) != 0 || !S_ISREG(ast.st_mode)) {
235 close(fd);
236 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN", archive_full,
237 "zip", kXR_IOError, "zip archive not a regular file");
238 }
239
240 /* CONSUME via the materialize-to-scratch seam: zip is built on random-access
241 * pread over the archive fd (+ sendfile for stored members), which a backend
242 * with no kernel fd cannot serve. When staging is in effect, copy the archive
243 * into a local POSIX scratch (anonymous fd) and read THAT. Capability + size
244 * gated; POSIX is a no-op (read the confined fd in place). Config:
245 * brix_zip_stage_dir + brix_zip_force_scratch + brix_zip_stage_max_bytes. */
246 {
247 const char *sdir = (conf->zip_stage_dir.len > 0)
248 ? (const char *) conf->zip_stage_dir.data : NULL;
249 off_t maxb = (off_t) conf->zip_stage_max_bytes;
250
251 /* The default POSIX archive already has a kernel fd — only materialize a
252 * local scratch copy when explicitly forced (a backend without CAP_FD is
253 * the future case, but this open path only sees POSIX archives today). */
254 if (conf->zip_force_scratch
255 && sdir != NULL && sdir[0] != '\0'
256 && ast.st_size <= maxb)
257 {
258 ngx_fd_t sfd = xvfs_stage_fd(fd, sdir);
259 if (sfd == NGX_INVALID_FILE) {
260 close(fd);
261 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN", archive_full,
262 "zip", kXR_IOError, "zip archive staging failed");
263 }
264 close(fd);
265 fd = sfd; /* same bytes/size — ast still valid */
266 ngx_log_error(NGX_LOG_INFO, c->log, 0,
267 "zip: archive staged to scratch (%O bytes)",
268 (off_t) ast.st_size);
269 }
270 }
271
272 cd_max = conf->zip_cd_max_bytes ? conf->zip_cd_max_bytes
273 : (size_t) (16 * 1024 * 1024);
274 zrc = brix_zip_find_member(fd, (off_t) ast.st_size, member, cd_max, &m);
275 if (zrc != BRIX_ZIP_OK) {
276 close(fd);
277 if (zrc == BRIX_ZIP_NOMEMBER) {
278 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN", archive_full,
279 "zip", kXR_NotFound, "zip member not found");
280 }
281 /* ECORRUPT covers corrupt/oversize/unsupported (encrypted, method,
282 * data-descriptor); EIO covers pread/alloc failure. */
283 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN", archive_full,
284 "zip",
285 (zrc == BRIX_ZIP_EIO) ? kXR_IOError : kXR_Unsupported,
286 "zip member unreadable");
287 }
288
289 idx = brix_alloc_fhandle(ctx);
290 if (idx < 0) {
291 close(fd);
292 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN", archive_full,
293 "zip", kXR_ServerError, "too many open files");
294 }
295
296 ctx->files[idx].fd = fd;
297 ctx->files[idx].readable = 1;
298 ctx->files[idx].writable = 0;
299 ctx->files[idx].from_cache = 0;
300 ctx->files[idx].is_regular = 1; /* member reads behave as a file */
301 ctx->files[idx].device = ast.st_dev;
302 ctx->files[idx].inode = ast.st_ino;
303 ctx->files[idx].cached_size = (off_t) m.uncomp_size;
304 ctx->files[idx].read_last_end = -1;
305 ctx->files[idx].read_ahead_end = 0;
306 ctx->files[idx].read_codec = (uint8_t) BRIX_CODEC_IDENTITY;
307 ctx->files[idx].write_codec = (uint8_t) BRIX_CODEC_IDENTITY;
308 ctx->files[idx].dashboard_slot = -1;
309
310 ctx->files[idx].zip_mode = 1;
311 ctx->files[idx].zip_method = m.method;
312 ctx->files[idx].zip_data_off = m.data_off;
313 ctx->files[idx].zip_comp_size = m.comp_size;
314 ctx->files[idx].zip_uncomp_size = m.uncomp_size;
315 ctx->files[idx].zip_crc32 = m.crc32;
316 ctx->files[idx].zip_inflate = NULL;
317 ctx->files[idx].zip_logical_pos = 0;
318 ctx->files[idx].zip_comp_pos = 0;
319
320 if (brix_set_fhandle_path(ctx, c, idx, archive_full) != NGX_OK) {
321 brix_free_fhandle(ctx, idx);
322 return NGX_ERROR;
323 }
324
325 ctx->files[idx].bytes_read = 0;
326 ctx->files[idx].bytes_written = 0;
327 ctx->files[idx].open_time = ngx_current_msec;
328
329 if (!ctx->is_bound) {
330 brix_session_handle_publish(ctx->login.sessid, idx, &ctx->files[idx]);
331 }
332
333 /* Build the kXR_open reply: the 4-byte handle by default, the full body +
334 * stat (member's UNCOMPRESSED size) when the client set kXR_retstat. */
335 {
336 ServerOpenBody body;
337 char statbuf[256];
338 u_char *buf;
339 size_t hbytes, bodylen, total;
340 ngx_flag_t want_stat = (options & kXR_retstat) ? 1 : 0;
341
342 ngx_memzero(&body, sizeof(body));
343 body.fhandle[0] = (u_char) idx;
344
345 statbuf[0] = '\0';
346 if (want_stat) {
347 snprintf(statbuf, sizeof(statbuf), "%llu %lld %d %ld",
348 (unsigned long long) ast.st_ino,
349 (long long) m.uncomp_size,
350 kXR_readable,
351 (long) ast.st_mtime);
352 }
353
354 hbytes = want_stat ? sizeof(ServerOpenBody) : sizeof(body.fhandle);
355 bodylen = hbytes + (want_stat ? strlen(statbuf) + 1 : 0);
356 total = XRD_RESPONSE_HDR_LEN + bodylen;
357
358 buf = ngx_palloc(c->pool, total);
359 if (buf == NULL) {
360 brix_free_fhandle(ctx, idx);
361 return NGX_ERROR;
362 }
363 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok, (uint32_t) bodylen,
364 (ServerResponseHdr *) buf);
365 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body, hbytes);
366 if (want_stat) {
367 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + sizeof(ServerOpenBody),
368 statbuf, strlen(statbuf) + 1);
369 }
370
371 brix_log_access(ctx, c, "OPEN", archive_full, "zip", 1, 0, NULL, 0);
372 BRIX_OP_OK(ctx, BRIX_OP_OPEN_RD);
373 return brix_queue_response(ctx, c, buf, total);
374 }

377ngx_int_t
378brix_zip_read(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
379 int64_t offset, size_t rlen)
380{
381 brix_file_t *fh = &ctx->files[idx];
382 u_char *buf;
383 ssize_t n;
384
385 /* Past EOF → empty (matches a normal read beyond file size). */
386 if ((uint64_t) offset >= fh->zip_uncomp_size) {
387 BRIX_OP_OK(ctx, BRIX_OP_READ);
388 return brix_send_ok(ctx, c, NULL, 0);
389 }
390 if ((uint64_t) offset + rlen > fh->zip_uncomp_size) {
391 rlen = (size_t) (fh->zip_uncomp_size - (uint64_t) offset);
392 }
393
394 if (fh->zip_method == BRIX_ZIP_METHOD_STORE) {
395 buf = ngx_palloc(c->pool, rlen);
396 if (buf == NULL) {
397 return brix_send_error(ctx, c, kXR_NoMemory, "zip read OOM");
398 }
399 n = zip_pread_full(fh->fd, buf, rlen,
400 (off_t) (fh->zip_data_off + (uint64_t) offset));
401 if (n < 0) {
402 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
403 fh->path ? fh->path : "-", "zip",
404 kXR_IOError, "zip member read failed");
405 }
406 fh->bytes_read += (size_t) n;
407 BRIX_OP_OK(ctx, BRIX_OP_READ);
408 return brix_send_ok(ctx, c, buf, (size_t) n);
409 }
410
411 /* Deflate (method 8): streaming raw inflate. */
412 buf = ngx_palloc(c->pool, rlen);
413 if (buf == NULL) {
414 return brix_send_error(ctx, c, kXR_NoMemory, "zip read OOM");
415 }
416 n = zip_deflate_read(fh, (uint64_t) offset, buf, rlen);
417 if (n < 0) {
418 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
419 fh->path ? fh->path : "-", "zip",
420 kXR_IOError, "zip member inflate failed");
421 }
422 fh->bytes_read += (size_t) n;
423 BRIX_OP_OK(ctx, BRIX_OP_READ);
424 return brix_send_ok(ctx, c, buf, (size_t) n);
425}
