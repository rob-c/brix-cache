1/*
2 * disconnect_report.c — metrics finalization + access-log reporting for
3 * root:// connection teardown (split from disconnect.c, phase-79).
4 *
5 * WHAT: The reporting half of connection teardown — finalizes the session
6 * metrics (connections_active, rx/tx byte totals) and emits the
7 * access-log records (a kXR_Cancelled line per still-open handle and the
8 * session-level throughput detail). The four public entry points
9 * (brix_disconnect_update_metrics, _sess_reason, _log_open_files,
10 * _format_session_detail) are invoked by brix_on_disconnect()
11 * (disconnect.c); brix_disconnect_file_bytes stays private here.
12 * WHY: disconnect.c owns the teardown ORDER and resource release; this file
13 * owns what the teardown REPORTS. Separating the two keeps each half
14 * focused and under the 500-line file-size cap. These helpers are pure
15 * reporting — they read ctx counters and emit log/metric side effects,
16 * with no bearing on cleanup ordering.
17 * HOW: Each helper takes ctx (and c/now where needed) explicitly and returns
18 * void, mirroring the resource-release helpers in disconnect.c. The
19 * cross-file entry points are declared in disconnect_internal.h.
20 */

41/* Finalize session metrics: decrement connections_active and accumulate the
42 * lifecycle rx/tx byte totals (with per-IP-version and per-protocol breakdowns),
43 * which must be committed here before ctx is destroyed. */
44
45void
46brix_disconnect_update_metrics(brix_ctx_t *ctx)
47{
48 if (ctx->metrics == NULL) {
49 return;
50 }
51
52 ngx_atomic_fetch_add(&ctx->metrics->connections_active,
53 (ngx_atomic_int_t) -1);
54 ngx_atomic_fetch_add(&ctx->metrics->bytes_rx_total,
55 (ngx_atomic_int_t) ctx->totals.bytes_written);
56 ngx_atomic_fetch_add(&ctx->metrics->bytes_tx_total,
57 (ngx_atomic_int_t) ctx->totals.bytes);
58
59 /* Per-IP-version byte accounting — mirrors the aggregate totals above. */
60 if (ctx->ip_version == AF_INET6) {
61 ngx_atomic_fetch_add(&ctx->metrics->bytes_rx_ipv6_total,
62 (ngx_atomic_int_t) ctx->totals.bytes_written);
63 ngx_atomic_fetch_add(&ctx->metrics->bytes_tx_ipv6_total,
64 (ngx_atomic_int_t) ctx->totals.bytes);
65 } else if (ctx->ip_version == AF_INET) {
66 ngx_atomic_fetch_add(&ctx->metrics->bytes_rx_ipv4_total,
67 (ngx_atomic_int_t) ctx->totals.bytes_written);
68 ngx_atomic_fetch_add(&ctx->metrics->bytes_tx_ipv4_total,
69 (ngx_atomic_int_t) ctx->totals.bytes);
70 }
71
72 /* Per-protocol byte accounting for native stream-layer traffic. */
73 if (ngx_strcmp(ctx->protocol_label, "root") == 0) {
74 ngx_atomic_fetch_add(&ctx->metrics->proto_root_bytes_rx_total,
75 (ngx_atomic_int_t) ctx->totals.bytes_written);
76 ngx_atomic_fetch_add(&ctx->metrics->proto_root_bytes_tx_total,
77 (ngx_atomic_int_t) ctx->totals.bytes);
78 }
79}

81/* Dominant byte count for an open handle in disconnect logging: bytes_written when
82 * any write occurred (upload), else bytes_read — so an interrupted upload is not
83 * logged as a zero-byte read. */
84
85static size_t
86brix_disconnect_file_bytes(const brix_file_t *file)
87{
88 /*
89 * A handle is either read-heavy or write-heavy in the access log. Prefer
90 * written bytes when present so interrupted uploads are not reported as
91 * zero-byte reads.
92 */
93 if (file->bytes_written > 0) {
94 return file->bytes_written;
95 }
96
97 return file->bytes_read;
98}

100/*
101 * WHAT: Derive the sesslog END reason for root connection teardown.
102 * WHY: END lines need a stable low-cardinality reason even though many nginx
103 * callbacks funnel through brix_on_disconnect().
104 * HOW: Prefer an explicit hint, then shutdown, timeout, socket error, and
105 * finally the normal client-disconnect case.
106 */
107brix_sess_end_t
108brix_disconnect_sess_reason(brix_ctx_t *ctx, ngx_connection_t *c)
109{
110 if (ctx->sess_end_hint_set) {
111 return ctx->sess_end_hint;
112 }
113 if (ngx_exiting || ngx_terminate) {
114 return BRIX_SESS_END_SHUTDOWN;
115 }
116 if (c != NULL && c->timedout) {
117 return BRIX_SESS_END_TIMEOUT;
118 }
119 if (c != NULL && c->error) {
120 return BRIX_SESS_END_ERROR;
121 }
122
123 return BRIX_SESS_END_CLIENT;
124}

126/* Emit a kXR_Cancelled access-log entry for every still-open handle when the
127 * connection drops (detail "interrupted X.XXMB/s" or "interrupted"), with duration
128 * measured from the original open_time via req_start reuse. */
129
130void
131brix_disconnect_log_open_files(brix_ctx_t *ctx, ngx_connection_t *c,
132 ngx_msec_t now)
133{
134 int handle_index;
135
136 for (handle_index = 0; handle_index < BRIX_MAX_FILES; handle_index++) {
137 brix_file_t *file;
138 char detail[64];
139 size_t byte_total;
140 ngx_msec_t duration_ms;
141
142 file = &ctx->files[handle_index];
143 if (file->fd < 0) {
144 continue;
145 }
146
147 byte_total = brix_disconnect_file_bytes(file);
148 duration_ms = now - file->open_time;
149
150 if (byte_total > 0 && duration_ms > 0) {
151 double mb_per_second;
152
153 mb_per_second = (double) byte_total / (double) duration_ms
154 / 1000.0;
155 snprintf(detail, sizeof(detail), "interrupted %.2fMB/s",
156 mb_per_second);
157 } else {
158 snprintf(detail, sizeof(detail), "interrupted");
159 }
160
161 /*
162 * Reuse req_start so the common access logger reports per-handle
163 * duration from the original open time, not from disconnect time.
164 */
165 ctx->req_start = file->open_time;
166 brix_log_access(ctx, c, "CLOSE", file->path, detail, 0,
167 kXR_Cancelled, "connection lost", byte_total);
168 if (file->sess_xfer.active && byte_total > file->sess_xfer.bytes) {
169 brix_sess_xfer_add(&file->sess_xfer,
170 (uint64_t) (byte_total - file->sess_xfer.bytes));
171 }
172 brix_sess_xfer_end(ctx->sess, &file->sess_xfer,
173 (ngx_exiting || ngx_terminate)
174 ? BRIX_SESS_XFER_SHUTDOWN
175 : BRIX_SESS_XFER_ABORTED);
176 }
177}

179/* Format session-level throughput for the final access-log entry: separate rx/tx
180 * MB/s when both occurred, otherwise a single aggregate. */
181
182void
183brix_disconnect_format_session_detail(brix_ctx_t *ctx, ngx_msec_t now,
184 char *detail, size_t detail_size, size_t *total_bytes)
185{
186 ngx_msec_t session_duration_ms;
187
188 *total_bytes = ctx->totals.bytes + ctx->totals.bytes_written;
189 session_duration_ms = now - ctx->totals.start;
190
191 if (*total_bytes == 0 || session_duration_ms == 0) {
192 snprintf(detail, detail_size, "-");
193 return;
194 }
195
196 if (ctx->totals.bytes_written > 0) {
197 double read_mbps;
198 double write_mbps;
199
200 read_mbps = (double) ctx->totals.bytes
201 / (double) session_duration_ms / 1000.0;
202 write_mbps = (double) ctx->totals.bytes_written
203 / (double) session_duration_ms / 1000.0;
204 snprintf(detail, detail_size, "rx=%.2fMB/s tx=%.2fMB/s",
205 read_mbps, write_mbps);
206 return;
207 }
208
209 snprintf(detail, detail_size, "%.2fMB/s",
210 (double) *total_bytes / (double) session_duration_ms / 1000.0);
211}
