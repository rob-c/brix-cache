23/* origin_request — send a generic 24-byte ClientRequestHdr (requestid + packed
24 * `body`) plus `payload`, then read the response into (*status, *rbody, *rdlen).
25 * The caller owns *rbody (free it). Returns 0 (response received — check *status)
26 * or -1 on a transport failure. Shared by the namespace/fattr ops below. */
27static int
28origin_request(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
29 uint16_t requestid, const uint8_t body[XRDW_BODY_LEN],
30 const void *payload, size_t plen, uint16_t *status, u_char **rbody,
31 uint32_t *rdlen, size_t rmax)
32{
33 size_t total = sizeof(ClientRequestHdr) + plen;
34 u_char *buf;
35 ClientRequestHdr *req;
36
37 buf = malloc(total);
38 if (buf == NULL) {
39 return -1;
40 }
41 ngx_memzero(buf, sizeof(ClientRequestHdr));
42 req = (ClientRequestHdr *) buf;
43 req->streamid[1] = 8; /* unused stream slot */
44 req->requestid = htons(requestid);
45 ngx_memcpy(req->body, body, XRDW_BODY_LEN);
46 req->dlen = htonl((kXR_int32) plen);
47 if (plen > 0) {
48 ngx_memcpy(buf + sizeof(ClientRequestHdr), payload, plen);
49 }
50
51 if (brix_cache_io_send(oc, buf, total) != 0) {
52 free(buf);
53 return -1;
54 }
55 free(buf);
56
57 *rbody = NULL;
58 return brix_cache_read_response(t, oc, status, rbody, rdlen, rmax);
59}

61/* Map a non-ok origin response to errno. A failure is a kXR_error frame whose
62 * body is [int32 errnum][msg]; decode the kXR errnum (kXR_NotFound, …) from it.
63 * (Some servers may also place the kXR code directly in the status word.) */
64static int
65origin_status_errno(uint16_t status, const u_char *body, uint32_t dlen)
66{
67 int errcode = (int) status;
68
69 if (status == kXR_error) {
70 const char *m = NULL;
71 size_t ml = 0;
72 (void) xrd_error_body_decode(body, dlen, &errcode, &m, &ml);
73 }
74 switch (errcode) {
75 case kXR_NotFound: return ENOENT;
76 case kXR_NotAuthorized: return EACCES;
77 case kXR_isDirectory: return EISDIR;
78 case kXR_ItExists: return ENOTEMPTY; /* non-empty dir: kXR_rmdir, or kXR_mv onto one */
79 default: return EIO;
80 }
81}

83/* brix_cache_origin_rename — kXR_mv old→new on the origin. Wire payload is
84 * "src ' ' dst" with arg1len=len(src). Returns 0, or -1 with errno set. */
85int
86brix_cache_origin_rename(brix_cache_fill_t *t,
87 brix_cache_origin_conn_t *oc, const char *src, const char *dst)
88{
89 uint8_t body[XRDW_BODY_LEN];
90 size_t sl = strlen(src), dl = strlen(dst), total = sl + 1 + dl;
91 char *payload;
92 uint16_t status;
93 uint32_t dlen;
94 u_char *rbody;
95 int rc;
96
97 if (sl == 0 || sl > 0x7fff) {
98 errno = EINVAL;
99 return -1;
100 }
101 payload = malloc(total);
102 if (payload == NULL) {
103 errno = ENOMEM;
104 return -1;
105 }
106 ngx_memcpy(payload, src, sl);
107 payload[sl] = ' ';
108 ngx_memcpy(payload + sl + 1, dst, dl);
109
110 {
111 xrdw_twopath_req_t b = { .arg1len = (int16_t) sl };
112 xrdw_twopath_req_pack(&b, body);
113 }
114 rc = origin_request(t, oc, kXR_mv, body, payload, total, &status, &rbody,
115 &dlen, 256);
116 free(payload);
117 if (rc != 0) {
118 errno = EIO;
119 return -1;
120 }
121 if (status != kXR_ok) {
122 errno = origin_status_errno(status, rbody, dlen);
123 free(rbody);
124 return -1;
125 }
126 free(rbody);
127 return 0;
128}

201/* Build "<path>\0[int16 rc=0]<name>\0" (+ "[int32 BE vlen]<value>") for a single-
202 * attribute fattr request. Returns a malloc'd buffer + *plen, or NULL (OOM). */
203static u_char *
204origin_fattr_payload(const char *path, const char *name, const void *val,
205 size_t vlen, int with_value, size_t *plen)
206{
207 size_t pn = strlen(path), nn = strlen(name);
208 size_t need = pn + 1 + 2 + nn + 1 + (with_value ? 4 + vlen : 0);
209 u_char *buf, *p;
210
211 buf = malloc(need);
212 if (buf == NULL) {
213 return NULL;
214 }
215 p = buf;
216 ngx_memcpy(p, path, pn); p += pn; *p++ = 0;
217 *p++ = 0; *p++ = 0; /* nvec int16 rc=0 */
218 ngx_memcpy(p, name, nn); p += nn; *p++ = 0;
219 if (with_value) {
220 uint32_t vbe = htonl((uint32_t) vlen);
221 ngx_memcpy(p, &vbe, 4); p += 4;
222 if (vlen > 0) { ngx_memcpy(p, val, vlen); p += vlen; }
223 }
224 *plen = (size_t) (p - buf);
225 return buf;
226}

228/* brix_cache_origin_getfattr — kXR_fattr Get of ONE attribute on `path`. Copies
229 * the value into buf[cap] and returns its length, 0 if absent, or -1 (errno). */
230ssize_t
231brix_cache_origin_getfattr(brix_cache_fill_t *t,
232 brix_cache_origin_conn_t *oc, const char *path, const char *name,
233 void *buf, size_t cap)
234{
235 uint8_t body[XRDW_BODY_LEN];
236 u_char *payload, *rbody = NULL, *after;
237 size_t plen, next;
238 uint16_t status, rc = 0;
239 uint32_t dlen, vlen;
240
241 payload = origin_fattr_payload(path, name, NULL, 0, 0, &plen);
242 if (payload == NULL) { errno = ENOMEM; return -1; }
243 {
244 xrdw_fattr_req_t b = { .subcode = kXR_fattrGet, .numattr = 1 };
245 xrdw_fattr_req_pack(&b, body);
246 }
247 if (origin_request(t, oc, kXR_fattr, body, payload, plen, &status, &rbody,
248 &dlen, 65536) != 0)
249 {
250 free(payload);
251 errno = EIO;
252 return -1;
253 }
254 free(payload);
255 if (status != kXR_ok) {
256 errno = origin_status_errno(status, rbody, dlen);
257 free(rbody);
258 return -1;
259 }
260 if (rbody == NULL || dlen < 2
261 || xrdp_fattr_nvec_parse(rbody, dlen, 2, &rc, NULL, NULL, &next) != 0)
262 {
263 free(rbody);
264 errno = EIO;
265 return -1;
266 }
267 if (rc != 0) { /* attribute not present */
268 free(rbody);
269 errno = ENODATA;
270 return -1;
271 }
272 after = rbody + next;
273 if (after + 4 > rbody + dlen) { free(rbody); errno = EIO; return -1; }
274 ngx_memcpy(&vlen, after, 4); vlen = ntohl(vlen); after += 4;
275 if (after + vlen > rbody + dlen) { vlen = (uint32_t) (rbody + dlen - after); }
276 if (buf != NULL && cap > 0) {
277 ngx_memcpy(buf, after, (vlen < cap) ? vlen : cap);
278 }
279 free(rbody);
280 return (ssize_t) vlen;
281}

412/* brix_cache_dirlist_add — append a strdup of name[0, len) to *dl, growing
413 * the array (doubling from 16) as needed. Returns 0 on success, -1 (errno
414 * ENOMEM) on allocation failure — caller aborts the fetch; the caller-owned
415 * dl still holds every name added so far and must still be freed. */
416static int
417brix_cache_dirlist_add(brix_cache_dirlist_t *dl, const char *name, size_t len)
418{
419 char *copy;
420
421 if (dl->count == dl->cap) {
422 size_t new_cap = (dl->cap == 0) ? 16 : dl->cap * 2;
423 char **grown = realloc(dl->names, new_cap * sizeof(*grown));
424
425 if (grown == NULL) {
426 errno = ENOMEM;
427 return -1;
428 }
429 dl->names = grown;
430 dl->cap = new_cap;
431 }
432
433 copy = malloc(len + 1);
434 if (copy == NULL) {
435 errno = ENOMEM;
436 return -1;
437 }
438 ngx_memcpy(copy, name, len);
439 copy[len] = '\0';
440
441 dl->names[dl->count++] = copy;
442 return 0;
443}

445/* brix_cache_dirlist_name_is_dot — 1 iff name (len bytes, not NUL-terminated)
446 * is "." or "..". The VFS readdir contract (brix_vfs_readdir, vfs_dir.c)
447 * never yields these to a protocol handler, so a remote origin that includes
448 * them in its kXR_dirlist body must be filtered here to match every other
449 * backend's opendir/readdir behaviour. */
450static int
451brix_cache_dirlist_name_is_dot(const char *name, size_t len)
452{
453 return (len == 1 && name[0] == '.')
454 || (len == 2 && name[0] == '.' && name[1] == '.');
455}

457/* brix_cache_dirlist_split — split one response frame's body on '\n' (the
458 * kXR_dirlist wire delimiter — src/protocols/root/dirlist/handler.c writes
459 * "<name>\n" per entry, plain-names mode has no stat/checksum trailer lines
460 * to skip) into individual names, appending each non-dot entry to *dl. A
461 * trailing partial name (no terminating '\n' — the final frame's last byte
462 * is NUL-terminated by the server, not '\n', per handler.c) is still emitted
463 * if non-empty, since it is the last name in the whole listing. Returns 0 on
464 * success, -1 (errno set) on an allocation failure partway through. */
465static int
466brix_cache_dirlist_split(brix_cache_dirlist_t *dl, const u_char *body,
467 uint32_t dlen)
468{
469 size_t start = 0, i;
470
471 for (i = 0; i < dlen; i++) {
472 if (body[i] != '\n') {
473 continue;
474 }
475 if (i > start && !brix_cache_dirlist_name_is_dot(
476 (const char *) body + start, i - start))
477 {
478 if (brix_cache_dirlist_add(dl, (const char *) body + start,
479 i - start) != 0)
480 {
481 return -1;
482 }
483 }
484 start = i + 1;
485 }
486 /* Trailing bytes with no '\n' (final frame's NUL-terminated tail, or a
487 * lone name with no delimiter at all) — treat as one more name. */
488 if (start < dlen && !brix_cache_dirlist_name_is_dot(
489 (const char *) body + start, dlen - start))
490 {
491 if (brix_cache_dirlist_add(dl, (const char *) body + start,
492 dlen - start) != 0)
493 {
494 return -1;
495 }
496 }
497 return 0;
498}

500/* brix_cache_origin_dirlist — kXR_dirlist <path> on the origin: plain name
501 * listing (no kXR_dstat/kXR_dcksm — dirlist.c's server handler only needs
502 * per-entry stat when a caller explicitly asks for it, and sd_xroot's
503 * opendir/readdir only need names, matching every other backend's directory
504 * iteration contract). Wire request: kXR_dirlist opcode, body = reserved(15)
505 * + options(1) at body[15] (xrdw_dirlist_req_pack; options=0 here — no
506 * kXR_dstat/kXR_dcksm/kXR_online), payload = the path (no NUL — dlen is the
507 * length, matching kXR_rm/kXR_rmdir's payload framing above). The server
508 * streams the listing as kXR_oksofar frames of newline-delimited names,
509 * ending with a single kXR_ok frame (src/protocols/root/dirlist/handler.c);
510 * this loops brix_cache_read_response until that final kXR_ok, splitting and
511 * appending each frame's body into *dl via brix_cache_dirlist_split.
512 * Returns 0 (dl populated), or -1 with errno set (dl left however far it
513 * got — still owned by the caller). */
514int
515brix_cache_origin_dirlist(brix_cache_fill_t *t,
516 brix_cache_origin_conn_t *oc, const char *path, brix_cache_dirlist_t *dl)
517{
518 uint8_t body[XRDW_BODY_LEN];
519 size_t pl = (path != NULL) ? strlen(path) : 0;
520 size_t total = sizeof(ClientRequestHdr) + pl;
521 u_char *buf;
522 ClientRequestHdr *req;
523
524 if (pl == 0 || pl > 0x7fff) {
525 errno = EINVAL;
526 return -1;
527 }
528
529 {
530 xrdw_dirlist_req_t b = { .options = 0 }; /* plain names, no stat */
531 xrdw_dirlist_req_pack(&b, body);
532 }
533
534 buf = malloc(total);
535 if (buf == NULL) {
536 errno = ENOMEM;
537 return -1;
538 }
539 ngx_memzero(buf, sizeof(ClientRequestHdr));
540 req = (ClientRequestHdr *) buf;
541 req->streamid[1] = 8; /* unused stream slot */
542 req->requestid = htons(kXR_dirlist);
543 ngx_memcpy(req->body, body, XRDW_BODY_LEN);
544 req->dlen = htonl((kXR_int32) pl);
545 ngx_memcpy(buf + sizeof(ClientRequestHdr), path, pl);
546
547 if (brix_cache_io_send(oc, buf, total) != 0) {
548 free(buf);
549 errno = EIO;
550 return -1;
551 }
552 free(buf);
553
554 for (;;) {
555 uint16_t status;
556 uint32_t dlen;
557 u_char *rbody = NULL;
558

564 }
565

568 free(rbody);
569 return -1;
570 }
571

573 free(rbody);

575 "cache origin dirlist returned invalid status");
576 errno = EIO;
577 return -1;
578 }
579
580 if (dlen > 0 && brix_cache_dirlist_split(dl, rbody, dlen) != 0) {
581 int saved_errno = errno;
582
583 free(rbody);
584 errno = saved_errno;
585 return -1;
586 }
587 free(rbody);
588
589 if (status == kXR_ok) {
590 return 0;
591 }
592 }
