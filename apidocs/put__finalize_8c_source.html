12/*
13 * s3_commit_put — W6b: commit the staged temp file to its final path, honouring
14 * an exclusive (create-if-absent) PutObject. When the request carried
15 * `If-None-Match: *` the commit uses renameat2(RENAME_NOREPLACE); otherwise the
16 * plain rename. Returns NGX_OK; or NGX_ERROR with ngx_errno preserved (EEXIST
17 * when an exclusive create lost the race — the caller maps that to 412). Runs
18 * on the event loop (all three PUT commit sites do), so the request ctx is live.
19 */
20ngx_int_t
21s3_commit_put(ngx_http_request_t *r, ngx_log_t *log, const char *root_canon,
22 brix_vfs_staged_t *staged, const char *final_path)
23{
24 ngx_http_s3_req_ctx_t *rx =
25 ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
26 size_t bytes;
27 ngx_int_t rc;
28 int e;
29
30 (void) root_canon; /* the staged handle carries its own final/tmp paths */
31 /* Publish through the VFS staged surface (routes through sd_stage when a stage
32 * store is composed; a local temp+rename otherwise) — §6.5/G9 unified staging. */
33 rc = brix_vfs_staged_commit(staged,
34 (rx != NULL && rx->exclusive_create) ? 1 : 0);
35
36 /* Unified ledger: S3 PutObject (chunked / aio path) is a STAGE publish — the
37 * same audit line as S3 POST, WebDAV PUT, and root:// uploads. The staged fd
38 * is already closed by the body handler before commit, so the committed
39 * object's size comes from a confined stat of the published path. */
40 if (rc == NGX_OK) {
41 brix_vfs_ctx_t fctx;
42 brix_vfs_stat_t fst;
43
44 brix_vfs_ctx_init(&fctx, r->pool, log, BRIX_PROTO_S3, root_canon,
45 NULL, 0 /* allow_write */, 0 /* is_tls */, NULL, final_path);
46 bytes = (brix_vfs_probe(&fctx, 1 /* no-follow */, &fst) == NGX_OK
47 && fst.is_regular) ? (size_t) fst.size : 0;
48 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path, NULL, bytes,
49 BRIX_XFER_OK, 0, log);
50 } else {
51 e = errno;
52 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path, NULL, 0,
53 BRIX_XFER_COMMIT_ERR, e, log);
54 errno = e; /* preserve EEXIST → 412 for the caller */
55 }
56 return rc;
57}

97/*
98 * s3_put_finalize_fs_error — finalize a PUT that failed at a filesystem op,
99 * mapping the captured errno to the correct HTTP status instead of a blanket 500.
100 *
101 * WHAT: A cross-tenant / DAC-denied create (e.g. alice PUT into bob's 0755 dir →
102 * EACCES on the staged temp create) is a forbidden request, not a server
103 * fault. Route the captured errno through brix_http_errno_to_status()
104 * (the same map S3 GET/CopyObject already use): EACCES/EPERM/EXDEV/ELOOP →
105 * 403 AccessDenied, ENOENT/ENOTDIR → 404 NoSuchKey, ENOSPC → 507, etc.
106 * Any errno that maps to 5xx falls back to the existing internal-error path.
107 * WHY: A genuine permission denial must surface as 403 (clean 4xx contract), not
108 * HTTP 500 — a 500 misreports a bounded, well-formed request as an internal
109 * error and is the robustness defect the impersonation red-team flagged.
110 * HOW: The caller MUST capture errno into saved_errno BEFORE any intervening
111 * syscall (logging, staged_abort) clobbers it, then pass it here.
112 */
113void
114s3_put_finalize_fs_error(ngx_http_request_t *r, int saved_errno)
115{
116 int status;
117 const char *code;
118 const char *message;
119
120 status = (int) brix_http_errno_to_status(saved_errno);
121
122 /* errno that has no clean 4xx contract (e.g. EIO) keeps the 500 path. */
123 if (status >= 500) {
124 s3_put_finalize_error(r);
125 return;
126 }
127
128 switch (status) {
129 case 403:
130 code = "AccessDenied";
131 message = "Access Denied.";
132 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_ACCESS_DENIED]);
133 break;
134 case 404:
135 code = "NoSuchKey";
136 message = "The specified key does not exist.";
137 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_NO_SUCH_KEY]);
138 break;
139 case 409:
140 code = "BucketAlreadyExists";
141 message = "The requested resource already exists.";
142 break;
143 case 414:
144 code = "KeyTooLongError";
145 message = "Your key is too long.";
146 break;
147 default:
148 code = "InvalidRequest";
149 message = "The request could not be completed.";
150 break;
151 }
152
153 brix_dashboard_http_error(r, message);
154 brix_dashboard_http_finish(r);
155 (void) s3_send_xml_error(r, (ngx_uint_t) status, code, message);
156 s3_metrics_finalize_request_method(r, BRIX_S3_METHOD_PUT,
157 (ngx_int_t) status);
158}

161void
162s3_put_finalize_empty_ok(ngx_http_request_t *r)
163{
164 ngx_int_t rc;
165
166 /*
167 * S3 PutObject succeeds with 200 + an ETag header. A PUT response must
168 * NEVER be reinterpreted as a conditional-GET result: when the client sends
169 * If-None-Match / If-Modified-Since, nginx core's not-modified filter would
170 * otherwise rewrite this 200 into a 304 Not Modified (304 is a GET/HEAD-only
171 * status — RFC 9110 §15.4.5). The object is written either way, but the
172 * status would be wrong and could break S3 SDKs that set conditional headers
173 * by default. Disable the not-modified filter for this response.
174 */
175 r->disable_not_modified = 1;
176 r->headers_out.status = NGX_HTTP_OK;
177 r->headers_out.content_length_n = 0;
178 ngx_http_send_header(r);
179
180 rc = ngx_http_send_special(r, NGX_HTTP_LAST);
181 brix_dashboard_http_finish(r);
182 s3_metrics_finalize_request_method(r, BRIX_S3_METHOD_PUT, rc);
183}

197 }
198 BRIX_S3_METRIC_INC(put_body_total[body_mode]);
199 s3_put_finalize_empty_ok(r);

237/*
238 * s3_put_finalize_codec_error — reject a PUT whose Content-Encoding could not be
239 * decoded (phase-42 W1). Maps the decoder's HTTP status to a clean S3 XML error
240 * instead of a blanket 500: a bomb-guard trip → 413 EntityTooLarge, an
241 * unsupported/disabled codec or a malformed/truncated stream → 400 InvalidRequest.
242 * The staged object has already been aborted by the caller, so no partial bytes
243 * are ever stored. A genuine I/O/internal failure stays a 500 (caller routes it
244 * to s3_put_finalize_error instead).
245 */
246void
247s3_put_finalize_codec_error(ngx_http_request_t *r, ngx_int_t status)
248{
249 const char *code = "InvalidRequest";
250 const char *msg = "The request body could not be decoded with the "
251 "requested Content-Encoding.";
252
253 if (status == NGX_HTTP_REQUEST_ENTITY_TOO_LARGE) {
254 code = "EntityTooLarge";
255 msg = "The decompressed body exceeds the maximum allowed size.";
256 }
257 brix_dashboard_http_error(r, "s3 content-encoding decode failed");
258 brix_dashboard_http_finish(r);
259 (void) s3_send_xml_error(r, status, code, msg);
260 s3_metrics_finalize_request_method(r, BRIX_S3_METHOD_PUT, status);
261}

264/*
265 * s3_put_checksum_failed — verify+echo the client's full-object checksum and, on
266 * a terminal failure, send the corresponding 400 response.
267 *
268 * Returns 1 when a failure response was already sent (the caller must stop), or
269 * 0 to proceed with the success finalize. A compute error (S3_CKSUM_ERROR) is
270 * non-fatal — the object stands, just without an echoed checksum header.
271 */
272int
273s3_put_checksum_failed(ngx_http_request_t *r, const char *fs_path,
274 const char *root_canon)
275{
276 switch (s3_put_checksum_apply(r, fs_path, root_canon)) {
277 case S3_CKSUM_MISMATCH:
278 s3_put_finalize_bad_digest(r);
279 return 1;
280 case S3_CKSUM_CONFLICT:
281 s3_put_finalize_invalid_request(r);
282 return 1;
283 default:
284 return 0; /* S3_CKSUM_OK or S3_CKSUM_ERROR — proceed */
285 }
286}

301/*
302 * s3_put_body_handler — ngx_http_read_client_request_body() completion
303 * callback for S3 PutObject.
304 *
305 * Retrieves the filesystem path from the per-request module context (set by
306 * ngx_http_s3_handler before calling ngx_http_read_client_request_body).
307 *
308 * Special case — directory sentinels: S3 clients represent directories as
309 * zero-byte objects whose key ends with the S3_DIR_SENTINEL suffix (typically
310 * "_$folder$"). These are handled by creating the real directory rather than
311 * a file.
312 *
313 * Normal case: write body to a temp file (O_EXCL for atomicity), then rename
314 * onto the final path. On any error the temp file is unlinked.
315 *
316 * Ownership: the response is finalised via s3_put_finalize_ok() which calls
317 * ngx_http_send_header() and ngx_http_send_special(). The function must not
318 * return NGX_DONE (the caller already incremented r->main->count via
319 * ngx_http_read_client_request_body).
