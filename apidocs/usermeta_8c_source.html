1/*
2 * usermeta.c — S3 user-defined object metadata (x-amz-meta-*).
3 *
4 * WHAT: Persist the x-amz-meta-<name> headers a client sends on PutObject /
5 * CopyObject, and echo them back on GET/HEAD — the S3 user-metadata
6 * contract every SDK relies on, and the slot the advisory unix-attr codec
7 * (x-amz-meta-xrd-unixattr) rides in for object-store POSIX-attr parity.
8 * WHY: The gateway stored none, so metadata round-trips silently dropped. The
9 * backend-metadata-parity work needs a real S3 endpoint that keeps user
10 * metadata so the shared sd_s3 driver's get/set-meta validate end-to-end
11 * and the S3 export backend reaches getattr/xattr parity.
12 * HOW: The whole user-metadata set is one URL-encoded "k=v&k=v" blob stored in
13 * a dedicated xattr (user.s3.usermeta) beside the object, via the VFS
14 * xattr surface (impersonation-correct, exactly like object tagging in
15 * tagging.c). Keys are lowercased (AWS contract). Invisible to every other
16 * code path; no new on-disk structure.
17 */

54/* Read the stored metadata blob into out (NUL-terminated). Returns length, 0 if
55 * none, -1 on error (denied / missing object). */
56static ssize_t
57s3_user_meta_load(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
58 const char *fs_path, char *out, size_t outsz)
59{
60 brix_vfs_ctx_t vctx;
61 ssize_t n;
62
63 s3_user_meta_vfs_ctx(r, fs_path, cf, &vctx);
64 n = brix_vfs_getxattr(&vctx, S3_USERMETA_XATTR, out, outsz - 1);
65 if (n < 0) {
66 /* phase74-fp: ENOTSUP == EOPNOTSUPP on Linux but they are distinct
67 * errnos on other POSIX systems — both are checked deliberately for
68 * portability, so the "equivalent nested operands" finding is moot. */
69 /* NOLINTNEXTLINE(misc-redundant-expression) */
70 if (errno == ENODATA || errno == ENOTSUP || errno == EOPNOTSUPP) {
71 return 0; /* object readable, just carries no user metadata */
72 }
73 return -1;
74 }
75 out[n] = '\0';
76 return n;
77}

91/*
92 * s3_user_meta_append_kv — append one already-lowercased key and its raw value
93 * to the "k=v&k=v" blob under construction.
94 *
95 * WHAT: Emits a "&" separator (only when *pos already holds a prior pair), then
96 * the percent-encoded key, a "=", and the percent-encoded value, advancing
97 * *pos past the bytes written. Returns NGX_OK, or NGX_ERROR if any encode
98 * fails or the result would overflow the out/outsz buffer.
99 * WHY: Keeping the separator/encode/'='/encode arithmetic in one linear helper
100 * is what lets the header-scan loop stay under the complexity cap — the
101 * loop decides which headers qualify, this owns the byte-exact framing so
102 * '=', '&' and spaces can never break the stored form.
103 * HOW: 1. When *pos is non-zero, bounds-check and write the '&' joiner.
104 * 2. URL-encode the key (empty passthrough set) and bounds-checked copy it.
105 * 3. Write the '=' delimiter.
106 * 4. URL-encode the value and bounds-checked copy it.
107 * 5. Publish the advanced cursor back through *pos.
108 */
109static ngx_int_t
110s3_user_meta_append_kv(char *out, size_t outsz, size_t *pos,
111 const char *klow, size_t knlen, const u_char *val, size_t vallen)
112{
113 char enc[1024];
114 size_t cursor = *pos;
115 size_t n;
116
117 if (cursor != 0) {
118 if (cursor + 1 >= outsz) {
119 return NGX_ERROR;
120 }
121 out[cursor++] = '&';
122 }
123
124 if (brix_http_urlencode((u_char *) klow, knlen, enc, sizeof(enc), "") < 0) {
125 return NGX_ERROR;
126 }
127 n = ngx_strlen(enc);
128 if (cursor + n + 1 >= outsz) {
129 return NGX_ERROR;
130 }
131 ngx_memcpy(out + cursor, enc, n);
132 cursor += n;
133 out[cursor++] = '=';
134
135 if (brix_http_urlencode(val, vallen, enc, sizeof(enc), "") < 0) {
136 return NGX_ERROR;
137 }
138 n = ngx_strlen(enc);
139 if (cursor + n >= outsz) {
140 return NGX_ERROR;
141 }
142 ngx_memcpy(out + cursor, enc, n);
143 cursor += n;
144
145 *pos = cursor;
146 return NGX_OK;
147}

149/*
150 * s3_user_meta_blob_from_headers — collect every x-amz-meta-<name> request
151 * header into one URL-encoded "k=v&k=v" blob (key lowercased, both components
152 * percent-encoded so '=', '&' and spaces never break the stored form). Sets
153 * *any when at least one header was found. Returns NGX_OK / NGX_ERROR (overflow).
154 */
155static ngx_int_t
156s3_user_meta_blob_from_headers(ngx_http_request_t *r, char *out, size_t outsz,
157 size_t *out_len, int *any)
158{
159 ngx_list_part_t *part = &r->headers_in.headers.part;
160 ngx_table_elt_t *h = part->elts;
161 const size_t plen = sizeof(S3_USERMETA_PREFIX) - 1;
162 size_t pos = 0;
163 ngx_uint_t i;
164
165 *any = 0;
166 for (i = 0; /* void */; i++) {
167 const u_char *kname;
168 size_t knlen, j;
169 char klow[256];
170
171 if (i >= part->nelts) {
172 if (part->next == NULL) {
173 break;
174 }
175 part = part->next;
176 h = part->elts;
177 i = 0;
178 }
179
180 if (h[i].key.len <= plen
181 || ngx_strncasecmp(h[i].key.data, (u_char *) S3_USERMETA_PREFIX,
182 plen) != 0)
183 {
184 continue;
185 }
186
187 kname = h[i].key.data + plen;
188 knlen = h[i].key.len - plen;
189 if (knlen == 0 || knlen >= sizeof(klow)) {
190 continue;
191 }
192 if (h[i].value.len > S3_USERMETA_KV_MAX) {
193 return NGX_ERROR;
194 }
195 for (j = 0; j < knlen; j++) {
196 klow[j] = (char) ngx_tolower(kname[j]);
197 }
198
199 if (s3_user_meta_append_kv(out, outsz, &pos, klow, knlen,
200 h[i].value.data, h[i].value.len) != NGX_OK)
201 {
202 return NGX_ERROR;
203 }
204 *any = 1;
205 }
206
207 out[pos] = '\0';
208 *out_len = pos;
209 return NGX_OK;
210}

235void
236s3_echo_user_metadata(ngx_http_request_t *r, const char *fs_path)
237{
238 ngx_http_s3_loc_conf_t *cf =
239 ngx_http_get_module_loc_conf(r, ngx_http_brix_s3_module);
240 char blob[S3_USERMETA_BLOB_MAX];
241 ssize_t blen;
242 char *p, *amp;
243
244 blen = s3_user_meta_load(r, cf, fs_path, blob, sizeof(blob));
245 if (blen <= 0) {
246 return; /* error, denied, or no user metadata — nothing to echo */
247 }
248
249 p = blob;
250 while (*p != '\0') {
251 char kbuf[256], vbuf[S3_USERMETA_KV_MAX + 1];
252 char *eq;
253
254 amp = strchr(p, '&');
255 if (amp != NULL) {
256 *amp = '\0';
257 }
258 eq = strchr(p, '=');
259 if (eq != NULL) {
260 *eq = '\0';
261 if (brix_http_urldecode((u_char *) p, ngx_strlen(p), kbuf,
262 sizeof(kbuf), BRIX_URLDECODE_PLUS_TO_SPACE)
263 == BRIX_URLDECODE_OK
264 && brix_http_urldecode((u_char *) (eq + 1),
265 ngx_strlen(eq + 1), vbuf, sizeof(vbuf),
266 BRIX_URLDECODE_PLUS_TO_SPACE) == BRIX_URLDECODE_OK
267 && kbuf[0] != '\0')
268 {
269 /* The header NAME must outlive this stack frame: set_header
270 * copies the value but stores the key pointer as-is, so build
271 * "x-amz-meta-<name>" in the request pool. */
272 size_t pfx = sizeof(S3_USERMETA_PREFIX) - 1;
273 size_t klen = ngx_strlen(kbuf);
274 u_char *hdr = ngx_pnalloc(r->pool, pfx + klen + 1);
275
276 if (hdr != NULL) {
277 ngx_memcpy(hdr, S3_USERMETA_PREFIX, pfx);
278 ngx_memcpy(hdr + pfx, kbuf, klen);
279 hdr[pfx + klen] = '\0';
280 (void) s3_set_header(r, (const char *) hdr, vbuf);
281 }
282 }
283 }
284 if (amp == NULL) {
285 break;
286 }
287 p = amp + 1;
288 }
289}
