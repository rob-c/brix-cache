26/* File: gsi_outbound_exchange.c — GSI round-2 driver for native TPC pull
27 * WHAT: tpc_outbound_gsi_exchange completes the outbound GSI handshake on a TPC
28 * pull socket after gsi_outbound_certreq.c has sent kXGC_certreq and received
29 * the server's kXGS_cert (`body`). It (optionally) verifies the server leaf cert
30 * against conf->gsi_store, serialises our proxy cert chain to PEM, and hands the
31 * server reply + proxy credential to the SHARED XrdSecgsi round-2 builder
32 * (brix_gsi_build_cert_response, src/gsi/gsi_core.c) — the exact
33 * implementation the native client uses. It then sends the resulting kXGC_cert
34 * (kXR_auth seq=4) and checks the server's final status == kXR_ok.
35 *
36 * WHY: The DH-variant detection, AES session-key agreement, proof-of-possession
37 * (signing the server rtag with our proxy key), encrypted cert-chain main and
38 * outer kXGC_cert assembly are identical to the client's. Routing both through
39 * one kernel removes a ~280-line parallel reimplementation (this file used to
40 * carry it) and gives the destination the signed-DH path for free: whatever DH
41 * variant the server offers, the shared builder handles it.
42 *
43 * HOW: optional X509_verify_cert(conf->gsi_store, server-leaf) → BIO_s_mem +
44 * PEM_write_bio_X509(chain) → brix_gsi_build_cert_response(body, dlen,
45 * chain_pem, pkey, &payload, &plen) → free(body) → tpc_send_kxr_auth(seq=4,
46 * payload) → tpc_recv_response → status==kXR_ok ? 0 : -1. Resource ownership:
47 * this function owns `body` (frees on every exit; the caller NULLs its copy) and
48 * the proxy-chain BIO + response payload it allocates; it does NOT free
49 * x/chain/pkey/certreq/cbio/kbio — gsi_outbound_certreq.c owns those.
50 * */
51
52/*
53 * tpc_gsi_exchange_cleanup — release the round-2 resources (each guard NULL-safe)
54 * and return rc. The shared kernel owns its own ephemeral DH keypair, so the only
55 * resources this driver holds are the server reply `body`, the proxy-chain PEM BIO
56 * and the malloc'd response payload. Replaces the former round_fail/done ladder.
57 */
58static int
59tpc_gsi_exchange_cleanup(int rc, u_char *body, BIO *chain_bio, u_char *payload)
60{
61 if (body) free(body);
62 if (chain_bio) BIO_free(chain_bio);
63 if (payload) free(payload);
64 return rc;
65}

67/*
68 * tpc_gsi_verify_server_cert — optional CA-store check of the server leaf cert.
69 *
70 * WHAT: When conf->gsi_store is configured AND the server included its kXRS_x509
71 * bucket, PEM-parse the leaf and run X509_verify_cert against our store. Returns
72 * 0 when verification succeeds, is skipped (no store / no bucket / unparseable
73 * leaf), or the store-ctx cannot be initialised (matching the original fall-
74 * through); returns -1 only on a present-and-invalid cert.
75 * WHY: We must not derive a shared secret with an unverified server when a store
76 * is present. GSI delegated proxies are legitimate here, so
77 * X509_V_FLAG_ALLOW_PROXY_CERTS is set — identical to the pre-refactor path.
78 * HOW: BIO_new_mem_buf → PEM_read_bio_X509 → X509_STORE_CTX_{new,init,set_flags}
79 * → X509_verify_cert; every OpenSSL handle freed on every exit (no leak).
80 */
81static int
82tpc_gsi_verify_server_cert(brix_tpc_pull_t *t, const u_char *body, uint32_t dlen)
83{
84 ngx_stream_brix_srv_conf_t *conf = t->conf;
85 const u_char *srv_pem;
86 size_t srv_pem_len;
87 BIO *mbio;
88 X509 *srv;
89 X509_STORE_CTX *sctx;
90 int rc = 0;
91
92 if (conf->gsi_store == NULL) {
93 return 0;
94 }
95 if (gsi_find_bucket(body, dlen, (uint32_t) kXRS_x509,
96 &srv_pem, &srv_pem_len) != 0
97 || srv_pem_len == 0)
98 {
99 return 0;
100 }
101
102 mbio = BIO_new_mem_buf(srv_pem, (int) srv_pem_len);
103 srv = PEM_read_bio_X509(mbio, NULL, NULL, NULL);
104 BIO_free(mbio);
105 if (srv == NULL) {
106 return 0;
107 }
108
109 sctx = X509_STORE_CTX_new();
110 if (sctx != NULL
111 && X509_STORE_CTX_init(sctx, conf->gsi_store, srv, NULL) == 1)
112 {
113 X509_STORE_CTX_set_flags(sctx, X509_V_FLAG_ALLOW_PROXY_CERTS);
114 if (X509_verify_cert(sctx) != 1) {
115 snprintf(t->err_msg, sizeof(t->err_msg),
116 "TPC GSI server certificate verification failed");
117 t->xrd_error = kXR_NotAuthorized;
118 rc = -1;
119 }
120 }
121 if (sctx != NULL) {
122 X509_STORE_CTX_free(sctx);
123 }
124 X509_free(srv);
125 return rc;
126}

145/*
146 * tpc_gsi_build_round2 — serialise our proxy chain to PEM and build the round-2
147 * kXGC_cert payload via the SHARED XrdSecgsi kernel.
148 *
149 * WHAT: PEM-encode the proxy cert chain r->chain (leaf first) into one contiguous
150 * blob and pass it plus the server's round-1 r->body to
151 * brix_gsi_build_cert_response, yielding the malloc'd r->payload / r->plen.
152 * Returns 0 on success, -1 on OOM or a builder failure (with t->err_msg /
153 * t->xrd_error set exactly as before).
154 * WHY: The DH-variant detection, AES session-key agreement, proof-of-possession
155 * and encrypted-chain assembly are identical to the native client's; routing
156 * through the one kernel keeps wire compatibility. chain + pkey are borrowed.
157 * HOW: BIO_s_mem → PEM_write_bio_X509(chain[i]) → BIO_get_mem_ptr →
158 * brix_gsi_build_cert_response. On any failure the chain BIO is freed here and
159 * r->chain_bio left NULL; on success r->chain_bio is returned to the caller to
160 * free in its normal cleanup ordering (preserving the original free/NULL order).
161 */
162static int
163tpc_gsi_build_round2(brix_tpc_pull_t *t, tpc_gsi_round2_t *r)
164{
165 BIO *cb;
166 BUF_MEM *bptr = NULL;
167 char err[160];
168 int ci;
169
170 cb = BIO_new(BIO_s_mem());
171 if (cb == NULL) {
172 snprintf(t->err_msg, sizeof(t->err_msg), "TPC GSI out of memory");
173 t->xrd_error = kXR_NoMemory;
174 return -1;
175 }
176 for (ci = 0; ci < sk_X509_num(r->chain); ci++) {
177 PEM_write_bio_X509(cb, sk_X509_value(r->chain, ci));
178 }
179 BIO_get_mem_ptr(cb, &bptr);
180
181 err[0] = '\0';
182 if (brix_gsi_build_cert_response(r->body, r->dlen,
183 (const uint8_t *) bptr->data,
184 bptr->length, r->pkey,
185 &r->payload, &r->plen,
186 err, sizeof(err)) != 0)
187 {
188 snprintf(t->err_msg, sizeof(t->err_msg),
189 "TPC GSI round-2 build failed: %s", err[0] ? err : "unknown");
190 t->xrd_error = kXR_NotAuthorized;
191 BIO_free(cb);
192 return -1;
193 }
194
195 r->chain_bio = cb;
196 return 0;
197}

199/* WHAT: GSI round-2 driver for outbound TPC — verify server cert (optional), build
200 * the kXGC_cert via the shared gsi_core kernel, send it + check kXR_ok. Returns 0
201 * or -1 with t->xrd_error set. Caller: gsi_outbound_certreq.c. */
202int
203tpc_outbound_gsi_exchange(brix_tpc_pull_t *t, int fd,
204 u_char *body, uint32_t dlen,
205 X509 *x, STACK_OF(X509) *chain, EVP_PKEY *pkey,
206 u_char *certreq, BIO *cbio, BIO *kbio)
207{
208 int rc = -1;
209 uint16_t status;
210 BIO *chain_bio = NULL;
211 u_char *payload = NULL;
212 uint32_t plen = 0;
213 tpc_gsi_round2_t r2 = { body, dlen, chain, pkey, NULL, NULL, 0 };
214
215 /*
216 * Optional: verify the server leaf cert using our CA store before we derive a
217 * shared secret with it. Skipped entirely when no store is configured or the
218 * server omitted its kXRS_x509 bucket — only a present-but-invalid cert aborts.
219 */
220 if (tpc_gsi_verify_server_cert(t, body, dlen) != 0) {
221 return tpc_gsi_exchange_cleanup(rc, body, chain_bio, payload);
222 }
223
224 /*
225 * Serialise our proxy chain to PEM and build the round-2 kXGC_cert via the
226 * SHARED XrdSecgsi kernel — the exact implementation the native client uses
227 * (proven wire-compatible vs real EOS + stock XrdSecgsi). chain PEM + pkey are
228 * borrowed (we still own/free them via the certreq.c finalizer).
229 */
230 if (tpc_gsi_build_round2(t, &r2) != 0) {
231 return tpc_gsi_exchange_cleanup(rc, body, r2.chain_bio, r2.payload);
232 }
233 chain_bio = r2.chain_bio;
234 payload = r2.payload;
235 plen = r2.plen;
236
237 /* The round-1 server body is consumed; free it (and the chain BIO) before we
238 * block on the round-2 round trip. NULL them so cleanup won't double-free. */
239 free(body);
240 body = NULL;
241 BIO_free(chain_bio);
242 chain_bio = NULL;
243
244 /* round 2: send our kXGC_cert (seq=4 — protocol=1, login=2, certreq=3). */
245 if (tpc_send_kxr_auth(t, fd, 4, payload, plen) != 0) {
246 snprintf(t->err_msg, sizeof(t->err_msg), "TPC GSI round-2 send failed");
247 t->xrd_error = kXR_ServerError;
248 return tpc_gsi_exchange_cleanup(rc, body, chain_bio, payload);
249 }
250 free(payload);
251 payload = NULL;
252
253 /* Await the server's verdict. kXR_ok == authenticated. */
254 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
255 snprintf(t->err_msg, sizeof(t->err_msg),
256 "TPC GSI final auth recv failed");
257 t->xrd_error = kXR_ServerError;
258 return tpc_gsi_exchange_cleanup(rc, body, chain_bio, payload);
259 }
260 if (status != kXR_ok) {
261 const char *emsg = (body != NULL && dlen > 4)
262 ? (const char *) (body + 4) : "";
263 snprintf(t->err_msg, sizeof(t->err_msg),
264 "TPC GSI authentication failed (status=%u): %.*s",
265 (unsigned) status,
266 (int) (dlen > 4 ? dlen - 4 : 0), emsg);
267 t->xrd_error = kXR_AuthFailed;
268 return tpc_gsi_exchange_cleanup(rc, body, chain_bio, payload);
269 }
270
271 rc = 0;
272 return tpc_gsi_exchange_cleanup(rc, body, chain_bio, payload);
273}
