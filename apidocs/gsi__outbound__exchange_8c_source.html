26/* File: gsi_outbound_exchange.c — GSI round-2 driver for native TPC pull
27 * WHAT: tpc_outbound_gsi_exchange completes the outbound GSI handshake on a TPC
28 * pull socket after gsi_outbound_certreq.c has sent kXGC_certreq and received
29 * the server's kXGS_cert (`body`). It (optionally) verifies the server leaf cert
30 * against conf->gsi_store, serialises our proxy cert chain to PEM, and hands the
31 * server reply + proxy credential to the SHARED XrdSecgsi round-2 builder
32 * (brix_gsi_build_cert_response, src/gsi/gsi_core.c) — the exact
33 * implementation the native client uses. It then sends the resulting kXGC_cert
34 * (kXR_auth seq=4) and checks the server's final status == kXR_ok.
35 *
36 * WHY: The DH-variant detection, AES session-key agreement, proof-of-possession
37 * (signing the server rtag with our proxy key), encrypted cert-chain main and
38 * outer kXGC_cert assembly are identical to the client's. Routing both through
39 * one kernel removes a ~280-line parallel reimplementation (this file used to
40 * carry it) and gives the destination the signed-DH path for free: whatever DH
41 * variant the server offers, the shared builder handles it.
42 *
43 * HOW: optional X509_verify_cert(conf->gsi_store, server-leaf) → BIO_s_mem +
44 * PEM_write_bio_X509(chain) → brix_gsi_build_cert_response(body, dlen,
45 * chain_pem, pkey, &payload, &plen) → free(body) → tpc_send_kxr_auth(seq=4,
46 * payload) → tpc_recv_response → status==kXR_ok ? 0 : -1. Resource ownership:
47 * this function owns `body` (frees on every exit; the caller NULLs its copy) and
48 * the proxy-chain BIO + response payload it allocates; it does NOT free
49 * x/chain/pkey/certreq/cbio/kbio — gsi_outbound_certreq.c owns those.
50 * */
51
52/*
53 * tpc_gsi_exchange_cleanup — release the round-2 resources (each guard NULL-safe)
54 * and return rc. The shared kernel owns its own ephemeral DH keypair, so the only
55 * resources this driver holds are the server reply `body`, the proxy-chain PEM BIO
56 * and the malloc'd response payload. Replaces the former round_fail/done ladder.
57 */
58static int
59tpc_gsi_exchange_cleanup(int rc, u_char *body, BIO *chain_bio, u_char *payload)
60{
61 if (body) free(body);
62 if (chain_bio) BIO_free(chain_bio);
63 if (payload) free(payload);
64 return rc;
65}
