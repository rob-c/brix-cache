1/*
2 * vfs_internal.h — implementation-private definitions shared by the vfs_*.c units.
3 *
4 * WHAT: Defines the real handle structs hidden behind vfs.h's opaque typedefs
5 * (brix_vfs_file_s, brix_vfs_dir_s), the inline confinement/write
6 * guards (brix_vfs_require_confined, brix_vfs_require_write), the
7 * ctx-path accessor (brix_vfs_ctx_path), the metrics/access-log observer
8 * helpers (brix_vfs_observe_ctx_op / brix_vfs_observe_file_op and the
9 * elapsed-usec/proto helpers they use), and the cross-unit prototypes
10 * (fill_stat, copy_path, adopt_fd, pread_full, pwrite_full).
11 *
12 * WHY: Every vfs_*.c file needs the same guard-then-syscall-then-observe
13 * pattern and the same handle layout. Centralising it here keeps the
14 * per-op files thin and guarantees that confinement re-verification and
15 * metric/log emission happen identically for every operation.
16 *
17 * HOW: The guards reject any ctx whose resolved path is empty or not confined
18 * (and, for writes, when allow_write is unset), setting errno. The
19 * observer helpers translate an rc/errno into an brix_err_class_t,
20 * compute latency from a start ngx_current_msec, then call
21 * brix_metric_op_done + brix_access_log_emit and restore errno so the
22 * caller can return it untouched. Only this header is included by the
23 * vfs_*.c units; protocol handlers include vfs.h instead.
24 */

145/* Build a transient storage-driver object view from a ctx + fd: the bound
146 * instance (or NULL for the default backend), that backend's driver, and the
147 * fd. Used to ask the backend to perform/decide a per-fd operation without the
148 * VFS hard-coding any concrete driver. "No explicit backend" resolves to
149 * brix_sd_default_driver() rather than naming POSIX. */
150static ngx_inline void
151brix_vfs_ctx_sd_obj(const brix_vfs_ctx_t *ctx, ngx_fd_t fd,
152 brix_sd_obj_t *obj)
153{
154 ngx_memzero(obj, sizeof(*obj));
155 obj->inst = ctx != NULL ? ctx->sd : NULL;
156 obj->driver = (obj->inst != NULL) ? obj->inst->driver
157 : brix_sd_default_driver();
158 obj->fd = fd;
159}

186/* Ask the handle's backend for a sendfile-able fd over [off, off+len), passing
187 * the VFS's storage-neutral zero-copy verdict; returns the fd, or
188 * NGX_INVALID_FILE when the backend declines (or has no read_sendfile_fd slot).
189 * This is the single place the VFS consults the backend's sendfile decision. */
190static ngx_inline ngx_fd_t
191brix_vfs_handle_sendfile_fd(const brix_vfs_file_t *fh, off_t off,
192 size_t len, unsigned want_zerocopy)
193{
194 brix_sd_obj_t obj;
195
196 brix_vfs_handle_sd_obj(fh, &obj);
197 if (obj.driver == NULL || obj.driver->read_sendfile_fd == NULL) {
198 return NGX_INVALID_FILE;
199 }
200 return obj.driver->read_sendfile_fd(&obj, off, len, want_zerocopy);
201}

216/* Read guard: assert the ctx has a non-empty, kernel-confined resolved path.
217 * Returns NGX_OK if confined, else NGX_ERROR with errno=EINVAL. Every wire op
218 * must pass this before touching the filesystem. */
219static ngx_inline ngx_int_t
220brix_vfs_require_confined(const brix_vfs_ctx_t *ctx)
221{
222 const char *path = brix_vfs_ctx_path(ctx);
223
224 if (ctx == NULL || path == NULL || path[0] == '\0'
225 || !ctx->resolved.is_confined)
226 {
227 errno = EINVAL;
228 return NGX_ERROR;
229 }
230
231 return NGX_OK;
232}
233
234/* Write guard: confinement check (as above) plus ctx->allow_write.
235 * Returns NGX_OK only when both hold; otherwise NGX_ERROR with errno=EINVAL
236 * (unconfined) or EACCES (write not permitted). */
237static ngx_inline ngx_int_t
238brix_vfs_require_write(const brix_vfs_ctx_t *ctx)
239{
240 if (brix_vfs_require_confined(ctx) != NGX_OK) {
241 return NGX_ERROR;
242 }
243
244 if (!ctx->allow_write) {
245 errno = EACCES;
246 return NGX_ERROR;
247 }
248
249 return NGX_OK;
250}

252/* Translate a namespace status into a faithful POSIX errno. The namespace layer
253 * sets res.sys_errno for syscall failures but leaves it 0 for the conditions it
254 * derives itself (notably BRIX_NS_NOT_EMPTY from its own emptiness probe), so
255 * callers that collapse a failed brix_ns_* result to errno must use this for
256 * the sys_errno==0 case rather than a blanket EIO — otherwise a non-empty rmdir
257 * surfaces as EIO/500 instead of ENOTEMPTY/409. */
258static ngx_inline int
259brix_vfs_ns_status_errno(brix_ns_status_t status)
260{
261 switch (status) {
262 case BRIX_NS_OK: return 0;
263 case BRIX_NS_NOT_FOUND: return ENOENT;
264 case BRIX_NS_DENIED: return EACCES;
265 case BRIX_NS_EXISTS: return EEXIST;
266 case BRIX_NS_CONFLICT: return ENOTDIR;
267 case BRIX_NS_NOT_EMPTY: return ENOTEMPTY;
268 case BRIX_NS_TOO_LONG: return ENAMETOOLONG;
269 case BRIX_NS_NO_SPACE: return ENOSPC;
270 case BRIX_NS_IO_ERROR: return EIO;
271 }
272
273 return EIO;
274}

288/* phase-56 D-1: a real monotonic timestamp in NANOseconds for op-latency.
289 * Replaces the cached ngx_current_msec, which (a) only advances on event-loop
290 * ticks — so a synchronous metadata op that never yields reported 0 µs — and
291 * (b) is millisecond-resolution, quantizing the whole sub-ms band to 0/1000 µs.
292 * CLOCK_MONOTONIC is vDSO-backed (~20 ns/call, lost in the syscalls the op
293 * already makes) and gives honest sub-µs deltas. NOT CLOCK_MONOTONIC_COARSE —
294 * that is also ~1-4 ms granularity and would only fix (a), not the resolution. */
295static ngx_inline uint64_t
296brix_vfs_now_ns(void)
297{
298 struct timespec ts;
299
300 clock_gettime(CLOCK_MONOTONIC, &ts);
301 return (uint64_t) ts.tv_sec * 1000000000ull + (uint64_t) ts.tv_nsec;
302}
303
304/* Latency since start_ns in MICROseconds (start is an brix_vfs_now_ns()
305 * snapshot). Clamps to 0 if the monotonic clock appears to have gone backwards. */
306static ngx_inline ngx_msec_t
307brix_vfs_elapsed_usec(uint64_t start_ns)
308{
309 uint64_t now_ns = brix_vfs_now_ns();
310
311 if (now_ns < start_ns) {
312 return 0;
313 }
314
315 return (ngx_msec_t) ((now_ns - start_ns) / 1000ull);
316}

318/* Post-op observer: derive the error class from rc/sys_errno, compute latency
319 * from start_msec, then emit one metric (brix_metric_op_done) and one access
320 * log line (brix_access_log_emit) for op. bytes is the transferred count;
321 * result may be NULL. Borrows path (does not copy). Restores errno=sys_errno on
322 * return so the caller can propagate it unchanged. */
323static ngx_inline void
324brix_vfs_observe_ctx_op(const brix_vfs_ctx_t *ctx, const char *path,
325 brix_metric_op_t op, const brix_vfs_io_result_t *result,
326 size_t bytes, ngx_int_t rc, int sys_errno, uint64_t start_ns)
327{
328 brix_err_class_t err;
329 ngx_msec_t latency_usec;
330
331 err = rc == NGX_OK ? BRIX_ERR_NONE
332 : brix_metric_err_from_errno(sys_errno);
333 latency_usec = brix_vfs_elapsed_usec(start_ns);
334
335 brix_metric_op_done(brix_vfs_metrics_proto(ctx), op, bytes,
336 latency_usec, err);
337
338 /* Per-backend storage byte totals (staged-commit writes, VFS-metered
339 * reads). ctx->sd == NULL is the default-POSIX instance. */
340 if (rc == NGX_OK && bytes > 0) {
341 brix_metric_backend_bytes(
342 ctx != NULL && ctx->sd != NULL ? brix_sd_backend_name(ctx->sd)
343 : "posix",
344 op, bytes);
345 }
346
347 brix_access_log_emit(ctx, path, op, result, bytes, err, latency_usec);
348
349 errno = sys_errno;
350}
