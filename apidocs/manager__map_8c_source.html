11/* ---- Store a copied, NUL-terminated host into a manager-map entry ----
12 *
13 * WHAT: Allocates hostlen+1 bytes from cf->pool, copies hostlen bytes from
14 * src, NUL-terminates, and records the length in entry->host. Returns NGX_OK
15 * on success, or NGX_ERROR if the pool allocation fails (no log emitted — the
16 * historical behaviour returns the error silently).
17 *
18 * WHY: Both endpoint forms (IPv6 literal and host:port) extract a host
19 * substring of differing bounds but store it identically; factoring the copy
20 * keeps that storage step in one place with no behavioural drift.
21 *
22 * HOW:
23 * 1. ngx_pnalloc(hostlen + 1); on NULL return NGX_ERROR.
24 * 2. ngx_memcpy hostlen bytes from src and write a trailing NUL.
25 * 3. Set entry->host.len = hostlen and return NGX_OK.
26 */
27static ngx_int_t
28brix_manager_map_store_host(ngx_conf_t *cf, brix_manager_map_t *entry,
29 const char *src, size_t hostlen)
30{
31 entry->host.data = ngx_pnalloc(cf->pool, hostlen + 1);
32 if (entry->host.data == NULL) {
33 return NGX_ERROR;
34 }
35 ngx_memcpy(entry->host.data, src, hostlen);
36 entry->host.data[hostlen] = '\0';
37 entry->host.len = hostlen;
38 return NGX_OK;
39}

41/* ---- Parse and validate a decimal port string into entry->port ----
42 *
43 * WHAT: Parses port_str as base-10 into a port number, requiring a fully
44 * consumed string and a value in 1..65535. On success stores the value in
45 * *out_port and returns NGX_OK; on any parse/range failure emits the emerg
46 * "invalid port" diagnostic (naming the original directive argument arg) and
47 * returns NGX_ERROR.
48 *
49 * WHY: The two endpoint forms both terminate in a colon-delimited port with
50 * identical validation and identical error text; one helper guarantees the
51 * bounds check and message stay identical across both branches.
52 *
53 * HOW:
54 * 1. strtol(port_str, &endp, 10).
55 * 2. Reject trailing garbage (*endp != '\0'), non-positive, or > 65535.
56 * 3. On rejection, emerg-log against arg and return NGX_ERROR.
57 * 4. Otherwise cast to uint16_t, store in *out_port, return NGX_OK.
58 */
59static ngx_int_t
60brix_manager_map_parse_port(ngx_conf_t *cf, const char *port_str,
61 ngx_str_t *arg, uint16_t *out_port)
62{
63 char *endp;
64 long pnum;
65
66 pnum = strtol(port_str, &endp, 10);
67 if (*endp != '\0' || pnum <= 0 || pnum > 65535) {
68 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
69 "brix_manager_map: invalid port in \"%V\"", arg);
70 return NGX_ERROR;
71 }
72 *out_port = (uint16_t) pnum;
73 return NGX_OK;
74}

76/* ---- Parse an IPv6-literal endpoint [addr]:port into a map entry ----
77 *
78 * WHAT: Parses the bracketed IPv6 literal form (addr_copy[0] == '['), storing
79 * the bracket-enclosed host and the trailing port into entry. Returns NGX_OK
80 * on success, or NGX_ERROR (emerg-logged for malformed bracket/port; silent
81 * for allocation failure) naming arg in diagnostics.
82 *
83 * WHY: The IPv6 grammar differs from the plain host:port grammar — it splits
84 * on the closing bracket and requires a ':' immediately after it — so it needs
85 * its own bounds computation before reusing the shared host/port storage.
86 *
87 * HOW:
88 * 1. Locate ']' with strchr; reject a missing/empty ("[]") host.
89 * 2. Require the byte after ']' to be ':', else "missing port".
90 * 3. host = bytes between '[' and ']'; store via store_host.
91 * 4. Parse the port starting two bytes past ']' (skip "]:").
92 */
93static ngx_int_t
94brix_manager_map_parse_ipv6(ngx_conf_t *cf, brix_manager_map_t *entry,
95 char *addr_copy, ngx_str_t *arg)
96{
97 char *rb;
98 size_t hostlen;
99
100 rb = strchr(addr_copy, ']');
101 if (rb == NULL || rb == addr_copy) {
102 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
103 "brix_manager_map: invalid IPv6 host \"%V\"", arg);
104 return NGX_ERROR;
105 }
106 if (*(rb + 1) != ':') {
107 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
108 "brix_manager_map: missing port in \"%V\"", arg);
109 return NGX_ERROR;
110 }
111
112 hostlen = (size_t) (rb - addr_copy - 1);
113 if (brix_manager_map_store_host(cf, entry, addr_copy + 1, hostlen)
114 != NGX_OK)
115 {
116 return NGX_ERROR;
117 }
118
119 return brix_manager_map_parse_port(cf, rb + 2, arg, &entry->port);
120}

122/* ---- Parse an IPv4/hostname endpoint host:port into a map entry ----
123 *
124 * WHAT: Parses the unbracketed host:port form, splitting on the LAST colon so
125 * bare IPv4 or hostnames parse correctly, and stores host/port into entry.
126 * Returns NGX_OK on success, or NGX_ERROR (emerg-logged for a missing port;
127 * silent for allocation failure) naming arg in diagnostics.
128 *
129 * WHY: Splitting on the final colon is what distinguishes this from the IPv6
130 * form and keeps a hostname containing no colon (plus its port) unambiguous;
131 * it must not be confused with the bracketed grammar.
132 *
133 * HOW:
134 * 1. strrchr for the last ':'; reject its absence as "missing port".
135 * 2. host = bytes before that colon; store via store_host.
136 * 3. Parse the port from the byte after the colon.
137 */
138static ngx_int_t
139brix_manager_map_parse_hostport(ngx_conf_t *cf, brix_manager_map_t *entry,
140 char *addr_copy, ngx_str_t *arg)
141{
142 char *colon;
143 size_t hostlen;
144
145 colon = strrchr(addr_copy, ':');
146 if (colon == NULL) {
147 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
148 "brix_manager_map: missing port in \"%V\"", arg);
149 return NGX_ERROR;
150 }
151
152 hostlen = (size_t) (colon - addr_copy);
153 if (brix_manager_map_store_host(cf, entry, addr_copy, hostlen) != NGX_OK) {
154 return NGX_ERROR;
155 }
156
157 return brix_manager_map_parse_port(cf, colon + 1, arg, &entry->port);
158}

160/* ---- Dispatch endpoint parsing by literal form ----
161 *
162 * WHAT: Selects the bracketed IPv6 parser when addr_copy begins with '[',
163 * otherwise the host:port parser, populating entry->host and entry->port.
164 * Returns the chosen parser's NGX_OK/NGX_ERROR.
165 *
166 * WHY: Isolating the one-byte form discriminator keeps the orchestrator flat
167 * and the two grammars in separate, independently reviewable helpers.
168 *
169 * HOW:
170 * 1. If addr_copy[0] == '[', delegate to the IPv6 parser.
171 * 2. Otherwise delegate to the host:port parser.
172 */
173static ngx_int_t
174brix_manager_map_parse_endpoint(ngx_conf_t *cf, brix_manager_map_t *entry,
175 char *addr_copy, ngx_str_t *arg)
176{
177 if (addr_copy[0] == '[') {
178 return brix_manager_map_parse_ipv6(cf, entry, addr_copy, arg);
179 }
180 return brix_manager_map_parse_hostport(cf, entry, addr_copy, arg);
181}

183/* ---- Append one prefix->endpoint entry to xcf->manager_map ----
184 *
185 * WHAT: Lazily creates xcf->manager_map, pushes a zeroed entry, normalises the
186 * prefix (policy conventions), parses the endpoint (IPv4/hostname or
187 * [IPv6]:port, port 1..65535), and logs a NOTICE. Returns NGX_CONF_OK, or
188 * NGX_CONF_ERROR (emerg-logged for user-facing failures) on any error.
189 *
190 * WHY: This is the `brix_manager_map <prefix> <host:port>` directive handler
191 * for CMS manager mode; the endpoint parsing is delegated to helpers so this
192 * stays a flat early-return orchestrator.
193 *
194 * HOW:
195 * 1. Ensure the array exists; push and zero a fresh entry.
196 * 2. Normalise value[1] into entry->prefix.
197 * 3. Copy value[2] into a NUL-terminated buffer for parsing.
198 * 4. Parse the endpoint into entry->host/entry->port.
199 * 5. Log the configured mapping and return NGX_CONF_OK.
200 */
201char *
202brix_conf_set_manager_map(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
203{
204 ngx_stream_brix_srv_conf_t *xcf = conf;
205 ngx_str_t *value;
206 brix_manager_map_t *entry;
207 char *addr_copy;
208
209 value = cf->args->elts;
210 (void) cmd;
211
212 if (xcf->manager_map == NULL) {
213 xcf->manager_map = ngx_array_create(cf->pool, 2,
214 sizeof(brix_manager_map_t));
215 if (xcf->manager_map == NULL) {
216 return NGX_CONF_ERROR;
217 }
218 }
219
220 entry = ngx_array_push(xcf->manager_map);
221 if (entry == NULL) {
222 return NGX_CONF_ERROR;
223 }
224
225 ngx_memzero(entry, sizeof(*entry));
226
227 /* Normalize and store the prefix path (policy-style) */
228 if (brix_normalize_policy_path(cf->pool, &value[1], &entry->prefix) != NGX_OK) {
229 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
230 "brix_manager_map: invalid path \"%V\"", &value[1]);
231 return NGX_CONF_ERROR;
232 }
233
234 /* Copy the host:port argument into a NUL-terminated buffer for parsing. */
235 addr_copy = brix_pstrdup_z(cf->pool, &value[2]);
236 if (addr_copy == NULL) {
237 return NGX_CONF_ERROR;
238 }
239
240 if (brix_manager_map_parse_endpoint(cf, entry, addr_copy, &value[2])
241 != NGX_OK)
242 {
243 return NGX_CONF_ERROR;
244 }
245
246 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
247 "brix: manager_map configured: prefix=%s backend=%s:%d",
248 (char *) entry->prefix.data, (char *) entry->host.data, (int) entry->port);
249
250 return NGX_CONF_OK;
251}
