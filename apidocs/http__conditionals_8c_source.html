1/*
2 * http_conditionals.c - shared HTTP conditional request checks (RFC 7232).
3 *
4 * WHAT: Evaluates If-None-Match, If-Match, If-Modified-Since, and Overwrite headers
5 * against resource metadata. Returns 304 Not Modified, 412 Precondition Failed,
6 * or NGX_OK (conditions pass) for each conditional type.
7 *
8 * WHY: RFC 7232 §3 requires servers to validate conditional requests before returning
9 * content. WebDAV PUT/MOVE/COPY and S3 GET/PUT all use these checks. Shared
10 * implementation ensures consistent behaviour across both modules.
11 *
12 * HOW: ETag path generates resource ETag from mtime/size, then compares against header
13 * list entries (case-insensitive, weak-equivalence support). Time path parses
14 * HTTP date via ngx_parse_http_time(), compares against st_mtime. Overwrite
15 * checks header value equals "F".
16*/
17

24/*
25 * brix_http_strip_weak - strip W/ prefix from ETag string.
26 *
27 * WHAT: If the data starts with "W/", advances data pointer by 2 and reduces len
28 * by 2. Otherwise leaves data and len unchanged.
29 *
30 * WHY: RFC 7232 §2.3 defines weak equivalence: W/"etag1" ≡ "etag1" for If-None-Match
31 * comparison. This helper normalises both resource ETag and header entries before
32 * string comparison when flags include BRIX_HTTP_COND_WEAK_EQUIV.
33 *
34 * HOW: len >= 2 && data[0]=='W' && data[1]=='/' → data+=2, len-=2.
35 */
36
37static void
38brix_http_strip_weak(const char **data, size_t *len)
39{
40 if (*len >= 2 && (*data)[0] == 'W' && (*data)[1] == '/') {
41 *data += 2;
42 *len -= 2;
43 }
44}

46ngx_int_t
47brix_http_etag_list_contains(const ngx_str_t *header, const char *etag,
48 unsigned flags)
49{
50 const char *p;
51 const char *end;
52 const char *etag_data;
53 size_t etag_len;
54
55 if (header == NULL || header->data == NULL || etag == NULL) {
56 return NGX_DECLINED;
57 }
58
59 p = (const char *) header->data;
60 end = p + header->len;
61 etag_data = etag;
62 etag_len = strlen(etag);
63
64 if (flags & BRIX_HTTP_COND_WEAK_EQUIV) {
65 brix_http_strip_weak(&etag_data, &etag_len);
66 }
67
68 while (p < end) {
69 const char *tok_start;
70 const char *tok_end;
71 size_t tok_len;
72
73 while (p < end && (*p == ' ' || *p == '\t' || *p == ',')) {
74 p++;
75 }
76 if (p >= end) {
77 break;
78 }
79
80 tok_start = p;
81 while (p < end && *p != ',') {
82 p++;
83 }
84 tok_end = p;
85
86 while (tok_end > tok_start
87 && (tok_end[-1] == ' ' || tok_end[-1] == '\t'))
88 {
89 tok_end--;
90 }
91
92 tok_len = (size_t) (tok_end - tok_start);
93 if (flags & BRIX_HTTP_COND_WEAK_EQUIV) {
94 brix_http_strip_weak(&tok_start, &tok_len);
95 }
96
97 if (tok_len == etag_len
98 && ngx_strncmp(tok_start, etag_data, tok_len) == 0)
99 {
100 return NGX_OK;
101 }
102 }
103
104 return NGX_DECLINED;
105}

107ngx_int_t
108brix_http_check_etag_preconditions(ngx_http_request_t *r,
109 ngx_flag_t resource_exists, const struct stat *st, unsigned etag_flags,
110 unsigned condition_flags)
111{
112 ngx_table_elt_t *if_match;
113 ngx_table_elt_t *if_none_match;
114 char etag_buf[64];
115 const char *etag;
116
117 if_match = r->headers_in.if_match;
118 if_none_match = r->headers_in.if_none_match;
119
120 if (if_match == NULL && if_none_match == NULL) {
121 return NGX_OK;
122 }
123
124 etag = NULL;
125 if (resource_exists && st != NULL) {
126 brix_http_etag_str(etag_buf, sizeof(etag_buf),
127 st->st_mtime, st->st_size, etag_flags);
128 etag = etag_buf;
129 }
130
131 if (if_match != NULL) {
132 if (!resource_exists) {
133 return NGX_HTTP_PRECONDITION_FAILED;
134 }
135
136 if (if_match->value.len == 1 && if_match->value.data[0] == '*') {
137 /* wildcard match passed; If-None-Match below may still fail */
138 } else if (brix_http_etag_list_contains(&if_match->value, etag,
139 condition_flags) != NGX_OK)
140 {
141 return NGX_HTTP_PRECONDITION_FAILED;
142 }
143 }
144
145 if (if_none_match != NULL) {
146 if (if_none_match->value.len == 1
147 && if_none_match->value.data[0] == '*')
148 {
149 return resource_exists ? NGX_HTTP_PRECONDITION_FAILED : NGX_OK;
150 }
151
152 if (resource_exists
153 && brix_http_etag_list_contains(&if_none_match->value, etag,
154 condition_flags) == NGX_OK)
155 {
156 return NGX_HTTP_PRECONDITION_FAILED;
157 }
158 }
159
160 return NGX_OK;
161}

163/*
164 * cond_header_present - is a conditional header usable?
165 *
166 * WHAT: True when the header exists with a non-empty value.
167 * WHY: An empty conditional header is invalid per RFC 9110; treating it as
168 * absent (rather than as a never-matching list) keeps a malformed client
169 * from turning every request into a 412.
170 * HOW: NULL / len check.
171 */
172static ngx_flag_t
173cond_header_present(const ngx_table_elt_t *h)
174{
175 return h != NULL && h->value.len > 0;
176}
177
178/*
179 * cond_is_wildcard - is the header value the `*` any-representation token?
180 *
181 * WHAT: True for a value that is exactly "*".
182 * WHY: RFC 9110 §13.1.1/§13.1.2: `*` selects any existing representation and
183 * must be recognised before list matching.
184 * HOW: Length-1 compare (nginx has already trimmed surrounding whitespace).
185 */
186static ngx_flag_t
187cond_is_wildcard(const ngx_str_t *v)
188{
189 return v->len == 1 && v->data[0] == '*';
190}

192ngx_int_t
193brix_http_eval_preconditions(ngx_http_request_t *r,
194 ngx_flag_t resource_exists, time_t mtime, off_t size,
195 unsigned etag_flags, unsigned condition_flags)
196{
197 ngx_table_elt_t *if_match = r->headers_in.if_match;
198 ngx_table_elt_t *if_none = r->headers_in.if_none_match;
199 ngx_table_elt_t *if_unmod = r->headers_in.if_unmodified_since;
200 ngx_table_elt_t *if_mod = r->headers_in.if_modified_since;
201 unsigned read_mode = condition_flags & BRIX_HTTP_COND_READ;
202 unsigned time_mode = condition_flags & BRIX_HTTP_COND_TIME;
203 char etag_buf[64];
204 const char *etag = NULL;
205
206 if (resource_exists) {
207 brix_http_etag_str(etag_buf, sizeof(etag_buf), mtime, size,
208 etag_flags);
209 etag = etag_buf;
210 }
211
212 /* 1. If-Match — fails (412) unless the representation exists and the
213 * validator selects it. */
214 if (cond_header_present(if_match)) {
215 if (!resource_exists) {
216 return NGX_HTTP_PRECONDITION_FAILED;
217 }
218 if (!cond_is_wildcard(&if_match->value)
219 && brix_http_etag_list_contains(&if_match->value, etag,
220 condition_flags) != NGX_OK)
221 {
222 return NGX_HTTP_PRECONDITION_FAILED;
223 }
224 } else if (time_mode && cond_header_present(if_unmod)) {
225 /* 2. If-Unmodified-Since — only when If-Match is absent. */
226 time_t t = ngx_parse_http_time(if_unmod->value.data,
227 if_unmod->value.len);
228 if (t != (time_t) NGX_ERROR && mtime > t) {
229 return NGX_HTTP_PRECONDITION_FAILED;
230 }
231 }
232
233 /* 3. If-None-Match — a match means the client already holds (or must not
234 * overwrite) this representation: 304 on reads, 412 on writes. */
235 if (cond_header_present(if_none)) {
236 ngx_flag_t selected;
237
238 if (cond_is_wildcard(&if_none->value)) {
239 selected = resource_exists;
240 } else {
241 selected = resource_exists
242 && brix_http_etag_list_contains(&if_none->value, etag,
243 condition_flags)
244 == NGX_OK;
245 }
246 if (selected) {
247 return read_mode ? NGX_HTTP_NOT_MODIFIED
248 : NGX_HTTP_PRECONDITION_FAILED;
249 }
250 } else if (read_mode && time_mode && cond_header_present(if_mod)) {
251 /* 4. If-Modified-Since — reads only, when If-None-Match is absent.
252 * `before` semantics: not modified since the given date ⇒ 304. */
253 time_t t = ngx_parse_http_time(if_mod->value.data, if_mod->value.len);
254 if (t != (time_t) NGX_ERROR && mtime <= t) {
255 return NGX_HTTP_NOT_MODIFIED;
256 }
257 }
258
259 return NGX_OK;
260}

262ngx_int_t
263brix_http_check_if_modified_since(ngx_http_request_t *r, time_t mtime)
264{
265 ngx_str_t ims;
266 time_t ims_time;
267
268 if (r->headers_in.if_modified_since == NULL) {
269 return NGX_OK;
270 }
271
272 ims = r->headers_in.if_modified_since->value;
273 ims_time = ngx_parse_http_time(ims.data, ims.len);
274 if (ims_time != NGX_ERROR && mtime <= ims_time) {
275 return NGX_HTTP_NOT_MODIFIED;
276 }
277
278 return NGX_OK;
279}
