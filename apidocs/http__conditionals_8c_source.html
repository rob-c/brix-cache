1/*
2 * http_conditionals.c - shared HTTP conditional request checks (RFC 7232).
3 *
4 * WHAT: Evaluates If-None-Match, If-Match, If-Modified-Since, and Overwrite headers
5 * against resource metadata. Returns 304 Not Modified, 412 Precondition Failed,
6 * or NGX_OK (conditions pass) for each conditional type.
7 *
8 * WHY: RFC 7232 §3 requires servers to validate conditional requests before returning
9 * content. WebDAV PUT/MOVE/COPY and S3 GET/PUT all use these checks. Shared
10 * implementation ensures consistent behaviour across both modules.
11 *
12 * HOW: ETag path generates resource ETag from mtime/size, then compares against header
13 * list entries (case-insensitive, weak-equivalence support). Time path parses
14 * HTTP date via ngx_parse_http_time(), compares against st_mtime. Overwrite
15 * checks header value equals "F".
16*/

24/*
25 * brix_http_strip_weak - strip W/ prefix from ETag string.
26 *
27 * WHAT: If the data starts with "W/", advances data pointer by 2 and reduces len
28 * by 2. Otherwise leaves data and len unchanged.
29 *
30 * WHY: RFC 7232 §2.3 defines weak equivalence: W/"etag1" ≡ "etag1" for If-None-Match
31 * comparison. This helper normalises both resource ETag and header entries before
32 * string comparison when flags include BRIX_HTTP_COND_WEAK_EQUIV.
33 *
34 * HOW: len >= 2 && data[0]=='W' && data[1]=='/' → data+=2, len-=2.
35 */
36
37static void
38brix_http_strip_weak(const char **data, size_t *len)
39{
40 if (*len >= 2 && (*data)[0] == 'W' && (*data)[1] == '/') {
41 *data += 2;
42 *len -= 2;
43 }
44}

46/*
47 * brix_http_next_etag_token - pull the next comma-separated entry from an ETag list.
48 *
49 * WHAT: Advances *cursor past leading separators, then captures one entry up to the
50 * next comma (trailing spaces/tabs trimmed). Returns 1 with tok_start/tok_len set
51 * when an entry was found, 0 (cursor at end) when the list is exhausted.
52 *
53 * WHY: If-Match / If-None-Match carry a comma-separated list of entity-tags
54 * (RFC 7232 §2.3, §3.1). Isolating the tokeniser keeps the comparison loop in
55 * brix_http_etag_list_contains flat and independently testable.
56 *
57 * HOW: 1. Skip leading ' ', '\t', ','. 2. If at end, publish cursor and return 0.
58 * 3. Mark start; scan to next ',' or end. 4. Trim trailing OWS. 5. Publish
59 * cursor, token start and length; return 1.
60 */
61
62static ngx_flag_t
63brix_http_next_etag_token(const char **cursor, const char *end,
64 const char **tok_start, size_t *tok_len)
65{
66 const char *p = *cursor;
67 const char *start;
68 const char *stop;
69
70 while (p < end && (*p == ' ' || *p == '\t' || *p == ',')) {
71 p++;
72 }
73 if (p >= end) {
74 *cursor = p;
75 return 0;
76 }
77
78 start = p;
79 while (p < end && *p != ',') {
80 p++;
81 }
82 stop = p;
83
84 while (stop > start && (stop[-1] == ' ' || stop[-1] == '\t')) {
85 stop--;
86 }
87
88 *cursor = p;
89 *tok_start = start;
90 *tok_len = (size_t) (stop - start);
91 return 1;
92}

94ngx_int_t
95brix_http_etag_list_contains(const ngx_str_t *header, const char *etag,
96 unsigned flags)
97{
98 const char *cursor;
99 const char *end;
100 const char *etag_data;
101 size_t etag_len;
102 const char *tok_start;
103 size_t tok_len;
104
105 if (header == NULL || header->data == NULL || etag == NULL) {
106 return NGX_DECLINED;
107 }
108
109 cursor = (const char *) header->data;
110 end = cursor + header->len;
111 etag_data = etag;
112 etag_len = strlen(etag);
113
114 if (flags & BRIX_HTTP_COND_WEAK_EQUIV) {
115 brix_http_strip_weak(&etag_data, &etag_len);
116 }
117
118 while (brix_http_next_etag_token(&cursor, end, &tok_start, &tok_len)) {
119 if (flags & BRIX_HTTP_COND_WEAK_EQUIV) {
120 brix_http_strip_weak(&tok_start, &tok_len);
121 }
122
123 if (tok_len == etag_len
124 && ngx_strncmp(tok_start, etag_data, tok_len) == 0)
125 {
126 return NGX_OK;
127 }
128 }
129
130 return NGX_DECLINED;
131}

133/*
134 * cond_header_present - is a conditional header usable?
135 *
136 * WHAT: True when the header exists with a non-empty value.
137 * WHY: An empty conditional header is invalid per RFC 9110; treating it as
138 * absent (rather than as a never-matching list) keeps a malformed client
139 * from turning every request into a 412.
140 * HOW: NULL / len check.
141 */
142static ngx_flag_t
143cond_header_present(const ngx_table_elt_t *h)
144{
145 return h != NULL && h->value.len > 0;
146}

148/*
149 * cond_is_wildcard - is the header value the `*` any-representation token?
150 *
151 * WHAT: True for a value that is exactly "*".
152 * WHY: RFC 9110 §13.1.1/§13.1.2: `*` selects any existing representation and
153 * must be recognised before list matching.
154 * HOW: Length-1 compare (nginx has already trimmed surrounding whitespace).
155 */
156static ngx_flag_t
157cond_is_wildcard(const ngx_str_t *v)
158{
159 return v->len == 1 && v->data[0] == '*';
160}

162/*
163 * brix_http_validator_selects - does an entity-tag validator select this resource?
164 *
165 * WHAT: Returns true when the header value (a `*` wildcard or a comma-separated
166 * entity-tag list) selects the current representation. A missing resource is
167 * never selected.
168 *
169 * WHY: RFC 7232 §3.1/§3.2 define the same "does this validator match the selected
170 * representation" test for both If-Match and If-None-Match — the two differ only
171 * in what they do with the answer. Sharing the predicate keeps that answer
172 * byte-identical across every conditional callsite.
173 *
174 * HOW: 1. No representation ⇒ not selected. 2. `*` ⇒ selected (a representation
175 * exists). 3. Otherwise the list must contain the resource's entity-tag.
176 */
177
178static ngx_flag_t
179brix_http_validator_selects(const ngx_str_t *value, ngx_flag_t resource_exists,
180 const char *etag, unsigned condition_flags)
181{
182 if (!resource_exists) {
183 return 0;
184 }
185 if (cond_is_wildcard(value)) {
186 return 1;
187 }
188 return brix_http_etag_list_contains(value, etag, condition_flags) == NGX_OK;
189}

191/*
192 * brix_http_eval_if_unmodified_since - apply an If-Unmodified-Since precondition.
193 *
194 * WHAT: Returns NGX_HTTP_PRECONDITION_FAILED (412) when the resource was modified
195 * after the header date, otherwise NGX_DECLINED (precondition passes / proceed).
196 *
197 * WHY: RFC 7232 §3.4: a write must not proceed if the representation changed since the
198 * client last saw it. An unparseable date is ignored (treated as passing) per the
199 * same section.
200 *
201 * HOW: Parse the HTTP date; if valid and mtime is strictly newer, fail with 412.
202 */
203
204static ngx_int_t
205brix_http_eval_if_unmodified_since(const ngx_table_elt_t *if_unmod, time_t mtime)
206{
207 time_t t = ngx_parse_http_time(if_unmod->value.data, if_unmod->value.len);
208
209 if (t != (time_t) NGX_ERROR && mtime > t) {
210 return NGX_HTTP_PRECONDITION_FAILED;
211 }
212 return NGX_DECLINED;
213}

215/*
216 * brix_http_eval_if_modified_since - apply an If-Modified-Since precondition.
217 *
218 * WHAT: Returns NGX_HTTP_NOT_MODIFIED (304) when the resource has not been modified
219 * since the header date, otherwise NGX_DECLINED (precondition passes / proceed).
220 *
221 * WHY: RFC 7232 §3.3: a GET may be short-circuited to 304 when the client's cached
222 * copy is still current. An unparseable date is ignored (treated as modified).
223 *
224 * HOW: Parse the HTTP date; if valid and mtime is at or before it, return 304.
225 */
226
227static ngx_int_t
228brix_http_eval_if_modified_since(const ngx_table_elt_t *if_mod, time_t mtime)
229{
230 time_t t = ngx_parse_http_time(if_mod->value.data, if_mod->value.len);
231
232 if (t != (time_t) NGX_ERROR && mtime <= t) {
233 return NGX_HTTP_NOT_MODIFIED;
234 }
235 return NGX_DECLINED;
236}

238ngx_int_t
239brix_http_check_etag_preconditions(ngx_http_request_t *r,
240 ngx_flag_t resource_exists, const struct stat *st, unsigned etag_flags,
241 unsigned condition_flags)
242{
243 ngx_table_elt_t *if_match;
244 ngx_table_elt_t *if_none_match;
245 char etag_buf[64];
246 const char *etag;
247
248 if_match = r->headers_in.if_match;
249 if_none_match = r->headers_in.if_none_match;
250
251 if (if_match == NULL && if_none_match == NULL) {
252 return NGX_OK;
253 }
254
255 etag = NULL;
256 if (resource_exists && st != NULL) {
257 brix_http_etag_str(etag_buf, sizeof(etag_buf),
258 st->st_mtime, st->st_size, etag_flags);
259 etag = etag_buf;
260 }
261
262 /* If-Match: fail unless the validator selects the existing representation. */
263 if (if_match != NULL
264 && !brix_http_validator_selects(&if_match->value, resource_exists, etag,
265 condition_flags))
266 {
267 return NGX_HTTP_PRECONDITION_FAILED;
268 }
269
270 /* If-None-Match: fail when the validator DOES select it (write semantics). */
271 if (if_none_match != NULL
272 && brix_http_validator_selects(&if_none_match->value, resource_exists,
273 etag, condition_flags))
274 {
275 return NGX_HTTP_PRECONDITION_FAILED;
276 }
277
278 return NGX_OK;
279}

281/*
282 * brix_http_eval_match_group - first RFC-7232 precedence step (If-Match, else
283 * If-Unmodified-Since).
284 *
285 * WHAT: Returns NGX_HTTP_PRECONDITION_FAILED (412) when the step's precondition fails,
286 * otherwise NGX_DECLINED (proceed to the If-None-Match step).
287 *
288 * WHY: RFC 7232 §6 fixes the order: If-Match takes precedence and, when it is absent,
289 * If-Unmodified-Since is evaluated in its place (the two are mutually exclusive at
290 * this step). Encapsulating the pair preserves that "else" relationship exactly.
291 *
292 * HOW: 1. If-Match present ⇒ 412 unless the validator selects the representation;
293 * otherwise proceed. 2. Else, in time mode, an If-Unmodified-Since header is
294 * applied. 3. Neither present ⇒ proceed.
295 */
296
297static ngx_int_t
298brix_http_eval_match_group(ngx_table_elt_t *if_match, ngx_table_elt_t *if_unmod,
299 ngx_flag_t resource_exists, const char *etag, unsigned condition_flags,
300 unsigned time_mode, time_t mtime)
301{
302 if (cond_header_present(if_match)) {
303 if (!brix_http_validator_selects(&if_match->value, resource_exists, etag,
304 condition_flags))
305 {
306 return NGX_HTTP_PRECONDITION_FAILED;
307 }
308 return NGX_DECLINED;
309 }
310
311 if (time_mode && cond_header_present(if_unmod)) {
312 return brix_http_eval_if_unmodified_since(if_unmod, mtime);
313 }
314
315 return NGX_DECLINED;
316}

318/*
319 * brix_http_eval_none_match_group - second RFC-7232 precedence step (If-None-Match,
320 * else If-Modified-Since).
321 *
322 * WHAT: Returns NGX_HTTP_NOT_MODIFIED (304) or NGX_HTTP_PRECONDITION_FAILED (412) when
323 * the step's precondition triggers, otherwise NGX_DECLINED (conditions pass).
324 *
325 * WHY: RFC 7232 §6 evaluates If-None-Match after the If-Match step and, when it is
326 * absent, evaluates If-Modified-Since in its place (reads only). A selecting
327 * If-None-Match means the client already holds the representation: 304 on reads,
328 * 412 on writes.
329 *
330 * HOW: 1. If-None-Match present ⇒ if the validator selects it, 304 for reads / 412 for
331 * writes; otherwise proceed. 2. Else, for reads in time mode, apply
332 * If-Modified-Since. 3. Neither present ⇒ proceed.
333 */
334
335static ngx_int_t
336brix_http_eval_none_match_group(ngx_table_elt_t *if_none,
337 ngx_table_elt_t *if_mod, ngx_flag_t resource_exists, const char *etag,
338 unsigned condition_flags, unsigned read_mode, unsigned time_mode,
339 time_t mtime)
340{
341 if (cond_header_present(if_none)) {
342 if (brix_http_validator_selects(&if_none->value, resource_exists, etag,
343 condition_flags))
344 {
345 return read_mode ? NGX_HTTP_NOT_MODIFIED
346 : NGX_HTTP_PRECONDITION_FAILED;
347 }
348 return NGX_DECLINED;
349 }
350
351 if (read_mode && time_mode && cond_header_present(if_mod)) {
352 return brix_http_eval_if_modified_since(if_mod, mtime);
353 }
354
355 return NGX_DECLINED;
356}

358ngx_int_t
359brix_http_eval_preconditions(ngx_http_request_t *r,
360 ngx_flag_t resource_exists, time_t mtime, off_t size,
361 unsigned etag_flags, unsigned condition_flags)
362{
363 ngx_table_elt_t *if_match = r->headers_in.if_match;
364 ngx_table_elt_t *if_none = r->headers_in.if_none_match;
365 ngx_table_elt_t *if_unmod = r->headers_in.if_unmodified_since;
366 ngx_table_elt_t *if_mod = r->headers_in.if_modified_since;
367 unsigned read_mode = condition_flags & BRIX_HTTP_COND_READ;
368 unsigned time_mode = condition_flags & BRIX_HTTP_COND_TIME;
369 char etag_buf[64];
370 const char *etag = NULL;
371 ngx_int_t decision;
372
373 if (resource_exists) {
374 brix_http_etag_str(etag_buf, sizeof(etag_buf), mtime, size,
375 etag_flags);
376 etag = etag_buf;
377 }
378
379 /* Step 1: If-Match, or (absent) If-Unmodified-Since — may 412. */
380 decision = brix_http_eval_match_group(if_match, if_unmod, resource_exists,
381 etag, condition_flags, time_mode,
382 mtime);
383 if (decision != NGX_DECLINED) {
384 return decision;
385 }
386
387 /* Step 2: If-None-Match, or (absent) If-Modified-Since — may 304/412. */
388 decision = brix_http_eval_none_match_group(if_none, if_mod, resource_exists,
389 etag, condition_flags, read_mode,
390 time_mode, mtime);
391 if (decision != NGX_DECLINED) {
392 return decision;
393 }
394
395 return NGX_OK;
396}
