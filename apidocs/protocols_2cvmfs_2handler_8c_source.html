1/* handler.c — the cvmfs:// content handler.
2 *
3 * WHAT: entry point for every request on an brix_cvmfs-enabled location:
4 * ctx setup → gate (method/class/reject/geo) → cache-tier open-or-fill
5 * → file response.
6 * WHY: cvmfs:// is a dedicated protocol — this handler owns the request
7 * end-to-end; nothing routes through the WebDAV dispatch.
8 * HOW: everything below the protocol seam is shared machinery: the serve
9 * step is the same offload → coalesced-fill → brix_vfs_open →
10 * brix_http_serve_file_ranged composition the WebDAV GET path
11 * drives, with Range/HEAD/conditional semantics from the shared
12 * helpers. The proxy-mode (T14) per-upstream override rides on the
13 * request ctx (convention #2).
14 */

32/* CVMFS-flavoured errno → HTTP status: origin-side trouble is a GATEWAY
33 * error (convention #3 — the origin transfer was bad, not the client
34 * request), everything else follows the shared table. */
35static ngx_uint_t
36cvmfs_errno_status(int err)
37{
38 if (err == ENOENT || err == ENOTDIR || err == ENAMETOOLONG) {
39 return NGX_HTTP_NOT_FOUND;
40 }
41 if (err == EACCES || err == EPERM || err == EXDEV || err == ELOOP) {
42 return NGX_HTTP_FORBIDDEN;
43 }
44 if (err == EIO) {
45 return NGX_HTTP_BAD_GATEWAY;
46 }
47 return brix_http_errno_to_status(err);
48}

50/* Re-entry trampoline for the off-loop cache fill: after the fill lands the

54cvmfs_reenter(ngx_http_request_t *r, void *data)

76/* Serve a classified CAS GET/HEAD from the tier: offload a socket-wire
77 * serve, coalesce a miss into one off-loop fill, then open + respond via
78 * the shared ranged-file pipeline. */
79static ngx_int_t
80cvmfs_tier_get(ngx_http_request_t *r, ngx_http_brix_cvmfs_loc_conf_t *lcf)
81{
82 ngx_http_brix_cvmfs_ctx_t *ctx =
83 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
84 char path[PATH_MAX];
85 const char *root, *key;
86 brix_vfs_ctx_t vctx;
87 brix_vfs_file_t *fh;
88 brix_vfs_stat_t vst;
89 int vfs_err = 0;
90 int is_tls = 0;
91 ngx_int_t rc;
92
93 root = (ctx != NULL && ctx->up_root != NULL) ? ctx->up_root
94 : lcf->common.root_canon;
95
96 /* fs path = export root + uri; a "/" root (the pure-cache-node anchor)
97 * contributes nothing so the path IS the uri. */
98 if (root[0] == '/' && root[1] == '\0') {
99 if (brix_str_cbuf(path, sizeof(path), &r->uri) == NULL) {
100 return NGX_HTTP_REQUEST_URI_TOO_LARGE;
101 }
102 } else {
103 size_t rn = ngx_strlen(root);
104
105 if (rn + r->uri.len >= sizeof(path)) {
106 return NGX_HTTP_REQUEST_URI_TOO_LARGE;
107 }
108 ngx_memcpy(path, root, rn);
109 ngx_memcpy(path + rn, r->uri.data, r->uri.len);
110 path[rn + r->uri.len] = '\0';
111 }
112
113#if (NGX_HTTP_SSL)
114 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
115#endif
116 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log,
117 BRIX_PROTO_CVMFS, root, "", /* allow_write */ 0, is_tls, NULL,
118 path);
119 vctx.sd = cvmfs_resolve_sd(r, lcf);
120 if (vctx.sd == NULL) {
121 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
122 "cvmfs: no storage backend registered for \"%s\" - check "
123 "brix_storage_backend", root);
124 return NGX_HTTP_INTERNAL_SERVER_ERROR;
125 }
126 key = brix_vfs_export_relative(&vctx, path);
127
128 /* Dashboard live-transfer record from TIER ENTRY, not serve time: the
129 * record then spans a cold fill's whole origin wait, so in-flight cvmfs
130 * fills are visible in the transfer table (the serve pipeline's own
131 * start is idempotent and rebinds this same slot; rejects never got
132 * here, so they never occupy one). Auto-freed with the request pool.
133 * `key` is the client-visible /cvmfs/... path — the synthetic proxy-mode
134 * registry root must not leak into the operator display. */
135 (void) brix_dashboard_http_start_identity(r, key, "anonymous", "",
136 BRIX_XFER_PROTO_CVMFS, BRIX_XFER_DIR_READ, "GET", -1);
137
138 /* socket-wire backends can't open/read on the event loop */
139 {
140 brix_http_serve_opts_t sopts;
141
142 ngx_memzero(&sopts, sizeof(sopts));
143 sopts.xfer_proto = BRIX_XFER_PROTO_CVMFS;
144 sopts.op_name = "GET";
145 sopts.identity = "anonymous";
146 sopts.etag_flags = BRIX_ETAG_WEAK;
147
148 rc = brix_http_serve_offload_remote(r, vctx.sd, key, path, &sopts,
149 &lcf->common, NULL);
150 if (rc == NGX_DONE) {
151 return NGX_DONE;
152 }
153 if (rc == NGX_ERROR) {
154 return NGX_HTTP_INTERNAL_SERVER_ERROR;
155 }
156 }
157
158 /* miss → one coalesced off-loop fill, re-entering this handler on land */
159 rc = brix_http_cache_fill_if_needed(r, vctx.sd, key, &lcf->common,
160 cvmfs_reenter, NULL);
161 if (rc == NGX_DONE) {
162 if (ctx != NULL) {
163 ctx->cache_status = BRIX_CVMFS_CACHE_FILL; /* $cvmfs_cache */
164 }
165 return NGX_DONE;
166 }
167 if (rc == NGX_ERROR) {
168 return NGX_HTTP_INTERNAL_SERVER_ERROR;
169 }
170
171 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
172 if (fh == NULL) {
173 return (ngx_int_t) cvmfs_errno_status(vfs_err);
174 }
175 if (ctx != NULL && ctx->cache_status == BRIX_CVMFS_CACHE_NONE) {
176 ctx->cache_status = BRIX_CVMFS_CACHE_HIT; /* $cvmfs_cache */
177 }
178 if (brix_vfs_file_stat(fh, &vst) != NGX_OK) {
179 brix_vfs_close(fh, r->connection->log);
180 return NGX_HTTP_INTERNAL_SERVER_ERROR;
181 }
182 if (vst.is_directory) {
183 brix_vfs_close(fh, r->connection->log);
184 return NGX_HTTP_FORBIDDEN;
185 }
186
187 rc = brix_http_check_if_modified_since(r, vst.mtime);
188 if (rc == NGX_HTTP_NOT_MODIFIED) {
189 brix_vfs_close(fh, r->connection->log);
190 r->headers_out.status = NGX_HTTP_NOT_MODIFIED;
191 r->headers_out.content_length_n = 0;
192 ngx_http_send_header(r);
193 return ngx_http_send_special(r, NGX_HTTP_LAST);
194 }
195 if (rc != NGX_OK) {
196 brix_vfs_close(fh, r->connection->log);
197 return rc;
198 }
199
200 r->allow_ranges = 1;
201
202 {
203 brix_http_serve_opts_t opts;
204 brix_http_serve_result_t result;
205
206 ngx_memzero(&opts, sizeof(opts));
207 opts.xfer_proto = BRIX_XFER_PROTO_CVMFS;
208 opts.op_name = "GET";
209 opts.identity = "anonymous";
210 opts.etag_flags = BRIX_ETAG_WEAK;
211
212 rc = brix_http_serve_file_ranged(r, fh, &vst, path, &opts, &result);
213 }
214 return rc;
215}

217/* Request-finalization observer: fires once when the request pool is torn
218 * down, with the FINAL response status — the one place every serve path
219 * (inline open, off-loop fill, passthrough) converges, so the negative
220 * memo (T13) sees every 404 regardless of which path produced it. */
221static void
222cvmfs_finalize_observe(void *data)
223{
224 ngx_http_request_t *r = data;
225 ngx_http_brix_cvmfs_loc_conf_t *lcf =
226 ngx_http_get_module_loc_conf(r, ngx_http_brix_cvmfs_module);
227 ngx_http_brix_cvmfs_ctx_t *ctx =
228 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
229 ngx_uint_t status = r->headers_out.status;
230
231 if (lcf != NULL) {
232 brix_cvmfs_notify_status(r, lcf, status);
233 }
234
235 /* Optional client-op trace: one line per classified request naming the
236 * traffic class, repository, path and final cache disposition + status.
237 * DEBUG normally (visible under error_log … debug), promoted to INFO by
238 * brix_cvmfs_trace. Correlates with the upstream-request line by path. */
239 if (ctx != NULL && lcf != NULL) {
240 static const char *cls_names[] = { "cas", "manifest", "geo", "reject" };
241 static const char *cache_names[] = { "-", "hit", "fill", "neg" };
242 ngx_uint_t level = lcf->cvmfs.trace ? NGX_LOG_INFO : NGX_LOG_DEBUG;
243
244 if (r->connection->log->log_level >= level) {
245 char safe[1024];
246 char raw[1024];
247 size_t n = ngx_min(r->uri.len, sizeof(raw) - 1);
248
249 /* r->uri.data is NOT NUL-terminated (points into the request
250 * buffer); copy the exact uri span before sanitizing. */
251 ngx_memcpy(raw, r->uri.data, n);
252 raw[n] = '\0';
253 brix_sanitize_log_string(raw, safe, sizeof(safe));
254 ngx_log_error(level, r->connection->log, 0,
255 "cvmfs-trace: client id=%uA class=%s repo=%*s path=%s "
256 "cache=%s status=%ui",
257 r->connection->number,
258 cls_names[ctx->url.cls <= CVMFS_URL_REJECT ? ctx->url.cls : 3],
259 ctx->url.repo != NULL ? ctx->url.repo_len : (size_t) 0,
260 ctx->url.repo != NULL ? ctx->url.repo : "",
261 safe,
262 cache_names[ctx->cache_status <= BRIX_CVMFS_CACHE_NEG
263 ? ctx->cache_status : 0],
264 status);
265 }
266 }
267
268 /* T16: fill/byte accounting off the FINAL disposition + status. The
269 * byte counts use the response body length (content_length_n) — the
270 * serve pipeline set it before headers went out. */
271 if (ctx == NULL) {
272 return;
273 }
274 if (ctx->cache_status == BRIX_CVMFS_CACHE_FILL) {
275 if (status == NGX_HTTP_OK || status == NGX_HTTP_PARTIAL_CONTENT) {
276 BRIX_CVMFS_METRIC_INC(fills_total);
277 if (r->headers_out.content_length_n > 0) {
278 BRIX_CVMFS_METRIC_ADD(bytes_served_fill_total,
279 (ngx_atomic_uint_t) r->headers_out.content_length_n);
280 }
281 brix_metric_cache_result(BRIX_PROTO_CVMFS, 0, 0);
282 if (ctx->repo != NULL) {
283 BRIX_ATOMIC_INC(&ctx->repo->fills_total);
284 BRIX_ATOMIC_INC(&ctx->repo->cache_misses_total);
285 if (ctx->url.cls == CVMFS_URL_CAS) {
286 BRIX_ATOMIC_INC(&ctx->repo->files_accessed_total);
287 }
288 if (r->headers_out.content_length_n > 0) {
289 BRIX_ATOMIC_ADD(&ctx->repo->bytes_served_fill_total,
290 (ngx_atomic_uint_t) r->headers_out.content_length_n);
291 }
292 }
293 } else if (status == NGX_HTTP_BAD_GATEWAY) {
294 BRIX_CVMFS_METRIC_INC(fill_failures_total);
295 if (ctx->repo != NULL) {
296 BRIX_ATOMIC_INC(&ctx->repo->fill_failures_total);
297 }
298 }
299 /* a 504 hold-expiry is NOT a definitive fill failure — the
300 * detached fill may still publish for the client's retry */
301 } else if (ctx->cache_status == BRIX_CVMFS_CACHE_HIT
302 && (status == NGX_HTTP_OK
303 || status == NGX_HTTP_PARTIAL_CONTENT))
304 {
305 if (r->headers_out.content_length_n > 0) {
306 BRIX_CVMFS_METRIC_ADD(bytes_served_hit_total,
307 (ngx_atomic_uint_t) r->headers_out.content_length_n);
308 }
309 brix_metric_cache_result(BRIX_PROTO_CVMFS, 1, 0);
310 if (ctx->repo != NULL) {
311 BRIX_ATOMIC_INC(&ctx->repo->cache_hits_total);
312 if (ctx->url.cls == CVMFS_URL_CAS) {
313 BRIX_ATOMIC_INC(&ctx->repo->files_accessed_total);
314 }
315 if (r->headers_out.content_length_n > 0) {
316 BRIX_ATOMIC_ADD(&ctx->repo->bytes_served_hit_total,
317 (ngx_atomic_uint_t) r->headers_out.content_length_n);
318 }
319 }
320 }
321}

323ngx_int_t
324ngx_http_brix_cvmfs_handler(ngx_http_request_t *r)
325{
326 ngx_http_brix_cvmfs_loc_conf_t *lcf =
327 ngx_http_get_module_loc_conf(r, ngx_http_brix_cvmfs_module);
328 ngx_http_brix_cvmfs_ctx_t *ctx;
329 ngx_int_t rc;
330
331 /* Re-entry after an off-loop fill runs the handler again on the same
332 * request — the existing ctx marks the observer as already registered. */
333 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
334 if (ctx == NULL) {
335 ngx_pool_cleanup_t *cln;
336
337 ctx = ngx_pcalloc(r->pool, sizeof(*ctx));
338 if (ctx == NULL) {
339 return NGX_HTTP_INTERNAL_SERVER_ERROR;
340 }
341 ngx_http_set_ctx(r, ctx, ngx_http_brix_cvmfs_module);
342
343 cln = ngx_pool_cleanup_add(r->pool, 0);
344 if (cln == NULL) {
345 return NGX_HTTP_INTERNAL_SERVER_ERROR;
346 }
347 cln->handler = cvmfs_finalize_observe;
348 cln->data = r;
349 }
350
351 rc = ngx_http_discard_request_body(r); /* GET/HEAD only proto */
352 if (rc != NGX_OK) {
353 return rc;
354 }
355
356 if (lcf->scvmfs) {
357 rc = brix_scvmfs_preamble(r, lcf); /* T22: TLS + authz */
358 if (rc != NGX_DECLINED) {
359 return rc;
360 }
361 }
362
363 rc = brix_cvmfs_gate(r, lcf); /* classify + police */
364 if (rc != NGX_DECLINED) {
365 return rc; /* reject status, passthrough NGX_DONE … */
366 }
367 return cvmfs_tier_get(r, lcf);
368}
