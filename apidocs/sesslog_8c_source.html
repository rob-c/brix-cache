40/*
41 * WHAT: Return the bounded string length for a possibly NULL C string.
42 * WHY: Formatters accept caller-owned strings from wire paths and identities;
43 * bounding before copies keeps stack scratch buffers finite.
44 * HOW: Count bytes until NUL or max is reached, treating NULL as "-".

46static size_t
47brix_sess_strnlen_or_dash(const char *s, size_t max)
48{
49 size_t n;
50
51 if (s == NULL) {
52 return 1;
53 }
54
55 for (n = 0; n < max && s[n] != '\0'; n++) {
56 /* count only */
57 }
58
59 return n;
60}

62/*
63 * WHAT: Default sanitizer used when unit callers pass NULL.
64 * WHY: The formatter remains usable in pure C tests without linking the nginx
65 * path helper; production glue injects the normal access-log sanitizer.
66 * HOW: Escape quotes, backslashes, controls, DEL, and non-ASCII as \xNN.
67 */
68static size_t
69brix_sess_default_sanitize(char *dst, size_t dst_size, const char *src,
70 size_t src_len)
71{
72 static const char hex[] = "0123456789abcdef";
73 size_t di;
74 size_t si;
75
76 if (dst_size == 0) {
77 return 0;
78 }
79
80 if (src == NULL) {
81 src = "-";
82 src_len = 1;
83 }
84
85 di = 0;
86 for (si = 0; si < src_len && di + 1 < dst_size; si++) {
87 unsigned char ch = (unsigned char) src[si];
88
89 if (ch == '"' || ch == '\\') {
90 if (di + 2 >= dst_size) {
91 break;
92 }
93 dst[di++] = '\\';
94 dst[di++] = (char) ch;
95 continue;
96 }
97
98 if (ch < 0x20 || ch >= 0x7f) {
99 if (di + 4 >= dst_size) {
100 break;
101 }
102 dst[di++] = '\\';
103 dst[di++] = 'x';
104 dst[di++] = hex[(ch >> 4) & 0x0f];
105 dst[di++] = hex[ch & 0x0f];
106 continue;
107 }
108
109 dst[di++] = (char) ch;
110 }
111
112 dst[di] = '\0';
113 return di;
114}

116/*
117 * WHAT: Append formatted text to a bounded line buffer.
118 * WHY: Every event must remain one complete parseable line even when values are
119 * long or malicious.
120 * HOW: Advance the cursor only when snprintf reports data that fit.
121 */
122static void
123brix_sess_append(brix_sess_buf_t *b, const char *fmt, ...)
124{
125 va_list ap;
126 int n;
127
128 if (b->rem == 0) {
129 return;
130 }
131
132 va_start(ap, fmt);
133 /* phase74-fp: fmt is always a compile-time string literal — every caller of
134 * this static varargs forwarder passes a literal format (audited; no table,
135 * no user input reaches fmt). */
136 n = vsnprintf(b->pos, b->rem, fmt, ap); /* NOLINT(clang-diagnostic-format-nonliteral) */
137 va_end(ap);
138
139 if (n < 0) {
140 return;
141 }
142
143 if ((size_t) n >= b->rem) {
144 b->used += b->rem - 1;
145 b->pos += b->rem - 1;
146 b->rem = 1;
147 b->pos[0] = '\0';
148 return;
149 }
150
151 b->pos += n;
152 b->rem -= (size_t) n;
153 b->used += (size_t) n;
154}

156/*
157 * WHAT: Append a fixed-order quoted field with sanitization.
158 * WHY: Paths, users, VOs, peers, and errors can contain arbitrary wire bytes;
159 * quoting them uniformly makes the session log split-safe.
160 * HOW: Clip the raw value, preserve the fact of truncation with "...", sanitize,
161 * and append key="value".
162 */
163static void
164brix_sess_append_quoted(brix_sess_buf_t *b, const char *key, const char *value,
165 size_t raw_cap, brix_sess_sanitize_fn san)
166{
167 char clipped[BRIX_SESSLOG_PATH_MAX + 4];
168 char safe[BRIX_SESSLOG_PATH_MAX * 4 + 8];
169 size_t src_len;
170 size_t copy_len;
171 size_t cap;
172
173 if (raw_cap > BRIX_SESSLOG_PATH_MAX) {
174 raw_cap = BRIX_SESSLOG_PATH_MAX;
175 }
176 if (raw_cap == 0) {
177 raw_cap = 1;
178 }
179
180 if (value == NULL) {
181 value = "-";
182 }
183
184 cap = raw_cap;
185 src_len = brix_sess_strnlen_or_dash(value, cap + 1);
186 copy_len = src_len;
187 if (copy_len > cap) {
188 copy_len = (cap > 3) ? cap - 3 : cap;
189 }
190
191 if (copy_len > 0) {
192 memcpy(clipped, value, copy_len);
193 }
194 if (src_len > cap && cap > 3) {
195 memcpy(clipped + copy_len, "...", 3);
196 copy_len += 3;
197 }
198 clipped[copy_len] = '\0';
199
200 if (san == NULL) {
201 san = brix_sess_default_sanitize;
202 }
203 (void) san(safe, sizeof(safe), clipped, copy_len);
204 brix_sess_append(b, " %s=\"%s\"", key, safe);
205}

207/*
208 * WHAT: Initialize a formatter cursor and make even tiny buffers safe.
209 * WHY: Public formatters should be NULL/tiny-buffer tolerant for tests and
210 * defensive production callers.
211 * HOW: Seed an empty string when there is room; all appends update rem/used.

213static brix_sess_buf_t
214brix_sess_buf(char *line, size_t line_max)
215{
216 brix_sess_buf_t b;
217
218 b.pos = line;
219 b.rem = line_max;
220 b.used = 0;
221 if (line != NULL && line_max > 0) {
222 line[0] = '\0';
223 } else {
224 b.rem = 0;
225 }

246/*
247 * WHAT: Calculate integer bytes per second for an XFER line.
248 * WHY: Operators need a grep-friendly transfer rate that cannot overflow on
249 * large files.
250 * HOW: Use 64-bit arithmetic and divide by max(duration, 1 ms).
251 */
252static uint64_t
253brix_sess_rate(uint64_t bytes, uint64_t dur_ms)
254{
255 uint64_t divisor = dur_ms == 0 ? 1 : dur_ms;
256
257 if (bytes > UINT64_MAX / 1000) {
258 return UINT64_MAX / divisor;
259 }
260
261 return (bytes * 1000) / divisor;
262}

458/*
459 * WHAT: One (numeric code -> stable token) mapping row.
460 * WHY: The three err-from-* families share an identical shape — a small fixed
461 * set of codes collapsing onto a low-cardinality token — so expressing each as
462 * data (a table) rather than a switch ladder removes the per-family branching
463 * complexity while keeping the emitted strings byte-identical.
464 * HOW: Plain value struct scanned linearly by brix_sess_err_lookup().
465 */
466typedef struct {
467 int code;
468 const char *token;
469} brix_sess_err_entry_t;
470
471/*
472 * WHAT: Resolve a numeric code against a fixed mapping table, returning the
473 * matched token or a formatted "code:<n>" fallback.
474 * WHY: All three public err-from-* accessors want the same lookup-then-fallback
475 * behaviour; centralising it keeps the fallback formatting (and the scratch/NULL
476 * contract) in exactly one place.
477 * HOW: (1) Linearly scan the table for an exact code match and return its token.
478 * (2) On no match, write "code:<code>" into scratch when a buffer is supplied,
479 * otherwise return the static "code:0" sentinel — matching the prior default arm.
480 */
481static const char *
482brix_sess_err_lookup(const brix_sess_err_entry_t *table, size_t count,
483 int code, char *scratch, size_t n)
484{
485 size_t i;
486
487 for (i = 0; i < count; i++) {
488 if (table[i].code == code) {
489 return table[i].token;
490 }
491 }
492
493 if (scratch != NULL && n > 0) {
494 snprintf(scratch, n, "code:%d", code);
495 return scratch;
496 }
497
498 return "code:0";
499}
