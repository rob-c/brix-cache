1/*
2 * sd_cache_forward.c — namespace / xattr / dir / staged-write forwarders.
3 *
4 * WHAT: The read-cache decorator's delegating vtable slots — every write /
5 * namespace / xattr / directory / staged-write op forwards straight to the
6 * wrapped source (a write also invalidates the cached copy). The one
7 * non-trivial slot is sd_cache_stat, which answers from a COMPLETE cinfo
8 * to keep a warm-object stat off the source.
9 *
10 * WHY: Split from sd_cache.c (phase-79) to keep every cache file under the
11 * ~500-line, one-concept-per-file cap. These forwarders are the "cache is
12 * transport-transparent above the seam" half of the driver — reviewable
13 * apart from the interposed read-open decision tree and the fill / slice
14 * machinery. The read cache only interposes READ-open; everything here is
15 * pass-through-plus-evict.
16 *
17 * HOW: Each slot reaches the source instance through SD_CACHE_SRC / the state's
18 * ->source (sd_cache_internal.h) and dispatches through the source
19 * driver's matching slot, returning NGX_ERROR / ENOTSUP when the source
20 * lacks it. unlink/rename/staged-open additionally brix_cstore_evict the
21 * affected key(s). All slots are non-static — the driver vtable in
22 * sd_cache.c wires them by name through sd_cache_internal.h. ZERO
23 * behaviour change from the pre-split file.
24 */

44ngx_int_t
45sd_cache_stat(brix_sd_instance_t *inst, const char *path,
46 brix_sd_stat_t *out)
47{
48 sd_cache_inst_state *st = SD_CACHE_ST(inst);
49 brix_sd_instance_t *s = SD_CACHE_SRC(inst);
50 brix_cache_cinfo_t ci;
51
52 /* A COMPLETE cached object answers stat from its cinfo — the same
53 * authoritative-hit doctrine as open() (section 6.4), and it keeps a stat of
54 * a warm object off the source (a remote source would otherwise take a
55 * blocking wire round-trip on the caller's thread — the event loop for the
56 * kXR_open pre-flight probe). A miss/partial falls through to the source. */
57 if (brix_cstore_cinfo_load(&st->cstore, path, &ci) == NGX_OK
58 && (ci.flags & BRIX_CINFO_F_COMPLETE))
59 {
60 ngx_memzero(out, sizeof(*out));
61 out->size = (off_t) ci.size;
62 out->mtime = (time_t) ci.mtime;
63 out->ctime = (time_t) ci.mtime;
64 out->mode = (mode_t) S_IFREG
65 | (mode_t) ((ci.mode != 0) ? (ci.mode & 0777) : 0644);
66 out->is_reg = 1;
67 return NGX_OK;
68 }
69
70 return s->driver->stat ? s->driver->stat(s, path, out) : NGX_ERROR;
71}

211sd_cache_staged_open(brix_sd_instance_t *inst, const char *final_path,
212 mode_t mode, int *err_out)
213{
214 sd_cache_inst_state *st = SD_CACHE_ST(inst);
215 brix_sd_instance_t *s = st->source;
216
217 if (s->driver->staged_open == NULL) {
218 if (err_out != NULL) {
219 *err_out = ENOSYS;
220 }
221 return NULL;
222 }
223 /* A staged publish replaces the object; drop any cached copy now. */
224 (void) brix_cstore_evict(&st->cstore, final_path);
225 return s->driver->staged_open(s, final_path, mode, err_out);
226}

228/* Credential-scoped staged_open: forwards the per-user cred into the source's
229 * staged_open_cred slot when the source driver implements it.
230 *
231 * WHAT: Evicts any cached copy (a staged write is a replacement) and delegates
232 * to the source via brix_sd_staged_open_maybe_cred so the backend driver
233 * can authenticate as the requesting user for the staged upload.
234 *
235 * WHY: Without this slot the cache decorator drops the credential on the floor
236 * when a caller uses brix_sd_staged_open_maybe_cred against it.
237 *
238 * HOW: Evict → brix_sd_staged_open_maybe_cred (cred forwarded to source). */
239brix_sd_staged_t *
240sd_cache_staged_open_cred(brix_sd_instance_t *inst, const char *final_path,
241 mode_t mode, const brix_sd_cred_t *cred, int *err_out)
242{
243 sd_cache_inst_state *st = SD_CACHE_ST(inst);
244 brix_sd_instance_t *s = st->source;
245
246 if (s->driver->staged_open == NULL && s->driver->staged_open_cred == NULL) {
247 if (err_out != NULL) {
248 *err_out = ENOSYS;
249 }
250 return NULL;
251 }
252 (void) brix_cstore_evict(&st->cstore, final_path);
253 return brix_sd_staged_open_maybe_cred(s, final_path, mode, cred, err_out);
254}
