8/*
9 * Client-mediated remote → remote copy: read from the source server and write to
10 * the destination server through this process (two independent sessions). This is
11 * NOT server-side third-party copy (kXR_tpc / M8); the bytes transit the client.
12 * Both opens go through brix_roundtrip, so each side independently follows any
13 * cluster redirect to its data server.
14 */
15/*
16 * WHAT: Tear down whatever the remote→remote copy actually acquired, in reverse
17 * order, and return the final rc.
18 * WHY: Both sessions/handles are released the same way from every exit point;
19 * extracting it lets the orchestrator return r2r_teardown(...) directly at
20 * each decision site instead of jumping to a shared label.
21 * HOW: Each `*_up` / `*open` flag gates the matching close so a partially
22 * initialised attempt frees only what it acquired. The destination handle
23 * is closed cleanly only on success (POSC discards a partial upload when the
24 * handle is abandoned on error); the source handle always closes silently.
25 */
26int
27r2r_teardown(brix_conn *sc, brix_conn *dc, brix_file *sf, brix_file *df,
28 int src_up, int dst_up, int sopen, int dopen, int rc,
29 brix_status *st)
30{
31 if (dopen && rc == 0) {
32 if (brix_file_close(dc, df, st) != 0) {
33 rc = -1;
34 }
35 }
36 if (sopen) {
37 brix_status throwaway;
38 brix_status_clear(&throwaway);
39 brix_file_close(sc, sf, &throwaway);
40 }
41 if (dst_up) {
42 brix_close(dc);
43 }
44 if (src_up) {
45 brix_close(sc);
46 }
47 return rc;
48}

51/*
52 * WHAT: Stream the source's `si->size` bytes through this process into the open
53 * destination handle. Returns 0 on a complete transfer, -1 (st set) otherwise.
54 * WHY: Isolating the read/write loop (and its scratch buffer) keeps the loop body
55 * free of cleanup jumps — it just reports success/failure to the orchestrator,
56 * which owns the connections and runs the staged teardown.
57 * HOW: malloc one chunk buffer, read from the source and write to the destination
58 * until si->size is reached, then free the buffer on every path.
59 */
60int
61r2r_stream_body(brix_conn *sc, brix_conn *dc, brix_file *sf, brix_file *df,
62 const brix_statinfo *si, const brix_copy_opts *o, brix_status *st)
63{
64 pump_remote_t src = { .c = sc, .f = sf, .pgrw = o->pgrw };
65 pump_remote_t sink = { .c = dc, .f = df, .pgrw = o->pgrw };
66
67 /* remote (known si->size) → remote; no progress here (historical). */
68 return transfer_pump(pump_src_remote, &src, pump_sink_remote, &sink,
69 si->size, NULL, si->size, st);
70}

73int
74copy_remote_to_remote(const brix_url *su, const brix_url *du,
75 const brix_copy_opts *o, const brix_opts *co, brix_status *st)
76{
77 brix_conn sc, dc;
78 brix_file sf, df;
79 brix_statinfo si;
80 int rc;
81
82 if (brix_connect(&sc, su, co, st) != 0) {
83 return -1;
84 }
85 if (brix_stat(&sc, su->path, &si, st) != 0) {
86 return r2r_teardown(&sc, &dc, &sf, &df, 1, 0, 0, 0, -1, st);
87 }
88 if (si.flags & kXR_isDir) {
89 brix_status_set(st, XRDC_EUSAGE, 0,
90 "source is a directory (recursive copy unsupported)");
91 return r2r_teardown(&sc, &dc, &sf, &df, 1, 0, 0, 0, -1, st);
92 }
93
94 if (brix_connect(&dc, du, co, st) != 0) {
95 return r2r_teardown(&sc, &dc, &sf, &df, 1, 0, 0, 0, -1, st);
96 }
97
98 if (brix_file_open_read(&sc, su->path, &sf, st) != 0) {
99 return r2r_teardown(&sc, &dc, &sf, &df, 1, 1, 0, 0, -1, st);
100 }
101 if (brix_file_open_write(&dc, du->path, o->force, o->posc, &df, st) != 0) {
102 return r2r_teardown(&sc, &dc, &sf, &df, 1, 1, 1, 0, -1, st);
103 }
104
105 rc = r2r_stream_body(&sc, &dc, &sf, &df, &si, o, st);
106
107 return r2r_teardown(&sc, &dc, &sf, &df, 1, 1, 1, 1, rc, st);
108}

111/*
112 * --cksum handling. `spec` is "<type>[:source|:print|:<value>]". Computes the
113 * named checksum over the local file `local_path` (the bytes we actually moved)
114 * and then, by mode:
115 * :source / :end2end → query the server (kXR_Qcksum) for `remote_path` on the
116 * already-open connection `c` and require they match;
117 * :<value> → require the local digest equals the given hex;
118 * :print / (none) → print "<type> <digest>" (unless silent).
119 * Returns XRDC_CK_OK (matched/printed), XRDC_CK_MISMATCH (digest known and WRONG
120 * — caller drops the destination), or XRDC_CK_UNVERIFIED (query/transport/usage
121 * error — the digest is UNKNOWN, so the caller keeps the file and only warns;
122 * deleting a byte-perfect download because a control-plane query hiccupped would
123 * be the inverse footgun).
124 */
125int
126cksum_verify(brix_conn *c, const char *remote_path, const char *local_path,
127 const char *spec, int silent, brix_status *st)
128{
129 char algo_name[32];
130 char local_hex[129];
131 const char *colon = strchr(spec, ':');
132 const char *mode = NULL;
133 size_t alen;
134 brix_cksum_algo algo;
135 int lfd;
136
137 alen = colon ? (size_t) (colon - spec) : strlen(spec);
138 if (alen == 0 || alen >= sizeof(algo_name)) {
139 brix_status_set(st, XRDC_EUSAGE, 0, "bad --cksum type");
140 return XRDC_CK_UNVERIFIED;
141 }
142 memcpy(algo_name, spec, alen);
143 algo_name[alen] = '\0';
144 mode = colon ? colon + 1 : NULL;
145
146 if (brix_cksum_algo_parse(algo_name, &algo) != 0) {
147 brix_status_set(st, XRDC_EUSAGE, 0,
148 "unsupported --cksum type \"%s\"", algo_name);
149 return XRDC_CK_UNVERIFIED;
150 }
151
152 if (local_path == NULL) {
153 /* stdio endpoint — nothing on disk to digest; skip rather than lie. */
154 if (!silent) {
155 fprintf(stderr, "xrdcp: --cksum skipped for stdin/stdout\n");
156 }
157 return XRDC_CK_OK;
158 }
159
160 lfd = open(local_path, O_RDONLY);
161 if (lfd < 0) {
162 brix_status_set(st, XRDC_EUSAGE, errno,
163 "open %s for checksum: %s", local_path, strerror(errno));
164 return XRDC_CK_UNVERIFIED;
165 }
166 if (brix_cksum_fd(lfd, algo, local_hex, sizeof(local_hex), st) != 0) {
167 close(lfd);
168 return XRDC_CK_UNVERIFIED;
169 }
170 close(lfd);
171
172 if (mode != NULL
173 && (strcmp(mode, "source") == 0 || strcmp(mode, "end2end") == 0)) {
174 char server_hex[129];
175 if (remote_path == NULL) {
176 brix_status_set(st, XRDC_EUSAGE, 0, "--cksum:source has no remote");
177 return XRDC_CK_UNVERIFIED;
178 }
179 if (brix_query_cksum(c, remote_path, algo_name,
180 server_hex, sizeof(server_hex), st) != 0) {
181 return XRDC_CK_UNVERIFIED; /* server digest UNKNOWN, not WRONG */
182 }
183 if (strcasecmp(local_hex, server_hex) != 0) {
184 /* A checksum mismatch is a data-integrity failure, not a transient
185 * framing fault — classify it non-retryable so no resilient loop
186 * spins re-verifying the same bytes. */
187 brix_status_set(st, XRDC_EINTEGRITY, 0,
188 "%s mismatch: local %s != server %s",
189 algo_name, local_hex, server_hex);
190 return XRDC_CK_MISMATCH;
191 }
192 if (!silent) {
193 printf("%s %s OK (matches server)\n", algo_name, local_hex);
194 }
195 return XRDC_CK_OK;
196 }
197
198 if (mode != NULL && strcmp(mode, "print") != 0) {
199 /* literal expected value */
200 if (strcasecmp(local_hex, mode) != 0) {
201 brix_status_set(st, XRDC_EINTEGRITY, 0,
202 "%s mismatch: got %s expected %s",
203 algo_name, local_hex, mode);
204 return XRDC_CK_MISMATCH;
205 }
206 if (!silent) {
207 printf("%s %s OK\n", algo_name, local_hex);
208 }
209 return XRDC_CK_OK;
210 }
211
212 /* print / no mode */
213 if (!silent) {
214 printf("%s %s\n", algo_name, local_hex);
215 }
216 return XRDC_CK_OK;
217}

220/* Mint a random TPC rendezvous key (hex of 16 /dev/urandom bytes). */
221int
222gen_tpc_key(char *out, size_t outsz)
223{
224 uint8_t raw[16];
225 int fd;
226
227 if (outsz < sizeof(raw) * 2 + 1) {
228 return -1;
229 }
230 fd = open("/dev/urandom", O_RDONLY);
231 if (fd < 0) {
232 return -1;
233 }
234 if (read(fd, raw, sizeof(raw)) != (ssize_t) sizeof(raw)) {
235 close(fd);
236 return -1;
237 }
238 close(fd);
239 brix_hex_encode(raw, sizeof(raw), out); /* shared lowercase hex */
240 return 0;
241}

244/*
245 * Server-side third-party copy (root://) in the STOCK XRootD dialect
246 * (XrdOucTPC cgiC2Dst/cgiC2Src + the XrdCl ThirdPartyCopyJob control order),
247 * which BOTH a stock XRootD destination and nginx-xrootd accept:
248 * 1. connect SRC + kXR_stat → size for oss.asize; the session has followed
249 * any cluster redirect, so its live host:port names the actual source
250 * data server (the client-side "placement" step).
251 * 2. open DST write with tpc.key=K &tpc.src=<shost:sport> &tpc.lfn=<spath>
252 * &tpc.stage=copy &oss.asize=N &tpc.dlg/spr/tpr/dlgon → dest = puller
253 * 3. kXR_sync DST — rendezvous setup (stock) / arm (nginx).
254 * 4. open SRC read with tpc.key=K &tpc.dst=<dhost> &tpc.stage=copy
255 * → the source authorizes the upcoming pull (stock authQ / nginx SHM
256 * key registry). nginx defers this open's reply until the pull completes
257 * (tpc_coord_defer surfaces the kXR_waitresp); stock replies immediately.
258 * 5. kXR_sync DST — trigger the pull and await completion (the reply may be
259 * deferred via kXR_waitresp → kXR_attn(asynresp); brix_recv unwraps it).
260 * The destination server connects to the source itself and pulls the bytes — no
261 * data transits this client (unlike copy_remote_to_remote).
262 *
263 * Dialect note: the legacy nginx-only form (tpc.src=root://host:port/path as a
264 * full URL, no tpc.lfn/tpc.stage, source opened BEFORE the destination) is NOT
265 * emitted any more — a stock destination cannot parse it ("Invalid address"),
266 * and a stock source cannot Match() its tpc.dst full-URL against the puller's
267 * hostname. nginx-xrootd parses both forms (src/tpc/engine/parse.c normalizes bare
268 * host[:port] + tpc.lfn), so the stock dialect is the one that works against
269 * every server. tpc.token_mode is an nginx extension; unknown tpc.* keys are
270 * ignored by both servers.
271 */
272/*
273 * WHAT: Tear down whatever the TPC rendezvous acquired — the two opaque request
274 * strings, the source/destination handles, and their connections — and
275 * return the final rc.
276 * WHY: Every exit point releases the same resources the same way; extracting it
277 * lets the orchestrator return tpc_teardown(...) directly at each decision
278 * site instead of jumping to a shared label.
279 * HOW: Each `*open` / `*_up` flag gates the matching close so a partially set-up
280 * rendezvous frees only what it acquired. The destination handle reports its
281 * close error into `st` only on success (so a failed sync's status survives);
282 * the source handle always closes silently. The opaque strings are freed
283 * unconditionally (they are allocated before any session is opened).
284 */
285int
286tpc_teardown(brix_conn *sc, brix_conn *dc, brix_file *sf, brix_file *df,
287 char *src_opaque, char *dst_opaque,
288 int su_up, int du_up, int sopen, int dopen, int rc, brix_status *st)
289{
290 if (dopen) {
291 brix_status tw; brix_status_clear(&tw);
292 brix_file_close(dc, df, rc == 0 ? st : &tw);
293 }
294 if (sopen) {
295 brix_status tw; brix_status_clear(&tw);
296 brix_file_close(sc, sf, &tw);
297 }
298 if (du_up) { brix_close(dc); }
299 if (su_up) { brix_close(sc); }
300 free(src_opaque);
301 free(dst_opaque);
302 return rc;
303}

306int
307copy_tpc(const brix_url *su, const brix_url *du, const brix_copy_opts *o,
308 const brix_opts *co, brix_status *st)
309{
310 brix_conn sc, dc;
311 brix_file sf, df;
312 brix_statinfo si;
313 char key[40];
314 char src_hp[XRDC_HOSTPORT_MAX];
315 char *src_opaque = NULL, *dst_opaque = NULL;
316 size_t need;
317 const char *tok = (o->tpc_token_mode && o->tpc_token_mode[0])
318 ? o->tpc_token_mode : NULL;
319
320 if (gen_tpc_key(key, sizeof(key)) != 0) {
321 brix_status_set(st, XRDC_ESOCK, 0, "cannot generate TPC key");
322 return -1;
323 }
324
325 /* 1. Source session + stat: learn the size for oss.asize and let the
326 * session follow any cluster redirect so src_hp names the actual source
327 * data server (the client-side equivalent of the stock placement step).
328 * The stat is BEST-EFFORT, like XrdCl's facultative placement stat — a
329 * minimal source that cannot answer kXR_stat must not kill the copy (the
330 * pull itself will surface any real problem); only a definitive "no such
331 * file" fails fast, before the destination is created. */
332 if (brix_connect(&sc, su, co, st) != 0) {
333 return -1;
334 }
335 si.size = -1;
336 if (brix_stat(&sc, su->path, &si, st) != 0) {
337 if (st->kxr == kXR_NotFound) {
338 return tpc_teardown(&sc, &dc, &sf, &df, NULL, NULL,
339 1, 0, 0, 0, -1, st);
340 }
341 si.size = -1; /* size unknown → oss.asize omitted below */
342 }
343 brix_format_host_port(sc.host, (uint16_t) sc.port, src_hp, sizeof(src_hp));
344
345 /* Heap-size the opaque strings to the actual host/path/key/token lengths. */
346 need = sizeof(src_hp) + strlen(su->host) + strlen(su->path)
347 + strlen(du->host) + strlen(du->path) + sizeof(key)
348 + (tok ? strlen(tok) : 0) + 192;
349 src_opaque = (char *) malloc(need);
350 dst_opaque = (char *) malloc(need);
351 if (src_opaque == NULL || dst_opaque == NULL) {
352 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
353 return tpc_teardown(&sc, &dc, &sf, &df, src_opaque, dst_opaque,
354 1, 0, 0, 0, -1, st);
355 }
356
357 /* 2. Destination open, stock cgiC2Dst field order. tpc.dlg names the
358 * originally-requested source endpoint and is inert while tpc.dlgon=0
359 * (this client never delegates); emitted for wire parity with XrdCl.
360 * oss.asize is only sent when the best-effort stat learned the size. */
361 {
362 char asize[48];
363 asize[0] = '\0';
364 if (si.size >= 0) {
365 snprintf(asize, sizeof(asize), "&oss.asize=%lld",
366 (long long) si.size);
367 }
368 snprintf(dst_opaque, need,
369 "tpc.key=%s&tpc.src=%s&tpc.lfn=%s&tpc.dlg=%s:%d&tpc.spr=root"
370 "&tpc.tpr=root&tpc.dlgon=0%s&tpc.stage=copy%s%s",
371 key, src_hp, su->path, su->host, su->port, asize,
372 tok ? "&tpc.token_mode=" : "", tok ? tok : "");
373 }
374
375 if (brix_connect(&dc, du, co, st) != 0) {
376 return tpc_teardown(&sc, &dc, &sf, &df, src_opaque, dst_opaque,
377 1, 0, 0, 0, -1, st);
378 }
379 if (brix_file_open_opaque(&dc, du->path, dst_opaque, 1, o->force, o->posc,
380 &df, st) != 0) {
381 return tpc_teardown(&sc, &dc, &sf, &df, src_opaque, dst_opaque,
382 1, 1, 0, 0, -1, st);
383 }
384
385 if (brix_file_sync(&dc, &df, st) != 0) { /* rendezvous setup / arm */
386 return tpc_teardown(&sc, &dc, &sf, &df, src_opaque, dst_opaque,
387 1, 1, 0, 1, -1, st);
388 }
389
390 /* 3. Source open, stock cgiC2Src form (tpc.dst = destination HOSTNAME as
391 * the source will see the puller connect — dc.host is the post-redirect
392 * endpoint). This registers/authorizes the key at the source. On nginx the
393 * source defers this open's reply (kXR_waitresp) until the pull completes;
394 * let brix_recv surface that deferral rather than block — the pull that
395 * satisfies it is only triggered by the sync below, so blocking here would
396 * deadlock the rendezvous. The source connection stays open through the
397 * transfer so the registration remains live. */
398 snprintf(src_opaque, need, "tpc.key=%s&tpc.dst=%s&tpc.stage=copy%s%s",
399 key, dc.host,
400 tok ? "&tpc.token_mode=" : "", tok ? tok : "");
401 sc.tpc_coord_defer = 1;
402 if (brix_file_open_opaque(&sc, su->path, src_opaque, 0, 0, 0, &sf, st) != 0) {
403 return tpc_teardown(&sc, &dc, &sf, &df, src_opaque, dst_opaque,
404 1, 1, 0, 1, -1, st);
405 }
406
407 if (dc.io.timeout_ms < 300000) {
408 dc.io.timeout_ms = 300000; /* 5 min for the deferred pull */
409 }
410 if (brix_file_sync(&dc, &df, st) != 0) { /* trigger + await completion */
411 return tpc_teardown(&sc, &dc, &sf, &df, src_opaque, dst_opaque,
412 1, 1, 1, 1, -1, st);
413 }
414
415 return tpc_teardown(&sc, &dc, &sf, &df, src_opaque, dst_opaque,
416 1, 1, 1, 1, 0, st);
417}
