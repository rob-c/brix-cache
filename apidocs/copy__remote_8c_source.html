8/*
9 * Client-mediated remote → remote copy: read from the source server and write to
10 * the destination server through this process (two independent sessions). This is
11 * NOT server-side third-party copy (kXR_tpc / M8); the bytes transit the client.
12 * Both opens go through brix_roundtrip, so each side independently follows any
13 * cluster redirect to its data server.
14 */
15/*
16 * WHAT: Tear down whatever the remote→remote copy actually acquired, in reverse
17 * order, and return the final rc.
18 * WHY: Both sessions/handles are released the same way from every exit point;
19 * extracting it lets the orchestrator return r2r_teardown(...) directly at
20 * each decision site instead of jumping to a shared label.
21 * HOW: Each `*_up` / `*open` flag gates the matching close so a partially
22 * initialised attempt frees only what it acquired. The destination handle
23 * is closed cleanly only on success (POSC discards a partial upload when the
24 * handle is abandoned on error); the source handle always closes silently.
25 */
26int
27r2r_teardown(brix_conn *sc, brix_conn *dc, brix_file *sf, brix_file *df,
28 int src_up, int dst_up, int sopen, int dopen, int rc,
29 brix_status *st)
30{
31 if (dopen && rc == 0) {
32 if (brix_file_close(dc, df, st) != 0) {
33 rc = -1;
34 }
35 }
36 if (sopen) {
37 brix_status throwaway;
38 brix_status_clear(&throwaway);
39 brix_file_close(sc, sf, &throwaway);
40 }
41 if (dst_up) {
42 brix_close(dc);
43 }
44 if (src_up) {
45 brix_close(sc);
46 }
47 return rc;
48}

51/*
52 * WHAT: Stream the source's `si->size` bytes through this process into the open
53 * destination handle. Returns 0 on a complete transfer, -1 (st set) otherwise.
54 * WHY: Isolating the read/write loop (and its scratch buffer) keeps the loop body
55 * free of cleanup jumps — it just reports success/failure to the orchestrator,
56 * which owns the connections and runs the staged teardown.
57 * HOW: malloc one chunk buffer, read from the source and write to the destination
58 * until si->size is reached, then free the buffer on every path.
59 */
60int
61r2r_stream_body(brix_conn *sc, brix_conn *dc, brix_file *sf, brix_file *df,
62 const brix_statinfo *si, const brix_copy_opts *o, brix_status *st)
63{
64 pump_remote_t src = { .c = sc, .f = sf, .pgrw = o->pgrw };
65 pump_remote_t sink = { .c = dc, .f = df, .pgrw = o->pgrw };
66
67 /* remote (known si->size) → remote; no progress here (historical). */
68 return transfer_pump(pump_src_remote, &src, pump_sink_remote, &sink,
69 si->size, NULL, si->size, st);
70}

73int
74copy_remote_to_remote(const brix_url *su, const brix_url *du,
75 const brix_copy_opts *o, const brix_opts *co, brix_status *st)
76{
77 brix_conn sc, dc;
78 brix_file sf, df;
79 brix_statinfo si;
80 int rc;
81
82 if (brix_connect(&sc, su, co, st) != 0) {
83 return -1;
84 }
85 if (brix_stat(&sc, su->path, &si, st) != 0) {
86 return r2r_teardown(&sc, &dc, &sf, &df, 1, 0, 0, 0, -1, st);
87 }
88 if (si.flags & kXR_isDir) {
89 brix_status_set(st, XRDC_EUSAGE, 0,
90 "source is a directory (recursive copy unsupported)");
91 return r2r_teardown(&sc, &dc, &sf, &df, 1, 0, 0, 0, -1, st);
92 }
93
94 if (brix_connect(&dc, du, co, st) != 0) {
95 return r2r_teardown(&sc, &dc, &sf, &df, 1, 0, 0, 0, -1, st);
96 }
97
98 if (brix_file_open_read(&sc, su->path, &sf, st) != 0) {
99 return r2r_teardown(&sc, &dc, &sf, &df, 1, 1, 0, 0, -1, st);
100 }
101 if (brix_file_open_write(&dc, du->path, o->force, o->posc, &df, st) != 0) {
102 return r2r_teardown(&sc, &dc, &sf, &df, 1, 1, 1, 0, -1, st);
103 }
104
105 rc = r2r_stream_body(&sc, &dc, &sf, &df, &si, o, st);
106
107 return r2r_teardown(&sc, &dc, &sf, &df, 1, 1, 1, 1, rc, st);
108}

111/*
112 * WHAT: A parsed --cksum spec: the resolved algorithm, its lowercase name (for
113 * kXR_Qcksum + messages), the optional mode suffix, and the silent flag.
114 * WHY: cksum_verify's spec grammar ("<type>[:source|:print|:<value>]") is parsed
115 * once up front; carrying the result (plus the caller's silent flag) in a
116 * small struct lets the digest and per-mode helpers take one argument
117 * instead of several loose values.
118 * HOW: `mode` points into the caller's `spec` (after the colon) or is NULL when
119 * no suffix was given; `name` is a NUL-terminated copy of the type token;
120 * `silent` suppresses the success prints.
121 */
122typedef struct {
123 brix_cksum_algo algo;
124 char name[32];
125 const char *mode;
126 int silent;
127} cksum_spec_t;
128
129
130/*
131 * WHAT: Parse `spec` into a resolved algorithm + mode (the "resolve-algo" step).
132 * Returns 0 on success; on a usage error sets `st` and returns -1.
133 * WHY: Splitting the grammar parse out of cksum_verify keeps the orchestrator a
134 * flat sequence (parse → digest → compare) with no nested branching.
135 * HOW: Split at the first ':' — the token before it is the algorithm name, the
136 * remainder (or NULL) is the mode. Reject an empty or over-long name, then
137 * resolve the name to an algo id via the shared parser.
138 */
139static int
140cksum_parse_spec(const char *spec, cksum_spec_t *out, brix_status *st)
141{
142 const char *colon = strchr(spec, ':');
143 size_t alen = colon ? (size_t) (colon - spec) : strlen(spec);
144
145 if (alen == 0 || alen >= sizeof(out->name)) {
146 brix_status_set(st, XRDC_EUSAGE, 0, "bad --cksum type");
147 return -1;
148 }
149 memcpy(out->name, spec, alen);
150 out->name[alen] = '\0';
151 out->mode = colon ? colon + 1 : NULL;
152
153 if (brix_cksum_algo_parse(out->name, &out->algo) != 0) {
154 brix_status_set(st, XRDC_EUSAGE, 0,
155 "unsupported --cksum type \"%s\"", out->name);
156 return -1;
157 }
158 return 0;
159}

162/*
163 * WHAT: Compute `spec->algo` over the local file `local_path` into `hex` (the
164 * "compute-local" step). Returns 0 on success, -1 (st set) otherwise.
165 * WHY: The open/digest/close sequence is a single self-contained unit with one
166 * cleanup path; isolating it removes an fd lifetime from the orchestrator.
167 * HOW: open() the file read-only, run the shared fd digest, and close on every
168 * path (including the digest-failure path).
169 */
170static int
171cksum_compute_local(const char *local_path, const cksum_spec_t *spec,
172 char *hex, size_t hexsz, brix_status *st)
173{
174 int lfd = open(local_path, O_RDONLY);
175
176 if (lfd < 0) {
177 brix_status_set(st, XRDC_EUSAGE, errno,
178 "open %s for checksum: %s", local_path, strerror(errno));
179 return -1;
180 }
181 if (brix_cksum_fd(lfd, spec->algo, hex, hexsz, st) != 0) {
182 close(lfd);
183 return -1;
184 }
185 close(lfd);
186 return 0;
187}

190/*
191 * WHAT: :source / :end2end mode — fetch the server's digest for `remote_path` and
192 * compare it to `local_hex` (the "fetch-remote + compare" step).
193 * WHY: The server round-trip and the mismatch classification are one logical
194 * branch of cksum_verify; extracting it keeps the orchestrator's mode
195 * dispatch a flat ladder of single-line calls.
196 * HOW: Require a remote path, query kXR_Qcksum, then case-insensitively compare.
197 * A query/transport failure is UNVERIFIED (digest UNKNOWN); a definite
198 * inequality is a non-retryable MISMATCH (EINTEGRITY).
199 */
200static int
201cksum_verify_source(brix_conn *c, const char *remote_path, const char *local_hex,
202 const cksum_spec_t *spec, brix_status *st)
203{
204 char server_hex[129];
205
206 if (remote_path == NULL) {
207 brix_status_set(st, XRDC_EUSAGE, 0, "--cksum:source has no remote");
208 return XRDC_CK_UNVERIFIED;
209 }
210 if (brix_query_cksum(c, remote_path, spec->name,
211 server_hex, sizeof(server_hex), st) != 0) {
212 return XRDC_CK_UNVERIFIED; /* server digest UNKNOWN, not WRONG */
213 }
214 if (strcasecmp(local_hex, server_hex) != 0) {
215 /* A checksum mismatch is a data-integrity failure, not a transient
216 * framing fault — classify it non-retryable so no resilient loop
217 * spins re-verifying the same bytes. */
218 brix_status_set(st, XRDC_EINTEGRITY, 0,
219 "%s mismatch: local %s != server %s",
220 spec->name, local_hex, server_hex);
221 return XRDC_CK_MISMATCH;
222 }
223 if (!spec->silent) {
224 printf("%s %s OK (matches server)\n", spec->name, local_hex);
225 }
226 return XRDC_CK_OK;
227}
228

230/*
231 * WHAT: :<value> mode — compare `local_hex` to the literal expected digest carried
232 * in `spec->mode` (the "compare-literal" step).
233 * WHY: The literal branch mirrors the :source branch but has no round-trip;
234 * splitting it keeps both compare paths at the same altitude.
235 * HOW: Case-insensitive compare; inequality is a non-retryable MISMATCH.
236 */
237static int
238cksum_verify_literal(const char *local_hex, const cksum_spec_t *spec,
239 brix_status *st)
240{
241 if (strcasecmp(local_hex, spec->mode) != 0) {
242 brix_status_set(st, XRDC_EINTEGRITY, 0,
243 "%s mismatch: got %s expected %s",
244 spec->name, local_hex, spec->mode);
245 return XRDC_CK_MISMATCH;
246 }
247 if (!spec->silent) {
248 printf("%s %s OK\n", spec->name, local_hex);
249 }
250 return XRDC_CK_OK;
251}

254/*
255 * --cksum handling. `spec` is "<type>[:source|:print|:<value>]". Computes the
256 * named checksum over the local file `local_path` (the bytes we actually moved)
257 * and then, by mode:
258 * :source / :end2end → query the server (kXR_Qcksum) for `remote_path` on the
259 * already-open connection `c` and require they match;
260 * :<value> → require the local digest equals the given hex;
261 * :print / (none) → print "<type> <digest>" (unless silent).
262 * Returns XRDC_CK_OK (matched/printed), XRDC_CK_MISMATCH (digest known and WRONG
263 * — caller drops the destination), or XRDC_CK_UNVERIFIED (query/transport/usage
264 * error — the digest is UNKNOWN, so the caller keeps the file and only warns;
265 * deleting a byte-perfect download because a control-plane query hiccupped would
266 * be the inverse footgun).
267 */
268int
269cksum_verify(brix_conn *c, const char *remote_path, const char *local_path,
270 const char *spec, int silent, brix_status *st)
271{
272 cksum_spec_t parsed = { 0 };
273 char local_hex[129];
274
275 if (cksum_parse_spec(spec, &parsed, st) != 0) {
276 return XRDC_CK_UNVERIFIED;
277 }
278 parsed.silent = silent;
279
280 if (local_path == NULL) {
281 /* stdio endpoint — nothing on disk to digest; skip rather than lie. */
282 if (!silent) {
283 fprintf(stderr, "xrdcp: --cksum skipped for stdin/stdout\n");
284 }
285 return XRDC_CK_OK;
286 }
287
288 if (cksum_compute_local(local_path, &parsed,
289 local_hex, sizeof(local_hex), st) != 0) {
290 return XRDC_CK_UNVERIFIED;
291 }
292
293 if (parsed.mode != NULL
294 && (strcmp(parsed.mode, "source") == 0
295 || strcmp(parsed.mode, "end2end") == 0)) {
296 return cksum_verify_source(c, remote_path, local_hex, &parsed, st);
297 }
298
299 if (parsed.mode != NULL && strcmp(parsed.mode, "print") != 0) {
300 return cksum_verify_literal(local_hex, &parsed, st);
301 }
302
303 /* print / no mode */
304 if (!silent) {
305 printf("%s %s\n", parsed.name, local_hex);
306 }
307 return XRDC_CK_OK;
308}

311/* Mint a random TPC rendezvous key (hex of 16 /dev/urandom bytes). */
312int
313gen_tpc_key(char *out, size_t outsz)
314{
315 uint8_t raw[16];
316 int fd;
317
318 if (outsz < sizeof(raw) * 2 + 1) {
319 return -1;
320 }
321 fd = open("/dev/urandom", O_RDONLY);
322 if (fd < 0) {
323 return -1;
324 }
325 if (read(fd, raw, sizeof(raw)) != (ssize_t) sizeof(raw)) {
326 close(fd);
327 return -1;
328 }
329 close(fd);
330 brix_hex_encode(raw, sizeof(raw), out); /* shared lowercase hex */
331 return 0;
332}

335/*
336 * Server-side third-party copy (root://) in the STOCK XRootD dialect
337 * (XrdOucTPC cgiC2Dst/cgiC2Src + the XrdCl ThirdPartyCopyJob control order),
338 * which BOTH a stock XRootD destination and nginx-xrootd accept:
339 * 1. connect SRC + kXR_stat → size for oss.asize; the session has followed
340 * any cluster redirect, so its live host:port names the actual source
341 * data server (the client-side "placement" step).
342 * 2. open DST write with tpc.key=K &tpc.src=<shost:sport> &tpc.lfn=<spath>
343 * &tpc.stage=copy &oss.asize=N &tpc.dlg/spr/tpr/dlgon → dest = puller
344 * 3. kXR_sync DST — rendezvous setup (stock) / arm (nginx).
345 * 4. open SRC read with tpc.key=K &tpc.dst=<dhost> &tpc.stage=copy
346 * → the source authorizes the upcoming pull (stock authQ / nginx SHM
347 * key registry). nginx defers this open's reply until the pull completes
348 * (tpc_coord_defer surfaces the kXR_waitresp); stock replies immediately.
349 * 5. kXR_sync DST — trigger the pull and await completion (the reply may be
350 * deferred via kXR_waitresp → kXR_attn(asynresp); brix_recv unwraps it).
351 * The destination server connects to the source itself and pulls the bytes — no
352 * data transits this client (unlike copy_remote_to_remote).
353 *
354 * Dialect note: the legacy nginx-only form (tpc.src=root://host:port/path as a
355 * full URL, no tpc.lfn/tpc.stage, source opened BEFORE the destination) is NOT
356 * emitted any more — a stock destination cannot parse it ("Invalid address"),
357 * and a stock source cannot Match() its tpc.dst full-URL against the puller's
358 * hostname. nginx-xrootd parses both forms (src/tpc/engine/parse.c normalizes bare
359 * host[:port] + tpc.lfn), so the stock dialect is the one that works against
360 * every server. tpc.token_mode is an nginx extension; unknown tpc.* keys are
361 * ignored by both servers.
362 */
363/*
364 * WHAT: Tear down whatever the TPC rendezvous acquired — the two opaque request
365 * strings, the source/destination handles, and their connections — and
366 * return the final rc.
367 * WHY: Every exit point releases the same resources the same way; extracting it
368 * lets the orchestrator return tpc_teardown(...) directly at each decision
369 * site instead of jumping to a shared label.
370 * HOW: Each `*open` / `*_up` flag gates the matching close so a partially set-up
371 * rendezvous frees only what it acquired. The destination handle reports its
372 * close error into `st` only on success (so a failed sync's status survives);
373 * the source handle always closes silently. The opaque strings are freed
374 * unconditionally (they are allocated before any session is opened).
375 */
376int
377tpc_teardown(brix_conn *sc, brix_conn *dc, brix_file *sf, brix_file *df,
378 char *src_opaque, char *dst_opaque,
379 int su_up, int du_up, int sopen, int dopen, int rc, brix_status *st)
380{
381 if (dopen) {
382 brix_status tw; brix_status_clear(&tw);
383 brix_file_close(dc, df, rc == 0 ? st : &tw);
384 }
385 if (sopen) {
386 brix_status tw; brix_status_clear(&tw);
387 brix_file_close(sc, sf, &tw);
388 }
389 if (du_up) { brix_close(dc); }
390 if (su_up) { brix_close(sc); }
391 free(src_opaque);
392 free(dst_opaque);
393 return rc;
394}

397/*
398 * WHAT: The mutable rendezvous state a copy_tpc pass threads through its setup /
399 * run / verify stages: both sessions, both handles, the two opaque strings,
400 * and the acquisition flags that gate teardown.
401 * WHY: copy_tpc was one long procedure whose every step both mutated this state
402 * and, on failure, called tpc_teardown with a hand-maintained flag vector.
403 * Bundling the state lets the stage helpers own their own error-teardown at
404 * the right altitude while the orchestrator stays a flat call sequence.
405 * HOW: `su_up`/`du_up`/`sopen`/`dopen` mirror tpc_teardown's flag arguments;
406 * each stage sets a flag the instant it acquires the matching resource.
407 */
408typedef struct {
409 brix_conn sc, dc;
410 brix_file sf, df;
411 char *src_opaque, *dst_opaque;
412 int su_up, du_up, sopen, dopen;
413} tpc_state_t;

416/*
417 * WHAT: The read-only inputs a copy_tpc pass shares across its stages: the source
418 * and destination URLs, the transfer/connection option blocks, the minted
419 * rendezvous key, the source's post-redirect host:port, the best-effort
420 * size (-1 = unknown), and the optional tpc.token_mode value (NULL = none).
421 * WHY: These values are computed once (in setup) and then only read by the run
422 * and verify stages; bundling them keeps each stage helper within the 5-arg
423 * budget instead of threading eight loose parameters.
424 * HOW: `key`/`src_hp` are filled by tpc_setup_source; `size`/`tok` likewise; the
425 * four pointers alias the caller's URL/option blocks and are never mutated.
426 */
427typedef struct {
428 const brix_url *su, *du;
429 const brix_copy_opts *o;
430 const brix_opts *co;
431 char key[40];
432 char src_hp[XRDC_HOSTPORT_MAX];
433 int64_t size;
434 const char *tok;
435} tpc_params_t;

438/*
439 * WHAT: Release everything `s` acquired via the frozen tpc_teardown extern and
440 * return `rc`.
441 * WHY: Every copy_tpc stage exits through the same teardown; a one-line adapter
442 * from the state struct to the flag-vector extern keeps each call site
443 * readable and the flag order in exactly one place.
444 * HOW: Forward the struct's handles, opaque strings, and acquisition flags to
445 * tpc_teardown unchanged.
446 */
447static int
448tpc_state_teardown(tpc_state_t *s, int rc, brix_status *st)
449{
450 return tpc_teardown(&s->sc, &s->dc, &s->sf, &s->df,
451 s->src_opaque, s->dst_opaque,
452 s->su_up, s->du_up, s->sopen, s->dopen, rc, st);
453}

456/*
457 * WHAT: Stage 1 (setup) — mint the key, open the source session, best-effort stat
458 * it, and format its post-redirect host:port into `src_hp`. On success sets
459 * `*size` (>=0 known, -1 unknown) and returns 0; on hard failure tears down
460 * and returns -1.
461 * WHY: The placement step mixes key generation, a session open, a facultative
462 * stat, and address formatting; isolating it keeps copy_tpc's top level a
463 * linear setup → run → verify.
464 * HOW: gen_tpc_key → brix_connect (marks su_up) → brix_stat. The stat is
465 * BEST-EFFORT like XrdCl's facultative placement stat: only a definitive
466 * kXR_NotFound fails fast (before the destination exists); any other stat
467 * error just leaves the size unknown so oss.asize is omitted downstream.
468 */
469static int
470tpc_setup_source(tpc_state_t *s, tpc_params_t *p, brix_status *st)
471{
472 brix_statinfo si;
473
474 if (gen_tpc_key(p->key, sizeof(p->key)) != 0) {
475 brix_status_set(st, XRDC_ESOCK, 0, "cannot generate TPC key");
476 return -1;
477 }
478
479 /* Source session + stat: learn the size for oss.asize and let the session
480 * follow any cluster redirect so src_hp names the actual source data
481 * server (the client-side equivalent of the stock placement step). */
482 if (brix_connect(&s->sc, p->su, p->co, st) != 0) {
483 return -1;
484 }
485 s->su_up = 1;
486
487 si.size = -1;
488 if (brix_stat(&s->sc, p->su->path, &si, st) != 0) {
489 if (st->kxr == kXR_NotFound) {
490 return tpc_state_teardown(s, -1, st);
491 }
492 si.size = -1; /* size unknown → oss.asize omitted below */
493 }
494 p->size = si.size;
495 brix_format_host_port(s->sc.host, (uint16_t) s->sc.port,
496 p->src_hp, sizeof(p->src_hp));
497 return 0;
498}

501/*
502 * WHAT: Build the destination cgiC2Dst opaque string into `dst_opaque`.
503 * WHY: The stock field order + conditional oss.asize is fiddly formatting; kept
504 * in one helper so the wire layout lives in a single place and copy_tpc's
505 * run stage reads as a sequence of opens.
506 * HOW: Emit tpc.key/src/lfn/dlg/spr/tpr/dlgon/stage in the stock order. tpc.dlg
507 * names the originally-requested source endpoint and is inert while
508 * tpc.dlgon=0 (this client never delegates; emitted for XrdCl wire parity).
509 * oss.asize is appended only when the best-effort stat learned the size.
510 */
511static void
512tpc_build_dst_opaque(char *dst_opaque, size_t need, const tpc_params_t *p)
513{
514 char asize[48];
515
516 asize[0] = '\0';
517 if (p->size >= 0) {
518 snprintf(asize, sizeof(asize), "&oss.asize=%lld", (long long) p->size);
519 }
520 snprintf(dst_opaque, need,
521 "tpc.key=%s&tpc.src=%s&tpc.lfn=%s&tpc.dlg=%s:%d&tpc.spr=root"
522 "&tpc.tpr=root&tpc.dlgon=0%s&tpc.stage=copy%s%s",
523 p->key, p->src_hp, p->su->path, p->su->host, p->su->port, asize,
524 p->tok ? "&tpc.token_mode=" : "", p->tok ? p->tok : "");
525}

528/*
529 * WHAT: Stage 2 (run) — allocate the opaque strings, open + arm the destination,
530 * then open the source to register the key. Returns 0 with both handles
531 * live; on any failure tears down and returns -1.
532 * WHY: The rendezvous is a fixed open/sync/open sequence; grouping it keeps
533 * copy_tpc's top level free of the per-step teardown branches.
534 * HOW: Heap-size and build both opaques, connect+open the destination (cgiC2Dst)
535 * and sync it (setup/arm), then open the source (cgiC2Src, tpc.dst =
536 * dc.host, the puller endpoint the source will see). The source open runs
537 * with tpc_coord_defer=1 so brix_recv surfaces the kXR_waitresp deferral
538 * rather than blocking — the pull that satisfies it is only triggered by
539 * the verify-stage sync, so blocking here would deadlock the rendezvous.
540 * The source connection stays open so the registration remains live.
541 */
542static int
543tpc_run_rendezvous(tpc_state_t *s, const tpc_params_t *p, brix_status *st)
544{
545 /* Heap-size the opaque strings to the actual host/path/key/token lengths. */
546 size_t need = XRDC_HOSTPORT_MAX + strlen(p->su->host) + strlen(p->su->path)
547 + strlen(p->du->host) + strlen(p->du->path) + 40
548 + (p->tok ? strlen(p->tok) : 0) + 192;
549
550 s->src_opaque = (char *) malloc(need);
551 s->dst_opaque = (char *) malloc(need);
552 if (s->src_opaque == NULL || s->dst_opaque == NULL) {
553 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");

561 }

564 p->o->posc, &s->df, st) != 0) {
565 return tpc_state_teardown(s, -1, st);
566 }
567 s->dopen = 1;
568
569 if (brix_file_sync(&s->dc, &s->df, st) != 0) { /* rendezvous setup / arm */
570 return tpc_state_teardown(s, -1, st);
571 }
572
573 snprintf(s->src_opaque, need, "tpc.key=%s&tpc.dst=%s&tpc.stage=copy%s%s",
574 p->key, s->dc.host,
575 p->tok ? "&tpc.token_mode=" : "", p->tok ? p->tok : "");
576 s->sc.tpc_coord_defer = 1;
577 if (brix_file_open_opaque(&s->sc, p->su->path, s->src_opaque, 0, 0, 0,
578 &s->sf, st) != 0) {
579 return tpc_state_teardown(s, -1, st);
580 }
581 s->sopen = 1;
582 return 0;
583}

588 const brix_opts *co, brix_status *st)
589{
590 tpc_state_t s = { 0 };
591 tpc_params_t p = {
592 .su = su, .du = du, .o = o, .co = co, .size = -1,
593 .tok = (o->tpc_token_mode && o->tpc_token_mode[0])
594 ? o->tpc_token_mode : NULL,
595 };
596
597 /* 1. Placement: key + source session + best-effort stat. */
598 if (tpc_setup_source(&s, &p, st) != 0) {
599 return -1;
600 }
601
602 /* 2. Rendezvous: open + arm destination, then register the key at source. */
603 if (tpc_run_rendezvous(&s, &p, st) != 0) {
604 return -1;
605 }
606
607 /* 3. Trigger + await completion (reply may be deferred via kXR_waitresp →
608 * kXR_attn(asynresp); brix_recv unwraps it). */
609 if (s.dc.io.timeout_ms < 300000) {
610 s.dc.io.timeout_ms = 300000; /* 5 min for the deferred pull */
611 }
612 if (brix_file_sync(&s.dc, &s.df, st) != 0) {
613 return tpc_state_teardown(&s, -1, st);
614 }
615
616 return tpc_state_teardown(&s, 0, st);
617}
