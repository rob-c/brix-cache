34int
35brix_ssi_match(ngx_stream_brix_srv_conf_t *conf, const char *path,
36 const char **service, size_t *service_len)
37{
38 const char *svc;
39 size_t n;
40
41 if (conf == NULL || !conf->ssi_enable || path == NULL) {
42 return 0;
43 }
44 if (ngx_strncmp(path, BRIX_SSI_PREFIX, BRIX_SSI_PREFIX_LEN) != 0) {
45 return 0;
46 }
47 svc = path + BRIX_SSI_PREFIX_LEN;
48 n = 0;
49 while (svc[n] != '\0' && svc[n] != '/' && svc[n] != '?') {
50 n++;
51 }
52 if (n == 0 || n >= 64) {
53 return 0;
54 }
55 *service = svc;
56 *service_len = n;
57 return 1;
58}

60/* ---- Gate + configure the CTA tape service during SSI open ----
61 *
62 * WHAT: For service name "cta", verifies `brix_ssi_service cta;` is enabled and
63 * pushes the journal path + executor backend down to the service. Returns 1 when
64 * the open may proceed (either non-CTA, or CTA enabled + configured); returns 0
65 * after sending a kXR_NotFound error and stashing its status in *out_rc.
66 *
67 * WHY: The CTA service exposes a storage-control surface that must stay dark
68 * unless explicitly enabled; isolating the gate keeps brix_ssi_open a flat
69 * early-return sequence and confines the temporary journal buffer to one scope.
70 *
71 * HOW: 1) Non-"cta" services short-circuit to proceed. 2) Reject when conf is
72 * absent or ssi_cta_enable is off — indistinguishable from an unknown service.
73 * 3) Copy the journal directive into a null-terminated buffer (bounded) and
74 * apply it with the executor flag.
75 */
76static int
77ssi_open_cta_gate(brix_ctx_t *ctx, ngx_connection_t *c,
78 ngx_stream_brix_srv_conf_t *conf, const char *svc,
79 ngx_int_t *out_rc)
80{
81 char jbuf[1024];
82 size_t jlen;
83
84 if (ngx_strcmp(svc, "cta") != 0) {
85 return 1;
86 }
87 if (conf == NULL || !conf->ssi_cta_enable) {
88 *out_rc = brix_send_error(ctx, c, kXR_NotFound, "unknown SSI service");
89 return 0;
90 }
91 jlen = conf->ssi_cta_journal.len;
92 if (jlen >= sizeof(jbuf)) {
93 jlen = sizeof(jbuf) - 1;
94 }
95 ngx_memcpy(jbuf, conf->ssi_cta_journal.data, jlen);
96 jbuf[jlen] = '\0';
97 brix_ssi_cta_configure(jbuf, conf->ssi_cta_executor == 1);
98 return 1;
99}

101/* ---- Build and queue the byte-exact SSI open reply ----
102 *
103 * WHAT: Assembles the response header + 12-byte ServerOpenBody (carrying the file
104 * handle in fhandle[0]), optionally followed by a synthesized StatInfo string,
105 * and queues it. Returns the queue result, or a kXR_NoMemory error status if the
106 * response buffer cannot be allocated.
107 *
108 * WHY: The libXrdSsi client opens with kXR_retstat and refuses the reply unless a
109 * StatInfo string follows the ServerOpenBody. Keeping the exact layout (header,
110 * then body, then optional "id size flags mtime\0") in one helper preserves the
111 * wire contract while shrinking the orchestrator.
112 *
113 * HOW: 1) Zero the body and stamp the handle index. 2) When want_stat, format a
114 * plausible stat (id 0, size 0, read+write flags, mtime now) matching the
115 * open-path format and grow bodylen by its NUL-terminated length. 3) Allocate the
116 * header+body(+stat) buffer, build the header for the current streamid, copy the
117 * body, append the stat string when present, and queue.
118 */
119static ngx_int_t
120ssi_open_send_reply(brix_ctx_t *ctx, ngx_connection_t *c, int idx, int want_stat)
121{
122 ServerOpenBody body;
123 char statbuf[128];
124 size_t total, bodylen;
125 u_char *buf;
126
127 ngx_memzero(&body, sizeof(body));
128 body.fhandle[0] = (u_char) idx;
129 bodylen = sizeof(ServerOpenBody);
130
131 statbuf[0] = '\0';
132 if (want_stat) {
133 ngx_snprintf((u_char *) statbuf, sizeof(statbuf), "%d %d %d %T%Z",
134 0, 0, kXR_readable | kXR_writable, ngx_time());
135 bodylen += ngx_strlen(statbuf) + 1;
136 }
137
138 total = XRD_RESPONSE_HDR_LEN + bodylen;
139 buf = ngx_palloc(c->pool, total);
140 if (buf == NULL) {
141 return brix_send_error(ctx, c, kXR_NoMemory, "ssi resp");
142 }
143 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok, (uint32_t) bodylen,
144 (ServerResponseHdr *) buf);
145 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body, sizeof(ServerOpenBody));
146 if (want_stat) {
147 size_t slen = ngx_strlen(statbuf) + 1;
148 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + sizeof(ServerOpenBody),
149 statbuf, slen);
150 }
151
152 return brix_queue_response(ctx, c, buf, total);
153}

155ngx_int_t
156brix_ssi_open(brix_ctx_t *ctx, ngx_connection_t *c,
157 const char *service, size_t service_len, uint16_t options)
158{
159 ngx_connection_t *conn = c;
160 brix_ssi_session_t *sess;
161 brix_ssi_provider_t prov;
162 char svc[64];
163 int idx, devnull;
164 ngx_int_t gate_rc;
165 int want_stat = (options & kXR_retstat) ? 1 : 0;
166
167 ngx_stream_brix_srv_conf_t *conf = ngx_stream_get_module_srv_conf(
168 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
169
170 ngx_memcpy(svc, service, service_len);
171 svc[service_len] = '\0';
172
173 if (!brix_ssi_provider_lookup(svc, &prov)) {
174 return brix_send_error(ctx, c, kXR_NotFound, "unknown SSI service");
175 }
176
177 if (!ssi_open_cta_gate(ctx, c, conf, svc, &gate_rc)) {
178 return gate_rc;
179 }
180
181 idx = brix_alloc_fhandle(ctx);
182 if (idx < 0) {
183 return brix_send_error(ctx, c, kXR_ServerError, "too many open files");
184 }
185
186 /* A real (harmless) fd keeps the slot "in use" so close()/free work via the
187 * normal path; all I/O is intercepted on ctx->files[idx].ssi first. */
188 devnull = open("/dev/null", O_RDONLY | O_CLOEXEC);
189 if (devnull < 0) {
190 brix_free_fhandle(ctx, idx);
191 return brix_send_error(ctx, c, kXR_ServerError, "ssi open");
192 }
193
194 sess = brix_ssi_session_create(conn->pool, service, service_len, &prov);
195 if (sess == NULL) {
196 close(devnull);
197 brix_free_fhandle(ctx, idx);
198 return brix_send_error(ctx, c, kXR_NoMemory, "ssi alloc");
199 }
200 /* Key the registry by the session's own address (unique per live session)
201 * so async completions resolve THIS session, not whichever handle shares the
202 * connection. The generation check catches a recycled address. */
203 sess->conn_id = (uintptr_t) sess;
204 if (conf != NULL) {
205 sess->max_inflight = (int) conf->ssi_max_inflight;
206 sess->request_max = conf->ssi_request_max;
207 sess->response_max = conf->ssi_response_max;
208 }
209 brix_ssi_registry_add(sess->conn_id, sess);
210
211 ctx->files[idx].fd = devnull;
212 ctx->files[idx].ssi = sess; /* now a session, not a req */
213 ctx->files[idx].readable = 1;
214 ctx->files[idx].writable = 1;
215 ctx->files[idx].is_regular = 0;
216
217 return ssi_open_send_reply(ctx, c, idx, want_stat);
218}

220ngx_int_t
221brix_ssi_query(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
222 const unsigned char *body, size_t body_len)
223{
224 brix_ssi_session_t *sess = ctx->files[idx].ssi;
225 brix_ssi_req_t *rq;
226 int cmd;
227 uint32_t id, size;
228 u_char *buf;
229 size_t total;
230
231 if (body_len < BRIX_SSI_RRINFO_LEN) {
232 return brix_send_error(ctx, c, kXR_ArgInvalid, "short SSI control");
233 }
234 brix_ssi_rrinfo_decode(body, &cmd, &id, &size);
235
236 if (cmd == BRIX_SSI_CMD_CAN) {
237 brix_ssi_session_drop(sess, id);
238 return brix_send_ok(ctx, c, NULL, 0);
239 }
240
241 /* response-wait (Rwt). For the synchronous core the response is ready once
242 * the request write completed; if the reqId is unknown or no response is
243 * pending yet, treat as a protocol error. */
244 rq = brix_ssi_session_req(sess, id, 0);
245 if (rq == NULL || !rq->responded) {
246 return brix_send_error(ctx, c, kXR_InvalidRequest,
247 "no SSI response pending");
248 }
249
250 if (rq->error) {
251 BRIX_SRV_METRIC_INC(ctx, ssi_errors_total);
252 return brix_send_error(ctx, c,
253 rq->err_code ? rq->err_code : kXR_ServerError,
254 rq->err_text[0] ? rq->err_text : "SSI error");
255 }
256
257 /*
258 * Streaming response: reply pendResp with metadata only (no inline data);
259 * the client pulls the body via kXR_read (GetResp maps pendResp -> isStream).
260 * Unary response: reply fullResp with the data inline.
261 */
262 if (rq->streaming) {
263 total = brix_ssi_reply_len(rq->meta_len, 0);
264 buf = ngx_palloc(c->pool, total);
265 if (buf == NULL) {
266 return brix_send_error(ctx, c, kXR_NoMemory, "ssi reply");
267 }
268 brix_ssi_reply_build(BRIX_SSI_ATTN_PEND, rq->meta, rq->meta_len,
269 NULL, 0, buf);
270 return brix_send_ok(ctx, c, buf, total);
271 }
272
273 total = brix_ssi_reply_len(rq->meta_len, rq->resp.len);
274 buf = ngx_palloc(c->pool, total);
275 if (buf == NULL) {
276 return brix_send_error(ctx, c, kXR_NoMemory, "ssi reply");
277 }
278 brix_ssi_reply_build(BRIX_SSI_ATTN_FULL, rq->meta, rq->meta_len,
279 rq->resp.data, rq->resp.len, buf);
280 return brix_send_ok(ctx, c, buf, total);
281}

283ngx_int_t
284brix_ssi_read(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
285 uint64_t offset, uint32_t rlen)
286{
287 brix_ssi_session_t *sess = ctx->files[idx].ssi;
288 brix_ssi_req_t *rq;
289 unsigned char off8[8];
290 int cmd;
291 uint32_t id, size;
292 size_t avail, n;
293
294 /* The SSI read offset field is a big-endian XrdSsiRRInfo; re-serialize the
295 * decoded 64-bit offset back to its 8 wire bytes and decode the reqId. */
296 off8[0] = (u_char) (offset >> 56); off8[1] = (u_char) (offset >> 48);
297 off8[2] = (u_char) (offset >> 40); off8[3] = (u_char) (offset >> 32);
298 off8[4] = (u_char) (offset >> 24); off8[5] = (u_char) (offset >> 16);
299 off8[6] = (u_char) (offset >> 8); off8[7] = (u_char) (offset);
300 brix_ssi_rrinfo_decode(off8, &cmd, &id, &size);
301
302 rq = brix_ssi_session_req(sess, id, 0);
303 if (rq == NULL) {
304 return brix_send_ok(ctx, c, NULL, 0); /* unknown reqId → nothing */
305 }
306
307 /* Write-until-read protocol: a request never given an explicit size is
308 * dispatched on its first read (the writes have all accumulated by now). */
309 if (!rq->dispatched) {
310 brix_ssi_dispatch(rq, sess->provider.process);
311 BRIX_SRV_METRIC_INC(ctx, ssi_requests_total);
312 }
313
314 if (!rq->responded || rq->error) {
315 return brix_send_ok(ctx, c, NULL, 0); /* nothing to stream */
316 }
317 if (rq->read_cursor >= rq->resp.len) {
318 return brix_send_ok(ctx, c, NULL, 0); /* EOF */
319 }
320 avail = rq->resp.len - rq->read_cursor;
321 n = (rlen < avail) ? rlen : avail;
322 {
323 u_char *p = rq->resp.data + rq->read_cursor;
324 rq->read_cursor += n;
325 return brix_send_ok(ctx, c, p, n);
326 }
