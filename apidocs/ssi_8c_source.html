30int
31brix_ssi_match(ngx_stream_brix_srv_conf_t *conf, const char *path,
32 const char **service, size_t *service_len)
33{
34 const char *svc;
35 size_t n;
36
37 if (conf == NULL || !conf->ssi_enable || path == NULL) {
38 return 0;
39 }
40 if (ngx_strncmp(path, BRIX_SSI_PREFIX, BRIX_SSI_PREFIX_LEN) != 0) {
41 return 0;
42 }
43 svc = path + BRIX_SSI_PREFIX_LEN;
44 n = 0;
45 while (svc[n] != '\0' && svc[n] != '/' && svc[n] != '?') {
46 n++;
47 }
48 if (n == 0 || n >= 64) {
49 return 0;
50 }
51 *service = svc;
52 *service_len = n;
53 return 1;
54}

56/*
57 * Responder state. The synchronous/submit phase fills only `rq` (ctx/c NULL, so
58 * alerts drop — there is no push channel yet). The completion phase (driven from
59 * the deferral timer) also carries ctx/c, so alerts are pushed live.
60 */
61typedef struct {
62 brix_ssi_req_t *rq;
63 brix_ctx_t *ctx; /* non-NULL only when a push channel exists */
64 ngx_connection_t *c;
65} ssi_resp_state_t;

67static void
68ssi_resp_set_metadata(brix_ssi_responder_t *r, const unsigned char *md,
69 size_t len)
70{
71 brix_ssi_req_t *rq = ((ssi_resp_state_t *) r->state)->rq;
72
73 if (len > BRIX_SSI_META_MAX) {
74 len = BRIX_SSI_META_MAX;
75 }
76 if (rq->meta == NULL) {
77 rq->meta = ngx_palloc(rq->pool, BRIX_SSI_META_MAX);
78 if (rq->meta == NULL) {
79 return;
80 }
81 }
82 ngx_memcpy(rq->meta, md, len);
83 rq->meta_len = len;
84}

86static void
87ssi_resp_set_response(brix_ssi_responder_t *r, const unsigned char *buf,
88 size_t len, int last)
89{
90 brix_ssi_req_t *rq = ((ssi_resp_state_t *) r->state)->rq;
91
92 /* A non-final chunk marks the response as streamed: it is delivered to the
93 * client via kXR_read pulls (pendResp) rather than inline in the reply. */
94 if (!last) {
95 rq->streaming = 1;
96 }
97 {
98 size_t cap = rq->response_max > 0 ? rq->response_max : BRIX_SSI_RESP_MAX;
99 if (brix_ssi_respbuf_append(&rq->resp, buf, len, cap) != 0) {
100 rq->error = 1;
101 rq->err_code = kXR_NoMemory;
102 return;
103 }
104 }
105 rq->responded = 1;
106}

108static void
109ssi_resp_alert(brix_ssi_responder_t *r, const unsigned char *buf, size_t len)
110{
111 ssi_resp_state_t *st = r->state;
112
113 /* Deliverable only when a push channel exists (the deferred completion phase)
114 * and the client is awaiting a pushed response. The synchronous core has no
115 * out-of-band channel, so it drops alerts (Phase-1 behaviour). */
116 if (st->ctx != NULL && st->rq->waiting) {
117 brix_ssi_deliver_alert(st->ctx, st->c, st->rq, buf, len);
118 }
119}

121static void
122ssi_resp_error(brix_ssi_responder_t *r, int code, const char *text)
123{
124 brix_ssi_req_t *rq = ((ssi_resp_state_t *) r->state)->rq;
125
126 rq->error = 1;
127 /* The service reports a POSIX errno (XrdSsi eNum); map it to a kXR code. */
128 rq->err_code = code ? (int) brix_kxr_from_errno(code) : kXR_ServerError;
129 if (text != NULL) {
130 ngx_cpystrn((u_char *) rq->err_text, (u_char *) text,
131 sizeof(rq->err_text));
132 }
133 rq->responded = 1;
134}

172/* Run the resolved service over the accumulated request (submit phase). The
173 * service may answer inline or call r->defer to be completed later. */
174static void
175ssi_dispatch(brix_ssi_req_t *rq, brix_ssi_process_fn process)
176{
177 brix_ssi_responder_t r;
178 ssi_resp_state_t st = { rq, NULL, NULL };
179
180 ssi_responder_init(&r, &st, ssi_resp_defer_accept);
181
182 rq->dispatched = 1;
183 if (process(rq->req, rq->req_len, &r) != 0 && !rq->responded) {
184 rq->error = 1;
185 rq->err_code = kXR_ServerError;
186 rq->responded = 1;
187 }
188}

190/* Re-invoke a deferred service to produce its response now (completion phase):
191 * defer is refused (so the service answers inline) and ctx/c are present (so the
192 * service's alerts are pushed live before the terminal response). */
193static void
194ssi_complete_deferred(brix_ctx_t *ctx, ngx_connection_t *c,
195 brix_ssi_session_t *sess, brix_ssi_req_t *rq)
196{
197 brix_ssi_responder_t r;
198 ssi_resp_state_t st = { rq, ctx, c };
199
200 ssi_responder_init(&r, &st, ssi_resp_defer_refuse);
201 if (sess->provider.process(rq->req, rq->req_len, &r) != 0 && !rq->responded) {
202 rq->error = 1;
203 rq->err_code = kXR_ServerError;
204 rq->responded = 1;
205 }
206}

208/* ------------------------------------------------------------------ */
209/* Async deferral: an event-loop timer drives the deferred completion. */
210/* ------------------------------------------------------------------ */

212/*
213 * The timer state is heap-allocated (NOT pool-allocated) so it has an independent
214 * lifetime: handle teardown cancels it (ngx_del_timer + free) before the
215 * connection pool dies, and the timer handler frees it on fire. It carries the
216 * session key + generation (never a raw session pointer) so the handler resolves
217 * a LIVE session through the registry — the use-after-free guard.
218 */
219typedef struct {
220 ngx_event_t timer;
221 brix_ctx_t *ctx;
222 ngx_connection_t *c;
223 uintptr_t session_key;
224 uint64_t generation;
225 uint32_t req_id;
226} ssi_defer_t;
227
228static void
229ssi_defer_cancel(void *p)
230{
231 ssi_defer_t *d = p;
232
233 if (d == NULL) {
234 return;
235 }
236 if (d->timer.timer_set) {
237 ngx_del_timer(&d->timer);
238 }
239 ngx_free(d);
240}
241
242static void
243ssi_defer_timer_handler(ngx_event_t *ev)
244{
245 ssi_defer_t *d = ev->data;
246 brix_ssi_session_t *s;
247 brix_ssi_req_t *rq;
248
249 /* Liveness gate: a NULL find means the session/connection is gone (teardown
250 * removed it). Do not touch ctx/c in that case. */
251 s = brix_ssi_registry_find(d->session_key, d->generation);
252 if (s != NULL) {
253 rq = brix_ssi_session_req(s, d->req_id, 0);
254 if (rq != NULL && !rq->responded) {
255 brix_ssi_dlv_kind kind;
256
257 rq->defer_ctx = NULL; /* consume the timer before delivering */
258 ssi_complete_deferred(d->ctx, d->c, s, rq);
259 /* A streamed response is signalled PEND (client pulls the body via
260 * kXR_read); a unary response is pushed inline; an error is terminal. */
261 kind = rq->error ? SSI_DLV_ERROR
262 : rq->streaming ? SSI_DLV_PEND
263 : SSI_DLV_RESPONSE;
264 brix_ssi_deliver(d->ctx, d->c, s, d->req_id, kind);
265 } else if (rq != NULL) {
266 rq->defer_ctx = NULL;
267 }
268 }
269 ngx_free(d);
270}

272static ngx_int_t
273ssi_defer_arm(brix_ctx_t *ctx, ngx_connection_t *c,
274 brix_ssi_session_t *sess, brix_ssi_req_t *rq)
275{
276 ssi_defer_t *d = ngx_alloc(sizeof(*d), c->log);
277
278 if (d == NULL) {
279 return NGX_ERROR;
280 }
281 ngx_memzero(&d->timer, sizeof(d->timer));
282 d->timer.handler = ssi_defer_timer_handler;
283 d->timer.data = d;
284 d->timer.log = c->log;
285 d->ctx = ctx;
286 d->c = c;
287 d->session_key = sess->conn_id;
288 d->generation = sess->generation;
289 d->req_id = rq->req_id;
290 rq->defer_ctx = d;
291 ngx_add_timer(&d->timer, 10); /* 10ms — proves the push without stalling */
292 return NGX_OK;
293}

295/*
296 * Teardown hook (called from brix_free_fhandle before the .ssi slot is cleared,
297 * on both kXR_close and connection disconnect): cancel any armed deferral timers
298 * and unregister the session, so no async completion can run against freed memory.
299 */
300void
301brix_ssi_handle_cleanup(void *ssi_session)
302{
303 brix_ssi_session_t *sess = ssi_session;
304 int i;
305
306 if (sess == NULL) {
307 return;
308 }
309 for (i = 0; i < BRIX_SSI_MAX_INFLIGHT; i++) {
310 if (sess->rr[i].in_use && sess->rr[i].defer_ctx != NULL) {
311 ssi_defer_cancel(sess->rr[i].defer_ctx);
312 sess->rr[i].defer_ctx = NULL;
313 }
314 }
315 brix_ssi_registry_remove(sess->conn_id);
316}

318ngx_int_t
319brix_ssi_open(brix_ctx_t *ctx, ngx_connection_t *c,
320 const char *service, size_t service_len, uint16_t options)
321{
322 ngx_connection_t *conn = c;
323 brix_ssi_session_t *sess;
324 brix_ssi_provider_t prov;
325 char svc[64];
326 int idx, devnull;
327 ServerOpenBody body;
328 size_t total, bodylen;
329 u_char *buf;
330 char statbuf[128];
331 int want_stat = (options & kXR_retstat) ? 1 : 0;
332
333 ngx_stream_brix_srv_conf_t *conf = ngx_stream_get_module_srv_conf(
334 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
335 int is_cta;
336
337 ngx_memcpy(svc, service, service_len);
338 svc[service_len] = '\0';
339
340 if (!brix_ssi_provider_lookup(svc, &prov)) {
341 return brix_send_error(ctx, c, kXR_NotFound, "unknown SSI service");
342 }
343
344 /* The CTA tape service exposes a storage-control surface — gate it behind
345 * `brix_ssi_service cta;`. When enabled, push the config down to the
346 * service (journal + executor backend). */
347 is_cta = (ngx_strcmp(svc, "cta") == 0);
348 if (is_cta) {
349 if (conf == NULL || !conf->ssi_cta_enable) {
350 return brix_send_error(ctx, c, kXR_NotFound, "unknown SSI service");
351 }
352 {
353 char jbuf[1024];
354 size_t jlen = conf->ssi_cta_journal.len;
355 if (jlen >= sizeof(jbuf)) {
356 jlen = sizeof(jbuf) - 1;
357 }
358 ngx_memcpy(jbuf, conf->ssi_cta_journal.data, jlen);
359 jbuf[jlen] = '\0';
360 brix_ssi_cta_configure(jbuf, conf->ssi_cta_executor == 1);
361 }
362 }
363
364 idx = brix_alloc_fhandle(ctx);
365 if (idx < 0) {
366 return brix_send_error(ctx, c, kXR_ServerError, "too many open files");
367 }
368
369 /* A real (harmless) fd keeps the slot "in use" so close()/free work via the
370 * normal path; all I/O is intercepted on ctx->files[idx].ssi first. */
371 devnull = open("/dev/null", O_RDONLY | O_CLOEXEC);
372 if (devnull < 0) {
373 brix_free_fhandle(ctx, idx);
374 return brix_send_error(ctx, c, kXR_ServerError, "ssi open");
375 }
376
377 sess = brix_ssi_session_create(conn->pool, service, service_len, &prov);
378 if (sess == NULL) {
379 close(devnull);
380 brix_free_fhandle(ctx, idx);
381 return brix_send_error(ctx, c, kXR_NoMemory, "ssi alloc");
382 }
383 /* Key the registry by the session's own address (unique per live session)
384 * so async completions resolve THIS session, not whichever handle shares the
385 * connection. The generation check catches a recycled address. */
386 sess->conn_id = (uintptr_t) sess;
387 if (conf != NULL) {
388 sess->max_inflight = (int) conf->ssi_max_inflight;
389 sess->request_max = conf->ssi_request_max;
390 sess->response_max = conf->ssi_response_max;
391 }
392 brix_ssi_registry_add(sess->conn_id, sess);
393
394 ctx->files[idx].fd = devnull;
395 ctx->files[idx].ssi = sess; /* now a session, not a req */
396 ctx->files[idx].readable = 1;
397 ctx->files[idx].writable = 1;
398 ctx->files[idx].is_regular = 0;
399
400 /*
401 * The libXrdSsi client opens the resource with kXR_retstat and refuses the
402 * reply unless a StatInfo string follows the 12-byte ServerOpenBody. For the
403 * virtual SSI resource we synthesize a plausible stat: id 0, size 0,
404 * read+write flags, mtime now — matching the open-path "id size flags mtime"
405 * format (open_resolved_file.c).
406 */
407 ngx_memzero(&body, sizeof(body));
408 body.fhandle[0] = (u_char) idx;
409 bodylen = sizeof(ServerOpenBody);
410
411 statbuf[0] = '\0';
412 if (want_stat) {
413 ngx_snprintf((u_char *) statbuf, sizeof(statbuf), "%d %d %d %T%Z",
414 0, 0, kXR_readable | kXR_writable, ngx_time());
415 bodylen += ngx_strlen(statbuf) + 1;
416 }
417
418 total = XRD_RESPONSE_HDR_LEN + bodylen;
419 buf = ngx_palloc(conn->pool, total);
420 if (buf == NULL) {
421 return brix_send_error(ctx, c, kXR_NoMemory, "ssi resp");
422 }
423 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok, (uint32_t) bodylen,
424 (ServerResponseHdr *) buf);
425 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body, sizeof(ServerOpenBody));
426 if (want_stat) {
427 size_t slen = ngx_strlen(statbuf) + 1;
428 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + sizeof(ServerOpenBody),
429 statbuf, slen);
430 }
431
432 return brix_queue_response(ctx, c, buf, total);
433}

435ngx_int_t
436brix_ssi_write(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
437 const unsigned char off8[8])
438{
439 brix_ssi_session_t *sess = ctx->files[idx].ssi;
440 brix_ssi_req_t *rq;
441 int cmd;
442 uint32_t id, size;
443 size_t n = ctx->recv.cur_dlen;
444
445 brix_ssi_rrinfo_decode(off8, &cmd, &id, &size);
446
447 if (cmd == BRIX_SSI_CMD_CAN) {
448 brix_ssi_session_drop(sess, id); /* request-phase cancel */
449 return brix_send_ok(ctx, c, NULL, 0);
450 }
451
452 rq = brix_ssi_session_req(sess, id, 1);
453 if (rq == NULL) {
454 return brix_send_error(ctx, c, kXR_Overloaded,
455 "too many concurrent SSI requests");
456 }
457 if (rq->dispatched) {
458 return brix_send_error(ctx, c, kXR_FileLocked,
459 "SSI request already dispatched");
460 }
461
462 if (!rq->have_size) {
463 rq->req_expected = size;
464 rq->have_size = 1;
465 }
466
467 if (n > 0) {
468 /* request cap: the configured limit, never above the compile-time hard
469 * ceiling (the request buffer is sized to that ceiling). */
470 size_t reqcap = (sess->request_max > 0 &&
471 sess->request_max < BRIX_SSI_REQ_MAX)
472 ? sess->request_max : BRIX_SSI_REQ_MAX;
473 if (rq->req_len + n > reqcap) {
474 return brix_send_error(ctx, c, kXR_ArgTooLong,
475 "SSI request too large");
476 }
477 if (rq->req == NULL) {
478 rq->req = ngx_palloc(c->pool, BRIX_SSI_REQ_MAX);
479 if (rq->req == NULL) {
480 return brix_send_error(ctx, c, kXR_NoMemory, "ssi req buf");
481 }
482 }
483 if (ctx->recv.payload != NULL) {
484 ngx_memcpy(rq->req + rq->req_len, ctx->recv.payload, n);
485 rq->req_len += n;
486 }
487 }
488
489 /* When the client declared a total request size (req_expected > 0), dispatch
490 * as soon as that many bytes have accumulated. When no size was given
491 * (req_expected == 0 — the simple write-until-read protocol), DEFER dispatch
492 * to the first read so several writes accumulate into one request instead of
493 * the first write firing it prematurely (which then rejected every following
494 * write as "already dispatched"). */
495 if (rq->req_expected > 0 && rq->req_len >= rq->req_expected) {
496 ssi_dispatch(rq, sess->provider.process);
497 BRIX_SRV_METRIC_INC(ctx, ssi_requests_total);
498
499 /* Deferred service: capture the submit streamid, arm the async
500 * completion, and ack with kXR_waitresp — the response is pushed later
501 * via kXR_attn (brix_ssi_deliver). */
502 if (rq->deferred && !rq->responded) {
503 ngx_memcpy(rq->defer_streamid, ctx->recv.cur_streamid,
504 sizeof(rq->defer_streamid));
505 rq->waiting = 1;
506 if (ssi_defer_arm(ctx, c, sess, rq) != NGX_OK) {
507 return brix_send_error(ctx, c, kXR_NoMemory, "ssi defer");
508 }
509 return brix_send_waitresp(ctx, c);
510 }
511 }
512
513 return brix_send_ok(ctx, c, NULL, 0);
514}

516ngx_int_t
517brix_ssi_query(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
518 const unsigned char *body, size_t body_len)
519{
520 brix_ssi_session_t *sess = ctx->files[idx].ssi;
521 brix_ssi_req_t *rq;
522 int cmd;
523 uint32_t id, size;
524 u_char *buf;
525 size_t total;
526
527 if (body_len < BRIX_SSI_RRINFO_LEN) {
528 return brix_send_error(ctx, c, kXR_ArgInvalid, "short SSI control");
529 }
530 brix_ssi_rrinfo_decode(body, &cmd, &id, &size);
531
532 if (cmd == BRIX_SSI_CMD_CAN) {
533 brix_ssi_session_drop(sess, id);
534 return brix_send_ok(ctx, c, NULL, 0);
535 }
536
537 /* response-wait (Rwt). For the synchronous core the response is ready once
538 * the request write completed; if the reqId is unknown or no response is
539 * pending yet, treat as a protocol error. */
540 rq = brix_ssi_session_req(sess, id, 0);
541 if (rq == NULL || !rq->responded) {
542 return brix_send_error(ctx, c, kXR_InvalidRequest,
543 "no SSI response pending");
544 }
545
546 if (rq->error) {
547 BRIX_SRV_METRIC_INC(ctx, ssi_errors_total);
548 return brix_send_error(ctx, c,
549 rq->err_code ? rq->err_code : kXR_ServerError,
550 rq->err_text[0] ? rq->err_text : "SSI error");
551 }
552
553 /*
554 * Streaming response: reply pendResp with metadata only (no inline data);
555 * the client pulls the body via kXR_read (GetResp maps pendResp -> isStream).
556 * Unary response: reply fullResp with the data inline.
557 */
558 if (rq->streaming) {
559 total = brix_ssi_reply_len(rq->meta_len, 0);
560 buf = ngx_palloc(c->pool, total);
561 if (buf == NULL) {
562 return brix_send_error(ctx, c, kXR_NoMemory, "ssi reply");
563 }
564 brix_ssi_reply_build(BRIX_SSI_ATTN_PEND, rq->meta, rq->meta_len,
565 NULL, 0, buf);
566 return brix_send_ok(ctx, c, buf, total);
567 }
568
569 total = brix_ssi_reply_len(rq->meta_len, rq->resp.len);
570 buf = ngx_palloc(c->pool, total);
571 if (buf == NULL) {
572 return brix_send_error(ctx, c, kXR_NoMemory, "ssi reply");
573 }
574 brix_ssi_reply_build(BRIX_SSI_ATTN_FULL, rq->meta, rq->meta_len,
575 rq->resp.data, rq->resp.len, buf);
576 return brix_send_ok(ctx, c, buf, total);
577}

579ngx_int_t
580brix_ssi_read(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
581 uint64_t offset, uint32_t rlen)
582{
583 brix_ssi_session_t *sess = ctx->files[idx].ssi;
584 brix_ssi_req_t *rq;
585 unsigned char off8[8];
586 int cmd;
587 uint32_t id, size;
588 size_t avail, n;
589
590 /* The SSI read offset field is a big-endian XrdSsiRRInfo; re-serialize the
591 * decoded 64-bit offset back to its 8 wire bytes and decode the reqId. */
592 off8[0] = (u_char) (offset >> 56); off8[1] = (u_char) (offset >> 48);
593 off8[2] = (u_char) (offset >> 40); off8[3] = (u_char) (offset >> 32);
594 off8[4] = (u_char) (offset >> 24); off8[5] = (u_char) (offset >> 16);
595 off8[6] = (u_char) (offset >> 8); off8[7] = (u_char) (offset);

597

599 if (rq == NULL) {
600 return brix_send_ok(ctx, c, NULL, 0); /* unknown reqId → nothing */
601 }
602
603 /* Write-until-read protocol: a request never given an explicit size is
604 * dispatched on its first read (the writes have all accumulated by now). */
605 if (!rq->dispatched) {
606 ssi_dispatch(rq, sess->provider.process);
607 BRIX_SRV_METRIC_INC(ctx, ssi_requests_total);
608 }
609
610 if (!rq->responded || rq->error) {
611 return brix_send_ok(ctx, c, NULL, 0); /* nothing to stream */
612 }
613 if (rq->read_cursor >= rq->resp.len) {
614 return brix_send_ok(ctx, c, NULL, 0); /* EOF */
615 }
616 avail = rq->resp.len - rq->read_cursor;
617 n = (rlen < avail) ? rlen : avail;
618 {
619 u_char *p = rq->resp.data + rq->read_cursor;
620 rq->read_cursor += n;
621 return brix_send_ok(ctx, c, p, n);
622 }
