4/*
5 * netopt.h — shared OS-level dead-peer reaping socket options.
6 *
7 * WHAT: one header-only helper that applies SO_KEEPALIVE (with tight
8 * TCP_KEEPIDLE/INTVL/CNT probes) and TCP_USER_TIMEOUT to a socket fd.
9 *
10 * WHY: Phase 39 introduced this block inline in src/connection/handler.c for the
11 * root:// accept path. Phase 50 needs the identical behaviour on the CMS
12 * client connect socket (src/cms/connect.c) and the CMS server accept socket
13 * (src/cms/server_handler.c) so a silently-dropped manager/data-node peer is
14 * reaped by the kernel even while no event-loop deadline is armed. Factoring it
15 * here keeps one definition and lets handler.c, connect.c and server_handler.c
16 * (a separate stream module) all share it via static inline — no new .c, so the
17 * build graph (the `config` source list) is unchanged.
18 *
19 * HOW: header-only `static ngx_inline`, compiled into each translation unit.
20 * Every setsockopt failure is deliberately NON-FATAL — a missing/denied option
21 * must never abort a connection. Each option is guarded by its feature macro so
22 * the helper compiles on platforms lacking the TCP_KEEP / USER_TIMEOUT opts.
23 */
24

30/*
31 * Apply dead-peer reaping options to fd.
32 * keepalive — enable SO_KEEPALIVE plus seconds-scale probes
33 * (30 s idle / 10 s interval / 3 probes), matching the
34 * Phase-39 root:// accept defaults.
35 * user_timeout_ms — TCP_USER_TIMEOUT in ms; 0 leaves the kernel default.
36 * Failures are swallowed (best-effort hardening, never a hard error).
37 */
38static ngx_inline void
39brix_apply_tcp_deadpeer_opts(ngx_socket_t fd, ngx_flag_t keepalive,
40 ngx_msec_t user_timeout_ms)
41{
42 if (keepalive) {
43 int on = 1;
44 (void) setsockopt(fd, SOL_SOCKET, SO_KEEPALIVE,
45 (const void *) &on, sizeof(on));
46#if defined(TCP_KEEPIDLE)
47 { int v = 30;
48 (void) setsockopt(fd, IPPROTO_TCP, TCP_KEEPIDLE,
49 (const void *) &v, sizeof(v)); }
50#endif
51#if defined(TCP_KEEPINTVL)
52 { int v = 10;
53 (void) setsockopt(fd, IPPROTO_TCP, TCP_KEEPINTVL,
54 (const void *) &v, sizeof(v)); }
55#endif
56#if defined(TCP_KEEPCNT)

72/*
73 * Select the TCP congestion-control algorithm on fd (e.g. "bbr", "cubic").
74 * algo — algorithm name; empty (len 0) leaves the kernel default untouched.
75 * Per-socket and best-effort: a missing/unavailable algorithm (e.g. the bbr
76 * module not loaded) fails the setsockopt silently and the connection proceeds
77 * on the kernel default — never a hard error. BBR is rate/RTT-model based and
78 * does not collapse cwnd on the spurious "loss" signals that packet reordering
79 * induces, so it improves throughput on reordering/lossy high-BDP links without
80 * any client change (the sender's behaviour governs a download).
81 */
82static ngx_inline void
83brix_apply_tcp_congestion(ngx_socket_t fd, ngx_str_t algo)
84{
85#if defined(TCP_CONGESTION)
86 if (algo.len > 0 && algo.len < 64) {
87 (void) setsockopt(fd, IPPROTO_TCP, TCP_CONGESTION,
88 (const void *) algo.data, (socklen_t) algo.len);
89 }
90#else
91 (void) fd;
92 (void) algo;
93#endif
94}
