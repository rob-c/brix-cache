1/*
2 * xfer.h — public contract for the unified durable-transfer engine.
3 *
4 * WHAT: One state machine (BEGIN -> MOVE -> COMMIT|ABORT) that the four durable
5 * write/transfer paths — normal staging, tape stage-out, proxy
6 * write-through, and TPC — are thin configurations of. This header is the
7 * only surface callers and movers share; the state-machine internals live
8 * in xfer_core.c, the byte movers in xfer_mover_*.c, audit in
9 * xfer_ledger.c, policy in xfer_policy.c, durability in xfer_journal.c /
10 * xfer_reconcile.c.
11 *
12 * WHY: Four subsystems each re-implemented commit/abort, confinement, async
13 * durability, external-process execution, and their own audit record. That
14 * fragmentation is the support burden (four places crash-safety can be
15 * wrong, two fork/reap models — only one crash-safe) and the auditing
16 * burden (four metric families, one access-log schema). This engine
17 * collapses them onto one envelope. See
18 * docs/superpowers/specs/2026-06-28-unified-durable-transfer-engine-design.md.
19 *
20 * HOW: The engine is introduced incrementally (spec §9). This header carries the
21 * full vocabulary up front so the seam is stable, but only the symbols a
22 * given phase has landed are defined in the .c files. Phase 1 lands the
23 * in-process pump mover (brix_xfer_pump_objects) extracted from
24 * compat/staged_file.c.
25 */
26

70/*
71 * Phase 1 — the delegated out-of-process agent harness.
72 *
73 * The single crash-safe way to run an external transfer process from an nginx
74 * worker. nginx's master SIGCHLD handler walks every SHM zone as an
75 * ngx_slab_pool_t and force-unlocks it; several module zones overwrite that
76 * header, so reaping ANY worker child SIGSEGVs the master. The agent is
77 * double-forked (reparented to init) so nginx never reaps it, and the agent —
78 * not nginx — reaps the transfer-command children. All blocking work
79 * (fork/exec/waitpid) runs in the agent, off the event loop, with no thread pool.
80 *
81 * Originally embedded in src/frm/stage.c for tape recall; lifted here so every
82 * external-process transfer (tape copycmd, write-through origin client, TPC curl)
83 * shares ONE fork/reap path. The harness is payload-agnostic: it shuttles
84 * fixed-size request/reply frames and invokes per-kind callbacks. Because the
85 * agent is a fork() (not exec), the callback function pointers are valid in the
86 * child.
87 */
88
89/* Per-kind agent behaviour. process() runs in the reparented AGENT (off the
90 * event loop; may fork/exec/waitpid its own children — SIGCHLD is SIG_DFL there).
91 * on_reply() and after_drain() run in the WORKER on the event loop. `data` is an
92 * opaque pointer passed to all three; agent-side mutations do not propagate back
93 * across the fork. */
94typedef struct {
95 size_t req_size; /* bytes per request frame */
96 size_t rep_size; /* bytes per reply frame */
97 void (*process)(const void *req, void *rep, void *data); /* agent */
98 void (*on_reply)(const void *rep, void *data); /* worker */
99 void (*after_drain)(void *data); /* worker, optional (may be 0) */
100 void *data;
101 const char *name; /* for log lines, e.g. "frm stage" */
102} brix_xfer_agent_ops_t;
103
104/* Worker-side handle for one agent. Zero-initialize before first attach; `fd < 0`
105 * means "no live agent" so dispatch fails closed. */
106typedef struct {
107 int fd; /* worker side of socketpair, or -1 */
108 ngx_connection_t *conn; /* nginx read-event connection */
109 const brix_xfer_agent_ops_t *ops;
110 ngx_log_t *log;
111 void *rep_buf; /* worker reply scratch, rep_size */
112} brix_xfer_agent_t;
113
114/* Spawn a reparented agent and register the worker side of its socketpair as an
115 * nginx read event. On any failure tears down and leaves a->fd = -1 (dispatch
116 * fails closed). Returns NGX_OK / NGX_ERROR. Also used internally for respawn. */
117ngx_int_t brix_xfer_agent_attach(brix_xfer_agent_t *a,
118 const brix_xfer_agent_ops_t *ops, ngx_log_t *log);
119
120/* Release the worker side (closes fd + frees the connection). Idempotent; the
121 * reparented agent observes EOF and exits on its own. */
122void brix_xfer_agent_teardown(brix_xfer_agent_t *a);
123
124/* Send one request frame (a->ops->req_size bytes) to the agent. Returns NGX_OK
125 * sent, NGX_AGAIN if the agent is backed up (retry later), NGX_ERROR on a dead
126 * or broken agent. */
127ngx_int_t brix_xfer_agent_dispatch(brix_xfer_agent_t *a, const void *req);

129/* ===================== Phase 2: the unified audit ledger =================== */

131/*
132 * One audit event, recorded at a transfer's terminal COMMIT or ABORT.
133 *
134 * Per the approved ledger model, metrics keep their existing per-subsystem names
135 * (each caller books its own as before — e.g. STAGE still books OP_WRITE); the
136 * ledger's contribution is the ONE consistent audit line emitted for every kind,
137 * so an operator has a single place to answer "what was published, by whom, by
138 * which path". The line is transport-agnostic — it works identically for stream
139 * root:// callers and HTTP S3/WebDAV callers, which have separate per-protocol
140 * access logs.
141 */
142typedef struct {
143 brix_xfer_kind_t kind;
144 const char *direction; /* "in" (into our storage) | "out" */
145 const char *path; /* final object path (sanitized on emit) */
146 const char *principal; /* authenticated identity, or NULL ("-") */
147 size_t bytes; /* committed object size */
148 brix_xfer_result_t result;
149 int sys_errno; /* meaningful when result != OK */
150 ngx_log_t *log;
151} brix_xfer_audit_t;
152
153/* Append one unified audit line for a terminal transfer. Best-effort: a sink it
154 * cannot open is warned once and skipped (auditing never blocks a transfer). The
155 * sink path is $BRIX_XFER_AUDIT_LOG, else <prefix>/logs/xfer_audit.log. */
156void brix_xfer_ledger_record(const brix_xfer_audit_t *ev);

158/* Stable human strings for a kind / result (used in the audit line and tests). */
159const char *brix_xfer_kind_str(brix_xfer_kind_t kind);

164/*
165 * Record a transfer reaching a terminal state. Every kind (STAGE/TAPE/WT, and
166 * later TPC) calls this at commit/abort instead of building an audit struct
167 * inline — one place, so the audit schema and (in a later step) the durable
168 * journal completion update have a single hook. Thin today (it emits the unified
169 * ledger line); it is the seam the journal/reconcile work attaches to without
170 * touching any caller again.

183/*
184 * Phase 1 — the in-process byte pump.
185 *
186 * Copy the whole source object into the destination object by reading through
187 * the SOURCE object's storage driver and writing through the DESTINATION
188 * object's driver. The two objects may live on different backends — this is the
189 * cross-mount move at the heart of every staged commit, so it is exactly where
190 * one side could later be an object/S3 store while the other stays POSIX. All
191 * raw byte I/O stays inside the SD backends (sd_posix.c et al.); this loop is
192 * driver-mediated and never touches a raw fd directly.
193 *
194 * Returns NGX_OK, or NGX_ERROR with errno set. EINTR is retried.
195 */
196ngx_int_t brix_xfer_pump_objects(brix_sd_obj_t *src, brix_sd_obj_t *dst);
