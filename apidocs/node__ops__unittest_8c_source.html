30/* Build an rrdata with C-string fields (lengths excluding NUL, as the real
31 * decoder produces). */
32static brix_cms_rrdata_t
33rr(const char *path, const char *path2, const char *mode)
34{
35 brix_cms_rrdata_t d;
36 memset(&d, 0, sizeof(d));
37 if (path) { d.path = (const unsigned char *) path; d.path_len = strlen(path); }
38 if (path2) { d.path2 = (const unsigned char *) path2; d.path2_len = strlen(path2); }
39 if (mode) { d.mode = (const unsigned char *) mode; d.mode_len = strlen(mode); }
40 return d;
41}
