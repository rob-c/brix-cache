31/* Build an rrdata with C-string fields (lengths excluding NUL, as the real
32 * decoder produces). */
33static brix_cms_rrdata_t
34rr(const char *path, const char *path2, const char *mode)
35{
36 brix_cms_rrdata_t d;
37 memset(&d, 0, sizeof(d));
38 if (path) { d.path = (const unsigned char *) path; d.path_len = strlen(path); }
39 if (path2) { d.path2 = (const unsigned char *) path2; d.path2_len = strlen(path2); }
40 if (mode) { d.mode = (const unsigned char *) mode; d.mode_len = strlen(mode); }
41 return d;
42}

149{
150 /* prepadd needs path + reqid; notify/prty pass through when present. */
151 brix_cms_rrdata_t d = rr("/atlas/f.root", NULL, NULL);
152 brix_cms_node_plan_t p;
153 d.reqid = (const unsigned char *) "42.7@mgr"; d.reqid_len = 8;
154 d.notify = (const unsigned char *) "udp://n:1"; d.notify_len = 9;
155 d.prty = (const unsigned char *) "2"; d.prty_len = 1;
156 CHECK(brix_cms_node_plan(K_PREPADD, &d, &p) == 0);
157 CHECK(p.action == XRDCMS_NACT_PREPADD);
158 CHECK(p.path && strcmp(p.path, "/atlas/f.root") == 0);
159 CHECK(p.reqid && strcmp(p.reqid, "42.7@mgr") == 0);
160 CHECK(p.notify && strcmp(p.notify, "udp://n:1") == 0);
