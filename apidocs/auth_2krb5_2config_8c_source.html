4/*
5 * config.c — configure-time Kerberos 5 (krb5) auth setup for the stream server
6 *
7 * WHAT: Implements brix_configure_krb5_auth(), invoked from server-conf
8 * post-processing to validate and pre-load the Kerberos state a server
9 * needs before it can accept "krb5" logins: the krb5_context, the parsed
10 * acceptor principal (krb5_principal_obj) and the keytab handle
11 * (krb5_keytab_obj). These are consumed at run time by
12 * brix_handle_krb5_auth() in auth.c.
13 *
14 * WHY: Doing this once at config time (rather than per connection) means
15 * misconfiguration — missing principal, unparseable name, unreadable
16 * keytab — fails the server fast with a clear ngx_conf_log_error at
17 * startup instead of denying every client later. It also avoids repeated
18 * keytab opens on the hot auth path.
19 *
20 * HOW: Returns NGX_OK immediately unless xcf->auth == BRIX_AUTH_KRB5. It
21 * requires brix_krb5_principal, then krb5_init_context, krb5_parse_name,
22 * and resolves the keytab (krb5_kt_resolve when brix_krb5_keytab is set,
23 * else krb5_kt_default). It probes the keytab with a start/end seq_get to
24 * confirm readability and logs the effective principal, keytab name and
25 * ip_check flag at NOTICE. All krb5 code is gated on BRIX_HAVE_KRB5; a
26 * build without it that nonetheless requests krb5 auth fails with EMERG.
27 * Errors are decoded via brix_krb5_error()/brix_krb5_free_error().
28 */

47/* ---- Require that a service principal was configured ----
48 *
49 * WHAT: Returns NGX_OK when brix_krb5_principal is set; otherwise logs an
50 * EMERG remediation diagnostic and returns NGX_ERROR.
51 *
52 * WHY: krb5 auth cannot present an acceptor identity to clients without a
53 * principal, so this precondition is checked before touching any krb5
54 * library state and fails the server startup with a precise hint.
55 *
56 * HOW: (1) If xcf->krb5.principal.len is non-zero, return NGX_OK. (2) Else emit
57 * the BRIX_DIAG_CONF EMERG guidance and return NGX_ERROR.
58 */
59static ngx_int_t
60brix_krb5_require_principal(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
61{
62 if (xcf->krb5.principal.len != 0) {
63 return NGX_OK;
64 }
65
66 BRIX_DIAG_CONF(NGX_LOG_EMERG, cf, 0,
67 "brix: krb5 auth is enabled but no service principal is set",
68 "brix_auth krb5 needs the service principal it presents to "
69 "clients, but brix_krb5_principal is missing",
70 "add e.g. brix_krb5_principal \"xrootd/host.example.org@REALM\"; "
71 "it must match a key in the keytab");
72 return NGX_ERROR;

75/* ---- Initialise the krb5_context ----
76 *
77 * WHAT: Creates xcf->krb5.context via krb5_init_context; returns NGX_OK on
78 * success or NGX_ERROR (after an EMERG log) on failure.
79 *
80 * WHY: Every subsequent krb5 call needs a context; establishing it in one place
81 * keeps the orchestrator flat and gives a single, clear failure point.
82 *
83 * HOW: (1) Call krb5_init_context. (2) On non-zero rc, log the numeric code at
84 * EMERG and return NGX_ERROR. (3) Otherwise return NGX_OK.
85 */
86static ngx_int_t
87brix_krb5_init_context(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
88{
89 krb5_error_code rc;
90
91 rc = krb5_init_context(&xcf->krb5.context);
92 if (rc != 0) {
93 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
94 "brix: krb5_init_context failed (%d)",

97 }

99 return NGX_OK;

102/* ---- Parse the configured principal into a krb5_principal_obj ----
103 *
104 * WHAT: Parses xcf->krb5.principal into xcf->krb5.principal_obj; returns NGX_OK
105 * on success or NGX_ERROR (after an EMERG log with the decoded krb5 error)
106 * on failure.
107 *
108 * WHY: The acceptor principal must be a valid krb5 name before it can be used to
109 * match keytab entries; parsing here surfaces malformed names at startup.
110 *
111 * HOW: (1) Call krb5_parse_name on the principal string. (2) On non-zero rc,
112 * decode the message with brix_krb5_error, log at EMERG, free the message,
113 * and return NGX_ERROR. (3) Otherwise return NGX_OK.
114 */
115static ngx_int_t
116brix_krb5_parse_principal(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
117{
118 krb5_error_code rc;
119 const char *kmsg;
120
121 rc = krb5_parse_name(xcf->krb5.context,
122 (const char *) xcf->krb5.principal.data,
123 &xcf->krb5.principal_obj);
124 if (rc != 0) {
125 kmsg = brix_krb5_error(xcf, rc);
126 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
127 "brix: cannot parse krb5 principal \"%V\": %s",
128 &xcf->krb5.principal, kmsg ? kmsg : "unknown");
129 brix_krb5_free_error(xcf, kmsg);
130 return NGX_ERROR;
131 }
132
133 return NGX_OK;
134}

136/* ---- Resolve and open the keytab ----
137 *
138 * WHAT: Opens xcf->krb5.keytab_obj from the configured keytab path, or the krb5
139 * default keytab when none is set; returns NGX_OK on success or NGX_ERROR
140 * (after an EMERG remediation log) on failure.
141 *
142 * WHY: The keytab holds the acceptor's long-term key; opening it at config time
143 * turns a missing/unreadable keytab into a fast startup failure rather than
144 * a per-connection auth error, and avoids repeated opens on the hot path.
145 *
146 * HOW: (1) If a keytab path is configured, krb5_kt_resolve it; else
147 * krb5_kt_default. (2) On non-zero rc, decode the message, emit the
148 * BRIX_DIAG_CONF EMERG guidance, free the message, and return NGX_ERROR.
149 * (3) Otherwise return NGX_OK.
150 */
151static ngx_int_t
152brix_krb5_open_keytab(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
153{
154 krb5_error_code rc;
155 const char *kmsg;
156
157 if (xcf->krb5.keytab.len > 0) {
158 rc = krb5_kt_resolve(xcf->krb5.context,
159 (const char *) xcf->krb5.keytab.data,
160 &xcf->krb5.keytab_obj);
161 } else {
162 rc = krb5_kt_default(xcf->krb5.context, &xcf->krb5.keytab_obj);
163 }
164 if (rc != 0) {
165 kmsg = brix_krb5_error(xcf, rc);
166 BRIX_DIAG_CONF(NGX_LOG_EMERG, cf, 0,
167 "brix: cannot open krb5 keytab \"%V\": %s",
168 "the keytab path is wrong, unreadable by the nginx user, or has "
169 "no key for the configured principal",
170 "point brix_krb5_keytab at the service keytab and grant the "
171 "nginx user read access (chmod 0400, correct owner); verify with "
172 "klist -k",
173 &xcf->krb5.keytab, kmsg ? kmsg : "unknown");
174 brix_krb5_free_error(xcf, kmsg);
175 return NGX_ERROR;
176 }
177
178 return NGX_OK;
179}

181/* ---- Confirm the keytab is readable ----
182 *
183 * WHAT: Probes xcf->krb5.keytab_obj with a start/end sequence traversal to
184 * confirm it can actually be read; returns NGX_OK on success or NGX_ERROR
185 * (after an EMERG log) on failure.
186 *
187 * WHY: krb5_kt_resolve/default succeed lazily without touching the file, so an
188 * explicit start_seq_get is the earliest point a truly unreadable keytab is
189 * detected — still at config time, before any client connects.
190 *
191 * HOW: (1) krb5_kt_start_seq_get; on non-zero rc decode+log at EMERG, free the
192 * message, return NGX_ERROR. (2) On success, release the cursor with
193 * krb5_kt_end_seq_get and return NGX_OK.
194 */
195static ngx_int_t
196brix_krb5_probe_keytab(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
197{
198 krb5_error_code rc;
199 const char *kmsg;
200 krb5_kt_cursor cursor;
201
202 rc = krb5_kt_start_seq_get(xcf->krb5.context, xcf->krb5.keytab_obj,
203 &cursor);
204 if (rc != 0) {
205 kmsg = brix_krb5_error(xcf, rc);
206 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
207 "brix: cannot read krb5 keytab: %s",
208 kmsg ? kmsg : "unknown");
209 brix_krb5_free_error(xcf, kmsg);
210 return NGX_ERROR;
211 }
212 (void) krb5_kt_end_seq_get(xcf->krb5.context, xcf->krb5.keytab_obj,
213 &cursor);
214
215 return NGX_OK;
216}

218/* ---- Log the resolved krb5 auth configuration at NOTICE ----
219 *
220 * WHAT: Emits a single NOTICE line reporting the effective acceptor principal,
221 * keytab name, and ip_check flag. No return value; purely diagnostic.
222 *
223 * WHY: A one-line confirmation of the exact principal/keytab actually resolved
224 * lets operators verify the running configuration matches intent without
225 * guessing at defaults.
226 *
227 * HOW: (1) Resolve the keytab name via krb5_kt_get_name, substituting
228 * "(unknown)" on failure. (2) Unparse the principal object, falling back to
229 * the configured principal string when unparse fails. (3) Log the NOTICE
230 * line. (4) Free the unparsed name if one was produced.
231 */
232static void
233brix_krb5_log_configured(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
234{
235 char kt_name[1024];
236 char *principal = NULL;
237
238 if (krb5_kt_get_name(xcf->krb5.context, xcf->krb5.keytab_obj,
239 kt_name, sizeof(kt_name)) != 0)
240 {
241 ngx_cpystrn((u_char *) kt_name, (u_char *) "(unknown)",
242 sizeof(kt_name));
243 }
244
245 if (krb5_unparse_name(xcf->krb5.context, xcf->krb5.principal_obj,
246 &principal) != 0)
247 {
248 principal = NULL;
249 }
250
251 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
252 "brix: krb5 auth configured - principal=%s "
253 "keytab=%s ip_check=%s",
254 principal != NULL ? principal
255 : (const char *) xcf->krb5.principal.data,
256 kt_name, xcf->krb5.ip_check ? "on" : "off");
257
258 if (principal != NULL) {
259 krb5_free_unparsed_name(xcf->krb5.context, principal);
260 }
261}

264/*
265 *
266 * WHAT: Validate krb5 directives and pre-load the krb5_context, acceptor
267 * principal and keytab into xcf for later use by the auth handler.
268 *
269 * WHY: Surfaces all Kerberos misconfiguration at config time so the server
270 * either starts ready to authenticate or fails to start with a precise
271 * diagnostic — clients never see an avoidable auth failure.
272 *
273 * HOW: No-op (NGX_OK) unless xcf->auth == BRIX_AUTH_KRB5. Otherwise it
274 * requires brix_krb5_principal, inits the context, parses the principal,
275 * resolves/opens the keytab (configured path or default), confirms the
276 * keytab is readable, and logs the resolved principal/keytab/ip_check.
277 * Returns NGX_ERROR on any failure (and EMERG when krb5 was requested in a
278 * build compiled without Kerberos support).
279 */
280ngx_int_t
281brix_configure_krb5_auth(ngx_conf_t *cf,
282 ngx_stream_brix_srv_conf_t *xcf)
283{
284#if (BRIX_HAVE_KRB5)
285 ngx_int_t rc;
286
287 if (xcf->auth != BRIX_AUTH_KRB5) {
288 return NGX_OK;
289 }
290
291 rc = brix_krb5_require_principal(cf, xcf);
292 if (rc != NGX_OK) {
293 return rc;
294 }
295
296 rc = brix_krb5_init_context(cf, xcf);
297 if (rc != NGX_OK) {
298 return rc;
299 }
300
301 rc = brix_krb5_parse_principal(cf, xcf);
302 if (rc != NGX_OK) {
303 return rc;
304 }
305
306 rc = brix_krb5_open_keytab(cf, xcf);
307 if (rc != NGX_OK) {
308 return rc;
309 }
310
311 rc = brix_krb5_probe_keytab(cf, xcf);
312 if (rc != NGX_OK) {
313 return rc;
314 }
315
316 brix_krb5_log_configured(cf, xcf);
317
318 return NGX_OK;
319#else
320 if (xcf->auth == BRIX_AUTH_KRB5) {
321 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
322 "brix_auth krb5 requested, but this build "
323 "was configured without Kerberos 5 support");
324 return NGX_ERROR;
325 }
326
327 return NGX_OK;
328#endif
329}
