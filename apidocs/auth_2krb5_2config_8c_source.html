4/*
5 * config.c — configure-time Kerberos 5 (krb5) auth setup for the stream server
6 *
7 * WHAT: Implements brix_configure_krb5_auth(), invoked from server-conf
8 * post-processing to validate and pre-load the Kerberos state a server
9 * needs before it can accept "krb5" logins: the krb5_context, the parsed
10 * acceptor principal (krb5_principal_obj) and the keytab handle
11 * (krb5_keytab_obj). These are consumed at run time by
12 * brix_handle_krb5_auth() in auth.c.
13 *
14 * WHY: Doing this once at config time (rather than per connection) means
15 * misconfiguration — missing principal, unparseable name, unreadable
16 * keytab — fails the server fast with a clear ngx_conf_log_error at
17 * startup instead of denying every client later. It also avoids repeated
18 * keytab opens on the hot auth path.
19 *
20 * HOW: Returns NGX_OK immediately unless xcf->auth == BRIX_AUTH_KRB5. It
21 * requires brix_krb5_principal, then krb5_init_context, krb5_parse_name,
22 * and resolves the keytab (krb5_kt_resolve when brix_krb5_keytab is set,
23 * else krb5_kt_default). It probes the keytab with a start/end seq_get to
24 * confirm readability and logs the effective principal, keytab name and
25 * ip_check flag at NOTICE. All krb5 code is gated on BRIX_HAVE_KRB5; a
26 * build without it that nonetheless requests krb5 auth fails with EMERG.
27 * Errors are decoded via brix_krb5_error()/brix_krb5_free_error().
28 */
29
30#if (BRIX_HAVE_KRB5)

48/*
49 *
50 * WHAT: Validate krb5 directives and pre-load the krb5_context, acceptor
51 * principal and keytab into xcf for later use by the auth handler.
52 *
53 * WHY: Surfaces all Kerberos misconfiguration at config time so the server
54 * either starts ready to authenticate or fails to start with a precise
55 * diagnostic — clients never see an avoidable auth failure.
56 *
57 * HOW: No-op (NGX_OK) unless xcf->auth == BRIX_AUTH_KRB5. Otherwise it
58 * requires brix_krb5_principal, inits the context, parses the principal,
59 * resolves/opens the keytab (configured path or default), confirms the
60 * keytab is readable, and logs the resolved principal/keytab/ip_check.
61 * Returns NGX_ERROR on any failure (and EMERG when krb5 was requested in a
62 * build compiled without Kerberos support).
63 */
64ngx_int_t
65brix_configure_krb5_auth(ngx_conf_t *cf,
66 ngx_stream_brix_srv_conf_t *xcf)
67{
68#if (BRIX_HAVE_KRB5)
69 krb5_error_code rc;
70 const char *kmsg;
71 krb5_kt_cursor cursor;
72 char kt_name[1024];
73 char *principal = NULL;
74
75 if (xcf->auth != BRIX_AUTH_KRB5) {
76 return NGX_OK;
77 }
78
79 if (xcf->krb5.principal.len == 0) {
80 BRIX_DIAG_CONF(NGX_LOG_EMERG, cf, 0,
81 "brix: krb5 auth is enabled but no service principal is set",
82 "brix_auth krb5 needs the service principal it presents to "
83 "clients, but brix_krb5_principal is missing",
84 "add e.g. brix_krb5_principal \"xrootd/host.example.org@REALM\"; "
85 "it must match a key in the keytab");
86 return NGX_ERROR;
87 }
88
89 rc = krb5_init_context(&xcf->krb5.context);
90 if (rc != 0) {
91 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
92 "brix: krb5_init_context failed (%d)",
93 (int) rc);
94 return NGX_ERROR;
95 }
96
97 rc = krb5_parse_name(xcf->krb5.context,
98 (const char *) xcf->krb5.principal.data,
99 &xcf->krb5.principal_obj);
100 if (rc != 0) {
101 kmsg = brix_krb5_error(xcf, rc);
102 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
103 "brix: cannot parse krb5 principal \"%V\": %s",
104 &xcf->krb5.principal, kmsg ? kmsg : "unknown");
105 brix_krb5_free_error(xcf, kmsg);
106 return NGX_ERROR;
107 }
108
109 if (xcf->krb5.keytab.len > 0) {
110 rc = krb5_kt_resolve(xcf->krb5.context,
111 (const char *) xcf->krb5.keytab.data,
112 &xcf->krb5.keytab_obj);
113 } else {
114 rc = krb5_kt_default(xcf->krb5.context, &xcf->krb5.keytab_obj);
115 }
116 if (rc != 0) {
117 kmsg = brix_krb5_error(xcf, rc);
118 BRIX_DIAG_CONF(NGX_LOG_EMERG, cf, 0,
119 "brix: cannot open krb5 keytab \"%V\": %s",
120 "the keytab path is wrong, unreadable by the nginx user, or has "
121 "no key for the configured principal",
122 "point brix_krb5_keytab at the service keytab and grant the "
123 "nginx user read access (chmod 0400, correct owner); verify with "
124 "klist -k",
125 &xcf->krb5.keytab, kmsg ? kmsg : "unknown");
126 brix_krb5_free_error(xcf, kmsg);
127 return NGX_ERROR;
128 }
129
130 rc = krb5_kt_start_seq_get(xcf->krb5.context, xcf->krb5.keytab_obj,
131 &cursor);
132 if (rc != 0) {
133 kmsg = brix_krb5_error(xcf, rc);
134 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
135 "brix: cannot read krb5 keytab: %s",
136 kmsg ? kmsg : "unknown");
137 brix_krb5_free_error(xcf, kmsg);
138 return NGX_ERROR;
139 }
140 (void) krb5_kt_end_seq_get(xcf->krb5.context, xcf->krb5.keytab_obj,
141 &cursor);
142
143 if (krb5_kt_get_name(xcf->krb5.context, xcf->krb5.keytab_obj,
144 kt_name, sizeof(kt_name)) != 0)
145 {
146 ngx_cpystrn((u_char *) kt_name, (u_char *) "(unknown)",
147 sizeof(kt_name));
148 }
149
150 if (krb5_unparse_name(xcf->krb5.context, xcf->krb5.principal_obj,
151 &principal) != 0)
152 {
153 principal = NULL;
154 }
155
156 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
157 "brix: krb5 auth configured - principal=%s "
158 "keytab=%s ip_check=%s",
159 principal != NULL ? principal
160 : (const char *) xcf->krb5.principal.data,
161 kt_name, xcf->krb5.ip_check ? "on" : "off");
162
163 if (principal != NULL) {
164 krb5_free_unparsed_name(xcf->krb5.context, principal);
165 }
166
167 return NGX_OK;
168#else
169 if (xcf->auth == BRIX_AUTH_KRB5) {
170 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
171 "brix_auth krb5 requested, but this build "
172 "was configured without Kerberos 5 support");
173 return NGX_ERROR;
174 }
175
176 return NGX_OK;
177#endif
178}
