1/*
2 * brixcvmfs_rw.c — the cvmfs-rw union driver: full POSIX write semantics on
3 * top of the read-only CVMFS-brix mount.
4 *
5 * WHAT: a FUSE ops table that unions a local writable upper tree
6 * (<mnt>/.brixwrites/upper, sibling of .brixcache, reached via a dirfd
7 * preserved from before the mount) over the CVMFS lower layer: create/
8 * write/truncate/mkdir/rename land in upper (copy-up on first write),
9 * deletes leave whiteout markers, and /.brixwrites itself is exposed as
10 * a read-write passthrough subtree so --overlay-list/--overlay-reset
11 * work on the live mount.
12 * WHY: jobs need scratch/patch space "inside" the software repo without ever
13 * touching upstream; the upper tree is a plain path-mirroring dir —
14 * inspectable, diffable, rm -rf-resettable.
15 * HOW: every op consults the overlay core (client/lib/fs/overlay.h) first
16 * and falls back to the brixcvmfs_op_* read-only ops for lower cases;
17 * upper always wins. Single-threaded (-s) like the ro driver, so no
18 * locking. Op handlers return 0/-errno.
19 */

48/* FUSE path → .brixwrites-relative when inside the passthrough subtree
49 * ("" = the subtree root), else NULL. */
50static const char *pt_rel(const char *path) {
51 static const size_t n = sizeof(BRIX_OV_DIRNAME) - 1;
52 if (strncmp(path + 1, BRIX_OV_DIRNAME, n) != 0) return NULL;
53 const char *rest = path + 1 + n;
54 if (rest[0] == '\0') return "";
55 if (rest[0] == '/') return rest + 1;
56 return NULL;
57}

90static int merged_exists(const char *path) {

100/* ---- union locate (shared upper/lower/whiteout resolution) --------------- */
101
102/*
103 * WHAT: one location snapshot of a path across both layers.
104 * WHY: open/unlink/rmdir/rename all need the same (upper state, lower dirent)
105 * pair; resolving it once keeps the per-op error ordering identical and
106 * the handlers small.
107 * HOW: `s`/`st` come from the overlay classify; `lower`/`e` from the CVMFS
108 * catalog resolve (1 found / 0 absent-or-not-consulted).
109 */
110typedef struct {
111 struct stat st; /* upper stat, valid when s == BRIX_OV_UPPER */
112 brix_ov_state s; /* upper / masked / none */
113 cvmfs_dirent_t e; /* lower dirent, valid when lower == 1 */
114 int lower; /* 1 found / 0 absent (or masked → not consulted) */
115} ov_loc_t;
116
117/*
118 * WHAT: classify `path` in upper, then resolve lower unless masked.
119 * WHY: mutation ops (unlink, rename source) need the lower dirent even when
120 * upper wins (to decide whether a whiteout is required), but a masked
121 * path returns -ENOENT before the catalog is ever consulted.
122 * HOW: masked short-circuits with lower = 0; catalog errors map to -EIO.
123 */
124static int ov_locate(const char *path, ov_loc_t *l) {
125 int rc = rw_classify(path, &l->st, &l->s);
126 if (rc != 0) return rc;
127 l->lower = 0;
128 if (l->s == BRIX_OV_MASKED) return 0;
129 l->lower = lower_resolve(path, &l->e);
130 return l->lower < 0 ? -EIO : 0;
131}
132
133/*
134 * WHAT: like ov_locate, but the catalog is consulted only when upper has no
135 * say at all (state NONE).
136 * WHY: open and rename-target only care about the merged VIEW — an upper hit
137 * or whiteout already decides the outcome, so lower must stay unresolved
138 * there (a masked path must look absent, never like a lower hit).
139 * HOW: upper/masked leave lower = 0; catalog errors map to -EIO.
140 */
141static int ov_locate_visible(const char *path, ov_loc_t *l) {
142 int rc = rw_classify(path, &l->st, &l->s);
143 if (rc != 0) return rc;
144 l->lower = l->s == BRIX_OV_NONE ? lower_resolve(path, &l->e) : 0;
145 return l->lower < 0 ? -EIO : 0;
146}

172/* ensure the upper parent chain of `path` exists (lower-mirrored modes) */
173static int rw_ensure_parents(const char *path) {
174 const char *rel = ov_rel(path);
175 const char *slash = strrchr(rel, '/');
176 if (slash == NULL) return 0; /* parent is the root */
177 char parent[1024];
178 size_t n = (size_t) (slash - rel);
179 if (n >= sizeof(parent)) return -ENAMETOOLONG;
180 memcpy(parent, rel, n);
181 parent[n] = '\0';
182 return brix_overlay_mkdirs(&g_ov, parent, rw_lower_dir_mode, NULL);
183}

185/* materialise lower file `path` (dirent e) in the upper tree */
186static int rw_copyup(const char *path, const cvmfs_dirent_t *e) {
187 int rc = rw_ensure_parents(path);
188 if (rc != 0) return rc;
189 struct stat lst = { 0 };
190 lst.st_size = (off_t) e->size;
191 lst.st_mode = S_IFREG | (e->mode & 07777);
192 lst.st_mtime = (time_t) e->mtime;
193 return brix_overlay_copyup(&g_ov, ov_rel(path), &lst, rw_read_lower, NULL);
194}

198/* fi->fh: 0 = lower-served (read-only), else upper/passthrough fd + 1 */
199static int fh_store(struct fuse_file_info *fi, int fd) {
200 fi->fh = (uint64_t) fd + 1;
201 return 0;

204/* does this open intend to modify the file's bytes? */
205static int rw_want_write(int oflags) {
206 return (oflags & O_ACCMODE) != O_RDONLY || (oflags & O_TRUNC);
207}
208
209/*
210 * WHAT: open a path the upper layer already owns.
211 * WHY: upper always wins — no lower consult, but O_CREAT|O_EXCL must still
212 * see the existing entry and directories can't be opened for data.
213 * HOW: strips O_CREAT (the file exists) and opens straight in upper.
214 */
215static int rw_open_upper(const char *path, const struct stat *st, int oflags,
216 mode_t mode, struct fuse_file_info *fi) {
217 if (S_ISDIR(st->st_mode)) return -EISDIR;
218 if ((oflags & (O_CREAT | O_EXCL)) == (O_CREAT | O_EXCL)) return -EEXIST;
219 int fd = brix_overlay_open(&g_ov, ov_rel(path), oflags & ~O_CREAT, mode);
220 return fd < 0 ? fd : fh_store(fi, fd);
221}
222
223/*
224 * WHAT: open a path visible in neither layer (masked or truly absent).
225 * WHY: only O_CREAT may conjure it; a fresh upper file over a whiteout must
226 * clear the whiteout or the new file would stay invisible.
227 * HOW: mirror the parent chain into upper, create, clear the whiteout.
228 */
229static int rw_open_create(const char *path, int oflags, mode_t mode,
230 struct fuse_file_info *fi) {
231 if (!(oflags & O_CREAT)) return -ENOENT;
232 int rc = rw_ensure_parents(path);
233 if (rc != 0) return rc;
234 int fd = brix_overlay_open(&g_ov, ov_rel(path), oflags, mode);
235 if (fd < 0) return fd;
236 brix_overlay_whiteout_clear(&g_ov, ov_rel(path));
237 return fh_store(fi, fd);
238}

240/*
241 * WHAT: open a path that only the lower layer has (dirent `e`).
242 * WHY: reads stay lower-served (fh = 0 → ro driver); the first write
243 * triggers copy-up so upper owns the bytes from then on.
244 * HOW: refuse dirs/symlinks for writing, copy-up, then open in upper with
245 * O_CREAT|O_EXCL stripped (copy-up just created the upper file).
246 */
247static int rw_open_lower(const char *path, const cvmfs_dirent_t *e, int oflags,
248 mode_t mode, struct fuse_file_info *fi) {
249 if (lower_is_dir(e)) return -EISDIR;
250 if (!rw_want_write(oflags)) { fi->fh = 0; return brixcvmfs_op_open(path, fi); }
251 if (!lower_is_file(e)) return -EPERM; /* symlink write-through: no */
252 if (oflags & O_EXCL) return -EEXIST;
253
254 int rc = rw_copyup(path, e); /* first write → copy-up */
255 if (rc != 0) return rc;
256 int fd = brix_overlay_open(&g_ov, ov_rel(path), oflags & ~(O_CREAT | O_EXCL), mode);
257 return fd < 0 ? fd : fh_store(fi, fd);
258}

260/*
261 * WHAT: the shared open/create/mknod entry — route to the owning layer.
262 * WHY: every open variant needs the same passthrough/reserved gating and the
263 * same union resolution before the per-layer semantics diverge.
264 * HOW: passthrough subtree → direct openat; else locate in the merged view
265 * and dispatch to the upper / create / lower helper.
266 */
267static int rw_open_common(const char *path, int oflags, mode_t mode,
268 struct fuse_file_info *fi) {
269 const char *pr = pt_rel(path);
270 if (pr != NULL) {
271 int fd = openat(g_writes_fd, pt_at(pr), oflags | O_NOFOLLOW, mode);
272 return fd < 0 ? -errno : fh_store(fi, fd);
273 }
274 if (rw_reserved(path))
275 return rw_want_write(oflags) || (oflags & O_CREAT) ? -EPERM : -ENOENT;
276
277 ov_loc_t l;
278 int rc = ov_locate_visible(path, &l);
279 if (rc != 0) return rc;
280 if (l.s == BRIX_OV_UPPER) return rw_open_upper(path, &l.st, oflags, mode, fi);
281 if (l.lower == 0) return rw_open_create(path, oflags, mode, fi);
282 return rw_open_lower(path, &l.e, oflags, mode, fi);
283}

333static int rw_truncate(const char *path, off_t len, struct fuse_file_info *fi) {
334 if (fi != NULL && fi->fh != 0)
335 return ftruncate((int) fi->fh - 1, len) != 0 ? -errno : 0;
336
337 const char *pr = pt_rel(path);
338 if (pr != NULL) {
339 int fd = openat(g_writes_fd, pt_at(pr), O_WRONLY | O_NOFOLLOW);
340 if (fd < 0) return -errno;
341 int rc = ftruncate(fd, len) != 0 ? -errno : 0;
342 close(fd);
343 return rc;
344 }
345 if (rw_reserved(path)) return -EPERM;
346
347 struct stat st;
348 brix_ov_state s;
349 int rc = rw_classify(path, &st, &s);
350 if (rc != 0) return rc;
351 if (s == BRIX_OV_UPPER) return brix_overlay_truncate(&g_ov, ov_rel(path), len);
352 if (s == BRIX_OV_MASKED) return -ENOENT;
353
354 cvmfs_dirent_t e;
355 int lower = lower_resolve(path, &e);
356 if (lower < 0) return -EIO;
357 if (lower == 0) return -ENOENT;
358 if (lower_is_dir(&e)) return -EISDIR;
359 rc = rw_copyup(path, &e);
360 return rc != 0 ? rc : brix_overlay_truncate(&g_ov, ov_rel(path), len);
361}

379static int pt_readdir(const char *pr, void *buf, fuse_fill_dir_t filler) {
380 int fd = openat(g_writes_fd, pt_at(pr), O_RDONLY | O_DIRECTORY | O_CLOEXEC);
381 if (fd < 0) return -errno;
382 DIR *d = fdopendir(fd); /* owns fd */
383 if (d == NULL) { int e = errno; close(fd); return -e; }
384 filler(buf, ".", NULL, 0, 0);
385 filler(buf, "..", NULL, 0, 0);
386 struct dirent *e;
387 while ((e = readdir(d)) != NULL)
388 if (strcmp(e->d_name, ".") != 0 && strcmp(e->d_name, "..") != 0)
389 filler(buf, e->d_name, NULL, 0, 0);
390 closedir(d);
391 return 0;
392}

408static int rw_readdir(const char *path, void *buf, fuse_fill_dir_t filler,
409 off_t off, struct fuse_file_info *fi, enum fuse_readdir_flags fl) {
410 (void) off; (void) fi; (void) fl;
411 const char *pr = pt_rel(path);
412 if (pr != NULL) return pt_readdir(pr, buf, filler);
413
414 cvmfs_client_refresh(brixcvmfs_client(), brixcvmfs_mono_now());
415
416 struct stat st;
417 brix_ov_state s;
418 int rc = rw_classify(path, &st, &s);
419 if (rc != 0) return rc;
420 if (s == BRIX_OV_MASKED) return -ENOENT;
421 if (s == BRIX_OV_UPPER && !S_ISDIR(st.st_mode)) return -ENOTDIR;
422
423 filler(buf, ".", NULL, 0, 0);
424 filler(buf, "..", NULL, 0, 0);
425 if (strcmp(path, "/") == 0)
426 filler(buf, BRIX_OV_DIRNAME, NULL, 0, 0); /* the passthrough subtree */
427
428 brix_ov_nameset set;
429 int opaque = 0;
430 rc = brix_overlay_read_upper(&g_ov, ov_rel(path), &set, &opaque);
431 if (rc != 0) return rc;
432
433 for (size_t i = 0; ; i++) { /* upper entries win */
434 char flag = 0;
435 const char *name = brix_ov_nameset_at(&set, i, &flag);
436 if (name == NULL) break;
437 if (flag == 'u') filler(buf, name, NULL, 0, 0);
438 }
439
440 if (!opaque) { /* merge the lower listing */
441 rw_lowdir_t c = { buf, filler, &set };
442 cvmfs_catalog_readdir(brixcvmfs_client()->root_catalog,
443 brixcvmfs_cat_path(path), rw_lowdir_emit, &c);
444 }
445 brix_ov_nameset_free(&set);
446 return 0;
447}

451static int rw_mkdir(const char *path, mode_t mode) {
452 const char *pr = pt_rel(path);
453 if (pr != NULL) return mkdirat(g_writes_fd, pr, mode) != 0 ? -errno : 0;
454 if (rw_reserved(path)) return -EPERM;
455 if (merged_exists(path)) return -EEXIST;
456
457 int rc = rw_ensure_parents(path);
458 if (rc != 0) return rc;
459 rc = brix_overlay_mkdir(&g_ov, ov_rel(path), mode);
460 if (rc != 0) return rc;
461 brix_overlay_whiteout_clear(&g_ov, ov_rel(path));
462
463 cvmfs_dirent_t e; /* re-created deleted lower dir */
464 if (lower_resolve(path, &e) == 1 && lower_is_dir(&e))
465 brix_overlay_set_opaque(&g_ov, ov_rel(path));
466 return 0;
467}

469static int rw_unlink(const char *path) {
470 const char *pr = pt_rel(path);
471 if (pr != NULL) return unlinkat(g_writes_fd, pr, 0) != 0 ? -errno : 0;
472 if (rw_reserved(path)) return -ENOENT; /* invisible vocabulary */
473
474 ov_loc_t l;
475 int rc = ov_locate(path, &l);
476 if (rc != 0) return rc;
477 if (l.s == BRIX_OV_MASKED) return -ENOENT;
478
479 if (l.s == BRIX_OV_UPPER) {
480 if (S_ISDIR(l.st.st_mode)) return -EISDIR;
481 rc = brix_overlay_unlink_upper(&g_ov, ov_rel(path));
482 if (rc != 0) return rc;
483 return l.lower == 1 ? brix_overlay_whiteout_set(&g_ov, ov_rel(path)) : 0;
484 }
485 if (l.lower == 0) return -ENOENT;
486 if (lower_is_dir(&l.e)) return -EISDIR;
487 return brix_overlay_whiteout_set(&g_ov, ov_rel(path));
488}

502/*
503 * WHAT: rmdir shape check — is `path` a directory in the merged view?
504 * WHY: rmdir must refuse masked/absent paths and non-directories in either
505 * layer, in exactly this order, before the emptiness scan pays for a
506 * catalog listing.
507 * HOW: classify upper first (its errors win), then resolve lower; fills `l`
508 * and `*lower_dir` for the removal step on success.
509 */
510static int rw_rmdir_shape(const char *path, ov_loc_t *l, int *lower_dir) {
511 int rc = rw_classify(path, &l->st, &l->s);
512 if (rc != 0) return rc;
513 if (l->s == BRIX_OV_MASKED) return -ENOENT;
514 if (l->s == BRIX_OV_UPPER && !S_ISDIR(l->st.st_mode)) return -ENOTDIR;
515
516 l->lower = lower_resolve(path, &l->e);
517 if (l->lower < 0) return -EIO;
518 *lower_dir = l->lower == 1 && lower_is_dir(&l->e);
519 if (l->s == BRIX_OV_NONE && !*lower_dir)
520 return l->lower == 1 ? -ENOTDIR : -ENOENT;
521 return 0;
522}

524/*
525 * WHAT: merged-empty check — does the union view of dir `path` hold anything?
526 * WHY: any real upper entry, or any un-whiteouted lower entry (unless the
527 * dir is opaque), blocks the removal — same rule as kernel overlayfs.
528 * HOW: scan the upper nameset for 'u' entries first; only when that comes up
529 * empty walk the lower listing with the whiteout set as a filter.
530 */
531static int rw_dir_merged_nonempty(const char *path, int lower_dir, int *nonempty) {
532 brix_ov_nameset set;
533 int opaque = 0;
534 int rc = brix_overlay_read_upper(&g_ov, ov_rel(path), &set, &opaque);
535 if (rc != 0) return rc;
536 *nonempty = 0;
537 for (size_t i = 0; !*nonempty; i++) {
538 char flag = 0;
539 const char *name = brix_ov_nameset_at(&set, i, &flag);
540 if (name == NULL) break;
541 if (flag == 'u') *nonempty = 1;
542 }
543 if (!*nonempty && lower_dir && !opaque) {
544 rw_lowcount_t c = { &set, 0 };
545 cvmfs_catalog_readdir(brixcvmfs_client()->root_catalog,
546 brixcvmfs_cat_path(path), rw_lowcount_cb, &c);
547 *nonempty = c.nonempty;
548 }
549 brix_ov_nameset_free(&set);
550 return 0;
551}

553static int rw_rmdir(const char *path) {
554 const char *pr = pt_rel(path);
555 if (pr != NULL) return unlinkat(g_writes_fd, pr, AT_REMOVEDIR) != 0 ? -errno : 0;
556 if (rw_reserved(path)) return -ENOENT;
557
558 ov_loc_t l;
559 int lower_dir = 0;
560 int rc = rw_rmdir_shape(path, &l, &lower_dir);
561 if (rc != 0) return rc;
562
563 int nonempty = 0;
564 rc = rw_dir_merged_nonempty(path, lower_dir, &nonempty);
565 if (rc != 0) return rc;
566 if (nonempty) return -ENOTEMPTY;
567
568 if (l.s == BRIX_OV_UPPER) {
569 rc = brix_overlay_rmdir_upper(&g_ov, ov_rel(path));
570 if (rc != 0) return rc;
571 }
572 return lower_dir ? brix_overlay_whiteout_set(&g_ov, ov_rel(path)) : 0;
573}

575/*
576 * WHAT: rename admission checks shared by every union rename.
577 * WHY: flag/vocabulary/NOREPLACE gating is identical for files and dirs and
578 * must run before either side is resolved.
579 * HOW: pure predicate chain, 0 or -err.
580 */
581static int rw_rename_preflight(const char *from, const char *to, unsigned int flags) {
582 if (flags & ~(unsigned int) RENAME_NOREPLACE) return -EINVAL;
583 if (rw_reserved(from) || rw_reserved(to)) return -EPERM;
584 if ((flags & RENAME_NOREPLACE) && merged_exists(to)) return -EEXIST;
585 return 0;
586}

588/* target sanity: renaming anything onto a visible directory is refused */
589static int rw_rename_target_ok(const char *to) {
590 ov_loc_t t;
591 int rc = ov_locate_visible(to, &t);
592 if (rc != 0) return rc;
593 if (t.s == BRIX_OV_UPPER && S_ISDIR(t.st.st_mode)) return -EISDIR;
594 if (t.lower == 1 && lower_is_dir(&t.e)) return -EISDIR;
595 return 0;
596}

598/*
599 * WHAT: the shared rename tail — move the upper entry `from` → `to`.
600 * WHY: files and pure-upper dirs finish identically: mirror the target's
601 * parent chain, move in upper, unmask the target, and (when the source
602 * still exists in lower) whiteout the old name so it disappears.
603 * HOW: `mask_from` says whether lower still shows the source.
604 */
605static int rw_rename_commit(const char *from, const char *to, int mask_from) {
606 int rc = rw_ensure_parents(to);
607 if (rc != 0) return rc;
608 rc = brix_overlay_rename_upper(&g_ov, ov_rel(from), ov_rel(to));
609 if (rc != 0) return rc;
610 brix_overlay_whiteout_clear(&g_ov, ov_rel(to));
611 return mask_from ? brix_overlay_whiteout_set(&g_ov, ov_rel(from)) : 0;
612}
613
614/*
615 * WHAT: directory rename — same-layer (pure-upper) fast path only.
616 * WHY: only a pure-upper dir moves atomically; anything still merged with
617 * lower is -EXDEV (mv falls back to copy+delete, exactly like kernel
618 * overlayfs without redirect_dir).
619 * HOW: an opaque upper dir over a lower dir counts as pure-upper.
620 */
621static int rw_rename_dir(const char *from, const char *to,
622 int from_upper_dir, int from_lower_dir) {
623 if (!from_upper_dir) return -EXDEV;
624 brix_ov_nameset set;
625 int opaque = 0;
626 int rc = brix_overlay_read_upper(&g_ov, ov_rel(from), &set, &opaque);
627 if (rc != 0) return rc;
628 brix_ov_nameset_free(&set);
629 if (from_lower_dir && !opaque) return -EXDEV;
630 return rw_rename_commit(from, to, from_lower_dir);
631}
632
633/*
634 * WHAT: file rename — cross-layer copy-up path.
635 * WHY: a lower-only file must be materialised in upper before it can move;
636 * lower symlinks never move (no copy-up semantics for them).
637 * HOW: copy-up when lower-only, then the shared commit tail.
638 */
639static int rw_rename_file(const char *from, const char *to, const ov_loc_t *f) {
640 if (f->s == BRIX_OV_NONE) {
641 if (f->lower == 0) return -ENOENT;
642 if (!lower_is_file(&f->e)) return -EPERM; /* lower symlinks don't move */
643 int rc = rw_copyup(from, &f->e);
644 if (rc != 0) return rc;
645 }
646 return rw_rename_commit(from, to, f->lower == 1);
647}

649static int rw_rename(const char *from, const char *to, unsigned int flags) {
650 const char *prf = pt_rel(from), *prt = pt_rel(to);
651 if ((prf != NULL) != (prt != NULL)) return -EXDEV; /* no crossing the seam */
652 if (prf != NULL)
653 return renameat(g_writes_fd, prf, g_writes_fd, prt) != 0 ? -errno : 0;
654
655 int rc = rw_rename_preflight(from, to, flags);
656 if (rc != 0) return rc;
657
658 ov_loc_t f;
659 rc = ov_locate(from, &f);
660 if (rc != 0) return rc;
661 if (f.s == BRIX_OV_MASKED) return -ENOENT;
662
663 rc = rw_rename_target_ok(to);
664 if (rc != 0) return rc;
665
666 int from_lower_dir = f.lower == 1 && lower_is_dir(&f.e);
667 int from_upper_dir = f.s == BRIX_OV_UPPER && S_ISDIR(f.st.st_mode);
668 if (from_upper_dir || from_lower_dir)
669 return rw_rename_dir(from, to, from_upper_dir, from_lower_dir);
670 return rw_rename_file(from, to, &f);
671}

673static int rw_symlink(const char *target, const char *path) {
674 const char *pr = pt_rel(path);
675 if (pr != NULL) return symlinkat(target, g_writes_fd, pr) != 0 ? -errno : 0;
676 if (rw_reserved(path)) return -EPERM;

685static int rw_readlink(const char *path, char *buf, size_t size) {
686 const char *pr = pt_rel(path);
687 if (pr != NULL) {
688 ssize_t n = readlinkat(g_writes_fd, pt_at(pr), buf, size - 1);
689 if (n < 0) return -errno;
690 buf[n] = '\0';
691 return 0;
692 }
693 brix_ov_state s;
694 struct stat st;
695 int rc = rw_classify(path, &st, &s);
696 if (rc != 0) return rc;
697 if (s == BRIX_OV_UPPER) return brix_overlay_readlink(&g_ov, ov_rel(path), buf, size);
698 if (s == BRIX_OV_MASKED) return -ENOENT;

702/* ---- metadata mutations --------------------------------------------------
703 * Shared shape: upper → apply; masked → ENOENT; lower file → copy-up then
704 * apply; lower dir → materialise the upper dir then apply. */
705
706typedef int (*rw_meta_fn)(const char *rel, void *arg);
707
708static int rw_meta_apply(const char *path, rw_meta_fn apply, void *arg) {
709 if (rw_reserved(path)) return -EPERM;
710
711 struct stat st;
712 brix_ov_state s;
713 int rc = rw_classify(path, &st, &s);
714 if (rc != 0) return rc;
715 if (s == BRIX_OV_UPPER) return apply(ov_rel(path), arg);
716 if (s == BRIX_OV_MASKED) return -ENOENT;
717
718 cvmfs_dirent_t e;
719 int lower = lower_resolve(path, &e);
720 if (lower < 0) return -EIO;
721 if (lower == 0) return -ENOENT;
722
723 if (lower_is_dir(&e)) {
724 rc = rw_ensure_parents(path);
725 if (rc == 0) rc = brix_overlay_mkdirs(&g_ov, ov_rel(path), rw_lower_dir_mode, NULL);
726 } else if (lower_is_file(&e)) {
727 rc = rw_copyup(path, &e);
728 } else {
729 return -EOPNOTSUPP; /* lower symlink metadata */
730 }
731 return rc != 0 ? rc : apply(ov_rel(path), arg);
732}

775/* answer for the user.overlay magic attribute, or NULL when not ours */
776static const char *rw_overlay_xattr(const char *path) {
777 struct stat st;
778 brix_ov_state s;
779 if (rw_classify(path, &st, &s) != 0) return NULL;
780 if (s == BRIX_OV_MASKED) return NULL;
781 cvmfs_dirent_t e;

787static int rw_getxattr(const char *path, const char *name, char *value, size_t size) {
788 if (pt_rel(path) != NULL) return -ENODATA;
789 if (strcmp(name, "user.overlay") == 0) {
790 const char *v = rw_overlay_xattr(path);
791 if (v == NULL) return -ENODATA;

799 struct stat st;

801 if (s == BRIX_OV_UPPER) return -ENODATA; /* cvmfs attrs are lower-only */
802 if (s == BRIX_OV_MASKED) return -ENODATA;
803 return brixcvmfs_op_getxattr(path, name, value, size);
804}

845int brixcvmfs_setup_rw(const char *mnt, const char *writes_override) {
846 char dir[600];
847 if (writes_override != NULL && writes_override[0] != '\0') {
848 snprintf(dir, sizeof(dir), "%s", writes_override);
849 } else {
850 mkdir(mnt, 0755); /* ensure the mountpoint exists */
851 snprintf(dir, sizeof(dir), "%s/" BRIX_OV_DIRNAME, mnt);
852 }
853 if (mkdir(dir, 0755) != 0 && errno != EEXIST) {
854 fprintf(stderr, "brixcvmfs: cannot create overlay dir %s\n", dir);
855 return -1;
856 }
857 g_writes_fd = open(dir, O_RDONLY | O_DIRECTORY | O_CLOEXEC);
858 if (g_writes_fd < 0 || brix_overlay_init(&g_ov, g_writes_fd) != 0) {
859 fprintf(stderr, "brixcvmfs: overlay at %s unusable\n", dir);
860 if (g_writes_fd >= 0) { close(g_writes_fd); g_writes_fd = -1; }
861 return -1;
862 }
863 return 0;
864}
