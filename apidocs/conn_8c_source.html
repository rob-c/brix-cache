1/*
2 * conn.c — connection / session bring-up and teardown.
3 *
4 * WHAT: connect → (20B handshake + kXR_protocol, pipelined) → kXR_login (anon) →
5 * ready; and a best-effort kXR_endsess + close on teardown.
6 * WHY: Everything else (stat/ls/get/put) needs a logged-in session; this is the
7 * one place that drives the handshake state machine.
8 * HOW: We send the 20-byte ClientInitHandShake and the 24-byte protocol request
9 * as one 44-byte segment (as modern clients do). The server replies with a
10 * handshake frame (streamid {0,0}) AND a protocol frame (streamid = our
11 * protocol request's id); we tolerate either ordering / a combined reply by
12 * keying on the streamid. Anonymous login sends no credential payload; if
13 * the server demands a security protocol we fail cleanly (auth is M4).
14 *
15 * wire: XProtocol.hh ClientInitHandShake — {0,0,0,htonl(4),htonl(2012=ROOTD_PQ)}.
16 * wire: XProtocol.hh ServerProtocolBody — pval[4] flags[4]; flags carry server caps.
17 * wire: XProtocol.hh ServerLoginBody — sessid[16] [+ "&P=..." security list].
18 */

37/* Well-known IGTF/grid CA trust directory shipped by fetch-crl / ca-policy RPMs.
38 * Grid server certs (dCache, EOS, ...) chain to these CAs, which are NOT in the
39 * OpenSSL system bundle, so falling back to it gives "unable to get local issuer
40 * certificate". */
41#define XRDC_GRID_CA_DIR "/etc/grid-security/certificates"

43/*
44 * Resolve the CA trust directory for TLS peer verification using the same search
45 * order as the stock xrootd/globus tooling, so the client trusts grid CAs out of
46 * the box:
47 * 1. explicit caller value (e.g. --ca-dir) 2. $X509_CERT_DIR
48 * 3. /etc/grid-security/certificates (if present and readable)
49 * 4. NULL ⇒ caller falls back to OpenSSL system defaults
50 * Returns a borrowed string (env/literal/argument); never allocates.
51 */
52const char *
53brix_resolve_ca_dir(const char *opt_ca_dir)
54{
55 const char *env;
56
57 if (opt_ca_dir != NULL && opt_ca_dir[0] != '\0') {
58 return opt_ca_dir;
59 }
60
61 env = getenv("X509_CERT_DIR");
62 if (env != NULL && env[0] != '\0') {
63 return env;
64 }
65
66 if (access(XRDC_GRID_CA_DIR, R_OK | X_OK) == 0) {
67 return XRDC_GRID_CA_DIR;
68 }
69
70 return NULL;
71}

73static void
74fill_username(char out[9])
75{
76 /* The login username is the OS identity advertised for monitoring/default
77 * mapping (real authz comes from the chosen sec protocol). Prefer $LOGNAME/
78 * $USER — set in every interactive/login shell — so the common path skips the
79 * getpwuid() NSS lookup entirely (it lazy-loads libnss_* on first call, a
80 * measurable per-process cost the async manager would otherwise pay once per
81 * parallel stream). Fall back to getpwuid() when the environment is unset. */
82 const char *name = getenv("LOGNAME");
83 struct passwd *pw;
84 size_t n;
85
86 if (name == NULL || name[0] == '\0') { name = getenv("USER"); }
87 if (name == NULL || name[0] == '\0') {
88 pw = getpwuid(geteuid());
89 name = (pw != NULL && pw->pw_name != NULL) ? pw->pw_name : "nobody";
90 }
91 n = strlen(name);
92
93 if (n > 8) { n = 8; } /* the wire field is 8 bytes; truncate */
94 memcpy(out, name, n);
95 out[n] = '\0'; /* NUL-terminated for xrd_pack_login_request */
96}

98/* Read one response frame raw (header + body), bypassing streamid checks; used
99 * for the handshake exchange where the first reply carries streamid {0,0}. */
100static int
101recv_raw(brix_conn *c, uint16_t *sid, uint16_t *status,
102 uint8_t *body, uint32_t bodycap, uint32_t *blen, brix_status *st)
103{
104 uint8_t hdr[XRD_RESPONSE_HDR_LEN];
105 uint32_t dlen;
106
107 if (brix_read_full(&c->io, hdr, sizeof(hdr), st) != 0) {
108 return -1;
109 }
110 xrd_resp_hdr_unpack(hdr, sid, status, &dlen); /* unaligned-safe */
111
112 if (dlen > bodycap) {
113 brix_status_set(st, XRDC_EPROTO, 0,
114 "handshake body too large (%u > %u)", dlen, bodycap);
115 return -1;
116 }
117 if (dlen > 0 && brix_read_full(&c->io, body, dlen, st) != 0) {
118 return -1;
119 }
120 *blen = dlen;
121 if (c->diag.wire_trace) { /* §15: trace handshake/protocol replies too */
122 brix_trace_frame(c, '<', *sid, *status, 0, dlen, body, dlen);
123 }
124 return 0;
125}

127static int
128do_handshake(brix_conn *c, uint16_t proto_sid, int want_tls, brix_status *st)
129{
130 uint8_t seg[XRD_HANDSHAKE_LEN + XRD_REQUEST_HDR_LEN];
131 ClientInitHandShake hs;
132 ClientProtocolRequest pr;
133 uint16_t sid, status;
134 uint8_t body[256];
135 uint32_t blen;
136 int saw_proto = 0;
137 int rounds;
138
139 xrd_pack_handshake(&hs);
140
141 /* The client owns its protocol streamid; ask for the security-requirements
142 * trailer (to learn the signing level) and advertise TLS capability,
143 * requiring TLS for roots:// / --tls. */
144 {
145 const uint8_t sid[2] = { (uint8_t) (proto_sid >> 8),
146 (uint8_t) (proto_sid & 0xff) };
147 uint8_t flags = (uint8_t) (kXR_secreqs | kXR_ableTLS |
148 (want_tls ? kXR_wantTLS : 0));
149 xrd_pack_protocol_request(&pr, sid, flags);
150 }
151
152 memcpy(seg, &hs, XRD_HANDSHAKE_LEN);
153 memcpy(seg + XRD_HANDSHAKE_LEN, &pr, XRD_REQUEST_HDR_LEN);
154
155 if (brix_write_full(&c->io, seg, sizeof(seg), st) != 0) {
156 return -1;
157 }
158 if (c->diag.wire_trace) { /* §15: the 20B init has no streamid/requestid */
159 fprintf(stderr, "> handshake-init (20B) + kXR_protocol sid=%u\n", proto_sid);
160 }
161
162 /* Expect up to two frames: a handshake reply (streamid {0,0}) and the
163 * protocol reply (streamid == proto_sid). Some servers may send only the
164 * protocol reply; key on the streamid rather than assume an ordering. */
165 for (rounds = 0; rounds < 2 && !saw_proto; rounds++) {
166 if (recv_raw(c, &sid, &status, body, sizeof(body), &blen, st) != 0) {
167 return -1;
168 }
169 if (status != kXR_ok) {
170 /* The server completed the framing and EXPLICITLY rejected the
171 * handshake (e.g. kXR_error because we asked for TLS on a non-TLS
172 * port). That is a permanent decision: classify it by the server's
173 * own status code (not the retryable XRDC_EPROTO framing-desync
174 * code) so the resilient loop fails fast instead of re-handshaking
175 * the same rejection until its stall window expires. */
176 brix_status_set(st, (int) status, 0,
177 "handshake: server status %u", status);
178 return -1;
179 }
180 if (sid == proto_sid) {
181 if (blen < sizeof(ServerProtocolBody)) {
182 brix_status_set(st, XRDC_EPROTO, 0,
183 "protocol reply too short (%u bytes)", blen);
184 return -1;
185 }
186 /* ServerProtocolBody = pval[4] flags[4]; capabilities in flags. */
187 c->server_flags = xrd_get_u32_be(body + 4); /* unaligned-safe */
188 /* Optional SecurityInfo trailer (present because we set kXR_secreqs):
189 * 4-byte header {0, hasSec, sec_count, 0}, sec_count*8 protocol
190 * entries, then a 6-byte 'S' section whose byte[4] is the signing
191 * security level. Defensive: default 0 (no signing). */
192 c->sec_level = 0;
193 if (blen >= 12) {
194 uint8_t sec_count = body[10];
195 size_t sp = 12u + (size_t) sec_count * 8u;
196 if (sp + 6u <= blen && body[sp] == 'S') {
197 c->sec_level = body[sp + 4];
198 }
199 }
200 saw_proto = 1;
201 }
202 /* else: handshake reply (streamid {0,0}); keep reading for the protocol. */
203 }
204
205 if (!saw_proto) {
206 brix_status_set(st, XRDC_EPROTO, 0, "no protocol reply from server");
207 return -1;
208 }
209 return 0;
210}

212static int
213do_login(brix_conn *c, const brix_opts *o, brix_status *st)
214{
215 ClientLoginRequest req;
216 uint16_t sid, status;
217 uint8_t *body = NULL;
218 uint32_t blen = 0;
219
220 /* streamid {0,0}: brix_send stamps the real streamid (and dlen) after
221 * packing. The username is the OS identity; advertise async-response
222 * capability. dlen=0 → anonymous (no credential/CGI payload). */
223 {
224 static const uint8_t sid0[2] = { 0, 0 };
225 char uname[9];
226 fill_username(uname);
227 xrd_pack_login_request(&req, sid0, (int32_t) getpid(), uname,
228 (uint8_t) (kXR_ver005 | kXR_asyncap));
229 }
230
231 if (brix_send(c, &req, NULL, 0, &sid, st) != 0) {
232 return -1;
233 }
234 if (brix_recv(c, sid, &status, &body, &blen, st) != 0) {
235 return -1;
236 }
237 if (status != kXR_ok) {
238 brix_status_set(st, XRDC_EPROTO, 0, "login: server status %u", status);
239 free(body);
240 return -1;
241 }
242 if (blen < BRIX_SESSION_ID_LEN) {
243 brix_status_set(st, XRDC_EPROTO, 0,
244 "login reply too short (%u bytes)", blen);
245 free(body);
246 return -1;
247 }
248 memcpy(c->sessid, body, BRIX_SESSION_ID_LEN);
249
250 /* Anything past the 16-byte sessid is a "&P=<proto>,..." security list:
251 * the server demands authentication. Hand off to the auth driver. */
252 if (blen > BRIX_SESSION_ID_LEN) {
253 char sec[256];
254 uint32_t n = blen - BRIX_SESSION_ID_LEN;
255 if (n >= sizeof(sec)) { n = sizeof(sec) - 1; }
256 memcpy(sec, body + BRIX_SESSION_ID_LEN, n);
257 sec[n] = '\0';
258 free(body);
259 snprintf(c->sec_list, sizeof(c->sec_list), "%s", sec); /* §15 explain */
260 return brix_authenticate(c, sec, o, st);
261 }
262
263 free(body);
264 return 0;
265}

267/* Establish the session on the already-set c->host/c->port using c->want_tls +
268 * c->opts: connect → handshake → [TLS] → login → auth. Resets per-connection state
269 * so it is safe to call again from brix_reconnect after a redirect. */
270static int
271brix_bringup_ex(brix_conn *c, int want_login, brix_status *st)
272{
273 c->io.fd = -1;
274 c->io.ssl = NULL;
275 c->ssl_ctx = NULL;
276 c->next_sid = 1;
277 c->server_flags = 0;
278 c->sec_level = 0;
279 c->signing_active = 0;
280 c->sig_seqno = 0;
281
282 /* §15: arm trace/timing from opts (the per-opcode RTT table on c persists
283 * across a redirect's reconnect, so the final summary aggregates all hops). */
284 c->diag.wire_trace = c->opts.wire_trace;
285 c->diag.timing = c->opts.timing;
286 c->diag.redir_trace = c->opts.redir_trace;
287
288 /* Bound the whole bring-up (connect + handshake + TLS + login) with the SHORT
289 * connect timeout, not the long steady-state I/O timeout: a firewall that
290 * completes the TCP handshake then black-holes the protocol bytes must fail
291 * promptly so the reconnect machinery can ride over it, rather than hanging
292 * the caller. The steady-state timeout is restored once the session is up. */
293 c->io.timeout_ms = brix_tmo_connect_ms();
294
295 c->diag.phase_ns[0] = brix_mono_ns(); /* §15.3: connect-phase breakdown */
296 c->io.fd = brix_tcp_connect(c->host, c->port, c->io.timeout_ms, st);
297 if (c->io.fd < 0) {
298 return -1;
299 }
300 c->diag.phase_ns[1] = brix_mono_ns(); /* tcp connected */
301
302 /* Reserve streamid 1 for the protocol request; subsequent ops start at 2. */
303 c->next_sid = 2;
304 if (do_handshake(c, 1, c->want_tls, st) != 0) {
305 brix_close(c);
306 return -1;
307 }
308
309 /* TLS decision — never silently downgrade. */
310 {
311 int have_tls = (c->server_flags & kXR_haveTLS) != 0;
312 int goto_tls = (c->server_flags & kXR_gotoTLS) != 0;
313 if (goto_tls || (c->want_tls && have_tls)) {
314 const char *ca = brix_resolve_ca_dir(c->opts.ca_dir);
315 if (brix_tls_upgrade(c, !c->opts.insecure_tls, c->opts.verify_host, ca, st) != 0) {
316 brix_close(c);
317 return -1;
318 }
319 } else if (c->want_tls && !have_tls) {
320 if (c->tls_strict || !c->opts.notlsok) {
321 brix_status_set(st, XRDC_EAUTH, 0,
322 "server offers no TLS; refusing cleartext "
323 "(use --notlsok with root:// to override)");
324 brix_close(c);
325 return -1;
326 }
327 /* root:// + --notlsok: proceed cleartext. */
328 }
329 }
330 c->diag.phase_ns[2] = brix_mono_ns(); /* tls negotiated (==tcp if cleartext) */
331
332 if (want_login && do_login(c, &c->opts, st) != 0) {
333 brix_close(c);
334 return -1;
335 }
336 c->diag.phase_ns[3] = brix_mono_ns(); /* login + auth done */
337
338 /* Session is up: switch from the short bring-up cap to the steady-state I/O
339 * timeout so a legitimately long read/write is not cut off. */
340 c->io.timeout_ms = brix_tmo_io_ms();
341 return 0;
342}

351int
352brix_bind(brix_conn *sec, const brix_conn *primary, brix_status *st)
353{
354 ClientBindRequest req;
355 uint16_t sid, status;
356 uint8_t *body = NULL;
357 uint32_t blen = 0;
358
359 /* A secondary stream re-runs handshake + kXR_protocol [+ TLS] against the
360 * SAME target but SKIPS kXR_login; the server inherits identity from the
361 * primary's session via kXR_bind{sessid}. (src/protocols/root/session/bind.c) */
362 memset(sec, 0, sizeof(*sec));
363 sec->io.timeout_ms = primary->io.timeout_ms;
364 sec->opts = primary->opts;
365 sec->want_tls = primary->want_tls;
366 sec->tls_strict = primary->tls_strict;
367 snprintf(sec->host, sizeof(sec->host), "%s", primary->host);
368 sec->port = primary->port;
369
370 if (brix_bringup_ex(sec, 0, st) != 0) {
371 return -1;
372 }
373
374 memset(&req, 0, sizeof(req));
375 req.requestid = htons(kXR_bind);
376 {
377 xrdw_sessid_req_t b;
378 memcpy(b.sessid, primary->sessid, BRIX_SESSION_ID_LEN);
379 xrdw_sessid_req_pack(&b, ((ClientRequestHdr *) &req)->body);
380 }
381
382 if (brix_send(sec, &req, NULL, 0, &sid, st) != 0
383 || brix_recv(sec, sid, &status, &body, &blen, st) != 0) {
384 /* Quiet teardown: a bound stream has no session of its own to end. */
385 brix_tls_free(sec);
386 if (sec->io.fd >= 0) { close(sec->io.fd); sec->io.fd = -1; }
387 return -1;
388 }
389 free(body); /* reply body = 1-byte pathid (server bookkeeping) */
390 return 0;
391}

393/* phase-49: arm the libxrdproto crypto (SHA/HMAC) exactly once, lazily, so every
394 * connecting tool gets working GSI/token digests without an explicit
395 * brix_crypto_init() call — removing the easy-to-forget "GSI silently breaks if

397 * any remaining explicit caller is harmless. */

399brix_crypto_init_void(void)

401 (void) brix_crypto_init(); /* int-returning; pthread_once needs void(void) */

411int
412brix_connect(brix_conn *c, const brix_url *u, const brix_opts *o, brix_status *st)
413{
414 brix_crypto_once();
415 memset(c, 0, sizeof(*c));
416 c->io.timeout_ms = brix_tmo_io_ms(); /* steady-state; bring-up uses the short cap */
417
418 if (o != NULL) {
419 c->opts = *o;
420 } else {
421 c->opts.verify_host = 1;
422 }
423
424 if (u->scheme != XRDC_SCHEME_ROOT && u->scheme != XRDC_SCHEME_ROOTS) {
425 brix_status_set(st, XRDC_EUSAGE, 0,
426 "native client speaks root:// / roots:// only (scheme %d)",
427 (int) u->scheme);
428 return -1;
429 }
430
431 snprintf(c->host, sizeof(c->host), "%s", u->host);
432 c->port = u->port;
433 /* Phase 40 (a): remember the ORIGINAL endpoint (the manager) so a dead
434 * redirect target can fall back here for a fresh server selection. Set once
435 * at connect; brix_reconnect deliberately does NOT touch it. */
436 snprintf(c->home_host, sizeof(c->home_host), "%s", u->host);
437 c->home_port = u->port;
438 /* WS-3: carry the single-slash flag forward so per-op error sites can fire
439 * the double-slash hint without re-parsing the original URL. */
440 c->single_slash_path = u->single_slash_path;
441 c->tls_strict = (u->scheme == XRDC_SCHEME_ROOTS);
442 c->want_tls = c->tls_strict || c->opts.want_tls;
443
444 /* §15.1: open the capture sink ONCE (here, not in bringup — so a redirect's
445 * reconnect appends rather than truncates). Frames are recorded by frame.c. */
446 if (c->opts.capture != NULL && c->opts.capture[0] != '\0') {
447 char ep[320];
448 c->diag.cap = brix_capture_open(c->opts.capture);
449 brix_format_host_port(c->host, (uint16_t) c->port, ep, sizeof(ep));
450 brix_capture_meta(c->diag.cap, "endpoint", ep);
451 }
452
453 {
454 int rc = brix_bringup(c, st);
455 if (rc == 0 && c->diag.cap != NULL) { /* snapshot negotiated session */
456 char buf[64], sx[2 * BRIX_SESSION_ID_LEN + 1];
457 int i;
458 snprintf(buf, sizeof(buf), "0x%x", (unsigned) c->server_flags);
459 brix_capture_meta(c->diag.cap, "caps", buf);
460 brix_capture_meta(c->diag.cap, "auth",
461 c->diag.chosen_auth ? c->diag.chosen_auth : "anon");
462 brix_capture_meta(c->diag.cap, "seclist",
463 c->sec_list[0] ? c->sec_list : "(none)");
464 for (i = 0; i < BRIX_SESSION_ID_LEN; i++) {
465 snprintf(sx + i * 2, 3, "%02x", c->sessid[i]);
466 }
467 brix_capture_meta(c->diag.cap, "sessid", sx);
468 }
469 return rc;
470 }
471}

473int
474brix_connect_no_login(brix_conn *c, const brix_url *u, const brix_opts *o,
475 brix_status *st)
476{
477 brix_crypto_once();
478 memset(c, 0, sizeof(*c));
479 c->io.timeout_ms = brix_tmo_io_ms(); /* steady-state; bring-up uses the short cap */
480 if (o != NULL) { c->opts = *o; } else { c->opts.verify_host = 1; }
481
482 if (u->scheme != XRDC_SCHEME_ROOT && u->scheme != XRDC_SCHEME_ROOTS) {
483 brix_status_set(st, XRDC_EUSAGE, 0,
484 "native client speaks root:// / roots:// only");
485 return -1;
486 }
487 snprintf(c->host, sizeof(c->host), "%s", u->host);
488 c->port = u->port;
489 snprintf(c->home_host, sizeof(c->home_host), "%s", u->host);
490 c->home_port = u->port;
491 c->single_slash_path = u->single_slash_path; /* WS-3: propagate for hint */
492 c->tls_strict = (u->scheme == XRDC_SCHEME_ROOTS);
493 /* TLS per the scheme: roots:// negotiates TLS (cert presented); a plain root://
494 * server that does not mandate TLS stays cleartext (then there is simply no cert,
495 * rather than a rejected wantTLS handshake). No kXR_login: cert inspection needs
496 * no identity, so this works even where anon login would be rejected. A server
497 * that sends kXR_gotoTLS still upgrades regardless. */
498 c->want_tls = c->tls_strict;
499 c->opts.notlsok = 1;
500 return brix_bringup_ex(c, 0 /*want_login*/, st);
501}

503int
504brix_reconnect(brix_conn *c, const char *host, int port, brix_status *st)
505{
506 /* Abandon the current transport (no endsess — we are leaving this server) but
507 * keep opts/want_tls/redirect-state, then re-establish against the new target. */
508 brix_tls_free(c);
509 if (c->io.fd >= 0) {
510 close(c->io.fd);
511 c->io.fd = -1;
512 }
513 snprintf(c->host, sizeof(c->host), "%s", host);
514 c->port = port;
515 return brix_bringup(c, st);
516}

518/* 1 if the session id is all-zero — i.e. no session was ever established (login
519 * never completed), so there is nothing to gracefully end. */
520static int
521sessid_is_zero(const uint8_t *s)
522{
523 int i;
524 for (i = 0; i < BRIX_SESSION_ID_LEN; i++) {
525 if (s[i] != 0) {
526 return 0;

532void
533brix_close(brix_conn *c)
534{
535 if (c != NULL && c->diag.timing) { /* §15: one summary per session at exit */
536 brix_timing_report(c);
537 }
538 if (c != NULL && c->diag.cap != NULL) { /* §15.1: flush + close the capture */
539 brix_capture_close(c->diag.cap);
540 c->diag.cap = NULL; /* idempotent: safe on double close */
541 }
542 if (c == NULL || c->io.fd < 0) {
543 return;
544 }
545
546 /* Best-effort graceful end-of-session: kXR_endsess{sessid[16]}, FIRE-AND-
547 * FORGET. We deliberately do NOT wait for the reply: against a black-holing
548 * peer (a misbehaving inline firewall) reading a reply that never comes would
549 * stall teardown for a full timeout, and the server tears the session down on
550 * socket close regardless. Skip it entirely when no session was established
551 * (sessid still zero — e.g. a connection that failed during handshake), and
552 * cap the send so even a wedged socket cannot block the close. */
553 if (!sessid_is_zero(c->sessid)) {
554 uint8_t req[XRD_REQUEST_HDR_LEN];
555 uint16_t sid;
556 brix_status throwaway;
557
558 if (c->io.timeout_ms <= 0 || c->io.timeout_ms > 2000) {
559 c->io.timeout_ms = 2000; /* teardown send must not hang */
560 }
561 memset(req, 0, sizeof(req));
562 req[2] = (uint8_t) (kXR_endsess >> 8);
563 req[3] = (uint8_t) (kXR_endsess & 0xff);

566 memcpy(b.sessid, c->sessid, BRIX_SESSION_ID_LEN);
567 xrdw_sessid_req_pack(&b, req + 4);
568 }

570 (void) brix_send(c, req, NULL, 0, &sid, &throwaway);
571 }
572
573 brix_tls_free(c); /* SSL_shutdown/free + SSL_CTX_free (no-op if none) */
574 close(c->io.fd);
575 c->io.fd = -1;
576 OPENSSL_cleanse(c->signing_key, sizeof(c->signing_key));
577}

579/*
580 * brix_explain_conn — narrate an established session: endpoint, server roles,
581 * negotiated caps, signing, the auth choice, the TLS state, and the session id.
582 *
583 * WHAT: a read-only report over the conn fields conn.c/auth.c already populated.
584 * WHY: both `xrdfs explain` and `xrddiag check` want the same human-readable
585 * picture of what the handshake actually negotiated — single-source it here.
586 * HOW: decode server_flags into roles/caps, defer the per-module auth detail to
587 * brix_auth_explain, and call brix_tls_info for the live cipher (flagging a
588 * gotoTLS→cleartext downgrade as a warning).
589 */
590void
591brix_explain_conn(brix_conn *c, const brix_opts *opts, FILE *out)
592{
593 const char *ver = NULL, *cipher = NULL;
594 unsigned f;
595 int i;
596
597 if (c == NULL || out == NULL) {
598 return;
599 }
600 f = (unsigned) c->server_flags;
601
602 fprintf(out, "Endpoint: %s:%d\n", c->host, c->port);
603
604 fprintf(out, "Roles: %s%s%s%s\n",
605 (f & kXR_isServer) ? "server " : "",
606 (f & kXR_isManager) ? "manager " : "",
607 (f & kXR_attrProxy) ? "proxy " : "",
608 (f & (kXR_isServer | kXR_isManager)) ? "" : "(unknown)");
609 fprintf(out, "Caps: TLS=%s%s pgread/pgwrite=%s POSC=%s\n",
610 (f & kXR_haveTLS) ? "available" : "no",
611 (f & kXR_gotoTLS) ? " (gotoTLS: required)" : "",
612 (f & kXR_suppgrw) ? "yes" : "no",
613 (f & kXR_supposc) ? "yes" : "no");
614 fprintf(out, "Signing: sec_level=%d%s\n", c->sec_level,
615 c->signing_active ? " (kXR_sigver active)" : "");
616
617 fprintf(out, "Auth:\n");
618 brix_auth_explain(c, opts != NULL ? opts : &c->opts, out);
619
620 if (brix_tls_info(c, &ver, &cipher)) {
621 fprintf(out, "TLS: active — %s / %s\n",
622 ver ? ver : "?", cipher ? cipher : "?");
623 } else if (f & kXR_gotoTLS) {
624 fprintf(out, "TLS: INACTIVE — WARNING: server advertised gotoTLS but "
625 "the session is cleartext (downgrade)\n");
626 } else {
627 fprintf(out, "TLS: inactive (cleartext)\n");
628 }
629
630 /* §15.2: introspect whatever credentials are in the environment — shown even
631 * when not the chosen protocol, so "you have a token but the server didn't
632 * offer ztn" and "your proxy is expired" are both visible. Best-effort. */
633 {
634 char *tok = brix_token_discover();
635 const char *proxy = getenv("X509_USER_PROXY");
636 if (tok != NULL || (proxy != NULL && proxy[0] != '\0')) {
637 fprintf(out, "Credentials (in environment):\n");
638 if (tok != NULL) {
639 brix_token_explain(tok, out);
640 free(tok);
641 }
642 if (proxy != NULL && proxy[0] != '\0') {
643 brix_gsi_cert_explain(proxy, out);
644 }
645 }
646 }
647
648 fprintf(out, "Session: ");
649 for (i = 0; i < (int) sizeof(c->sessid); i++) {
650 fprintf(out, "%02x", c->sessid[i]);
651 }
652 fprintf(out, "\n");
653}
