7/*
8 * Multiple flags can be OR'd together. The most common combinations:
9 * read-only: kXR_open_read
10 * create-write: kXR_open_updt | kXR_new
11 * overwrite: kXR_open_updt | kXR_delete
12 * append: kXR_open_apnd
13 * retstat-open: kXR_open_read | kXR_retstat (saves a separate kXR_stat)
14 */

45/* ------------------------------------------------------------------ */
46/* Stat response flags (combined into <flags> in the ASCII stat body) */
47/* ------------------------------------------------------------------ */
48/*
49 * The on-wire stat body is a space-separated ASCII string:
50 * "<id> <size> <flags> <mtime>"
51 *
52 * <flags> is a decimal integer formed by OR-ing the kXR_* values below.
53 * kXR_file (0) is the default; set only the bits that apply.
54 *
55 * FIELD ORDER WARNING: <size> comes BEFORE <flags> in the wire format.
56 * Swapping them causes the client to misread kXR_isDir (2) or kXR_readable
57 * (16) as the file size, producing confusing "file size = 16" bugs.
58 */
59#define kXR_file 0 /* plain regular file (no flag bit) */
60#define kXR_xset 1 /* executable (x bit set) or searchable dir */
61#define kXR_isDir 2 /* path is a directory */
62#define kXR_other 4 /* special file — not a regular file or dir */
63#define kXR_offline 8 /* file data is not currently online/cached */
64#define kXR_readable 16 /* caller has read permission */
65#define kXR_writable 32 /* caller has write permission */
66#define kXR_poscpend 64 /* POSC (persist-on-close) file, not yet closed */
67#define kXR_bkpexist 128 /* a backup copy exists at this site */
68#define kXR_statAttrCache 256 /* per-file stat hint: xattr metadata is cached locally;
69 server can satisfy kXR_fattr without a round-trip.
70 Local extension — not part of upstream protocol spec.
71 Named kXR_statAttrCache to avoid collision with the
72 protocol-level kXR_attrCache (0x80) flag below. */
73#define kXR_cachersp 512 /* file is served from a local read-through cache
74 (xcache node); set when conf->cache is on and
75 the file exists in the local cache_root.
76 XRootD clients use this to distinguish a cache
77 server from a plain data server. */

226/* ------------------------------------------------------------------ */
227/* kXR_pgwrite — paged write with CRC32 integrity */
228/* ------------------------------------------------------------------ */
229/*
230 * Each page fragment in a pgwrite payload starts with a 4-byte big-endian
231 * CRC32c checksum followed by up to XRD_PGWRITE_PAGESZ bytes of data:
232 * [crc32c_be[4]][data[0..4095]] [crc32c_be[4]][data[0..4095]] ...
233 * The first and last fragments may be smaller when the file offset is
234 * unaligned or the write ends mid-page.
235 * Total payload = (full_pages * 4100) + (last_page_size + 4) if partial.
236 */
237#define XRD_PGWRITE_PAGESZ 4096 /* bytes of data per page */
238#define XRD_PGWRITE_CKSZ 4 /* bytes of CRC32c before each page */
239#define XRD_PGWRITE_UNITSZ (XRD_PGWRITE_PAGESZ + XRD_PGWRITE_CKSZ) /* 4100 */
240
241/* ------------------------------------------------------------------ */
242/* kXR_pgread — paged read with CRC32c integrity */
243/* ------------------------------------------------------------------ */
244/*
245 * The server response body interleaves data and checksums as
246 * [data[4096]][crc32c_be[4]] per page fragment. This is intentionally the
247 * inverse order of the client-to-server kXR_pgwrite payload.
248 *
249 * kXR_pgPageSZ — page size in bytes
250 * kXR_pgPageBL — log2(page size) = 12, used to convert byte offsets
251 * to page indices: page_index = byte_offset >> kXR_pgPageBL
252 * kXR_pgUnitSZ — total bytes per page unit (data + checksum)
253 * kXR_pgRetry — reqflags bit: server should retry any failed pages
254 * kXR_AnyPath — pathid value meaning "server chooses the path"
255 */
256#define kXR_pgPageSZ 4096 /* bytes per data page */
257#define kXR_pgPageBL 12 /* bits: log2(4096) */
258#define kXR_pgUnitSZ (kXR_pgPageSZ + 4) /* 4100 bytes */
259#define kXR_pgRetry 0x01 /* retry bad pages */
260#define kXR_AnyPath 0xff /* let server choose */
261
262/* CSE (checksum-error) retransmit caps — mirror stock XProtocol limits.

265 * file (the Fob capacity; over → kXR_TooManyErrs). */
266#define kXR_pgMaxEpr 128 /* max errs per request */
