9/* cms_addr_is_loopback — classify the resolved manager address as on-host
10 * (127.0.0.0/8 or ::1). A loopback manager is reached with no network latency, so
11 * it gets the most aggressive cold-start settle profile. */
12static ngx_uint_t
13cms_addr_is_loopback(struct sockaddr *sa)
14{
15 if (sa == NULL) {
16 return 0;
17 }
18
19 if (sa->sa_family == AF_INET) {
20 struct sockaddr_in *sin = (struct sockaddr_in *) sa;
21 return (ntohl(sin->sin_addr.s_addr) & 0xFF000000u) == 0x7F000000u;
22 }
23
24#if (NGX_HAVE_INET6)
25 if (sa->sa_family == AF_INET6) {
26 struct sockaddr_in6 *sin6 = (struct sockaddr_in6 *) sa;
27 return IN6_IS_ADDR_LOOPBACK(&sin6->sin6_addr) ? 1 : 0;
28 }
29#endif
30
31 return 0;
32}

66/* ngx_brix_cms_schedule_retry — schedule the next reconnect after a failure.
67 *
68 * Two regimes, chosen automatically so every caller (connect.c and recv.c) gets the
69 * right behaviour without having to know which failure path it is on:
70 *
71 * - COLD-START FAST-RETRY: a node that has NEVER yet logged in is racing its
72 * manager's listen socket (most acutely when the whole mesh boots together on
73 * one host). Retry on the short fixed fast-retry interval for a bounded window
74 * so the mesh settles in tens of ms instead of seconds. Quiet (debug) — a brief
75 * race is expected, not WARN-worthy.
76 * - EXPONENTIAL BACKOFF: once the fast-retry window expires, or once this node has
77 * ever logged in (so a failure is a real outage, not a cold start), fall back to
78 * the doubling+jitter backoff. The genuinely-unreachable diagnostic is surfaced
79 * here, sparsely.
80 *
81 * Gating fast-retry on pre-first-login + a bounded window is what guarantees it can
82 * never become a busy-spin (cf. the self-rearming-0ms-timer footgun).
83 */
84
85void
86ngx_brix_cms_schedule_retry(ngx_brix_cms_ctx_t *ctx)
87{
88 ngx_msec_t delay;
89 ngx_msec_t max_backoff;
90 ngx_msec_t now;
91
92 /* Never schedule a reconnect once the worker is draining — a pending retry
93 * timer would otherwise keep the exiting worker alive to the shutdown
94 * timeout for no purpose. */
95 if (ngx_exiting) {
96 return;
97 }
98
99 /* Cold-start fast-retry window. */
100 if (!ctx->ever_logged_in && ctx->fast_window > 0) {
101 now = ngx_current_msec;
102
103 /* Start the window on the first failure of this cold start. */
104 if (ctx->fast_deadline == 0) {
105 ctx->fast_deadline = now + ctx->fast_window;
106 }
107
108 if ((ngx_msec_int_t) (ctx->fast_deadline - now) > 0) {
109 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
110 "brix: CMS fast-retry to %V (attempt %ui)",
111 &ctx->conf->cms.manager, ctx->connect_attempts);
112 ngx_brix_cms_schedule(ctx, ctx->fast_retry);
113 return;
114 }
115
116 /* Window expired: the manager is genuinely unreachable for a same-host
117 * cold start — surface the actionable diagnostic, then fall through to the
118 * sparse exponential backoff below. */
119 BRIX_DIAG_WARN(ctx->cycle->log, 0,
120 "xrootd[cms]: cannot reach cluster manager %V",
121 "the cmsd is down, the address/port is wrong, or a firewall blocks "
122 "the connection",
123 "confirm cmsd is listening and that brix_cms_manager matches its "
124 "host:port; this node stays OUT of the cluster until it connects",
125 &ctx->conf->cms.manager);
126 }
127
128 /* Cap max backoff at 10× the heartbeat interval so a short cms_interval
129 * (e.g. 2s for tests) also gives short reconnect windows. */
130 max_backoff = (ngx_msec_t) ctx->conf->cms.interval * 10000;
131 if (max_backoff > NGX_BRIX_CMS_BACKOFF_MAX) {
132 max_backoff = NGX_BRIX_CMS_BACKOFF_MAX;
133 }
134
135 delay = ctx->backoff;
136 if (ctx->backoff < max_backoff) {
137 ctx->backoff *= 2;
138 if (ctx->backoff > max_backoff) {
139 ctx->backoff = max_backoff;
140 }
141 }
142
143 /*
144 * Phase 39 (WS7): add up to +25% random jitter so many workers/nodes that lost
145 * the manager at the same instant do not reconnect in lockstep (a thundering
146 * herd that would re-overload a recovering manager). ngx_random() is nginx's
147 * PRNG; jitter only ever lengthens the delay, never shortens it.
148 */
149 if (delay > 0) {
150 delay += (ngx_msec_t) (ngx_random() % (delay / 4 + 1));
151 }
152
153 ngx_brix_cms_schedule(ctx, delay);
154}

186static void
187ngx_brix_cms_write_handler(ngx_event_t *ev)
188{
189 ngx_connection_t *c;
190 ngx_brix_cms_ctx_t *ctx;
191
192 c = ev->data;
193 ctx = c->data;
194
195 ngx_log_debug3(NGX_LOG_DEBUG_EVENT, c->write->log, 0,
196 "brix: CMS write handler called timedout=%d "
197 "logged_in=%d c=%p",
198 (int) ev->timedout,
199 ctx ? (int) ctx->logged_in : -1, c);
200
201 if (ev->timedout) {
202 /* A connect/first-write that never completed. Route through the connect
203 * retry policy: pre-first-login it may fast-retry (bounded window), else it
204 * backs off — never the multi-second backoff for a same-host cold start. */
205 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, ev->log, 0,
206 "brix: CMS connect/write timed out");
207 ngx_brix_cms_disconnect(ctx);
208 ngx_brix_cms_schedule_retry(ctx);
209 return;
210 }
211
212 if (c->write->timer_set) {
213 ngx_del_timer(c->write);
214 }
215
216 if (!ctx->logged_in) {
217 if (ngx_brix_cms_send_login(ctx) != NGX_OK) {
218 /* The TCP connect was refused/half-open (on loopback, ECONNREFUSED
219 * surfaces here as a writable-with-error event, not at connect()).
220 * This is exactly the same-host "manager not listening yet" case → the
221 * fast-retry policy, not the multi-second backoff. */
222 ngx_brix_cms_disconnect(ctx);
223 ngx_brix_cms_schedule_retry(ctx);
224 return;
225 }
226
227 ctx->logged_in = 1;
228 ctx->backoff = ngx_min((ngx_msec_t) ctx->conf->cms.interval * 1000,
229 (ngx_msec_t) NGX_BRIX_CMS_BACKOFF_INITIAL);
230
231 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
232 "brix: CMS login sent to %V",
233 &ctx->conf->cms.manager);
234
235 /* First successful login of this boot: leave fast-retry mode (any future
236 * reconnect is a real outage → backoff) and report the settle time. */
237 if (!ctx->ever_logged_in) {
238 ctx->ever_logged_in = 1;
239 ctx->fast_deadline = 0;
240 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
241 "brix: CMS registered with %V after %uL ms "
242 "(%ui connect attempt(s), %s)",
243 &ctx->conf->cms.manager,
244 (brix_phase_now_ns() - ctx->start_ns) / 1000000ull,
245 ctx->connect_attempts,
246 ctx->is_loopback ? "loopback" : "remote");
247 }
248
249 /*
250 * Announce traffic state (Resume|noStage) immediately after login so a
251 * real cmsd manager marks this disk-only node active and eligible for
252 * selection; without it the manager keeps us suspended and never
253 * redirects clients here.
254 */
255 if (ngx_brix_cms_send_status(ctx) != NGX_OK) {
256 ngx_brix_cms_disconnect(ctx);
257 ngx_brix_cms_schedule_retry(ctx);
258 return;
259 }
260
261 /*
262 * WS1: arm the manager-silence deadline ONCE, at the login transition.
263 * It is deliberately NOT re-armed on our own heartbeat sends (that would
264 * let our outbound traffic mask a manager that has gone silent); recv.c
265 * resets it only when a frame actually arrives FROM the manager. So it
266 * measures time since the last manager activity and, on expiry, the recv
267 * handler reconnects us to a healthy manager.
268 */
269 ngx_brix_cms_arm_read_deadline(ctx);
270 }
271
272 if (ngx_brix_cms_send_load(ctx) != NGX_OK) {
273 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
274 "brix: CMS write handler: send_load failed");
275 ngx_brix_cms_disconnect(ctx);
276 ngx_brix_cms_schedule_retry(ctx);
277 return;
278 }
279
280 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
281 ngx_brix_cms_disconnect(ctx);
282 ngx_brix_cms_schedule_retry(ctx);
283 return;
284 }
285
286 ngx_brix_cms_schedule(ctx, (ngx_msec_t) ctx->conf->cms.interval * 1000);
287
288 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, ev->log, 0,
289 "brix: CMS write handler: heartbeat sent");
290}

292/* ngx_brix_cms_connect — start a TCP connect to the CMS manager via
293 * ngx_event_connect_peer, install the read/write handlers, and arm the connect
294 * timeout (NGX_BRIX_CMS_CONNECT_TIMEOUT = 5s); called from the timer when
295 * ctx->connection == NULL. */
296
297static void
298ngx_brix_cms_connect(ngx_brix_cms_ctx_t *ctx)
299{
300 ngx_int_t rc;
301 ngx_connection_t *c;
302
303 ctx->connect_attempts++;
304
305 ngx_memzero(&ctx->peer, sizeof(ctx->peer));
306 ctx->peer.sockaddr = ctx->conf->cms.addr->sockaddr;
307 ctx->peer.socklen = ctx->conf->cms.addr->socklen;
308 ctx->peer.name = &ctx->conf->cms.addr->name;
309 ctx->peer.get = ngx_event_get_peer;
310 ctx->peer.log = ctx->cycle->log;
311 ctx->peer.log_error = NGX_ERROR_ERR;
312
313 rc = ngx_event_connect_peer(&ctx->peer);
314 if (rc == NGX_ERROR || rc == NGX_DECLINED || ctx->peer.connection == NULL) {
315 BRIX_DIAG_WARN(ctx->cycle->log, 0,
316 "xrootd[cms]: cannot reach cluster manager %V",
317 "the cmsd is down, the address/port is wrong, or a firewall "
318 "blocks the connection",
319 "confirm cmsd is listening and that brix_cms_manager matches "
320 "its host:port; this node will keep retrying and stays OUT of the "
321 "cluster until it connects",
322 &ctx->conf->cms.manager);
323 ngx_brix_cms_schedule_retry(ctx);
324 return;
325 }
326
327 c = ctx->peer.connection;
328 ctx->connection = c;
329 ctx->logged_in = 0;
330 ctx->in_pos = 0;
331 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
332
333 c->data = ctx;
334 c->read->handler = ngx_brix_cms_read_handler;
335 c->write->handler = ngx_brix_cms_write_handler;
336
337 /*
338 * WS5: OS-level dead-peer reaping on the manager socket, so a silently-
339 * dropped manager is torn down by the kernel even between event-loop
340 * deadlines. Best-effort; failures are non-fatal.
341 */
342 brix_apply_tcp_deadpeer_opts(c->fd, ctx->conf->cms.tcp_keepalive,
343 ctx->conf->cms.tcp_user_timeout);
344
345 if (rc == NGX_AGAIN) {
346 /*
347 * WS2: bound the connect + first-write readiness window with
348 * cms_send_timeout (operator-tunable); the write handler's ev->timedout
349 * path reconnects with backoff. Falls back to the fixed connect timeout
350 * if the knob is disabled (0).
351 */
352 ngx_msec_t connect_tmo = ctx->conf->cms.send_timeout > 0
353 ? ctx->conf->cms.send_timeout
354 : NGX_BRIX_CMS_CONNECT_TIMEOUT;
355 ngx_add_timer(c->write, connect_tmo);
356 return;
357 }
358
359 ngx_brix_cms_write_handler(c->write);
360}

362/* ngx_brix_cms_timer — the CMS timer handler: when connected, send a load
363 * heartbeat every cms_interval seconds (disconnect + backoff on failure); when
364 * ctx->connection == NULL, trigger a reconnect. */
365
366static void
367ngx_brix_cms_timer(ngx_event_t *ev)
368{
369 ngx_brix_cms_ctx_t *ctx;
370
371 ctx = ev->data;
372
373 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ev->log, 0,
374 "brix: CMS timer fired connection=%p logged_in=%d",
375 ctx->connection,
376 ctx->connection ? (int) ctx->logged_in : -1);
377
378 /* Worker shutting down: drop the manager link immediately (so the manager
379 * sees us leave at once) and do not reschedule the heartbeat. */
380 if (ngx_exiting) {
381 ngx_brix_cms_disconnect(ctx);
382 return;
383 }
384
385 if (ctx->connection == NULL) {
386 ngx_brix_cms_connect(ctx);
387 return;
388 }
389
390 if (ngx_brix_cms_send_load(ctx) != NGX_OK) {
391 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
392 "brix: CMS load heartbeat failed");
393 ngx_brix_cms_disconnect(ctx);
394 ngx_brix_cms_schedule_retry(ctx);
395 return;
396 }
397
398 ngx_brix_cms_schedule(ctx, (ngx_msec_t) ctx->conf->cms.interval * 1000);
399}

401/* ngx_brix_cms_start — worker-init entry point (from config/process.c): allocate
402 * the CMS context, seed the reconnect backoff from cms_interval (capped at
403 * NGX_BRIX_CMS_BACKOFF_INITIAL), and schedule the first connect after
404 * NGX_BRIX_CMS_INITIAL_DELAY (1s). Each worker keeps its own connection. */
405
406void
407ngx_brix_cms_start(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *conf)
408{
409 ngx_brix_cms_ctx_t *ctx;
410
411 if (conf->cms.addr == NULL || conf->cms.ctx != NULL) {
412 return;
413 }
414
415 ctx = ngx_pcalloc(cycle->pool, sizeof(ngx_brix_cms_ctx_t));
416 if (ctx == NULL) {
417 ngx_log_error(NGX_LOG_ERR, cycle->log, 0,
418 "brix: CMS heartbeat allocation failed");
419 return;
420 }
421
422 ctx->cycle = cycle;
423 ctx->conf = conf;
424 ctx->backoff = ngx_min((ngx_msec_t) conf->cms.interval * 1000,
425 (ngx_msec_t) NGX_BRIX_CMS_BACKOFF_INITIAL);
426 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
427 ctx->start_ns = brix_phase_now_ns();
428
429 /*
430 * Resolve the cold-start settle profile once. A loopback manager (same host)
431 * is reached with no network latency, so it gets the most aggressive defaults;
432 * an explicit directive overrides the locality default. The fast-retry interval
433 * is floored so a misconfigured "0" can never become a busy connect-storm.
434 */
435 ctx->is_loopback = cms_addr_is_loopback(conf->cms.addr->sockaddr);
436
437 ctx->fast_retry = (conf->cms.connect_retry != NGX_CONF_UNSET_MSEC)
438 ? conf->cms.connect_retry
439 : (ctx->is_loopback ? NGX_BRIX_CMS_FASTRETRY_LOOPBACK
440 : NGX_BRIX_CMS_FASTRETRY_REMOTE);
441 if (ctx->fast_retry < NGX_BRIX_CMS_FASTRETRY_FLOOR) {
442 ctx->fast_retry = NGX_BRIX_CMS_FASTRETRY_FLOOR;
443 }
444 ctx->fast_window = ctx->is_loopback ? NGX_BRIX_CMS_FASTWIN_LOOPBACK
445 : NGX_BRIX_CMS_FASTWIN_REMOTE;
446
447 ctx->timer.handler = ngx_brix_cms_timer;
448 ctx->timer.data = ctx;
449 ctx->timer.log = cycle->log;
450 /*
451 * The heartbeat re-arms itself every cms_interval (connect.c:194,298). Mark
452 * it cancelable so a draining worker (ngx_exiting after `nginx -s reload`)
453 * exits as soon as its connections close, instead of being held alive to the
454 * shutdown timeout by the next pending heartbeat. The cluster keepalive is
455 * best-effort; dropping a final tick at teardown is harmless (the reconnect
456 * path is already ngx_exiting-guarded). Matches the io_uring panic timer.
457 */
458 ctx->timer.cancelable = 1;
459
460 conf->cms.ctx = ctx;
461
462 {
463 /* First-connect delay: directive override, else the locality default
464 * (0 for loopback — connect immediately; the fast-retry handles a miss). */
465 ngx_msec_t init_delay = (conf->cms.initial_delay != NGX_CONF_UNSET_MSEC)
466 ? conf->cms.initial_delay
467 : (ctx->is_loopback ? NGX_BRIX_CMS_INITDELAY_LOOPBACK
468 : NGX_BRIX_CMS_INITDELAY_REMOTE);
469 ngx_brix_cms_schedule(ctx, init_delay);
470 }
471
472 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
473 "brix: CMS heartbeat starting for manager %V",
474 &conf->cms.manager);
475}
