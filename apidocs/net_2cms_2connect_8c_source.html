10/* cms_addr_is_loopback — classify the resolved manager address as on-host
11 * (127.0.0.0/8 or ::1). A loopback manager is reached with no network latency, so
12 * it gets the most aggressive cold-start settle profile. */
13static ngx_uint_t
14cms_addr_is_loopback(struct sockaddr *sa)
15{
16 if (sa == NULL) {
17 return 0;
18 }
19
20 if (sa->sa_family == AF_INET) {
21 struct sockaddr_in *sin = (struct sockaddr_in *) sa;
22 return (ntohl(sin->sin_addr.s_addr) & 0xFF000000u) == 0x7F000000u;
23 }
24
25#if (NGX_HAVE_INET6)
26 if (sa->sa_family == AF_INET6) {
27 struct sockaddr_in6 *sin6 = (struct sockaddr_in6 *) sa;
28 return IN6_IS_ADDR_LOOPBACK(&sin6->sin6_addr) ? 1 : 0;
29 }
30#endif
31
32 return 0;
33}

48{

160/* ngx_brix_cms_schedule_retry — schedule the next reconnect after a failure.
161 *
162 * Two regimes, chosen automatically so every caller (connect.c and recv.c) gets the
163 * right behaviour without having to know which failure path it is on:
164 *
165 * - COLD-START FAST-RETRY: a node that has NEVER yet logged in is racing its
166 * manager's listen socket (most acutely when the whole mesh boots together on
167 * one host). Retry on the short fixed fast-retry interval for a bounded window
168 * so the mesh settles in tens of ms instead of seconds. Quiet (debug) — a brief
169 * race is expected, not WARN-worthy.
170 * - EXPONENTIAL BACKOFF: once the fast-retry window expires, or once this node has
171 * ever logged in (so a failure is a real outage, not a cold start), fall back to
172 * the doubling+jitter backoff. The genuinely-unreachable diagnostic is surfaced
173 * here, sparsely.
174 *
175 * Gating fast-retry on pre-first-login + a bounded window is what guarantees it can
176 * never become a busy-spin (cf. the self-rearming-0ms-timer footgun).
177 */
178
179void
180ngx_brix_cms_schedule_retry(ngx_brix_cms_ctx_t *ctx)
181{
182 ngx_msec_t delay;
183 ngx_msec_t max_backoff;
184 ngx_msec_t now;
185
186 /* Never schedule a reconnect once the worker is draining — a pending retry
187 * timer would otherwise keep the exiting worker alive to the shutdown
188 * timeout for no purpose. */
189 if (ngx_exiting) {
190 return;
191 }
192
193 /* Cold-start fast-retry window. */
194 if (!ctx->ever_logged_in && ctx->fast_window > 0) {
195 now = ngx_current_msec;
196
197 /* Start the window on the first failure of this cold start. */
198 if (ctx->fast_deadline == 0) {
199 ctx->fast_deadline = now + ctx->fast_window;
200 }
201
202 if ((ngx_msec_int_t) (ctx->fast_deadline - now) > 0) {
203 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
204 "brix: CMS fast-retry to %V (attempt %ui)",
205 &ctx->conf->cms.manager, ctx->connect_attempts);
206 ngx_brix_cms_schedule(ctx, ctx->fast_retry);
207 return;
208 }
209
210 /* Window expired: the manager is genuinely unreachable for a same-host
211 * cold start — surface the actionable diagnostic, then fall through to the
212 * sparse exponential backoff below. */
213 BRIX_DIAG_WARN(ctx->cycle->log, 0,
214 "xrootd[cms]: cannot reach cluster manager %V",
215 "the cmsd is down, the address/port is wrong, or a firewall blocks "
216 "the connection",
217 "confirm cmsd is listening and that brix_cms_manager matches its "
218 "host:port; this node stays OUT of the cluster until it connects",
219 &ctx->conf->cms.manager);
220 }
221
222 /* Cap max backoff at 10× the heartbeat interval so a short cms_interval
223 * (e.g. 2s for tests) also gives short reconnect windows. */
224 max_backoff = (ngx_msec_t) ctx->conf->cms.interval * 10000;
225 if (max_backoff > NGX_BRIX_CMS_BACKOFF_MAX) {
226 max_backoff = NGX_BRIX_CMS_BACKOFF_MAX;
227 }
228
229 delay = ctx->backoff;
230 if (ctx->backoff < max_backoff) {
231 ctx->backoff *= 2;
232 if (ctx->backoff > max_backoff) {
233 ctx->backoff = max_backoff;
234 }
235 }
236
237 /*
238 * Phase 39 (WS7): add up to +25% random jitter so many workers/nodes that lost
239 * the manager at the same instant do not reconnect in lockstep (a thundering
240 * herd that would re-overload a recovering manager). ngx_random() is nginx's
241 * PRNG; jitter only ever lengthens the delay, never shortens it.
242 */
243 if (delay > 0) {
244 delay += (ngx_msec_t) (ngx_random() % (delay / 4 + 1));
245 }
246
247 ngx_brix_cms_schedule(ctx, delay);
248}

285static void
286ngx_brix_cms_write_handler(ngx_event_t *ev)
287{
288 ngx_connection_t *c;
289 ngx_brix_cms_ctx_t *ctx;
290
291 c = ev->data;
292 ctx = c->data;
293
294 ngx_log_debug3(NGX_LOG_DEBUG_EVENT, c->write->log, 0,
295 "brix: CMS write handler called timedout=%d "
296 "logged_in=%d c=%p",
297 (int) ev->timedout,
298 ctx ? (int) ctx->logged_in : -1, c);
299
300 if (ev->timedout) {
301 /* A connect/first-write that never completed. Route through the connect
302 * retry policy: pre-first-login it may fast-retry (bounded window), else it
303 * backs off — never the multi-second backoff for a same-host cold start. */
304 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, ev->log, 0,
305 "brix: CMS connect/write timed out");
306 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_TIMEOUT);
307 ngx_brix_cms_disconnect(ctx);
308 ngx_brix_cms_schedule_retry(ctx);
309 return;
310 }
311
312 if (c->write->timer_set) {
313 ngx_del_timer(c->write);
314 }
315
316 if (!ctx->logged_in) {
317 if (ngx_brix_cms_send_login(ctx) != NGX_OK) {
318 /* The TCP connect was refused/half-open (on loopback, ECONNREFUSED
319 * surfaces here as a writable-with-error event, not at connect()).
320 * This is exactly the same-host "manager not listening yet" case → the
321 * fast-retry policy, not the multi-second backoff. */
322 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
323 ngx_brix_cms_disconnect(ctx);
324 ngx_brix_cms_schedule_retry(ctx);
325 return;
326 }
327
328 ctx->logged_in = 1;
329 ctx->backoff = ngx_min((ngx_msec_t) ctx->conf->cms.interval * 1000,
330 (ngx_msec_t) NGX_BRIX_CMS_BACKOFF_INITIAL);
331
332 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
333 "brix: CMS login sent to %V",
334 &ctx->conf->cms.manager);
335
336 /* First successful login of this boot: leave fast-retry mode (any future
337 * reconnect is a real outage → backoff) and report the settle time. */
338 if (!ctx->ever_logged_in) {
339 ctx->ever_logged_in = 1;
340 ctx->fast_deadline = 0;
341 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
342 "brix: CMS registered with %V after %uL ms "
343 "(%ui connect attempt(s), %s)",
344 &ctx->conf->cms.manager,
345 (brix_phase_now_ns() - ctx->start_ns) / 1000000ull,
346 ctx->connect_attempts,
347 ctx->is_loopback ? "loopback" : "remote");
348 }
349
350 /*
351 * Announce traffic state (Resume|noStage) immediately after login so a
352 * real cmsd manager marks this disk-only node active and eligible for
353 * selection; without it the manager keeps us suspended and never
354 * redirects clients here.
355 */
356 if (ngx_brix_cms_send_status(ctx) != NGX_OK) {
357 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
358 ngx_brix_cms_disconnect(ctx);
359 ngx_brix_cms_schedule_retry(ctx);
360 return;
361 }
362 ngx_brix_cms_log_registration(ctx);
363
364 /*
365 * WS1: arm the manager-silence deadline ONCE, at the login transition.
366 * It is deliberately NOT re-armed on our own heartbeat sends (that would
367 * let our outbound traffic mask a manager that has gone silent); recv.c
368 * resets it only when a frame actually arrives FROM the manager. So it
369 * measures time since the last manager activity and, on expiry, the recv
370 * handler reconnects us to a healthy manager.
371 */
372 ngx_brix_cms_arm_read_deadline(ctx);
373 }
374
375 if (ngx_brix_cms_send_load(ctx) != NGX_OK) {
376 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
377 "brix: CMS write handler: send_load failed");
378 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
379 ngx_brix_cms_disconnect(ctx);
380 ngx_brix_cms_schedule_retry(ctx);
381 return;
382 }
383
384 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
385 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
386 ngx_brix_cms_disconnect(ctx);
387 ngx_brix_cms_schedule_retry(ctx);
388 return;
389 }
390
391 ngx_brix_cms_schedule(ctx, (ngx_msec_t) ctx->conf->cms.interval * 1000);
392
393 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, ev->log, 0,
394 "brix: CMS write handler: heartbeat sent");
395}

397/* ngx_brix_cms_connect — start a TCP connect to the CMS manager via
398 * ngx_event_connect_peer, install the read/write handlers, and arm the connect
399 * timeout (NGX_BRIX_CMS_CONNECT_TIMEOUT = 5s); called from the timer when
400 * ctx->connection == NULL. */
401
402static void
403ngx_brix_cms_connect(ngx_brix_cms_ctx_t *ctx)
404{
405 ngx_int_t rc;
406 ngx_connection_t *c;
407
408 ctx->connect_attempts++;
409
410 ngx_memzero(&ctx->peer, sizeof(ctx->peer));
411 ctx->peer.sockaddr = ctx->conf->cms.addr->sockaddr;
412 ctx->peer.socklen = ctx->conf->cms.addr->socklen;
413 ctx->peer.name = &ctx->conf->cms.addr->name;
414 ctx->peer.get = ngx_event_get_peer;
415 ctx->peer.log = ctx->cycle->log;
416 ctx->peer.log_error = NGX_ERROR_ERR;
417
418 rc = ngx_event_connect_peer(&ctx->peer);
419 if (rc == NGX_ERROR || rc == NGX_DECLINED || ctx->peer.connection == NULL) {
420 BRIX_DIAG_WARN(ctx->cycle->log, 0,
421 "xrootd[cms]: cannot reach cluster manager %V",
422 "the cmsd is down, the address/port is wrong, or a firewall "
423 "blocks the connection",
424 "confirm cmsd is listening and that brix_cms_manager matches "
425 "its host:port; this node will keep retrying and stays OUT of the "
426 "cluster until it connects",
427 &ctx->conf->cms.manager);
428 ngx_brix_cms_schedule_retry(ctx);
429 return;
430 }
431
432 c = ctx->peer.connection;
433 ctx->connection = c;
434 ctx->logged_in = 0;
435 ctx->in_pos = 0;
436 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
437
438 c->data = ctx;
439 c->read->handler = ngx_brix_cms_read_handler;
440 c->write->handler = ngx_brix_cms_write_handler;
441 ngx_brix_cms_sess_begin(ctx);
442
443 /*
444 * WS5: OS-level dead-peer reaping on the manager socket, so a silently-
445 * dropped manager is torn down by the kernel even between event-loop
446 * deadlines. Best-effort; failures are non-fatal.
447 */
448 brix_apply_tcp_deadpeer_opts(c->fd, ctx->conf->cms.tcp_keepalive,
449 ctx->conf->cms.tcp_user_timeout);
450
451 if (rc == NGX_AGAIN) {
452 /*
453 * WS2: bound the connect + first-write readiness window with
454 * cms_send_timeout (operator-tunable); the write handler's ev->timedout
455 * path reconnects with backoff. Falls back to the fixed connect timeout
456 * if the knob is disabled (0).
457 */
458 ngx_msec_t connect_tmo = ctx->conf->cms.send_timeout > 0
459 ? ctx->conf->cms.send_timeout
460 : NGX_BRIX_CMS_CONNECT_TIMEOUT;
461 ngx_add_timer(c->write, connect_tmo);
462 return;
463 }
464
465 ngx_brix_cms_write_handler(c->write);
466}

468/* ngx_brix_cms_timer — the CMS timer handler: when connected, send a load
469 * heartbeat every cms_interval seconds (disconnect + backoff on failure); when
470 * ctx->connection == NULL, trigger a reconnect. */
471
472static void
473ngx_brix_cms_timer(ngx_event_t *ev)
474{
475 ngx_brix_cms_ctx_t *ctx;
476
477 ctx = ev->data;
478
479 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ev->log, 0,
480 "brix: CMS timer fired connection=%p logged_in=%d",
481 ctx->connection,
482 ctx->connection ? (int) ctx->logged_in : -1);
483
484 /* Worker shutting down: drop the manager link immediately (so the manager
485 * sees us leave at once) and do not reschedule the heartbeat. */
486 if (ngx_exiting) {
487 ngx_brix_cms_disconnect(ctx);
488 return;
489 }
490
491 if (ctx->connection == NULL) {
492 ngx_brix_cms_connect(ctx);
493 return;
494 }
495
496 if (ngx_brix_cms_send_load(ctx) != NGX_OK) {
497 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
498 "brix: CMS load heartbeat failed");
499 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
500 ngx_brix_cms_disconnect(ctx);
501 ngx_brix_cms_schedule_retry(ctx);
502 return;
503 }
504
505 ngx_brix_cms_schedule(ctx, (ngx_msec_t) ctx->conf->cms.interval * 1000);
506}

508/* ngx_brix_cms_start — worker-init entry point (from config/process.c): allocate
509 * the CMS context, seed the reconnect backoff from cms_interval (capped at
510 * NGX_BRIX_CMS_BACKOFF_INITIAL), and schedule the first connect after
511 * NGX_BRIX_CMS_INITIAL_DELAY (1s). Each worker keeps its own connection. */
512
513void
514ngx_brix_cms_start(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *conf)
515{
516 ngx_brix_cms_ctx_t *ctx;
517
518 if (conf->cms.addr == NULL || conf->cms.ctx != NULL) {
519 return;
520 }
521
522 ctx = ngx_pcalloc(cycle->pool, sizeof(ngx_brix_cms_ctx_t));
523 if (ctx == NULL) {
524 ngx_log_error(NGX_LOG_ERR, cycle->log, 0,
525 "brix: CMS heartbeat allocation failed");
526 return;
527 }
528
529 ctx->cycle = cycle;
530 ctx->conf = conf;
531 ctx->backoff = ngx_min((ngx_msec_t) conf->cms.interval * 1000,
532 (ngx_msec_t) NGX_BRIX_CMS_BACKOFF_INITIAL);
533 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
534 ctx->start_ns = brix_phase_now_ns();
535
536 /*
537 * Resolve the cold-start settle profile once. A loopback manager (same host)
538 * is reached with no network latency, so it gets the most aggressive defaults;
539 * an explicit directive overrides the locality default. The fast-retry interval
540 * is floored so a misconfigured "0" can never become a busy connect-storm.
541 */
542 ctx->is_loopback = cms_addr_is_loopback(conf->cms.addr->sockaddr);
543
544 ctx->fast_retry = (conf->cms.connect_retry != NGX_CONF_UNSET_MSEC)
545 ? conf->cms.connect_retry
546 : (ctx->is_loopback ? NGX_BRIX_CMS_FASTRETRY_LOOPBACK
547 : NGX_BRIX_CMS_FASTRETRY_REMOTE);
548 if (ctx->fast_retry < NGX_BRIX_CMS_FASTRETRY_FLOOR) {
549 ctx->fast_retry = NGX_BRIX_CMS_FASTRETRY_FLOOR;
550 }
551 ctx->fast_window = ctx->is_loopback ? NGX_BRIX_CMS_FASTWIN_LOOPBACK
552 : NGX_BRIX_CMS_FASTWIN_REMOTE;
553
554 ctx->timer.handler = ngx_brix_cms_timer;
555 ctx->timer.data = ctx;
556 ctx->timer.log = cycle->log;
557 /*
558 * The heartbeat re-arms itself every cms_interval (connect.c:194,298). Mark
559 * it cancelable so a draining worker (ngx_exiting after `nginx -s reload`)
560 * exits as soon as its connections close, instead of being held alive to the
561 * shutdown timeout by the next pending heartbeat. The cluster keepalive is
562 * best-effort; dropping a final tick at teardown is harmless (the reconnect
563 * path is already ngx_exiting-guarded). Matches the io_uring panic timer.
564 */
565 ctx->timer.cancelable = 1;
566
567 conf->cms.ctx = ctx;
568
569 {
570 /* First-connect delay: directive override, else the locality default
571 * (0 for loopback — connect immediately; the fast-retry handles a miss). */
572 ngx_msec_t init_delay = (conf->cms.initial_delay != NGX_CONF_UNSET_MSEC)
573 ? conf->cms.initial_delay
574 : (ctx->is_loopback ? NGX_BRIX_CMS_INITDELAY_LOOPBACK
575 : NGX_BRIX_CMS_INITDELAY_REMOTE);
576 ngx_brix_cms_schedule(ctx, init_delay);
577 }
578
579 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
580 "brix: CMS heartbeat starting for manager %V",
581 &conf->cms.manager);
582}
