11/* cms_addr_is_loopback — classify the resolved manager address as on-host
12 * (127.0.0.0/8 or ::1). A loopback manager is reached with no network latency, so
13 * it gets the most aggressive cold-start settle profile. */
14static ngx_uint_t
15cms_addr_is_loopback(struct sockaddr *sa)
16{
17 if (sa == NULL) {
18 return 0;
19 }
20
21 if (sa->sa_family == AF_INET) {
22 struct sockaddr_in *sin = (struct sockaddr_in *) sa;
23 return (ntohl(sin->sin_addr.s_addr) & 0xFF000000u) == 0x7F000000u;
24 }
25
26#if (NGX_HAVE_INET6)
27 if (sa->sa_family == AF_INET6) {
28 struct sockaddr_in6 *sin6 = (struct sockaddr_in6 *) sa;
29 return IN6_IS_ADDR_LOOPBACK(&sin6->sin6_addr) ? 1 : 0;
30 }
31#endif
32
33 return 0;
34}

47static brix_sess_end_t
48ngx_brix_cms_end_reason(ngx_brix_cms_ctx_t *ctx, ngx_connection_t *c)
49{
50 if (ngx_exiting || ngx_terminate) {

52 }

55 return ctx->sess_end_hint;
56 }

161/* ngx_brix_cms_schedule_retry — schedule the next reconnect after a failure.
162 *
163 * Two regimes, chosen automatically so every caller (connect.c and recv.c) gets the
164 * right behaviour without having to know which failure path it is on:
165 *
166 * - COLD-START FAST-RETRY: a node that has NEVER yet logged in is racing its
167 * manager's listen socket (most acutely when the whole mesh boots together on
168 * one host). Retry on the short fixed fast-retry interval for a bounded window
169 * so the mesh settles in tens of ms instead of seconds. Quiet (debug) — a brief
170 * race is expected, not WARN-worthy.
171 * - EXPONENTIAL BACKOFF: once the fast-retry window expires, or once this node has
172 * ever logged in (so a failure is a real outage, not a cold start), fall back to
173 * the doubling+jitter backoff. The genuinely-unreachable diagnostic is surfaced
174 * here, sparsely.
175 *
176 * Gating fast-retry on pre-first-login + a bounded window is what guarantees it can
177 * never become a busy-spin (cf. the self-rearming-0ms-timer footgun).
178 */
179
180void
181ngx_brix_cms_schedule_retry(ngx_brix_cms_ctx_t *ctx)
182{
183 ngx_msec_t delay;
184 ngx_msec_t max_backoff;
185 ngx_msec_t now;
186
187 /* Never schedule a reconnect once the worker is draining — a pending retry
188 * timer would otherwise keep the exiting worker alive to the shutdown
189 * timeout for no purpose. */
190 if (ngx_exiting) {
191 return;
192 }
193
194 /* Cold-start fast-retry window. */
195 if (!ctx->ever_logged_in && ctx->fast_window > 0) {
196 now = ngx_current_msec;
197
198 /* Start the window on the first failure of this cold start. */
199 if (ctx->fast_deadline == 0) {
200 ctx->fast_deadline = now + ctx->fast_window;
201 }
202
203 if ((ngx_msec_int_t) (ctx->fast_deadline - now) > 0) {
204 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ctx->cycle->log, 0,
205 "brix: CMS fast-retry to %V (attempt %ui)",
206 &ctx->conf->cms.manager, ctx->connect_attempts);
207 ngx_brix_cms_schedule(ctx, ctx->fast_retry);
208 return;
209 }
210
211 /* Window expired: the manager is genuinely unreachable for a same-host
212 * cold start — surface the actionable diagnostic, then fall through to the
213 * sparse exponential backoff below. */
214 BRIX_DIAG_WARN(ctx->cycle->log, 0,
215 "xrootd[cms]: cannot reach cluster manager %V",
216 "the cmsd is down, the address/port is wrong, or a firewall blocks "
217 "the connection",
218 "confirm cmsd is listening and that brix_cms_manager matches its "
219 "host:port; this node stays OUT of the cluster until it connects",
220 &ctx->conf->cms.manager);
221 }
222
223 /* Cap max backoff at 10× the heartbeat interval so a short cms_interval
224 * (e.g. 2s for tests) also gives short reconnect windows. */
225 max_backoff = (ngx_msec_t) ctx->conf->cms.interval * 10000;
226 if (max_backoff > NGX_BRIX_CMS_BACKOFF_MAX) {
227 max_backoff = NGX_BRIX_CMS_BACKOFF_MAX;
228 }
229
230 delay = ctx->backoff;
231 if (ctx->backoff < max_backoff) {
232 ctx->backoff *= 2;
233 if (ctx->backoff > max_backoff) {
234 ctx->backoff = max_backoff;
235 }
236 }
237
238 /*
239 * Phase 39 (WS7): add up to +25% random jitter so many workers/nodes that lost
240 * the manager at the same instant do not reconnect in lockstep (a thundering
241 * herd that would re-overload a recovering manager). ngx_random() is nginx's
242 * PRNG; jitter only ever lengthens the delay, never shortens it.
243 */
244 if (delay > 0) {
245 delay += (ngx_msec_t) (ngx_random() % (delay / 4 + 1));
246 }
247
248 ngx_brix_cms_schedule(ctx, delay);
249}

251/* ngx_brix_cms_disconnect — close the TCP connection, drop its timers, NULL
252 * ctx->connection, clear logged_in, and reset inbuf for the next reconnect; called
253 * on I/O errors or timeouts. */
254
255void
256ngx_brix_cms_disconnect(ngx_brix_cms_ctx_t *ctx)
257{
258 ngx_connection_t *c;
259
260 c = ctx->connection;
261 if (c == NULL) {
262 return;
263 }
264
265 if (c->read->timer_set) {
266 ngx_del_timer(c->read);
267 }
268
269 if (c->write->timer_set) {
270 ngx_del_timer(c->write);
271 }
272
273 brix_sess_end(ctx->sess, ngx_brix_cms_end_reason(ctx, c));
274 ctx->sess = NULL;
275 ctx->sess_attempt_logged = 0;
276 ctx->sess_end_hint_set = 0;
277
278 ngx_close_connection(c);
279
280 ctx->connection = NULL;
281 ctx->logged_in = 0;
282 ctx->in_pos = 0;
283 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
284}

286static void
287ngx_brix_cms_write_handler(ngx_event_t *ev)
288{
289 ngx_connection_t *c;
290 ngx_brix_cms_ctx_t *ctx;
291
292 c = ev->data;
293 ctx = c->data;
294
295 ngx_log_debug3(NGX_LOG_DEBUG_EVENT, c->write->log, 0,
296 "brix: CMS write handler called timedout=%d "
297 "logged_in=%d c=%p",
298 (int) ev->timedout,
299 ctx ? (int) ctx->logged_in : -1, c);
300
301 if (ev->timedout) {
302 /* A connect/first-write that never completed. Route through the connect
303 * retry policy: pre-first-login it may fast-retry (bounded window), else it
304 * backs off — never the multi-second backoff for a same-host cold start. */
305 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, ev->log, 0,
306 "brix: CMS connect/write timed out");
307 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_TIMEOUT);
308 ngx_brix_cms_disconnect(ctx);
309 ngx_brix_cms_schedule_retry(ctx);
310 return;
311 }
312
313 if (c->write->timer_set) {
314 ngx_del_timer(c->write);
315 }
316
317 if (!ctx->logged_in) {
318 if (ngx_brix_cms_send_login(ctx) != NGX_OK) {
319 /* The TCP connect was refused/half-open (on loopback, ECONNREFUSED
320 * surfaces here as a writable-with-error event, not at connect()).
321 * This is exactly the same-host "manager not listening yet" case → the
322 * fast-retry policy, not the multi-second backoff. */
323 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
324 ngx_brix_cms_disconnect(ctx);
325 ngx_brix_cms_schedule_retry(ctx);
326 return;
327 }
328
329 ctx->logged_in = 1;
330 ctx->backoff = ngx_min((ngx_msec_t) ctx->conf->cms.interval * 1000,
331 (ngx_msec_t) NGX_BRIX_CMS_BACKOFF_INITIAL);
332
333 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
334 "brix: CMS login sent to %V",
335 &ctx->conf->cms.manager);
336 brix_cms_log_action(ev->log, "login",
337 (const char *) ctx->conf->cms.manager.data,
338 "out", NULL, 1,
339 ctx->conf->manager_mode
340 ? "sub-manager registering up (Manager bit)"
341 : "leaf/client registering up");
342
343 /* First successful login of this boot: leave fast-retry mode (any future
344 * reconnect is a real outage → backoff) and report the settle time. */
345 if (!ctx->ever_logged_in) {
346 ctx->ever_logged_in = 1;
347 ctx->fast_deadline = 0;
348 ngx_log_error(NGX_LOG_NOTICE, ev->log, 0,
349 "brix: CMS registered with %V after %uL ms "
350 "(%ui connect attempt(s), %s)",
351 &ctx->conf->cms.manager,
352 (brix_phase_now_ns() - ctx->start_ns) / 1000000ull,
353 ctx->connect_attempts,
354 ctx->is_loopback ? "loopback" : "remote");
355 }
356
357 /*
358 * Announce traffic state (Resume|noStage) immediately after login so a
359 * real cmsd manager marks this disk-only node active and eligible for
360 * selection; without it the manager keeps us suspended and never
361 * redirects clients here.
362 */
363 if (ngx_brix_cms_send_status(ctx) != NGX_OK) {
364 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
365 ngx_brix_cms_disconnect(ctx);
366 ngx_brix_cms_schedule_retry(ctx);
367 return;
368 }
369 ngx_brix_cms_log_registration(ctx);
370
371 /*
372 * WS1: arm the manager-silence deadline ONCE, at the login transition.
373 * It is deliberately NOT re-armed on our own heartbeat sends (that would
374 * let our outbound traffic mask a manager that has gone silent); recv.c
375 * resets it only when a frame actually arrives FROM the manager. So it
376 * measures time since the last manager activity and, on expiry, the recv
377 * handler reconnects us to a healthy manager.
378 */
379 ngx_brix_cms_arm_read_deadline(ctx);
380 }
381
382 if (ngx_brix_cms_send_load(ctx) != NGX_OK) {
383 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
384 "brix: CMS write handler: send_load failed");
385 brix_cms_log_action(ev->log, "load",
386 (const char *) ctx->conf->cms.manager.data,
387 "out", NULL, 0, "send_load failed");
388 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
389 ngx_brix_cms_disconnect(ctx);
390 ngx_brix_cms_schedule_retry(ctx);
391 return;
392 }
393
394 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
395 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
396 ngx_brix_cms_disconnect(ctx);
397 ngx_brix_cms_schedule_retry(ctx);
398 return;
399 }
400
401 ngx_brix_cms_schedule(ctx, (ngx_msec_t) ctx->conf->cms.interval * 1000);
402
403 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, ev->log, 0,
404 "brix: CMS write handler: heartbeat sent");
405}

407/* ngx_brix_cms_connect — start a TCP connect to the CMS manager via
408 * ngx_event_connect_peer, install the read/write handlers, and arm the connect
409 * timeout (NGX_BRIX_CMS_CONNECT_TIMEOUT = 5s); called from the timer when
410 * ctx->connection == NULL. */
411
412static void
413ngx_brix_cms_connect(ngx_brix_cms_ctx_t *ctx)
414{
415 ngx_int_t rc;
416 ngx_connection_t *c;
417
418 ctx->connect_attempts++;
419
420 ngx_memzero(&ctx->peer, sizeof(ctx->peer));
421 ctx->peer.sockaddr = ctx->conf->cms.addr->sockaddr;
422 ctx->peer.socklen = ctx->conf->cms.addr->socklen;
423 ctx->peer.name = &ctx->conf->cms.addr->name;
424 ctx->peer.get = ngx_event_get_peer;
425 ctx->peer.log = ctx->cycle->log;
426 ctx->peer.log_error = NGX_ERROR_ERR;
427
428 rc = ngx_event_connect_peer(&ctx->peer);
429 if (rc == NGX_ERROR || rc == NGX_DECLINED || ctx->peer.connection == NULL) {
430 BRIX_DIAG_WARN(ctx->cycle->log, 0,
431 "xrootd[cms]: cannot reach cluster manager %V",
432 "the cmsd is down, the address/port is wrong, or a firewall "
433 "blocks the connection",
434 "confirm cmsd is listening and that brix_cms_manager matches "
435 "its host:port; this node will keep retrying and stays OUT of the "
436 "cluster until it connects",
437 &ctx->conf->cms.manager);
438 ngx_brix_cms_schedule_retry(ctx);
439 return;
440 }
441
442 c = ctx->peer.connection;
443 ctx->connection = c;
444 ctx->logged_in = 0;
445 ctx->in_pos = 0;
446 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
447
448 c->data = ctx;
449 c->read->handler = ngx_brix_cms_read_handler;
450 c->write->handler = ngx_brix_cms_write_handler;
451 ngx_brix_cms_sess_begin(ctx);
452
453 /*
454 * WS5: OS-level dead-peer reaping on the manager socket, so a silently-
455 * dropped manager is torn down by the kernel even between event-loop
456 * deadlines. Best-effort; failures are non-fatal.
457 */
458 brix_apply_tcp_deadpeer_opts(c->fd, ctx->conf->cms.tcp_keepalive,
459 ctx->conf->cms.tcp_user_timeout);
460
461 if (rc == NGX_AGAIN) {
462 /*
463 * WS2: bound the connect + first-write readiness window with
464 * cms_send_timeout (operator-tunable); the write handler's ev->timedout
465 * path reconnects with backoff. Falls back to the fixed connect timeout
466 * if the knob is disabled (0).
467 */
468 ngx_msec_t connect_tmo = ctx->conf->cms.send_timeout > 0
469 ? ctx->conf->cms.send_timeout
470 : NGX_BRIX_CMS_CONNECT_TIMEOUT;
471 ngx_add_timer(c->write, connect_tmo);
472 return;
473 }
474
475 ngx_brix_cms_write_handler(c->write);
476}

478/* ngx_brix_cms_timer — the CMS timer handler: when connected, send a load
479 * heartbeat every cms_interval seconds (disconnect + backoff on failure); when
480 * ctx->connection == NULL, trigger a reconnect. */
481
482static void
483ngx_brix_cms_timer(ngx_event_t *ev)
484{
485 ngx_brix_cms_ctx_t *ctx;
486
487 ctx = ev->data;
488
489 ngx_log_debug2(NGX_LOG_DEBUG_EVENT, ev->log, 0,
490 "brix: CMS timer fired connection=%p logged_in=%d",
491 ctx->connection,
492 ctx->connection ? (int) ctx->logged_in : -1);
493
494 /* Worker shutting down: drop the manager link immediately (so the manager
495 * sees us leave at once) and do not reschedule the heartbeat. */
496 if (ngx_exiting) {
497 ngx_brix_cms_disconnect(ctx);
498 return;
499 }
500
501 if (ctx->connection == NULL) {
502 ngx_brix_cms_connect(ctx);
503 return;
504 }
505
506 if (ngx_brix_cms_send_load(ctx) != NGX_OK) {
507 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
508 "brix: CMS load heartbeat failed");
509 brix_cms_log_action(ev->log, "load",
510 (const char *) ctx->conf->cms.manager.data,
511 "out", NULL, 0, "heartbeat send failed");
512 ngx_brix_cms_set_end_hint(ctx, BRIX_SESS_END_ERROR);
513 ngx_brix_cms_disconnect(ctx);
514 ngx_brix_cms_schedule_retry(ctx);
515 return;
516 }
517
518 brix_cms_log_action(ev->log, "load",
519 (const char *) ctx->conf->cms.manager.data,
520 "out", NULL, 1,
521 ctx->conf->manager_mode ? "aggregate space heartbeat"
522 : "free-space heartbeat");
523
524 ngx_brix_cms_schedule(ctx, (ngx_msec_t) ctx->conf->cms.interval * 1000);
525}

527/* ngx_brix_cms_start — worker-init entry point (from config/process.c): allocate
528 * the CMS context, seed the reconnect backoff from cms_interval (capped at
529 * NGX_BRIX_CMS_BACKOFF_INITIAL), and schedule the first connect after
530 * NGX_BRIX_CMS_INITIAL_DELAY (1s). Each worker keeps its own connection. */
531
532void
533ngx_brix_cms_start(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *conf)
534{
535 ngx_brix_cms_ctx_t *ctx;
536
537 if (conf->cms.addr == NULL || conf->cms.ctx != NULL) {
538 return;
539 }
540
541 ctx = ngx_pcalloc(cycle->pool, sizeof(ngx_brix_cms_ctx_t));
542 if (ctx == NULL) {
543 ngx_log_error(NGX_LOG_ERR, cycle->log, 0,
544 "brix: CMS heartbeat allocation failed");
545 return;
546 }
547
548 ctx->cycle = cycle;
549 ctx->conf = conf;
550 ctx->backoff = ngx_min((ngx_msec_t) conf->cms.interval * 1000,
551 (ngx_msec_t) NGX_BRIX_CMS_BACKOFF_INITIAL);
552 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
553 ctx->start_ns = brix_phase_now_ns();
554
555 /*
556 * Resolve the cold-start settle profile once. A loopback manager (same host)
557 * is reached with no network latency, so it gets the most aggressive defaults;
558 * an explicit directive overrides the locality default. The fast-retry interval
559 * is floored so a misconfigured "0" can never become a busy connect-storm.
560 */
561 ctx->is_loopback = cms_addr_is_loopback(conf->cms.addr->sockaddr);
562
563 ctx->fast_retry = (conf->cms.connect_retry != NGX_CONF_UNSET_MSEC)
564 ? conf->cms.connect_retry
565 : (ctx->is_loopback ? NGX_BRIX_CMS_FASTRETRY_LOOPBACK
566 : NGX_BRIX_CMS_FASTRETRY_REMOTE);
567 if (ctx->fast_retry < NGX_BRIX_CMS_FASTRETRY_FLOOR) {
568 ctx->fast_retry = NGX_BRIX_CMS_FASTRETRY_FLOOR;
569 }
570 ctx->fast_window = ctx->is_loopback ? NGX_BRIX_CMS_FASTWIN_LOOPBACK
571 : NGX_BRIX_CMS_FASTWIN_REMOTE;
572
573 ctx->timer.handler = ngx_brix_cms_timer;
574 ctx->timer.data = ctx;
575 ctx->timer.log = cycle->log;
576 /*
577 * The heartbeat re-arms itself every cms_interval (connect.c:194,298). Mark
578 * it cancelable so a draining worker (ngx_exiting after `nginx -s reload`)
579 * exits as soon as its connections close, instead of being held alive to the
580 * shutdown timeout by the next pending heartbeat. The cluster keepalive is
581 * best-effort; dropping a final tick at teardown is harmless (the reconnect
582 * path is already ngx_exiting-guarded). Matches the io_uring panic timer.
583 */
584 ctx->timer.cancelable = 1;
585
586 conf->cms.ctx = ctx;
587
588 {
589 /* First-connect delay: directive override, else the locality default
590 * (0 for loopback — connect immediately; the fast-retry handles a miss). */
591 ngx_msec_t init_delay = (conf->cms.initial_delay != NGX_CONF_UNSET_MSEC)
592 ? conf->cms.initial_delay
593 : (ctx->is_loopback ? NGX_BRIX_CMS_INITDELAY_LOOPBACK
594 : NGX_BRIX_CMS_INITDELAY_REMOTE);
595 ngx_brix_cms_schedule(ctx, init_delay);
596 }
597
598 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
599 "brix: CMS heartbeat starting for manager %V",
600 &conf->cms.manager);
601}

603/*
604 * CMS auto-role derivation.
605 *
606 * The role a node plays in the cmsd cluster is derived — not hand-configured —
607 * from two facts already present on the merged server conf:
608 *
609 * manager_capable = manager_mode (set by `brix_manager_mode on`, or
610 * auto-derived from `brix_cms_server on`; see
611 * server_module.c: a block that accepts downstream nodes
612 * IS a manager).
613 * has_upstream = a `brix_cms_manager host:port` resolved into cms.addr,
614 * i.e. this node registers UP to a parent manager.
615 *
616 * manager_capable & !has_upstream -> MANAGER (top of the tree)
617 * manager_capable & has_upstream -> SUB-MANAGER (accepts downstream AND
618 * registers up + aggregates)
619 * !manager_capable & has_upstream -> CLIENT (leaf data server / relay)
620 * !manager_capable & !has_upstream -> NONE (not a cmsd node)
621 */
622typedef enum {
623 BRIX_CMS_ROLE_NONE = 0,
624 BRIX_CMS_ROLE_CLIENT,
625 BRIX_CMS_ROLE_MANAGER,
626 BRIX_CMS_ROLE_SUBMANAGER
627} brix_cms_role_t;
628
629static brix_cms_role_t
630brix_cms_role_derive(ngx_flag_t manager_capable, ngx_flag_t has_upstream)
631{
632 if (manager_capable) {
633 return has_upstream ? BRIX_CMS_ROLE_SUBMANAGER : BRIX_CMS_ROLE_MANAGER;
634 }
635 return has_upstream ? BRIX_CMS_ROLE_CLIENT : BRIX_CMS_ROLE_NONE;
636}
637
638static const char *
639brix_cms_role_name(brix_cms_role_t role)
640{
641 switch (role) {
642 case BRIX_CMS_ROLE_MANAGER: return "manager";
643 case BRIX_CMS_ROLE_SUBMANAGER: return "sub-manager";
644 case BRIX_CMS_ROLE_CLIENT: return "client";
645 default: return "none";
646 }
647}
648
649/*
650 * brix_cms_role_worker_init — per-worker CMS bring-up + role proof for ONE
651 * server block. Called for every stream server (process.c), including a
652 * dedicated cms-only manager/sub-manager block whose main-module data path is
653 * disabled (common.enable off) — that block still owns a cmsd role and, when it
654 * has an upstream, an outbound client to start.
655 *
656 * Two jobs:
657 * 1. PROOF-IN-LOGS: emit exactly one NOTICE naming this node's derived role
658 * (manager / sub-manager / client) and how it was derived. Worker 0 only,
659 * because every worker shares this config and N identical lines help no one.
660 * 2. SINGLE UPSTREAM CONNECTION: start the outbound CMS client on WORKER 0
661 * ONLY. A stock cmsd admits one connection per node identity (SID =
662 * host:listen_port, identical across workers), so N per-worker connections
663 * collide as "already logged in" and get 30s-blacklisted. Gating the
664 * client to a single worker makes brix register cleanly with a STOCK
665 * upstream cmsd and removes the self-collision entirely. (Tradeoff: the
666 * §7.1 pending-locate bridge, which needs the client on the worker holding
667 * the suspended session, is unavailable in this single-connection mode —
668 * aggregation, which is what a sub-manager needs, works over one link.)
669 */
670void
671brix_cms_role_worker_init(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *xcf)
672{
673 brix_cms_role_t role;
674 ngx_flag_t has_upstream;
675 ngx_flag_t manager_capable;
676
677 has_upstream = (xcf->cms.addr != NULL);
678 manager_capable = (xcf->manager_mode == 1);
679 role = brix_cms_role_derive(manager_capable, has_upstream);
680
681 if (role == BRIX_CMS_ROLE_NONE) {
682 return;
683 }
684
685 if (ngx_worker == 0) {
686 if (has_upstream) {
687 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
688 "brix: cmsd role: this node is a %s "
689 "(listen :%d, upstream_manager=%V, "
690 "accepts_downstream=%s, aggregates_up=%s)",
691 brix_cms_role_name(role), (int) xcf->listen_port,
692 &xcf->cms.manager,
693 manager_capable ? "yes" : "no",
694 role == BRIX_CMS_ROLE_SUBMANAGER ? "yes" : "no");
695 } else {
696 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
697 "brix: cmsd role: this node is a %s "
698 "(listen :%d, upstream_manager=none, "
699 "accepts_downstream=%s, aggregates_up=no)",
700 brix_cms_role_name(role), (int) xcf->listen_port,
701 manager_capable ? "yes" : "no");
702 }
703 }
704
705 /* Single outbound connection: worker 0 only (see the note above). */
706 if (has_upstream && ngx_worker == 0) {
707 ngx_brix_cms_start(cycle, xcf);
708 }
709}
