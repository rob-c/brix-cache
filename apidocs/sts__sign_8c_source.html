1/*
2 * sts_sign.c — SigV4 request construction for the S3 STS exchange (phase-79).
3 *
4 * WHAT: build the canonical (sorted, percent-encoded) STS query string and
5 * SigV4-sign it with the node's backend *service* credentials over the
6 * "sts" service, producing the exact query AWS/MinIO recomputes and checks.
7 * WHY: split out of sts.c (formerly 815 lines) as the pure request-building
8 * concern — no I/O, no allocation — so each STS file owns one job. The
9 * byte output is unchanged from the pre-split code: same canonical layout,
10 * same signing-key derivation, same signature.
11 * HOW: sts_build_action_qs emits Action=AssumeRole (+RoleArn/RoleSessionName)
12 * or Action=GetSessionToken in SigV4 canonical byte order; sts_sign_query
13 * forms the canonical request + string-to-sign, derives the signing key
14 * with brix_sigv4_signing_key(), and appends "&X-Amz-Signature=<hex>".
15 *
16 * The reused building blocks:
17 * - SigV4 signing key: brix_sigv4_signing_key() (core/compat/sigv4.h)
18 * - HMAC / SHA-256: brix_hmac_sha256/brix_sha256 (core/compat/crypto.h)
19 */

26/*
27 * Lowercase-hex-encode `n` bytes of `in` into `out` (needs 2*n+1 bytes, incl.
28 * NUL). Used for the payload hash and the final signature, both of which SigV4
29 * carries as hex.
30 */
31static void
32sts_hex(const uint8_t *in, size_t n, char *out)
33{
34 static const char hx[] = "0123456789abcdef";
35 size_t i;
36
37 for (i = 0; i < n; i++) {
38 out[2 * i] = hx[(in[i] >> 4) & 0xf];
39 out[2 * i + 1] = hx[in[i] & 0xf];
40 }
41 out[2 * n] = '\0';
42}

45/*
46 * sts_sign_query — build and SigV4-sign the STS request query string.
47 *
48 * WHAT: given the sorted action query (`action_qs`, already canonical and
49 * percent-encoded, WITHOUT the trailing X-Amz-Signature) and the target
50 * host, compute the SigV4 signature and return the full signed query.
51 * WHY: the STS endpoint authenticates the node by SigV4 over its service
52 * credentials; this is the exact string AWS/MinIO recomputes and checks.
53 * HOW: canonical request = "GET\n/\n<qs>\nhost:<h>\n\nhost\nUNSIGNED?..." with an
54 * empty-body SHA-256; string-to-sign wraps it with the credential scope;
55 * signing key from brix_sigv4_signing_key over "sts"; signature =
56 * HMAC(key, string_to_sign). Result "<qs>&X-Amz-Signature=<hex>" is
57 * written NUL-terminated into `out` (size `outsz`). NGX_OK / NGX_ERROR.
58 *
59 * The timestamps, host and config are read from `req` (amzdate is
60 * "YYYYMMDDTHHMMSSZ", datestamp its "YYYYMMDD" prefix).
61 */
62ngx_int_t
63sts_sign_query(const sts_req_t *req, const char *action_qs,
64 char *out, size_t outsz)
65{
66 const brix_s3_sts_conf_t *cf = req->cf;
67 const char *host = req->host;
68 const char *amzdate = req->amzdate;
69 const char *datestamp = req->datestamp;
70 uint8_t empty_hash[32];
71 char empty_hex[65];
72 char canonical[4096];
73 uint8_t canon_hash[32];
74 char canon_hex[65];
75 char scope[128];
76 char to_sign[512];
77 uint8_t key[32];
78 uint8_t sig[32];
79 char sig_hex[65];
80 int n;
81
82 /* Empty request body — STS AssumeRole carries all params in the query. */
83 if (!brix_sha256((const uint8_t *) "", 0, empty_hash)) {
84 return NGX_ERROR;
85 }
86 sts_hex(empty_hash, 32, empty_hex);
87
88 n = ngx_snprintf((u_char *) canonical, sizeof(canonical),
89 "GET\n/\n%s\nhost:%s\nx-amz-date:%s\n\nhost;x-amz-date\n%s",
90 action_qs, host, amzdate, empty_hex) - (u_char *) canonical;
91 if (n <= 0 || (size_t) n >= sizeof(canonical)) {
92 return NGX_ERROR;
93 }
94 if (!brix_sha256((const uint8_t *) canonical, (size_t) n, canon_hash)) {
95 return NGX_ERROR;
96 }
97 sts_hex(canon_hash, 32, canon_hex);
98
99 ngx_snprintf((u_char *) scope, sizeof(scope), "%s/%V/sts/aws4_request%Z",
100 datestamp, &cf->region);
101
102 n = ngx_snprintf((u_char *) to_sign, sizeof(to_sign),
103 "AWS4-HMAC-SHA256\n%s\n%s\n%s",
104 amzdate, scope, canon_hex) - (u_char *) to_sign;
105 if (n <= 0 || (size_t) n >= sizeof(to_sign)) {
106 return NGX_ERROR;
107 }
108
109 if (!brix_sigv4_signing_key(cf->svc_sk.data, cf->svc_sk.len,
110 datestamp, (const char *) cf->region.data, "sts", key)) {
111 return NGX_ERROR;
112 }
113 if (!brix_hmac_sha256(key, 32, (const uint8_t *) to_sign, (size_t) n, sig)) {
114 return NGX_ERROR;
115 }
116 sts_hex(sig, 32, sig_hex);
117
118 n = ngx_snprintf((u_char *) out, outsz, "%s&X-Amz-Signature=%s%Z",
119 action_qs, sig_hex) - (u_char *) out;
120 if (n <= 0 || (size_t) n >= outsz) {
121 return NGX_ERROR;
122 }
123 return NGX_OK;
124}

127/*
128 * sts_build_action_qs — build the canonical (sorted, encoded) STS query string,
129 * WITHOUT the trailing X-Amz-Signature.
130 *
131 * Parameters appear in SigV4 canonical (byte-sorted-by-name) order. When a role
132 * ARN is configured we emit Action=AssumeRole (+ RoleArn/RoleSessionName);
133 * otherwise Action=GetSessionToken. `req->credential` is the pre-encoded
134 * "AKID%2FDATE%2Fregion%2Fsts%2Faws4_request" X-Amz-Credential value.
135 */
136ngx_int_t
137sts_build_action_qs(const sts_req_t *req, char *out, size_t outsz)
138{
139 const brix_s3_sts_conf_t *cf = req->cf;
140 const char *rsn = req->rsn;
141 const char *credential = req->credential;
142 const char *amzdate = req->amzdate;
143 int ttl = req->ttl;
144 int n;
145
146 if (cf->role_arn.len > 0) {
147 n = ngx_snprintf((u_char *) out, outsz,
148 "Action=AssumeRole"
149 "&DurationSeconds=%d"
150 "&RoleArn=%V"
151 "&RoleSessionName=%s"
152 "&Version=2011-06-15"
153 "&X-Amz-Algorithm=AWS4-HMAC-SHA256"
154 "&X-Amz-Credential=%s"
155 "&X-Amz-Date=%s"
156 "&X-Amz-SignedHeaders=host%%3Bx-amz-date%Z",
157 ttl, &cf->role_arn, rsn, credential, amzdate)
158 - (u_char *) out;
159 } else {
160 n = ngx_snprintf((u_char *) out, outsz,
161 "Action=GetSessionToken"
162 "&DurationSeconds=%d"
163 "&Version=2011-06-15"
164 "&X-Amz-Algorithm=AWS4-HMAC-SHA256"
165 "&X-Amz-Credential=%s"
166 "&X-Amz-Date=%s"
167 "&X-Amz-SignedHeaders=host%%3Bx-amz-date%Z",
168 ttl, credential, amzdate)
169 - (u_char *) out;
170 }
171
172 if (n <= 0 || (size_t) n >= outsz) {
173 return NGX_ERROR;
174 }
175 return NGX_OK;
176}
