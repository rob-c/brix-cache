7/*
8 * identity.c — builder/accessors for the protocol-agnostic principal.
9 *
10 * WHAT: Implements brix_identity_t lifecycle (alloc) and the setters that
11 * populate it from the various wire-auth flavours — GSI DN, JWT/token
12 * claims, SSS user, S3 access key — plus const accessors
13 * (brix_identity_dn_cstr / _subject_cstr / _vo_csv_cstr), a token
14 * scope check (brix_identity_check_token_scope), and a one-line audit
15 * summary (brix_identity_describe).
16 *
17 * WHY: Each protocol verifies credentials its own way, but policy, ACL, and
18 * audit code must reason about a single canonical principal shape. This
19 * module is the only place that knows how to translate per-protocol
20 * credential fragments into that shape and to keep the structured arrays
21 * (vo_list, scopes, token_scopes) consistent with the flat compatibility
22 * views (vo_csv, scope_raw) that current hot paths still read.
23 *
24 * HOW: Every string lands in the caller's ngx_pool_t via the internal
25 * brix_identity_set_cstr (NUL-terminated copy so the values can be
26 * passed to C string APIs). CSV group lists are tokenised by
27 * brix_identity_split_csv and space-separated OAuth scopes by
28 * brix_identity_split_spaces into ngx_str_t arrays. Any setter that
29 * records a credential ORs the matching BRIX_AUTHN_* bit into
30 * auth_method and marks is_authenticated; token claims additionally
31 * cache the parsed scope table and derive has_read_scope /
32 * has_write_scope for fast policy decisions.
33 */

35/*
36 * Append one ngx_str_t copy of [start, start+len) to *array, lazily creating
37 * the array on first use. Empty/NULL inputs are a successful no-op so callers
38 * can feed raw tokeniser output without pre-filtering.
39 */
40static ngx_int_t
41brix_identity_push_str(ngx_pool_t *pool, ngx_array_t **array,
42 const char *start, size_t len)
43{
44 ngx_str_t *elt;
45
46 if (pool == NULL || array == NULL || start == NULL || len == 0) {
47 return NGX_OK;
48 }
49
50 if (*array == NULL) {
51 *array = ngx_array_create(pool, 4, sizeof(ngx_str_t));
52 if (*array == NULL) {
53 return NGX_ERROR;
54 }
55 }
56
57 elt = ngx_array_push(*array);
58 if (elt == NULL) {
59 return NGX_ERROR;
60 }
61
62 elt->data = ngx_pnalloc(pool, len + 1);
63 if (elt->data == NULL) {
64 return NGX_ERROR;
65 }
66
67 ngx_memcpy(elt->data, start, len);
68 elt->data[len] = '\0';
69 elt->len = len;
70
71 return NGX_OK;
72}

74/*
75 * Split a comma-separated list (e.g. VO/group CSV) into individual ngx_str_t
76 * elements, skipping empty fields and runs of commas.
77 */
78static ngx_int_t
79brix_identity_split_csv(ngx_pool_t *pool, ngx_array_t **array,
80 const char *csv)
81{
82 const char *p, *start;
83
84 if (csv == NULL || csv[0] == '\0') {
85 return NGX_OK;
86 }
87
88 p = csv;
89 while (*p != '\0') {
90 while (*p == ',') {
91 p++;
92 }
93 start = p;
94 while (*p != '\0' && *p != ',') {
95 p++;
96 }
97 if ((size_t) (p - start) > 0
98 && brix_identity_push_str(pool, array, start,
99 (size_t) (p - start)) != NGX_OK)
100 {
101 return NGX_ERROR;
102 }
103 }
104
105 return NGX_OK;
106}

288/*
289 * Populate the identity from a verified token's claims: sets subject (sub),
290 * issuer (iss), raw scope string, and groups; mirrors `sub` into `dn` when
291 * present; tokenises the raw scope into the scopes array; caches up to
292 * BRIX_MAX_TOKEN_SCOPES parsed scope entries; and derives has_read_scope /
293 * has_write_scope from their read/write/create flags. Marks the principal
294 * token-authenticated.
295 */
296ngx_int_t
297brix_identity_set_token_claims(brix_identity_t *id, ngx_pool_t *pool,
298 const brix_token_claims_t *claims)
299{
300 int i;
301
302 if (id == NULL || claims == NULL) {
303 return NGX_ERROR;
304 }
305
306 if (brix_identity_set_subject(id, pool, claims->sub,
307 BRIX_AUTHN_TOKEN) != NGX_OK
308 || brix_identity_set_cstr(pool, &id->issuer, claims->iss) != NGX_OK
309 || brix_identity_set_cstr(pool, &id->scope_raw,
310 claims->scope_raw) != NGX_OK
311 || brix_identity_set_vos_csv(id, pool, claims->groups) != NGX_OK)
312 {
313 return NGX_ERROR;
314 }
315
316 if (claims->sub[0] != '\0'
317 && brix_identity_set_cstr(pool, &id->dn, claims->sub) != NGX_OK)
318 {
319 return NGX_ERROR;
320 }
321
322 if (brix_identity_split_spaces(pool, &id->scopes,
323 claims->scope_raw) != NGX_OK)
324 {
325 return NGX_ERROR;
326 }
327
328 id->token_scope_count = claims->scope_count;
329 if (id->token_scope_count > BRIX_MAX_TOKEN_SCOPES) {
330 id->token_scope_count = BRIX_MAX_TOKEN_SCOPES;
331 }
332
333 for (i = 0; i < id->token_scope_count; i++) {
334 id->token_scopes[i] = claims->scopes[i];
335 if (claims->scopes[i].read) {
336 id->has_read_scope = 1;
337 }
338 if (claims->scopes[i].write || claims->scopes[i].create) {
339 id->has_write_scope = 1;
340 }
341 }
342
343 id->auth_method |= BRIX_AUTHN_TOKEN;
344 id->is_authenticated = 1;
345
346 return NGX_OK;
347}

394/*
395 * Authorise `logical_path` against the cached token scopes. Returns NGX_OK
396 * (allow) immediately for non-token principals — scope checks apply only to
397 * token auth; otherwise delegates to brix_token_check_write/read depending
398 * on need_write and returns NGX_OK only when the scope grants access.
399 */
400ngx_int_t
401brix_identity_check_token_scope(const brix_identity_t *id,
402 const char *logical_path, int need_write)
403{
404 if (id == NULL || !(id->auth_method & BRIX_AUTHN_TOKEN)) {
405 return NGX_OK;
406 }
407
408 /* phase-59 W1: when authed via a multi-issuer registry, enforce the
409 * issuer's base_path/restricted_path gate + strategy ladder per path. */
410 if (id->token_issuer != NULL) {
411 brix_token_op_e op = need_write ? BRIX_TOKEN_OP_WRITE
412 : BRIX_TOKEN_OP_READ;
413 brix_token_claims_t c;
414 ngx_memzero(&c, sizeof(c));
415 c.scope_count = id->token_scope_count;
416 ngx_memcpy(c.scopes, id->token_scopes,
417 sizeof(brix_token_scope_t) * id->token_scope_count);
418 return brix_token_authz_strategy(
419 (const brix_token_issuer_t *) id->token_issuer,
420 &c, logical_path, op) ? NGX_OK : NGX_ERROR;
421 }
422
423 if (need_write) {
424 return brix_token_check_write(id->token_scopes,
425 id->token_scope_count,
426 logical_path)
427 ? NGX_OK : NGX_ERROR;
428 }
429
430 return brix_token_check_read(id->token_scopes,
431 id->token_scope_count,
432 logical_path)
433 ? NGX_OK : NGX_ERROR;
434}

441brix_identity_describe(const brix_identity_t *id, ngx_pool_t *pool)
