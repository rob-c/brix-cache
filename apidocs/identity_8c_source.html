6/*
7 * identity.c — builder/accessors for the protocol-agnostic principal.
8 *
9 * WHAT: Implements brix_identity_t lifecycle (alloc) and the setters that
10 * populate it from the various wire-auth flavours — GSI DN, JWT/token
11 * claims, SSS user, S3 access key — plus const accessors
12 * (brix_identity_dn_cstr / _subject_cstr / _vo_csv_cstr), a token
13 * scope check (brix_identity_check_token_scope), and a one-line audit
14 * summary (brix_identity_describe).
15 *
16 * WHY: Each protocol verifies credentials its own way, but policy, ACL, and
17 * audit code must reason about a single canonical principal shape. This
18 * module is the only place that knows how to translate per-protocol
19 * credential fragments into that shape and to keep the structured arrays
20 * (vo_list, scopes, token_scopes) consistent with the flat compatibility
21 * views (vo_csv, scope_raw) that current hot paths still read.
22 *
23 * HOW: Every string lands in the caller's ngx_pool_t via the internal
24 * brix_identity_set_cstr (NUL-terminated copy so the values can be
25 * passed to C string APIs). CSV group lists are tokenised by
26 * brix_identity_split_csv and space-separated OAuth scopes by
27 * brix_identity_split_spaces into ngx_str_t arrays. Any setter that
28 * records a credential ORs the matching BRIX_AUTHN_* bit into
29 * auth_method and marks is_authenticated; token claims additionally
30 * cache the parsed scope table and derive has_read_scope /
31 * has_write_scope for fast policy decisions.
32 */
33
34/*
35 * Append one ngx_str_t copy of [start, start+len) to *array, lazily creating
36 * the array on first use. Empty/NULL inputs are a successful no-op so callers
37 * can feed raw tokeniser output without pre-filtering.
38 */
39static ngx_int_t
40brix_identity_push_str(ngx_pool_t *pool, ngx_array_t **array,
41 const char *start, size_t len)
42{
43 ngx_str_t *elt;
44
45 if (pool == NULL || array == NULL || start == NULL || len == 0) {
46 return NGX_OK;
47 }
48
49 if (*array == NULL) {
50 *array = ngx_array_create(pool, 4, sizeof(ngx_str_t));
51 if (*array == NULL) {
52 return NGX_ERROR;
53 }
54 }
55
56 elt = ngx_array_push(*array);
57 if (elt == NULL) {
58 return NGX_ERROR;
59 }
60
61 elt->data = ngx_pnalloc(pool, len + 1);
62 if (elt->data == NULL) {
63 return NGX_ERROR;
64 }
65
66 ngx_memcpy(elt->data, start, len);
67 elt->data[len] = '\0';
68 elt->len = len;
69
70 return NGX_OK;
71}

73/*
74 * Split a comma-separated list (e.g. VO/group CSV) into individual ngx_str_t
75 * elements, skipping empty fields and runs of commas.
76 */
77static ngx_int_t
78brix_identity_split_csv(ngx_pool_t *pool, ngx_array_t **array,
79 const char *csv)
80{
81 const char *p, *start;
82
83 if (csv == NULL || csv[0] == '\0') {
84 return NGX_OK;
85 }
86
87 p = csv;
88 while (*p != '\0') {
89 while (*p == ',') {
90 p++;
91 }
92 start = p;
93 while (*p != '\0' && *p != ',') {
94 p++;
95 }
96 if ((size_t) (p - start) > 0
97 && brix_identity_push_str(pool, array, start,
98 (size_t) (p - start)) != NGX_OK)
99 {
100 return NGX_ERROR;
101 }
102 }
103
104 return NGX_OK;
105}

264/*
265 * brix_identity_derive_attrs — split each comma-separated VOMS FQAN / token
266 * group into (vorg, role, group) and store them as three index-aligned CSVs for
267 * the xrdacc engine. An FQAN like "/cms/Role=production/Capability=NULL" yields
268 * vorg="cms", role="production", group="/cms"; "Role=NULL" and a plain group
269 * name ("cms") yield an empty role. Each output is a prefix/substring of the
270 * input, so an strlen(vo_csv)-sized buffer always suffices.
271 */
272static ngx_int_t
273brix_identity_derive_attrs(brix_identity_t *id, ngx_pool_t *pool,
274 const char *vo_csv)
275{
276 size_t inlen = (vo_csv != NULL) ? ngx_strlen(vo_csv) : 0;
277 u_char *vb, *rb, *gb;
278 size_t vl = 0, rl = 0, gl = 0;
279 const char *p, *toks;
280 int first = 1;
281
282 ngx_str_null(&id->acc_vorg_csv);
283 ngx_str_null(&id->acc_role_csv);
284 ngx_str_null(&id->acc_group_csv);
285 if (inlen == 0) {
286 return NGX_OK;
287 }
288
289 vb = ngx_pnalloc(pool, inlen + 1);
290 rb = ngx_pnalloc(pool, inlen + 1);
291 gb = ngx_pnalloc(pool, inlen + 1);
292 if (vb == NULL || rb == NULL || gb == NULL) {
293 return NGX_ERROR;
294 }
295
296 p = toks = vo_csv;
297 for (;;) {
298 if (*p == ',' || *p == '\0') {
299 const char *tok = toks;
300 size_t tl = (size_t) (p - toks);
301 const char *role = NULL, *grp, *rp, *cp, *vstart, *vend;
302 size_t rlen = 0, glen, vlen;
303
304 while (tl > 0 && (*tok == ' ' || *tok == '\t')) { tok++; tl--; }
305 while (tl > 0 && (tok[tl - 1] == ' ' || tok[tl - 1] == '\t')) { tl--; }
306
307 grp = tok;
308 glen = tl;
309 rp = memmem(tok, tl, "/Role=", 6);
310 if (rp != NULL) {
311 const char *rs = rp + 6, *re;
312 glen = (size_t) (rp - tok); /* group = before /Role= */
313 re = memchr(rs, '/', (size_t) ((tok + tl) - rs));
314 if (re == NULL) { re = tok + tl; }
315 role = rs;
316 rlen = (size_t) (re - rs);
317 if (rlen == 4 && ngx_strncmp(role, "NULL", 4) == 0) { rlen = 0; }
318 } else if ((cp = memmem(tok, tl, "/Capability=", 12)) != NULL) {
319 glen = (size_t) (cp - tok); /* strip trailing capability */
320 }
321
322 vstart = grp;
323 vlen = glen;
324 if (vlen > 0 && *vstart == '/') { vstart++; vlen--; }
325 vend = memchr(vstart, '/', vlen);
326 if (vend != NULL) { vlen = (size_t) (vend - vstart); }
327
328 if (!first) { vb[vl++] = ','; rb[rl++] = ','; gb[gl++] = ','; }
329 first = 0;
330 if (vlen) { ngx_memcpy(vb + vl, vstart, vlen); vl += vlen; }
331 if (rlen) { ngx_memcpy(rb + rl, role, rlen); rl += rlen; }
332 if (glen) { ngx_memcpy(gb + gl, grp, glen); gl += glen; }
333
334 if (*p == '\0') { break; }
335 toks = p + 1;
336 }
337 p++;
338 }
339 vb[vl] = '\0'; rb[rl] = '\0'; gb[gl] = '\0';
340 id->acc_vorg_csv.data = vb; id->acc_vorg_csv.len = vl;
341 id->acc_role_csv.data = rb; id->acc_role_csv.len = rl;
342 id->acc_group_csv.data = gb; id->acc_group_csv.len = gl;
343 return NGX_OK;
344}

369/*
370 * Populate the identity from a verified token's claims: sets subject (sub),
371 * issuer (iss), raw scope string, and groups; mirrors `sub` into `dn` when
372 * present; tokenises the raw scope into the scopes array; caches up to
373 * BRIX_MAX_TOKEN_SCOPES parsed scope entries; and derives has_read_scope /
374 * has_write_scope from their read/write/create flags. Marks the principal
375 * token-authenticated.
376 */
377ngx_int_t
378brix_identity_set_token_claims(brix_identity_t *id, ngx_pool_t *pool,
379 const brix_token_claims_t *claims)
380{
381 int i;
382
383 if (id == NULL || claims == NULL) {
384 return NGX_ERROR;
385 }
386
387 if (brix_identity_set_subject(id, pool, claims->sub,
388 BRIX_AUTHN_TOKEN) != NGX_OK
389 || brix_identity_set_cstr(pool, &id->issuer, claims->iss) != NGX_OK
390 || brix_identity_set_cstr(pool, &id->scope_raw,
391 claims->scope_raw) != NGX_OK
392 || brix_identity_set_vos_csv(id, pool, claims->groups) != NGX_OK)
393 {
394 return NGX_ERROR;
395 }
396
397 if (claims->sub[0] != '\0'
398 && brix_identity_set_cstr(pool, &id->dn, claims->sub) != NGX_OK)
399 {
400 return NGX_ERROR;
401 }
402
403 if (brix_identity_split_spaces(pool, &id->scopes,
404 claims->scope_raw) != NGX_OK)
405 {
406 return NGX_ERROR;
407 }
408
409 id->token_scope_count = claims->scope_count;
410 if (id->token_scope_count > BRIX_MAX_TOKEN_SCOPES) {
411 id->token_scope_count = BRIX_MAX_TOKEN_SCOPES;
412 }
413
414 for (i = 0; i < id->token_scope_count; i++) {
415 id->token_scopes[i] = claims->scopes[i];
416 if (claims->scopes[i].read) {
417 id->has_read_scope = 1;
418 }
419 if (claims->scopes[i].write || claims->scopes[i].create) {
420 id->has_write_scope = 1;
421 }
422 }
423
424 id->auth_method |= BRIX_AUTHN_TOKEN;
425 id->is_authenticated = 1;
426
427 return NGX_OK;
428}

475/*
476 * Authorise `logical_path` against the cached token scopes. Returns NGX_OK
477 * (allow) immediately for non-token principals — scope checks apply only to
478 * token auth; otherwise delegates to brix_token_check_write/read depending
479 * on need_write and returns NGX_OK only when the scope grants access.
480 */
481ngx_int_t
482brix_identity_check_token_scope(const brix_identity_t *id,
483 const char *logical_path, int need_write)
484{
485 if (id == NULL || !(id->auth_method & BRIX_AUTHN_TOKEN)) {
486 return NGX_OK;
487 }
488
489 /* phase-59 W1: when authed via a multi-issuer registry, enforce the
490 * issuer's base_path/restricted_path gate + strategy ladder per path. */
491 if (id->token_issuer != NULL) {
492 brix_token_op_e op = need_write ? BRIX_TOKEN_OP_WRITE
493 : BRIX_TOKEN_OP_READ;
494 brix_token_claims_t c;
495 ngx_memzero(&c, sizeof(c));
496 c.scope_count = id->token_scope_count;
497 ngx_memcpy(c.scopes, id->token_scopes,
498 sizeof(brix_token_scope_t) * id->token_scope_count);
499 return brix_token_authz_strategy(
500 (const brix_token_issuer_t *) id->token_issuer,
501 &c, logical_path, op) ? NGX_OK : NGX_ERROR;
502 }
503
504 if (need_write) {
505 return brix_token_check_write(id->token_scopes,
506 id->token_scope_count,
507 logical_path)
508 ? NGX_OK : NGX_ERROR;
509 }
510
511 return brix_token_check_read(id->token_scopes,
512 id->token_scope_count,
513 logical_path)
514 ? NGX_OK : NGX_ERROR;
515}
