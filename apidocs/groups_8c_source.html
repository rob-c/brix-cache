1/*
2 * groups.c — OS/NIS group resolution for the xrdacc engine (XrdAccGroups).
3 *
4 * WHAT: resolves a user to its Unix supplementary groups (getpwnam +
5 * getgrouplist) and tests NIS netgroup membership (innetgr), so `g` records
6 * can match a user's OS gidlist and `n` records can match NIS netgroups —
7 * beyond the VO/token groups the credential itself carries. A per-worker
8 * TTL cache (gidlifetime) keeps the slow NSS lookups off the hot path.
9 *
10 * WHY: completes XrdAcc parity (acc.gidlifetime / acc.pgo / acc.nisdomain).
11 * The access engine reaches this layer only through the function pointers
12 * installed by brix_acc_groups_init(), so the decision engine stays
13 * testable without the OS.
14 *
15 * HOW: a small open-addressed per-worker cache keyed by username holds the
16 * resolved group-name list with an expiry; the resolver copies cached names
17 * into the caller's request pool. netgroup membership is a thin innetgr()
18 * wrapper honouring the configured NIS domain. Per-worker, single-threaded:
19 * no locking needed.
20 */

45/* Parse a space/comma-separated gid list into the retran table (replaces the
46 * previous set). Non-numeric / overflow entries are skipped (best effort). */
47void
48brix_acc_groups_set_gidretran(const char *gidlist)
49{
50 const char *p = gidlist;
51
52 acc_retrancnt = 0;
53 if (gidlist == NULL) {
54 return;
55 }
56 while (*p != '\0' && acc_retrancnt < BRIX_ACC_MAX_RETRAN) {
57 while (*p == ' ' || *p == ',' || *p == '\t') {
58 p++;
59 }
60 if (*p < '0' || *p > '9') {
61 break;
62 }
63 acc_retrangid[acc_retrancnt++] = (gid_t) ngx_atoi((u_char *) p,
64 ngx_strlen(p));
65 while (*p >= '0' && *p <= '9') {
66 p++;
67 }
68 }
69}

71/* Dotran: a gid in the retran list resolves to no usable group name (its
72 * gid->name mapping is ambiguous under NIS). Returns 1 to skip the gid. */
73static int
74acc_gid_retran(gid_t gid)
75{
76 int i;
77 for (i = 0; i < acc_retrancnt; i++) {
78 if (acc_retrangid[i] == gid) {
79 return 1;
80 }
81 }
82 return 0;
83}

108/*
109 * Phase 51 (E3): bound the blocking NSS group resolution under system pressure.
110 * getpwnam/getgrouplist/getgrgid block the single-threaded event loop when the
111 * backing directory service (NIS/LDAP) is slow or down. Two guards:
112 * - NEGATIVE CACHE: an unknown / no-group user is cached (count 0) for a short
113 * TTL so a flood of distinct unknown users cannot re-block on every request.
114 * - CIRCUIT BREAKER: after N consecutive SLOW resolutions the breaker opens for
115 * a cooldown, during which lookups fail fast to "no supplementary groups"
116 * (the conservative degraded behaviour — group-based grants are withheld, the
117 * user's primary identity still works) instead of re-blocking the worker.
118 */
119#define ACC_NSS_NEG_TTL_SECS 60 /* negative-cache lifetime */
120#define ACC_NSS_SLOW_MS 2000 /* a resolve over this is "slow" */
121#define ACC_NSS_TRIP_COUNT 5 /* consecutive slow resolves → open */
122#define ACC_NSS_COOLDOWN_SECS 10 /* breaker stays open this long */
123
124static struct {
125 int consecutive_slow;
126 time_t open_until; /* breaker open (skip NSS) until this time; 0 = closed */
127} acc_nss_breaker;
128
129/* Monotonic milliseconds for measuring a blocking call's wall time (ngx_current_msec
130 * is not updated while a syscall blocks, so it cannot time one). */
131static int64_t
132acc_monotonic_ms(void)
133{
134 struct timespec ts;
135 if (clock_gettime(CLOCK_MONOTONIC, &ts) != 0) {
136 return 0;
137 }
138 return (int64_t) ts.tv_sec * 1000 + ts.tv_nsec / 1000000;
139}

141static unsigned
142acc_user_hash(const char *user)
143{
144 unsigned h = 5381;
145 while (*user) { h = ((h << 5) + h) ^ (unsigned char) *user++; }
146 return h % ACC_GRP_CACHE_SLOTS;
147}

162/* Resolve a user's Unix groups via NSS into a fresh malloc'd vector.
163 * Returns count (>=0) and *namesp (malloc'd, NULL if 0), or -1 on unknown user. */
164static int
165acc_resolve_unix(const char *user, char ***namesp)
166{
167 struct passwd *pw;
168 gid_t gids[64];
169 int ng = (int) (sizeof(gids) / sizeof(gids[0]));
170 char **names;
171 int out = 0, i;
172
173 *namesp = NULL;
174
175 pw = getpwnam(user);
176 if (pw == NULL) {
177 return -1; /* not a local user: no Unix groups */
178 }
179
180 if (acc_primary_only) {
181 gids[0] = pw->pw_gid;
182 ng = 1;
183 } else if (getgrouplist(user, pw->pw_gid, gids, &ng) < 0) {
184 /* buffer too small: cap at what we have room for */
185 ng = (int) (sizeof(gids) / sizeof(gids[0]));
186 }
187 if (ng <= 0) {
188 return 0;
189 }
190
191 names = malloc((size_t) ng * sizeof(char *));
192 if (names == NULL) {
193 return 0;
194 }
195 for (i = 0; i < ng; i++) {
196 struct group *gr;
197 if (acc_gid_retran(gids[i])) {
198 continue; /* ambiguous shared gid — skip its name */
199 }
200 gr = getgrgid(gids[i]);
201 if (gr != NULL && gr->gr_name != NULL) {
202 names[out] = strdup(gr->gr_name);
203 if (names[out] != NULL) { out++; }
204 }
205 }
206 if (out == 0) { free(names); names = NULL; }
207 *namesp = names;
208 return out;
209}

211/* Public resolver (installed into the engine): user -> request-pool array of
212 * group-name strings, using the TTL cache. */
213static ngx_array_t *
214brix_acc_unix_groups(ngx_pool_t *pool, const char *user)
215{
216 acc_grp_cache_t *e;
217 ngx_array_t *out;
218 time_t now;
219 int i;
220
221 if (user == NULL || *user == '\0' || ngx_strlen(user) >= ACC_GRP_USER_MAX) {
222 return NULL;
223 }
224 now = time(NULL);
225 e = &acc_grp_cache[acc_user_hash(user)];
226
227 /* Cache hit on the same user and not expired? */
228 if (!(e->user[0] != '\0' && e->expiry > now && ngx_strcmp(e->user, user) == 0)) {
229 char **names;
230 int cnt;
231 int64_t t0, elapsed;
232
233 /* E3: circuit breaker open → fail fast to "no groups", don't touch NSS. */
234 if (acc_nss_breaker.open_until > now) {
235 return NULL;
236 }
237
238 t0 = acc_monotonic_ms();
239 cnt = acc_resolve_unix(user, &names);
240 elapsed = acc_monotonic_ms() - t0;
241
242 /* E3: trip the breaker on sustained slow resolutions. */
243 if (elapsed >= ACC_NSS_SLOW_MS) {
244 if (++acc_nss_breaker.consecutive_slow >= ACC_NSS_TRIP_COUNT) {
245 acc_nss_breaker.open_until = now + ACC_NSS_COOLDOWN_SECS;
246 acc_nss_breaker.consecutive_slow = 0;
247 BRIX_RESIL_METRIC_INC(acc_nss_breaker_open_total);
248 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
249 "brix_acc: NSS group lookup slow (%L ms) — "
250 "opening circuit breaker for %ds (failing to "
251 "no-supplementary-groups)",
252 (long long) elapsed, ACC_NSS_COOLDOWN_SECS);
253 }
254 } else {
255 acc_nss_breaker.consecutive_slow = 0;
256 }
257
258 acc_cache_free(e);
259
260 /* E3: negative-cache an unknown / no-group user (short TTL) so a flood of
261 * distinct misses cannot re-block NSS on every request. */
262 if (cnt <= 0) {
263 ngx_memcpy(e->user, user, ngx_strlen(user) + 1);
264 e->names = NULL;
265 e->count = 0;
266 e->expiry = now + ngx_min(ACC_NSS_NEG_TTL_SECS, acc_gidlifetime);
267 return NULL;
268 }
269
270 ngx_memcpy(e->user, user, ngx_strlen(user) + 1);
271 e->names = names;
272 e->count = cnt;
273 e->expiry = now + acc_gidlifetime;
274 }
275
276 if (e->count == 0) {
277 return NULL;
278 }
279
280 out = ngx_array_create(pool, (ngx_uint_t) e->count, sizeof(char *));
281 if (out == NULL) {
282 return NULL;
283 }
284 for (i = 0; i < e->count; i++) {
285 char **slot = ngx_array_push(out);
286 size_t len = ngx_strlen(e->names[i]);
287 char *copy = ngx_pnalloc(pool, len + 1);
288 if (slot == NULL || copy == NULL) {
289 return NULL;
290 }
291 ngx_memcpy(copy, e->names[i], len + 1);
292 *slot = copy;
293 }
294 return out;
295}
