1/*
2 * groups.c — OS/NIS group resolution for the xrdacc engine (XrdAccGroups).
3 *
4 * WHAT: resolves a user to its Unix supplementary groups (getpwnam +
5 * getgrouplist) and tests NIS netgroup membership (innetgr), so `g` records
6 * can match a user's OS gidlist and `n` records can match NIS netgroups —
7 * beyond the VO/token groups the credential itself carries. A per-worker
8 * TTL cache (gidlifetime) keeps the slow NSS lookups off the hot path.
9 *
10 * WHY: completes XrdAcc parity (acc.gidlifetime / acc.pgo / acc.nisdomain).
11 * The access engine reaches this layer only through the function pointers
12 * installed by brix_acc_groups_init(), so the decision engine stays
13 * testable without the OS.
14 *
15 * HOW: a small open-addressed per-worker cache keyed by username holds the
16 * resolved group-name list with an expiry; the resolver copies cached names
17 * into the caller's request pool. netgroup membership is a thin innetgr()
18 * wrapper honouring the configured NIS domain. Per-worker, single-threaded:
19 * no locking needed.
20 */

45/* Parse a space/comma-separated gid list into the retran table (replaces the
46 * previous set). Non-numeric / overflow entries are skipped (best effort). */
47void
48brix_acc_groups_set_gidretran(const char *gidlist)
49{
50 const char *p = gidlist;
51
52 acc_retrancnt = 0;
53 if (gidlist == NULL) {
54 return;
55 }
56 while (*p != '\0' && acc_retrancnt < BRIX_ACC_MAX_RETRAN) {
57 while (*p == ' ' || *p == ',' || *p == '\t') {
58 p++;
59 }
60 if (*p < '0' || *p > '9') {
61 break;
62 }
63 acc_retrangid[acc_retrancnt++] = (gid_t) ngx_atoi((u_char *) p,
64 ngx_strlen(p));
65 while (*p >= '0' && *p <= '9') {
66 p++;
67 }
68 }
69}

77 for (i = 0; i < acc_retrancnt; i++) {

109/*
110 * Phase 51 (E3): bound the blocking NSS group resolution under system pressure.
111 * getpwnam/getgrouplist/getgrgid block the single-threaded event loop when the
112 * backing directory service (NIS/LDAP) is slow or down. Two guards:
113 * - NEGATIVE CACHE: an unknown / no-group user is cached (count 0) for a short
114 * TTL so a flood of distinct unknown users cannot re-block on every request.
115 * - CIRCUIT BREAKER: after N consecutive SLOW resolutions the breaker opens for
116 * a cooldown, during which lookups fail fast to "no supplementary groups"
117 * (the conservative degraded behaviour — group-based grants are withheld, the
118 * user's primary identity still works) instead of re-blocking the worker.
119 */
120#define ACC_NSS_NEG_TTL_SECS 60 /* negative-cache lifetime */
121#define ACC_NSS_SLOW_MS 2000 /* a resolve over this is "slow" */
122#define ACC_NSS_TRIP_COUNT 5 /* consecutive slow resolves → open */
123#define ACC_NSS_COOLDOWN_SECS 10 /* breaker stays open this long */
124
125static struct {
126 int consecutive_slow;
127 time_t open_until; /* breaker open (skip NSS) until this time; 0 = closed */
128} acc_nss_breaker;
129
130/* Monotonic milliseconds for measuring a blocking call's wall time (ngx_current_msec
131 * is not updated while a syscall blocks, so it cannot time one). */
132static int64_t
133acc_monotonic_ms(void)
134{
135 struct timespec ts;
136 if (clock_gettime(CLOCK_MONOTONIC, &ts) != 0) {
137 return 0;
138 }
139 return (int64_t) ts.tv_sec * 1000 + ts.tv_nsec / 1000000;
140}

142static unsigned
143acc_user_hash(const char *user)
144{
145 unsigned h = ACC_DJB2_SEED;
146 while (*user) { h = ((h << 5) + h) ^ (unsigned char) *user++; }
147 return h % ACC_GRP_CACHE_SLOTS;
148}

163/* Resolve a user's Unix groups via NSS into a fresh malloc'd vector.
164 * Returns count (>=0) and *namesp (malloc'd, NULL if 0), or -1 on unknown user. */
165static int
166acc_resolve_unix(const char *user, char ***namesp)
167{
168 struct passwd *pw;
169 gid_t gids[64];
170 int ng = (int) (sizeof(gids) / sizeof(gids[0]));
171 char **names;
172 int out = 0, i;
173
174 *namesp = NULL;
175
176 pw = getpwnam(user);
177 if (pw == NULL) {
178 return -1; /* not a local user: no Unix groups */
179 }
180
181 if (acc_primary_only) {
182 gids[0] = pw->pw_gid;
183 ng = 1;
184 } else if (getgrouplist(user, pw->pw_gid, gids, &ng) < 0) {
185 /* buffer too small: cap at what we have room for */
186 ng = (int) (sizeof(gids) / sizeof(gids[0]));
187 }
188 if (ng <= 0) {
189 return 0;
190 }
191
192 names = malloc((size_t) ng * sizeof(char *));
193 if (names == NULL) {
194 return 0;
195 }
196 for (i = 0; i < ng; i++) {
197 struct group *gr;
198 if (acc_gid_retran(gids[i])) {
199 continue; /* ambiguous shared gid — skip its name */
200 }
201 gr = getgrgid(gids[i]);
202 if (gr != NULL && gr->gr_name != NULL) {
203 names[out] = strdup(gr->gr_name);
204 if (names[out] != NULL) { out++; }
205 }
206 }
207 if (out == 0) { free(names); names = NULL; }
208 *namesp = names;
209 return out;
210}

212/* ---- Record one NSS resolve latency into the circuit breaker ----
213 *
214 * WHAT: given the wall time of a completed NSS group resolve and the current
215 * time, updates the shared breaker state. A resolve at or over ACC_NSS_SLOW_MS
216 * increments the consecutive-slow counter and, on the ACC_NSS_TRIP_COUNT-th
217 * consecutive slow resolve, opens the breaker for ACC_NSS_COOLDOWN_SECS (and
218 * emits the resiliency metric + a WARN). A fast resolve resets the counter.
219 *
220 * WHY: extracted from brix_acc_unix_groups so the resolver orchestrator stays
221 * within the complexity cap; isolating the breaker-trip decision here keeps
222 * the slow/fast accounting in one nameable place with no behaviour change.
223 *
224 * HOW:
225 * 1. If elapsed >= ACC_NSS_SLOW_MS, pre-increment consecutive_slow.
226 * 2. When that count reaches ACC_NSS_TRIP_COUNT, set open_until = now +
227 * cooldown, reset the counter, bump the breaker-open metric, and log WARN.
228 * 3. Otherwise (fast resolve) reset consecutive_slow to 0.
229 */
230static void
231acc_nss_note_latency(int64_t elapsed, time_t now)
232{
233 /* E3: trip the breaker on sustained slow resolutions. */
234 if (elapsed >= ACC_NSS_SLOW_MS) {
235 if (++acc_nss_breaker.consecutive_slow >= ACC_NSS_TRIP_COUNT) {
236 acc_nss_breaker.open_until = now + ACC_NSS_COOLDOWN_SECS;
237 acc_nss_breaker.consecutive_slow = 0;
238 BRIX_RESIL_METRIC_INC(acc_nss_breaker_open_total);
239 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
240 "brix_acc: NSS group lookup slow (%L ms) — "
241 "opening circuit breaker for %ds (failing to "
242 "no-supplementary-groups)",
243 (long long) elapsed, ACC_NSS_COOLDOWN_SECS);
244 }
245 } else {
246 acc_nss_breaker.consecutive_slow = 0;
247 }
248}

250/* ---- Refresh one expired/absent cache entry from NSS ----
251 *
252 * WHAT: (re)populates cache entry e for user via a fresh NSS resolve, subject to
253 * the circuit breaker. Returns NGX_OK when e now holds a real group-name list
254 * (cnt > 0) that the caller should render; returns NGX_DECLINED when the caller
255 * must yield "no supplementary groups" — either because the breaker is open
256 * (NSS untouched, e left as-is) or because the user resolved to zero groups
257 * (negatively cached with a short TTL). e is always left in a self-consistent
258 * state on the NGX_DECLINED no-group path (count 0).
259 *
260 * WHY: extracted from brix_acc_unix_groups to keep that resolver under the
261 * complexity cap. The NGX_DECLINED/NGX_OK split preserves the original invariant
262 * that a breaker-open miss returns immediately WITHOUT rendering any stale entry.
263 *
264 * HOW:
265 * 1. If the breaker is open (open_until > now), return NGX_DECLINED without
266 * touching NSS or e (matches the original early return).
267 * 2. Time the acc_resolve_unix() call with the monotonic clock and feed the
268 * elapsed time to acc_nss_note_latency() to maintain the breaker.
269 * 3. Free the old entry contents via acc_cache_free().
270 * 4. On cnt <= 0, negatively cache the username with a short TTL and return
271 * NGX_DECLINED.
272 * 5. Otherwise store the resolved name vector with the full gidlifetime TTL and
273 * return NGX_OK.
274 */
275static ngx_int_t
276acc_cache_refresh(acc_grp_cache_t *e, const char *user, time_t now)
277{
278 char **names;
279 int cnt;
280 int64_t t0, elapsed;
281
282 /* E3: circuit breaker open → fail fast to "no groups", don't touch NSS. */
283 if (acc_nss_breaker.open_until > now) {
284 return NGX_DECLINED;
285 }
286
287 t0 = acc_monotonic_ms();
288 cnt = acc_resolve_unix(user, &names);
289 elapsed = acc_monotonic_ms() - t0;
290
291 acc_nss_note_latency(elapsed, now);
292
293 acc_cache_free(e);
294
295 /* E3: negative-cache an unknown / no-group user (short TTL) so a flood of
296 * distinct misses cannot re-block NSS on every request. */
297 if (cnt <= 0) {
298 ngx_memcpy(e->user, user, ngx_strlen(user) + 1);
299 e->names = NULL;
300 e->count = 0;
301 e->expiry = now + ngx_min(ACC_NSS_NEG_TTL_SECS, acc_gidlifetime);
302 return NGX_DECLINED;
303 }
304
305 ngx_memcpy(e->user, user, ngx_strlen(user) + 1);
306 e->names = names;
307 e->count = cnt;
308 e->expiry = now + acc_gidlifetime;
309 return NGX_OK;
310}

312/* ---- Copy a cache entry's group names into a request-pool array ----
313 *
314 * WHAT: builds an ngx_array_t of char* in pool, each element a pool-owned
315 * null-terminated copy of one cached group name. Returns the array, or NULL on
316 * any allocation failure.
317 *
318 * WHY: the cached name vector is malloc'd and per-worker-persistent; callers need
319 * request-lifetime copies. Isolating the copy loop keeps the resolver flat and
320 * under the complexity cap.
321 *
322 * HOW:
323 * 1. Create the array sized to e->count.
324 * 2. For each cached name, push a slot and pnalloc a len+1 copy in pool.
325 * 3. memcpy the name (including its terminator) and store the copy.
326 * 4. Return NULL on any allocation failure encountered along the way.
327 */
328static ngx_array_t *
329acc_cache_to_array(ngx_pool_t *pool, acc_grp_cache_t *e)
330{
331 ngx_array_t *out;
332 int i;
333
334 out = ngx_array_create(pool, (ngx_uint_t) e->count, sizeof(char *));
335 if (out == NULL) {
336 return NULL;
337 }
338 for (i = 0; i < e->count; i++) {
339 char **slot = ngx_array_push(out);
340 size_t len = ngx_strlen(e->names[i]);
341 char *copy = ngx_pnalloc(pool, len + 1);
342 if (slot == NULL || copy == NULL) {
343 return NULL;
344 }
345 ngx_memcpy(copy, e->names[i], len + 1);
346 *slot = copy;
347 }
348 return out;
349}

351/* Public resolver (installed into the engine): user -> request-pool array of
352 * group-name strings, using the TTL cache. */
353static ngx_array_t *
354brix_acc_unix_groups(ngx_pool_t *pool, const char *user)
355{
356 acc_grp_cache_t *e;
357 time_t now;
358
359 if (user == NULL || *user == '\0' || ngx_strlen(user) >= ACC_GRP_USER_MAX) {
360 return NULL;
361 }
362 now = time(NULL);
363 e = &acc_grp_cache[acc_user_hash(user)];
364
365 /* Cache miss (absent, expired, or different user) → refresh from NSS. A
366 * breaker-open or no-group refresh yields "no supplementary groups" now,
367 * without rendering any stale entry. */
368 if (!(e->user[0] != '\0' && e->expiry > now && ngx_strcmp(e->user, user) == 0)) {
369 if (acc_cache_refresh(e, user, now) != NGX_OK) {
370 return NULL;
371 }
372 }
373
374 if (e->count == 0) {
375 return NULL;
376 }
377 return acc_cache_to_array(pool, e);
378}
