18/* Per-export instance state (inst->state): the resolved origin params. For a
19 * cache-constructed instance `conf` is the real export conf; for a registry-built
20 * PRIMARY backend (a stream/http export with no stream conf) `synth` owns a
21 * minimal conf carrying just cache_origin_host/port/tls (+ trusted_ca, TLS-only). */

36/* ---- shared origin-open machinery (phase-79 file-size split) --------------
37 *
38 * The per-open object state, the open request bundle, and the connect →
39 * bootstrap → kXR_open orchestrator live in sd_xroot.c (the driver core) and
40 * are shared with the object I/O + open + stat path (sd_xroot_io.c). The
41 * credential-copy and teardown helpers are also reused by the staged-write
42 * path (sd_xroot_staged.c). Driver-private: not part of the public surface. */

44/* Per-open state: a live origin connection + open file handle + the synthetic
45 * fill-task the origin functions need (conf + clean_path + error scratch). */
46typedef struct {

54/* Inputs to a single origin file open. Bundling the seven loose arguments into a
55 * struct keeps sd_xroot_origin_open() and its callers (open_common, stat,
56 * stat_cred) below the parameter-count gate. `size_out`/`err_out` may be NULL;
57 * the remaining fields are required. */
58typedef struct {
59 ngx_stream_brix_srv_conf_t *conf; /* origin params (real or synthetic) */
60 const brix_sd_cred_t *cred; /* per-user credential, or NULL */
61 const char *path; /* remote path to open */
62 int want_write; /* 1 = writable open, 0 = read open */
63 mode_t mode; /* create mode (write opens only) */
64 off_t *size_out; /* out: file size (may be NULL) */
65 int *err_out; /* out: errno on failure (may be NULL) */
66} sd_xroot_origin_open_req_t;
