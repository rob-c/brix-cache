1/*
2 * xrootdfs_main.c — unified `xrootdfs` FUSE front-end (multi-call, like `xrd`).
3 *
4 * WHAT: A single `xrootdfs` binary that runs EITHER driver: the default
5 * async/resilient one (root:// + http(s)/WebDAV), or — with `--legacy` —
6 * the simple synchronous one. One tool, one man page, a mode flag.
7 * WHY: The two drivers were separate binaries; a single binary with `--legacy`
8 * is simpler to ship and matches how the `xrd` front-end selects behavior.
9 * HOW: Scan argv for a leading `--legacy` token, strip it, and dispatch to the
10 * matching driver entry point (xrootdfs_drivers.h). The legacy driver tags
11 * its own FUSE subtype (fuse.xrootdfs_legacy) so mount listings still tell
12 * the two modes apart.
13 *
14 * Clean-room: only the two in-tree driver entry points + libc.
15 */

22{
23 /* A leading `--legacy` (before the positional endpoint) selects the simple
24 * synchronous driver. We accept it anywhere among the leading options and
25 * strip it so the chosen driver sees its own native argv. */
26 for (int i = 1; i < argc; i++) {
27 if (strcmp(argv[i], "--legacy") == 0) {
28 for (int j = i; j < argc - 1; j++) {
29 argv[j] = argv[j + 1];
30 }
31 argv[--argc] = NULL;
