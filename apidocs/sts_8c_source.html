1/*
2 * sts.c — AWS STS AssumeRole / GetSessionToken client (phase-70 §5.5).
3 *
4 * WHAT: exchange the node's long-lived backend S3 *service* credential for
5 * short-lived temporary credentials scoped to the inbound principal.
6 * WHY: an inbound S3 request proves knowledge of a secret via SigV4 but never
7 * transmits it, so the origin cannot be handed the caller's key. The node
8 * instead calls STS with its own service key, tagging the call with the
9 * caller's identity (RoleSessionName), and forwards the resulting
10 * temporary (ak, sk, session) to the origin. Pure passthrough is
11 * impossible by design; this is the EXCHANGE path.
12 * HOW: this file is the orchestrator — it validates the config, derives the
13 * identity/timestamps, then drives the pure request builder (sts_sign.c)
14 * and the transport + XML parser (sts_http.c) in a flat phase sequence:
15 * prepare → perform → finish. Secrets and session tokens are copied into
16 * the caller's pool and never logged. The SigV4 signing and the HTTP/XML
17 * work live in the sibling files (declared in sts_internal.h).
18 *
19 * File split (phase-79):
20 * - sts.c — this orchestrator + small identity/URL/TTL helpers
21 * - sts_sign.c — canonical query construction + SigV4 signing
22 * - sts_http.c — libcurl transport + libxml2 response parsing
23 * - sts_internal.h — the structs, size cap and seam entry points they share
24 */

38/*
39 * Copy a NUL-terminated C string into `dst` as a pool-allocated ngx_str_t.
40 * A NULL source yields an empty (but non-NULL .data) string so callers can
41 * always treat *session as valid. Returns NGX_OK / NGX_ERROR (OOM).
42 */
43static ngx_int_t
44sts_pool_cstr(ngx_pool_t *pool, ngx_str_t *dst, const char *src)
45{
46 size_t len = (src != NULL) ? ngx_strlen(src) : 0;
47 u_char *p;
48
49 p = ngx_pnalloc(pool, len + 1);
50 if (p == NULL) {
51 return NGX_ERROR;
52 }
53 if (len > 0) {
54 ngx_memcpy(p, src, len);
55 }
56 p[len] = '\0';
57 dst->data = p;
58 dst->len = len;
59 return NGX_OK;
60}

83/*
84 * sts_rsn_char_ok — is `c` a byte AWS permits verbatim in a RoleSessionName?
85 *
86 * WHAT: predicate for the AWS-permitted RoleSessionName alphabet
87 * ([A-Za-z0-9+=,.@:/-]); every other byte is mapped to '_' by the caller.
88 * WHY: isolating the alphabet check keeps the sanitising loop's cyclomatic
89 * complexity low and puts the (byte-frozen) allowed set in one place.
90 * HOW: a punctuation lookup string plus the three alphanumeric ranges; returns
91 * 1 for permitted bytes, 0 otherwise. Semantics identical to the former
92 * inline `||` chain.
93 */
94static int
95sts_rsn_char_ok(char c)
96{
97 if ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
98 || (c >= '0' && c <= '9'))
99 {
100 return 1;
101 }
102 return strchr("+=,.@:/-", c) != NULL && c != '\0';
103}

106/*
107 * sts_rsn_source — pick the RoleSessionName source string from the identity:
108 * prefer the subject (JWT sub or S3 access key), fall back to the DN, else
109 * "anonymous". Returns a NUL-terminated C string (never NULL).
110 */
111static const char *
112sts_rsn_source(const brix_identity_t *id)
113{
114 if (id != NULL && id->subject.len > 0) {
115 return (const char *) id->subject.data;
116 }
117 if (id != NULL && id->dn.len > 0) {
118 return (const char *) id->dn.data;
119 }
120 return "anonymous";
121}

124/*
125 * Derive a RoleSessionName from the inbound identity, sanitised to the
126 * AWS-permitted alphabet with everything else mapped to '_', and truncated to
127 * 64 chars. Written NUL-terminated into out[65].
128 */
129static void
130sts_role_session_name(const brix_identity_t *id, char out[65])
131{
132 const char *src = sts_rsn_source(id);
133 size_t i;
134
135 for (i = 0; i < 64 && src[i] != '\0'; i++) {
136 out[i] = sts_rsn_char_ok(src[i]) ? src[i] : '_';
137 }
138 if (i == 0) {
139 ngx_memcpy(out, "anonymous", 9);
140 i = 9;
141 }
142 out[i] = '\0';
143}

146/*
147 * Parse a URL of the form "scheme://host[:port][/...]" into a pool-copied,
148 * NUL-terminated host[:port] authority for the SigV4 "host" header. Returns
149 * NGX_OK / NGX_ERROR on a malformed endpoint.
150 */
151static ngx_int_t
152sts_host_from_url(ngx_pool_t *pool, const ngx_str_t *url, char **host_out)
153{
154 const char *s = (const char *) url->data;
155 const char *p;
156 const char *end;
157 size_t n;
158 char *h;
159
160 p = strstr(s, "://");
161 p = (p != NULL) ? p + 3 : s;
162 for (end = p; *end != '\0' && *end != '/'; end++) { /* to path or NUL */ }
163 n = (size_t) (end - p);
164 if (n == 0) {
165 return NGX_ERROR;
166 }
167 h = ngx_pnalloc(pool, n + 1);
168 if (h == NULL) {
169 return NGX_ERROR;
170 }
171 ngx_memcpy(h, p, n);
172 h[n] = '\0';
173 *host_out = h;
174 return NGX_OK;
175}

182/*
183 * Format the current UTC time as SigV4's "YYYYMMDDTHHMMSSZ" into amzdate[17]
184 * and its "YYYYMMDD" datestamp prefix into datestamp[9]. Returns NGX_OK, or
185 * NGX_ERROR if the clock cannot be read.
186 */
187static ngx_int_t
188sts_now(char amzdate[17], char datestamp[9])
189{
190 time_t t = time(NULL);
191 struct tm g;
192
193 if (gmtime_r(&t, &g) == NULL) {
194 return NGX_ERROR;
195 }
196 if (strftime(amzdate, 17, "%Y%m%dT%H%M%SZ", &g) != 16) {
197 return NGX_ERROR;
198 }
199 ngx_memcpy(datestamp, amzdate, 8);
200 datestamp[8] = '\0';
201 return NGX_OK;
202}

207 *
208 * WHAT: check the mandatory pointers and the config completeness the exchange
209 * needs (endpoint/region/service key). WHY: fail closed with the exact

213static ngx_int_t

217 if (pool == NULL || cf == NULL || out->ak == NULL || out->sk == NULL

233/*
234 * sts_prepare — populate `req` and render the fully-signed request URL.
235 *
236 * WHAT: derive the clamped TTL, timestamps, host, RoleSessionName and encoded
237 * X-Amz-Credential, then build+sign the canonical query and format the
238 * final "<endpoint>?<signed-query>" into url[urlsz].
239 * WHY: isolates all pure request construction from transport, keeping the
240 * entry point a linear phase sequence. Byte output is unchanged — the same
241 * calls in the same order, only the {cf,host,...} cluster now travels in
242 * `req` instead of as loose args.
243 * HOW: clock/host/credential each early-return on overflow or failure with the
244 * original error string; then sts_build_action_qs → sts_sign_query →
245 * url. NGX_OK / NGX_ERROR.
246 */
247static ngx_int_t
248sts_prepare(ngx_pool_t *pool, sts_req_t *req, char *url, size_t urlsz,
249 ngx_log_t *log)
250{
251 const brix_s3_sts_conf_t *cf = req->cf;
252 char action_qs[2048];
253 char signed_qs[2560];
254 char *host = NULL;
255 int n;
256
257 req->ttl = sts_clamp_ttl(cf->ttl_secs);
258
259 if (sts_now(req->amzdate, req->datestamp) != NGX_OK) {
260 ngx_log_error(NGX_LOG_ERR, log, 0, "brix_sts: clock read failed");
261 return NGX_ERROR;
262 }
263 if (sts_host_from_url(pool, &cf->endpoint, &host) != NGX_OK) {
264 ngx_log_error(NGX_LOG_ERR, log, 0,
265 "brix_sts: malformed STS endpoint");
266 return NGX_ERROR;
267 }
268 req->host = host;
269
270 sts_role_session_name(req->id, req->rsn);
271
272 /* X-Amz-Credential value with the scope slashes percent-encoded (%2F). */
273 n = ngx_snprintf((u_char *) req->credential, sizeof(req->credential),
274 "%V%%2F%s%%2F%V%%2Fsts%%2Faws4_request%Z",
275 &cf->svc_ak, req->datestamp, &cf->region)
276 - (u_char *) req->credential;
277 if (n <= 0 || (size_t) n >= sizeof(req->credential)) {
278 ngx_log_error(NGX_LOG_ERR, log, 0, "brix_sts: credential too long");
279 return NGX_ERROR;
280 }
281
282 if (sts_build_action_qs(req, action_qs, sizeof(action_qs)) != NGX_OK) {
283 ngx_log_error(NGX_LOG_ERR, log, 0, "brix_sts: query build overflow");
284 return NGX_ERROR;
285 }
286
287 if (sts_sign_query(req, action_qs, signed_qs, sizeof(signed_qs)) != NGX_OK)
288 {
289 ngx_log_error(NGX_LOG_ERR, log, 0, "brix_sts: SigV4 signing failed");
290 return NGX_ERROR;
291 }
292
293 n = ngx_snprintf((u_char *) url, urlsz, "%V?%s%Z",
294 &cf->endpoint, signed_qs) - (u_char *) url;
295 if (n <= 0 || (size_t) n >= urlsz) {
296 ngx_log_error(NGX_LOG_ERR, log, 0, "brix_sts: URL too long");
297 return NGX_ERROR;
298 }
299 return NGX_OK;
300}

303/*
304 * sts_perform — fetch the signed URL and capture a successful (HTTP 200) body.
305 *
306 * WHAT: allocate the bounded response buffer, GET the URL, and require a 200.
307 * WHY: separates transport + status policy from parsing; the non-200 branch
308 * keeps the original "AssumeRole for <rsn> returned HTTP <n>" message.
309 * HOW: fills `resp` (its buffer pool-allocated) and early-returns NGX_ERROR on
310 * transport failure or a non-200 status; NGX_OK leaves the body in `resp`.
311 */
312static ngx_int_t
313sts_perform(ngx_pool_t *pool, const char *url, const char *rsn,
314 sts_resp_t *resp, ngx_log_t *log)
315{
316 long http_status = 0;
317
318 resp->buf = ngx_pnalloc(pool, BRIX_STS_RESP_MAX);
319 if (resp->buf == NULL) {
320 return NGX_ERROR;
321 }
322 resp->len = 0;
323 resp->cap = BRIX_STS_RESP_MAX;
324
325 if (sts_http_get(url, resp, &http_status, log) != NGX_OK) {
326 return NGX_ERROR;
327 }
328 if (http_status != 200) {
329 ngx_log_error(NGX_LOG_ERR, log, 0,
330 "brix_sts: STS AssumeRole for \"%s\" returned HTTP %l",
331 rsn, http_status);
332 return NGX_ERROR;
333 }
334 return NGX_OK;
335}

338/*
339 * sts_finish — parse the STS XML body and pool-copy the credentials out.
340 *
341 * WHAT: extract AccessKeyId/SecretAccessKey/SessionToken and copy each into the
342 * caller's ngx_str_t out slots. WHY: the last phase — parse then persist,
343 * failing closed on either. HOW: sts_parse_response into fixed buffers,
344 * then sts_pool_cstr for each; NGX_OK / NGX_ERROR. Secrets never logged.
345 */
346static ngx_int_t
347sts_finish(ngx_pool_t *pool, const sts_resp_t *resp,
348 const brix_s3_sts_out_t *out, ngx_log_t *log)
349{
350 char r_ak[256];
351 char r_sk[512];
352 char r_session[8192];
353 sts_creds_buf_t creds = {
354 r_ak, sizeof(r_ak),
355 r_sk, sizeof(r_sk),
356 r_session, sizeof(r_session)
357 };
358
359 if (sts_parse_response(resp->buf, resp->len, &creds, log) != NGX_OK) {
360 return NGX_ERROR;
361 }
362
363 if (sts_pool_cstr(pool, out->ak, r_ak) != NGX_OK
364 || sts_pool_cstr(pool, out->sk, r_sk) != NGX_OK
365 || sts_pool_cstr(pool, out->session, r_session) != NGX_OK)
366 {
367 return NGX_ERROR;
368 }
369 return NGX_OK;
370}

374brix_s3_sts_assume(ngx_pool_t *pool, const brix_identity_t *id,

401 return NGX_OK;
