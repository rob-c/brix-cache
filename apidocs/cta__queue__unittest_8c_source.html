96static void test_journal_round_trip(void)
97{
98 const char *path = "/tmp/cta_queue_ut.journal";
99 uint64_t a_id, b_id;
100 remove(path);
101
102 /* first run: submit two requests, advance one to COMPLETE */
103 {
104 brix_cta_queue_t *q = cta_queue_create();
105 cta_request_t r = mk(CTA_OP_ARCHIVE);
106 cta_req_t *a, *b;
107 CHECK(cta_queue_open_journal(q, path) == 0);
108 a = cta_queue_submit(q, &r, "alice");
109 b = cta_queue_submit(q, &r, "bob");
110 a_id = a->id; b_id = b->id;
111 cta_queue_transition(a, CTA_ST_QUEUED);
112 cta_queue_transition(a, CTA_ST_ACTIVE);
113 cta_queue_transition(a, CTA_ST_COMPLETE);
114 cta_queue_destroy(q);
115 }
116 /* second run: replay restores both entries with their latest state */
117 {
118 brix_cta_queue_t *q = cta_queue_create();
119 cta_req_t *a, *b;
120 CHECK(cta_queue_open_journal(q, path) == 0);
121 a = cta_queue_find(q, a_id);
122 b = cta_queue_find(q, b_id);
123 CHECK(a != NULL && a->state == CTA_ST_COMPLETE);
124 CHECK(b != NULL && b->state == CTA_ST_SUBMITTED);
125 CHECK(strcmp(a->owner, "alice") == 0);
126 /* next_id advanced past the replayed ids */
127 CHECK(cta_queue_submit(q, &(cta_request_t){0}, "x")->id > b_id);
128 cta_queue_destroy(q);
129 }
130 remove(path);
131}
