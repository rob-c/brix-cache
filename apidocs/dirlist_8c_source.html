1/*
2 * Async kXR_dirlist via nginx thread pool.
3 *
4 * WHAT: Offloads the entire directory iteration + stat + checksum work to a
5 * thread-pool worker, freeing the nginx event loop to serve other
6 * connections while a large or slow-media directory is being scanned.
7 *
8 * WHY: directory iteration + fstatat(2) + fgetxattr(2)/hash-on-miss can block for tens
9 * to hundreds of milliseconds on spinning disk or network-backed storage.
10 * Running these on the event loop stalls all connections sharing that
11 * worker process. Posting to the nginx thread pool lets I/O block on a
12 * dedicated OS thread instead.
13 *
14 * HOW:
15 * Main thread (brix_handle_dirlist, handler.c):
16 * 1. Auth, path resolution and algorithm validation run synchronously.
17 * 2. A response buffer is allocated from c->pool (large block → ngx_pfree
18 * can release it after drain).
19 * 3. Fields are copied into brix_dirlist_aio_t; the task is posted.
20 * 4. Returns NGX_OK immediately; ctx->state = XRD_ST_AIO.
21 *
22 * Worker thread (brix_dirlist_aio_thread):
23 * - Opens the directory independently (no nginx state touched).
24 * - Iterates entries: skip "." / ".." / unsafe names.
25 * - Calls fstatat, brix_dirlist_checksum_token (xattr fast path or
26 * full hash), and builds the complete XRootD wire response directly into
27 * t->response. Wire format: one or more kXR_oksofar frames followed by
28 * a single kXR_ok frame.
29 * - No pool allocation in this function — only POSIX syscalls and pure
30 * computation.
31 *
32 * Done callback (brix_dirlist_aio_done, main thread):
33 * - Checks ctx->destroyed; on error sends kXR_IOError.
34 * - On success: wraps t->response in a memory ngx_buf_t + ngx_chain_t and
35 * calls brix_queue_response_chain(ctx, c, chain, t->response) so the
36 * buffer is freed via brix_release_read_buffer (ngx_pfree) once the
37 * chain has been fully drained.
38 * - Calls brix_aio_resume to re-arm the read event.
39 *
40 * Fallback: if the thread pool is not configured (conf->common.thread_pool == NULL)
41 * or the queue is full, brix_handle_dirlist falls back to the synchronous
42 * loop already present in handler.c — behaviour is identical to pre-AIO.
43 *
44 * Truncation: if the directory listing would exceed BRIX_DIRLIST_AIO_RESPONSE_MAX
45 * (4 MiB), the thread sets io_errno = E2BIG and the done callback returns
46 * kXR_IOError. The operator can increase the limit or rely on the sync path.
47 */

67/*
68 * brix_dirlist_aio_thread — worker: execute confined dirlist VFS job.
69 *
70 * THREAD SAFETY RULES (enforced here):
71 * - NO pool allocation (ngx_palloc / ngx_pcalloc / ngx_palloc_large).
72 * - NO nginx connection state access (c->log is stale once on a thread).
73 * - Use only the VFS worker-safe execution core and task-owned fields.
74 */
75void
76brix_dirlist_aio_thread(void *data, ngx_log_t *log)
77{
78 ngx_thread_task_t *task = data;
79 brix_dirlist_aio_t *t = task->ctx;
80 brix_vfs_job_t job;
81
82 t->response_len = 0;
83 t->io_errno = 0;
84 t->err_msg[0] = '\0';
85
86 brix_vfs_job_opendir_init(&job, t->dirfd, t->response, t->response_cap,
87 t->streamid, t->want_stat, t->want_cksum,
88 t->resolved, t->cksum_algo, log,
89 t->err_msg, sizeof(t->err_msg));
90 brix_vfs_io_execute(&job);
91
92 t->response_len = job.out_size;
93 t->io_errno = job.io_errno;
94
95 if (t->dirfd >= 0) {
96 close(t->dirfd);
97 t->dirfd = -1;
98 }
99}

101/*
102 * brix_dirlist_aio_done — main-thread completion callback.
103 *
104 * Fires on the nginx event loop after the thread pool posts the result.
105 * Responsibilities:
106 * 1. Guard against stale connection (ctx->destroyed check).
107 * 2. On I/O error: send kXR_error frame, release response buffer, resume.
108 * 3. On success: wrap t->response in a memory chain and queue it.
109 * The buffer is released via brix_release_read_buffer (ngx_pfree)
110 * after the chain has been fully drained.
111 * 4. Call brix_aio_resume() to re-arm the appropriate event.
112 */
113void
114brix_dirlist_aio_done(ngx_event_t *ev)
115{
116 ngx_thread_task_t *task = ev->data;
117 brix_dirlist_aio_t *t = task->ctx;
118 brix_ctx_t *ctx = t->ctx;
119 ngx_connection_t *c = t->c;
120
121 /* Restore connection state; abort silently if connection was destroyed. */
122 if (!brix_aio_restore_request(ctx, t->streamid)) {
123 return;
124 }
125
126 if (t->io_errno) {
127 uint16_t xrd_err;
128 char msg[128];
129
130 xrd_err = brix_kxr_from_errno(t->io_errno);
131
132 snprintf(msg, sizeof(msg), "dirlist: %s",
133 t->err_msg[0] ? t->err_msg : strerror(t->io_errno));
134 ngx_log_error(NGX_LOG_ERR, c->log, 0,
135 "brix: dirlist AIO failed: %s", msg);
136 brix_log_access(ctx, c, "DIRLIST", t->resolved, "-",
137 0, xrd_err, msg, 0);
138 BRIX_OP_ERR(ctx, BRIX_OP_DIRLIST);
139
140 brix_release_read_buffer(ctx, c, t->response);
141 t->response = NULL;
142 brix_send_error(ctx, c, xrd_err, msg);
143 brix_aio_resume(c);
144 return;
145 }
146
147 brix_log_access(ctx, c, "DIRLIST", t->resolved,
148 t->want_cksum ? "dcksm"
149 : (t->want_stat ? "stat" : "-"),
150 1, 0, NULL, 0);
151 BRIX_OP_OK(ctx, BRIX_OP_DIRLIST);
152
153 /*
154 * The worker already built a contiguous wire response containing every
155 * kXR_oksofar frame plus the final kXR_ok frame. Queue it as a plain
156 * buffer so an empty final body still sends its 8-byte header and so the
157 * owned backing allocation is retained across partial writes.
158 */
159 brix_queue_response_base(ctx, c, t->response, t->response_len,
160 t->response);
161 if (ctx->state != XRD_ST_SENDING) {
162 brix_release_read_buffer(ctx, c, t->response);
163 }
164
165 brix_aio_resume(c);
166}
