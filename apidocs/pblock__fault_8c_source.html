20/* Instance-level rule cache. Refreshed only when the ctl epoch advances; the
21 * mutex guards a refresh (rare — a metadata boundary) against concurrent worker
22 * threads sharing the instance. */

36static int
37errno_by_name(const char *s, size_t len)
38{
39 static const struct { const char *n; int e; } tab[] = {
40 { "EIO", EIO }, { "ENOSPC", ENOSPC }, { "EACCES", EACCES },
41 { "EROFS", EROFS }, { "EAGAIN", EAGAIN }, { "EPERM", EPERM },
42 { "EDQUOT", EDQUOT }, { "EBUSY", EBUSY },
43 };
44 size_t i;
45
46 for (i = 0; i < sizeof(tab) / sizeof(tab[0]); i++) {
47 if (strlen(tab[i].n) == len && memcmp(tab[i].n, s, len) == 0) {
48 return tab[i].e;
49 }
50 }
51 return 0;
52}

54/* Parse a "fault.<op>" value: space-separated key=val tokens
55 * (errno=NAME, after_bytes=N, short=N) into *out (zeroed first). */
56static void
57parse_fault(const char *v, pblock_rule_t *out)
58{
59 const char *p = v;
60
61 memset(out, 0, sizeof(*out));
62 out->after_bytes = -1;
63 out->short_to = -1;
64 if (v == NULL) {
65 return;
66 }
67 while (*p != '\0') {
68 while (*p == ' ') {
69 p++;
70 }
71 if (*p == '\0') {
72 break;
73 }
74 {
75 const char *tok_end = p;
76 const char *eq;
77
78 while (*tok_end != '\0' && *tok_end != ' ') {
79 tok_end++;
80 }
81 eq = memchr(p, '=', (size_t) (tok_end - p));
82 if (eq != NULL) {
83 size_t klen = (size_t) (eq - p);
84 const char *val = eq + 1;
85 size_t vlen = (size_t) (tok_end - val);
86
87 if (klen == 5 && memcmp(p, "errno", 5) == 0) {
88 out->inj_errno = errno_by_name(val, vlen);
89 } else if (klen == 11 && memcmp(p, "after_bytes", 11) == 0) {
90 out->after_bytes = strtoll(val, NULL, 10);
91 } else if (klen == 5 && memcmp(p, "short", 5) == 0) {
92 out->short_to = (int) strtol(val, NULL, 10);
93 }
94 }
95 p = tok_end;
96 }
97 }
98}

140/* Reload the cached rules from ctl (caller holds the mutex). */
141static void
142lab_refresh_locked(pblock_lab_state_t *ls)
143{
144 char buf[256];
145
146 memset(&ls->r, 0, sizeof(ls->r));
147 memset(&ls->w, 0, sizeof(ls->w));
148 ls->r.after_bytes = ls->w.after_bytes = -1;
149 ls->r.short_to = ls->w.short_to = -1;
150
151 if (pblock_ctl_get(ls->cat, "fault.pread", buf, sizeof(buf)) == 1) {
152 parse_fault(buf, &ls->r);
153 }
154 if (pblock_ctl_get(ls->cat, "fault.pwrite", buf, sizeof(buf)) == 1) {
155 parse_fault(buf, &ls->w);
156 }
157 ls->r.rate_bps = ctl_int(ls->cat, "shape.read_bps");
158 ls->w.rate_bps = ctl_int(ls->cat, "shape.write_bps");
159 ls->open_ms = (int) ctl_int(ls->cat, "shape.open_ms");
160 ls->crash_at[0] = '\0';
161 (void) pblock_ctl_get(ls->cat, "crash.at", ls->crash_at,
162 sizeof(ls->crash_at));
163 ls->loaded = 1;
164}

185pblock_lab_obj_t *
186pblock_lab_snapshot(pblock_lab_state_t *ls, const char *path)
187{
188 pblock_lab_obj_t *lo;
189 int64_t epoch;
190 int open_ms;
191
192 (void) path; /* Wave A: rules are export-global, not per-path */
193 if (ls == NULL) {
194 return NULL;
195 }
196
197 pthread_mutex_lock(&ls->mtx);
198 epoch = pblock_ctl_epoch(ls->cat);
199 if (!ls->loaded || (epoch >= 0 && epoch != ls->epoch)) {
200 lab_refresh_locked(ls);
201 ls->epoch = epoch;
202 }
203 if (rule_inert(&ls->r) && rule_inert(&ls->w) && ls->open_ms == 0) {
204 pthread_mutex_unlock(&ls->mtx);
205 return NULL;
206 }
207 lo = calloc(1, sizeof(*lo));
208 if (lo == NULL) {
209 pthread_mutex_unlock(&ls->mtx);
210 return NULL;
211 }
212 lo->r = ls->r;
213 lo->w = ls->w;
214 open_ms = ls->open_ms;
215 pthread_mutex_unlock(&ls->mtx);
216
217 sleep_ms(open_ms); /* F8 open delay (outside the lock) */
218 return lo;
219}

229int
230pblock_lab_io_gate(pblock_lab_obj_t *lo, int is_write, size_t *len, off_t off)
231{
232 const pblock_rule_t *r = is_write ? &lo->w : &lo->r;
233 int64_t *cum = is_write ? &lo->wbytes : &lo->rbytes;
234
235 (void) off;
236
237 /* F8 pacing: sleep proportional to the requested transfer. Capped at 5s so a
238 * pathological rate can never wedge an AIO-pool thread indefinitely. */
239 if (r->rate_bps > 0 && *len > 0) {
240 uint64_t ns = (uint64_t) *len * 1000000000ULL / (uint64_t) r->rate_bps;
241 struct timespec ts;
242
243 if (ns > 5000000000ULL) {
244 ns = 5000000000ULL;
245 }
246 ts.tv_sec = (time_t) (ns / 1000000000ULL);
247 ts.tv_nsec = (long) (ns % 1000000000ULL);
248 (void) nanosleep(&ts, NULL);
249 }
250
251 /* F1 short op: clamp the transfer length. */
252 if (r->short_to >= 0 && (size_t) r->short_to < *len) {
253 *len = (size_t) r->short_to;
254 }
255
256 /* F1 error injection: fire once the cumulative threshold is reached (or
257 * immediately when after_bytes < 0). */
258 if (r->inj_errno != 0) {
259 if (r->after_bytes < 0 || *cum >= r->after_bytes) {
260 return r->inj_errno;
261 }
262 }
263
264 *cum += (int64_t) *len;
265 return 0;
266}

270void
271pblock_lab_crash(pblock_lab_state_t *ls, const char *point)
272{
273 int64_t epoch;
274 int hit;
275
276 if (ls == NULL || point == NULL) {
277 return;
278 }
279
280 pthread_mutex_lock(&ls->mtx);
281 epoch = pblock_ctl_epoch(ls->cat);
282 if (!ls->loaded || (epoch >= 0 && epoch != ls->epoch)) {
283 lab_refresh_locked(ls);
284 ls->epoch = epoch;
285 }
286 hit = (ls->crash_at[0] != '\0' && strcmp(ls->crash_at, point) == 0);
287 pthread_mutex_unlock(&ls->mtx);
288
289 if (hit) {
290 /* Simulate a power loss at this durability boundary: the worker dies
291 * without flushing anything further; nginx's master respawns it. The
292 * post-restart catalog↔blocks residue is what pblock-fsck classifies. */
293 _exit(PBLOCK_CRASH_EXIT);
294 }
295}
