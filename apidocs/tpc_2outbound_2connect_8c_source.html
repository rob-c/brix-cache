1/* File: connect.c — DNS resolution and TCP connect to TPC source server
2 * WHAT: Resolves the remote XRootD origin host via getaddrinfo, validates each candidate address against SSRF policy rules (allow_local/allow_private), creates a blocking socket with configurable receive/send timeouts, then performs non-blocking connect followed by poll-based wait for connection completion. Also provides brix_tpc_check_src_policy as an SSRF preflight wrapper that resolves host+port without creating the socket — used before kXR_open destination file creation to validate source addresses early.
3 *
4 * WHY: Native TPC pull requires nginx to establish a TCP connection to the remote root:// origin server before it can send handshake frames and read the file. DNS resolution must iterate over all addrinfo candidates (IPv4/IPv6) because some may be unreachable; SSRF policy validation prevents connecting to localhost or private ranges when configured to reject them. Non-blocking connect with poll-based wait avoids blocking the nginx event loop while still respecting configurable timeout limits. The preflight check enables early source address validation before allocating destination file resources.
5 *
6 * HOW: getaddrinfo(src_host, port_str, hints SOCK_STREAM+AF_UNSPEC) → iterate addrinfo candidates → brix_net_target_check_addr with SSRF policy → socket(family,socktype,proto) → setsockopt SO_RCVTIMEO/SO_SNDTIMEO → fcntl O_NONBLOCK → connect() → if EINPROGRESS then poll POLLOUT with TPC_CONNECT_TIMEOUT_SEC → getsockopt SO_ERROR to verify zero → restore original flags. Returns connected fd or -1 on failure. Preflight: brix_net_target_check_dns resolves host+port without creating socket, returns 0/-1.
7 * */

25/* WHAT: DNS resolve + TCP connect to TPC origin — getaddrinfo → SSRF policy check per candidate → socket with SO_RCVTIMEO/SO_SNDTIMEO → non-blocking connect via poll. Returns connected fd or -1. */
26
27int
28tpc_connect(brix_tpc_pull_t *t)
29{
30 struct addrinfo hints, *res, *rp;
31 char port_str[16];
32 int fd = -1;
33 uint16_t src_port;
34
35 src_port = t->src_port ? t->src_port : 1094;
36 snprintf(port_str, sizeof(port_str), "%u", (unsigned) src_port);
37
38 ngx_memzero(&hints, sizeof(hints));
39 hints.ai_socktype = SOCK_STREAM;
40 hints.ai_family = AF_UNSPEC;
41
42 if (getaddrinfo(t->src_host, port_str, &hints, &res) != 0) {
43 snprintf(t->err_msg, sizeof(t->err_msg),
44 "TPC DNS resolution failed for %s", t->src_host);
45 return -1;
46 }
47
48 for (rp = res; rp != NULL; rp = rp->ai_next) {
49 brix_net_target_policy_t policy;
50 char ssrf_err[128];
51
52 ngx_memzero(&policy, sizeof(policy));
53 policy.allow_local = t->conf->tpc_allow_local;
54 policy.allow_private = t->conf->tpc_allow_private;
55
56 if (brix_net_target_check_addr(rp->ai_addr, &policy,
57 ssrf_err, sizeof(ssrf_err))
58 != NGX_OK)
59 {
60 snprintf(t->err_msg, sizeof(t->err_msg),
61 "TPC source host %s: %s", t->src_host, ssrf_err);
62 freeaddrinfo(res);
63 return -1;
64 }
65
66 fd = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
67 if (fd < 0) {
68 continue;
69 }
70
71 brix_apply_socket_io_timeouts(fd, TPC_IO_TIMEOUT_SEC);
72
73 /*
74 * SciTags (phase-34): stamp the IPv6 flow label on the OUTBOUND pull
75 * socket before connecting (codes resolved on the event loop in
76 * start_pull). No-op on IPv4 / when not marked; fail-open.
77 */
78 if (t->pmark_exp != 0) {
79 (void) brix_pmark_flowlabel_apply_addr(fd, rp->ai_addr,
80 rp->ai_addrlen, t->pmark_exp, t->pmark_act, ngx_cycle->log);
81 }
82
83 if (brix_connect_fd_deadline(fd, rp->ai_addr, rp->ai_addrlen,
84 TPC_CONNECT_TIMEOUT_SEC * 1000) == 0)
85 {
86 break;
87 }
88
89 close(fd);
90 fd = -1;
91 }
92 freeaddrinfo(res);
93
94 if (fd < 0) {
95 snprintf(t->err_msg, sizeof(t->err_msg),
96 "TPC connect to %s failed", t->src_host);
97 return -1;
98 }
99
100 return fd;
101}

126 return -1;
127 }
128
129 ngx_memzero(&target, sizeof(target));

132 target.port = src_port ? src_port : 1094;

137 policy.allow_private = allow_private;

140 err_msg, err_msg_sz) != NGX_OK)
141 {
142 return -1;
143 }
144
145 return 0;
