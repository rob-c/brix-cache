19/* Case-insensitive: does the header line start with `name`? */
20static int
21hdr_is(const unsigned char *line, size_t len, const char *name)
22{
23 size_t nl = strlen(name);
24 if (len < nl) {
25 return 0;
26 }
27 for (size_t i = 0; i < nl; i++) {
28 if (tolower(line[i]) != tolower((unsigned char) name[i])) {
29 return 0;
30 }
31 }
32 return 1;
33}

44/* Emit a Transfer-Encoding header line with its name obfuscated per `mode`. */
45static void
46emit_te_obfuscated(unsigned char *out, size_t cap, size_t *o,
47 const unsigned char *line, size_t llen, int mode, int lf)
48{
49 const unsigned char *colon = memchr(line, ':', llen);
50 if (!colon) {
51 emit_line(out, cap, o, line, llen, lf);
52 return;
53 }
54 size_t namelen = (size_t) (colon - line);
55 size_t rest = llen - namelen; /* colon + value */
56 if (mode == 1) { /* space before the colon */
57 fp_bufcat(out, cap, o, line, namelen);
58 fp_bufputc(out, cap, o, ' ');
59 fp_bufcat(out, cap, o, colon, rest);
60 } else if (mode == 2) { /* leading tab (line folding) */
61 fp_bufputc(out, cap, o, '\t');
62 fp_bufcat(out, cap, o, line, llen);
63 } else { /* '-' -> '_' inside the name */
64 for (size_t k = 0; k < namelen; k++) {
65 fp_bufputc(out, cap, o, line[k] == '-' ? '_' : line[k]);
66 }
67 fp_bufcat(out, cap, o, colon, rest);
68 }
69 fp_buf_eol(out, cap, o, lf);
70}

72/* Append the added/injected headers after the original header lines. */
73static void

95 fp_bufcat(out, cap, o, ": ", 2);
96 fp_bufcat(out, cap, o, c->inj_val, (size_t) c->inj_val_len);
97 fp_buf_eol(out, cap, o, lf);

102size_t
103fp_http_rewrite(const unsigned char *in, size_t n,
104 unsigned char *out, size_t outcap,
105 const fp_http_cfg *c, fp_http_stats *st, int *applied)
106{
107 *applied = 0;
108 const unsigned char *end = NULL;
109 for (size_t i = 0; i + 3 < n; i++) {
110 if (in[i] == '\r' && in[i + 1] == '\n' &&
111 in[i + 2] == '\r' && in[i + 3] == '\n') {
112 end = in + i;
113 break;
114 }
115 }
116 if (!end) {
117 return 0; /* no header block here */
118 }
119 *applied = 1;
120 st->msgs++;
121 int lf = c->naked_lf;
122 size_t o = 0;
123 size_t hlen = (size_t) (end - in); /* header bytes, no terminator */
124
125 size_t ls = 0;
126 while (ls <= hlen) {
127 size_t le = ls;
128 while (le + 1 <= hlen && !(in[le] == '\r' && in[le + 1] == '\n')) {
129 le++;
130 }
131 if (le > hlen) {
132 le = hlen;
133 }
134 const unsigned char *line = in + ls;
135 size_t llen = le - ls;
136 if (llen > 0) {
137 if (c->obfuscate_te && hdr_is(line, llen, "transfer-encoding")) {
138 emit_te_obfuscated(out, outcap, &o, line, llen, c->obfuscate_te, lf);
139 st->te_obf++;
140 } else {
141 emit_line(out, outcap, &o, line, llen, lf);
142 }
143 }
144 if (le >= hlen) {
145 break;
146 }
147 ls = le + 2;
148 }
149
150 emit_added_headers(out, outcap, &o, c, st, lf);
151 fp_buf_eol(out, outcap, &o, lf); /* blank line: end of headers */
152 if (lf) {
153 st->lf_converted++;
154 }
155
156 const unsigned char *body = end + 4; /* body follows CRLFCRLF */
157 fp_bufcat(out, outcap, &o, body, (size_t) (in + n - body));
158
159 if (c->append_len > 0) {
160 fp_bufcat(out, outcap, &o, c->append, (size_t) c->append_len);
161 st->appended++;
162 }
163 return o;
164}
