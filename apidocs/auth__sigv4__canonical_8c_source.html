31/*
32 * qsort comparator implementing SigV4's "sort by name, then by value" ordering.
33 *
34 * SigV4 sorts on the byte values of the *encoded* params, but because percent-
35 * encoding is order-preserving for the unreserved-only alphabet used here, a raw
36 * byte compare of the decoded bytes yields the same ordering — so we compare the
37 * decoded buffers directly and avoid encoding twice.
38 *
39 * Names are not NUL-terminated to a common length, so a plain strcmp/memcmp over
40 * the longer length would read past one buffer. We instead memcmp() the common
41 * prefix (the shorter of the two lengths); only if that prefix is equal does the
42 * shorter string sort first (the length tiebreak below). This reproduces standard
43 * lexicographic ordering where "a" < "ab". Value comparison repeats the same
44 * prefix-then-length scheme as the secondary key.
45 */
46static int
47qparam_cmp(const void *a, const void *b)
48{
49 const qparam_t *pa = (const qparam_t *) a;
50 const qparam_t *pb = (const qparam_t *) b;
51 size_t ml = pa->name_len < pb->name_len ? pa->name_len : pb->name_len;
52 int r = memcmp(pa->name, pb->name, ml);
53 if (r != 0) return r;
54 /* common prefix equal -> shorter name is the lesser ("a" < "ab") */
55 if (pa->name_len != pb->name_len) {
56 return (pa->name_len < pb->name_len) ? -1 : 1;
57 }
58
59 ml = pa->value_len < pb->value_len ? pa->value_len : pb->value_len;
60 r = memcmp(pa->value, pb->value, ml);
61 if (r != 0) return r;
62 return (pa->value_len < pb->value_len) ? -1
63 : (pa->value_len > pb->value_len) ? 1 : 0;
64}

74/*
75 * Parse one "name=value" (or bare-flag "name") query segment spanning
76 * [segment_start, segment_end), with `equals` pointing at the first '=' in the
77 * segment or NULL when there is none, url-decoding the name and value into *p.
78 * Returns 1 if the parameter was stored, 0 if it should be skipped — a decode
79 * failure on either half, or the X-Amz-Signature parameter when skip_signature
80 * is set (it must not appear in the string it signs).
81 */
82static int
83sigv4_store_param(qparam_t *p, const u_char *segment_start,
84 const u_char *segment_end, const u_char *equals, ngx_flag_t skip_signature)
85{
86 size_t raw_name_len;
87 size_t raw_value_len;
88 const u_char *raw_value;
89
90 /* No '=' in the segment -> a bare flag parameter: the whole segment is the
91 * name and the value is empty (""), which SigV4 still canonicalises as
92 * "name=". When '=' is present, name is everything before it and value is
93 * everything after (excluding the '=' itself). */
94 raw_name_len = (size_t) ((equals ? equals : segment_end) - segment_start);
95 raw_value = equals ? equals + 1 : (u_char *) "";
96 raw_value_len = equals ? (size_t) (segment_end - equals - 1) : 0;
97
98 if (brix_http_urldecode(segment_start, raw_name_len,
99 (char *) p->name, sizeof(p->name),
100 BRIX_URLDECODE_PLUS_TO_SPACE |
101 BRIX_URLDECODE_REJECT_NUL) != BRIX_URLDECODE_OK)
102 {
103 return 0;
104 }
105 p->name_len = strlen((char *) p->name);
106
107 if (skip_signature
108 && p->name_len == 15
109 && ngx_strcmp(p->name, (u_char *) "X-Amz-Signature") == 0)
110 {
111 return 0;
112 }
113
114 if (brix_http_urldecode(raw_value, raw_value_len,
115 (char *) p->value, sizeof(p->value),
116 BRIX_URLDECODE_PLUS_TO_SPACE |
117 BRIX_URLDECODE_REJECT_NUL) != BRIX_URLDECODE_OK)
118 {
119 return 0;
120 }
121 p->value_len = strlen((char *) p->value);
122 return 1;
123}

125size_t
126build_canonical_qs(const u_char *qs, size_t qslen, ngx_flag_t skip_signature,
127 u_char *out, size_t outsz)
128{
129/* WHY: SigV4 canonical query string requires parameters sorted alphabetically
130 * by name then value, with names and values percent-encoded — this is the
131 * exact string that clients compute and sign. Skipping X-Amz-Signature from
132 * the canonical qs ensures the signature field itself is not part of what
133 * it signs (self-referential). Parameter count capped at 64 to bound stack
134 * allocation of params[] array. */
135 /* Parse query string into at most 64 parameters */
136 qparam_t params[64];
137 int param_count = 0;
138 const u_char *cursor = qs;
139 const u_char *end = qs + qslen;
140
141 while (cursor < end && param_count < 64) {
142 const u_char *segment_start = cursor;
143 const u_char *segment_end = cursor;
144 const u_char *equals = NULL;
145
146 /* Scan one "name=value" segment up to the next '&' (or end). Record the
147 * FIRST '=' only: a value may legitimately contain '=' bytes, so any
148 * '=' after the first belongs to the value, not the name/value split. */
149 while (segment_end < end && *segment_end != '&') {
150 if (*segment_end == '=' && equals == NULL) {
151 equals = segment_end;
152 }
153 segment_end++;
154 }
155
156 if (segment_start < segment_end
157 && sigv4_store_param(&params[param_count], segment_start,
158 segment_end, equals, skip_signature))
159 {
160 param_count++;
161 }
162
163 /* Step past the '&' delimiter; if segment_end already hit `end` there is
164 * no delimiter to skip, so leave cursor at end to terminate the loop. */
165 cursor = (segment_end < end) ? segment_end + 1 : segment_end;
166 }
167
168 qsort(params, (size_t) param_count, sizeof(params[0]), qparam_cmp);
169
170 /* Emit the sorted params as "enc(name)=enc(value)&..." into out[]. Every
171 * append is guarded against outsz so a short buffer truncates cleanly rather
172 * than overflowing; one trailing byte is always reserved for the final NUL
173 * written after the loop (hence the strict "< outsz", not "<="). */
174 size_t oi = 0;
175 u_char enc[1024];
176
177 for (int i = 0; i < param_count; i++) {
178 if (i > 0 && oi + 1 < outsz) {
179 out[oi++] = '&';
180 }
181 size_t n;
182 n = uriencode_param(params[i].name, params[i].name_len,
183 enc, sizeof(enc));
184 if (oi + n < outsz) {
185 ngx_memcpy(out + oi, enc, n);
186 oi += n;
187 }
188 if (oi + 1 < outsz) out[oi++] = '=';
189 n = uriencode_param(params[i].value, params[i].value_len,
190 enc, sizeof(enc));
191 if (oi + n < outsz) {
192 ngx_memcpy(out + oi, enc, n);
193 oi += n;
194 }
195 }
196 out[oi] = '\0';
197 return oi;
198}
