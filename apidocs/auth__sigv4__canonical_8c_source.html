2 * auth.c — AWS Signature Version 4 (HMAC-SHA256) verification.

31/*
32 * qsort comparator implementing SigV4's "sort by name, then by value" ordering.
33 *
34 * SigV4 sorts on the byte values of the *encoded* params, but because percent-
35 * encoding is order-preserving for the unreserved-only alphabet used here, a raw
36 * byte compare of the decoded bytes yields the same ordering — so we compare the
37 * decoded buffers directly and avoid encoding twice.
38 *
39 * Names are not NUL-terminated to a common length, so a plain strcmp/memcmp over
40 * the longer length would read past one buffer. We instead memcmp() the common
41 * prefix (the shorter of the two lengths); only if that prefix is equal does the
42 * shorter string sort first (the length tiebreak below). This reproduces standard
43 * lexicographic ordering where "a" < "ab". Value comparison repeats the same
44 * prefix-then-length scheme as the secondary key.
45 */
46static int
47qparam_cmp(const void *a, const void *b)
48{
49 const qparam_t *pa = (const qparam_t *) a;
50 const qparam_t *pb = (const qparam_t *) b;
51 size_t ml = pa->name_len < pb->name_len ? pa->name_len : pb->name_len;
52 int r = memcmp(pa->name, pb->name, ml);
53 if (r != 0) return r;
54 /* common prefix equal -> shorter name is the lesser ("a" < "ab") */
55 if (pa->name_len != pb->name_len) {
56 return (pa->name_len < pb->name_len) ? -1 : 1;
57 }
58
59 ml = pa->value_len < pb->value_len ? pa->value_len : pb->value_len;
60 r = memcmp(pa->value, pb->value, ml);
61 if (r != 0) return r;
62 return (pa->value_len < pb->value_len) ? -1
63 : (pa->value_len > pb->value_len) ? 1 : 0;
64}

74/*
75 * Parse one "name=value" (or bare-flag "name") query segment spanning
76 * [segment_start, segment_end), with `equals` pointing at the first '=' in the
77 * segment or NULL when there is none, url-decoding the name and value into *p.
78 * Returns 1 if the parameter was stored, 0 if it should be skipped — a decode
79 * failure on either half, or the X-Amz-Signature parameter when skip_signature
80 * is set (it must not appear in the string it signs).
81 */
82static int
83sigv4_store_param(qparam_t *p, const u_char *segment_start,
84 const u_char *segment_end, const u_char *equals, ngx_flag_t skip_signature)
85{
86 size_t raw_name_len;
87 size_t raw_value_len;
88 const u_char *raw_value;
89
90 /* No '=' in the segment -> a bare flag parameter: the whole segment is the
91 * name and the value is empty (""), which SigV4 still canonicalises as
92 * "name=". When '=' is present, name is everything before it and value is
93 * everything after (excluding the '=' itself). */
94 raw_name_len = (size_t) ((equals ? equals : segment_end) - segment_start);
95 raw_value = equals ? equals + 1 : (u_char *) "";
96 raw_value_len = equals ? (size_t) (segment_end - equals - 1) : 0;
97
98 if (brix_http_urldecode(segment_start, raw_name_len,
99 (char *) p->name, sizeof(p->name),
100 BRIX_URLDECODE_PLUS_TO_SPACE |
101 BRIX_URLDECODE_REJECT_NUL) != BRIX_URLDECODE_OK)
102 {
103 return 0;
104 }
105 p->name_len = strlen((char *) p->name);
106
107 if (skip_signature
108 && p->name_len == 15
109 && ngx_strcmp(p->name, (u_char *) "X-Amz-Signature") == 0)
110 {
111 return 0;
112 }
113
114 if (brix_http_urldecode(raw_value, raw_value_len,
115 (char *) p->value, sizeof(p->value),
116 BRIX_URLDECODE_PLUS_TO_SPACE |
117 BRIX_URLDECODE_REJECT_NUL) != BRIX_URLDECODE_OK)
118 {
119 return 0;
120 }
121 p->value_len = strlen((char *) p->value);
122 return 1;
123}

125/* ---- Scan one query-string segment, locating its end and first '=' ----
126 *
127 * WHAT: Given the start of a segment and the end of the query string, returns a
128 * pointer to the segment end (the next '&' or the query-string end, whichever
129 * comes first) and writes the position of the FIRST '=' within the segment to
130 * *equals_out, or NULL when the segment contains no '='.
131 *
132 * WHY: The name/value split in a SigV4 query parameter is defined by the first
133 * '=' only — a value may legitimately contain further '=' bytes, so those later
134 * '=' bytes belong to the value, not the split. Isolating the scan keeps the
135 * parse loop flat and the "first-equals-wins" rule stated in exactly one place,
136 * which is load-bearing for byte-identical canonicalisation.
137 *
138 * HOW:
139 * 1. Walk from segment_start until end or a '&' delimiter is reached.
140 * 2. Record the position of the first '=' seen (leave NULL if none).
141 * 3. Publish the first-'=' position via *equals_out and return the end pointer.
142 */
143static const u_char *
144sigv4_scan_segment(const u_char *segment_start, const u_char *end,
145 const u_char **equals_out)
146{
147 const u_char *segment_end = segment_start;
148 const u_char *equals = NULL;
149
150 while (segment_end < end && *segment_end != '&') {
151 if (*segment_end == '=' && equals == NULL) {
152 equals = segment_end;
153 }
154 segment_end++;
155 }
156
157 *equals_out = equals;
158 return segment_end;
159}

161/* ---- Parse a raw query string into decoded, storable parameters ----
162 *
163 * WHAT: Splits [qs, qs+qslen) on '&' into at most max_params segments, decodes
164 * each into params[], and returns the number of parameters stored. Segments
165 * that are empty or rejected by sigv4_store_param (decode failure, or the
166 * X-Amz-Signature parameter when skip_signature is set) are dropped.
167 *
168 * WHY: The canonical query string is the exact byte sequence a client signs, so
169 * the parse must reproduce SigV4's segmentation precisely. Capping at max_params
170 * bounds the caller's fixed-size params[] stack array. Skipping X-Amz-Signature
171 * keeps the signature field out of the string it signs (self-referential).
172 *
173 * HOW:
174 * 1. Walk the query string one '&'-delimited segment at a time.
175 * 2. For each segment, locate its end and first '=' via sigv4_scan_segment.
176 * 3. Store non-empty segments accepted by sigv4_store_param, counting them.
177 * 4. Step past the '&' delimiter, or stop when the query-string end is hit.
178 */
179static int
180sigv4_parse_qs(const u_char *qs, size_t qslen, ngx_flag_t skip_signature,
181 qparam_t *params, int max_params)
182{
183 int param_count = 0;
184 const u_char *cursor = qs;
185 const u_char *end = qs + qslen;
186
187 while (cursor < end && param_count < max_params) {
188 const u_char *segment_start = cursor;
189 const u_char *equals = NULL;
190 const u_char *segment_end =
191 sigv4_scan_segment(segment_start, end, &equals);
192
193 if (segment_start < segment_end
194 && sigv4_store_param(&params[param_count], segment_start,
195 segment_end, equals, skip_signature))
196 {
197 param_count++;
198 }
199
200 /* Step past the '&' delimiter; if segment_end already hit `end` there is
201 * no delimiter to skip, so leave cursor at end to terminate the loop. */
202 cursor = (segment_end < end) ? segment_end + 1 : segment_end;
203 }
204
205 return param_count;
206}

208/* ---- Percent-encode one component and append it to the output buffer ----
209 *
210 * WHAT: Percent-encodes [src, src+slen) and appends the result at out[oi],
211 * returning the new write offset. If the encoded bytes would not fit within
212 * outsz (keeping the strict "< outsz" reservation for the caller's trailing
213 * NUL), nothing is written and oi is returned unchanged.
214 *
215 * WHY: Every append into the canonical buffer must be bounds-guarded so a short
216 * buffer truncates cleanly rather than overflowing; centralising the encode-and-
217 * guard step keeps that safety rule identical for both the name and the value.
218 *
219 * HOW:
220 * 1. Percent-encode the source component into a local scratch buffer.
221 * 2. Append it only if oi + encoded_len stays strictly below outsz.
222 * 3. Return the advanced offset (or the original offset when it did not fit).
223 */
224static size_t
225sigv4_append_encoded(u_char *out, size_t oi, size_t outsz,
226 const u_char *src, size_t slen)
227{
228 u_char enc[1024];
229 size_t n = uriencode_param(src, slen, enc, sizeof(enc));
230
231 if (oi + n < outsz) {
232 ngx_memcpy(out + oi, enc, n);
233 oi += n;
234 }
235 return oi;
236}

238/* ---- Emit sorted parameters as the canonical "name=value&..." string ----
239 *
240 * WHAT: Writes the already-sorted params[] as "enc(name)=enc(value)" joined by
241 * '&' into out[], NUL-terminates it, and returns the byte length written
242 * (excluding the NUL).
243 *
244 * WHY: This is the final canonical query string that both client and server feed
245 * into the signature. The '&' separator, the '=' between name and value, and the
246 * per-append bounds guard must be byte-identical to what clients compute, so the
247 * emission is expressed once here with reserved space for the trailing NUL.
248 *
249 * HOW:
250 * 1. For each parameter, prepend '&' before all but the first (bounds-guarded).
251 * 2. Append the encoded name, then a '=', then the encoded value.
252 * 3. NUL-terminate the buffer and return the written length.
253 */
254static size_t
255sigv4_emit_canonical_qs(const qparam_t *params, int param_count,
256 u_char *out, size_t outsz)
257{
258 size_t oi = 0;
259
260 for (int i = 0; i < param_count; i++) {
261 if (i > 0 && oi + 1 < outsz) {
262 out[oi++] = '&';
263 }
264 oi = sigv4_append_encoded(out, oi, outsz, params[i].name,
265 params[i].name_len);
266 if (oi + 1 < outsz) {
267 out[oi++] = '=';
268 }
269 oi = sigv4_append_encoded(out, oi, outsz, params[i].value,
270 params[i].value_len);
271 }
272
273 out[oi] = '\0';
274 return oi;
275}

277/* ---- Build the SigV4 canonical query string ----
278 *
279 * WHAT: Parses the raw query string [qs, qs+qslen), sorts its parameters, and
280 * writes the canonical "enc(name)=enc(value)&..." form into out[] (NUL-
281 * terminated), returning the byte length written (excluding the NUL).
282 *
283 * WHY: SigV4 requires parameters sorted alphabetically by name then value with
284 * names and values percent-encoded — this is the exact string clients compute
285 * and sign, so it must be reproduced byte-for-byte. Parameter count is capped at
286 * 64 to bound the stack allocation of the params[] array.
287 *
288 * HOW:
289 * 1. Parse the query string into at most 64 decoded parameters.
290 * 2. Sort them by name then value using qparam_cmp (SigV4 ordering).
291 * 3. Emit the sorted parameters as the canonical query string.
292 */
293size_t
294build_canonical_qs(const u_char *qs, size_t qslen, ngx_flag_t skip_signature,
295 u_char *out, size_t outsz)
296{
297 qparam_t params[64];
298 int param_count;
299
300 param_count = sigv4_parse_qs(qs, qslen, skip_signature, params, 64);
301 qsort(params, (size_t) param_count, sizeof(params[0]), qparam_cmp);
302 return sigv4_emit_canonical_qs(params, param_count, out, outsz);
303}
