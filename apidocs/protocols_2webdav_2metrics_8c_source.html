11static brix_metric_op_t
12webdav_unified_op(ngx_uint_t method)
13{
14 switch (method) {
15 case BRIX_WEBDAV_METHOD_GET:
16 return BRIX_METRIC_OP_READ;
17 case BRIX_WEBDAV_METHOD_HEAD:
18 return BRIX_METRIC_OP_STAT;
19 case BRIX_WEBDAV_METHOD_PUT:
20 return BRIX_METRIC_OP_WRITE;
21 case BRIX_WEBDAV_METHOD_DELETE:
22 return BRIX_METRIC_OP_DELETE;
23 case BRIX_WEBDAV_METHOD_MKCOL:
24 return BRIX_METRIC_OP_MKDIR;
25 case BRIX_WEBDAV_METHOD_COPY:
26 return BRIX_METRIC_OP_TPC;
27 case BRIX_WEBDAV_METHOD_PROPFIND:
28 return BRIX_METRIC_OP_DIRLIST;
29 default:
30 return BRIX_METRIC_OP_STAT;
31 }
32}

34static brix_sess_mode_t
35webdav_sess_mode(ngx_http_request_t *r, ngx_uint_t method)
36{
37 if (method == BRIX_WEBDAV_METHOD_GET) {
38 return BRIX_SESS_MODE_READ;
39 }
40 if (method == BRIX_WEBDAV_METHOD_PUT
41 || method == BRIX_WEBDAV_METHOD_MKCOL
42 || (r != NULL && r->method == NGX_HTTP_MOVE))
43 {
44 return BRIX_SESS_MODE_WRITE;
45 }
46 if (method == BRIX_WEBDAV_METHOD_DELETE) {
47 return BRIX_SESS_MODE_DELETE;
48 }
49 if (method == BRIX_WEBDAV_METHOD_COPY) {
50 return BRIX_SESS_MODE_COPY;
51 }
52 if (method == BRIX_WEBDAV_METHOD_PROPFIND) {
53 ngx_str_t depth = brix_http_get_header(r, "Depth");
54 if (depth.len == 1 && depth.data[0] == '0') {
55 return BRIX_SESS_MODE_META;
56 }
57 return BRIX_SESS_MODE_LIST;
58 }
59
60 return BRIX_SESS_MODE_META;
61}

132static int64_t
133webdav_sess_xfer_expected(ngx_http_request_t *r, brix_sess_mode_t mode)
134{
135 if (r == NULL) {
136 return -1;
137 }
138
139 if (mode == BRIX_SESS_MODE_READ || mode == BRIX_SESS_MODE_COPY) {
140 return r->headers_out.content_length_n >= 0
141 ? (int64_t) r->headers_out.content_length_n : -1;
142 }
143
144 if (mode == BRIX_SESS_MODE_WRITE) {
145 return r->headers_in.content_length_n >= 0
146 ? (int64_t) r->headers_in.content_length_n : -1;
147 }
148
149 return -1;
150}

152static uint64_t
153webdav_sess_xfer_bytes(ngx_http_request_t *r, brix_sess_mode_t mode,
154 int64_t expected)
155{
156 if (mode == BRIX_SESS_MODE_READ || mode == BRIX_SESS_MODE_COPY) {
157 return expected > 0 ? (uint64_t) expected : 0;
158 }
159
160 if (mode == BRIX_SESS_MODE_WRITE && r != NULL
161 && r->headers_in.content_length_n > 0)
162 {
163 return (uint64_t) r->headers_in.content_length_n;
164 }
165
166 return 0;
167}

169static void
170webdav_sess_start_xfer(ngx_http_request_t *r, brix_sess_t *sess,
171 const char *path, brix_sess_mode_t mode)
172{
173 ngx_http_brix_webdav_req_ctx_t *ctx;
174
175 if (!webdav_sess_mode_has_xfer(mode)) {
176 return;
177 }
178
179 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
180 if (ctx == NULL || ctx->sess_xfer_started) {
181 return;
182 }
183
184 brix_sess_xfer_start(sess, &ctx->sess_xfer, path, mode, -1);
185 ctx->sess_xfer_started = ctx->sess_xfer.active ? 1 : 0;
186}

188static void
189webdav_sess_finish_xfer(ngx_http_request_t *r, brix_sess_t *sess,
190 brix_sess_mode_t mode, ngx_uint_t status)
191{
192 ngx_http_brix_webdav_req_ctx_t *ctx;
193 int64_t expected;
194 uint64_t bytes;
195
196 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
197 if (ctx == NULL || !ctx->sess_xfer_started) {
198 return;
199 }
200
201 if (mode == BRIX_SESS_MODE_COPY && ctx->sess_xfer.expected >= 0) {
202 expected = ctx->sess_xfer.expected;
203 bytes = ctx->sess_xfer.bytes;
204 } else {
205 expected = webdav_sess_xfer_expected(r, mode);
206 bytes = webdav_sess_xfer_bytes(r, mode, expected);
207 }
208 ctx->sess_xfer.expected = expected;
209 if (bytes > ctx->sess_xfer.bytes) {
210 brix_sess_xfer_add(&ctx->sess_xfer, bytes - ctx->sess_xfer.bytes);
211 }
212
213 brix_sess_xfer_end(sess, &ctx->sess_xfer,
214 status < NGX_HTTP_BAD_REQUEST
215 ? BRIX_SESS_XFER_COMPLETE
216 : BRIX_SESS_XFER_ABORTED);
217 ctx->sess_xfer_started = 0;
218}

249/**
250 * WHAT: Map nginx HTTP method to the WebDAV metric enum for Prometheus request counting.
251 *
252 * Returns a value from BRIX_WEBDAV_METHOD_* enum corresponding to the incoming
253 * request's HTTP method. Uses r->method (nginx internal enum) for OPTIONS/HEAD/GET/
254 * PUT/DELETE; uses r->method_name string comparison for WebDAV-specific methods
255 * MKCOL, COPY, and PROPFIND (which nginx doesn't have built-in enums for). Returns
256 * BRIX_WEBDAV_METHOD_OTHER as fallback for unknown or custom methods. All callers
257 * use this to index into the requests_total[METHOD] metric counter before processing
258 * begins or after completing an operation.
259 */
260ngx_uint_t
261webdav_metrics_method(ngx_http_request_t *r)
262{
263 const brix_http_operation_t *op;
264
265 op = brix_http_operation_find(r, brix_webdav_operations,
266 brix_webdav_operations_count);
267
268 return op ? op->metric_slot : BRIX_WEBDAV_METHOD_OTHER;
269}

271/**
272 * WHAT: Increment the WebDAV requests_total counter for this request's method.
273 *
274 * Called early in each handler to record that a request arrived for a specific HTTP
275 * method. Uses webdav_metrics_method() to classify the method, then increments the
276 * requests_total[method] Prometheus metric via BRIX_WEBDAV_METRIC_INC(). This is
277 * typically called at the start of every WebDAV operation handler (GET, PUT, DELETE, etc.)
278 * before any processing begins. Does not track status — only counts request arrivals.
279 */
280void
281webdav_metrics_request(ngx_http_request_t *r)
282{
283 ngx_uint_t method = webdav_metrics_method(r);
284
285 BRIX_WEBDAV_METRIC_INC(requests_total[method]);
286}

288/**
289 * WHAT: Increment the WebDAV responses_total counter by method and status class.
290 *
291 * Called at end of each handler to record the outcome of processing. Determines the HTTP
292 * status from either rc (error code), r->headers_out.status (explicitly set response), or
293 * defaults to NGX_HTTP_OK if no status was set. Maps to a low-cardinality status class via
294 * webdav_metrics_status_class(), then increments responses_total[method][status_class].
295 * Returns immediately if rc == NGX_DONE (request already finalized, metrics should not be
296 * double-counted). Used by all handlers after completing their operation logic.
297 */
298void
299webdav_metrics_response(ngx_http_request_t *r, ngx_int_t rc)
300{
301 ngx_uint_t method;
302 ngx_uint_t status;
303 ngx_uint_t status_class;
304 brix_sess_t *sess;
305 char path[BRIX_SESSLOG_PATH_MAX];
306 char errscratch[BRIX_SESSLOG_ERR_MAX];
307
308 if (rc == NGX_DONE) {
309 return;
310 }
311
312 method = webdav_metrics_method(r);
313
314 status = brix_http_effective_status(r, rc);
315 status_class = brix_http_status_class(status);
316 sess = webdav_sess(r);
317 brix_sess_result(sess, status < NGX_HTTP_BAD_REQUEST,
318 brix_http_sess_uri(r, path, sizeof(path)),
319 webdav_sess_mode(r, method),
320 status < NGX_HTTP_BAD_REQUEST ? NULL
321 : brix_sesslog_err_from_http((int) status,
322 errscratch,
323 sizeof(errscratch)));
324 webdav_sess_finish_xfer(r, sess, webdav_sess_mode(r, method), status);
325 BRIX_WEBDAV_METRIC_INC(responses_total[method][status_class]);
326 brix_metric_op_done(BRIX_PROTO_WEBDAV, webdav_unified_op(method),
327 0, 0,
328 brix_metric_err_from_http_status(status));
329}

361/**
362 * WHAT: Emit a status-only (empty-body) response and finalize the request.
363 *
364 * Captures the recurring four-line tail used by status-only outcomes
365 * (201/204/… with no body): set the status and a zero content length, send the
366 * header, then finalize via the send_special result so response metrics are
367 * recorded. Equivalent to:
368 * r->headers_out.status = status;
369 * r->headers_out.content_length_n = 0;
370 * ngx_http_send_header(r);
371 * webdav_metrics_finalize_request(r, ngx_http_send_special(r, NGX_HTTP_LAST));
372 *
373 * The caller still owns flow control: this finalizes the request, so it must be
374 * the last action on `r` and the caller should return immediately after. Add
375 * any response headers (e.g. Location for 201) before calling.
376 */
377void
378webdav_send_status_only(ngx_http_request_t *r, ngx_uint_t status)
379{
380 r->headers_out.status = status;
381 r->headers_out.content_length_n = 0;
382 ngx_http_send_header(r);
383 webdav_metrics_finalize_request(r, ngx_http_send_special(r, NGX_HTTP_LAST));
384}
