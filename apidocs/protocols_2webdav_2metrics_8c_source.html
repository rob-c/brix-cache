33/**
34 * WHAT: Map nginx HTTP method to the WebDAV metric enum for Prometheus request counting.
35 *
36 * Returns a value from BRIX_WEBDAV_METHOD_* enum corresponding to the incoming
37 * request's HTTP method. Uses r->method (nginx internal enum) for OPTIONS/HEAD/GET/
38 * PUT/DELETE; uses r->method_name string comparison for WebDAV-specific methods
39 * MKCOL, COPY, and PROPFIND (which nginx doesn't have built-in enums for). Returns
40 * BRIX_WEBDAV_METHOD_OTHER as fallback for unknown or custom methods. All callers
41 * use this to index into the requests_total[METHOD] metric counter before processing
42 * begins or after completing an operation.
43 */
44ngx_uint_t
45webdav_metrics_method(ngx_http_request_t *r)
46{
47 const brix_http_operation_t *op;
48
49 op = brix_http_operation_find(r, brix_webdav_operations,
50 brix_webdav_operations_count);
51
52 return op ? op->metric_slot : BRIX_WEBDAV_METHOD_OTHER;
53}

55/**
56 * WHAT: Increment the WebDAV requests_total counter for this request's method.
57 *
58 * Called early in each handler to record that a request arrived for a specific HTTP
59 * method. Uses webdav_metrics_method() to classify the method, then increments the
60 * requests_total[method] Prometheus metric via BRIX_WEBDAV_METRIC_INC(). This is
61 * typically called at the start of every WebDAV operation handler (GET, PUT, DELETE, etc.)
62 * before any processing begins. Does not track status — only counts request arrivals.
63 */
64void
65webdav_metrics_request(ngx_http_request_t *r)
66{
67 ngx_uint_t method = webdav_metrics_method(r);
68
69 BRIX_WEBDAV_METRIC_INC(requests_total[method]);
70}

72/**
73 * WHAT: Increment the WebDAV responses_total counter by method and status class.
74 *
75 * Called at end of each handler to record the outcome of processing. Determines the HTTP
76 * status from either rc (error code), r->headers_out.status (explicitly set response), or
77 * defaults to NGX_HTTP_OK if no status was set. Maps to a low-cardinality status class via
78 * webdav_metrics_status_class(), then increments responses_total[method][status_class].
79 * Returns immediately if rc == NGX_DONE (request already finalized, metrics should not be
80 * double-counted). Used by all handlers after completing their operation logic.
81 */
82void
83webdav_metrics_response(ngx_http_request_t *r, ngx_int_t rc)
84{
85 ngx_uint_t method;
86 ngx_uint_t status;
87 ngx_uint_t status_class;
88
89 if (rc == NGX_DONE) {
90 return;
91 }
92
93 method = webdav_metrics_method(r);
94
95 status = brix_http_effective_status(r, rc);
96 status_class = brix_http_status_class(status);
97 BRIX_WEBDAV_METRIC_INC(responses_total[method][status_class]);
98 brix_metric_op_done(BRIX_PROTO_WEBDAV, webdav_unified_op(method),
99 0, 0,
100 brix_metric_err_from_http_status(status));
101}

103/**
104 * WHAT: Wrapper helper — records response metrics then returns the original rc value.
105 *
106 * Convenience function that calls webdav_metrics_response(r, rc) and immediately returns
107 * rc to the caller. Used by handlers that want to record metrics before returning an nginx
108 * result code without needing two separate statement lines. Equivalent to:
109 * webdav_metrics_response(r, rc); return rc;

111ngx_int_t
112webdav_metrics_return(ngx_http_request_t *r, ngx_int_t rc)
113{
114 webdav_metrics_response(r, rc);

133/**
134 * WHAT: Emit a status-only (empty-body) response and finalize the request.
135 *
136 * Captures the recurring four-line tail used by status-only outcomes
137 * (201/204/… with no body): set the status and a zero content length, send the
138 * header, then finalize via the send_special result so response metrics are
139 * recorded. Equivalent to:
140 * r->headers_out.status = status;
141 * r->headers_out.content_length_n = 0;
142 * ngx_http_send_header(r);
143 * webdav_metrics_finalize_request(r, ngx_http_send_special(r, NGX_HTTP_LAST));
144 *
145 * The caller still owns flow control: this finalizes the request, so it must be
146 * the last action on `r` and the caller should return immediately after. Add
147 * any response headers (e.g. Location for 201) before calling.
148 */
149void
150webdav_send_status_only(ngx_http_request_t *r, ngx_uint_t status)
151{
152 r->headers_out.status = status;
153 r->headers_out.content_length_n = 0;
154 ngx_http_send_header(r);
155 webdav_metrics_finalize_request(r, ngx_http_send_special(r, NGX_HTTP_LAST));
156}
