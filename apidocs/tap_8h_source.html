4/*
5 * tap.h — protocol observation tap (decoder + sink fan-out).
6 *
7 * WHAT: turns raw XRootD wire bytes into a decoded frame descriptor and fans each
8 * frame out to registered sinks (audit / metrics / capture / inspection). Fed by
9 * both proxy modes: the terminating proxy (full plaintext) and the transparent
10 * relay (whatever travels in cleartext).
11 * WHY: one decoder + one fan-out, shared, instead of per-consumer frame parsing.
12 * HOW: pure C — no nginx, no allocation, no OpenSSL — so it embeds in any consumer
13 * and unit-tests standalone. Reuses src/protocol/frame_hdr.h + opcodes.h.
14 */

74/* ---- streaming decoder (tap_stream.c) ----
75 * Feeds the byte-relay: one per direction. Reassembles frame headers across
76 * arbitrary chunk boundaries, captures the path for path-bearing requests (up to
77 * BRIX_TAP_PATH_CAP bytes), and emits each frame to `tap`. Non-path frames
78 * (and all responses) emit as soon as the header is complete; the payload is
79 * skipped, never buffered whole. */
80#define BRIX_TAP_PATH_CAP 1024
81
82/* The client opens with a fixed 20-byte ClientInitHandShake before any request
83 * framing (coalesced with the first kXR_protocol in v5); the C2U decoder skips it
84 * so request framing stays aligned. The server's reply is a standard-format
85 * response, so the U2C decoder needs no preamble skip. */
86#define BRIX_TAP_C2U_PREAMBLE 20
