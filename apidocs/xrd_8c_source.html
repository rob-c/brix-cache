91/* Exec `tool` (found next to this binary, else via PATH) with argv. Does not return
92 * on success. */
93void
94exec_tool(const char *tool, char **argv)
95{
96 char self[PATH_MAX];
97 ssize_t n = readlink("/proc/self/exe", self, sizeof(self) - 1);
98
99 if (n > 0) {
100 char dirbuf[PATH_MAX];
101 char path[PATH_MAX];
102 char *dir;
103 self[n] = '\0';
104 snprintf(dirbuf, sizeof(dirbuf), "%s", self);
105 dir = dirname(dirbuf);
106 if ((size_t) snprintf(path, sizeof(path), "%s/%s", dir, tool) < sizeof(path)
107 && access(path, X_OK) == 0) {
108 execv(path, argv);
109 }
110 }
111 execvp(tool, argv); /* fall back to $PATH */
112 fprintf(stderr, "xrd: cannot run %s: %s\n", tool, strerror(errno));
113 _exit(127);
114}

117/* Map an fs-verb path-position arg to what xrdfs expects: a root:// URL (or an alias
118 * resolving to one) becomes its path component (host/port must match `ehost:eport`);
119 * anything else (a bare path or a flag) is passed through. Returns a malloc'd string,
120 * or NULL with *mismatch=1 if the arg targets a different endpoint. */
121char *
122map_fs_arg(const char *arg, const char *ehost, int eport, int *mismatch)
123{
124 char resolved[XRDC_PATH_MAX];
125 brix_url u;
126 brix_status st;
127
128 *mismatch = 0;
129 brix_status_clear(&st);
130 brix_alias_resolve(arg, resolved, sizeof(resolved));
131 if (brix_url_parse(resolved, &u, &st) == 0
132 && (u.scheme == XRDC_SCHEME_ROOT || u.scheme == XRDC_SCHEME_ROOTS)) {
133 if (strcmp(u.host, ehost) != 0 || u.port != eport) {
134 *mismatch = 1;
135 return NULL;
136 }
137 return strdup(u.path[0] != '\0' ? u.path : "/");
138 }
139 return strdup(arg); /* a bare path or a flag — verbatim */
140}

388/*
389 * WHAT: find the FIRST arg (argv[2..]) that resolves to a root:// URL; it fixes
390 * the connect endpoint (path depth doesn't matter — `root://h//` targets
391 * the root). Returns 1 with *sp filled, else 0.
392 * WHY: scanning (rather than assuming argv[2]) lets flags precede the endpoint,
393 * e.g. `xrd df -h root://h//` or `xrd ln -s root://h//tgt root://h//link`.
394 * HOW: alias-resolve + URL-parse each arg; on the first root/roots hit,
395 * rebuild `scheme://host:port` (bracketing IPv6 hosts) into sp->endpoint.
396 */
397static int
398fs_find_endpoint(int argc, char **argv, fs_split_t *sp)
399{
400 char resolved[XRDC_PATH_MAX];
401 brix_status st;
402 int i;
403
404 sp->ep_idx = -1;
405 for (i = 2; i < argc; i++) {
406 brix_status_clear(&st);
407 brix_alias_resolve(argv[i], resolved, sizeof(resolved));
408 if (brix_url_parse(resolved, &sp->u, &st) == 0
409 && (sp->u.scheme == XRDC_SCHEME_ROOT
410 || sp->u.scheme == XRDC_SCHEME_ROOTS)) {
411 const char *scheme =
412 (sp->u.scheme == XRDC_SCHEME_ROOTS) ? "roots" : "root";
413 int v6 = (strchr(sp->u.host, ':') != NULL);
414 snprintf(sp->endpoint, sizeof(sp->endpoint), "%s://%s%s%s:%d", scheme,
415 v6 ? "[" : "", sp->u.host, v6 ? "]" : "", sp->u.port);
416 sp->ep_idx = i;
417 return 1;
418 }
419 }
420 return 0;
421}

424/*
425 * WHAT: append the mapped fs-verb args (argv[2..]) to nv starting at slot k.
426 * Returns the next free slot, or -1 (with the error printed) when an arg
427 * targets a different endpoint.
428 * WHY: map every arg — the endpoint-bearing URL and any further same-endpoint
429 * URL/alias become their path components; flags and bare paths pass
430 * through. So flags-before-endpoint and multi-path verbs (mv/ln) work.
431 * HOW: the endpoint slot emits an explicit path only when the URL carried one
432 * (a bare `root://h//`, path "/" or empty, leaves the verb to default);
433 * every other slot goes through map_fs_arg().
434 */
435static int
436fs_map_split_args(char **nv, int k, int argc, char **argv, fs_split_t *sp)
437{
438 int i;
439
440 for (i = 2; i < argc; i++) {
441 int mism = 0;
442 char *m;
443 if (i == sp->ep_idx) {
444 if (sp->u.path[0] == '/' && sp->u.path[1] != '\0') {
445 nv[k++] = strdup(sp->u.path);
446 }
447 continue;
448 }
449 m = map_fs_arg(argv[i], sp->u.host, sp->u.port, &mism);
450 if (mism) {
451 fprintf(stderr, "xrd %s: every path must be on the same endpoint "
452 "(%s)\n", argv[1], sp->endpoint);
453 return -1;
454 }
455 nv[k++] = m;
456 }
457 return k;
458}

461/*
462 * WHAT: filesystem verb -> exec `xrdfs <endpoint> <verb> [paths...]`.
463 * WHY: xrdfs separates the connect endpoint from the path, so when the target
464 * is a full root:// URL (or an alias that resolves to one) carrying a
465 * path, split it: `xrd stat root://h//d/f` -> `xrdfs root://h:port stat
466 * /d/f`. A bare host:port (or anything not a root:// URL) is passed
467 * through unchanged.
468 * HOW: fs_find_endpoint() scans for the endpoint URL; split mode maps every
469 * arg via fs_map_split_args(), pass-through mode copies args verbatim.
470 */
471static int
472cmd_fs_verb(int argc, char **argv)
473{
474 const char *cmd = argv[1];
475 fs_split_t sp;
476 char **nv;
477 int i, k = 0, split;
478
479 if (argc < 3) {
480 fprintf(stderr, "xrd %s: needs an <endpoint>\n", cmd);
481 return 50;
482 }
483 split = fs_find_endpoint(argc, argv, &sp);
484 nv = (char **) malloc((size_t) (argc + 3) * sizeof(char *));
485 if (nv == NULL) { fprintf(stderr, "xrd: out of memory\n"); return 51; }
486 nv[k++] = (char *) "xrdfs";
487 if (split) {
488 nv[k++] = sp.endpoint; /* connect endpoint (host:port) */
489 nv[k++] = (char *) cmd; /* the verb */
490 k = fs_map_split_args(nv, k, argc, argv, &sp);
491 if (k < 0) {
492 free(nv);
493 return 50;
494 }
495 } else {
496 nv[k++] = argv[2]; /* bare endpoint as given */
497 nv[k++] = (char *) cmd;
498 for (i = 3; i < argc; i++) { /* paths/flags verbatim */
499 nv[k++] = argv[i];
500 }
501 }
502 nv[k] = NULL;
503 exec_tool("xrdfs", nv);
504 return 127; /* unreachable: exec_tool does not return */
505}

508/*
509 * WHAT: emit a did-you-mean hint when the user typed an unrecognised xrd
510 * command; search both the named xrd dispatch table and FS_VERBS.
511 * WHY: spec WS-7: every unknown-command site must offer a suggestion when
512 * one exists within DL distance ≤ 2 (TTY-gated, C3 compliant).
513 * HOW: build a merged NULL-terminated names array — dispatch-table names in
514 * their historical order, skipping '-'-prefixed flag aliases, then
515 * FS_VERBS — and pass it to brix_suggest().
516 */
517static void
518report_unknown_command(const char *cmd)
519{
520 const char *all_names[80]; /* FS_VERBS(36) + named commands(24) + pad */
521 int n = 0;
522 int i;
523 const char *suggestion;
524
525 for (i = 0; XRD_DISPATCH[i].name != NULL && n < 79; i++) {
526 if (XRD_DISPATCH[i].name[0] != '-') {
527 all_names[n++] = XRD_DISPATCH[i].name;
528 }
529 }
530 for (i = 0; FS_VERBS[i] != NULL && n < 79; i++) {
531 all_names[n++] = FS_VERBS[i];
532 }
533 all_names[n] = NULL;
534
535 fprintf(stderr, "xrd: unknown command '%s'\n\n", cmd);
536 suggestion = brix_suggest(cmd, all_names);
537 if (suggestion != NULL) {
538 brix_cli_hint("hint: did you mean '%s'?\n", suggestion);
539 }
540}
