91/* Exec `tool` (found next to this binary, else via PATH) with argv. Does not return
92 * on success. */
93void
94exec_tool(const char *tool, char **argv)
95{
96 char self[PATH_MAX];
97 ssize_t n = readlink("/proc/self/exe", self, sizeof(self) - 1);
98
99 if (n > 0) {
100 char dirbuf[PATH_MAX];
101 char path[PATH_MAX];
102 char *dir;
103 self[n] = '\0';
104 snprintf(dirbuf, sizeof(dirbuf), "%s", self);
105 dir = dirname(dirbuf);
106 if ((size_t) snprintf(path, sizeof(path), "%s/%s", dir, tool) < sizeof(path)
107 && access(path, X_OK) == 0) {
108 execv(path, argv);
109 }
110 }
111 execvp(tool, argv); /* fall back to $PATH */
112 fprintf(stderr, "xrd: cannot run %s: %s\n", tool, strerror(errno));
113 _exit(127);
114}

117/* Map an fs-verb path-position arg to what xrdfs expects: a root:// URL (or an alias
118 * resolving to one) becomes its path component (host/port must match `ehost:eport`);
119 * anything else (a bare path or a flag) is passed through. Returns a malloc'd string,
120 * or NULL with *mismatch=1 if the arg targets a different endpoint. */
121char *
122map_fs_arg(const char *arg, const char *ehost, int eport, int *mismatch)
123{
124 char resolved[XRDC_PATH_MAX];
125 brix_url u;
126 brix_status st;
127
128 *mismatch = 0;
129 brix_status_clear(&st);
130 brix_alias_resolve(arg, resolved, sizeof(resolved));
131 if (brix_url_parse(resolved, &u, &st) == 0
132 && (u.scheme == XRDC_SCHEME_ROOT || u.scheme == XRDC_SCHEME_ROOTS)) {
133 if (strcmp(u.host, ehost) != 0 || u.port != eport) {
134 *mismatch = 1;
135 return NULL;
136 }
137 return strdup(u.path[0] != '\0' ? u.path : "/");
138 }
139 return strdup(arg); /* a bare path or a flag — verbatim */
140}

260 /* filesystem verb. xrdfs separates the connect endpoint from the path, so when
261 * the target is a full root:// URL (or an alias that resolves to one) carrying a
262 * path, split it: `xrd stat root://h//d/f` -> `xrdfs root://h:port stat /d/f`.
263 * A bare host:port (or anything not a root:// URL) is passed through unchanged. */
264 if (is_fs_verb(cmd)) {
265 char resolved[XRDC_PATH_MAX];
266 char endpoint[320];
267 brix_url u;
268 brix_status st;
269 char **nv;
270 int i, k = 0, split = 0, ep_idx = -1;
271
272 if (argc < 3) {
273 fprintf(stderr, "xrd %s: needs an <endpoint>\n", cmd);
274 return 50;
275 }
276 /* Find the FIRST arg that resolves to a root:// URL; it fixes the connect
277 * endpoint (path depth doesn't matter — `root://h//` targets the root).
278 * Scanning (rather than assuming argv[2]) lets flags precede the endpoint,
279 * e.g. `xrd df -h root://h//` or `xrd ln -s root://h//tgt root://h//link`. */
280 for (i = 2; i < argc; i++) {
281 brix_status_clear(&st);
282 brix_alias_resolve(argv[i], resolved, sizeof(resolved));
283 if (brix_url_parse(resolved, &u, &st) == 0
284 && (u.scheme == XRDC_SCHEME_ROOT || u.scheme == XRDC_SCHEME_ROOTS)) {
285 const char *scheme = (u.scheme == XRDC_SCHEME_ROOTS) ? "roots" : "root";
286 int v6 = (strchr(u.host, ':') != NULL);
287 snprintf(endpoint, sizeof(endpoint), "%s://%s%s%s:%d", scheme,
288 v6 ? "[" : "", u.host, v6 ? "]" : "", u.port);
289 split = 1;
290 ep_idx = i;
291 break;
292 }
293 }
294 nv = (char **) malloc((size_t) (argc + 3) * sizeof(char *));
295 if (nv == NULL) { fprintf(stderr, "xrd: out of memory\n"); return 51; }
296 nv[k++] = (char *) "xrdfs";
297 if (split) {
298 nv[k++] = endpoint; /* connect endpoint (host:port) */
299 nv[k++] = (char *) cmd; /* the verb */
300 /* Map every arg: the endpoint-bearing URL and any further same-endpoint
301 * URL/alias become their path components; flags and bare paths pass
302 * through. So flags-before-endpoint and multi-path verbs (mv/ln) work. */
303 for (i = 2; i < argc; i++) {
304 int mism = 0;
305 char *m;
306 if (i == ep_idx) {
307 /* Emit an explicit path only when the URL carried one; a bare
308 * `root://h//` (path "/" or empty) leaves the verb to default. */
309 if (u.path[0] == '/' && u.path[1] != '\0') {
310 nv[k++] = strdup(u.path);
311 }
312 continue;
313 }
314 m = map_fs_arg(argv[i], u.host, u.port, &mism);
315 if (mism) {
316 fprintf(stderr, "xrd %s: every path must be on the same endpoint "
317 "(%s)\n", cmd, endpoint);
318 free(nv);
319 return 50;
320 }
321 nv[k++] = m;
322 }
323 } else {
324 nv[k++] = argv[2]; /* bare endpoint as given */
325 nv[k++] = (char *) cmd;
326 for (i = 3; i < argc; i++) { /* paths/flags verbatim */
327 nv[k++] = argv[i];
328 }
329 }
330 nv[k] = NULL;
331 exec_tool("xrdfs", nv);
332 }
