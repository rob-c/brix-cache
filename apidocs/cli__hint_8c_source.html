2 * cli_hint.c — TTY-gated usability hint helpers (spec §2 C3, WS-1/WS-3/WS-7).

4 * WHAT: brix_cli_hints_enabled() / brix_cli_hint() / brix_cli_hint_once() +
5 * brix_hint_sanitize_url() / brix_hint_url_double_slash() /
6 * brix_hint_doctor_referral().
7 * WHY: one place to audit the C3 compatibility gate so hints never leak into
8 * scripts, pipelines, or cron jobs (non-TTY stderr = unchanged output).
9 * HOW: isatty(STDERR_FILENO) && (BRIX_NO_HINTS unset or "0") — computed once

35/*
36 * WHAT: test whether BRIX_NO_HINTS suppresses all hints.
37 * WHY: two spellings accepted: unset (hints on) or the string "0" (hints on);
38 * any other value, including "" and "1", suppresses hints.
39 * HOW: getenv returns NULL when unset; the string "0" is the only set value
40 * that leaves hints on (following the common shell boolean convention).
41 */
42static int
43no_hints_env(void)
44{
45 const char *v = getenv("BRIX_NO_HINTS");
46 if (v == NULL) {
47 return 0; /* unset → hints are ON */
48 }
49 if (v[0] == '0' && v[1] == '\0') {
50 return 0; /* "0" → hints are ON */
51 }
52 return 1; /* any other value → suppress hints */
53}

55int
56brix_cli_hints_enabled(void)
57{
58 /*
59 * WHAT: lazy, per-process computation of the hint gate.
60 * WHY: isatty() and getenv() are cheap but called at every log site; a
61 * static cache avoids repeated syscalls and makes the gate consistent
62 * even if BRIX_NO_HINTS is mutated mid-process.
63 * HOW: 0 = unknown, 1 = enabled, -1 = disabled (tristate).
64 */
65 static int cached = 0;
66
67 if (cached == 0) {
68 cached = (isatty(STDERR_FILENO) && !no_hints_env()) ? 1 : -1;
69 }
70 return cached > 0;
71}

91void
92brix_cli_hint_once(const char *key, const char *fmt, ...)
93{
94 /*
95 * WHAT: emit at most one hint per (process, key).
96 * WHY: hints fired inside loops (e.g. per-file divergence warnings) would
97 * flood an interactive session; once-per-key suppresses duplicates.
98 * HOW: fixed 16-slot static table of seen key pointers/strings. strcmp
99 * match so string-literal and heap keys both work. When the table
100 * is full (16 distinct keys seen), new keys are silently dropped
101 * (resource-bounded, no malloc): spec WS-1 contract is to emit at
102 * most once per distinct key among the first 16 keys.
103 */
104 static const char *seen[HINT_ONCE_MAX];
105 static int seen_count = 0;
106 va_list ap;
107 int i;
108
109 if (!brix_cli_hints_enabled()) {
110 return;
111 }
112 /* Check whether this key was already emitted. */
113 for (i = 0; i < seen_count; i++) {
114 if (strcmp(seen[i], key) == 0) {
115 return;
116 }
117 }
118 /* Register the key (silently drop if table is full). */
119 if (seen_count < HINT_ONCE_MAX) {
120 seen[seen_count++] = key;
121 } else {
122 return; /* table full: drop new keys silently (spec WS-1 contract) */
123 }
124 va_start(ap, fmt);
125 vfprintf(stderr, fmt, ap);
126 va_end(ap);
127}

129void
130brix_hint_sanitize_url(const char *url_str, char *out, size_t outsz)
131{
132 /*
133 * WHAT: copy up to HINT_URL_MAX input bytes of url_str into out, escaping
134 * every byte < 0x20 or > 0x7e as "\xNN" (lowercase hex).
135 * WHY: spec WS-7 / WS-3: URLs in hint output must not carry terminal
136 * escape sequences (a hostile URL like "root://\x1b[31m…" could
137 * inject ANSI escapes into the terminal if printed verbatim).
138 * Escaping (rather than replacing) keeps the hostile bytes visible
139 * to the user in a safe, greppable form.
140 * HOW: byte-by-byte copy; printable ASCII (0x20–0x7e) passes through,
141 * everything else emits 4 bytes "\xNN". Stops at HINT_URL_MAX input
142 * bytes, NUL, or when out has no room for the next token. Callers
143 * size out for the worst case: HINT_URL_MAX*4 + 1.
144 */
145 static const char hexdig[] = "0123456789abcdef";
146 size_t i;
147 size_t o = 0;
148
149 if (out == NULL || outsz == 0) {
150 return;
151 }
152 out[0] = '\0';
153 if (url_str == NULL) {
154 return;
155 }
156 for (i = 0; i < HINT_URL_MAX && url_str[i] != '\0'; i++) {
157 unsigned char c = (unsigned char) url_str[i];
158
159 if (c < 0x20 || c > 0x7e) {
160 if (o + 4 >= outsz) {
161 break; /* no room for a full \xNN token + NUL */
162 }
163 out[o++] = '\\';
164 out[o++] = 'x';
165 out[o++] = hexdig[c >> 4];
166 out[o++] = hexdig[c & 0x0f];
167 } else {
168 if (o + 1 >= outsz) {
169 break; /* no room for the byte + NUL */
170 }
171 out[o++] = (char) c;
172 }
173 }
174 out[o] = '\0';
175}

177void
178brix_hint_url_double_slash(const brix_status *st, const brix_url *url)
179{
180 /*
181 * WHAT: emit the double-slash convention hint once per process when a
182 * NOT-FOUND-class failure occurred on a URL that had a single slash
183 * between authority and path.
184 * WHY: spec WS-3: users who type root://host/path instead of
185 * root://host//path see a not-found error with no indication of the
186 * conventional double-slash requirement; this hint fixes that gap.
187 * Gating on not-found matters: an auth failure or I/O error on a
188 * single-slash URL is NOT caused by the missing slash, so hinting
189 * there would mislead.
190 * HOW: require url->single_slash_path (set by url.c when collapse did
191 * NOT fire) AND a not-found-class status: kXR_NotFound covers both
192 * the wire error and HTTP-404-mapped failures (webfile.c maps 404 →
193 * kXR_NotFound); sys_errno==ENOENT covers locally-mapped misses.
194 * Emit via brix_cli_hint_once so it fires at most once per process.
195 */
196 int not_found;
197
198 if (st == NULL || url == NULL || !url->single_slash_path) {
199 return;
200 }
201 not_found = (st->kxr == kXR_NotFound || st->sys_errno == ENOENT);
202 if (!not_found) {
203 return;
204 }
205 brix_cli_hint_once("url-double-slash",
206 "hint: XRootD URLs take a double slash before absolute paths —\n"
207 " try root://HOST//PATH (you wrote a single '/'); see man xrd, URLS.\n");
208}

210void
211brix_hint_doctor_referral(const brix_status *st, const char *url_str)
212{
213 /*
214 * WHAT: emit a "diagnose with: xrddiag check <url>" hint once per process
215 * when a failure is auth-class (kXR_NotAuthorized / kXR_AuthFailed /
216 * XRDC_EAUTH — all three live in the st->kxr field: the local
217 * XRDC_E* sentinels are negative precisely so they can share the
218 * field with the positive kXR_* wire codes).
219 * WHY: spec WS-7: auth failures are rarely self-diagnosable from the
220 * error message alone; xrddiag check exercises the full credential
221 * + auth handshake and reports the root cause.
222 * HOW: check st->kxr for auth codes; sanitize url_str (control bytes →
223 * \xNN, 128-byte input cap) before printing; use
224 * brix_cli_hint_once("doctor") to suppress repeats.
225 */
226 char safe_url[HINT_URL_MAX * 4 + 1];
227 int is_auth;
228
229 if (st == NULL) {
230 return;
231 }
232 is_auth = (st->kxr == kXR_NotAuthorized
233 || st->kxr == kXR_AuthFailed
234 || st->kxr == XRDC_EAUTH);
235 if (!is_auth) {
236 return;
237 }
238 brix_hint_sanitize_url(url_str, safe_url, sizeof(safe_url));
239 if (safe_url[0] != '\0') {
240 brix_cli_hint_once("doctor",
241 "hint: diagnose with: xrddiag check %s\n", safe_url);
242 } else {
243 brix_cli_hint_once("doctor",
244 "hint: diagnose with: xrddiag check <endpoint>\n");
245 }
