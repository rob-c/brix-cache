1/*
2 * audit_handler.c — LOG-phase outcome classification + audit line.
3 *
4 * WHAT: after the response went out, maps the final status to a guard
5 * outcome (404 -> notfound, 401/403 -> authfail), runs guard_classify_post,
6 * and appends one audit line per flagged request.
7 * WHY: post-response signals (not-found storms, repeated auth failures) only
8 * exist once the backend has answered — the LOG phase is where the status
9 * is final and every request passes through exactly once.
10 * HOW: skip requests already audited by the ACCESS bounce (request ctx flag),
11 * rebuild guard_request_t with outcome fields, classify, write.
12 */
13
14#include "guard_http.h"

16/* ---- Final HTTP status -> guard outcome ----
17 *
18 * WHAT: maps the response status to the guard's outcome classes: 404 ->
19 * NOTFOUND, 401/403 -> AUTHFAIL, other 4xx/5xx -> ERROR, else OK.
20 *
21 * WHY: classify_post keys on outcomes, not raw statuses, so the same
22 * thresholds mean the same thing across HTTP and root://.
23 *
24 * HOW: 1. Compare against the nginx status constants in signal order.
25 */
26static guard_outcome_t
27status_to_outcome(ngx_uint_t status)
28{
29 if (status == NGX_HTTP_NOT_FOUND) {
30 return OUTCOME_NOTFOUND; /* 404 */
31 }
32 if (status == NGX_HTTP_UNAUTHORIZED || status == NGX_HTTP_FORBIDDEN) {
33 return OUTCOME_AUTHFAIL; /* 401 / 403 */
34 }
35 if (status >= NGX_HTTP_BAD_REQUEST) {
36 return OUTCOME_ERROR;
37 }
38 return OUTCOME_OK;
39}
40
41/* ---- LOG-phase handler: audit post-response signals ----
42 *
43 * WHAT: always returns NGX_OK (LOG handlers cannot affect the response);
44 * appends an audit line when classify_post flags the final outcome.
45 *
46 * WHY: one line per notfound/authfail is the raw material for fail2ban's
47 * threshold jails; requests the ACCESS phase already bounced were audited
48 * there and must not double-log.
49 *
50 * HOW: 1. Skip unguarded locations and ACCESS-bounced requests (ctx flag).
51 * 2. Rebuild guard_request_t and stamp the final status/outcome.
52 * 3. classify_post; write the audit line only on a flagged reason.
53 */
54ngx_int_t
55ngx_http_brix_guard_log_handler(ngx_http_request_t *r)
56{
57 ngx_http_brix_guard_loc_conf_t *lcf;
58 ngx_http_brix_guard_ctx_t *ctx;
59 guard_request_t req;
60 guard_reason_t reason;
61 char pathbuf[BRIX_GUARD_PATH_BUF];
62 char ipbuf[BRIX_GUARD_IP_BUF];
63
64 lcf = ngx_http_get_module_loc_conf(r, ngx_http_brix_guard_module);
65 if (!lcf->enable) {
66 return NGX_OK;
67 }
68 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_guard_module);
69 if (ctx != NULL && ctx->bounced) {
70 return NGX_OK; /* already logged at ACCESS phase (pre-reason) */
71 }
72
73 ngx_http_brix_guard_build_request(r, lcf, &req,
74 pathbuf, sizeof(pathbuf), ipbuf, sizeof(ipbuf));
75 req.status_code = (int) r->headers_out.status;
76 req.outcome = status_to_outcome(r->headers_out.status);
77
78 reason = guard_classify_post(&lcf->ruleset, &req);
79 if (reason != GUARD_R_NONE) {
80 ngx_http_brix_guard_write_audit(r, lcf, &req, reason);
81 }
82 return NGX_OK;
83}
