1/*
2 * pblock_refs.h — Phase-83 F10: content-addressed dedup + refcounted blobs.
3 *
4 * WHAT: The foundation of wave D: a driver-owned `blobs(blob_id, refcount,
5 * size, block_size, content_hash)` table that lets several `objects` rows
6 * share one physical blob. server_copy becomes a refcount bump
7 * (O(metadata) copy-on-write), publish-time dedup folds identical content
8 * onto an existing blob, and any write to a shared blob first breaks the
9 * share by copying the blocks to a fresh blob (block granularity makes
10 * that cheap).
11 *
12 * WHY: Exercises every CAP_SERVER_COPY caller (kXR clone, WebDAV COPY) against
13 * real copy-on-write semantics, and gives F6 snapshots / F11 versioning
14 * their O(metadata) foundation.
15 *
16 * HOW: Armed by the `dedup=1` static opt (its own gate, like csi/locks — this
17 * is a capability, not a lab toy); an init failure leaves the
18 * byte-for-byte production path. A row's absence means "refcount 1"
19 * (legacy blobs from before the gate was set), so enabling dedup on a
20 * populated export is safe. The content hash is the whole-object CRC-32
21 * from the wverify accumulator fed on the write path; a hash match is
22 * only a CANDIDATE — dedup always byte-verifies both blobs before
23 * linking, so a CRC collision (or a forged row) can never alias content.
24 * Share-break happens at OPEN for write-intent opens (a metadata
25 * boundary), keeping the byte hot path free of refcount reads.
26 * ngx-free (libc + sqlite3); BRIX_HAVE_SQLITE-gated.
27 *
28 * Requires: pblock_store.h (pblock_state_t), sd_pblock_catalog.h (pblock_meta)
29 * before inclusion.
30 */

56/* Release one reference to a blob — the single unlink/overwrite entry for BOTH
57 * gate states: with refs off (or on the last reference) it removes the block
58 * files and the blob's csi rows exactly like the pre-F10 code; with a live
59 * shared reference it only decrements. A DB error keeps the blocks (an orphan
60 * for `pblock-fsck --gc`, never a shared-block removal). */
61void pblock_refs_release(const pblock_state_t *st, const char *blob_id,
62 int64_t size, int64_t bs);

64/* Publish-time dedup: with the object's row already pointing at meta->blob_id,
65 * look for a byte-identical existing blob (hash+size+block_size candidates,
66 * then a full byte compare) and fold onto it — bump the survivor, repoint the
67 * row, release our now-redundant blob. Skips (tracking the blob as-is) when
68 * crc_ok is 0, when our blob is already shared, or when no candidate survives
69 * verification. Returns 1 (folded), 0 (kept), -1/errno. */
70int pblock_refs_dedup_publish(pblock_state_t *st, const char *path,
71 pblock_meta *meta, uint32_t crc, int crc_ok);
72
73/* Copy-on-write share-break for a write-intent open: when meta->blob_id is
74 * shared, copy its blocks to a fresh private blob (or start empty when `trunc`
75 * — the content is being replaced anyway), repoint the object's row, carry the
76 * csi rows over, and release the old reference. Updates meta->blob_id in
77 * place. 0 (private — possibly a no-op) or -1/errno. */
78int pblock_refs_break_share(pblock_state_t *st, const char *path,
