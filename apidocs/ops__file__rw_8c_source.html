8/*
9 * phase-42 W4 — inflate one inline-compressed kXR_read frame. The server
10 * compressed `out_cap`-bounded plaintext as a single self-contained codec frame,
11 * so a guard with out_cap = the requested length is sufficient (the data is our
12 * own server's, so the expansion-ratio bomb guard is left off). Returns the
13 * plaintext length produced, or -1 on a malformed/oversized frame.
14 */
15ssize_t
16brix_inflate_frame(uint8_t codec, const uint8_t *comp, size_t comp_len,
17 void *out, size_t out_cap, brix_status *st)
18{
19 brix_codec_guard_t guard;
20 brix_codec_stream_t *s;
21 size_t in_pos = 0, out_pos = 0;
22
23 if (comp_len == 0) {
24 return 0; /* empty frame ⇒ zero plaintext bytes (EOF / empty range) */
25 }
26
27 memset(&guard, 0, sizeof(guard));
28 guard.out_cap = out_cap; /* plaintext cannot exceed the requested len */
29 guard.max_ratio = 0;
30
31 s = brix_codec_open((brix_codec_id_t) codec,
32 BRIX_CODEC_DIR_DECOMPRESS, -1, &guard);
33 if (s == NULL) {
34 brix_status_set(st, XRDC_EPROTO, 0,
35 "inline read decompression unavailable (codec %u)",
36 (unsigned) codec);
37 return -1;
38 }
39
40 for (;;) {
41 size_t prev_in = in_pos, prev_out = out_pos;
42 brix_codec_rc_t rc = brix_codec_step(s, comp, comp_len, &in_pos,
43 (uint8_t *) out, out_cap,
44 &out_pos, 1 /* finish */);
45 if (rc == BRIX_CODEC_END) {
46 break;
47 }
48 if (rc != BRIX_CODEC_OK
49 || (in_pos == prev_in && out_pos == prev_out))
50 {
51 brix_codec_close(s);
52 brix_status_set(st, XRDC_EPROTO, 0,
53 "corrupt inline-compressed read frame");
54 return -1;
55 }
56 }
57
58 brix_codec_close(s);
59 return (ssize_t) out_pos;
60}

63ssize_t
64brix_file_read(brix_conn *c, brix_file *f, int64_t offset, void *buf, size_t len,
65 brix_status *st)
66{
67 ClientReadRequest req;
68 uint16_t sid, status;
69 size_t total = 0;
70 uint8_t *comp = NULL; /* accumulated codec frame (compressed only) */
71 size_t comp_cap = 0;
72
73 memset(&req, 0, sizeof(req));
74 req.requestid = htons(kXR_read);
75 {
76 xrdw_read_req_t b = { .offset = offset, .rlen = (int32_t) len };
77 memcpy(b.fhandle, f->fhandle, XRD_FHANDLE_LEN);
78 xrdw_read_req_pack(&b, ((ClientRequestHdr *) &req)->body);
79 }
80
81 if (brix_send(c, &req, NULL, &sid, st) != 0) {
82 return -1;
83 }
84
85 /* One or more frames (kXR_oksofar* then kXR_ok) make up the read result.
86 * Plaintext handles copy straight into buf; compressed handles accumulate
87 * the whole frame into `comp` and inflate it once the final kXR_ok arrives. */
88 for (;;) {
89 uint8_t *body = NULL;
90 uint32_t blen = 0;
91 brix_resp_out out = { &status, &body, &blen };
92 if (brix_recv(c, sid, &out, st) != 0) {
93 free(comp);
94 return -1;
95 }
96 if (blen > 0) {
97 if (f->read_codec == 0) {
98 if (total + blen > len) {
99 free(body);
100 brix_status_set(st, XRDC_EPROTO, 0,
101 "read returned more than requested");
102 return -1;
103 }
104 memcpy((uint8_t *) buf + total, body, blen);
105 total += blen;
106 } else {
107 /* Accumulate compressed bytes (cap at 2x len + slack — a frame
108 * of plaintext <= len cannot legitimately exceed that). */
109 size_t max_comp = len + (len / 2) + 4096;
110 if (total + blen > max_comp) {
111 free(body);
112 free(comp);
113 brix_status_set(st, XRDC_EPROTO, 0,
114 "compressed read frame too large");
115 return -1;
116 }
117 if (total + blen > comp_cap) {
118 size_t ncap = comp_cap ? comp_cap * 2 : 65536;
119 uint8_t *p;
120 while (ncap < total + blen) {
121 ncap *= 2;
122 }
123 p = (uint8_t *) realloc(comp, ncap);
124 if (p == NULL) {
125 free(body);
126 free(comp);
127 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
128 return -1;
129 }
130 comp = p;
131 comp_cap = ncap;
132 }
133 memcpy(comp + total, body, blen);
134 total += blen;
135 }
136 }
137 free(body);
138 if (status == kXR_ok) {
139 break;
140 }
141 }
142
143 if (f->read_codec == 0) {
144 return (ssize_t) total;
145 }
146
147 {
148 ssize_t plain = brix_inflate_frame(f->read_codec, comp, total,
149 buf, len, st);
150 free(comp);
151 return plain;
152 }
153}

156/*
157 * phase-42 W5: compress `in_len` plaintext bytes as ONE self-contained codec
158 * frame for an inline-write upload (the inverse of brix_inflate_frame). Returns
159 * a malloc'd buffer (caller frees) and sets *out_len, or NULL on failure.
160 */
161uint8_t *
162brix_deflate_frame(uint8_t codec, const void *in, size_t in_len, size_t *out_len,
163 brix_status *st)
164{
165 size_t cap = in_len + in_len / 2 + 4096;
166 uint8_t *out = malloc(cap);
167 brix_codec_stream_t *s;
168 size_t in_pos = 0, out_pos = 0;
169
170 if (out == NULL) {
171 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
172 return NULL;
173 }
174 s = brix_codec_open((brix_codec_id_t) codec,
175 BRIX_CODEC_DIR_COMPRESS, -1, NULL);
176 if (s == NULL) {
177 free(out);
178 brix_status_set(st, XRDC_EPROTO, 0,
179 "inline write compression unavailable (codec %u)",
180 (unsigned) codec);
181 return NULL;
182 }
183 for ( ;; ) {
184 size_t prev_in = in_pos, prev_out = out_pos;
185 brix_codec_rc_t rc = brix_codec_step(s, in, in_len, &in_pos,
186 out, cap, &out_pos, 1 /*finish*/);
187 uint8_t *p;
188 if (rc == BRIX_CODEC_END) {
189 break;
190 }
191 if (rc != BRIX_CODEC_OK
192 || (in_pos == prev_in && out_pos == prev_out))
193 {
194 brix_codec_close(s);
195 free(out);
196 brix_status_set(st, XRDC_EPROTO, 0, "compress failed");
197 return NULL;
198 }
199 /* OK with room remaining: the codec made progress but isn't finished (e.g.
200 * lz4 consumes its input in chunks) — call again into the same buffer.
201 * Only a genuinely full buffer needs a grow (preserving bytes via realloc).
202 * cap is pre-sized to > the worst-case compressed size, so growth is rare. */
203 if (out_pos < cap) {
204 continue;
205 }
206 cap *= 2;
207 p = realloc(out, cap);
208 if (p == NULL) {
209 brix_codec_close(s);
210 free(out);
211 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
212 return NULL;
213 }
214 out = p;
215 }
216 brix_codec_close(s);
217 *out_len = out_pos;
218 return out;
219}

275/* ---- Accumulate a kXR_readv reply into one contiguous buffer ----
276 *
277 * WHAT: Reads the kXR_oksofar* then terminal kXR_ok frames of an in-flight
278 * kXR_readv response into a single heap buffer, returning 0 with *out_acc /
279 * *out_len set (caller frees *out_acc) or -1 on recv/allocation/size-cap
280 * failure (buffer freed internally, so the caller must NOT free on -1).
281 *
282 * WHY: The readv reply may be split across many kXR_oksofar frames; segment
283 * scatter needs the whole reply as one linear span. Splitting accumulation out
284 * of brix_file_readv keeps both the receive loop and the segment walk below the
285 * complexity cap without altering the wire behaviour.
286 *
287 * HOW:
288 * 1. Loop receiving frames with brix_recv into a growable buffer `acc`.
289 * 2. Reject any reply that would exceed XRDC_VEC_MAXBYTES.
290 * 3. Grow capacity by doubling (from 65536) when the next frame won't fit,
291 * preserving already-accumulated bytes via realloc.
292 * 4. Append each frame body, then stop once a kXR_ok status is seen.
293 */
294static int
295brix_readv_accumulate(brix_conn *c, uint16_t sid, uint8_t **out_acc,
296 size_t *out_len, brix_status *st)
297{
298 uint8_t *acc = NULL;
299 size_t acc_len = 0, acc_cap = 0;
300 uint16_t status;
301
302 for (;;) { /* accumulate kXR_oksofar* then kXR_ok */
303 uint8_t *body = NULL;
304 uint32_t blen = 0;
305 brix_resp_out out = { &status, &body, &blen };
306 if (brix_recv(c, sid, &out, st) != 0) {
307 free(acc);
308 return -1;
309 }
310 if (blen > 0) {
311 if (acc_len + blen > XRDC_VEC_MAXBYTES) {
312 free(body); free(acc);
313 brix_status_set(st, XRDC_EPROTO, 0, "readv: response too large");
314 return -1;
315 }
316 if (acc_len + blen > acc_cap) {
317 size_t nc = acc_cap ? acc_cap : 65536;
318 uint8_t *na;
319 while (nc < acc_len + blen) { nc *= 2; }
320 na = (uint8_t *) realloc(acc, nc);
321 if (na == NULL) {
322 free(body); free(acc);
323 brix_status_set(st, XRDC_EPROTO, 0, "readv: out of memory");
324 return -1;
325 }
326 acc = na; acc_cap = nc;
327 }
328 memcpy(acc + acc_len, body, blen);
329 acc_len += blen;
330 }
331 free(body);
332 if (status == kXR_ok) {
333 break;
334 }
335 }
336 *out_acc = acc;
337 *out_len = acc_len;
338 return 0;
339}

342/* ---- Scatter an accumulated kXR_readv reply into caller segment buffers ----
343 *
344 * WHAT: Walks the accumulated reply `acc` (length acc_len) in request order,
345 * copying each segment's data into segs[i].buf and recording the actual byte
346 * count in segs[i].got. Returns the total bytes scattered, or -1 on a truncated
347 * segment header or body. Does not free `acc` (the caller owns it).
348 *
349 * WHY: The reply interleaves, per segment, a 16-byte readahead_list header
350 * carrying the ACTUAL read length followed by that many data bytes; segments
351 * must be recovered in request order. Isolating this walk from the receive loop
352 * keeps brix_file_readv under the complexity cap with no behaviour change.
353 *
354 * HOW:
355 * 1. For each of the nseg segments, verify a full 16-byte header remains.
356 * 2. Decode the per-segment length rl via the shared segment codec.
357 * 3. Verify rl data bytes remain, then copy min(rl, segs[i].len) into the
358 * segment buffer and set segs[i].got.
359 * 4. Advance the cursor past the data and sum the copied bytes.
360 */
361static ssize_t
362brix_readv_scatter(const uint8_t *acc, size_t acc_len,
363 brix_readv_seg *segs, size_t nseg, brix_status *st)
364{
365 size_t i, cursor = 0;
366 ssize_t total = 0;
367
368 for (i = 0; i < nseg; i++) { /* [readahead_list 16B][data rlen] per seg */
369 uint32_t rl;
370 size_t cp;
371 if (cursor + BRIX_READV_SEGSIZE > acc_len) {
372 brix_status_set(st, XRDC_EPROTO, 0, "readv: truncated segment header");
373 return -1;
374 }
375 rl = brix_readv_seg_rlen(acc + cursor); /* shared segment codec */
376 cursor += BRIX_READV_SEGSIZE;
377 if (cursor + rl > acc_len) {
378 brix_status_set(st, XRDC_EPROTO, 0, "readv: truncated segment data");
379 return -1;
380 }
381 cp = (rl <= segs[i].len) ? rl : segs[i].len;
382 memcpy(segs[i].buf, acc + cursor, cp);
383 segs[i].got = cp; /* actual bytes for this segment (≤ len) */
384 total += (ssize_t) cp;
385 cursor += rl;
386 }
387 return total;
388}

391/*
392 * kXR_readv (3025) — scatter-gather read. The request payload is nseg readahead_list
393 * entries (fhandle[4] + rlen[4 BE] + offset[8 BE]); the response interleaves, per
394 * segment, a 16-byte header (with the ACTUAL read length) then that many data bytes,
395 * possibly split across kXR_oksofar frames. We accumulate the whole reply, then walk
396 * the segments in request order into each seg.buf. Returns total bytes read.
397 */
398ssize_t
399brix_file_readv(brix_conn *c, brix_file *f, brix_readv_seg *segs,
400 size_t nseg, brix_status *st)
401{
402 ClientRequestHdr req;
403 uint8_t *payload;
404 uint16_t sid;
405 uint8_t *acc = NULL;
406 size_t acc_len = 0, i;
407 ssize_t total;
408
409 if (nseg == 0 || nseg > XRDC_VEC_MAXSEGS) {
410 brix_status_set(st, XRDC_EUSAGE, 0, "readv: bad segment count");
411 return -1;
412 }
413 payload = (uint8_t *) malloc(nseg * BRIX_READV_SEGSIZE);
414 if (payload == NULL) {
415 brix_status_set(st, XRDC_EPROTO, 0, "readv: out of memory");
416 return -1;
417 }
418 for (i = 0; i < nseg; i++) {
419 /* Shared segment-header codec — same layout the server packs/parses. */
420 brix_readv_seg_pack(payload + i * BRIX_READV_SEGSIZE, f->fhandle,
421 (uint32_t) segs[i].len, (uint64_t) segs[i].offset);
422 }
423 memset(&req, 0, sizeof(req));
424 req.requestid = htons(kXR_readv);
425 xrdw_empty_req_pack(((ClientRequestHdr *) &req)->body);
426 {
427 brix_payload pl = { payload, (uint32_t) (nseg * BRIX_READV_SEGSIZE) };
428 if (brix_send(c, &req, &pl, &sid, st) != 0) {
429 free(payload);
430 return -1;
431 }
432 }
433 free(payload);
434
435 if (brix_readv_accumulate(c, sid, &acc, &acc_len, st) != 0) {
436 return -1;
437 }
438
439 total = brix_readv_scatter(acc, acc_len, segs, nseg, st);
440 free(acc);
441 return total;
442}

445/*
446 * kXR_writev (3031) — scatter-gather write. Wire framing (stock
447 * XrdXrootdProtocol::do_WriteV): the header dlen covers ONLY the nseg*16-byte
448 * write_list descriptor block (fhandle[4] + wlen[4 BE] + offset[8 BE]
449 * back-to-back); the concatenated segment data streams immediately after the
450 * frame and the server recovers its length as sum(wlen). Counting the data
451 * inside dlen is rejected by stock servers with kXR_ArgInvalid ("Write vector
452 * is invalid") followed by a link drop. do_sync sets kXR_wv_doSync (fsync
453 * each touched handle). The write is all-or-nothing.
454 */
455int
456brix_file_writev(brix_conn *c, brix_file *f, const brix_writev_seg *segs,
457 size_t nseg, int do_sync, brix_status *st)
458{
459 ClientWriteVRequest req;
460 uint8_t *payload, *p;
461 uint16_t sid, status;
462 uint8_t *body = NULL;
463 uint32_t blen = 0;
464 size_t i, total_data = 0, plen;
465
466 if (nseg == 0 || nseg > XRDC_VEC_MAXSEGS) {
467 brix_status_set(st, XRDC_EUSAGE, 0, "writev: bad segment count");
468 return -1;
469 }
470 /* Bound the aggregate payload so the size_t sum can't wrap and plen stays
471 * inside the uint32_t wire dlen field (256MiB, matching readv's response cap). */
472 for (i = 0; i < nseg; i++) {
473 if (segs[i].len > XRDC_VEC_MAXBYTES
474 || total_data > XRDC_VEC_MAXBYTES - segs[i].len) {
475 brix_status_set(st, XRDC_EUSAGE, 0,
476 "writev: payload exceeds %u bytes", XRDC_VEC_MAXBYTES);
477 return -1;
478 }
479 total_data += segs[i].len;
480 }
481 plen = nseg * 16 + total_data;
482 payload = (uint8_t *) malloc(plen);
483 if (payload == NULL) {
484 brix_status_set(st, XRDC_EPROTO, 0, "writev: out of memory");
485 return -1;
486 }
487 for (i = 0; i < nseg; i++) { /* descriptor block */
488 uint8_t *e = payload + i * 16;
489 uint32_t wl = htonl((uint32_t) segs[i].len);
490 uint64_t of = htobe64((uint64_t) segs[i].offset);
491 memcpy(e, f->fhandle, XRD_FHANDLE_LEN);
492 memcpy(e + 4, &wl, 4);
493 memcpy(e + 8, &of, 8);
494 }
495 p = payload + nseg * 16; /* concatenated data block */
496 for (i = 0; i < nseg; i++) {
497 memcpy(p, segs[i].data, segs[i].len);
498 p += segs[i].len;
499 }
500 memset(&req, 0, sizeof(req));
501 req.requestid = htons(kXR_writev);
502 {
503 xrdw_writev_req_t b = { .options = (uint8_t) (do_sync ? kXR_wv_doSync : 0) };
504 xrdw_writev_req_pack(&b, ((ClientRequestHdr *) &req)->body);
505 }
506 /* dlen frames only the descriptor block; the data bytes ride after it. */
507 {
508 brix_payload_ext pe = { payload, (uint32_t) plen,
509 (uint32_t) (nseg * 16) };
510 if (brix_send_ext(c, &req, &pe, &sid, st) != 0) {
511 free(payload);
512 return -1;
513 }
514 }
515 free(payload);
516 {
517 brix_resp_out out = { &status, &body, &blen };
518 if (brix_recv(c, sid, &out, st) != 0) {
519 return -1;
520 }
521 }
522 free(body);
523 return 0;
524}
