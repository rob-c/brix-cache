11/*
12 * WHAT: Per-level make-directory result codes shared by the recursive-mkdir
13 * driver. A level is either freshly created, already present (EEXIST),
14 * or a hard error.
15 *
16 * WHY: The three public variants (plain / confined-canon / beneath) differ
17 * only in HOW one level is created and in whether policy runs on EEXIST.
18 * Normalising the syscall outcome to these three states lets one shared
19 * walk drive all of them while each variant keeps identical semantics.
20 *
21 * HOW: Backend hooks return BRIX_MKDIR_CREATED on a fresh mkdir,
22 * BRIX_MKDIR_EXISTED when the syscall failed with EEXIST, and
23 * BRIX_MKDIR_ERROR for any other failure (errno preserved).
24 */
25#define BRIX_MKDIR_ERROR (-1)
26#define BRIX_MKDIR_CREATED (0)
27#define BRIX_MKDIR_EXISTED (1)

29/*
30 * WHAT: File-local descriptor bundling everything the recursive-mkdir walk
31 * needs: the level-creation backend, its bound arguments, the group
32 * policy rules, and whether a policy failure is fatal.
33 *
34 * WHY: Passing this single context keeps every helper at or below the
35 * five-parameter limit and makes the per-variant behaviour explicit and
36 * data-driven rather than duplicated across three near-identical loops.
37 *
38 * HOW: make_level() is invoked for each prefix in tmp and returns a
39 * BRIX_MKDIR_* code. root_len is the length of root_canon and is used by
40 * the beneath backend to derive the export-relative path. policy_on_exist
41 * selects the two behavioural regimes: the plain variant applies policy
42 * only on fresh create and treats failure as fatal (policy_on_exist=0);
43 * the confined variants apply policy best-effort on every level whenever
44 * rules are set and ignore the result (policy_on_exist=1).
45 */

58/*
59 * WHAT: Copy <src> into a caller-owned tmp[PATH_MAX] buffer, stripping a single
60 * trailing '/', and reject over-long paths.
61 *
62 * WHY: All three variants begin with byte-identical normalisation; sharing it
63 * removes duplication and guarantees the length/ENAMETOOLONG guard stays
64 * consistent (a divergence here could silently truncate a confined path).
65 *
66 * HOW: snprintf into tmp; on overflow set errno=ENAMETOOLONG and return -1.
67 * Store the resulting length via *out_len and drop a lone trailing slash.
68 * Returns 0 on success.
69 */
70static int
71brix_mkdir_copy_tmp(char *tmp, const char *src, int *out_len)
72{
73 int n;
74
75 n = snprintf(tmp, PATH_MAX, "%s", src);
76 if (n < 0 || (size_t) n >= (size_t) PATH_MAX) {
77 errno = ENAMETOOLONG;
78 return -1;
79 }
80
81 if (n > 0 && tmp[n - 1] == '/') {
82 tmp[n - 1] = '\0';
83 n--;
84 }
85
86 *out_len = n;
87 return 0;
88}

90/*
91 * WHAT: Validate that the normalised <tmp> lies under <root_canon> and return a
92 * pointer to the first character after the root prefix (the start of the
93 * relative portion to walk).
94 *
95 * WHY: This is the security-critical confinement gate shared by the confined
96 * and beneath variants: a resolved path that is not a sub-path of the
97 * export root must be rejected (EXDEV) so recursive creation can never
98 * escape the export even if a symlink was swapped in after resolution.
99 *
100 * HOW: For a root of "/" the whole path (after the leading slash) is relative.
101 * Otherwise require a byte-exact root prefix followed by end-of-string or
102 * '/'; a mismatch yields EXDEV, an exact match (path == root) yields
103 * EEXIST. Returns the walk-start pointer, or NULL with errno set.
104 */
105static char *
106brix_mkdir_confined_start(char *tmp, const char *root_canon, size_t root_len)
107{
108 if (root_len == 1 && root_canon[0] == '/') {
109 return tmp + 1;
110 }
111
112 if (strncmp(tmp, root_canon, root_len) != 0
113 || (tmp[root_len] != '\0' && tmp[root_len] != '/'))
114 {
115 errno = EXDEV;
116 return NULL;
117 }
118
119 if (tmp[root_len] == '\0') {
120 errno = EEXIST;
121 return NULL;
122 }
123
124 return tmp + root_len + 1;
125}

127/*
128 * WHAT: Apply parent-group policy to a just-processed directory level per the
129 * walk's behavioural regime.
130 *
131 * WHY: The plain variant enforces policy only on freshly-created levels and
132 * fails the whole operation if policy application errors; the confined
133 * variants apply policy best-effort on every level and ignore the result.
134 * Centralising the branch keeps those two contracts byte-identical to the
135 * original inline code.
136 *
137 * HOW: No-op when rules are unset. When policy_on_exist is false, run only if
138 * the level was freshly created and propagate NGX_ERROR as a fatal -1.
139 * When policy_on_exist is true, run unconditionally and discard the
140 * result. Returns 0 to continue, -1 to abort.
141 */
142static int
143brix_mkdir_apply_policy(const brix_mkdir_walk_t *w, const char *dir, int made)
144{
145 if (w->rules == NULL) {
146 return 0;
147 }
148
149 if (!w->policy_on_exist) {
150 if (made != BRIX_MKDIR_CREATED) {
151 return 0;
152 }
153 if (brix_apply_parent_group_policy_path(w->log, dir, w->rules)
154 == NGX_ERROR)
155 {
156 return -1;
157 }
158 return 0;
159 }
160
161 (void) brix_apply_parent_group_policy_path(w->log, dir, w->rules);
162 return 0;
163}

165/*
166 * WHAT: Create one directory level <dir> and apply policy for it.
167 *
168 * WHY: Every prefix and the final target go through the same create-then-policy
169 * sequence; factoring it into one helper removes the four inline copies
170 * and keeps the EEXIST-is-success rule in a single place.
171 *
172 * HOW: Invoke the backend make_level hook. A BRIX_MKDIR_ERROR aborts with the
173 * backend's errno preserved. Otherwise defer to brix_mkdir_apply_policy()
174 * with the create/existed outcome. Returns 0 to continue, -1 to abort.
175 */
176static int
177brix_mkdir_one_level(const brix_mkdir_walk_t *w, const char *dir)
178{
179 int made;
180
181 made = w->make_level(w, dir);
182 if (made == BRIX_MKDIR_ERROR) {
183 return -1;
184 }
185
186 return brix_mkdir_apply_policy(w, dir, made);
187}

189/*
190 * WHAT: Walk <tmp> from the relative-portion start pointer <p> to the target,
191 * creating every intermediate directory and then the target itself.
192 *
193 * WHY: This is the single shared component walk behind all three public
194 * variants — the only place the '/'-by-'/' truncation logic lives, so a
195 * fix or audit touches one code path.
196 *
197 * HOW: Advance p; at each '/', temporarily terminate to expose the prefix,
198 * create that level, then restore the '/' before continuing (restoring
199 * even on the error path so the buffer stays well-formed). After the loop
200 * create the full path once more for the final level. Returns 0 / -1.
201 */
202static int
203brix_mkdir_walk(const brix_mkdir_walk_t *w, char *tmp, char *p)
204{
205 for (; *p; p++) {
206 if (*p != '/') {
207 continue;
208 }
209
210 *p = '\0';
211 if (brix_mkdir_one_level(w, tmp) != 0) {
212 *p = '/';
213 return -1;
214 }
215 *p = '/';
216 }
217
218 return brix_mkdir_one_level(w, tmp);
219}

221/*
222 * WHAT: make_level backend for the unconstrained variant — a plain mkdir(2).
223 *
224 * WHY: Used where the caller only needs directory creation without export
225 * confinement (e.g. rename intermediate directories).
226 *
227 * HOW: mkdir(dir, mode); map success to CREATED, EEXIST to EXISTED, any other
228 * failure to ERROR with errno intact.

230static int
231brix_mkdir_level_plain(const brix_mkdir_walk_t *w, const char *dir)

293 * WHAT: Creates directories recursively from root to target path, applying parent group policy at each newly-created intermediate level when log and rules are provided. Creating each level individually ensures intermediate directories inherit the correct parent group policy from their newly-created parent, preventing misconfigured permissions on nested paths. The policy application happens only when mkdir succeeds (not when EEXIST is returned) — existing parents retain their original policy.
296 *
297 * HOW: Normalise path into tmp[PATH_MAX] (strip trailing '/'), then drive the shared component walk from tmp+1 with the plain-mkdir backend and fatal-on-create policy (policy_on_exist=0). 299int
300brix_mkdir_recursive_policy(const char *path, mode_t mode,

302{
