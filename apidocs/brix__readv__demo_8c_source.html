1/*
2 * brix_readv_demo.c — exercise libbrix's scatter-gather read (kXR_readv) with
3 * caller-chosen, variably-sized segments.
4 *
5 * Issues ONE brix_file_readv() for every "<off>:<len>" segment given on the
6 * command line, writes each segment's actually-delivered bytes (in request
7 * order, concatenated) to <outfile>, and prints one line per segment:
8 * seg <i> <off> <req_len> <got>
9 * plus a trailing "total <bytes>". `got` may be < req_len when the server caps
10 * the element (brix_readv_segment_size) or the read runs short of EOF — which
11 * is exactly the variable-block behaviour a test wants to verify.
12 *
13 * cc brix_readv_demo.c $(pkg-config --cflags --libs libbrix) -o demo
14 */

54 }

56 segs[i].len = (size_t) len;
57 segs[i].buf = malloc(len ? len : 1); /* caller-owned, >= len */
58 if (segs[i].buf == NULL) { fprintf(stderr, "oom\n"); return 1; }
59 }
60
61 brix_status_clear(&st);
62 if (brix_url_parse(argv[1], &u, &st) != 0) {
63 fprintf(stderr, "url: %s\n", st.msg); return 1;
64 }
65 if (brix_connect(&c, &u, NULL, &st) != 0) {
66 fprintf(stderr, "connect: %s\n", st.msg); return 1;
67 }
68 if (brix_file_open_read(&c, argv[2], &f, &st) != 0) {
69 fprintf(stderr, "open: %s\n", st.msg); brix_close(&c); return 1;
70 }
71
72 total = brix_file_readv(&c, &f, segs, nseg, &st);
73 if (total < 0) {
74 fprintf(stderr, "readv: %s\n", st.msg);
75 brix_file_close(&c, &f, &st); brix_close(&c); return 1;
76 }
77
78 out = fopen(argv[3], "wb");
79 if (out == NULL) { fprintf(stderr, "cannot open outfile\n"); return 1; }
80 for (i = 0; i < nseg; i++) {
81 printf("seg %zu %lld %zu %zu\n", i, (long long) segs[i].offset,
82 segs[i].len, segs[i].got);
83 if (segs[i].got > 0) {
84 fwrite(segs[i].buf, 1, segs[i].got, out);
85 }
86 free(segs[i].buf);
87 }
88 fclose(out);
89 printf("total %zd\n", total);
90 free(segs);
91
92 brix_file_close(&c, &f, &st);
93 brix_close(&c);
