1/*
2 * crc64.c - parameterized 64-bit CRC kernel (pure, ngx-free).
3 *
4 * WHAT: CRC-64/XZ and CRC-64/NVME, both reflected with init/xorout all-FF, via a
5 * 256-entry reflected lookup table per variant; plus a zlib-style GF(2)
6 * combine for deriving a whole-object CRC from per-part CRCs (S3 multipart
7 * FULL_OBJECT). See crc64.h for the variant/polynomial/encoding contract.
8 * WHY: Stock XRootD ships no crc64 calculator; this is the single shared engine
9 * behind kXR_Qcksum, WebDAV Digest, S3 x-amz-checksum-crc64nvme and the
10 * native client — the crc32c.c analogue for 64-bit checksums.
11 * HOW: Tables are filled once at load by a constructor (single-threaded, before
12 * any worker thread forks, like crc32c's hw-table init), so the hot path is
13 * table lookups + XORs. No goto; no hardware path (x86 baseline has no
14 * CRC64 instruction). Correctness of each table is asserted in the unit
15 * test tests/unit/test_crc64.c against the published check constants
16 * (XZ 0x995DC9BBDF1939FA, NVME 0xAE8B14860A799888).
17 */
18#include "crc64.h"

20/*
21 * Reflected (bit-reversed) polynomials. The CRC catalogue lists the NORMAL
22 * (MSB-first) forms — XZ 0x42F0E1EBA9EA3693, NVME 0xAD93D23594C93659 — but a
23 * reflected (right-shifting) table-driven implementation uses the bit-reversed
24 * constant. XZ's reflected form is identical to Go's crc64.ECMA; NVME's matches
25 * the Linux kernel crc64_rocksoft polynomial.
26 */
27#define BRIX_CRC64_XZ_POLY_REFL 0xC96C5795D7870F42ULL
28#define BRIX_CRC64_NVME_POLY_REFL 0x9A6C9329AC4BC9B5ULL
29
30#define BRIX_CRC64_TABLE_SIZE 256
31#define BRIX_CRC64_GF2_DIM 64
32
33static uint64_t brix_crc64_xz_tab[BRIX_CRC64_TABLE_SIZE];
34static uint64_t brix_crc64_nvme_tab[BRIX_CRC64_TABLE_SIZE];

36/* Build the 256-entry reflected table for one reflected polynomial. */
37static void
38brix_crc64_build_table(uint64_t *tab, uint64_t poly_refl)
39{
40 int i, j;
41
42 for (i = 0; i < BRIX_CRC64_TABLE_SIZE; i++) {
43 uint64_t crc = (uint64_t) i;
44
45 for (j = 0; j < 8; j++) {
46 crc = (crc & 1) ? (crc >> 1) ^ poly_refl : (crc >> 1);
47 }
48 tab[i] = crc;
49 }
50}

80uint64_t
81brix_crc64_extend(brix_crc64_variant_t variant, uint64_t crc,
82 const void *buf, size_t len)
83{
84 const unsigned char *p;
85 const uint64_t *tab;
86
87 tab = brix_crc64_table_for(variant);
88 if (tab == NULL) {
89 return crc;
90 }
91 if (buf == NULL && len != 0) {
92 return crc;
93 }
94
95 p = (const unsigned char *) buf;
96
97 crc = ~crc; /* init = all-FF (rolling-safe) */
98 while (len--) {
99 crc = tab[(crc ^ *p++) & 0xFFu] ^ (crc >> 8);
100 }
101 return ~crc; /* xorout = all-FF */
102}

110/*
111 * GF(2) linear-algebra combine (the zlib crc32_combine construction at 64-bit
112 * width). A CRC register update over a run of zero bytes is a linear operator on
113 * the 64-bit state; combining crc(A) with crc(B) means advancing crc(A) across
114 * |B| zero bytes, then XORing crc(B). Operators are 64x64 bit-matrices stored as
115 * uint64_t[64] (column n = image of basis vector e_n).
116 */
117static uint64_t
118brix_crc64_gf2_times(const uint64_t *mat, uint64_t vec)
119{
120 uint64_t sum = 0;
121
122 while (vec) {
123 if (vec & 1) {
124 sum ^= *mat;
125 }
126 vec >>= 1;
127 mat++;
128 }
129 return sum;
130}

132/* square = mat * mat over GF(2). */
133static void
134brix_crc64_gf2_square(uint64_t *square, const uint64_t *mat)
135{
136 int n;
137
138 for (n = 0; n < BRIX_CRC64_GF2_DIM; n++) {
139 square[n] = brix_crc64_gf2_times(mat, mat[n]);
140 }
141}

143uint64_t
144brix_crc64_combine(brix_crc64_variant_t variant, uint64_t crc1,
145 uint64_t crc2, uint64_t len2)
146{
147 uint64_t even[BRIX_CRC64_GF2_DIM]; /* even-power-of-two zeros operator */
148 uint64_t odd[BRIX_CRC64_GF2_DIM]; /* odd-power-of-two zeros operator */
149 uint64_t poly_refl;
150 uint64_t row;
151 int n;
152
153 poly_refl = brix_crc64_poly_for(variant);
154 if (poly_refl == 0 || len2 == 0) {
155 return crc1;
156 }
157
158 /* Operator for a single shifted-in zero bit. */
159 odd[0] = poly_refl;
160 row = 1;
161 for (n = 1; n < BRIX_CRC64_GF2_DIM; n++) {
162 odd[n] = row;
163 row <<= 1;
164 }
165
166 brix_crc64_gf2_square(even, odd); /* two zero bits */
167 brix_crc64_gf2_square(odd, even); /* four zero bits */
168
169 /* Apply len2 zero BYTES to crc1: the first square below yields the operator
170 * for one zero byte (eight zero bits) in `even`, then each loop iteration
171 * doubles the zero-run, consuming one bit of len2 per half-iteration. */
172 do {
173 brix_crc64_gf2_square(even, odd);
174 if (len2 & 1) {
175 crc1 = brix_crc64_gf2_times(even, crc1);
176 }
177 len2 >>= 1;
178 if (len2 == 0) {
179 break;
180 }
181 brix_crc64_gf2_square(odd, even);
182 if (len2 & 1) {
183 crc1 = brix_crc64_gf2_times(odd, crc1);
184 }
185 len2 >>= 1;
186 } while (len2 != 0);
187
188 return crc1 ^ crc2;
189}
