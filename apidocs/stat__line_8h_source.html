1/*
2 * stat_line.h — XRootD stat-response line grammar (single source of truth).
3 *
4 * WHAT: the ASCII body of a kXR_stat / kXR_statx reply (and each kXR_dstat
5 * directory entry) is a space-separated line:
6 * "<id> <size> <flags> <mtime>"
7 * with an OPTIONAL extended tail some servers (EOS) append:
8 * " <ctime> <atime> <mode-octal> <owner> <group>"
9 * This header holds BOTH directions of that one grammar:
10 * - brix_statline_format: fields -> line (the server is the encoder)
11 * - brix_statline_parse: line -> fields (the client is the decoder)
12 * WHY: the encoder lived in the module (src/path/stat_body.c, via snprintf) and
13 * the decoder lived in the native client (client/lib/ops_meta.c, via sscanf)
14 * with NOTHING but human discipline keeping the two in step. They are exact
15 * inverse operations on a wire-visible textual spec: if the field order, the
16 * octal-mode encoding, or the mtime units ever drift on one side, stat/dirlist
17 * interop breaks silently. Co-locating both directions (with a round-trip
18 * contract) makes the grammar a single audited artifact.
19 * HOW: header-only static inlines — no ngx, no allocation, no OpenSSL — so the
20 * same code compiles into both the nginx module and the ngx-free client.
21 * The flag *values* (kXR_isDir/kXR_readable/...) are caller policy and stay
22 * at the call sites; this header only owns the line's lexical shape.
23 *
24 * ROUND-TRIP CONTRACT: for any (id, size, flags, mtime),
25 * brix_statline_parse(buf-from-brix_statline_format(id,size,flags,mtime))
26 * recovers the same four mandatory fields (have_ext == 0).
27 *
28 * Clean-room: grammar from src/protocol wire docs (vs XProtocol stat reply).
29 */

37/* Optional extended tail (EOS long-stat form). All zeroed when absent. */
38typedef struct {
39 int have_ext; /* 1 if the 5-field tail was present, else 0 */
40 long ctime;
41 long atime;
42 unsigned mode; /* decoded from the octal mode token on the wire */
43 char owner[64];
44 char group[64];
45} brix_statline_ext;

47/*
48 * Encode the mandatory 4-field stat line. Returns snprintf's value (the number of
49 * bytes that would be written, excluding the NUL) so callers can detect
50 * truncation. The integer widths mirror the historical wire encoding exactly:
51 * id as %llu, size as %lld, flags as %d, mtime as %ld.
52 */
53static inline int
54brix_statline_format(char *out, size_t outsz, unsigned long long id,
55 long long size, int flags, long mtime)
56{
57 return snprintf(out, outsz, "%llu %lld %d %ld", id, size, flags, mtime);
58}
59
60/*
61 * Decode a stat line. Fills the four mandatory fields (any out-pointer may be
62 * NULL) and, when `ext` is non-NULL, the optional EOS tail (have_ext records
63 * whether it was present; the mode token is interpreted as octal). Returns 0 on
64 * success, -1 if the line does not carry at least the four mandatory fields.
65 */
66static inline int
67brix_statline_parse(const char *s, uint64_t *id, int64_t *size,
68 int *flags, long *mtime, brix_statline_ext *ext)
69{
70 unsigned long long pid = 0;
71 long long psize = 0;
72 int pflags = 0;
73 long pmtime = 0, pctime = 0, patime = 0;
74 char modebuf[32], owner[64], group[64];
75 int nf;
76
77 modebuf[0] = owner[0] = group[0] = '\0';
78
79 nf = sscanf(s, "%llu %lld %d %ld %ld %ld %31s %63s %63s",
80 &pid, &psize, &pflags, &pmtime,
81 &pctime, &patime, modebuf, owner, group);
82 if (nf < 4) {
83 return -1;
84 }
85
86 if (id != NULL) { *id = (uint64_t) pid; }
87 if (size != NULL) { *size = (int64_t) psize; }
88 if (flags != NULL) { *flags = pflags; }
89 if (mtime != NULL) { *mtime = pmtime; }
90
91 if (ext != NULL) {
92 ext->have_ext = 0;
93 ext->ctime = 0;
94 ext->atime = 0;
95 ext->mode = 0;
96 ext->owner[0] = '\0';
97 ext->group[0] = '\0';
98
99 if (nf >= 9) {
100 ext->have_ext = 1;
101 ext->ctime = pctime;
102 ext->atime = patime;
103 ext->mode = (unsigned) strtoul(modebuf, NULL, 8);
104 snprintf(ext->owner, sizeof(ext->owner), "%s", owner);
105 snprintf(ext->group, sizeof(ext->group), "%s", group);
106 }
107 }
108
109 return 0;
110}
