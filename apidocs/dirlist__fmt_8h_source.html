1/*
2 * dirlist_fmt.h — kXR_dirlist response envelope grammar (single source of truth).
3 *
4 * WHAT: the dstat lead-in sentinel that a kXR_dirlist reply carries when the
5 * request set kXR_dstat / kXR_dcksm. The body then begins with this fixed
6 * 10-byte sentinel, followed by "<name>\n<statline>\n" pairs (the stat line
7 * itself is owned by protocol/stat_line.h).
8 * WHY: the server EMITS the sentinel (twice — the streaming and buffered dirlist
9 * paths in src/dirlist/handler.c) and the client MATCHES its 9-byte prefix
10 * to decide whether to enter stat-pairing mode (client/lib/ops_meta.c). The
11 * literal was hand-written in all three places; the stock XRootD client keys
12 * on exactly the 9-byte prefix `.\n0 0 0 0`, so a stray byte here silently
13 * drops the whole listing into "every line is a filename" mode. One
14 * definition keeps emit and match in lockstep.
15 * HOW: header-only macros — no ngx, no allocation — so the same constant compiles
16 * into the nginx module and the ngx-free client. The match length is derived
17 * from the emit literal (lead-in minus its trailing newline) so the two can
18 * never drift.
19 *
20 * Clean-room: sentinel from XProtocol dirlist dstat docs (DirectoryList::dStatPrefix).
21 */
22#ifndef BRIX_PROTOCOL_DIRLIST_FMT_H
23#define BRIX_PROTOCOL_DIRLIST_FMT_H
24
25#include <stddef.h>
26
27/* What the server prepends to a dstat/dcksm listing (10 bytes incl. trailing \n). */
28#define BRIX_DSTAT_LEADIN ".\n0 0 0 0\n"
29#define BRIX_DSTAT_LEADIN_LEN (sizeof(BRIX_DSTAT_LEADIN) - 1)
30
31/* What the client matches at the head of the body to detect stat-pairing mode:
32 * the lead-in without its trailing newline (9 bytes). Derived, never re-spelled. */
33#define BRIX_DSTAT_PREFIX_LEN (BRIX_DSTAT_LEADIN_LEN - 1)
34
35#endif /* BRIX_PROTOCOL_DIRLIST_FMT_H */
