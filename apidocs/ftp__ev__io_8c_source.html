5/*
6 * ftp_ev_io.c — the non-blocking control-channel engine.
7 *
8 * WHAT: the session entry point installed as the stream content handler, the
9 * read/write event handlers, the buffered-reply flusher, the command-framing
10 * loop, and idempotent session teardown.
11 *
12 * WHY: this is the piece that makes the gateway event-driven. Where the sync
13 * engine blocks in read()/write(), here every I/O either completes or yields
14 * NGX_AGAIN back to nginx's event loop, so one worker interleaves many sessions.
15 *
16 * HOW: brix_ftp_ev_process() is the heart — a single loop that (1) drains any
17 * queued reply, (2) frames one complete command line out of the inbound buffer
18 * and dispatches it, or (3) reads more bytes. It runs until an operation would
19 * block (arm the matching event + idle timer and return) or the session ends.
20 * Both event handlers funnel into it, so the flush-then-frame ordering is
21 * identical whether we were woken to read or to write.
22 */
23

25/* Try to frame one CRLF-terminated command line out of buf[bpos..blen).
26 *
27 * On success returns a NUL-terminated pointer into ->buf (CRLF stripped) and
28 * advances ->bpos past the line. Returns NULL when no complete line is buffered
29 * yet; sets *too_long when the buffer is full with no newline in sight (an
30 * unterminated over-long line — the caller drops the session). */
31static char *
32ev_take_line(ftp_ev_t *fc, int *too_long)
33{
34 u_char *nl;
35 size_t linelen;
36
37 *too_long = 0;
38
39 if (fc->bpos < fc->blen) {
40 nl = memchr(fc->buf + fc->bpos, '\n', fc->blen - fc->bpos);
41 if (nl != NULL) {
42 u_char *line = fc->buf + fc->bpos;
43 linelen = (size_t) (nl - line); /* bytes before the '\n' */
44 fc->bpos += linelen + 1; /* consume through '\n' */
45 if (linelen > 0 && line[linelen - 1] == '\r') {
46 linelen--; /* strip CR */
47 }
48 line[linelen] = '\0';
49 return (char *) line;
50 }
51 }
52
53 /* No newline in the buffered bytes. Compact the consumed prefix so the next
54 * recv() has room; if the whole buffer is then full, the peer sent an
55 * over-long unterminated line. */
56 if (fc->bpos > 0) {
57 ngx_memmove(fc->buf, fc->buf + fc->bpos, fc->blen - fc->bpos);
58 fc->blen -= fc->bpos;
59 fc->bpos = 0;
60 }
61 if (fc->blen == sizeof(fc->buf)) {
62 *too_long = 1;
63 }
64 return NULL;
65}

68/* Send as much of the queued reply as the socket accepts. Returns NGX_OK once
69 * the buffer is fully drained (and reset), NGX_AGAIN if the socket is full, or
70 * NGX_ERROR on a send failure / peer close. */
71ngx_int_t
72brix_ftp_ev_flush(ftp_ev_t *fc)
73{
74 ngx_connection_t *c = fc->c;
75
76 while (fc->ob_pos < fc->ob_len) {
77 ssize_t n = c->send(c, fc->ob + fc->ob_pos, fc->ob_len - fc->ob_pos);
78 if (n > 0) {
79 fc->ob_pos += (size_t) n;
80 continue;
81 }
82 if (n == NGX_AGAIN) {
83 return NGX_AGAIN;
84 }
85 return NGX_ERROR; /* NGX_ERROR or 0 (EOF) */
86 }
87 fc->ob_pos = fc->ob_len = 0;
88 return NGX_OK;
89}

112/* The command→reply state machine. Drives one step at a time until an operation
113 * would block (arm the relevant event and return) or the session is finalized. */
114static void
115brix_ftp_ev_process(ftp_ev_t *fc)
116{
117 ngx_connection_t *c = fc->c;
118
119 for ( ;; ) {
120 char *line;
121 int too_long;
122 ngx_int_t rc;
123 ssize_t n;
124
125 /* (1) Drain any queued reply before touching the next command. */
126 if (fc->ob_len > fc->ob_pos) {
127 rc = brix_ftp_ev_flush(fc);
128 if (rc == NGX_AGAIN) {
129 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
130 brix_ftp_ev_finalize(fc, NGX_STREAM_INTERNAL_SERVER_ERROR);
131 }
132 ngx_add_timer(c->write, BRIX_FTP_EV_IO_TIMEO);
133 return;
134 }
135 if (rc == NGX_ERROR) {
136 brix_ftp_ev_finalize(fc, NGX_STREAM_OK);
137 return;
138 }
139 if (fc->state == FTP_EV_ST_CLOSING) { /* QUIT reply delivered */
140 brix_ftp_ev_finalize(fc, NGX_STREAM_OK);
141 return;
142 }
143 }
144
145 /* A data transfer holds the control channel half-open: once its 150 has
146 * drained, the data connection's own events drive the transfer to a 226/
147 * 550 (brix_ftp_ev_data_finish resumes us). Don't frame the next command
148 * meanwhile — FTP is half-duplex, and reading ahead would race the pump. */
149 if (fc->state == FTP_EV_ST_XFER) {
150 return;
151 }
152
153 /* (2) Frame and dispatch one complete command line. */
154 line = ev_take_line(fc, &too_long);
155 if (too_long) {
156 ngx_log_error(NGX_LOG_INFO, c->log, 0,
157 "brix: GridFTP(ev) over-long command line — closing");
158 brix_ftp_ev_finalize(fc, NGX_STREAM_OK);
159 return;
160 }
161 if (line != NULL) {
162 if (line[0] == '\0') {
163 continue; /* ignore blank line */
164 }
165 rc = brix_ftp_ev_dispatch(fc, line);
166 if (rc == NGX_ERROR) {
167 brix_ftp_ev_finalize(fc, NGX_STREAM_OK);
168 return;
169 }
170 if (rc == NGX_DONE) {
171 fc->state = FTP_EV_ST_CLOSING; /* flush, then close */
172 }
173 continue; /* loop: flush the reply */
174 }
175
176 /* (3) No complete line buffered — read more. */
177 n = c->recv(c, fc->buf + fc->blen, sizeof(fc->buf) - fc->blen);
178 if (n == NGX_AGAIN) {
179 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
180 brix_ftp_ev_finalize(fc, NGX_STREAM_INTERNAL_SERVER_ERROR);
181 return;
182 }
183 ngx_add_timer(c->read, BRIX_FTP_EV_IO_TIMEO);
184 return;
185 }
186 if (n == 0 || n == NGX_ERROR) {
187 brix_ftp_ev_finalize(fc, NGX_STREAM_OK);
188 return;
189 }
190 fc->blen += (size_t) n;
191 }
192}

258brix_ftp_ev_handler(ngx_stream_session_t *s)

265

269 return;
270 }

272 if (fc->ob == NULL) {
273 ngx_stream_finalize_session(s, NGX_STREAM_INTERNAL_SERVER_ERROR);
274 return;
275 }
276 fc->s = s;
277 fc->c = c;
278 fc->conf = conf;
279 fc->state = FTP_EV_ST_CMD;
280 fc->pasv_fd = -1;
281 fc->allo_size = -1; /* no ALLO declared yet */
282 fc->prot = 'C';
283 fc->cwd[0] = '/';
284 fc->cwd[1] = '\0';
285
286 ngx_stream_set_ctx(s, fc, ngx_stream_brix_ftp_module);

290
291 ngx_log_error(NGX_LOG_INFO, c->log, 0,

294
295 if (brix_ftp_ev_reply(fc, "220 BriX GridFTP Gateway ready\r\n") != NGX_OK) {
296 brix_ftp_ev_finalize(fc, NGX_STREAM_INTERNAL_SERVER_ERROR);

298 }

300

302}
