1/*
2 * WHAT: brix_read_compressed() serves a kXR_read for a handle that was opened
3 * with the opaque "?xrootd.compress=<codec>" while the server has
4 * brix_read_compress on. It reads a bounded plaintext window from the
5 * file, compresses it as ONE self-contained codec frame, and queues the
6 * compressed bytes as a single kXR_read response. The negotiating client
7 * (which learned the codec from the kXR_open reply cpsize/cptype) inflates
8 * the frame back to plaintext.
9 *
10 * WHY: This is an opt-in extension beyond stock XRootD (whose root:// data path
11 * is always plaintext) — the swiss-army goal of trading CPU for bytes on
12 * the wire for compressible data. It is deliberately ISOLATED from the
13 * proven hot read path in read.c: brix_handle_read() routes here only
14 * when read_codec != IDENTITY, so the default (uncompressed) path — its
15 * sendfile fast path, windowed streaming and AIO pipeline — is completely
16 * untouched and byte-identical. pgread/readv never call this, preserving
17 * the pgread kXR_status + per-page CRC32c invariant byte-for-byte.
18 *
19 * HOW: Each request is an independent whole-range frame, so reads stay offset-
20 * addressable and resumable (re-reading the same offset reproduces the
21 * same plaintext, hence an equivalent frame). The plaintext window is
22 * staged in ctx->rd.read_scratch and the codec output in ctx->rd.cmp_scratch —
23 * both raw-alloc'd session-lifetime keep-slots (brix_get_pool_scratch /
24 * brix_release_read_buffer). To stay event-loop friendly the VFS-core
25 * read is synchronous but clamped to BRIX_READ_CHUNK_MAX; a larger
26 * client request simply gets a short (but legal) read and asks again.
27 */

40/* Worst-case codec expansion bound for `plain` bytes: every supported codec
41 * (zlib/zstd/xz/brotli/bzip2) stays well under +50% + a small fixed header for
42 * any input, so this single bound is safe without a per-codec compressBound. */
43static size_t
44brix_codec_max_out(size_t plain)
45{
46 return plain + (plain / 2) + 4096;
47}
48
49/*
50 * Compress `in_len` plaintext bytes into `out` (capacity *out_cap) as one finished
51 * codec frame. Returns the produced length, or (size_t)-1 on failure. Grows the
52 * caller's rd.cmp_scratch via BRIX_GET_SCRATCH if the codec needs more room than
53 * the initial bound (defensive — the bound is sized so this never trips).
54 */
55static size_t
56brix_compress_frame(brix_ctx_t *ctx, ngx_connection_t *c,
57 uint8_t codec_id, const u_char *in, size_t in_len)
58{
59 brix_codec_stream_t *s;
60 size_t cap;
61 size_t in_pos = 0, out_pos = 0;
62 brix_codec_rc_t rc;
63
64 cap = brix_codec_max_out(in_len);
65
66 if (BRIX_GET_SCRATCH(ctx, c, rd.cmp_scratch, rd.cmp_scratch_size, cap) == NULL) {
67 return (size_t) -1;
68 }
69
70 s = brix_codec_open((brix_codec_id_t) codec_id,
71 BRIX_CODEC_DIR_COMPRESS, -1, NULL);
72 if (s == NULL) {
73 return (size_t) -1;
74 }
75
76 /* cap = brix_codec_max_out(in_len) is >= the worst-case compressed size for
77 * every supported codec, so the single output buffer never fills before END.
78 * A codec may still need several step() calls (e.g. lz4 consumes its input in
79 * chunks, returning OK with room remaining) — keep calling until END. Only a
80 * genuinely full buffer before END would be an overflow, which max_out rules
81 * out; treat it as an error rather than a (lossy) realloc-grow. */
82 for ( ;; ) {
83 size_t prev_in = in_pos;
84 size_t prev_out = out_pos;
85
86 rc = brix_codec_step(s, in, in_len, &in_pos,
87 ctx->rd.cmp_scratch, ctx->rd.cmp_scratch_size, &out_pos,
88 1 /* finish */);
89
90 if (rc == BRIX_CODEC_END) {
91 break;
92 }
93 if (rc != BRIX_CODEC_OK
94 || out_pos >= ctx->rd.cmp_scratch_size
95 || (in_pos == prev_in && out_pos == prev_out))
96 {
97 brix_codec_close(s);
98 return (size_t) -1;
99 }
100 /* OK with room remaining: the codec made progress but isn't finished;
101 * call again to drain the rest of the frame into the same buffer. */
102 }
103
104 brix_codec_close(s);
105 return out_pos;
106}

108ngx_int_t
109brix_read_compressed(brix_ctx_t *ctx, ngx_connection_t *c,
110 ngx_stream_brix_srv_conf_t *rconf,
111 int idx, off_t offset, size_t rlen)
112{
113 int fd = ctx->files[idx].fd;
114 uint8_t codec_id = ctx->files[idx].read_codec;
115 off_t file_size;
116 size_t data_total, clen;
117 u_char *plain;
118 ssize_t nread;
119 ngx_chain_t *rsp_chain;
120 ngx_int_t rc;
121
122 /*
123 * Determine readable extent. Read-only handles use the size cached at open
124 * (stable); writable handles re-stat so same-session writes are visible.
125 */
126 if (!ctx->files[idx].writable) {
127 file_size = ctx->files[idx].cached_size;
128 } else {
129 struct stat st;
130 if (fstat(fd, &st) != 0) {
131 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
132 ctx->files[idx].path, "-",
133 kXR_IOError, strerror(errno));
134 }
135 file_size = st.st_size;
136 }
137
138 /* Clamp to bytes present and to a single inline window (short reads are a
139 * legal XRootD response — the client re-reads from the next offset). */
140 if (offset >= file_size) {
141 data_total = 0;
142 } else {
143 off_t avail = file_size - offset;
144 data_total = (avail < (off_t) rlen) ? (size_t) avail : rlen;
145 }
146 if (data_total > BRIX_READ_CHUNK_MAX) {
147 data_total = BRIX_READ_CHUNK_MAX;
148 }
149
150 if (data_total == 0) {
151 /* EOF / empty range: an empty frame inflates to zero bytes. */
152 BRIX_OP_OK(ctx, BRIX_OP_READ);
153 return brix_send_ok(ctx, c, NULL, 0);
154 }
155
156 /* Stage the plaintext window. */
157 plain = BRIX_GET_SCRATCH(ctx, c, rd.read_scratch, rd.read_scratch_size,
158 data_total);
159 if (plain == NULL) {
160 return NGX_ERROR;
161 }
162
163 brix_prefetch_read_file(c->log, &ctx->files[idx], offset, data_total,
164 file_size);
165
166 {
167 brix_vfs_job_t job;
168
169 brix_vfs_job_read_init(&job, fd, offset, data_total, plain,
170 data_total, 0);
171 brix_vfs_io_execute(&job);
172 nread = job.nio;
173 if (job.io_errno != 0 || nread < 0) {
174 int err = job.io_errno != 0 ? job.io_errno : EIO;
175
176 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
177 ctx->files[idx].path, "-",
178 kXR_IOError, strerror(err));
179 }
180 }
181 data_total = (size_t) nread;
182
183 if (data_total == 0) {
184 BRIX_OP_OK(ctx, BRIX_OP_READ);
185 return brix_send_ok(ctx, c, NULL, 0);
186 }
187
188 /* Compress the window into rd.cmp_scratch as one self-contained frame. */
189 clen = brix_compress_frame(ctx, c, codec_id, plain, data_total);
190 if (clen == (size_t) -1) {
191 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
192 ctx->files[idx].path, "-",
193 kXR_IOError, "inline read compression failed");
194 }
195
196 /* Accounting mirrors the plaintext read path: counters reflect PLAINTEXT
197 * bytes served (what the client receives after inflate), not wire bytes. */
198 ctx->files[idx].bytes_read += data_total;
199 ctx->totals.bytes += data_total;
200 brix_rl_charge_ctx(ctx, clen); /* bandwidth: charge actual wire bytes */
201
202 if (ctx->files[idx].dashboard_slot >= 0 &&
203 ngx_brix_dashboard_shm_zone != NULL)
204 {
205 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
206 ctx->files[idx].dashboard_slot,
207 (ngx_atomic_int_t) data_total,
208 (int64_t) ngx_current_msec);
209 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
210 ctx->files[idx].dashboard_slot, "read");
211 }
212
213 if (rconf->access_log_fd != NGX_INVALID_FILE) {
214 char read_detail[80];
215
216 snprintf(read_detail, sizeof(read_detail), "%lld+%zu z=%zu",
217 (long long) offset, data_total, clen);
218 brix_log_access(ctx, c, "READ", ctx->files[idx].path,
219 read_detail, 1, 0, NULL, data_total);
220 }
221 BRIX_OP_OK(ctx, BRIX_OP_READ);
222
223 /* Frame the compressed bytes as a single kXR_read response over rd.cmp_scratch
224 * (a keep-slot, so brix_release_read_buffer leaves it for reuse). */
225 rsp_chain = brix_build_chunked_chain(ctx, c, ctx->rd.cmp_scratch, clen);
226 if (rsp_chain == NULL) {
227 return NGX_ERROR;
228 }
229
230 rc = brix_queue_response_chain(ctx, c, rsp_chain, ctx->rd.cmp_scratch);
231 if (rc != NGX_OK || ctx->state != XRD_ST_SENDING) {
232 brix_release_read_buffer(ctx, c, ctx->rd.cmp_scratch);
233 }
234 return rc;
235}
