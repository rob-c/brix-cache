1/*
2 * WHAT: brix_read_compressed() serves a kXR_read for a handle that was opened
3 * with the opaque "?xrootd.compress=<codec>" while the server has
4 * brix_read_compress on. It reads a bounded plaintext window from the
5 * file, compresses it as ONE self-contained codec frame, and queues the
6 * compressed bytes as a single kXR_read response. The negotiating client
7 * (which learned the codec from the kXR_open reply cpsize/cptype) inflates
8 * the frame back to plaintext.
9 *
10 * WHY: This is an opt-in extension beyond stock XRootD (whose root:// data path
11 * is always plaintext) — the swiss-army goal of trading CPU for bytes on
12 * the wire for compressible data. It is deliberately ISOLATED from the
13 * proven hot read path in read.c: brix_handle_read() routes here only
14 * when read_codec != IDENTITY, so the default (uncompressed) path — its
15 * sendfile fast path, windowed streaming and AIO pipeline — is completely
16 * untouched and byte-identical. pgread/readv never call this, preserving
17 * the pgread kXR_status + per-page CRC32c invariant byte-for-byte.
18 *
19 * HOW: Each request is an independent whole-range frame, so reads stay offset-
20 * addressable and resumable (re-reading the same offset reproduces the
21 * same plaintext, hence an equivalent frame). The plaintext window is
22 * staged in ctx->rd.read_scratch and the codec output in ctx->rd.cmp_scratch —
23 * both raw-alloc'd session-lifetime keep-slots (brix_get_pool_scratch /
24 * brix_release_read_buffer). To stay event-loop friendly the VFS-core
25 * read is synchronous but clamped to BRIX_READ_CHUNK_MAX; a larger
26 * client request simply gets a short (but legal) read and asks again.
27 */

40/* Worst-case codec expansion bound for `plain` bytes: every supported codec
41 * (zlib/zstd/xz/brotli/bzip2) stays well under +50% + a small fixed header for
42 * any input, so this single bound is safe without a per-codec compressBound. */
43static size_t
44brix_codec_max_out(size_t plain)
45{
46 return plain + (plain / 2) + 4096;
47}

49/*
50 * Compress `in_len` plaintext bytes into `out` (capacity *out_cap) as one finished
51 * codec frame. Returns the produced length, or (size_t)-1 on failure. Grows the
52 * caller's rd.cmp_scratch via BRIX_GET_SCRATCH if the codec needs more room than
53 * the initial bound (defensive — the bound is sized so this never trips).
54 */
55static size_t
56brix_compress_frame(brix_ctx_t *ctx, ngx_connection_t *c,
57 uint8_t codec_id, const u_char *in, size_t in_len)
58{
59 brix_codec_stream_t *s;
60 size_t cap;
61 size_t in_pos = 0, out_pos = 0;
62 brix_codec_rc_t rc;
63
64 cap = brix_codec_max_out(in_len);
65
66 if (BRIX_GET_SCRATCH(ctx, c, rd.cmp_scratch, rd.cmp_scratch_size, cap) == NULL) {
67 return (size_t) -1;
68 }
69
70 s = brix_codec_open((brix_codec_id_t) codec_id,
71 BRIX_CODEC_DIR_COMPRESS, -1, NULL);
72 if (s == NULL) {
73 return (size_t) -1;
74 }
75
76 /* cap = brix_codec_max_out(in_len) is >= the worst-case compressed size for
77 * every supported codec, so the single output buffer never fills before END.
78 * A codec may still need several step() calls (e.g. lz4 consumes its input in
79 * chunks, returning OK with room remaining) — keep calling until END. Only a
80 * genuinely full buffer before END would be an overflow, which max_out rules
81 * out; treat it as an error rather than a (lossy) realloc-grow. */
82 for ( ;; ) {
83 size_t prev_in = in_pos;
84 size_t prev_out = out_pos;
85
86 rc = brix_codec_step(s, in, in_len, &in_pos,
87 ctx->rd.cmp_scratch, ctx->rd.cmp_scratch_size, &out_pos,
88 1 /* finish */);
89
90 if (rc == BRIX_CODEC_END) {
91 break;
92 }
93 if (rc != BRIX_CODEC_OK
94 || out_pos >= ctx->rd.cmp_scratch_size
95 || (in_pos == prev_in && out_pos == prev_out))
96 {
97 brix_codec_close(s);
98 return (size_t) -1;
99 }
100 /* OK with room remaining: the codec made progress but isn't finished;
101 * call again to drain the rest of the frame into the same buffer. */
102 }
103
104 brix_codec_close(s);
105 return out_pos;
106}

108/* ---- Compute the readable extent and clamped inline window for this request ----
109 *
110 * WHAT: Resolves the handle's current file size into *file_size and the number
111 * of plaintext bytes to stage this request into *data_total (0 at/after
112 * EOF). Returns NGX_OK, or NGX_ERROR with *io_err set to the fstat errno
113 * when a writable handle cannot be re-stat'd.
114 *
115 * WHY: Read-only handles use the size cached at open (stable); writable handles
116 * re-stat so same-session writes are visible. Isolating this branch keeps
117 * the orchestrator flat and the size/clamp policy in one testable place.
118 *
119 * HOW: 1. Pick cached_size for read-only handles, else fstat the fd (errno out
120 * on failure). 2. Clamp the window to bytes present past offset — a
121 * short read is a legal XRootD response, the client re-reads the next
122 * offset. 3. Cap to a single inline window (BRIX_READ_CHUNK_MAX).
123 */
124static ngx_int_t
125brix_compressed_extent(brix_ctx_t *ctx, int idx, off_t offset, size_t rlen,
126 off_t *file_size, size_t *data_total, int *io_err)
127{
128 off_t size;
129 size_t total;
130
131 if (!ctx->files[idx].writable) {
132 size = ctx->files[idx].cached_size;
133 } else {
134 struct stat st;
135 if (fstat(ctx->files[idx].fd, &st) != 0) {
136 *io_err = errno;
137 return NGX_ERROR;
138 }
139 size = st.st_size;
140 }
141
142 if (offset >= size) {
143 total = 0;
144 } else {
145 off_t avail = size - offset;
146 total = (avail < (off_t) rlen) ? (size_t) avail : rlen;
147 }
148 if (total > BRIX_READ_CHUNK_MAX) {
149 total = BRIX_READ_CHUNK_MAX;
150 }
151
152 *file_size = size;
153 *data_total = total;
154 return NGX_OK;
155}

157/* ---- Read the plaintext window synchronously through the VFS core ----
158 *
159 * WHAT: Reads up to data_total bytes at offset from the handle into plain via a
160 * VFS read job, storing the byte count in *nread_out. Returns NGX_OK, or
161 * NGX_ERROR with *io_err set (io_errno, or EIO for a bare short/negative
162 * return) on failure.
163 *
164 * WHY: All file-data byte I/O must flow through the VFS core seam; extracting
165 * the job dance keeps the orchestrator free of raw job-struct plumbing and
166 * preserves the exact errno→message mapping of the inline version.
167 *
168 * HOW: 1. Init a read job for the handle fd. 2. Execute it synchronously.
169 * 3. On io_errno or a negative return, surface an errno (EIO fallback).
170 * 4. Otherwise publish the non-negative byte count.
171 */
172static ngx_int_t
173brix_compressed_read_window(brix_ctx_t *ctx, int idx, off_t offset,
174 u_char *plain, size_t data_total,
175 size_t *nread_out, int *io_err)
176{
177 brix_vfs_job_t job;
178 ssize_t nread;
179
180 brix_vfs_job_read_init(&job, ctx->files[idx].fd, offset, data_total, plain,
181 data_total, 0);
182 brix_vfs_io_execute(&job);
183 nread = job.nio;
184 if (job.io_errno != 0 || nread < 0) {
185 *io_err = job.io_errno != 0 ? job.io_errno : EIO;
186 return NGX_ERROR;
187 }
188
189 *nread_out = (size_t) nread;
190 return NGX_OK;
191}

193/* ---- Record metrics, dashboard, and access-log side effects for a served frame ----
194 *
195 * WHAT: Advances the per-handle and per-session byte counters, charges the
196 * rate-limiter, updates the dashboard slot, writes the access-log line,
197 * and marks the READ op OK. No return value.
198 *
199 * WHY: Accounting mirrors the plaintext read path — counters reflect PLAINTEXT
200 * bytes served (what the client receives after inflate), while the
201 * rate-limiter is charged the actual compressed wire bytes. Grouping the
202 * side effects at one edge keeps the orchestrator a pure control flow.
203 *
204 * HOW: 1. Add plaintext bytes to handle + session totals, charge wire bytes to
205 * the limiter. 2. If a dashboard slot is active, publish bytes and a
206 * read op. 3. When an access-log fd is configured, emit the framed
207 * detail line. 4. Increment the READ ok metric.

240/* ---- Serve a kXR_read for a compress-negotiated handle as one codec frame ----
241 *
242 * WHAT: Reads a bounded plaintext window, compresses it into one self-contained
243 * codec frame over rd.cmp_scratch, and queues it as a single kXR_read
244 * response. Returns the queue rc, NGX_ERROR on scratch/chain failure, or
245 * a sent-error rc for stat/IO/compression failures.
246 *
247 * WHY: Keeps the compressed read path a short linear orchestration: extent,
248 * stage, read, compress, account, frame — each step a named helper — so
249 * the byte-exact framing and codec selection are easy to audit.
250 *
251 * HOW: 1. Resolve the readable window (short-circuit an empty frame at EOF).
252 * 2. Stage plaintext into rd.read_scratch and prefetch. 3. Read the
253 * window through the VFS core (empty read → empty frame). 4. Compress
254 * into rd.cmp_scratch. 5. Record side effects. 6. Frame and queue the
255 * compressed bytes, releasing the keep-slot if it was not handed off.
256 */
257ngx_int_t
258brix_read_compressed(brix_ctx_t *ctx, ngx_connection_t *c,
259 ngx_stream_brix_srv_conf_t *rconf,
260 int idx, off_t offset, size_t rlen)
261{
262 uint8_t codec_id = ctx->files[idx].read_codec;
263 off_t file_size;
264 size_t data_total, clen, nread;
265 u_char *plain;
266 ngx_chain_t *rsp_chain;
267 ngx_int_t rc;
268 int io_err = 0;
269
270 if (brix_compressed_extent(ctx, idx, offset, rlen, &file_size,
271 &data_total, &io_err) != NGX_OK) {
272 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
273 ctx->files[idx].path, "-",
274 kXR_IOError, strerror(io_err));
275 }
276
277 if (data_total == 0) {
278 /* EOF / empty range: an empty frame inflates to zero bytes. */
279 BRIX_OP_OK(ctx, BRIX_OP_READ);
280 return brix_send_ok(ctx, c, NULL, 0);
281 }
282
283 /* Stage the plaintext window. */
284 plain = BRIX_GET_SCRATCH(ctx, c, rd.read_scratch, rd.read_scratch_size,
285 data_total);
286 if (plain == NULL) {
287 return NGX_ERROR;
288 }
289
290 brix_prefetch_read_file(c->log, &ctx->files[idx], offset, data_total,
291 file_size);
292
293 if (brix_compressed_read_window(ctx, idx, offset, plain, data_total,
294 &nread, &io_err) != NGX_OK) {
295 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
296 ctx->files[idx].path, "-",
297 kXR_IOError, strerror(io_err));
298 }
299 data_total = nread;
300
301 if (data_total == 0) {
302 BRIX_OP_OK(ctx, BRIX_OP_READ);
303 return brix_send_ok(ctx, c, NULL, 0);
304 }
305
306 /* Compress the window into rd.cmp_scratch as one self-contained frame. */
307 clen = brix_compress_frame(ctx, c, codec_id, plain, data_total);
308 if (clen == (size_t) -1) {
309 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
310 ctx->files[idx].path, "-",
311 kXR_IOError, "inline read compression failed");
312 }
313
314 brix_compressed_account(ctx, c, rconf, idx, offset, data_total, clen);
315
316 /* Frame the compressed bytes as a single kXR_read response over rd.cmp_scratch
317 * (a keep-slot, so brix_release_read_buffer leaves it for reuse). */
318 rsp_chain = brix_build_chunked_chain(ctx, c, ctx->rd.cmp_scratch, clen);
319 if (rsp_chain == NULL) {
320 return NGX_ERROR;
321 }
322
323 rc = brix_queue_response_chain(ctx, c, rsp_chain, ctx->rd.cmp_scratch);
324 if (rc != NGX_OK || ctx->state != XRD_ST_SENDING) {
325 brix_release_read_buffer(ctx, c, ctx->rd.cmp_scratch);
326 }
327 return rc;
328}
