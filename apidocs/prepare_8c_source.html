13/*
14 * WHAT: kXR_prepare — local-storage staging hint. (kXR_QPrep status query lives
15 * in prepare_qprep.c.)
16 * prepare accepts newline-separated path lists, validates each against auth/ACLs/filesystem existence,
17 * optionally invokes a configured staging command (e.g., xrdcp to tape), returns request ID for later status queries.
18 * QPrep queries staging status of prior prepare paths — returns "A <path>" (available) or "M <path>" (missing).
19 *
20 * WHY: Staging workflows require clients to submit file lists before actual transfer, enabling servers to initiate
21 * tape nearline retrieval or other pre-transfer operations. prepare stores request ID + path list for QPrep status
22 * queries. Disk-only servers return immediate results since files are either present or absent. Cancel/evict options
23 * allow clients to abort pending staging operations without penalty.
24 *
25 * HOW: brix_handle_prepare() parses ClientPrepareRequest — extracts optionX via ntohs, checks kXR_wmode+allow_write,
26 * cancel/evict as noops, payload presence. Pre-allocates stage_paths/stage_bufs if collect_stage (kXR_stage + prepare_command). Parses
27 * newline-separated paths: extract_path → has_forbidden_component() → resolve_path → authdb(vo_acl token_scope) → stat(S_ISDIR check).
28 * Fills out_resolved for staging collection. Stores saved payload in ctx->prepare.paths, sets reqid="0", invokes staging command best-effort,
29 * returns "0" as response. brix_query_prep_status() parses payload skipping reqid line — uses inline paths or falls back to stored
30 * prepare_paths. Allocates resp buffer src_len*2+64, resolves each path + auth chain + stat(S_ISREG) → writes 'A'/'M' prefix per path,
31 * NUL-terminates and sends response.
32 */

34/*
35 * WHAT: Return the stable owner string used for FRM stream prepare records and
36 * cancel checks. Authenticated callers use their canonical identity DN/token
37 * subject; anonymous callers get a per-login owner key derived from sessid.
38 * WHY: FRM cancel authorization compares stored requester strings. Leaving
39 * anonymous records owner-less lets any later anonymous session cancel another
40 * session's stage request by guessing its reqid.
41 * HOW: Borrow the identity string when present. Otherwise render
42 * "anon-session:" plus the 16-byte session id as lowercase hex into caller
43 * storage and return that buffer. */
44static const char *
45brix_prepare_owner_key(brix_ctx_t *ctx, char *anon_key, size_t anon_key_sz)
46{
47 static const char hex[] = "0123456789abcdef";
48 static const char prefix[] = "anon-session:";
49 const char *dn;
50 size_t pos;
51 ngx_uint_t i;
52
53 if (ctx == NULL) {
54 return NULL;
55 }
56
57 dn = brix_identity_dn_cstr(ctx->identity);
58 if (dn != NULL && dn[0] != '\0') {
59 return dn;
60 }
61
62 if (anon_key == NULL || anon_key_sz < sizeof(prefix) + 32) {
63 return NULL;
64 }
65
66 ngx_memcpy(anon_key, prefix, sizeof(prefix) - 1);
67 pos = sizeof(prefix) - 1;
68 for (i = 0; i < 16; i++) {
69 u_char b = ctx->login.sessid[i];
70 anon_key[pos++] = hex[b >> 4];
71 anon_key[pos++] = hex[b & 0x0f];
72 }
73 anon_key[pos] = '\0';
74 return anon_key;
75}

89/* Phase 35: kXR_prepare + kXR_cancel — delete the named request from the durable
90 * queue. The reqid is the first payload line. Idempotent: an unknown reqid still
91 * returns kXR_ok (no enumeration oracle). */
92static ngx_int_t
93brix_prepare_handle_cancel(brix_ctx_t *ctx, ngx_connection_t *c,
94 ngx_stream_brix_srv_conf_t *conf)
95{
96 char reqid[BRIX_STAGE_REQID_LEN];
97 char owner_key[BRIX_PREPARE_OWNER_KEY_MAX];
98 brix_stage_registry_t *reg = brix_stage_registry_singleton();
99 const u_char *p, *end;
100 size_t n;
101
102 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
103 return brix_prepare_send_fail(ctx, c, "-", kXR_ArgMissing,
104 "prepare requestid not specified");
105 }
106 p = ctx->recv.payload;
107 end = ctx->recv.payload + ctx->recv.cur_dlen;
108 while (p < end && *p != '\n' && *p != '\r' && *p != '\0') {
109 p++;
110 }
111 n = (size_t) (p - ctx->recv.payload);
112 if (n == 0 || n >= sizeof(reqid)) {
113 return brix_prepare_send_fail(ctx, c, "-", kXR_ArgMissing,
114 "prepare requestid not specified");
115 }
116 ngx_memcpy(reqid, ctx->recv.payload, n);
117 reqid[n] = '\0';
118
119 /* FRM-1: a request may only be cancelled by the owner that created it.
120 * Anonymous stream callers are scoped to their login session id, so another
121 * anonymous session cannot cancel by guessing a durable reqid. */
122 if (brix_stage_request_owner_check(reg, reqid,
123 brix_prepare_owner_key(ctx, owner_key,
124 sizeof(owner_key)),
125 c->log) != NGX_OK)
126 {
127 brix_log_access(ctx, c, "PREPARE", reqid, "cancel-denied", 0,
128 kXR_NotAuthorized, NULL, 0);
129 return brix_prepare_send_fail(ctx, c, reqid, kXR_NotAuthorized,
130 "not the owner of this request");
131 }
132
133 (void) brix_stage_request_delete(reg, reqid, c->log); /* idempotent */
134 brix_log_access(ctx, c, "PREPARE", reqid, "cancel", 1, kXR_ok, NULL, 0);
135 return brix_send_ok(ctx, c, NULL, 0);
136}

138/*
139 * Shared state for the kXR_prepare path-scan pipeline (alloc → scan → emit).
140 * Passed by pointer so the three phases read/accumulate the same counters and
141 * staging buffers without a long positional argument list. group_reqid points
142 * at a caller-owned BRIX_STAGE_REQID_LEN buffer.
143 */
144typedef struct {
145 ngx_stream_brix_srv_conf_t *conf;
146 uint16_t options; /* req.options snapshot */
147 ngx_flag_t need_resolved; /* fill out_resolved (legacy cmd OR enqueue) */
148 ngx_flag_t do_enqueue; /* enqueue into the durable FRM queue */
149 ngx_flag_t collect_stage; /* collect paths for brix_prepare_command */
150 const char **stage_paths; /* pool array, stage_max entries */

152 ngx_uint_t stage_max;

156 char *group_reqid; /* first durable reqid = client handle */

159/*

162 * bounded to avoid excessive allocation). No-op when need_resolved is clear.

164static ngx_int_t
165prepare_alloc_stage_arrays(ngx_connection_t *c, prepare_scan_t *sc, size_t dlen)

168 return NGX_OK;

172 if (sc->stage_max > BRIX_PREPARE_CMD_MAX_PATHS) {

174 }
175
176 sc->stage_paths = ngx_palloc(c->pool, sizeof(const char *) * sc->stage_max);

184/*

186 * it for the legacy staging command. The first durable reqid becomes the client
187 * handle. The caller advances stage_count after this returns.

189static void
190prepare_enqueue_resolved(brix_ctx_t *ctx, ngx_connection_t *c,

192{

195 char rq[BRIX_STAGE_REQID_LEN];

198 * anonymous-session key, so the cancel-owner check has the same
199 * owner string. */

219 BRIX_STAGE_REQID_LEN);

221 /* NOTE (engine-integration step): driving the recall — the former
222 * frm_stage_kick() — moves to brix_stage_submit(RECALL) + the engine
223 * scheduler; the composable sd_frm backend also faults the recall on

225 }

231/*
232 * Parse the newline-separated payload, validate + authorize each path via
233 * brix_prepare_check_path, and enqueue/collect the ones that pass. Returns
234 * NGX_OK when the whole list scanned; NGX_DONE or an error rc when a path failed
235 * (check_path already sent the response) — the caller maps NGX_DONE to NGX_OK.
236 */
237static ngx_int_t
238prepare_scan_paths(brix_ctx_t *ctx, ngx_connection_t *c, prepare_scan_t *sc)
239{
240 const u_char *p = ctx->recv.payload;
241 const u_char *end = ctx->recv.payload + ctx->recv.cur_dlen;
242
243 while (p < end) {
244 const u_char *line;
245 size_t line_len;
246 ngx_int_t rc;
247 char *out_resolved;
248
249 line = p;
250 while (p < end && *p != '\n') {
251 p++;
252 }
253
254 line_len = (size_t) (p - line);
255 if (p < end && *p == '\n') {
256 p++;
257 }
258
259 while (line_len > 0
260 && (line[line_len - 1] == '\r'
261 || line[line_len - 1] == '\0'))
262 {
263 line_len--;
264 }
265
266 if (line_len == 0) {
267 continue;
268 }
269
270 sc->paths++;
271
272 /* Point out_resolved at the next slot in the staging buffer (when the
273 * legacy command OR the durable queue needs the resolved path) so
274 * brix_prepare_check_path fills it in-place. */
275 if (sc->need_resolved && sc->stage_count < sc->stage_max) {
276 out_resolved = sc->stage_bufs + sc->stage_count * PATH_MAX;
277 out_resolved[0] = '\0';
278 } else {
279 out_resolved = NULL;
280 }
281
282 rc = brix_prepare_check_path(ctx, c, sc->conf, line, line_len,
283 (sc->options & kXR_noerrs) != 0,
284 &sc->missing, out_resolved);
285 if (rc != NGX_OK) {
286 return rc; /* NGX_DONE (response sent) or a hard error */
287 }
288
289 if (out_resolved != NULL && out_resolved[0] != '\0') {
290 prepare_enqueue_resolved(ctx, c, sc, out_resolved);
291 sc->stage_count++;
292 }
293 }
294
295 return NGX_OK;
296}

352 * with the kXR_ok response in a single buffer to avoid a double-queue race

354 * by the background command; we cannot know when they arrive, so the
355 * notification is omitted and a warning is logged. */

357 if (sc->missing == 0) {

361 size_t attn_len;

363 u_char *buf;
364

366 "prepare reqid=%s complete", resp_reqid);
367 if (notify_len >= sizeof(notify_msg)) {
368 notify_len = sizeof(notify_msg) - 1;
369 }
370 attn_len = brix_attn_asyncms_frame_len(notify_len);
371 total = ok_len + attn_len;
372
373 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
374
375 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
376 (uint32_t) resp_reqid_len,
377 (ServerResponseHdr *) buf);
378 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, resp_reqid, resp_reqid_len);
379 brix_build_attn_asyncms_frame(buf + ok_len, notify_msg, notify_len);
380
381 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, c->log, 0,
382 "brix: sending kXR_prepare ok + kXR_attn asyncms notify");
383
384 return brix_queue_response(ctx, c, buf, total);
385
386 } else {
387 ngx_log_error(NGX_LOG_WARN, c->log, 0,
388 "brix: kXR_notify requested but %ui path(s) still being "
389 "staged; async completion notification not supported",

391 }
392 }

394 return brix_send_ok(ctx, c, (u_char *) resp_reqid,

442/* WHY: kXR_prepare validates each path in a prepare request against auth, ACLs, and filesystem existence before accepting it for staging. 513/* HOW: Parses ClientPrepareRequest from ctx->recv.hdr_buf — extracts optionX via ntohs(req->optionX). Checks kXR_stage + prepare_command.len > 0 → collect_stage=1.
