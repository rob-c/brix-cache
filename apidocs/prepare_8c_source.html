13/*
14 * WHAT: kXR_prepare — local-storage staging hint. (kXR_QPrep status query lives
15 * in prepare_qprep.c.)
16 * prepare accepts newline-separated path lists, validates each against auth/ACLs/filesystem existence,
17 * optionally invokes a configured staging command (e.g., xrdcp to tape), returns request ID for later status queries.
18 * QPrep queries staging status of prior prepare paths — returns "A <path>" (available) or "M <path>" (missing).
19 *
20 * WHY: Staging workflows require clients to submit file lists before actual transfer, enabling servers to initiate
21 * tape nearline retrieval or other pre-transfer operations. prepare stores request ID + path list for QPrep status
22 * queries. Disk-only servers return immediate results since files are either present or absent. Cancel/evict options
23 * allow clients to abort pending staging operations without penalty.
24 *
25 * HOW: brix_handle_prepare() parses ClientPrepareRequest — extracts optionX via ntohs, checks kXR_wmode+allow_write,
26 * cancel/evict as noops, payload presence. Pre-allocates stage_paths/stage_bufs if collect_stage (kXR_stage + prepare_command). Parses
27 * newline-separated paths: extract_path → has_forbidden_component() → resolve_path → authdb(vo_acl token_scope) → stat(S_ISDIR check).
28 * Fills out_resolved for staging collection. Stores saved payload in ctx->prepare.paths, sets reqid="0", invokes staging command best-effort,
29 * returns "0" as response. brix_query_prep_status() parses payload skipping reqid line — uses inline paths or falls back to stored
30 * prepare_paths. Allocates resp buffer src_len*2+64, resolves each path + auth chain + stat(S_ISREG) → writes 'A'/'M' prefix per path,
31 * NUL-terminates and sends response.
32 */

34static ngx_flag_t
35brix_prepare_has_forbidden_component(const char *path)
36{
37 const char *p = path;
38
39 while (*p != '\0') {
40 const char *seg;
41 size_t len;
42
43 while (*p == '/') {
44 p++;
45 }
46
47 seg = p;
48 while (*p != '\0' && *p != '/') {
49 p++;
50 }
51
52 len = (size_t) (p - seg);
53 if ((len == 1 && seg[0] == '.')
54 || (len == 2 && seg[0] == '.' && seg[1] == '.'))
55 {
56 return 1;
57 }
58 }
59
60 return 0;
61}

65/*
66 * WHAT: Return the stable owner string used for FRM stream prepare records and
67 * cancel checks. Authenticated callers use their canonical identity DN/token
68 * subject; anonymous callers get a per-login owner key derived from sessid.
69 * WHY: FRM cancel authorization compares stored requester strings. Leaving
70 * anonymous records owner-less lets any later anonymous session cancel another
71 * session's stage request by guessing its reqid.
72 * HOW: Borrow the identity string when present. Otherwise render
73 * "anon-session:" plus the 16-byte session id as lowercase hex into caller
74 * storage and return that buffer. */
75static const char *
76brix_prepare_owner_key(brix_ctx_t *ctx, char *anon_key, size_t anon_key_sz)
77{
78 static const char hex[] = "0123456789abcdef";
79 static const char prefix[] = "anon-session:";
80 const char *dn;
81 size_t pos;
82 ngx_uint_t i;
83
84 if (ctx == NULL) {
85 return NULL;
86 }
87
88 dn = brix_identity_dn_cstr(ctx->identity);
89 if (dn != NULL && dn[0] != '\0') {
90 return dn;
91 }
92
93 if (anon_key == NULL || anon_key_sz < sizeof(prefix) + 32) {
94 return NULL;
95 }
96
97 ngx_memcpy(anon_key, prefix, sizeof(prefix) - 1);
98 pos = sizeof(prefix) - 1;
99 for (i = 0; i < 16; i++) {
100 u_char b = ctx->login.sessid[i];
101 anon_key[pos++] = hex[b >> 4];
102 anon_key[pos++] = hex[b & 0x0f];
103 }
104 anon_key[pos] = '\0';
105 return anon_key;
106}

132/*
133 * WHAT: run the three prepare authorization tiers (authdb VO/ACL, VO identity
134 * ACL, token scope) on one resolved path.
135 * WHY: the existing-file and noerrs-absent branches of check_path must apply
136 * the SAME gate on the SAME paths (verdict parity between "exists" and
137 * "absent") — factoring it here removes the duplication AND guarantees the
138 * two branches can never drift apart. Authorization is a property of the
139 * identity + logical path, not of on-disk existence.
140 * HOW: each tier that denies sends its specific error via check_fail and its rc
141 * (NGX_DONE/error) is returned; NGX_OK only when all three pass.

185static ngx_int_t
186brix_prepare_check_path(brix_ctx_t *ctx, ngx_connection_t *c,
187 ngx_stream_brix_srv_conf_t *conf, const u_char *line, size_t line_len,
188 ngx_flag_t noerrs, ngx_uint_t *missing,
189 char *out_resolved) /* PATH_MAX buffer filled with absolute path on
190 auth-pass paths; '\0' if path cannot be resolved.
191 Pass NULL when staging collection is not needed. */
192{
193 char pathbuf[BRIX_MAX_PATH + 1];
194 char full_path[PATH_MAX];
195 struct stat st;
196
197 if (line_len > BRIX_MAX_PATH) {
198 return brix_prepare_check_fail(ctx, c, "-", kXR_ArgTooLong,
199 "prepare path too long");
200 }
201
202 if (!brix_extract_path(c->log, line, line_len, pathbuf,
203 sizeof(pathbuf), 1)) {
204 return brix_prepare_check_fail(ctx, c, "-", kXR_ArgInvalid,
205 "invalid prepare path");
206 }
207
208 if (brix_prepare_has_forbidden_component(pathbuf)) {
209 return brix_prepare_check_fail(ctx, c, pathbuf, kXR_ArgInvalid,
210 "invalid prepare path");
211 }
212
213 /* phase74-fp: pathbuf is the request path, full_path the output buf. */
214 brix_beneath_full_path(conf->common.root_canon, pathbuf, /* NOLINT(readability-suspicious-call-argument) */
215 full_path, sizeof(full_path));
216
217 /*
218 * CONTRACT: the same "file is absent" condition (ENOENT/ENOTDIR from the
219 * confined stat) has two outcomes selected by the kXR_noerrs flag:
220 * - noerrs set → not an error. The path is counted in *missing and the
221 * request still succeeds, so a client can prepare/stage files that do
222 * not exist on disk yet (tape nearline recall, not-yet-cached objects).
223 * - noerrs clear → kXR_NotFound, failing the request on the first miss.
224 * EACCES/EPERM and any other errno always fail regardless of noerrs.
225 */
226 if (brix_stat_beneath(conf->rootfd, pathbuf, &st) != 0) {
227 if ((errno == ENOENT || errno == ENOTDIR) && noerrs) {
228 (*missing)++;
229 /* SECURITY: authorization is a property of the IDENTITY + LOGICAL
230 * PATH, not of on-disk existence. A prepare/stage of a not-yet-
231 * materialised object (tape nearline recall, not-yet-cached) must
232 * still prove the caller may READ/STAGE this namespace path —
233 * otherwise an unauthorized principal drives recalls or enumerates
234 * the namespace via prepare, and later serves the recalled bytes from
235 * the shared cache. Run the SAME three tiers, on the SAME paths, as
236 * the existing-file branch below (verdict parity between "exists" and
237 * "absent"); only then supply the staging path. */
238 {
239 ngx_int_t arc = prepare_path_authz(ctx, c, conf, pathbuf,
240 full_path);
241 if (arc != NGX_OK) {
242 return arc;
243 }
244 }
245 /* For staging: supply absolute path even if file doesn't exist yet
246 * (tape nearline / not-yet-created). */
247 if (out_resolved != NULL) {
248 ngx_cpystrn((u_char *) out_resolved, (u_char *) full_path,
249 PATH_MAX);
250 }
251 return NGX_OK;
252 }
253 return prepare_stat_error(ctx, c, pathbuf, full_path);
254 }
255
256 {
257 ngx_int_t arc = prepare_path_authz(ctx, c, conf, pathbuf, full_path);
258 if (arc != NGX_OK) {
259 return arc;
260 }
261 }
262
263 /* Copy the absolute export path for the staging command; only authorized
264 * paths reach here so the staging hook can trust the value. */
265 if (out_resolved != NULL) {
266 ngx_cpystrn((u_char *) out_resolved, (u_char *) full_path, PATH_MAX);
267 }
268
269 if (S_ISDIR(st.st_mode)) {
270 if (noerrs) {
271 (*missing)++;
272 return NGX_OK;
273 }
274 return brix_prepare_check_fail(ctx, c, pathbuf, kXR_isDirectory,
275 "prepare target is a directory");
276 }
277
278 return NGX_OK;
279}

281/* Phase 35: kXR_prepare + kXR_cancel — delete the named request from the durable
282 * queue. The reqid is the first payload line. Idempotent: an unknown reqid still
283 * returns kXR_ok (no enumeration oracle). */
284static ngx_int_t
285brix_prepare_handle_cancel(brix_ctx_t *ctx, ngx_connection_t *c,
286 ngx_stream_brix_srv_conf_t *conf)
287{
288 char reqid[BRIX_STAGE_REQID_LEN];
289 char owner_key[BRIX_PREPARE_OWNER_KEY_MAX];
290 brix_stage_registry_t *reg = brix_stage_registry_singleton();
291 const u_char *p, *end;
292 size_t n;
293
294 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
295 return brix_prepare_send_fail(ctx, c, "-", kXR_ArgMissing,
296 "prepare requestid not specified");
297 }
298 p = ctx->recv.payload;
299 end = ctx->recv.payload + ctx->recv.cur_dlen;
300 while (p < end && *p != '\n' && *p != '\r' && *p != '\0') {
301 p++;
302 }
303 n = (size_t) (p - ctx->recv.payload);
304 if (n == 0 || n >= sizeof(reqid)) {
305 return brix_prepare_send_fail(ctx, c, "-", kXR_ArgMissing,
306 "prepare requestid not specified");
307 }
308 ngx_memcpy(reqid, ctx->recv.payload, n);
309 reqid[n] = '\0';
310
311 /* FRM-1: a request may only be cancelled by the owner that created it.
312 * Anonymous stream callers are scoped to their login session id, so another
313 * anonymous session cannot cancel by guessing a durable reqid. */
314 if (brix_stage_request_owner_check(reg, reqid,
315 brix_prepare_owner_key(ctx, owner_key,
316 sizeof(owner_key)),
317 c->log) != NGX_OK)
318 {
319 brix_log_access(ctx, c, "PREPARE", reqid, "cancel-denied", 0,
320 kXR_NotAuthorized, NULL, 0);
321 return brix_prepare_send_fail(ctx, c, reqid, kXR_NotAuthorized,
322 "not the owner of this request");
323 }
324
325 (void) brix_stage_request_delete(reg, reqid, c->log); /* idempotent */
326 brix_log_access(ctx, c, "PREPARE", reqid, "cancel", 1, kXR_ok, NULL, 0);
327 return brix_send_ok(ctx, c, NULL, 0);
328}

330/*
331 * Shared state for the kXR_prepare path-scan pipeline (alloc → scan → emit).
332 * Passed by pointer so the three phases read/accumulate the same counters and
333 * staging buffers without a long positional argument list. group_reqid points
334 * at a caller-owned BRIX_STAGE_REQID_LEN buffer.
335 */
336typedef struct {
337 ngx_stream_brix_srv_conf_t *conf;
338 uint16_t options; /* req.options snapshot */
339 ngx_flag_t need_resolved; /* fill out_resolved (legacy cmd OR enqueue) */
340 ngx_flag_t do_enqueue; /* enqueue into the durable FRM queue */
341 ngx_flag_t collect_stage; /* collect paths for brix_prepare_command */
342 const char **stage_paths; /* pool array, stage_max entries */
343 char *stage_bufs; /* pool array, stage_max * PATH_MAX */
344 ngx_uint_t stage_max;
345 ngx_uint_t stage_count; /* accumulated resolved paths */
346 ngx_uint_t paths; /* non-empty lines seen */
347 ngx_uint_t missing; /* absent-but-authorized paths (noerrs) */
348 char *group_reqid; /* first durable reqid = client handle */
349} prepare_scan_t;

351/*
352 * Pre-allocate the staging collection arrays when a resolved path list is
353 * needed. Cap at BRIX_PREPARE_CMD_MAX_PATHS (typically dlen/2 paths, but
354 * bounded to avoid excessive allocation). No-op when need_resolved is clear.
355 */
356static ngx_int_t
357prepare_alloc_stage_arrays(ngx_connection_t *c, prepare_scan_t *sc, size_t dlen)
358{
359 if (!sc->need_resolved) {
360 return NGX_OK;
361 }
362
363 sc->stage_max = dlen / 2 + 1;
364 if (sc->stage_max > BRIX_PREPARE_CMD_MAX_PATHS) {
365 sc->stage_max = BRIX_PREPARE_CMD_MAX_PATHS;
366 }
367
368 sc->stage_paths = ngx_palloc(c->pool, sizeof(const char *) * sc->stage_max);
369 sc->stage_bufs = ngx_palloc(c->pool, (size_t) sc->stage_max * PATH_MAX);
370 if (sc->stage_paths == NULL || sc->stage_bufs == NULL) {
371 return NGX_ERROR;
372 }
373 return NGX_OK;
374}

376/*
377 * Accept one authorized, resolved path: durably enqueue it (FRM) and/or collect
378 * it for the legacy staging command. The first durable reqid becomes the client
379 * handle. The caller advances stage_count after this returns.
380 */
381static void
382prepare_enqueue_resolved(brix_ctx_t *ctx, ngx_connection_t *c,
383 prepare_scan_t *sc, char *out_resolved)
384{
385 if (sc->do_enqueue) {
386 brix_stage_request_view_t v;
387 char rq[BRIX_STAGE_REQID_LEN];
388 ngx_int_t arc;
389 /* Record the canonical identity DN/token subject, or a scoped
390 * anonymous-session key, so the cancel-owner check has the same
391 * owner string. */
392 char owner_key[BRIX_PREPARE_OWNER_KEY_MAX];
393 const char *rdn = brix_prepare_owner_key(ctx, owner_key,
394 sizeof(owner_key));
395
396 ngx_memzero(&v, sizeof(v));
397 v.lfn = out_resolved;
398 v.requester_dn = (rdn != NULL && rdn[0] != '\0') ? rdn : NULL;
399 v.tod_expire = (int64_t) time(NULL)
400 + (int64_t) (sc->conf->frm.stage_ttl / 1000);
401
402 arc = brix_stage_request_add(brix_stage_registry_singleton(),
403 &v, rq, sizeof(rq), c->log);
404 if (arc == NGX_ERROR) {
405 ngx_log_error(NGX_LOG_ERR, c->log, 0,
406 "brix: stage request add failed for \"%s\"",
407 out_resolved);
408 } else if (sc->group_reqid[0] == '\0') {
409 /* The first request id is the handle returned to the client. */
410 ngx_cpystrn((u_char *) sc->group_reqid, (u_char *) rq,
411 BRIX_STAGE_REQID_LEN);
412 }
413 /* NOTE (engine-integration step): driving the recall — the former
414 * frm_stage_kick() — moves to brix_stage_submit(RECALL) + the engine
415 * scheduler; the composable sd_frm backend also faults the recall on
416 * read, so the request is durably recorded here. */
417 }
418 if (sc->collect_stage) {
419 sc->stage_paths[sc->stage_count] = out_resolved;
420 }
421}

423/*
424 * Parse the newline-separated payload, validate + authorize each path via
425 * brix_prepare_check_path, and enqueue/collect the ones that pass. Returns
426 * NGX_OK when the whole list scanned; NGX_DONE or an error rc when a path failed
427 * (check_path already sent the response) — the caller maps NGX_DONE to NGX_OK.
428 */
429static ngx_int_t
430prepare_scan_paths(brix_ctx_t *ctx, ngx_connection_t *c, prepare_scan_t *sc)
431{
432 const u_char *p = ctx->recv.payload;
433 const u_char *end = ctx->recv.payload + ctx->recv.cur_dlen;
434
435 while (p < end) {
436 const u_char *line;
437 size_t line_len;
438 ngx_int_t rc;
439 char *out_resolved;
440
441 line = p;
442 while (p < end && *p != '\n') {
443 p++;
444 }
445
446 line_len = (size_t) (p - line);
447 if (p < end && *p == '\n') {
448 p++;
449 }
450
451 while (line_len > 0
452 && (line[line_len - 1] == '\r'
453 || line[line_len - 1] == '\0'))
454 {
455 line_len--;
456 }
457
458 if (line_len == 0) {
459 continue;
460 }
461
462 sc->paths++;
463
464 /* Point out_resolved at the next slot in the staging buffer (when the
465 * legacy command OR the durable queue needs the resolved path) so
466 * brix_prepare_check_path fills it in-place. */
467 if (sc->need_resolved && sc->stage_count < sc->stage_max) {
468 out_resolved = sc->stage_bufs + sc->stage_count * PATH_MAX;
469 out_resolved[0] = '\0';
470 } else {
471 out_resolved = NULL;
472 }
473
474 rc = brix_prepare_check_path(ctx, c, sc->conf, line, line_len,
475 (sc->options & kXR_noerrs) != 0,
476 &sc->missing, out_resolved);
477 if (rc != NGX_OK) {
478 return rc; /* NGX_DONE (response sent) or a hard error */
479 }
480
481 if (out_resolved != NULL && out_resolved[0] != '\0') {
482 prepare_enqueue_resolved(ctx, c, sc, out_resolved);
483 sc->stage_count++;
484 }
485 }
486
487 return NGX_OK;
488}

490/*
491 * kXR_stage tail: persist the path list for later kXR_QPrep queries, launch the
492 * legacy staging command if configured, and (kXR_notify) fold an asyncms
493 * "complete" notification into the ok response when nothing is still missing.
494 * Returns the queued-response rc.

496static ngx_int_t
497prepare_emit_stage(brix_ctx_t *ctx, ngx_connection_t *c, prepare_scan_t *sc)
498{
499 u_char *saved;
500 const char *resp_reqid;
501 size_t resp_reqid_len;
502
503 /* The request handle: the durable reqid when enqueued, else legacy "0". */
504 resp_reqid = (sc->do_enqueue && sc->group_reqid[0] != '\0')
505 ? sc->group_reqid : "0";
506 resp_reqid_len = ngx_strlen(resp_reqid);
507
508 saved = ngx_alloc((size_t) ctx->recv.cur_dlen + 1, c->log);
509 if (saved == NULL) {
510 return NGX_ERROR;
511 }
512
513 ngx_memcpy(saved, ctx->recv.payload, ctx->recv.cur_dlen);
514 saved[ctx->recv.cur_dlen] = '\0';
515
516 if (ctx->prepare.paths != NULL) {

518 }

521 sizeof(ctx->prepare.reqid));
522 ctx->prepare.paths = saved;
523 ctx->prepare.paths_len = ctx->recv.cur_dlen;
524
525 /* Invoke the staging command if configured and paths were collected. */
526 if (sc->collect_stage && sc->stage_count > 0) {
527 if (brix_prepare_invoke_command(c->log, sc->conf,

529 (sc->options & kXR_coloc) != 0)
530 != NGX_OK)

535 }
536 } else if (sc->collect_stage && sc->stage_count == 0) {
537 ngx_log_error(NGX_LOG_WARN, c->log, 0,
538 "brix: kXR_stage set but no resolvable paths"
539 " for prepare_command");

542 /* kXR_notify: send kXR_attn + kXR_asyncms notification when all requested
543 * files are already on disk (missing == 0). The notification is combined
544 * with the kXR_ok response in a single buffer to avoid a double-queue race
545 * on EAGAIN write paths. When missing > 0 some files are still being staged
546 * by the background command; we cannot know when they arrive, so the
547 * notification is omitted and a warning is logged. */
548 if (sc->options & kXR_notify) {
549 if (sc->missing == 0) {
550 char notify_msg[96];
551 size_t notify_len;
552 size_t ok_len = XRD_RESPONSE_HDR_LEN + resp_reqid_len;
553 size_t attn_len;
554 size_t total;
555 u_char *buf;
556

564

567 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
568 (uint32_t) resp_reqid_len,
569 (ServerResponseHdr *) buf);
570 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, resp_reqid, resp_reqid_len);
571 brix_build_attn_asyncms_frame(buf + ok_len, notify_msg, notify_len);
572
573 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, c->log, 0,
574 "brix: sending kXR_prepare ok + kXR_attn asyncms notify");
575
576 return brix_queue_response(ctx, c, buf, total);
577
578 } else {

580 "brix: kXR_notify requested but %ui path(s) still being "
581 "staged; async completion notification not supported",
582 sc->missing);
583 }
584 }
585
586 return brix_send_ok(ctx, c, (u_char *) resp_reqid,
587 (uint32_t) resp_reqid_len);

653 sc.options = req.options;
654 sc.collect_stage = (req.options & kXR_stage) && conf->prepare_command.len > 0;
655 sc.do_enqueue = (req.options & kXR_stage) && conf->frm.enable
656 && brix_stage_registry_singleton() != NULL;
657 sc.need_resolved = sc.collect_stage || sc.do_enqueue;
658 sc.group_reqid = group_reqid;

662 if (rc != NGX_DECLINED) {
663 return rc; /* read-only reject, cancel, or evict — response sent */
664 }
665
666 if (ctx->recv.cur_dlen == 0 || ctx->recv.payload == NULL) {
667 return brix_prepare_send_fail(ctx, c, "-", kXR_ArgMissing,
668 "prepare file list is missing");
669 }
670

672 return NGX_ERROR;
673 }

676 if (rc == NGX_DONE) {
677 return NGX_OK;
678 }
679 if (rc != NGX_OK) {
680 return rc;
681 }
682
683 if (sc.paths == 0) {
684 return brix_prepare_send_fail(ctx, c, "-", kXR_ArgMissing,
685 "prepare file list is empty");
686 }
687
688 snprintf(detail, sizeof(detail),
689 "paths=%u missing=%u opts=0x%02x optx=0x%04x%s",
690 (unsigned int) sc.paths, (unsigned int) sc.missing,
691 (unsigned int) req.options, (unsigned int) optionx,
692 (req.options & kXR_coloc) ? " (coloc)" : "");

695
696 /* kXR_stage: save the path list for kXR_QPrep queries, launch the staging
697 * command, and (kXR_notify) fold in an asyncms completion notification. */
698 if (req.options & kXR_stage) {
699 return prepare_emit_stage(ctx, c, &sc);
700 }
701
702 return brix_send_ok(ctx, c, NULL, 0);
703}
