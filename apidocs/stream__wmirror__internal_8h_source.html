1/*
2 * stream_wmirror_internal.h — declarations shared across the XRootD stream
3 * data-write mirror files after the phase-79 file-size split.
4 *
5 * WHAT: Cross-declares the handful of symbols and the two state structs that
6 * straddle the boundaries between the three data-write-mirror files, plus
7 * the shared BRIX_WMIR_METRIC_INC counter macro.
8 * WHY: stream_wmirror.c (824 lines) split into three focused files under the
9 * 500-line cap:
10 * stream_wmirror.c — per-connection accumulation + the public
11 * on_open / observe / cleanup hooks
12 * stream_wmirror_replay.c — detached-replay lifecycle: socket setup,
13 * event handlers, flush, teardown, launch
14 * stream_wmirror_state.c — replay protocol state machine: per-phase
15 * frame build/send + step decisions + dispatch
16 * The accumulation side launches a replay (wmir_launch); the replay's read
17 * handler hands each parsed frame to the state machine (wmir_dispatch); the
18 * state machine drains/queues frames (wmir_flush) and tears down on any
19 * error (wmir_finish). Exactly those cross-file symbols become non-static
20 * and are declared here; everything else stays file-local.
21 * HOW: All three .c files include this header; none of these symbols is part of
22 * the mirror's public surface (stream_wmirror.h).
23 *
24 * Requires: core/ngx_brix_module.h (ngx + wire types, XRD_RESPONSE_HDR_LEN) and
25 * observability/metrics/metrics_macros.h (brix_metrics_shared) — both
26 * pulled in below so this header stands alone.
27 */

36/* Data-write mirror counters live in the shared root metrics struct (low
37 * cardinality, no per-file labels per metrics INVARIANT 8). Shared by the

40#define BRIX_WMIR_METRIC_INC(field) \

47/*
48 * A single per-file accumulator slot. The accumulation side (stream_wmirror.c)
49 * owns and fills it keyed by the primary's file-handle index; wmir_launch
50 * (stream_wmirror_replay.c) reads it to seed a detached replay, stealing f->data.
51 */
52typedef struct {
53 unsigned active:1; /* a write-open is accumulating */
54 unsigned aborted:1; /* cap exceeded / non-seq / pgwrite */
55 u_char open_hdr[24]; /* client's open request header (24 B) */
56 u_char *open_payload; /* malloc: open path (+cgi) */
57 uint32_t open_dlen;
58 u_char *data; /* malloc/realloc: accumulated bytes */
59 size_t data_len;
60 off_t next_off; /* expected next contiguous write offset */
61} brix_wmirror_file_t;

64/*
65 * Replay state machine. The first three are the shared bootstrap (identical to
66 * the read mirror); the last three are this mirror's stateful create sequence.
67 * Each phase names the frame we have SENT and are now awaiting a response for.
68 * Linear progression only — any non-ok response aborts via wmir_finish(r, 0).
69 */
70typedef enum {
71 WMIR_HANDSHAKE = 0,

74 WMIR_OPEN, /* sent open, awaiting open response (shadow fhandle) */
75 WMIR_WRITE, /* sent write, awaiting response */
76 WMIR_CLOSE, /* sent close, awaiting response */
77} wmir_phase_t;
