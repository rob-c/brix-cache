40/*
41 * brix_webdav_require_vo <path> <vo> — append a VO ACL rule (mirrors the stream
42 * brix_require_vo). The rule path is normalized here; the realpath is finalized at
43 * startup (brix_finalize_vo_rules, from merge_loc_conf).
44 */
45char *
46webdav_conf_require_vo(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
47{
48 ngx_http_brix_webdav_loc_conf_t *wlcf = conf;
49 ngx_str_t *value;
50 brix_vo_rule_t *rule;
51
52 (void) cmd;
53 value = cf->args->elts; /* value[1] = path, value[2] = vo */
54
55 if (wlcf->vo_rules == NGX_CONF_UNSET_PTR || wlcf->vo_rules == NULL) {
56 wlcf->vo_rules = ngx_array_create(cf->pool, 2, sizeof(brix_vo_rule_t));
57 if (wlcf->vo_rules == NULL) {
58 return NGX_CONF_ERROR;
59 }
60 }
61
62 rule = ngx_array_push(wlcf->vo_rules);
63 if (rule == NULL) {
64 return NGX_CONF_ERROR;
65 }
66 ngx_memzero(rule, sizeof(*rule));
67
68 if (brix_normalize_policy_path(cf->pool, &value[1], &rule->path) != NGX_OK) {
69 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
70 "brix_webdav_require_vo: invalid path \"%V\"", &value[1]);
71 return NGX_CONF_ERROR;
72 }
73
74 if (brix_copy_conf_string(cf, &value[2], &rule->vo) != NGX_CONF_OK) {
75 return NGX_CONF_ERROR;
76 }
77
78 return NGX_CONF_OK;

111/*
112 * brix_webdav_dig_export <name> <dir> — register a named read-only diagnostic
113 * export (§3). The dir is realpath'd at config time into the export's `canon` (the
114 * RESOLVE_BENEATH anchor); a non-existent dir is a config error so misconfiguration
115 * is caught at startup, not at request time.
116 */
117char *
118webdav_conf_dig_export(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
119{
120 ngx_http_brix_webdav_loc_conf_t *wlcf = conf;
121 ngx_str_t *value;
122 brix_dig_export_t *ex;
123 char rp[PATH_MAX];
124
125 (void) cmd;
126
127 if (wlcf->dig_exports == NGX_CONF_UNSET_PTR || wlcf->dig_exports == NULL) {
128 wlcf->dig_exports = ngx_array_create(cf->pool, 2,
129 sizeof(brix_dig_export_t));
130 if (wlcf->dig_exports == NULL) {
131 return NGX_CONF_ERROR;
132 }
133 }
134
135 value = cf->args->elts; /* value[1]=name value[2]=dir */
136
137 if (realpath((const char *) value[2].data, rp) == NULL) {
138 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
139 "brix_webdav_dig_export: cannot resolve \"%V\"",
140 &value[2]);
141 return NGX_CONF_ERROR;
142 }
143
144 ex = ngx_array_push(wlcf->dig_exports);
145 if (ex == NULL) {
146 return NGX_CONF_ERROR;
147 }
148 ngx_memzero(ex, sizeof(*ex));
149 ex->name = value[1];
150 ex->dir = value[2];
151 ex->canon.len = ngx_strlen(rp);
152 ex->canon.data = ngx_pnalloc(cf->pool, ex->canon.len + 1);
153 if (ex->canon.data == NULL) {
154 return NGX_CONF_ERROR;
155 }
156 ngx_memcpy(ex->canon.data, rp, ex->canon.len + 1);
157
158 return NGX_CONF_OK;
159}

166char *
167webdav_conf_proxy_auth(ngx_conf_t *cf, ngx_command_t *cmd, void *conf_ptr)
168{
169 ngx_http_brix_webdav_loc_conf_t *conf = conf_ptr;
170 ngx_str_t *value = cf->args->elts;
171
172 (void) cmd;
173
174 if (ngx_strcmp(value[1].data, "anonymous") == 0) {
175 conf->upstream_auth = WEBDAV_PROXY_AUTH_ANONYMOUS;
176 return NGX_CONF_OK;
177 }
178 if (ngx_strcmp(value[1].data, "forward") == 0) {

180 return NGX_CONF_OK;
181 }
182 if (ngx_strcmp(value[1].data, "token") == 0) {
183 if (cf->args->nelts < 3) {
184 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
185 "brix_webdav_proxy_auth token requires a token value");
186 return NGX_CONF_ERROR;
187 }
188 conf->upstream_auth = WEBDAV_PROXY_AUTH_TOKEN;
189 conf->upstream_auth_token = value[2];
190 return NGX_CONF_OK;
191 }
192 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
193 "brix_webdav_proxy_auth: unknown policy \"%V\""
194 " (expected anonymous, forward, or token <value>)",
195 &value[1]);
196 return NGX_CONF_ERROR;
197}

241/*
242 * Parse one "brix_webdav_open_file_cache" parameter token. Recognised tokens:
243 * max=N -> *max (must be > 0)
244 * inactive=T -> *inactive (nginx time spec)
245 * off -> *off = 1
246 * Returns NGX_OK if the token was recognised and valid, NGX_ERROR for an unknown
247 * token or an out-of-range value; the caller logs the offending token on error.
248 */
249ngx_int_t
250webdav_open_file_cache_arg(ngx_str_t *arg, ngx_int_t *max, time_t *inactive,
251 ngx_flag_t *off)
252{
253 ngx_str_t s;
254
255 if (ngx_strncmp(arg->data, "max=", 4) == 0) {
256 *max = ngx_atoi(arg->data + 4, arg->len - 4);
257 return (*max <= 0) ? NGX_ERROR : NGX_OK;
258 }
259
260 if (ngx_strncmp(arg->data, "inactive=", 9) == 0) {
261 s.len = arg->len - 9;
262 s.data = arg->data + 9;
263 *inactive = ngx_parse_time(&s, 1);
264 return (*inactive == (time_t) NGX_ERROR) ? NGX_ERROR : NGX_OK;
265 }
266
267 if (ngx_strcmp(arg->data, "off") == 0) {
268 *off = 1;
269 return NGX_OK;
270 }
271
272 return NGX_ERROR; /* unknown token */
273}
