1/*
2 * vfs_mkdir.c — VFS directory creation.
3 *
4 * WHAT: Implements brix_vfs_mkdir(), which creates the resolved ctx path as a
5 * directory with the given mode, optionally creating missing parent
6 * components (`parents`).
7 *
8 * WHY: kXR_mkdir and WebDAV MKCOL need one write-gated, confined mkdir with the
9 * same parent-creation semantics and a single metric/access-log emission
10 * as the other namespace mutators.
11 *
12 * HOW: Enforces brix_vfs_require_write() and a non-NULL root_canon, then
13 * delegates to brix_ns_mkdir() (namespace layer) passing mode and the
14 * parents flag. The namespace status is mapped back to errno (sys_errno or
15 * EIO) and observed as BRIX_METRIC_OP_MKDIR on every path.
16 */
17#include "vfs_internal.h"
18#include "fs/path/path.h" /* brix_chmod_confined_canon (impersonation-aware) */

20/* vfs_backend_mkpath_leaf — leaf-aware recursive mkdir for non-POSIX backends.
21 *
22 * WHAT: Creates each prefix of `logical` in turn through brix_sd_mkdir_maybe_cred
23 * on `leaf`, tolerating EEXIST at each level. Like brix_vfs_backend_mkpath
24 * but dispatches on the leaf driver and threads a per-user credential so
25 * the remote origin receives the user's identity on every mkdir call.
26 *
27 * WHY: brix_vfs_backend_mkpath resolves the backend via brix_vfs_backend_resolve
28 * (which returns the TOP instance) and calls sd->driver->mkdir directly,
29 * bypassing both the leaf unwrap and the credential. When `parents` is set
30 * and use_cred is non-zero, we must present the user credential to the leaf
31 * for every intermediate mkdir — not just the final one. A local helper
32 * that already has `leaf` and `cred` avoids touching the shared
33 * brix_vfs_backend_mkpath signature (used by callers without cred).
34 *
35 * HOW: Walk `logical` component-by-component, accumulating the path prefix in
36 * `acc`, calling brix_sd_mkdir_maybe_cred(leaf, acc, mode, cred) for each
37 * non-root prefix (EEXIST-tolerant). Returns 0 on success, -1 with errno
38 * set on a real failure. NGX_DECLINED (positive) when the leaf has no
39 * mkdir slot (caller falls back to brix_vfs_backend_mkpath). */
40static int
41vfs_backend_mkpath_leaf(brix_sd_instance_t *leaf, const char *logical,
42 mode_t mode, const brix_sd_cred_t *cred)
43{
44 char acc[PATH_MAX];
45 size_t i = 0, j = 0;
46
47 if (leaf == NULL || leaf->driver->mkdir == NULL) {
48 /* Not a real failure — the caller falls back to
49 * brix_vfs_backend_mkpath — but set errno for clarity to any caller
50 * that inspects it before checking the NGX_DECLINED return. */
51 errno = ENOTSUP;
52 return NGX_DECLINED;
53 }
54
55 while (logical[i] != '\0') {
56 if (logical[i] != '/') {
57 i++;
58 continue;
59 }
60 if (j + 1 >= sizeof(acc)) {
61 errno = ENAMETOOLONG;
62 return -1;
63 }
64 acc[j++] = '/';
65 i++;
66 while (logical[i] != '\0' && logical[i] != '/') {
67 if (j + 1 >= sizeof(acc)) {
68 errno = ENAMETOOLONG;
69 return -1;
70 }
71 acc[j++] = logical[i++];
72 }
73 acc[j] = '\0';
74 if (j > 1
75 && brix_sd_mkdir_maybe_cred(leaf, acc, mode, cred) != NGX_OK
76 && errno != EEXIST) {
77 return -1;
78 }
79 }
80 return 0;
81}

83/* vfs_mkdir_req_t — the resolved, gated mkdir request threaded to the backend
84 * dispatch. WHAT: bundles ctx, the target `path`, mode, the parents flag, and
85 * the OP_MKDIR observation start timestamp. WHY: keeps the backend helpers'
86 * parameter lists small (one request pointer instead of five scalars) without
87 * introducing globals — all state is passed explicitly. HOW: filled once in
88 * brix_vfs_mkdir and passed by const pointer down the backend path. */
89typedef struct {
90 brix_vfs_ctx_t *ctx;
91 const char *path;
92 mode_t mode;
93 unsigned parents;
94 uint64_t start;
95} vfs_mkdir_req_t;

97/* vfs_backend_mkdir_dispatch — run the resolved mkdir against a non-POSIX
98 * driver, honouring `parents` and a per-user credential.
99 *
100 * WHAT: Given the already-gated `leaf` instance, credential, and use_cred flag,
101 * creates the request's export-relative path as a directory. For the
102 * parents case it walks the whole prefix chain (leaf-aware+cred when
103 * use_cred, else brix_vfs_backend_mkpath); otherwise a single
104 * brix_sd_mkdir_maybe_cred. Returns NGX_OK / NGX_ERROR with errno set.
105 *
106 * WHY: Isolates the three-way mkdir dispatch (no-slot / parents / single) from
107 * brix_vfs_mkdir so the caller stays under the complexity cap. The gate
108 * (caps, credential) is resolved by the caller and passed in explicitly —
109 * this helper is pure dispatch with side effects at the driver edge.
110 *
111 * HOW: Rejects a NULL mkdir slot with ENOTSUP. For `parents` picks the
112 * leaf-aware cred-threaded mkpath when use_cred, else the root-canon
113 * mkpath, mapping a 0/-1 return to NGX_OK/NGX_ERROR. The non-parents case
114 * returns brix_sd_mkdir_maybe_cred directly. */
115static ngx_int_t
116vfs_backend_mkdir_dispatch(const vfs_mkdir_req_t *req,
117 const brix_sd_driver_t *drv, brix_sd_instance_t *leaf,
118 int use_cred, const brix_sd_cred_t *cred)
119{
120 brix_vfs_ctx_t *ctx = req->ctx;
121 const char *logical = brix_vfs_export_relative(ctx, req->path);
122
123 if (drv->mkdir == NULL) {
124 errno = ENOTSUP;
125 return NGX_ERROR;
126 }
127 if (req->parents) {
128 int mkrc;
129
130 if (use_cred) {
131 /* Leaf-aware, cred-threaded mkpath for the per-user case. */
132 mkrc = vfs_backend_mkpath_leaf(leaf, logical, req->mode, cred);
133 } else {
134 mkrc = brix_vfs_backend_mkpath(ctx->root_canon, logical, req->mode,
135 ctx->log);
136 }
137 return (mkrc == 0) ? NGX_OK : NGX_ERROR;
138 }
139 return brix_sd_mkdir_maybe_cred(leaf, logical, req->mode,
140 use_cred ? cred : NULL);
141}

143/* vfs_mkdir_backend — mkdir path for a non-POSIX driver (caps gate + cred gate
144 * + dispatch + observe).
145 *
146 * WHAT: Enforces the catalog-mutation capability, resolves any per-user
147 * credential, dispatches the mkdir via vfs_backend_mkdir_dispatch, and
148 * emits the single OP_MKDIR observation. Returns the mkdir result.
149 *
150 * WHY: The non-POSIX backend case carries most of brix_vfs_mkdir's branching;
151 * hoisting it here keeps brix_vfs_mkdir a thin router (write gate,
152 * root_canon, backend-vs-POSIX) under the complexity cap without changing
153 * any ordering, error path, or observation.
154 *
155 * HOW: Rejects a driver lacking BRIX_SD_CAP_DIRS_WRITE with EPERM. When the
156 * credential gate is active, resolves the cred (EACCES/cred_err on
157 * failure). Both early exits observe before returning, exactly as the
158 * success path does. The credential is zeroed before the gate so an
159 * inactive-kind pointer is never handed to the driver. */
160static ngx_int_t
161vfs_mkdir_backend(const vfs_mkdir_req_t *req, const brix_sd_driver_t *drv)
162{
163 brix_vfs_ctx_t *ctx = req->ctx;
164 const char *path = req->path;
165 uint64_t start = req->start;
166 brix_sd_instance_t *leaf = brix_vfs_ns_leaf(ctx->sd);
167 brix_sd_ucred_t store;
168 brix_sd_cred_t cred;
169 ngx_int_t rc;
170 int saved_errno;
171 int use_cred = 0, cred_err = 0;
172
173 /* Zero before the gate: it fills only the active credential kind; an
174 * unzeroed cred hands a garbage inactive pointer to the driver's
175 * cred slot (bearer PASSTHROUGH would leave x509_proxy dangling). */
176 ngx_memzero(&store, sizeof(store));
177 ngx_memzero(&cred, sizeof(cred));
178
179 /* phase-71: catalog-mutation capability gate. A backend that can list
180 * directories (CAP_DIRS) but not mutate the catalog (no CAP_DIRS_WRITE)
181 * rejects mkdir uniformly rather than relying on a NULL vtable slot. */
182 if (!(drv->caps & BRIX_SD_CAP_DIRS_WRITE)) {
183 saved_errno = EPERM;
184 errno = saved_errno;
185 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
186 NGX_ERROR, saved_errno, start);
187 return NGX_ERROR;
188 }
189 if (brix_vfs_cred_gate_active(ctx)) {
190 if (brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err)
191 != NGX_OK)
192 {
193 saved_errno = cred_err ? cred_err : EACCES;
194 errno = saved_errno;
195 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR,
196 NULL, 0, NGX_ERROR, saved_errno,
197 start);
198 return NGX_ERROR;
199 }
200 }
201
202 rc = vfs_backend_mkdir_dispatch(req, drv, leaf, use_cred, &cred);
203 brix_sd_ucred_wipe(&store); /* secret consumed by mkdir; erase (A-4/T4) */
204
205 saved_errno = errno;
206 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
207 rc, saved_errno, start);
208 return rc;
209}

211/* Create the resolved ctx path as a directory (mode), creating parents when
212 * `parents`. Write-gated and confined; metered as OP_MKDIR. */
213ngx_int_t
214brix_vfs_mkdir(brix_vfs_ctx_t *ctx, mode_t mode, unsigned parents)
215{
216 brix_ns_result_t res;
217 const char *path;
218 uint64_t start;
219 int saved_errno;
220 const brix_sd_driver_t *drv;
221
222 start = brix_vfs_now_ns();
223 path = brix_vfs_ctx_path(ctx);
224
225 if (brix_vfs_require_write(ctx) != NGX_OK) {
226 saved_errno = errno;
227 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
228 NGX_ERROR, saved_errno, start);
229 return NGX_ERROR;
230 }
231
232 if (ctx->root_canon == NULL) {
233 errno = EINVAL;
234 saved_errno = errno;
235 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
236 NGX_ERROR, saved_errno, start);
237 return NGX_ERROR;
238 }
239
240 /* Non-POSIX backend: create the directory entry in the driver's namespace.
241 * The catalog enforces POSIX parent existence (a lone mkdir with missing
242 * ancestors is ENOENT), so `parents` walks the whole chain through the
243 * driver — prefix by prefix, EEXIST-tolerant.
244 * Dispatch on the leaf instance so brix_sd_mkdir_maybe_cred finds the leaf
245 * driver's mkdir_cred slot (decorators have only plain relays). */
246 drv = brix_vfs_ctx_driver(ctx);
247 if (drv != NULL) {
248 vfs_mkdir_req_t req;
249
250 req.ctx = ctx;
251 req.path = path;
252 req.mode = mode;
253 req.parents = parents;
254 req.start = start;
255 return vfs_mkdir_backend(&req, drv);
256 }
257
258 res = brix_ns_mkdir(ctx->log, ctx->root_canon,
259 brix_vfs_ctx_path(ctx), mode,
260 parents ? 1 : 0);
261 if (res.status == BRIX_NS_OK) {
262 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
263 NGX_OK, 0, start);
264 return NGX_OK;
265 }
266
267 errno = res.sys_errno != 0 ? res.sys_errno : EIO;
268 saved_errno = errno;
269 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
270 NGX_ERROR, saved_errno, start);
271 return NGX_ERROR;
272}

274/*
275 * brix_vfs_chmod — change the resolved path's permission bits through the VFS
276 * seam. Like brix_vfs_mkdir it delegates to the impersonation-aware confined
277 * helper (brix_chmod_confined_canon) rather than a raw fchmodat, so under
278 * impersonation the chmod is performed by the broker as the mapped user.
279 */
280ngx_int_t
281brix_vfs_chmod(brix_vfs_ctx_t *ctx, mode_t mode)
282{
283 if (brix_vfs_require_write(ctx) != NGX_OK) {
284 return NGX_ERROR;
285 }
286 if (ctx->root_canon == NULL) {
287 errno = EINVAL;
288 return NGX_ERROR;
289 }
290 /* A non-POSIX backend mutates mode through its setattr slot (e.g. pblock
291 * updates the catalog row). A backend with no setattr slot has no mutable
292 * metadata (block/object data-only namespaces) — accept as a no-op success so
293 * MKCOL/PUT flows that chmod do not fail. */
294 {
295 const brix_sd_driver_t *drv = brix_vfs_ctx_driver(ctx);
296 if (drv != NULL) {
297 brix_sd_ucred_t store;
298 brix_sd_cred_t cred;
299 brix_sd_setattr_t attr;
300 ngx_int_t chmod_rc;
301 int use_cred = 0, cred_err = 0;
302
303 /* Zero before the gate: it fills only the active credential kind;
304 * an unzeroed cred hands a garbage inactive pointer to the driver
305 * cred slot (bearer PASSTHROUGH would leave x509_proxy dangling). */
306 ngx_memzero(&cred, sizeof(cred));
307
308 if (brix_vfs_cred_gate_active(ctx)) {
309 if (brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err)
310 != NGX_OK)
311 {
312 errno = cred_err ? cred_err : EACCES;
313 return NGX_ERROR;
314 }
315 }
316 if (drv->setattr == NULL) {
317 brix_sd_ucred_wipe(&store); /* resolved but unused; erase */
318 return NGX_OK;
319 }
320 ngx_memzero(&attr, sizeof(attr));
321 attr.set_mode = 1;
322 attr.mode = mode;
323 /* Dispatch on the leaf so brix_sd_setattr_maybe_cred finds the
324 * leaf driver's setattr_cred slot (decorators relay to plain). */
325 chmod_rc = brix_sd_setattr_maybe_cred(brix_vfs_ns_leaf(ctx->sd),
326 brix_vfs_export_relative(ctx, brix_vfs_ctx_path(ctx)),
327 &attr, use_cred ? &cred : NULL) == NGX_OK
328 ? NGX_OK : NGX_ERROR;
329 brix_sd_ucred_wipe(&store); /* secret consumed; erase (A-4/T4) */
330 return chmod_rc;
331 }
332 }
333 if (brix_chmod_confined_canon(ctx->log, ctx->root_canon,
334 brix_vfs_ctx_path(ctx), mode) != 0) {
335 return NGX_ERROR; /* errno set by the helper */
336 }
337 return NGX_OK;
338}
