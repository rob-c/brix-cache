1/*
2 * vfs_mkdir.c — VFS directory creation.
3 *
4 * WHAT: Implements brix_vfs_mkdir(), which creates the resolved ctx path as a
5 * directory with the given mode, optionally creating missing parent
6 * components (`parents`).
7 *
8 * WHY: kXR_mkdir and WebDAV MKCOL need one write-gated, confined mkdir with the
9 * same parent-creation semantics and a single metric/access-log emission
10 * as the other namespace mutators.
11 *
12 * HOW: Enforces brix_vfs_require_write() and a non-NULL root_canon, then
13 * delegates to brix_ns_mkdir() (namespace layer) passing mode and the
14 * parents flag. The namespace status is mapped back to errno (sys_errno or
15 * EIO) and observed as BRIX_METRIC_OP_MKDIR on every path.
16 */
17#include "vfs_internal.h"
18#include "fs/path/path.h" /* brix_chmod_confined_canon (impersonation-aware) */
19
20/* Create the resolved ctx path as a directory (mode), creating parents when
21 * `parents`. Write-gated and confined; metered as OP_MKDIR. */
22ngx_int_t
23brix_vfs_mkdir(brix_vfs_ctx_t *ctx, mode_t mode, unsigned parents)
24{
25 brix_ns_result_t res;
26 const char *path;
27 uint64_t start;
28 int saved_errno;
29 const brix_sd_driver_t *drv;
30
31 start = brix_vfs_now_ns();
32 path = brix_vfs_ctx_path(ctx);
33
34 if (brix_vfs_require_write(ctx) != NGX_OK) {
35 saved_errno = errno;
36 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
37 NGX_ERROR, saved_errno, start);
38 return NGX_ERROR;
39 }
40
41 if (ctx->root_canon == NULL) {
42 errno = EINVAL;
43 saved_errno = errno;
44 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
45 NGX_ERROR, saved_errno, start);
46 return NGX_ERROR;
47 }
48
49 /* Non-POSIX backend: create the directory entry in the driver's
50 * namespace. The catalog enforces POSIX parent existence (a lone mkdir
51 * with missing ancestors is ENOENT), so `parents` walks the whole chain
52 * through the driver — prefix by prefix, EEXIST-tolerant. */
53 drv = brix_vfs_ctx_driver(ctx);
54 if (drv != NULL) {
55 ngx_int_t rc;
56
57 if (drv->mkdir == NULL) {
58 errno = ENOSYS;
59 rc = NGX_ERROR;
60 } else if (parents) {
61 rc = (brix_vfs_backend_mkpath(ctx->root_canon,
62 brix_vfs_export_relative(ctx, path), mode,
63 ctx->log) == 0) ? NGX_OK : NGX_ERROR;
64 } else {
65 rc = drv->mkdir(ctx->sd, brix_vfs_export_relative(ctx, path),
66 mode);
67 }
68
69 saved_errno = errno;
70 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
71 rc, saved_errno, start);
72 return rc;
73 }
74
75 res = brix_ns_mkdir(ctx->log, ctx->root_canon,
76 brix_vfs_ctx_path(ctx), mode,
77 parents ? 1 : 0);
78 if (res.status == BRIX_NS_OK) {
79 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
80 NGX_OK, 0, start);
81 return NGX_OK;
82 }
83
84 errno = res.sys_errno != 0 ? res.sys_errno : EIO;
85 saved_errno = errno;
86 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_MKDIR, NULL, 0,
87 NGX_ERROR, saved_errno, start);
88 return NGX_ERROR;
89}

91/*
92 * brix_vfs_chmod — change the resolved path's permission bits through the VFS
93 * seam. Like brix_vfs_mkdir it delegates to the impersonation-aware confined
94 * helper (brix_chmod_confined_canon) rather than a raw fchmodat, so under
95 * impersonation the chmod is performed by the broker as the mapped user.
96 */
97ngx_int_t
98brix_vfs_chmod(brix_vfs_ctx_t *ctx, mode_t mode)
99{
100 if (brix_vfs_require_write(ctx) != NGX_OK) {
101 return NGX_ERROR;
102 }
103 if (ctx->root_canon == NULL) {
104 errno = EINVAL;
105 return NGX_ERROR;
106 }
107 /* A non-POSIX backend mutates mode through its setattr slot (e.g. pblock
108 * updates the catalog row). A backend with no setattr slot has no mutable
109 * metadata (block/object data-only namespaces) — accept as a no-op success so
110 * MKCOL/PUT flows that chmod do not fail. */
111 {
112 const brix_sd_driver_t *drv = brix_vfs_ctx_driver(ctx);
113 if (drv != NULL) {
114 brix_sd_setattr_t attr;
115 if (drv->setattr == NULL) {
116 return NGX_OK;
117 }
118 ngx_memzero(&attr, sizeof(attr));
119 attr.set_mode = 1;
120 attr.mode = mode;
121 return drv->setattr(ctx->sd,
122 brix_vfs_export_relative(ctx, brix_vfs_ctx_path(ctx)),
123 &attr) == NGX_OK ? NGX_OK : NGX_ERROR;
124 }
125 }
126 if (brix_chmod_confined_canon(ctx->log, ctx->root_canon,
127 brix_vfs_ctx_path(ctx), mode) != 0) {
128 return NGX_ERROR; /* errno set by the helper */
129 }
130 return NGX_OK;
131}

133/*
134 * brix_vfs_setattr — apply kXR_setattr (times and/or owner) to the resolved
135 * path through the VFS seam. A non-POSIX backend routes to its setattr slot
136 * (no-op success if it has none); the default POSIX path uses the
137 * impersonation-aware confined utimensat/fchownat helper so under impersonation
138 * the change is performed by the broker as the mapped user. The unified slot also
139 * carries mode, so a backend satisfies chmod and setattr through one entry point —
140 * kXR_setattr itself never sets mode (that is kXR_chmod's job).
141 */
142ngx_int_t
143brix_vfs_setattr(brix_vfs_ctx_t *ctx, const brix_sd_setattr_t *attr)
144{
145 if (brix_vfs_require_write(ctx) != NGX_OK) {
146 return NGX_ERROR;
147 }
148 if (ctx->root_canon == NULL || attr == NULL) {
149 errno = EINVAL;
150 return NGX_ERROR;
151 }
152
153 {
154 const brix_sd_driver_t *drv = brix_vfs_ctx_driver(ctx);
155 if (drv != NULL) {
156 if (drv->setattr == NULL) {
157 return NGX_OK; /* no mutable metadata — no-op success */
158 }
159 return drv->setattr(ctx->sd,
160 brix_vfs_export_relative(ctx, brix_vfs_ctx_path(ctx)),
161 attr) == NGX_OK ? NGX_OK : NGX_ERROR;
162 }
163 }
164
165 {
166 struct timespec times[2];
167 times[0] = attr->atime;
168 times[1] = attr->mtime;
169 if (brix_setattr_confined_canon(ctx->log, ctx->root_canon,
170 brix_vfs_ctx_path(ctx), attr->set_times, times,
171 attr->set_owner, attr->uid, attr->gid) != 0) {
172 return NGX_ERROR;
173 }
174 if (attr->set_mode
175 && brix_chmod_confined_canon(ctx->log, ctx->root_canon,
176 brix_vfs_ctx_path(ctx), attr->mode) != 0) {
177 return NGX_ERROR;
178 }
179 }
180 return NGX_OK;
181}
