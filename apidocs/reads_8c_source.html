4/*
5 * reads.c — thread-pool offload for the stream kXR_read / kXR_pgread opcodes.
6 *
7 * WHAT: Three read paths share this file. (1) Windowed memory read
8 * (brix_read_window_pump / _emit) streams a large memory-backed
9 * (TLS / non-regular) read as fill->drain->fill chunks. (2) Plain
10 * kXR_read AIO (brix_read_aio_thread / _done) offloads a single
11 * file reads to a worker thread and returns one chained response.
12 * (3) pgread AIO (brix_pgread_aio_thread / _done) does the same but
13 * interleaves a per-page CRC32C into the wire output.
14 *
15 * WHY: File I/O (and the pgread CRC32C loop) can block; running them on the
16 * nginx event-loop thread would stall every other connection on this
17 * worker. The thread pool absorbs the blocking syscall and CPU-bound
18 * checksum so the event loop stays responsive.
19 *
20 * HOW: Each path splits into a *_thread half (runs on a worker thread, may
21 * only touch the task struct — never ctx/connection/pool) and a *_done
22 * half (runs back on the event loop, owns all protocol/state mutation
23 * and chain building). The two halves communicate only through the
24 * brix_{read,pgread}_aio_t task struct carried by ngx_thread_task_t.
25 * Every *_done first re-validates the connection via
26 * brix_aio_restore_stream/_request (the stream may have died while the
27 * task ran) and ends by calling brix_aio_resume() to re-arm events.
28 */
29

34/*
35 * brix_read_window_emit — build + queue one window's wire chunk from
36 * rd.read_scratch[0..nread), advancing the continuation state. Status is
37 * kXR_oksofar for every window except the last (or a short read at EOF), which
38 * is kXR_ok. Returns NGX_ERROR if the read failed or the chain could not be
39 * built (an error response has already been sent); otherwise NGX_OK, with
40 * ctx->state == XRD_ST_SENDING if the chunk is still draining and
41 * ctx->rd.win_active cleared when this was the final window.
42 */
43static ngx_int_t
44brix_read_window_emit(brix_ctx_t *ctx, ngx_connection_t *c,
45 ssize_t nread, int io_errno)
46{
47 ngx_chain_t *chain;
48 uint16_t status;
49 size_t got;
50
51 if (nread < 0) {
52 ctx->rd.win_active = 0;
53 ctx->state = XRD_ST_REQ_HEADER;
54 ctx->recv.hdr_pos = 0;
55 BRIX_OP_ERR(ctx, BRIX_OP_READ);
56 brix_send_error(ctx, c, kXR_IOError,
57 io_errno ? strerror(io_errno) : "async read error");
58 return NGX_ERROR;
59 }
60
61 got = (size_t) nread;
62 ctx->files[ctx->rd.win_idx].bytes_read += got;
63 ctx->totals.bytes += got;
64
65 if (got < ctx->rd.win_remaining) {
66 ctx->rd.win_remaining -= got;
67 ctx->rd.win_offset += (off_t) got;
68 } else {
69 ctx->rd.win_remaining = 0;
70 }
71
72 /* Last planned window, or a short read (EOF), terminates the response. */
73 status = (ctx->rd.win_remaining == 0 || got == 0) ? kXR_ok : kXR_oksofar;
74
75 chain = brix_build_window_chain(ctx, c, ctx->rd.read_scratch, got, status);
76 if (chain == NULL) {
77 ctx->rd.win_active = 0;
78 ctx->state = XRD_ST_REQ_HEADER;
79 return NGX_ERROR;
80 }
81
82 if (status == kXR_ok) {
83 ctx->rd.win_active = 0;
84 BRIX_OP_OK(ctx, BRIX_OP_READ);
85 }
86
87 ctx->state = XRD_ST_REQ_HEADER;
88 ctx->recv.hdr_pos = 0;
89 brix_queue_response_chain(ctx, c, chain, ctx->rd.read_scratch);
90 if (ctx->state != XRD_ST_SENDING) {
91 brix_release_read_buffer(ctx, c, ctx->rd.read_scratch); /* no-op slot */
92 }
93 return NGX_OK;
94}

96/*
97 * brix_read_window_pump — read the next window into rd.read_scratch and emit it,
98 * looping while sends complete synchronously. Posts an AIO task when a thread
99 * pool is available (returns with state XRD_ST_AIO; brix_read_aio_done resumes
100 * the pump); otherwise reads the window inline (bounded to one window). When
101 * the windowed read finishes it resumes the event loop for the next request.
102 */
103void
104brix_read_window_pump(brix_ctx_t *ctx, ngx_connection_t *c,
105 ngx_stream_brix_srv_conf_t *rconf)
106{
107 for ( ;; ) {
108 size_t want;
109 u_char *databuf;
110 ssize_t nread;
111
112 if (!ctx->rd.win_active) {
113 brix_aio_resume(c);
114 return;
115 }
116
117 want = ctx->rd.win_remaining < (size_t) BRIX_READ_WINDOW
118 ? ctx->rd.win_remaining : (size_t) BRIX_READ_WINDOW;
119
120 databuf = BRIX_GET_SCRATCH(ctx, c, rd.read_scratch, rd.read_scratch_size,
121 want);
122 if (databuf == NULL) {
123 ctx->rd.win_active = 0;
124 ctx->state = XRD_ST_REQ_HEADER;
125 brix_send_error(ctx, c, kXR_NoMemory, "read window alloc failed");
126 brix_aio_resume(c);
127 return;
128 }
129 brix_budget_sync(ctx);
130
131 if (rconf->common.thread_pool != NULL) {
132 ngx_thread_task_t *task = ctx->rd.read_aio_task;
133 brix_read_aio_t *t;
134 ngx_flag_t posted = 0;
135
136 /*
137 * One task struct is allocated once per stream and cached on
138 * ctx->rd.read_aio_task, then reused across every window of this read
139 * (and across later reads) to avoid a pool allocation per window.
140 * On reuse the task is dirty from its last trip through the pool:
141 * task->next must be cleared (the pool threads it onto its run
142 * queue) and event.complete reset to 0 (ngx_post_event set it when
143 * the previous completion fired) or the next post would be ignored.
144 */
145 if (task == NULL) {
146 task = ngx_thread_task_alloc(c->pool,
147 sizeof(brix_read_aio_t));
148 if (task != NULL) {
149 ctx->rd.read_aio_task = task;
150 }
151 } else {
152 task->next = NULL;
153 task->event.complete = 0;
154 }
155
156 if (task != NULL) {
157 t = task->ctx;
158 t->c = c;
159 t->ctx = ctx;
160 t->conf = rconf;
161 t->fd = ctx->rd.win_fd;
162 t->handle_idx = ctx->rd.win_idx;
163 t->offset = ctx->rd.win_offset;
164 t->rlen = want;
165 t->databuf = databuf;
166 /*
167 * Snapshot the 2-word streamid into the task so the completion
168 * callback can verify it still matches the live ctx — by the
169 * time the worker finishes, this connection may have been torn
170 * down and the slot reused by an unrelated stream.
171 */
172 t->streamid[0] = ctx->rd.win_streamid[0];
173 t->streamid[1] = ctx->rd.win_streamid[1];
174 t->nread = 0;
175 t->io_errno = 0;
176 brix_task_bind(task, brix_read_aio_thread,
177 brix_read_aio_done);
178 (void) brix_aio_post_task(ctx, c, rconf->common.thread_pool,
179 task, "brix: window task post failed, sync fallback",
180 &posted);
181 if (posted) {
182 return; /* async: done callback resumes the pump */
183 }
184 /* post failed (queue full): fall through to the inline read. */
185 }
186 }
187
188 /*
189 * Inline fallback (no pool configured, or post failed): do the blocking
190 * VFS read on the event-loop thread for this one window only, then let
191 * the for(;;) loop pick up the next window. Bounded to a single window
192 * so a large read can never monopolise the loop for more than
193 * BRIX_READ_WINDOW.
194 */
195 {
196 brix_vfs_job_t job;
197
198 brix_vfs_job_read_init(&job, ctx->rd.win_fd,
199 ctx->rd.win_offset, want, databuf,
200 want, 0);
201 brix_vfs_io_execute(&job);
202 nread = job.nio;
203 if (brix_read_window_emit(ctx, c, nread, job.io_errno)
204 == NGX_ERROR)
205 {
206 brix_aio_resume(c);
207 return;
208 }
209 }
210 if (ctx->state == XRD_ST_SENDING) {
211 return; /* async send: send.c resumes the pump on drain */
212 }
213 /* sync send complete → loop reads the next window */
214 }
215}

217/*
218 * brix_read_aio_thread — thread-pool worker for kXR_read.
219 *
220 * Runs on a worker thread; must not touch nginx state, connection pools, or
221 * any field that is not owned by the task struct. Only the blocking VFS core
222 * syscall belongs here; all protocol work happens in the done callback on the
223 * main thread.
224 *
225 * t->nread: set to pread return value (< 0 on error, 0 on EOF, > 0 on success).
226 * t->io_errno: saved errno on failure.
227 */
228void
229brix_read_aio_thread(void *data, ngx_log_t *log)
230{
231 brix_read_aio_t *t = data;
232 brix_vfs_job_t job;
233
234 /*
235 * Worker threads execute the VFS-owned thread-safe core only; all protocol
236 * state updates stay on the event-loop side in the completion callback.
237 */
238 brix_vfs_job_read_init(&job, t->fd, t->offset, t->rlen,
239 t->databuf, t->rlen, 0);
240 job.csi = t->csi; /* phase-59 W2: verify in the worker */
241 brix_vfs_job_set_obj(&job, &t->obj); /* Layer 3: route via driver if bound */
242 brix_vfs_io_execute(&job);
243
244 t->nread = job.nio;
245 t->io_errno = job.io_errno; /* CSI mismatch surfaces as EIO here */
246}

248/*
249 * brix_read_aio_done — main-thread completion callback for kXR_read AIO.
250 *
251 * Called by nginx's event loop after the thread pool posts the result via
252 * ngx_post_event. Responsibilities:
253 * 1. Guard against stale connection (ctx->destroyed check via restore_stream).
254 * 2. On I/O error: send kXR_IOError, release databuf, resume event loop.
255 * 3. On success: build a response chain (chunked if > 16 MiB), update per-
256 * handle and session byte counters, queue the chain.
257 * 4. Call brix_aio_resume() to re-arm the appropriate event (write or read).
258 *
259 * NOTE: if the chain send blocks (XRD_ST_SENDING), databuf is kept alive as
260 * wchain_base and freed by brix_release_pending_buffer after full drain.
261 */
262void
263brix_read_aio_done(ngx_event_t *ev)
264{
265 ngx_thread_task_t *task = ev->data;
266 brix_read_aio_t *t = task->ctx;
267 brix_ctx_t *ctx = t->ctx;
268 ngx_connection_t *c = t->c;
269 ngx_chain_t *rsp_chain;
270
271 if (!brix_aio_restore_stream(ctx, t->streamid)) {
272 return;
273 }
274
275 if (ctx->rd.win_active) {
276 /*
277 * Phase 31 W2.1: this completion belongs to one window of a windowed
278 * read. Emit its chunk, then continue (next window) or finish.
279 */
280 if (brix_read_window_emit(ctx, c, t->nread, t->io_errno)
281 == NGX_ERROR)
282 {
283 brix_aio_resume(c);
284 return;
285 }
286 /*
287 * After emit, exactly one of three things is true and each takes a
288 * different path (no fall-through — every branch returns):
289 * a) chunk is still draining (XRD_ST_SENDING): hand off; send.c will
290 * re-enter the pump once the socket has flushed this window.
291 * b) chunk sent synchronously and more windows remain
292 * (rd_win_active still set): drive the next window now.
293 * c) windowed read complete (rd_win_active cleared by emit): the read
294 * is done, so resume the normal request event loop.
295 */
296 if (ctx->state == XRD_ST_SENDING) {
297 return; /* (a) send.c resumes the pump when the chunk drains */
298 }
299 if (ctx->rd.win_active) {
300 brix_read_window_pump(ctx, c, t->conf); /* (b) sync-sent: next window */
301 return;
302 }
303 brix_aio_resume(c); /* (c) finished */
304 return;
305 }
306
307 if (t->nread < 0) {
308 ctx->state = XRD_ST_REQ_HEADER;
309 ctx->recv.hdr_pos = 0;
310 brix_release_read_buffer(ctx, c, t->databuf);
311 BRIX_OP_ERR(ctx, BRIX_OP_READ);
312 brix_send_error(ctx, c, kXR_IOError,
313 t->io_errno ? strerror(t->io_errno) : "async read error");
314 brix_aio_resume(c);
315 return;
316 }
317
318 ctx->files[t->handle_idx].bytes_read += (size_t) t->nread;
319 ctx->totals.bytes += (size_t) t->nread;
320 BRIX_OP_OK(ctx, BRIX_OP_READ);
321
322 rsp_chain = brix_build_chunked_chain(ctx, c,
323 t->databuf, (size_t) t->nread);
324 if (rsp_chain == NULL) {
325 brix_release_read_buffer(ctx, c, t->databuf);
326 ctx->state = XRD_ST_REQ_HEADER;
327 brix_aio_resume(c);
328 return;
329 }
330
331 ctx->state = XRD_ST_REQ_HEADER;
332 ctx->recv.hdr_pos = 0;
333
334 brix_queue_response_chain(ctx, c, rsp_chain, t->databuf);
335 if (ctx->state != XRD_ST_SENDING) {
336 brix_release_read_buffer(ctx, c, t->databuf);
337 } else {
338 /*
339 * Parked and draining: the buffer is a per-in-flight rd_pool slot and the
340 * header is per-slot, so this memory-backed read is SAFE to pipeline. Note
341 * the cold-AIO path does not yet ACTUALLY pipeline: recv already suspended
342 * on this read's AIO and brix_aio_resume() only drains the write side, so
343 * the next read is not issued until this response drains. Setting the flag
344 * is correct and forward-compatible with the non-suspending read-AIO change
345 * that will let cold reads pipeline like the warm-cache inline path does.
346 * Single-chunk only (non-windowed read <= BRIX_READ_WINDOW < CHUNK_MAX).
347 */
348 ctx->out.resp_pipelinable = 1;
349 }
350 brix_aio_resume(c);
351}

392/*
393 * brix_pgread_aio_done — response builder for pgread AIO completion.
394 *
395 * pgread wire format ([CRC32C(4)][page data] × N) cannot use
396 * brix_build_chunked_chain and requires direct chain construction with a
397 * ServerStatusResponse_pgRead header.

399void
400brix_pgread_aio_done(ngx_event_t *ev)
401{

404 brix_ctx_t *ctx = t->ctx;
405 ngx_connection_t *c = t->c;

407 ngx_chain_t *rsp_chain;
408 ngx_stream_brix_srv_conf_t *rconf;
409
410 if (!brix_aio_restore_request(ctx, t->streamid)) {
411 return;
412 }
413
414 if (t->nread < 0) {
415 brix_release_read_buffer(ctx, c, t->scratch);
416 BRIX_OP_ERR(ctx, BRIX_OP_PGREAD);
417 brix_send_error(ctx, c, kXR_IOError,
418 t->io_errno ? strerror(t->io_errno) : "async pgread error");
419 brix_aio_resume(c);
420 return;
421 }
422
423 /*

425 * buffer at all. The client reads the header, sees zero payload bytes, and
426 * treats it as end-of-data — there are no pages, hence no CRC32C words.

441 * own per-page CRC32c and must be sent verbatim behind the pgRead status
442 * header — never through brix_build_chunked_chain (wrong kXR_ok framing).
443 * Shared with the synchronous handler via brix_build_pgread_chain. */

445 (uint32_t) t->out_size);
446 if (rsp_chain == NULL) {
447 brix_release_read_buffer(ctx, c, t->scratch);
448 brix_aio_resume(c);
449 return;
450 }
451
452 ctx->files[t->handle_idx].bytes_read += (size_t) t->nread;
453 ctx->totals.bytes += (size_t) t->nread;
454
455 rconf = ngx_stream_get_module_srv_conf(
456 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
457 if (rconf->access_log_fd != NGX_INVALID_FILE) {
458 char detail[64];
459 snprintf(detail, sizeof(detail), "%lld+%zu",
460 (long long) t->offset, (size_t) t->nread);
461 brix_log_access(ctx, c, "PGREAD", ctx->files[t->handle_idx].path,
462 detail, 1, 0, NULL, (size_t) t->nread);
463 }
464 BRIX_OP_OK(ctx, BRIX_OP_PGREAD);
465
466 brix_queue_response_chain(ctx, c, rsp_chain, t->scratch);
467 if (ctx->state != XRD_ST_SENDING) {
468 brix_release_read_buffer(ctx, c, t->scratch);
469 }
470 brix_aio_resume(c);
471}
