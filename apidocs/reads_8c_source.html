4/*
5 * reads.c — thread-pool offload for the stream kXR_read / kXR_pgread opcodes.
6 *
7 * WHAT: Three read paths share this file. (1) Windowed memory read
8 * (brix_read_window_pump / _emit) streams a large memory-backed
9 * (TLS / non-regular) read as fill->drain->fill chunks. (2) Plain
10 * kXR_read AIO (brix_read_aio_thread / _done) offloads a single
11 * file reads to a worker thread and returns one chained response.
12 * (3) pgread AIO (brix_pgread_aio_thread / _done) does the same but
13 * interleaves a per-page CRC32C into the wire output.
14 *
15 * WHY: File I/O (and the pgread CRC32C loop) can block; running them on the
16 * nginx event-loop thread would stall every other connection on this
17 * worker. The thread pool absorbs the blocking syscall and CPU-bound
18 * checksum so the event loop stays responsive.
19 *
20 * HOW: Each path splits into a *_thread half (runs on a worker thread, may
21 * only touch the task struct — never ctx/connection/pool) and a *_done
22 * half (runs back on the event loop, owns all protocol/state mutation
23 * and chain building). The two halves communicate only through the
24 * brix_{read,pgread}_aio_t task struct carried by ngx_thread_task_t.
25 * Every *_done first re-validates the connection via
26 * brix_aio_restore_stream/_request (the stream may have died while the
27 * task ran) and ends by calling brix_aio_resume() to re-arm events.
28 */

53/*
54 * brix_read_window_emit — build + queue one window's wire chunk from
55 * rd.read_scratch[0..nread), advancing the continuation state. Status is
56 * kXR_oksofar for every window except the last (or a short read at EOF), which
57 * is kXR_ok. Returns NGX_ERROR if the read failed or the chain could not be
58 * built (an error response has already been sent); otherwise NGX_OK, with
59 * ctx->state == XRD_ST_SENDING if the chunk is still draining and
60 * ctx->rd.win_active cleared when this was the final window.
61 */
62static ngx_int_t
63brix_read_window_emit(brix_ctx_t *ctx, ngx_connection_t *c,
64 ssize_t nread, int io_errno)
65{
66 ngx_chain_t *chain;
67 uint16_t status;
68 size_t got;
69
70 if (nread < 0) {
71 brix_read_io_failure_log(c->log, "windowed", ctx->rd.win_fd,
72 ctx->rd.win_offset, ctx->rd.win_remaining,
73 io_errno);
74 ctx->rd.win_active = 0;
75 ctx->state = XRD_ST_REQ_HEADER;
76 ctx->recv.hdr_pos = 0;
77 BRIX_OP_ERR(ctx, BRIX_OP_READ);
78 brix_send_error(ctx, c, kXR_IOError,
79 io_errno ? strerror(io_errno) : "async read error");
80 return NGX_ERROR;
81 }
82
83 got = (size_t) nread;
84 ctx->files[ctx->rd.win_idx].bytes_read += got;
85 ctx->totals.bytes += got;
86
87 if (got < ctx->rd.win_remaining) {
88 ctx->rd.win_remaining -= got;
89 ctx->rd.win_offset += (off_t) got;
90 } else {
91 ctx->rd.win_remaining = 0;
92 }
93
94 /* Last planned window, or a short read (EOF), terminates the response. */
95 status = (ctx->rd.win_remaining == 0 || got == 0) ? kXR_ok : kXR_oksofar;
96
97 chain = brix_build_window_chain(ctx, c, ctx->rd.read_scratch, got, status);
98 if (chain == NULL) {
99 ctx->rd.win_active = 0;
100 ctx->state = XRD_ST_REQ_HEADER;
101 return NGX_ERROR;
102 }
103
104 if (status == kXR_ok) {
105 ctx->rd.win_active = 0;
106 BRIX_OP_OK(ctx, BRIX_OP_READ);
107 }
108
109 ctx->state = XRD_ST_REQ_HEADER;
110 ctx->recv.hdr_pos = 0;
111 brix_queue_response_chain(ctx, c, chain, ctx->rd.read_scratch);
112 if (ctx->state != XRD_ST_SENDING) {
113 brix_release_read_buffer(ctx, c, ctx->rd.read_scratch); /* no-op slot */
114 }
115 return NGX_OK;
116}

118/*
119 * brix_read_window_pump — read the next window into rd.read_scratch and emit it,
120 * looping while sends complete synchronously. Posts an AIO task when a thread
121 * pool is available (returns with state XRD_ST_AIO; brix_read_aio_done resumes
122 * the pump); otherwise reads the window inline (bounded to one window). When
123 * the windowed read finishes it resumes the event loop for the next request.
124 */
125void
126brix_read_window_pump(brix_ctx_t *ctx, ngx_connection_t *c,
127 ngx_stream_brix_srv_conf_t *rconf)
128{
129 for ( ;; ) {
130 size_t want;
131 u_char *databuf;
132 ssize_t nread;
133
134 if (!ctx->rd.win_active) {
135 brix_aio_resume(c);
136 return;
137 }
138
139 want = ctx->rd.win_remaining < (size_t) BRIX_READ_WINDOW
140 ? ctx->rd.win_remaining : (size_t) BRIX_READ_WINDOW;
141
142 databuf = BRIX_GET_SCRATCH(ctx, c, rd.read_scratch, rd.read_scratch_size,
143 want);
144 if (databuf == NULL) {
145 ctx->rd.win_active = 0;
146 ctx->state = XRD_ST_REQ_HEADER;
147 brix_send_error(ctx, c, kXR_NoMemory, "read window alloc failed");
148 brix_aio_resume(c);
149 return;
150 }
151 brix_budget_sync(ctx);
152
153 if (rconf->common.thread_pool != NULL) {
154 ngx_thread_task_t *task = ctx->rd.read_aio_task;
155 brix_read_aio_t *t;
156 ngx_flag_t posted = 0;
157
158 /*
159 * One task struct is allocated once per stream and cached on
160 * ctx->rd.read_aio_task, then reused across every window of this read
161 * (and across later reads) to avoid a pool allocation per window.
162 * On reuse the task is dirty from its last trip through the pool:
163 * task->next must be cleared (the pool threads it onto its run
164 * queue) and event.complete reset to 0 (ngx_post_event set it when
165 * the previous completion fired) or the next post would be ignored.
166 */
167 if (task == NULL) {
168 task = ngx_thread_task_alloc(c->pool,
169 sizeof(brix_read_aio_t));
170 if (task != NULL) {
171 ctx->rd.read_aio_task = task;
172 }
173 } else {
174 task->next = NULL;
175 task->event.complete = 0;
176 }
177
178 if (task != NULL) {
179 t = task->ctx;
180 t->c = c;
181 t->ctx = ctx;
182 t->conf = rconf;
183 t->fd = ctx->rd.win_fd;
184 t->handle_idx = ctx->rd.win_idx;
185 t->offset = ctx->rd.win_offset;
186 t->rlen = want;
187 t->databuf = databuf;
188 /*
189 * Snapshot the 2-word streamid into the task so the completion
190 * callback can verify it still matches the live ctx — by the
191 * time the worker finishes, this connection may have been torn
192 * down and the slot reused by an unrelated stream.
193 */
194 t->streamid[0] = ctx->rd.win_streamid[0];
195 t->streamid[1] = ctx->rd.win_streamid[1];
196 t->nread = 0;
197 t->io_errno = 0;
198 /*
199 * The task struct is shared with the single-shot buffered path
200 * (read_post_aio), so csi/obj still hold the LAST read's handle
201 * state. The worker routes the pread through t->obj's driver
202 * (which carries its own fd) whenever obj.driver != NULL, so a
203 * stale obj sends this window to the previous handle's fd —
204 * wrong file if the number was recycled, EBADF if closed or
205 * write-only — and a stale csi is a dangling pointer once that
206 * handle closed. Rebind both to the windowed read's own handle
207 * on every post.
208 */
209 t->csi = ctx->files[ctx->rd.win_idx].csi;
210 t->obj = ctx->files[ctx->rd.win_idx].sd_obj;
211 brix_task_bind(task, brix_read_aio_thread,
212 brix_read_aio_done);
213 (void) brix_aio_post_task(ctx, c, rconf->common.thread_pool,
214 task, "brix: window task post failed, sync fallback",
215 &posted);
216 if (posted) {
217 return; /* async: done callback resumes the pump */
218 }
219 /* post failed (queue full): fall through to the inline read. */
220 }
221 }
222
223 /*
224 * Inline fallback (no pool configured, or post failed): do the blocking
225 * VFS read on the event-loop thread for this one window only, then let
226 * the for(;;) loop pick up the next window. Bounded to a single window
227 * so a large read can never monopolise the loop for more than
228 * BRIX_READ_WINDOW.
229 */
230 {
231 brix_vfs_job_t job;
232
233 brix_vfs_job_read_init(&job, ctx->rd.win_fd,
234 ctx->rd.win_offset, want, databuf,
235 want, 0);
236 /* Same handle binding as the posted task above: verify CSI pages
237 * and route driver-backed handles through their storage object. */
238 job.csi = ctx->files[ctx->rd.win_idx].csi;
239 brix_vfs_job_set_obj(&job, &ctx->files[ctx->rd.win_idx].sd_obj);
240 brix_vfs_io_execute(&job);
241 nread = job.nio;
242 if (brix_read_window_emit(ctx, c, nread, job.io_errno)
243 == NGX_ERROR)
244 {
245 brix_aio_resume(c);
246 return;
247 }
248 }
249 if (ctx->state == XRD_ST_SENDING) {
250 return; /* async send: send.c resumes the pump on drain */
251 }
252 /* sync send complete → loop reads the next window */
253 }
254}

256/*
257 * brix_read_aio_thread — thread-pool worker for kXR_read.
258 *
259 * Runs on a worker thread; must not touch nginx state, connection pools, or
260 * any field that is not owned by the task struct. Only the blocking VFS core
261 * syscall belongs here; all protocol work happens in the done callback on the
262 * main thread.
263 *
264 * t->nread: set to pread return value (< 0 on error, 0 on EOF, > 0 on success).
265 * t->io_errno: saved errno on failure.
266 */
267void
268brix_read_aio_thread(void *data, ngx_log_t *log)
269{
270 brix_read_aio_t *t = data;
271 brix_vfs_job_t job;
272
273 /*
274 * Worker threads execute the VFS-owned thread-safe core only; all protocol
275 * state updates stay on the event-loop side in the completion callback.
276 */
277 brix_vfs_job_read_init(&job, t->fd, t->offset, t->rlen,
278 t->databuf, t->rlen, 0);
279 job.csi = t->csi; /* phase-59 W2: verify in the worker */
280 brix_vfs_job_set_obj(&job, &t->obj); /* Layer 3: route via driver if bound */
281 brix_vfs_io_execute(&job);
282
283 t->nread = job.nio;
284 t->io_errno = job.io_errno; /* CSI mismatch surfaces as EIO here */
285}

287/*
288 * brix_read_aio_done — main-thread completion callback for kXR_read AIO.
289 *
290 * Called by nginx's event loop after the thread pool posts the result via
291 * ngx_post_event. Responsibilities:
292 * 1. Guard against stale connection (ctx->destroyed check via restore_stream).
293 * 2. On I/O error: send kXR_IOError, release databuf, resume event loop.
294 * 3. On success: build a response chain (chunked if > 16 MiB), update per-
295 * handle and session byte counters, queue the chain.
296 * 4. Call brix_aio_resume() to re-arm the appropriate event (write or read).
297 *
298 * NOTE: if the chain send blocks (XRD_ST_SENDING), databuf is kept alive as
299 * wchain_base and freed by brix_release_pending_buffer after full drain.
300 */
301void
302brix_read_aio_done(ngx_event_t *ev)
303{
304 ngx_thread_task_t *task = ev->data;
305 brix_read_aio_t *t = task->ctx;
306 brix_ctx_t *ctx = t->ctx;
307 ngx_connection_t *c = t->c;
308 ngx_chain_t *rsp_chain;
309
310 if (!brix_aio_restore_stream(ctx, t->streamid)) {
311 return;
312 }
313
314 if (ctx->rd.win_active) {
315 /*
316 * Phase 31 W2.1: this completion belongs to one window of a windowed
317 * read. Emit its chunk, then continue (next window) or finish.
318 */
319 if (brix_read_window_emit(ctx, c, t->nread, t->io_errno)
320 == NGX_ERROR)
321 {
322 brix_aio_resume(c);
323 return;
324 }
325 /*
326 * After emit, exactly one of three things is true and each takes a
327 * different path (no fall-through — every branch returns):
328 * a) chunk is still draining (XRD_ST_SENDING): hand off; send.c will
329 * re-enter the pump once the socket has flushed this window.
330 * b) chunk sent synchronously and more windows remain
331 * (rd_win_active still set): drive the next window now.
332 * c) windowed read complete (rd_win_active cleared by emit): the read
333 * is done, so resume the normal request event loop.
334 */
335 if (ctx->state == XRD_ST_SENDING) {
336 return; /* (a) send.c resumes the pump when the chunk drains */
337 }
338 if (ctx->rd.win_active) {
339 brix_read_window_pump(ctx, c, t->conf); /* (b) sync-sent: next window */
340 return;
341 }
342 brix_aio_resume(c); /* (c) finished */
343 return;
344 }
345
346 if (t->nread < 0) {
347 brix_read_io_failure_log(c->log, "read-aio", t->fd, t->offset,
348 t->rlen, t->io_errno);
349 ctx->state = XRD_ST_REQ_HEADER;
350 ctx->recv.hdr_pos = 0;
351 brix_release_read_buffer(ctx, c, t->databuf);
352 BRIX_OP_ERR(ctx, BRIX_OP_READ);
353 brix_send_error(ctx, c, kXR_IOError,
354 t->io_errno ? strerror(t->io_errno) : "async read error");
355 brix_aio_resume(c);
356 return;
357 }
358
359 ctx->files[t->handle_idx].bytes_read += (size_t) t->nread;
360 ctx->totals.bytes += (size_t) t->nread;
361 BRIX_OP_OK(ctx, BRIX_OP_READ);
362
363 rsp_chain = brix_build_chunked_chain(ctx, c,
364 t->databuf, (size_t) t->nread);
365 if (rsp_chain == NULL) {
366 brix_release_read_buffer(ctx, c, t->databuf);
367 ctx->state = XRD_ST_REQ_HEADER;
368 brix_aio_resume(c);
369 return;
370 }
371
372 ctx->state = XRD_ST_REQ_HEADER;
373 ctx->recv.hdr_pos = 0;
374
375 brix_queue_response_chain(ctx, c, rsp_chain, t->databuf);
376 if (ctx->state != XRD_ST_SENDING) {
377 brix_release_read_buffer(ctx, c, t->databuf);
378 } else {
379 /*
380 * Parked and draining: the buffer is a per-in-flight rd_pool slot and the
381 * header is per-slot, so this memory-backed read is SAFE to pipeline. Note
382 * the cold-AIO path does not yet ACTUALLY pipeline: recv already suspended
383 * on this read's AIO and brix_aio_resume() only drains the write side, so
384 * the next read is not issued until this response drains. Setting the flag
385 * is correct and forward-compatible with the non-suspending read-AIO change
386 * that will let cold reads pipeline like the warm-cache inline path does.
387 * Single-chunk only (non-windowed read <= BRIX_READ_WINDOW < CHUNK_MAX).
388 */
389 ctx->out.resp_pipelinable = 1;
390 }
391 brix_aio_resume(c);
392}

433/*

435 *
436 * pgread wire format ([CRC32C(4)][page data] × N) cannot use
437 * brix_build_chunked_chain and requires direct chain construction with a

460 brix_aio_resume(c);

486 (uint32_t) t->out_size);
487 if (rsp_chain == NULL) {
488 brix_release_read_buffer(ctx, c, t->scratch);
489 brix_aio_resume(c);
490 return;
491 }
492
493 ctx->files[t->handle_idx].bytes_read += (size_t) t->nread;
494 ctx->totals.bytes += (size_t) t->nread;
495
496 rconf = ngx_stream_get_module_srv_conf(
497 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
498 if (rconf->access_log_fd != NGX_INVALID_FILE) {
499 char detail[64];
500 snprintf(detail, sizeof(detail), "%lld+%zu",
501 (long long) t->offset, (size_t) t->nread);
502 brix_log_access(ctx, c, "PGREAD", ctx->files[t->handle_idx].path,
503 detail, 1, 0, NULL, (size_t) t->nread);
504 }
505 BRIX_OP_OK(ctx, BRIX_OP_PGREAD);
506
507 brix_queue_response_chain(ctx, c, rsp_chain, t->scratch);
508 if (ctx->state != XRD_ST_SENDING) {
509 brix_release_read_buffer(ctx, c, t->scratch);
510 }
511 brix_aio_resume(c);
512}
