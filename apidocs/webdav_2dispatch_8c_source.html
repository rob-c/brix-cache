1/*
2 * dispatch.c - WebDAV content handler and method routing.
3 *
4 * WHAT: nginx HTTP content handler — routes authenticated, pre-validated
5 * WebDAV requests to per-method handlers. Authentication, CORS, write
6 * permission and token scope checks are performed upstream in the access phase
7 * handler (access.c); path and lock checks stay in the native method handlers.
8 *
9 * WHY: Keeping all WebDAV methods in this module gives consistent metrics,
10 * logging, lock checks, and path confinement without relying on nginx's
11 * built-in ngx_http_dav_module.
12 *
13 * HOW: For each method, performs any method-specific pre-work not already
14 * done by the access handler (lock check for PUT/MOVE before reading body,
15 * TPC header inspection for COPY) then calls the appropriate handler.
16 * Returns NGX_DECLINED for unrecognised methods so nginx returns 405.
17 */

45{
46 ngx_http_brix_webdav_req_ctx_t *rx;

74static ngx_int_t
75webdav_dispatch_macaroon(ngx_http_request_t *r,
76 ngx_http_brix_webdav_loc_conf_t *conf)
77{
78 static const char discovery_path[] = "/.well-known/oauth-authorization-server";
79 static const char token_path[] = "/.oauth2/token";
80 ngx_int_t rc;
81
82 if (r->uri.len >= sizeof(discovery_path) - 1
83 && ngx_memcmp(r->uri.data + r->uri.len - (sizeof(discovery_path) - 1),
84 discovery_path, sizeof(discovery_path) - 1) == 0)
85 {
86 return webdav_metrics_return(r, webdav_handle_macaroon_discovery(r));
87 }
88
89 if (r->uri.len >= sizeof(token_path) - 1
90 && ngx_memcmp(r->uri.data + r->uri.len - (sizeof(token_path) - 1),
91 token_path, sizeof(token_path) - 1) == 0)
92 {
93 if (r->method != NGX_HTTP_POST) {
94 return webdav_metrics_return(r, NGX_HTTP_NOT_ALLOWED);
95 }
96 rc = brix_http_read_body(r, webdav_handle_macaroon_token);
97 return (rc == NGX_DONE) ? NGX_DONE : webdav_metrics_return(r, rc);
98 }
99
100 if (conf->dig_enable) {
101 rc = brix_dig_handle(r);
102 if (rc != NGX_DECLINED) {
103 return webdav_metrics_return(r, rc);
104 }
105 }
106
107 if (r->method == NGX_HTTP_POST) {
108 ngx_str_t ct = brix_http_get_header(r, "Content-Type");
109 static const char mr[] = "application/macaroon-request";
110 if (ct.len >= sizeof(mr) - 1
111 && ngx_strncasecmp(ct.data, (u_char *) mr, sizeof(mr) - 1) == 0)
112 {
113 rc = brix_http_read_body(r, webdav_handle_macaroon_request);
114 return (rc == NGX_DONE) ? NGX_DONE : webdav_metrics_return(r, rc);
115 }
116 }
117
118 return NGX_DECLINED;

392/*
393 * Phase 40 wrapper: bracket the whole synchronous dispatch with the impersonation
394 * principal taken from the authenticated identity (no-op unless map mode). This
395 * covers every method that completes inline (GET/HEAD/DELETE/MKCOL/COPY/MOVE/
396 * PROPFIND/...). Methods that read a body asynchronously (PUT/LOCK) return
397 * NGX_DONE here — the principal is cleared on return and re-established inside the
398 * body callback (see webdav_handle_put_body), so it never leaks across the event
399 * loop while a body is pending.
400 */
401ngx_int_t
402ngx_http_brix_webdav_handler(ngx_http_request_t *r)
403{
404 ngx_http_brix_webdav_req_ctx_t *rx =
405 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
406 ngx_int_t rc;
407
408 brix_imp_request_begin(rx != NULL ? rx->identity : NULL);
409 rc = webdav_dispatch_inner(r);
410 brix_imp_request_end();
411 return rc;
412}

414static ngx_int_t
415webdav_dispatch_inner(ngx_http_request_t *r)
416{
417 ngx_http_brix_webdav_loc_conf_t *conf;
418 ngx_int_t rc;
419
420 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
421 if (!conf->common.enable) {
422 return NGX_DECLINED;
423 }
424
425 /*
426 * Fast teardown: the worker is draining (graceful reload/quit). Reject a
427 * freshly-arrived request with 503 + Retry-After and force the connection
428 * closed, rather than starting a transfer this worker is about to abandon.
429 * The client's resilient layer retries against the new worker and resumes
430 * a download from its last byte via a Range GET. Requests already mid-flight
431 * are untouched (their handler has long since returned); ngx_close_idle_
432 * connections() drops idle keepalive connections for us.
433 */
434 if (ngx_exiting) {
435 return webdav_dispatch_draining(r);
436 }
437
438 /* AGPL-3.0 sec.13: offer remote users the source (X-Source header). */
439 brix_http_source_offer(r);
440 webdav_sess_begin_request(r);
441 webdav_sess_attempt_request(r);
442
443 /*
444 * SciTags packet marking (phase-34). TPC (COPY) is always marked; plain
445 * GET/PUT only when brix_pmark_http_plain is on (default off = XRootD
446 * parity). Begun here post-auth; ended via a request-pool cleanup.
447 */
448 webdav_dispatch_pmark(r, conf);
449
450 /* OPTIONS: access handler already added CORS headers and tracked the
451 * pre-flight metric; content handler owns the Allow response. */
452 if (r->method == NGX_HTTP_OPTIONS) {
453 return webdav_metrics_return(r, webdav_handle_options(r));
454 }
455
456 /* Macaroon token issuance endpoints — handled before proxy mode because
457 * the server issues tokens for itself, not the upstream. */
458 rc = webdav_dispatch_macaroon(r, conf);
459 if (rc != NGX_DECLINED) {
460 return rc;
461 }
462
463 /* Phase-2 Task 8: opt-in proxy-upload delegation endpoint. Handled before
464 * proxy mode because the server stores the credential itself, not the
465 * upstream. Path match mirrors the macaroon endpoints above: suffix
466 * match against a well-known path, off by default (falls through to
467 * normal dispatch → 404, matching the "endpoint off = not special"
468 * contract). */
469 rc = webdav_dispatch_delegation_upload(r, conf);
470 if (rc != NGX_DECLINED) {
471 return rc;
472 }
473
474 /* Phase-3 Task 4: standard GridSite two-step getProxyReq/putProxy REST
475 * flow, alongside the T8 proxy-upload form above. Same
476 * brix_delegation_endpoint gate (off by default = both paths fall
477 * through to ordinary dispatch, same "off = not special" contract as
478 * T8). Two routes under the SAME well-known path as T8 but with a
479 * trailing segment, so they are matched separately from T8's exact
480 * suffix match (which requires no trailing "/…" after
481 * "/.well-known/brix-delegation"):
482 * GET /.well-known/brix-delegation/request — getProxyReq (step 1)
483 * PUT /.well-known/brix-delegation/<id> — putProxy (step 2)
484 * request_handle is synchronous (no body to read — GET carries none)
485 * and follows the ordinary synchronous-handler contract (like
486 * webdav_handle_options): called directly and its return value passed
487 * through webdav_metrics_return(), NOT NGX_DONE — it does not read a
488 * body asynchronously, so there is no pending async work for NGX_DONE
489 * to represent; returning NGX_DONE here would have nginx's content
490 * phase finalize the request a second time (see delegation.h's doc
491 * comment on webdav_delegation_request_handle). */
492 rc = webdav_dispatch_delegation_gridsite(r, conf);
493 if (rc != NGX_DECLINED) {
494 return rc;
495 }
496
497 /* WLCG HTTP Tape REST API (/api/v1/…) — handled before proxy mode because
498 * the server answers it locally, not the upstream. POST dispatches async
499 * body reading (NGX_DONE); GET/DELETE return inline. */
500 rc = webdav_dispatch_tape_rest(r, conf);
501 if (rc != NGX_DECLINED) {
502 return rc;
503 }
504
505 rc = webdav_dispatch_method(r, conf);
506 if (rc != NGX_DECLINED) {
507 return rc;
508 }
509
510 /* Unrecognised method: send 405 with Allow header (RFC 7231 §6.5.5). */
511 return webdav_dispatch_not_allowed(r, conf);
512}
