1/*
2 * cli_cksum.c — shared front-end for the local-or-root:// checksum CLIs.
3 *
4 * WHAT: brix_cli_cksum_main() is the whole body of the xrdcrc32c / xrdcrc64 /
5 * xrdadler32 tools: given a program name, a wire algorithm name and the
6 * matching local-digest enum, it checksums either a LOCAL file (streamed
7 * through brix_cksum_fd) or a root:// file (kXR_query/kXR_Qcksum via
8 * brix_query_cksum) and prints "<hex> <path>".
9 * WHY: The three checksum tools were byte-identical except for one algorithm
10 * string + enum; this collapses them to a ~6-line main() each, so a new
11 * algorithm tool is a one-liner and the dispatch logic lives in one place.
12 * HOW: strstr("://") selects the root:// path (parse URL, require root/roots,
13 * connect, query) vs the local path (open O_RDONLY, brix_cksum_fd). Exit
14 * codes mirror the stock tools byte-for-byte: argument/URL-parse errors are
15 * XRDC_EXIT_USAGE; any failure to PRODUCE a checksum (connect, query, local
16 * open, or digest) returns the caller's `err_exit` (xrdadler32 → 1,
17 * xrdcrc32c → 3, xrdcrc64 → 1) rather than the XrdCl-style shellcode, which
18 * diverged from stock (54=NotFound) and tripped scripts that branch on $?.
19 */

30brix_cli_cksum_main(const char *prog, const char *algo_name,
31 brix_cksum_algo algo, const char *arg, int err_exit)
32{
33 brix_status st;
34 char hex[129];
35
36 /* --help / --version before using arg as a path. */
37 if (arg != NULL) {
38 if (strcmp(arg, "--version") == 0) {
39 printf("%s (BriX-Cache client) %s\n", prog, brix_client_version());
40 return 0;
41 }
42 if (strcmp(arg, "--help") == 0 || strcmp(arg, "-h") == 0) {
43 printf("usage: %s <local-path | root://host[:port]//path>\n"
44 BRIX_USAGE_FOOTER("%s"),
45 prog, prog);
46 return 0;
47 }
48 }
49
50 if (arg == NULL) {
51 fprintf(stderr, "usage: %s <local-path | root://host[:port]//path>\n",
52 prog);
53 return XRDC_EXIT_USAGE;
54 }
55 brix_crypto_init();
56 brix_status_clear(&st);
57
58 if (strstr(arg, "://") != NULL) {
59 brix_url u;
60 brix_conn c;
61
62 if (brix_url_parse(arg, &u, &st) != 0
63 || (u.scheme != XRDC_SCHEME_ROOT && u.scheme != XRDC_SCHEME_ROOTS)) {
64 fprintf(stderr, "%s: %s\n", prog, st.msg);
65 return XRDC_EXIT_USAGE;
66 }
67 if (brix_connect(&c, &u, NULL, &st) != 0) {
68 fprintf(stderr, "%s: connect: %s\n", prog, st.msg);
69 return err_exit;
70 }
71 if (brix_query_cksum(&c, u.path, algo_name, hex, sizeof(hex), &st) != 0) {
72 fprintf(stderr, "%s: %s\n", prog, st.msg);
73 brix_close(&c);
74 return err_exit;
75 }
76 brix_close(&c);
77 printf("%s %s\n", hex, u.path);
78 return 0;
79 }
80
81 {
82 int fd = open(arg, O_RDONLY);
83 if (fd < 0) {
84 fprintf(stderr, "%s: open %s failed\n", prog, arg);
85 return err_exit;
86 }
87 if (brix_cksum_fd(fd, algo, hex, sizeof(hex), &st) != 0) {
88 fprintf(stderr, "%s: %s\n", prog, st.msg);
89 close(fd);
90 return err_exit;
91 }
92 close(fd);
93 printf("%s %s\n", hex, arg);
94 return 0;
95 }
96}
