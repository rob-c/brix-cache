1/*
2 * tpc_thread.c - thread-pool wrapper for HTTP-TPC blocking curl transfers.
3 *
4 * Moves the fork/waitpid blocking from the nginx event-loop worker to a
5 * background thread so that other connections on the same worker are not
6 * stalled for the duration of a large file transfer (which can be minutes).
7 *
8 * Compiled only when NGX_THREADS is defined (--with-threads configure flag).
9 * When thread pool is not available at runtime (conf->common.thread_pool == NULL),
10 * webdav_tpc_post_thread_task returns NGX_DECLINED and the caller falls back
11 * to the synchronous path in tpc.c.
12 */

46/*
47 * Add this transfer to the cross-process TPC registry, mapping src/dst by
48 * direction: push reads local_path and writes the remote `url`; pull reads `url`
49 * and writes dest_path. Returns the non-zero transfer id, or 0 if the registry
50 * is full.
51 */
52static uint64_t
53webdav_tpc_thread_register(ngx_http_request_t *r, int is_push,
54 const char *url, const char *local_path,
55 const char *dest_path)
56{
57 brix_tpc_transfer_t transfer;
58 ngx_str_t src;
59 ngx_str_t dst;
60
61 ngx_memzero(&transfer, sizeof(transfer));
62
63 if (is_push) {
64 src.data = (u_char *) local_path;
65 src.len = ngx_strlen(local_path);
66 dst.data = (u_char *) url;
67 dst.len = ngx_strlen(url);
68 transfer.direction = BRIX_TPC_DIR_PUSH;
69 } else {
70 src.data = (u_char *) url;
71 src.len = ngx_strlen(url);
72 dst.data = (u_char *) dest_path;
73 dst.len = ngx_strlen(dest_path);
74 transfer.direction = BRIX_TPC_DIR_PULL;
75 }
76
77 transfer.protocol = BRIX_TPC_PROTO_WEBDAV;
78 transfer.src_url = src;
79 transfer.dst_path = dst;
80 transfer.state = BRIX_TPC_STATE_PENDING;
81
82 return brix_tpc_registry_add(&transfer, r->connection->log);
83}

85/*
86 * WHAT: mark this transfer failed in the registry and bump the direction's
87 * TPC error metric, using the bytes recorded so far.
88 * WHY: every worker-body failure exit performs this identical two-call
89 * bookkeeping; centralising it keeps the ordering (registry then metric)
90 * and the counter values consistent across all exit paths.
91 * HOW: reads t->transfer_id / t->bytes_transferred / t->is_push and issues the
92 * registry ERROR update followed by the ERROR metric. Thread-safe: both
93 * callees are thread-safe and no shared state is added.
94 */
95static void
96tpc_thread_fail(tpc_thread_ctx_t *t)
97{
98 (void) brix_tpc_registry_update(t->transfer_id, t->bytes_transferred,
99 BRIX_TPC_STATE_ERROR, t->log);
100 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV,
101 t->is_push ? BRIX_TPC_DIR_PUSH
102 : BRIX_TPC_DIR_PULL,
103 BRIX_TPC_METRIC_ERROR,
104 t->bytes_transferred, t->log);
105}

107/*
108 * WHAT: record the on-disk size of the local transfer file into
109 * t->bytes_transferred, leaving it unchanged if the stat fails.
110 * WHY: both the push success path and the pull post-curl path report the
111 * committed byte count from the temp/source file size.
112 * HOW: best-effort stat() of t->local_path (raw is correct here — the temp
113 * lives in the svc-owned staging domain, not an export path).
114 */
115static void
116tpc_thread_record_size(tpc_thread_ctx_t *t)
117{
118 struct stat st;
119
120 if (stat(t->local_path, &st) == 0) { /* vfs-seam-allow: TPC local transfer temp */
121 t->bytes_transferred = st.st_size;
122 }
123}

125/*
126 * WHAT: SSRF preflight for the remote URL; returns NGX_OK to proceed or
127 * NGX_ERROR after fully finalising the failure (status/registry/metric).
128 * WHY: getaddrinfo() blocks, so the resolve-and-vet must run on the worker
129 * thread before curl forks; a blocked address must be rejected exactly
130 * once with a 403 and error bookkeeping.
131 * HOW: parses the URL under the conf-derived policy, checks DNS, and on
132 * failure sets t->http_status=403 and calls tpc_thread_fail(). A NULL
133 * URL is treated as "nothing to vet" and returns NGX_OK.
134 */
135static ngx_int_t
136tpc_thread_ssrf_preflight(tpc_thread_ctx_t *t)
137{
138 brix_net_target_t net_target;
139 brix_net_target_policy_t net_policy;
140 ngx_str_t url_str;
141 char ssrf_err[256];
142
143 if (t->url == NULL) {
144 return NGX_OK;
145 }
146
147 url_str.data = (u_char *) t->url;
148 url_str.len = ngx_strlen(t->url);
149
150 ngx_memzero(&net_policy, sizeof(net_policy));
151 net_policy.require_https = 1;
152 net_policy.allow_root_scheme = 0;
153 net_policy.allow_local = t->conf->tpc_allow_local;
154 net_policy.allow_private = t->conf->tpc_allow_private;
155 net_policy.default_https_port = 443;
156
157 if (brix_net_target_parse(NULL, &url_str, &net_target,
158 ssrf_err, sizeof(ssrf_err)) == NGX_OK
159 && brix_net_target_check_dns(&net_target, &net_policy,
160 ssrf_err, sizeof(ssrf_err)) == NGX_OK)
161 {
162 return NGX_OK;
163 }
164
165 ngx_log_error(NGX_LOG_WARN, t->log, 0,
166 "brix_webdav: HTTP-TPC SSRF check blocked "
167 "transfer to %s: %s",
168 t->url, ssrf_err);
169 t->http_status = NGX_HTTP_FORBIDDEN;
170 tpc_thread_fail(t);
171 return NGX_ERROR;
172}

174/*
175 * WHAT: run the push (local → remote) curl transfer and finalise its outcome.
176 * WHY: the push direction has its own success/error bookkeeping (byte count
177 * from the source file, PUSH_SUCCESS webdav metric, 201 status) distinct
178 * from pull; isolating it keeps the worker body flat.
179 * HOW: invokes webdav_tpc_run_curl_push; on NGX_OK records size, marks the
180 * registry DONE, bumps success metrics and sets 201; otherwise
181 * tpc_thread_fail() and propagates the curl status. Preserves the exact
182 * marker/counter update order of the original inline branch.
183 */
184static void
185tpc_thread_run_push(tpc_thread_ctx_t *t)
186{
187 ngx_int_t rc;
188
189 rc = webdav_tpc_run_curl_push(t->log, t->conf, t->url,
190 t->local_path, t->transfer_headers,
191 t->transfer_id);
192 if (rc != NGX_OK) {
193 tpc_thread_fail(t);
194 t->http_status = rc;
195 return;
196 }
197
198 tpc_thread_record_size(t);
199 (void) brix_tpc_registry_update(t->transfer_id, t->bytes_transferred,
200 BRIX_TPC_STATE_DONE, t->log);
201 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV, BRIX_TPC_DIR_PUSH,
202 BRIX_TPC_METRIC_SUCCESS,
203 t->bytes_transferred, t->log);
204 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_PUSH_SUCCESS]);
205 t->http_status = NGX_HTTP_CREATED;
206}

208/*
209 * WHAT: run the pull (remote → local temp) curl transfer, single- or
210 * multi-stream; returns NGX_OK if bytes landed in the temp, else the
211 * failure status.
212 * WHY: the pull byte fetch is a distinct stage from the atomic commit; on
213 * failure the temp must be removed and the transfer marked errored before
214 * the commit stage is ever reached.
215 * HOW: dispatches to the multi- or single-stream curl runner by t->n_streams;
216 * on error unlinks the confined temp, calls tpc_thread_fail(), sets
217 * t->http_status and returns NGX_ERROR. On success records the temp size
218 * and returns NGX_OK.
219 */
220static ngx_int_t
221tpc_thread_pull_fetch(tpc_thread_ctx_t *t)
222{
223 ngx_int_t rc;
224
225 if (t->n_streams > 1) {
226 rc = webdav_tpc_run_curl_pull_multi(t->log, t->conf, t->url,
227 t->local_path, t->transfer_headers,
228 t->n_streams, t->transfer_id, NULL,
229 t->user_cert, t->user_key);
230 } else {
231 rc = webdav_tpc_run_curl_pull(t->log, t->conf, t->url, t->local_path,
232 t->transfer_headers, t->transfer_id,
233 t->user_cert, t->user_key);
234 }
235
236 if (rc != NGX_OK) {
237 (void) brix_unlink_confined_canon(t->log, t->root_canon,
238 t->local_path, 0);
239 tpc_thread_fail(t);
240 t->http_status = rc;
241 return NGX_ERROR;
242 }
243
244 tpc_thread_record_size(t);
245 return NGX_OK;
246}

248/*
249 * WHAT: atomically commit the pull temp into dest_path; returns NGX_OK on
250 * success (with the final 201/204 status set) or NGX_ERROR after full
251 * failure bookkeeping.
252 * WHY: mirrors the synchronous path's commit policy — no-overwrite uses link()
253 * (EEXIST ⇒ 412 racing create), overwrite uses rename(); every failure
254 * unlinks the temp so no partial file is left and bumps COMMIT_ERROR.
255 * HOW: branches on t->overwrite, always unlinks the temp afterwards, and on
256 * any failure marks the commit-error metric, calls tpc_thread_fail() and
257 * sets the appropriate 4xx/5xx status. On success sets 204 if the dest
258 * pre-existed else 201.
259 */
260static ngx_int_t
261tpc_thread_pull_commit(tpc_thread_ctx_t *t)
262{
263 int err;
264
265 if (!t->overwrite) {
266 if (brix_link_confined_canon(t->log, t->root_canon,
267 t->local_path, t->dest_path) != 0) {
268 err = errno;
269 (void) brix_unlink_confined_canon(t->log, t->root_canon,
270 t->local_path, 0);
271 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_COMMIT_ERROR]);
272 tpc_thread_fail(t);
273 t->http_status = (err == EEXIST) ? NGX_HTTP_PRECONDITION_FAILED
274 : NGX_HTTP_INTERNAL_SERVER_ERROR;
275 return NGX_ERROR;
276 }
277 (void) brix_unlink_confined_canon(t->log, t->root_canon,
278 t->local_path, 0);
279 } else if (brix_rename_confined_canon(t->log, t->root_canon,
280 t->local_path, t->dest_path) != 0) {
281 brix_log_safe_path(t->log, NGX_LOG_ERR, ngx_errno,
282 "brix_webdav: HTTP-TPC rename failed for: \"%s\"",
283 t->dest_path);
284 (void) brix_unlink_confined_canon(t->log, t->root_canon,
285 t->local_path, 0);
286 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_COMMIT_ERROR]);
287 tpc_thread_fail(t);
288 t->http_status = NGX_HTTP_INTERNAL_SERVER_ERROR;
289 return NGX_ERROR;
290 }
291
292 t->http_status = t->existed ? NGX_HTTP_NO_CONTENT : NGX_HTTP_CREATED;
293 return NGX_OK;
294}

296/*
297 * WHAT: run the full pull direction — fetch the remote bytes then atomically
298 * commit the temp into place, reporting success bookkeeping.
299 * WHY: keeps the two pull stages (fetch, commit) sequenced with a single
300 * success finaliser, matching the original inline flow.
301 * HOW: early-returns if either stage fails (each already finalised its own
302 * error); on full success marks the registry DONE, bumps the PULL_SUCCESS
303 * metrics. The status was set by the commit stage.
304 */
305static void
306tpc_thread_run_pull(tpc_thread_ctx_t *t)
307{
308 if (tpc_thread_pull_fetch(t) != NGX_OK) {
309 return;
310 }
311 if (tpc_thread_pull_commit(t) != NGX_OK) {
312 return;
313 }
314
315 (void) brix_tpc_registry_update(t->transfer_id, t->bytes_transferred,
316 BRIX_TPC_STATE_DONE, t->log);
317 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV, BRIX_TPC_DIR_PULL,
318 BRIX_TPC_METRIC_SUCCESS,
319 t->bytes_transferred, t->log);
320 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_PULL_SUCCESS]);
321}

323/*
324 * Thread-pool worker body. RUNS ON A BACKGROUND THREAD, not the nginx event
325 * loop: it MUST NOT touch the event loop, send a response, or call most ngx_http
326 * APIs. Its only outputs are the t->* result fields (http_status,
327 * bytes_transferred) plus the side-effecting registry/metrics/filesystem calls,
328 * which are thread-safe. tpc_thread_done (below) consumes the results back on
329 * the event loop. Performs the SSRF preflight, marks the transfer active, and
330 * dispatches to the push or pull stage (pull atomically commits the temp file).
331 */
332static void
333tpc_thread_func(void *data, ngx_log_t *log)
334{
335 tpc_thread_ctx_t *t = data;
336
337 (void) log; /* use t->log for request-level context */
338
339 if (tpc_thread_ssrf_preflight(t) != NGX_OK) {
340 return;
341 }
342
343 (void) brix_tpc_registry_update(t->transfer_id, 0,
344 BRIX_TPC_STATE_ACTIVE, t->log);
345
346 if (t->is_push) {
347 tpc_thread_run_push(t);
348 } else {
349 tpc_thread_run_pull(t);
350 }
351}

353/*
354 * Completion handler — RUNS BACK ON THE EVENT LOOP after tpc_thread_func
355 * finishes. This is the async re-entry point: it reads the result fields the
356 * worker set, removes the registry entry, and sends the final HTTP response
357 * (201/204 on success, the error status otherwise), then finalizes the request.
358 * Pairs with the r->main->count++ taken in webdav_tpc_post_thread_task.
359 */
360static void
361tpc_thread_done(ngx_event_t *ev)
362{
363 ngx_thread_task_t *task = ev->data;
364 tpc_thread_ctx_t *t = task->ctx;
365 ngx_http_request_t *r = t->r;
366 ngx_int_t status = t->http_status;
367
368 (void) brix_tpc_registry_remove(t->transfer_id,
369 r->connection != NULL
370 ? r->connection->log : t->log);
371
372 /* Unified transfer ledger: one audit line shared with STAGE/TAPE/WT. A push
373 * sends a local object OUT to a remote dest; a pull brings data IN to the
374 * local dest (for a pull, local_path is the staging temp, so the audited
375 * object is dest_path). */
376 {
377 int ok = (status == NGX_HTTP_CREATED
378 || status == NGX_HTTP_NO_CONTENT);
379 const char *path = t->is_push
380 ? (t->local_path[0] ? t->local_path : t->url)
381 : (t->dest_path[0] ? t->dest_path : t->url);
382 brix_xfer_finish(BRIX_XFER_TPC, t->is_push ? "out" : "in",
383 path, NULL, (size_t) t->bytes_transferred,
384 ok ? BRIX_XFER_OK
385 : (t->is_push ? BRIX_XFER_DST_ERR
386 : BRIX_XFER_SRC_ERR),
387 0, t->log);
388 }
389
390 if (status == NGX_HTTP_CREATED || status == NGX_HTTP_NO_CONTENT) {
391 if (t->bytes_transferred > 0) {
392 brix_dashboard_http_add(r,
393 (ngx_atomic_int_t) t->bytes_transferred);
394 }
395 brix_dashboard_http_finish(r);
396 webdav_send_status_only(r, (ngx_uint_t) status);
397 } else {
398 brix_dashboard_http_error(r, "webdav TPC failed");
399 brix_dashboard_http_finish(r);
400 webdav_metrics_finalize_request(r, status);
401 }
402}

426/*
427 * webdav_tpc_post_thread_task — post pull or push curl transfer to thread pool.
428 *
429 * Returns NGX_DONE on success (caller must return NGX_DONE to nginx).
430 * Returns NGX_DECLINED when conf->common.thread_pool is NULL (caller uses sync path).
431 * Returns NGX_ERROR on allocation or post failure.
432 *
433 * local_path and dest_path are copied into the task context so stack-allocated
434 * caller buffers are safe to reuse after this call returns.
435 */
436ngx_int_t
437webdav_tpc_post_thread_task(ngx_http_request_t *r,
438 ngx_http_brix_webdav_loc_conf_t *conf,
439 int is_push, ngx_flag_t existed, ngx_flag_t overwrite,
440 const char *url, const char *local_path,
441 const char *dest_path,
442 ngx_array_t *transfer_headers,
443 ngx_uint_t n_streams,
444 const char *user_cert, const char *user_key)
445{
446 ngx_thread_task_t *task;
447 tpc_thread_ctx_t *t;
448
449 if (conf->common.thread_pool == NULL) {
450 return NGX_DECLINED;
451 }
452
453 if (url == NULL || local_path == NULL || (!is_push && dest_path == NULL)) {
454 return NGX_ERROR;
455 }
456
457 task = ngx_thread_task_alloc(r->pool, sizeof(tpc_thread_ctx_t));
458 if (task == NULL) {
459 return NGX_ERROR;
460 }
461
462 t = task->ctx;
463 t->r = r;
464 t->conf = conf;
465 t->log = r->connection->log;
466 t->is_push = is_push;
467 t->existed = existed;
468 t->overwrite = overwrite;
469 t->url = url;
470 t->transfer_headers = transfer_headers;
471 t->user_cert = user_cert;
472 t->user_key = user_key;
473 t->n_streams = (is_push || n_streams < 1) ? 1 : n_streams;
474 t->transfer_id = webdav_tpc_thread_register(r, is_push, url, local_path,
475 dest_path);
476 if (t->transfer_id == 0) {
477 return NGX_HTTP_SERVICE_UNAVAILABLE;
478 }
479
480 tpc_thread_ctx_copy_paths(t, conf, local_path, dest_path);
481
482 brix_task_bind(task, tpc_thread_func, tpc_thread_done);
483 task->event.log = r->connection->log;
484
485 if (ngx_thread_task_post(conf->common.thread_pool, task) != NGX_OK) {
486 (void) brix_tpc_registry_remove(t->transfer_id,
487 r->connection->log);
488 return NGX_ERROR;
489 }
490
491 /* Hold a reference on the request so nginx does not destroy it while the
492 * transfer runs on the background thread; tpc_thread_done finalizes it,
493 * releasing this count. Returning NGX_DONE keeps the request alive. */
494 r->main->count++;
495 return NGX_DONE;
496}
