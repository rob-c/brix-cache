1/*
2 * introspect.c — Phase 21 Step C: real-time OIDC token revocation via a
3 * non-blocking introspection subrequest.
4 *
5 * Design (modelled on nginx's auth_request module):
6 *
7 * - A dedicated NGX_HTTP_ACCESS_PHASE handler, separate from the main webdav
8 * access handler, so the subrequest suspend/resume re-entry replays ONLY
9 * this handler (the main auth handler is not re-run, and its metrics are
10 * not double-counted).
11 *
12 * - The operator defines an internal location that POSTs to the IdP's
13 * RFC 7662 /introspect endpoint, e.g.:
14 *
15 * location = /internal/introspect {
16 * internal;
17 * proxy_method POST;
18 * proxy_set_header Content-Type application/x-www-form-urlencoded;
19 * proxy_set_body "token=$arg_token";
20 * proxy_pass https://iam.example.org/introspect;
21 * }
22 *
23 * and names it with `brix_webdav_token_introspect_loc /internal/introspect;`.
24 * We fire a subrequest at that URI carrying `?token=<jwt>` (JWTs are
25 * base64url, so they are URL-safe as a query value).
26 *
27 * - Fast path: the token fingerprint (SHA-256) is checked against a Phase-20
28 * KV revoke zone first; a cached "revoked" marker short-circuits to 403
29 * without an IdP round-trip. Only negative results are cached (for
30 * introspect_ttl seconds) so revocation propagates quickly while valid
31 * tokens are not re-introspected on every request (the Phase-20 token
32 * validation cache already amortises those).
33 *
34 * - Failure policy: if the IdP is unreachable or returns a non-200, the
35 * request proceeds when fail_open is on (WLCG default) or is rejected with
36 * 403 when off.
37 */

52/*
53 * Parse an RFC 7662 introspection response body. Returns 1 when the JSON
54 * object has "active": true, 0 otherwise (including parse failure — a malformed
55 * response is treated as "not active" only by the caller's policy, never here).
56 */
57static int
58webdav_introspect_parse_active(const u_char *body, size_t len)
59{
60 json_error_t jerr;
61 json_t *root, *active;
62 char *buf;
63 int result = 0;
64
65 if (body == NULL || len == 0) {
66 return 0;
67 }
68 buf = ngx_alloc(len + 1, ngx_cycle->log);
69 if (buf == NULL) {
70 return 0;
71 }
72 ngx_memcpy(buf, body, len);
73 buf[len] = '\0';
74
75 root = json_loads(buf, JSON_REJECT_DUPLICATES, &jerr);
76 ngx_free(buf);
77 if (root == NULL) {
78 return 0;
79 }
80
81 active = json_object_get(root, "active");
82 if (json_is_boolean(active) && json_is_true(active)) {
83 result = 1;
84 }
85 json_decref(root);
86 return result;
87}

89/* Subrequest completion: record active/revoked verdict on the parent ctx. */
90static ngx_int_t
91webdav_introspect_done(ngx_http_request_t *r, void *data, ngx_int_t rc)
92{
93 webdav_introspect_data_t *d = data;
94 ngx_http_brix_webdav_req_ctx_t *pctx = d->parent_ctx;
95 ngx_http_brix_webdav_loc_conf_t *conf = d->conf;
96 ngx_uint_t status;
97 int active;
98
99 status = r->headers_out.status;
100
101 if (rc == NGX_OK && status == NGX_HTTP_OK && r->upstream != NULL) {
102 ngx_buf_t *b = &r->upstream->buffer;
103 active = webdav_introspect_parse_active(b->pos,
104 (size_t) (b->last - b->pos));
105
106 /* Cache only negative (revoked) results so revocation propagates. */
107 if (!active && conf->revoke_kv != NULL && d->have_key) {
108 (void) brix_kv_set(conf->revoke_kv, d->key, 32,
109 (const u_char *) "1", 1,
110 (ngx_msec_t) conf->introspect_ttl * 1000);
111 }
112 } else {
113 /* IdP unreachable / error → honour the configured failure policy. */
114 active = conf->introspect_fail_open ? 1 : 0;
115 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
116 "brix_webdav: token introspection failed "
117 "(rc=%i status=%ui) — %s",
118 rc, status,
119 conf->introspect_fail_open ? "allowing (fail-open)"
120 : "denying (fail-closed)");
121 }
122
123 pctx->introspect_active = active ? 1 : 0;
124 pctx->introspect_done = 1;
125 return rc;
126}

128static ngx_int_t
129webdav_introspect_fire(ngx_http_request_t *r,
130 ngx_http_brix_webdav_req_ctx_t *ctx,
131 ngx_http_brix_webdav_loc_conf_t *conf,
132 const ngx_str_t *token, const u_char key[32], int have_key)
133{
134 ngx_http_request_t *sr;
135 ngx_http_post_subrequest_t *ps;
136 webdav_introspect_data_t *d;
137 ngx_str_t args;
138 u_char *p;
139
140 d = ngx_pcalloc(r->pool, sizeof(*d));
141 ps = ngx_palloc(r->pool, sizeof(*ps));
142 if (d == NULL || ps == NULL) {
143 return NGX_ERROR;
144 }
145 d->parent_ctx = ctx;
146 d->conf = conf;
147 d->have_key = have_key ? 1 : 0;
148 if (have_key) {
149 ngx_memcpy(d->key, key, 32);
150 }
151 ps->handler = webdav_introspect_done;
152 ps->data = d;
153
154 /* args = "token=<jwt>" (JWT is base64url → URL-safe as a query value). */
155 args.len = sizeof("token=") - 1 + token->len;
156 args.data = ngx_pnalloc(r->pool, args.len);
157 if (args.data == NULL) {
158 return NGX_ERROR;
159 }
160 p = ngx_cpymem(args.data, "token=", sizeof("token=") - 1);
161 ngx_memcpy(p, token->data, token->len);
162
163 if (ngx_http_subrequest(r, &conf->introspect_loc, &args, &sr, ps,
164 NGX_HTTP_SUBREQUEST_WAITED
165 | NGX_HTTP_SUBREQUEST_IN_MEMORY) != NGX_OK)
166 {
167 return NGX_ERROR;
168 }
169
170 return NGX_OK;
171}

173ngx_int_t
174webdav_introspect_access_handler(ngx_http_request_t *r)
175{
176 ngx_http_brix_webdav_loc_conf_t *conf;
177 ngx_http_brix_webdav_req_ctx_t *ctx;
178 ngx_str_t auth, bearer;
179 u_char key[32];
180 int have_key = 0;
181
182 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
183 if (conf == NULL || conf->introspect_loc.len == 0) {
184 return NGX_DECLINED; /* introspection not configured */
185 }
186
187 /* Only main requests; never introspect our own subrequest. */
188 if (r != r->main) {
189 return NGX_DECLINED;
190 }
191
192 /* Needs a Bearer token to introspect. */
193 if (r->headers_in.authorization == NULL) {
194 return NGX_DECLINED;
195 }
196 auth = r->headers_in.authorization->value;
197 if (brix_http_extract_bearer(&auth, &bearer) != NGX_OK
198 || bearer.len == 0)
199 {
200 return NGX_DECLINED;
201 }
202
203 /* This access handler may run before the main webdav access handler that
204 * normally allocates the request context, so get-or-create it here. The
205 * struct layout matches xrdhttp_get_ctx(), so a later call there reuses it. */
206 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
207 if (ctx == NULL) {
208 BRIX_PCALLOC_OR_RETURN(ctx, r->pool, sizeof(*ctx), NGX_ERROR);
209 ngx_http_set_ctx(r, ctx, ngx_http_brix_webdav_module);
210 }
211
212 /* Resume path: the subrequest already produced a verdict. */
213 if (ctx->introspect_done) {
214 if (!ctx->introspect_active) {
215 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
216 "brix_webdav: bearer token rejected by"
217 " introspection (revoked)");
218 return NGX_HTTP_FORBIDDEN;
219 }
220 return NGX_DECLINED; /* active → allow, continue */
221 }
222
223 /* Fast path: token fingerprint already known-revoked. */
224 if (brix_sha256(bearer.data, bearer.len, key) == 1) {
225 have_key = 1;
226 if (conf->revoke_kv != NULL) {
227 u_char v[4];
228 size_t vl = sizeof(v);
229 if (brix_kv_get(conf->revoke_kv, key, 32, v, &vl) == 1) {
230 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
231 "brix_webdav: bearer token rejected"
232 " (revocation cache hit)");
233 return NGX_HTTP_FORBIDDEN;
234 }
235 }
236 }
237
238 /* Slow path: fire the introspection subrequest and suspend. */
239 if (webdav_introspect_fire(r, ctx, conf, &bearer, key, have_key)
240 != NGX_OK)
241 {
242 return conf->introspect_fail_open ? NGX_DECLINED : NGX_HTTP_FORBIDDEN;
243 }
244
245 return NGX_AGAIN; /* suspend until subrequest completes */
246}

249char *
250webdav_conf_revoke_cache(ngx_conf_t *cf, ngx_command_t *cmd, void *conf_ptr)
251{
252 ngx_http_brix_webdav_loc_conf_t *conf = conf_ptr;
253 ngx_str_t *value = cf->args->elts;
254 ngx_str_t zone = ngx_null_string;
255 ngx_uint_t i;
256 brix_kv_t *kv;
257
258 (void) cmd;
259
260 for (i = 1; i < cf->args->nelts; i++) {
261 if (ngx_strncmp(value[i].data, "zone=", 5) == 0) {
262 zone.data = value[i].data + 5;
263 zone.len = value[i].len - 5;
264 } else {
265 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
266 "invalid brix_webdav_revoke_cache parameter \"%V\"",
267 &value[i]);
268 return NGX_CONF_ERROR;
269 }
270 }
271 if (zone.len == 0) {
272 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
273 "brix_webdav_revoke_cache requires zone=<name>");
274 return NGX_CONF_ERROR;
275 }
276
277 kv = brix_kv_find(&zone);
278 if (kv == NULL) {
279 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
280 "brix_webdav_revoke_cache: unknown zone \"%V\" "
281 "(declare it with brix_kv_zone first)", &zone);
282 return NGX_CONF_ERROR;
283 }
284 if (kv->key_max < 32 || kv->val_max < 1) {
285 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
286 "brix_webdav_revoke_cache zone \"%V\" too small: need key>=32 val>=1",
287 &zone);
288 return NGX_CONF_ERROR;
289 }
290
291 conf->revoke_kv = kv;
292 return NGX_CONF_OK;
293}
