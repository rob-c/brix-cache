16/*
17 * WHAT: This file provides shared cryptographic helpers for Simple Shared Secret (SSS) authentication.
18 * Includes big-endian 32/64-bit read/write primitives for wire protocol framing, software CRC32c implementation
19 * for integrity verification (SSS uses CRC32 not HMAC), Blowfish-CFB encryption/decryption for challenge-response,
20 * and OpenSSL 3.x legacy provider loading for SHA-256 digest availability. All helpers are static or internal —
21 * used exclusively by sss/auth.c and sss/key_parse.c during SSS authentication flow.
22 *
23 * WHY: SSS authentication requires wire-format parsing (big-endian integers), integrity verification via CRC32 appended
24 * to plaintext before encryption, symmetric challenge encryption using Blowfish-CFB with zero IV, and OpenSSL 3.x
25 * compatibility for SHA-256 digest fetching from legacy provider. These helpers centralize crypto operations preventing
26 * duplication across SSS module files and ensuring consistent wire-format handling.
27 *
28 * HOW: Four helper categories → big-endian read/write (brix_sss_read_be32/be64, brix_sss_write_be32) for wire protocol parsing;
29 * CRC32 software implementation (brix_sss_crc32) using standard polynomial 0xedb88320u for integrity verification;
30 * Blowfish-CFB encryption/decryption (brix_sss_bf32_crypt) with EVP_CIPHER_CTX and zero IV for challenge-response;
31 * OpenSSL legacy provider loading (brix_sss_load_legacy_provider) ensuring SHA-256 digest availability on OpenSSL 3.x. */

33uint32_t
34brix_sss_read_be32(const u_char *p)
35{
36 return ((uint32_t) p[0] << 24)
37 | ((uint32_t) p[1] << 16)
38 | ((uint32_t) p[2] << 8)
39 | (uint32_t) p[3];
40}

42uint64_t
43brix_sss_read_be64(const u_char *p)
44{
45 return ((uint64_t) brix_sss_read_be32(p) << 32)
46 | (uint64_t) brix_sss_read_be32(p + 4);

118/*
119 * WHAT: File-local working state for one SSS blob verification pass. Carries the
120 * inputs (keytab, lifetime, wire blob) and the derived intermediates
121 * (selected key, located cipher span, decrypted-plaintext scratch and its
122 * length) between the verify sub-steps so each helper takes a single ctx
123 * rather than a long parameter list.
124 *
125 * WHY: brix_sss_verify_blob is an extern with an out-of-file caller
126 * (net/cms/server_auth.c) so its signature is frozen at 8 params; the
127 * decomposition threads this ctx internally instead. `clear` bounds the
128 * decrypt scratch — SSS credentials over any transport (XRootD kXR_auth or
129 * CMS kYR_xauth frame) are far smaller than this — and MUST be cleansed on
130 * every exit that touched it.
131 *
132 * HOW: brix_sss_verify_blob zero-inits one of these, then calls the header /
133 * decrypt / crc-check / identity sub-steps in sequence, each reading and
134 * writing only ctx fields. `clear_len` is the plaintext length with the
135 * trailing 4-byte CRC stripped.
136 */
137typedef struct {
138 ngx_array_t *keys;
139 time_t lifetime;
140 const u_char *blob;
141 size_t blob_len;
142 const brix_sss_key_t *key;
143 const u_char *cipher;
144 size_t cipher_len;
145 u_char clear[8192];
146 size_t clear_len;
147} brix_sss_blob_ctx_t;

149/*
150 * WHAT: Validates the SSS credential wire framing (minimum length, magic bytes,
151 * encryption marker, key-name size, and header-length bound) and returns
152 * the derived header length that separates the header from the ciphertext.
153 *
154 * WHY: Isolating pure framing validation keeps the CCN of each verify sub-step
155 * within the gate and lets the key-lookup step assume a well-formed header.
156 * Every framing failure is a fail-closed deny.
157 *
158 * HOW: Checks the "sss\0" magic and BRIX_SSS_ENC_BF32 enc marker, bounds the
159 * key-name size (multiple-of-8, <= BRIX_SSS_NAME_MAX, NUL-terminated),
160 * derives hdr_len = BRIX_SSS_HDR_LEN + kn_size, and bounds it against the
161 * blob length. Writes the header length through *hdr_len_out. Returns
162 * NGX_OK on a well-formed header, NGX_ERROR (with err populated) otherwise.
163 */
164static ngx_int_t
165brix_sss_blob_validate_framing(const brix_sss_blob_ctx_t *ctx,
166 size_t *hdr_len_out, char *err, size_t errsz)
167{
168 const u_char *blob = ctx->blob;
169 size_t hdr_len;
170 uint8_t kn_size;
171
172 if (blob == NULL
173 || ctx->blob_len < BRIX_SSS_HDR_LEN + BRIX_SSS_DATA_HDR_LEN + 4)
174 {
175 snprintf(err, errsz, "sss credential too short");
176 return NGX_ERROR;
177 }
178
179 if (blob[0] != 's' || blob[1] != 's' || blob[2] != 's' || blob[3] != '\0'
180 || blob[7] != BRIX_SSS_ENC_BF32)
181 {
182 snprintf(err, errsz, "sss credential bad magic/enc");
183 return NGX_ERROR;
184 }
185
186 kn_size = blob[6];
187 if (kn_size != 0 && (kn_size > BRIX_SSS_NAME_MAX || (kn_size & 0x07))) {
188 snprintf(err, errsz, "sss credential bad key-name size");
189 return NGX_ERROR;
190 }
191
192 hdr_len = BRIX_SSS_HDR_LEN + kn_size;
193 if (hdr_len >= ctx->blob_len || (kn_size && blob[hdr_len - 1] != '\0')) {
194 snprintf(err, errsz, "sss credential malformed header");
195 return NGX_ERROR;
196 }
197
198 *hdr_len_out = hdr_len;
199 return NGX_OK;
200}

202/*
203 * WHAT: Validates the SSS credential wire header, then locates the matching
204 * keytab entry by key id and records the ciphertext span into the ctx.
205 *
206 * WHY: Every framing and key-lookup failure is a fail-closed deny; keeping them
207 * in one linear step lets the orchestrator read as a sequence and lets the
208 * decrypt step assume a validated cipher span.
209 *
210 * HOW: Defers framing checks to brix_sss_blob_validate_framing, then reads the
211 * big-endian key id, resolves it against the keytab, and sets ctx->cipher /
212 * ctx->cipher_len (bounded against the scratch size). Returns NGX_OK on a
213 * well-formed header with a live key, NGX_ERROR (with err populated) on any
214 * framing or lookup failure.
215 */
216static ngx_int_t
217brix_sss_blob_parse_header(brix_sss_blob_ctx_t *ctx, char *err, size_t errsz)
218{
219 size_t hdr_len = 0;
220 int64_t key_id;
221 ngx_int_t rc;
222
223 rc = brix_sss_blob_validate_framing(ctx, &hdr_len, err, errsz);
224 if (rc != NGX_OK) {
225 return rc;
226 }
227
228 key_id = (int64_t) brix_sss_read_be64(ctx->blob + 8);
229 ctx->key = brix_sss_find_key_arr(ctx->keys, key_id);
230 if (ctx->key == NULL) {
231 snprintf(err, errsz, "sss key id %lld not in keytab", (long long) key_id);
232 return NGX_ERROR;
233 }
234
235 ctx->cipher = ctx->blob + hdr_len;
236 ctx->cipher_len = ctx->blob_len - hdr_len;
237 if (ctx->cipher_len <= 4 || ctx->cipher_len > sizeof(ctx->clear)) {
238 snprintf(err, errsz, "sss ciphertext length out of range");
239 return NGX_ERROR;
240 }
241
242 return NGX_OK;
243}

245/*
246 * WHAT: Decrypts the located ciphertext span with the selected SSS key into the
247 * ctx plaintext scratch and verifies the trailing CRC-32 over the payload.
248 *
249 * WHY: A wrong key or tampered credential decrypts to garbage whose appended CRC
250 * will not match — this is the security-load-bearing gate that turns a bad
251 * key into a clean deny rather than a forged identity. On any failure that
252 * has touched the scratch it must be cleansed by the caller.
253 *
254 * HOW: Runs brix_sss_bf32_crypt in decrypt mode; on success strips the trailing
255 * 4-byte CRC to yield ctx->clear_len, reads the wire CRC big-endian, and
256 * compares it to brix_sss_crc32 over the payload. Also enforces the minimum
257 * data-header length. Returns NGX_OK when the CRC matches and the plaintext
258 * is long enough, NGX_ERROR (with err populated) otherwise. The scratch is
259 * left populated for the caller's cleanse on both paths.
260 */
261static ngx_int_t
262brix_sss_blob_decrypt_verify_crc(brix_sss_blob_ctx_t *ctx, char *err,
263 size_t errsz)
264{
265 size_t out_len;
266 uint32_t got_crc, want_crc;
267
268 if (brix_sss_bf32_crypt(0, ctx->key->key, ctx->key->key_len,
269 ctx->cipher, ctx->cipher_len,
270 ctx->clear, sizeof(ctx->clear),
271 &out_len) != NGX_OK
272 || out_len <= 4)
273 {
274 snprintf(err, errsz, "sss decrypt failed");
275 return NGX_ERROR;
276 }
277
278 ctx->clear_len = out_len - 4;
279 got_crc = brix_sss_read_be32(ctx->clear + ctx->clear_len);
280 want_crc = brix_sss_crc32(ctx->clear, ctx->clear_len);
281 if (got_crc != want_crc || ctx->clear_len < BRIX_SSS_DATA_HDR_LEN) {
282 snprintf(err, errsz, "sss CRC mismatch (wrong key or tampered)");
283 return NGX_ERROR;
284 }
285
286 return NGX_OK;
287}

289/*
290 * WHAT: Enforces the SSS replay window on a CRC-verified plaintext by checking
291 * its generation timestamp against the configured credential lifetime.
292 *
293 * WHY: A CRC-valid credential minted long ago is a replay; the replay window is
294 * frozen and rejecting the expired credential keeps stale captures from
295 * authenticating. The caller cleanses the scratch on the reject path.
296 *
297 * HOW: Reads the big-endian gen_time at offset 32, computes `now` relative to
298 * BRIX_SSS_BASE_TIME, and denies when gen_time + lifetime has elapsed.
299 * Returns NGX_OK if still within window, NGX_ERROR (with err populated) if
300 * expired.
301 */
302static ngx_int_t
303brix_sss_blob_check_replay(brix_sss_blob_ctx_t *ctx, char *err, size_t errsz)
304{
305 uint32_t gen_time = brix_sss_read_be32(ctx->clear + 32);
306 uint32_t now = (uint32_t) (ngx_time() - BRIX_SSS_BASE_TIME);
307
308 if (gen_time + (uint32_t) ctx->lifetime <= now) {
309 snprintf(err, errsz, "sss credential expired (replay window)");
310 return NGX_ERROR;
311 }
312
313 return NGX_OK;
314}

316/*
317 * WHAT: Applies the credential options byte and, when requested, parses the
318 * decrypted identity TLV block into the caller's identity struct.
319 *
320 * WHY: SNDLID credentials are unsupported and must decline (not error) so the
321 * caller can distinguish them; identity parsing is the last step before a
322 * credential is trusted. The caller cleanses the scratch on every path.
323 *
324 * HOW: Reads the options byte at offset 39; returns NGX_DECLINED for
325 * BRIX_SSS_OPT_SNDLID. Otherwise, when id_out is non-NULL, forwards the
326 * identity payload (past the data header) to brix_sss_parse_identity.
327 * Returns NGX_OK on success, NGX_DECLINED for SNDLID, NGX_ERROR (with err
328 * populated) on a parse failure.
329 */
330static ngx_int_t
331brix_sss_blob_finish_identity(brix_sss_blob_ctx_t *ctx,
332 brix_sss_identity_t *id_out, char *err, size_t errsz)
333{
334 uint8_t options = ctx->clear[39];
335
336 if (options == BRIX_SSS_OPT_SNDLID) {
337 snprintf(err, errsz, "sss SNDLID credential unsupported");
338 return NGX_DECLINED;

340

343 ctx->clear_len - BRIX_SSS_DATA_HDR_LEN,

345 {
346 snprintf(err, errsz, "sss identity parse failed");
347 return NGX_ERROR;
348 }
349
350 return NGX_OK;
351}

353ngx_int_t
354brix_sss_verify_blob(ngx_array_t *keys, time_t lifetime,
355 const u_char *blob, size_t blob_len,
356 brix_sss_identity_t *id_out, const brix_sss_key_t **key_out,
357 char *err, size_t errsz)
358{
359 brix_sss_blob_ctx_t ctx = {0};
360 ngx_int_t rc;
361
362 ctx.keys = keys;
363 ctx.lifetime = lifetime;
364 ctx.blob = blob;
365 ctx.blob_len = blob_len;
366
367 if (key_out) {
368 *key_out = NULL;
369 }
370
371 /* Header framing + key lookup + ciphertext location: no scratch touched yet. */
372 rc = brix_sss_blob_parse_header(&ctx, err, errsz);
373 if (rc != NGX_OK) {
374 return rc;
375 }
376
377 /* From here the plaintext scratch is populated on every path and MUST be
378 * cleansed before return. */
379 rc = brix_sss_blob_decrypt_verify_crc(&ctx, err, errsz);
380 if (rc == NGX_OK) {
381 rc = brix_sss_blob_check_replay(&ctx, err, errsz);
382 }
383 if (rc == NGX_OK) {
384 rc = brix_sss_blob_finish_identity(&ctx, id_out, err, errsz);
385 }
386
387 OPENSSL_cleanse(ctx.clear, sizeof(ctx.clear));
388
389 if (rc == NGX_OK && key_out) {
390 *key_out = ctx.key;
391 }
392 return rc;
393}
