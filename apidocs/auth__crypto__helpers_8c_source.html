16/*
17 * WHAT: This file provides shared cryptographic helpers for Simple Shared Secret (SSS) authentication.
18 * Includes big-endian 32/64-bit read/write primitives for wire protocol framing, software CRC32c implementation
19 * for integrity verification (SSS uses CRC32 not HMAC), Blowfish-CFB encryption/decryption for challenge-response,
20 * and OpenSSL 3.x legacy provider loading for SHA-256 digest availability. All helpers are static or internal —
21 * used exclusively by sss/auth.c and sss/key_parse.c during SSS authentication flow.
22 *
23 * WHY: SSS authentication requires wire-format parsing (big-endian integers), integrity verification via CRC32 appended
24 * to plaintext before encryption, symmetric challenge encryption using Blowfish-CFB with zero IV, and OpenSSL 3.x
25 * compatibility for SHA-256 digest fetching from legacy provider. These helpers centralize crypto operations preventing
26 * duplication across SSS module files and ensuring consistent wire-format handling.
27 *
28 * HOW: Four helper categories → big-endian read/write (brix_sss_read_be32/be64, brix_sss_write_be32) for wire protocol parsing;
29 * CRC32 software implementation (brix_sss_crc32) using standard polynomial 0xedb88320u for integrity verification;
30 * Blowfish-CFB encryption/decryption (brix_sss_bf32_crypt) with EVP_CIPHER_CTX and zero IV for challenge-response;
31 * OpenSSL legacy provider loading (brix_sss_load_legacy_provider) ensuring SHA-256 digest availability on OpenSSL 3.x. */

33uint32_t
34brix_sss_read_be32(const u_char *p)
35{
36 return ((uint32_t) p[0] << 24)
37 | ((uint32_t) p[1] << 16)
38 | ((uint32_t) p[2] << 8)
39 | (uint32_t) p[3];
40}
41
42uint64_t
43brix_sss_read_be64(const u_char *p)
44{
45 return ((uint64_t) brix_sss_read_be32(p) << 32)
46 | (uint64_t) brix_sss_read_be32(p + 4);
47}

54{

56 p[1] = (u_char) (v >> 16);
57 p[2] = (u_char) (v >> 8);
58 p[3] = (u_char) v;

197 now = (uint32_t) (ngx_time() - BRIX_SSS_BASE_TIME);
198 if (gen_time + (uint32_t) lifetime <= now) {

201 return NGX_ERROR;
202 }

205 if (options == BRIX_SSS_OPT_SNDLID) {
206 OPENSSL_cleanse(clear, sizeof(clear));
207 snprintf(err, errsz, "sss SNDLID credential unsupported");
208 return NGX_DECLINED;
209 }
210

213 clear_len - BRIX_SSS_DATA_HDR_LEN,

215 {
216 OPENSSL_cleanse(clear, sizeof(clear));
217 snprintf(err, errsz, "sss identity parse failed");
218 return NGX_ERROR;
219 }
220
221 OPENSSL_cleanse(clear, sizeof(clear));
222 if (key_out) {
223 *key_out = key;
224 }
225 return NGX_OK;
226}
