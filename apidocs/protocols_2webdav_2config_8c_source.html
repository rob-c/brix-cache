1/*
2 * config.c - WebDAV location config create/merge and startup validation.
3 *
4 * WHAT: nginx HTTP module location configuration lifecycle — allocates a fresh config struct with all fields set to NGX_CONF_UNSET sentinel values (create_loc_conf), then merges parent→child inheritance chain applying ngx_conf_merge_* macros to resolve defaults (merge_loc_conf). After merging, performs startup validation: canonicalizes the export root path, validates CA/CRL file/directory paths, builds a cached OpenSSL X509_STORE from configured certificates and CRLs, loads JWKS keys for token auth, and validates TPC curl binary paths. Returns NGX_CONF_OK on success or NGX_CONF_ERROR with emerg-level log messages on failure.
5 *
6 * WHY: WebDAV requires many interdependent config values (CA store, root path, CORS origins, token issuer/audience, upstream URL) that must be validated together before accepting traffic. The merge chain ensures location-level directives inherit from server and main context defaults, while the validation phase catches configuration errors at postconfig time so nginx -t fails early rather than a worker crashing on first request. Building the CA store during config merge (not per-request) eliminates repeated X509_STORE construction cost.
7 *
8 * HOW: create_loc_conf allocates with ngx_pcalloc and sets every field to NGX_CONF_UNSET or NGX_CONF_UNSET_UINT; merge_loc_conf applies ngx_conf_merge_* macros for each field, then conditionally validates root path, CA/CRL paths, CORS origins, JWKS load, TPC binary paths, and upstream URL parsing. On validation success builds X509_STORE via webdav_build_ca_store() and attaches a pool cleanup handler for automatic deallocation on worker exit. The base-merge/validation and proxy/mirror/token clusters live in the sibling files config_merge.c and config_proxy.c (see config_internal.h); this file keeps the create/merge entrypoints, the CA-store cleanup, CORS validation, and the endpoint startup summary.
9 */

58/*
59 * Reject any configured CORS origin that is empty or contains control bytes.
60 * These strings are later reflected into Access-Control-Allow-Origin response
61 * headers, so a CR/LF in one would enable header injection — validate at config
62 * time so a bad value fails `nginx -t` instead of poisoning responses.
63 */
64ngx_int_t
65webdav_validate_cors_origins(ngx_conf_t *cf,
66 ngx_http_brix_webdav_loc_conf_t *conf)
67{
68 ngx_str_t *origins;
69 ngx_uint_t i;
70
71 if (conf->cors_origins == NULL) {
72 return NGX_OK;
73 }
74
75 origins = conf->cors_origins->elts;
76 for (i = 0; i < conf->cors_origins->nelts; i++) {
77 if (origins[i].len == 0
78 || webdav_tpc_str_has_ctl(origins[i].data, origins[i].len))
79 {
80 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
81 "brix_webdav: invalid CORS origin \"%V\"",
82 &origins[i]);
83 return NGX_ERROR;
84 }
85 }
86
87 return NGX_OK;
88}

204 * Merge parent (server/outer location) config into this location and run all
205 * startup validation. Three phases: (1) inherit/default every directive via
206 * ngx_conf_merge_*; (2) when WebDAV is enabled, resolve+confine the export root,
207 * validate CA/CRL/JWKS/TPC paths, and build the cached X509 CA store once;
208 * (3) resolve the upstream-proxy and traffic-mirror backends. Returns
209 * NGX_CONF_ERROR (with an emerg log) on any validation failure so `nginx -t`

212/*
213 * WHAT: short human label for the merged brix_webdav_auth mode (for the
214 * startup summary), so the log reads "optional (anonymous allowed)" not "1".
215 */
216static const char *
217webdav_auth_name(ngx_uint_t auth)
218{
219 switch (auth) {
220 case WEBDAV_AUTH_REQUIRED: return "required";
221 case WEBDAV_AUTH_OPTIONAL: return "optional (anonymous allowed)";
222 default: return "none (anonymous)";
223 }

226/*
227 * WHAT: is this location's WebDAV config a distinct endpoint, or just inherited
228 * unchanged from its parent location? `brix_webdav on` at server scope is
229 * inherited by every nested location; printing a banner for each would be
230 * noise, so we only print where the user-visible facts differ from the parent.
231 */
232ngx_uint_t
233webdav_summary_is_new(ngx_http_brix_webdav_loc_conf_t *conf,
234 ngx_http_brix_webdav_loc_conf_t *prev)
235{
236 if (prev == NULL || !prev->common.enable) {
237 return 1;
238 }
239 return (conf->auth != prev->auth
240 || conf->common.allow_write != prev->common.allow_write
241 || ngx_strcmp(conf->common.root_canon,
242 prev->common.root_canon) != 0);
243}

246 * WHAT: emit the friendly NOTICE banner for one WebDAV (davs://) endpoint —
247 * export, read/write, auth mode, which credential types are accepted, CRL
248 * posture, and TPC/proxy mode — plus WARN notes for valid-but-risky settings.
249 * WHY: give davs:// admins the same first-run confirmation the root:// banner

253 */

258void
259webdav_log_endpoint_summary(ngx_conf_t *cf,
260 ngx_http_brix_webdav_loc_conf_t *conf)
261{
262 ngx_uint_t has_x509 = (conf->cadir.len > 0 || conf->cafile.len > 0
263 || conf->proxy_certs);
264 ngx_uint_t has_token = (conf->jwks_key_count > 0);
265
266 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
267 "brix: WebDAV (davs://) endpoint ready — export \"%V\" (%s), auth: %s",
268 &conf->common.root,
269 conf->common.allow_write ? "read-write" : "read-only",
270 webdav_auth_name(conf->auth));
271
272 if (has_x509 || has_token) {
273 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
274 "brix: credentials accepted:%s%s",
275 has_x509 ? " x509/GSI-proxy" : "",
276 has_token ? " bearer-token" : "");
277 }
278 if (conf->crl.len > 0) {
279 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
280 "brix: revocation: CRL \"%V\"", &conf->crl);
281 }
282 if (conf->tpc) {
283 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
284 "brix: third-party copy (TPC COPY) enabled");
285 }
286 if (conf->upstream_proxy) {
287 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
288 "brix: mode: proxy — forwards to backend \"%V\"",
289 &conf->upstream_url);
290 }
291
292 webdav_log_endpoint_warnings(cf, conf, has_x509, has_token);
293}

295/* Valid-but-noteworthy settings, surfaced explicitly for a first-time admin so
296 * that permissive-by-configuration setups (revocation off, anonymous writes,
297 * auth-required-but-no-verifier) are never silent. Split out of the endpoint
298 * summary to keep each function's branching within the readability gate. */
299static void
300webdav_log_endpoint_warnings(ngx_conf_t *cf,
301 ngx_http_brix_webdav_loc_conf_t *conf, ngx_uint_t has_x509,
302 ngx_uint_t has_token)
303{
304 if (has_x509 && conf->crl.len == 0) {
305 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
306 "brix: NOTE: x509/GSI is accepted but no CRL is configured — "
307 "REVOKED certificates will be ACCEPTED (set brix_webdav_crl)");
308 }
309 if (conf->common.allow_write && conf->auth != WEBDAV_AUTH_REQUIRED) {
310 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
311 "brix: NOTE: writes are enabled but authentication is not "
312 "required — anonymous clients may be able to create/modify/delete "
313 "files (set brix_webdav_auth required)");
314 }
315 if (conf->auth == WEBDAV_AUTH_REQUIRED && !has_x509 && !has_token
316 && !conf->upstream_proxy)
317 {
318 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
319 "brix: NOTE: auth is required but no x509 CA or token JWKS is "
320 "configured — every client will be rejected (set "
321 "brix_webdav_cadir and/or brix_webdav_token_jwks)");
322 }
323}

325char *
326ngx_http_brix_webdav_merge_loc_conf(ngx_conf_t *cf,
327 void *parent, void *child)
328{
329 ngx_http_brix_webdav_loc_conf_t *prev = parent;
330 ngx_http_brix_webdav_loc_conf_t *conf = child;
331
332 if (webdav_merge_base_conf(cf, prev, conf) != NGX_CONF_OK) {
333 return NGX_CONF_ERROR;
334 }
335 if (webdav_validate_webdav_enabled(cf, prev, conf) != NGX_CONF_OK) {
336 return NGX_CONF_ERROR;
337 }
338 if (webdav_merge_auth_token_conf(cf, prev, conf) != NGX_CONF_OK) {
339 return NGX_CONF_ERROR;
340 }
341 if (webdav_merge_upstream_conf(cf, prev, conf) != NGX_CONF_OK) {
342 return NGX_CONF_ERROR;
343 }
344 if (webdav_merge_mirror_and_summary(cf, prev, conf) != NGX_CONF_OK) {
345 return NGX_CONF_ERROR;
346 }
347
348 return NGX_CONF_OK;
349}
