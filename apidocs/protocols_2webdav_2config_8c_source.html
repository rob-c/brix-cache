1/*
2 * config.c - WebDAV location config create/merge and startup validation.
3 *
4 * WHAT: nginx HTTP module location configuration lifecycle — allocates a fresh config struct with all fields set to NGX_CONF_UNSET sentinel values (create_loc_conf), then merges parent→child inheritance chain applying ngx_conf_merge_* macros to resolve defaults (merge_loc_conf). After merging, performs startup validation: canonicalizes the export root path, validates CA/CRL file/directory paths, builds a cached OpenSSL X509_STORE from configured certificates and CRLs, loads JWKS keys for token auth, and validates TPC curl binary paths. Returns NGX_CONF_OK on success or NGX_CONF_ERROR with emerg-level log messages on failure.
5 *
6 * WHY: WebDAV requires many interdependent config values (CA store, root path, CORS origins, token issuer/audience, upstream URL) that must be validated together before accepting traffic. The merge chain ensures location-level directives inherit from server and main context defaults, while the validation phase catches configuration errors at postconfig time so nginx -t fails early rather than a worker crashing on first request. Building the CA store during config merge (not per-request) eliminates repeated X509_STORE construction cost.
7 *
8 * HOW: create_loc_conf allocates with ngx_pcalloc and sets every field to NGX_CONF_UNSET or NGX_CONF_UNSET_UINT; merge_loc_conf applies ngx_conf_merge_* macros for each field, then conditionally validates root path, CA/CRL paths, CORS origins, JWKS load, TPC binary paths, and upstream URL parsing. On validation success builds X509_STORE via webdav_build_ca_store() and attaches a pool cleanup handler for automatic deallocation on worker exit.
9 */

56/*
57 * Reject any configured CORS origin that is empty or contains control bytes.
58 * These strings are later reflected into Access-Control-Allow-Origin response
59 * headers, so a CR/LF in one would enable header injection — validate at config
60 * time so a bad value fails `nginx -t` instead of poisoning responses.
61 */
62static ngx_int_t
63webdav_validate_cors_origins(ngx_conf_t *cf,
64 ngx_http_brix_webdav_loc_conf_t *conf)
65{
66 ngx_str_t *origins;
67 ngx_uint_t i;
68
69 if (conf->cors_origins == NULL) {
70 return NGX_OK;
71 }
72
73 origins = conf->cors_origins->elts;
74 for (i = 0; i < conf->cors_origins->nelts; i++) {
75 if (origins[i].len == 0
76 || webdav_tpc_str_has_ctl(origins[i].data, origins[i].len))
77 {
78 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
79 "brix_webdav: invalid CORS origin \"%V\"",
80 &origins[i]);
81 return NGX_ERROR;
82 }
83 }
84
85 return NGX_OK;
86}

242/*
243 * WHAT: emit the friendly NOTICE banner for one WebDAV (davs://) endpoint —
244 * export, read/write, auth mode, which credential types are accepted, CRL
245 * posture, and TPC/proxy mode — plus WARN notes for valid-but-risky settings.
246 * WHY: give davs:// admins the same first-run confirmation the root:// banner
247 * gives (visible in `nginx -t`), and surface the same foot-guns: revocation
248 * off, anonymous writes, auth-required-but-no-credentials. Logged at merge
249 * time (per location), the only hook that reliably sees every WebDAV location.
250 */
251static void
252webdav_log_endpoint_summary(ngx_conf_t *cf,
253 ngx_http_brix_webdav_loc_conf_t *conf)
254{
255 ngx_uint_t has_x509 = (conf->cadir.len > 0 || conf->cafile.len > 0
256 || conf->proxy_certs);
257 ngx_uint_t has_token = (conf->jwks_key_count > 0);
258
259 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
260 "brix: WebDAV (davs://) endpoint ready — export \"%V\" (%s), auth: %s",
261 &conf->common.root,
262 conf->common.allow_write ? "read-write" : "read-only",
263 webdav_auth_name(conf->auth));
264
265 if (has_x509 || has_token) {
266 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
267 "brix: credentials accepted:%s%s",
268 has_x509 ? " x509/GSI-proxy" : "",
269 has_token ? " bearer-token" : "");
270 }
271 if (conf->crl.len > 0) {
272 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
273 "brix: revocation: CRL \"%V\"", &conf->crl);
274 }
275 if (conf->tpc) {
276 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
277 "brix: third-party copy (TPC COPY) enabled");
278 }
279 if (conf->upstream_proxy) {
280 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
281 "brix: mode: proxy — forwards to backend \"%V\"",
282 &conf->upstream_url);
283 }
284
285 /* Valid-but-noteworthy settings, surfaced explicitly for a first-time admin. */
286 if (has_x509 && conf->crl.len == 0) {
287 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
288 "brix: NOTE: x509/GSI is accepted but no CRL is configured — "
289 "REVOKED certificates will be ACCEPTED (set brix_webdav_crl)");
290 }
291 if (conf->common.allow_write && conf->auth != WEBDAV_AUTH_REQUIRED) {
292 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
293 "brix: NOTE: writes are enabled but authentication is not "
294 "required — anonymous clients may be able to create/modify/delete "
295 "files (set brix_webdav_auth required)");
296 }
297 if (conf->auth == WEBDAV_AUTH_REQUIRED && !has_x509 && !has_token
298 && !conf->upstream_proxy)
299 {
300 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
301 "brix: NOTE: auth is required but no x509 CA or token JWKS is "
302 "configured — every client will be rejected (set "
303 "brix_webdav_cadir and/or brix_webdav_token_jwks)");
304 }
305}
