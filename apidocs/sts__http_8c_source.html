1/*
2 * sts_http.c — STS transport + response parsing (phase-79 split).
3 *
4 * WHAT: fetch the signed STS URL over http/https with libcurl and parse the
5 * AssumeRole/GetSessionToken XML reply into AccessKeyId / SecretAccessKey
6 * / SessionToken. Returns the HTTP status to the caller and the extracted
7 * credentials in the caller's fixed buffers.
8 * WHY: split out of sts.c (formerly 815 lines) as the side-effecting edge —
9 * network I/O and XML parsing — kept apart from the pure request builder
10 * (sts_sign.c) and the orchestrator (sts.c). Behaviour is byte-for-byte
11 * the pre-split code: same curl hardening, same bounded body, same
12 * name-directed XML descent, same log strings (secrets never logged).
13 * HOW: sts_http_get mirrors webdav/tpc_curl.c — TLS verified, protocols pinned
14 * to http/https, body streamed into a BRIX_STS_RESP_MAX-bounded buffer.
15 * sts_parse_response walks the libxml2 tree for the credential elements;
16 * a no-libxml2 build fails closed with an explicit message.
17 *
18 * The reused building blocks:
19 * - outbound HTTP: libcurl, mirroring src/protocols/webdav/tpc_curl.c
20 * - XML parsing: libxml2 (guarded by BRIX_HAVE_LIBXML2)
21 */

36/*
37 * libcurl write callback: append the response body into a growable pool buffer,
38 * bounded by BRIX_STS_RESP_MAX so a hostile/huge response cannot exhaust memory.
39 * Returns 0 (short write → curl aborts) on overflow or OOM.
40 */
41static size_t
42sts_write_cb(void *ptr, size_t size, size_t nmemb, void *userdata)
43{
44 sts_resp_t *r = userdata;
45 size_t n = size * nmemb;
46
47 if (n == 0) {
48 return 0;
49 }
50 if (r->len + n > BRIX_STS_RESP_MAX) {
51 return 0;
52 }
53 ngx_memcpy(r->buf + r->len, ptr, n);
54 r->len += n;
55 return n;
56}

59/*
60 * sts_http_get — GET the signed STS URL and capture the response body.
61 *
62 * Restricts the transfer to http/https, enforces TLS verification, and streams
63 * the body into `resp` (a pre-sized pool buffer). Returns NGX_OK with the HTTP
64 * status in *http_status, or NGX_ERROR on a transport error (logged, no
65 * secrets). Mirrors the curl setup in webdav_tpc_run_curl_core().
66 */
67ngx_int_t
68sts_http_get(const char *url, sts_resp_t *resp, long *http_status,
69 ngx_log_t *log)
70{
71 CURL *curl;
72 CURLcode res;
73 char errbuf[CURL_ERROR_SIZE];
74 ngx_int_t rc = NGX_ERROR;
75
76 curl = curl_easy_init();
77 if (curl == NULL) {
78 ngx_log_error(NGX_LOG_ERR, log, 0, "brix_sts: curl_easy_init failed");
79 return NGX_ERROR;
80 }
81
82 errbuf[0] = '\0';
83 curl_easy_setopt(curl, CURLOPT_ERRORBUFFER, errbuf);
84 curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
85 curl_easy_setopt(curl, CURLOPT_URL, url);
86#ifdef CURLOPT_PROTOCOLS_STR
87 curl_easy_setopt(curl, CURLOPT_PROTOCOLS_STR, "http,https");
88#else
89 curl_easy_setopt(curl, CURLOPT_PROTOCOLS,
90 (long) (CURLPROTO_HTTP | CURLPROTO_HTTPS));
91#endif
92 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYPEER, 1L);
93 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYHOST, 2L);
94 curl_easy_setopt(curl, CURLOPT_CONNECTTIMEOUT, 10L);
95 curl_easy_setopt(curl, CURLOPT_TIMEOUT, 30L);
96 curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, sts_write_cb);
97 curl_easy_setopt(curl, CURLOPT_WRITEDATA, resp);
98
99 res = curl_easy_perform(curl);
100 if (res == CURLE_OK) {
101 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, http_status);
102 rc = NGX_OK;
103 } else {
104 ngx_log_error(NGX_LOG_WARN, log, 0, "brix_sts: STS request failed: %s",
105 errbuf[0] ? errbuf : curl_easy_strerror(res));
106 }
107
108 curl_easy_cleanup(curl);
109 return rc;
110}

119/*
120 * Recursively search the parse tree for the first element named `name` and copy
121 * its text content into out[outsz]. Returns 1 if found, 0 otherwise. The STS
122 * AssumeRole/GetSessionToken responses nest <Credentials> under result and
123 * response wrappers, so a name-directed descent is simpler than a fixed path.
124 */
125static int
126sts_xml_find(xmlNodePtr node, const char *name, char *out, size_t outsz)
127{
128 xmlNodePtr n;
129
130 for (n = node; n != NULL; n = n->next) {
131 if (n->type == XML_ELEMENT_NODE
132 && xmlStrcmp(n->name, (const xmlChar *) name) == 0)
133 {
134 xmlChar *txt = xmlNodeGetContent(n);
135 if (txt != NULL) {
136 ngx_snprintf((u_char *) out, outsz, "%s%Z", (char *) txt);
137 xmlFree(txt);
138 return 1;
139 }
140 }
141 if (n->children != NULL
142 && sts_xml_find(n->children, name, out, outsz))
143 {
144 return 1;
145 }
146 }
147 return 0;
148}

151/*
152 * sts_parse_response — extract AccessKeyId / SecretAccessKey / SessionToken
153 * from the STS XML body into the caller's fixed buffers. Returns NGX_OK when at
154 * least AccessKeyId and SecretAccessKey are present; NGX_ERROR (with the fault
155 * logged, secrets excluded) otherwise. SessionToken is optional.
156 */
157ngx_int_t
158sts_parse_response(const u_char *body, size_t len,
159 const sts_creds_buf_t *creds, ngx_log_t *log)
160{
161 xmlDocPtr doc;
162 xmlNodePtr root;
163 ngx_int_t rc = NGX_ERROR;
164
165 creds->ak[0] = creds->sk[0] = creds->session[0] = '\0';
166
167 doc = xmlReadMemory((const char *) body, (int) len, "sts.xml", NULL,
168 XML_PARSE_NONET | XML_PARSE_NOERROR | XML_PARSE_NOWARNING);
169 if (doc == NULL) {
170 ngx_log_error(NGX_LOG_ERR, log, 0,
171 "brix_sts: unparseable STS response");
172 return NGX_ERROR;
173 }
174
175 root = xmlDocGetRootElement(doc);
176 if (root != NULL
177 && sts_xml_find(root, "AccessKeyId", creds->ak, creds->aksz)
178 && sts_xml_find(root, "SecretAccessKey", creds->sk, creds->sksz)
179 && creds->ak[0] != '\0' && creds->sk[0] != '\0')
180 {
181 (void) sts_xml_find(root, "SessionToken", creds->session,
182 creds->sesssz);
183 rc = NGX_OK;
184 } else {
185 ngx_log_error(NGX_LOG_ERR, log, 0,
186 "brix_sts: STS response missing credential fields");
187 }
188
189 xmlFreeDoc(doc);
190 return rc;
191}
