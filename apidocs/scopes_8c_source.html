1/*
2 * scopes.c — WLCG scope parsing and path authorization.
3 *
4 * WHAT: Parses space-separated WLCG "permission:path" scope claims into structured
5 * brix_token_scope_t entries, then checks read/write access against parsed
6 * scopes for a given request path. Recognizes storage.read/write/create/modify/stage
7 * permissions with exact-length matching to prevent substring attacks.
8 * WHY: WLCG tokens encode authorization as space-separated scope claims (e.g.,
9 * "storage.read:/atlas/reco"). This module extracts permission and path components,
10 * applies prefix-based path matching with boundary checks ("/data" must not match "/database"),
11 * and provides read/write access decision functions for use by validate.c and s3/auth.c.
12 * HOW: parse_scopes() tokenizes space-separated input → splits on ":" separator → copies path
13 * component (default "/" if empty) → sets permission flags via exact-length memcmp → returns
14 * count of parsed scopes. check_read()/check_write() iterate over parsed scopes, checking
15 * read/write/create flags against scope_path_matches() prefix comparison with boundary enforcement.
16 */
17

45/* WHAT: Copies a permission:path path component into the brix_token_scope_t structure, enforcing BRIX_SCOPE_PATH_MAX boundary. When path_len == 0 (no path specified in "permission:" token), defaults to "/" which matches all paths per WLCG token profile convention. Enforces buffer overflow prevention via truncation at BRIX_SCOPE_PATH_MAX - 1 when input exceeds capacity.
46 * WHY: WLCG scope tokens use "storage.read:/atlas/reco" format where the path component grants access only to specific directories. Defaulting empty paths to "/" follows WLCG convention (unscoped permission = full access). Buffer overflow guard prevents malicious tokens with oversized path components from corrupting downstream scope validation logic.
47 * HOW: Three-step → if path_len == 0, set scope->path = "/" default; else truncate at BRIX_SCOPE_PATH_MAX - 1 for boundary safety; memcpy(path) into scope->path with null termination at truncated length. */
48
49static void
50token_scope_copy_path(brix_token_scope_t *scope, const char *path,
51 size_t path_len)
52{
53 if (path_len == 0) {
54 scope->path[0] = '/';
55 scope->path[1] = '\0';
56 return;
57 }
58
59 if (path_len >= BRIX_SCOPE_PATH_MAX) {
60 path_len = BRIX_SCOPE_PATH_MAX - 1;
61 }
62
63 memcpy(scope->path, path, path_len);
64 scope->path[path_len] = '\0';
65}

67/* WHAT: Parses a WLCG "storage.*" permission string into boolean flags on brix_token_scope_t using exact-length memcmp comparison. Recognizes five permissions: storage.read (12 chars), storage.write (13 chars), storage.create (14 chars), storage.modify (14 chars), and storage.stage (13 chars — treated as read-only). Length-based matching prevents substring attacks where "storage.re" could incorrectly match "storage.read".
68 * WHY: WLCG token scope claims use standardized permission strings; exact-length memcmp ensures precise matching without partial-string ambiguity. Storage.stage is mapped to read-only (not write) per WLCG staging semantics — staged files are read-accessible but not writable until committed. Length-based validation prevents malformed tokens from granting unintended permissions through substring overlap.
69 * HOW: Sequential memcmp comparison against five known permission strings with exact length checks (12/13/14 chars), setting scope->read/write/create/modify/stage flags accordingly on match. Returns immediately after first positive match; storage.stage falls-through sets read=1 as final case. */
70
71static void
72token_scope_set_permission(brix_token_scope_t *scope,
73 const char *permission, size_t permission_len)
74{
75 if (permission_len == 12
76 && memcmp(permission, "storage.read", 12) == 0)
77 {
78 scope->read = 1;
79 return;
80 }
81
82 if (permission_len == 13
83 && memcmp(permission, "storage.write", 13) == 0)
84 {
85 scope->write = 1;
86 return;
87 }
88
89 if (permission_len == 14
90 && memcmp(permission, "storage.create", 14) == 0)
91 {
92 scope->create = 1;
93 return;
94 }
95
96 if (permission_len == 14
97 && memcmp(permission, "storage.modify", 14) == 0)
98 {
99 scope->modify = 1;
100 return;
101 }
102
103 if (permission_len == 13
104 && memcmp(permission, "storage.stage", 13) == 0)
105 {
106 scope->read = 1;
107 }
108}

110/*
111 * brix_token_parse_scopes — parse the WLCG "scope" claim into a structured
112 * list.
113 *
114 * The scope claim is a space-separated list of "permission:path" tokens
115 * as defined by the WLCG token profile. Examples:
116 * "storage.read:/atlas/reco storage.write:/atlas/output"
117 *
118 * Each token is parsed into an brix_token_scope_t; at most max_scopes
119 * entries are written to scopes[].
120 *
121 * Preconditions: scopes[] must have room for max_scopes elements.
122 * Postconditions: scopes[0..return_value-1] are initialised.
123 * Returns: number of scope entries written (may be 0 on empty input).
124 */
125int
126brix_token_parse_scopes(const char *scope_str,
127 brix_token_scope_t *scopes, int max_scopes)
128{
129 const char *cursor;
130 int count;
131
132 if (scope_str == NULL || scope_str[0] == '\0' || max_scopes <= 0) {
133 return 0;
134 }
135
136 cursor = scope_str;
137 count = 0;
138
139 while (*cursor != '\0' && count < max_scopes) {
140 const char *entry_start;
141 const char *entry_end;
142 const char *colon;
143 const char *path;
144 size_t entry_len;
145 size_t permission_len;
146 size_t path_len;
147
148 cursor = token_scope_skip_spaces(cursor);
149 if (*cursor == '\0') {
150 break;
151 }
152
153 entry_start = cursor;
154 entry_end = token_scope_end(entry_start);
155 entry_len = (size_t) (entry_end - entry_start);
156
157 colon = memchr(entry_start, ':', entry_len);
158 if (colon == NULL) {
159 cursor = entry_end;
160 continue;
161 }
162
163 permission_len = (size_t) (colon - entry_start);
164 path = colon + 1;
165 path_len = (size_t) (entry_end - path);
166
167 memset(&scopes[count], 0, sizeof(scopes[count]));
168 token_scope_copy_path(&scopes[count], path, path_len);
169 token_scope_set_permission(&scopes[count], entry_start,
170 permission_len);
171
172 /*
173 * Preserve the previous behavior: an unknown storage.* permission
174 * still consumes a parsed slot, but grants no read/write capability.
175 */
176 count++;
177 cursor = entry_end;
178 }
179
180 return count;
181}

183/*
184 * brix_token_scope_path_matches — decide whether scope_path covers
185 * request_path.
186 *
187 * Rules:
188 * - scope "/" matches every path.
189 * - A trailing "/" in scope_path is ignored for comparison purposes.
190 * - scope_path matches request_path only if scope_path is a prefix of
191 * request_path AND the next character in request_path is either '/'
192 * or NUL.
193 *
194 * Path-prefix attack prevention: "/data" must not match "/database".
195 * The boundary check (next == '/' || next == '\0') enforces this.
196 *
197 * Preconditions: both scope_path and request_path are NUL-terminated,
198 * absolute paths (start with '/').
199 * Returns: 1 if scope covers the request path, 0 otherwise.
200 *
201 * Exposed (non-static) so the SciTokens issuer registry (issuer_registry.c)
202 * reuses the exact same boundary-checked prefix logic for base_path /
203 * restricted_path scoping (phase-59 W1).
204 */
205int
206brix_token_scope_path_matches(const char *scope_path,
207 const char *request_path)
208{
209 size_t scope_len;
210 char next;
211
212 scope_len = strlen(scope_path);
213 if (scope_len == 1 && scope_path[0] == '/') {
214 return 1;
215 }
216
217 if (scope_len > 1 && scope_path[scope_len - 1] == '/') {
218 scope_len--;
219 }
220
221 if (strncmp(scope_path, request_path, scope_len) != 0) {
222 return 0;
223 }
224
225 /* path-prefix attack: "/data" must not match "/database" */
226 next = request_path[scope_len];
227 return (next == '\0' || next == '/');
228}

230/*
231 * brix_token_check_read — test whether any scope grants storage.read
232 * access to path.
233 *
234 * Preconditions: scopes[0..scope_count-1] produced by brix_token_parse_scopes().
235 * Returns: 1 if access is granted, 0 if denied.
236 */
237int
238brix_token_check_read(const brix_token_scope_t *scopes, int scope_count,
239 const char *path)
240{
241 int scope_index;
242
243 for (scope_index = 0; scope_index < scope_count; scope_index++) {
244 if (scopes[scope_index].read
245 && brix_token_scope_path_matches(scopes[scope_index].path, path))
246 {
247 return 1;
248 }
249 }
250
251 return 0;
252}

254/*
255 * brix_token_check_write — test whether any scope grants storage.write or
256 * storage.create access to path.
257 *
258 * Both write and create permission are treated as sufficient for write ops;
259 * this matches the WLCG token profile intent where "create" is a write-like
260 * capability restricted to new objects.
261 *
262 * Returns: 1 if access is granted, 0 if denied.
263 */
264int
265brix_token_check_write(const brix_token_scope_t *scopes, int scope_count,
266 const char *path)
267{
268 int scope_index;
269
270 for (scope_index = 0; scope_index < scope_count; scope_index++) {
271 /*
272 * "modify" is a write-like WLCG capability (overwrite/modify/delete of an
273 * existing object); it was parsed into the scope (scopes.h) and produced by
274 * the macaroon parser (MANAGE -> storage.modify) but consulted by NO authz
275 * decision — so any JWT/macaroon presenting storage.modify was wrongly
276 * denied every write. Honor it alongside write/create (the codebase already
277 * conflates create-new with overwrite in this coarse write gate).
278 */
279 if ((scopes[scope_index].write || scopes[scope_index].create
280 || scopes[scope_index].modify)
281 && brix_token_scope_path_matches(scopes[scope_index].path, path))
282 {
283 return 1;
284 }
285 }
286
287 return 0;
288}
