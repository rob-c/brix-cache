8/* Create LPATH and every missing parent directory (local `mkdir -p`).
9 *
10 * WHY: the recursive download nests the copied tree under the destination
11 * (recursive_dest_root → e.g. <dst>/<basename>), so the very first directory
12 * created can be two levels below an existing parent (<dst> itself may not yet
13 * exist). A single mkdir() would fail with ENOENT on that missing parent; the
14 * remote upload side already creates parents (brix_mkdir parents=1), so the
15 * local side must match. Idempotent: an existing component is not an error
16 * (deeper failures such as a non-directory in the path still surface ENOTDIR).
17 * Returns 0 on success, -1 with errno set on the first uncreatable component. */
18static int
19local_mkdirs(const char *lpath, mode_t mode)
20{
21 char buf[XRDC_PATH_MAX];
22 size_t n = strlen(lpath);
23 size_t i;
24
25 if (n == 0 || n >= sizeof(buf)) {
26 errno = (n == 0) ? EINVAL : ENAMETOOLONG;
27 return -1;
28 }
29 memcpy(buf, lpath, n + 1);
30 for (i = 1; i < n; i++) { /* each intermediate component */
31 if (buf[i] != '/') {
32 continue;
33 }
34 buf[i] = '\0';
35 if (mkdir(buf, mode) != 0 && errno != EEXIST) {
36 return -1;
37 }
38 buf[i] = '/';
39 }
40 if (mkdir(buf, mode) != 0 && errno != EEXIST) { /* the leaf */
41 return -1;
42 }
43 return 0;
44}

47/* 1 = the remote file (rpath, over the walker's open conn) and the local file
48 * (lpath) have the same digest under o->sync_cksum_algo; 0 = differ or
49 * undeterminable.
50 *
51 * WHAT: the XRDC_SYNC_CKSUM content check for the recursive walkers.
52 * WHY: brix_sync_should_skip is only the size gate for CKSUM mode; the byte
53 * comparison happens here. Any failure (bad algo, query error, open or
54 * read error) returns 0 so the walker COPIES — an undeterminable
55 * comparison must never skip (silent skip on error is data loss).
56 * HOW: server digest via kXR_Qcksum on the already-open conn (no reconnect
57 * per file), local digest via brix_cksum_fd; case-insensitive hex compare. */
58static int
59sync_cksum_match(brix_conn *c, const char *rpath, const char *lpath,
60 const brix_copy_opts *o)
61{
62 const char *algo = o->sync_cksum_algo ? o->sync_cksum_algo : "adler32";
63 brix_cksum_algo a;
64 brix_status st;
65 char rhex[132], lhex[132];
66 int fd, rc;
67
68 brix_status_clear(&st);
69 if (brix_cksum_algo_parse(algo, &a) != 0) {
70 return 0;
71 }
72 if (brix_query_cksum(c, rpath, algo, rhex, sizeof(rhex), &st) != 0) {
73 return 0;
74 }
75 fd = open(lpath, O_RDONLY);
76 if (fd < 0) {
77 return 0;
78 }
79 rc = brix_cksum_fd(fd, a, lhex, sizeof(lhex), &st);
80 close(fd);
81 if (rc != 0) {
82 return 0;
83 }
84 return strcasecmp(rhex, lhex) == 0;
85}

88/* Recursively remove a local tree (mirror-delete helper).
89 *
90 * WHAT: post-order deletion of a local path without following symlinks.
91 * WHY: called by mirror_delete_local to prune extra local paths that have no
92 * corresponding remote entry. lstat (not stat) is used throughout so a
93 * symlink inside the destination tree can never escape the root and delete
94 * unrelated data.
95 * HOW: lstat to classify; dirs → opendir + recurse each child + rmdir; files
96 * → unlink. Returns 0 on success, -1 with errno set on first failure. */
97static int
98local_rmtree(const char *path)
99{
100 struct stat sb;
101
102 if (lstat(path, &sb) != 0) {
103 return -1;
104 }
105 if (S_ISDIR(sb.st_mode)) {
106 DIR *d = opendir(path);
107 struct dirent *de;
108 if (d == NULL) { return -1; }
109 while ((de = readdir(d)) != NULL) {
110 char child[XRDC_PATH_MAX];
111 if (de->d_name[0] == '.'
112 && (de->d_name[1] == '\0'
113 || (de->d_name[1] == '.' && de->d_name[2] == '\0'))) {
114 continue;
115 }
116 if ((size_t) snprintf(child, sizeof(child), "%s/%s", path,
117 de->d_name) >= sizeof(child)) {
118 closedir(d);
119 errno = ENAMETOOLONG;
120 return -1;
121 }
122 if (local_rmtree(child) != 0) { closedir(d); return -1; }
123 }
124 closedir(d);
125 return rmdir(path);
126 }
127 return unlink(path);
128}

131/* Prune local entries under lpath that are absent from the remote listing.
132 *
133 * WHAT: the download-direction mirror pass for xrdcp -r --sync --delete.
134 * WHY: after the download loop the ents[] array holds every name the remote
135 * directory contains; a local child absent from that set is an extra
136 * that must be removed to keep the local tree a faithful mirror of the
137 * remote. Excluded paths are outside the sync scope and must never be
138 * deleted (the filter gate enforces this).
139 * HOW: opendir(lpath); for each local entry, linear-scan ents[] by name; if
140 * absent AND brix_copy_filter_match passes → dry-run print or delete via
141 * local_rmtree (for dirs) / unlink (for files). */
142static void
143mirror_delete_local(const char *lpath, const brix_dirent *ents, size_t nents,
144 const char *rel, const brix_copy_opts *o)
145{
146 DIR *d;
147 struct dirent *de;
148
149 d = opendir(lpath);
150 if (d == NULL) {
151 fprintf(stderr, "xrdcp: --delete: opendir %s: %s\n", lpath, strerror(errno));
152 return;
153 }
154 while ((de = readdir(d)) != NULL) {
155 char relc[XRDC_PATH_MAX];
156 char lchild[XRDC_PATH_MAX];
157 size_t i;
158 int in_remote = 0;
159
160 if (de->d_name[0] == '.'
161 && (de->d_name[1] == '\0'
162 || (de->d_name[1] == '.' && de->d_name[2] == '\0'))) {
163 continue;
164 }
165 /* Is this local name present in the remote listing? */
166 for (i = 0; i < nents; i++) {
167 if (strcmp(ents[i].name, de->d_name) == 0) { in_remote = 1; break; }
168 }
169 if (in_remote) {
170 continue;
171 }
172 /* Build relative and absolute paths for the filter check and deletion. */
173 if ((size_t) snprintf(relc, sizeof(relc), "%s%s%s",
174 rel, rel[0] ? "/" : "", de->d_name) >= sizeof(relc)
175 || (size_t) snprintf(lchild, sizeof(lchild), "%s/%s",
176 lpath, de->d_name) >= sizeof(lchild)) {
177 fprintf(stderr, "xrdcp: --delete: path too long under %s\n", lpath);
178 continue;
179 }
180 /* Excluded entries are outside the sync scope — never delete them. */
181 if (!brix_copy_filter_match(o, relc)) {
182 continue;
183 }
184 if (o->dry_run) {
185 printf("[dry-run] delete %s\n", lchild);
186 } else {
187 /* lstat first: for regular files (and symlinks) use unlink directly
188 * rather than the full recursive rmtree; local_rmtree is only
189 * needed for directories. lstat avoids following a symlink that
190 * might point outside the tree. */
191 struct stat csb;
192 if (lstat(lchild, &csb) == 0 && S_ISDIR(csb.st_mode)) {
193 if (local_rmtree(lchild) != 0) {
194 fprintf(stderr, "xrdcp: --delete: cannot remove %s: %s\n",
195 lchild, strerror(errno));
196 }
197 } else if (unlink(lchild) != 0) {
198 fprintf(stderr, "xrdcp: --delete: cannot remove %s: %s\n",
199 lchild, strerror(errno));
200 }
201 }
202 }
203 closedir(d);
204}

207/* Prune remote entries under rpath that have no local counterpart in lpath.
208 *
209 * WHAT: the upload-direction mirror pass for xrdcp -r --sync --delete.
210 * WHY: after the upload loop the local tree is authoritative; a remote entry
211 * with no local counterpart must be removed so the remote mirrors the
212 * local. Excluded paths are outside the sync scope and must never be
213 * deleted.
214 * HOW: brix_dirlist(rpath) for the remote snapshot; for each remote entry,
215 * lstat the local counterpart; if absent AND filter passes → dry-run
216 * print or delete (brix_rmtree for dirs, brix_rm for files).
217 * GUARD: this pass treats the LOCAL tree as authoritative, so it MUST NOT run
218 * alongside --remove-source (which deletes each local source after upload):
219 * the sources would already be gone here and every uploaded remote copy
220 * would be purged. xrdcp rejects that flag combination at parse time
221 * (see the --delete/--remove-source guard in main), so this function is
222 * only ever reached with the local tree intact. */
223static void
224mirror_delete_remote(brix_conn *c, const char *rpath, const char *lpath,
225 const char *rel, const brix_copy_opts *o)
226{
227 brix_dirent *ents = NULL;
228 brix_status st;
229 size_t n = 0, i;
230
231 brix_status_clear(&st);
232 if (brix_dirlist(c, rpath, 1, &ents, &n, &st) != 0) {
233 fprintf(stderr, "xrdcp: --delete: dirlist %s: %s\n", rpath, st.msg);
234 return;
235 }
236 for (i = 0; i < n; i++) {
237 char relc[XRDC_PATH_MAX];
238 char rchild[XRDC_PATH_MAX];
239 char lchild[XRDC_PATH_MAX];
240 struct stat sb;
241 int is_dir;
242
243 if (ents[i].name[0] == '.'
244 && (ents[i].name[1] == '\0'
245 || (ents[i].name[1] == '.' && ents[i].name[2] == '\0'))) {
246 continue;
247 }
248 if ((size_t) snprintf(relc, sizeof(relc), "%s%s%s",
249 rel, rel[0] ? "/" : "", ents[i].name) >= sizeof(relc)
250 || (size_t) snprintf(rchild, sizeof(rchild), "%s/%s",
251 rpath, ents[i].name) >= sizeof(rchild)
252 || (size_t) snprintf(lchild, sizeof(lchild), "%s/%s",
253 lpath, ents[i].name) >= sizeof(lchild)) {
254 fprintf(stderr, "xrdcp: --delete: path too long under %s\n", rpath);
255 continue;
256 }
257 /* Keep the remote entry if the local counterpart exists. */
258 if (lstat(lchild, &sb) == 0) {
259 continue;
260 }
261 /* Excluded entries are outside the sync scope — never delete them. */
262 if (!brix_copy_filter_match(o, relc)) {
263 continue;
264 }
265 /* Determine if the remote entry is a directory. If dirlist provided no stat
266 * info (have_stat=0), do a defensive stat to resolve the type before deletion.
267 * Without this, a stat-less directory would be treated as a file, causing
268 * brix_rm to fail confusingly. If the defensive stat also fails, skip the
269 * entry and warn rather than guessing. */
270 is_dir = 0;
271 if (ents[i].have_stat) {
272 is_dir = (ents[i].st.flags & kXR_isDir) ? 1 : 0;
273 } else {
274 brix_statinfo si;
275 brix_status sst;
276 brix_status_clear(&sst);
277 if (brix_stat(c, rchild, &si, &sst) == 0) {
278 is_dir = (si.flags & kXR_isDir) ? 1 : 0;
279 } else {
280 fprintf(stderr, "xrdcp: --delete: cannot stat %s, skipping\n", rchild);
281 continue;
282 }
283 }
284 /* For dry-run, only print; do not call brix_rmtree with BRIX_RMTREE_DRYRUN
285 * because that would walk the remote tree (I/O), bloating the dry-run output.
286 * Instead, we guard the call itself so dry-run performs no remote I/O. */
287 if (o->dry_run) {
288 printf("[dry-run] delete %s\n", rchild);
289 } else if (is_dir) {
290 brix_status_clear(&st);
291 if (brix_rmtree(c, rchild, 0, NULL, NULL, &st) != 0) {
292 fprintf(stderr, "xrdcp: --delete: cannot remove %s: %s\n",
293 rchild, st.msg);
294 }
295 } else {
296 brix_status_clear(&st);
297 if (brix_rm(c, rchild, &st) != 0) {
298 fprintf(stderr, "xrdcp: --delete: cannot remove %s: %s\n",
299 rchild, st.msg);
300 }
301 }
302 }
303 free(ents);
304}

307/* Recurse a remote tree (rpath) under conn c into local lpath.
308 *
309 * WHAT: downloads every regular file under rpath into the mirrored local path,
310 * applying --exclude/--include filters and honoring --dry-run and --sync
311 * (size/mtime/cksum up-to-date skip per o->sync_cmp).
312 * WHY: threading `rel` (path relative to the copy root, "" at the top level)
313 * through every recursive call lets brix_copy_filter_match compare
314 * both the full relative path and its basename, so pattern semantics are
315 * consistent at any depth.
316 * HOW: skip mkdir in dry-run mode; build relc for each entry; apply filter
317 * then sync-skip then dry-run guard before calling copy_one_r2l; recurse
318 * with relc. The sync-skip runs before the dry-run print so that
319 * `-r --sync --dry-run` only lists files that would actually be copied. */
320int
321copy_tree_download(brix_conn *c, const char *rpath, const char *lpath,
322 const char *rel, const brix_copy_opts *o, brix_status *st)
323{
324 brix_dirent *ents = NULL;
325 size_t n = 0, i;
326
327 if (!o->dry_run && local_mkdirs(lpath, 0755) != 0) {
328 brix_status_set(st, XRDC_ESOCK, errno, "mkdir %s: %s", lpath, strerror(errno));
329 return -1;
330 }
331 if (brix_dirlist(c, rpath, 1, &ents, &n, st) != 0) {
332 return -1;
333 }
334 for (i = 0; i < n; i++) {
335 char relc[XRDC_PATH_MAX];
336 char rc[XRDC_PATH_MAX], lc[XRDC_PATH_MAX];
337 if (ents[i].name[0] == '.'
338 && (ents[i].name[1] == '\0'
339 || (ents[i].name[1] == '.' && ents[i].name[2] == '\0'))) {
340 continue;
341 }
342 /* Build the path of this entry relative to the copy root so that
343 * brix_copy_filter_match can match at both full-rel and basename. */
344 if ((size_t) snprintf(relc, sizeof(relc), "%s%s%s",
345 rel, rel[0] ? "/" : "", ents[i].name) >= sizeof(relc)) {
346 brix_status_set(st, XRDC_EUSAGE, 0,
347 "recursive copy: rel path too long under %s", rpath);
348 free(ents);
349 return -1;
350 }
351 if ((size_t) snprintf(rc, sizeof(rc), "%s/%s", rpath, ents[i].name)
352 >= sizeof(rc)
353 || (size_t) snprintf(lc, sizeof(lc), "%s/%s", lpath, ents[i].name)
354 >= sizeof(lc)) {
355 brix_status_set(st, XRDC_EUSAGE, 0,
356 "recursive copy: path too long under %s", rpath);
357 free(ents);
358 return -1;
359 }
360 if (ents[i].have_stat && (ents[i].st.flags & kXR_isDir)) {
361 if (copy_tree_download(c, rc, lc, relc, o, st) != 0) {
362 free(ents);
363 return -1;
364 }
365 if (o->remove_source && !o->dry_run) {
366 brix_status rst;
367 brix_status_clear(&rst);
368 (void) brix_rmdir(c, rc, &rst);
369 }
370 } else {
371 /* For a symlink (kXR_other) the dirlist size is the lstat size — the
372 * LENGTH OF THE LINK TARGET PATH, not the bytes the server serves
373 * when it follows the link on open. Trusting it truncates the copy
374 * (a link to a 10-byte file named "two.txt" would copy 7 bytes), so
375 * read to EOF (expected = -1) for those; regular files keep their
376 * real size, whose short-read guard still catches truncation. */
377 int64_t expected = (ents[i].have_stat
378 && !(ents[i].st.flags & kXR_other))
379 ? ents[i].st.size : -1;
380 if (!brix_copy_filter_match(o, relc)) {
381 continue;
382 }
383 /* --sync: skip an up-to-date destination. Only when the remote
384 * side has trustworthy dirlist stat data (no kXR_other — a
385 * symlink's listed size is the link-target-path length, not the
386 * served bytes) AND the local side stats as a regular file; any
387 * undeterminable side falls through to the copy (data-loss rule).
388 * Runs before the dry-run print so -r --sync --dry-run only lists
389 * files that would actually be transferred. */
390 if (o->sync && ents[i].have_stat && !(ents[i].st.flags & kXR_other)) {
391 struct stat sb;
392 if (stat(lc, &sb) == 0 && S_ISREG(sb.st_mode)
393 && brix_sync_should_skip(o->sync_cmp,
394 (long long) ents[i].st.size,
395 (long long) ents[i].st.mtime,
396 (long long) sb.st_size,
397 (long long) sb.st_mtime)
398 && (o->sync_cmp != XRDC_SYNC_CKSUM
399 || sync_cksum_match(c, rc, lc, o))) {
400 continue; /* up-to-date — skip */
401 }
402 }
403 if (o->dry_run) {
404 printf("[dry-run] copy %s -> %s\n", rc, lc);
405 continue;
406 }
407 if (copy_one_r2l(c, rc, lc, expected, st) != 0) {
408 free(ents);
409 return -1;
410 }
411 if (o->remove_source && !o->dry_run) {
412 brix_status rst;
413 brix_status_clear(&rst);
414 (void) brix_rm(c, rc, &rst);
415 }
416 }
417 }
418 /* Mirror-delete pass: remove local entries that the remote no longer has.
419 * Runs only when --sync --delete is active; ents[] is still live here so
420 * the remote snapshot is coherent with the copies just done above. */
421 if (o->sync_delete) {
422 mirror_delete_local(lpath, ents, n, rel, o);
423 }
424 free(ents);
425 return 0;
426}

429/* Recurse a local tree (lpath) into a remote tree (rpath) under conn c.
430 *
431 * WHAT: uploads every regular file under lpath to the mirrored remote path,
432 * applying --exclude/--include filters and honoring --dry-run and --sync
433 * (size/mtime/cksum up-to-date skip per o->sync_cmp).
434 * WHY: same rel-threading rationale as copy_tree_download — filter patterns
435 * must see the full relative path so they behave consistently at depth.
436 * HOW: skip brix_mkdir in dry-run mode; build relc from de->d_name; apply
437 * filter then sync-skip then dry-run guard before copy_one_l2r; recurse
438 * with relc. Sync runs before dry-run so -r --sync --dry-run only lists
439 * files that would actually be transferred. */
440int
441copy_tree_upload(brix_conn *c, const char *lpath, const char *rpath,
442 const char *rel, const brix_copy_opts *o, brix_status *st)
443{
444 DIR *d;
445 struct dirent *de;
446 brix_status mst;
447
448 brix_status_clear(&mst);
449 if (!o->dry_run) {
450 (void) brix_mkdir(c, rpath, 0755, 1 /*parents*/, &mst); /* may already exist */
451 }
452 d = opendir(lpath);
453 if (d == NULL) {
454 brix_status_set(st, XRDC_ESOCK, errno, "opendir %s: %s", lpath, strerror(errno));
455 return -1;
456 }
457 while ((de = readdir(d)) != NULL) {
458 char relc[XRDC_PATH_MAX];
459 char lc[XRDC_PATH_MAX], rc[XRDC_PATH_MAX];
460 struct stat sb;
461 if (de->d_name[0] == '.'
462 && (de->d_name[1] == '\0'
463 || (de->d_name[1] == '.' && de->d_name[2] == '\0'))) {
464 continue;
465 }
466 /* Build the path of this entry relative to the copy root so that
467 * brix_copy_filter_match can match at both full-rel and basename. */
468 if ((size_t) snprintf(relc, sizeof(relc), "%s%s%s",
469 rel, rel[0] ? "/" : "", de->d_name) >= sizeof(relc)) {
470 brix_status_set(st, XRDC_EUSAGE, 0,
471 "recursive copy: path too long under %s", lpath);
472 closedir(d);
473 return -1;
474 }
475 if ((size_t) snprintf(lc, sizeof(lc), "%s/%s", lpath, de->d_name)
476 >= sizeof(lc)
477 || (size_t) snprintf(rc, sizeof(rc), "%s/%s", rpath, de->d_name)
478 >= sizeof(rc)) {
479 brix_status_set(st, XRDC_EUSAGE, 0,
480 "recursive copy: path too long under %s", lpath);
481 closedir(d);
482 return -1;
483 }
484 /* lstat (not stat) so symlinks are detected, not followed — a link to a
485 * parent directory would otherwise drive unbounded recursion. */
486 if (lstat(lc, &sb) != 0) {
487 continue; /* vanished between readdir and stat — skip */
488 }
489 if (S_ISLNK(sb.st_mode)) {
490 continue; /* skip symlinks (loop-safe; mirrors official -r default) */
491 }
492 if (S_ISDIR(sb.st_mode)) {
493 if (copy_tree_upload(c, lc, rc, relc, o, st) != 0) {
494 closedir(d);
495 return -1;
496 }
497 if (o->remove_source && !o->dry_run) {
498 (void) rmdir(lc);
499 }
500 } else if (S_ISREG(sb.st_mode)) {
501 if (!brix_copy_filter_match(o, relc)) {
502 continue;
503 }
504 /* --sync: skip an up-to-date remote destination. A failed remote
505 * stat (missing file, error) or a directory in the way falls
506 * through to the copy — never skip on an undeterminable compare.
507 * Runs before the dry-run print so -r --sync --dry-run only lists
508 * files that would actually be transferred. */
509 if (o->sync) {
510 brix_statinfo si;
511 brix_status sst;
512 brix_status_clear(&sst);
513 if (brix_stat(c, rc, &si, &sst) == 0 && !(si.flags & kXR_isDir)
514 && brix_sync_should_skip(o->sync_cmp,
515 (long long) sb.st_size,
516 (long long) sb.st_mtime,
517 (long long) si.size,
518 (long long) si.mtime)
519 && (o->sync_cmp != XRDC_SYNC_CKSUM
520 || sync_cksum_match(c, rc, lc, o))) {
521 continue; /* up-to-date — skip */
522 }
523 }
524 if (o->dry_run) {
525 printf("[dry-run] copy %s -> %s\n", lc, rc);
526 continue;
527 }
528 if (copy_one_l2r(c, lc, rc, o, st) != 0) {
529 closedir(d);
530 return -1;
531 }
532 if (o->remove_source && !o->dry_run) {
533 (void) unlink(lc);
534 }
535 }
536 }
537 closedir(d);
538 /* Mirror-delete pass: remove remote entries that the local tree no longer
539 * has. Runs only when --sync --delete is active; lists the remote
540 * directory here (post-upload) so the remote snapshot is fresh. */
541 if (o->sync_delete) {
542 mirror_delete_remote(c, rpath, lpath, rel, o);
543 }
544 return 0;
545}

548/* Build the recursive copy's destination root from the source path, using
549 * rsync-style trailing-slash semantics:
550 *
551 * src WITHOUT a trailing slash ("dir") -> NEST the tree under the source's
552 * last path component: <dst>/<basename(dir)>/... This matches stock
553 * `xrdcp -r <dir> <dst>` and avoids silently merging two differently-named
554 * source trees into one flattened destination.
555 * src WITH a trailing slash ("dir/") -> FLAT mirror: copy the CONTENTS of
556 * the source straight into <dst>/... (no extra basename level), exactly as
557 * `rsync dir/ dst/` does.
558 *
559 * A degenerate basename ('.', '/', or empty — e.g. the whole-export `//.`/`//`
560 * forms) has no meaningful name to nest under, so the destination is likewise
561 * used verbatim. Returns 0 on success, -1 if the composed path would overflow. */
562int
563recursive_dest_root(const char *dstdir, const char *srcpath,
564 char *out, size_t outsz)
565{
566 size_t len = strlen(srcpath);
567 const char *base;
568 size_t blen, dl, i;
569 const char *sep;
570
571 /* Trailing slash on the source => flat mirror (copy contents into <dst>). */
572 if (len > 0 && srcpath[len - 1] == '/') {
573 return ((size_t) snprintf(out, outsz, "%s", dstdir) >= outsz) ? -1 : 0;
574 }
575
576 base = srcpath;
577 for (i = len; i > 0; i--) {
578 if (srcpath[i - 1] == '/') { base = srcpath + i; break; }
579 }
580 blen = (size_t) (srcpath + len - base);
581
582 if (blen == 0 || (blen == 1 && base[0] == '.')) { /* nothing to nest */
583 return ((size_t) snprintf(out, outsz, "%s", dstdir) >= outsz) ? -1 : 0;
584 }
585 dl = strlen(dstdir);
586 sep = (dl > 0 && dstdir[dl - 1] == '/') ? "" : "/";
587 return ((size_t) snprintf(out, outsz, "%s%s%.*s", dstdir, sep,
588 (int) blen, base) >= outsz) ? -1 : 0;
589}

592/* Recursive copy entry: connect once, walk the source tree. Direction-aware. */
593int
594copy_recursive(const brix_url *su, const brix_url *du, int download,
595 const brix_copy_opts *o, const brix_opts *co, brix_status *st)
596{
597 brix_conn c;
598 int rc;
599 char destroot[XRDC_PATH_MAX];
600
601 /* Nest under the source basename (stock parity); see recursive_dest_root. */
602 if (recursive_dest_root(du->path, su->path, destroot, sizeof(destroot)) != 0) {
603 brix_status_set(st, XRDC_EUSAGE, 0,
604 "recursive copy: destination path too long");
605 return -1;
606 }
607
608 if (download) {
609 if (brix_connect(&c, su, co, st) != 0) { return -1; }
610 rc = copy_tree_download(&c, su->path, destroot, "", o, st);
611 /* Best-effort: remove the source root dir when the whole tree succeeded.
612 * The walker already removed each file and subdir, so the root will only
613 * succeed when nothing was filtered; failure is silently ignored. */
614 if (rc == 0 && o->remove_source && !o->dry_run) {
615 brix_status rst;
616 brix_status_clear(&rst);
617 (void) brix_rmdir(&c, su->path, &rst);
618 }
619 } else {
620 if (brix_connect(&c, du, co, st) != 0) { return -1; }
621 rc = copy_tree_upload(&c, su->path, destroot, "", o, st);
622 /* Best-effort: remove the local source root after a fully-successful
623 * upload walk (only succeeds if the directory is now empty). */
624 if (rc == 0 && o->remove_source && !o->dry_run) {
625 (void) rmdir(su->path);
626 }
627 }
628 brix_close(&c);
629 return rc;
630}

633/* web transfer (davs:// / http(s):// / s3://) — production GET/PUT over */
634/* the streaming HTTP client. Auth: WebDAV bearer token or S3 SigV4. */
635
636
637/* Build the auth header block for a web request into hdrs[] (may be empty for an
638 * anonymous endpoint). S3 → SigV4 (host signed as "host:port" to match the Host
639 * header we send); WebDAV/HTTP → Authorization: Bearer if a token is available.
640 *
641 * co carries the credential store (co->cred); when set the store is tried first
642 * for both the bearer token and S3 keys, falling back to opts/env on failure so
643 * env-sourced credentials behave identically to today. */
644int
645web_auth_headers(const brix_weburl *u, const char *method,
646 const brix_copy_opts *o, const brix_opts *co,
647 char *hdrs, size_t hdrsz, brix_status *st)
648{
649 hdrs[0] = '\0';
650 if (u->is_s3) {
651 const char *ak = (o && o->s3_access) ? o->s3_access : NULL;
652 const char *sk = (o && o->s3_secret) ? o->s3_secret : NULL;
653 const char *rg = (o && o->s3_region) ? o->s3_region : getenv("AWS_DEFAULT_REGION");
654 brix_cred_view sv;
655 char host[300], payhash[65];
656
657 /* Prefer the cred store for S3 keys when no explicit opts override. */
658 if ((ak == NULL || sk == NULL) && co != NULL && co->cred != NULL) {
659 if (brix_cred_acquire(co->cred, XRDC_CRED_S3KEYS, 0, &sv, st) == 0) {
660 if (ak == NULL) { ak = sv.s3_access; }
661 if (sk == NULL) { sk = sv.s3_secret; }
662 } else {
663 brix_status_clear(st);
664 }
665 }
666 /* Fall through to env when store not set or acquire failed. */
667 if (ak == NULL) { ak = getenv("AWS_ACCESS_KEY_ID"); }
668 if (sk == NULL) { sk = getenv("AWS_SECRET_ACCESS_KEY"); }
669
670 if (ak == NULL || sk == NULL) {
671 return 0; /* anonymous — server may permit unsigned access */
672 }
673 if (rg == NULL) { rg = "us-east-1"; }
674 /* A '?' would split path vs query in the server's canonical request but
675 * we sign the whole path as CanonicalURI — reject rather than mis-sign. */
676 if (strchr(u->path, '?') != NULL) {
677 brix_status_set(st, XRDC_EUSAGE, 0,
678 "s3: query strings in the URL are not supported");
679 return -1;
680 }
681 /* The SigV4 signed host MUST match the wire Host header byte-for-byte; that
682 * header brackets IPv6 literals ([::1]:9000), so sign the same form. */
683 brix_format_host_port(u->host, (uint16_t) u->port, host, sizeof(host));
684 /* UNSIGNED-PAYLOAD for every method: the body isn't folded into the
685 * signature (it streams), which both nginx-xrootd's S3 and real AWS accept. */
686 snprintf(payhash, sizeof(payhash), "UNSIGNED-PAYLOAD");
687 if (brix_s3_sign_v4(method, host, u->path, ak, sk, rg, payhash,
688 hdrs, hdrsz) != 0) {
689 brix_status_set(st, XRDC_EAUTH, 0, "s3: failed to build SigV4 signature");
690 return -1;
691 }
692 return 0;
693 }
694 {
695 const char *tok = (o && o->bearer) ? o->bearer : NULL;
696 brix_cred_view bv;
697
698 /* Prefer the cred store for the bearer token when no explicit opt override. */
699 if (tok == NULL && co != NULL && co->cred != NULL) {
700 if (brix_cred_acquire(co->cred, XRDC_CRED_BEARER, 0, &bv, st) == 0
701 && bv.token != NULL) {
702 tok = bv.token;
703 } else {
704 brix_status_clear(st);
705 }
706 }
707 /* Fall through to env when store not set or acquire failed. */
708 if (tok == NULL) { tok = getenv("BEARER_TOKEN"); }
709
710 if (tok != NULL && tok[0] != '\0') {
711 int n = snprintf(hdrs, hdrsz, "Authorization: Bearer %s\r\n", tok);
712 if (n < 0 || (size_t) n >= hdrsz) {
713 brix_status_set(st, XRDC_EUSAGE, 0, "bearer token too long");
714 return -1;
715 }
716 }
717 }
718 return 0;
719}

722int
723copy_web_download(const brix_weburl *su, const brix_url *du, int to_stdout,
724 const brix_copy_opts *o, const brix_opts *co, brix_status *st)
725{
726 char hdrs[8192];
727 char tmp[XRDC_PATH_MAX];
728 int outfd, status = 0, rc;
729 long long blen = 0;
730
731 if (web_auth_headers(su, "GET", o, co, hdrs, sizeof(hdrs), st) != 0) {
732 return -1;
733 }
734 if (to_stdout) {
735 rc = brix_http_download(su->host, su->port, su->tls, su->path,
736 hdrs[0] ? hdrs : NULL, co ? co->verify_host : 1,
737 co ? co->ca_dir : NULL, STDOUT_FILENO,
738 XRDC_WEB_TIMEOUT_MS, &status, &blen, st);
739 return rc;
740 }
741 /* Refuse to overwrite an existing destination unless -f. */
742 if (!(o && o->force) && access(du->path, F_OK) == 0) {
743 brix_status_set(st, XRDC_EUSAGE, 0,
744 "destination exists (use -f to overwrite): %s", du->path);
745 return -1;
746 }
747 /* Download to a temp sibling and atomically rename on success: a failed
748 * transfer must never truncate or delete a pre-existing destination. */
749 outfd = open_download_temp(du->path, tmp, sizeof(tmp), st);
750 if (outfd < 0) {
751 return -1;
752 }
753 rc = brix_http_download(su->host, su->port, su->tls, su->path,
754 hdrs[0] ? hdrs : NULL, co ? co->verify_host : 1,
755 co ? co->ca_dir : NULL, outfd, XRDC_WEB_TIMEOUT_MS,
756 &status, &blen, st);
757 close(outfd);
758 rc = atomic_dest_finish(tmp, du->path, rc, st);
759 if (rc != 0) {
760 return rc;
761 }
762 if (o && !o->silent) {
763 fprintf(stderr, "xrdcp: downloaded %lld bytes (HTTP %d)\n", blen, status);
764 }
765 return 0;
766}
