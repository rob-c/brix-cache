8/* Create LPATH and every missing parent directory (local `mkdir -p`).
9 *
10 * WHY: the recursive download nests the copied tree under the destination
11 * (recursive_dest_root → e.g. <dst>/<basename>), so the very first directory
12 * created can be two levels below an existing parent (<dst> itself may not yet
13 * exist). A single mkdir() would fail with ENOENT on that missing parent; the
14 * remote upload side already creates parents (brix_mkdir parents=1), so the
15 * local side must match. Idempotent: an existing component is not an error
16 * (deeper failures such as a non-directory in the path still surface ENOTDIR).
17 * Returns 0 on success, -1 with errno set on the first uncreatable component. */
18static int
19local_mkdirs(const char *lpath, mode_t mode)
20{
21 char buf[XRDC_PATH_MAX];
22 size_t n = strlen(lpath);
23 size_t i;
24
25 if (n == 0 || n >= sizeof(buf)) {
26 errno = (n == 0) ? EINVAL : ENAMETOOLONG;
27 return -1;
28 }
29 memcpy(buf, lpath, n + 1);
30 for (i = 1; i < n; i++) { /* each intermediate component */
31 if (buf[i] != '/') {
32 continue;
33 }
34 buf[i] = '\0';
35 if (mkdir(buf, mode) != 0 && errno != EEXIST) {
36 return -1;
37 }
38 buf[i] = '/';
39 }
40 if (mkdir(buf, mode) != 0 && errno != EEXIST) { /* the leaf */
41 return -1;
42 }
43 return 0;
44}

47/* 1 = NAME is "." or ".." (the self/parent entries every directory walk must
48 * skip); 0 otherwise. Shared by every tree walker and mirror pass below so
49 * the dot test is written exactly once. */
50static int
51dirent_is_dot(const char *name)
52{
53 return name[0] == '.'
54 && (name[1] == '\0' || (name[1] == '.' && name[2] == '\0'));
55}

58/* Compose the copy-root-relative path of a child entry: "<rel>/<name>", with
59 * no leading '/' when rel is "" (the top level). Every walker threads this
60 * form into brix_copy_filter_match so patterns behave identically at any
61 * depth. Returns 0 on success, -1 when the result would not fit. */
62static int
63rel_join(const char *rel, const char *name, char *out, size_t outsz)
64{
65 return ((size_t) snprintf(out, outsz, "%s%s%s",
66 rel, rel[0] ? "/" : "", name) >= outsz) ? -1 : 0;
67}

70/* Compose "<dir>/<name>" into out. Returns 0 on success, -1 when the result
71 * would not fit (callers report their own context-specific message). */
72static int
73path_join(const char *dir, const char *name, char *out, size_t outsz)
74{
75 return ((size_t) snprintf(out, outsz, "%s/%s", dir, name) >= outsz)
76 ? -1 : 0;

80/* 1 = the remote file (rpath, over the walker's open conn) and the local file
81 * (lpath) have the same digest under o->sync_cksum_algo; 0 = differ or
82 * undeterminable.
83 *
84 * WHAT: the XRDC_SYNC_CKSUM content check for the recursive walkers.
85 * WHY: brix_sync_should_skip is only the size gate for CKSUM mode; the byte
86 * comparison happens here. Any failure (bad algo, query error, open or
87 * read error) returns 0 so the walker COPIES — an undeterminable
88 * comparison must never skip (silent skip on error is data loss).
89 * HOW: server digest via kXR_Qcksum on the already-open conn (no reconnect
90 * per file), local digest via brix_cksum_fd; case-insensitive hex compare. */
91static int
92sync_cksum_match(brix_conn *c, const char *rpath, const char *lpath,
93 const brix_copy_opts *o)
94{
95 const char *algo = o->sync_cksum_algo ? o->sync_cksum_algo : "adler32";
96 brix_cksum_algo a;
97 brix_status st;
98 char rhex[132], lhex[132];
99 int fd, rc;
100
101 brix_status_clear(&st);
102 if (brix_cksum_algo_parse(algo, &a) != 0) {
103 return 0;
104 }
105 if (brix_query_cksum(c, rpath, algo, rhex, sizeof(rhex), &st) != 0) {
106 return 0;
107 }
108 fd = open(lpath, O_RDONLY);
109 if (fd < 0) {
110 return 0;
111 }
112 rc = brix_cksum_fd(fd, a, lhex, sizeof(lhex), &st);
113 close(fd);
114 if (rc != 0) {
115 return 0;
116 }
117 return strcasecmp(rhex, lhex) == 0;
118}

121/* Recursively remove a local tree (mirror-delete helper).
122 *
123 * WHAT: post-order deletion of a local path without following symlinks.
124 * WHY: called by mirror_delete_local to prune extra local paths that have no
125 * corresponding remote entry. lstat (not stat) is used throughout so a
126 * symlink inside the destination tree can never escape the root and delete
127 * unrelated data.
128 * HOW: lstat to classify; dirs → opendir + recurse each child + rmdir; files
129 * → unlink. Returns 0 on success, -1 with errno set on first failure. */
130static int
131local_rmtree(const char *path)
132{
133 struct stat sb;
134
135 if (lstat(path, &sb) != 0) {
136 return -1;
137 }
138 if (S_ISDIR(sb.st_mode)) {
139 DIR *d = opendir(path);
140 struct dirent *de;
141 if (d == NULL) { return -1; }
142 while ((de = readdir(d)) != NULL) {
143 char child[XRDC_PATH_MAX];
144 if (dirent_is_dot(de->d_name)) {
145 continue;
146 }
147 if (path_join(path, de->d_name, child, sizeof(child)) != 0) {
148 closedir(d);
149 errno = ENAMETOOLONG;
150 return -1;
151 }
152 if (local_rmtree(child) != 0) { closedir(d); return -1; }
153 }
154 closedir(d);
155 return rmdir(path);
156 }
157 return unlink(path);
158}

161/* 1 = NAME appears in the remote listing ents[0..nents); 0 otherwise.
162 * The tombstone presence probe for the download-direction mirror pass. */
163static int
164name_in_ents(const brix_dirent *ents, size_t nents, const char *name)
165{
166 size_t i;
167
168 for (i = 0; i < nents; i++) {
169 if (strcmp(ents[i].name, name) == 0) {
170 return 1;
171 }
172 }
173 return 0;

177/* Delete one extra LOCAL path for the mirror pass (or print it in dry-run).
178 *
179 * WHY: shared tombstone-removal tail of mirror_delete_local. lstat first:
180 * for regular files (and symlinks) use unlink directly rather than the full
181 * recursive rmtree; local_rmtree is only needed for directories. lstat
182 * avoids following a symlink that might point outside the tree. */
183static void
184mirror_local_remove(const char *lchild, const brix_copy_opts *o)
185{
186 struct stat csb;
187
188 if (o->dry_run) {
189 printf("[dry-run] delete %s\n", lchild);
190 return;
191 }
192 if (lstat(lchild, &csb) == 0 && S_ISDIR(csb.st_mode)) {
193 if (local_rmtree(lchild) != 0) {
194 fprintf(stderr, "xrdcp: --delete: cannot remove %s: %s\n",
195 lchild, strerror(errno));
196 }
197 } else if (unlink(lchild) != 0) {
198 fprintf(stderr, "xrdcp: --delete: cannot remove %s: %s\n",
199 lchild, strerror(errno));
200 }
201}

204/* Prune local entries under lpath that are absent from the remote listing.
205 *
206 * WHAT: the download-direction mirror pass for xrdcp -r --sync --delete.
207 * WHY: after the download loop the ents[] array holds every name the remote
208 * directory contains; a local child absent from that set is an extra
209 * that must be removed to keep the local tree a faithful mirror of the
210 * remote. Excluded paths are outside the sync scope and must never be
211 * deleted (the filter gate enforces this).
212 * HOW: opendir(lpath); for each local entry, scan ents[] by name; if absent
213 * AND brix_copy_filter_match passes → mirror_local_remove (dry-run
214 * print, or delete via local_rmtree for dirs / unlink for files). */
215static void
216mirror_delete_local(const char *lpath, const brix_dirent *ents, size_t nents,
217 const char *rel, const brix_copy_opts *o)
218{
219 DIR *d;
220 struct dirent *de;
221
222 d = opendir(lpath);
223 if (d == NULL) {
224 fprintf(stderr, "xrdcp: --delete: opendir %s: %s\n", lpath, strerror(errno));
225 return;
226 }
227 while ((de = readdir(d)) != NULL) {
228 char relc[XRDC_PATH_MAX];
229 char lchild[XRDC_PATH_MAX];
230
231 if (dirent_is_dot(de->d_name)) {
232 continue;
233 }
234 /* Keep the local entry if the remote listing still has this name. */
235 if (name_in_ents(ents, nents, de->d_name)) {
236 continue;
237 }
238 /* Build relative and absolute paths for the filter check and deletion. */
239 if (rel_join(rel, de->d_name, relc, sizeof(relc)) != 0
240 || path_join(lpath, de->d_name, lchild, sizeof(lchild)) != 0) {
241 fprintf(stderr, "xrdcp: --delete: path too long under %s\n", lpath);
242 continue;
243 }
244 /* Excluded entries are outside the sync scope — never delete them. */
245 if (!brix_copy_filter_match(o, relc)) {
246 continue;
247 }
248 mirror_local_remove(lchild, o);
249 }
250 closedir(d);
251}

254/* Resolve whether the remote entry E (absolute path rchild) is a directory.
255 *
256 * WHY: if dirlist provided no stat info (have_stat=0), do a defensive stat to
257 * resolve the type before deletion. Without this, a stat-less directory
258 * would be treated as a file, causing brix_rm to fail confusingly. If the
259 * defensive stat also fails, warn and return -1 (caller skips the entry)
260 * rather than guessing. Returns 0 with *is_dir set on success. */
261static int
262remote_entry_is_dir(brix_conn *c, const brix_dirent *e, const char *rchild,
263 int *is_dir)
264{
265 brix_statinfo si;
266 brix_status sst;
267
268 if (e->have_stat) {
269 *is_dir = (e->st.flags & kXR_isDir) ? 1 : 0;
270 return 0;
271 }
272 brix_status_clear(&sst);
273 if (brix_stat(c, rchild, &si, &sst) == 0) {
274 *is_dir = (si.flags & kXR_isDir) ? 1 : 0;
275 return 0;
276 }
277 fprintf(stderr, "xrdcp: --delete: cannot stat %s, skipping\n", rchild);
278 return -1;
279}

282/* Delete one extra REMOTE path for the mirror pass (or print it in dry-run).
283 *
284 * WHY: shared tombstone-removal tail of mirror_delete_remote. For dry-run,
285 * only print; do not call brix_rmtree with BRIX_RMTREE_DRYRUN because that
286 * would walk the remote tree (I/O), bloating the dry-run output. Instead,
287 * the call itself is guarded so dry-run performs no remote I/O. */
288static void
289mirror_remote_remove(brix_conn *c, const char *rchild, int is_dir,
290 const brix_copy_opts *o)
291{
292 brix_status st;
293
294 if (o->dry_run) {
295 printf("[dry-run] delete %s\n", rchild);
296 return;
297 }
298 brix_status_clear(&st);
299 if (is_dir) {
300 if (brix_rmtree(c, rchild, 0, NULL, NULL, &st) != 0) {
301 fprintf(stderr, "xrdcp: --delete: cannot remove %s: %s\n",
302 rchild, st.msg);
303 }
304 } else if (brix_rm(c, rchild, &st) != 0) {
305 fprintf(stderr, "xrdcp: --delete: cannot remove %s: %s\n",
306 rchild, st.msg);
307 }
308}

311/* Prune remote entries under rpath that have no local counterpart in lpath.
312 *
313 * WHAT: the upload-direction mirror pass for xrdcp -r --sync --delete.
314 * WHY: after the upload loop the local tree is authoritative; a remote entry
315 * with no local counterpart must be removed so the remote mirrors the
316 * local. Excluded paths are outside the sync scope and must never be
317 * deleted.
318 * HOW: brix_dirlist(rpath) for the remote snapshot; for each remote entry,
319 * lstat the local counterpart; if absent AND filter passes →
320 * mirror_remote_remove (dry-run print, or brix_rmtree for dirs /
321 * brix_rm for files, type resolved by remote_entry_is_dir).
322 * GUARD: this pass treats the LOCAL tree as authoritative, so it MUST NOT run
323 * alongside --remove-source (which deletes each local source after upload):
324 * the sources would already be gone here and every uploaded remote copy
325 * would be purged. xrdcp rejects that flag combination at parse time
326 * (see the --delete/--remove-source guard in main), so this function is
327 * only ever reached with the local tree intact. */
328static void
329mirror_delete_remote(brix_conn *c, const char *rpath, const char *lpath,
330 const char *rel, const brix_copy_opts *o)
331{
332 brix_dirent *ents = NULL;
333 brix_status st;
334 size_t n = 0, i;
335
336 brix_status_clear(&st);
337 if (brix_dirlist(c, rpath, 1, &ents, &n, &st) != 0) {
338 fprintf(stderr, "xrdcp: --delete: dirlist %s: %s\n", rpath, st.msg);
339 return;
340 }
341 for (i = 0; i < n; i++) {
342 char relc[XRDC_PATH_MAX];
343 char rchild[XRDC_PATH_MAX];
344 char lchild[XRDC_PATH_MAX];
345 struct stat sb;
346 int is_dir;
347
348 if (dirent_is_dot(ents[i].name)) {
349 continue;
350 }
351 if (rel_join(rel, ents[i].name, relc, sizeof(relc)) != 0
352 || path_join(rpath, ents[i].name, rchild, sizeof(rchild)) != 0
353 || path_join(lpath, ents[i].name, lchild, sizeof(lchild)) != 0) {
354 fprintf(stderr, "xrdcp: --delete: path too long under %s\n", rpath);
355 continue;
356 }
357 /* Keep the remote entry if the local counterpart exists. */
358 if (lstat(lchild, &sb) == 0) {
359 continue;
360 }
361 /* Excluded entries are outside the sync scope — never delete them. */
362 if (!brix_copy_filter_match(o, relc)) {
363 continue;
364 }
365 if (remote_entry_is_dir(c, &ents[i], rchild, &is_dir) != 0) {
366 continue;
367 }
368 mirror_remote_remove(c, rchild, is_dir, o);
369 }
370 free(ents);
371}

374/* 1 = the download of remote entry E (rc) may be skipped because the local
375 * destination (lc) is up to date under --sync; 0 = copy.
376 *
377 * WHY: only when the remote side has trustworthy dirlist stat data (no
378 * kXR_other — a symlink's listed size is the link-target-path length, not
379 * the served bytes) AND the local side stats as a regular file; any
380 * undeterminable side falls through to the copy (data-loss rule). Runs
381 * before the dry-run print so -r --sync --dry-run only lists files that
382 * would actually be transferred. */
383static int
384dl_sync_skip(const copy_walk_ctx *w, const brix_dirent *e,
385 const char *rc, const char *lc)
386{
387 struct stat sb;
388
389 if (!w->o->sync || !e->have_stat || (e->st.flags & kXR_other)) {
390 return 0;
391 }
392 return stat(lc, &sb) == 0 && S_ISREG(sb.st_mode)
393 && brix_sync_should_skip(w->o->sync_cmp,
394 (long long) e->st.size,
395 (long long) e->st.mtime,
396 (long long) sb.st_size,
397 (long long) sb.st_mtime)
398 && (w->o->sync_cmp != XRDC_SYNC_CKSUM
399 || sync_cksum_match(w->c, rc, lc, w->o));
400}

403/* Transfer one remote FILE entry E to the local path lc (download walk).
404 *
405 * HOW: filter → sync-skip → dry-run guard → copy_one_r2l → --remove-source.
406 * For a symlink (kXR_other) the dirlist size is the lstat size — the LENGTH
407 * OF THE LINK TARGET PATH, not the bytes the server serves when it follows
408 * the link on open. Trusting it truncates the copy (a link to a 10-byte
409 * file named "two.txt" would copy 7 bytes), so read to EOF (expected = -1)
410 * for those; regular files keep their real size, whose short-read guard
411 * still catches truncation. Returns 0 (copied or skipped) or -1 (fatal,
412 * w->st set by copy_one_r2l). */
413static int
414tree_dl_file(const copy_walk_ctx *w, const brix_dirent *e,
415 const char *relc, const char *rc, const char *lc)
416{
417 int64_t expected = (e->have_stat && !(e->st.flags & kXR_other))
418 ? e->st.size : -1;
419
420 if (!brix_copy_filter_match(w->o, relc)) {
421 return 0;
422 }
423 if (dl_sync_skip(w, e, rc, lc)) {
424 return 0; /* up-to-date — skip */
425 }
426 if (w->o->dry_run) {
427 printf("[dry-run] copy %s -> %s\n", rc, lc);
428 return 0;
429 }
430 if (copy_one_r2l(w->c, rc, lc, expected, w->st) != 0) {
431 return -1;
432 }
433 if (w->o->remove_source && !w->o->dry_run) {
434 brix_status rst;
435 brix_status_clear(&rst);
436 (void) brix_rm(w->c, rc, &rst);
437 }
438 return 0;
439}

442/* Handle one remote directory entry E of the download walk.
443 *
444 * HOW: skip dots; build the copy-root-relative path (relc) plus the remote
445 * and local child paths; directories recurse via copy_tree_download (then
446 * best-effort rmdir under --remove-source), everything else goes through
447 * tree_dl_file. Returns 0 to continue the walk, -1 fatal (w->st set). */
448static int
449tree_dl_entry(const copy_walk_ctx *w, const brix_dirent *e)
450{
451 char relc[XRDC_PATH_MAX];
452 char rc[XRDC_PATH_MAX], lc[XRDC_PATH_MAX];
453
454 if (dirent_is_dot(e->name)) {
455 return 0;
456 }
457 /* Build the path of this entry relative to the copy root so that
458 * brix_copy_filter_match can match at both full-rel and basename. */
459 if (rel_join(w->rel, e->name, relc, sizeof(relc)) != 0) {
460 brix_status_set(w->st, XRDC_EUSAGE, 0,
461 "recursive copy: rel path too long under %s", w->rpath);
462 return -1;
463 }
464 if (path_join(w->rpath, e->name, rc, sizeof(rc)) != 0
465 || path_join(w->lpath, e->name, lc, sizeof(lc)) != 0) {
466 brix_status_set(w->st, XRDC_EUSAGE, 0,
467 "recursive copy: path too long under %s", w->rpath);
468 return -1;
469 }
470 if (e->have_stat && (e->st.flags & kXR_isDir)) {
471 copy_walk_ctx cw = { w->c, rc, lc, relc, w->o, w->st };
472 if (copy_tree_download(&cw) != 0) {
473 return -1;
474 }
475 if (w->o->remove_source && !w->o->dry_run) {
476 brix_status rst;
477 brix_status_clear(&rst);
478 (void) brix_rmdir(w->c, rc, &rst);
479 }
480 return 0;
481 }
482 return tree_dl_file(w, e, relc, rc, lc);
483}

486/* Recurse a remote tree (rpath) under conn c into local lpath.
487 *
488 * WHAT: downloads every regular file under rpath into the mirrored local path,
489 * applying --exclude/--include filters and honoring --dry-run and --sync
490 * (size/mtime/cksum up-to-date skip per o->sync_cmp).
491 * WHY: threading `rel` (path relative to the copy root, "" at the top level)
492 * through every recursive call lets brix_copy_filter_match compare
493 * both the full relative path and its basename, so pattern semantics are
494 * consistent at any depth.
495 * HOW: skip mkdir in dry-run mode; list the remote directory; delegate each
496 * entry to tree_dl_entry (filter → sync-skip → dry-run → copy_one_r2l).
497 * The sync-skip runs before the dry-run print so that `-r --sync
498 * --dry-run` only lists files that would actually be copied. */
499int
500copy_tree_download(const copy_walk_ctx *w)
501{
502 brix_dirent *ents = NULL;
503 size_t n = 0, i;
504
505 if (!w->o->dry_run && local_mkdirs(w->lpath, 0755) != 0) {
506 brix_status_set(w->st, XRDC_ESOCK, errno, "mkdir %s: %s",
507 w->lpath, strerror(errno));
508 return -1;
509 }
510 if (brix_dirlist(w->c, w->rpath, 1, &ents, &n, w->st) != 0) {
511 return -1;
512 }
513 for (i = 0; i < n; i++) {
514 if (tree_dl_entry(w, &ents[i]) != 0) {
515 free(ents);
516 return -1;
517 }
518 }
519 /* Mirror-delete pass: remove local entries that the remote no longer has.
520 * Runs only when --sync --delete is active; ents[] is still live here so
521 * the remote snapshot is coherent with the copies just done above. */
522 if (w->o->sync_delete) {
523 mirror_delete_local(w->lpath, ents, n, w->rel, w->o);
524 }
525 free(ents);
526 return 0;
527}

588/* Handle one local directory entry NAME of the upload walk.
589 *
590 * HOW: skip dots; build the copy-root-relative path (relc) plus the local
591 * and remote child paths; lstat (not stat) so symlinks are detected, not
592 * followed — a link to a parent directory would otherwise drive unbounded
593 * recursion; symlinks are skipped (loop-safe; mirrors official -r default).
594 * Directories recurse via copy_tree_upload (then best-effort rmdir under
595 * --remove-source); regular files go through tree_ul_file. Returns 0 to
596 * continue the walk, -1 fatal (w->st set). */
597static int
598tree_ul_entry(const copy_walk_ctx *w, const char *name)
599{
600 char relc[XRDC_PATH_MAX];
601 char lc[XRDC_PATH_MAX], rc[XRDC_PATH_MAX];
602 struct stat sb;
603
604 if (dirent_is_dot(name)) {

608 * brix_copy_filter_match can match at both full-rel and basename. */

613 "recursive copy: path too long under %s", w->lpath);
614 return -1;

617 return 0; /* vanished between readdir and stat — skip */

620 return 0; /* skip symlinks (loop-safe; mirrors official -r default) */

622 if (S_ISDIR(sb.st_mode)) {

639/* Recurse a local tree (lpath) into a remote tree (rpath) under conn c.
640 *
641 * WHAT: uploads every regular file under lpath to the mirrored remote path,
642 * applying --exclude/--include filters and honoring --dry-run and --sync
643 * (size/mtime/cksum up-to-date skip per o->sync_cmp).
644 * WHY: same rel-threading rationale as copy_tree_download — filter patterns
645 * must see the full relative path so they behave consistently at depth.
646 * HOW: skip brix_mkdir in dry-run mode; delegate each readdir entry to
647 * tree_ul_entry (filter → sync-skip → dry-run → copy_one_l2r). Sync
648 * runs before dry-run so -r --sync --dry-run only lists files that

684/* Build the recursive copy's destination root from the source path, using
685 * rsync-style trailing-slash semantics:
686 *
687 * src WITHOUT a trailing slash ("dir") -> NEST the tree under the source's
688 * last path component: <dst>/<basename(dir)>/... This matches stock
689 * `xrdcp -r <dir> <dst>` and avoids silently merging two differently-named
690 * source trees into one flattened destination.
691 * src WITH a trailing slash ("dir/") -> FLAT mirror: copy the CONTENTS of
692 * the source straight into <dst>/... (no extra basename level), exactly as
693 * `rsync dir/ dst/` does.
694 *
695 * A degenerate basename ('.', '/', or empty — e.g. the whole-export `//.`/`//`
696 * forms) has no meaningful name to nest under, so the destination is likewise
697 * used verbatim. Returns 0 on success, -1 if the composed path would overflow. */
698int
699recursive_dest_root(const char *dstdir, const char *srcpath,
700 char *out, size_t outsz)
701{
702 size_t len = strlen(srcpath);
703 const char *base;
704 size_t blen, dl, i;
705 const char *sep;
706
707 /* Trailing slash on the source => flat mirror (copy contents into <dst>). */
708 if (len > 0 && srcpath[len - 1] == '/') {
709 return ((size_t) snprintf(out, outsz, "%s", dstdir) >= outsz) ? -1 : 0;
710 }
711
712 base = srcpath;
713 for (i = len; i > 0; i--) {
714 if (srcpath[i - 1] == '/') { base = srcpath + i; break; }
715 }
716 blen = (size_t) (srcpath + len - base);
717
718 if (blen == 0 || (blen == 1 && base[0] == '.')) { /* nothing to nest */
719 return ((size_t) snprintf(out, outsz, "%s", dstdir) >= outsz) ? -1 : 0;
720 }
721 dl = strlen(dstdir);
722 sep = (dl > 0 && dstdir[dl - 1] == '/') ? "" : "/";
723 return ((size_t) snprintf(out, outsz, "%s%s%.*s", dstdir, sep,
724 (int) blen, base) >= outsz) ? -1 : 0;
725}

728/* Recursive copy entry: connect once, walk the source tree. Direction-aware. */
729int
730copy_recursive(const copy_recurse_req *rq, brix_status *st)
731{
732 const brix_url *su = rq->su;
733 const brix_copy_opts *o = rq->o;
734 brix_conn c;
735 int rc;
736 char destroot[XRDC_PATH_MAX];
737
738 /* Nest under the source basename (stock parity); see recursive_dest_root. */
739 if (recursive_dest_root(rq->du->path, su->path, destroot,
740 sizeof(destroot)) != 0) {
741 brix_status_set(st, XRDC_EUSAGE, 0,
742 "recursive copy: destination path too long");
743 return -1;
744 }
745
746 if (rq->download) {
747 copy_walk_ctx w = { &c, su->path, destroot, "", o, st };
748 if (brix_connect(&c, su, rq->co, st) != 0) { return -1; }
749 rc = copy_tree_download(&w);
750 /* Best-effort: remove the source root dir when the whole tree succeeded.
751 * The walker already removed each file and subdir, so the root will only
752 * succeed when nothing was filtered; failure is silently ignored. */
753 if (rc == 0 && o->remove_source && !o->dry_run) {
754 brix_status rst;
755 brix_status_clear(&rst);
756 (void) brix_rmdir(&c, su->path, &rst);
757 }
758 } else {
759 copy_walk_ctx w = { &c, destroot, su->path, "", o, st };
760 if (brix_connect(&c, rq->du, rq->co, st) != 0) { return -1; }
761 rc = copy_tree_upload(&w);
762 /* Best-effort: remove the local source root after a fully-successful
763 * upload walk (only succeeds if the directory is now empty). */
764 if (rc == 0 && o->remove_source && !o->dry_run) {
765 (void) rmdir(su->path);
766 }
767 }
768 brix_close(&c);
769 return rc;
770}

777/* Resolve the S3 access/secret key pair for SigV4 signing.
778 *
779 * WHY: precedence is frozen — explicit opts first, then the credential store
780 * (co->cred) when set, then the AWS_* environment, so env-sourced credentials
781 * behave identically to today. A store-acquire failure is not an error: the
782 * status is cleared and resolution falls through to the environment. Either
783 * pointer may come back NULL (anonymous access — the caller decides). */
784static void
785s3_resolve_keys(const brix_copy_opts *o, const brix_opts *co,
786 const char **ak, const char **sk, brix_status *st)
787{
788 brix_cred_view sv;
789
790 *ak = (o && o->s3_access) ? o->s3_access : NULL;
791 *sk = (o && o->s3_secret) ? o->s3_secret : NULL;
792
793 /* Prefer the cred store for S3 keys when no explicit opts override. */
794 if ((*ak == NULL || *sk == NULL) && co != NULL && co->cred != NULL) {
795 if (brix_cred_acquire(co->cred, XRDC_CRED_S3KEYS, 0, &sv, st) == 0) {
796 if (*ak == NULL) { *ak = sv.s3_access; }
797 if (*sk == NULL) { *sk = sv.s3_secret; }
798 } else {
799 brix_status_clear(st);
800 }
801 }
802 /* Fall through to env when store not set or acquire failed. */
803 if (*ak == NULL) { *ak = getenv("AWS_ACCESS_KEY_ID"); }
804 if (*sk == NULL) { *sk = getenv("AWS_SECRET_ACCESS_KEY"); }
805}
806
807
808/* Build the S3 SigV4 Authorization block for a->u into hdrs[].
809 *
810 * HOW: resolve keys (opts → store → env; both missing = anonymous, empty
811 * hdrs); host is signed as "host:port" to match the wire Host header
812 * byte-for-byte (brackets IPv6 literals); UNSIGNED-PAYLOAD for every method
813 * because the body streams and is not folded into the signature (both
814 * nginx-xrootd's S3 and real AWS accept that). Returns 0/-1 (a->st set). */
815static int
816auth_hdr_s3(const web_auth_ctx *a, char *hdrs, size_t hdrsz)
817{
818 const brix_weburl *u = a->u;
819 const brix_copy_opts *o = a->o;
820 brix_status *st = a->st;
821 const char *ak, *sk;
822 const char *rg = (o && o->s3_region) ? o->s3_region
823 : getenv("AWS_DEFAULT_REGION");
824 char host[300], payhash[65];
825
826 s3_resolve_keys(o, a->co, &ak, &sk, st);
827 if (ak == NULL || sk == NULL) {
828 return 0; /* anonymous — server may permit unsigned access */
829 }
830 if (rg == NULL) { rg = "us-east-1"; }
831 /* A '?' would split path vs query in the server's canonical request but
832 * we sign the whole path as CanonicalURI — reject rather than mis-sign. */
833 if (strchr(u->path, '?') != NULL) {
834 brix_status_set(st, XRDC_EUSAGE, 0,
835 "s3: query strings in the URL are not supported");
836 return -1;
837 }
838 /* The SigV4 signed host MUST match the wire Host header byte-for-byte; that
839 * header brackets IPv6 literals ([::1]:9000), so sign the same form. */
840 brix_format_host_port(u->host, (uint16_t) u->port, host, sizeof(host));
841 /* UNSIGNED-PAYLOAD for every method: the body isn't folded into the
842 * signature (it streams), which both nginx-xrootd's S3 and real AWS accept. */
843 snprintf(payhash, sizeof(payhash), "UNSIGNED-PAYLOAD");
844 if (brix_s3_sign_v4(a->method, host, u->path, ak, sk, rg, payhash,
845 hdrs, hdrsz) != 0) {
846 brix_status_set(st, XRDC_EAUTH, 0, "s3: failed to build SigV4 signature");
847 return -1;
848 }
849 return 0;
850}

853/* Build the WebDAV/HTTP bearer Authorization header into hdrs[] (left empty
854 * for an anonymous endpoint).
855 *
856 * WHY: precedence is frozen — explicit opts first, then the credential store
857 * (co->cred) when set, then $BEARER_TOKEN; a store-acquire failure clears the
858 * status and falls through to the environment. Returns 0/-1 (st set). */
859static int
860auth_hdr_bearer(const brix_copy_opts *o, const brix_opts *co,
861 char *hdrs, size_t hdrsz, brix_status *st)
862{
863 const char *tok = (o && o->bearer) ? o->bearer : NULL;
864 brix_cred_view bv;
865
866 /* Prefer the cred store for the bearer token when no explicit opt override. */
867 if (tok == NULL && co != NULL && co->cred != NULL) {
868 if (brix_cred_acquire(co->cred, XRDC_CRED_BEARER, 0, &bv, st) == 0
869 && bv.token != NULL) {
870 tok = bv.token;
871 } else {
872 brix_status_clear(st);
873 }
874 }
875 /* Fall through to env when store not set or acquire failed. */
876 if (tok == NULL) { tok = getenv("BEARER_TOKEN"); }
877
878 if (tok != NULL && tok[0] != '\0') {
879 int n = snprintf(hdrs, hdrsz, "Authorization: Bearer %s\r\n", tok);
880 if (n < 0 || (size_t) n >= hdrsz) {
881 brix_status_set(st, XRDC_EUSAGE, 0, "bearer token too long");
882 return -1;
883 }
884 }
885 return 0;
886}

889/* Build the auth header block for a web request into hdrs[] (may be empty for an
890 * anonymous endpoint). S3 → SigV4 (host signed as "host:port" to match the Host
891 * header we send); WebDAV/HTTP → Authorization: Bearer if a token is available.
892 *
893 * a->co carries the credential store (co->cred); when set the store is tried
894 * first for both the bearer token and S3 keys, falling back to opts/env on
895 * failure so env-sourced credentials behave identically to today. */
896int
897web_auth_headers(const web_auth_ctx *a, char *hdrs, size_t hdrsz)
898{
899 hdrs[0] = '\0';

901 return auth_hdr_s3(a, hdrs, hdrsz);
902 }
903 return auth_hdr_bearer(a->o, a->co, hdrs, hdrsz, a->st);
904}

958 * transfer must never truncate or delete a pre-existing destination. */

964 close(io.outfd);
965 rc = atomic_dest_finish(tmp, du->path, rc, st);
966 if (rc != 0) {
967 return rc;
968 }
969 if (o && !o->silent) {
970 fprintf(stderr, "xrdcp: downloaded %lld bytes (HTTP %d)\n",
971 io.blen, io.status);
972 }
973 return 0;
