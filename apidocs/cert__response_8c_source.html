6/*
7 * cert_response.c — GSI round 1: reply to kXGC_certreq with kXGS_cert.
8 *
9 * Generates an ephemeral ffdhe2048 DH key (from the per-worker keypool, so
10 * keygen never runs on the event thread), advertises the keyable cipher/md
11 * lists, optionally signs the DH public (signed-DH) and the client rtag with the
12 * server RSA key, and emits the kXGS_cert bucket list (kXRS_puk/cipher,
13 * cipher_alg, md_alg, x509, main). Each function is documented at its
14 * definition below; the wire layout is described inline at the assembly sites.
15 */

17/*
18 * gsi_certreq_version — best-effort read of the client's advertised XrdSecgsi
19 * version from the certreq's kXRS_version bucket (a 4-byte int). Encoders
20 * differ on byte order (our client emits big-endian; stock XrdSut writes the
21 * host-order int), so accept whichever interpretation lands in the plausible
22 * XrdSecgsi range; 0 when absent/implausible (caller then treats as legacy).
23 */
24static uint32_t
25gsi_certreq_version(const u_char *payload, size_t plen)
26{
27 const u_char *vb = NULL;
28 size_t vlen = 0;
29 uint32_t raw, be;
30
31 if (gsi_find_bucket(payload, plen, (uint32_t) kXRS_version, &vb, &vlen) != 0
32 || vlen < 4)
33 {
34 return 0;
35 }
36 ngx_memcpy(&raw, vb, 4); /* host-order as written by stock XrdSut */
37 be = ntohl(raw); /* big-endian as written by our client */
38 if (be >= 10000 && be <= 99999) {
39 return be;
40 }
41 if (raw >= 10000 && raw <= 99999) {
42 return raw;
43 }
44 return 0;
45}

47/*
48 * gsi_certreq_clnt_opts — read the client's kXRS_clnt_opts (int32 kOpts* flags)
49 * from the certreq payload. XrdSut marshals it host-order; our client emits it
50 * big-endian (gsi_core.c). The flags are a small bitmask (< 256), so pick the
51 * interpretation whose value is a plausible flag word. Returns 0 if absent.
52 */
53static uint32_t
54gsi_certreq_clnt_opts(const u_char *payload, size_t plen)
55{
56 const u_char *ob = NULL;
57 size_t olen = 0;
58 uint32_t raw, be;
59
60 if (gsi_find_bucket(payload, plen, (uint32_t) kXRS_clnt_opts, &ob, &olen) != 0
61 || olen < 4)
62 {
63 return 0;
64 }
65 ngx_memcpy(&raw, ob, 4);
66 be = ntohl(raw);
67 if (be != 0 && be < 4096) {
68 return be;
69 }

76/*
77 * gsi_use_signed_dh — resolve the signed-DH decision for this handshake from
78 * the operator policy (conf->gsi_signed_dh) and the client's advertised
79 * version. REQUIRE always signs; AUTO signs only modern (>=10400) clients;
80 * OFF never signs. An unknown client version (0) is treated as legacy.
81 */
82static int
83gsi_use_signed_dh(ngx_uint_t policy, uint32_t client_version)
84{
85 if (policy == BRIX_GSI_SDH_REQUIRE) {
86 return 1;
87 }
88 if (policy == BRIX_GSI_SDH_AUTO) {
89 return client_version >= BRIX_GSI_VERS_DHSIGNED;
90 }
91 return 0;
92}

94/*
95 * Phase 52 (WS-A): build the advertised kXRS_cipher_alg list into out[] (NUL-
96 * terminated) from the configured preference (or the built-in default, aes-128-
97 * cbc first), keeping ONLY ciphers this build can actually key — so a client
98 * never selects one we cannot decrypt (e.g. bf-cbc without the OpenSSL legacy
99 * provider). Falls back to "aes-128-cbc" if nothing usable remains.

109 size_t out_n = 0;
110
111 out[0] = '\0';
112 while (*p && out_n + 1 < outsz) {
113 const char *start = p;
114 char name[24];
115 size_t n;
116 brix_gsi_cipher_t tmp;
117
118 while (*p && *p != ':') { p++; }
119 n = (size_t) (p - start);
120 if (n > 0 && n < sizeof(name)) {
121 ngx_memcpy(name, start, n);
122 name[n] = '\0';
123 if (brix_gsi_cipher_lookup(name, &tmp)
124 && out_n + n + 1 < outsz) {
125 if (out_n > 0) { out[out_n++] = ':'; }
126 ngx_memcpy(out + out_n, name, n);
127 out_n += n;
128 out[out_n] = '\0';
129 }
130 }
131 if (*p == ':') { p++; }
132 }
133 if (out_n == 0) {
134 ngx_cpystrn((u_char *) out, (u_char *) "aes-128-cbc", outsz);
135 }
