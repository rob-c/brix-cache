6/*
7 * cert_response.c — GSI round 1: reply to kXGC_certreq with kXGS_cert.
8 *
9 * Generates an ephemeral ffdhe2048 DH key (from the per-worker keypool, so
10 * keygen never runs on the event thread), advertises the keyable cipher/md
11 * lists, optionally signs the DH public (signed-DH) and the client rtag with the
12 * server RSA key, and emits the kXGS_cert bucket list (kXRS_puk/cipher,
13 * cipher_alg, md_alg, x509, main). Each function is documented at its
14 * definition below; the wire layout is described inline at the assembly sites.
15 */

17/*
18 * gsi_certreq_version — best-effort read of the client's advertised XrdSecgsi
19 * version from the certreq's kXRS_version bucket (a 4-byte int). Encoders
20 * differ on byte order (our client emits big-endian; stock XrdSut writes the
21 * host-order int), so accept whichever interpretation lands in the plausible
22 * XrdSecgsi range; 0 when absent/implausible (caller then treats as legacy).
23 */
24static uint32_t
25gsi_certreq_version(const u_char *payload, size_t plen)
26{
27 const u_char *vb = NULL;
28 size_t vlen = 0;
29 uint32_t raw, be;
30
31 if (gsi_find_bucket(payload, plen, (uint32_t) kXRS_version, &vb, &vlen) != 0
32 || vlen < 4)
33 {
34 return 0;
35 }
36 ngx_memcpy(&raw, vb, 4); /* host-order as written by stock XrdSut */
37 be = ntohl(raw); /* big-endian as written by our client */
38 if (be >= 10000 && be <= 99999) {
39 return be;
40 }
41 if (raw >= 10000 && raw <= 99999) {
42 return raw;
43 }
44 return 0;
45}

47/*
48 * gsi_certreq_clnt_opts — read the client's kXRS_clnt_opts (int32 kOpts* flags)
49 * from the certreq payload. XrdSut marshals it host-order; our client emits it
50 * big-endian (gsi_core.c). The flags are a small bitmask (< 256), so pick the
51 * interpretation whose value is a plausible flag word. Returns 0 if absent.
52 */
53static uint32_t
54gsi_certreq_clnt_opts(const u_char *payload, size_t plen)
55{
56 const u_char *ob = NULL;
57 size_t olen = 0;
58 uint32_t raw, be;
59
60 if (gsi_find_bucket(payload, plen, (uint32_t) kXRS_clnt_opts, &ob, &olen) != 0
61 || olen < 4)
62 {
63 return 0;
64 }
65 ngx_memcpy(&raw, ob, 4);
66 be = ntohl(raw);
67 if (be != 0 && be < 4096) {
68 return be;
69 }
70 if (raw != 0 && raw < 4096) {
71 return raw;
72 }
73 return 0;
74}

76/*
77 * gsi_use_signed_dh — resolve the signed-DH decision for this handshake from
78 * the operator policy (conf->gsi_signed_dh) and the client's advertised
79 * version. REQUIRE always signs; AUTO signs only modern (>=10400) clients;
80 * OFF never signs. An unknown client version (0) is treated as legacy.
81 */
82static int
83gsi_use_signed_dh(ngx_uint_t policy, uint32_t client_version)
84{
85 if (policy == BRIX_GSI_SDH_REQUIRE) {
86 return 1;
87 }
88 if (policy == BRIX_GSI_SDH_AUTO) {
89 return client_version >= BRIX_GSI_VERS_DHSIGNED;
90 }
91 return 0;
92}

94/*
95 * Phase 52 (WS-A): build the advertised kXRS_cipher_alg list into out[] (NUL-
96 * terminated) from the configured preference (or the built-in default, aes-128-
97 * cbc first), keeping ONLY ciphers this build can actually key — so a client
98 * never selects one we cannot decrypt (e.g. bf-cbc without the OpenSSL legacy
99 * provider). Falls back to "aes-128-cbc" if nothing usable remains.
100 */
101static void
102gsi_build_cipher_alg_list(const ngx_stream_brix_srv_conf_t *conf,
103 char *out, size_t outsz)
104{
105 const char *src = (conf->gsi_ciphers.len > 0)
106 ? (const char *) conf->gsi_ciphers.data
107 : brix_gsi_cipher_default_list();
108 const char *p = src;
109 size_t out_n = 0;
110
111 out[0] = '\0';
112 while (*p && out_n + 1 < outsz) {
113 const char *start = p;
114 char name[24];
115 size_t n;
116 brix_gsi_cipher_t tmp;
117
118 while (*p && *p != ':') { p++; }
119 n = (size_t) (p - start);
120 if (n > 0 && n < sizeof(name)) {
121 ngx_memcpy(name, start, n);
122 name[n] = '\0';
123 if (brix_gsi_cipher_lookup(name, &tmp)
124 && out_n + n + 1 < outsz) {
125 if (out_n > 0) { out[out_n++] = ':'; }
126 ngx_memcpy(out + out_n, name, n);
127 out_n += n;
128 out[out_n] = '\0';
129 }
130 }
131 if (*p == ':') { p++; }
132 }
133 if (out_n == 0) {
134 ngx_cpystrn((u_char *) out, (u_char *) "aes-128-cbc", outsz);
135 }
136}

138/*
139 * gsi_cert_chunks_t — the computed byte chunks the kXGS_cert framer emits, one
140 * field per top-level bucket payload. Populated by the build/sign helpers and
141 * consumed by gsi_frame_cert_response() so the wire assembly takes one struct
142 * instead of a long parameter list; every field is a plain pointer+length view
143 * into c->pool memory (no ownership transfer). Zero-initialised at the call
144 * site so an unset chunk frames as an empty (len 0) payload deterministically.
145 */
146typedef struct {
147 uint32_t pub_type; /* kXRS_puk (unsigned) or kXRS_cipher */
148 const u_char *pub_data; /* DH-public bucket payload */
149 size_t pub_len;

151 size_t calg_len;

161/*
162 * gsi_record_handshake_opts — WHAT: resolve and record the two per-handshake
163 * policy decisions on the connection (signed-DH variant, client delegation
164 * opts) from the certreq payload + operator config. WHY: round 2
165 * (parse_x509.c) must agree on the signed-DH padding/IV, and a later delegation
166 * round must pick the flow the client advertised, so both are latched onto
167 * ctx->gsi here. HOW: apply the operator policy against the client's advertised
168 * version; a REQUIRE policy with no server key degrades to unsigned rather than
169 * failing the auth. Returns the resolved signed-DH flag; logs the delegation
170 * opts at INFO (byte-identical to the pre-split line).
171 */
172static int
173gsi_record_handshake_opts(brix_ctx_t *ctx, ngx_connection_t *c,
174 const ngx_stream_brix_srv_conf_t *conf)
175{
176 int signed_dh = gsi_use_signed_dh(conf->gsi_signed_dh,
177 gsi_certreq_version(ctx->recv.payload,
178 ctx->recv.cur_dlen));
179
180 if (signed_dh && conf->gsi_key == NULL) {
181 signed_dh = 0;
182 }
183 ctx->gsi.signed_dh = signed_dh;
184
185 /* §F6: capture the client's advertised delegation mode (kXRS_clnt_opts) so a
186 * later delegation round picks the flow the client actually supports —
187 * kOptsFwdPxy (forward) vs kOptsDlgPxy / kOptsSigReq (sign-request). */
188 ctx->gsi.clnt_opts = gsi_certreq_clnt_opts(ctx->recv.payload,
189 ctx->recv.cur_dlen);
190 ngx_log_error(NGX_LOG_INFO, c->log, 0,
191 "brix: GSI client delegation opts=0x%02xd (fwd=%d sign=%d dlg=%d)",
192 (unsigned) ctx->gsi.clnt_opts,
193 (ctx->gsi.clnt_opts & 0x2) ? 1 : 0,
194 (ctx->gsi.clnt_opts & 0x4) ? 1 : 0,
195 (ctx->gsi.clnt_opts & 0x1) ? 1 : 0);
196
197 return signed_dh;
198}

200/*
201 * gsi_build_dh_public — WHAT: take/generate an ephemeral ffdhe2048 DH key and
202 * build the kXRS_puk blob ("<PEM params>---BPUB---<pub hex>---EPUB--") into a
203 * c->pool buffer. WHY: this is the DH-public body the kXGS_cert advertises; the
204 * key is latched on ctx->gsi.dh_key for round-2 shared-secret derivation. HOW:
205 * pop a pre-generated key from the per-worker pool (keygen never runs on the
206 * event thread under a certreq burst) or inline-keygen on an empty pool; own the
207 * key on this connection. On any failure the DH key is freed and NGX_ERROR is
208 * returned. Returns NGX_OK with *puk_out / *puk_len_out set on success.
209 */
210static ngx_int_t
211gsi_build_dh_public(brix_ctx_t *ctx, ngx_connection_t *c,
212 const ngx_stream_brix_srv_conf_t *conf,
213 u_char **puk_out, size_t *puk_len_out)
214{
215 EVP_PKEY *dhkey = NULL;
216 BIGNUM *pub_bn = NULL;
217 char *pub_hex = NULL;
218 BIO *bio;
219 BUF_MEM *bptr;
220 char puk_buf[4096];
221 int puk_written;
222 u_char *puk_blob;
223 size_t puk_len;
224
225 if (!brix_gsi_keypool_pop(conf->common.thread_pool, c->log, &dhkey)) {
226 dhkey = brix_gsi_dh_keygen();
227 if (dhkey == NULL) {
228 return NGX_ERROR;
229 }
230 }
231
232 if (!EVP_PKEY_get_bn_param(dhkey, "pub", &pub_bn)) {
233 EVP_PKEY_free(dhkey);
234 return NGX_ERROR;
235 }
236 pub_hex = BN_bn2hex(pub_bn);
237 BN_free(pub_bn);
238 if (pub_hex == NULL) {
239 EVP_PKEY_free(dhkey);
240 return NGX_ERROR;
241 }
242
243 bio = BIO_new(BIO_s_mem());
244 if (bio == NULL) {
245 OPENSSL_free(pub_hex);
246 EVP_PKEY_free(dhkey);
247 return NGX_ERROR;
248 }
249 PEM_write_bio_Parameters(bio, dhkey);
250 ctx->gsi.dh_key = dhkey; /* ownership transfers to the connection */
251 BIO_get_mem_ptr(bio, &bptr);
252
253 puk_written = snprintf(puk_buf, sizeof(puk_buf),
254 "%.*s---BPUB---%s---EPUB--",
255 (int) bptr->length, bptr->data, pub_hex);
256 BIO_free(bio);
257 OPENSSL_free(pub_hex);
258
259 if (puk_written <= 0 || (size_t) puk_written >= sizeof(puk_buf)) {
260 return NGX_ERROR;
261 }
262 puk_len = (size_t) puk_written;
263
264 BRIX_PALLOC_OR_RETURN(puk_blob, c->pool, puk_len, NGX_ERROR);
265 ngx_memcpy(puk_blob, puk_buf, puk_len);
266
267 *puk_out = puk_blob;
268 *puk_len_out = puk_len;
269 return NGX_OK;
270}

272/*
273 * gsi_select_dh_public_bucket — WHAT: choose the DH-public bucket payload+type
274 * for the kXGS_cert. WHY: unsigned (default) advertises the bare Public() blob
275 * as kXRS_puk; signed-DH (>=10400) advertises the SAME blob RSA-signed with the
276 * server key (XrdCryptosslRSA::EncryptPrivate) as kXRS_cipher, which the client
277 * recovers via DecryptPublic against the kXRS_x509 cert we also send —
278 * authenticating the DH parameters in transit. HOW: on the signed path,
279 * RSA-encrypt-private into a c->pool buffer sized for the signature; on failure
280 * WARN and return NGX_ERROR. On entry *ch already carries the unsigned default
281 * (pub_data/pub_len = the bare puk blob); the signed path overwrites it and sets
282 * pub_type to kXRS_cipher, the unsigned path only stamps kXRS_puk. Returns
283 * NGX_OK on success.
284 */
285static ngx_int_t
286gsi_select_dh_public_bucket(ngx_connection_t *c,
287 const ngx_stream_brix_srv_conf_t *conf, int signed_dh,
288 gsi_cert_chunks_t *ch)
289{
290 if (signed_dh) {
291 size_t cap = ch->pub_len
292 + 2 * (size_t) EVP_PKEY_size(conf->gsi_key) + 64;
293 u_char *sig = ngx_palloc(c->pool, cap);
294 size_t pub_len = sig ? brix_gsi_rsa_encrypt_private(conf->gsi_key,
295 ch->pub_data, ch->pub_len, sig, cap) : 0;
296
297 if (pub_len == 0) {
298 ngx_log_error(NGX_LOG_WARN, c->log, 0,
299 "brix: GSI signed-DH: failed to sign DH public");
300 return NGX_ERROR;
301 }
302 ch->pub_data = sig;
303 ch->pub_len = pub_len;
304 ch->pub_type = (uint32_t) kXRS_cipher;
305 return NGX_OK;
306 }
307
308 ch->pub_type = (uint32_t) kXRS_puk;
309 return NGX_OK;
310}

312/*
313 * gsi_sign_client_rtag — WHAT: RSA-PKCS1 sign the client's random tag (kXRS_rtag
314 * inside the certreq's kXRS_main container) with the server key. WHY: the signed
315 * rtag proves possession of the server private key to the client and rides in the
316 * response's nested kXRS_main container. HOW: locate the client rtag; if present
317 * and the whole sign sequence succeeds, emit the signature into a c->pool buffer;
318 * any step failing leaves *rtag_out NULL / *rtag_len_out 0 (unsigned response,
319 * not an error — matches the pre-split fall-through). No return code: the
320 * out-params carry the result.
321 */
322static void
323gsi_sign_client_rtag(brix_ctx_t *ctx, ngx_connection_t *c,
324 const ngx_stream_brix_srv_conf_t *conf,
325 u_char **rtag_out, size_t *rtag_len_out)
326{
327 const u_char *main_data = NULL;
328 size_t main_dlen = 0;
329 const u_char *clnt_rtag = NULL;
330 size_t clnt_rtlen = 0;
331
332 *rtag_out = NULL;
333 *rtag_len_out = 0;
334
335 if (gsi_find_bucket(ctx->recv.payload, ctx->recv.cur_dlen,
336 (uint32_t) kXRS_main, &main_data, &main_dlen) == 0) {
337 gsi_find_bucket(main_data, main_dlen,
338 (uint32_t) kXRS_rtag, &clnt_rtag, &clnt_rtlen);
339 }
340
341 if (clnt_rtag && clnt_rtlen > 0) {
342 EVP_PKEY_CTX *sctx = EVP_PKEY_CTX_new(conf->gsi_key, NULL);
343
344 if (sctx) {
345 size_t slen = (size_t) EVP_PKEY_size(conf->gsi_key);
346 u_char *signed_rtag = ngx_palloc(c->pool, slen);
347
348 if (signed_rtag
349 && EVP_PKEY_sign_init(sctx) > 0
350 && EVP_PKEY_CTX_set_rsa_padding(sctx, RSA_PKCS1_PADDING) > 0
351 && EVP_PKEY_sign(sctx, signed_rtag, &slen,
352 clnt_rtag, clnt_rtlen) > 0)
353 {
354 *rtag_out = signed_rtag;
355 *rtag_len_out = slen;
356 }
357 EVP_PKEY_CTX_free(sctx);
358 }
359 }
360}

362/*
363 * gsi_build_main_container — WHAT: build the nested kXRS_main container the
364 * kXGS_cert carries as a single top-level kXRS_main bucket. WHY: it is NOT a
365 * flat field — it re-encapsulates its own protocol header plus an inner bucket
366 * list. Inner wire layout (network byte order throughout):
367 * "gsi\0" protocol name (4 bytes, NUL-padded)
368 * kXGS_cert opcode (uint32, 4 bytes)
369 * [ kXRS_signed_rtag bucket ] optional: type(4) + len(4) + signature
370 * kXRS_none terminator (uint32, 4 bytes)
371 * HOW: reserve the fixed 12 bytes (name + opcode + terminator) plus, when the
372 * rtag was signed, the 8-byte bucket header and signature; then walk mp
373 * byte-by-byte to emit the layout. Returns NGX_OK with *buf_out / *len_out set.
374 */
375static ngx_int_t
376gsi_build_main_container(ngx_connection_t *c,
377 const u_char *signed_rtag, size_t signed_rtag_len,
378 u_char **buf_out, size_t *len_out)
379{
380 size_t main_len = 4 + 4 + 4;
381 u_char *main_buf;
382 u_char *mp;
383
384 if (signed_rtag_len > 0) {
385 main_len += 4 + 4 + signed_rtag_len;
386 }
387
388 BRIX_PALLOC_OR_RETURN(main_buf, c->pool, main_len, NGX_ERROR);
389
390 mp = main_buf;
391 mp[0] = 'g';
392 mp[1] = 's';
393 mp[2] = 'i';
394 mp[3] = '\0';
395 mp += 4;
396 *(uint32_t *) mp = htonl(kXGS_cert);
397 mp += 4;
398 if (signed_rtag_len > 0) {
399 *(uint32_t *) mp = htonl(kXRS_signed_rtag);
400 mp += 4;
401 *(uint32_t *) mp = htonl((uint32_t) signed_rtag_len);
402 mp += 4;
403 ngx_memcpy(mp, signed_rtag, signed_rtag_len);
404 mp += signed_rtag_len;
405 }
406 *(uint32_t *) mp = htonl(kXRS_none);
407
408 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
409 "brix: kXGS_cert signed rtag=%uz bytes main_len=%uz",
410 signed_rtag_len, main_len);
411
412 *buf_out = main_buf;
413 *len_out = main_len;
414 return NGX_OK;
415}

417/*
418 * gsi_frame_cert_response — WHAT: assemble and queue the kXGS_cert response frame
419 * from the computed chunks. WHY: this is the single wire-emission site — the
420 * top-level bucket list (kXRS_puk/cipher, kXRS_cipher_alg, kXRS_md_alg, kXRS_x509,
421 * kXRS_main, kXRS_none) wrapped in the kXR_authmore server header. HOW: sum the
422 * body length, allocate the c->pool frame, write the response header, then walk p
423 * byte-by-byte emitting each bucket (type, len, payload) in network byte order;
424 * byte layout is frozen and identical to the pre-split assembly. Returns the
425 * brix_queue_response() result (NGX_OK on queued).
426 */
427static ngx_int_t
428gsi_frame_cert_response(brix_ctx_t *ctx, ngx_connection_t *c,
429 const gsi_cert_chunks_t *ch)
430{
431 size_t body_len = 4 + 4
432 + 4 + 4 + ch->pub_len
433 + 4 + 4 + ch->calg_len
434 + 4 + 4 + ch->malg_len
435 + 4 + 4 + ch->cert_len
436 + 4 + 4 + ch->main_len
437 + 4;
438 size_t total = XRD_RESPONSE_HDR_LEN + body_len;
439 u_char *buf;
440 u_char *p;
441
442 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
443
444 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_authmore,
445 (uint32_t) body_len, (ServerResponseHdr *) buf);
446
447 p = buf + XRD_RESPONSE_HDR_LEN;
448
449 p[0] = 'g';
450 p[1] = 's';
451 p[2] = 'i';
452 p[3] = '\0';
453 p += 4;
454 *(uint32_t *) p = htonl(kXGS_cert);
455 p += 4;
456
457 *(uint32_t *) p = htonl(ch->pub_type);
458 p += 4;
459 *(uint32_t *) p = htonl((uint32_t) ch->pub_len);
460 p += 4;
461 ngx_memcpy(p, ch->pub_data, ch->pub_len);
462 p += ch->pub_len;
463
464 *(uint32_t *) p = htonl(kXRS_cipher_alg);
465 p += 4;
466 *(uint32_t *) p = htonl((uint32_t) ch->calg_len);
467 p += 4;
468 ngx_memcpy(p, ch->cipher_alg, ch->calg_len);
469 p += ch->calg_len;
470
471 *(uint32_t *) p = htonl(kXRS_md_alg);
472 p += 4;
473 *(uint32_t *) p = htonl((uint32_t) ch->malg_len);
474 p += 4;
475 ngx_memcpy(p, ch->md_alg, ch->malg_len);
476 p += ch->malg_len;
477
478 *(uint32_t *) p = htonl(kXRS_x509);
479 p += 4;
480 *(uint32_t *) p = htonl((uint32_t) ch->cert_len);
481 p += 4;
482 ngx_memcpy(p, ch->cert_pem, ch->cert_len);
483 p += ch->cert_len;
484
485 *(uint32_t *) p = htonl(kXRS_main);
486 p += 4;
487 *(uint32_t *) p = htonl((uint32_t) ch->main_len);
488 p += 4;
489 ngx_memcpy(p, ch->main_buf, ch->main_len);
490 p += ch->main_len;
491
492 *(uint32_t *) p = htonl(kXRS_none);
493
494 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, c->log, 0,
495 "brix: kXGS_cert sent cert_len=%uz puk_len=%uz main_len=%uz",
496 ch->cert_len, ch->puk_len, ch->main_len);
497
498 return brix_queue_response(ctx, c, buf, total);
499}

501/*
502 * Respond to kXGC_certreq with kXGS_cert. Orchestrates the four
503 * equal-behavior phases: latch handshake opts, build the DH-public body, sign
504 * the client rtag into the nested kXRS_main container, and frame+queue the
505 * response. Wire bytes are frozen — see the per-helper doc blocks.
506 */
507
508ngx_int_t
509brix_gsi_send_cert(brix_ctx_t *ctx, ngx_connection_t *c)
510{
511 ngx_stream_brix_srv_conf_t *conf;
512 gsi_cert_chunks_t ch = {0};
513 char cipher_alg[96];
514 u_char *puk_blob = NULL;
515 size_t puk_len = 0;
516 u_char *signed_rtag = NULL;
517 size_t signed_rtag_len = 0;
518 int signed_dh;
519
520 conf = ngx_stream_get_module_srv_conf(ctx->session,
521 ngx_stream_brix_module);
522
523 if (conf->gsi_cert_pem == NULL || conf->gsi_cert_pem_len == 0) {
524 return NGX_ERROR;
525 }
526 ch.cert_pem = conf->gsi_cert_pem;
527 ch.cert_len = conf->gsi_cert_pem_len;
528
529 /* Advertise the kXRS_cipher_alg list (config preference or default, keyable
530 * ciphers only). See gsi_build_cipher_alg_list(). */
531 gsi_build_cipher_alg_list(conf, cipher_alg, sizeof(cipher_alg));
532 ch.cipher_alg = cipher_alg;
533 ch.calg_len = ngx_strlen(cipher_alg);
534
535 ch.md_alg = "sha256:sha1";
536 ch.malg_len = strlen(ch.md_alg);
537
538 signed_dh = gsi_record_handshake_opts(ctx, c, conf);
539
540 if (gsi_build_dh_public(ctx, c, conf, &puk_blob, &puk_len) != NGX_OK) {
541 return NGX_ERROR;
542 }
543 ch.puk_len = puk_len;
544 ch.pub_data = puk_blob; /* unsigned default; signed path overwrites */
545 ch.pub_len = puk_len;
546
547 if (gsi_select_dh_public_bucket(c, conf, signed_dh, &ch) != NGX_OK) {
548 return NGX_ERROR;
549 }
550
551 gsi_sign_client_rtag(ctx, c, conf, &signed_rtag, &signed_rtag_len);
552
553 if (gsi_build_main_container(c, signed_rtag, signed_rtag_len,
554 (u_char **) &ch.main_buf, &ch.main_len)
555 != NGX_OK)
556 {
557 return NGX_ERROR;
558 }
559
560 return gsi_frame_cert_response(ctx, c, &ch);
561}
