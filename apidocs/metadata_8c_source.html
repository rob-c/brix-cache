4/*
5 * WHAT: kXR_QStats, kXR_Qxattr, kXR_QFinfo, kXR_QFSinfo, kXR_Qvisa, kXR_Qopaque, kXR_Qopaquf, kXR_Qopaqug — metadata and plugin-style query handlers.
6 * QStats returns XML-formatted server statistics (connections, bytes, uptime); Qxattr lists extended attributes on a file path;
7 * QFinfo returns "0" as placeholder; QFSinfo delegates to space handler; Qvisa/opaques return FSctl/fctl unsupported responses
8 * matching reference XRootD behavior since nginx-xrootd does not embed the XrdOfs plugin layer.
9 *
10 * WHY: These queries provide server observability (stats), filesystem metadata (xattr, finfo), and extension hooks (visa/opaques) that clients use
11 * to discover server capabilities and file properties. QStats XML matches reference format for monitoring dashboards; xattr returns oss.* key-value
12 * pairs including file type, size, timestamps, and user.U.* extended attributes for HEP data provenance tracking. Opaque queries return unsupported
13 * to maintain protocol compatibility with clients that send FSctl/fctl requests expecting a consistent response shape.
14 *
15 * HOW: Three shared static helpers handle common patterns: arg_missing() logs + sends kXR_ArgMissing error; fsctl_unsupported()/fctl_unsupported()
16 * log + send kXR_Unsupported for plugin operations. payload_equals() compares wire payload against expected text with null-termination handling.
17 * Public APIs follow security chain (extract_path → resolve_path → authdb → vo_acl) where applicable, then delegate to stat/listxattr/getxattr syscalls
18 * or return placeholder/unsupported responses. QStats reads metrics struct + local socket address for port extraction; xattr builds oss.* key-value string
19 * with listxattr iteration filtering user.U.* prefixes.
20 */

46static ngx_flag_t
47brix_query_payload_equals(brix_ctx_t *ctx, const char *text)
48{
49 size_t len, text_len;
50
51 if (ctx->recv.payload == NULL) {
52 return 0;
53 }
54
55 len = ctx->recv.cur_dlen;
56 if (len > 0 && ctx->recv.payload[len - 1] == '\0') {
57 len--;
58 }
59
60 text_len = strlen(text);
61 return (len == text_len
62 && ngx_memcmp(ctx->recv.payload, text, text_len) == 0);

66 * connections, bytes rx/tx, timestamp, listening port). */
67ngx_int_t
68brix_query_stats(brix_ctx_t *ctx, ngx_connection_t *c)
69{
70 char resp[1024];
71 int port = 0;
72 long conns_active = 0, conns_total = 0;
73 long bytes_in = 0, bytes_out = 0;
74 time_t now = time(NULL);
75 int n;
76
77 if (ctx->metrics) {
78 port = (int) ctx->metrics->port;
79 conns_active = (long) ctx->metrics->connections_active;
80 conns_total = (long) ctx->metrics->connections_total;
81 bytes_in = (long) ctx->metrics->bytes_rx_total;
82 bytes_out = (long) ctx->metrics->bytes_tx_total;
83 }
84
85 if (port == 0 && c->local_sockaddr != NULL) {
86 if (c->local_sockaddr->sa_family == AF_INET) {
87 struct sockaddr_in *sin =
88 (struct sockaddr_in *) c->local_sockaddr;
89 port = (int) ntohs(sin->sin_port);
90 } else if (c->local_sockaddr->sa_family == AF_INET6) {
91 struct sockaddr_in6 *sin6 =
92 (struct sockaddr_in6 *) c->local_sockaddr;
93 port = (int) ntohs(sin6->sin6_port);
94 }
95 }
96
97 n = snprintf(resp, sizeof(resp) - 1,
98 "<statistics id=\"xrootd\" ver=\"" BRIX_SERVER_VERSION "\""
99 " tos=\"%ld\" pgm=\"" BRIX_SERVER_NAME "\">"
100 "<stats id=\"info\"><host>localhost</host><port>%d</port>"
101 "<name>" BRIX_SERVER_NAME "</name></stats>"
102 "<stats id=\"link\"><num>%ld</num><tot>%ld</tot>"
103 "<in>%ld</in><out>%ld</out><ctime>0</ctime>"
104 "<ltime>0</ltime><sfps>0</sfps></stats>"
105 "</statistics>",
106 (long) now, port,
107 conns_active, conns_total,
108 bytes_in, bytes_out);
109
110 brix_log_access(ctx, c, "QUERY", "-", "stats", 1, 0, NULL, 0);
111 BRIX_OP_OK(ctx, BRIX_OP_QUERY_STATS);
112 return brix_send_ok(ctx, c, resp, (uint32_t) (n + 1));
113}

115/* brix_query_xattr — kXR_Qxattr: list a path's extended attributes through the
116 * full security chain (extract → resolve → authdb → VO ACL → stat), returning the
117 * oss.* key-values plus any user.U.*-prefixed xattrs. */
118ngx_int_t
119brix_query_xattr(brix_ctx_t *ctx, ngx_connection_t *c,
120 ngx_stream_brix_srv_conf_t *conf)
121{
122 char pathbuf[BRIX_MAX_PATH + 1];
123 char full_path[PATH_MAX];
124 char resp[4096];
125 int pos = 0;
126 char raw_list[4096];
127 ssize_t list_sz;
128 brix_vfs_ctx_t vctx;
129 brix_vfs_stat_t vst;
130 char ftype;
131 char facc;
132
133 if (ctx->recv.cur_dlen == 0 || ctx->recv.payload == NULL) {
134 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_XATTR);
135 return brix_send_error(ctx, c, kXR_ArgMissing,
136 "xattr: path required");
137 }
138
139 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
140 pathbuf, sizeof(pathbuf), 1)) {
141 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_XATTR);
142 return brix_send_error(ctx, c, kXR_ArgInvalid, "invalid path");
143 }
144
145 /* phase74-fp: pathbuf is the request path, full_path the output buf. */
146 brix_beneath_full_path(conf->common.root_canon, pathbuf, /* NOLINT(readability-suspicious-call-argument) */
147 full_path, sizeof(full_path));
148
149 if (brix_auth_gate(ctx, c, BRIX_OP_QUERY_XATTR, "QUERY",
150 pathbuf, full_path, conf,
151 BRIX_AUTH_READ, 0) != NGX_OK) {
152 return ctx->write_rc;
153 }
154
155 /* Stat + xattr list/get all flow through the VFS (one ctx, confined to the
156 * export root). probe (follow) replaces the raw stat; the OP_STAT metric is
157 * suppressed (probe) so only the enclosing QUERY op is accounted. */
158 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
159 conf->common.root_canon, NULL, conf->common.allow_write,
160 0 /* is_tls */, NULL, full_path);
161
162 if (brix_vfs_probe(&vctx, 0 /* follow */, &vst) != NGX_OK) {
163 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_XATTR);
164 return brix_send_error(ctx, c, brix_kxr_from_errno(errno),
165 strerror(errno));
166 }
167
168 if (vst.is_regular) {
169 ftype = 'f';
170 } else if (vst.is_directory) {
171 ftype = 'd';
172 } else {
173 ftype = 'o';
174 }
175
176 facc = (vst.mode & S_IWUSR) ? 'w' : 'r';
177
178 pos = snprintf(resp, sizeof(resp) - 1,
179 "oss.cgroup=default&oss.type=%c&oss.used=%lld"
180 "&oss.mt=%ld&oss.ct=%ld&oss.at=%ld"
181 "&oss.u=*&oss.g=*&oss.fs=%c",
182 ftype, (long long) vst.size,
183 (long) vst.mtime, (long) vst.ctime, (long) vst.atime,
184 facc);
185
186 list_sz = brix_vfs_listxattr(&vctx, raw_list, sizeof(raw_list));
187 if (list_sz > 0) {
188 char *lp = raw_list;
189 char *lend = raw_list + list_sz;
190
191 while (lp < lend && pos < (int) sizeof(resp) - 256) {
192 size_t nlen = strlen(lp);
193
194 if (strncmp(lp, "user.U.", 7) == 0 && nlen > 7) {
195 char val[1024];
196 ssize_t vlen;
197
198 vlen = brix_vfs_getxattr(&vctx, lp, val, sizeof(val) - 1);
199 if (vlen >= 0) {
200 val[vlen] = '\0';
201 pos += snprintf(resp + pos, sizeof(resp) - pos - 1,
202 "&%s=%.*s", lp + 5, (int) vlen, val);
203 }
204 }
205
206 lp += nlen + 1;
207 }
208 }
209
210 brix_log_access(ctx, c, "QUERY", pathbuf, "xattr", 1, 0, NULL, 0);
211 BRIX_OP_OK(ctx, BRIX_OP_QUERY_XATTR);
212
213 return brix_send_ok(ctx, c, resp, (uint32_t) (pos + 1));
214}
