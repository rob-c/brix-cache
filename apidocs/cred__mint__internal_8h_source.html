1/*
2 * cred_mint_internal.h — internal helpers shared between cred_mint.c and
3 * cred_mint_cert.c.
4 *
5 * WHAT: The MINT_LOG NULL-tolerant log macro, the mint_ca_t / mint_material_t
6 * OpenSSL-object bundles, and the prototype for mint_build_cert() — the
7 * one EC-keygen + X509 build/sign entry point that crosses the
8 * cred_mint.c <-> cred_mint_cert.c split. Implementation detail of the
9 * cred_mint helper, NOT part of the public cred_mint.h surface.
10 *
11 * WHY: Splitting the cert-minting lifecycle into cred_mint_cert.c keeps each
12 * translation unit under the file-size guard while preserving the
13 * verbatim mint logic. Only mint_build_cert() and the two object-bundle
14 * structs cross the split; the CN sanitiser, subject builder, serial
15 * packer and field-setter stay static in cred_mint_cert.c.
16 *
17 * HOW: Prototypes + shared aggregates + one macro only; include cred_mint.h
18 * for the ngx types and the OpenSSL headers for the X509/EVP_PKEY
19 * struct members.
20 */

29/* ---- MINT_LOG ---------------------------------------------------------
30 * WHAT: ngx_log_error(), but tolerant of log==NULL (a no-op in that case).
31 * WHY: ngx_log_error() is a macro that dereferences `log` directly
32 * (`(log)->log_level >= level`) with no NULL guard of its own; callers
33 * of brix_cred_mint() in real request paths always carry ctx->log, but
34 * the standalone C unit test (tests/c/test_cred_mint.c) legitimately
35 * has no nginx log cycle available and passes NULL — guard every call
36 * site rather than change ngx_log_error's own contract. */
37#define MINT_LOG(level, log, err, ...) \
38 do { if ((log) != NULL) { ngx_log_error(level, log, err, __VA_ARGS__); } } while (0)

40/* ---- mint_ca_t / mint_material_t -----------------------------------------
41 * WHAT: mint_ca_t bundles the loaded mint-CA cert+key (the signer);
42 * mint_material_t bundles the minted leaf cert+key plus the CA cert that
43 * forms the trust-chain tail written into the PEM.
44 * WHY: Groups the OpenSSL objects that always travel together so the mint
45 * helpers stay at ≤5 parameters without adding globals — the pointers
46 * and their free ordering are unchanged, only their call shape.
47 * HOW: Plain aggregates filled on the stack; ownership stays with the caller
48 * (each helper documents which objects it allocates or borrows). */
49typedef struct {
50 X509 *cert;
51 EVP_PKEY *key;
52} mint_ca_t;
53
54typedef struct {
55 X509 *cert; /* minted leaf (owned) */
56 EVP_PKEY *key; /* minted leaf key (owned) */
57 X509 *ca_cert; /* borrowed: mint CA cert, PEM trust-chain tail */
58} mint_material_t;
59
60/* ---- mint_build_cert -----------------------------------------------------
61 * Generate a fresh EC P-256 keypair and a signed X509 whose subject is
62 * "/O=brix-minted/CN=<sanitized principal>", issuer = the mint CA's subject,
63 * notBefore=now, notAfter=now+ttl_secs, signed by ca->key with SHA-256. On
64 * success fills out->cert/out->key (leaves out->ca_cert untouched). Lives in
65 * cred_mint_cert.c; called from mint_generate() in cred_mint.c. */
66int mint_build_cert(const mint_ca_t *ca, const char *principal, int ttl_secs,
67 mint_material_t *out, ngx_log_t *log);

69#endif /* BRIX_FS_BACKEND_CRED_MINT_INTERNAL_H */
