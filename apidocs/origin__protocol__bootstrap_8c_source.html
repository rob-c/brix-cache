29/* Extract the "gsi" protocol's parameter substring from a login advert that may
30 * carry several "&P=<proto>,<parms>" entries (e.g. "&P=ztn,v:10000&P=gsi,v:10600,
31 * c:ssl,ca:HASH"). Returns a pointer INTO `parms` just past "gsi," (the v:/c:/ca:
32 * list brix_gsi_parse_parms wants), or NULL when gsi is not advertised. */
33static const char *
34cache_origin_gsi_parms(const char *parms, size_t plen)
35{
36 static const char needle[] = "gsi,";
37 size_t i;
38
39 if (parms == NULL || plen < sizeof(needle) - 1) {
40 return NULL;
41 }
42 for (i = 0; i + (sizeof(needle) - 1) <= plen; i++) {
43 if (ngx_strncmp(parms + i, needle, sizeof(needle) - 1) == 0) {
44 return parms + i + (sizeof(needle) - 1);
45 }
46 }
47 return NULL;
48}

73/* origin_expect_frame — read one origin reply into fr and require kXR_ok. WHAT:
74 * shared reply validator for the bootstrap wire steps. HOW: on transport/timeout
75 * failure brix_cache_read_response has already set the task error; on a non-ok
76 * status the body is freed and fail_msg becomes a kXR_ServerError. On success
77 * fr->body is the caller's to free. Returns 0 / -1. */
78static int
79origin_expect_frame(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
80 uint32_t max_dlen, const char *fail_msg, origin_frame_t *fr)
81{
82 fr->body = NULL;
83 if (brix_cache_read_response(t, oc, &fr->status, &fr->body,
84 &fr->dlen, max_dlen) != 0) {
85 return -1;
86 }
87 if (fr->status != kXR_ok) {
88 free(fr->body);
89 fr->body = NULL;
90 brix_cache_set_error(t, kXR_ServerError, 0, fail_msg);
91 return -1;
92 }
93 return 0;
94}

120/* origin_bs_protocol — bootstrap step 2: kXR_protocol negotiation, with the
121 * cleartext-handshake-then-TLS-upgrade a stock XRootD `roots://` origin demands.
122 *
123 * WHY: a TLS-for-ztn origin answers the CLEARTEXT kXR_protocol request with a
124 * kXR_gotoTLS advert and expects the client to upgrade THIS fd to TLS before
125 * kXR_login/auth — an immediate SSL_connect at byte 0 (the old behaviour) never
126 * reaches this exchange, so the origin was unreachable. When brix_cache_origin_tls
127 * is set we advertise kXR_ableTLS so the origin knows it may request the upgrade.
128 * HOW: send the protocol request on the connector streamid (advertising ableTLS
129 * when configured), inspect the reply flags, and:
130 * - gotoTLS + tls on → brix_cache_origin_tls_upgrade (every later frame rides
131 * TLS via io.c once oc->ssl is set)
132 * - gotoTLS + tls off → refuse (kXR_TLSRequired) rather than a mid-session surprise
133 * - no gotoTLS → stay cleartext
134 * Returns 0 / -1 (task error set). */
135static int
136origin_bs_protocol(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc)
137{
138 ClientProtocolRequest pr;
139 origin_frame_t fr;
140 static const uint8_t sid[2] = { 0, 1 }; /* cache-origin connector streamid */
141 uint8_t cap;
142 uint32_t flags = 0;
143
144 /* Advertise TLS capability so a TLS-requiring origin will answer with
145 * kXR_gotoTLS; with tls off we send no flag and behaviour is unchanged. */
146 cap = t->conf->cache_origin_tls ? (uint8_t) kXR_ableTLS : 0;
147 xrd_pack_protocol_request(&pr, sid, cap);
148
149 if (brix_cache_io_send(oc, &pr, sizeof(pr)) != 0) {
150 brix_cache_set_error(t, kXR_ServerError, errno,
151 "cache origin protocol write failed");
152 return -1;
153 }
154 if (origin_expect_frame(t, oc, sizeof(ServerProtocolBody),
155 "cache origin protocol negotiation failed",
156 &fr) != 0) {
157 return -1;
158 }
159
160 if (fr.dlen >= sizeof(ServerProtocolBody)) {
161 ServerProtocolBody *pb = (ServerProtocolBody *) fr.body;
162
163 flags = (uint32_t) ntohl(pb->flags);
164 }
165 free(fr.body);
166
167 if (flags & kXR_gotoTLS) {
168 if (!t->conf->cache_origin_tls) {
169 brix_cache_set_error(t, kXR_TLSRequired, 0,
170 "cache origin requires TLS; enable brix_cache_origin_tls");
171 return -1;
172 }
173 /* In-place upgrade of the connected fd BEFORE kXR_login/auth: the origin
174 * cert is verified against the configured origin CA (synth->trusted_ca)
175 * with hostname binding inside brix_cache_origin_tls_upgrade. */
176 if (brix_cache_origin_tls_upgrade(t, oc,
177 &t->conf->cache_origin_host) != 0)
178 {
179 return -1; /* upgrade helper set t's error */
180 }
181 }
182 return 0;
183}

185/* origin_bs_login — bootstrap step 3: anonymous kXR_login (user 'xrd', capver
186 * kXR_ver005). WHAT: sends the login and reads the reply into fr WITHOUT
187 * validating the status — kXR_authmore and an auth-advert-bearing kXR_ok are
188 * both legitimate here, so the caller owns the status decision (and fr->body).
189 * Returns 0 with fr populated, -1 on send/read failure (task error set). */
190static int
191origin_bs_login(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
192 origin_frame_t *fr)
193{
194 ClientLoginRequest lr;
195 static const uint8_t sid[2] = { 0, 1 }; /* cache-origin connector streamid */
196
197 xrd_pack_login_request(&lr, sid, (int32_t) ngx_pid, "xrd", kXR_ver005);
198
199 if (brix_cache_io_send(oc, &lr, sizeof(lr)) != 0) {

203 }
204
205 fr->body = NULL;
206 if (brix_cache_read_response(t, oc, &fr->status, &fr->body,
207 &fr->dlen, 4096) != 0) {
208 return -1;
209 }
210 return 0;
211}

213/* origin_bs_parse_advert — pure parse of the login auth advert into ad. WHAT:
214 * detects "&P=..." (needs_auth), the ztn/sss protocol names, and copies the gsi
215 * v:/c:/ca: list out of the (about-to-be-freed) body, stopping at the next "&P="
216 * entry so a co-advertised ztn block isn't mis-parsed. No I/O, no task state. */
217static void
218origin_bs_parse_advert(const u_char *parms, size_t plen,
219 origin_auth_advert_t *ad)
220{
221 const char *gp;
222
223 ad->needs_auth = (ngx_strlchr((u_char *) parms,
224 (u_char *) parms + plen, '=') != NULL);
225 ad->has_ztn = (ngx_strnstr((u_char *) parms, "ztn", plen) != NULL);
226 ad->has_sss = (ngx_strnstr((u_char *) parms, "sss", plen) != NULL);
227 ad->has_gsi = 0;
228 ad->gsi_parms[0] = '\0';
229
230 gp = cache_origin_gsi_parms((const char *) parms, plen);
231 if (gp != NULL) {
232 const char *amp = gp;
233 size_t end = (size_t) ((const char *) parms + plen - gp);
234 size_t i;
235
236 for (i = 0; i < end && amp[i] != '&'; i++) { /* find terminator */ }
237 if (i >= sizeof(ad->gsi_parms)) { i = sizeof(ad->gsi_parms) - 1; }
238 ngx_memcpy(ad->gsi_parms, gp, i);
239 ad->gsi_parms[i] = '\0';
240 ad->has_gsi = 1;
241 }
242}

265/* origin_bs_auth_dispatch — credential ladder for an auth-demanding advert.
266 * WHY the ordering: per-user overrides WIN over every static service credential
267 * (the session must carry the user's identity, never the service's), and x509 vs
268 * bearer are mutually exclusive — at most one will be non-empty. A per-user
269 * credential must NEVER fall back to a service credential: the operator
270 * provisioned it for a reason, and silent fallback would change the presented
271 * identity. Returns the auth round-trip result, or -1 with kXR_AuthFailed when
272 * no advertised protocol matches a configured credential. */
273static int
274origin_bs_auth_dispatch(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
275 const origin_auth_advert_t *ad)
276{
277 if (t->cred_x509_proxy[0] != '\0') {
278 if (ad->has_gsi) {
279 return brix_cache_origin_auth_gsi(t, oc, ad->gsi_parms,
280 t->cred_x509_proxy);
281 }
282 brix_cache_set_error(t, kXR_AuthFailed, 0,
283 "origin does not advertise gsi for the per-user credential");
284 return -1;
285 }
286 if (t->cred_bearer[0] != '\0') {
287 if (ad->has_ztn) {
288 ngx_str_t bt = {
289 ngx_strlen(t->cred_bearer),
290 (u_char *) t->cred_bearer
291 };
292 return brix_cache_origin_auth_ztn(t, oc, &bt);
293 }
294 brix_cache_set_error(t, kXR_AuthFailed, 0,
295 "origin does not advertise ztn for the per-user bearer credential");
296 return -1;
297 }
298 if (ad->has_ztn && t->conf->cache_origin_bearer.len > 0) {
299 return brix_cache_origin_auth_ztn(t, oc,
300 &t->conf->cache_origin_bearer);
301 }
302 if (ad->has_gsi && t->conf->cache_origin_x509_proxy.len > 0) {
303 return brix_cache_origin_auth_gsi(t, oc, ad->gsi_parms,
304 (const char *) t->conf->cache_origin_x509_proxy.data);
305 }
306 if (ad->has_sss && t->conf->cache_origin_sss_keytab.len > 0) {
307 return brix_cache_origin_auth_sss(t, oc,
308 (const char *) t->conf->cache_origin_sss_keytab.data);
309 }
310 brix_cache_set_error(t, kXR_AuthFailed, 0, origin_bs_auth_fail_msg(t));
311 return -1;
312}

341/* brix_cache_origin_bootstrap — three-phase XRootD connection bootstrap over a
342 * CLEARTEXT TCP socket: ClientInitHandShake → kXR_protocol negotiation → anonymous
343 * kXR_login (user 'xrd', capver kXR_ver005, streamid[1]=1). A stock XRootD
344 * `roots://` origin answers the cleartext kXR_protocol with a kXR_gotoTLS advert;
345 * origin_bs_protocol then upgrades THIS fd to TLS in place (verified against the
346 * origin CA) BEFORE login/auth, so the ztn/GSI credential exchange rides the
347 * encrypted channel. When the origin demands auth
348 * (kXR_authmore) and a bearer token is configured, a ztn kXR_auth completes the
349 * session. Every cache fill needs a valid session before reading. HOW: one
350 * static helper per wire step (origin_bs_handshake/_protocol/_login), then the
351 * login status decides between the advert-driven credential dispatch and the
352 * advert-less kXR_authmore fallback. Returns 0 on success, -1 on any phase
353 * failure. */
354int
355brix_cache_origin_bootstrap(brix_cache_fill_t *t,
356 brix_cache_origin_conn_t *oc)
357{
358 origin_frame_t fr;
359
360 if (origin_bs_handshake(t, oc) != 0) {
361 return -1;
362 }
363 if (origin_bs_protocol(t, oc) != 0) {
364 return -1;
365 }
366 if (origin_bs_login(t, oc, &fr) != 0) {
367 return -1;
368 }
369
370 /* A kXR_ok login on an AUTHENTICATED origin still carries an auth advert:
371 * body = sessid(16) + "&P=<proto>,..." (anonymous origins send only the 16-byte
372 * sessid). So a kXR_ok with a "&P=" parameter block means the session is NOT yet
373 * authenticated — present the configured bearer via ztn (§14/C-3). kXR_authmore
374 * is the mid-protocol variant; handle it the same way. */
375 if ((fr.status == kXR_ok || fr.status == kXR_authmore)
376 && fr.dlen > BRIX_SESSION_ID_LEN)
377 {
378 origin_auth_advert_t ad;
379
380 origin_bs_parse_advert(fr.body + BRIX_SESSION_ID_LEN,
381 fr.dlen - BRIX_SESSION_ID_LEN, &ad);
382 free(fr.body);
383
384 if (ad.needs_auth) {
385 return origin_bs_auth_dispatch(t, oc, &ad);
386 }
387 return 0;
388 }
389 free(fr.body);
390
391 if (fr.status == kXR_authmore) {
392 return origin_bs_authmore_fallback(t, oc);
393 }
394 if (fr.status != kXR_ok) {
395 brix_cache_set_error(t, kXR_ServerError, 0,
396 "cache origin anonymous login failed");
397 return -1;
398 }
399
400 return 0;
401}
