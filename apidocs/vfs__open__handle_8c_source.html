1/*
2 * vfs_open_handle.c — VFS handle lifecycle, close, and the brix_vfs_file_*
3 * accessors.
4 *
5 * WHAT: Implements brix_vfs_close() and every read-only accessor over an open
6 * brix_vfs_file_t: fd / sd_obj / pread / sendfile-fd / can-sendfile /
7 * backend-name / path / size / mtime / from_cache / file_stat, plus the
8 * phase-71 memfd sendfile-proxy materialiser.
9 *
10 * WHY: Once brix_vfs_open() (vfs_open.c) hands back a confined, already-fstat'd
11 * handle, everything else is pure handle-state access and teardown. Keeping
12 * it separate from the open cascade keeps both files under the size cap and
13 * isolates the "given a handle" surface from the "build a handle" surface.
14 *
15 * HOW: Accessors read cached metadata captured at adopt time; brix_vfs_file_stat()
16 * answers from that cache on a read-only handle (stat_current) and otherwise
17 * re-stats through the backend's fstat slot. brix_vfs_file_sendfile_fd()
18 * delegates the zero-copy decision to the backend and, for a fd-less
19 * CAP_MEMFILE backend, materialises a handle-owned memfd once.
20 * brix_vfs_close() releases any memfd then the descriptor through the
21 * backend's close slot.
22 */
23#include "vfs_internal.h"
24#include "vfs_backend_registry.h"
25#include <sys/mman.h> /* memfd_create (phase-71 step 2 memfd sendfile proxy) */
26#include "core/compat/log_diag.h"

28ngx_int_t
29brix_vfs_close(brix_vfs_file_t *fh, ngx_log_t *log)
30{
31 if (fh == NULL) {
32 return NGX_OK;
33 }
34
35 /* phase-71 step 2: release a materialised memfd sendfile proxy, if any. This
36 * runs before the fd-based early-out below because a CAP_MEMFILE backend has
37 * obj.fd == NGX_INVALID_FILE yet may still own a memfd. */
38 if (fh->memfd != NGX_INVALID_FILE) {
39 (void) ngx_close_file(fh->memfd);
40 fh->memfd = NGX_INVALID_FILE;
41 }
42
43 if (fh->obj.driver == NULL || fh->obj.fd == NGX_INVALID_FILE) {
44 return NGX_OK;
45 }
46
47 /* Release the descriptor through the backend's close slot; the driver
48 * marks obj.fd invalid. The VFS keeps the error-log wrapper. */
49 if (fh->obj.driver->close(&fh->obj) != NGX_OK) {
50 BRIX_DIAG_ERR(log != NULL ? log : fh->log, ngx_errno,
51 "xrootd[disk]: close failed for \"%s\"",
52 "a deferred write error surfaced at close — typically the "
53 "filesystem filled up (ENOSPC) or the device returned an I/O error",
54 "check free space and dmesg for disk errors; the file may be "
55 "incomplete, so the client's write should be treated as failed",
56 fh->path != NULL ? fh->path : "-");
57 return NGX_ERROR;
58 }
59
60 return NGX_OK;
61}

109/* brix_vfs_memfile_materialize — phase-71 step 2 memfd sendfile proxy.
110 *
111 * WHAT: For a CAP_MEMFILE backend that exposes no kernel fd (obj.fd invalid,
112 * read_sendfile_fd declines), pread the whole object into an anonymous
113 * memfd ONCE and cache it on fh->memfd, so the VFS can hand every backend
114 * a uniform seekable fd for the sendfile / file-backed serve path.
115 * WHY: Removes the last backend-identity branch in the serve path: callers stop
116 * special-casing "no fd → build a memory buffer myself" and use one fd path.
117 * HOW: memfd_create + a pread→write loop through the driver's worker-safe pread
118 * slot; the fd is owned by the handle and closed in brix_vfs_close. Returns
119 * the cached fd on repeat calls. NGX_INVALID_FILE on any failure (the caller
120 * then falls back to its legacy memory-backed path — no behaviour change). */
121static ngx_fd_t
122brix_vfs_memfile_materialize(brix_vfs_file_t *fh)
123{
124 brix_sd_obj_t obj;
125 ngx_fd_t fd;
126 off_t off;
127 u_char buf[65536];
128
129 if (fh->memfd != NGX_INVALID_FILE) {
130 return fh->memfd; /* already materialised */
131 }
132
133 brix_vfs_handle_sd_obj(fh, &obj);
134 if (obj.driver == NULL || obj.driver->pread == NULL
135 || !(brix_sd_caps(obj.inst) & BRIX_SD_CAP_MEMFILE))
136 {
137 return NGX_INVALID_FILE;
138 }
139
140 fd = (ngx_fd_t) memfd_create("brix-vfs-memfile", MFD_CLOEXEC);
141 if (fd == NGX_INVALID_FILE) {
142 return NGX_INVALID_FILE;
143 }
144
145 for (off = 0; off < fh->size; /* advanced below */) {
146 size_t want = (size_t) ngx_min((off_t) sizeof(buf), fh->size - off);
147 ssize_t n = obj.driver->pread(&obj, buf, want, off);
148
149 if (n <= 0) {
150 (void) ngx_close_file(fd);
151 return NGX_INVALID_FILE;
152 }
153 if (write(fd, buf, (size_t) n) != n) {
154 (void) ngx_close_file(fd);
155 return NGX_INVALID_FILE;
156 }
157 off += n;
158 }
159
160 if (lseek(fd, 0, SEEK_SET) == (off_t) -1) {
161 (void) ngx_close_file(fd);
162 return NGX_INVALID_FILE;
163 }
164
165 fh->memfd = fd;
166 return fd;
167}

169/* The fd only when the backend elects to back a zero-copy transfer of the whole
170 * object, else NGX_INVALID_FILE. The decision is delegated to the backend's
171 * read_sendfile_fd slot (want_zerocopy=1: the HTTP serve helper applies the
172 * TLS/cleartext choice itself). For a fd-less CAP_MEMFILE backend the VFS
173 * materialises a handle-owned memfd (phase-71 step 2) so the serve path is a
174 * uniform seekable fd for every backend. The contract gate for callers that
175 * build a sendfile / file-backed response. */
176ngx_fd_t
177brix_vfs_file_sendfile_fd(const brix_vfs_file_t *fh)
178{
179 ngx_fd_t fd;
180
181 if (fh == NULL) {
182 return NGX_INVALID_FILE;
183 }
184 fd = brix_vfs_handle_sendfile_fd(fh, 0, (size_t) fh->size, 1);
185 if (fd != NGX_INVALID_FILE) {
186 return fd;
187 }
188 /* fh is const by contract, but the memfd cache is an internal materialisation
189 * that does not change the observable file — safe to fill lazily here. */
190 return brix_vfs_memfile_materialize((brix_vfs_file_t *) fh);
191}

237ngx_int_t
238brix_vfs_file_stat(const brix_vfs_file_t *fh, brix_vfs_stat_t *stat_out)
239{
240 brix_sd_stat_t st;
241 brix_sd_obj_t obj;
242
243 if (fh == NULL || stat_out == NULL
244 || (fh->obj.fd == NGX_INVALID_FILE && fh->obj.driver == NULL))
245 {
246 /* A driver-backed handle (object/remote backend) has no kernel fd; it
247 * answers from cached metadata or the driver's fstat slot below. */
248 errno = EINVAL;
249 return NGX_ERROR;
250 }
251
252 /*
253 * phase-45 W2/R1: when the metadata cached at adopt time is authoritative,
254 * answer from it and skip a redundant fstat(2). This is the common read
255 * path (S3/WebDAV GET open the fd then immediately stat it). stat_current is
256 * set by adopt_fd only for read-only handles, whose file cannot change
257 * through them; a writable handle has it clear and always takes a live fstat.
258 */
259 if (fh->stat_current) {
260 ngx_memzero(stat_out, sizeof(*stat_out));
261 stat_out->size = fh->size;
262 stat_out->mtime = fh->mtime;
263 stat_out->ctime = fh->ctime;
264 stat_out->mode = (ngx_uint_t) fh->mode;
265 stat_out->ino = fh->ino;
266 stat_out->is_directory = S_ISDIR(fh->mode) ? 1 : 0;
267 stat_out->is_regular = S_ISREG(fh->mode) ? 1 : 0;
268 return NGX_OK;
269 }
270
271 /* Live re-stat through the backend's fstat slot (not a direct fstat(2)). */
272 brix_vfs_handle_sd_obj(fh, &obj);
273 if (obj.driver->fstat == NULL || obj.driver->fstat(&obj, &st) != NGX_OK) {
274 return NGX_ERROR;
275 }
276
277 brix_vfs_sd_stat_to_vfs(&st, stat_out);
278 return NGX_OK;
279}
