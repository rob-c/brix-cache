1/*
2 * server_conf.c — server-block config lifecycle: allocate, merge, enable.
3 *
4 * create_srv_conf() allocates one srv_conf per server block with every field at
5 * an NGX_CONF_UNSET sentinel (or NULL), so the merge step can tell an omitted
6 * directive from one explicitly set. merge_srv_conf() applies nginx parent→child
7 * inheritance (ngx_conf_merge_* for scalars, brix_merge_arrays() for the
8 * VO/group/manager-map rule sets); it is a flat orchestrator that delegates each
9 * configuration area to a per-area entry point. enable() handles
10 * "xrootd on|off;", swapping in our session handler when on.
11 *
12 * Phase-79 file-size split: the per-area MERGE helpers were moved into three
13 * focused sibling files, keeping this file under the 500-line cap. The five
14 * area entry points the orchestrator calls are declared in
15 * server_conf_internal.h and defined in:
16 * server_conf_merge_security.c — identity/crypto + storage-plane merges
17 * server_conf_merge_cluster.c — TPC + cluster/session merges
18 * server_conf_merge_proxy_net.c — upstream/proxy + network-fault merge
19 * The create-side per-area initialisers stay here, beside the creator they serve.
20 */

261/*
262 * This function creates a new server-level configuration object for nginx.
263 * It allocates memory and initializes every field to an "unset" or NULL state,
264 * allowing the merge step later to distinguish between missing directives and
265 * explicitly configured values.
266 */
267void *
268ngx_stream_brix_create_srv_conf(ngx_conf_t *cf)
269{
270 ngx_stream_brix_srv_conf_t *conf;
271
272 /*
273 * nginx allocates one per-server config object during parsing and then
274 * merges parent/child scopes later. Start everything in an explicit
275 * "unset" or NULL state so the merge step can tell whether a directive
276 * was omitted or configured intentionally.
277 *
278 * Scalar fields use nginx's UNSET sentinels when they participate in merge
279 * logic; runtime-only objects start out NULL/invalid and are created later
280 * during postconfiguration once parsing has finished. The per-field init is
281 * split into one helper per configuration area, mirroring the merge helpers.
282 */
283 conf = ngx_pcalloc(cf->pool, sizeof(ngx_stream_brix_srv_conf_t));
284 if (conf == NULL) {
285 return NULL;
286 }
287
288 ngx_http_brix_shared_init(&conf->common);
289 brix_create_srv_security(conf);
290 brix_create_srv_storage(conf);
291 brix_create_srv_cluster(conf);
292 brix_create_srv_proxy_net(conf);
293
294 return conf;
295}

297/*
298 * Standard nginx parent→child scope inheritance. Delegates to one per-area entry
299 * point (each defined in a sibling server_conf_merge_*.c and declared in
300 * server_conf_internal.h), invoked in the original linear order so cross-area
301 * derivations still observe their already-merged inputs (e.g. writethrough's
302 * origin/decision in proxy_net depends on the cache fields settled in storage;
303 * CMS timeout derivation in cluster depends on the merged cms_interval). Any
304 * entry point returning NGX_CONF_ERROR aborts the merge (the error is already
305 * logged).
306 */
307char *
308ngx_stream_brix_merge_srv_conf(ngx_conf_t *cf, void *parent, void *child)
309{
310 ngx_stream_brix_srv_conf_t *prev = parent;
311 ngx_stream_brix_srv_conf_t *conf = child;
312
313 if (brix_merge_srv_security(cf, conf, prev) != NGX_CONF_OK) {
314 return NGX_CONF_ERROR;
315 }
316 if (brix_merge_srv_storage(cf, conf, prev) != NGX_CONF_OK) {
317 return NGX_CONF_ERROR;
318 }
319 brix_merge_srv_tpc(conf, prev);
320 if (brix_merge_srv_cluster(cf, conf, prev) != NGX_CONF_OK) {
321 return NGX_CONF_ERROR;
322 }
323 if (brix_merge_srv_proxy_net(cf, conf, prev) != NGX_CONF_OK) {
324 return NGX_CONF_ERROR;
325 }
326
327 return NGX_CONF_OK;
328}

330/*
331 * ngx_stream_brix_enable - handler for the "xrootd on|off;" directive.
332 */
333char *
334ngx_stream_brix_enable(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
335{
336 ngx_stream_brix_srv_conf_t *xcf = conf;
337 ngx_stream_core_srv_conf_t *cscf;
338 char *rv;
339
340 rv = ngx_conf_set_flag_slot(cf, cmd, conf);
341 if (rv != NGX_CONF_OK) {
342 return rv;
343 }
344
345 /* Explicit `xrootd off;` leaves the server block as a normal stream server. */
346 if (!xcf->common.enable) {
347 return NGX_CONF_OK;
348 }
349
350 /*
351 * The stream core owns the accept loop; enabling the directive swaps in
352 * our session handler for this server block.
353 */
354 cscf = ngx_stream_conf_get_module_srv_conf(cf, ngx_stream_core_module);
355 cscf->handler = ngx_stream_brix_handler;
356
357 return NGX_CONF_OK;
358}
