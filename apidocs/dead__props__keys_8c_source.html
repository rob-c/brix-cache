1/*
2 * dead_props_keys.c - WebDAV dead-property xattr key encode/decode/validate.
3 *
4 * The flat "user." xattr keyspace cannot hold arbitrary XML namespace URIs or
5 * element names directly, so a dead property is keyed by a hex-encoded
6 * (namespace, local-name) pair under a fixed prefix. This file owns that
7 * encoding: building a key, decoding a key back, and validating that a decoded
8 * local name is safe to splice into a PROPFIND response. Split verbatim from
9 * dead_props.c; consumed by it via dead_props_internal.h.
10 */

26/*
27 * WHAT: True if `c` is an ASCII letter or '_' (the XML NameStartChar subset we
28 * permit for a dead-property local name).
29 * WHY: Splitting the character-class test out of the validator collapses the
30 * validator's branch fan-out (each range test is one branch) into a
31 * single call, keeping the validator's complexity low and the accepted
32 * set stated in exactly one place.
33 * HOW: Pure predicate over the raw byte; no side effects.
34 */
35static ngx_flag_t
36webdav_xml_name_start_char(unsigned char c)
37{
38 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z') || c == '_';
39}
40
41/*
42 * WHAT: True if `c` is a permitted XML NameChar for a dead-property local name
43 * (a NameStartChar, or a digit, '-', or '.').
44 * WHY: Same rationale as webdav_xml_name_start_char — one predicate per
45 * character class keeps the validator loop a single-branch scan.
46 * HOW: Reuses the start-char predicate, then admits the trailing-only extras.
47 */
48static ngx_flag_t
49webdav_xml_name_char(unsigned char c)
50{
51 return webdav_xml_name_start_char(c)
52 || (c >= '0' && c <= '9') || c == '-' || c == '.';
53}

55/*
56 * Validate that a decoded local name is a safe XML element name before it is
57 * spliced back into a PROPFIND response. This is the injection guard: a name
58 * read from an xattr is attacker-influenced, so we restrict it to the XML
59 * NameStartChar/NameChar subset (ASCII letters, '_', then also digits, '-',
60 * '.') and reject anything else (notably '<', '>', '&', '/', whitespace).
61 */
62ngx_flag_t
63webdav_dead_prop_xml_name_ok(const char *name)
64{
65 const unsigned char *p = (const unsigned char *) name;
66
67 if (p == NULL || *p == '\0') {
68 return 0;
69 }
70
71 if (!webdav_xml_name_start_char(*p)) {
72 return 0;
73 }
74
75 for (p++; *p != '\0'; p++) {
76 if (!webdav_xml_name_char(*p)) {
77 return 0;
78 }
79 }
80
81 return 1;
82}

84/*
85 * Encode (namespace URI, local name) into the flat xattr key documented above:
86 * PREFIX + hex(ns) + '.' + hex(local). Writes a NUL-terminated string into
87 * out[0..outsz). Every append is bounds-checked against `left` and returns
88 * NGX_ERROR (caller maps to ENAMETOOLONG) rather than truncating, since a
89 * truncated key would silently collide with a different property.
90 */
91ngx_int_t
92webdav_dead_prop_attr_name(const char *ns, const char *local,
93 char *out, size_t outsz)
94{
95 static const char hex[] = "0123456789abcdef";
96 const unsigned char *p;
97 char *d = out;
98 size_t left = outsz;
99 size_t n;
100
101 /* Prefix is fixed-length and always written first. */
102 n = WEBDAV_DEAD_PROP_PREFIX_LEN;
103 if (left <= n) {
104 return NGX_ERROR;
105 }
106 ngx_memcpy(d, WEBDAV_DEAD_PROP_PREFIX, n);
107 d += n;
108 left -= n;
109
110 /* hex(namespace): two output bytes per source byte; need >2 left so a
111 * trailing '.' and NUL still fit. */
112 for (p = (const unsigned char *) ns; p != NULL && *p != '\0'; p++) {
113 if (left <= 2) {
114 return NGX_ERROR;
115 }
116 *d++ = hex[*p >> 4];
117 *d++ = hex[*p & 0x0f];
118 left -= 2;
119 }
120
121 if (left <= 1) {
122 return NGX_ERROR;
123 }
124 *d++ = '.'; /* the one literal separator; source dots are hex "2e" */
125 left--;
126
127 /* hex(local name) */
128 for (p = (const unsigned char *) local; p != NULL && *p != '\0'; p++) {
129 if (left <= 2) {
130 return NGX_ERROR;
131 }
132 *d++ = hex[*p >> 4];
133 *d++ = hex[*p & 0x0f];
134 left -= 2;
135 }
136
137 if (left == 0) {
138 return NGX_ERROR;
139 }
140 *d = '\0';
141 return NGX_OK;
142}

144/*
145 * Inverse of the hex encoder: decode `len` hex chars into len/2 raw bytes plus
146 * a NUL. Returns NULL on odd length or any non-hex digit (a corrupted/foreign
147 * xattr key), which the caller treats as "not one of ours" rather than fatal.
148 */
149static char *
150webdav_dead_prop_decode_hex(ngx_pool_t *pool, const char *hex, size_t len)
151{
152 char *out;
153 size_t i;
154
155 if ((len & 1) != 0) { /* hex always comes in pairs */
156 return NULL;
157 }
158
159 BRIX_PNALLOC_OR_RETURN(out, pool, len / 2 + 1, NULL);
160
161 for (i = 0; i < len; i += 2) {
162 int hi = brix_hex_from_char((unsigned char) hex[i]);
163 int lo = brix_hex_from_char((unsigned char) hex[i + 1]);
164 if (hi < 0 || lo < 0) {
165 return NULL;
166 }
167 out[i / 2] = (char) ((hi << 4) | lo);
168 }
169 out[len / 2] = '\0';
170 return out;
171}

173/*
174 * Parse one listxattr entry back into (namespace, local name).
175 * Returns NGX_DECLINED for any key that is not a well-formed dead-property key
176 * (wrong prefix, no '.' separator, bad hex, or a local name that fails the XML
177 * safety check) so the caller can skip foreign xattrs silently.
178 */
179ngx_int_t
180webdav_dead_prop_decode_attr(ngx_pool_t *pool, const char *attr,
181 char **ns_out, char **local_out)
182{
183 const char *payload;
184 const char *dot;
185
186 if (ngx_strncmp(attr, WEBDAV_DEAD_PROP_PREFIX,
187 WEBDAV_DEAD_PROP_PREFIX_LEN) != 0)
188 {
189 return NGX_DECLINED;
190 }
191
192 payload = attr + WEBDAV_DEAD_PROP_PREFIX_LEN;
193 dot = strchr(payload, '.');
194 if (dot == NULL) {
195 return NGX_DECLINED;
196 }
197
198 *ns_out = webdav_dead_prop_decode_hex(pool, payload,
199 (size_t) (dot - payload));
200 *local_out = webdav_dead_prop_decode_hex(pool, dot + 1, strlen(dot + 1));
201 if (*ns_out == NULL || *local_out == NULL
202 || !webdav_dead_prop_xml_name_ok(*local_out))
203 {
204 return NGX_DECLINED;
205 }
206
207 return NGX_OK;
208}
