1/*
2 * config.c — pmark per-server config lifecycle + directive parsing.
3 *
4 * WHAT: Initialise/merge the brix_pmark_conf_t embedded in the shared config
5 * preamble (config/shared_conf.h), and the custom directive setters for the
6 * repeatable / multi-token directives (firefly_dest, map_experiment,
7 * map_activity, domain). Simple flag/str directives use the stock
8 * ngx_conf_set_*_slot setters wired directly in each module's command table.
9 *
10 * WHY: pmark config is shared across the stream (root://) and HTTP (WebDAV/S3)
11 * modules. Because ngx_http_brix_shared_conf_t is the FIRST member of every
12 * protocol conf struct, a custom setter can cast its `conf` argument straight
13 * to ngx_http_brix_shared_conf_t* and reach ->pmark, so ONE setter serves all
14 * three protocols.
15 *
16 * HOW: init sets NGX_CONF_UNSET sentinels; merge applies SciTags-sane defaults
17 * (opt-in disabled; firefly + flowlabel + scitag-cgi on; domain remote; firefly
18 * port 10514). Arrays inherit from the parent when the child did not set them.
19 */

39char *
40brix_pmark_conf_merge(ngx_conf_t *cf, brix_pmark_conf_t *prev,
41 brix_pmark_conf_t *conf)
42{
43 ngx_conf_merge_value(conf->enable, prev->enable, 0);
44 ngx_conf_merge_value(conf->firefly, prev->firefly, 1);
45 ngx_conf_merge_value(conf->flowlabel, prev->flowlabel, 1);
46 ngx_conf_merge_value(conf->scitag_cgi, prev->scitag_cgi, 1);
47 ngx_conf_merge_value(conf->firefly_origin, prev->firefly_origin, 0);
48 ngx_conf_merge_value(conf->http_plain, prev->http_plain, 0);
49 ngx_conf_merge_msec_value(conf->echo, prev->echo, 0);
50 /* XRootD's pmark ffecho enforces a 30s floor on the "ongoing" refresh —
51 * a shorter interval multiplies firefly UDP per active flow for no
52 * monitoring benefit. Mirror it: warn and raise rather than reject so a
53 * config written against stock xrootd semantics keeps working. */
54 if (conf->echo > 0 && conf->echo < 30000) {
55 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
56 "brix_pmark_echo %M ms is below the 30s minimum; raised to 30s",
57 conf->echo);
58 conf->echo = 30000;
59 }
60 ngx_conf_merge_uint_value(conf->domain, prev->domain,
61 BRIX_PMARK_DOMAIN_REMOTE);
62 ngx_conf_merge_str_value(conf->appname, prev->appname, "nginx-xrootd");
63 ngx_conf_merge_str_value(conf->defsfile, prev->defsfile, "");
64
65 if (conf->firefly_dest == NULL) { conf->firefly_dest = prev->firefly_dest; }
66 if (conf->exp_rules == NULL) { conf->exp_rules = prev->exp_rules; }
67 if (conf->act_rules == NULL) { conf->act_rules = prev->act_rules; }
68
69 return NGX_CONF_OK;
70}

81char *
82brix_pmark_set_firefly_dest(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
83{
84 brix_pmark_conf_t *pm = pmark_conf(conf);
85 ngx_str_t *value = cf->args->elts;
86 ngx_str_t *dest;
87
88 (void) cmd;
89
90 if (pm->firefly_dest == NULL) {
91 pm->firefly_dest = ngx_array_create(cf->pool, 2, sizeof(ngx_str_t));
92 if (pm->firefly_dest == NULL) {
93 return NGX_CONF_ERROR;
94 }
95 }
96
97 dest = ngx_array_push(pm->firefly_dest);
98 if (dest == NULL) {
99 return NGX_CONF_ERROR;
100 }
101
102 /* Stored verbatim as "host[:port]"; resolved to a sockaddr per worker at
103 * init (defaults to port 10514 when no :port is given). */
104 *dest = value[1];
105 return NGX_CONF_OK;
106}

116
117 if (ngx_strcmp(value[1].data, "any") == 0) {
118 pm->domain = BRIX_PMARK_DOMAIN_ANY;
119 } else if (ngx_strcmp(value[1].data, "local") == 0) {
120 pm->domain = BRIX_PMARK_DOMAIN_LOCAL;

122 pm->domain = BRIX_PMARK_DOMAIN_REMOTE;
123 } else {
124 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
125 "invalid brix_pmark_domain \"%V\" (use any|local|remote)",
126 &value[1]);
127 return NGX_CONF_ERROR;
128 }
129 return NGX_CONF_OK;

139

141
142 if (pm->exp_rules == NULL) {

145 if (pm->exp_rules == NULL) {
146 return NGX_CONF_ERROR;
147 }
148 }
149
150 rule = ngx_array_push(pm->exp_rules);
151 if (rule == NULL) {
152 return NGX_CONF_ERROR;
153 }
154 ngx_memzero(rule, sizeof(*rule));
155
156 /* default <exp> | path <glob> <exp> | vo <name> <exp> */
157 if (ngx_strcmp(value[1].data, "default") == 0 && cf->args->nelts == 3) {
158 rule->kind = BRIX_PMARK_EXP_DEFAULT;
159 rule->exp_name = value[2];
160 } else if (ngx_strcmp(value[1].data, "path") == 0 && cf->args->nelts == 4) {
161 rule->kind = BRIX_PMARK_EXP_PATH;
162 rule->match = value[2];
163 rule->exp_name = value[3];
164 } else if (ngx_strcmp(value[1].data, "vo") == 0 && cf->args->nelts == 4) {
165 rule->kind = BRIX_PMARK_EXP_VO;
166 rule->match = value[2];
167 rule->exp_name = value[3];
168 } else {
169 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
170 "brix_pmark_map_experiment: use "
171 "\"default <exp>\" | \"path <glob> <exp>\" | \"vo <name> <exp>\"");
172 return NGX_CONF_ERROR;
173 }
174 return NGX_CONF_OK;
175}
