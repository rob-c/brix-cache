1/*
2 * xrdhttp_stats.c — ?xrd.stats XML statistics endpoint for the XrdHttp protocol.
3 *
4 * WHAT: Serves the XRootD statistics XML format at any URL that includes
5 * the query parameter "xrd.stats". The response is consumed by XRootD
6 * collector infrastructure (xrd_mon, XrdCns) and by xrdfs "query stats".
7 *
8 * WHY: XrdHttp clients and monitoring agents poll "GET /path?xrd.stats" to
9 * obtain server performance data in the same format that the native XRootD
10 * protocol serves via kXR_query / kXR_QStats. Providing this endpoint makes
11 * nginx-xrootd observable by the same monitoring stack without requiring a
12 * separate sidecar.
13 *
14 * HOW: Reads WebDAV metrics from the shared-memory zone (atomic reads, safe
15 * from any worker) and formats them into a minimal XML document compatible
16 * with XRootD 5.x stats output. Fields that have no direct mapping (e.g.,
17 * scheduler threads, poll-set state) are emitted as zero to avoid parse
18 * failures in collector software.
19 *
20 * SECURITY: The response contains only aggregate counters — no path names,
21 * bucket names, principal names, or IP addresses are included. Collectors
22 * that wish to restrict access to this endpoint should do so via nginx
23 * location blocks (allow/deny directives or auth_basic).
24 */

41 * Build the XRootD-compatible stats XML document into buf (null-terminated).
42 * Returns the number of bytes written (excluding the terminating NUL), or -1

45static int
46build_stats_xml(char *buf, size_t bufsz, ngx_http_request_t *r)
47{
48 ngx_brix_metrics_t *shm;
49 ngx_brix_webdav_metrics_t zero;
50 ngx_brix_webdav_metrics_t *m;
51 char hostname[256];
52 time_t now;
53 ngx_atomic_t gets, puts, deletes, total_req;
54 ngx_atomic_t bytes_in, bytes_out;
55 ngx_atomic_t auth_ok, auth_fail;
56 int listen_port;
57 int len;
58
59 shm = brix_metrics_shared();
60 if (shm != NULL) {
61 m = &shm->webdav;
62 } else {
63 ngx_memzero(&zero, sizeof(zero));
64 m = &zero;
65 }
66
67 now = time(NULL);
68
69 if (gethostname(hostname, sizeof(hostname)) != 0) {
70 ngx_cpystrn((u_char *) hostname, (u_char *) "localhost",
71 sizeof(hostname));
72 }
73 hostname[sizeof(hostname) - 1] = '\0';
74
75 /* Derive the actual listening port from the incoming connection. */
76 {
77 struct sockaddr_storage ss;
78 socklen_t sslen = sizeof(ss);
79
80 if (getsockname(r->connection->fd, (struct sockaddr *)(void *) &ss,
81 &sslen) == 0) {
82 if (ss.ss_family == AF_INET) {
83 listen_port = ntohs(((struct sockaddr_in *)(void *) &ss)->sin_port);
84 } else if (ss.ss_family == AF_INET6) {
85 listen_port = ntohs(((struct sockaddr_in6 *)(void *) &ss)->sin6_port);
86 } else {
87 listen_port = 8443;
88 }
89 } else {
90 listen_port = 8443;
91 }
92 }
93
94 gets = m->requests_total[BRIX_WEBDAV_METHOD_GET]
95 + m->requests_total[BRIX_WEBDAV_METHOD_HEAD];
96 puts = m->requests_total[BRIX_WEBDAV_METHOD_PUT];
97 deletes = m->requests_total[BRIX_WEBDAV_METHOD_DELETE];
98
99 total_req = 0;
100 for (ngx_uint_t i = 0; i < BRIX_WEBDAV_NMETHODS; i++) {
101 total_req += m->requests_total[i];
102 }
103
104 bytes_in = m->bytes_rx_total;
105 bytes_out = m->bytes_tx_total;
106
107 auth_ok = m->auth_total[BRIX_WEBDAV_AUTH_RESULT_CERT_OK]
108 + m->auth_total[BRIX_WEBDAV_AUTH_RESULT_TOKEN_OK]
109 + m->auth_total[BRIX_WEBDAV_AUTH_RESULT_ANONYMOUS];
110 auth_fail = m->auth_total[BRIX_WEBDAV_AUTH_RESULT_REJECTED];
111
112 len = snprintf(buf, bufsz,
113 "<?xml version=\"1.0\" encoding=\"utf-8\"?>\n"
114 "<statistics tod=\"%ld\" ver=\"" BRIX_SERVER_VERSION "\" src=\"%s:%d\""
115 " ins=\"anon\" pid=\"%d\" pgm=\"" BRIX_SERVER_NAME "\">\n"
116
117 " <stats id=\"info\">\n"
118 " <host>%s</host>\n"
119 " <port>%d</port>\n"
120 " <name>" BRIX_SERVER_NAME "</name>\n"
121 " <role>server</role>\n"
122 " </stats>\n"
123
124 " <stats id=\"sched\">\n"
125 " <jobs>0</jobs><inq>0</inq><maxinq>0</maxinq>\n"
126 " <threads>0</threads><idle>0</idle>\n"
127 " <tcr>0</tcr><tde>0</tde><tlimr>0</tlimr>\n"
128 " </stats>\n"
129
130 " <stats id=\"link\">\n"
131 " <num>%lld</num><maxn>0</maxn>\n"
132 " <tot>%lld</tot>\n"
133 " <in>%lld</in><out>%lld</out>\n"
134 " <ctime>0</ctime><tmo>0</tmo><stall>0</stall><sfps>0</sfps>\n"
135 " </stats>\n"
136
137 " <stats id=\"poll\">\n"
138 " <att>0</att><en>0</en><ev>0</ev><int>0</int>\n"
139 " </stats>\n"
140
141 " <stats id=\"buff\">\n"
142 " <reqs>0</reqs><mem>0</mem><buffs>0</buffs>\n"
143 " <adj>0</adj><xlreqs>0</xlreqs>\n"
144 " </stats>\n"
145
146 " <stats id=\"xrootd\">\n"
147 " <ops>\n"
148 " <open>%lld</open>\n"
149 " <getf>%lld</getf>\n"
150 " <put>%lld</put>\n"
151 " <rf>%lld</rf>\n"
152 " <rd>%lld</rd>\n"
153 " <pr>0</pr><rv>0</rv>\n"
154 " </ops>\n"
155 " <aio><num>0</num><max>0</max><rej>0</rej></aio>\n"
156 " <err>%lld</err><dly>0</dly><rdr>0</rdr>\n"
157 " <lgn>\n"
158 " <num>%lld</num>\n"
159 " <af>%lld</af><ua>0</ua><ur>0</ur>\n"
160 " </lgn>\n"
161 " </stats>\n"
162
163 " <stats id=\"http\">\n"
164 " <requests>%lld</requests>\n"
165 " <bytes_in>%lld</bytes_in>\n"
166 " <bytes_out>%lld</bytes_out>\n"
167 " <tpc_pull>%lld</tpc_pull>\n"
168 " <tpc_push>%lld</tpc_push>\n"
169 " <range_partial>%lld</range_partial>\n"
170 " </stats>\n"
171
172 "</statistics>\n",
173
174 /* header */
175 (long) now, hostname, listen_port, (int) ngx_getpid(),
176 /* info */
177 hostname, listen_port,
178 /* link */
179 (long long) total_req, /* num — approximated as total requests */
180 (long long) total_req, /* tot */
181 (long long) bytes_in, /* in */
182 (long long) bytes_out, /* out */
183 /* xrootd ops */
184 (long long) (gets + puts), /* open */
185 (long long) gets, /* getf */
186 (long long) puts, /* put */
187 (long long) deletes, /* rf (remove file) */
188 (long long) m->requests_total[BRIX_WEBDAV_METHOD_GET], /* rd */
189 /* err */
190 (long long) (m->auth_total[BRIX_WEBDAV_AUTH_RESULT_REJECTED]
191 + m->responses_total[0][BRIX_HTTP_STATUS_5XX]),
192 /* lgn */
193 (long long) auth_ok,
194 (long long) auth_fail,
195 /* http */
196 (long long) total_req,
197 (long long) bytes_in,
198 (long long) bytes_out,
199 (long long) m->tpc_total[BRIX_WEBDAV_TPC_PULL_STARTED],
200 (long long) m->tpc_total[BRIX_WEBDAV_TPC_PUSH_STARTED],
201 (long long) m->range_total[BRIX_WEBDAV_RANGE_PARTIAL]
202 );
203
204 if (len < 0 || (size_t) len >= bufsz) {
205 return -1;
206 }
207
208 return len;
209}

211ngx_int_t
212xrdhttp_handle_stats_query(ngx_http_request_t *r)
213{
214 char buf[XRDHTTP_STATS_BUF_MAX];
215 int len;
216 ngx_buf_t *b;
217 ngx_chain_t out;
218 ngx_table_elt_t *h;
219
220 len = build_stats_xml(buf, sizeof(buf), r);
221 if (len < 0) {
222 return NGX_HTTP_INTERNAL_SERVER_ERROR;
223 }
224
225 r->headers_out.status = NGX_HTTP_OK;
226 r->headers_out.content_length_n = len;
227 r->allow_ranges = 0;
228
229 h = ngx_list_push(&r->headers_out.headers);
230 if (h == NULL) {
231 return NGX_HTTP_INTERNAL_SERVER_ERROR;
232 }
233 h->hash = 1;
234 ngx_str_set(&h->key, "Content-Type");
235 ngx_str_set(&h->value, "text/xml; charset=utf-8");
236
237 /* Cache-Control: no-store — stats are always current. */
238 h = ngx_list_push(&r->headers_out.headers);
239 if (h == NULL) {
240 return NGX_HTTP_INTERNAL_SERVER_ERROR;
241 }
242 h->hash = 1;
243 ngx_str_set(&h->key, "Cache-Control");
244 ngx_str_set(&h->value, "no-store");
245
246 (void) xrdhttp_add_response_headers(r, NGX_HTTP_OK);
247
248 ngx_int_t rc = ngx_http_send_header(r);
249 if (rc == NGX_ERROR || r->header_only) {
250 return rc;
251 }
252
253 b = ngx_create_temp_buf(r->pool, (size_t) len);
254 if (b == NULL) {
255 return NGX_HTTP_INTERNAL_SERVER_ERROR;
256 }
257 ngx_memcpy(b->pos, buf, (size_t) len);
258 b->last = b->pos + len;
259 b->last_buf = 1;
260 b->last_in_chain = 1;
261
262 out.buf = b;
263 out.next = NULL;
264
265 return ngx_http_output_filter(r, &out);
266}
