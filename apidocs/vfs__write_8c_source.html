1/*
2 * vfs_write.c — the shared full-write primitive.
3 *
4 * WHAT: Implements brix_vfs_pwrite_full(), the EINTR-safe / short-write-safe
5 * pwrite loop used across the VFS (the I/O core in vfs_io_core.c and the
6 * per-buffer copiers).
7 *
8 * WHY: Every VFS write has to apply the same short-write/EINTR retry policy, and
9 * every data byte has to traverse the Storage Driver seam rather than a raw
10 * syscall. Concentrating the loop here guarantees both, once.
11 *
12 * HOW: brix_vfs_pwrite_full() wraps the fd in a POSIX storage-driver object and
13 * drives driver.pwrite() in a loop until exactly len bytes are written,
14 * retrying on EINTR and treating a 0-byte pwrite as EIO.
15 */
16#include "vfs_internal.h"
17#include "fs/backend/sd.h"
18
19/* EINTR-safe, short-write-safe pwrite loop. Writes exactly len bytes at offset
20 * or returns NGX_ERROR; a 0-byte pwrite is treated as EIO. */
21ngx_int_t
22brix_vfs_pwrite_full(ngx_fd_t fd, const u_char *buf, size_t len,
23 off_t offset)
24{
25 size_t done = 0;
26 brix_sd_obj_t obj;
27
28 /* Route every data byte write through the Storage Driver seam (phase-55); the
29 * EINTR/short-write loop policy stays here in the VFS. The syscall lives in the
30 * backend so a non-POSIX driver slots in unchanged. (Reverts phase-56 A-1.) */
31 brix_sd_posix_wrap(&obj, fd);
32
33 while (done < len) {
34 ssize_t n = brix_sd_posix_driver.pwrite(&obj, buf + done, len - done,
35 offset + (off_t) done);
36
37 if (n < 0) {
38 if (errno == EINTR) {
39 continue;
40 }
41 return NGX_ERROR;
42 }
43
44 if (n == 0) {
45 errno = EIO;
46 return NGX_ERROR;
47 }
48
49 done += (size_t) n;
50 }
51
52 return NGX_OK;
53}
