1/*
2 * sec_protocol.h — kXR_login "&P=" security-protocol list parser (shared).
3 *
4 * WHAT: after kXR_login the server advertises the auth methods it accepts as a
5 * parameter block of "&P=<name>,<args>" entries (e.g.
6 * "&P=gsi,v:10300,c:ssl&P=ztn,..."). brix_sec_proto_advertised tells a
7 * client whether a given protocol is offered and, optionally, returns that
8 * entry's argument string.
9 * WHY: TWO XRootD-client implementations parse this same list to choose which
10 * credential to send: the native client's auth driver (client/lib/auth.c)
11 * and the server's own TPC-outbound auth selector (server-as-client,
12 * src/tpc/gsi/gsi_outbound_finish.c). The latter hand-rolled a LOOSER check —
13 * a bare strstr(parms, "ztn")/"gsi") with no "&P=" anchor or name-boundary
14 * test — which can false-match the substring anywhere in the block (another
15 * protocol's args, a trailing hostname) and pick the wrong auth method.
16 * Sharing the native client's anchored parser de-duplicates the grammar AND
17 * tightens the server's selection.
18 * HOW: header-only static inline over libc (strstr/strncmp/memcpy) — no ngx, no
19 * allocation — so the same code compiles into the nginx module and the
20 * ngx-free client. `parms` may be NULL for a presence-only query.
21 *
22 * Clean-room: "&P=<proto>,<args>" grammar from the XRootD login security list.
23 */

25#define BRIX_PROTOCOL_SEC_PROTOCOL_H

30/*
31 * Is "&P=<name>" advertised in `seclist`? Matches only on the "&P=" anchor with a
32 * name boundary (',' / '&' / end), so a protocol name embedded in another entry's
33 * arguments does not false-match. When `parms` is non-NULL and `psz` > 0, the
34 * matched entry's argument string (the text after the name's comma, up to the next
35 * "&P=" or end) is copied in NUL-terminated and truncated to fit. Returns 1 if the
36 * protocol is advertised, else 0.
37 */
38static inline int
39brix_sec_proto_advertised(const char *seclist, const char *name,
40 char *parms, size_t psz)
41{
42 size_t nlen;
43 const char *p;
44
45 if (parms != NULL && psz > 0) {
46 parms[0] = '\0';
47 }
48 if (seclist == NULL || name == NULL) {
49 return 0;
50 }
51
52 nlen = strlen(name);
53 p = seclist;
54 while ((p = strstr(p, "&P=")) != NULL) {
55 const char *n = p + 3;
56 if (strncmp(n, name, nlen) == 0
57 && (n[nlen] == ',' || n[nlen] == '&' || n[nlen] == '\0')) {
58 if (parms != NULL && psz > 0) {
59 const char *args = (n[nlen] == ',') ? n + nlen + 1 : "";
60 const char *end = strstr(args, "&P=");
61 size_t alen = end ? (size_t) (end - args) : strlen(args);
62 if (alen >= psz) {
63 alen = psz - 1;
64 }
65 memcpy(parms, args, alen);
66 parms[alen] = '\0';
67 }
68 return 1;
69 }
70 p = n;
71 }
72 return 0;
73}
