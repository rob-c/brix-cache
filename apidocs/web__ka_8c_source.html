1/*
2 * web_ka.c — persistent keep-alive HTTP/1.1 transport + PROPFIND exchange.
3 *
4 * WHAT: brix_kaconn (connect/disconnect/read-headers/read-body) and brix_webmeta
5 * (a kaconn + fixed auth) with brix_webmeta_propfind. One request in
6 * flight; NOT thread-safe — pool it (brix_cpool) or serialise it.
7 * WHY: the FUSE web-metadata path (getattr/readdir) reconnected per call via
8 * the one-shot brix_http_req. Keep-alive + a small pool removes the
9 * per-op TCP+TLS handshake, matching the read path (webfile_io.c) and the
10 * recent brixcvmfs connection-reuse work.
11 * HOW: the transport/codec mirrors webfile_io.c (web_connect/web_read_some/
12 * header parse); the PROPFIND loop mirrors brix_webfile_pread's deadline-
13 * bounded reconnect-on-sever. No goto.
14 *
15 * Clean-room: composes the public brix_* connection API only.
16 */

69int
70brix_kaconn_connect(brix_kaconn *k, brix_status *st)
71{
72 memset(&k->io, 0, sizeof(k->io));
73 k->io.fd = brix_tcp_connect(k->host, k->port, k->timeout_ms, st);
74 if (k->io.fd < 0) {
75 return -1;
76 }
77 k->io.timeout_ms = k->timeout_ms;
78 if (k->tls && brix_tls_client(&k->io, k->host, k->verify, k->verify,
79 k->ca_dir[0] ? k->ca_dir : NULL,
80 &k->tls_ctx, st) != 0) {
81 close(k->io.fd);
82 k->io.fd = -1;
83 return -1;
84 }
85 k->connected = 1;
86 return 0;
87}

89/* Read up to n bytes (branches on TLS). >0 bytes, 0 EOF, -1 error. Mirrors
90 * web_read_some (webfile_io.c:51), keyed on brix_kaconn. */
91static ssize_t
92kaconn_read_some(brix_kaconn *k, void *buf, size_t n, brix_status *st)
93{
94 if (k->io.ssl != NULL) {
95 size_t got = 0;
96 if (brix_tls_read_some(&k->io, buf, n, &got, st) != 0) {
97 return -1;
98 }
99 return (ssize_t) got;
100 }
101 struct pollfd pfd;
102 ssize_t r;
103 int pr;
104 pfd.fd = k->io.fd; pfd.events = POLLIN; pfd.revents = 0;
105 do { pr = poll(&pfd, 1, k->io.timeout_ms); } while (pr < 0 && errno == EINTR);
106 if (pr <= 0) {
107 brix_status_set(st, XRDC_ESOCK, pr == 0 ? ETIMEDOUT : errno, "ka read");
108 return -1;
109 }
110 do { r = read(k->io.fd, buf, n); } while (r < 0 && errno == EINTR);
111 if (r < 0) {
112 brix_status_set(st, XRDC_ESOCK, errno, "ka read: %s", strerror(errno));
113 return -1;
114 }
115 return r;
116}

118/* Case-insensitive Content-Length lookup (mirrors hdr_clen). */
119static long long
120ka_hdr_clen(const char *hdrs)
121{
122 const char *p = hdrs;
123 while ((p = strchr(p, '\n')) != NULL) {
124 p++;
125 if (strncasecmp(p, "Content-Length:", 15) == 0) {
126 return strtoll(p + 15, NULL, 10);
127 }
128 }
129 if (strncasecmp(hdrs, "Content-Length:", 15) == 0) {
130 return strtoll(hdrs + 15, NULL, 10);
131 }
132 return -1;
133}

135int
136brix_kaconn_read_headers(brix_kaconn *k, char *hbuf, size_t hbufsz,
137 brix_ka_hdr *out, brix_status *st)
138{
139 size_t hlen = 0;
140 char *eoh = NULL;
141 int status = 0;
142 long long clen;
143
144 for (;;) {
145 ssize_t r;
146 if (hlen + 1 >= hbufsz) {
147 brix_kaconn_disconnect(k);
148 brix_status_set(st, XRDC_EPROTO, 0, "ka: header too large");
149 return -1;
150 }
151 r = kaconn_read_some(k, hbuf + hlen, hbufsz - 1 - hlen, st);
152 if (r <= 0) {
153 brix_kaconn_disconnect(k);
154 if (r == 0) {
155 brix_status_set(st, XRDC_ESOCK, 0, "ka: peer closed");
156 }
157 return -1;
158 }
159 hlen += (size_t) r;
160 hbuf[hlen] = '\0';
161 eoh = strstr(hbuf, "\r\n\r\n");
162 if (eoh != NULL) {
163 break;
164 }
165 }
166 if (strncmp(hbuf, "HTTP/", 5) == 0) {
167 const char *sp = strchr(hbuf, ' ');
168 status = sp ? atoi(sp + 1) : 0;
169 }
170 /* Parse Content-Length off a NUL-terminated header block, then restore the
171 * first body byte. clen stays -1 when absent — the caller decides (a non-2xx
172 * response is rejected without needing a body). */
173 {
174 char saved = eoh[0];
175 eoh[0] = '\0';
176 clen = ka_hdr_clen(hbuf);
177 eoh[0] = saved;
178 }
179 if (clen > KA_BODY_MAX) {
180 brix_kaconn_disconnect(k);
181 brix_status_set(st, XRDC_EPROTO, 0, "ka: body too large");
182 return -1;
183 }
184 out->status = status;
185 out->clen = clen;
186 out->body_start = eoh + 4;
187 out->body_buffered = hlen - (size_t) (out->body_start - hbuf);
188 return 1;
189}

191int
192brix_kaconn_read_body(brix_kaconn *k, const brix_ka_hdr *hdr,
193 char **body_out, size_t *len_out, brix_status *st)
194{
195 size_t total = (size_t) hdr->clen;
196 char *body = malloc(total + 1);
197 size_t have = hdr->body_buffered < total ? hdr->body_buffered : total;
198
199 if (body == NULL) {
200 brix_kaconn_disconnect(k);
201 brix_status_set(st, XRDC_EPROTO, 0, "ka: out of memory");
202 return -1;
203 }
204 memcpy(body, hdr->body_start, have);
205 if (have < total
206 && brix_read_full(&k->io, body + have, total - have, st) != 0) {
207 brix_kaconn_disconnect(k); /* sever mid-body → drop the slot */
208 free(body);
209 return -1;
210 }
211 body[total] = '\0';
212 *body_out = body;
213 *len_out = total;
214 return 0;
215}

231/* Build + write "PROPFIND path HTTP/1.1 … Depth: d {auth}". On write fault the
232 * kaconn is disconnected (caller reconnects + retries). */
233static int
234webmeta_send_req(brix_webmeta *m, const char *path, int depth, brix_status *st)
235{
236 char req[3200];
237 int rn = snprintf(req, sizeof(req),
238 "PROPFIND %s HTTP/1.1\r\nHost: %s\r\nUser-Agent: xrootdfs\r\n"
239 "Accept: */*\r\nConnection: keep-alive\r\nDepth: %d\r\n%s\r\n",
240 path[0] ? path : "/", m->ka.hostport,
241 depth ? 1 : 0, m->auth);
242 if (rn < 0 || (size_t) rn >= sizeof(req)) {
243 brix_status_set(st, XRDC_EUSAGE, 0, "PROPFIND: request too long");
244 return -1;
245 }
246 if (brix_write_full(&m->ka.io, req, (size_t) rn, st) != 0) {
247 brix_kaconn_disconnect(&m->ka);
248 return -1;
249 }
250 return 0;
251}

253/* status → kXR (matches web_range_read_headers:206-212). 0 ok, -1 err (st set). */
254static int
255webmeta_status_map(int status, brix_status *st)
256{
257 if (status == 207 || status == 200) {
258 return 0;
259 }
260 if (status == 404) {
261 brix_status_set(st, kXR_NotFound, 0, "not found");
262 } else if (status == 401 || status == 403) {
263 brix_status_set(st, kXR_NotAuthorized, 0, "HTTP %d", status);
264 } else {
265 brix_status_set(st, XRDC_EPROTO, 0, "PROPFIND HTTP %d", status);
266 }
267 return -1;
268}

270int
271brix_webmeta_propfind(brix_webmeta *m, const char *path, int depth,
272 char **body_out, size_t *len_out, brix_status *st)
273{
274 unsigned attempt = 0;
275 int window_ms = webfile_window_ms();
276 uint64_t deadline = brix_mono_ns() + (uint64_t) window_ms * 1000000ULL;
277
278 *body_out = NULL;
279 *len_out = 0;
280
281 for (;;) {
282 if (m->ka.connected || brix_kaconn_connect(&m->ka, st) == 0) {
283 if (webmeta_send_req(m, path, depth, st) == 0) {
284 char hbuf[KA_HDR_MAX + 1];
285 brix_ka_hdr h;
286 if (brix_kaconn_read_headers(&m->ka, hbuf, sizeof(hbuf), &h, st)
287 == 1) {
288 if (webmeta_status_map(h.status, st) != 0) {
289 /* non-2xx: the body was not consumed, so the socket is
290 * misaligned — drop it (next checkout reconnects). The
291 * status (NotFound/NotAuthorized/…) is non-retryable, so
292 * the retry gate below returns immediately. */
293 brix_kaconn_disconnect(&m->ka);
294 return -1;
295 }
296 if (h.clen < 0) {
297 brix_kaconn_disconnect(&m->ka);
298 brix_status_set(st, XRDC_EPROTO, 0,
299 "PROPFIND: no Content-Length");
300 return -1;
301 }
302 if (brix_kaconn_read_body(&m->ka, &h, body_out, len_out, st)
303 == 0) {
304 return 0; /* success */
305 }
306 }
307 }
308 }
309 if (window_ms <= 0 || !brix_status_retryable(st)
310 || brix_mono_ns() >= deadline) {
311 return -1;
312 }
313 brix_backoff_sleep_fast(attempt++);
314 }
