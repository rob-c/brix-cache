4/*
5 * metrics_internal.h — private contract of the /metrics HTTP exporter module.
6 *
7 * WHAT: the buffered Prometheus-text writer (metrics_writer_t + mw_* helpers)
8 * and the per-subsystem "export" entry points (one per .c) that the
9 * handler calls to stream the counters out of the shared-memory metrics
10 * block (ngx_brix_metrics_t, defined in metrics.h) as Prometheus text.
11 * WHY: the counters live in SHM and are written on the hot path by every
12 * protocol; this read-only exporter is a SEPARATE nginx HTTP module
13 * (ngx_http_brix_metrics_module) so /metrics can be scraped without
14 * touching the data plane. Splitting the exporters per subsystem
15 * (stream/webdav/s3/cluster/ratelimit/...) keeps each .c small.
16 * HOW: mw_init/mw_printf/mw_emit_* append into a pool-backed ngx_chain_t of
17 * 64 KiB buffers; mw_finish caps the chain; the handler sends it as the
18 * response body. The brix_export_* functions each read one region of
19 * *shm and emit its HELP/TYPE/counter lines via the writer.
20 */

41/* Grow-only buffered writer: mw_printf/mw_emit_* append text into a chain of
42 * METRICS_BUF_SIZE buffers (pool-allocated), starting a new buffer when the
43 * current tail fills. The completed chain becomes the /metrics response body. */
44typedef struct {
45 ngx_pool_t *pool;
46 ngx_chain_t *head;
47 ngx_chain_t *tail;
48 u_char *pos; /* current write cursor in tail buffer */
49 u_char *last; /* one-past-end pointer for the tail buffer */
50 size_t total; /* total bytes emitted across the whole chain */
51} metrics_writer_t;

62/* Append printf-formatted text to the chain, rolling over to a fresh buffer if
63 * the current tail can't hold the output (a single line must fit in one empty

65 * a line that overflows even an empty buffer. NOTE: the mw_emit_* helpers ignore
