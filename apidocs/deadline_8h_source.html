1/*
2 * deadline.h — Phase 39 network-fault resilience: steady-state per-connection
3 * read / send deadlines for the root:// stream plane.
4 *
5 * WHAT: small, idempotent helpers that arm/disarm the client read (c->read) and
6 * write (c->write) timers used to shed slowloris / silently-stalled / half-open
7 * peers. Every helper is a no-op unless the operator enabled the matching
8 * directive (brix_handshake_timeout / brix_read_timeout / brix_send_timeout),
9 * whose merged value is cached on ctx at accept (src/connection/handler.c).
10 *
11 * WHY: official XRootD arms no steady-state deadline — a 1-byte-per-readWait
12 * trickle, a half-open peer, or a silent mid-PDU stall is never timed out, so it
13 * pins ctx + scratch heap + fds + budget + a concurrency slot indefinitely.
14 * These deadlines bound that, and so exceed official's behaviour under loss.
15 *
16 * HOW: read_deadline_armed / send_deadline_armed track OUR ownership of the timer,
17 * so arm/disarm are idempotent. The read deadline is armed ONLY at a genuine
18 * incompletion (the recv loop's NGX_AGAIN suspend) and disarmed at the
19 * PDU-complete boundary; under healthy back-to-back pipelined reads (which never
20 * block mid-PDU) it is never armed, so the Phase-29 keep-reading branches never
21 * touch the worker timer rbtree. Because the recv loop always disarms at the
22 * PDU-complete boundary — which precedes every dispatch/sub-system handoff — the
23 * read timer is provably never armed while in XRD_ST_AIO/SENDING/UPSTREAM/PROXY/
24 * WAITING_*, so rev->timedout can never fire and finalize the session while an
25 * in-flight AIO task still references ctx (the documented UAF hazard).
26 */

35/*
36 * Arm c->read's deadline for an in-progress obligation, if we have not already
37 * armed it and the relevant deadline is enabled.
38 * - Pre-auth: bound the WHOLE unauthenticated phase (handshake_timeout_ms) — any
39 * wait while unauthenticated is suspicious, so an unauthenticated stall cannot
40 * squat a connection slot.
41 * - Authed: bound only an IN-PROGRESS partial PDU (read_timeout_ms). An idle
42 * keepalive connection sitting at a fresh request boundary (REQ_HEADER,
43 * hdr_pos == 0) is deliberately left alone so long-lived xrdcp sessions that
44 * pause between operations are not killed.
45 * Idempotent: a no-op when already armed, so it never resets the deadline on each
46 * partial byte (the slowloris-resistant property — the deadline measures time to
47 * COMPLETE the PDU, not time since the last byte).
48 */
49static ngx_inline void
50brix_arm_read_deadline(ngx_connection_t *c, brix_ctx_t *ctx)
51{
52 ngx_msec_t t;
53
54 if (ctx->deadline.read_armed) {
55 return;
56 }
57
58 if (ctx->login.auth_done) {
59 if (ctx->state == XRD_ST_REQ_HEADER && ctx->recv.hdr_pos == 0) {
60 return; /* idle between requests — allow keepalive */
61 }
62 t = ctx->deadline.read_ms;
63 } else {
64 t = ctx->deadline.handshake_ms;
65 }
66
67 if (t == 0) {
68 return; /* deadline disabled */
69 }
70
71 ngx_add_timer(c->read, t);
72 ctx->deadline.read_armed = 1;
73}

75/* Disarm the read deadline we armed (idempotent; never touches a timer we did not
76 * arm — e.g. the CMS/FRM WAITING timers that share c->read). */
77static ngx_inline void
78brix_disarm_read_deadline(ngx_connection_t *c, brix_ctx_t *ctx)
79{
80 if (!ctx->deadline.read_armed) {
81 return;
82 }
83 if (c->read->timer_set) {
84 ngx_del_timer(c->read);
85 }
86 ctx->deadline.read_armed = 0;
87}

89/*
90 * Arm / refresh c->write's response-drain deadline. Called from
91 * brix_schedule_write_resume on every park and re-park — i.e. once per write
92 * event that still has queued data to drain — so the deadline RESETS on each
93 * drain-progress opportunity (an EPOLLOUT means the socket accepted more bytes).
94 * It therefore fires only after send_timeout_ms with NO progress at all (a stuck /
95 * half-open consumer that has stopped reading), and never false-fires on a
96 * legitimate slow-but-steady large/pipelined transfer. Reset frequency is
97 * per-write-event (per socket-buffer refill), never per-byte or per-slot, so the
98 * bulk-throughput drain takes one cheap rbtree reposition per refill cycle.
99 * Disarmed by brix_disarm_send_deadline when the output queue fully drains.
100 * Bounds a slow / half-open consumer that would otherwise pin the parked out_ring
101 * slots + the read_scratch windows they reference forever.
102 */
103static ngx_inline void
104brix_arm_send_deadline(ngx_connection_t *c, brix_ctx_t *ctx)
105{
106 if (ctx->deadline.send_ms == 0) {
107 return; /* deadline disabled */
108 }
109 ngx_add_timer(c->write, ctx->deadline.send_ms);
110 ctx->deadline.send_armed = 1;
111}
112
113static ngx_inline void
114brix_disarm_send_deadline(ngx_connection_t *c, brix_ctx_t *ctx)
115{
116 if (!ctx->deadline.send_armed) {
117 return;
118 }
119 if (c->write->timer_set) {
120 ngx_del_timer(c->write);
121 }
122 ctx->deadline.send_armed = 0;
123}
