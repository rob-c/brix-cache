54/* echo-async: the deferred sibling of echo. On submit it asks to defer (the
55 * server replies kXR_waitresp and pushes the result later); when re-invoked for
56 * completion (defer unavailable) it echoes the request inline. The defer-or-respond
57 * shape is the template every async service follows. */
58static int
59svc_echo_async(const unsigned char *req, size_t req_len, brix_ssi_responder_t *r)
60{
61 if (r->defer != NULL && r->defer(r) == 0) {
62 return 0; /* submit phase: deferred, completed later */
63 }
64 r->set_response(r, req, req_len, 1); /* completion phase: respond inline */
65 return 0;
66}

68/* stream-async: the deferred sibling of stream. On submit it defers; on
69 * completion it emits the response as three chunks (last=0,0,1), so the framework
70 * signals the client pendResp and the client drains the buffer via kXR_read. */
71static int
72svc_stream_async(const unsigned char *req, size_t req_len, brix_ssi_responder_t *r)
73{
74 (void) req;
75 (void) req_len;
76 if (r->defer != NULL && r->defer(r) == 0) {
77 return 0; /* submit phase: deferred */
78 }
79 r->set_response(r, (const unsigned char *) "part-A|", 7, 0);
80 r->set_response(r, (const unsigned char *) "part-B|", 7, 0);
81 r->set_response(r, (const unsigned char *) "part-C", 6, 1);
82 return 0;
83}
