33/* Whitelist-validate a hostname (1=ok, 0=reject). Allows letters, digits, '.',
34 * '-', and ':' (for IPv6 literals); bounded to RFC-1035's 253 chars. Reject,
35 * never sanitise — the value flows into the cluster registry and downstream
36 * connection logic. */
37int
38admin_validate_hostname(const char *host)
39{
40 size_t i, n;
41
42 if (host == NULL) {
43 return 0;
44 }
45 n = ngx_strlen(host);
46 if (n == 0 || n > 253) {
47 return 0;
48 }
49 for (i = 0; i < n; i++) {
50 unsigned char c = (unsigned char) host[i];
51 if (!((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
52 || (c >= '0' && c <= '9') || c == '.' || c == '-'
53 || c == ':')) /* ':' allowed for IPv6 literals */
54 {
55 return 0;
56 }
57 }
58 return 1;
59}

62/*
63 * WHAT: True (1) if `c` is an ASCII letter or digit, else 0.
64 * WHY: The three admin whitelist validators (hostname/paths/url) all share this
65 * alphanumeric base class; naming it once keeps each per-char predicate a
66 * flat OR of named checks instead of an inline char-range ladder (§8.6).
67 * HOW: Pure range test over the case letters and the digit run.
68 */
69static int
70admin_char_is_alnum(unsigned char c)
71{
72 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
73 || (c >= '0' && c <= '9');
74}

116/*
117 * WHAT: Authenticate an admin write request via two independent factors:
118 * a source-IP CIDR allowlist and a bearer-token secret.
119 * WHY: The admin API mutates cluster routing and proxy backends; it must fail
120 * closed. If neither factor is configured the API is treated as disabled
121 * (denied) so it is never inadvertently open.
122 * HOW: Compute cidr_ok and secret_ok independently. The bearer is compared in
123 * constant time (CRYPTO_memcmp) and length-checked first to avoid timing
124 * leaks. Combination rule:
125 * require_both ON -> every CONFIGURED factor must pass (AND);
126 * require_both OFF -> either configured factor passing is enough (OR).
127 */
128/*
129 * WHAT: True (1) if the request's source IP matches the admin CIDR allowlist.
130 * WHY: One of the two independent admin auth factors; naming it keeps the
131 * orchestrator flat and the sockaddr-null guard co-located with its use.
132 * HOW: Require a resolved peer sockaddr, then ngx_cidr_match against the
133 * configured allow array. Frozen: a missing sockaddr never matches.
134 */
135static int
136admin_auth_cidr_ok(ngx_http_request_t *r,
137 const ngx_http_brix_dashboard_loc_conf_t *conf)
138{
139 return r->connection->sockaddr != NULL
140 && ngx_cidr_match(r->connection->sockaddr, conf->admin_allow) == NGX_OK;
141}

144/*
145 * WHAT: True (1) if the request carries a bearer token equal to the admin
146 * secret, else 0.
147 * WHY: The second independent admin auth factor. Kept a pure predicate so the
148 * constant-time comparison and its length pre-gate live in one place.
149 * HOW: Extract the bearer, length-gate before the constant-time compare so
150 * unequal lengths short-circuit without a content-dependent comparison,
151 * then CRYPTO_memcmp against the configured secret. Frozen: no Authorization
152 * header never matches.
153 */
154static int
155admin_auth_secret_ok(ngx_http_request_t *r,
156 const ngx_http_brix_dashboard_loc_conf_t *conf)
157{
158 ngx_str_t bearer;
159 ngx_str_t hdr;
160
161 if (r->headers_in.authorization == NULL) {
162 return 0;
163 }
164 hdr = r->headers_in.authorization->value;
165 return brix_http_extract_bearer(&hdr, &bearer) == NGX_OK
166 && bearer.len == conf->admin_secret.len
167 && bearer.len > 0
168 && CRYPTO_memcmp(bearer.data, conf->admin_secret.data, bearer.len) == 0;
169}

172/*
173 * WHAT: Combine the two configured/passed auth factors into a final verdict.
174 * WHY: Isolates the AND/OR combination rule from the factor evaluation so the
175 * verdict logic is reviewable on its own; behavior byte-frozen.
176 * HOW: require_both ON -> every CONFIGURED factor must pass (an unconfigured
177 * factor is not required); OFF -> any single configured factor passing is
178 * enough (OR).
179 */
180static brix_admin_auth_result_t
181admin_auth_combine(const ngx_http_brix_dashboard_loc_conf_t *conf,
182 int has_allow, int cidr_ok, int has_secret, int secret_ok)
183{
184 if (conf->admin_require_both) {
185 if (has_allow && !cidr_ok) return BRIX_ADMIN_AUTH_DENIED;
186 if (has_secret && !secret_ok) return BRIX_ADMIN_AUTH_DENIED;
187 return BRIX_ADMIN_AUTH_OK;
188 }
189 return (cidr_ok || secret_ok)
190 ? BRIX_ADMIN_AUTH_OK : BRIX_ADMIN_AUTH_DENIED;
191}

194brix_admin_auth_result_t
195brix_admin_check_auth(ngx_http_request_t *r,
196 const ngx_http_brix_dashboard_loc_conf_t *conf)
197{
198 int has_allow = (conf->admin_allow != NULL && conf->admin_allow->nelts > 0);
199 int has_secret = (conf->admin_secret.len > 0);
200 int cidr_ok = 0;
201 int secret_ok = 0;
202
203 if (!has_allow && !has_secret) {
204 return BRIX_ADMIN_AUTH_DENIED; /* admin API not configured */
205 }
206
207 if (has_allow && admin_auth_cidr_ok(r, conf)) {
208 cidr_ok = 1;
209 }
210 if (has_secret && admin_auth_secret_ok(r, conf)) {
211 secret_ok = 1;
212 }
213
214 return admin_auth_combine(conf, has_allow, cidr_ok, has_secret, secret_ok);
215}

232/*
233 * WHAT: Completion callback invoked by nginx once the admin request body is
234 * buffered. Reassembles the body, parses it as a JSON object, and invokes
235 * the per-route handler stashed in the request ctx.
236 * WHY: Body reading is async; this is the re-entry point after
237 * brix_admin_read_body returned NGX_DONE. It owns finalizing the request
238 * on every path (success or error) via ngx_http_finalize_request.
239 * HOW: Sum buffer sizes; if the body spilled to a temp file (not in memory) it
240 * exceeded client_body_buffer_size and is rejected as too large. Enforce
241 * ADMIN_MAX_BODY, then concatenate and json_loadb. Non-object JSON is
242 * rejected so handlers can assume json_object_get works.
243 */
244void
245brix_admin_body_callback(ngx_http_request_t *r)
246{
247 brix_admin_body_ctx_t *bctx;
248 ngx_chain_t *cl;
249 u_char *buf;
250 size_t total = 0, off = 0;
251 json_t *parsed;
252 json_error_t jerr;
253 ngx_int_t rc;
254
255 bctx = ngx_http_get_module_ctx(r, ngx_http_brix_dashboard_module);
256
257 if (r->request_body == NULL || r->request_body->bufs == NULL) {
258 ngx_http_finalize_request(r,
259 admin_send_error(r, NGX_HTTP_BAD_REQUEST, "empty_body"));
260 return;
261 }
262
263 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
264 if (ngx_buf_in_memory(cl->buf)) {
265 total += cl->buf->last - cl->buf->pos;
266 } else {
267 /* Body spilled to a temp file — reject (too large for admin JSON). */
268 ngx_http_finalize_request(r,
269 admin_send_error(r, NGX_HTTP_REQUEST_ENTITY_TOO_LARGE,
270 "body_too_large"));
271 return;
272 }
273 }
274
275 if (total == 0 || total > ADMIN_MAX_BODY) {
276 ngx_http_finalize_request(r,
277 admin_send_error(r, NGX_HTTP_REQUEST_ENTITY_TOO_LARGE,
278 "body_too_large"));
279 return;
280 }
281
282 buf = ngx_palloc(r->pool, total);
283 if (buf == NULL) {
284 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
285 return;
286 }
287 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
288 size_t n = cl->buf->last - cl->buf->pos;
289 ngx_memcpy(buf + off, cl->buf->pos, n);
290 off += n;
291 }
292
293 parsed = json_loadb((char *) buf, total, 0, &jerr);
294 if (parsed == NULL || !json_is_object(parsed)) {
295 if (parsed != NULL) { json_decref(parsed); }
296 ngx_http_finalize_request(r,
297 admin_send_error(r, NGX_HTTP_BAD_REQUEST, "invalid_json"));
298 return;
299 }
300
301 rc = bctx->handler(r, parsed);
302 json_decref(parsed);
303 ngx_http_finalize_request(r, rc);
304}

308 * Kick off async body reading for a body-bearing admin route. Stashes `handler`
309 * in the request ctx so the shared body callback can dispatch to it once the
310 * body is available. Returns NGX_DONE on the normal async path (the callback
311 * will finalize), or an error status if nginx failed to start the read.

313ngx_int_t
314brix_admin_read_body(ngx_http_request_t *r,

316{

350/*
351 * WHAT: Match the required "http://" or "https://" scheme prefix; on success
352 * store the offset of the first host byte in *off and return 1, else 0.
353 * WHY: Isolating the scheme decision keeps admin_validate_url a flat sequence
354 * of named guards; the two-arm scheme test is the bulk of its branching.
355 * HOW: Longest-prefix first (https before http) so "https://" is not
356 * mis-parsed as "http://" + a leading 's'; offsets frozen at 8 / 7.
357 */
358static int
359admin_url_scheme_off(const char *url, size_t *off)
360{
361 if (ngx_strncmp(url, "https://", 8) == 0) {
362 *off = 8;
363 return 1;
364 }
365 if (ngx_strncmp(url, "http://", 7) == 0) {
366 *off = 7;
367 return 1;
368 }
369 return 0;
370}

373/*
374 * Whitelist-validate a backend URL: "http://" or "https://" followed by a
375 * host[:port][/path] composed only of safe characters. Reject — never
376 * sanitise — before handing the string to ngx_parse_url() / the SHM pool.
377 */
378int
379admin_validate_url(const char *url)
380{
381 size_t i, n, off = 0;
382 const char *host;
383
384 if (url == NULL) {
385 return 0;
386 }
387 n = ngx_strlen(url);
388 if (n < 9 || n >= 512) { /* shortest is "http://h" + something */
389 return 0;
390 }
391 if (!admin_url_scheme_off(url, &off)) {
392 return 0;
393 }
394 host = url + off;
395 if (*host == '\0' || *host == '/' || *host == ':') {
396 return 0; /* empty host */
397 }
398 for (i = off; i < n; i++) {
399 if (!admin_url_char_ok((unsigned char) url[i])) {
400 return 0;
401 }
402 }
403 return 1;
404}

445 * returns the response status when the URI is this endpoint; leaves

477 * WHAT: Route the "/cluster/servers/{host}/{port}[/action]" sub-resource. Sets
478 * *matched per whether the URI is under that prefix.
479 * WHY: Isolates the per-server sub-action ladder (drain/undrain/delete/upsert)

579/*
580 * WHAT: Type of an admin endpoint-group router: on a URI match it sets *matched
581 * and returns the response status; otherwise it clears *matched.
582 * WHY: Lets brix_admin_dispatch drive the endpoint groups as a static table
583 * (§8.6 table-driven dispatch) instead of a branch ladder.
584 * HOW: Uniform (request, matched) -> status signature shared by every group.
585 */
586typedef ngx_int_t (*admin_route_fn)(ngx_http_request_t *r, int *matched);

589ngx_int_t
590brix_admin_dispatch(ngx_http_request_t *r)
591{
592 static const admin_route_fn routes[] = {
593 admin_route_io_uring,
594 admin_route_cluster,
595 admin_route_proxy,
596 };
597 ngx_http_brix_dashboard_loc_conf_t *conf;
598 ngx_uint_t i;
599

601

603 admin_audit(r, "auth", NULL, "forbidden");
604 return admin_send_error(r, NGX_HTTP_FORBIDDEN, "forbidden");
605 }
606
607 for (i = 0; i < sizeof(routes) / sizeof(routes[0]); i++) {
608 int matched = 0;
609 ngx_int_t rc = routes[i](r, &matched);
610 if (matched) {
611 return rc;
612 }
613 }
614
615 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
616}
