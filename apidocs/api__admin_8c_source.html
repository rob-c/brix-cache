33/* Whitelist-validate a hostname (1=ok, 0=reject). Allows letters, digits, '.',
34 * '-', and ':' (for IPv6 literals); bounded to RFC-1035's 253 chars. Reject,
35 * never sanitise — the value flows into the cluster registry and downstream
36 * connection logic. */
37int
38admin_validate_hostname(const char *host)
39{
40 size_t i, n;
41
42 if (host == NULL) {
43 return 0;
44 }
45 n = ngx_strlen(host);
46 if (n == 0 || n > 253) {
47 return 0;
48 }
49 for (i = 0; i < n; i++) {
50 unsigned char c = (unsigned char) host[i];
51 if (!((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
52 || (c >= '0' && c <= '9') || c == '.' || c == '-'
53 || c == ':')) /* ':' allowed for IPv6 literals */
54 {
55 return 0;
56 }
57 }
58 return 1;
59}

90/*
91 * WHAT: Authenticate an admin write request via two independent factors:
92 * a source-IP CIDR allowlist and a bearer-token secret.
93 * WHY: The admin API mutates cluster routing and proxy backends; it must fail
94 * closed. If neither factor is configured the API is treated as disabled
95 * (denied) so it is never inadvertently open.
96 * HOW: Compute cidr_ok and secret_ok independently. The bearer is compared in
97 * constant time (CRYPTO_memcmp) and length-checked first to avoid timing
98 * leaks. Combination rule:
99 * require_both ON -> every CONFIGURED factor must pass (AND);
100 * require_both OFF -> either configured factor passing is enough (OR).
101 */
102brix_admin_auth_result_t
103brix_admin_check_auth(ngx_http_request_t *r,
104 const ngx_http_brix_dashboard_loc_conf_t *conf)
105{
106 int has_allow = (conf->admin_allow != NULL && conf->admin_allow->nelts > 0);
107 int has_secret = (conf->admin_secret.len > 0);
108 int cidr_ok = 0;
109 int secret_ok = 0;
110
111 if (!has_allow && !has_secret) {
112 return BRIX_ADMIN_AUTH_DENIED; /* admin API not configured */
113 }
114
115 if (has_allow && r->connection->sockaddr != NULL
116 && ngx_cidr_match(r->connection->sockaddr, conf->admin_allow) == NGX_OK)
117 {
118 cidr_ok = 1;
119 }
120
121 /* Bearer check: length-gate before the constant-time compare so unequal
122 * lengths short-circuit without a content-dependent comparison. */
123 if (has_secret && r->headers_in.authorization != NULL) {
124 ngx_str_t bearer;
125 ngx_str_t hdr = r->headers_in.authorization->value;
126 if (brix_http_extract_bearer(&hdr, &bearer) == NGX_OK
127 && bearer.len == conf->admin_secret.len
128 && bearer.len > 0
129 && CRYPTO_memcmp(bearer.data, conf->admin_secret.data,
130 bearer.len) == 0)
131 {
132 secret_ok = 1;
133 }
134 }
135
136 /* AND mode: a configured factor that fails is fatal; an unconfigured factor
137 * is simply not required. */
138 if (conf->admin_require_both) {
139 if (has_allow && !cidr_ok) return BRIX_ADMIN_AUTH_DENIED;
140 if (has_secret && !secret_ok) return BRIX_ADMIN_AUTH_DENIED;
141 return BRIX_ADMIN_AUTH_OK;
142 }
143 /* OR mode: any single configured factor passing grants access. */
144 return (cidr_ok || secret_ok)
145 ? BRIX_ADMIN_AUTH_OK : BRIX_ADMIN_AUTH_DENIED;
146}

163/*
164 * WHAT: Completion callback invoked by nginx once the admin request body is
165 * buffered. Reassembles the body, parses it as a JSON object, and invokes
166 * the per-route handler stashed in the request ctx.
167 * WHY: Body reading is async; this is the re-entry point after
168 * brix_admin_read_body returned NGX_DONE. It owns finalizing the request
169 * on every path (success or error) via ngx_http_finalize_request.
170 * HOW: Sum buffer sizes; if the body spilled to a temp file (not in memory) it
171 * exceeded client_body_buffer_size and is rejected as too large. Enforce
172 * ADMIN_MAX_BODY, then concatenate and json_loadb. Non-object JSON is
173 * rejected so handlers can assume json_object_get works.
174 */
175void
176brix_admin_body_callback(ngx_http_request_t *r)
177{
178 brix_admin_body_ctx_t *bctx;
179 ngx_chain_t *cl;
180 u_char *buf;
181 size_t total = 0, off = 0;
182 json_t *parsed;
183 json_error_t jerr;
184 ngx_int_t rc;
185
186 bctx = ngx_http_get_module_ctx(r, ngx_http_brix_dashboard_module);
187
188 if (r->request_body == NULL || r->request_body->bufs == NULL) {
189 ngx_http_finalize_request(r,
190 admin_send_error(r, NGX_HTTP_BAD_REQUEST, "empty_body"));
191 return;
192 }
193
194 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
195 if (ngx_buf_in_memory(cl->buf)) {
196 total += cl->buf->last - cl->buf->pos;
197 } else {
198 /* Body spilled to a temp file — reject (too large for admin JSON). */
199 ngx_http_finalize_request(r,
200 admin_send_error(r, NGX_HTTP_REQUEST_ENTITY_TOO_LARGE,
201 "body_too_large"));
202 return;
203 }
204 }
205
206 if (total == 0 || total > ADMIN_MAX_BODY) {
207 ngx_http_finalize_request(r,
208 admin_send_error(r, NGX_HTTP_REQUEST_ENTITY_TOO_LARGE,
209 "body_too_large"));
210 return;
211 }
212
213 buf = ngx_palloc(r->pool, total);
214 if (buf == NULL) {
215 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
216 return;
217 }
218 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
219 size_t n = cl->buf->last - cl->buf->pos;
220 ngx_memcpy(buf + off, cl->buf->pos, n);
221 off += n;
222 }
223
224 parsed = json_loadb((char *) buf, total, 0, &jerr);
225 if (parsed == NULL || !json_is_object(parsed)) {
226 if (parsed != NULL) { json_decref(parsed); }
227 ngx_http_finalize_request(r,
228 admin_send_error(r, NGX_HTTP_BAD_REQUEST, "invalid_json"));
229 return;
230 }
231
232 rc = bctx->handler(r, parsed);
233 json_decref(parsed);
234 ngx_http_finalize_request(r, rc);
235}

238/*
239 * Kick off async body reading for a body-bearing admin route. Stashes `handler`
240 * in the request ctx so the shared body callback can dispatch to it once the
241 * body is available. Returns NGX_DONE on the normal async path (the callback
242 * will finalize), or an error status if nginx failed to start the read.
243 */
244ngx_int_t
245brix_admin_read_body(ngx_http_request_t *r,
246 brix_admin_body_handler_t handler)
247{
248 brix_admin_body_ctx_t *bctx;
249 ngx_int_t rc;
250
251 bctx = ngx_palloc(r->pool, sizeof(*bctx));
252 if (bctx == NULL) {
253 return NGX_HTTP_INTERNAL_SERVER_ERROR;
254 }
255 bctx->handler = handler;
256 ngx_http_set_ctx(r, bctx, ngx_http_brix_dashboard_module);
257
258 rc = ngx_http_read_client_request_body(r, brix_admin_body_callback);
259 if (rc >= NGX_HTTP_SPECIAL_RESPONSE) {
260 return rc;
261 }
262 return NGX_DONE;
263}

266/*
267 * Whitelist-validate a backend URL: "http://" or "https://" followed by a
268 * host[:port][/path] composed only of safe characters. Reject — never
269 * sanitise — before handing the string to ngx_parse_url() / the SHM pool.
270 */
271int
272admin_validate_url(const char *url)
273{
274 size_t i, n, off;
275 const char *host;
276
277 if (url == NULL) {
278 return 0;
279 }
280 n = ngx_strlen(url);
281 if (n < 9 || n >= 512) { /* shortest is "http://h" + something */
282 return 0;
283 }
284 if (ngx_strncmp(url, "https://", 8) == 0) {
285 off = 8;
286 } else if (ngx_strncmp(url, "http://", 7) == 0) {
287 off = 7;
288 } else {
289 return 0;
290 }
291 host = url + off;
292 if (*host == '\0' || *host == '/' || *host == ':') {
293 return 0; /* empty host */
294 }
295 for (i = off; i < n; i++) {
296 unsigned char c = (unsigned char) url[i];
297 if (!((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
298 || (c >= '0' && c <= '9') || c == '.' || c == '-'
299 || c == '_' || c == ':' || c == '/'))
300 {
301 return 0;
302 }
303 }
304 return 1;
305}

356 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");

360 sizeof(ADMIN_PREFIX "cluster/servers/") - 1) == 0)

383 }
384

386 if (r->method == NGX_HTTP_POST) {
387 return brix_admin_read_body(r, admin_proxy_add);

389 if (r->method == NGX_HTTP_GET) {
390 return admin_proxy_list(r);
391 }
392 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
393 }
394 if (r->uri.len > sizeof(ADMIN_PREFIX "proxy/backends/") - 1
395 && ngx_strncmp(r->uri.data, ADMIN_PREFIX "proxy/backends/",
396 sizeof(ADMIN_PREFIX "proxy/backends/") - 1) == 0)
397 {
398 uint32_t id = 0;
399 char action[16];
400 if (admin_parse_proxy_uri(r, &id, action, sizeof(action)) != NGX_OK) {
401 return admin_send_error(r, NGX_HTTP_BAD_REQUEST, "bad_uri");
402 }
403 return admin_proxy_one(r, action, id);
404 }
405
406 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
407}
