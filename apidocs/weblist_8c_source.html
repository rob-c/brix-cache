1/*
2 * weblist.c — list the files under a WebDAV collection (for recursive web copy).
3 *
4 * WHAT: brix_webdav_list() issues one PROPFIND Depth: infinity against a davs/http
5 * collection and returns the absolute server paths of every FILE beneath it
6 * (collections themselves are skipped — subdirs are recreated locally from the
7 * file paths).
8 * WHY: The XRootD/WebDAV wire has no "give me the tree" transfer op, so `xrdcp -r`
9 * over davs:// must enumerate the collection itself and copy each file. Keeping
10 * the enumeration here (a new file) leaves the copy engine untouched.
11 * HOW: PROPFIND over the existing HTTP client; scan the multistatus body for each
12 * <D:response> block, take its <D:href> and treat it as a file unless the block
13 * carries <D:collection/>. hrefs are percent-decoded; an absolute-URL href is
14 * reduced to its path. Bounded entry count.
15 *
16 * Clean-room: parses the documented WebDAV multistatus shape this module emits
17 * (src/protocols/webdav/propfind.c), not any client library.
18 */

30/* Page-invariant inputs for one ListObjectsV2 listing.
31 *
32 * WHAT: Holds everything that stays constant across the paginated GETs — bucket
33 * name, pre-encoded prefix, SigV4 signing host + credentials + region — so the
34 * per-page request builder takes one context pointer instead of a long arg list.
35 * WHY: brix_s3_list's extern signature is frozen (declared in brix_net.h); the
36 * consolidation is confined to this file's static page machinery.
37 * HOW: Populated once in brix_s3_list, then passed by const pointer to
38 * s3_list_build_request for each page.
39 */

42 char encp[XRDC_PATH_MAX * 3]; /* pre-encoded object prefix */
43 char hostport[300]; /* bracketed Host, byte-matching the wire */
44 int have_prefix;
45 const char *ak;
46 const char *sk;
47 const char *region;
48} s3_list_req_t;

50/* Growable string-vector accumulator shared by the page/block scanners.
51 *
52 * WHAT: Bundles the (array, count, capacity) triple that push() grows so the
53 * scanners pass one accumulator pointer instead of three out-params.
54 * WHY: Both the S3 key scan and the WebDAV block scan feed the same array; a single
55 * struct keeps every helper under the 5-parameter gate.
56 * HOW: Zero-initialised by the public entry, threaded by pointer into every scanner,
57 * and its arr/n handed to brix_strv_free on error and to *keys / *paths on success.
58 */
59typedef struct {
60 char **arr;
61 size_t n;
62 size_t cap;
63} weblist_acc_t;

65/* Caller-owned output buffers for one built request (path + signed headers).
66 *
67 * WHAT: Carries the destination path and header buffers (with their sizes) for
68 * s3_list_build_request, so it stays within the parameter gate.
69 * WHY: brix_s3_list's signature is frozen; the buffer plumbing is confined here.
70 * HOW: Points at the paging loop's stack buffers; filled per page then consumed by
71 * brix_http_req.

80/* Append a strdup'd copy of s to the accumulator's growable array. 0 / -1. */
81static int
82push(weblist_acc_t *acc, const char *s)
83{
84 if (acc->n == acc->cap) {
85 size_t nc = acc->cap ? acc->cap * 2 : 64;
86 char **na = (char **) realloc(acc->arr, nc * sizeof(char *));
87 if (na == NULL) { return -1; }
88 acc->arr = na;
89 acc->cap = nc;
90 }
91 acc->arr[acc->n] = strdup(s);
92 if (acc->arr[acc->n] == NULL) { return -1; }
93 acc->n++;
94 return 0;
95}

106/* Copy the text between <tag> and </tag> (first occurrence at/after *p) into out;
107 * advances *p past the close tag (even on truncation, so a scan loop can continue
108 * past an over-long value). Returns 1 if found and it fit, 0 if not found, -1 if
109 * found but too long for out (out holds the truncated prefix). Callers that must
110 * not act on a partial value (e.g. a continuation token) treat -1 as a hard error. */
111static int
112xml_tag(const char **p, const char *otag, const char *ctag, char *out, size_t osz)
113{
114 const char *s = strstr(*p, otag);
115 const char *e;
116 size_t n;
117 int truncated = 0;
118 if (s == NULL) { return 0; }
119 s += strlen(otag);
120 e = strstr(s, ctag);
121 if (e == NULL) { return 0; }
122 n = (size_t) (e - s);
123 if (n >= osz) { n = osz - 1; truncated = 1; }
124 memcpy(out, s, n);
125 out[n] = '\0';
126 *p = e + strlen(ctag);
127 return truncated ? -1 : 1;
128}

130/* Build the SORTED, RFC-3986-encoded ListObjectsV2 canonical query string into out
131 * (params: continuation-token < list-type < prefix). encp = pre-encoded prefix.
132 * Every append is length-checked. 0 on success, -1 if it would not fit. */
133static int
134s3_canon_qs(const char *token, const char *encp, int have_prefix,
135 char *out, size_t outsz)
136{
137 char enctok[8192];
138 size_t qn = 0;
139 int w;
140
141 if (token[0] != '\0') {
142 if (brix_http_urlencode((const unsigned char *) token, strlen(token),
143 enctok, sizeof(enctok), "") < 0) {
144 return -1;
145 }
146 w = snprintf(out + qn, outsz - qn, "continuation-token=%s&", enctok);
147 if (w < 0 || (size_t) w >= outsz - qn) { return -1; }
148 qn += (size_t) w;
149 }
150 w = snprintf(out + qn, outsz - qn, "list-type=2");
151 if (w < 0 || (size_t) w >= outsz - qn) { return -1; }
152 qn += (size_t) w;
153 if (have_prefix) {
154 w = snprintf(out + qn, outsz - qn, "&prefix=%s", encp);
155 if (w < 0 || (size_t) w >= outsz - qn) { return -1; }
156 qn += (size_t) w;
157 }
158 return 0;
159}

161/* ---- Split an s3:// list path into bucket + object-prefix ----
162 *
163 * WHAT: Parses u->path ("/bucket[/prefix...]") into bucket[] and prefix[]. Returns
164 * 0 on success, -1 (with *st set to XRDC_EUSAGE) if the path is not rooted or
165 * the bucket segment does not fit bucket[].
166 * WHY: ListObjectsV2 addresses a bucket in the path and the object prefix in the
167 * query string, so the two components must be separated before signing.
168 * HOW: 1. Require a leading '/'. 2. Find the first '/' after it: absent → the whole
169 * tail is the bucket and prefix is empty; present → copy the bucket segment
170 * (length-checked into req->bucket) and take everything after it as the prefix.
171 */
172static int
173s3_list_split_path(const brix_weburl *u, s3_list_req_t *req,
174 char *prefix, size_t prefixsz, brix_status *st)
175{
176 const char *bsl;
177
178 if (u->path[0] != '/') {
179 brix_status_set(st, XRDC_EUSAGE, 0, "s3 list: bad path");
180 return -1;
181 }
182 bsl = strchr(u->path + 1, '/');
183 if (bsl == NULL) {
184 snprintf(req->bucket, sizeof(req->bucket), "%s", u->path + 1);
185 prefix[0] = '\0';
186 return 0;
187 }
188 {
189 size_t bl = (size_t) (bsl - (u->path + 1));
190 if (bl >= sizeof(req->bucket)) {
191 brix_status_set(st, XRDC_EUSAGE, 0, "s3: bucket too long");
192 return -1;
193 }
194 memcpy(req->bucket, u->path + 1, bl);
195 req->bucket[bl] = '\0';
196 }
197 snprintf(prefix, prefixsz, "%s", bsl + 1);
198 return 0;
199}

201/* ---- Build + sign the request path for one ListObjectsV2 page ----
202 *
203 * WHAT: Renders the canonical URI ("/bucket") and query string (from the current
204 * continuation token) into path[], and, when credentials are supplied, the
205 * SigV4 Authorization headers into hdrs[]. Returns 0, or -1 (with *st set) on
206 * overflow (XRDC_EUSAGE) or a signing failure (XRDC_EAUTH).
207 * WHY: Each page is a distinct signed GET; isolating the per-page URL/signature
208 * assembly keeps the paging loop to control flow only.
209 * HOW: 1. Assemble canon_uri "/bucket". 2. Build the sorted canonical query string
210 * (token < list-type < prefix). 3. Compose "uri?qs" into path[]. 4. If ak/sk
211 * are both non-empty, sign the GET over the same bracketed host used on the
212 * wire and emit the Authorization headers; otherwise leave hdrs empty.
213 */
214static int
215s3_list_build_request(const s3_list_req_t *req, const char *token,
216 const s3_list_out_t *out, brix_status *st)
217{
218 char canon_uri[300], canon_qs[XRDC_PATH_MAX * 4];
219
220 snprintf(canon_uri, sizeof(canon_uri), "/%s", req->bucket);
221 if (s3_canon_qs(token, req->encp, req->have_prefix, canon_qs, sizeof(canon_qs)) != 0
222 || (size_t) snprintf(out->path, out->pathsz, "%s?%s", canon_uri, canon_qs)
223 >= out->pathsz) {
224 brix_status_set(st, XRDC_EUSAGE, 0, "s3 list: prefix/continuation too long");
225 return -1;
226 }
227 out->hdrs[0] = '\0';
228 if (req->ak != NULL && req->sk != NULL && req->ak[0] != '\0' && req->sk[0] != '\0') {
229 if (brix_s3_sign_v4_q("GET", req->hostport, canon_uri, canon_qs, req->ak,
230 req->sk, req->region, "UNSIGNED-PAYLOAD",
231 out->hdrs, out->hdrssz) != 0) {
232 brix_status_set(st, XRDC_EAUTH, 0, "s3 list: failed to sign request");
233 return -1;
234 }
235 }
236 return 0;
237}

239/* ---- Append every <Key> in one ListObjectsV2 page body to the result array ----
240 *
241 * WHAT: Scans r->body for <Key>...</Key> elements and pushes each onto *arr / *n / *cap
242 * until the tag stream ends or XRDC_WEBLIST_MAX is reached. Returns 0, or -1
243 * (with *st set to XRDC_EPROTO) on an over-long key or an allocation failure.
244 * WHY: Object keys are the payload of the listing; keeping the scan separate frees
245 * the pager from the per-key error handling.
246 * HOW: 1. Loop xml_tag over the body. 2. Stop cleanly at end-of-tags or the cap.
247 * 3. A truncated key (kr < 0) is fatal — a clipped key yields a wrong URL.
248 * 4. push() each key; propagate an OOM as a hard error.
249 */
250static int
251s3_list_scan_keys(const brix_http_resp *r, weblist_acc_t *acc, brix_status *st)
252{
253 const char *p = r->body ? r->body : "";
254
255 for (;;) {
256 char key[XRDC_PATH_MAX];
257 int kr = xml_tag(&p, "<Key>", "</Key>", key, sizeof(key));
258 if (kr == 0 || acc->n >= XRDC_WEBLIST_MAX) {
259 break;
260 }
261 if (kr < 0) {
262 /* A key longer than our buffer would yield a wrong download URL —
263 * fail loudly rather than silently dropping or mangling it. */
264 brix_status_set(st, XRDC_EPROTO, 0,
265 "s3 list: object key exceeds %zu bytes", sizeof(key));
266 return -1;
267 }
268 if (push(acc, key) != 0) {
269 brix_status_set(st, XRDC_EPROTO, 0, "s3 list: out of memory");
270 return -1;
271 }
272 }
273 return 0;
274}

276/* ---- Extract the continuation token for the next ListObjectsV2 page ----
277 *
278 * WHAT: Reads IsTruncated / NextContinuationToken from r->body into token[] and sets
279 * *more to 1 when another page follows. Returns 0, or -1 (with *st set to
280 * XRDC_EPROTO) if the token is present but longer than token[].
281 * WHY: Pagination correctness hinges on carrying the full token forward; a clipped
282 * token would silently corrupt the listing, so truncation must be fatal.
283 * HOW: 1. Reset token to empty and *more to 0. 2. Only when IsTruncated == "true"
284 * read NextContinuationToken. 3. A truncated token is a hard error; otherwise
285 * *more reflects whether a token was found.
286 */
287static int
288s3_list_next_token(const brix_http_resp *r, char *token, size_t tokensz,
289 int *more, brix_status *st)
290{
291 const char *tp = r->body ? r->body : "";
292 char trunc[8];
293
294 token[0] = '\0';
295 *more = 0;
296 if (xml_tag(&tp, "<IsTruncated>", "</IsTruncated>", trunc, sizeof(trunc)) == 1
297 && strcmp(trunc, "true") == 0) {
298 const char *np = r->body;
299 int tr = xml_tag(&np, "<NextContinuationToken>",
300 "</NextContinuationToken>", token, tokensz);
301 if (tr < 0) {
302 /* token truncated → continuing would corrupt the listing */
303 brix_status_set(st, XRDC_EPROTO, 0,
304 "s3 list: continuation token too long");
305 return -1;
306 }
307 *more = (tr == 1);
308 }
309 return 0;
310}

312int
313brix_s3_list(const brix_weburl *u, const char *ak, const char *sk,
314 const char *region, int verify, const char *ca_dir,
315 char ***keys, size_t *n_out, brix_status *st)
316{
317 s3_list_req_t req;
318 weblist_acc_t acc = { NULL, 0, 0 };
319 char prefix[XRDC_PATH_MAX];
320 /* The server's NextContinuationToken is the b64url of the last returned key
321 * (keys up to S3_MAX_KEY=4096 → token up to ~5.5 KB), so this MUST be large
322 * enough or pagination silently corrupts. */
323 char token[8192];
324 int rounds = 0;
325
326 memset(&req, 0, sizeof(req));
327 *keys = NULL;
328 *n_out = 0;
329 if (region == NULL || region[0] == '\0') { region = "us-east-1"; }
330 req.ak = ak;
331 req.sk = sk;
332 req.region = region;
333 /* u->path = "/bucket[/prefix...]" → split bucket vs prefix. */
334 if (s3_list_split_path(u, &req, prefix, sizeof(prefix), st) != 0) {
335 return -1;
336 }
337 /* The SigV4 signed host MUST match the wire Host header byte-for-byte; that
338 * header is built bracketed for IPv6 literals ([::1]:9000), so sign the same. */
339 brix_format_host_port(u->host, (uint16_t) u->port, req.hostport, sizeof(req.hostport));
340 if (brix_http_urlencode((const unsigned char *) prefix, strlen(prefix),
341 req.encp, sizeof(req.encp), "") < 0) {
342 brix_status_set(st, XRDC_EUSAGE, 0, "s3: prefix too long");
343 return -1;
344 }
345 req.have_prefix = (prefix[0] != '\0');
346 token[0] = '\0';
347
348 do {
349 char path[XRDC_PATH_MAX * 5];
350 char hdrs[2048];
351 s3_list_out_t out = { path, sizeof(path), hdrs, sizeof(hdrs) };
352 brix_http_resp r;
353 int more = 0;
354
355 if (++rounds > 100000) {
356 brix_strv_free(acc.arr, acc.n);
357 brix_status_set(st, XRDC_EPROTO, 0, "s3 list: too many pages");
358 return -1;
359 }
360 if (s3_list_build_request(&req, token, &out, st) != 0) {
361 brix_strv_free(acc.arr, acc.n);
362 return -1;
363 }
364 if (brix_http_req(u->host, u->port, u->tls, "GET", path, hdrs[0] ? hdrs : NULL,
365 NULL, 0, XRDC_WEBLIST_TIMEOUT, verify, ca_dir, &r, st) != 0) {
366 brix_strv_free(acc.arr, acc.n);
367 return -1;
368 }
369 if (r.status != 200) {
370 brix_status_set(st, XRDC_EPROTO, 0, "s3 ListObjectsV2 returned HTTP %d", r.status);
371 brix_http_resp_free(&r);
372 brix_strv_free(acc.arr, acc.n);
373 return -1;
374 }
375 if (s3_list_scan_keys(&r, &acc, st) != 0
376 || s3_list_next_token(&r, token, sizeof(token), &more, st) != 0) {
377 brix_http_resp_free(&r);
378 brix_strv_free(acc.arr, acc.n);
379 return -1;
380 }
381 brix_http_resp_free(&r);
382 if (!more) { break; }
383 } while (1);
384
385 *keys = acc.arr;
386 *n_out = acc.n;
387 return 0;
388}

438/* ---- Append the file path of one <D:response> block, if it is a file ----
439 *
440 * WHAT: Given a single multistatus response block [p, end), extracts its <D:href>,
441 * skips it when the block carries <D:collection/>, and otherwise pushes the
442 * percent-decoded path onto *arr / *n / *cap. Returns 0 (including for skipped
443 * collections and hrefless/malformed blocks), or -1 (with *st set to
444 * XRDC_EPROTO) on an allocation failure.
445 * WHY: The response scan reduces to "per block, emit its file"; pulling the block
446 * logic out keeps the scan loop to control flow only.
447 * HOW: 1. Locate <D:href>..</D:href> inside the block; absent → nothing to emit.
448 * 2. If the block also holds <D:collection>, it is a directory → skip. 3. Else
449 * percent-decode the href (flags=0 keeps a literal '+'), reduce an absolute-URL
450 * href to its path component, and push it.
451 */
452static int
453webdav_list_emit_block(const char *p, const char *end, weblist_acc_t *acc,
454 brix_status *st)
455{
456 const char *h = strstr(p, "<D:href>");
457 const char *he;
458 const char *col;
459 char href[XRDC_PATH_MAX];
460 const char *path = href;
461
462 if (h == NULL || h >= end) {
463 return 0;
464 }
465 h += 8;
466 he = strstr(h, "</D:href>");
467 if (he == NULL || he >= end) {
468 return 0;
469 }
470 col = strstr(p, "<D:collection");
471 if (col != NULL && col < end) {
472 return 0; /* directory — subdirs are recreated from the file paths */
473 }
474 /* flags=0: keep a literal '+' (it is a real path byte in an href, not a
475 * form-encoded space). */
476 if (brix_http_urldecode((const unsigned char *) h, (size_t) (he - h),
477 href, sizeof(href), 0) != BRIX_URLDECODE_OK) {
478 href[0] = '\0'; /* overflow/malformed → skip content */
479 }
480 /* reduce an absolute-URL href to its path component */
481 if (strstr(href, "://") != NULL) {
482 char *sl = strstr(href, "://");
483 char *ps = strchr(sl + 3, '/');
484 path = ps ? ps : "/";
485 }
486 if (push(acc, path) != 0) {
487 brix_status_set(st, XRDC_EPROTO, 0, "webdav list: out of memory");
488 return -1;
489 }
490 return 0;
491}

493/* ---- Scan a PROPFIND multistatus body, collecting every file path ----
494 *
495 * WHAT: Walks each <D:response>..</D:response> block in body and appends the file
496 * paths onto *arr / *n / *cap, up to XRDC_WEBLIST_MAX. Returns 0, or -1 (with *st
497 * set) on an allocation failure inside a block.
498 * WHY: Separates the block-boundary iteration from the per-block extraction so the
499 * public entry stays a thin request/dispatch shell.
500 * HOW: 1. Find each "<D:response" and its closing "</D:response>"; stop at an
501 * unterminated block or the entry cap. 2. Hand the block to
502 * webdav_list_emit_block. 3. Advance past the close tag.
503 */
504static int
505webdav_list_scan_body(const char *body, weblist_acc_t *acc, brix_status *st)
506{
507 const char *p = body ? body : "";
508
509 while ((p = strstr(p, "<D:response")) != NULL && acc->n < XRDC_WEBLIST_MAX) {
510 const char *end = strstr(p, "</D:response>");
511 if (end == NULL) {
512 break;
513 }
514 if (webdav_list_emit_block(p, end, acc, st) != 0) {
515 return -1;
516 }
517 p = end + 12;
518 }
519 return 0;
520}
