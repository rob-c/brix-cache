1/*
2 * weblist.c — list the files under a WebDAV collection (for recursive web copy).
3 *
4 * WHAT: brix_webdav_list() issues one PROPFIND Depth: infinity against a davs/http
5 * collection and returns the absolute server paths of every FILE beneath it
6 * (collections themselves are skipped — subdirs are recreated locally from the
7 * file paths).
8 * WHY: The XRootD/WebDAV wire has no "give me the tree" transfer op, so `xrdcp -r`
9 * over davs:// must enumerate the collection itself and copy each file. Keeping
10 * the enumeration here (a new file) leaves the copy engine untouched.
11 * HOW: PROPFIND over the existing HTTP client; scan the multistatus body for each
12 * <D:response> block, take its <D:href> and treat it as a file unless the block
13 * carries <D:collection/>. hrefs are percent-decoded; an absolute-URL href is
14 * reduced to its path. Bounded entry count.
15 *
16 * Clean-room: parses the documented WebDAV multistatus shape this module emits
17 * (src/protocols/webdav/propfind.c), not any client library.
18 */

30/* Append a strdup'd copy of [s,s+len) to a growable array. 0 / -1. */
31static int
32push(char ***arr, size_t *n, size_t *cap, const char *s)
33{
34 if (*n == *cap) {
35 size_t nc = *cap ? *cap * 2 : 64;
36 char **na = (char **) realloc(*arr, nc * sizeof(char *));
37 if (na == NULL) { return -1; }
38 *arr = na;
39 *cap = nc;
40 }
41 (*arr)[*n] = strdup(s);
42 if ((*arr)[*n] == NULL) { return -1; }
43 (*n)++;
44 return 0;
45}

56/* Copy the text between <tag> and </tag> (first occurrence at/after *p) into out;
57 * advances *p past the close tag (even on truncation, so a scan loop can continue
58 * past an over-long value). Returns 1 if found and it fit, 0 if not found, -1 if
59 * found but too long for out (out holds the truncated prefix). Callers that must
60 * not act on a partial value (e.g. a continuation token) treat -1 as a hard error. */
61static int
62xml_tag(const char **p, const char *otag, const char *ctag, char *out, size_t osz)
63{
64 const char *s = strstr(*p, otag);
65 const char *e;
66 size_t n;
67 int truncated = 0;
68 if (s == NULL) { return 0; }
69 s += strlen(otag);
70 e = strstr(s, ctag);
71 if (e == NULL) { return 0; }
72 n = (size_t) (e - s);
73 if (n >= osz) { n = osz - 1; truncated = 1; }
74 memcpy(out, s, n);
75 out[n] = '\0';
76 *p = e + strlen(ctag);
77 return truncated ? -1 : 1;
78}

80/* Build the SORTED, RFC-3986-encoded ListObjectsV2 canonical query string into out
81 * (params: continuation-token < list-type < prefix). encp = pre-encoded prefix.
82 * Every append is length-checked. 0 on success, -1 if it would not fit. */
83static int
84s3_canon_qs(const char *token, const char *encp, int have_prefix,
85 char *out, size_t outsz)
86{
87 char enctok[8192];
88 size_t qn = 0;
89 int w;
90
91 if (token[0] != '\0') {
92 if (brix_http_urlencode((const unsigned char *) token, strlen(token),
93 enctok, sizeof(enctok), "") < 0) {
94 return -1;
95 }
96 w = snprintf(out + qn, outsz - qn, "continuation-token=%s&", enctok);
97 if (w < 0 || (size_t) w >= outsz - qn) { return -1; }
98 qn += (size_t) w;
99 }
100 w = snprintf(out + qn, outsz - qn, "list-type=2");
101 if (w < 0 || (size_t) w >= outsz - qn) { return -1; }
102 qn += (size_t) w;
103 if (have_prefix) {
104 w = snprintf(out + qn, outsz - qn, "&prefix=%s", encp);
105 if (w < 0 || (size_t) w >= outsz - qn) { return -1; }
106 qn += (size_t) w;
107 }
108 return 0;
109}

111int
112brix_s3_list(const brix_weburl *u, const char *ak, const char *sk,
113 const char *region, int verify, const char *ca_dir,
114 char ***keys, size_t *n_out, brix_status *st)
115{
116 char bucket[256], prefix[XRDC_PATH_MAX], encp[XRDC_PATH_MAX * 3];
117 char hostport[300];
118 char **arr = NULL;
119 size_t n = 0, cap = 0;
120 const char *bsl;
121 /* The server's NextContinuationToken is the b64url of the last returned key
122 * (keys up to S3_MAX_KEY=4096 → token up to ~5.5 KB), so this MUST be large
123 * enough or pagination silently corrupts. */
124 char token[8192];
125 int rounds = 0;
126
127 *keys = NULL;
128 *n_out = 0;
129 if (region == NULL || region[0] == '\0') { region = "us-east-1"; }
130 /* u->path = "/bucket[/prefix...]" → split bucket vs prefix. */
131 if (u->path[0] != '/') {
132 brix_status_set(st, XRDC_EUSAGE, 0, "s3 list: bad path");
133 return -1;
134 }
135 bsl = strchr(u->path + 1, '/');
136 if (bsl == NULL) {
137 snprintf(bucket, sizeof(bucket), "%s", u->path + 1);
138 prefix[0] = '\0';
139 } else {
140 size_t bl = (size_t) (bsl - (u->path + 1));
141 if (bl >= sizeof(bucket)) { brix_status_set(st, XRDC_EUSAGE, 0, "s3: bucket too long"); return -1; }
142 memcpy(bucket, u->path + 1, bl);
143 bucket[bl] = '\0';
144 snprintf(prefix, sizeof(prefix), "%s", bsl + 1);
145 }
146 /* The SigV4 signed host MUST match the wire Host header byte-for-byte; that
147 * header is built bracketed for IPv6 literals ([::1]:9000), so sign the same. */
148 brix_format_host_port(u->host, (uint16_t) u->port, hostport, sizeof(hostport));
149 if (brix_http_urlencode((const unsigned char *) prefix, strlen(prefix),
150 encp, sizeof(encp), "") < 0) {
151 brix_status_set(st, XRDC_EUSAGE, 0, "s3: prefix too long");
152 return -1;
153 }
154 token[0] = '\0';
155
156 do {
157 char canon_uri[300], canon_qs[XRDC_PATH_MAX * 4], path[XRDC_PATH_MAX * 5];
158 char hdrs[2048];
159 brix_http_resp r;
160 const char *p;
161 char trunc[8];
162 int more = 0;
163
164 if (++rounds > 100000) {
165 brix_strv_free(arr, n);
166 brix_status_set(st, XRDC_EPROTO, 0, "s3 list: too many pages");
167 return -1;
168 }
169 snprintf(canon_uri, sizeof(canon_uri), "/%s", bucket);
170 if (s3_canon_qs(token, encp, prefix[0] != '\0', canon_qs, sizeof(canon_qs)) != 0
171 || (size_t) snprintf(path, sizeof(path), "%s?%s", canon_uri, canon_qs)
172 >= sizeof(path)) {
173 brix_strv_free(arr, n);
174 brix_status_set(st, XRDC_EUSAGE, 0, "s3 list: prefix/continuation too long");
175 return -1;
176 }
177
178 hdrs[0] = '\0';
179 if (ak != NULL && sk != NULL && ak[0] != '\0' && sk[0] != '\0') {
180 if (brix_s3_sign_v4_q("GET", hostport, canon_uri, canon_qs, ak, sk,
181 region, "UNSIGNED-PAYLOAD", hdrs, sizeof(hdrs)) != 0) {
182 brix_strv_free(arr, n);
183 brix_status_set(st, XRDC_EAUTH, 0, "s3 list: failed to sign request");
184 return -1;
185 }
186 }
187 if (brix_http_req(u->host, u->port, u->tls, "GET", path, hdrs[0] ? hdrs : NULL,
188 NULL, 0, XRDC_WEBLIST_TIMEOUT, verify, ca_dir, &r, st) != 0) {
189 brix_strv_free(arr, n);
190 return -1;
191 }
192 if (r.status != 200) {
193 brix_status_set(st, XRDC_EPROTO, 0, "s3 ListObjectsV2 returned HTTP %d", r.status);
194 brix_http_resp_free(&r);
195 brix_strv_free(arr, n);
196 return -1;
197 }
198 p = r.body ? r.body : "";
199 for (;;) {
200 char key[XRDC_PATH_MAX];
201 int kr = xml_tag(&p, "<Key>", "</Key>", key, sizeof(key));
202 if (kr == 0 || n >= XRDC_WEBLIST_MAX) {
203 break;
204 }
205 if (kr < 0) {
206 /* A key longer than our buffer would yield a wrong download URL —
207 * fail loudly rather than silently dropping or mangling it. */
208 brix_http_resp_free(&r);
209 brix_strv_free(arr, n);
210 brix_status_set(st, XRDC_EPROTO, 0,
211 "s3 list: object key exceeds %zu bytes", sizeof(key));
212 return -1;
213 }
214 if (push(&arr, &n, &cap, key) != 0) {
215 brix_http_resp_free(&r);
216 brix_strv_free(arr, n);
217 brix_status_set(st, XRDC_EPROTO, 0, "s3 list: out of memory");
218 return -1;
219 }
220 }
221 /* pagination: IsTruncated + NextContinuationToken */
222 token[0] = '\0';
223 {
224 const char *tp = r.body ? r.body : "";
225 if (xml_tag(&tp, "<IsTruncated>", "</IsTruncated>", trunc, sizeof(trunc)) == 1
226 && strcmp(trunc, "true") == 0) {
227 const char *np = r.body;
228 int tr = xml_tag(&np, "<NextContinuationToken>",
229 "</NextContinuationToken>", token, sizeof(token));
230 if (tr < 0) {
231 /* token truncated → continuing would corrupt the listing */
232 brix_http_resp_free(&r);
233 brix_strv_free(arr, n);
234 brix_status_set(st, XRDC_EPROTO, 0,
235 "s3 list: continuation token too long");
236 return -1;
237 }
238 more = (tr == 1);
239 }
240 }
241 brix_http_resp_free(&r);
242 if (!more) { break; }
243 } while (1);
244
245 *keys = arr;
246 *n_out = n;
247 return 0;
248}

250int
251brix_webdav_mkcol(const brix_weburl *u, const char *path, const char *bearer,
252 int verify, const char *ca_dir, brix_status *st)
253{
254 char headers[8192];
255 brix_http_resp r;
256 int ok;
257
258 if (bearer != NULL && bearer[0] != '\0') {
259 snprintf(headers, sizeof(headers), "Authorization: Bearer %s\r\n", bearer);
260 } else {
261 headers[0] = '\0';
262 }
263 if (brix_http_req(u->host, u->port, u->tls, "MKCOL", path,
264 headers[0] ? headers : NULL, NULL, 0, XRDC_WEBLIST_TIMEOUT,
265 verify, ca_dir, &r, st) != 0) {
266 return -1;
267 }
268 /* 201 = created; 200 = ok; 405 (Method Not Allowed) / 301 = the collection
269 * already exists → idempotent success. Anything else is a real failure. */
270 ok = (r.status == 201 || r.status == 200 || r.status == 405 || r.status == 301);
271 if (!ok) {
272 brix_status_set(st, XRDC_EPROTO, 0, "MKCOL returned HTTP %d", r.status);
273 }
274 brix_http_resp_free(&r);
275 return ok ? 0 : -1;
276}

278int
279brix_webdav_list(const brix_weburl *u, const char *bearer, int verify,
280 const char *ca_dir, char ***paths, size_t *n_out, brix_status *st)
281{
282 char headers[8192];
283 brix_http_resp r;
284 char **arr = NULL;
285 size_t n = 0, cap = 0;
286 const char *p;
287 int rc = -1;
288
289 *paths = NULL;
290 *n_out = 0;
291 if (u->is_s3) {
292 brix_status_set(st, XRDC_EUSAGE, 0, "recursive copy: s3:// listing not supported yet");
293 return -1;
294 }
295 if (bearer != NULL && bearer[0] != '\0') {
296 snprintf(headers, sizeof(headers),
297 "Depth: infinity\r\nAuthorization: Bearer %s\r\n", bearer);
298 } else {
299 snprintf(headers, sizeof(headers), "Depth: infinity\r\n");
300 }
301 if (brix_http_req(u->host, u->port, u->tls, "PROPFIND", u->path, headers,
302 NULL, 0, XRDC_WEBLIST_TIMEOUT, verify, ca_dir, &r, st) != 0) {
303 return -1;
304 }
305 if (r.status != 207 && r.status != 200) {
306 brix_status_set(st, XRDC_EPROTO, 0, "PROPFIND returned HTTP %d", r.status);
307 brix_http_resp_free(&r);
308 return -1;
309 }
310
311 p = r.body ? r.body : "";
312 while ((p = strstr(p, "<D:response")) != NULL && n < XRDC_WEBLIST_MAX) {
313 const char *end = strstr(p, "</D:response>");
314 const char *h, *he;
315 int is_dir;
316 if (end == NULL) {
317 break;
318 }
319 h = strstr(p, "<D:href>");
320 if (h != NULL && h < end) {
321 h += 8;
322 he = strstr(h, "</D:href>");
323 if (he != NULL && he < end) {
324 const char *col = strstr(p, "<D:collection");
325 is_dir = (col != NULL && col < end);
326 if (!is_dir) {
327 char href[XRDC_PATH_MAX];
328 const char *path = href;
329 /* flags=0: keep a literal '+' (it is a real path byte in an
330 * href, not a form-encoded space). */
331 if (brix_http_urldecode((const unsigned char *) h,
332 (size_t) (he - h), href,
333 sizeof(href), 0)
334 != BRIX_URLDECODE_OK) {
335 href[0] = '\0'; /* overflow/malformed → skip content */
336 }
337 /* reduce an absolute-URL href to its path component */
338 if (strstr(href, "://") != NULL) {
339 char *sl = strstr(href, "://");
340 char *ps = strchr(sl + 3, '/');
341 path = ps ? ps : "/";
342 }
343 if (push(&arr, &n, &cap, path) != 0) {
344 brix_strv_free(arr, n);
345 brix_http_resp_free(&r);
346 brix_status_set(st, XRDC_EPROTO, 0, "webdav list: out of memory");
347 return -1;
348 }
349 }
350 }
351 }
352 p = end + 12;
353 }
354 brix_http_resp_free(&r);
355 *paths = arr;
356 *n_out = n;
357 rc = 0;
358 return rc;
359}
