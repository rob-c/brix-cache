19/*
20 * WHAT: Send the 201 Created <PostResponse> XML document (Location/Bucket/Key/
21 * ETag) used when the form requested success_action_status=201.
22 * HOW: Build the Location URL (absolute if a Host header is present, else a
23 * path), then assemble the XML via the XML_APPEND* helper macros.
24 */
25ngx_int_t
26s3_post_send_created(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
27 const s3_post_form_t *form, const char *etag)
28{
29 ngx_buf_t *b;
30 ngx_chain_t out;
31 u_char *xml;
32 size_t xml_capacity = 8192;
33 size_t xml_len = 0;
34 ngx_int_t rc;
35 ngx_str_t host;
36 char location[S3_MAX_KEY + 512];
37
38 BRIX_PNALLOC_OR_RETURN(xml, r->pool, xml_capacity, NGX_HTTP_INTERNAL_SERVER_ERROR);
39
40 /* Absolute URL when a Host header is available; otherwise a bare path. */
41 host = r->headers_in.host ? r->headers_in.host->value
42 : (ngx_str_t) ngx_null_string;
43 if (host.len > 0) {
44 snprintf(location, sizeof(location), "http://%.*s/%.*s/%s",
45 (int) host.len, host.data,
46 (int) cf->bucket.len, cf->bucket.data, form->key);
47 } else {
48 snprintf(location, sizeof(location), "/%.*s/%s",
49 (int) cf->bucket.len, cf->bucket.data, form->key);
50 }
51
52 /* Assemble the <PostResponse> body (mechanical field-by-field build). */
53 XML_APPEND("<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
54 "<PostResponse>");
55 XML_APPEND_ELEM("Location", location, strlen(location));
56 XML_APPEND_ELEM("Bucket", cf->bucket.data, cf->bucket.len);
57 XML_APPEND_ELEM("Key", form->key, strlen(form->key));
58 XML_APPEND_ELEM("ETag", etag, strlen(etag));
59 XML_APPEND("</PostResponse>");
60
61 b = ngx_create_temp_buf(r->pool, xml_len);
62 if (b == NULL) {
63 return NGX_HTTP_INTERNAL_SERVER_ERROR;
64 }
65 ngx_memcpy(b->pos, xml, xml_len);
66 b->last = b->pos + xml_len;
67 b->last_buf = 1;
68
69 out.buf = b;
70 out.next = NULL;
71
72 r->headers_out.status = NGX_HTTP_CREATED;
73 r->headers_out.content_length_n = (off_t) xml_len;
74 r->headers_out.content_type.len = sizeof("application/xml") - 1;
75 r->headers_out.content_type.data = (u_char *) "application/xml";
76 r->headers_out.content_type_len = r->headers_out.content_type.len;
77
78 rc = ngx_http_send_header(r);
79 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
80 return rc;
81 }
82
83 return ngx_http_output_filter(r, &out);
84}

87/*
88 * WHAT: Produce the client-requested success response after a stored object.
89 * WHY: The S3 POST form controls the success shape via two optional fields,
90 * checked in precedence order:
91 * - success_action_redirect: 303 redirect to the client-supplied URL
92 * (validated for control chars first, since it becomes a Location).
93 * - success_action_status: 200 / 201 / 204 (default 204 if unset).
94 * Anything else is a client error.
95 */
96ngx_int_t
97s3_post_send_success(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
98 const s3_post_form_t *form, const char *etag)
99{
100 /* Redirect takes precedence over success_action_status when both are set. */
101 if (form->success_redirect[0] != '\0') {
102 /* Reject control chars: this value is reflected into a Location header. */
103 if (brix_http_str_has_ctl((u_char *) form->success_redirect,
104 strlen(form->success_redirect)))
105 {
106 return s3_post_error(r, NGX_HTTP_BAD_REQUEST, "InvalidArgument",
107 "success_action_redirect is invalid.");
108 }
109 if (s3_set_header(r, "Location", form->success_redirect) != NGX_OK) {
110 return NGX_HTTP_INTERNAL_SERVER_ERROR;
111 }
112 return s3_post_send_empty(r, NGX_HTTP_SEE_OTHER);
113 }
114
115 /* Default success status is 204 No Content when unspecified. */
116 if (form->success_status[0] == '\0'
117 || strcmp(form->success_status, "204") == 0)
118 {
119 return s3_post_send_empty(r, NGX_HTTP_NO_CONTENT);
120 }
121
122 if (strcmp(form->success_status, "200") == 0) {
123 return s3_post_send_empty(r, NGX_HTTP_OK);
124 }
125
126 if (strcmp(form->success_status, "201") == 0) {
127 return s3_post_send_created(r, cf, form, etag);
128 }
129
130 return s3_post_error(r, NGX_HTTP_BAD_REQUEST, "InvalidArgument",
131 "success_action_status must be 200, 201, or 204.");
132}
