26/* 1 iff this process's /tmp is a systemd PrivateTmp mount — the /tmp
27 * mountpoint's root inside its filesystem is a /tmp/systemd-private-<id>/tmp
28 * subtree. Parsed from /proc/self/mountinfo (fields: id parent major:minor
29 * ROOT MOUNTPOINT ...). Config-domain read, best-effort (0 on any failure). */
30static int
31brix_tmp_is_systemd_private(void)
32{
33 FILE *f;
34 char line[1024];
35 int private_tmp = 0;
36
37 f = fopen("/proc/self/mountinfo", "re");
38 if (f == NULL) {
39 return 0;
40 }
41 while (fgets(line, sizeof(line), f) != NULL) {
42 char root[512], mnt[512];
43
44 if (sscanf(line, "%*d %*d %*31s %511s %511s", root, mnt) == 2
45 && strcmp(mnt, "/tmp") == 0
46 && strstr(root, "/systemd-private-") != NULL)
47 {
48 private_tmp = 1;
49 break;
50 }
51 }
52 (void) fclose(f);
53 return private_tmp;
54}

56/* Shout — loudly, per store — when a stage/cache store lives under a systemd
57 * PrivateTmp /tmp: that mount is PRIVATE to the service and DELETED on every
58 * service stop/restart, so staged-but-unflushed writes and cached objects do
59 * not survive. The shipped unit sets PrivateTmp=true, making this the default
60 * deployment posture — the operator must be told, not left to find out from a
61 * post-restart empty spool. */
62void
63brix_tier_warn_private_tmp(ngx_conf_t *cf,
64 const ngx_http_brix_shared_conf_t *common)
65{
66 const ngx_str_t *stores[2];
67 int i, shouted = 0;
68
69 if (!brix_tmp_is_systemd_private()) {
70 return;
71 }
72 stores[0] = &common->stage_store;
73 stores[1] = &common->cache_store;
74
75 for (i = 0; i < 2; i++) {
76 if (stores[i]->len == 0
77 || ngx_strnstr(stores[i]->data, ":/tmp/", stores[i]->len) == NULL)
78 {
79 continue;
80 }
81 if (!shouted) {
82 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
83 "brix: ==================== PrivateTmp DETECTED "
84 "====================");
85 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
86 "brix: this service runs with systemd PrivateTmp: /tmp is "
87 "PRIVATE to the service and is DELETED on every service "
88 "stop/restart.");
89 shouted = 1;
90 }
91 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
92 "brix: the store \"%V\" therefore does NOT survive a restart — "
93 "staged-but-unflushed writes / cached objects are LOST. Point "
94 "it at a durable path or set PrivateTmp=no in the unit.",
95 stores[i]);
96 }
97 if (shouted) {
98 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
99 "brix: ==========================================================="
100 "====");
101 }
102}

104/* Build "<base>/<sanitised backend URL>.<worker-uid>" for the brix-managed
105 * default stage store: every backend-URL byte outside [A-Za-z0-9._-] becomes
106 * '_', capped at 160 bytes, so distinct backends get distinct filesystem-safe
107 * spool directories under the one managed base. The runtime worker uid is
108 * part of the leaf name: the base is shared between every instance on the
109 * host, and two instances of different identities (a root-launched gateway
110 * spooling as nobody next to an unprivileged dev instance) hitting the SAME
111 * backend URL must not fight over one 0700 leaf only one of them can use. */
112static void
113brix_tier_default_stage_dir(const ngx_str_t *backend, uid_t worker_uid,
114 char *dir, size_t cap)
115{
116 size_t pos = sizeof(BRIX_TIER_DEFAULT_STAGE_BASE "/") - 1;
117 size_t i;
118
119 ngx_cpystrn((u_char *) dir, (u_char *) BRIX_TIER_DEFAULT_STAGE_BASE "/",
120 cap);
121 for (i = 0; i < backend->len && i < 160 && pos + 1 < cap; i++) {
122 u_char c = backend->data[i];
123 int keep = (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')
124 || (c >= '0' && c <= '9') || c == '.' || c == '_'
125 || c == '-';
126
127 dir[pos++] = keep ? (char) c : '_';
128 }
129 dir[pos] = '\0';
130 if (pos < cap) {
131 (void) snprintf(dir + pos, cap - pos, ".%lu",
132 (unsigned long) worker_uid);
133 }
134}

136/*
137 * brix_tier_default_stage_store() — zero-config write staging for a
138 * whole-object remote gateway (WebDAV/S3 backend).
139 *
140 * WHAT: When a writable export forwards to a whole-object remote backend and
141 * the operator configured NO stage tier at all, provision the stock
142 * stage decorator over a brix-managed posix store at
143 * /tmp/staging/<sanitised-backend-url>.<worker-uid>: create the
144 * directory (0711
145 * base, 0700 leaf chown'd to the worker user, exactly as
146 * brix_shared_credential_dir_ensure hands over /dev/shm/brix-creds),
147 * point common->stage_store at "posix:<dir>", flip stage_enable on,
148 * and shout a multi-line [warn] banner naming the directory, the tmpfs
149 * option and the opt-outs.
150 *
151 * WHY: Without a stage tier the remote driver rejects random-offset writes
152 * at the cap (kXR_Unsupported / EINVAL) — correct but hostile as the
153 * default posture of a POSIX-speaking gateway. Routing the default
154 * through the SAME tier grammar the operator would write by hand
155 * (brix_stage on + brix_stage_store posix:…) keeps one code path:
156 * tier_parse_local validates the directory like an export root and the
157 * stock sd_stage decorator does all buffering and flushing — no bespoke
158 * staging machinery.
159 *
160 * HOW: Build the sanitised leaf path; mkdir base 0711 (root-owned, traverse
161 * only) and leaf 0700; chown the leaf to ccf->user when the master runs
162 * as root (the workers write the spool). On ANY failure warn and return
163 * with staging left off — writes then require strictly sequential
164 * offsets, exactly as before — never fatal. On success rewrite
165 * stage_store (cf->pool) + stage_enable and emit the banner.
166 */
167void
168brix_tier_default_stage_store(ngx_conf_t *cf,
169 ngx_http_brix_shared_conf_t *common)
170{
171 char dir[PATH_MAX];
172 u_char *url;
173 size_t url_len;
174 struct stat st;
175 uid_t want_uid;
176 gid_t want_gid;
177
178 brix_shared_worker_dir_ids(cf, &want_uid, &want_gid);
179 brix_tier_default_stage_dir(&common->storage_backend, want_uid,
180 dir, sizeof(dir));
181
182 if (brix_mkdir_recursive(BRIX_TIER_DEFAULT_STAGE_BASE, 0711) != 0
183 || (mkdir(dir, 0700) != 0 && errno != EEXIST))
184 {
185 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
186 "brix: cannot create the default stage store \"%s\" — "
187 "write staging stays OFF; random writes to the remote backend "
188 "will be rejected (set brix_stage_store, or \"brix_stage off\" "
189 "to silence this warning)", dir);
190 return;
191 }
192 if (stat(dir, &st) != 0 || !S_ISDIR(st.st_mode)) {
193 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
194 "brix: default stage store \"%s\" is not a usable directory — "
195 "write staging stays OFF", dir);
196 return;
197 }
198
199 /* The master parses config as root but the workers write the spool as
200 * the RUNTIME worker identity — the `user` account, or the always-on
201 * de-escalation target (brix_worker_user/nobody) for a root-capable
202 * worker — hand the directory to them, exactly as ngx_create_paths does
203 * for the temp paths. */
204 if (geteuid() == 0 && st.st_uid != want_uid
205 && chown(dir, want_uid, want_gid) != 0)
206 {
207 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
208 "brix: cannot chown the default stage store \"%s\" to the "
209 "worker user — write staging stays OFF", dir);
210 return;
211 }
212
213 /* An unprivileged master cannot repossess a foreign leaf (a squatter, or
214 * debris from another identity despite the uid-suffixed name) — refuse it
215 * and keep the documented never-fatal contract: warn, staging stays OFF,
216 * rather than adopting a store the later tier validation would [emerg] on
217 * (or worse, one a hostile local user controls). */
218 if (geteuid() != 0
219 && (st.st_uid != geteuid() || access(dir, W_OK | X_OK) != 0))
220 {
221 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
222 "brix: the default stage store \"%s\" exists but is not owned/"
223 "writable by this user (owner uid %lu, we are uid %lu) — "
224 "write staging stays OFF; remove the directory or set "
225 "brix_stage_store", dir, (unsigned long) st.st_uid,
226 (unsigned long) geteuid());
227 return;
228 }
229
230 url_len = sizeof("posix:") - 1 + ngx_strlen(dir);
231 url = ngx_pnalloc(cf->pool, url_len + 1);
232 if (url == NULL) {
233 return;
234 }
235 ngx_sprintf(url, "posix:%s%Z", dir);
236 common->stage_store.data = url;
237 common->stage_store.len = url_len;
238 common->stage_enable = 1;
239
240 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
241 "brix: ==================== DEFAULT WRITE-STAGING ===================");
242 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
243 "brix: no stage tier is configured for the whole-object remote "
244 "backend \"%V\";", &common->storage_backend);
245 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
246 "brix: random writes will be buffered in \"%s\",", dir);
247 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
248 "brix: a directory MANAGED BY BRIX (created, filled and cleaned by "
249 "brix — do not place files there; mount it tmpfs if you want "
250 "RAM-tier performance).");
251 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
252 "brix: override with \"brix_stage_store <store-url>\", or opt out "
253 "with \"brix_stage off\" (writes must then be strictly sequential).");
254 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
255 "brix: ===============================================================");
256}
