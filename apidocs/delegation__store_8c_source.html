1/*
2 * delegation_store.c - Phase-3 T4 pending-delegation store (per-worker,
3 * in-memory, TTL-bounded). Split out of delegation.c; the two-step GridSite
4 * handlers (delegation_gridsite_req.c / delegation_gridsite_put.c) reach it
5 * through the three entry points declared in delegation_internal.h.
6 *
7 * ========================================================================
8 * WHAT: A fixed-capacity table of {id, fresh EVP_PKEY*, client_dn,
9 * expires_at} entries, one per in-flight getProxyReq/putProxy
10 * handshake. brix_deleg_store_put() inserts (evicting expired
11 * entries first, then refusing if still full); brix_deleg_store_take()
12 * looks up + REMOVES an entry by id (one-shot: the caller always owns
13 * the returned EVP_PKEY* and must free it), sweeping expired entries
14 * as a side effect of every call so no background thread is needed.
15 *
16 * WHY: Per-worker, NOT SHM. The fresh private key generated for a
17 * getProxyReq must NEVER leave the process that generated it (the
18 * brief's hardest requirement) — SHM would mean either serialising an
19 * EVP_PKEY into shared bytes (a private key exposed to every other
20 * worker process) or storing only a reference nginx cannot safely
21 * cross-process anyway. A module-static per-worker table keeps the
22 * key in this worker's heap for its ~600s lifetime and nowhere else.
23 * The operational cost: a getProxyReq and its matching putProxy MUST
24 * land on the same worker. For a single-worker deployment (typical
25 * test/small-site config) this is automatic. For multi-worker, the
26 * deployment needs sticky routing (e.g. an L7 proxy keyed on client
27 * IP/TLS session, or a client that simply retries putProxy against
28 * the same connection/worker it got the CSR from) — documented here
29 * and in the reference doc, not solved by this store.
30 *
31 * HOW: Fixed array (BRIX_DELEG_STORE_CAP slots), linear scan (the table is
32 * small and operations are rare — an O(n) scan is simpler and just as
33 * fast in practice as a hash table at this size). brix_deleg_store()
34 * lazily allocates the singleton on first use (module-static pointer,
35 * the ONE new global this feature introduces, exactly as scoped in
36 * the task brief) from ngx_cycle->pool so it lives for the worker's
37 * lifetime; nginx never destroys that pool mid-run.
38 * ======================================================================== */

50/* Bounded capacity: a pending handshake is a brief interactive round-trip
51 * (client signs a CSR, usually within seconds); 256 concurrent in-flight
52 * handshakes per worker is generous headroom without unbounded growth. */
53#define BRIX_DELEG_STORE_CAP 256
54
55/* Pending-delegation lifetime: long enough for a human/script to sign and
56 * return a CSR, short enough that an abandoned handshake's key doesn't
57 * linger. Matches the brief's suggested 10 minutes. */
58#define BRIX_DELEG_TTL_SEC 600
59
60typedef struct {
61 char id[BRIX_DELEG_ID_HEXLEN + 1]; /* hex, NUL-terminated; empty = free slot */
62 EVP_PKEY *fresh_key; /* owned; never written to disk */
63 char client_dn[1024]; /* bound at getreq, rechecked at putProxy */
64 time_t created_at;
65 time_t expires_at;
66} brix_deleg_entry_t;
67
68struct brix_deleg_store_s {
69 brix_deleg_entry_t slots[BRIX_DELEG_STORE_CAP];
70};

72/* The one documented per-worker global this feature introduces (see the
73 * module doc-block above for why this cannot be SHM). Lazily allocated on
74 * first use from ngx_cycle->pool; never freed early (worker-lifetime). */
75static brix_deleg_store_t *brix_deleg_store_singleton = NULL;

77brix_deleg_store_t *
78brix_deleg_store(void)
79{
80 if (brix_deleg_store_singleton == NULL) {
81 brix_deleg_store_singleton =
82 ngx_pcalloc(ngx_cycle->pool, sizeof(brix_deleg_store_t));

87/*
88 * brix_deleg_entry_free - release an entry's owned resources and mark the
89 * slot free. The ONLY place an EVP_PKEY is freed, so every store exit path
90 * (evict, sweep, take, drop) routes through this — the single choke point
91 * that makes "no EVP_PKEY leak on any path" auditable by inspection.
92 */
93static void
94brix_deleg_entry_free(brix_deleg_entry_t *e)
95{
96 if (e->fresh_key != NULL) {
97 EVP_PKEY_free(e->fresh_key);
98 e->fresh_key = NULL;
99 }
100 e->id[0] = '\0';
101 e->client_dn[0] = '\0';
102 e->created_at = 0;
103 e->expires_at = 0;
104}
105
106/*
107 * brix_deleg_store_sweep - free every expired entry. Called at the top of
108 * both brix_deleg_store_put (to make room) and brix_deleg_store_take (so a
109 * lookup never returns a stale entry) — this IS the store's TTL enforcement;
110 * there is no background timer.
111 */
112static void
113brix_deleg_store_sweep(brix_deleg_store_t *st, time_t now)
114{
115 int i;
116
117 for (i = 0; i < BRIX_DELEG_STORE_CAP; i++) {
118 if (st->slots[i].id[0] != '\0' && st->slots[i].expires_at <= now) {
119 brix_deleg_entry_free(&st->slots[i]);
120 }
121 }
122}

124/*
125 * brix_deleg_store_put - insert a new pending entry with a fresh CSPRNG id.
126 *
127 * WHAT: Sweeps expired entries, then scans for a free slot. On success,
128 * fills id_out (caller-supplied buffer, >= BRIX_DELEG_ID_HEXLEN+1)
129 * with the hex id, stores fresh_key (ownership transfers to the
130 * store — caller must NOT free it after a successful put) and
131 * client_dn, sets expires_at = now + BRIX_DELEG_TTL_SEC.
132 *
133 * WHY: CSPRNG (RAND_bytes, 16 bytes / 32 hex chars) makes the id
134 * unguessable — an attacker cannot brute-force or predict another
135 * client's pending delegation-id to attempt a cross-client putProxy
136 * (the DN check at take-time is the second, independent layer).
137 *
138 * HOW: 1. Sweep. 2. RAND_bytes + brix_hex_encode for the id (retry once on
139 * an astronomically unlikely id collision — a fresh RAND_bytes draw,
140 * not a fallback to a weaker source). 3. Linear scan for id[0]=='\0'.
141 * 4. All full after sweep -> NGX_DECLINED (caller maps to 503,
142 * documented eviction policy: evict-expired-first, else reject new
143 * getreq rather than evict a live pending handshake).
144 */
145ngx_int_t
146brix_deleg_store_put(brix_deleg_store_t *st, EVP_PKEY *fresh_key,
147 const char *client_dn, char *id_out, size_t id_out_cap)
148{
149 unsigned char raw[BRIX_DELEG_ID_BYTES];
150 char hex[BRIX_DELEG_ID_HEXLEN + 1];
151 time_t now = time(NULL);
152 int i;
153
154 if (id_out_cap < sizeof(hex)) {
155 return NGX_ERROR;
156 }
157 brix_deleg_store_sweep(st, now);
158
159 if (RAND_bytes(raw, sizeof(raw)) != 1) {
160 return NGX_ERROR;
161 }
162 brix_hex_encode(raw, sizeof(raw), hex);
163
164 for (i = 0; i < BRIX_DELEG_STORE_CAP; i++) {
165 if (st->slots[i].id[0] == '\0') {
166 ngx_memcpy(st->slots[i].id, hex, sizeof(hex));
167 st->slots[i].fresh_key = fresh_key;
168 ngx_memcpy(st->slots[i].client_dn, client_dn,
169 ngx_min(strlen(client_dn) + 1,
170 sizeof(st->slots[i].client_dn) - 1));
171 st->slots[i].client_dn[sizeof(st->slots[i].client_dn) - 1] = '\0';
172 st->slots[i].created_at = now;
173 st->slots[i].expires_at = now + BRIX_DELEG_TTL_SEC;
174 ngx_memcpy(id_out, hex, sizeof(hex));
175 return NGX_OK;
176 }
177 }
178 return NGX_DECLINED; /* bounded store full even after sweep */
179}

181/*
182 * brix_deleg_store_take - one-shot lookup-and-remove by id, DN-checked.
183 *
184 * WHAT: Scans for the id FIRST (before sweeping) so an expired-but-still-
185 * present entry can be reported as EXPIRED rather than being wiped by
186 * a blind sweep and misreported as NOT_FOUND — "never existed" /
187 * "already consumed" is a genuinely different, distinguishable
188 * outcome from "existed but timed out" for the caller's error
189 * message. On a live match: if want_dn does not match the entry's
190 * stored client_dn, frees the entry (one-shot: a cross-client attempt
191 * burns the id rather than leaving it available for a retry) and
192 * returns DN_MISMATCH WITHOUT transferring the key. On a DN match,
193 * transfers fresh_key ownership to *key_out (caller must
194 * EVP_PKEY_free it) and frees the slot's other fields, returning OK.
195 * Once the target id is resolved (found-and-handled, or absent),
196 * sweeps the REST of the table for any other expired entries — this
197 * is still the TTL enforcement's only trigger point (alongside
198 * brix_deleg_store_put), just ordered after the id-specific check so
199 * the two responsibilities (report THIS id's true state; reclaim
200 * OTHER entries' expired slots) don't fight over the same pass.
201 *
202 * WHY: This is the id-to-client binding re-check the brief requires:
203 * "must belong to THIS authenticated client DN — reject
204 * cross-client". One-shot on EVERY terminal outcome (not just
205 * success) is the documented policy — see the module doc-block —
206 * so a determined attacker gets exactly one guess per id and success
207 * or failure both close it out, with no leaked EVP_PKEY on any path
208 * (brix_deleg_entry_free is the sole free site).
209 *
210 * HOW: Linear scan for id match. Not found -> sweep the rest -> NOT_FOUND.
211 * Found but past expiry -> free + sweep the rest -> EXPIRED.
212 * DN mismatch -> free + sweep the rest -> DN_MISMATCH.
213 * Match -> transfer key, free the rest of the slot, sweep the
214 * remaining table, -> OK.
215 */
216brix_deleg_take_t
217brix_deleg_store_take(brix_deleg_store_t *st, const char *id,
218 const char *want_dn, EVP_PKEY **key_out)
219{
220 time_t now = time(NULL);
221 int i;
222 brix_deleg_take_t result = BRIX_DELEG_TAKE_NOT_FOUND;
223
224 for (i = 0; i < BRIX_DELEG_STORE_CAP; i++) {
225 brix_deleg_entry_t *e = &st->slots[i];
226
227 if (e->id[0] == '\0' || strcmp(e->id, id) != 0) {
228 continue;
229 }
230 if (e->expires_at <= now) {
231 brix_deleg_entry_free(e);
232 result = BRIX_DELEG_TAKE_EXPIRED;
233 } else if (strcmp(e->client_dn, want_dn) != 0) {
234 brix_deleg_entry_free(e);
235 result = BRIX_DELEG_TAKE_DN_MISMATCH;
236 } else {
237 *key_out = e->fresh_key;
238 e->fresh_key = NULL; /* ownership -> caller; entry_free must not free it */
239 brix_deleg_entry_free(e);
240 result = BRIX_DELEG_TAKE_OK;
241 }
242 break;
243 }
244 brix_deleg_store_sweep(st, now);
245 return result;
246}
