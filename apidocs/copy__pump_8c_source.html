8int
9write_all(int fd, const uint8_t *buf, size_t n, brix_status *st)
10{
11 size_t off = 0;
12 while (off < n) {
13 ssize_t w = write(fd, buf + off, n - off);
14 if (w < 0) {
15 if (errno == EINTR) {
16 continue;
17 }
18 brix_status_set(st, XRDC_ESOCK, errno, "local write: %s", strerror(errno));
19 return -1;
20 }
21 off += (size_t) w;
22 }
23 return 0;
24}

27/* Remote source/sink over an open handle; ->pgrw selects paged I/O + per-page
28 * CRC32c (kXR_pgread/pgwrite) vs the plain kXR_read/write path.
29 *
30 * Resilience (download source, when ->resilient): on a transport fault the read
31 * is retried within a max_stall deadline — reconnecting the session and REOPENING
32 * the file at the same absolute offset (idempotent), with fast backoff. The read
33 * size adapts: it starts large (full throughput on a clean link) and HALVES on
34 * each sever down to XRDC_RESILIENT_FLOOR, so a lossy link converges on a request
35 * small enough to get through. This is what lets a one-shot xrdcp ride out loss
36 * the way the FUSE driver's mfile layer does. */
37
38/* Reconnect the session (to the original endpoint, re-selecting a data server)
39 * and reopen the source file — replacing the dead handle. 0 / -1 (st set). */
40int
41pump_remote_reopen(pump_remote_t *r, brix_status *st)
42{
43 const char *host = (r->c->home_host[0] != '\0') ? r->c->home_host : r->c->host;
44 int port = (r->c->home_port != 0) ? r->c->home_port : r->c->port;
45 if (brix_reconnect(r->c, host, port, st) != 0) {
46 return -1;
47 }
48 brix_file nf;
49 int rc = (r->opaque != NULL && r->opaque[0] != '\0')
50 ? brix_file_open_opaque(r->c, r->path, r->opaque, 0, 0, 0, &nf, st)
51 : brix_file_open_read(r->c, r->path, &nf, st);
52 if (rc != 0) {
53 return -1;
54 }
55 *r->f = nf;
56 return 0;
57}

60ssize_t
61pump_src_remote(void *ctx, uint8_t *buf, int64_t off, size_t cap, brix_status *st)
62{
63 pump_remote_t *r = (pump_remote_t *) ctx;
64
65 if (!r->resilient) {
66 return r->pgrw ? brix_file_pgread(r->c, r->f, off, buf, cap, st)
67 : brix_file_read(r->c, r->f, off, buf, cap, st);
68 }
69
70 uint64_t deadline = brix_mono_ns() + (uint64_t) r->max_stall_ms * 1000000ULL;
71 unsigned attempt = 0;
72 for (;;) {
73 size_t want = (cap < r->cur_chunk) ? cap : r->cur_chunk;
74 ssize_t n = r->pgrw ? brix_file_pgread(r->c, r->f, off, buf, want, st)
75 : brix_file_read(r->c, r->f, off, buf, want, st);
76 if (n >= 0) {
77 return n;
78 }
79 if (!brix_status_retryable(st) || brix_copy_quit_requested()
80 || brix_mono_ns() >= deadline) {
81 return -1;
82 }
83 /* Transport fault: shrink the request so it is likelier to get through,
84 * then reconnect+reopen and retry at the same offset. */
85 if (r->cur_chunk > XRDC_RESILIENT_FLOOR) {
86 r->cur_chunk /= 2;
87 }
88 brix_backoff_sleep_fast(attempt++);
89 (void) pump_remote_reopen(r, st); /* best-effort; loop re-tries */
90 }
91}

113int
114pump_sink_remote(void *ctx, const uint8_t *buf, int64_t off, size_t n,
115 brix_status *st)
116{
117 pump_remote_t *r = (pump_remote_t *) ctx;
118
119 if (!r->resilient) {
120 return r->pgrw ? brix_file_pgwrite(r->c, r->f, off, buf, n, st)
121 : brix_file_write(r->c, r->f, off, buf, n, st);
122 }
123
124 /* Resilient upload (server has brix_upload_resume on): on a transport
125 * fault, reconnect + reopen-in-place and re-issue THIS buffer at the same
126 * absolute offset. The bytes below `off` were already acked, so the server
127 * has them on the preserved partial — re-writing the current span is either
128 * filling the gap or an idempotent overwrite, so there is never a hole. */
129 uint64_t deadline = brix_mono_ns() + (uint64_t) r->max_stall_ms * 1000000ULL;
130 unsigned attempt = 0;
131 for (;;) {
132 int rc = r->pgrw ? brix_file_pgwrite(r->c, r->f, off, buf, n, st)
133 : brix_file_write(r->c, r->f, off, buf, n, st);
134 if (rc == 0) {
135 return 0;
136 }
137 if (!brix_status_retryable(st) || brix_copy_quit_requested()
138 || brix_mono_ns() >= deadline) {
139 return -1;
140 }
141 brix_backoff_sleep_fast(attempt++);
142 (void) pump_sink_reopen(r, st); /* best-effort; loop re-tries */
143 }
144}

150pump_src_local(void *ctx, uint8_t *buf, int64_t off, size_t cap, brix_status *st)
151{

153 (void) off;

155 ssize_t r = read(fd, buf, cap);

203/*
204 * The loop. expected >= 0 = known length (stop at expected; a 0-read before it
205 * is a short-read error); expected < 0 = EOF-driven (a 0-read is the clean end).
206 * When o->progress is set it fires after each drained chunk with (off,
207 * progress_total), plus one final (off, off) at EOF to mirror the historical
208 * "100%" upload tick; pass o == NULL to suppress progress entirely. Cancel
209 * (g_brix_copy_quit) aborts with EINTR. Owns its CHUNK buffer. Returns 0 / -1.
210 */
211int
212transfer_pump(pump_src_fn src, void *sctx, pump_sink_fn sink, void *kctx,
213 int64_t expected, const brix_copy_opts *o, int64_t progress_total,
214 brix_status *st)
215{
216 uint8_t *buf;
217 int64_t off = 0;
218 int rc = -1;
219
220 buf = (uint8_t *) malloc(XRDC_COPY_CHUNK);
221 if (buf == NULL) {
222 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
223 return -1;
224 }
225
226 for (;;) {
227 size_t cap = XRDC_COPY_CHUNK;
228 ssize_t n;
229
230 /* Completion is tested BEFORE the cancel flag: a byte-complete known-size
231 * transfer must report success even if the sticky g_brix_copy_quit was
232 * set on the final chunk. This mirrors the original head-controlled
233 * `while (off < si->size)` loops (download/r2r), where completion was the
234 * loop condition and the cancel check lived inside the body — so a SIGINT
235 * landing at the finish line never discarded a perfect file. For an
236 * EOF-driven pump (expected < 0) this is skipped and cancel is checked
237 * first, exactly as the original `for (;;)` upload/recursive loops did. */
238 if (expected >= 0 && off >= expected) {
239 rc = 0;
240 break;
241 }
242 if (g_brix_copy_quit) {
243 brix_status_set(st, XRDC_ESOCK, EINTR, "transfer cancelled (signal)");
244 break; /* rc stays -1 → caller drops the partial/temp */
245 }
246 if (expected >= 0 && (int64_t) cap > expected - off) {
247 cap = (size_t) (expected - off);
248 }
249
250 n = src(sctx, buf, off, cap, st);
251 if (n < 0) {
252 break;
253 }
254 if (n == 0) {
255 if (expected >= 0) {
256 brix_status_set(st, XRDC_EPROTO, 0,
257 "short read: got %lld of %lld bytes",
258 (long long) off, (long long) expected);
259 break;
260 }
261 rc = 0; /* EOF — full body streamed */
262 if (o != NULL && o->progress != NULL) {
263 o->progress(o->progress_arg, (long long) off, (long long) off);
264 }
265 break;
266 }
267 if (sink(kctx, buf, off, (size_t) n, st) != 0) {
268 break;
269 }
270 off += n;
271 if (o != NULL && o->progress != NULL) {
272 o->progress(o->progress_arg, (long long) off,
273 (long long) progress_total);
274 }
275 }
276
277 free(buf);
278 return rc;
279}
