1/*
2 * stat_flags.h — XRootD stat `flags` field semantics (single source of truth).
3 *
4 * WHAT: the meaning of the bits in the `flags` field of a kXR_stat reply
5 * (kXR_isDir / kXR_other / kXR_readable / kXR_xset / ...), expressed once
6 * for both halves of the boundary:
7 * - brix_stat_flags_from_mode: struct-stat mode -> kXR flags (server encode)
8 * - brix_stat_mode_from_flags: kXR flags -> POSIX file-type+perm bits (client decode)
9 * WHY: this is the companion to stat_line.h — that header owns the LINE shape
10 * ("<id> <size> <flags> <mtime>"); this one owns what the `flags` integer in
11 * that line MEANS. The server set the bits (src/path/stat_body.c) and the
12 * client turned them back into an st_mode (client/lib/posix_map.c) with the
13 * bit meanings written out independently on each side — so the dir/other/
14 * readable/xset semantics were duplicated and free to drift.
15 * HOW: header-only static inlines over <sys/stat.h> + the kXR_* bit constants —
16 * no ngx/alloc/OpenSSL — so the same code compiles into the nginx module and
17 * the ngx-free client. FUSE-specific policy (st_nlink, st_size, st_ino,
18 * st_blksize) stays in the client; only the file-type + permission bits,
19 * which the flag spec actually dictates, live here.
20 *
21 * The two functions are NOT strict inverses: the decoder handles every bit a
22 * peer xrootd server might send, while the encoder emits the file-type +
23 * permission bits the reference StatGen sets. They share the bit *definitions*,
24 * not a round trip.
25 *
26 * Clean-room: bit meanings from src/protocol/flags.h (vs XProtocol stat flags).
27 */

29#define BRIX_PROTOCOL_STAT_FLAGS_H

35/*
36 * Encode the kXR stat `flags` field from a POSIX mode (server side). `extra_flags`
37 * carries any caller-supplied bits (e.g. kXR_offline for a tape-resident file).
38 * Mirrors the historical derivation exactly: directory -> kXR_isDir; non-regular
39 * non-dir -> kXR_other; any read permission bit -> kXR_readable.
40 */
41static inline int
42brix_stat_flags_from_mode(mode_t st_mode, int extra_flags)
43{
44 int flags = extra_flags;
45
46 if (S_ISDIR(st_mode)) {
47 flags |= kXR_isDir;
48 } else if (!S_ISREG(st_mode)) {
49 flags |= kXR_other;
50 }
51
52 if (st_mode & (S_IRUSR | S_IRGRP | S_IROTH)) {
53 flags |= kXR_readable;
54 }
55
56 return flags;
57}
58
59/*
60 * Encode the kXR stat `flags` field from a full struct stat (server side),
61 * mirroring the reference XrdXrootdProtocol::StatGen exactly. Unlike the
62 * mode-only helper above, this sets kXR_writable and kXR_xset in addition to
63 * kXR_readable, applying the owner/group/other permission check against the
64 * server's own (effective) uid/gid — which is whom the export is read/written
65 * as. `extra_flags` carries caller-supplied bits (e.g. kXR_offline/kXR_cachersp).
66 *
67 * read bit set & (owner&uid match | group&gid match | other) -> kXR_readable
68 * write bit set & (... same ...) -> kXR_writable
69 * exec bit set & (... same ...) -> kXR_xset
70 * directory -> kXR_isDir ; non-dir non-regular -> kXR_other
71 */
72static inline int
73brix_stat_flags_from_stat(const struct stat *st, uid_t myuid, gid_t mygid,
74 int extra_flags)
75{
76 int flags = extra_flags;
77 mode_t m = st->st_mode;
78
79 if ((m & (S_IRUSR | S_IRGRP | S_IROTH))
80 && (((m & S_IRUSR) && myuid == st->st_uid)
81 || ((m & S_IRGRP) && mygid == st->st_gid)
82 || (m & S_IROTH))) {
83 flags |= kXR_readable;
84 }
85 if ((m & (S_IWUSR | S_IWGRP | S_IWOTH))
86 && (((m & S_IWUSR) && myuid == st->st_uid)
87 || ((m & S_IWGRP) && mygid == st->st_gid)
88 || (m & S_IWOTH))) {
89 flags |= kXR_writable;
90 }
91 if ((m & (S_IXUSR | S_IXGRP | S_IXOTH))
92 && (((m & S_IXUSR) && myuid == st->st_uid)
93 || ((m & S_IXGRP) && mygid == st->st_gid)
94 || (m & S_IXOTH))) {
95 flags |= kXR_xset;
96 }
97
98 if (S_ISDIR(m)) {
99 flags |= kXR_isDir;
100 } else if (!S_ISREG(m)) {
101 flags |= kXR_other;
102 }
103
104 return flags;
105}

107/*
108 * Decode the kXR stat `flags` field into a POSIX file-type + permission mode
109 * (client side). `allow_symlink` is set by callers that issued an lstat and want a
110 * non-regular/non-dir entry surfaced as a symlink (size = target length, so the
111 * kernel follows up with readlink); a plain stat passes 0 and gets a regular file.
112 * kXR_isDir -> S_IFDIR | 0755
113 * kXR_other (lstat only) -> S_IFLNK | 0777
114 * otherwise -> S_IFREG | (kXR_xset ? 0755 : 0644)
115 */
116static inline mode_t
117brix_stat_mode_from_flags(int flags, int allow_symlink)
118{
119 if (flags & kXR_isDir) {
120 return (mode_t) (S_IFDIR | 0755);
121 }
122 if (allow_symlink && (flags & kXR_other)) {
123 return (mode_t) (S_IFLNK | 0777);
124 }
125 return (mode_t) (S_IFREG | ((flags & kXR_xset) ? 0755 : 0644));
126}
