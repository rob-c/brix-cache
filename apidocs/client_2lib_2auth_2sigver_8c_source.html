1/*
2 * sigver.c — client request signing (kXR_sigver), the mirror of the server's
3 * src/protocols/root/handshake/sigver.c verifier.
4 *
5 * WHAT: When the session has a GSI signing key and the server's advertised
6 * security level requires a signature for an opcode, prepend a kXR_sigver
7 * frame carrying HMAC-SHA256(signing_key, seqno_be(8) || request_hdr(24) ||
8 * payload). The server replies kXR_ok, then verifies the following request
9 * against the recorded HMAC.
10 * WHY: XRootD high-security configs (security_level >= 2) reject unsigned
11 * mutating/open requests; this lets the native client interoperate with them.
12 * HOW: Gated on c->sec_level (parsed from the protocol reply's SecurityInfo) so it
13 * is a no-op on the common level-0 servers. seqno is monotonic per
14 * connection. Opcode policy is copied verbatim from the server.
15 *
16 * NOTE: not exercised by the test harness (its GSI servers run at level 0); the
17 * implementation follows the server verifier byte-for-byte. See the
18 * clean-room log for the validation gap.
19 */

30int
31brix_sigver_maybe(brix_conn *c, const uint8_t *hdr24, const void *payload,
32 uint32_t plen, brix_status *st)
33{
34 uint16_t reqid;
35 uint64_t seq;
36 uint8_t mac[32];
37 ClientSigverRequest sv;
38 uint16_t sid;
39
40 reqid = xrd_get_u16_be(hdr24 + 2); /* unaligned-safe */
41
42 if (!c->signing_active || c->sec_level < 2
43 || !brix_gsi_sigver_required(reqid, c->sec_level)) {
44 return 0; /* signing not required for this op */
45 }
46
47 seq = ++c->sig_seqno;
48
49 /* Shared HMAC (libxrdproto gsi_core) over seqno_be || request header ||
50 * payload — byte-identical to what the server verifies. The client always
51 * covers the payload (nodata = 0). */
52 if (!brix_gsi_sigver_hmac(c->signing_key, seq, hdr24, payload, plen, 0,
53 mac)) {
54 brix_status_set(st, XRDC_EAUTH, 0, "sigver: HMAC failed");
55 return -1;
56 }
57
58 /* Build + send the kXR_sigver frame (its own streamid). */
59 memset(&sv, 0, sizeof(sv));
60 sid = c->next_sid++;
61 sv.streamid[0] = (uint8_t) (sid >> 8);
62 sv.streamid[1] = (uint8_t) (sid & 0xff);
63 sv.requestid = htons(kXR_sigver);
64 {
65 xrdw_sigver_req_t b = { .expectrid = reqid, .version = 0,
66 .flags = 0, /* payload is included in the HMAC */
67 .seqno = seq, .crypto = kXR_SHA256_sig };
68 xrdw_sigver_req_pack(&b, ((ClientRequestHdr *) &sv)->body);
69 }
70 sv.dlen = (kXR_int32) htonl(32);
71
72 if (brix_write_full(&c->io, &sv, sizeof(sv), st) != 0
73 || brix_write_full(&c->io, mac, 32, st) != 0) {
74 return -1;
75 }
76 /*
77 * kXR_sigver is a request PREFIX: a spec-conformant server (stock XRootD,
78 * and now this module) sends NO response on success — only kXR_SigVerErr on
79 * failure, which surfaces as the covered request's reply. So do NOT read an
80 * ack here; the next recv is the covered request's response. Reading a
81 * non-existent ack would consume that reply (or block).
82 */
83 return 0;
84}
