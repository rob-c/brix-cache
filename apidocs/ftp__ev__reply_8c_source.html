3/*
4 * ftp_ev_reply.c — reply framing for the event engine.
5 *
6 * WHAT: append-only reply helpers that write into the connection's outbound
7 * buffer (->ob): a printf-style status reply, an AUTH/ADAT token continuation,
8 * plus the base64 codecs the RFC 2228 handshake needs.
9 *
10 * WHY: the engine is non-blocking, so replies are never written inline — they are
11 * queued into ->ob and drained by brix_ftp_ev_flush() under the event loop. Once
12 * the GSI security layer is active every status reply is transparently GSS-wrapped
13 * into a 631/632/633 safe/private frame; the pre-auth handshake (220/334/335/235)
14 * is emitted while ->sec_active is still 0 and so goes out cleartext.
15 *
16 * HOW: ev_out_put() is the single choke point that copies bytes into ->ob with a
17 * hard capacity check. Because commands are processed lock-step (the engine
18 * drains ->ob before framing the next command), at most one command's replies are
19 * resident at a time, so the fixed 64 KiB buffer cannot be overrun by a
20 * well-behaved reply path.
21 */

24/* Copy `len` reply bytes into the outbound buffer. Fails closed (NGX_ERROR) if
25 * the reply would exceed the buffer — a reply that large is a bug, not traffic. */
26static ngx_int_t
27ev_out_put(ftp_ev_t *fc, const u_char *data, size_t len)
28{
29 if (len > BRIX_FTP_EV_OB_CAP - fc->ob_len) {
30 ngx_log_error(NGX_LOG_ERR, fc->c->log, 0,
31 "brix: GridFTP(ev) reply exceeds outbound buffer (%uz)",
32 len);
33 return NGX_ERROR;
34 }
35 ngx_memcpy(fc->ob + fc->ob_len, data, len);
36 fc->ob_len += len;
37 return NGX_OK;
38}

74/* GSS-wrap `n` reply bytes and queue them as an RFC 2228 safe/private reply
75 * ("631/632/633 <b64>") over the protected control channel. */
76static ngx_int_t
77ev_reply_wrapped(ftp_ev_t *fc, const u_char *plain, size_t n)
78{
79 ngx_str_t tok, b64;
80 u_char *line, *p;
81 size_t need;
82
83 if (brix_gssapi_wrap(fc->gss, plain, n, &tok) != NGX_OK
84 || ev_b64_encode(fc->c->pool, tok.data, tok.len, &b64) != NGX_OK)
85 {
86 return NGX_ERROR;
87 }
88 need = b64.len + ngx_strlen(fc->wrap_code) + sizeof(" \r\n");
89 line = ngx_pnalloc(fc->c->pool, need);
90 if (line == NULL) {
91 return NGX_ERROR;
92 }
93 p = ngx_slprintf(line, line + need, "%s %V\r\n", fc->wrap_code, &b64);
94 return ev_out_put(fc, line, (size_t) (p - line));
95}
