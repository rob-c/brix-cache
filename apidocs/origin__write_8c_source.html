23/* brix_cache_origin_open_write — kXR_open (update + delete + mkpath) to mirror a
24 * local file onto the origin: truncate the destination and create missing parent
25 * dirs (where supported) for an atomic write-through replacement. Returns 0 with
26 * fhandle set, -1 on error or redirect. */
27int
28brix_cache_origin_open_write(brix_cache_fill_t *t,
29 brix_cache_origin_conn_t *oc, const char *path, uint16_t mode_bits,
30 u_char fhandle[XRD_FHANDLE_LEN])
31{
32 size_t pathlen, total;
33 u_char *buf;
34 ClientOpenRequest *req;
35 uint16_t status;
36 uint32_t dlen;
37 u_char *body;
38
39 if (path == NULL || path[0] == '\0') {
40 brix_cache_set_error(t, kXR_ArgInvalid, 0,
41 "write-through origin path missing");
42 return -1;
43 }
44
45 pathlen = strlen(path);
46 total = sizeof(ClientOpenRequest) + pathlen;
47
48 buf = malloc(total);
49 if (buf == NULL) {
50 brix_cache_set_error(t, kXR_NoMemory, 0,
51 "write-through origin open allocation failed");
52 return -1;
53 }
54
55 ngx_memzero(buf, total);
56 req = (ClientOpenRequest *) buf;
57 req->streamid[1] = 2;
58 req->requestid = htons(kXR_open);
59 /*
60 * Replace the origin copy atomically from the write-through point of view:
61 * open for update, create missing parents if the origin supports mkpath,
62 * and truncate the destination before streaming the local contents.
63 */
64 {
65 xrdw_open_req_t b = {
66 .mode = (uint16_t) (mode_bits != 0 ? mode_bits : 0644),
67 .options = kXR_open_updt | kXR_delete | kXR_mkpath
68 };
69 xrdw_open_req_pack(&b, ((ClientRequestHdr *) buf)->body);
70 }
71 req->dlen = htonl((kXR_int32) pathlen);
72 ngx_memcpy(buf + sizeof(*req), path, pathlen);
73
74 if (brix_cache_io_send(oc, buf, total) != 0) {
75 free(buf);
76 brix_cache_set_error(t, kXR_ServerError, errno,
77 "write-through origin open write failed");
78 return -1;
79 }
80 free(buf);
81
82 body = NULL;
83 if (brix_cache_read_response(t, oc, &status, &body, &dlen,
84 BRIX_MAX_PATH + 256) != 0) {
85 return -1;
86 }
87
88 if (status == kXR_error) {
89 brix_cache_set_origin_error(t, body, dlen,
90 "write-through origin open failed");
91 free(body);
92 return -1;
93 }
94 if (status == kXR_redirect) {
95 free(body);
96 brix_cache_set_error(t, kXR_Unsupported, 0,
97 "write-through origin redirected open; direct "
98 "data server origin is required");
99 return -1;
100 }
101 /* A kXR_open reply is a bare 4-byte fhandle; the cpsize/cptype trailer of
102 * ServerOpenBody (12 bytes) only follows when kXR_compress or kXR_retstat was
103 * requested — and the write-through open requests neither. Require only the
104 * fhandle (XRD_FHANDLE_LEN), not the full struct, or a conformant origin's
105 * minimal 4-byte response is wrongly rejected (which aborted the flush and
106 * left the origin file half-written). The cache never uses cpsize/cptype. */
107 if (status != kXR_ok || dlen < XRD_FHANDLE_LEN) {
108 free(body);
109 brix_cache_set_error(t, kXR_ServerError, 0,
110 "write-through origin open invalid response");
111 return -1;
112 }
113
114 ngx_memcpy(fhandle, ((ServerOpenBody *) body)->fhandle, XRD_FHANDLE_LEN);
115 free(body);
116 return 0;
117}

119/* brix_cache_origin_close_file — send kXR_close for the fhandle and discard the
120 * reply (close status only matters for errors, which don't invalidate data already
121 * written to disk). Every opened file must be closed before reconnect/finish. */
122void
123brix_cache_origin_close_file(brix_cache_origin_conn_t *oc,
124 const u_char fhandle[XRD_FHANDLE_LEN])
125{
126 ClientCloseRequest req;
127 uint16_t rsp_status;
128 uint32_t dlen;
129 u_char *body;
130 brix_cache_fill_t dummy;
131
132 ngx_memzero(&req, sizeof(req));
133 req.streamid[1] = 2;
134 req.requestid = htons(kXR_close);
135 ngx_memcpy(req.fhandle, fhandle, XRD_FHANDLE_LEN);
136 req.dlen = 0;
137
138 (void) brix_cache_io_send(oc, &req, sizeof(req));
139
140 ngx_memzero(&dummy, sizeof(dummy));
141 dummy.result = NGX_OK;
142 body = NULL;
143 if (brix_cache_read_response(&dummy, oc, &rsp_status, &body, &dlen,
144 4096) == 0) {
145 free(body);
146 }
147}

149/* brix_cache_origin_write_chunk — kXR_write a payload at a big-endian 64-bit
150 * offset (htobe64, XRootD wire format); the reply must be kXR_ok with dlen=0.
151 * Returns 0 on success, -1 on error. */
152int
153brix_cache_origin_write_chunk(brix_cache_fill_t *t,
154 brix_cache_origin_conn_t *oc, const u_char fhandle[XRD_FHANDLE_LEN],
155 uint64_t offset, const u_char *data, size_t len)
156{
157 ClientWriteRequest req;
158 uint16_t status;
159 uint32_t dlen;
160 u_char *body;
161
162 if (len > INT32_MAX) {
163 brix_cache_set_error(t, kXR_ArgTooLong, 0,
164 "write-through origin write too large");
165 return -1;
166 }
167
168 ngx_memzero(&req, sizeof(req));
169 req.streamid[1] = 3;
170 req.requestid = htons(kXR_write);
171 ngx_memcpy(req.fhandle, fhandle, XRD_FHANDLE_LEN);
172 req.offset = (kXR_int64) htobe64(offset);
173 req.dlen = htonl((kXR_int32) len);
174
175 if (brix_cache_io_send(oc, &req, sizeof(req)) != 0
176 || (len > 0 && brix_cache_io_send(oc, data, len) != 0))
177 {
178 brix_cache_set_error(t, kXR_ServerError, errno,
179 "write-through origin write failed");
180 return -1;
181 }
182
183 body = NULL;
184 if (brix_cache_read_response(t, oc, &status, &body, &dlen,
185 4096) != 0) {
186 return -1;
187 }
188
189 if (status == kXR_error) {
190 brix_cache_set_origin_error(t, body, dlen,
191 "write-through origin write rejected");
192 free(body);
193 return -1;
194 }
195
196 free(body);
197 if (status != kXR_ok || dlen != 0) {
198 brix_cache_set_error(t, kXR_ServerError, 0,
199 "write-through origin write invalid response");
200 return -1;
201 }
202
203 return 0;
204}

302/* Positional write into a fill sink: a driver staged-write handle (driver-backed
303 * cache) or a raw POSIX fd. Keeps the origin read loop backend-agnostic. */
304int
305brix_cache_sink_pwrite(brix_cache_sink_t *sink, const void *buf, size_t len,
306 off_t off)
307{
308 if (sink->mem != NULL) {
309 /* In-memory sink: a positional copy into the caller's buffer, bounds-
310 * checked against mem_cap. Used by the root:// remote driver's pread. */
311 if (off < 0 || (size_t) off + len > sink->mem_cap) {
312 errno = EINVAL;
313 return -1;
314 }
315 ngx_memcpy(sink->mem + off, buf, len);
316 return 0;
317 }
318 if (sink->staged != NULL) {
319 ssize_t n = sink->staged->inst->driver->staged_write(sink->staged, buf,
320 len, off);
321 return (n == (ssize_t) len) ? 0 : -1;
322 }
323 return brix_cache_fd_write_all(sink->fd, buf, len, off);
324}
