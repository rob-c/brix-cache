40/*
41 * WHAT: Locate the "algo:path"/"algo path" prefix separator (':' or ' ') within the NUL-bounded
42 * leading run of the payload, reporting the algorithm length and the separator offset.
43 * WHY: The legacy non-standard wire form places the algorithm as a "<algo><sep><path>" prefix;
44 * it must be distinguished from a bare path before the path is extracted. Isolating the scan
45 * keeps qcksum_select_algo linear and free of the raw index loop.
46 * HOW: strnlen bounds the scan to the printable prefix, then a single forward pass returns the
47 * first ':'/' ' offset (also the algorithm length) via *alg_len; returns 1 iff a separator
48 * was found, 0 otherwise (out-params zeroed).
49 */
50static ngx_flag_t
51brix_qcksum_find_algo_sep(const u_char *payload, size_t payload_len,
52 size_t *sep_off, size_t *alg_len)
53{
54 size_t wire_len = strnlen((const char *) payload, payload_len);
55
56 *sep_off = 0;
57 *alg_len = 0;
58
59 for (size_t i = 0; i < wire_len; i++) {
60 if (payload[i] == ':' || payload[i] == ' ') {
61 *sep_off = i;
62 *alg_len = i;
63 return 1;
64 }
65 }
66 return 0;
67}

69/*
70 * WHAT: Parse the standard "?cks.type=<algo>" CGI field off the request path, overriding any
71 * legacy prefix algorithm. Writes the algorithm into *algo; sets *bad_algo=1 on an
72 * unrecognized value.
73 * WHY: Stock XRootD clients and EOS request the algorithm via a "?cks.type=<algo>" CGI (XrdCl
74 * appends exactly "?cks.type=<algo>"); honoring it makes `xrdfs ... cksum -a <algo>` and
75 * `xrdcp --cksum` select correctly here, not only via the non-standard prefix form.
76 * HOW: Find '?', then walk '&'-separated fields; for the "cks.type=" field trim the trailing wire
77 * NUL/CR/LF (last-field form has no '&', so the NUL lands inside the value and would corrupt
78 * "adler32\0" → unknown algorithm) before parsing. *bad_algo distinguishes "no cks.type"
79 * (leave algo untouched) from "cks.type present but unknown" (caller must reject).
80 */
81static void
82brix_qcksum_apply_cks_type_cgi(const u_char *path_payload,
83 size_t path_payload_len, char *algo, size_t algo_sz, ngx_flag_t *bad_algo)
84{
85 static const char key[] = "cks.type=";
86 const u_char *qmark;
87 const u_char *p;
88 const u_char *end;
89
90 *bad_algo = 0;
91
92 qmark = memchr(path_payload, '?', path_payload_len);
93 if (qmark == NULL) {
94 return;
95 }
96
97 p = qmark + 1;
98 end = path_payload + path_payload_len;
99 while (p < end) {
100 const u_char *amp = memchr(p, '&', (size_t) (end - p));
101 size_t flen = amp ? (size_t) (amp - p) : (size_t) (end - p);
102
103 while (flen > 0 && (p[flen - 1] == '\0'
104 || p[flen - 1] == '\r'
105 || p[flen - 1] == '\n')) {
106 flen--;
107 }
108 if (flen > sizeof(key) - 1
109 && ngx_strncmp(p, key, sizeof(key) - 1) == 0) {
110 if (!brix_query_parse_algorithm(p + sizeof(key) - 1,
111 flen - (sizeof(key) - 1), algo, algo_sz)) {
112 *bad_algo = 1;
113 }
114 return;
115 }
116 if (amp == NULL) {
117 return;
118 }
119 p = amp + 1;
120 }
121}

123/*
124 * WHAT: Select the checksum algorithm for a path-based Qcksum and split the payload into the
125 * path portion. On success returns NGX_OK with the path_payload / path_payload_len out-params
126 * pointing at the request path; on an unknown algorithm sends kXR_ArgInvalid, sets *out_rc to
127 * that send result, and returns NGX_DECLINED.
128 * WHY: Two independent algorithm sources feed one selection: the legacy "<algo><sep><path>"
129 * prefix and the standard "?cks.type=<algo>" CGI (which overrides the prefix). INVARIANT 9:
130 * crc64 (CRC-64/XZ) and crc64nvme (CRC-64/NVME) are DISTINCT names to brix_checksum_parse —
131 * this selector never conflates them; it only forwards the wire token to the parser.
132 * HOW: Detect the prefix separator (brix_qcksum_find_algo_sep) and, when present and non-empty,
133 * parse the prefix algorithm and advance the path payload past it. Then apply the CGI
134 * override (brix_qcksum_apply_cks_type_cgi), which wins. Any unknown token → error send.
135 */
136static ngx_int_t
137brix_qcksum_select_algo(brix_qcksum_req_t *rq, const u_char **path_payload,
138 size_t *path_payload_len, ngx_int_t *out_rc)
139{
140 const u_char *payload = rq->ctx->recv.payload;
141 size_t payload_len = (size_t) rq->ctx->recv.cur_dlen;
142 size_t sep_off = 0;
143 size_t alg_len = 0;
144 ngx_flag_t bad_algo = 0;
145
146 *path_payload = payload;
147 *path_payload_len = payload_len;
148
149 if (brix_qcksum_find_algo_sep(payload, payload_len, &sep_off, &alg_len)
150 && alg_len > 0 && alg_len + 1 < payload_len)
151 {
152 if (!brix_query_parse_algorithm(payload, alg_len, rq->algo,
153 sizeof(rq->algo))) {
154 BRIX_OP_ERR(rq->ctx, BRIX_OP_QUERY_CKSUM);
155 *out_rc = brix_send_error(rq->ctx, rq->c, kXR_ArgInvalid,
156 "unknown checksum algorithm");
157 return NGX_DECLINED;
158 }
159 *path_payload = payload + sep_off + 1;
160 *path_payload_len = payload_len - (alg_len + 1);
161 }
162
163 brix_qcksum_apply_cks_type_cgi(*path_payload, *path_payload_len,
164 rq->algo, sizeof(rq->algo), &bad_algo);
165 if (bad_algo) {
166 BRIX_OP_ERR(rq->ctx, BRIX_OP_QUERY_CKSUM);
167 *out_rc = brix_send_error(rq->ctx, rq->c, kXR_ArgInvalid,
168 "unknown checksum algorithm");
169 return NGX_DECLINED;
170 }
171 return NGX_OK;
172}

174/*
175 * WHAT: Manager / redirector bounce for a path-based Qcksum. Returns NGX_DECLINED when the caller
176 * should proceed with the local resolve; otherwise sets *out_rc to the terminal result
177 * (redirect sent, not-found sent, NGX_AGAIN for an async CMS locate) and returns NGX_OK.
178 * WHY: A pure redirector keeps no local copy, so a kXR_Qcksum — file metadata about one path —
179 * must be bounced to the data server that authoritatively holds it, exactly as kXR_stat and
180 * kXR_open do. Isolating the bounce keeps the orchestrator's happy path flat.
181 * HOW: tried/triedrc convergence → not-found once every holder was visited; registry hit →
182 * redirect; registry miss with a CMS parent → async kYR_locate (NGX_AGAIN); all misses →
183 * NGX_DECLINED so the caller falls through to the local resolve (404).
184 */
185static ngx_int_t
186brix_qcksum_manager_bounce(brix_qcksum_req_t *rq, ngx_int_t *out_rc)
187{
188 brix_ctx_t *ctx = rq->ctx;
189 ngx_connection_t *c = rq->c;
190 ngx_stream_brix_srv_conf_t *conf = rq->conf;
191 const char *pathbuf = rq->pathbuf;
192 char redir_host[256];
193 uint16_t redir_port;
194
195 if (!conf->manager_mode) {
196 return NGX_DECLINED;
197 }
198
199 /* tried/triedrc: converge to not-found once the client has visited
200 * every server holding this path (avoids the redirect-limit loop).
201 * Terminal outcomes are reported to the caller via *out_rc (mirroring
202 * BRIX_RETURN_ERR / BRIX_RETURN_REDIR, whose embedded `return` cannot be
203 * used from a helper that also signals "declined → local resolve"). */
204 if (brix_manager_tried_exhausted(ctx->recv.payload, ctx->recv.cur_dlen,
205 pathbuf)) {
206 brix_log_access(ctx, c, "QUERY", pathbuf, "cksum",
207 0, kXR_NotFound, "file not found on any data server",
208 0);
209 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSUM);
210 *out_rc = brix_send_error(ctx, c, kXR_NotFound,
211 "file not found on any data server");
212 return NGX_OK;
213 }
214
215 if (brix_srv_select(pathbuf, 0, redir_host,
216 sizeof(redir_host), &redir_port)) {
217 brix_log_access(ctx, c, "QUERY", pathbuf, "registry",
218 1, kXR_ok, NULL, 0);
219 BRIX_OP_OK(ctx, BRIX_OP_QUERY_CKSUM);
220 *out_rc = brix_send_redirect(ctx, c, redir_host, redir_port);
221 return NGX_OK;
222 }
223
224 /* Registry miss — ask the CMS parent via kYR_locate (async). */
225 if (conf->cms.ctx != NULL) {
226 uint32_t streamid;
227
228 streamid = ngx_brix_cms_next_streamid(conf->cms.ctx);
229 if (brix_pending_insert(streamid, ngx_pid, c->fd, c->number,
230 ctx->recv.cur_streamid,
231 conf->cms.locate_timeout) == NGX_OK)
232 {
233 ctx->cms_wait_streamid = streamid;
234 ctx->state = XRD_ST_WAITING_CMS;
235 ngx_add_timer(c->read, conf->cms.locate_timeout);
236 if (ngx_brix_cms_send_locate(conf->cms.ctx, streamid,
237 pathbuf) == NGX_OK)
238 {
239 *out_rc = NGX_AGAIN;
240 return NGX_OK;
241 }
242 ngx_del_timer(c->read);
243 ctx->state = XRD_ST_REQ_HEADER;
244 brix_pending_remove(streamid, ngx_pid);
245 }
246 }
247 /* Registry + CMS miss: fall through to the local resolve (404). */
248 return NGX_DECLINED;
249}

309/*
310 * WHAT: Open the resolved path through the VFS seam for checksum computation, returning the open
311 * handle. On failure returns NULL and sets *out_rc to the terminal result (cache-origin
312 * redirect on a read-through miss, otherwise an errno-mapped error send).
313 * WHY: A checksum read is a full-content origin fetch, so the open mirrors the davs/S3 GET
314 * confinement + credential-bind path exactly (impersonation-aware, metered, opt-in mint).
315 * Keeping the cache-miss redirect beside the open localizes the "not yet pulled" decision.
316 * HOW: Init a VFS ctx bound to the storage credential dir + mint CA + delegation, open O_READ.
317 * On ENOENT with a configured cache origin → redirect the metadata query there (the origin
318 * holds the authoritative bytes); otherwise map errno → kXR and send.
319 */
320static brix_vfs_file_t *
321brix_qcksum_open(brix_qcksum_req_t *rq, ngx_int_t *out_rc)
322{
323 brix_ctx_t *ctx = rq->ctx;
324 ngx_connection_t *c = rq->c;
325 ngx_stream_brix_srv_conf_t *conf = rq->conf;
326 const char *pathbuf = rq->pathbuf;
327 const char *full_path = rq->full_path;
328 brix_vfs_ctx_t vctx;
329 brix_vfs_file_t *fh;
330 int vfs_err = 0;
331
332 /* Confined read-open through the VFS seam (impersonation-aware, metered).
333 * The fd backs the backend-agnostic checksum kernel; the handle is closed
334 * via brix_vfs_close on every exit (sync below, or the aio done cb). */
335 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
336 conf->common.root_canon, NULL, conf->common.allow_write,
337 0 /* is_tls */, ctx->identity, full_path);
338 brix_vfs_ctx_bind_backend_cred(&vctx,
339 &conf->common.storage_credential_dir,
340 conf->common.storage_credential_fallback);
341 /* Phase-3 T1: opt-in credential minting, mirroring the davs/S3 GET
342 * mint bind — a checksum read is a full-content origin fetch. */
343 brix_vfs_ctx_bind_backend_mint(&vctx,
344 &conf->common.storage_credential_mint_ca_cert,
345 &conf->common.storage_credential_mint_ca_key,
346 conf->common.storage_credential_mint_ttl);
347 brix_root_vfs_bind_deleg(ctx, conf, &vctx);
348 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
349 errno = vfs_err;
350
351 if (fh != NULL) {
352 return fh;
353 }
354
355 /*
356 * Read-through cache miss: the file has not been pulled from the origin
357 * yet (a cached hit is computed locally above). The origin holds the
358 * authoritative bytes — identical to anything we would later cache — so
359 * redirect the metadata query there rather than returning "not found".
360 */
361 if (errno == ENOENT && conf->cache_origin_host.len > 0) {
362 char origin_host[256];
363 size_t hlen = conf->cache_origin_host.len < sizeof(origin_host)
364 ? conf->cache_origin_host.len : sizeof(origin_host) - 1;
365
366 ngx_memcpy(origin_host, conf->cache_origin_host.data, hlen);
367 origin_host[hlen] = '\0';
368 /* Equivalent to BRIX_RETURN_REDIR but through the *out_rc out-param
369 * (the macro embeds its own `return`, which cannot leave a helper that
370 * signals failure by returning NULL). */
371 brix_log_access(ctx, c, "QUERY", pathbuf, "cache-origin",
372 1, kXR_ok, NULL, 0);
373 BRIX_OP_OK(ctx, BRIX_OP_QUERY_CKSUM);
374 *out_rc = brix_send_redirect(ctx, c, origin_host,
375 conf->cache_origin_port);
376 return NULL;
377 }
378 /* Equivalent to BRIX_RETURN_ERR via the *out_rc out-param. */
379 {
380 uint16_t code = brix_kxr_from_errno(errno);
381 brix_log_access(ctx, c, "QUERY", full_path, "cksum",
382 0, code, strerror(errno), 0);
383 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSUM);
384 *out_rc = brix_send_error(ctx, c, code, strerror(errno));
385 }
386 return NULL;
387}

389/*
390 * WHAT: Try to offload a whole-object checksum to the thread pool from an open VFS handle.
391 * Returns NGX_OK when the task was posted (*handled=1) OR the fall-through-to-sync case
392 * (*handled=0); returns a terminal rc with *handled=1 when the task could not be created or
393 * posted (the handle is closed on those failure exits).
394 * WHY: Async execution prevents the event loop blocking on large-file checksums. The path
395 * variant owns the handle (close_fd=1: the done cb releases the VFS handle), unlike the
396 * handle variant which computes on an already-open fd it does not own.
397 * HOW: Alloc the task, populate the brix_cksum_aio_t (fd/fh/obj/streamid/algo/resolved), bind
398 * the thread + done callbacks, then brix_aio_post_task. A full queue leaves *handled=0 so
399 * the caller reuses the still-open handle for a synchronous compute.
400 */
401static ngx_int_t
402brix_qcksum_try_async(brix_qcksum_req_t *rq, brix_vfs_file_t *fh, int fd,
403 ngx_flag_t *handled)
404{
405 brix_ctx_t *ctx = rq->ctx;
406 ngx_connection_t *c = rq->c;
407 ngx_stream_brix_srv_conf_t *conf = rq->conf;
408 brix_cksum_aio_t *t;
409 ngx_thread_task_t *task;
410 ngx_flag_t posted = 0;
411
412 *handled = 0;
413
414 if (conf->common.thread_pool == NULL) {
415 return NGX_OK;
416 }
417
418 task = ngx_thread_task_alloc(c->pool, sizeof(brix_cksum_aio_t));
419 if (task == NULL) {
420 brix_vfs_close(fh, c->log);
421 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSUM);
422 *handled = 1;
423 return brix_send_error(ctx, c, kXR_NoMemory, "out of memory");
424 }
425
426 t = task->ctx;
427 t->ctx = ctx;
428 t->c = c;
429 t->conf = conf;
430 t->fd = fd;
431 t->fh = fh; /* done cb releases the VFS handle */
432 t->close_fd = 1;
433 brix_vfs_file_sd_obj(fh, &t->obj); /* Layer 3: whole-object checksum */
434 ngx_memcpy(t->streamid, ctx->recv.cur_streamid, 2);
435 ngx_cpystrn((u_char *) t->algo, (u_char *) rq->algo, sizeof(t->algo));
436 ngx_cpystrn((u_char *) t->resolved, (u_char *) rq->full_path,
437 sizeof(t->resolved));
438 t->error_code = 0;
439
440 brix_task_bind(task, brix_cksum_aio_thread, brix_cksum_aio_done);
441 task->ctx = t;
442
443 if (brix_aio_post_task(ctx, c, conf->common.thread_pool, task,
444 "cksum thread pool queue full, using sync",
445 &posted) != NGX_OK)
446 {
447 brix_vfs_close(fh, c->log);
448 *handled = 1;
449 return NGX_ERROR;
450 }
451
452 if (posted) {
453 *handled = 1;
454 return NGX_OK;
455 }
456 /* not posted → fall through to the sync path, reusing the open handle. */
457 return NGX_OK;
458}

493/* WHAT: Parses a payload containing "algo:path", resolves the path through security checks (authdb, VO ACL, token scope),
494 * opens the file confined, and delegates checksum computation to brix_query_build_checksum(). Returns hex-formatted result.
495 * WHY: kXR_Qcksum can query either open-file handles or arbitrary paths; this handler implements the path-based variant with
496 * full security chain verification before accessing any filesystem resource.
497 * HOW: resolve (algo select + path extract + manager bounce + auth gate) → VFS open (or cache-origin redirect) →
498 * try async offload, else compute synchronously + reply. Each stage early-returns on its terminal outcome. */
499
500ngx_int_t
501brix_query_cksum_path(brix_ctx_t *ctx, ngx_connection_t *c,
502 ngx_stream_brix_srv_conf_t *conf)
503{
504 brix_qcksum_req_t rq;
505 brix_vfs_file_t *fh;
506 int fd;
507 ngx_int_t rc = NGX_ERROR;
508 ngx_flag_t handled = 0;
509

511 rq.ctx = ctx;
