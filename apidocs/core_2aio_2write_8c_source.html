7/*
8 * brix_write_aio_thread — thread-pool worker for kXR_write / kXR_pgwrite.
9 *
10 * Runs on a worker thread; must not touch nginx state, connection pools, or
11 * any field that is not owned by the task struct.
12 */
13void
14brix_write_aio_thread(void *data, ngx_log_t *log)
15{
16 brix_write_aio_t *t = data;
17 brix_vfs_job_t job;
18
19 /*
20 * Off-loop blocking write. Capture errno into the task struct immediately:
21 * errno is thread-local and would be clobbered before the done callback runs
22 * on the main thread. No nginx state may be touched from here.
23 */
24 brix_vfs_job_write_init(&job, t->fd, t->offset, t->data, t->len);
25 job.csi = t->csi; /* phase-59 W2: update tags in worker */
26 brix_vfs_job_set_obj(&job, &t->obj); /* Layer 3: route via driver if bound */
27 brix_vfs_io_execute(&job);
28
29 t->nwritten = job.nio;
30 t->io_errno = job.io_errno;
31}

33/*
34 * Access-log outcome for one WRITE AIO completion. Bundles the four
35 * per-outcome fields so the shared logger stays within the parameter budget and
36 * so success/error callsites name their arguments explicitly.
37 */
38typedef struct {
39 int ok; /* 1 = success line, 0 = error line */
40 int xrd_error; /* kXR error code on the error line, 0 otherwise */
41 const char *errmsg; /* error detail string, NULL on success */
42 size_t nbytes; /* bytes written on success, 0 on error */
43} brix_write_aio_logrec_t;

45/*
46 * WHAT: Emit one WRITE access-log line for this AIO completion, gated on the
47 * access-log fd being open.
48 * WHY: Every outcome (error, short-write, success) in both the pipelined and
49 * serial paths logs the same "<req_offset>+<len>" detail through the same
50 * fd-open guard; centralizing the snprintf + guard keeps the log bytes and
51 * the gate identical across all six callsites (the tests grep these lines).
52 * HOW: Skip when access_log_fd is closed; otherwise format the fixed
53 * "%lld+%zu" detail from req_offset/len and forward every field of the
54 * outcome record verbatim to brix_log_access. c is the live connection
55 * (callers run this only after the destroyed/restore guards).
56 */
57static void
58brix_write_aio_log(brix_ctx_t *ctx, ngx_connection_t *c,
59 ngx_stream_brix_srv_conf_t *rconf, brix_write_aio_t *t,
60 const brix_write_aio_logrec_t *rec)
61{
62 char detail[64];
63
64 if (rconf->access_log_fd == NGX_INVALID_FILE) {
65 return;
66 }
67
68 snprintf(detail, sizeof(detail), "%lld+%zu",
69 (long long) t->req_offset, t->len);
70 brix_log_access(ctx, c, "WRITE", t->path, detail,

74/*
75 * WHAT: Commit the success-path accounting and cache/journal side effects for a
76 * completed single-region write.
77 * WHY: The pipelined and serial success paths perform the identical
78 * bytes_written accounting, write-through dirty-marking, and recovery-journal
79 * record; sharing them keeps the two paths in lockstep so a fix to one can
80 * never drift from the other.
81 * HOW: Add nwritten to the per-handle and totals byte counters, then, when the
82 * handle has write-through enabled, mark the written range dirty (the API
83 * takes the LAST inclusive offset, hence req_offset + nwritten - 1) and,
84 * when journalling is enabled, record the committed write. Pure accounting
85 * plus cache/journal calls — no wire I/O.
86 */
87static void
88brix_write_aio_commit(brix_ctx_t *ctx, brix_write_aio_t *t)
89{
90 ctx->files[t->handle_idx].bytes_written += (size_t) t->nwritten;
91 ctx->totals.bytes_written += (size_t) t->nwritten;
92
93 if (ctx->files[t->handle_idx].wt_enabled) {
94 brix_wt_mark_dirty(ctx, t->handle_idx,
95 t->req_offset + (int64_t) t->nwritten - 1,
96 (size_t) t->nwritten);
97 }
98
99 if (ctx->files[t->handle_idx].wrts_enabled) {
100 brix_wrts_record(&ctx->files[t->handle_idx], t->req_offset,
101 (uint32_t) t->nwritten);
102 }
103}

105/*
106 * WHAT: Handle the pipelined plain-kXR_write completion (the !is_pgwrite path).
107 * WHY: A pipelined write must not disturb the recv loop, which has kept
108 * receiving and posting further writes while this one ran; we restore only
109 * the streamid for this ack's frame, reply asynchronously (parked in the
110 * out_ring, drained by the write event), then nudge the read side in case
111 * recv throttled on the pipeline depth.
112 * HOW: Restore the streamid (on failure the connection is dead — run any held
113 * deferred teardown and return). Classify the outcome into errmsg
114 * (hard error / short write / NULL=success). On error: log, count, and
115 * send kXR_IOError under resp_async, then schedule a read resume. On
116 * success: commit accounting via brix_write_aio_commit, log, count, and
117 * send kXR_ok under resp_async, then schedule a read resume. Never touches
118 * the recv state/hdr_pos.
119 */
120static void
121brix_write_aio_done_pipelined(brix_ctx_t *ctx, ngx_connection_t *c,
122 ngx_stream_brix_srv_conf_t *rconf,
123 brix_write_aio_t *t, ngx_int_t op)
124{
125 const char *errmsg = NULL;
126
127 if (!brix_aio_restore_stream(ctx, t->streamid)) {
128 if (ctx->out.finalize_pending && ctx->out.wr_inflight == 0) {
129 brix_run_deferred_teardown(ctx, c);
130 }
131 return;
132 }
133
134 if (t->nwritten < 0) {
135 errmsg = t->io_errno ? strerror(t->io_errno) : "async write error";
136 } else if ((size_t) t->nwritten < t->len) {
137 errmsg = "short write (disk full?)";
138 }
139
140 if (errmsg != NULL) {
141 brix_write_aio_logrec_t rec = { 0, kXR_IOError, errmsg, 0 };
142 brix_write_aio_log(ctx, c, rconf, t, &rec);
143 BRIX_OP_ERR(ctx, op);
144 ctx->out.resp_async = 1;
145 (void) brix_send_error(ctx, c, kXR_IOError, errmsg);
146 ctx->out.resp_async = 0;
147 (void) brix_schedule_read_resume(c);
148 return;
149 }
150
151 brix_write_aio_commit(ctx, t);
152 {
153 brix_write_aio_logrec_t rec = { 1, 0, NULL, (size_t) t->nwritten };
154 brix_write_aio_log(ctx, c, rconf, t, &rec);
155 }
156 BRIX_OP_OK(ctx, op);
157
158 ctx->out.resp_async = 1;
159 (void) brix_send_ok(ctx, c, NULL, 0);
160 ctx->out.resp_async = 0;
161 (void) brix_schedule_read_resume(c);
162}

164/*
165 * WHAT: Send the success reply for a completed serial write and re-arm the
166 * connection.
167 * WHY: The serial reply framing differs by opcode: kXR_pgwrite expects a
168 * kXR_status response whose "info" offset echoes the REQUEST offset (where
169 * the data landed, matching the reference do_pgWrite — not offset+len,
170 * which diverged from stock in test_conf_pgio), and a page that failed
171 * CRC32c gets a CSE retransmit frame instead; plain kXR_write expects a
172 * bare kXR_ok.
173 * HOW: For pgwrite, send the CSE retransmit list when bad_page_count > 0 else a
174 * plain pgwrite status at req_offset; for plain write, send kXR_ok. Then
175 * resume the recv loop for the next op.
176 */
177static void
178brix_write_aio_serial_reply_ok(brix_ctx_t *ctx, ngx_connection_t *c,
179 brix_write_aio_t *t)
180{
181 if (t->is_pgwrite) {
182 /* CSE: one or more pages failed CRC32c. The data is already on disk
183 * (accept-then-correct); reply with the retransmit list so the client
184 * resends those pages with kXR_pgRetry. */
185 if (t->bad_page_count > 0) {
186 brix_send_pgwrite_cse(ctx, c, t->req_offset,
187 t->bad_pages, t->bad_page_count);
188 } else {
189 brix_send_pgwrite_status(ctx, c, t->req_offset);
190 }
191 } else {
192 brix_send_ok(ctx, c, NULL, 0);
193 }
194
195 brix_aio_resume(c);
196}

198/*
199 * WHAT: Handle the serial kXR_pgwrite / non-pipelined kXR_write completion.
200 * WHY: pgwrite is not pipelined, so the recv loop is suspended in XRD_ST_AIO; it
201 * is safe to restore the FULL request context (state + hdr_pos) and drive
202 * the response + resume synchronously.
203 * HOW: Restore the request (on failure the connection is dead — return). On a
204 * hard error (nwritten < 0) or a short write (nwritten < len) — which the
205 * kXR_write protocol cannot acknowledge partially — log, count, send
206 * kXR_IOError, and resume. Otherwise commit accounting via
207 * brix_write_aio_commit, log the success, count it, and send the
208 * opcode-appropriate success reply.
209 */
210static void
211brix_write_aio_done_serial(brix_ctx_t *ctx, ngx_connection_t *c,
212 ngx_stream_brix_srv_conf_t *rconf,
213 brix_write_aio_t *t, ngx_int_t op)
214{
215 if (!brix_aio_restore_request(ctx, t->streamid)) {
216 return;
217 }
218
219 if (t->nwritten < 0) {
220 const char *errmsg = t->io_errno ? strerror(t->io_errno)
221 : "async write error";
222 brix_write_aio_logrec_t rec = { 0, kXR_IOError, errmsg, 0 };
223 brix_write_aio_log(ctx, c, rconf, t, &rec);
224 BRIX_OP_ERR(ctx, op);
225 brix_send_error(ctx, c, kXR_IOError, errmsg);
226 brix_aio_resume(c);

228 }

230 /*

238 "short write (disk full?)", 0 };
239 brix_write_aio_log(ctx, c, rconf, t, &rec);
240 BRIX_OP_ERR(ctx, op);
241 brix_send_error(ctx, c, kXR_IOError, "short write (disk full?)");
242 brix_aio_resume(c);
243 return;
244 }

246 brix_write_aio_commit(ctx, t);

249 brix_write_aio_log(ctx, c, rconf, t, &rec);

256/*
257 * brix_write_aio_done — main-thread completion callback for kXR_write / kXR_pgwrite AIO.
258 *
259 * WHAT: Bridge the thread-pool write worker back to the event loop: free the
260 * detached payload, keep the pipelining counter honest, run any deferred
261 * teardown on a torn-down connection, then dispatch to the pipelined or
262 * serial completion handler.
263 * WHY: The two reply disciplines are structurally different (async out-ring vs
264 * synchronous restore/resume) and each has its own liveness guard; splitting
265 * them keeps this callback a flat guard-then-dispatch sequence.
266 * HOW: Free payload_to_free unconditionally (we own it regardless of liveness);
267 * decrement wr_inflight for a plain write BEFORE the liveness check so a
268 * deferred teardown fires on the LAST completion; on ctx->destroyed run the
269 * held teardown (last pipelined completion only) and return; otherwise fetch
270 * rconf and call the pipelined or serial handler.
271 */
272void
273brix_write_aio_done(ngx_event_t *ev)
274{
275 ngx_thread_task_t *task = ev->data;
276 brix_write_aio_t *t = task->ctx;
277 brix_ctx_t *ctx = t->ctx;
278 ngx_connection_t *c = t->c;
279 ngx_stream_brix_srv_conf_t *rconf;
280 ngx_int_t op = BRIX_OP_WRITE;
281 ngx_flag_t pipelined = !t->is_pgwrite;
282
283 /*
284 * Free the detached payload first, before the restore guard can bail out.
285 * The worker copied the wire bytes into this heap buffer so the recv buffer
286 * could be reused while the write was in flight; we own it regardless of
287 * whether the connection still exists, so free it unconditionally to avoid
288 * a leak on the early-return path below.
289 */
290 if (t->payload_to_free) {
291 ngx_free(t->payload_to_free);
292 t->payload_to_free = NULL;
293 }
294
295 /*
296 * Write pipelining: a plain kXR_write is no longer in flight. Decrement
297 * BEFORE the liveness check so a deferred teardown (client disconnected while
298 * writes were still running) fires on the LAST completion, once no pwrite
299 * references ctx/fds.
300 */
301 if (pipelined && ctx->out.wr_inflight > 0) {
302 ctx->out.wr_inflight--;
303 }
304
305 if (ctx->destroyed) {
306 /*
307 * Connection torn down — or a disconnect/timeout was deferred because
308 * writes were in flight. Touch nothing further except, on the last
309 * pipelined completion, running the teardown that was held off.
310 */
311 if (pipelined && ctx->out.finalize_pending && ctx->out.wr_inflight == 0) {
312 brix_run_deferred_teardown(ctx, c); /* frees ctx — return now */
313 }
314 return;
315 }
316
317 rconf = ngx_stream_get_module_srv_conf(
318 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
319
320 if (pipelined) {
321 brix_write_aio_done_pipelined(ctx, c, rconf, t, op);
322 return;
323 }
324
325 brix_write_aio_done_serial(ctx, c, rconf, t, op);
326}

328/*
329 * brix_writev_write_aio_thread — thread-pool worker for kXR_writev multi-segment write.
330 *
331 * Runs on a worker thread. io_error=1 on hard failure, io_error=2 on short write.
332 * If do_sync is set, fsync(2) is called per unique fd after all segments succeed.
333 */
334void
335brix_writev_write_aio_thread(void *data, ngx_log_t *log)
336{
337 brix_writev_aio_t *t = data;
338 brix_vfs_job_t job;
339
340 t->bytes_total = 0;
341 t->io_error = 0;
342 t->err_msg[0] = '\0';
343
344 /*
345 * Write each segment to its own (fd, offset). On the first failing segment
346 * we record the error and return immediately — earlier segments may already
347 * be on disk, but the protocol reply is all-or-nothing, so we do not attempt
348 * to roll back or continue. io_error encodes the failure kind for the done
349 * callback: 1 = hard pwrite error, 2 = short write.
350 */
351 ngx_memzero(&job, sizeof(job));
352 job.op = BRIX_VFS_IO_WRITEV;
353 job.segs = t->segs;
354 job.nsegs = t->n_segs;
355 job.do_sync = t->do_sync ? 1 : 0;
356 job.err_msg = t->err_msg;
357 job.err_msg_cap = sizeof(t->err_msg);
358
359 brix_vfs_io_execute(&job);
360
361 t->bytes_total = job.out_size;
362 if (job.io_errno != 0) {
363 t->io_error = job.short_io ? 2 : 1;
364 if (t->err_msg[0] == '\0') {
365 snprintf(t->err_msg, sizeof(t->err_msg),
366 "writev I/O error: %s", strerror(job.io_errno));
367 }
368 }
369}

400 }
401

411 * the main thread (the worker only does I/O). Each segment may target a

416 brix_file_t *file = &ctx->files[t->segs[i].handle_idx];

419

428 brix_wt_mark_dirty(ctx, t->segs[i].handle_idx,
429 t->segs[i].offset + (off_t) t->segs[i].wlen - 1,

431 }
432 }
433 }
434 ctx->totals.bytes_written += t->bytes_total;
435
436 rconf = ngx_stream_get_module_srv_conf(
437 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
438 if (rconf->access_log_fd != NGX_INVALID_FILE) {
439 char detail[64];
440 snprintf(detail, sizeof(detail), "%zu_segs", t->n_segs);
441 brix_log_access(ctx, c, "WRITEV", "-", detail, 1, 0, NULL,
442 t->bytes_total);
443 }
444 BRIX_OP_OK(ctx, BRIX_OP_WRITEV);

447 brix_aio_resume(c);
