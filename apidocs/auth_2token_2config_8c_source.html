5/* ---- Build the multi-issuer token registry from brix_token_config ----
6 *
7 * WHAT: Constructs a brix_token_registry_t from the brix_token_config file and
8 * stores it on xcf->token_registry. Warns (non-fatally) when the legacy
9 * single-issuer directives are also set, since the registry supersedes
10 * them. Returns NGX_OK on success, NGX_ERROR if registry construction
11 * fails.
12 *
13 * WHY: The multi-issuer registry (phase-59 W1) is the modern replacement for the
14 * single brix_token_jwks/_issuer/_audience directives. When both are
15 * present the registry wins; the operator is warned so they understand the
16 * single-issuer settings are being ignored rather than merged.
17 *
18 * HOW: 1) If any single-issuer directive is non-empty, emit a WARN that the
19 * registry supersedes them.
20 * 2) Call brix_token_registry_build() with the config path and the
21 * capability authz mode; return NGX_ERROR on failure.
22 * 3) Publish the built registry on xcf and return NGX_OK.
23 */
24static ngx_int_t
25brix_token_configure_registry(ngx_conf_t *cf,
26 ngx_stream_brix_srv_conf_t *xcf)
27{
28 brix_token_registry_t *reg = NULL;
29
30 /* Multi-issuer registry (phase-59 W1): when brix_token_config is set it
31 * supersedes the single-issuer token_jwks/_issuer/_audience directives. */
32 if (xcf->token_issuer.len || xcf->token_audience.len
33 || xcf->token_jwks.len)
34 {
35 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
36 "brix_token_config supersedes brix_token_issuer/"
37 "_audience/_jwks (single-issuer directives ignored)");
38 }
39 if (brix_token_registry_build(cf,
40 (const char *) xcf->token_config.data,
41 BRIX_AUTHZ_CAPABILITY, &reg) != NGX_OK)
42 {
43 return NGX_ERROR;
44 }
45 xcf->token_registry = reg;
46 return NGX_OK;
47}

49/* ---- Require the three single-issuer token directives ----
50 *
51 * WHAT: Verifies that brix_token_jwks, brix_token_issuer and brix_token_audience
52 * are all non-empty. Returns NGX_OK when all three are present, otherwise
53 * logs an emerg-level error and returns NGX_ERROR.
54 *
55 * WHY: Single-issuer token authentication cannot function without a JWKS key
56 * file, an issuer to validate token claims against, and an audience to
57 * restrict which tokens are accepted. Catching the omission at
58 * configuration-validation time (nginx -t) prevents starting a server with
59 * auth that would reject every client at runtime.
60 *
61 * HOW: 1) If any of the three ngx_str_t fields has zero length, log an emerg
62 * error naming all three directives and return NGX_ERROR.
63 * 2) Otherwise return NGX_OK.
64 */
65static ngx_int_t
66brix_token_require_single_issuer_fields(ngx_conf_t *cf,
67 ngx_stream_brix_srv_conf_t *xcf)
68{
69 if (xcf->token_jwks.len == 0 || xcf->token_issuer.len == 0
70 || xcf->token_audience.len == 0)
71 {
72 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
73 "brix_auth token/both requires "
74 "brix_token_jwks, "
75 "brix_token_issuer and brix_token_audience");
76 return NGX_ERROR;
77 }
78
79 return NGX_OK;
80}

82/* ---- Validate, load and register the JWKS keys ----
83 *
84 * WHAT: Validates the brix_token_jwks path (regular file, readable), loads its
85 * public keys into xcf->jwks_keys, registers a pool cleanup for the loaded
86 * keys, and records the file mtime for the refresh timer. Returns NGX_OK on
87 * success, NGX_ERROR on any validation, load, or cleanup-registration
88 * failure.
89 *
90 * WHY: JWT verification needs the issuer's public keys resident in memory. A
91 * loaded key count below zero indicates file corruption or invalid format
92 * (distinct from mere absence), which must fail configuration rather than
93 * allow a server with unusable keys. The mtime lets the refresh timer detect
94 * later changes to the JWKS file.
95 *
96 * HOW: 1) Validate the JWKS path via brix_validate_path() (regular file, R_OK);
97 * return NGX_ERROR on failure.
98 * 2) Load keys via brix_jwks_load(); a negative count is a load failure —
99 * log emerg and return NGX_ERROR.
100 * 3) If any keys loaded, register a pool cleanup for them; return NGX_ERROR
101 * if that fails.
102 * 4) stat() the file and record st_mtime; a stat failure is ignored so the
103 * mtime simply stays unset.
104 * 5) Return NGX_OK.
105 */
106static ngx_int_t
107brix_token_load_and_register_jwks(ngx_conf_t *cf,
108 ngx_stream_brix_srv_conf_t *xcf)
109{
110 struct stat st;
111
112 if (brix_validate_path(cf, "brix_token_jwks",
113 &xcf->token_jwks,
114 BRIX_PATH_REGULAR_FILE, R_OK)
115 != NGX_OK)
116 {
117 return NGX_ERROR;
118 }
119
120 xcf->jwks_key_count = brix_jwks_load(
121 cf->log, (const char *) xcf->token_jwks.data,
122 xcf->jwks_keys, BRIX_MAX_JWKS_KEYS);
123
124 if (xcf->jwks_key_count < 0) {
125 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
126 "brix: failed to load JWKS from \"%s\"",
127 xcf->token_jwks.data);
128 return NGX_ERROR;
129 }
130
131 if (xcf->jwks_key_count > 0
132 && brix_jwks_register_cleanup(cf->pool, xcf->jwks_keys,
133 &xcf->jwks_key_count) != NGX_OK)
134 {
135 return NGX_ERROR;
136 }
137
138 /* Record mtime so the refresh timer can detect changes */
139 if (stat((const char *) xcf->token_jwks.data, &st) == 0) {
140 xcf->jwks_mtime = st.st_mtime;
141 }
142
143 return NGX_OK;
144}

167/* ---- Validate and materialise token/JWT auth configuration ----
168 *
169 * WHAT: Validates token auth configuration when auth mode is BRIX_AUTH_TOKEN or
170 * BRIX_AUTH_BOTH. Dispatches to the multi-issuer registry when
171 * brix_token_config is set; otherwise requires the single-issuer
172 * directives, loads the JWKS keys, and logs a summary. Returns NGX_OK when
173 * auth is not token-based or is fully configured, NGX_ERROR on any
174 * validation failure.
175 *
176 * WHY: Missing or invalid token auth prerequisites would cause runtime auth
177 * failures for all clients — nginx -t must catch this at configuration
178 * validation time rather than allowing the server to start with broken
179 * auth. Config setup runs once during nginx startup on the main process
180 * thread; there is no concurrent access after initialization.
181 *
182 * HOW: 1) Early-exit NGX_OK if auth mode is not token/both.
183 * 2) If brix_token_config is set, delegate entirely to the multi-issuer
184 * registry builder and return its result.
185 * 3) Require the three single-issuer directives; return NGX_ERROR if any is
186 * missing.
187 * 4) Validate, load, and register the JWKS keys (and capture mtime); return
188 * NGX_ERROR on failure.
189 * 5) Log the configuration summary and return NGX_OK.
190 */
191ngx_int_t
192brix_configure_token_auth(ngx_conf_t *cf,
193 ngx_stream_brix_srv_conf_t *xcf)
194{
195 if (xcf->auth != BRIX_AUTH_TOKEN && xcf->auth != BRIX_AUTH_BOTH) {
196 return NGX_OK;
197 }
198
199 /* Multi-issuer registry (phase-59 W1) supersedes single-issuer directives. */
200 if (xcf->token_config.len > 0) {
201 return brix_token_configure_registry(cf, xcf);
202 }
203
204 if (brix_token_require_single_issuer_fields(cf, xcf) != NGX_OK) {
205 return NGX_ERROR;
206 }
207
208 if (brix_token_load_and_register_jwks(cf, xcf) != NGX_OK) {
209 return NGX_ERROR;
210 }
211
212 brix_token_log_summary(cf, xcf);
213
214 return NGX_OK;
215}
