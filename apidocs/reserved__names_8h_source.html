1/*
2 * reserved_names.h — the single source of truth for INTERNAL on-disk artifact
3 * names that must NEVER be exposed to a client.
4 *
5 * WHAT: brix_is_internal_name(path) returns non-zero when the final path
6 * component is a service-internal metadata/staging artifact — a cache sidecar
7 * (.cinfo / .xrdcinfo / .meta), a stage-out crash-recovery marker (.commit),
8 * or an in-flight upload temp (…​.xrd-tmp.… / …​.xrdresume.…). These files can
9 * land inside a client-visible export namespace: the upload temps do so by
10 * default (when no separate stage dir is configured, they sit adjacent to the
11 * target file), and the cache sidecars do so if the cache/state tree is
12 * misconfigured under an export root.
13 *
14 * WHY: Every client-facing enumeration (root:// dirlist, WebDAV PROPFIND/SEARCH,
15 * S3 ListObjects) and every direct-access-by-name path (root:// open/stat/
16 * statx, WebDAV GET/HEAD, S3 GetObject) must treat these as invisible — a
17 * listing skips them and a direct request returns NotFound — so their
18 * contents, sizes, mtimes, and even their existence (which leaks residency /
19 * in-progress-upload activity) never reach a client. This is the one predicate
20 * all of those checkpoints call, so the reserved set is defined exactly once.
21 * The internal cache scanner (cstore) uses the same predicate to skip sidecars
22 * when enumerating cached objects.
23 *
24 * NOTE: these suffixes/infixes are RESERVED — a client-supplied name that matches
25 * is treated as internal (hidden / NotFound). The patterns are XRootD-specific
26 * and distinctive to keep collisions with genuine user data negligible.
27 *
28 * HOW: pure lexical test on the basename; no allocation, no I/O. Accepts either a
29 * full path or a bare name (the basename is located internally).
30 */

37brix_name_has_suffix(const char *name, size_t n, const char *suf, size_t suflen)
38{
39 return n >= suflen && memcmp(name + n - suflen, suf, suflen) == 0;

42static inline int
43brix_is_internal_name(const char *path)
44{
45 const char *name;
46 size_t n;
47
48 if (path == NULL) {
49 return 0;
50 }
51 name = strrchr(path, '/');
52 name = (name != NULL) ? name + 1 : path; /* basename */
53 n = strlen(name);
54 if (n == 0) {
55 return 0;
56 }
57
58 /* Cache metadata sidecars (block-present bitmap + origin metadata). */
59 if (brix_name_has_suffix(name, n, ".cinfo", 6)
60 || brix_name_has_suffix(name, n, ".xrdcinfo", 9)
61 || brix_name_has_suffix(name, n, ".meta", 5))
62 {
63 return 1;
64 }
65 /* Stage-out crash-recovery marker (records a pending commit's final path). */
66 if (brix_name_has_suffix(name, n, ".commit", 7)) {
67 return 1;
68 }
69 /* In-flight upload temps: <base>.xrd-tmp.<pid>.<rand> (atomic direct write)
70 * and <base>.xrdresume.<hash>.part (resumable Content-Range PUT). Matched by
71 * their distinctive infix so the trailing pid/hash/.part does not matter. */
72 if (strstr(name, ".xrd-tmp.") != NULL
73 || strstr(name, ".xrdresume.") != NULL)
74 {
75 return 1;
76 }
77
78 return 0;
79}
