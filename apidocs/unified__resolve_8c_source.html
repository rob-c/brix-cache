33static ngx_int_t
34brix_stat_type_cstr(const char *path)
35{
36 struct stat st;
37
38 if (stat(path, &st) != 0) {
39 if (errno == ENOENT) {
40 return BRIX_PATH_TYPE_NOT_FOUND;
41 }
42 return NGX_ERROR;
43 }
44
45 return (ngx_int_t) (st.st_mode & S_IFMT);
46}

48/*
49 * brix_finish_resolved — final confinement + stat + output for a canonical path.
50 *
51 * WHAT: Re-checks resolved_path against the export root, stats it for its type
52 * (tolerating absence under the missing-tail/parent opts), enforces
53 * require_directory, then copies the confined path into ctx->resolved and
54 * populates ctx->result.
55 *
56 * WHY: Every resolution path (existing target, missing tail, missing parents)
57 * converges here so containment and result publication happen in exactly
58 * one place. Taking ctx collapses the former eight parameters to three.
59 *
60 * HOW: Reads root_canon/log/depth from ctx; writes ctx->resolved/ctx->result.
61 * Ordering (within-root → stat → require_directory → copy → publish) is
62 * byte-for-byte unchanged from the pre-refactor implementation.
63 */
64brix_path_status_t
65brix_finish_resolved(const brix_resolve_ctx_t *ctx, const char *resolved_path,
66 brix_path_opts_t opts)
67{
68 ngx_int_t type;
69 brix_path_status_t rc;
70
71 if (!brix_path_within_root(ctx->root_canon, resolved_path)) {
72 brix_path_warn(ctx->log, "brix: path traversal attempt", resolved_path);
73 return BRIX_PATH_STATUS_INVALID;
74 }
75
76 type = brix_stat_type_cstr(resolved_path);
77 if (type == NGX_ERROR) {
78 if (opts.allow_missing_tail || opts.allow_missing_parents) {
79 type = BRIX_PATH_TYPE_NOT_FOUND;
80 } else {
81 return BRIX_PATH_STATUS_ERROR;
82 }
83 }
84
85 if (opts.require_directory && type != S_IFDIR) {
86 return BRIX_PATH_STATUS_INVALID;
87 }
88
89 rc = brix_copy_cstr(ctx->resolved, ctx->resolvsz, resolved_path);
90 if (rc != BRIX_PATH_STATUS_OK) {
91 return rc;
92 }
93
94 if (ctx->result != NULL) {
95 ctx->result->resolved.data = (u_char *) ctx->resolved;
96 ctx->result->resolved.len = strlen(ctx->resolved);
97 ctx->result->type = type;
98 ctx->result->depth = ctx->depth;
99 ctx->result->is_confined = 1;
100 }
101
102 return BRIX_PATH_STATUS_OK;
103}

105/*
106 * brix_resolve_missing_tail — resolve a write target whose FINAL component does
107 * not yet exist (create-a-new-leaf).
108 *
109 * WHAT: Splits candidate into parent dir + base name, canonicalises the parent
110 * with realpath(), confirms the parent is within the export root, rebuilds
111 * the absolute target (canonical parent + base), and finalises it under
112 * allow_missing_tail semantics via brix_finish_resolved.
113 *
114 * WHY: New-leaf writes cannot realpath() the whole path, so containment is
115 * enforced on the existing parent instead. Taking ctx keeps this at three
116 * parameters while preserving the write-path confinement contract.
117 *
118 * HOW: Copies into fixed PATH_MAX buffers with bounds checks; the lone-"/"
119 * parent and root-parent snprintf format selection are unchanged.
120 */
121brix_path_status_t
122brix_resolve_missing_tail(const brix_resolve_ctx_t *ctx, const char *candidate)
123{
124 char parent[PATH_MAX];
125 char parent_canon[PATH_MAX];
126 char rebuilt[PATH_MAX];
127 char base_buf[PATH_MAX];
128 char *slash;
129 const char *base;
130 size_t base_len;
131 int n;
132 brix_path_opts_t final_opts;
133
134 if (strlen(candidate) >= sizeof(parent)) {
135 return BRIX_PATH_STATUS_TOO_LONG;
136 }
137
138 ngx_memcpy(parent, candidate, strlen(candidate) + 1);
139 slash = strrchr(parent, '/');
140 if (slash == NULL) {
141 return BRIX_PATH_STATUS_INVALID;
142 }
143
144 base = slash + 1;
145 if (*base == '\0') {
146 return BRIX_PATH_STATUS_INVALID;
147 }
148
149 base_len = strlen(base);
150 if (base_len >= sizeof(base_buf)) {
151 return BRIX_PATH_STATUS_TOO_LONG;
152 }
153 ngx_memcpy(base_buf, base, base_len + 1);
154
155 if (slash == parent) {
156 slash[1] = '\0';
157 } else {
158 *slash = '\0';
159 }
160
161 if (realpath(parent, parent_canon) == NULL) {
162 if (errno == ENOENT) {
163 return BRIX_PATH_STATUS_NOT_FOUND;
164 }
165 return BRIX_PATH_STATUS_ERROR;
166 }
167
168 if (!brix_path_within_root(ctx->root_canon, parent_canon)) {
169 brix_path_warn(ctx->log, "brix: path traversal attempt in write",
170 parent_canon);
171 return BRIX_PATH_STATUS_INVALID;
172 }
173
174 n = snprintf(rebuilt, sizeof(rebuilt),
175 (strcmp(parent_canon, "/") == 0) ? "/%s" : "%s/%s",
176 parent_canon, base_buf);
177 if (n < 0 || (size_t) n >= sizeof(rebuilt)) {
178 return BRIX_PATH_STATUS_TOO_LONG;
179 }
180
181 final_opts = (brix_path_opts_t) { 0 };
182 final_opts.allow_missing_tail = 1;
183 final_opts.is_write_operation = 1;
184
185 return brix_finish_resolved(ctx, rebuilt, final_opts);
186}

188/*
189 * brix_walk_to_existing_ancestor — trim ancestor[] to its deepest EXISTING
190 * component and canonicalise it.
191 *
192 * WHAT: Repeatedly realpath()s ancestor; on ENOENT strips the trailing
193 * component and retries, canonicalising into ancestor_canon once a real
194 * directory is reached (or the "/" root).
195 *
196 * WHY: Missing-parents (recursive-mkdir/PUT) resolution must confine against
197 * the nearest existing ancestor. Extracting the walk loop keeps the caller
198 * under the complexity cap while preserving the exact climb order.
199 *
200 * HOW: Mutates ancestor in place (trailing-slash rule unchanged) and writes the
201 * canonical form into ancestor_canon; returns NOT_FOUND/ERROR on failure,
202 * OK once ancestor_canon holds an existing path.
203 */
204static brix_path_status_t
205brix_walk_to_existing_ancestor(char *ancestor, char *ancestor_canon)
206{
207 char *slash;
208
209 for ( ;; ) {
210 if (realpath(ancestor, ancestor_canon) != NULL) {
211 return BRIX_PATH_STATUS_OK;
212 }
213
214 if (errno != ENOENT) {
215 return BRIX_PATH_STATUS_ERROR;
216 }
217
218 slash = strrchr(ancestor, '/');
219 if (slash == NULL) {
220 return BRIX_PATH_STATUS_NOT_FOUND;
221 }
222
223 if (slash == ancestor) {
224 slash[1] = '\0';
225 } else {
226 *slash = '\0';
227 }
228
229 if (strcmp(ancestor, "/") == 0) {
230 if (realpath(ancestor, ancestor_canon) == NULL) {
231 return BRIX_PATH_STATUS_NOT_FOUND;
232 }
233 return BRIX_PATH_STATUS_OK;
234 }
235 }
236}

238/*
239 * brix_resolve_missing_parents — resolve a target whose intermediate parents do
240 * not yet exist (recursive mkdir / HTTP PUT-style missing suffix).
241 *
242 * WHAT: Walks candidate back to its deepest existing, canonicalised ancestor
243 * (brix_walk_to_existing_ancestor), confirms containment, re-attaches the
244 * missing suffix to that canonical ancestor, and finalises under
245 * allow_missing_parents semantics.
246 *
247 * WHY: With multiple absent components, confinement is enforced on the deepest
248 * real ancestor. Taking ctx keeps this at three parameters and preserves
249 * the suffix-splice and root-vs-nonroot format selection exactly.
250 *
251 * HOW: Uses fixed PATH_MAX buffers with bounds checks; suffix offset math and
252 * snprintf format branch are byte-for-byte the pre-refactor logic.
253 */
254brix_path_status_t
255brix_resolve_missing_parents(const brix_resolve_ctx_t *ctx,
256 const char *candidate)
257{
258 char ancestor[PATH_MAX];
259 char ancestor_canon[PATH_MAX];
260 char rebuilt[PATH_MAX];
261 const char *suffix;
262 size_t ancestor_len;
263 int n;
264 brix_path_status_t rc;
265 brix_path_opts_t final_opts;
266
267 if (strlen(candidate) >= sizeof(ancestor)) {
268 return BRIX_PATH_STATUS_TOO_LONG;
269 }
270
271 ngx_memcpy(ancestor, candidate, strlen(candidate) + 1);
272
273 rc = brix_walk_to_existing_ancestor(ancestor, ancestor_canon);
274 if (rc != BRIX_PATH_STATUS_OK) {
275 return rc;
276 }
277
278 if (!brix_path_within_root(ctx->root_canon, ancestor_canon)) {
279 brix_path_warn(ctx->log, "brix: path traversal attempt", ancestor_canon);
280 return BRIX_PATH_STATUS_INVALID;
281 }
282
283 ancestor_len = strlen(ancestor);
284 suffix = candidate + ancestor_len;
285 if (ancestor_len == 1 && ancestor[0] == '/') {
286 suffix = candidate + 1;
287 n = snprintf(rebuilt, sizeof(rebuilt),
288 (strcmp(ancestor_canon, "/") == 0) ? "/%s" : "%s/%s",
289 ancestor_canon, suffix);
290 } else {
291 n = snprintf(rebuilt, sizeof(rebuilt), "%s%s", ancestor_canon, suffix);
292 }
293
294 if (n < 0 || (size_t) n >= sizeof(rebuilt)) {
295 return BRIX_PATH_STATUS_TOO_LONG;
296 }
297
298 final_opts = (brix_path_opts_t) { 0 };
299 final_opts.allow_missing_parents = 1;
300
301 return brix_finish_resolved(ctx, rebuilt, final_opts);
302}
