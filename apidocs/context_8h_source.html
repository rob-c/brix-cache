12/*
13 * Per-connection context (brix_ctx_t).
14 *
15 * One instance per TCP connection, allocated from the nginx connection pool.
16 * The state machine runs on a single nginx worker thread; only one request
17 * is "in flight" at a time — XRootD multiplexing is handled via streamid
18 * matching on the client side, but the server serialises responses.
19 *
20 * Requires: state.h, file.h, tunables.h, protocol/protocol.h,
21 * token/token.h, metrics/metrics.h, and nginx/OpenSSL headers
22 * before inclusion.
23 */
24/*
25 * Output-queue slot (Phase 29 pipelining).
26 *
27 * Each slot owns the full send state for ONE in-flight response: the
28 * flat-buffer tail (small error/ok/status frames) OR the chain tail (read data,
29 * dirlist chunks), plus the reusable header/data/file chain structs that back a
30 * single-chunk read response. Bundling these per-slot (instead of one set of
31 * singletons on the connection) is what lets more than one response be in flight
32 * without the builders aliasing each other's chain memory.
33 *
34 * Only one of wbuf / wchain is active in a given slot at a time. wbuf points at
35 * the next unsent byte; wbuf_base is the owned allocation freed once the slot
36 * drains. wchain holds the remaining chain links; wchain_base (if set) backs a
37 * chain link and is freed after the chain fully drains. The read_fast_* structs
38 * are the pre-zeroed chain/buf/file objects reused for the common one-chunk read
39 * response so that path allocates nothing from the pool per request.
40 */
41typedef struct {
42 u_char *wbuf; /* pointer to the next byte to send */
43 size_t wbuf_len; /* total bytes in the buffer */
44 size_t wbuf_pos; /* bytes successfully sent so far */
45 u_char *wbuf_base; /* allocation base (freed when send completes) */
46
47 ngx_chain_t *wchain; /* remaining chain links to send */
48 off_t wchain_pending; /* unsent bytes in wchain, for cheap resume */
49 u_char *wchain_base; /* optional backing buffer, freed after drain */
50
51 ngx_chain_t read_fast_hdr_chain;
52 ngx_chain_t read_fast_body_chain;
53 ngx_buf_t read_fast_hdr_buf;
54 ngx_buf_t read_fast_body_buf;
55 ngx_file_t read_fast_file;
56
57 /*
58 * Per-slot response header storage (Phase 29 single-chunk; Phase 32 WS2
59 * widened to BRIX_SLOT_HDR_MAX so a MULTI-chunk sendfile read also keeps
60 * all its per-chunk ServerResponseHdrs here instead of the shared
61 * read_hdr_scratch). Owning the headers per-slot is what lets multiple
62 * (single- or multi-chunk) reads pipeline without their headers clobbering
63 * each other while a prior response is still draining.
64 */
65 u_char hdr_bytes[BRIX_SLOT_HDR_MAX];
66} brix_resp_slot_t;

68/*
69 * Phase 32 WS3 — concurrent-AIO read pipeline.
70 *
71 * A pool of per-in-flight-read buffers + thread tasks lets multiple
72 * memory-backed kXR_reads be outstanding at once (pipelined) instead of the
73 * single read_scratch / read_aio_task serial pair. Each entry is owned by one
74 * in-flight read from dispatch until the response it backs has fully drained
75 * from its out_ring slot, at which point brix_release_read_buffer() returns it
76 * to the pool. Buffers are raw ngx_alloc'd (Phase-31 discipline) and freed on
77 * disconnect. rd_inflight counts entries currently in use.
78 */
79typedef struct {
80 u_char *buf; /* raw-alloc'd data buffer (ngx_alloc/ngx_free) */
81 size_t size; /* allocated size of buf */
82 ngx_thread_task_t *task; /* per-entry thread task (kXR_read AIO) */
83 unsigned in_use:1; /* 1 while an in-flight read owns this entry */
84} brix_read_slot_t;
