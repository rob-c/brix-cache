28ssize_t
29sd_s3_get_meta(sd_s3_file *f, const char *name, const sd_s3_meta_buf *out,
30 char *errbuf, size_t errcap)
31{
32 char auth[SD_S3_AUTH_HDRS_CAP];
33 char hname[160];
34 brix_s3_resp_t resp;
35 int n;
36
37 if (f == NULL || name == NULL || out == NULL || out->buf == NULL
38 || out->cap == 0)
39 {
40 sd_s3_set_err(errbuf, errcap, "s3 get-meta: bad parameters");
41 return -1;
42 }
43 n = snprintf(hname, sizeof(hname), "x-amz-meta-%s", name);
44 if (n < 0 || (size_t) n >= sizeof(hname)) {
45 sd_s3_set_err(errbuf, errcap, "s3 get-meta: attribute name too long");
46 return -1;
47 }
48 if (sd_s3_sign(f, "HEAD", "", auth, sizeof(auth)) != 0) {
49 sd_s3_set_err(errbuf, errcap, "s3 HEAD: SigV4 sign failed on %s", f->key);
50 return -1;
51 }
52 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "HEAD",
53 f->key, auth, NULL, 0, f->timeout_ms, &resp,
54 errbuf, errcap) != 0)
55 {
56 return -1;
57 }
58 if (resp.status != 200) {
59 int rc = sd_s3_status_err(resp.status, "HEAD", f->key, errbuf, errcap);
60 f->transport->resp_free(&resp);
61 return rc; /* -1 */
62 }
63 if (f->transport->resp_header(&resp, hname, out->buf, out->cap) != 0) {
64 f->transport->resp_free(&resp);
65 out->buf[0] = '\0';
66 return 0; /* attribute absent */
67 }
68 f->transport->resp_free(&resp);
69 return (ssize_t) strlen(out->buf);
70}

72/* One "user.<name>" listxattr entry from a raw "x-amz-meta-<name>: v" header
73 * line. Appends to buf[*used..cap) when buf is non-NULL and always advances
74 * *need by the entry size; the caller compares need vs cap at the end. */
75static void
76sd_s3_list_meta_emit(const char *name, size_t nlen, char *buf, size_t cap,
77 size_t *need)
78{
79 size_t entry = sizeof("user.") - 1 + nlen + 1;
80
81 if (buf != NULL && *need + entry <= cap) {
82 memcpy(buf + *need, "user.", sizeof("user.") - 1);
83 memcpy(buf + *need + sizeof("user.") - 1, name, nlen);
84 buf[*need + entry - 1] = '\0';
85 }
86 *need += entry;
87}

89/* Scan the raw response header block for "x-amz-meta-<name>:" lines and emit
90 * each as a "user.<name>" listxattr entry (the advisory blob is skipped — it
91 * surfaces as POSIX attrs, not as a user xattr). Returns the total byte need;
92 * the caller compares it against cap. */
93static size_t
94sd_s3_list_meta_scan(const char *hdrs, char *buf, size_t cap)
95{
96 static const char pfx[] = "x-amz-meta-";
97 const size_t pfxlen = sizeof(pfx) - 1;
98 const char *p;
99 size_t need = 0;
100
101 for (p = hdrs; *p != '\0'; ) {
102 const char *eol = p + strcspn(p, "\r\n");
103 const char *colon;
104
105 if ((size_t) (eol - p) > pfxlen
106 && strncasecmp(p, pfx, pfxlen) == 0
107 && (colon = memchr(p, ':', (size_t) (eol - p))) != NULL
108 && colon > p + pfxlen)
109 {
110 const char *name = p + pfxlen;
111 size_t nlen = (size_t) (colon - name);
112
113 /* The advisory blob surfaces as POSIX attrs, not as a user xattr. */
114 if (nlen != sizeof(BRIX_META_ADVISORY_S3META) - 1
115 || strncasecmp(name, BRIX_META_ADVISORY_S3META, nlen) != 0)
116 {
117 sd_s3_list_meta_emit(name, nlen, buf, cap, &need);
118 }
119 }
120 p = eol + strspn(eol, "\r\n");
121 }
122 return need;
123}

125ssize_t
126sd_s3_list_meta(sd_s3_file *f, char *buf, size_t cap,
127 char *errbuf, size_t errcap)
128{
129 char auth[SD_S3_AUTH_HDRS_CAP];
130 brix_s3_resp_t resp;
131 const char *hdrs;
132 size_t need;
133
134 if (f == NULL) {
135 sd_s3_set_err(errbuf, errcap, "s3 list-meta: bad parameters");
136 errno = EINVAL;
137 return -1;
138 }
139 if (f->transport->resp_headers_raw == NULL) {
140 sd_s3_set_err(errbuf, errcap,
141 "s3 list-meta: transport cannot enumerate headers");
142 errno = ENOTSUP;
143 return -1;
144 }
145 if (sd_s3_sign(f, "HEAD", "", auth, sizeof(auth)) != 0) {
146 sd_s3_set_err(errbuf, errcap, "s3 HEAD: SigV4 sign failed on %s", f->key);
147 errno = EIO;
148 return -1;
149 }
150 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "HEAD",
151 f->key, auth, NULL, 0, f->timeout_ms, &resp,
152 errbuf, errcap) != 0)
153 {
154 errno = EIO;
155 return -1;
156 }
157 if (resp.status != 200) {
158 int rc = sd_s3_status_err(resp.status, "HEAD", f->key, errbuf, errcap);
159 f->transport->resp_free(&resp);
160 return rc; /* -1, errno mapped from the HTTP status */
161 }
162 hdrs = f->transport->resp_headers_raw(&resp);
163 if (hdrs == NULL) {
164 f->transport->resp_free(&resp);
165 sd_s3_set_err(errbuf, errcap, "s3 list-meta: no raw header block");
166 errno = ENOTSUP;
167 return -1;
168 }
169
170 need = sd_s3_list_meta_scan(hdrs, buf, cap);
171 f->transport->resp_free(&resp);
172
173 if (buf == NULL || cap == 0) {
174 return (ssize_t) need; /* listxattr(2) size probe */
175 }
176 if (need > cap) {
177 sd_s3_set_err(errbuf, errcap, "s3 list-meta: buffer too small");
178 errno = ERANGE;
179 return -1;
180 }
181 return (ssize_t) need;
182}

209/* One header for the extended SigV4 signer (name lowercase, value verbatim). */
210typedef struct { const char *name; const char *value; } sd_s3_sign_hdr_t;

212/* What to sign: the request identity a caller hands to sd_s3_sign_ext. */
213typedef struct {
214 const char *method; /* HTTP verb */
215 const char *canon_qs; /* canonical query string (NULL = "") */
216 const sd_s3_sign_hdr_t *extra; /* additional x-amz-* headers */
217 size_t n_extra; /* count of extra headers (max 32) */
218} sd_s3_sign_req_t;

221sd_s3_sign_hdr_cmp(const void *a, const void *b)
222{
223 return strcmp(((const sd_s3_sign_hdr_t *) a)->name,

227/*
228 * sd_s3_sign_ctx_t — file-local working state for the extended SigV4 signer.
229 * Carries the request identity (method / query string / extra headers), the
230 * fixed signed-header values (host, x-amz-date, payload hash) and every
231 * intermediate the pipeline stages hand to each other (sorted header set,
232 * canonical-request hash, credential scope, signed-header list, signature).
233 * One struct instead of 7+ loose parameters per helper.
234 */
235typedef struct {
236 const char *method; /* HTTP verb being signed */
237 const char *canon_qs; /* canonical query string ("" ok) */
238 const char *payload_hex; /* payload hash (UNSIGNED-PAYLOAD) */
239 const sd_s3_sign_hdr_t *extra; /* caller's additional headers */
240 size_t n_extra;
241 sd_s3_sign_hdr_t all[3 + 32]; /* fixed + extra, sorted for canon */
242 size_t n_all;
243 char host[300]; /* Host header value */
244 char amzdate[20]; /* x-amz-date (ISO8601 basic) */
245 char datestamp[12]; /* YYYYMMDD for the scope */
246 char enc_uri[2048]; /* URI-encoded object key */
247 char signed_list[2048]; /* "a;b;c" SignedHeaders */
248 char canon_hex[65]; /* SHA-256(canonical request) hex */
249 char scope[160]; /* date/region/s3/aws4_request */
250 char sighex[65]; /* final signature, hex */
251} sd_s3_sign_ctx_t;

253/*
254 * sd_s3_sign_prepare — stage 1: bind the request identity into the signing
255 * context. Formats host:port, stamps the current UTC time, URI-encodes the
256 * object key, then assembles the full signed-header set (fixed host /
257 * x-amz-content-sha256 / x-amz-date plus the caller's extras) sorted by name
258 * as SigV4 canonicalization requires. 0 / -1 (key too long to encode).
259 */
260static int
261sd_s3_sign_prepare(const sd_s3_file *f, sd_s3_sign_ctx_t *sx)
262{
263 size_t i;
264
265 brix_format_host_port(f->host, (uint16_t) f->port, sx->host,
266 sizeof(sx->host));
267 sd_s3_utc_now(sx->amzdate, sx->datestamp);
268 if (brix_http_urlencode((const unsigned char *) f->key, strlen(f->key),
269 sx->enc_uri, sizeof(sx->enc_uri), "/") < 0)
270 {
271 return -1;
272 }
273
274 sx->n_all = 0;
275 sx->all[sx->n_all].name = "host";
276 sx->all[sx->n_all++].value = sx->host;
277 sx->all[sx->n_all].name = "x-amz-content-sha256";
278 sx->all[sx->n_all++].value = sx->payload_hex;
279 sx->all[sx->n_all].name = "x-amz-date";
280 sx->all[sx->n_all++].value = sx->amzdate;
281 for (i = 0; i < sx->n_extra; i++) {
282 sx->all[sx->n_all++] = sx->extra[i];
283 }
284 qsort(sx->all, sx->n_all, sizeof(sx->all[0]), sd_s3_sign_hdr_cmp);
285 return 0;
286}

288/*
289 * sd_s3_canonical_request — stage 2: build the SigV4 canonical request
290 * (method \n uri \n qs \n sorted "name:value\n" headers \n signed-list \n
291 * payload-hash) and hash it into sx->canon_hex; also accumulates the
292 * ";"-joined SignedHeaders list the Authorization line reuses. The canonical
293 * buffer itself never leaves this frame — only its SHA-256 matters. 0 / -1
294 * on truncation.
295 */
296static int
297sd_s3_canonical_request(sd_s3_sign_ctx_t *sx)
298{
299 char canon[16384];
300 size_t i, sl = 0;
301 int off, cn;
302
303 off = snprintf(canon, sizeof(canon), "%s\n%s\n%s\n", sx->method,
304 sx->enc_uri, sx->canon_qs);
305 if (off < 0 || (size_t) off >= sizeof(canon)) {
306 return -1;
307 }
308 for (i = 0; i < sx->n_all; i++) {
309 cn = snprintf(canon + off, sizeof(canon) - (size_t) off, "%s:%s\n",
310 sx->all[i].name, sx->all[i].value);
311 if (cn < 0 || (size_t) off + (size_t) cn >= sizeof(canon)) {
312 return -1;
313 }
314 off += cn;
315 cn = snprintf(sx->signed_list + sl, sizeof(sx->signed_list) - sl,
316 "%s%s", (i ? ";" : ""), sx->all[i].name);
317 if (cn < 0 || sl + (size_t) cn >= sizeof(sx->signed_list)) {
318 return -1;
319 }
320 sl += (size_t) cn;
321 }
322 cn = snprintf(canon + off, sizeof(canon) - (size_t) off, "\n%s\n%s",
323 sx->signed_list, sx->payload_hex);
324 if (cn < 0 || (size_t) off + (size_t) cn >= sizeof(canon)) {
325 return -1;
326 }
327 sd_s3_sha256_hex(canon, strlen(canon), sx->canon_hex);
328 return 0;
329}

331/*
332 * sd_s3_string_to_sign — stage 3a: derive the credential scope
333 * (date/region/s3/aws4_request → sx->scope) and format the SigV4
334 * string-to-sign ("AWS4-HMAC-SHA256\n date \n scope \n canon-hash") into the
335 * caller's buffer. 0 / -1 on truncation.
336 */
337static int
338sd_s3_string_to_sign(const sd_s3_file *f, sd_s3_sign_ctx_t *sx,
339 char *sts, size_t stscap)
340{
341 int cn;
342
343 cn = snprintf(sx->scope, sizeof(sx->scope), "%s/%s/s3/aws4_request",
344 sx->datestamp, f->region);
345 if (cn < 0 || (size_t) cn >= sizeof(sx->scope)) {
346 return -1;
347 }
348 cn = snprintf(sts, stscap, "AWS4-HMAC-SHA256\n%s\n%s\n%s", sx->amzdate,
349 sx->scope, sx->canon_hex);
350 if (cn < 0 || (size_t) cn >= stscap) {
351 return -1;
352 }
353 return 0;
354}

356/*
357 * sd_s3_derive_key_and_sign — stage 3b: run the SigV4 key-derivation chain
358 * (secret → date → region → service → aws4_request) and HMAC the
359 * string-to-sign with the derived key, hex-encoding the signature into
360 * sx->sighex. 0 / -1 if any crypto primitive fails.
361 */
362static int
363sd_s3_derive_key_and_sign(const sd_s3_file *f, sd_s3_sign_ctx_t *sx,
364 const char *sts)
365{
366 uint8_t k4[32], sig[32];
367
368 if (!brix_sigv4_signing_key((const uint8_t *) f->sk, strlen(f->sk),
369 sx->datestamp, f->region, "s3", k4)
370 || !brix_hmac_sha256(k4, 32, (const uint8_t *) sts, strlen(sts), sig))
371 {
372 return -1;
373 }
374 brix_hex_encode(sig, 32, sx->sighex);
375 return 0;
376}

378/*
379 * sd_s3_emit_auth_header — stage 4: write the complete request header block
380 * the transport sends: x-amz-date, x-amz-content-sha256, every extra header
381 * verbatim (in caller order — the sorted set was for canonicalization only),
382 * then the Authorization line carrying credential scope, SignedHeaders and
383 * the signature. 0 / -1 on truncation.
384 */
385static int
386sd_s3_emit_auth_header(const sd_s3_file *f, const sd_s3_sign_ctx_t *sx,
387 char *hdrs, size_t hdrsz)
388{
389 size_t i;
390 int off, cn;
391
392 off = snprintf(hdrs, hdrsz,
393 "x-amz-date: %s\r\nx-amz-content-sha256: %s\r\n",
394 sx->amzdate, sx->payload_hex);
395 if (off < 0 || (size_t) off >= hdrsz) {
396 return -1;
397 }
398 for (i = 0; i < sx->n_extra; i++) {
399 cn = snprintf(hdrs + off, hdrsz - (size_t) off, "%s: %s\r\n",
400 sx->extra[i].name, sx->extra[i].value);
401 if (cn < 0 || (size_t) off + (size_t) cn >= hdrsz) {
402 return -1;
403 }
404 off += cn;
405 }
406 cn = snprintf(hdrs + off, hdrsz - (size_t) off,
407 "Authorization: AWS4-HMAC-SHA256 Credential=%s/%s, "
408 "SignedHeaders=%s, Signature=%s\r\n",
409 f->ak, sx->scope, sx->signed_list, sx->sighex);
410 if (cn < 0 || (size_t) off + (size_t) cn >= hdrsz) {
411 return -1;
412 }
413 return 0;
414}

416/*
417 * sd_s3_sign_ext — SigV4 over an arbitrary set of additional x-amz-* headers (on
418 * top of the fixed host / x-amz-content-sha256 / x-amz-date). Used by set-meta,
419 * whose copy-onto-self carries x-amz-copy-source, x-amz-metadata-directive and
420 * the x-amz-meta-* pairs — all of which AWS requires in the signed header set.
421 * Emits the complete request header block to send (the extra headers + the
422 * Authorization line). 0 / -1. Reuses every kernel sd_s3_sign uses. Pure
423 * pipeline: prepare → canonical request → string-to-sign → sign → emit.
424 */
425static int
426sd_s3_sign_ext(const sd_s3_file *f, const sd_s3_sign_req_t *req,
427 char *hdrs, size_t hdrsz)
428{
429 sd_s3_sign_ctx_t sx;
430 char sts[640];
431
432 if (req->n_extra > 32) {
433 return -1;
434 }
435 sx.method = req->method;
436 sx.canon_qs = (req->canon_qs != NULL) ? req->canon_qs : "";
437 sx.payload_hex = "UNSIGNED-PAYLOAD";
438 sx.extra = req->extra;
439 sx.n_extra = req->n_extra;
440
441 if (sd_s3_sign_prepare(f, &sx) != 0
442 || sd_s3_canonical_request(&sx) != 0
443 || sd_s3_string_to_sign(f, &sx, sts, sizeof(sts)) != 0
444 || sd_s3_derive_key_and_sign(f, &sx, sts) != 0)
445 {
446 return -1;
447 }
448 return sd_s3_emit_auth_header(f, &sx, hdrs, hdrsz);
449}

451static int
452sd_s3_set_meta_f(sd_s3_file *f, const sd_s3_meta_kv *kv, size_t nkv,
453 char *errbuf, size_t errcap)
454{
455 char auth[SD_S3_AUTH_HDRS_CAP];
456 sd_s3_sign_hdr_t extra[2 + 32];
457 sd_s3_sign_req_t req;
458 char names[32][160];
459 char lname[160];
460 size_t n_extra = 0, i, j, nl;
461 brix_s3_resp_t resp;
462
463 if (nkv > 32) {
464 sd_s3_set_err(errbuf, errcap, "s3 set-meta: too many attributes");
465 return -1;
466 }
467 /* Self-copy with REPLACE: update only metadata, never re-upload the bytes. */
468 extra[n_extra].name = "x-amz-copy-source";
469 extra[n_extra++].value = f->key;
470 extra[n_extra].name = "x-amz-metadata-directive";
471 extra[n_extra++].value = "REPLACE";
472
473 for (i = 0; i < nkv; i++) {
474 if (kv[i].name == NULL || kv[i].value == NULL) {
475 sd_s3_set_err(errbuf, errcap, "s3 set-meta: null attribute");
476 return -1;
477 }
478 nl = strlen(kv[i].name);
479 if (nl >= sizeof(lname)) {
480 sd_s3_set_err(errbuf, errcap, "s3 set-meta: attribute name too long");
481 return -1;
482 }
483 for (j = 0; j < nl; j++) { /* AWS lowercases user-metadata names */
484 char c = kv[i].name[j];
485 lname[j] = (c >= 'A' && c <= 'Z') ? (char) (c - 'A' + 'a') : c;
486 }
487 lname[nl] = '\0';
488 if (snprintf(names[i], sizeof(names[i]), "x-amz-meta-%s", lname)
489 >= (int) sizeof(names[i]))
490 {
491 sd_s3_set_err(errbuf, errcap, "s3 set-meta: attribute name too long");
492 return -1;
493 }
494 extra[n_extra].name = names[i];
495 extra[n_extra++].value = kv[i].value;
496 }
497
498 req.method = "PUT";
499 req.canon_qs = "";
500 req.extra = extra;
501 req.n_extra = n_extra;
502 if (sd_s3_sign_ext(f, &req, auth, sizeof(auth)) != 0) {
503 sd_s3_set_err(errbuf, errcap, "s3 set-meta: SigV4 sign failed on %s",
504 f->key);
505 return -1;
506 }
507 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "PUT",
508 f->key, auth, NULL, 0, f->timeout_ms, &resp,
509 errbuf, errcap) != 0)
510 {
511 return -1;
512 }
513 if (resp.status != 200) {
514 int rc = sd_s3_status_err(resp.status, "CopyObject(REPLACE)", f->key,
515 errbuf, errcap);
516 f->transport->resp_free(&resp);
517 return rc;
518 }
519 f->transport->resp_free(&resp);
520 return 0;
521}
