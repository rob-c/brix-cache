1/*
2 * sd_s3_meta.c — S3 object metadata (x-amz-meta-*) get/set + POSIX unixattr
3 * advisory encoding. Split out of sd_s3.c: the metadata path (HEAD to read
4 * user metadata, and metadata-copy PUT to set it) with its own extended SigV4
5 * signer (sd_s3_sign_ext, which must sign the x-amz-meta-* headers). Uses the
6 * shared sd_s3_file layout + signing primitives via sd_s3_internal.h; the public
7 * entry points (sd_s3_get_meta/set_meta/get_unixattr/set_unixattr) are declared
8 * in sd_s3.h.

28ssize_t
29sd_s3_get_meta(sd_s3_file *f, const char *name, char *buf, size_t cap,
30 char *errbuf, size_t errcap)
31{
32 char auth[SD_S3_AUTH_HDRS_CAP];
33 char hname[160];
34 brix_s3_resp_t resp;
35 int n;
36
37 if (f == NULL || name == NULL || buf == NULL || cap == 0) {
38 sd_s3_set_err(errbuf, errcap, "s3 get-meta: bad parameters");
39 return -1;
40 }
41 n = snprintf(hname, sizeof(hname), "x-amz-meta-%s", name);
42 if (n < 0 || (size_t) n >= sizeof(hname)) {
43 sd_s3_set_err(errbuf, errcap, "s3 get-meta: attribute name too long");
44 return -1;
45 }
46 if (sd_s3_sign(f, "HEAD", "", auth, sizeof(auth)) != 0) {
47 sd_s3_set_err(errbuf, errcap, "s3 HEAD: SigV4 sign failed on %s", f->key);
48 return -1;
49 }
50 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "HEAD",
51 f->key, auth, NULL, 0, f->timeout_ms, &resp,
52 errbuf, errcap) != 0)
53 {
54 return -1;
55 }
56 if (resp.status != 200) {
57 int rc = sd_s3_status_err(resp.status, "HEAD", f->key, errbuf, errcap);
58 f->transport->resp_free(&resp);
59 return rc; /* -1 */
60 }
61 if (f->transport->resp_header(&resp, hname, buf, cap) != 0) {
62 f->transport->resp_free(&resp);
63 buf[0] = '\0';
64 return 0; /* attribute absent */
65 }
66 f->transport->resp_free(&resp);
67 return (ssize_t) strlen(buf);
68}

70int
71sd_s3_get_unixattr(sd_s3_file *f, brix_meta_advisory_t *out,
72 char *errbuf, size_t errcap)
73{
74 char blob[512];
75 ssize_t n;
76
77 if (out == NULL) {
78 return -1;
79 }
80 n = sd_s3_get_meta(f, BRIX_META_ADVISORY_S3META, blob, sizeof(blob),
81 errbuf, errcap);
82 if (n < 0) {
83 return -1;
84 }
85 if (n == 0) {
86 return 0; /* object carries no advisory metadata */
87 }
88 if (brix_meta_advisory_decode(blob, (size_t) n, out) != 0) {
89 sd_s3_set_err(errbuf, errcap, "s3 get-unixattr: blob decode failed");
90 return -1;
91 }
92 return 1;
93}

105/*
106 * sd_s3_sign_ext — SigV4 over an arbitrary set of additional x-amz-* headers (on
107 * top of the fixed host / x-amz-content-sha256 / x-amz-date). Used by set-meta,
108 * whose copy-onto-self carries x-amz-copy-source, x-amz-metadata-directive and
109 * the x-amz-meta-* pairs — all of which AWS requires in the signed header set.
110 * Emits the complete request header block to send (the extra headers + the
111 * Authorization line). 0 / -1. Reuses every kernel sd_s3_sign uses.
112 */
113static int
114sd_s3_sign_ext(const sd_s3_file *f, const char *method, const char *canon_qs,
115 const sd_s3_sign_hdr_t *extra, size_t n_extra,
116 char *hdrs, size_t hdrsz)
117{
118 const char *payload_hex = "UNSIGNED-PAYLOAD";
119 char host[300], amzdate[20], datestamp[12];
120 char canon_hex[65], sighex[65];
121 char scope[160], sts[640], enc_uri[2048];
122 char canon[16384], signed_list[2048];
123 sd_s3_sign_hdr_t all[3 + 32];
124 uint8_t k4[32], sig[32];
125 size_t n_all = 0, i, sl = 0;
126 int off, cn;
127
128 if (n_extra > 32) {
129 return -1;
130 }
131 if (canon_qs == NULL) {
132 canon_qs = "";
133 }
134 brix_format_host_port(f->host, (uint16_t) f->port, host, sizeof(host));
135 sd_s3_utc_now(amzdate, datestamp);
136 if (brix_http_urlencode((const unsigned char *) f->key, strlen(f->key),
137 enc_uri, sizeof(enc_uri), "/") < 0)
138 {
139 return -1;
140 }
141
142 all[n_all].name = "host"; all[n_all++].value = host;
143 all[n_all].name = "x-amz-content-sha256"; all[n_all++].value = payload_hex;
144 all[n_all].name = "x-amz-date"; all[n_all++].value = amzdate;
145 for (i = 0; i < n_extra; i++) {
146 all[n_all++] = extra[i];
147 }
148 qsort(all, n_all, sizeof(all[0]), sd_s3_sign_hdr_cmp);
149
150 off = snprintf(canon, sizeof(canon), "%s\n%s\n%s\n", method, enc_uri,
151 canon_qs);
152 if (off < 0 || (size_t) off >= sizeof(canon)) {
153 return -1;
154 }
155 for (i = 0; i < n_all; i++) {
156 cn = snprintf(canon + off, sizeof(canon) - (size_t) off, "%s:%s\n",
157 all[i].name, all[i].value);
158 if (cn < 0 || (size_t) (off + cn) >= sizeof(canon)) {
159 return -1;
160 }
161 off += cn;
162 cn = snprintf(signed_list + sl, sizeof(signed_list) - sl, "%s%s",
163 (i ? ";" : ""), all[i].name);
164 if (cn < 0 || sl + (size_t) cn >= sizeof(signed_list)) {
165 return -1;
166 }
167 sl += (size_t) cn;
168 }
169 cn = snprintf(canon + off, sizeof(canon) - (size_t) off, "\n%s\n%s",
170 signed_list, payload_hex);
171 if (cn < 0 || (size_t) (off + cn) >= sizeof(canon)) {
172 return -1;
173 }
174 sd_s3_sha256_hex(canon, strlen(canon), canon_hex);
175
176 cn = snprintf(scope, sizeof(scope), "%s/%s/s3/aws4_request", datestamp,
177 f->region);
178 if (cn < 0 || (size_t) cn >= sizeof(scope)) {
179 return -1;
180 }
181 cn = snprintf(sts, sizeof(sts), "AWS4-HMAC-SHA256\n%s\n%s\n%s", amzdate,
182 scope, canon_hex);
183 if (cn < 0 || (size_t) cn >= sizeof(sts)) {
184 return -1;
185 }
186 if (!brix_sigv4_signing_key((const uint8_t *) f->sk, strlen(f->sk),
187 datestamp, f->region, "s3", k4)
188 || !brix_hmac_sha256(k4, 32, (const uint8_t *) sts, strlen(sts), sig))
189 {
190 return -1;
191 }
192 brix_hex_encode(sig, 32, sighex);
193
194 off = snprintf(hdrs, hdrsz,
195 "x-amz-date: %s\r\nx-amz-content-sha256: %s\r\n",
196 amzdate, payload_hex);
197 if (off < 0 || (size_t) off >= hdrsz) {
198 return -1;
199 }
200 for (i = 0; i < n_extra; i++) {
201 cn = snprintf(hdrs + off, hdrsz - (size_t) off, "%s: %s\r\n",
202 extra[i].name, extra[i].value);
203 if (cn < 0 || (size_t) (off + cn) >= hdrsz) {
204 return -1;
205 }
206 off += cn;
207 }
208 cn = snprintf(hdrs + off, hdrsz - (size_t) off,
209 "Authorization: AWS4-HMAC-SHA256 Credential=%s/%s, "
210 "SignedHeaders=%s, Signature=%s\r\n",
211 f->ak, scope, signed_list, sighex);
212 if (cn < 0 || (size_t) (off + cn) >= hdrsz) {
213 return -1;
214 }
215 return 0;
216}

218static int
219sd_s3_set_meta_f(sd_s3_file *f, const sd_s3_meta_kv *kv, size_t nkv,
220 char *errbuf, size_t errcap)
221{
222 char auth[SD_S3_AUTH_HDRS_CAP];
223 sd_s3_sign_hdr_t extra[2 + 32];
224 char names[32][160];
225 char lname[160];
226 size_t n_extra = 0, i, j, nl;
227 brix_s3_resp_t resp;
228
229 if (nkv > 32) {
230 sd_s3_set_err(errbuf, errcap, "s3 set-meta: too many attributes");
231 return -1;
232 }
233 /* Self-copy with REPLACE: update only metadata, never re-upload the bytes. */
234 extra[n_extra].name = "x-amz-copy-source";
235 extra[n_extra++].value = f->key;
236 extra[n_extra].name = "x-amz-metadata-directive";
237 extra[n_extra++].value = "REPLACE";
238
239 for (i = 0; i < nkv; i++) {
240 if (kv[i].name == NULL || kv[i].value == NULL) {
241 sd_s3_set_err(errbuf, errcap, "s3 set-meta: null attribute");
242 return -1;
243 }
244 nl = strlen(kv[i].name);
245 if (nl >= sizeof(lname)) {
246 sd_s3_set_err(errbuf, errcap, "s3 set-meta: attribute name too long");
247 return -1;
248 }
249 for (j = 0; j < nl; j++) { /* AWS lowercases user-metadata names */
250 char c = kv[i].name[j];
251 lname[j] = (c >= 'A' && c <= 'Z') ? (char) (c - 'A' + 'a') : c;
252 }
253 lname[nl] = '\0';
254 if (snprintf(names[i], sizeof(names[i]), "x-amz-meta-%s", lname)
255 >= (int) sizeof(names[i]))
256 {
257 sd_s3_set_err(errbuf, errcap, "s3 set-meta: attribute name too long");
258 return -1;
259 }
260 extra[n_extra].name = names[i];
261 extra[n_extra++].value = kv[i].value;
262 }
263
264 if (sd_s3_sign_ext(f, "PUT", "", extra, n_extra, auth, sizeof(auth)) != 0) {
265 sd_s3_set_err(errbuf, errcap, "s3 set-meta: SigV4 sign failed on %s",
266 f->key);
267 return -1;
268 }
269 if (f->transport->request(f->tctx, f->host, f->port, f->tls, "PUT",
270 f->key, auth, NULL, 0, f->timeout_ms, &resp,
271 errbuf, errcap) != 0)
272 {
273 return -1;
274 }
275 if (resp.status != 200) {
276 int rc = sd_s3_status_err(resp.status, "CopyObject(REPLACE)", f->key,
277 errbuf, errcap);
278 f->transport->resp_free(&resp);
279 return rc;
280 }
281 f->transport->resp_free(&resp);
282 return 0;
283}
