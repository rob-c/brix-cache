1/*
2 * cstr.h — bounded ngx_str_t → NUL-terminated C-string conversion.
3 *
4 * WHAT: brix_cbuf_copy() copies len bytes into a caller buffer and
5 * NUL-terminates, refusing (returns NULL, buffer untouched) when the
6 * result would not fit; brix_str_cbuf() is the ngx_str_t wrapper;
7 * brix_pstrdup_z() pool-allocates len+1 bytes and NUL-terminates.
8 *
9 * WHY: ~40 call sites hand-rolled the same bounds-check + ngx_memcpy +
10 * `buf[len] = '\0'` triple — each one an independently auditable
11 * off-by-one risk. One reviewed helper replaces per-site review.
12 *
13 * HOW: The core is ngx-free (define BRIX_CSTR_NO_NGX to drop the ngx
14 * wrappers) so tests/cstr_unittest.c compiles standalone with plain
15 * gcc, the same pattern as core/compat/af_policy.h.
16 */

24/* Copy len bytes into buf and NUL-terminate. Returns buf, or NULL (buffer
25 * untouched) when len+1 would exceed bufsize or buf is NULL. */
26static inline const char *
27brix_cbuf_copy(char *buf, size_t bufsize, const void *data, size_t len)
28{
29 if (buf == NULL || bufsize == 0 || len >= bufsize) {
30 return NULL;
31 }
32 if (len > 0) {
33 memcpy(buf, data, len);
34 }
35 buf[len] = '\0';
36 return buf;
37}

48/* Pool-duplicate an ngx_str_t as a NUL-terminated C string (len+1 bytes).
49 * Returns NULL on allocation failure. */
50static inline char *
51brix_pstrdup_z(ngx_pool_t *pool, const ngx_str_t *s)
52{
53 char *p;
54
55 p = ngx_pnalloc(pool, s->len + 1);
56 if (p == NULL) {
57 return NULL;
58 }
59 ngx_memcpy(p, s->data, s->len);
60 p[s->len] = '\0';
61 return p;
62}
