1/*
2 * idmap.c — identity -> local UNIX account resolution (phase 40).
3 *
4 * WHAT: brix_idmap_resolve() maps an authenticated principal (GSI DN / token
5 * sub / SSS user / krb5 localname) to {uid, primary gid, supplementary gids}
6 * via (1) an optional grid-mapfile (DN -> local username), (2) a direct
7 * getpwnam() of the principal, then (3) a squash-to-default or deny policy.
8 * A per-process TTL cache keeps the NSS lookups off the hot path.
9 *
10 * WHY: This is the pure, self-contained foundation of per-request impersonation
11 * (the privileged broker consumes it). It is the only place numeric uid/gid
12 * are derived; it enforces the safety floor (never resolve to uid 0 / below
13 * min_uid), so the broker can never be asked to act as a system account.
14 *
15 * HOW: Modelled on src/acc/groups.c (getpwnam + getgrouplist + a fixed-slot,
16 * strcmp, TTL cache). The grid-mapfile is parsed once at init into a small
17 * array of (quoted-DN, username) pairs; resolution checks it first, then falls
18 * back to a direct getpwnam. No goto; pure helpers, side effects at the edges.
19 */
20

86/*
87 * Parse one grid-mapfile line of the classic form:
88 * "<quoted DN>" <local-username>
89 * Comments (#) and blank lines are skipped. Returns 1 and fills the two out
90 * params (malloc'd) on a usable mapping, 0 to skip the line, -1 on OOM.
91 */
92static int
93idmap_gridmap_parse_line(const char *line, char **dn_out, char **user_out)
94{
95 const char *p = line;
96 const char *dn_start, *dn_end, *u_start, *u_end;
97
98 while (*p == ' ' || *p == '\t') { p++; }
99 if (*p == '#' || *p == '\0' || *p == '\n' || *p != '"') {
100 return 0; /* comment, blank, or unquoted -> skip */
101 }
102 dn_start = ++p; /* after the opening quote */
103 while (*p != '"' && *p != '\0' && *p != '\n') { p++; }
104 if (*p != '"') {
105 return 0; /* unterminated quote -> skip */
106 }
107 dn_end = p++; /* at closing quote */
108
109 while (*p == ' ' || *p == '\t') { p++; }
110 u_start = p;
111 while (*p != '\0' && *p != '\n' && *p != ' ' && *p != '\t') { p++; }
112 u_end = p;
113 if (u_end == u_start || (size_t) (dn_end - dn_start) >= IDMAP_PRINC_MAX) {
114 return 0; /* no username, or DN too long -> skip */
115 }
116
117 *dn_out = strndup(dn_start, (size_t) (dn_end - dn_start));
118 *user_out = strndup(u_start, (size_t) (u_end - u_start));
119 if (*dn_out == NULL || *user_out == NULL) {
120 free(*dn_out);
121 free(*user_out);
122 return -1;
123 }
124 return 1;
125}

127/* Load the grid-mapfile at `path` into idmap_gridmap[]. Returns NGX_OK (incl.
128 * the "no path" case) or NGX_ERROR on IO/OOM. */
129static ngx_int_t
130idmap_gridmap_load(const char *path, ngx_log_t *log)
131{
132 FILE *fp;
133 char line[1024];
134 size_t cap = 0;
135
136 idmap_gridmap_free();
137 if (path == NULL || path[0] == '\0') {
138 return NGX_OK; /* no mapfile configured */
139 }
140
141 fp = fopen(path, "re");
142 if (fp == NULL) {
143 if (log != NULL) {
144 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno,
145 "impersonate: cannot open grid-mapfile \"%s\"", path);
146 }
147 return NGX_ERROR;
148 }
149
150 while (fgets(line, sizeof(line), fp) != NULL) {
151 char *dn = NULL, *user = NULL;
152 int r = idmap_gridmap_parse_line(line, &dn, &user);
153
154 if (r < 0) {
155 fclose(fp);
156 idmap_gridmap_free();
157 return NGX_ERROR;
158 }
159 if (r == 0) {
160 continue;
161 }
162 if (idmap_gridmap_n == cap) {
163 size_t ncap = cap ? cap * 2 : 16;
164 idmap_gridmap_entry_t *ne =
165 realloc(idmap_gridmap, ncap * sizeof(*ne));
166 if (ne == NULL) {
167 free(dn); free(user);
168 fclose(fp);
169 idmap_gridmap_free();
170 return NGX_ERROR;
171 }
172 idmap_gridmap = ne;
173 cap = ncap;
174 }
175 idmap_gridmap[idmap_gridmap_n].dn = dn;
176 idmap_gridmap[idmap_gridmap_n].user = user;
177 idmap_gridmap_n++;
178 }
179 fclose(fp);
180
181 if (log != NULL) {
182 ngx_log_error(NGX_LOG_NOTICE, log, 0,
183 "impersonate: loaded %uz grid-mapfile entries from \"%s\"",
184 idmap_gridmap_n, path);
185 }
186 return NGX_OK;
187}

204/* True if gid is reserved (0 / below the effective floor) or on the forbidden
205 * privileged-group list. Used to scan the FULL group set on overflow. */
206static int
207idmap_gid_reserved_or_forbidden(gid_t gid)
208{
209 int k;
210
211 if (gid == 0 || gid < (gid_t) idmap_min_uid) {
212 return 1;
213 }
214 for (k = 0; k < idmap_forbidden_gids_n; k++) {
215 if (gid == idmap_forbidden_gids[k]) {
216 return 1;
217 }
218 }
219 return 0;
220}

222/*
223 * Resolve a local username to {uid, gid, supplementary gids}. Returns 0 and
224 * fills *out on success, -1 if the user does not exist, or -2 if the user is a
225 * member of a RESERVED or FORBIDDEN group anywhere in their (possibly >32) group
226 * set — caller must treat both negatives as deny. Mirrors the getpwnam/
227 * getgrouplist usage in src/acc/groups.c, but keeps the numeric ids and, crucially,
228 * fails CLOSED when the group set overflows the 32-slot buffer.
229 */
230static int
231idmap_resolve_user(const char *user, brix_idmap_creds_t *out)
232{
233 struct passwd *pw;
234 gid_t gids[BRIX_IDMAP_MAXGROUPS];
235 int ng = BRIX_IDMAP_MAXGROUPS;
236 int i;
237
238 pw = getpwnam(user);
239 if (pw == NULL) {
240 return -1;
241 }
242
243 out->uid = pw->pw_uid;
244 out->gid = pw->pw_gid;
245
246 if (idmap_primary_only) {
247 out->groups[0] = pw->pw_gid;
248 out->ngroups = 1;
249 return 0;
250 }
251
252 if (getgrouplist(user, pw->pw_gid, gids, &ng) < 0) {
253 /*
254 * OVERFLOW: the user belongs to MORE than BRIX_IDMAP_MAXGROUPS groups.
255 * `ng` now holds the TRUE total; `gids` holds only a 32-entry PREFIX in
256 * /etc/group scan order (NOT sorted). A reserved or forbidden group could
257 * sit past the cap, so checking only the prefix would fail OPEN — a
258 * dropped group is a MISSED deny. Re-resolve the FULL set and REFUSE if
259 * ANY group is reserved (0 / < floor) or forbidden. Common on multi-VO
260 * HEP hosts where users exceed 32 group memberships.
261 */
262 int total = ng;
263 gid_t *full = malloc((size_t) total * sizeof(gid_t));
264 int j;
265
266 if (full == NULL) {
267 return -1; /* OOM -> deny (fail closed) */
268 }
269 if (getgrouplist(user, pw->pw_gid, full, &total) < 0) {
270 free(full);
271 return -1; /* grew again / still failing -> deny */
272 }
273 for (j = 0; j < total; j++) {
274 if (idmap_gid_reserved_or_forbidden(full[j])) {
275 free(full);
276 return -2; /* reserved/forbidden group -> DENY */
277 }
278 }
279 /* Whole set is clean; keep a capped subset for the impersonation setgroups
280 * (a subset only ever GRANTS LESS, so it is fail-safe for the op). */
281 for (i = 0; i < total && i < BRIX_IDMAP_MAXGROUPS; i++) {
282 out->groups[i] = full[i];
283 }
284 out->ngroups = i;
285 free(full);
286 return 0;
287 }
288 if (ng < 1) {
289 gids[0] = pw->pw_gid;
290 ng = 1;
291 }
292 for (i = 0; i < ng && i < BRIX_IDMAP_MAXGROUPS; i++) {
293 out->groups[i] = gids[i];
294 }
295 out->ngroups = i;
296 return 0;
297}

299/*
300 * The single authoritative reserved-id test (declared in impersonate.h, used by
301 * BOTH this mapping layer and the broker's syscall-edge guard). A credential set
302 * is "privileged" — and must be REFUSED — if its primary uid, primary gid, or ANY
303 * supplementary gid is 0 or strictly below `floor`, or if ngroups is out of
304 * range. Pure: no syscalls, no globals.
305 */
306int
307brix_imp_creds_privileged(const brix_idmap_creds_t *cr, uid_t floor,
308 uint32_t *out_id, char *out_kind)
309{
310 int i;
311
312 if (cr == NULL) {
313 if (out_id != NULL) { *out_id = 0; }
314 if (out_kind != NULL) { *out_kind = 'n'; }
315 return 1;
316 }
317 if (cr->ngroups < 0 || cr->ngroups > BRIX_IDMAP_MAXGROUPS) {
318 if (out_id != NULL) { *out_id = (uint32_t) cr->ngroups; }
319 if (out_kind != NULL) { *out_kind = 'n'; }
320 return 1;
321 }
322 if (cr->uid == 0 || cr->uid < floor) {
323 if (out_id != NULL) { *out_id = (uint32_t) cr->uid; }
324 if (out_kind != NULL) { *out_kind = 'u'; }
325 return 1;
326 }
327 if (cr->gid == 0 || cr->gid < (gid_t) floor) {
328 if (out_id != NULL) { *out_id = (uint32_t) cr->gid; }
329 if (out_kind != NULL) { *out_kind = 'g'; }
330 return 1;
331 }
332 for (i = 0; i < cr->ngroups; i++) {
333 if (cr->groups[i] == 0 || cr->groups[i] < (gid_t) floor) {
334 if (out_id != NULL) { *out_id = (uint32_t) cr->groups[i]; }
335 if (out_kind != NULL) { *out_kind = 's'; }
336 return 1;
337 }
338 }
339 return 0;
340}

375/*
376 * Parse a comma/whitespace-separated NAME list and resolve each name to a numeric
377 * id via the supplied resolver, adding it to the matching forbidden set. `is_user`
378 * selects getpwnam (uid) vs getgrnam (gid). Names that do not resolve on this host
379 * are skipped (a privileged group absent here cannot be a target anyway). `list`
380 * is "" => use the built-in default.
381 */
382static void
383idmap_forbid_load(const char *list, const char *def, int is_user, ngx_log_t *log)
384{
385 const char *p = (list != NULL && list[0] != '\0') ? list : def;
386 char name[256];
387
388 while (*p != '\0') {
389 size_t n = 0;
390
391 while (*p == ',' || *p == ' ' || *p == '\t') { p++; }
392 while (*p != '\0' && *p != ',' && *p != ' ' && *p != '\t'
393 && n < sizeof(name) - 1)
394 {
395 name[n++] = *p++;
396 }
397 name[n] = '\0';
398 while (*p != '\0' && *p != ',' && *p != ' ' && *p != '\t') { p++; }
399 if (n == 0) {
400 continue;
401 }
402 if (is_user) {
403 struct passwd *pw = getpwnam(name);
404 if (pw != NULL) {
405 idmap_forbid_uid(pw->pw_uid);
406 }
407 } else {
408 struct group *gr = getgrnam(name);
409 if (gr != NULL) {
410 idmap_forbid_gid(gr->gr_gid);
411 }
412 }
413 }
414 (void) log;
415}

431/* True if any of the creds' gids (primary or supplementary) is a forbidden
432 * (privileged) group — even when its numeric gid is >= the floor. */
433static int
434idmap_creds_have_forbidden_group(const brix_idmap_creds_t *cr)
435{
436 int i, j;
437
438 for (j = 0; j < idmap_forbidden_gids_n; j++) {
439 if (cr->gid == idmap_forbidden_gids[j]) {
440 return 1;
441 }
442 for (i = 0; i < cr->ngroups; i++) {
443 if (cr->groups[i] == idmap_forbidden_gids[j]) {
444 return 1;
445 }
446 }
447 }
448 return 0;
449}

451/*
452 * Accept these creds as an impersonation target only if NONE of:
453 * - any reserved/floor id (uid/primary gid/supplementary gid < floor or 0),
454 * - the target uid is an explicitly forbidden account (service/system),
455 * - the user is a member of a forbidden privileged group (sudo/wheel/...).
456 */
457static int
458idmap_creds_allowed(const brix_idmap_creds_t *cr)
459{
460 return !brix_imp_creds_privileged(cr, idmap_min_uid, NULL, NULL)
461 && !idmap_uid_forbidden(cr->uid)
462 && !idmap_creds_have_forbidden_group(cr);
463}

466static ngx_uint_t
467idmap_hash(const char *s)
468{
469 ngx_uint_t h = 5381;
470
471 while (*s) {
472 h = ((h << 5) + h) ^ (ngx_uint_t) (unsigned char) *s++; /* djb2-xor */
473 }
474 return h & (IDMAP_CACHE_SLOTS - 1);
475}

478ngx_int_t
479brix_idmap_init(const brix_idmap_conf_t *conf, ngx_log_t *log)
480{
481 if (conf == NULL) {
482 return NGX_ERROR;
483 }
484
485 idmap_ttl = (conf->cache_ttl > 0) ? (time_t) conf->cache_ttl
486 : BRIX_IDMAP_DEFAULT_TTL;
487 idmap_min_uid = (conf->min_uid > 0) ? conf->min_uid
488 : BRIX_IDMAP_DEFAULT_MIN_UID;
489 /*
490 * Clamp the effective floor UP to the absolute hard floor: ids below
491 * BRIX_IMP_HARD_MIN_ID can never be impersonated, even if an admin sets a
492 * lower brix_idmap_min_uid. (A lower value would in any case be caught at
493 * the broker's syscall edge; clamping here turns it into a clean deny instead
494 * of a fatal broker abort.)
495 */
496 if (idmap_min_uid < BRIX_IMP_HARD_MIN_ID) {
497 if (log != NULL) {
498 ngx_log_error(NGX_LOG_NOTICE, log, 0,
499 "impersonate: brix_idmap_min_uid %d raised to the hard "
500 "reserved-id floor %d (uids/gids below %d can never be "
501 "impersonated)", (int) idmap_min_uid,
502 BRIX_IMP_HARD_MIN_ID, BRIX_IMP_HARD_MIN_ID);
503 }
504 idmap_min_uid = BRIX_IMP_HARD_MIN_ID;
505 }
506 idmap_primary_only = conf->primary_only ? 1 : 0;
507
508 /*
509 * Resolve the deny-lists (account names -> uids, privileged group names ->
510 * gids) once, here. Reset first so a hot-reload re-reads them. The nginx
511 * worker uid is ALWAYS forbidden as a target so the gateway cannot be
512 * impersonated as itself. Names are taken from config or the built-in
513 * defaults; unresolved names are skipped.
514 */
515 idmap_forbidden_uids_n = 0;
516 idmap_forbidden_gids_n = 0;
517 {
518 char buf[1024];
519 const char *users = NULL;
520 if (conf->forbidden_users.len > 0
521 && brix_str_cbuf(buf, sizeof(buf), &conf->forbidden_users) != NULL) {
522 users = buf;
523 }
524 idmap_forbid_load(users, BRIX_IMP_DEFAULT_FORBIDDEN_USERS, 1, log);
525 }
526 {
527 char buf[1024];
528 const char *groups = NULL;
529 if (conf->forbidden_groups.len > 0
530 && brix_str_cbuf(buf, sizeof(buf), &conf->forbidden_groups) != NULL) {
531 groups = buf;
532 }
533 idmap_forbid_load(groups, BRIX_IMP_DEFAULT_FORBIDDEN_GROUPS, 0, log);
534 }
535 if (conf->worker_uid != 0 && conf->worker_uid != (uid_t) NGX_CONF_UNSET_UINT) {
536 idmap_forbid_uid(conf->worker_uid);
537 }
538 if (log != NULL) {
539 ngx_log_error(NGX_LOG_NOTICE, log, 0,
540 "impersonate: deny-lists loaded (%d forbidden uids, %d "
541 "forbidden gids)", idmap_forbidden_uids_n,
542 idmap_forbidden_gids_n);
543 if (idmap_forbidden_uids_n >= IDMAP_FORBID_MAX
544 || idmap_forbidden_gids_n >= IDMAP_FORBID_MAX) {
545 ngx_log_error(NGX_LOG_WARN, log, 0,
546 "impersonate: forbidden deny-list hit the %d-entry cap; "
547 "names beyond it were NOT enforced — reduce the list or "
548 "raise IDMAP_FORBID_MAX", IDMAP_FORBID_MAX);
549 }
550 }
551
552 idmap_default_user[0] = '\0';
553 if (conf->default_user.len > 0) {
554 (void) brix_str_cbuf(idmap_default_user, sizeof(idmap_default_user),
555 &conf->default_user);
556 }
557
558 ngx_memzero(idmap_cache, sizeof(idmap_cache)); /* drop the cache */
559
560 {
561 char pbuf[IDMAP_PRINC_MAX];
562 const char *path = NULL;
563 if (conf->gridmap_path.len > 0
564 && brix_str_cbuf(pbuf, sizeof(pbuf), &conf->gridmap_path) != NULL)
565 {
566 path = pbuf;
567 }
568 return idmap_gridmap_load(path, log);
569 }
570}

572ngx_int_t
573brix_idmap_resolve(const brix_idmap_conf_t *conf, const char *principal,
574 brix_idmap_creds_t *out, ngx_log_t *log)
575{
576 idmap_cache_slot_t *slot;
577 const char *user;
578 brix_idmap_creds_t creds;
579 time_t now;
580 int rc;
581
582 (void) conf; /* config is installed via brix_idmap_init() */
583
584 if (principal == NULL || out == NULL
585 || principal[0] == '\0' || ngx_strlen(principal) >= IDMAP_PRINC_MAX)
586 {
587 return BRIX_IDMAP_DENY;
588 }
589
590 now = time(NULL);
591 slot = &idmap_cache[idmap_hash(principal)];
592
593 /* Cache hit on the same principal, not expired. */
594 if (slot->expiry > now && strcmp(slot->principal, principal) == 0) {
595 if (slot->rc == BRIX_IDMAP_OK || slot->rc == BRIX_IDMAP_SQUASH) {
596 *out = slot->creds;
597 }
598 return slot->rc;
599 }
600
601 /* (1) grid-mapfile, else (2) the principal as a literal username. */
602 user = idmap_gridmap_lookup(principal);
603 if (user == NULL) {
604 user = principal;
605 }
606
607 rc = BRIX_IDMAP_DENY;
608 if (idmap_resolve_user(user, &creds) == 0 && idmap_creds_allowed(&creds)) {
609 rc = BRIX_IDMAP_OK;
610 } else if (idmap_default_user[0] != '\0'
611 && idmap_resolve_user(idmap_default_user, &creds) == 0
612 && idmap_creds_allowed(&creds))
613 {
614 rc = BRIX_IDMAP_SQUASH; /* unmapped/forbidden -> squash account */
615 }
616
617 if (rc != BRIX_IDMAP_OK && rc != BRIX_IDMAP_SQUASH && log != NULL) {
618 ngx_log_error(NGX_LOG_INFO, log, 0,
619 "impersonate: no UNIX mapping for principal \"%s\" "
620 "(user=\"%s\") -> deny", principal, user);
621 }
622
623 /* Cache the verdict (incl. denies, to bound repeated NSS misses). */
624 ngx_memcpy(slot->principal, principal, ngx_strlen(principal) + 1);
625 slot->rc = rc;
