7/*
8 * Per-identity bucket state stored verbatim as the KV value.
9 *
10 * WHAT: the entire mutable token-bucket state for one identity — token count
11 * plus the wall-clock timestamp the count was last brought current.
12 * WHY: the KV value is an opaque fixed-width byte blob (memcpy'd in/out of the
13 * shared slot), so the struct IS the on-shm record; its size must not
14 * exceed the zone's val_max (checked at config time, see directive).
15 * HOW: read with brix_kv_get into a stack copy, mutated locally, written
16 * back with brix_kv_set. No pointers — safe to copy across the
17 * process boundary into shared memory. sizeof() is typically 16 bytes
18 * (4-byte tokens + ngx_msec_t, modulo alignment padding).
19 */
20typedef struct {
21 uint32_t tokens; /* tokens currently available */
22 ngx_msec_t last_refill; /* ngx_current_msec of the last refill */
23} brix_rl_val_t;
24
25/* KV key is "rl:" + raw identity bytes; the prefix namespaces rate-limit
26 * entries so a single zone could be shared with other "rl:"-free consumers. */
27#define BRIX_RL_KEY_PREFIX_LEN 3 /* "rl:" */
28#define BRIX_RL_ID_MAX 256 /* identity bytes hashed; longer ids are truncated */
29
30/*
31 * brix_rate_limit_check() — one token-bucket admission test for `id`.
32 *
33 * WHAT: refill the identity's bucket by elapsed time, spend one token, and
34 * return NGX_OK (admit) or NGX_DECLINED (empty bucket — reject).
35 * WHY: bucket state is shared across all workers via the KV zone, so the
36 * limit is global, not per-worker.
37 * HOW (locking / consistency): this function takes NO lock itself. Each of
38 * the brix_kv_get and brix_kv_set calls grabs and releases the
39 * zone's shared spinlock independently (see kv.c). The read-modify-write
40 * is therefore NOT atomic: two workers can both read the same token count
41 * and both spend it, so the effective limit may over-admit slightly under
42 * contention. This is the deliberate "best-effort" contract documented
43 * in rate_limit.h — acceptable for request throttling, and it keeps the
44 * critical section to a single O(1) probe with no callback inside it.
45 */
46ngx_int_t
47brix_rate_limit_check(const brix_rate_limit_conf_t *rl,
48 const char *id, size_t id_len)
49{
50 u_char key[BRIX_RL_KEY_PREFIX_LEN + BRIX_RL_ID_MAX];
51 size_t key_len;
52 brix_rl_val_t v;
53 size_t vlen = sizeof(v);
54 ngx_msec_t now, elapsed, ttl;
55 uint32_t refill;
56
57 /* Fail-open: a missing zone or zeroed rate/burst means the directive was
58 * never configured for this scope, so admit unconditionally rather than
59 * dividing by zero in the ttl computation below. */
60 if (rl == NULL || rl->kv == NULL || rl->rate == 0 || rl->burst == 0) {
61 return NGX_OK; /* disabled — admit */
62 }
63 /* Clamp the identity to the fixed key buffer; the stack `key` array is
64 * exactly PREFIX_LEN + BRIX_RL_ID_MAX bytes, so this bounds the memcpy
65 * below and prevents a stack overflow on an attacker-influenced id_len. */
66 if (id_len > BRIX_RL_ID_MAX) {
67 id_len = BRIX_RL_ID_MAX;
68 }
69
70 /* Build "rl:" + id in place; key_len is the exact span hashed by the KV
71 * layer (no NUL terminator — these are length-counted opaque bytes). */
72 key[0] = 'r'; key[1] = 'l'; key[2] = ':';
73 ngx_memcpy(key + BRIX_RL_KEY_PREFIX_LEN, id, id_len);
74 key_len = BRIX_RL_KEY_PREFIX_LEN + id_len;
75
76 now = ngx_current_msec;
77
78 /* Pre-seed `v` with a full, freshly-refilled bucket. brix_kv_get only
79 * overwrites &v on a hit, so on a miss (new identity, or an entry the KV
80 * layer lazily evicted as TTL-expired) these defaults stand — a first-seen
81 * caller starts with the full burst allowance. vlen is passed by-ref and
82 * clamped by the KV layer to the stored val_len. */
83 v.tokens = rl->burst;
84 v.last_refill = now;
85 (void) brix_kv_get(rl->kv, key, key_len, &v, &vlen);
86
87 /* Refill proportional to elapsed time. When no whole token has accrued,
88 * leave last_refill untouched so the fractional remainder keeps building
89 * across calls rather than being discarded. */
90 /* elapsed is unsigned ngx_msec_t arithmetic. In normal operation now >=
91 * last_refill; should the monotonic-ish clock ever go backwards the wrap
92 * yields a huge elapsed, but refill is then clamped to burst below, so the
93 * worst case is one over-full bucket, never a deficit. */
94 elapsed = now - v.last_refill;
95 if (elapsed > 0) {
96 /* tokens accrued = elapsed_ms * rate_per_sec / 1000. 64-bit product
97 * before the divide avoids overflow for large elapsed*rate; the result
98 * fits uint32_t because it is immediately capped at burst. */
99 refill = (uint32_t) ((uint64_t) elapsed * rl->rate / 1000ULL);
100 if (refill > 0) {
101 uint64_t t = (uint64_t) v.tokens + refill;
102 v.tokens = (t > rl->burst) ? rl->burst : (uint32_t) t;
103 /* Only advance last_refill when at least one whole token accrued.
104 * Leaving it untouched when refill == 0 preserves the sub-token
105 * fractional remainder: it keeps accumulating in `elapsed` across
106 * calls instead of being silently rounded down to zero each time,
107 * which would otherwise cap the effective rate below `rate`. */
108 v.last_refill = now;
109 }
110 }
111
112 /* Per-entry TTL = time to refill an empty bucket to full + 1s slack.
113 * WHY: once this many ms pass with no traffic the bucket would be full
114 * anyway, so letting the KV layer lazily evict the idle entry on its next
115 * probe is free — it just re-seeds to a full bucket on the next request.
116 * Division by rl->rate is safe: the rate==0 fail-open guard above
117 * guarantees rate is non-zero here. */
118 ttl = (ngx_msec_t) ((uint64_t) rl->burst * 1000 / rl->rate) + 1000;
119
120 /* Empty bucket: persist the (refilled-but-still-zero) state so last_refill
121 * advances, then reject. We still write so the next caller's elapsed is
122 * measured from now, not from the original last_refill. */
123 if (v.tokens == 0) {
124 (void) brix_kv_set(rl->kv, key, key_len, &v, sizeof(v), ttl);
125 return NGX_DECLINED; /* bucket empty — reject */
126 }
127
128 /* Spend one token for this request and write the bucket back. The set may
129 * silently fail (zone full at the 0.5 load-factor cap, see kv.c); we ignore
130 * that and admit — failing open is the right call for a throttle whose
131 * state store is momentarily saturated. */
132 v.tokens--;
133 (void) brix_kv_set(rl->kv, key, key_len, &v, sizeof(v), ttl);
134 return NGX_OK;
135}

137/*
138 * brix_rl_parse_rate() — parse a "rate=<N>r/s" or "rate=<N>" token.
139 *
140 * WHAT: read the numeric rate out of `arg` (already known to start "rate="),
141 * storing it in *rate and returning NGX_OK; on a non-positive or
142 * non-numeric value it logs the config error and returns NGX_ERROR.
143 * WHY: isolates the one parameter that has an optional "r/s" suffix so the
144 * dispatcher stays a flat prefix ladder; the caller maps NGX_ERROR to
145 * NGX_CONF_ERROR after this function has already emitted the message.
146 * HOW: 1. Point past the 5-byte "rate=" prefix. 2. Drop a trailing "r/s"
147 * (requires len > 3 so a bare "r/s" is not mistaken for zero digits).
148 * 3. ngx_atoi the remaining digits; reject NGX_ERROR or n <= 0.
149 * 4. Store the widened value.
150 */
151static ngx_int_t
152brix_rl_parse_rate(ngx_conf_t *cf, ngx_str_t *arg, ngx_uint_t *rate)
153{
154 u_char *p = arg->data + 5;
155 size_t len = arg->len - 5;
156 ngx_int_t n;
157
158 /* accept "<N>r/s" or bare "<N>" */
159 if (len > 3 && ngx_strncmp(p + len - 3, "r/s", 3) == 0) {
160 len -= 3;
161 }
162 n = ngx_atoi(p, len);
163 if (n == NGX_ERROR || n <= 0) {
164 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
165 "invalid brix_rate_limit rate \"%V\"", arg);
166 return NGX_ERROR;
167 }
168 *rate = (ngx_uint_t) n;
169 return NGX_OK;
170}

172/*
173 * brix_rl_parse_burst() — parse a "burst=<N>" token.
174 *
175 * WHAT: read the numeric burst out of `arg` (already known to start "burst="),
176 * storing it in *burst and returning NGX_OK; on a non-positive or
177 * non-numeric value it logs the config error and returns NGX_ERROR.
178 * WHY: mirrors brix_rl_parse_rate so each parameter owns its own validation
179 * and error string, keeping the dispatcher branch-free of parse detail.
180 * HOW: 1. ngx_atoi the bytes past the 6-byte "burst=" prefix. 2. Reject
181 * NGX_ERROR or n <= 0 with the config error. 3. Store the value.
182 */
183static ngx_int_t
184brix_rl_parse_burst(ngx_conf_t *cf, ngx_str_t *arg, ngx_uint_t *burst)
185{
186 ngx_int_t n = ngx_atoi(arg->data + 6, arg->len - 6);
187
188 if (n == NGX_ERROR || n <= 0) {
189 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
190 "invalid brix_rate_limit burst \"%V\"", arg);
191 return NGX_ERROR;
192 }
193 *burst = (ngx_uint_t) n;
194 return NGX_OK;
195}

197/*
198 * brix_rl_parse_key() — parse a "key=dn" / "key=ip" token.
199 *
200 * WHAT: set *key_ip to 1 for "ip" or 0 for "dn" and return NGX_OK; any other
201 * value logs the config error and returns NGX_ERROR.
202 * WHY: the keying identity selects which field brix_rate_limit_check() hashes;
203 * strict matching keeps a typo like "key=ipv6" a hard config error rather
204 * than a silent prefix match on "ip".
205 * HOW: require the whole token to be exactly 6 bytes ("key=" + 2) and compare
206 * the two-byte tail against "ip" then "dn"; fall through to the error.
207 */
208static ngx_int_t
209brix_rl_parse_key(ngx_conf_t *cf, ngx_str_t *arg, ngx_uint_t *key_ip)
210{
211 /* len == 6 pins the value to exactly two bytes ("key=" + 2), so
212 * "ip"/"dn" match strictly and e.g. "key=ipv6" is rejected rather
213 * than prefix-matching "ip". */
214 if (arg->len == 6 && ngx_strncmp(arg->data + 4, "ip", 2) == 0) {
215 *key_ip = 1;
216 return NGX_OK;
217 }
218 if (arg->len == 6 && ngx_strncmp(arg->data + 4, "dn", 2) == 0) {
219 *key_ip = 0;
220 return NGX_OK;
221 }
222 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
223 "brix_rate_limit key must be dn or ip, got \"%V\"", arg);
224 return NGX_ERROR;
225}

227/*
228 * brix_rl_parse_arg() — dispatch one directive token to its parameter parser.
229 *
230 * WHAT: classify `arg` by its "name=" prefix and route it to the matching
231 * parser, updating the referenced accumulators; returns NGX_OK when the
232 * token was recognised and valid, NGX_ERROR when it was unknown or the
233 * sub-parser rejected it (the error is logged by whichever detected it).
234 * WHY: keeps brix_rate_limit_directive() a flat loop over tokens — all the
235 * per-parameter knowledge lives in the small parsers above.
236 * HOW: 1. "zone=" copies the name tail into *zone (always valid). 2. "rate=",
237 * "burst=", "key=" defer to their parsers. 3. Anything else is an
238 * unknown parameter and is reported here.
239 */
240static ngx_int_t
241brix_rl_parse_arg(ngx_conf_t *cf, ngx_str_t *arg, ngx_str_t *zone,
242 ngx_uint_t *rate, ngx_uint_t *burst, ngx_uint_t *key_ip)
243{
244 if (ngx_strncmp(arg->data, "zone=", 5) == 0) {
245 zone->data = arg->data + 5;
246 zone->len = arg->len - 5;
247 return NGX_OK;
248 }
249 if (ngx_strncmp(arg->data, "rate=", 5) == 0) {
250 return brix_rl_parse_rate(cf, arg, rate);
251 }
252 if (ngx_strncmp(arg->data, "burst=", 6) == 0) {
253 return brix_rl_parse_burst(cf, arg, burst);
254 }
255 if (ngx_strncmp(arg->data, "key=", 4) == 0) {
256 return brix_rl_parse_key(cf, arg, key_ip);
257 }
258 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
259 "invalid brix_rate_limit parameter \"%V\"", arg);
260 return NGX_ERROR;
261}

263/*
264 * brix_rl_resolve_zone() — look up and size-check the named KV zone.
265 *
266 * WHAT: find the zone named `zone`, verify its value slots can hold a whole
267 * brix_rl_val_t, and return the handle in *out; returns NGX_OK on success
268 * or NGX_ERROR (with the config error logged) when the zone is missing or
269 * too small.
270 * WHY: the size guard runs once at config time so brix_rate_limit_check() can
271 * memcpy the bucket struct into the slot without re-checking sizes on
272 * every request.
273 * HOW: 1. brix_kv_find by name (the zone was registered earlier by
274 * brix_kv_zone in the same main block). 2. Reject a NULL lookup.
275 * 3. Reject a val_max below sizeof(brix_rl_val_t). 4. Publish the handle.
276 */
277static ngx_int_t
278brix_rl_resolve_zone(ngx_conf_t *cf, ngx_str_t *zone, brix_kv_t **out)
279{
280 brix_kv_t *kv = brix_kv_find(zone);
281
282 if (kv == NULL) {
283 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
284 "brix_rate_limit: unknown zone \"%V\" "
285 "(declare it with brix_kv_zone first)", zone);
286 return NGX_ERROR;
287 }
288 /* Refuse a zone whose value slots cannot hold a whole bucket record;
289 * otherwise brix_kv_set would truncate/reject the value at runtime and
290 * the limiter would silently malfunction. */
291 if (kv->val_max < sizeof(brix_rl_val_t)) {
292 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
293 "brix_rate_limit zone \"%V\" too small: need val>=%uz",
294 zone, sizeof(brix_rl_val_t));
295 return NGX_ERROR;
296 }
297 *out = kv;
298 return NGX_OK;
299}

301/*
302 * brix_rate_limit_directive() — parse and bind one
303 * brix_rate_limit zone=<name> rate=<N>r/s burst=<N> [key=dn|ip];
304 *
305 * WHAT: validate the args, resolve the named KV zone, and populate the
306 * brix_rate_limit_conf_t embedded in the caller's conf struct.
307 * WHY: this is the only place rl->kv is set, so a parse failure here leaves
308 * kv == NULL and brix_rate_limit_check() fails open (admits).
309 * HOW: cmd->offset is the byte offset of the embedded conf field within the
310 * owning module's conf struct — the usual ngx_command_t pattern for a
311 * sub-struct that has no dedicated NGX_*_CONF_OFFSET slot. 1. Fold each
312 * token through brix_rl_parse_arg. 2. Require the mandatory trio.
313 * 3. Resolve+size-check the zone. 4. Commit the fields.
314 */
315char *
316brix_rate_limit_directive(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
317{
318 /* Reach the embedded conf field by raw offset (see doc comment). */
319 brix_rate_limit_conf_t *rl =
320 (brix_rate_limit_conf_t *) ((char *) conf + cmd->offset);
321 ngx_str_t *value = cf->args->elts;
322 ngx_str_t zone = ngx_null_string;
323 ngx_uint_t rate = 0;
324 ngx_uint_t burst = 0;
325 ngx_uint_t key_ip = 0;
326 ngx_uint_t i;
327 brix_kv_t *kv;
328
329 for (i = 1; i < cf->args->nelts; i++) {
330 if (brix_rl_parse_arg(cf, &value[i], &zone, &rate, &burst, &key_ip)
331 != NGX_OK)
332 {
333 return NGX_CONF_ERROR;
334 }
335 }
336
337 if (zone.len == 0 || rate == 0 || burst == 0) {
338 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
339 "brix_rate_limit requires zone=<name> rate=<N>r/s burst=<N>");
340 return NGX_CONF_ERROR;
341 }
342
343 if (brix_rl_resolve_zone(cf, &zone, &kv) != NGX_OK) {
344 return NGX_CONF_ERROR;
345 }
346
347 rl->kv = kv;
348 rl->rate = rate;
349 rl->burst = burst;
350 rl->key_ip = key_ip;
351 return NGX_CONF_OK;
352}
