1/*
2 * fs_walk_remove.c - confined recursive tree removal (split from fs_walk.c).
3 *
4 * WHAT: Recursively deletes a directory tree confined to an export root_canon.
5 * brix_fs_remove_tree_confined() drives the opendir/readdir loop and the
6 * final rmdir; brix_fs_remove_tree_entry() handles one child per iteration
7 * (recurse into subdirs, unlink files) through the beneath API.
8 *
9 * WHY: XRootD DEL/MOVE/COPY on collections and S3 AbortMultipartUpload cleanup
10 * need recursive child deletion. The confined check (root_canon) ensures no
11 * entry escapes the root path boundary, preventing ../ traversal attacks.
12 * Split from fs_walk.c to keep both halves under the file-size cap; the walk
13 * helpers (dot-entry / join-path / dir-is-empty / callback walk) stay there,
14 * the tree-removal cluster lives here. Pure code motion, no behaviour change.
15 *
16 * HOW: opendir → readdir loop per entry: join path, strip to the root-relative
17 * form, lstat via the beneath helper. If dir → recurse. If file →
18 * brix_unlink_beneath(). After all children processed → brix_unlink_beneath()
19 * with the rmdir flag on the directory itself. Reuses brix_fs_is_dot_entry()
20 * and brix_fs_join_path() from fs_walk.h (both public API).
21 */

31/*
32 * brix_fs_remove_tree_entry - remove one child during a confined tree walk.
33 *
34 * WHAT: Handles a single directory entry of brix_fs_remove_tree_confined(): skips
35 * dot entries, joins and strips the child path, lstats it via the beneath
36 * helper, then recurses into subdirectories or unlinks files through the
37 * beneath API. Returns NGX_OK to keep iterating, NGX_ERROR to stop.
38 *
39 * WHY: Extracted from the readdir loop so the outer function stays under the
40 * complexity/length budget while every mutation still flows through the
41 * beneath API (the export-root boundary). Behaviour is identical to the
42 * original inline body: dot and ENOENT entries are skipped; any join/strip
43 * failure, non-ENOENT lstat, failed recursion, or failed unlink stops the
44 * walk and (for the syscall failures) logs and leaves errno set.
45 *
46 * HOW: Guarded early returns; on the strip-escape case errno is set to EXDEV to
47 * match the original, and NGX_ERROR signals the loop to break.
48 */

56static ngx_int_t
57brix_fs_remove_tree_entry(ngx_log_t *log, const char *root_canon, int rootfd,
58 const char *path, const struct dirent *de, ngx_uint_t depth)
59{
60 char child[PATH_MAX];
61 struct stat st;
62 const char *child_rel;
63
64 if (brix_fs_is_dot_entry(de->d_name)) {
65 return NGX_OK;
66 }
67
68 if (brix_fs_join_path(path, de->d_name, child, sizeof(child)) != NGX_OK) {
69 return NGX_ERROR;
70 }
71
72 child_rel = brix_beneath_strip_root(root_canon, child);
73 if (child_rel == NULL) {
74 errno = EXDEV;
75 return NGX_ERROR;
76 }
77
78 /* lstat via the beneath helper so it runs AS THE MAPPED USER under
79 * impersonation (the raw lstat would fail EACCES inside a 0700 user-owned
80 * staging dir); off impersonation it is the same local lstat. */
81 if (brix_lstat_beneath(rootfd, child_rel, &st) != 0) {
82 if (errno == ENOENT) {
83 return NGX_OK;
84 }
85 ngx_log_error(NGX_LOG_ERR, log, errno,
86 "brix: remove-tree lstat failed \"%s\"", child);
87 return NGX_ERROR;
88 }
89
90 if (S_ISDIR(st.st_mode)) {
91 return brix_fs_remove_tree_at(log, root_canon, child, depth + 1);
92 }
93
94 if (brix_unlink_beneath(rootfd, child_rel, 0) != 0) {
95 ngx_log_error(NGX_LOG_ERR, log, errno,
96 "brix: remove-tree unlink failed \"%s\"", child);
97 return NGX_ERROR;
98 }
99
100 return NGX_OK;
101}

103/*
104 * brix_fs_remove_tree_confined - recursively delete directory tree confined to root_canon.
105 *
106 * WHAT: Opens path via opendir(), iterates entries (skip dots), lstat each child,
107 * recurses into subdirectories, unlinks files via brix_unlink_confined_canon().
108 * After processing children, rmdir the directory itself. Returns NGX_OK on
109 * full removal or NGX_ERROR on any failure.
110 *
111 * WHY: XRootD DEL/MOVE/COPY on collections requires recursive child deletion. The
112 * confined check (root_canon) ensures no entry escapes the root path boundary,
113 * preventing ../ traversal attacks.
114 *
115 * HOW: opendir → readdir loop per entry: join path, lstat. If dir → recurse.
116 * If file → brix_unlink_confined_canon(root_canon, child, 0). After all
117 * children processed → brix_unlink_confined_canon(root_canon, path, 1) (rmdir).
118 */
119
120static ngx_int_t
121brix_fs_remove_tree_at(ngx_log_t *log, const char *root_canon,
122 const char *path, ngx_uint_t depth)
123{
124 DIR *dp;
125 struct dirent *de;
126 ngx_int_t rc;
127 int rootfd;
128 const char *path_rel;
129
130 /* D-6/T2: independent C-stack recursion — bound it so a hostile deep tree
131 * aborts cleanly instead of overflowing the worker stack. */
132 if (depth > BRIX_FS_TREE_MAX_DEPTH) {
133 ngx_log_error(NGX_LOG_ERR, log, 0,
134 "brix: remove-tree depth cap (%d) exceeded at \"%s\"",
135 BRIX_FS_TREE_MAX_DEPTH, path);
136 errno = ELOOP;
137 return NGX_ERROR;
138 }
139
140 /*
141 * The opendir/readdir/lstat iteration below is NOT a beneath site: it walks
142 * by absolute child path built from single, kernel-supplied dirent names
143 * (no client `..`), and lstat does not follow the final component, so a
144 * dangling/outward symlink entry is seen as a symlink and unlinked as one
145 * (its target is never opened or followed). The actual mutations — every
146 * unlink and the final rmdir — DO go through the beneath API, which is where
147 * the export-root boundary is enforced. A readdir-relative
148 * openat(dirfd,...) walk would be a defence-in-depth refinement but is not a
149 * confinement gap here.
150 */
151 rootfd = brix_beneath_open_root(root_canon);
152 if (rootfd < 0) {
153 ngx_log_error(NGX_LOG_ERR, log, errno,
154 "brix: remove-tree cannot open root for \"%s\"", path);
155 return NGX_ERROR;
156 }
157 path_rel = brix_beneath_strip_root(root_canon, path);
158 if (path_rel == NULL) {
159 close(rootfd);
160 errno = EXDEV;
161 return NGX_ERROR;
162 }
163
164 /* Open the dir AS THE MAPPED USER under impersonation (broker fd + fdopendir):
165 * a multipart staging dir is owned 0700 by the mapped user, so a raw worker
166 * opendir() here fails EACCES and breaks AbortMultipartUpload / the cleanup.
167 * Off impersonation this is a plain opendir(). */
168 dp = brix_opendir_confined_canon(log, root_canon, path);
169 if (dp == NULL) {
170 close(rootfd);
171 if (errno == ENOENT) {
172 return NGX_OK;
173 }
174 ngx_log_error(NGX_LOG_ERR, log, errno,
175 "brix: remove-tree opendir failed \"%s\"", path);
176 return NGX_ERROR;
177 }
178
179 rc = NGX_OK;
180
181 while ((de = readdir(dp)) != NULL) {
182 if (brix_fs_remove_tree_entry(log, root_canon, rootfd, path, de, depth)
183 != NGX_OK)
184 {
185 rc = NGX_ERROR;
186 break;
187 }
188 }
189
190 closedir(dp);
191
192 if (rc == NGX_OK
193 && brix_unlink_beneath(rootfd, path_rel, 1) != 0)
194 {
195 ngx_log_error(NGX_LOG_ERR, log, errno,
196 "brix: remove-tree rmdir failed \"%s\"", path);
197 rc = NGX_ERROR;
198 }
199
200 close(rootfd);
201 return rc;
202}
