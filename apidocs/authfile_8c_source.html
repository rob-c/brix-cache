1/*
2 * authfile.c — XrdAcc authorization database parser (XrdAccAuthFile/ConfigDB).
3 *
4 * WHAT: brix_acc_authfile_parse() reads a stock XRootD `authdb` file and
5 * builds a complete brix_acc_tables_t generation: the per-category name
6 * lists (users/groups/hosts/netgroups/orgs/roles/templates), the default
7 * (`u *`) and fungible (`u =`) lists, the domain (`h .suffix`) list, and the
8 * compound-identity (`=`) definitions wired to their exclusive (`x`) or
9 * inclusive (`s`) rule capabilities.
10 *
11 * WHY: faithful port of XrdAccConfig::ConfigDBrec()/idDef() plus the
12 * XrdAccAuthFile tokenizer, so a site can point `brix_authdb_format xrdacc`
13 * at an existing XRootD authdb and get identical structure.
14 *
15 * HOW: this unit is the orchestrator — it loads the file (<= 1 MiB), drives the
16 * tokenizer (authfile_tokenize.c) to accumulate each logical record, dispatches
17 * each record by its type letter to the per-type handlers (authfile_record.c),
18 * then finalizes the exclusive/inclusive rule ordering. All storage comes from
19 * the generation's own pool for a one-shot atomic free.
20 */

29/*
30 * acc_finalize_rules — split the parsed `=` definitions into the exclusive
31 * (sx_list, file order) and inclusive (sy_list) chains, dropping any definition
32 * that never received an x/s rule (caps == NULL). Runs once after the whole
33 * file parses so the access engine sees the two ordered rule lists.
34 */
35static void
36acc_finalize_rules(brix_acc_tables_t *tabs, ngx_log_t *log)
37{
38 brix_acc_idrule_t *id, *next;
39 brix_acc_idrule_t *sx_tail = NULL, *sy_tail = NULL;
40
41 for (id = tabs->id_defs; id != NULL; id = next) {
42 next = id->next;
43 id->next = NULL;
44
45 if (id->caps == NULL) {
46 ngx_log_error(NGX_LOG_WARN, log, 0,
47 "xrootd authdb: id \"%s\" defined but never used "
48 "(no x/s rule) — ignored", id->name);
49 continue;
50 }
51 if (id->rule >= 0) { /* exclusive, file order */
52 if (sx_tail == NULL) {
53 tabs->sx_list = id;
54 } else {
55 sx_tail->next = id;
56 }
57 sx_tail = id;
58 } else { /* inclusive */
59 if (sy_tail == NULL) {
60 tabs->sy_list = id;
61 } else {
62 sy_tail->next = id;
63 }
64 sy_tail = id;
65 }
66 }
67}

69/*
70 * acc_dispatch_record — validate the one-letter record type and name of an
71 * accumulated record and route it to its type handler (=, x, s, or a named
72 * g/h/n/o/r/t/u). Returns NGX_OK or NGX_ERROR (logged). Bumps rule_count for
73 * diagnostics on every well-formed record.
74 */
75static ngx_int_t
76acc_dispatch_record(acc_parse_ctx_t *pc, char **w, ngx_uint_t n)
77{
78 char rtype;
79
80 if (w[0][0] == '\0' || w[0][1] != '\0') {
81 BRIX_DIAG_EMERG(pc->log, 0,
82 "xrootd authdb: invalid record type \"%s\"",
83 "each record must begin with a single-letter type in column 1",
84 "valid record types are = x s g h n o r t u; fix this line in "
85 "the authdb file",
86 w[0]);
87 return NGX_ERROR;
88 }
89 rtype = w[0][0];
90
91 if (n < 2) {
92 ngx_log_error(NGX_LOG_EMERG, pc->log, 0,
93 "xrootd authdb: record name missing after '%c'", rtype);
94 return NGX_ERROR;
95 }
96
97 pc->tabs->rule_count++;
98
99 switch (rtype) {
100 case '=': return acc_record_iddef(pc, w, n);
101 case 'x': return acc_record_rule(pc, w, n, 1);
102 case 's': return acc_record_rule(pc, w, n, 0);
103 case 'g': case 'h': case 'n': case 'o':
104 case 'r': case 't': case 'u':
105 return acc_record_named(pc, rtype, w, n);
106 default:
107 ngx_log_error(NGX_LOG_EMERG, pc->log, 0,
108 "xrootd authdb: invalid id type \"%c\"", rtype);
109 return NGX_ERROR;
110 }
111}

123/*
124 * acc_load_authfile — open, size-check (max 1 MiB) and slurp the authdb `file`
125 * into a fresh pool buffer. On success fills `ld` (buf/fsize/mtime) and returns
126 * ACC_LOAD_OK; an empty file returns ACC_LOAD_EMPTY (a valid deny-all db, mtime
127 * still set); any error returns ACC_LOAD_ERROR (logged). Split out of the
128 * parser so the main function is just load then tokenize then finalize.
129 */
130static int
131acc_load_authfile(ngx_log_t *log, const char *file, ngx_pool_t *pool,
132 acc_loaded_t *ld)
133{
134 ngx_fd_t fd;
135 ngx_file_info_t fi;
136 ssize_t nread;
137 size_t n;
138
139 fd = ngx_open_file((u_char *) file, NGX_FILE_RDONLY, NGX_FILE_OPEN, 0);
140 if (fd == NGX_INVALID_FILE) {
141 BRIX_DIAG_EMERG(log, ngx_errno,
142 "xrootd authdb: cannot open \"%s\"",
143 "the path in brix_acc_authdb is wrong or unreadable by the "
144 "nginx user",
145 "check the path exists and the nginx master/worker user can read "
146 "it (the OS reason is appended below)",
147 file);
148 return ACC_LOAD_ERROR;
149 }
150 if (ngx_fd_info(fd, &fi) == NGX_FILE_ERROR) {
151 ngx_close_file(fd);
152 return ACC_LOAD_ERROR;
153 }
154 n = (size_t) ngx_file_size(&fi);
155 if (n > BRIX_ACC_AUTHDB_MAX) {
156 BRIX_DIAG_EMERG(log, 0,
157 "xrootd authdb \"%s\" exceeds the 1 MiB limit",
158 "the file is larger than the parser accepts — usually a wrong "
159 "path (pointing at a data file) or a runaway generated authdb",
160 "confirm brix_acc_authdb points at the authorization file, not "
161 "something else; split or trim it below 1 MiB",
162 file);
163 ngx_close_file(fd);
164 return ACC_LOAD_ERROR;
165 }
166
167 ld->mtime = ngx_file_mtime(&fi);
168 ld->fsize = n;
169
170 if (n == 0) {
171 ngx_close_file(fd);
172 return ACC_LOAD_EMPTY; /* empty authdb = deny-all, valid */
173 }
174
175 ld->buf = ngx_pnalloc(pool, n);
176 if (ld->buf == NULL) {
177 ngx_close_file(fd);
178 return ACC_LOAD_ERROR;
179 }
180 nread = ngx_read_fd(fd, ld->buf, n);
181 ngx_close_file(fd);
182 if (nread < 0 || (size_t) nread != n) {
183 BRIX_DIAG_EMERG(log, 0,
184 "xrootd authdb: short read of \"%s\"",
185 "the file changed size or an I/O error occurred while reading it",
186 "make sure the authdb is written atomically (write-then-rename), "
187 "then re-run nginx -t",
188 file);
189 return ACC_LOAD_ERROR;
190 }
191 return ACC_LOAD_OK;
192}

194/*
195 * acc_tokenize_and_dispatch — run the tokenizer over `buf`/`fsize`, accumulating
196 * each logical record's words and dispatching them through the parse context.
197 * Returns NGX_OK when the whole buffer parsed, or NGX_ERROR on the first
198 * tokenizer alloc failure or record error (which is already logged).
199 */
200static ngx_int_t
201acc_tokenize_and_dispatch(acc_parse_ctx_t *pc, u_char *buf, size_t fsize)
202{
203 acc_tok_t tok;
204 ngx_array_t *words; /* of char* for the current record */
205 int tok_rc;
206
207 words = ngx_array_create(pc->tabs->pool, 16, sizeof(char *));
208 if (words == NULL) {
209 return NGX_ERROR;
210 }
211
212 tok.p = buf;
213 tok.end = buf + fsize;
214 tok.pool = pc->tabs->pool;
215
216 do {
217 char *w;
218 tok_rc = acc_tok_next(&tok, &w);
219
220 if (tok_rc == ACC_TOK_WORD) {
221 char **slot = ngx_array_push(words);
222 if (slot == NULL) { return NGX_ERROR; }
223 *slot = w;
224 continue;
225 }
226
227 /* End of record or EOF: dispatch any accumulated words. */
228 if (words->nelts > 0) {
229 if (acc_dispatch_record(pc, (char **) words->elts, words->nelts)
230 != NGX_OK)
231 {
232 return NGX_ERROR;
233 }
234 words->nelts = 0;
235 }
236 } while (tok_rc != ACC_TOK_EOF);
237
238 return NGX_OK;
239}

241brix_acc_tables_t *
242brix_acc_authfile_parse(ngx_log_t *log, const char *file,
243 char spacechar, ngx_int_t uri_decode)
244{
245 brix_acc_tables_t *tabs;
246 ngx_pool_t *pool;
247 acc_parse_ctx_t pc;
248 acc_loaded_t ld = { NULL, 0, 0 };
249 int load;
250
251 pool = ngx_create_pool(16 * 1024, log);
252 if (pool == NULL) {
253 return NULL;
254 }
255 tabs = ngx_pcalloc(pool, sizeof(*tabs));
256 if (tabs == NULL) {
257 ngx_destroy_pool(pool);
258 return NULL;
259 }
260
261 load = acc_load_authfile(log, file, pool, &ld);
262 if (load == ACC_LOAD_ERROR) {
263 ngx_destroy_pool(pool);
264 return NULL;
265 }
266
267 tabs->pool = pool;
268 tabs->mtime = ld.mtime;
269 tabs->parse_spacechar = spacechar;
270 tabs->parse_uridecode = uri_decode;
271
272 if (load == ACC_LOAD_EMPTY) {
273 return tabs; /* empty authdb = deny-all, valid */
274 }
275
276 pc.tabs = tabs;
277 pc.id_tail = NULL;
278 pc.excl_seq = 0;
279 pc.log = log;
280
281 if (acc_tokenize_and_dispatch(&pc, ld.buf, ld.fsize) != NGX_OK) {
282 ngx_destroy_pool(pool);
283 return NULL;
284 }
285
286 acc_finalize_rules(tabs, log);
287 return tabs;
288}
