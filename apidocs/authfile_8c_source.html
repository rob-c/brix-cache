1/*
2 * authfile.c — XrdAcc authorization database parser (XrdAccAuthFile/ConfigDB).
3 *
4 * WHAT: brix_acc_authfile_parse() reads a stock XRootD `authdb` file and
5 * builds a complete brix_acc_tables_t generation: the per-category name
6 * lists (users/groups/hosts/netgroups/orgs/roles/templates), the default
7 * (`u *`) and fungible (`u =`) lists, the domain (`h .suffix`) list, and the
8 * compound-identity (`=`) definitions wired to their exclusive (`x`) or
9 * inclusive (`s`) rule capabilities.
10 *
11 * WHY: faithful port of XrdAccConfig::ConfigDBrec()/idDef() plus the
12 * XrdAccAuthFile tokenizer, so a site can point `brix_authdb_format xrdacc`
13 * at an existing XRootD authdb and get identical structure.
14 *
15 * HOW: the whole file (<= 1 MiB) is tokenized into records — one logical line
16 * each, with `\`-at-end-of-line continuation and `#` comments — then each
17 * record (`<type> <name> <pairs...>`) is dispatched by its type letter. All
18 * storage comes from the generation's own pool for a one-shot atomic free.
19 */
20

38static char *
39acc_pstrdup(ngx_pool_t *pool, const u_char *s, size_t len)
40{
41 char *d = ngx_pnalloc(pool, len + 1);
42 if (d == NULL) {
43 return NULL;
44 }
45 ngx_memcpy(d, s, len);
46 d[len] = '\0';
47 return d;

50/*
51 * acc_tok_next — return the next word (ACC_TOK_WORD, *word set), an end-of-record
52 * marker (ACC_TOK_EOR, at a non-continued newline), or ACC_TOK_EOF. Whitespace
53 * separates words; `\` immediately before a newline continues the record; `#`
54 * starts a comment to end of line.
55 */
56static int
57acc_tok_next(acc_tok_t *t, char **word)
58{
59 u_char *start;
60
61 for (;;) {
62 if (t->p >= t->end) {
63 return ACC_TOK_EOF;
64 }
65 switch (*t->p) {
66 case ' ': case '\t': case '\r':
67 t->p++;
68 continue;
69 case '\\':
70 /* Backslash at end of line continues the record (join lines). */
71 if (t->p + 1 < t->end && t->p[1] == '\n') {
72 t->p += 2;
73 continue;
74 }
75 if (t->p + 2 < t->end && t->p[1] == '\r' && t->p[2] == '\n') {
76 t->p += 3;
77 continue;
78 }
79 break; /* otherwise a literal '\' begins a word (path escape) */
80 case '#':
81 while (t->p < t->end && *t->p != '\n') {
82 t->p++;
83 }
84 continue;
85 case '\n':
86 t->p++;
87 return ACC_TOK_EOR;
88 default:
89 break;
90 }
91 break; /* a word starts here */
92 }
93
94 start = t->p;
95 while (t->p < t->end
96 && *t->p != ' ' && *t->p != '\t'
97 && *t->p != '\n' && *t->p != '\r')
98 {
99 t->p++;
100 }
101
102 *word = acc_pstrdup(t->pool, start, (size_t) (t->p - start));
103 return (*word == NULL) ? ACC_TOK_EOF : ACC_TOK_WORD;
104}

107/*
108 * acc_subspace — in-place replace every `sc` with a real space in an identity
109 * name, so a name containing spaces can be written without quoting (ports
110 * XrdAccConfig::subSpace). No-op when sc is 0. Applied to g/o/r/u names only.
111 */
112static void
113acc_subspace(char *id, char sc)
114{
115 if (sc == '\0' || id == NULL) {
116 return;
117 }
118 for (; *id != '\0'; id++) {
119 if (*id == sc) {
120 *id = ' ';
121 }
122 }
123}

125/*
126 * acc_uri_decode — return a pool copy of `s` with %XX escapes decoded (ports
127 * the XrdOucUri::Decode applied to path tokens under `encoding pct path`). A
128 * malformed escape is left literal. Returns `s` unchanged on alloc failure.
129 */
130static const char *
131acc_uri_decode(ngx_pool_t *pool, const char *s)
132{
133 size_t n = ngx_strlen(s);
134 char *out = ngx_pnalloc(pool, n + 1);
135 char *w;
136 size_t i;
137
138 if (out == NULL) {
139 return s;
140 }
141 for (w = out, i = 0; i < n; i++) {
142 if (s[i] == '%' && i + 2 < n) {
143 ngx_int_t hi = ngx_hextoi((u_char *) &s[i + 1], 2);
144 if (hi >= 0) { /* both nibbles were hex digits */
145 *w++ = (char) hi;
146 i += 2;
147 continue;
148 }
149 }
150 *w++ = s[i];
151 }
152 *w = '\0';
153 return out;
154}

173/* Build a capability node for a concrete path + privilege string. */
174static brix_acc_cap_t *
175acc_cap_path(ngx_pool_t *pool, const char *path, const char *privs,
176 ngx_log_t *log)
177{
178 brix_acc_cap_t *cap;
179 brix_acc_priv_caps_t pc;
180 int i;
181
182 if (brix_acc_parse_privs(privs, ngx_strlen(privs), &pc) != 0) {
183 BRIX_DIAG_EMERG(log, 0,
184 "xrootd authdb: invalid privileges \"%s\"",
185 "a privilege token contains an unknown letter",
186 "valid privilege letters are a(ll) d(elete) i(nsert) k(lock) "
187 "l(ookup) n(rename) r(ead) w(rite), optionally prefixed '-' to "
188 "deny; fix this record in the authdb file",
189 privs);
190 return NULL;
191 }
192
193 BRIX_PCALLOC_OR_RETURN(cap, pool, sizeof(*cap), NULL);
194 cap->path = (char *) path;
195 cap->plen = (int) ngx_strlen(path);
196 cap->caps = pc;
197
198 /* Locate a "@=" template-substitution point. */
199 for (i = 0; i < cap->plen; i++) {
200 if (cap->path[i] == '@' && cap->path[i + 1] == '=') {
201 cap->pins = i;
202 cap->prem = cap->plen - i - 2;
203 break;
204 }
205 }
206 return cap;
207}

209/*
210 * acc_build_caps — parse the trailing `<path|template> <priv>` pairs of a record
211 * (words[start..n]) into a capability list. Returns NGX_OK and sets *out, or
212 * NGX_ERROR (logged). Mirrors the getPP loop in ConfigDBrec.
213 */
214static ngx_int_t
215acc_build_caps(ngx_pool_t *pool, brix_acc_tables_t *tabs,
216 char **words, ngx_uint_t start, ngx_uint_t n,
217 brix_acc_cap_t **out, ngx_log_t *log)
218{
219 brix_acc_cap_t *head = NULL, *tail = NULL, *cap;
220 ngx_uint_t i = start;
221
222 while (i < n) {
223 char *path = words[i++];
224 int istmplt;
225
226 if (path[0] == '\\') {
227 if (path[1] == '\0') {
228 ngx_log_error(NGX_LOG_EMERG, log, 0,
229 "xrootd authdb: object id missing after '\\'");
230 return NGX_ERROR;
231 }
232 path++; /* escaped: a real path, '\' stripped */
233 istmplt = 0;
234 } else {
235 istmplt = (path[0] != '/');
236 }
237
238 if (istmplt) {
239 /* Template indirection: reference a previously-defined `t` list. */
240 brix_acc_cap_t *tcaps = brix_acc_named_find(tabs->t_list, path);
241 if (tcaps == NULL) {
242 ngx_log_error(NGX_LOG_EMERG, log, 0,
243 "xrootd authdb: missing template \"%s\"", path);
244 return NGX_ERROR;
245 }
246 BRIX_PCALLOC_OR_RETURN(cap, pool, sizeof(*cap), NGX_ERROR);
247 cap->tmpl = tcaps;
248 } else {
249 if (i >= n) {
250 ngx_log_error(NGX_LOG_EMERG, log, 0,
251 "xrootd authdb: missing privs for path \"%s\"",
252 path);
253 return NGX_ERROR;
254 }
255 /* encoding pct path: URI-decode the path token before storing. */
256 if (tabs->parse_uridecode) {
257 path = (char *) acc_uri_decode(pool, path);
258 }
259 cap = acc_cap_path(pool, path, words[i++], log);
260 if (cap == NULL) {
261 return NGX_ERROR;
262 }
263 }
264
265 if (tail == NULL) {
266 head = tail = cap;
267 } else {
268 tail->next = cap;
269 tail = cap;
270 }
271 }
272
273 if (head == NULL) {
274 return NGX_ERROR; /* no capabilities specified */
275 }
276 *out = head;
277 return NGX_OK;
278}

287/*
288 * acc_iddef_set — bind one `=` selector value into its slot, rejecting a
289 * duplicate selector and applying spacechar to identity values (all but host).
290 * Returns NGX_OK or NGX_ERROR (logged) — replaces the former `goto dup`.
291 */
292static ngx_int_t
293acc_iddef_set(char **slot, char *val, char sel, const char *defname,
294 char spacechar, ngx_log_t *log)
295{
296 if (*slot != NULL) {
297 ngx_log_error(NGX_LOG_EMERG, log, 0,
298 "xrootd authdb: id selector '%c' twice for %s",
299 sel, defname);
300 return NGX_ERROR;
301 }
302 if (sel != 'h') { /* spacechar: identity names only, not host */
303 acc_subspace(val, spacechar);
304 }
305 *slot = val;
306 return NGX_OK;
307}

309/* `=` record: define a compound identity (selectors only, caps filled by x/s). */
310static ngx_int_t
311acc_record_iddef(brix_acc_tables_t *tabs, char **w, ngx_uint_t n,
312 brix_acc_idrule_t **id_tail, ngx_log_t *log)
313{
314 brix_acc_idrule_t *id;
315 ngx_uint_t i;
316
317 if ((n - 2) % 2 != 0 || n < 4) {
318 ngx_log_error(NGX_LOG_EMERG, log, 0,
319 "xrootd authdb: `= %s` needs <selector value> pairs",
320 w[1]);
321 return NGX_ERROR;
322 }
323 for (id = tabs->id_defs; id != NULL; id = id->next) {
324 if (ngx_strcmp(id->name, w[1]) == 0) {
325 ngx_log_error(NGX_LOG_EMERG, log, 0,
326 "xrootd authdb: duplicate id definition \"%s\"", w[1]);
327 return NGX_ERROR;
328 }
329 }
330
331 BRIX_PCALLOC_OR_RETURN(id, tabs->pool, sizeof(*id), NGX_ERROR);
332 id->name = w[1];
333 id->rule = INT_MIN; /* "no x/s rule attached yet" */
334
335 for (i = 2; i + 1 < n; i += 2) {
336 char sel = w[i][0];
337 char *val = w[i + 1];
338 char sc = tabs->parse_spacechar;
339 ngx_int_t rc;
340
341 if (w[i][1] != '\0' || !acc_selector_ok(sel)) {
342 ngx_log_error(NGX_LOG_EMERG, log, 0,
343 "xrootd authdb: invalid id selector \"%s\" for %s",
344 w[i], w[1]);
345 return NGX_ERROR;
346 }
347 switch (sel) {
348 case 'g': rc = acc_iddef_set(&id->grp, val, sel, w[1], sc, log); break;
349 case 'h': rc = acc_iddef_set(&id->host, val, sel, w[1], sc, log);
350 if (rc == NGX_OK) { id->hlen = (int) ngx_strlen(val); }
351 break;
352 case 'o': rc = acc_iddef_set(&id->org, val, sel, w[1], sc, log); break;
353 case 'r': rc = acc_iddef_set(&id->role, val, sel, w[1], sc, log); break;
354 default: rc = acc_iddef_set(&id->user, val, sel, w[1], sc, log); break;
355 }
356 if (rc != NGX_OK) {
357 return NGX_ERROR;
358 }
359 }
360
361 /* Append in file order. */
362 if (*id_tail == NULL) {
363 tabs->id_defs = id;
364 } else {
365 (*id_tail)->next = id;
366 }
367 *id_tail = id;
368 return NGX_OK;
369}

371/* `x`/`s` record: attach capabilities (and exclusive/inclusive flag) to a def. */
372static ngx_int_t
373acc_record_rule(brix_acc_tables_t *tabs, char **w, ngx_uint_t n,
374 int exclusive, int *excl_seq, ngx_log_t *log)
375{
376 brix_acc_idrule_t *id;
377
378 for (id = tabs->id_defs; id != NULL; id = id->next) {
379 if (ngx_strcmp(id->name, w[1]) == 0) {
380 break;
381 }
382 }
383 if (id == NULL) {
384 ngx_log_error(NGX_LOG_EMERG, log, 0,
385 "xrootd authdb: missing id definition \"%s\"", w[1]);
386 return NGX_ERROR;
387 }
388 if (id->caps != NULL) {
389 ngx_log_error(NGX_LOG_EMERG, log, 0,
390 "xrootd authdb: duplicate rule for id \"%s\"", w[1]);
391 return NGX_ERROR;
392 }
393 if (acc_build_caps(tabs->pool, tabs, w, 2, n, &id->caps, log) != NGX_OK) {
394 return NGX_ERROR;
395 }
396 id->rule = exclusive ? (*excl_seq)++ : -1;
397 return NGX_OK;
398}

400/* g/h/n/o/r/t/u record: bind a name to a capability list. */
401static ngx_int_t
402acc_record_named(brix_acc_tables_t *tabs, char rtype, char **w, ngx_uint_t n,
403 ngx_log_t *log)
404{
405 brix_acc_named_t **head = NULL;
406 brix_acc_cap_t *caps;
407 char *name = w[1];
408 int alluser = 0, anyuser = 0, domain = 0;
409
410 switch (rtype) {
411 case 'g': head = &tabs->g_list; break;
412 case 'n': head = &tabs->n_list; break;
413 case 'o': head = &tabs->o_list; break;
414 case 'r': head = &tabs->r_list; break;
415 case 't': head = &tabs->t_list; break;
416 case 'h':
417 domain = (name[0] == '.');
418 head = domain ? &tabs->d_list : &tabs->h_list;
419 break;
420 case 'u':
421 alluser = (name[0] == '*' && name[1] == '\0');
422 anyuser = (name[0] == '=' && name[1] == '\0');
423 head = &tabs->u_list;
424 break;
425 default:
426 return NGX_ERROR;
427 }
428
429 /* spacechar: substitute in identity names only (g/o/r/u), never the `u *`/
430 * `u =` wildcards, host, netgroup or template names (XrdAcc ConfigDBrec). */
431 if (rtype == 'g' || rtype == 'o' || rtype == 'r'
432 || (rtype == 'u' && !alluser && !anyuser))
433 {
434 acc_subspace(name, tabs->parse_spacechar);
435 }
436
437 /* Duplicate detection. */
438 if ((alluser && tabs->z_list != NULL)
439 || (anyuser && tabs->x_list != NULL)
440 || (!alluser && !anyuser
441 && brix_acc_named_find(*head, name) != NULL))
442 {
443 ngx_log_error(NGX_LOG_EMERG, log, 0,
444 "xrootd authdb: duplicate rule for id \"%s\"", name);
445 return NGX_ERROR;
446 }
447
448 if (acc_build_caps(tabs->pool, tabs, w, 2, n, &caps, log) != NGX_OK) {
449 ngx_log_error(NGX_LOG_EMERG, log, 0,
450 "xrootd authdb: no capabilities for \"%s\"", name);
451 return NGX_ERROR;
452 }
453
454 if (alluser) {
455 tabs->z_list = caps;
456 } else if (anyuser) {
457 tabs->x_list = caps;
458 } else if (acc_named_prepend(tabs->pool, head, name, caps) == NULL) {
459 return NGX_ERROR;
460 }
461 return NGX_OK;
462}

465static void
466acc_finalize_rules(brix_acc_tables_t *tabs, ngx_log_t *log)
467{
468 brix_acc_idrule_t *id, *next;
469 brix_acc_idrule_t *sx_tail = NULL, *sy_tail = NULL;
470
471 for (id = tabs->id_defs; id != NULL; id = next) {
472 next = id->next;
473 id->next = NULL;
474
475 if (id->caps == NULL) {
476 ngx_log_error(NGX_LOG_WARN, log, 0,
477 "xrootd authdb: id \"%s\" defined but never used "
478 "(no x/s rule) — ignored", id->name);
479 continue;
480 }
481 if (id->rule >= 0) { /* exclusive, file order */
482 if (sx_tail == NULL) {
483 tabs->sx_list = id;
484 } else {
485 sx_tail->next = id;
486 }
487 sx_tail = id;
488 } else { /* inclusive */
489 if (sy_tail == NULL) {
490 tabs->sy_list = id;
491 } else {
492 sy_tail->next = id;
493 }
494 sy_tail = id;
495 }
496 }
497}

500static ngx_int_t
501acc_dispatch_record(brix_acc_tables_t *tabs, char **w, ngx_uint_t n,
502 brix_acc_idrule_t **id_tail, int *excl_seq, ngx_log_t *log)
503{
504 char rtype;
505
506 if (w[0][0] == '\0' || w[0][1] != '\0') {
507 BRIX_DIAG_EMERG(log, 0,
508 "xrootd authdb: invalid record type \"%s\"",
509 "each record must begin with a single-letter type in column 1",
510 "valid record types are = x s g h n o r t u; fix this line in "
511 "the authdb file",
512 w[0]);
513 return NGX_ERROR;
514 }
515 rtype = w[0][0];
516
517 if (n < 2) {
518 ngx_log_error(NGX_LOG_EMERG, log, 0,
519 "xrootd authdb: record name missing after '%c'", rtype);
520 return NGX_ERROR;
521 }
522
523 tabs->rule_count++;
524
525 switch (rtype) {
526 case '=': return acc_record_iddef(tabs, w, n, id_tail, log);
527 case 'x': return acc_record_rule(tabs, w, n, 1, excl_seq, log);
528 case 's': return acc_record_rule(tabs, w, n, 0, excl_seq, log);
529 case 'g': case 'h': case 'n': case 'o':
530 case 'r': case 't': case 'u':
531 return acc_record_named(tabs, rtype, w, n, log);
532 default:
533 ngx_log_error(NGX_LOG_EMERG, log, 0,
534 "xrootd authdb: invalid id type \"%c\"", rtype);
535 return NGX_ERROR;
536 }
537}

539brix_acc_tables_t *
540brix_acc_authfile_parse(ngx_log_t *log, const char *file,
541 char spacechar, ngx_int_t uri_decode)
542{
543 brix_acc_tables_t *tabs;
544 brix_acc_idrule_t *id_tail = NULL;
545 ngx_pool_t *pool;
546 ngx_fd_t fd;
547 ngx_file_info_t fi;
548 u_char *buf;
549 ssize_t nread;
550 size_t fsize;
551 acc_tok_t tok;
552 ngx_array_t *words; /* of char* for the current record */
553 int excl_seq = 0, rc, tok_rc;
554 ngx_int_t status = NGX_OK;
555
556 fd = ngx_open_file((u_char *) file, NGX_FILE_RDONLY, NGX_FILE_OPEN, 0);
557 if (fd == NGX_INVALID_FILE) {
558 BRIX_DIAG_EMERG(log, ngx_errno,
559 "xrootd authdb: cannot open \"%s\"",
560 "the path in brix_acc_authdb is wrong or unreadable by the "
561 "nginx user",
562 "check the path exists and the nginx master/worker user can read "
563 "it (the OS reason is appended below)",
564 file);
565 return NULL;
566 }
567 if (ngx_fd_info(fd, &fi) == NGX_FILE_ERROR) {
568 ngx_close_file(fd);
569 return NULL;
570 }
571 fsize = (size_t) ngx_file_size(&fi);
572 if (fsize > BRIX_ACC_AUTHDB_MAX) {
573 BRIX_DIAG_EMERG(log, 0,
574 "xrootd authdb \"%s\" exceeds the 1 MiB limit",
575 "the file is larger than the parser accepts — usually a wrong "
576 "path (pointing at a data file) or a runaway generated authdb",
577 "confirm brix_acc_authdb points at the authorization file, not "
578 "something else; split or trim it below 1 MiB",
579 file);
580 ngx_close_file(fd);
581 return NULL;
582 }
583
584 pool = ngx_create_pool(16 * 1024, log);
585 if (pool == NULL) {
586 ngx_close_file(fd);
587 return NULL;
588 }
589 tabs = ngx_pcalloc(pool, sizeof(*tabs));
590 if (tabs == NULL) {
591 ngx_destroy_pool(pool);
592 ngx_close_file(fd);
593 return NULL;
594 }
595 tabs->pool = pool;
596 tabs->mtime = ngx_file_mtime(&fi);
597 tabs->parse_spacechar = spacechar;
598 tabs->parse_uridecode = uri_decode;
599
600 if (fsize == 0) {
601 ngx_close_file(fd);
602 return tabs; /* empty authdb = deny-all, valid */
603 }
604
605 buf = ngx_pnalloc(pool, fsize);
606 if (buf == NULL) {
607 ngx_destroy_pool(pool);
608 ngx_close_file(fd);
609 return NULL;
610 }
611 nread = ngx_read_fd(fd, buf, fsize);
612 ngx_close_file(fd);
613 if (nread < 0 || (size_t) nread != fsize) {
614 BRIX_DIAG_EMERG(log, 0,
615 "xrootd authdb: short read of \"%s\"",
616 "the file changed size or an I/O error occurred while reading it",
617 "make sure the authdb is written atomically (write-then-rename), "
618 "then re-run nginx -t",
619 file);
620 ngx_destroy_pool(pool);
621 return NULL;
622 }
623
624 words = ngx_array_create(pool, 16, sizeof(char *));
625 if (words == NULL) {
626 ngx_destroy_pool(pool);
627 return NULL;
628 }
629
630 tok.p = buf;
631 tok.end = buf + fsize;
632 tok.pool = pool;
633
634 do {
635 char *w;
636 tok_rc = acc_tok_next(&tok, &w);
637
638 if (tok_rc == ACC_TOK_WORD) {
639 char **slot = ngx_array_push(words);
640 if (slot == NULL) { status = NGX_ERROR; break; }
641 *slot = w;
642 continue;
643 }
644
645 /* End of record or EOF: dispatch any accumulated words. */
646 if (words->nelts > 0) {
647 rc = acc_dispatch_record(tabs, (char **) words->elts, words->nelts,
648 &id_tail, &excl_seq, log);
649 if (rc != NGX_OK) { status = NGX_ERROR; break; }
650 words->nelts = 0;
651 }
652 } while (tok_rc != ACC_TOK_EOF);
653
654 if (status != NGX_OK) {
655 ngx_destroy_pool(pool);
656 return NULL;
657 }
658
659 acc_finalize_rules(tabs, log);
660 return tabs;
661}
