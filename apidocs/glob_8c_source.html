1/*
2 * glob.c — client-side wildcard expansion for root:// URLs.
3 *
4 * WHAT: brix_glob() expands a root:// URL whose final path component contains glob
5 * metacharacters (* ? [) into the list of matching full URLs.
6 * WHY: The XRootD wire protocol has no server-side globbing for transfer/metadata
7 * ops, so a "swiss-army-knife" client must expand patterns itself and hide
8 * that from the user — a quoted wildcard source just works.
9 * HOW: Parse the URL, split the path into <dir>/<pattern>, open one connection,
10 * dirlist <dir>, fnmatch each entry against <pattern> (FNM_PERIOD so '*' does
11 * not match dotfiles, mirroring the shell), and rebuild a full URL per match.
12 *
13 * Scope: single-level (the wildcard must be in the LAST path component); multi-level
14 * patterns (a wildcard in a non-final component) are not expanded. Clean-room.
15 */
16#include "brix.h"
17
18#include <fnmatch.h>
19#include <stdlib.h>
20#include <string.h>

25{
26 return s != NULL && strpbrk(s, "*?[") != NULL;

29int
30brix_glob(const char *url, const brix_opts *co, char ***out, size_t *n_out,
31 brix_status *st)
32{
33 brix_url u;
34 brix_conn c;
35 brix_dirent *ents = NULL;
36 char **arr = NULL;
37 char dir[XRDC_PATH_MAX];
38 char pattern[XRDC_NAME_MAX];
39 const char *scheme, *slash, *dsep;
40 size_t ne = 0, cap = 0, n = 0, i, dlen;
41
42 *out = NULL;
43 *n_out = 0;
44 if (brix_url_parse(url, &u, st) != 0) {
45 return -1;
46 }
47 if (u.scheme != XRDC_SCHEME_ROOT && u.scheme != XRDC_SCHEME_ROOTS) {
48 brix_status_set(st, XRDC_EUSAGE, 0, "glob: only root:// URLs are expanded");
49 return -1;
50 }
51 slash = strrchr(u.path, '/');
52 if (slash == NULL) {
53 brix_status_set(st, XRDC_EUSAGE, 0, "glob: malformed path");
54 return -1;
55 }
56 if (strlen(slash + 1) >= sizeof(pattern)) {
57 brix_status_set(st, XRDC_EUSAGE, 0, "glob: pattern too long");
58 return -1; /* reject — never truncate (a clipped wildcard mis-matches) */
59 }
60 memcpy(pattern, slash + 1, strlen(slash + 1) + 1);
61 if (!brix_has_glob(pattern)) {
62 brix_status_set(st, XRDC_EUSAGE, 0, "glob: no wildcard in last path component");
63 return -1;
64 }
65 dlen = (size_t) (slash - u.path);
66 if (dlen == 0) {
67 dir[0] = '/';
68 dir[1] = '\0';
69 } else {
70 if (dlen >= sizeof(dir)) {
71 brix_status_set(st, XRDC_EUSAGE, 0, "glob: directory path too long");
72 return -1;
73 }
74 memcpy(dir, u.path, dlen);
75 dir[dlen] = '\0';
76 }
77
78 if (brix_connect(&c, &u, co, st) != 0) {
79 return -1;
80 }
81 if (brix_dirlist(&c, dir, 0 /*no stat*/, &ents, &ne, st) != 0) {
82 brix_close(&c);
83 return -1;
84 }
85 scheme = (u.scheme == XRDC_SCHEME_ROOTS) ? "roots" : "root";
86 dsep = (dir[0] != '\0' && dir[strlen(dir) - 1] == '/') ? "" : "/";
87 {
88 int v6 = (strchr(u.host, ':') != NULL); /* IPv6 literal */
89 const char *hb = v6 ? "[" : "";
90 const char *he = v6 ? "]" : "";
91 int err = 0;
92 for (i = 0; i < ne && !err; i++) {
93 char full[XRDC_PATH_MAX + 320];
94 int wn;
95 /* dirlist entries are single path components; reject anything else so a
96 * hostile/odd name can't yield a traversal-shaped URL. */
97 if (strchr(ents[i].name, '/') != NULL
98 || strcmp(ents[i].name, ".") == 0 || strcmp(ents[i].name, "..") == 0) {
99 continue;
100 }
101 if (fnmatch(pattern, ents[i].name, FNM_PERIOD | FNM_PATHNAME) != 0) {
102 continue;
103 }
104 /* Re-bracket an IPv6 host so the rebuilt URL round-trips. */
105 wn = snprintf(full, sizeof(full), "%s://%s%s%s:%d/%s%s%s",
106 scheme, hb, u.host, he, u.port, dir, dsep, ents[i].name);
107 if (wn < 0 || (size_t) wn >= sizeof(full)) {
108 continue; /* skip a match whose rebuilt URL would truncate */
109 }
110 if (n == cap) {
111 size_t nc = cap ? cap * 2 : 16;
112 char **na = (char **) realloc(arr, nc * sizeof(char *));
113 if (na == NULL) { err = 1; break; }
114 arr = na;
115 cap = nc;
116 }
117 arr[n] = strdup(full);
118 if (arr[n] == NULL) { err = 1; break; }
119 n++;
120 }
121 free(ents);
122 brix_close(&c);
123 if (err) {
124 brix_glob_free(arr, n);
125 brix_status_set(st, XRDC_EPROTO, 0, "glob: out of memory");
126 return -1;
127 }
128 }
129 *out = arr;
130 *n_out = n;
131 return (int) n;
132}
