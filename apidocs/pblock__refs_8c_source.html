1/*
2 * pblock_refs.c — F10 refcounted blobs + content-addressed dedup for pblock.
3 *
4 * WHAT: Implements pblock_refs.h: the `blobs` refcount table, reference
5 * bookkeeping (track/bump/count/release), publish-time dedup with a
6 * mandatory byte-verify, and the copy-on-write share-break.
7 *
8 * HOW: Row absence = implicit refcount 1 (legacy blobs), so every reader
9 * treats "no row" and "refcount 1" identically. A hash match is never
10 * trusted on its own: dedup byte-compares both blobs through the block
11 * engine before linking. Release fails CLOSED on a DB error — blocks are
12 * kept (an fsck-collectable orphan at worst), never removed while a
13 * sibling row may still reference them. ngx-free (libc + sqlite3);
14 * BRIX_HAVE_SQLITE-gated.
15 */

56int
57pblock_refs_track(const pblock_state_t *st, const char *blob_id,
58 int64_t size, int64_t bs, uint32_t crc, int crc_ok)
59{
60 sqlite3_stmt *q;
61 char hash[24];
62 int rc;
63
64 if (crc_ok) {
65 snprintf(hash, sizeof(hash), "crc32:%08x", crc);
66 } else {
67 hash[0] = '\0';
68 }
69 q = cat_prepare(st->cat,
70 "INSERT INTO blobs(blob_id, refcount, size, block_size, content_hash)"
71 " VALUES(?1, 1, ?2, ?3, ?4)"
72 " ON CONFLICT(blob_id) DO UPDATE SET"
73 " size = excluded.size, block_size = excluded.block_size,"
74 " content_hash = excluded.content_hash;");
75 if (q == NULL) {
76 return -1;
77 }
78 sqlite3_bind_text(q, 1, blob_id, -1, SQLITE_STATIC);
79 sqlite3_bind_int64(q, 2, size);
80 sqlite3_bind_int64(q, 3, bs);
81 sqlite3_bind_text(q, 4, hash, -1, SQLITE_STATIC);
82 rc = sqlite3_step(q);
83 sqlite3_finalize(q);
84 return rc == SQLITE_DONE ? 0 : cat_fail(EIO);
85}

87int
88pblock_refs_bump(const pblock_state_t *st, const char *blob_id,
89 int64_t size, int64_t bs)
90{
91 sqlite3_stmt *q;
92 int rc;
93
94 /* A missing row carries the implicit legacy reference — created here as 2:
95 * that reference plus the new one. */
96 q = cat_prepare(st->cat,
97 "INSERT INTO blobs(blob_id, refcount, size, block_size, content_hash)"
98 " VALUES(?1, 2, ?2, ?3, '')"
99 " ON CONFLICT(blob_id) DO UPDATE SET refcount = refcount + 1;");
100 if (q == NULL) {
101 return -1;
102 }
103 sqlite3_bind_text(q, 1, blob_id, -1, SQLITE_STATIC);
104 sqlite3_bind_int64(q, 2, size);
105 sqlite3_bind_int64(q, 3, bs);
106 rc = sqlite3_step(q);
107 sqlite3_finalize(q);
108 return rc == SQLITE_DONE ? 0 : cat_fail(EIO);
109}

111int
112pblock_refs_count(const pblock_state_t *st, const char *blob_id)
113{
114 sqlite3_stmt *q;
115 int n = 1; /* absent row = implicit single ref */
116 int rc;
117
118 q = cat_prepare(st->cat,
119 "SELECT refcount FROM blobs WHERE blob_id = ?1;");
120 if (q == NULL) {
121 return -1;
122 }
123 sqlite3_bind_text(q, 1, blob_id, -1, SQLITE_STATIC);
124 rc = sqlite3_step(q);
125 if (rc == SQLITE_ROW) {
126 n = sqlite3_column_int(q, 0);
127 if (n < 1) {
128 n = 1;
129 }
130 } else if (rc != SQLITE_DONE) {
131 sqlite3_finalize(q);
132 return cat_fail(EIO);
133 }
134 sqlite3_finalize(q);
135 return n;
136}

164void
165pblock_refs_release(const pblock_state_t *st, const char *blob_id,
166 int64_t size, int64_t bs)
167{
168 int n;
169
170 if (!st->refs) { /* gate off: the pre-F10 removal */
171 refs_remove(st, blob_id, size, bs);
172 return;
173 }
174 n = pblock_refs_count(st, blob_id);
175 if (n < 0) {
176 return; /* fail closed: keep the blocks — an
177 * fsck orphan beats removing bytes a
178 * sibling row may still reference */
179 }
180 if (n <= 1) {
181 refs_drop_row(st, blob_id);
182 refs_remove(st, blob_id, size, bs);
183 return;
184 }
185 {
186 sqlite3_stmt *q = cat_prepare(st->cat,
187 "UPDATE blobs SET refcount = refcount - 1 WHERE blob_id = ?1;");
188
189 if (q != NULL) {
190 sqlite3_bind_text(q, 1, blob_id, -1, SQLITE_STATIC);
191 (void) sqlite3_step(q);
192 sqlite3_finalize(q);
193 }
194 }
195}

197/* refs_repoint — point the object row for `path` at a different blob. The
198 * nscache holds the old row, so it must be invalidated. 0 or -1/errno. */
199static int
200refs_repoint(pblock_state_t *st, const char *path, const char *blob_id)
201{
202 sqlite3_stmt *q;
203 int rc;
204
205 q = cat_prepare(st->cat,
206 "UPDATE objects SET blob_id = ?2 WHERE path = ?1;");
207 if (q == NULL) {
208 return -1;
209 }
210 sqlite3_bind_text(q, 1, path, -1, SQLITE_STATIC);
211 sqlite3_bind_text(q, 2, blob_id, -1, SQLITE_STATIC);
212 rc = sqlite3_step(q);
213 sqlite3_finalize(q);
214 if (rc != SQLITE_DONE) {
215 return cat_fail(EIO);
216 }
217 nscache_inval(st->cat, path);
218 return 0;
219}

221/* refs_same_bytes — full byte compare of two blobs over [0, size) through the
222 * block engine (transient block fds). 1 identical, 0 different, -1/errno. */
223static int
224refs_same_bytes(const pblock_state_t *st, const char *a, const char *b,
225 int64_t size, int64_t bs)
226{
227 char ba[32768], bb[32768];
228 int64_t off = 0;
229
230 while (off < size) {
231 size_t chunk = (size_t) (size - off) < sizeof(ba)
232 ? (size_t) (size - off) : sizeof(ba);
233 ssize_t ra = pblock_read_blocks(st, a, bs, -1, ba, chunk, (off_t) off);
234 ssize_t rb = pblock_read_blocks(st, b, bs, -1, bb, chunk, (off_t) off);
235
236 if (ra < 0 || rb < 0) {
237 return -1;
238 }
239 if (ra != rb || memcmp(ba, bb, (size_t) ra) != 0) {
240 return 0;
241 }
242 if (ra == 0) { /* both short (hole tail): equal so far */
243 break;
244 }
245 off += ra;
246 }
247 return 1;
248}

250int
251pblock_refs_dedup_publish(pblock_state_t *st, const char *path,
252 pblock_meta *meta, uint32_t crc, int crc_ok)
253{
254 sqlite3_stmt *q;
255 char hash[24];
256 char match[PBLOCK_BLOB_ID_CAP];
257 int found = 0;
258
259 /* No trustworthy whole-object hash, an empty object, or a blob that is
260 * already shared (folding it would strand the sibling): just (re)track. */
261 if (!crc_ok || meta->size <= 0
262 || pblock_refs_count(st, meta->blob_id) != 1)
263 {
264 return pblock_refs_track(st, meta->blob_id, meta->size,
265 meta->block_size, crc, crc_ok);
266 }
267
268 snprintf(hash, sizeof(hash), "crc32:%08x", crc);
269 q = cat_prepare(st->cat,
270 "SELECT blob_id FROM blobs"
271 " WHERE content_hash = ?1 AND size = ?2 AND block_size = ?3"
272 " AND blob_id != ?4 AND refcount >= 1;");
273 if (q == NULL) {
274 return pblock_refs_track(st, meta->blob_id, meta->size,
275 meta->block_size, crc, crc_ok);
276 }
277 sqlite3_bind_text(q, 1, hash, -1, SQLITE_STATIC);
278 sqlite3_bind_int64(q, 2, meta->size);
279 sqlite3_bind_int64(q, 3, meta->block_size);
280 sqlite3_bind_text(q, 4, meta->blob_id, -1, SQLITE_STATIC);
281 while (!found && sqlite3_step(q) == SQLITE_ROW) {
282 const unsigned char *id = sqlite3_column_text(q, 0);
283
284 if (id == NULL) {
285 continue;
286 }
287 snprintf(match, sizeof(match), "%s", (const char *) id);
288 /* The hash only nominates: a CRC collision (or a forged row) must
289 * never alias content, so every candidate is byte-verified. */
290 if (refs_same_bytes(st, meta->blob_id, match, meta->size,
291 meta->block_size) == 1)
292 {
293 found = 1;
294 }
295 }
296 sqlite3_finalize(q);
297
298 if (!found) {
299 return pblock_refs_track(st, meta->blob_id, meta->size,
300 meta->block_size, crc, crc_ok);
301 }
302
303 if (pblock_refs_bump(st, match, meta->size, meta->block_size) != 0) {
304 return pblock_refs_track(st, meta->blob_id, meta->size,
305 meta->block_size, crc, crc_ok);
306 }
307 if (refs_repoint(st, path, match) != 0) {
308 pblock_refs_release(st, match, meta->size, meta->block_size);
309 return -1;
310 }
311 pblock_refs_release(st, meta->blob_id, meta->size, meta->block_size);
312 snprintf(meta->blob_id, sizeof(meta->blob_id), "%s", match);
313 return 1;
314}

316int
317pblock_refs_break_share(pblock_state_t *st, const char *path,
318 pblock_meta *meta, int trunc)
319{
320 char fresh[PBLOCK_BLOB_ID_CAP];
321 int64_t last, i;
322 int n;
323
324 n = pblock_refs_count(st, meta->blob_id);
325 if (n < 0) {
326 errno = EIO; /* can't prove the blob is private —
327 * refuse the write open, never write
328 * through a possibly-shared blob */
329 return -1;
330 }
331 if (n <= 1) {
332 return 0; /* already private */
333 }
334
335 if (pblock_gen_blob_id(fresh) != 0
336 || pblock_ensure_obj_dir(st, fresh) != 0)
337 {
338 return -1;
339 }
340
341 if (trunc) {
342 /* Content is being replaced: start from an empty block 0 (the open
343 * path expects the file to exist) instead of copying doomed bytes. */
344 char bp[PATH_MAX];
345 int fd;
346
347 if (pblock_block_path(st, fresh, 0, bp, sizeof(bp)) != 0) {
348 return -1;
349 }
350 fd = open(bp, O_RDWR | O_CREAT | O_EXCL, 0600);
351 if (fd < 0) {
352 return -1;
353 }
354 close(fd);
355 } else {
356 last = pblock_last_block(meta->size, meta->block_size);
357 for (i = 0; i <= last; i++) {
358 char sp[PATH_MAX], dp[PATH_MAX];
359
360 if (pblock_block_path(st, meta->blob_id, i, sp, sizeof(sp)) != 0
361 || pblock_block_path(st, fresh, i, dp, sizeof(dp)) != 0
362 || pblock_copy_one_block(sp, dp) < 0)
363 {
364 int err = errno;
365
366 pblock_remove_blocks(st, fresh, meta->size, meta->block_size);
367 errno = err;
368 return -1;
369 }
370 }
371 }
372
373 if (refs_repoint(st, path, fresh) != 0) {
374 int err = errno;
375
376 pblock_remove_blocks(st, fresh, meta->size, meta->block_size);
377 errno = err;
378 return -1;
379 }
380 if (st->csi && !trunc) {
381 /* Carry the at-rest CRCs to the private copy (same bytes, same tags);
382 * best-effort — missing tags read as "unset", never as corruption. */
383 sqlite3_stmt *q = cat_prepare(st->cat,
384 "INSERT OR REPLACE INTO csi(blob_id, block_no, crc)"
385 " SELECT ?2, block_no, crc FROM csi WHERE blob_id = ?1;");
386
387 if (q != NULL) {
388 sqlite3_bind_text(q, 1, meta->blob_id, -1, SQLITE_STATIC);
389 sqlite3_bind_text(q, 2, fresh, -1, SQLITE_STATIC);
390 (void) sqlite3_step(q);
391 sqlite3_finalize(q);
392 }
393 }
394 (void) pblock_refs_track(st, fresh, meta->size, meta->block_size, 0, 0);
395 pblock_refs_release(st, meta->blob_id, meta->size, meta->block_size);
396 snprintf(meta->blob_id, sizeof(meta->blob_id), "%s", fresh);
397 return 0;
398}
