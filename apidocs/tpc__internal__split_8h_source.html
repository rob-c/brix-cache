1/*
2 * tpc_internal_split.h — cross-file declarations shared between the four
3 * translation units of the WebDAV HTTP-TPC COPY handler after the phase-79
4 * file-size split.
5 *
6 * WHAT: Publishes the pull-context struct and the small set of file-local helpers
7 * that are called across the tpc.c / tpc_copy.c / tpc_push.c / tpc_pull.c
8 * boundaries: the shared request helpers (identity, session-xfer note,
9 * authorize, registry add, subject-token extract, bearer-header append)
10 * that stay in tpc.c; the COPY request parsing/credential helpers in
11 * tpc_copy.c; the push entry in tpc_push.c; and the pull prepare/accounting/
12 * execute entries in tpc_pull.c.
13 *
14 * WHY: tpc.c was one 1324-line file — far over the 500-line cap. It is split by
15 * concept into the shared helpers + COPY dispatcher (stays in tpc.c), the
16 * COPY request parsing + OAuth2/OIDC credential + user-proxy helpers
17 * (tpc_copy.c), the push handler (tpc_push.c), and the staged pull
18 * execution — target prepare, the marker/thread/sync tiers, commit and
19 * success finalisation (tpc_pull.c). Each file reaches the others only
20 * through the entry points declared here, so exactly those helpers become
21 * non-static. Nothing here is exported beyond the WebDAV TPC module — the
22 * public COPY entry remains ngx_http_brix_webdav_tpc_handle_copy in
23 * webdav_tpc.h.
24 *
25 * HOW: All four translation units include this header (in addition to webdav.h
26 * and webdav_tpc.h). No behaviour changes: the split only relocates the
27 * same functions and un-statics the ones called across the new boundary.
28 *
29 * Requires: webdav.h (request/config types), core/compat/staged_file.h
30 * (brix_staged_file_t), <sys/stat.h> (struct stat) before inclusion.
31 */

40/*
41 * WHAT: Bundle of the shared state threaded through the file-local HTTP-TPC pull
42 * exec helpers (target preparation, the three transfer tiers, commit and
43 * success finalisation).
44 * WHY: Each pull helper otherwise repeated the same long positional parameter
45 * list (request, config, source URL, staged temp file, transfer headers,
46 * stream count and the overwrite/existed flags). Collecting them into one
47 * stack-local context keeps every helper at =5 params and makes the shared
48 * pull state explicit without any global state.
49 * HOW: Zero-initialised with ngx_memzero by ngx_http_brix_webdav_tpc_handle_copy,
50 * populated once with the exact values previously passed positionally, then
51 * handed to each helper by pointer. `path`/`path_len`/`sb`/`staged` alias
52 * the caller's own stack storage (unchanged addresses); `transfer_id` is
53 * produced by the sync tier and consumed by commit/finish. Field values and
54 * read order are identical, so the pull semantics (probe, authorize, staged
55 * open, marker/thread/sync tiers, link-vs-rename commit, registry/dashboard
56 * ordering) are unchanged.
57 */
58typedef struct {
59 ngx_http_request_t *r; /* current request */
60 ngx_http_brix_webdav_loc_conf_t *conf; /* location config */
61 const char *source_url; /* remote https source */
62 char *path; /* confined local target path */
63 size_t path_len; /* capacity of `path` */
64 struct stat *sb; /* target stat (existence/size) */
65 ngx_array_t *transfer_headers; /* outbound curl headers */
66 brix_staged_file_t *staged; /* atomic-commit temp file */
67 const char *user_cert; /* per-user pull-leg cert (or NULL) */
68 const char *user_key; /* per-user pull-leg key (or NULL) */
69 uint64_t transfer_id; /* registry id (sync tier) */
70 ngx_uint_t n_streams; /* negotiated parallel streams */
71 ngx_flag_t overwrite; /* Overwrite: T/F policy */
72 ngx_flag_t existed; /* target pre-existed */
73} webdav_tpc_pull_ctx_t;
