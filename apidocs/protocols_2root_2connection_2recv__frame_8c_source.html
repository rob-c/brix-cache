11/*
12 * recv_frame.c — the READ side of the recv framing loop (see recv_frame.h):
13 * deferred-request drain, fresh-request housekeeping, non-XRootD handoff, and
14 * reading/accumulating the next PDU unit. The process side (buffer management,
15 * body extension, header/payload dispatch) lives in recv_process.c. Bodies are
16 * the original ngx_stream_brix_recv loop-body blocks moved verbatim; only
17 * loop-exit statements (break/continue/return) became step codes.
18 */
19
20brix_recv_step_t
21brix_recv_run_deferred(ngx_stream_session_t *s, ngx_connection_t *c,
22 ngx_stream_brix_srv_conf_t *conf, brix_ctx_t *ctx, ngx_event_t *rev,
23 size_t *rx_pending)
24{
25 ngx_int_t rc;
26
27 /*
28 * A non-pipelinable request was fully read (header in cur_*, any payload in
29 * payload_buf) while pipelined reads OR writes were still in flight. Keep
30 * it parked until BOTH the ack/response queue (out.count) and the in-flight
31 * pwrites (wr_inflight) have drained — a write completion or the send-side
32 * drain re-enters recv to re-check. This is what lets a kXR_close safely
33 * follow a burst of pipelined writes: every pwrite lands before the retire.
34 */
35 if (ctx->out.count > 0 || ctx->out.wr_inflight > 0) {
36 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
37 return BRIX_RECV_STEP_BREAK;
38 }
39 return BRIX_RECV_STEP_RETURN;
40 }
41
42 ctx->out.recv_deferred = 0;
43
44 /*
45 * Reinstall the parked request's stream-id: a pipelined pwrite/read ack
46 * completing during the park went through brix_aio_restore_stream(), which
47 * overwrote recv.cur_streamid with the completed op's sid. Dispatching the
48 * deferred request with that stale sid misdirects its reply onto a sid the
49 * client has already retired (and may have recycled for a newer request),
50 * while the deferred request itself never gets answered — the client's
51 * stream timeout then kills the whole multiplexed connection.
52 */
53 ctx->recv.cur_streamid[0] = ctx->out.deferred_streamid[0];
54 ctx->recv.cur_streamid[1] = ctx->out.deferred_streamid[1];
55
56 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, *rx_pending);
57 *rx_pending = 0;
58 rc = brix_dispatch(ctx, c, conf);
59 if (rc == NGX_ERROR) {
60 return BRIX_RECV_STEP_BREAK;
61 }
62 if (ctx->state == XRD_ST_AIO) {
63 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
64 return BRIX_RECV_STEP_BREAK;
65 }
66 return BRIX_RECV_STEP_RETURN;
67 }
68 if (ctx->state != XRD_ST_SENDING) {
69 if (ctx->tls_pending) {
70 brix_start_tls(ctx, c, conf);
71 return BRIX_RECV_STEP_RETURN;
72 }
73 ctx->state = XRD_ST_REQ_HEADER;
74 ctx->recv.hdr_pos = 0;
75 }
76 return BRIX_RECV_STEP_CONTINUE;
77}
78
79/*
80 * Fresh-request-boundary housekeeping at the top of a REQ_HEADER read (hdr_pos
81 * == 0): apply pipelining backpressure, trim grown scratch, reconcile the heap
82 * budget, and handle graceful-quit teardown of an idle keepalive. RETURN when
83 * the connection suspended or was torn down; NEXT to proceed reading the header.
84 */
85static brix_recv_step_t
86brix_recv_header_prep(ngx_stream_session_t *s, ngx_connection_t *c,
87 brix_ctx_t *ctx, size_t rx_pending)
88{
89 /*
90 * Phase 29 pipelining backpressure: if pipeline_depth responses are already
91 * queued/draining (out.count) OR pwrites are still in flight (wr_inflight),
92 * stop reading new requests and suspend with state=SENDING. Bounding
93 * out.count + wr_inflight at pipeline_depth caps BOTH in-flight pwrites and
94 * queued acks, so the out_ring can never overflow. (Both counters are 0 for
95 * the common serial case, so this is a no-op there.)
96 */
97 if (ctx->out.count + ctx->out.wr_inflight >= ctx->out.pipeline_depth) {
98 ctx->state = XRD_ST_SENDING;
99 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, rx_pending);
100 return BRIX_RECV_STEP_RETURN;
101 }
102
103 /*
104 * Top of a fresh request with the queue empty — the previous response is
105 * fully sent, so no chain references the scratch buffers. Phase 31: trim
106 * scratch a prior large read/pgwrite grew back to the streaming window, then
107 * reconcile the transfer-heap footprint with the SHM-global budget. Gated
108 * on out.count==0 so a pipelined read still in flight is never disturbed.
109 */
110 if (ctx->out.count == 0 && ctx->out.wr_inflight == 0) {
111 brix_trim_scratch(ctx, c);
112 brix_budget_sync(ctx);
113
114 /*
115 * Fast teardown: the worker is draining and this connection is quiescent
116 * at a fresh request boundary (no queued response, no in-flight pwrite).
117 * Close now rather than parking; the client reconnects to the new worker
118 * and resumes. BUT only when no file is open: a connection holding an
119 * open handle is mid-transfer (a streaming read parked between chunks, or
120 * a cache slice-fill in progress) — forcing a reconnect there loses the
121 * in-flight fill and surfaces as a spurious kXR_NotFound. Let the active
122 * transfer finish on the old worker (bounded by worker_shutdown_timeout).
123 */
124 if (ngx_exiting && !brix_ctx_has_open_file(ctx)) {
125 brix_on_disconnect(ctx, c);
126 brix_close_all_files(ctx);
127 ngx_stream_finalize_session(s, NGX_STREAM_OK);
128 return BRIX_RECV_STEP_RETURN;
129 }
130
131 /*
132 * Mark idle so a graceful quit's ngx_close_idle_connections() drops this
133 * parked keepalive immediately. An open handle means an active transfer,
134 * not a keepalive idle — leave it un-idle so the reload lets it complete.
135 */
136 c->idle = brix_ctx_has_open_file(ctx) ? 0 : 1;
137 }
138
139 return BRIX_RECV_STEP_NEXT;
140}

142/*
143 * The handshake's first byte is non-zero — not an XRootD client hello (the
144 * client hello begins with a zero streamid word); it is an HTTP method letter or
145 * a TLS 0x16. Splice to the local HTTP/WebDAV listener when brix_http_handoff
146 * is configured (one port serves both protocols), else close. RETURN when the
147 * relay took the connection; BREAK to close it.
148 */
149static brix_recv_step_t
150brix_recv_handle_nonxrootd(ngx_stream_session_t *s, ngx_connection_t *c,
151 ngx_stream_brix_srv_conf_t *conf, brix_ctx_t *ctx)
152{
153 if (conf->http_handoff_addr != NULL
154 && brix_http_handoff_start(s, c, conf,
155 ctx->recv.hdr_buf, ctx->recv.hdr_pos) == NGX_OK)
156 {
157 return BRIX_RECV_STEP_RETURN; /* the relay owns the connection now */
158 }
159 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
160 "brix: non-XRootD client (first byte 0x%02xd)"
161 " closing immediately",
162 (unsigned) ctx->recv.hdr_buf[0]);
163 return BRIX_RECV_STEP_BREAK;
164}
165
166/*
167 * Fold `avail` newly-received bytes into the current PDU state and decide the
168 * loop action: detect a non-XRootD handshake byte (→ handle_nonxrootd), CONTINUE
169 * on a short read (more of this unit still to come), or NEXT when the unit is now
170 * complete.
171 */
172static brix_recv_step_t
173brix_recv_advance(ngx_stream_session_t *s, ngx_connection_t *c,
174 ngx_stream_brix_srv_conf_t *conf, brix_ctx_t *ctx, size_t avail,
175 size_t need, size_t *rx_pending)
176{
177 *rx_pending += avail;
178
179 if (ctx->state == XRD_ST_HANDSHAKE) {
180 ctx->recv.hdr_pos += avail;
181
182 if (ctx->recv.hdr_pos >= 1 && ctx->recv.hdr_buf[0] != 0) {
183 return brix_recv_handle_nonxrootd(s, c, conf, ctx);
184 }
185
186 } else if (ctx->state == XRD_ST_REQ_HEADER) {
187 ctx->recv.hdr_pos += avail;
188
189 } else {
190 ctx->recv.payload_pos += avail;
191 }
192
193 if (avail < need) {
194 return BRIX_RECV_STEP_CONTINUE;
195 }
196
197 return BRIX_RECV_STEP_NEXT;
198}

200brix_recv_step_t
201brix_recv_read_frame(ngx_stream_session_t *s, ngx_connection_t *c,
202 ngx_stream_brix_srv_conf_t *conf, brix_ctx_t *ctx, ngx_event_t *rev,
203 size_t *rx_pending)
204{
205 u_char *dest;
206 size_t need;
207 ssize_t n;
208
209 if (ctx->state == XRD_ST_HANDSHAKE) {
210 dest = ctx->recv.hdr_buf + ctx->recv.hdr_pos;
211 need = XRD_HANDSHAKE_LEN - ctx->recv.hdr_pos;
212
213 } else if (ctx->state == XRD_ST_REQ_HEADER) {
214 if (ctx->recv.hdr_pos == 0) {
215 brix_recv_step_t st = brix_recv_header_prep(s, c, ctx, *rx_pending);
216 if (st != BRIX_RECV_STEP_NEXT) {
217 return st;
218 }
219 }
220 dest = ctx->recv.hdr_buf + ctx->recv.hdr_pos;
221 need = XRD_REQUEST_HDR_LEN - ctx->recv.hdr_pos;
222
223 } else {
224 /* cur_body_extra is non-zero only for kXR_writev (segment data streams
225 * after the dlen-framed descriptor block) and for kXR_chkpoint/ckpXeq
226 * (the sub-request body streams after the dlen-framed sub-header). */
227 dest = ctx->recv.payload + ctx->recv.payload_pos;
228 need = (size_t) ctx->recv.cur_dlen + ctx->recv.cur_body_extra
229 - ctx->recv.payload_pos;
230 }
231
232 if (need > 0) {
233 rev->available = -1;
234 n = c->recv(c, dest, need);
235
236 if (n == NGX_AGAIN) {
237 ngx_log_debug(NGX_LOG_DEBUG_STREAM, c->log, 0,
238 "brix: recv AGAIN st=%d hdr_pos=%uz avail=%d"
239 " ready=%d active=%d",
240 (int) ctx->state, ctx->recv.hdr_pos,
241 rev->available, (int) rev->ready,
242 (int) rev->active);
243 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
244 return BRIX_RECV_STEP_BREAK;
245 }
246 /* Phase 39: genuine incompletion — arm the read deadline once
247 * (idempotent, so repeated partial reads of the same PDU do NOT
248 * reset it), bounding time-to-complete against a slowloris. */
249 brix_arm_read_deadline(c, ctx);
250 BRIX_SRV_METRIC_ADD(ctx, wire_bytes_rx_total, *rx_pending);
251 return BRIX_RECV_STEP_RETURN;
252 }
253
254 if (n == NGX_ERROR || n == 0) {
255 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, c->log, 0,
256 "brix: client disconnected");
257 if (brix_defer_teardown_if_writing(ctx, c, NGX_STREAM_OK)) {
258 return BRIX_RECV_STEP_RETURN;
259 }
260 brix_on_disconnect(ctx, c);
261 brix_close_all_files(ctx);
262 ngx_stream_finalize_session(s, NGX_STREAM_OK);
263 return BRIX_RECV_STEP_RETURN;
264 }
265
266 return brix_recv_advance(s, c, conf, ctx, (size_t) n, need, rx_pending);
267 }
268
269 return BRIX_RECV_STEP_NEXT;
270}
