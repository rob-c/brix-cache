48/* Return the first direct child element of `parent` with the given local name,
49 * or NULL. NULL-safe in `parent` so callers can chain lookups without
50 * intermediate NULL checks. */
51static xmlNodePtr
52webdav_search_find_child(xmlNodePtr parent, const char *name)
53{
54 xmlNodePtr n;
55
56 for (n = parent != NULL ? parent->children : NULL; n != NULL; n = n->next) {
57 if (n->type == XML_ELEMENT_NODE
58 && xmlStrcmp(n->name, BAD_CAST name) == 0)
59 {
60 return n;
61 }
62 }
63
64 return NULL;
65}

67/*
68 * Depth-first search the subtree for the first DAV:literal element and copy its
69 * text content into q->literal (truncated to the fixed buffer). Stops at the
70 * first match: the supported query subset has only one literal term. q->literal
71 * being non-empty is the sentinel that unwinds the recursion early.
72 */
73static void
74webdav_search_find_literal(xmlNodePtr n, webdav_search_query_t *q)
75{
76 for (; n != NULL; n = n->next) {
77 if (n->type == XML_ELEMENT_NODE
78 && xmlStrcmp(n->name, BAD_CAST "literal") == 0)
79 {
80 xmlChar *text = xmlNodeGetContent(n);
81 if (text != NULL) {
82 ngx_cpystrn((u_char *) q->literal, text, sizeof(q->literal));
83 xmlFree(text);
84 }
85 return;
86 }
87
88 webdav_search_find_literal(n->children, q);
89 if (q->literal[0] != '\0') {
90 return;
91 }
92 }
93}

95/*
96 * Parse the SEARCH request body into a query descriptor.
97 * Expects DAV:searchrequest > DAV:basicsearch; extracts the scope depth
98 * (0 / 1 / infinity) and an optional DAV:contains literal. Returns
99 * NGX_HTTP_BAD_REQUEST for a missing/oversized/malformed body, NGX_OK otherwise.
100 * The libxml2 parse flags are the same hardened set as PROPFIND (no network,
101 * no external entities, no HUGE) to block entity-expansion DoS.
102 */
103static ngx_int_t
104webdav_search_parse(ngx_http_request_t *r, webdav_search_query_t *q)
105{
106 u_char *body = NULL;
107 size_t body_len = 0;
108 ngx_int_t rc;
109 xmlDocPtr doc;
110 xmlNodePtr root, basic, from, scope, depth, where;
111 int opts = XML_PARSE_NONET | XML_PARSE_NOERROR | XML_PARSE_NOWARNING;
112#if defined(XML_PARSE_NO_XXE)
113 opts |= XML_PARSE_NO_XXE;
114#endif
115
116 q->depth = 0;
117 q->literal[0] = '\0';
118
119 rc = brix_http_body_read_all(r, WEBDAV_SEARCH_BODY_MAX, &body, &body_len);
120 if (rc != NGX_OK || body_len == 0) {
121 return NGX_HTTP_BAD_REQUEST;
122 }
123
124 doc = xmlReadMemory((const char *) body, (int) body_len,
125 "search.xml", NULL, opts);
126 if (doc == NULL) {
127 return NGX_HTTP_BAD_REQUEST;
128 }
129
130 root = xmlDocGetRootElement(doc);
131 basic = webdav_search_find_child(root, "basicsearch");
132 if (root == NULL || xmlStrcmp(root->name, BAD_CAST "searchrequest") != 0
133 || basic == NULL)
134 {
135 xmlFreeDoc(doc);
136 return NGX_HTTP_BAD_REQUEST;
137 }
138
139 from = webdav_search_find_child(basic, "from");
140 scope = webdav_search_find_child(from, "scope");
141 depth = webdav_search_find_child(scope, "depth");
142 if (depth != NULL) {
143 xmlChar *text = xmlNodeGetContent(depth);
144 if (text != NULL) {
145 if (xmlStrcmp(text, BAD_CAST "1") == 0) {
146 q->depth = 1;
147 } else if (xmlStrcmp(text, BAD_CAST "infinity") == 0) {
148 q->depth = -1;
149 }
150 xmlFree(text);
151 }
152 }
153
154 where = webdav_search_find_child(basic, "where");
155 if (where != NULL) {
156 webdav_search_find_literal(where->children, q);
157 }
158
159 xmlFreeDoc(doc);
160 return NGX_OK;
161}

163/*
164 * Predicate: does this href satisfy the query? An empty literal matches
165 * everything; otherwise the literal must appear as a substring of the last path
166 * segment (displayname), matching the conservative DAV:contains semantics.
167 */
168static ngx_flag_t
169webdav_search_matches(const char *href, const webdav_search_query_t *q)
170{
171 const char *name;
172
173 if (q->literal[0] == '\0') {
174 return 1; /* no WHERE clause → match all */
175 }
176
177 /* Reduce href to its final segment (the displayname). */
178 name = href + strlen(href);
179 while (name > href && *(name - 1) != '/') {
180 name--;
181 }
182
183 return strstr(name, q->literal) != NULL;
184}

186/*
187 * Append one <D:response> for `href` to the multistatus chain, but only if it
188 * matches the query (non-matches return NGX_OK without emitting anything).
189 * The href is XML-escaped before interpolation.
190 */
191static ngx_int_t
192webdav_search_append_response(ngx_http_request_t *r, ngx_chain_t **head,
193 ngx_chain_t **tail, const char *href, const webdav_search_query_t *q)
194{
195 char *safe_href;
196
197 if (!webdav_search_matches(href, q)) {
198 return NGX_OK;
199 }
200
201 safe_href = webdav_escape_xml_text(r->pool, href);
202 if (safe_href == NULL) {
203 return NGX_ERROR;
204 }
205
206 return brix_http_chain_appendf(r->pool, head, tail,
207 "<D:response><D:href>%s</D:href>"
208 "<D:status>HTTP/1.1 200 OK</D:status></D:response>",
209 safe_href) == NULL
210 ? NGX_ERROR : NGX_OK;
211}

217/*
218 * WHAT: Test whether a directory listing entry should be skipped entirely.
219 * WHY: SEARCH never surfaces dotfiles or the internal sidecar/upload-temp
220 * namespace, and readdir already elides "."/".." — factoring the filter
221 * out keeps the walk loop focused on emitting matches.
222 * HOW: Returns 1 (skip) for a leading '.' or a reserved internal name, else 0.
223 */
224static int
225webdav_search_entry_skip(const char *dname)
226{
227 if (dname[0] == '.') {
228 return 1; /* skip hidden files (search never lists dotfiles) */
229 }
230 if (brix_is_internal_name(dname)) {
231 return 1; /* hide internal sidecars / upload temps (not dotfiles) */
232 }
233 return 0;
234}

236/*
237 * WHAT: Compose the child href = base_href + dname with exactly one separating
238 * '/', into the caller-provided buffer.
239 * WHY: base_href may or may not already end in '/'; the walk must never emit a
240 * double slash or a truncated href, and both truncation branches share the
241 * same "skip this entry" outcome.
242 * HOW: Picks the join format from base_href's trailing byte; returns NGX_OK on
243 * success or NGX_DECLINED when the formatted result would be truncated (a
244 * caller sentinel to skip the entry — no bytes are emitted).
245 */
246static ngx_int_t
247webdav_search_child_href(const char *base_href, const char *dname,
248 char *child_href, size_t child_href_size)
249{
250 size_t blen = strlen(base_href);
251 int n;
252
253 if (blen == 0 || base_href[blen - 1] != '/') {
254 n = snprintf(child_href, child_href_size, "%s/%s", base_href, dname);
255 } else {
256 n = snprintf(child_href, child_href_size, "%s%s", base_href, dname);
257 }
258
259 return (size_t) n >= child_href_size ? NGX_DECLINED : NGX_OK;
260}

262/*
263 * WHAT: Decide whether a listing entry is a directory to recurse into.
264 * WHY: d_type answers directly on most filesystems; only a DT_UNKNOWN result
265 * needs a fallback, and that fallback must be a confined *no-follow* probe
266 * so a trailing symlink is never followed into recursion.
267 * HOW: Returns 1 for BRIX_VFS_DT_DIR; for BRIX_VFS_DT_UNKNOWN runs a confined
268 * no-follow VFS probe of child_path; else 0.
269 */
270static int
271webdav_search_entry_is_dir(webdav_search_walk_ctx_t *w,
272 const char *root_canon, brix_vfs_dirent_kind_t dkind,
273 const char *child_path)
274{
275 brix_vfs_ctx_t pctx;
276 brix_vfs_stat_t vst;
277
278 if (dkind == BRIX_VFS_DT_DIR) {
279 return 1;
280 }
281 if (dkind != BRIX_VFS_DT_UNKNOWN) {
282 return 0;
283 }
284
285 brix_vfs_ctx_init(&pctx, w->r->pool, w->r->connection->log,
286 BRIX_PROTO_WEBDAV, root_canon, NULL, 0, 0, NULL, child_path);
287 return brix_vfs_probe(&pctx, 1 /* no-follow */, &vst) == NGX_OK
288 && vst.is_directory;
289}

291/*
292 * WHAT: Process one already-composed child (append its response, then recurse
293 * into it when the scope is depth-infinity and it is a directory).
294 * WHY: Isolating per-entry emission + descent from the enumeration loop keeps
295 * both the loop and this stage single-purpose and below the complexity cap.
296 * HOW: Appends the child response (NGX_ERROR propagates to the caller to abort
297 * the walk); bumps the shared counter; for a depth==infinity directory,
298 * builds the sub_href (skipping on truncation) and recurses. Returns
299 * NGX_OK on success or NGX_ERROR on an unrecoverable append/recursion
300 * failure.
301 */
302static ngx_int_t
303webdav_search_visit_child(webdav_search_walk_ctx_t *w, const char *root_canon,
304 brix_vfs_dirent_kind_t dkind, const char *child_path,
305 const char *child_href)
306{
307 if (webdav_search_append_response(w->r, w->head, w->tail, child_href, w->q)
308 != NGX_OK)
309 {
310 return NGX_ERROR;
311 }
312 (*w->count)++;
313
314 if (w->q->depth == -1
315 && webdav_search_entry_is_dir(w, root_canon, dkind, child_path))
316 {
317 char sub_href[WEBDAV_MAX_PATH + 3];
318
319 if ((size_t) snprintf(sub_href, sizeof(sub_href), "%s/", child_href)
320 < sizeof(sub_href)
321 && webdav_search_walk(w, child_path, sub_href) != NGX_OK)
322 {
323 return NGX_ERROR;
324 }
325 }
326
327 return NGX_OK;
328}

330/*
331 * WHAT: Enumerate one open directory, visiting each qualifying child.
332 * WHY: Split from the walk so the open/confine/close lifecycle and the per-entry
333 * loop are separate responsibilities.
334 * HOW: Loops readdir until end/error or the result-set cap; skips filtered and
335 * truncated/un-joinable entries; delegates emission+descent to
336 * webdav_search_visit_child. Returns NGX_OK to finish the directory or
337 * NGX_ERROR to abort the whole walk.
338 */
339static ngx_int_t
340webdav_search_scan_dir(webdav_search_walk_ctx_t *w, brix_vfs_dir_t *dp,
341 const char *root_canon, const char *dir_path, const char *base_href)
342{
343 for ( ;; ) {
344 ngx_str_t name;
345 brix_vfs_dirent_kind_t dkind;
346 const char *dname;
347 char child_path[WEBDAV_MAX_PATH];
348 char child_href[WEBDAV_MAX_PATH + 2];
349
350 if (*w->count >= WEBDAV_SEARCH_MAX_ENTRIES) {
351 break; /* result-set cap reached: stop scanning this dir */
352 }
353
354 /* "."/".." are filtered by readdir_kind; the entry KIND comes from d_type
355 * (no per-entry stat). A symlink/special is listed but never recursed. */
356 if (brix_vfs_readdir_kind(dp, &name, &dkind) != NGX_OK) {
357 break; /* NGX_DONE (end) or error → stop */
358 }
359 dname = (const char *) name.data;
360
361 if (webdav_search_entry_skip(dname)) {
362 continue;
363 }
364 if (brix_fs_join_path(dir_path, dname, child_path,
365 sizeof(child_path)) != NGX_OK)
366 {
367 continue;
368 }
369 if (webdav_search_child_href(base_href, dname, child_href,
370 sizeof(child_href)) != NGX_OK)
371 {
372 continue; /* href would truncate: skip this entry */
373 }
374
375 if (webdav_search_visit_child(w, root_canon, dkind, child_path,
376 child_href) != NGX_OK)
377 {
378 return NGX_ERROR;
379 }
380 }
381
382 return NGX_OK;
383}

385/*
386 * WHAT: Recursively scan dir_path, appending a response for each matching
387 * descendant. Recurses only when depth==infinity.
388 * WHY: Entry point for the SEARCH namespace walk; owns the confidentiality gate
389 * and the VFS directory lifecycle while delegating enumeration.
390 * HOW: Applies the Phase-40 mapped-user access gate (skip subtree on denial),
391 * opens the directory NON-metered through the VFS, scans it, and always
392 * closes the handle. The shared counter caps the result set. Unreadable
393 * dirs are skipped (NGX_OK); an append/recursion failure returns NGX_ERROR
394 * with the directory closed first.
395 */
396static ngx_int_t
397webdav_search_walk(webdav_search_walk_ctx_t *w, const char *dir_path,
398 const char *base_href)
399{
400 brix_vfs_ctx_t wctx;
401 brix_vfs_dir_t *dp;
402 ngx_int_t rc;
403 ngx_http_brix_webdav_loc_conf_t *wdcf =
404 ngx_http_get_module_loc_conf(w->r, ngx_http_brix_webdav_module);
405 const char *root_canon = (const char *) wdcf->common.root_canon;
406
407 /*
408 * Phase 40 confidentiality gate (mirrors propfind): under impersonation the
409 * worker uid may be able to readdir() a directory the *mapped* user cannot.
410 * Ask the broker to open it as the mapped user first; on denial skip this
411 * subtree silently rather than enumerate it with the worker's credentials.
412 * No-op (returns NGX_OK) when impersonation is off.
413 */
414 if (brix_dirlist_access_ok(w->r->connection->log, wdcf->common.root_canon,
415 dir_path) != NGX_OK)
416 {
417 return NGX_OK; /* mapped user may not list this dir — no leak */
418 }
419
420 /* Enumerate through the VFS (broker fdopendir under impersonation), NON-metered:
421 * a depth-infinity SEARCH must not emit one OP_DIRLIST per visited subdir (the
422 * SEARCH op accounts for the whole walk). */
423 brix_vfs_ctx_init(&wctx, w->r->pool, w->r->connection->log,
424 BRIX_PROTO_WEBDAV, wdcf->common.root_canon, NULL, 0 /* allow_write */,
425 0 /* is_tls */, NULL, dir_path);
426 dp = brix_vfs_opendir_quiet(&wctx, NULL);
427 if (dp == NULL) {
428 return NGX_OK; /* unreadable subtree: skip silently */
429 }
430
431 rc = webdav_search_scan_dir(w, dp, root_canon, dir_path, base_href);
432
433 brix_vfs_closedir(dp, w->r->connection->log);
434 return rc;
435}

437/*
438 * WHAT: Build the 207 multistatus body chain for the resolved SEARCH scope.
439 * WHY: Splitting body construction from response finalization keeps each stage
440 * single-purpose; this stage owns the XML framing and the walk fan-out.
441 * HOW: Emits the multistatus prologue, the scope's own <D:response>, then (for
442 * depth 1 / infinity on a directory scope) the matching descendants via
443 * webdav_search_walk, and the closing tag. The chain endpoints and entry
444 * counter live in the caller-owned walk ctx `w`; the scope href is r->uri
445 * truncated to the buffer, exactly as before. Returns NGX_OK, or
446 * NGX_HTTP_INTERNAL_SERVER_ERROR on any chain-append / walk failure.
447 */
448static ngx_int_t
449webdav_search_build_body(webdav_search_walk_ctx_t *w, const char *path,
450 const struct stat *sb, const webdav_search_query_t *q)
451{
452 ngx_http_request_t *r = w->r;
453 char href[WEBDAV_MAX_PATH + 2];
454 size_t uri_len;
455
456 if (brix_http_chain_appendf(r->pool, w->head, w->tail,
457 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
458 "<D:multistatus xmlns:D=\"DAV:\">") == NULL)
459 {
460 return NGX_HTTP_INTERNAL_SERVER_ERROR;
461 }
462
463 uri_len = r->uri.len < sizeof(href) - 1 ? r->uri.len : sizeof(href) - 1;
464 ngx_memcpy(href, r->uri.data, uri_len);
465 href[uri_len] = '\0';
466
467 if (webdav_search_append_response(r, w->head, w->tail, href, q) != NGX_OK) {
468 return NGX_HTTP_INTERNAL_SERVER_ERROR;
469 }
470
471 if ((q->depth == 1 || q->depth == -1) && S_ISDIR(sb->st_mode)
472 && webdav_search_walk(w, path, href) != NGX_OK)
473 {
474 return NGX_HTTP_INTERNAL_SERVER_ERROR;
475 }
476
477 if (brix_http_chain_appendf(r->pool, w->head, w->tail,
478 "</D:multistatus>") == NULL)
479 {
480 return NGX_HTTP_INTERNAL_SERVER_ERROR;
481 }
482
483 return NGX_OK;
484}

486/*
487 * WHAT: Finalize the 207 multistatus response: mark the chain end, set status /
488 * length / Content-Type, then send headers and the body.
489 * WHY: Isolates the nginx output plumbing from body construction so both stay
490 * below the complexity cap.
491 * HOW: Flags the tail as last buffer, sums the body length, sets the 207 status
492 * and the XML Content-Type header, sends the header (short-circuiting on
493 * error / header_only exactly as before) and streams the chain. Returns
494 * the ngx_http_send_header / ngx_http_output_filter result or
495 * NGX_HTTP_INTERNAL_SERVER_ERROR on header-list push failure.
496 */
497static ngx_int_t
498webdav_search_finalize(ngx_http_request_t *r, ngx_chain_t *head,
499 ngx_chain_t *tail)
500{
501 off_t total_len = 0;
502 ngx_chain_t *lc;
503 ngx_table_elt_t *h;
504 ngx_int_t rc;
505
506 if (tail != NULL) {
507 tail->buf->last_buf = 1;
508 tail->buf->last_in_chain = 1;
509 }
510
511 for (lc = head; lc != NULL; lc = lc->next) {
512 total_len += lc->buf->last - lc->buf->pos;
513 }
514
515 r->headers_out.status = 207;
516 r->headers_out.content_length_n = total_len;
517
518 h = ngx_list_push(&r->headers_out.headers);
519 if (h == NULL) {
520 return NGX_HTTP_INTERNAL_SERVER_ERROR;
521 }
522 h->hash = 1;
523 ngx_str_set(&h->key, "Content-Type");
524 ngx_str_set(&h->value, "application/xml; charset=\"utf-8\"");
525
526 rc = ngx_http_send_header(r);
527 if (rc == NGX_ERROR || r->header_only) {
528 return rc;
529 }
530
531 return ngx_http_output_filter(r, head);
532}

534/*
535 * Execute the SEARCH after the body is available: resolve+stat the scope,
536 * parse the query, then build a 207 multistatus listing the scope itself plus
537 * any matching children (depth 1) or descendants (infinity). Returns an HTTP
538 * status / NGX_* code.
539 */
540static ngx_int_t
541webdav_search_do(ngx_http_request_t *r)
542{
543 char path[WEBDAV_MAX_PATH];
544 struct stat sb;
545 ngx_int_t rc;
546 webdav_search_query_t q;
547 ngx_chain_t *head = NULL, *tail = NULL;
548 ngx_uint_t count = 0;
549 webdav_search_walk_ctx_t w;
550
551 rc = webdav_resolve_stat(r, path, sizeof(path), &sb);
552 if (rc != NGX_OK) {
553 return rc;
554 }
555
556 rc = webdav_search_parse(r, &q);
557 if (rc != NGX_OK) {
558 return rc;
559 }
560
561 w.r = r;
562 w.head = &head;
563 w.tail = &tail;
564 w.count = &count;
565 w.q = &q;
566
567 rc = webdav_search_build_body(&w, path, &sb, &q);
568 if (rc != NGX_OK) {
569 return rc;
570 }
571
572 return webdav_search_finalize(r, head, tail);
573}

575/*
576 * Body-ready callback (async re-entry point): runs the search and finalizes the
577 * request, also recording WebDAV metrics for the operation.
578 *
579 * Phase 40: SEARCH walks the namespace exactly like PROPFIND, so it is read
580 * asynchronously and the outer dispatch wrapper has already cleared the
581 * impersonation principal by the time this callback fires. Re-establish it for
582 * the (synchronous) walk so the directory stat/opendir and confidentiality
583 * checks run as the mapped user — otherwise SEARCH would enumerate as the
584 * unprivileged worker and could leak entries the mapped user cannot read.
585 * No-op unless map mode is active.
586 */
587static void
588webdav_search_body_handler(ngx_http_request_t *r)
589{
590 ngx_http_brix_webdav_req_ctx_t *rx =
591 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
592 ngx_int_t rc;
593
594 brix_imp_request_begin(rx != NULL ? rx->identity : NULL);
595 rc = webdav_search_do(r);
596 brix_imp_request_end();
597
598 webdav_metrics_finalize_request(r, rc);
599}
