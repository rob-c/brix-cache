33/* Return the first direct child element of `parent` with the given local name,
34 * or NULL. NULL-safe in `parent` so callers can chain lookups without
35 * intermediate NULL checks. */
36static xmlNodePtr
37webdav_search_find_child(xmlNodePtr parent, const char *name)
38{
39 xmlNodePtr n;
40
41 for (n = parent != NULL ? parent->children : NULL; n != NULL; n = n->next) {
42 if (n->type == XML_ELEMENT_NODE
43 && xmlStrcmp(n->name, BAD_CAST name) == 0)
44 {
45 return n;
46 }
47 }
48
49 return NULL;
50}

52/*
53 * Depth-first search the subtree for the first DAV:literal element and copy its
54 * text content into q->literal (truncated to the fixed buffer). Stops at the
55 * first match: the supported query subset has only one literal term. q->literal
56 * being non-empty is the sentinel that unwinds the recursion early.
57 */
58static void
59webdav_search_find_literal(xmlNodePtr n, webdav_search_query_t *q)
60{
61 for (; n != NULL; n = n->next) {
62 if (n->type == XML_ELEMENT_NODE
63 && xmlStrcmp(n->name, BAD_CAST "literal") == 0)
64 {
65 xmlChar *text = xmlNodeGetContent(n);
66 if (text != NULL) {
67 ngx_cpystrn((u_char *) q->literal, text, sizeof(q->literal));
68 xmlFree(text);
69 }
70 return;
71 }
72
73 webdav_search_find_literal(n->children, q);
74 if (q->literal[0] != '\0') {
75 return;
76 }
77 }
78}

80/*
81 * Parse the SEARCH request body into a query descriptor.
82 * Expects DAV:searchrequest > DAV:basicsearch; extracts the scope depth
83 * (0 / 1 / infinity) and an optional DAV:contains literal. Returns
84 * NGX_HTTP_BAD_REQUEST for a missing/oversized/malformed body, NGX_OK otherwise.
85 * The libxml2 parse flags are the same hardened set as PROPFIND (no network,
86 * no external entities, no HUGE) to block entity-expansion DoS.
87 */
88static ngx_int_t
89webdav_search_parse(ngx_http_request_t *r, webdav_search_query_t *q)
90{
91 u_char *body = NULL;
92 size_t body_len = 0;
93 ngx_int_t rc;
94 xmlDocPtr doc;
95 xmlNodePtr root, basic, from, scope, depth, where;
96 int opts = XML_PARSE_NONET | XML_PARSE_NOERROR | XML_PARSE_NOWARNING;
97#if defined(XML_PARSE_NO_XXE)
98 opts |= XML_PARSE_NO_XXE;
99#endif
100
101 q->depth = 0;
102 q->literal[0] = '\0';
103
104 rc = brix_http_body_read_all(r, WEBDAV_SEARCH_BODY_MAX, &body, &body_len);
105 if (rc != NGX_OK || body_len == 0) {
106 return NGX_HTTP_BAD_REQUEST;
107 }
108
109 doc = xmlReadMemory((const char *) body, (int) body_len,
110 "search.xml", NULL, opts);
111 if (doc == NULL) {
112 return NGX_HTTP_BAD_REQUEST;
113 }
114
115 root = xmlDocGetRootElement(doc);
116 basic = webdav_search_find_child(root, "basicsearch");
117 if (root == NULL || xmlStrcmp(root->name, BAD_CAST "searchrequest") != 0
118 || basic == NULL)
119 {
120 xmlFreeDoc(doc);
121 return NGX_HTTP_BAD_REQUEST;
122 }
123
124 from = webdav_search_find_child(basic, "from");
125 scope = webdav_search_find_child(from, "scope");
126 depth = webdav_search_find_child(scope, "depth");
127 if (depth != NULL) {
128 xmlChar *text = xmlNodeGetContent(depth);
129 if (text != NULL) {
130 if (xmlStrcmp(text, BAD_CAST "1") == 0) {
131 q->depth = 1;
132 } else if (xmlStrcmp(text, BAD_CAST "infinity") == 0) {
133 q->depth = -1;
134 }
135 xmlFree(text);
136 }
137 }
138
139 where = webdav_search_find_child(basic, "where");
140 if (where != NULL) {
141 webdav_search_find_literal(where->children, q);
142 }
143
144 xmlFreeDoc(doc);
145 return NGX_OK;
146}

148/*
149 * Predicate: does this href satisfy the query? An empty literal matches
150 * everything; otherwise the literal must appear as a substring of the last path
151 * segment (displayname), matching the conservative DAV:contains semantics.
152 */
153static ngx_flag_t
154webdav_search_matches(const char *href, const webdav_search_query_t *q)
155{
156 const char *name;
157
158 if (q->literal[0] == '\0') {
159 return 1; /* no WHERE clause → match all */
160 }
161
162 /* Reduce href to its final segment (the displayname). */
163 name = href + strlen(href);
164 while (name > href && *(name - 1) != '/') {
165 name--;
166 }
167
168 return strstr(name, q->literal) != NULL;
169}

171/*
172 * Append one <D:response> for `href` to the multistatus chain, but only if it
173 * matches the query (non-matches return NGX_OK without emitting anything).
174 * The href is XML-escaped before interpolation.
175 */
176static ngx_int_t
177webdav_search_append_response(ngx_http_request_t *r, ngx_chain_t **head,
178 ngx_chain_t **tail, const char *href, const webdav_search_query_t *q)
179{
180 char *safe_href;
181
182 if (!webdav_search_matches(href, q)) {
183 return NGX_OK;
184 }
185
186 safe_href = webdav_escape_xml_text(r->pool, href);
187 if (safe_href == NULL) {
188 return NGX_ERROR;
189 }
190
191 return brix_http_chain_appendf(r->pool, head, tail,
192 "<D:response><D:href>%s</D:href>"
193 "<D:status>HTTP/1.1 200 OK</D:status></D:response>",
194 safe_href) == NULL
195 ? NGX_ERROR : NGX_OK;
196}

198/*
199 * Recursively scan dir_path, appending a response for each matching descendant.
200 * `count` is shared across the recursion and capped at WEBDAV_SEARCH_MAX_ENTRIES
201 * to bound the result set. Recurses only when depth==infinity. Unreadable dirs
202 * are skipped; on append error the DIR is closed before returning NGX_ERROR.
203 */
204static ngx_int_t
205webdav_search_walk(ngx_http_request_t *r, ngx_chain_t **head,
206 ngx_chain_t **tail, const char *dir_path, const char *base_href,
207 ngx_uint_t *count, const webdav_search_query_t *q)
208{
209 brix_vfs_ctx_t wctx;
210 brix_vfs_dir_t *dp;
211 ngx_http_brix_webdav_loc_conf_t *wdcf =
212 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
213
214 /*
215 * Phase 40 confidentiality gate (mirrors propfind): under impersonation the
216 * worker uid may be able to readdir() a directory the *mapped* user cannot.
217 * Ask the broker to open it as the mapped user first; on denial skip this
218 * subtree silently rather than enumerate it with the worker's credentials.
219 * No-op (returns NGX_OK) when impersonation is off.
220 */
221 if (brix_dirlist_access_ok(r->connection->log, wdcf->common.root_canon,
222 dir_path) != NGX_OK)
223 {
224 return NGX_OK; /* mapped user may not list this dir — no leak */
225 }
226
227 /* Enumerate through the VFS (broker fdopendir under impersonation), NON-metered:
228 * a depth-infinity SEARCH must not emit one OP_DIRLIST per visited subdir (the
229 * SEARCH op accounts for the whole walk). */
230 brix_vfs_ctx_init(&wctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
231 wdcf->common.root_canon, NULL, 0 /* allow_write */, 0 /* is_tls */,
232 NULL, dir_path);
233 dp = brix_vfs_opendir_quiet(&wctx, NULL);
234 if (dp == NULL) {
235 return NGX_OK; /* unreadable subtree: skip silently */
236 }
237
238 for ( ;; ) {
239 ngx_str_t name;
240 brix_vfs_dirent_kind_t dkind;
241 const char *dname;
242 char child_path[WEBDAV_MAX_PATH];
243 char child_href[WEBDAV_MAX_PATH + 2];
244 size_t blen;
245 int is_dir;
246 ngx_int_t rrc;
247
248 if (*count >= WEBDAV_SEARCH_MAX_ENTRIES) {
249 break; /* result-set cap reached: stop scanning this dir */
250 }
251
252 /* "."/".." are filtered by readdir_kind; the entry KIND comes from d_type
253 * (no per-entry stat). A symlink/special is listed but never recursed. */
254 rrc = brix_vfs_readdir_kind(dp, &name, &dkind);
255 if (rrc != NGX_OK) {
256 break; /* NGX_DONE (end) or error → stop */
257 }
258 dname = (const char *) name.data;
259
260 if (dname[0] == '.') {
261 continue; /* skip hidden files (search never lists dotfiles) */
262 }
263 if (brix_is_internal_name(dname)) {
264 continue; /* hide internal sidecars / upload temps (not dotfiles) */
265 }
266
267 if (brix_fs_join_path(dir_path, dname, child_path,
268 sizeof(child_path)) != NGX_OK)
269 {
270 continue;
271 }
272
273 /* Compose child href = base_href + name with exactly one separating
274 * '/'; skip the entry if the result would be truncated. */
275 blen = strlen(base_href);
276 if (blen == 0 || base_href[blen - 1] != '/') {
277 if ((size_t) snprintf(child_href, sizeof(child_href), "%s/%s",
278 base_href, dname) >= sizeof(child_href))
279 {
280 continue;
281 }
282 } else if ((size_t) snprintf(child_href, sizeof(child_href), "%s%s",
283 base_href, dname)
284 >= sizeof(child_href))
285 {
286 continue;
287 }
288
289 if (webdav_search_append_response(r, head, tail, child_href, q)
290 != NGX_OK)
291 {
292 brix_vfs_closedir(dp, r->connection->log);
293 return NGX_ERROR;
294 }
295 (*count)++;
296
297 /* Recurse only into directories. d_type gives the answer directly; on a
298 * DT_UNKNOWN filesystem fall back to a confined no-follow probe (so a
299 * trailing symlink is never followed into recursion). */
300 is_dir = (dkind == BRIX_VFS_DT_DIR);
301 if (dkind == BRIX_VFS_DT_UNKNOWN) {
302 brix_vfs_ctx_t pctx;
303 brix_vfs_stat_t vst;
304
305 brix_vfs_ctx_init(&pctx, r->pool, r->connection->log,
306 BRIX_PROTO_WEBDAV, wdcf->common.root_canon, NULL, 0, 0, NULL,
307 child_path);
308 is_dir = (brix_vfs_probe(&pctx, 1 /* no-follow */, &vst) == NGX_OK
309 && vst.is_directory);
310 }
311
312 if (q->depth == -1 && is_dir) {
313 char sub_href[WEBDAV_MAX_PATH + 3];
314 if ((size_t) snprintf(sub_href, sizeof(sub_href), "%s/",
315 child_href) < sizeof(sub_href)
316 && webdav_search_walk(r, head, tail, child_path, sub_href,
317 count, q) != NGX_OK)
318 {
319 brix_vfs_closedir(dp, r->connection->log);
320 return NGX_ERROR;
321 }
322 }
323 }
324
325 brix_vfs_closedir(dp, r->connection->log);
326 return NGX_OK;
327}

329/*
330 * Execute the SEARCH after the body is available: resolve+stat the scope,
331 * parse the query, then build a 207 multistatus listing the scope itself plus
332 * any matching children (depth 1) or descendants (infinity). Returns an HTTP
333 * status / NGX_* code.
334 */
335static ngx_int_t
336webdav_search_do(ngx_http_request_t *r)
337{
338 char path[WEBDAV_MAX_PATH];
339 char href[WEBDAV_MAX_PATH + 2];
340 struct stat sb;
341 ngx_int_t rc;
342 webdav_search_query_t q;
343 ngx_chain_t *head = NULL, *tail = NULL, *lc;
344 ngx_uint_t count = 0;
345 off_t total_len = 0;
346 ngx_table_elt_t *h;
347 size_t uri_len;
348
349 rc = webdav_resolve_stat(r, path, sizeof(path), &sb);
350 if (rc != NGX_OK) {
351 return rc;
352 }
353
354 rc = webdav_search_parse(r, &q);
355 if (rc != NGX_OK) {
356 return rc;
357 }
358
359 if (brix_http_chain_appendf(r->pool, &head, &tail,
360 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
361 "<D:multistatus xmlns:D=\"DAV:\">") == NULL)
362 {
363 return NGX_HTTP_INTERNAL_SERVER_ERROR;
364 }
365
366 uri_len = r->uri.len < sizeof(href) - 1 ? r->uri.len : sizeof(href) - 1;
367 ngx_memcpy(href, r->uri.data, uri_len);
368 href[uri_len] = '\0';
369
370 if (webdav_search_append_response(r, &head, &tail, href, &q) != NGX_OK) {
371 return NGX_HTTP_INTERNAL_SERVER_ERROR;
372 }
373
374 if ((q.depth == 1 || q.depth == -1) && S_ISDIR(sb.st_mode)
375 && webdav_search_walk(r, &head, &tail, path, href, &count, &q)
376 != NGX_OK)
377 {
378 return NGX_HTTP_INTERNAL_SERVER_ERROR;
379 }
380
381 if (brix_http_chain_appendf(r->pool, &head, &tail,
382 "</D:multistatus>") == NULL)
383 {
384 return NGX_HTTP_INTERNAL_SERVER_ERROR;
385 }
386
387 if (tail != NULL) {
388 tail->buf->last_buf = 1;
389 tail->buf->last_in_chain = 1;
390 }
391
392 for (lc = head; lc != NULL; lc = lc->next) {
393 total_len += lc->buf->last - lc->buf->pos;
394 }
395
396 r->headers_out.status = 207;
397 r->headers_out.content_length_n = total_len;
398
399 h = ngx_list_push(&r->headers_out.headers);
400 if (h == NULL) {
401 return NGX_HTTP_INTERNAL_SERVER_ERROR;
402 }
403 h->hash = 1;
404 ngx_str_set(&h->key, "Content-Type");
405 ngx_str_set(&h->value, "application/xml; charset=\"utf-8\"");
406
407 rc = ngx_http_send_header(r);
408 if (rc == NGX_ERROR || r->header_only) {
409 return rc;
410 }
411
412 return ngx_http_output_filter(r, head);
413}

415/*
416 * Body-ready callback (async re-entry point): runs the search and finalizes the
417 * request, also recording WebDAV metrics for the operation.
418 *
419 * Phase 40: SEARCH walks the namespace exactly like PROPFIND, so it is read
420 * asynchronously and the outer dispatch wrapper has already cleared the
421 * impersonation principal by the time this callback fires. Re-establish it for
422 * the (synchronous) walk so the directory stat/opendir and confidentiality
423 * checks run as the mapped user — otherwise SEARCH would enumerate as the
424 * unprivileged worker and could leak entries the mapped user cannot read.
425 * No-op unless map mode is active.
426 */
427static void
428webdav_search_body_handler(ngx_http_request_t *r)
429{
430 ngx_http_brix_webdav_req_ctx_t *rx =
431 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
432 ngx_int_t rc;
433
434 brix_imp_request_begin(rx != NULL ? rx->identity : NULL);
435 rc = webdav_search_do(r);
436 brix_imp_request_end();
437
438 webdav_metrics_finalize_request(r, rc);
439}
