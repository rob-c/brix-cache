17/* POST body handler: flip the io_uring runtime kill switch. Body {"enabled":
18 * bool} — false disables io_uring fleet-wide on every worker's next op (the
19 * no-reload CVE-response switch); true re-enables it. 404 if io_uring was not
20 * enabled in the config (no SHM flag to flip). */
21ngx_int_t
22admin_io_uring_set(ngx_http_request_t *r, json_t *body)
23{
24 json_t *en = json_object_get(body, "enabled");
25 int enabled;
26
27 if (!json_is_boolean(en)) {
28 admin_audit(r, "io_uring", NULL, "bad_request");
29 return admin_send_error(r, NGX_HTTP_BAD_REQUEST, "missing_field");
30 }
31 enabled = json_is_true(en);
32
33 if (brix_uring_killswitch_set(enabled ? 0 : 1) != NGX_OK) {
34 admin_audit(r, "io_uring", NULL, "not_enabled");
35 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "io_uring_not_enabled");
36 }
37
38 admin_audit(r, "io_uring", enabled ? "enable" : "disable", "ok");
39 {
40 json_t *root = json_object();
41 if (root == NULL) {
42 return NGX_HTTP_INTERNAL_SERVER_ERROR;
43 }
44 dashboard_json_set_schema(root);
45 json_object_set_new(root, "result",
46 json_string(enabled ? "enabled" : "disabled"));
47 return dashboard_json_send(r, NGX_HTTP_OK, root);
48 }
49}

68/* Directive setter for `brix_admin_allow <cidr>...`: append each CIDR arg to
69 * the loc-conf allowlist array (created lazily). NGX_DONE from ngx_ptocidr means
70 * the address had non-zero host bits, which is a warning, not an error. */
71char *
72brix_admin_set_allow(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
73{
74 ngx_http_brix_dashboard_loc_conf_t *lcf = conf;
75 ngx_str_t *value = cf->args->elts;
76 ngx_uint_t i;
77
78 (void) cmd;
79
80 if (lcf->admin_allow == NULL) {
81 lcf->admin_allow = ngx_array_create(cf->pool, cf->args->nelts - 1,
82 sizeof(ngx_cidr_t));
83 if (lcf->admin_allow == NULL) {
84 return NGX_CONF_ERROR;
85 }
86 }
87
88 for (i = 1; i < cf->args->nelts; i++) {
89 ngx_cidr_t *cidr = ngx_array_push(lcf->admin_allow);
90 ngx_int_t rc;
91 if (cidr == NULL) {
92 return NGX_CONF_ERROR;
93 }
94 rc = ngx_ptocidr(&value[i], cidr);
95 if (rc == NGX_ERROR) {
96 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
97 "invalid CIDR \"%V\" in brix_admin_allow",
98 &value[i]);
99 return NGX_CONF_ERROR;
100 }
101 if (rc == NGX_DONE) {
102 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
103 "low address bits of \"%V\" in brix_admin_allow were ignored",
104 &value[i]);
105 }
106 }
107 return NGX_CONF_OK;
108}

111/*
112 * WHAT: Directive setter for `brix_admin_secret <file>`: load the bearer token
113 * from a file at config time.
114 * WHY: Keeping the secret in a file (not inline in nginx.conf) limits exposure;
115 * the transient stack copy is OPENSSL_cleanse'd on every exit path so it
116 * does not linger in memory after parsing.
117 * HOW: Read the file, strip trailing whitespace/newlines, reject empty or
118 * sub-ADMIN_SECRET_MIN tokens (brute-force resistance), then copy the
119 * trimmed token into the pool.
120 */
121char *
122brix_admin_set_secret(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
123{
124 ngx_http_brix_dashboard_loc_conf_t *lcf = conf;
125 ngx_str_t *value = cf->args->elts;
126 ngx_str_t path = value[1];
127 ngx_file_t file;
128 u_char rbuf[ADMIN_SECRET_MAX];
129 ssize_t n;
130 size_t len;
131
132 (void) cmd;
133
134 if (lcf->admin_secret.len > 0) {
135 return "is duplicate";
136 }
137 if (ngx_conf_full_name(cf->cycle, &path, 1) != NGX_OK) {
138 return NGX_CONF_ERROR;
139 }
140
141 ngx_memzero(&file, sizeof(file));
142 file.name = path;
143 file.log = cf->log;
144 file.fd = ngx_open_file(path.data, NGX_FILE_RDONLY, NGX_FILE_OPEN, 0);
145 if (file.fd == NGX_INVALID_FILE) {
146 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
147 "brix_admin_secret: cannot open \"%V\"", &path);
148 return NGX_CONF_ERROR;
149 }
150
151 n = ngx_read_file(&file, rbuf, sizeof(rbuf), 0);
152 ngx_close_file(file.fd);
153 if (n <= 0) {
154 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
155 "brix_admin_secret: \"%V\" is empty or unreadable",
156 &path);
157 return NGX_CONF_ERROR;
158 }
159
160 /* Trim trailing whitespace/newlines. */
161 len = (size_t) n;
162 while (len > 0 && (rbuf[len - 1] == '\n' || rbuf[len - 1] == '\r'
163 || rbuf[len - 1] == ' ' || rbuf[len - 1] == '\t')) {
164 len--;
165 }
166 if (len == 0) {
167 OPENSSL_cleanse(rbuf, sizeof(rbuf));
168 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
169 "brix_admin_secret: \"%V\" contains no token",
170 &path);
171 return NGX_CONF_ERROR;
172 }
173 /* W6/E2 — reject trivially short secrets that invite brute force. */
174 if (len < ADMIN_SECRET_MIN) {
175 OPENSSL_cleanse(rbuf, sizeof(rbuf));
176 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
177 "brix_admin_secret: \"%V\" token is too short "
178 "(%uz bytes; need >= %d)", &path, len,
179 (int) ADMIN_SECRET_MIN);
180 return NGX_CONF_ERROR;
181 }
182
183 lcf->admin_secret.data = ngx_pnalloc(cf->pool, len);
184 if (lcf->admin_secret.data == NULL) {
185 OPENSSL_cleanse(rbuf, sizeof(rbuf));
186 return NGX_CONF_ERROR;
187 }
188 ngx_memcpy(lcf->admin_secret.data, rbuf, len);
189 lcf->admin_secret.len = len;
190 /* W6/F1 — scrub the transient stack copy of the secret. */
191 OPENSSL_cleanse(rbuf, sizeof(rbuf));
192 return NGX_CONF_OK;
193}

202/*
203 * ---- Directive setter for `brix_admin_rate_limit` ----
204 *
205 * WHAT: Parses `brix_admin_rate_limit off` (disable the throttle) or
206 * `brix_admin_rate_limit <writes/min> [<reads/min>]` into the loc-conf
207 * per-minute fields. Returns NGX_CONF_OK or a config-error string.
208 *
209 * WHY: The admin API must not be a DoS vector, but sites legitimately poll it
210 * hard under load — so the limits are operator-tunable per method class,
211 * with a value of 0 meaning "unlimited" for that class.
212 *
213 * HOW: 1. Reject a duplicate declaration.
214 * 2. A single literal "off" arg clears the enable flag and returns.
215 * 3. Parse the mandatory writes/min; parse reads/min when present
216 * (absent leaves it UNSET so the merge default applies).
217 */
218char *
219brix_admin_set_rate_limit(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
220{
221 ngx_http_brix_dashboard_loc_conf_t *lcf = conf;
222 ngx_str_t *value = cf->args->elts;
223 ngx_int_t per_min;
224
225 (void) cmd;
226
227 if (lcf->admin_rl_enable != NGX_CONF_UNSET) {
228 return "is duplicate";
229 }
230
231 if (value[1].len == 3 && ngx_strncmp(value[1].data, "off", 3) == 0) {
232 if (cf->args->nelts != 2) {
233 return "\"off\" takes no further arguments";
234 }
235 lcf->admin_rl_enable = 0;
236 return NGX_CONF_OK;
237 }
238
239 per_min = ngx_atoi(value[1].data, value[1].len);
240 if (per_min == NGX_ERROR) {
241 return "first argument must be \"off\" or write requests/minute";
242 }
243 lcf->admin_rl_write_pm = (ngx_uint_t) per_min;
244
245 if (cf->args->nelts == 3) {
246 per_min = ngx_atoi(value[2].data, value[2].len);
247 if (per_min == NGX_ERROR) {
248 return "second argument must be read requests/minute";
249 }
250 lcf->admin_rl_read_pm = (ngx_uint_t) per_min;
251 }
252
253 lcf->admin_rl_enable = 1;
254 return NGX_CONF_OK;
255}

258/*
259 * ---- Build one per-IP admin throttle rule ----
260 *
261 * WHAT: Fills `rule` as a per-IP leaky-bucket request-rate rule draining at
262 * `per_min` requests/minute against `zone`. per_min == 0 leaves
263 * req_rate 0, which brix_rl_check treats as "no limit".
264 *
265 * WHY: The read and write rules differ only in their per-minute value; one
266 * builder keeps the fixed-point conversion and burst policy in one place.
267 *
268 * HOW: 1. Zero the rule and bind zone + IP key dimension.
269 * 2. Convert requests/minute to the limiter's req/s × BRIX_RL_REQ_SCALE
270 * fixed point.
271 * 3. Grant a burst of a quarter-minute's traffic (min 10) so bursty but
272 * legitimate clients (dashboard page loads, scripted sweeps) are not
273 * clipped by the steady-state rate.
274 */
275static void
276admin_rl_build_rule(brix_rl_rule_t *rule, ngx_uint_t per_min,
277 brix_rl_zone_t *zone)
278{
279 ngx_memzero(rule, sizeof(*rule));
280 rule->key_type = BRIX_RL_KEY_IP;
281 rule->zone = zone;
282 rule->req_rate = per_min * BRIX_RL_REQ_SCALE / 60;
283 rule->req_burst = per_min / 4 > 10 ? per_min / 4 : 10;
284}

287/*
288 * ---- Finalize the admin API throttle at config merge time ----
289 *
290 * WHAT: Declares (or re-attaches) the dedicated "brix_admin_api" SHM zone and
291 * builds the read/write per-IP rules from the merged per-minute values.
292 * Returns NGX_OK, or NGX_ERROR when the zone cannot be created.
293 *
294 * WHY: The throttle is on by default with no configuration, so the zone must
295 * be provisioned by the module itself rather than by an operator
296 * `brix_rate_limit_zone` declaration; brix_rl_zone_add is idempotent, so
297 * multiple dashboard locations share the one segment.
298 *
299 * HOW: 1. brix_rl_zone_add the fixed-name zone (reuses an existing handle).
300 * 2. Build the write and read rules bound to that zone.
301 */
302ngx_int_t
303brix_admin_rl_finalize(ngx_conf_t *cf, ngx_http_brix_dashboard_loc_conf_t *lcf)
304{
305 static ngx_str_t zone_name = ngx_string("brix_admin_api");
306 brix_rl_zone_t *zone = NULL;
307
308 if (brix_rl_zone_add(cf, &zone_name, BRIX_ADMIN_RL_ZONE_SIZE, &zone)
309 != NGX_OK)
310 {
311 return NGX_ERROR;
312 }
313 admin_rl_build_rule(&lcf->admin_rl_write, lcf->admin_rl_write_pm, zone);
314 admin_rl_build_rule(&lcf->admin_rl_read, lcf->admin_rl_read_pm, zone);
315 return NGX_OK;
316}
