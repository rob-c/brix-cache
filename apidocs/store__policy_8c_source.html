44brix_x509_oneline(X509_NAME *name, char *buf, size_t buflen)
45{
46 char *s;
47
48 if (buflen == 0) {
49 return buf;
50 }
51 buf[0] = '\0';
52 if (name == NULL) {
53 return buf;
54 }
55 s = X509_NAME_oneline(name, NULL, 0);
56 if (s != NULL) {
57 size_t n = strlen(s);
58 if (n >= buflen) {
59 n = buflen - 1;
60 }
61 memcpy(buf, s, n);
62 buf[n] = '\0';
63 OPENSSL_free(s);
64 }
65 return buf;
66}

70static int
71sp_entry_push(brix_sp_table_t *t, brix_sp_entry_t e)
72{
73 if (t->nents == t->cap) {
74 size_t ncap = t->cap ? t->cap * 2 : 8;
75 brix_sp_entry_t *ne = realloc(t->ents, ncap * sizeof(*ne));
76 if (ne == NULL) {
77 return -1;
78 }
79 t->ents = ne;
80 t->cap = ncap;
81 }
82 t->ents[t->nents++] = e;
83 return 0;
84}

86/* Read up to BRIX_SP_FILE_MAX bytes of path into a malloc'd buffer. */
87static char *
88sp_read_file(const char *path, size_t *out_len)
89{
90 struct stat st;
91 int fd;
92 char *buf;
93 ssize_t n;
94 size_t total = 0;
95
96 fd = open(path, O_RDONLY | O_CLOEXEC);
97 if (fd < 0) {
98 return NULL;
99 }
100 if (fstat(fd, &st) != 0 || !S_ISREG(st.st_mode)
101 || st.st_size > BRIX_SP_FILE_MAX)
102 {
103 close(fd);
104 return NULL;
105 }
106 buf = malloc((size_t) st.st_size + 1);
107 if (buf == NULL) {
108 close(fd);
109 return NULL;
110 }
111 while (total < (size_t) st.st_size) {
112 n = read(fd, buf + total, (size_t) st.st_size - total);
113 if (n < 0) {
114 free(buf);
115 close(fd);
116 return NULL;
117 }
118 if (n == 0) {
119 break;
120 }
121 total += (size_t) n;
122 }
123 close(fd);
124 buf[total] = '\0';
125 *out_len = total;
126 return buf;
127}

137brix_sp_table_t *
138brix_sp_table_build(const char *cadir, void *log, brix_sp_log_fn log_fn)
139{
140 brix_sp_table_t *t;
141 DIR *dir;
142 struct dirent *ent;
143
144 t = calloc(1, sizeof(*t));
145 if (t == NULL) {
146 return NULL;
147 }
148 if (cadir == NULL) {
149 return t; /* empty table: nothing is ever "present" */
150 }
151
152 dir = opendir(cadir);
153 if (dir == NULL) {
154 return t;
155 }
156
157 while ((ent = readdir(dir)) != NULL) {
158 const char *name = ent->d_name;
159 size_t nlen = strlen(name);
160 const char *suf = ".signing_policy";
161 size_t slen = strlen(suf);
162 char path[PATH_MAX];
163 char *fbuf;
164 size_t flen;
165 char errbuf[256];
166 brix_sp_entry_t e;
167 char *endp;
168 int n;
169
170 if (nlen <= slen || strcmp(name + nlen - slen, suf) != 0) {
171 continue;
172 }
173
174 n = snprintf(path, sizeof(path), "%s/%s", cadir, name);
175 if (n < 0 || (size_t) n >= sizeof(path)) {
176 continue;
177 }
178
179 memset(&e, 0, sizeof(e));
180 e.hash = strtoul(name, &endp, 16);
181 e.have_hash = (endp != name && *endp == '.');
182
183 fbuf = sp_read_file(path, &flen);
184 if (fbuf == NULL) {
185 e.malformed = 1;
186 sp_log(log, log_fn, BRIX_SP_LOG_WARN,
187 "signing_policy: unreadable file (rejecting its CA)");
188 } else {
189 e.policy = brix_sp_parse(fbuf, flen, errbuf, sizeof(errbuf));
190 free(fbuf);
191 if (e.policy == NULL) {
192 e.malformed = 1;
193 sp_log(log, log_fn, BRIX_SP_LOG_WARN,
194 "signing_policy: malformed file (rejecting its CA)");
195 }
196 }
197
198 if (sp_entry_push(t, e) != 0) {
199 if (e.policy != NULL) {
200 brix_sp_free(e.policy);
201 }
202 }
203 }
204
205 closedir(dir);
206 return t;
207}

226/*
227 * Resolve the table entry (if any) whose stem hash matches this CA, trying the
228 * canonical (new) then the legacy (old) subject-name hash.
229 */
230static const brix_sp_entry_t *
231sp_find_by_hash(const brix_sp_table_t *t, X509 *ca)
232{
233 unsigned long h_new = X509_subject_name_hash(ca);
234 unsigned long h_old = X509_subject_name_hash_old(ca);
235
236 for (size_t i = 0; i < t->nents; i++) {
237 if (t->ents[i].have_hash
238 && (t->ents[i].hash == h_new || t->ents[i].hash == h_old))
239 {
240 return &t->ents[i];
241 }
242 }
243 return NULL;
244}

246/*
247 * Fallback: some parsed policy names this CA DN even if its file was oddly
248 * named (stem not hex). Returns the entry, or NULL.
249 */
250static const brix_sp_entry_t *
251sp_find_by_dn(const brix_sp_table_t *t, const char *ca_dn)
252{
253 for (size_t i = 0; i < t->nents; i++) {
254 if (t->ents[i].policy != NULL
255 && brix_sp_ca_dn_present(t->ents[i].policy, ca_dn))
256 {
257 return &t->ents[i];
258 }
259 }
260 return NULL;
261}

263int
264brix_sp_table_check(const brix_sp_table_t *t, brix_sp_mode_t mode,
265 X509 *ca, X509 *subject)
266{
267 const brix_sp_entry_t *e;
268 char ca_dn[1024];
269 char subj_dn[1024];
270
271 if (mode == BRIX_SP_MODE_OFF || t == NULL || ca == NULL || subject == NULL) {
272 return 1;
273 }
274
275 brix_x509_oneline(X509_get_subject_name(ca), ca_dn, sizeof(ca_dn));
276
277 e = sp_find_by_hash(t, ca);
278 if (e == NULL) {
279 e = sp_find_by_dn(t, ca_dn);
280 }
281
282 if (e == NULL) {
283 /* No policy file present for this CA. */
284 return mode == BRIX_SP_MODE_REQUIRE ? 0 : 1;
285 }
286
287 if (e->malformed || e->policy == NULL) {
288 return 0; /* present but unusable → fail closed */
289 }
290
291 if (!brix_sp_ca_dn_present(e->policy, ca_dn)) {
292 return 0; /* file present but names the wrong CA → fail closed */
293 }
294
295 brix_x509_oneline(X509_get_subject_name(subject), subj_dn, sizeof(subj_dn));
296 return brix_sp_subject_allowed(e->policy, ca_dn, subj_dn) ? 1 : 0;
297}
