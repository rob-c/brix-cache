48char *
49brix_x509_oneline(X509_NAME *name, char *buf, size_t buflen)
50{
51 char *s;
52
53 if (buflen == 0) {
54 return buf;
55 }
56 buf[0] = '\0';
57 if (name == NULL) {
58 return buf;
59 }
60 s = X509_NAME_oneline(name, NULL, 0);
61 if (s != NULL) {
62 size_t n = strlen(s);
63 if (n >= buflen) {
64 n = buflen - 1;
65 }
66 memcpy(buf, s, n);
67 buf[n] = '\0';
68 OPENSSL_free(s);
69 }
70 return buf;

75static int
76sp_entry_push(brix_sp_table_t *t, brix_sp_entry_t e)
77{
78 if (t->nents == t->cap) {
79 size_t ncap = t->cap ? t->cap * 2 : 8;
80 brix_sp_entry_t *ne = realloc(t->ents, ncap * sizeof(*ne));
81 if (ne == NULL) {
82 return -1;
83 }
84 t->ents = ne;
85 t->cap = ncap;
86 }
87 t->ents[t->nents++] = e;
88 return 0;

91/* Read up to BRIX_SP_FILE_MAX bytes of path into a malloc'd buffer. */
92static char *
93sp_read_file(const char *path, size_t *out_len)
94{
95 struct stat st;
96 int fd;
97 char *buf;
98 ssize_t n;
99 size_t total = 0;
100
101 fd = open(path, O_RDONLY | O_CLOEXEC);
102 if (fd < 0) {
103 return NULL;
104 }
105 if (fstat(fd, &st) != 0 || !S_ISREG(st.st_mode)
106 || st.st_size > BRIX_SP_FILE_MAX)
107 {
108 close(fd);
109 return NULL;
110 }
111 buf = malloc((size_t) st.st_size + 1);
112 if (buf == NULL) {
113 close(fd);
114 return NULL;
115 }
116 while (total < (size_t) st.st_size) {
117 n = read(fd, buf + total, (size_t) st.st_size - total);
118 if (n < 0) {
119 free(buf);
120 close(fd);
121 return NULL;
122 }
123 if (n == 0) {
124 break;
125 }
126 total += (size_t) n;
127 }
128 close(fd);
129 buf[total] = '\0';
130 *out_len = total;
131 return buf;
132}

142brix_sp_table_t *
143brix_sp_table_build(const char *cadir, void *log, brix_sp_log_fn log_fn)
144{
145 brix_sp_table_t *t;
146 DIR *dir;
147 struct dirent *ent;
148
149 t = calloc(1, sizeof(*t));
150 if (t == NULL) {
151 return NULL;
152 }
153 if (cadir == NULL) {
154 return t; /* empty table: nothing is ever "present" */
155 }
156
157 dir = opendir(cadir);
158 if (dir == NULL) {
159 return t;
160 }
161
162 while ((ent = readdir(dir)) != NULL) {
163 const char *name = ent->d_name;
164 size_t nlen = strlen(name);
165 const char *suf = ".signing_policy";
166 size_t slen = strlen(suf);
167 char path[PATH_MAX];
168 char *fbuf;
169 size_t flen;
170 char errbuf[256];
171 brix_sp_entry_t e;
172 char *endp;
173 int n;
174
175 if (nlen <= slen || strcmp(name + nlen - slen, suf) != 0) {
176 continue;
177 }
178
179 n = snprintf(path, sizeof(path), "%s/%s", cadir, name);
180 if (n < 0 || (size_t) n >= sizeof(path)) {
181 continue;
182 }
183
184 memset(&e, 0, sizeof(e));
185 e.hash = strtoul(name, &endp, 16);
186 e.have_hash = (endp != name && *endp == '.');
187
188 fbuf = sp_read_file(path, &flen);
189 if (fbuf == NULL) {
190 e.malformed = 1;
191 sp_log(log, log_fn, BRIX_SP_LOG_WARN,
192 "signing_policy: unreadable file (rejecting its CA)");
193 } else {
194 e.policy = brix_sp_parse(fbuf, flen, errbuf, sizeof(errbuf));
195 free(fbuf);
196 if (e.policy == NULL) {
197 e.malformed = 1;
198 sp_log(log, log_fn, BRIX_SP_LOG_WARN,
199 "signing_policy: malformed file (rejecting its CA)");
200 }
201 }
202
203 if (sp_entry_push(t, e) != 0) {
204 if (e.policy != NULL) {
205 brix_sp_free(e.policy);
206 }
207 }
208 }
209
210 closedir(dir);
211 return t;
212}

231/*
232 * Resolve the table entry (if any) whose stem hash matches this CA, trying the
233 * canonical (new) then the legacy (old) subject-name hash.
234 */
235static const brix_sp_entry_t *
236sp_find_by_hash(const brix_sp_table_t *t, X509 *ca)
237{
238 unsigned long h_new = X509_subject_name_hash(ca);
239 unsigned long h_old = X509_subject_name_hash_old(ca);
240
241 for (size_t i = 0; i < t->nents; i++) {
242 if (t->ents[i].have_hash
243 && (t->ents[i].hash == h_new || t->ents[i].hash == h_old))
244 {
245 return &t->ents[i];
246 }
247 }
248 return NULL;
249}

251/*
252 * Fallback: some parsed policy names this CA DN even if its file was oddly
253 * named (stem not hex). Returns the entry, or NULL.
254 */
255static const brix_sp_entry_t *
256sp_find_by_dn(const brix_sp_table_t *t, const char *ca_dn)
257{
258 for (size_t i = 0; i < t->nents; i++) {
259 if (t->ents[i].policy != NULL
260 && brix_sp_ca_dn_present(t->ents[i].policy, ca_dn))
261 {
262 return &t->ents[i];
263 }
264 }
265 return NULL;
266}

268int
269brix_sp_table_check(const brix_sp_table_t *t, brix_sp_mode_t mode,
270 X509 *ca, X509 *subject)
271{
272 const brix_sp_entry_t *e;
273 char ca_dn[1024];
274 char subj_dn[1024];
275
276 if (mode == BRIX_SP_MODE_OFF || t == NULL || ca == NULL || subject == NULL) {
277 return 1;
278 }
279
280 brix_x509_oneline(X509_get_subject_name(ca), ca_dn, sizeof(ca_dn));
281
282 e = sp_find_by_hash(t, ca);
283 if (e == NULL) {
284 e = sp_find_by_dn(t, ca_dn);
285 }
286
287 if (e == NULL) {
288 /* No policy file present for this CA. */
289 return mode == BRIX_SP_MODE_REQUIRE ? 0 : 1;
290 }
291
292 if (e->malformed || e->policy == NULL) {
293 return 0; /* present but unusable → fail closed */
294 }
295
296 if (!brix_sp_ca_dn_present(e->policy, ca_dn)) {
297 return 0; /* file present but names the wrong CA → fail closed */
298 }
299
300 brix_x509_oneline(X509_get_subject_name(subject), subj_dn, sizeof(subj_dn));
301 return brix_sp_subject_allowed(e->policy, ca_dn, subj_dn) ? 1 : 0;
302}

307static const char *BRIX_PX_LIMITED_OID = "1.3.6.1.4.1.3536.1.1.1.9";

309brix_px_kind_t
310brix_px_classify(X509 *cert)
311{
312 PROXY_CERT_INFO_EXTENSION *pci;
313 brix_px_kind_t kind = BRIX_PX_NONE;
314
315 if (X509_get_extension_flags(cert) & EXFLAG_PROXY) {
316 kind = BRIX_PX_FULL;
317 pci = X509_get_ext_d2i(cert, NID_proxyCertInfo, NULL, NULL);
318 if (pci != NULL) {
319 char oid[128];
320 int n = OBJ_obj2txt(oid, sizeof(oid),
321 pci->proxyPolicy->policyLanguage, 1);
322 if (n > 0 && strcmp(oid, BRIX_PX_LIMITED_OID) == 0) {
323 kind = BRIX_PX_LIMITED;
324 }
325 PROXY_CERT_INFO_EXTENSION_free(pci);
326 }
327 return kind;
328 }
329
330 /* Legacy Globus proxy: last RDN is CN=proxy or CN=limited proxy. */
331 {
332 X509_NAME *nm = X509_get_subject_name(cert);
333 int last = X509_NAME_entry_count(nm) - 1;
334 if (last >= 0) {
335 X509_NAME_ENTRY *e = X509_NAME_get_entry(nm, last);
336 ASN1_STRING *v = X509_NAME_ENTRY_get_data(e);
337 const unsigned char *s = ASN1_STRING_get0_data(v);
338 int len = ASN1_STRING_length(v);
339 if (len == (int) sizeof("limited proxy") - 1
340 && strncasecmp((const char *) s, "limited proxy", len) == 0)
341 {
342 return BRIX_PX_LIMITED;
343 }
344 if (len == (int) sizeof("proxy") - 1
345 && strncasecmp((const char *) s, "proxy", len) == 0)
346 {
347 return BRIX_PX_FULL;
348 }
349 }
350 }
351 return BRIX_PX_NONE;
352}

354int
355brix_proxy_chain_ok(STACK_OF(X509) *chain)
356{
357 int n, i;
358 int seen_limited = 0;
359
360 if (chain == NULL) {
361 return 1;
362 }
363
364 n = sk_X509_num(chain);
365 for (i = n - 1; i >= 0; i--) { /* root .. leaf */
366 brix_px_kind_t kind = brix_px_classify(sk_X509_value(chain, i));
367
368 if (kind == BRIX_PX_LIMITED) {
369 seen_limited = 1;
370 } else if (kind == BRIX_PX_FULL && seen_limited) {
371 return 0; /* full proxy beneath a limited one — escalation */
372 }
373 }
374 return 1;
375}

379/* Recognised RFC 3820 / Globus proxy policy-language OIDs. */
380static const char *BRIX_PX_OID_IMPERSONATION = "1.3.6.1.5.5.7.21.1";

384static int
385brix_dn_has_control_bytes(X509_NAME *nm)
386{
387 int i, n = X509_NAME_entry_count(nm);
388 for (i = 0; i < n; i++) {
389 X509_NAME_ENTRY *e = X509_NAME_get_entry(nm, i);
390 ASN1_STRING *v = X509_NAME_ENTRY_get_data(e);
391 const unsigned char *s = ASN1_STRING_get0_data(v);
392 int len = ASN1_STRING_length(v);
393 int j;
394 for (j = 0; j < len; j++) {
395 if (s[j] < 0x20 || s[j] == 0x7f) {
396 return 1; /* embedded control or NUL byte */
397 }
398 }
399 }
400 return 0;
401}

403int
404brix_cert_policy_violation(X509 *cert)
405{
406 EVP_PKEY *pk;
407 int is_proxy = (X509_get_extension_flags(cert) & EXFLAG_PROXY) != 0;
408
409 /* Key strength (IGTF: RSA >= 2048, EC >= 256). */
410 pk = X509_get0_pubkey(cert);
411 if (pk != NULL) {
412 int id = EVP_PKEY_base_id(pk);
413 int bits = EVP_PKEY_bits(pk);
414 if ((id == EVP_PKEY_RSA || id == EVP_PKEY_RSA2 || id == EVP_PKEY_DSA)
415 && bits < 2048) {
416 return 1;
417 }
418 if (id == EVP_PKEY_EC && bits < 256) {
419 return 1;
420 }
421 }
422
423 /* Signature algorithm: reject MD5 and SHA-1 based signatures. */
424 {
425 int sig_nid = X509_get_signature_nid(cert);
426 int md_nid = NID_undef, pk_nid = NID_undef;
427 if (OBJ_find_sigid_algs(sig_nid, &md_nid, &pk_nid)) {
428 if (md_nid == NID_md5 || md_nid == NID_sha1 || md_nid == NID_md2
429 || md_nid == NID_md4) {
430 return 1;
431 }
432 }
433 }
434
435 /* Serial number: positive and <= 20 octets (RFC 5280 §4.1.2.2). Proxies
436 * are exempt from the ceiling — grid proxies use large derived serials. */
437 {
438 const ASN1_INTEGER *sn = X509_get0_serialNumber(cert);
439 BIGNUM *bn = ASN1_INTEGER_to_BN(sn, NULL);
440 if (bn != NULL) {
441 int bad = BN_is_zero(bn) || BN_is_negative(bn)
442 || (!is_proxy && BN_num_bytes(bn) > 20);
443 BN_free(bn);
444 if (bad) {
445 return 1;
446 }
447 }
448 }
449
450 /* DN sanity: no embedded control/NUL bytes (RFC 5280 §4.1.2.6). */
451 if (brix_dn_has_control_bytes(X509_get_subject_name(cert))
452 || brix_dn_has_control_bytes(X509_get_issuer_name(cert))) {
453 return 1;
454 }
455
456 return 0;
457}

459int
460brix_leaf_purpose_violation(X509 *leaf)
461{
462 /* extendedKeyUsage: if present, must allow client auth. */
463 EXTENDED_KEY_USAGE *eku = X509_get_ext_d2i(leaf, NID_ext_key_usage, NULL,
464 NULL);
465 if (eku != NULL) {
466 int i, ok = 0;
467 for (i = 0; i < sk_ASN1_OBJECT_num(eku); i++) {
468 int nid = OBJ_obj2nid(sk_ASN1_OBJECT_value(eku, i));
469 if (nid == NID_client_auth || nid == NID_anyExtendedKeyUsage) {
470 ok = 1;
471 break;
472 }
473 }
474 sk_ASN1_OBJECT_pop_free(eku, ASN1_OBJECT_free);
475 if (!ok) {
476 return 1;
477 }
478 }
479
480 /* keyUsage: if present, must assert digitalSignature (RSA/ECDSA client
481 * auth) OR keyAgreement (fixed-ECDH client auth). */
482 {
483 uint32_t flags = X509_get_key_usage(leaf); /* UINT32_MAX if absent */
484 if (flags != UINT32_MAX
485 && !(flags & (KU_DIGITAL_SIGNATURE | KU_KEY_AGREEMENT))) {
486 return 1;
487 }
488 }
489 return 0;
490}

492int
493brix_proxy_pci_ok(STACK_OF(X509) *chain)
494{
495 int n, i;
496
497 if (chain == NULL) {
498 return 1;
499 }
500 n = sk_X509_num(chain);
501 for (i = 0; i < n; i++) {
502 X509 *cert = sk_X509_value(chain, i);
503 int loc = X509_get_ext_by_NID(cert, NID_proxyCertInfo, -1);
504 X509_EXTENSION *ext;
505 PROXY_CERT_INFO_EXTENSION *pci;
506 char oid[128];
507 int ok_oid;
508
509 if (loc < 0) {
510 continue; /* no proxyCertInfo — legacy/EEC handled elsewhere */
511 }
512 ext = X509_get_ext(cert, loc);
513 if (!X509_EXTENSION_get_critical(ext)) {
514 return 0; /* proxyCertInfo present but not critical (RFC 3820 §3.1) */
515 }
516 pci = X509_get_ext_d2i(cert, NID_proxyCertInfo, NULL, NULL);
517 if (pci == NULL) {
518 return 0; /* malformed proxyCertInfo */
519 }
520 oid[0] = '\0';
521 OBJ_obj2txt(oid, sizeof(oid), pci->proxyPolicy->policyLanguage, 1);
522 PROXY_CERT_INFO_EXTENSION_free(pci);
523 ok_oid = (strcmp(oid, BRIX_PX_OID_IMPERSONATION) == 0)
524 || (strcmp(oid, BRIX_PX_OID_INDEPENDENT) == 0)
525 || (strcmp(oid, BRIX_PX_LIMITED_OID) == 0);
526 if (!ok_oid) {
527 return 0; /* unrecognised proxy policy language (RFC 3820 §3.2) */
528 }
529 }
530 return 1;
531}

535/*
536 * WHAT: check_issued override accepting a name-matching issuer for an RFC 3820
537 * proxy subject even when its authorityKeyIdentifier does not match the
538 * issuer's subjectKeyIdentifier.
539 * WHY: xrdgsiproxy/voms-proxy-init copy the EEC's own AKID into the delegated
540 * proxy, so OpenSSL's default check_issued rejects the signing EEC as the
541 * proxy's issuer (AKID_SKID_MISMATCH) and reports "unable to get local
542 * issuer". The reference XRootD chain selects issuers by subject name +
543 * signature; match that.
544 * HOW: defer to X509_check_issued; on the AKID objections, accept when the
545 * subject is a recognised proxy (EXFLAG_PROXY). The RSA signature is
546 * still verified afterwards, so this relaxes selection, not trust.
547 */
548static int
549brix_sp_proxy_check_issued(X509_STORE_CTX *ctx, X509 *subject, X509 *issuer)
550{
551 int rv;
552
553 (void) ctx;
554
555 rv = X509_check_issued(issuer, subject);
556 if (rv == X509_V_OK) {
557 return 1;
558 }
559 /*
560 * The authorityKeyIdentifier is advisory (RFC 5280 §4.2.1.1): a mismatch
561 * must not prevent selecting a name-matching issuer. Accept the AKID/SKID
562 * and AKID issuer-serial objections for ANY subject (not just proxies) —
563 * the issuer's signature over the subject is still verified afterwards by
564 * X509_verify_cert, so this relaxes issuer *selection*, never trust. This
565 * also covers delegated grid proxies whose copied AKID points at the CA.
566 */
567 if (rv == X509_V_ERR_AKID_SKID_MISMATCH
568 || rv == X509_V_ERR_AKID_ISSUER_SERIAL_MISMATCH)
569 {
570 return 1;
571 }
572 return 0;
573}

575/*
576 * WHAT: verify callback used only in BRIX_CRL_MODE_TRY.
577 * WHY: "try" checks revocation where a CRL exists but tolerates a CA that has
578 * none; a stale (expired) CRL stays fatal (staleness is evidence).
579 * HOW: downgrade only X509_V_ERR_UNABLE_TO_GET_CRL to success; every other
580 * verdict (CRL_HAS_EXPIRED, CERT_REVOKED, ...) stands.
581 */
582static int
583brix_crl_try_verify_cb(int ok, X509_STORE_CTX *ctx)
584{
585 if (ok) {
586 return 1;
587 }
588 if (X509_STORE_CTX_get_error(ctx) == X509_V_ERR_UNABLE_TO_GET_CRL) {
589 return 1;
590 }
591 return 0;
592}

594int
595brix_store_configure(X509_STORE *store, const char *cadir,
596 unsigned long extra_flags, int crl_count,
597 brix_sp_mode_t sp_mode, int crl_mode,
598 void *log, brix_sp_log_fn log_fn)
599{
600 brix_sp_table_t *table;
601
602 if (store == NULL) {
603 return -1;
604 }
605
606 if (extra_flags != 0) {
607 X509_STORE_set_flags(store, extra_flags);
608 }
609 /* AKID is advisory: tolerate an AKID mismatch when selecting a name-matching
610 * issuer on every store (webdav and GSI); the signature is still verified. */
611 X509_STORE_set_check_issued(store, brix_sp_proxy_check_issued);
612
613 if (crl_mode == BRIX_CRL_MODE_REQUIRE
614 || (crl_mode == BRIX_CRL_MODE_TRY && crl_count > 0))
615 {
616 X509_STORE_set_flags(store, X509_V_FLAG_CRL_CHECK
617 | X509_V_FLAG_CRL_CHECK_ALL | X509_V_FLAG_USE_DELTAS);
618 }
619 if (crl_mode == BRIX_CRL_MODE_TRY) {
620 X509_STORE_set_verify_cb(store, brix_crl_try_verify_cb);
621 }
622
623 if (cadir == NULL && sp_mode == BRIX_SP_MODE_REQUIRE) {
624 sp_log(log, log_fn, BRIX_SP_LOG_WARN,
625 "signing_policy: \"require\" needs a hashed CA directory, not a "
626 "bundle file");
627 return -1;
628 }
629
630 table = brix_sp_table_build(cadir, log, log_fn);
631 if (table == NULL) {
632 return -1;
633 }
634 if (!brix_store_policy_attach(store, table, sp_mode, crl_mode)) {
635 brix_sp_table_free(table);
636 return -1;
637 }
638 return 0;
639}

672int
673brix_store_policy_attach(X509_STORE *store, brix_sp_table_t *table,
674 brix_sp_mode_t sp_mode, int crl_mode)
675{
676 brix_store_policy_t *sp;
677 int idx = sp_store_ex_index();
678
679 if (store == NULL || idx < 0) {
680 return 0;
681 }
682 sp = calloc(1, sizeof(*sp));
683 if (sp == NULL) {
684 return 0;
685 }
686 sp->table = table;
