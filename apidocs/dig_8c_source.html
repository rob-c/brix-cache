1/*
2 * dig.c — XrdDig remote diagnostics handler (§3). See dig.h.
3 *
4 * Security model (all enforced here, fail-closed):
5 * - default off (brix_webdav_dig off → handler declines → normal 404 path).
6 * - read-only: only GET/HEAD; any other method → 405.
7 * - confinement: the kernel openat2(RESOLVE_BENEATH) primitive
8 * (brix_open_beneath) anchored at the export's realpath — "../" and symlink
9 * escapes are impossible regardless of the requested relative path.
10 * - authorization: a principal→export allow-file; an anonymous principal, an
11 * unset/unreadable allow-file, or no matching rule all DENY (403).
12 */
13

28/* The authenticated principal: token subject if present, else GSI DN; NULL when
29 * the request is anonymous (no usable identity). */
30static const char *
31dig_principal(ngx_http_brix_webdav_req_ctx_t *ctx)
32{
33 const char *p;
34
35 if (ctx == NULL || ctx->identity == NULL) {
36 return NULL;
37 }
38 p = brix_identity_subject_cstr(ctx->identity);
39 if (p != NULL && p[0] != '\0') {
40 return p;
41 }
42 p = brix_identity_dn_cstr(ctx->identity);
43 if (p != NULL && p[0] != '\0') {
44 return p;
45 }
46 return NULL;
47}

49/*
50 * dig_authorize — fail-closed allow-file check.
51 *
52 * Grammar (first matching rule wins; '#' starts a comment):
53 * <principal-or-*> <export>[,<export>...]
54 * '*' matches any non-anonymous principal. Returns NGX_OK only on an explicit
55 * match; every other outcome (no file, unopenable, anonymous, no rule) denies.
56 */
57static ngx_int_t
58dig_authorize(const char *authfile, const char *principal, const char *export)
59{
60 FILE *fp;
61 char line[1024];
62 int fd;
63 ngx_int_t verdict = NGX_ERROR;
64
65 if (authfile == NULL || authfile[0] == '\0' || principal == NULL) {
66 return NGX_ERROR;
67 }
68 fd = open(authfile, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
69 if (fd < 0) {
70 return NGX_ERROR;
71 }
72 fp = fdopen(fd, "r");
73 if (fp == NULL) {
74 close(fd);
75 return NGX_ERROR;
76 }
77 while (fgets(line, sizeof(line), fp) != NULL) {
78 char *hash = strchr(line, '#');
79 char *who, *exports, *save = NULL, *save2 = NULL, *tok;
80 int matched = 0;
81
82 if (hash != NULL) {
83 *hash = '\0';
84 }
85 who = strtok_r(line, " \t\r\n", &save);
86 exports = strtok_r(NULL, " \t\r\n", &save);
87 if (who == NULL || exports == NULL) {
88 continue;
89 }
90 if (strcmp(who, "*") != 0 && strcmp(who, principal) != 0) {
91 continue;
92 }
93 for (tok = strtok_r(exports, ",", &save2); tok != NULL;
94 tok = strtok_r(NULL, ",", &save2))
95 {
96 if (strcmp(tok, export) == 0) {
97 matched = 1;
98 break;
99 }
100 }
101 if (matched) {
102 verdict = NGX_OK;
103 break;
104 }
105 }
106 fclose(fp);
107 return verdict;
108}

110/* Map a confined-open errno to an HTTP status. A RESOLVE_BENEATH escape attempt
111 * surfaces as EXDEV/ELOOP/EACCES → 403; a genuine miss → 404. */
112static ngx_int_t
113dig_open_status(int err)
114{
115 switch (err) {
116 case ENOENT:
117 case ENOTDIR:
118 return NGX_HTTP_NOT_FOUND;
119 default:
120 return NGX_HTTP_FORBIDDEN; /* EXDEV/ELOOP/EACCES/EPERM = blocked escape */
121 }
122}

124ngx_int_t
125brix_dig_handle(ngx_http_request_t *r)
126{
127 ngx_http_brix_webdav_loc_conf_t *conf;
128 ngx_http_brix_webdav_req_ctx_t *ctx;
129 brix_dig_export_t *exports, *match = NULL;
130 const u_char *rest, *end, *sl;
131 char export[128];
132 char rel[PATH_MAX];
133 char canon[PATH_MAX];
134 const char *principal;
135 size_t exlen, rlen;
136 ngx_uint_t i;
137 int fd;
138 struct stat st;
139 brix_vfs_ctx_t vctx;
140 brix_vfs_file_t *fh;
141 brix_vfs_stat_t vst;
142 char full[PATH_MAX];
143 int vfs_err = 0;
144 int is_tls = 0;
145 ngx_int_t rc;
146 ngx_pool_cleanup_t *cln;
147 ngx_pool_cleanup_file_t *clnf;
148 ngx_file_t *file;
149 ngx_buf_t *b;
150 ngx_chain_t out;
151
152 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
153 if (!conf->dig_enable) {
154 return NGX_DECLINED;
155 }
156 if (r->uri.len <= BRIX_DIG_PREFIX_LEN
157 || ngx_strncmp(r->uri.data, (u_char *) BRIX_DIG_PREFIX,
158 BRIX_DIG_PREFIX_LEN) != 0)
159 {
160 return NGX_DECLINED;
161 }
162 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
163 return NGX_HTTP_NOT_ALLOWED;
164 }
165
166 /* Split "<export>/<rel>" (r->uri is decoded but NOT NUL-terminated). */
167 rest = r->uri.data + BRIX_DIG_PREFIX_LEN;
168 end = r->uri.data + r->uri.len;
169 sl = memchr(rest, '/', (size_t) (end - rest));
170 if (sl == NULL || sl == rest) {
171 return NGX_HTTP_NOT_FOUND; /* need both an export and a file */
172 }
173 exlen = (size_t) (sl - rest);
174 rlen = (size_t) (end - (sl + 1));
175 if (exlen == 0 || exlen >= sizeof(export)
176 || rlen == 0 || rlen >= sizeof(rel))
177 {
178 return NGX_HTTP_NOT_FOUND;
179 }
180 ngx_memcpy(export, rest, exlen);
181 export[exlen] = '\0';
182 ngx_memcpy(rel, sl + 1, rlen);
183 rel[rlen] = '\0';
184
185 /* Resolve the export name → its realpath anchor. */
186 if (conf->dig_exports == NULL) {
187 return NGX_HTTP_NOT_FOUND;
188 }
189 exports = conf->dig_exports->elts;
190 for (i = 0; i < conf->dig_exports->nelts; i++) {
191 if (exports[i].name.len == exlen
192 && ngx_strncmp(exports[i].name.data, (u_char *) export, exlen) == 0)
193 {
194 match = &exports[i];
195 break;
196 }
197 }
198 if (match == NULL) {
199 return NGX_HTTP_NOT_FOUND;
200 }
201
202 /* Authorize (fail-closed). */
203 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
204 principal = dig_principal(ctx);
205 if (dig_authorize((const char *) conf->dig_auth_file.data, principal,
206 export) != NGX_OK)
207 {
208 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
209 "xrootd dig: deny principal=\"%s\" export=\"%s\"",
210 principal ? principal : "(anonymous)", export);
211 return NGX_HTTP_FORBIDDEN;
212 }
213
214 /* Serve through the VFS seam, confined under the export realpath
215 * (brix_vfs_open uses the same kernel RESOLVE_BENEATH open underneath). */
216 if (brix_str_cbuf(canon, sizeof(canon), &match->canon) == NULL) {
217 return NGX_HTTP_INTERNAL_SERVER_ERROR;
218 }
219
220 if ((size_t) snprintf(full, sizeof(full), "%s/%s", canon, rel)
221 >= sizeof(full))
222 {
223 return NGX_HTTP_INTERNAL_SERVER_ERROR;
224 }
225#if (NGX_HTTP_SSL)
226 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
227#endif
228 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
229 canon, NULL, 0 /* allow_write */, is_tls, NULL, full);
230
231 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
232 if (fh == NULL) {
233 return dig_open_status(vfs_err);
234 }
235 if (brix_vfs_file_stat(fh, &vst) != NGX_OK || !vst.is_regular) {
236 brix_vfs_close(fh, r->connection->log);
237 return NGX_HTTP_NOT_FOUND; /* directories/specials not served */
238 }
239 ngx_memzero(&st, sizeof(st));
240 st.st_size = vst.size;
241 st.st_mtime = vst.mtime;
242 st.st_mode = (mode_t) vst.mode;
243
244 /* Zero-copy serve fd, gated on the backend's CAP_SENDFILE (the pool cleanup
245 * below closes it; the VFS handle owns no other resource). */
246 fd = brix_vfs_file_sendfile_fd(fh);
247 if (fd == NGX_INVALID_FILE) {
248 brix_vfs_close(fh, r->connection->log);
249 return NGX_HTTP_INTERNAL_SERVER_ERROR;
250 }
251
252 /* Ensure the fd is closed when the request pool is destroyed (after the
253 * response is fully sent, including across events for large files). */
254 cln = ngx_pool_cleanup_add(r->pool, sizeof(ngx_pool_cleanup_file_t));
255 if (cln == NULL) {
256 close(fd);
257 return NGX_HTTP_INTERNAL_SERVER_ERROR;
258 }
259 cln->handler = ngx_pool_cleanup_file;
260 clnf = cln->data;
261 clnf->fd = fd;
262 clnf->name = (u_char *) "dig";
263 clnf->log = r->connection->log;
264
265 r->headers_out.status = NGX_HTTP_OK;
266 r->headers_out.content_length_n = st.st_size;
267 r->headers_out.last_modified_time = st.st_mtime;
268 r->headers_out.content_type.len = sizeof("text/plain") - 1;
269 r->headers_out.content_type.data = (u_char *) "text/plain";
270 r->headers_out.content_type_len = r->headers_out.content_type.len;
271
272 rc = ngx_http_send_header(r);
273 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only || st.st_size == 0) {
274 return rc;
275 }
276
277 file = ngx_pcalloc(r->pool, sizeof(ngx_file_t));
278 b = ngx_pcalloc(r->pool, sizeof(ngx_buf_t));
279 if (file == NULL || b == NULL) {
280 return NGX_HTTP_INTERNAL_SERVER_ERROR;
281 }
282 file->fd = fd;
283 file->log = r->connection->log;
284
285 b->file = file;
286 b->file_pos = 0;
287 b->file_last = st.st_size;
288 b->in_file = 1;
289 b->last_buf = (r == r->main) ? 1 : 0;
290 b->last_in_chain = 1;
291
292 out.buf = b;
293 out.next = NULL;
294 return ngx_http_output_filter(r, &out);
295}
