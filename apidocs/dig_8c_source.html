1/*
2 * dig.c — XrdDig remote diagnostics handler (§3). See dig.h.
3 *
4 * Security model (all enforced here, fail-closed):
5 * - default off (brix_webdav_dig off → handler declines → normal 404 path).
6 * - read-only: only GET/HEAD; any other method → 405.
7 * - confinement: the kernel openat2(RESOLVE_BENEATH) primitive
8 * (brix_open_beneath) anchored at the export's realpath — "../" and symlink
9 * escapes are impossible regardless of the requested relative path.
10 * - authorization: a principal→export allow-file; an anonymous principal, an
11 * unset/unreadable allow-file, or no matching rule all DENY (403).
12 */

28/* The authenticated principal: token subject if present, else GSI DN; NULL when
29 * the request is anonymous (no usable identity). */
30static const char *
31dig_principal(ngx_http_brix_webdav_req_ctx_t *ctx)
32{
33 const char *p;
34
35 if (ctx == NULL || ctx->identity == NULL) {
36 return NULL;
37 }
38 p = brix_identity_subject_cstr(ctx->identity);
39 if (p != NULL && p[0] != '\0') {
40 return p;
41 }
42 p = brix_identity_dn_cstr(ctx->identity);
43 if (p != NULL && p[0] != '\0') {
44 return p;
45 }
46 return NULL;
47}

49/*
50 * dig_authorize — fail-closed allow-file check.
51 *
52 * Grammar (first matching rule wins; '#' starts a comment):
53 * <principal-or-*> <export>[,<export>...]
54 * '*' matches any non-anonymous principal. Returns NGX_OK only on an explicit
55 * match; every other outcome (no file, unopenable, anonymous, no rule) denies.
56 */
57static ngx_int_t
58dig_authorize(const char *authfile, const char *principal, const char *export)
59{
60 FILE *fp;
61 char line[1024];
62 int fd;
63 ngx_int_t verdict = NGX_ERROR;
64
65 if (authfile == NULL || authfile[0] == '\0' || principal == NULL) {
66 return NGX_ERROR;
67 }
68 fd = open(authfile, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
69 if (fd < 0) {
70 return NGX_ERROR;
71 }
72 fp = fdopen(fd, "r");
73 if (fp == NULL) {
74 close(fd);
75 return NGX_ERROR;
76 }
77 while (fgets(line, sizeof(line), fp) != NULL) {
78 char *hash = strchr(line, '#');
79 char *who, *exports, *save = NULL, *save2 = NULL, *tok;
80 int matched = 0;
81
82 if (hash != NULL) {
83 *hash = '\0';
84 }
85 who = strtok_r(line, " \t\r\n", &save);
86 exports = strtok_r(NULL, " \t\r\n", &save);
87 if (who == NULL || exports == NULL) {
88 continue;
89 }
90 if (strcmp(who, "*") != 0 && strcmp(who, principal) != 0) {
91 continue;
92 }
93 for (tok = strtok_r(exports, ",", &save2); tok != NULL;
94 tok = strtok_r(NULL, ",", &save2))
95 {
96 if (strcmp(tok, export) == 0) {
97 matched = 1;
98 break;
99 }
100 }
101 if (matched) {
102 verdict = NGX_OK;
103 break;
104 }
105 }
106 /* Read-only stream: no buffered writes can be lost, so a close error
107 * cannot change the already-computed verdict. */
108 (void) fclose(fp);
109 return verdict;
110}

112/* Map a confined-open errno to an HTTP status. A RESOLVE_BENEATH escape attempt
113 * surfaces as EXDEV/ELOOP/EACCES → 403; a genuine miss → 404. */
114static ngx_int_t
115dig_open_status(int err)
116{
117 switch (err) {
118 case ENOENT:
119 case ENOTDIR:
120 return NGX_HTTP_NOT_FOUND;
121 default:
122 return NGX_HTTP_FORBIDDEN; /* EXDEV/ELOOP/EACCES/EPERM = blocked escape */
123 }
124}

126/*
127 * dig_req_t — the small per-request working set threaded across the phase
128 * helpers of brix_dig_handle().
129 *
130 * WHAT: carries the parsed target (export name + relative file), the matched
131 * export descriptor, the opened VFS handle + its stat, and the serve fd
132 * from one phase to the next.
133 * WHY: keeps each phase helper a single-responsibility step with explicit
134 * inputs/outputs (coding-standards §8) instead of a wall of locals in one
135 * giant function; state is passed, never reached for.
136 * HOW: the orchestrator zero-inits one on the stack and hands its address to
137 * each step, which fills the fields it owns.
138 */
139typedef struct {
140 char export[128]; /* NUL-terminated export name */
141 char rel[PATH_MAX]; /* NUL-terminated relative file */
142 brix_dig_export_t *match; /* matched export descriptor */
143 brix_vfs_file_t *fh; /* opened VFS handle */
144 brix_vfs_stat_t vst; /* stat of the opened file */
145 int fd; /* zero-copy serve fd */
146} dig_req_t;

148/*
149 * dig_precheck — decline/deny the request before any parsing work.
150 *
151 * WHAT: gates on the feature flag, the reserved URI prefix, and the read-only
152 * method allow-list (GET/HEAD).
153 * WHY: a single fail-fast gate keeps the security contract (default-off,
154 * read-only) in one auditable place at the top of the flow.
155 * HOW: returns NGX_DECLINED (not ours), NGX_HTTP_NOT_ALLOWED (bad method), or
156 * NGX_OK (proceed).
157 */
158static ngx_int_t
159dig_precheck(ngx_http_request_t *r, ngx_http_brix_webdav_loc_conf_t *conf)
160{
161 if (!conf->dig_enable) {
162 return NGX_DECLINED;
163 }
164 if (r->uri.len <= BRIX_DIG_PREFIX_LEN
165 || ngx_strncmp(r->uri.data, (u_char *) BRIX_DIG_PREFIX,
166 BRIX_DIG_PREFIX_LEN) != 0)
167 {
168 return NGX_DECLINED;
169 }
170 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
171 return NGX_HTTP_NOT_ALLOWED;
172 }
173 return NGX_OK;
174}

176/*
177 * dig_parse_target — split the URI tail into "<export>/<rel>".
178 *
179 * WHAT: extracts the export name and the relative file path from the decoded
180 * (but NOT NUL-terminated) r->uri, NUL-terminating both into req.
181 * WHY: pure parsing isolated from I/O so the length/bounds checks that keep
182 * the fixed buffers safe are trivially reviewable.
183 * HOW: finds the first '/' after the prefix; both halves must be non-empty and
184 * fit their buffers, else NGX_HTTP_NOT_FOUND.
185 */
186static ngx_int_t
187dig_parse_target(ngx_http_request_t *r, dig_req_t *req)
188{
189 const u_char *rest, *end, *sl;
190 size_t exlen, rlen;
191
192 rest = r->uri.data + BRIX_DIG_PREFIX_LEN;
193 end = r->uri.data + r->uri.len;
194 sl = memchr(rest, '/', (size_t) (end - rest));
195 if (sl == NULL || sl == rest) {
196 return NGX_HTTP_NOT_FOUND; /* need both an export and a file */
197 }
198 exlen = (size_t) (sl - rest);
199 rlen = (size_t) (end - (sl + 1));
200 if (exlen == 0 || exlen >= sizeof(req->export)
201 || rlen == 0 || rlen >= sizeof(req->rel))
202 {
203 return NGX_HTTP_NOT_FOUND;
204 }
205 ngx_memcpy(req->export, rest, exlen);
206 req->export[exlen] = '\0';
207 ngx_memcpy(req->rel, sl + 1, rlen);
208 req->rel[rlen] = '\0';
209 return NGX_OK;
210}

212/*
213 * dig_match_export — resolve the export name to its declared descriptor.
214 *
215 * WHAT: linear-scans the configured export list for a name equal to req->export
216 * and stores the match (the realpath BENEATH anchor lives on it).
217 * WHY: an unknown export is a genuine miss, not an error — keeping the lookup
218 * separate makes the 404 semantics explicit.
219 * HOW: NGX_OK with req->match set, or NGX_HTTP_NOT_FOUND.
220 */
221static ngx_int_t
222dig_match_export(ngx_http_brix_webdav_loc_conf_t *conf, dig_req_t *req)
223{
224 brix_dig_export_t *exports;
225 size_t exlen;
226 ngx_uint_t i;
227
228 if (conf->dig_exports == NULL) {
229 return NGX_HTTP_NOT_FOUND;
230 }
231 exlen = ngx_strlen(req->export);
232 exports = conf->dig_exports->elts;
233 for (i = 0; i < conf->dig_exports->nelts; i++) {
234 if (exports[i].name.len == exlen
235 && ngx_strncmp(exports[i].name.data,
236 (u_char *) req->export, exlen) == 0)
237 {
238 req->match = &exports[i];
239 return NGX_OK;
240 }
241 }
242 return NGX_HTTP_NOT_FOUND;
243}

245/*
246 * dig_authz — fail-closed authorization for the resolved export.
247 *
248 * WHAT: derives the request principal and checks it against the allow-file for
249 * the requested export, logging a WARN on denial.
250 * WHY: concentrates the fail-closed decision (anonymous / no file / no rule all
251 * deny) at one edge, with the deny log string frozen.
252 * HOW: NGX_OK on an explicit allow, else NGX_HTTP_FORBIDDEN.
253 */
254static ngx_int_t
255dig_authz(ngx_http_request_t *r, ngx_http_brix_webdav_loc_conf_t *conf,
256 dig_req_t *req)
257{
258 ngx_http_brix_webdav_req_ctx_t *ctx;
259 const char *principal;
260
261 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
262 principal = dig_principal(ctx);
263 if (dig_authorize((const char *) conf->dig_auth_file.data, principal,
264 req->export) != NGX_OK)
265 {
266 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
267 "xrootd dig: deny principal=\"%s\" export=\"%s\"",
268 principal ? principal : "(anonymous)", req->export);
269 return NGX_HTTP_FORBIDDEN;
270 }
271 return NGX_OK;
272}

274/*
275 * dig_open_confined — open the target file through the VFS seam.
276 *
277 * WHAT: builds the confined VFS context under the export realpath, opens the
278 * file, stats it (must be a regular file), and captures the serve fd.
279 * WHY: the confinement anchor (RESOLVE_BENEATH under match->canon) is the load-
280 * bearing security seam — keeping open+stat+fd together makes the handle
281 * lifecycle (close-on-failure) local and correct.
282 * HOW: on success stores req->fh / req->vst / req->fd and returns NGX_OK; on
283 * failure closes any partial handle and returns the mapped HTTP status.
284 */
285static ngx_int_t
286dig_open_confined(ngx_http_request_t *r, dig_req_t *req)
287{
288 char canon[PATH_MAX];
289 char full[PATH_MAX];
290 brix_vfs_ctx_t vctx;
291 int vfs_err = 0;
292 int is_tls = 0;
293
294 /* Serve through the VFS seam, confined under the export realpath
295 * (brix_vfs_open uses the same kernel RESOLVE_BENEATH open underneath). */
296 if (brix_str_cbuf(canon, sizeof(canon), &req->match->canon) == NULL) {
297 return NGX_HTTP_INTERNAL_SERVER_ERROR;
298 }
299 if ((size_t) snprintf(full, sizeof(full), "%s/%s", canon, req->rel)
300 >= sizeof(full))
301 {
302 return NGX_HTTP_INTERNAL_SERVER_ERROR;
303 }
304#if (NGX_HTTP_SSL)
305 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
306#endif
307 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
308 canon, NULL, 0 /* allow_write */, is_tls, NULL, full);
309
310 req->fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
311 if (req->fh == NULL) {
312 return dig_open_status(vfs_err);
313 }
314 if (brix_vfs_file_stat(req->fh, &req->vst) != NGX_OK
315 || !req->vst.is_regular)
316 {
317 brix_vfs_close(req->fh, r->connection->log);
318 return NGX_HTTP_NOT_FOUND; /* directories/specials not served */
319 }
320
321 /* Zero-copy serve fd, gated on the backend's CAP_SENDFILE (a pool cleanup
322 * closes it; the VFS handle owns no other resource). */
323 req->fd = brix_vfs_file_sendfile_fd(req->fh);
324 if (req->fd == NGX_INVALID_FILE) {
325 brix_vfs_close(req->fh, r->connection->log);
326 return NGX_HTTP_INTERNAL_SERVER_ERROR;
327 }
328 return NGX_OK;
329}

331/*
332 * dig_register_fd_cleanup — tie the serve fd's lifetime to the request pool.
333 *
334 * WHAT: registers a pool cleanup that closes req->fd when the request pool is
335 * destroyed (after the response is fully sent, across events for large
336 * files).
337 * WHY: the fd must outlive this function (sendfile streams asynchronously) yet
338 * never leak — the pool cleanup is the correct owner.
339 * HOW: NGX_OK on success; on allocation failure closes the fd and returns
340 * NGX_HTTP_INTERNAL_SERVER_ERROR.
341 */
342static ngx_int_t
343dig_register_fd_cleanup(ngx_http_request_t *r, int fd)
344{
345 ngx_pool_cleanup_t *cln;
346 ngx_pool_cleanup_file_t *clnf;
347
348 cln = ngx_pool_cleanup_add(r->pool, sizeof(ngx_pool_cleanup_file_t));
349 if (cln == NULL) {
350 close(fd);
351 return NGX_HTTP_INTERNAL_SERVER_ERROR;
352 }
353 cln->handler = ngx_pool_cleanup_file;
354 clnf = cln->data;
355 clnf->fd = fd;
356 clnf->name = (u_char *) "dig";
357 clnf->log = r->connection->log;
358 return NGX_OK;
359}

361/*
362 * dig_send_response — emit headers and (for GET) stream the file body.
363 *
364 * WHAT: sets the 200 response headers from the captured stat, sends them, and —
365 * unless header-only or empty — appends a single sendfile buffer to the
366 * output chain.
367 * WHY: the response edge (all the side effects) lives in one place; the byte
368 * layout, content-type, and header set are frozen.
369 * HOW: returns the header rc directly when there is no body to stream, else the
370 * ngx_http_output_filter result.
371 */
372static ngx_int_t
373dig_send_response(ngx_http_request_t *r, dig_req_t *req)
374{
375 ngx_file_t *file;
376 ngx_buf_t *b;
377 ngx_chain_t out;
378 ngx_int_t rc;
379 off_t size = (off_t) req->vst.size;
380
381 r->headers_out.status = NGX_HTTP_OK;
382 r->headers_out.content_length_n = size;
383 r->headers_out.last_modified_time = (time_t) req->vst.mtime;
384 r->headers_out.content_type.len = sizeof("text/plain") - 1;
385 r->headers_out.content_type.data = (u_char *) "text/plain";
386 r->headers_out.content_type_len = r->headers_out.content_type.len;
387
388 rc = ngx_http_send_header(r);
389 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only || size == 0) {
390 return rc;
391 }
392
393 file = ngx_pcalloc(r->pool, sizeof(ngx_file_t));
394 b = ngx_pcalloc(r->pool, sizeof(ngx_buf_t));
395 if (file == NULL || b == NULL) {
396 return NGX_HTTP_INTERNAL_SERVER_ERROR;
397 }
398 file->fd = req->fd;
399 file->log = r->connection->log;
400
401 b->file = file;
402 b->file_pos = 0;
403 b->file_last = size;
404 b->in_file = 1;
405 b->last_buf = (r == r->main) ? 1 : 0;
406 b->last_in_chain = 1;
407
408 out.buf = b;
409 out.next = NULL;
410 return ngx_http_output_filter(r, &out);
411}

413/*
414 * brix_dig_handle — XrdDig diagnostics entry point (see dig.h).
415 *
416 * WHAT: orchestrates the fixed sequence precheck → parse → match → authorize →
417 * open (confined) → register fd cleanup → send response.
418 * WHY: each step is a single-responsibility helper; the handler reads as a flat
419 * rc-and-early-return chain (coding-standards §8), the complexity living
420 * in the named steps.
421 * HOW: any step returning non-NGX_OK is the final result (NGX_DECLINED / an
422 * HTTP status / the output-filter rc); NGX_OK means proceed.
423 */
424ngx_int_t
425brix_dig_handle(ngx_http_request_t *r)
426{
427 ngx_http_brix_webdav_loc_conf_t *conf;
428 dig_req_t req;
429 ngx_int_t rc;
430
431 ngx_memzero(&req, sizeof(req));
432
433 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
434
435 rc = dig_precheck(r, conf);
436 if (rc != NGX_OK) {
437 return rc;
438 }
439 rc = dig_parse_target(r, &req);
440 if (rc != NGX_OK) {
441 return rc;
442 }
443 rc = dig_match_export(conf, &req);
444 if (rc != NGX_OK) {
445 return rc;
446 }
447 rc = dig_authz(r, conf, &req);
448 if (rc != NGX_OK) {
449 return rc;
450 }
451 rc = dig_open_confined(r, &req);
452 if (rc != NGX_OK) {
453 return rc;
454 }
455 rc = dig_register_fd_cleanup(r, req.fd);
456 if (rc != NGX_OK) {
457 return rc;
458 }
459 return dig_send_response(r, &req);
460}
