1/*
2 * gsi_upstream_login.c — Phase-4b GSI delegation: threaded blocking login to the
3 * upstream AS THE USER + fd handoff into the async relay (Tasks 3+4).
4 *
5 * WHY: the proven in-process GSI client (the cache origin client) is BLOCKING —
6 * it runs the certreq/cert handshake on a thread-pool worker. The terminating
7 * proxy is event-driven. So for `brix_tap_proxy_auth gsi` we offload the whole
8 * connect+GSI-login (presenting the client's DELEGATED proxy) to a thread, then
9 * promote the authenticated fd to the proxy's async forward/relay.
10 *
11 * HOW: thread body builds a synthetic origin conf (host/port + the delegated proxy
12 * path + the proxy server's CA store) and reuses brix_cache_origin_connect +
13 * _bootstrap; on success it transfers the authenticated fd. The completion
14 * handler (event loop) wraps that fd in an ngx_connection_t already at IDLE
15 * (bootstrap done), unlinks the temp credential, and dispatches the saved
16 * request. Lifecycle guarded by client_ctx->destroyed (the worker may finish
17 * after the client vanished). GSI auth is cleartext-data here (tls=0), which is
18 * what the tap observes and what the fd handoff supports.
19 */

48/* Thread-pool worker: blocking connect + GSI login presenting the delegated proxy.
49 * Reuses the cache origin client over a throwaway synthetic conf. On success the
50 * authenticated fd is transferred out of `oc` so origin_close does not take it. */
51static void
52proxy_gsi_login_thread(void *data, ngx_log_t *log)
53{
54 proxy_gsi_login_t *g = data;
55 ngx_stream_brix_srv_conf_t *synth = calloc(1, sizeof(*synth));
56 brix_cache_fill_t *t = calloc(1, sizeof(*t));
57 brix_cache_origin_conn_t oc;
58
59 (void) log;
60 g->result_fd = -1;
61
62 if (synth == NULL || t == NULL) {
63 free(synth);
64 free(t);
65 return;
66 }
67
68 oc.fd = -1;
69 oc.ssl = NULL;
70 oc.ssl_ctx = NULL;
71
72 synth->cache_origin_host.data = (u_char *) g->host;
73 synth->cache_origin_host.len = ngx_strlen(g->host);
74 synth->cache_origin_port = g->port;
75 synth->cache_origin_tls = 0; /* cleartext GSI auth */
76 synth->cache_origin_family = (ngx_uint_t) g->family;
77 synth->cache_origin_x509_proxy.data = (u_char *) g->deleg_path;
78 synth->cache_origin_x509_proxy.len = ngx_strlen(g->deleg_path);
79 synth->gsi_store = g->gsi_store; /* borrowed; do not free */
80 t->conf = synth;
81
82 if (brix_cache_origin_connect(t, &oc) == 0
83 && brix_cache_origin_bootstrap(t, &oc) == 0)
84 {
85 g->result_fd = oc.fd; /* transfer ownership */
86 oc.fd = -1;
87 }
88
89 brix_cache_origin_close(&oc); /* frees SSL (if any); fd already taken */
90 free(synth);
91 free(t);
92}

95/* Build the upstream ngx_connection_t around the authenticated fd (already
96 * post-login), wired to the proxy relay handlers, at IDLE. Returns NGX_OK. */
97static ngx_int_t
98proxy_gsi_promote_fd(brix_proxy_ctx_t *proxy, int fd)
99{
100 ngx_connection_t *client_conn = proxy->client_conn;
101 ngx_connection_t *uconn;
102
103 uconn = ngx_get_connection(fd, client_conn->log);
104 if (uconn == NULL) {
105 return NGX_ERROR;
106 }
107 uconn->pool = ngx_create_pool(512, client_conn->log);
108 if (uconn->pool == NULL) {
109 ngx_free_connection(uconn);
110 return NGX_ERROR;
111 }
112 if (ngx_nonblocking(fd) == -1) {
113 ngx_destroy_pool(uconn->pool);
114 ngx_free_connection(uconn);
115 return NGX_ERROR;
116 }
117
118 uconn->data = proxy;
119 uconn->recv = ngx_recv;
120 uconn->send = ngx_send;
121 uconn->recv_chain = ngx_recv_chain;
122 uconn->send_chain = ngx_send_chain;
123 uconn->log = client_conn->log;
124 uconn->read->log = client_conn->log;
125 uconn->write->log = client_conn->log;
126 uconn->read->handler = brix_proxy_read_handler;
127 uconn->write->handler = brix_proxy_write_handler;
128
129 proxy->conn = uconn;
130 proxy->state = XRD_PX_IDLE; /* bootstrap already done in the thread */
131 proxy->from_pool = 1; /* => skip the bootstrap path */
132
133 if (ngx_handle_read_event(uconn->read, 0) != NGX_OK) {
134 return NGX_ERROR;
135 }
136 return NGX_OK;
137}

140/* Event-loop completion: the thread finished. Unlink the temp credential, then —
141 * if the client is still alive — promote the authenticated fd and dispatch. */
142static void
143proxy_gsi_login_done(ngx_event_t *ev)
144{
145 ngx_thread_task_t *task = ev->data;
146 proxy_gsi_login_t *g = task->ctx;
147 brix_proxy_ctx_t *proxy = g->proxy;
148
149 if (g->deleg_path[0] != '\0') {
150 (void) unlink(g->deleg_path); /* vfs-seam-allow: config-domain delegated GSI proxy credential temp (not export storage) */
151 }
152
153 /* Client vanished while we were logging in: drop the authed fd and stop. */
154 if (proxy->client_ctx == NULL || proxy->client_ctx->destroyed) {
155 if (g->result_fd >= 0) {
156 ngx_close_socket(g->result_fd);
157 }
158 return;
159 }
160
161 if (g->result_fd < 0) {
162 brix_proxy_abort(proxy, "proxy: GSI upstream login failed");
163 return;
164 }
165
166 if (proxy_gsi_promote_fd(proxy, g->result_fd) != NGX_OK) {
167 ngx_close_socket(g->result_fd);
168 brix_proxy_abort(proxy, "proxy: GSI upstream handoff failed");
169 return;
170 }
171
172 if (proxy->saved_req != NULL) {
173 brix_proxy_dispatch_pending(proxy);
174 } else {
175 proxy->client_ctx->state = XRD_ST_REQ_HEADER;
176 (void) brix_schedule_read_resume(proxy->client_conn);
177 }
178}

182brix_proxy_gsi_connect_async(brix_proxy_ctx_t *proxy,
183 ngx_stream_brix_srv_conf_t *conf, ngx_str_t *host, uint16_t port)
184{
185 ngx_connection_t *c = proxy->client_conn;
186 brix_ctx_t *ctx = proxy->client_ctx;
187 ngx_thread_task_t *task;
188 proxy_gsi_login_t *g;
189
190 if (conf->common.thread_pool == NULL) {
191 ngx_log_error(NGX_LOG_ERR, c->log, 0,
192 "brix_tap_proxy_auth gsi: a thread_pool is required");
193 return NGX_ERROR;
194 }
195 if (ctx->gsi.deleg_proxy_pem == NULL || ctx->gsi.deleg_proxy_len == 0) {
196 ngx_log_error(NGX_LOG_ERR, c->log, 0,
197 "xrootd tap proxy: no delegated X.509 proxy from the client "
198 "(client must delegate; is the client GSI with delegation?)");
199 return NGX_ERROR;
200 }
201
202 task = ngx_thread_task_alloc(c->pool, sizeof(proxy_gsi_login_t));
203 if (task == NULL) {
204 return NGX_ERROR;
205 }
206 g = task->ctx;
207
208 if (brix_proxy_gsi_write_pem_temp(ctx->gsi.deleg_proxy_pem,
209 ctx->gsi.deleg_proxy_len,
210 g->deleg_path, sizeof(g->deleg_path)) != 0)
211 {
212 ngx_log_error(NGX_LOG_ERR, c->log, ngx_errno,
213 "xrootd tap proxy: cannot persist delegated proxy");
214 return NGX_ERROR;
215 }
216
217 ngx_cpystrn((u_char *) g->host, host->data, sizeof(g->host));
218 g->port = port;
219 g->family = BRIX_AF_AUTO;
220 g->gsi_store = conf->gsi_store; /* the proxy server's CA store (borrowed) */
221 g->log = c->log;
222 g->proxy = proxy;
223 g->client_conn = c;
224 g->result_fd = -1;
225
226 brix_task_bind(task, proxy_gsi_login_thread, proxy_gsi_login_done);
227
228 if (ngx_thread_task_post(conf->common.thread_pool, task) != NGX_OK) {
229 (void) unlink(g->deleg_path); /* vfs-seam-allow: config-domain delegated GSI proxy credential temp (not export storage) */
230 ngx_log_error(NGX_LOG_ERR, c->log, 0,
231 "xrootd tap proxy: GSI login thread post failed");
232 return NGX_ERROR;
233 }
234
235 /* Park the client read loop until the login completes. */
236 proxy->client_ctx->state = XRD_ST_PROXY;
237 return NGX_OK;
238}
