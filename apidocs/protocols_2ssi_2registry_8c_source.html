4 * A fixed table is the right shape here: the entry count is bounded by concurrent

20void
21brix_ssi_registry_add(uintptr_t conn_id, brix_ssi_session_t *s)
22{
23 int i, free_slot = -1;
24
25 if (s == NULL) {
26 return;
27 }
28 for (i = 0; i < BRIX_SSI_REGISTRY_SLOTS; i++) {
29 if (ssi_reg[i].in_use && ssi_reg[i].conn_id == conn_id) {
30 ssi_reg[i].generation = s->generation; /* refresh a recycled id */
31 ssi_reg[i].s = s;
32 return;
33 }
34 if (!ssi_reg[i].in_use && free_slot < 0) {
35 free_slot = i;
36 }
37 }
38 if (free_slot < 0) {
39 return; /* table full: async delivery for this conn fails safe (drops) */
40 }
41 ssi_reg[free_slot].conn_id = conn_id;
42 ssi_reg[free_slot].generation = s->generation;
43 ssi_reg[free_slot].s = s;
44 ssi_reg[free_slot].in_use = 1;
45}
