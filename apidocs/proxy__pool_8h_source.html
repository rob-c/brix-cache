1/*
2 * proxy_pool.h — Phase 23 dynamic WebDAV proxy backend pool (shared memory).
3 *
4 * Enabled per-location with `brix_webdav_proxy_dynamic on`. Unlike the
5 * Phase 21 config-pool backend array (immutable until reload, per-worker COW),
6 * this pool lives in a shared-memory zone so the REST admin API can add,
7 * remove, drain, and undrain backends at runtime across all workers.
8 *
9 * Selection is weighted round-robin skipping DRAINING/DEAD entries. Each entry
10 * carries an atomic in_flight counter (incremented when proxy.c starts an
11 * upstream connection, decremented in finalize) so an operator can drain a
12 * backend and poll until in_flight == 0 before removing it.
13 */
14#ifndef BRIX_WEBDAV_PROXY_POOL_H
