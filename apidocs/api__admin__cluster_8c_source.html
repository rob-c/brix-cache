8/*
9 * Copy the URI tail after "/brix/api/v1/admin/cluster/servers/" into a caller
10 * buffer.
11 *
12 * WHAT: validate the fixed admin prefix and copy the remaining path into a
13 * NUL-terminated, mutable local buffer.
14 * WHY: request URIs are neither NUL-terminated nor safe to mutate in place;
15 * later parsing splits the copy destructively.
16 * HOW: reject when the prefix is absent or the tail is empty or overlong,
17 * then memcpy the tail and NUL-terminate. Returns NGX_OK on success.
18 */
19static ngx_int_t
20admin_copy_server_tail(ngx_http_request_t *r, char *tail, size_t tail_size)
21{
22 static const char pfx[] = "/brix/api/v1/admin/cluster/servers/";
23 size_t pfxlen = sizeof(pfx) - 1;
24 size_t taillen;
25
26 if (r->uri.len <= pfxlen
27 || ngx_strncmp(r->uri.data, pfx, pfxlen) != 0)
28 {
29 return NGX_ERROR;
30 }
31
32 taillen = r->uri.len - pfxlen;
33 if (taillen == 0 || taillen >= tail_size) {
34 return NGX_ERROR;
35 }
36 ngx_memcpy(tail, r->uri.data + pfxlen, taillen);
37 tail[taillen] = '\0';
38 return NGX_OK;
39}

42/*
43 * Split the mutable tail copy into up to three '/'-delimited segments.
44 *
45 * WHAT: partition `tail` into seg[0]=host, seg[1]=port, seg[2]=optional action.
46 * WHY: the admin server URI encodes host/port/action positionally; each field
47 * is a bare path segment.
48 * HOW: in-place split — replace each '/' with NUL and start the next segment
49 * at the following char, stopping after the third segment. The third
50 * segment captured the remainder, so trim it at any further '/' (e.g. a
51 * trailing slash). Returns the segment count.
52 */
53static ngx_uint_t
54admin_split_server_segments(char *tail, char *seg[3])
55{
56 ngx_uint_t nseg = 0;
57 char *p;
58
59 p = tail;
60 seg[nseg++] = p;
61 while (*p != '\0' && nseg < 3) {
62 if (*p == '/') {
63 *p = '\0';
64 seg[nseg++] = p + 1;
65 }
66 p++;
67 }
68 if (nseg == 3) {
69 for (p = seg[2]; *p != '\0'; p++) {
70 if (*p == '/') { *p = '\0'; break; }
71 }
72 }
73 return nseg;
74}

77/*
78 * Extract the host segment into `host_out`, stripping IPv6 brackets.
79 *
80 * WHAT: validate the host segment and copy it out in registry-canonical form.
81 * WHY: a literal IPv6 host is sent bracketed in the URI path ("[::1]") because
82 * it contains colons, but the cluster registry stores the address bare;
83 * an unstripped host would never match a registered entry.
84 * HOW: reject an empty or overlong segment, drop surrounding brackets when
85 * present, re-validate, then copy into `host_out`. Returns NGX_OK on
86 * success.
87 */
88static ngx_int_t
89admin_extract_host(const char *seg0, char *host_out, size_t host_size)
90{
91 const char *h = seg0;
92 size_t hl = ngx_strlen(h);
93 char hbuf[512];
94
95 if (h[0] == '\0' || hl >= host_size) {
96 return NGX_ERROR;
97 }
98 if (h[0] == '[' && hl >= 2 && h[hl - 1] == ']') {
99 if (hl - 1 >= sizeof(hbuf)) {
100 return NGX_ERROR;
101 }
102 ngx_memcpy(hbuf, h + 1, hl - 2);
103 hbuf[hl - 2] = '\0';
104 h = hbuf;
105 }
106 if (h[0] == '\0' || ngx_strlen(h) >= host_size) {
107 return NGX_ERROR;
108 }
109 ngx_cpystrn((u_char *) host_out, (u_char *) h, host_size);
110 return NGX_OK;
111}

114/*
115 * Parse the port segment into a validated 16-bit port.
116 *
117 * WHAT: convert the port segment to a uint16_t in range 1..65535.
118 * WHY: a well-formed server URI must carry a usable TCP port.
119 * HOW: ngx_atoi the segment and reject non-numeric or out-of-range values.
120 * Returns NGX_OK on success and writes `*port_out`.
121 */
122static ngx_int_t
123admin_parse_port(const char *seg1, uint16_t *port_out)
124{
125 ngx_int_t port;
126
127 port = ngx_atoi((u_char *) seg1, ngx_strlen(seg1));
128 if (port == NGX_ERROR || port <= 0 || port > 65535) {
129 return NGX_ERROR;
130 }
131 *port_out = (uint16_t) port;
132 return NGX_OK;
133}

136/*
137 * Parse the tail after "/brix/api/v1/admin/cluster/servers/" into host,
138 * port, and an optional trailing action ("drain"/"undrain"/""). Returns
139 * NGX_OK on success.
140 */
141ngx_int_t
142admin_parse_server_uri(ngx_http_request_t *r, char *host_out, size_t host_size,
143 uint16_t *port_out, char *action_out, size_t action_size)
144{
145 char tail[512];
146 char *seg[3] = { NULL, NULL, NULL };
147 ngx_uint_t nseg;
148
149 if (admin_copy_server_tail(r, tail, sizeof(tail)) != NGX_OK) {
150 return NGX_ERROR;
151 }
152
153 nseg = admin_split_server_segments(tail, seg);
154 if (nseg < 2) {
155 return NGX_ERROR;
156 }
157
158 if (admin_extract_host(seg[0], host_out, host_size) != NGX_OK) {
159 return NGX_ERROR;
160 }
161 if (admin_parse_port(seg[1], port_out) != NGX_OK) {
162 return NGX_ERROR;
163 }
164
165 action_out[0] = '\0';
166 if (nseg == 3 && seg[2][0] != '\0' && ngx_strlen(seg[2]) < action_size) {
167 ngx_cpystrn((u_char *) action_out, (u_char *) seg[2], action_size);
168 }
169 return NGX_OK;
170}

173/* POST/PUT body handler: upsert a server into the cluster registry. Validates
174 * required fields and host/paths character whitelists, clamps util_pct to 0..100
175 * and free_mb to >=0, then registers and audits. */
176ngx_int_t
177admin_cluster_register(ngx_http_request_t *r, json_t *body)
178{
179 const char *host = json_string_value(json_object_get(body, "host"));
180 const char *paths = json_string_value(json_object_get(body, "paths"));
181 json_int_t port = json_integer_value(json_object_get(body, "port"));
182 json_int_t free_mb = json_integer_value(json_object_get(body, "free_mb"));
183 json_int_t util_pct = json_integer_value(json_object_get(body, "util_pct"));
184
185 if (host == NULL || paths == NULL || port <= 0 || port > 65535) {
186 admin_audit(r, "cluster/register", host, "bad_request");
187 return admin_send_error(r, NGX_HTTP_BAD_REQUEST, "missing_field");
188 }
189 if (!admin_validate_hostname(host) || !admin_validate_paths(paths)) {
190 admin_audit(r, "cluster/register", host, "invalid_field");
191 return admin_send_error(r, NGX_HTTP_BAD_REQUEST, "invalid_field");
192 }
193 if (util_pct < 0) util_pct = 0;
194 if (util_pct > 100) util_pct = 100;
195 if (free_mb < 0) free_mb = 0;
196
197 brix_srv_register(host, (uint16_t) port, paths,
198 (uint32_t) free_mb, (uint32_t) util_pct);
199 brix_dashboard_event_add(BRIX_DASH_EVENT_NAMESPACE, 0, 0,
200 "admin: server registered", host);
201 admin_audit(r, "cluster/register", host, "registered");
202 return admin_send_ok(r, "registered");
203}
