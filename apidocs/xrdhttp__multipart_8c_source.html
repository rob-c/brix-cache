1/*
2 * xrdhttp_multipart.c — multipart/byteranges GET for XrdHttp vector reads.
3 *
4 * WHAT: Implements HTTP multipart/byteranges response (RFC 7233 §4.1) for GET requests
5 * that carry multiple comma-separated byte ranges. This maps to XRootD's kXR_readv
6 * (vector read) operation when accessed via the XrdHttp protocol dialect: clients such
7 * as ROOT TFile and xrdcp --prefer-xrdhttp submit a single GET with a multi-range
8 * Range header to retrieve non-contiguous data blocks in one round-trip.
9 *
10 * WHY: Without multipart range support, XrdHttp clients that need N data blocks must
11 * issue N sequential GETs, negating the latency benefit of XRootD's readv. A single
12 * multipart GET reduces round-trips to one regardless of how many blocks are needed.
13 *
14 * HOW: After the file is already open (fd) and stat'd (sb):
15 * 1. Parse all "bytes=a-b, c-d, ..." ranges from the Range header into a fixed-size
16 * array (up to XRDHTTP_MAX_RANGES entries; excess ranges are silently ignored).
17 * 2. Validate each range: clamp end to file size; drop zero-length ranges.
18 * 3. Compute total Content-Length (sum of per-part headers + data + boundaries).
19 * 4. Build an ngx_chain_t: per-range part = boundary+headers (memory buf) +
20 * file data (file buf). Append final boundary marker.
21 * 5. Send headers and chain through ngx_http_output_filter().
22 *
23 * TLS/cleartext buffer rule (from AGENTS.md §INVARIANTS):
24 * - TLS connections: use memory-backed ngx_buf_t (b->memory=1).
25 * - Cleartext: use file-backed ngx_buf_t (b->in_file=1) for sendfile.
26 * Here we use file-backed buffers for data sections (sendfile-eligible) and
27 * memory buffers for part headers and boundary markers in both cases.
28 * This is safe: nginx's sendfile filter correctly handles mixed chains.
29 *
30 * FD ownership: if fd_from_table==1 the fd is owned by the fd-cache; we set
31 * clnf->fd = NGX_INVALID_FILE for each part's cleanup so the cache retains it.
32 */

47/*
48 * WHAT: Per-request assembly context for the multipart body build.
49 * WHY: The content-length pass and the chain-build pass share the same
50 * invariant inputs (request, source fd + its path/ownership, file size, and a
51 * caller-owned formatting scratch buffer). Bundling them keeps each helper's
52 * signature small and its data flow explicit rather than threading 6+ params.
53 * HOW: Populated once in the orchestrator and passed by const pointer to the
54 * length/chain helpers; nothing here is mutated by those helpers.
55 */
56typedef struct {
57 ngx_http_request_t *r;
58 ngx_fd_t fd;
59 const char *path;
60 int fd_from_table;
61 off_t file_size;
62 char *scratch;
63 size_t scratch_size;
64} mp_ctx_t;

66/*
67 * Append a memory buffer containing <text> (NUL-terminated) to the chain.
68 * Allocates the text copy from r->pool. Returns NGX_OK or NGX_ERROR.
69 */
70static ngx_int_t
71chain_append_text(ngx_http_request_t *r,
72 ngx_chain_t **tail, const char *text,
73 ngx_chain_t **new_link_out)
74{
75 ngx_buf_t *b;
76 ngx_chain_t *link;
77 size_t len = ngx_strlen(text);
78
79 BRIX_PCALLOC_OR_RETURN(b, r->pool, sizeof(ngx_buf_t), NGX_ERROR);
80 b->pos = b->start = ngx_pnalloc(r->pool, len);
81 if (b->pos == NULL) {
82 return NGX_ERROR;
83 }
84 ngx_memcpy(b->pos, text, len);
85 b->last = b->end = b->pos + len;
86 b->memory = 1;
87
88 link = ngx_alloc_chain_link(r->pool);
89 if (link == NULL) {
90 return NGX_ERROR;
91 }
92 link->buf = b;
93 link->next = NULL;
94
95 if (*tail != NULL) {
96 (*tail)->next = link;
97 }
98 *tail = link;
99
100 if (new_link_out) {
101 *new_link_out = link;
102 }
103 return NGX_OK;
104}

107/*
108 * WHAT: Format one part's MIME header block into <buf> and return its length.
109 * WHY: The Content-Length pre-pass and the chain-build pass need byte-identical
110 * part headers; a single formatter guarantees the emitted bytes match the
111 * pre-computed length (RFC 7233 §4.1 boundary/Content-Range framing frozen).
112 * HOW: snprintf the fixed 4-line block (boundary, Content-Type, Content-Range,
113 * blank) with this range's start/end and the file size. Returns the snprintf
114 * result (>=0 on success, <0 on encoding error) so callers can fail-closed.
115 */
116static int
117mp_format_part_header(char *buf, size_t buf_size,
118 const brix_byte_range_t *range, off_t file_size)
119{
120 return snprintf(buf, buf_size,
121 "--" XRDHTTP_BOUNDARY "\r\n"
122 "Content-Type: application/octet-stream\r\n"
123 "Content-Range: bytes %lld-%lld/%lld\r\n"
124 "\r\n",
125 (long long) range->start,
126 (long long) range->end,
127 (long long) file_size);
128}

131/*
132 * WHAT: Compute the total multipart/byteranges body length into *total_out.
133 * WHY: The response Content-Length must be known before any buffer is sent;
134 * it is the exact sum of per-part header bytes + data bytes + the "\r\n"
135 * after each part + the closing boundary marker.
136 * HOW: For each range, format its header (via the shared formatter) to learn
137 * the header byte count, then add header + data + 2 (trailing CRLF). Finally
138 * add the closing "--BOUNDARY--\r\n". Returns NGX_OK, or NGX_ERROR if any
139 * header fails to format. <ctx> carries the file size and scratch buffer.
140 */
141static ngx_int_t
142mp_compute_content_length(const mp_ctx_t *ctx, const brix_byte_range_t *ranges,
143 ngx_uint_t nranges, off_t *total_out)
144{
145 off_t total = 0;
146 ngx_uint_t i;
147
148 for (i = 0; i < nranges; i++) {
149 off_t part_data_len = ranges[i].end - ranges[i].start + 1;
150 int hdr_len = mp_format_part_header(ctx->scratch, ctx->scratch_size,
151 &ranges[i], ctx->file_size);
152 if (hdr_len < 0) {
153 return NGX_ERROR;
154 }
155 total += hdr_len + part_data_len + 2 /* \r\n after data */;
156 }
157 /* Final boundary: --BOUNDARY--\r\n */
158 total += sizeof("--" XRDHTTP_BOUNDARY "--\r\n") - 1;
159
160 *total_out = total;
161 return NGX_OK;
162}

165/*
166 * WHAT: Append one multipart part (header + optional file data + trailing CRLF)
167 * to the response chain, tracking the chain head.
168 * WHY: The per-range emission is the load-bearing byte sequence; isolating it
169 * keeps the orchestrator flat and each append's error path a single return.
170 * HOW: Format the part header into <scratch>, append it as a memory buf, then
171 * for non-empty ranges append the file-backed data buf (sendfile-eligible;
172 * own_fd = !fd_from_table so the fd-cache retains a shared fd), then append
173 * the "\r\n" separator. *head is set to the first link produced if still
174 * NULL. Returns NGX_OK or NGX_ERROR.
175 */
176static ngx_int_t
177mp_append_part(const mp_ctx_t *ctx, ngx_chain_t **head, ngx_chain_t **tail,
178 const brix_byte_range_t *range)
179{
180 off_t part_data_len = range->end - range->start + 1;
181 ngx_chain_t *link = NULL;
182
183 mp_format_part_header(ctx->scratch, ctx->scratch_size, range,
184 ctx->file_size);
185
186 if (chain_append_text(ctx->r, tail, ctx->scratch, &link) != NGX_OK) {
187 return NGX_ERROR;
188 }
189 if (*head == NULL) {
190 *head = link;
191 }
192
193 if (part_data_len > 0) {
194 if (brix_http_chain_append_file_range(ctx->r, tail, ctx->fd, ctx->path,
195 range->start, range->end,
196 !ctx->fd_from_table) != NGX_OK)
197 {
198 return NGX_ERROR;
199 }
200 if (*head == NULL) {
201 *head = *tail;
202 }
203 }
204
205 if (chain_append_text(ctx->r, tail, "\r\n", &link) != NGX_OK) {
206 return NGX_ERROR;
207 }
208 if (*head == NULL) {
209 *head = link;
210 }
211 return NGX_OK;
212}

215/*
216 * WHAT: Build the complete multipart body chain for all ranges and mark its
217 * last buffer. Returns the chain head via *head_out.
218 * WHY: Separating chain assembly from header/metric emission keeps the
219 * orchestrator a linear sequence of named phases.
220 * HOW: Append each part (mp_append_part), then the closing boundary marker,
221 * then flag the tail buf as last_buf/last_in_chain. Returns NGX_OK or
222 * NGX_ERROR.
223 */
224static ngx_int_t
225mp_build_chain(const mp_ctx_t *ctx, const brix_byte_range_t *ranges,
226 ngx_uint_t nranges, ngx_chain_t **head_out)
227{
228 ngx_chain_t *head = NULL;
229 ngx_chain_t *tail = NULL;
230 ngx_chain_t *link = NULL;
231 ngx_uint_t i;
232
233 for (i = 0; i < nranges; i++) {
234 if (mp_append_part(ctx, &head, &tail, &ranges[i]) != NGX_OK) {
235 return NGX_ERROR;
236 }
237 }
238
239 /* Final boundary. */
240 if (chain_append_text(ctx->r, &tail, "--" XRDHTTP_BOUNDARY "--\r\n",
241 &link) != NGX_OK)
242 {
243 return NGX_ERROR;
244 }
245 if (head == NULL) {
246 head = link;
247 }
248
249 /* Mark last buffer. */
250 if (tail != NULL) {
251 tail->buf->last_buf = 1;
252 tail->buf->last_in_chain = 1;
253 }
254
255 *head_out = head;
256 return NGX_OK;
257}

260/*
261 * WHAT: Set the 206 status, Content-Length, and multipart Content-Type header.
262 * WHY: Header emission is a distinct concern from body assembly; the
263 * multipart/byteranges Content-Type + boundary bytes are wire-frozen.
264 * HOW: Assign status/content_length_n, disable nginx's own range emission,
265 * then push a Content-Type header carrying the boundary. Returns NGX_OK or
266 * NGX_ERROR.
267 */
268static ngx_int_t
269mp_set_response_headers(ngx_http_request_t *r, off_t total_len)
270{
271 ngx_table_elt_t *h;
272
273 r->headers_out.status = NGX_HTTP_PARTIAL_CONTENT;
274 r->headers_out.content_length_n = total_len;
275 r->allow_ranges = 0; /* we're emitting the ranges ourselves */
276
277 h = ngx_list_push(&r->headers_out.headers);
278 if (h == NULL) {
279 return NGX_ERROR;
280 }
281 h->hash = 1;
282 ngx_str_set(&h->key, "Content-Type");
283 h->value.data = ngx_pstrdup(r->pool, &(ngx_str_t){
284 sizeof("multipart/byteranges; boundary=" XRDHTTP_BOUNDARY) - 1,
285 (u_char *) "multipart/byteranges; boundary=" XRDHTTP_BOUNDARY
286 });
287 if (h->value.data == NULL) {
288 return NGX_ERROR;
289 }
290 h->value.len = sizeof("multipart/byteranges; boundary=" XRDHTTP_BOUNDARY) - 1;
291 return NGX_OK;
292}

295/*
296 * WHAT: Account the transferred data-payload bytes (data only, no framing) to
297 * the WebDAV byte-transfer metrics, split by address family.
298 * WHY: Metric labels must stay low-cardinality; only total + per-family byte
299 * counters are bumped, using the summed data length (not the framed length).
300 * HOW: Sum each range's data length, add to the total counter and to the IPv6
301 * or IPv4 counter based on the connection's socket family.
302 */
303static void
304mp_account_metrics(ngx_http_request_t *r,
305 const brix_byte_range_t *ranges, ngx_uint_t nranges)
306{
307 off_t data_bytes = 0;
308 ngx_uint_t i;
309
310 for (i = 0; i < nranges; i++) {
311 data_bytes += ranges[i].end - ranges[i].start + 1;
312 }
313
314 BRIX_WEBDAV_METRIC_ADD(bytes_tx_total, (size_t) data_bytes);
315 if (r->connection && r->connection->sockaddr
316 && r->connection->sockaddr->sa_family == AF_INET6) {
317 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv6_total, (size_t) data_bytes);
318 } else {
319 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv4_total, (size_t) data_bytes);
320 }
321}

324/*
325 * WHAT: Copy the request URI into <path> (used as the file-buf path for
326 * sendfile cleanup), NUL-terminating and bounding to the buffer.
327 * WHY: The file-range bufs record a path for their cleanup handler; it must be
328 * a bounded C string derived from the (non-NUL-terminated) ngx_str_t URI.
329 * HOW: Start empty, and if the URI fits, cpystrn it (which NUL-terminates).
330 */
331static void
332mp_resolve_path(ngx_http_request_t *r, char *path, size_t path_size)
333{
334 path[0] = '\0';
335 if (r->uri.len > 0 && r->uri.len < path_size - 1) {
336 ngx_cpystrn((u_char *) path, r->uri.data,
337 ngx_min(r->uri.len + 1, path_size));
338 }

342ngx_int_t
343xrdhttp_handle_multipart_get(ngx_http_request_t *r,
344 ngx_fd_t fd,
345 const struct stat *sb,
346 int fd_from_table)
347{
348 brix_byte_range_t ranges[XRDHTTP_MAX_RANGES];
349 brix_range_vector_opts_t opts;
350 ngx_uint_t nranges = 0;
351 off_t total_len = 0;
352 ngx_chain_t *head = NULL;
353 char path[WEBDAV_MAX_PATH];
354 char part_hdr[256];
355 ngx_int_t rc;
356
357 /* Get the resolved path for file bufs (used in cleanup). */
358 mp_resolve_path(r, path, sizeof(path));
359
360 if (r->headers_in.range == NULL
361 || r->headers_in.range->value.len < 7
362 || ngx_strncmp(r->headers_in.range->value.data, "bytes=", 6) != 0)
363 {
364 return NGX_HTTP_BAD_REQUEST;
365 }
366
367 ngx_memzero(&opts, sizeof(opts));
368 opts.max_ranges = XRDHTTP_MAX_RANGES;
369 opts.allow_suffix = 1;
370 opts.allow_open_ended = 1;
371 opts.drop_unsatisfiable = 1;
372
373 if (brix_http_parse_range_vector(r->headers_in.range->value.data + 6,
374 r->headers_in.range->value.len - 6,
375 sb->st_size, &opts,
376 ranges, &nranges) != NGX_OK
377 || nranges == 0)
378 {
379 /* All ranges unsatisfiable. */
380 r->headers_out.status = NGX_HTTP_RANGE_NOT_SATISFIABLE;
381 r->headers_out.content_length_n = 0;
382 ngx_http_send_header(r);
383 return ngx_http_send_special(r, NGX_HTTP_LAST);
384 }
385
386 mp_ctx_t ctx = {
387 .r = r,
388 .fd = fd,
389 .path = path,
390 .fd_from_table = fd_from_table,
391 .file_size = sb->st_size,
392 .scratch = part_hdr,
393 .scratch_size = sizeof(part_hdr),
394 };
395
396 if (mp_compute_content_length(&ctx, ranges, nranges, &total_len) != NGX_OK) {
397 return NGX_HTTP_INTERNAL_SERVER_ERROR;
398 }
399
400 if (mp_build_chain(&ctx, ranges, nranges, &head) != NGX_OK) {
401 return NGX_HTTP_INTERNAL_SERVER_ERROR;
402 }
403
404 if (mp_set_response_headers(r, total_len) != NGX_OK) {
405 return NGX_HTTP_INTERNAL_SERVER_ERROR;
406 }
407
408 mp_account_metrics(r, ranges, nranges);
409
410 /* Inject XrdHttp response headers before send. */
411 (void) xrdhttp_add_response_headers(r, NGX_HTTP_PARTIAL_CONTENT);
412
413 rc = ngx_http_send_header(r);
414 if (rc == NGX_ERROR || r->header_only) {
415 return rc;
416 }
417
418 return ngx_http_output_filter(r, head);
419}
