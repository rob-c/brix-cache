1/*
2 * xrdhttp_multipart.c — multipart/byteranges GET for XrdHttp vector reads.
3 *
4 * WHAT: Implements HTTP multipart/byteranges response (RFC 7233 §4.1) for GET requests
5 * that carry multiple comma-separated byte ranges. This maps to XRootD's kXR_readv
6 * (vector read) operation when accessed via the XrdHttp protocol dialect: clients such
7 * as ROOT TFile and xrdcp --prefer-xrdhttp submit a single GET with a multi-range
8 * Range header to retrieve non-contiguous data blocks in one round-trip.
9 *
10 * WHY: Without multipart range support, XrdHttp clients that need N data blocks must
11 * issue N sequential GETs, negating the latency benefit of XRootD's readv. A single
12 * multipart GET reduces round-trips to one regardless of how many blocks are needed.
13 *
14 * HOW: After the file is already open (fd) and stat'd (sb):
15 * 1. Parse all "bytes=a-b, c-d, ..." ranges from the Range header into a fixed-size
16 * array (up to XRDHTTP_MAX_RANGES entries; excess ranges are silently ignored).
17 * 2. Validate each range: clamp end to file size; drop zero-length ranges.
18 * 3. Compute total Content-Length (sum of per-part headers + data + boundaries).
19 * 4. Build an ngx_chain_t: per-range part = boundary+headers (memory buf) +
20 * file data (file buf). Append final boundary marker.
21 * 5. Send headers and chain through ngx_http_output_filter().
22 *
23 * TLS/cleartext buffer rule (from AGENTS.md §INVARIANTS):
24 * - TLS connections: use memory-backed ngx_buf_t (b->memory=1).
25 * - Cleartext: use file-backed ngx_buf_t (b->in_file=1) for sendfile.
26 * Here we use file-backed buffers for data sections (sendfile-eligible) and
27 * memory buffers for part headers and boundary markers in both cases.
28 * This is safe: nginx's sendfile filter correctly handles mixed chains.
29 *
30 * FD ownership: if fd_from_table==1 the fd is owned by the fd-cache; we set
31 * clnf->fd = NGX_INVALID_FILE for each part's cleanup so the cache retains it.
32 */

46/*
47 * Append a memory buffer containing <text> (NUL-terminated) to the chain.
48 * Allocates the text copy from r->pool. Returns NGX_OK or NGX_ERROR.
49 */
50static ngx_int_t
51chain_append_text(ngx_http_request_t *r,
52 ngx_chain_t **tail, const char *text,
53 ngx_chain_t **new_link_out)
54{
55 ngx_buf_t *b;
56 ngx_chain_t *link;
57 size_t len = ngx_strlen(text);
58
59 BRIX_PCALLOC_OR_RETURN(b, r->pool, sizeof(ngx_buf_t), NGX_ERROR);
60 b->pos = b->start = ngx_pnalloc(r->pool, len);
61 if (b->pos == NULL) {
62 return NGX_ERROR;
63 }
64 ngx_memcpy(b->pos, text, len);
65 b->last = b->end = b->pos + len;
66 b->memory = 1;
67
68 link = ngx_alloc_chain_link(r->pool);
69 if (link == NULL) {
70 return NGX_ERROR;
71 }
72 link->buf = b;
73 link->next = NULL;
74
75 if (*tail != NULL) {
76 (*tail)->next = link;
77 }
78 *tail = link;
79
80 if (new_link_out) {
81 *new_link_out = link;
82 }
83 return NGX_OK;
84}

87ngx_int_t
88xrdhttp_handle_multipart_get(ngx_http_request_t *r,
89 ngx_fd_t fd,
90 const struct stat *sb,
91 int fd_from_table)
92{
93 brix_byte_range_t ranges[XRDHTTP_MAX_RANGES];
94 brix_range_vector_opts_t opts;
95 ngx_uint_t nranges;
96 off_t total_len = 0;
97 ngx_chain_t *head = NULL;
98 ngx_chain_t *tail = NULL;
99 ngx_chain_t *link;
100 ngx_table_elt_t *h;
101 char path[WEBDAV_MAX_PATH];
102 char part_hdr[256];
103 char len_str[32];
104 ngx_uint_t i;
105
106 /* Get the resolved path for file bufs (used in cleanup). */
107 path[0] = '\0';
108 if (r->uri.len > 0 && r->uri.len < sizeof(path) - 1) {
109 ngx_cpystrn((u_char *) path, r->uri.data,
110 ngx_min(r->uri.len + 1, sizeof(path)));
111 }
112
113 if (r->headers_in.range == NULL
114 || r->headers_in.range->value.len < 7
115 || ngx_strncmp(r->headers_in.range->value.data, "bytes=", 6) != 0)
116 {
117 return NGX_HTTP_BAD_REQUEST;
118 }
119
120 ngx_memzero(&opts, sizeof(opts));
121 opts.max_ranges = XRDHTTP_MAX_RANGES;
122 opts.allow_suffix = 1;
123 opts.allow_open_ended = 1;
124 opts.drop_unsatisfiable = 1;
125
126 if (brix_http_parse_range_vector(r->headers_in.range->value.data + 6,
127 r->headers_in.range->value.len - 6,
128 sb->st_size, &opts,
129 ranges, &nranges) != NGX_OK
130 || nranges == 0)
131 {
132 /* All ranges unsatisfiable. */
133 r->headers_out.status = NGX_HTTP_RANGE_NOT_SATISFIABLE;
134 r->headers_out.content_length_n = 0;
135 ngx_http_send_header(r);
136 return ngx_http_send_special(r, NGX_HTTP_LAST);
137 }
138
139 /* Compute Content-Length */ for (i = 0; i < nranges; i++) {
140 off_t part_data_len = ranges[i].end - ranges[i].start + 1;
141
142 /* Per-part header:
143 * --BOUNDARY\r\n
144 * Content-Type: application/octet-stream\r\n
145 * Content-Range: bytes START-END/SIZE\r\n
146 * \r\n
147 */
148 int hdr_len = snprintf(part_hdr, sizeof(part_hdr),
149 "--" XRDHTTP_BOUNDARY "\r\n"
150 "Content-Type: application/octet-stream\r\n"
151 "Content-Range: bytes %lld-%lld/%lld\r\n"
152 "\r\n",
153 (long long) ranges[i].start,
154 (long long) ranges[i].end,
155 (long long) sb->st_size);
156 if (hdr_len < 0) {
157 return NGX_HTTP_INTERNAL_SERVER_ERROR;
158 }
159
160 total_len += hdr_len + part_data_len + 2 /* \r\n after data */;
161 }
162 /* Final boundary: --BOUNDARY--\r\n */
163 total_len += sizeof("--" XRDHTTP_BOUNDARY "--\r\n") - 1;
164
165 /* Build response chain */ for (i = 0; i < nranges; i++) {
166 off_t part_data_len = ranges[i].end - ranges[i].start + 1;
167
168 /* Part header. */
169 snprintf(part_hdr, sizeof(part_hdr),
170 "--" XRDHTTP_BOUNDARY "\r\n"
171 "Content-Type: application/octet-stream\r\n"
172 "Content-Range: bytes %lld-%lld/%lld\r\n"
173 "\r\n",
174 (long long) ranges[i].start,
175 (long long) ranges[i].end,
176 (long long) sb->st_size);
177
178 if (chain_append_text(r, &tail, part_hdr, &link) != NGX_OK) {
179 return NGX_HTTP_INTERNAL_SERVER_ERROR;
180 }
181 if (head == NULL) {
182 head = link;
183 }
184
185 /* File data for this range. */
186 if (part_data_len > 0) {
187 if (brix_http_chain_append_file_range(r, &tail, fd, path,
188 ranges[i].start,
189 ranges[i].end,
190 !fd_from_table) != NGX_OK)
191 {
192 return NGX_HTTP_INTERNAL_SERVER_ERROR;
193 }
194 if (head == NULL) {
195 head = tail;
196 }
197 }
198
199 /* CRLF between data and next boundary. */
200 if (chain_append_text(r, &tail, "\r\n", &link) != NGX_OK) {
201 return NGX_HTTP_INTERNAL_SERVER_ERROR;
202 }
203 if (head == NULL) {
204 head = link;
205 }
206 }
207
208 /* Final boundary. */
209 if (chain_append_text(r, &tail,
210 "--" XRDHTTP_BOUNDARY "--\r\n",
211 &link) != NGX_OK)
212 {
213 return NGX_HTTP_INTERNAL_SERVER_ERROR;
214 }
215 if (head == NULL) {
216 head = link;
217 }
218
219 /* Mark last buffer. */
220 if (tail != NULL) {
221 tail->buf->last_buf = 1;
222 tail->buf->last_in_chain = 1;
223 }
224
225 /* Set response headers */ r->headers_out.status = NGX_HTTP_PARTIAL_CONTENT;
226 r->headers_out.content_length_n = total_len;
227 r->allow_ranges = 0; /* we're emitting the ranges ourselves */
228
229 h = ngx_list_push(&r->headers_out.headers);
230 if (h == NULL) {
231 return NGX_HTTP_INTERNAL_SERVER_ERROR;
232 }
233 h->hash = 1;
234 ngx_str_set(&h->key, "Content-Type");
235 h->value.data = ngx_pstrdup(r->pool, &(ngx_str_t){
236 sizeof("multipart/byteranges; boundary=" XRDHTTP_BOUNDARY) - 1,
237 (u_char *) "multipart/byteranges; boundary=" XRDHTTP_BOUNDARY
238 });
239 if (h->value.data == NULL) {
240 return NGX_HTTP_INTERNAL_SERVER_ERROR;
241 }
242 h->value.len = sizeof("multipart/byteranges; boundary=" XRDHTTP_BOUNDARY) - 1;
243
244 /* Add bytes-transferred metric for the total data payload only. */
245 off_t data_bytes = 0;
246 for (i = 0; i < nranges; i++) {
247 data_bytes += ranges[i].end - ranges[i].start + 1;
248 }
249 snprintf(len_str, sizeof(len_str), "%lld", (long long) data_bytes);
250 BRIX_WEBDAV_METRIC_ADD(bytes_tx_total, (size_t) data_bytes);
251 if (r->connection && r->connection->sockaddr
252 && r->connection->sockaddr->sa_family == AF_INET6) {
253 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv6_total, (size_t) data_bytes);
254 } else {
255 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv4_total, (size_t) data_bytes);
256 }
257
258 /* Inject XrdHttp response headers before send. */
259 (void) xrdhttp_add_response_headers(r, NGX_HTTP_PARTIAL_CONTENT);
260
261 ngx_int_t rc = ngx_http_send_header(r);
262 if (rc == NGX_ERROR || r->header_only) {
263 return rc;
264 }
265
266 return ngx_http_output_filter(r, head);
267}
