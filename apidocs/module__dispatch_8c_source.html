18/* Exact URI match (length and bytes). */
19static ngx_int_t
20dashboard_uri_eq(ngx_str_t uri, const char *literal)
21{
22 size_t len = ngx_strlen(literal);
23
24 return uri.len == len && ngx_memcmp(uri.data, literal, len) == 0;
25}
26
27/* Strict prefix match: uri must be STRICTLY longer than `literal` (uri.len >
28 * len), so a bare collection path does not match its own "<path>/" prefix form —
29 * letting _eq and _prefix routes coexist (e.g. ".../transfers" vs
30 * ".../transfers/<id>"). */
31static ngx_int_t
32dashboard_uri_prefix(ngx_str_t uri, const char *literal)
33{
34 size_t len = ngx_strlen(literal);
35
36 return uri.len > len && ngx_memcmp(uri.data, literal, len) == 0;
37}

39/*
40 * Route-table dispatch (main_handler decomposition).
41 *
42 * Two ordered `static const` tables replace the former if-ladder while keeping
43 * byte-identical routing. Every match test is either `dashboard_uri_eq` (exact)
44 * or `dashboard_uri_prefix` (strictly-longer prefix); both are mutually
45 * exclusive across distinct literals, so within a table order does not change
46 * which literal a URI matches. The ORDER BETWEEN groups is what stays frozen:
47 * concrete API/handler routes first, then the admin prefix, then the generic
48 * "/api/v1/" catch-all, then the page/login tail — most-specific first.
49 */

51/* How a route row matches its literal against the request URI. */
52typedef enum {

54 DASH_MATCH_PREFIX /* strict prefix (dashboard_uri_prefix) */

107/*
108 * WHAT: True when `uri` matches `route`'s literal under its match kind.
109 * WHY: Single predicate keeps the two dispatch loops branch-free of the
110 * eq-vs-prefix distinction, preserving the exact original semantics.
111 * HOW: Delegate to the same _eq/_prefix helpers the if-ladder used.
112 */
113static ngx_int_t
114dashboard_api_route_matches(ngx_str_t uri, const dashboard_api_route_t *route)
115{
116 if (route->match == DASH_MATCH_PREFIX) {
117 return dashboard_uri_prefix(uri, route->literal);
118 }
119 return dashboard_uri_eq(uri, route->literal);
120}

122/*
123 * WHAT: Try the concrete versioned-API routes; on the first match, dispatch to
124 * the JSON API handler for that endpoint and store the result in *out.
125 * WHY: Collapses ten identical eq/prefix -> api_handler branches into one
126 * ordered scan without changing which URI hits which endpoint id.
127 * HOW: Linear scan of dashboard_api_routes (exact rows are mutually exclusive;
128 * order is irrelevant within them). Returns NGX_OK on a match, NGX_DECLINED
129 * when no API route applies so the caller falls through to later groups.
130 */
131static ngx_int_t
132dashboard_dispatch_api_route(ngx_http_request_t *r, ngx_str_t uri,
133 ngx_int_t *out)
134{
135 size_t i;
136
137 for (i = 0; i < sizeof(dashboard_api_routes)
138 / sizeof(dashboard_api_routes[0]); i++) {
139 if (dashboard_api_route_matches(uri, &dashboard_api_routes[i])) {
140 *out = ngx_http_brix_dashboard_api_handler(r,
141 dashboard_api_routes[i].endpoint);
142 return NGX_OK;
143 }
144 }
145 return NGX_DECLINED;
146}

148/*
149 * WHAT: Try the dedicated-handler routes; on the first exact match, invoke that
150 * handler and store its result in *out.
151 * WHY: Collapses the config/files/download/vfs* / scan exact-match branches into
152 * one scan; all rows are exact and mutually exclusive.
153 * HOW: Linear scan of dashboard_handler_routes. Returns NGX_OK on a match,
154 * NGX_DECLINED otherwise (caller continues to the prefix/catch-all/tail).
155 */
156static ngx_int_t
157dashboard_dispatch_handler_route(ngx_http_request_t *r, ngx_str_t uri,
158 ngx_int_t *out)
159{
160 size_t i;
161
162 for (i = 0; i < sizeof(dashboard_handler_routes)
163 / sizeof(dashboard_handler_routes[0]); i++) {
164 if (dashboard_uri_eq(uri, dashboard_handler_routes[i].literal)) {
165 *out = dashboard_handler_routes[i].handler(r);
166 return NGX_OK;
167 }
168 }
169 return NGX_DECLINED;
170}

172/* Main content handler dispatcher */
173/*
174 * WHAT: Content handler installed at the dashboard location; routes the request
175 * URI to the page, login, compat-JSON, versioned-API, or admin handler.
176 * HOW: Groups are tried MOST-SPECIFIC FIRST so exact matches win over prefix
177 * matches (e.g. ".../transfers" before ".../transfers/<id>", and the
178 * known v1 endpoints before the catch-all "/api/v1/" 404). Per-endpoint
179 * auth lives inside the called handlers, not here. Unmatched -> 404.
180 */
181ngx_int_t
182ngx_http_brix_dashboard_main_handler(ngx_http_request_t *r)
183{
184 ngx_http_brix_dashboard_loc_conf_t *conf;
185 ngx_str_t uri;
186 ngx_int_t rc = NGX_DECLINED;
187
188 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
189 if (!conf->enable) {
190 return NGX_HTTP_NOT_FOUND;
191 }
192
193 /* AGPL-3.0 sec.13: offer remote users the source (X-Source header). */
194 brix_http_source_offer(r);
195
196 uri = r->uri;
197
198 if (dashboard_dispatch_api_route(r, uri, &rc) == NGX_OK) {
199 return rc;
200 }
201 if (dashboard_dispatch_handler_route(r, uri, &rc) == NGX_OK) {
202 return rc;
203 }
204
205 /* Phase 23: admin write API (auth + method routing inside dispatch). */
206 if (uri.len >= sizeof("/brix/api/v1/admin/") - 1
207 && ngx_memcmp(uri.data, "/brix/api/v1/admin/",
208 sizeof("/brix/api/v1/admin/") - 1) == 0)
209 {
210 return brix_admin_dispatch(r);
211 }
212
213 /* Catch-all for unknown /api/v1/ paths: return the API's structured 404
214 * (must come AFTER every concrete v1 route above). */
215 if (uri.len > sizeof("/brix/api/v1/") - 1
216 && ngx_memcmp(uri.data, "/brix/api/v1/",
217 sizeof("/brix/api/v1/") - 1) == 0)
218 {
219 return ngx_http_brix_dashboard_api_handler(r,
220 BRIX_DASHBOARD_API_V1_NOT_FOUND);
221 }
222
223 if (dashboard_uri_eq(uri, "/brix/login"))
224 {
225 return ngx_http_brix_dashboard_login_handler(r);
226 }
227
228 if (dashboard_uri_eq(uri, "/brix") || dashboard_uri_eq(uri, "/brix/")) {
229 return ngx_http_brix_dashboard_page_handler(r);
230 }
231
232 return NGX_HTTP_NOT_FOUND;
233}
