4/*
5 * WHAT: Start an upstream XRootD connection — context allocation, DNS/TCP setup, bootstrap buffer build,
6 * and connect initiation for transparent proxy mode.
7 * WHY: When a client connects in proxy mode, nginx must lazily open a backend XRootD server connection
8 * on the first post-login opcode. This file owns that startup sequence: allocating upstream context,
9 * resolving the upstream address (pre-configured or per-request DNS), creating a non-blocking socket,
10 * arming event handlers, and building the initial handshake/protocol/login bootstrap bytes.
11 * HOW: Two-path address resolution — fast path uses pre-resolved sockaddr from config (no DNS on event loop);
12 * fallback path calls getaddrinfo() per-request with warning logged at startup. Non-blocking socket +
13 * ngx_get_connection + pool setup + read/write handler assignment + timer for kXR_wait retry. Bootstrap
14 * buffer contains handshake zeros, protocol request, and login request concatenated in wire order.
15 */

20/*
21 * WHAT: Allocate the upstream context, link it to the client, and snapshot the
22 * in-flight opcode/streamid/path plus opcode-specific option fields.
23 * Returns NGX_OK once `up->req_*` is fully populated; NGX_ERROR (after cleanup
24 * where an `up` exists) on alloc failure, missing payload, or path extraction.
25 * WHY: The socket is opened on the first post-login opcode, so the request that
26 * triggered it must be captured verbatim — it is replayed to the backend after
27 * bootstrap so the server's reply matches the client's request byte-for-byte.
28 * HOW: pcalloc the ctx from the client pool, link it both ways, copy the streamid;
29 * extract the confined path from the payload; then, for locate/open only,
30 * unpack the wire header body (network byte order) to preserve options (and
31 * open's mode). *out_up is set once allocation succeeds so the caller can
32 * report failure without re-cleaning.
33 */
34static ngx_int_t
35brix_upstream_start_snapshot_request(brix_ctx_t *ctx, ngx_connection_t *c,
36 brix_upstream_t **out_up)
37{
38 brix_upstream_t *up;
39
40 up = ngx_pcalloc(c->pool, sizeof(brix_upstream_t));
41 if (up == NULL) {
42 return NGX_ERROR;
43 }
44
45 up->client_ctx = ctx;
46 up->client_conn = c;
47 ctx->upstream = up;
48 *out_up = up;
49
50 /* Snapshot the opcode/streamid that triggered the open — these are replayed to the
51 * backend verbatim after bootstrap so the server's reply matches the client's. */
52 up->req_opcode = ctx->recv.cur_reqid;
53 up->req_streamid[0] = ctx->recv.cur_streamid[0];
54 up->req_streamid[1] = ctx->recv.cur_streamid[1];
55
56 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
57 return NGX_ERROR;
58 }
59
60 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
61 up->req_path, sizeof(up->req_path), 1))
62 {
63 return NGX_ERROR;
64 }
65
66 /* Preserve opcode-specific request fields straight off the wire header (network
67 * byte order) so the replayed request to the backend is byte-identical to the
68 * client's. Only locate/open carry options we must reproduce; open also carries mode. */
69 if (ctx->recv.cur_reqid == kXR_locate) {
70 xrdw_locate_req_t lr;
71
72 xrdw_locate_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &lr);
73 up->req_options = lr.options;
74 } else if (ctx->recv.cur_reqid == kXR_open) {
75 xrdw_open_req_t oreq;
76
77 xrdw_open_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &oreq);
78 up->req_options = oreq.options;
79 up->req_open_mode = oreq.mode;
80 }
81
82 return NGX_OK;
83}

85/*
86 * WHAT: Produce a connectable, non-blocking upstream socket and the sockaddr to
87 * connect it to. Returns the fd (>=0) and fills *addr and *addrlen on success;
88 * returns NGX_INVALID_FILE and sets *dns_failed on a hard DNS error.
89 * WHY: Two address-resolution paths must be tried in a fixed order without touching
90 * the event loop when avoidable — config pre-resolution first, per-request DNS
91 * only as a fallback — and the caller distinguishes "cannot resolve" from
92 * "no usable address" for logging.
93 * HOW: Fast path uses conf->upstream_addr (resolved at config time): open a socket
94 * of the address family, set non-blocking, and copy the sockaddr. Fallback path
95 * calls brix_resolve_connect_socket (blocking getaddrinfo, first family that
96 * yields a non-blocking socket wins) and reports BRIX_RESOLVE_ERR_DNS via
97 * *dns_failed. *addrlen stays 0 only on the failure paths.
98 */
99static int
100brix_upstream_start_resolve_socket(ngx_stream_brix_srv_conf_t *conf,
101 struct sockaddr_storage *addr,
102 socklen_t *addrlen, ngx_int_t *dns_failed)
103{
104 int fd;
105
106 *dns_failed = 0;
107
108 if (conf->upstream_addr != NULL) {
109 /* fast path: address pre-resolved at config time — no DNS on event loop */
110 struct sockaddr *sa = conf->upstream_addr->sockaddr;
111
112 fd = ngx_socket(sa->sa_family, SOCK_STREAM, 0);
113 if (fd != (int) NGX_INVALID_FILE) {
114 if (ngx_nonblocking(fd) == NGX_ERROR) {
115 ngx_close_socket(fd);
116 fd = (int) NGX_INVALID_FILE;
117 } else {
118 ngx_memcpy(addr, sa, conf->upstream_addr->socklen);
119 *addrlen = conf->upstream_addr->socklen;
120 }
121 }
122 return fd;
123 }
124
125 /* fallback: resolve per-request (blocks event loop; logged as warning at startup).
126 * The first family yielding a non-blocking socket wins; a no-usable-socket
127 * result falls through to the shared `fd == INVALID` check in the caller. */
128 brix_resolve_status_t rstatus;
129
130 fd = brix_resolve_connect_socket((char *) conf->upstream_host.data,
131 (unsigned) conf->upstream_port,
132 BRIX_AF_AUTO,
133 addr, addrlen, &rstatus);
134 if (fd == (int) NGX_INVALID_FILE && rstatus == BRIX_RESOLVE_ERR_DNS) {
135 *dns_failed = 1;
136 }
137 return fd;
138}

140/*
141 * WHAT: Wrap a connected-to-be socket fd in an nginx connection with its own pool
142 * and the upstream read/write handlers armed. Returns the connection, or NULL
143 * (after closing the fd and any half-built connection) on failure.
144 * WHY: The raw fd must participate in the event loop; from the moment ngx_get_connection
145 * succeeds, teardown must also ngx_free_connection — this helper keeps that
146 * ordering in one place so its failure paths never leak.
147 * HOW: ngx_get_connection, then create a small per-connection pool (response bodies
148 * and TLS scratch live here so they are reclaimed with the connection), then wire
149 * the recv/send vtable, log, and read/write handlers. On any failure the fd (and
150 * connection, if obtained) are released before returning NULL.
151 */
152static ngx_connection_t *
153brix_upstream_start_wrap_conn(int fd, brix_upstream_t *up, ngx_log_t *log)
154{
155 ngx_connection_t *uconn;
156
157 /* Wrap the raw fd in an nginx connection so it participates in the event loop.
158 * From here failures must ngx_free_connection() in addition to closing the socket. */
159 uconn = ngx_get_connection(fd, log);
160 if (uconn == NULL) {
161 ngx_close_socket(fd);
162 return NULL;
163 }
164
165 /* Per-connection pool: response bodies and TLS scratch are allocated here so they
166 * are reclaimed when the upstream connection is destroyed. */
167 uconn->pool = ngx_create_pool(512, log);
168 if (uconn->pool == NULL) {
169 ngx_free_connection(uconn);
170 ngx_close_socket(fd);
171 return NULL;
172 }
173
174 uconn->data = up;
175 uconn->recv = ngx_recv;
176 uconn->send = ngx_send;
177 uconn->recv_chain = ngx_recv_chain;
178 uconn->send_chain = ngx_send_chain;
179 uconn->log = log;
180 uconn->read->handler = brix_upstream_read_handler;
181 uconn->write->handler = brix_upstream_write_handler;
182 uconn->read->log = log;
183 uconn->write->log = log;
184
185 return uconn;
186}

188/*
189 * WHAT: Build the three-message bootstrap into the upstream connection's pool and
190 * install it as the write buffer. Returns NGX_OK, or NGX_ERROR on alloc failure.
191 * WHY: The opening handshake + protocol + login bytes are sent before any client
192 * opcode is replayed; staging them as a single wbuf lets the flush loop drain
193 * them across partial writes with wbuf_pos as the cursor.
194 * HOW: Size for the fixed handshake plus one ClientProtocolRequest and one
195 * ClientLoginRequest, palloc from uconn->pool, serialize via
196 * brix_upstream_build_bootstrap, then point up->wbuf at it with pos = 0.
197 */
198static ngx_int_t
199brix_upstream_start_build_wbuf(brix_upstream_t *up, ngx_connection_t *uconn)
200{
201 size_t bslen;
202 u_char *bsbuf;
203
204 /* Build the bootstrap write buffer: the three opening wire messages concatenated in
205 * send order — handshake (fixed-size zero/init bytes) + protocol request + login
206 * request. This is sent before any client opcode is replayed. wbuf_pos tracks how
207 * much has drained across partial writes. */
208 bslen = XRD_HANDSHAKE_LEN
209 + sizeof(ClientProtocolRequest)
210 + sizeof(ClientLoginRequest);
211 bsbuf = ngx_palloc(uconn->pool, bslen);
212 if (bsbuf == NULL) {
213 return NGX_ERROR;
214 }
215 brix_upstream_build_bootstrap(bsbuf);
216 up->wbuf = bsbuf;
217 up->wbuf_len = bslen;
218 up->wbuf_pos = 0;
219
220 return NGX_OK;
221}

223/*
224 * WHAT: Kick off the non-blocking connect, arm the write event, and — on an immediate
225 * local connect — start the bootstrap by flushing it. Returns NGX_OK once the
226 * connect is armed/initiated; NGX_ERROR on connect, event-arm, or flush failure.
227 * WHY: This is the final commit step; every failure here must roll the client session
228 * state back to REQ_HEADER so the client can retry, which the caller does on the
229 * NGX_ERROR return.
230 * HOW: connect() — EINPROGRESS is the normal async case (write handler finishes via
231 * SO_ERROR), rc==0 is an immediate connect, anything else is a hard fail.
232 * ngx_handle_write_event arms the write side in both good cases. For rc==0 only,
233 * advance to BOOTSTRAP/HANDSHAKE, reset the response accumulator, and flush.
234 * The socket fd is taken from uconn->fd (set by ngx_get_connection).
235 */
236static ngx_int_t
237brix_upstream_start_connect(brix_upstream_t *up, ngx_connection_t *uconn,
238 ngx_stream_brix_srv_conf_t *conf,
239 const struct sockaddr_storage *addr,
240 socklen_t addrlen)
241{
242 ngx_int_t rc;
243
244 /* Non-blocking connect: EINPROGRESS is the normal async case (write handler finishes
245 * it via SO_ERROR); rc==0 is an immediate local connect; anything else is a hard fail. */
246 rc = connect(uconn->fd, (const struct sockaddr *)(const void *) addr, addrlen);
247 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
248 ngx_log_error(NGX_LOG_ERR, uconn->log, ngx_socket_errno,
249 "brix: upstream connect to %s:%d failed",
250 (char *) conf->upstream_host.data,
251 (int) conf->upstream_port);
252 return NGX_ERROR;
253 }
254
255 /* Arm the write event in both cases: EINPROGRESS needs it to learn of connect
256 * completion; rc==0 needs it to drain any bootstrap bytes that block on this write. */
257 if (ngx_handle_write_event(uconn->write, 0) != NGX_OK) {
258 return NGX_ERROR;
259 }
260
261 /* Immediate connect: skip the CONNECTING wait and start bootstrap now — advance to
262 * BOOTSTRAP/HANDSHAKE, reset the response-header accumulator, and flush bootstrap. */
263 if (rc == 0) {
264 up->state = XRD_UP_BOOTSTRAP;
265 up->bs_phase = XRD_UP_BS_HANDSHAKE;
266 up->rhdr_pos = 0;
267
268 if (brix_upstream_flush(up) == NGX_ERROR) {
269 return NGX_ERROR;
270 }
271 }
272
273 return NGX_OK;
274}

276/*
277 * WHAT: Lazily open the backend XRootD connection for the in-flight client opcode and
278 * kick off the async connect. Returns NGX_OK once the connect is armed/initiated;
279 * NGX_ERROR (after cleanup) on any setup failure.
280 * WHY: In proxy mode the upstream socket is created on demand at the first post-login
281 * opcode, so this captures the opcode/streamid/path/options that triggered it —
282 * they must be replayed to the backend once bootstrap completes.
283 * HOW: alloc+link the upstream ctx, snapshot the request, resolve the address (config
284 * fast path or per-request DNS), create a non-blocking socket + nginx connection,
285 * build the handshake/protocol/login bootstrap into the write buffer, then connect().
286 * On EINPROGRESS the write handler finishes the connect; on immediate success (rc==0)
287 * we flush the bootstrap here. Any failure path rolls ctx->state back to REQ_HEADER.
288 */
289ngx_int_t
290brix_upstream_start(brix_ctx_t *ctx, ngx_connection_t *c,
291 ngx_stream_brix_srv_conf_t *conf)
292{
293 brix_upstream_t *up = NULL;
294 ngx_connection_t *uconn;
295 int fd;
296 struct sockaddr_storage chosen_addr;
297 socklen_t chosen_addrlen = 0; /* set whenever fd is valid;
298 0 only on the (rejected)
299 fd==INVALID paths */
300 ngx_int_t dns_failed;
301
302 if (brix_upstream_start_snapshot_request(ctx, c, &up) != NGX_OK) {
303 if (up != NULL) {
304 brix_upstream_cleanup(up);
305 }
306 return NGX_ERROR;
307 }
308
309 fd = brix_upstream_start_resolve_socket(conf, &chosen_addr,
310 &chosen_addrlen, &dns_failed);
311 if (dns_failed) {
312 ngx_log_error(NGX_LOG_ERR, c->log, 0,
313 "brix: upstream: cannot resolve \"%s\"",
314 (char *) conf->upstream_host.data);
315 brix_upstream_cleanup(up);
316 return NGX_ERROR;
317 }
318
319 if (fd == (int) NGX_INVALID_FILE) {
320 ngx_log_error(NGX_LOG_ERR, c->log, 0,
321 "brix: upstream: no usable address for \"%s\"",
322 (char *) conf->upstream_host.data);
323 brix_upstream_cleanup(up);
324 return NGX_ERROR;
325 }
326
327 uconn = brix_upstream_start_wrap_conn(fd, up, c->log);
328 if (uconn == NULL) {
329 brix_upstream_cleanup(up);
330 return NGX_ERROR;
331 }
332
333 up->conn = uconn;
334 up->state = XRD_UP_CONNECTING;
335
336 if (brix_upstream_start_build_wbuf(up, uconn) != NGX_OK) {
337 brix_upstream_cleanup(up);
338 return NGX_ERROR;
339 }
340
341 /* Mark the client session as proxied; client-side reads now route to upstream. */
342 ctx->state = XRD_ST_UPSTREAM;
343
344 if (brix_upstream_start_connect(up, uconn, conf,
345 &chosen_addr, chosen_addrlen) != NGX_OK)
346 {
347 brix_upstream_cleanup(up);
348 ctx->state = XRD_ST_REQ_HEADER;
349 return NGX_ERROR;
350 }
351
352 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
353 "brix: upstream connecting to %s:%d",
354 (char *) conf->upstream_host.data,
355 (int) conf->upstream_port);
356
357 return NGX_OK;
358}
