1/*
2 * status.c — error carrier + kXR name lookup + shell exit-code mapping.
3 *
4 * WHAT: Format a human message into brix_status, name a kXR_* code, and derive a
5 * process exit code the way scripts expect.
6 * WHY: xrdcp/xrdfs print a useful diagnostic and exit non-zero in a stable way;
7 * callers branch on $? so the codes must be predictable.
8 * HOW: Local/socket faults map to 51, usage to 50, any server error to 54

55/*
56 * Classify a failed status as RETRYABLE (transient — a reconnect + re-issue, or a
57 * later attempt, may succeed) vs FATAL (re-issuing cannot help). The async
58 * resilience layer (aio.c) uses this to decide whether to transparently reconnect
59 * and re-drive a request rather than surface the error. Returns 1 = retryable.
60 *
61 * Retryable: transport faults (socket/connect/timeout/reset → XRDC_ESOCK), a frame
62 * desync that a fresh session heals (XRDC_EPROTO), and the server's own "try again"
63 * family (Overloaded / inProgress / noserver / ServerError). Everything else —
64 * auth, not-found, bad-arg, exists, no-space, checksum, unsupported, local missing
65 * feature support, … — is fatal: the request is well-formed but the answer is
66 * "no", and retrying just amplifies load.
67 */
68int
69brix_status_retryable(const brix_status *st)
70{
71 if (st == NULL || st->kxr == 0) {
72 return 0;
73 }
74 switch (st->kxr) {
75 case XRDC_ESOCK: /* transport: connect/timeout/reset/peer-closed */
76 case XRDC_EPROTO: /* desync — a clean reconnect re-syncs framing */
77 case kXR_Overloaded: /* server explicitly busy */
78 case kXR_inProgress: /* operation transiently in progress elsewhere */
79 case kXR_noserver: /* cluster: no server right now, maybe shortly */
80 case kXR_ServerError: /* generic server-side fault, often transient */
81 return 1;
82 default:
83 return 0;
84 }
85}

87int
88brix_shellcode(const brix_status *st)
89{
90 if (st == NULL || st->kxr == 0) {
91 return 0;
92 }
93 switch (st->kxr) {
94 case XRDC_ESOCK: return 51; /* socket/connect/timeout */
95 case XRDC_ERESOLVE: return 51; /* permanent name-resolution failure */
96 case XRDC_EREDIRECT: return 52; /* redirect loop / budget exhausted */
97 case XRDC_EPROTO: return 52; /* malformed server frame */
98 case XRDC_EUSAGE: return 50; /* CLI / argument error */
99 case XRDC_EAUTH: return 53; /* auth needed/failed */
100 case XRDC_EINTEGRITY: return 51;/* data corruption → I/O-error class */
101 case XRDC_EUNSUPPORTED: return 54; /* local feature unsupported */
102 default: return 54; /* a server kXR_error response */
103 }

106/*
107 * Map a failed status to a NEGATIVE errno, for the POSIX layers (FUSE / preload)
108 * that must hand the kernel a -errno. Server kXR_* codes translate per the
109 * project's canonical errno↔kXR table (CLAUDE.md); the local XRDC_E* sentinels
110 * fall back to st->sys_errno when set, else a sensible default. A clean status
111 * (kxr == 0) maps to 0 (success).
112 */
113int
114brix_kxr_to_errno(const brix_status *st)
115{
116 int e;
117
118 if (st == NULL || st->kxr == 0) {
119 return 0;
120 }
121
122 /* Server kXR_* wire codes go through the shared canonical kXR↔errno table
123 * (libxrdproto), so the two directions stay in sync with the module's
124 * errno→kXR. A 0 return means "not a kXR wire error" — fall through to the
125 * client-only XRDC_E* sentinel handling below. */
126 e = brix_errno_from_kxr((uint16_t) st->kxr);
127 if (e != 0) {
128 return -e;
129 }
130
131 switch (st->kxr) {
132 /* Local-side sentinels: prefer the captured sys_errno, else a default. */
133 case XRDC_ESOCK: return st->sys_errno ? -st->sys_errno : -EIO;
134 case XRDC_EPROTO: return -EPROTO;
135 case XRDC_EUSAGE: return -EINVAL;
136 case XRDC_EAUTH: return -EACCES;
137 case XRDC_EINTEGRITY: return -EIO; /* data corruption surfaces as EIO */
138 case XRDC_EUNSUPPORTED: return -ENOTSUP;
139 default: return st->sys_errno ? -st->sys_errno : -EIO;
140 }
141}
