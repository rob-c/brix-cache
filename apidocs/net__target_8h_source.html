1/*
2 * net_target.h — shared external transfer target parser and SSRF policy.
3 *
4 * Centralises URL parsing and DNS-based address classification for all
5 * outbound connections made by the module: native root:// TPC, WebDAV
6 * HTTP-TPC (curl), and any future S3 remote-copy feature.
7 *
8 * Two public helpers:
9 * brix_net_target_parse() — split URL into scheme/host/port/path
10 * brix_net_target_check_dns() — resolve host and reject prohibited addrs
11 *
12 * check_dns() calls getaddrinfo(), which blocks. It MUST be called only
13 * from a background thread (ngx_thread_pool_run or equivalent), never from
14 * the nginx event-loop worker.
15 */
16

24/*
25 * Parsed view of an outbound URL. All ngx_str_t fields point into the
26 * original URL buffer — no pool allocation is performed by the parser.
27 * The caller must keep the original buffer alive while this struct is used.
28 */
29typedef struct {
30 ngx_str_t raw_url; /* the full original URL */
31 ngx_str_t scheme; /* e.g. "https", "root" (no trailing colon) */
32 ngx_str_t host; /* hostname or IP literal (no brackets for IPv6) */
33 ngx_str_t path; /* everything from the first "/" onward, or empty */
34 uint16_t port; /* 0 when not explicitly present in the URL */
35 ngx_flag_t has_port; /* 1 when port was explicitly present */
36} brix_net_target_t;
37
38/*
39 * Policy controlling which destinations are acceptable.
40 *
41 * require_https: 1 → reject any non-https scheme URL
42 * allow_root_scheme: 1 → also accept root:// URLs (used by native TPC)
43 * allow_local: 0 → reject loopback (127/8, ::1) and link-local
44 * allow_private: 0 → reject RFC-1918 (10/8, 172.16/12, 192.168/16)
45 * and IPv6 ULA (fc00::/7)
46 * default_https_port: port substituted when URL has no port and scheme=https
47 * default_root_port: port substituted when URL has no port and scheme=root
48 */
49typedef struct {
50 ngx_flag_t require_https;
51 ngx_flag_t allow_root_scheme;
52 ngx_flag_t allow_local;
53 ngx_flag_t allow_private;
54 uint16_t default_https_port;
55 uint16_t default_root_port;
56} brix_net_target_policy_t;

58/*
59 * brix_net_target_parse — split a URL string into its components.
60 *
61 * The pool parameter is accepted for API consistency but not used; all
62 * fields in *out point into url->data. Returns NGX_OK on success.
63 * On failure writes a NUL-terminated message into err[0..errsz) and
64 * returns NGX_ERROR.
65 *
66 * Recognised URL forms:
67 * scheme://host/path
68 * scheme://host:port/path
69 * scheme://[ipv6addr]:port/path
70 */
71ngx_int_t brix_net_target_parse(ngx_pool_t *pool,
72 const ngx_str_t *url, brix_net_target_t *out,
73 char *err, size_t errsz);

75/*
76 * brix_net_target_check_addr — classify a single resolved sockaddr.
77 *
78 * Returns NGX_ERROR when the address is in a prohibited range under policy,
79 * NGX_OK otherwise. Can be called from any context (no DNS, no blocking).
80 * Used by native TPC connect loop to check each resolved address inline.
81 */
82ngx_int_t brix_net_target_check_addr(const struct sockaddr *sa,

86/*
87 * brix_net_target_check_dns — resolve host and verify against policy.
88 *
89 * Calls getaddrinfo(3) on target->host (BLOCKING — call from thread only).
90 * Rejects the target if any resolved address is in a prohibited range under
91 * the given policy. Returns NGX_OK when all addresses are permitted.
92 * On rejection writes a NUL-terminated message into err[0..errsz).
93 */
94ngx_int_t brix_net_target_check_dns(
95 const brix_net_target_t *target,
96 const brix_net_target_policy_t *policy,
97 char *err, size_t errsz);

99/*
100 * brix_net_target_check_dns_pin — like check_dns, but also returns the first
101 * permitted resolved address as a numeric string in out_ip[0..out_ipsz).
102 *
103 * Closes the DNS-rebind TOCTOU window: the caller validates and resolves once,
104 * then pins out_ip into the transfer agent (e.g. curl CURLOPT_RESOLVE) so the
105 * connection target cannot differ from the address that passed policy. Same
106 * BLOCKING/thread-only contract as check_dns(). Returns NGX_OK with out_ip
107 * filled, NGX_ERROR otherwise (err filled).
108 */
109ngx_int_t brix_net_target_check_dns_pin(
110 const brix_net_target_t *target,
111 const brix_net_target_policy_t *policy,
112 char *out_ip, size_t out_ipsz,
113 char *err, size_t errsz);
114
115/*
116 * brix_net_host_chars_valid — reject host strings that could inject control
117 * bytes or alternate schemes into a redirect/registration string.
118 *
119 * Allows only the characters legal in a hostname or IPv4/IPv6 literal:
120 * [A-Za-z0-9.:-]. Rejects empty, anything with '/', whitespace, NUL, or any
121 * byte outside that set (so "evil\r\nLocation:" or "http://x" never reach a
122 * client that parses a redirect host). Returns 1 if valid, 0 otherwise.
123 */
124int brix_net_host_chars_valid(const char *host, size_t len);
