1/*
2 * config_proxy.c - WebDAV token-load + upstream-proxy + traffic-mirror merge cluster.
3 *
4 * WHAT: the merge-time helpers that load JWKS/multi-issuer token config
5 * (webdav_merge_auth_token_conf), resolve the upstream-proxy backends whether
6 * dynamic-pool or static (webdav_merge_upstream_conf and its helpers), and merge
7 * the Phase-24 traffic-mirror settings then emit the per-endpoint startup summary
8 * (webdav_merge_mirror_and_summary).
9 * WHY: split VERBATIM out of config.c to keep each translation unit under the
10 * file-size gate; behaviour is unchanged. The three entrypoints called by the
11 * merge glue in config.c (webdav_merge_auth_token_conf, webdav_merge_upstream_conf,
12 * webdav_merge_mirror_and_summary) are declared in config_internal.h, as are the
13 * summary helpers (webdav_summary_is_new / webdav_log_endpoint_summary) they call
14 * back into.
15 * HOW: pure move — same includes, same helpers, same control flow as before.
16 */

50char *
51webdav_merge_auth_token_conf(ngx_conf_t *cf, ngx_http_brix_webdav_loc_conf_t *prev,
52 ngx_http_brix_webdav_loc_conf_t *conf)
53{
54 if (conf->token_jwks.len > 0) {
55 int rc;
56
57 rc = brix_jwks_load(cf->log,
58 (const char *) conf->token_jwks.data,
59 conf->jwks_keys, BRIX_MAX_JWKS_KEYS);
60 if (rc < 0) {
61 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
62 "brix_webdav: failed to load JWKS from \"%V\"",
63 &conf->token_jwks);
64 return NGX_CONF_ERROR;
65 }
66 conf->jwks_key_count = rc;
67
68 if (rc > 0
69 && brix_jwks_register_cleanup(cf->pool, conf->jwks_keys,
70 &conf->jwks_key_count) != NGX_OK)
71 {
72 return NGX_CONF_ERROR;
73 }
74 }
75
76 /* Multi-issuer registry (phase-59 W1) — only build it on a leaf location
77 * that actually set brix_webdav_token_config (token_registry stays the
78 * inherited value otherwise). */
79 if (conf->token_config.len > 0 && conf->token_registry == NULL) {
80 brix_token_registry_t *reg = NULL;
81
82 if (brix_token_registry_build(cf,
83 (const char *) conf->token_config.data,
84 BRIX_AUTHZ_CAPABILITY, &reg) != NGX_OK)
85 {
86 return NGX_CONF_ERROR;
87 }
88 conf->token_registry = reg;
89 }
90
91
92 return NGX_CONF_OK;
93}

121static void
122webdav_try_inherit_static_backends(ngx_http_brix_webdav_loc_conf_t *prev,
123 ngx_http_brix_webdav_loc_conf_t *conf)
124{
125 if (conf->upstream_urls == NULL && prev->upstream_urls != NULL) {
126 conf->upstream_urls = prev->upstream_urls;
127 }
128 if (conf->upstream_url.len == 0 && prev->upstream_url.len > 0) {
129 conf->upstream_url = prev->upstream_url;
130 }
131 if (prev->upstream_backends == NULL
132 || conf->upstream_urls != prev->upstream_urls
133 || prev->upstream_url.len != conf->upstream_url.len
134 || (conf->upstream_url.len > 0
135 && ngx_memcmp(prev->upstream_url.data, conf->upstream_url.data,
136 conf->upstream_url.len) != 0))
137 {
138 return;
139 }
140 conf->upstream_backends = prev->upstream_backends;
141 conf->upstream_resolved = prev->upstream_resolved;
142 conf->upstream_host = prev->upstream_host;
143 conf->upstream_url_base = prev->upstream_url_base;
144 conf->upstream_ssl = prev->upstream_ssl;
145 conf->upstream_conf = prev->upstream_conf;
146#if (NGX_HTTP_SSL)
147 conf->upstream_ssl_ctx = prev->upstream_ssl_ctx;
148#endif
149}

151static char *
152webdav_build_static_proxy_backends(ngx_conf_t *cf,
153 ngx_http_brix_webdav_loc_conf_t *prev,
154 ngx_http_brix_webdav_loc_conf_t *conf)
155{
156 static ngx_str_t webdav_proxy_hide_headers[] = {
157 ngx_null_string
158 };
159 ngx_hash_init_t hh;
160
161 webdav_try_inherit_static_backends(prev, conf);
162 if (conf->upstream_backends != NULL) {
163 return webdav_validate_proxy_token(cf, conf);
164 }
165
166 if (webdav_proxy_build_backends(cf, conf) != NGX_OK) {
167 return NGX_CONF_ERROR;
168 }
169
170 hh.max_size = 512;
171 hh.bucket_size = ngx_align(64, ngx_cacheline_size);
172 hh.name = "webdav_proxy_hide_headers_hash";
173 hh.pool = cf->pool;
174 hh.temp_pool = NULL;
175 if (ngx_http_upstream_hide_headers_hash(cf, &conf->upstream_conf,
176 &prev->upstream_conf, webdav_proxy_hide_headers, &hh) != NGX_OK)
177 {
178 return NGX_CONF_ERROR;
179 }
180 return webdav_validate_proxy_token(cf, conf);
181}

183char *
184webdav_merge_upstream_conf(ngx_conf_t *cf, ngx_http_brix_webdav_loc_conf_t *prev,
185 ngx_http_brix_webdav_loc_conf_t *conf)
186{
187 ngx_conf_merge_value(conf->upstream_proxy, prev->upstream_proxy, 0);
188 ngx_conf_merge_str_value(conf->upstream_url, prev->upstream_url, "");
189 ngx_conf_merge_uint_value(conf->upstream_auth, prev->upstream_auth,
190 WEBDAV_PROXY_AUTH_ANONYMOUS);
191 ngx_conf_merge_str_value(conf->upstream_auth_token,
192 prev->upstream_auth_token, "");
193 ngx_conf_merge_msec_value(conf->upstream_conf.connect_timeout,
194 prev->upstream_conf.connect_timeout, 0);
195 ngx_conf_merge_msec_value(conf->upstream_conf.send_timeout,
196 prev->upstream_conf.send_timeout, 0);
197 ngx_conf_merge_msec_value(conf->upstream_conf.read_timeout,
198 prev->upstream_conf.read_timeout, 0);
199 ngx_conf_merge_uint_value(conf->upstream_max_fails,
200 prev->upstream_max_fails, 3);
201 ngx_conf_merge_msec_value(conf->upstream_fail_timeout,
202 prev->upstream_fail_timeout, 30000);
203 ngx_conf_merge_value(conf->proxy_pool_enabled, prev->proxy_pool_enabled, 0);
204
205 if (!conf->upstream_proxy) {
206 return NGX_CONF_OK;
207 }
208 if (conf->proxy_pool_enabled) {
209 return webdav_setup_dynamic_proxy_pool(cf, prev, conf);
210 }
211 if (conf->upstream_backends == NULL) {
212 return webdav_build_static_proxy_backends(cf, prev, conf);
213 }
214 return NGX_CONF_OK;
215}

217char *
218webdav_merge_mirror_and_summary(ngx_conf_t *cf, ngx_http_brix_webdav_loc_conf_t *prev,
219 ngx_http_brix_webdav_loc_conf_t *conf)
220{
221 /* Phase 24: traffic mirror — inherit parent targets, derive enabled, and
222 * build the shadow upstream conf (timeouts/TLS/hide-headers) when active. */
223 if (conf->mirror.targets == NULL) {
224 conf->mirror.targets = prev->mirror.targets;
225 }
226 ngx_conf_merge_str_value(conf->mirror.token, prev->mirror.token, "");
227 ngx_conf_merge_uint_value(conf->mirror.sample_pct, prev->mirror.sample_pct, 100);
228 ngx_conf_merge_uint_value(conf->mirror.method_mask, prev->mirror.method_mask,
229 BRIX_MIRROR_M_DEFAULT);
230 ngx_conf_merge_value(conf->mirror.strip_auth, prev->mirror.strip_auth, 1);
231 ngx_conf_merge_value(conf->mirror.log_diverge, prev->mirror.log_diverge, 1);
232 ngx_conf_merge_msec_value(conf->mirror.timeout_ms, prev->mirror.timeout_ms, 5000);
233 ngx_conf_merge_value(conf->mirror.mirror_writes,
234 prev->mirror.mirror_writes, 0);
235 conf->mirror.enabled = (conf->mirror.targets != NULL
236 && conf->mirror.targets->nelts > 0) ? 1 : 0;
237
238 if (conf->mirror.enabled
239 && conf->mirror_upstream_conf.connect_timeout == 0)
240 {
241 if (brix_http_mirror_setup(cf, conf, prev) != NGX_OK) {
242 return NGX_CONF_ERROR;
243 }
244 }
245
246 /*
247 * Friendly per-endpoint startup summary (visible in `nginx -t` output and
248 * at boot). Only for a location that actually enables WebDAV and whose
249 * config is not merely inherited unchanged from its parent — see
250 * webdav_summary_is_new(). Mirrors the root:// banner in the stream
251 * postconfiguration.
252 */
253 if (conf->common.enable && webdav_summary_is_new(conf, prev)) {
254 webdav_log_endpoint_summary(cf, conf);
255 }
256
257
258 return NGX_CONF_OK;
259}
