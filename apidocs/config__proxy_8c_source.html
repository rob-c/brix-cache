1/*
2 * config_proxy.c - WebDAV token-load + traffic-mirror merge cluster.
3 *
4 * WHAT: the merge-time helpers that load JWKS/multi-issuer token config
5 * (webdav_merge_auth_token_conf) and merge the Phase-24 traffic-mirror settings
6 * then emit the per-endpoint startup summary (webdav_merge_mirror_and_summary).
7 * WHY: split out of config.c to keep each translation unit under the file-size
8 * gate. The two entrypoints called by the merge glue in config.c
9 * (webdav_merge_auth_token_conf, webdav_merge_mirror_and_summary) are declared
10 * in config_internal.h, as are the summary helpers (webdav_summary_is_new /
11 * webdav_log_endpoint_summary) they call back into.
12 * NOTE: the WebDAV reverse-proxy transport (brix_webdav_proxy*) was retired; its
13 * merge helpers were removed with it.
14 */

47char *
48webdav_merge_auth_token_conf(ngx_conf_t *cf, ngx_http_brix_webdav_loc_conf_t *prev,
49 ngx_http_brix_webdav_loc_conf_t *conf)
50{
51 if (conf->token_jwks.len > 0) {
52 int rc;
53
54 rc = brix_jwks_load(cf->log,
55 (const char *) conf->token_jwks.data,
56 conf->jwks_keys, BRIX_MAX_JWKS_KEYS);
57 if (rc < 0) {
58 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
59 "brix_webdav: failed to load JWKS from \"%V\"",
60 &conf->token_jwks);
61 return NGX_CONF_ERROR;
62 }
63 conf->jwks_key_count = rc;
64
65 if (rc > 0
66 && brix_jwks_register_cleanup(cf->pool, conf->jwks_keys,
67 &conf->jwks_key_count) != NGX_OK)
68 {
69 return NGX_CONF_ERROR;
70 }
71 }
72
73 /* Multi-issuer registry (phase-59 W1) — only build it on a leaf location
74 * that actually set brix_webdav_token_config (token_registry stays the
75 * inherited value otherwise). */
76 if (conf->token_config.len > 0 && conf->token_registry == NULL) {
77 brix_token_registry_t *reg = NULL;
78
79 if (brix_token_registry_build(cf,
80 (const char *) conf->token_config.data,
81 BRIX_AUTHZ_CAPABILITY, &reg) != NGX_OK)
82 {
83 return NGX_CONF_ERROR;
84 }
85 conf->token_registry = reg;
86 }
87
88
89 return NGX_CONF_OK;
90}

92char *
93webdav_merge_mirror_and_summary(ngx_conf_t *cf, ngx_http_brix_webdav_loc_conf_t *prev,
94 ngx_http_brix_webdav_loc_conf_t *conf)
95{
96 /* Phase 24: traffic mirror — inherit parent targets, derive enabled, and
97 * build the shadow upstream conf (timeouts/TLS/hide-headers) when active. */
98 if (conf->mirror.targets == NULL) {
99 conf->mirror.targets = prev->mirror.targets;
100 }
101 ngx_conf_merge_str_value(conf->mirror.token, prev->mirror.token, "");
102 ngx_conf_merge_uint_value(conf->mirror.sample_pct, prev->mirror.sample_pct, 100);
103 ngx_conf_merge_uint_value(conf->mirror.method_mask, prev->mirror.method_mask,
104 BRIX_MIRROR_M_DEFAULT);
105 ngx_conf_merge_value(conf->mirror.strip_auth, prev->mirror.strip_auth, 1);
106 ngx_conf_merge_value(conf->mirror.log_diverge, prev->mirror.log_diverge, 1);
107 ngx_conf_merge_msec_value(conf->mirror.timeout_ms, prev->mirror.timeout_ms, 5000);
108 ngx_conf_merge_value(conf->mirror.mirror_writes,
109 prev->mirror.mirror_writes, 0);
110 conf->mirror.enabled = (conf->mirror.targets != NULL
111 && conf->mirror.targets->nelts > 0) ? 1 : 0;
112
113 if (conf->mirror.enabled
114 && conf->mirror_upstream_conf.connect_timeout == 0)
115 {
116 if (brix_http_mirror_setup(cf, conf, prev) != NGX_OK) {
117 return NGX_CONF_ERROR;
118 }
119 }
120
121 /*
122 * Friendly per-endpoint startup summary (visible in `nginx -t` output and
123 * at boot). Only for a location that actually enables WebDAV and whose
124 * config is not merely inherited unchanged from its parent — see
125 * webdav_summary_is_new(). Mirrors the root:// banner in the stream
126 * postconfiguration.
127 */
128 if (conf->common.enable && webdav_summary_is_new(conf, prev)) {
129 webdav_log_endpoint_summary(cf, conf);
130 }
131
132
133 return NGX_CONF_OK;
134}
