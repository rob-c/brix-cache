1/*
2 * sd_ceph_internal.h — cross-TU internals shared by the Ceph/RADOS driver's
3 * implementation files (sd_ceph.c, sd_ceph_io.c, sd_ceph_object.c,
4 * sd_ceph_cred.c).
5 *
6 * WHAT: The driver body was split (file-size guard) across four translation
7 * units. This header carries the pieces they share: the private struct
8 * definitions (the live connection, the per-export state incl. its
9 * cred-conn cache, the per-open object state, and the bundled open
10 * request) plus the declarations of every driver-private function that is
11 * DEFINED in one of those files but REFERENCED from another (the vtable op
12 * functions the descriptor in sd_ceph.c wires up, and the handful of
13 * helpers used across the split).
14 *
15 * WHY: These types/functions are implementation-private (they never appear in
16 * the public sd_ceph.h API — that file declares only the pure LFN->oid
17 * helpers, the per-export config struct, the driver symbol, and the
18 * shared oid-level conn/oid API). Keeping them here rather than in
19 * sd_ceph.h keeps the private surface out of the public header while
20 * still letting the four implementation files agree on one definition.
21 *
22 * HOW: Include-guarded; pulls in the public sd_ceph.h (which, under
23 * BRIX_HAVE_CEPH, brings librados + fs/backend/sd.h) and the striper
24 * header (rados_striper_t is embedded in sd_ceph_state_t). Everything
25 * below the include block is gated on BRIX_HAVE_CEPH, mirroring the
26 * driver body it serves.
27 */

39/* sd_ceph_conn_s — a live librados connection (the full definition of the
40 * opaque sd_ceph_conn_t declared in sd_ceph.h). Defined here, early in the
41 * driver body, rather than down beside sd_ceph_conn_create/_destroy: it must
42 * be a COMPLETE type before sd_ceph_state_t's cred-conn cache entries and
43 * sd_ceph_cleanup (both further down but still ahead of the "shared oid-
44 * level layer" section) can dereference `conn->refs`.
45 *
46 * `refs` is a PIN count: every open object currently resolved onto this
47 * connection (sd_ceph_obj_state_t.conn) holds one ref, taken at open
48 * (sd_ceph_conn_pin) and dropped at close (sd_ceph_conn_unpin). `doomed`
49 * marks a connection with no future path to reuse — either (a) evicted/
50 * removed from the cred-conn cache table while still pinned (a future
51 * lookup can never find it again), or (b) a transient connection that was
52 * never inserted into the table at all (created when every cache slot was
53 * pinned; sd_ceph_cred_conn marks it doomed at birth, before returning it,
54 * since nothing else could ever reach it to free it otherwise) — in both
55 * cases the connection could not be (or never will be) destroyed at the
56 * point it stops being reachable via the table, because refs>0 (or will
57 * become >0 the moment the caller pins it). The ref-dropping unpin that
58 * brings refs to 0 performs the deferred rados_ioctx_destroy/
59 * rados_shutdown. This refcount/pin/deferred-destroy design fixes two
60 * historical bugs: (1) UAF — an in-use connection is never torn down out
61 * from under a still-open handle's pread/pwrite/fstat, no matter how much
62 * LRU eviction pressure other (user,keyring) opens generate; (2) leak — a
63 * transient (all-slots-pinned) connection is guaranteed exactly one
64 * deferred destroy, at its own last unpin, instead of being abandoned. */
65struct sd_ceph_conn_s {
66 rados_t cluster;
67 rados_ioctx_t ioctx;
68 ngx_pool_t *pool;
69 unsigned connected:1;
70 unsigned doomed:1;
71 unsigned refs;
72};

74/* Sizes for the per-user cred-conn cache key (user id + keyring path). These
75 * intentionally MIRROR fs/backend/ucred.h's BRIX_UCRED_CEPH_USER_MAX /
76 * BRIX_UCRED_CEPH_KEYRING_MAX (the source of the strings cached here) rather
77 * than including ucred.h directly: ucred.h pulls in ngx_config.h/ngx_core.h
78 * unconditionally (it has no XRDPROTO_NO_NGX branch, unlike sd.h), which
79 * would break the ngx-free standalone build this file supports (the live
80 * driver tests under tests/ceph/ compile sd_ceph.c with -DXRDPROTO_NO_NGX
81 * and no nginx headers at all). If ucred.h's bounds ever change, update these
82 * to match — sd_ceph.c's cred-conn cache entries must stay large enough to
83 * hold whatever ucred.c can populate brix_sd_cred_t.ceph_user/ceph_keyring
84 * with. */
85#define SD_CEPH_CRED_USER_MAX 128
86#define SD_CEPH_CRED_KEYRING_MAX 1024
87
88/* Bound on the per-export cache of per-user (ceph_user, ceph_keyring) librados
89 * connections (ceph-peruser item). A small bounded LRU: cred-scoped opens are
90 * rare relative to plain opens (only credential-gated exports use them), and
91 * each entry holds a live rados_t/ioctx pair, so the cap keeps worst-case
92 * connection/fd usage predictable under a many-user credential directory. */
93#define SD_CEPH_CRED_CONN_CACHE_MAX 8
94
95/* Bound on a single xattr value we buffer for the size-probe path. */
96#define SD_CEPH_XATTR_MAX (64u * 1024)

98/* One cached per-user connection: the (user, keyring) key it was opened for,
99 * the connection itself, and an LRU generation counter (higher = more
100 * recently used) so eviction picks the least-recently-used entry. */

132/* Driver-private per-open state (obj->state): the object id + a cached size.
133 * `striped` is set when the object is stored in the stock-XrdCeph libradosstriper
134 * layout and must be read back through the striper to reassemble its bytes.
135 * `ioctx` is the connection this OPEN was resolved against — the export's
136 * static service ioctx for a plain open, or a per-user cred-conn's ioctx for
137 * a cred-scoped open (ceph-peruser item). All raw byte ops (pread/pwrite/
138 * ftruncate/fstat) MUST use this field rather than reaching back through
139 * obj->inst->state->ioctx, or a per-user open would silently operate under
140 * the service account after the initial open — defeating the credential
141 * scoping the whole feature exists to provide.
142 * `conn` is NON-NULL only for a cred-scoped open: it is the pinned
143 * sd_ceph_conn_t this object's `ioctx` was drawn from (sd_ceph_conn_pin at
144 * open, sd_ceph_conn_unpin at close). This is the fix for the cred-conn UAF
145 * — as long as `conn` is non-NULL and pinned, the LRU cache can never
146 * destroy the underlying rados_t/ioctx while this handle is still open,
147 * no matter how many other (user,keyring) opens run concurrently. A plain
148 * (service-credential) open leaves `conn` NULL: the export's own ioctx is
149 * instance-lived (destroyed only at sd_ceph_cleanup, never LRU-evicted) and
150 * needs no pin. */
151typedef struct {
152 char oid[1024];
153 uint64_t size;
154 rados_ioctx_t ioctx;
155 sd_ceph_conn_t *conn;
156 unsigned for_write:1;
157 unsigned striped:1;
158} sd_ceph_obj_state_t;

160/* sd_ceph_open_req_t — bundled inputs for one object open (plain or
161 * cred-scoped). Collapsing the seven separate open parameters into a single
162 * request record keeps sd_ceph_open_on_ioctx and its extracted probe/apply
163 * helpers within the parameter budget and lets each helper take the whole
164 * request by const pointer instead of re-threading the individual fields.
165 * - st : export state (key_prefix + shared striper cache)
166 * - ioctx : the connection (service or per-user) rados_* calls run on
167 * - pin_conn : non-NULL only for a cred-scoped open (pinned on success)
168 * - path : the logical path to resolve to an object id
169 * - sd_flags : BRIX_SD_O_* open intent (create/excl/trunc/write)
170 * - mode : POSIX mode (unused by RADOS; carried for signature parity) */
171typedef struct {
172 brix_sd_instance_t *inst;
173 sd_ceph_state_t *st;
174 rados_ioctx_t ioctx;
175 sd_ceph_conn_t *pin_conn;
176 const char *path;
177 int sd_flags;
178 mode_t mode;
179} sd_ceph_open_req_t;
