28int
29brix_scan_emitq_submit(brix_scan_emitq_t *q, uint64_t seq, void *payload,
30 brix_scan_emit_cb cb, void *ctx)
31{
32 unsigned idx;
33 int released = 0;
34
35 /* already emitted, or beyond the reorder window ⇒ caller error */
36 if (seq < q->next || seq >= q->next + q->window) {
37 return -1;
38 }
39 idx = (unsigned) (seq % q->window);
40 if (q->used[idx]) {
41 return -1; /* duplicate seq in-flight */
42 }
43 q->slots[idx] = payload;
44 q->used[idx] = 1;
45
46 /* drain every now-contiguous record in order */
47 while (q->used[q->next % q->window]) {
48 unsigned i = (unsigned) (q->next % q->window);
49 void *p = q->slots[i];
50
51 q->used[i] = 0;
52 q->slots[i] = NULL;
53 cb(ctx, q->next, p);
54 q->next++;
55 released++;
56 }
57 return released;
58}
