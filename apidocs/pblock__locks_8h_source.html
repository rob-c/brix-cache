1/*
2 * pblock_locks.h — Phase-83 F15 byte-range locks / mandatory lease
3 * enforcement for pblock.
4 *
5 * WHAT: A driver-owned `locks(path, off, len, mode, owner, expires_at)`
6 * catalog table plus MANDATORY enforcement: a live foreign lease
7 * refuses conflicting opens at open time (whole-file leases —
8 * `len = 0` means the whole file), denies overlapping pwrites through
9 * an open-time snapshot (range leases, `len > 0`), and blocks the
10 * bypass routes — unlink, rename and staged publish of a leased path.
11 * Modes: 'W' excludes other writers, 'X' excludes everyone else.
12 * Refusals are errno EBUSY, which the protocol edge maps to
13 * kXR_FileLocked / HTTP 423.
14 *
15 * WHY: Real concurrent-writer exclusion (two STORs to one path), the
16 * "single-writer" guarantee TPC destinations want, and a mandatory
17 * backing model for WebDAV LOCK semantics — enforced by the storage
18 * driver, not advisory.
19 *
20 * HOW: Leases are rows (owner = synthetic catalog uid; `expires_at` is a
21 * unix-seconds deadline so a crashed client can never wedge the
22 * export — expired rows are ignored everywhere). Tests plant and
23 * release rows via sqlite3, the standard ctl channel. Enforcement DB
24 * reads happen only at metadata boundaries (open/unlink/rename/
25 * commit); the pwrite check is a pure scan of the per-handle snapshot
26 * taken at open — per the phase design, a conflicting later open is
27 * refused at *its* open, which is what makes per-I/O DB hits
28 * unnecessary. A later-acquired lease does not affect already-open
29 * handles (the documented snapshot model). Gated by its own
30 * `locks=1` static opt; ngx-free (libc + sqlite3),
31 * BRIX_HAVE_SQLITE-gated.
32 *
33 * Requires: pblock_store.h, sd_pblock_internal.h and fs/backend/sd.h before
34 * inclusion.
35 */

39/* One snapshotted foreign range lease (os->lock_rng array element). Named
40 * struct so sd_pblock_internal.h can hold the pointer without including us. */
41typedef struct pblock_lock_rng_s {

43 int64_t len; /* > 0: range; 0 rows are refused at open instead */
44 int64_t expires_at; /* unix seconds; ignored once in the past */
