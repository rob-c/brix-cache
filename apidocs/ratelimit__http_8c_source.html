1/*
2 * ratelimit_http.c — Phase 25 HTTP/WebDAV enforcement.
3 *
4 * An NGX_HTTP_ACCESS_PHASE handler (registered after the auth handler so the
5 * identity is populated) evaluates every rule attached to the location. A
6 * request-rate throttle returns 429 with Retry-After (unless `nodelay`); a
7 * bandwidth rule that is already overflowing returns 429 too, and otherwise
8 * stashes the matched rule + key on the request ctx so the body filter can
9 * charge the actual bytes sent.
10 */

12#include "protocols/webdav/webdav.h"
13#include "core/http/http_headers.h"
14#include "observability/metrics/metrics_macros.h"

27static ngx_int_t
28rl_reject(ngx_http_request_t *r, brix_rl_rule_t *rule, uint32_t wait_sec,
29 const char *key_str)
30{
31 BRIX_RL_METRIC_INC(rl_throttled_http_total);
32
33 if (!rule->nodelay && wait_sec > 0) {
34 char buf[NGX_INT_T_LEN + 1];
35 ngx_snprintf((u_char *) buf, sizeof(buf), "%ud%Z", wait_sec);
36 (void) brix_http_set_header(r, "Retry-After", buf, NULL);
37 }
38
39 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
40 "xrootd rate limit: 429 for \"%V\" (key=%s, wait=%uD)",
41 &r->uri, key_str, wait_sec);
42
43 return NGX_HTTP_TOO_MANY_REQUESTS;
44}

46/*
47 * WHAT: Ensure the WebDAV request ctx exists, allocating+installing it lazily.
48 * WHY: The body/log phase charges bandwidth and releases concurrency slots
49 * via this ctx; the auth gate may not have created one (anonymous
50 * WebDAV), so a rate-limit dimension that needs to stash state must be
51 * able to conjure it on demand.
52 * HOW: Return the existing ctx unchanged; otherwise pcalloc from the request
53 * pool and install it, mirroring the original inline behaviour exactly
54 * (a failed pcalloc leaves the slot unset and returns NULL).
55 */
56static ngx_http_brix_webdav_req_ctx_t *
57rl_ensure_ctx(ngx_http_request_t *r, ngx_http_brix_webdav_req_ctx_t *wctx)
58{
59 if (wctx != NULL) {
60 return wctx;
61 }
62 wctx = ngx_pcalloc(r->pool, sizeof(*wctx));
63 if (wctx != NULL) {
64 ngx_http_set_ctx(r, wctx, ngx_http_brix_webdav_module);
65 }
66 return wctx;
67}

69/*
70 * WHAT: Resolve the request path lazily, only the first time a VOLUME rule is
71 * encountered.
72 * WHY: Only VOLUME rules key on the path; resolving it is a syscall we skip
73 * entirely when no such rule matches, and do at most once per request.
74 * HOW: No-op unless this is a VOLUME rule and the path has not been resolved
75 * yet; on resolve failure the buffer is emptied (rl_req.path aliases it),
76 * and *have_path is latched so the work never repeats.
77 */
78static void
79rl_resolve_volume_path(ngx_http_request_t *r,
80 ngx_http_brix_webdav_loc_conf_t *lcf, const brix_rl_rule_t *rule,
81 char *path, int *have_path)
82{
83 if (rule->key_type != BRIX_RL_KEY_VOLUME || *have_path) {
84 return;
85 }
86 if (ngx_http_brix_webdav_resolve_path(r, lcf->common.root_canon,
87 path, BRIX_RL_PATH_LEN) != NGX_OK)
88 {
89 path[0] = '\0';
90 }
91 *have_path = 1;
92}

94/*
95 * WHAT: Evaluate the bandwidth dimension of one rule.
96 * WHY: An already-overflowing bandwidth key rejects immediately; otherwise
97 * the matched rule+key are stashed so the log phase can charge the bytes
98 * actually sent.
99 * HOW: Pre-check; on NGX_AGAIN return the 429 verdict. On pass, ensure a ctx
100 * (updating *wctx_p) and record the key+rule when one is available.
101 * Returns NGX_OK to continue, or the reject code to propagate.
102 */
103static ngx_int_t
104rl_check_bw(ngx_http_request_t *r, brix_rl_rule_t *rule,
105 const char *key_str, ngx_http_brix_webdav_req_ctx_t **wctx_p)
106{
107 ngx_http_brix_webdav_req_ctx_t *wctx;
108 uint32_t wait_sec;
109
110 if (rule->bw_rate <= 0) {
111 return NGX_OK;
112 }
113 if (brix_rl_bw_check(rule, key_str, &wait_sec) == NGX_AGAIN) {
114 return rl_reject(r, rule, wait_sec, key_str);
115 }
116 wctx = rl_ensure_ctx(r, *wctx_p);
117 *wctx_p = wctx;
118 if (wctx != NULL) {
119 ngx_cpystrn((u_char *) wctx->rl_key_str, (u_char *) key_str,
120 sizeof(wctx->rl_key_str));
121 wctx->rl_bw_rule = rule;
122 }
123 return NGX_OK;
124}

126/*
127 * WHAT: Evaluate the concurrency dimension (W7) of one rule.
128 * WHY: Reserve at most one in-flight slot per request so the log-phase
129 * release pairs exactly; an exhausted key rejects with 429.
130 * HOW: Skip unless the rule caps concurrency and no slot is held yet. On
131 * acquire failure return the reject code. On success, ensure a ctx to
132 * record the release target and latch *conc_held; if no ctx can be
133 * allocated, hand the slot back immediately rather than leak it.
134 * Returns NGX_OK to continue, or the reject code to propagate.
135 */
136static ngx_int_t
137rl_check_conc(ngx_http_request_t *r, brix_rl_rule_t *rule,
138 const char *key_str, ngx_http_brix_webdav_req_ctx_t **wctx_p,
139 int *conc_held)
140{
141 ngx_http_brix_webdav_req_ctx_t *wctx;
142
143 if (rule->req_conc <= 0 || *conc_held) {
144 return NGX_OK;
145 }
146 if (brix_rl_conc_acquire(rule, key_str) == NGX_AGAIN) {
147 return rl_reject(r, rule, 0, key_str);
148 }
149 wctx = rl_ensure_ctx(r, *wctx_p);
150 *wctx_p = wctx;
151 if (wctx != NULL) {
152 ngx_cpystrn((u_char *) wctx->rl_conc_key, (u_char *) key_str,
153 sizeof(wctx->rl_conc_key));
154 wctx->rl_conc_rule = rule;
155 *conc_held = 1;
156 } else {
157 /* No ctx to stash the release target — give the slot back now rather
158 * than leak it (degrades to no concurrency cap). */
159 brix_rl_conc_release(rule, key_str);
160 }
161 return NGX_OK;
162}

164/*
165 * WHAT: Evaluate every rate-limit dimension of one matched rule.
166 * WHY: Keeps the per-rule verdict logic (rate → bandwidth → concurrency, in
167 * that short-circuit order) in one place so the loop body stays flat.
168 * HOW: Request-rate rejects on NGX_AGAIN; bandwidth and concurrency delegate
169 * to their helpers (which may update *wctx_p / *conc_held). Returns
170 * NGX_OK to continue the loop, or a 429 code to return from the handler.
171 */
172static ngx_int_t
173rl_apply_rule(ngx_http_request_t *r, brix_rl_rule_t *rule,
174 const char *key_str, ngx_http_brix_webdav_req_ctx_t **wctx_p,
175 int *conc_held)
176{
177 uint32_t wait_sec;
178 ngx_int_t rc;
179
180 if (rule->req_rate > 0
181 && brix_rl_check(rule, key_str, &wait_sec) == NGX_AGAIN)
182 {
183 return rl_reject(r, rule, wait_sec, key_str);
184 }
185
186 rc = rl_check_bw(r, rule, key_str, wctx_p);
187 if (rc != NGX_OK) {
188 return rc;
189 }
190
191 return rl_check_conc(r, rule, key_str, wctx_p, conc_held);
192}

194ngx_int_t
195brix_rl_http_access_handler(ngx_http_request_t *r)
196{
197 ngx_http_brix_webdav_loc_conf_t *lcf;
198 ngx_http_brix_webdav_req_ctx_t *wctx;
199 brix_rl_rule_t *rules;
200 rl_key_req_t rl_req;
201 ngx_uint_t i;
202 char key_str[BRIX_RL_KEY_LEN];
203 char path[BRIX_RL_PATH_LEN];
204 int have_path = 0;
205 int conc_held = 0; /* W7: one slot/request */
206 ngx_int_t rc;
207
208 if (r != r->main) {
209 return NGX_DECLINED; /* subrequests inherit the parent's verdict */
210 }
211
212 lcf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
213 if (lcf == NULL || lcf->rl_rules == NULL || lcf->rl_rules->nelts == 0) {
214 return NGX_DECLINED;
215 }
216 /*
217 * In WebDAV proxy mode the webdav module's request-ctx slot is owned by the
218 * proxy (webdav_proxy_ctx_t, set in proxy.c), NOT the rate-limit req_ctx.
219 * Touching it here would acquire a concurrency slot the log phase can never
220 * release (it would read the proxy ctx as a req_ctx -> garbage rule pointer
221 * -> crash, see brix_rl_http_log_handler). Rate limiting does not apply to
222 * proxied requests; leave the slot to the proxy.
223 */
224 if (lcf->upstream_proxy) {
225 return NGX_DECLINED;
226 }
227 wctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
228
229 rules = lcf->rl_rules->elts;
230 path[0] = '\0';
231
232 /*
233 * Resolve the identity handles once, before the per-rule loop: the unified
234 * identity from the WebDAV ctx and the connection address. brix_rl_key_http
235 * then reads this bundle rather than re-deriving them per call. `path` is
236 * still filled lazily below (only VOLUME rules need it); rl_req.path aliases
237 * the buffer so those rules see the resolved value.
238 */
239 rl_req.id = wctx ? wctx->identity : NULL;
240 rl_req.wctx = wctx;
241 rl_req.ip = &r->connection->addr_text;
242 rl_req.path = path;
243
244 for (i = 0; i < lcf->rl_rules->nelts; i++) {
245
246 rl_resolve_volume_path(r, lcf, &rules[i], path, &have_path);
247
248 rc = brix_rl_key_http(&rules[i], &rl_req, key_str, sizeof(key_str));
249 if (rc != NGX_OK) { continue; } /* volume prefix / key did not match */
250
251 rc = rl_apply_rule(r, &rules[i], key_str, &wctx, &conc_held);
252 if (rc != NGX_OK) {
253 return rc;
254 }
255 }
256
257 return NGX_DECLINED; /* allowed — fall through to the next phase */
258}

261/* bandwidth charge (LOG phase) * The WebDAV file-serve path sends bytes via a thread-pool/sendfile context
262 * that does not reliably traverse a chained body filter, so bandwidth is
263 * charged once per request in the log phase from the known response size. */
264ngx_int_t
265brix_rl_http_log_handler(ngx_http_request_t *r)
266{
267 ngx_http_brix_webdav_loc_conf_t *lcf;
268 ngx_http_brix_webdav_req_ctx_t *wctx;
269 off_t nbytes;
270
271 if (r != r->main) {
272 return NGX_OK;
273 }
274 /*
275 * In proxy mode the webdav module ctx slot holds a webdav_proxy_ctx_t, not
276 * our req_ctx — reading rl_conc_rule/rl_conc_key off it would dereference a
277 * garbage rule pointer (observed SIGSEGV in brix_rl_conc_release). The
278 * access handler likewise skips proxied locations, so there is never a slot
279 * to release here. Bail before interpreting the foreign ctx.
280 */
281 lcf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
282 if (lcf == NULL || lcf->upstream_proxy) {
283 return NGX_OK;
284 }
285 wctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
286 if (wctx == NULL) {
287 return NGX_OK;
288 }
289
290 /* W7 — release the concurrency slot exactly once (the log phase runs for
291 * every finalized request, including errors/aborts, so this never leaks). */
292 if (wctx->rl_conc_rule != NULL && wctx->rl_conc_key[0] != '\0') {
293 brix_rl_conc_release((brix_rl_rule_t *) wctx->rl_conc_rule,
294 wctx->rl_conc_key);
295 wctx->rl_conc_rule = NULL;
296 wctx->rl_conc_key[0] = '\0';
297 }
298
299 if (wctx->rl_bw_rule == NULL || wctx->rl_key_str[0] == '\0') {
300 return NGX_OK;
301 }
302
303 nbytes = r->headers_out.content_length_n;
304 if (nbytes <= 0) {
305 /* Fallback to bytes actually written on the connection minus headers
306 * (best-effort when the length was not declared, e.g. chunked). */
307 nbytes = (off_t) r->connection->sent - (off_t) r->header_size;
308 }
309 if (nbytes > 0) {
310 brix_rl_charge_bytes((brix_rl_rule_t *) wctx->rl_bw_rule,
311 wctx->rl_key_str, (size_t) nbytes);
312 }
313 return NGX_OK;
314}
