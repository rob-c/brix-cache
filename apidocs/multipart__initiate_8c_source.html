1/*
2 * WHAT: S3-compatible InitiateMultipartUpload endpoint. Handles POST /bucket/key?uploads
3 * requests to create a new multipart upload session. Generates an opaque upload ID from
4 * timestamp components, creates a staging directory for partial uploads, and returns
5 * the AWS-standard XML response body with Bucket/Key/UploadId elements.
6 */
7
8/* WHY: Multipart uploads are essential for large file transfers (>5GB) where single PUT
9 * operations would timeout or consume excessive memory. The upload ID provides a unique
10 * session identifier that ties together all subsequent UploadPart and CompleteMultipartUpload
11 * requests — each part must reference the same upload_id to be included in the final assembly.
12 * Staging directory isolation (0700 permissions) prevents cross-upload contamination between
13 * concurrent multipart sessions. */
14
15/* HOW: Three-phase operation. Phase 1: Generate upload ID using ngx_gettimeofday() + pid —
16 * three 8-hex-digit fields produce 96 bits of uniqueness (seconds + microseconds + nginx worker
17 * PID). Note: initial attempt used ngx_snprintf with %l format; the fallback snprintf on lines
18 * 40-44 uses plain %lx to avoid ngx_snprintf's trailing 'l' suffix characters. Phase 2: Create
19 * staging directory via s3_get_mpu_dir() + brix_mkdir_confined_canon() — confined path ensures
20 * staging stays under the export root with 0700 permissions for upload isolation. Phase 3: Build
21 * AWS-standard XML response body using snprintf into 512-byte stack buffer, convert to ngx_buf_t
22 * via ngx_create_temp_buf(), send header + output filter chain. Three OOM/error paths all return
23 * NGX_HTTP_INTERNAL_SERVER_ERROR with metric increment. */
24

37ngx_int_t
38s3_handle_multipart_initiate(ngx_http_request_t *r,
39 const char *fs_path,
40 ngx_http_s3_loc_conf_t *cf,
41 const char *key_str)
42{
43 char upload_id[25]; /* 8+8+8 hex digits + NUL */
44 char mpu_dir[PATH_MAX];
45 char xml_buf[512];
46 ngx_buf_t *b;
47 struct timeval tv;
48 u_char *p;
49 size_t xml_len;
50
51 /* Generate a simple opaque upload ID from timestamp + usec + pid.
52 * Three 8-hex fields gives 96 bits of uniqueness — sufficient for a
53 * staging token that is only valid for the lifetime of the upload. */
54 ngx_gettimeofday(&tv);
55 p = ngx_snprintf((u_char *)upload_id, sizeof(upload_id) - 1,
56 "%08xl%08xl%08xl",
57 (unsigned long)tv.tv_sec,
58 (unsigned long)tv.tv_usec,
59 (unsigned long)ngx_pid);
60 *p = '\0';
61
62 /* Trim the format suffix characters ('l') that ngx_snprintf emits for %l */
63 /* Simpler: just use fixed hex with snprintf since upload_id is not a path */
64 snprintf(upload_id, sizeof(upload_id), "%08lx%08lx%08lx",
65 (unsigned long)tv.tv_sec,
66 (unsigned long)tv.tv_usec,
67 (unsigned long)ngx_pid);
68
69 /* Phase 39 (WS8/HTTP-2): before creating this upload's staging dir,
70 * opportunistically reap abandoned staging dirs in the SAME directory whose
71 * mtime is idle past brix_s3_mpu_max_age (a client that never sent
72 * Complete/Abort). Bounded to one readdir; off (0) by default. */
73 if (cf->mpu_max_age > 0) {
74 (void) s3_mpu_reap_stale(r, cf, fs_path, (time_t) cf->mpu_max_age);
75 }
76
77 s3_get_mpu_dir(fs_path, upload_id, mpu_dir, sizeof(mpu_dir));
78
79 if (brix_mkdir_confined_canon(r->connection->log, cf->common.root_canon,
80 mpu_dir, 0700) != 0)
81 {
82 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, errno,
83 "s3 initiate_mpu: mkdir(\"%s\") failed", mpu_dir);
84 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
85 return NGX_HTTP_INTERNAL_SERVER_ERROR;
86 }
87
88 /* Build the XML response body */
89 xml_len = (size_t)snprintf(xml_buf, sizeof(xml_buf),
90 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
91 "<InitiateMultipartUploadResult"
92 " xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">\n"
93 " <Bucket>%s</Bucket>\n"
94 " <Key>%s</Key>\n"
95 " <UploadId>%s</UploadId>\n"
96 "</InitiateMultipartUploadResult>\n",
97 cf->bucket.data ? (const char *)cf->bucket.data : "default",
98 key_str,
99 upload_id);
100
101 if (xml_len >= sizeof(xml_buf)) {
102 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
103 return NGX_HTTP_INTERNAL_SERVER_ERROR;
104 }
105
106 b = ngx_create_temp_buf(r->pool, xml_len);
107 if (b == NULL) {
108 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
109 return NGX_HTTP_INTERNAL_SERVER_ERROR;
110 }
111 ngx_memcpy(b->pos, xml_buf, xml_len);
112 b->last = b->pos + xml_len;
113 b->last_buf = 1;
114
115 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
116 (ngx_str_t) ngx_string("application/xml"), b);
117}
