9/*
10 * ftp_ev_mode_e.c — non-blocking MODE E (GFD.020 §3.4) extended-block transfers.
11 *
12 * WHAT: the RETR framing pump (frame the source as offset-addressed extended
13 * blocks over one data connection), plus the 111 restart / 112 perf markers the
14 * sender emits. The STOR reassembly receiver lives in ftp_ev_mode_e_recv.c.
15 *
16 * WHY: the sync engine drives MODE E with a blocking poll() across the listener
17 * and every live stream (ftp_stor_mode_e). Here each stream is its own nginx
18 * connection with its own read event, and the listener keeps accepting: globus
19 * opens all parallel streams at once and handshakes their TLS concurrently, so the
20 * receiver must bring them up promptly and interleave their reads under the event
21 * loop — never draining one before accepting the next.
22 *
23 * HOW: RETR reuses the single data connection (ftp_ev_data.c brings it up, TLS and
24 * all); brix_ftp_ev_retr_mode_e_start() installs a write pump that prepends the
25 * 17-byte header to each chunk and closes with an EOF|EOD trailer.
26 */

72
73 for ( ;; ) {
74 if (dc->buf_pos < dc->buf_len) {
75 ssize_t n = c->send(c, dc->buf + dc->buf_pos,
76 dc->buf_len - dc->buf_pos);
77 if (n > 0) {
78 dc->buf_pos += (size_t) n;
79 continue;
80 }
81 if (n == NGX_AGAIN) {

83 brix_ftp_ev_data_finish(dc, NGX_ERROR);
84 return;
85 }
86 ngx_add_timer(wev, BRIX_FTP_EV_IO_TIMEO);
87 return;
88 }

90 return;
91 }
92
93 if (dc->eb_phase == 2) { /* trailer sent */
94 brix_ftp_ev_data_finish(dc, NGX_OK);
95 return;
96 }
97 if (dc->eb_phase == 1) { /* frame the EOF|EOD end */
98 /* No payload; the OFFSET field carries the total EOD count (1). */
99 ftp_eb_pack(dc->buf, (u_char) (FTP_EB_EOF | FTP_EB_EOD), 0, 1);
100 dc->buf_len = FTP_EB_HDR;
101 dc->buf_pos = 0;
102 dc->eb_phase = 2;
103 continue;
104 }
105
106 if (dc->off >= dc->size) { /* whole file framed */
107 dc->eb_phase = 1;
108 continue;
109 }
110 {
111 size_t want = (dc->size - dc->off > BRIX_FTP_EV_XFER_BUF)
112 ? BRIX_FTP_EV_XFER_BUF
113 : (size_t) (dc->size - dc->off);
114 ssize_t got = brix_vfs_file_pread(dc->fh, dc->buf + FTP_EB_HDR,
115 want, dc->off);
116 if (got <= 0) {
117 brix_ftp_ev_data_finish(dc, NGX_ERROR);
118 return;
119 }
120 ftp_eb_pack(dc->buf, 0 /* data */, (uint64_t) got, (uint64_t) dc->off);
121 dc->off += got;
122 dc->buf_len = FTP_EB_HDR + (size_t) got;
123 dc->buf_pos = 0;
124 if (dc->off - dc->eb_marked >= FTP_EV_EB_MARKER_BYTES) {
125 ev_eb_marker_perf(fc, dc->off);
