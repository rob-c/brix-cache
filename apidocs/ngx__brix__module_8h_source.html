4/*
5 * ngx_brix_module.h — umbrella internal header for the nginx XRootD stream module.
6 *
7 * Every .c file in the module includes this header. It provides:
8 * 1. All system, nginx, and OpenSSL headers needed across the codebase.
9 * 2. Core type definitions (via src/types/ sub-headers).
10 * 3. Forward declarations of every subsystem's public API.
11 *
12 * To find a specific type, see the corresponding src/types/ file:
13 * types/tunables.h — BRIX_* size limits, auth constants, metric macros
14 * types/identity.h — brix_identity_t (verified principal state)
15 * types/state.h — brix_state_t enum, opaque forward decls
16 * types/file.h — brix_file_t (per-open-file bookkeeping)
17 * types/context.h — brix_ctx_t (per-connection state)
18 * types/config.h — ngx_stream_brix_srv_conf_t (per-server config)
19 *
20 * Maintainer map for non-XRootD specialists:
21 * - stream/ owns the nginx module descriptor, command table, and lifecycle hooks.
22 * - config/ owns directive parsing, config inheritance, and startup validation.
23 * - connection/ owns nginx event wiring and the byte-accumulation state machine.
24 * - handshake/ owns the initial client hello and opcode dispatcher.
25 * - session/ owns protocol/login/auth/liveness requests.
26 * - read/ owns file-handle lifecycle: stat/open/read/readv/close.
27 * - write/ owns storage mutation: write/pgwrite/sync/truncate/mkdir/rm/rmdir/mv/chmod.
28 * - query/ owns kXR_query and kXR_prepare.
29 * - dirlist/ owns kXR_dirlist (directory listing with optional dStat).
30 * - fattr/ owns kXR_fattr (file extended attribute operations).
31 * - path/ translates untrusted client paths and enforces VO ACL policy.
32 * - response/ owns response formatting and the wire send helpers.
33 * - aio/ owns async pread/pwrite (thread-pool path) and response chain builders.
34 * - upstream/ owns the XRootD upstream redirector query flow.
35 * - cache/ owns read-through cache fill and lock management.
36 * - cms/ owns CMS manager heartbeat/registration.
37 * - manager/ owns the server registry (cluster/redirector mode) and CMS server handler.
38 * - gsi/ owns GSI/x509 certificate operations.
39 * - voms/ owns VOMS attribute-certificate extraction (runtime dlopen).
40 * - token/ owns JWT/WLCG bearer-token validation and scope checks.
41 * - metrics/ owns Prometheus shared-memory counters.
42 * - protocol/ owns XRootD wire-format constants and packed structs (headers only).
43 */

245/*
246 * Include architecture — why the headers look circular but are safe:
247 *
248 * Every sub-header (e.g. read/read.h) starts with:
249 * #include "../ngx_brix_module.h"
250 * so it can use brix_ctx_t, ngx_stream_brix_srv_conf_t, etc.
251 *
252 * This file then #includes those same sub-headers below to pull their
253 * function declarations into every translation unit that includes us.
254 * The resulting include cycle (A includes B, B includes A) is broken
255 * by the include guard at the top of this file: when B tries to include
256 * A a second time, BRIX_NGX_BRIX_MODULE_H is already defined and the
257 * re-include is a no-op. By the time B is processed, all types defined
258 * earlier in this file are already visible to B.
259 *
260 * The net effect: every .c file that does #include "ngx_brix_module.h"
261 * automatically gets the declarations from every subsystem header too.
262 * No .c file needs to know which sub-header declares a particular function.
263 */
