8/* Human-readable name for an BRIX_AUTH_* enum value (for the startup log). */
9static const char *
10brix_auth_mode_name(ngx_uint_t auth)
11{
12 switch (auth) {
13 case BRIX_AUTH_NONE: return "none (anonymous)";
14 case BRIX_AUTH_GSI: return "GSI/x509";
15 case BRIX_AUTH_TOKEN: return "bearer token";
16 case BRIX_AUTH_BOTH: return "GSI or token";
17 case BRIX_AUTH_SSS: return "shared-secret (sss)";
18 case BRIX_AUTH_UNIX: return "unix (self-asserted)";
19 case BRIX_AUTH_KRB5: return "Kerberos 5";
20 case BRIX_AUTH_HOST: return "host allowlist";
21 case BRIX_AUTH_PWD: return "password (pwd)";
22 default: return "unknown";
23 }
24}

26/* Log a one-time NOTICE summary of the effective server config (auth, roots,
27 * ports, enabled features). */
28static void
29brix_log_startup_summary(ngx_log_t *log, ngx_stream_brix_srv_conf_t *xcf)
30{
31 ngx_log_error(NGX_LOG_NOTICE, log, 0,
32 "brix: root:// endpoint ready — export \"%V\" (%s), auth: %s",
33 &xcf->common.root,
34 xcf->common.allow_write ? "read-write" : "read-only",
35 brix_auth_mode_name(xcf->auth));
36
37 if (xcf->crl.len > 0) {
38 if (xcf->crl_reload > 0) {
39 ngx_log_error(NGX_LOG_NOTICE, log, 0,
40 "brix: revocation: CRL \"%V\", reloaded every %T s",
41 &xcf->crl, (time_t) xcf->crl_reload);
42 } else {
43 ngx_log_error(NGX_LOG_NOTICE, log, 0,
44 "brix: revocation: CRL \"%V\", loaded once at startup "
45 "(set brix_crl_reload for periodic refresh)",
46 &xcf->crl);
47 }
48 }
49
50 if (xcf->jwks_key_count > 0) {
51 ngx_log_error(NGX_LOG_NOTICE, log, 0,
52 "brix: token validation: %d JWKS key(s) loaded",
53 xcf->jwks_key_count);
54 }
55
56 if (xcf->manager_mode) {
57 ngx_log_error(NGX_LOG_NOTICE, log, 0,
58 "brix: mode: cluster manager — redirects clients to data "
59 "servers (does not serve local files)");
60 }
61 if (xcf->proxy.enable) {
62 ngx_log_error(NGX_LOG_NOTICE, log, 0,
63 "brix: mode: proxy — forwards client traffic to a backend");
64 }
65 if (xcf->cache) {
66 ngx_log_error(NGX_LOG_NOTICE, log, 0,
67 "brix: mode: read-through cache in front of an origin");
68 }
69
70 /* Valid-but-noteworthy settings a first-time admin should see explicitly. */
71 if (xcf->auth == BRIX_AUTH_NONE) {
72 ngx_log_error(NGX_LOG_NOTICE, log, 0,
73 "brix: NOTE: no authentication required — this endpoint is "
74 "OPEN to anonymous clients (set brix_auth to require "
75 "credentials)");
76 }
77 if ((xcf->auth == BRIX_AUTH_GSI || xcf->auth == BRIX_AUTH_BOTH)
78 && xcf->crl.len == 0)
79 {
80 ngx_log_error(NGX_LOG_WARN, log, 0,
81 "brix: NOTE: GSI auth is enabled but no CRL is configured — "
82 "REVOKED certificates will be ACCEPTED (set brix_crl to a CRL "
83 "file/dir, e.g. /etc/grid-security/certificates)");
84 }
85 if (xcf->common.allow_write) {
86 ngx_log_error(NGX_LOG_NOTICE, log, 0,
87 "brix: NOTE: write access is enabled — authorized clients can "
88 "create, modify and delete files under the export root");
89 }
90}

234/*
235 * Phase 35: bind the FRM durable stage queue + its SHM hot-index zone. A
236 * single queue is shared by every frm-enabled server block (Phase 0). The
237 * index zone-init callback reconciles file → index in the master before
238 * fork; workers open their own fds in init_process.
239 */
240static ngx_int_t
241postconf_stage_waiter(ngx_conf_t *cf, ngx_stream_core_main_conf_t *cmcf,
242 ngx_stream_core_srv_conf_t **cscfp)
243{
244 ngx_stream_brix_srv_conf_t *xcf;
245 ngx_uint_t i;
246 ngx_uint_t frm_peak = 0;
247 int any_stage = 0;
248
249 for (i = 0; i < cmcf->servers.nelts; i++) {
250 xcf = postconf_srv_conf(cscfp, i);
251 if (xcf->common.enable && xcf->frm.enable) {
252 any_stage = 1;
253 if (xcf->frm.max_inflight > frm_peak) {
254 frm_peak = xcf->frm.max_inflight;
255 }
256 }
257 }
258
259 /* Async-recall waiter SHM zone (clients parked on kXR_waitresp), sized to
260 * a couple× the in-flight bound; harmless when async is off. The legacy
261 * FRM durable queue + SHM index are retired — staging requests live in the
262 * composable registry (opened per-worker in init_process). */
263 if (any_stage
264 && brix_stage_waiter_configure(cf, frm_peak * 2 + 64) != NGX_OK)
265 {
266 return NGX_ERROR;
267 }
268
269 return NGX_OK;
270}

376/* Stream-module postconfiguration hook: validate config, build the runtime
377 * objects (TLS/PKI/SHM/CMS), wire handlers, and log the startup summary.
378 * Returns NGX_OK / NGX_ERROR. */
379ngx_int_t
380ngx_stream_brix_postconfiguration(ngx_conf_t *cf)
381{
382 ngx_stream_core_main_conf_t *cmcf;
383 ngx_stream_core_srv_conf_t **cscfp;
384 brix_phase_timer_t pt;
385
386 /* Master-side config-build cost breakdown (one NOTICE line at the end). */
387 brix_phase_timer_start(&pt);
388
389 cmcf = ngx_stream_conf_get_module_main_conf(cf, ngx_stream_core_module);
390 cscfp = cmcf->servers.elts;
391
392 /*
393 * Attempt to load libvomsapi.so.1 via dlopen. If the library is not
394 * present we continue; config validation below rejects brix_require_vo
395 * directives when VOMS is unavailable.
396 */
397 (void) brix_voms_init(cf->log);
398
399 if (postconf_prepare_servers(cf, cmcf, cscfp) != NGX_OK) {
400 return NGX_ERROR;
401 }
402 brix_phase_mark(&pt, "prepare"); /* server prep + GSI/TLS/token/sss/krb5 */
403
404 if (postconf_finalize_policies(cf, cmcf, cscfp) != NGX_OK) {
405 return NGX_ERROR;
406 }
407 brix_phase_mark(&pt, "policy");
408
409 if (brix_configure_metrics(cf, cmcf) != NGX_OK) {
410 return NGX_ERROR;
411 }
412
413 if (brix_configure_dashboard(cf) != NGX_OK) {
414 return NGX_ERROR;
415 }
416
417 if (postconf_shared_registries(cf, cmcf, cscfp) != NGX_OK) {
418 return NGX_ERROR;
419 }
420 brix_phase_mark(&pt, "registries"); /* metrics/dashboard/session/srv/tpc SHM */
421
422 if (postconf_stage_waiter(cf, cmcf, cscfp) != NGX_OK) {
423 return NGX_ERROR;
424 }
425
426 brix_phase_mark(&pt, "frm");
427
428 if (brix_configure_thread_pools(cf, cmcf) != NGX_OK) {
429 return NGX_ERROR;
430 }
431
432 /*
433 * Phase 44: io_uring fail-fast. `brix_io_uring on` is a hard requirement —
434 * if it cannot be satisfied (not compiled in, or the runtime probe fails on
435 * this host) refuse to start so the operator is not silently downgraded.
436 * The runtime probe is consulted here (per-process, seccomp-accurate), not
437 * at parse time. `off`/`auto` always pass.
438 */
439 if (brix_uring_validate_conf(cf) != NGX_OK) {
440 return NGX_ERROR;
441 }
442
443#if (BRIX_HAVE_LIBURING)
444 if (postconf_uring_killswitch(cf, cmcf, cscfp) != NGX_OK) {
445 return NGX_ERROR;
446 }
447#endif
448
449 if (postconf_impersonation(cf, cmcf, cscfp) != NGX_OK) {
450 return NGX_ERROR;
451 }
452
453 postconf_log_summaries(cf, cmcf, cscfp);
454
455 brix_phase_mark(&pt, "pools_uring");
456 brix_phase_timer_log(&pt, cf->log, "xrootd postconfig");
457
458 return NGX_OK;
459}
