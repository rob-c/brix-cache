1/*
2 * vfs_walk.c — VFS confined recursive walk (bulk / off-thread traversal).
3 *
4 * WHAT: Implements brix_vfs_walk(), a thread-safe, non-allocating, non-metered
5 * confined traversal of a rootfd-relative target. A regular-file target
6 * fires the caller's per-file callback once; a directory target recurses,
7 * firing the callback for each regular file it contains (to max_depth /
8 * max_files). Symlinks and special files are never followed or reported.
9 *
10 * WHY: Bulk consumers — kXR_Qckscan (checksum scan), recursive copy/remove —
11 * run on a thread-pool worker (or at startup) and enumerate whole subtrees.
12 * The metered, pool-allocating handle API (brix_vfs_open/opendir) is
13 * neither thread-safe (it allocates on the request pool) nor appropriate
14 * (it would emit one OP_DIRLIST/OP_OPEN per visited node). This entry point
15 * keeps the confined open/opendir/fstatat INSIDE the VFS (so callers never
16 * reach for a confined-helper directly) while staying allocation- and
17 * metric-free, and books nothing — the enclosing protocol op accounts for
18 * the whole walk.
19 *
20 * HOW: brix_vfs_walk() stats the target via the confined rootfd path
21 * (brix_stat_beneath): a regular file → open (when open_files) + one cb;
22 * a directory → vfs_walk_dir(). The recursion opens each directory with
23 * openat2 RESOLVE_BENEATH (brix_open_beneath O_DIRECTORY) + fdopendir,
24 * classifies each child by a dirfd-relative AT_SYMLINK_NOFOLLOW fstatat
25 * (immune to a concurrent rename), recurses into directories and opens +
26 * fires the callback for regular files. A per-entry open/stat failure
27 * SKIPS that entry (a raced unlink or an unreadable file does not abort the
28 * scan); only an opendir failure or a callback abort stops the walk. No
29 * goto (coding-standards): early-return + closedir on every exit.
30 */

40/*
41 * vfs_walk_ctx_t — the invariant traversal state threaded through the confined
42 * recursive walk.
43 *
44 * WHAT: Bundles every parameter that stays constant across the whole walk (the
45 * confinement root, the caller's callback + cookie, the filter options and
46 * the opendir-failure error sink) plus the single mutable counter shared by
47 * the recursion (nfiles). Only `depth` — which changes on every recursive
48 * descent — is passed as an explicit per-call argument.
49 * WHY: vfs_walk_dir()/vfs_walk_emit_file() each carried a 8–10 parameter list
50 * (log/rootfd/opts/cb/cookie/nfiles/errmsg/errsz), which is param-bloat
51 * (coding-standards §8) and forced every recursive call to re-thread the
52 * same eight values. Promoting a file-local context makes the data flow
53 * explicit (one pointer), keeps the helpers small and drops the signatures
54 * under the 5-param gate without touching the frozen extern brix_vfs_walk()
55 * contract.
56 * HOW: brix_vfs_walk() zero-initialises one ctx on its stack from its own
57 * arguments and passes &ctx (plus a starting depth) into vfs_walk_dir();
58 * the recursion carries the same &ctx unchanged and only advances depth.
59 * nfiles is a live counter inside the ctx (the file cap is process-wide for
60 * the walk), so it is written through the shared pointer, not copied.
61 */
62typedef struct {
63 ngx_log_t *log;
64 int rootfd;
65 const brix_vfs_walk_opts_t *opts;
66 brix_vfs_walk_file_cb cb;
67 void *cookie;
68 ngx_uint_t nfiles; /* regular files reached so far */
69 char *errmsg; /* opendir-failure sink (may be NULL) */
70 size_t errsz;
71} vfs_walk_ctx_t;

73/* Join a parent logical path + child name into out (export-relative). Special-
74 * cases parent=="/" so the root yields "/name", not "//name". Returns 0 (skip)
75 * on truncation. */
76static int
77vfs_walk_join(const char *parent, const char *name, char *out, size_t outsz)
78{
79 int n;
80
81 if (parent[0] == '/' && parent[1] == '\0') {
82 n = snprintf(out, outsz, "/%s", name);
83 } else {
84 n = snprintf(out, outsz, "%s/%s", parent, name);
85 }
86 return (n > 0 && (size_t) n < outsz);
87}

89/* Open one regular file (when requested), fire the callback, close the fd.
90 * Returns the callback's code, or NGX_OK when open_files and the open failed
91 * (a per-entry open failure is a soft skip during a directory walk). */
92static ngx_int_t
93vfs_walk_emit_file(const vfs_walk_ctx_t *ctx, const char *logical,
94 const struct stat *st, int soft_open_fail)
95{
96 brix_vfs_stat_t vst;
97 ngx_int_t rc;
98 int fd = -1;
99
100 if (ctx->opts->open_files) {
101 fd = brix_open_beneath(ctx->rootfd, logical, O_RDONLY, 0);
102 if (fd < 0) {
103 return soft_open_fail ? NGX_OK : NGX_ERROR;
104 }
105 }
106
107 brix_vfs_fill_stat(st, &vst);
108 rc = ctx->cb(ctx->cookie, logical, &vst, fd);
109
110 if (fd >= 0) {
111 (void) ngx_close_file(fd);
112 }
113 return rc;
114}

119/* Open one directory for a confined walk. On success returns an open DIR* (the
120 * caller closedir()s it); on failure returns NULL and records the opendir-error
121 * message in the ctx sink. RESOLVE_BENEATH O_DIRECTORY refuses to escape the
122 * root or open a non-directory. fdopendir adopts the fd on success (closedir
123 * releases it); we close it ourselves only on the fdopendir failure path to
124 * avoid a leak and a double-close. */
125static DIR *
126vfs_walk_opendir(const vfs_walk_ctx_t *ctx, const char *logical_dir)
127{
128 DIR *dh;
129 int dfd;
130
131 dfd = brix_open_beneath(ctx->rootfd, logical_dir, O_RDONLY | O_DIRECTORY, 0);
132 if (dfd < 0) {
133 if (ctx->errmsg != NULL && ctx->errsz > 0) {
134 snprintf(ctx->errmsg, ctx->errsz, "opendir failed: %s",
135 strerror(errno));
136 }
137 return NULL;
138 }
139
140 dh = fdopendir(dfd);
141 if (dh == NULL) {
142 (void) ngx_close_file(dfd);
143 if (ctx->errmsg != NULL && ctx->errsz > 0) {
144 snprintf(ctx->errmsg, ctx->errsz, "opendir failed: %s",
145 strerror(errno));
146 }
147 return NULL;
148 }
149 return dh;
150}

152/* Per-entry disposition returned by vfs_walk_entry(): keep scanning the
153 * directory, or stop the whole walk and propagate `code` to the caller. */
154typedef struct {
155 int stop; /* non-zero → abort the walk, returning `code` */
156 ngx_int_t code; /* the abort code (a callback result or NGX_ERROR) */
157} vfs_walk_step_t;
158
159/* Classify and handle one readdir entry. `dh` supplies the dirfd for a
160 * rename-immune fstatat; on a directory it recurses, on a regular file it counts
161 * against the file cap and fires the callback. Any per-entry open/stat failure is
162 * a soft skip (returns {0}); only a recursion/callback abort sets stop=1. */
163static vfs_walk_step_t
164vfs_walk_entry(vfs_walk_ctx_t *ctx, DIR *dh, const char *logical_dir,
165 const struct dirent *de, ngx_uint_t depth)
166{
167 vfs_walk_step_t step = { 0, NGX_OK };
168 char child[BRIX_MAX_PATH + 1];
169 struct stat st;
170 int dfd;
171
172 if (brix_fs_is_dot_entry(de->d_name)) {
173 return step;
174 }
175 if (!vfs_walk_join(logical_dir, de->d_name, child, sizeof(child))) {
176 return step; /* path too long — skip */
177 }
178
179 /* stat by (dirfd, name) — not full path — so we inspect the exact entry
180 * readdir returned, immune to a concurrent rename. NOFOLLOW: a symlink is
181 * neither S_ISDIR nor S_ISREG, so it is silently skipped (never traversed or
182 * reported). */
183 dfd = dirfd(dh);
184 if (dfd < 0 || fstatat(dfd, de->d_name, &st, AT_SYMLINK_NOFOLLOW) != 0) {
185 return step; /* unstattable (or dirfd failure) — skip the entry */
186 }
187
188 if (S_ISDIR(st.st_mode)) {
189 ngx_int_t rc = vfs_walk_dir(ctx, child, depth + 1);
190 if (rc != NGX_OK) {
191 step.stop = 1;
192 step.code = rc;
193 }
194 return step;
195 }
196
197 if (!S_ISREG(st.st_mode)) {
198 return step; /* symlink / special → skip */
199 }
200
201 /* Regular file: the file cap counts every file reached (before any open), so
202 * a skipped/unreadable file still consumes budget — bounding syscalls. */
203 if (ctx->opts->max_files != 0 && ctx->nfiles >= ctx->opts->max_files) {
204 return step;
205 }
206 ctx->nfiles++;
207
208 {
209 ngx_int_t rc = vfs_walk_emit_file(ctx, child, &st,
210 1 /* soft open-fail: skip this entry */);
211 if (rc != NGX_OK) {
212 step.stop = 1;
213 step.code = rc; /* callback abort */
214 }
215 }
216 return step;
217}

219/* Recursive directory body. Returns NGX_OK on success, NGX_ERROR on an opendir
220 * failure (errmsg set), or a callback abort code. */
221static ngx_int_t
222vfs_walk_dir(vfs_walk_ctx_t *ctx, const char *logical_dir, ngx_uint_t depth)
223{
224 DIR *dh;
225 struct dirent *de;
226
227 if (depth > ctx->opts->max_depth) {
228 return NGX_OK; /* depth cap: prune, not an error */
229 }
230
231 dh = vfs_walk_opendir(ctx, logical_dir);
232 if (dh == NULL) {
233 return NGX_ERROR;
234 }
235
236 for ( ;; ) {
237 vfs_walk_step_t step;
238
239 errno = 0;
240 de = readdir(dh);
241 if (de == NULL) {
242 break; /* end of stream (errno==0) or a readdir error — stop here */
243 }
244
245 step = vfs_walk_entry(ctx, dh, logical_dir, de, depth);
246 if (step.stop) {
247 (void) closedir(dh);
248 return step.code;
249 }
250 }
251
252 (void) closedir(dh);
253 return NGX_OK;
254}

256/* Walk a confined rootfd-relative target. See vfs.h for the full contract. */
257ngx_int_t
258brix_vfs_walk(ngx_log_t *log, int rootfd, const char *logical,
259 const brix_vfs_walk_opts_t *opts, brix_vfs_walk_file_cb cb, void *cookie,
260 brix_vfs_walk_target_t *target_out, char *errmsg, size_t errsz)
261{
262 struct stat st;
263 vfs_walk_ctx_t ctx = { log, rootfd, opts, cb, cookie, 0, errmsg, errsz };
264
265 if (target_out != NULL) {
266 *target_out = BRIX_VFS_WALK_NONE;
267 }
268 if (opts == NULL || cb == NULL || logical == NULL) {
269 errno = EINVAL;
270 return NGX_ERROR;
271 }
272
273 if (brix_stat_beneath(rootfd, logical, &st) != 0) {
274 return NGX_DECLINED; /* target not found */
275 }
276
277 if (S_ISREG(st.st_mode)) {
278 if (target_out != NULL) {
279 *target_out = BRIX_VFS_WALK_FILE;
280 }
281 /* Single-file target: an open failure here IS an error (not a skip). */
282 return vfs_walk_emit_file(&ctx, logical, &st, 0 /* hard open-fail */);
283 }
284
285 if (S_ISDIR(st.st_mode)) {
286 if (target_out != NULL) {
287 *target_out = BRIX_VFS_WALK_DIR;
288 }
289 return vfs_walk_dir(&ctx, logical, 0);
290 }
291
292 if (target_out != NULL) {
293 *target_out = BRIX_VFS_WALK_OTHER;
294 }
295 return NGX_OK; /* not a file or directory — caller inspects *target_out */
296}

352/*
353 * brix_vfs_backend_mkpath — recursively create `logical` (export-relative,
354 * leading-slash) and every missing parent through a NON-default backend driver's
355 * mkdir slot, tolerating EEXIST. The driver namespace is inherently confined to
356 * the export, so no rootfd anchoring is needed. Returns NGX_DECLINED for the
357 * default POSIX export (resolve == NULL) or a driver without a mkdir slot, so the
358 * caller falls back to its own confined/group-policy mkpath; 0 on success;
359 * -1 with errno set on a real failure.
360 */
361int
362brix_vfs_backend_mkpath(const char *root_canon, const char *logical,
363 mode_t mode, ngx_log_t *log)
364{
365 brix_sd_instance_t *sd = brix_vfs_backend_resolve(root_canon, log);
366 char acc[PATH_MAX];
367 size_t i = 0, j = 0;
368
369 if (sd == NULL || sd->driver == NULL || sd->driver->mkdir == NULL) {
370 /* Not a real failure — the caller falls back to its own confined/
371 * group-policy mkpath — but set errno for clarity to any caller that
372 * inspects it before checking the NGX_DECLINED return. */
373 errno = ENOSYS;
374 return NGX_DECLINED;
375 }
376
377 /* Create each prefix in turn: "/a/b/c" → "/a", "/a/b", "/a/b/c". */
378 while (logical[i] != '\0') {
379 if (logical[i] != '/') {
380 i++; /* defensive: skip a leading non-slash */
381 continue;
382 }
383 if (j + 1 >= sizeof(acc)) {
384 errno = ENAMETOOLONG;
385 return -1;
386 }
387 acc[j++] = '/';
388 i++;
389 while (logical[i] != '\0' && logical[i] != '/') {
390 if (j + 1 >= sizeof(acc)) {
391 errno = ENAMETOOLONG;
392 return -1;
393 }
394 acc[j++] = logical[i++];
395 }
396 acc[j] = '\0';
397 if (j > 1 /* skip the bare "/" root */
398 && sd->driver->mkdir(sd, acc, mode) != NGX_OK
399 && errno != EEXIST) {
400 return -1;
401 }
402 }
403 return 0;
404}
