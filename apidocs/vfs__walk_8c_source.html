1/*
2 * vfs_walk.c — VFS confined recursive walk (bulk / off-thread traversal).
3 *
4 * WHAT: Implements brix_vfs_walk(), a thread-safe, non-allocating, non-metered
5 * confined traversal of a rootfd-relative target. A regular-file target
6 * fires the caller's per-file callback once; a directory target recurses,
7 * firing the callback for each regular file it contains (to max_depth /
8 * max_files). Symlinks and special files are never followed or reported.
9 *
10 * WHY: Bulk consumers — kXR_Qckscan (checksum scan), recursive copy/remove —
11 * run on a thread-pool worker (or at startup) and enumerate whole subtrees.
12 * The metered, pool-allocating handle API (brix_vfs_open/opendir) is
13 * neither thread-safe (it allocates on the request pool) nor appropriate
14 * (it would emit one OP_DIRLIST/OP_OPEN per visited node). This entry point
15 * keeps the confined open/opendir/fstatat INSIDE the VFS (so callers never
16 * reach for a confined-helper directly) while staying allocation- and
17 * metric-free, and books nothing — the enclosing protocol op accounts for
18 * the whole walk.
19 *
20 * HOW: brix_vfs_walk() stats the target via the confined rootfd path
21 * (brix_stat_beneath): a regular file → open (when open_files) + one cb;
22 * a directory → vfs_walk_dir(). The recursion opens each directory with
23 * openat2 RESOLVE_BENEATH (brix_open_beneath O_DIRECTORY) + fdopendir,
24 * classifies each child by a dirfd-relative AT_SYMLINK_NOFOLLOW fstatat
25 * (immune to a concurrent rename), recurses into directories and opens +
26 * fires the callback for regular files. A per-entry open/stat failure
27 * SKIPS that entry (a raced unlink or an unreadable file does not abort the
28 * scan); only an opendir failure or a callback abort stops the walk. No
29 * goto (coding-standards): early-return + closedir on every exit.
30 */

42/*
43 * vfs_walk_ctx_t — the invariant traversal state threaded through the confined
44 * recursive walk.
45 *
46 * WHAT: Bundles every parameter that stays constant across the whole walk (the
47 * confinement root, the caller's callback + cookie, the filter options and
48 * the opendir-failure error sink) plus the single mutable counter shared by
49 * the recursion (nfiles). Only `depth` — which changes on every recursive
50 * descent — is passed as an explicit per-call argument.
51 * WHY: vfs_walk_dir()/vfs_walk_emit_file() each carried a 8–10 parameter list
52 * (log/rootfd/opts/cb/cookie/nfiles/errmsg/errsz), which is param-bloat
53 * (coding-standards §8) and forced every recursive call to re-thread the
54 * same eight values. Promoting a file-local context makes the data flow
55 * explicit (one pointer), keeps the helpers small and drops the signatures
56 * under the 5-param gate without touching the frozen extern brix_vfs_walk()
57 * contract.
58 * HOW: brix_vfs_walk() zero-initialises one ctx on its stack from its own
59 * arguments and passes &ctx (plus a starting depth) into vfs_walk_dir();
60 * the recursion carries the same &ctx unchanged and only advances depth.
61 * nfiles is a live counter inside the ctx (the file cap is process-wide for
62 * the walk), so it is written through the shared pointer, not copied.
63 */
64typedef struct {
65 ngx_log_t *log;
66 int rootfd;
67 const brix_vfs_walk_opts_t *opts;
68 brix_vfs_walk_file_cb cb;
69 void *cookie;
70 ngx_uint_t nfiles; /* regular files reached so far */
71 char *errmsg; /* opendir-failure sink (may be NULL) */
72 size_t errsz;
73} vfs_walk_ctx_t;

75/* Join a parent logical path + child name into out (export-relative). Special-
76 * cases parent=="/" so the root yields "/name", not "//name". Returns 0 (skip)
77 * on truncation. */
78static int
79vfs_walk_join(const char *parent, const char *name, char *out, size_t outsz)
80{
81 int n;
82
83 if (parent[0] == '/' && parent[1] == '\0') {
84 n = snprintf(out, outsz, "/%s", name);
85 } else {
86 n = snprintf(out, outsz, "%s/%s", parent, name);
87 }
88 return (n > 0 && (size_t) n < outsz);
89}

91/* Open one regular file (when requested), fire the callback, close the fd.
92 * Returns the callback's code, or NGX_OK when open_files and the open failed
93 * (a per-entry open failure is a soft skip during a directory walk). */
94static ngx_int_t
95vfs_walk_emit_file(const vfs_walk_ctx_t *ctx, const char *logical,
96 const struct stat *st, int soft_open_fail)
97{
98 brix_vfs_stat_t vst;
99 ngx_int_t rc;
100 int fd = -1;
101
102 if (ctx->opts->open_files) {
103 fd = brix_open_beneath(ctx->rootfd, logical, O_RDONLY, 0);
104 if (fd < 0) {
105 return soft_open_fail ? NGX_OK : NGX_ERROR;
106 }
107 }
108
109 brix_vfs_fill_stat(st, &vst);
110 rc = ctx->cb(ctx->cookie, logical, &vst, fd);
111
112 if (fd >= 0) {
113 (void) ngx_close_file(fd);
114 }
115 return rc;
116}

121/* Open one directory for a confined walk. On success returns an open DIR* (the
122 * caller closedir()s it); on failure returns NULL and records the opendir-error
123 * message in the ctx sink. RESOLVE_BENEATH O_DIRECTORY refuses to escape the
124 * root or open a non-directory. fdopendir adopts the fd on success (closedir
125 * releases it); we close it ourselves only on the fdopendir failure path to
126 * avoid a leak and a double-close. */
127static DIR *
128vfs_walk_opendir(const vfs_walk_ctx_t *ctx, const char *logical_dir)
129{
130 DIR *dh;
131 int dfd;
132
133 dfd = brix_open_beneath(ctx->rootfd, logical_dir, O_RDONLY | O_DIRECTORY, 0);
134 if (dfd < 0) {
135 if (ctx->errmsg != NULL && ctx->errsz > 0) {
136 snprintf(ctx->errmsg, ctx->errsz, "opendir failed: %s",
137 strerror(errno));
138 }
139 return NULL;
140 }
141
142 dh = fdopendir(dfd);
143 if (dh == NULL) {
144 (void) ngx_close_file(dfd);
145 if (ctx->errmsg != NULL && ctx->errsz > 0) {
146 snprintf(ctx->errmsg, ctx->errsz, "opendir failed: %s",
147 strerror(errno));
148 }
149 return NULL;
150 }
151 return dh;
152}

154/* Per-entry disposition returned by vfs_walk_entry(): keep scanning the
155 * directory, or stop the whole walk and propagate `code` to the caller. */
156typedef struct {
157 int stop; /* non-zero → abort the walk, returning `code` */
158 ngx_int_t code; /* the abort code (a callback result or NGX_ERROR) */
159} vfs_walk_step_t;
160
161/* Classify and handle one readdir entry. `dh` supplies the dirfd for a
162 * rename-immune fstatat; on a directory it recurses, on a regular file it counts
163 * against the file cap and fires the callback. Any per-entry open/stat failure is
164 * a soft skip (returns {0}); only a recursion/callback abort sets stop=1. */
165static vfs_walk_step_t
166vfs_walk_entry(vfs_walk_ctx_t *ctx, DIR *dh, const char *logical_dir,
167 const struct dirent *de, ngx_uint_t depth)
168{
169 vfs_walk_step_t step = { 0, NGX_OK };
170 char child[BRIX_MAX_PATH + 1];
171 struct stat st;
172 int dfd;
173
174 if (brix_fs_is_dot_entry(de->d_name)) {
175 return step;
176 }
177 if (!vfs_walk_join(logical_dir, de->d_name, child, sizeof(child))) {
178 return step; /* path too long — skip */
179 }
180
181 /* stat by (dirfd, name) — not full path — so we inspect the exact entry
182 * readdir returned, immune to a concurrent rename. NOFOLLOW: a symlink is
183 * neither S_ISDIR nor S_ISREG, so it is silently skipped (never traversed or
184 * reported). */
185 dfd = dirfd(dh);
186 if (dfd < 0 || fstatat(dfd, de->d_name, &st, AT_SYMLINK_NOFOLLOW) != 0) {
187 return step; /* unstattable (or dirfd failure) — skip the entry */
188 }
189
190 if (S_ISDIR(st.st_mode)) {
191 ngx_int_t rc = vfs_walk_dir(ctx, child, depth + 1);
192 if (rc != NGX_OK) {
193 step.stop = 1;
194 step.code = rc;
195 }
196 return step;
197 }
198
199 if (!S_ISREG(st.st_mode)) {
200 return step; /* symlink / special → skip */
201 }
202
203 /* Regular file: the file cap counts every file reached (before any open), so
204 * a skipped/unreadable file still consumes budget — bounding syscalls. */
205 if (ctx->opts->max_files != 0 && ctx->nfiles >= ctx->opts->max_files) {
206 return step;
207 }
208 ctx->nfiles++;
209
210 {
211 ngx_int_t rc = vfs_walk_emit_file(ctx, child, &st,
212 1 /* soft open-fail: skip this entry */);
213 if (rc != NGX_OK) {
214 step.stop = 1;
215 step.code = rc; /* callback abort */
216 }
217 }
218 return step;
219}

221/* Recursive directory body. Returns NGX_OK on success, NGX_ERROR on an opendir
222 * failure (errmsg set), or a callback abort code. */
223static ngx_int_t
224vfs_walk_dir(vfs_walk_ctx_t *ctx, const char *logical_dir, ngx_uint_t depth)
225{
226 DIR *dh;
227 struct dirent *de;
228
229 if (depth > ctx->opts->max_depth) {
230 return NGX_OK; /* depth cap: prune, not an error */
231 }
232
233 dh = vfs_walk_opendir(ctx, logical_dir);
234 if (dh == NULL) {
235 return NGX_ERROR;
236 }
237
238 for ( ;; ) {
239 vfs_walk_step_t step;
240
241 errno = 0;
242 de = readdir(dh);
243 if (de == NULL) {
244 break; /* end of stream (errno==0) or a readdir error — stop here */
245 }
246
247 step = vfs_walk_entry(ctx, dh, logical_dir, de, depth);
248 if (step.stop) {
249 (void) closedir(dh);
250 return step.code;
251 }
252 }
253
254 (void) closedir(dh);
255 return NGX_OK;
256}

258/* Walk a confined rootfd-relative target. See vfs.h for the full contract. */
259ngx_int_t
260brix_vfs_walk(ngx_log_t *log, int rootfd, const char *logical,
261 const brix_vfs_walk_opts_t *opts, brix_vfs_walk_file_cb cb, void *cookie,
262 brix_vfs_walk_target_t *target_out, char *errmsg, size_t errsz)
263{
264 struct stat st;
265 vfs_walk_ctx_t ctx = { log, rootfd, opts, cb, cookie, 0, errmsg, errsz };
266
267 if (target_out != NULL) {
268 *target_out = BRIX_VFS_WALK_NONE;
269 }
270 if (opts == NULL || cb == NULL || logical == NULL) {
271 errno = EINVAL;
272 return NGX_ERROR;
273 }
274
275 if (brix_stat_beneath(rootfd, logical, &st) != 0) {
276 return NGX_DECLINED; /* target not found */
277 }
278
279 if (S_ISREG(st.st_mode)) {
280 if (target_out != NULL) {
281 *target_out = BRIX_VFS_WALK_FILE;
282 }
283 /* Single-file target: an open failure here IS an error (not a skip). */
284 return vfs_walk_emit_file(&ctx, logical, &st, 0 /* hard open-fail */);
285 }
286
287 if (S_ISDIR(st.st_mode)) {
288 if (target_out != NULL) {
289 *target_out = BRIX_VFS_WALK_DIR;
290 }
291 return vfs_walk_dir(&ctx, logical, 0);
292 }
293
294 if (target_out != NULL) {
295 *target_out = BRIX_VFS_WALK_OTHER;
296 }
297 return NGX_OK; /* not a file or directory — caller inspects *target_out */
298}

343/*
344 * brix_vfs_backend_mkpath — recursively create `logical` (export-relative,
345 * leading-slash) and every missing parent through a NON-default backend driver's
346 * mkdir slot, tolerating EEXIST. The driver namespace is inherently confined to
347 * the export, so no rootfd anchoring is needed. Returns NGX_DECLINED for the
348 * default POSIX export (resolve == NULL) or a driver without a mkdir slot, so the
349 * caller falls back to its own confined/group-policy mkpath; 0 on success;
350 * -1 with errno set on a real failure.
351 */
352int
353brix_vfs_backend_mkpath(const char *root_canon, const char *logical,
354 mode_t mode, ngx_log_t *log)
355{
356 brix_sd_instance_t *sd = brix_vfs_backend_resolve(root_canon, log);
357 char acc[PATH_MAX];
358 size_t i = 0, j = 0;
359
360 if (sd == NULL || sd->driver == NULL || sd->driver->mkdir == NULL) {
361 /* Not a real failure — the caller falls back to its own confined/
362 * group-policy mkpath — but set errno for clarity to any caller that
363 * inspects it before checking the NGX_DECLINED return. */
364 errno = ENOSYS;
365 return NGX_DECLINED;
366 }
367
368 /* Create each prefix in turn: "/a/b/c" → "/a", "/a/b", "/a/b/c". */
369 while (logical[i] != '\0') {
370 if (logical[i] != '/') {
371 i++; /* defensive: skip a leading non-slash */
372 continue;
373 }
374 if (j + 1 >= sizeof(acc)) {
375 errno = ENAMETOOLONG;
376 return -1;
377 }
378 acc[j++] = '/';
379 i++;
380 while (logical[i] != '\0' && logical[i] != '/') {
381 if (j + 1 >= sizeof(acc)) {
382 errno = ENAMETOOLONG;
383 return -1;
384 }
385 acc[j++] = logical[i++];
386 }
387 acc[j] = '\0';
388 if (j > 1 /* skip the bare "/" root */
389 && sd->driver->mkdir(sd, acc, mode) != NGX_OK
390 && errno != EEXIST) {
391 return -1;
392 }
393 }
394 return 0;
395}

446/*
447 * vfs_copy_ctx_t — the invariant state of a confined copytree recursion.
448 *
449 * WHAT: Bundles every parameter that stays constant across a whole tree copy
450 * (the log, the confinement root, the preserve-xattrs flag and the
451 * per-entry metadata callback + cookie). Only the moving src/dst paths are
452 * passed as explicit per-node arguments.
453 * WHY: The per-entry / per-subdir copytree helpers otherwise each carried a
454 * 7–8 parameter list (log/root_canon/src/dst/preserve_xattrs/meta_cb/
455 * cookie), which is param-bloat (coding-standards §8). A file-local context
456 * makes the shared data flow one pointer and keeps the extracted helpers
457 * under the 5-param gate without disturbing the frozen extern
458 * brix_vfs_copytree() contract.
459 * HOW: brix_vfs_copytree() zero-initialises one ctx on its stack from its own
460 * arguments and passes &ctx (plus the moving src/dst) into the loop helper;
461 * the recursion re-enters brix_vfs_copytree(), which rebuilds an identical
462 * ctx for that subtree — so the context never outlives one frame.
463 */

466 const char *root_canon;
467 int preserve_xattrs;
468 brix_vfs_copy_meta_cb meta_cb;

470} vfs_copy_ctx_t;

472/* Recursively copy one confined child directory src_child→dst_child: mkdir the
473 * dst (tolerating EEXIST), copy fattrs + run meta_cb when requested, then recurse
474 * via brix_vfs_copytree. Factored out of the copytree loop to keep that loop's
475 * per-entry dispatch flat. Returns NGX_OK on success, NGX_ERROR on any step. */

483 {
484 return NGX_ERROR;

499/* Copy one readdir entry of a copytree walk. Builds the src/dst child paths,
500 * lstat-nofollow classifies the entry (a vanished/unreadable child is a soft
501 * skip), then dispatches to the subdir recursion or the file copy. Returns
502 * NGX_OK to continue the enclosing loop (including on a skip), NGX_ERROR to
503 * abort it. */
504static ngx_int_t
505vfs_copytree_entry(const vfs_copy_ctx_t *ctx, const char *src, const char *dst,
506 const struct dirent *de)
507{
508 char src_child[BRIX_MAX_PATH];
509 char dst_child[BRIX_MAX_PATH];
510 struct stat sb;
511
512 if (brix_fs_is_dot_entry(de->d_name)) {
513 return NGX_OK;
514 }
515 if ((size_t) snprintf(src_child, sizeof(src_child), "%s/%s",
516 src, de->d_name) >= sizeof(src_child)
517 || (size_t) snprintf(dst_child, sizeof(dst_child), "%s/%s",
518 dst, de->d_name) >= sizeof(dst_child))
519 {
520 return NGX_ERROR;
521 }
522
523 /* lstat nofollow: never resolve a symlink out of the export. */
524 if (brix_lstat_confined_canon(ctx->log, ctx->root_canon, src_child, &sb, 1)
525 != 0)
526 {
527 return NGX_OK; /* vanished/unreadable child — skip */
528 }
529
530 if (S_ISDIR(sb.st_mode)) {
531 return vfs_copytree_subdir(ctx, src_child, dst_child, sb.st_mode);
532 }
533 if (S_ISREG(sb.st_mode)) {
534 return brix_vfs_copyfile(ctx->log, ctx->root_canon, src_child, dst_child,
535 ctx->preserve_xattrs, ctx->meta_cb, ctx->cookie);
536 }
537 return NGX_OK; /* symlink / special → skip */
538}

540/* Recursively copy a confined directory tree src→dst. See vfs.h. */
541ngx_int_t
542brix_vfs_copytree(ngx_log_t *log, const char *root_canon, const char *src,
543 const char *dst, int preserve_xattrs, brix_vfs_copy_meta_cb meta_cb,
544 void *cookie)
545{
546 vfs_copy_ctx_t ctx = { log, root_canon, preserve_xattrs, meta_cb, cookie };
547 DIR *dp;
548 struct dirent *de;
549 ngx_int_t rc = NGX_OK;
550
551 /* Enumerate as the mapped user (a 0700 user-private collection would EACCES
552 * a bare worker opendir). */
553 dp = brix_opendir_confined_canon(log, root_canon, src);
554 if (dp == NULL) {
555 return NGX_ERROR;
556 }
557
558 while ((de = readdir(dp)) != NULL) {
559 rc = vfs_copytree_entry(&ctx, src, dst, de);
560 if (rc != NGX_OK) {
561 break;
562 }
563 }
564
565 (void) closedir(dp);
566 return rc;
