1/*
2 * vfs_walk.c — VFS confined recursive walk (bulk / off-thread traversal).
3 *
4 * WHAT: Implements brix_vfs_walk(), a thread-safe, non-allocating, non-metered
5 * confined traversal of a rootfd-relative target. A regular-file target
6 * fires the caller's per-file callback once; a directory target recurses,
7 * firing the callback for each regular file it contains (to max_depth /
8 * max_files). Symlinks and special files are never followed or reported.
9 *
10 * WHY: Bulk consumers — kXR_Qckscan (checksum scan), recursive copy/remove —
11 * run on a thread-pool worker (or at startup) and enumerate whole subtrees.
12 * The metered, pool-allocating handle API (brix_vfs_open/opendir) is
13 * neither thread-safe (it allocates on the request pool) nor appropriate
14 * (it would emit one OP_DIRLIST/OP_OPEN per visited node). This entry point
15 * keeps the confined open/opendir/fstatat INSIDE the VFS (so callers never
16 * reach for a confined-helper directly) while staying allocation- and
17 * metric-free, and books nothing — the enclosing protocol op accounts for
18 * the whole walk.
19 *
20 * HOW: brix_vfs_walk() stats the target via the confined rootfd path
21 * (brix_stat_beneath): a regular file → open (when open_files) + one cb;
22 * a directory → vfs_walk_dir(). The recursion opens each directory with
23 * openat2 RESOLVE_BENEATH (brix_open_beneath O_DIRECTORY) + fdopendir,
24 * classifies each child by a dirfd-relative AT_SYMLINK_NOFOLLOW fstatat
25 * (immune to a concurrent rename), recurses into directories and opens +
26 * fires the callback for regular files. A per-entry open/stat failure
27 * SKIPS that entry (a raced unlink or an unreadable file does not abort the
28 * scan); only an opendir failure or a callback abort stops the walk. No
29 * goto (coding-standards): early-return + closedir on every exit.
30 */

42/* Join a parent logical path + child name into out (export-relative). Special-
43 * cases parent=="/" so the root yields "/name", not "//name". Returns 0 (skip)
44 * on truncation. */
45static int
46vfs_walk_join(const char *parent, const char *name, char *out, size_t outsz)
47{
48 int n;
49
50 if (parent[0] == '/' && parent[1] == '\0') {
51 n = snprintf(out, outsz, "/%s", name);
52 } else {
53 n = snprintf(out, outsz, "%s/%s", parent, name);
54 }
55 return (n > 0 && (size_t) n < outsz);
56}

58/* Open one regular file (when requested), fire the callback, close the fd.
59 * Returns the callback's code, or NGX_OK when open_files and the open failed
60 * (a per-entry open failure is a soft skip during a directory walk). */
61static ngx_int_t
62vfs_walk_emit_file(ngx_log_t *log, int rootfd, const char *logical,
63 const struct stat *st, const brix_vfs_walk_opts_t *opts,
64 brix_vfs_walk_file_cb cb, void *cookie, int soft_open_fail)
65{
66 brix_vfs_stat_t vst;
67 ngx_int_t rc;
68 int fd = -1;
69
70 if (opts->open_files) {
71 fd = brix_open_beneath(rootfd, logical, O_RDONLY, 0);
72 if (fd < 0) {
73 return soft_open_fail ? NGX_OK : NGX_ERROR;
74 }
75 }
76
77 brix_vfs_fill_stat(st, &vst);
78 rc = cb(cookie, logical, &vst, fd);
79
80 if (fd >= 0) {
81 (void) ngx_close_file(fd);
82 }
83 return rc;
84}

86/* Recursive directory body. Returns NGX_OK on success, NGX_ERROR on an opendir
87 * failure (errmsg set), or a callback abort code. */
88static ngx_int_t
89vfs_walk_dir(ngx_log_t *log, int rootfd, const char *logical_dir,
90 const brix_vfs_walk_opts_t *opts, brix_vfs_walk_file_cb cb, void *cookie,
91 ngx_uint_t depth, ngx_uint_t *nfiles, char *errmsg, size_t errsz)
92{
93 DIR *dh;
94 struct dirent *de;
95 int dfd;
96
97 if (depth > opts->max_depth) {
98 return NGX_OK; /* depth cap: prune, not an error */
99 }
100
101 /* RESOLVE_BENEATH O_DIRECTORY: refuses to escape the root or open a non-dir. */
102 dfd = brix_open_beneath(rootfd, logical_dir, O_RDONLY | O_DIRECTORY, 0);
103 if (dfd < 0) {
104 if (errmsg != NULL && errsz > 0) {
105 snprintf(errmsg, errsz, "opendir failed: %s", strerror(errno));
106 }
107 return NGX_ERROR;
108 }
109
110 /* fdopendir adopts dfd on success (closedir releases it); close it ourselves
111 * only on the fdopendir failure path to avoid a leak and a double-close. */
112 dh = fdopendir(dfd);
113 if (dh == NULL) {
114 (void) ngx_close_file(dfd);
115 if (errmsg != NULL && errsz > 0) {
116 snprintf(errmsg, errsz, "opendir failed: %s", strerror(errno));
117 }
118 return NGX_ERROR;
119 }
120
121 for ( ;; ) {
122 char child[BRIX_MAX_PATH + 1];
123 struct stat st;
124 ngx_int_t rc;
125
126 errno = 0;
127 de = readdir(dh);
128 if (de == NULL) {
129 break; /* end of stream (errno==0) or a readdir error — stop here */
130 }
131
132 if (brix_fs_is_dot_entry(de->d_name)) {
133 continue;
134 }
135 if (!vfs_walk_join(logical_dir, de->d_name, child, sizeof(child))) {
136 continue; /* path too long — skip */
137 }
138
139 /* stat by (dirfd, name) — not full path — so we inspect the exact entry
140 * readdir returned, immune to a concurrent rename. NOFOLLOW: a symlink
141 * is neither S_ISDIR nor S_ISREG, so it is silently skipped (never
142 * traversed or reported). */
143 if (fstatat(dirfd(dh), de->d_name, &st, AT_SYMLINK_NOFOLLOW) != 0) {
144 continue;
145 }
146
147 if (S_ISDIR(st.st_mode)) {
148 rc = vfs_walk_dir(log, rootfd, child, opts, cb, cookie, depth + 1,
149 nfiles, errmsg, errsz);
150 if (rc != NGX_OK) {
151 (void) closedir(dh);
152 return rc;
153 }
154 continue;
155 }
156
157 if (!S_ISREG(st.st_mode)) {
158 continue; /* symlink / special → skip */
159 }
160
161 /* Regular file: file cap counts every file reached (before any open), so
162 * a skipped/unreadable file still consumes budget — bounding syscalls. */
163 if (opts->max_files != 0 && *nfiles >= opts->max_files) {
164 continue;
165 }
166 (*nfiles)++;
167
168 rc = vfs_walk_emit_file(log, rootfd, child, &st, opts, cb, cookie,
169 1 /* soft open-fail: skip this entry */);
170 if (rc != NGX_OK) {
171 (void) closedir(dh);
172 return rc; /* callback abort */
173 }
174 }
175
176 (void) closedir(dh);
177 return NGX_OK;
178}

180/* Walk a confined rootfd-relative target. See vfs.h for the full contract. */
181ngx_int_t
182brix_vfs_walk(ngx_log_t *log, int rootfd, const char *logical,
183 const brix_vfs_walk_opts_t *opts, brix_vfs_walk_file_cb cb, void *cookie,
184 brix_vfs_walk_target_t *target_out, char *errmsg, size_t errsz)
185{
186 struct stat st;
187 ngx_uint_t nfiles = 0;
188
189 if (target_out != NULL) {
190 *target_out = BRIX_VFS_WALK_NONE;
191 }
192 if (opts == NULL || cb == NULL || logical == NULL) {
193 errno = EINVAL;
194 return NGX_ERROR;
195 }
196
197 if (brix_stat_beneath(rootfd, logical, &st) != 0) {
198 return NGX_DECLINED; /* target not found */
199 }
200
201 if (S_ISREG(st.st_mode)) {
202 if (target_out != NULL) {
203 *target_out = BRIX_VFS_WALK_FILE;
204 }
205 /* Single-file target: an open failure here IS an error (not a skip). */
206 return vfs_walk_emit_file(log, rootfd, logical, &st, opts, cb, cookie,
207 0 /* hard open-fail */);
208 }
209
210 if (S_ISDIR(st.st_mode)) {
211 if (target_out != NULL) {
212 *target_out = BRIX_VFS_WALK_DIR;
213 }
214 return vfs_walk_dir(log, rootfd, logical, opts, cb, cookie, 0, &nfiles,
215 errmsg, errsz);
216 }
217
218 if (target_out != NULL) {
219 *target_out = BRIX_VFS_WALK_OTHER;
220 }
221 return NGX_OK; /* not a file or directory — caller inspects *target_out */
222}

267/*
268 * brix_vfs_backend_mkpath — recursively create `logical` (export-relative,
269 * leading-slash) and every missing parent through a NON-default backend driver's
270 * mkdir slot, tolerating EEXIST. The driver namespace is inherently confined to
271 * the export, so no rootfd anchoring is needed. Returns NGX_DECLINED for the
272 * default POSIX export (resolve == NULL) or a driver without a mkdir slot, so the
273 * caller falls back to its own confined/group-policy mkpath; 0 on success;
274 * -1 with errno set on a real failure.
275 */
276int
277brix_vfs_backend_mkpath(const char *root_canon, const char *logical,
278 mode_t mode, ngx_log_t *log)
279{
280 brix_sd_instance_t *sd = brix_vfs_backend_resolve(root_canon, log);
281 char acc[PATH_MAX];
282 size_t i = 0, j = 0;
283
284 if (sd == NULL || sd->driver == NULL || sd->driver->mkdir == NULL) {
285 return NGX_DECLINED;
286 }
287
288 /* Create each prefix in turn: "/a/b/c" → "/a", "/a/b", "/a/b/c". */
289 while (logical[i] != '\0') {
290 if (logical[i] != '/') {
291 i++; /* defensive: skip a leading non-slash */
292 continue;
293 }
294 if (j + 1 >= sizeof(acc)) {
295 errno = ENAMETOOLONG;
296 return -1;
297 }
298 acc[j++] = '/';
299 i++;
300 while (logical[i] != '\0' && logical[i] != '/') {
301 if (j + 1 >= sizeof(acc)) {
302 errno = ENAMETOOLONG;
303 return -1;
304 }
305 acc[j++] = logical[i++];
306 }
307 acc[j] = '\0';
308 if (j > 1 /* skip the bare "/" root */
309 && sd->driver->mkdir(sd, acc, mode) != NGX_OK
310 && errno != EEXIST) {
311 return -1;
312 }
313 }
314 return 0;
315}

317/* Copy a single confined regular file src→dst (impersonation-aware via the
318 * confined-canon helpers; thread-safe — no pool allocation). See vfs.h. */
319ngx_int_t
320brix_vfs_copyfile(ngx_log_t *log, const char *root_canon, const char *src,
321 const char *dst, int preserve_xattrs, brix_vfs_copy_meta_cb meta_cb,
322 void *cookie)
323{
324 struct stat sb;
325 int src_fd;
326 int dst_fd;
327 ngx_int_t rc;
328
329 /* stat as the mapped user (src may be a 0600 file under a 0700 parent the
330 * worker cannot itself traverse); follow semantics (copy the link target). */
331 if (brix_lstat_confined_canon(log, root_canon, src, &sb, 0) != 0) {
332 return NGX_ERROR;
333 }
334
335 src_fd = brix_open_confined_canon(log, root_canon, src,
336 O_RDONLY | O_CLOEXEC, 0);
337 if (src_fd < 0) {
338 return NGX_ERROR;
339 }
340
341 dst_fd = brix_open_confined_canon(log, root_canon, dst,
342 O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC,
343 sb.st_mode & 0777);
344 if (dst_fd < 0) {
345 (void) ngx_close_file(src_fd);
346 return NGX_ERROR;
347 }
348
349 rc = brix_copy_range(log, src_fd, 0, dst_fd, 0, (size_t) sb.st_size,
350 src, dst);
351
352 (void) ngx_close_file(src_fd);
353 (void) ngx_close_file(dst_fd);
354
355 if (rc == NGX_OK) {
356 if (preserve_xattrs) {
357 brix_ns_copy_fattrs(log, src, dst);
358 }
359 if (meta_cb != NULL) {
360 rc = meta_cb(cookie, src, dst, 0 /* is_dir */);
361 }
362 }
363 return rc;
364}

366/* Recursively copy a confined directory tree src→dst. See vfs.h. */
367ngx_int_t
368brix_vfs_copytree(ngx_log_t *log, const char *root_canon, const char *src,
369 const char *dst, int preserve_xattrs, brix_vfs_copy_meta_cb meta_cb,
370 void *cookie)
371{
372 DIR *dp;
373 struct dirent *de;
374 char src_child[BRIX_MAX_PATH];
375 char dst_child[BRIX_MAX_PATH];
376 struct stat sb;
377 ngx_int_t rc = NGX_OK;
378
379 /* Enumerate as the mapped user (a 0700 user-private collection would EACCES
380 * a bare worker opendir). */
381 dp = brix_opendir_confined_canon(log, root_canon, src);
382 if (dp == NULL) {
383 return NGX_ERROR;
384 }
385
386 while ((de = readdir(dp)) != NULL) {
387 if (brix_fs_is_dot_entry(de->d_name)) {
388 continue;
389 }
390
391 if ((size_t) snprintf(src_child, sizeof(src_child), "%s/%s",
392 src, de->d_name) >= sizeof(src_child)
393 || (size_t) snprintf(dst_child, sizeof(dst_child), "%s/%s",
394 dst, de->d_name) >= sizeof(dst_child))
395 {
396 rc = NGX_ERROR;
397 break;
398 }
399
400 /* lstat nofollow: never resolve a symlink out of the export. */
401 if (brix_lstat_confined_canon(log, root_canon, src_child, &sb, 1) != 0)
402 {
403 continue; /* vanished/unreadable child — skip */
404 }
405
406 if (S_ISDIR(sb.st_mode)) {
407 if (brix_mkdir_confined_canon(log, root_canon, dst_child,
408 sb.st_mode & 0777) != 0
409 && errno != EEXIST)
410 {
411 rc = NGX_ERROR;
412 break;
413 }
414 if (preserve_xattrs) {
415 brix_ns_copy_fattrs(log, src_child, dst_child);
416 }
417 if (meta_cb != NULL
418 && meta_cb(cookie, src_child, dst_child, 1 /* is_dir */)
419 != NGX_OK)
420 {
421 rc = NGX_ERROR;
422 break;
423 }
424 if (brix_vfs_copytree(log, root_canon, src_child, dst_child,
425 preserve_xattrs, meta_cb, cookie) != NGX_OK)
426 {
427 rc = NGX_ERROR;
428 break;
429 }
430 } else if (S_ISREG(sb.st_mode)) {
431 if (brix_vfs_copyfile(log, root_canon, src_child, dst_child,
432 preserve_xattrs, meta_cb, cookie) != NGX_OK)
433 {
434 rc = NGX_ERROR;
435 break;
436 }
437 }
438 }
439
440 (void) closedir(dp);
441 return rc;
442}
