8/*
9 * ftp_ev_dispatch.c — the command dispatcher and the small stateless verbs.
10 *
11 * WHAT: brix_ftp_ev_dispatch() frames a single command line into verb+arg,
12 * enforces the pre-auth gate, and routes to the right handler; the transfer-
13 * parameter and session verbs whose entire implementation is a single reply
14 * (TYPE/MODE/STRU/ALLO/PWD/CDUP/NOOP/SYST/OPTS/REST/ABOR/SITE/PBSZ/PROT/DCAU/
15 * USER/PASS/QUIT/FEAT) are handled inline here; the heavier namespace, security,
16 * and transfer verbs delegate to their own files.
17 *
18 * WHY: this mirrors the sync engine's ftp_dispatch() one-to-one so the two engines
19 * accept exactly the same grammar and emit the same reply codes — the invariant
20 * the transition depends on. Keeping the one-line verbs here (rather than one
21 * trivial function each) keeps the dispatcher readable without inflating the file
22 * count with near-empty handlers.
23 *
24 * HOW: each handler queues its reply via brix_ftp_ev_reply() and returns NGX_OK
25 * to continue, NGX_DONE on QUIT (flush then close), or NGX_ERROR on a fatal reply
26 * failure. The dispatcher never performs I/O itself — the engine flushes.
27 */
28

30/* Verbs accepted before the client has authenticated: the login verbs, the RFC
31 * 2228 security handshake (which sets ->authed once ADAT completes), and
32 * stateless capability probes. Everything else is gated behind login. */
33static int
34ev_verb_preauth(const char *verb)
35{
36 static const char *const ok[] = {
37 "USER", "PASS", "AUTH", "ADAT", "PBSZ", "PROT",
38 "MIC", "CONF", "ENC", "QUIT", "FEAT", "SYST",
39 "NOOP", "HELP", "OPTS", NULL
40 };
41 int i;
42
43 for (i = 0; ok[i] != NULL; i++) {
44 if (strcasecmp(verb, ok[i]) == 0) {
45 return 1;
46 }
47 }
48 return 0;
49}
