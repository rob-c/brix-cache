1/*
2 * sd_cache.c - the generic read-cache decorator (section 12.1). See header.
3 *
4 * The decorator forwards every write / namespace / xattr / dir / staged op to the
5 * wrapped `source` and interposes only the READ-open path: a COMPLETE cached
6 * object is served from the cache store (cstore), a miss is filled from the source
7 * into the store and recorded in a cinfo, and a write invalidates the cached copy.
8 * Served read objects are the store's own objects, so byte I/O bypasses the
9 * decorator. A sick cache degrades to a source read - it never fails a read or
10 * serves wrong bytes (section 16).
11 *
12 * This file owns the interposed read-open decision tree, the driver vtable, the
13 * create/destroy lifecycle, and the async-fill offload seam. The pieces it drives
14 * live in three siblings after the phase-79 size split (all reached through
15 * sd_cache_internal.h): the whole-file fill spine (sd_cache_fill.c), the
16 * slice/partial machinery + partial byte slots (sd_cache_partial.c), and the
17 * namespace/xattr/dir/staged-write forwarders (sd_cache_forward.c).
18 */

46/* Serve a COMPLETE cached object for `path`, if one exists.
47 *
48 * WHAT: The authoritative-hit fast path — a COMPLETE cinfo serves straight
49 * from the cache store (freshness is checked at fill time via verify,
50 * not on every read - section 6.4).
51 *
52 * WHY: Splitting the hit decision from the miss machinery keeps the open
53 * decision tree flat: the caller falls through to the miss path on any
54 * NULL (no cinfo, not COMPLETE, or the object vanished under us).
55 *
56 * HOW: cinfo load + COMPLETE check, then brix_cstore_serve_open. Presents the
57 * ORIGIN perms recorded in the cinfo, not the physical store object's
58 * bits (which are forced owner-writable so the cinfo xattr can be
59 * maintained). A cached object is a regular file; 0 = pre-mode cinfo →
60 * keep the store bits (snap left untouched). */
61static brix_sd_obj_t *
62cache_open_serve_hit(sd_cache_inst_state *st, const char *path, int *err_out)
63{
64 brix_cache_cinfo_t ci;
65 brix_sd_obj_t *obj;
66
67 if (brix_cstore_cinfo_load(&st->cstore, path, &ci) != NGX_OK
68 || !(ci.flags & BRIX_CINFO_F_COMPLETE))
69 {
70 return NULL;
71 }
72 obj = brix_cstore_serve_open(&st->cstore, path, err_out);
73 if (obj == NULL) {
74 return NULL; /* the cached object vanished under us - refill */
75 }
76 if (ci.mode != 0) {
77 obj->snap.mode = (mode_t) S_IFREG | (mode_t) (ci.mode & 0777);
78 }
79 ngx_log_debug1(NGX_LOG_DEBUG_CORE, st->log, 0,
80 "sd_cache: hit \"%s\"", path);
81 return obj;
82}

84/* Park the open on an in-flight nearline (tape) recall.
85 *
86 * WHAT: For a NEARLINE source, kicks the async recall for `path` and reports
87 * whether the open should park (fail soft with EAGAIN).
88 *
89 * WHY: A nearline miss is an async recall the open parks on (section 9.2).
90 * The waiter that parks/wakes the open lands with the frm driver
91 * (SP4/SP5); until then a NEARLINE source cannot be served here, so
92 * fail soft with EAGAIN rather than block.
93 *
94 * HOW: Returns 1 (recall in flight — *err_out = EAGAIN; the HTTP plane
95 * answers 202 "staging" + Retry-After; a retry re-polls the recall and,
96 * once the MSS brings the object online, takes the normal miss-fill and
97 * serves, SP5 §9.2) or 0 (not nearline, or already online — proceed to
98 * a normal fill). */
99static int
100cache_open_recall_parked(sd_cache_inst_state *st, brix_sd_instance_t *src,
101 const char *path, int *err_out)
102{
103 char reqid[40];
104 ngx_int_t rr;
105
106 if ((brix_sd_caps(src) & BRIX_SD_CAP_NEARLINE) == 0
107 || src->driver->recall == NULL)
108 {
109 return 0;
110 }
111 rr = src->driver->recall(src, path, reqid);
112 if (rr != NGX_AGAIN) {
113 return 0; /* NGX_OK: already online - a normal fill follows */
114 }
115 ngx_log_debug1(NGX_LOG_DEBUG_CORE, st->log, 0,
116 "sd_cache: nearline recall of \"%s\" in flight (staging)", path);
117 if (err_out != NULL) {
118 *err_out = EAGAIN;
119 }
120 return 1;
121}

123/* Serve a cache MISS: partial-serve when slice mode applies, else fill+serve.
124 *
125 * WHAT: The miss half of the read-open decision — build a partial (on-demand
126 * block-fill) object in slice mode, or run the whole-file fill and serve
127 * the cached copy.
128 *
129 * WHY: A miss or a PARTIAL cinfo both take the slice path (a COMPLETE object
130 * was already served by the hit helper); a non-default slice_size on a
131 * LOCAL cache store fills on demand (section 6.5) instead of pulling the
132 * whole object.
133 *
134 * HOW: Each strategy falls through to the next on failure; NULL means the
135 * caller degrades to a plain source read (a sick cache never fails a
136 * read, section 16). */
137static brix_sd_obj_t *
138cache_open_miss_serve(brix_sd_instance_t *inst, sd_cache_inst_state *st,
139 const char *path, const brix_sd_cred_t *cred, int *err_out)
140{
141 brix_sd_obj_t *obj;
142
143 if (st->policy.slice_size > 0
144 && st->cstore.meta_mode == BRIX_CMETA_LOCAL)
145 {
146 obj = sd_cache_partial_open(inst, st, path, cred, err_out);
147 if (obj != NULL) {
148 ngx_log_debug1(NGX_LOG_DEBUG_CORE, st->log, 0,
149 "sd_cache: partial-serve \"%s\"", path);
150 return obj;
151 }
152 /* partial open failed - fall through to a whole-file fill / source read */
153 }
154
155 if (sd_cache_fill(st, path, cred) == NGX_OK) {
156 obj = brix_cstore_serve_open(&st->cstore, path, err_out);
157 if (obj != NULL) {
158 ngx_log_debug1(NGX_LOG_DEBUG_CORE, st->log, 0,
159 "sd_cache: filled \"%s\"", path);
160 return obj;
161 }
162 }
163 return NULL;
164}

166/* Common open implementation shared by sd_cache_open (no cred) and
167 * sd_cache_open_cred (with cred).
168 *
169 * WHAT: The interposed read-open decision tree for the cache decorator.
170 *
171 * WHY: Extracting the body into a common helper avoids duplicating the entire
172 * open logic across two vtable slots (plain and cred-scoped).
173 *
174 * HOW: Write/create/trunc → passthrough + evict. A COMPLETE hit → serve from
175 * the store (cache_open_serve_hit). A miss → partial-serve or fill+serve
176 * (cache_open_miss_serve, cred threaded through to the source open).
177 * Failed or declined → degrade to the source. rq->cred may be NULL
178 * (service-credential path). */
179static brix_sd_obj_t *
180sd_cache_open_common(brix_sd_instance_t *inst, const char *path,
181 const sd_cache_open_req_t *rq, int *err_out)
182{
183 sd_cache_inst_state *st = SD_CACHE_ST(inst);
184 brix_sd_instance_t *src = st->source;
185 brix_sd_obj_t *obj;
186
187 /* A composed source always defines .open (brix_tier_build never yields a
188 * driver without it), but guard uniformly with the fill/partial-open paths
189 * (sd_cache_fill, sd_cache_partial_open) rather than relying on that alone. */
190 if (src->driver->open == NULL) {
191 if (err_out != NULL) { *err_out = ENOSYS; }
192 errno = ENOSYS;
193 return NULL;
194 }
195
196 /* WRITE / CREATE / TRUNC: pass through and invalidate the cached copy. */
197 if (rq->sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC)) {
198 obj = brix_sd_open_maybe_cred(src, path, rq->sd_flags, rq->mode,
199 rq->cred, err_out);
200 if (obj != NULL) {
201 (void) brix_cstore_evict(&st->cstore, path);
202 }
203 return obj;
204 }
205
206 /* READ: a COMPLETE cinfo is an authoritative hit. */
207 obj = cache_open_serve_hit(st, path, err_out);
208 if (obj != NULL) {
209 return obj;
210 }
211
212 /* Path-filtered out: serve straight from the source, never cache. */
213 if (!sd_cache_admit(&st->policy, path, -1)) {
214 return brix_sd_open_maybe_cred(src, path, rq->sd_flags, rq->mode,
215 rq->cred, err_out);
216 }
217
218 /* Nearline (tape) source: park the open on an in-flight recall (§9.2). */
219 if (cache_open_recall_parked(st, src, path, err_out)) {
220 return NULL;
221 }
222
223 /* MISS: partial-serve (slice mode) or fill from the source + serve. */
224 obj = cache_open_miss_serve(inst, st, path, rq->cred, err_out);
225 if (obj != NULL) {
226 return obj;
227 }
228
229 /* Declined or failed: serve from the source (a sick cache never fails a read,
230 * section 16). */
231 return brix_sd_open_maybe_cred(src, path, rq->sd_flags, rq->mode,
232 rq->cred, err_out);
233}

337 inst->log = log;
338 inst->pool = NULL;

359/* ---- async-fill seam (SP2 "shell -> full") --------------------------------
360 * The decorator's open() runs the miss-fill INLINE - correct on a worker thread
361 * (the stream fill task, a WebDAV/S3 PUT worker) but a stall on the event loop
362 * when the fill reads a remote source or writes a remote store (a socket wire
363 * client cannot do blocking I/O on the un-pumped loop; an in-process store just
364 * freezes the worker for the transfer). The HTTP read plane therefore probes
365 * whether an inline open would block, runs the fill on the thread pool, and
366 * re-enters. These three entrypoints expose exactly that - without making the
367 * SD open() contract asynchronous. See src/shared/http_cache_fill.c. */
368
369int
370brix_sd_cache_instance_is(const brix_sd_instance_t *inst)
371{
372 return (inst != NULL && inst->driver == &brix_sd_cache_driver) ? 1 : 0;
373}

393/* Would a read-open of `key` block the calling thread on slow (remote) I/O? 1
394 * only for a cache MISS whose whole-file fill would touch a non-local tier - the
395 * source exposes no local fd (a remote read: xroot/http/s3/ceph) or the cache
396 * store is not a local POSIX dir (a remote write: e.g. a rados store). A COMPLETE
397 * hit, a slice-mode object (open returns without filling), a local->local copy,
398 * or a non-cache instance all return 0 (serve inline). No blocking call - the
399 * cinfo probe hits the per-worker L1 / a local sidecar. */
400int
401brix_sd_cache_fill_needs_offload(brix_sd_instance_t *inst, const char *key)
402{
403 sd_cache_inst_state *st;
404 brix_cache_cinfo_t ci;
405 int src_slow;
406 int store_slow;
407
408 if (!brix_sd_cache_instance_is(inst) || key == NULL) {
409 return 0;
410 }
411 st = SD_CACHE_ST(inst);
412
413 /* A COMPLETE cached object is served from the store with no fill —
414 * unless its phase-68 TTL has passed (an expired manifest refills; the
415 * failed-refill path serves it stale within the 10x-TTL bound). */
416 if (brix_cstore_cinfo_load(&st->cstore, key, &ci) == NGX_OK
417 && (ci.flags & BRIX_CINFO_F_COMPLETE))
418 {
419 if (!(st->policy.cvmfs_manifest_ttl > 0
420 && brix_cache_cinfo_expired(&ci, time(NULL)) == 1))
421 {
422 return 0;
423 }
424 /* expired: fall through to the miss logic (refill if a slow tier) */
425 }
426 /* Slice/partial mode (LOCAL store): open() returns a partial object without a
427 * whole-file fill, so the open call itself does not block. */
428 if (st->policy.slice_size > 0
429 && st->cstore.meta_mode == BRIX_CMETA_LOCAL)
430 {
431 return 0;
432 }
433 /* A miss: the inline open would run the whole-file fill. Offload iff a slow
434 * tier is involved - a remote source read or a remote store write. */
435 src_slow = (brix_sd_caps(st->source) & BRIX_SD_CAP_FD) == 0;
436 store_slow = (st->cstore.meta_mode != BRIX_CMETA_LOCAL);
437 return (src_slow || store_slow) ? 1 : 0;
438}
