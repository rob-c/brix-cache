1/*
2 * sd_cache.c - the generic read-cache decorator (section 12.1). See header.
3 *
4 * The decorator forwards every write / namespace / xattr / dir / staged op to the
5 * wrapped `source` and interposes only the READ-open path: a COMPLETE cached
6 * object is served from the cache store (cstore), a miss is filled from the source
7 * into the store and recorded in a cinfo, and a write invalidates the cached copy.
8 * Served read objects are the store's own objects, so byte I/O bypasses the
9 * decorator. A sick cache degrades to a source read - it never fails a read or
10 * serves wrong bytes (section 16).
11 */
12#include "sd_cache.h"
13#include "sd_cache_internal.h" /* sd_cache_inst_state + SD_CACHE_ST/SRC */
14#include "sd_cache_policy.h" /* admission + repo-metrics (split out) */
15#include "protocols/cvmfs/classify.h" /* phase-68 manifest-TTL stamping */
16#include "observability/metrics/metrics.h" /* phase-68 T16 counters */
17#include "observability/metrics/metrics_macros.h"
18#include "fs/cache/cstore.h"
19#include "fs/backend/http/sd_http.h" /* per-upstream fill attribution */

33/* ---- the fill spine (source -> cache store) ------------------------------- */
34
35/* Fill `key` from the source into the cache store and record its cinfo. Returns
36 * NGX_OK (object cached + cinfo stored), NGX_DECLINED (policy declined to cache it
37 * - serve from source), or NGX_ERROR (a fill failure - serve from source). */
38static ngx_int_t
39sd_cache_fill(sd_cache_inst_state *st, const char *key)
40{
41 brix_sd_instance_t *src = st->source;
42 brix_sd_obj_t *so;
43 brix_sd_staged_t *staged;
44 brix_sd_stat_t snap;
45 u_char *buf;
46 off_t off = 0;
47 int err = 0;
48 int verified;
49 mode_t fmode;
50 brix_cache_cinfo_t ci;
51 struct timespec fill_t0; /* T16: per-upstream fill duration */
52
53 (void) clock_gettime(CLOCK_MONOTONIC, &fill_t0);
54
55 if (src->driver->open == NULL) {
56 errno = ENOSYS;
57 return NGX_ERROR;
58 }
59 so = src->driver->open(src, key, BRIX_SD_O_READ, 0, &err);
60 if (so == NULL) {
61 if (sd_cache_stale_serve_ok(st, key)) {
62 return NGX_OK; /* bounded stale-if-error (phase-68) */
63 }
64 errno = err ? err : EIO;
65 return NGX_ERROR;
66 }
67 /* Every op on the OPEN OBJECT dispatches through so->driver, NOT src->driver:
68 * a decorator source (sd_stage) legitimately returns the tier-below's object
69 * from a read open ("read byte-I/O bypasses the decorator"), so the object's
70 * vtable can differ from the instance's — dispatching object ops through the
71 * instance vtable reinterprets a foreign object state (type confusion). */
72 if (so->driver->pread == NULL) {
73 brix_sd_obj_release(so);
74 errno = ENOSYS;
75 return NGX_ERROR;
76 }
77 /* open() does not guarantee a populated snap (the posix driver fstats lazily),
78 * so fstat the object for an accurate size/mtime/mode - the cinfo validity and
79 * the cached file's permission bits both depend on it. */
80 snap = so->snap;
81 if (so->driver->fstat != NULL) {
82 (void) so->driver->fstat(so, &snap);
83 }
84 /* SECURITY + correctness: the physical cache-store object is a svc-owned
85 * artifact that aggregates MANY users' bytes under one tree. Per-user
86 * authorization is enforced at the protocol gate (open_cache.c), and the
87 * CLIENT-FACING mode is carried in the cinfo record and served by
88 * sd_cache_stat() — decoupled from this physical mode. So force 0600:
89 * - owner rw is REQUIRED (XATTR meta_mode stores user.xrd.cinfo ON this
90 * object; Linux refuses user.* xattrs on a non-writable inode — a
91 * read-only 0444 source would otherwise block cinfo persistence, G3);
92 * - NO group/other bits, so a mapped low-priv uid cannot read another
93 * user's cached bytes by direct filesystem access (the source mode's
94 * 0644 previously leaked here). snap.mode still reaches clients via the
95 * cinfo record (ci.mode below, served by sd_cache_stat). */
96 fmode = S_IRUSR | S_IWUSR; /* 0600 — svc-owned cache artifact */
97
98 if (!sd_cache_admit(&st->policy, key, snap.size)) {
99 brix_sd_obj_release(so);
100 return NGX_DECLINED; /* too big / filtered - do not cache */
101 }
102
103 staged = brix_cstore_fill_open(&st->cstore, key, fmode);
104 if (staged == NULL) {
105 ngx_log_error(NGX_LOG_WARN, st->log, errno,
106 "sd_cache: fill_open on the cache store failed for \"%s\" - not cached",
107 key);
108 brix_sd_obj_release(so);
109 return NGX_ERROR;
110 }
111 buf = malloc(SD_CACHE_CHUNK);
112 if (buf == NULL) {
113 brix_cstore_fill_abort(staged);
114 brix_sd_obj_release(so);
115 errno = ENOMEM;
116 return NGX_ERROR;
117 }
118
119 for ( ;; ) {
120 ssize_t r = so->driver->pread(so, buf, SD_CACHE_CHUNK, off);
121
122 if (r < 0) {
123 int read_err = errno; /* capture BEFORE any cleanup call */
124
125 if (read_err == EINTR) {
126 continue;
127 }
128 free(buf);
129 brix_cstore_fill_abort(staged);
130 brix_sd_obj_release(so);
131 if (sd_cache_stale_serve_ok(st, key)) {
132 return NGX_OK; /* bounded stale-if-error (phase-68) */
133 }
134 /* Restore the READ's errno. sd_cache_stale_serve_ok() (and the
135 * cleanup calls) stat the absent cache entry and leave errno
136 * ENOENT — propagating that would make the fill layer classify a
137 * merely-broken origin connection (EIO/ETIMEDOUT) as the origin's
138 * definitive 404, turning a transient network fault into a
139 * cached, client-poisoning "not found" instead of a retry. */
140 errno = read_err;
141 return NGX_ERROR;
142 }
143 if (r == 0) {
144 break;
145 }
146 if (brix_cstore_fill_write(staged, buf, (size_t) r, off) < 0) {
147 free(buf);
148 brix_cstore_fill_abort(staged);
149 brix_sd_obj_release(so);
150 return NGX_ERROR;
151 }
152 off += r;
153 if (snap.size > 0 && off >= snap.size) {
154 break; /* size known: skip the EOF-probe round-trip */
155 }
156 }
157 free(buf);
158 brix_sd_obj_release(so);
159
160 /* phase-68: digest-verify the staged bytes BEFORE the commit publishes
161 * them (cvmfs-cas: the key names its own sha1 — no origin digest). A
162 * MISMATCH is quarantined for evidence and the fill fails (the client
163 * sees a gateway error and retries); an ERROR fails closed. Non-CAS keys
164 * (manifests) come back UNVERIFIED and commit as before. */
165 verified = 0;
166 if (st->policy.verify == BRIX_CACHE_VERIFY_CVMFS_CAS) {
167 const char *pp = (staged->inst->driver->staged_path != NULL)
168 ? staged->inst->driver->staged_path(staged) : NULL;
169 brix_cache_verify_result_e vr;
170
171 vr = (pp != NULL)
172 ? brix_cache_verify_cvmfs_cas(pp, key, st->log, NULL, NULL)
173 : BRIX_CACHE_VERIFY_ERROR;
174 if (vr == BRIX_CACHE_VERIFY_MISMATCH) {
175 ngx_brix_cvmfs_repo_metrics_t *rm = sd_cache_repo_metrics(st,
176 key);
177
178 BRIX_CVMFS_METRIC_INC(verify_failures_total);
179 if (rm != NULL) {
180 BRIX_ATOMIC_INC(&rm->verify_failures_total);
181 }
182 sd_cache_note_origin_bytes(st, key, off); /* WAN cost is WAN cost */
183 /* a verify mismatch IS an upstream fill failure — the bytes came
184 * from the origin but did not publish. */
185 sd_cache_note_upstream(st, 0, off, sd_cache_ms_since(&fill_t0));
186 brix_cache_quarantine_part(pp, st->policy.quarantine_dir,
187 st->log);
188 brix_cstore_fill_abort(staged); /* part already renamed away */
189 errno = EBADMSG; /* digest mismatch — T20 budgets retries */
190 return NGX_ERROR;
191 }
192 if (vr == BRIX_CACHE_VERIFY_ERROR) {
193 ngx_log_error(NGX_LOG_ERR, st->log, errno,
194 "sd_cache: cvmfs-cas verify could not run for \"%s\" - "
195 "failing the fill closed", key);
196 brix_cstore_fill_abort(staged);
197 errno = EIO;
198 return NGX_ERROR;
199 }
200 verified = (vr == BRIX_CACHE_VERIFY_VERIFIED);
201 }
202
203 sd_cache_note_origin_bytes(st, key, off); /* WAN in, this attempt */
204 sd_cache_note_upstream(st, 1, off, sd_cache_ms_since(&fill_t0));
205
206 if (brix_cstore_fill_commit(staged) != NGX_OK) {
207 return NGX_ERROR;
208 }
209
210 /* Whole-file COMPLETE cinfo (the 1 MiB granule keys validity; the present
211 * bitmap is all-set). A partial/slice fill is section 6.5 / SP2. */
212 ngx_memzero(&ci, sizeof(ci));
213 ci.magic = BRIX_CACHE_CINFO_MAGIC;
214 ci.version = BRIX_CACHE_CINFO_VERSION;
215 ci.block_size = BRIX_CACHE_DIRTY_BLOCK;
216 ci.size = (uint64_t) off;
217 ci.mtime = (uint64_t) snap.mtime;
218 ci.mode = (uint32_t) (snap.mode & 0777); /* origin perms — served back
219 * so a read-only source is not
220 * masked by the owner-writable
221 * physical store object. */
222 ci.nblocks = brix_cache_cinfo_nblocks((uint64_t) off, ci.block_size);
223 ci.flags = BRIX_CINFO_F_COMPLETE
224 | (verified ? BRIX_CINFO_F_VERIFIED : 0);
225 ci.filled_at = (uint64_t) time(NULL);
226 if (st->policy.cvmfs_manifest_ttl > 0 && sd_cache_is_manifest_key(key)) {
227 brix_cache_cinfo_set_expires(&ci,
228 (time_t) ci.filled_at + st->policy.cvmfs_manifest_ttl);
229 }
230
231 if (brix_cstore_cinfo_store(&st->cstore, key, &ci) != NGX_OK) {
232 /* The object is cached but unrecorded - a safe miss (refill) next time. */
233 ngx_log_error(NGX_LOG_WARN, st->log, errno,
234 "sd_cache: cinfo store failed for \"%s\" - will refill on next read",
235 key);
236 }
237 return NGX_OK;
238}

240/* ---- slice/partial caching (section 6.5) ----------------------------------
241 * When policy.slice_size > 0 on a LOCAL posix cache store, an object is cached in
242 * fixed-size BLOCKS filled on demand: a read range-fills only the blocks it
243 * touches (source pread -> cache pwrite -> cinfo present-bit), so a Range request
244 * never pulls the whole object. The served object carries this state and ranges-
245 * fill on pread; it is never sendfiled (it may have holes). A fully-filled object's
246 * cinfo becomes COMPLETE and subsequent opens take the whole-file hit fast path. */
247
248typedef struct {
249 brix_sd_instance_t *source;
250 brix_sd_obj_t *src_obj; /* lazily opened on the first miss */
251 int cache_fd; /* the RW (sparse) cache object */
252 off_t size;
253 uint32_t block_size;
254 uint32_t mode; /* origin perm bits recorded in cinfo */
255 uint64_t mtime;
256 uint64_t nblocks;
257 uint8_t *bitmap; /* present blocks (in-memory mirror) */
258 size_t bitmap_len;
259 ngx_log_t *log;
260 char key[1024];
261 char cache_path[PATH_MAX]; /* for cinfo record_block */
262} sd_cache_partial_t;
263
264/* Fetch block `blk` from the source into the cache object + mark it present. */
265static int
266sd_cache_fill_block(sd_cache_partial_t *p, uint64_t blk)
267{
268 off_t bstart = (off_t) blk * p->block_size;
269 size_t blen;
270 u_char *bbuf;
271 off_t got = 0;
272 int e = 0;
273
274 if (bstart >= p->size) {
275 return 0;
276 }
277 blen = (size_t) ((bstart + (off_t) p->block_size <= p->size)
278 ? p->block_size : (p->size - bstart));
279
280 if (p->src_obj == NULL) {
281 if (p->source->driver->open == NULL) {
282 errno = ENOSYS;
283 return -1;
284 }
285 p->src_obj = p->source->driver->open(p->source, p->key, BRIX_SD_O_READ,
286 0, &e);
287 if (p->src_obj == NULL) {
288 errno = e ? e : EIO;
289 return -1;
290 }
291 /* Object ops dispatch through the OBJECT's driver (a decorator source
292 * returns the tier-below's object from a read open — see sd_cache_fill). */
293 if (p->src_obj->driver->pread == NULL) {
294 errno = ENOSYS;
295 return -1;
296 }
297 }
298 bbuf = malloc(blen);
299 if (bbuf == NULL) {
300 errno = ENOMEM;
301 return -1;
302 }
303 while ((size_t) got < blen) {
304 ssize_t r = p->src_obj->driver->pread(p->src_obj, bbuf + got,
305 blen - (size_t) got, bstart + got);
306 if (r < 0) {
307 if (errno == EINTR) { continue; }
308 free(bbuf);
309 return -1;
310 }
311 if (r == 0) {
312 break; /* short source vs stat - stop */
313 }
314 got += r;
315 }
316 {
317 off_t w = 0;
318
319 while (w < got) {
320 ssize_t n = pwrite(p->cache_fd, bbuf + w, (size_t) (got - w),
321 bstart + w);
322 if (n < 0) {
323 if (errno == EINTR) { continue; }
324 free(bbuf);
325 return -1;
326 }
327 w += n;
328 }
329 }
330 free(bbuf);
331
332 (void) brix_cache_cinfo_record_block(p->cache_path, (uint64_t) p->size,
333 p->block_size, p->mtime, p->mode, blk,
334 p->log);
335 if (p->bitmap != NULL && blk < p->nblocks) {
336 brix_cache_cinfo_mark_block(p->bitmap, blk);
337 }
338 return 0;
339}

341/* Open a partial-serve object for `key` (slice mode). NULL + *err_out on failure. */
342static brix_sd_obj_t *
343sd_cache_partial_open(brix_sd_instance_t *inst, sd_cache_inst_state *st,
344 const char *key, int *err_out)
345{
346 brix_sd_instance_t *src = st->source;
347 brix_sd_stat_t snap;
348 sd_cache_partial_t *p;
349 brix_sd_obj_t *o;
350 char cpath[PATH_MAX];
351 uint32_t bs;
352 int fd;
353
354 if (src->driver->stat == NULL
355 || src->driver->stat(src, key, &snap) != NGX_OK)
356 {
357 if (err_out != NULL) { *err_out = ENOENT; }
358 return NULL;
359 }
360 bs = (uint32_t) st->policy.slice_size;
361 /* Force owner rw ONLY (0600): the partial object is re-opened O_RDWR for every
362 * incremental block fill (a read-only 0444 source would EACCES the second open
363 * and silently fall back to a whole-file fill, §6.5). SECURITY: no group/other
364 * bits — this svc-owned cache artifact must not be directly readable by a mapped
365 * low-priv uid. The origin perms are carried in the cinfo and served back to
366 * clients (see sd_cache_stat / the READ hit), decoupled from this physical mode. */
367 fd = brix_cstore_partial_open(&st->cstore, key,
368 (mode_t) (S_IRUSR | S_IWUSR),
369 snap.size, cpath, sizeof(cpath));
370 if (fd < 0) {
371 if (err_out != NULL) { *err_out = errno ? errno : EIO; }
372 return NULL;
373 }
374 p = calloc(1, sizeof(*p));
375 o = calloc(1, sizeof(*o));
376 if (p == NULL || o == NULL) {
377 (void) close(fd);
378 free(p);
379 free(o);
380 if (err_out != NULL) { *err_out = ENOMEM; }
381 return NULL;
382 }
383 p->source = src;
384 p->cache_fd = fd;
385 p->size = snap.size;
386 p->block_size = bs;
387 p->mode = (uint32_t) (snap.mode & 0777); /* origin perms → cinfo */
388 p->mtime = (uint64_t) snap.mtime;
389 p->nblocks = brix_cache_cinfo_nblocks((uint64_t) snap.size, bs);
390 p->bitmap_len = brix_cache_cinfo_bitmap_len(p->nblocks);
391 p->log = st->log;
392 ngx_cpystrn((u_char *) p->key, (u_char *) key, sizeof(p->key));
393 ngx_cpystrn((u_char *) p->cache_path, (u_char *) cpath, sizeof(p->cache_path));
394
395 /* Adopt a previously-recorded present bitmap (an earlier partial fill) when it
396 * matches this object's geometry; otherwise start all-absent. */
397 {
398 brix_cache_cinfo_t hdr;
399 uint8_t *bm = NULL;
400 size_t bl = 0;
401
402 if (brix_cache_cinfo_load(cpath, &hdr, &bm, &bl) == NGX_OK
403 && bl == p->bitmap_len && hdr.size == (uint64_t) snap.size
404 && hdr.block_size == bs)
405 {
406 p->bitmap = bm;
407 } else {
408 free(bm);
409 p->bitmap = (p->bitmap_len > 0) ? calloc(1, p->bitmap_len) : NULL;
410 }
411 }
412
413 o->driver = inst->driver; /* our pread/close/fstat range-fill */
414 o->inst = inst;
415 o->fd = NGX_INVALID_FILE; /* a partial object is never sendfiled */
416 o->snap = snap;
417 o->state = p;
418 o->heap_shell = 1;
419 return o;
420}

422/* ---- the interposed read/write open --------------------------------------- */
423
424static brix_sd_obj_t *
425sd_cache_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
426 mode_t mode, int *err_out)
427{
428 sd_cache_inst_state *st = SD_CACHE_ST(inst);
429 brix_sd_instance_t *src = st->source;
430 brix_cache_cinfo_t ci;
431 brix_sd_obj_t *obj;
432
433 /* WRITE / CREATE / TRUNC: pass through and invalidate the cached copy. */
434 if (sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC)) {
435 obj = src->driver->open(src, path, sd_flags, mode, err_out);
436 if (obj != NULL) {
437 (void) brix_cstore_evict(&st->cstore, path);
438 }
439 return obj;
440 }
441
442 /* READ: a COMPLETE cinfo is an authoritative hit (freshness is checked at fill
443 * time via verify, not on every read - section 6.4). */
444 if (brix_cstore_cinfo_load(&st->cstore, path, &ci) == NGX_OK
445 && (ci.flags & BRIX_CINFO_F_COMPLETE))
446 {
447 obj = brix_cstore_serve_open(&st->cstore, path, err_out);
448 if (obj != NULL) {
449 /* Present the ORIGIN perms recorded in the cinfo, not the physical
450 * store object's bits (which are forced owner-writable so the cinfo
451 * xattr can be maintained). A cached object is a regular file; 0 =
452 * pre-mode cinfo → keep the store bits (snap left untouched). */
453 if (ci.mode != 0) {
454 obj->snap.mode = (mode_t) S_IFREG | (mode_t) (ci.mode & 0777);
455 }
456 ngx_log_debug1(NGX_LOG_DEBUG_CORE, st->log, 0,
457 "sd_cache: hit \"%s\"", path);
458 return obj;
459 }
460 /* the cached object vanished under us - fall through to refill */
461 }
462
463 /* Path-filtered out: serve straight from the source, never cache. */
464 if (!sd_cache_admit(&st->policy, path, -1)) {
465 return src->driver->open(src, path, sd_flags, mode, err_out);
466 }
467
468 /* Nearline (tape) source: a miss is an async recall the open parks on
469 * (section 9.2). The waiter that parks/wakes the open lands with the frm
470 * driver (SP4/SP5); until then a NEARLINE source cannot be served here, so
471 * fail soft with EAGAIN rather than block. */
472 if ((brix_sd_caps(src) & BRIX_SD_CAP_NEARLINE) != 0
473 && src->driver->recall != NULL)
474 {
475 char reqid[40];
476 ngx_int_t rr = src->driver->recall(src, path, reqid);
477
478 if (rr == NGX_AGAIN) {
479 /* Recall in flight: the open "parks" as EAGAIN. The HTTP plane answers
480 * 202 "staging" + Retry-After; a retry re-polls the recall and, once
481 * the MSS brings the object online, takes the normal miss-fill below
482 * (tape -> cache store) and serves (SP5 §9.2). */
483 ngx_log_debug1(NGX_LOG_DEBUG_CORE, st->log, 0,
484 "sd_cache: nearline recall of \"%s\" in flight (staging)", path);
485 if (err_out != NULL) {
486 *err_out = EAGAIN;
487 }
488 return NULL;
489 }
490 /* NGX_OK: already online - fall through to a normal fill. */
491 }
492
493 /* Slice/partial caching: a non-default slice_size on a LOCAL cache store fills
494 * on demand (section 6.5) instead of pulling the whole object. A miss or a
495 * PARTIAL cinfo both take this path; a COMPLETE object was already served above. */
496 if (st->policy.slice_size > 0
497 && st->cstore.meta_mode == BRIX_CMETA_LOCAL)
498 {
499 obj = sd_cache_partial_open(inst, st, path, err_out);
500 if (obj != NULL) {
501 ngx_log_debug1(NGX_LOG_DEBUG_CORE, st->log, 0,
502 "sd_cache: partial-serve \"%s\"", path);
503 return obj;
504 }
505 /* partial open failed - fall through to a whole-file fill / source read */
506 }
507
508 /* MISS: fill from the source, then serve the cached copy. */
509 if (sd_cache_fill(st, path) == NGX_OK) {
510 obj = brix_cstore_serve_open(&st->cstore, path, err_out);
511 if (obj != NULL) {
512 ngx_log_debug1(NGX_LOG_DEBUG_CORE, st->log, 0,
513 "sd_cache: filled \"%s\"", path);
514 return obj;
515 }
516 }
517
518 /* Declined or failed: serve from the source (a sick cache never fails a read,
519 * section 16). */
520 return src->driver->open(src, path, sd_flags, mode, err_out);
521}

523/* ---- namespace / xattr / dir forwarders (delegate to the source) ---------- */
524
525static ngx_int_t
526sd_cache_stat(brix_sd_instance_t *inst, const char *path,
527 brix_sd_stat_t *out)
528{
529 sd_cache_inst_state *st = SD_CACHE_ST(inst);
530 brix_sd_instance_t *s = SD_CACHE_SRC(inst);
531 brix_cache_cinfo_t ci;
532
533 /* A COMPLETE cached object answers stat from its cinfo — the same
534 * authoritative-hit doctrine as open() (section 6.4), and it keeps a stat of
535 * a warm object off the source (a remote source would otherwise take a
536 * blocking wire round-trip on the caller's thread — the event loop for the
537 * kXR_open pre-flight probe). A miss/partial falls through to the source. */
538 if (brix_cstore_cinfo_load(&st->cstore, path, &ci) == NGX_OK
539 && (ci.flags & BRIX_CINFO_F_COMPLETE))
540 {
541 ngx_memzero(out, sizeof(*out));
542 out->size = (off_t) ci.size;
543 out->mtime = (time_t) ci.mtime;
544 out->ctime = (time_t) ci.mtime;
545 out->mode = (mode_t) S_IFREG
546 | (mode_t) ((ci.mode != 0) ? (ci.mode & 0777) : 0644);
547 out->is_reg = 1;
548 return NGX_OK;
549 }
550
551 return s->driver->stat ? s->driver->stat(s, path, out) : NGX_ERROR;
552}

555sd_cache_unlink(brix_sd_instance_t *inst, const char *path, int is_dir)
556{
557 sd_cache_inst_state *st = SD_CACHE_ST(inst);
558 brix_sd_instance_t *s = st->source;

560

564 }

896/* ---- async-fill seam (SP2 "shell -> full") --------------------------------
897 * The decorator's open() runs the miss-fill INLINE - correct on a worker thread
898 * (the stream fill task, a WebDAV/S3 PUT worker) but a stall on the event loop
899 * when the fill reads a remote source or writes a remote store (a socket wire
900 * client cannot do blocking I/O on the un-pumped loop; an in-process store just
901 * freezes the worker for the transfer). The HTTP read plane therefore probes
902 * whether an inline open would block, runs the fill on the thread pool, and
903 * re-enters. These three entrypoints expose exactly that - without making the
904 * SD open() contract asynchronous. See src/shared/http_cache_fill.c. */
905
906int
907brix_sd_cache_instance_is(const brix_sd_instance_t *inst)
908{
909 return (inst != NULL && inst->driver == &brix_sd_cache_driver) ? 1 : 0;
910}

912/* Would a read-open of `key` block the calling thread on slow (remote) I/O? 1
913 * only for a cache MISS whose whole-file fill would touch a non-local tier - the
914 * source exposes no local fd (a remote read: xroot/http/s3/ceph) or the cache
915 * store is not a local POSIX dir (a remote write: e.g. a rados store). A COMPLETE
916 * hit, a slice-mode object (open returns without filling), a local->local copy,
917 * or a non-cache instance all return 0 (serve inline). No blocking call - the
918 * cinfo probe hits the per-worker L1 / a local sidecar. */
919int
920brix_sd_cache_fill_needs_offload(brix_sd_instance_t *inst, const char *key)
921{
922 sd_cache_inst_state *st;
923 brix_cache_cinfo_t ci;
924 int src_slow;
925 int store_slow;
926
927 if (!brix_sd_cache_instance_is(inst) || key == NULL) {
928 return 0;
929 }
930 st = SD_CACHE_ST(inst);
931
932 /* A COMPLETE cached object is served from the store with no fill —
933 * unless its phase-68 TTL has passed (an expired manifest refills; the
934 * failed-refill path serves it stale within the 10x-TTL bound). */
935 if (brix_cstore_cinfo_load(&st->cstore, key, &ci) == NGX_OK
936 && (ci.flags & BRIX_CINFO_F_COMPLETE))
937 {
938 if (!(st->policy.cvmfs_manifest_ttl > 0
939 && brix_cache_cinfo_expired(&ci, time(NULL)) == 1))
940 {
941 return 0;
942 }
943 /* expired: fall through to the miss logic (refill if a slow tier) */
944 }
945 /* Slice/partial mode (LOCAL store): open() returns a partial object without a
946 * whole-file fill, so the open call itself does not block. */
947 if (st->policy.slice_size > 0
948 && st->cstore.meta_mode == BRIX_CMETA_LOCAL)
949 {
950 return 0;
951 }
952 /* A miss: the inline open would run the whole-file fill. Offload iff a slow
953 * tier is involved - a remote source read or a remote store write. */
954 src_slow = (brix_sd_caps(st->source) & BRIX_SD_CAP_FD) == 0;
955 store_slow = (st->cstore.meta_mode != BRIX_CMETA_LOCAL);
956 return (src_slow || store_slow) ? 1 : 0;
957}
958
959/* Run the whole-file fill for `key` (source -> cache store + cinfo) on the
960 * CALLING thread - the worker-thread half of the offload. NGX_OK (cached),
961 * NGX_DECLINED (admission declined - not cached), NGX_ERROR (fill failure).
962 * Safe off the event loop: pure driver pread/pwrite + cstore ops, no nginx pool. */
963ngx_int_t
964brix_sd_cache_fill_key(brix_sd_instance_t *inst, const char *key)
965{
966 if (!brix_sd_cache_instance_is(inst) || key == NULL) {
967 errno = EINVAL;
968 return NGX_ERROR;
969 }
970 return sd_cache_fill(SD_CACHE_ST(inst), key);
971}
