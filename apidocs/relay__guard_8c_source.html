1/*
2 * relay_guard.c — frame classification + drop enforcement for the relay.
3 *
4 * WHAT: maps decoded tap frames onto the pure-C guard core: request opcodes
5 * -> op-classes for the pre-verdict (junk path signatures / grammar),
6 * kXR_error responses -> outcomes for the post-signal (notfound/authfail),
7 * and one guard audit line per flagged frame on the relay's log.
8 * WHY: a root:// scanner probing junk paths through the relay must be cut
9 * off before the backend serves it, and weak signals must reach fail2ban
10 * in the same key=value shape the HTTP guard emits.
11 * HOW: pure mapping helpers feed guard_classify_pre/post; the only side
12 * effects are the drop flag and the audit log line. The relay pump checks
13 * the flag after every tap feed and tears the connection down.
14 */
15

21/* ---- kXR_* request opcode -> guard op-class ----
22 *
23 * WHAT: maps the wire requestid to the guard's protocol-agnostic op-class;
24 * unlisted opcodes are GUARD_OP_UNKNOWN.
25 *
26 * WHY: the "root" grammar profile permits op-classes, not raw opcodes, so
27 * ruleset semantics stay identical across the three guarded surfaces.
28 *
29 * HOW: 1. switch over the opcodes the relay can meaningfully classify
30 * (session/handshake ops map to HANDSHAKE so login flows are never
31 * grammar-bounced).
32 * 2. Every other opcode inside the legal request range (kXR_auth ..
33 * kXR_clone: ping/close/sync/bind/fattr/prepare/…) is legitimate
34 * housekeeping -> INFO; only out-of-range requestids are UNKNOWN
35 * (a non-XRootD client talking to the port).
36 */
37static guard_op_class_t
38opcode_to_op(uint16_t opcode)
39{
40 switch (opcode) {
41 case kXR_open:
42 case kXR_read:
43 case kXR_readv:
44 case kXR_pgread:
45 case kXR_locate:
46 case kXR_statx: return GUARD_OP_READ;
47 case kXR_write:
48 case kXR_pgwrite:
49 case kXR_writev:
50 case kXR_mkdir:
51 case kXR_mv:
52 case kXR_chmod:
53 case kXR_chkpoint:
54 case kXR_truncate: return GUARD_OP_WRITE;
55 case kXR_dirlist: return GUARD_OP_LIST;
56 case kXR_rm:
57 case kXR_rmdir: return GUARD_OP_DELETE;
58 case kXR_stat:
59 case kXR_query: return GUARD_OP_INFO;
60 case kXR_prepare: return GUARD_OP_STAGE;
61 case kXR_login:
62 case kXR_auth:
63 case kXR_protocol:
64 case kXR_bind:
65 case kXR_endsess:
66 case kXR_sigver: return GUARD_OP_HANDSHAKE;
67 default:
68 if (opcode >= kXR_auth && opcode <= kXR_clone) {
69 return GUARD_OP_INFO; /* legal housekeeping (ping/close/…) */
70 }
71 return GUARD_OP_UNKNOWN; /* off-spec requestid */
72 }
73}

75/* ---- kXR_error errnum -> guard outcome ----
76 *
77 * WHAT: maps the error code carried in a kXR_error response payload (surfaced
78 * as frame->errnum by the tap) to the outcomes the guard flags.
79 *
80 * WHY: kXR_NotFound storms and kXR_NotAuthorized streaks are the root://
81 * equivalents of HTTP 404/401 — the same fail2ban thresholds apply.
82 *
83 * HOW: 1. kXR_NotFound -> NOTFOUND; kXR_NotAuthorized -> AUTHFAIL.
84 * 2. Any other kXR_error -> ERROR; non-error statuses -> OK.
85 */
86static guard_outcome_t
87errnum_to_outcome(uint16_t status, uint32_t errnum)
88{
89 if (status != kXR_error) {
90 return OUTCOME_OK;
91 }
92 if (errnum == kXR_NotFound) {
93 return OUTCOME_NOTFOUND;
94 }
95 if (errnum == kXR_NotAuthorized) {
96 return OUTCOME_AUTHFAIL;
97 }
98 return OUTCOME_ERROR;
99}

101/* ---- Emit one guard audit line on the relay log ----
102 *
103 * WHAT: formats req/reason with the shared guard formatter (cached ISO-8601
104 * timestamp) and logs it at INFO on the relay's stable log.
105 *
106 * WHY: the stream relay has no per-location audit file; the error log is the
107 * operator-visible destination, and the line body keeps the exact
108 * key=value contract so it stays grep/fail2ban-parseable.
109 *
110 * HOW: 1. Copy the cached ISO-8601 clock (adapters own the clock).
111 * 2. guard_audit_format; log only complete lines.
112 */
113static void
114relay_guard_audit(brix_relay_guard_t *g, const guard_request_t *req,
115 guard_reason_t reason)
116{
117 char line[1280];
118 char ts[sizeof("YYYY-MM-DDThh:mm:ss+00:00")];
119 size_t ts_len;
120
121 ts_len = ngx_cached_http_log_iso8601.len;
122 if (ts_len >= sizeof(ts)) {
123 ts_len = sizeof(ts) - 1;
124 }
125 ngx_memcpy(ts, ngx_cached_http_log_iso8601.data, ts_len);
126 ts[ts_len] = '\0';
127
128 if (guard_audit_format(req, reason, ts, line, sizeof(line)) > 0) {
129 ngx_log_error(NGX_LOG_INFO, g->log, 0, "%s", line);
130 }
131}

133/* ---- Initialize the per-relay guard state ----
134 *
135 * WHAT: zeroes the state; when enabled, builds the ruleset (built-in scanner
136 * signatures + "root" profile grammar) and copies the client address.
137 *
138 * WHY: the ruleset is per-connection state assembled once at relay start —
139 * the sink then classifies allocation-free for the connection's life.
140 *
141 * HOW: 1. Zero; store log; bail if disabled.
142 * 2. guard_ruleset_init + default signatures + "root" profile.
143 * 3. NUL-terminate the client addr_text copy (audit line ip= field).
144 */
145void
146brix_relay_guard_init(brix_relay_guard_t *g, int enable,
147 const ngx_str_t *client_addr, ngx_log_t *log)
148{
149 size_t ip_len;
150
151 ngx_memzero(g, sizeof(*g));
152 g->log = log;
153 if (!enable) {
154 return;
155 }
156 g->enable = 1;
157
158 guard_ruleset_init(&g->rules);
159 guard_ruleset_add_default_signatures(&g->rules);
160 guard_ruleset_load_profile(&g->rules, "root");
161
162 ip_len = client_addr->len < sizeof(g->ip) - 1
163 ? client_addr->len : sizeof(g->ip) - 1;
164 ngx_memcpy(g->ip, client_addr->data, ip_len);
165 g->ip[ip_len] = '\0';
166}

168/* ---- Tap sink: classify one decoded frame ----
169 *
170 * WHAT: C2U request frames run the pre-verdict (signature/grammar) and set
171 * the drop flag on BOUNCE; U2C response frames run the post-signal
172 * (notfound/authfail). Either way a flagged frame emits one audit line.
173 *
174 * WHY: this is the whole guard enforcement surface for root:// — the relay
175 * never terminates the protocol, so classification rides the tap.
176 *
177 * HOW: 1. No-op when disabled.
178 * 2. NUL-terminate + sanitize the wire path (INVARIANT: every logged
179 * wire string passes brix_sanitize_log_string).
180 * 3. Requests: opcode->op, classify_pre, set drop + status=0 audit.
181 * 4. Responses: errnum->outcome, classify_post.
182 * 5. Log the audit line when a reason fired.
183 */
184void
185brix_relay_guard_sink(void *ctx, const brix_tap_frame_t *f,
186 brix_tap_dir_t dir, const uint8_t *payload, size_t payload_len)
187{
188 brix_relay_guard_t *g = ctx;
189 guard_request_t req;
190 guard_reason_t why = GUARD_R_NONE;
191 char raw_path[BRIX_TAP_PATH_CAP + 1];
192 char san_path[BRIX_TAP_PATH_CAP + 1];
193 size_t raw_len = 0;
194
195 (void) payload;
196 (void) payload_len;
197
198 if (!g->enable || g->drop) {
199 return;
200 }
201
202 if (f->path != NULL && f->path_len > 0) {
203 raw_len = f->path_len < BRIX_TAP_PATH_CAP
204 ? f->path_len : BRIX_TAP_PATH_CAP;
205 ngx_memcpy(raw_path, f->path, raw_len);
206 }
207 raw_path[raw_len] = '\0';
208
209 req.ip = g->ip;
210 req.proto = "root";
211 req.path_len = brix_sanitize_log_string(raw_path, san_path,
212 sizeof(san_path));
213 req.path = san_path;
214 req.cred_present = 0; /* the relay never sees the credential */
215 req.status_code = 0;
216
217 if (dir == BRIX_TAP_C2U && f->is_request) {
218 req.op = opcode_to_op(f->opcode);
219 req.outcome = OUTCOME_PENDING;
220 if (guard_classify_pre(&g->rules, &req, &why) == GUARD_BOUNCE) {
221 g->drop = 1; /* the pump tears the relay down */
222 }
223 } else if (dir == BRIX_TAP_U2C && !f->is_request) {
224 req.op = GUARD_OP_UNKNOWN;
225 req.status_code = (int) f->status;
226 req.outcome = errnum_to_outcome(f->status, f->errnum);
227 why = guard_classify_post(&g->rules, &req);
228 }
229
230 if (why != GUARD_R_NONE) {
231 relay_guard_audit(g, &req, why);
232 }
233}
