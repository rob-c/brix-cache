8/* Expand one source into `out`: a root:// glob via brix_glob, a local glob via
9 * glob(3), otherwise the literal. Returns 0; -1 only on a hard (alloc) failure.
10 * A glob that matches nothing appends nothing and warns. */
11/* Like brix_has_glob, but a '?' that introduces an opaque section
12 * (root://...?key=val, e.g. ?xrdcl.unzip=member) is the opaque separator, not a
13 * wildcard — so only the path before it is considered for globbing. */
14int
15source_has_glob(const char *s)
16{
17 const char *q = strchr(s, '?');
18
19 if (q != NULL && strchr(q, '=') != NULL) {
20 size_t plen = (size_t) (q - s), i;
21 for (i = 0; i < plen; i++) {
22 if (s[i] == '*' || s[i] == '[' || s[i] == '?') {
23 return 1;
24 }
25 }
26 return 0;
27 }
28 return brix_has_glob(s);
29}

32int
33expand_source(const char *s_in, const brix_opts *co, char ***out, size_t *n, size_t *cap)
34{
35 char rs[XRDC_PATH_MAX];
36 const char *s;
37 brix_alias_resolve(s_in, rs, sizeof(rs)); /* ~/.xrdrc: name:suffix -> URL */
38 s = rs;
39 if (!source_has_glob(s)) {
40 return str_append(out, n, cap, s);
41 }
42 if (brix_is_web_url(s)) {
43 return str_append(out, n, cap, s); /* web globbing not supported; literal */
44 }
45 if (is_root_url(s)) {
46 char **m = NULL;
47 size_t nm = 0, i;
48 brix_status st;
49 brix_status_clear(&st);
50 if (brix_glob(s, co, &m, &nm, &st) < 0) {
51 if (st.kxr == XRDC_EUSAGE) {
52 return str_append(out, n, cap, s); /* genuinely not a glob → literal */
53 }
54 /* Hard failure (connect/dirlist/auth): surface it; don't silently fall
55 * back to copying the literal '*' pattern as a filename. */
56 fprintf(stderr, "xrdcp: glob %s: %s\n", s, st.msg);
57 return 0;
58 }
59 if (nm == 0) {
60 fprintf(stderr, "xrdcp: no matches for %s\n", s);
61 }
62 for (i = 0; i < nm; i++) {
63 if (str_append(out, n, cap, m[i]) != 0) { brix_glob_free(m, nm); return -1; }
64 }
65 brix_glob_free(m, nm);
66 return 0;
67 }
68 {
69 glob_t g;
70 int rc = glob(s, 0, NULL, &g);
71 if (rc == 0) {
72 size_t i;
73 for (i = 0; i < g.gl_pathc; i++) {
74 if (str_append(out, n, cap, g.gl_pathv[i]) != 0) { globfree(&g); return -1; }
75 }
76 globfree(&g);
77 return 0;
78 }
79 globfree(&g);
80 fprintf(stderr, "xrdcp: no matches for %s\n", s);
81 return 0;
82 }
83}

86/* mkdir -p of the directory component of `filepath` (errors other than EEXIST are
87 * left for the subsequent open to report). */
88void
89mkdirs_for(const char *filepath)
90{
91 char tmp[XRDC_PATH_MAX];
92 char *slash, *p;
93 snprintf(tmp, sizeof(tmp), "%s", filepath);
94 slash = strrchr(tmp, '/');
95 if (slash == NULL || slash == tmp) {
96 return;
97 }
98 *slash = '\0';
99 for (p = tmp + 1; *p != '\0'; p++) {
100 if (*p == '/') {
101 *p = '\0';
102 (void) mkdir(tmp, 0755);
103 *p = '/';
104 }
105 }
106 (void) mkdir(tmp, 0755);
107}

110/* MKCOL each ancestor collection of `rel` under `base` on web endpoint `du`
111 * (davs/http only; S3 keys are flat → no-op). Idempotent, created top-down so a
112 * deeper PUT never hits 409. 0 / -1 (st set). */
113int
114mkcol_parents(const brix_weburl *du, const char *base, const char *rel,
115 const char *bearer, const brix_opts *co, brix_status *st)
116{
117 char acc[XRDC_PATH_MAX * 2];
118 const char *slash;
119 size_t blen;
120
121 if (du->is_s3) { return 0; }
122 blen = (size_t) snprintf(acc, sizeof(acc), "%s", base);
123 if (blen >= sizeof(acc)) { return -1; }
124 for (slash = strchr(rel, '/'); slash != NULL; slash = strchr(slash + 1, '/')) {
125 int w = snprintf(acc + blen, sizeof(acc) - blen, "/%.*s",
126 (int) (slash - rel), rel);
127 if (w < 0 || blen + (size_t) w >= sizeof(acc)) { return -1; }
128 if (brix_webdav_mkcol(du, acc, bearer, co ? co->verify_host : 1,
129 co ? co->ca_dir : NULL, st) != 0) {
130 return -1;
131 }
132 }
133 return 0;
134}

137/* Place one recursively-listed source file `srcurl` at `rel` under `dstroot`,
138 * which may be a LOCAL directory (mkdir -p + copy) or a WEB collection (MKCOL the
139 * parent collections + a web->web relay via copy_one_with_retry). Centralising the
140 * destination side lets both the WebDAV and S3 recursive sources copy to a local
141 * tree OR another web endpoint (web->web). 0 / -1 (st set).
142 *
143 * Filter and dry-run are applied here — before any I/O — so they cover all
144 * recursive sources (WebDAV, S3). A filtered or dry-run item returns 0 (handled,
145 * not a failure) so the caller counts it in `ok`, not `fail`. */
146int
147recursive_place(const char *dstroot, const char *rel, const char *srcurl,
148 const brix_copy_opts *fo, const brix_opts *co, int retries,
149 brix_status *st)
150{
151 if (fo != NULL && !brix_copy_filter_match(fo, rel)) {
152 return 0; /* filtered — not an error, skip silently */
153 }
154 if (fo != NULL && fo->dry_run) {
155 printf("[dry-run] copy %s -> %s/%s\n", srcurl, dstroot, rel);
156 return 0;
157 }
158 if (brix_is_web_url(dstroot)) {
159 brix_weburl du;
160 char dbase[XRDC_PATH_MAX], dsturl[XRDC_PATH_MAX * 2 + 320];
161 const char *dbearer;
162 size_t blen;
163 if (brix_weburl_parse(dstroot, &du) != 0) {
164 brix_status_set(st, XRDC_EUSAGE, 0, "bad web dst URL %s", dstroot);
165 return -1;
166 }
167 snprintf(dbase, sizeof(dbase), "%s", du.path);
168 blen = strlen(dbase);
169 while (blen > 0 && dbase[blen - 1] == '/') { dbase[--blen] = '\0'; }
170 dbearer = (fo != NULL && fo->bearer != NULL) ? fo->bearer
171 : getenv("BEARER_TOKEN");
172 if (mkcol_parents(&du, dbase, rel, dbearer, co, st) != 0) {
173 return -1;
174 }
175 if ((size_t) snprintf(dsturl, sizeof(dsturl), "%s://%s:%d%s/%s",
176 web_scheme_str(du.proto), du.host, du.port, dbase, rel)
177 >= sizeof(dsturl)) {
178 brix_status_set(st, XRDC_EUSAGE, 0, "web->web dst path too long");
179 return -1;
180 }
181 return copy_one_with_retry(srcurl, dsturl, fo, co, retries, st);
182 } else {
183 char dstfile[XRDC_PATH_MAX];
184 if ((size_t) snprintf(dstfile, sizeof(dstfile), "%s/%s", dstroot, rel)
185 >= sizeof(dstfile)) {
186 brix_status_set(st, XRDC_EUSAGE, 0, "dst path too long");
187 return -1;
188 }
189 mkdirs_for(dstfile);
190 return copy_one_with_retry(srcurl, dstfile, fo, co, retries, st);
191 }
192}

195/* For a recursive copy whose destination is a web COLLECTION, MKCOL the base
196 * collection once (idempotent) so the first per-file PUT doesn't 409. No-op for a
197 * local destination or an S3 bucket (flat keys need no collections). */
198void
199ensure_web_dst_base(const char *dstroot, const brix_copy_opts *fo,
200 const brix_opts *co)
201{
202 brix_weburl du;
203 char dbase[XRDC_PATH_MAX];
204 const char *dbearer;
205 size_t blen;
206 brix_status st;
207
208 if (!brix_is_web_url(dstroot) || brix_weburl_parse(dstroot, &du) != 0
209 || du.is_s3 || (fo != NULL && fo->dry_run)) {
210 return;
211 }
212 snprintf(dbase, sizeof(dbase), "%s", du.path);
213 blen = strlen(dbase);
214 while (blen > 0 && dbase[blen - 1] == '/') { dbase[--blen] = '\0'; }
215 if (dbase[0] == '\0') { return; } /* root collection needs no MKCOL */
216 dbearer = (fo != NULL && fo->bearer != NULL) ? fo->bearer
217 : getenv("BEARER_TOKEN");
218 brix_status_clear(&st);
219 (void) brix_webdav_mkcol(&du, dbase, dbearer, co ? co->verify_host : 1,
220 co ? co->ca_dir : NULL, &st); /* best-effort */
221}

224/* Recursively download an s3:// prefix into dstdir: ListObjectsV2 (paginated,
225 * SigV4-signed) then copy each key to dstdir/<key-minus-prefix>, preserving the
226 * tree. `fo` is a non-recursive opts copy carrying the S3 creds. Returns 0/1. */
227int
228recursive_s3_download(const brix_weburl *u, const char *dstdir,
229 const brix_copy_opts *fo, const brix_opts *co, int retries)
230{
231 brix_status st;
232 char **keys = NULL;
233 char bucket[256], prefix[XRDC_PATH_MAX];
234 const char *ak, *sk, *region, *scheme, *bsl;
235 size_t n = 0, i, plen, ok = 0, fail = 0;
236
237 brix_status_clear(&st);
238 ak = fo->s3_access ? fo->s3_access : getenv("AWS_ACCESS_KEY_ID");
239 sk = fo->s3_secret ? fo->s3_secret : getenv("AWS_SECRET_ACCESS_KEY");
240 region = fo->s3_region ? fo->s3_region : getenv("AWS_DEFAULT_REGION");
241 if (brix_s3_list(u, ak, sk, region, co ? co->verify_host : 1,
242 co ? co->ca_dir : NULL, &keys, &n, &st) != 0) {
243 fprintf(stderr, "xrdcp: s3 list: %s\n", st.msg);
244 return 1;
245 }
246 /* split u->path "/bucket[/prefix]" into bucket + prefix (no trailing slash).
247 * One guarded memcpy for the bucket covers both the no-slash and prefix cases
248 * (brix_s3_list already validated the same split, but stay defensive). */
249 bsl = strchr(u->path + 1, '/');
250 {
251 size_t bl = bsl ? (size_t) (bsl - (u->path + 1)) : strlen(u->path + 1);
252 if (bl >= sizeof(bucket)) {
253 fprintf(stderr, "xrdcp: s3 bucket name too long\n");
254 brix_strv_free(keys, n);
255 return 1;
256 }
257 memcpy(bucket, u->path + 1, bl);
258 bucket[bl] = '\0';
259 if (bsl != NULL) {
260 snprintf(prefix, sizeof(prefix), "%s", bsl + 1);
261 } else {
262 prefix[0] = '\0';
263 }
264 }
265 plen = strlen(prefix);
266 while (plen > 0 && prefix[plen - 1] == '/') { prefix[--plen] = '\0'; }
267 scheme = web_scheme_str(u->proto);
268 ensure_web_dst_base(dstdir, fo, co); /* s3->web: create the dst collection */
269
270 for (i = 0; i < n; i++) {
271 const char *key = keys[i];
272 const char *rel;
273 char relbuf[XRDC_NAME_MAX];
274 char srcurl[XRDC_PATH_MAX + 320];
275 brix_status cst;
276
277 if (plen == 0) {
278 rel = key;
279 } else if (strncmp(key, prefix, plen) == 0 && (key[plen] == '/' || key[plen] == '\0')) {
280 rel = key + plen + (key[plen] == '/' ? 1 : 0);
281 } else {
282 path_basename(key, relbuf, sizeof(relbuf)); /* fallback: flatten */
283 rel = relbuf;
284 }
285 if (*rel == '\0') {
286 continue;
287 }
288 if (rel_is_unsafe(rel)) {
289 fprintf(stderr, "xrdcp: refusing unsafe key from server: %s\n", key);
290 fail++;
291 continue;
292 }
293 if ((size_t) snprintf(srcurl, sizeof(srcurl), "%s://%s:%d/%s/%s",
294 scheme, u->host, u->port, bucket, key) >= sizeof(srcurl)) {
295 fail++;
296 continue;
297 }
298 /* dstdir may be a local dir OR a web collection (recursive s3->web). */
299 brix_status_clear(&cst);
300 if (recursive_place(dstdir, rel, srcurl, fo, co, retries, &cst) == 0) {
301 ok++;
302 if (!fo->silent) {
303 fprintf(stderr, "[%zu/%zu] %s -> %s/%s\n", ok + fail, n, srcurl, dstdir, rel);
304 }
305 } else {
306 fail++;
307 fprintf(stderr, "xrdcp: %s: %s\n", srcurl, cst.msg);
308 }
309 }
310 brix_strv_free(keys, n);
311 if (!fo->silent) {
312 fprintf(stderr, "xrdcp: %zu copied, %zu failed (recursive s3)\n", ok, fail);
313 }
314 return (fail == 0) ? 0 : 1;
315}

318int
319recursive_web_download(const char *src, const char *dstdir, const brix_copy_opts *o,
320 const brix_opts *co, int retries)
321{
322 brix_weburl u;
323 brix_status st;
324 brix_copy_opts fo;
325 char **paths = NULL;
326 char prefix[XRDC_PATH_MAX];
327 const char *bearer, *scheme;
328 size_t n = 0, i, plen, ok = 0, fail = 0;
329
330 /* Each listed file is a plain (non-recursive) copy; clear the -r flag so the
331 * per-file brix_copy doesn't bounce off the "no recursive web" guard. */
332 fo = *o;
333 fo.recursive = 0;
334 brix_status_clear(&st);
335 if (brix_weburl_parse(src, &u) != 0) {
336 fprintf(stderr, "xrdcp: bad web URL %s\n", src);
337 return 1;
338 }
339 if (u.is_s3) {
340 return recursive_s3_download(&u, dstdir, &fo, co, retries);
341 }
342 bearer = (o != NULL && o->bearer != NULL) ? o->bearer : getenv("BEARER_TOKEN");
343 if (brix_webdav_list(&u, bearer, co ? co->verify_host : 1,
344 co ? co->ca_dir : NULL, &paths, &n, &st) != 0) {
345 fprintf(stderr, "xrdcp: list %s: %s\n", src, st.msg);
346 return 1;
347 }
348 snprintf(prefix, sizeof(prefix), "%s", u.path);
349 plen = strlen(prefix);
350 while (plen > 1 && prefix[plen - 1] == '/') { prefix[--plen] = '\0'; }
351 scheme = web_scheme_str(u.proto);
352 ensure_web_dst_base(dstdir, &fo, co); /* web->web: create the dst collection */
353
354 for (i = 0; i < n; i++) {
355 const char *path = paths[i];
356 const char *rel;
357 char relbuf[XRDC_NAME_MAX];
358 char srcurl[XRDC_PATH_MAX + 320];
359 brix_status cst;
360
361 if (strncmp(path, prefix, plen) == 0 && (path[plen] == '/' || path[plen] == '\0')) {
362 rel = path + plen + (path[plen] == '/' ? 1 : 0);
363 } else {
364 path_basename(path, relbuf, sizeof(relbuf)); /* fallback: flatten */
365 rel = relbuf;
366 }
367 if (*rel == '\0') {
368 continue;
369 }
370 /* Security: a hostile server must not traverse out of dstdir via the href. */
371 if (rel_is_unsafe(rel)) {
372 fprintf(stderr, "xrdcp: refusing unsafe path from server: %s\n", path);
373 fail++;
374 continue;
375 }
376 if ((size_t) snprintf(srcurl, sizeof(srcurl), "%s://%s:%d%s",
377 scheme, u.host, u.port, path) >= sizeof(srcurl)) {
378 fail++;
379 continue;
380 }
381 /* dstdir may be a local dir OR a web collection (recursive web->web). */
382 brix_status_clear(&cst);
383 if (recursive_place(dstdir, rel, srcurl, &fo, co, retries, &cst) == 0) {
384 ok++;
385 if (o == NULL || !o->silent) {
386 fprintf(stderr, "[%zu/%zu] %s -> %s/%s\n", ok + fail, n, srcurl, dstdir, rel);
387 }
388 } else {
389 fail++;
390 fprintf(stderr, "xrdcp: %s: %s\n", srcurl, cst.msg);
391 }
392 }
393 brix_strv_free(paths, n);
394 if (o == NULL || !o->silent) {
395 fprintf(stderr, "xrdcp: %zu copied, %zu failed (recursive web)\n", ok, fail);
396 }
397 return (fail == 0) ? 0 : 1;
398}

416/* Recursively walk a local directory, MKCOL'ing each WebDAV collection (davs/http
417 * only — S3 keys are flat) and PUT'ing each regular file. `rel` is the path of the
418 * current directory relative to the upload root ("" at the top). Symlinks and
419 * special files are skipped (only real dirs + regular files are uploaded). */
420void
421web_upload_walk(web_upload_ctx *c, const char *localdir, const char *rel)
422{
423 DIR *d = opendir(localdir);
424 struct dirent *de;
425
426 if (d == NULL) {
427 fprintf(stderr, "xrdcp: cannot open %s: %s\n", localdir, strerror(errno));
428 c->fail++;
429 return;
430 }
431 while ((de = readdir(d)) != NULL) {
432 char childlocal[XRDC_PATH_MAX];
433 char childrel[XRDC_PATH_MAX];
434 struct stat sb;
435
436 if (strcmp(de->d_name, ".") == 0 || strcmp(de->d_name, "..") == 0) {
437 continue;
438 }
439 if ((size_t) snprintf(childlocal, sizeof(childlocal), "%s/%s",
440 localdir, de->d_name) >= sizeof(childlocal)
441 || (rel[0] == '\0'
442 ? (size_t) snprintf(childrel, sizeof(childrel), "%s", de->d_name)
443 : (size_t) snprintf(childrel, sizeof(childrel), "%s/%s",
444 rel, de->d_name)) >= sizeof(childrel)) {
445 fprintf(stderr, "xrdcp: path too long under %s\n", localdir);
446 c->fail++;
447 continue;
448 }
449 if (lstat(childlocal, &sb) != 0) {
450 fprintf(stderr, "xrdcp: stat %s: %s\n", childlocal, strerror(errno));
451 c->fail++;
452 continue;
453 }
454 if (S_ISDIR(sb.st_mode)) {
455 /* Create the remote collection (top-down) before descending so the
456 * child PUTs/MKCOLs don't hit 409 Conflict. S3 has no real dirs.
457 * Under --dry-run, skip the MKCOL but still recurse so files get
458 * printed. */
459 if (!c->u->is_s3 && (c->fo == NULL || !c->fo->dry_run)) {
460 char rpath[XRDC_PATH_MAX * 2];
461 brix_status mst;
462 brix_status_clear(&mst);
463 if (web_join(c->base, childrel, rpath, sizeof(rpath)) != 0
464 || brix_webdav_mkcol(c->u, rpath, c->bearer,
465 c->co ? c->co->verify_host : 1,
466 c->co ? c->co->ca_dir : NULL, &mst) != 0) {
467 fprintf(stderr, "xrdcp: mkcol %s: %s\n", childrel, mst.msg);
468 c->fail++;
469 continue; /* skip this subtree — its files would 409 */
470 }
471 }
472 web_upload_walk(c, childlocal, childrel);
473 } else if (S_ISREG(sb.st_mode)) {
474 char rurl[XRDC_PATH_MAX * 2 + 320];
475 char rpath[XRDC_PATH_MAX * 2];
476 brix_status cst;
477 /* Apply --exclude/--include and --dry-run before any I/O. */
478 if (c->fo != NULL && !brix_copy_filter_match(c->fo, childrel)) {
479 continue; /* filtered — skip, not a failure */
480 }
481 if (web_join(c->base, childrel, rpath, sizeof(rpath)) != 0
482 || (size_t) snprintf(rurl, sizeof(rurl), "%s://%s:%d%s",
483 c->scheme, c->u->host, c->u->port, rpath)
484 >= sizeof(rurl)) {
485 fprintf(stderr, "xrdcp: remote path too long for %s\n", childrel);
486 c->fail++;
487 continue;
488 }
489 if (c->fo != NULL && c->fo->dry_run) {
490 printf("[dry-run] copy %s -> %s\n", childlocal, rurl);
491 c->ok++;
492 continue;
493 }
494 brix_status_clear(&cst);
495 if (copy_one_with_retry(childlocal, rurl, c->fo, c->co, c->retries,
496 &cst) == 0) {
497 c->ok++;
498 if (c->fo == NULL || !c->fo->silent) {
499 fprintf(stderr, "[%zu] %s -> %s\n", c->ok + c->fail,
500 childlocal, rurl);
501 }
502 } else {
503 c->fail++;
504 fprintf(stderr, "xrdcp: %s: %s\n", rurl, cst.msg);
505 }
506 }
507 /* else: symlink / fifo / device — skip (not uploaded) */
508 }
509 closedir(d);
510}

513/* Recursively upload a local directory's CONTENTS into a web (davs/http/s3)
514 * collection: `xrdcp -r ./dir davs://h/coll/` → coll/<files-under-dir>. The wire
515 * has no recursive transfer op, so walk locally + MKCOL + per-file PUT. Returns
516 * 0 if every file uploaded, 1 otherwise. */
517int
518recursive_web_upload(const char *localdir, const char *dst, const brix_copy_opts *o,
519 const brix_opts *co, int retries)
520{
521 brix_weburl u;
522 brix_copy_opts fo;
523 char base[XRDC_PATH_MAX];
524 web_upload_ctx c;
525 size_t blen;
526
527 if (brix_weburl_parse(dst, &u) != 0) {
528 fprintf(stderr, "xrdcp: bad web URL %s\n", dst);
529 return 1;
530 }
531 /* Each file is a plain (non-recursive) copy so brix_copy's "no recursive web"
532 * guard doesn't trip. */
533 fo = *o;
534 fo.recursive = 0;
535
536 snprintf(base, sizeof(base), "%s", u.path);
537 blen = strlen(base);
538 while (blen > 0 && base[blen - 1] == '/') { base[--blen] = '\0'; }
539
540 c.u = &u;
541 c.base = base;
542 c.scheme = web_scheme_str(u.proto);
543 c.bearer = (o != NULL && o->bearer != NULL) ? o->bearer : getenv("BEARER_TOKEN");
544 c.fo = &fo;
545 c.co = co;
546 c.retries = retries;
547 c.ok = 0;
548 c.fail = 0;
549
550 /* Ensure the destination collection itself exists (idempotent). Root ("")
551 * and S3 buckets need no MKCOL. Under --dry-run, skip to avoid creating
552 * the base collection on the remote. */
553 if (!u.is_s3 && base[0] != '\0' && !fo.dry_run) {
554 brix_status mst;
555 brix_status_clear(&mst);
556 if (brix_webdav_mkcol(&u, base, c.bearer, co ? co->verify_host : 1,
557 co ? co->ca_dir : NULL, &mst) != 0) {
558 fprintf(stderr, "xrdcp: mkcol %s: %s\n", base, mst.msg);
559 /* proceed anyway — PUTs will surface any real problem */
560 }
561 }
562
563 web_upload_walk(&c, localdir, "");
564 if (o == NULL || !o->silent) {
565 fprintf(stderr, "xrdcp: %zu copied, %zu failed (recursive web upload)\n",
566 c.ok, c.fail);
567 }
568 return (c.fail == 0) ? 0 : 1;
569}
