2 * vfs_backend_config_s3.c — the s3:// and root(s)://-or-local-driver origin
3 * parsers plus their per-driver registry-entry builders (phase-79 file-size
4 * split of vfs_backend_config.c).
5 *
6 * WHAT: Parses "s3://host[:port]/bucket" into a read-only S3 source backend and
7 * "root://host:port" / "roots://host:port" into a remote root:// primary
8 * backend (any other value falling through to the local driver name via

24/* Register a read-only S3 source backend for an export (phase-64): the export's
25 * bytes live in a remote S3 bucket, served over the shared libcurl S3 transport
26 * (signed Range GET). bucket is the path-style bucket name; port defaults to
27 * 80/443 by tls when the URL omits it. The driver is CAP_RANGE_READ only, so an
28 * S3 primary is read-only (writes are rejected, exactly like an http:// primary). */
29void
30brix_vfs_backend_config_s3(const char *root_canon, const char *host,
31 int port, int tls, const char *bucket)
32{
33 brix_vfs_backend_entry_t *e;
34
35 if (root_canon == NULL || root_canon[0] == '\0' || host == NULL
36 || host[0] == '\0' || bucket == NULL || bucket[0] == '\0'
37 || port <= 0 || port > 65535)
38 {
39 return;
40 }
41 e = brix_vfs_backend_entry_get_or_create(root_canon);
42 if (e == NULL) {
43 return;
44 }
45 ngx_memcpy(e->backend, "s3", sizeof("s3"));
46 ngx_cpystrn((u_char *) e->origin_host, (u_char *) host,
47 sizeof(e->origin_host));
48 e->origin_port = port;
49 e->origin_tls = tls;
50 ngx_cpystrn((u_char *) e->origin_path, (u_char *) bucket,
51 sizeof(e->origin_path)); /* origin_path carries the bucket */
52 e->inst = NULL; /* rebuilt on next resolve */
53}

87/* Split an s3:// authority "host[:port]" into host length and numeric port,
88 * defaulting to 7480 (radosgw) when no ":port" is present. Returns NGX_OK, or
89 * NGX_ERROR after an [emerg] for a bad port. */
90static ngx_int_t
91vfs_parse_s3_port(ngx_conf_t *cf, const u_char *authority,
92 size_t authority_len, vfs_origin_parse_t *out)
93{
94 const u_char *colon = NULL;
95 size_t i;
96
97 for (i = authority_len; i > 0; i--) {
98 if (authority[i - 1] == ':') {
99 colon = authority + i - 1;
100 break;
101 }
102 }
103
104 if (colon == NULL) {
105 out->host_len = authority_len;
106 out->port = 7480; /* radosgw S3 default */
107 return NGX_OK;
108 }
109
110 {
111 ngx_int_t pn;
112 pn = ngx_atoi((u_char *) colon + 1,
113 (size_t) (authority + authority_len - (colon + 1)));
114 if (pn == NGX_ERROR || pn <= 0 || pn > 65535) {
115 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
116 "brix_storage_backend: invalid s3 origin port");
117 return NGX_ERROR;
118 }
119 out->host_len = (size_t) (colon - authority);
120 out->port = (int) pn;
121 }
122
123 return NGX_OK;
124}

126/* Parse "s3://host[:port]/bucket" into host/port/base(bucket). Returns NGX_OK,
127 * or NGX_ERROR after an [emerg] for a malformed URL. */
128static ngx_int_t
129vfs_parse_s3_url(ngx_conf_t *cf, const ngx_str_t *sb, vfs_origin_parse_t *out)
130{
131 u_char *authority = sb->data + sizeof("s3://") - 1;
132 size_t rem = sb->len - (sizeof("s3://") - 1);
133 size_t i, slash = rem, authority_len, path_len, host_len;
134 const char *path;
135
136 for (i = 0; i < rem; i++) {
137 if (authority[i] == '/') {
138 slash = i;
139 break;
140 }
141 }
142 authority_len = slash;
143 path = (slash < rem) ? (const char *) (authority + slash) : "";
144 path_len = (slash < rem) ? rem - slash : 0;
145
146 if (vfs_parse_s3_port(cf, authority, authority_len, out) != NGX_OK) {
147 return NGX_ERROR;
148 }
149 host_len = out->host_len;
150 while (path_len > 0 && path[0] == '/') {
151 path++;
152 path_len--;
153 }
154 if (host_len == 0 || host_len >= out->host_cap || path_len == 0
155 || path_len >= out->base_cap)
156 {
157 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
158 "brix_storage_backend: s3:// needs \"//host[:port]/bucket\"");
159 return NGX_ERROR;
160 }
161
162 ngx_memcpy(out->host, authority, host_len);
163 out->host[host_len] = '\0';
164 ngx_memcpy(out->base, path, path_len);
165 out->base[path_len] = '\0';
166 return NGX_OK;
167}

169/* "s3://host[:port]/bucket" → a read-only S3 source backend. Returns NGX_OK on a
170 * handled segment, NGX_ERROR after an [emerg] for a malformed one, or
171 * NGX_DECLINED when the scheme is not ours. */
172ngx_int_t
173vfs_parse_s3_origin(ngx_conf_t *cf, const char *root_canon, const ngx_str_t *sb)
174{
175 char host[256], bucket[256];
176 vfs_origin_parse_t parsed;
177
178 /* "s3://host[:port]/bucket" → a read-only S3 source backend (path-style): the
179 * first path segment is the bucket. Default port 7480 (radosgw), tls off — the
180 * same defaults the generic tier parser uses for an s3 store. */
181 if (sb->len < sizeof("s3://") - 1
182 || ngx_strncmp(sb->data, "s3://", sizeof("s3://") - 1) != 0)
183 {
184 return NGX_DECLINED;
185 }
186
187 ngx_memzero(&parsed, sizeof(parsed));
188 parsed.host = host;
189 parsed.host_cap = sizeof(host);
190 parsed.base = bucket; /* origin_path carries the bucket */
191 parsed.base_cap = sizeof(bucket);
192
193 if (vfs_parse_s3_url(cf, sb, &parsed) != NGX_OK) {
194 return NGX_ERROR;
195 }
196
197 brix_vfs_backend_config_s3(root_canon, parsed.host, parsed.port, 0,
198 parsed.base);
199 return NGX_OK;
200
201}

203/* "root://host:port" / "roots://host:port" → a remote root:// primary backend;
204 * any other value is a local driver name (pblock/posix) handled by
205 * brix_vfs_backend_config. Returns NGX_OK, or NGX_ERROR after an [emerg] for a
206 * malformed remote origin. */
207ngx_int_t
208vfs_parse_xroot_or_driver_origin(ngx_conf_t *cf, const char *root_canon,
209 const ngx_str_t *sb, size_t block_size, int family)
210{
211 u_char *addr = NULL;
212 size_t addrn = 0;
213 int is_roots = 0;
214 /* "root://host:port" / "roots://host:port" → a remote root:// primary backend;
215 * any other value is a local driver name (pblock/posix) handled as before. */
216 if (sb->len > sizeof("roots://") - 1
217 && ngx_strncmp(sb->data, "roots://", sizeof("roots://") - 1) == 0)
218 {
219 addr = sb->data + sizeof("roots://") - 1;
220 addrn = sb->len - (sizeof("roots://") - 1);
221 is_roots = 1;
222 } else if (sb->len > sizeof("root://") - 1
223 && ngx_strncmp(sb->data, "root://", sizeof("root://") - 1) == 0)
224 {
225 addr = sb->data + sizeof("root://") - 1;
226 addrn = sb->len - (sizeof("root://") - 1);
227 }
228
229 if (addr == NULL) {
230 brix_vfs_backend_config(root_canon, sb, block_size);
231 return NGX_OK;
232 }
233
234 {
235 u_char *colon = NULL;
236 size_t i, hostn;
237 ngx_int_t portnum;
238 char host[256];
239
240 /* Split host:port on the LAST colon (a bracketed [v6]:port keeps it). */
241 for (i = addrn; i > 0; i--) {
242 if (addr[i - 1] == ':') { colon = addr + i - 1; break; }
243 }
244 if (colon == NULL) {
245 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
246 "brix_storage_backend: remote origin needs host:port");
247 return NGX_ERROR;
248 }
249 hostn = (size_t) (colon - addr);
250 portnum = ngx_atoi(colon + 1, (size_t) (addr + addrn - (colon + 1)));
251 if (hostn == 0 || hostn >= sizeof(host) || portnum == NGX_ERROR
252 || portnum <= 0 || portnum > 65535)
253 {
254 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
255 "brix_storage_backend: invalid remote origin host:port");
256 return NGX_ERROR;
257 }
258 ngx_memcpy(host, addr, hostn);
259 host[hostn] = '\0';
260 brix_vfs_backend_config_xroot(root_canon, host, (int) portnum,
261 is_roots, family);
262 }
263
264 return NGX_OK;
265}
