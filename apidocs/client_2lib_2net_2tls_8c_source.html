1/*
2 * tls.c — in-protocol TLS for the native client (roots:// and GSI+TLS).
3 *
4 * WHAT: Upgrade the live socket to a TLS session (after the kXR_protocol reply,
5 * before kXR_login) and transfer bytes transparently through it. Server
6 * certificate verification against a CA hash dir ($X509_CERT_DIR) is on by
7 * default; a name-check escape (--noverifyhost) relaxes only the hostname.
8 * WHY: XRootD encrypts the control/data stream in-protocol; the same SSL session
9 * then carries login, GSI auth, and all file ops.
10 * HOW: The fd goes non-blocking at upgrade time; SSL_connect and the read/write
11 * loops drive OpenSSL's WANT_READ/WANT_WRITE through the same poll(2) +
12 * timeout discipline the cleartext path uses. sock.c branches to these when
13 * io->ssl != NULL, so no other file changes for TLS.
14 */

42/* poll one fd for `events`; >0 ready, 0 timeout, <0 error (EINTR retried). */
43static int
44wait_io(int fd, short events, int timeout_ms)
45{
46 struct pollfd pfd;
47 int pr;
48
49 pfd.fd = fd;
50 pfd.events = events;
51 do {
52 pr = poll(&pfd, 1, timeout_ms);
53 } while (pr < 0 && errno == EINTR && !brix_copy_quit_requested());
54 return pr;
55}

57/* Clamp the idle timeout to a whole-operation deadline; see read_wait_ms in
58 * sock.c for the rationale (slow-drip guard). -1 = deadline passed. */
59static int
60tls_read_wait_ms(int timeout_ms, uint64_t deadline_ns)
61{
62 if (deadline_ns == 0) {
63 return timeout_ms;
64 }
65 {
66 uint64_t now = brix_mono_ns();
67 int64_t rem_ms;
68
69 if (now >= deadline_ns) {
70 return -1;
71 }
72 rem_ms = (int64_t) ((deadline_ns - now) / 1000000ULL);
73 if (rem_ms <= 0) {
74 return -1;
75 }
76 if (timeout_ms > 0 && (int64_t) timeout_ms < rem_ms) {
77 return timeout_ms;
78 }
79 return (int) rem_ms;
80 }
81}

83int
84brix_tls_read(brix_io *io, void *buf, size_t n, brix_status *st)
85{
86 SSL *ssl = (SSL *) io->ssl;
87 uint8_t *p = (uint8_t *) buf;
88 size_t got = 0;
89 /* Shared absolute cutoff for the whole logical operation (armed once by
90 * brix_io_stall_arm), so a multi-record TLS read shares one budget rather
91 * than re-arming per record. 0 = disabled. */
92 uint64_t deadline_ns = io->stall_deadline_ns;
93
94 while (got < n) {
95 size_t nread = 0;
96 int ok, err;
97
98 if (brix_copy_quit_requested()) { /* Phase 40 (a): prompt cancel */
99 brix_status_set(st, XRDC_ESOCK, EINTR, "transfer cancelled (signal)");
100 return -1;
101 }
102 ERR_clear_error(); /* so SSL_get_error reflects THIS op, not a stale queue */
103 ok = SSL_read_ex(ssl, p + got, n - got, &nread);
104
105 if (ok == 1) {
106 got += nread;
107 continue;
108 }
109 err = SSL_get_error(ssl, 0);
110 if (err == SSL_ERROR_WANT_READ || err == SSL_ERROR_WANT_WRITE) {
111 short ev = (err == SSL_ERROR_WANT_READ) ? POLLIN : POLLOUT;
112 int wait_ms = tls_read_wait_ms(io->timeout_ms, deadline_ns);
113 int pr;
114
115 if (wait_ms < 0) {
116 brix_status_set(st, XRDC_ESOCK, ETIMEDOUT,
117 "TLS read exceeded slow-drip deadline "
118 "(%d ms, got %zu/%zu bytes)",
119 io->stall_deadline_ms, got, n);
120 return -1;
121 }
122 pr = wait_io(io->fd, ev, wait_ms);
123 if (pr == 0) {
124 if (deadline_ns != 0 && brix_mono_ns() >= deadline_ns) {
125 brix_status_set(st, XRDC_ESOCK, ETIMEDOUT,
126 "TLS read exceeded slow-drip deadline "
127 "(%d ms, got %zu/%zu bytes)",
128 io->stall_deadline_ms, got, n);
129 return -1;
130 }
131 brix_status_set(st, XRDC_ESOCK, ETIMEDOUT, "TLS read timed out");
132 return -1;
133 }
134 if (pr < 0) {
135 brix_status_set(st, XRDC_ESOCK, errno, "poll(tls read): %s",
136 strerror(errno));
137 return -1;
138 }
139 continue;
140 }
141 if (err == SSL_ERROR_ZERO_RETURN) {
142 brix_status_set(st, XRDC_ESOCK, 0,
143 "TLS connection closed by peer (read %zu/%zu)", got, n);
144 return -1;
145 }
146 if (err == SSL_ERROR_SYSCALL) {
147 if (errno == EINTR || errno == EAGAIN) {
148 continue;
149 }
150 brix_status_set(st, XRDC_ESOCK, errno, "TLS read: %s",
151 errno ? strerror(errno) : "unexpected EOF");
152 return -1;
153 }
154 tls_err(st, XRDC_ESOCK, "TLS read");
155 return -1;
156 }
157 return 0;
158}

211/* Drive a non-blocking SSL_connect to completion (poll on WANT_*). 0 / -1. */
212static int
213tls_do_connect(SSL *ssl, brix_io *io, int verify_peer, brix_status *st)
214{
215 for (;;) {
216 int ret, err;
217
218 ERR_clear_error();
219 ret = SSL_connect(ssl);
220 if (ret == 1) {
221 break;
222 }
223 err = SSL_get_error(ssl, ret);
224 if (err == SSL_ERROR_WANT_READ || err == SSL_ERROR_WANT_WRITE) {
225 short ev = (err == SSL_ERROR_WANT_READ) ? POLLIN : POLLOUT;
226 if (wait_io(io->fd, ev, io->timeout_ms) <= 0) {
227 brix_status_set(st, XRDC_ESOCK, 0, "TLS handshake timed out");
228 return -1;
229 }
230 continue;
231 }
232 {
233 long vr = SSL_get_verify_result(ssl);
234 if (vr != X509_V_OK) {
235 brix_status_set(st, XRDC_EAUTH, 0, "TLS verify failed: %s",
236 X509_verify_cert_error_string(vr));
237 } else {
238 tls_err(st, XRDC_ESOCK, "SSL_connect");
239 }
240 }
241 return -1;
242 }
243
244 if (verify_peer) {
245 long vr = SSL_get_verify_result(ssl);
246 if (vr != X509_V_OK) {
247 brix_status_set(st, XRDC_EAUTH, 0, "TLS verify failed: %s",
248 X509_verify_cert_error_string(vr));
249 return -1;
250 }
251 }
252 return 0;
253}

255int
256brix_tls_upgrade(brix_conn *c, int verify_peer, int verify_host,
257 const char *ca_dir, brix_status *st)
258{
259 SSL_CTX *ctx;
260 SSL *ssl;
261 int fl;
262
263 ctx = SSL_CTX_new(TLS_client_method());
264 if (ctx == NULL) {
265 tls_err(st, XRDC_ESOCK, "SSL_CTX_new");
266 return -1;
267 }
268 SSL_CTX_set_min_proto_version(ctx, TLS1_2_VERSION);
269
270 if (verify_peer) {
271 SSL_CTX_set_verify(ctx, SSL_VERIFY_PEER, NULL);
272 if (ca_dir != NULL && ca_dir[0] != '\0') {
273 if (SSL_CTX_load_verify_locations(ctx, NULL, ca_dir) != 1) {
274 tls_err(st, XRDC_EAUTH, "load CA dir");
275 SSL_CTX_free(ctx);
276 return -1;
277 }
278 } else {
279 SSL_CTX_set_default_verify_paths(ctx);
280 }
281 /* Permit proxy certs in the chain (GSI deployments). */
282 X509_VERIFY_PARAM_set_flags(SSL_CTX_get0_param(ctx),
283 X509_V_FLAG_ALLOW_PROXY_CERTS);
284 } else {
285 SSL_CTX_set_verify(ctx, SSL_VERIFY_NONE, NULL);
286 }
287
288 ssl = SSL_new(ctx);
289 if (ssl == NULL) {
290 tls_err(st, XRDC_ESOCK, "SSL_new");
291 SSL_CTX_free(ctx);
292 return -1;
293 }
294
295 if (verify_peer && verify_host && c->host[0] != '\0') {
296 SSL_set1_host(ssl, c->host);
297 }
298
299 /* Non-blocking for poll-driven handshake/IO; SSL handles record framing. */
300 fl = fcntl(c->io.fd, F_GETFL, 0);
301 if (fl >= 0) {
302 (void) fcntl(c->io.fd, F_SETFL, fl | O_NONBLOCK);
303 }
304 SSL_set_fd(ssl, c->io.fd);
305 SSL_set_connect_state(ssl);
306
307 if (tls_do_connect(ssl, &c->io, verify_peer, st) != 0) {
308 SSL_free(ssl);
309 SSL_CTX_free(ctx);
310 return -1;
311 }
312
313 c->io.ssl = ssl; /* struct ssl_st* == SSL* */
314 c->ssl_ctx = ctx;
315 return 0;
316}

318/*
319 * Read UP TO n bytes (one record) over TLS — unlike brix_tls_read (which fills
320 * exactly n for root:// framing), this is a stream read for the HTTP client: sets
321 * *got to the bytes read (0 = clean EOF / peer close) and returns 0, or -1 on error.
322 */
323int
324brix_tls_read_some(brix_io *io, void *buf, size_t n, size_t *got, brix_status *st)
325{
326 SSL *ssl = (SSL *) io->ssl;
327 size_t nread = 0;
328
329 *got = 0;
330 for (;;) {
331 int ok, err;
332 ERR_clear_error();
333 ok = SSL_read_ex(ssl, buf, n, &nread);
334 if (ok == 1) {
335 *got = nread;
336 return 0;
337 }
338 err = SSL_get_error(ssl, 0);
339 if (err == SSL_ERROR_ZERO_RETURN) {
340 return 0; /* clean close → *got stays 0 (EOF) */
341 }
342 if (err == SSL_ERROR_WANT_READ || err == SSL_ERROR_WANT_WRITE) {
343 short ev = (err == SSL_ERROR_WANT_READ) ? POLLIN : POLLOUT;
344 int pr = wait_io(io->fd, ev, io->timeout_ms);
345 if (pr <= 0) {
346 brix_status_set(st, XRDC_ESOCK, pr == 0 ? ETIMEDOUT : errno,
347 "TLS read %s", pr == 0 ? "timed out" : "poll failed");
348 return -1;
349 }
350 continue;
351 }
352 if (err == SSL_ERROR_SYSCALL && errno == 0) {
353 return 0; /* unclean EOF — treat as end of body */
354 }
355 if (err == SSL_ERROR_SYSCALL && (errno == EINTR || errno == EAGAIN)) {
356 continue;
357 }
358 tls_err(st, XRDC_ESOCK, "TLS read");
359 return -1;
360 }
361}

390 tls_err(st, XRDC_EAUTH, "load CA dir");
391 SSL_CTX_free(ctx);
392 return -1;
393 }
394 } else {
395 SSL_CTX_set_default_verify_paths(ctx);
396 }

411 if (verify_peer && verify_host) {

421
422 if (tls_do_connect(ssl, io, verify_peer, st) != 0) {

424 SSL_CTX_free(ctx);

426 }

428 *out_ctx = ctx;
429 return 0;

488/* Join the cert's DNS subjectAltNames into out[] as a comma-separated list. */
489static void
490peer_collect_sans(X509 *cert, char *out, size_t outsz)
491{
492 GENERAL_NAMES *gens;
493 int i, n;
494 size_t off = 0;
495
496 out[0] = '\0';
497 gens = (GENERAL_NAMES *) X509_get_ext_d2i(cert, NID_subject_alt_name, NULL, NULL);
498 if (gens == NULL) {
499 return;
500 }
501 n = sk_GENERAL_NAME_num(gens);
502 for (i = 0; i < n; i++) {
503 GENERAL_NAME *g = sk_GENERAL_NAME_value(gens, i);
504 const char *dns;
505 int len;
506 if (g->type != GEN_DNS) {
507 continue;
508 }
509 dns = (const char *) ASN1_STRING_get0_data(g->d.dNSName);
510 len = ASN1_STRING_length(g->d.dNSName);
511 if (len <= 0 || off + (size_t) len + 2 >= outsz) {
512 continue;
513 }
514 if (off > 0) { out[off++] = ','; }
515 memcpy(out + off, dns, (size_t) len);
516 off += (size_t) len;
517 out[off] = '\0';
518 }
519 GENERAL_NAMES_free(gens);
520}

522int
523brix_tls_peer_cert_info(const brix_conn *c, brix_cert_info *out)
524{
525 SSL *ssl = (SSL *) c->io.ssl;
526 X509 *cert;
527 const ASN1_TIME *nb, *na;
528 struct tm tmv;
529 time_t now = time(NULL);
530 int dd = 0, ds = 0;
531
532 memset(out, 0, sizeof(*out));
533 if (ssl == NULL) {
534 return -1; /* cleartext: no peer cert */
535 }
536 cert = SSL_get_peer_certificate(ssl); /* bumps refcount; free below */
537 if (cert == NULL) {
538 return -1;
539 }
540 out->have = 1;
541 X509_NAME_oneline(X509_get_subject_name(cert), out->subject, sizeof(out->subject));
542 X509_NAME_oneline(X509_get_issuer_name(cert), out->issuer, sizeof(out->issuer));
543 out->self_signed = (X509_check_issued(cert, cert) == X509_V_OK);
544 out->host_match = (c->host[0] != '\0'
545 && X509_check_host(cert, c->host, 0, 0, NULL) == 1);
546
547 nb = X509_get0_notBefore(cert);
548 na = X509_get0_notAfter(cert);
549 if (ASN1_TIME_to_tm(nb, &tmv)) { out->not_before = (long) timegm(&tmv); }
550 if (ASN1_TIME_to_tm(na, &tmv)) { out->not_after = (long) timegm(&tmv); }
551 out->expired = (out->not_after != 0 && out->not_after < (long) now);
552 out->not_yet_valid = (out->not_before != 0 && out->not_before > (long) now);
553 if (ASN1_TIME_diff(&dd, &ds, NULL, na)) {
554 out->days_left = dd; /* whole days; negative once expired */
555 }
556 peer_collect_sans(cert, out->sans, sizeof(out->sans));
557
558 X509_free(cert);
559 return 0;
560}
