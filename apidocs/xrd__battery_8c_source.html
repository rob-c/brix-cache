22/* Append a result. status: >0 pass, 0 fail, <0 skipped. */
23void
24bat_add(xrd_battery *b, const char *name, int status, const char *fmt, ...)
25{
26 xrd_check *c;
27 va_list ap;
28 if (b->n >= XRD_MAX_CHECKS) { return; }
29 c = &b->checks[b->n++];
30 snprintf(c->name, sizeof(c->name), "%s", name);
31 c->ok = (status > 0);
32 c->skipped = (status < 0);
33 va_start(ap, fmt);
34 vsnprintf(c->detail, sizeof(c->detail), fmt, ap);
35 va_end(ap);
36 if (status < 0) { b->nskip++; }
37 else if (status > 0) { b->npass++; }
38 else { b->nfail++; }
39}

42/* Fill `buf` (size n) with a deterministic, position-dependent pattern. */
43void
44fill_pattern(uint8_t *buf, size_t n)
45{
46 size_t i;
47 for (i = 0; i < n; i++) { buf[i] = (uint8_t) ((i * 7u + 3u) & 0xff); }
48}

51/* Write `buf` to an anonymous tmpfile and return its fd (rewound), or -1. */
52int
53tmpfile_with(const uint8_t *buf, size_t n)
54{
55 FILE *f = tmpfile();
56 int fd;
57 if (f == NULL) { return -1; }
58 if (n > 0 && fwrite(buf, 1, n, f) != n) { fclose(f); return -1; }
59 fflush(f);
60 fd = dup(fileno(f));
61 fclose(f);
62 if (fd >= 0) { lseek(fd, 0, SEEK_SET); }
63 return fd;
64}

67/* fd-backed pull source for brix_http_upload: the battery's body is an anonymous
68 * tmpfile (generated diagnostic payload, not export storage), so a plain pread by
69 * offset is the source — storage callers pass a VFS-backed source instead. */
70static ssize_t
71bat_upload_src_fd(void *ctx, uint8_t *buf, int64_t off, size_t cap, brix_status *st)
72{
73 ssize_t r = pread(*(int *) ctx, buf, cap, (off_t) off); /* vfs-seam-allow: anonymous tmpfile diagnostic payload, not export storage */
74 if (r < 0) {
75 brix_status_set(st, XRDC_ESOCK, errno, "pread: %s", strerror(errno));
76 }
77 return r;
78}

81/* The native root:// functional battery: always-safe reads, then (do_write) a full
82 * write/read/verify/checksum/metadata cycle under a temp dir that is cleaned up. */
83void
84battery_root(const brix_url *u, const brix_opts *o, int do_write, xrd_battery *b)
85{
86 brix_conn c;
87 brix_status st;
88 brix_statinfo si;
89 brix_dirent *ents = NULL;
90 size_t nents = 0;
91 char reply[1024];
92 int ext_sa = 0, ext_sl = 0, ext_rl = 0, ext_ln = 0;
93
94 snprintf(b->protocol, sizeof(b->protocol), "root");
95 brix_status_clear(&st);
96 if (brix_connect(&c, u, o, &st) != 0) {
97 snprintf(b->err, sizeof(b->err), "%s", st.msg);
98 return;
99 }
100 b->reachable = 1;
101
102 /* read-only methods */ brix_status_clear(&st);
103 if (brix_stat(&c, "/", &si, &st) == 0) { bat_add(b, "stat", 1, "/ flags=0x%x", si.flags); }
104 else { bat_add(b, "stat", 0, "%s", st.msg); }
105
106 brix_status_clear(&st);
107 if (brix_dirlist(&c, "/", 0, &ents, &nents, &st) == 0) {
108 bat_add(b, "dirlist", 1, "%zu entries", nents);
109 free(ents); ents = NULL;
110 } else { bat_add(b, "dirlist", 0, "%s", st.msg); }
111
112 brix_status_clear(&st);
113 if (brix_statvfs(&c, "/", reply, sizeof(reply), &st) == 0) { bat_add(b, "statvfs", 1, "ok"); }
114 else { bat_add(b, "statvfs", 0, "%s", st.msg); }
115
116 brix_status_clear(&st);
117 if (brix_query(&c, kXR_Qconfig, "chksum", reply, sizeof(reply), &st) == 0) {
118 char *nl = strchr(reply, '\n'); if (nl) { *nl = '\0'; }
119 bat_add(b, "query-config", 1, "%s", reply);
120 } else { bat_add(b, "query-config", 0, "%s", st.msg); }
121
122 /* negative: a traversal path must not resolve outside the export */
123 brix_status_clear(&st);
124 {
125 int rc = brix_stat(&c, "/../../../../etc/passwd", &si, &st);
126 bat_add(b, "path-confinement", rc != 0 ? 1 : 0,
127 rc != 0 ? "escape rejected" : "LEAKED /etc/passwd");
128 }
129
130 (void) brix_ext_probe(&c, &ext_sa, &ext_sl, &ext_rl, &ext_ln, &st);
131
132 if (!do_write) {
133 bat_add(b, "write-suite", -1, "skipped (pass --rw to run write tests)");
134 brix_close(&c);
135 return;
136 }
137
138 /* read/write cycle under a temp dir */ {
139 char dir[128], file[200], file2[200];
140 uint8_t payload[8192], rbuf[8192];
141 char srvck[160], locck[160];
142 long pid = (long) getpid();
143 int ok, rc, sym_left = 0;
144
145 snprintf(dir, sizeof(dir), "/.xrd_doctor_%ld", pid);
146 snprintf(file, sizeof(file), "%s/probe.bin", dir);
147 snprintf(file2, sizeof(file2), "%s/probe.moved.bin", dir);
148 fill_pattern(payload, sizeof(payload));
149
150 brix_status_clear(&st);
151 rc = brix_mkdir(&c, dir, 0755, 1, &st);
152 bat_add(b, "mkdir", rc == 0 ? 1 : 0, "%s", rc == 0 ? dir : st.msg);
153
154 /* write */
155 {
156 brix_file f;
157 ok = 0;
158 brix_status_clear(&st);
159 if (brix_file_open_write(&c, file, 1, 0, &f, &st) == 0) {
160 ok = (brix_file_write(&c, &f, 0, payload, sizeof(payload), &st) == 0);
161 brix_file_close(&c, &f, &st);
162 }
163 bat_add(b, "write", ok ? 1 : 0, ok ? "%zu bytes" : "%s",
164 ok ? sizeof(payload) : (size_t) 0, ok ? "" : st.msg);
165 }
166 /* read-back + byte-exact verify */
167 {
168 brix_file f;
169 ssize_t got = -1;
170 int match = 0;
171 brix_status_clear(&st);
172 if (brix_file_open_read(&c, file, &f, &st) == 0) {
173 got = brix_file_read(&c, &f, 0, rbuf, sizeof(rbuf), &st);
174 brix_file_close(&c, &f, &st);
175 match = (got == (ssize_t) sizeof(payload)
176 && memcmp(rbuf, payload, sizeof(payload)) == 0);
177 }
178 bat_add(b, "read-verify", match ? 1 : 0,
179 match ? "byte-exact %zd bytes" : "mismatch/short (%s)",
180 match ? got : 0, match ? "" : st.msg);
181 }
182 /* readv (two segments) */
183 {
184 brix_file f;
185 brix_readv_seg segs[2];
186 int match = 0;
187 uint8_t s0[64], s1[128];
188 segs[0].offset = 0; segs[0].len = sizeof(s0); segs[0].buf = s0; segs[0].got = 0;
189 segs[1].offset = 1000; segs[1].len = sizeof(s1); segs[1].buf = s1; segs[1].got = 0;
190 brix_status_clear(&st);
191 if (brix_file_open_read(&c, file, &f, &st) == 0) {
192 if (brix_file_readv(&c, &f, segs, 2, &st) >= 0) {
193 match = (memcmp(s0, payload, sizeof(s0)) == 0
194 && memcmp(s1, payload + 1000, sizeof(s1)) == 0);
195 }
196 brix_file_close(&c, &f, &st);
197 }
198 bat_add(b, "readv", match ? 1 : 0, match ? "2 segs verified" : "%s", st.msg);
199 }
200 /* checksum: server adler32 vs locally computed adler32 of the payload */
201 {
202 int fd = tmpfile_with(payload, sizeof(payload));
203 int verified = 0;
204 brix_status_clear(&st);
205 if (fd >= 0
206 && brix_cksum_fd(fd, XRDC_CK_ADLER32, locck, sizeof(locck), &st) == 0
207 && brix_query_cksum(&c, file, "adler32", srvck, sizeof(srvck), &st) == 0) {
208 verified = (strcmp(locck, srvck) == 0);
209 }
210 if (fd >= 0) { close(fd); }
211 bat_add(b, "checksum-verify", verified ? 1 : 0,
212 verified ? "adler32 %s matches" : "server/local differ or n/a",
213 verified ? srvck : "");
214 }
215 /* setattr times (xrdfs.ext) */
216 if (ext_sa) {
217 struct timespec ts[2];
218 ts[0].tv_sec = ts[1].tv_sec = 0;
219 ts[0].tv_nsec = ts[1].tv_nsec = UTIME_NOW;
220 brix_status_clear(&st);
221 rc = brix_setattr(&c, file, 1, ts, 0, (uint32_t) -1, (uint32_t) -1, &st);
222 bat_add(b, "setattr-times", rc == 0 ? 1 : 0, "%s", rc == 0 ? "mtime set" : st.msg);
223 } else { bat_add(b, "setattr-times", -1, "server lacks xrdfs.ext"); }
224 /* xattr set/get/del (xrdfs.ext fattr is always present, but gate on a probe) */
225 {
226 char val[64]; size_t vlen = 0;
227 int okset, okget, okdel;
228 brix_status_clear(&st);
229 okset = (brix_fattr_set(&c, file, "doctor", "ok", 2, 0, &st) == 0);
230 okget = okset && (brix_fattr_get(&c, file, "doctor", val, sizeof(val), &vlen, &st) == 0
231 && vlen == 2 && memcmp(val, "ok", 2) == 0);
232 okdel = okget && (brix_fattr_del(&c, file, "doctor", &st) == 0);
233 bat_add(b, "xattr", okdel ? 1 : (okset ? 0 : -1),
234 okdel ? "set/get/del roundtrip" : (okset ? "%s" : "not supported"),
235 okdel ? "" : st.msg);
236 }
237 /* symlink + readlink (xrdfs.ext). Note: if the server's rm resolves through
238 * the final symlink it cannot unlink the link itself — we detect that and skip
239 * the dir's rmdir rather than report a phantom "not empty" failure. */
240 if (ext_sl && ext_rl) {
241 char lp[200], tgt[256];
242 ssize_t rl;
243 int made;
244 snprintf(lp, sizeof(lp), "%s/probe.link", dir);
245 brix_status_clear(&st);
246 made = (brix_symlink(&c, file, lp, &st) == 0);
247 if (made && (rl = brix_readlink(&c, lp, tgt, sizeof(tgt), &st)) > 0
248 && strcmp(tgt, file) == 0) {
249 brix_status rs;
250 brix_status_clear(&rs);
251 if (brix_rm(&c, lp, &rs) != 0) {
252 sym_left = 1;
253 bat_add(b, "symlink+readlink", 1,
254 "create+readlink ok; unlink unsupported (rm follows the link)");
255 } else {
256 bat_add(b, "symlink+readlink", 1, "create/readlink/unlink ok");
257 }
258 } else {
259 bat_add(b, "symlink+readlink", made ? 0 : 0, "%s", st.msg);
260 if (made) { brix_status rs; brix_status_clear(&rs);
261 if (brix_rm(&c, lp, &rs) != 0) { sym_left = 1; } }
262 }
263 } else { bat_add(b, "symlink+readlink", -1, "server lacks xrdfs.ext"); }
264 /* rename */
265 brix_status_clear(&st);
266 rc = brix_mv(&c, file, file2, &st);
267 bat_add(b, "rename", rc == 0 ? 1 : 0, "%s", rc == 0 ? "moved" : st.msg);
268 /* truncate */
269 brix_status_clear(&st);
270 rc = brix_truncate(&c, file2, 10, &st);
271 bat_add(b, "truncate", rc == 0 ? 1 : 0, "%s", rc == 0 ? "to 10 bytes" : st.msg);
272 /* rm the file, then rmdir the now-empty temp dir (cleanup) */
273 brix_status_clear(&st);
274 rc = brix_rm(&c, file2, &st);
275 bat_add(b, "rm", rc == 0 ? 1 : 0, "%s", rc == 0 ? "removed" : st.msg);
276 if (sym_left) {
277 bat_add(b, "rmdir", -1,
278 "skipped: temp dir retains a symlink the server cannot unlink");
279 } else {
280 brix_status_clear(&st);
281 rc = brix_rmdir(&c, dir, &st);
282 bat_add(b, "rmdir", rc == 0 ? 1 : 0, "%s", rc == 0 ? "removed" : st.msg);
283 }
284 }
285 brix_close(&c);
286}

289/* The WebDAV/HTTP functional battery: OPTIONS + PROPFIND (read), then (do_write) a
290 * MKCOL/PUT/GET-verify/PROPFIND/MOVE/DELETE cycle under a temp collection. bearer NULL
291 * = anonymous. */
292void
293battery_web(const brix_weburl *u, int do_write, const char *bearer, int verify,
294 xrd_battery *b)
295{
296 brix_http_resp resp;
297 brix_status st;
298 const char *ca = brix_resolve_ca_dir(NULL);
299 char authhdr[1200];
300 const char *xtra = NULL;
301
302 snprintf(b->protocol, sizeof(b->protocol), "%s", u->tls ? "https" : "http");
303 if (bearer != NULL && bearer[0] != '\0') {
304 snprintf(authhdr, sizeof(authhdr), "Authorization: Bearer %s\r\n", bearer);
305 xtra = authhdr;
306 }
307
308 brix_status_clear(&st);
309 if (brix_http_req(u->host, u->port, u->tls, "OPTIONS", "/", xtra, NULL, 0,
310 5000, verify, ca, &resp, &st) != 0) {
311 snprintf(b->err, sizeof(b->err), "%s", st.msg);
312 bat_add(b, "OPTIONS", 0, "%s", st.msg);
313 return;
314 }
315 b->reachable = 1;
316 {
317 char dav[160] = "";
318 brix_http_header(&resp, "DAV", dav, sizeof(dav));
319 bat_add(b, "OPTIONS", (resp.status >= 200 && resp.status < 500) ? 1 : 0,
320 "HTTP %d%s%s", resp.status, dav[0] ? " DAV=" : "", dav);
321 }
322 brix_http_resp_free(&resp);
323
324 {
325 const char *body = "<?xml version=\"1.0\"?><propfind xmlns=\"DAV:\"><allprop/></propfind>";
326 char hdr[1400];
327 snprintf(hdr, sizeof(hdr), "Depth: 0\r\nContent-Type: application/xml\r\n%s",
328 xtra ? xtra : "");
329 brix_status_clear(&st);
330 if (brix_http_req(u->host, u->port, u->tls, "PROPFIND", "/", hdr, body,
331 strlen(body), 5000, verify, ca, &resp, &st) == 0) {
332 bat_add(b, "PROPFIND", resp.status == 207 ? 1 : 0, "HTTP %d", resp.status);
333 brix_http_resp_free(&resp);
334 } else { bat_add(b, "PROPFIND", 0, "%s", st.msg); }
335 }
336
337 if (!do_write) { bat_add(b, "write-suite", -1, "skipped (pass --rw)"); return; }
338
339 {
340 char dir[160], fpath[256], mpath[256], dst[2048];
341 uint8_t payload[4096], rbuf[4096];
342 long pid = (long) getpid();
343 int fd, st_code = 0, ok;
344 long long blen = 0;
345
346 snprintf(dir, sizeof(dir), "/.xrd_doctor_%ld/", pid);
347 snprintf(fpath, sizeof(fpath), "%.150sprobe.bin", dir);
348 snprintf(mpath, sizeof(mpath), "%.150sprobe.moved.bin", dir);
349 fill_pattern(payload, sizeof(payload));
350
351 /* MKCOL */
352 brix_status_clear(&st);
353 if (brix_http_req(u->host, u->port, u->tls, "MKCOL", dir, xtra, NULL, 0,
354 5000, verify, ca, &resp, &st) == 0) {
355 bat_add(b, "MKCOL", (resp.status == 201 || resp.status == 405) ? 1 : 0,
356 "HTTP %d", resp.status);
357 brix_http_resp_free(&resp);
358 } else { bat_add(b, "MKCOL", 0, "%s", st.msg); }
359
360 /* PUT */
361 fd = tmpfile_with(payload, sizeof(payload));
362 brix_status_clear(&st);
363 if (fd >= 0 && brix_http_upload(u->host, u->port, u->tls, fpath, xtra,
364 bat_upload_src_fd, &fd,
365 (long long) sizeof(payload), verify, ca, 10000,
366 &st_code, &st) == 0) {
367 bat_add(b, "PUT", (st_code >= 200 && st_code < 300) ? 1 : 0, "HTTP %d", st_code);
368 } else { bat_add(b, "PUT", 0, "%s", st.msg); }
369 if (fd >= 0) { close(fd); }
370
371 /* GET + byte-exact verify */
372 fd = tmpfile_with(NULL, 0);
373 brix_status_clear(&st);
374 if (fd >= 0 && brix_http_download(u->host, u->port, u->tls, fpath, xtra, verify,
375 ca, fd, 10000, &st_code, &blen, &st) == 0
376 && blen == (long long) sizeof(payload)) {
377 lseek(fd, 0, SEEK_SET);
378 ok = (read(fd, rbuf, sizeof(rbuf)) == (ssize_t) sizeof(payload)
379 && memcmp(rbuf, payload, sizeof(payload)) == 0);
380 bat_add(b, "GET-verify", ok ? 1 : 0, ok ? "byte-exact %lld" : "mismatch", blen);
381 } else { bat_add(b, "GET-verify", 0, "HTTP %d %s", st_code, st.msg); }
382 if (fd >= 0) { close(fd); }
383
384 /* MOVE */
385 snprintf(dst, sizeof(dst), "Destination: %s://%s:%d%s\r\n%s",
386 u->tls ? "https" : "http", u->host, u->port, mpath, xtra ? xtra : "");
387 brix_status_clear(&st);
388 if (brix_http_req(u->host, u->port, u->tls, "MOVE", fpath, dst, NULL, 0,
389 5000, verify, ca, &resp, &st) == 0) {
390 bat_add(b, "MOVE", (resp.status >= 200 && resp.status < 300) ? 1 : 0,
391 "HTTP %d", resp.status);
392 brix_http_resp_free(&resp);
393 } else { bat_add(b, "MOVE", 0, "%s", st.msg); }
394
395 /* DELETE the (moved) file and the collection */
396 brix_status_clear(&st);
397 if (brix_http_req(u->host, u->port, u->tls, "DELETE", mpath, xtra, NULL, 0,
398 5000, verify, ca, &resp, &st) == 0) {
399 bat_add(b, "DELETE", (resp.status >= 200 && resp.status < 300) ? 1 : 0,
400 "HTTP %d", resp.status);
401 brix_http_resp_free(&resp);
402 } else { bat_add(b, "DELETE", 0, "%s", st.msg); }
403 { brix_status rs; brix_status_clear(&rs);
404 if (brix_http_req(u->host, u->port, u->tls, "DELETE", dir, xtra, NULL, 0,
405 5000, verify, ca, &resp, &rs) == 0) { brix_http_resp_free(&resp); } }
406 }
407}

410/* The S3 functional battery: ListObjectsV2 (read), then (do_write) a SigV4-signed
411 * PUT/GET-verify/DELETE of a temp object. ak/sk NULL = anonymous (writes skipped). */
412void
413battery_s3(const brix_weburl *u, int do_write, const char *ak, const char *sk,
414 const char *region, int verify, xrd_battery *b)
415{
416 brix_status st;
417 const char *ca = brix_resolve_ca_dir(NULL);
418 char **keys = NULL;
419 size_t nk = 0;
420
421 snprintf(b->protocol, sizeof(b->protocol), "s3");
422 brix_status_clear(&st);
423 if (brix_s3_list(u, ak, sk, region, verify, ca, &keys, &nk, &st) != 0) {
424 snprintf(b->err, sizeof(b->err), "%s", st.msg);
425 bat_add(b, "list-objects", 0, "%s", st.msg);
426 return;
427 }
428 b->reachable = 1;
429 bat_add(b, "list-objects", 1, "%zu keys", nk);
430 brix_strv_free(keys, nk);
431
432 if (!do_write) { bat_add(b, "write-suite", -1, "skipped (pass --rw)"); return; }
433 if (ak == NULL || sk == NULL) {
434 bat_add(b, "write-suite", -1, "no AWS_ACCESS_KEY_ID/SECRET — writes skipped");
435 return;
436 }
437 {
438 uint8_t payload[2048], rbuf[2048];
439 char uri[320], phash[80], hdrs[2048], reqhdr[2100];
440 long pid = (long) getpid();
441 int fd, st_code = 0, ok;
442 long long blen = 0;
443 const char *bucket_path = (u->path[0] == '/') ? u->path : "/";
444
445 fill_pattern(payload, sizeof(payload));
446 snprintf(uri, sizeof(uri), "%.250s/.xrd_doctor_%ld.bin",
447 (strcmp(bucket_path, "/") == 0) ? "" : bucket_path, pid);
448 if (uri[0] != '/') { /* ensure path-style leading slash */
449 memmove(uri + 1, uri, strlen(uri) + 1);
450 uri[0] = '/';
451 }
452
453 /* PUT (body hash signed) */
454 brix_s3_sha256_hex(payload, sizeof(payload), phash);
455 brix_status_clear(&st);
456 if (brix_s3_sign_v4("PUT", u->host, uri, ak, sk, region, phash, hdrs, sizeof(hdrs)) == 0) {
457 fd = tmpfile_with(payload, sizeof(payload));
458 if (fd >= 0 && brix_http_upload(u->host, u->port, u->tls, uri, hdrs,
459 bat_upload_src_fd, &fd,
460 (long long) sizeof(payload), verify, ca, 10000,
461 &st_code, &st) == 0) {
462 bat_add(b, "PUT", (st_code >= 200 && st_code < 300) ? 1 : 0, "HTTP %d", st_code);
463 } else { bat_add(b, "PUT", 0, "%s", st.msg); }
464 if (fd >= 0) { close(fd); }
465 } else { bat_add(b, "PUT", 0, "sign failed"); }
466
467 /* GET + verify (empty-body hash) */
468 brix_s3_sha256_hex("", 0, phash);
469 brix_status_clear(&st);
470 if (brix_s3_sign_v4("GET", u->host, uri, ak, sk, region, phash, hdrs, sizeof(hdrs)) == 0) {
471 fd = tmpfile_with(NULL, 0);
472 if (fd >= 0 && brix_http_download(u->host, u->port, u->tls, uri, hdrs, verify,
473 ca, fd, 10000, &st_code, &blen, &st) == 0
474 && blen == (long long) sizeof(payload)) {
475 lseek(fd, 0, SEEK_SET);
476 ok = (read(fd, rbuf, sizeof(rbuf)) == (ssize_t) sizeof(payload)
477 && memcmp(rbuf, payload, sizeof(payload)) == 0);
478 bat_add(b, "GET-verify", ok ? 1 : 0, ok ? "byte-exact %lld" : "mismatch", blen);
479 } else { bat_add(b, "GET-verify", 0, "HTTP %d %s", st_code, st.msg); }
480 if (fd >= 0) { close(fd); }
481 } else { bat_add(b, "GET-verify", 0, "sign failed"); }
482
483 /* DELETE */
484 brix_s3_sha256_hex("", 0, phash);
485 brix_status_clear(&st);
486 if (brix_s3_sign_v4("DELETE", u->host, uri, ak, sk, region, phash, hdrs, sizeof(hdrs)) == 0) {
487 brix_http_resp resp;
488 (void) reqhdr;
489 if (brix_http_req(u->host, u->port, u->tls, "DELETE", uri, hdrs, NULL, 0,
490 5000, verify, ca, &resp, &st) == 0) {
491 bat_add(b, "DELETE", (resp.status >= 200 && resp.status < 300) ? 1 : 0,
492 "HTTP %d", resp.status);

503{
504 memset(b, 0, sizeof(*b));
505 snprintf(b->endpoint, sizeof(b->endpoint), "%s", endpoint);
506

508 brix_weburl w;
509 if (brix_weburl_parse(endpoint, &w) != 0) {
510 snprintf(b->protocol, sizeof(b->protocol), "web");
511 snprintf(b->err, sizeof(b->err), "unparseable web URL");
512 return;
513 }
514 if (w.is_s3) {

516 battery_s3(&w, do_write, getenv("AWS_ACCESS_KEY_ID"),
517 getenv("AWS_SECRET_ACCESS_KEY"),
518 region ? region : "us-east-1", verify, b);
519 } else {
520 char *tok = brix_token_discover();
521 battery_web(&w, do_write, tok, verify, b);
522 if (tok != NULL) { free(tok); }
523 }
524 return;
525 }
526 {
527 brix_url u;
528 brix_opts o;
529 brix_status st;
530 memset(&o, 0, sizeof(o));
531 o.verify_host = verify;
532 brix_status_clear(&st);
533 snprintf(b->protocol, sizeof(b->protocol), "root");
534 if (brix_endpoint_parse(endpoint, &u, &st) != 0) {
535 snprintf(b->err, sizeof(b->err), "%s", st.msg);
536 return;
537 }
538 battery_root(&u, &o, do_write, b);
539 }
