1/*
2 * api_snapshot_panels.c - snapshot cache / storage / cluster panel builders.
3 *
4 * WHAT: Owns the three "infrastructure panel" fills of the dashboard snapshot —
5 * "cache" (write-through + per-listener cache stats), "storage" (VFS
6 * export census + per-backend byte I/O), and "cluster" (server registry
7 * heartbeats). Each is a `dashboard_fill_*(json_t *target, ...)` that
8 * populates keys directly on a caller-supplied object.
9 * WHY: Split out of api_snapshot.c (phase-79 file-size split) to keep both
10 * files under the 500-line ceiling. These panels form one cohesive
11 * cluster: they all read the metrics/VFS/registry SHM planes and emit
12 * redact-aware infra detail, distinct from the totals/protocol/event
13 * collectors and the top-level endpoint builders that stay in
14 * api_snapshot.c. Behavior is byte-identical to the pre-split code.
15 * HOW: The cross-file seam lives in dashboard_api_internal.h — dashboard_fill_
16 * cache / _storage / _cluster are the entry points the snapshot and the
17 * flat v1 endpoint builders (in api_snapshot.c) call; every other helper
18 * here is file-static.
19 */
20#include "dashboard_api_internal.h"
21#include "fs/vfs/vfs_backend_registry.h" /* export census for "storage" */

24/*
25 * WHAT: Write-through byte/handle counters, summed across all listener slots.
26 * WHY: dashboard_fill_cache reports both the disabled (all-zero) and live
27 * shapes of the write_through object; carrying the tallies in one struct
28 * keeps the sum loop and the JSON builder decoupled.
29 */
30typedef struct {
31 uint64_t dirty;
32 uint64_t pending;
33 uint64_t success;
34 uint64_t errors;
35 uint64_t bytes;
36} dashboard_wt_totals_t;

39/*
40 * WHAT: Build the "write_through" JSON object from a set of counter totals.
41 * WHY: Both the no-data path (all zero) and the live path emit an identical
42 * key set; one builder guarantees byte-identical output for either.
43 * HOW: "enabled" is inferred true iff any counter is non-zero (no dedicated
44 * configured-flag lives in SHM). Returns NULL on OOM — the caller passes
45 * that straight to json_object_set_new, which tolerates NULL.
46 */
47static json_t *
48dashboard_build_write_through(const dashboard_wt_totals_t *wt_t)
49{
50 json_t *wt = json_object();
51 if (!wt) { return NULL; }
52 json_object_set_new(wt, "enabled",
53 (wt_t->dirty || wt_t->pending || wt_t->success
54 || wt_t->errors || wt_t->bytes)
55 ? json_true() : json_false());
56 json_object_set_new(wt, "dirty_handles", json_integer((json_int_t) wt_t->dirty));
57 json_object_set_new(wt, "flush_pending", json_integer((json_int_t) wt_t->pending));
58 json_object_set_new(wt, "flush_success_total", json_integer((json_int_t) wt_t->success));
59 json_object_set_new(wt, "flush_errors_total", json_integer((json_int_t) wt_t->errors));
60 json_object_set_new(wt, "flush_bytes_total", json_integer((json_int_t) wt_t->bytes));
61 return wt;
62}

83/*
84 * WHAT: Build the per-listener cache JSON entry for one in-use, cache-enabled
85 * server slot.
86 * WHY: Isolating the entry shape keeps the listener loop flat and lets the
87 * stat/eviction fields live next to their comments.
88 * HOW: Ratios are stored in SHM as parts-per-million integers (lock-free);
89 * divide by 1e6 for a 0..1 float. `redact` drops the listen port (infra
90 * detail) for the anonymous tier. Returns NULL on OOM.
91 */
92static json_t *
93dashboard_build_cache_listener(const ngx_brix_srv_metrics_t *srv,
94 ngx_uint_t redact)
95{
96 brix_fs_usage_t fsu;
97 json_t *entry = json_object();
98
99 if (!entry) { return NULL; }
100 if (!redact) { /* listen port is infra detail — omit for anonymous */
101 json_object_set_new(entry, "port", json_integer((json_int_t) srv->port));
102 }
103 json_object_set_new(entry, "auth", json_string(srv->auth));
104 json_object_set_new(entry, "eviction_threshold_ratio",
105 json_real((double) srv->cache_eviction_threshold / 1000000.0));
106 json_object_set_new(entry, "evictions_total",
107 json_integer((json_int_t) srv->cache_evictions_total));
108 json_object_set_new(entry, "evicted_bytes_total",
109 json_integer((json_int_t) srv->cache_evicted_bytes_total));
110 json_object_set_new(entry, "eviction_errors_total",
111 json_integer((json_int_t) srv->cache_eviction_errors_total));
112
113 if (brix_fs_usage_stat(srv->cache_root, &fsu) == NGX_OK) {
114 json_object_set_new(entry, "occupancy_ratio",
115 json_real((double) fsu.occupancy_ppm / 1000000.0));
116 json_object_set_new(entry, "bytes_total",
117 json_integer((json_int_t) fsu.total_bytes));
118 json_object_set_new(entry, "bytes_used",
119 json_integer((json_int_t) fsu.occupancy_bytes));
120 json_object_set_new(entry, "bytes_available",
121 json_integer((json_int_t) fsu.available_bytes));
122 }
123 return entry;
124}

127/*
128 * WHAT: Build the "listeners" array — one entry per in-use, cache-enabled slot.
129 * WHY: Splits the array construction from the per-entry shape so the caller
130 * stays a single set_new call.
131 * HOW: Returns NULL on OOM of the array itself; per-entry OOM skips that slot.
132 */
133static json_t *
134dashboard_build_cache_listeners(const ngx_brix_metrics_t *met, ngx_uint_t redact)
135{
136 json_t *listeners = json_array();
137 ngx_uint_t i;
138
139 if (!listeners) { return NULL; }
140 for (i = 0; i < BRIX_METRICS_MAX_SERVERS; i++) {
141 const ngx_brix_srv_metrics_t *srv = &met->servers[i];
142 json_t *entry;
143
144 if (!srv->in_use || !srv->cache_enabled) { continue; }
145 entry = dashboard_build_cache_listener(srv, redact);
146 if (!entry) { continue; }
147 json_array_append_new(listeners, entry);
148 }
149 return listeners;
150}

153/*
154 * dashboard_fill_cache — adds "enabled", "listeners", and "write_through"
155 * directly to `target`. Called with target=root for v1/cache (flat shape)
156 * and with target=sub-object for snapshot (nested under "cache").
157 */
158void
159dashboard_fill_cache(json_t *target, ngx_uint_t redact)
160{
161 ngx_brix_metrics_t *met;
162 ngx_uint_t i;
163 ngx_uint_t enabled = 0;
164 dashboard_wt_totals_t wt_t = { 0, 0, 0, 0, 0 };
165
166 if (ngx_brix_shm_zone == NULL
167 || ngx_brix_shm_zone->data == NULL
168 || ngx_brix_shm_zone->data == (void *) 1)
169 {
170 dashboard_fill_cache_empty(target);
171 return;
172 }
173
174 met = ngx_brix_shm_zone->data;
175 for (i = 0; i < BRIX_METRICS_MAX_SERVERS; i++) {
176 if (met->servers[i].in_use && met->servers[i].cache_enabled) { enabled = 1; }
177 wt_t.dirty += (uint64_t) met->servers[i].wt_dirty_handles;
178 wt_t.pending += (uint64_t) met->servers[i].wt_flush_pending;
179 wt_t.success += (uint64_t) met->servers[i].wt_flush_success_total;
180 wt_t.errors += (uint64_t) met->servers[i].wt_flush_error_total;
181 wt_t.bytes += (uint64_t) met->servers[i].wt_flush_bytes_total;
182 }
183
184 json_object_set_new(target, "enabled", enabled ? json_true() : json_false());
185 json_object_set_new(target, "listeners", dashboard_build_cache_listeners(met, redact));
186 json_object_set_new(target, "write_through", dashboard_build_write_through(&wt_t));
187}

190/*
191 * dashboard_fill_storage — adds "exports" (per-export backend identity plus
192 * live-statvfs capacity for local backends) and "io" (per-backend byte
193 * totals) to `target`. The identity source is the config-time VFS backend
194 * registry (the same census behind /vfs and the storage-backend info gauge),
195 * so it covers stream AND http exports. redact (anonymous tier) omits the
196 * export root and origin host — numbers only, matching the cache panel.
197 */
198/*
199 * WHAT: Build one "exports" entry from a VFS backend census record.
200 * WHY: Keeps the per-export identity + capacity shape in one place so the
201 * array loop stays flat.
202 * HOW: "remote" is the inverse of a local (posix/pblock) backend; live-statvfs
203 * capacity is only statted for local backends. `redact` drops the export
204 * root and origin host (infra detail) for the anonymous tier. Returns
205 * NULL on OOM.
206 */
207static json_t *
208dashboard_build_storage_export(const brix_vfs_backend_info_t *info,
209 ngx_uint_t redact)
210{
211 brix_fs_usage_t fsu;
212 json_t *e = json_object();
213 int local;
214
215 if (e == NULL) { return NULL; }
216
217 local = ngx_strcmp(info->backend, "posix") == 0
218 || ngx_strcmp(info->backend, "pblock") == 0;
219
220 json_object_set_new(e, "backend", json_string(info->backend));
221 json_object_set_new(e, "remote", local ? json_false() : json_true());
222 json_object_set_new(e, "staging",
223 info->staging ? json_true() : json_false());
224 if (!redact) { /* export path + upstream host are infra detail */
225 json_object_set_new(e, "root", json_string(info->root_canon));
226 if (info->host != NULL && info->host[0] != '\0') {
227 json_object_set_new(e, "origin_host", json_string(info->host));
228 json_object_set_new(e, "origin_port",
229 json_integer((json_int_t) info->port));
230 }
231 }
232 if (local
233 && brix_fs_usage_stat(info->root_canon, &fsu) == NGX_OK)
234 {
235 json_object_set_new(e, "bytes_total",
236 json_integer((json_int_t) fsu.total_bytes));
237 json_object_set_new(e, "bytes_used",
238 json_integer((json_int_t) fsu.occupancy_bytes));
239 json_object_set_new(e, "bytes_available",
240 json_integer((json_int_t) fsu.available_bytes));
241 json_object_set_new(e, "occupancy_ratio",
242 json_real((double) fsu.occupancy_ppm / 1000000.0));
243 }
244 return e;
245}

248/*
249 * WHAT: Build the "exports" array from the config-time VFS backend registry.
250 * WHY: The identity census covers both stream and http exports; splitting the
251 * loop out keeps dashboard_fill_storage a pair of set_new calls.
252 * HOW: Returns NULL on OOM of the array; per-entry OOM/failed-info skips.
253 */
254static json_t *
255dashboard_build_storage_exports(ngx_uint_t redact)
256{
257 json_t *exports = json_array();
258 ngx_uint_t i, n;
259
260 if (exports == NULL) { return NULL; }
261 n = brix_vfs_backend_export_count();
262 for (i = 0; i < n; i++) {
263 brix_vfs_backend_info_t info;
264 json_t *e;
265
266 if (brix_vfs_backend_export_info(i, &info) != NGX_OK) {
267 continue;
268 }
269 e = dashboard_build_storage_export(&info, redact);
270 if (e == NULL) {
271 continue;
272 }
273 json_array_append_new(exports, e);
274 }
275 return exports;
276}

279/*
280 * WHAT: Build the "io" object — per-backend read/written byte totals.
281 * WHY: Isolates the SHM read + idle-backend filtering from the export census.
282 * HOW: Returns NULL on OOM of the object; when the metrics zone is absent the
283 * object is emitted empty (no backends). Idle backends (0/0) are omitted.
284 */
285static json_t *
286dashboard_build_storage_io(void)
287{
288 json_t *io = json_object();
289
290 if (io == NULL) { return NULL; }
291 if (ngx_brix_shm_zone != NULL
292 && ngx_brix_shm_zone->data != NULL
293 && ngx_brix_shm_zone->data != (void *) 1)
294 {
295 ngx_brix_metrics_t *met = ngx_brix_shm_zone->data;
296 int id;
297
298 for (id = 0; id < BRIX_FS_ID_COUNT; id++) {
299 uint64_t rd = (uint64_t) met->unified.io_bytes_read_backend[id];
300 uint64_t wr = (uint64_t) met->unified.io_bytes_written_backend[id];
301 json_t *b;
302
303 if (rd == 0 && wr == 0) {
304 continue; /* idle backends stay out of the JSON */
305 }
306 b = json_object();
307 if (b == NULL) {
308 continue;
309 }
310 json_object_set_new(b, "bytes_read_total",
311 json_integer((json_int_t) rd));
312 json_object_set_new(b, "bytes_written_total",
313 json_integer((json_int_t) wr));
314 json_object_set_new(io, brix_fs_id_name(id), b);
315 }
316 }
317 return io;
318}

343/*
344 * dashboard_fill_cluster — adds "stale_after_ms" and "servers" directly to
345 * `target`. Called with target=root for v1/cluster and target=sub-object for
346 * snapshot.
347 */
348void
349dashboard_fill_cluster(json_t *target, ngx_pool_t *pool, int64_t now_ms,
350 const ngx_http_brix_dashboard_loc_conf_t *conf, ngx_uint_t redact)
351{
352 brix_srv_snapshot_entry_t *entries;
353 ngx_uint_t n, i;
354 json_t *servers;
355
356 json_object_set_new(target, "stale_after_ms",
357 json_integer((json_int_t) conf->cluster_stale_after_ms));
358
359 servers = json_array();
360 entries = ngx_pcalloc(pool, sizeof(*entries) * BRIX_SRV_REGISTRY_SLOTS);
361 if (entries == NULL || servers == NULL) {
362 json_object_set_new(target, "servers", servers ? servers : json_array());
363 return;
364 }
365
366 n = brix_srv_snapshot(entries, BRIX_SRV_REGISTRY_SLOTS,
367 (ngx_msec_t) now_ms);
368 for (i = 0; i < n; i++) {
369 int64_t age = now_ms >= (int64_t) entries[i].last_seen
370 ? now_ms - (int64_t) entries[i].last_seen : 0;
371 json_t *srv = json_object();
372 if (!srv) { continue; }
373 json_object_set_new(srv, "host", json_string(redact ? "[redacted]" : entries[i].host));
374 if (!redact) { /* port + exported paths are infra detail for anonymous */
375 json_object_set_new(srv, "port", json_integer((json_int_t) entries[i].port));
376 json_object_set_new(srv, "paths", json_string(entries[i].paths));
377 /* Phase 89 W9: virtual network id ("" = unset) + staging capability. */
378 json_object_set_new(srv, "vnid", json_string(entries[i].vnid));
379 json_object_set_new(srv, "stage",
380 entries[i].stage ? json_true() : json_false());
381 }
382 json_object_set_new(srv, "free_mb", json_integer((json_int_t) entries[i].free_mb));
383 json_object_set_new(srv, "util_pct", json_integer((json_int_t) entries[i].util_pct));
384 /* Phase 89 W4: heartbeat machine-load percentage (0 until reported). */
385 json_object_set_new(srv, "load_pct", json_integer((json_int_t) entries[i].load_pct));
386 json_object_set_new(srv, "last_seen", json_integer((json_int_t) entries[i].last_seen));
387 json_object_set_new(srv, "heartbeat_age_ms", json_integer((json_int_t) age));
388 json_object_set_new(srv, "stale",
389 age > (int64_t) conf->cluster_stale_after_ms ? json_true() : json_false());
390 /* Phase 23: a non-zero blacklist means the server is drained/blacklisted. */
391 json_object_set_new(srv, "draining",
392 entries[i].blacklisted_until != 0 ? json_true() : json_false());
393 json_array_append_new(servers, srv);
394 }
395 json_object_set_new(target, "servers", servers);
396}
