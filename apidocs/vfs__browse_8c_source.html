1/*
2 * dashboard/vfs_browse.c — VFS export browser for the monitoring UI.
3 *
4 * WHAT: three admin-auth-only endpoints that browse the LOGICAL namespace of
5 * every registered export through the VFS (never raw POSIX):
6 *
7 * GET /xrootd/api/v1/vfs the export census
8 * GET /xrootd/api/v1/vfs/files?export=&path= JSON directory listing
9 * GET /xrootd/api/v1/vfs/download?export=&path= stream one file
10 *
11 * WHY: the older /files browser (files.c) walks a HOST directory tree
12 * (brix_dashboard_browse_root) with raw readdir/statx — right for
13 * logs and spool dirs, wrong for storage: a pblock export's on-disk
14 * shape is catalog.db + packed blobs, and a ceph export has no host
15 * tree at all. Routing every namespace/data op through brix_vfs_*
16 * shows the export exactly as a client sees it, for ANY backend the
17 * registry composed (phase-62: the VFS is the sole storage truth).
18 *
19 * SECURITY:
20 * - Always admin-auth (ngx_http_brix_dashboard_check_auth) — never the
21 * anonymous tier: this surface exposes stored user data.
22 * - Opt-in: disabled (404) unless `brix_dashboard_vfs_browse on`.
23 * - Read-only by construction: the vctx binds with allow_write=0, so even
24 * a coding slip below cannot reach a write path.
25 * - Input path must be absolute, NUL-free, and ".."-free; the VFS then
26 * re-confines every open/stat to the export root at the kernel level
27 * (openat2 RESOLVE_IN_ROOT in the backends).
28 */

101/* Extract + URL-decode ?path= as an EXPORT-ABSOLUTE path ("/" default).
102 * Rejects: non-absolute, embedded NUL, any ".." segment, over-long. The VFS
103 * kernel confinement below is the real wall; this normalises + fails fast. */
104static ngx_int_t
105vfs_browse_get_path(ngx_http_request_t *r, char *out, size_t outsz)
106{
107 ngx_str_t raw;
108 u_char *dst, *src;
109 size_t n, i;
110
111 if (ngx_http_arg(r, (u_char *) "path", 4, &raw) != NGX_OK
112 || raw.len == 0)
113 {
114 ngx_memcpy(out, "/", 2);
115 return NGX_OK;
116 }
117 if (raw.len >= outsz) {
118 return NGX_ERROR;
119 }
120
121 dst = (u_char *) out;
122 src = raw.data;
123 ngx_unescape_uri(&dst, &src, raw.len, 0);
124 n = (size_t) (dst - (u_char *) out);
125 out[n] = '\0';
126
127 if (out[0] != '/' || strlen(out) != n) {
128 return NGX_ERROR; /* relative, or embedded NUL */
129 }
130 for (i = 0; i + 1 < n; i++) {
131 if (out[i] == '.' && out[i + 1] == '.'
132 && (i == 0 || out[i - 1] == '/')
133 && (out[i + 2] == '/' || out[i + 2] == '\0'))
134 {
135 return NGX_ERROR; /* ".." segment */
136 }
137 }
138 /* strip a trailing '/' (except the root itself) for stable echo/joins */
139 if (n > 1 && out[n - 1] == '/') {
140 out[n - 1] = '\0';
141 }
142 return NGX_OK;
143}

145/* Join the export root and the export-absolute path into abs[]. A pure-cache
146 * root of "/" must not double the slash (the cvmfs handler's rule). */
147static ngx_int_t
148vfs_browse_abs_path(const char *root_canon, const char *path,
149 char *abs, size_t abssz)
150{
151 int n;
152
153 if (root_canon[0] == '/' && root_canon[1] == '\0') {
154 n = snprintf(abs, abssz, "%s", path);
155 } else {
156 n = snprintf(abs, abssz, "%s%s", root_canon,
157 (path[0] == '/' && path[1] == '\0') ? "" : path);
158 }
159 return (n > 0 && (size_t) n < abssz) ? NGX_OK : NGX_ERROR;
160}

177/* GET /xrootd/api/v1/vfs — the export census (index, root, backend, origin). */
178ngx_int_t
179ngx_http_brix_dashboard_vfs_exports_handler(ngx_http_request_t *r)
180{
181 ngx_http_brix_dashboard_loc_conf_t *conf;
182 json_t *root, *arr;
183 ngx_uint_t i, n;
184 ngx_int_t rc;
185
186 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
187 rc = vfs_browse_preamble(r, conf);
188 if (rc != NGX_OK) {
189 return rc;
190 }
191
192 root = json_object();
193 arr = json_array();
194 if (root == NULL || arr == NULL) {
195 if (root) { json_decref(root); }
196 if (arr) { json_decref(arr); }
197 return NGX_HTTP_INTERNAL_SERVER_ERROR;
198 }
199
200 n = brix_vfs_backend_export_count();
201 for (i = 0; i < n; i++) {
202 brix_vfs_backend_info_t info;
203 json_t *e;
204
205 if (brix_vfs_backend_export_info(i, &info) != NGX_OK) {
206 continue;
207 }
208 e = json_object();
209 if (e == NULL) {
210 continue;
211 }
212 json_object_set_new(e, "index", json_integer((json_int_t) i));
213 json_object_set_new(e, "root", json_string(info.root_canon));
214 json_object_set_new(e, "backend", json_string(info.backend));
215 if (info.host[0] != '\0') {
216 json_object_set_new(e, "origin_host", json_string(info.host));
217 json_object_set_new(e, "origin_port",
218 json_integer((json_int_t) info.port));
219 }
220 json_object_set_new(e, "staging", info.staging ? json_true()
221 : json_false());
222 if (json_array_append_new(arr, e) != 0) {
223 continue;
224 }
225 }
226
227 dashboard_json_set_schema(root);
228 json_object_set_new(root, "exports", arr);
229 return dashboard_json_send(r, NGX_HTTP_OK, root);
230}

232/* One listing entry via the VFS: kind from d_type, size/mtime from a confined
233 * per-entry stat (skipped for OTHER — symlinks/specials are shown, never
234 * followed). Returns NULL on alloc failure (caller skips the entry). */
235static json_t *
236vfs_browse_entry(ngx_http_request_t *r,
237 const brix_vfs_backend_info_t *info, const char *abs_dir,
238 const ngx_str_t *name, brix_vfs_dirent_kind_t kind)
239{
240 char child[PATH_MAX];
241 char namez[NAME_MAX + 1];
242 brix_vfs_ctx_t cctx;
243 brix_vfs_stat_t st;
244 json_t *o;
245 const char *type;
246 int n;
247
248 if (name->len > NAME_MAX) {
249 return NULL;
250 }
251 ngx_memcpy(namez, name->data, name->len);
252 namez[name->len] = '\0';
253
254 ngx_memzero(&st, sizeof(st));
255 if (kind != BRIX_VFS_DT_OTHER) {
256 n = snprintf(child, sizeof(child), "%s%s%s", abs_dir,
257 (abs_dir[strlen(abs_dir) - 1] == '/') ? "" : "/", namez);
258 if (n <= 0 || (size_t) n >= sizeof(child)) {
259 return NULL;
260 }
261 vfs_browse_ctx(r, info, child, &cctx);
262 if (brix_vfs_stat(&cctx, &st) != NGX_OK) {
263 ngx_memzero(&st, sizeof(st)); /* vanished mid-scan: list bare */
264 }
265 if (kind == BRIX_VFS_DT_UNKNOWN) {
266 kind = st.is_directory ? BRIX_VFS_DT_DIR
267 : st.is_regular ? BRIX_VFS_DT_REG
268 : BRIX_VFS_DT_OTHER;
269 }
270 }
271 type = (kind == BRIX_VFS_DT_DIR) ? "dir"
272 : (kind == BRIX_VFS_DT_REG) ? "file" : "other";
273
274 o = json_object();
275 if (o == NULL) {
276 return NULL;
277 }
278 json_object_set_new(o, "name", json_string(namez));
279 json_object_set_new(o, "type", json_string(type));
280 json_object_set_new(o, "size", json_integer((json_int_t) st.size));
281 json_object_set_new(o, "mtime", json_integer((json_int_t) st.mtime));
282 return o;
283}

286ngx_int_t
287ngx_http_brix_dashboard_vfs_files_handler(ngx_http_request_t *r)
288{
289 ngx_http_brix_dashboard_loc_conf_t *conf;

291 brix_vfs_ctx_t vctx;
292 brix_vfs_dir_t *dh;

294 char abs[PATH_MAX];
295 ngx_uint_t idx, count = 0;
296 int err = 0, truncated = 0;
297 json_t *root, *arr;
298 ngx_int_t rc;
299
300 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
301 rc = vfs_browse_preamble(r, conf);
302 if (rc != NGX_OK) {
303 return rc;
304 }
305 if (vfs_browse_get_export(r, &idx) != NGX_OK
306 || brix_vfs_backend_export_info(idx, &info) != NGX_OK
307 || vfs_browse_get_path(r, path, sizeof(path)) != NGX_OK
308 || vfs_browse_abs_path(info.root_canon, path, abs, sizeof(abs))
309 != NGX_OK)
310 {
311 return NGX_HTTP_BAD_REQUEST;
312 }
313
314 vfs_browse_ctx(r, &info, abs, &vctx);
315 dh = brix_vfs_opendir(&vctx, &err);
316 if (dh == NULL) {
317 return vfs_browse_status(err);
318 }
319
320 root = json_object();
321 arr = json_array();
322 if (root == NULL || arr == NULL) {
323 if (root) { json_decref(root); }
324 if (arr) { json_decref(arr); }
325 brix_vfs_closedir(dh, r->connection->log);
326 return NGX_HTTP_INTERNAL_SERVER_ERROR;
327 }
328
329 for ( ;; ) {
330 ngx_str_t name;
331 brix_vfs_dirent_kind_t kind;
332 json_t *e;
333
334 rc = brix_vfs_readdir_kind(dh, &name, &kind);
335 if (rc == NGX_DONE) {
336 break;
337 }
338 if (rc != NGX_OK) {
339 break; /* stream error: serve what we have */
340 }
341 if (count >= DASHBOARD_VFS_MAX_ENTRIES) {
342 truncated = 1;
343 break;
344 }
345 e = vfs_browse_entry(r, &info, abs, &name, kind);
346 if (e == NULL || json_array_append_new(arr, e) != 0) {
347 continue;
348 }
349 count++;
350 }
351 brix_vfs_closedir(dh, r->connection->log);
352
353 dashboard_json_set_schema(root);
354 json_object_set_new(root, "export", json_integer((json_int_t) idx));
355 json_object_set_new(root, "backend", json_string(info.backend));
356 json_object_set_new(root, "path", json_string(path));
357 json_object_set_new(root, "truncated", truncated ? json_true()
358 : json_false());
359 json_object_set_new(root, "entries", arr);
360 return dashboard_json_send(r, NGX_HTTP_OK, root);
361}
