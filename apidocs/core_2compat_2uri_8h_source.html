41/*
42 * brix_http_urldecode — decode percent-encoded HTTP input into a C string.
43 *
44 * WHAT: Scans src[0..src_len] for '%' escape sequences and decodes them to the
45 * corresponding byte value. Well-formed escapes (%HH with valid hex digits)
46 * are decoded; malformed '%' sequences (missing digits, invalid hex) are
47 * preserved verbatim — nginx-lenient strategy.
48 *
49 * WHY: WebDAV query parameters, S3 canonical request strings, and XrdHttp paths
50 * arrive percent-encoded from HTTP clients. This function decodes them to
51 * raw bytes suitable for path resolution, auth signing, or storage ops.
52 *
53 * HOW: Iterate src byte-by-byte. When '%' is found, read the next two bytes as
54 * hex nibbles via brix_hex_from_char(). If both are valid hex, combine
55 * them into a decoded byte and write to dst. If either nibble is invalid,
56 * copy the '%' verbatim. Check for NUL_BYTE when REJECT_NUL flag is set.
57 * Track overflow at each write (di+1 >= dst_sz). Always null-terminate on OK.
58 *
59 * Parameters:
60 * src — input bytes to decode (may contain '%', '+', raw chars)
61 * src_len — length of the input buffer
62 * dst — output buffer for decoded result (must have at least 2 bytes)
63 * dst_sz — total size of the output buffer
64 * flags — combination of BRIX_URLDECODE_* flags (0 = default behaviour)
65 *
66 * Returns:
67 * BRIX_URLDECODE_OK — success; NUL terminator written at dst[di]
68 * BRIX_URLDECODE_OVERFLOW — output buffer too small for decoded result
69 * BRIX_URLDECODE_NUL_BYTE — %00 encountered with REJECT_NUL flag set
70 * BRIX_URLDECODE_BADARG — dst is NULL or dst_sz < 2
71 */
72int brix_http_urldecode(const unsigned char *src, size_t src_len,
73 char *dst, size_t dst_sz, unsigned flags);

75/*
76 * brix_http_urlencode — RFC 3986 percent-encoder.
77 *
78 * WHAT: Encodes src[0..srclen] into HTTP percent-encoded form. Unreserved
79 * characters per RFC 3986 ([A-Za-z0-9._~-]) plus any character in
80 * safe_extra pass through unchanged. All others are encoded as %XX
81 * with uppercase hex digits.
82 *
83 * WHY: S3 canonical request construction, WebDAV URL generation, and XrdHttp
84 * path encoding all need RFC 3986-compliant percent-encoding. The encoder
85 * produces deterministic output suitable for HMAC-SHA256 signing (S3).
86 *
87 * HOW: Iterate src bytes. Check against the unreserved set + safe_extra chars;
88 * if matched, copy directly to dst. Otherwise emit '%' followed by two
89 * uppercase hex nibbles via h[]. Track overflow at each write position.
90 * Null-terminate on success.
91 *
92 * Parameters:
93 * src — input bytes to encode
94 * srclen — length of the input buffer
95 * dst — output buffer for encoded result (must have at least 1 byte)
96 * dstsz — total size of the output buffer
97 * safe_extra — additional characters treated as unreserved (e.g. "/" for path segments);
98 * NULL means only RFC 3986 unreserved chars pass through
99 *
100 * Returns:
101 * >= 0 — number of bytes written (excluding NUL terminator) on success
102 * -1 — output buffer too small, or dst is NULL / dstsz < 1
103 */
104ssize_t brix_http_urlencode(const unsigned char *src, size_t srclen,
105 char *dst, size_t dstsz, const char *safe_extra);
