1/*
2 * metrics/cluster.c — Prometheus gauges for the shared-memory server registry.
3 *
4 * Exports per-server free-space, utilisation, heartbeat age, and blacklist
5 * status so operators can monitor cluster membership without polling the
6 * dashboard API. Unlike other metrics files this one reads directly from the
7 * registry SHM zone (via brix_srv_snapshot) rather than from the stream
8 * module's per-worker metrics zone.
9 */
10
11#include "metrics_internal.h"
12#include "metrics_macros.h"
13#include "net/manager/registry.h"
14
15/*
16 * WHAT: emit the per-server cluster gauges and aggregate health-check counters
17 * into the supplied metrics writer.
18 * WHY: operators need live visibility into registry membership (which data
19 * servers the manager knows about, their reported space/load, heartbeat
20 * freshness, and blacklist state) without scraping the dashboard JSON API.
21 * HOW: take a point-in-time copy of the registry SHM via brix_srv_snapshot()
22 * (which acquires the registry spinlock, copies under lock, and releases —
23 * so everything below operates on the local stack array `entries` and
24 * holds no lock), then walk that copy once per metric family. Each family
25 * is its own emit loop so the HELP/TYPE banner is written exactly once
26 * ahead of the per-server samples (Prometheus requires the banner before
27 * any sample of that family).
28 */
29void
30brix_export_cluster_metrics(metrics_writer_t *mw)
31{
32 /* Stack snapshot buffer; sized to the compile-time slot maximum so a full
33 * registry always fits without a heap allocation in the scrape path. */
34 brix_srv_snapshot_entry_t entries[BRIX_SRV_REGISTRY_SLOTS];
35 ngx_uint_t n, i;
36 ngx_msec_t now;
37 /* 255 (max host) + 1 (':') + 5 (port digits) + 1 ('\0') = 262; use 512. */
38 char srv_label[512];
39
40 /* Registry SHM is only created in manager/redirector mode; with no zone
41 * there is no cluster to report — emit nothing (not even the count gauge). */
42 if (brix_srv_shm_zone == NULL) {
43 return;
44 }
45
46 /* Sample the clock once so the heartbeat-age and blacklist comparisons
47 * below all reference the same instant as the snapshot was taken. */
48 now = ngx_current_msec;
49 n = brix_srv_snapshot(entries, BRIX_SRV_REGISTRY_SLOTS, now);
50
51 mw_printf(mw,
52 "# HELP brix_cluster_servers_registered "
53 "Number of data servers currently in the cluster registry.\n"
54 "# TYPE brix_cluster_servers_registered gauge\n"
55 "brix_cluster_servers_registered %u\n",
56 (unsigned int) n);
57
58 /* Count gauge already emitted; with no occupied slots there are no
59 * per-server families to follow, so stop before the emit loops. */
60 if (n == 0) {
61 return;
62 }
63
64 /* Build the {server="host:port"} label into the shared srv_label buffer.
65 * %.255s bounds the host copy to the registry's host[256] field width so a
66 * non-NUL-terminated or oversized host can never overrun srv_label[512].
67 * Re-invoked at the top of every emit loop because the buffer is reused. */
68#define CLUSTER_LABEL(idx) \
69 snprintf(srv_label, sizeof(srv_label), "%.255s:%u", \
70 entries[(idx)].host, (unsigned int) entries[(idx)].port)
71
72 mw_printf(mw,
73 "# HELP brix_cluster_server_free_megabytes "
74 "Free disk space (MB) last reported by each registered data server.\n"
75 "# TYPE brix_cluster_server_free_megabytes gauge\n");
76 for (i = 0; i < n; i++) {
77 CLUSTER_LABEL(i);
78 mw_printf(mw,
79 "brix_cluster_server_free_megabytes{server=\"%s\"} %u\n",
80 srv_label, (unsigned int) entries[i].free_mb);
81 }
82
83 mw_printf(mw,
84 "# HELP brix_cluster_server_utilization_percent "
85 "Storage utilisation percent (0-100) last reported by each data server.\n"
86 "# TYPE brix_cluster_server_utilization_percent gauge\n");
87 for (i = 0; i < n; i++) {
88 CLUSTER_LABEL(i);
89 mw_printf(mw,
90 "brix_cluster_server_utilization_percent{server=\"%s\"} %u\n",
91 srv_label, (unsigned int) entries[i].util_pct);
92 }
93
94 mw_printf(mw,
95 "# HELP brix_cluster_server_last_seen_seconds "
96 "Seconds since this data server last sent a CMS heartbeat to the manager.\n"
97 "# TYPE brix_cluster_server_last_seen_seconds gauge\n");
98 for (i = 0; i < n; i++) {
99 /* last_seen is the ngx_current_msec captured at the server's last
100 * heartbeat; age is now - last_seen converted ms -> s. The (now >=
101 * last_seen) guard avoids an unsigned underflow producing a huge bogus
102 * age if last_seen is momentarily ahead of `now` (clock skew between
103 * the worker that recorded it and this scrape) — clamp to 0 instead. */
104 double age_s = (now >= entries[i].last_seen)
105 ? (double)(now - entries[i].last_seen) / 1000.0
106 : 0.0;
107 CLUSTER_LABEL(i);
108 mw_printf(mw,
109 "brix_cluster_server_last_seen_seconds{server=\"%s\"} %.3f\n",
110 srv_label, age_s);
111 }
112
113 mw_printf(mw,
114 "# HELP brix_cluster_server_blacklisted "
115 "1 if the server is temporarily blacklisted after a CMS disconnect, "
116 "0 if available.\n"
117 "# TYPE brix_cluster_server_blacklisted gauge\n");
118 for (i = 0; i < n; i++) {
119 /* blacklisted_until==0 means "available"; any non-zero value is a
120 * future ngx_current_msec deadline. The entry is still blacklisted
121 * only while that deadline is in the future (> now); a deadline that
122 * has already elapsed reports 0 even though the registry hasn't yet
123 * lazily cleared the field. */
124 int bl = (entries[i].blacklisted_until != 0
125 && entries[i].blacklisted_until > now) ? 1 : 0;
126 CLUSTER_LABEL(i);
127 mw_printf(mw,
128 "brix_cluster_server_blacklisted{server=\"%s\"} %d\n",
129 srv_label, bl);
130 }
131
132 /* error_count doubles as the cumulative CMS-disconnect counter for the
133 * entry; exposed as a Prometheus counter (monotonic within a slot's life,
134 * but resets if the slot is recycled to a different server). */
135 mw_printf(mw,
136 "# HELP brix_cluster_server_disconnect_total "
137 "Cumulative CMS disconnect count for each registered data server.\n"
138 "# TYPE brix_cluster_server_disconnect_total counter\n");
139 for (i = 0; i < n; i++) {
140 CLUSTER_LABEL(i);
141 mw_printf(mw,
142 "brix_cluster_server_disconnect_total{server=\"%s\"} %u\n",
143 srv_label, (unsigned int) entries[i].error_count);
144 }
145
146#undef CLUSTER_LABEL
147
148 /* Phase 22 — aggregate health-check counters (no per-server labels, per
149 * INVARIANT 8: low-cardinality only; per-server HC state is on the
150 * dashboard snapshot API). */
151 {
152 /* These four counters live in the stream module's shared metrics zone
153 * (not the registry SHM walked above), so fetch that mapping separately;
154 * it is NULL until the metrics zone is initialised. */
155 ngx_brix_metrics_t *m = brix_metrics_shared();
156 if (m != NULL) {
157 mw_printf(mw,
158 "# HELP brix_cluster_hc_probes_total "
159 "Active health-check probes started.\n"
160 "# TYPE brix_cluster_hc_probes_total counter\n"
161 "brix_cluster_hc_probes_total %lu\n"
162 "# HELP brix_cluster_hc_pass_total "
163 "Health-check probes that passed.\n"
164 "# TYPE brix_cluster_hc_pass_total counter\n"
165 "brix_cluster_hc_pass_total %lu\n"
166 "# HELP brix_cluster_hc_fail_total "
167 "Health-check probes that failed or timed out.\n"
168 "# TYPE brix_cluster_hc_fail_total counter\n"
169 "brix_cluster_hc_fail_total %lu\n"
170 "# HELP brix_cluster_hc_blacklist_total "
171 "Servers blacklisted by health checking.\n"
172 "# TYPE brix_cluster_hc_blacklist_total counter\n"
173 "brix_cluster_hc_blacklist_total %lu\n",
174 /* fetch_add(&x, 0) is the lock-free idiom for an atomic READ of
175 * an ngx_atomic_t — adding zero leaves the value unchanged while
176 * returning a consistent snapshot without taking a lock. */
177 (unsigned long) ngx_atomic_fetch_add(&m->hc_probes_total, 0),
178 (unsigned long) ngx_atomic_fetch_add(&m->hc_pass_total, 0),
179 (unsigned long) ngx_atomic_fetch_add(&m->hc_fail_total, 0),
180 (unsigned long) ngx_atomic_fetch_add(&m->hc_blacklist_total, 0));
181 }
182 }
183}
