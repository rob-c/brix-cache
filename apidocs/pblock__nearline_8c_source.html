1/*
2 * pblock_nearline.c — F4 nearline/tape residency simulation for pblock.
3 *
4 * WHAT: Implements pblock_nearline.h: the `nearline(path, res)` table, the
5 * pure residency read, the bounded synchronous recall (ctl-driven
6 * latency/failure), and namespace row maintenance.
7 *
8 * HOW: The table stores only demoted paths — absence means ONLINE, so a

45int
46pblock_nearline_res(const pblock_state_t *st, const char *path,
47 brix_sd_residency_t *out)
48{
49 sqlite3_stmt *q;
50
51 *out = BRIX_SD_RES_ONLINE;
52 q = cat_prepare(st->cat, "SELECT res FROM nearline WHERE path = ?1;");
53 if (q == NULL) {
54 return -1;
55 }
56 sqlite3_bind_text(q, 1, path, -1, SQLITE_STATIC);
57 if (sqlite3_step(q) == SQLITE_ROW) {
58 int64_t r = sqlite3_column_int64(q, 0);
59
60 *out = (r >= BRIX_SD_RES_ONLINE && r <= BRIX_SD_RES_LOST)
61 ? (brix_sd_residency_t) r : BRIX_SD_RES_LOST;
62 }
63 sqlite3_finalize(q);
64 return 0;
65}

67/* Set (or clear, res == ONLINE — absence IS online) the residency row. */
68static int
69nearline_set(const pblock_state_t *st, const char *path,
70 brix_sd_residency_t res)
71{
72 sqlite3_stmt *q;
73 int rc;
74
75 if (res == BRIX_SD_RES_ONLINE) {
76 q = cat_prepare(st->cat, "DELETE FROM nearline WHERE path = ?1;");
77 } else {
78 q = cat_prepare(st->cat,
79 "INSERT INTO nearline VALUES(?1, ?2)"
80 " ON CONFLICT(path) DO UPDATE SET res = excluded.res;");
81 }
82 if (q == NULL) {
83 return -1;
84 }
85 sqlite3_bind_text(q, 1, path, -1, SQLITE_STATIC);
86 if (res != BRIX_SD_RES_ONLINE) {
87 sqlite3_bind_int64(q, 2, (int64_t) res);
88 }
89 rc = sqlite3_step(q) == SQLITE_DONE ? 0 : -1;
90 sqlite3_finalize(q);
91 return rc;
92}

94int
95pblock_nearline_recall(const pblock_state_t *st, const char *path)
96{
97 brix_sd_residency_t res;
98 char val[64], failkey[1088];
99 int64_t ms = 0;
100
101 if (pblock_nearline_res(st, path, &res) != 0) {
102 return 0; /* fail-open: an unreadable simulation
103 * table must not take the export down */
104 }
105 if (res == BRIX_SD_RES_ONLINE) {
106 return 0;
107 }
108 if (res == BRIX_SD_RES_LOST) {
109 errno = ENOENT;
110 return -1;
111 }
112
113 /* NEARLINE/OFFLINE: simulate the tape robot — sleep the configured
114 * latency, then land the outcome. */
115 if (pblock_ctl_get(st->cat, "nearline.recall_ms", val, sizeof(val)) == 1) {
116 ms = strtoll(val, NULL, 10);
117 if (ms < 0) {
118 ms = 0;
119 }
120 if (ms > NL_RECALL_MS_CAP) {
121 ms = NL_RECALL_MS_CAP;
122 }
123 }
124 if (ms > 0) {
125 struct timespec ts = { ms / 1000, (ms % 1000) * 1000000L };
126
127 while (nanosleep(&ts, &ts) != 0 && errno == EINTR) {
128 /* resume the remainder */
129 }
130 }
131
132 snprintf(failkey, sizeof(failkey), "nearline.fail.%s", path);
133 if (pblock_ctl_get(st->cat, failkey, val, sizeof(val)) == 1) {
134 (void) nearline_set(st, path, BRIX_SD_RES_LOST);
135 errno = EIO;
136 return -1;
137 }
138 if (nearline_set(st, path, BRIX_SD_RES_ONLINE) != 0) {
139 errno = EIO;
140 return -1;
141 }
142 return 0;
143}
