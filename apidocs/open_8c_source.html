16/*
17 * open.c — read-side cache hit path: serve a file from the local cache tree.
18 *
19 * WHAT: The VFS cache-open hook (brix_cache_open) plus its helpers: mapping a
20 * resolved export path to its cache-tree path (brix_cache_path_for_resolved),
21 * validating a hit against the origin (brix_cache_validate_meta), and
22 * bumping the access time for LRU (brix_cache_record_access).
23 *
24 * WHY: When read-through caching is on, reads should be satisfied from the
25 * local cache whenever a complete, fresh copy exists, avoiding an origin
26 * round-trip. This file decides "is there a usable cache hit?" and, if so,
27 * adopts the open fd into the VFS layer.
28 *
29 * HOW: brix_cache_open() declines (NGX_DECLINED) for anything that must not
30 * be served from cache — caching disabled, BRIX_VFS_O_NOCACHE, or any
31 * write/create/trunc/append intent — so callers fall back to the origin.
32 * Otherwise it derives the cache path under cache_root_canon, checks
33 * brix_cache_file_ready(), opens it O_RDONLY|O_NOFOLLOW|O_CLOEXEC, and
34 * confirms freshness via the meta sidecar (size + mtime) before handing
35 * the fd to brix_vfs_adopt_fd(). Confinement here is path-construction
36 * + O_NOFOLLOW rather than RESOLVE_BENEATH (the cache tree is a different
37 * directory from the export rootfd) — see the in-function comment.
38 */

40/*
41 * brix_cache_path_for_resolved — map an export path to its cache-tree path.
42 *
43 * Strips root_canon from the resolved export path and re-roots the remaining
44 * suffix under cache_root_canon, writing the result to out (NUL-terminated).
45 * Rejects (NGX_ERROR) NULL/empty args, a resolved path not under root_canon, a
46 * malformed suffix (EINVAL), or a result that would overflow out (ENAMETOOLONG).
47 * Note: this is a pure lexical remap, not a confinement check.
48 */
49ngx_int_t
50brix_cache_path_for_resolved(const char *cache_root_canon,
51 const char *root_canon, const char *resolved, char *out, size_t outsz)
52{
53 size_t cache_len;
54 size_t root_len;
55 size_t suffix_len;
56 const char *suffix;
57
58 if (cache_root_canon == NULL || root_canon == NULL || resolved == NULL
59 || out == NULL || outsz == 0 || cache_root_canon[0] == '\0'
60 || root_canon[0] == '\0' || resolved[0] == '\0')
61 {
62 errno = EINVAL;
63 return NGX_ERROR;
64 }
65
66 root_len = strlen(root_canon);
67 if (ngx_strncmp((u_char *) resolved, (u_char *) root_canon, root_len)
68 != 0)
69 {
70 errno = EINVAL;
71 return NGX_ERROR;
72 }
73
74 if (root_len == 1 && root_canon[0] == '/') {
75 suffix = (resolved[1] != '\0') ? resolved : "";
76 } else {
77 suffix = resolved + root_len;
78 }
79 if (suffix[0] != '\0' && suffix[0] != '/') {
80 errno = EINVAL;
81 return NGX_ERROR;
82 }
83
84 cache_len = strlen(cache_root_canon);
85 suffix_len = strlen(suffix);
86 if (cache_len == 1 && cache_root_canon[0] == '/' && suffix[0] == '/') {
87 suffix++;
88 suffix_len--;
89 }
90 if (cache_len + suffix_len >= outsz) {
91 errno = ENAMETOOLONG;
92 return NGX_ERROR;
93 }
94
95 ngx_memcpy(out, cache_root_canon, cache_len);
96 ngx_memcpy(out + cache_len, suffix, suffix_len + 1);
97
98 return NGX_OK;
99}

101/*
102 * brix_cache_validate_meta — confirm a cache hit still matches the origin.
103 *
104 * Reads the sidecar meta for cache_path and compares its recorded size and
105 * mtime against the live cache file stat (st). Returns NGX_OK if they agree,
106 * NGX_DECLINED otherwise (no meta, or stale: errno set to ESTALE on mismatch),
107 * so the caller falls back to the origin.
108 */
109static ngx_int_t
110brix_cache_validate_meta(const char *cache_path, const struct stat *st,
111 ngx_log_t *log)
112{
113 brix_cache_meta_t meta;
114
115 if (brix_cache_meta_read(log, cache_path, &meta) != NGX_OK) {
116 return NGX_DECLINED;
117 }
118
119 if (meta.size != (uint64_t) st->st_size
120 || meta.mtime != (uint64_t) st->st_mtime)
121 {
122 errno = ESTALE;
123 return NGX_DECLINED;
124 }
125
126 return NGX_OK;
127}

129/*
130 * brix_cache_open — VFS cache-open hook: serve a read from the cache if able.
131 *
132 * Declines (NGX_DECLINED) when caching is disabled, BRIX_VFS_O_NOCACHE is
133 * set, or any write/create/trunc/append flag is present, and on a cache miss,
134 * not-ready file, or stale meta. On a validated hit it opens the cache file
135 * read-only and adopts the fd via brix_vfs_adopt_fd(), returning *fh_out and
136 * NGX_OK. Returns NGX_ERROR on hard I/O errors (errno set).
137 */
138ngx_int_t
139brix_cache_open(brix_vfs_ctx_t *ctx, ngx_uint_t flags,
140 brix_vfs_file_t **fh_out)
141{
142 char cache_path[PATH_MAX];
143 const char *resolved;
144 struct stat st;
145 ngx_int_t rc;
146
147 if (fh_out == NULL) {
148 errno = EINVAL;
149 return NGX_ERROR;
150 }
151 *fh_out = NULL;
152
153 if (ctx == NULL || !ctx->cache_enabled
154 || ctx->cache_root_canon == NULL || ctx->cache_root_canon[0] == '\0'
155 || (flags & BRIX_VFS_O_NOCACHE)
156 || (flags & (BRIX_VFS_O_WRITE | BRIX_VFS_O_CREATE
157 | BRIX_VFS_O_TRUNC | BRIX_VFS_O_APPEND)))
158 {
159 return NGX_DECLINED;
160 }
161
162 resolved = brix_vfs_ctx_path(ctx);
163 if (brix_cache_path_for_resolved(ctx->cache_root_canon,

169 }

171 /*
172 * Exclusively-VFS hit-serve: the cache file is opened through the cache
173 * STORAGE driver (POSIX driver on the cache rootfd by default, or a configured
174 * backend), keyed on the cache namespace. The driver owns confinement; for a
175 * driver-backed cache the bytes live in the driver's namespace, not as a POSIX
176 * file. The normal read path memory-serves when the backend cannot sendfile.
177 */
178 {
179 brix_sd_instance_t *inst =
180 brix_cache_storage_by_root(ctx->cache_root_canon);
181 const char *key;
182 brix_sd_stat_t sd_st;
183 brix_sd_obj_t *o;
184 int e = 0;
185
186 if (inst == NULL) {
187 return NGX_DECLINED; /* no cache storage on this root */
188 }
189 /* key = the suffix under cache_root (what the driver keys its namespace on). */
190 key = cache_path + ngx_strlen(ctx->cache_root_canon);
191 if (key[0] != '/') {
192 return NGX_DECLINED;
193 }
194
195 if (inst->driver->stat(inst, key, &sd_st) != NGX_OK) {
196 return (errno == ENOENT || errno == ENOTDIR) ? NGX_DECLINED
197 : NGX_ERROR;
198 }
199 if (!sd_st.is_reg) {
200 return NGX_DECLINED;
201 }
202
203 /* Validate against the .meta sidecar (origin freshness) at the POSIX state
204 * path (== cache_path for a co-located cache). Synthesize a struct stat
205 * from the driver snapshot. */
206 ngx_memzero(&st, sizeof(st));
207 st.st_size = sd_st.size;
208 st.st_mtime = sd_st.mtime;
209 st.st_mode = S_IFREG;
210 {
211 const char *state_root =
212 brix_cache_state_root_by_root(ctx->cache_root_canon);
213 char sidecar[PATH_MAX];
214
215 if (state_root == NULL
216 || brix_cache_sidecar_path(ctx->cache_root_canon, state_root,
217 cache_path, sidecar, sizeof(sidecar)) != 0
218 || brix_cache_validate_meta(sidecar, &st, ctx->log) != NGX_OK)
219 {
220 return NGX_DECLINED; /* obj not opened yet — nothing to close */
221 }
222 }
223
224 o = inst->driver->open(inst, key, BRIX_SD_O_READ, 0, &e);
225 if (o == NULL) {
226 errno = e;
227 return (e == ENOENT || e == ENOTDIR) ? NGX_DECLINED : NGX_ERROR;
228 }
229
230 rc = brix_vfs_adopt_obj(ctx, key, o, 0 /* read-only */, fh_out);
231 if (rc != NGX_OK) {
232 int err = errno;
233 (void) inst->driver->close(o);
234 if (o->heap_shell) {
235 free(o);
236 }
237 errno = err;
238 }
239 return rc;
240 }
