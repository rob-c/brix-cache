16/*
17 * open.c — read-side cache hit path: serve a file from the local cache tree.
18 *
19 * WHAT: The VFS cache-open hook (brix_cache_open) plus its helpers: mapping a
20 * resolved export path to its cache-tree path (brix_cache_path_for_resolved),
21 * validating a hit against the origin (brix_cache_validate_meta), and
22 * bumping the access time for LRU (brix_cache_record_access).
23 *
24 * WHY: When read-through caching is on, reads should be satisfied from the
25 * local cache whenever a complete, fresh copy exists, avoiding an origin
26 * round-trip. This file decides "is there a usable cache hit?" and, if so,
27 * adopts the open fd into the VFS layer.
28 *
29 * HOW: brix_cache_open() declines (NGX_DECLINED) for anything that must not
30 * be served from cache — caching disabled, BRIX_VFS_O_NOCACHE, or any
31 * write/create/trunc/append intent — so callers fall back to the origin.
32 * Otherwise it derives the cache path under cache_root_canon, checks
33 * brix_cache_file_ready(), opens it O_RDONLY|O_NOFOLLOW|O_CLOEXEC, and
34 * confirms freshness via the meta sidecar (size + mtime) before handing
35 * the fd to brix_vfs_adopt_fd(). Confinement here is path-construction
36 * + O_NOFOLLOW rather than RESOLVE_BENEATH (the cache tree is a different
37 * directory from the export rootfd) — see the in-function comment.
38 */

40/*
41 * brix_cache_path_args_valid — reject NULL/empty inputs to the path remap.
42 *
43 * WHAT: True iff every string arg is non-NULL and non-empty and outsz > 0.
44 * WHY: Centralizes the guard so the remap body deals only with well-formed
45 * inputs, preserving the original EINVAL-on-bad-args contract.
46 * HOW: Pure boolean over the four strings plus outsz; no side effects.

50 const char *resolved, const char *out, size_t outsz)

52 return cache_root_canon != NULL && root_canon != NULL && resolved != NULL
53 && out != NULL && outsz != 0 && cache_root_canon[0] != '\0'
54 && root_canon[0] != '\0' && resolved[0] != '\0';

57/*
58 * brix_cache_export_suffix — derive the export-relative suffix of resolved.
59 *
60 * WHAT: Confirms resolved is under root_canon and returns the trailing suffix
61 * (the part re-rooted under the cache tree), or NULL if resolved is not
62 * under root_canon or the suffix is malformed (not '' or starting '/').
63 * WHY: Isolates the root-stripping + suffix-shape check from assembly so each
64 * branch of the original function is a single, testable step.
65 * HOW: Prefix-compares root_canon, special-cases root_canon == "/", then
66 * validates the leading character of the remainder.
67 */
68static const char *
69brix_cache_export_suffix(const char *root_canon, const char *resolved)
70{
71 size_t root_len;
72 const char *suffix;
73
74 root_len = strlen(root_canon);
75 if (ngx_strncmp((u_char *) resolved, (u_char *) root_canon, root_len) != 0) {
76 return NULL;
77 }
78
79 if (root_len == 1 && root_canon[0] == '/') {
80 suffix = (resolved[1] != '\0') ? resolved : "";
81 } else {
82 suffix = resolved + root_len;
83 }
84
85 if (suffix[0] != '\0' && suffix[0] != '/') {
86 return NULL;
87 }
88
89 return suffix;
90}

92/*
93 * brix_cache_path_for_resolved — map an export path to its cache-tree path.
94 *
95 * Strips root_canon from the resolved export path and re-roots the remaining
96 * suffix under cache_root_canon, writing the result to out (NUL-terminated).
97 * Rejects (NGX_ERROR) NULL/empty args, a resolved path not under root_canon, a
98 * malformed suffix (EINVAL), or a result that would overflow out (ENAMETOOLONG).
99 * Note: this is a pure lexical remap, not a confinement check.
100 */
101ngx_int_t
102brix_cache_path_for_resolved(const char *cache_root_canon,
103 const char *root_canon, const char *resolved, char *out, size_t outsz)
104{
105 size_t cache_len;
106 size_t suffix_len;
107 const char *suffix;
108
109 if (!brix_cache_path_args_valid(cache_root_canon, root_canon, resolved,
110 out, outsz))
111 {
112 errno = EINVAL;
113 return NGX_ERROR;
114 }
115
116 suffix = brix_cache_export_suffix(root_canon, resolved);
117 if (suffix == NULL) {
118 errno = EINVAL;
119 return NGX_ERROR;
120 }
121
122 cache_len = strlen(cache_root_canon);
123 suffix_len = strlen(suffix);
124 if (cache_len == 1 && cache_root_canon[0] == '/' && suffix[0] == '/') {
125 suffix++;
126 suffix_len--;
127 }
128 if (cache_len + suffix_len >= outsz) {
129 errno = ENAMETOOLONG;
130 return NGX_ERROR;
131 }
132
133 ngx_memcpy(out, cache_root_canon, cache_len);
134 ngx_memcpy(out + cache_len, suffix, suffix_len + 1);
135
136 return NGX_OK;
137}

139/*
140 * brix_cache_validate_meta — confirm a cache hit still matches the origin.
141 *
142 * Reads the sidecar meta for cache_path and compares its recorded size and
143 * mtime against the live cache file stat (st). Returns NGX_OK if they agree,
144 * NGX_DECLINED otherwise (no meta, or stale: errno set to ESTALE on mismatch),
145 * so the caller falls back to the origin.
146 */
147static ngx_int_t
148brix_cache_validate_meta(const char *cache_path, const struct stat *st,
149 ngx_log_t *log)
150{
151 brix_cache_meta_t meta;
152
153 if (brix_cache_meta_read(log, cache_path, &meta) != NGX_OK) {
154 return NGX_DECLINED;
155 }
156
157 if (meta.size != (uint64_t) st->st_size
158 || meta.mtime != (uint64_t) st->st_mtime)
159 {
160 errno = ESTALE;
161 return NGX_DECLINED;
162 }
163
164 return NGX_OK;
165}

167/*
168 * brix_cache_open_declined — should brix_cache_open decline before touching disk?
169 *
170 * WHAT: True when this open must never be served from cache — caching disabled
171 * or unconfigured, BRIX_VFS_O_NOCACHE requested, or any write-intent flag
172 * (write/create/trunc/append) present.
173 * WHY: Concentrates the "serve-vs-fall-back-to-origin" gate in one place; this
174 * decision is authz-transparency-sensitive and must not drift.
175 * HOW: Pure predicate over ctx state and flags; no side effects.
176 */
177static ngx_int_t
178brix_cache_open_declined(const brix_vfs_ctx_t *ctx, ngx_uint_t flags)
179{
180 return ctx == NULL || !ctx->cache_enabled
181 || ctx->cache_root_canon == NULL || ctx->cache_root_canon[0] == '\0'
182 || (flags & BRIX_VFS_O_NOCACHE)
183 || (flags & (BRIX_VFS_O_WRITE | BRIX_VFS_O_CREATE
184 | BRIX_VFS_O_TRUNC | BRIX_VFS_O_APPEND));
185}

187/*
188 * brix_cache_hit_fresh — is cache_path a regular, origin-fresh cache hit?
189 *
190 * WHAT: Stats key through the cache storage driver (inst) and validates the
191 * result against the .meta sidecar for cache_path. Returns NGX_OK on a
192 * usable hit, NGX_DECLINED on a miss / non-regular / stale entry, or
193 * NGX_ERROR on a hard stat error (errno preserved from the driver).
194 * WHY: Extracts the freshness decision — which entries serve vs fall back to
195 * the origin — so the ordering (stat, then meta) is stated exactly once.
196 * HOW: Driver stat → is_reg check → synthesize struct stat from the snapshot
197 * → resolve the sidecar path → brix_cache_validate_meta.
198 */
199static ngx_int_t
200brix_cache_hit_fresh(const brix_vfs_ctx_t *ctx, brix_sd_instance_t *inst,
201 const char *key, const char *cache_path)
202{
203 brix_sd_stat_t sd_st;
204 struct stat st;
205 const char *state_root;
206 char sidecar[PATH_MAX];
207
208 if (inst->driver->stat(inst, key, &sd_st) != NGX_OK) {
209 return (errno == ENOENT || errno == ENOTDIR) ? NGX_DECLINED
210 : NGX_ERROR;
211 }
212 if (!sd_st.is_reg) {
213 return NGX_DECLINED;
214 }
215
216 /* Validate against the .meta sidecar (origin freshness) at the POSIX state
217 * path (== cache_path for a co-located cache). Synthesize a struct stat
218 * from the driver snapshot. */
219 ngx_memzero(&st, sizeof(st));
220 st.st_size = sd_st.size;
221 st.st_mtime = sd_st.mtime;
222 st.st_mode = S_IFREG;
223
224 state_root = brix_cache_state_root_by_root(ctx->cache_root_canon);
225 if (state_root == NULL
226 || brix_cache_sidecar_path(ctx->cache_root_canon, state_root,
227 cache_path, sidecar, sizeof(sidecar)) != 0
228 || brix_cache_validate_meta(sidecar, &st, ctx->log) != NGX_OK)
229 {
230 return NGX_DECLINED; /* obj not opened yet — nothing to close */
231 }
232
233 return NGX_OK;
234}

236/*
237 * brix_cache_open_adopt — open a validated cache entry and adopt its fd.
238 *
239 * WHAT: Opens key through the storage driver read-only and hands the object to
240 * the VFS via brix_vfs_adopt_obj, publishing *fh_out on success. On an
241 * open failure returns NGX_DECLINED (miss) or NGX_ERROR; on an adopt
242 * failure it closes/frees the object and re-raises the original errno.
243 * WHY: Isolates the resource-owning step (open → adopt, with cleanup on the
244 * error edge) from the freshness decision above it.
245 * HOW: driver->open → on NULL classify errno; else brix_vfs_adopt_obj, undo on
246 * failure.
247 */
248static ngx_int_t
249brix_cache_open_adopt(brix_vfs_ctx_t *ctx, brix_sd_instance_t *inst,
250 const char *key, brix_vfs_file_t **fh_out)
251{
252 brix_sd_obj_t *o;
253 ngx_int_t rc;
254 int e = 0;
255
256 o = inst->driver->open(inst, key, BRIX_SD_O_READ, 0, &e);
257 if (o == NULL) {
258 errno = e;
259 return (e == ENOENT || e == ENOTDIR) ? NGX_DECLINED : NGX_ERROR;
260 }
261
262 rc = brix_vfs_adopt_obj(ctx, key, o, 0 /* read-only */, fh_out);
263 if (rc != NGX_OK) {
264 int err = errno;
265 (void) inst->driver->close(o);
266 if (o->heap_shell) {
267 free(o);
268 }
269 errno = err;
270 }
271 return rc;
272}

274/*
275 * brix_cache_open — VFS cache-open hook: serve a read from the cache if able.
276 *
277 * Declines (NGX_DECLINED) when caching is disabled, BRIX_VFS_O_NOCACHE is
278 * set, or any write/create/trunc/append flag is present, and on a cache miss,
279 * not-ready file, or stale meta. On a validated hit it opens the cache file
280 * read-only and adopts the fd via brix_vfs_adopt_fd(), returning *fh_out and
281 * NGX_OK. Returns NGX_ERROR on hard I/O errors (errno set).
282 */
283ngx_int_t
284brix_cache_open(brix_vfs_ctx_t *ctx, ngx_uint_t flags,
285 brix_vfs_file_t **fh_out)
286{
287 char cache_path[PATH_MAX];
288 const char *resolved;
289 const char *key;
290 brix_sd_instance_t *inst;
291 ngx_int_t rc;
292
293 if (fh_out == NULL) {
294 errno = EINVAL;
295 return NGX_ERROR;
296 }
297 *fh_out = NULL;
298
299 if (brix_cache_open_declined(ctx, flags)) {
300 return NGX_DECLINED;
301 }
302
303 resolved = brix_vfs_ctx_path(ctx);
304 if (brix_cache_path_for_resolved(ctx->cache_root_canon,
305 ctx->root_canon, resolved,
306 cache_path, sizeof(cache_path))
307 != NGX_OK)
308 {
309 return NGX_DECLINED;
310 }
311
312 /*
313 * Exclusively-VFS hit-serve: the cache file is opened through the cache
314 * STORAGE driver (POSIX driver on the cache rootfd by default, or a configured
315 * backend), keyed on the cache namespace. The driver owns confinement; for a
316 * driver-backed cache the bytes live in the driver's namespace, not as a POSIX
317 * file. The normal read path memory-serves when the backend cannot sendfile.
318 */
319 inst = brix_cache_storage_by_root(ctx->cache_root_canon);
320 if (inst == NULL) {
321 return NGX_DECLINED; /* no cache storage on this root */
322 }
323
324 /* key = the suffix under cache_root (what the driver keys its namespace on). */
325 key = cache_path + ngx_strlen(ctx->cache_root_canon);
326 if (key[0] != '/') {
327 return NGX_DECLINED;
328 }
329
330 rc = brix_cache_hit_fresh(ctx, inst, key, cache_path);
331 if (rc != NGX_OK) {
332 return rc;
333 }
334
335 return brix_cache_open_adopt(ctx, inst, key, fh_out);
336}

339 * brix_cache_record_access — refresh a cache file's atime for LRU eviction.
340 *
341 * Touches cache_path's access time (atime only, via utimensat with UTIME_NOW /
342 * UTIME_OMIT) so the eviction pass in evict_policy.c sees it as recently used.
343 * The bytes argument is currently unused. Returns NGX_OK on success or where
344 * UTIME_OMIT is unavailable, NGX_ERROR on bad args or a failed utimensat.
345 */
346ngx_int_t
347brix_cache_record_access(const char *cache_path, size_t bytes,
348 ngx_log_t *log)
349{
350 (void) bytes;
351
352 if (cache_path == NULL || cache_path[0] == '\0') {
353 errno = EINVAL;
354 return NGX_ERROR;
355 }
356
357#if defined(UTIME_OMIT)
358 {
359 struct timespec ts[2];
360
361 ts[0].tv_sec = 0;
362 ts[0].tv_nsec = UTIME_NOW;
363 ts[1].tv_sec = 0;
364 ts[1].tv_nsec = UTIME_OMIT;
365
366 if (utimensat(AT_FDCWD, cache_path, ts, 0) != 0) {
367 if (log != NULL) {
368 ngx_log_error(NGX_LOG_DEBUG, log, errno,
369 "brix: cache access timestamp update failed \"%s\"",
370 cache_path);
371 }
372 return NGX_ERROR;
373 }
374 }
375#else
376 (void) log;
377#endif
378
379 return NGX_OK;
380}
