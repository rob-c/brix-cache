1/*
2 * proto_exclusive.c - enforce "one brix protocol per location, per port".
3 *
4 * WHAT: config-load validation that at most one brix protocol (webdav, s3,
5 * cvmfs) is enabled in any one location, and that all brix-enabled
6 * locations under a single listen port speak the same protocol.
7 * WHY: the unified storage directives (brix_cache_store, brix_export, ...)
8 * are per-location; a single owning protocol keeps their meaning
9 * unambiguous, and one-protocol-per-port is the deployment model.
10 * HOW: called from the WebDAV postconfiguration (which runs after ALL module
11 * merges AND after nginx has finalised each server's location structures,
12 * so every enable flag is final). Walks cmcf->ports -> addrs -> servers;
13 * for each server it reads the server-level conf and then the finalised
14 * location structures (static_locations tree, regex_locations and
15 * named_locations), OR-ing each protocol's enable flag into a per-
16 * location, per-server and per-port mask. More than one bit set at any
17 * level is a configuration error. brix_scvmfs is a layer on cvmfs, not a
18 * second protocol; the stream-plane handoff mux is out of scope (stream,
19 * not http).
20 *
21 * Note on structures: by postconfiguration time nginx has already run
22 * ngx_http_init_locations / ngx_http_init_static_location_trees, which
23 * split named + regex locations out of clcf->locations into separate
24 * arrays and reorganise the prefix locations into clcf->static_locations.
25 * Walking the raw clcf->locations queue is therefore unreliable; this
26 * walker reads the finalised structures instead.
27 */

37#define BRIX_PROTO_NONE 0u
38#define BRIX_PROTO_WEBDAV (1u << 0)
39#define BRIX_PROTO_S3 (1u << 1)
40#define BRIX_PROTO_CVMFS (1u << 2)

43/*
44 * WHAT: Human-readable directive name for a single protocol bit.
45 * WHY: Error diagnostics name the offending protocols by their directive.
46 * HOW: Tests the bits in a fixed order; the caller passes a mask that has
47 * the wanted protocol bit set (extra bits are ignored).
48 */
49static const char *
50brix_proto_name(ngx_uint_t bit)
51{
52 if (bit & BRIX_PROTO_WEBDAV) {
53 return "brix_webdav";
54 }
55 if (bit & BRIX_PROTO_S3) {
56 return "brix_s3";
57 }
58 return "brix_cvmfs";
59}

62/*
63 * WHAT: The set of brix protocols this location's conf array enables.
64 * WHY: Each protocol keeps its enable flag in its own loc-conf; the walker
65 * needs one combined mask per location.
66 * HOW: Reads each module's loc-conf via its ctx_index and OR-s the bit for
67 * every protocol whose enable flag merged to on.
68 */
69static ngx_uint_t
70brix_proto_mask(void **loc_conf)
71{
72 ngx_uint_t mask = BRIX_PROTO_NONE;
73 ngx_http_brix_webdav_loc_conf_t *w;
74 ngx_http_s3_loc_conf_t *s;
75 ngx_http_brix_cvmfs_loc_conf_t *c;
76
77 w = loc_conf[ngx_http_brix_webdav_module.ctx_index];
78 s = loc_conf[ngx_http_brix_s3_module.ctx_index];
79 c = loc_conf[ngx_http_brix_cvmfs_module.ctx_index];
80
81 if (w != NULL && w->common.enable == 1) {
82 mask |= BRIX_PROTO_WEBDAV;
83 }
84 if (s != NULL && s->common.enable == 1) {
85 mask |= BRIX_PROTO_S3;
86 }
87 if (c != NULL && c->cvmfs.enable == 1) {
88 mask |= BRIX_PROTO_CVMFS;
89 }
90 return mask;
91}

94/*
95 * WHAT: Fold one conf level's protocol mask into *server_mask, rejecting a
96 * level that enables more than one protocol itself.
97 * WHY: Every individual location (and the server level) must own at most one
98 * protocol, and its bit must contribute to the per-port aggregate.
99 * HOW: Computes the level's mask; a mask with more than one bit set is a
100 * per-location error naming the level. Otherwise OR-s it into the total.
101 */
102static ngx_int_t
103brix_proto_add_level(ngx_conf_t *cf, void **loc_conf, ngx_str_t *name,
104 ngx_uint_t *server_mask)
105{
106 ngx_uint_t mask;
107
108 mask = brix_proto_mask(loc_conf);
109 if (mask & (mask - 1)) { /* more than one bit set */
110 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
111 "two brix protocols enabled in location \"%V\" - "
112 "one brix protocol per location", name);
113 return NGX_ERROR;
114 }
115 *server_mask |= mask;
116 return NGX_OK;
117}

125/*
126 * WHAT: Fold a NULL-terminated array of location confs into *server_mask.
127 * WHY: Both regex (clcf->regex_locations) and named (cscf->named_locations)
128 * locations are stored this way, apart from the static-location tree.
129 * HOW: Walks each entry as a full location (recursing into its own subtree).
130 */
131static ngx_int_t
132brix_proto_walk_loc_array(ngx_conf_t *cf, ngx_http_core_loc_conf_t **arr,
133 ngx_uint_t *server_mask)
134{
135 ngx_uint_t i;
136
137 if (arr == NULL) {
138 return NGX_OK;
139 }
140 for (i = 0; arr[i] != NULL; i++) {
141 if (brix_proto_walk_location(cf, arr[i], server_mask) != NGX_OK) {
142 return NGX_ERROR;
143 }
144 }
145 return NGX_OK;
146}

149/*
150 * WHAT: Recurse the finalised static-location tree, folding every node's
151 * location into *server_mask.
152 * WHY: After ngx_http_init_static_location_trees the prefix/exact locations
153 * live only in this tree, not in a flat queue.
154 * HOW: Visits each node's exact and inclusive location, then descends the
155 * binary-search siblings (left/right) and the nested subtree (tree).
156 */
157static ngx_int_t
158brix_proto_walk_tree(ngx_conf_t *cf, ngx_http_location_tree_node_t *node,
159 ngx_uint_t *server_mask)
160{
161 if (node == NULL) {
162 return NGX_OK;
163 }
164
165 if (node->exact != NULL
166 && brix_proto_walk_location(cf, node->exact, server_mask) != NGX_OK)
167 {
168 return NGX_ERROR;
169 }
170 if (node->inclusive != NULL
171 && brix_proto_walk_location(cf, node->inclusive, server_mask) != NGX_OK)
172 {
173 return NGX_ERROR;
174 }
175
176 if (brix_proto_walk_tree(cf, node->left, server_mask) != NGX_OK
177 || brix_proto_walk_tree(cf, node->tree, server_mask) != NGX_OK
178 || brix_proto_walk_tree(cf, node->right, server_mask) != NGX_OK)
179 {
180 return NGX_ERROR;
181 }
182 return NGX_OK;
183}

186/*
187 * WHAT: Fold one location and all of its descendants into *server_mask.
188 * WHY: A location owns its own protocol and may contain nested prefix, regex
189 * and physically-nested child locations, all of which count toward the
190 * per-port aggregate.
191 * HOW: Adds this location's own mask, then descends its static-location tree
192 * and its regex-location array (each a full location conf to recurse on).
193 */
194static ngx_int_t
195brix_proto_walk_location(ngx_conf_t *cf, ngx_http_core_loc_conf_t *clcf,
196 ngx_uint_t *server_mask)
197{
198 if (brix_proto_add_level(cf, clcf->loc_conf, &clcf->name, server_mask)
199 != NGX_OK)
200 {
201 return NGX_ERROR;
202 }
203
204 if (brix_proto_walk_tree(cf, clcf->static_locations, server_mask)
205 != NGX_OK)
206 {
207 return NGX_ERROR;
208 }
209
210#if (NGX_PCRE)
211 if (brix_proto_walk_loc_array(cf, clcf->regex_locations, server_mask)
212 != NGX_OK)
213 {
214 return NGX_ERROR;
215 }
216#endif
217 return NGX_OK;
218}

221/*
222 * WHAT: Fold one server's whole location structure into *mask.
223 * WHY: The per-port check aggregates over the servers bound to that port; a
224 * server-level enable (a loc-conf directive outside any location) counts
225 * too.
226 * HOW: Reads the server-level conf array (the server's core loc-conf leaves
227 * loc_conf NULL, so the server ctx array is the source of truth), then
228 * walks the server's static-location tree, regex locations and named
229 * (@name) locations.
230 */
231static ngx_int_t
232brix_proto_server_mask(ngx_conf_t *cf, ngx_http_core_srv_conf_t *cscf,
233 ngx_uint_t *mask)
234{
235 ngx_http_core_loc_conf_t *clcf;
236
237 if (brix_proto_add_level(cf, cscf->ctx->loc_conf, &cscf->server_name, mask)
238 != NGX_OK)
239 {
240 return NGX_ERROR;
241 }
242
243 clcf = cscf->ctx->loc_conf[ngx_http_core_module.ctx_index];
244
245 if (brix_proto_walk_tree(cf, clcf->static_locations, mask) != NGX_OK) {
246 return NGX_ERROR;
247 }
248
249#if (NGX_PCRE)
250 if (brix_proto_walk_loc_array(cf, clcf->regex_locations, mask) != NGX_OK) {
251 return NGX_ERROR;
252 }
253#endif
254
255 return brix_proto_walk_loc_array(cf, cscf->named_locations, mask);
256}

259/*
260 * WHAT: Reject a configuration that mixes brix protocols per location or per
261 * listen port.
262 * WHY: See the file header — one owning protocol per location keeps the
263 * unified storage directives unambiguous, and one protocol per port is
264 * the deployment model.
265 * HOW: Walks cmcf->ports; for each port, aggregates the protocol mask of every
266 * server bound to every address on it. A location owning more than one
267 * protocol fails inside the walk; a port whose aggregate mask has more
268 * than one bit set fails here with the two conflicting protocol names.
269 */
270ngx_int_t
271brix_http_proto_exclusive_check(ngx_conf_t *cf)
272{
273 ngx_uint_t p, a, s;
274 ngx_uint_t port_mask, srv_mask;
275 ngx_http_conf_port_t *port;
276 ngx_http_conf_addr_t *addr;
277 ngx_http_core_srv_conf_t **cscf;
278 ngx_http_core_main_conf_t *cmcf;
279
280 cmcf = ngx_http_conf_get_module_main_conf(cf, ngx_http_core_module);
281 if (cmcf->ports == NULL) {
282 return NGX_OK;
283 }
284
285 port = cmcf->ports->elts;
286 for (p = 0; p < cmcf->ports->nelts; p++) {
287 port_mask = BRIX_PROTO_NONE;
288 addr = port[p].addrs.elts;
289
290 for (a = 0; a < port[p].addrs.nelts; a++) {
291 cscf = addr[a].servers.elts;
292 for (s = 0; s < addr[a].servers.nelts; s++) {
293 srv_mask = BRIX_PROTO_NONE;
294 if (brix_proto_server_mask(cf, cscf[s], &srv_mask) != NGX_OK) {
295 return NGX_ERROR;
296 }
297 port_mask |= srv_mask;
298 }
299 }
300
301 if (port_mask & (port_mask - 1)) { /* more than one bit set */
302 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
303 "%s and %s both enabled under listen port %ui - "
304 "one brix protocol per port",
305 brix_proto_name(port_mask & (port_mask - 1)),
306 brix_proto_name(port_mask & ~(port_mask - 1)),
307 (ngx_uint_t) port[p].port); /* host order (ngx_inet_get_port) */
308 return NGX_ERROR;
309 }
310 }
311 return NGX_OK;
312}
