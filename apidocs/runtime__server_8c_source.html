31static int
32brix_server_storage_backend_is_remote(const ngx_str_t *sb)
33{
34 return (sb->len > sizeof("root://") - 1
35 && ngx_strncmp(sb->data, "root://", sizeof("root://") - 1) == 0)
36 || (sb->len > sizeof("roots://") - 1
37 && ngx_strncmp(sb->data, "roots://", sizeof("roots://") - 1) == 0)
38 || (sb->len > sizeof("http://") - 1
39 && ngx_strncmp(sb->data, "http://", sizeof("http://") - 1) == 0)
40 || (sb->len > sizeof("https://") - 1
41 && ngx_strncmp(sb->data, "https://", sizeof("https://") - 1) == 0);
42}

44/*
45 * brix_server_guard_remote_authz — config-time guardrail for an origin-scheme
46 * backend whose export setup will never run (P80.5, finding 1.5).
47 *
48 * WHAT: Rejects at config time the combination of an origin-scheme
49 * brix_storage_backend (s3/http/xroot/...), configured authorization
50 * rules (brix_authdb / brix_require_vo / brix_inherit_parent_group),
51 * and a server mode that skips brix_server_setup_export — the exact
52 * gate below (brix_server_has_runtime_export).
53 *
54 * WHY: When the export setup is skipped, root_canon stays "/" and the path
55 * join produces "//<wire-path>" while every authdb/VO rule was
56 * canonicalized to "/<path>" at load — nothing ever matches and every
57 * op dies "3010 authdb denied" with zero hint at the cause. The lab
58 * burned hours on this silent everything-denied server; make it a loud
59 * config error instead.
60 *
61 * HOW: Pure predicate over already-merged conf fields; emits one EMERG
62 * naming the "//path vs /path" mechanics and which engine breaks on
63 * which side (xrdacc authorizes the LOGICAL wire path; native authdb /
64 * VO ACLs authorize the RESOLVED backing path).
65 */
66static ngx_int_t
67brix_server_guard_remote_authz(ngx_conf_t *cf,
68 ngx_stream_brix_srv_conf_t *xcf)
69{
70 int has_rules;
71
72 if (brix_server_has_runtime_export(xcf)
73 || !brix_storage_backend_is_remote(&xcf->common))
74 {
75 return NGX_OK;
76 }
77
78 has_rules = xcf->authdb.len > 0
79 || (xcf->vo_rules != NULL && xcf->vo_rules->nelts > 0)
80 || (xcf->group_rules != NULL && xcf->group_rules->nelts > 0);
81 if (!has_rules) {
82 return NGX_OK;
83 }
84
85 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
86 "brix_storage_backend \"%V\" with authorization rules "
87 "(brix_authdb/brix_require_vo/brix_inherit_parent_group) requires a "
88 "runtime export: this server mode (manager/supervisor/proxy) skips "
89 "brix_export setup, so the authz gate would check \"//<path>\" "
90 "(root_canon \"/\" + wire path) against rules canonicalized to "
91 "\"/<path>\" — nothing matches and EVERY request is denied 3010. "
92 "xrdacc rules break on the logical wire path; native authdb/VO ACL "
93 "rules break on the resolved backing path. Set brix_export to a real "
94 "local directory (it aligns both sides), or drop the authz rules "
95 "from this server block",
96 &xcf->common.storage_backend);
97 return NGX_ERROR;
98}

100static ngx_int_t
101brix_server_set_storage_credential(ngx_conf_t *cf,
102 ngx_stream_brix_srv_conf_t *xcf)
103{
104 char cred_z[256];
105 char bearer[4096];
106 const brix_credential_t *cred;
107 brix_vfs_backend_cred_t bcred;
108
109 if (xcf->common.storage_credential.len == 0) {
110 return NGX_OK;
111 }
112 ngx_cpystrn((u_char *) cred_z, xcf->common.storage_credential.data,
113 ngx_min(xcf->common.storage_credential.len + 1,
114 sizeof(cred_z)));
115 cred = brix_credential_lookup(cred_z);
116 if (cred == NULL) {
117 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
118 "brix_storage_credential: no brix_credential \"%V\"",
119 &xcf->common.storage_credential);
120 return NGX_ERROR;
121 }
122 if (brix_credential_to_backend_cred(cred, bearer, sizeof(bearer),
123 &bcred, cf->log) != NGX_OK)
124 {
125 return NGX_ERROR;
126 }
127 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
128 "brix: backend credential \"%V\" for \"%s\": gsi=%s key=%s bearer=%s",
129 &xcf->common.storage_credential, xcf->common.root_canon,
130 bcred.x509_proxy ? bcred.x509_proxy : "(none)",
131 bcred.x509_key ? bcred.x509_key : "(in-proxy/none)",
132 bcred.bearer ? "set" : "(none)");
133 brix_vfs_backend_set_credential(xcf->common.root_canon, &bcred);
134 return NGX_OK;

146 return NGX_OK;
147 }

170 xcf->cache_origin_bearer.data = bp;
171 xcf->cache_origin_bearer.len = bl;
172 }
173 if (cred->x509_proxy.len > 0) {
174 xcf->cache_origin_x509_proxy = cred->x509_proxy;
175 } else if (cred->x509_cert.len > 0) {
176 xcf->cache_origin_x509_proxy = cred->x509_cert;
177 xcf->cache_origin_x509_key = cred->x509_key;
178 }
179 if (cred->ca_dir.len > 0) {
180 xcf->cache_origin_ca_dir = cred->ca_dir;
181 }
182 return NGX_OK;

238 return NGX_ERROR;

245 return NGX_ERROR;

274static ngx_int_t
275brix_server_validate_cache_watermarks(ngx_conf_t *cf,
276 ngx_stream_brix_srv_conf_t *xcf)
277{
278 if (xcf->cache_eviction_threshold == 0
279 || xcf->cache_eviction_threshold >= 1000000)
280 {
281 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
282 "brix_cache_eviction_threshold must be greater than 0 "
283 "and less than 1.0");
284 return NGX_ERROR;
285 }
286 /* The percent-grammar spelling first, so a bad brix_cache_evict_at/_to
287 * pair gets a percent-worded error instead of the ppm one below (the seed
288 * in server_conf_merge_security.c multiplies by 10000). */
289 if ((xcf->common.cache_evict_at != NGX_CONF_UNSET_UINT
290 && (xcf->common.cache_evict_at == 0
291 || xcf->common.cache_evict_at >= 100))
292 || (xcf->common.cache_evict_to != NGX_CONF_UNSET_UINT
293 && (xcf->common.cache_evict_to == 0
294 || xcf->common.cache_evict_to
295 >= (xcf->common.cache_evict_at == NGX_CONF_UNSET_UINT
296 ? 90 : xcf->common.cache_evict_at))))
297 {
298 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
299 "brix_cache_evict_to must be a percent greater than 0 and "
300 "less than brix_cache_evict_at (which must be < 100)");
301 return NGX_ERROR;
302 }
303 if (xcf->reaper.high_watermark == 0
304 || xcf->reaper.high_watermark >= 1000000
305 || xcf->reaper.low_watermark == 0
306 || xcf->reaper.low_watermark >= xcf->reaper.high_watermark)
307 {
308 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
309 "brix_cache_low_watermark must be greater than 0 and less "
310 "than brix_cache_high_watermark (which must be < 1.0)");
311 return NGX_ERROR;
312 }
313 return NGX_OK;
314}

316static ngx_int_t
317brix_server_validate_cache(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
318{
319 if (!xcf->cache) {
320 return NGX_OK;
321 }
322 if (xcf->common.allow_write && !xcf->wt.enable) {
323 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
324 "brix_cache is read-only and requires "
325 "brix_allow_write off (or enable brix_write_through)");
326 return NGX_ERROR;
327 }
328 if (brix_server_validate_cache_stage_backend(cf, xcf) != NGX_OK) {
329 return NGX_ERROR;
330 }
331 if (xcf->cache_root.len == 0
332 || !brix_server_storage_backend_is_remote(&xcf->common.storage_backend))
333 {
334 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
335 "brix_cache on requires brix_cache_export and a remote "
336 "brix_storage_backend (root://host:port); the retired "
337 "brix_cache_origin model is the tier grammar now "
338 "(brix_storage_backend + brix_cache_store)");
339 return NGX_ERROR;
340 }
341 if (brix_validate_path(cf, "brix_cache_export", &xcf->cache_root,
342 BRIX_PATH_DIRECTORY, R_OK | W_OK | X_OK)
343 != NGX_OK)
344 {
345 return NGX_ERROR;
346 }
347 if (xcf->cache_lock_timeout <= 0) {
348 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
349 "brix_cache_lock_timeout must be greater than zero");
350 return NGX_ERROR;
351 }
352 if (brix_server_validate_cache_watermarks(cf, xcf) != NGX_OK) {
353 return NGX_ERROR;
354 }
355 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
356 "brix: cache enabled root=%V origin=%V tls=%s "
357 "lock_timeout=%ds eviction_threshold=0.%06ui",
358 &xcf->cache_root, &xcf->cache_origin,
359 xcf->cache_origin_tls ? "on" : "off",
360 (int) xcf->cache_lock_timeout,
361 xcf->cache_eviction_threshold);
362 return NGX_OK;
363}

429/* Prepare one server block at postconfiguration: validate the configured root
430 * is an existing, accessible directory (access mode matching the write policy)
431 * and check the cache configuration, before the block accepts connections.
432 * Returns NGX_OK, or NGX_ERROR (emerg-logged) on any invalid resource. */
433ngx_int_t
434brix_config_prepare_server(ngx_conf_t *cf,
435 ngx_stream_brix_srv_conf_t *xcf)
436{
437 /* Hard read-only switch: force allow_write off before any allow_write-dependent
438 * setup, so every write gate rejects (root:// require_write, write-open, ...). */
439 brix_shared_apply_read_only(&xcf->common, cf->log);
440
441 /* Phase-4b: a GSI tap proxy must capture the client's delegated proxy to
442 * present it upstream — auto-enable delegation receipt if the admin didn't. */
443 if (xcf->proxy.enable && xcf->proxy.auth == BRIX_PROXY_AUTH_GSI
444 && !xcf->tpc_delegate)
445 {
446 xcf->tpc_delegate = 1;
447 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
448 "brix_tap_proxy_auth gsi: enabling GSI proxy delegation capture");
449 }
450
451 if (brix_server_guard_remote_authz(cf, xcf) != NGX_OK) {
452 return NGX_ERROR;
453 }
454 if (brix_server_setup_export(cf, xcf) != NGX_OK) {
455 return NGX_ERROR;
456 }
457 if (brix_server_validate_cache(cf, xcf) != NGX_OK) {
458 return NGX_ERROR;
459 }
460 if (brix_server_validate_wt_stage(cf, xcf) != NGX_OK) {
461 return NGX_ERROR;
462 }
463 if (brix_server_setup_logging(cf, xcf) != NGX_OK) {
464 return NGX_ERROR;
465 }
466 if (brix_server_setup_tls(cf, xcf) != NGX_OK) {
467 return NGX_ERROR;
468 }
469
470 return NGX_OK;
471}
