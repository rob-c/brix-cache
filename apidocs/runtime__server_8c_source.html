23static int
24brix_server_has_runtime_export(const ngx_stream_brix_srv_conf_t *xcf)
25{
26 return !xcf->manager_mode && !xcf->caps.supervisor
27 && xcf->manager_map == NULL && !xcf->proxy.enable;

30static int
31brix_server_storage_backend_is_remote(const ngx_str_t *sb)
32{
33 return (sb->len > sizeof("root://") - 1
34 && ngx_strncmp(sb->data, "root://", sizeof("root://") - 1) == 0)
35 || (sb->len > sizeof("roots://") - 1
36 && ngx_strncmp(sb->data, "roots://", sizeof("roots://") - 1) == 0)
37 || (sb->len > sizeof("http://") - 1
38 && ngx_strncmp(sb->data, "http://", sizeof("http://") - 1) == 0)
39 || (sb->len > sizeof("https://") - 1
40 && ngx_strncmp(sb->data, "https://", sizeof("https://") - 1) == 0);
41}

43/*
44 * brix_server_guard_remote_authz — config-time guardrail for an origin-scheme
45 * backend whose export setup will never run (P80.5, finding 1.5).
46 *
47 * WHAT: Rejects at config time the combination of an origin-scheme
48 * brix_storage_backend (s3/http/xroot/...), configured authorization
49 * rules (brix_authdb / brix_require_vo / brix_inherit_parent_group),
50 * and a server mode that skips brix_server_setup_export — the exact
51 * gate below (brix_server_has_runtime_export).
52 *
53 * WHY: When the export setup is skipped, root_canon stays "/" and the path
54 * join produces "//<wire-path>" while every authdb/VO rule was
55 * canonicalized to "/<path>" at load — nothing ever matches and every
56 * op dies "3010 authdb denied" with zero hint at the cause. The lab
57 * burned hours on this silent everything-denied server; make it a loud
58 * config error instead.
59 *
60 * HOW: Pure predicate over already-merged conf fields; emits one EMERG
61 * naming the "//path vs /path" mechanics and which engine breaks on
62 * which side (xrdacc authorizes the LOGICAL wire path; native authdb /
63 * VO ACLs authorize the RESOLVED backing path).
64 */
65static ngx_int_t
66brix_server_guard_remote_authz(ngx_conf_t *cf,
67 ngx_stream_brix_srv_conf_t *xcf)
68{
69 int has_rules;
70
71 if (brix_server_has_runtime_export(xcf)
72 || !brix_storage_backend_is_remote(&xcf->common))
73 {
74 return NGX_OK;
75 }
76
77 has_rules = xcf->authdb.len > 0
78 || (xcf->vo_rules != NULL && xcf->vo_rules->nelts > 0)
79 || (xcf->group_rules != NULL && xcf->group_rules->nelts > 0);
80 if (!has_rules) {
81 return NGX_OK;
82 }
83
84 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
85 "brix_storage_backend \"%V\" with authorization rules "
86 "(brix_authdb/brix_require_vo/brix_inherit_parent_group) requires a "
87 "runtime export: this server mode (manager/supervisor/proxy) skips "
88 "brix_export setup, so the authz gate would check \"//<path>\" "
89 "(root_canon \"/\" + wire path) against rules canonicalized to "
90 "\"/<path>\" — nothing matches and EVERY request is denied 3010. "
91 "xrdacc rules break on the logical wire path; native authdb/VO ACL "
92 "rules break on the resolved backing path. Set brix_export to a real "
93 "local directory (it aligns both sides), or drop the authz rules "
94 "from this server block",
95 &xcf->common.storage_backend);
96 return NGX_ERROR;
97}

99static ngx_int_t
100brix_server_set_storage_credential(ngx_conf_t *cf,
101 ngx_stream_brix_srv_conf_t *xcf)
102{
103 char cred_z[256];
104 char bearer[4096];
105 const brix_credential_t *cred;
106 brix_vfs_backend_cred_t bcred;
107
108 if (xcf->common.storage_credential.len == 0) {
109 return NGX_OK;
110 }
111 ngx_cpystrn((u_char *) cred_z, xcf->common.storage_credential.data,

122 &bcred, cf->log) != NGX_OK)
123 {
124 return NGX_ERROR;
125 }

127 "brix: backend credential \"%V\" for \"%s\": gsi=%s key=%s bearer=%s",
128 &xcf->common.storage_credential, xcf->common.root_canon,
129 bcred.x509_proxy ? bcred.x509_proxy : "(none)",
130 bcred.x509_key ? bcred.x509_key : "(in-proxy/none)",
131 bcred.bearer ? "set" : "(none)");

133 return NGX_OK;

160 }
161 if (bearer[0] != '\0') {
162 size_t bl = ngx_strlen(bearer);
163 u_char *bp = ngx_pnalloc(cf->pool, bl + 1);
164

169 xcf->cache_origin_bearer.data = bp;
170 xcf->cache_origin_bearer.len = bl;
171 }
172 if (cred->x509_proxy.len > 0) {
173 xcf->cache_origin_x509_proxy = cred->x509_proxy;
174 } else if (cred->x509_cert.len > 0) {
175 xcf->cache_origin_x509_proxy = cred->x509_cert;
176 xcf->cache_origin_x509_key = cred->x509_key;
177 }
178 if (cred->ca_dir.len > 0) {
179 xcf->cache_origin_ca_dir = cred->ca_dir;
180 }
181 return NGX_OK;

244 return NGX_ERROR;

273static ngx_int_t
274brix_server_validate_cache_watermarks(ngx_conf_t *cf,
275 ngx_stream_brix_srv_conf_t *xcf)
276{
277 if (xcf->cache_eviction_threshold == 0
278 || xcf->cache_eviction_threshold >= 1000000)
279 {
280 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
281 "brix_cache_eviction_threshold must be greater than 0 "
282 "and less than 1.0");
283 return NGX_ERROR;
284 }
285 /* The percent-grammar spelling first, so a bad brix_cache_evict_at/_to
286 * pair gets a percent-worded error instead of the ppm one below (the seed
287 * in server_conf_merge_security.c multiplies by 10000). */
288 if ((xcf->common.cache_evict_at != NGX_CONF_UNSET_UINT
289 && (xcf->common.cache_evict_at == 0
290 || xcf->common.cache_evict_at >= 100))
291 || (xcf->common.cache_evict_to != NGX_CONF_UNSET_UINT
292 && (xcf->common.cache_evict_to == 0
293 || xcf->common.cache_evict_to
294 >= (xcf->common.cache_evict_at == NGX_CONF_UNSET_UINT
295 ? 90 : xcf->common.cache_evict_at))))
296 {
297 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
298 "brix_cache_evict_to must be a percent greater than 0 and "
299 "less than brix_cache_evict_at (which must be < 100)");
300 return NGX_ERROR;
301 }
302 if (xcf->reaper.high_watermark == 0
303 || xcf->reaper.high_watermark >= 1000000
304 || xcf->reaper.low_watermark == 0
305 || xcf->reaper.low_watermark >= xcf->reaper.high_watermark)
306 {
307 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
308 "brix_cache_low_watermark must be greater than 0 and less "
309 "than brix_cache_high_watermark (which must be < 1.0)");
310 return NGX_ERROR;
311 }
312 return NGX_OK;
313}

315static ngx_int_t
316brix_server_validate_cache(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
317{
318 if (!xcf->cache) {
319 return NGX_OK;
320 }
321 if (xcf->common.allow_write && !xcf->wt.enable) {
322 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
323 "brix_cache is read-only and requires "
324 "brix_allow_write off (or enable brix_write_through)");
325 return NGX_ERROR;

328 return NGX_ERROR;
329 }
330 if (xcf->cache_root.len == 0
331 || !brix_server_storage_backend_is_remote(&xcf->common.storage_backend))
332 {
333 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
334 "brix_cache on requires brix_cache_export and a remote "
335 "brix_storage_backend (root://host:port); the retired "
336 "brix_cache_origin model is the tier grammar now "
337 "(brix_storage_backend + brix_cache_store)");
338 return NGX_ERROR;
339 }
340 if (brix_validate_path(cf, "brix_cache_export", &xcf->cache_root,
341 BRIX_PATH_DIRECTORY, R_OK | W_OK | X_OK)
342 != NGX_OK)
343 {
344 return NGX_ERROR;
345 }
346 if (xcf->cache_lock_timeout <= 0) {
347 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
348 "brix_cache_lock_timeout must be greater than zero");
349 return NGX_ERROR;
350 }
351 if (brix_server_validate_cache_watermarks(cf, xcf) != NGX_OK) {
352 return NGX_ERROR;
353 }
354 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
355 "brix: cache enabled root=%V origin=%V tls=%s "
356 "lock_timeout=%ds eviction_threshold=0.%06ui",
357 &xcf->cache_root, &xcf->cache_origin,
358 xcf->cache_origin_tls ? "on" : "off",
359 (int) xcf->cache_lock_timeout,
360 xcf->cache_eviction_threshold);
361 return NGX_OK;
362}

421#if (NGX_SSL)
422/*
423 * brix_tls_ctx_enable_verify — turn on real peer authentication for an outbound
424 * client SSL_CTX.
425 *
426 * WHAT: Sets SSL_VERIFY_PEER (so a bad or untrusted chain fails the handshake)
427 * and, when `host` is non-empty, pins the expected certificate hostname on
428 * the context's verify parameters (so a valid cert for the WRONG host is
429 * also rejected).
430 * WHY: Loading a trust store without SSL_VERIFY_PEER is inert — OpenSSL clients
431 * default to SSL_VERIFY_NONE, so an on-path attacker presenting any
432 * CA-valid (or self-signed) cert completes the handshake and the proxy /
433 * redirector re-sends kXR_login over the attacker's channel. Chain-only
434 * verification still accepts any CA-valid cert for a different host, so the
435 * name check is folded in via the CTX verify param (mirrors the cache
436 * origin path in fs/cache/origin_connection.c). Applied on the shared CTX
437 * so every connection it spawns inherits verification; a failure then
438 * aborts at the existing `!handshaked` check in the handshake callbacks.
439 * HOW: SSL_CTX_set_verify + X509_VERIFY_PARAM_set1_host on SSL_CTX_get0_param.
440 */
441static ngx_int_t
442brix_tls_ctx_enable_verify(ngx_conf_t *cf, SSL_CTX *ctx, const ngx_str_t *host)
443{
444 SSL_CTX_set_verify(ctx, SSL_VERIFY_PEER, NULL);
445
446 if (host != NULL && host->len > 0 && host->data != NULL) {
447 X509_VERIFY_PARAM *param = SSL_CTX_get0_param(ctx);
448 char hbuf[256];
449 size_t hl = host->len < sizeof(hbuf) - 1
450 ? host->len : sizeof(hbuf) - 1;
451
452 ngx_memcpy(hbuf, host->data, hl);
453 hbuf[hl] = '\0';
454 X509_VERIFY_PARAM_set_hostflags(param,
455 X509_CHECK_FLAG_NO_PARTIAL_WILDCARDS);
456 if (X509_VERIFY_PARAM_set1_host(param, hbuf, 0) != 1) {
457 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
458 "brix: failed to pin upstream TLS verify host \"%V\"", host);
459 return NGX_ERROR;
460 }
461 }
462 return NGX_OK;
463}

466static ngx_int_t
467brix_server_setup_tls(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
468{
469#if (NGX_SSL)
470 if (xcf->proxy.enable && xcf->proxy.upstream_tls) {
471 xcf->proxy.tls_ctx = ngx_pcalloc(cf->pool, sizeof(ngx_ssl_t));
472 if (xcf->proxy.tls_ctx == NULL) {
473 return NGX_ERROR;
474 }
475 xcf->proxy.tls_ctx->log = cf->log;
476 if (ngx_ssl_create(xcf->proxy.tls_ctx,
477 NGX_SSL_TLSv1_2 | NGX_SSL_TLSv1_3,
478 NULL) != NGX_OK)
479 {
480 return NGX_ERROR;
481 }
482 if (xcf->proxy.upstream_tls_ca.len > 0) {
483 if (ngx_ssl_trusted_certificate(cf, xcf->proxy.tls_ctx,
484 &xcf->proxy.upstream_tls_ca,
485 5) != NGX_OK)
486 {
487 return NGX_ERROR;
488 }
489 if (xcf->proxy.upstream_ssl_verify) {
490 if (brix_tls_ctx_enable_verify(cf, xcf->proxy.tls_ctx->ctx,
491 (xcf->proxy.upstream_tls_name.len > 0)
492 ? &xcf->proxy.upstream_tls_name : &xcf->proxy.host)
493 != NGX_OK)
494 {
495 return NGX_ERROR;
496 }
497 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
498 "brix: proxy upstream TLS CA loaded from \"%V\" — peer "
499 "verification (chain + host) enabled",
500 &xcf->proxy.upstream_tls_ca);
501 } else {
502 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
503 "brix: proxy upstream TLS CA loaded from \"%V\" but "
504 "brix_tap_proxy_upstream_tls_verify is off — the peer is "
505 "UNVERIFIED and the hop is MITM-able",
506 &xcf->proxy.upstream_tls_ca);
507 }
508 } else if (xcf->proxy.upstream_ssl_verify) {
509 /* A-1: fail closed — an unauthenticated TLS upstream re-sends the
510 * client's kXR_login over an attacker-controllable channel. */
511 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
512 "brix: brix_tap_proxy_upstream_tls is on but no "
513 "brix_tap_proxy_upstream_tls_ca is set — refusing an "
514 "unauthenticated, MITM-able TLS upstream (set the CA, or "
515 "brix_tap_proxy_upstream_tls_verify off to opt out)");
516 return NGX_ERROR;
517 } else {
518 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
519 "brix: brix_tap_proxy_upstream_tls is on with verification "
520 "explicitly off and no CA — the upstream TLS peer is UNVERIFIED "
521 "and the hop is MITM-able");
522 }
523 }
524 if (xcf->upstream_tls) {
525 xcf->upstream_tls_ctx = ngx_pcalloc(cf->pool, sizeof(ngx_ssl_t));
526 if (xcf->upstream_tls_ctx == NULL) {
527 return NGX_ERROR;
528 }

553 "verification (chain + host) enabled",

555 } else {

575 }
576 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
577 "brix: upstream redirector TLS enabled (kXR_gotoTLS support)");
578 }
579#else
580 (void) cf;
581 (void) xcf;
582#endif
583 return NGX_OK;
584}

586/* Prepare one server block at postconfiguration: validate the configured root
587 * is an existing, accessible directory (access mode matching the write policy)
588 * and check the cache configuration, before the block accepts connections.
589 * Returns NGX_OK, or NGX_ERROR (emerg-logged) on any invalid resource. */
590ngx_int_t
591brix_config_prepare_server(ngx_conf_t *cf,
592 ngx_stream_brix_srv_conf_t *xcf)
593{
594 /* Hard read-only switch: force allow_write off before any allow_write-dependent
595 * setup, so every write gate rejects (root:// require_write, write-open, ...). */
596 brix_shared_apply_read_only(&xcf->common, cf->log);
597
598 /* Phase-4b: a GSI tap proxy must capture the client's delegated proxy to
599 * present it upstream — auto-enable delegation receipt if the admin didn't. */
600 if (xcf->proxy.enable && xcf->proxy.auth == BRIX_PROXY_AUTH_GSI
601 && !xcf->tpc_delegate)
602 {
603 xcf->tpc_delegate = 1;
604 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
605 "brix_tap_proxy_auth gsi: enabling GSI proxy delegation capture");
606 }
607
608 if (brix_server_guard_remote_authz(cf, xcf) != NGX_OK) {
609 return NGX_ERROR;
610 }
611 if (brix_server_setup_export(cf, xcf) != NGX_OK) {
612 return NGX_ERROR;
613 }
614 if (brix_server_validate_cache(cf, xcf) != NGX_OK) {
615 return NGX_ERROR;
616 }
617 if (brix_server_validate_wt_stage(cf, xcf) != NGX_OK) {
618 return NGX_ERROR;
619 }
620 if (brix_server_setup_logging(cf, xcf) != NGX_OK) {
621 return NGX_ERROR;
622 }
623 if (brix_server_setup_tls(cf, xcf) != NGX_OK) {
624 return NGX_ERROR;
625 }
626
627 return NGX_OK;
628}
