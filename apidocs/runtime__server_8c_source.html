16/* Directive setter for a tier store-URL directive: arg[1] = the store URL (into the
17 * ngx_str_t at cmd->offset); args[2..] = trailing credential=/block_size= params
18 * (into the ngx_array_t* at the field offset carried in cmd->post). See header. */
19char *
20brix_conf_set_store_slot(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
21{
22 char *p = conf;
23 ngx_str_t *url = (ngx_str_t *) (p + cmd->offset);
24 ngx_array_t **args = (ngx_array_t **) (p + (uintptr_t) cmd->post);
25 ngx_str_t *value = cf->args->elts;
26 ngx_uint_t i;
27
28 if (url->data != NULL) {
29 return "is duplicate";
30 }
31 *url = value[1]; /* the store URL token */
32
33 if (cf->args->nelts > 2) {
34 *args = ngx_array_create(cf->pool, cf->args->nelts - 2, sizeof(ngx_str_t));
35 if (*args == NULL) {
36 return NGX_CONF_ERROR;
37 }
38 for (i = 2; i < cf->args->nelts; i++) {
39 ngx_str_t *a = ngx_array_push(*args);
40
41 if (a == NULL) {
42 return NGX_CONF_ERROR;
43 }
44 *a = value[i]; /* "credential=..." / "block_size=..." */
45 }
46 }
47 return NGX_CONF_OK;
48}

50/* A LOCAL storage backend NAMES THE EXPORT TREE — the fully composable replacement
51 * for brix_root. Rewrites common->root from the backend URL and anchors root_canon
52 * there. No-op for a remote/non-local backend (root://, tape://, http://, a cache
53 * origin, or none). Shared by all three protocol finalisers; called BEFORE the
54 * export-root prep. Two forms:
55 * posix:<path> → root = <path>; CLEAR the backend (default POSIX driver).
56 * pblock://<path> → root = /<path> (one leading '/' guaranteed: "pblock://x"→
57 * "/x", "pblock:///abs"→"//abs"→/abs); KEEP the "pblock"
58 * driver; create the block-store directory on init if needed. */
59void
60brix_storage_backend_posix_root(ngx_http_brix_shared_conf_t *common)
61{
62 ngx_str_t *sb = &common->storage_backend;
63
64 if (sb->len > sizeof("posix:") - 1
65 && ngx_strncmp(sb->data, "posix:", sizeof("posix:") - 1) == 0)
66 {
67 u_char *p = sb->data + (sizeof("posix:") - 1);
68 size_t plen = sb->len - (sizeof("posix:") - 1);
69
70 /* Accept both posix:<path> and posix://<path>: collapse leading "//" runs to
71 * one '/' so the root is a single canonical path (a stray double slash
72 * propagates into common->root and breaks raw prefix-strip path uses).
73 * "posix://abs" with an absolute path yields "///abs" → "/abs". */
74 while (plen >= 2 && p[0] == '/' && p[1] == '/') {
75 p++;
76 plen--;
77 }
78 common->root.data = p;
79 common->root.len = plen;
80 sb->len = 0; /* default POSIX driver */
81 return;
82 }
83
84 if (sb->len > sizeof("pblock://") - 1
85 && ngx_strncmp(sb->data, "pblock://", sizeof("pblock://") - 1) == 0)
86 {
87 size_t base = sizeof("pblock://") - 1; /* past "pblock://" */
88
89 /* Yield EXACTLY one leading '/': "pblock:///abs" already has it (offset
90 * `base`); "pblock://rel" gains it by keeping the prior '/' (offset base-1).
91 * A double slash would break the write-through's prefix-strip path derivation. */
92 if (sb->data[base] == '/') {
93 common->root.data = sb->data + base;
94 common->root.len = sb->len - base;
95 } else {
96 common->root.data = sb->data + base - 1;
97 common->root.len = sb->len - base + 1;
98 }
99 sb->len = sizeof("pblock") - 1; /* the bare "pblock" driver */
100 (void) brix_mkdir_recursive((const char *) common->root.data, 0755);
101 }
102}

104/* 1 iff the storage backend is REMOTE (the export's bytes live off-box: a root://
105 * origin, http(s)://, s3://, tape://, ceph). For such an export the LOCAL root_canon
106 * is only a namespace anchor — never written — so its export-root prep must not
107 * demand W_OK (a pure remote-backed node defaults root_canon to "/", which is not
108 * writable). Local backends (posix/pblock, or none) return 0 and keep the W_OK
109 * check. Called after brix_storage_backend_posix_root (posix:/pblock:// already
110 * rewritten away). */
111int
112brix_storage_backend_is_remote(const ngx_http_brix_shared_conf_t *common)
113{
114 static const char *const schemes[] = {
115 "root://", "roots://", "http://", "https://", "s3://",
116 "tape://", "frm://", "rados://", "ceph:", "cephfsro:", NULL
117 };
118 const ngx_str_t *sb = &common->storage_backend;
119 int i;
120
121 for (i = 0; schemes[i] != NULL; i++) {
122 size_t n = ngx_strlen(schemes[i]);
123
124 if (sb->len >= n && ngx_strncmp(sb->data, schemes[i], n) == 0) {
125 return 1;
126 }
127 }
128 return 0;
129}

131/* Register the export's phase-64 composable cache/stage tiers (additive over the
132 * storage backend). Parses the cache_store / stage_store URLs (operator errors are
133 * [emerg], failing nginx -t) and records the tier cfg + policy on the backend
134 * registry, which composes the sd_cache / sd_stage decorators per worker. Shared by
135 * all three protocol finalisers (§4.4) - it reads only the common preamble. */
136ngx_int_t
137brix_tier_register_stores(ngx_conf_t *cf, ngx_http_brix_shared_conf_t *common)
138{
139 char err[256];
140
141 /* G8 (P4/§9.4): a nearline (tape) backend is unservable without a cache tier
142 * as the recall target - reject at config time. "frm://" is the tape:// alias. */
143 {
144 const ngx_str_t *sb = &common->storage_backend;
145 int is_nearline =
146 (sb->len > sizeof("tape://") - 1
147 && ngx_strncmp(sb->data, "tape://", sizeof("tape://") - 1) == 0)
148 || (sb->len > sizeof("frm://") - 1
149 && ngx_strncmp(sb->data, "frm://", sizeof("frm://") - 1) == 0);
150
151 if (is_nearline && common->cache_store.len == 0) {
152 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
153 "brix: a \"tape://\"/\"frm://\" backend is nearline and requires "
154 "brix_cache_store (the recall target); add a cache tier");
155 return NGX_ERROR;
156 }
157 }
158
159 if (common->cache_store.len > 0) {
160 brix_tier_cfg_t cfg;
161 brix_cache_policy_t pol;
162
163 if (brix_tier_parse_store(cf, &common->cache_store,
164 common->cache_store_args, BRIX_TIER_CACHE, &cfg, err,
165 sizeof(err)) != NGX_OK)
166 {
167 return NGX_ERROR; /* [emerg] already logged */
168 }
169 ngx_memzero(&pol, sizeof(pol));
170 pol.enabled = 1;
171 pol.max_file_size = common->cache_max_object;
172 pol.evict_at = common->cache_evict_at;
173 pol.evict_to = common->cache_evict_to;
174 pol.meta_mode = (int) common->cache_meta_mode;
175 pol.batch_cinfo = (common->cache_batch_cinfo == 2)
176 ? -1 : (int) common->cache_batch_cinfo;
177 pol.l1_entries = common->cache_index_cache;
178 pol.slice_size = common->cache_slice_size;
179 /* phase-68: digest verification on fill (cvmfs-cas today). The verify
180 * runs on the staged temp BEFORE commit, which needs the store's
181 * staged_path — a local posix store; reject other stores loudly. */
182 pol.verify = (common->cache_verify_mode == NGX_CONF_UNSET_UINT)
183 ? BRIX_CACHE_VERIFY_OFF
184 : (brix_cache_verify_mode_e) common->cache_verify_mode;
185 if (pol.verify == BRIX_CACHE_VERIFY_CVMFS_CAS
186 && ngx_strcmp(cfg.driver, "posix") != 0)
187 {
188 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
189 "brix_cache_verify cvmfs-cas requires a local posix "
190 "cache store (got \"%s\")", cfg.driver);
191 return NGX_ERROR;
192 }
193 pol.cvmfs_manifest_ttl = common->cache_manifest_ttl;
194 if (common->cache_quarantine_dir.len > 0) {
195 ngx_cpystrn((u_char *) pol.quarantine_dir,
196 common->cache_quarantine_dir.data,
197 ngx_min(common->cache_quarantine_dir.len + 1,
198 sizeof(pol.quarantine_dir)));
199 }
200 brix_vfs_backend_config_cache_store(common->root_canon, &cfg, &pol);
201 }
202
203 if (common->stage_enable == 1) {
204 brix_tier_cfg_t cfg;
205 brix_stage_policy_t spol;
206
207 if (common->stage_store.len == 0) {
208 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
209 "brix_stage on requires brix_stage_store");
210 return NGX_ERROR;
211 }
212 if (brix_tier_parse_store(cf, &common->stage_store,
213 common->stage_store_args, BRIX_TIER_STAGE, &cfg, err,
214 sizeof(err)) != NGX_OK)
215 {
216 return NGX_ERROR;
217 }
218 ngx_memzero(&spol, sizeof(spol));
219 spol.enabled = 1;
220 spol.flush_mode = common->stage_flush_async ? BRIX_WT_MODE_ASYNC
221 : BRIX_WT_MODE_SYNC;
222 brix_vfs_backend_config_stage_store(common->root_canon, &cfg, &spol);
223 }
224
225 return NGX_OK;
226}

228/* Prepare one server block at postconfiguration: validate the configured root
229 * is an existing, accessible directory (access mode matching the write policy)
230 * and check the cache configuration, before the block accepts connections.
231 * Returns NGX_OK, or NGX_ERROR (emerg-logged) on any invalid resource. */
232ngx_int_t
233brix_config_prepare_server(ngx_conf_t *cf,
234 ngx_stream_brix_srv_conf_t *xcf)
235{
236 /* Hard read-only switch: force allow_write off before any allow_write-dependent
237 * setup, so every write gate rejects (root:// require_write, write-open, ...). */
238 brix_shared_apply_read_only(&xcf->common, cf->log);
239
240 /* Phase-4b: a GSI tap proxy must capture the client's delegated proxy to
241 * present it upstream — auto-enable delegation receipt if the admin didn't. */
242 if (xcf->proxy.enable && xcf->proxy.auth == BRIX_PROXY_AUTH_GSI
243 && !xcf->tpc_delegate)
244 {
245 xcf->tpc_delegate = 1;
246 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
247 "brix_tap_proxy_auth gsi: enabling GSI proxy delegation capture");
248 }
249
250 if (!xcf->manager_mode && !xcf->caps.supervisor
251 && xcf->manager_map == NULL && !xcf->proxy.enable) {
252 brix_export_root_opts_t root_opts;
253
254 /* A "posix:<path>" storage backend NAMES THE LOCAL EXPORT TREE — the fully
255 * composable replacement for brix_export. Anchor the export root at <path>
256 * and clear the backend so the default POSIX driver serves it directly. */
257 brix_storage_backend_posix_root(&xcf->common);
258
259 root_opts.directive_name = "brix_export";
260 root_opts.allow_write = xcf->common.allow_write
261 && !brix_storage_backend_is_remote(&xcf->common);
262 /* A pure cache node needs no local export tree: brix_cache_store is the
263 * physical FSAL and brix_cache_root the advertised root. brix_root then
264 * defaults to "/" (server_conf.c) — the cache serves the "/" namespace,
265 * filling from the backend into the store. */
266 root_opts.required = 1;
267 root_opts.canon_size = sizeof(xcf->common.root_canon);
268 if (brix_prepare_export_root(cf, &xcf->common.root, &root_opts,
269 xcf->common.root_canon) != NGX_CONF_OK)
270 {
271 return NGX_ERROR;
272 }
273 /* SP4: reap interrupted NON-staged direct-write temps under a real export
274 * tree (brix_tmp_reap_register itself refuses "/" — a pure cache node,
275 * whose root_canon is "/", keeps its in-flight temps in the store tier). */
276 brix_tmp_reap_register(xcf->common.root_canon);
277
278 /* Register the export's selected storage backend for VFS resolution at
279 * request time. A "root://host:port" value makes the export's PRIMARY
280 * storage a REMOTE XRootD server (read + write-through via sd_xroot); a
281 * driver name (e.g. "pblock") selects a local backend; default POSIX is a
282 * no-op. */
283 if (brix_vfs_backend_config_str(cf, xcf->common.root_canon,
284 &xcf->common.storage_backend, xcf->common.pblock_block_size,
285 (int) xcf->cache_origin_family)
286 != NGX_OK)
287 {
288 return NGX_ERROR;
289 }
290
291 /* §14: attach the named brix_credential's bearer token to the source
292 * backend (today consumed by sd_http as Authorization: Bearer). Resolved at
293 * config time so a missing credential fails loudly; an empty token_file is
294 * an error too. No brix_storage_credential ⇒ anonymous. */
295 if (xcf->common.storage_credential.len > 0) {
296 char cred_z[256];
297 char bearer[4096];
298 const brix_credential_t *cred;
299
300 ngx_cpystrn((u_char *) cred_z, xcf->common.storage_credential.data,
301 ngx_min(xcf->common.storage_credential.len + 1,
302 sizeof(cred_z)));
303 cred = brix_credential_lookup(cred_z);
304 if (cred == NULL) {
305 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
306 "brix_storage_credential: no brix_credential \"%V\"",
307 &xcf->common.storage_credential);
308 return NGX_ERROR;
309 }
310 if (brix_credential_bearer(cred, bearer, sizeof(bearer), cf->log)
311 != NGX_OK)
312 {
313 return NGX_ERROR;
314 }
315 {
316 brix_vfs_backend_cred_t bcred;
317
318 ngx_memzero(&bcred, sizeof(bcred));
319 bcred.bearer = (bearer[0] != '\0') ? bearer : NULL;
320 bcred.x509_proxy = (cred->x509_proxy.len > 0)
321 ? (const char *) cred->x509_proxy.data : NULL;
322 bcred.ca_dir = (cred->ca_dir.len > 0)
323 ? (const char *) cred->ca_dir.data : NULL;
324 bcred.s3_access_key = (cred->s3_access_key.len > 0)
325 ? (const char *) cred->s3_access_key.data : NULL;
326 bcred.s3_secret_key = (cred->s3_secret_key.len > 0)
327 ? (const char *) cred->s3_secret_key.data : NULL;
328 bcred.s3_region = (cred->s3_region.len > 0)
329 ? (const char *) cred->s3_region.data : NULL;
330 bcred.sss_keytab = (cred->sss_keytab.len > 0)
331 ? (const char *) cred->sss_keytab.data : NULL;
332 brix_vfs_backend_set_credential(xcf->common.root_canon, &bcred);
333 }
334 }
335
336 /* §14 + C-3/C-5: the write-back flush authenticates to its wt_origin with a
337 * credential too (ztn bearer). Resolve into cache_origin_bearer, which the
338 * C-5 driver write-back already threads into its sd_xroot dest. */
339 if (xcf->wt.credential.len > 0) {
340 char cred_z[256];
341 char bearer[4096];
342 const brix_credential_t *cred;
343
344 ngx_cpystrn((u_char *) cred_z, xcf->wt.credential.data,
345 ngx_min(xcf->wt.credential.len + 1, sizeof(cred_z)));
346 cred = brix_credential_lookup(cred_z);
347 if (cred == NULL) {
348 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
349 "brix_wt_credential: no brix_credential \"%V\"",
350 &xcf->wt.credential);
351 return NGX_ERROR;
352 }
353 if (brix_credential_bearer(cred, bearer, sizeof(bearer), cf->log)
354 != NGX_OK)
355 {
356 return NGX_ERROR;
357 }
358 if (bearer[0] != '\0') {
359 size_t bl = ngx_strlen(bearer);
360 u_char *bp = ngx_pnalloc(cf->pool, bl + 1);
361
362 if (bp == NULL) {
363 return NGX_ERROR;
364 }
365 ngx_memcpy(bp, bearer, bl + 1);
366 xcf->cache_origin_bearer.data = bp;
367 xcf->cache_origin_bearer.len = bl;
368 }
369 /* GSI write-back: the proxy + verify-CA the C-5 dest presents/uses. */
370 if (cred->x509_proxy.len > 0) {
371 xcf->cache_origin_x509_proxy = cred->x509_proxy;
372 }
373 if (cred->ca_dir.len > 0) {
374 xcf->cache_origin_ca_dir = cred->ca_dir;
375 }
376 }
377
378 /* Optional fast-cache upload staging device. Canonicalize once here; an
379 * unset/empty value leaves upload_stage_dir_canon empty (stage adjacent
380 * to the destination). A configured-but-bad path fails config loudly. */
381 if (xcf->upload_stage_dir.len > 0) {
382 brix_export_root_opts_t stage_opts;
383 stage_opts.directive_name = "brix_stage_dir";
384 stage_opts.allow_write = 1;
385 stage_opts.required = 0;
386 stage_opts.canon_size = sizeof(xcf->upload_stage_dir_canon);
387 if (brix_prepare_export_root(cf, &xcf->upload_stage_dir,
388 &stage_opts, xcf->upload_stage_dir_canon) != NGX_CONF_OK)
389 {
390 return NGX_ERROR;
391 }
392 /* Track for the stage-out reaper (finishes interrupted cache->storage
393 * commits across restarts). */
394 brix_stage_dir_register(xcf->upload_stage_dir_canon);
395 }
396
397 /* Phase-64: register the composable cache/stage tiers (sd_cache / sd_stage
398 * decorators composed over the backend, per worker). */
399 if (brix_tier_register_stores(cf, &xcf->common) != NGX_OK) {
400 return NGX_ERROR;
401 }

403 /* HARD config guard: the effective cache/state sidecar tree (folds
404 * cache_state_root / cache_store cstore / cache_root) and the upload
405 * stage dir must live OUTSIDE the export — otherwise .cinfo/.meta
406 * sidecars and upload temps would sit in the client-visible namespace.
407 * A nesting is a deploy-blocking error, not a warning. */
408 if (brix_assert_dir_outside_export(cf, "cache state/sidecar tree",
409 xcf->common.root_canon, brix_cache_state_root(xcf)) != NGX_OK
410 || brix_assert_dir_outside_export(cf, "brix_stage_dir",
411 xcf->common.root_canon, xcf->upload_stage_dir_canon) != NGX_OK)
412 {
413 return NGX_ERROR;
414 }
415 }
416
417 if (xcf->cache) {
418 /* The read-through cache is read-only UNLESS write-through is enabled,
419 * in which case write handles are accepted and mirrored to the origin
420 * at kXR_sync/kXR_close (see src/cache/writethrough_flush.c). */
421 if (xcf->common.allow_write && !xcf->wt.enable) {
422 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
423 "brix_cache is read-only and requires "
424 "brix_allow_write off (or enable brix_write_through)");
425 return NGX_ERROR;
426 }

428 /* §14: brix_cache_storage_backend is retired — a driver-backed cache
429 * is the tier grammar's brix_cache_store. */
430
431 /* Write-back staging cache backend (its own root). Same sidecar rule. */
432 if (xcf->cache_wt_stage_backend.len > 0) {
433 if (xcf->cache_wt_stage_root.len == 0) {
434 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
435 "brix_cache_wt_stage_backend requires "
436 "brix_cache_wt_stage_root");
437 return NGX_ERROR;
438 }
439 if (xcf->cache_state_root.len == 0) {
440 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
441 "brix_cache_wt_stage_backend requires a POSIX "
442 "brix_cache_state_root for its sidecars");
443 return NGX_ERROR;
444 }
445 brix_vfs_backend_config((const char *) xcf->cache_wt_stage_root.data,
446 &xcf->cache_wt_stage_backend,
447 xcf->cache_wt_stage_block_size);
448 }
449
450 {
451 /* C-1 (phase-63): the cache source may be a remote brix_storage_backend
452 * (root://...) instead of a separate brix_cache_origin — the cache then
453 * fills from the registered backend (open_or_fill resolves it). */
454 ngx_str_t *sb = &xcf->common.storage_backend;
455 int remote_source =
456 (sb->len > sizeof("root://") - 1
457 && ngx_strncmp(sb->data, "root://", sizeof("root://") - 1) == 0)
458 || (sb->len > sizeof("roots://") - 1
459 && ngx_strncmp(sb->data, "roots://",
460 sizeof("roots://") - 1) == 0)
461 || (sb->len > sizeof("http://") - 1
462 && ngx_strncmp(sb->data, "http://",
463 sizeof("http://") - 1) == 0)
464 || (sb->len > sizeof("https://") - 1
465 && ngx_strncmp(sb->data, "https://",
466 sizeof("https://") - 1) == 0);
467
468 /* §14: the legacy brix_cache_origin is retired — an `brix_cache
469 * on` cache fills from the export's REMOTE storage backend. */
470 if (xcf->cache_root.len == 0 || !remote_source) {
471 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
472 "brix_cache on requires brix_cache_export and a remote "
473 "brix_storage_backend (root://host:port); the retired "
474 "brix_cache_origin model is the tier grammar now "
475 "(brix_storage_backend + brix_cache_store)");
476 return NGX_ERROR;
477 }
478 }
479
480 if (brix_validate_path(cf, "brix_cache_export",
481 &xcf->cache_root,
482 BRIX_PATH_DIRECTORY,
483 R_OK | W_OK | X_OK)
484 != NGX_OK)
485 {
486 return NGX_ERROR;
487 }
488
489 if (xcf->cache_lock_timeout <= 0) {
490 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
491 "brix_cache_lock_timeout must be greater than zero");
492 return NGX_ERROR;
493 }
494
495 if (xcf->cache_eviction_threshold == 0
496 || xcf->cache_eviction_threshold >= 1000000)
497 {
498 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
499 "brix_cache_eviction_threshold must be greater than 0 "
500 "and less than 1.0");
501 return NGX_ERROR;
502 }

505 * satisfy this; an explicit pair that inverts it is a config error. */
506 if (xcf->reaper.high_watermark == 0
507 || xcf->reaper.high_watermark >= 1000000
508 || xcf->reaper.low_watermark == 0
509 || xcf->reaper.low_watermark >= xcf->reaper.high_watermark)

515 }
516
517 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
518 "brix: cache enabled root=%V origin=%V tls=%s "

520 &xcf->cache_root, &xcf->cache_origin,
521 xcf->cache_origin_tls ? "on" : "off",
522 (int) xcf->cache_lock_timeout,
523 xcf->cache_eviction_threshold);
524 }

526 /* Write-back-staging backpressure validation. Independent of the read cache —
527 * write-through staging exists whenever brix_write_through is on, so this
528 * runs at the server level. When a HIGH watermark is set it needs a staging
529 * root to measure, and the pair must satisfy 0 < low < high < 1.0. */
530 if (xcf->cache_wt_stage_high_watermark > 0) {
531 if (xcf->cache_wt_stage_root.len == 0) {
532 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
533 "brix_wt_stage_high_watermark requires "
534 "brix_cache_wt_stage_root");
535 return NGX_ERROR;
536 }
537 if (xcf->cache_wt_stage_high_watermark >= 1000000
538 || xcf->cache_wt_stage_low_watermark == 0
539 || xcf->cache_wt_stage_low_watermark
540 >= xcf->cache_wt_stage_high_watermark)
541 {
542 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
543 "brix_wt_stage_low_watermark must be greater than 0 and "
544 "less than brix_wt_stage_high_watermark (which must be < 1.0)");
545 return NGX_ERROR;
546 }
547 }
548
549 /*
550 * Access log handling mirrors nginx conventions: empty means disabled by
551 * default, the literal string "off" suppresses logging explicitly, and
552 * any other value is treated as the path to append to.
553 */
554 if (xcf->access_log.len > 0
555 && ngx_strcmp(xcf->access_log.data, (u_char *) "off") != 0)
556 {
557 /*
558 * Register the log with nginx (cycle->open_files) rather than opening a
559 * raw fd here. The master opens it during ngx_init_cycle, reopens it on
560 * USR1 (dup2 onto the same fd number), and closes it cleanly when the
561 * cycle is torn down — so log rotation and `nginx -s reload` no longer
562 * leak fds or keep writing to a rotated inode. Each worker captures
563 * file->fd into access_log_fd in init_process (the fd is not yet open at
564 * this config-time point).
565 */
566 xcf->access_log_file = ngx_conf_open_file(cf->cycle, &xcf->access_log);
567 if (xcf->access_log_file == NULL) {
568 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
569 "brix: cannot register access log \"%V\"",
570 &xcf->access_log);
571 return NGX_ERROR;
572 }
573
574 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
575 "brix: access log \"%V\" registered",
576 &xcf->access_log);
577 }
578
579 if (xcf->proxy.enable
580 && xcf->proxy.audit_log.len > 0
581 && ngx_strcmp(xcf->proxy.audit_log.data, (u_char *) "off") != 0)
582 {
583 /* nginx-managed handle; see the access-log note above. */
584 xcf->proxy.audit_log_file = ngx_conf_open_file(cf->cycle,
585 &xcf->proxy.audit_log);
586 if (xcf->proxy.audit_log_file == NULL) {
587 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
588 "brix: cannot register proxy audit log \"%V\"",
589 &xcf->proxy.audit_log);
590 return NGX_ERROR;
591 }
592
593 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
594 "brix: proxy audit log \"%V\" registered",
595 &xcf->proxy.audit_log);
596 }
597
598#if (NGX_SSL)
599 if (xcf->proxy.enable && xcf->proxy.upstream_tls) {
600 xcf->proxy.tls_ctx = ngx_pcalloc(cf->pool, sizeof(ngx_ssl_t));
601 if (xcf->proxy.tls_ctx == NULL) {
602 return NGX_ERROR;
603 }
604 xcf->proxy.tls_ctx->log = cf->log;
605 if (ngx_ssl_create(xcf->proxy.tls_ctx,
606 NGX_SSL_TLSv1_2 | NGX_SSL_TLSv1_3,
607 NULL) != NGX_OK)
608 {
609 return NGX_ERROR;
610 }
611
612 /* Peer certificate verification (optional; enabled by directive). */
613 if (xcf->proxy.upstream_tls_ca.len > 0) {
614 if (ngx_ssl_trusted_certificate(cf, xcf->proxy.tls_ctx,
615 &xcf->proxy.upstream_tls_ca,
616 5 /* chain depth */)
617 != NGX_OK)
618 {
619 return NGX_ERROR;
620 }
621 /* ngx_ssl_trusted_certificate sets SSL_VERIFY_PEER internally */
622 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
623 "brix: proxy upstream TLS CA loaded from \"%V\"",
624 &xcf->proxy.upstream_tls_ca);
625 }
626 }
627
628 /* Upstream redirector TLS (for mid-stream kXR_gotoTLS upgrade). */
629 if (xcf->upstream_tls) {
630 xcf->upstream_tls_ctx = ngx_pcalloc(cf->pool, sizeof(ngx_ssl_t));
631 if (xcf->upstream_tls_ctx == NULL) {
632 return NGX_ERROR;
633 }
634 xcf->upstream_tls_ctx->log = cf->log;
635 if (ngx_ssl_create(xcf->upstream_tls_ctx,
636 NGX_SSL_TLSv1_2 | NGX_SSL_TLSv1_3,
637 NULL) != NGX_OK)
638 {
639 return NGX_ERROR;
640 }
641
642 if (xcf->upstream_tls_ca.len > 0) {
643 if (ngx_ssl_trusted_certificate(cf, xcf->upstream_tls_ctx,
644 &xcf->upstream_tls_ca,
645 5 /* chain depth */)
646 != NGX_OK)
647 {
648 return NGX_ERROR;
649 }
650 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
651 "brix: upstream redirector TLS CA loaded from \"%V\"",
652 &xcf->upstream_tls_ca);
653 }
654
655 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
656 "brix: upstream redirector TLS enabled (kXR_gotoTLS support)");
657 }
658#endif
659
660 return NGX_OK;
661}
