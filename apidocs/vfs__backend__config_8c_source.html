13void
14brix_vfs_backend_config(const char *root_canon, const ngx_str_t *name,
15 size_t block_size)
16{
17 brix_vfs_backend_entry_t *e;
18
19 if (root_canon == NULL || root_canon[0] == '\0' || name == NULL
20 || name->len == 0)
21 {
22 return;
23 }
24 /* Bare local driver names register here: "pblock", and — phase-68 —
25 * an EXPLICIT "posix". Naming posix (the default) was a silent no-op
26 * before; registering it makes the export visible to census surfaces
27 * (the dashboard VFS browser, /metrics backend info) while leaving
28 * every config that doesn't name a backend exactly as it was. */
29 if (!((name->len == sizeof("pblock") - 1
30 && ngx_strncmp(name->data, "pblock", sizeof("pblock") - 1) == 0)
31 || (name->len == sizeof("posix") - 1
32 && ngx_strncmp(name->data, "posix", sizeof("posix") - 1) == 0)))
33 {
34 return;
35 }
36
37 /* Dedup on root_canon so a config reload updates rather than appends
38 * (an existing entry keeps its backend and only refreshes block_size). */
39 e = brix_vfs_backend_entry_find(root_canon);
40 if (e != NULL) {
41 e->block_size = (int64_t) block_size;
42 e->inst = NULL; /* rebuilt on next resolve */
43 return;
44 }
45 e = brix_vfs_backend_entry_get_or_create(root_canon);
46 if (e == NULL) {
47 return;
48 }
49 if (name->data[0] == 'p' && name->data[1] == 'o') {
50 ngx_memcpy(e->backend, "posix", sizeof("posix"));
51 } else {
52 ngx_memcpy(e->backend, "pblock", sizeof("pblock"));
53 }
54 e->block_size = (int64_t) block_size;
55}

282/* Parse ONE "http(s)://host[:port][/base]" segment into host/port/tls/base.
283 * Returns NGX_OK, or NGX_ERROR after an [emerg] for a malformed segment. */
284static ngx_int_t
285vfs_parse_http_origin(ngx_conf_t *cf, u_char *seg, size_t segn,
286 char *host, size_t host_cap, int *port_out, int *tls_out,
287 char *base, size_t base_cap)
288{
289 u_char *h = NULL;
290 size_t hn = 0;
291 int htls = 0;
292 size_t i, slash, hplen, pathlen, hostn;
293 u_char *colon = NULL;
294 const char *path;
295 int port;
296
297 if (segn > sizeof("https://") - 1
298 && ngx_strncmp(seg, "https://", sizeof("https://") - 1) == 0)
299 {
300 h = seg + sizeof("https://") - 1;
301 hn = segn - (sizeof("https://") - 1);
302 htls = 1;
303 } else if (segn > sizeof("http://") - 1
304 && ngx_strncmp(seg, "http://", sizeof("http://") - 1) == 0)
305 {
306 h = seg + sizeof("http://") - 1;
307 hn = segn - (sizeof("http://") - 1);
308 } else {
309 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
310 "brix_storage_backend: every |-separated origin must be an "
311 "http:// or https:// URL");
312 return NGX_ERROR;
313 }
314
315 slash = hn;
316 for (i = 0; i < hn; i++) {
317 if (h[i] == '/') { slash = i; break; }
318 }
319 hplen = slash;
320 path = (slash < hn) ? (const char *) (h + slash) : "";
321 pathlen = (slash < hn) ? hn - slash : 0;
322
323 for (i = hplen; i > 0; i--) {
324 if (h[i - 1] == ':') { colon = h + i - 1; break; }
325 }
326 if (colon != NULL) {
327 ngx_int_t pn = ngx_atoi(colon + 1, (size_t) (h + hplen - (colon + 1)));
328
329 hostn = (size_t) (colon - h);
330 if (pn == NGX_ERROR || pn <= 0 || pn > 65535) {
331 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
332 "brix_storage_backend: invalid http origin port");
333 return NGX_ERROR;
334 }
335 port = (int) pn;
336 } else {
337 hostn = hplen;
338 port = htls ? 443 : 80;
339 }
340 if (hostn == 0 || hostn >= host_cap || pathlen >= base_cap) {
341 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
342 "brix_storage_backend: invalid http origin host/path");
343 return NGX_ERROR;
344 }
345 ngx_memcpy(host, h, hostn);
346 host[hostn] = '\0';
347 if (pathlen > 0) {
348 ngx_memcpy(base, path, pathlen);
349 base[pathlen] = '\0';
350 } else {
351 base[0] = '\0';
352 }
353 *port_out = port;
354 *tls_out = htls;
355 return NGX_OK;
356}

358/* Append one failover endpoint to the export's http backend entry (phase-68
359 * T11). Must run AFTER brix_vfs_backend_config_http registered endpoint 0
360 * for the same root. */
361static ngx_int_t
362vfs_backend_add_http_endpoint(ngx_conf_t *cf, const char *root_canon,
363 const char *host, int port, int tls, const char *base)
364{
365 brix_vfs_backend_entry_t *e = brix_vfs_backend_entry_find(root_canon);
366 size_t n;
367
368 if (e == NULL || ngx_strcmp(e->backend, "http") != 0) {
369 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
370 "brix_storage_backend: internal - endpoint list before primary");
371 return NGX_ERROR;
372 }
373 n = sizeof(e->http_extra) / sizeof(e->http_extra[0]);
374 if ((size_t) e->n_http_extra >= n) {
375 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
376 "brix_storage_backend: too many |-separated origins (max %uz)",
377 n + 1);
378 return NGX_ERROR;
379 }
380 if (ngx_strlen(host) >= sizeof(e->http_extra[0].host)
381 || ngx_strlen(base) >= sizeof(e->http_extra[0].base))
382 {
383 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
384 "brix_storage_backend: origin host/base too long for the "
385 "endpoint table");
386 return NGX_ERROR;
387 }
388 ngx_cpystrn((u_char *) e->http_extra[e->n_http_extra].host,
389 (u_char *) host, sizeof(e->http_extra[0].host));
390 e->http_extra[e->n_http_extra].port = port;
391 e->http_extra[e->n_http_extra].tls = tls;
392 ngx_cpystrn((u_char *) e->http_extra[e->n_http_extra].base,
393 (u_char *) base, sizeof(e->http_extra[0].base));
394 e->n_http_extra++;
395 e->inst = NULL; /* rebuilt on next resolve */
396 return NGX_OK;
397}

400ngx_int_t
401brix_vfs_backend_config_str(ngx_conf_t *cf, const char *root_canon,
402 const ngx_str_t *sb, size_t block_size, int family)
403{
404 u_char *addr = NULL;
405 size_t addrn = 0;
406 int is_roots = 0;
407
408 if (sb == NULL) {
409 return NGX_OK;
410 }
411
412 /* An export that names NO backend is the default-POSIX case. Phase-68 made
413 * an EXPLICIT "posix" register; register the default too so the census
414 * surfaces (dashboard /vfs + storage panel, /metrics info + capacity
415 * gauges) see the most common configuration. Guard root_canon "/": a pure
416 * cache node's namespace anchor is the whole host fs — never a census row. */
417 if (sb->len == 0) {
418 if (root_canon != NULL && root_canon[0] == '/'
419 && root_canon[1] != '\0')

428 /* "cephfsro:<meta_pool>+<data_pool>[@<conf>][?assume_quiesced=1]" → read-only
429 * CephFS-via-RADOS backend. Both pools are required. The fs MUST be quiesced;
430 * the operator asserts that with the "?assume_quiesced=1" suffix (the driver
431 * refuses to bind otherwise — fail-closed). Checked before "ceph:" since that
432 * is a prefix-distinct but related scheme. */
433 if (sb->len > sizeof("cephfsro:") - 1
434 && ngx_strncmp(sb->data, "cephfsro:", sizeof("cephfsro:") - 1) == 0)
435 {
436 const u_char *rest = sb->data + (sizeof("cephfsro:") - 1);
437 size_t restn = sb->len - (sizeof("cephfsro:") - 1);
438 char meta[256] = "", data[256] = "", conf[1024] = "", q[64] = "";
439 size_t i, mn = 0, dn = 0, cn = 0, qn = 0;
440 int phase = 0; /* 0=meta, 1=data (+), 2=conf (@), 3=query (?) */
441 int quiesced, live;
442
443 for (i = 0; i < restn; i++) {
444 u_char c = rest[i];
445 if (c == '+' && phase == 0) { phase = 1; continue; }
446 if (c == '@' && phase < 2) { phase = 2; continue; }
447 if (c == '?') { phase = 3; continue; }
448 if (phase == 0 && mn + 1 < sizeof(meta)) { meta[mn++] = (char) c; }
449 else if (phase == 1 && dn + 1 < sizeof(data)) { data[dn++] = (char) c; }
450 else if (phase == 2 && cn + 1 < sizeof(conf)) { conf[cn++] = (char) c; }
451 else if (phase == 3 && qn + 1 < sizeof(q)) { q[qn++] = (char) c; }
452 }
453 meta[mn] = '\0'; data[dn] = '\0'; conf[cn] = '\0'; q[qn] = '\0';
454
455 if (meta[0] == '\0' || data[0] == '\0') {
456 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0, "brix_storage_backend: "
457 "cephfsro needs both pools (cephfsro:<meta>+<data>)");
458 return NGX_ERROR;
459 }
460 quiesced = (ngx_strstr(q, "assume_quiesced=1") != NULL);
461 live = (ngx_strstr(q, "live=1") != NULL);
462 if (!quiesced && !live) {
463 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0, "brix_storage_backend: "
464 "cephfsro requires a consistency assertion: "
465 "?assume_quiesced=1 (fs frozen — MDS down/failed, journal flushed) "
466 "or ?live=1 (still mounted — best-effort eventually-consistent "
467 "reads with optimistic revalidation + retry)");
468 return NGX_ERROR;
469 }
470 brix_vfs_backend_config_cephfs_ro(root_canon, meta, data, conf,
471 quiesced, live);
472 return NGX_OK;
473 }

475 /* "ceph:<pool>[@<conf>][?<key_prefix>]" → flat, block-only RADOS object
476 * backend (the pure-librados reference). The pool is required; conf defaults
477 * to /etc/ceph/ceph.conf; everything after '@' (to an optional '?') is the
478 * ceph.conf path; after '?' is the object-key prefix. */
479 {
480 const char *backend = NULL;
481 size_t skip = 0;
482
483 if (sb->len > sizeof("ceph:") - 1
484 && ngx_strncmp(sb->data, "ceph:", sizeof("ceph:") - 1) == 0)
485 {
486 backend = "ceph"; skip = sizeof("ceph:") - 1;
487 }
488
489 if (backend != NULL) {
490 const u_char *rest = sb->data + skip;
491 size_t restn = sb->len - skip;
492 char pool[256] = "", conf[1024] = "", prefix[256] = "";
493 size_t i, pn = 0, cn = 0, xn = 0;
494 int phase = 0; /* 0=pool, 1=conf (@), 2=prefix (?) */
495
496 for (i = 0; i < restn; i++) {
497 u_char c = rest[i];
498 if (c == '@' && phase == 0) { phase = 1; continue; }
499 if (c == '?') { phase = 2; continue; }
500 if (phase == 0 && pn + 1 < sizeof(pool)) { pool[pn++] = (char) c; }
501 else if (phase == 1 && cn + 1 < sizeof(conf)) { conf[cn++] = (char) c; }
502 else if (phase == 2 && xn + 1 < sizeof(prefix)) { prefix[xn++] = (char) c; }
503 }
504 pool[pn] = '\0'; conf[cn] = '\0'; prefix[xn] = '\0';
505 if (pool[0] == '\0') {
506 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
507 "brix_storage_backend: %s: needs a pool (%s:<pool>)",
508 backend, backend);
509 return NGX_ERROR;
510 }
511 brix_vfs_backend_config_ceph(root_canon, backend, pool, conf, prefix);
512 return NGX_OK;
513 }

516 /* "rados://<pool>[/<namespace>]" → the flat librados object backend (the URL
517 * alias of "ceph:<pool>"); the tail after the pool is the object-key prefix.
518 * Cluster addressing comes from /etc/ceph/ceph.conf, so there is no host:port
519 * authority — the authority slot is the pool. */
520 if (sb->len >= sizeof("rados://") - 1
521 && ngx_strncmp(sb->data, "rados://", sizeof("rados://") - 1) == 0)
522 {
523 const u_char *rest = sb->data + sizeof("rados://") - 1;
524 size_t restn = sb->len - (sizeof("rados://") - 1);
525 char pool[256] = "";
526 char prefix[256] = "";
527 size_t i, slash = restn;
528
529 for (i = 0; i < restn; i++) {
530 if (rest[i] == '/') { slash = i; break; }
531 }
532 if (slash > 0 && slash < sizeof(pool)) {
533 ngx_memcpy(pool, rest, slash);
534 pool[slash] = '\0';
535 }
536 if (slash < restn && (restn - slash - 1) < sizeof(prefix)) {
537 ngx_memcpy(prefix, rest + slash + 1, restn - slash - 1); /* drop '/' */
538 prefix[restn - slash - 1] = '\0';
539 }
540 if (pool[0] == '\0') {
541 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
542 "brix_storage_backend: rados:// needs \"//<pool>[/<namespace>]\"");
543 return NGX_ERROR;
544 }
545 brix_vfs_backend_config_ceph(root_canon, "ceph", pool, NULL, prefix);
546 return NGX_OK;
547 }

549 /* "tape://<adapter>/<base>" (or its alias "frm://...") → a nearline (tape/MSS)
550 * backend served by sd_frm (phase-64 SP5). The authority host names the MSS
551 * adapter ("" / "stub" = the built-in stub); the path is the adapter's MSS
552 * base. */
553 {
554 const u_char *rest = NULL;
555 size_t restn = 0;
556
557 if (sb->len > sizeof("tape://") - 1
558 && ngx_strncmp(sb->data, "tape://", sizeof("tape://") - 1) == 0)
559 {
560 rest = sb->data + sizeof("tape://") - 1;
561 restn = sb->len - (sizeof("tape://") - 1);
562 } else if (sb->len > sizeof("frm://") - 1
563 && ngx_strncmp(sb->data, "frm://", sizeof("frm://") - 1) == 0)
564 {
565 rest = sb->data + sizeof("frm://") - 1;
566 restn = sb->len - (sizeof("frm://") - 1);
567 }
568
569 if (rest != NULL) {
570 char adapter[64] = "";
571 char base[1024] = "";
572 size_t i, slash = restn;
573
574 for (i = 0; i < restn; i++) {
575 if (rest[i] == '/') { slash = i; break; }
576 }
577 if (slash > 0 && slash < sizeof(adapter)) {
578 ngx_memcpy(adapter, rest, slash);
579 adapter[slash] = '\0';
580 }
581 if (slash < restn && (restn - slash) < sizeof(base)) {
582 ngx_memcpy(base, rest + slash, restn - slash); /* keeps leading '/' */
583 base[restn - slash] = '\0';
584 }
585 if (base[0] != '/') {
586 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
587 "brix_storage_backend: tape://|frm:// needs "
588 "\"//<adapter>/<base-path>\"");
589 return NGX_ERROR;
590 }
591 brix_vfs_backend_config_tape(root_canon, adapter, base);
592 return NGX_OK;
593 }
594 }

596 /* "http://host[:port]/base" / "https://..." → a read-only HTTP SOURCE
597 * backend. phase-68 T11: a PIPE-SEPARATED ordered list of such URLs
598 * registers a multi-endpoint origin set with health-scored read failover
599 * (mirrors the CVMFS_SERVER_URL syntax); the first URL is the primary
600 * and the write target. */
601 if (vfs_backend_is_http_url(sb->data, sb->len)) {
602 u_char *seg = sb->data;
603 u_char *end = sb->data + sb->len;
604 int first = 1;
605
606 while (seg < end) {
607 u_char *pipe = ngx_strlchr(seg, end, '|');
608 size_t segn = (pipe != NULL) ? (size_t) (pipe - seg)
609 : (size_t) (end - seg);
610 char host[256], base[1024];
611 int port, htls;
612
613 if (vfs_parse_http_origin(cf, seg, segn, host, sizeof(host),
614 &port, &htls, base, sizeof(base))
615 != NGX_OK)
616 {
617 return NGX_ERROR; /* [emerg] already logged */
618 }
619 if (first) {
620 brix_vfs_backend_config_http(root_canon, host, port, htls,
621 base);
622 first = 0;
623 } else if (vfs_backend_add_http_endpoint(cf, root_canon, host,
624 port, htls, base)
625 != NGX_OK)
626 {
627 return NGX_ERROR;
628 }
629 seg = (pipe != NULL) ? pipe + 1 : end;
630 }
631 return NGX_OK;
632 }
633
634 /* "s3://host[:port]/bucket" → a read-only S3 source backend (path-style): the
635 * first path segment is the bucket. Default port 7480 (radosgw), tls off — the
636 * same defaults the generic tier parser uses for an s3 store. */
637 if (sb->len >= sizeof("s3://") - 1
638 && ngx_strncmp(sb->data, "s3://", sizeof("s3://") - 1) == 0)
639 {
640 u_char *h = sb->data + sizeof("s3://") - 1;
641 size_t hn = sb->len - (sizeof("s3://") - 1);
642 size_t i, slash = hn, hplen, pathlen;
643 u_char *colon = NULL;
644 const char *path;
645 char host[256], bucket[256];
646 size_t hostn;
647 int port;
648
649 for (i = 0; i < hn; i++) {
650 if (h[i] == '/') { slash = i; break; }
651 }
652 hplen = slash;
653 path = (slash < hn) ? (const char *) (h + slash) : "";
654 pathlen = (slash < hn) ? hn - slash : 0;
655
656 for (i = hplen; i > 0; i--) {
657 if (h[i - 1] == ':') { colon = h + i - 1; break; }
658 }
659 if (colon != NULL) {
660 ngx_int_t pn = ngx_atoi(colon + 1, (size_t) (h + hplen - (colon + 1)));
661 hostn = (size_t) (colon - h);
662 if (pn == NGX_ERROR || pn <= 0 || pn > 65535) {
663 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
664 "brix_storage_backend: invalid s3 origin port");
665 return NGX_ERROR;
666 }
667 port = (int) pn;
668 } else {
669 hostn = hplen;
670 port = 7480; /* radosgw S3 default */
671 }
672 while (pathlen > 0 && path[0] == '/') { path++; pathlen--; } /* → bucket */
673 if (hostn == 0 || hostn >= sizeof(host) || pathlen == 0
674 || pathlen >= sizeof(bucket))
675 {
676 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
677 "brix_storage_backend: s3:// needs \"//host[:port]/bucket\"");
678 return NGX_ERROR;
679 }
680 ngx_memcpy(host, h, hostn);
681 host[hostn] = '\0';
682 ngx_memcpy(bucket, path, pathlen);
683 bucket[pathlen] = '\0';
684 brix_vfs_backend_config_s3(root_canon, host, port, 0, bucket);
685 return NGX_OK;
686 }
687
688 /* "root://host:port" / "roots://host:port" → a remote root:// primary backend;
689 * any other value is a local driver name (pblock/posix) handled as before. */
690 if (sb->len > sizeof("roots://") - 1
691 && ngx_strncmp(sb->data, "roots://", sizeof("roots://") - 1) == 0)
692 {
693 addr = sb->data + sizeof("roots://") - 1;
694 addrn = sb->len - (sizeof("roots://") - 1);
695 is_roots = 1;
696 } else if (sb->len > sizeof("root://") - 1
697 && ngx_strncmp(sb->data, "root://", sizeof("root://") - 1) == 0)
698 {
699 addr = sb->data + sizeof("root://") - 1;
700 addrn = sb->len - (sizeof("root://") - 1);
701 }
702
703 if (addr == NULL) {
704 brix_vfs_backend_config(root_canon, sb, block_size);
705 return NGX_OK;
706 }
707
708 {
709 u_char *colon = NULL;
710 size_t i, hostn;
711 ngx_int_t portnum;
712 char host[256];
713
714 /* Split host:port on the LAST colon (a bracketed [v6]:port keeps it). */
715 for (i = addrn; i > 0; i--) {
716 if (addr[i - 1] == ':') { colon = addr + i - 1; break; }
717 }
718 if (colon == NULL) {
719 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
720 "brix_storage_backend: remote origin needs host:port");
721 return NGX_ERROR;
722 }
723 hostn = (size_t) (colon - addr);
724 portnum = ngx_atoi(colon + 1, (size_t) (addr + addrn - (colon + 1)));
725 if (hostn == 0 || hostn >= sizeof(host) || portnum == NGX_ERROR
726 || portnum <= 0 || portnum > 65535)
727 {
728 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
729 "brix_storage_backend: invalid remote origin host:port");
730 return NGX_ERROR;
731 }
732 ngx_memcpy(host, addr, hostn);
733 host[hostn] = '\0';
734 brix_vfs_backend_config_xroot(root_canon, host, (int) portnum,
735 is_roots, family);
736 }
737 return NGX_OK;
738}

814/* Runtime twin of the config-time http registration (phase-68 T14 proxy
815 * mode): register (or reuse) a synthetic per-upstream export entry keyed
816 * `up_root`, whose source is the http origin `host`:`port` and whose cache
817 * tier mirrors the entry at `template_root` with its store path suffixed
818 * `store_suffix` — so objects from different Stratum-1s can never alias in
819 * the store. Runs on the worker's event loop only (the entry table is
820 * per-process after fork; no cross-thread access). Returns NGX_OK, or
821 * NGX_ERROR when the table is full / the paths overflow. */
822ngx_int_t
823brix_vfs_backend_register_http_upstream(const char *up_root,
824 const char *template_root, const char *host, int port, int tls,
825 const char *store_suffix)
826{
827 brix_vfs_backend_entry_t *e, *tpl;
828
829 e = brix_vfs_backend_entry_find(up_root);
830 if (e != NULL) {
831 return NGX_OK; /* already registered */
832 }
833 tpl = brix_vfs_backend_entry_find(template_root);
834 e = brix_vfs_backend_entry_get_or_create(up_root);
835 if (e == NULL) {
836 return NGX_ERROR; /* table full */
837 }
838 ngx_memcpy(e->backend, "http", sizeof("http"));
839 ngx_cpystrn((u_char *) e->origin_host, (u_char *) host,
840 sizeof(e->origin_host));
841 e->origin_port = port;
842 e->origin_tls = tls;
843 e->origin_path[0] = '\0'; /* the request URI carries the full path */
844
845 if (tpl != NULL && tpl->cache_tier.configured) {
846 size_t n;
847
848 e->cache_tier = tpl->cache_tier;
849 e->cache_policy = tpl->cache_policy;
850 n = ngx_strlen(e->cache_tier.path);
851 if (store_suffix != NULL && store_suffix[0] != '\0') {
852 if (n + ngx_strlen(store_suffix) + 1 >= sizeof(e->cache_tier.path)) {
853 return NGX_ERROR;
854 }
855 ngx_cpystrn((u_char *) e->cache_tier.path + n,
856 (u_char *) store_suffix,
857 sizeof(e->cache_tier.path) - n);
858 /* the cstore mkdirs each KEY's parents but expects its own root
859 * to exist — create the per-upstream subtree now (local store) */
860 if (ngx_strcmp(e->cache_tier.driver, "posix") == 0) {
861 (void) brix_mkdir_recursive(e->cache_tier.path, 0755);
862 }
863 }
864 }
865 e->inst = NULL;
866 return NGX_OK;
867}
