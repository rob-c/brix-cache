19void
20brix_vfs_backend_config(const char *root_canon, const ngx_str_t *name,
21 size_t block_size)
22{
23 brix_vfs_backend_entry_t *e;
24
25 if (root_canon == NULL || root_canon[0] == '\0' || name == NULL
26 || name->len == 0)
27 {
28 return;
29 }
30 /* Bare local driver names register here: "pblock", and — phase-68 —
31 * an EXPLICIT "posix". Naming posix (the default) was a silent no-op
32 * before; registering it makes the export visible to census surfaces
33 * (the dashboard VFS browser, /metrics backend info) while leaving
34 * every config that doesn't name a backend exactly as it was. */
35 if (!((name->len == sizeof("pblock") - 1
36 && ngx_strncmp(name->data, "pblock", sizeof("pblock") - 1) == 0)
37 || (name->len == sizeof("posix") - 1
38 && ngx_strncmp(name->data, "posix", sizeof("posix") - 1) == 0)))
39 {
40 return;
41 }
42
43 /* Dedup on root_canon so a config reload updates rather than appends
44 * (an existing entry keeps its backend and only refreshes block_size). */
45 e = brix_vfs_backend_entry_find(root_canon);
46 if (e != NULL) {
47 e->block_size = (int64_t) block_size;
48 e->inst = NULL; /* rebuilt on next resolve */
49 return;
50 }
51 e = brix_vfs_backend_entry_get_or_create(root_canon);
52 if (e == NULL) {
53 return;
54 }
55 if (name->data[0] == 'p' && name->data[1] == 'o') {
56 ngx_memcpy(e->backend, "posix", sizeof("posix"));
57 } else {
58 ngx_memcpy(e->backend, "pblock", sizeof("pblock"));
59 }
60 e->block_size = (int64_t) block_size;
61}

107ngx_int_t
108brix_vfs_backend_config_str(ngx_conf_t *cf, const char *root_canon,
109 const ngx_str_t *sb, size_t block_size, int family)
110{
111 ngx_int_t rc;
112
113 if (sb == NULL) {
114 return NGX_OK;
115 }
116
117 /* An export that names NO backend is the default-POSIX case. Phase-68 made
118 * an EXPLICIT "posix" register; register the default too so the census
119 * surfaces (dashboard /vfs + storage panel, /metrics info + capacity
120 * gauges) see the most common configuration. Guard root_canon "/": a pure
121 * cache node's namespace anchor is the whole host fs — never a census row. */
122 if (sb->len == 0) {
123 if (root_canon != NULL && root_canon[0] == '/'
124 && root_canon[1] != '\0')
125 {
126 static const ngx_str_t posix_name = ngx_string("posix");
127
128 brix_vfs_backend_config(root_canon, &posix_name, block_size);
129 }
130 return NGX_OK;
131 }
132
133 rc = vfs_parse_cephfsro_origin(cf, root_canon, sb);
134 if (rc != NGX_DECLINED) { return rc; }
135 rc = vfs_parse_ceph_origin(cf, root_canon, sb);
136 if (rc != NGX_DECLINED) { return rc; }
137 rc = vfs_parse_rados_origin(cf, root_canon, sb);
138 if (rc != NGX_DECLINED) { return rc; }
139 rc = vfs_parse_tape_origin(cf, root_canon, sb);
140 if (rc != NGX_DECLINED) { return rc; }
141 rc = vfs_parse_http_origin_list(cf, root_canon, sb);
142 if (rc != NGX_DECLINED) { return rc; }
143 rc = vfs_parse_s3_origin(cf, root_canon, sb);
144 if (rc != NGX_DECLINED) { return rc; }
145 return vfs_parse_xroot_or_driver_origin(cf, root_canon, sb, block_size, family);
146}

178/* Endpoint (host,port) at `idx` of the http backend registered at
179 * `root_canon` — index 0 is the primary, 1.. the T11 failover list. Returns
180 * 0, or -1 past the end / not an http backend. Pointers alias the registry's
181 * stable storage. */
182int
183brix_vfs_backend_http_endpoint_at(const char *root_canon, int idx,
184 const char **host, int *port)
185{
186 brix_vfs_backend_entry_t *e = brix_vfs_backend_entry_find(root_canon);
187
188 if (e == NULL || ngx_strcmp(e->backend, "http") != 0 || idx < 0
189 || idx > e->n_http_extra)
190 {
191 return -1;
192 }
193 if (idx == 0) {
194 *host = e->origin_host;
195 *port = e->origin_port;
196 } else {
197 *host = e->http_extra[idx - 1].host;
198 *port = e->http_extra[idx - 1].port;
199 }
200 return 0;
201}

206brix_vfs_backend_set_http_ranks(const char *root_canon, const int *ranks,

208{

222/* Runtime twin of the config-time http registration (phase-68 T14 proxy
223 * mode): register (or reuse) a synthetic per-upstream export entry keyed
224 * `up_root`, whose source is the http origin `host`:`port` and whose cache
225 * tier mirrors the entry at `template_root` with its store path suffixed
226 * `store_suffix` — so objects from different Stratum-1s can never alias in
227 * the store. Runs on the worker's event loop only (the entry table is
228 * per-process after fork; no cross-thread access). Returns NGX_OK, or
229 * NGX_ERROR when the table is full / the paths overflow. */
230ngx_int_t
231brix_vfs_backend_register_http_upstream(const char *up_root,
232 const char *template_root, const char *host, int port, int tls,
233 const char *store_suffix)
234{
235 brix_vfs_backend_entry_t *e, *tpl;
236
237 e = brix_vfs_backend_entry_find(up_root);
238 if (e != NULL) {
239 return NGX_OK; /* already registered */
240 }
241 tpl = brix_vfs_backend_entry_find(template_root);
242 e = brix_vfs_backend_entry_get_or_create(up_root);
243 if (e == NULL) {
244 return NGX_ERROR; /* table full */
245 }
246 ngx_memcpy(e->backend, "http", sizeof("http"));
247 ngx_cpystrn((u_char *) e->origin_host, (u_char *) host,
248 sizeof(e->origin_host));
249 e->origin_port = port;
250 e->origin_tls = tls;
251 e->origin_path[0] = '\0'; /* the request URI carries the full path */
252
253 if (tpl != NULL && tpl->cache_tier.configured) {
254 size_t n;
255
256 e->cache_tier = tpl->cache_tier;
257 e->cache_policy = tpl->cache_policy;
258 n = ngx_strlen(e->cache_tier.path);
259 if (store_suffix != NULL && store_suffix[0] != '\0') {
260 if (n + ngx_strlen(store_suffix) + 1 >= sizeof(e->cache_tier.path)) {
261 return NGX_ERROR;
262 }
263 ngx_cpystrn((u_char *) e->cache_tier.path + n,
264 (u_char *) store_suffix,
265 sizeof(e->cache_tier.path) - n);
266 /* the cstore mkdirs each KEY's parents but expects its own root
267 * to exist — create the per-upstream subtree now (local store) */
268 if (ngx_strcmp(e->cache_tier.driver, "posix") == 0) {
269 (void) brix_mkdir_recursive(e->cache_tier.path, 0755);
270 }
271 }
272 }
273 e->inst = NULL;
274 return NGX_OK;
275}
