1/* ---- File: src/core/negcache/negcache_core.h — negative-path backoff core ----
2 *
3 * The nginx-free heart of the E-4 DoS-resilience "negative cache": a per-
4 * principal sliding-window counter of *missing-path* lookups (kXR_stat /
5 * kXR_locate that resolve to kXR_NotFound). When one principal's miss rate
6 * crosses a threshold inside the window the slot ARMS, and note() then paces the
7 * principal to at most one served miss per backoff interval — every excess miss
8 * gets a kXR_wait. Because the stock XRootD client answers kXR_wait by sleeping
9 * and re-sending the *same* request, this guarantees each lookup still completes
10 * (on its retry, one interval later) while a path-enumeration / stat-harvest
11 * loop is throttled to ~one path per backoff interval. Legitimate clients — who
12 * rarely miss and so never arm — are untouched.
13 *
14 * No ngx_* dependency, so tests/c/test_negcache.c links this file directly and
15 * drives the trip / decay / per-principal-isolation behaviour under plain C.
16 * The ngx wrapper (negcache.c) owns the SHM slot array + mutex and derives the
17 * principal hash; this file is pure arithmetic over a caller-owned slot array.
18 */
19#ifndef BRIX_NEGCACHE_CORE_H
20#define BRIX_NEGCACHE_CORE_H

22#include <stdint.h>

24/*
25 * One direct-mapped slot: a principal's recent miss count within a window.
26 * key == 0 marks an empty slot (a principal whose hash is 0 is remapped to 1 by
27 * the core), so the SHM table needs no separate in-use flag. A hash collision
28 * between two principals reuses the slot by victim-replacement — this is a
29 * throttle, never a correctness oracle, so a collision merely resets the loser's
30 * counter (weakening the throttle slightly, never mis-throttling a bystander).
31 */
32typedef struct {
33 uint32_t key; /* principal hash (0 = empty) */
34 uint32_t misses; /* misses accumulated in the current window */
35 uint64_t window_ms; /* start of the current window (ms, monotonic) */
36 uint64_t served_ms; /* last armed miss let through (0 = none this window)*/
37} brix_negcache_slot_t;
38
39/* Per-call tunables (carried from the server's brix_negcache_backoff rule). */
40typedef struct {
41 unsigned threshold; /* misses within a window that ARM the throttle(>=1)*/
42 unsigned window_ms; /* sliding-window length in ms (>= 1) */
43 unsigned backoff_s; /* kXR_wait seconds + min gap between served misses */
44} brix_negcache_params_t;
45
46/*
47 * Record one missing-path lookup for `key_hash` at `now_ms` and return the
48 * kXR_wait seconds to impose (0 = let the NotFound through). Once the slot is
49 * armed (>= threshold misses in the window), the first miss of each backoff
50 * interval is served (returns 0) and every earlier miss in that interval is
51 * throttled (returns backoff_s) — so a flood is paced without ever wedging a
52 * request. Fail-open: returns 0 on a NULL/empty slot array or a misconfigured
53 * params block. The caller must serialise concurrent calls on the same slot
54 * array (the SHM wrapper holds the zone mutex around this call).
55 */
56unsigned brix_negcache_core_note(brix_negcache_slot_t *slots, unsigned nslots,
57 uint32_t key_hash, uint64_t now_ms, const brix_negcache_params_t *p);
