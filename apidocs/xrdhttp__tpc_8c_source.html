1/*
2 * xrdhttp_tpc.c — XrdHttp TPC / checksum / digest support for the nginx-xrootd
3 * WebDAV module.
4 *
5 * WHAT: Split verbatim from xrdhttp.c:
6 * - TPC shim: synthesise Source:/Destination: headers from ?tpc.src= /
7 * ?tpc.dst=, and inject an Authorization: Bearer from X-Xrootd-Tpc-Token.
8 * - Checksum: Digest: response header computed on-demand from the open fd for
9 * xrd.want.cksum / Want-Digest requests.
10 * - Streaming Digest body filter: fold the response body through adler32 and
11 * emit a Digest trailer for XrdHttp clients.
12 *
13 * WHY: Groups the TPC-header + checksum/digest concern in one focused file so
14 * xrdhttp.c stays under the file-size cap. See xrdhttp.c for request parsing,
15 * xrdhttp_response.c for the status/redirect dialect, and xrdhttp.h for the API.
16 *
17 * HOW: Behaviour is identical to the original combined file — functions were
18 * moved verbatim.
19 */

42/* TPC header injection */
43ngx_int_t
44xrdhttp_inject_tpc_headers(ngx_http_request_t *r)
45{
46 xrdhttp_req_ctx_t *ctx;
47 ngx_table_elt_t *h;
48
49 ctx = (xrdhttp_req_ctx_t *)
50 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
51 if (ctx == NULL) {
52 return NGX_OK;
53 }
54
55 /*
56 * ?tpc.src= → inject Source: header (if not already present).
57 * This allows XrdHttp clients to signal TPC pull via URL instead of header.
58 */
59 if (ctx->tpc_src[0]
60 && webdav_tpc_find_header(r, "Source", sizeof("Source") - 1) == NULL)
61 {
62 brix_net_target_t target;
63 ngx_str_t url;
64 char err[256];
65
66 url.data = (u_char *) ctx->tpc_src;
67 url.len = ngx_strlen(ctx->tpc_src);
68
69 if (brix_net_target_parse(NULL, &url, &target, err, sizeof(err))
70 != NGX_OK)
71 {
72 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
73 "xrdhttp: invalid tpc.src URL: %s; ignored", err);
74 } else if (ngx_strncasecmp(url.data, (u_char *) "https://", 8) != 0) {
75 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
76 "xrdhttp: tpc.src must be https://; ignored");
77 } else {
78 if (brix_http_request_header_add(r, "Source", ctx->tpc_src,
79 NULL) != NGX_OK)
80 {
81 return NGX_ERROR;
82 }
83 }
84 }
85
86 /*
87 * ?tpc.dst= → inject Destination: header (if not already present).
88 */
89 if (ctx->tpc_dst[0]
90 && webdav_tpc_find_header(r, "Destination",
91 sizeof("Destination") - 1) == NULL)
92 {
93 brix_net_target_t target;
94 ngx_str_t url;
95 char err[256];
96
97 url.data = (u_char *) ctx->tpc_dst;
98 url.len = ngx_strlen(ctx->tpc_dst);
99
100 if (brix_net_target_parse(NULL, &url, &target, err, sizeof(err))
101 != NGX_OK)
102 {
103 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
104 "xrdhttp: invalid tpc.dst URL: %s; ignored", err);
105 } else if (ngx_strncasecmp(url.data, (u_char *) "https://", 8) != 0) {
106 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
107 "xrdhttp: tpc.dst must be https://; ignored");
108 } else {
109 if (brix_http_request_header_add(r, "Destination",
110 ctx->tpc_dst,
111 NULL) != NGX_OK)
112 {
113 return NGX_ERROR;
114 }
115 }
116 }
117
118 /*
119 * X-Xrootd-Tpc-Token → if Authorization: Bearer is absent, inject from
120 * the captured TPC token. This allows TPC delegation via this header.
121 */
122 if (ctx->tpc_token[0]
123 && r->headers_in.authorization == NULL)
124 {
125 char bearer_buf[sizeof("Bearer ") + XRDHTTP_TPC_KEY_MAX];
126 snprintf(bearer_buf, sizeof(bearer_buf), "Bearer %s", ctx->tpc_token);
127
128 if (brix_http_request_header_add(r, "Authorization", bearer_buf,
129 &h) != NGX_OK)
130 {
131 return NGX_ERROR;
132 }
133 /* Point nginx's fast-path field at this new entry. */
134 r->headers_in.authorization = h;
135 }
136
137 return NGX_OK;
138}

140/* checksum header injection */
141ngx_int_t
142xrdhttp_add_checksum_header(ngx_http_request_t *r,
143 ngx_fd_t fd,
144 const struct stat *sb)
145{
146 xrdhttp_req_ctx_t *ctx;
147 /* algo (32) + '=' + sha256 hex (64) + NUL = 98; round up generously */
148 char hdr_value[256];
149 char algo[32];
150 brix_checksum_alg_t alg;
151
152 ctx = (xrdhttp_req_ctx_t *)
153 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
154 if (ctx == NULL || !ctx->want_cksum[0] || fd == NGX_INVALID_FILE) {
155 return NGX_OK;
156 }
157
158 if (sb->st_size > (off_t) 2 * 1024 * 1024 * 1024LL) {
159 /* Skip checksum computation for files > 2 GiB to bound latency. */
160 ngx_log_debug0(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
161 "xrdhttp: skipping checksum for large file (> 2 GiB)");
162 return NGX_OK;
163 }
164
165 if (brix_checksum_parse(ctx->want_cksum, strlen(ctx->want_cksum),
166 &alg, algo, sizeof(algo)) == NGX_OK)
167 {
168 brix_integrity_info_t info;
169 brix_integrity_opts_t iopts;
170
171 ngx_memzero(&iopts, sizeof(iopts));
172 iopts.allow_xattr_cache = 1;
173 iopts.update_xattr_cache = 1;
174
175 if (brix_integrity_get_fd(r->connection->log, fd, NULL, "<xrdhttp>",
176 algo, &iopts, &info) != NGX_OK
177 || brix_integrity_format_http_digest(&info, hdr_value,
178 sizeof(hdr_value)) != NGX_OK)
179 {
180 return NGX_OK;
181 }
182 } else {
183 char safe_alg[sizeof(ctx->want_cksum) * 4];
184 brix_sanitize_log_string(ctx->want_cksum,
185 safe_alg, sizeof(safe_alg));
186 ngx_log_debug1(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
187 "xrdhttp: unsupported checksum algorithm \"%s\"",
188 safe_alg);
189 return NGX_OK;
190 }
191
192 return brix_http_set_header(r, "Digest", hdr_value, NULL);
193}

195/* streaming Digest body filter (Phase 21 Step B) */
196ngx_int_t
197xrdhttp_digest_body_filter(ngx_http_request_t *r, ngx_chain_t *in,
198 ngx_http_output_body_filter_pt next)
199{
200 xrdhttp_req_ctx_t *ctx;
201 ngx_chain_t *cl;
202 ngx_uint_t saw_last = 0;
203
204 ctx = (xrdhttp_req_ctx_t *)
205 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
206
207 if (ctx == NULL || !ctx->is_xrdhttp || !ctx->compute_digest) {
208 return next(r, in); /* pass-through */
209 }
210
211 if (ctx->adler == 0) {
212 ctx->adler = (uint32_t) adler32(0L, Z_NULL, 0); /* seed = 1 */
213 }
214
215 for (cl = in; cl != NULL; cl = cl->next) {
216 ngx_buf_t *b = cl->buf;
217
218 if (ngx_buf_in_memory(b) && b->last > b->pos) {
219 ctx->adler = (uint32_t) adler32((uLong) ctx->adler, b->pos,
220 (uInt) (b->last - b->pos));
221 }
222 if (b->last_buf || b->last_in_chain) {
223 saw_last = 1;
224 }
225 }
226
227 /* On the final buffer, queue the Digest as a response trailer. nginx only
228 * transmits trailers on chunked HTTP/1.1 and HTTP/2/3 responses; for a
229 * fixed Content-Length GET the fd-based xrdhttp_add_checksum_header() path
230 * supplies the Digest header instead, so this is purely additive. */
231 if (saw_last && !ctx->digest_emitted) {
232 ngx_table_elt_t *h = ngx_list_push(&r->headers_out.trailers);
233 if (h != NULL) {
234 u_char *v = ngx_pnalloc(r->pool, sizeof("adler32=") - 1 + 8 + 1);
235 if (v != NULL) {
236 size_t n = ngx_sprintf(v, "adler32=%08xD", ctx->adler) - v;
237 h->hash = 1;
238 ngx_str_set(&h->key, "Digest");
239 h->value.data = v;
240 h->value.len = n;
241 }
242 }
243 ctx->digest_emitted = 1;
244 }
245
246 return next(r, in);
247}
