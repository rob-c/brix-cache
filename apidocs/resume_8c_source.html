10/*
11 * Re-enter the XRootD event flow after a thread-pool completion.
12 *
13 * All three helpers are called from AIO done callbacks running on the nginx
14 * main thread (posted via ngx_post_event). They check ctx->destroyed first
15 * so that a callback that fires after the client has disconnected can abort
16 * without touching freed memory.
17 */
18
19/*
20 * brix_aio_restore_stream — restore the stream-id context field so the
21 * subsequent response builder uses the correct stream-id from the request that
22 * triggered the AIO.
23 *
24 * Returns 1 if the connection is still alive, 0 if it has been destroyed.
25 */
26ngx_flag_t
27brix_aio_restore_stream(brix_ctx_t *ctx, const u_char streamid[2])
28{
29 if (ctx->destroyed) {
30 return 0;
31 }
32
33 ctx->recv.cur_streamid[0] = streamid[0];
34 ctx->recv.cur_streamid[1] = streamid[1];
35
36 return 1;
37}

39/*
40 * brix_aio_restore_request — like brix_aio_restore_stream, but also
41 * resets the connection state back to XRD_ST_REQ_HEADER so the recv loop
42 * can accept the next request.
43 *
44 * Call this from done callbacks that complete the request cycle (i.e. the
45 * response has already been queued before calling brix_aio_resume).
46 */
47ngx_flag_t
48brix_aio_restore_request(brix_ctx_t *ctx, const u_char streamid[2])
49{
50 if (!brix_aio_restore_stream(ctx, streamid)) {
51 return 0;
52 }
53
54 ctx->state = XRD_ST_REQ_HEADER;
55 ctx->recv.hdr_pos = 0;
56
57 return 1;
58}

60/*
61 * brix_aio_post_task — post a pre-built ngx_thread_task_t to the thread
62 * pool and transition ctx->state to XRD_ST_AIO.
63 *
64 * If pool is NULL (thread pool not configured), returns NGX_OK without posting
65 * and leaves *posted = 0 so the caller falls back to synchronous I/O.
66 *
67 * If ngx_thread_task_post() fails (pool queue full), logs a warning at
68 * NGX_LOG_WARN and returns NGX_OK with *posted = 0 — again the caller falls
69 * back to synchronous I/O rather than failing the request.
70 *
71 * On success: *posted = 1, ctx->state = XRD_ST_AIO.
72 */
73ngx_int_t
74brix_aio_post_task(brix_ctx_t *ctx, ngx_connection_t *c,
75 ngx_thread_pool_t *pool, ngx_thread_task_t *task,
76 const char *fallback_log, ngx_flag_t *posted)
77{
78 *posted = 0;
79
80 /*
81 * Phase 44 tier 1: io_uring. Compiled out entirely in a stub build (no
82 * liburing), so this is exactly the historical thread-pool path there. The
83 * ring is selected only when it is up for this worker, not killed at
84 * runtime, the op is mapped, and the ring is not full; any prep/submit
85 * failure leaves *posted = 0 and falls through to the thread pool below.
86 */
87#if (BRIX_HAVE_LIBURING)
88 {
89 brix_uring_t *u = brix_uring_worker();
90 brix_uring_op_e op = brix_uring_op_for(task);
91
92 if (u != NULL && u->enabled && op != XRD_URING_OP_NONE) {
93 if (!brix_uring_disabled(u) && u->inflight < u->queue_depth
94 && brix_uring_submit(ctx, c, task, op, posted) == NGX_OK
95 && *posted)
96 {
97 /* metrics: a mapped op that the ring accepted. */
98 BRIX_SRV_METRIC_INC(ctx, io_uring_ops_total);
99 if (ctx->metrics != NULL && ctx->metrics->io_uring_active == 0) {
100 ctx->metrics->io_uring_active = 1;
101 }
102 ctx->state = XRD_ST_AIO;
103 return NGX_OK;
104 }
105 /* metrics: a mapped op the ring could not take (killed / full /
106 * submit failed) — it falls through to the pool below. */
107 BRIX_SRV_METRIC_INC(ctx, io_uring_fallback_total);
108 }
109 }
110#endif
111
112 /* Tier 2: thread pool (always built; the fallback for an unmapped op, a
113 * full ring, or a stub build). */
114 if (pool == NULL) {
115 return NGX_OK; /* tier 3: caller does inline sync I/O */
116 }
117
118 if (ngx_thread_task_post(pool, task) != NGX_OK) {
119 ngx_log_error(NGX_LOG_WARN, c->log, 0, "%s", fallback_log);
120 return NGX_OK;
121 }
122
123 ctx->state = XRD_ST_AIO;
124 *posted = 1;
125
126 return NGX_OK;
127}

129/*
130 * brix_aio_resume — schedule the next event-loop step after an AIO task
131 * completes and its done callback has already queued the response.
132 *
133 * If the response put the connection into XRD_ST_SENDING (it could not be
134 * sent immediately), arm the write event. Otherwise arm the read event so
135 * pipelined requests buffered in the kernel are processed without waiting for
136 * a new epoll notification.
137 *
138 * Guards against ctx->destroyed (client disconnected while AIO was in flight).
139 */
140void
141brix_aio_resume(ngx_connection_t *c)
142{
143 ngx_stream_session_t *s;
144 brix_ctx_t *ctx;
145
146 /*
147 * AIO completion runs outside the normal recv path. If the completion
148 * queued a response that still has bytes pending, resume the write side
149 * first. Once there is no pending response, post the read side so already
150 * arrived pipelined requests run before the next epoll_wait.
151 */
152
153 s = c->data;
154 ctx = ngx_stream_get_module_ctx(s, ngx_stream_brix_module);
155 if (ctx == NULL || ctx->destroyed) {
156 return;
157 }
158
159 ngx_log_debug5(NGX_LOG_DEBUG_STREAM, c->log, 0,
160 "brix: aio_resume state=%d ravail=%d rready=%d "
161 "wready=%d wposted=%d",
162 (int) ctx->state, c->read->available,
163 (int) c->read->ready, (int) c->write->ready,
164 (int) c->write->posted);
165
166 if (ctx->state == XRD_ST_SENDING) {
167 /* Resuming a suspended sender — force the write event to run so the
168 * parked out_ring drains even if wev->ready is a stale 0 (the pipeline
169 * saturation stall). brix_send() re-arms via schedule_write_resume on
170 * a genuine NGX_AGAIN, so this cannot busy-loop. */
171 if (brix_ensure_write_event(c) != NGX_OK) {
172 brix_aio_resume_fail(ctx, c, s);
173 }
174 return;
175 }
176
177 if (brix_schedule_read_resume(c) != NGX_OK) {
178 brix_aio_resume_fail(ctx, c, s);
179 }
180}

182/*
183 * brix_aio_resume_fail — finalize after a resume-scheduling failure (epoll
184 * re-arm error), honouring the deferred-teardown guard. brix_aio_resume runs
185 * inside an AIO done handler: sibling completions for this connection (e.g.
186 * further pipelined pwrites) may still sit in ngx_posted_events behind us, and
187 * their events live in c->pool — finalizing here without the guard destroys
188 * that pool and corrupts the posted queue when the loop reaches them.
189 */
190static void
191brix_aio_resume_fail(brix_ctx_t *ctx, ngx_connection_t *c,
192 ngx_stream_session_t *s)
193{
194 if (brix_defer_teardown_if_writing(ctx, c,
195 NGX_STREAM_INTERNAL_SERVER_ERROR))
196 {
197 return;
198 }
199 brix_on_disconnect(ctx, c);
200 brix_close_all_files(ctx);
201 ngx_stream_finalize_session(s, NGX_STREAM_INTERNAL_SERVER_ERROR);
202}
