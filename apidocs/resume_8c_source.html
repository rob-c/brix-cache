5/*
6 * Re-enter the XRootD event flow after a thread-pool completion.
7 *
8 * All three helpers are called from AIO done callbacks running on the nginx
9 * main thread (posted via ngx_post_event). They check ctx->destroyed first
10 * so that a callback that fires after the client has disconnected can abort
11 * without touching freed memory.
12 */

14/*
15 * brix_aio_restore_stream — restore the stream-id context field so the
16 * subsequent response builder uses the correct stream-id from the request that
17 * triggered the AIO.
18 *
19 * Returns 1 if the connection is still alive, 0 if it has been destroyed.
20 */
21ngx_flag_t
22brix_aio_restore_stream(brix_ctx_t *ctx, const u_char streamid[2])
23{
24 if (ctx->destroyed) {
25 return 0;
26 }
27
28 ctx->recv.cur_streamid[0] = streamid[0];
29 ctx->recv.cur_streamid[1] = streamid[1];
30
31 return 1;
32}

34/*
35 * brix_aio_restore_request — like brix_aio_restore_stream, but also
36 * resets the connection state back to XRD_ST_REQ_HEADER so the recv loop
37 * can accept the next request.
38 *
39 * Call this from done callbacks that complete the request cycle (i.e. the
40 * response has already been queued before calling brix_aio_resume).
41 */
42ngx_flag_t
43brix_aio_restore_request(brix_ctx_t *ctx, const u_char streamid[2])
44{
45 if (!brix_aio_restore_stream(ctx, streamid)) {
46 return 0;
47 }
48
49 ctx->state = XRD_ST_REQ_HEADER;
50 ctx->recv.hdr_pos = 0;
51
52 return 1;
53}

55/*
56 * brix_aio_post_task — post a pre-built ngx_thread_task_t to the thread
57 * pool and transition ctx->state to XRD_ST_AIO.
58 *
59 * If pool is NULL (thread pool not configured), returns NGX_OK without posting
60 * and leaves *posted = 0 so the caller falls back to synchronous I/O.
61 *
62 * If ngx_thread_task_post() fails (pool queue full), logs a warning at
63 * NGX_LOG_WARN and returns NGX_OK with *posted = 0 — again the caller falls
64 * back to synchronous I/O rather than failing the request.
65 *
66 * On success: *posted = 1, ctx->state = XRD_ST_AIO.
67 */
68ngx_int_t
69brix_aio_post_task(brix_ctx_t *ctx, ngx_connection_t *c,
70 ngx_thread_pool_t *pool, ngx_thread_task_t *task,
71 const char *fallback_log, ngx_flag_t *posted)
72{
73 *posted = 0;
74
75 /*
76 * Phase 44 tier 1: io_uring. Compiled out entirely in a stub build (no
77 * liburing), so this is exactly the historical thread-pool path there. The
78 * ring is selected only when it is up for this worker, not killed at
79 * runtime, the op is mapped, and the ring is not full; any prep/submit
80 * failure leaves *posted = 0 and falls through to the thread pool below.
81 */
82#if (BRIX_HAVE_LIBURING)
83 {
84 brix_uring_t *u = brix_uring_worker();
85 brix_uring_op_e op = brix_uring_op_for(task);
86
87 if (u != NULL && u->enabled && op != XRD_URING_OP_NONE) {
88 if (!brix_uring_disabled(u) && u->inflight < u->queue_depth
89 && brix_uring_submit(ctx, c, task, op, posted) == NGX_OK
90 && *posted)
91 {
92 /* metrics: a mapped op that the ring accepted. */
93 BRIX_SRV_METRIC_INC(ctx, io_uring_ops_total);
94 if (ctx->metrics != NULL && ctx->metrics->io_uring_active == 0) {
95 ctx->metrics->io_uring_active = 1;
96 }
97 ctx->state = XRD_ST_AIO;
98 return NGX_OK;
99 }
100 /* metrics: a mapped op the ring could not take (killed / full /
101 * submit failed) — it falls through to the pool below. */
102 BRIX_SRV_METRIC_INC(ctx, io_uring_fallback_total);
103 }
104 }
105#endif
106
107 /* Tier 2: thread pool (always built; the fallback for an unmapped op, a
108 * full ring, or a stub build). */
109 if (pool == NULL) {
110 return NGX_OK; /* tier 3: caller does inline sync I/O */
111 }
112
113 if (ngx_thread_task_post(pool, task) != NGX_OK) {
114 ngx_log_error(NGX_LOG_WARN, c->log, 0, "%s", fallback_log);
115 return NGX_OK;
116 }
117
118 ctx->state = XRD_ST_AIO;
119 *posted = 1;
120
121 return NGX_OK;
122}

124/*
125 * brix_aio_resume — schedule the next event-loop step after an AIO task
126 * completes and its done callback has already queued the response.
127 *
128 * If the response put the connection into XRD_ST_SENDING (it could not be
129 * sent immediately), arm the write event. Otherwise arm the read event so
130 * pipelined requests buffered in the kernel are processed without waiting for
131 * a new epoll notification.
132 *
133 * Guards against ctx->destroyed (client disconnected while AIO was in flight).
134 */
135void
136brix_aio_resume(ngx_connection_t *c)
137{
138 ngx_stream_session_t *s;
139 brix_ctx_t *ctx;
140
141 /*
142 * AIO completion runs outside the normal recv path. If the completion
143 * queued a response that still has bytes pending, resume the write side
144 * first. Once there is no pending response, post the read side so already
145 * arrived pipelined requests run before the next epoll_wait.
146 */
147
148 s = c->data;
149 ctx = ngx_stream_get_module_ctx(s, ngx_stream_brix_module);
150 if (ctx == NULL || ctx->destroyed) {
151 return;
152 }
153
154 ngx_log_debug5(NGX_LOG_DEBUG_STREAM, c->log, 0,
155 "brix: aio_resume state=%d ravail=%d rready=%d "
156 "wready=%d wposted=%d",
157 (int) ctx->state, c->read->available,
158 (int) c->read->ready, (int) c->write->ready,
159 (int) c->write->posted);
160
161 if (ctx->state == XRD_ST_SENDING) {
162 if (brix_schedule_write_resume(c) != NGX_OK) {
163 ngx_stream_finalize_session(c->data, NGX_STREAM_INTERNAL_SERVER_ERROR);
164 }
165 return;
166 }
167
168 if (brix_schedule_read_resume(c) != NGX_OK) {
169 ngx_stream_finalize_session(c->data, NGX_STREAM_INTERNAL_SERVER_ERROR);
170 }
171}
