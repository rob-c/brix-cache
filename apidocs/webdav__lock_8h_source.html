26/* setxattr the encoded lock onto `path`. Pass flags=XATTR_CREATE for atomic
27 * cross-worker lock acquisition: NGX_DECLINED means another worker won the race
28 * (EEXIST -> caller maps to 423). NGX_OK / NGX_ERROR otherwise. Takes the
29 * request (not just a log) so the lock xattr is written as the mapped user under
30 * impersonation (root_canon is derived from the request's loc conf). */

32 const webdav_lock_xattr_t *e, int flags);

36 webdav_lock_xattr_t *e);
37/* removexattr the lock on `path`. Idempotent: NGX_OK even if absent; NGX_ERROR

41/*
42 * webdav_lock_startup_sweep — recursively remove every persisted lock xattr
43 * (WEBDAV_LOCK_XATTR_KEY) under root_canon. Used at startup when

50/* Walk path -> export root checking each level's lock xattr (O(path_depth)
51 * getxattr calls); expired locks are lazily deleted in passing. NGX_HTTP_LOCKED
52 * if an unexpired lock not matched by the request If header covers `path`; 500
53 * on a getxattr fault; NGX_OK otherwise. need_write is currently advisory. */
54ngx_int_t webdav_check_locks(ngx_http_request_t *r, const char *path,
55 int need_write);
56/* webdav_check_locks() plus, when `path` is a directory, a recursive descendant
57 * scan (opendir walk). Use for DELETE/COPY/MOVE on collections so a lock on any
58 * child blocks the op. Same return codes as webdav_check_locks(). */
59ngx_int_t webdav_check_locks_tree(ngx_http_request_t *r, const char *path);
60/* Append <D:lockdiscovery> for `path` to the head/tail chain (r->pool), reading
61 * the live lock xattr (empty element if none/expired; owner XML-escaped; expired
62 * locks lazily deleted). NGX_OK / NGX_ERROR on chain alloc failure. */
63ngx_int_t webdav_lock_append_discovery(ngx_http_request_t *r, const char *path,
64 ngx_chain_t **head, ngx_chain_t **tail);
65/* Append the static <D:supportedlock> element (exclusive+shared write) to the
66 * head/tail chain (r->pool). NGX_OK / NGX_ERROR on chain alloc failure. */
67ngx_int_t webdav_lock_append_supported(ngx_http_request_t *r,
68 ngx_chain_t **head, ngx_chain_t **tail);
