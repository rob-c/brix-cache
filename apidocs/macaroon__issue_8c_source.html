1/*
2 * macaroon_issue.c — WLCG macaroon token issuance.
3 *
4 * WHAT: Mints new single-root macaroon tokens by constructing the binary
5 * packet sequence (location, identifier, first-party caveats, signature),
6 * computing the HMAC-SHA256 chain, and base64url-encoding the result.
7 *
8 * WHY: The existing macaroon.c only validates tokens. The macaroon issuance
9 * endpoint (POST /.oauth2/token) needs to mint new tokens for WLCG
10 * third-party delegation — a client authenticates, requests a scoped token,
11 * and the server issues a macaroon that can be delegated to a TPC agent.
12 *
13 * HOW: Packet wire format: [4-hex plen][label][ ][value][\n]
14 * where plen includes the 4 hex chars. The signature packet omits the
15 * trailing newline (matches XrdMacaroons convention and the test helper).
16 * HMAC chain: sig = HMAC(root_key, identifier_value), then for each
17 * first-party caveat: sig = HMAC(sig, caveat_value). The location packet
18 * is informational only and does not enter the HMAC chain.
19 */

31/*
32 * write_packet — append one macaroon packet to the output buffer.
33 *
34 * Format: [4-hex plen][label][ ][value][\n] (newline added when add_newline=1)
35 * plen counts all bytes including the 4-hex prefix itself.
36 *
37 * Updates *buf and *rem. Returns bytes written, or -1 on overflow.
38 */
39static int
40write_packet(u_char **buf, size_t *rem,
41 const char *label, size_t label_len,
42 const u_char *value, size_t val_len,
43 int add_newline)
44{
45 size_t plen = 4 + label_len + 1 + val_len + (size_t)(add_newline ? 1 : 0);
46 char hexlen[5];
47
48 if (plen > *rem || plen > 0xFFFF) {
49 return -1;
50 }
51
52 snprintf(hexlen, sizeof(hexlen), "%04x", (unsigned int) plen);
53 ngx_memcpy(*buf, hexlen, 4);
54 *buf += 4;
55 ngx_memcpy(*buf, label, label_len);
56 *buf += label_len;
57 **buf = ' ';
58 (*buf)++;
59 ngx_memcpy(*buf, value, val_len);
60 *buf += val_len;
61 if (add_newline) {
62 **buf = '\n';
63 (*buf)++;
64 }
65 *rem -= plen;
66 return (int) plen;
67}

70brix_macaroon_issue(ngx_log_t *log,
71 const u_char *root_key, size_t root_key_len,
72 const char *location,
73 const char *identifier,
74 const char *activities,
75 const char *path,
76 time_t expiry,
77 char *out_b64, size_t out_b64sz)
78{
79 u_char bin[ISSUE_BIN_MAX];
80 u_char *p;
81 size_t rem;
82 u_char sig[32];
83 unsigned int sig_len = 32;
84
85 if (!identifier || !identifier[0] || !root_key || root_key_len == 0) {
86 ngx_log_error(NGX_LOG_ERR, log, 0,
87 "macaroon_issue: missing identifier or root key");
88 return NGX_ERROR;
89 }
90 if (out_b64sz < BRIX_MACAROON_ISSUE_OUT_MAX) {
91 ngx_log_error(NGX_LOG_ERR, log, 0,
92 "macaroon_issue: output buffer too small");
93 return NGX_ERROR;
94 }
95
96 p = bin;
97 rem = sizeof(bin);
98
99 /* location packet — informational only, does not update HMAC chain */
100 if (location && location[0]) {
101 if (write_packet(&p, &rem, "location", 8,
102 (const u_char *) location, strlen(location), 1) < 0)
103 {
104 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
105 return NGX_ERROR;
106 }
107 }
108
109 /* identifier packet — initialises the HMAC chain */
110 {
111 size_t idlen = strlen(identifier);
112 if (write_packet(&p, &rem, "identifier", 10,
113 (const u_char *) identifier, idlen, 1) < 0)
114 {
115 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
116 return NGX_ERROR;
117 }
118 /* sig = HMAC(root_key, identifier_value) */
119 HMAC(EVP_sha256(), root_key, (int) root_key_len,
120 (const u_char *) identifier, (int) idlen, sig, &sig_len);
121 }
122
123 /* activity: caveat — WLCG activity list (e.g. "DOWNLOAD,LIST") */
124 if (activities && activities[0]) {
125 char act_cav[128];
126 size_t act_len;
127 u_char next_sig[32];
128
129 act_len = (size_t) snprintf(act_cav, sizeof(act_cav),
130 "activity:%s", activities);
131 if (act_len >= sizeof(act_cav)) {
132 ngx_log_error(NGX_LOG_ERR, log, 0,
133 "macaroon_issue: activity caveat too long");
134 return NGX_ERROR;
135 }
136 if (write_packet(&p, &rem, "cid", 3,
137 (const u_char *) act_cav, act_len, 1) < 0)
138 {
139 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
140 return NGX_ERROR;
141 }
142 /* sig = HMAC(sig, caveat_value) */
143 HMAC(EVP_sha256(), sig, 32, (const u_char *) act_cav, (int) act_len,
144 next_sig, &sig_len);
145 ngx_memcpy(sig, next_sig, 32);
146 }
147
148 /* path: caveat — path prefix constraint */
149 if (path && path[0]) {
150 char path_cav[1024];
151 size_t path_cav_len;
152 u_char next_sig[32];
153
154 path_cav_len = (size_t) snprintf(path_cav, sizeof(path_cav),
155 "path:%s", path);
156 if (path_cav_len >= sizeof(path_cav)) {
157 ngx_log_error(NGX_LOG_ERR, log, 0,
158 "macaroon_issue: path caveat too long");
159 return NGX_ERROR;
160 }
161 if (write_packet(&p, &rem, "cid", 3,
162 (const u_char *) path_cav, path_cav_len, 1) < 0)
163 {
164 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
165 return NGX_ERROR;
166 }
167 HMAC(EVP_sha256(), sig, 32,
168 (const u_char *) path_cav, (int) path_cav_len,
169 next_sig, &sig_len);
170 ngx_memcpy(sig, next_sig, 32);
171 }
172
173 /* before: caveat — expiry in ISO8601 UTC */
174 if (expiry > 0) {
175 char exp_cav[64];
176 size_t exp_cav_len;
177 u_char next_sig[32];
178 struct tm tm_val;
179
180 gmtime_r(&expiry, &tm_val);
181 exp_cav_len = (size_t) snprintf(exp_cav, sizeof(exp_cav),
182 "before:%04d-%02d-%02dT%02d:%02d:%02dZ",
183 tm_val.tm_year + 1900, tm_val.tm_mon + 1, tm_val.tm_mday,
184 tm_val.tm_hour, tm_val.tm_min, tm_val.tm_sec);
185 if (exp_cav_len >= sizeof(exp_cav)) {
186 ngx_log_error(NGX_LOG_ERR, log, 0,
187 "macaroon_issue: expiry caveat overflow");
188 return NGX_ERROR;
189 }
190 if (write_packet(&p, &rem, "cid", 3,
191 (const u_char *) exp_cav, exp_cav_len, 1) < 0)
192 {
193 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
194 return NGX_ERROR;
195 }
196 HMAC(EVP_sha256(), sig, 32,
197 (const u_char *) exp_cav, (int) exp_cav_len,
198 next_sig, &sig_len);
199 ngx_memcpy(sig, next_sig, 32);
200 }
201
202 /* signature packet: [4-hex plen]["signature "][32 raw bytes]
203 * No trailing newline — matches XrdMacaroons wire convention. */
204 {
205 /* plen = 4 (hex header) + 10 ("signature ") + 32 (sig bytes) = 46 */
206 size_t plen = 46;
207 char hexlen[5];
208
209 if (plen > rem) {
210 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
211 return NGX_ERROR;
212 }
213 snprintf(hexlen, sizeof(hexlen), "%04x", (unsigned int) plen);
214 ngx_memcpy(p, hexlen, 4);
215 p += 4;
216 ngx_memcpy(p, "signature ", 10);
217 p += 10;
218 ngx_memcpy(p, sig, 32);
219 p += 32;
220 }
221
222 b64url_encode((const char *) bin, (size_t)(p - bin), out_b64, out_b64sz);
223 return NGX_OK;
224}
