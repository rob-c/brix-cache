1/*
2 * macaroon_issue.c — WLCG macaroon token issuance.
3 *
4 * WHAT: Mints new single-root macaroon tokens by constructing the binary
5 * packet sequence (location, identifier, first-party caveats, signature),
6 * computing the HMAC-SHA256 chain, and base64url-encoding the result.
7 *
8 * WHY: The existing macaroon.c only validates tokens. The macaroon issuance
9 * endpoint (POST /.oauth2/token) needs to mint new tokens for WLCG
10 * third-party delegation — a client authenticates, requests a scoped token,
11 * and the server issues a macaroon that can be delegated to a TPC agent.
12 *
13 * HOW: Packet wire format: [4-hex plen][label][ ][value][\n]
14 * where plen includes the 4 hex chars. The signature packet omits the
15 * trailing newline (matches XrdMacaroons convention and the test helper).
16 * HMAC chain: sig = HMAC(root_key, identifier_value), then for each
17 * first-party caveat: sig = HMAC(sig, caveat_value). The location packet
18 * is informational only and does not enter the HMAC chain.
19 */

31/*
32 * issue_buf_t — output buffer cursor for macaroon packet assembly.
33 *
34 * WHAT: Bundles the write cursor and remaining capacity so packet writers
35 * take one context parameter instead of a (buf, rem) pointer pair.
36 * WHY: Keeps helper signatures at or below the 5-parameter limit and makes
37 * the shared mutable state explicit (single owner, threaded by value).
38 * HOW: `p` advances as bytes are appended; `rem` decreases by the same amount.
39 * `start` is retained so the caller can compute the final packet length.
40 */
41typedef struct {
42 u_char *start;
43 u_char *p;
44 size_t rem;
45} issue_buf_t;

47/*
48 * write_packet — append one newline-terminated macaroon packet to the buffer.
49 *
50 * WHAT: Emits [4-hex plen][label][ ][value][\n] into `buf`, where plen counts
51 * all bytes including the 4-hex prefix and the trailing newline.
52 * WHY: Every newline-terminated macaroon element (location, identifier,
53 * caveats) shares this exact wire encoding, so a single writer preserves
54 * byte-for-byte layout. The signature packet is newline-less and has its
55 * own writer (emit_signature).
56 * HOW: Compute plen, reject on overflow, then copy the fields in order and
57 * advance the cursor. Returns bytes written, or -1 on overflow.
58 */
59static int
60write_packet(issue_buf_t *buf,
61 const char *label, size_t label_len,
62 const u_char *value, size_t val_len)
63{
64 size_t plen = 4 + label_len + 1 + val_len + 1;
65 char hexlen[5];
66
67 if (plen > buf->rem || plen > 0xFFFF) {
68 return -1;
69 }
70
71 snprintf(hexlen, sizeof(hexlen), "%04x", (unsigned int) plen);
72 ngx_memcpy(buf->p, hexlen, 4);
73 buf->p += 4;
74 ngx_memcpy(buf->p, label, label_len);
75 buf->p += label_len;
76 *buf->p = ' ';
77 buf->p++;
78 ngx_memcpy(buf->p, value, val_len);
79 buf->p += val_len;
80 *buf->p = '\n';
81 buf->p++;
82 buf->rem -= plen;
83 return (int) plen;
84}

86/*
87 * chain_hmac — advance the macaroon HMAC-SHA256 chain by one caveat.
88 *
89 * WHAT: Computes sig = HMAC(sig, msg) and stores the 32-byte result back
90 * into `sig` in place.
91 * WHY: Each first-party caveat re-keys the chain with the previous signature;
92 * centralising it keeps the crypto call order identical across caveats.
93 * HOW: HMAC into a scratch buffer (the key aliases the output otherwise) and
94 * copy the 32 bytes back over `sig`.
95 */
96static void
97chain_hmac(u_char sig[32], const u_char *msg, size_t msg_len)
98{
99 u_char next_sig[32];
100 unsigned int next_len = 32;
101
102 HMAC(EVP_sha256(), sig, 32, msg, (int) msg_len, next_sig, &next_len);
103 ngx_memcpy(sig, next_sig, 32);
104}

106/*
107 * emit_cid_caveat — write one "cid" caveat packet and extend the HMAC chain.
108 *
109 * WHAT: Appends the caveat value as a cid packet and folds it into `sig`.
110 * WHY: The activity, path, and before caveats differ only in how their value
111 * string is formatted; this helper unifies the write-then-chain step.
112 * HOW: write_packet (with newline) then chain_hmac over the same value bytes,
113 * preserving the original per-caveat order. Returns NGX_OK or NGX_ERROR.
114 */
115static ngx_int_t
116emit_cid_caveat(issue_buf_t *buf, u_char sig[32],
117 const char *cav, size_t cav_len, ngx_log_t *log)
118{
119 if (write_packet(buf, "cid", 3, (const u_char *) cav, cav_len) < 0) {
120 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
121 return NGX_ERROR;
122 }
123 chain_hmac(sig, (const u_char *) cav, cav_len);
124 return NGX_OK;
125}

127/*
128 * emit_location — append the informational location packet (no HMAC chaining).
129 *
130 * WHAT: Writes the location: packet when a non-empty location is supplied.
131 * WHY: The location is advisory metadata and must NOT enter the HMAC chain,
132 * so it is handled separately from the caveat helpers.
133 * HOW: Skip when absent; otherwise write_packet with a trailing newline.
134 * Returns NGX_OK (including the skipped case) or NGX_ERROR on overflow.
135 */
136static ngx_int_t
137emit_location(issue_buf_t *buf, const char *location, ngx_log_t *log)
138{
139 if (!location || !location[0]) {
140 return NGX_OK;
141 }
142 if (write_packet(buf, "location", 8,
143 (const u_char *) location, strlen(location)) < 0)
144 {
145 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
146 return NGX_ERROR;
147 }
148 return NGX_OK;
149}

151/*
152 * emit_identifier — append the identifier packet and seed the HMAC chain.
153 *
154 * WHAT: Writes the identifier: packet and initialises sig = HMAC(root_key, id).
155 * WHY: The identifier both appears on the wire and roots the signature chain;
156 * it is the one element keyed by the root key rather than by a prior sig.
157 * HOW: write_packet (with newline), then a single HMAC keyed by root_key.
158 * Returns NGX_OK or NGX_ERROR on overflow.
159 */
160static ngx_int_t
161emit_identifier(issue_buf_t *buf, u_char sig[32],
162 const u_char *root_key, size_t root_key_len,
163 const char *identifier, ngx_log_t *log)
164{
165 size_t idlen = strlen(identifier);
166 unsigned int sig_len = 32;
167
168 if (write_packet(buf, "identifier", 10,
169 (const u_char *) identifier, idlen) < 0)
170 {
171 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
172 return NGX_ERROR;
173 }
174 /* sig = HMAC(root_key, identifier_value) */
175 HMAC(EVP_sha256(), root_key, (int) root_key_len,
176 (const u_char *) identifier, (int) idlen, sig, &sig_len);
177 return NGX_OK;
178}

180/*
181 * emit_activity_caveat — append the WLCG "activity:" caveat if requested.
182 *
183 * WHAT: Formats "activity:<activities>" and emits it as a cid caveat.
184 * WHY: Encodes the granted WLCG activity list (e.g. "DOWNLOAD,LIST").
185 * HOW: snprintf into a fixed buffer, guard against truncation, then delegate
186 * to emit_cid_caveat. Skips silently when no activities are supplied.
187 */
188static ngx_int_t
189emit_activity_caveat(issue_buf_t *buf, u_char sig[32],
190 const char *activities, ngx_log_t *log)
191{
192 char act_cav[128];
193 size_t act_len;
194
195 if (!activities || !activities[0]) {
196 return NGX_OK;
197 }
198 act_len = (size_t) snprintf(act_cav, sizeof(act_cav),
199 "activity:%s", activities);
200 if (act_len >= sizeof(act_cav)) {
201 ngx_log_error(NGX_LOG_ERR, log, 0,
202 "macaroon_issue: activity caveat too long");
203 return NGX_ERROR;
204 }
205 return emit_cid_caveat(buf, sig, act_cav, act_len, log);
206}

236/*
237 * emit_expiry_caveat — append the "before:" ISO8601-UTC expiry caveat.
238 *
239 * WHAT: Formats "before:YYYY-MM-DDThh:mm:ssZ" and emits it as a cid caveat.
240 * WHY: Bounds token validity in time; omitted when expiry <= 0.
241 * HOW: gmtime_r + snprintf, guard truncation, delegate to emit_cid_caveat.
242 */
243static ngx_int_t
244emit_expiry_caveat(issue_buf_t *buf, u_char sig[32],
245 time_t expiry, ngx_log_t *log)
246{
247 char exp_cav[64];
248 size_t exp_cav_len;
249 struct tm tm_val;
250
251 if (expiry <= 0) {
252 return NGX_OK;
253 }
254 gmtime_r(&expiry, &tm_val);
255 exp_cav_len = (size_t) snprintf(exp_cav, sizeof(exp_cav),
256 "before:%04d-%02d-%02dT%02d:%02d:%02dZ",
257 tm_val.tm_year + 1900, tm_val.tm_mon + 1, tm_val.tm_mday,
258 tm_val.tm_hour, tm_val.tm_min, tm_val.tm_sec);
259 if (exp_cav_len >= sizeof(exp_cav)) {
260 ngx_log_error(NGX_LOG_ERR, log, 0,
261 "macaroon_issue: expiry caveat overflow");
262 return NGX_ERROR;
263 }
264 return emit_cid_caveat(buf, sig, exp_cav, exp_cav_len, log);
265}

267/*
268 * emit_signature — append the terminal signature packet.
269 *
270 * WHAT: Writes [4-hex plen]["signature "][32 raw sig bytes] with NO trailing
271 * newline, matching the XrdMacaroons wire convention.
272 * WHY: The signature packet is the only element carrying raw HMAC bytes and
273 * must be emitted last, verbatim, once the chain is complete.
274 * HOW: plen is fixed at 46 (4 + 10 + 32); guard remaining space, then copy the
275 * hex header, label, and signature. Returns NGX_OK or NGX_ERROR.
276 */
277static ngx_int_t
278emit_signature(issue_buf_t *buf, const u_char sig[32], ngx_log_t *log)
279{
280 /* plen = 4 (hex header) + 10 ("signature ") + 32 (sig bytes) = 46 */
281 size_t plen = 46;
282 char hexlen[5];
283
284 if (plen > buf->rem) {
285 ngx_log_error(NGX_LOG_ERR, log, 0, "macaroon_issue: buffer overflow");
286 return NGX_ERROR;
287 }
288 snprintf(hexlen, sizeof(hexlen), "%04x", (unsigned int) plen);
289 ngx_memcpy(buf->p, hexlen, 4);
290 buf->p += 4;
291 ngx_memcpy(buf->p, "signature ", 10);
292 buf->p += 10;
293 ngx_memcpy(buf->p, sig, 32);
294 buf->p += 32;
295 return NGX_OK;
296}

298ngx_int_t
299brix_macaroon_issue(ngx_log_t *log,
300 const u_char *root_key, size_t root_key_len,
301 const char *location,
302 const char *identifier,
303 const char *activities,
304 const char *path,
305 time_t expiry,
306 char *out_b64, size_t out_b64sz)
307{
308 u_char bin[ISSUE_BIN_MAX];
309 u_char sig[32] = {0};
310 issue_buf_t buf;
311
312 if (!identifier || !identifier[0] || !root_key || root_key_len == 0) {
313 ngx_log_error(NGX_LOG_ERR, log, 0,
314 "macaroon_issue: missing identifier or root key");
315 return NGX_ERROR;
316 }
317 if (out_b64sz < BRIX_MACAROON_ISSUE_OUT_MAX) {
318 ngx_log_error(NGX_LOG_ERR, log, 0,
319 "macaroon_issue: output buffer too small");
320 return NGX_ERROR;
321 }
322
323 buf.start = bin;
324 buf.p = bin;
325 buf.rem = sizeof(bin);
326
327 /* Assemble packets in wire order; each helper preserves the exact
328 * write-then-HMAC sequence of the original implementation. */
329 if (emit_location(&buf, location, log) != NGX_OK
330 || emit_identifier(&buf, sig, root_key, root_key_len,
331 identifier, log) != NGX_OK
332 || emit_activity_caveat(&buf, sig, activities, log) != NGX_OK
333 || emit_path_caveat(&buf, sig, path, log) != NGX_OK
334 || emit_expiry_caveat(&buf, sig, expiry, log) != NGX_OK
335 || emit_signature(&buf, sig, log) != NGX_OK)
336 {
337 return NGX_ERROR;
338 }
339
340 b64url_encode((const char *) bin, (size_t)(buf.p - buf.start),
341 out_b64, out_b64sz);
342 return NGX_OK;
343}
