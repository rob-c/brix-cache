49/* sd_s3_utc_now — current UTC as SigV4 amzdate (YYYYMMDDTHHMMSSZ) + datestamp
50 * (YYYYMMDD), computed self-contained from the epoch (civil-from-days) so it
51 * needs neither <time.h> (shadowed here) nor the module-only compat/time.c. */
52void
53sd_s3_utc_now(char amzdate[20], char datestamp[12])
54{
55 struct timeval tv;
56 long long secs, days, rem, z, era, y;
57 unsigned doe, yoe, doy, mp, d, m;
58 int hh, mm, ss;
59
60 gettimeofday(&tv, NULL);
61 secs = (long long) tv.tv_sec;
62 days = secs / 86400;
63 rem = secs % 86400;
64 if (rem < 0) { rem += 86400; days -= 1; }
65 hh = (int) (rem / 3600);
66 mm = (int) ((rem % 3600) / 60);
67 ss = (int) (rem % 60);
68
69 /* Howard Hinnant's civil_from_days (days since 1970-01-01, UTC). */
70 z = days + 719468;
71 era = (z >= 0 ? z : z - 146096) / 146097;
72 doe = (unsigned) (z - era * 146097);
73 yoe = (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365;
74 y = (long long) yoe + era * 400;
75 doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
76 mp = (5 * doy + 2) / 153;
77 d = doy - (153 * mp + 2) / 5 + 1;
78 m = mp < 10 ? mp + 3 : mp - 9;
79 y += (m <= 2);
80
81 /* Bound the year to 4 digits (real epochs are well within 0000-9999) so the
82 * fixed amzdate[20]/datestamp[12] buffers are provably large enough. */
83 {
84 int yr = (int) (((y % 10000) + 10000) % 10000);
85 unsigned mo = m % 100, da = d % 100;
86 snprintf(datestamp, 12, "%04d%02u%02u", yr, mo, da);
87 snprintf(amzdate, 20, "%04d%02u%02uT%02u%02u%02uZ",
88 yr, mo, da, hh % 100, mm % 100, ss % 100);
89 }
90}

100/* sd_s3_sign_ex — build the SigV4 x-amz-date / content-sha256 / Authorization
101 * header block for `method` on this object (payload = UNSIGNED-PAYLOAD), OPTIONALLY
102 * folding one extra signed header `ck_name: ck_val` into the canonical request,
103 * the SignedHeaders list, and the emitted header block. Used to attach an
104 * origin-verifiable body checksum (x-amz-checksum-crc32, base64) on PUT /
105 * UploadPart so the origin itself rejects a wire-corrupted body with BadDigest
106 * (brix_backend_put_checksum, fault-hardening #12). ck_name MUST be a lowercase
107 * header token that sorts AFTER "host" and BEFORE "x-amz-content-sha256"
108 * (x-amz-checksum-* satisfies this) so the fixed canonical ordering below holds.
109 * ck_name==NULL reproduces the plain three-header signature byte-for-byte.
110 * Ported from xrdc_s3_sign_v4_q; all kernels are shared. 0 / -1. */
111int
112sd_s3_sign_ex(const sd_s3_file *f, const char *method, const char *canon_qs,
113 const char *ck_name, const char *ck_val, char *hdrs, size_t hdrsz)
114{
115 const char *payload_hex = "UNSIGNED-PAYLOAD";
116 char host[300];
117 char amzdate[20], datestamp[12];
118 char canon_hex[65], sighex[65];
119 char canon[8192], scope[160], sts[640], enc_uri[2048];
120 char ck_canon[256], ck_signed[64], ck_emit[256];
121 uint8_t k4[32], sig[32];
122 int cn;
123
124 if (canon_qs == NULL) { canon_qs = ""; }
125 if (ck_name != NULL && (ck_val == NULL || ck_val[0] == '\0')) {
126 return -1; /* header requested but no value */
127 }
128
129 /* The one optional header is emitted into three renderings: the canonical
130 * "name:value\n" line (folded in sorted order between host and
131 * x-amz-content-sha256), its SignedHeaders token, and the wire header line.
132 * Absent → all three empty, leaving the plain signature unchanged. */
133 if (ck_name != NULL) {
134 cn = snprintf(ck_canon, sizeof(ck_canon), "%s:%s\n", ck_name, ck_val);
135 if (cn < 0 || (size_t) cn >= sizeof(ck_canon)) { return -1; }
136 cn = snprintf(ck_signed, sizeof(ck_signed), "%s;", ck_name);
137 if (cn < 0 || (size_t) cn >= sizeof(ck_signed)) { return -1; }
138 cn = snprintf(ck_emit, sizeof(ck_emit), "%s: %s\r\n", ck_name, ck_val);
139 if (cn < 0 || (size_t) cn >= sizeof(ck_emit)) { return -1; }
140 } else {
141 ck_canon[0] = ck_signed[0] = ck_emit[0] = '\0';
142 }
143
144 brix_format_host_port(f->host, (uint16_t) f->port, host, sizeof(host));
145 sd_s3_utc_now(amzdate, datestamp);
146
147 if (brix_http_urlencode((const unsigned char *) f->key, strlen(f->key),
148 enc_uri, sizeof(enc_uri), "/") < 0) {
149 return -1;
150 }
151 cn = snprintf(canon, sizeof(canon),
152 "%s\n%s\n%s\nhost:%s\n%sx-amz-content-sha256:%s\nx-amz-date:%s\n\n"
153 "host;%sx-amz-content-sha256;x-amz-date\n%s",
154 method, enc_uri, canon_qs, host, ck_canon, payload_hex, amzdate,
155 ck_signed, payload_hex);
156 if (cn < 0 || (size_t) cn >= sizeof(canon)) {
157 return -1;
158 }
159 sd_s3_sha256_hex(canon, strlen(canon), canon_hex);
160
161 cn = snprintf(scope, sizeof(scope), "%s/%s/s3/aws4_request",
162 datestamp, f->region);
163 if (cn < 0 || (size_t) cn >= sizeof(scope)) {
164 return -1;
165 }
166 cn = snprintf(sts, sizeof(sts), "AWS4-HMAC-SHA256\n%s\n%s\n%s",
167 amzdate, scope, canon_hex);
168 if (cn < 0 || (size_t) cn >= sizeof(sts)) {
169 return -1;
170 }
171
172 if (!brix_sigv4_signing_key((const uint8_t *) f->sk, strlen(f->sk),
173 datestamp, f->region, "s3", k4)
174 || !brix_hmac_sha256(k4, 32, (const uint8_t *) sts, strlen(sts), sig)) {
175 return -1;
176 }
177 brix_hex_encode(sig, 32, sighex);
178
179 cn = snprintf(hdrs, hdrsz,
180 "x-amz-date: %s\r\nx-amz-content-sha256: %s\r\n%s"
181 "Authorization: AWS4-HMAC-SHA256 Credential=%s/%s, "
182 "SignedHeaders=host;%sx-amz-content-sha256;x-amz-date, Signature=%s\r\n",
183 amzdate, payload_hex, ck_emit, f->ak, scope, ck_signed, sighex);
184 if (cn < 0 || (size_t) cn >= hdrsz) {
185 return -1;
186 }
187 return 0;
188}

199/* Map a non-2xx HTTP status to -1 with a diagnostic; returns -1 always. Sets
200 * errno to the matching POSIX class (EACCES for 401/403, ENOENT for 404, else
201 * EIO) so callers can recover the error category — the message channel alone
202 * loses it. The client S3 VFS maps this errno to XRDC_EAUTH/ENOENT; the server
203 * S3 backend maps it to kXR via the canonical errno↔kXR table. */
204int
205sd_s3_status_err(int status, const char *op, const char *key,
206 char *errbuf, size_t errcap)
207{
208 if (status == 401 || status == 403) {
209 sd_s3_set_err(errbuf, errcap,
210 "s3 %s: auth failed (HTTP %d) on %s — check AK/SK/region",
211 op, status, key);
212 errno = EACCES;
213 } else if (status == 404) {
214 sd_s3_set_err(errbuf, errcap, "s3 %s: not found (HTTP 404) on %s",
215 op, key);
216 errno = ENOENT;
217 } else {
218 sd_s3_set_err(errbuf, errcap, "s3 %s: HTTP %d on %s", op, status, key);
219 errno = EIO;
220 }
221 return -1;
222}
