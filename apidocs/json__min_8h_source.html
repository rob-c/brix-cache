1/*
2 * json_min.h — minimal, dependency-free JSON value extractor (libxrdproto).
3 *
4 * WHAT: Extract one top-level object member's value from a JSON document, with
5 * no external dependency (libc only — NO jansson). String values are
6 * unescaped; scalar values (number/true/false/null) are returned as their
7 * raw token text.
8 * WHY: The native client (xrddiag/explain) needs to peek at a few JWT claims for
9 * DIAGNOSTIC display only, and must not link jansson. The previous client
10 * helper was a fragile strstr() hack that mis-parsed escaped quotes and
11 * could match a key inside a value; this is a proper string-aware scanner.
12 * The server keeps its jansson-based src/token/json.c for the security-
13 * critical JWKS/JWT paths — this is strictly the lightweight diagnostic core.
14 * HOW: A small non-recursive scanner walks the top-level object, correctly
15 * skipping over string contents and nested objects/arrays so a key is only
16 * matched in object-key position at depth 1.
17 *
18 * This is a DIAGNOSTIC helper, not a validating parser: it does not reject every
19 * malformed document, only enough to extract a well-formed member safely.
20 */
21#ifndef BRIX_JSON_MIN_H
22#define BRIX_JSON_MIN_H

24#include <stddef.h>

26/*
27 * Extract the value of top-level object key `key` from the JSON document in
28 * [json, json+len). A string value has its JSON escapes decoded (including
29 * \uXXXX, with surrogate pairs, to UTF-8); a scalar value (number/true/false/
30 * null) is copied as its raw token text. An object/array value is NOT copied
31 * (returns 0) since it has no flat-string form. The result is written
32 * NUL-terminated into out[outsz], truncated to fit.
33 *
34 * Returns 1 if the key was found and a string/scalar value copied, 0 otherwise
35 * (not found, value is object/array, malformed input, or out too small).
36 */
37int brix_json_get_str(const char *json, size_t len, const char *key,
