1/*
2 * vfs_deleg_bind.c — VFS delegation bag binding + ctx mode reporting
3 * (phase-70 §4, §5.1). Split verbatim from vfs_deleg.c.
4 *
5 * WHAT: The capture-side seam that constructs the per-request live-cred bag and
6 * reports the resolved delegation mode back to the cred gate:
7 * brix_vfs_ctx_bind_backend_deleg() — hang a borrowed bag on a VFS ctx.
8 * brix_vfs_deleg_set_exchange() — stamp EXCHANGE conf onto the bag.
9 * brix_vfs_deleg_bind() — allocate + fill + bind the bag.
10 * brix_vfs_backend_mode() — report the ctx's resolved mode.
11 * brix_vfs_backend_accepts_proxy() — does the leaf backend take a proxy?
12 * brix_vfs_deleg_snapshot() — copy out mode+bearer for a child ctx.
13 *
14 * WHY: These functions only touch the bag layout and the ctx; they share no
15 * statics with the PASSTHROUGH/EXCHANGE materialiser in vfs_deleg.c, so
16 * they live here to keep each file well under the size budget. All entry
17 * points are public (declared in vfs.h via vfs_internal.h).
18 */
19#include "vfs_internal.h"

21/* ---- brix_vfs_ctx_bind_backend_deleg ---------------------------------------
22 *
23 * WHAT: Store a borrowed pointer to the front door's live-cred bag on the ctx.
24 *
25 * WHY: Kept separate from brix_vfs_ctx_bind_backend_cred (dir-based select) so
26 * the ~35 existing bind sites are untouched and a request opts in to
27 * delegation explicitly. The bag carries BYTES, not a directory.
28 *
29 * HOW: The bag (and its byte fields) are owned by the caller's request pool
30 * and must outlive the VFS op; a NULL bag leaves the ctx on the SELECT
31 * path. No copy is made. */
32void
33brix_vfs_ctx_bind_backend_deleg(brix_vfs_ctx_t *vctx, brix_deleg_live_t *live)
34{
35 if (vctx == NULL) {
36 return;
37 }
38
39 vctx->deleg_live = live;
40}

42/* ---- brix_vfs_deleg_set_exchange -------------------------------------------
43 *
44 * WHAT: Populate the EXCHANGE conf (endpoint + client creds + audience) on the
45 * ctx's already-bound live-cred bag.
46 *
47 * WHY: The cred gate needs the RFC-8693 endpoint to trade a live bearer for a
48 * backend-audienced token (§5.4). The bag layout is private to the VFS, so
49 * this setter is the one place the borrowed conf strings are stamped onto
50 * it — kept separate from brix_vfs_deleg_bind so the capture site stays a
51 * single call plus an optional exchange-conf call (no signature churn on
52 * the bind path shared by PASSTHROUGH).
53 *
54 * HOW: A no-op when no bag is bound (nothing forwardable was captured) or the
55 * endpoint is empty (EXCHANGE then degrades to verbatim passthrough in the
56 * gate). All strings are borrowed (conf-owned, NUL-terminated). */
57void
58brix_vfs_deleg_set_exchange(brix_vfs_ctx_t *vctx,
59 const ngx_str_t *endpoint, const ngx_str_t *client_id,
60 const ngx_str_t *client_secret, const ngx_str_t *audience)
61{
62 brix_deleg_live_t *live;
63
64 if (vctx == NULL || vctx->deleg_live == NULL) {
65 return;
66 }
67 if (endpoint == NULL || endpoint->len == 0 || endpoint->data == NULL) {
68 return;
69 }
70
71 live = vctx->deleg_live;
72 live->tx.endpoint = *endpoint;
73 if (client_id != NULL) {
74 live->tx.client_id = *client_id;
75 }
76 if (client_secret != NULL) {
77 live->tx.client_secret = *client_secret;
78 }
79 if (audience != NULL) {
80 live->tx_audience = *audience;
81 }
82}

84/* ---- brix_vfs_deleg_bind ---------------------------------------------------
85 *
86 * WHAT: Allocate a live-cred bag from `pool`, fill it with the captured
87 * forwardable credential bytes (bearer text and/or full proxy PEM) plus
88 * the resolved mode, and bind it onto `vctx`. A no-op (SELECT path) when
89 * `mode` is BRIX_CRED_SELECT or nothing forwardable was captured.
90 *
91 * WHY: This is the single constructor for the bag, kept here (not in a protocol
92 * handler) because brix_deleg_live_t's layout is private to the VFS. Every
93 * front-door bind site calls this uniformly right after
94 * brix_vfs_ctx_bind_backend_cred, so the "conf mode + captured bytes →
95 * bound bag" step is stated once rather than copy-pasted per protocol.
96 *
97 * HOW: Borrows the byte ranges (owned by the caller's request pool). When no
98 * credential bytes are present the bag is not bound and the ctx stays on
99 * the dir-based SELECT path — so an operator enabling passthrough without a
100 * forwardable credential in-hand degrades to SELECT rather than deny here
101 * (the deny decision belongs to the cred gate, not the capture). */
102ngx_int_t
103brix_vfs_deleg_bind(ngx_pool_t *pool, brix_vfs_ctx_t *vctx,
104 enum brix_cred_mode mode, const ngx_str_t *bearer,
105 const ngx_str_t *proxy_pem)
106{
107 brix_deleg_live_t *live;
108 int have_proxy;
109 int have_bearer;
110
111 if (pool == NULL || vctx == NULL || mode == BRIX_CRED_SELECT) {
112 return NGX_OK;
113 }
114
115 have_proxy = (proxy_pem != NULL && proxy_pem->len > 0
116 && proxy_pem->data != NULL);
117 have_bearer = (bearer != NULL && bearer->len > 0 && bearer->data != NULL);
118
119 if (!have_proxy && !have_bearer) {
120 return NGX_OK; /* nothing forwardable captured — stay on SELECT */
121 }
122
123 live = ngx_pcalloc(pool, sizeof(*live));
124 if (live == NULL) {
125 return NGX_ERROR;
126 }
127
128 live->mode = mode;
129 if (have_proxy) {
130 live->have_proxy_pem = 1;
131 live->proxy_pem = *proxy_pem;
132 }
133 if (have_bearer) {
134 live->bearer = *bearer;
135 }
136
137 brix_vfs_ctx_bind_backend_deleg(vctx, live);
138 return NGX_OK;
139}

141/* ---- brix_vfs_backend_mode -------------------------------------------------
142 *
143 * WHAT: Report the delegation mode resolved for this ctx.
144 *
145 * WHY: The cred gate (vfs_cred.c) branches on the mode before the SELECT logic;
146 * keeping the lookup here means the "no bag ⇒ SELECT" default is stated
147 * once.
148 *
149 * HOW: Returns the bound bag's mode, or BRIX_CRED_SELECT when no bag is bound. */
150enum brix_cred_mode
151brix_vfs_backend_mode(brix_vfs_ctx_t *vctx)
152{
153 if (vctx == NULL || vctx->deleg_live == NULL) {
154 return BRIX_CRED_SELECT;
155 }
156
157 return vctx->deleg_live->mode;
158}

160/* ---- brix_vfs_backend_accepts_proxy ----------------------------------------
161 *
162 * WHAT: Report whether the ctx's resolved leaf backend consumes a forwarded
163 * X.509 proxy PEM.
164 *
165 * WHY: A protocol that forwards a captured proxy by default (the gsiftp→xrootd
166 * gateway) must not bind a proxy bag on a backend that cannot use one — the
167 * cred gate would then deny (EACCES) a request that should have served on
168 * the service credential. Gating the bind on this predicate keeps the
169 * default-on delegation scoped to proxy-capable backends (xroot, s3).
170 *
171 * HOW: brix_sd_cred_accept on the resolved leaf; NULL-safe (default-POSIX
172 * resolves to a NULL sd whose accept mask is 0). */
173int
174brix_vfs_backend_accepts_proxy(brix_vfs_ctx_t *vctx)
175{
176 if (vctx == NULL || vctx->sd == NULL) {
177 return 0;
178 }
179 return (brix_sd_cred_accept(brix_vfs_ns_leaf(vctx->sd))
180 & BRIX_SD_CRED_PROXY_PEM) ? 1 : 0;
181}

183/* ---- brix_vfs_deleg_snapshot -----------------------------------------------
184 *
185 * WHAT: Copy out the ctx's bound delegation mode + bearer bytes so a derived
186 * ctx (e.g. a recurse-child) can re-bind the same credential.
187 *
188 * WHY: Some ops build fresh child ctxs from a lightweight local struct rather
189 * than the parent ctx (root:// fattr recurse). Those children must carry
190 * the same passthrough credential; this borrows the mode+bearer so the
191 * child can re-bind via brix_vfs_deleg_bind without the bag layout leaking.
192 *
193 * HOW: No bound bag ⇒ mode=SELECT + empty bearer. Otherwise the bag's mode and
194 * a borrowed view of its bearer (same lifetime as the source). The proxy
195 * PEM is deliberately not exposed — it is a materialised secret. */
196void
197brix_vfs_deleg_snapshot(const brix_vfs_ctx_t *vctx,
198 enum brix_cred_mode *mode, ngx_str_t *bearer)
199{
200 if (mode != NULL) {
201 *mode = BRIX_CRED_SELECT;
202 }
203 if (bearer != NULL) {
204 ngx_str_null(bearer);
205 }
206
207 if (vctx == NULL || vctx->deleg_live == NULL) {
208 return;
209 }
210
211 if (mode != NULL) {
212 *mode = vctx->deleg_live->mode;
213 }
214 if (bearer != NULL) {
215 *bearer = vctx->deleg_live->bearer;
216 }
217}
