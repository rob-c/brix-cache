20static int
21parse_valid(const char *s)
22{
23 /* "HH:MM" → hours (rounded up on minutes); a bare integer → hours. */
24 const char *colon = strchr(s, ':');
25 if (colon != NULL) {
26 int h = atoi(s);
27 int m = atoi(colon + 1);
28 int hours = h + (m > 0 ? 1 : 0);
29 return hours > 0 ? hours : 12;
30 }
31 return atoi(s) > 0 ? atoi(s) : 12;
32}

54/* Cursor over the argument vector shared by every option parser. */
55typedef struct {
56 char **argv;
57 int argc;
58 int i;
59} arg_cursor;

61/*
62 * WHAT: Match the argument at the cursor against a long/short option pair and,
63 * on a hit, capture the following argument into *out.
64 * WHY: Every subcommand parses options as identical `-x`/`--x VALUE` pairs; a
65 * single matcher removes the repeated four-term boolean ladder that drove
66 * main's complexity while keeping the exact accepted spellings.
67 * HOW: On a spelling match with a value available, advance the cursor past the
68 * value, store it, and return 1; otherwise leave state untouched, return 0.
69 */
70static int
71opt_take_value(arg_cursor *ac, const char *short_name, const char *long_name,
72 const char **out)
73{
74 char *cur = ac->argv[ac->i];
75
76 if ((strcmp(cur, short_name) == 0 || strcmp(cur, long_name) == 0)
77 && ac->i + 1 < ac->argc)
78 {
79 *out = ac->argv[++ac->i];
80 return 1;
81 }
82 return 0;
83}

85/*
86 * WHAT: Parse the `init` option vector into a zeroed brix_proxy_opts.
87 * WHY: Isolates init's five-option grammar (valid/cert/key/out/bits) from the
88 * dispatch flow; -valid and -bits need numeric conversion, the rest are
89 * raw argv pointers, matching the original per-flag behavior exactly.
90 * HOW: Walk argv[2..]; each flag is an option pair captured via opt_take_value
91 * into a scratch string then applied. Any unmatched token → 0 (usage).
92 */
93static int
94init_parse_opts(int argc, char **argv, brix_proxy_opts *o)
95{
96 arg_cursor ac = { argv, argc, 0 };
97 const char *val;
98
99 memset(o, 0, sizeof(*o));
100
101 for (ac.i = 2; ac.i < argc; ac.i++) {
102 val = NULL;
103 if (opt_take_value(&ac, "-valid", "--valid", &val)) {
104 o->valid_hours = parse_valid(val);
105 } else if (opt_take_value(&ac, "-cert", "--cert", &val)) {
106 o->user_cert = val;
107 } else if (opt_take_value(&ac, "-key", "--key", &val)) {
108 o->user_key = val;
109 } else if (opt_take_value(&ac, "-out", "--out", &val)) {
110 o->out_path = val;
111 } else if (opt_take_value(&ac, "-bits", "--bits", &val)) {
112 o->bits = atoi(val);
113 } else {
114 return 0;
115 }
116 }
117 return 1;
118}

120/*
121 * WHAT: Run `xrdgsiproxy init`, returning the process exit code.
122 * WHY: Owns init end-to-end — parse, create the proxy, then echo it back —
123 * so the dispatcher stays a flat table lookup.
124 * HOW: Parse opts (usage=50 on error); brix_proxy_create failure surfaces via
125 * brix_shellcode; success re-reads the fresh proxy with brix_proxy_info.
126 */
127static int
128cmd_init(int argc, char **argv, brix_status *st)
129{
130 brix_proxy_opts o;
131 int rc;
132
133 if (!init_parse_opts(argc, argv, &o)) {
134 usage(argv[0]);
135 return 50;
136 }
137 rc = brix_proxy_create(&o, st);
138 if (rc != 0) {
139 fprintf(stderr, "xrdgsiproxy: init: %s\n", st->msg);
140 return brix_shellcode(st);
141 }
142 return brix_proxy_info(o.out_path, stdout, st); /* echo what we made */
143}

145/*
146 * WHAT: Parse the shared `-file`/`--file FILE` option of info/destroy.
147 * WHY: info and destroy accept exactly one option with identical grammar;
148 * one parser keeps their acceptance sets in lockstep.
149 * HOW: Walk argv[2..] capturing the file path; any other token → 0 (usage).
150 */
151static int
152file_parse_opts(int argc, char **argv, const char **file)
153{
154 arg_cursor ac = { argv, argc, 0 };
155
156 *file = NULL;
157 for (ac.i = 2; ac.i < argc; ac.i++) {
158 if (!opt_take_value(&ac, "-file", "--file", file)) {
159 return 0;
160 }
161 }
162 return 1;
163}

165/*
166 * WHAT: Run `xrdgsiproxy info`, returning the process exit code.
167 * WHY: info is tolerant of an absent proxy (mirroring stock xrdgsiproxy): a
168 * missing file is a soft "not found" notice + exit 1, not a hard error.
169 * HOW: Parse -file; brix_proxy_info on success returns 0; XRDC_ENOENT prints
170 * the notice and returns 1; any other failure escalates via brix_shellcode.
171 */
172static int
173cmd_info(int argc, char **argv, brix_status *st)
174{
175 const char *file;
176 int rc;
177
178 if (!file_parse_opts(argc, argv, &file)) {
179 usage(argv[0]);
180 return 50;
181 }
182 rc = brix_proxy_info(file, stdout, st);
183 if (rc == 0) {
184 return 0;
185 }
186 if (st->kxr == XRDC_ENOENT) {
187 printf("%s\n", st->msg);
188 return 1;
189 }
190 fprintf(stderr, "xrdgsiproxy: info: %s\n", st->msg);
191 return brix_shellcode(st);
192}

219typedef int (*cmd_fn)(int argc, char **argv, brix_status *st);

221static const struct {
222 const char *name;
223 cmd_fn fn;
224} cmd_table[] = {
225 { "init", cmd_init },
226 { "info", cmd_info },
227 { "destroy", cmd_destroy },
228};

230/*
231 * WHAT: Resolve a verb string to its handler, or NULL if unknown.
232 * WHY: Keeps the table private and main a flat lookup + call.
233 * HOW: Linear scan of the small fixed table.
234 */
235static cmd_fn
236cmd_lookup(const char *cmd)
237{
238 size_t n;
239
240 for (n = 0; n < sizeof(cmd_table) / sizeof(cmd_table[0]); n++) {
241 if (strcmp(cmd, cmd_table[n].name) == 0) {
242 return cmd_table[n].fn;
243 }
244 }
245 return NULL;
246}

248int
249main(int argc, char **argv)
250{
251 brix_status st;
252 const char *cmd;
253 cmd_fn fn;
254
255 if (argc < 2) { usage(argv[0]); return 50; }
256 cmd = argv[1];
257
258 /* --help / --version as the first argument. */
259 if (strcmp(cmd, "--version") == 0) {
260 printf("%s (BriX-Cache client) %s\n", brix_prog_base(argv[0]),
261 brix_client_version());
262 return 0;
263 }
264 if (strcmp(cmd, "--help") == 0 || strcmp(cmd, "-h") == 0) {
265 usage_fp(stdout, argv[0]);
266 return 0;
267 }
268
269 brix_crypto_init();
270 brix_status_clear(&st);
271
272 fn = cmd_lookup(cmd);
273 if (fn == NULL) {
274 usage(argv[0]);
275 return 50;
276 }
277 return fn(argc, argv, &st);
