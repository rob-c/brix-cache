18static int
19parse_valid(const char *s)
20{
21 /* "HH:MM" → hours (rounded up on minutes); a bare integer → hours. */
22 const char *colon = strchr(s, ':');
23 if (colon != NULL) {
24 int h = atoi(s);
25 int m = atoi(colon + 1);
26 int hours = h + (m > 0 ? 1 : 0);
27 return hours > 0 ? hours : 12;
28 }
29 return atoi(s) > 0 ? atoi(s) : 12;
30}

51main(int argc, char **argv)
52{
53 brix_status st;
54 const char *cmd;
55 int i, rc;
56
57 if (argc < 2) { usage(); return 50; }
58 cmd = argv[1];
59
60 /* --help / --version as the first argument. */
61 if (strcmp(cmd, "--version") == 0) {
62 printf("xrdgsiproxy (BriX-Cache client) %s\n", brix_client_version());
63 return 0;
64 }
65 if (strcmp(cmd, "--help") == 0 || strcmp(cmd, "-h") == 0) {
66 usage_fp(stdout);
67 return 0;
68 }
69 brix_crypto_init();
70 brix_status_clear(&st);
71
72 if (strcmp(cmd, "init") == 0) {
73 brix_proxy_opts o;
74 memset(&o, 0, sizeof(o));
75 for (i = 2; i < argc; i++) {
76 if ((strcmp(argv[i], "-valid") == 0 || strcmp(argv[i], "--valid") == 0)
77 && i + 1 < argc) { o.valid_hours = parse_valid(argv[++i]); }
78 else if ((strcmp(argv[i], "-cert") == 0 || strcmp(argv[i], "--cert") == 0)
79 && i + 1 < argc) { o.user_cert = argv[++i]; }
80 else if ((strcmp(argv[i], "-key") == 0 || strcmp(argv[i], "--key") == 0)
81 && i + 1 < argc) { o.user_key = argv[++i]; }
82 else if ((strcmp(argv[i], "-out") == 0 || strcmp(argv[i], "--out") == 0)
83 && i + 1 < argc) { o.out_path = argv[++i]; }
84 else if ((strcmp(argv[i], "-bits") == 0 || strcmp(argv[i], "--bits") == 0)
85 && i + 1 < argc) { o.bits = atoi(argv[++i]); }
86 else { usage(); return 50; }
87 }
88 rc = brix_proxy_create(&o, &st);
89 if (rc != 0) {
90 fprintf(stderr, "xrdgsiproxy: init: %s\n", st.msg);
91 return brix_shellcode(&st);
92 }
93 return brix_proxy_info(o.out_path, stdout, &st); /* echo what we made */
94 }
95
96 if (strcmp(cmd, "info") == 0 || strcmp(cmd, "destroy") == 0) {
97 const char *file = NULL;
98 int is_info = (strcmp(cmd, "info") == 0);
99 for (i = 2; i < argc; i++) {
100 if ((strcmp(argv[i], "-file") == 0 || strcmp(argv[i], "--file") == 0)
101 && i + 1 < argc) { file = argv[++i]; }
102 else { usage(); return 50; }
103 }
104 rc = is_info ? brix_proxy_info(file, stdout, &st)
105 : brix_proxy_destroy(file, &st);
106 if (rc == 0) {
107 return 0;
108 }
109 /* `info` is tolerant of an absent proxy, mirroring stock xrdgsiproxy:
110 * emit a plain "not found" notice and exit 1 rather than escalating a
111 * missing file to a hard usage error. Any other failure (e.g. a present
112 * but unparseable proxy) still surfaces via the normal error path. */
113 if (is_info && st.kxr == XRDC_ENOENT) {
114 printf("%s\n", st.msg);
115 return 1;
116 }
117 fprintf(stderr, "xrdgsiproxy: %s: %s\n", cmd, st.msg);
118 return brix_shellcode(&st);
119 }
120
121 usage();
122 return 50;
123}
