17/* bootstrap response handling */
18/*
19 * WHAT: Advance the upstream login handshake one step, consuming the response
20 * that the read handler just finished accumulating (proxy->resp_status /
21 * resp_dlen / resp_body) and, where the next step requires sending,
22 * arming the write side with the next request frame.
23 * WHY: Connecting to an upstream XRootD server is a fixed conversation —
24 * hello -> kXR_protocol -> kXR_login -> (optional kXR_auth) — that must
25 * complete before any client request can be forwarded. Each leg is a
26 * separate wire round-trip, so the proxy drives it as a state machine
27 * (proxy->bs_phase) re-entered once per upstream response rather than
28 * blocking; the event loop never waits.
29 * HOW: Switch on bs_phase. Each arm validates the just-received response,
30 * optionally builds+flushes the next request, and either advances
31 * bs_phase or returns early (when it issued a send and must wait for the
32 * reply). The single fallthrough at the bottom frees the response
33 * accumulator; reaching XRD_PX_BS_DONE flips proxy->state to IDLE and
34 * releases any request queued during bootstrap.
35 *
36 * RE-ENTRY: this is called by brix_proxy_read_handler each time a complete
37 * upstream frame arrives. Arms that send (AUTH legs) return early and are
38 * resumed by the next read event; non-sending arms fall through to the
39 * shared reset/finish tail. The auth-send arms (FORWARD bearer, SSS,
40 * FORWARD token-file fallback, login-sec hint) are structurally identical
41 * — build frame -> free+reset resp accumulator -> set wbuf -> flush ->
42 * arm write if partial -> return — and share proxy_bs_queue_auth_frame;
43 * they differ only in how the credential is sourced.
44 */
45/* BS_LOGIN bootstrap phase: handle the upstream login reply — kXR_authmore
46 * (forward bearer / SSS / token-file), login failure, and the token-only
47 * login-sec hint (proactive P=sss / P=ztn). Returns 1 when it has issued a
48 * frame or aborted and the caller must return; 0 when the phase is complete
49 * (bs_phase advanced to DONE) and the caller falls through to the shared
50 * reset/finish tail. */
51static int
52brix_proxy_bs_login(brix_proxy_ctx_t *proxy)
53{
54 if (proxy->resp_status == kXR_authmore) {
55 /* Upstream requests authentication — handle by configured policy.
56 * Effective policy: per-upstream override if set, else global
57 * conf->proxy.auth. */
58 return proxy_bs_do_auth(proxy);
59 }
60
61 if (proxy->resp_status != kXR_ok) {
62 proxy_bs_auth_error(proxy, 1, "upstream login failed");
63 return 1;
64 }
65
66 if (proxy_bs_login_sec_hint(proxy)) {
67 return 1;
68 }
69
70 proxy->bs_phase = XRD_PX_BS_DONE;
71
72 return 0;
73}

75/*
76 * WHAT: Validate the upstream kXR_protocol reply (BS_PROTOCOL phase).
77 * WHY: kXR_protocol reply body: [4 bytes pval][4 bytes flags][...]. The
78 * server's capability flags live at body offset 4; kXR_gotoTLS there
79 * means the server wants the connection upgraded to TLS now. We only
80 * reach this code on a cleartext connection (TLS-from-start uses a
81 * different path), so an in-band upgrade request is unsupported.
82 * HOW: Returns NGX_OK when the reply is acceptable; NGX_ERROR after
83 * aborting the proxy (bad status or gotoTLS demand).
84 */
85static ngx_int_t
86proxy_bs_check_protocol_reply(brix_proxy_ctx_t *proxy)
87{
88 if (proxy->resp_status != kXR_ok) {
89 brix_proxy_abort(proxy, "upstream kXR_protocol failed");
90 return NGX_ERROR;
91 }
92
93 if (proxy->resp_dlen >= 8) {
94 uint32_t flags_be;
95 ngx_memcpy(&flags_be, proxy->resp_body + 4, sizeof(flags_be));
96 if (ntohl(flags_be) & kXR_gotoTLS) {
97#if (NGX_SSL)
98 /* gotoTLS from upstream is only valid when proxy_upstream_tls
99 * was NOT set (it's the upstream's choice to require TLS after
100 * the connection was made without TLS). Not supported yet. */
101#endif
102 brix_proxy_abort(proxy,
103 "upstream requires TLS upgrade after connect "
104 "(use brix_proxy_upstream_tls on to start with TLS)");
105 return NGX_ERROR;
106 }
107 }
108
109 return NGX_OK;
110}

112/*
113 * WHAT: Adopt the bootstrapped upstream connection into service — flip the
114 * proxy to IDLE, publish success metrics, restore failure budgets, and
115 * hand the fd over to whichever request is waiting.
116 * WHY: This is the single point where the upstream stops being "in
117 * bootstrap" and becomes usable by the client relay; keeping the
118 * transition in one helper freezes the handoff timing.
119 * HOW: Marks the upstream healthy, clears the consecutive-failure budget,
120 * refills the reconnect budget, then either forwards the request that
121 * was queued during bootstrap (dispatch_pending handles bound-secondary
122 * lazy-open) or resumes the client read loop.
123 */
124static void
125proxy_bs_adopt_conn(brix_proxy_ctx_t *proxy)
126{
127 proxy->state = XRD_PX_IDLE;
128 brix_proxy_up_mark_ok(proxy);
129 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connects_total);
130 BRIX_PROXY_UP_INC(proxy, upstream_connects_total);
131
132 /* The upstream accepted the forwarded credential — this connection is
133 * healthy again, so clear the consecutive-failure budget. */
134 if (proxy->client_ctx != NULL) {
135 proxy->client_ctx->proxy_fail_count = 0;
136 }
137
138 /* Restore the full reconnect budget on every successful bootstrap */
139 if (proxy->conf != NULL) {
140 proxy->reconnect_left = (int) proxy->conf->proxy.reconnect_attempts;
141 }
142
143 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, proxy->client_conn->log, 0,
144 "xrootd proxy: upstream bootstrap done");
145
146 /* If a request was queued during bootstrap, forward it now.
147 * brix_proxy_dispatch_pending handles bound-secondary lazy-open. */
148 if (proxy->saved_req != NULL) {
149 brix_proxy_dispatch_pending(proxy);
150 return;
151 }
152
153 /* No deferred request — just resume the client read loop */
154 {
155 brix_ctx_t *ctx = proxy->client_ctx;
156 ctx->state = XRD_ST_REQ_HEADER;
157 brix_schedule_read_resume(proxy->client_conn);
158 }
159}

161void
162brix_proxy_handle_bootstrap(brix_proxy_ctx_t *proxy)
163{
164 /* State machine over the upstream login conversation; see bs_phase enum. */
165 switch (proxy->bs_phase) {
166
167 case XRD_PX_BS_HANDSHAKE:
168 /* Server hello: the 16-byte handshake reply begins with two zero
169 * 32-bit words, and the read handler parsed status from the bytes that
170 * happen to fall in the zero prefix, so resp_status is normally 0 ==
171 * kXR_ok. A non-zero value here means the bytes did not line up as a
172 * valid hello, i.e. this is not an XRootD server speaking. */
173 if (proxy->resp_status != kXR_ok) {
174 brix_proxy_abort(proxy, "bad handshake from upstream");
175 return;
176 }
177 proxy->bs_phase = XRD_PX_BS_PROTOCOL;
178 break;
179
180 case XRD_PX_BS_PROTOCOL:
181 if (proxy_bs_check_protocol_reply(proxy) != NGX_OK) {
182 return;
183 }
184 proxy->bs_phase = XRD_PX_BS_LOGIN;
185 break;
186
187 case XRD_PX_BS_LOGIN:
188 if (brix_proxy_bs_login(proxy)) {
189 return;
190 }
191 break;
192
193 case XRD_PX_BS_AUTH:
194 if (proxy->resp_status != kXR_ok) {
195 proxy_bs_auth_error(proxy, 1, "upstream rejected forwarded token");
196 return;
197 }
198 proxy->bs_phase = XRD_PX_BS_DONE;
199 break;
200
201 default:
202 brix_proxy_abort(proxy, "proxy: invalid bootstrap phase");
203 return;
204 }
205
206 /* Shared tail: reached only by arms that fell through (i.e. did NOT issue a
207 * send and return early). Reset the accumulator for whatever comes next. */
208 /* Reset response accumulator for the next bootstrap message or first req */
209 proxy_bs_reset_resp(proxy);
210
211 if (proxy->bs_phase != XRD_PX_BS_DONE) {
212 return; /* caller (read_handler) will continue the loop */
213 }
214
215 /* Bootstrap complete — transition to IDLE */
216 proxy_bs_adopt_conn(proxy);
217}
