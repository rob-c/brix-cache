17/* auth-frame builder */
18/*
19 * Builds a kXR_auth frame carrying a ztn (bearer token) credential.
20 *
21 * Wire format expected by the nginx-xrootd server (gsi/auth.c):
22 * Header bytes 16-19 (= cur_body[12:15]): "ztn\0" ← credential type
23 * Header bytes 20-23 (dlen): 4 + token_len
24 * Payload bytes 0-3: "ztn\0" ← payload prefix
25 * Payload bytes 4..: JWT token
26 *
27 * Returns the total frame length (24-byte header + 4 + token_len).
28 * buf must be at least 24 + 4 + token_len bytes.
29 */
30static size_t
31proxy_build_auth_ztn(u_char *buf, const char *token, size_t token_len)
32{
33 uint16_t rid = htons(kXR_auth);
34 uint32_t dlen_be = htonl((uint32_t)(4 + token_len));
35
36 ngx_memzero(buf, 24);
37 ngx_memcpy(buf + 2, &rid, 2);
38 /* credtype "ztn\0" at header body offset 12 (= buf offset 16) */
39 buf[16] = 'z'; buf[17] = 't'; buf[18] = 'n'; buf[19] = '\0';
40 ngx_memcpy(buf + 20, &dlen_be, 4);
41 /* payload: "ztn\0" prefix followed by JWT token */
42 buf[24] = 'z'; buf[25] = 't'; buf[26] = 'n'; buf[27] = '\0';
43 ngx_memcpy(buf + 28, token, token_len);
44
45 return 28 + token_len;
46}

48/* upstream auth resolution * WHAT: Resolve the effective upstream auth policy and (for SSS) the key to use
49 * for THIS upstream: a per-upstream `auth`/`sss:<keyname>` override wins,
50 * else the global brix_proxy_auth + first configured key.
51 * WHY: Both the kXR_authmore branch and the kXR_ok login-sec-hint branch need
52 * the same decision; factoring it keeps them in lock-step.
53 * HOW: Writes *eff_auth (BRIX_PROXY_AUTH_*) and *eff_key (NULL unless an SSS
54 * key was selected). */
55static void
56proxy_resolve_upstream_auth(brix_proxy_ctx_t *proxy, ngx_uint_t *eff_auth,
57 const brix_sss_key_t **eff_key)
58{
59 ngx_stream_brix_srv_conf_t *conf = proxy->conf;
60
61 *eff_auth = conf ? conf->proxy.auth : BRIX_PROXY_AUTH_ANONYMOUS;
62 *eff_key = NULL;
63
64 if (proxy->upstream_idx >= 0
65 && conf != NULL
66 && conf->proxy.upstreams != NULL
67 && proxy->upstream_idx < (int) conf->proxy.upstreams->nelts)
68 {
69 brix_proxy_upstream_t *u =
70 (brix_proxy_upstream_t *) conf->proxy.upstreams->elts
71 + proxy->upstream_idx;
72
73 if (u->auth >= 0) {
74 *eff_auth = (ngx_uint_t) u->auth;
75 }
76 if (*eff_auth == BRIX_PROXY_AUTH_SSS
77 && conf->sss_keys != NULL
78 && conf->sss_keys->nelts > 0)
79 {
80 brix_sss_key_t *keys = conf->sss_keys->elts;
81 ngx_uint_t ki;
82 if (u->sss_keyname[0] != '\0') {
83 for (ki = 0; ki < conf->sss_keys->nelts; ki++) {
84 if (ngx_strcmp(keys[ki].name, u->sss_keyname) == 0) {
85 *eff_key = &keys[ki];
86 break;
87 }
88 }
89 }
90 if (*eff_key == NULL) {
91 *eff_key = keys; /* fall back to first key */
92 }
93 }
94 }
95}

97/* send an SSS kXR_auth credential to the upstream * WHAT: Build an SSS credential from `key`, wrap it in a kXR_auth request, and
98 * arm the write side; advance bs_phase to BS_AUTH.
99 * WHY: Used by BOTH the kXR_authmore path and the kXR_ok login-sec-hint path
100 * (our own server advertises SSS via the latter), so the logic lives once.
101 * HOW: Returns NGX_OK once the frame is queued (caller must return), or
102 * NGX_ERROR after aborting the proxy on any failure. */
103static ngx_int_t
104proxy_send_sss_auth(brix_proxy_ctx_t *proxy, const brix_sss_key_t *key)
105{
106 u_char cred[512];
107 size_t cred_len;
108 u_char *frame;
109 size_t frame_len;
110
111 if (brix_sss_build_proxy_credential(key, key->user,
112 cred, sizeof(cred), &cred_len) != NGX_OK)
113 {
114 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_auth_errors);
115 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
116 brix_proxy_abort(proxy, "proxy: SSS credential build failed");
117 return NGX_ERROR;
118 }
119
120 frame_len = XRD_REQUEST_HDR_LEN + cred_len;
121 frame = ngx_palloc(proxy->conn->pool, frame_len);
122 if (frame == NULL) {
123 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_auth_errors);
124 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
125 brix_proxy_abort(proxy, "proxy: OOM for SSS frame");
126 return NGX_ERROR;
127 }

129 /* kXR_auth request header: reqid at [2:3], credtype "sss\0" at [16:19]
130 * (the server routes on this field — leaving it zero yields "unknown
131 * credtype"), dlen at [20:23], SSS credential body after the 24-byte hdr. */
132 ngx_memzero(frame, XRD_REQUEST_HDR_LEN);
133 frame[2] = (kXR_auth >> 8) & 0xFF;
134 frame[3] = kXR_auth & 0xFF;
135 frame[16] = 's'; frame[17] = 's'; frame[18] = 's'; frame[19] = '\0';
136 {
137 uint32_t dlen_be = htonl((uint32_t) cred_len);
138 ngx_memcpy(frame + 20, &dlen_be, 4);
139 }
140 ngx_memcpy(frame + XRD_REQUEST_HDR_LEN, cred, cred_len);
141
142 if (proxy->resp_body != NULL) {
143 ngx_free(proxy->resp_body);
144 proxy->resp_body = NULL;
145 }
146 proxy->rhdr_pos = 0;
147 proxy->resp_dlen = 0;
148 proxy->resp_body_pos = 0;
149
150 proxy->wbuf = frame;
151 proxy->wbuf_len = frame_len;
152 proxy->wbuf_pos = 0;
153 proxy->bs_phase = XRD_PX_BS_AUTH;
154
155 if (brix_proxy_flush(proxy) == NGX_ERROR) {
156 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_auth_errors);
157 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
158 brix_proxy_abort(proxy, "proxy: SSS frame send failed");
159 return NGX_ERROR;
160 }
161 if (proxy->wbuf_pos < proxy->wbuf_len) {
162 if (ngx_handle_write_event(proxy->conn->write, 0) != NGX_OK) {
163 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_auth_errors);
164 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
165 brix_proxy_abort(proxy, "proxy: write arm for SSS failed");
166 return NGX_ERROR;
167 }
168 }
169 return NGX_OK;
170}

172/* bootstrap response handling */
173/*
174 * WHAT: Advance the upstream login handshake one step, consuming the response
175 * that the read handler just finished accumulating (proxy->resp_status /
176 * resp_dlen / resp_body) and, where the next step requires sending,
177 * arming the write side with the next request frame.
178 * WHY: Connecting to an upstream XRootD server is a fixed conversation —
179 * hello -> kXR_protocol -> kXR_login -> (optional kXR_auth) — that must
180 * complete before any client request can be forwarded. Each leg is a
181 * separate wire round-trip, so the proxy drives it as a state machine
182 * (proxy->bs_phase) re-entered once per upstream response rather than
183 * blocking; the event loop never waits.
184 * HOW: Switch on bs_phase. Each arm validates the just-received response,
185 * optionally builds+flushes the next request, and either advances
186 * bs_phase or returns early (when it issued a send and must wait for the
187 * reply). The single fallthrough at the bottom frees the response
188 * accumulator; reaching XRD_PX_BS_DONE flips proxy->state to IDLE and
189 * releases any request queued during bootstrap.
190 *
191 * RE-ENTRY: this is called by brix_proxy_read_handler each time a complete
192 * upstream frame arrives. Arms that send (AUTH legs) return early and are
193 * resumed by the next read event; non-sending arms fall through to the
194 * shared reset/finish tail. The four auth-send arms (FORWARD bearer, SSS,
195 * FORWARD token-file fallback, login-sec hint) are structurally identical:
196 * build frame -> free+reset resp accumulator -> set wbuf -> flush -> arm
197 * write if partial -> return. They differ only in how the credential is
198 * sourced.
199 */
200/* BS_LOGIN bootstrap phase: handle the upstream login reply — kXR_authmore
201 * (forward bearer / SSS / token-file), login failure, and the token-only
202 * login-sec hint (proactive P=sss / P=ztn). Returns 1 when it has issued a
203 * frame or aborted and the caller must return; 0 when the phase is complete
204 * (bs_phase advanced to DONE) and the caller falls through to the shared
205 * reset/finish tail. */
206static int
207brix_proxy_bs_login(brix_proxy_ctx_t *proxy)
208{
209 if (proxy->resp_status == kXR_authmore) {
210 /* Upstream requests authentication — handle by configured policy.
211 * Effective policy: per-upstream override if set, else global conf->proxy.auth. */
212 ngx_stream_brix_srv_conf_t *conf = proxy->conf;
213 ngx_uint_t eff_auth;
214 const brix_sss_key_t *eff_key;
215
216 proxy_resolve_upstream_auth(proxy, &eff_auth, &eff_key);
217
218 /* auth-send arm 1 of 4: FORWARD using the client's bearer * Preferred FORWARD path: the connecting client already presented a
219 * WLCG bearer token, so re-present it verbatim to the upstream. */
220 if (eff_auth == BRIX_PROXY_AUTH_FORWARD
221 && proxy->client_ctx != NULL
222 && proxy->client_ctx->bearer_token[0] != '\0')
223 {
224 /* Forward the client's WLCG bearer token as a ztn credential */
225 const char *token = proxy->client_ctx->bearer_token;
226 size_t token_len = ngx_strlen(token);
227 size_t frame_len = 28 + token_len; /* hdr24 + len4 + jwt */
228 u_char *frame;
229
230 frame = ngx_palloc(proxy->conn->pool, frame_len);
231 if (frame == NULL) {
232 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
233 upstream_auth_errors);
234 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
235 brix_proxy_abort(proxy, "proxy: OOM building auth frame");
236 return 1;
237 }
238
239 proxy_build_auth_ztn(frame, token, token_len);
240
241 /* Clear the just-consumed response so the accumulator is ready
242 * for the kXR_auth reply, then install the auth frame as the
243 * write buffer and advance to BS_AUTH before flushing. */
244 if (proxy->resp_body != NULL) {
245 ngx_free(proxy->resp_body);
246 proxy->resp_body = NULL;
247 }
248 proxy->rhdr_pos = 0;
249 proxy->resp_dlen = 0;
250 proxy->resp_body_pos = 0;
251
252 proxy->wbuf = frame;
253 proxy->wbuf_len = frame_len;
254 proxy->wbuf_pos = 0;
255 proxy->bs_phase = XRD_PX_BS_AUTH;
256
257 /* flush() may only partially write under backpressure; if so,
258 * arm the write event so the rest goes out later. Either way we
259 * return and wait for the BS_AUTH reply on the next read event. */
260 if (brix_proxy_flush(proxy) == NGX_ERROR) {
261 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
262 upstream_auth_errors);
263 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
264 brix_proxy_abort(proxy, "proxy: auth frame send failed");
265 return 1;
266 }
267 if (proxy->wbuf_pos < proxy->wbuf_len) {
268 if (ngx_handle_write_event(proxy->conn->write, 0) != NGX_OK) {
269 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
270 upstream_auth_errors);
271 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
272 brix_proxy_abort(proxy,
273 "proxy: write arm for auth failed");
274 }
275 }
276 return 1;
277 }
278
279 /* auth-send arm 2 of 4: SSS shared-secret credential */ if (eff_auth == BRIX_PROXY_AUTH_SSS && eff_key != NULL) {
280 (void) proxy_send_sss_auth(proxy, eff_key);
281 return 1; /* frame queued, or proxy aborted on failure */
282 }

284 /* auth-send arm 3 of 4: FORWARD token-file fallback * FORWARD fallback — client has no bearer token but
285 * brix_upstream_token_file is configured (credential bridge).
286 * ftok is function-static (64 KiB) to keep it off the stack; safe
287 * because nginx workers are single-threaded on the event loop and
288 * the token is copied into the heap frame before any yield. */
289 if (eff_auth == BRIX_PROXY_AUTH_FORWARD
290 && conf != NULL
291 && conf->upstream_token_file.len > 0)
292 {
293 static u_char ftok[PROXY_UPSTREAM_BEARER_MAX];
294 size_t flen = 0;
295 size_t fframe_len;
296 u_char *fframe;
297

299 ftok, sizeof(ftok), &flen,
300 proxy->client_conn->log,
301 "proxy authmore-file token") != NGX_OK
302 || flen == 0)
303 {
304 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
305 upstream_auth_errors);
306 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
307 brix_proxy_abort(proxy,
308 "upstream requires auth; set brix_upstream_token_file");
309 return 1;
310 }
311 fframe_len = 28 + flen;
312 fframe = ngx_palloc(proxy->conn->pool, fframe_len);
313 if (fframe == NULL) {
314 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
315 upstream_auth_errors);
316 brix_proxy_abort(proxy, "proxy: OOM for file token frame");
317 return 1;
318 }
319 proxy_build_auth_ztn(fframe, (const char *) ftok, flen);
320 if (proxy->resp_body != NULL) {
321 ngx_free(proxy->resp_body);
322 proxy->resp_body = NULL;
323 }
324 proxy->rhdr_pos = 0;
325 proxy->resp_dlen = 0;
326 proxy->resp_body_pos = 0;
327 proxy->wbuf = fframe;
328 proxy->wbuf_len = fframe_len;
329 proxy->wbuf_pos = 0;
330 proxy->bs_phase = XRD_PX_BS_AUTH;
331 if (brix_proxy_flush(proxy) == NGX_ERROR) {
332 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
333 upstream_auth_errors);
334 brix_proxy_abort(proxy,
335 "proxy: file token frame send failed");
336 return 1;
337 }
338 if (proxy->wbuf_pos < proxy->wbuf_len) {
339 if (ngx_handle_write_event(proxy->conn->write, 0) != NGX_OK) {
340 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
341 upstream_auth_errors);
342 brix_proxy_abort(proxy,
343 "proxy: write arm for file token failed");
344 }
345 }
346 return 1;
347 }

349 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_auth_errors);
350 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
351 brix_proxy_abort(proxy,
352 "upstream requires authentication "
353 "(set brix_proxy_auth forward or sss)");
354 return 1;
355 }
356 if (proxy->resp_status != kXR_ok) {
357 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_auth_errors);
358 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
359 brix_proxy_abort(proxy, "upstream login failed");
360 return 1;

363 /*
364 * Token-only servers embed a security challenge in the login response:
365 * [sessid:16][&P=ztn,v:10000]
366 * The client must proactively send kXR_auth (no explicit kXR_authmore).
367 * Detect the hint and inject the token from file or client bearer.
368 */
369 /* The security hint string follows the fixed 16-byte session id in the
370 * login body, so step past it before scanning. resp_body was NUL-padded
371 * by the read handler (alloc dlen+1), making strstr() safe here. */
372 if (proxy->resp_dlen > BRIX_SESSION_ID_LEN
373 && proxy->resp_body != NULL)
374 {
375 const char *parms =
376 (const char *)(proxy->resp_body + BRIX_SESSION_ID_LEN);
377
378 /* auth-send arm 5 of 5: proactive SSS on login-sec hint * The nginx-xrootd server (and stock xrootd) advertise SSS via the
379 * kXR_ok login response sec hint ("&P=sss,..."), NOT via kXR_authmore.
380 * If this upstream's effective policy is SSS, build and send the SSS
381 * credential unprompted — without this, SSS upstream auth silently
382 * never happens and the forwarded open is rejected NotAuthorized. */
383 if (strstr(parms, "P=sss") != NULL) {
384 ngx_uint_t sss_auth;
385 const brix_sss_key_t *sss_key;
386
387 proxy_resolve_upstream_auth(proxy, &sss_auth, &sss_key);
388 if (sss_auth == BRIX_PROXY_AUTH_SSS && sss_key != NULL) {
389 (void) proxy_send_sss_auth(proxy, sss_key);
390 return 1; /* frame queued, or proxy aborted on failure */
391 }
392 }
393
394 /* auth-send arm 4 of 5: proactive ztn on login-sec hint * Server advertised "P=ztn" but did not send kXR_authmore, so we
395 * must send kXR_auth unprompted. Source the token from the client
396 * bearer (FORWARD) first, else from the upstream token file. */
397 if (strstr(parms, "P=ztn") != NULL) {
398 ngx_stream_brix_srv_conf_t *lconf = proxy->conf;
399 const char *ltoken = NULL;
400 size_t ltoken_len = 0;
401 static u_char lftok[PROXY_UPSTREAM_BEARER_MAX];
402
403 /* Prefer client's bearer token (FORWARD mode) */
404 if (proxy->client_ctx != NULL
405 && proxy->client_ctx->bearer_token[0] != '\0')
406 {
407 ltoken = proxy->client_ctx->bearer_token;
408 ltoken_len = ngx_strlen(ltoken);
409 }
410 /* Fallback: read from upstream_token_file */
411 else if (lconf != NULL
412 && lconf->upstream_token_file.len > 0)
413 {
414 size_t lflen = 0;
415 if (brix_token_read_file(&lconf->upstream_token_file,
416 lftok, sizeof(lftok), &lflen,
417 proxy->client_conn->log,
418 "proxy login-sec ztn") == NGX_OK
419 && lflen > 0)
420 {
421 ltoken = (const char *) lftok;
422 ltoken_len = lflen;
423 }
424 }
425
426 if (ltoken != NULL && ltoken_len > 0) {
427 size_t lframe_len = 28 + ltoken_len;
428 u_char *lframe = ngx_palloc(proxy->conn->pool, lframe_len);
429 if (lframe == NULL) {
430 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
431 upstream_auth_errors);
432 brix_proxy_abort(proxy,
433 "proxy: OOM for login-sec token frame");
434 return 1;
435 }
436 proxy_build_auth_ztn(lframe, ltoken, ltoken_len);
437 if (proxy->resp_body != NULL) {
438 ngx_free(proxy->resp_body);
439 proxy->resp_body = NULL;
440 }
441 proxy->rhdr_pos = 0;
442 proxy->resp_dlen = 0;
443 proxy->resp_body_pos = 0;
444 proxy->wbuf = lframe;
445 proxy->wbuf_len = lframe_len;
446 proxy->wbuf_pos = 0;
447 proxy->bs_phase = XRD_PX_BS_AUTH;
448 if (brix_proxy_flush(proxy) == NGX_ERROR) {
449 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
450 upstream_auth_errors);
451 brix_proxy_abort(proxy,
452 "proxy: login-sec token frame send failed");
453 return 1;
454 }
455 if (proxy->wbuf_pos < proxy->wbuf_len) {
456 if (ngx_handle_write_event(proxy->conn->write, 0)
457 != NGX_OK)
458 {
459 BRIX_PROXY_METRIC_INC(proxy->client_ctx,
460 upstream_auth_errors);
461 brix_proxy_abort(proxy,
462 "proxy: write arm for login-sec failed");
463 }
464 }
465 return 1;
466 }
467 }
468 }
469
470 proxy->bs_phase = XRD_PX_BS_DONE;
471
472 return 0;
473}

475void
476brix_proxy_handle_bootstrap(brix_proxy_ctx_t *proxy)
477{
478 /* State machine over the upstream login conversation; see bs_phase enum. */
479 switch (proxy->bs_phase) {
480
481 case XRD_PX_BS_HANDSHAKE:
482 /* Server hello: the 16-byte handshake reply begins with two zero
483 * 32-bit words, and the read handler parsed status from the bytes that
484 * happen to fall in the zero prefix, so resp_status is normally 0 ==
485 * kXR_ok. A non-zero value here means the bytes did not line up as a
486 * valid hello, i.e. this is not an XRootD server speaking. */
487 if (proxy->resp_status != kXR_ok) {
488 brix_proxy_abort(proxy, "bad handshake from upstream");
489 return;
490 }
491 proxy->bs_phase = XRD_PX_BS_PROTOCOL;
492 break;
493
494 case XRD_PX_BS_PROTOCOL:
495 if (proxy->resp_status != kXR_ok) {
496 brix_proxy_abort(proxy, "upstream kXR_protocol failed");
497 return;
498 }
499 /* kXR_protocol reply body: [4 bytes pval][4 bytes flags][...]. The
500 * server's capability flags live at body offset 4; kXR_gotoTLS there
501 * means the server wants the connection upgraded to TLS now. We only
502 * reach this code on a cleartext connection (TLS-from-start uses a
503 * different path), so an in-band upgrade request is unsupported. */
504 if (proxy->resp_dlen >= 8) {
505 uint32_t flags_be;
506 ngx_memcpy(&flags_be, proxy->resp_body + 4, sizeof(flags_be));
507 if (ntohl(flags_be) & kXR_gotoTLS) {
508#if (NGX_SSL)
509 /* gotoTLS from upstream is only valid when proxy_upstream_tls
510 * was NOT set (it's the upstream's choice to require TLS after
511 * the connection was made without TLS). Not supported yet. */
512#endif
513 brix_proxy_abort(proxy,
514 "upstream requires TLS upgrade after connect "
515 "(use brix_proxy_upstream_tls on to start with TLS)");
516 return;
517 }
518 }
519 proxy->bs_phase = XRD_PX_BS_LOGIN;
520 break;
521
522 case XRD_PX_BS_LOGIN:
523 if (brix_proxy_bs_login(proxy)) {
524 return;
525 }
526 break;
527
528 case XRD_PX_BS_AUTH:
529 if (proxy->resp_status != kXR_ok) {
530 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_auth_errors);
531 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
532 brix_proxy_abort(proxy, "upstream rejected forwarded token");
533 return;
534 }
535 proxy->bs_phase = XRD_PX_BS_DONE;
536 break;
537
538 default:
539 brix_proxy_abort(proxy, "proxy: invalid bootstrap phase");
540 return;
541 }
542
543 /* Shared tail: reached only by arms that fell through (i.e. did NOT issue a
544 * send and return early). Reset the accumulator for whatever comes next. */
545 /* Reset response accumulator for the next bootstrap message or first req */
546 if (proxy->resp_body != NULL) {
547 ngx_free(proxy->resp_body);
548 proxy->resp_body = NULL;
549 }
550 proxy->rhdr_pos = 0;
551 proxy->resp_dlen = 0;
552 proxy->resp_body_pos = 0;
553
554 if (proxy->bs_phase != XRD_PX_BS_DONE) {
555 return; /* caller (read_handler) will continue the loop */
556 }
557
558 /* Bootstrap complete — transition to IDLE */
559 proxy->state = XRD_PX_IDLE;
560 brix_proxy_up_mark_ok(proxy);
561 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connects_total);
562 BRIX_PROXY_UP_INC(proxy, upstream_connects_total);
563
564 /* The upstream accepted the forwarded credential — this connection is
565 * healthy again, so clear the consecutive-failure budget. */
566 if (proxy->client_ctx != NULL) {
567 proxy->client_ctx->proxy_fail_count = 0;
568 }
569
570 /* Restore the full reconnect budget on every successful bootstrap */
