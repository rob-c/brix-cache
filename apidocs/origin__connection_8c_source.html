17/* brix_cache_origin_close — free everything an origin connection holds (SSL,
18 * SSL_CTX, socket fd) and NULL the pointers; symmetric with origin_connect and
19 * called on every error path. Order matters: SSL shutdown before free, SSL_CTX
20 * before the fd close, fd last, so the SSL layer never dangles. */
21
22void
23brix_cache_origin_close(brix_cache_origin_conn_t *oc)
24{
25 if (oc->ssl != NULL) {
26 SSL_shutdown(oc->ssl);
27 SSL_free(oc->ssl);
28 oc->ssl = NULL;
29 }
30
31 if (oc->ssl_ctx != NULL) {
32 SSL_CTX_free(oc->ssl_ctx);
33 oc->ssl_ctx = NULL;
34 }
35
36 if (oc->fd >= 0) {
37 close(oc->fd);
38 oc->fd = -1;
39 }
40}

42/* origin_tls_load_verify — load the origin's CA trust anchor into ssl_ctx,
43 * choosing CAfile vs CApath by stat(). WHAT: brix_trusted_ca (synth->trusted_ca)
44 * may be a single PEM bundle OR a hashed CA directory (grid-security style); the
45 * two OpenSSL slots are not interchangeable, so a directory passed as CAfile
46 * silently loads nothing and the peer verify then falls back to the system store.
47 * WHY: the origin CA is operator-configured (e->origin_ca_dir → synth->trusted_ca)
48 * and is usually a directory. HOW: stat → S_ISDIR ? CApath : CAfile; empty trust
49 * anchor → system default paths. Returns 0 on a usable store, -1 on load failure. */
50static int
51origin_tls_load_verify(SSL_CTX *ctx, const ngx_str_t *ca)
52{
53 if (ca == NULL || ca->len == 0 || ca->data == NULL) {
54 (void) SSL_CTX_set_default_verify_paths(ctx);
55 return 0;
56 }
57 {
58 char path[PATH_MAX];
59 size_t n = ca->len < sizeof(path) - 1 ? ca->len : sizeof(path) - 1;
60 struct stat st;
61 int is_dir;
62 int ok;
63
64 ngx_memcpy(path, ca->data, n);
65 path[n] = '\0';
66
67 is_dir = (stat(path, &st) == 0 && S_ISDIR(st.st_mode));
68 ok = is_dir
69 ? SSL_CTX_load_verify_locations(ctx, NULL, path)
70 : SSL_CTX_load_verify_locations(ctx, path, NULL);
71 return ok == 1 ? 0 : -1;
72 }
73}

75/* brix_cache_origin_tls_upgrade — perform a blocking client TLS handshake over
76 * the ALREADY-CONNECTED origin fd (oc->fd), storing the negotiated SSL on oc->ssl
77 * so every subsequent frame rides TLS via the io.c SSL_read/SSL_write path.
78 *
79 * WHAT: build a per-connection SSL_CTX (peer verify against synth->trusted_ca,
80 * CAfile-or-CApath), bind the expected hostname + SNI, and SSL_connect.
81 * WHY: a stock XRootD `roots://` origin negotiates kXR_protocol over CLEARTEXT
82 * and then advertises kXR_gotoTLS; the client must upgrade THIS fd in place
83 * before kXR_login/auth. The same helper also serves any TLS-from-connect
84 * caller (call it right after connect). Runs on a blocking fill worker, so
85 * it uses a synchronous SSL_connect rather than nginx's async handshake.
86 * HOW: SSL_CTX_new(TLS_client) → SSL_VERIFY_PEER → origin_tls_load_verify →
87 * SSL_set1_host + SNI (MITM: chain-only would accept any CA-valid cert) →
88 * SSL_set_fd(oc->fd) → SSL_connect. On failure t's error is set; any partly
89 * built oc->ssl/oc->ssl_ctx stay attached to oc for the caller's
90 * brix_cache_origin_close to release (same contract as connect_addr).
91 * Returns 0 with oc->ssl live, -1 on failure (t error set). */
92int
93brix_cache_origin_tls_upgrade(brix_cache_fill_t *t,
94 brix_cache_origin_conn_t *oc, const ngx_str_t *host)
95{
96 oc->ssl_ctx = SSL_CTX_new(TLS_client_method());
97 if (oc->ssl_ctx == NULL) {
98 brix_cache_set_error(t, kXR_ServerError, 0,
99 "cache origin TLS context failed");
100 return -1;
101 }
102 SSL_CTX_set_min_proto_version(oc->ssl_ctx, TLS1_2_VERSION);
103 SSL_CTX_set_verify(oc->ssl_ctx, SSL_VERIFY_PEER, NULL);
104
105 if (origin_tls_load_verify(oc->ssl_ctx, &t->conf->trusted_ca) != 0) {
106 brix_cache_set_error(t, kXR_ServerError, 0,
107 "cache origin CA load failed");
108 return -1;
109 }
110
111 oc->ssl = SSL_new(oc->ssl_ctx);
112 if (oc->ssl == NULL) {
113 brix_cache_set_error(t, kXR_ServerError, 0,
114 "cache origin TLS allocation failed");
115 return -1;
116 }
117
118 /*
119 * Verify the presented cert is actually FOR this host: SSL_VERIFY_PEER only
120 * checks the chain, not the name, so without SSL_set1_host an on-path attacker
121 * with ANY CA-valid cert could impersonate the origin. Fold the name check
122 * into the handshake verification and set SNI so the origin serves the right
123 * cert.
124 */
125 if (host != NULL && host->len > 0 && host->data != NULL) {
126 char vhost[256];
127 size_t hl = host->len < sizeof(vhost) - 1
128 ? host->len : sizeof(vhost) - 1;
129
130 ngx_memcpy(vhost, host->data, hl);
131 vhost[hl] = '\0';
132 SSL_set_hostflags(oc->ssl, X509_CHECK_FLAG_NO_PARTIAL_WILDCARDS);
133 if (SSL_set1_host(oc->ssl, vhost) != 1) {
134 brix_cache_set_error(t, kXR_ServerError, 0,
135 "cache origin TLS host-verify setup failed");
136 return -1;
137 }
138 (void) SSL_set_tlsext_host_name(oc->ssl, vhost);
139 }
140 SSL_set_fd(oc->ssl, oc->fd);
141
142 if (SSL_connect(oc->ssl) != 1) {
143 char ebuf[160];
144
145 ERR_error_string_n(ERR_get_error(), ebuf, sizeof(ebuf));
146 ngx_log_error(NGX_LOG_ERR, ngx_cycle->log, 0,
147 "brix: cache origin TLS handshake failed: %s", ebuf);
148 brix_cache_set_error(t, kXR_TLSRequired, 0,
149 "cache origin TLS handshake failed");
150 return -1;
151 }
152 return 0;
153}

155/* brix_cache_origin_connect_addr — getaddrinfo, try each result with a
156 * non-blocking poll-timeout connect (avoids the ~2min TCP retransmit stall), set
157 * SO_RCVTIMEO/SO_SNDTIMEO. TLS is NOT engaged here for a root:// origin: a stock
158 * XRootD `roots://` peer does the kXR_protocol exchange over CLEARTEXT and then
159 * advertises kXR_gotoTLS, so brix_cache_origin_bootstrap upgrades the fd in place
160 * (brix_cache_origin_tls_upgrade) after the protocol reply. Runs in a blocking
161 * fill thread-pool worker, so it uses poll() rather than the event loop;
162 * BRIX_CACHE_IO_TIMEOUT bounds an unreachable peer. */
163int
164brix_cache_origin_connect_addr(brix_cache_fill_t *t,
165 brix_cache_origin_conn_t *oc, const ngx_str_t *host, uint16_t portnum)
166{
167 struct addrinfo hints;
168 struct addrinfo *res, *rp;
169 char port[16];
170 int rc;
171
172 oc->fd = -1;
173 oc->ssl_ctx = NULL;
174 oc->ssl = NULL;
175 res = NULL;
176 rc = -1;
177
178 if (host == NULL || host->len == 0 || host->data == NULL
179 || portnum == 0)
180 {
181 brix_cache_set_error(t, kXR_ServerError, 0,
182 "cache origin address not configured");
183 return -1;
184 }
185
186 snprintf(port, sizeof(port), "%u", (unsigned) portnum);
187
188 ngx_memzero(&hints, sizeof(hints));
189 hints.ai_socktype = SOCK_STREAM;
190 {
191 ngx_uint_t fam = (t->conf != NULL)
192 ? t->conf->cache_origin_family : (ngx_uint_t) BRIX_AF_AUTO;
193 if (fam == NGX_CONF_UNSET_UINT) {
194 fam = (ngx_uint_t) BRIX_AF_AUTO;
195 }
196 hints.ai_family = (int) fam; /* AUTO==AF_UNSPEC tries every family */
197 }
198
199 if (getaddrinfo((char *) host->data, port, &hints, &res) != 0) {
200 brix_cache_set_error(t, kXR_ServerError, 0,
201 "cache origin DNS resolution failed");
202 return -1;
203 }
204
205 for (rp = res; rp != NULL; rp = rp->ai_next) {
206 int fd;
207
208 fd = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
209 if (fd < 0) {
210 continue;
211 }
212
213 brix_apply_socket_io_timeouts(fd, BRIX_CACHE_IO_TIMEOUT);
214
215 if (brix_connect_fd_deadline(fd, rp->ai_addr, rp->ai_addrlen,
216 BRIX_CACHE_IO_TIMEOUT * 1000) == 0)
217 {
218 oc->fd = fd;
219 rc = 0;
220 break;
221 }
222
223 close(fd);
224 }
225
226 freeaddrinfo(res);
227
228 if (rc != 0) {
229 brix_cache_set_syserror(t, kXR_ServerError,
230 "cache origin connect failed");
231 return -1;
232 }
233
234 /* TLS is deferred: the root:// bootstrap negotiates kXR_protocol over
235 * cleartext and upgrades on the origin's kXR_gotoTLS advert. See
236 * brix_cache_origin_tls_upgrade + brix_cache_origin_bootstrap. */
237 return 0;
238}
