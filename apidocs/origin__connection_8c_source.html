14/* brix_cache_origin_close — free everything an origin connection holds (SSL,
15 * SSL_CTX, socket fd) and NULL the pointers; symmetric with origin_connect and
16 * called on every error path. Order matters: SSL shutdown before free, SSL_CTX
17 * before the fd close, fd last, so the SSL layer never dangles. */
18
19void
20brix_cache_origin_close(brix_cache_origin_conn_t *oc)
21{
22 if (oc->ssl != NULL) {
23 SSL_shutdown(oc->ssl);
24 SSL_free(oc->ssl);
25 oc->ssl = NULL;
26 }
27
28 if (oc->ssl_ctx != NULL) {
29 SSL_CTX_free(oc->ssl_ctx);
30 oc->ssl_ctx = NULL;
31 }
32
33 if (oc->fd >= 0) {
34 close(oc->fd);
35 oc->fd = -1;
36 }
37}
