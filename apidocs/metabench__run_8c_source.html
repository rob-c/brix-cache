1/*
2 * metabench_run.c — threaded GSI runner for the metadata storm.
3 *
4 * WHAT: plan->workers threads, each holding ONE persistent GSI connection, run a
5 * deterministic mkdir/chmod/truncate/stat (CREATE) or rm/rmdir (REMOVE)
6 * program over its own /w<id> subtree, timing every op.
7 * WHY: isolate the pblock backend under concurrency — the cost measured is the
8 * backend's, not repeated handshakes or process startup.
9 * HOW: one brix_conn per thread (brix_conn is one-request-in-flight and not
10 * thread-safe; one-per-thread is the documented MT pattern). Latencies are
11 * merged and reduced to p50/p95/p99 by the pure metabench helpers.
12 */
13#include "metabench_run.h"
14#include <pthread.h>
15#include <stdlib.h>
16#include <string.h>

18/* one worker's slice: connect (GSI) once, run its subtree, record latencies. */
19typedef struct {
20 const brix_url *url;
21 const brix_opts *opts;
22 const metabench_plan *plan;
23 metabench_phase phase;
24 int id;
25 double *lat_ms; /* capacity lat_cap, filled lat_n */
26 long lat_cap;
27 long lat_n;
28 long ops, failures;
29} mb_worker;

31/* record one op's latency (drop silently once the per-worker buffer is full). */
32static void mb_record(mb_worker *w, uint64_t t0)
33{
34 double ms = (double) (brix_mono_ns() - t0) / 1e6;
35 if (w->lat_n < w->lat_cap) {
36 w->lat_ms[w->lat_n++] = ms;
37 }
38}

40/* CREATE: mkdir+chmod the tree, then create+chmod+stat each file. The stat
41 * asserts the chmod persisted (mode read-back), so a silent mode loss fails. */
42static void mb_create(mb_worker *w, brix_conn *c)
43{
44 const metabench_plan *p = w->plan;
45 char path[512];
46 brix_status st;
47 uint64_t t0;
48
49 snprintf(path, sizeof(path), "/w%d", w->id);
50 t0 = brix_mono_ns();
51 if (brix_mkdir(c, path, 0755, 0, &st) != 0) { w->failures++; }
52 w->ops++; mb_record(w, t0);
53
54 t0 = brix_mono_ns();
55 if (brix_chmod(c, path, 0755, &st) != 0) { w->failures++; }
56 w->ops++; mb_record(w, t0);
57
58 for (int d = 0; d < p->dirs_per_worker; d++) {
59 snprintf(path, sizeof(path), "/w%d/d%d", w->id, d);
60 t0 = brix_mono_ns();
61 if (brix_mkdir(c, path, 0700, 0, &st) != 0) { w->failures++; }
62 w->ops++; mb_record(w, t0);
63
64 t0 = brix_mono_ns();
65 if (brix_chmod(c, path, 0700, &st) != 0) { w->failures++; }
66 w->ops++; mb_record(w, t0);
67
68 for (int f = 0; f < p->files_per_dir; f++) {
69 snprintf(path, sizeof(path), "/w%d/d%d/f%d", w->id, d, f);
70
71 /* create an empty file: open kXR_new (force=0) then close — the same
72 * metadata-only path `xrdfs touch` uses (truncate(2) needs an
73 * existing target, so it is NOT a create). */
74 brix_file fh;
75 t0 = brix_mono_ns();
76 if (brix_file_open_write(c, path, 0 /* new */, 0 /* posc */, &fh, &st) != 0) {
77 w->failures++;
78 } else if (brix_file_close(c, &fh, &st) != 0) {
79 w->failures++;
80 }
81 w->ops++; mb_record(w, t0);
82
83 t0 = brix_mono_ns();
84 if (brix_chmod(c, path, 0640, &st) != 0) { w->failures++; }
85 w->ops++; mb_record(w, t0);
86
87 brix_statinfo si;
88 t0 = brix_mono_ns();
89 if (brix_stat(c, path, &si, &st) != 0) {
90 w->failures++;
91 } else if (si.have_ext && (si.mode & 0777) != 0640) {
92 w->failures++; /* chmod did not persist */
93 }
94 w->ops++; mb_record(w, t0);
95 }
96 }
97}

99/* REMOVE: rm files, rmdir dirs bottom-up; leaves the store as it started. */
100static void mb_remove(mb_worker *w, brix_conn *c)
101{
102 const metabench_plan *p = w->plan;
103 char path[512];
104 brix_status st;
105 uint64_t t0;
106
107 for (int d = 0; d < p->dirs_per_worker; d++) {
108 for (int f = 0; f < p->files_per_dir; f++) {
109 snprintf(path, sizeof(path), "/w%d/d%d/f%d", w->id, d, f);
110 t0 = brix_mono_ns();
111 if (brix_rm(c, path, &st) != 0) { w->failures++; }
112 w->ops++; mb_record(w, t0);
113 }
114 snprintf(path, sizeof(path), "/w%d/d%d", w->id, d);
115 t0 = brix_mono_ns();
116 if (brix_rmdir(c, path, &st) != 0) { w->failures++; }
117 w->ops++; mb_record(w, t0);
118 }
119
120 snprintf(path, sizeof(path), "/w%d", w->id);
121 t0 = brix_mono_ns();
122 if (brix_rmdir(c, path, &st) != 0) { w->failures++; }
123 w->ops++; mb_record(w, t0);
124}

126/* thread entry: one GSI session, then the phase program. A connect failure marks
127 * every planned op for this worker as failed (so the run cannot pass). */
128static void *mb_thread(void *arg)
129{
130 mb_worker *w = arg;
131 brix_conn c;
132 brix_status st;
133
134 brix_status_clear(&st);
135 if (brix_connect(&c, w->url, w->opts, &st) != 0) {
136 w->failures = w->lat_cap;
137 w->ops = w->lat_cap;
138 return NULL;
139 }
140
141 if (w->phase == MB_CREATE) {
142 mb_create(w, &c);
143 } else {
144 mb_remove(w, &c);
145 }
146
147 brix_close(&c);
148 return NULL;
149}

151/* ---- Populate the per-worker slice array from the shared plan ----
152 *
153 * WHAT: fills wk[0..W-1] with the immutable run inputs (url/opts/plan/phase,
154 * sequential id, latency-buffer capacity) and allocates each worker's
155 * zeroed latency buffer. No return value; a failed lat_ms calloc leaves a
156 * NULL buffer that mb_record tolerates (it only writes when lat_n<lat_cap).
157 *
158 * WHY: every worker needs the same read-only context plus its own private
159 * latency storage; hoisting this out of metabench_run keeps the
160 * orchestrator flat and below the complexity cap.
161 *
162 * HOW: 1. iterate i over [0,W); 2. copy the shared pointers and scalar id/cap;
163 * 3. calloc a per_cap-entry double buffer for that worker.
164 */
165static void mb_init_workers(mb_worker *wk, int W, const brix_url *url,
166 const brix_opts *opts, const metabench_plan *plan,
167 metabench_phase phase, long per_cap)
168{
169 for (int i = 0; i < W; i++) {
170 wk[i].url = url;
171 wk[i].opts = opts;
172 wk[i].plan = plan;
173 wk[i].phase = phase;
174 wk[i].id = i;
175 wk[i].lat_cap = per_cap;
176 wk[i].lat_ms = calloc((size_t) per_cap, sizeof(double));
177 }
178}
179
180/* ---- Launch, join, and time the worker threads ----
181 *
182 * WHAT: spawns W threads on mb_thread (one persistent GSI connection each),
183 * joins them all, and returns the wall-clock duration in seconds.
184 *
185 * WHY: the storm's throughput and per-op rates are derived from this single
186 * wall interval; measuring it around exactly the create/join bracket keeps
187 * the timing honest and confines the pthread plumbing to one helper.
188 *
189 * HOW: 1. sample a monotonic start; 2. pthread_create each worker slice;
190 * 3. pthread_join each in order; 4. return elapsed nanoseconds / 1e9.
191 */
192static double mb_run_workers(mb_worker *wk, pthread_t *th, int W)
193{
194 uint64_t t0 = brix_mono_ns();
195 for (int i = 0; i < W; i++) {
196 pthread_create(&th[i], NULL, mb_thread, &wk[i]);
197 }
198 for (int i = 0; i < W; i++) {
199 pthread_join(th[i], NULL);
200 }
201 return (double) (brix_mono_ns() - t0) / 1e9;
202}

204/* ---- Sum the recorded latency counts across all workers ----
205 *
206 * WHAT: returns the total number of latency samples recorded by wk[0..W-1]
207 * (each worker's lat_n), used to size the merged latency array.
208 *
209 * WHY: the merged buffer must be exactly large enough for every sample; a
210 * separate counting pass lets the caller allocate once before merging.
211 *
212 * HOW: 1. accumulate wk[i].lat_n over i in [0,W); 2. return the sum.
213 */
214static long mb_total_lat(const mb_worker *wk, int W)
215{
216 long total_lat = 0;
217 for (int i = 0; i < W; i++) {
218 total_lat += wk[i].lat_n;
219 }
220 return total_lat;
221}

223/* ---- Merge worker results into the output and the combined latency array ----
224 *
225 * WHAT: folds each worker's ops/failures and per-worker record into out, copies
226 * every recorded latency into all[], frees each worker's lat_ms, and
227 * returns the number of samples written to all[]. When all is NULL
228 * (merged-buffer alloc failed) the inner copy is skipped and 0 is returned.
229 *
230 * WHY: aggregation is the bulk of the run's branching; isolating it keeps the
231 * orchestrator flat while preserving the exact fold order and the
232 * per-worker latency-buffer ownership (freed here, where it is consumed).
233 *
234 * HOW: 1. iterate i over [0,W); 2. add ops/failures into totals and fill the
235 * per_worker[i] row (ops_per_sec guarded on wall>0); 3. append each of
236 * wk[i].lat_n samples into all[k++] only while all is non-NULL; 4. free
237 * wk[i].lat_ms; 5. return k, the merged sample count.
238 */
239static long mb_aggregate(metabench_result *out, mb_worker *wk, int W,
240 double *all, double wall)
241{
242 long k = 0;
243 for (int i = 0; i < W; i++) {
244 out->total_ops += wk[i].ops;
245 out->failures += wk[i].failures;
246 out->per_worker[i].id = i;
247 out->per_worker[i].ops = wk[i].ops;
248 out->per_worker[i].failures = wk[i].failures;
249 out->per_worker[i].ops_per_sec = wall > 0 ? (double) wk[i].ops / wall : 0;
250 for (long j = 0; j < wk[i].lat_n && all != NULL; j++) {
251 all[k++] = wk[i].lat_ms[j];
252 }
253 free(wk[i].lat_ms);
254 }
255 return k;
256}

258/* ---- Decide pass/fail for the completed storm ----
259 *
260 * WHAT: returns 0 when the run passed (zero failures AND, if a p99 ceiling is
261 * set, the measured p99 is within it) and -1 otherwise.
262 *
263 * WHY: the verdict couples several conditions; naming it keeps the exit
264 * criterion in one place and out of the orchestrator's branch budget. A
265 * p99_ceil_ms of 0 means "no latency ceiling", so only failures gate.
266 *
267 * HOW: 1. require out->failures == 0; 2. AND (p99_ceil_ms == 0 OR p99 within
268 * ceiling); 3. map the boolean to 0 (pass) / -1 (fail).
269 */
270static int mb_verdict(const metabench_result *out, const metabench_plan *plan)
271{
272 int ok = (out->failures == 0)
273 && (plan->p99_ceil_ms == 0 || out->p99_ms <= plan->p99_ceil_ms);
274 return ok ? 0 : -1;
275}

277int metabench_run(const brix_url *url, const brix_opts *opts,
278 const metabench_plan *plan, metabench_phase phase,
279 metabench_result *out, brix_status *st)
280{
281 (void) st;
282
283 int W = plan->workers;
284 if (W > 64) {
285 W = 64;
286 }
287 long per_cap = metabench_storm_ops(plan) / plan->workers + 8;
288
289 mb_worker *wk = calloc((size_t) W, sizeof(*wk));
290 pthread_t *th = calloc((size_t) W, sizeof(*th));
291 if (wk == NULL || th == NULL) {
292 free(wk);
293 free(th);
294 return -1;
295 }
296
297 mb_init_workers(wk, W, url, opts, plan, phase, per_cap);
298
299 double wall = mb_run_workers(wk, th, W);
300
301 memset(out, 0, sizeof(*out));
302 out->phase = phase;
303 out->nworkers = W;
304 out->wall_s = wall;
305
306 long total_lat = mb_total_lat(wk, W);
307 double *all = calloc((size_t) (total_lat ? total_lat : 1), sizeof(double));
308 long k = mb_aggregate(out, wk, W, all, wall);
309
310 out->ops_per_sec = wall > 0 ? (double) out->total_ops / wall : 0;
311 out->p50_ms = metabench_percentile(all, (size_t) k, 50.0);
312 out->p95_ms = metabench_percentile(all, (size_t) k, 95.0);
313 out->p99_ms = metabench_percentile(all, (size_t) k, 99.0);
314
315 free(all);
316 free(wk);
317 free(th);
318
319 return mb_verdict(out, plan);
320}
