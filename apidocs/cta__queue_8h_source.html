4/*
5 * cta_queue.h — CTA request-queue state machine.
6 *
7 * WHAT: a per-worker registry of archive/retrieve/cancel/query requests with a
8 * lifecycle state machine and an owner/admin-gated cancel.
9 * WHY: the flagship CTA SSI service tracks each request's progress; `query`
10 * lists the queue and `cancel` aborts a prior request.
11 * HOW: a fixed table allocated once per worker (long-lived, so malloc — not a
12 * connection pool). Transitions are validated against a table, never an
13 * ad-hoc branch ladder. Pure C, standalone-testable.
14 */
