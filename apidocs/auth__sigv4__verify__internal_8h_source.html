1/*
2 * auth_sigv4_verify_internal.h — cross-file declarations for the S3 SigV4
3 * verification split (auth_sigv4_verify.c / _time.c / _crypto.c).
4 *
5 * WHAT: Declares the two result structs threaded through one SigV4
6 * verification (the resolved request timestamp and the server-computed
7 * signature material) plus the handful of functions that cross a
8 * translation-unit boundary after the phase-79 file-size split: the
9 * auth-result metric sink, the timestamp resolver, and the two crypto
10 * steps (canonical-request signature compute + constant-time compare).
11 * It also re-exports the two sibling SigV4 entry points the split files
12 * consume (the canonical query-string builder and the header getter).
13 * WHY: auth_sigv4_verify.c exceeded the ~500-line file-size guard, so it was
14 * carved into three cohesive units — the verifier orchestrator (verify.c),
15 * the timestamp parse + clock-skew rules (verify_time.c), and the
16 * byte-frozen canonicalisation + HMAC signing-key + signature compare
17 * (verify_crypto.c). Grouping the shared structs and the crossing entry
18 * points here keeps every definition in exactly one place while preserving
19 * byte-identical canonical-request construction, string-to-sign, HMAC
20 * signing-key derivation, and constant-time signature comparison of the
21 * original single-file implementation. INVARIANT §6: this is SigV4-only
22 * machinery — it shares no logic with WLCG bearer-token auth.
23 * HOW: Requires "s3.h" (nginx request types, ngx_http_s3_loc_conf_t, u_char,
24 * ngx_uint_t) and "s3_auth_internal.h" (sigv4_components_t) before
25 * inclusion; both are included here. Every SigV4-verify .c file includes
26 * this header. The declared symbols are non-static at their definitions so
27 * the split links (a symbol defined in one file and referenced from
28 * another must carry external linkage AND appear here).
29 */

36/*
37 * s3_amz_date_out_t — resolved SigV4 request timestamp.
38 *
39 * WHAT: Carries the amz timestamp resolved by s3_sigv4_resolve_request_time:
40 * an inline storage buffer for the header path plus the (buf, len) view
41 * the string-to-sign consumes.
42 * WHY: Bundles the header-scratch buffer and its two out-params into one
43 * object so the resolver stays under the 5-param gate and the buffer's
44 * lifetime (it must outlive the resolve call) is explicit at the callsite.
45 * HOW: For the header path `date` points into `buf`; for the presigned path it
46 * points into comp's own storage. `len` is the strlen of `date`.
47 */
48typedef struct {
49 char buf[32]; /* header-path scratch; must outlive resolve */
50 const char *date; /* resolved timestamp (into buf or comp) */
51 size_t len; /* strlen(date) */
52} s3_amz_date_out_t;

54/*
55 * s3_sigv4_sig_out_t — server-computed SigV4 signature material.
56 *
57 * WHAT: Outputs of s3_sigv4_compute_signature: the derived signing key (k4)
58 * and the 64-char lowercase-hex server signature.
59 * WHY: Bundles the two crypto outputs so the computer stays under the 5-param
60 * gate; k4 is later needed by s3_sigv4_finish for chunk-sig retention.
61 * HOW: Zero-initialise at declaration; populated only on the NGX_OK return.
62 */
63typedef struct {
64 u_char k4[32]; /* derived signing key */
65 char computed_hex[65]; /* server signature, 64 hex + NUL */
66} s3_sigv4_sig_out_t;
