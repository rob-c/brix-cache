4/*
5 * brix_json_escape — write the ESCAPED body of a JSON string (no quotes).
6 *
7 * WHAT: emits `s` to `out` with the JSON string escapes applied; a NULL `s`
8 * emits nothing.
9 * WHY: these strings routinely carry untrusted, server-supplied wire text
10 * (paths, status messages, directory-entry names), so control bytes,
11 * quotes and backslashes MUST be escaped or a hostile value could break
12 * out of the string and forge JSON structure — a security requirement.
13 * HOW: per-byte switch for the short escapes (" \\ \n \r \t); any other
14 * control byte (<0x20) or non-ASCII byte (>=0x80) becomes a \uXXXX escape;
15 * printable ASCII passes through verbatim.
16 */
17void
18brix_json_escape(FILE *out, const char *s)
19{
20 const unsigned char *p;
21 if (s == NULL) { return; } /* NULL input: emit nothing; brix_json_fputs's surrounding quotes yield "" */
22 p = (const unsigned char *) s;
23 for (; *p != '\0'; p++) {
24 switch (*p) {
25 case '"': fputs("\\\"", out); break;
26 case '\\': fputs("\\\\", out); break;
27 case '\n': fputs("\\n", out); break;
28 case '\r': fputs("\\r", out); break;
29 case '\t': fputs("\\t", out); break;
30 default:
31 if (*p < 0x20 || *p >= 0x80) {
32 fprintf(out, "\\u%04x", (unsigned) *p);
33 } else {
34 fputc(*p, out);
35 }
36 }
37 }
38}

40/*
41 * brix_json_fputs — write a complete, quoted, escaped JSON string.
42 *
43 * WHAT: emits `"<escaped s>"` to `out` (a NULL `s` yields an empty string "").
44 * WHY: the single choke-point for emitting any JSON string value or key, so
45 * every string in the output is quoted AND escaped the same safe way.
46 * HOW: writes the opening quote, delegates the body to brix_json_escape, then
47 * writes the closing quote.
48 */
49void
50brix_json_fputs(FILE *out, const char *s)
51{
52 fputc('"', out);
53 brix_json_escape(out, s);
54 fputc('"', out);
55}

57/*
58 * brix_json_kv_str — write a "key":"value" pair, optionally comma-terminated.
59 *
60 * WHAT: emits `"<k>":"<v>"` (both escaped) plus a trailing ',' when `comma`.
61 * WHY: object fields carrying string values are the common case; folding the
62 * colon and optional separator here keeps every call site uniform.
63 * HOW: brix_json_fputs for the key, a ':', brix_json_fputs for the value, then
64 * a ',' iff `comma` is nonzero.
65 */
66void
67brix_json_kv_str(FILE *out, const char *k, const char *v, int comma)
68{
69 brix_json_fputs(out, k);
70 fputc(':', out);
71 brix_json_fputs(out, v);
72 if (comma) { fputc(',', out); }
73}

75/*
76 * brix_json_kv_ll — write a "key":<number> pair, optionally comma-terminated.
77 *
78 * WHAT: emits `"<k>":<v>` with `v` as a decimal long long, plus a trailing ','
79 * when `comma`.
80 * WHY: numeric fields (sizes, mtimes, counts) must appear unquoted so JSON
81 * consumers read them as numbers, not strings.
82 * HOW: brix_json_fputs for the (escaped) key, then a printf of ":%lld" and the
83 * optional separator.

85void
86brix_json_kv_ll(FILE *out, const char *k, long long v, int comma)
