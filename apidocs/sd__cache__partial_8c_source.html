2 * sd_cache_partial.c — slice / partial-object caching (section 6.5).
3 *
4 * WHAT: The on-demand block-fill path: when policy.slice_size > 0 on a LOCAL
5 * posix cache store, an object is cached in fixed-size BLOCKS filled as
6 * reads touch them. Owns sd_cache_partial_open (build the partial-serve
7 * object) and the decorator's own byte slots reached only for such an
8 * object (pread range-fill, close, fstat, and the never-sendfile slot).
9 *
10 * WHY: Split from sd_cache.c (phase-79) to keep every cache file under the
11 * ~500-line, one-concept-per-file cap. Slice mode is a self-contained
12 * strategy — a sparse cache fd, a present bitmap, and per-user credentials
13 * captured for deferred fills — reviewable apart from the whole-file fill
14 * spine (sd_cache_fill.c) and the vtable adapters (sd_cache.c).
15 *
16 * HOW: A read range-fills only the blocks it touches (source pread -> cache
17 * pwrite -> cinfo present-bit) via brix_cstore_serve_pread's callback, so
18 * a Range request never pulls the whole object. The served object carries
19 * this state and ranges-fill on pread; it is never sendfiled (it may have
20 * holes). A fully-filled object's cinfo becomes COMPLETE and subsequent
21 * opens take the whole-file hit fast path. The five cross-called symbols
22 * (sd_cache_partial_open + the four byte slots) are non-static and reach
23 * sd_cache.c through sd_cache_internal.h. ZERO behaviour change.
24 */

42/* ---- slice/partial caching (section 6.5) ----------------------------------
43 * When policy.slice_size > 0 on a LOCAL posix cache store, an object is cached in
44 * fixed-size BLOCKS filled on demand: a read range-fills only the blocks it
45 * touches (source pread -> cache pwrite -> cinfo present-bit), so a Range request
46 * never pulls the whole object. The served object carries this state and ranges-
47 * fill on pread; it is never sendfiled (it may have holes). A fully-filled object's
48 * cinfo becomes COMPLETE and subsequent opens take the whole-file hit fast path. */
49
50typedef struct {
51 brix_sd_instance_t *source;
52 brix_sd_obj_t *src_obj; /* lazily opened on the first miss */
53 int cache_fd; /* the RW (sparse) cache object */
54 off_t size;
55 uint32_t block_size;
56 uint32_t mode; /* origin perm bits recorded in cinfo */
57 uint64_t mtime;
58 uint64_t nblocks;
59 uint8_t *bitmap; /* present blocks (in-memory mirror) */
60 size_t bitmap_len;
61 ngx_log_t *log;
62 char key[1024];
63 char cache_path[PATH_MAX]; /* for cinfo record_block */
64 /* Per-user credential copies for deferred (range-fill) source opens.
65 * A partial-fill block may be filled on a later pread after the request
66 * context — and its brix_sd_cred_t — is gone; embedding NUL-terminated
67 * copies here ensures later opens can still authenticate as the owner.
68 * cred_proxy[0] == '\0' means no per-user credential (service cred). */
69 char cred_proxy[1024]; /* x509_proxy path, or "" */
70 char cred_key[128]; /* ucred key stem, or "" */
71 char cred_principal[512]; /* principal string, or "" */
72} sd_cache_partial_t;

74/* Lazily open the partial object's source for a deferred range-fill.
75 *
76 * WHAT: Ensures p->src_obj is an open, pread-capable source object, opening it
77 * on the first missing-block fill.
78 *
79 * WHY: Block fills occur on pread calls after the request context is gone;
80 * the open must happen against the credential captured at partial-open
81 * time so the fill still authenticates as the original user.
82 *
83 * HOW: Authenticate as the original user when a credential was captured at
84 * partial-open time; otherwise fall back to the service credential.
85 * The stack brix_sd_cred_t only needs x509_proxy/key/principal — the
86 * cred_dir/fallback_deny fields are not needed for a range-fill re-open
87 * (the proxy path was already resolved at object-open time).
88 * Returns 0 (src_obj usable) or -1 with errno set. */
89static int
90partial_source_ensure(sd_cache_partial_t *p)
91{
92 int e = 0;
93
94 if (p->src_obj != NULL) {
95 return 0;
96 }
97 if (p->source->driver->open == NULL) {
98 errno = ENOSYS;
99 return -1;
100 }
101 if (p->cred_proxy[0] != '\0') {
102 brix_sd_cred_t rc;
103
104 ngx_memzero(&rc, sizeof(rc));
105 rc.x509_proxy = p->cred_proxy;
106 rc.key = p->cred_key;
107 rc.principal = p->cred_principal;
108 p->src_obj = brix_sd_open_maybe_cred(p->source, p->key,
109 BRIX_SD_O_READ, 0, &rc, &e);
110 } else {
111 p->src_obj = p->source->driver->open(p->source, p->key,
112 BRIX_SD_O_READ, 0, &e);
113 }
114 if (p->src_obj == NULL) {
115 errno = e ? e : EIO;
116 return -1;
117 }
118 /* Object ops dispatch through the OBJECT's driver (a decorator source
119 * returns the tier-below's object from a read open — see sd_cache_fill). */
120 if (p->src_obj->driver->pread == NULL) {
121 errno = ENOSYS;
122 return -1;
123 }
124 return 0;
125}

127/* Copy one block's bytes source -> sparse cache object.
128 *
129 * WHAT: preads [bstart, bstart+blen) from the open source object into a heap
130 * buffer and pwrites the bytes into the cache fd at the same offset.
131 *
132 * WHY: The read and write retry loops (EINTR, short reads against a stale
133 * stat) are pure byte plumbing — separated from the block bookkeeping
134 * (cinfo record + bitmap mark) that publishes the block as present.
135 *
136 * HOW: A short source read (r == 0 before blen) stops early and writes what
137 * was read — the block is still recorded present by the caller, matching
138 * the "short source vs stat" doctrine. Returns 0 or -1 with errno set. */
139static int
140partial_block_copy(sd_cache_partial_t *p, off_t bstart, size_t blen)
141{
142 u_char *bbuf;
143 off_t got = 0;
144 off_t w = 0;
145
146 bbuf = malloc(blen);
147 if (bbuf == NULL) {
148 errno = ENOMEM;
149 return -1;
150 }
151 while ((size_t) got < blen) {
152 ssize_t r = p->src_obj->driver->pread(p->src_obj, bbuf + got,
153 blen - (size_t) got, bstart + got);
154 if (r < 0) {
155 if (errno == EINTR) { continue; }
156 free(bbuf);
157 return -1;
158 }
159 if (r == 0) {
160 break; /* short source vs stat - stop */
161 }
162 got += r;
163 }
164 while (w < got) {
165 ssize_t n = pwrite(p->cache_fd, bbuf + w, (size_t) (got - w),
166 bstart + w);
167 if (n < 0) {
168 if (errno == EINTR) { continue; }
169 free(bbuf);
170 return -1;
171 }
172 w += n;
173 }
174 free(bbuf);
175 return 0;
176}

178/* Fetch block `blk` from the source into the cache object + mark it present. */
179static int
180sd_cache_fill_block(sd_cache_partial_t *p, uint64_t blk)
181{
182 off_t bstart = (off_t) blk * p->block_size;
183 size_t blen;
184
185 if (bstart >= p->size) {
186 return 0;
187 }
188 blen = (size_t) ((bstart + (off_t) p->block_size <= p->size)
189 ? p->block_size : (p->size - bstart));
190
191 if (partial_source_ensure(p) != 0) {
192 return -1;
193 }
194 if (partial_block_copy(p, bstart, blen) != 0) {
195 return -1;
196 }
197
198 (void) brix_cache_cinfo_record_block(p->cache_path, (uint64_t) p->size,
199 p->block_size, p->mtime, p->mode, blk,
200 p->log);
201 if (p->bitmap != NULL && blk < p->nblocks) {
202 brix_cache_cinfo_mark_block(p->bitmap, blk);
203 }
204 return 0;
205}

207/* Capture the per-user credential for deferred block fills.
208 *
209 * WHAT: Copies x509_proxy/key/principal from `cred` into the partial state's
210 * embedded NUL-terminated buffers.
211 *
212 * WHY: Block fills occur on pread calls after the request context — and its
213 * brix_sd_cred_t — is gone; capturing the resolved credential at open
214 * time ensures later fills authenticate as the original user rather
215 * than the service account.
216 *
217 * HOW: Only copies when the proxy path is non-empty; otherwise cred_proxy
218 * stays '\0' (which triggers the service-credential path in
219 * partial_source_ensure). */
220static void
221partial_capture_cred(sd_cache_partial_t *p, const brix_sd_cred_t *cred)
222{
223 if (cred == NULL || cred->x509_proxy == NULL || cred->x509_proxy[0] == '\0') {
224 return;
225 }
226 ngx_cpystrn((u_char *) p->cred_proxy, (u_char *) cred->x509_proxy,
227 sizeof(p->cred_proxy));
228 ngx_cpystrn((u_char *) p->cred_key,
229 (u_char *) (cred->key ? cred->key : ""),
230 sizeof(p->cred_key));
231 ngx_cpystrn((u_char *) p->cred_principal,
232 (u_char *) (cred->principal ? cred->principal : ""),
233 sizeof(p->cred_principal));
234}

236/* Adopt or initialize the partial object's present bitmap.
237 *
238 * WHAT: Loads a previously-recorded present bitmap (an earlier partial fill)
239 * into p->bitmap when it matches this object's geometry; otherwise
240 * starts all-absent.
241 *
242 * WHY: Re-opening a partially-filled object must not forget which blocks are
243 * already present — but a stale bitmap (size or block-size changed)
244 * would serve wrong bytes, so geometry mismatch discards it.
245 *
246 * HOW: brix_cache_cinfo_load from the cache path; adopt only when the bitmap
247 * length, recorded size, and block size all match; otherwise free the
248 * loaded bitmap and calloc a zeroed one (NULL for an empty object). */
249static void
250partial_adopt_bitmap(sd_cache_partial_t *p, const char *cpath)
251{
252 brix_cache_cinfo_t hdr;
253 uint8_t *bm = NULL;
254 size_t bl = 0;
255
256 if (brix_cache_cinfo_load(cpath, &hdr, &bm, &bl) == NGX_OK
257 && bl == p->bitmap_len && hdr.size == (uint64_t) p->size
258 && hdr.block_size == p->block_size)
259 {
260 p->bitmap = bm;
261 return;
262 }
263 free(bm);
264 p->bitmap = (p->bitmap_len > 0) ? calloc(1, p->bitmap_len) : NULL;
265}

267/* Open a partial-serve object for `key` (slice mode).
268 *
269 * WHAT: Allocates and wires a sd_cache_partial_t for on-demand block fills from
270 * the source, recording the per-user credential for deferred re-opens.
271 *
272 * WHY: Block fills occur on pread calls after the request context is gone;
273 * capturing the resolved credential at open time ensures later fills
274 * authenticate as the original user rather than the service account.
275 *
276 * HOW: Stats the source, opens the sparse cache object (0600, see below),
277 * wires the partial state (geometry + credential + present bitmap via
278 * partial_capture_cred / partial_adopt_bitmap), and returns a heap
279 * object shell whose driver is this decorator (range-fill pread).
280 * Returns the new object, or NULL with *err_out set on failure. */
281brix_sd_obj_t *
282sd_cache_partial_open(brix_sd_instance_t *inst, sd_cache_inst_state *st,
283 const char *key, const brix_sd_cred_t *cred, int *err_out)
284{
285 brix_sd_instance_t *src = st->source;
286 brix_sd_stat_t snap;
287 sd_cache_partial_t *p;
288 brix_sd_obj_t *o;
289 char cpath[PATH_MAX];
290 uint32_t bs;
291 int fd;
292
293 if (src->driver->stat == NULL
294 || src->driver->stat(src, key, &snap) != NGX_OK)
295 {
296 if (err_out != NULL) { *err_out = ENOENT; }
297 return NULL;
298 }
299 bs = (uint32_t) st->policy.slice_size;
300 /* Force owner rw ONLY (0600): the partial object is re-opened O_RDWR for every
301 * incremental block fill (a read-only 0444 source would EACCES the second open
302 * and silently fall back to a whole-file fill, §6.5). SECURITY: no group/other
303 * bits — this svc-owned cache artifact must not be directly readable by a mapped
304 * low-priv uid. The origin perms are carried in the cinfo and served back to
305 * clients (see sd_cache_stat / the READ hit), decoupled from this physical mode. */
306 fd = brix_cstore_partial_open(&st->cstore, key,
307 (mode_t) (S_IRUSR | S_IWUSR),
308 snap.size, cpath, sizeof(cpath));
309 if (fd < 0) {
310 if (err_out != NULL) { *err_out = errno ? errno : EIO; }
311 return NULL;
312 }
313 p = calloc(1, sizeof(*p));
314 o = calloc(1, sizeof(*o));
315 if (p == NULL || o == NULL) {
316 (void) close(fd);
317 free(p);
318 free(o);
319 if (err_out != NULL) { *err_out = ENOMEM; }
320 return NULL;
321 }
322 p->source = src;
323 p->cache_fd = fd;
324 p->size = snap.size;
325 p->block_size = bs;
326 p->mode = (uint32_t) (snap.mode & 0777); /* origin perms → cinfo */
327 p->mtime = (uint64_t) snap.mtime;
328 p->nblocks = brix_cache_cinfo_nblocks((uint64_t) snap.size, bs);
329 p->bitmap_len = brix_cache_cinfo_bitmap_len(p->nblocks);
330 p->log = st->log;
331 ngx_cpystrn((u_char *) p->key, (u_char *) key, sizeof(p->key));
332 ngx_cpystrn((u_char *) p->cache_path, (u_char *) cpath, sizeof(p->cache_path));
333
334 partial_capture_cred(p, cred);
335 partial_adopt_bitmap(p, cpath);
336
337 o->driver = inst->driver; /* our pread/close/fstat range-fill */
338 o->inst = inst;
339 o->fd = NGX_INVALID_FILE; /* a partial object is never sendfiled */
340 o->snap = snap;
341 o->state = p;
342 o->heap_shell = 1;
343 return o;
344}

373ngx_int_t
374sd_cache_close(brix_sd_obj_t *obj)
375{
376 sd_cache_partial_t *p = (obj != NULL) ? obj->state : NULL;
377
378 if (p != NULL) {
379 brix_sd_obj_release(p->src_obj); /* close + free heap shell if any */
380 if (p->cache_fd >= 0) {
381 (void) close(p->cache_fd);
382 }
383 free(p->bitmap);
384 free(p);
385 }
386 /* The shell is NOT freed here. driver->close releases the decorator state
387 * (source obj + cache fd + bitmap) only; the malloc'd shell (heap_shell) is
388 * owned by the caller that holds the object by pointer — brix_sd_obj_release()
389 * and the VFS adopt-fail path both do `close(o); if (o->heap_shell) free(o);`,
390 * and brix_vfs_adopt_obj() frees the original after copying it by value (and
391 * zeroes heap_shell on the embedded copy). Freeing it here double-frees the
392 * shell the moment this object is released by pointer or an adopt fails.
393 */
394 return NGX_OK;
395}
