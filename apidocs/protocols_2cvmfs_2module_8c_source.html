1/*
2 * module.c — nginx directive table, config lifecycle, and HTTP module context
3 * for the dedicated cvmfs:// protocol plane.
4 *
5 * WHAT: Three responsibilities, mirroring src/protocols/s3/module.c:
6 * 1. Config lifecycle (create_loc_conf / merge_loc_conf): allocates
7 * ngx_http_brix_cvmfs_loc_conf_t (shared preamble + cvmfs knobs),
8 * merges main→srv→loc, and when enable=1 anchors the export root,
9 * registers the composable storage backend + cache/stage tiers.
10 * 2. Handler install (ngx_http_brix_cvmfs_set): the "brix_cvmfs"
11 * directive parses its flag and installs ngx_http_brix_cvmfs_handler
12 * as the location's content handler — the location IS the protocol
13 * endpoint; no WebDAV dispatch is involved.
14 * 3. Directive table: the brix_cvmfs_* family, including the two
15 * per-protocol tier directives over the shared preamble.
16 *
17 * WHY: cvmfs:// is a first-class protocol. A CVMFS site cache is read-only
18 * by construction (allow_write is forced off) and typically a PURE cache
19 * node: no local export tree. When no root is configured the root
20 * anchors at "/" (the same pure-cache-node semantics the stream plane
21 * uses): the location serves the "/" namespace, filling from the http
22 * origin backend into the cache store.
23 *
24 * HOW: create/merge follow the s3 module's manual common.* init/merge idiom
25 * (established phase-64 gotcha: no shared_init). Postconfiguration
26 * resolves the thread pool per server exactly as s3 does — the cache
27 * fill/offload helpers need it.
28 */

93/* Parse "<lat>:<lon>" into two doubles. 0 on success. */
94static int
95cvmfs_parse_latlon(const ngx_str_t *v, double *lat, double *lon)
96{
97 char buf[64], *colon, *end;
98
99 if (v->len == 0 || v->len >= sizeof(buf)) {
100 return -1;
101 }
102 ngx_memcpy(buf, v->data, v->len);
103 buf[v->len] = '\0';
104 colon = strchr(buf, ':');
105 if (colon == NULL) {
106 return -1;
107 }
108 *colon = '\0';
109 *lat = strtod(buf, &end);
110 if (end == buf || *end != '\0' || *lat < -90.0 || *lat > 90.0) {
111 return -1;
112 }
113 *lon = strtod(colon + 1, &end);
114 if (end == colon + 1 || *end != '\0' || *lon < -180.0 || *lon > 180.0) {
115 return -1;
116 }
117 return 0;
118}

120/* brix_cvmfs_upstream_allow <host> [host ...] — append EVERY argument to
121 * the allowlist. The stock ngx_conf_set_str_array_slot keeps only the first
122 * argument per directive, so a site list written on one line silently
123 * allowed just its first Stratum-1 (observed in the field: every other
124 * Tier-1 answered 403 and clients failed over to — and pinned on — the sole
125 * surviving host). Both forms now work: one directive per host, or one
126 * directive listing them all. */
127static char *
128cvmfs_conf_upstream_allow(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
129{
130 ngx_http_brix_cvmfs_loc_conf_t *c = conf;
131 ngx_str_t *value, *slot;
132 ngx_uint_t i;
133
134 (void) cmd;
135
136 if (c->cvmfs.upstream_allow == NGX_CONF_UNSET_PTR) {
137 c->cvmfs.upstream_allow = ngx_array_create(cf->pool, 4,
138 sizeof(ngx_str_t));
139 if (c->cvmfs.upstream_allow == NULL) {
140 return NGX_CONF_ERROR;
141 }
142 }
143
144 value = cf->args->elts;
145 for (i = 1; i < cf->args->nelts; i++) {
146 slot = ngx_array_push(c->cvmfs.upstream_allow);
147 if (slot == NULL) {
148 return NGX_CONF_ERROR;
149 }
150 *slot = value[i];
151 }
152 return NGX_CONF_OK;
153}

155/* Geo mode (T19): every configured endpoint must have coordinates and
156 * brix_cvmfs_here must be set — rank once by great-circle distance and
157 * record the ranks on the backend entry (applied at instance build). */
158static char *
159cvmfs_geo_rank_config(ngx_conf_t *cf, ngx_http_brix_cvmfs_loc_conf_t *conf)
160{
161 double here_lat, here_lon;
162 double metric[SD_HTTP_EP_MAX];
163 int ranks[SD_HTTP_EP_MAX];
164 const char *host;
165 int port, idx, n;
166 brix_cvmfs_coord_t *coords;
167 ngx_uint_t i;
168
169 if (conf->cvmfs.here.len == 0
170 || cvmfs_parse_latlon(&conf->cvmfs.here, &here_lat, &here_lon) != 0)
171 {
172 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
173 "brix_cvmfs_origin_select geo requires brix_cvmfs_here "
174 "<lat>:<lon>");
175 return NGX_CONF_ERROR;
176 }
177 if (conf->cvmfs.origin_coords == NULL
178 || conf->cvmfs.origin_coords->nelts == 0)
179 {
180 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
181 "brix_cvmfs_origin_select geo requires one "
182 "brix_cvmfs_origin_coords per configured origin");
183 return NGX_CONF_ERROR;
184 }
185 coords = conf->cvmfs.origin_coords->elts;
186
187 for (n = 0; n < SD_HTTP_EP_MAX; n++) {
188 int matched = 0;
189
190 if (brix_vfs_backend_http_endpoint_at(conf->common.root_canon, n,
191 &host, &port) != 0)
192 {
193 break;
194 }
195 for (i = 0; i < conf->cvmfs.origin_coords->nelts; i++) {
196 size_t hl = ngx_strlen(host);
197
198 if (coords[i].host.len != hl
199 || ngx_strncasecmp(coords[i].host.data, (u_char *) host, hl)
200 != 0)
201 {
202 continue;
203 }
204 if (coords[i].port != 0 && (int) coords[i].port != port) {
205 continue;
206 }
207 metric[n] = brix_cvmfs_haversine_km(here_lat, here_lon,
208 coords[i].lat,
209 coords[i].lon);
210 matched = 1;
211 break;
212 }
213 if (!matched) {
214 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
215 "brix_cvmfs_origin_select geo: no brix_cvmfs_origin_coords "
216 "for origin %s:%d", host, port);
217 return NGX_CONF_ERROR;
218 }
219 }
220 if (n == 0) {
221 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
222 "brix_cvmfs_origin_select geo requires an http(s) "
223 "brix_storage_backend");
224 return NGX_CONF_ERROR;
225 }
226 for (idx = n; idx < SD_HTTP_EP_MAX; idx++) {
227 ranks[idx] = 0;
228 }
229 brix_cvmfs_rank_by_metric(metric, n, ranks);
230 brix_vfs_backend_set_http_ranks(conf->common.root_canon, ranks,
231 SD_HTTP_EP_MAX);
232
233 /* Record the computed ordering so an operator can confirm at startup that,
234 * e.g., RAL really did rank ahead of CERN from this site's coordinates.
235 * One line per origin with its great-circle distance and resulting rank.
236 * WARN, not NOTICE: at config-parse time cf->log is still the prefix log
237 * at NGX_LOG_ERR, so a NOTICE would be silently dropped and never reach
238 * `nginx -t` output or the startup log — the level that is actually seen
239 * is the point of the line. It fires once and reports a decision, not a
240 * fault. */
241 for (idx = 0; idx < n; idx++) {
242 if (brix_vfs_backend_http_endpoint_at(conf->common.root_canon, idx,
243 &host, &port) != 0)
244 {
245 break;
246 }
247 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
248 "brix_cvmfs_origin_select geo [selection report]: origin "
249 "%s:%d is %.0f km from here (%.4f:%.4f) -> rank %d%s",
250 host, port, metric[idx], here_lat, here_lon, ranks[idx],
251 (ranks[idx] == 0) ? " (preferred: reads try this origin first)"
252 : " (failover only)");
253 }
254 return NGX_CONF_OK;
255}

257/* brix_cvmfs_origin_coords <host[:port]> <lat>:<lon> — geographic position
258 * of one origin (multi). An entry with a port matches only that endpoint. */
259static char *
260ngx_http_brix_cvmfs_set_coords(ngx_conf_t *cf, ngx_command_t *cmd,
261 void *conf)
262{
263 ngx_http_brix_cvmfs_loc_conf_t *lcf = conf;
264 ngx_str_t *value = cf->args->elts;
265 brix_cvmfs_coord_t *c;
266 u_char *colon;
267
268 (void) cmd;
269 if (lcf->cvmfs.origin_coords == NGX_CONF_UNSET_PTR
270 || lcf->cvmfs.origin_coords == NULL)
271 {
272 lcf->cvmfs.origin_coords =
273 ngx_array_create(cf->pool, 4, sizeof(brix_cvmfs_coord_t));
274 if (lcf->cvmfs.origin_coords == NULL) {
275 return NGX_CONF_ERROR;
276 }
277 }
278 c = ngx_array_push(lcf->cvmfs.origin_coords);
279 if (c == NULL) {
280 return NGX_CONF_ERROR;
281 }
282
283 c->host = value[1];
284 c->port = 0;
285 colon = ngx_strlchr(value[1].data, value[1].data + value[1].len, ':');
286 if (colon != NULL) {
287 ngx_int_t p = ngx_atoi(colon + 1,
288 (size_t) (value[1].data + value[1].len
289 - (colon + 1)));
290
291 if (p < 1 || p > 65535) {
292 return "has an invalid port";
293 }
294 c->port = (in_port_t) p;
295 c->host.len = (size_t) (colon - value[1].data);
296 }
297 if (cvmfs_parse_latlon(&value[2], &c->lat, &c->lon) != 0) {
298 return "has invalid <lat>:<lon> coordinates";
299 }
300 return NGX_CONF_OK;
301}

303/*
304 * brix_cvmfs_reject_unsupported() — EMERG at config load for storage grammar
305 * that cvmfs cannot honour.
306 *
307 * WHAT: Checks three merged common-preamble fields for features that are
308 * structurally incompatible with a read-only content-addressed site cache:
309 * (1) staging — brix_stage on / brix_stage_store, (2) object slicing —
310 * brix_cache_slice_size, and (3) explicit write permission — brix_allow_write.
311 *
312 * WHY: Without an explicit rejection at config-load time these directives
313 * silently pass nginx -t, misleading operators into believing their config
314 * is active when it is not. cvmfs is structurally read-only; writes and
315 * staging have no meaning (CAS objects are immutable whole objects), and
316 * slicing never applies because the cache fill and CAS verification run on
317 * the whole object. A loud nginx -t rejection surfaces operator mistakes
318 * before traffic is served.
319 *
320 * HOW: Called at the top of the cvmfs-enabled branch in merge_loc_conf,
321 * after brix_http_common_adopt() and ngx_http_brix_shared_merge() have
322 * resolved unified directive values into conf->common (so brix_stage,
323 * brix_allow_write, and brix_cache_slice_size are visible). Returns
324 * NGX_CONF_ERROR on any violation; NGX_CONF_OK otherwise. The caller
325 * must check the return and propagate NGX_CONF_ERROR to nginx.
326 */
327static char *
328brix_cvmfs_reject_unsupported(ngx_conf_t *cf,
329 ngx_http_brix_cvmfs_loc_conf_t *conf)
330{
331 /* Staging (brix_stage on / brix_stage_store) implies mutability: a cvmfs
332 * cache is a read-through CAS store; write-back staging makes no sense. */
333 if (conf->common.stage_enable == 1 || conf->common.stage_store.len > 0) {
334 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
335 "brix_stage/brix_stage_store: cvmfs is a read-only protocol; "
336 "staging is not supported");
337 return NGX_CONF_ERROR;
338 }
339
340 /* CAS object slicing: cvmfs objects are immutable and must be filled and
341 * verified as whole units; per-slice CAS verification is undefined. */
342 if (conf->common.cache_slice_size > 0) {
343 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
344 "brix_cache_slice_size: cvmfs CAS objects are immutable whole "
345 "objects; slicing is not supported");
346 return NGX_CONF_ERROR;
347 }
348
349 /* Explicit write permission: the hard force (conf->common.allow_write = 0)
350 * below already prevents accidental writes, but an explicit on is a
351 * config mistake that deserves a loud rejection rather than silent no-op. */
352 if (conf->common.allow_write == 1) {
353 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
354 "brix_allow_write on: cvmfs is a read-only protocol; "
355 "write permission cannot be granted");
356 return NGX_CONF_ERROR;
357 }
358
359 return NGX_CONF_OK;
360}

362static char *
363ngx_http_brix_cvmfs_merge_loc_conf(ngx_conf_t *cf, void *parent, void *child)
364{
365 ngx_http_brix_cvmfs_loc_conf_t *prev = parent;
366 ngx_http_brix_cvmfs_loc_conf_t *conf = child;
367
368 /* Unified directives (brix_export, brix_cache_store, brix_cache_verify,
369 * ...) live in the common module; pull the merged values for this location
370 * into our embedded preamble before protocol merge applies defaults. */
371 brix_http_common_adopt(cf, &conf->common);
372
373 /* Merge enable first — before shared_merge — so we can distinguish a
374 * cvmfs location from a non-cvmfs one when pre-seeding the verify default
375 * below. enable merges only from prev->cvmfs.enable: it has no dependency
376 * on any common.* field set by shared_merge, so this ordering is safe. */
377 ngx_conf_merge_value(conf->cvmfs.enable, prev->cvmfs.enable, 0);
378
379 /* cvmfs default: fills are verified against their CAS SHA-1 content
380 * address. Pre-seed before shared_merge, which turns NGX_CONF_UNSET_UINT
381 * into 0 (BRIX_CACHE_VERIFY_OFF) and would otherwise suppress this default.
382 *
383 * brix_http_common_adopt above copied any user-set value from the common
384 * module into conf->common.cache_verify_mode. The common module's own
385 * merge is inheritance-only (no defaults), so if the user never touched
386 * brix_cache_verify at ANY scope, the common module's conf remains
387 * NGX_CONF_UNSET_UINT and nothing is copied — the field is still UNSET
388 * here. If the user DID set it, adopt carried the explicit value over and
389 * the guard below will not fire, respecting the operator's choice.
390 *
391 * prev->common.cache_verify_mode is intentionally NOT checked: at location
392 * level, prev is the server-level cvmfs conf, which has already been
393 * through shared_merge (setting it to 0), so checking prev would never
394 * allow the default to apply. The UNSET check on conf is sufficient.
395 *
396 * The guard on cvmfs.enable confines this to cvmfs locations only. */
397 if (conf->cvmfs.enable == 1
398 && conf->common.cache_verify_mode == NGX_CONF_UNSET_UINT)
399 {
400 conf->common.cache_verify_mode = BRIX_CACHE_VERIFY_CVMFS_CAS;
401 }
402
403 /* Shared common.* preamble. Two deliberate gains over the old manual
404 * block: read_only now forces allow_write off (hardening; cvmfs is a
405 * read path anyway), and common.ktls merges to its default — inert here
406 * (nothing under protocols/cvmfs/ reads it). */
407 if (ngx_http_brix_shared_merge(cf, &prev->common, &conf->common, "")
408 != NGX_CONF_OK)
409 {
410 return NGX_CONF_ERROR;
411 }
412 ngx_conf_merge_sec_value(conf->cvmfs.manifest_ttl, prev->cvmfs.manifest_ttl,
413 61);
414 ngx_conf_merge_sec_value(conf->cvmfs.negative_ttl, prev->cvmfs.negative_ttl,
415 10);
416 ngx_conf_merge_str_value(conf->cvmfs.quarantine_dir,
417 prev->cvmfs.quarantine_dir, "");
418 ngx_conf_merge_ptr_value(conf->cvmfs.upstream_allow,
419 prev->cvmfs.upstream_allow, NULL);
420 ngx_conf_merge_uint_value(conf->cvmfs.upstream_max, prev->cvmfs.upstream_max,
421 8);
422 ngx_conf_merge_uint_value(conf->cvmfs.origin_select,
423 prev->cvmfs.origin_select,
424 BRIX_CVMFS_SELECT_RTT);
425 ngx_conf_merge_ptr_value(conf->cvmfs.origin_coords,
426 prev->cvmfs.origin_coords, NULL);
427 ngx_conf_merge_str_value(conf->cvmfs.here, prev->cvmfs.here, "");
428 ngx_conf_merge_sec_value(conf->cvmfs.rtt_interval, prev->cvmfs.rtt_interval,
429 60);
430 ngx_conf_merge_sec_value(conf->cvmfs.client_hold, prev->cvmfs.client_hold,
431 25);
432 ngx_conf_merge_sec_value(conf->cvmfs.fill_max_life,
433 prev->cvmfs.fill_max_life, 300);
434 ngx_conf_merge_value(conf->cvmfs.trace, prev->cvmfs.trace, 0);
435 /* Trace promotion is process-wide (the shared origin transport has no
436 * per-location handle): any cvmfs location turning it on promotes the
437 * upstream-request lines to INFO for every worker (set pre-fork). */
438 if (conf->cvmfs.trace) {
439 brix_origin_trace_set(1);
440 }
441
442 /* Upstream stall detection + force-through retry (2026-07-03). The transport
443 * bounds and the force-primary toggle are process-wide operator policy (the
444 * shared curl transport / sd_http driver have no per-location handle), set
445 * pre-fork here beside the trace flag so every worker inherits them. */
446 ngx_conf_merge_sec_value(conf->cvmfs.origin_connect_timeout,
447 prev->cvmfs.origin_connect_timeout, 2);
448 ngx_conf_merge_sec_value(conf->cvmfs.origin_stall_timeout,
449 prev->cvmfs.origin_stall_timeout, 4);
450 ngx_conf_merge_uint_value(conf->cvmfs.origin_stall_bytes,
451 prev->cvmfs.origin_stall_bytes, 1);
452 ngx_conf_merge_sec_value(conf->cvmfs.origin_attempt_timeout,
453 prev->cvmfs.origin_attempt_timeout, 0);
454 ngx_conf_merge_value(conf->cvmfs.origin_reuse_conn,
455 prev->cvmfs.origin_reuse_conn, 1);
456 ngx_conf_merge_uint_value(conf->cvmfs.fill_retry_policy,
457 prev->cvmfs.fill_retry_policy,
458 BRIX_CVMFS_RETRY_FAILOVER);
459 ngx_conf_merge_value(conf->cvmfs.shared_cache, prev->cvmfs.shared_cache, 0);
460 ngx_conf_merge_value(conf->cvmfs.unified_origin, prev->cvmfs.unified_origin,
461 0);
462 /* unified_origin serves every proxy request from the location's configured
463 * origin backend, so that backend MUST be an http(s) origin set (ideally a
464 * "|"-separated multi-endpoint one for the failover that hides a dead
465 * origin). Fail loudly at config time rather than 500 per request. */
466 if (conf->cvmfs.enable && conf->cvmfs.unified_origin
467 && (conf->common.storage_backend.len < 4
468 || ngx_strncasecmp(conf->common.storage_backend.data,
469 (u_char *) "http", 4) != 0))
470 {
471 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
472 "brix_cvmfs_unified_origin on requires brix_storage_backend "
473 "to name an http(s) origin set, e.g. "
474 "\"http://s1a:8000|http://s1b:8000|http://s1c:8000\" (the '|'-list "
475 "is the ranked failover set that hides a dead Stratum-1)");
476 return NGX_CONF_ERROR;
477 }
478 if (conf->cvmfs.enable) {
479 brix_s3_origin_timeouts_set(
480 (long) conf->cvmfs.origin_connect_timeout * 1000,
481 (long) conf->cvmfs.origin_stall_timeout,
482 (long) conf->cvmfs.origin_stall_bytes,
483 (long) conf->cvmfs.origin_attempt_timeout * 1000);
484 brix_s3_origin_reuse_set(conf->cvmfs.origin_reuse_conn ? 1 : 0);
485 if (conf->cvmfs.fill_retry_policy == BRIX_CVMFS_RETRY_FORCE_PRIMARY) {
486 sd_http_force_primary_set(1);
487 }
488 }
489
490 /* Server-side geo answering (2026-07-03). */
491 ngx_conf_merge_uint_value(conf->cvmfs.geo_answer, prev->cvmfs.geo_answer,
492 BRIX_CVMFS_GEO_PASSTHROUGH);
493 ngx_conf_merge_sec_value(conf->cvmfs.geo_cache_ttl, prev->cvmfs.geo_cache_ttl,
494 60);
495 ngx_conf_merge_uint_value(conf->cvmfs.geo_max_servers,
496 prev->cvmfs.geo_max_servers, 16);
497
498 ngx_conf_merge_value(conf->scvmfs, prev->scvmfs, 0);
499 ngx_conf_merge_uint_value(conf->scvmfs_authz, prev->scvmfs_authz,
500 BRIX_SCVMFS_AUTHZ_NONE);
501 ngx_conf_merge_str_value(conf->scvmfs_token_issuers,
502 prev->scvmfs_token_issuers, "");
503 if (conf->scvmfs_registry == NULL) {
504 conf->scvmfs_registry = prev->scvmfs_registry;
505 }
506
507 /* scvmfs (T22, EXPERIMENTAL) is a LAYER on cvmfs — structural, checked
508 * at config time. Bearer mode needs the issuer registry to exist. */
509 if (conf->scvmfs) {
510 if (!conf->cvmfs.enable) {
511 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
512 "brix_scvmfs requires brix_cvmfs on");
513 return NGX_CONF_ERROR;
514 }
515 if (conf->scvmfs_authz == BRIX_SCVMFS_AUTHZ_BEARER) {
516 if (conf->scvmfs_token_issuers.len == 0) {
517 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
518 "brix_scvmfs_authz bearer requires "
519 "brix_scvmfs_token_issuers <scitokens.cfg>");
520 return NGX_CONF_ERROR;
521 }
522 if (conf->scvmfs_registry == NULL) {
523 brix_token_registry_t *reg = NULL;
524
525 if (brix_token_registry_build(cf,
526 (const char *) conf->scvmfs_token_issuers.data,
527 BRIX_AUTHZ_CAPABILITY, &reg) != NGX_OK)
528 {
529 return NGX_CONF_ERROR;
530 }
531 conf->scvmfs_registry = reg;
532 }
533 }
534 }
535
536 if (conf->cvmfs.enable) {
537 brix_export_root_opts_t root_opts;
538
539 /* Reject storage grammar cvmfs cannot honour before doing anything
540 * else: staging, slicing, and explicit writes make no sense for a
541 * read-only content-addressed site cache. */
542 if (brix_cvmfs_reject_unsupported(cf, conf) != NGX_CONF_OK) {
543 return NGX_CONF_ERROR;
544 }
545
546 /* CVMFS is a read-only protocol: no directive can enable writes. */
547 conf->common.allow_write = 0;
548
549 /* "posix:<path>" backend names the local export tree (composable
550 * brix_root replacement) — same rewrite every protocol applies. */
551 brix_storage_backend_posix_root(&conf->common);
552
553 /* Pure cache node (the normal CVMFS shape): no local export tree —
554 * anchor the namespace at "/" exactly like the stream plane's pure
555 * cache node; the location serves the "/" namespace, filling from
556 * the http backend into the cache store. */
557 if (conf->common.root.len == 0) {
558 ngx_str_set(&conf->common.root, "/");
559 }
560
561 root_opts.directive_name = "brix_cvmfs";
562 root_opts.allow_write = 0;
563 root_opts.required = 1;
564 root_opts.canon_size = sizeof(conf->common.root_canon);
565 if (brix_prepare_export_root(cf, &conf->common.root, &root_opts,
566 conf->common.root_canon) != NGX_CONF_OK)
567 {
568 return NGX_CONF_ERROR;
569 }
570
571 /* Persistent confinement rootfd (openat2 RESOLVE_BENEATH anchor). */
572 if (brix_http_open_rootfd(cf, &conf->common) != NGX_CONF_OK) {
573 return NGX_CONF_ERROR;
574 }
575
576 /* Register the composable storage backend (phase-63): the http(s)
577 * Stratum-1 origin URL routes every VFS op to sd_http. */
578 if (brix_vfs_backend_config_str(cf, conf->common.root_canon,
579 &conf->common.storage_backend, conf->common.pblock_block_size,
580 BRIX_AF_AUTO)
581 != NGX_OK)
582 {
583 return NGX_CONF_ERROR;
584 }
585
586 /* Verify-mismatch evidence lands in the protocol's quarantine dir;
587 * MANIFEST-class fills get the protocol's TTL stamped (T12). */
588 conf->common.cache_quarantine_dir = conf->cvmfs.quarantine_dir;
589 conf->common.cache_manifest_ttl = conf->cvmfs.manifest_ttl;
590 /* T20 never-drop deadlines for the shared fill machinery */
591 conf->common.cache_client_hold = conf->cvmfs.client_hold;
592 conf->common.cache_fill_max_life = conf->cvmfs.fill_max_life;
593
594 /* Phase-64: compose the cache/stage tiers over the backend. */
595 if (brix_tier_register_stores(cf, &conf->common) != NGX_OK) {
596 return NGX_CONF_ERROR;
597 }
598
599 /* T19 origin selection: geo ranks compute once at config time;
600 * rtt registers the per-worker probe; static keeps configured
601 * order (all ranks 0 — the pick is order-stable on ties). */
602 if (conf->cvmfs.origin_select == BRIX_CVMFS_SELECT_GEO) {
603 if (cvmfs_geo_rank_config(cf, conf) != NGX_CONF_OK) {
604 return NGX_CONF_ERROR;
605 }
606 } else if (conf->cvmfs.origin_select == BRIX_CVMFS_SELECT_RTT) {
607 brix_cvmfs_rtt_register(conf->common.root_canon,
608 conf->cvmfs.rtt_interval,
609 &conf->common.thread_pool_name);
610 }
611
612 /* WARN (not NOTICE — config-parse NOTICE is dropped at cf->log ERR
613 * level) when coordinates were supplied but origin_select is not geo:
614 * the coordinates are silently ignored in all other modes and the
615 * operator may not realise their geo config has no effect. */
616 if (conf->cvmfs.origin_select != BRIX_CVMFS_SELECT_GEO
617 && conf->cvmfs.origin_coords != NULL)
618 {
619 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
620 "brix_cvmfs_origin_coords set but brix_cvmfs_origin_select "
621 "is not geo — coordinates are ignored");
622 }
623 }
624
625 return NGX_CONF_OK;
626}

734/*
735 * Post-config: resolve the async-I/O thread pool per server (fill/offload
736 * helpers need it) — identical mechanics to the s3 module.
737 */
738static ngx_int_t
739ngx_http_brix_cvmfs_postconfiguration(ngx_conf_t *cf)
740{
741 ngx_http_core_main_conf_t *cmcf;
742 ngx_http_core_srv_conf_t **cscfp;
743 ngx_http_brix_cvmfs_loc_conf_t *lcf;
744 static ngx_str_t default_pool_name = ngx_string("default");
745 ngx_str_t *pool_name;
746 ngx_uint_t s;
747
748 /* An HTTP-only cvmfs node has no stream block, so the SHM zones normally
749 * created by the stream postconfiguration must be ensured here: the
750 * metrics table (or every counter INC is a silent no-op) AND the
751 * dashboard transfer/events/history zones (or every live-transfer
752 * record silently fails and the dashboard stays empty). Both are
753 * idempotent — ngx_shared_memory_add returns an existing zone. */
754 if (brix_metrics_ensure_zone(cf) != NGX_OK) {
755 return NGX_ERROR;
756 }
757 if (brix_configure_dashboard(cf) != NGX_OK) {
758 return NGX_ERROR;
759 }
760
761 cmcf = ngx_http_conf_get_module_main_conf(cf, ngx_http_core_module);
762 cscfp = cmcf->servers.elts;
763
764 for (s = 0; s < cmcf->servers.nelts; s++) {
765 ngx_http_conf_ctx_t *ctx = cscfp[s]->ctx;
766
767 lcf = ctx->loc_conf[ngx_http_brix_cvmfs_module.ctx_index];
768 if (lcf == NULL || !lcf->cvmfs.enable) {
769 continue;
770 }
771
772 pool_name = (lcf->common.thread_pool_name.len > 0)
773 ? &lcf->common.thread_pool_name
774 : &default_pool_name;
775
776 lcf->common.thread_pool = ngx_thread_pool_get(cf->cycle, pool_name);
777 if (lcf->common.thread_pool == NULL) {
778 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
779 "brix_cvmfs: thread pool \"%V\" not found - "
780 "async cache fills disabled (add a thread_pool directive)",
781 pool_name);
782 }
783 }
784
785 return NGX_OK;
786}
