1/*
2 * module.c — nginx directive table, config lifecycle, and HTTP module context
3 * for the dedicated cvmfs:// protocol plane.
4 *
5 * WHAT: The module's nginx wiring, mirroring src/protocols/s3/module.c:
6 * 1. Config lifecycle head: create_loc_conf allocates
7 * ngx_http_brix_cvmfs_loc_conf_t (shared preamble + cvmfs knobs) and
8 * sets the UNSET sentinels; the merge is delegated to sibling files
9 * (see below).
10 * 2. Handler install (ngx_http_brix_cvmfs_set): the "brix_cvmfs"
11 * directive parses its flag and installs ngx_http_brix_cvmfs_handler
12 * as the location's content handler — the location IS the protocol
13 * endpoint; no WebDAV dispatch is involved.
14 * 3. Directive table: the brix_cvmfs_* family (split into the two
15 * directives_*.inc fragments) and its enums.
16 * 4. nginx $cvmfs_* variables + pre/post-configuration + init-process
17 * and the ngx_module_t definition.
18 *
19 * WHY: cvmfs:// is a first-class protocol. A CVMFS site cache is read-only
20 * by construction (allow_write is forced off) and typically a PURE cache
21 * node: no local export tree. When no root is configured the root
22 * anchors at "/" (the same pure-cache-node semantics the stream plane
23 * uses): the location serves the "/" namespace, filling from the http
24 * origin backend into the cache store.
25 *
26 * HOW: create follows the s3 module's manual common.* init idiom (established
27 * phase-64 gotcha: no shared_init). The location merge grew past the
28 * file-size gate and is split across three siblings, wired here through
29 * cvmfs_module_internal.h: cvmfs_module_merge.c owns the merge-field
30 * helpers + the ngx_http_brix_cvmfs_merge_loc_conf orchestrator installed
31 * in the module context; cvmfs_module_build.c owns the enabled-branch
32 * export/backend/cache build; cvmfs_module_georank.c owns the geographic
33 * origin-ranking and coordinate/allow-list directive setters referenced by
34 * the directive table. Postconfiguration resolves the thread pool per
35 * server exactly as s3 does — the cache fill/offload helpers need it.
36 */

107static ngx_int_t
108cvmfs_var_set(ngx_http_request_t *r, ngx_http_variable_value_t *v,
109 const char *val)
110{
111 size_t n = ngx_strlen(val);
112 u_char *p = ngx_pnalloc(r->pool, n);
113
114 if (p == NULL) {
115 return NGX_ERROR;
116 }
117 ngx_memcpy(p, val, n);
118 v->data = p;
119 v->len = n;
120 v->valid = 1;
121 v->no_cacheable = 0;
122 v->not_found = 0;
123 return NGX_OK;
124}

126static ngx_int_t

129{

132 static const char *names[] = { "cas", "manifest", "geo", "reject" };

136 return cvmfs_var_set(r, v, "-");

156static ngx_int_t
157cvmfs_var_origin(ngx_http_request_t *r, ngx_http_variable_value_t *v,
158 uintptr_t data)
159{
160 ngx_http_brix_cvmfs_loc_conf_t *lcf =
161 ngx_http_get_module_loc_conf(r, ngx_http_brix_cvmfs_module);
162 ngx_http_brix_cvmfs_ctx_t *ctx =
163 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
164 brix_sd_instance_t *inst;
165 char buf[300];
166 const char *root;
167
168 (void) data;
169 if (ctx == NULL || ctx->cache_status != BRIX_CVMFS_CACHE_FILL
170 || lcf == NULL)
171 {
172 return cvmfs_var_set(r, v, "-");
173 }
174 root = (ctx->up_root != NULL) ? ctx->up_root : lcf->common.root_canon;
175 inst = brix_vfs_backend_resolve(root, r->connection->log);
176 while (inst != NULL && ngx_strcmp(inst->driver->name, "http") != 0) {
177 inst = brix_sd_cache_source_instance(inst);
178 }
179 if (inst == NULL || sd_http_last_origin(inst, buf, sizeof(buf)) != 0) {
180 return cvmfs_var_set(r, v, "-");
181 }
182 return cvmfs_var_set(r, v, buf);
183}

192static ngx_int_t
193ngx_http_brix_cvmfs_preconfiguration(ngx_conf_t *cf)
194{
195 ngx_http_variable_t *v, *nv;
196
197 for (v = ngx_http_brix_cvmfs_vars; v->name.len; v++) {
198 nv = ngx_http_add_variable(cf, &v->name, v->flags);
199 if (nv == NULL) {
200 return NGX_ERROR;
201 }
202 nv->get_handler = v->get_handler;
203 nv->data = v->data;
204 }
205 return NGX_OK;

208/*
209 * Post-config: resolve the async-I/O thread pool per server (fill/offload
210 * helpers need it) — identical mechanics to the s3 module.
211 */
212static ngx_int_t
213ngx_http_brix_cvmfs_postconfiguration(ngx_conf_t *cf)
214{
215 ngx_http_core_main_conf_t *cmcf;
216 ngx_http_core_srv_conf_t **cscfp;
217 ngx_http_brix_cvmfs_loc_conf_t *lcf;
218 static ngx_str_t default_pool_name = ngx_string("default");
219 ngx_str_t *pool_name;
220 ngx_uint_t s;
221
222 /* An HTTP-only cvmfs node has no stream block, so the SHM zones normally
223 * created by the stream postconfiguration must be ensured here: the
224 * metrics table (or every counter INC is a silent no-op) AND the
225 * dashboard transfer/events/history zones (or every live-transfer
226 * record silently fails and the dashboard stays empty). Both are
227 * idempotent — ngx_shared_memory_add returns an existing zone. */
228 if (brix_metrics_ensure_zone(cf) != NGX_OK) {
229 return NGX_ERROR;
230 }
231 if (brix_configure_dashboard(cf) != NGX_OK) {
232 return NGX_ERROR;
233 }
234
235 cmcf = ngx_http_conf_get_module_main_conf(cf, ngx_http_core_module);
236 cscfp = cmcf->servers.elts;
237
238 for (s = 0; s < cmcf->servers.nelts; s++) {
239 ngx_http_conf_ctx_t *ctx = cscfp[s]->ctx;
240
241 lcf = ctx->loc_conf[ngx_http_brix_cvmfs_module.ctx_index];
242 if (lcf == NULL || !lcf->cvmfs.enable) {
243 continue;
244 }
245
246 pool_name = (lcf->common.thread_pool_name.len > 0)
247 ? &lcf->common.thread_pool_name
248 : &default_pool_name;
249
250 lcf->common.thread_pool = ngx_thread_pool_get(cf->cycle, pool_name);
251 if (lcf->common.thread_pool == NULL) {
252 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
253 "brix_cvmfs: thread pool \"%V\" not found - "
254 "async cache fills disabled (add a thread_pool directive)",
255 pool_name);
256 }
257 }
258
259 return NGX_OK;
260}
