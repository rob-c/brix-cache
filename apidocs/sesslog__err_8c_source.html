7/*
8 * WHAT: One (numeric code -> stable token) mapping row.
9 * WHY: The three err-from-* families share an identical shape — a small fixed
10 * set of codes collapsing onto a low-cardinality token — so expressing each as
11 * data (a table) rather than a switch ladder removes the per-family branching
12 * complexity while keeping the emitted strings byte-identical.
13 * HOW: Plain value struct scanned linearly by brix_sess_err_lookup().
14 */
15typedef struct {
16 int code;
17 const char *token;
18} brix_sess_err_entry_t;
19
20/*
21 * WHAT: Resolve a numeric code against a fixed mapping table, returning the
22 * matched token or a formatted "code:<n>" fallback.
23 * WHY: All three public err-from-* accessors want the same lookup-then-fallback
24 * behaviour; centralising it keeps the fallback formatting (and the scratch/NULL
25 * contract) in exactly one place.
26 * HOW: (1) Linearly scan the table for an exact code match and return its token.
27 * (2) On no match, write "code:<code>" into scratch when a buffer is supplied,
28 * otherwise return the static "code:0" sentinel — matching the prior default arm.
29 */
30static const char *
31brix_sess_err_lookup(const brix_sess_err_entry_t *table, size_t count,
32 int code, char *scratch, size_t n)
33{
34 size_t i;
35
36 for (i = 0; i < count; i++) {
37 if (table[i].code == code) {
38 return table[i].token;
39 }
40 }
41
42 if (scratch != NULL && n > 0) {
43 snprintf(scratch, n, "code:%d", code);
44 return scratch;
45 }
46
47 return "code:0";
48}
