7/*
8 * WHAT: Bound-secondary lazy-open relay cluster — handle the synthetic
9 * kXR_open response, translate the upstream fhandle into the reserved
10 * local slot (or free it on failure), chain any further pending lazy
11 * opens, and dispatch the saved read/readv that triggered the open.
12 *
13 * WHY: A bound secondary defers its kXR_open until the first read arrives.
14 * When the upstream open completes the proxy must resolve the local↔
15 * upstream fhandle mapping, re-issue the queued request with the
16 * translated handle, and resume the client read loop — all transparently.
17 *
18 * HOW: brix_proxy_relay_lazy_open() is invoked from
19 * brix_proxy_relay_to_client() (forward_relay_response.c). It records the
20 * open, releases the response body, drives brix_proxy_lazy_open_next() for
21 * any remaining reserved handles, then brix_proxy_dispatch_saved_read()
22 * to forward the deferred read with fhandle translation applied.
23 */

93brix_proxy_translate_saved_readv(brix_proxy_ctx_t *proxy, u_char *rreq,
94 size_t rlen)
95{
96 u_char *pl = rreq + XRD_REQUEST_HDR_LEN;
97 size_t pos = 0;
98 size_t pdlen = rlen > XRD_REQUEST_HDR_LEN ? rlen - XRD_REQUEST_HDR_LEN : 0;
99
100 while (pos + 16 <= pdlen) {
101 int cfh = (int)(unsigned char) pl[pos];
102 if (cfh >= 0 && cfh < BRIX_MAX_FILES
103 && proxy->fh_map[cfh].upstream_fh >= 0)
104 {
105 pl[pos] = (u_char)(unsigned int) proxy->fh_map[cfh].upstream_fh;
106 }
107 pos += 16;
108 }

113brix_proxy_translate_saved_read(brix_proxy_ctx_t *proxy, u_char *rreq,
114 size_t rlen, int local_fh)
115{
116 uint16_t saved_rid;
117 int upstream_fh;
118
119 saved_rid = ntohs(((ClientRequestHdr *)(void *) rreq)->requestid);
120 if (saved_rid == kXR_read || saved_rid == kXR_pgread) {
121 upstream_fh = (local_fh >= 0 && local_fh < BRIX_MAX_FILES)
122 ? proxy->fh_map[local_fh].upstream_fh : -1;
123 if (upstream_fh >= 0) {
124 rreq[4] = (u_char)(unsigned int) upstream_fh;
125 }
126 return local_fh;
127 }
128 if (saved_rid == kXR_readv) {
129 return brix_proxy_translate_saved_readv(proxy, rreq, rlen);
130 }
131 return local_fh;

210/* relay upstream response to client */
211/* Handle the synthetic kXR_open response for a bound-secondary lazy open:
212 * translate the upstream fhandle into the reserved local slot (or free it on
213 * failure) and resume the client read loop. Returns 1 when this was a lazy-
214 * open response and the caller must return; 0 otherwise. */
215int
216brix_proxy_relay_lazy_open(brix_proxy_ctx_t *proxy)
217{
218 brix_ctx_t *ctx = proxy->client_ctx;
219 ngx_connection_t *c = proxy->client_conn;
220 int local_fh = proxy->fwd_local_fh;
221
222 if (!proxy->fwd_is_lazy_open) {
223 return 0;
224 }
225
226 proxy->fwd_is_lazy_open = 0;
227
228 if (proxy->resp_status != kXR_ok) {
229 brix_proxy_lazy_open_failed(proxy, ctx, c, local_fh);
230 return 1;
231 }
232
233 brix_proxy_lazy_record_open(proxy, local_fh);
234 brix_proxy_lazy_release_body(proxy);
235
236 if (brix_proxy_lazy_open_next(proxy, ctx, c)) {
237 return 1;
238 }
239
240 if (brix_proxy_dispatch_saved_read(proxy, c)) {
241 return 1;
242 }
243
244 proxy->state = XRD_PX_IDLE;
245 ctx->state = XRD_ST_REQ_HEADER;
246 brix_schedule_read_resume(c);
247 return 1;
248}
