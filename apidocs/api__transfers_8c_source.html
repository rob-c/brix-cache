8/*
9 * WHAT: Point-in-time snapshot of one SHM transfer slot.
10 * WHY: Slots live in cross-process SHM and are mutated by worker processes
11 * without locking. Copying every field into this file-local struct BEFORE
12 * any jansson allocation prevents a field changing mid-build (torn rows)
13 * and avoids holding SHM pointers across allocator calls. Passing the
14 * struct by pointer keeps every JSON-stage helper below to <=5 params
15 * without reintroducing globals.
16 * NOTE: char[] copies mirror the SHM field sizes and are always NUL-terminated
17 * since SHM strings may be unterminated if truncated. The derived scalars
18 * (idle_ms, avg_bps, instant_bps) are computed once here.
19 */
20typedef struct {
21 int64_t last_ms;
22 int64_t bytes;
23 int64_t start_ms;
24 int64_t idle_ms;
25 uint64_t avg_bps;
26 uint64_t instant_bps;
27 char client_ip[BRIX_DASHBOARD_IP_LEN];
28 char identity[BRIX_DASHBOARD_IDENTITY_LEN];
29 char vo[BRIX_DASHBOARD_VO_LEN];
30 char path[BRIX_DASHBOARD_PATH_LEN];
31 char op[BRIX_DASHBOARD_OP_LEN];
32 char last_error[BRIX_DASHBOARD_REASON_LEN];
33 char remote_host[BRIX_DASHBOARD_HOST_LEN];
34 char remote_path[BRIX_DASHBOARD_PATH_LEN];
35} dashboard_xfer_snapshot_t;

38/*
39 * WHAT: Decay the published EWMA instant rate for read-only display.
40 * WHY: The owning worker only re-folds slot->instant_bps while bytes flow
41 * (transfer_table.c), so a transfer sitting in an inter-burst gap would
42 * otherwise show its last burst rate frozen. Fold in the zero-rate sample
43 * intervals elapsed since the last worker sample (same alpha = 1/4) so the
44 * shown rate eases toward zero during the gap.
45 * HOW: Purely local — never writes the SHM slot. The loop is bounded:
46 * instant_bps reaches 0 within ~log_{4/3} iterations.
47 */
48static uint64_t
49dashboard_decay_instant_bps(const brix_transfer_slot_t *slot, int64_t now_ms)
50{
51 uint64_t instant_bps = (uint64_t) slot->instant_bps;
52 int64_t sample_ms = (int64_t) slot->last_sample_ms;
53 int64_t missed = (sample_ms > 0 && now_ms > sample_ms)
54 ? (now_ms - sample_ms) / BRIX_XFER_SAMPLE_MS : 0;
55
56 while (missed-- > 0 && instant_bps > 0) {
57 instant_bps = (instant_bps * 3) / 4; /* EWMA fold with raw = 0 */
58 }
59 return instant_bps;
60}

63/*
64 * WHAT: NUL-terminate one fixed-size char[] copied from an SHM string.
65 * WHY: SHM strings may be unterminated if truncated; every string field must be
66 * forced-terminated after the ngx_memcpy before jansson touches it.

68static void
69dashboard_snap_str(char *dst, const char *src, size_t size)
70{
71 ngx_memcpy(dst, src, size);
72 dst[size - 1] = '\0';

76/*
77 * WHAT: Copy an SHM transfer slot into a stable local snapshot.
78 * WHY: See dashboard_xfer_snapshot_t — this is the barriered, one-shot capture
79 * that decouples the JSON build from concurrent worker mutation.
80 * HOW: Issue a memory barrier, copy scalars, derive idle_ms/avg_bps/instant_bps,
81 * then copy and terminate each string field.
82 */
83static void
84dashboard_snapshot_slot(dashboard_xfer_snapshot_t *snap,
85 const brix_transfer_slot_t *slot, int64_t now_ms)
86{
87 /* Barrier: ensure subsequent reads see a coherent view of the SHM slot
88 * rather than reordered/cached fields from a concurrent writer. */
89 ngx_memory_barrier();
90
91 snap->last_ms = (int64_t) slot->last_ms;
92 snap->bytes = (int64_t) slot->bytes;
93 snap->start_ms = slot->start_ms;
94 snap->idle_ms = (snap->last_ms > 0 && now_ms >= snap->last_ms)
95 ? now_ms - snap->last_ms : 0;
96 snap->avg_bps = dashboard_avg_bps(snap->bytes, snap->start_ms, now_ms);
97 snap->instant_bps = dashboard_decay_instant_bps(slot, now_ms);
98
99 dashboard_snap_str(snap->client_ip, slot->client_ip, sizeof(snap->client_ip));
100 dashboard_snap_str(snap->identity, slot->identity, sizeof(snap->identity));
101 dashboard_snap_str(snap->vo, slot->vo, sizeof(snap->vo));
102 dashboard_snap_str(snap->path, slot->path, sizeof(snap->path));
103 dashboard_snap_str(snap->op, slot->op, sizeof(snap->op));
104 dashboard_snap_str(snap->last_error, slot->last_error, sizeof(snap->last_error));
105 dashboard_snap_str(snap->remote_host, slot->tpc_remote_host, sizeof(snap->remote_host));
106 dashboard_snap_str(snap->remote_path, slot->tpc_remote_path_hint, sizeof(snap->remote_path));
107}

214/*
215 * WHAT: Serialise one transfer slot into a JSON object.
216 * WHY: Snapshots the SHM slot once (dashboard_snapshot_slot), then emits each
217 * cohesive field group through a dedicated stage helper. This keeps the
218 * point-in-time-consistency guarantee (no torn rows, no SHM pointer held
219 * across allocation) while staying under the complexity budget.
220 * NOTE: v1_fields gates the richer v1 schema; detail_fields adds per-op counters
221 * and the session hash (only the single-transfer detail endpoint sets it).
222 */
223json_t *
224dashboard_build_transfer_object(
225 const ngx_http_brix_dashboard_loc_conf_t *conf,
226 const brix_transfer_slot_t *slot, int64_t now_ms,
227 ngx_uint_t v1_fields, ngx_uint_t detail_fields, ngx_uint_t redact)
228{
229 dashboard_xfer_snapshot_t snap;
230 json_t *obj;
231
232 ngx_memzero(&snap, sizeof(snap));
233 dashboard_snapshot_slot(&snap, slot, now_ms);
234
235 obj = json_object();
236 if (!obj) { return NULL; }
237
238 dashboard_emit_base_fields(obj, slot, &snap, redact);
239
240 /* Compat (/xrootd/transfers) shape stops here; v1 adds the fields below. */
241 if (!v1_fields) { return obj; }
242
243 dashboard_emit_v1_fields(obj, conf, slot, &snap, redact);
244 dashboard_emit_tpc_object(obj, slot, &snap, redact);
245
246 if (detail_fields) {
247 dashboard_emit_detail_fields(obj, slot);
248 }
249
250 return obj;
251}

254json_t *
255dashboard_build_transfer_rows(int64_t now_ms,
256 const ngx_http_brix_dashboard_loc_conf_t *conf, ngx_uint_t v1_fields,
257 ngx_uint_t redact)
258{
259 brix_transfer_table_t *tbl;
260 json_t *arr;
261 ngx_uint_t i;
262
263 arr = json_array();
264 if (!arr) { return NULL; }
265
266 if (ngx_brix_dashboard_shm_zone == NULL
267 || ngx_brix_dashboard_shm_zone->data == NULL
268 || ngx_brix_dashboard_shm_zone->data == (void *) 1)
269 {
270 return arr;
271 }
272
273 tbl = ngx_brix_dashboard_shm_zone->data;
274 for (i = 0; i < BRIX_DASHBOARD_MAX_TRANSFERS; i++) {
275 brix_transfer_slot_t *slot = &tbl->slots[i];
276 int64_t last_ms;
277 json_t *obj;
278
279 if (slot->in_use == 0) { continue; }
280
281 /* Opportunistic GC: a slot whose last activity is older than STALE_GC_MS
282 * almost certainly leaked (worker died without freeing it). Reclaim it
283 * here on the read path — there is no separate sweeper — and record an
284 * event so the cleanup is visible, then skip emitting the dead row. */
285 last_ms = (int64_t) slot->last_ms;
286 if (last_ms > 0 && now_ms - last_ms > STALE_GC_MS) {
287 brix_dashboard_event_add(BRIX_DASH_EVENT_DASHBOARD,
288 slot->proto, 0,
289 "stale active transfer cleaned up",
290 slot->path);
291 brix_transfer_slot_free(tbl, (int) i);
292 continue;
293 }
294
295 obj = dashboard_build_transfer_object(conf, slot, now_ms, v1_fields, 0, redact);
296 if (obj) { json_array_append_new(arr, obj); }
297 }
298 return arr;
299}

365/*
366 * Parse the numeric transfer id from "/brix/api/v1/transfers/<id>".
367 * Strict: the tail must be present and consist solely of decimal digits — any
368 * non-digit (path traversal, trailing slash, query) yields NGX_DECLINED so the
369 * caller returns 400 rather than risking a misparse.
370 */
371ngx_int_t
372dashboard_parse_detail_id(ngx_http_request_t *r, uint32_t *id)
373{
374 const char *prefix = "/brix/api/v1/transfers/";
375 size_t prefix_len = sizeof("/brix/api/v1/transfers/") - 1;
376 uint32_t value = 0;
377 size_t i;
378
379 if (r->uri.len <= prefix_len
380 || ngx_memcmp(r->uri.data, prefix, prefix_len) != 0)
381 {
382 return NGX_DECLINED;
383 }
384
385 /* Accumulate base-10; reject on the first non-digit byte. */
386 for (i = prefix_len; i < r->uri.len; i++) {
387 u_char c = r->uri.data[i];
388 if (c < '0' || c > '9') { return NGX_DECLINED; }
389 value = value * 10u + (uint32_t) (c - '0');
390 }
391
392 *id = value;
393 return NGX_OK;
394}
