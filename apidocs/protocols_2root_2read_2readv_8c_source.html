20/* ---- File-local kXR_readv request working state ----
21 *
22 * WHAT: Bundles the per-request fields shared by brix_handle_readv's helper
23 * steps (config, wire view, segment count, per-element cap, and the two
24 * allocations once they exist). Not a wire structure — purely in-process.
25 *
26 * WHY: The validate / build-descriptors / async-post / sync-execute steps each
27 * need most of these fields; threading one struct keeps helper parameter
28 * counts within the gate and makes the data flow explicit.
29 *
30 * HOW: The orchestrator zero-inits it, fills the constant fields, then hands a
31 * pointer to each step; steps that allocate (response_buffer, segment_descs)
32 * write back into the same struct.
33 */

35 ngx_stream_brix_srv_conf_t *rconf;
36 readahead_list *wire_segments;
37 size_t segment_count;
38 size_t readv_seg_max;
39 size_t max_response_bytes;
40 u_char *response_buffer;

44/* ---- Validate handles and compute the response upper bound ----
45 *
46 * WHAT: First pass over the wire segments: validates every file handle and
47 * accumulates req->max_response_bytes (capping each element at readv_seg_max).
48 * Returns 1 on success; returns 0 and stores the terminating status in
49 * *rc_out on a bad handle or an over-limit total.
50 *
51 * WHY: Allocation must happen only after the whole request is known sane; this
52 * isolates the sanity pass from the buffer setup and keeps the error triplet
53 * (metric + send_error) at a single named step.
54 *
55 * HOW:
56 * 1. For each segment decode the handle index and requested length.
57 * 2. Reject an invalid handle via brix_validate_read_handle (sets *rc_out).
58 * 3. Clamp the length to the per-element cap and add the framed size.
59 * 4. Reject the request if the running total exceeds BRIX_MAX_READV_TOTAL.
60 */
61static ngx_flag_t
62brix_readv_validate_and_size(brix_ctx_t *ctx, ngx_connection_t *c,
63 brix_readv_req_t *req, ngx_int_t *rc_out)
64{
65 size_t segment_index;
66
67 req->max_response_bytes = 0;
68 for (segment_index = 0; segment_index < req->segment_count;
69 segment_index++)
70 {
71 int handle_index;
72 uint32_t read_length;
73 ngx_int_t validate_rc;
74
75 handle_index =
76 (int) (unsigned char) req->wire_segments[segment_index].fhandle[0];
77 read_length =
78 (uint32_t) ntohl((uint32_t) req->wire_segments[segment_index].rlen);
79
80 if (!brix_validate_read_handle(ctx, c, handle_index, "READV",
81 BRIX_OP_READV, &validate_rc)) {
82 *rc_out = validate_rc;
83 return 0;
84 }
85
86 if ((size_t) read_length > req->readv_seg_max) {
87 read_length = (uint32_t) req->readv_seg_max;
88 }
89 req->max_response_bytes += BRIX_READV_SEGSIZE + read_length;
90
91 if (req->max_response_bytes > BRIX_MAX_READV_TOTAL) {
92 BRIX_OP_ERR(ctx, BRIX_OP_READV);
93 *rc_out = brix_send_error(ctx, c, kXR_ArgTooLong,
94 "readv total would exceed server limit");
95 return 0;
96 }
97 }
98
99 return 1;
100}

102/* ---- Build the response wire body and the segment descriptor array ----
103 *
104 * WHAT: Lays out the final kXR_readv response body in req->response_buffer and
105 * fills req->segment_descs so each descriptor's payload pointer names the
106 * exact place preadv() will land that segment's bytes. No return value; both
107 * buffers are pre-allocated by the orchestrator.
108 *
109 * WHY: The body layout and descriptor construction is a single pure pass with
110 * no I/O; isolating it keeps brix_handle_readv flat and the byte layout (the
111 * 4-byte fhandle, 4-byte length, 8-byte offset header per segment) reviewable
112 * in one place.
113 *
114 * HOW:
115 * 1. Walk a cursor over the response buffer, one framed segment at a time.
116 * 2. For each segment write the fhandle, clamped big-endian length, and raw
117 * offset into the leading header, then record fd/obj/offset/length and the
118 * header/payload pointers into the descriptor.
119 * 3. Advance the cursor past the header and clamped payload region.
120 */
121static void
122brix_readv_build_descriptors(brix_ctx_t *ctx, brix_readv_req_t *req)
123{
124 u_char *response_cursor = req->response_buffer;
125 size_t segment_index;
126
127 for (segment_index = 0; segment_index < req->segment_count;
128 segment_index++)
129 {
130 int handle_index;
131 uint32_t read_length;
132 uint32_t read_length_be;
133
134 handle_index =
135 (int) (unsigned char) req->wire_segments[segment_index].fhandle[0];
136 read_length = (uint32_t) ntohl(
137 (uint32_t) req->wire_segments[segment_index].rlen);
138
139 if ((size_t) read_length > req->readv_seg_max) {
140 read_length = (uint32_t) req->readv_seg_max;
141 }
142
143 ngx_memcpy(response_cursor,
144 req->wire_segments[segment_index].fhandle, 4);
145 read_length_be = htonl(read_length);
146 ngx_memcpy(response_cursor + 4, &read_length_be, 4);
147 ngx_memcpy(response_cursor + 8,
148 &req->wire_segments[segment_index].offset, 8);
149
150 req->segment_descs[segment_index].fd =
151 ctx->files[handle_index].fd;
152 req->segment_descs[segment_index].obj =
153 ctx->files[handle_index].sd_obj; /* Layer 3: driver or zeroed */
154 req->segment_descs[segment_index].handle_index = handle_index;
155 req->segment_descs[segment_index].offset = (off_t) (int64_t)
156 be64toh((uint64_t) req->wire_segments[segment_index].offset);
157 req->segment_descs[segment_index].read_length = read_length;
158 req->segment_descs[segment_index].header_read_length_ptr =
159 response_cursor + 4;
160 req->segment_descs[segment_index].payload_ptr =
161 response_cursor + BRIX_READV_SEGSIZE;
162
163 response_cursor += BRIX_READV_SEGSIZE + read_length;
164 }
165}

167/* ---- Try to offload the readv I/O onto the AIO thread pool ----
168 *
169 * WHAT: When a thread pool is configured, populates the reusable readv AIO task
170 * and posts it. Returns 1 if the task was posted (the caller must return
171 * NGX_OK immediately); returns 0 to fall back to synchronous execution. On an
172 * allocation failure it frees the descriptors + response buffer and returns 1
173 * with *rc_out = NGX_ERROR.
174 *
175 * WHY: The async offload is an optional fast path; separating it lets the
176 * orchestrator read as "post; else run synchronously" and keeps the task-setup
177 * fields in one place.
178 *
179 * HOW:
180 * 1. No thread pool → return 0 (sync path).
181 * 2. Reuse ctx->rd.readv_aio_task, allocating it on first use (free + return
182 * on OOM); otherwise reset its completion state.
183 * 3. Fill the task context (connection, ctx, segments, buffer, stream id),
184 * bind the thread/done callbacks, and post it.
185 * 4. Return 1 only when brix_aio_post_task reports it was actually posted.
186 */
187static ngx_flag_t
188brix_readv_try_post_async(brix_ctx_t *ctx, ngx_connection_t *c,
189 brix_readv_req_t *req, ngx_int_t *rc_out)
190{
191 ngx_thread_task_t *task;
192 brix_readv_aio_t *t;
193 ngx_flag_t posted;
194
195 if (req->rconf->common.thread_pool == NULL) {
196 return 0;
197 }
198
199 task = ctx->rd.readv_aio_task;
200 if (task == NULL) {
201 task = ngx_thread_task_alloc(c->pool, sizeof(brix_readv_aio_t));
202 if (task == NULL) {
203 ngx_free(req->segment_descs);
204 brix_release_read_buffer(ctx, c, req->response_buffer);
205 *rc_out = NGX_ERROR;
206 return 1;
207 }
208 ctx->rd.readv_aio_task = task;
209 } else {
210 task->next = NULL;
211 task->event.complete = 0;
212 }
213
214 t = task->ctx;
215 t->c = c;
216 t->ctx = ctx;
217 t->segment_count = req->segment_count;
218 t->segments = req->segment_descs;
219 t->response_buffer = req->response_buffer;
220 t->bytes_read_total = 0;
221 t->response_bytes = 0;
222 t->io_error = 0;
223 t->streamid[0] = ctx->recv.cur_streamid[0];
224 t->streamid[1] = ctx->recv.cur_streamid[1];
225
226 brix_task_bind(task, brix_readv_aio_thread, brix_readv_aio_done);
227
228 (void) brix_aio_post_task(ctx, c, req->rconf->common.thread_pool, task,
229 "brix: thread_task_post failed, falling back to sync readv",
230 &posted);
231 if (posted) {
232 *rc_out = NGX_OK;
233 return 1;
234 }
235
236 return 0;

239/* ---- Run the readv I/O synchronously and send the assembled response ----
240 *
241 * WHAT: Executes the vectored read through the VFS I/O seam, updates per-handle
242 * and total byte counters, emits the access-log line, and queues the chunked
243 * response chain. Returns the status to propagate to the dispatcher; on I/O
244 * error emits the READV error triplet with kXR_IOError.
245 *
246 * WHY: This is the fallback path (no thread pool, or post declined); isolating
247 * it keeps the orchestrator flat and puts all response-side side effects
248 * (byte accounting, logging, framing, send) behind one named step.
249 *
250 * HOW:
251 * 1. Build a BRIX_VFS_IO_READV job over the descriptors and execute it.
252 * 2. On I/O error free the buffers and send the kXR_IOError triplet.
253 * 3. Add each segment's read_length to its handle's counter, then free descs.
254 * 4. Log access when enabled, bump metrics/total bytes.
255 * 5. Build the chunked chain and queue it, releasing the buffer on failure or
256 * when the connection is no longer sending.
257 */
258static ngx_int_t
259brix_readv_execute_sync(brix_ctx_t *ctx, ngx_connection_t *c,
260 brix_readv_req_t *req)
261{
262 size_t bytes_read_total = 0;
263 size_t response_bytes;
264 size_t segment_index;
265 ngx_chain_t *rsp_chain;
266 char error_message[128];
267 brix_vfs_job_t job;
268
269 error_message[0] = '\0';
270 ngx_memzero(&job, sizeof(job));
271 job.op = BRIX_VFS_IO_READV;
272 job.segs = req->segment_descs;
273 job.nsegs = req->segment_count;
274 job.err_msg = error_message;
275 job.err_msg_cap = sizeof(error_message);
276
277 brix_vfs_io_execute(&job);
278
279 if (job.io_errno != 0) {
280 ngx_free(req->segment_descs);
281 brix_release_read_buffer(ctx, c, req->response_buffer);
282 BRIX_OP_ERR(ctx, BRIX_OP_READV);
283 return brix_send_error(ctx, c, kXR_IOError,
284 error_message[0] ? error_message
285 : "readv I/O error");
286 }
287
288 bytes_read_total = (size_t) job.nio;
289 response_bytes = job.out_size;
290
291 for (segment_index = 0; segment_index < req->segment_count;
292 segment_index++)
293 {
294 ctx->files[req->segment_descs[segment_index].handle_index].bytes_read +=
295 req->segment_descs[segment_index].read_length;
296 }
297 ngx_free(req->segment_descs);
298
299 if (req->rconf->access_log_fd != NGX_INVALID_FILE) {
300 char detail[64];
301
302 snprintf(detail, sizeof(detail), "%zu_segs", req->segment_count);
303 brix_log_access(ctx, c, "READV", "-", detail, 1, 0, NULL,
304 bytes_read_total);
305 }
306 BRIX_OP_OK(ctx, BRIX_OP_READV);
307 ctx->totals.bytes += bytes_read_total;
308
309 rsp_chain = brix_build_chunked_chain(ctx, c, req->response_buffer,
310 response_bytes);
311 if (rsp_chain == NULL) {
312 brix_release_read_buffer(ctx, c, req->response_buffer);
313 return NGX_ERROR;
314 }
315
316 {
317 ngx_int_t rc = brix_queue_response_chain(ctx, c, rsp_chain,
318 req->response_buffer);
319
320 if (rc != NGX_OK || ctx->state != XRD_ST_SENDING) {
321 brix_release_read_buffer(ctx, c, req->response_buffer);
322 }
323 return rc;
324 }
325}

327/* Handle kXR_readv — validate the request (payload a multiple of the readv
328 * element size, segment lengths within bounds), read all segments, and assemble
329 * the chained response. */
330ngx_int_t
331brix_handle_readv(brix_ctx_t *ctx, ngx_connection_t *c)
332{
333 /* phase-42 W4 invariant: readv is ALWAYS plaintext — it never consults
334 * read_codec. Inline read compression applies only to single-segment
335 * kXR_read; readv's interleaved per-segment framing (and its pgread-style
336 * integrity expectations) stay byte-for-byte intact. */
337 brix_readv_req_t req;
338 ngx_int_t rc;
339
340 ngx_memzero(&req, sizeof(req));
341
342 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0 ||
343 (ctx->recv.cur_dlen % BRIX_READV_SEGSIZE) != 0)
344 {
345 BRIX_OP_ERR(ctx, BRIX_OP_READV);
346 return brix_send_error(ctx, c, kXR_ArgInvalid,
347 "malformed readv request");
348 }
349
350 req.wire_segments = (readahead_list *) ctx->recv.payload;
351 req.segment_count = ctx->recv.cur_dlen / BRIX_READV_SEGSIZE;
352
353 /* Phase 27 W2/F1: explicit segment-count cap at the callsite (defense in
354 * depth over the recv-layer payload cap). */
355 if (req.segment_count == 0 || req.segment_count > BRIX_READV_MAXSEGS) {
356 BRIX_OP_ERR(ctx, BRIX_OP_READV);
357 return brix_send_error(ctx, c, kXR_ArgTooLong,
358 "readv segment count exceeds server limit");
359 }
360
361 req.rconf = ngx_stream_get_module_srv_conf(
362 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
363
364 /* Per-element read cap = the configured readv segment size (official
365 * maxReadv_ior, advertised via Qconfig readv_ior_max). An element requesting
366 * more is served SHORT (capped): our native client reads each segment's
367 * actual returned length (see test_readv_variable_blocks). Well-behaved
368 * clients honour the advertised readv_ior_max and never request more; a
369 * client that ignores it and cannot handle a short element (e.g. XrdCl's
370 * VectorRead) must therefore size its elements within the cap. */
371 req.readv_seg_max = req.rconf->readv_segment_size;
372
373 /*
374 * First pass validates every file handle and computes the upper bound for
375 * the single response scratch buffer. Allocation happens only after the
376 * whole request is known to be sane.
377 */
378 if (!brix_readv_validate_and_size(ctx, c, &req, &rc)) {
379 return rc;
380 }
381
382 brix_prefetch_readv_segments(ctx, c, req.wire_segments, req.segment_count,
383 req.readv_seg_max);
384
385 /*
386 * Phase 31 W4: kXR_readv assembles its whole response (up to
387 * BRIX_MAX_READV_TOTAL = 256 MiB) in rd.read_scratch. Admit it against the
388 * SHM-global transfer budget so a burst of large readv requests cannot blow
389 * the memory cap; over budget, defer with kXR_wait and let the client
390 * re-issue. (readv is not yet windowed like kXR_read — a single large readv
391 * still allocates its full response; the budget bounds the aggregate.)
392 */
393 if (!brix_budget_admit(ctx, req.rconf->memory_budget,
394 req.max_response_bytes)) {
395 return brix_send_wait(ctx, c, 1);
396 }
397
398 req.response_buffer = BRIX_GET_SCRATCH(ctx, c, rd.read_scratch,
399 rd.read_scratch_size,
400 req.max_response_bytes);
401 if (req.response_buffer == NULL) {
402 return NGX_ERROR;
403 }
404
405 /* Charge the assembled-response footprint to the budget promptly. */
406 brix_budget_sync(ctx);
407
408 req.segment_descs = brix_alloc_array(c->log, req.segment_count,
409 sizeof(brix_readv_seg_desc_t));
410 if (req.segment_descs == NULL) {
411 brix_release_read_buffer(ctx, c, req.response_buffer);
412 return NGX_ERROR;
413 }
414
415 /*
416 * Build the final wire body up front. The descriptor payload pointers are
417 * the exact places where preadv() will land each segment's bytes.
418 */
419 brix_readv_build_descriptors(ctx, &req);
420
421 /* Prefer the AIO offload; on a successful post the response completes in the
422 * done callback and we return its status directly. */
423 if (brix_readv_try_post_async(ctx, c, &req, &rc)) {
424 return rc;
425 }
426
427 /* No thread pool, or the post declined — read synchronously and reply. */
428 return brix_readv_execute_sync(ctx, c, &req);
429}
