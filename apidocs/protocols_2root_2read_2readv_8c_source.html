118 run_bytes += (size_t) segments[segment_index + k].read_length;

146 snprintf(error_message, error_message_len,
147 "readv past EOF at seg %zu", segment_index);
148 free(ranges);
149 return NGX_ERROR;
150 }
151
152 *bytes_read_total += run_bytes;
153 segment_index += run_count;
154 }
155
156 free(ranges);
