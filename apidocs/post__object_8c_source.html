16/*
17 * WHAT: Copy `len` raw form bytes into a fixed C-string buffer `dst`.
18 * WHY: Form values are attacker-controlled; we reject embedded NUL bytes so a
19 * value cannot be truncated when later treated as a C string (smuggling a
20 * different key/policy past length-based checks). HOW: bounds-check, scan
21 * for NUL, then copy and terminate.
22 * Returns NGX_OK, or NGX_ERROR if it would overflow or contains a NUL.
23 */
24ngx_int_t
25s3_post_copy_text(const u_char *data, size_t len, char *dst, size_t dstsz)
26{
27 size_t i;
28
29 if (len >= dstsz) {
30 return NGX_ERROR;
31 }
32
33 /* Reject embedded NUL so dst is an unambiguous C string downstream. */
34 for (i = 0; i < len; i++) {
35 if (data[i] == '\0') {
36 return NGX_ERROR;
37 }
38 }
39
40 ngx_memcpy(dst, data, len);
41 dst[len] = '\0';
42 return NGX_OK;
43}

46/*
47 * WHAT: Find the first occurrence of `needle` within the binary buffer `hay`.
48 * WHY: The multipart body is binary (file bytes may contain NUL), so libc
49 * strstr() cannot be used to locate boundary/CRLF markers. HOW: naive
50 * O(n*m) byte scan, which is adequate for the small needles used here.
51 * Returns a pointer into `hay`, or NULL if not found.
52 */
53u_char *
54s3_memmem(u_char *hay, size_t hay_len, const u_char *needle,
55 size_t needle_len)
56{
57 size_t i;
58
59 if (needle_len == 0 || hay_len < needle_len) {
60 return NULL;
61 }
62
63 for (i = 0; i <= hay_len - needle_len; i++) {
64 if (hay[i] == needle[0]
65 && ngx_memcmp(hay + i, needle, needle_len) == 0)
66 {
67 return hay + i;
68 }
69 }
70
71 return NULL;
72}

75/*
76 * WHAT: Reduce a client-supplied filename to its basename, in place.
77 * WHY: Browsers may send a full path (e.g. "C:\\Users\\x\\f.txt" or
78 * "/home/x/f.txt") in the multipart filename; only the final component is
79 * meaningful and stripping leading path segments removes a directory-
80 * traversal vector before the name is used in ${filename} expansion. HOW:
81 * take whichever of the last '/' or last '\\' appears later, then shift
82 * the tail (incl. its NUL) down to the front of the buffer.
83 */
84void
85s3_post_basename(char *s)
86{
87 char *slash;
88 char *bslash;
89 char *base;
90
91 slash = strrchr(s, '/');
92 bslash = strrchr(s, '\\');
93 base = slash > bslash ? slash : bslash; /* later separator wins */
94
95 if (base != NULL) {
96 ngx_memmove(s, base + 1, strlen(base + 1) + 1);
97 }
98}

101/*
102 * WHAT: Commit the uploaded file bytes to `fs_path` and compute its ETag.
103 * WHY: Uses the VFS staged-write lifecycle (write to a temp, atomically rename
104 * on success) so a partial/failed upload never leaves a corrupt object
105 * visible — the same durability contract as PUT. The VFS layer keeps every
106 * op confined to root_canon and meters the publish (OP_WRITE) + access log.
107 * HOW: ensure the parent directory exists (brix_vfs_mkdir, parents=1), open a
108 * staged handle (brix_vfs_staged_open), pwrite the whole buffer to its
109 * fd, commit (atomic rename), then stat the result for the ETag.
110 * Cleanup: any write error calls brix_vfs_staged_abort (discard temp) before
111 * returning NGX_ERROR, so no orphan temp file is left behind.
112 * Returns NGX_OK (object committed, etag set) or NGX_ERROR.
113 */
114ngx_int_t
115s3_post_write_object(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
116 const s3_post_form_t *form, const char *fs_path, char *etag,
117 size_t etag_sz)
118{
119 brix_vfs_ctx_t vctx;
120 brix_vfs_writer_t *w;
121 int vfs_err;
122 struct stat sb;
123
124 /* Create the key's parent directories (S3 keys may embed '/') through the
125 * VFS mkdir (parents=1). EEXIST is benign; any other mkdir failure aborts
126 * the write. */
127 {
128 char parent[PATH_MAX];
129 char *last_slash;
130 size_t flen = strlen(fs_path);
131
132 if (flen < sizeof(parent)) {
133 ngx_memcpy(parent, fs_path, flen + 1);
134 last_slash = strrchr(parent, '/');
135 /* Skip if the slash is the root itself (last_slash == parent). */
136 if (last_slash && last_slash != parent) {
137 brix_vfs_ctx_t pctx;
138
139 *last_slash = '\0';
140 s3_build_vfs_ctx(r, parent, cf, &pctx);
141 if (brix_vfs_mkdir(&pctx, 0755, 1 /* parents */) != NGX_OK
142 && errno != EEXIST)
143 {
144 return NGX_ERROR;
145 }
146 }
147 }
148 }
149
150 s3_build_vfs_ctx(r, fs_path, cf, &vctx);
151 w = brix_vfs_writer_open(&vctx, BRIX_VFS_O_ATOMIC, cf->common.verify_write,
152 &vfs_err);
153 if (w == NULL) {
154 errno = vfs_err;
155 return NGX_ERROR;
156 }
157
158 /* Write the full in-memory file part through the unified write session — it
159 * routes to the backend's driver (in-place fd or staged object upload), so a
160 * non-POSIX backend is no longer bypassed by a raw fd-keyed pwrite. */
161 if (brix_vfs_writer_write(w, form->file_data, form->file_len, 0) != NGX_OK) {
162 brix_vfs_writer_abort(w);
163 return NGX_ERROR;
164 }
165
166 /* Atomically publish the staged temp as the final object (folds the
167 * verify-on-write read-back when the export opts in). */
168 if (brix_vfs_writer_commit(w) != NGX_OK) {
169 return NGX_ERROR;
170 }
171
172 /* Best-effort ETag: a stat failure here is non-fatal (empty ETag). The VFS
173 * probe is confined + no-follow (so a symlink at the final name is not
174 * followed) and non-metered (the PostObject op already accounts for this
175 * upload; a per-upload OP_STAT would inflate the stat counter). */
176 {
177 brix_vfs_stat_t vst;
178
179 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) == NGX_OK) {
180 ngx_memzero(&sb, sizeof(sb));
181 sb.st_mtime = vst.mtime;
182 sb.st_size = vst.size;
183 s3_etag(&sb, etag, etag_sz);
184 (void) s3_set_header(r, "ETag", etag);
185 } else {
186 etag[0] = '\0';
187 }
188 }
189
190 return NGX_OK;
191}

194/*
195 * WHAT: Request-body callback for S3 browser POST Object uploads.
196 * WHY: nginx invokes this once the full request body is buffered (registered as
197 * the body handler), so the multipart payload is available in memory here.
198 * This drives the whole pipeline; on any failure it sends the appropriate
199 * S3 error and finalizes per-method metrics — there is no return value
200 * because nginx owns the request lifecycle from a body handler.
201 * HOW (ordered pipeline, each step finalizes metrics on failure):
202 * parse boundary -> read body -> parse multipart -> require key+file ->
203 * expand ${filename} and resolve the confined fs path -> verify the signed
204 * policy -> write the object -> send the success response.
205 * NOTE: verifying the SIGNATURE before resolving/writing is intentional — but
206 * path resolution happens before the (potentially expensive) crypto since
207 * a bad key is a cheap reject; the object is only written post-verify.
208 */
209

214 * broker rather than the unprivileged worker. No-op unless map mode.
215 */

228void
229s3_post_object_body_handler_inner(ngx_http_request_t *r)
230{
231 ngx_http_s3_loc_conf_t *cf;
232 u_char *body;
233 size_t body_len;
234 char boundary[256];
235 s3_post_form_t form;
236 char fs_path[PATH_MAX];
237 char etag[48];
238 ngx_int_t rc;
239
240 cf = ngx_http_get_module_loc_conf(r, ngx_http_brix_s3_module);
241 ngx_memzero(&form, sizeof(form));
242
243 if (s3_post_boundary(r, boundary, sizeof(boundary)) != NGX_OK) {
244 s3_metrics_finalize_request_method(
245 r, BRIX_S3_METHOD_POST,
246 s3_post_error(r, NGX_HTTP_BAD_REQUEST, "MalformedPOSTRequest",
247 "POST Object requires multipart/form-data."));
248 return;
249 }
250
251 rc = brix_http_body_read_all(r, S3_POST_MAX_BODY, &body, &body_len);
252 if (rc == NGX_DECLINED) {
253 s3_metrics_finalize_request_method(
254 r, BRIX_S3_METHOD_POST,
255 s3_post_error(r, NGX_HTTP_REQUEST_ENTITY_TOO_LARGE,
256 "EntityTooLarge", "POST body is too large."));
257 return;
258 }
259 if (rc != NGX_OK) {
260 s3_metrics_finalize_request_method(r, BRIX_S3_METHOD_POST,
261 NGX_HTTP_INTERNAL_SERVER_ERROR);
262 return;
263 }
264
265 if (s3_post_parse_form(r, body, body_len, boundary, &form) != NGX_OK) {
266 s3_metrics_finalize_request_method(
267 r, BRIX_S3_METHOD_POST,
268 s3_post_error(r, NGX_HTTP_BAD_REQUEST, "MalformedPOSTRequest",
269 "The multipart form-data body is invalid."));
270 return;
271 }
272
273 if (!form.have_file || form.key[0] == '\0') {
274 s3_metrics_finalize_request_method(
275 r, BRIX_S3_METHOD_POST,
276 s3_post_error(r, NGX_HTTP_BAD_REQUEST, "InvalidArgument",
277 "POST Object requires key and file fields."));
278 return;
279 }
280
281 if (s3_post_expand_filename(r, &form) != NGX_OK
282 || !s3_resolve_key(cf->common.root_canon, form.key,
283 fs_path, sizeof(fs_path),
284 cf->common.cache_store_endpoint))
285 {
286 s3_metrics_finalize_request_method(
287 r, BRIX_S3_METHOD_POST,
288 s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
289 "Access Denied."));
290 return;
291 }
292
293 rc = s3_post_verify_policy(r, cf, &form);
294 if (rc != NGX_OK) {
295 s3_metrics_finalize_request_method(r, BRIX_S3_METHOD_POST, rc);
296 return;
297 }
298
299 if (s3_post_write_object(r, cf, &form, fs_path, etag, sizeof(etag))
300 != NGX_OK)
301 {
302 /*
303 * A confined create that fails with EACCES/EPERM/EXDEV is a forbidden
304 * write, not a server fault, and must surface as 403 AccessDenied (the
305 * same contract the shared errno table gives every other handler).
306 * Under impersonation (`brix_impersonation map`) the create is brokered
307 * as the mapped user, so a missing/unmappable principal or a DAC-denied
308 * target dir lands here — a clean 403, never a 500. Genuine I/O faults
309 * (EIO/ENOSPC/...) keep their 5xx mapping. The staged-file pattern
310 * leaves no orphan object behind on any of these paths.
311 */
312 int werrno = errno;
313
314 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
315 if (brix_http_map_errno(werrno) == NGX_HTTP_FORBIDDEN) {
316 s3_metrics_finalize_request_method(
317 r, BRIX_S3_METHOD_POST,
318 s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
319 "Access Denied."));
320 } else {
321 s3_metrics_finalize_request_method(
322 r, BRIX_S3_METHOD_POST,
323 (ngx_int_t) brix_http_map_errno(werrno));
324 }
325 return;
326 }
327
328 BRIX_S3_METRIC_ADD(bytes_rx_total, form.file_len);
329
330 s3_metrics_finalize_request_method(
331 r, BRIX_S3_METHOD_POST,
332 s3_post_send_success(r, cf, &form, etag));
333}
