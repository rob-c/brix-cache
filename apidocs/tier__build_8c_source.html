1/*
2 * tier_build.c - turn a tier cfg into a bound SD instance, and compose the stack
3 * (phase-64 section 5, Appendix C). See tier.h.
4 *
5 * brix_tier_build is the SINGLE driver-name dispatch in the whole storage plane:
6 * it maps a parsed tier cfg onto the existing per-driver create function. A driver
7 * not yet implemented as a tier (rados, tape) is a tracked "needs development"
8 * (P1) - it logs a [warn] and returns NULL so the export is held inactive, never a
9 * crash. brix_tier_build_stack composes cache(stage(backend)) via the sd_cache /
10 * sd_stage decorators and enforces tape-requires-cache (P4/G8). The result is
11 * memoised per worker; the VFS resolves to it blind (P3, G4/G5).
12 */

27/* Resolve a tier's §14 credential into NUL-terminated bearer / proxy / ca-dir
28 * strings (empty when the tier is anonymous or the field is unset). */
29static void
30tier_resolve_creds(const brix_tier_cfg_t *t, char *bearer, size_t bcap,
31 char *proxy, size_t pcap, char *cadir, size_t ccap, ngx_log_t *log)
32{
33 const brix_credential_t *c = t->credential;
34
35 bearer[0] = '\0';
36 proxy[0] = '\0';
37 cadir[0] = '\0';
38 if (c == NULL) {
39 return;
40 }
41 (void) brix_credential_bearer(c, bearer, bcap, log); /* "" when none */
42 if (c->x509_proxy.len > 0) {
43 (void) brix_str_cbuf(proxy, pcap, &c->x509_proxy);
44 }
45 if (c->ca_dir.len > 0) {
46 (void) brix_str_cbuf(cadir, ccap, &c->ca_dir);
47 }
48}

50/* "needs development": a recognised driver that is not yet a tier in this build.
51 * Marked possibly-unused: its only callers are the #else arms of the optional-
52 * library driver branches (pblock/sqlite, rados/ceph), so a build with ALL those
53 * libraries present (e.g. the ceph container) references it from no arm. */
54static brix_sd_instance_t *tier_needs_dev(ngx_log_t *log, const char *driver,
55 const char *sp) __attribute__((unused));

57static brix_sd_instance_t *
58tier_needs_dev(ngx_log_t *log, const char *driver, const char *sp)
59{
60 ngx_log_error(NGX_LOG_WARN, log, 0,
61 "xrootd tier: the \"%s\" driver is not yet implemented as a storage tier "
62 "(needs development - phase-64 %s); this export is inactive until then",
63 driver, sp);
64 return NULL;
65}

67/* ---- the single driver-name dispatch (Appendix C) ------------------------- */
68
69brix_sd_instance_t *
70brix_tier_build(const brix_tier_cfg_t *t, ngx_log_t *log)
71{
72 if (t == NULL || !t->configured) {
73 return NULL;
74 }
75
76 if (ngx_strcmp(t->driver, "posix") == 0) {
77 int err = 0;
78
79 return brix_sd_instance_create(ngx_cycle->pool, log, "posix",
80 (void *) t->path, &err);
81 }
82
83 if (ngx_strcmp(t->driver, "pblock") == 0) {
84#if BRIX_HAVE_SQLITE
85 brix_sd_pblock_conf_t conf;
86 int err = 0;
87
88 ngx_memzero(&conf, sizeof(conf));
89 conf.root = t->path;
90 conf.busy_timeout_ms = 5000;
91 conf.block_size = (int64_t) t->block_size;
92 return brix_sd_instance_create(ngx_cycle->pool, log, "pblock", &conf,
93 &err);
94#else
95 return tier_needs_dev(log, "pblock (needs libsqlite3)", "build");
96#endif
97 }
98
99 if (ngx_strcmp(t->driver, "xroot") == 0) {
100 char bearer[4096];
101 char proxy[1024];
102 char cadir[1024];
103
104 tier_resolve_creds(t, bearer, sizeof(bearer), proxy, sizeof(proxy),
105 cadir, sizeof(cadir), log);
106 return brix_sd_xroot_create_origin(t->host, t->port, t->tls,
107 BRIX_AF_AUTO,
108 (bearer[0] != '\0') ? bearer : NULL,
109 (proxy[0] != '\0') ? proxy : NULL,
110 (cadir[0] != '\0') ? cadir : NULL, NULL /* sss_keytab */, log);
111 }
112
113 if (ngx_strcmp(t->driver, "http") == 0) {
114 brix_sd_http_cfg_t cfg;
115 char bearer[4096];
116 char proxy[1024];
117 char cadir[1024];
118
119 tier_resolve_creds(t, bearer, sizeof(bearer), proxy, sizeof(proxy),
120 cadir, sizeof(cadir), log);
121 ngx_memzero(&cfg, sizeof(cfg));
122 cfg.host = t->host;
123 cfg.port = t->port;
124 cfg.tls = t->tls;
125 cfg.base_path = t->path;
126 cfg.transport = &brix_s3_origin_curl_transport;
127 cfg.timeout_ms = 60000;
128 cfg.bearer_token = (bearer[0] != '\0') ? bearer : NULL;
129 return brix_sd_http_create(&cfg, log);
130 }
131
132 if (ngx_strcmp(t->driver, "s3") == 0) {
133 brix_sd_remote_cfg_t cfg;
134 const char *bucket = t->path;
135
136 /* The §14 credential block carries no S3 access/secret/region yet (that
137 * representation is SP3, section 21); SP1 builds an anonymous/public S3
138 * source - read works for public buckets. */
139 ngx_memzero(&cfg, sizeof(cfg));
140 cfg.scheme = BRIX_SD_REMOTE_S3;
141 ngx_cpystrn((u_char *) cfg.host, (u_char *) t->host, sizeof(cfg.host));
142 cfg.port = t->port;
143 cfg.tls = t->tls;
144 if (bucket[0] == '/') {
145 bucket++;
146 }
147 ngx_cpystrn((u_char *) cfg.bucket, (u_char *) bucket, sizeof(cfg.bucket));
148 cfg.timeout_ms = 60000;
149 cfg.transport = &brix_s3_origin_curl_transport;
150 return brix_sd_remote_create(&cfg, log);
151 }
152
153 if (ngx_strcmp(t->driver, "rados") == 0
154 || ngx_strcmp(t->driver, "ceph") == 0) {
155#if BRIX_HAVE_CEPH
156 /* rados://POOL[/namespace] / ceph://POOL[/namespace] -> the ceph (librados)
157 * driver; the path tail is the object key prefix. The cstore over it uses
158 * XATTR cinfo mode (the cache state lives on the RADOS object). */
159 brix_sd_ceph_conf_t conf;
160 const char *prefix = (t->path[0] == '/') ? t->path + 1 : t->path;
161 int err = 0;
162
163 ngx_memzero(&conf, sizeof(conf));
164 conf.pool = t->host;
165 conf.conf_file = NULL; /* default /etc/ceph/ceph.conf */
166 conf.key_prefix = (prefix[0] != '\0') ? prefix : NULL;
167 return brix_sd_instance_create(ngx_cycle->pool, log, "ceph", &conf,
168 &err);
169#else
170 return tier_needs_dev(log, "rados (needs librados at build time)", "SP3");
171#endif
172 }
173 if (ngx_strcmp(t->driver, "tape") == 0
174 || ngx_strcmp(t->driver, "frm") == 0) {
175 /* the tape://|frm:// authority host selects the MSS adapter ("" = stub); the
176 * path is the adapter's MSS base (the stub's local tape dir). */
177 return brix_sd_frm_create(t->host, t->path, log);
178 }
179
180 ngx_log_error(NGX_LOG_ERR, log, 0,
181 "xrootd tier: unknown driver \"%s\"", t->driver);
182 return NULL; /* operator error - parse_store should have rejected it */
183}

185/* ---- compose the stack (cache outermost -> stage -> backend) -------------- */
186
187brix_sd_instance_t *
188brix_tier_build_stack(brix_tier_stack_t *s, ngx_log_t *log)
189{
190 brix_sd_instance_t *top;
191 brix_sd_instance_t *store;
192
193 if (s == NULL) {
194 return NULL;
195 }
196 if (s->composed != NULL) {
197 return s->composed; /* memoised in this worker */
198 }
199
200 top = brix_tier_build(&s->backend, log);
201 if (top == NULL) {
202 return NULL;
203 }
204
205 /* G8 (P4): a nearline (tape) backend cannot be served without a cache tier as
206 * the recall target. The operator-facing [emerg] is raised at config time once
207 * the tier config wiring lands (section 9.4); this is the build-time guard. */
208 if (s->backend.nearline && !s->cache_store.configured) {
209 ngx_log_error(NGX_LOG_ERR, log, 0,
210 "xrootd tier: a nearline (tape) backend requires brix_cache (the "
211 "recall target) - export held inactive");
212 return NULL;
213 }
214
215 if (s->stage_store.configured && s->stage.enabled) {
216 store = brix_tier_build(&s->stage_store, log);
217 if (store == NULL) {
218 return NULL;
219 }
220 top = brix_sd_stage_create(top, store, &s->stage, NULL, log);
221 if (top == NULL) {
222 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
223 "xrootd tier: stage decorator init failed");
224 return NULL;
225 }
226 }
227
228 if (s->cache_store.configured && s->cache.enabled) {
229 const char *local_root;
230
231 store = brix_tier_build(&s->cache_store, log);
232 if (store == NULL) {
233 return NULL;
234 }
235 /* LOCAL cinfo only for a posix cache store; a remote store keeps cinfo as
236 * a store xattr (SP2), so pass NULL there. */
237 local_root = (ngx_strcmp(s->cache_store.driver, "posix") == 0
238 && s->cache_store.path[0] != '\0')
239 ? s->cache_store.path : NULL;
240 top = brix_sd_cache_create(top, store, &s->cache, local_root, log);
241 if (top == NULL) {
242 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
243 "xrootd tier: cache decorator init failed");
244 return NULL;
245 }
246 }
247
248 s->composed = top;
249 return top;
250}
