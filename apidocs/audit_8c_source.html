1/*
2 * audit.c — authorization grant/deny audit log (XrdAccAudit).
3 *
4 * WHAT: brix_acc_audit() emits one structured line per authorization decision
5 * when auditing is enabled — deny-only, grant-only, or both — mirroring
6 * XRootD's `acc.audit`. Untrusted fields (id, host, path come off the wire)
7 * are sanitised so a crafted path cannot inject log lines.
8 *
9 * WHY: operators need an auth trail; the `native` engine only logs denials ad
10 * hoc. This is the XrdAcc-parity audit sink, gated by brix_authdb_audit.
11 *
12 * HOW: a level bitmask (1=deny, 2=grant) decides whether to log; the line is
13 * "<id>@<host> grant|deny <op> <path>" with control/quote bytes escaped.
14 */

21/* Copy src into dst (capacity cap) escaping control bytes, quotes and
22 * backslashes to keep the audit line single-valued and injection-safe. */
23static void
24acc_audit_sanitize(u_char *dst, size_t cap, const char *src)
25{
26 size_t i = 0;
27
28 if (src == NULL) { src = "-"; }
29 while (*src && i + 4 < cap) {
30 u_char c = (u_char) *src++;
31 if (c == '\\' || c == '"') {
32 dst[i++] = '\\';
33 dst[i++] = c;
34 } else if (c < 0x20 || c >= 0x7f) {
35 dst[i++] = '?';
36 } else {
37 dst[i++] = c;
38 }
39 }
40 dst[i] = '\0';
41}

43void
44brix_acc_audit(ngx_log_t *log, ngx_uint_t level, int granted,
45 const char *op, const char *id, const char *host,
46 const char *path)
47{
48 u_char idbuf[256], hostbuf[256], pathbuf[1024];
49
50 if (level == 0) {
51 return;
52 }
53 if (granted && !(level & BRIX_ACC_AUDIT_GRANT)) {
54 return;
55 }
56 if (!granted && !(level & BRIX_ACC_AUDIT_DENY)) {
57 return;
58 }
59
60 acc_audit_sanitize(idbuf, sizeof(idbuf), id);
61 acc_audit_sanitize(hostbuf, sizeof(hostbuf), host);
62 acc_audit_sanitize(pathbuf, sizeof(pathbuf), path);
63
64 ngx_log_error(NGX_LOG_NOTICE, log, 0,
65 "xrootd authz: %s@%s %s %s \"%s\"",
66 idbuf, hostbuf, granted ? "grant" : "deny",
67 (op != NULL) ? op : "?", pathbuf);
68}
