20/*
21 * xattr naming scheme: an xattr key is
22 * "user.nginx_xrootd.webdav." <hex(namespace-URI)> "." <hex(local-name)>
23 * Both the XML namespace URI and the local element name are lowercase-hex
24 * encoded so that arbitrary URI bytes (':', '/', '.', etc.) survive the
25 * "user." flat keyspace, and the single literal '.' separator is unambiguous
26 * (real dots inside the URI/name become "2e", never a bare '.').
27 * NAME_MAX 255 is the Linux xattr-key limit; VALUE_MAX/LIST_MAX cap how much
28 * we will read back so a hostile filesystem cannot force unbounded allocs.
29 */
30#define WEBDAV_DEAD_PROP_PREFIX "user.nginx_xrootd.webdav."
31#define WEBDAV_DEAD_PROP_PREFIX_LEN (sizeof(WEBDAV_DEAD_PROP_PREFIX) - 1)
32#define WEBDAV_DEAD_PROP_NAME_MAX 255
33#define WEBDAV_DEAD_PROP_VALUE_MAX 16384
34#define WEBDAV_DEAD_PROP_LIST_MAX 65536
