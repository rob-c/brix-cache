1/*
2 * authfile_tokenize.c — the XrdAccAuthFile lexer for the `authdb` parser.
3 *
4 * WHAT: acc_tok_next() splits the slurped authdb buffer into logical records —
5 * whitespace-separated words, one record per line, with `\`-at-end-of-line
6 * continuation and `#` comments — returning ACC_TOK_WORD / ACC_TOK_EOR /
7 * ACC_TOK_EOF. Words are pool-copied and NUL-terminated for the record layer.
8 *
9 * WHY: the byte-level scanner is a self-contained concern with no knowledge of
10 * record types or the tables it feeds; splitting it out of authfile.c keeps the
11 * orchestrator focused on load→dispatch→finalize and each unit under the
12 * file-size standard. Its tokenization is byte-for-byte the stock
13 * XrdAccAuthFile behaviour so an existing authdb parses identically.
14 *
15 * HOW: acc_tok_skip_gap() advances over separators, `\`-newline continuations
16 * and comments to classify the next byte; acc_tok_next() then scans a word (or
17 * emits the record/EOF marker). All allocation comes from the caller's pool.
18 */
19
20#include "authfile_internal.h"

22/*
23 * acc_pstrdup — pool-copy `len` bytes of `s` into a fresh NUL-terminated buffer.
24 *
25 * WHAT: returns a pool-owned NUL-terminated copy of the word, or NULL on alloc
26 * failure.
27 * WHY: wire words are not NUL-terminated slices of the mmap'd buffer; the record
28 * layer wants C strings it can strcmp/strlen without carrying lengths around.
29 * HOW: 1) allocate len+1 from the pool; 2) memcpy the bytes; 3) append '\0'.
30 */
31static char *
32acc_pstrdup(ngx_pool_t *pool, const u_char *s, size_t len)
33{
34 char *d = ngx_pnalloc(pool, len + 1);
35 if (d == NULL) {
36 return NULL;
37 }
38 ngx_memcpy(d, s, len);
39 d[len] = '\0';
40 return d;
41}

43/* Outcome of skipping inter-token gap: what the scanner is now positioned on. */
44enum { ACC_GAP_EOF = 0, ACC_GAP_EOR, ACC_GAP_WORD };
45
46/*
47 * acc_tok_bslash_is_cont — is the '\' at t->p an end-of-line continuation
48 * (`<newline>` or `<cr><newline>`) rather than the start of an escaped path
49 * word? Advances t->p past the continuation and returns 1 when it is; leaves
50 * t->p untouched and returns 0 otherwise.
51 */
52static int
53acc_tok_bslash_is_cont(acc_tok_t *t)
54{
55 if (t->p + 1 < t->end && t->p[1] == '\n') {
56 t->p += 2;
57 return 1;
58 }
59 if (t->p + 2 < t->end && t->p[1] == '\r' && t->p[2] == '\n') {
60 t->p += 3;
61 return 1;
62 }
63 return 0;
64}

66/*
67 * acc_tok_skip_gap — advance t->p over separators (space/tab/cr), `\`-newline
68 * continuations and `#` comments, stopping at the first byte that starts a word,
69 * a record-terminating newline, or end of input. Returns ACC_GAP_WORD /
70 * ACC_GAP_EOR (having consumed the newline) / ACC_GAP_EOF accordingly. Split
71 * out of acc_tok_next so the classify step is separate from word scanning.
72 */
73static int
74acc_tok_skip_gap(acc_tok_t *t)
75{
76 for (;;) {
77 if (t->p >= t->end) {
78 return ACC_GAP_EOF;
79 }
80 switch (*t->p) {
81 case ' ': case '\t': case '\r':
82 t->p++;
83 continue;
84 case '\\':
85 if (acc_tok_bslash_is_cont(t)) {
86 continue;
87 }
88 return ACC_GAP_WORD; /* literal '\' begins a word (path escape) */
89 case '#':
90 while (t->p < t->end && *t->p != '\n') {
91 t->p++;
92 }
93 continue;
94 case '\n':
95 t->p++;
96 return ACC_GAP_EOR;
97 default:
98 return ACC_GAP_WORD;
99 }
100 }
101}

103/*
104 * acc_tok_next — return the next word (ACC_TOK_WORD, *word set), an end-of-record
105 * marker (ACC_TOK_EOR, at a non-continued newline), or ACC_TOK_EOF. Whitespace
106 * separates words; `\` immediately before a newline continues the record; `#`
107 * starts a comment to end of line.
108 */
109int
110acc_tok_next(acc_tok_t *t, char **word)
111{
112 u_char *start;
113
114 switch (acc_tok_skip_gap(t)) {
115 case ACC_GAP_EOF: return ACC_TOK_EOF;
116 case ACC_GAP_EOR: return ACC_TOK_EOR;
117 default: break; /* ACC_GAP_WORD: a word starts at t->p */
118 }
119
120 start = t->p;
121 while (t->p < t->end
122 && *t->p != ' ' && *t->p != '\t'
123 && *t->p != '\n' && *t->p != '\r')
124 {
125 t->p++;
126 }
127
128 *word = acc_pstrdup(t->pool, start, (size_t) (t->p - start));
129 return (*word == NULL) ? ACC_TOK_EOF : ACC_TOK_WORD;
130}
