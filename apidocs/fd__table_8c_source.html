14/* brix_alloc_fhandle — return the first free slot (fd < 0) in ctx->files; the
15 * index becomes the one-byte on-wire fhandle, so it is bounded by BRIX_MAX_FILES.
16 * -1 when all slots are occupied. Single-owner per connection (event thread, no
17 * locking); a slot is only reused once its fd is closed. */
18int
19brix_alloc_fhandle(brix_ctx_t *ctx)
20{
21 int handle_index;
22
23 /*
24 * The XRootD wire handle stores this slot number in one byte. Keep all
25 * allocation/validation paths using the same bounded table.
26 */
27 for (handle_index = 0; handle_index < BRIX_MAX_FILES; handle_index++) {
28 if (ctx->files[handle_index].fd < 0) {
29 return handle_index;
30 }
31 }
32
33 return -1;
34}

37 * Used by the recv-loop drain gate: a connection with an open file is mid-transfer
38 * (a streaming read parked between kXR_read chunks), so a draining worker must let

41int
42brix_ctx_has_open_file(const brix_ctx_t *ctx)

44 int handle_index;
45
46 for (handle_index = 0; handle_index < BRIX_MAX_FILES; handle_index++) {
47 if (ctx->files[handle_index].fd >= 0) {
48 return 1;
49 }
50 }
51
52 return 0;

55/* brix_set_fhandle_path — store a heap copy (ngx_alloc, NOT pool, so it outlives
56 * the kXR_open request) of the canonical path in the slot, freeing any prior path
57 * first; brix_free_fhandle owns the free. NGX_OK, or NGX_ERROR on bad bounds or
58 * allocation failure. Heap (not pool) avoids fragmentation across open/close cycles. */
59ngx_int_t
60brix_set_fhandle_path(brix_ctx_t *ctx, ngx_connection_t *c,
61 int handle_index, const char *path)
62{
63 char *path_copy;
64 size_t path_bytes;
65
66 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES
67 || path == NULL)
68 {
69 return NGX_ERROR;
70 }
71
72 path_bytes = ngx_strlen(path) + 1;
73 path_copy = ngx_alloc(path_bytes, c->log);
74 if (path_copy == NULL) {
75 return NGX_ERROR;
76 }
77
78 ngx_memcpy(path_copy, path, path_bytes);
79
80 if (ctx->files[handle_index].path != NULL) {
81 ngx_free(ctx->files[handle_index].path);
82 }
83
84 ctx->files[handle_index].path = path_copy;
85 return NGX_OK;
86}

93brix_local_file_matches_shared_handle(const brix_file_t *file,
94 const brix_shared_handle_entry_t *shared)
95{

107/* brix_reopen_bound_read_handle — reopen a bound secondary's stale fd to match the
108 * primary's shared entry via a cache-aware confined open (O_RDONLY|O_NOCTTY|O_CLOEXEC),
109 * refreshing all file metadata from the fresh fstat; NGX_DECLINED (revoke) if the
110 * filesystem object's device/inode changed. */
111
112static ngx_int_t
113brix_reopen_bound_read_handle(brix_ctx_t *ctx, ngx_connection_t *c,
114 int handle_index, const brix_shared_handle_entry_t *shared)
115{
116 ngx_stream_brix_srv_conf_t *conf;
117 brix_file_t *file;
118 struct stat st;
119 int fd;
120 int open_flags;
121
122 if (!shared->readable || shared->path[0] == '\0') {
123 return NGX_DECLINED;
124 }
125
126 conf = ngx_stream_get_module_srv_conf(ctx->session,
127 ngx_stream_brix_module);
128 if (conf == NULL) {
129 return NGX_ERROR;
130 }
131
132 /*
133 * The primary published a canonical, ACL-checked path. The secondary
134 * still opens it with the same confinement helper so a stale or corrupted
135 * shared-memory entry cannot escape the configured export root.
136 */
137 open_flags = O_RDONLY | O_NOCTTY | O_CLOEXEC;
138 if (shared->from_cache) {
139 fd = open(shared->path, open_flags); /* vfs-seam-allow: separate server-managed cache-root domain (from_cache), opened as worker */
140 } else {
141 /* shared->path is the absolute path; strip root_canon to get the
142 * path relative to rootfd for openat2 RESOLVE_BENEATH. */
143 const char *rel = shared->path;
144 size_t root_len = strlen(conf->common.root_canon);
145 if (root_len > 0
146 && ngx_strncmp((u_char *) shared->path,
147 (u_char *) conf->common.root_canon,
148 root_len) == 0
149 && shared->path[root_len] == '/')
150 {
151 rel = shared->path + root_len;
152 }
153 fd = brix_vfs_open_fd_at(conf->rootfd, rel, open_flags, 0);
154 }
155
156 if (fd < 0) {
157 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, errno,
158 "brix: bound handle reopen failed handle=%d path=%s",
159 handle_index, shared->path);
160 return NGX_DECLINED;
161 }
162
163 if (fstat(fd, &st) != 0) {
164 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, errno,
165 "brix: fstat failed on bound handle=%d",
166 handle_index);
167 close(fd);
168 return NGX_DECLINED;
169 }
170
171 if (st.st_dev != shared->device || st.st_ino != shared->inode) {
172 close(fd);
173 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
174 "brix: bound handle=%d path changed before reopen",
175 handle_index);
176 return NGX_DECLINED;
177 }
178
179 file = &ctx->files[handle_index];
180 file->fd = fd;
181 file->readable = 1;
182 file->writable = 0;
183 file->from_cache = shared->from_cache ? 1 : 0;
184 file->is_regular = S_ISREG(st.st_mode) ? 1 : 0;
185 file->device = st.st_dev;
186 file->inode = st.st_ino;
187 file->cached_size = (off_t) st.st_size;
188 file->read_last_end = -1;
189 file->read_ahead_end = 0;
190 file->bytes_read = 0;
191 file->bytes_written = 0;
192 file->open_time = ngx_current_msec;
193
194 if (brix_set_fhandle_path(ctx, c, handle_index, shared->path)
195 != NGX_OK)
196 {
197 brix_free_fhandle(ctx, handle_index);
198 return NGX_ERROR;
199 }
200
201 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
202 "brix: bound handle=%d reopened shared path=%s",
203 handle_index, shared->path);
204 return NGX_OK;
205}

207/* brix_ensure_read_handle — for an unbound session, NGX_OK iff the fd exists. For
208 * a bound secondary, re-validate against the primary's shared entry every time: if
209 * the primary closed/reused the slot, revoke (free + NGX_DECLINED); if stale but
210 * still published, reopen a fresh matching fd. Invariant: a bound secondary never
211 * reads from a handle no longer in the primary's published table. */
212
213ngx_int_t
214brix_ensure_read_handle(brix_ctx_t *ctx, ngx_connection_t *c,
215 int handle_index)
216{
217 brix_shared_handle_entry_t shared;
218
219 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES) {
220 return NGX_DECLINED;
221 }
222
223 if (!ctx->is_bound) {
224 /* fd >= 0 (POSIX) OR a driver-backed object/remote handle (no kernel fd;
225 * reads route through sd_obj.driver via the buffered serve path). */
226 return (ctx->files[handle_index].fd >= 0
227 || ctx->files[handle_index].sd_obj.driver != NULL)
228 ? NGX_OK : NGX_DECLINED;
229 }
230
231 /*
232 * A bound stream only follows the primary session's current handle table.
233 * Re-check the shared slot on every read request so a primary close, reuse,
234 * or session teardown immediately revokes the secondary's local fd.
235 */
236 if (!brix_session_handle_lookup_hint(ctx->bound_sessid, handle_index,
237 &ctx->files[handle_index]
238 .shared_handle_slot_hint,
239 &shared))
240 {
241 if (ctx->files[handle_index].fd >= 0) {
242 brix_free_fhandle(ctx, handle_index);
243 }
244 return NGX_DECLINED;
245 }
246
247 if (brix_local_file_matches_shared_handle(&ctx->files[handle_index],
248 &shared))
249 {
250 return NGX_OK;
251 }
252
253 if (ctx->files[handle_index].fd >= 0
254 || ctx->files[handle_index].path != NULL)
255 {
256 brix_free_fhandle(ctx, handle_index);
257 }
258
259 return brix_reopen_bound_read_handle(ctx, c, handle_index, &shared);
260}

284/* brix_free_fhandle — full teardown of one slot: close the fd, unpublish from
285 * shared memory (unbound only), unlink checkpoint/POSC staging files, free the heap
286 * path, and reset every field to its zero/null state. Called on session end, error
287 * recovery, and kXR_close completion. */
288
289void
290brix_free_fhandle(brix_ctx_t *ctx, int handle_index)
291{
292 brix_file_t *file;
293 ngx_stream_brix_srv_conf_t *conf;
294 const char *root_canon;
295 ngx_log_t *tlog;
296
297 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES) {
298 return;
299 }
300
301 file = &ctx->files[handle_index];
302
303 /* The teardown has no request ctx, but the srv conf still yields the export
304 * root_canon so handle-owned staging temps can be confined-unlinked. */
305 conf = ngx_stream_get_module_srv_conf(ctx->session, ngx_stream_brix_module);
306 root_canon = (conf != NULL) ? conf->common.root_canon : NULL;
307 tlog = (ctx->session != NULL && ctx->session->connection != NULL)
308 ? ctx->session->connection->log : NULL;
309
310 if (!ctx->is_bound && file->fd >= 0) {
311 brix_session_handle_unpublish(ctx->login.sessid, handle_index);
312 }
313
314 if (file->sd_obj.driver != NULL) {
315 /* Layer 3: a driver-backed handle owns its descriptor(s) and commits any
316 * pending metadata (e.g. catalog size/mtime) in its own close(); do NOT
317 * also raw-close file->fd, which is the driver's block-0 descriptor. */
318 (void) file->sd_obj.driver->close(&file->sd_obj);
319 file->sd_obj.driver = NULL;
320 file->sd_obj.inst = NULL;
321 file->sd_obj.state = NULL;
322 file->sd_obj.fd = -1;
323 } else if (file->fd >= 0) {
324 close(file->fd);
325 }
326
327 /* xmeta P3: flush + free the CSI engine (merges folded block CRCs
328 * into the file's metadata record). */
329 if (file->csi != NULL) {
330 brix_csi_close((brix_csi_t *) file->csi);
331 ngx_free(file->csi);
332 file->csi = NULL;
333 }
334
335 /*
336 * Checkpoint files are temporary rollback state owned by the handle.
337 * Normal close/free removes any abandoned checkpoint path.
338 */
339 if (file->ckp_path != NULL) {
340 fhandle_unlink_staging(file->ckp_path, root_canon, tlog);
341 ngx_free(file->ckp_path);
342 }
343
344 /*
345 * POSC cleanup: if posc_final_path is still set here the session ended
346 * without a clean kXR_close (disconnect, error, or abandon). Unlink the
347 * temp staging file so partial writes are not left on disk.
348 *
349 * EXCEPTION — upload resume (is_resume): the partial is deterministic and
350 * identity-keyed, so it is PRESERVED on a non-clean close. That is the
351 * whole mechanism: the same client reconnecting re-opens it in place (no
352 * truncate) and resumes from its offset. Abandoned partials are reclaimed
353 * by an operator glob-clean ("*.xrdresume.*.part") rather than here.
354 */
355 if (file->posc_final_path != NULL) {
356 if (file->path != NULL && !file->is_resume) {
357 fhandle_unlink_staging(file->path, root_canon, tlog);
358 }
359 ngx_free(file->posc_final_path);
360 }
361
362 if (file->path != NULL) {
363 ngx_free(file->path);
364 }
365
366 if (file->tpc_transfer_id != 0) {
367 (void) brix_tpc_registry_remove(file->tpc_transfer_id, NULL);
368 }
369
370 brix_wt_mark_clean(ctx, handle_index);
371
372 /* ZIP member handle: release the deflate inflate stream (if any) and clear
373 * the zip_* state so a reused slot cannot be mistaken for a zip handle. */
374 brix_zip_handle_cleanup(file);
375 file->zip_mode = 0;
376 file->zip_method = 0;
377 file->zip_data_off = 0;
378 file->zip_comp_size = 0;
379 file->zip_uncomp_size = 0;
380 file->zip_crc32 = 0;
381 file->zip_logical_pos = 0;
382 file->zip_comp_pos = 0;
383
384 file->fd = -1;
385 ngx_memzero(&file->sd_obj, sizeof(file->sd_obj)); /* Layer 3: clear driver obj */
386 file->shared_handle_slot_hint = -1; /* Phase 33 C2: drop the cached SHM slot */
387 file->readable = 0;
388 file->writable = 0;
389 file->from_cache = 0;
390 file->bytes_read = 0;
391 file->bytes_written = 0;
392 file->open_time = 0;
393 file->path = NULL;
394 /* §7: cancel SSI async-deferral timers + unregister the session before the
395 * .ssi pointer is dropped (runs on close + disconnect; NULL-safe). */
396 brix_ssi_handle_cleanup(file->ssi);
397 file->ssi = NULL; /* §7: SSI state is connection-pool owned */
398 file->is_regular = 0;
399 file->device = 0;
400 file->inode = 0;
401 file->cached_size = 0;
402 file->read_last_end = -1;
403 file->read_ahead_end = 0;
404 file->ckp_path = NULL;
405 file->ckp_size = 0;
406 file->posc_final_path = NULL;
407 file->is_resume = 0;
408 file->tpc_destination = 0;
409 file->tpc_armed = 0;
410 file->tpc_started = 0;
411 file->tpc_done = 0;
412 file->tpc_key[0] = '\0';
413 file->tpc_org[0] = '\0';
414 file->tpc_src_host[0] = '\0';
415 file->tpc_src_port = 0;
416 file->tpc_src_path[0] = '\0';
417 file->tpc_token_mode[0] = '\0';
418 file->tpc_transfer_id = 0;
419 file->wt_enabled = 0;
420 file->wt_policy = 0;
421 file->wt_mode_bits = 0;
422 file->wt_dirty_offset = -1;
423 file->wt_bytes_written = 0;
424 file->wt_flush_task = NULL;
425 file->wt_flush_pending = 0;
426 file->wrts_enabled = 0;
427 file->wrts_head = 0;
428 file->wrts_count = 0;
429 file->wrts_gen = 0;
430 brix_pgw_fob_reset(file);
431}

467/* brix_validate_read_handle — two phases: (1) brix_ensure_read_handle confirms
468 * the fd and refreshes bound secondaries; (2) the readable capability bit. 1 iff
469 * both pass, else logs + sends kXR_FileNotOpen/kXR_ServerError (phase 1) or
470 * kXR_NotAuthorized (phase 2). Re-checking the bit per read guards against drift. */
471
472ngx_flag_t
473brix_validate_read_handle(brix_ctx_t *ctx, ngx_connection_t *c,
474 int handle_index, const char *verb, ngx_uint_t op, ngx_int_t *rc)
475{
476 ngx_int_t ensure_rc;
477
478 ensure_rc = brix_ensure_read_handle(ctx, c, handle_index);
479 if (ensure_rc != NGX_OK) {
480 if (ensure_rc == NGX_ERROR) {
481 BRIX_BAIL_ERR(ctx, c, op, verb, "-", "-", kXR_ServerError,
482 "could not prepare file handle", rc);
483 }
484
485 BRIX_BAIL_ERR(ctx, c, op, verb, "-", "-", kXR_FileNotOpen,
486 "invalid file handle", rc);
487 }
488
489 /*
490 * Authorization and path checks happened at open time. Later handle I/O
491 * verifies the capability bit recorded on the handle rather than resolving
492 * the path again on every read.
493 */
494 if (!ctx->files[handle_index].readable) {
495 BRIX_BAIL_ERR(ctx, c, op, verb, ctx->files[handle_index].path, "-",
496 kXR_NotAuthorized, "file not open for reading", rc);
497 }
498
499 return 1;
500}
