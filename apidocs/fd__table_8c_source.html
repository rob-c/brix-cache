14/* brix_alloc_fhandle — return the first free slot (fd < 0) in ctx->files; the
15 * index becomes the one-byte on-wire fhandle, so it is bounded by BRIX_MAX_FILES.
16 * -1 when all slots are occupied. Single-owner per connection (event thread, no
17 * locking); a slot is only reused once its fd is closed. */
18int
19brix_alloc_fhandle(brix_ctx_t *ctx)
20{
21 int handle_index;
22
23 /*
24 * The XRootD wire handle stores this slot number in one byte. Keep all
25 * allocation/validation paths using the same bounded table.
26 */
27 for (handle_index = 0; handle_index < BRIX_MAX_FILES; handle_index++) {
28 if (ctx->files[handle_index].fd < 0) {
29 return handle_index;
30 }
31 }
32
33 return -1;
34}

36/* brix_ctx_has_open_file — 1 if any handle slot is occupied (fd >= 0), else 0.
37 * Used by the recv-loop drain gate: a connection with an open file is mid-transfer
38 * (a streaming read parked between kXR_read chunks), so a draining worker must let
39 * it finish rather than fast-teardown at the request boundary — a forced mid-stream
40 * reconnect loses the in-flight fill. Single-owner per connection (event thread). */
41int
42brix_ctx_has_open_file(const brix_ctx_t *ctx)
43{
44 int handle_index;
45
46 for (handle_index = 0; handle_index < BRIX_MAX_FILES; handle_index++) {
47 if (ctx->files[handle_index].fd >= 0) {
48 return 1;
49 }
50 }
51
52 return 0;
53}

55/* brix_set_fhandle_path — store a heap copy (ngx_alloc, NOT pool, so it outlives
56 * the kXR_open request) of the canonical path in the slot, freeing any prior path
57 * first; brix_free_fhandle owns the free. NGX_OK, or NGX_ERROR on bad bounds or
58 * allocation failure. Heap (not pool) avoids fragmentation across open/close cycles. */
59ngx_int_t
60brix_set_fhandle_path(brix_ctx_t *ctx, ngx_connection_t *c,
61 int handle_index, const char *path)
62{
63 char *path_copy;
64 size_t path_bytes;
65
66 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES
67 || path == NULL)
68 {
69 return NGX_ERROR;
70 }
71
72 path_bytes = ngx_strlen(path) + 1;
73 path_copy = ngx_alloc(path_bytes, c->log);
74 if (path_copy == NULL) {
75 return NGX_ERROR;
76 }
77
78 ngx_memcpy(path_copy, path, path_bytes);
79
80 if (ctx->files[handle_index].path != NULL) {
81 ngx_free(ctx->files[handle_index].path);
82 }
83
84 ctx->files[handle_index].path = path_copy;
85 return NGX_OK;
86}

88/* brix_local_file_matches_shared_handle — 1 iff the local fd still corresponds to
89 * the primary's published handle (fd valid + readable, same device/inode/path), used
90 * by brix_ensure_read_handle to decide whether a bound secondary must reopen. */
91
92static ngx_flag_t
93brix_local_file_matches_shared_handle(const brix_file_t *file,
94 const brix_shared_handle_entry_t *shared)
95{
96 if (file->fd < 0 || file->path == NULL || !file->readable
97 || !shared->readable)
98 {
99 return 0;
100 }
101
102 return file->device == shared->device
103 && file->inode == shared->inode
104 && ngx_strcmp(file->path, shared->path) == 0;
105}

107/* brix_reopen_bound_read_handle — reopen a bound secondary's stale fd to match the
108 * primary's shared entry via a cache-aware confined open (O_RDONLY|O_NOCTTY|O_CLOEXEC),
109 * refreshing all file metadata from the fresh fstat; NGX_DECLINED (revoke) if the
110 * filesystem object's device/inode changed. */
111
112static ngx_int_t
113brix_reopen_bound_read_handle(brix_ctx_t *ctx, ngx_connection_t *c,
114 int handle_index, const brix_shared_handle_entry_t *shared)
115{
116 ngx_stream_brix_srv_conf_t *conf;
117 brix_file_t *file;
118 struct stat st;
119 int fd;
120 int open_flags;
121
122 if (!shared->readable || shared->path[0] == '\0') {
123 return NGX_DECLINED;
124 }
125
126 conf = ngx_stream_get_module_srv_conf(ctx->session,
127 ngx_stream_brix_module);
128 if (conf == NULL) {
129 return NGX_ERROR;
130 }
131
132 /*
133 * The primary published a canonical, ACL-checked path. The secondary
134 * still opens it with the same confinement helper so a stale or corrupted
135 * shared-memory entry cannot escape the configured export root.
136 */
137 open_flags = O_RDONLY | O_NOCTTY | O_CLOEXEC;
138 if (shared->from_cache) {
139 fd = open(shared->path, open_flags | O_NOFOLLOW); /* vfs-seam-allow: separate server-managed cache-root domain (from_cache), opened as worker; O_NOFOLLOW so a stray symlink in the svc-owned cache tree is not followed */
140 } else {
141 /* shared->path is the absolute path; strip root_canon to get the
142 * path relative to rootfd for openat2 RESOLVE_BENEATH. */
143 const char *rel = shared->path;
144 size_t root_len = strlen(conf->common.root_canon);
145 if (root_len > 0
146 && ngx_strncmp((u_char *) shared->path,
147 (u_char *) conf->common.root_canon,
148 root_len) == 0
149 && shared->path[root_len] == '/')
150 {
151 rel = shared->path + root_len;
152 }
153 fd = brix_vfs_open_fd_at(conf->rootfd, rel, open_flags, 0);
154 }
155
156 if (fd < 0) {
157 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, errno,
158 "brix: bound handle reopen failed handle=%d path=%s",
159 handle_index, shared->path);
160 return NGX_DECLINED;
161 }
162
163 if (fstat(fd, &st) != 0) {
164 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, errno,
165 "brix: fstat failed on bound handle=%d",
166 handle_index);
167 close(fd);
168 return NGX_DECLINED;
169 }
170
171 if (st.st_dev != shared->device || st.st_ino != shared->inode) {
172 close(fd);
173 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
174 "brix: bound handle=%d path changed before reopen",
175 handle_index);
176 return NGX_DECLINED;
177 }
178
179 file = &ctx->files[handle_index];
180 file->fd = fd;
181 file->readable = 1;
182 file->writable = 0;
183 file->from_cache = shared->from_cache ? 1 : 0;
184 file->is_regular = S_ISREG(st.st_mode) ? 1 : 0;
185 file->device = st.st_dev;
186 file->inode = st.st_ino;
187 file->cached_size = (off_t) st.st_size;
188 file->read_last_end = -1;
189 file->read_ahead_end = 0;
190 file->bytes_read = 0;
191 file->bytes_written = 0;
192 file->open_time = ngx_current_msec;
193
194 if (brix_set_fhandle_path(ctx, c, handle_index, shared->path)
195 != NGX_OK)
196 {
197 brix_free_fhandle(ctx, handle_index);
198 return NGX_ERROR;
199 }
200
201 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
202 "brix: bound handle=%d reopened shared path=%s",
203 handle_index, shared->path);
204 return NGX_OK;
205}

207/* brix_ensure_read_handle — for an unbound session, NGX_OK iff the fd exists. For
208 * a bound secondary, re-validate against the primary's shared entry every time: if
209 * the primary closed/reused the slot, revoke (free + NGX_DECLINED); if stale but
210 * still published, reopen a fresh matching fd. Invariant: a bound secondary never
211 * reads from a handle no longer in the primary's published table. */
212
213ngx_int_t
214brix_ensure_read_handle(brix_ctx_t *ctx, ngx_connection_t *c,
215 int handle_index)
216{
217 brix_shared_handle_entry_t shared;
218
219 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES) {
220 return NGX_DECLINED;
221 }
222
223 if (!ctx->is_bound) {
224 /* fd >= 0 (POSIX) OR a driver-backed object/remote handle (no kernel fd;
225 * reads route through sd_obj.driver via the buffered serve path). */
226 return (ctx->files[handle_index].fd >= 0
227 || ctx->files[handle_index].sd_obj.driver != NULL)
228 ? NGX_OK : NGX_DECLINED;
229 }
230
231 /*
232 * A bound stream only follows the primary session's current handle table.
233 * Re-check the shared slot on every read request so a primary close, reuse,
234 * or session teardown immediately revokes the secondary's local fd.
235 */
236 if (!brix_session_handle_lookup_hint(ctx->bound_sessid, handle_index,
237 &ctx->files[handle_index]
238 .shared_handle_slot_hint,
239 &shared))
240 {
241 if (ctx->files[handle_index].fd >= 0) {
242 brix_free_fhandle(ctx, handle_index);
243 }
244 return NGX_DECLINED;
245 }
246
247 if (brix_local_file_matches_shared_handle(&ctx->files[handle_index],
248 &shared))
249 {
250 return NGX_OK;
251 }
252
253 if (ctx->files[handle_index].fd >= 0
254 || ctx->files[handle_index].path != NULL)
255 {
256 brix_free_fhandle(ctx, handle_index);
257 }
258
259 return brix_reopen_bound_read_handle(ctx, c, handle_index, &shared);
260}

266 * active fd; otherwise logs + sends kXR_FileNotOpen and returns 0. The prerequisite
267 * check before any read/write capability check. */
268
269ngx_flag_t
270brix_validate_file_handle(brix_ctx_t *ctx, ngx_connection_t *c,
271 int handle_index, const char *verb, ngx_uint_t op, ngx_int_t *rc)
272{
273 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES
274 || (ctx->files[handle_index].fd < 0
275 && ctx->files[handle_index].sd_obj.driver == NULL
276 && ctx->files[handle_index].writer == NULL))
277 {
278 /* A driver-backed handle (object/remote backend) is "open" with no kernel
279 * fd — data I/O routes through sd_obj.driver, so fd < 0 is normal there.
280 * A whole-object staged write handle (writer != NULL) is likewise "open"
281 * with no fd — byte I/O routes through the write session (phase-70). */
282 BRIX_BAIL_ERR(ctx, c, op, verb, "-", "-",
283 kXR_FileNotOpen, "invalid file handle", rc);
284 }
285
286 return 1;
287}

289/* brix_validate_read_handle — two phases: (1) brix_ensure_read_handle confirms
290 * the fd and refreshes bound secondaries; (2) the readable capability bit. 1 iff
291 * both pass, else logs + sends kXR_FileNotOpen/kXR_ServerError (phase 1) or
292 * kXR_NotAuthorized (phase 2). Re-checking the bit per read guards against drift. */
293
294ngx_flag_t
295brix_validate_read_handle(brix_ctx_t *ctx, ngx_connection_t *c,
296 int handle_index, const char *verb, ngx_uint_t op, ngx_int_t *rc)
297{
298 ngx_int_t ensure_rc;
299
300 ensure_rc = brix_ensure_read_handle(ctx, c, handle_index);
301 if (ensure_rc != NGX_OK) {
302 if (ensure_rc == NGX_ERROR) {
303 BRIX_BAIL_ERR(ctx, c, op, verb, "-", "-", kXR_ServerError,
304 "could not prepare file handle", rc);
305 }
306
307 BRIX_BAIL_ERR(ctx, c, op, verb, "-", "-", kXR_FileNotOpen,
308 "invalid file handle", rc);
309 }
310
311 /*
312 * Authorization and path checks happened at open time. Later handle I/O
313 * verifies the capability bit recorded on the handle rather than resolving
314 * the path again on every read.
315 */
316 if (!ctx->files[handle_index].readable) {
317 BRIX_BAIL_ERR(ctx, c, op, verb, ctx->files[handle_index].path, "-",
318 kXR_NotAuthorized, "file not open for reading", rc);
319 }
320
321 return 1;
322}
