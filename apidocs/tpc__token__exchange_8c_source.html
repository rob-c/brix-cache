1/*
2 * tpc_token_exchange.c — RFC 8693 OAuth2 token-exchange path for native XRootD
3 * TPC outbound delegated authentication (split from tpc_token.c, phase-79).
4 *
5 * WHAT: Fetches a delegated OAuth2 access token by exchanging the destination's
6 * own bearer (subject) token at an external OAuth2 token endpoint, via a
7 * fork/exec of curl. The public entry point tpc_token_rfc8693 is invoked
8 * by tpc_fetch_delegated_token (tpc_token.c) for token_mode
9 * "token-exchange".
10 * WHY: When a TPC source requires the destination to authenticate as a
11 * different identity than the client, RFC 8693 converts the destination's
12 * bearer file subject token into an access token scoped for the remote
13 * origin. Keeping this path in its own file holds each half of the token
14 * fetcher under the file-size cap and focused on one delegation mode.
15 * HOW: read subject token → stage the form body in a private temp file →
16 * build the curl argv → run curl and capture stdout → parse the JSON
17 * access_token (shared parser in tpc_token.c). Every step records
18 * err_msg / xrd_error on failure and the body temp file is unlinked on
19 * every exit path.
20 */

38/* WHAT: Reads the local bearer (subject) token from tpc_outbound_bearer_file
39 * into subject_token. Returns 0 on success, -1 with err_msg/xrd_error set
40 * (kXR_ArgInvalid when unconfigured/empty, kXR_IOError on read failure).
41 * WHY: The RFC 8693 exchange needs the destination's own bearer as the subject
42 * token; isolating the read keeps its errno-dependent error mapping out of the
43 * orchestrator.
44 * HOW: brix_token_read_file(); on non-NGX_OK map errno==EINVAL to the
45 * "not configured or empty" ArgInvalid case, anything else to IOError. */
46static int
47tpc_rfc8693_read_subject(brix_tpc_pull_t *t, char *subject_token,
48 size_t subject_token_sz)
49{
50 if (brix_token_read_file(&t->conf->tpc_outbound_bearer_file,
51 (u_char *) subject_token, subject_token_sz,
52 NULL, NULL, "TPC token")
53 != NGX_OK)
54 {
55 if (errno == EINVAL) {
56 snprintf(t->err_msg, sizeof(t->err_msg),
57 "TPC token: bearer file not configured or empty");
58 t->xrd_error = kXR_ArgInvalid;
59 } else {
60 snprintf(t->err_msg, sizeof(t->err_msg),
61 "TPC token: cannot read bearer file: %s", strerror(errno));
62 t->xrd_error = kXR_IOError;
63 }
64 return -1;
65 }
66
67 return 0;
68}

70/* WHAT: Builds the form-urlencoded RFC 8693 request body from subject_token and
71 * stages it in a private temp file, returning the temp path in body_file.
72 * Returns 0 on success, -1 with err_msg/xrd_error=kXR_IOError on temp-file
73 * failure.
74 * WHY: Staging the body in a 0600 file (unpredictable name, exclusive create)
75 * in a private 0700 dir instead of on curl's command line keeps the
76 * bearer/subject token out of the process listing (proc cmdline). The caller
77 * unlinks body_file on every exit path.
78 * HOW: ngx_snprintf the grant_type/subject_token/resource/audience/scope body;
79 * hand the exact formatted length to the shared brix_cred_stage_write(), which
80 * stages it in the per-uid /dev/shm/brix-creds.<euid> tmpfs dir (never
81 * world-traversable /tmp — fail-closed) and returns the resolved path.
82 * subject_token is a JWT (URL-safe base64), so it is form-safe without
83 * URL-encoding. */
84static int
85tpc_rfc8693_stage_body(brix_tpc_pull_t *t, const char *subject_token,
86 char *body_file, size_t body_file_sz)
87{
88 char body_buf[4096];
89 u_char *p;
90
91 p = ngx_snprintf((u_char *) body_buf, sizeof(body_buf),
92 "grant_type=urn:ietf:params:oauth:grant-type:"
93 "token-exchange"
94 "&subject_token=%s"
95 "&resource=%s"
96 "&audience=%s"
97 "&scope=%s",
98 subject_token,
99 t->src_host,
100 t->src_host,
101 t->token_scope);
102
103 /* A-5: stage in the shared private 0700 tmpfs facility, fail-closed. */
104 if (brix_cred_stage_write("tpc_token_body_", body_buf,
105 (size_t) (p - (u_char *) body_buf),
106 body_file, body_file_sz) != 0)
107 {
108 OPENSSL_cleanse(body_buf, sizeof(body_buf));
109 snprintf(t->err_msg, sizeof(t->err_msg),
110 "TPC token: cannot stage credential body privately (%s) — "
111 "refusing to fall back to world-readable /tmp",
112 strerror(errno));
113 t->xrd_error = kXR_IOError;
114 return -1;
115 }
116
117 /* The live subject token has been persisted to the 0600 file; scrub the
118 * heap copy so it does not linger in worker memory (shared with A-4). */
119 OPENSSL_cleanse(body_buf, sizeof(body_buf));
120
121 return 0;
122}

124/* WHAT: Fills curl_argv with the token-exchange curl invocation (no shell,
125 * execvp-style) sourcing the form body from body_file.
126 * WHY: Building the argv in one place keeps the curl option set (and the
127 * security-relevant -f fail-on-HTTP-error, "--" options terminator, and optional
128 * client basic-auth) reviewable and the orchestrator flat.
129 * HOW: append curl -s -S -f -X POST -H Content-Type; when both client_id and
130 * client_secret are configured, append -u id secret (curl joins per its -u
131 * convention); append -d body_file, the "--" end-of-options terminator, the
132 * token endpoint URL, and a NULL terminator. */
133static void
134tpc_rfc8693_build_argv(brix_tpc_pull_t *t, char **curl_argv,
135 const char *body_file)
136{
137 int argc = 0;
138
139 /*
140 * -s -S : silent but still print errors; -f : fail (non-zero exit) on
141 * HTTP >= 400 so the caller's exit-status check catches auth failures.
142 */
143 curl_argv[argc++] = "curl";
144 curl_argv[argc++] = "-s";
145 curl_argv[argc++] = "-S";
146 curl_argv[argc++] = "-f";
147 curl_argv[argc++] = "-X";
148 curl_argv[argc++] = "POST";
149 curl_argv[argc++] = "-H";
150 curl_argv[argc++] = "Content-Type: application/x-www-form-urlencoded";
151
152 /* Optional OAuth2 client basic auth: curl -u "id:secret" — only added when
153 * both halves are configured. NOTE: -u takes a single "id:secret" arg, so
154 * these two array slots become two separate argv entries that curl joins
155 * via the -u/value convention. */
156 if (t->conf->tpc_outbound_client_id.len > 0
157 && t->conf->tpc_outbound_client_secret.len > 0) {
158 curl_argv[argc++] = "-u";
159 curl_argv[argc++] = (char *) t->conf->tpc_outbound_client_id.data;
160 curl_argv[argc++] = (char *) t->conf->tpc_outbound_client_secret.data;
161 }
162
163 /* -d data: the form body, sourced from the staged temp file path. */
164 curl_argv[argc++] = "-d";
165 curl_argv[argc++] = (char *) body_file;
166 /* W3 — end-of-options terminator so the endpoint URL can never be parsed
167 * as a curl option even if a misconfigured endpoint begins with '-'. */
168 curl_argv[argc++] = "--";
169 curl_argv[argc++] = (char *) t->conf->tpc_outbound_token_endpoint.data;
170 curl_argv[argc++] = NULL;
171}

173/* WHAT: Runs curl synchronously (via the shared SIGCHLD-safe capture helper),
174 * captures its stdout into buf, and unlinks the staged body file. Returns 0 on
175 * success, -1 with err_msg/xrd_error set on subprocess failure or non-zero curl
176 * exit.
177 * WHY: Centralizing the capture + status mapping + temp-file cleanup keeps every
178 * exit path unlinking body_file exactly once. A non-zero rc means pipe/fork
179 * failed or curl was signal-killed (kXR_ServerError); a non-zero child exit
180 * (including curl -f on HTTP >= 400) is an auth failure (kXR_AuthFailed).
181 * HOW: brix_subprocess_capture(); on rc!=0 unlink+ServerError; on child exit!=0
182 * unlink+AuthFailed; otherwise unlink and return 0 with the reply in buf. */
183static int
184tpc_rfc8693_run_curl(brix_tpc_pull_t *t, char **curl_argv,
185 const char *body_file, char *buf, size_t bufsz)
186{
187 int ec = -1;
188
189 if (brix_subprocess_capture(curl_argv, buf, bufsz, NULL, &ec) != 0) {
190 snprintf(t->err_msg, sizeof(t->err_msg),
191 "TPC token: token-exchange subprocess failed "
192 "(pipe/fork or signal)");
193 unlink(body_file); /* vfs-seam-allow: staged credential temp, not export storage */
194 t->xrd_error = kXR_ServerError;
195 return -1;
196 }
197 if (ec != 0) {
198 snprintf(t->err_msg, sizeof(t->err_msg),
199 "TPC token: token exchange failed (curl exit %d)", ec);
200 unlink(body_file); /* vfs-seam-allow: staged credential temp, not export storage */
201 t->xrd_error = kXR_AuthFailed;
202 return -1;
203 }
204
205 /* Success: body file no longer needed; remove before parsing the reply. */
206 unlink(body_file); /* vfs-seam-allow: staged credential temp, not export storage */
207 return 0;
208}

210/* WHAT: Fetches a delegated OAuth2/OIDC access token via an RFC 8693
211 * token-exchange POST to the external OAuth2 endpoint (fork/exec curl). Returns
212 * 0 on success (token in token_out), -1 with err_msg/xrd_error set on failure.
213 * WHY: When the TPC source requires a delegated token (a different identity from
214 * the client), RFC 8693 converts the destination's bearer file subject token
215 * into an access token scoped for the remote origin. The subject token, its
216 * staging, the curl argv, and the run/cleanup are each isolated so the exchange
217 * reads as a short linear sequence.
218 * HOW: tpc_rfc8693_read_subject() -> tpc_rfc8693_stage_body() ->
219 * tpc_rfc8693_build_argv() -> tpc_rfc8693_run_curl() ->
220 * tpc_token_parse_access_token(). Any step's failure returns -1 with the error
221 * already recorded. */
222int
223tpc_token_rfc8693(brix_tpc_pull_t *t, char *token_out, size_t token_out_sz)
224{
225 char buf[TPC_TOKEN_MAX_LEN + 256];
226 char *curl_argv[20];
227 char body_file[NGX_MAX_PATH];
228 char subject_token[TPC_TOKEN_MAX_LEN];
229
230 if (tpc_rfc8693_read_subject(t, subject_token, sizeof(subject_token)) != 0) {
231 return -1;
232 }
233
234 if (tpc_rfc8693_stage_body(t, subject_token, body_file, sizeof(body_file))
235 != 0)
236 {
237 return -1;
238 }
239
240 tpc_rfc8693_build_argv(t, curl_argv, body_file);
241
242 if (tpc_rfc8693_run_curl(t, curl_argv, body_file, buf, sizeof(buf)) != 0) {
243 return -1;
244 }
245
246 if (tpc_token_parse_access_token(buf, token_out, token_out_sz) != 0) {
247 t->xrd_error = kXR_AuthFailed;
248 return -1;
249 }
250
251 return 0;
252}
