33/*
34 * Extract the requested checksum algorithm from the dirlist request payload.
35 * The client may append a CGI parameter like "?cks.type=sha256" to specify
36 * which algorithm to use. If no parameter is present, defaults to adler32.
37 * Returns NGX_OK on success, NGX_DECLINED if an unsupported algorithm was
38 * requested (bad_algo contains the rejected name for error reporting).
39 */
40
41ngx_int_t
42brix_dirlist_checksum_algorithm(const u_char *payload, size_t payload_len,
43 char *algo, size_t algo_sz, char *bad_algo, size_t bad_algo_sz)
44{
45 const u_char *qmark, *p, *end;
46
47 ngx_cpystrn((u_char *) algo, (u_char *) "adler32", algo_sz);
48 bad_algo[0] = '\0';
49
50 if (payload == NULL || payload_len == 0) {
51 return NGX_OK;
52 }
53
54 if (payload[payload_len - 1] == '\0') {
55 payload_len--;
56 }
57
58 qmark = memchr(payload, '?', payload_len);
59 if (qmark == NULL) {
60 return NGX_OK;
61 }
62
63 p = qmark + 1;
64 end = payload + payload_len;
65
66 while (p < end) {
67 const u_char *amp;
68 size_t field_len;
69 static const char key[] = "cks.type=";
70
71 amp = memchr(p, '&', (size_t) (end - p));
72 field_len = amp ? (size_t) (amp - p) : (size_t) (end - p);
73
74 if (field_len > sizeof(key) - 1
75 && ngx_strncmp(p, key, sizeof(key) - 1) == 0)
76 {
77 const u_char *value = p + sizeof(key) - 1;
78 size_t value_len = field_len - (sizeof(key) - 1);
79 ngx_int_t rc;
80
81 rc = brix_dirlist_parse_algorithm(value, value_len,
82 algo, algo_sz);
83 if (rc == NGX_DECLINED || rc == NGX_ERROR) {
84 size_t copy_len = value_len;
85
86 if (copy_len >= bad_algo_sz) {
87 copy_len = bad_algo_sz - 1;
88 }
89 ngx_memcpy(bad_algo, value, copy_len);
90 bad_algo[copy_len] = '\0';
91 return NGX_DECLINED;
92 }
93
94 return NGX_OK;
95 }
96
97 if (amp == NULL) {
98 break;
99 }
100 p = amp + 1;
101 }
102
103 return NGX_OK;
104}

106/*
107 * Compute a checksum token for a single directory entry.
108 * Opens the file, computes the digest using the requested algorithm (adler32,
109 * crc32, crc32c, md5, sha1, or sha256), and writes the result as
110 * "algo:hexvalue" to out. Returns "algo:none" if the file is not regular or
111 * checksum fails.
112 */
113void
114brix_dirlist_checksum_token(ngx_log_t *log, int dfd,
115 const char *name, const char *path, const struct stat *st,
116 const char *algo, char *out, size_t outsz)
117{
118 int fd;
119
120 if (!S_ISREG(st->st_mode)) {
121 snprintf(out, outsz, "%s:none", algo);
122 return;
123 }
124
125 fd = openat(dfd, name, O_RDONLY | O_CLOEXEC | O_NOFOLLOW); /* vfs-seam-allow: dirfd-relative open within an already-VFS-opened confined dir stream (brix_vfs_dir_fd) */
126 if (fd < 0) {
127 snprintf(out, outsz, "%s:none", algo);
128 return;
129 }
130
131 {
132 brix_integrity_info_t info;
133 brix_integrity_opts_t iopts;
134
135 ngx_memzero(&iopts, sizeof(iopts));
136 iopts.allow_xattr_cache = 1;
137 iopts.update_xattr_cache = 1;
138
139 if (brix_integrity_get_fd(log, fd, NULL, path, algo, &iopts, &info) == NGX_OK) {
140 snprintf(out, outsz, "%s:%s", info.alg_name, info.hex);
141 } else {
142 snprintf(out, outsz, "%s:none", algo);
143 }
144 }
145
146 close(fd);
147}

149/*
150 * Format a dcksm stat body for one directory entry.
151 * Produces the 9-field line: inode size flags mtime ctime atime mode uid gid.
152 * Sets kXR_isDir or kXR_other flags based on file type, and readable/writable/xset
153 * flags based on permission bits. This is used when kXR_dcksm is requested.
154 */
155void
156brix_dirlist_make_dcksm_stat_body(const struct stat *st, char *out,
157 size_t outsz)
158{
159 int flags = 0;
160
161 if (S_ISDIR(st->st_mode)) {
162 flags |= kXR_isDir;
163
164 } else if (!S_ISREG(st->st_mode)) {
165 flags |= kXR_other;
166 }
167
168 if (st->st_mode & (S_IRUSR | S_IRGRP | S_IROTH)) {
169 flags |= kXR_readable;
170 }
171
172 if (st->st_mode & (S_IWUSR | S_IWGRP | S_IWOTH)) {
173 flags |= kXR_writable;
174 }
175
176 if (st->st_mode & (S_IXUSR | S_IXGRP | S_IXOTH)) {
177 flags |= kXR_xset;
178 }
179
180 snprintf(out, outsz, "%llu %lld %d %ld %ld %ld %04o %u %u",
181 (unsigned long long) st->st_ino,
182 (long long) st->st_size,
183 flags,
184 (long) st->st_mtime,
185 (long) st->st_ctime,
186 (long) st->st_atime,
187 (unsigned int) (st->st_mode & 07777),
188 (unsigned int) st->st_uid,
189 (unsigned int) st->st_gid);
190}
