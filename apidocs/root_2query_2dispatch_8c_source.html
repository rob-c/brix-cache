1/*
2 * WHAT: Dispatch the kXR_query opcode to one of 14+ sub-handlers based on the request's infotype field. Each infotype value routes to a specialized handler: checksum computation, filesystem space inquiry, server configuration query, statistics collection, extended attribute listing, file info retrieval, and various opaque/visa extension hooks. Returns kXR_Unsupported for unrecognized infotypes with debug logging.
3 */
4
5/* WHY: The XRootD query protocol consolidates diverse server-side operations into a single opcode (kXR_query) rather than requiring separate opcodes for each operation type. This reduces wire protocol complexity while enabling flexible sub-protocol extension — new query types can be added by registering handlers without changing the core dispatcher or client code. The typedef struct brix_ckscan_aio_t and extern declarations cross-reference checksum scan AIO infrastructure defined in separate files (checksum_ckscan_common.c, checksum_ckscan_async.c) to maintain modular architecture while keeping dispatcher self-contained. */
6
7/* HOW: Sequential infotype comparison chain using ntohs() to convert big-endian 16-bit value from wire format into host byte order. Each if-block calls a dedicated handler function (brix_query_prep_status through brix_query_opaqug) — handlers return NGX_OK/NGX_ERROR or send error responses directly. The last kXR_Qvisa case includes an additional precondition check: ctx->recv.cur_dlen must be zero (no pending data length) before proceeding to visa query. After all comparisons, debug log the unsupported infotype value and call brix_send_error() with kXR_Unsupported status. */

16/* ---- Route the common single-target query infotypes ----
17 *
18 * WHAT: Map one of the nine plain query infotypes (prep-status, checksum,
19 * ckscan, space, config, stats, xattr, finfo, fsinfo) to its dedicated handler
20 * and return that handler's status (NGX_OK/NGX_ERROR, or a value produced by an
21 * error-send). Returns the NGX_DECLINED sentinel when infotype belongs to none
22 * of these — signalling the caller to try the extended group.
23 *
24 * WHY: These infotypes route unconditionally to a single handler with no
25 * per-type preconditions, so grouping them keeps the top-level dispatcher's
26 * cyclomatic complexity within budget while preserving the exact 1:1 routing.
27 * NGX_DECLINED is used as the miss sentinel (never NGX_OK) so a handler that
28 * legitimately returns NGX_OK is never misread as "not matched, keep looking".
29 *
30 * HOW:
31 * 1. Compare infotype against each of the nine query codes in order.
32 * 2. On the first match, tail-call the matching handler and return its status.
33 * 3. If nothing matches, return NGX_DECLINED.
34 */
35static ngx_int_t
36brix_query_route_common(brix_ctx_t *ctx, ngx_connection_t *c,
37 ngx_stream_brix_srv_conf_t *conf, xrdw_query_req_t *req, uint16_t infotype)
38{
39 if (infotype == kXR_QPrep) {
40 return brix_query_prep_status(ctx, c, conf);
41 }
42
43 if (infotype == kXR_Qcksum) {
44 return brix_query_cksum(ctx, c, conf, req);
45 }
46
47 if (infotype == kXR_Qckscan) {
48 return brix_query_ckscan(ctx, c, conf);
49 }
50
51 if (infotype == kXR_Qspace) {
52 return brix_query_space(ctx, c, conf);
53 }
54
55 if (infotype == kXR_Qconfig) {
56 return brix_query_config(ctx, c, conf);
57 }
58
59 if (infotype == kXR_QStats) {
60 return brix_query_stats(ctx, c);
61 }
62
63 if (infotype == kXR_Qxattr) {
64 return brix_query_xattr(ctx, c, conf);
65 }
66
67 if (infotype == kXR_QFinfo) {
68 return brix_query_finfo(ctx, c);
69 }
70
71 if (infotype == kXR_QFSinfo) {
72 return brix_query_fsinfo(ctx, c, conf);
73 }
74
75 return NGX_DECLINED;

78/* ---- Route the extended query infotypes with per-type preconditions ----
79 *
80 * WHAT: Map the visa and opaque-family infotypes (visa, opaque, opaquf, opaqug)
81 * to their handlers, applying each type's precondition, and return the resulting
82 * status. Returns the NGX_DECLINED sentinel when infotype belongs to none of
83 * these.
84 *
85 * WHY: Unlike the common group, these infotypes carry extra routing logic — visa
86 * rejects a non-empty data length, and opaqug diverts SSI response-wait/cancel
87 * requests to the SSI engine before falling back to the generic handler. Keeping
88 * them in a separate helper isolates that branching so neither this helper nor
89 * the top-level dispatcher exceeds the complexity budget, with identical routing.
90 *
91 * HOW:
92 * 1. For kXR_Qvisa, reject a non-zero cur_dlen via BRIX_RETURN_ERR, else call
93 * brix_query_visa.
94 * 2. For kXR_Qopaque / kXR_Qopaquf, tail-call the matching handler.
95 * 3. For kXR_Qopaqug, route to the SSI engine when fhandle[0] names an open SSI
96 * session, else fall back to brix_query_opaqug.
97 * 4. If nothing matches, return NGX_DECLINED.
98 */
99static ngx_int_t
100brix_query_route_extended(brix_ctx_t *ctx, ngx_connection_t *c,
101 ngx_stream_brix_srv_conf_t *conf, xrdw_query_req_t *req, uint16_t infotype)
102{
103 if (infotype == kXR_Qvisa) {
104 if (ctx->recv.cur_dlen != 0) {
105 BRIX_RETURN_ERR(ctx, c, BRIX_OP_QUERY_VISA, "QUERY",
106 "-", "visa", kXR_ArgInvalid,
107 "Invalid information query type code");
108 }
109
110 return brix_query_visa(ctx, c, req);
111 }
112
113 if (infotype == kXR_Qopaque) {
114 return brix_query_opaque(ctx, c);
115 }
116
117 if (infotype == kXR_Qopaquf) {
118 return brix_query_opaquf(ctx, c, conf);
119 }
120
121 if (infotype == kXR_Qopaqug) {
122 /* §7 XrdSsi: libXrdSsi waits for / cancels a response with a
123 * File::Fcntl == kXR_query(kXR_Qopaqug) whose fhandle is the SSI session
124 * and whose body is an XrdSsiRRInfo. Route those to the SSI engine; all
125 * other opaqug queries keep the generic handler. */
126 int fh = (int) (unsigned char) req->fhandle[0];
127 if (fh >= 0 && fh < BRIX_MAX_FILES && ctx->files[fh].ssi != NULL) {
128 return brix_ssi_query(ctx, c, fh, ctx->recv.payload, ctx->recv.cur_dlen);
129 }
130 return brix_query_opaqug(ctx, c, req);
131 }
132
133 return NGX_DECLINED;
134}

136/* ---- Top-level kXR_query dispatcher ----
137 *
138 * WHAT: Unpack the kXR_query request header, select the query sub-protocol by
139 * infotype, and return the chosen handler's status. Replies kXR_Unsupported for
140 * unrecognized infotypes after a debug log line.
141 *
142 * WHY: kXR_query multiplexes many server operations onto one opcode; this
143 * function is the single routing point. Splitting the routing across two grouped
144 * helpers keeps each function within the complexity budget without altering the
145 * wire-level behavior clients depend on.
146 *
147 * HOW:
148 * 1. Unpack the request body into req and read its infotype.
149 * 2. Try the common group; return its status unless it declines.
150 * 3. Try the extended group; return its status unless it declines.
151 * 4. If both decline, debug-log the infotype and send kXR_Unsupported.
152 */
153ngx_int_t
154brix_handle_query(brix_ctx_t *ctx, ngx_connection_t *c,
155 ngx_stream_brix_srv_conf_t *conf)
156{
157 xrdw_query_req_t req;
158 uint16_t infotype;
159 ngx_int_t rc;
160
161 xrdw_query_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
162 infotype = req.infotype;
163
164 rc = brix_query_route_common(ctx, c, conf, &req, infotype);
165 if (rc != NGX_DECLINED) {
166 return rc;
167 }
168
169 rc = brix_query_route_extended(ctx, c, conf, &req, infotype);
170 if (rc != NGX_DECLINED) {
171 return rc;
172 }
173
174 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
175 "brix: kXR_query unsupported infotype=%d",
176 (int) infotype);
177 return brix_send_error(ctx, c, kXR_Unsupported,
178 "query type not supported");
179}
