1/*
2 * WHAT: Dispatch the kXR_query opcode to one of 14+ sub-handlers based on the request's infotype field. Each infotype value routes to a specialized handler: checksum computation, filesystem space inquiry, server configuration query, statistics collection, extended attribute listing, file info retrieval, and various opaque/visa extension hooks. Returns kXR_Unsupported for unrecognized infotypes with debug logging.
3 */
4
5/* WHY: The XRootD query protocol consolidates diverse server-side operations into a single opcode (kXR_query) rather than requiring separate opcodes for each operation type. This reduces wire protocol complexity while enabling flexible sub-protocol extension — new query types can be added by registering handlers without changing the core dispatcher or client code. The typedef struct brix_ckscan_aio_t and extern declarations cross-reference checksum scan AIO infrastructure defined in separate files (checksum_ckscan_common.c, checksum_ckscan_async.c) to maintain modular architecture while keeping dispatcher self-contained. */
6
7/* HOW: Sequential infotype comparison chain using ntohs() to convert big-endian 16-bit value from wire format into host byte order. Each if-block calls a dedicated handler function (brix_query_prep_status through brix_query_opaqug) — handlers return NGX_OK/NGX_ERROR or send error responses directly. The last kXR_Qvisa case includes an additional precondition check: ctx->recv.cur_dlen must be zero (no pending data length) before proceeding to visa query. After all comparisons, debug log the unsupported infotype value and call brix_send_error() with kXR_Unsupported status. */
8
9#include "query_internal.h"
10#include "protocols/ssi/ssi.h" /* §7 XrdSsi response-wait via kXR_query(Qopaqug) */

16ngx_int_t
17brix_handle_query(brix_ctx_t *ctx, ngx_connection_t *c,
18 ngx_stream_brix_srv_conf_t *conf)
19{
20 xrdw_query_req_t req;
21 uint16_t infotype;
22
23 xrdw_query_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
24 infotype = req.infotype;
25
26 if (infotype == kXR_QPrep) {
27 return brix_query_prep_status(ctx, c, conf);
28 }
29
30 if (infotype == kXR_Qcksum) {
31 return brix_query_cksum(ctx, c, conf, &req);
32 }
33
34 if (infotype == kXR_Qckscan) {
35 return brix_query_ckscan(ctx, c, conf);
36 }
37
38 if (infotype == kXR_Qspace) {
39 return brix_query_space(ctx, c, conf);
40 }
41
42 if (infotype == kXR_Qconfig) {
43 return brix_query_config(ctx, c, conf);
44 }
45
46 if (infotype == kXR_QStats) {
47 return brix_query_stats(ctx, c);
48 }
49
50 if (infotype == kXR_Qxattr) {
51 return brix_query_xattr(ctx, c, conf);
52 }
53
54 if (infotype == kXR_QFinfo) {
55 return brix_query_finfo(ctx, c);
56 }
57
58 if (infotype == kXR_QFSinfo) {
59 return brix_query_fsinfo(ctx, c, conf);
60 }
61
62 if (infotype == kXR_Qvisa) {
63 if (ctx->recv.cur_dlen != 0) {
64 BRIX_RETURN_ERR(ctx, c, BRIX_OP_QUERY_VISA, "QUERY",
65 "-", "visa", kXR_ArgInvalid,
66 "Invalid information query type code");
67 }
68
69 return brix_query_visa(ctx, c, &req);
70 }
71
72 if (infotype == kXR_Qopaque) {
73 return brix_query_opaque(ctx, c);
74 }
75
76 if (infotype == kXR_Qopaquf) {
77 return brix_query_opaquf(ctx, c, conf);
78 }
79
80 if (infotype == kXR_Qopaqug) {
81 /* §7 XrdSsi: libXrdSsi waits for / cancels a response with a
82 * File::Fcntl == kXR_query(kXR_Qopaqug) whose fhandle is the SSI session
83 * and whose body is an XrdSsiRRInfo. Route those to the SSI engine; all
84 * other opaqug queries keep the generic handler. */
85 int fh = (int) (unsigned char) req.fhandle[0];
86 if (fh >= 0 && fh < BRIX_MAX_FILES && ctx->files[fh].ssi != NULL) {
87 return brix_ssi_query(ctx, c, fh, ctx->recv.payload, ctx->recv.cur_dlen);
88 }
89 return brix_query_opaqug(ctx, c, &req);
90 }
91
92 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
93 "brix: kXR_query unsupported infotype=%d",
94 (int) infotype);
95 return brix_send_error(ctx, c, kXR_Unsupported,
96 "query type not supported");
97}
