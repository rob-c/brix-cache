8/*
9 * Phase 40 (a): build a per-transfer-UNIQUE temp path
10 * "<dst>.xrdcp-tmp.<pid>.<seq>". The pid alone is NOT unique under `-j`, whose
11 * batch workers are threads sharing one pid — two same-basename sources copied
12 * into one directory would otherwise collide on an identical temp name and
13 * interleave-corrupt it (then rename a garbage file into place, reported as
14 * success). A process-wide atomic sequence makes every concurrent transfer's
15 * temp distinct; pids already differ across separate processes. Returns 0, or -1
16 * if the composed path would not fit.
17 */
18int
19make_temp_path(const char *dst, char *out, size_t outsz)
20{
21 static atomic_ulong seq;
22 unsigned long s = atomic_fetch_add(&seq, 1ul);
23 if ((size_t) snprintf(out, outsz, "%s.xrdcp-tmp.%ld.%lu",
24 dst, (long) getpid(), s) >= outsz) {
25 return -1;
26 }
27 return 0;
28}

31/*
32 * open_download_temp — create a fresh private temp next to `dst` for the
33 * download+atomic-rename, and hand back its fd and name.
34 *
35 * WHY: the temp name is predictable (pid + counter), so an attacker with write
36 * access to the destination directory could pre-create it as a symlink and
37 * redirect our O_TRUNC onto a victim-owned file. O_EXCL refuses a
38 * pre-existing name and O_NOFOLLOW refuses a symlink, closing that race; we
39 * regenerate the name and retry on a stale collision so a leftover temp from
40 * a killed run doesn't wedge the transfer. Returns an fd (caller closes) and
41 * fills tmp[], or -1 with *st set.
42 */
43int
44open_download_temp(const char *dst, char *tmp, size_t tmpsz, brix_status *st)
45{
46 int attempt;
47
48 for (attempt = 0; attempt < 64; attempt++) {
49 int fd;
50 if (make_temp_path(dst, tmp, tmpsz) != 0) {
51 brix_status_set(st, XRDC_EUSAGE, 0, "destination path too long: %s", dst);
52 return -1;
53 }
54 fd = open(tmp, O_WRONLY | O_CREAT | O_EXCL | O_NOFOLLOW, 0644);
55 if (fd >= 0) {
56 return fd;
57 }
58 if (errno != EEXIST) {
59 brix_status_set(st, XRDC_EUSAGE, errno, "open %s: %s",
60 tmp, strerror(errno));
61 return -1;
62 }
63 }
64 brix_status_set(st, XRDC_EUSAGE, EEXIST,
65 "could not create a unique temp for %s", dst);
66 return -1;
67}

70/*
71 * Commit (or discard) a temp destination: on rc==0 rename `tmp`→`dest`
72 * atomically (downgrading rc to -1 with st set if the rename fails); on rc!=0
73 * drop the temp. Returns the final rc. Shared by every local-dest writer so the
74 * "atomic dest" guarantee lives in one place.
75 */
76int
77atomic_dest_finish(const char *tmp, const char *dest, int rc, brix_status *st)
78{
79 if (rc == 0) {
80 if (rename(tmp, dest) != 0) {
81 brix_status_set(st, XRDC_ESOCK, errno, "rename %s -> %s: %s",
82 tmp, dest, strerror(errno));
83 unlink(tmp);
84 return -1;
85 }
86 return 0;
87 }
88 unlink(tmp); /* drop the partial/cancelled/mismatched temp */
89 return rc;
90}

94/*
95 * WHAT: The invariant inputs of one resilient-open retry loop — the connection,
96 * the path being opened, the compress opaque (NULL if none), and the stall
97 * deadline within which reconnect+reopen attempts are made.
98 * WHY: Both the download and upload paths open their remote handle with the same
99 * "open → retryable? reconnect → retry until the stall deadline" loop.
100 * Bundling its shared inputs into one file-local struct lets a single helper
101 * carry that logic for both directions without a wide parameter list and
102 * keeps download_stream_body / upload_stream_body flat early-return bodies.
103 * HOW: `c` is the (already-connected) control connection; `path` is the remote
104 * object; `opaque` rides the open when --compress is active (NULL otherwise);
105 * `deadline_ns` is the absolute brix_mono_ns() cutoff computed from the
106 * stall window. Populated by the caller; consumed by csctx_reopen_home /
107 * the direction-specific open helpers.
108 */
109typedef struct {
110 brix_conn *c;
111 const char *path;
112 const char *opaque; /* compress opaque or NULL */
113 uint64_t deadline_ns; /* absolute retry cutoff (brix_mono_ns scale) */
114} copy_stream_ctx_t;

117/*
118 * WHAT: Reconnect a control connection to its home (redirector) endpoint,
119 * falling back to the currently-connected host/port when no home is set.
120 * WHY: Every resilient open/close retry re-establishes the session against the
121 * redirector (home_host/home_port) so a reopen re-runs the full locate,
122 * not a stale data-server address. Confining that host/port selection to
123 * one helper removes the duplicated `home ?: current` picking from both
124 * the download and upload retry loops.
125 * HOW: Prefer c->home_host / c->home_port when populated; otherwise use the
126 * live c->host / c->port. The reconnect result is intentionally ignored
127 * (the caller re-attempts the open and re-checks the deadline).
128 */
129static void
130csctx_reopen_home(brix_conn *c, brix_status *st)
131{
132 const char *h = (c->home_host[0] != '\0') ? c->home_host : c->host;
133 int p = (c->home_port != 0) ? c->home_port : c->port;
134 (void) brix_reconnect(c, h, p, st);
135}

138/*
139 * WHAT: Decide whether a failed resilient attempt may be retried, sleeping a
140 * backoff step when it can.
141 * WHY: The download open, upload open, and upload close-commit loops share one
142 * give-up rule — stop on a non-retryable status, an operator cancel, or a
143 * blown stall deadline — and otherwise back off before the next attempt.
144 * One predicate keeps that rule identical across all three callers.
145 * HOW: Returns 1 (retry, after brix_backoff_sleep_fast) when the status is
146 * retryable, no cancel is pending, and the deadline is still in the future;
147 * returns 0 (give up) otherwise. `*attempt` is advanced on a retry so the
148 * backoff grows.
149 */
150static int
151csctx_retry_gate(const brix_status *st, uint64_t deadline_ns, unsigned *attempt)
152{
153 if (!brix_status_retryable(st) || brix_copy_quit_requested()
154 || brix_mono_ns() >= deadline_ns) {
155 return 0;
156 }
157 brix_backoff_sleep_fast((*attempt)++);
158 return 1;
159}

162/*
163 * WHAT: Open the download source read handle, retrying reconnect+reopen within
164 * the stall deadline on a transport fault.
165 * WHY: The open is the last single-RTT step of setup and is just as sever-prone
166 * as connect/stat on a lossy link, so it rides the same resilient loop.
167 * Split out so download_stream_body stays a flat sequence.
168 * HOW: Issues brix_file_open_opaque when a compress opaque is set (ctx->opaque),
169 * else brix_file_open_read; on failure applies csctx_retry_gate and, if it
170 * permits, reconnects to home and retries. Returns 0 with *f open, or -1
171 * (st set) once the retry gate gives up.
172 */
173static int
174download_open_resilient(const copy_stream_ctx_t *ctx, brix_file *f,
175 brix_status *st)
176{
177 unsigned attempt = 0;
178 for (;;) {
179 int orc = (ctx->opaque != NULL)
180 ? brix_file_open_opaque(ctx->c, ctx->path,
181 ctx->opaque, 0, 0, 0, f, st)
182 : brix_file_open_read(ctx->c, ctx->path, f, st);
183 if (orc == 0) {
184 return 0;
185 }
186 if (!csctx_retry_gate(st, ctx->deadline_ns, &attempt)) {
187 return -1;
188 }
189 csctx_reopen_home(ctx->c, st); /* re-establish, then reopen */
190 }
191}

194/*
195 * WHAT: Open the source for read, stream the known-size body to the local sink,
196 * then close the remote handle — the whole "remote file is open" lifetime.
197 * WHY: Confining the open-read handle (and its secondary streams + scratch buf)
198 * to one helper lets the caller stay a flat early-return sequence: the file
199 * is always closed here, on every path, without a shared cleanup jump.
200 * HOW: open_read → streams_open(&ss) → pump(src, sink/sinkctx, si->size) →
201 * file_close. The caller supplies (sink, sinkctx): either a VFS file via
202 * pump_sink_local_vfs + pump_local_t, or the stdout fd via pump_sink_local.
203 * Returns 0 on a complete transfer, -1 (st set) otherwise. On open_read
204 * failure the streams are left untouched (ss.n stays 0, so the caller's
205 * streams_close is a no-op) — mirroring the original NULL-init. The
206 * connection is owned by the caller so it can run the post-transfer checksum
207 * before tearing down. NOTE: the invariant inputs (c/su/si/o/ss) ride in
208 * a download_body_ctx so the extern stays under the 5-parameter gate; the
209 * (sink, sinkctx) pump pair and st stay free (per-callsite / out-param).
210 */
211int
212download_stream_body(const download_body_ctx *j, pump_sink_fn sink,
213 void *sinkctx, brix_status *st)
214{
215 brix_conn *c = j->c;
216 const brix_url *su = j->su;
217 const brix_statinfo *si = j->si;
218 const brix_copy_opts *o = j->o;
219 brix_streamset *ss = j->ss;
220 brix_file f;
221 pump_remote_t src = {0};
222 int rc;
223 int stall = copy_stall_ms(o, 60000);
224 char opq[80];
225 copy_stream_ctx_t ctx = {0};
226
227 ctx.c = c;
228 ctx.path = su->path;
229 ctx.deadline_ns = brix_mono_ns() + (uint64_t) stall * 1000000ULL;
230
231 /* phase-42 W4: request inline read compression when --compress was given.
232 * Rides the open opaque; the server confirms via the open reply and
233 * brix_file_read transparently inflates. A server that doesn't support it
234 * just returns plaintext (f.read_codec stays 0), so this is always safe. */
235 if (o->compress != NULL && o->compress[0] != '\0') {
236 snprintf(opq, sizeof(opq), "xrootd.compress=%s", o->compress);
237 ctx.opaque = opq;
238 }
239 /* Open the remote handle, retried within max_stall on a transport fault
240 * (reconnecting between attempts). */
241 if (download_open_resilient(&ctx, &f, st) != 0) {
242 return -1;
243 }
244
245 /* M8: attach N-1 bound secondary streams to the (post-redirect) session. */
246 brix_streams_open(ss, c, o->streams, st);
247
248 /* remote (known si->size) → caller-supplied sink, with progress. Resilient:
249 * a sever mid-read reconnects + reopens at offset and adapts the request size,
250 * so a one-shot download rides out a flaky/lossy link. */
251 src.c = c;
252 src.f = &f;
253 src.pgrw = o->pgrw;
254 src.resilient = 1;
255 src.path = su->path;
256 src.opaque = ctx.opaque;
257 src.max_stall_ms = copy_stall_ms(o, 60000);
258 src.cur_chunk = XRDC_COPY_CHUNK;
259 rc = transfer_pump(pump_src_remote, &src, sink, sinkctx,
260 si->size, o, si->size, st);
261
262 {
263 brix_status throwaway;
264 brix_status_clear(&throwaway);
265 brix_file_close(c, &f, rc == 0 ? st : &throwaway);
266 }
267 return rc;
268}

271/*
272 * WHAT: The invariant inputs of one local download — the (caller-owned) control
273 * connection, the source/destination URLs, the source stat info, the
274 * options, and the bound-streams set.
275 * WHY: copy_download runs the transfer down two branches (stdout / local file)
276 * that both thread the same six inputs into their body + cksum helpers.
277 * Bundling them into one file-local struct keeps every download helper
278 * under the 5-parameter gate and makes the branch bodies a flat sequence.
279 * HOW: Populated once by copy_download after resilient_setup; passed by const
280 * pointer to download_to_stdout / download_to_local_file, which read
281 * su/du/si/o and pass c/ss through to download_stream_body.
282 */
283typedef struct {
284 brix_conn *c;
285 const brix_url *su;
286 const brix_url *du;
287 const brix_statinfo *si;
288 const brix_copy_opts *o;
289 brix_streamset *ss;
290} download_job_t;

293/*
294 * WHAT: Reconcile a completed download's checksum verdict into the transfer rc,
295 * emitting the "downloaded but NOT verified" note on an unverified query.
296 * WHY: Both the stdout and local-file download paths apply the identical
297 * MISMATCH→fail / UNVERIFIED→warn-and-clear rule after a good transfer;
298 * one helper keeps that policy in a single place and off copy_download's
299 * two branches.
300 * HOW: Runs cksum_verify(local_path may be NULL for stdout); a MISMATCH returns
301 * -1 (caller drops any committed file), an UNVERIFIED prints the note
302 * (unless silent) and clears st (a query hiccup is not a transfer failure),
303 * and an OK returns 0. `local_path` NULL ≡ stdout; the note names du->path.
304 */
305static int
306download_reconcile_cksum(const download_job_t *job, const char *local_path,
307 brix_status *st)
308{
309 const brix_copy_opts *o = job->o;
310 int ck = cksum_verify(job->c, job->su->path, local_path, o->cksum,
311 o->silent, st);
312 if (ck == XRDC_CK_MISMATCH) {
313 return -1;
314 }
315 if (ck == XRDC_CK_UNVERIFIED) {
316 if (!o->silent) {
317 fprintf(stderr, "xrdcp: %s downloaded but checksum NOT verified: "
318 "%s\n", job->du->path, st->msg);
319 }
320 brix_status_clear(st);
321 }
322 return 0;
323}

326/*
327 * WHAT: Download the source body straight to stdout (no temp, no VFS, no commit),
328 * then verify its checksum.
329 * WHY: The stdio destination has no on-disk file to commit or unlink, so it is
330 * the simple half of copy_download; splitting it out keeps each branch a
331 * flat sequence.
332 * HOW: Pumps to STDOUT_FILENO via pump_sink_local; on success, when --cksum was
333 * given, reconciles the verdict with a NULL local path (cksum_verify skips
334 * gracefully, since stdout has no file). Connection is caller-owned.
335 */
336static int
337download_to_stdout(const download_job_t *job, brix_status *st)
338{
339 int stdoutfd = STDOUT_FILENO;
340 download_body_ctx dj = { job->c, job->su, job->si, job->o, job->ss };
341 int rc = download_stream_body(&dj, pump_sink_local, &stdoutfd, st);
342 if (rc == 0 && job->o->cksum != NULL) {
343 rc = download_reconcile_cksum(job, NULL, st);
344 }
345 return rc;
346}

349/*
350 * WHAT: Download the source body into a VFS-backed temp at du->path, then commit
351 * (fsync+rename) and verify the checksum, dropping the file on any failure.
352 * WHY: The local-file path adds the atomic temp+rename lifecycle and integrity
353 * drop-on-mismatch that the stdout path lacks; confining the VFS handle to
354 * this helper keeps its acquire/commit/abort/close linear (no shared jump).
355 * HOW: Opens du->path WRITE (+FORCE when -f) via brix_vfs_open, streams into it,
356 * and on success commits. A committed transfer with --cksum is reconciled;
357 * a genuine MISMATCH unlinks the committed-but-bad file and fails. On any
358 * failure before commit the temp is aborted (unlinked). Connection is
359 * caller-owned.
360 */
361static int
362download_to_local_file(const download_job_t *job, brix_status *st)
363{
364 const brix_copy_opts *o = job->o;
365 brix_vfs_file *vf = NULL;
366 brix_vfs_open_opts vopts = {0};
367 pump_local_t lc;
368 int committed = 0;
369 int rc;
370
371 vopts.io_uring = o->io_uring;
372 vopts.expected_size = job->si->size;
373 vopts.cred = NULL;
374
375 if (brix_vfs_open(job->du->path,
376 XRDC_VFS_WRITE | (o->force ? XRDC_VFS_FORCE : 0),
377 &vopts, &vf, st) != 0) {
378 return -1;
379 }
380
381 lc.vf = vf;
382 {
383 download_body_ctx dj = { job->c, job->su, job->si, o, job->ss };
384 rc = download_stream_body(&dj, pump_sink_local_vfs, &lc, st);
385 }
386
387 /* Commit on success (fsync + rename temp→final); only then verify the
388 * checksum against the committed file. A genuine MISMATCH drops the
389 * committed file and returns error — it is an integrity failure, not a
390 * transient fault. A query hiccup (UNVERIFIED) keeps the good bytes. */
391 if (rc == 0) {
392 rc = brix_vfs_commit(vf, st);
393 if (rc == 0) {
394 committed = 1;
395 if (o->cksum != NULL) {
396 rc = download_reconcile_cksum(job, job->du->path, st);
397 if (rc != 0) {
398 unlink(job->du->path); /* drop committed-but-bad file */
399 }
400 }
401 }
402 }
403 if (rc != 0 && !committed) {
404 brix_vfs_abort(vf); /* discard the partial temp */
405 }
406 brix_vfs_close(vf);
407 return rc;
408}

411int
412copy_download(const brix_url *su, const brix_url *du, const brix_copy_opts *o,
413 const brix_opts *co, brix_status *st)
414{
415 brix_conn c;
416 brix_statinfo si;
417 brix_streamset ss;
418 download_job_t job = {0};
419 int to_stdout = (du->scheme == XRDC_SCHEME_STDIO);
420 int stall = copy_stall_ms(o, 60000);
421 int rc;
422
423 ss.n = 0; /* so the streams teardown is a no-op if we never bind */
424 if (resilient_setup(&c, su, co, &si, stall, st) != 0) {
425 return -1;
426 }
427 if (si.flags & kXR_isDir) {
428 brix_status_set(st, XRDC_EUSAGE, 0, "source is a directory (use -r, M5)");
429 brix_close(&c);
430 return -1;
431 }
432
433 job.c = &c;
434 job.su = su;
435 job.du = du;
436 job.si = &si;
437 job.o = o;
438 job.ss = &ss;
439
440 if (to_stdout) {
441 rc = download_to_stdout(&job, st);
442 brix_streams_close(&ss);
443 brix_close(&c);
444 return rc;
445 }
446
447 /* Local file path: existence-check preserving the original error message,
448 * then open via VFS (atomic temp+rename and optional io_uring inside the
449 * backend). commit() does fsync+rename; abort() unlinks the temp on any
450 * failure or checksum mismatch so the final destination is never partial. */
451 if (!o->force && access(du->path, F_OK) == 0) {
452 brix_status_set(st, XRDC_EUSAGE, 0,
453 "destination exists (use -f to overwrite): %s",
454 du->path);
455 brix_close(&c);
456 return -1;
457 }
458
459 rc = download_to_local_file(&job, st);
460
461 brix_streams_close(&ss);
462 brix_close(&c);
463 return rc;
464}

467/*
468 * WHAT: Open the upload destination write handle, retrying reconnect+reopen
469 * within the stall deadline on a transport fault.
470 * WHY: A restart can hit during connect/open, before the write loop's resilient
471 * sink is reached; retrying the INITIAL open is safe because nothing is
472 * written yet, so a fresh create/truncate retry is idempotent (matches
473 * download_open_resilient and brix_rfile_open_write). Split out so
474 * upload_stream_body stays a flat sequence.
475 * HOW: Issues brix_file_open_opaque (write=1, force, posc) when a compress
476 * opaque is set (ctx->opaque), else brix_file_open_write; on failure applies
477 * csctx_retry_gate and, if it permits, reconnects to home and retries.
478 * Returns 0 with *f open, or -1 (st set) once the retry gate gives up.
479 * `force`/`posc` are the destination open flags.
480 */
481static int
482upload_open_resilient(const copy_stream_ctx_t *ctx, int force, int posc,
483 brix_file *f, brix_status *st)
484{
485 unsigned attempt = 0;
486 for (;;) {
487 int orc = (ctx->opaque != NULL)
488 ? brix_file_open_opaque(ctx->c, ctx->path,
489 ctx->opaque, 1, force, posc,
490 f, st)
491 : brix_file_open_write(ctx->c, ctx->path, force, posc, f, st);
492 if (orc == 0) {
493 return 0;
494 }
495 if (!csctx_retry_gate(st, ctx->deadline_ns, &attempt)) {
496 return -1;
497 }
498 csctx_reopen_home(ctx->c, st);
499 }

503/*
504 * WHAT: Resiliently perform the upload's final close, which is the COMMIT
505 * (renames the staged partial onto the destination).
506 * WHY: A restart landing on the final close would otherwise leave a
507 * fully-written-but-uncommitted partial. Retrying the close within the
508 * stall window — reconnecting + reopening IN PLACE (no truncate) between
509 * attempts — commits it, since the bytes are all present on the server.
510 * Split out so upload_stream_body stays flat.
511 * HOW: Loops brix_file_close (sink->c / sink->f are the connection + handle);
512 * on failure, gives up on cancel/deadline, else backs off and reopens the
513 * partial in place (pump_sink_reopen) to re-commit. If reopen-in-place
514 * fails a PRIOR close may already have committed with its ack lost to the
515 * sever — for a known size (total >= 0) confirm the commit by the
516 * destination's (sink->path) size and treat as success; otherwise fall back
517 * to retryability. Returns 0 (committed) or -1 (st set).
518 */
519static int
520upload_close_commit_resilient(pump_remote_t *sink, int64_t total,
521 brix_status *st)
522{
523 uint64_t deadline = brix_mono_ns()
524 + (uint64_t) sink->max_stall_ms * 1000000ULL;
525 unsigned attempt = 0;
526 for (;;) {
527 if (brix_file_close(sink->c, sink->f, st) == 0) {
528 return 0;
529 }
530 if (brix_copy_quit_requested() || brix_mono_ns() >= deadline) {
531 return -1;
532 }
533 brix_backoff_sleep_fast(attempt++);
534 if (pump_sink_reopen(sink, st) == 0) {
535 continue; /* reopened the partial in place — loop re-commits */
536 }
537 /*
538 * Reopen-in-place failed. A PRIOR close may have already committed
539 * (renamed the staged partial onto the destination) with its ack lost
540 * to the sever — so the partial is gone and reopen-update NotFounds.
541 * Confirm the commit by the destination's size and treat as success.
542 * (total < 0 = stdin: no known size, so fall back to retryability.)
543 */
544 if (total >= 0) {
545 brix_statinfo si;
546 if (brix_stat(sink->c, sink->path, &si, st) == 0
547 && si.size == total) {
548 brix_status_clear(st);
549 return 0;
550 }
551 }
552 if (!brix_status_retryable(st) || brix_mono_ns() >= deadline) {
553 return -1;
554 }
555 }
556}

559/*
560 * WHAT: Verify a persisted upload's checksum, reconciling the verdict into rc.
561 * WHY: After the commit the file is on the server, so its checksum can be
562 * compared against the local source digest; confining that policy here
563 * keeps upload_stream_body's tail a single call.
564 * HOW: For stdin (XRDC_SCHEME_STDIO) there is no on-disk file, so passes NULL
565 * as the local path (cksum_verify skips gracefully). MISMATCH returns -1;
566 * UNVERIFIED prints the note (unless silent) and clears st (a could-not-
567 * verify is not a transfer failure); OK returns 0.
568 */
569static int
570upload_reconcile_cksum(brix_conn *c, const brix_url *su, const brix_url *du,
571 const brix_copy_opts *o, brix_status *st)
572{

574 int ck = cksum_verify(c, du->path, ck_local, o->cksum, o->silent, st);
575 if (ck == XRDC_CK_MISMATCH) {
576 return -1;
577 }
578 if (ck == XRDC_CK_UNVERIFIED) {

580 fprintf(stderr, "xrdcp: uploaded but checksum NOT verified: %s\n",
581 st->msg);
582 }
583 brix_status_clear(st); /* could-not-verify is not a transfer failure */
584 }
585 return 0;

589/*
590 * WHAT: Connect the destination, open it for write, stream bytes from the
591 * caller-supplied (src, srcctx) into it, then tear the whole remote side
592 * down (file close on success, checksum, bound streams, connection) — the
593 * entire "destination session is up" lifetime.
594 * WHY: Confining the connection / write handle / secondary streams to one helper
595 * keeps copy_upload() a flat early-return sequence whose only lingering
596 * resource is the caller-owned VFS handle. Both pre-open failure paths
597 * (connect, open_write) return early without entering the finish teardown.
598 * HOW: connect → open_write → streams_open → transfer_pump(src→remote) → finish.
599 * `total` is the known source size for progress and resilient-close checks
600 * (-1 for stdin / unknown). src is either pump_src_local (stdin) or
601 * pump_src_local_vfs (local file via brix_vfs). su->path is used as the
602 * local checksum source path (NULL ≡ stdin → cksum_verify skips gracefully).
603 * NOTE: the 8-parameter signature is a frozen extern (called from
604 * copy_block.c) — decomposed body-only, per phase-75 G4.
605 */
606int
607upload_stream_body(const upload_body_ctx *j, pump_src_fn src, void *srcctx,
608 brix_status *st)
609{
610 const brix_url *su = j->su;
611 const brix_url *du = j->du;
612 const brix_copy_opts *o = j->o;
613 const brix_opts *co = j->co;
614 int64_t total = j->total;
615 brix_conn c;
616 brix_file f;
617 brix_streamset ss;
618 pump_remote_t sink = {0};
619 int rc;
620 int stall = copy_stall_ms(o, 60000);
621 char opq[80];
622 copy_stream_ctx_t ctx = {0};
623
624 ss.n = 0;
625 if (brix_connect_resilient(&c, du, co, st) != 0) {
626 return -1;
627 }
628
629 /* phase-42 W5: request inline write compression when --compress was given —
630 * the server decompresses each payload on ingest (brix_file_write compresses
631 * transparently once the handle's write_codec is learned). A server that
632 * doesn't support it returns plaintext (write_codec stays 0), so this is safe.
633 * Streams are disabled under write compression (the secondaries would carry
634 * raw payloads the server can't frame). */
635 ctx.c = &c;
636 ctx.path = du->path;
637 ctx.deadline_ns = brix_mono_ns() + (uint64_t) stall * 1000000ULL;
638 if (o->compress != NULL && o->compress[0] != '\0') {
639 snprintf(opq, sizeof(opq), "xrootd.compress=%s", o->compress);
640 ctx.opaque = opq;
641 }
642 /* Resilient INITIAL open (retry+reconnect within the stall window). */
643 if (upload_open_resilient(&ctx, o->force, o->posc, &f, st) != 0) {
644 brix_close(&c);
645 return -1;
646 }
647
648 /* M8: attach N-1 bound secondary streams to the (post-redirect) session.
649 * Skip them when write compression is active (see above). */
650 if (f.write_codec == 0) {
651 brix_streams_open(&ss, &c, o->streams, st);
652 }
653
654 /* local src → remote (EOF-driven), with progress (total = file size or -1).
655 * The sink is resilient: a transport sever mid-upload reconnects, reopens the
656 * destination IN PLACE (no truncate) and re-issues from the same offset, so an
657 * upload survives an nginx restart and resumes from where it left off. This
658 * needs the bytes below the offset to still be on the server: true for a
659 * direct-to-final write (default, posc off) and for a server with
660 * brix_upload_resume on (deterministic preserved partial). Re-issuing the
661 * same buffer at the same offset is idempotent. */
662 sink.c = &c;
663 sink.f = &f;
664 sink.pgrw = o->pgrw;
665 sink.resilient = 1;
666 sink.path = du->path;
667 sink.posc = o->posc;
668 sink.max_stall_ms = copy_stall_ms(o, 60000);
669 rc = transfer_pump(src, srcctx, pump_sink_remote, &sink, -1, o, total, st);
670
671 /* Only close the remote file cleanly on success: with POSC, abandoning the
672 * handle (connection teardown without close) makes the server discard the
673 * partial upload, which is exactly the atomicity we want on error. The close
674 * is the COMMIT, so it is retried resiliently within the stall window. */
675 if (rc == 0) {
676 rc = upload_close_commit_resilient(&sink, total, st);
677 }
678 /* The file is persisted after close — verify its checksum now (connection
679 * still open), comparing our local source digest against the server's. */
680 if (rc == 0 && o->cksum != NULL) {
681 rc = upload_reconcile_cksum(&c, su, du, o, st);
682 }
683 brix_streams_close(&ss);
684 brix_close(&c);
685 return rc;
686}

689int
690copy_upload(const brix_url *su, const brix_url *du, const brix_copy_opts *o,
691 const brix_opts *co, brix_status *st)
692{
693 if (su->scheme == XRDC_SCHEME_STDIO) {
694 /* stdio path: pump from raw STDIN_FILENO; no VFS open */
695 int stdinfd = STDIN_FILENO;
696 upload_body_ctx uj = { su, du, o, co, -1 /* size unknown */ };
697 return upload_stream_body(&uj, pump_src_local, &stdinfd, st);
698 }
699
700 /* Local file path: open via VFS (io_uring selection inside the backend) */
701 {
702 brix_vfs_file *vf = NULL;
703 brix_vfs_open_opts vopts;
704 brix_vfs_stat vst;
705 brix_status tmp_st;
706 pump_local_t lc;
707 int64_t total = -1;
708 int rc;
709
710 vopts.io_uring = o->io_uring;
711 vopts.expected_size = -1; /* read-only open; hint unused */
712 vopts.cred = NULL;
713
714 if (brix_vfs_open(su->path, XRDC_VFS_READ, &vopts, &vf, st) != 0) {
715 return -1;
716 }
717 brix_status_clear(&tmp_st);
718 if (brix_vfs_fstat(vf, &vst, &tmp_st) == 0) {
719 total = vst.size;
720 }
721
722 lc.vf = vf;
723 {
724 upload_body_ctx uj = { su, du, o, co, total };
725 rc = upload_stream_body(&uj, pump_src_local_vfs, &lc, st);
726 }
727 brix_vfs_close(vf);
728 return rc;
729 }
730}

743int
744copy_web_upload(const brix_url *su, const brix_weburl *du, const brix_copy_opts *o,
745 const brix_opts *co, brix_status *st)
746{
747 char hdrs[8192];
748 brix_vfs_file *vf = NULL;
749 brix_vfs_open_opts vopts;
750 brix_vfs_stat vst;
751 int status = 0, rc;
752
753 if (su->scheme == XRDC_SCHEME_STDIO) {
754 brix_status_set(st, XRDC_EUSAGE, 0,
755 "web upload needs a regular local file (Content-Length); "
756 "stdin not supported");
757 return -1;
758 }
759
760 /* Open the local source through the VFS (byte I/O dispatches to the shared
761 * SD driver), then fstat it for the Content-Length the PUT must promise. */
762 vopts.io_uring = o ? o->io_uring : 0;
763 vopts.expected_size = -1; /* read-only open; hint unused */
764 vopts.cred = NULL;
765 if (brix_vfs_open(su->path, XRDC_VFS_READ, &vopts, &vf, st) != 0) {
766 return -1;
767 }
768 if (brix_vfs_fstat(vf, &vst, st) != 0) {
769 brix_vfs_close(vf);
770 return -1;
771 }
772 if (vst.is_dir) {
773 /* st_size is only a reliable Content-Length for a non-directory file. */
774 brix_vfs_close(vf);
775 brix_status_set(st, XRDC_EUSAGE, 0,
776 "web upload source must be a regular file: %s", su->path);
777 return -1;
778 }
779 {
780 web_auth_ctx a = { du, "PUT", o, co, st };
781 if (web_auth_headers(&a, hdrs, sizeof(hdrs)) != 0) {
782 brix_vfs_close(vf);
783 return -1;
784 }
785 }
786 {
787 /* Resilient by default: Content-Range PUT chunks that reconnect + resume
788 * from the server's durable offset, so the upload survives an nginx
789 * restart (server brix_webdav_upload_resume). A plain server commits on
790 * the first whole-range chunk, so a single-shot upload still works. */
791 int stall = copy_stall_ms(o, XRDC_DEFAULT_MAX_STALL_MS);
792 rc = brix_http_upload_resumable(du->host, du->port, du->tls, du->path,
793 hdrs[0] ? hdrs : NULL, web_upload_src_vfs, vf,
794 (long long) vst.size,
795 co ? co->verify_host : 1, co ? co->ca_dir : NULL,
796 XRDC_WEB_TIMEOUT_MS, stall, &status, st);
797 }
798 brix_vfs_close(vf);
799 if (rc == 0 && o && !o->silent) {
800 fprintf(stderr, "xrdcp: uploaded %lld bytes (HTTP %d)\n",
801 (long long) vst.size, status);
802 }
803 return rc;
804}

808int
809copy_web(const char *src, const char *dst, const brix_copy_opts *o,
810 const brix_opts *co, brix_status *st)
811{

814

817 "recursive copy is not supported for web (davs/s3) endpoints");
818 return -1;
819 }
820 if (src_web && !dst_web) { /* download: web → local/stdout */
821 brix_weburl su;
822 brix_url du;
823 if (brix_weburl_parse(src, &su) != 0) {
824 brix_status_set(st, XRDC_EUSAGE, 0, "bad web source URL");
825 return -1;
826 }
827 if (brix_url_parse(dst, &du, st) != 0) {
828 return -1;
829 }
830 if (du.scheme != XRDC_SCHEME_LOCAL && du.scheme != XRDC_SCHEME_STDIO) {
831 brix_status_set(st, XRDC_EUSAGE, 0,
832 "web download destination must be local or '-'");
833 return -1;
834 }
835 {
836 web_dl_req rq = { &su, &du, du.scheme == XRDC_SCHEME_STDIO, o, co };
837 return copy_web_download(&rq, st);
838 }
839 }
840 if (!src_web && dst_web) { /* upload: local → web */
841 brix_url su;
842 brix_weburl du;
843 if (brix_url_parse(src, &su, st) != 0) {
844 return -1;
845 }
846 if (brix_weburl_parse(dst, &du) != 0) {
847 brix_status_set(st, XRDC_EUSAGE, 0, "bad web destination URL");
848 return -1;
849 }
850 if (su.scheme != XRDC_SCHEME_LOCAL && su.scheme != XRDC_SCHEME_STDIO) {
851 brix_status_set(st, XRDC_EUSAGE, 0,
852 "web upload source must be a local file");
853 return -1;
854 }
855 return copy_web_upload(&su, &du, o, co, st);
856 }
857 brix_status_set(st, XRDC_EUSAGE, 0,
858 "web→web copy is not supported (stage via a local file)");
859 return -1;
860}
