3 * closing a handle onto an object's block-0 fd + cached catalog metadata.
4 *
5 * WHAT: Owns sd_pblock_open / sd_pblock_open_as / sd_pblock_close (the .open /
6 * .close vtable slots named in the descriptor in sd_pblock.c, plus the
7 * identity-parameterised open_as boundary the *_cred slots call) and their
8 * private helpers (pblock_make_obj + the create/existing/absent open lanes).
9 *
10 * WHY: Split from sd_pblock.c (phase-79/guard burndown) to hold every pblock

16 * HOW: Same ngx-free (libc + sqlite, malloc-owned state) contract as the rest of
17 * the driver, compiled only when the build found libsqlite3
18 * (BRIX_HAVE_SQLITE) so a no-sqlite build stays empty.

59/* pblock_make_obj — wrap a block-0 fd + cached metadata in a heap object. On
60 * failure the fd is closed and NULL/errno returned. */
61static brix_sd_obj_t *
62pblock_make_obj(brix_sd_instance_t *inst, const char *path, int fd,
63 const pblock_meta *meta)
64{
65 brix_sd_obj_t *obj;
66 pblock_obj_t *os;
67
68 obj = calloc(1, sizeof(*obj));
69 os = calloc(1, sizeof(*os));
70 if (obj == NULL || os == NULL) {
71 free(obj);
72 free(os);
73 if (fd >= 0) {
74 close(fd);
75 }
76 errno = ENOMEM;
77 return NULL;
78 }
79
80 os->st = inst->state;
81 os->meta = *meta;
82 os->block_size = meta->block_size;
83 snprintf(os->path, sizeof(os->path), "%s", path);
84 snprintf(os->blob_id, sizeof(os->blob_id), "%s", meta->blob_id);
85
86 /* Phase-83: resolve any lab fault/shape rule for this handle ONCE, at open
87 * (a metadata boundary). NULL when lab is off or no rule applies — the hot
88 * byte path then skips the gate entirely. */
89 os->lab = pblock_lab_snapshot(os->st->lab, path);
90
91 /* F3: snapshot the at-rest per-block CRCs ONCE at open so the hot read path
92 * verifies against memory (no DB). csi_dlo/dhi is the empty written extent
93 * (dlo > dhi) until the first write widens it. */
94 os->csi_dlo = INT64_MAX;
95 os->csi_dhi = 0;
96 if (os->st->csi && !meta->is_dir) {
97 int64_t nblk = meta->size > 0
98 ? pblock_last_block(meta->size, os->block_size) + 1 : 0;
99
100 (void) pblock_csi_load(os->st->cat, os->blob_id, nblk,
101 &os->csi_crc, &os->csi_n);
102 }
103
104 /* F5: snapshot how far this handle may grow the file ONCE at open (quota
105 * off ⇒ INT64_MAX) — pwrite/copy_range enforce it without touching the DB. */
106 os->quota_max = pblock_quota_max_size(os->st, meta->uid, meta->size);
107
108 obj->driver = inst->driver;
109 obj->inst = inst;
110 obj->fd = fd >= 0 ? fd : NGX_INVALID_FILE;
111 obj->state = os;
112 obj->heap_shell = 1; /* malloc'd shell; a VFS adopter frees it after copy */
113
114 /* Capture the open-time metadata snapshot the VFS reports to callers (file
115 * size, mode, mtime) — without this the adopting handle would see a zeroed
116 * snap and treat the file as empty (immediate EOF on read). */
117 ngx_memzero(&obj->snap, sizeof(obj->snap));
118 obj->snap.size = meta->size;
119 obj->snap.mtime = meta->mtime;
120 obj->snap.ctime = meta->ctime;
121 obj->snap.mode = meta->mode;
122 obj->snap.ino = (ino_t) pblock_fnv(path);
123 obj->snap.uid = meta->uid;
124 obj->snap.gid = meta->gid;
125 obj->snap.is_dir = meta->is_dir ? 1 : 0;
126 obj->snap.is_reg = meta->is_dir ? 0 : 1;
127
128 /* F6: count live regular-file handles so a snapshot restore can refuse
129 * (EBUSY) rather than swap the namespace out from under an open file.
130 * Balanced by the matching decrement in sd_pblock_close. */
131 if (os->st->snap && !meta->is_dir) {
132 __atomic_add_fetch(&os->st->open_files, 1, __ATOMIC_RELEASE);
133 }
134 return obj;
135}

137/* pblock_open_create — create a brand-new file: a fresh per-object dir + block 0,
138 * plus its catalog row owned by (uid, gid) — 0/0 for the service itself.
139 * Returns the object or NULL/errno (*err_out set). */
140static brix_sd_obj_t *
141pblock_open_create(brix_sd_instance_t *inst, const char *path, mode_t mode,
142 uint32_t uid, uint32_t gid, int *err_out)
143{
144 pblock_state_t *st = inst->state;
145 pblock_meta meta;
146 brix_sd_obj_t *obj;
147 char block0[PATH_MAX];
148 int fd;
149
150 if (pblock_quota_admit(st, uid, 0, 1) != 0) { /* F5 */
151 if (err_out != NULL) { *err_out = errno; }
152 return NULL;
153 }
154
155 memset(&meta, 0, sizeof(meta));
156 if (pblock_gen_blob_id(meta.blob_id) != 0
157 || pblock_ensure_obj_dir(st, meta.blob_id) != 0
158 || pblock_block_path(st, meta.blob_id, 0, block0, sizeof(block0)) != 0)
159 {
160 if (err_out != NULL) { *err_out = errno; }
161 return NULL;
162 }
163
164 fd = open(block0, O_RDWR | O_CREAT | O_EXCL, 0600);
165 if (fd < 0) {
166 if (err_out != NULL) { *err_out = errno; }
167 return NULL;
168 }
169
170 meta.is_dir = 0;
171 meta.size = 0;
172 meta.block_size = st->block_size;
173 meta.mtime = meta.ctime = pblock_now();
174 meta.mode = S_IFREG | (mode & 0777);
175 meta.uid = uid;
176 meta.gid = gid;
177 snprintf(meta.xform, sizeof(meta.xform), "%s", /* F12/F13: record kind */
178 pblock_xform_name(st->xform.kind));
179
180 if (pblock_catalog_put(st->cat, path, &meta) != 0) {
181 int err = errno;
182
183 close(fd);
184 pblock_remove_blocks(st, meta.blob_id, 0, st->block_size);
185 if (err_out != NULL) { *err_out = err; }
186 return NULL;
187 }
188
189 obj = pblock_make_obj(inst, path, fd, &meta);
190 if (obj == NULL && err_out != NULL) {
191 *err_out = errno;
192 }
193 if (obj != NULL && st->refs) { /* F10 */
194 (void) pblock_refs_track(st, meta.blob_id, 0, meta.block_size, 0, 0);
195 ((pblock_obj_t *) obj->state)->wv = brix_wverify_begin();
196 }
197 return obj;
198}

200/* pblock_open_existing — open an existing file's block 0; honours O_TRUNC on
201 * write via the block-aware truncate. Returns the object or NULL/errno. */
202static brix_sd_obj_t *
203pblock_open_existing(brix_sd_instance_t *inst, const char *path,
204 pblock_meta *meta, int sd_flags, int *err_out)
205{
206 pblock_state_t *st = inst->state;
207 brix_sd_obj_t *obj;
208 char block0[PATH_MAX];
209 int want_write = sd_flags & BRIX_SD_O_WRITE;
210 int fd;
211
212 /* F12/F13: refuse to serve a file whose recorded transform does not match the
213 * export's configured one (wrong key/codec ⇒ garbage). A config error, not a
214 * data error — fail the open rather than hand back undecodable bytes. */
215 if (pblock_xform_kind_from_name(meta->xform) != st->xform.kind) {
216 if (err_out != NULL) { *err_out = EIO; }
217 return NULL;
218 }
219
220 /* F10: never write through a shared blob — a write-intent open first gives
221 * this object a private copy (empty on O_TRUNC: the bytes are doomed). */
222 if (st->refs && want_write
223 && pblock_refs_break_share(st, path, meta,
224 (sd_flags & BRIX_SD_O_TRUNC) != 0) != 0)
225 {
226 if (err_out != NULL) { *err_out = errno; }
227 return NULL;
228 }
229
230 if (pblock_block_path(st, meta->blob_id, 0, block0, sizeof(block0)) != 0) {
231 if (err_out != NULL) { *err_out = errno; }
232 return NULL;
233 }
234 fd = open(block0, want_write ? O_RDWR : O_RDONLY);
235 if (fd < 0) {
236 if (err_out != NULL) { *err_out = errno; }
237 return NULL;
238 }
239
240 obj = pblock_make_obj(inst, path, fd, meta);
241 if (obj == NULL) {
242 if (err_out != NULL) { *err_out = errno; }
243 return NULL;
244 }
245
246 if (want_write && (sd_flags & BRIX_SD_O_TRUNC) && meta->size > 0) {
247 if (sd_pblock_ftruncate(obj, 0) != NGX_OK) {
248 int err = errno;
249
250 obj->driver->close(obj); /* frees state + closes fd */
251 free(obj); /* … and the malloc'd shell */
252 if (err_out != NULL) { *err_out = err; }
253 return NULL;
254 }
255 }
256 if (st->refs && want_write && (sd_flags & BRIX_SD_O_TRUNC)) {
257 /* F10: content is being rewritten from scratch — this handle can grow
258 * an honest whole-object CRC, making the blob a dedup candidate. */
259 ((pblock_obj_t *) obj->state)->wv = brix_wverify_begin();
260 }
261 return obj;
262}

264/* F15: mandatory lease gate — before both the create and overwrite lanes, since
265 * a whole-file lease guards the *name* (a create under a foreign lease is as much
266 * a conflicting write as an overwrite). Live foreign 'X' refuses everyone; a live
267 * foreign whole-file 'W' refuses write-intent opens. EBUSY maps to kXR_FileLocked
268 * / HTTP 423 at the protocol edge. Returns 1 if the open must be refused (errno +
269 * *err_out set), 0 to proceed. */
270static int
271pblock_open_locked(pblock_state_t *st, const char *path, int sd_flags,
272 uint32_t uid, int *err_out)
273{
274 if (st->locks
275 && pblock_locks_open_check(st, path,
276 (sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE
277 | BRIX_SD_O_TRUNC | BRIX_SD_O_APPEND)) != 0,
278 uid) != 0)
279 {
280 if (err_out != NULL) { *err_out = errno; }
281 return 1;
282 }
283 return 0;
284}

286/* Existing-file open after the dir/lock gates: the F9 visibility-lag hide, the
287 * O_CREATE|O_EXCL conflict, and the F4 nearline recall, then the real open. */
288static brix_sd_obj_t *
289pblock_open_existing_gated(brix_sd_instance_t *inst, pblock_state_t *st,
290 const char *path, pblock_meta *meta, int sd_flags, int *err_out)
291{
292 /* F9: visibility lag — a freshly created path is ENOENT to other readers
293 * for the armed window. Write-intent opens are exempt: the writer/overwriter
294 * lane must never see its own phantom ENOENT (the S3 session monotonic-read
295 * guarantee — a handle that wrote reads through its own open snapshot and is
296 * untouched by design). */
297 if (st->lab != NULL
298 && !(sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE
299 | BRIX_SD_O_TRUNC))
300 && pblock_anomaly_hidden(st, path))
301 {
302 if (err_out != NULL) { *err_out = ENOENT; }
303 return NULL;
304 }
305 if ((sd_flags & BRIX_SD_O_CREATE) && (sd_flags & BRIX_SD_O_EXCL)) {
306 if (err_out != NULL) { *err_out = EEXIST; }
307 return NULL;
308 }
309 /* F4: a demoted (nearline/offline) file must be recalled before it serves —
310 * a bounded synchronous recall like sd_frm's, so a plain export exercises the
311 * lane without a cache tier in front. A LOST object or failed recall errors
312 * out (never silently serves the bytes a real tape system could not). O_TRUNC
313 * replaces the content, so it skips the recall and comes back ONLINE below via
314 * the write path. */
315 if (st->nearline && !(sd_flags & BRIX_SD_O_TRUNC)
316 && pblock_nearline_recall(st, path) != 0)
317 {
318 if (err_out != NULL) { *err_out = errno; }
319 return NULL;
320 }
321 return pblock_open_existing(inst, path, meta, sd_flags, err_out);
322}

324/* Absent path: create it when O_CREATE is set (recording the F9 anomaly on the
325 * new name), else ENOENT. */
326static brix_sd_obj_t *
327pblock_open_absent(brix_sd_instance_t *inst, pblock_state_t *st,
328 const char *path, mode_t mode, uint32_t uid, uint32_t gid,
329 int sd_flags, int *err_out)
330{
331 brix_sd_obj_t *obj;
332
333 if (!(sd_flags & BRIX_SD_O_CREATE)) {
334 if (err_out != NULL) { *err_out = ENOENT; }
335 return NULL;
336 }
337 obj = pblock_open_create(inst, path, mode, uid, gid, err_out);

339 pblock_anomaly_created(st, path); /* F9 */
340 }
341 return obj;
342}

344/* sd_pblock_open_as — the open implementation, parameterized by the owner
345 * (uid, gid) recorded on a CREATE. The plain vtable slot passes 0/0 (the
346 * service); the identity-aware open_cred slot (sd_pblock_cred.c) passes the
347 * requester's resolved catalog ids so new files are owned by their creator. */
348static brix_sd_obj_t *
349pblock_open_as_inner(brix_sd_instance_t *inst, const char *path, int sd_flags,
350 mode_t mode, uint32_t uid, uint32_t gid, int *err_out)
351{
352 pblock_state_t *st = inst->state;
353 pblock_meta meta;
354 int rc;
355
356 rc = pblock_catalog_lookup(st->cat, path, &meta);
357 if (rc < 0) {
358 if (err_out != NULL) { *err_out = errno; }
359 return NULL;
360 }
361
362 if (pblock_open_locked(st, path, sd_flags, uid, err_out)) {
363 return NULL;
364 }
365
366 if (rc == 0 && meta.is_dir) {
367 if (sd_flags & BRIX_SD_O_WRITE) {
368 if (err_out != NULL) { *err_out = EISDIR; }
369 return NULL;
370 }
371 return pblock_make_obj(inst, path, NGX_INVALID_FILE, &meta);
372 }
373
374 if (rc == 0) { /* existing file */
375 return pblock_open_existing_gated(inst, st, path, &meta, sd_flags,
376 err_out);
377 }
378
379 return pblock_open_absent(inst, st, path, mode, uid, gid, sd_flags, err_out);
380}

382/* sd_pblock_open_as — the identity-parameterised open, wrapping the real
383 * implementation with the F17 audit record. The wrapper (not the inner) is the
384 * audit boundary so every terminal outcome — dir open, existing file, create,
385 * or an error — is recorded exactly once with the actor identity and result. */
386brix_sd_obj_t *
387sd_pblock_open_as(brix_sd_instance_t *inst, const char *path, int sd_flags,
388 mode_t mode, uint32_t uid, uint32_t gid, int *err_out)
389{
390 pblock_state_t *st = inst->state;
391 brix_sd_obj_t *obj;
392 int myerr = 0;
393
394 obj = pblock_open_as_inner(inst, path, sd_flags, mode, uid, gid, &myerr);
395 if (obj != NULL && st->locks) {
396 /* F15: snapshot the foreign range leases into the handle — the pwrite
397 * check is then a pure scan (per the phase design, a conflicting
398 * later open is refused at *its* open, so no per-I/O DB hits). */
399 pblock_locks_snapshot(st, obj->state, uid);
400 }
401 if (st->audit) {
402 char aux[32];
403
404 snprintf(aux, sizeof(aux), "flags=%d", sd_flags);
405 pblock_audit_log(st->cat, "open", path, aux, uid, gid,
406 obj ? 0 : -1, obj ? 0 : myerr);
407 }
408 if (obj == NULL && err_out != NULL) {
409 *err_out = myerr;
410 }
411 return obj;
412}

422ngx_int_t
423sd_pblock_close(brix_sd_obj_t *obj)
424{
425 pblock_obj_t *os;
426 ngx_int_t rc = NGX_OK;
427
428 if (obj == NULL) {
429 return NGX_OK;
430 }
431 os = obj->state;
432
433 if (os != NULL && os->dirty) {
434 pblock_meta old;
435 int have_old = 0;
436
437 /* F9: the pre-update row is what a stale stat will serve — capture it
438 * before the touch publishes the new size/mtime. */
439 if (os->st->lab != NULL) {
440 have_old = pblock_catalog_lookup(os->st->cat, os->path, &old) == 0;
441 }
442 pblock_lab_crash(os->st->lab, "before_catalog_update"); /* F7 */
443 if (pblock_quota_touch_admit(os->st, os->path, os->meta.uid,
444 os->meta.size) != 0 /* F5 */
445 || pblock_catalog_touch(os->st->cat, os->path, os->meta.size,
446 os->meta.mtime) != 0)
447 {
448 rc = NGX_ERROR;
449 }
450 pblock_lab_crash(os->st->lab, "after_catalog_update"); /* F7 */
451 if (rc == NGX_OK && have_old) {
452 pblock_anomaly_updated(os->st, os->path, old.size,
453 old.mtime); /* F9 */
454 }
455 os->dirty = 0;
456 }
457 if (obj->fd != NGX_INVALID_FILE) {
458 if (close(obj->fd) != 0) {
459 rc = NGX_ERROR;
460 }
461 obj->fd = NGX_INVALID_FILE;
462 }
463 if (os != NULL) {
464 if (os->st->audit) { /* F17: fold per-I/O totals here */
465 char aux[64];
466
467 snprintf(aux, sizeof(aux), "r=%lld w=%lld mb=%lld",
468 (long long) os->a_rbytes, (long long) os->a_wbytes,
469 (long long) os->a_maxblock);
470 pblock_audit_log(os->st->cat, "close", os->path, aux,
471 os->meta.uid, os->meta.gid,
472 rc == NGX_OK ? 0 : -1, 0);
473 }
474 if (os->st->csi) { /* F3: persist CRCs for written blocks */
475 (void) pblock_csi_flush(os->st, os->blob_id, os->meta.size,
476 os->block_size, os->csi_dlo, os->csi_dhi);
477 }
478 if (os->wv != NULL) { /* F10: publish-time dedup fold */
479 if (os->st->refs && rc == NGX_OK) {
480 uint32_t crc = 0;
481 off_t total = 0;
482 int ok;
483
484 ok = brix_wverify_expected(os->wv, &crc, &total) == 0
485 && (int64_t) total == os->meta.size;
486 (void) pblock_refs_dedup_publish(os->st, os->path, &os->meta,
487 crc, ok);
488 }
489 brix_wverify_free(os->wv);
490 }
491 if (os->st->snap && !os->meta.is_dir) { /* F6: release the handle count */
492 __atomic_sub_fetch(&os->st->open_files, 1, __ATOMIC_RELEASE);
493 }
494 free(os->csi_crc);
495 free(os->lock_rng); /* F15 per-open lease snapshot */
496 pblock_lab_obj_free(os->lab); /* Phase-83 per-open snapshot */
497 }
498 free(os);
499 obj->state = NULL;
500 /* The obj shell is NOT freed here: it may be embedded in the adopter's
501 * handle (the VFS copies *obj by value). Its owner — the VFS adopter (via
502 * obj->heap_shell), the internal open-error paths, or the standalone test —
503 * frees the malloc'd shell. */
504 return rc;
505}
