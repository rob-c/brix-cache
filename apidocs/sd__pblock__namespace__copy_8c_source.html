1/*
2 * sd_pblock_namespace_copy.c — server-side copy vtable slots for the pblock
3 * storage driver.
4 *
5 * WHAT: Implements the server_copy metadata-plane slots of
6 * brix_sd_pblock_driver: sd_pblock_server_copy_as (owner-parameterised)
7 * and sd_pblock_server_copy (the plain 0/0 service slot). A copy either
8 * shares the source's blob copy-on-write (F10 refs) or physically
9 * duplicates every block under a fresh blob id, then publishes the
10 * destination catalog row owned by the copier.
11 *
12 * WHY: Split out of sd_pblock_namespace.c (file-size guard burndown) to keep
13 * every pblock file under the one-concept size cap. The byte/metadata copy
14 * path is self-contained — its three static helpers (CoW share, block copy,
15 * physical copy) are used only by the two public slots here — so it lives
16 * apart from the namespace mutations (stat/unlink/mkdir/setattr/rename),
17 * directory iteration and xattr CRUD that stay in sd_pblock_namespace.c.
18 * The two public functions are non-static because the driver descriptor
19 * names them; declarations live in sd_pblock_internal.h.
20 *
21 * HOW: Every operation reaches the namespace through the catalog API
22 * (sd_pblock_catalog.h) and touches block files only through the packed-block
23 * engine (pblock_store.h) — no raw byte syscalls here. ngx-free (libc +
24 * catalog + engine) so the module and standalone unit test compile it
25 * identically. Gated by BRIX_HAVE_SQLITE like the rest of the backend.
26 */

59/* F10 CoW copy: dst shares src's blob — O(metadata), no bytes move; the first
60 * write to either row breaks the share at open. The dst row is owned by
61 * (uid, gid). Returns NGX_OK / NGX_ERROR. */
62static ngx_int_t
63pblock_copy_cow(pblock_state_t *st, const pblock_meta *smeta, const char *dst,
64 uint32_t uid, uint32_t gid, off_t *bytes_out)
65{
66 pblock_meta dexist, dmeta;
67 int dhad = pblock_catalog_lookup(st->cat, dst, &dexist) == 0;
68
69 if (pblock_refs_bump(st, smeta->blob_id, smeta->size,
70 smeta->block_size) != 0)
71 {
72 return NGX_ERROR;
73 }
74 memset(&dmeta, 0, sizeof(dmeta));
75 memcpy(dmeta.blob_id, smeta->blob_id, sizeof(dmeta.blob_id));
76 dmeta.is_dir = 0;
77 dmeta.size = smeta->size;
78 dmeta.block_size = smeta->block_size;
79 dmeta.mtime = dmeta.ctime = pblock_now();
80 dmeta.mode = smeta->mode;
81 dmeta.uid = uid;
82 dmeta.gid = gid;
83 if (pblock_catalog_put(st->cat, dst, &dmeta) != 0) {
84 int err = errno;
85
86 pblock_refs_release(st, smeta->blob_id, smeta->size,
87 smeta->block_size);
88 errno = err;
89 return NGX_ERROR;
90 }
91 if (dhad && !dexist.is_dir) { /* the replaced dst's blob loses a ref */
92 pblock_refs_release(st, dexist.blob_id, dexist.size,
93 dexist.block_size);
94 }
95 if (st->lab != NULL) { /* F9 */
96 if (dhad) {
97 pblock_anomaly_updated(st, dst, dexist.size, dexist.mtime);
98 } else {
99 pblock_anomaly_created(st, dst);
100 }
101 }
102 if (bytes_out != NULL) {
103 *bytes_out = (off_t) smeta->size;
104 }
105 /* No csi flush: the shared blob's integrity rows already exist. */
106 if (st->audit) { /* F17 */
107 char aux[32];
108
109 snprintf(aux, sizeof(aux), "cow=1 w=%lld", (long long) smeta->size);
110 pblock_audit_log(st->cat, "copy", dst, aux, uid, gid, 0, 0);
111 }
112 return NGX_OK;
113}

115/* Physically copy every block of `src_blob` to `dst_blob`. Returns 0, or -1
116 * with errno set from the failing call (the caller unwinds the partial dst). */
117static int
118pblock_copy_blocks(pblock_state_t *st, const char *src_blob,
119 const char *dst_blob, int64_t size, int64_t block_size)
120{
121 int64_t last = pblock_last_block(size, block_size);
122 int64_t i;
123
124 for (i = 0; i <= last; i++) {
125 char sp[PATH_MAX], dp[PATH_MAX];
126
127 if (pblock_block_path(st, src_blob, i, sp, sizeof(sp)) != 0
128 || pblock_block_path(st, dst_blob, i, dp, sizeof(dp)) != 0
129 || pblock_copy_one_block(sp, dp) < 0)
130 {
131 return -1;
132 }
133 }
134 return 0;
135}

137/* Full byte copy: allocate a fresh blob, copy every block, then publish the dst
138 * row owned by (uid, gid). Returns NGX_OK / NGX_ERROR (partial blob unwound). */
139static ngx_int_t
140pblock_copy_physical(pblock_state_t *st, const pblock_meta *smeta,
141 const char *dst, uint32_t uid, uint32_t gid, off_t *bytes_out)
142{
143 pblock_meta dmeta, dexist;
144 int dhad = 0;
145
146 memset(&dmeta, 0, sizeof(dmeta));
147 if (pblock_gen_blob_id(dmeta.blob_id) != 0
148 || pblock_ensure_obj_dir(st, dmeta.blob_id) != 0)
149 {
150 return NGX_ERROR;
151 }
152
153 if (pblock_copy_blocks(st, smeta->blob_id, dmeta.blob_id, smeta->size,
154 smeta->block_size) != 0)
155 {
156 int err = errno;
157
158 pblock_remove_blocks(st, dmeta.blob_id, smeta->size,
159 smeta->block_size);
160 errno = err;
161 return NGX_ERROR;
162 }
163
164 dmeta.is_dir = 0;
165 dmeta.size = smeta->size;
166 dmeta.block_size = smeta->block_size;
167 dmeta.mtime = dmeta.ctime = pblock_now();
168 dmeta.mode = smeta->mode;
169 dmeta.uid = uid;
170 dmeta.gid = gid;
171
172 /* F9: is this copy a create or an overwrite of dst? The pre-put row is what
173 * a stale stat will serve. */
174 if (st->lab != NULL) {
175 dhad = pblock_catalog_lookup(st->cat, dst, &dexist) == 0;
176 }
177 if (pblock_catalog_put(st->cat, dst, &dmeta) != 0) {
178 int err = errno;
179
180 pblock_remove_blocks(st, dmeta.blob_id, smeta->size,
181 smeta->block_size);
182 errno = err;
183 return NGX_ERROR;
184 }
185 if (st->lab != NULL) {
186 if (dhad) {
187 pblock_anomaly_updated(st, dst, dexist.size, dexist.mtime);
188 } else {
189 pblock_anomaly_created(st, dst);
190 }
191 }
192 if (bytes_out != NULL) {
193 *bytes_out = (off_t) smeta->size;
194 }
195 if (st->csi) { /* F3: tag the copy */
196 (void) pblock_csi_flush(st, dmeta.blob_id, dmeta.size,
197 dmeta.block_size, 0, INT64_MAX);
198 }
199 if (st->audit) { /* F17 */
200 char aux[32];
201
202 snprintf(aux, sizeof(aux), "w=%lld", (long long) smeta->size);
203 pblock_audit_log(st->cat, "copy", dst, aux, uid, gid, 0, 0);
204 }
205 return NGX_OK;
206}

208/* sd_pblock_server_copy_as — server-side copy whose destination row is owned
209 * by (uid, gid): the copier, not the source's owner (POSIX cp semantics). The
210 * plain slot passes 0/0 (service); server_copy_cred passes the requester. */
211ngx_int_t
212sd_pblock_server_copy_as(brix_sd_instance_t *inst, const char *src,
213 const char *dst, off_t *bytes_out, uint32_t uid, uint32_t gid)
214{
215 pblock_state_t *st = inst->state;
216 pblock_meta smeta;
217 int rc;
218
219 rc = pblock_catalog_lookup(st->cat, src, &smeta);
220 if (rc < 0) {
221 return NGX_ERROR;
222 }
223 if (rc == 1) {
224 errno = ENOENT;
225 return NGX_ERROR;
226 }
227 if (smeta.is_dir) {
228 errno = EISDIR;
229 return NGX_ERROR;
230 }
231
232 if (st->quota) { /* F5 */
233 pblock_meta dexist;
234 int drc = pblock_catalog_lookup(st->cat, dst, &dexist);
235
236 if (pblock_quota_admit(st, uid,
237 (int64_t) smeta.size - (drc == 0 ? dexist.size : 0),
238 drc == 0 ? 0 : 1) != 0)
239 {
240 return NGX_ERROR;
241 }
242 }
243
244 if (st->refs) { /* F10: CoW copy */
245 return pblock_copy_cow(st, &smeta, dst, uid, gid, bytes_out);
246 }
247 return pblock_copy_physical(st, &smeta, dst, uid, gid, bytes_out);
248}
