2 * tmp_path.c — uniform temporary-file path construction for all subsystems.
3 *
4 * All three subsystems that write data atomically (WebDAV COPY, WebDAV TPC
5 * pull, S3 PUT) previously used ad-hoc suffix formats. Centralising the
6 * format means:
7 *
8 * - operators can clean orphaned temp files with a single glob:
9 * find /export -name "*.xrd-tmp.*" -mtime +1 -delete
10 * - the uniqueness strategy (pid + ngx_random()) is consistent everywhere
11 * - a future "scan for stale tmps" monitor only needs one pattern

27/*
28 * WHAT: Construct a unique temporary file path based on a base (final) path.
29 *
30 * WHY: Atomic writes need a temp file that won't collide with existing files or concurrent
31 * writers. The format <base>.xrd-tmp.<pid>.<random> guarantees uniqueness across processes
32 * and retries, while the .xrd-tmp. prefix lets operators glob-clean orphaned temps.
33 *
34 * HOW: snprintf combines base_path + ".xrd-tmp." + getpid() + ngx_random(). Returns NGX_OK
35 * on success, NGX_ERROR if truncation occurs (n >= out_sz or n < 0). Caller uses this
36 * path with O_EXCL open to guarantee no collision.
37 *
38 * Parameters:
39 * base_path — the final destination path (used as prefix for temp name)
40 * out — output buffer to receive the constructed tmp path
41 * out_sz — size of out buffer
42 */
43ngx_int_t
44brix_make_tmp_path(const char *base_path, char *out, size_t out_sz)
45{
46 int n;
47
48 n = snprintf(out, out_sz, "%s.xrd-tmp.%ld.%u",
49 base_path, (long) getpid(), (unsigned) ngx_random());
50 if (n < 0 || (size_t) n >= out_sz) {
51 return NGX_ERROR;
52 }
53
54 return NGX_OK;
55}

57/*
58 * WHAT: Construct the DETERMINISTIC resume-staging path for an upload, keyed to
59 * the authenticated principal AND the final path.
60 *
61 * WHY: Upload resume across an nginx restart needs a reconnecting client's
62 * re-open of the same final path (by the same identity) to land on the
63 * SAME staging file so the already-written bytes survive and the transfer
64 * continues from its offset. A random per-open temp (brix_make_tmp_path)
65 * cannot satisfy that — the second open would never find the first temp.
66 * Hashing the principal into the name also isolates users: a different
67 * identity derives a different staging file and can never reclaim another's
68 * partial (and the server only ever derives the name from the AUTHENTICATED
69 * identity, never from client-supplied data).
70 *
71 * HOW: out = "<base>.xrdresume.<hex16(SHA-256(principal "\0" base))>.part".
72 * The base prefix keeps the staging file adjacent to the destination so the
73 * commit rename(2) stays same-filesystem/atomic; the 16-byte (128-bit) hash
74 * prefix makes cross-(identity,path) collisions negligible. The
75 * ".xrdresume." infix lets operators glob-clean stale partials:
76 * find /export -name "*.xrdresume.*.part" -mtime +1 -delete
77 *
78 * Parameters:
79 * base_path — final destination absolute path
80 * principal — authenticated identity (DN / token subject); "" or NULL for an
81 * anonymous endpoint (all anonymous uploads to a path then share,
82 * consistent with that endpoint having no per-user isolation)
83 * out/out_sz — output buffer
84 *
85 * Returns NGX_OK, or NGX_ERROR on hash failure or truncation.
86 */
87ngx_int_t
88brix_make_resume_path(const char *base_path, const char *principal,
89 const char *stage_dir, char *out, size_t out_sz)
90{
91 uint8_t digest[32];
92 char hexbuf[33];
93 u_char concat[768 + PATH_MAX];
94 size_t wlen, blen, clen;
95 int n;
96 const char *who = (principal != NULL && principal[0] != '\0')
97 ? principal : "anonymous";
98
99 /* Hash principal || '\0' || base_path (the NUL keeps the two fields
100 * unambiguous so distinct (who, base) pairs cannot alias). */
101 wlen = strnlen(who, 767);
102 blen = strnlen(base_path, PATH_MAX - 1);
103 clen = 0;
104 ngx_memcpy(concat, who, wlen); clen += wlen;
105 concat[clen++] = '\0';
106 ngx_memcpy(concat + clen, base_path, blen); clen += blen;
107
108 /* brix_sha256 follows the OpenSSL convention: 1 = success, 0 = failure. */
109 if (brix_sha256(concat, clen, digest) != 1) {
110 return NGX_ERROR;
111 }
112
113 brix_hex_encode(digest, 16, hexbuf); /* 16 bytes -> 32 hex chars + NUL */
114
115 if (stage_dir != NULL && stage_dir[0] != '\0') {
116 /* Stage on the configured fast device. The basename is the
117 * server-generated hash (no client-controlled component), so the flat
118 * stage dir holds every in-progress upload without collision and the
119 * commit later moves it onto the storage. */
120 n = snprintf(out, out_sz, "%s/%s.xrdresume.part", stage_dir, hexbuf);
121 } else {
122 /* Stage adjacent to the destination (same filesystem → atomic rename). */
123 n = snprintf(out, out_sz, "%s.xrdresume.%s.part", base_path, hexbuf);
124 }
125 if (n < 0 || (size_t) n >= out_sz) {
126 return NGX_ERROR;
127 }
128
129 return NGX_OK;
130}

132/* ---- orphaned atomic-write temp reaper (phase-64 SP4) ---------------------
133 * A non-staged (direct) write streams the body to a "<final>.xrd-tmp.<pid>.<rand>"
134 * temp and renames it onto the final path on commit. A crash/kill mid-write leaves
135 * that temp orphaned in the export tree. Unlike a STAGED write (recoverable via the
136 * stage_engine reconcile), a broken direct write is simply discarded - so on
137 * startup we REAP these orphans. The owning pid is in the name: a temp whose pid is
138 * still alive is an IN-FLIGHT write of a draining worker (a reload) and is KEPT;
139 * only dead-owner (or malformed) temps are removed - safe under reload. */
140
141#define BRIX_TMP_REAP_MAX 64
142static char s_tmp_reap_roots[BRIX_TMP_REAP_MAX][PATH_MAX];
143static ngx_uint_t s_tmp_reap_count;
144static ngx_uint_t s_tmp_reaped; /* nftw has no ctx; worker-0 startup is single-threaded */
145
146void
147brix_tmp_reap_register(const char *export_root)
148{
149 ngx_uint_t i;
150
151 if (export_root == NULL || export_root[0] == '\0'
152 || strlen(export_root) >= PATH_MAX)
153 {
154 return;
155 }
156 /* Never reap "/" — a cache node anchored at the root namespace (no brix_root)
157 * yields root_canon "/", but nftw-walking the whole host filesystem (and
158 * unlinking matching temps anywhere) is catastrophic. A pure cache node has no
159 * local export-tree temps to reap (its in-flight writes live in the store
160 * tier), so there is nothing to walk here. */
161 if (export_root[0] == '/' && export_root[1] == '\0') {
162 return;
163 }
164 for (i = 0; i < s_tmp_reap_count; i++) {
165 if (strcmp(s_tmp_reap_roots[i], export_root) == 0) {
166 return; /* dedup across server blocks */
167 }
168 }
169 if (s_tmp_reap_count >= BRIX_TMP_REAP_MAX) {
170 return;
171 }
172 memcpy(s_tmp_reap_roots[s_tmp_reap_count], export_root,
173 strlen(export_root) + 1);
174 s_tmp_reap_count++;
175}

177/* nftw visitor: unlink a "*.xrd-tmp.<pid>.*" file whose owner pid is dead. */
178static int
179brix_tmp_reap_cb(const char *fpath, const struct stat *sb, int typeflag,
180 struct FTW *ftwbuf)
181{
182 const char *base;
183 const char *mark;
184 long pid;
185 char *end;
186
187 (void) sb;
188 (void) ftwbuf;
189 if (typeflag != FTW_F) {
190 return 0;
191 }
192 base = strrchr(fpath, '/');
193 base = (base != NULL) ? base + 1 : fpath;
194 mark = strstr(base, ".xrd-tmp.");
195 if (mark == NULL) {
196 return 0;
197 }
198 pid = strtol(mark + sizeof(".xrd-tmp.") - 1, &end, 10);
199 if (pid > 0 && end != mark + sizeof(".xrd-tmp.") - 1) {
200 /* A live owner = an in-flight write (a draining worker during reload). A
201 * dead owner (ESRCH) = an orphan. EPERM (foreign owner) = keep, be safe. */
202 if (kill((pid_t) pid, 0) == 0 || errno == EPERM) {
203 return 0;
204 }
205 }
206 if (unlink(fpath) == 0) {
207 s_tmp_reaped++;
208 }
209 return 0;
210}

212ngx_uint_t
213brix_tmp_reap_all(ngx_log_t *log)
214{
215 ngx_uint_t i;
216
217 s_tmp_reaped = 0;
218 for (i = 0; i < s_tmp_reap_count; i++) {
219 (void) nftw(s_tmp_reap_roots[i], brix_tmp_reap_cb, 16, FTW_PHYS);
220 }
221 if (s_tmp_reaped > 0 && log != NULL) {
222 ngx_log_error(NGX_LOG_NOTICE, log, 0,
223 "brix: reaped %ui orphaned upload temp(s) from interrupted "
224 "non-staged write(s)", s_tmp_reaped);
225 }
226 return s_tmp_reaped;
227}
