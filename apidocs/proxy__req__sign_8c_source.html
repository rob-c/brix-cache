1/* File: proxy_req_sign.c — RFC-3820 GSI proxy issuance/signing (phase-57 §F6)
2 * WHAT: brix_gsi_sign_pxyreq() ISSUES a proxy certificate: given the delegating
3 * client's signer EEC/proxy (PEM) + private key and a DER X509_REQ produced by
4 * brix_gsi_build_pxyreq(), it validates the request, builds an X.509 v3 proxy
5 * (serial = request CN, subject = request subject, issuer = signer, request
6 * pubkey, validity = signer's remaining lifetime, keyUsage inherited, critical
7 * proxyCertInfo with the RFC-3820 monotonic path length) and self-signs it with
8 * the signer key (SHA-256), returning the malloc'd proxy PEM.
9 *
10 * WHY: This is the delegating client's half of F6 multi-hop delegation — the
11 * kXGC_sigpxy step that turns the server's request into a usable proxy. Split
12 * out of proxy_req.c (phase-79 file-size guard) as one cohesive unit; the
13 * request build stays in proxy_req.c and the delegated-chain assembly lives in
14 * proxy_req_assemble.c. Ported faithfully from stock XrdSecgsi
15 * XrdCrypto/XrdCryptosslgsiAux.cc::XrdCryptosslX509SignProxyReq.
16 *
17 * HOW: parse signer PEM + request DER → subject must be '<signer>/CN=<serial>' →
18 * signer must be unexpired (proxy inherits its remaining lifetime) → verify the
19 * request self-signature → populate the v3 proxy skeleton → copy the signer's
20 * non-SAN/non-pci extensions (keyUsage required, SAN rejected) and add a fresh
21 * critical proxyCertInfo whose path length is min(request, signer-1) → X509_sign
22 * → PEM export. No goto: a single NULL-safe cleanup (sgn_fail) frees everything
23 * not handed to the caller. The shared proxyCertInfo builder pxr_make_pci_ext()
24 * and the GSI OID constants come from proxy_req_internal.h. */

40/* pcPathLengthConstraint of an X509_REQ's proxyCertInfo, or -1 if none. */
41static int
42pxr_req_pathlen(X509_REQ *req)
43{
44 STACK_OF(X509_EXTENSION) *exts = X509_REQ_get_extensions(req);
45 int depth = -1, i;
46
47 for (i = 0; exts != NULL && i < sk_X509_EXTENSION_num(exts); i++) {
48 X509_EXTENSION *e = sk_X509_EXTENSION_value(exts, i);
49 char oid[128];
50
51 OBJ_obj2txt(oid, sizeof(oid), X509_EXTENSION_get_object(e), 1);
52 if (strcmp(oid, GSI_PROXYCERTINFO_OID) == 0) {
53 const unsigned char *p =
54 ASN1_STRING_get0_data(X509_EXTENSION_get_data(e));
55 int len =
56 ASN1_STRING_length(X509_EXTENSION_get_data(e));
57 PROXY_CERT_INFO_EXTENSION *in =
58 d2i_PROXY_CERT_INFO_EXTENSION(NULL, &p, len);
59
60 if (in != NULL) {
61 if (in->pcPathLengthConstraint != NULL) {
62 depth = (int) ASN1_INTEGER_get(in->pcPathLengthConstraint);
63 }
64 PROXY_CERT_INFO_EXTENSION_free(in);
65 }
66 break;
67 }
68 }
69 if (exts != NULL) {
70 sk_X509_EXTENSION_pop_free(exts, X509_EXTENSION_free);
71 }
72 return depth;
73}

90/* Record msg in the caller's error buffer, free every owned resource, and
91 * return -1 — the single cleanup tail for the whole issuance (no goto).
92 * Called with msg == NULL on the success path purely to free leftovers. */
93static int
94sgn_fail(sgn_ctx *x, const char *msg)
95{
96 if (x->err != NULL && x->errcap > 0 && msg != NULL) {
97 snprintf(x->err, x->errcap, "%s", msg);
98 }
99 if (x->sbio) BIO_free(x->sbio);
100 if (x->signer) X509_free(x->signer);
101 if (x->req) X509_REQ_free(x->req);
102 if (x->reqpub) EVP_PKEY_free(x->reqpub);
103 if (x->proxy) X509_free(x->proxy);
104 if (x->pciext) X509_EXTENSION_free(x->pciext);
105 if (x->obio) BIO_free(x->obio);
106 return -1;
107}

109/* Copy the signer's extensions into the proxy (keyUsage required, SAN rejected,
110 * proxyCertInfo skipped); report the signer's pci pathlen via *indepth. -1 err. */
111static int
112sgn_copy_signer_exts(X509 *signer, X509 *proxy, int *indepth, int *haskeyusage)
113{
114 int n = X509_get_ext_count(signer);
115 int i;
116
117 *indepth = -1;
118 *haskeyusage = 0;
119 for (i = 0; i < n; i++) {
120 X509_EXTENSION *e = X509_get_ext(signer, i);
121 char oid[128];
122
123 OBJ_obj2txt(oid, sizeof(oid), X509_EXTENSION_get_object(e), 1);
124 if (strcmp(oid, GSI_PROXYCERTINFO_OID) == 0
125 || strcmp(oid, GSI_PROXYCERTINFO_OLD_OID) == 0) {
126 const unsigned char *p =
127 ASN1_STRING_get0_data(X509_EXTENSION_get_data(e));
128 int len =
129 ASN1_STRING_length(X509_EXTENSION_get_data(e));
130 PROXY_CERT_INFO_EXTENSION *in =
131 d2i_PROXY_CERT_INFO_EXTENSION(NULL, &p, len);
132
133 if (in != NULL) {
134 if (in->pcPathLengthConstraint != NULL) {
135 *indepth = (int) ASN1_INTEGER_get(in->pcPathLengthConstraint);
136 }
137 PROXY_CERT_INFO_EXTENSION_free(in);
138 }
139 continue; /* do not copy the pci */
140 }
141 if (strcmp(oid, GSI_SUBJ_ALT_NAME_OID) == 0) {
142 return -1; /* SAN not allowed in a proxy */
143 }
144 if (strcmp(oid, GSI_KEY_USAGE_OID) == 0) {
145 *haskeyusage = 1;
146 }
147 if (X509_add_ext(proxy, e, -1) == 0) { /* X509_add_ext dups */
148 return -1;
149 }
150 }
151 return 0;
152}

154/* Parse the signer cert (PEM) into x->signer and the request (DER) into
155 * x->req. 0 on success; -1 via sgn_fail (everything freed) on error. */
156static int
157sgn_parse_inputs(sgn_ctx *x, const uint8_t *signer_pem, size_t signer_pem_len,
158 const uint8_t *req_der, size_t req_len)
159{
160 const unsigned char *p;
161
162 x->sbio = BIO_new_mem_buf(signer_pem, (int) signer_pem_len);
163 x->signer = x->sbio ? PEM_read_bio_X509(x->sbio, NULL, NULL, NULL) : NULL;
164 if (x->signer == NULL) {
165 return sgn_fail(x, "gsi sign: cannot parse signer cert");
166 }
167 p = req_der;
168 x->req = d2i_X509_REQ(NULL, &p, (long) req_len);
169 if (x->req == NULL) {
170 return sgn_fail(x, "gsi sign: cannot parse request");
171 }
172 return 0;
173}

175/* Enforce that the request subject is exactly '<signer subject>/CN=<serial>'
176 * and extract the serial. 0 on success; -1 via sgn_fail on error. */
177static int
178sgn_check_subject(sgn_ctx *x, long *serial)
179{
180 char sname[1024], rname[1024];
181 size_t slen;
182 char *endp;
183
184 if (X509_NAME_oneline(X509_get_subject_name(x->signer), sname,
185 sizeof(sname)) == NULL
186 || X509_NAME_oneline(X509_REQ_get_subject_name(x->req), rname,
187 sizeof(rname)) == NULL) {
188 return sgn_fail(x, "gsi sign: subject undefined");
189 }
190 slen = strlen(sname);
191 if (strncmp(rname, sname, slen) != 0
192 || strncmp(rname + slen, "/CN=", 4) != 0) {
193 return sgn_fail(x,
194 "gsi sign: request subject not '<signer>/CN=<serial>'");
195 }
196 *serial = strtol(rname + slen + 4, &endp, 10);
197 if (endp == rname + slen + 4 || *endp != '\0' || *serial < 0) {
198 return sgn_fail(x, "gsi sign: bad serial in request CN");
199 }
200 return 0;
201}

203/* The signer must not be expired; *timeleft gets its remaining lifetime in
204 * seconds (the proxy inherits it). 0 on success; -1 via sgn_fail on error. */
205static int
206sgn_check_signer_validity(sgn_ctx *x, int *timeleft)
207{
208 int days = 0, secs = 0;
209
210 /* seconds left = diff(now, signer notAfter) */
211 if (ASN1_TIME_diff(&days, &secs, NULL, X509_get0_notAfter(x->signer)) != 1) {
212 return sgn_fail(x, "gsi sign: cannot read signer expiry");
213 }
214 *timeleft = days * 86400 + secs;
215 if (*timeleft <= 0) {
216 return sgn_fail(x, "gsi sign: signer certificate expired");
217 }
218 return 0;
219}

221/* Verify the request's self-signature (leaves the pubkey in x->reqpub for the
222 * proxy build). 0 on success; -1 via sgn_fail on error. */
223static int
224sgn_verify_req_signature(sgn_ctx *x)
225{
226 x->reqpub = X509_REQ_get_pubkey(x->req);
227 if (x->reqpub == NULL || X509_REQ_verify(x->req, x->reqpub) != 1) {
228 return sgn_fail(x, "gsi sign: request self-signature invalid");
229 }
230 return 0;
231}

233/* Build the skeleton proxy cert in x->proxy: version 3, serial from the CN,
234 * subject from the request, issuer = signer, request pubkey, validity
235 * [now, now+timeleft]. 0 on success; -1 via sgn_fail on error. */
236static int
237sgn_populate_proxy(sgn_ctx *x, long serial, int timeleft)
238{
239 x->proxy = X509_new();
240 if (x->proxy == NULL
241 || X509_set_version(x->proxy, 2L) != 1
242 || ASN1_INTEGER_set(X509_get_serialNumber(x->proxy),
243 (long) serial) != 1
244 || X509_set_subject_name(x->proxy,
245 X509_REQ_get_subject_name(x->req)) != 1
246 || X509_set_issuer_name(x->proxy, X509_get_subject_name(x->signer)) != 1
247 || X509_set_pubkey(x->proxy, x->reqpub) != 1
248 || X509_gmtime_adj(X509_getm_notBefore(x->proxy), 0) == NULL
249 || X509_gmtime_adj(X509_getm_notAfter(x->proxy), timeleft) == NULL) {
250 return sgn_fail(x, "gsi sign: cannot populate proxy");
251 }
252 return 0;
253}

255/* Copy the signer's extensions into the proxy (keyUsage required, SAN
256 * rejected) and add a fresh critical proxyCertInfo whose path length is the
257 * RFC-3820 monotonic min(request, signer-1) — pxr_make_pci_ext decrements
258 * once more. 0 on success; -1 via sgn_fail on error. */
259static int
260sgn_add_extensions(sgn_ctx *x)
261{
262 int indepth, reqdepth, outdepth, haskeyusage;
263
264 if (sgn_copy_signer_exts(x->signer, x->proxy, &indepth, &haskeyusage) != 0) {
265 return sgn_fail(x,
266 "gsi sign: bad signer extensions (SAN, or copy failed)");
267 }
268 if (!haskeyusage) {
269 return sgn_fail(x, "gsi sign: signer lacks keyUsage");
270 }
271
272 /* Path-length: min(req, signer-1); the helper decrements once more. */
273 reqdepth = pxr_req_pathlen(x->req);
274 outdepth = (reqdepth < indepth) ? reqdepth : (indepth - 1);
275 x->pciext = pxr_make_pci_ext(outdepth);
276 if (x->pciext == NULL || X509_add_ext(x->proxy, x->pciext, -1) == 0) {
277 return sgn_fail(x, "gsi sign: cannot add proxyCertInfo");
278 }
279 return 0;
280}

282/* PEM-export the signed proxy into a malloc'd NUL-terminated buffer for the
283 * caller. 0 on success; -1 via sgn_fail on error. */
284static int
285sgn_export_pem(sgn_ctx *x, uint8_t **proxy_pem, size_t *proxy_len)
286{
287 char *pem = NULL;
288 long pemlen;
289
290 x->obio = BIO_new(BIO_s_mem());
291 if (x->obio == NULL || PEM_write_bio_X509(x->obio, x->proxy) != 1) {
292 return sgn_fail(x, "gsi sign: PEM export failed");
293 }
294 pemlen = BIO_get_mem_data(x->obio, &pem);
295 *proxy_pem = malloc((size_t) pemlen + 1);
296 if (*proxy_pem == NULL) {
297 return sgn_fail(x, "gsi sign: out of memory");
298 }
299 memcpy(*proxy_pem, pem, (size_t) pemlen);
300 (*proxy_pem)[pemlen] = '\0';
301 *proxy_len = (size_t) pemlen;
302 return 0;
303}

316int
317brix_gsi_sign_pxyreq(const brix_gsi_blob_t *signer_pem, EVP_PKEY *signer_key,
318 const brix_gsi_blob_t *req_der,
319 brix_gsi_buf_t *proxy_pem, const brix_gsi_err_t *err)
320{
321 sgn_ctx x;
322 long serial = 0;
323 int timeleft = 0;
324
325 memset(&x, 0, sizeof(x));
326 if (err != NULL) {
327 x.err = err->buf;
328 x.errcap = err->cap;
329 }
330 if (!sgn_args_ok(signer_pem, signer_key, req_der, proxy_pem)) {
331 return sgn_fail(&x, "gsi sign: bad arguments");
332 }
333
334 if (sgn_parse_inputs(&x, signer_pem->data, signer_pem->len,
335 req_der->data, req_der->len) != 0
336 || sgn_check_subject(&x, &serial) != 0
337 || sgn_check_signer_validity(&x, &timeleft) != 0
338 || sgn_verify_req_signature(&x) != 0
339 || sgn_populate_proxy(&x, serial, timeleft) != 0
340 || sgn_add_extensions(&x) != 0) {
341 return -1; /* helpers already freed via sgn_fail */
342 }
343
344 /* Sign with the issuer's key. */
345 if (X509_sign(x.proxy, signer_key, EVP_sha256()) <= 0) {
346 return sgn_fail(&x, "gsi sign: signing failed");
347 }
348
349 if (sgn_export_pem(&x, &proxy_pem->data, &proxy_pem->len) != 0) {
350 return -1;
351 }
352 (void) sgn_fail(&x, NULL);
353 return 0;
354}
