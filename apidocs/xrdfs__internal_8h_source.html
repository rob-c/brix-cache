26/*
27 * WHAT: emit credential + WS-3 double-slash hints at xrdfs root:// error sites.
28 * WHY: consolidates the three-step hint sequence (cred hint + doctor referral +
29 * double-slash) into one call so every root:// op error site is covered
30 * without repeating the URL construction logic. Replaces every standalone
31 * brix_cred_hint_for_status(&st, ..., stderr) call in the xrdfs handlers.
32 * HOW: constructs a host[:port] endpoint string from c->host/c->port for the
33 * doctor-referral hint (WS-7); builds a minimal brix_url snapshot carrying
34 * c->single_slash_path for the double-slash hint (WS-3). Both hints are
35 * TTY-gated and fire at most once per process (brix_cli_hint_once).
36 */
37static inline void
38xrdfs_op_hints(const brix_status *st, int want_write, const brix_conn *c)
39{
40 char epbuf[300]; /* host(256) + ':' + port digits + NUL */
41 brix_url snap;
42
43 /* Build a host:port endpoint string for the doctor-referral hint (WS-7). */
44 if (c->port > 0) {
45 snprintf(epbuf, sizeof(epbuf), "%s:%d", c->host, c->port);
46 } else {
47 snprintf(epbuf, sizeof(epbuf), "%s", c->host);
48 }
49 brix_cred_hint_for_status_url(st, want_write, stderr, epbuf);
50
51 /* Build a minimal URL snapshot for the double-slash hint (WS-3). */
52 memset(&snap, 0, sizeof(snap));
53 snap.scheme = XRDC_SCHEME_ROOT;
54 snprintf(snap.host, sizeof(snap.host), "%s", c->host);
55 snap.port = c->port;
56 snap.single_slash_path = c->single_slash_path;
57 brix_hint_url_double_slash(st, &snap);
58}
