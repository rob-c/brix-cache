1/*
2 * privs.c — XrdAcc privilege algebra (operation requirements, parser, test).
3 *
4 * WHAT: implements the three pure functions declared in privs.h — the
5 * operation->required-privilege table (brix_acc_op_needs), the grant test
6 * (brix_acc_test), and the single-letter privilege parser
7 * (brix_acc_parse_privs).
8 *
9 * WHY: faithful ports of XrdAccAccess::Test() (the need[] table) and
10 * XrdAccConfig::PrivsConvert() so the `xrdacc` engine reproduces stock XRootD
11 * authorization decisions bit-for-bit. Pure leaf — depends only on privs.h.
12 *
13 * HOW: a static need[] table indexed by Access_Operation; Test() returns
14 * (need & priv) == need; PrivsConvert() walks the letter string accumulating
15 * into a 2-slot table that the single '-' advances from positive to negative.
16 */
17
18#include "privs.h"

20/*
21 * need[] — privilege bits required per operation, 1:1 with brix_acc_op_t.
22 * Mirrors XrdAccAccess::Test()'s table. AOP_EXCL_* (13,14) require 0xffff so
23 * they can never be satisfied by the privilege model (exclusivity is enforced
24 * elsewhere). We additionally pin AOP_STAGE/AOP_POLL (15,16) — XRootD's own
25 * table stops at index 14 and reads out of bounds for these; we map them to
26 * their documented composites instead.
27 */
28static const brix_acc_privs_t brix_acc_op_need[BRIX_AOP_LAST + 1] = {
29 [BRIX_AOP_ANY] = BRIX_ACC_PRIV_NONE,
30 [BRIX_AOP_CHMOD] = BRIX_ACC_PRIV_CHMOD,
31 [BRIX_AOP_CHOWN] = BRIX_ACC_PRIV_CHMOD, /* Chown == Chmod (0x063) */
32 [BRIX_AOP_CREATE] = BRIX_ACC_PRIV_CREATE,
33 [BRIX_AOP_DELETE] = BRIX_ACC_PRIV_DELETE,
34 [BRIX_AOP_INSERT] = BRIX_ACC_PRIV_INSERT,
35 [BRIX_AOP_LOCK] = BRIX_ACC_PRIV_LOCK,
36 [BRIX_AOP_MKDIR] = BRIX_ACC_PRIV_INSERT, /* Mkdir == Insert */
37 [BRIX_AOP_READ] = BRIX_ACC_PRIV_READ,
38 [BRIX_AOP_READDIR] = BRIX_ACC_PRIV_READ, /* Readdir == Read */
39 [BRIX_AOP_RENAME] = BRIX_ACC_PRIV_RENAME,
40 [BRIX_AOP_STAT] = BRIX_ACC_PRIV_LOOKUP, /* Stat == Lookup */
41 [BRIX_AOP_UPDATE] = BRIX_ACC_PRIV_UPDATE,
42 [BRIX_AOP_EXCL_CREATE] = 0xffffu,
43 [BRIX_AOP_EXCL_INSERT] = 0xffffu,
44 [BRIX_AOP_STAGE] = BRIX_ACC_PRIV_STAGE,
45 [BRIX_AOP_POLL] = BRIX_ACC_PRIV_POLL,
46};

48brix_acc_privs_t
49brix_acc_op_needs(brix_acc_op_t op)
50{
51 if (op < 0 || op > BRIX_AOP_LAST) {
52 return 0xffffu; /* unknown operation -> unsatisfiable -> deny */
53 }
54 return brix_acc_op_need[op];
55}

57int
58brix_acc_test(brix_acc_privs_t priv, brix_acc_op_t op)
59{
60 brix_acc_privs_t need = brix_acc_op_needs(op);
61
62 /* Grant iff every required bit is present. For AOP_ANY (need==0) this is
63 * trivially true; the engine returns the privilege set directly for that. */
64 return (brix_acc_privs_t) (need & priv) == need;
65}

81int
82brix_acc_parse_privs(const char *s, size_t len, brix_acc_priv_caps_t *caps)
83{
84 brix_acc_privs_t tab[2] = { BRIX_ACC_PRIV_NONE, BRIX_ACC_PRIV_NONE };
85 int slot = 0; /* 0 = positive, 1 = negative */
86 size_t i;
87
88 for (i = 0; i < len; i++) {
89 switch (s[i]) {
90 case 'a': tab[slot] |= BRIX_ACC_PRIV_ALL; break;
91 case 'd': tab[slot] |= BRIX_ACC_PRIV_DELETE; break;
92 case 'i': tab[slot] |= BRIX_ACC_PRIV_INSERT; break;
93 case 'k': tab[slot] |= BRIX_ACC_PRIV_LOCK; break;
94 case 'l': tab[slot] |= BRIX_ACC_PRIV_LOOKUP; break;
95 case 'n': tab[slot] |= BRIX_ACC_PRIV_RENAME; break;
96 case 'r': tab[slot] |= BRIX_ACC_PRIV_READ; break;
97 case 'w': tab[slot] |= BRIX_ACC_PRIV_WRITE; break;
98 case '-':
99 /* Exactly one '-' is allowed: it switches to negative privileges. */
100 if (slot != 0) {
101 return -1;
102 }
103 slot = 1;
104 break;
105 default:
106 return -1; /* unknown privilege character */
107 }
108 }
109
110 caps->pprivs = tab[0];
111 caps->nprivs = tab[1];
112 return 0;
113}
