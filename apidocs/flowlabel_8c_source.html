1/*
2 * flowlabel.c — in-band IPv6 Flow Label marking (REQUIRED SciTags technique).
3 *
4 * WHAT: Stamp the SciTags flow-id into the 20-bit IPv6 Flow Label of a connected
5 * socket's egress packets, so routers/NRENs can classify the traffic without
6 * seeing the out-of-band firefly. This is the path XRootD declared but never
7 * implemented (its marking site is an empty `// { TODO??? }` at
8 * XrdNetPMarkCfg.cc:240-248); we provide the real Linux kernel calls.
9 *
10 * WHY: The flow label is the per-packet technique; firefly is per-flow reporting.
11 * The phase-34 design requires BOTH. Marking is fail-open: IPv4, IPv4-mapped,
12 * a disabled config, or a kernel that refuses the specific label all degrade to
13 * "not labelled" (firefly still runs) and NEVER fail a transfer.
14 *
15 * HOW: Reserve the exact label for the connected peer via
16 * setsockopt(IPV6_FLOWLABEL_MGR) with IPV6_FL_A_GET|IPV6_FL_F_CREATE, then
17 * enable IPV6_FLOWINFO_SEND so the kernel writes it into outgoing headers. The
18 * struct/constants live only in <linux/in6.h>, which clashes with
19 * <netinet/in.h>; to avoid the conflict we declare the (stable, append-only)
20 * kernel ABI here under private names. A one-time per-worker probe records
21 * whether this host even permits setting a specific label (needs the right
22 * net.ipv6 sysctls / CAP_NET_ADMIN for some ranges).
23 */

47struct pmark_flowlabel_req {
48 struct in6_addr flr_dst;
49 uint32_t flr_label; /* network byte order, low 20 bits */
50 uint8_t flr_action;
51 uint8_t flr_share;

62ngx_int_t
63brix_pmark_flowlabel_usable(ngx_log_t *log)
64{
65 int fd;
66 struct pmark_flowlabel_req fl;
67
68 if (pmark_fl_usable >= 0) {
69 return pmark_fl_usable ? NGX_OK : NGX_DECLINED;
70 }
71 pmark_fl_usable = 0;
72
73 fd = (int) ngx_socket(AF_INET6, SOCK_DGRAM, 0);
74 if (fd < 0) {
75 BRIX_DIAG(NGX_LOG_NOTICE, log, ngx_errno,
76 "pmark: IPv6 flow-label marking disabled (no AF_INET6 socket)",
77 "this host has IPv6 disabled, so SciTags cannot set the IPv6 "
78 "flow label; packet marking falls back to firefly UDP only",
79 "if you want flow-label marking, enable IPv6 on the host; "
80 "otherwise this is expected and firefly marking still works");
81 return NGX_DECLINED;
82 }
83
84 /* Try to lease a specific label toward the loopback address. If the kernel
85 * refuses (EPERM / sysctl off), specific-label marking is not possible. */
86 ngx_memzero(&fl, sizeof(fl));
87 fl.flr_dst = in6addr_loopback;
88 /* A representative in-range structural label (exp/act minima) just to learn
89 * whether the kernel will lease a SPECIFIC label on this host at all. */
90 fl.flr_label = htonl(brix_pmark_flowlabel_encode(BRIX_PMARK_EXP_MIN,
91 BRIX_PMARK_ACT_MIN));
92 fl.flr_action = PMARK_FL_A_GET;
93 fl.flr_flags = PMARK_FL_F_CREATE;
94 fl.flr_share = PMARK_FL_S_EXCL;
95
96 if (setsockopt(fd, IPPROTO_IPV6, IPV6_FLOWLABEL_MGR, &fl, sizeof(fl)) == 0) {
97 pmark_fl_usable = 1;
98 } else {
99 ngx_log_error(NGX_LOG_NOTICE, log, ngx_errno,
100 "pmark: IPv6 flow-label marking unavailable "
101 "(check net.ipv6.flowlabel_state_ranges / CAP_NET_ADMIN); "
102 "firefly-only");
103 }
104
105 ngx_close_socket(fd);
106 return pmark_fl_usable ? NGX_OK : NGX_DECLINED;
107}

110/* Lease the encoded label on `fd` toward IPv6 destination `dst6` and enable
111 * IPV6_FLOWINFO_SEND. Shared by the connected (getpeername) and explicit-dest
112 * (libcurl opensocket) entry points. */
113static ngx_int_t
114pmark_flowlabel_lease(int fd, const struct in6_addr *dst6, ngx_uint_t exp,
115 ngx_uint_t act, ngx_log_t *log)
116{
117 struct pmark_flowlabel_req fl;
118 uint32_t label;
119 int on = 1;
120
121 /* Structural bits (community + activity) plus 5 random entropy bits, set once
122 * here per flow so same-(exp,act) flows hash differently for ECMP (spec §4). */
123 label = brix_pmark_flowlabel_encode(exp, act)
124 | ((uint32_t) ngx_random() & BRIX_PMARK_FL_ENTROPY_MASK);
125
126 ngx_memzero(&fl, sizeof(fl));
127 fl.flr_dst = *dst6;
128 fl.flr_label = htonl(label);
129 fl.flr_action = PMARK_FL_A_GET;
130 fl.flr_flags = PMARK_FL_F_CREATE;
131 fl.flr_share = PMARK_FL_S_EXCL;
132
133 if (setsockopt(fd, IPPROTO_IPV6, IPV6_FLOWLABEL_MGR, &fl, sizeof(fl)) != 0) {
134 BRIX_PMARK_METRIC_INC(pmark_flowlabel_failed_total);
135 ngx_log_debug1(NGX_LOG_DEBUG_CORE, log, ngx_errno,
136 "pmark: flow-label lease failed (label=0x%05xui)", label);
137 return NGX_DECLINED;
138 }
139 (void) setsockopt(fd, IPPROTO_IPV6, IPV6_FLOWINFO_SEND, &on, sizeof(on));
140
141 BRIX_PMARK_METRIC_INC(pmark_flowlabel_set_total);
142 ngx_log_debug1(NGX_LOG_DEBUG_CORE, log, 0,
143 "pmark: stamped IPv6 flow label 0x%05xui", label);
144 return NGX_OK;
145}

148ngx_int_t
149brix_pmark_flowlabel_apply(ngx_connection_t *c, int fd, ngx_uint_t exp,
150 ngx_uint_t act, ngx_log_t *log)
151{
152 struct sockaddr_storage ss;
153 socklen_t len = sizeof(ss);
154 struct sockaddr_in6 *sin6;
155
156 (void) c;
157
158 if (fd < 0 || brix_pmark_flowlabel_usable(log) != NGX_OK) {
159 return NGX_DECLINED;
160 }
161 if (getpeername(fd, (struct sockaddr *) &ss, &len) != 0
162 || ss.ss_family != AF_INET6)
163 {
164 return NGX_DECLINED; /* IPv4 (or no peer) → firefly only */
165 }
166 sin6 = (struct sockaddr_in6 *) &ss;
167 if (IN6_IS_ADDR_V4MAPPED(&sin6->sin6_addr)) {
168 return NGX_DECLINED; /* ::ffff: → not real IPv6 */
169 }
170 return pmark_flowlabel_lease(fd, &sin6->sin6_addr, exp, act, log);
171}
