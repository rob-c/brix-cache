35static void
36write_file(const char *path, const unsigned char *data, size_t len)
37{
38 int fd = open(path, O_WRONLY | O_CREAT | O_TRUNC, 0644);
39
40 if (fd < 0 || write(fd, data, len) != (ssize_t) len) {
41 perror("write_file");
42 exit(2);
43 }
44 close(fd);
45}

48main(void)
49{
50 char dir[] = "/tmp/csi_ut.XXXXXX";
51 char path[4400];
52 unsigned char data[3 * G]; /* 3 full blocks */
53 brix_csi_t w, r;
54 size_t i;
55
56 if (mkdtemp(dir) == NULL) {
57 perror("mkdtemp");
58 return 2;
59 }
60 snprintf(path, sizeof(path), "%s/a.bin", dir);
61 for (i = 0; i < sizeof(data); i++) {
62 data[i] = (unsigned char) (i * 31u + 7u);
63 }
64 write_file(path, data, sizeof(data));
65
66 /* ---- write handle: fold + flush creates the record ---- */
67 CHECK(brix_csi_open(&w, path, G, 1) == BRIX_CSI_OK, "write open");
68 CHECK(brix_csi_write_update(&w, data, 0, sizeof(data)) == BRIX_CSI_OK,
69 "fold 3 aligned blocks");
70 CHECK(brix_csi_flush(&w) == BRIX_CSI_OK, "flush writes the record");
71 brix_csi_close(&w);
72
73 /* the record is the file's own xmeta (xattr or sidecar) */
74 {
75 brix_xmeta_t xm;
76
77 CHECK(brix_xmeta_path_load(path, &xm) == BRIX_XMETA_OK
78 && xm.have_blockcrc && xm.blockcrc != NULL
79 && xm.buffer_size == (int64_t) G && xm.nblocks == 3
80 && xm.blockcrc[0] != 0 && xm.blockcrc[1] != 0
81 && xm.blockcrc[2] != 0,
82 "record holds 3 computed block CRCs");
83 CHECK(xm.no_cksum_time == 0, "fully tagged: no_cksum_time clear");
84 brix_xmeta_free(&xm);
85 }
86
87 /* ---- read handle: verify clean / corrupt / restored ---- */
88 CHECK(brix_csi_open(&r, path, G, 0) == BRIX_CSI_OK,
89 "read open sees a verifiable record");
90 CHECK(brix_csi_verify_read(&r, data, 0, sizeof(data)) == BRIX_CSI_OK,
91 "verify clean data passes");
92
93 data[G + 10] ^= 0xFF; /* corrupt block 1 */
94 CHECK(brix_csi_verify_read(&r, data, 0, sizeof(data))
95 == BRIX_CSI_MISMATCH, "corrupt block detected");
96 CHECK(brix_csi_verify_read(&r, data, 0, G) == BRIX_CSI_OK,
97 "read not spanning the bad block passes");
98 CHECK(brix_csi_verify_read(&r, data + G, G, G)
99 == BRIX_CSI_MISMATCH, "exact bad-block read fails");
100 /* a read only PARTIALLY covering the bad block skips it (hot-path rule) */
101 CHECK(brix_csi_verify_read(&r, data + G, G, G / 2) == BRIX_CSI_OK,
102 "partial-edge block is not verified");
103
104 /* trust_fs bypass */
105 r.trust_fs = 1;
106 CHECK(brix_csi_verify_read(&r, data, 0, sizeof(data)) == BRIX_CSI_OK,
107 "trust_fs skips verify on corrupt block");
108 r.trust_fs = 0;
109
110 data[G + 10] ^= 0xFF; /* restore */
111 CHECK(brix_csi_verify_read(&r, data, 0, sizeof(data)) == BRIX_CSI_OK,
112 "restored data verifies");
113 brix_csi_close(&r);
114
115 /* ---- unaligned write: edge blocks recomputed at flush ---- */
116 memset(data + 100, 0x5A, G); /* mutate an unaligned window */
117 write_file(path, data, sizeof(data));
118 CHECK(brix_csi_open(&w, path, G, 1) == BRIX_CSI_OK, "reopen write");
119 CHECK(brix_csi_write_update(&w, data + 100, 100, G) == BRIX_CSI_OK,
120 "fold unaligned write (no full block covered)");
121 CHECK(brix_csi_flush(&w) == BRIX_CSI_OK, "flush recomputes edges");
122 brix_csi_close(&w);
123 CHECK(brix_csi_open(&r, path, G, 0) == BRIX_CSI_OK, "read reopen");
124 CHECK(brix_csi_verify_read(&r, data, 0, sizeof(data)) == BRIX_CSI_OK,
125 "post-unaligned-write data verifies");
126 brix_csi_close(&r);
127
128 /* ---- growth: flush re-geometries the record ---- */
129 {
130 unsigned char *big = malloc(5 * G);
131
132 memcpy(big, data, sizeof(data));
133 memset(big + 3 * G, 0xA5, 2 * G);
134 write_file(path, big, 5 * G);
135 brix_csi_open(&w, path, G, 1);
136 brix_csi_write_update(&w, big + 3 * G, 3 * G, 2 * G);
137 CHECK(brix_csi_flush(&w) == BRIX_CSI_OK, "flush after growth");
138 brix_csi_close(&w);
139 {
140 brix_xmeta_t xm;
141
142 CHECK(brix_xmeta_path_load(path, &xm) == BRIX_XMETA_OK
143 && xm.nblocks == 5 && xm.file_size == 5 * G,
144 "record re-geometried to 5 blocks");
145 brix_xmeta_free(&xm);
146 }
147 CHECK(brix_csi_open(&r, path, G, 0) == BRIX_CSI_OK, "read grown");
148 CHECK(brix_csi_verify_read(&r, big, 0, 5 * G) == BRIX_CSI_OK,
149 "grown file verifies end-to-end");
150 brix_csi_close(&r);
151 free(big);
152 }

154 /* ---- load-once: a read handle snapshots the record at open and keeps
155 * verifying against it even after the on-disk record is removed. A per-read
156 * reload implementation would see no record and stop catching corruption —
157 * this is the regression guard for the multi-threaded read hot path. ---- */
158 {
159 char p3[4400];
160 brix_csi_t rc;
161 unsigned char d3[2 * G];
162 size_t k;
163
164 for (k = 0; k < sizeof(d3); k++) {
165 d3[k] = (unsigned char) (k * 17u + 3u);
166 }
167 snprintf(p3, sizeof(p3), "%s/cached.bin", dir);
168 write_file(p3, d3, sizeof(d3));
169
170 CHECK(brix_csi_open(&w, p3, G, 1) == BRIX_CSI_OK, "cached: write open");
171 CHECK(brix_csi_write_update(&w, d3, 0, sizeof(d3)) == BRIX_CSI_OK,
172 "cached: fold blocks");
173 CHECK(brix_csi_flush(&w) == BRIX_CSI_OK, "cached: flush writes record");
174 brix_csi_close(&w);
175
176 CHECK(brix_csi_open(&rc, p3, G, 0) == BRIX_CSI_OK,
177 "cached: read open loads the record once");
178 /* Drop both on-disk carriers: a load-once handle keeps its snapshot. */
179 brix_xmeta_path_remove(p3);
180 CHECK(brix_csi_verify_read(&rc, d3, 0, sizeof(d3)) == BRIX_CSI_OK,
181 "cached: clean verify survives record removal (load-once)");
182 d3[G + 5] ^= 0xFF; /* corrupt block 1 */
183 CHECK(brix_csi_verify_read(&rc, d3, 0, sizeof(d3)) == BRIX_CSI_MISMATCH,
184 "cached: corruption still caught after removal (load-once)");
185 brix_csi_close(&rc);
186 }
187
188 /* ---- no record: read open reports NOTAGS ---- */
189 {
190 char p2[4400];
191 brix_csi_t n;
192
193 snprintf(p2, sizeof(p2), "%s/naked.bin", dir);
194 write_file(p2, data, G);
195 CHECK(brix_csi_open(&n, p2, G, 0) == BRIX_CSI_NOTAGS,
196 "unrecorded file reads as NOTAGS (csi_require gate)");
197 }
198
199 printf("csi_unittest: %d checks, %d failed\n", g_checks, g_failed);
200 return g_failed == 0 ? 0 : 1;
