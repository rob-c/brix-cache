35/*
36 * WHAT: check_issued override accepting a name-matching issuer for an RFC 3820
37 * proxy subject even when its authorityKeyIdentifier does not match the
38 * issuer's subjectKeyIdentifier.
39 * WHY: xrdgsiproxy/voms-proxy-init copy the EEC's own AKID into the delegated
40 * proxy, so OpenSSL's default check_issued rejects the signing EEC as the
41 * proxy's issuer (AKID_SKID_MISMATCH) and reports "unable to get local
42 * issuer". The reference XRootD chain selects issuers by subject name +
43 * signature; match that.
44 * HOW: defer to X509_check_issued; on the AKID objections, accept when the
45 * subject is a recognised proxy (EXFLAG_PROXY). The RSA signature is
46 * still verified afterwards, so this relaxes selection, not trust.
47 */
48static int
49brix_sp_proxy_check_issued(X509_STORE_CTX *ctx, X509 *subject, X509 *issuer)
50{
51 int rv;
52
53 (void) ctx;
54
55 rv = X509_check_issued(issuer, subject);
56 if (rv == X509_V_OK) {
57 return 1;
58 }
59 /*
60 * The authorityKeyIdentifier is advisory (RFC 5280 §4.2.1.1): a mismatch
61 * must not prevent selecting a name-matching issuer. Accept the AKID/SKID
62 * and AKID issuer-serial objections for ANY subject (not just proxies) —
63 * the issuer's signature over the subject is still verified afterwards by
64 * X509_verify_cert, so this relaxes issuer *selection*, never trust. This
65 * also covers delegated grid proxies whose copied AKID points at the CA.
66 */
67 if (rv == X509_V_ERR_AKID_SKID_MISMATCH
68 || rv == X509_V_ERR_AKID_ISSUER_SERIAL_MISMATCH)
69 {
70 return 1;
71 }
72 return 0;
73}

75/*
76 * WHAT: verify callback used only in BRIX_CRL_MODE_TRY.
77 * WHY: "try" checks revocation where a CRL exists but tolerates a CA that has
78 * none; a stale (expired) CRL stays fatal (staleness is evidence).
79 * HOW: downgrade only X509_V_ERR_UNABLE_TO_GET_CRL to success; every other
80 * verdict (CRL_HAS_EXPIRED, CERT_REVOKED, ...) stands.
81 */
82static int
83brix_crl_try_verify_cb(int ok, X509_STORE_CTX *ctx)
84{
85 if (ok) {
86 return 1;
87 }
88 if (X509_STORE_CTX_get_error(ctx) == X509_V_ERR_UNABLE_TO_GET_CRL) {
89 return 1;
90 }
91 return 0;
92}

94int
95brix_store_configure(X509_STORE *store, const char *cadir,
96 unsigned long extra_flags, int crl_count,
97 brix_sp_mode_t sp_mode, int crl_mode,
98 void *log, brix_sp_log_fn log_fn)
99{
100 brix_sp_table_t *table;
101
102 if (store == NULL) {
103 return -1;
104 }
105
106 if (extra_flags != 0) {
107 X509_STORE_set_flags(store, extra_flags);
108 }
109 /* AKID is advisory: tolerate an AKID mismatch when selecting a name-matching
110 * issuer on every store (webdav and GSI); the signature is still verified. */
111 X509_STORE_set_check_issued(store, brix_sp_proxy_check_issued);
112
113 if (crl_mode == BRIX_CRL_MODE_REQUIRE
114 || (crl_mode == BRIX_CRL_MODE_TRY && crl_count > 0))
115 {
116 X509_STORE_set_flags(store, X509_V_FLAG_CRL_CHECK
117 | X509_V_FLAG_CRL_CHECK_ALL | X509_V_FLAG_USE_DELTAS);
118 }
119 if (crl_mode == BRIX_CRL_MODE_TRY) {
120 X509_STORE_set_verify_cb(store, brix_crl_try_verify_cb);
121 }
122
123 if (cadir == NULL && sp_mode == BRIX_SP_MODE_REQUIRE) {
124 sp_log(log, log_fn, BRIX_SP_LOG_WARN,
125 "signing_policy: \"require\" needs a hashed CA directory, not a "
126 "bundle file");
127 return -1;
128 }
129
130 table = brix_sp_table_build(cadir, log, log_fn);
131 if (table == NULL) {
132 return -1;
133 }
134 if (!brix_store_policy_attach(store, table, sp_mode, crl_mode)) {
135 brix_sp_table_free(table);
136 return -1;
137 }
138 return 0;
139}

172int
173brix_store_policy_attach(X509_STORE *store, brix_sp_table_t *table,
174 brix_sp_mode_t sp_mode, int crl_mode)
175{
176 brix_store_policy_t *sp;
177 int idx = sp_store_ex_index();
178
179 if (store == NULL || idx < 0) {
180 return 0;
181 }
182 sp = calloc(1, sizeof(*sp));
183 if (sp == NULL) {
184 return 0;
185 }
186 sp->table = table;
187 sp->sp_mode = sp_mode;
188 sp->crl_mode = crl_mode;
189
190 if (!X509_STORE_set_ex_data(store, idx, sp)) {
191 free(sp);
192 return 0;
193 }
194 return 1;
195}
