1/*
2 * backend_async_root.c — root:// park/resume adapter for the durable
3 * backend-async mutation queue (see fs/xfer/backend_async_queue.{c,h}).
4 *
5 * WHAT: Bridges a kXR_rm / kXR_rmdir request into the backend-async queue: on
6 * enqueue the connection is parked in XRD_ST_WAITING_BAQ (the recv loop
7 * yields on that state, half-duplex, so no further PDU is read and the
8 * request's streamid stays current); when the batch flushes, the waker
9 * (baq_root_done) sends the real reply on that streamid and resumes reads.
10 *
11 * WHY: The reply must be delivered LATE on the original streamid after the bulk
12 * backend flush. root:// is half-duplex per connection, so simply not
13 * replying parks the synchronous client in recv — exactly the "block until
14 * flush" contract — with no interim kXR_wait/waitresp needed.
15 *
16 * HOW: One park record per connection (ctx->baq_park), reused across sequential
17 * async mutations (only one can be in flight at a time). It is the queue's
18 * opaque `client` token and is read back by the waker. On teardown the
19 * disconnect funnel calls brix_baq_drop_client(ctx->baq_park) so the flush
20 * never wakes a freed connection; the durable journal record stays on disk
21 * and replays at the next flush or at boot.
22 *
23 * NOTE: The queued syscall runs in the worker's own credential context at drain
24 * time (the per-request impersonation broker state is not captured across
25 * the park). Authorization is still enforced synchronously by brix_auth_gate
26 * BEFORE the enqueue, so an unauthorized mutation is never queued; but an
27 * export that relies on per-user impersonation for the mutation ITSELF
28 * should not enable brix_backend_async.
29 */

43/* Per-connection park record: everything the waker needs to send the late reply
44 * on the original streamid (which ctx->recv.cur_streamid still holds — the recv
45 * loop reads no new PDU while WAITING_BAQ). */
46typedef struct {
47 brix_ctx_t *ctx;
48 ngx_connection_t *c;
49 ngx_uint_t op_id; /* BRIX_OP_RM / _RMDIR / _MV slot */
50 const char *verb; /* "RM" / "RMDIR" / "MV" (static) */
51 brix_baq_op_t baq_op; /* for RMDIR ENOENT→OK idempotency */
52 char resolved[PATH_MAX]; /* access-log path (src for MV) */
53 char detail[PATH_MAX]; /* access-log detail ("-" or MV dst) */
54} brix_baq_root_park_t;
55
56/*
57 * baq_root_done — the queue waker for a parked root:// mutation.
58 *
59 * Runs when the batch flushes (deferred posted event on the size trigger, or the
60 * coalesce timer on the time trigger). Sends the real result on the parked
61 * connection and re-arms its read loop. Mirrors op_table.c's exec_rmdir: a
62 * missing RMDIR target is success (stock do_Rmdir tolerates ENOENT).
63 */
64static void
65baq_root_done(void *client, int op_errno)
66{
67 brix_baq_root_park_t *park = client;
68 brix_ctx_t *ctx = park->ctx;
69 ngx_connection_t *c = park->c;
70
71 if (op_errno == ENOENT && park->baq_op == BRIX_BAQ_RMDIR) {
72 op_errno = 0; /* idempotent_missing (rmdir parity) */
73 }
74
75 /* Set REQ_HEADER before sending: brix_send_* overrides to XRD_ST_SENDING on a
76 * partial write, and the recv handoff gate yields on SENDING until it drains
77 * — the same ordering the CMS/FRM resume paths use. */
78 ctx->state = XRD_ST_REQ_HEADER;
79
80 if (op_errno == 0) {
81 brix_log_access(ctx, c, park->verb, park->resolved, park->detail,
82 1, kXR_ok, NULL, 0);
83 BRIX_OP_OK(ctx, park->op_id);
84 (void) brix_send_ok(ctx, c, NULL, 0);
85 } else {
86 uint16_t code = brix_kxr_from_errno(op_errno);
87 const char *msg = brix_kxr_err_string(op_errno);
88
89 brix_log_access(ctx, c, park->verb, park->resolved, park->detail,
90 0, code, msg, 0);
91 BRIX_OP_ERR(ctx, park->op_id);
92 (void) brix_send_error(ctx, c, code, msg);
93 }
94
95 /* Leave ctx->baq_park pointing at this record — it is reused for the next
96 * async mutation on this connection (brix_baq_drop_client is a no-op when
97 * nothing is queued, so an unrelated later disconnect stays safe). */
98 (void) brix_schedule_read_resume(c);
99}

101int
102brix_root_backend_async_try(brix_ctx_t *ctx, ngx_connection_t *c,
103 ngx_stream_brix_srv_conf_t *conf, const brix_op_desc_t *d,
104 const char *resolved)
105{
106 brix_baq_root_park_t *park;
107 brix_baq_op_t baq_op;
108
109 if (!conf->backend_async) {
110 return 0; /* feature off — run inline */
111 }
112
113 /* Only namespace-removal mutations are queueable today. Anything else (CHMOD,
114 * MKDIR-via-other-paths, …) runs synchronously. */
115 switch (d->opcode) {
116 case kXR_rm: baq_op = BRIX_BAQ_UNLINK; break;
117 case kXR_rmdir: baq_op = BRIX_BAQ_RMDIR; break;
118 default: return 0;
119 }
120
121 /* One reused park record per connection (only one mutation is ever parked at
122 * a time — WAITING_BAQ yields the recv loop, so a second is never read). */
123 park = ctx->baq_park;
124 if (park == NULL) {
125 park = ngx_pcalloc(c->pool, sizeof(*park));
126 if (park == NULL) {
127 return 0; /* OOM — fall back to inline */
128 }
129 ctx->baq_park = park;
130 }
131 park->ctx = ctx;
132 park->c = c;
133 park->op_id = d->op_id;
134 park->verb = d->name;
135 park->baq_op = baq_op;
136 (void) snprintf(park->resolved, sizeof(park->resolved), "%s", resolved);
137 park->detail[0] = '-';
138 park->detail[1] = '\0';
139
140 /* Park BEFORE enqueue so any teardown that races the enqueue sees a consistent
141 * WAITING_BAQ state. The size-triggered drain is a deferred posted event, so
142 * the waker cannot fire before this call returns. */
143 ctx->state = XRD_ST_WAITING_BAQ;
144
145 if (brix_baq_enqueue(baq_op, conf->common.root_canon, resolved,
146 NULL /* no dst */, 0 /* mode */,
147 conf->backend_async_batch, conf->backend_async_wait,
148 baq_root_done, park) != NGX_OK)
149 {
150 ctx->state = XRD_ST_REQ_HEADER; /* enqueue refused — un-park, run inline */
151 return 0;
152 }
153
154 return 1; /* parked; caller returns NGX_OK */
155}

157int
158brix_root_backend_async_mv_try(brix_ctx_t *ctx, ngx_connection_t *c,
159 ngx_stream_brix_srv_conf_t *conf, const char *src_resolved,
160 const char *dst_resolved)
161{
162 brix_baq_root_park_t *park;
163
164 if (!conf->backend_async) {
165 return 0; /* feature off — run inline */
166 }
167
168 /* Reuse the per-connection park record (only one mutation is ever parked at
169 * a time — WAITING_BAQ yields the recv loop). The caller has already gated on
170 * a NON-EXISTENT destination, so this is a pure create: the queue's
171 * overwrite=0 rename can only succeed or fail source-side (ENOENT/ENOSPC/…),
172 * whose generic errno→kXR mapping matches mv_execute exactly. The dst-exists
173 * ladder (EEXIST/EISDIR/ENOTEMPTY) never reaches here — those stay inline. */
174 park = ctx->baq_park;
175 if (park == NULL) {
176 park = ngx_pcalloc(c->pool, sizeof(*park));
177 if (park == NULL) {
178 return 0; /* OOM — fall back to inline */
179 }
180 ctx->baq_park = park;
181 }
182 park->ctx = ctx;
183 park->c = c;
184 park->op_id = BRIX_OP_MV;
185 park->verb = "MV";
186 park->baq_op = BRIX_BAQ_RENAME;
187 (void) snprintf(park->resolved, sizeof(park->resolved), "%s", src_resolved);
188 (void) snprintf(park->detail, sizeof(park->detail), "%s", dst_resolved);
189
190 ctx->state = XRD_ST_WAITING_BAQ;
191
192 if (brix_baq_enqueue(BRIX_BAQ_RENAME, conf->common.root_canon, src_resolved,
193 dst_resolved, 0 /* mode */, conf->backend_async_batch,
194 conf->backend_async_wait, baq_root_done, park) != NGX_OK)
195 {
196 ctx->state = XRD_ST_REQ_HEADER; /* enqueue refused — un-park, run inline */
197 return 0;
198 }
199
200 return 1; /* parked; caller returns NGX_OK */
