1/*
2 * dashboard/files.c — admin file browser + downloader for the monitoring UI.
3 *
4 * Two admin-auth-only endpoints, both confined to brix_dashboard_browse_root
5 * via the kernel openat2(RESOLVE_BENEATH) primitive (brix_open_beneath), so a
6 * client-supplied ?path= can never escape the configured root:
7 *
8 * GET /xrootd/api/v1/files?path=<rel> JSON listing of a directory:
9 * each entry's name, type, size, local owner (uid -> username), and
10 * modification + creation (statx btime) timestamps.
11 * GET /xrootd/api/v1/download?path=<rel> stream one regular file as an
12 * application/octet-stream attachment.
13 *
14 * SECURITY:
15 * - Always admin-auth (ngx_http_brix_dashboard_check_auth) — never the
16 * anonymous read tier, mirroring the config-download endpoint.
17 * - Disabled unless brix_dashboard_browse_root is configured (endpoints 404).
18 * - Confinement is the kernel's RESOLVE_BENEATH; per-entry statx uses
19 * AT_SYMLINK_NOFOLLOW so a symlink is reported, never followed.
20 */

39/* Map a confined-open failure to an HTTP status. A RESOLVE_BENEATH escape
40 * attempt surfaces as EXDEV/ELOOP — report it as 403 (forbidden), not 500. */
41static ngx_int_t
42dashboard_open_status(int e)
43{
44 if (e == ENOENT || e == ENOTDIR) {
45 return NGX_HTTP_NOT_FOUND;
46 }
47 if (e == EACCES || e == EPERM || e == EXDEV || e == ELOOP) {
48 return NGX_HTTP_FORBIDDEN;
49 }
50 return NGX_HTTP_INTERNAL_SERVER_ERROR;
51}

54/* Resolve uid -> local username; falls back to the decimal uid on lookup miss.
55 * A 1-deep cache short-circuits the common "whole dir owned by one user" case. */
56static void
57dashboard_owner_name(uid_t uid, char *out, size_t outsz)
58{
59 static uid_t s_cached_uid;
60 static char s_cached_name[64];
61 static int s_have_cache;
62
63 struct passwd pw;
64 struct passwd *res = NULL;
65 char buf[1024];
66
67 if (s_have_cache && s_cached_uid == uid) {
68 ngx_snprintf((u_char *) out, outsz, "%s%Z", s_cached_name);
69 return;
70 }
71 if (getpwuid_r(uid, &pw, buf, sizeof(buf), &res) == 0 && res != NULL
72 && res->pw_name != NULL)
73 {
74 ngx_snprintf((u_char *) out, outsz, "%s%Z", res->pw_name);
75 ngx_snprintf((u_char *) s_cached_name, sizeof(s_cached_name), "%s%Z",
76 res->pw_name);
77 s_cached_uid = uid;
78 s_have_cache = 1;
79 return;
80 }
81 ngx_snprintf((u_char *) out, outsz, "%ud%Z", (ngx_uint_t) uid);
82}

84/*
85 * Extract + URL-decode the ?path= query argument into out[] as a path RELATIVE
86 * to the browse root (leading '/' stripped; empty/"/" => "."). Rejects an
87 * embedded NUL (decode artefact / smuggling). Returns NGX_OK / NGX_ERROR.
88 * The kernel RESOLVE_BENEATH at open time is what actually enforces confinement;
89 * this only normalises the input.
90 */
91static ngx_int_t
92dashboard_files_get_path(ngx_http_request_t *r, char *out, size_t outsz)
93{
94 ngx_str_t raw;
95 u_char *dst, *src;
96 size_t n;
97
98 if (ngx_http_arg(r, (u_char *) "path", 4, &raw) != NGX_OK
99 || raw.len == 0)
100 {
101 ngx_memcpy(out, ".", 2);
102 return NGX_OK;
103 }
104 if (raw.len >= outsz) {
105 return NGX_ERROR;
106 }
107
108 /* URL-decode in place into out (decoded length <= encoded length). */
109 dst = (u_char *) out;
110 src = raw.data;
111 ngx_unescape_uri(&dst, &src, raw.len, 0);
112 n = (size_t) (dst - (u_char *) out);
113 out[n] = '\0';
114
115 if (ngx_strlchr((u_char *) out, (u_char *) out + n, '\0') != NULL) {
116 return NGX_ERROR; /* embedded NUL */
117 }
118
119 /* Strip a leading '/' so the value is relative to the browse root; map the
120 * root itself to "." (what openat2 wants for the anchor dir). */
121 {
122 char *p = out;
123 while (*p == '/') {
124 p++;
125 }
126 if (*p == '\0') {
127 ngx_memcpy(out, ".", 2);
128 } else if (p != out) {
129 ngx_memmove(out, p, strlen(p) + 1);
130 }
131 }
132 return NGX_OK;
133}

136/* Build one entry object from a statx result. Returns a json_t (caller owns) or
137 * NULL on OOM. */
138static json_t *
139dashboard_files_entry(const char *name, const struct statx *stx)
140{
141 json_t *o = json_object();
142 char owner[64];
143
144 if (o == NULL) { return NULL; }
145 dashboard_owner_name((uid_t) stx->stx_uid, owner, sizeof(owner));
146
147 json_object_set_new(o, "name", json_string(name));
148 json_object_set_new(o, "type",
149 S_ISDIR(stx->stx_mode) ? json_string("dir")
150 : S_ISREG(stx->stx_mode) ? json_string("file")
151 : json_string("other"));
152 json_object_set_new(o, "size", json_integer((json_int_t) stx->stx_size));
153 json_object_set_new(o, "owner", json_string(owner));
154 json_object_set_new(o, "uid", json_integer((json_int_t) stx->stx_uid));
155 json_object_set_new(o, "mtime",
156 json_integer((json_int_t) stx->stx_mtime.tv_sec));
157 /* Creation time (birth) when the filesystem supports it; 0 => client falls
158 * back to mtime. */
159 json_object_set_new(o, "btime",
160 json_integer((json_int_t)
161 ((stx->stx_mask & STATX_BTIME) ? stx->stx_btime.tv_sec : 0)));
162 return o;
163}

165/* GET /xrootd/api/v1/files?path=<rel> */
166ngx_int_t
167ngx_http_brix_dashboard_files_handler(ngx_http_request_t *r)
168{
169 ngx_http_brix_dashboard_loc_conf_t *conf;
170 char relpath[PATH_MAX];
171 int rootfd, dirfd;
172 DIR *dp;
173 struct dirent *de;
174 json_t *root, *arr;
175 ngx_int_t rc;
176 ngx_uint_t count = 0;
177 int truncated = 0;
178
179 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
180 if (conf->browse_root_canon[0] == '\0') {
181 return NGX_HTTP_NOT_FOUND; /* feature disabled */
182 }
183 rc = ngx_http_brix_dashboard_check_auth(r, conf, 0);
184 if (rc != NGX_OK) { return rc; }
185 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
186 return NGX_HTTP_NOT_ALLOWED;
187 }
188 brix_http_source_offer(r);
189
190 if (dashboard_files_get_path(r, relpath, sizeof(relpath)) != NGX_OK) {
191 return NGX_HTTP_BAD_REQUEST;
192 }
193
194 rootfd = brix_beneath_open_root(conf->browse_root_canon);
195 if (rootfd < 0) {
196 return NGX_HTTP_INTERNAL_SERVER_ERROR;
197 }
198 dirfd = brix_open_beneath(rootfd, relpath,
199 O_RDONLY | O_DIRECTORY | O_CLOEXEC, 0);
200 close(rootfd);
201 if (dirfd < 0) {
202 return dashboard_open_status(ngx_errno);
203 }
204 dp = fdopendir(dirfd);
205 if (dp == NULL) {
206 close(dirfd);
207 return NGX_HTTP_INTERNAL_SERVER_ERROR;
208 }
209
210 root = json_object();
211 arr = json_array();
212 if (root == NULL || arr == NULL) {
213 if (root) { json_decref(root); }
214 if (arr) { json_decref(arr); }
215 closedir(dp);
216 return NGX_HTTP_INTERNAL_SERVER_ERROR;
217 }
218
219 while ((de = readdir(dp)) != NULL) {
220 struct statx stx;
221 json_t *e;
222
223 if (de->d_name[0] == '.'
224 && (de->d_name[1] == '\0'
225 || (de->d_name[1] == '.' && de->d_name[2] == '\0')))
226 {
227 continue; /* skip "." and ".." (UI navigates via breadcrumb) */
228 }
229 if (count >= DASHBOARD_FILES_MAX_ENTRIES) {
230 truncated = 1;
231 break;
232 }
233 if (statx(dirfd, de->d_name, AT_SYMLINK_NOFOLLOW,
234 STATX_TYPE | STATX_SIZE | STATX_UID | STATX_MTIME
235 | STATX_BTIME, &stx) != 0)
236 {
237 continue; /* vanished mid-scan / unreadable — skip */
238 }
239 e = dashboard_files_entry(de->d_name, &stx);
240 if (e == NULL || json_array_append_new(arr, e) != 0) {
241 continue;
242 }
243 count++;
244 }
245 closedir(dp); /* also closes dirfd */
246
247 /* Re-derive the normalised path for echo (strip the "." root sentinel). */
248 dashboard_json_set_schema(root);
249 json_object_set_new(root, "path",
250 json_string((relpath[0] == '.' && relpath[1] == '\0') ? "/" : relpath));
251 json_object_set_new(root, "truncated", truncated ? json_true() : json_false());
252 json_object_set_new(root, "entries", arr);
253
254 return dashboard_json_send(r, NGX_HTTP_OK, root);
255}

257/* GET /xrootd/api/v1/download?path=<rel> */
258/* Build a safe Content-Disposition filename from the basename of relpath:
259 * strip the directory, then any byte that could break the header (quote,
260 * control, backslash) becomes '_'. */
261static void
262dashboard_download_filename(const char *relpath, char *out, size_t outsz)
263{
264 const char *base = strrchr(relpath, '/');
265 size_t i;
266
267 base = (base != NULL) ? base + 1 : relpath;
268 if (base[0] == '\0') {
269 base = "download";
270 }
271 for (i = 0; base[i] != '\0' && i + 1 < outsz; i++) {
272 unsigned char c = (unsigned char) base[i];
273 out[i] = (c < 0x20 || c == '"' || c == '\\' || c == 0x7f) ? '_'
274 : (char) c;
275 }
276 out[i] = '\0';
277}

279ngx_int_t
280ngx_http_brix_dashboard_download_handler(ngx_http_request_t *r)
281{
282 ngx_http_brix_dashboard_loc_conf_t *conf;
283 char relpath[PATH_MAX];
284 char fname[256];
285 char disp[320];
286 int rootfd, fd;
287 struct stat sb;
288 ngx_int_t rc;
289 ngx_table_elt_t *h;
290 int n;
291
292 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
293 if (conf->browse_root_canon[0] == '\0') {
294 return NGX_HTTP_NOT_FOUND;
295 }
296 rc = ngx_http_brix_dashboard_check_auth(r, conf, 0);
297 if (rc != NGX_OK) { return rc; }
298 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
299 return NGX_HTTP_NOT_ALLOWED;
300 }
301 brix_http_source_offer(r);
302
303 if (dashboard_files_get_path(r, relpath, sizeof(relpath)) != NGX_OK
304 || (relpath[0] == '.' && relpath[1] == '\0'))
305 {
306 return NGX_HTTP_BAD_REQUEST; /* the root itself is not a download */
307 }
308
309 rootfd = brix_beneath_open_root(conf->browse_root_canon);
310 if (rootfd < 0) {
311 return NGX_HTTP_INTERNAL_SERVER_ERROR;
312 }
313 fd = brix_open_beneath(rootfd, relpath,
314 O_RDONLY | O_NOFOLLOW | O_CLOEXEC, 0);
315 close(rootfd);
316 if (fd < 0) {
317 return dashboard_open_status(ngx_errno);
318 }
319 if (fstat(fd, &sb) != 0) {
320 close(fd);
321 return NGX_HTTP_INTERNAL_SERVER_ERROR;
322 }
323 if (!S_ISREG(sb.st_mode)) {
324 close(fd);
325 return NGX_HTTP_NOT_FOUND; /* only regular files are downloadable */
326 }
327
328 dashboard_download_filename(relpath, fname, sizeof(fname));
329
330 r->headers_out.status = NGX_HTTP_OK;
331 r->headers_out.content_length_n = (off_t) sb.st_size;
332 r->headers_out.content_type =
333 (ngx_str_t) ngx_string("application/octet-stream");
334 r->headers_out.content_type_len = r->headers_out.content_type.len;
335 r->headers_out.last_modified_time = sb.st_mtime;
336
337 n = ngx_snprintf((u_char *) disp, sizeof(disp),
338 "attachment; filename=\"%s\"", fname) - (u_char *) disp;
339 h = ngx_list_push(&r->headers_out.headers);
340 if (h != NULL) {
341 h->hash = 1;
342 ngx_str_set(&h->key, "Content-Disposition");
343 h->value.len = (size_t) n;
344 h->value.data = ngx_pnalloc(r->pool, (size_t) n);
345 if (h->value.data == NULL) { close(fd); return NGX_HTTP_INTERNAL_SERVER_ERROR; }
346 ngx_memcpy(h->value.data, disp, (size_t) n);
347 }
348 h = ngx_list_push(&r->headers_out.headers);
349 if (h != NULL) {
350 h->hash = 1;
351 ngx_str_set(&h->key, "Cache-Control");
352 ngx_str_set(&h->value, "no-store");
353 }
354
355 if (r->method == NGX_HTTP_HEAD || sb.st_size == 0) {
356 rc = ngx_http_send_header(r);
357 close(fd);
358 if (rc == NGX_ERROR || rc > NGX_OK) { return rc; }
359 return ngx_http_send_special(r, NGX_HTTP_LAST);
360 }
361
362 /* Stream the file (sendfile on cleartext; nginx reads it for TLS). The
363 * helper sends headers, builds the file buf, and closes fd via pool cleanup. */
364 return brix_http_send_file_range(r, fd, relpath, 0, (off_t) sb.st_size, 1);
365}
