1/*
2 * dashboard/files.c — admin file browser + downloader for the monitoring UI.
3 *
4 * Two admin-auth-only endpoints, both confined to brix_dashboard_browse_root
5 * via the kernel openat2(RESOLVE_BENEATH) primitive (brix_open_beneath), so a
6 * client-supplied ?path= can never escape the configured root:
7 *
8 * GET /xrootd/api/v1/files?path=<rel> JSON listing of a directory:
9 * each entry's name, type, size, local owner (uid -> username), and
10 * modification + creation (statx btime) timestamps.
11 * GET /xrootd/api/v1/download?path=<rel> stream one regular file as an
12 * application/octet-stream attachment.
13 *
14 * SECURITY:
15 * - Always admin-auth (ngx_http_brix_dashboard_check_auth) — never the
16 * anonymous read tier, mirroring the config-download endpoint.
17 * - Disabled unless brix_dashboard_browse_root is configured (endpoints 404).
18 * - Confinement is the kernel's RESOLVE_BENEATH; per-entry statx uses
19 * AT_SYMLINK_NOFOLLOW so a symlink is reported, never followed.
20 */

39/* Map a confined-open failure to an HTTP status. A RESOLVE_BENEATH escape
40 * attempt surfaces as EXDEV/ELOOP — report it as 403 (forbidden), not 500. */
41static ngx_int_t
42dashboard_open_status(int e)
43{
44 if (e == ENOENT || e == ENOTDIR) {
45 return NGX_HTTP_NOT_FOUND;
46 }
47 if (e == EACCES || e == EPERM || e == EXDEV || e == ELOOP) {
48 return NGX_HTTP_FORBIDDEN;
49 }
50 return NGX_HTTP_INTERNAL_SERVER_ERROR;
51}

54/* Resolve uid -> local username; falls back to the decimal uid on lookup miss.
55 * A 1-deep cache short-circuits the common "whole dir owned by one user" case. */
56static void
57dashboard_owner_name(uid_t uid, char *out, size_t outsz)
58{
59 static uid_t s_cached_uid;
60 static char s_cached_name[64];
61 static int s_have_cache;
62
63 struct passwd pw;
64 struct passwd *res = NULL;
65 char buf[1024];
66
67 if (s_have_cache && s_cached_uid == uid) {
68 ngx_snprintf((u_char *) out, outsz, "%s%Z", s_cached_name);
69 return;
70 }
71 if (getpwuid_r(uid, &pw, buf, sizeof(buf), &res) == 0 && res != NULL
72 && res->pw_name != NULL)
73 {
74 ngx_snprintf((u_char *) out, outsz, "%s%Z", res->pw_name);
75 ngx_snprintf((u_char *) s_cached_name, sizeof(s_cached_name), "%s%Z",
76 res->pw_name);
77 s_cached_uid = uid;
78 s_have_cache = 1;
79 return;
80 }
81 ngx_snprintf((u_char *) out, outsz, "%ud%Z", (ngx_uint_t) uid);
82}

84/*
85 * Extract + URL-decode the ?path= query argument into out[] as a path RELATIVE
86 * to the browse root (leading '/' stripped; empty/"/" => "."). Rejects an
87 * embedded NUL (decode artefact / smuggling). Returns NGX_OK / NGX_ERROR.
88 * The kernel RESOLVE_BENEATH at open time is what actually enforces confinement;
89 * this only normalises the input.
90 */
91static ngx_int_t
92dashboard_files_get_path(ngx_http_request_t *r, char *out, size_t outsz)
93{
94 ngx_str_t raw;
95 u_char *dst, *src;
96 size_t n;
97
98 if (ngx_http_arg(r, (u_char *) "path", 4, &raw) != NGX_OK
99 || raw.len == 0)
100 {
101 ngx_memcpy(out, ".", 2);
102 return NGX_OK;
103 }
104 if (raw.len >= outsz) {
105 return NGX_ERROR;
106 }
107
108 /* URL-decode in place into out (decoded length <= encoded length). */
109 dst = (u_char *) out;
110 src = raw.data;
111 ngx_unescape_uri(&dst, &src, raw.len, 0);
112 n = (size_t) (dst - (u_char *) out);
113 out[n] = '\0';
114
115 if (ngx_strlchr((u_char *) out, (u_char *) out + n, '\0') != NULL) {
116 return NGX_ERROR; /* embedded NUL */
117 }
118
119 /* Strip a leading '/' so the value is relative to the browse root; map the
120 * root itself to "." (what openat2 wants for the anchor dir). */
121 {
122 char *p = out;
123 while (*p == '/') {
124 p++;
125 }
126 if (*p == '\0') {
127 ngx_memcpy(out, ".", 2);
128 } else if (p != out) {
129 ngx_memmove(out, p, strlen(p) + 1);
130 }
131 }
132 return NGX_OK;
133}

136/* Build one entry object from a statx result. Returns a json_t (caller owns) or
137 * NULL on OOM. */
138static json_t *
139dashboard_files_entry(const char *name, const struct statx *stx)
140{
141 json_t *o = json_object();
142 char owner[64];
143
144 if (o == NULL) { return NULL; }
145 dashboard_owner_name((uid_t) stx->stx_uid, owner, sizeof(owner));
146
147 json_object_set_new(o, "name", json_string(name));
148 json_object_set_new(o, "type",
149 S_ISDIR(stx->stx_mode) ? json_string("dir")
150 : S_ISREG(stx->stx_mode) ? json_string("file")
151 : json_string("other"));
152 json_object_set_new(o, "size", json_integer((json_int_t) stx->stx_size));
153 json_object_set_new(o, "owner", json_string(owner));
154 json_object_set_new(o, "uid", json_integer((json_int_t) stx->stx_uid));
155 json_object_set_new(o, "mtime",
156 json_integer((json_int_t) stx->stx_mtime.tv_sec));
157 /* Creation time (birth) when the filesystem supports it; 0 => client falls
158 * back to mtime. */
159 json_object_set_new(o, "btime",
160 json_integer((json_int_t)
161 ((stx->stx_mask & STATX_BTIME) ? stx->stx_btime.tv_sec : 0)));
162 return o;
163}

165/*
166 * WHAT: Run the admin-auth + method + feature-enabled preamble common to both
167 * file endpoints, then extract the confined relative ?path= into out[].
168 * WHY: files_handler and download_handler share an identical gate; factoring it
169 * keeps each orchestrator flat and the security checks in one place.
170 * HOW: feature-disabled/auth/method map to their HTTP status exactly as before
171 * (auth returns its own rc); on success emit the AGPL source offer and
172 * normalise the path. Returns NGX_OK when the caller may proceed, else the
173 * HTTP status (or auth rc) to return verbatim.
174 */
175static ngx_int_t
176dashboard_files_gate(ngx_http_request_t *r,
177 ngx_http_brix_dashboard_loc_conf_t *conf,
178 char *out, size_t outsz)
179{
180 ngx_int_t rc;
181
182 if (conf->browse_root_canon[0] == '\0') {
183 return NGX_HTTP_NOT_FOUND; /* feature disabled */
184 }
185 rc = ngx_http_brix_dashboard_check_auth(r, conf, 0);
186 if (rc != NGX_OK) { return rc; }
187 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
188 return NGX_HTTP_NOT_ALLOWED;
189 }
190 brix_http_source_offer(r);
191
192 if (dashboard_files_get_path(r, out, outsz) != NGX_OK) {
193 return NGX_HTTP_BAD_REQUEST;
194 }
195 return NGX_OK;
196}

198/*
199 * WHAT: Open the confined directory named by relpath under the browse root.
200 * WHY: Isolates the two-fd RESOLVE_BENEATH open + fdopendir dance (with its
201 * own error mapping) from the listing orchestrator.
202 * HOW: open the root anchor, openat2-beneath the relpath as a directory, hand
203 * the fd to fdopendir. On success *out_dp is set and *out_dirfd holds the
204 * still-open descriptor (owned by the DIR*, closed by closedir). Returns
205 * NGX_OK or the HTTP status to return; *out_dp is NULL on any error.
206 */
207static ngx_int_t
208dashboard_files_opendir(ngx_http_brix_dashboard_loc_conf_t *conf,
209 const char *relpath, DIR **out_dp, int *out_dirfd)
210{
211 int rootfd, dirfd;
212
213 *out_dp = NULL;
214 *out_dirfd = -1;
215
216 rootfd = brix_beneath_open_root(conf->browse_root_canon);
217 if (rootfd < 0) {
218 return NGX_HTTP_INTERNAL_SERVER_ERROR;
219 }
220 dirfd = brix_open_beneath(rootfd, relpath,
221 O_RDONLY | O_DIRECTORY | O_CLOEXEC, 0);
222 close(rootfd);
223 if (dirfd < 0) {
224 return dashboard_open_status(ngx_errno);
225 }
226 *out_dp = fdopendir(dirfd);
227 if (*out_dp == NULL) {
228 close(dirfd);
229 return NGX_HTTP_INTERNAL_SERVER_ERROR;
230 }
231 *out_dirfd = dirfd;
232 return NGX_OK;
233}

235/*
236 * WHAT: Read every visible entry of an open directory into the JSON array arr,
237 * capped at DASHBOARD_FILES_MAX_ENTRIES.
238 * WHY: Pulls the readdir/statx/skip loop out of the orchestrator so the handler
239 * stays a flat sequence; entry emission stays byte-identical.
240 * HOW: skip "."/".."; per-entry AT_SYMLINK_NOFOLLOW statx (a symlink is
241 * reported, never followed); vanished/unreadable entries are skipped;
242 * on hitting the cap set *truncated and stop. dirfd is the statx anchor.
243 */
244static void
245dashboard_files_scan(DIR *dp, int dirfd, json_t *arr, int *truncated)
246{
247 struct dirent *de;
248 ngx_uint_t count = 0;
249
250 while ((de = readdir(dp)) != NULL) {
251 struct statx stx;
252 json_t *e;
253
254 if (de->d_name[0] == '.'
255 && (de->d_name[1] == '\0'
256 || (de->d_name[1] == '.' && de->d_name[2] == '\0')))
257 {
258 continue; /* skip "." and ".." (UI navigates via breadcrumb) */
259 }
260 if (count >= DASHBOARD_FILES_MAX_ENTRIES) {
261 *truncated = 1;
262 break;
263 }
264 if (statx(dirfd, de->d_name, AT_SYMLINK_NOFOLLOW,
265 STATX_TYPE | STATX_SIZE | STATX_UID | STATX_MTIME
266 | STATX_BTIME, &stx) != 0)
267 {
268 continue; /* vanished mid-scan / unreadable — skip */
269 }
270 e = dashboard_files_entry(de->d_name, &stx);
271 if (e == NULL || json_array_append_new(arr, e) != 0) {
272 continue;
273 }
274 count++;
275 }
276}

278/*
279 * WHAT: Build the JSON listing document (schema + echoed path + truncated flag +
280 * entries array) and send it as the 200 response.
281 * WHY: Separates document assembly + send from the scan, and localises the two
282 * json_object()/json_array() OOM guards.
283 * HOW: allocate root+arr (503 on OOM); scan the directory into arr; echo the
284 * normalised path (root sentinel "." -> "/"); send. closedir(dp) also
285 * closes the underlying dirfd. Returns the HTTP result of the send.
286 */
287static ngx_int_t
288dashboard_files_render(ngx_http_request_t *r, DIR *dp, int dirfd,
289 const char *relpath)
290{
291 json_t *root, *arr;
292 int truncated = 0;
293
294 root = json_object();
295 arr = json_array();
296 if (root == NULL || arr == NULL) {
297 if (root) { json_decref(root); }
298 if (arr) { json_decref(arr); }
299 closedir(dp);
300 return NGX_HTTP_INTERNAL_SERVER_ERROR;
301 }
302
303 dashboard_files_scan(dp, dirfd, arr, &truncated);
304 closedir(dp); /* also closes dirfd */
305
306 /* Re-derive the normalised path for echo (strip the "." root sentinel). */
307 dashboard_json_set_schema(root);
308 json_object_set_new(root, "path",
309 json_string((relpath[0] == '.' && relpath[1] == '\0') ? "/" : relpath));
310 json_object_set_new(root, "truncated", truncated ? json_true() : json_false());
311 json_object_set_new(root, "entries", arr);
312
313 return dashboard_json_send(r, NGX_HTTP_OK, root);
314}

336/* GET /xrootd/api/v1/download?path=<rel> */
337/* Build a safe Content-Disposition filename from the basename of relpath:
338 * strip the directory, then any byte that could break the header (quote,
339 * control, backslash) becomes '_'. */
340static void
341dashboard_download_filename(const char *relpath, char *out, size_t outsz)
342{
343 const char *base = strrchr(relpath, '/');
344 size_t i;
345
346 base = (base != NULL) ? base + 1 : relpath;
347 if (base[0] == '\0') {
348 base = "download";
349 }
350 for (i = 0; base[i] != '\0' && i + 1 < outsz; i++) {
351 unsigned char c = (unsigned char) base[i];
352 out[i] = (c < 0x20 || c == '"' || c == '\\' || c == 0x7f) ? '_'
353 : (char) c;
354 }
355 out[i] = '\0';
356}

358/*
359 * WHAT: Confined-open the regular file named by relpath and fstat it into *sb.
360 * WHY: Isolates the RESOLVE_BENEATH open + type check (with error mapping) so
361 * the download orchestrator stays a flat sequence.
362 * HOW: open beneath the root with O_NOFOLLOW (a symlink is never followed);
363 * fstat; reject non-regular as 404. On success *out_fd holds the open fd
364 * (caller owns it); on any error *out_fd is -1 and the fd is closed here.
365 * Returns NGX_OK or the HTTP status to return verbatim.
366 */
367static ngx_int_t
368dashboard_download_open(ngx_http_brix_dashboard_loc_conf_t *conf,
369 const char *relpath, int *out_fd, struct stat *sb)
370{
371 int rootfd, fd;
372
373 *out_fd = -1;
374
375 rootfd = brix_beneath_open_root(conf->browse_root_canon);
376 if (rootfd < 0) {
377 return NGX_HTTP_INTERNAL_SERVER_ERROR;
378 }
379 fd = brix_open_beneath(rootfd, relpath,
380 O_RDONLY | O_NOFOLLOW | O_CLOEXEC, 0);
381 close(rootfd);
382 if (fd < 0) {
383 return dashboard_open_status(ngx_errno);
384 }
385 if (fstat(fd, sb) != 0) {
386 close(fd);
387 return NGX_HTTP_INTERNAL_SERVER_ERROR;
388 }
389 if (!S_ISREG(sb->st_mode)) {
390 close(fd);
391 return NGX_HTTP_NOT_FOUND; /* only regular files are downloadable */
392 }
393 *out_fd = fd;
394 return NGX_OK;
395}

397/*
398 * WHAT: Set the octet-stream download response headers (status, length, type,
399 * last-modified, Content-Disposition attachment, Cache-Control no-store).
400 * WHY: Separates header assembly (pure header mutation) from the fd lifecycle
401 * and the send, keeping the orchestrator flat.
402 * HOW: derive a header-safe attachment filename from the basename; push the two
403 * extra headers. A Content-Disposition value OOM is the one failure and
404 * maps to 503; the caller then closes the fd. Returns NGX_OK or the HTTP
405 * status to return.
406 */
407static ngx_int_t
408dashboard_download_set_headers(ngx_http_request_t *r, const struct stat *sb,
409 const char *relpath)
410{
411 char fname[256];
412 char disp[320];
413 ngx_table_elt_t *h;
414 int n;
415
416 dashboard_download_filename(relpath, fname, sizeof(fname));
417
418 r->headers_out.status = NGX_HTTP_OK;
419 r->headers_out.content_length_n = (off_t) sb->st_size;
420 r->headers_out.content_type =
421 (ngx_str_t) ngx_string("application/octet-stream");
422 r->headers_out.content_type_len = r->headers_out.content_type.len;
423 r->headers_out.last_modified_time = sb->st_mtime;
424
425 n = ngx_snprintf((u_char *) disp, sizeof(disp),
426 "attachment; filename=\"%s\"", fname) - (u_char *) disp;
427 h = ngx_list_push(&r->headers_out.headers);
428 if (h != NULL) {
429 h->hash = 1;
430 ngx_str_set(&h->key, "Content-Disposition");
431 h->value.len = (size_t) n;
432 h->value.data = ngx_pnalloc(r->pool, (size_t) n);
433 if (h->value.data == NULL) { return NGX_HTTP_INTERNAL_SERVER_ERROR; }
434 ngx_memcpy(h->value.data, disp, (size_t) n);
435 }
436 h = ngx_list_push(&r->headers_out.headers);
437 if (h != NULL) {
438 h->hash = 1;
439 ngx_str_set(&h->key, "Cache-Control");
440 ngx_str_set(&h->value, "no-store");
441 }
442 return NGX_OK;
443}

445/*
446 * WHAT: Emit the download response body for an open regular file fd.
447 * WHY: Splits the HEAD/empty short-circuit from the streaming path so each is a
448 * single linear branch owning the fd's fate.
449 * HOW: for HEAD or a zero-length file, send headers + a LAST special and close
450 * fd here; otherwise hand fd to brix_http_send_file_range, which sends
451 * headers, builds the file buf, and closes fd via pool cleanup. Returns
452 * the HTTP result.
453 */
454static ngx_int_t
455dashboard_download_body(ngx_http_request_t *r, int fd, const char *relpath,
456 const struct stat *sb)
457{
458 ngx_int_t rc;
459
460 if (r->method == NGX_HTTP_HEAD || sb->st_size == 0) {
461 rc = ngx_http_send_header(r);
462 close(fd);
463 if (rc == NGX_ERROR || rc > NGX_OK) { return rc; }
464 return ngx_http_send_special(r, NGX_HTTP_LAST);
465 }
466
467 /* Stream the file (sendfile on cleartext; nginx reads it for TLS). The
468 * helper sends headers, builds the file buf, and closes fd via pool cleanup. */
469 return brix_http_send_file_range(r, fd, relpath, 0, (off_t) sb->st_size, 1);
470}
