33/* ---- Bring up the durable stage engine (phase-64 SP4) for this worker ----
34 *
35 * WHAT: Initialises the stage journal and, on worker 0 only, replays staged
36 * flushes interrupted by a crash and reaps orphaned direct-write temp files.
37 * No return value — the engine degrades to in-memory when the journal dir is
38 * unset and reconcile/reap are best-effort recovery.
39 *
40 * WHY: A crash can strand a staged FLUSH mid-flight and leave partial
41 * "<final>.xrd-tmp.<dead-pid>.*" files in the export tree; without this the
42 * write never reaches the backend and the partials leak disk forever.
43 *
44 * HOW:
45 * 1. Init the stage engine with the opt-in $BRIX_STAGE_JOURNAL_DIR
46 * (unset = in-memory, no recovery).
47 * 2. On worker 0: replay any journalled staged FLUSH (brix_stage_reconcile).
48 * 3. On worker 0: reap orphaned non-staged direct-write temporaries — the
49 * broken write is discarded; the client retries (§11.3).
50 */
51static void
52brix_init_stage_engine_worker(ngx_cycle_t *cycle)
53{
54 /* phase-64 SP4: durable stage journal + restart reconcile. The journal dir is
55 * opt-in via $BRIX_STAGE_JOURNAL_DIR (unset = in-memory, no recovery). On a
56 * restart worker 0 replays any staged FLUSH left in flight by a crash so the
57 * write reaches the backend (only staged writes are recoverable - a non-staged
58 * direct write's partial is reaped, not replayed; §11.3). */
59 brix_stage_engine_init(getenv("BRIX_STAGE_JOURNAL_DIR"));
60 if (ngx_worker == 0) {
61 brix_stage_reconcile(NULL);
62 /* Clean up the OTHER half: a NON-staged direct write interrupted by the
63 * crash left an orphan "<final>.xrd-tmp.<dead-pid>.*" in the export tree -
64 * reap it (the broken write is discarded; the client retries). */
65 (void) brix_tmp_reap_all(cycle->log);
66 }
67}

69/* ---- Warm the per-worker GSI ephemeral-DH key pool ----
70 *
71 * WHAT: Seeds the GSI DH key pool from the first GSI-enabled server block's
72 * configuration. No return value — a config with no GSI server is a no-op
73 * and pool warm-up failures degrade to on-demand keygen.
74 *
75 * WHY: Phase 33 — kXGC_certreq must never run keygen on the event thread
76 * under a concurrent handshake burst. Only a small seed is generated
77 * synchronously here; the pool fills to its configured size off the event
78 * thread (via the GSI server's thread pool) so worker startup is not blocked
79 * on the full warm-up.
80 *
81 * HOW:
82 * 1. Skip unless the server scan found a GSI/BOTH block.
83 * 2. Init the keypool with that block's thread pool, size, and seed count.
84 */
85static void
86brix_init_gsi_keypool(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *gsi_xcf)
87{
88 if (gsi_xcf == NULL) {
89 return;
90 }
91 brix_gsi_keypool_init(cycle, gsi_xcf->common.thread_pool,
92 gsi_xcf->gsi_keypool_size,
93 gsi_xcf->gsi_keypool_seed);
94}

96/* ---- Warn once when openat2(2) confinement is unavailable ----
97 *
98 * WHAT: Emits a persistent startup WARN when the running kernel lacks
99 * openat2(2). No return value — this is diagnostics only; the fallback path
100 * still works.
101 *
102 * WHY: F-05 — without openat2 (kernel < 5.6 or built without the headers),
103 * path confinement falls back to the O_NOFOLLOW segment-by-segment walk
104 * which has a narrow TOCTOU window; operators must know confinement is
105 * degraded so they can plan a kernel upgrade.
106 *
107 * HOW:
108 * 1. Probe runtime availability (brix_openat2_runtime_available).
109 * 2. Log the degraded-confinement warning when unavailable.
110 */
111static void
112brix_warn_openat2_unavailable(ngx_cycle_t *cycle)
113{
114 if (!brix_openat2_runtime_available()) {
115 ngx_log_error(NGX_LOG_WARN, cycle->log, 0,
116 "brix: openat2(2) is not available on this system "
117 "(requires Linux kernel 5.6+). Path confinement falls "
118 "back to O_NOFOLLOW traversal, which has a TOCTOU race "
119 "window on multi-tenant storage. Upgrade the kernel for "
120 "strongest path confinement.");
121 }
122}

124/*
125 * Worker process init: start CRL reload timers for every server block that
126 * has brix_crl_reload configured. Timers are per-worker because each
127 * nginx worker process has its own event loop and its own copy of the config
128 * pointers (but the X509_STORE* is shared within a worker).
129 */
130/* ---- Per-worker init after fork: bring up all worker-scoped subsystems ----
131 *
132 * WHAT: Initialises crypto, the stage engine + scheduler, the proxy pool,
133 * io_uring, every enabled server block (via brix_init_one_server), the GSI
134 * keypool, the identity broker, and the worker-0 maintenance reapers.
135 * Returns NGX_OK / NGX_ERROR (NGX_ERROR aborts the worker).
136 *
137 * WHY: nginx workers are forked from the master and inherit config but not
138 * per-process resources (fds, timers, rings, key pools); everything a worker
139 * owns must be (re)built here, in the frozen order the reload-semantics
140 * contract documents.
141 *
142 * HOW:
143 * 1. Start the boot-cost phase timer and init OpenSSL crypto (fatal).
144 * 2. Bring up the stage engine and arm the per-worker stage-flush
145 * scheduler — BEFORE the stream-config early-return so HTTP-only
146 * (WebDAV/S3) workers get them too.
147 * 3. Early-return NGX_OK when there is no stream config.
148 * 4. Warn once when openat2(2) is unavailable (degraded confinement).
149 * 5. Init the proxy pool and this worker's io_uring ring (fatal under
150 * `brix_io_uring on`).
151 * 6. For each enabled server block: note GSI/manager presence, then run
152 * the per-server init ladder (fatal on error).
153 * 7. Warm the GSI keypool, arm the pending-locate reaper, connect the
154 * identity broker, arm the stage-out reaper.
155 * 8. Log the per-worker boot-cost breakdown and return NGX_OK.
156 */
157ngx_int_t
158ngx_stream_brix_init_process(ngx_cycle_t *cycle)
159{
160 ngx_stream_core_main_conf_t *cmcf;
161 ngx_stream_core_srv_conf_t **cscfp;
162 ngx_stream_brix_srv_conf_t *xcf;
163 ngx_uint_t i;
164 ngx_stream_brix_srv_conf_t *gsi_xcf = NULL; /* first GSI block: keypool cfg */
165 ngx_uint_t manager_seen = 0;
166 brix_phase_timer_t pt;
167 u_char ctx[64];
168
169 /* Permanent per-worker boot-cost breakdown (one NOTICE line at the end). */
170 brix_phase_timer_start(&pt);
171
172 if (!brix_crypto_init()) {
173 ngx_log_error(NGX_LOG_EMERG, cycle->log, 0,
174 "brix: failed to initialise OpenSSL crypto primitives");
175 return NGX_ERROR;
176 }
177
178 brix_init_stage_engine_worker(cycle);
179
180 /* arm the per-worker async stage-flush scheduler. Done BEFORE the stream-config
181 * early-return below so it runs in HTTP-only (WebDAV/S3) workers too - the
182 * deferred-flush queue is per-worker and protocol-agnostic; the tick is a no-op
183 * when the queue is empty. */
184 brix_init_stage_sched_timer(cycle);
185
186 /* Shed ALL worker capabilities + set NO_NEW_PRIVS in EVERY worker, BEFORE the
187 * stream-config early-return below, so HTTP-only (WebDAV/S3) workers are
188 * hardened too — not just stream/root:// workers or `map` mode. A worker never
189 * needs caps; a root-configured worker must not keep them (D-3 companion). */
190 brix_imp_worker_harden(cycle->log);
191
192 /* De-escalate a root-capable worker (root, or a non-root CAP_SETUID service
193 * account) down to the confined brix_worker_user (default nobody) BEFORE any
194 * backend init, broker connect, or the seccomp install below — so pre-auth
195 * credential parsing never runs as a root-capable identity, in EVERY mode and
196 * for HTTP-only (WebDAV/S3) workers too (they reach this before the early
197 * return below; their seccomp install runs later in the WebDAV init_process).
198 * Fail-closed: a worker that cannot reach a confined account refuses to run. */
199 if (brix_imp_worker_deescalate(cycle->log) != NGX_OK) {
200 return NGX_ERROR;
201 }
202
203 cmcf = ngx_stream_cycle_get_module_main_conf(cycle, ngx_stream_core_module);
204 if (cmcf == NULL) {
205 return NGX_OK;
206 }
207
208 brix_warn_openat2_unavailable(cycle);
209
210 cscfp = cmcf->servers.elts;
211
212 brix_proxy_pool_init();
213
214 /*
215 * Phase 44: bring up this worker's optional io_uring ring (after the proxy
216 * pool, same lifetime as every other per-worker async resource). A no-op
217 * unless a server block enabled it; under `brix_io_uring on` a bring-up
218 * failure returns NGX_ERROR so the worker refuses to run on the thread pool
219 * (§32.7 backstop). Under `auto` it degrades silently.
220 */
221 if (brix_uring_init_worker(cycle) != NGX_OK) {
222 return NGX_ERROR;
223 }
224 brix_phase_mark(&pt, "uring");
225
226 for (i = 0; i < cmcf->servers.nelts; i++) {
227 xcf = ngx_stream_conf_get_module_srv_conf(cscfp[i],
228 ngx_stream_brix_module);
229
230 if (!xcf->common.enable) {
231 continue;
232 }
233
234 if ((xcf->auth == BRIX_AUTH_GSI || xcf->auth == BRIX_AUTH_BOTH)
235 && gsi_xcf == NULL)
236 {
237 /* Phase 33: warm the GSI DH key pool below — keypool sizing +
238 * thread pool come from the first GSI block. */
239 gsi_xcf = xcf;
240 }
241
242 if (xcf->manager_mode) {
243 manager_seen = 1; /* A4: arm the pending-locate reaper below */
244 }
245
246 if (brix_init_one_server(cycle, xcf) != NGX_OK) {
247 return NGX_ERROR;
248 }
249 }
250
251 brix_phase_mark(&pt, "servers");
252
253 brix_init_gsi_keypool(cycle, gsi_xcf);
254 brix_phase_mark(&pt, "keypool");
255
256 brix_init_pending_reap_timer(cycle, manager_seen);
257
258 /* Phase 40: connect this worker to the identity broker (no-op unless
259 * brix_impersonation=map; lazily reconnects if the broker isn't up yet). */
260 brix_imp_init_worker(cycle);
261
262 brix_init_stage_reap_timer(cycle);
263
264 ngx_snprintf(ctx, sizeof(ctx) - 1, "xrootd init_process[w%ui]%Z", ngx_worker);
265 brix_phase_timer_log(&pt, cycle->log, (const char *) ctx);
266
267 /* D-3: install the seccomp syscall filter LAST, once every one-shot setup
268 * syscall above has run, so only the steady-state serving set must be on the
269 * allowlist. Idempotent + process-global (strictest across ALL brix servers,
270 * stream + http): for an HTTP-only config this init_process early-returned
271 * above and the install happens in the WebDAV init_process instead, so
272 * WebDAV/S3-only workers are filtered too. Fails the worker closed if an
273 * audit/enforce filter cannot be loaded — never serve unfiltered. */
274 if (brix_seccomp_install_once(cycle) != NGX_OK) {
275 return NGX_ERROR;
276 }
277
278 return NGX_OK;
279}

281/* Per-worker teardown at exit: release process-scoped resources. */
282void
283brix_exit_process(ngx_cycle_t *cycle)
284{
285 ngx_stream_core_main_conf_t *cmcf;
286 ngx_stream_core_srv_conf_t **cscfp;
287 ngx_stream_brix_srv_conf_t *xcf;
288 ngx_uint_t i;
289
290 /*
291 * Phase 27 W6: best-effort explicit LeakSanitizer check at worker exit.
292 * nginx's daemon exit path does not reliably reach LSan's own atexit hook,
293 * so a sanitizer build would otherwise never report leaks. Placed before
294 * any early return so it runs for http-only configs too. Compiled out
295 * entirely in normal builds — zero production effect.
296 *
297 * NOTE: whether this actually fires depends on nginx reaching exit_process
298 * at shutdown, which is platform/signal-dependent (it was observed NOT to
299 * run under WSL2 in dev). See docs/03-configuration/build-guide.md.
300 */
301#if defined(__SANITIZE_ADDRESS__)
302 __lsan_do_recoverable_leak_check();
303#endif
304
305 brix_sesslog_shutdown_flush();
306
307 /* Fast teardown: drop any idle pooled upstream connections so this draining
308 * worker releases authenticated upstream sockets immediately (with a clean
309 * FIN) rather than leaving process-exit to reap them. Idempotent and safe
310 * when proxy mode was never used (the pool is simply empty). */
311 brix_proxy_pool_shutdown();
312
313 /* Phase 44: tear down this worker's io_uring ring (no-op if never up). */
314 brix_uring_exit_worker(cycle);
315
316 cmcf = ngx_stream_cycle_get_module_main_conf(cycle, ngx_stream_core_module);
317 if (cmcf == NULL) {
318 return;
319 }
320
321 cscfp = cmcf->servers.elts;
322 for (i = 0; i < cmcf->servers.nelts; i++) {
323 xcf = ngx_stream_conf_get_module_srv_conf(cscfp[i],
324 ngx_stream_brix_module);
325 if (!xcf->common.enable) {
326 continue;
327 }
328 if (xcf->rootfd >= 0) {
329 close(xcf->rootfd);
330 xcf->rootfd = -1;
331 }
332 brix_cache_storage_cleanup(xcf);
333 }
334
335 brix_crypto_cleanup();
336}
