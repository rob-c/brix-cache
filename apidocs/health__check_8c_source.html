1/*
2 * health_check.c — Phase 22 active stream health checks (see health_check.h).
3 *
4 * The probe is a self-contained async XRootD client connection. Its wire
5 * framing mirrors the proven upstream bootstrap (src/upstream/bootstrap.c):
6 * the bootstrap write buffer is built by the shared
7 * brix_upstream_build_bootstrap(), and responses are read as uniform 8-byte
8 * ServerResponseHdr + dlen-byte body frames, exactly as the upstream read loop
9 * does. Unlike the upstream path it carries no client context — on bootstrap
10 * completion it sends kXR_ping (or kXR_stat "/") and reports the verdict to the
11 * registry rather than forwarding a saved client request.
12 *
13 * A single deadline timer (hc_timeout_ms) bounds the whole probe; a hung server
14 * that accepts the TCP connection but never answers is failed on timeout.
15 */

36/*
37 * Probe state machine, advanced one step per complete response frame received.
38 * Each phase consumes the server's reply to the request the previous phase (or
39 * the pipelined bootstrap) sent: HANDSHAKE -> PROTOCOL -> LOGIN -> PROBE. See
40 * brix_hc_dispatch() for the per-phase transitions and early-exit verdicts.
41 */
42typedef enum {
43 XRD_HC_HANDSHAKE = 0,
44 XRD_HC_PROTOCOL,
45 XRD_HC_LOGIN,
46 XRD_HC_PROBE,
47} brix_hc_phase_t;

123/* write side */
124/*
125 * Drain hc->wbuf to the socket without blocking. Returns NGX_OK once the whole
126 * buffer is sent (and read events re-armed so the reply can arrive), NGX_AGAIN
127 * if the socket is full (write event re-armed; brix_hc_write_handler resumes
128 * here), or NGX_ERROR on a fatal send error.
129 */
130static ngx_int_t
131brix_hc_flush(brix_hc_ctx_t *hc)
132{
133 ngx_connection_t *c = hc->conn;
134 ssize_t n;
135
136 /* Partial-write loop: send() may consume only part of the buffer, so track
137 * wbuf_pos and resume from there on the next writable event. */
138 while (hc->wbuf_pos < hc->wbuf_len) {
139 n = c->send(c, hc->wbuf + hc->wbuf_pos, hc->wbuf_len - hc->wbuf_pos);
140 if (n > 0) {
141 hc->wbuf_pos += (size_t) n;
142 continue;
143 }
144 if (n == NGX_AGAIN) {
145 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
146 return NGX_ERROR;
147 }
148 return NGX_AGAIN;
149 }
150 return NGX_ERROR;
151 }
152 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
153 return NGX_ERROR;
154 }
155 return NGX_OK;
156}

158/*
159 * Build and send the final liveness probe (kXR_ping or kXR_stat "/") after a
160 * successful login, transitioning to XRD_HC_PROBE. The reply's status is the
161 * pass/fail verdict, decided back in brix_hc_dispatch().
162 */
163static void
164brix_hc_send_probe(brix_hc_ctx_t *hc)
165{
166 if (hc->probe_type == BRIX_HC_TYPE_STAT) {
167 /* kXR_stat request followed by a 1-byte path body "/". Over-allocate
168 * by 1 so the path char sits immediately after the fixed header. */
169 size_t total = sizeof(ClientStatRequest) + 1; /* path "/" */
170 ClientStatRequest *r = ngx_palloc(hc->pool, total);
171 if (r == NULL) { brix_hc_finish(hc, 0); return; }
172 ngx_memzero(r, sizeof(*r));
173 /* streamid is a 2-byte client tag echoed in the reply; {0,2} marks this
174 * as a health request (see ping branch below). */
175 r->streamid[0] = 0;
176 r->streamid[1] = 2;
177 r->requestid = htons(kXR_stat); /* opcode, network order */
178 r->dlen = htonl((kXR_int32) 1); /* body length = 1 ("/") */
179 ((u_char *) r)[sizeof(*r)] = '/'; /* append path after header */
180 hc->wbuf = (u_char *) r;
181 hc->wbuf_len = total;
182 } else {
183 ClientPingRequest *r = ngx_palloc(hc->pool, sizeof(*r));
184 if (r == NULL) { brix_hc_finish(hc, 0); return; }
185 ngx_memzero(r, sizeof(*r));
186 r->streamid[0] = 0;
187 r->streamid[1] = 2; /* streamid 2 marks a health request */
188 r->requestid = htons(kXR_ping);
189 r->dlen = 0; /* ping carries no body */
190 hc->wbuf = (u_char *) r;
191 hc->wbuf_len = sizeof(*r);
192 }
193 /* Reset both write and read accumulators for this fresh request/response. */
194 hc->wbuf_pos = 0;
195 hc->rhdr_pos = 0;
196 hc->resp_dlen = 0;
197 hc->resp_body = NULL;
198 hc->resp_body_pos = 0;
199 hc->phase = XRD_HC_PROBE;
200
201 if (brix_hc_flush(hc) == NGX_ERROR) {
202 brix_hc_finish(hc, 0);
203 }
204}

206/*
207 * Writable-event handler. First writable event after a non-blocking connect()
208 * confirms (or rejects) the TCP connection via SO_ERROR; subsequent calls just
209 * resume a partially-flushed write buffer.
210 */
211static void
212brix_hc_write_handler(ngx_event_t *wev)
213{
214 ngx_connection_t *c = wev->data;
215 brix_hc_ctx_t *hc = c->data;
216 ngx_int_t rc;
217
218 if (wev->timedout) {
219 brix_hc_finish(hc, 0);
220 return;
221 }
222
223 /* Non-blocking connect completes asynchronously: the first writable event
224 * means connect() finished — SO_ERROR carries success (0) or the failure
225 * code, since connect() itself returned EINPROGRESS earlier. */
226 if (hc->connecting) {
227 int err = 0;
228 socklen_t len = sizeof(err);
229 if (getsockopt(c->fd, SOL_SOCKET, SO_ERROR, (char *) &err, &len) == -1
230 || err != 0)
231 {
232 BRIX_DIAG_WARN(hc->log, err, /* may be 0 */
233 "brix: health check failed to connect to %s:%d",
234 "the cluster member is down or unreachable from the manager",
235 "check that member's xrootd/data service and the network path "
236 "to it; it is marked DOWN and clients are routed elsewhere "
237 "until it passes again",
238 hc->host, (int) hc->port);
239 brix_hc_finish(hc, 0);
240 return;
241 }
242 hc->connecting = 0;
243 }
244
245 rc = brix_hc_flush(hc);
246 if (rc == NGX_ERROR) {
247 brix_hc_finish(hc, 0);
248 }
249}

252/* read side */
253/* Accumulate one full ServerResponseHdr + body frame; return NGX_OK when a
254 * complete frame is ready, NGX_AGAIN to wait for more, NGX_ERROR on close. */
255static ngx_int_t
256brix_hc_recv_frame(brix_hc_ctx_t *hc)
257{
258 ngx_connection_t *c = hc->conn;
259 ssize_t n;
260
261 /* Stage 1: fill the fixed 8-byte header, possibly across several reads
262 * (rhdr_pos tracks how much has arrived). Only once it is complete do we
263 * decode status/dlen and learn the body length. */
264 if (hc->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
265 size_t need = XRD_RESPONSE_HDR_LEN - hc->rhdr_pos;
266 n = c->recv(c, hc->rhdr + hc->rhdr_pos, need);
267 if (n == NGX_AGAIN) { return NGX_AGAIN; }
268 if (n <= 0) { return NGX_ERROR; } /* 0 = peer closed */
269 hc->rhdr_pos += (size_t) n;
270 if (hc->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
271 return NGX_AGAIN;
272 }
273 /* Header complete: decode the wire fields (big-endian on the wire). */
274 {
275 ServerResponseHdr *h = (ServerResponseHdr *) (void *) hc->rhdr;
276 hc->resp_status = ntohs(h->status);
277 hc->resp_dlen = ntohl(h->dlen);
278 }
279 if (hc->resp_dlen > 0) {
280 /* Bound attacker/garbage-controlled body length: a probe reply is
281 * tiny, so cap the allocation rather than trust the wire dlen. */
282 if (hc->resp_dlen > 4096) { /* bound probe response bodies */
283 return NGX_ERROR;
284 }
285 hc->resp_body = ngx_palloc(c->pool, hc->resp_dlen);
286 if (hc->resp_body == NULL) { return NGX_ERROR; }
287 hc->resp_body_pos = 0;
288 }
289 }
290
291 /* Stage 2: fill the dlen-byte body (again possibly across several reads).
292 * Skipped entirely when dlen == 0. */
293 if (hc->resp_body_pos < hc->resp_dlen) {
294 size_t need = hc->resp_dlen - hc->resp_body_pos;
295 n = c->recv(c, hc->resp_body + hc->resp_body_pos, need);
296 if (n == NGX_AGAIN) { return NGX_AGAIN; }
297 if (n <= 0) { return NGX_ERROR; }
298 hc->resp_body_pos += (size_t) n;
299 if (hc->resp_body_pos < hc->resp_dlen) {
300 return NGX_AGAIN;
301 }
302 }
303
304 return NGX_OK;
305}

308/*
309 * Handshake-done callback for a Step-F TLS-upgraded probe. Mirrors the
310 * upstream brix_upstream_tls_handshake_done(): verify the handshake (and the
311 * peer, explicitly), restore the probe's event handlers, and resend a fresh
312 * kXR_login over the TLS channel — the probe then continues LOGIN -> PROBE
313 * exactly as on cleartext. Any failure is a probe failure: a server that
314 * demands TLS but cannot complete it cannot serve TLS clients.
315 */
316static void
317brix_hc_tls_handshake_done(ngx_connection_t *c)
318{
319 brix_hc_ctx_t *hc = c->data;
320 ClientLoginRequest *req;
321
322 if (!c->ssl->handshaked) {
323 ngx_log_error(NGX_LOG_WARN, hc->log, 0,
324 "brix: health check: %s:%d TLS handshake failed",
325 hc->host, (int) hc->port);
326 brix_hc_finish(hc, 0);
327 return;
328 }
329 if (SSL_get_verify_result(c->ssl->connection) != X509_V_OK) {
330 ngx_log_error(NGX_LOG_WARN, hc->log, 0,
331 "brix: health check: %s:%d TLS peer verification failed",
332 hc->host, (int) hc->port);
333 brix_hc_finish(hc, 0);
334 return;
335 }
336
337 hc->tls = 1;
338

340 c->read->handler = brix_hc_read_handler;

357 hc->resp_dlen = 0;

359 hc->resp_body_pos = 0;
360

362 brix_hc_finish(hc, 0);
363 }

367/*
368 * XRD_HC_PROTOCOL step of brix_hc_dispatch(): consume the kXR_protocol reply
369 * and decide how the probe proceeds. Returns NGX_DONE when the step reached a
370 * verdict or issued its own follow-up send (the caller must return without
371 * touching hc — it may already be freed), or NGX_OK when the phase advanced to
372 * LOGIN and the caller should fall through to the accumulator reset.
373 */
374static ngx_int_t
375brix_hc_dispatch_protocol(brix_hc_ctx_t *hc)
376{
377 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return NGX_DONE; }
378
379 /* If the server demands TLS we don't probe deeper (no TLS in the
380 * probe path); it answered at the protocol level, so treat it as
381 * alive rather than blacklisting it.
382 *
383 * ServerResponseBody_Protocol layout: bytes 0-3 = pval (protocol
384 * version), bytes 4-7 = flags. Read the flags word at offset 4 and
385 * test the kXR_gotoTLS bit. Guard on dlen >= 8 so short/old replies
386 * that omit the flags simply fall through to the LOGIN phase. */
387 if (hc->resp_dlen >= 8) {
388 uint32_t flags_be;
389 ngx_memcpy(&flags_be, hc->resp_body + 4, sizeof(flags_be));
390 if (ntohl(flags_be) & kXR_gotoTLS) {
391#if (NGX_SSL)
392 /* Step F: with an outbound TLS ctx configured, probe DEEP —
393 * upgrade this connection and continue LOGIN -> PROBE over
394 * TLS, exactly as the upstream bootstrap does (the server
395 * discards the plaintext login pre-sent in the pipelined
396 * bootstrap). The probe deadline timer keeps running. */
397 if (hc->conf->upstream_tls
398 && hc->conf->upstream_tls_ctx != NULL)
399 {
400 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, hc->log, 0,
401 "brix: health check: %s:%d wants TLS; upgrading probe",
402 hc->host, (int) hc->port);
403 if (brix_outbound_start_tls(hc->conf->upstream_tls_ctx,
404 hc->conn, hc->host,
405 brix_hc_tls_handshake_done)
406 != NGX_OK)
407 {
408 brix_hc_finish(hc, 0);
409 }
410 return NGX_DONE;
411 }
412#endif
413 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, hc->log, 0,
414 "brix: health check: %s:%d wants TLS; "
415 "treating protocol-OK as alive", hc->host, (int) hc->port);
416 brix_hc_finish(hc, 1);
417 return NGX_DONE;
418 }
419 }
420 if (hc->tls_capable) {
421 /* No gotoTLS: send the deferred login now, in cleartext. (A
422 * TLS-capable probe never pipelined it — see brix_hc_start.) */
423 ClientLoginRequest *req = ngx_palloc(hc->pool, sizeof(*req));
424 if (req == NULL) { brix_hc_finish(hc, 0); return NGX_DONE; }
425 brix_upstream_build_login(req);
426 hc->wbuf = (u_char *) req;
427 hc->wbuf_len = sizeof(*req);
428 hc->wbuf_pos = 0;
429 hc->phase = XRD_HC_LOGIN;
430 hc->rhdr_pos = 0;
431 hc->resp_dlen = 0;
432 hc->resp_body = NULL;
433 hc->resp_body_pos = 0;
434 if (brix_hc_flush(hc) == NGX_ERROR) { brix_hc_finish(hc, 0); }
435 return NGX_DONE;
436 }
437 hc->phase = XRD_HC_LOGIN;
438 return NGX_OK;
439}

441/*
442 * Consume one complete response frame and advance the probe state machine.
443 * Several phases short-circuit to a verdict (alive/dead) without reaching the
444 * final PROBE phase — see the per-case comments. On a non-terminal transition
445 * it resets the accumulator and re-posts the read event so any already-buffered
446 * pipelined reply is processed in the same event cycle.
447 */
448
449static void
450brix_hc_dispatch(brix_hc_ctx_t *hc)
451{
452 switch (hc->phase) {
453
454 case XRD_HC_HANDSHAKE:
455 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return; }
456 hc->phase = XRD_HC_PROTOCOL;
457 break;
458
459 case XRD_HC_PROTOCOL:
460 if (brix_hc_dispatch_protocol(hc) == NGX_DONE) {
461 return;
462 }
463 break;
464
465 case XRD_HC_LOGIN:
466 /* authmore => server is alive but wants credentials; we don't carry
467 * any, so accept protocol liveness rather than fail. */
468 if (hc->resp_status == kXR_authmore) {
469 brix_hc_finish(hc, 1);
470 return;
471 }
472 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return; }
473 brix_hc_send_probe(hc); /* sets phase = PROBE, sends ping/stat */
474 return;
475
476 case XRD_HC_PROBE:
477 brix_hc_finish(hc, hc->resp_status == kXR_ok ? 1 : 0);
478 return;
479 }
480
481 /* Reset accumulator for the next frame and post a synthetic read so any
482 * pipelined bytes already in the socket buffer are processed this cycle. */
483 hc->rhdr_pos = 0;
484 hc->resp_dlen = 0;
485 hc->resp_body = NULL;
486 hc->resp_body_pos = 0;
487
488 if (ngx_handle_read_event(hc->conn->read, 0) != NGX_OK) {
489 brix_hc_finish(hc, 0);
490 return;
491 }
492 ngx_post_event(hc->conn->read, &ngx_posted_events);
493}

495/*
496 * Readable-event handler. Pulls exactly one complete frame per invocation and
497 * hands it to brix_hc_dispatch(), which re-arms (or posts) the read event for
498 * the next frame; this keeps each event-loop iteration bounded.
499 */
500static void
501brix_hc_read_handler(ngx_event_t *rev)
502{
503 ngx_connection_t *c = rev->data;
504 brix_hc_ctx_t *hc = c->data;
505 ngx_int_t rc;
506
507 if (rev->timedout) {
508 brix_hc_finish(hc, 0);
509 return;
510 }
511
512 /* Loop is structural only: every branch returns. It exists so a future
513 * "read another frame inline" change has an obvious place to continue. */
514 for ( ;; ) {
515 rc = brix_hc_recv_frame(hc);
516 if (rc == NGX_AGAIN) {
517 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
518 brix_hc_finish(hc, 0);
519 }
520 return;
521 }
522 if (rc == NGX_ERROR) {
523 brix_hc_finish(hc, 0);
524 return;
525 }
526 brix_hc_dispatch(hc);
527 return; /* dispatch re-arms / posts as needed */
528 }
529}

550/*
551 * Single termination point for a probe (called from every error/success path).
552 * Reports the verdict to the shared registry, bumps metrics, and tears down all
553 * probe state. MUST be the last thing any caller does with hc: it destroys the
554 * pool that backs hc itself, so hc is dangling on return.
555 */
556static void
557brix_hc_finish(brix_hc_ctx_t *hc, int passed)
558{
559 /* Stash the pool before any registry call: hc lives inside it and is freed
560 * by the ngx_destroy_pool() at the end. */
561 ngx_pool_t *pool = hc->pool;
562
563 /* Cancel the deadline timer and close the socket before reporting, so a
564 * verdict callback can never observe a half-torn-down probe. */
565 if (hc->tev.timer_set) {
566 ngx_del_timer(&hc->tev);
567 }
568 if (hc->conn != NULL) {
569 ngx_close_connection(hc->conn);
570 hc->conn = NULL;
571 }
572
573 if (passed) {
574 brix_srv_hc_pass(hc->host, hc->port);
575 BRIX_HC_METRIC_INC(hc_pass_total);
576 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, hc->log, 0,
577 "brix: health check: %s:%d passed",
578 hc->host, (int) hc->port);
579 } else {
580 if (brix_srv_hc_fail(hc->host, hc->port,
581 hc->threshold, hc->blacklist_ms))
582 {
583 BRIX_HC_METRIC_INC(hc_blacklist_total);
584 ngx_log_error(NGX_LOG_WARN, hc->log, 0,
585 "brix: health check: %s:%d blacklisted after"
586 " %ui consecutive failures",
587 hc->host, (int) hc->port, (ngx_uint_t) hc->threshold);
588 }
589 BRIX_HC_METRIC_INC(hc_fail_total);
590 }
591
592 ngx_destroy_pool(pool); /* frees hc itself (last use above) */
593}

596 * Pre-connection failure cleanup for brix_hc_start(): the socket/connection
597 * could not be set up before hc took ownership, so the resolver result (if any)

615 * Allocation failures before the probe is viable count as neither pass nor
616 * fail toward blacklisting (threshold/blacklist_ms = 0); they only clear
617 * hc_in_progress so the slot is re-probed next interval.

619static brix_hc_ctx_t *

649/*
650 * Resolve host:port for brix_hc_start(). Returns the addrinfo list (owned by
651 * the caller), or NULL after reporting a real probe failure (configured
652 * threshold/blacklist) and destroying the probe pool.

654static struct addrinfo *
655brix_hc_resolve(brix_hc_ctx_t *hc, const char *host, uint16_t port)
656{
657 struct addrinfo hints, *res = NULL;
658 char portstr[8];

660 ngx_memzero(&hints, sizeof(hints));
661 hints.ai_family = AF_UNSPEC;
662 hints.ai_socktype = SOCK_STREAM;

664 (int) port);

666 ngx_log_error(NGX_LOG_WARN, hc->log, 0,

670 hc->conf->hc.blacklist_ms);

672 return NULL;
673 }
674 return res;

677/*
678 * Create the non-blocking probe socket and wire it into an nginx connection
679 * for brix_hc_start(). Returns NGX_OK with hc->conn owning the fd, or
680 * NGX_ERROR after full pre-connection cleanup (socket closed by hand,
681 * brix_hc_pre_connect_fail() frees res, records the failure, destroys the
682 * pool).
683 */
684static ngx_int_t
685brix_hc_open_conn(ngx_cycle_t *cycle, brix_hc_ctx_t *hc,
686 struct addrinfo *res, const char *host, uint16_t port)
687{
688 ngx_connection_t *c;
689 ngx_socket_t fd;
690
691 /* Until hc->conn is wired up, the socket is owned manually and must be
692 * closed by hand on error (the pool does not know about a bare fd). These
693 * pre-connection failures count as a real probe failure (they use the
694 * configured threshold/blacklist). */
695 fd = ngx_socket(res->ai_family, SOCK_STREAM, 0);
696 if (fd == (ngx_socket_t) -1 || ngx_nonblocking(fd) == -1) {
697 if (fd != (ngx_socket_t) -1) { ngx_close_socket(fd); }
698 brix_hc_pre_connect_fail(res, host, port, hc->conf, hc->pool);
699 return NGX_ERROR;
700 }
701
702 c = ngx_get_connection(fd, cycle->log);
703 if (c == NULL) {
704 ngx_close_socket(fd);
705 brix_hc_pre_connect_fail(res, host, port, hc->conf, hc->pool);
706 return NGX_ERROR;
707 }
708 /* From here the connection (and therefore the fd) is owned by hc->conn and
709 * released through brix_hc_finish() -> ngx_close_connection(). */
710 c->pool = hc->pool;
711 c->data = hc;
712 c->recv = ngx_recv;
713 c->send = ngx_send;
714 c->read->handler = brix_hc_read_handler;
715 c->write->handler = brix_hc_write_handler;
716 c->read->log = c->write->log = cycle->log;
717 hc->conn = c;
718 return NGX_OK;
719}

721/*
722 * Build the pipelined bootstrap write buffer for brix_hc_start(). Returns
723 * NGX_OK with hc->wbuf/wbuf_len/wbuf_pos queued for sending, or NGX_ERROR on
724 * allocation failure (no cleanup here; the caller reports the verdict).
725 */
726static ngx_int_t
727brix_hc_queue_bootstrap(brix_hc_ctx_t *hc)
728{
729 size_t bslen;
730
731 /* Step F: with an outbound TLS ctx the probe can complete a kXR_gotoTLS
732 * upgrade, so advertise kXR_ableTLS — a brix server only answers gotoTLS
733 * to TLS-capable clients (it lets others finish in cleartext, which would
734 * silently skip the deep-probe path). A TLS-capable probe must NOT
735 * pipeline the plaintext login behind the protocol request: on gotoTLS
736 * the server hands every pending cleartext byte to the TLS layer, and a
737 * pre-sent login corrupts the handshake ("packet length too long"). So
738 * it sends handshake + protocol only and defers login until the protocol
739 * verdict. Without a ctx the flag byte stays 0 and the full pipelined
740 * bootstrap is byte-identical to pre-Step-F probes. */
741#if (NGX_SSL)
742 hc->tls_capable = (hc->conf->upstream_tls
743 && hc->conf->upstream_tls_ctx != NULL);
744#endif
745 /* Always allocate (and build) all three frames; wbuf_len truncates the
746 * SEND to handshake + protocol for a TLS-capable probe. */
747 bslen = XRD_HANDSHAKE_LEN + sizeof(ClientProtocolRequest)
748 + sizeof(ClientLoginRequest);
749 hc->wbuf = ngx_palloc(hc->pool, bslen);
750 if (hc->wbuf == NULL) {
751 return NGX_ERROR;
752 }
753 brix_upstream_build_bootstrap_flags(hc->wbuf,
754 hc->tls_capable ? kXR_ableTLS : 0);
755 hc->wbuf_len = hc->tls_capable
756 ? bslen - sizeof(ClientLoginRequest) : bslen;
757 hc->wbuf_pos = 0;
758 return NGX_OK;
759}

761/*
762 * Kick off the connect() for brix_hc_start() and hand the probe to the event
763 * loop. Frees the resolver result; every failure path ends in
764 * brix_hc_finish(), so the claim is always released.
765 */
766static void
767brix_hc_begin_connect(brix_hc_ctx_t *hc, struct addrinfo *res)
768{
769 int rc;
770
771 /* Non-blocking connect: rc == 0 means it completed immediately (typical for
772 * localhost/loopback); EINPROGRESS means it is in flight; any other error
773 * is a hard failure. Free the resolver result either way before branching. */
774 rc = connect(hc->conn->fd, res->ai_addr, res->ai_addrlen);
775 freeaddrinfo(res);
776
777 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
778 brix_hc_finish(hc, 0);
779 return;
780 }
781 if (ngx_handle_write_event(hc->conn->write, 0) != NGX_OK) {
782 brix_hc_finish(hc, 0);
783 return;
784 }
785
786 if (rc == 0) {
787 /* Connected synchronously: send the bootstrap now; the reply arrives
788 * via the read handler. */
789 hc->connecting = 0;
790 if (brix_hc_flush(hc) == NGX_ERROR) {
791 brix_hc_finish(hc, 0);
792 }
793 } else {
794 /* In progress: the first writable event runs brix_hc_write_handler,
795 * which confirms SO_ERROR and then flushes the bootstrap. */
796 hc->connecting = 1; /* finish in the write handler */
797 }
798}

800/*
801 * Launch one probe against host:port (the slot was already claimed by
802 * brix_srv_hc_claim()). Sets up a non-blocking connection, queues the
803 * pipelined bootstrap, arms the deadline timer, and starts connect(). Control
804 * then returns to the event loop; the probe completes asynchronously in the
805 * read/write handlers, always ending at brix_hc_finish().
806 *
807 * Ownership note: every early-exit path here MUST release the claim — either
808 * via brix_hc_finish() (once hc exists and reports a verdict) or via a direct
809 * brix_srv_hc_fail() (before hc is usable). The helpers above uphold this on
810 * their own failure paths. Otherwise the slot stays hc_in_progress=1 forever
811 * and is never probed again.
812 */
813static void
814brix_hc_start(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *conf,
815 const char *host, uint16_t port)
816{
817 brix_hc_ctx_t *hc;
818 struct addrinfo *res;
819
820 hc = brix_hc_ctx_create(cycle, conf, host, port);
821 if (hc == NULL) {
822 return;
823 }
824
825 /* Resolve target. */
826 res = brix_hc_resolve(hc, host, port);
827 if (res == NULL) {
828 return;
829 }
830
831 if (brix_hc_open_conn(cycle, hc, res, host, port) != NGX_OK) {
832 return;
833 }
834
835 if (brix_hc_queue_bootstrap(hc) != NGX_OK) {
836 freeaddrinfo(res);
837 brix_hc_finish(hc, 0);
838 return;
839 }
840
841 /* Single deadline for the whole probe. */
842 hc->tev.handler = brix_hc_timeout_handler;
843 hc->tev.data = hc;
844 hc->tev.log = cycle->log;
845 ngx_add_timer(&hc->tev, conf->hc.timeout_ms);
846
847 BRIX_HC_METRIC_INC(hc_probes_total);
848
849 brix_hc_begin_connect(hc, res);
850}

852/*
853 * Recurring scan tick. Re-arms itself first (so a probe-launch failure can't
854 * stop future scans), then tries to claim at most one due server from the
855 * shared registry and probe it. Claiming exactly one per tick is what spreads
856 * load and guarantees a single worker probes each server per interval.
857 */
858static void
859brix_hc_timer_handler(ngx_event_t *ev)
860{
861 brix_hc_mgr_t *mgr = ev->data;
862 ngx_stream_brix_srv_conf_t *conf = mgr->conf;
863 char host[256];
864 uint16_t port;
865 ngx_msec_t next_due = conf->hc.interval_ms;
866 ngx_msec_t delay;
867
868 /* Claim is atomic under the registry spinlock and self-rate-limits via
869 * hc_next_check, so concurrent workers never double-probe one server. */
870 if (brix_srv_hc_claim(host, sizeof(host), &port,
871 conf->hc.interval_ms, &next_due))
872 {
873 brix_hc_start(mgr->cycle, conf, host, port);
874 /* More servers may be due now — spread the remaining probes at the
875 * per-slot cadence rather than firing them all this tick. */
876 delay = mgr->scan_interval_ms;
877 } else {
878 /* Nothing due: sleep until the soonest server becomes due instead of
879 * waking at the sub-second floor to find nothing (the old idle drain).
880 * Floored by the spread cadence (burst guard) and capped by the probe
881 * interval (so a freshly-registered server is probed within one). */
882 delay = next_due;
883 if (delay < mgr->scan_interval_ms) {
884 delay = mgr->scan_interval_ms;
885 }
886 if (delay > conf->hc.interval_ms) {
887 delay = conf->hc.interval_ms;
888 }
889 }
890
891 /* Re-arm (bounded to [scan_interval, hc_interval]) — a probe-launch
892 * failure or empty registry can never stop future scans. Suppressed once
893 * the worker is exiting so the scan loop releases the draining worker. */
894 if (!ngx_exiting) {
895 ngx_add_timer(ev, delay);
896 }
897}
