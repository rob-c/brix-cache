1/*
2 * health_check.c — Phase 22 active stream health checks (see health_check.h).
3 *
4 * The probe is a self-contained async XRootD client connection. Its wire
5 * framing mirrors the proven upstream bootstrap (src/upstream/bootstrap.c):
6 * the bootstrap write buffer is built by the shared
7 * brix_upstream_build_bootstrap(), and responses are read as uniform 8-byte
8 * ServerResponseHdr + dlen-byte body frames, exactly as the upstream read loop
9 * does. Unlike the upstream path it carries no client context — on bootstrap
10 * completion it sends kXR_ping (or kXR_stat "/") and reports the verdict to the
11 * registry rather than forwarding a saved client request.
12 *
13 * A single deadline timer (hc_timeout_ms) bounds the whole probe; a hung server
14 * that accepts the TCP connection but never answers is failed on timeout.
15 */

27/*
28 * Probe state machine, advanced one step per complete response frame received.
29 * Each phase consumes the server's reply to the request the previous phase (or
30 * the pipelined bootstrap) sent: HANDSHAKE -> PROTOCOL -> LOGIN -> PROBE. See
31 * brix_hc_dispatch() for the per-phase transitions and early-exit verdicts.
32 */
33typedef enum {
34 XRD_HC_HANDSHAKE = 0,
35 XRD_HC_PROTOCOL,
36 XRD_HC_LOGIN,
37 XRD_HC_PROBE,
38} brix_hc_phase_t;

102/* write side */
103/*
104 * Drain hc->wbuf to the socket without blocking. Returns NGX_OK once the whole
105 * buffer is sent (and read events re-armed so the reply can arrive), NGX_AGAIN
106 * if the socket is full (write event re-armed; brix_hc_write_handler resumes
107 * here), or NGX_ERROR on a fatal send error.
108 */
109static ngx_int_t
110brix_hc_flush(brix_hc_ctx_t *hc)
111{
112 ngx_connection_t *c = hc->conn;
113 ssize_t n;
114
115 /* Partial-write loop: send() may consume only part of the buffer, so track
116 * wbuf_pos and resume from there on the next writable event. */
117 while (hc->wbuf_pos < hc->wbuf_len) {
118 n = c->send(c, hc->wbuf + hc->wbuf_pos, hc->wbuf_len - hc->wbuf_pos);
119 if (n > 0) {
120 hc->wbuf_pos += (size_t) n;
121 continue;
122 }
123 if (n == NGX_AGAIN) {
124 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
125 return NGX_ERROR;
126 }
127 return NGX_AGAIN;
128 }
129 return NGX_ERROR;
130 }
131 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
132 return NGX_ERROR;
133 }
134 return NGX_OK;
135}

137/*
138 * Build and send the final liveness probe (kXR_ping or kXR_stat "/") after a
139 * successful login, transitioning to XRD_HC_PROBE. The reply's status is the
140 * pass/fail verdict, decided back in brix_hc_dispatch().
141 */
142static void
143brix_hc_send_probe(brix_hc_ctx_t *hc)
144{
145 if (hc->probe_type == BRIX_HC_TYPE_STAT) {
146 /* kXR_stat request followed by a 1-byte path body "/". Over-allocate
147 * by 1 so the path char sits immediately after the fixed header. */
148 size_t total = sizeof(ClientStatRequest) + 1; /* path "/" */
149 ClientStatRequest *r = ngx_palloc(hc->pool, total);
150 if (r == NULL) { brix_hc_finish(hc, 0); return; }
151 ngx_memzero(r, sizeof(*r));
152 /* streamid is a 2-byte client tag echoed in the reply; {0,2} marks this
153 * as a health request (see ping branch below). */
154 r->streamid[0] = 0;
155 r->streamid[1] = 2;
156 r->requestid = htons(kXR_stat); /* opcode, network order */
157 r->dlen = htonl((kXR_int32) 1); /* body length = 1 ("/") */
158 ((u_char *) r)[sizeof(*r)] = '/'; /* append path after header */
159 hc->wbuf = (u_char *) r;
160 hc->wbuf_len = total;
161 } else {
162 ClientPingRequest *r = ngx_palloc(hc->pool, sizeof(*r));
163 if (r == NULL) { brix_hc_finish(hc, 0); return; }
164 ngx_memzero(r, sizeof(*r));
165 r->streamid[0] = 0;
166 r->streamid[1] = 2; /* streamid 2 marks a health request */
167 r->requestid = htons(kXR_ping);
168 r->dlen = 0; /* ping carries no body */
169 hc->wbuf = (u_char *) r;
170 hc->wbuf_len = sizeof(*r);
171 }
172 /* Reset both write and read accumulators for this fresh request/response. */
173 hc->wbuf_pos = 0;
174 hc->rhdr_pos = 0;
175 hc->resp_dlen = 0;
176 hc->resp_body = NULL;
177 hc->resp_body_pos = 0;
178 hc->phase = XRD_HC_PROBE;
179
180 if (brix_hc_flush(hc) == NGX_ERROR) {
181 brix_hc_finish(hc, 0);
182 }
183}

185/*
186 * Writable-event handler. First writable event after a non-blocking connect()
187 * confirms (or rejects) the TCP connection via SO_ERROR; subsequent calls just
188 * resume a partially-flushed write buffer.
189 */
190static void
191brix_hc_write_handler(ngx_event_t *wev)
192{
193 ngx_connection_t *c = wev->data;
194 brix_hc_ctx_t *hc = c->data;
195 ngx_int_t rc;
196
197 if (wev->timedout) {
198 brix_hc_finish(hc, 0);
199 return;
200 }
201
202 /* Non-blocking connect completes asynchronously: the first writable event
203 * means connect() finished — SO_ERROR carries success (0) or the failure
204 * code, since connect() itself returned EINPROGRESS earlier. */
205 if (hc->connecting) {
206 int err = 0;
207 socklen_t len = sizeof(err);
208 if (getsockopt(c->fd, SOL_SOCKET, SO_ERROR, (char *) &err, &len) == -1
209 || err != 0)
210 {
211 BRIX_DIAG_WARN(hc->log, err, /* may be 0 */
212 "brix: health check failed to connect to %s:%d",
213 "the cluster member is down or unreachable from the manager",
214 "check that member's xrootd/data service and the network path "
215 "to it; it is marked DOWN and clients are routed elsewhere "
216 "until it passes again",
217 hc->host, (int) hc->port);
218 brix_hc_finish(hc, 0);
219 return;
220 }
221 hc->connecting = 0;
222 }
223
224 rc = brix_hc_flush(hc);
225 if (rc == NGX_ERROR) {
226 brix_hc_finish(hc, 0);
227 }
228}

231/* read side */
232/* Accumulate one full ServerResponseHdr + body frame; return NGX_OK when a
233 * complete frame is ready, NGX_AGAIN to wait for more, NGX_ERROR on close. */
234static ngx_int_t
235brix_hc_recv_frame(brix_hc_ctx_t *hc)
236{
237 ngx_connection_t *c = hc->conn;
238 ssize_t n;
239
240 /* Stage 1: fill the fixed 8-byte header, possibly across several reads
241 * (rhdr_pos tracks how much has arrived). Only once it is complete do we
242 * decode status/dlen and learn the body length. */
243 if (hc->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
244 size_t need = XRD_RESPONSE_HDR_LEN - hc->rhdr_pos;
245 n = c->recv(c, hc->rhdr + hc->rhdr_pos, need);
246 if (n == NGX_AGAIN) { return NGX_AGAIN; }
247 if (n <= 0) { return NGX_ERROR; } /* 0 = peer closed */
248 hc->rhdr_pos += (size_t) n;
249 if (hc->rhdr_pos < XRD_RESPONSE_HDR_LEN) {
250 return NGX_AGAIN;
251 }
252 /* Header complete: decode the wire fields (big-endian on the wire). */
253 {
254 ServerResponseHdr *h = (ServerResponseHdr *) (void *) hc->rhdr;
255 hc->resp_status = ntohs(h->status);
256 hc->resp_dlen = ntohl(h->dlen);
257 }
258 if (hc->resp_dlen > 0) {
259 /* Bound attacker/garbage-controlled body length: a probe reply is
260 * tiny, so cap the allocation rather than trust the wire dlen. */
261 if (hc->resp_dlen > 4096) { /* bound probe response bodies */
262 return NGX_ERROR;
263 }
264 hc->resp_body = ngx_palloc(c->pool, hc->resp_dlen);
265 if (hc->resp_body == NULL) { return NGX_ERROR; }
266 hc->resp_body_pos = 0;
267 }
268 }
269
270 /* Stage 2: fill the dlen-byte body (again possibly across several reads).
271 * Skipped entirely when dlen == 0. */
272 if (hc->resp_body_pos < hc->resp_dlen) {
273 size_t need = hc->resp_dlen - hc->resp_body_pos;
274 n = c->recv(c, hc->resp_body + hc->resp_body_pos, need);
275 if (n == NGX_AGAIN) { return NGX_AGAIN; }
276 if (n <= 0) { return NGX_ERROR; }
277 hc->resp_body_pos += (size_t) n;
278 if (hc->resp_body_pos < hc->resp_dlen) {
279 return NGX_AGAIN;
280 }
281 }
282
283 return NGX_OK;
284}

286/*
287 * Consume one complete response frame and advance the probe state machine.
288 * Several phases short-circuit to a verdict (alive/dead) without reaching the
289 * final PROBE phase — see the per-case comments. On a non-terminal transition
290 * it resets the accumulator and re-posts the read event so any already-buffered
291 * pipelined reply is processed in the same event cycle.
292 */
293
294static void
295brix_hc_dispatch(brix_hc_ctx_t *hc)
296{
297 switch (hc->phase) {
298
299 case XRD_HC_HANDSHAKE:
300 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return; }
301 hc->phase = XRD_HC_PROTOCOL;
302 break;
303
304 case XRD_HC_PROTOCOL:
305 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return; }
306 /* If the server demands TLS we don't probe deeper (no TLS in the
307 * probe path); it answered at the protocol level, so treat it as
308 * alive rather than blacklisting it.
309 *
310 * ServerResponseBody_Protocol layout: bytes 0-3 = pval (protocol
311 * version), bytes 4-7 = flags. Read the flags word at offset 4 and
312 * test the kXR_gotoTLS bit. Guard on dlen >= 8 so short/old replies
313 * that omit the flags simply fall through to the LOGIN phase. */
314 if (hc->resp_dlen >= 8) {
315 uint32_t flags_be;
316 ngx_memcpy(&flags_be, hc->resp_body + 4, sizeof(flags_be));
317 if (ntohl(flags_be) & kXR_gotoTLS) {
318 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, hc->log, 0,
319 "brix: health check: %s:%d wants TLS; "
320 "treating protocol-OK as alive", hc->host, (int) hc->port);
321 brix_hc_finish(hc, 1);
322 return;
323 }
324 }
325 hc->phase = XRD_HC_LOGIN;
326 break;
327
328 case XRD_HC_LOGIN:
329 /* authmore => server is alive but wants credentials; we don't carry
330 * any, so accept protocol liveness rather than fail. */
331 if (hc->resp_status == kXR_authmore) {
332 brix_hc_finish(hc, 1);
333 return;
334 }
335 if (hc->resp_status != kXR_ok) { brix_hc_finish(hc, 0); return; }
336 brix_hc_send_probe(hc); /* sets phase = PROBE, sends ping/stat */
337 return;
338
339 case XRD_HC_PROBE:
340 brix_hc_finish(hc, hc->resp_status == kXR_ok ? 1 : 0);
341 return;
342 }
343
344 /* Reset accumulator for the next frame and post a synthetic read so any
345 * pipelined bytes already in the socket buffer are processed this cycle. */
346 hc->rhdr_pos = 0;
347 hc->resp_dlen = 0;
348 hc->resp_body = NULL;
349 hc->resp_body_pos = 0;
350
351 if (ngx_handle_read_event(hc->conn->read, 0) != NGX_OK) {
352 brix_hc_finish(hc, 0);
353 return;
354 }
355 ngx_post_event(hc->conn->read, &ngx_posted_events);
356}

358/*
359 * Readable-event handler. Pulls exactly one complete frame per invocation and
360 * hands it to brix_hc_dispatch(), which re-arms (or posts) the read event for
361 * the next frame; this keeps each event-loop iteration bounded.
362 */
363static void
364brix_hc_read_handler(ngx_event_t *rev)
365{
366 ngx_connection_t *c = rev->data;
367 brix_hc_ctx_t *hc = c->data;
368 ngx_int_t rc;
369
370 if (rev->timedout) {
371 brix_hc_finish(hc, 0);
372 return;
373 }
374
375 /* Loop is structural only: every branch returns. It exists so a future
376 * "read another frame inline" change has an obvious place to continue. */
377 for ( ;; ) {
378 rc = brix_hc_recv_frame(hc);
379 if (rc == NGX_AGAIN) {
380 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
381 brix_hc_finish(hc, 0);
382 }
383 return;
384 }
385 if (rc == NGX_ERROR) {
386 brix_hc_finish(hc, 0);
387 return;
388 }
389 brix_hc_dispatch(hc);
390 return; /* dispatch re-arms / posts as needed */
391 }
392}

413/*
414 * Single termination point for a probe (called from every error/success path).
415 * Reports the verdict to the shared registry, bumps metrics, and tears down all
416 * probe state. MUST be the last thing any caller does with hc: it destroys the
417 * pool that backs hc itself, so hc is dangling on return.
418 */
419static void
420brix_hc_finish(brix_hc_ctx_t *hc, int passed)
421{
422 /* Stash the pool before any registry call: hc lives inside it and is freed
423 * by the ngx_destroy_pool() at the end. */
424 ngx_pool_t *pool = hc->pool;
425
426 /* Cancel the deadline timer and close the socket before reporting, so a
427 * verdict callback can never observe a half-torn-down probe. */
428 if (hc->tev.timer_set) {
429 ngx_del_timer(&hc->tev);
430 }
431 if (hc->conn != NULL) {
432 ngx_close_connection(hc->conn);
433 hc->conn = NULL;
434 }
435
436 if (passed) {
437 brix_srv_hc_pass(hc->host, hc->port);
438 BRIX_HC_METRIC_INC(hc_pass_total);
439 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, hc->log, 0,
440 "brix: health check: %s:%d passed",
441 hc->host, (int) hc->port);
442 } else {
443 if (brix_srv_hc_fail(hc->host, hc->port,
444 hc->threshold, hc->blacklist_ms))
445 {
446 BRIX_HC_METRIC_INC(hc_blacklist_total);
447 ngx_log_error(NGX_LOG_WARN, hc->log, 0,
448 "brix: health check: %s:%d blacklisted after"
449 " %ui consecutive failures",
450 hc->host, (int) hc->port, (ngx_uint_t) hc->threshold);
451 }
452 BRIX_HC_METRIC_INC(hc_fail_total);
453 }
454
455 ngx_destroy_pool(pool); /* frees hc itself (last use above) */
456}

475/*
476 * Launch one probe against host:port (the slot was already claimed by
477 * brix_srv_hc_claim()). Sets up a non-blocking connection, queues the
478 * pipelined bootstrap, arms the deadline timer, and starts connect(). Control
479 * then returns to the event loop; the probe completes asynchronously in the
480 * read/write handlers, always ending at brix_hc_finish().
481 *
482 * Ownership note: every early-exit path here MUST release the claim — either
483 * via brix_hc_finish() (once hc exists and reports a verdict) or via a direct
484 * brix_srv_hc_fail() (before hc is usable). Otherwise the slot stays
485 * hc_in_progress=1 forever and is never probed again.
486 */
487static void
488brix_hc_start(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *conf,
489 const char *host, uint16_t port)
490{
491 ngx_pool_t *pool;
492 brix_hc_ctx_t *hc;
493 ngx_connection_t *c;
494 struct addrinfo hints, *res = NULL;
495 char portstr[8];
496 ngx_socket_t fd;
497 size_t bslen;
498 int rc;
499
500 /* Allocation failures before the probe is viable count as neither pass nor
501 * fail toward blacklisting (threshold/blacklist_ms = 0); they only clear
502 * hc_in_progress so the slot is re-probed next interval. */
503 pool = ngx_create_pool(1024, cycle->log);
504 if (pool == NULL) {
505 brix_srv_hc_fail(host, port, 0, 0); /* release the claim */
506 return;
507 }
508 hc = ngx_pcalloc(pool, sizeof(*hc));
509 if (hc == NULL) {
510 ngx_destroy_pool(pool);
511 brix_srv_hc_fail(host, port, 0, 0);
512 return;
513 }
514 hc->pool = pool;
515 hc->log = cycle->log;
516 hc->phase = XRD_HC_HANDSHAKE;
517 hc->probe_type = conf->hc.type;
518 hc->threshold = (uint32_t) conf->hc.threshold;
519 hc->blacklist_ms = conf->hc.blacklist_ms;
520 hc->port = port;
521 ngx_cpystrn((u_char *) hc->host, (u_char *) host, sizeof(hc->host));
522
523 /* Resolve target. */
524 ngx_memzero(&hints, sizeof(hints));
525 hints.ai_family = AF_UNSPEC;
526 hints.ai_socktype = SOCK_STREAM;
527 (void) ngx_snprintf((u_char *) portstr, sizeof(portstr), "%d%Z",
528 (int) port);
529 if (getaddrinfo(host, portstr, &hints, &res) != 0 || res == NULL) {
530 ngx_log_error(NGX_LOG_WARN, cycle->log, 0,
531 "brix: health check: cannot resolve %s:%d",
532 host, (int) port);
533 brix_srv_hc_fail(host, port, conf->hc.threshold,
534 conf->hc.blacklist_ms);
535 ngx_destroy_pool(pool);
536 return;
537 }
538
539 /* Until hc->conn is wired up, the socket is owned manually and must be
540 * closed by hand on error (the pool does not know about a bare fd). These
541 * pre-connection failures jump to `fail`, which counts as a real probe
542 * failure (uses the configured threshold/blacklist). */
543 fd = ngx_socket(res->ai_family, SOCK_STREAM, 0);
544 if (fd == (ngx_socket_t) -1 || ngx_nonblocking(fd) == -1) {
545 if (fd != (ngx_socket_t) -1) { ngx_close_socket(fd); }
546 brix_hc_pre_connect_fail(res, host, port, conf, pool);
547 return;
548 }
549
550 c = ngx_get_connection(fd, cycle->log);
551 if (c == NULL) {
552 ngx_close_socket(fd);
553 brix_hc_pre_connect_fail(res, host, port, conf, pool);
554 return;
555 }
556 /* From here the connection (and therefore the fd) is owned by hc->conn and
557 * released through brix_hc_finish() -> ngx_close_connection(). */
558 c->pool = pool;
559 c->data = hc;
560 c->recv = ngx_recv;
561 c->send = ngx_send;
562 c->read->handler = brix_hc_read_handler;
563 c->write->handler = brix_hc_write_handler;
564 c->read->log = c->write->log = cycle->log;
565 hc->conn = c;
566
567 /* Pipelined bootstrap: handshake + protocol + login. */
568 bslen = XRD_HANDSHAKE_LEN + sizeof(ClientProtocolRequest)
569 + sizeof(ClientLoginRequest);
570 hc->wbuf = ngx_palloc(pool, bslen);
571 if (hc->wbuf == NULL) {
572 freeaddrinfo(res);
573 brix_hc_finish(hc, 0);
574 return;
575 }
576 brix_upstream_build_bootstrap(hc->wbuf);
577 hc->wbuf_len = bslen;
578 hc->wbuf_pos = 0;
579
580 /* Single deadline for the whole probe. */
581 hc->tev.handler = brix_hc_timeout_handler;
582 hc->tev.data = hc;
583 hc->tev.log = cycle->log;
584 ngx_add_timer(&hc->tev, conf->hc.timeout_ms);
585
586 BRIX_HC_METRIC_INC(hc_probes_total);
587
588 /* Non-blocking connect: rc == 0 means it completed immediately (typical for
589 * localhost/loopback); EINPROGRESS means it is in flight; any other error
590 * is a hard failure. Free the resolver result either way before branching. */
591 rc = connect(fd, res->ai_addr, res->ai_addrlen);
592 freeaddrinfo(res);
593 res = NULL;
594
595 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
596 brix_hc_finish(hc, 0);
597 return;
598 }
599 if (ngx_handle_write_event(c->write, 0) != NGX_OK) {
600 brix_hc_finish(hc, 0);
601 return;
602 }
603
604 if (rc == 0) {
605 /* Connected synchronously: send the bootstrap now; the reply arrives
606 * via the read handler. */
607 hc->connecting = 0;
608 if (brix_hc_flush(hc) == NGX_ERROR) {
609 brix_hc_finish(hc, 0);
610 }
611 } else {
612 /* In progress: the first writable event runs brix_hc_write_handler,
613 * which confirms SO_ERROR and then flushes the bootstrap. */
614 hc->connecting = 1; /* finish in the write handler */
615 }
616 return;
617}

619/*
620 * Recurring scan tick. Re-arms itself first (so a probe-launch failure can't
621 * stop future scans), then tries to claim at most one due server from the
622 * shared registry and probe it. Claiming exactly one per tick is what spreads
623 * load and guarantees a single worker probes each server per interval.
624 */
625static void
626brix_hc_timer_handler(ngx_event_t *ev)
627{
628 brix_hc_mgr_t *mgr = ev->data;
629 ngx_stream_brix_srv_conf_t *conf = mgr->conf;
630 char host[256];
631 uint16_t port;
632 ngx_msec_t next_due = conf->hc.interval_ms;
633 ngx_msec_t delay;
634
635 /* Claim is atomic under the registry spinlock and self-rate-limits via
636 * hc_next_check, so concurrent workers never double-probe one server. */
637 if (brix_srv_hc_claim(host, sizeof(host), &port,
638 conf->hc.interval_ms, &next_due))
639 {
640 brix_hc_start(mgr->cycle, conf, host, port);
641 /* More servers may be due now — spread the remaining probes at the
642 * per-slot cadence rather than firing them all this tick. */
643 delay = mgr->scan_interval_ms;
644 } else {
645 /* Nothing due: sleep until the soonest server becomes due instead of
646 * waking at the sub-second floor to find nothing (the old idle drain).
647 * Floored by the spread cadence (burst guard) and capped by the probe
648 * interval (so a freshly-registered server is probed within one). */
649 delay = next_due;
650 if (delay < mgr->scan_interval_ms) {
651 delay = mgr->scan_interval_ms;
652 }
653 if (delay > conf->hc.interval_ms) {
654 delay = conf->hc.interval_ms;
655 }
656 }
657
658 /* Re-arm (bounded to [scan_interval, hc_interval]) — a probe-launch
659 * failure or empty registry can never stop future scans. Suppressed once
660 * the worker is exiting so the scan loop releases the draining worker. */
661 if (!ngx_exiting) {
662 ngx_add_timer(ev, delay);
663 }
664}

666void
667brix_hc_manager_start(ngx_cycle_t *cycle,
668 ngx_stream_brix_srv_conf_t *conf)
669{
670 brix_hc_mgr_t *mgr;
671 ngx_msec_t scan;
672
673 if (!conf->hc.enabled || conf->hc.interval_ms == 0) {
674 return;
675 }
676
677 mgr = ngx_pcalloc(cycle->pool, sizeof(*mgr));
678 if (mgr == NULL) {
679 return;
680 }
681 mgr->cycle = cycle;
682 mgr->conf = conf;
683
684 /* Spread probes across the interval: one claim attempt every
685 * interval/slots ms, floored at 100 ms. */
686 scan = conf->hc.interval_ms / BRIX_SRV_REGISTRY_SLOTS;
687 mgr->scan_interval_ms = scan < 100 ? 100 : scan;
688
689 mgr->timer.handler = brix_hc_timer_handler;
690 mgr->timer.data = mgr;
691 mgr->timer.log = cycle->log;
692
693 ngx_add_timer(&mgr->timer, 2000); /* let CMS connections settle first */
694
695 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
696 "brix: health check manager started "
697 "(interval=%Ms timeout=%Ms scan=%Ms)",
698 conf->hc.interval_ms, conf->hc.timeout_ms,
699 mgr->scan_interval_ms);
700}
