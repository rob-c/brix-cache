1/*
2 * health_check.c — Phase 22 active stream health checks (see health_check.h).
3 *
4 * The probe is a self-contained async XRootD client connection. Its wire
5 * framing mirrors the proven upstream bootstrap (src/upstream/bootstrap.c):
6 * the bootstrap write buffer is built by the shared
7 * brix_upstream_build_bootstrap(), and responses are read as uniform 8-byte
8 * ServerResponseHdr + dlen-byte body frames, exactly as the upstream read loop
9 * does. Unlike the upstream path it carries no client context — on bootstrap
10 * completion it sends kXR_ping (or kXR_stat "/") and reports the verdict to the
11 * registry rather than forwarding a saved client request.
12 *
13 * A single deadline timer (hc_timeout_ms) bounds the whole probe; a hung server
14 * that accepts the TCP connection but never answers is failed on timeout.
15 */

60/*
61 * Single termination point for a probe (called from every error/success path).
62 * Reports the verdict to the shared registry, bumps metrics, and tears down all
63 * probe state. MUST be the last thing any caller does with hc: it destroys the
64 * pool that backs hc itself, so hc is dangling on return.
65 */
66void
67brix_hc_finish(brix_hc_ctx_t *hc, int passed)
68{
69 /* Stash the pool before any registry call: hc lives inside it and is freed
70 * by the ngx_destroy_pool() at the end. */
71 ngx_pool_t *pool = hc->pool;
72
73 /* Cancel the deadline timer and close the socket before reporting, so a
74 * verdict callback can never observe a half-torn-down probe. */
75 if (hc->tev.timer_set) {
76 ngx_del_timer(&hc->tev);
77 }
78 if (hc->conn != NULL) {
79 ngx_close_connection(hc->conn);
80 hc->conn = NULL;
81 }
82
83 if (passed) {
84 brix_srv_hc_pass(hc->host, hc->port);
85 BRIX_HC_METRIC_INC(hc_pass_total);
86 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, hc->log, 0,
87 "brix: health check: %s:%d passed",
88 hc->host, (int) hc->port);
89 } else {
90 if (brix_srv_hc_fail(hc->host, hc->port,
91 hc->threshold, hc->blacklist_ms))
92 {
93 BRIX_HC_METRIC_INC(hc_blacklist_total);
94 ngx_log_error(NGX_LOG_WARN, hc->log, 0,
95 "brix: health check: %s:%d blacklisted after"
96 " %ui consecutive failures",
97 hc->host, (int) hc->port, (ngx_uint_t) hc->threshold);
98 }
99 BRIX_HC_METRIC_INC(hc_fail_total);
100 }
101
102 ngx_destroy_pool(pool); /* frees hc itself (last use above) */
103}

122/*
123 * Allocate the probe pool + ctx and seed the identity/config fields for
124 * brix_hc_start(). Returns the ctx, or NULL after releasing the claim.
125 * Allocation failures before the probe is viable count as neither pass nor
126 * fail toward blacklisting (threshold/blacklist_ms = 0); they only clear
127 * hc_in_progress so the slot is re-probed next interval.
128 */
129static brix_hc_ctx_t *
130brix_hc_ctx_create(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *conf,
131 const char *host, uint16_t port)
132{
133 ngx_pool_t *pool;
134 brix_hc_ctx_t *hc;
135
136 pool = ngx_create_pool(1024, cycle->log);
137 if (pool == NULL) {
138 brix_srv_hc_fail(host, port, 0, 0); /* release the claim */
139 return NULL;
140 }
141 hc = ngx_pcalloc(pool, sizeof(*hc));
142 if (hc == NULL) {
143 ngx_destroy_pool(pool);
144 brix_srv_hc_fail(host, port, 0, 0);
145 return NULL;
146 }
147 hc->pool = pool;
148 hc->log = cycle->log;
149 hc->conf = conf;
150 hc->phase = XRD_HC_HANDSHAKE;
151 hc->probe_type = conf->hc.type;
152 hc->threshold = (uint32_t) conf->hc.threshold;
153 hc->blacklist_ms = conf->hc.blacklist_ms;
154 hc->port = port;
155 ngx_cpystrn((u_char *) hc->host, (u_char *) host, sizeof(hc->host));
156 return hc;
157}

159/*
160 * Resolve host:port for brix_hc_start(). Returns the addrinfo list (owned by
161 * the caller), or NULL after reporting a real probe failure (configured
162 * threshold/blacklist) and destroying the probe pool.
163 */
164static struct addrinfo *
165brix_hc_resolve(brix_hc_ctx_t *hc, const char *host, uint16_t port)
166{
167 struct addrinfo hints, *res = NULL;
168 char portstr[8];
169
170 ngx_memzero(&hints, sizeof(hints));
171 hints.ai_family = AF_UNSPEC;
172 hints.ai_socktype = SOCK_STREAM;
173 (void) ngx_snprintf((u_char *) portstr, sizeof(portstr), "%d%Z",
174 (int) port);
175 if (getaddrinfo(host, portstr, &hints, &res) != 0 || res == NULL) {
176 ngx_log_error(NGX_LOG_WARN, hc->log, 0,
177 "brix: health check: cannot resolve %s:%d",
178 host, (int) port);
179 brix_srv_hc_fail(host, port, hc->conf->hc.threshold,
180 hc->conf->hc.blacklist_ms);
181 ngx_destroy_pool(hc->pool);
182 return NULL;
183 }
184 return res;
185}

187/*
188 * Create the non-blocking probe socket and wire it into an nginx connection
189 * for brix_hc_start(). Returns NGX_OK with hc->conn owning the fd, or
190 * NGX_ERROR after full pre-connection cleanup (socket closed by hand,
191 * brix_hc_pre_connect_fail() frees res, records the failure, destroys the
192 * pool).
193 */
194static ngx_int_t
195brix_hc_open_conn(ngx_cycle_t *cycle, brix_hc_ctx_t *hc,
196 struct addrinfo *res, const char *host, uint16_t port)
197{
198 ngx_connection_t *c;
199 ngx_socket_t fd;
200
201 /* Until hc->conn is wired up, the socket is owned manually and must be
202 * closed by hand on error (the pool does not know about a bare fd). These
203 * pre-connection failures count as a real probe failure (they use the
204 * configured threshold/blacklist). */
205 fd = ngx_socket(res->ai_family, SOCK_STREAM, 0);
206 if (fd == (ngx_socket_t) -1 || ngx_nonblocking(fd) == -1) {
207 if (fd != (ngx_socket_t) -1) { ngx_close_socket(fd); }
208 brix_hc_pre_connect_fail(res, host, port, hc->conf, hc->pool);
209 return NGX_ERROR;
210 }
211
212 c = ngx_get_connection(fd, cycle->log);
213 if (c == NULL) {
214 ngx_close_socket(fd);
215 brix_hc_pre_connect_fail(res, host, port, hc->conf, hc->pool);
216 return NGX_ERROR;
217 }
218 /* From here the connection (and therefore the fd) is owned by hc->conn and
219 * released through brix_hc_finish() -> ngx_close_connection(). */
220 c->pool = hc->pool;
221 c->data = hc;
222 c->recv = ngx_recv;
223 c->send = ngx_send;
224 c->read->handler = brix_hc_read_handler;
225 c->write->handler = brix_hc_write_handler;
226 c->read->log = c->write->log = cycle->log;
227 hc->conn = c;
228 return NGX_OK;
229}

231/*
232 * Build the pipelined bootstrap write buffer for brix_hc_start(). Returns
233 * NGX_OK with hc->wbuf/wbuf_len/wbuf_pos queued for sending, or NGX_ERROR on
234 * allocation failure (no cleanup here; the caller reports the verdict).
235 */
236static ngx_int_t
237brix_hc_queue_bootstrap(brix_hc_ctx_t *hc)
238{
239 size_t bslen;
240
241 /* Step F: with an outbound TLS ctx the probe can complete a kXR_gotoTLS
242 * upgrade, so advertise kXR_ableTLS — a brix server only answers gotoTLS
243 * to TLS-capable clients (it lets others finish in cleartext, which would
244 * silently skip the deep-probe path). A TLS-capable probe must NOT
245 * pipeline the plaintext login behind the protocol request: on gotoTLS
246 * the server hands every pending cleartext byte to the TLS layer, and a
247 * pre-sent login corrupts the handshake ("packet length too long"). So
248 * it sends handshake + protocol only and defers login until the protocol
249 * verdict. Without a ctx the flag byte stays 0 and the full pipelined
250 * bootstrap is byte-identical to pre-Step-F probes. */
251#if (NGX_SSL)
252 hc->tls_capable = (hc->conf->upstream_tls
253 && hc->conf->upstream_tls_ctx != NULL);
254#endif
255 /* Always allocate (and build) all three frames; wbuf_len truncates the
256 * SEND to handshake + protocol for a TLS-capable probe. */
257 bslen = XRD_HANDSHAKE_LEN + sizeof(ClientProtocolRequest)
258 + sizeof(ClientLoginRequest);
259 hc->wbuf = ngx_palloc(hc->pool, bslen);
260 if (hc->wbuf == NULL) {
261 return NGX_ERROR;
262 }
263 brix_upstream_build_bootstrap_flags(hc->wbuf,
264 hc->tls_capable ? kXR_ableTLS : 0);
265 hc->wbuf_len = hc->tls_capable
266 ? bslen - sizeof(ClientLoginRequest) : bslen;
267 hc->wbuf_pos = 0;
268 return NGX_OK;
269}

271/*
272 * Kick off the connect() for brix_hc_start() and hand the probe to the event
273 * loop. Frees the resolver result; every failure path ends in
274 * brix_hc_finish(), so the claim is always released.
275 */
276static void
277brix_hc_begin_connect(brix_hc_ctx_t *hc, struct addrinfo *res)
278{
279 int rc;
280
281 /* Non-blocking connect: rc == 0 means it completed immediately (typical for
282 * localhost/loopback); EINPROGRESS means it is in flight; any other error
283 * is a hard failure. Free the resolver result either way before branching. */
284 rc = connect(hc->conn->fd, res->ai_addr, res->ai_addrlen);
285 freeaddrinfo(res);
286
287 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
288 brix_hc_finish(hc, 0);
289 return;
290 }
291 if (ngx_handle_write_event(hc->conn->write, 0) != NGX_OK) {
292 brix_hc_finish(hc, 0);
293 return;
294 }
295
296 if (rc == 0) {
297 /* Connected synchronously: send the bootstrap now; the reply arrives
298 * via the read handler. */
299 hc->connecting = 0;
300 if (brix_hc_flush(hc) == NGX_ERROR) {
301 brix_hc_finish(hc, 0);
302 }
303 } else {
304 /* In progress: the first writable event runs brix_hc_write_handler,
305 * which confirms SO_ERROR and then flushes the bootstrap. */
306 hc->connecting = 1; /* finish in the write handler */
307 }
308}

310/*
311 * Launch one probe against host:port (the slot was already claimed by
312 * brix_srv_hc_claim()). Sets up a non-blocking connection, queues the
313 * pipelined bootstrap, arms the deadline timer, and starts connect(). Control
314 * then returns to the event loop; the probe completes asynchronously in the
315 * read/write handlers, always ending at brix_hc_finish().
316 *
317 * Ownership note: every early-exit path here MUST release the claim — either
318 * via brix_hc_finish() (once hc exists and reports a verdict) or via a direct
319 * brix_srv_hc_fail() (before hc is usable). The helpers above uphold this on
320 * their own failure paths. Otherwise the slot stays hc_in_progress=1 forever
321 * and is never probed again.
322 */
323static void
324brix_hc_start(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *conf,
325 const char *host, uint16_t port)
326{
327 brix_hc_ctx_t *hc;
328 struct addrinfo *res;
329
330 hc = brix_hc_ctx_create(cycle, conf, host, port);
331 if (hc == NULL) {
332 return;
333 }
334
335 /* Resolve target. */
336 res = brix_hc_resolve(hc, host, port);
337 if (res == NULL) {
338 return;
339 }
340
341 if (brix_hc_open_conn(cycle, hc, res, host, port) != NGX_OK) {
342 return;
343 }
344
345 if (brix_hc_queue_bootstrap(hc) != NGX_OK) {
346 freeaddrinfo(res);
347 brix_hc_finish(hc, 0);
348 return;
349 }
350
351 /* Single deadline for the whole probe. */
352 hc->tev.handler = brix_hc_timeout_handler;
353 hc->tev.data = hc;
354 hc->tev.log = cycle->log;
355 ngx_add_timer(&hc->tev, conf->hc.timeout_ms);
356
357 BRIX_HC_METRIC_INC(hc_probes_total);
358
359 brix_hc_begin_connect(hc, res);
360}

362/*
363 * Recurring scan tick. Re-arms itself first (so a probe-launch failure can't
364 * stop future scans), then tries to claim at most one due server from the
365 * shared registry and probe it. Claiming exactly one per tick is what spreads
366 * load and guarantees a single worker probes each server per interval.
367 */
368static void
369brix_hc_timer_handler(ngx_event_t *ev)
370{
371 brix_hc_mgr_t *mgr = ev->data;
372 ngx_stream_brix_srv_conf_t *conf = mgr->conf;
373 char host[256];
374 uint16_t port;
375 ngx_msec_t next_due = conf->hc.interval_ms;
376 ngx_msec_t delay;
377
378 /* Claim is atomic under the registry spinlock and self-rate-limits via
379 * hc_next_check, so concurrent workers never double-probe one server. */
380 if (brix_srv_hc_claim(host, sizeof(host), &port,
381 conf->hc.interval_ms, &next_due))
382 {
383 brix_hc_start(mgr->cycle, conf, host, port);
384 /* More servers may be due now — spread the remaining probes at the
385 * per-slot cadence rather than firing them all this tick. */
386 delay = mgr->scan_interval_ms;
387 } else {
388 /* Nothing due: sleep until the soonest server becomes due instead of
389 * waking at the sub-second floor to find nothing (the old idle drain).
390 * Floored by the spread cadence (burst guard) and capped by the probe
391 * interval (so a freshly-registered server is probed within one). */
392 delay = next_due;
393 if (delay < mgr->scan_interval_ms) {
394 delay = mgr->scan_interval_ms;
395 }
396 if (delay > conf->hc.interval_ms) {
397 delay = conf->hc.interval_ms;
398 }
399 }
400
401 /* Re-arm (bounded to [scan_interval, hc_interval]) — a probe-launch
402 * failure or empty registry can never stop future scans. Suppressed once
403 * the worker is exiting so the scan loop releases the draining worker. */
404 if (!ngx_exiting) {
405 ngx_add_timer(ev, delay);
406 }
407}

409void
410brix_hc_manager_start(ngx_cycle_t *cycle,
411 ngx_stream_brix_srv_conf_t *conf)
412{
413 brix_hc_mgr_t *mgr;
414 ngx_msec_t scan;
415
416 if (!conf->hc.enabled || conf->hc.interval_ms == 0) {
417 return;
418 }
419
420 mgr = ngx_pcalloc(cycle->pool, sizeof(*mgr));
421 if (mgr == NULL) {
422 return;
423 }
424 mgr->cycle = cycle;
425 mgr->conf = conf;
426
427 /* Spread probes across the interval: one claim attempt every
428 * interval/slots ms, floored at 100 ms. */
429 scan = conf->hc.interval_ms / BRIX_SRV_REGISTRY_SLOTS;
430 mgr->scan_interval_ms = scan < 100 ? 100 : scan;
431
432 mgr->timer.handler = brix_hc_timer_handler;
433 mgr->timer.data = mgr;
434 mgr->timer.log = cycle->log;
435
436 ngx_add_timer(&mgr->timer, 2000); /* let CMS connections settle first */
437
438 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
439 "brix: health check manager started "
440 "(interval=%Ms timeout=%Ms scan=%Ms)",
441 conf->hc.interval_ms, conf->hc.timeout_ms,
442 mgr->scan_interval_ms);
443}
