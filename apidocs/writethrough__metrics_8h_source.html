4/*
5 * writethrough_metrics.h — header-only helpers for write-through metric
6 * accounting (the "dirty handle" + flush-queue gauges exported on /metrics).
7 *
8 * WHAT: small ngx_inline helpers that keep two classes of counter consistent:
9 * (1) the per-handle dirty state (wt_dirty_handles gauge) — incremented the
10 * first time a handle goes dirty and decremented exactly once when it is
11 * flushed/closed, and (2) the flush-queue + flush-outcome counters.
12 * WHY: these counters are touched from several call sites (write, flush, close,
13 * abort); centralising the inc/dec logic here keeps a handle from being
14 * double-counted or a gauge from going negative under concurrency.
15 * HOW: gauges use a guarded decrement (never drop below zero); the dirty-handle
16 * transitions key off file->wt_dirty_offset (< 0 == clean) so the gauge
17 * moves only on the clean<->dirty EDGE, not on every write.
18 *
19 * Include after ngx_brix_module.h so brix_ctx_t, brix_file_t, and the
20 * BRIX_* metric macros are already visible.
21 */
22
23/* Race-safe gauge decrement: only decrements when the counter is > 0, so a
24 * double-clean or a lost increment can never drive the gauge negative. */
25static ngx_inline void
26brix_wt_metric_dec_if_positive(ngx_atomic_t *counter)
27{
28 if (counter != NULL && ngx_atomic_fetch_add(counter, 0) > 0) {
29 ngx_atomic_fetch_add(counter, (ngx_atomic_int_t) -1);
30 }
31}

33/* Record that handle `idx` now holds unflushed (dirty) data ending at
34 * dirty_offset. Bumps the wt_dirty_handles gauge ONLY on the clean->dirty edge
35 * (first dirty write of this handle, detected via wt_dirty_offset < 0); later
36 * writes just accumulate wt_bytes_written and advance the offset. No-op unless
37 * write-through is enabled for the handle. */
38static ngx_inline void

40 size_t bytes_written)
41{
42 brix_file_t *file;
43

45 return;
46 }
47

61/* Clear handle `idx`'s dirty state after a successful flush/close. Decrements
62 * the wt_dirty_handles gauge once (guarded) only if the handle was actually
63 * dirty, then resets the offset/byte accounting. Pairs 1:1 with the edge
64 * increment in brix_wt_mark_dirty. */
65static ngx_inline void
66brix_wt_mark_clean(brix_ctx_t *ctx, int idx)
67{
68 brix_file_t *file;
69
70 if (ctx == NULL || idx < 0 || idx >= BRIX_MAX_FILES) {
71 return;
72 }
73
74 file = &ctx->files[idx];
75 if (file->wt_dirty_offset >= 0 && ctx->metrics != NULL) {
76 brix_wt_metric_dec_if_positive(&ctx->metrics->wt_dirty_handles);
77 }
78
79 file->wt_dirty_offset = -1;
80 file->wt_bytes_written = 0;
81}
