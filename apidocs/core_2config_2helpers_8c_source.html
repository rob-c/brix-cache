8/* Validate a configured path at startup: existence (stat), type
9 * (file/dir/either), and access mode (R/W/X_OK). A NULL/empty path is
10 * optional and skipped. Returns NGX_OK, or NGX_ERROR with an emerg log
11 * naming the failure. */
12ngx_int_t
13brix_validate_path(ngx_conf_t *cf, const char *label, const ngx_str_t *path,
14 brix_path_kind_t kind, int access_mode)
15{
16 struct stat st;
17
18 if (path == NULL || path->len == 0 || path->data == NULL) {
19 return NGX_OK;
20 }
21
22 if (stat((char *) path->data, &st) != 0) {
23 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
24 "brix: %s path \"%s\" is not accessible",
25 label, path->data);
26 return NGX_ERROR;
27 }
28
29 switch (kind) {
30 case BRIX_PATH_REGULAR_FILE:
31 if (!S_ISREG(st.st_mode)) {
32 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
33 "brix: %s path \"%s\" must be a regular file",
34 label, path->data);
35 return NGX_ERROR;
36 }
37 break;
38
39 case BRIX_PATH_DIRECTORY:
40 if (!S_ISDIR(st.st_mode)) {
41 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
42 "brix: %s path \"%s\" must be a directory",
43 label, path->data);
44 return NGX_ERROR;
45 }
46 break;
47
48 case BRIX_PATH_FILE_OR_DIRECTORY:
49 if (!S_ISREG(st.st_mode) && !S_ISDIR(st.st_mode)) {
50 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
51 "brix: %s path \"%s\" must be a file or directory",
52 label, path->data);
53 return NGX_ERROR;
54 }
55 break;
56 }
57
58 if (access_mode != 0 && access((char *) path->data, access_mode) != 0) {
59 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
60 "brix: %s path \"%s\" failed permission check",
61 label, path->data);
62 return NGX_ERROR;
63 }
64
65 return NGX_OK;
66}

68/* Copy an ngx_str_t into a NUL-terminated C string in cf->pool (ngx_str_t is
69 * not NUL-terminated, so strtol/strchr/etc. need this). Returns NGX_CONF_OK,
70 * or NGX_CONF_ERROR on allocation failure. */
71char *
72brix_copy_conf_string(ngx_conf_t *cf, const ngx_str_t *src, ngx_str_t *dst)
73{
74 dst->data = ngx_pnalloc(cf->pool, src->len + 1);
75 if (dst->data == NULL) {
76 return NGX_CONF_ERROR;
77 }
78
79 ngx_memcpy(dst->data, src->data, src->len);
80 dst->data[src->len] = '\0';
81 dst->len = src->len;
82 return NGX_CONF_OK;
83}
