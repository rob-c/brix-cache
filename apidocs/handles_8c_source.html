1/*
2 * WHAT: This file implements the shared memory handle table used by kXR_bind
3 * secondary connections. Primary connections publish readable file handle
4 * metadata so another nginx worker can reopen and validate the same file.
5 *
6 * WHY: nginx workers do not share descriptors opened after fork. Bound
7 * streams therefore cannot inherit a raw fd from the primary connection; they
8 * need a shared, validated description of the handle instead.
9 *
10 * HOW: All operations are serialized by the handle-zone mutex. Entries are
11 * keyed by sessid + handle_index and carry path, readable/writable flags,
12 * cache status, device, inode, and size metadata for reopen validation.
13 */

34ngx_int_t
35brix_handle_shm_init_zone(ngx_shm_zone_t *shm_zone, void *data)
36{
37 ngx_flag_t fresh;
38 brix_shared_handle_table_t *tbl;
39
40 /*
41 * Allocate the table FROM the slab pool (not over shm.addr) so nginx's
42 * ngx_unlock_mutexes() — which treats shm.addr as an ngx_slab_pool_t and
43 * force-unlocks its mutex on every child death — finds an intact slab
44 * header instead of our clobbered struct. brix_shm_table_alloc() also
45 * zeroes the table and creates the process-local mutex from the table's
46 * first member (the ngx_shmtx_sh_t lock).
47 */
48 tbl = brix_shm_table_alloc(shm_zone, data,
49 sizeof(brix_shared_handle_table_t),
50 &brix_handle_mutex, &fresh);
51 if (tbl == NULL) {
52 return NGX_ERROR;
53 }
54
55 /*
56 * The fixed-size slots[] array has no extra fields to initialise beyond the
57 * zeroing the helper already performed on a fresh allocation, so there is
58 * nothing more to do here (live state must be preserved on reuse).
59 */
60 (void) fresh;
61 return NGX_OK;
62}

73/*
74 *
75 * WHAT: Shares file handle metadata with other workers enabling bound stream
76 * secondary connections to read primary-published handles. Write-only
77 * handles are not published because bound streams are read-only data channels.
78 *
79 * WHY: xrdcp secondary connections established via kXR_bind may arrive at
80 * different workers than the primary connection. Device/inode metadata lets
81 * the secondary verify the reopened path still refers to the original file.
82 */
83void
84brix_session_handle_publish(const u_char sessid[BRIX_SESSION_ID_LEN],
85 int handle_index, const brix_file_t *file)
86{
87 brix_shared_handle_table_t *tbl;
88 brix_shared_handle_entry_t *entry;
89 ngx_uint_t i, free_slot;
90 size_t path_len;
91
92 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES
93 || file == NULL || file->fd < 0)
94 {
95 return;
96 }
97
98 tbl = handle_table();
99 if (tbl == NULL) {
100 return;
101 }
102
103 ngx_shmtx_lock(&brix_handle_mutex);
104
105 free_slot = BRIX_SESSION_HANDLE_SLOTS;
106 entry = NULL;
107
108 for (i = 0; i < BRIX_SESSION_HANDLE_SLOTS; i++) {
109 if (brix_shared_handle_same_key(&tbl->slots[i], sessid,
110 handle_index))
111 {
112 entry = &tbl->slots[i];
113 break;
114 }
115
116 if (!tbl->slots[i].in_use
117 && free_slot == BRIX_SESSION_HANDLE_SLOTS)
118 {
119 free_slot = i;
120 }
121 }
122
123 /*
124 * Bound streams are read-only data channels. Publishing a write-only
125 * primary handle would create an attractive misuse path, so treat it as
126 * removal of any stale shared entry for the same slot.
127 */
128 if (!file->readable || file->path == NULL) {
129 if (entry != NULL) {
130 ngx_memzero(entry, sizeof(*entry));
131 }
132 ngx_shmtx_unlock(&brix_handle_mutex);
133 return;
134 }
135
136 path_len = ngx_strlen(file->path);
137 if (path_len > BRIX_MAX_PATH) {
138 if (entry != NULL) {
139 ngx_memzero(entry, sizeof(*entry));
140 }
141 ngx_shmtx_unlock(&brix_handle_mutex);
142 return;
143 }
144
145 if (entry == NULL) {
146 if (free_slot == BRIX_SESSION_HANDLE_SLOTS) {
147 ngx_shmtx_unlock(&brix_handle_mutex);
148 return;
149 }
150 entry = &tbl->slots[free_slot];
151 }
152
153 ngx_memzero(entry, sizeof(*entry));
154 ngx_memcpy(entry->sessid, sessid, BRIX_SESSION_ID_LEN);
155 entry->handle_index = (uint8_t) handle_index;
156 entry->readable = file->readable ? 1 : 0;
157 entry->writable = file->writable ? 1 : 0;
158 entry->from_cache = file->from_cache ? 1 : 0;
159 entry->is_regular = file->is_regular ? 1 : 0;
160 entry->device = file->device;
161 entry->inode = file->inode;
162 entry->cached_size = file->cached_size;
163 ngx_cpystrn((u_char *) entry->path, (u_char *) file->path,
164 sizeof(entry->path));
165 entry->in_use = 1;
166
167 ngx_shmtx_unlock(&brix_handle_mutex);
168}

170/*
171 *
172 * WHAT: Retrieves published handle metadata for bound stream read requests.
173 *
174 * WHY: Bound streams can reopen a primary-published handle in their own worker
175 * and validate path identity against the stored device/inode tuple.
176 */
177/*
178 *
179 * WHAT: Same as brix_session_handle_lookup() but with a caller-owned slot
180 * hint (Phase 33 C2). A bound secondary re-validates its published handle on
181 * EVERY read; *inout_slot remembers the slot matched last time so the common
182 * case becomes an O(1) direct check instead of a full table scan.
183 *
184 * WHY: The scan is the per-read cost on parallel bound streams (xrdcp
185 * --sources). Correctness is unchanged: the lock is still held and the FULL
186 * key (in_use + sessid + handle_index) is re-checked, so a primary close/reuse
187 * (which clears in_use) fails the fast path and falls through to the scan →
188 * revocation, exactly as before. *inout_slot is refreshed on a scan match and
189 * reset to -1 on a miss (and may be NULL for callers that don't cache).
190 */
191int
192brix_session_handle_lookup_hint(const u_char sessid[BRIX_SESSION_ID_LEN],
193 int handle_index, int *inout_slot, brix_shared_handle_entry_t *out)
194{
195 brix_shared_handle_table_t *tbl;
196 ngx_uint_t i;
197 int found = 0;
198
199 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES || out == NULL) {
200 return 0;
201 }
202
203 tbl = handle_table();
204 if (tbl == NULL) {
205 return 0;
206 }
207
208 ngx_shmtx_lock(&brix_handle_mutex);
209
210 /*
211 * Fast path: re-check the slot matched on the previous read directly. Still
212 * under the lock and still verifying the full key, so this is a pure scan
213 * elimination — never a weaker check.
214 */
215 if (inout_slot != NULL && *inout_slot >= 0
216 && (ngx_uint_t) *inout_slot < BRIX_SESSION_HANDLE_SLOTS
217 && brix_shared_handle_same_key(&tbl->slots[*inout_slot], sessid,
218 handle_index))
219 {
220 ngx_memcpy(out, &tbl->slots[*inout_slot], sizeof(*out));
221 out->path[BRIX_MAX_PATH] = '\0';
222 ngx_shmtx_unlock(&brix_handle_mutex);
223 return 1;
224 }
225
226 for (i = 0; i < BRIX_SESSION_HANDLE_SLOTS; i++) {
227 if (brix_shared_handle_same_key(&tbl->slots[i], sessid,
228 handle_index))
229 {
230 ngx_memcpy(out, &tbl->slots[i], sizeof(*out));
231 out->path[BRIX_MAX_PATH] = '\0';
232 if (inout_slot != NULL) {
233 *inout_slot = (int) i; /* refresh the hint for next read */
234 }
235 found = 1;
236 break;
237 }
238 }
239
240 if (!found && inout_slot != NULL) {
241 *inout_slot = -1; /* drop a now-stale hint so the next read rescans */
242 }
243
244 ngx_shmtx_unlock(&brix_handle_mutex);
245 return found;
246}

256 *
257 * WHAT: Removes one published handle entry during kXR_close.
258 *
259 * WHY: Secondary connections must not continue reading a primary handle after
260 * that handle has been closed or reused.
261 */
262void
263brix_session_handle_unpublish(const u_char sessid[BRIX_SESSION_ID_LEN],
264 int handle_index)
265{
266 brix_shared_handle_table_t *tbl;
267 ngx_uint_t i;
268
269 if (handle_index < 0 || handle_index >= BRIX_MAX_FILES) {
270 return;
271 }
272
273 tbl = handle_table();
274 if (tbl == NULL) {
275 return;
276 }
277
278 ngx_shmtx_lock(&brix_handle_mutex);
279
280 for (i = 0; i < BRIX_SESSION_HANDLE_SLOTS; i++) {
281 if (brix_shared_handle_same_key(&tbl->slots[i], sessid,
282 handle_index))
283 {
284 ngx_memzero(&tbl->slots[i], sizeof(tbl->slots[i]));
285 break;
286 }
287 }
288
289 ngx_shmtx_unlock(&brix_handle_mutex);
290}
