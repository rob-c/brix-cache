72int
73s3_list_cache_get(ngx_http_request_t *r, const char *root,
74 const char *prefix, const char *delimiter, time_t dir_mtime,
75 ngx_msec_t ttl_ms, s3_entry_t **out_items, int *out_n)
76{
77 s3_lc_slot_t *s = NULL;
78 s3_entry_t *items;
79 int i;
80
81 for (i = 0; i < S3_LIST_CACHE_SLOTS; i++) {
82 if (s3_lc_matches(&s3_lc_slots[i], root, prefix, delimiter)) {
83 s = &s3_lc_slots[i];
84 break;
85 }
86 }
87 if (s == NULL) {
88 return 0; /* not cached */
89 }
90
91 /* Invalidate on a top-level mutation or TTL expiry. */
92 if (s->dir_mtime != dir_mtime
93 || (ngx_msec_t) (ngx_current_msec - s->stored_at) >= ttl_ms)
94 {
95 s3_lc_slot_clear(s);
96 return 0;
97 }
98
99 /* Hit: materialise an s3_entry_t array in the request pool, duplicating the
100 * keys (stat fields left zero for the lazy per-page s3_entry_fill_stat). */
101 items = ngx_pcalloc(r->pool, sizeof(s3_entry_t) * (size_t) s->n);
102 if (items == NULL) {
103 return 0; /* fall back to a walk */
104 }
105 for (i = 0; i < s->n; i++) {
106 size_t klen = strlen(s->entries[i].key);
107 char *kp = ngx_pnalloc(r->pool, klen + 1);
108 if (kp == NULL) {
109 return 0;
110 }
111 ngx_memcpy(kp, s->entries[i].key, klen + 1);
112 items[i].key = kp;
113 items[i].is_prefix = s->entries[i].is_prefix;
114 }
115
116 s->seq = ++s3_lc_seq; /* mark most-recently-used */
117 *out_items = items;
118 *out_n = s->n;
119 return 1;
120}

122void
123s3_list_cache_put(ngx_log_t *log, const char *root, const char *prefix,
124 const char *delimiter, time_t dir_mtime, const s3_entry_t *items, int n)
125{
126 s3_lc_slot_t *victim = NULL;
127 s3_lc_entry_t *copy;
128 uint64_t min_seq = (uint64_t) -1;
129 int i;
130
131 if (n > S3_LIST_CACHE_MAX_ENTRIES) {
132 static int warned = 0;
133 if (!warned) {
134 warned = 1;
135 ngx_log_error(NGX_LOG_INFO, log, 0,
136 "s3 list-cache: %d entries exceeds cap %d; "
137 "this listing is not cached", n,
138 S3_LIST_CACHE_MAX_ENTRIES);
139 }
140 return;
141 }
142
143 /* Reuse a matching slot, else a free slot, else evict the LRU. */
144 for (i = 0; i < S3_LIST_CACHE_SLOTS; i++) {
145 if (s3_lc_matches(&s3_lc_slots[i], root, prefix, delimiter)) {
146 victim = &s3_lc_slots[i];
147 break;
148 }
149 if (!s3_lc_slots[i].used) {
150 victim = &s3_lc_slots[i]; /* prefer a free slot */
151 } else if (victim == NULL || !victim->used) {
152 /* keep the best free candidate; otherwise track LRU below */
153 }
154 }
155 if (victim == NULL || victim->used) {
156 for (i = 0; i < S3_LIST_CACHE_SLOTS; i++) {
157 if (s3_lc_slots[i].used && s3_lc_slots[i].seq < min_seq) {
158 min_seq = s3_lc_slots[i].seq;
159 victim = &s3_lc_slots[i];
160 }
161 }
162 }
163 if (victim == NULL) {
164 return;
165 }
166
167 copy = ngx_alloc(sizeof(s3_lc_entry_t) * (size_t) n, log);
168 if (copy == NULL) {
169 return;
170 }
171 for (i = 0; i < n; i++) {
172 copy[i].key = s3_lc_strdup(items[i].key);
173 copy[i].is_prefix = items[i].is_prefix;
174 if (copy[i].key == NULL) { /* OOM: unwind + bail */
175 while (--i >= 0) {
176 ngx_free(copy[i].key);
177 }
178 ngx_free(copy);
179 return;
180 }
181 }
182
183 s3_lc_slot_clear(victim); /* free any prior content */
184 victim->root = s3_lc_strdup(root);
185 victim->prefix = s3_lc_strdup(prefix);
186 victim->delimiter = s3_lc_strdup(delimiter);
187 if (victim->root == NULL || victim->prefix == NULL
188 || victim->delimiter == NULL)
189 {
190 for (i = 0; i < n; i++) {
191 ngx_free(copy[i].key);
192 }
193 ngx_free(copy);
194 s3_lc_slot_clear(victim);
195 return;
196 }
197 victim->dir_mtime = dir_mtime;
198 victim->stored_at = ngx_current_msec;
199 victim->entries = copy;
200 victim->n = n;
201 victim->seq = ++s3_lc_seq;
202 victim->used = 1;
203}
