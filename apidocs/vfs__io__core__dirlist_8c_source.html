1/*
2 * vfs_io_core_dirlist.c — worker-safe kXR_dirlist OPENDIR response builder.
3 *
4 * WHAT: Implements brix_vfs_io_execute_opendir(), the OPENDIR arm of the
5 * POD-only VFS I/O execution core. It fdopendir's a dup of the confined
6 * rootfd and builds the complete flat kXR_dirlist wire response (optional
7 * dstat lead-in, filtered entries with fd-relative fstat/checksum, chunked
8 * kXR_oksofar headers, final kXR_ok) straight into the job's caller-owned
9 * buffer, mutating only the job OUT fields.
10 *
11 * WHY: Split out of vfs_io_core.c for the file-size guard. The dirlist scan and
12 * wire-framing helpers form one self-contained concept — the only I/O op
13 * that emits a protocol response body — so they live in their own unit and
14 * share brix_vfs_io_set_error_message / the dispatch prototype with the
15 * core via vfs_io_core_internal.h.
16 *
17 * HOW: The dispatcher in vfs_io_core.c calls brix_vfs_io_execute_opendir(); all
18 * other helpers here are file-local statics using only thread-safe
19 * primitives, raw syscalls, and the job's POD buffers.
20 */

46/* brix_vfs_io_dirlist_fail — record an OPENDIR failure (io_errno=err, nio=-1)

59/* brix_vfs_io_dirlist_name_unsafe — 1 if a dir entry must be skipped: "."/".."
60 * , the ".nginx-xrootd" control-file prefix, or any control/DEL byte (the
61 * newline-delimited kXR_dirlist body would otherwise be corrupted/polluted). */
62static ngx_flag_t
63brix_vfs_io_dirlist_name_unsafe(const char *name)
64{
65 const u_char *p;
66
67 if (name[0] == '.' && (name[1] == '\0'
68 || (name[1] == '.' && name[2] == '\0')))
69 {
70 return 1;
71 }
72
73 if (ngx_strncmp(name, ".nginx-xrootd",
74 sizeof(".nginx-xrootd") - 1) == 0)
75 {
76 return 1;
77 }
78
79 for (p = (const u_char *) name; *p != '\0'; p++) {
80 if (*p < 0x20 || *p == 0x7f) {
81 return 1;
82 }
83 }
84
85 return 0;
86}

99/* brix_vfs_io_dirlist_need_new_chunk — if the next entry won't fit the current
100 * chunk, flush a kXR_oksofar chunk and reset the data cursor; fails E2BIG when one
101 * more full chunk would exceed job->buf_cap (preserving the old fixed-buffer bound). */
102static ngx_int_t
103brix_vfs_io_dirlist_need_new_chunk(brix_vfs_job_t *job, u_char *out,
104 size_t *base, u_char **cdata, size_t *cdpos, size_t need)
105{
106 if (*cdpos + need <= BRIX_VFS_DIRLIST_CHUNK_CAP) {
107 return NGX_OK;
108 }
109
110 brix_vfs_io_dirlist_flush_chunk(job, out, base, *cdpos);
111 if (*base + XRD_RESPONSE_HDR_LEN + BRIX_VFS_DIRLIST_CHUNK_CAP
112 > job->buf_cap)
113 {
114 char message[96];
115
116 snprintf(message, sizeof(message),
117 "listing too large for AIO buffer (%zu bytes)",
118 job->buf_cap);
119 brix_vfs_io_dirlist_fail(job, E2BIG, message);
120 return NGX_ERROR;
121 }
122
123 *cdata = out + *base + XRD_RESPONSE_HDR_LEN;
124 *cdpos = 0;
125 return NGX_OK;
126}

128/* brix_vfs_io_dirlist_stat_entry — for job->want_stat, fstatat the entry
129 * (no-follow) and format the stat body plus optional checksum token, adding the
130 * extra wire bytes to *need; NGX_DECLINED skips a vanished/unstattable entry. */
131static ngx_int_t
132brix_vfs_io_dirlist_stat_entry(brix_vfs_job_t *job, int dfd,
133 const char *name, char *statbuf, size_t statbuf_cap,
134 char *cksum_token, size_t cksum_token_cap, size_t *need)
135{
136 struct stat entry_st;
137
138 statbuf[0] = '\0';
139 cksum_token[0] = '\0';
140
141 if (!job->want_stat) {
142 return NGX_OK;
143 }
144
145 /* dfd < 0 (a failed dirfd upstream) and any fstatat failure — vanished
146 * entry or otherwise — take the same path: skip the entry. */
147 if (dfd < 0 || fstatat(dfd, name, &entry_st, AT_SYMLINK_NOFOLLOW) != 0) {
148 return NGX_DECLINED;
149 }
150
151 if (job->want_cksum) {
152 char entry_path[PATH_MAX];
153 const char *algo;
154 int n;
155
156 brix_dirlist_make_dcksm_stat_body(&entry_st, statbuf, statbuf_cap);
157 algo = job->cksum_algo != NULL ? job->cksum_algo : "unknown";
158 n = snprintf(entry_path, sizeof(entry_path), "%s/%s",
159 job->path != NULL ? job->path : "", name);
160 if (n < 0 || (size_t) n >= sizeof(entry_path)) {
161 snprintf(cksum_token, cksum_token_cap, "%s:none", algo);
162 } else {
163 brix_dirlist_checksum_token(job->log, dfd, name, entry_path,
164 &entry_st, algo, cksum_token,
165 cksum_token_cap);
166 }
167 *need += strlen(cksum_token) + sizeof(" [ ]") - 1;
168
169 } else {
170 brix_make_stat_body(&entry_st, 0, 0, statbuf, statbuf_cap);
171 }
172
173 *need += strlen(statbuf) + 1;
174 return NGX_OK;
175}

177/* brix_vfs_io_dirlist_emit_entry — append the pre-counted name, optional stat
178 * body, and optional checksum token into the chunk, advancing *cdpos (the byte
179 * appends live in one helper so capacity invariants are easy to audit). */
180static void
181brix_vfs_io_dirlist_emit_entry(brix_vfs_job_t *job, u_char *cdata,
182 size_t *cdpos, const char *name, size_t nlen, const char *statbuf,
183 const char *cksum_token)
184{
185 ngx_memcpy(cdata + *cdpos, name, nlen);
186 *cdpos += nlen;
187 cdata[(*cdpos)++] = '\n';
188
189 if (job->want_stat) {
190 size_t slen;
191
192 slen = strlen(statbuf);
193 ngx_memcpy(cdata + *cdpos, statbuf, slen);
194 *cdpos += slen;
195
196 if (job->want_cksum) {
197 int n;
198
199 n = snprintf((char *) (cdata + *cdpos),
200 BRIX_VFS_DIRLIST_CHUNK_CAP - *cdpos,
201 " [ %s ]", cksum_token);
202 if (n > 0) {
203 *cdpos += (size_t) n;
204 }
205 }
206
207 cdata[(*cdpos)++] = '\n';
208 }
209}

211/* brix_vfs_io_execute_opendir — build the complete kXR_dirlist flat response in
212 * job->buf by fdopendir'ing a dup of the confined rootfd (so the scan stays
213 * anchored to the directory approved on the event loop, not reopened by path):
214 * optional dstat lead-in, then filtered entries with fd-relative fstat/checksum,
215 * then the final kXR_ok header. */
216void
217brix_vfs_io_execute_opendir(brix_vfs_job_t *job)
218{
219 DIR *dp;
220 int scanfd;
221 u_char *out;
222 u_char *cdata;
223 size_t base;
224 size_t cdpos;
225
226 if (job->rootfd < 0 || job->buf == NULL) {
227 brix_vfs_io_dirlist_fail(job, EINVAL, "invalid opendir job");
228 return;
229 }
230
231 if (job->buf_cap < (size_t) (XRD_RESPONSE_HDR_LEN
232 + BRIX_VFS_DIRLIST_CHUNK_CAP))
233 {
234 brix_vfs_io_dirlist_fail(job, ENOMEM, "response buffer too small");
235 return;
236 }
237
238 scanfd = dup(job->rootfd);
239 if (scanfd < 0) {
240 brix_vfs_io_dirlist_fail(job, errno, NULL);
241 return;
242 }
243
244 dp = fdopendir(scanfd);
245 if (dp == NULL) {
246 int err;
247
248 err = errno;
249 close(scanfd);
250 brix_vfs_io_dirlist_fail(job, err, NULL);
251 return;
252 }
253
254 out = job->buf;
255 base = 0;
256 cdpos = 0;
257 cdata = out + XRD_RESPONSE_HDR_LEN;
258
259 if (job->want_stat) {
260 ngx_memcpy(cdata, BRIX_DSTAT_LEADIN, BRIX_DSTAT_LEADIN_LEN);
261 cdpos = BRIX_DSTAT_LEADIN_LEN;
262 }
263
264 for (;;) {
265 struct dirent *de;
266 const char *name;
267 size_t nlen;
268 size_t need;
269 char statbuf[256];
270 char cksum_token[EVP_MAX_MD_SIZE * 2 + 64];
271
272 errno = 0;
273 de = readdir(dp);
274 if (de == NULL) {
275 if (errno != 0) {
276 int err = errno;
277
278 closedir(dp);
279 brix_vfs_io_dirlist_fail(job, err, NULL);
280 return;
281 }
282 break;
283 }
284
285 name = de->d_name;
286 if (brix_vfs_io_dirlist_name_unsafe(name)) {
287 continue;
288 }
289
290 nlen = strlen(name);
291 need = nlen + 1;
292 if (brix_vfs_io_dirlist_stat_entry(job, dirfd(dp), name,
293 statbuf, sizeof(statbuf),
294 cksum_token, sizeof(cksum_token),
295 &need)
296 == NGX_DECLINED)
297 {
298 continue;
299 }
300
301 if (brix_vfs_io_dirlist_need_new_chunk(job, out, &base, &cdata,
302 &cdpos, need)
303 != NGX_OK)
304 {
305 closedir(dp);
306 return;
307 }
308
309 brix_vfs_io_dirlist_emit_entry(job, cdata, &cdpos, name, nlen,
310 statbuf, cksum_token);
311 }
312
313 closedir(dp);
314
315 if (cdpos > 0) {
316 cdata[cdpos - 1] = '\0';
317 }
318 brix_build_resp_hdr(job->streamid, kXR_ok, (uint32_t) cdpos,
319 (ServerResponseHdr *) (out + base));
320 job->out_size = base + XRD_RESPONSE_HDR_LEN + cdpos;
321 job->nio = 0;
322}
