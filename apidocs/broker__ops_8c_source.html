18/* openat2(rootfd, rel, RESOLVE_BENEATH). Returns fd or -errno. */
19int
20imp_openat2(int rootfd, const char *rel, uint32_t flags, uint32_t mode)
21{
22 struct open_how how;
23 long fd;
24
25 ngx_memzero(&how, sizeof(how));
26 how.flags = flags | O_CLOEXEC;
27 /*
28 * openat2() is stricter than open()/openat(): it rejects (EINVAL) any
29 * how.mode bit outside 07777. Callers legitimately pass a full struct
30 * stat st_mode (e.g. staged_file copying a source's permissions during a
31 * WebDAV/S3 COPY), which carries the S_IFMT type bits. Mask to the
32 * permission bits, exactly as the worker-local do_openat2() in
33 * src/path/beneath.c does, so a struct-stat mode is accepted instead of
34 * failing the whole impersonated COPY with EINVAL.
35 */
36 how.mode = (flags & O_CREAT) ? (mode & 07777) : 0;
37 how.resolve = RESOLVE_BENEATH | RESOLVE_NO_MAGICLINKS;
38
39 fd = syscall(SYS_openat2, rootfd, rel, &how, sizeof(how));
40 return (fd < 0) ? -errno : (int) fd;
41}

44/*
45 * Open the PARENT directory of `rel` beneath rootfd (O_PATH|O_DIRECTORY) and
46 * point *base at the final path component. Returns the parent fd or -errno;
47 * rejects a trailing "." / ".." / empty base. `scratch` (>= IMP_PATH_MAX) holds
48 * the mutated copy that *base points into.
49 */
50int
51imp_open_parent(int rootfd, const char *rel, char *scratch, const char **base)
52{
53 char *slash;
54 int pfd;
55
56 if (ngx_strlen(rel) >= IMP_PATH_MAX) {
57 return -ENAMETOOLONG;
58 }
59 ngx_memcpy(scratch, rel, ngx_strlen(rel) + 1);
60
61 slash = strrchr(scratch, '/');
62 if (slash == NULL) {
63 *base = scratch; /* parent is rootfd itself */
64 pfd = imp_openat2(rootfd, ".", O_PATH | O_DIRECTORY, 0);
65 } else {
66 *slash = '\0';
67 *base = slash + 1;
68 pfd = imp_openat2(rootfd, imp_rel(scratch), O_PATH | O_DIRECTORY, 0);
69 }
70
71 if ((*base)[0] == '\0'
72 || ngx_strcmp(*base, ".") == 0 || ngx_strcmp(*base, "..") == 0)
73 {
74 if (pfd >= 0) { close(pfd); }
75 return -EINVAL;
76 }
77 return pfd; /* fd or -errno from imp_openat2 */
78}

101/*
102 * imp_xattr_open — open the xattr target file beneath rootfd as the (already
103 * impersonated) mapped user for an f*xattr op. O_RDONLY|O_NONBLOCK is enough:
104 * f{get,set,remove,list}xattr check inode permission against the caller's
105 * (impersonated) creds at the call, independent of the fd's open mode, while the
106 * RESOLVE_BENEATH open both confines the path and enforces the mapped user's DAC
107 * to even reach the file. O_NONBLOCK avoids blocking on a FIFO. Returns fd or
108 * -errno. (Rare write-only files — mode 0200 — cannot be opened O_RDONLY; such
109 * a SETXATTR fails EACCES, an acceptable corner the path-based fallback avoided.)
110 */
111int
112imp_xattr_open(int rootfd, const char *rel)
113{
114 return imp_openat2(rootfd, rel, O_RDONLY | O_NONBLOCK, 0);
115}

118/*
119 * imp_xattr_name_ok — the broker only services the `user.` xattr namespace (all
120 * the module's xattr users live there: locks, dead properties, checksum cache).
121 * Refusing every other namespace is defence-in-depth: it denies any attempt to
122 * drive the broker into setting security.* / system.* / trusted.* attributes,
123 * independent of what the (unprivileged) mapped user's own creds would allow.

125int
126imp_xattr_name_ok(const char *name)
127{
128 return ngx_strncmp(name, "user.", 5) == 0;

132/*
133 * imp_xattr_filter_user — restrict a flistxattr(2) result (NUL-separated attribute
134 * names, total `len` bytes including terminators) to the `user.` namespace,
135 * repacking the kept names densely in place and returning the new length. Without
136 * this the broker would hand the worker the NAMES of every attribute on the file,
137 * including system.* / security.* / trusted.* set by root — an information leak
138 * (the worker cannot read their VALUES, but it learns they exist). Matches the
139 * same user.-only policy the per-name ops enforce via imp_xattr_name_ok().
140 */
141size_t
142imp_xattr_filter_user(char *list, size_t len)
143{
144 size_t in = 0, out = 0;
145
146 while (in < len) {
147 size_t entry = 0;
148 while (in + entry < len && list[in + entry] != '\0') {
149 entry++;
150 }
151 if (in + entry >= len) {
152 break; /* no NUL terminator: malformed tail */
153 }
154 entry++; /* include the NUL */
155 if (imp_xattr_name_ok(list + in)) {
156 if (out != in) {
157 ngx_memmove(list + out, list + in, entry);
158 }
159 out += entry;
160 }
161 in += entry;
162 }
163 return out;
164}

168/*
169 * renameat, optionally with RENAME_NOREPLACE (atomic create-if-absent). When
170 * `noreplace` is set and the kernel/filesystem lacks RENAME_NOREPLACE
171 * (ENOSYS/EINVAL) it falls back to a plain renameat so behaviour degrades to the
172 * legacy last-writer-wins rather than spuriously failing; on a modern kernel
173 * (>=3.15) the exclusive path is taken and a pre-existing dst yields EEXIST.
174 * Returns 0 on success, -1 with errno set.
175 */
176int
177imp_do_rename(int sfd, const char *sbase, int dfd, const char *dbase,
178 int noreplace)
179{
180 if (!noreplace) {
181 /* phase72-fp: sfd/sbase ARE the old (source) pair — order is correct */
182 return renameat(sfd, sbase, dfd, dbase); /* NOLINT(readability-suspicious-call-argument) */
183 }
184 if (syscall(SYS_renameat2, sfd, sbase, dfd, dbase,
185 (unsigned int) RENAME_NOREPLACE) == 0) {
186 return 0;
187 }
188 if (errno == ENOSYS || errno == EINVAL) {
189 /* phase72-fp: sfd/sbase ARE the old (source) pair — order is correct */
190 return renameat(sfd, sbase, dfd, dbase); /* NOLINT(readability-suspicious-call-argument) */
191 }
192 return -1;
193}

196/*
197 * imp_op_open — IMP_OP_OPEN: openat2(RESOLVE_BENEATH) and hand the fd back.
198 * WHY the O_NONBLOCK dance: a FIFO opened O_RDONLY blocks until a writer
199 * arrives, and a device node can block in its open method — either would wedge
200 * this single broker process, denying service to *every* worker/tenant (a
201 * cross-tenant DoS) since one bad path in the export hangs the whole
202 * impersonation channel. With O_NONBLOCK the open returns immediately for
203 * every file type, then we inspect what we actually got and only hand back
204 * regular files and directories — the only types the gateway data plane ever
205 * serves. HOW: open O_NONBLOCK -> fstat -> type gate -> clear O_NONBLOCK.
206 */
207static int
208imp_op_open(const imp_op_ctx_t *c)
209{
210 struct stat st;
211 int fd, fl, rc;
212
213 fd = imp_openat2(c->rootfd, c->rel, c->req->open_flags | O_NONBLOCK,
214 c->req->mode);
215 if (fd < 0) {
216 return fd;
217 }
218 if (fstat(fd, &st) != 0) {
219 rc = -errno;
220 close(fd);
221 return rc;
222 }
223 /*
224 * Reject FIFOs, sockets, and character/block devices: the gateway only
225 * reads/writes regular files and enumerates directories. Handing the
226 * worker a device or pipe fd would both leak a non-file capability and
227 * (for the data plane's blocking pread/pwrite/sendfile) reintroduce the
228 * very stall we just avoided. Fail closed with EOPNOTSUPP.
229 */
230 if (!S_ISREG(st.st_mode) && !S_ISDIR(st.st_mode)) {
231 close(fd);
232 return -EOPNOTSUPP;
233 }
234 /*
235 * Restore blocking semantics for the returned fd: O_NONBLOCK was only
236 * needed to survive the open(2); the worker's data plane expects a
237 * blocking regular-file/dir fd (it has no EAGAIN retry path). On a
238 * regular file/dir O_NONBLOCK is otherwise a no-op, so clearing it is
239 * always safe.
240 */
241 fl = fcntl(fd, F_GETFL);
242 if (fl != -1) {
243 (void) fcntl(fd, F_SETFL, fl & ~O_NONBLOCK);
244 }
245 *c->out_fd = fd;
246 c->rep->rep_flags |= IMP_REP_HAS_FD;
247 return 0;
248}

251/*
252 * imp_op_stat — IMP_OP_STAT / IMP_OP_LSTAT: confined stat into rep->st.
253 * WHY via O_PATH + AT_EMPTY_PATH: the RESOLVE_BENEATH open both confines the
254 * path and (for LSTAT, O_NOFOLLOW) pins the symlink itself. HOW: openat2 ->
255 * fstatat("", AT_EMPTY_PATH) -> imp_fill_stat.
256 */
257static int
258imp_op_stat(const imp_op_ctx_t *c)
259{
260 struct stat st;
261 int fd, rc;
262
263 fd = imp_openat2(c->rootfd, c->rel,
264 O_PATH | (c->req->op == IMP_OP_LSTAT ? O_NOFOLLOW : 0), 0);
265 if (fd < 0) {
266 return fd;
267 }
268 rc = fstatat(fd, "", &st, AT_EMPTY_PATH) == 0 ? 0 : -errno;
269 close(fd);
270 if (rc != 0) {
271 return rc;
272 }
273 imp_fill_stat(&c->rep->st, &st);
274 c->rep->rep_flags |= IMP_REP_HAS_STAT;
275 return 0;
276}

299/*
300 * imp_op_xattr_read — IMP_OP_GETXATTR / IMP_OP_LISTXATTR: read xattr value or
301 * name list into the trailing reply payload. WHY the two gates: GETXATTR is
302 * user.-namespace-only per name (imp_xattr_name_ok), and LISTXATTR withholds
303 * non-user.* attribute NAMES from the worker (imp_xattr_filter_user) — the
304 * per-name GET/SET/REMOVE ops are already user.-only.
305 */
306static int
307imp_op_xattr_read(const imp_op_ctx_t *c)
308{
309 int fd;
310 ssize_t n;
311
312 if (c->req->op == IMP_OP_GETXATTR && !imp_xattr_name_ok(c->req->path2)) {
313 return -EPERM;
314 }
315 fd = imp_xattr_open(c->rootfd, c->rel);
316 if (fd < 0) {
317 return fd;
318 }
319 if (c->req->op == IMP_OP_GETXATTR) {
320 n = fgetxattr(fd, c->req->path2, c->data_out, c->data_max);
321 } else {
322 n = flistxattr(fd, c->data_out, c->data_max);
323 }
324 close(fd);
325 if (n < 0) {
326 return -errno; /* ENODATA / ERANGE / EACCES ... */
327 }
328 if (c->req->op == IMP_OP_LISTXATTR) {
329 n = (ssize_t) imp_xattr_filter_user(c->data_out, (size_t) n);
330 }
331 c->rep->data_len = (uint32_t) n; /* value / name list -> reply payload */
332 c->rep->rep_flags |= IMP_REP_HAS_DATA;
333 return 0;
334}

417/*
418 * Perform the requested op while impersonated. The caller fills every ctx
419 * field except `rel` (derived here from c->req->path). Returns 0 and may set
420 * *c->out_fd (>=0, ownership transferred to the caller) and/or c->rep->st; or
421 * returns -errno. `c->data_out` (size `c->data_max`) receives a trailing
422 * reply payload for READLINK / GETXATTR / LISTXATTR; rep->data_len +
423 * IMP_REP_HAS_DATA are set in that case. `c->data_in` (size `c->data_in_len`)
424 * is the inbound payload for SETXATTR (the value).
425 * HOW: last-line reserved-identity guard, then table dispatch to the
426 * imp_op_<name>() helper for the opcode; unknown op -> -ENOSYS (fail closed).
427 */
428int
429imp_do_op(imp_op_ctx_t *c)
430{
431 ngx_uint_t i;
432
433 /*
434 * Last-line guard, at the file-op execution point itself: the effective
435 * fs-credentials MUST be non-reserved here (imp_become guarantees it). Read
436 * them back one more time and refuse the op rather than ever run a file
437 * operation as uid/gid 0 or < the hard floor. Cheap (two syscalls), and it
438 * closes the window completely — no file op runs under a reserved identity.
439 */
440 if ((uid_t) setfsuid((uid_t) -1) < (uid_t) BRIX_IMP_HARD_MIN_ID
441 || (gid_t) setfsgid((gid_t) -1) < (gid_t) BRIX_IMP_HARD_MIN_ID)
442 {
443 return -EPERM;
444 }
445
446 c->rel = imp_rel(c->req->path);
447
448 for (i = 0; i < sizeof(imp_op_table) / sizeof(imp_op_table[0]); i++) {
449 if (imp_op_table[i].op == c->req->op) {
450 return imp_op_table[i].fn(c);
451 }
452 }
453 return -ENOSYS;
454}
