1/*
2 * ext_ops.c — vendor extension opcode handlers: kXR_setattr / kXR_symlink /
3 * kXR_readlink / kXR_link.
4 *
5 * WHAT: POSIX-completeness operations the base XRootD protocol has no wire op for:
6 * set-mtime/atime + chown (kXR_setattr), symbolic links (kXR_symlink /
7 * kXR_readlink) and hard links (kXR_link). They let a FUSE mount honour
8 * `cp -p`, `touch -d`, chown, and `ln`/`ln -s` against this gateway.
9 * WHY: Capability-negotiated extensions (advertised via kXR_Qconfig "xrdfs.ext");
10 * the native client only emits them when advertised, so a stock client never
11 * triggers these handlers and a stock server never receives the opcodes.
12 * HOW: Same shape as the other namespace handlers — extract + resolve the path(s)
13 * (brix_resolve_op_path / brix_path_resolve_beneath), run the three-tier
14 * auth gate, then perform the operation through a ROOT-CONFINED *at() helper
15 * (resolve_confined_ops.c) so a symlink/parent swap cannot escape the export
16 * root. setattr's variable attribute block is parsed from the payload prefix.
17 *
18 * Wire formats: see src/protocol/wire_vendor_ext.h.
19 */

35/*
36 * kXR_setattr — set timestamps (utimens) and/or owner (chown) on a path.
37 * Payload = 44-byte big-endian attribute prefix + NUL-terminated path
38 * (see wire_vendor_ext.h). mode is intentionally not handled (kXR_chmod covers it).
39 */
40ngx_int_t
41brix_handle_setattr(brix_ctx_t *ctx, ngx_connection_t *c,
42 ngx_stream_brix_srv_conf_t *conf)
43{
44 const u_char *p;
45 int32_t flags;
46 struct timespec times[2];
47 int32_t uid, gid;
48 char reqpath[BRIX_MAX_PATH + 1];
49 char resolved[PATH_MAX];
50
51 if (ctx->recv.payload == NULL
52 || ctx->recv.cur_dlen <= (uint32_t) BRIX_SETATTR_PREFIX_LEN) {
53 return brix_send_error(ctx, c, kXR_ArgMissing, "setattr: short payload");
54 }
55
56 p = ctx->recv.payload;
57 {
58 /* Shared 44-byte attribute-prefix codec (libxrdproto) — same offsets the
59 * client encodes with, so the two cannot drift. */
60 xrdp_setattr_t a;
61 xrdp_setattr_prefix_unpack(p, &a);
62 flags = (int32_t) a.flags;
63 times[0].tv_sec = (time_t) a.atime_sec;
64 times[0].tv_nsec = (long) a.atime_nsec;
65 times[1].tv_sec = (time_t) a.mtime_sec;
66 times[1].tv_nsec = (long) a.mtime_nsec;
67 uid = a.uid;
68 gid = a.gid;
69 }
70
71 if (!brix_extract_path(c->log, p + BRIX_SETATTR_PREFIX_LEN,
72 ctx->recv.cur_dlen - BRIX_SETATTR_PREFIX_LEN,
73 reqpath, sizeof(reqpath), 1)) {
74 return brix_send_error(ctx, c, kXR_ArgInvalid, "setattr: invalid path");
75 }
76
77 if (brix_path_resolve_beneath(conf, c->log, reqpath, BRIX_PATH_EXISTING,
78 resolved, sizeof(resolved)) != NGX_OK) {
79 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CHMOD, "SETATTR", reqpath, "-",
80 kXR_NotFound, "not found");
81 }
82
83 if (brix_auth_gate(ctx, c, BRIX_OP_CHMOD, "SETATTR",
84 reqpath, resolved, conf,
85 BRIX_AUTH_UPDATE, 1) != NGX_OK) {
86 return ctx->write_rc;
87 }
88
89 {
90 brix_vfs_ctx_t vctx;
91 brix_sd_setattr_t attr;
92
93 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
94 conf->common.root_canon, NULL, conf->common.allow_write,
95 0 /* is_tls */, ctx->identity, resolved);
96 brix_vfs_ctx_bind_backend_cred(&vctx,
97 &conf->common.storage_credential_dir,
98 conf->common.storage_credential_fallback);
99 brix_root_vfs_bind_deleg(ctx, conf, &vctx);
100
101 ngx_memzero(&attr, sizeof(attr));
102 attr.set_times = (flags & kXR_sa_times) ? 1 : 0;
103 attr.set_owner = (flags & kXR_sa_owner) ? 1 : 0;
104 attr.atime = times[0];
105 attr.mtime = times[1];
106 attr.uid = (uid_t) uid;
107 attr.gid = (gid_t) gid;
108
109 if (brix_vfs_setattr(&vctx, &attr) != NGX_OK) {
110 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CHMOD, "SETATTR", resolved, "-",
111 brix_kxr_from_errno(errno), strerror(errno));
112 }
113 }
114
115 BRIX_RETURN_OK(ctx, c, BRIX_OP_CHMOD, "SETATTR", resolved, "-", 0);
116}

118/*
119 * kXR_symlink — create a symbolic link. Wire (mirrors kXR_mv): payload =
120 * target + ' ' + linkpath, arg1len = strlen(target). The target is stored
121 * verbatim (NOT path-validated — it is link content, traversal-confined at open
122 * time); only linkpath is resolved + confined.
123 */
124ngx_int_t
125brix_handle_symlink(brix_ctx_t *ctx, ngx_connection_t *c,
126 ngx_stream_brix_srv_conf_t *conf)
127{
128 xrdw_twopath_req_t req;
129 char target[PATH_MAX];
130 char link_buf[BRIX_MAX_PATH + 1];
131 char link_resolved[PATH_MAX];
132 int16_t tlen;
133 size_t link_len;
134
135 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
136 return brix_send_error(ctx, c, kXR_ArgMissing, "symlink: no paths");
137 }
138 xrdw_twopath_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
139 tlen = req.arg1len;
140 if (tlen <= 0 || (uint32_t) (tlen + 1) >= ctx->recv.cur_dlen
141 || (size_t) tlen >= sizeof(target)) {
142 return brix_send_error(ctx, c, kXR_ArgInvalid, "symlink: bad arg1len");
143 }
144 if (ctx->recv.payload[tlen] != ' ') {
145 return brix_send_error(ctx, c, kXR_ArgInvalid, "symlink: bad separator");
146 }
147 /* target is verbatim link content — copy as-is, reject an embedded NUL. */
148 if (memchr(ctx->recv.payload, '\0', (size_t) tlen) != NULL) {
149 return brix_send_error(ctx, c, kXR_ArgInvalid, "symlink: NUL in target");
150 }
151 memcpy(target, ctx->recv.payload, (size_t) tlen);
152 target[tlen] = '\0';
153
154 link_len = (size_t) ctx->recv.cur_dlen - (size_t) tlen - 1;
155 if (link_len == 0
156 || !brix_extract_path(c->log, ctx->recv.payload + tlen + 1, link_len,
157 link_buf, sizeof(link_buf), 1)) {
158 return brix_send_error(ctx, c, kXR_ArgInvalid, "symlink: invalid linkpath");
159 }
160
161 /* The link location's parent must exist; the link itself must not (WRITE). */
162 if (brix_path_resolve_beneath(conf, c->log, link_buf, BRIX_PATH_WRITE,
163 link_resolved, sizeof(link_resolved)) != NGX_OK) {
164 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MKDIR, "SYMLINK", link_buf, "-",
165 kXR_NotFound, "invalid link path");
166 }
167 if (brix_auth_gate(ctx, c, BRIX_OP_MKDIR, "SYMLINK",
168 link_buf, link_resolved, conf,
169 BRIX_AUTH_UPDATE, 1) != NGX_OK) {
170 return ctx->write_rc;
171 }
172
173 if (brix_symlink_confined_canon(c->log, conf->common.root_canon,
174 target, link_resolved) != 0) {
175 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MKDIR, "SYMLINK", link_resolved, "-",
176 brix_kxr_from_errno(errno), strerror(errno));
177 }
178
179 BRIX_RETURN_OK(ctx, c, BRIX_OP_MKDIR, "SYMLINK", link_resolved, "-", 0);
180}

182/*
183 * kXR_link — create a hard link. Wire (mirrors kXR_mv): payload =
184 * oldpath + ' ' + newpath, arg1len = strlen(oldpath). Both paths are resolved +
185 * confined; the existing source is linked to a new destination.
186 */
187ngx_int_t
188brix_handle_link(brix_ctx_t *ctx, ngx_connection_t *c,
189 ngx_stream_brix_srv_conf_t *conf)
190{
191 xrdw_twopath_req_t req;
192 char src_buf[BRIX_MAX_PATH + 1];
193 char dst_buf[BRIX_MAX_PATH + 1];
194 char src_resolved[PATH_MAX];
195 char dst_resolved[PATH_MAX];
196 int16_t src_len;
197 size_t dst_len;
198
199 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
200 return brix_send_error(ctx, c, kXR_ArgMissing, "link: no paths");
201 }
202 xrdw_twopath_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
203 src_len = req.arg1len;
204 if (src_len <= 0 || (uint32_t) (src_len + 1) >= ctx->recv.cur_dlen) {
205 return brix_send_error(ctx, c, kXR_ArgInvalid, "link: bad arg1len");
206 }
207 if (ctx->recv.payload[src_len] != ' ') {
208 return brix_send_error(ctx, c, kXR_ArgInvalid, "link: bad separator");
209 }
210 dst_len = (size_t) ctx->recv.cur_dlen - (size_t) src_len - 1;
211 if (dst_len == 0
212 || !brix_extract_path(c->log, ctx->recv.payload, (size_t) src_len,
213 src_buf, sizeof(src_buf), 1)
214 || !brix_extract_path(c->log, ctx->recv.payload + src_len + 1, dst_len,
215 dst_buf, sizeof(dst_buf), 1)) {
216 return brix_send_error(ctx, c, kXR_ArgInvalid, "link: invalid path");
217 }
218
219 if (brix_path_resolve_beneath(conf, c->log, src_buf, BRIX_PATH_EXISTING,
220 src_resolved, sizeof(src_resolved)) != NGX_OK) {
221 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MKDIR, "LINK", src_buf, "-",
222 kXR_NotFound, "source not found");
223 }
224 if (brix_auth_gate(ctx, c, BRIX_OP_MKDIR, "LINK",
225 src_buf, src_resolved, conf,
226 BRIX_AUTH_UPDATE, 1) != NGX_OK) {
227 return ctx->write_rc;
228 }
229 if (brix_path_resolve_beneath(conf, c->log, dst_buf, BRIX_PATH_WRITE,
230 dst_resolved, sizeof(dst_resolved)) != NGX_OK) {
231 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MKDIR, "LINK", dst_buf, "-",
232 kXR_NotFound, "invalid destination path");
233 }
234 if (brix_auth_gate(ctx, c, BRIX_OP_MKDIR, "LINK",
235 dst_buf, dst_resolved, conf,
236 BRIX_AUTH_UPDATE, 1) != NGX_OK) {
237 return ctx->write_rc;
238 }
239
240 if (brix_link_confined_canon(c->log, conf->common.root_canon,
241 src_resolved, dst_resolved) != 0) {
242 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MKDIR, "LINK", dst_resolved, "-",
243 brix_kxr_from_errno(errno), strerror(errno));
244 }
245
246 BRIX_RETURN_OK(ctx, c, BRIX_OP_MKDIR, "LINK", dst_resolved, "-", 0);
247}

249/*
250 * kXR_readlink — return a symlink's target. Read-side op (no write gate). The
251 * response body is the raw target bytes (not NUL-terminated on the wire).
252 */
253ngx_int_t
254brix_handle_readlink(brix_ctx_t *ctx, ngx_connection_t *c,
255 ngx_stream_brix_srv_conf_t *conf)
256{
257 char reqpath[BRIX_MAX_PATH + 1];
258 char resolved[PATH_MAX];
259 char target[PATH_MAX];
260 ssize_t n;
261
262 /* NOEXIST (not EXISTING): the EXISTING gate follow-stats the path, which fails
263 * for a DANGLING symlink (target missing). readlinkat below is the real check —
264 * it reads the link itself without following it, and ENOENTs a missing path. */
265 if (brix_resolve_op_path(ctx, c, BRIX_OP_STAT, "READLINK", conf,
266 BRIX_PATH_NOEXIST,
267 reqpath, sizeof(reqpath),
268 resolved, sizeof(resolved)) != NGX_OK) {
269 return ctx->write_rc;
270 }
271 if (brix_auth_gate(ctx, c, BRIX_OP_STAT, "READLINK",
272 reqpath, resolved, conf,
273 BRIX_AUTH_READ, 0) != NGX_OK) {
274 return ctx->write_rc;
275 }
276
277 n = brix_readlink_confined_canon(c->log, conf->common.root_canon,
278 resolved, target, sizeof(target) - 1);
279 if (n < 0) {
280 BRIX_RETURN_ERR(ctx, c, BRIX_OP_STAT, "READLINK", resolved, "-",
281 brix_kxr_from_errno(errno), strerror(errno));
282 }
283
284 /* Body = the link target. Log + count, then send (own body → not RETURN_OK). */
285 brix_log_access(ctx, c, "READLINK", resolved, "-", 1, kXR_ok, NULL, (size_t) n);
286 BRIX_OP_OK(ctx, BRIX_OP_STAT);
287 return brix_send_ok(ctx, c, target, (uint32_t) n);
288}
