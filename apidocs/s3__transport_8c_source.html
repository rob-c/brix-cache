38/* ---- stall detection (fast-fail bounds) -----------------------------------
39 * The shared vtable's per-request `timeout_ms` is only an OUTER ceiling. A
40 * connection that blackholes at connect, or opens and then delivers no bytes
41 * (DPI on the path, or an origin rate-limiting a single node), otherwise hangs
42 * for the whole ceiling — burning a CVMFS client's hold window on one attempt
43 * so the retry/failover machinery never runs. These process-global bounds make
44 * such a connection fail in SECONDS, converting a hang into a fast transport
45 * error the fill loop can retry (with a fresh curl handle = a fresh connection)
46 * inside the hold budget. Set pre-fork from the cvmfs merge, inherited by every
47 * worker — operator policy, not per-request data, so it rides beside the trace
48 * flag rather than widening the transport signature (shared with the client
49 * build). 0 = leave libcurl's default (feature off). */
50static long g_origin_connect_ms; /* CURLOPT_CONNECTTIMEOUT_MS (ms) */
51static long g_origin_stall_secs; /* CURLOPT_LOW_SPEED_TIME (seconds) */
52static long g_origin_stall_bytes; /* CURLOPT_LOW_SPEED_LIMIT (bytes/s) */
53static long g_origin_attempt_ms; /* CURLOPT_TIMEOUT_MS cap (ms) */
54
55void
56brix_s3_origin_timeouts_set(long connect_ms, long stall_secs,
57 long stall_bytes_per_s, long attempt_ms)
58{
59 g_origin_connect_ms = connect_ms > 0 ? connect_ms : 0;
60 g_origin_stall_secs = stall_secs > 0 ? stall_secs : 0;
61 g_origin_stall_bytes = stall_bytes_per_s > 0 ? stall_bytes_per_s : 0;
62 g_origin_attempt_ms = attempt_ms > 0 ? attempt_ms : 0;
63}

95{
96 curl_version_info_data *vi;
97
98 switch (ver) {
99 case 11:
100 return 1; /* every libcurl speaks HTTP/1.1 */
101 case 20:
102 case 21:
103 vi = curl_version_info(CURLVERSION_NOW);
104 return (vi != NULL && (vi->features & CURL_VERSION_HTTP2)) ? 1 : 0;
105 case 30:
106#ifdef CURL_VERSION_HTTP3
107 vi = curl_version_info(CURLVERSION_NOW);
108 return (vi != NULL && (vi->features & CURL_VERSION_HTTP3)) ? 1 : 0;
109#else
110 return 0; /* built against pre-H3 headers */
111#endif
112 default:
113 return 0;
114 }

117/* Map the operator policy onto CURLOPT_HTTP_VERSION for this request. Unset
118 * (the default) sets nothing — libcurl's own policy stays untouched. */
119static void
120s3o_apply_http_version(CURL *curl)
121{
122 switch (g_origin_http_version) {
123 case 11:
124 curl_easy_setopt(curl, CURLOPT_HTTP_VERSION,
125 (long) CURL_HTTP_VERSION_1_1);
126 break;
127 case 20:
128 /* ALPN h2 over TLS / h2c Upgrade over cleartext; libcurl falls back
129 * to 1.1 by itself when the origin does not negotiate. */
130 curl_easy_setopt(curl, CURLOPT_HTTP_VERSION,
131 (long) CURL_HTTP_VERSION_2_0);
132 break;
133 case 21:
134 /* Cleartext h2 with prior knowledge: no Upgrade dance, no fallback —
135 * the origin must speak h2c directly (nghttpd/haproxy h2c listener). */
136 curl_easy_setopt(curl, CURLOPT_HTTP_VERSION,
137 (long) CURL_HTTP_VERSION_2_PRIOR_KNOWLEDGE);
138 break;
139#ifdef CURL_VERSION_HTTP3
140 case 30:
141 curl_easy_setopt(curl, CURLOPT_HTTP_VERSION,
142 (long) CURL_HTTP_VERSION_3);
143 break;
144#endif
145 default:
146 break; /* 0 = unset: libcurl default */
147 }
148}

172static long
173s3o_ms_since(const struct timespec *t0)
174{
175 struct timespec now;
176 long ms;
177
178 if (clock_gettime(CLOCK_MONOTONIC, &now) != 0) {
179 return 0;
180 }
181 ms = (now.tv_sec - t0->tv_sec) * 1000L
182 + (now.tv_nsec - t0->tv_nsec) / 1000000L;
183 return ms < 0 ? 0 : ms;
184}

196/* Append `n` bytes to a growable buffer; returns 0 / -1 (OOM). */
197static int
198s3o_buf_append(char **buf, size_t *len, size_t *cap, const char *src, size_t n)
199{
200 if (*len + n + 1 > *cap) {
201 size_t newcap = (*cap == 0) ? 8192 : *cap;
202 char *nb;
203
204 while (*len + n + 1 > newcap) {
205 newcap *= 2;
206 }
207 nb = realloc(*buf, newcap);
208 if (nb == NULL) {
209 return -1;
210 }
211 *buf = nb;
212 *cap = newcap;
213 }
214 memcpy(*buf + *len, src, n);
215 *len += n;
216 (*buf)[*len] = '\0';
217 return 0;
218}

244/* Split the CRLF-separated header block sd_s3 built into a curl_slist, dropping
245 * blank lines and any trailing CR. Returns the list (caller frees) or NULL. */
246static struct curl_slist *
247s3o_build_slist(const char *headers)
248{
249 struct curl_slist *list = NULL;
250 const char *p = headers;
251
252 while (p != NULL && *p != '\0') {
253 const char *eol = strpbrk(p, "\r\n");
254 size_t linelen = (eol != NULL) ? (size_t) (eol - p) : strlen(p);
255
256 if (linelen > 0) {
257 char line[1024];
258
259 if (linelen >= sizeof(line)) {
260 linelen = sizeof(line) - 1;
261 }
262 memcpy(line, p, linelen);
263 line[linelen] = '\0';
264 list = curl_slist_append(list, line);
265 }
266 if (eol == NULL) {
267 break;
268 }
269 p = eol + strspn(eol, "\r\n"); /* skip the CR/LF run */
270 }
271 return list;
272}

274/* s3o_trace_t — the fields of one upstream-request trace line.
275 *
276 * WHAT: Bundles the request-identity (method/host/port/path) plus the outcome
277 * (status/bytes/duration/error) that s3o_trace() formats into a single
278 * line.
279 * WHY: The trace inputs were passed as 8 loose params; a small file-local
280 * descriptor keeps the emit signature at one argument and lets the two
281 * call sites (transport failure vs HTTP outcome) fill a stack struct in
282 * place with no behavior change.
283 * HOW: `status` < 0 marks a transport-level failure (curl error text in
284 * `err`); otherwise it is the HTTP status and `err` is NULL. `path` is
285 * wire-derived and gets sanitized by s3o_trace(). */
286typedef struct {
287 const char *method;
288 const char *host;
289 int port;
290 const char *path; /* path_and_query, wire-derived → sanitized */
291 int status; /* < 0 = transport failure, else HTTP status */
292 size_t bytes;
293 long dur_ms;
294 const char *err; /* curl error text when status < 0, else NULL */
295 const char *proto; /* negotiated HTTP version token, or NULL */
296} s3o_trace_t;
297
298/* Emit one upstream-request trace line from a filled s3o_trace_t. Logged at
299 * DEBUG normally, promoted to INFO under brix_cvmfs_trace. */
300static void
301s3o_trace(const s3o_trace_t *t)
302{
303 ngx_uint_t level = g_origin_trace_info ? NGX_LOG_INFO : NGX_LOG_DEBUG;
304 char safe[1024];
305
306 if (ngx_cycle == NULL || ngx_cycle->log == NULL
307 || ngx_cycle->log->log_level < level)
308 {
309 return; /* below the configured level: cheap */
310 }
311 brix_sanitize_log_string(t->path != NULL ? t->path : "",
312 safe, sizeof(safe));
313 if (t->status < 0) {
314 ngx_log_error(level, ngx_cycle->log, 0,
315 "cvmfs-trace: upstream %s http://%s:%d%s FAILED (%s) dur_ms=%l",
316 t->method, t->host, t->port, safe,
317 t->err ? t->err : "transport error", t->dur_ms);
318 } else {
319 ngx_log_error(level, ngx_cycle->log, 0,
320 "cvmfs-trace: upstream %s http://%s:%d%s status=%d bytes=%uz "
321 "host=%s:%d dur_ms=%l proto=%s",
322 t->method, t->host, t->port, safe, t->status, t->bytes,
323 t->host, t->port, t->dur_ms,
324 t->proto != NULL ? t->proto : "?");
325 }
326}

328/* ---- connection reuse (one long-lived connection per fill thread) ----------
329 * A fresh curl handle per request opened AND tore down a new TCP connection for
330 * EVERY origin request — ruinous on a high-latency link, where each object then
331 * paid the full TCP+HTTP handshake and a TCP slow-start ramp from a cold
332 * congestion window (so small objects finished before reaching full speed).
333 * Instead we keep ONE handle per fill thread and curl_easy_reset() it between
334 * requests: reset clears the per-request options but PRESERVES the live
335 * connection pool and DNS cache, so the 2nd..Nth request to the same Stratum-1
336 * reuses the already-open, already-warmed keep-alive connection — no handshake,
337 * no re-resolve, no slow-start restart. Per-thread because a curl easy handle
338 * cannot be shared across threads concurrently; a handful of warm connections
339 * replace thousands of cold ones. The pthread_key destructor cleans the handle
340 * up when the fill thread exits. */
341static pthread_key_t g_curl_key;
342static pthread_once_t g_curl_key_once = PTHREAD_ONCE_INIT;
343
344static void
345s3o_curl_thread_free(void *handle)
346{
347 if (handle != NULL) {
348 curl_easy_cleanup((CURL *) handle);
349 }
350}
351
352static void
353s3o_curl_key_init(void)
354{
355 (void) pthread_key_create(&g_curl_key, s3o_curl_thread_free);
356}

358/* This thread's persistent handle, reset and ready for a new request with its
359 * connection pool intact. Creates it on first use; NULL only on alloc failure. */
360static CURL *
361s3o_curl_acquire(void)
362{
363 CURL *handle;
364
365 (void) pthread_once(&g_curl_key_once, s3o_curl_key_init);
366 handle = (CURL *) pthread_getspecific(g_curl_key);
367 if (handle != NULL) {
368 curl_easy_reset(handle); /* keeps live connections + DNS cache */
369 return handle;
370 }
371 handle = curl_easy_init();
372 if (handle != NULL) {
373 (void) pthread_setspecific(g_curl_key, handle);
374 }
375 return handle;
376}

378/* s3o_apply_ca — point libcurl at the operator-configured trusted CA for origin
379 * TLS verification.
380 *
381 * WHAT: When `ca_path` is a non-empty filesystem path, set CURLOPT_CAPATH (for a
382 * hashed CA directory) or CURLOPT_CAINFO (for a single PEM bundle file) so
383 * the origin's server certificate is verified against THAT trust anchor.
384 * WHY: The https backend leg (phase-70) fetches from an origin whose server cert
385 * is signed by an operator/site CA that is NOT in the system bundle (e.g. a
386 * test PKI or a private grid CA). Without this the handshake fails with
387 * "SSL peer certificate was not OK". Mirrors brix_pelican_set_ca().
388 * HOW: A NULL/empty path leaves libcurl's default system bundle untouched, so a
389 * real-S3 / public-https origin (whose cert chains to a public root) keeps
390 * working. The path type is probed with stat(): a directory → CAPATH (a
391 * hashed-symlink CA dir), anything else (a regular file, or unstattable but
392 * operator-supplied) → CAINFO (a bundle). Verification stays ON regardless;
393 * this only widens WHICH roots are trusted, never disables the check. */
394static void
395s3o_apply_ca(CURL *curl, const char *ca_path)
396{
397 struct stat st;
398
399 if (ca_path == NULL || ca_path[0] == '\0') {
400 return; /* keep libcurl's system bundle */
401 }
402 if (stat(ca_path, &st) == 0 && S_ISDIR(st.st_mode)) {
403 curl_easy_setopt(curl, CURLOPT_CAPATH, ca_path);
404 } else {
405 curl_easy_setopt(curl, CURLOPT_CAINFO, ca_path);
406 }
407}

409/* s3o_request_t — one origin request's inputs, shared by the plain and
410 * cred-scoped transport slots.
411 *
412 * WHAT: The immutable inputs of a single synchronous libcurl request: the
413 * target (host/port/tls), the HTTP call (method/path/headers/body), the
414 * caller timeout, the OPTIONAL mutual-TLS client cert, and the operator CA
415 * context (`tctx`).
416 * WHY: Both vtable entry points (s3o_request / s3o_request_cred) pass the same
417 * dozen-plus arguments straight through to one shared body; collapsing
418 * them into a file-local descriptor keeps the shared body and its extracted
419 * helpers at one argument instead of fourteen, with no behavior change.
420 * HOW: `tctx` is the transport's OPTIONAL context — a NUL-terminated CA
421 * file-or-dir PATH (the operator's trusted CA for origin TLS), or NULL for
422 * libcurl's system bundle; applied via s3o_apply_ca(). `client_cert_pem`
423 * is NULL for the plain slot; on a `tls` request a non-empty path is
424 * presented as the mutual-TLS client cert (cert chain + key in one PEM). */
425typedef struct {
426 void *tctx; /* operator CA path (via s3o_apply_ca), or NULL */
427 const char *host;
428 int port;
429 int tls;
430 const char *method;
431 const char *path_and_query;
432 const char *headers;
433 const void *body;
434 size_t body_len;
435 int timeout_ms;
436 const char *client_cert_pem; /* mutual-TLS client PEM, or NULL (plain slot) */
437} s3o_request_t;

439/* s3o_build_headers — build the curl_slist for one request.
440 *
441 * WHAT: Parses the pre-built header block into a curl_slist and appends the
442 * forced Host header plus the Expect-suppression line.
443 * WHY: sd_s3 signs the SigV4 canonical host as "host:port" for EVERY port, so
444 * libcurl's default of dropping the port on 80/443 would break the
445 * signature — the Host header must be sent byte-for-byte as signed. Expect:
446 * 100-continue is suppressed because some S3 servers stall on it and sd_s3
447 * does not rely on it.
448 * HOW: Returns the list (caller frees via curl_slist_free_all); byte-identical
449 * to the inlined sequence it replaces. */
450static struct curl_slist *
451s3o_build_headers(const s3o_request_t *req)
452{
453 struct curl_slist *slist = s3o_build_slist(req->headers);
454 char host_hdr[300];
455
456 snprintf(host_hdr, sizeof(host_hdr), "Host: %s:%d", req->host, req->port);
457 slist = curl_slist_append(slist, host_hdr);
458 slist = curl_slist_append(slist, "Expect:");
459 return slist;
460}

462/* s3o_apply_timeouts — apply the per-attempt ceiling and the fast-fail bounds.
463 *
464 * WHAT: Sets CURLOPT_TIMEOUT_MS (the per-attempt total ceiling) and, when the
465 * operator configured them, CURLOPT_CONNECTTIMEOUT_MS and the low-speed
466 * stall bounds.
467 * WHY: The process-global attempt cap (operator policy) wins over the caller's
468 * timeout so a whole connect+transfer is abandoned promptly and the fill
469 * loop retries on a fresh connection inside the client's window; the
470 * connect and low-speed bounds turn a blackholed connect or a
471 * connected-but-silent origin into a fast transport error.
472 * HOW: A 0 operator value leaves libcurl's default; the stall pair takes effect
473 * only together. Byte-identical to the inlined option sequence. */
474static void
475s3o_apply_timeouts(CURL *curl, int timeout_ms)
476{
477 long total_ms = (g_origin_attempt_ms > 0) ? g_origin_attempt_ms
478 : (timeout_ms > 0 ? timeout_ms : 60000);
479
480 curl_easy_setopt(curl, CURLOPT_TIMEOUT_MS, total_ms);
481 curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
482
483 if (g_origin_connect_ms > 0) {
484 curl_easy_setopt(curl, CURLOPT_CONNECTTIMEOUT_MS, g_origin_connect_ms);
485 }
486 if (g_origin_stall_secs > 0 && g_origin_stall_bytes > 0) {
487 curl_easy_setopt(curl, CURLOPT_LOW_SPEED_TIME, g_origin_stall_secs);
488 curl_easy_setopt(curl, CURLOPT_LOW_SPEED_LIMIT, g_origin_stall_bytes);
489 }
490}

492/* s3o_apply_reuse — apply the connection-reuse policy for this request.
493 *
494 * WHAT: Either forbids reuse (fresh connection per request) or enables warm
495 * keep-alive reuse with the middlebox-death guards (TCP keepalive + a
496 * bounded max-age).
497 * WHY: Reusing one warm connection per fill thread amortizes the handshake and
498 * TCP slow-start on a high-latency link; the no-reuse escape hatch restores
499 * the pre-2026-07-03 one-connection-per-request behaviour on a hostile
500 * network where a reaped-but-reused connection would time out.
501 * HOW: Reads the process-global g_origin_no_reuse flag; byte-identical to the
502 * inlined branch it replaces. */
503static void
504s3o_apply_reuse(CURL *curl)
505{
506 if (g_origin_no_reuse) {
507 curl_easy_setopt(curl, CURLOPT_FORBID_REUSE, 1L);
508 curl_easy_setopt(curl, CURLOPT_FRESH_CONNECT, 1L);
509 return;
510 }
511 curl_easy_setopt(curl, CURLOPT_MAXCONNECTS, 16L);
512 curl_easy_setopt(curl, CURLOPT_TCP_KEEPALIVE, 1L);
513 curl_easy_setopt(curl, CURLOPT_TCP_KEEPIDLE, 15L);
514 curl_easy_setopt(curl, CURLOPT_TCP_KEEPINTVL, 15L);
515#ifdef CURLOPT_MAXAGE_CONN
516 curl_easy_setopt(curl, CURLOPT_MAXAGE_CONN, 20L);
517#endif
518}

520/* s3o_apply_method — select the HTTP method and attach any request body.
521 *
522 * WHAT: For HEAD sets NOBODY; for methods other than GET sets the custom request
523 * verb and, when present, the POST body. GET needs no options (libcurl's
524 * default).
525 * WHY: sd_s3 drives HEAD/GET/PUT/DELETE over this one transport; the verb and
526 * body wiring is the only method-specific curl state.
527 * HOW: Byte-identical to the inlined method dispatch it replaces. */
528static void
529s3o_apply_method(CURL *curl, const s3o_request_t *req)
530{
531 if (strcmp(req->method, "HEAD") == 0) {
532 curl_easy_setopt(curl, CURLOPT_NOBODY, 1L);
533 return;
534 }
535 if (strcmp(req->method, "GET") == 0) {
536 return;
537 }
538 curl_easy_setopt(curl, CURLOPT_CUSTOMREQUEST, req->method);
539 if (req->body != NULL && req->body_len > 0) {
540 curl_easy_setopt(curl, CURLOPT_POSTFIELDS, req->body);
541 curl_easy_setopt(curl, CURLOPT_POSTFIELDSIZE_LARGE,
542 (curl_off_t) req->body_len);
543 } else {
544 curl_easy_setopt(curl, CURLOPT_POSTFIELDSIZE_LARGE, (curl_off_t) 0);
545 }
546}

548/* s3o_apply_tls — apply TLS verification, the operator CA, and mutual-TLS cert.
549 *
550 * WHAT: On a `tls` request, enables peer/host verification, points libcurl at
551 * the operator CA (via s3o_apply_ca), and — when a client PEM was supplied
552 * — presents it as the mutual-TLS client certificate.
553 * WHY: Origin TLS verification is the operator's trust decision (verify by
554 * default so a misconfigured origin fails loudly); the mutual-TLS cert
555 * authenticates the origin hop AS the end user (phase-70 §5.1 GSI
556 * pass-through/select over an https backend leg).
557 * HOW: A non-TLS request is a no-op. Verification stays ON regardless; the CA
558 * context only widens which roots are trusted. The client cert is read
559 * (cert + key) from the one combined PEM and, because the handle is reset
560 * per request, never leaks into a later anonymous/bearer request.
561 * Byte-identical to the inlined TLS block it replaces. */
562static void
563s3o_apply_tls(CURL *curl, const s3o_request_t *req)
564{
565 if (!req->tls) {
566 return;
567 }
568 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYPEER, 1L);
569 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYHOST, 2L);
570 s3o_apply_ca(curl, (const char *) req->tctx);
571 if (req->client_cert_pem != NULL && req->client_cert_pem[0] != '\0') {
572 curl_easy_setopt(curl, CURLOPT_SSLCERTTYPE, "PEM");
573 curl_easy_setopt(curl, CURLOPT_SSLKEYTYPE, "PEM");

579/* s3o_configure — set every per-request curl option for `req` on `curl`.
580 *
581 * WHAT: Sets the URL, header list, write/header capture callbacks, then the
582 * timeout, reuse, method and TLS option groups.
583 * WHY: Concentrates the whole per-request curl configuration behind one call so
584 * the request body reads as build → configure → perform → finish.
585 * HOW: `r` is the response-capture buffer bound to the write/header callbacks;
586 * `slist` is the caller-owned header list. Option order is byte-identical
587 * to the pre-split sequence. */
588static void
589s3o_configure(CURL *curl, const s3o_request_t *req, s3o_resp_t *r,
590 struct curl_slist *slist)
591{
592 char url[2048];
593
594 snprintf(url, sizeof(url), "%s://%s:%d%s",
595 req->tls ? "https" : "http", req->host, req->port,
596 req->path_and_query);
597
598 curl_easy_setopt(curl, CURLOPT_URL, url);
599 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, slist);
600 curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, s3o_write_cb);
601 curl_easy_setopt(curl, CURLOPT_WRITEDATA, r);
602 curl_easy_setopt(curl, CURLOPT_HEADERFUNCTION, s3o_header_cb);
603 curl_easy_setopt(curl, CURLOPT_HEADERDATA, r);
604
605 s3o_apply_timeouts(curl, req->timeout_ms);
606 s3o_apply_reuse(curl);
607 s3o_apply_http_version(curl);
608 s3o_apply_method(curl, req);
609 s3o_apply_tls(curl, req);
610}

612/* s3o_request_impl — the shared request body for the plain and cred-scoped
613 * transport slots.
614 *
615 * WHAT: Performs one synchronous libcurl request described by `req`, capturing
616 * the response into a heap s3o_resp_t handed back through `resp->opaque`.
617 * WHY: The GSI-over-https backend leg (phase-70 §5.1) authenticates the origin
618 * hop AS the end user by presenting the user's proxy PEM as the TLS client
619 * cert — the only structural difference from an anonymous/bearer request is
620 * the mutual-TLS options (in req->client_cert_pem), so both entry points
621 * share one body.
622 * HOW: Acquires this thread's persistent warm handle, builds the header list,
623 * configures every per-request option (s3o_configure), performs the
624 * transfer and emits the trace line. A per-request curl handle is reset
625 * between uses, so TLS/cert options do not leak into the next request. */
626static int
627s3o_request_impl(const s3o_request_t *req, brix_s3_resp_t *resp,
628 char *errbuf, size_t errcap)
629{
630 CURL *curl;
631 CURLcode res;
632 struct curl_slist *slist = NULL;
633 s3o_resp_t *r;
634 long status = 0;
635 struct timespec t0;
636 s3o_trace_t tr = { 0 };
637
638 (void) clock_gettime(CLOCK_MONOTONIC, &t0);
639
640 curl = s3o_curl_acquire(); /* this thread's persistent, warm handle */
641 if (curl == NULL) {
642 if (errbuf && errcap) { snprintf(errbuf, errcap, "curl handle unavailable"); }
643 return -1;
644 }
645
646 r = calloc(1, sizeof(*r));
647 if (r == NULL) {
648 if (errbuf && errcap) { snprintf(errbuf, errcap, "out of memory"); }
649 return -1; /* handle stays in TLS, reused next call */
650 }
651
652 slist = s3o_build_headers(req);
653 s3o_configure(curl, req, r, slist);
654
655 tr.method = req->method;
656 tr.host = req->host;
657 tr.port = req->port;
658 tr.path = req->path_and_query;
659
660 res = curl_easy_perform(curl);
661 if (res != CURLE_OK) {
662 if (errbuf && errcap) {
663 snprintf(errbuf, errcap, "curl: %s", curl_easy_strerror(res));
664 }
665 tr.status = -1;
666 tr.dur_ms = s3o_ms_since(&t0);
667 tr.err = curl_easy_strerror(res);
668 s3o_trace(&tr);
669 curl_slist_free_all(slist);
670 free(r->body);
671 free(r->hdrs);
672 free(r);
673 return -1; /* handle persists (curl drops a dead
674 connection from the pool by itself) */
675 }
676
677 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &status);
678 curl_slist_free_all(slist);
679
680 resp->status = (int) status;
681 resp->opaque = r;
682 tr.status = (int) status;
683 tr.bytes = r->body_len;
684 tr.dur_ms = s3o_ms_since(&t0);
685 tr.proto = s3o_negotiated_proto(curl);
686 s3o_trace(&tr);
687 return 0;
688}

740static int
741s3o_resp_header(const brix_s3_resp_t *resp, const char *name,
742 char *out, size_t outcap)
743{
744 const s3o_resp_t *r = resp->opaque;
745 size_t namelen = strlen(name);
746 const char *p;
747
748 if (r == NULL || r->hdrs == NULL) {
749 return -1;
750 }
751 for (p = r->hdrs; *p != '\0'; ) {
752 const char *eol = strpbrk(p, "\r\n");
753 size_t linelen = (eol != NULL) ? (size_t) (eol - p) : strlen(p);
754
755 if (linelen > namelen && p[namelen] == ':'
756 && strncasecmp(p, name, namelen) == 0)
757 {
758 const char *v = p + namelen + 1;
759 size_t vlen;
760
761 while (v < p + linelen && (*v == ' ' || *v == '\t')) { v++; }
762 vlen = (size_t) (p + linelen - v);
763 if (vlen >= outcap) { vlen = outcap - 1; }
764 memcpy(out, v, vlen);
765 out[vlen] = '\0';
766 return 0;
767 }
768 if (eol == NULL) { break; }
769 p = eol + strspn(eol, "\r\n");
770 }
771 return -1;
772}
