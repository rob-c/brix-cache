37/* ---- stall detection (fast-fail bounds) -----------------------------------
38 * The shared vtable's per-request `timeout_ms` is only an OUTER ceiling. A
39 * connection that blackholes at connect, or opens and then delivers no bytes
40 * (DPI on the path, or an origin rate-limiting a single node), otherwise hangs
41 * for the whole ceiling — burning a CVMFS client's hold window on one attempt
42 * so the retry/failover machinery never runs. These process-global bounds make
43 * such a connection fail in SECONDS, converting a hang into a fast transport
44 * error the fill loop can retry (with a fresh curl handle = a fresh connection)
45 * inside the hold budget. Set pre-fork from the cvmfs merge, inherited by every
46 * worker — operator policy, not per-request data, so it rides beside the trace
47 * flag rather than widening the transport signature (shared with the client
48 * build). 0 = leave libcurl's default (feature off). */
49static long g_origin_connect_ms; /* CURLOPT_CONNECTTIMEOUT_MS (ms) */
50static long g_origin_stall_secs; /* CURLOPT_LOW_SPEED_TIME (seconds) */
51static long g_origin_stall_bytes; /* CURLOPT_LOW_SPEED_LIMIT (bytes/s) */
52static long g_origin_attempt_ms; /* CURLOPT_TIMEOUT_MS cap (ms) */
53

55brix_s3_origin_timeouts_set(long connect_ms, long stall_secs,

59 g_origin_stall_secs = stall_secs > 0 ? stall_secs : 0;
60 g_origin_stall_bytes = stall_bytes_per_s > 0 ? stall_bytes_per_s : 0;

64/* Connection reuse toggle (default ON = reuse). Reusing one keep-alive
65 * connection per fill thread amortizes the handshake + TCP slow-start on a
66 * high-latency link — but on a path with a middlebox that silently reaps idle
67 * connections, a reused-but-dead connection makes the next request time out
68 * (no RST to trigger curl's fresh-connection retry). Set OFF to force a fresh
69 * connection per request (the pre-2026-07-03 behaviour) when reuse hurts more
70 * than it helps on a hostile network. Set pre-fork from the cvmfs merge. */
71static int g_origin_no_reuse; /* 0 = reuse (default), 1 = fresh per req */
72
73void
74brix_s3_origin_reuse_set(int reuse_on)
75{
76 g_origin_no_reuse = reuse_on ? 0 : 1;
77}

79static long
80s3o_ms_since(const struct timespec *t0)
81{
82 struct timespec now;
83 long ms;
84
85 if (clock_gettime(CLOCK_MONOTONIC, &now) != 0) {
86 return 0;
87 }
88 ms = (now.tv_sec - t0->tv_sec) * 1000L
89 + (now.tv_nsec - t0->tv_nsec) / 1000000L;
90 return ms < 0 ? 0 : ms;
91}

103/* Append `n` bytes to a growable buffer; returns 0 / -1 (OOM). */
104static int
105s3o_buf_append(char **buf, size_t *len, size_t *cap, const char *src, size_t n)
106{
107 if (*len + n + 1 > *cap) {
108 size_t newcap = (*cap == 0) ? 8192 : *cap;
109 char *nb;
110
111 while (*len + n + 1 > newcap) {
112 newcap *= 2;
113 }
114 nb = realloc(*buf, newcap);
115 if (nb == NULL) {
116 return -1;
117 }
118 *buf = nb;
119 *cap = newcap;
120 }
121 memcpy(*buf + *len, src, n);
122 *len += n;
123 (*buf)[*len] = '\0';
124 return 0;
125}

151/* Split the CRLF-separated header block sd_s3 built into a curl_slist, dropping
152 * blank lines and any trailing CR. Returns the list (caller frees) or NULL. */
153static struct curl_slist *
154s3o_build_slist(const char *headers)
155{
156 struct curl_slist *list = NULL;
157 const char *p = headers;
158
159 while (p != NULL && *p != '\0') {
160 const char *eol = strpbrk(p, "\r\n");
161 size_t linelen = (eol != NULL) ? (size_t) (eol - p) : strlen(p);
162
163 if (linelen > 0) {
164 char line[1024];
165
166 if (linelen >= sizeof(line)) {
167 linelen = sizeof(line) - 1;
168 }
169 memcpy(line, p, linelen);
170 line[linelen] = '\0';
171 list = curl_slist_append(list, line);
172 }
173 if (eol == NULL) {
174 break;
175 }
176 p = eol + strspn(eol, "\r\n"); /* skip the CR/LF run */
177 }
178 return list;
179}

181/* Emit one upstream-request trace line. `status` < 0 means a transport-level
182 * failure (curl error in `err`); otherwise it is the HTTP status. Logged at
183 * DEBUG normally, promoted to INFO under brix_cvmfs_trace. Path is
184 * wire-derived → sanitized. */
185static void
186s3o_trace(const char *method, const char *host, int port,
187 const char *path_and_query, int status, size_t bytes,
188 long dur_ms, const char *err)
189{
190 ngx_uint_t level = g_origin_trace_info ? NGX_LOG_INFO : NGX_LOG_DEBUG;
191 char safe[1024];
192
193 if (ngx_cycle == NULL || ngx_cycle->log == NULL
194 || ngx_cycle->log->log_level < level)
195 {
196 return; /* below the configured level: cheap */
197 }
198 brix_sanitize_log_string(path_and_query != NULL ? path_and_query : "",
199 safe, sizeof(safe));
200 if (status < 0) {
201 ngx_log_error(level, ngx_cycle->log, 0,
202 "cvmfs-trace: upstream %s http://%s:%d%s FAILED (%s) dur_ms=%l",
203 method, host, port, safe, err ? err : "transport error", dur_ms);
204 } else {
205 ngx_log_error(level, ngx_cycle->log, 0,
206 "cvmfs-trace: upstream %s http://%s:%d%s status=%d bytes=%uz "
207 "host=%s:%d dur_ms=%l",
208 method, host, port, safe, status, bytes, host, port, dur_ms);
209 }
210}

212/* ---- connection reuse (one long-lived connection per fill thread) ----------
213 * A fresh curl handle per request opened AND tore down a new TCP connection for
214 * EVERY origin request — ruinous on a high-latency link, where each object then
215 * paid the full TCP+HTTP handshake and a TCP slow-start ramp from a cold
216 * congestion window (so small objects finished before reaching full speed).
217 * Instead we keep ONE handle per fill thread and curl_easy_reset() it between
218 * requests: reset clears the per-request options but PRESERVES the live
219 * connection pool and DNS cache, so the 2nd..Nth request to the same Stratum-1
220 * reuses the already-open, already-warmed keep-alive connection — no handshake,
221 * no re-resolve, no slow-start restart. Per-thread because a curl easy handle
222 * cannot be shared across threads concurrently; a handful of warm connections
223 * replace thousands of cold ones. The pthread_key destructor cleans the handle
224 * up when the fill thread exits. */
225static pthread_key_t g_curl_key;
226static pthread_once_t g_curl_key_once = PTHREAD_ONCE_INIT;
227
228static void
229s3o_curl_thread_free(void *handle)
230{
231 if (handle != NULL) {
232 curl_easy_cleanup((CURL *) handle);
233 }
234}

236static void
237s3o_curl_key_init(void)
238{
239 (void) pthread_key_create(&g_curl_key, s3o_curl_thread_free);
240}

242/* This thread's persistent handle, reset and ready for a new request with its
243 * connection pool intact. Creates it on first use; NULL only on alloc failure. */
244static CURL *
245s3o_curl_acquire(void)
246{
247 CURL *handle;
248
249 (void) pthread_once(&g_curl_key_once, s3o_curl_key_init);
250 handle = (CURL *) pthread_getspecific(g_curl_key);
251 if (handle != NULL) {
252 curl_easy_reset(handle); /* keeps live connections + DNS cache */
253 return handle;
254 }
255 handle = curl_easy_init();
256 if (handle != NULL) {
257 (void) pthread_setspecific(g_curl_key, handle);
258 }
259 return handle;
260}

262static int
263s3o_request(void *tctx, const char *host, int port, int tls,
264 const char *method, const char *path_and_query,
265 const char *headers, const void *body, size_t body_len,
266 int timeout_ms, brix_s3_resp_t *resp, char *errbuf, size_t errcap)
267{
268 CURL *curl;
269 CURLcode res;
270 struct curl_slist *slist = NULL;
271 s3o_resp_t *r;
272 char url[2048];
273 char host_hdr[300];
274 long status = 0;
275 struct timespec t0;
276
277 (void) tctx;
278 (void) clock_gettime(CLOCK_MONOTONIC, &t0);
279
280 curl = s3o_curl_acquire(); /* this thread's persistent, warm handle */
281 if (curl == NULL) {
282 if (errbuf && errcap) { snprintf(errbuf, errcap, "curl handle unavailable"); }
283 return -1;
284 }
285
286 r = calloc(1, sizeof(*r));
287 if (r == NULL) {
288 if (errbuf && errcap) { snprintf(errbuf, errcap, "out of memory"); }
289 return -1; /* handle stays in TLS, reused next call */
290 }
291
292 snprintf(url, sizeof(url), "%s://%s:%d%s",
293 tls ? "https" : "http", host, port, path_and_query);
294
295 slist = s3o_build_slist(headers);
296 /* Force the Host header to "host:port" — sd_s3 signs the SigV4 canonical host
297 * as host:port for EVERY port (brix_format_host_port always appends it), so
298 * libcurl's default of omitting the port on a default port (80/443) would
299 * break the signature. Always sending the port keeps the Host header byte-for-
300 * byte what was signed. */
301 snprintf(host_hdr, sizeof(host_hdr), "Host: %s:%d", host, port);
302 slist = curl_slist_append(slist, host_hdr);
303 /* Disable libcurl's automatic Expect: 100-continue on PUT/POST (some S3
304 * servers stall on it); sd_s3 does not rely on it. */
305 slist = curl_slist_append(slist, "Expect:");
306
307 curl_easy_setopt(curl, CURLOPT_URL, url);
308 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, slist);
309 curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, s3o_write_cb);
310 curl_easy_setopt(curl, CURLOPT_WRITEDATA, r);
311 curl_easy_setopt(curl, CURLOPT_HEADERFUNCTION, s3o_header_cb);
312 curl_easy_setopt(curl, CURLOPT_HEADERDATA, r);
313 /* Per-attempt total ceiling: the process-global attempt cap (operator
314 * policy) wins over the caller's timeout so a whole connect+transfer is
315 * abandoned at g_origin_attempt_ms — this is the "give up after Ns and let
316 * the fill loop retry on a fresh connection inside the client's window"
317 * knob. 0 = fall back to the caller's timeout (or libcurl's 60s default). */
318 {
319 long total_ms = (g_origin_attempt_ms > 0) ? g_origin_attempt_ms
320 : (timeout_ms > 0 ? timeout_ms : 60000);
321 curl_easy_setopt(curl, CURLOPT_TIMEOUT_MS, total_ms);
322 }
323 curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
324
325 if (g_origin_no_reuse) {
326 /* A/B / hostile-network escape hatch: behave like the old transport,
327 * one fresh connection per request (no stale-reuse risk, but pays the
328 * handshake + slow-start every time). */
329 curl_easy_setopt(curl, CURLOPT_FORBID_REUSE, 1L);
330 curl_easy_setopt(curl, CURLOPT_FRESH_CONNECT, 1L);
331 } else {
332 /* Reuse one warm connection per origin (the connection cache spans all
333 * endpoints so failover stays warm). Guard against a middlebox that
334 * reaps idle connections: probe with TCP keepalive every 15s so a dead
335 * connection is detected, and never reuse one idle longer than 20s —
336 * during a mount burst requests are back-to-back so reuse still lands,
337 * but a connection left idle past the danger window is reopened fresh. */
338 curl_easy_setopt(curl, CURLOPT_MAXCONNECTS, 16L);
339 curl_easy_setopt(curl, CURLOPT_TCP_KEEPALIVE, 1L);
340 curl_easy_setopt(curl, CURLOPT_TCP_KEEPIDLE, 15L);
341 curl_easy_setopt(curl, CURLOPT_TCP_KEEPINTVL, 15L);
342#ifdef CURLOPT_MAXAGE_CONN
343 curl_easy_setopt(curl, CURLOPT_MAXAGE_CONN, 20L);
344#endif
345 }
346
347 /* Fast-fail bounds (operator policy; 0 = leave libcurl's default). A stuck
348 * connect fails at g_origin_connect_ms; a connection that opens but then
349 * stays below g_origin_stall_bytes B/s for g_origin_stall_secs is aborted
350 * with CURLE_OPERATION_TIMEDOUT — the "stuck before any data" case. Both
351 * surface as a transport error the fill loop retries on a fresh handle. */
352 if (g_origin_connect_ms > 0) {
353 curl_easy_setopt(curl, CURLOPT_CONNECTTIMEOUT_MS, g_origin_connect_ms);
354 }
355 if (g_origin_stall_secs > 0 && g_origin_stall_bytes > 0) {
356 curl_easy_setopt(curl, CURLOPT_LOW_SPEED_TIME, g_origin_stall_secs);
357 curl_easy_setopt(curl, CURLOPT_LOW_SPEED_LIMIT, g_origin_stall_bytes);
358 }
359
360 if (strcmp(method, "HEAD") == 0) {
361 curl_easy_setopt(curl, CURLOPT_NOBODY, 1L);
362 } else if (strcmp(method, "GET") != 0) {
363 curl_easy_setopt(curl, CURLOPT_CUSTOMREQUEST, method);
364 if (body != NULL && body_len > 0) {
365 curl_easy_setopt(curl, CURLOPT_POSTFIELDS, body);
366 curl_easy_setopt(curl, CURLOPT_POSTFIELDSIZE_LARGE,
367 (curl_off_t) body_len);
368 } else {
369 curl_easy_setopt(curl, CURLOPT_POSTFIELDSIZE_LARGE, (curl_off_t) 0);
370 }
371 }
372
373 if (tls) {
374 /* Origin TLS verification is the operator's trust decision; default to
375 * verifying (a misconfigured origin should fail loudly, not silently). */
376 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYPEER, 1L);
377 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYHOST, 2L);
378 }
379
380 res = curl_easy_perform(curl);
381 if (res != CURLE_OK) {
382 if (errbuf && errcap) {
383 snprintf(errbuf, errcap, "curl: %s", curl_easy_strerror(res));
384 }
385 s3o_trace(method, host, port, path_and_query, -1, 0,
386 s3o_ms_since(&t0), curl_easy_strerror(res));
387 curl_slist_free_all(slist);
388 free(r->body);
389 free(r->hdrs);
390 free(r);
391 return -1; /* handle persists (curl drops a dead
392 connection from the pool by itself) */
393 }
394
395 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &status);
396 curl_slist_free_all(slist);
397
398 resp->status = (int) status;
399 resp->opaque = r;
400 s3o_trace(method, host, port, path_and_query, (int) status, r->body_len,
401 s3o_ms_since(&t0), NULL);
402 return 0;
403}

405static int
406s3o_resp_header(const brix_s3_resp_t *resp, const char *name,
407 char *out, size_t outcap)
408{
409 const s3o_resp_t *r = resp->opaque;
410 size_t namelen = strlen(name);
411 const char *p;
412
413 if (r == NULL || r->hdrs == NULL) {
414 return -1;
415 }
416 for (p = r->hdrs; *p != '\0'; ) {
417 const char *eol = strpbrk(p, "\r\n");
418 size_t linelen = (eol != NULL) ? (size_t) (eol - p) : strlen(p);
419
420 if (linelen > namelen && p[namelen] == ':'
421 && strncasecmp(p, name, namelen) == 0)
422 {
423 const char *v = p + namelen + 1;
424 size_t vlen;
425
426 while (v < p + linelen && (*v == ' ' || *v == '\t')) { v++; }
427 vlen = (size_t) (p + linelen - v);
428 if (vlen >= outcap) { vlen = outcap - 1; }
429 memcpy(out, v, vlen);
430 out[vlen] = '\0';
431 return 0;
432 }
433 if (eol == NULL) { break; }
434 p = eol + strspn(eol, "\r\n");
435 }
436 return -1;
437}
