28static long
29s3o_ms_since(const struct timespec *t0)
30{
31 struct timespec now;
32 long ms;
33
34 if (clock_gettime(CLOCK_MONOTONIC, &now) != 0) {
35 return 0;
36 }
37 ms = (now.tv_sec - t0->tv_sec) * 1000L
38 + (now.tv_nsec - t0->tv_nsec) / 1000000L;
39 return ms < 0 ? 0 : ms;
40}

42/* Append `n` bytes to a growable buffer; returns 0 / -1 (OOM). */
43static int
44s3o_buf_append(char **buf, size_t *len, size_t *cap, const char *src, size_t n)
45{
46 if (*len + n + 1 > *cap) {
47 size_t newcap = (*cap == 0) ? 8192 : *cap;
48 char *nb;
49
50 while (*len + n + 1 > newcap) {
51 newcap *= 2;
52 }
53 nb = realloc(*buf, newcap);
54 if (nb == NULL) {
55 return -1;
56 }
57 *buf = nb;
58 *cap = newcap;
59 }
60 memcpy(*buf + *len, src, n);
61 *len += n;
62 (*buf)[*len] = '\0';
63 return 0;
64}

90/* s3o_request_impl — the shared request body for the plain and cred-scoped
91 * transport slots.
92 *
93 * WHAT: Performs one synchronous libcurl request described by `req`, capturing
94 * the response into a heap s3o_resp_t handed back through `resp->opaque`.
95 * WHY: The GSI-over-https backend leg (phase-70 §5.1) authenticates the origin
96 * hop AS the end user by presenting the user's proxy PEM as the TLS client
97 * cert — the only structural difference from an anonymous/bearer request is
98 * the mutual-TLS options (in req->client_cert_pem), so both entry points
99 * share one body.
100 * HOW: Acquires this thread's persistent warm handle, builds the header list,
101 * configures every per-request option (s3o_configure), performs the
102 * transfer and emits the trace line. A per-request curl handle is reset
103 * between uses, so TLS/cert options do not leak into the next request. */
104static int
105s3o_request_impl(const s3o_request_t *req, brix_s3_resp_t *resp,
106 char *errbuf, size_t errcap)
107{
108 CURL *curl;
109 CURLcode res;
110 struct curl_slist *slist = NULL;
111 s3o_resp_t *r;
112 long status = 0;
113 struct timespec t0;
114 s3o_trace_t tr = { 0 };
115
116 (void) clock_gettime(CLOCK_MONOTONIC, &t0);
117
118 curl = s3o_curl_acquire(); /* this thread's persistent, warm handle */
119 if (curl == NULL) {
120 if (errbuf && errcap) { snprintf(errbuf, errcap, "curl handle unavailable"); }
121 return -1;
122 }
123
124 r = calloc(1, sizeof(*r));
125 if (r == NULL) {
126 if (errbuf && errcap) { snprintf(errbuf, errcap, "out of memory"); }
127 return -1; /* handle stays in TLS, reused next call */
128 }
129
130 slist = s3o_build_headers(req);
131 s3o_configure(curl, req, r, slist);
132
133 tr.method = req->method;
134 tr.host = req->host;
135 tr.port = req->port;
136 tr.path = req->path_and_query;
137
138 res = curl_easy_perform(curl);
139 if (res != CURLE_OK) {
140 if (errbuf && errcap) {
141 snprintf(errbuf, errcap, "curl: %s", curl_easy_strerror(res));
142 }
143 tr.status = -1;
144 tr.dur_ms = s3o_ms_since(&t0);
145 tr.err = curl_easy_strerror(res);
146 s3o_trace(&tr);
147 curl_slist_free_all(slist);
148 free(r->body);
149 free(r->hdrs);
150 free(r);
151 return -1; /* handle persists (curl drops a dead
152 connection from the pool by itself) */
153 }
154
155 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &status);
156 curl_slist_free_all(slist);
157
158 resp->status = (int) status;
159 resp->opaque = r;
160 tr.status = (int) status;
161 tr.bytes = r->body_len;
162 tr.dur_ms = s3o_ms_since(&t0);
163 tr.proto = s3o_negotiated_proto(curl);
164 s3o_trace(&tr);
165 return 0;
166}

218static int
219s3o_resp_header(const brix_s3_resp_t *resp, const char *name,
220 char *out, size_t outcap)
221{
222 const s3o_resp_t *r = resp->opaque;
223 size_t namelen = strlen(name);
224 const char *p;
225
226 if (r == NULL || r->hdrs == NULL) {
227 return -1;
228 }
229 for (p = r->hdrs; *p != '\0'; ) {
230 const char *eol = strpbrk(p, "\r\n");
231 size_t linelen = (eol != NULL) ? (size_t) (eol - p) : strlen(p);
232
233 if (linelen > namelen && p[namelen] == ':'
234 && strncasecmp(p, name, namelen) == 0)
235 {
236 const char *v = p + namelen + 1;
237 size_t vlen;
238
239 while (v < p + linelen && (*v == ' ' || *v == '\t')) { v++; }
240 vlen = (size_t) (p + linelen - v);
241 if (vlen >= outcap) { vlen = outcap - 1; }
242 memcpy(out, v, vlen);
243 out[vlen] = '\0';
244 return 0;
245 }
246 if (eol == NULL) { break; }
247 p = eol + strspn(eol, "\r\n");
248 }
249 return -1;
250}

261static const void *
262s3o_resp_body(const brix_s3_resp_t *resp, size_t *len)
263{
264 const s3o_resp_t *r = resp->opaque;
265
266 if (r == NULL) {
267 if (len) { *len = 0; }
268 return NULL;
269 }
270 if (len) { *len = r->body_len; }
271 return r->body;
272}

274static void
275s3o_resp_free(brix_s3_resp_t *resp)
276{
277 s3o_resp_t *r;
278
279 if (resp == NULL || resp->opaque == NULL) {
280 return;
281 }
282 r = resp->opaque;
283 free(r->body);
284 free(r->hdrs);
285 free(r);
286 resp->opaque = NULL;
287}
