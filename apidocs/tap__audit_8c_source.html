2 * tap_audit.c — single-line JSON formatter for a tapped frame.
3 *
4 * Pure string building into a caller buffer (no I/O, no allocation): the consumer
5 * decides where the line goes (error.log, a dedicated audit file). Path bytes from
6 * the wire are JSON-escaped (", \\, and control bytes) and bounded by path_len —

38/* Append a JSON-escaped, length-bounded string value. Returns 0 on overflow. */
39static int
40tap_json_append_escaped(char *out, size_t outsz, size_t *pos,
41 const uint8_t *s, size_t slen)
42{
43 size_t i;
44 for (i = 0; i < slen; i++) {
45 unsigned char ch = s[i];
46 char esc[8];
47 const char *seg;
48 size_t seglen;
49 if (ch == '"' || ch == '\\') {
50 esc[0] = '\\'; esc[1] = (char) ch; seglen = 2; seg = esc;
51 } else if (ch < 0x20) {
52 seglen = (size_t) snprintf(esc, sizeof(esc), "\\u%04x", ch);
53 seg = esc;
54 } else {
55 esc[0] = (char) ch; seglen = 1; seg = esc;
56 }
57 if (*pos + seglen >= outsz) { return 0; }
58 memcpy(out + *pos, seg, seglen);
59 *pos += seglen;
60 }
61 return 1;
62}

64size_t
65brix_tap_audit_format(const brix_tap_frame_t *f, brix_tap_dir_t dir,
66 char *out, size_t outsz)
67{
68 const char *dirs = (dir == BRIX_TAP_C2U) ? "c2u" : "u2c";
69 const char *opn;
70 size_t pos;
71 int n;
72
73 if (f == NULL || out == NULL || outsz == 0) {
74 return 0;
75 }
76
77 /* Fixed prefix + numeric fields via snprintf (bounded). */
78 if (f->is_request) {
79 opn = tap_opcode_name(f->opcode);
80 if (opn != NULL) {
81 n = snprintf(out, outsz,
82 "{\"dir\":\"%s\",\"streamid\":%u,\"op\":\"%s\",\"dlen\":%u",
83 dirs, f->streamid, opn, f->dlen);
84 } else {
85 n = snprintf(out, outsz,
86 "{\"dir\":\"%s\",\"streamid\":%u,\"op\":%u,\"dlen\":%u",
87 dirs, f->streamid, f->opcode, f->dlen);
88 }
89 } else {
90 n = snprintf(out, outsz,
91 "{\"dir\":\"%s\",\"streamid\":%u,\"status\":%u,\"dlen\":%u",
92 dirs, f->streamid, f->status, f->dlen);
93 }
94 if (n < 0 || (size_t) n >= outsz) {
95 return 0;
96 }
97 pos = (size_t) n;
98
99 /* Optional escaped path. */
100 if (f->path != NULL && f->path_len > 0) {
101 const char *pkey = ",\"path\":\"";
102 size_t klen = strlen(pkey);
103 if (pos + klen >= outsz) { return 0; }
104 memcpy(out + pos, pkey, klen);
105 pos += klen;
106 if (!tap_json_append_escaped(out, outsz, &pos, f->path, f->path_len)) {
107 return 0;
108 }
109 if (pos + 1 >= outsz) { return 0; }
110 out[pos++] = '"';
111 }
112
113 if (pos + 1 >= outsz) { return 0; } /* room for '}' + NUL */
114 out[pos++] = '}';
115 out[pos] = '\0';
116 return pos;
117}
