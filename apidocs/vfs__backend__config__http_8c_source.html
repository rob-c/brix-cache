2 * vfs_backend_config_http.c — the http(s):// origin parser (with T11
3 * multi-endpoint failover lists) and the http per-driver registry-entry
4 * builder (phase-79 file-size split of vfs_backend_config.c).
5 *
6 * WHAT: Parses "http://host[:port][/base]" / "https://..." backend origins —
7 * including a pipe-separated ordered list that registers a ranked
8 * multi-endpoint origin set (phase-68 T11) — into registry entries, and
9 * registers each endpoint (primary via brix_vfs_backend_config_http,
10 * failovers via the http_extra list).

13 * every file under the 500-line cap with byte-for-byte identical parse

25/* Register a read-only HTTP source backend for an export (endpoint 0). host/port
26 * are required; base_path is the URL base ("" / "/sub"). Resets the T11 failover
27 * list so a reload re-registers cleanly. */
28void
29brix_vfs_backend_config_http(const char *root_canon, const char *host,
30 int port, int tls, const char *base_path, int put_checksum)
31{
32 brix_vfs_backend_entry_t *e;
33
34 if (root_canon == NULL || root_canon[0] == '\0' || host == NULL
35 || host[0] == '\0' || port <= 0 || port > 65535)
36 {
37 return;
38 }
39 e = brix_vfs_backend_entry_get_or_create(root_canon);
40 if (e == NULL) {
41 return;
42 }
43 ngx_memcpy(e->backend, "http", sizeof("http"));
44 ngx_cpystrn((u_char *) e->origin_host, (u_char *) host,
45 sizeof(e->origin_host));
46 e->origin_port = port;
47 e->origin_tls = tls;
48 ngx_cpystrn((u_char *) e->origin_path, (u_char *) (base_path ? base_path : ""),
49 sizeof(e->origin_path));
50 e->origin_put_checksum = put_checksum ? 1 : 0; /* #12 */
51 e->n_http_extra = 0; /* reload resets the T11 list */
52 e->inst = NULL; /* rebuilt on next resolve */
53}

55/* 1 iff `data`/`len` starts with an http:// or https:// scheme. */
56static int
57vfs_backend_is_http_url(const u_char *data, size_t len)
58{
59 return (len > sizeof("http://") - 1
60 && ngx_strncmp(data, "http://", sizeof("http://") - 1) == 0)
61 || (len > sizeof("https://") - 1
62 && ngx_strncmp(data, "https://", sizeof("https://") - 1) == 0);
63}

65/* Strip a leading http:// or https:// scheme off one origin segment, returning
66 * the host-and-path remainder plus its TLS flag. Split out of the origin parser
67 * so each function stays within the readability gate. */
68static ngx_int_t
69vfs_http_origin_strip_scheme(ngx_conf_t *cf, u_char *seg, size_t segn,
70 vfs_origin_parse_t *out)
71{
72 if (segn > sizeof("https://") - 1
73 && ngx_strncmp(seg, "https://", sizeof("https://") - 1) == 0)
74 {
75 out->rest = seg + sizeof("https://") - 1;
76 out->rest_len = segn - (sizeof("https://") - 1);
77 out->tls = 1;
78 return NGX_OK;
79 }
80 if (segn > sizeof("http://") - 1
81 && ngx_strncmp(seg, "http://", sizeof("http://") - 1) == 0)
82 {
83 out->rest = seg + sizeof("http://") - 1;
84 out->rest_len = segn - (sizeof("http://") - 1);
85 out->tls = 0;
86 return NGX_OK;
87 }
88 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
89 "brix_storage_backend: every |-separated origin must be an "
90 "http:// or https:// URL");
91 return NGX_ERROR;
92}

94/* Split the host-and-port prefix "host[:port]" (length hplen) into host length
95 * and numeric port, defaulting to the scheme port when no ":port" is present.
96 * Split out of the origin parser to keep its branching within the readability
97 * gate. Returns NGX_OK, or NGX_ERROR after an [emerg] for a bad port. */
98static ngx_int_t
99vfs_http_origin_host_port(ngx_conf_t *cf, u_char *h, size_t hplen, int htls,
100 vfs_origin_parse_t *out)
101{
102 u_char *colon = NULL;
103 size_t i;
104
105 for (i = hplen; i > 0; i--) {
106 if (h[i - 1] == ':') { colon = h + i - 1; break; }
107 }
108 if (colon != NULL) {
109 ngx_int_t pn = ngx_atoi(colon + 1, (size_t) (h + hplen - (colon + 1)));
110
111 if (pn == NGX_ERROR || pn <= 0 || pn > 65535) {
112 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
113 "brix_storage_backend: invalid http origin port");
114 return NGX_ERROR;
115 }
116 out->host_len = (size_t) (colon - h);
117 out->port = (int) pn;
118 return NGX_OK;
119 }
120 out->host_len = hplen;
121 out->port = htls ? 443 : 80;
122 return NGX_OK;
123}

125/* Parse ONE "http(s)://host[:port][/base]" segment into host/port/tls/base.
126 * Returns NGX_OK, or NGX_ERROR after an [emerg] for a malformed segment. */
127static ngx_int_t
128vfs_parse_http_origin(ngx_conf_t *cf, u_char *seg, size_t segn,
129 vfs_origin_parse_t *out)
130{
131 u_char *h;
132 size_t hn;
133 size_t i, slash, hplen, pathlen, hostn;
134 const char *path;
135
136 if (vfs_http_origin_strip_scheme(cf, seg, segn, out) != NGX_OK) {
137 return NGX_ERROR;
138 }
139 h = out->rest;
140 hn = out->rest_len;
141
142 slash = hn;
143 for (i = 0; i < hn; i++) {
144 if (h[i] == '/') { slash = i; break; }
145 }
146 hplen = slash;
147 path = (slash < hn) ? (const char *) (h + slash) : "";
148 pathlen = (slash < hn) ? hn - slash : 0;
149
150 /* Trim an optional "?opt=val" query suffix (e.g. ?put_checksum=1, #12) off the
151 * URL base so the write-target path stays clean; the list parser scans the raw
152 * segment for the option itself. */
153 {
154 size_t q;
155 for (q = 0; q < pathlen; q++) {
156 if (path[q] == '?') { pathlen = q; break; }
157 }
158 }
159
160 if (vfs_http_origin_host_port(cf, h, hplen, out->tls, out) != NGX_OK) {
161 return NGX_ERROR;
162 }
163 hostn = out->host_len;
164 if (hostn == 0 || hostn >= out->host_cap || pathlen >= out->base_cap) {
165 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
166 "brix_storage_backend: invalid http origin host/path");
167 return NGX_ERROR;
168 }
169 ngx_memcpy(out->host, h, hostn);
170 out->host[hostn] = '\0';
171 if (pathlen > 0) {
172 ngx_memcpy(out->base, path, pathlen);
173 out->base[pathlen] = '\0';
174 } else {
175 out->base[0] = '\0';
176 }
177 return NGX_OK;
178}

180/* Append one failover endpoint to the export's http backend entry (phase-68
181 * T11). Must run AFTER brix_vfs_backend_config_http registered endpoint 0
182 * for the same root. */
183static ngx_int_t
184vfs_backend_add_http_endpoint(ngx_conf_t *cf, const char *root_canon,
185 const vfs_origin_parse_t *parsed)
186{
187 brix_vfs_backend_entry_t *e = brix_vfs_backend_entry_find(root_canon);
188 size_t n;
189
190 if (e == NULL || ngx_strcmp(e->backend, "http") != 0) {
191 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
192 "brix_storage_backend: internal - endpoint list before primary");
193 return NGX_ERROR;
194 }
195 n = sizeof(e->http_extra) / sizeof(e->http_extra[0]);
196 if ((size_t) e->n_http_extra >= n) {
197 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
198 "brix_storage_backend: too many |-separated origins (max %uz)",
199 n + 1);
200 return NGX_ERROR;
201 }
202 if (ngx_strlen(parsed->host) >= sizeof(e->http_extra[0].host)
203 || ngx_strlen(parsed->base) >= sizeof(e->http_extra[0].base))
204 {
205 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
206 "brix_storage_backend: origin host/base too long for the "
207 "endpoint table");
208 return NGX_ERROR;
209 }
210 ngx_cpystrn((u_char *) e->http_extra[e->n_http_extra].host,
211 (u_char *) parsed->host, sizeof(e->http_extra[0].host));
212 e->http_extra[e->n_http_extra].port = parsed->port;
213 e->http_extra[e->n_http_extra].tls = parsed->tls;
214 ngx_cpystrn((u_char *) e->http_extra[e->n_http_extra].base,
215 (u_char *) parsed->base, sizeof(e->http_extra[0].base));
216 e->n_http_extra++;
217 e->inst = NULL; /* rebuilt on next resolve */
218 return NGX_OK;
219}

221/* "http://host[:port]/base" / "https://..." → a read-only HTTP SOURCE backend,
222 * as a pipe-separated ordered failover list (phase-68 T11). Returns NGX_OK on a
223 * handled set, NGX_ERROR after an [emerg] for a malformed one, or NGX_DECLINED
224 * when the scheme is not ours. */
225ngx_int_t
226vfs_parse_http_origin_list(ngx_conf_t *cf, const char *root_canon,
227 const ngx_str_t *sb)
228{
229 /* "http://host[:port]/base" / "https://..." → a read-only HTTP SOURCE
230 * backend. phase-68 T11: a PIPE-SEPARATED ordered list of such URLs
231 * registers a multi-endpoint origin set with health-scored read failover
232 * (mirrors the CVMFS_SERVER_URL syntax); the first URL is the primary
233 * and the write target. */
234 if (vfs_backend_is_http_url(sb->data, sb->len)) {
235 u_char *seg = sb->data;
236 u_char *end = sb->data + sb->len;
237 int first = 1;
238 /* #12: opt-in origin-enforced body integrity. "...?put_checksum=1" makes the
239 * commit PUT (endpoint 0, the write target) carry Content-MD5 so the origin
240 * validates the bytes and rejects a wire-corrupted upload. Off by default:
241 * an origin that ignores Content-MD5 stays working untouched. */
242 int put_checksum = (ngx_strstr(sb->data, "put_checksum=1") != NULL);
243
244 while (seg < end) {
245 u_char *pipe = ngx_strlchr(seg, end, '|');
246 size_t segn = (pipe != NULL) ? (size_t) (pipe - seg)
247 : (size_t) (end - seg);
248 char host[256], base[1024];
249 vfs_origin_parse_t parsed;
250
251 ngx_memzero(&parsed, sizeof(parsed));
252 parsed.host = host;
253 parsed.host_cap = sizeof(host);
254 parsed.base = base;
255 parsed.base_cap = sizeof(base);
256
257 if (vfs_parse_http_origin(cf, seg, segn, &parsed) != NGX_OK) {
258 return NGX_ERROR; /* [emerg] already logged */
259 }
260 if (first) {
261 brix_vfs_backend_config_http(root_canon, parsed.host,
262 parsed.port, parsed.tls,
263 parsed.base, put_checksum);
264 first = 0;
265 } else if (vfs_backend_add_http_endpoint(cf, root_canon, &parsed)
266 != NGX_OK)
267 {
268 return NGX_ERROR;
269 }
270 seg = (pipe != NULL) ? pipe + 1 : end;
271 }
272 return NGX_OK;
273 }
274
275
276 return NGX_DECLINED;
277}
