1/*
2 * WHAT: Export stream-side user identity and VO (virtual organisation) tracking
3 * metrics to Prometheus text exposition format.
4 *
5 * WHY: XRootD sessions carry VOMS attributes (VO membership) and user identities
6 * (DN or token sub). Tracking these per-session allows operators to see which
7 * VO groups consume most bandwidth, how many unique users connect over time,
8 * and whether the bounded tracking tables are overflowing. Unlike request-level
9 * metrics this is a gauge-style view of active session identity distribution.
10 *
11 * HOW: Iterates two bounded LRU tables in shared memory (vo_global + user_tracking)
12 * using atomic fetch-add(0) for lock-free reads. Emit four metric families:
13 * 1) brix_vo_bytes_tx/rx_total — per-VO byte counters (counter)
14 * 2) brix_vo_requests_total — per-VO request count (counter)
15 * 3) brix_vo_overflow_total — evicted VO entries since process start (counter)
16 * 4) brix_unique_users_current/gauge + total/counter + evictions/counter
17 * — active users, lifetime users, and LRU eviction count (gauge + counter)
18 * 5) brix_user_sessions_total{hash} — session count per hashed user ID (gauge)
19 * VO names truncated to BRIX_VO_NAME_LEN characters; users identified by
20 * FNV-1a hash of DN or token sub. Empty slots are skipped.
21 */
22
23#include "metrics_internal.h"

25void
26brix_export_stream_tracking_metrics(metrics_writer_t *mw,
27 ngx_brix_metrics_t *shm)
28{
29 ngx_uint_t i;
30
31 mw_printf(mw,
32 "# HELP brix_vo_bytes_tx_total "
33 "Bytes sent to clients grouped by virtual organisation. "
34 "VO names are truncated to %d characters; the metric family has one entry per VO.\n"
35 "# TYPE brix_vo_bytes_tx_total counter\n", BRIX_VO_NAME_LEN - 1);
36 for (i = 0; i < BRIX_VO_MAX_TRACKED; i++) {
37 ngx_brix_vo_slot_t *vo = &shm->vo_global.slots[i];
38 if (!vo->name[0]) { continue; }
39 mw_printf(mw, "brix_vo_bytes_tx_total{vo=\"%s\"} %lu\n",
40 vo->name, (unsigned long) ngx_atomic_fetch_add(&vo->bytes_tx_total, 0));
41 }
42
43 mw_printf(mw,
44 "# HELP brix_vo_bytes_rx_total "
45 "Bytes received from clients grouped by virtual organisation. "
46 "VO names are truncated to %d characters.\n"
47 "# TYPE brix_vo_bytes_rx_total counter\n", BRIX_VO_NAME_LEN - 1);
48 for (i = 0; i < BRIX_VO_MAX_TRACKED; i++) {
49 ngx_brix_vo_slot_t *vo = &shm->vo_global.slots[i];
50 if (!vo->name[0]) { continue; }
51 mw_printf(mw, "brix_vo_bytes_rx_total{vo=\"%s\"} %lu\n",
52 vo->name, (unsigned long) ngx_atomic_fetch_add(&vo->bytes_rx_total, 0));
53 }
54
55 mw_printf(mw,
56 "# HELP brix_vo_requests_total "
57 "Requests grouped by virtual organisation. VO names are truncated.\n"
58 "# TYPE brix_vo_requests_total counter\n");
59 for (i = 0; i < BRIX_VO_MAX_TRACKED; i++) {
60 ngx_brix_vo_slot_t *vo = &shm->vo_global.slots[i];
61 if (!vo->name[0]) { continue; }
62 mw_printf(mw, "brix_vo_requests_total{vo=\"%s\"} %lu\n",
63 vo->name, (unsigned long) ngx_atomic_fetch_add(&vo->requests_total, 0));
64 }
65
66 mw_printf(mw,
67 "# HELP brix_vo_overflow_total "
68 "VO entries that exceeded the tracking limit and were evicted.\n"
69 "# TYPE brix_vo_overflow_total counter\n");
70 mw_printf(mw, "brix_vo_overflow_total %lu\n",
71 (unsigned long) ngx_atomic_fetch_add(&shm->vo_global.overflow_total, 0));
72
73 mw_printf(mw,
74 "# HELP brix_unique_users_current "
75 "Currently tracked unique user identities (bounded LRU, max %d). "
76 "Users are identified by DN or token sub via FNV-1a hash.\n"
77 "# TYPE brix_unique_users_current gauge\n", BRIX_USERS_MAX_TRACKED);
78 mw_printf(mw, "brix_unique_users_current %lu\n",
79 (unsigned long) ngx_atomic_fetch_add(&shm->user_tracking.unique_count, 0));
80
81 mw_printf(mw,
82 "# HELP brix_unique_users_total "
83 "Lifetime unique user identities seen since process start. "
84 "Never decremented.\n"
85 "# TYPE brix_unique_users_total counter\n");
86 mw_printf(mw, "brix_unique_users_total %lu\n",
87 (unsigned long) ngx_atomic_fetch_add(&shm->user_tracking.total_unique, 0));
88
89 mw_printf(mw,
90 "# HELP brix_user_evictions_total "
91 "User identity slots evicted from the tracking table.\n"
92 "# TYPE brix_user_evictions_total counter\n");
93 mw_printf(mw, "brix_user_evictions_total %lu\n",
94 (unsigned long) ngx_atomic_fetch_add(&shm->user_tracking.evictions_total, 0));
95
96 mw_printf(mw,
97 "# HELP brix_user_sessions_total "
98 "Sessions per tracked user identity. Sum across all entries equals total authenticated sessions.\n"
99 "# TYPE brix_user_sessions_total gauge\n");
100 for (i = 0; i < BRIX_USERS_MAX_TRACKED; i++) {
101 ngx_brix_user_slot_t *u = &shm->user_tracking.slots[i];
102 if (!u->id_hash) { continue; }
103 mw_printf(mw, "brix_user_sessions_total{hash=%08x} %lu\n",
104 u->id_hash, (unsigned long) ngx_atomic_fetch_add(&u->sessions_total, 0));
105 }
