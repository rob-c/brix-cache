4/*
5 * ssi_service.h — native compiled-in SSI service interface + registry.
6 *
7 * WHAT: a request handler produces a response through a responder callback set
8 * (Approach 1 in the design). Synchronous services call set_response inline;
9 * async/streaming services retain the responder and drive it from a
10 * thread-pool job. This replaces the C++ XrdSsiService/XrdSsiResponder ABI
11 * with a native C interface.
12 * WHY: byte-exact SSI without the plugin ABI; the registry resolves the open
13 * resource path "/<service>" to a handler.
14 * HOW: pure C so the handlers + registry are unit-testable standalone with a
15 * recording responder.
16 */

22/*
23 * Responder ops a service calls to deliver its result.
24 * set_metadata : optional metadata blob (delivered before/with the response).
25 * set_response : response data; last=1 marks the final chunk (last=0 streams).
26 * alert : out-of-band progress alert.
27 * error : terminal error with an SSI error code + text.
28 * All borrow the passed buffers for the duration of the call.
29 */

33 void (*set_response)(brix_ssi_responder_t *r,

36 const unsigned char *buf, size_t len);
37 void (*error)(brix_ssi_responder_t *r, int code, const char *text);
38 /*
39 * defer: a service that wants to answer later calls this. Returns 0 if the
40 * deferral was accepted (submit phase — the server replies kXR_waitresp and
41 * the service is re-invoked later to produce its response), or -1 if deferral
42 * is unavailable (completion phase — the service must respond inline now).
43 * A service that ignores defer stays fully synchronous (Phase-1 behaviour).
44 */
45 int (*defer)(brix_ssi_responder_t *r);
46 /*
47 * svc_slot: returns a pointer to a per-request void* cookie that persists from
48 * the submit call to the deferred completion call, letting a stateful service
49 * correlate the two phases (e.g. stash its request-queue entry). May be NULL
50 * if the host does not provide per-request service state.
51 */
52 void **(*svc_slot)(brix_ssi_responder_t *r);
53 void *state; /* responder implementation private data */
54};
