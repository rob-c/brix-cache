5/*
6 * opaque_validate.c — byte-hygiene gate for the XRootD CGI opaque string
7 * (hyper-hardening-plan §D-2, CWE-88/CWE-93/CWE-117 injection at the parse edge).
8 *
9 * WHAT: brix_opaque_illegal_byte() scans a NUL-terminated opaque query string
10 * (everything after '?' in a wire path: oss.* / tpc.* / auth.* / xrd.* /
11 * xrdcl.* key=value pairs) and reports the first byte outside the set a
12 * well-formed opaque needs. The caller rejects the request (kXR_ArgInvalid)
13 * before any handler parses, logs, or forwards the string.
14 *
15 * WHY: The opaque is consumed by several handlers (TPC source/dest URLs,
16 * delegated-token modes, compression negotiation, ZIP member selection)
17 * and is logged and, for TPC, spliced into an OUTBOUND request. A raw
18 * control byte is a log/CRLF-injection and request-smuggling primitive; a
19 * shell/quoting metacharacter is a command-injection primitive if any value
20 * ever reaches a shell; a high-bit byte is non-conforming and a mojibake /
21 * filter-evasion vector. None of these appear in a legitimate opaque — real
22 * clients percent-encode anything outside the unreserved/structural set — so
23 * rejecting them at the single parse edge is a zero-false-positive gate that
24 * stops the whole class before it fans out to the handlers.
25 *
26 * HOW: A 256-entry permit table, built once from an explicit allow string, is
27 * consulted per byte. Permitted = URL-unreserved (alnum . - _ ~),
28 * path/authority (/ : @), percent-encoding (% +), and CGI structure
29 * (= & ; , ?). ';' is the legacy CGI pair separator (equivalent to '&');
30 * stock XRootD accepts "k=v;other=z", so rejecting it would break parity
31 * and is not an injection concern here — the opaque is parsed, never handed
32 * to a shell. Everything else — 0x00-0x1F, 0x7F, 0x80-0xFF, and the printable
33 * metacharacters space ! " # $ ' ( ) * < > [ \ ] ^ ` { | } — is rejected.
34 * Pure C, no nginx/libc string deps: safe to call from the wire edge.
35 */
36
37/* The bytes a legitimate XRootD CGI opaque is built from. Anything else is an
38 * injection primitive that a conforming client would percent-encode. Note '?'
39 * is permitted because a tpc.src value may carry a nested URL query. */
40static const char BRIX_OPAQUE_ALLOWED[] =
41 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
42 "abcdefghijklmnopqrstuvwxyz"
43 "0123456789"
44 ".-_~" /* URL unreserved */
45 "/:@[]" /* path / authority — [] bracket an IPv6 literal host (RFC 3986
46 * IP-literal), which XrdCl emits in a tpc.src URL for an IPv6 (or
47 * IPv4-mapped, "[::ffff:127.0.0.1]") source; parse.c/launch.c
48 * decompose and re-bracket it, so the bytes are legitimate and
49 * required for native-TPC parity — not shell metacharacters here
50 * (the opaque is parsed, never handed to a shell). */
51 "%+" /* percent-encoding, plus-as-space */
52 "=&,?"; /* CGI structure: '&' key/val separator, '=' assignment, ',' list,
53 * '?' nested-query. NOT ';' — XRootD splits opaque only on '&'
54 * (XrdOucEnv.cc and XrdCl URL::SetParams both tokenize on "&"
55 * exclusively), so ';' is never a legitimate separator on the wire;
56 * permitting it only widened the injection surface (command
57 * chaining) with no interop benefit. */
58

73/*
74 * Scan opaque[] for the first byte outside the permitted set. Returns 1 and
75 * writes the offending byte to *bad (if non-NULL) on rejection; returns 0 when
76 * every byte is permitted (including the empty string).
77 *
78 * The permit table has no mutable state after the one-time fill, so the lazy
79 * init is race-benign under nginx's per-worker single-threaded event loop: a
80 * concurrent init would recompute the identical table. (TPC pull threads never
81 * reach here — validation runs on the main-loop open edge before dispatch.)
82 */
83int
84brix_opaque_illegal_byte(const char *opaque, unsigned char *bad)
85{
86 const unsigned char *s;
87
88 if (opaque == NULL) {
89 return 0;
90 }
91 if (!brix_opaque_permit_ready) {
92 brix_opaque_permit_init();
93 }
94
95 for (s = (const unsigned char *) opaque; *s != '\0'; s++) {
96 if (!brix_opaque_permit[*s]) {
97 if (bad != NULL) {
98 *bad = *s;
99 }
100 return 1;
101 }
102 }
103 return 0;
104}

106/* ------------------------------------------------------------------------- *
107 * Tier 2 — opt-in schema validation (brix_opaque_strict).
108 *
109 * XRootD opaque keys are namespaced "<ns>.<name>" (plus a couple of bare keys
110 * a stock client sends). Strict mode recognizes the known namespaces and bare
111 * keys and rejects anything else, and type-enforces the keys brix assigns a
112 * type to. It is OFF by default; stock accepts these unchecked, so this only
113 * bites a deployment that deliberately opts into the stricter posture.
114 *
115 * SCOPE (documented, not a bug): the walk splits on top-level '&'. A value that
116 * embeds a nested URL query (a tpc.src / tpc.dst carrying "root://h//p?a=b&c=d")
117 * would have its nested '&c=d' seen as a sibling pair. Conforming clients
118 * percent-encode a nested query, and TPC endpoints pass source/dest CGI in the
119 * dedicated tpc.scgi / tpc.dcgi keys, so this does not arise in practice — but
120 * it is why strict mode is opt-in rather than always-on.
121 * ------------------------------------------------------------------------- */
122

209/*
210 * Validate one "key=value" segment[0..seg_len). Returns a verdict and, on a
211 * violation, copies the offending key into keybuf. An empty segment (from a
212 * stray "&&" / trailing '&') is vacuously OK.
213 */
214static int
215brix_opaque_check_segment(const char *seg, size_t seg_len, char *keybuf,
216 size_t keybuf_len)
217{
218 size_t key_len;
219 const char *val;
220 size_t val_len;
221 size_t i;
222
223 if (seg_len == 0) {
224 return BRIX_OPAQUE_SCHEMA_OK;
225 }
226
227 key_len = brix_opaque_key_len(seg, seg_len);
228 val = seg + key_len + (key_len < seg_len ? 1 : 0);
229 val_len = seg_len - key_len - (key_len < seg_len ? 1 : 0);
230
231 /* Typed keys: brix assigns oss.asize an unsigned-integer type. A typed key
232 * is by definition recognized, so this also satisfies the namespace rule. */
233 if (brix_opaque_key_eq(seg, key_len, "oss.asize")) {
234 if (!brix_opaque_is_uint(val, val_len)) {
235 brix_opaque_copy_key(seg, key_len, keybuf, keybuf_len);
236 return BRIX_OPAQUE_SCHEMA_BAD_TYPE;
237 }
238 return BRIX_OPAQUE_SCHEMA_OK;
239 }
240
241 for (i = 0; BRIX_OPAQUE_NAMESPACES[i] != NULL; i++) {
242 if (brix_opaque_key_has_prefix(seg, key_len,
243 BRIX_OPAQUE_NAMESPACES[i])) {
244 return BRIX_OPAQUE_SCHEMA_OK;
245 }
246 }
247 for (i = 0; BRIX_OPAQUE_BARE_KEYS[i] != NULL; i++) {
248 if (brix_opaque_key_eq(seg, key_len, BRIX_OPAQUE_BARE_KEYS[i])) {
249 return BRIX_OPAQUE_SCHEMA_OK;
250 }
251 }
252
253 brix_opaque_copy_key(seg, key_len, keybuf, keybuf_len);
254 return BRIX_OPAQUE_SCHEMA_UNKNOWN_KEY;
255}

257int
258brix_opaque_schema_check(const char *opaque, char *keybuf, size_t keybuf_len)
259{
260 const char *seg;
261 size_t remaining;
262
263 if (keybuf != NULL && keybuf_len > 0) {
264 keybuf[0] = '\0';
265 }
266 if (opaque == NULL) {
267 return BRIX_OPAQUE_SCHEMA_OK;
268 }
269 if (*opaque == '?') { /* tolerate a stray leading '?' */
270 opaque++;
271 }
272
273 seg = opaque;
274 for (remaining = 0; seg[remaining] != '\0'; remaining++) { /* strlen */ }
275
276 while (remaining > 0) {
277 const char *amp = NULL;
278 size_t seg_len;
279 size_t i;
280 int verdict;
281
282 for (i = 0; i < remaining; i++) {
283 if (seg[i] == '&' || seg[i] == ';') { /* '&' and legacy ';' both split */
284 amp = seg + i;
285 break;
286 }
287 }
288 seg_len = (amp != NULL) ? (size_t) (amp - seg) : remaining;
289
290 verdict = brix_opaque_check_segment(seg, seg_len, keybuf, keybuf_len);
291 if (verdict != BRIX_OPAQUE_SCHEMA_OK) {
292 return verdict;
293 }
294
295 if (amp == NULL) {
296 break;
297 }
298 remaining -= seg_len + 1;
299 seg = amp + 1;
300 }
301
302 return BRIX_OPAQUE_SCHEMA_OK;
303}
