8/* `authdb <path>` — load identity-based ACL rules. Requires brix_auth gsi,
9 * token, or both; stores the path and parses its entries into authdb_rules.
10 * Returns NGX_CONF_OK, or NGX_CONF_ERROR (emerg-logged) on a bad prerequisite
11 * or parse error. */
12char *
13brix_conf_set_authdb(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
14{
15 ngx_stream_brix_srv_conf_t *xcf = conf;
16 ngx_str_t *value;
17
18 value = cf->args->elts;
19
20 /*
21 * The auth-mode requirement (authdb needs gsi/token for the native engine)
22 * is validated at merge time, where `brix_authdb_format` is final — the
23 * xrdacc engine also authorizes anonymous `u *` rules, so it is exempt.
24 * (Directive order means xcf->auth / acc_format are not yet settled here.)
25 */
26 xcf->authdb = value[1];
27
28 if (xcf->authdb_rules == NULL) {
29 xcf->authdb_rules = ngx_array_create(cf->pool, 4,
30 sizeof(brix_authdb_rule_t));
31 if (xcf->authdb_rules == NULL) {
32 return NGX_CONF_ERROR;
33 }
34 }
35
36 if (brix_parse_authdb(cf, &xcf->authdb, xcf->authdb_rules) != NGX_OK) {
37 return NGX_CONF_ERROR;
38 }
39
40 return NGX_CONF_OK;
41}

43/* `require_vo <vo>[ <path>]` — append a VO ACL rule to vo_rules. Returns
44 * NGX_CONF_OK, or NGX_CONF_ERROR (emerg-logged) on bad args. */
45char *
46brix_conf_set_require_vo(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
47{
48 ngx_stream_brix_srv_conf_t *xcf = conf;
49 ngx_str_t *value;
50 brix_vo_rule_t *rule;
51
52 value = cf->args->elts;
53 (void) cmd;
54
55 if (xcf->vo_rules == NULL) {
56 xcf->vo_rules = ngx_array_create(cf->pool, 2, sizeof(brix_vo_rule_t));
57 if (xcf->vo_rules == NULL) {
58 return NGX_CONF_ERROR;
59 }
60 }
61
62 rule = ngx_array_push(xcf->vo_rules);
63 if (rule == NULL) {
64 return NGX_CONF_ERROR;
65 }
66
67 ngx_memzero(rule, sizeof(*rule));
68
69 if (brix_normalize_policy_path(cf->pool, &value[1], &rule->path) != NGX_OK) {
70 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
71 "brix_require_vo: invalid path \"%V\"", &value[1]);
72 return NGX_CONF_ERROR;
73 }
74
75 if (brix_copy_conf_string(cf, &value[2], &rule->vo) != NGX_CONF_OK) {
76 return NGX_CONF_ERROR;
77 }
78
79 return NGX_CONF_OK;

119/* Postconfiguration finalization for the policy rules: validate cross-directive
120 * prerequisites once every directive has settled. Returns NGX_OK / NGX_ERROR. */
121ngx_int_t
122brix_config_finalize_policy(ngx_conf_t *cf,
123 ngx_stream_brix_srv_conf_t *xcf)
124{
125 if (xcf->vo_rules != NULL
126 && xcf->auth != BRIX_AUTH_GSI
127 && xcf->auth != BRIX_AUTH_TOKEN
128 && xcf->auth != BRIX_AUTH_BOTH)
129 {
130 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
131 "brix_require_vo requires brix_auth gsi, token or both");
132 return NGX_ERROR;
133 }
134
135 if (xcf->vo_rules != NULL) {
136 if (!brix_voms_available()) {
137 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
138 "brix_require_vo requires libvomsapi.so.1 at runtime "
139 "(install voms-libs on EL9)");
140 return NGX_ERROR;
141 }
142 if (xcf->vomsdir.len == 0 || xcf->voms_cert_dir.len == 0) {
143 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
144 "brix_require_vo requires brix_vomsdir and brix_voms_cert_dir");
145 return NGX_ERROR;
146 }
147
148 if (brix_validate_path(cf, "brix_vomsdir", &xcf->vomsdir,
149 BRIX_PATH_DIRECTORY, R_OK | X_OK)
150 != NGX_OK
151 || brix_validate_path(cf, "brix_voms_cert_dir",
152 &xcf->voms_cert_dir,
153 BRIX_PATH_DIRECTORY, R_OK | X_OK)
154 != NGX_OK)
155 {
156 return NGX_ERROR;
157 }
158 }
159 if (brix_finalize_vo_rules(cf->log, &xcf->common.root, xcf->vo_rules)
160 != NGX_OK)
161 {
162 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
163 "brix: failed to finalize brix_require_vo rules for root \"%V\"",
164 &xcf->common.root);
165 return NGX_ERROR;
166 }
167
168 if (brix_finalize_authdb_rules(cf->log, &xcf->common.root, xcf->authdb_rules)
169 != NGX_OK)
170 {
171 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
172 "brix: failed to finalize brix_authdb rules for root \"%V\"",
173 &xcf->common.root);
174 return NGX_ERROR;
175 }
176
177 if (brix_finalize_group_rules(cf->log, &xcf->common.root,
178 xcf->group_rules) != NGX_OK)
179 {
180 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
181 "brix: failed to finalize brix_inherit_parent_group rules for root \"%V\"",
182 &xcf->common.root);
183 return NGX_ERROR;
184 }
185
186 return NGX_OK;
187}
