1/*
2 * lifecycle_internal.h — shared state for the impersonation nginx-lifecycle glue.
3 *
4 * The lifecycle glue is split across three translation units that all read the
5 * single process-global settings block:
6 * - lifecycle.c config-directive parsing + validation
7 * - lifecycle_broker.c master-side privileged-broker spawn + init_module
8 * - lifecycle_worker.c worker cap-drop + client connect + per-request begin/end
9 * This header names the settings struct and exports the one global they share so
10 * the split stays behaviour-identical to the pre-split single file.
11 */
