1/*
2 * auth.c — authentication driver.
3 *
4 * WHAT: After kXR_login returns a "&P=<proto>,..." security list, parse it, pick
5 * a protocol we have credentials for (preference: gsi > ztn > unix, honoring
6 * an explicit --auth override), and drive the kXR_auth/kXR_authmore loop via
7 * the selected sec module.
8 * WHY: Turns the server's auth demand into a completed session, protocol-agnostic.
9 * HOW: Each kXR_auth carries the module's credtype + payload; the server replies
10 * kXR_authmore (feed the body back to module.more) or kXR_ok (done). Rounds
11 * are capped to defend against a server that never finishes.
12 */

23static int
24send_auth(brix_conn *c, const char credtype[4], const uint8_t *payload,
25 uint32_t plen, uint16_t *sid, brix_status *st)
26{
27 ClientAuthRequest req;
28 brix_payload pl = { payload, plen };
29
30 memset(&req, 0, sizeof(req));
31 req.requestid = htons(kXR_auth);
32 {
33 xrdw_auth_req_t b;
34 memcpy(b.credtype, credtype, 4);
35 xrdw_auth_req_pack(&b, ((ClientRequestHdr *) &req)->body);
36 }
37 return brix_send(c, &req, &pl, sid, st);
38}

40/* Run one protocol's kXR_auth/authmore exchange to completion. 0 / -1. */
41static int
42run_module(brix_conn *c, const brix_sec_module *m, const char *parms,
43 brix_status *st)
44{
45 uint8_t *payload = NULL;
46 uint32_t plen = 0;
47 int round;
48
49 if (m->first(c, parms, &payload, &plen, st) != 0) {
50 return -1;
51 }
52
53 for (round = 0; round < XRDC_AUTH_MAX_ROUNDS; round++) {
54 uint16_t sid, status;
55 uint8_t *body = NULL;
56 uint32_t blen = 0;
57 int rc;
58 brix_resp_out out = { &status, &body, &blen };
59
60 rc = send_auth(c, m->credtype, payload, plen, &sid, st);
61 free(payload);
62 payload = NULL;
63 plen = 0;
64 if (rc != 0) {
65 return -1;
66 }
67 if (brix_recv(c, sid, &out, st) != 0) {
68 return -1; /* kXR_error → st already set */
69 }
70 if (status == kXR_ok) {
71 free(body);
72 if (m->cleanup != NULL) {
73 m->cleanup(c);
74 }
75 return 0;
76 }
77 if (status == kXR_authmore) {
78 if (m->more == NULL) {
79 free(body);
80 brix_status_set(st, XRDC_EAUTH, 0,
81 "%s: server asked for more rounds (unsupported)",
82 m->name);
83 return -1;
84 }
85 rc = m->more(c, body, blen, &payload, &plen, st);
86 free(body);
87 if (rc != 0) {
88 return -1;
89 }
90 if (payload == NULL) {
91 brix_status_set(st, XRDC_EAUTH, 0,
92 "%s: auth ended without server ok", m->name);
93 return -1;
94 }
95 continue;
96 }
97 free(body);
98 brix_status_set(st, XRDC_EAUTH, 0, "%s: unexpected auth status %u",
99 m->name, status);
100 return -1;
101 }
102
103 free(payload);
104 brix_status_set(st, XRDC_EAUTH, 0, "%s: too many auth rounds", m->name);
105 return -1;
106}

108/*
109 * build_module_list — fill mods[] with every sec module, in preference order.
110 *
111 * WHAT: Populate the caller's fixed 7-slot array with the sec modules in the
112 * fixed preference order gsi > ztn > krb5 > sss > unix > host > pwd and
113 * return how many slots were written.
114 * WHY: Both the driver (brix_authenticate) and the read-only explainer
115 * (brix_auth_explain) must walk the SAME modules in the SAME order for
116 * "why skipped" to match the real choice; sharing one builder keeps them
117 * from drifting apart.
118 * HOW: gsi is only listed when its module is linked in (brix_sec_gsi may be
119 * NULL); the rest are appended unconditionally — a NULL slot there is
120 * filtered later at try-time, never here.
121 */
122static int
123build_module_list(const brix_sec_module *mods[7])
124{
125 int nmods = 0;
126
127 if (brix_sec_gsi() != NULL) {
128 mods[nmods++] = brix_sec_gsi();
129 }
130 mods[nmods++] = brix_sec_token();
131 mods[nmods++] = brix_sec_krb5();
132 mods[nmods++] = brix_sec_sss();
133 mods[nmods++] = brix_sec_unix();
134 mods[nmods++] = brix_sec_host();
135 mods[nmods++] = brix_sec_pwd();
136 return nmods;
137}

139/*
140 * module_candidate — decide whether one sec module is usable right now.
141 *
142 * WHAT: Return 1 (and fill parms with the server's advertised parameters for
143 * this protocol) when the module m should be driven, 0 when it must be
144 * skipped.
145 * WHY: Factors the four skip conditions of the try-loop out of the driver so
146 * the driver stays within complexity budget; the decision itself is
147 * unchanged and load-bearing — each branch guards a distinct reason a
148 * credential must NOT be presented.
149 * HOW: Skip a NULL slot; skip anything the --auth override filtered out; skip
150 * a protocol the server did not advertise (this call also fills parms as
151 * a side effect, so it must run for the surviving module); skip a module
152 * that reports no local credentials. Checks run in this exact order to
153 * preserve short-circuit behavior and the parms side effect.
154 */
155static int
156module_candidate(brix_conn *c, const brix_sec_module *m, const brix_opts *o,
157 const char *seclist, char *parms, size_t parmslen)
158{
159 if (m == NULL) {
160 return 0;
161 }
162 if (o != NULL && o->auth_force != NULL
163 && strcmp(o->auth_force, m->name) != 0) {
164 return 0;
165 }
166 if (!brix_sec_proto_advertised(seclist, m->name, parms, parmslen)) {
167 return 0;
168 }
169 if (m->have_creds != NULL && !m->have_creds(c)) {
170 return 0;
171 }
172 return 1;
173}

175/*
176 * auth_warn_fallback — tell the user a credentialed protocol failed.
177 *
178 * WHAT: Emit a one-line stderr warning naming the protocol that failed and the
179 * reason captured in st, then let the caller fall back to the next module.
180 * WHY: A protocol we HAD credentials for failed. Without this, the driver
181 * silently falls back (classically gsi to unix) and the user sees only a
182 * confusing downstream authorization error from the server (e.g. EOS
183 * "unauthorized identity used") instead of the real cause (e.g. the GSI
184 * handshake the server rejected). stderr keeps data output (stdout) clean.
185 * HOW: Print m->name and st->msg, substituting "no detail" when st carries no
186 * message text.
187 */
188static void
189auth_warn_fallback(const brix_sec_module *m, const brix_status *st)
190{
191 fprintf(stderr,
192 "brix: warning: '%s' authentication failed (%s); "
193 "falling back to the next offered protocol\n",
194 m->name, st->msg[0] ? st->msg : "no detail");
195}

197int
198brix_authenticate(brix_conn *c, const char *seclist, const brix_opts *o,
199 brix_status *st)
200{
201 const brix_sec_module *mods[7];
202 int nmods;
203 int i;
204 int tried = 0;
205
206 /* force_anon: complete the login but present NO credential. Used by the
207 * remote-doctor auth-suite to test the SERVER's own enforcement (does it
208 * reject unauthenticated operations?) independent of the client's creds. */
209 if (o != NULL && o->force_anon) {
210 return 0;
211 }
212
213 nmods = build_module_list(mods);
214
215 for (i = 0; i < nmods; i++) {
216 const brix_sec_module *m = mods[i];
217 char parms[256];
218
219 if (!module_candidate(c, m, o, seclist, parms, sizeof(parms))) {
220 continue;
221 }
222
223 tried = 1;
224 if (run_module(c, m, parms, st) == 0) {
225 c->diag.chosen_auth = m->name; /* §15 explain */
226 return 0;
227 }
228 auth_warn_fallback(m, st);
229 }
230
231 if (!tried) {
232 brix_status_set(st, XRDC_EAUTH, 0,
233 "no usable auth protocol for server list \"%s\"%s",
234 seclist,
235 (o != NULL && o->auth_force) ? " (with --auth filter)" : "");
236 }
237 return -1;
238}

240void
241brix_auth_explain(brix_conn *c, const brix_opts *o, FILE *out)
242{
243 const brix_sec_module *mods[7];
244 int nmods, i, picked = 0;
245 char parms[256];
246
247 nmods = build_module_list(mods);
248
249 fprintf(out, " sec list: %s\n",
250 c->sec_list[0] ? c->sec_list : "(none — anonymous)");
251 fprintf(out, " authenticated with: %s\n",
252 c->diag.chosen_auth ? c->diag.chosen_auth : "(none — anonymous login)");
253
254 /* Re-derive the would-be choice read-only, so "why skipped" matches the real
255 * driver's preference order (gsi > ztn > unix). */
256 for (i = 0; i < nmods; i++) {
257 const brix_sec_module *m = mods[i];
258 const char *why;
259 if (m == NULL) {
260 continue;
261 }
262 if (o != NULL && o->auth_force != NULL
263 && strcmp(o->auth_force, m->name) != 0) {
264 why = "skipped (--auth filter)";
265 } else if (!brix_sec_proto_advertised(c->sec_list, m->name, parms, sizeof(parms))) {
266 why = "not offered by server";
267 } else if (m->have_creds != NULL && !m->have_creds(c)) {
268 why = "offered, but no local credentials";
269 } else if (!picked) {
270 why = "offered + creds present → preferred";
271 picked = 1;
272 } else {
273 why = "offered + creds present (lower preference)";
274 }
275 fprintf(out, " %-6s %s\n", m->name, why);
276 }
