1/*
2 * sts.h — AWS STS (Security Token Service) client for backend credential
3 * EXCHANGE (phase-70 §5.5).
4 *
5 * WHAT: issue an AssumeRole call (falling back to GetSessionToken when no role
6 * ARN is configured) against a backend STS endpoint and return the
7 * resulting short-lived (AccessKeyId, SecretAccessKey, SessionToken).
8 * WHY: an S3 SigV4 secret is never transmitted, so the inbound access-key id
9 * alone cannot be forwarded to the origin. Instead the node holds a
10 * long-lived backend S3 *service* credential and exchanges it — scoped to
11 * the inbound principal via RoleSessionName — for temporary credentials
12 * the origin trusts. Pure passthrough is impossible by design.
13 * HOW: build the AssumeRole GET query, SigV4-sign it with the node's service
14 * credentials over the "sts" service, POST/GET via libcurl, and parse the
15 * XML (libxml2) or JSON response into pool-copied out params. The service
16 * secret and the returned secret/session token are never logged.
17 */

30 * role_arn — role to AssumeRole into; empty selects GetSessionToken

44/*
45 * brix_s3_sts_out_t — the (ak, sk, session) output triple of an STS exchange,
46 * bundled so callers hand one struct pointer instead of three loose out-params.
47 *
48 * WHAT: three borrowed ngx_str_t slots the exchange fills with the temporary
49 * credentials — AccessKeyId, SecretAccessKey and (optional) SessionToken.
50 * WHY: keeps brix_s3_sts_assume() within the ≤5-argument budget and gives the
51 * three logically-coupled outputs a single named home; the slots are the
52 * caller's own (borrowed, not owned by the STS layer).
53 * HOW: the caller declares three ngx_str_t, points the fields at them, and on
54 * NGX_OK reads back pool-allocated, NUL-terminated copies (session may be
55 * empty when the response omits a SessionToken).
56 */
57typedef struct {

59 ngx_str_t *sk;
60 ngx_str_t *session;
61} brix_s3_sts_out_t;

63/*
64 * Exchange the node's service credential for temporary credentials scoped to
65 * the inbound identity. `id->subject`/`id->dn` map to RoleSessionName so the
66 * origin audit log attributes actions to the real principal.
67 *
68 * On NGX_OK, out->ak, out->sk and out->session are set to pool-allocated,
69 * NUL-terminated copies of the returned credentials (session may be empty if

73 */
