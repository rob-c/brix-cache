8/* paged I/O with per-page CRC32c (kXR_pgread / kXR_pgwrite, M6) *
9 * Both replies use kXR_status (4007) framing, NOT the kXR_ok path, so they are
10 * read here rather than via brix_recv. One status frame is:
11 * ServerResponseHdr{status=kXR_status, dlen=24}
12 * ServerResponseBody_Status{crc32c[4], streamID[2], requestid[1], resptype[1],
13 * reserved[4], dlen[4]} (16 bytes)
14 * trailing offset[8] (pgRead/pgWrite body)
15 * The header crc32c covers the 20 bytes streamID..offset; bdy.dlen is the size of
16 * the page-data that follows the 24-byte body (0 for pgwrite). Page units are
17 * [crc32c_be 4][data ≤4096], aligned to the FILE offset (short first page).
18 */

21/* Read and validate one kXR_status frame's 24-byte body. On success sets
22 * *resptype (0=Final,1=Partial), *pgdlen (page-data bytes that follow), and
23 * *foff (the frame's file offset). kXR_error is surfaced via st. 0 / -1. */
24int
25read_status_frame(brix_conn *c, uint16_t want_sid, uint8_t *resptype,
26 uint32_t *pgdlen, int64_t *foff, brix_status *st)
27{
28 uint8_t hdr[XRD_RESPONSE_HDR_LEN];
29 uint8_t sb[XRDC_PG_STATUSBODY];
30 uint16_t sid, stat;
31 uint32_t dlen, want_crc, got_crc;
32 uint64_t off_be;
33
34 if (brix_read_full(&c->io, hdr, sizeof(hdr), st) != 0) {
35 return -1;
36 }
37 xrd_resp_hdr_unpack(hdr, &sid, &stat, &dlen); /* unaligned-safe */
38
39 if (stat == kXR_error) {
40 uint8_t *eb = NULL;
41 int errnum = 0;
42 if (dlen > 0 && dlen <= XRDC_DLEN_MAX) {
43 eb = (uint8_t *) malloc(dlen);
44 if (eb != NULL && brix_read_full(&c->io, eb, dlen, st) == 0) {
45 const char *emsg = "";
46 size_t emlen = 0;
47 /* bounded msg slice — eb is not NUL-terminated. */
48 xrd_error_body_decode(eb, dlen, &errnum, &emsg, &emlen);
49 brix_status_set(st, errnum, 0, "%.*s (%s)", (int) emlen,
50 emsg ? emsg : "", brix_kxr_name(errnum));
51 }
52 }
53 free(eb);
54 if (st->kxr == 0) {
55 brix_status_set(st, XRDC_EPROTO, 0, "pg op error (status %u)", stat);
56 }
57 return -1;
58 }
59 if (stat != kXR_status) {
60 brix_status_set(st, XRDC_EPROTO, 0,
61 "expected kXR_status, got %u", stat);
62 return -1;
63 }
64 if (want_sid != 0xffff && sid != want_sid) {
65 brix_status_set(st, XRDC_EPROTO, 0,
66 "stream id mismatch (got %u, want %u)", sid, want_sid);
67 return -1;
68 }
69 if (dlen != XRDC_PG_STATUSBODY) {
70 brix_status_set(st, XRDC_EPROTO, 0,
71 "bad kXR_status body length %u (want %d)",
72 dlen, XRDC_PG_STATUSBODY);
73 return -1;
74 }
75 if (brix_read_full(&c->io, sb, sizeof(sb), st) != 0) {
76 return -1;
77 }
78
79 /* crc32c covers sb[4..24): streamID(2) requestid(1) resptype(1) reserved(4)
80 * dlen(4) offset(8) = 20 bytes. */
81 want_crc = xrd_get_u32_be(sb); /* unaligned-safe */
82 got_crc = brix_crc32c_value(sb + 4, XRDC_PG_STATUSBODY - 4);
83 if (want_crc != got_crc) {
84 brix_status_set(st, XRDC_EINTEGRITY, 0,
85 "kXR_status header CRC mismatch (got %08x want %08x)",
86 got_crc, want_crc);
87 return -1;
88 }
89
90 *resptype = sb[7];
91 *pgdlen = xrd_get_u32_be(sb + 12); /* unaligned-safe */
92 memcpy(&off_be, sb + 16, 8);
93 *foff = (int64_t) be64toh(off_be);
94 return 0;
95}

98/* Decode one frame's page-data buffer (units of [crc 4][data], file-offset
99 * aligned at file_off) into dst, verifying each page's CRC32c. Returns decoded
100 * data bytes, or -1 (st set) on a CRC mismatch or malformed framing. */
101ssize_t
102decode_pages(const uint8_t *pg, uint32_t pglen, int64_t file_off,
103 uint8_t *dst, size_t dstcap, brix_status *st)
104{
105 int64_t bad = file_off;
106 ssize_t n = xrdp_pg_decode(pg, (size_t) pglen, file_off, dst, dstcap, &bad);
107
108 if (n == -1) {
109 brix_status_set(st, XRDC_EINTEGRITY, 0,
110 "pgread CRC mismatch at offset %lld", (long long) bad);
111 return -1;
112 }
113 if (n < 0) {
114 brix_status_set(st, XRDC_EPROTO, 0, "pgread malformed page framing");
115 return -1;
116 }
117 return n;
118}

121ssize_t
122brix_file_pgread(brix_conn *c, brix_file *f, int64_t offset, void *buf,
123 size_t len, brix_status *st)
124{
125 ClientPgReadRequest req;
126 uint16_t sid;
127 size_t total = 0;
128
129 memset(&req, 0, sizeof(req));
130 req.requestid = htons(kXR_pgread);
131 {
132 xrdw_pgread_req_t b = { .offset = offset, .rlen = (int32_t) len };
133 memcpy(b.fhandle, f->fhandle, XRD_FHANDLE_LEN);
134 xrdw_pgread_req_pack(&b, ((ClientRequestHdr *) &req)->body);
135 }
136 /* dlen (offset 20) is set to 0 by brix_send (no args payload). */
137
138 if (brix_send(c, &req, NULL, &sid, st) != 0) {
139 return -1;
140 }
141
142 /* Accumulate kXR_status frames until resptype=Final (the module sends one
143 * Final frame per request; Partial is handled defensively). */
144 for (;;) {
145 uint8_t resptype = 0;
146 uint32_t pgdlen = 0;
147 int64_t foff = 0;
148 uint8_t *pg;
149 ssize_t decoded;
150
151 if (read_status_frame(c, sid, &resptype, &pgdlen, &foff, st) != 0) {
152 return -1;
153 }
154 if (pgdlen == 0) {
155 if (resptype == kXR_FinalResult) {
156 break;
157 }
158 continue;
159 }
160 pg = (uint8_t *) malloc(pgdlen);
161 if (pg == NULL) {
162 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (%u)", pgdlen);
163 return -1;
164 }
165 if (brix_read_full(&c->io, pg, pgdlen, st) != 0) {
166 free(pg);
167 return -1;
168 }
169 decoded = decode_pages(pg, pgdlen, foff,
170 (uint8_t *) buf + total, len - total, st);
171 free(pg);
172 if (decoded < 0) {
173 return -1;
174 }
175 total += (size_t) decoded;
176 if (resptype == kXR_FinalResult) {
177 break;
178 }
179 }
180 return (ssize_t) total;
181}

186/* Resend one page (kXR_pgRetry) from the original buffer and report the server's
187 * verdict. pgoff is the page's file offset; the page data is sliced out of buf
188 * at (pgoff - base). Returns 0 = corrected, 1 = still bad, -1 = error (st set). */
189int
190pgwrite_retry_one(brix_conn *c, brix_file *f, const uint8_t *buf, int64_t base,
191 size_t len, int64_t pgoff, brix_status *st)
192{
193 ClientPgWriteRequest req;
194 uint16_t sid;
195 uint8_t rp[kXR_pgPageSZ + 4];
196 size_t doff = (size_t) (pgoff - base);
197 size_t to_boundary = (size_t) kXR_pgPageSZ
198 - (size_t) (pgoff & (int64_t) (kXR_pgPageSZ - 1));
199 size_t remaining, page_len, rplen;
200 uint8_t resptype = 0;
201 uint32_t pgdlen = 0;
202 int64_t foff = 0;
203
204 if (pgoff < base || doff >= len) {
205 brix_status_set(st, XRDC_EPROTO, 0,
206 "pgwrite CSE offset %lld out of range", (long long) pgoff);
207 return -1;
208 }
209 remaining = len - doff;
210 page_len = remaining < to_boundary ? remaining : to_boundary;
211 rplen = xrdp_pg_encode(buf + doff, page_len, pgoff, rp);
212
213 memset(&req, 0, sizeof(req));
214 req.requestid = htons(kXR_pgwrite);
215 {
216 xrdw_pgwrite_req_t b = { .offset = pgoff, .pathid = 0,
217 .reqflags = kXR_pgRetry };
218 memcpy(b.fhandle, f->fhandle, XRD_FHANDLE_LEN);
219 xrdw_pgwrite_req_pack(&b, ((ClientRequestHdr *) &req)->body);
220 }
221
222 {
223 brix_payload pl = { rp, (uint32_t) rplen };
224 if (brix_send(c, &req, &pl, &sid, st) != 0) {
225 return -1;
226 }
227 }
228 if (read_status_frame(c, sid, &resptype, &pgdlen, &foff, st) != 0) {
229 return -1;
230 }
231 if (pgdlen != 0) {
232 /* Still bad — drain and discard the CSE trailer, report not-yet-clean. */
233 uint8_t *cse = (uint8_t *) malloc(pgdlen);
234 if (cse != NULL) {
235 (void) brix_read_full(&c->io, cse, pgdlen, st);
236 free(cse);
237 }
238 return 1;
239 }
240 return 0;
241}

244/* ---- Drive one corrupt page back to a clean server verdict ----
245 *
246 * WHAT: Resend the page at file offset pgoff (sliced from buf at pgoff-base)
247 * via kXR_pgRetry up to XRDC_PGW_MAX_RETRY times. Returns 0 once the server
248 * accepts the page, or -1 (st set) if it stays corrupt past the retry budget
249 * or a wire error occurs.
250 *
251 * WHY: The pgwrite CSE handler must apply an identical bounded-retry policy to
252 * every failed page. Isolating that per-page loop keeps the CSE trailer walk
253 * flat and keeps the retry/verdict bookkeeping in one testable place.
254 *
255 * HOW:
256 * 1. Loop up to XRDC_PGW_MAX_RETRY: resend the page via pgwrite_retry_one;
257 * a verdict of 0 (corrected) or -1 (error) breaks out immediately, 1
258 * (still bad) retries.
259 * 2. verdict > 0 after the budget is exhausted -> set XRDC_EINTEGRITY, -1.
260 * 3. verdict < 0 -> propagate the already-set error, -1.
261 * 4. verdict 0 -> page corrected, 0.
262 */
263static int
264pgwrite_correct_one_page(brix_conn *c, brix_file *f, const uint8_t *buf,
265 int64_t base, size_t len, int64_t pgoff,
266 brix_status *st)
267{
268 int attempt, verdict = 1;
269
270 for (attempt = 0; attempt < XRDC_PGW_MAX_RETRY; attempt++) {
271 verdict = pgwrite_retry_one(c, f, buf, base, len, pgoff, st);
272 if (verdict <= 0) {
273 break; /* 0 = corrected, -1 = error */
274 }
275 }
276 if (verdict > 0) {
277 brix_status_set(st, XRDC_EINTEGRITY, 0,
278 "pgwrite page %lld uncorrectable after %d retries",
279 (long long) pgoff, XRDC_PGW_MAX_RETRY);
280 return -1;
281 }
282 if (verdict < 0) {
283 return -1;
284 }
285 return 0;
286}

289/* ---- Read and act on a pgwrite CSE (checksum-error) trailer ----
290 *
291 * WHAT: Read the pgdlen-byte CSE trailer the server sent after a pgwrite,
292 * validate its shape, and resend every listed bad page via
293 * pgwrite_correct_one_page. Returns 0 when all pages end clean, -1 (st set)
294 * on OOM, a malformed trailer, or an uncorrectable/errored page. Does NOT own
295 * the caller's request payload.
296 *
297 * WHY: The server writes the data but reports pages whose CRC32c failed on the
298 * wire (our encode is always correct); the client must retransmit those pages.
299 * Confining the trailer read and bad-page walk here keeps brix_file_pgwrite a
300 * flat send/read orchestrator.
301 *
302 * HOW:
303 * 1. Allocate and read the trailer; OOM or a short read -> -1 (st set).
304 * 2. Trailer layout is cseCRC(4) dlFirst(2) dlLast(2) then int64 bof[n];
305 * reject any length < 8 or not 8-aligned in the tail as malformed.
306 * 3. For each of the n bad-page offsets, decode the big-endian int64 and
307 * drive it clean via pgwrite_correct_one_page, stopping on the first
308 * failure.
309 * 4. Free the trailer and return the accumulated verdict.
310 */
311static int
312pgwrite_handle_cse(brix_conn *c, brix_file *f, const uint8_t *buf,
313 int64_t base, size_t len, uint32_t pgdlen, brix_status *st)
314{
315 uint8_t *cse = (uint8_t *) malloc(pgdlen);
316 size_t nbad, i;
317 int rc_ret = 0;
318
319 if (cse == NULL || brix_read_full(&c->io, cse, pgdlen, st) != 0) {
320 if (cse == NULL) {
321 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (%u)", pgdlen);
322 }
323 free(cse);
324 return -1;
325 }
326
327 /* Trailer: cseCRC(4) dlFirst(2) dlLast(2) then int64 bof[n]. */
328 if (pgdlen < 8 || ((pgdlen - 8) % 8) != 0) {
329 free(cse);
330 brix_status_set(st, XRDC_EPROTO, 0,
331 "malformed pgwrite CSE trailer (%u bytes)", pgdlen);
332 return -1;
333 }
334 nbad = (size_t) (pgdlen - 8) / 8;
335
336 for (i = 0; i < nbad && rc_ret == 0; i++) {
337 uint64_t bo_be;
338 int64_t bo;
339
340 memcpy(&bo_be, cse + 8 + i * 8, 8);
341 bo = (int64_t) be64toh(bo_be);
342 rc_ret = pgwrite_correct_one_page(c, f, buf, base, len, bo, st);
343 }
344
345 free(cse);
346 return rc_ret;
347}

350int
351brix_file_pgwrite(brix_conn *c, brix_file *f, int64_t offset, const void *buf,
352 size_t len, brix_status *st)
353{
354 ClientPgWriteRequest req;
355 uint16_t sid;
356 uint8_t *payload = NULL;
357 size_t cap, plen = 0;
358 uint8_t resptype = 0;
359 uint32_t pgdlen = 0;
360 int64_t foff = 0;
361
362 /* Worst case: one 4-byte CRC per page plus a short first/last page → at most
363 * (len/pagesz + 2) checksums. Build the [crc][data] payload via the shared
364 * page-mode encoder (libxrdproto) so client-encode == server-decode. */
365 cap = len + ((len / kXR_pgPageSZ) + 2) * 4;
366 payload = (uint8_t *) malloc(cap);
367 if (payload == NULL) {
368 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (%zu)", cap);
369 return -1;
370 }
371 plen = xrdp_pg_encode((const uint8_t *) buf, len, offset, payload);
372
373 memset(&req, 0, sizeof(req));
374 req.requestid = htons(kXR_pgwrite);
375 {
376 xrdw_pgwrite_req_t b = { .offset = offset, .pathid = 0, .reqflags = 0 };
377 memcpy(b.fhandle, f->fhandle, XRD_FHANDLE_LEN);
378 xrdw_pgwrite_req_pack(&b, ((ClientRequestHdr *) &req)->body);
379 }
380 /* dlen (offset 20) is set to plen by brix_send. */
381
382 {
383 brix_payload pl = { payload, (uint32_t) plen };
384 if (brix_send(c, &req, &pl, &sid, st) != 0) {
385 free(payload);
386 return -1;
387 }
388 }
389 if (read_status_frame(c, sid, &resptype, &pgdlen, &foff, st) != 0) {
390 free(payload); /* kXR_ChkSumErr (CRC rejected) surfaces here */
391 return -1;
392 }
393 if (pgdlen != 0) {
394 /* CSE: the server wrote the data but reported pages that failed CRC32c
395 * (typically wire corruption — our encode is always correct). Read the
396 * retransmit list and resend each page with kXR_pgRetry until the server
397 * accepts it or we exhaust the bounded attempts. */
398 int rc_ret = pgwrite_handle_cse(c, f, (const uint8_t *) buf, offset,
399 len, pgdlen, st);
400 free(payload);
401 return rc_ret;
402 }
403 free(payload);
404 return 0;
405}
