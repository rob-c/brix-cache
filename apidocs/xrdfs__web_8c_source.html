8/* http(s)/WebDAV endpoints — read-only namespace ops (ls, stat). */
9/* WHAT: When the endpoint is an http/https/dav/davs URL, xrdfs speaks */
10/* WebDAV PROPFIND for metadata (the webfile.c transport, shared */
11/* with the FUSE driver) instead of the binary root:// protocol — */
12/* so `xrdfs https://host/path ls` works like the official client.*/
13/* WHY: A WebDAV endpoint has no root:// session; only the read-only */
14/* metadata ops map cleanly. Mutating/file ops report clearly that */
15/* they need a root:// endpoint. */

20web_build_path(const char *base, const char *cwd, const char *arg,

22{

24 build_path(cwd, (arg != NULL && arg[0] != '\0') ? arg : ".", rel, sizeof(rel));

26 snprintf(out, outsz, "%s", (base != NULL && base[0] != '\0') ? base : "/");
27 } else {
28 snprintf(out, outsz, "%s%s", base, rel);

33/* WebDAV equivalent of ls_print_dir() — same output format, PROPFIND source. */
34int
35web_ls_print_dir(const web_ctx *w, const char *path, int want_long,
36 int recursive, int human, brix_status *st)
37{
38 brix_dirent *ents = NULL;
39 size_t n = 0, k, plen;
40 char prefix[XRDC_PATH_MAX + 2];
41 const char *sep;
42
43 if (brix_web_readdir(w->u, path, w->bearer, w->verify, w->ca_dir,
44 &ents, &n, st) != 0) {
45 return -1;
46 }
47 plen = strlen(path);
48 sep = (plen > 0 && path[plen - 1] == '/') ? "" : "/";
49 snprintf(prefix, sizeof(prefix), "%s%s", path, sep);
50
51 for (k = 0; k < n; k++) {
52 if (want_long && ents[k].have_stat) {
53 int f = ents[k].st.flags;
54 char td = (f & kXR_isDir) ? 'd' : '-';
55 char r = (f & kXR_readable) ? 'r' : '-';
56 char wc = (f & kXR_writable) ? 'w' : '-';
57 char szs[32];
58 fmt_size(ents[k].st.size, szs, sizeof(szs), human);
59 printf("%c%c%c %12s %s%s\n", td, r, wc, szs, prefix, ents[k].name);
60 } else {
61 printf("%s%s\n", prefix, ents[k].name);
62 }
63 }
64 if (recursive) {
65 for (k = 0; k < n; k++) {
66 char full[XRDC_PATH_MAX];
67 if (is_dot(ents[k].name)
68 || !(ents[k].have_stat && (ents[k].st.flags & kXR_isDir))) {
69 continue;
70 }
71 if (join_path(path, ents[k].name, full, sizeof(full)) != 0) {
72 continue;
73 }
74 printf("\n%s:\n", full);
75 if (web_ls_print_dir(w, full, want_long, 1, human, st) != 0) {
76 free(ents);
77 return -1;
78 }
79 }
80 }
81 free(ents);
82 return 0;
83}

86int
87web_ls(const web_ctx *w, const char *cwd, int argc, char **argv)
88{
89 brix_status st;
90 char path[XRDC_PATH_MAX];
91 const char *arg = NULL;
92 int want_long = 0, recursive = 0, human = 0, i;
93
94 for (i = 1; i < argc; i++) {
95 if (strcmp(argv[i], "-l") == 0) { want_long = 1; }
96 else if (strcmp(argv[i], "-R") == 0) { recursive = 1; }
97 else if (strcmp(argv[i], "-h") == 0) { human = 1; }
98 else if (strcmp(argv[i], "-lR") == 0 || strcmp(argv[i], "-Rl") == 0) {
99 want_long = 1; recursive = 1;
100 } else { arg = argv[i]; }
101 }
102 web_build_path(w->base, cwd, arg, path, sizeof(path));
103 brix_status_clear(&st);
104 if (web_ls_print_dir(w, path, want_long, recursive, human, &st) != 0) {
105 fprintf(stderr, "xrdfs: ls %s: %s\n", path, st.msg);
106 xrdfs_web_hints(&st, 0, w); /* WS-7 doctor referral */
107 return brix_shellcode(&st);
108 }
109 return 0;
110}

133/* Dispatch one command against a WebDAV endpoint. Read-only metadata ops only. */
134int
135web_dispatch(const web_ctx *w, int argc, char **argv)
136{
137 const char *cmd = argv[0];
138 if (strcmp(cmd, "ls") == 0) { return web_ls(w, "/", argc, argv); }
139 if (strcmp(cmd, "stat") == 0) { return web_stat(w, "/", argc, argv); }
140 fprintf(stderr, "xrdfs: '%s' is not supported over an http(s)/WebDAV endpoint "
141 "(read-only metadata only: ls, stat). Use a root:// endpoint for "
142 "the full command set.\n", cmd);
143 return 50;
144}
