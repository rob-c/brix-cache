8/* http(s)/WebDAV endpoints — read-only namespace ops (ls, stat). */
9/* WHAT: When the endpoint is an http/https/dav/davs URL, xrdfs speaks */
10/* WebDAV PROPFIND for metadata (the webfile.c transport, shared */
11/* with the FUSE driver) instead of the binary root:// protocol — */
12/* so `xrdfs https://host/path ls` works like the official client.*/
13/* WHY: A WebDAV endpoint has no root:// session; only the read-only */
14/* metadata ops map cleanly. Mutating/file ops report clearly that */
15/* they need a root:// endpoint. */

33/* ---- Print one directory entry in xrdfs ls format ----
34 *
35 * WHAT: Emits a single line for one PROPFIND-derived dirent to stdout. In long
36 * mode (want_long) with stat data present, prints the drwx-style flag
37 * triplet, a right-justified size (human-readable when human is set), and
38 * the full "prefix + name" path; otherwise prints just "prefix + name".
39 * No return value; output-only.
40 *
41 * WHY: Factored out of web_ls_print_dir so the per-entry formatting (with its
42 * flag/size ternaries) is a single-purpose helper, keeping the directory
43 * walker's cyclomatic complexity within the project cap and matching the
44 * exact one-line-per-entry format of the root:// ls_print_dir().
45 *
46 * HOW: 1. If long output is requested and the entry carries stat data, derive
47 * the dir/readable/writable flag characters and format the size.
48 * 2. Print the long-form line with flags, size, prefix, and name.
49 * 3. Otherwise print the short form: prefix followed by name.
50 */
51static void
52web_print_one_entry(const brix_dirent *ent, const char *prefix,
53 int want_long, int human)
54{
55 if (want_long && ent->have_stat) {
56 int f = ent->st.flags;
57 char td = (f & kXR_isDir) ? 'd' : '-';
58 char r = (f & kXR_readable) ? 'r' : '-';
59 char wc = (f & kXR_writable) ? 'w' : '-';
60 char szs[32];
61 fmt_size(ent->st.size, szs, sizeof(szs), human);
62 printf("%c%c%c %12s %s%s\n", td, r, wc, szs, prefix, ent->name);
63 } else {
64 printf("%s%s\n", prefix, ent->name);
65 }
66}

69/* ---- Recurse into subdirectories for xrdfs ls -R over WebDAV ----
70 *
71 * WHAT: For each subdirectory in a directory listing, prints the "full:" header
72 * and recursively lists its contents via web_ls_print_dir(). Returns 0 on
73 * success, -1 if any recursive listing fails (with st describing the
74 * failure). Skips "." / ".." entries, non-directories, and any child whose
75 * joined path would overflow.
76 *
77 * WHY: Separated from web_ls_print_dir so the recursion loop is an independent,
78 * single-purpose step. This keeps the walker's complexity within the cap
79 * while preserving the exact skip rules and recursive descent order.
80 *
81 * HOW: 1. Iterate the entries; skip dot entries and anything that is not a
82 * stat-bearing directory.
83 * 2. Build the child's full path; skip entries that do not fit.
84 * 3. Print the "\nfull:\n" section header, then recurse; on the first
85 * recursion failure return -1, otherwise return 0.
86 */
87static int
88web_ls_recurse_children(const web_ctx *w, const char *path,
89 const brix_dirent *ents, size_t n,
90 int want_long, int human, brix_status *st)
91{
92 size_t k;
93
94 for (k = 0; k < n; k++) {
95 char full[XRDC_PATH_MAX];
96 if (is_dot(ents[k].name)
97 || !(ents[k].have_stat && (ents[k].st.flags & kXR_isDir))) {
98 continue;
99 }
100 if (join_path(path, ents[k].name, full, sizeof(full)) != 0) {
101 continue;
102 }
103 printf("\n%s:\n", full);
104 if (web_ls_print_dir(w, full, want_long, 1, human, st) != 0) {
105 return -1;
106 }
107 }
108 return 0;
109}

112/* WebDAV equivalent of ls_print_dir() — same output format, PROPFIND source. */
113int
114web_ls_print_dir(const web_ctx *w, const char *path, int want_long,
115 int recursive, int human, brix_status *st)
116{
117 brix_dirent *ents = NULL;
118 size_t n = 0, k, plen;
119 char prefix[XRDC_PATH_MAX + 2];
120 const char *sep;
121
122 if (brix_web_readdir(w->u, path, w->bearer, w->verify, w->ca_dir,
123 &ents, &n, st) != 0) {
124 return -1;
125 }
126 plen = strlen(path);
127 sep = (plen > 0 && path[plen - 1] == '/') ? "" : "/";
128 snprintf(prefix, sizeof(prefix), "%s%s", path, sep);
129
130 for (k = 0; k < n; k++) {
131 web_print_one_entry(&ents[k], prefix, want_long, human);
132 }
133 if (recursive
134 && web_ls_recurse_children(w, path, ents, n, want_long, human, st) != 0) {
135 free(ents);
136 return -1;
137 }
138 free(ents);
139 return 0;
140}

143int
144web_ls(const web_ctx *w, const char *cwd, int argc, char **argv)
145{
146 brix_status st;
147 char path[XRDC_PATH_MAX];
148 const char *arg = NULL;
149 int want_long = 0, recursive = 0, human = 0, i;
150
151 for (i = 1; i < argc; i++) {
152 if (strcmp(argv[i], "-l") == 0) { want_long = 1; }
153 else if (strcmp(argv[i], "-R") == 0) { recursive = 1; }
154 else if (strcmp(argv[i], "-h") == 0) { human = 1; }
155 else if (strcmp(argv[i], "-lR") == 0 || strcmp(argv[i], "-Rl") == 0) {
156 want_long = 1; recursive = 1;
157 } else { arg = argv[i]; }
158 }
159 web_build_path(w->base, cwd, arg, path, sizeof(path));
160 brix_status_clear(&st);
161 if (web_ls_print_dir(w, path, want_long, recursive, human, &st) != 0) {
162 fprintf(stderr, "xrdfs: ls %s: %s\n", path, st.msg);
163 xrdfs_web_hints(&st, 0, w); /* WS-7 doctor referral */
164 return brix_shellcode(&st);
165 }
166 return 0;
167}

191int
192web_dispatch(const web_ctx *w, int argc, char **argv)
193{
194 const char *cmd = argv[0];
195 if (strcmp(cmd, "ls") == 0) { return web_ls(w, "/", argc, argv); }
196 if (strcmp(cmd, "stat") == 0) { return web_stat(w, "/", argc, argv); }
197 fprintf(stderr, "xrdfs: '%s' is not supported over an http(s)/WebDAV endpoint "
198 "(read-only metadata only: ls, stat). Use a root:// endpoint for "
199 "the full command set.\n", cmd);
200 return 50;
201}
