1/*
2 * s3.c — AWS Signature Version 4 (path-style) for the native client.
3 *
4 * WHAT: brix_s3_sign_v4() builds the x-amz-date / x-amz-content-sha256 /
5 * Authorization header block for any method+path+body-hash; the sha256
6 * helpers compute the payload hashes those headers carry.
7 * WHY: Both xrdcp's s3:// transfers and xrddiag's s3 probe need the identical
8 * signer — keeping one copy in the lib (rather than a private copy in each
9 * app) is the clean-room equivalent of XrdClS3's shared signing code.
10 * HOW: Canonical-request → string-to-sign → HMAC-SHA256 signing-key chain, using
11 * libxrdproto's SHA-256/HMAC kernels (compat/crypto.h). The CanonicalURI is
12 * percent-encoded (RFC-3986 unreserved + '/') to match the server's own
13 * canonicalization. Signed headers are host;x-amz-content-sha256;x-amz-date.
14 *
15 * Clean-room: implements the published SigV4 algorithm (AWS docs), not any SDK.
16 */

35/* ---- Build the SigV4 canonical request and return its SHA-256 hex ----
36 *
37 * WHAT: Percent-encodes the URI, assembles the canonical request
38 * (method, URI, canonical query, the three signed headers, the trailing
39 * SignedHeaders list, and the payload hash), and writes its lowercase
40 * SHA-256 hex (65 bytes incl. NUL) into canon_hex. Returns 0 on success,
41 * -1 if the URI cannot be encoded or the canonical request would overflow.
42 *
43 * WHY: The canonical-request hash is the first HMAC input; if this string
44 * differs by a single byte from what the server re-derives, the signature
45 * mismatches and the request 403s. Isolating its construction keeps the
46 * signer's stage boundaries explicit and each stage independently testable.
47 *
48 * HOW: 1. URL-encode uri with the shared RFC-3986 encoder (unreserved + '/'),
49 * byte-identical to the server's verify path.
50 * 2. snprintf the canonical request; treat truncation as a hard failure so
51 * a short buffer never silently signs a different string.
52 * 3. Hash the canonical request into canon_hex.
53 */
54static int
55brix_s3_sigv4_build_canon_hash(const char *method, const char *host,
56 const char *uri, const char *canon_qs,
57 const char *payload_hex, const char *amzdate,
58 char *canon_hex)
59{
60 char enc_uri[2048];
61 char canon[8192];
62 int cn;
63
64 /* Shared SigV4 CanonicalURI encoder (RFC-3986 unreserved + '/'), byte-identical
65 * to the server's verify path so client-signs == server-verifies. */
66 if (brix_http_urlencode((const unsigned char *) uri, strlen(uri),
67 enc_uri, sizeof(enc_uri), "/") < 0) {
68 return -1;
69 }
70 /* canonical request: method\nURI\nCANONICAL_QUERY\nheaders\nsignedHeaders\nhash.
71 * canon_qs must already be sorted + RFC-3986 encoded (the server re-derives the
72 * same string from the request query). */
73 cn = snprintf(canon, sizeof(canon),
74 "%s\n%s\n%s\nhost:%s\nx-amz-content-sha256:%s\nx-amz-date:%s\n\n"
75 "host;x-amz-content-sha256;x-amz-date\n%s",
76 method, enc_uri, canon_qs, host, payload_hex, amzdate, payload_hex);
77 /* A truncated canonical request would sign a different string than the server
78 * verifies → a confusing 403. Fail cleanly so the caller reports "can't sign". */
79 if (cn < 0 || (size_t) cn >= sizeof(canon)) {
80 return -1;
81 }
82 brix_s3_sha256_hex(canon, strlen(canon), canon_hex);
83 return 0;
84}

86/* ---- Build the credential scope and the string-to-sign ----
87 *
88 * WHAT: Formats the credential scope (datestamp/region/s3/aws4_request) into
89 * scope and the AWS4-HMAC-SHA256 string-to-sign (algorithm, amzdate, scope,
90 * canonical-request hash) into sts. Returns 0 on success, -1 if either
91 * snprintf would overflow its caller-supplied buffer.
92 *
93 * WHY: region is user/env-controlled (--s3-region / $AWS_DEFAULT_REGION); a
94 * truncated scope or string-to-sign would sign a different string than the
95 * Authorization header advertises → a confusing 403. Guarding both here
96 * keeps that contract in one place, matching the canonical-request guard.
97 *
98 * HOW: 1. snprintf the credential scope; fail on truncation.
99 * 2. snprintf the string-to-sign from amzdate, scope, and canon_hex; fail
100 * on truncation.
101 */
102static int
103brix_s3_sigv4_build_sts(const char *amzdate, const char *datestamp,
104 const char *region, const char *canon_hex,
105 char *scope, size_t scopesz, char *sts, size_t stssz)
106{
107 int cn;
108
109 cn = snprintf(scope, scopesz, "%s/%s/s3/aws4_request", datestamp, region);
110 if (cn < 0 || (size_t) cn >= scopesz) {
111 return -1;
112 }
113 cn = snprintf(sts, stssz, "AWS4-HMAC-SHA256\n%s\n%s\n%s",
114 amzdate, scope, canon_hex);
115 if (cn < 0 || (size_t) cn >= stssz) {
116 return -1;
117 }
118 return 0;
119}

121/* ---- Derive the signing key, sign the STS, and emit the header block ----
122 *
123 * WHAT: Runs the 4-round HMAC signing-key derivation, HMAC-signs the
124 * string-to-sign, and writes the x-amz-date / x-amz-content-sha256 /
125 * Authorization header block into hdrs. Returns 0 on success, -1 if the
126 * key derivation or HMAC fails or the header block would overflow hdrs.
127 *
128 * WHY: This is the final stage where the derived secret becomes bytes on the
129 * wire; a truncated Authorization header advertises a signature the server
130 * cannot reproduce, so truncation must fail cleanly rather than emit a
131 * partial header.
132 *
133 * HOW: 1. Derive the signing key with the shared libxrdproto 4-round chain and
134 * HMAC-sign the string-to-sign; bail on any crypto failure.
135 * 2. Hex-encode the signature.
136 * 3. snprintf the header block (Credential uses the same scope that was
137 * signed); fail on truncation.
138 */
139static int
140brix_s3_sigv4_sign_and_emit(const char *sk, const char *datestamp,
141 const char *region, const char *sts,
142 const char *ak, const char *payload_hex,
143 const char *amzdate, const char *scope,
144 char *hdrs, size_t hdrsz)
145{
146 uint8_t k4[32], sig[32];
147 char sighex[65];
148 int cn;
149
150 /* Shared 4-round signing-key derive (libxrdproto), then sign the STS. */
151 if (!brix_sigv4_signing_key((const uint8_t *) sk, strlen(sk),
152 datestamp, region, "s3", k4)
153 || !brix_hmac_sha256(k4, 32, (const uint8_t *) sts, strlen(sts), sig)) {
154 return -1;
155 }
156 brix_hex_encode(sig, 32, sighex);
157
158 cn = snprintf(hdrs, hdrsz,
159 "x-amz-date: %s\r\nx-amz-content-sha256: %s\r\n"
160 "Authorization: AWS4-HMAC-SHA256 Credential=%s/%s, "
161 "SignedHeaders=host;x-amz-content-sha256;x-amz-date, Signature=%s\r\n",
162 amzdate, payload_hex, ak, scope, sighex);
163 if (cn < 0 || (size_t) cn >= hdrsz) {
164 return -1;
165 }
166 return 0;
167}

169int
170brix_s3_sign_v4_q(const char *method, const char *host, const char *uri,
171 const char *canon_qs, const char *ak, const char *sk,
172 const char *region, const char *payload_hex,
173 char *hdrs, size_t hdrsz)
174{
175 char amzdate[20], datestamp[12];
176 char canon_hex[65];
177 char scope[160], sts[640];
178 time_t now = time(NULL);
179 struct tm tmv;
180
181 if (canon_qs == NULL) { canon_qs = ""; }
182 if (sk == NULL || ak == NULL || region == NULL || payload_hex == NULL) {
183 return -1;
184 }
185 if (gmtime_r(&now, &tmv) == NULL) {
186 return -1;
187 }
188 strftime(amzdate, sizeof(amzdate), "%Y%m%dT%H%M%SZ", &tmv);
189 strftime(datestamp, sizeof(datestamp), "%Y%m%d", &tmv);
190
191 if (brix_s3_sigv4_build_canon_hash(method, host, uri, canon_qs,
192 payload_hex, amzdate, canon_hex) != 0) {
193 return -1;
194 }
195 if (brix_s3_sigv4_build_sts(amzdate, datestamp, region, canon_hex,
196 scope, sizeof(scope), sts, sizeof(sts)) != 0) {
197 return -1;
198 }
199 return brix_s3_sigv4_sign_and_emit(sk, datestamp, region, sts, ak,
200 payload_hex, amzdate, scope, hdrs, hdrsz);
201}
