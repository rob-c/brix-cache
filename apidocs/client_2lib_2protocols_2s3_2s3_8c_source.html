1/*
2 * s3.c — AWS Signature Version 4 (path-style) for the native client.
3 *
4 * WHAT: brix_s3_sign_v4() builds the x-amz-date / x-amz-content-sha256 /
5 * Authorization header block for any method+path+body-hash; the sha256
6 * helpers compute the payload hashes those headers carry.
7 * WHY: Both xrdcp's s3:// transfers and xrddiag's s3 probe need the identical
8 * signer — keeping one copy in the lib (rather than a private copy in each
9 * app) is the clean-room equivalent of XrdClS3's shared signing code.
10 * HOW: Canonical-request → string-to-sign → HMAC-SHA256 signing-key chain, using
11 * libxrdproto's SHA-256/HMAC kernels (compat/crypto.h). The CanonicalURI is
12 * percent-encoded (RFC-3986 unreserved + '/') to match the server's own
13 * canonicalization. Signed headers are host;x-amz-content-sha256;x-amz-date.
14 *
15 * Clean-room: implements the published SigV4 algorithm (AWS docs), not any SDK.
16 */
17#include "brix.h"
18#include "core/compat/crypto.h" /* brix_sha256 / brix_hmac_sha256 */
19#include "core/compat/hex.h" /* brix_hex_encode */
20#include "core/compat/uri.h" /* brix_http_urlencode (shared SigV4 canonical URI) */
21#include "core/compat/sigv4.h" /* brix_sigv4_signing_key (shared 4-round HMAC chain) */

35int
36brix_s3_sign_v4_q(const char *method, const char *host, const char *uri,
37 const char *canon_qs, const char *ak, const char *sk,
38 const char *region, const char *payload_hex,
39 char *hdrs, size_t hdrsz)
40{
41 char amzdate[20], datestamp[12];
42 char canon_hex[65];
43 char canon[8192], scope[160], sts[640], enc_uri[2048];
44 uint8_t k4[32], sig[32];
45 char sighex[65];
46 time_t now = time(NULL);
47 struct tm tmv;
48 int cn;
49
50 if (canon_qs == NULL) { canon_qs = ""; }
51 if (sk == NULL || ak == NULL || region == NULL || payload_hex == NULL) {
52 return -1;
53 }
54 if (gmtime_r(&now, &tmv) == NULL) {
55 return -1;
56 }
57 strftime(amzdate, sizeof(amzdate), "%Y%m%dT%H%M%SZ", &tmv);
58 strftime(datestamp, sizeof(datestamp), "%Y%m%d", &tmv);
59
60 /* Shared SigV4 CanonicalURI encoder (RFC-3986 unreserved + '/'), byte-identical
61 * to the server's verify path so client-signs == server-verifies. */
62 if (brix_http_urlencode((const unsigned char *) uri, strlen(uri),
63 enc_uri, sizeof(enc_uri), "/") < 0) {
64 return -1;
65 }
66 /* canonical request: method\nURI\nCANONICAL_QUERY\nheaders\nsignedHeaders\nhash.
67 * canon_qs must already be sorted + RFC-3986 encoded (the server re-derives the
68 * same string from the request query). */
69 cn = snprintf(canon, sizeof(canon),
70 "%s\n%s\n%s\nhost:%s\nx-amz-content-sha256:%s\nx-amz-date:%s\n\n"
71 "host;x-amz-content-sha256;x-amz-date\n%s",
72 method, enc_uri, canon_qs, host, payload_hex, amzdate, payload_hex);
73 /* A truncated canonical request would sign a different string than the server
74 * verifies → a confusing 403. Fail cleanly so the caller reports "can't sign". */
75 if (cn < 0 || (size_t) cn >= sizeof(canon)) {
76 return -1;
77 }
78 brix_s3_sha256_hex(canon, strlen(canon), canon_hex);
79
80 /* region is user/env-controlled (--s3-region / $AWS_DEFAULT_REGION); a
81 * truncated scope/sts/hdrs would sign a different string than it advertises →
82 * a confusing 403. Check every snprintf and fail cleanly instead (same
83 * contract as the canon guard above). */
84 cn = snprintf(scope, sizeof(scope), "%s/%s/s3/aws4_request", datestamp, region);
85 if (cn < 0 || (size_t) cn >= sizeof(scope)) {
86 return -1;
87 }
88 cn = snprintf(sts, sizeof(sts), "AWS4-HMAC-SHA256\n%s\n%s\n%s",
89 amzdate, scope, canon_hex);
90 if (cn < 0 || (size_t) cn >= sizeof(sts)) {
91 return -1;
92 }
93
94 /* Shared 4-round signing-key derive (libxrdproto), then sign the STS. */
95 if (!brix_sigv4_signing_key((const uint8_t *) sk, strlen(sk),
96 datestamp, region, "s3", k4)
97 || !brix_hmac_sha256(k4, 32, (const uint8_t *) sts, strlen(sts), sig)) {
98 return -1;
99 }
100 brix_hex_encode(sig, 32, sighex);
101
102 cn = snprintf(hdrs, hdrsz,
103 "x-amz-date: %s\r\nx-amz-content-sha256: %s\r\n"
104 "Authorization: AWS4-HMAC-SHA256 Credential=%s/%s, "
105 "SignedHeaders=host;x-amz-content-sha256;x-amz-date, Signature=%s\r\n",
106 amzdate, payload_hex, ak, scope, sighex);
107 if (cn < 0 || (size_t) cn >= hdrsz) {
108 return -1;
109 }
110 return 0;
111}
