1/*
2 * vfs_wverify.c — VFS seam for self-computed write-verify (read-back compare).
3 *
4 * WHAT: brix_vfs_wverify_check() — given a write-side CRC accumulator and a
5 * FRESH read-only handle on the just-written object, re-read the object
6 * through its storage driver and confirm the persisted bytes match what
7 * was written.
8 * WHY: An object backend (pblock, rados) keeps no single kernel file whose
9 * bytes are the object, so "did the write actually land?" can only be
10 * answered by reading it back through the same driver. This is the reusable
11 * seam any protocol write path (GridFTP STOR, WebDAV/root/S3 PUT) can call
12 * after close to fail a silently-corrupted or short write.
13 * HOW: brix_wverify_expected() gives the whole-object CRC-32 and length the
14 * write should have produced (only when the writes covered [0, size) with
15 * no gap/overlap); the read handle's size must match, and a driver-routed
16 * brix_cksum_u32_obj(BRIX_CK_CRC32, …) over the reopened object must equal
17 * the expected CRC. Any divergence → NGX_ERROR (the caller unlinks + fails
18 * the transfer).
19 */
20#include "vfs_internal.h"
21#include "core/compat/wverify.h"
22#include "core/compat/checksum_core.h"

24ngx_int_t
25brix_vfs_wverify_check(brix_wverify_t *w, brix_vfs_file_t *rfh)
26{
27 brix_sd_obj_t obj;
28 uint32_t expect = 0, actual = 0;
29 off_t total = 0;
30
31 if (w == NULL || rfh == NULL) {
32 return NGX_ERROR;
33 }
34
35 /* What the write path should have persisted: a single [0, total) run. A gap,
36 * an overlap, or a degraded accumulator all fail closed here. */
37 if (brix_wverify_expected(w, &expect, &total) != 0) {
38 return NGX_ERROR;
39 }
40 if (total != brix_vfs_file_size(rfh)) {
41 return NGX_ERROR; /* persisted length disagrees */
42 }
43
44 /* Read the object back through the SAME driver and compare content. */
45 brix_vfs_file_sd_obj(rfh, &obj);
46 if (obj.driver == NULL
47 || brix_cksum_u32_obj(BRIX_CK_CRC32, &obj, &actual) != 0)
48 {
49 return NGX_ERROR;
50 }
51 return (actual == expect) ? NGX_OK : NGX_ERROR;
52}
