4/*
5 * ftp_eblock.h — GridFTP MODE E extended-block framing (GFD.020 §3.4).
6 *
7 * WHAT: the 17-byte extended-block header codec (descriptor + count + offset),
8 * the descriptor flag constants, and the committed-range overlap guard shared by
9 * both gateway engines' MODE E paths.
10 *
11 * WHY: MODE E frames a transfer as offset-addressed records that arrive out of
12 * order across up to `Parallelism` data connections — there is no shared stream
13 * cursor, so every block carries its own absolute file offset and a receiver must
14 * reject a block that overlaps an already-committed range (corruption or a
15 * deliberate overwrite, §2.6 R4). The wire codec and the overlap test are pure,
16 * self-contained functions; keeping one definition here means the event engine's
17 * RETR framer and STOR receiver (ev/ftp_ev_mode_e.c) frame identically and cannot
18 * drift. Marker formatting and the reassembly state machine stay in the engine
19 * (they touch its reply path and I/O model).
20 *
21 * HOW: header-only `static ngx_inline` — each translation unit inlines its own
22 * copy from this single source; GCC does not warn on an unused static inline, so
23 * a TU that needs only the codec (RETR framing) and not the range guard compiles
24 * clean under -Werror.
25 */

32/* 17-byte header, network byte order: 1-byte descriptor + 8-byte count + 8-byte
33 * offset, then `count` payload bytes. An EOF-flagged block carries no payload
34 * and puts the total EOD count in its OFFSET field (globus convention); each data
35 * connection ends with exactly one EOD. */
36#define FTP_EB_HDR 17
37#define FTP_EB_EOF 0x40 /* no payload; offset field = total EOD count */
38#define FTP_EB_EOD 0x08 /* last block on this data connection */

42ftp_eb_pack(u_char *h, u_char desc, uint64_t count, uint64_t offset)
43{
44 int i;
45 h[0] = desc;
46 for (i = 0; i < 8; i++) { h[1 + i] = (u_char) (count >> (56 - 8 * i)); }
47 for (i = 0; i < 8; i++) { h[9 + i] = (u_char) (offset >> (56 - 8 * i)); }
48}

51static ngx_inline void
52ftp_eb_unpack(const u_char *h, u_char *desc, uint64_t *count, uint64_t *offset)
53{
54 uint64_t c = 0, o = 0;
55 int i;
56 *desc = h[0];
57 for (i = 0; i < 8; i++) { c = (c << 8) | h[1 + i]; }
58 for (i = 0; i < 8; i++) { o = (o << 8) | h[9 + i]; }
59 *count = c;
60 *offset = o;
61}
