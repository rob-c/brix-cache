37/* pb_close — close an object and free its malloc'd shell. driver->close frees the
38 * per-open state + fd but not the obj struct (the VFS adopts that by value), so a
39 * direct caller owns the shell. */
40ngx_int_t
41pb_close(brix_sd_obj_t *o)
42{
43 ngx_int_t rc = D->close(o);
44
45 free(o);
46 return rc;
47}

51/* write_file — create `path`, write `data`, close. Returns 0 or -1. */
52int
53write_file(brix_sd_instance_t *inst, const char *path, const char *data,
54 size_t len)
55{
56 int err = 0;
57 brix_sd_obj_t *o;
58 ssize_t n;
59
60 o = D->open(inst, path,
61 BRIX_SD_O_WRITE | BRIX_SD_O_READ | BRIX_SD_O_CREATE,
62 0644, &err);
63 if (o == NULL) {
64 return -1;
65 }
66 n = D->pwrite(o, data, len, 0);
67 pb_close(o);
68 return (n == (ssize_t) len) ? 0 : -1;
69}

71/* read_file — open `path` read-only, read up to cap bytes at 0. Returns bytes
72 * read or -1. */
73ssize_t
74read_file(brix_sd_instance_t *inst, const char *path, char *buf, size_t cap)
75{
76 int err = 0;
77 brix_sd_obj_t *o;
78 ssize_t n;
79
80 o = D->open(inst, path, BRIX_SD_O_READ, 0, &err);
81 if (o == NULL) {
82 errno = err;
83 return -1;
84 }
85 n = D->pread(o, buf, cap, 0);
86 pb_close(o);
87 return n;
