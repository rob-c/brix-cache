343 int id;
344 int ops;
345 int ok;

348/* Each thread writes + reads back its own private files through the SHARED
349 * instance (shared SQLite connection + blob dir), proving the per-worker handle
350 * is safe across a thread pool and that blob I/O runs in parallel. */
351static void *
352thread_body(void *p)
353{
354 thread_arg *a = p;
355 int i;
356
357 for (i = 0; i < a->ops; i++) {
358 char path[64];
359 char buf[32];
360 char want[32];
361 int wn = snprintf(want, sizeof(want), "t%d-op%d", a->id, i);
362
363 snprintf(path, sizeof(path), "/conc/t%d-%d", a->id, i);
364 if (write_file(a->inst, path, want, (size_t) wn) != 0) {
365 return NULL;
366 }
367 if (read_file(a->inst, path, buf, sizeof(buf)) != wn
368 || memcmp(buf, want, (size_t) wn) != 0)
369 {
370 return NULL;
371 }
372 }
373 a->ok = 1;
374 return NULL;
375}

400/* Each forked child opens its OWN instance on the same root and writes a batch;
401 * the parent then verifies every child's files landed. Exercises cross-process
402 * WAL locking + the busy-timeout retry path. */
403static void
404test_processes(const char *root, brix_sd_instance_t *inst)
405{
406 enum { NPROC = 4, NOPS = 25 };
407 pid_t pids[NPROC];
408 int i, total = 0, rc;
409 brix_sd_dir_t *dir;
410 brix_sd_dirent_t de;

412
413 CHECK(D->mkdir(inst, "/mp", 0755) == NGX_OK, "mkdir mp");
414
415 for (i = 0; i < NPROC; i++) {
416 pid_t pid = fork();
417

436 D->cleanup(&cinst);
437 _exit(bad ? 1 : 0);

456 }
457 D->closedir(dir);

1153/* F14: catalog enumeration is a flat scan reporting every stored file (never a
1154 * directory) at any depth in one pass — three independent oracles (enumerate
1155 * count, enumerate size-sum, a recursive namespace walk) must agree. The scan is
1156 * UNCONFINED: a nested object comes back by its full path with no subtree gate,
1157 * which is precisely why enumeration is a service-plane inventory verb — there is
1158 * no cred-scoped variant, so it is unreachable from user protocols. An aborting
1159 * callback stops the walk early yet the slot still returns NGX_OK. */
1160typedef struct {
1161 int files; /* non-directory rows reported */
1162 int64_t bytes; /* summed size (valid only when want_stat) */
1163 int saw_nested; /* the nested /d/c path was reported */
1164 int stop_after; /* 0 = run to completion, else abort after N */
1165} enum_probe_t;
1166
1167static int
1168enum_probe_cb(void *ctx, const brix_sd_catalog_ent_t *ent)
1169{
1170 enum_probe_t *p = ctx;
1171
1172 if (ent->key == NULL) {
1173 return 0;
1174 }
1175 p->files++;
1176 if (ent->have_stat) {
1177 p->bytes += (int64_t) ent->size;
1178 }
1179 if (ent->path != NULL && strcmp(ent->path, "/d/c") == 0) {
1180 p->saw_nested = 1;
1181 }
1182 return (p->stop_after && p->files >= p->stop_after) ? 1 : 0;
1183}
1184
1185/* Independent oracle: recursively walk the namespace, counting regular files and
1186 * summing their stat sizes. opendir/readdir/stat only — no catalog SELECT. */
1187static void
1188enum_walk(brix_sd_instance_t *inst, const char *dir, int *files, int64_t *bytes)
1189{
1190 brix_sd_dir_t *d;
1191 brix_sd_dirent_t de;
1192 int err = 0;
1193
1194 d = D->opendir(inst, dir, &err);
1195 if (d == NULL) {
1196 return;
1197 }
1198 while (D->readdir(d, &de) == NGX_OK) {
1199 char child[512];
1200 brix_sd_stat_t st;
1201
1202 (void) snprintf(child, sizeof(child), "%s%s%s", dir,
1203 (strcmp(dir, "/") == 0) ? "" : "/", de.name);
1204 if (D->stat(inst, child, &st) != NGX_OK) {
1205 continue;
1206 }
1207 if (st.is_dir) {
1208 enum_walk(inst, child, files, bytes);
1209 } else {
1210 (*files)++;
1211 *bytes += (int64_t) st.size;
1212 }
1213 }
1214 D->closedir(d);
1215}

1349/* staged_put — publish `data` at `path` through the atomic staged path (the same
