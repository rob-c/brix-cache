44/* pb_close — close an object and free its malloc'd shell. driver->close frees the
45 * per-open state + fd but not the obj struct (the VFS adopts that by value), so a
46 * direct caller owns the shell. */
47static ngx_int_t
48pb_close(brix_sd_obj_t *o)
49{
50 ngx_int_t rc = D->close(o);
51
52 free(o);
53 return rc;
54}

58/* write_file — create `path`, write `data`, close. Returns 0 or -1. */
59static int
60write_file(brix_sd_instance_t *inst, const char *path, const char *data,
61 size_t len)
62{
63 int err = 0;
64 brix_sd_obj_t *o;
65 ssize_t n;
66
67 o = D->open(inst, path,
68 BRIX_SD_O_WRITE | BRIX_SD_O_READ | BRIX_SD_O_CREATE,
69 0644, &err);
70 if (o == NULL) {
71 return -1;
72 }
73 n = D->pwrite(o, data, len, 0);
74 pb_close(o);
75 return (n == (ssize_t) len) ? 0 : -1;
76}

339typedef struct {
340 brix_sd_instance_t *inst;
341 int id;
342 int ops;
343 int ok;
344} thread_arg;
345
346/* Each thread writes + reads back its own private files through the SHARED
347 * instance (shared SQLite connection + blob dir), proving the per-worker handle
348 * is safe across a thread pool and that blob I/O runs in parallel. */
349static void *
350thread_body(void *p)
351{
352 thread_arg *a = p;
353 int i;
354
355 for (i = 0; i < a->ops; i++) {
356 char path[64];
357 char buf[32];
358 char want[32];
359 int wn = snprintf(want, sizeof(want), "t%d-op%d", a->id, i);
360
361 snprintf(path, sizeof(path), "/conc/t%d-%d", a->id, i);
362 if (write_file(a->inst, path, want, (size_t) wn) != 0) {
363 return NULL;
364 }
365 if (read_file(a->inst, path, buf, sizeof(buf)) != wn
366 || memcmp(buf, want, (size_t) wn) != 0)
367 {
368 return NULL;
369 }
370 }
371 a->ok = 1;
372 return NULL;
373}

398/* Each forked child opens its OWN instance on the same root and writes a batch;
399 * the parent then verifies every child's files landed. Exercises cross-process
400 * WAL locking + the busy-timeout retry path. */
401static void
402test_processes(const char *root, brix_sd_instance_t *inst)
403{
404 enum { NPROC = 4, NOPS = 25 };
405 pid_t pids[NPROC];
406 int i, total = 0, rc;
407 brix_sd_dir_t *dir;
408 brix_sd_dirent_t de;
409 int err = 0;
410
411 CHECK(D->mkdir(inst, "/mp", 0755) == NGX_OK, "mkdir mp");
412
413 for (i = 0; i < NPROC; i++) {
414 pid_t pid = fork();
415
416 if (pid == 0) {
417 brix_sd_instance_t cinst = {0};
418 brix_sd_pblock_conf_t conf = { root, 5000, 0 };
419 int j, bad = 0;
420
421 cinst.driver = D;
422 if (D->init(&cinst, &conf) != NGX_OK) {
423 _exit(2);
424 }
425 for (j = 0; j < NOPS; j++) {
426 char path[64];
427
428 snprintf(path, sizeof(path), "/mp/p%d-%d", (int) getpid(), j);
429 if (write_file(&cinst, path, "x", 1) != 0) {
430 bad = 1;
431 break;
432 }
433 }
434 D->cleanup(&cinst);
435 _exit(bad ? 1 : 0);
436 }
437 CHECK(pid > 0, "fork %d", i);
438 pids[i] = pid;
439 }
440
441 for (i = 0; i < NPROC; i++) {
442 int status = 0;
443
444 waitpid(pids[i], &status, 0);
445 CHECK(WIFEXITED(status) && WEXITSTATUS(status) == 0,
446 "child %d exit status %d", i, status);
447 }
448
449 dir = D->opendir(inst, "/mp", &err);
450 CHECK(dir != NULL, "opendir mp");
451 if (dir != NULL) {
452 while ((rc = D->readdir(dir, &de)) == NGX_OK) {
453 total++;
454 }
455 D->closedir(dir);
456 }
457 CHECK(total == NPROC * NOPS, "expected %d files, found %d", NPROC * NOPS,
458 total);
459}

465static int
466blk_walk_cb(const char *path, const struct stat *sb, int type, struct FTW *ftw)
467{
468 (void) path;
469 (void) ftw;
470 if (type == FTW_F) {
471 g_blk_count++;
472 g_blk_total += (long) sb->st_size;
473 if ((long) sb->st_size > g_blk_max) {
474 g_blk_max = (long) sb->st_size;
475 }
476 }
477 return 0;
478}

480/* scan_blocks — tally the on-disk block files under <root>/data: count, total
481 * bytes, and the largest single block (which must never exceed block_size). */
482static void
483scan_blocks(const char *root)
484{
485 char data[512];
486
487 snprintf(data, sizeof(data), "%s/data", root);
488 g_blk_count = g_blk_total = g_blk_max = 0;
489 nftw(data, blk_walk_cb, 16, FTW_PHYS);
490}

492/* pblock_open_block_export — fresh export with a small block size so striping is
493 * exercised with a handful of bytes. Returns 0/-1 and fills *inst + *root. */
494static int
495open_block_export(brix_sd_instance_t *inst, char *root, int64_t block_size)
496{
497 static brix_sd_pblock_conf_t conf; /* root must outlive the instance */
498
499 conf.root = root;
500 conf.busy_timeout_ms = 2000;
501 conf.block_size = block_size;
502 memset(inst, 0, sizeof(*inst));
503 inst->driver = D;
504 return D->init(inst, &conf) == NGX_OK ? 0 : -1;
505}
