2 * config_merge.c - WebDAV base directive merge + startup validation cluster.
3 *
4 * WHAT: the parent→child directive inheritance/default pass (webdav_merge_base_conf)

51{

55 brix_http_common_adopt(cf, &conf->common);

58 * WebDAV exports default to "/" (pure cache nodes serve the whole ns). */

60 != NGX_CONF_OK)

214static char *
215webdav_set_storage_credential(ngx_conf_t *cf,
216 ngx_http_brix_webdav_loc_conf_t *conf)
217{
218 char cred_z[256];
219 char bearer[4096];
220 const brix_credential_t *cred;
221 brix_vfs_backend_cred_t bcred;
222
223 if (conf->common.storage_credential.len == 0) {
224 return NGX_CONF_OK;
225 }
226 ngx_cpystrn((u_char *) cred_z, conf->common.storage_credential.data,
227 ngx_min(conf->common.storage_credential.len + 1,
228 sizeof(cred_z)));
229 cred = brix_credential_lookup(cred_z);
230 if (cred == NULL) {
231 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
232 "brix_webdav_storage_credential: no brix_credential \"%V\"",
233 &conf->common.storage_credential);
234 return NGX_CONF_ERROR;
235 }
236 if (brix_credential_to_backend_cred(cred, bearer, sizeof(bearer),
237 &bcred, cf->log) != NGX_OK)
238 {
239 return NGX_CONF_ERROR;
240 }
241 brix_vfs_backend_set_credential(conf->common.root_canon, &bcred);
242 return NGX_CONF_OK;
243}

318static char *
319webdav_validate_auth_paths(ngx_conf_t *cf,
320 ngx_http_brix_webdav_loc_conf_t *conf)
321{
322 /* auth optional/required needs at least ONE credential verifier: an x509
323 * CA (cadir/cafile), a token verifier (JWKS / issuer registry / macaroon
324 * secret), or a Basic password db — otherwise every client is rejected
325 * and the misconfiguration should fail `nginx -t`, not surface as 403s. */
326 if ((conf->auth == WEBDAV_AUTH_OPTIONAL
327 || conf->auth == WEBDAV_AUTH_REQUIRED)
328 && conf->cadir.len == 0 && conf->cafile.len == 0
329 && conf->token_jwks.len == 0 && conf->token_config.len == 0
330 && conf->token_macaroon_secret.len == 0 && conf->pwd_file.len == 0)
331 {
332 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
333 "brix_webdav: auth optional/required needs a credential verifier "
334 "— brix_webdav_cadir/cafile, brix_webdav_token_jwks/config/"
335 "macaroon_secret, or brix_webdav_pwd_file");
336 return NGX_CONF_ERROR;
337 }
338 /* E-1: a writable export that does not REQUIRE authentication lets an
339 * unauthenticated client create/overwrite/delete objects. Warn always;
340 * refuse under strict security. */
341 if (conf->common.allow_write && conf->auth != WEBDAV_AUTH_REQUIRED
342 && brix_shared_security_gate(cf, conf->common.strict_security,
343 "WebDAV export permits unauthenticated writes "
344 "(brix_allow_write on but brix_webdav_auth is not required)",
345 "brix_webdav_auth required") != NGX_OK)
346 {
347 return NGX_CONF_ERROR;
348 }
349 if (webdav_validate_path(cf, "brix_webdav_cadir", &conf->cadir,
350 WEBDAV_PATH_DIRECTORY, R_OK | X_OK) != NGX_OK
351 || webdav_validate_path(cf, "brix_webdav_cafile", &conf->cafile,
352 WEBDAV_PATH_REGULAR_FILE, R_OK) != NGX_OK
353 || webdav_validate_path(cf, "brix_webdav_crl", &conf->crl,
354 WEBDAV_PATH_FILE_OR_DIRECTORY, R_OK) != NGX_OK)
355 {
356 return NGX_CONF_ERROR;
357 }
358 if (webdav_validate_cors_origins(cf, conf) != NGX_OK) {
359 return NGX_CONF_ERROR;
360 }
361 return NGX_CONF_OK;
362}

364static char *
365webdav_build_ca_store_once(ngx_conf_t *cf,
366 ngx_http_brix_webdav_loc_conf_t *conf)
367{
368 X509_STORE *store;
369 ngx_pool_cleanup_t *cln;
370 int crl_count = 0;
371
372 if (conf->auth != WEBDAV_AUTH_OPTIONAL
373 && conf->auth != WEBDAV_AUTH_REQUIRED)
374 {
375 return NGX_CONF_OK;
376 }
377 /* Token-only / Basic-only exports carry no x509 trust anchors: there is
378 * no store to build, and the cert tier declines at request time. */
379 if (conf->cadir.len == 0 && conf->cafile.len == 0) {
380 return NGX_CONF_OK;
381 }
382 store = webdav_build_ca_store(cf->log, conf, &crl_count);
383 if (store == NULL) {
384 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
385 "brix_webdav: failed to build cached CA store");
386 return NGX_CONF_ERROR;
387 }
388 (void) webdav_check_pki_consistency(cf->log, conf);
389 cln = ngx_pool_cleanup_add(cf->pool, 0);
390 if (cln == NULL) {
391 X509_STORE_free(store);
392 return NGX_CONF_ERROR;
393 }
394 cln->handler = webdav_x509_store_cleanup;
395 cln->data = store;
396 conf->ca_store = store;
397 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
398 "brix_webdav: cached CA store built for root=\"%V\" crls=%d",
399 &conf->common.root, crl_count);
400 return NGX_CONF_OK;
401}
