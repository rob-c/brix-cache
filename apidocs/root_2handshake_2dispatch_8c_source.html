8/*
9 * Request routing overview
10 * ========================
11 *
12 * brix_dispatch() tries each of the four dispatch functions in order.
13 * Each returns BRIX_DISPATCH_CONTINUE if the opcode is not its own;
14 * otherwise it handles the request and returns an ngx_int_t result.
15 *
16 * dispatch_session.c — protocol, login, auth, bind, endsess, ping, set
17 * proxy/forward.c — all post-login opcodes when brix_proxy is on
18 * dispatch_read.c — open(read), stat, statx, read, readv, pgread,
19 * close, dirlist, locate, query, prepare
20 * dispatch_write.c — open(write), write, pgwrite, writev, sync,
21 * truncate, mkdir, rm, rmdir, mv, chmod, fattr,
22 * clone, chkpoint
23 * dispatch_signing.c — sigver (must be last; inspects every request)
24 *
25 * Adding a new opcode: determine its category above, add a case to the
26 * matching dispatch_*.c file, then see docs/contributing.md §5 for the
27 * full checklist.
28 */
29

31brix_dispatch(brix_ctx_t *ctx, ngx_connection_t *c,
32 ngx_stream_brix_srv_conf_t *conf)
33{
34 ngx_int_t rc;
35
36 /* Every dispatched request resets the per-request timing origin for logging. */
37 ctx->req_start = ngx_current_msec;
38
39 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
40 "brix: dispatch reqid=%d", (int) ctx->recv.cur_reqid);
41
42 rc = brix_verify_pending_sigver(ctx, c);
43 if (rc != BRIX_DISPATCH_CONTINUE) {
44 return rc;
45 }
46
47 rc = brix_signing_enforce_level(ctx, c, conf);
48 if (rc != BRIX_DISPATCH_CONTINUE) {
49 return rc;
50 }
51
52 rc = brix_dispatch_session_opcode(ctx, c, conf);
53 if (rc != BRIX_DISPATCH_CONTINUE) {
54 return rc;
55 }
56
57 /*
58 * D-1: enforce the session-posture floor (brix_min_sec_level). Placed after
59 * the session opcodes are dispatched so the login/auth/protocol/bind/TLS-
60 * upgrade handshake is never blocked; every opcode reaching here is a
61 * data/metadata request, so a below-floor (cleartext, or anonymous under
62 * intense) session is refused before it can touch the proxy or the FS.
63 */
64 rc = brix_min_sec_enforce(ctx, c, conf);
65 if (rc != BRIX_DISPATCH_CONTINUE) {
66 return rc;
67 }
68
69 /*
70 * Proxy mode: all post-login file-system opcodes go to the upstream.
71 *
72 * SECURITY (fail-closed): gate on ctx->login.auth_done, NOT merely ctx->login.logged_in.
73 * kXR_login only sets logged_in; for any configured auth (gsi/token/sss/...)
74 * auth_done stays 0 until kXR_auth completes (login.c). The session opcodes
75 * (login/auth/bind/...) are already handled above, so anything reaching here
76 * is a file-system opcode that MUST be authenticated before we forward it to
77 * the upstream under the proxy's own bridged credentials. Gating on
78 * logged_in alone let a client send kXR_login then skip kXR_auth and still
79 * reach upstream resources. Anonymous mode (auth=none) sets auth_done=1 at
80 * login, so this remains a no-op gate there. This mirrors the require_auth
81 * gate enforced on the direct (non-proxy) read/write dispatchers.
82 */
83 if (conf->proxy.enable && ctx->login.auth_done) {
84 return brix_proxy_dispatch(ctx, c, conf);
85 }
86
87 /* Phase 25: advanced rate limiting / traffic shaping. Gates data-plane
88 * read/write opcodes; a throttled request is answered with kXR_wait and the
89 * gate returns the send result (NGX_DECLINED means "proceed normally"). */
90 rc = brix_rl_stream_gate(ctx, c, conf);
91 if (rc != NGX_DECLINED) {
92 return rc;
93 }
94
95 /*
96 * Phase 40: bracket the confined-FS dispatchers with the impersonation
97 * principal taken from the authenticated identity. begin()/end() are no-ops
98 * unless brix_impersonation=map; when active they make the beneath helpers
99 * (open/stat/mkdir/...) route to the broker as the mapped UNIX user for the
100 * duration of this synchronous dispatch, then restore the worker identity.
101 * The data plane (kXR_read/write on the already-open fd) and the mirror/RL
102 * paths run unbracketed — they need no impersonation.
103 */
104 brix_imp_request_begin(ctx->identity);
105 rc = brix_dispatch_read_opcode(ctx, c, conf);
106 brix_imp_request_end();
107 if (rc != BRIX_DISPATCH_CONTINUE) {
108 /* Phase 24: fire-and-forget replay of this read to the shadow server(s).
109 * No-op unless brix_stream_mirror_url is configured; the primary
110 * response has already been queued to the client above. */
111 brix_stream_mirror_maybe(ctx, c, conf, rc);
112 /* Phase 24 W3: kXR_close finalises a data-write mirror (open/write/close
113 * is dispatched across the read+write tables). */
114 brix_stream_wmirror_observe(ctx, c, conf, rc);
115 return rc;
116 }
117
118 brix_imp_request_begin(ctx->identity);
119 rc = brix_dispatch_write_opcode(ctx, c, conf);
120 brix_imp_request_end();
121 if (rc != BRIX_DISPATCH_CONTINUE) {
122 /* Phase 24 write mirroring (W1): replay self-contained metadata mutations
123 * (mkdir/rm/rmdir/mv/truncate/chmod) to the shadow. No-op unless
124 * brix_mirror_writes is on and the op is listed in brix_mirror_opcodes;
125 * the primary response was already queued to the client above. */
126 brix_stream_mirror_maybe(ctx, c, conf, rc);
127 /* Phase 24 W3: accumulate kXR_write / kXR_pgwrite payloads for the
128 * data-write mirror (replayed to the shadow on kXR_close). */
129 brix_stream_wmirror_observe(ctx, c, conf, rc);
130 return rc;
131 }
132
133 rc = brix_dispatch_signing_opcode(ctx, c);
134 if (rc != BRIX_DISPATCH_CONTINUE) {
135 return rc;
136 }
137
138 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
139 "brix: unsupported request %d",
140 (int) ctx->recv.cur_reqid);
141 /* An unrecognized opcode is kXR_InvalidRequest ("Invalid request code"),
142 * matching the reference (XrdXrootdProtocol.cc:608); kXR_Unsupported is
143 * reserved for a recognized op the backend cannot perform (ENOTSUP). */
144 return brix_send_error(ctx, c, kXR_InvalidRequest,
145 "Invalid request code");
146}
