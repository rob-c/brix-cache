1/*
2 * http_mirror_config.c — config-time wiring for the Phase 24 HTTP/WebDAV traffic
3 * mirror (see http_mirror.h).
4 *
5 * WHAT: Owns the merge-time upstream-conf builder (timeouts, buffers, TLS ctx,
6 * hide-headers hash) and the two directive setters `brix_mirror_url` (append a
7 * resolved shadow target) and `brix_mirror_methods` (parse the method mask).
8 * WHY: split out of http_mirror.c (phase-79 file-size cap) so all config-plane
9 * code — which runs only at nginx configuration time and shares nothing with the
10 * request-time path — lives in one focused file.
11 * HOW: brix_http_mirror_setup() is called from the WebDAV loc-conf merge; the two
12 * setters are registered in the WebDAV directive table (src/webdav/module.c) and
13 * populate conf->mirror. All three are declared in http_mirror.h.
14 */
15#include "http_mirror.h"

84char *
85brix_http_mirror_set_url(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
86{
87 ngx_http_brix_webdav_loc_conf_t *wlcf = conf;
88 ngx_str_t *value = cf->args->elts;
89 ngx_str_t url = value[1];
90 brix_mirror_target_t *t;
91 ngx_url_t u;
92 size_t scheme_len;
93 ngx_uint_t ssl;
94 in_port_t default_port;
95 u_char *p;
96
97 (void) cmd;
98
99 if (ngx_strncasecmp(url.data, (u_char *) "https://", 8) == 0) {
100 ssl = 1; scheme_len = 8; default_port = 443;
101 } else if (ngx_strncasecmp(url.data, (u_char *) "http://", 7) == 0) {
102 ssl = 0; scheme_len = 7; default_port = 80;
103 } else {
104 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
105 "brix_mirror_url: \"%V\" must start with http:// or https://",
106 &url);
107 return NGX_CONF_ERROR;
108 }
109
110 if (wlcf->mirror.targets == NULL) {
111 wlcf->mirror.targets = ngx_array_create(cf->pool,
112 BRIX_MIRROR_MAX_TARGETS, sizeof(brix_mirror_target_t));
113 if (wlcf->mirror.targets == NULL) { return NGX_CONF_ERROR; }
114 }
115 if (wlcf->mirror.targets->nelts >= BRIX_MIRROR_MAX_TARGETS) {
116 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
117 "brix_mirror_url: at most %d targets supported",
118 BRIX_MIRROR_MAX_TARGETS);
119 return NGX_CONF_ERROR;
120 }
121
122 ngx_memzero(&u, sizeof(ngx_url_t));
123 u.url.data = url.data + scheme_len;
124 u.url.len = url.len - scheme_len;
125 u.uri_part = 1;
126 u.default_port = default_port;
127 if (ngx_parse_url(cf->pool, &u) != NGX_OK || u.naddrs == 0) {
128 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
129 "brix_mirror_url: cannot resolve \"%V\"%s%s", &url,
130 u.err ? ": " : "", u.err ? u.err : "");
131 return NGX_CONF_ERROR;
132 }
133
134 t = ngx_array_push(wlcf->mirror.targets);
135 if (t == NULL) { return NGX_CONF_ERROR; }
136 ngx_memzero(t, sizeof(*t));
137 t->url = url;
138 t->ssl = ssl;
139 t->port = u.port;
140
141 /* Host: header value — "host" or "host:port" if non-default. */
142 if (u.port == default_port) {
143 t->host = u.host;
144 } else {
145 p = ngx_pnalloc(cf->pool, u.host.len + 1 + sizeof("65535") - 1);
146 if (p == NULL) { return NGX_CONF_ERROR; }
147 t->host.data = p;
148 t->host.len = ngx_sprintf(p, "%V:%d", &u.host, (int) u.port) - p;
149 }
150
151 /* url_base — "scheme://host[:port]" for logging. */
152 {
153 size_t base = scheme_len + t->host.len;
154 p = ngx_pnalloc(cf->pool, base);
155 if (p == NULL) { return NGX_CONF_ERROR; }
156 ngx_memcpy(p, url.data, scheme_len);
157 ngx_memcpy(p + scheme_len, t->host.data, t->host.len);
158 t->url_base.data = p;
159 t->url_base.len = base;
160 }
161
162 if (u.addrs[0].socklen > sizeof(t->sockaddr)) { return NGX_CONF_ERROR; }
163 ngx_memcpy(&t->sockaddr, u.addrs[0].sockaddr, u.addrs[0].socklen);
164 t->socklen = u.addrs[0].socklen;
165
166 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
167 "brix: WebDAV mirror target %V (ssl=%d)", &t->url_base, (int) ssl);
168 return NGX_CONF_OK;
169}

171char *
172brix_http_mirror_set_methods(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
173{
174 ngx_http_brix_webdav_loc_conf_t *wlcf = conf;
175 ngx_str_t *value = cf->args->elts;
176 ngx_uint_t i, mask = 0;
177
178 (void) cmd;
179
180 for (i = 1; i < cf->args->nelts; i++) {
181 ngx_str_t *v = &value[i];
182 if (ngx_strcasecmp(v->data, (u_char *) "GET") == 0) mask |= BRIX_MIRROR_M_GET;
183 else if (ngx_strcasecmp(v->data, (u_char *) "HEAD") == 0) mask |= BRIX_MIRROR_M_HEAD;
184 else if (ngx_strcasecmp(v->data, (u_char *) "PROPFIND") == 0) mask |= BRIX_MIRROR_M_PROPFIND;
185 else if (ngx_strcasecmp(v->data, (u_char *) "OPTIONS") == 0) mask |= BRIX_MIRROR_M_OPTIONS;
186 /* Write methods (require brix_mirror_writes on; isolated shadow). */
187 else if (ngx_strcasecmp(v->data, (u_char *) "PUT") == 0) mask |= BRIX_MIRROR_M_PUT;
188 else if (ngx_strcasecmp(v->data, (u_char *) "DELETE") == 0) mask |= BRIX_MIRROR_M_DELETE;
189 else if (ngx_strcasecmp(v->data, (u_char *) "MKCOL") == 0) mask |= BRIX_MIRROR_M_MKCOL;
190 else if (ngx_strcasecmp(v->data, (u_char *) "MOVE") == 0) mask |= BRIX_MIRROR_M_MOVE;
191 else if (ngx_strcasecmp(v->data, (u_char *) "COPY") == 0) mask |= BRIX_MIRROR_M_COPY;
192 else {
193 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
194 "brix_mirror_methods: unsupported method \"%V\" (one of"
195 " GET HEAD PROPFIND OPTIONS PUT DELETE MKCOL MOVE COPY;"
196 " write methods also need brix_mirror_writes on)", v);
197 return NGX_CONF_ERROR;
198 }
199 }
200 wlcf->mirror.method_mask = mask;
201 return NGX_CONF_OK;
202}
