40/* ---- Report whether a byte range is already resident in the read-ahead buffer ----
41 *
42 * WHAT: Returns non-zero when [off, off+n) lies entirely within the currently
43 * buffered window [rbuf_off, rbuf_off+rbuf_len); returns zero otherwise (buffer
44 * empty, range starts before the window, or range extends past the window end).
45 *
46 * WHY: The refill decision reads clearest as "refill unless already cached". A
47 * single named predicate keeps the caller from carrying an inverted compound
48 * condition that is easy to get subtly wrong.
49 *
50 * HOW:
51 * 1. A window with rbuf_len == 0 holds nothing, so the range is not cached.
52 * 2. The range must start at or after the window start (off >= rbuf_off).
53 * 3. The range end, measured from the window start, must fit within rbuf_len.
54 */
55static int
56iobuf_range_cached(const brix_iobuf *b, off_t off, size_t n)
57{
58 return b->rbuf_len > 0 && off >= b->rbuf_off
59 && (size_t) (off - b->rbuf_off) + n <= b->rbuf_len;
60}

62/* ---- Refill the read-ahead buffer at a new offset ----
63 *
64 * WHAT: Issues one backend pread into rbuf starting at off and records the new
65 * window (rbuf_off/rbuf_len). Returns 0 on success, or -1 on backend error (with
66 * st set by the backend).
67 *
68 * WHY: Isolates the fetch-and-record side effect so the serve path stays pure and
69 * the read orchestrator stays a flat sequence.
70 *
71 * HOW:
72 * 1. Prefetch the whole read-ahead window on sequential access (off == ra_next),
73 * otherwise fetch only the requested n bytes so random reads don't
74 * repeatedly over-fetch.
75 * 2. pread into rbuf; propagate a negative result as -1.
76 * 3. Record the window origin and the (possibly short, at EOF) fill length.
77 */
78static int
79iobuf_refill(brix_iobuf *b, off_t off, size_t n, brix_status *st)
80{
81 ssize_t r;
82 size_t fill = (off == b->ra_next) ? b->ra_size : n;
83
84 r = b->pread(b->be, (int64_t) off, b->rbuf, fill, st);
85 if (r < 0) {
86 return -1;
87 }
88 b->rbuf_off = off;
89 b->rbuf_len = (size_t) r;
90 return 0;
91}

93/* ---- Copy the requested bytes out of the read-ahead buffer ----
94 *
95 * WHAT: Copies up to n bytes for offset off out of rbuf into out and returns the
96 * number of bytes copied (a short count only happens at EOF); returns 0 when off
97 * falls outside the buffered window.
98 *
99 * WHY: Keeps the memcpy bounds arithmetic in one pure helper so the orchestrator
100 * only has to advance the read-ahead cursor by the returned count.
101 *
102 * HOW:
103 * 1. If off is outside [rbuf_off, rbuf_off+rbuf_len), nothing is available.
104 * 2. Clamp the copy to whichever is smaller: the bytes left in the window or n.
105 * 3. memcpy from the in-window position and return the copied count.
106 */
107static size_t
108iobuf_serve(const brix_iobuf *b, off_t off, char *out, size_t n)
109{
110 size_t inbuf;
111 size_t avail;
112
113 if (!(off >= b->rbuf_off && (size_t) (off - b->rbuf_off) < b->rbuf_len)) {
114 return 0;
115 }
116 inbuf = b->rbuf_len - (size_t) (off - b->rbuf_off);
117 avail = (inbuf < n) ? inbuf : n;
118 memcpy(out, b->rbuf + (off - b->rbuf_off), avail);
119 return avail;
120}

122ssize_t
123brix_iobuf_read(brix_iobuf *b, off_t off, char *out, size_t n, brix_status *st)
124{
125 size_t avail;
126
127 /* Coherence: a read on a writable handle must see pending buffered writes. */
128 if (b->wbuf_len > 0 && brix_iobuf_flush(b, st) != 0) {
129 return -1;
130 }
131
132 /* Direct read for writable handles, when read-ahead is off, or for reads
133 * larger than the buffer (no benefit from caching those). */
134 if (b->writable || b->ra_size == 0 || n >= b->ra_size) {
135 return b->pread(b->be, (int64_t) off, out, n, st);
136 }
137
138 if (b->rbuf == NULL) {
139 b->rbuf = malloc(b->ra_size);
140 if (b->rbuf == NULL) { /* fall back to a direct read on OOM */
141 return b->pread(b->be, (int64_t) off, out, n, st);
142 }
143 }
144
145 /* Refill unless [off, off+n) is already buffered. */
146 if (!iobuf_range_cached(b, off, n) && iobuf_refill(b, off, n, st) != 0) {
147 return -1;
148 }
149
150 /* Serve from the buffer (a short count only happens at EOF). */
151 avail = iobuf_serve(b, off, out, n);
152 b->ra_next = off + (off_t) avail;
153 return (ssize_t) avail;
154}

156int
157brix_iobuf_write(brix_iobuf *b, off_t off, const char *in, size_t n,
158 brix_status *st)
159{
160 /* Write-back disabled, or a write at/above the buffer size: flush any pending
161 * buffer and write straight through. */
162 if (b->wb_size == 0 || n >= b->wb_size) {
163 if (brix_iobuf_flush(b, st) != 0) {
164 return -1;
165 }
166 return b->pwrite(b->be, (int64_t) off, in, n, st);
167 }
168
169 if (b->wbuf == NULL) {
170 b->wbuf = malloc(b->wb_size);
171 if (b->wbuf == NULL) { /* fall back to a direct write on OOM */
172 return b->pwrite(b->be, (int64_t) off, in, n, st);
173 }
174 }
175
176 /* A non-contiguous write, or one that won't fit, flushes the current buffer
177 * first; then this write starts (or extends) a fresh contiguous run. */
178 if (b->wbuf_len > 0
179 && (off != b->wbuf_off + (off_t) b->wbuf_len
180 || b->wbuf_len + n > b->wb_size)) {
181 if (brix_iobuf_flush(b, st) != 0) {
182 return -1;
183 }
184 }
185 if (b->wbuf_len == 0) {
186 b->wbuf_off = off;
187 }
188 memcpy(b->wbuf + b->wbuf_len, in, n);
189 b->wbuf_len += n;
190 return 0;
191}
