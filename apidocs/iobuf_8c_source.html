28brix_iobuf_flush(brix_iobuf *b, brix_status *st)
29{
30 if (b->wbuf == NULL || b->wbuf_len == 0) {
31 return 0;
32 }
33 if (b->pwrite(b->be, (int64_t) b->wbuf_off, b->wbuf, b->wbuf_len, st) != 0) {

40ssize_t
41brix_iobuf_read(brix_iobuf *b, off_t off, char *out, size_t n, brix_status *st)
42{
43 ssize_t r;
44 size_t avail;
45
46 /* Coherence: a read on a writable handle must see pending buffered writes. */
47 if (b->wbuf_len > 0 && brix_iobuf_flush(b, st) != 0) {
48 return -1;
49 }
50
51 /* Direct read for writable handles, when read-ahead is off, or for reads
52 * larger than the buffer (no benefit from caching those). */
53 if (b->writable || b->ra_size == 0 || n >= b->ra_size) {
54 return b->pread(b->be, (int64_t) off, out, n, st);
55 }
56
57 if (b->rbuf == NULL) {
58 b->rbuf = malloc(b->ra_size);
59 if (b->rbuf == NULL) { /* fall back to a direct read on OOM */
60 return b->pread(b->be, (int64_t) off, out, n, st);
61 }
62 }
63
64 /* Refill unless [off, off+n) is already buffered. Prefetch the whole window
65 * only on sequential access so random reads don't repeatedly over-fetch. */
66 if (!(b->rbuf_len > 0 && off >= b->rbuf_off
67 && (size_t) (off - b->rbuf_off) + n <= b->rbuf_len)) {
68 size_t fill = (off == b->ra_next) ? b->ra_size : n;
69
70 r = b->pread(b->be, (int64_t) off, b->rbuf, fill, st);
71 if (r < 0) {
72 return -1;
73 }
74 b->rbuf_off = off;
75 b->rbuf_len = (size_t) r;
76 }
77
78 /* Serve from the buffer (a short count only happens at EOF). */
79 avail = 0;
80 if (off >= b->rbuf_off && (size_t) (off - b->rbuf_off) < b->rbuf_len) {
81 size_t inbuf = b->rbuf_len - (size_t) (off - b->rbuf_off);
82 avail = (inbuf < n) ? inbuf : n;
83 memcpy(out, b->rbuf + (off - b->rbuf_off), avail);
84 }
85 b->ra_next = off + (off_t) avail;
86 return (ssize_t) avail;
87}

89int
90brix_iobuf_write(brix_iobuf *b, off_t off, const char *in, size_t n,
91 brix_status *st)
92{
93 /* Write-back disabled, or a write at/above the buffer size: flush any pending
94 * buffer and write straight through. */
95 if (b->wb_size == 0 || n >= b->wb_size) {
96 if (brix_iobuf_flush(b, st) != 0) {
97 return -1;
98 }
99 return b->pwrite(b->be, (int64_t) off, in, n, st);
100 }
101
102 if (b->wbuf == NULL) {
103 b->wbuf = malloc(b->wb_size);
104 if (b->wbuf == NULL) { /* fall back to a direct write on OOM */
105 return b->pwrite(b->be, (int64_t) off, in, n, st);
106 }
107 }
108
109 /* A non-contiguous write, or one that won't fit, flushes the current buffer
110 * first; then this write starts (or extends) a fresh contiguous run. */
111 if (b->wbuf_len > 0
112 && (off != b->wbuf_off + (off_t) b->wbuf_len
113 || b->wbuf_len + n > b->wb_size)) {
114 if (brix_iobuf_flush(b, st) != 0) {
115 return -1;
116 }
117 }
118 if (b->wbuf_len == 0) {
119 b->wbuf_off = off;
120 }
121 memcpy(b->wbuf + b->wbuf_len, in, n);
122 b->wbuf_len += n;
123 return 0;
124}
