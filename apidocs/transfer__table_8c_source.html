6/*
7 * dashboard/transfer_table.c — live transfer slot operations and SHM init.
8 *
9 * WHAT: Owns the static ngx_shmtx_t that guards slot allocation, implements
10 * the SHM zone init callback (called by nginx when the zone is first
11 * mapped or re-attached on reload), and provides the four public API
12 * functions that stream workers call around file open/IO/close.
13 *
14 * CONCURRENCY MODEL:
15 * brix_transfer_slot_alloc_ex() — acquires mutex (brief, O(512) scan)
16 * brix_transfer_slot_update_bytes() — lock-free atomics only
17 * brix_transfer_slot_free() — lock-free atomic CAS
18 * brix_transfer_slot_free_all_for_session() — acquires mutex (disconnect)
19 * JSON exporter (api.c) — lock-free scan; tolerates tears
20 *
21 * STALE SLOT GC:
22 * The JSON exporter calls slot_free() on any slot whose last_ms is more than
23 * 60 seconds old. This catches slots that somehow escaped the close/disconnect
24 * cleanup path (e.g., due to a future bug or worker crash).
25 */
26
27static ngx_shmtx_t brix_dashboard_mutex;

29/* SHM init callback */
30ngx_int_t
31ngx_brix_dashboard_shm_init(ngx_shm_zone_t *shm_zone, void *data)
32{
33 ngx_flag_t fresh;
34 brix_transfer_table_t *tbl;
35
36 /*
37 * Allocate the table FROM the slab pool (never over shm.addr) so nginx's
38 * slab-pool header — which ngx_unlock_mutexes() dereferences on every child
39 * death — survives. The helper zeroes a fresh table and (re-)creates the
40 * mutex from the leading ngx_shmtx_sh_t lock on fresh, reload, and re-attach.
41 * On reuse the live slots are preserved; transfers in flight survive reload.
42 */
43 tbl = brix_shm_table_alloc(shm_zone, data,
44 sizeof(brix_transfer_table_t),
45 &brix_dashboard_mutex, &fresh);
46 if (tbl == NULL) {
47 return NGX_ERROR;
48 }
49
50 /* No fresh-only field inits: the helper's memzero leaves next_serial = 0
51 * and every slot free, which is the entire first-startup state. */
52 (void) fresh;
53
54 return NGX_OK;
55}

73int
74brix_transfer_slot_alloc_ex(brix_transfer_table_t *t,
75 const u_char sessid[16], const char *client_ip,
76 const char *identity, const char *vo, const char *path, const char *op,
77 uint8_t direction, uint8_t proto, int64_t expected_bytes, int64_t now_ms)
78{
79 int free_idx = -1;
80 brix_transfer_slot_t *slot;
81 int i;
82
83 ngx_shmtx_lock(&brix_dashboard_mutex);
84
85 for (i = 0; i < BRIX_DASHBOARD_MAX_TRANSFERS; i++) {
86 if (t->slots[i].in_use == 0) {
87 free_idx = i;
88 break;
89 }
90 }
91
92 if (free_idx < 0) {
93 ngx_shmtx_unlock(&brix_dashboard_mutex);
94 brix_dashboard_event_add(BRIX_DASH_EVENT_DASHBOARD, proto, 0,
95 "active transfer table full", path);
96 return -1; /* table full — transfer proceeds untracked */
97 }
98
99 slot = &t->slots[free_idx];
100 ngx_memzero(slot, sizeof(*slot));
101
102 if (sessid != NULL) {
103 ngx_memcpy(slot->sessid, sessid, 16);
104 }
105
106 dashboard_copy_field(slot->client_ip, sizeof(slot->client_ip), client_ip);
107 dashboard_copy_field(slot->identity, sizeof(slot->identity),
108 identity ? identity : "anonymous");
109 dashboard_copy_field(slot->vo, sizeof(slot->vo), vo);
110 dashboard_copy_field(slot->path, sizeof(slot->path), path);
111 dashboard_copy_field(slot->op, sizeof(slot->op), op ? op : "transfer");
112
113 slot->worker_pid = ngx_pid;
114 slot->direction = direction;
115 slot->proto = proto;
116 slot->state = BRIX_XFER_STATE_ACTIVE;
117 slot->expected_bytes = expected_bytes;
118 slot->start_ms = now_ms;
119 slot->last_ms = (ngx_atomic_t) now_ms;
120 slot->state_since_ms = (ngx_atomic_t) now_ms;
121 slot->serial = ++t->next_serial;
122 slot->in_use = 1; /* publish last — readers see a complete slot */
123
124 ngx_shmtx_unlock(&brix_dashboard_mutex);
125 return free_idx;
126}

140void
141brix_transfer_slot_update_bytes(brix_transfer_table_t *t,
142 int slot_idx, ngx_atomic_int_t nbytes, int64_t now_ms)
143{
144 brix_transfer_slot_t *slot;
145
146 if (slot_idx < 0 || slot_idx >= BRIX_DASHBOARD_MAX_TRANSFERS) {
147 return;
148 }
149
150 slot = &t->slots[slot_idx];
151
152 if (slot->in_use == 0) {
153 return; /* slot was freed between the check and this update */
154 }
155
156 if (nbytes > 0) {
157 ngx_atomic_fetch_add(&slot->bytes, nbytes);
158 }
159 slot->state = BRIX_XFER_STATE_ACTIVE;
160 slot->last_ms = (ngx_atomic_t) now_ms; /* 64-bit aligned write; atomic on x86_64 */
161
162 /*
163 * EWMA-smoothed instantaneous rate. Only the owning worker writes these
164 * three fields (the exporter is read-only), so a plain read-modify-write is
165 * race-free under the existing concurrency model. We fold a new sample at
166 * most once per BRIX_XFER_SAMPLE_MS: rate over the elapsed window, blended
167 * into the previous value with alpha = 1/4 (new = raw/4 + prev*3/4). This
168 * turns a bursty client (e.g. xrdcp --xrate: idle, then an 8 MiB burst) into
169 * a steady published rate instead of a 0↔line-rate sawtooth. Decay toward
170 * zero during the idle gap is applied read-only by the exporter (api.c).
171 */
172 {
173 int64_t sample_ms = (int64_t) slot->last_sample_ms;
174
175 if (sample_ms == 0) {
176 /* First update — seed the window without emitting a rate. */
177 slot->last_sample_ms = (ngx_atomic_t) now_ms;
178 slot->bytes_last_sample = slot->bytes;
179 } else {
180 int64_t dt = now_ms - sample_ms;
181
182 if (dt >= BRIX_XFER_SAMPLE_MS) {
183 int64_t db = (int64_t) slot->bytes
184 - (int64_t) slot->bytes_last_sample;
185 uint64_t raw = (db > 0) ? (uint64_t) (db * 1000 / dt) : 0;
186 uint64_t prev = (uint64_t) slot->instant_bps;
187
188 slot->instant_bps = (ngx_atomic_t) ((raw + prev * 3) / 4);
189 slot->bytes_last_sample = slot->bytes;
190 slot->last_sample_ms = (ngx_atomic_t) now_ms;
191 }
192 }
193 }
194}

276

278 return;
279 }
280
281 slot = &t->slots[slot_idx];
282 if (slot->in_use == 0 || op == NULL) {
283 return;

285
286 if (ngx_strcasecmp((u_char *) op, (u_char *) "read") == 0

291 } else if (ngx_strcasecmp((u_char *) op, (u_char *) "write") == 0
292 || ngx_strcasecmp((u_char *) op, (u_char *) "PUT") == 0
293 || ngx_strcasecmp((u_char *) op, (u_char *) "PutObject") == 0
294 || ngx_strcasecmp((u_char *) op, (u_char *) "UploadPart") == 0)
295 {
296 ngx_atomic_fetch_add(&slot->write_ops, 1);
297 } else if (ngx_strcasecmp((u_char *) op, (u_char *) "sync") == 0
298 || ngx_strcasecmp((u_char *) op, (u_char *) "commit") == 0)
299 {
300 ngx_atomic_fetch_add(&slot->sync_ops, 1);
301 } else if (ngx_strcasecmp((u_char *) op, (u_char *) "close") == 0) {
302 ngx_atomic_fetch_add(&slot->close_ops, 1);
303 }
