3/*
4 * credential.c — parse and validate TPC delegated credentials.
5 *
6 * WHAT: Implements the brix_tpc_credential.h interface: sniff a raw
7 * credential string into a typed brix_tpc_credential_t
8 * (brix_tpc_credential_parse), reject empty/expired credentials
9 * (brix_tpc_credential_validate), and name a credential type for logs
10 * (brix_tpc_credential_type_name).
11 *
12 * WHY: The shared TPC core accepts a credential from either transport in a free
13 * form (a bearer token, optionally "Bearer "-prefixed, or a GSI proxy PEM)
14 * and must classify it without baking transport-specific assumptions into
15 * the core. Centralising the sniffing here keeps the format heuristics and
16 * the expiry check in one auditable place.
17 *
18 * HOW: parse() trims surrounding whitespace/CRLF, treats empty or "none" as
19 * BRIX_TPC_CREDENTIAL_NONE, then routes on the caller hint or on the
20 * "Bearer " / "-----BEGIN" markers, stripping the "Bearer " prefix for
21 * tokens. The static brix_tpc_copy_credential_str() duplicates the value
22 * into the pool (NUL-terminated) when a pool is supplied, or aliases the
23 * caller's bytes when pool is NULL. validate() enforces non-empty bodies
24 * per type and compares expires_at against ngx_time().
25 */

108/*
109 * Sniff raw_credential into a typed *cred. Honours an optional type hint and
110 * the "Bearer "/"-----BEGIN" markers; trims surrounding whitespace and CRLF.
111 * Empty or "none" yields BRIX_TPC_CREDENTIAL_NONE. When pool is non-NULL the
112 * value is copied (NUL-terminated) into it, otherwise *cred aliases the input.
113 * Returns NGX_OK, NGX_DECLINED for an unrecognised format, or NGX_ERROR on a
114 * NULL cred / allocation failure.
115 */
116ngx_int_t
117brix_tpc_credential_parse(const ngx_str_t *raw_credential,
118 brix_tpc_credential_type_t hint, brix_tpc_credential_t *cred,
119 ngx_pool_t *pool, ngx_log_t *log)
120{
121 const u_char *data;
122 size_t len;
123
124 if (cred == NULL) {
125 return NGX_ERROR;
126 }
127
128 ngx_memzero(cred, sizeof(*cred));
129 cred->type = BRIX_TPC_CREDENTIAL_NONE;
130
131 if (raw_credential == NULL || raw_credential->data == NULL
132 || raw_credential->len == 0)
133 {
134 return NGX_OK;
135 }
136
137 data = raw_credential->data;
138 len = raw_credential->len;
139
140 while (len > 0 && (*data == ' ' || *data == '\t')) {
141 data++;
142 len--;
143 }
144 while (len > 0 && (data[len - 1] == ' ' || data[len - 1] == '\t'
145 || data[len - 1] == '\n' || data[len - 1] == '\r')) {
146 len--;
147 }
148
149 if (len == 0 || brix_tpc_str_equals(data, len, "none")) {
150 cred->type = BRIX_TPC_CREDENTIAL_NONE;
151 return NGX_OK;
152 }
153
154 if (hint == BRIX_TPC_CREDENTIAL_TOKEN
155 || brix_tpc_starts_with(data, len, "Bearer "))
156 {
157 if (brix_tpc_starts_with(data, len, "Bearer ")) {
158 data += sizeof("Bearer ") - 1;
159 len -= sizeof("Bearer ") - 1;
160 }
161
162 cred->type = BRIX_TPC_CREDENTIAL_TOKEN;
163 if (brix_tpc_copy_credential_str(&cred->bearer, data, len, pool)
164 != NGX_OK)
165 {
166 return NGX_ERROR;
167 }
168 return NGX_OK;
169 }
170
171 if (hint == BRIX_TPC_CREDENTIAL_PROXY
172 || brix_tpc_starts_with(data, len, "-----BEGIN"))
173 {
174 cred->type = BRIX_TPC_CREDENTIAL_PROXY;
175 if (brix_tpc_copy_credential_str(&cred->proxy_pem, data, len, pool)
176 != NGX_OK)
177 {
178 return NGX_ERROR;
179 }
180 return NGX_OK;
181 }
182
183 ngx_log_error(NGX_LOG_WARN, log, 0,
184 "brix_tpc: unsupported credential format");
185 return NGX_DECLINED;
186}

188/*
189 * Validate a parsed credential: PROXY/TOKEN must carry a non-empty body, and a
190 * non-zero expires_at must be in the future relative to ngx_time(). NONE always
191 * passes. Returns NGX_OK, NGX_DECLINED if empty/expired/invalid-type, or
192 * NGX_ERROR on a NULL cred.
193 */
194ngx_int_t
195brix_tpc_credential_validate(const brix_tpc_credential_t *cred,
196 ngx_log_t *log)
197{
198 time_t now;
199
200 if (cred == NULL) {
201 return NGX_ERROR;
202 }
203
204 switch (cred->type) {
205 case BRIX_TPC_CREDENTIAL_NONE:
206 return NGX_OK;
207
208 case BRIX_TPC_CREDENTIAL_PROXY:
209 if (cred->proxy_pem.data == NULL || cred->proxy_pem.len == 0) {
210 ngx_log_error(NGX_LOG_WARN, log, 0,
211 "brix_tpc: empty proxy credential");
212 return NGX_DECLINED;
213 }
214 break;
215
216 case BRIX_TPC_CREDENTIAL_TOKEN:
217 if (cred->bearer.data == NULL || cred->bearer.len == 0) {
218 ngx_log_error(NGX_LOG_WARN, log, 0,
219 "brix_tpc: empty bearer credential");
220 return NGX_DECLINED;
221 }
222 break;
223
224 default:
225 ngx_log_error(NGX_LOG_WARN, log, 0,
226 "brix_tpc: invalid credential type %d",
227 (int) cred->type);
228 return NGX_DECLINED;
229 }
230
231 if (cred->expires_at != 0) {
232 now = ngx_time();
233 if (cred->expires_at <= now) {
234 ngx_log_error(NGX_LOG_WARN, log, 0,
235 "brix_tpc: %s credential expired",
236 brix_tpc_credential_type_name(cred->type));
237 return NGX_DECLINED;
238 }
239 }
240
241 return NGX_OK;
242}
