1/*
2 * client.c — the unprivileged worker side of the impersonation broker (phase 40).
3 *
4 * WHAT: Worker-resident client for the privileged root broker. Holds one
5 * persistent AF_UNIX connection per worker, a per-request "current principal",
6 * and the op wrappers (open/stat/mkdir/unlink/rmdir/rename/link/truncate/chmod/
7 * chown) that the confined beneath helpers call when `map` mode is active. Each
8 * wrapper performs one synchronous request/reply round-trip; OPEN receives the
9 * resulting fd back over SCM_RIGHTS.
10 *
11 * WHY: The worker is fully unprivileged — it cannot setfsuid or hold any
12 * capability. Delegating the open/metadata syscalls to the broker is what lets
13 * files be owned by, and DAC be enforced for, the real mapped user while the
14 * worker stays a plain svc-uid process. The round-trip is synchronous because
15 * these helpers were already blocking syscalls in the worker's view; the data
16 * plane (read/write on the returned fd) never touches the broker.
17 *
18 * HOW: A tiny state singleton (one connection fd + the remembered socket path +
19 * the current principal). write_full sends a fixed imp_req_t; recv_reply reads
20 * the fixed imp_rep_t and, for OPEN, the SCM_RIGHTS fd. A broken connection
21 * (broker respawn) is reconnected once and the op retried; a second failure
22 * fails closed (errno=EIO). No goto; pure helpers, side effects at the edges.
23 */

44/* Open a fresh connected socket to imp_sock_path. Returns fd or -1 (errno). */
45static int
46imp_dial(void)
47{
48 struct sockaddr_un addr;
49 int fd;
50 size_t plen = ngx_strlen(imp_sock_path);
51
52 if (plen == 0 || plen >= sizeof(addr.sun_path)) {
53 errno = EINVAL;
54 return -1;
55 }
56 fd = socket(AF_UNIX, SOCK_STREAM | SOCK_CLOEXEC, 0);
57 if (fd < 0) {
58 return -1;
59 }
60 ngx_memzero(&addr, sizeof(addr));
61 addr.sun_family = AF_UNIX;
62 ngx_memcpy(addr.sun_path, imp_sock_path, plen + 1);
63 if (connect(fd, (struct sockaddr *) &addr, sizeof(addr)) != 0) {
64 int e = errno;
65 close(fd);
66 errno = e;
67 return -1;
68 }
69 return fd;
70}

93ngx_int_t
94brix_imp_client_connect(const char *path, ngx_log_t *log)
95{
96 size_t plen;
97
98 if (path == NULL) {
99 return NGX_ERROR;
100 }
101 plen = ngx_strlen(path);
102 if (plen == 0 || plen >= sizeof(imp_sock_path)) {
103 return NGX_ERROR;
104 }
105 ngx_memcpy(imp_sock_path, path, plen + 1);
106 imp_log = log;
107 imp_enabled = 1;
108
109 imp_drop_conn();
110 if (imp_ensure_conn() < 0) {
111 if (log) ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
112 "impersonate client: connect to broker \"%s\" failed",
113 path);
114 return NGX_ERROR;
115 }
116 return NGX_OK;
117}

119int
120brix_imp_client_active(void)
121{
122 /*
123 * Route the confined FS op to the broker when map mode is active AND we are
124 * either (a) inside a request that resolved to a real principal, or (b)
125 * inside a request whose authenticated identity yielded NO mappable principal
126 * (empty subject / no DN) — case (b) still goes to the broker, which denies
127 * the empty principal (idmap can't map "") so the op FAILS CLOSED instead of
128 * silently falling back to the unprivileged WORKER (which owns the export and
129 * can read svc-only files — a real fail-open). Outside any request bracket
130 * (housekeeping: checkpoint recovery, FRM) there is no principal AND no active
131 * request, so we run locally as the worker, as intended.
132 */
133 return imp_enabled && (imp_principal[0] != '\0' || imp_in_request);
134}

172/* Write exactly n bytes. Returns 0 ok, -1 error. */
173static int
174imp_write_full(int fd, const void *buf, size_t n)
175{
176 const u_char *p = buf;
177 size_t sent = 0;
178
179 while (sent < n) {
180 ssize_t w = write(fd, p + sent, n - sent);
181 if (w < 0) {
182 if (errno == EINTR) { continue; }
183 return -1;
184 }
185 sent += (size_t) w;
186 }
187 return 0;
188}

190/*
191 * Receive one imp_rep_t and, when present, the SCM_RIGHTS fd (IMP_REP_HAS_FD) and
192 * the trailing data payload (IMP_REP_HAS_DATA, READLINK target) into data_buf.
193 * Returns 0 on success (*out_fd is the received fd or -1), -1 on IO error.
194 */
195static int
196imp_recv_reply(int fd, imp_rep_t *rep, int *out_fd, char *data_buf,
197 size_t data_bufsz)
198{
199 struct msghdr msg;
200 struct iovec iov[2];
201 union {
202 struct cmsghdr align;
203 char buf[CMSG_SPACE(sizeof(int))];
204 } cmsgbuf;
205 struct cmsghdr *c;
206 ssize_t n;
207
208 *out_fd = -1;
209
210 ngx_memzero(&msg, sizeof(msg));
211 ngx_memzero(&cmsgbuf, sizeof(cmsgbuf));
212 iov[0].iov_base = rep;
213 iov[0].iov_len = sizeof(*rep);
214 msg.msg_iov = iov;
215 msg.msg_iovlen = 1;
216 /* A second iovec receives any trailing payload (READLINK target / GETXATTR
217 * value / LISTXATTR names) IN THE SAME message. data_buf must be large enough
218 * for the op's reply payload: a READLINK target is bounded by PATH_MAX, while
219 * GET/LISTXATTR replies are bounded by IMP_XATTR_MAX (imp_get_or_list sizes its
220 * buffer accordingly). recvmsg never writes past iov_len, and the data_len
221 * cross-check below fails closed on any mismatch, so an over-long reply is
222 * rejected rather than overflowing or desyncing. */
223 if (data_buf != NULL && data_bufsz > 0) {
224 iov[1].iov_base = data_buf;
225 iov[1].iov_len = data_bufsz;
226 msg.msg_iovlen = 2;
227 }
228 msg.msg_control = cmsgbuf.buf;
229 msg.msg_controllen = sizeof(cmsgbuf.buf);
230
231 do {
232 n = recvmsg(fd, &msg, MSG_CMSG_CLOEXEC);
233 } while (n < 0 && errno == EINTR);
234
235 if (n < (ssize_t) sizeof(*rep)) {
236 return -1; /* short read / EOF / error */
237 }
238 /* A reply that declares trailing data we gave nowhere to receive (no iov[1])
239 * cannot be consumed without desyncing — reject explicitly and fail closed. */
240 if (rep->data_len > 0 && (data_buf == NULL || data_bufsz == 0)) {
241 return -1;
242 }
243 /* The frame declares its trailing length; if what we received does not match,
244 * the stream is desynced — fail so the caller drops + reconnects. */
245 if ((size_t) (n - (ssize_t) sizeof(*rep)) != (size_t) rep->data_len) {
246 return -1;
247 }
248
249 for (c = CMSG_FIRSTHDR(&msg); c != NULL; c = CMSG_NXTHDR(&msg, c)) {
250 if (c->cmsg_level == SOL_SOCKET && c->cmsg_type == SCM_RIGHTS
251 && c->cmsg_len == CMSG_LEN(sizeof(int)))
252 {
253 ngx_memcpy(out_fd, CMSG_DATA(c), sizeof(int));
254 }
255 }
256 return 0;
257}

281/*
282 * One request/reply exchange with transparent single reconnect. On success
283 * returns 0 and fills *rep / *out_fd; on a transport failure returns -1
284 * (errno=EIO). The op is retried exactly once across a dropped connection
285 * (broker respawn), never more — a persistently dead broker fails closed.
286 *
287 * `data_in` (length `data_in_len`) is an optional inbound payload written
288 * immediately after the fixed frame (SETXATTR value); the broker reads exactly
289 * req->req_data_len trailing bytes. `data_buf`/`data_bufsz` receive an optional
290 * outbound reply payload (READLINK/GETXATTR/LISTXATTR). A retry re-sends the
291 * whole [frame][data_in] on a fresh connection, so a half-written payload on a
292 * dying connection never desyncs the broker.
293 */
294int
295imp_exchange(const imp_req_t *req, imp_rep_t *rep, int *out_fd,
296 char *data_buf, size_t data_bufsz,
297 const void *data_in, size_t data_in_len)
298{
299 int attempt;
300
301 for (attempt = 0; attempt < 2; attempt++) {
302 int fd = imp_ensure_conn();
303 if (fd < 0) {
304 continue; /* could not dial; retry once */
305 }
306 if (imp_write_full(fd, req, sizeof(*req)) != 0
307 || (data_in_len > 0
308 && imp_write_full(fd, data_in, data_in_len) != 0)
309 || imp_recv_reply(fd, rep, out_fd, data_buf, data_bufsz) != 0)
310 {
311 imp_drop_conn(); /* transport broke — reconnect & retry */
312 continue;
313 }
314 return 0;
315 }
316
317 if (imp_log) {
318 ngx_log_error(NGX_LOG_ERR, imp_log, 0,
319 "impersonate client: broker unreachable (op=%ud)", req->op);
320 }
321 errno = EIO;
322 return -1;
323}
