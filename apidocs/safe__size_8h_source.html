1/*
2 * safe_size.h — Phase 27 W1: overflow-checked size arithmetic and array
3 * allocation.
4 *
5 * WHAT: Header-only helpers that compute `a*b` / `a+b` with overflow detection
6 * and allocate `n*sz`-byte arrays that return NULL (rather than a truncated,
7 * heap-overflowing buffer) when the size computation would wrap.
8 *
9 * WHY: Anything whose size or count comes off the wire is attacker-controlled.
10 * A `malloc(n * sizeof(*p))` where `n` derives from a client-supplied length or
11 * segment count can silently wrap on 32-bit `size_t` math (or even 64-bit with
12 * a large enough `n`), producing a tiny allocation that the caller then writes
13 * `n` elements into — a classic heap overflow. These helpers make the overflow
14 * a clean NULL/error instead.
15 *
16 * HOW: Built on the compiler's `__builtin_*_overflow` intrinsics (GCC/Clang;
17 * the module is already built with one of these). All functions are
18 * `static ngx_inline` so there is no new translation unit and no ./configure
19 * change — include the header and call them.
20 *
21 * Adoption: convert every wire-driven `n * sizeof(...)` and `len + 1` size
22 * computation to these helpers (start with the readv segment array and the
23 * eviction-candidate realloc growth).
24 */
25#ifndef BRIX_SHARED_SAFE_SIZE_H
26#define BRIX_SHARED_SAFE_SIZE_H

36/* Multiply: *out = a*b. Returns NGX_OK on success, NGX_ERROR on overflow
37 * (in which case *out is left unspecified — callers must not use it). */
38static ngx_inline ngx_int_t
39brix_size_mul(size_t a, size_t b, size_t *out)
40 __attribute__((warn_unused_result));

42static ngx_inline ngx_int_t
43brix_size_mul(size_t a, size_t b, size_t *out)
44{
45#if defined(__GNUC__) || defined(__clang__)
46 if (__builtin_mul_overflow(a, b, out)) {
47 return NGX_ERROR;
48 }
49 return NGX_OK;
50#else
51 if (a != 0 && b > (size_t) -1 / a) {
52 return NGX_ERROR;
53 }
54 *out = a * b;
55 return NGX_OK;
56#endif
57}

90static ngx_inline void *
91brix_palloc_array(ngx_pool_t *pool, size_t n, size_t sz)
92{
93 size_t total;
94
95 if (brix_size_mul(n, sz, &total) != NGX_OK || total == 0) {
96 return NULL;
97 }
98 return ngx_palloc(pool, total);
