16/*
17 * brix_build_single_sendfile_chain — build a header+file chain for a single
18 * chunk sendfile response (data_total <= BRIX_READ_CHUNK_MAX).
19 *
20 * WHAT: Constructs an nginx ngx_chain_t with two links where the second link is
21 * a file-backed buffer (in_file=1) pointing to an open fd at offset..offset+data.
22 * The header link contains ServerResponseHdr; the data link enables direct kernel
23 * sendfile(2) — bypassing read()+write() and reducing CPU utilization on large reads.
24 *
25 * WHY: Non-TLS (cleartext) connections can leverage kernel-level zero-copy for superior
26 * throughput. TLS connections automatically fall back to memory-backed chains because
27 * nginx's SSL layer cannot wrap sendfile(2). This function handles the single-chunk
28 * case; multi-chunk responses are built by brix_build_sendfile_chain via iteration.
29 * Setting in_file=1 is critical: it tells nginx's output_filter to invoke sendfile()
30 * syscall directly rather than reading into user space and writing back out.
31 *
32 * HOW: 1) Acquires rd.read_hdr_scratch for the header buffer (same as memory chain).
33 * 2) Calls brix_build_resp_hdr() with ctx->recv.cur_streamid, kXR_ok, data_total.
34 * 3) Memzeros fast structs including read_fast_file (ngx_file_t).
35 * 4) Configures hdr_buf → hdr_chain (header memory-backed).
36 * 5) If data_total==0 returns header-only; otherwise configures:
37 * - file.fd = fd, file.name = path
38 * - body_buf.file = &file, body_buf.in_file = 1
39 * - body_buf.file_pos = offset, body_buf.file_last = offset + data
40 * 6) Links hdr_chain → body_chain; sets last_buf/last_in_chain on final buffer.
41 * 7) If base_out is non-NULL, stores hdrbuf pointer for deferred free after send.
42 */

106/*
107 * brix_build_sendfile_chain — build a multi-chunk sendfile chain for reads
108 * from a kernel file descriptor (zero-copy path for non-TLS connections).
109 *
110 * WHAT: Constructs an nginx ngx_chain_t where each wire frame consists of:
111 * 1. A header buffer (memory-backed, contains ServerResponseHdr)
112 * 2. A file buffer (in_file=1, points to fd at offset+di .. offset+di+chunk)
113 *
114 * This enables nginx's send_chain to invoke sendfile(2) syscalls directly —
115 * bypassing read()+write() round-trips and reducing CPU utilization on large
116 * transfers. Zero-copy semantics are preserved: data never enters user space.
117 *
118 * WHY: Non-TLS connections can use the kernel-level zero-copy path for superior
119 * throughput. TLS connections automatically fall back to read+write because
120 * nginx's SSL layer cannot wrap sendfile — no explicit handling needed here.
121 * Multi-chunk support handles reads exceeding BRIX_READ_CHUNK_MAX (16 MiB).
122 *
123 * HOW: Delegates to brix_build_single_sendfile_chain for small responses.
124 * For large responses, iterates over n_chunks allocating one ngx_file_t per
125 * data link (all referencing the same fd at successive offsets). hdrbuf is
126 * packed contiguously like chunked memory chain; base_out tracks allocation
127 * for deferred free after send completion.
128 */
129ngx_chain_t *
130brix_build_sendfile_chain(brix_ctx_t *ctx, ngx_connection_t *c,
131 int fd, const char *path, off_t offset, size_t data_total,
132 u_char **base_out)
133{
134 size_t n_chunks, last_size;
135 u_char *hdrbuf;
136 ngx_chain_t *head = NULL, *tail = NULL;
137 size_t di = 0;
138 size_t chunk;
139
140 if (base_out != NULL) {
141 *base_out = NULL;
142 }
143
144 if (data_total <= BRIX_READ_CHUNK_MAX) {
145 return brix_build_single_sendfile_chain(ctx, c, fd, path, offset,
146 data_total, base_out);
147 }
148
149 /* Same chunk-count math as brix_build_chunked_chain, via the shared
150 * brix_chunk_geometry helper: ceil-divide, then remap a zero final remainder
151 * back to a full CHUNK_MAX (the last frame must carry the trailing bytes,
152 * never zero). */
153 brix_chunk_geometry(data_total, &n_chunks, &last_size);
154
155 /*
156 * Phase 32 WS2: keep the per-chunk headers in THIS slot's private header
157 * buffer (slot->hdr_bytes), not the shared rd.read_hdr_scratch, so a multi-chunk
158 * sendfile read can pipeline — the next read built into another slot cannot
159 * clobber these headers while this response is still draining. rlen is
160 * capped at BRIX_READ_REQUEST_MAX so n_chunks*8 <= BRIX_SLOT_HDR_MAX
161 * always; the guard is defence-in-depth. The data links are file-backed
162 * (no heap) and the headers are slot-owned, so there is nothing to free —
163 * base_out stays NULL.
164 */
165 if (n_chunks * XRD_RESPONSE_HDR_LEN > BRIX_SLOT_HDR_MAX) {
166 return NULL;
167 }
168 hdrbuf = ctx->out.ring[ctx->out.tail].hdr_bytes;
169
170 ctx->out.resp_pipelinable = 1;
171
172 for (chunk = 0; chunk < n_chunks; chunk++) {
173 size_t chunk_data;
174 uint16_t status;
175 u_char *hptr;
176
177 /* Same frame layout as the memory path: full CHUNK_MAX per chunk
178 * except the last (last_size); kXR_oksofar on every frame but the
179 * final kXR_ok, all sharing ctx->recv.cur_streamid. */
180 chunk_data = (chunk < n_chunks - 1) ? BRIX_READ_CHUNK_MAX
181 : last_size;
182 status = (chunk == n_chunks - 1) ? kXR_ok : kXR_oksofar;
183 hptr = hdrbuf + chunk * XRD_RESPONSE_HDR_LEN;
184
185 brix_build_resp_hdr(ctx->recv.cur_streamid, status,
186 (uint32_t) chunk_data,
187 (ServerResponseHdr *) hptr);
188
189 if (brix_chain_append_mem(c->pool, &head, &tail, hptr,
190 hptr + XRD_RESPONSE_HDR_LEN) != NGX_OK)
191 {
192 return NULL;
193 }
194
195 if (chunk_data > 0) {
196 /*
197 * in_file=1 (set inside brix_chain_append_file) is what makes nginx
198 * emit sendfile(2) for this link instead of read()+write(). Every
199 * data link shares the SAME fd; di walks the byte offset so chunk k
200 * covers fd bytes [offset+di .. offset+di+chunk_data) — successive
201 * windows of one open file, no per-chunk seek or copy.
202 */
203 off_t file_pos = offset + (off_t) di;
204
205 if (brix_chain_append_file(c->pool, c, &head, &tail, fd, path,
206 file_pos,
207 file_pos + (off_t) chunk_data) != NGX_OK)
208 {
209 return NULL;
210 }
211 di += chunk_data;
212 }
213 }
214
215 if (tail != NULL) {
216 tail->buf->last_buf = 1;
217 tail->buf->last_in_chain = 1;
218 }
219
220 return head;
221}

223/*
224 * brix_build_pgread_chain — assemble the kXR_pgread response chain:
225 * [ServerStatusResponse_pgRead header] -> [encoded page data].
226 *
227 * `data` points at out_size bytes of already-CRC-encoded page-mode wire output
228 * (the gapped [CRC32c][page]+ layout produced by the in-place encoder), and
229 * carries its own per-page checksums — so it is sent verbatim, never through
230 * brix_build_chunked_chain (which would prepend a kXR_ok framing wrong for
231 * pgread). Both the synchronous handler (src/read/pgread.c) and the thread-pool
232 * AIO completion (brix_pgread_aio_done) build exactly this chain, so it lives
233 * here once. Returns the chain head, or NULL on a pool allocation failure — the
234 * caller owns its scratch buffer and handles its own error/cleanup path.
235 */
236ngx_chain_t *
237brix_build_pgread_chain(brix_ctx_t *ctx, ngx_connection_t *c,
238 int64_t offset, u_char *data, uint32_t out_size)
239{
240 ServerStatusResponse_pgRead *hdr_buf;
241 ngx_chain_t *cl_hdr, *cl_data;
242 ngx_buf_t *bd;
243
244 hdr_buf = ngx_palloc(c->pool, sizeof(*hdr_buf));
245 if (hdr_buf == NULL) {
246 return NULL;
247 }
248 brix_build_pgread_status(ctx, offset, out_size, hdr_buf);
249
250 cl_hdr = ngx_alloc_chain_link(c->pool);
251 if (cl_hdr == NULL) {
252 return NULL;
253 }
254 cl_hdr->buf = ngx_calloc_buf(c->pool);
255 if (cl_hdr->buf == NULL) {
256 return NULL;
257 }
258 cl_hdr->buf->pos = (u_char *) hdr_buf;
259 cl_hdr->buf->last = cl_hdr->buf->pos + sizeof(*hdr_buf);
260 cl_hdr->buf->memory = 1;
261 cl_hdr->buf->last_buf = 0;
262
263 cl_data = ngx_alloc_chain_link(c->pool);
264 bd = ngx_calloc_buf(c->pool);
265 if (cl_data == NULL || bd == NULL) {
266 return NULL;
267 }
268 bd->pos = data;
269 bd->last = data + out_size;
270 bd->memory = 1;
271 bd->last_buf = 1;
272 cl_data->buf = bd;
273 cl_data->next = NULL;
274
275 cl_hdr->next = cl_data;
276 return cl_hdr;
277}
