53{

65/* Map the POSIX open(2) flags the kXR_open path computed back to the backend-
66 * neutral BRIX_SD_O_* intent the storage driver understands. The driver
67 * re-derives its own native flags from these (the POSIX driver re-expands to
68 * O_*), so a non-POSIX backend never sees Linux-specific bits. */
69static int
70brix_open_oflags_to_sd(int oflags, int is_readable, int is_write)
71{
72 int sd = 0;
73
74 if (is_readable) { sd |= BRIX_SD_O_READ; }
75 if (is_write) { sd |= BRIX_SD_O_WRITE; }
76 if (oflags & O_CREAT) { sd |= BRIX_SD_O_CREATE; }
77 if (oflags & O_EXCL) { sd |= BRIX_SD_O_EXCL; }
78 if (oflags & O_TRUNC) { sd |= BRIX_SD_O_TRUNC; }
79 if (oflags & O_APPEND) { sd |= BRIX_SD_O_APPEND; }
80 return sd;
81}

83/* Driver-backed kXR_open (Layer 3): open `logical` through the export's storage
84 * driver into the handle's sd_obj, then synthesize a struct stat from the
85 * driver's captured open snapshot so the rest of the open path (bookkeeping,
86 * size reporting) is backend-agnostic. The handle's bare `fd` becomes the
87 * driver's representative descriptor (a block-0 fd for CAP_FD backends, or
88 * NGX_INVALID_FILE for a pure object store) and all subsequent byte I/O routes
89 * through fh->sd_obj.driver. Writes *out_fd and *st on success and returns
90 * NGX_OK; on failure sets errno and returns NGX_ERROR (the caller maps errno to
91 * the kXR error exactly as for a POSIX open). */
92static ngx_int_t
93brix_open_resolved_via_driver(brix_sd_instance_t *sd, const char *logical,
94 int oflags, int is_readable, int is_write, mode_t create_mode,
95 brix_file_t *fh, int *out_fd, struct stat *st)
96{
97 int sd_flags = brix_open_oflags_to_sd(oflags, is_readable,
98 is_write);
99 int oerr = 0;
100 brix_sd_obj_t *obj;
101
102 obj = sd->driver->open(sd, logical, sd_flags, create_mode, &oerr);
103 if (obj == NULL) {
104 errno = (oerr != 0) ? oerr : EIO;
105 return NGX_ERROR;
106 }
107
108 /* Adopt the object by value into the handle. A driver that malloc'd the obj
109 * shell (heap_shell) hands ownership of the COPY to us; free the now-
110 * redundant shell. The embedded copy is not itself a heap shell. */
111 fh->sd_obj = *obj;
112 if (obj->heap_shell) {
113 free(obj);
114 }
115 fh->sd_obj.heap_shell = 0;
116
117 /* The driver's open may defer metadata (the POSIX driver deliberately skips
118 * the fstat at open — see sd_posix_open). Populate the snapshot now via the
119 * driver's own fstat so the synthesized `struct stat` (and hence the handle's
120 * cached_size) is correct: a driver-aware fstat reports the LOGICAL object
121 * size (e.g. pblock's whole-object size, not block 0). Without this the
122 * cached_size stays 0 and the buffered read path — inline read compression,
123 * and any non-sendfile serve — sees EOF immediately and returns nothing. */
124 if (sd->driver->fstat != NULL && fh->sd_obj.snap.size == 0) {
125 (void) sd->driver->fstat(&fh->sd_obj, &fh->sd_obj.snap);
126 }
127
128 ngx_memzero(st, sizeof(*st));
129 st->st_size = fh->sd_obj.snap.size;
130 st->st_mtime = fh->sd_obj.snap.mtime;
131 st->st_ctime = fh->sd_obj.snap.ctime;
132 st->st_ino = fh->sd_obj.snap.ino;
133 st->st_mode = (fh->sd_obj.snap.mode != 0)
134 ? fh->sd_obj.snap.mode
135 : (fh->sd_obj.snap.is_dir ? (S_IFDIR | 0755)
136 : (S_IFREG | 0644));
137
138 /* The driver snapshot carries no device id, so st_dev would stay 0 and the
139 * published-handle table would record device 0. A bound secondary reopens
140 * the file for itself (a real fstat, real st_dev) and revalidates device+
141 * inode against the published entry — a 0 vs real-device mismatch would then
142 * wrongly revoke every bound read (kXR_error). For a driver with a real
143 * backing descriptor (the POSIX driver's fd is the file itself), capture the
144 * real device here so the published identity matches the secondary's reopen. */
145 if (fh->sd_obj.fd >= 0) {
146 struct stat rst;
147 if (fstat(fh->sd_obj.fd, &rst) == 0) {
148 st->st_dev = rst.st_dev;
149
150 /* Anti-wedge parity with the non-driver path's S_ISREG gate: a FIFO,
151 * socket, or device is not a servable byte stream. The confined open
152 * forced O_NONBLOCK so the open(2) itself could not park the worker,
153 * but a subsequent read/write would spin on EAGAIN — refuse it here,
154 * where the real st_mode is available (the driver snapshot may not
155 * carry a mode). A directory keeps its own handling upstream; only
156 * special files are cut. */
157 if (!S_ISREG(rst.st_mode) && !S_ISDIR(rst.st_mode)) {
158 (void) sd->driver->close(&fh->sd_obj);
159 fh->sd_obj.driver = NULL;
160 fh->sd_obj.inst = NULL;
161 fh->sd_obj.state = NULL;
162 fh->sd_obj.fd = -1;
163 errno = EINVAL;
164 return NGX_ERROR;
165 }
166 }
167 }
168
169 *out_fd = fh->sd_obj.fd; /* block-0 fd, or NGX_INVALID_FILE (-1) */
170 return NGX_OK;
171}

257 /* Write-through decision evaluation (mirrors XrdPfc::Cache::Decide())
258 * WHAT: Evaluate write-through policy at kXR_open time and cache the result on the handle.
259 * This is called once per open — the cached wt_policy determines close-time flush behavior.
260 *
261 * WHY: Mirrors XrdPfcDecision::Decide() from official XRootD PFC module (Cache::Attach()).
262 * Caching at open time avoids repeated policy evaluation for every write operation,
263 * reduces latency, and ensures consistent close-time behavior across the session.
264 *
265 * HOW: Policy callback flow (src/cache/writethrough_decision.h):
266 * 1. conf->wt.decision.fn(resolved, options, &conf->wt.decision) — default is brix_wt_default_decide()
267 * 2. Default engine checks: size filter → deny prefixes → allow prefixes → ALLOW_ASYNC (default)
268 * 3. Cache result on handle: ctx->files[idx].wt_policy = decision, wt_enabled = (decision != DENY),
269 * wt_dirty_offset = -1 (clean state), wt_bytes_written = 0
270 *
271 * Decision outcomes are cached for a future close-time write-back implementation:
272 * DENY → no write-back; local-only writes, handle treated as non-WT
273 * ALLOW_SYNC → synchronous flush to origin before closing handle (blocks)
274 * ALLOW_ASYNC→ schedule async thread-pool flush, return immediately to client */
275
276/* WT decision policy engine (default: prefix-based)
277 * WHAT: brix_wt_default_decide() — built-in prefix-based policy engine.
278 * External plugins can provide their own fn pointer for custom policies.
279 *
280 * WHY: Provides sensible defaults for most deployments without requiring external plugin setup.
281 * Prefix-based matching is O(n) where n = prefix length — acceptable for typical paths (/data/, /atlas/).
282 *
283 * HOW: Decision logic order (src/cache/writethrough_decision.c):
284 * 1. Size filter: if file > max_write_through_bytes AND no include regex match → DENY
285 * 2. Deny prefixes: any deny_prefix matches → DENY (deny takes precedence)
286 * 3. Allow prefixes: if allow list configured AND none match → DENY (whitelist mode)
287 * 4. Default: ALLOW_ASYNC (mirrors XrdPfcAllowDecision, sync preferred for local origins) */
288static void
289brix_open_decide_writethrough(brix_ctx_t *ctx, ngx_connection_t *c,
290 ngx_stream_brix_srv_conf_t *conf, const char *resolved, uint16_t options,
291 int idx, mode_t create_mode, ngx_flag_t is_write, ngx_flag_t wt_via_stage)
292{
293 if (is_write && conf->wt.enable) {
294 brix_wt_decision_t decision = BRIX_WT_DECISION_DENY;
295
296 if (conf->wt.decision.fn != NULL) {
297 decision = conf->wt.decision.fn(resolved, options, &conf->wt.decision);
298 }
299
300 /* wt sd_stage handles flush on the storage path (sync job / close), NOT via
301 * the close-time run_flush — so leave wt_enabled clear for them. */
302 ctx->files[idx].wt_enabled = (!wt_via_stage
303 && decision != BRIX_WT_DECISION_DENY) ? 1 : 0;
304 ctx->files[idx].wt_policy = decision;
305 ctx->files[idx].wt_mode_bits = create_mode;
306 ctx->files[idx].wt_dirty_offset = -1; /* no dirty writes yet */
307 ctx->files[idx].wt_bytes_written = 0;
308
309 ctx->files[idx].wt_flush_task = NULL;
310 ctx->files[idx].wt_flush_pending = 0;
311
312 if (c->log->log_level & NGX_LOG_DEBUG_STREAM) {
313 char wt_log_path[512];
314
315 brix_sanitize_log_string(resolved, wt_log_path,
316 sizeof(wt_log_path));
317 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
318 "brix: wt decision=%s path=%s",
319 decision == BRIX_WT_DECISION_DENY ? "DENY" :
320 decision == BRIX_WT_DECISION_ALLOW_SYNC ? "ALLOW_SYNC" : "ALLOW_ASYNC",
321 wt_log_path);
322 }
323 }

347 * path (a driver no-fd handle already has st), format "ino size flags mtime"
348 * into statbuf, and clear *want_stat when the stat is unavailable. */
349static void
350brix_open_build_retstat(brix_ctx_t *ctx, int idx, int fd, struct stat *st,
351 ngx_flag_t *want_stat, char *statbuf, size_t statbuf_sz)
352{
353 statbuf[0] = '\0';

355 /* A driver-backed no-fd handle (e.g. RADOS) has fd == NGX_INVALID_FILE;

374 (unsigned long long) st->st_ino,

422 /* The POSIX-fd path stats the fd to validate type and clear O_NONBLOCK. A
423 * driver-backed open already synthesized `st` from the driver snapshot and
424 * its directory/type rejection happens at the driver (EISDIR mapped above)
425 * and the pre-flight VFS probe, so this fd-specific block is skipped. */
426static ngx_int_t
427brix_open_validate_fd(brix_ctx_t *ctx, ngx_connection_t *c, const char *resolved,
428 int fd, struct stat *st, ngx_flag_t is_write, ngx_flag_t driver_backed)
429{
430 if (!driver_backed) {
431 if (fstat(fd, st) != 0) {
432 int err = errno;
433
434 close(fd);
435 BRIX_RETURN_ERR(ctx, c,
436 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
437 "OPEN", resolved, is_write ? "wr" : "rd",
438 kXR_IOError, strerror(err));
439 }
440
441 if (S_ISDIR(st->st_mode)) {
442 close(fd);
443 BRIX_RETURN_ERR(ctx, c,
444 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
445 "OPEN", resolved, is_write ? "wr" : "rd",
446 kXR_isDirectory, "is a directory");
447 }

449 /* Only regular files are servable byte streams. A FIFO, socket, device
450 * or other special file was opened O_NONBLOCK (so the open could not
451 * wedge the worker); refuse to serve it rather than let a read/write spin
452 * on EAGAIN. A staged write always lands on a freshly O_CREAT'd regular
453 * temp, so this only ever rejects a pre-existing special file. */
454 if (!S_ISREG(st->st_mode)) {
455 close(fd);
456 BRIX_RETURN_ERR(ctx, c,
457 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
458 "OPEN", resolved, is_write ? "wr" : "rd",
459 kXR_IOError, "not a regular file");
460 }
461
462 /* The fd is a confirmed regular file: drop O_NONBLOCK so every downstream
463 * read/write/sendfile sees ordinary blocking semantics (a no-op for local
464 * regular files, but it keeps the fd's flags unsurprising for callers). */
465 {
466 int fl = fcntl(fd, F_GETFL);
467 if (fl != -1 && (fl & O_NONBLOCK)) {
468 (void) fcntl(fd, F_SETFL, fl & ~O_NONBLOCK);
469 }
470 }
471 }
472 return NGX_OK;
473}

476 /* A write open of an existing DIRECTORY must be rejected up front with
477 * kXR_isDirectory (stock parity: O_WRONLY on a directory fails EISDIR).
478 * Without this the staging path below would derive a ".part" FILE from the
479 * directory's name, create it, and wrongly report success — diverging from
480 * stock. The read side is rejected symmetrically just below. */
481static ngx_int_t
482brix_open_check_write_target(brix_ctx_t *ctx, ngx_connection_t *c,
483 ngx_stream_brix_srv_conf_t *conf, const char *resolved, ngx_flag_t is_write)
484{
485 if (is_write) {
486 brix_vfs_stat_t dst;
487 /* A final path that is itself a symlink must be rejected for write: we
488 * never write THROUGH an in-root link. The direct-open mapping enforces
489 * this with O_NOFOLLOW on the final component (ELOOP), but the staging
490 * path opens a randomly-named temp instead of the final and would commit
491 * over the link on rename — so guard it here. lstat (no-follow) reports
492 * the link as itself; resolution is already confined to the export, so
493 * this catches an in-export link with EITHER an in-root or outward target
494 * without following it. */
495 if (brix_open_probe(c->log, conf->common.root_canon, resolved, 1,
496 &dst) && S_ISLNK((mode_t) dst.mode)) {
497 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
498 resolved, "wr", kXR_NotAuthorized,
499 "refusing to write through a symlink");
500 }
501 if (brix_open_probe(c->log, conf->common.root_canon, resolved, 0,
502 &dst) && dst.is_directory) {
503 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
504 resolved, "wr", kXR_isDirectory,
505 "is a directory");
506 }
507 }
508 /* NGX_DECLINED = allow/continue. A rejection above sends its own error and
509 * returns that send's rc (NGX_OK on success), so the caller MUST test for
510 * NGX_DECLINED — not NGX_OK — to tell "continue" from "error already sent"
511 * (else it falls through and sends a second, conflicting reply). */
512 return NGX_DECLINED;
513}

516 /* Phase C: two-tier write-back-staging backpressure. When write-through
517 * staging is configured with watermarks, shed new write-opens while the
518 * staging filesystem is full — delay in the soft band (kXR_wait, the client
519 * retries), reject at the hard cap (kXR_Overloaded). Runs before any handle or
520 * staging-temp allocation, so a shed write consumes nothing. Reads never reach
521 * here. */
522static ngx_int_t /* the value to return from the caller, or NGX_DECLINED = allow/continue */
523brix_open_stage_backpressure(brix_ctx_t *ctx, ngx_connection_t *c,
524 ngx_stream_brix_srv_conf_t *conf, const char *resolved, ngx_flag_t is_write)
525{
526 if (is_write && conf->wt.enable) {
527 switch (brix_wt_stage_admit(conf)) {
528 case BRIX_WT_ADMIT_WAIT:
529 brix_metric_wt_stage_throttled(0 /* wait */);
530 brix_log_access(ctx, c, "OPEN", resolved, "wr-staging-wait",
531 0, 0, "write-back staging busy; retry", 0);
532 return brix_send_wait(ctx, c, BRIX_WT_STAGE_WAIT_SECS);
533 case BRIX_WT_ADMIT_REJECT:
534 brix_metric_wt_stage_throttled(1 /* reject */);
535 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN", resolved,
536 "wr", kXR_Overloaded,
537 "write-back staging area full");
538 case BRIX_WT_ADMIT_ALLOW:
539 default:
540 break;
541 }
542 }
543 return NGX_DECLINED;
544}

547 /* When staging is active the kXR_new exclusive-create check must run against
548 * the FINAL path, not the staging temp (which is what actually gets O_EXCL):
549 * staging never opens the final, so without this an exclusive create over an
550 * existing object would wrongly succeed and overwrite it on commit.
551 *
552 * This is an EXCLUSIVE-create check, so it only applies when kXR_new is set
553 * WITHOUT kXR_delete — exactly the O_EXCL condition in the direct-open
554 * mapping (open_flags.h: kXR_new adds O_EXCL "unless kXR_delete is also
555 * set"). With kXR_delete also present the client explicitly asked to
556 * truncate/overwrite any existing object (the delete intent wins over the
557 * new intent), so an existing final must NOT be rejected — staging will
558 * replace it on the commit rename. Without this guard a delete+new overwrite
559 * (e.g. xrdcp -f, or a kXR_recoverWrts reopen) was wrongly rejected with
560 * kXR_ItExists. */
561static ngx_int_t
562brix_open_check_exclusive_create(brix_ctx_t *ctx, ngx_connection_t *c,
563 ngx_stream_brix_srv_conf_t *conf, const char *resolved, ngx_flag_t stage,
564 uint16_t options)
565{
566 if (stage && (options & kXR_new) && !(options & kXR_delete)) {
567 brix_vfs_stat_t fst;
568 if (brix_open_probe(c->log, conf->common.root_canon, resolved, 0,
569 &fst)) {
570 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
571 resolved, "wr", kXR_ItExists,
572 "file already exists");
573 }
574 }
575 /* NGX_DECLINED = allow/continue; a kXR_ItExists rejection above already sent
576 * its reply and returned that send's rc — the caller tests for NGX_DECLINED,
577 * not NGX_OK, so a sent error stops the open instead of falling through to
578 * open the staging temp and send a second (fhandle) reply. */
579 return NGX_DECLINED;
580}

583/* Build the POSC/resume staging temp path into posc_temp_path (resume = a
584 * deterministic identity-keyed partial; POSC = a random ".posc.*" temp on the
585 * final path's fs). Returns NGX_OK, or an error response on a too-long path. */
586static ngx_int_t
587brix_open_build_stage_temp(brix_ctx_t *ctx, ngx_connection_t *c,
588 ngx_stream_brix_srv_conf_t *conf, const char *resolved, ngx_flag_t use_resume,
589 ngx_flag_t use_posc, char *posc_temp_path, size_t posc_temp_path_sz)
590{
591 if (use_resume) {
592 /* Deterministic, identity-keyed: a reconnecting client re-opening the
593 * same final path lands on the SAME partial and resumes from its
594 * offset. Anonymous (empty dn) shares per-path (no per-user isolation
595 * on such an endpoint anyway). When brix_stage_dir is set the partial
596 * lives on that fast device and the close-time commit moves it to the
597 * destination (cross-device copy). */
598 const char *principal = ctx->login.dn[0] ? ctx->login.dn : NULL;
599 const char *stage_dir = conf->upload_stage_dir_canon[0]
600 ? conf->upload_stage_dir_canon : NULL;
601 if (brix_make_resume_path(resolved, principal, stage_dir,
602 posc_temp_path, posc_temp_path_sz)
603 != NGX_OK) {
604 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
605 resolved, "wr", kXR_ServerError,
606 "resume temp path too long");
607 }
608 } else if (use_posc) {
609 if (brix_make_tmp_path(resolved, posc_temp_path,
610 posc_temp_path_sz) != NGX_OK) {
611 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
612 resolved, "wr", kXR_ServerError,
613 "POSC temp path too long");
614 }
615 }
616 /* NGX_DECLINED = built OK / nothing to build (continue); a path-too-long
617 * rejection above already sent its reply — caller tests NGX_DECLINED, not
618 * NGX_OK, so a sent error is not mistaken for "continue". */
619 return NGX_DECLINED;
620}

623/*
624 *
625 * WHAT: Opens the actual file on disk and allocates a file handle (fhandle). Called after path resolution.
626 * This function performs the POSIX open(2) call with proper security guarantees including:
627 * - POSC mode: staging temp file for persist-on-successful-close writes
628 * - Confined open: brix_open_confined() prevents post-open path escape attacks
629 * - Handle allocation: brix_alloc_fhandle() assigns a slot (0–255) in fd_table.c
630 * - Bookkeeping initialization: readable/writable flags, cache origin, inode/device tracking,
631 * byte counters, timestamps, read-ahead state.
632 *
633 * WHY: This is the bridge between path resolution and data transfer. The resolved file handle
634 * carries all metadata reused by subsequent opcodes (read/pgread/readv/write/close). POSC
635 * protects against crash loss of partial writes; confined open prevents symlink escapes;
636 * handle allocation enforces the 0–255 fd-table limit.
637 *
638 * HOW: Determine POSIX flags from options/mode_bits → build POSC staging path if kXR_posc set →
639 * allocate fhandle slot → open via O_CLOEXEC (cache) or brix_open_confined() (non-cache) →
640 * stat the fd to validate regular file and populate handle metadata → set fhandle path field +
641 * posc_final_path if POSC active → apply parent group policy on write opens → evaluate WT
642 * decision policy at open time → build ServerOpenBody with fhandle + optional retstat → queue response.
643 */
644
645ngx_int_t
646brix_open_resolved_file(brix_ctx_t *ctx, ngx_connection_t *c,
647 ngx_stream_brix_srv_conf_t *conf,
648 const char *resolved, uint16_t options,
649 uint16_t mode_bits, ngx_flag_t is_write,
650 uint8_t codec)
651{
652 int idx, fd, oflags;
653 int is_readable;
654 ServerOpenBody body;
655 struct stat st;
656 char statbuf[256];
657 u_char *buf;
658 size_t bodylen, total;
659 ngx_flag_t want_stat;
660 ngx_flag_t from_cache;
661 /*
662 * POSC (persist-on-successful-close): when kXR_posc is set on a write
663 * open we stage writes to a temp file and rename to the final path only
664 * on a clean kXR_close. If the session drops mid-write the temp file is
665 * unlinked by brix_free_fhandle() (via the path field + posc_final_path
666 * sentinel). We build posc_temp_path here; it is used as the actual
667 * filesystem target for the open(2) call below.
668 */
669 char posc_temp_path[PATH_MAX];
670 ngx_flag_t use_posc = (is_write && (options & kXR_posc)) ? 1 : 0;
671 /*
672 * Upload resume (brix_upload_resume on): stage EVERY writable open to a
673 * deterministic identity-keyed partial that survives a disconnect, so a
674 * reconnecting client resumes in place. This is a superset of POSC staging
675 * (same temp-then-rename commit), so `stage` drives the open + commit and
676 * use_resume only changes (a) the temp path is deterministic, not random,
677 * and (b) the partial is preserved — not unlinked — on a non-clean close.
678 */
679 ngx_flag_t use_resume = (is_write && conf->upload_resume) ? 1 : 0;
680 ngx_flag_t stage = use_posc || use_resume;
681
682 want_stat = (options & kXR_retstat) ? 1 : 0;
683 from_cache = (conf->cache
684 && conf->cache_root.len > 0
685 && ngx_strncmp((u_char *) resolved,
686 conf->cache_root.data,
687 conf->cache_root.len) == 0);
688
689 /* Reject a write open onto a symlink or existing directory (split out). */
690 {
691 ngx_int_t wtc = brix_open_check_write_target(ctx, c, conf, resolved,
692 is_write);
693 if (wtc != NGX_DECLINED) {
694 return wtc; /* rejected: error already sent, propagate its rc */
695 }
696 }
697
698 /* Two-tier write-back-staging backpressure: shed new write-opens when the
699 * staging fs is full (kXR_wait / kXR_Overloaded) (split out). */
700 {
701 ngx_int_t bprc = brix_open_stage_backpressure(ctx, c, conf, resolved,
702 is_write);
703 if (bprc != NGX_DECLINED) {
704 return bprc;
705 }
706 }
707
708 /*
709 * Build the POSC staging temp path: same directory as the final path,
710 * with a ".posc.<pid>.<random>" suffix appended. This keeps the temp
711 * file on the same filesystem as the destination so rename(2) is atomic.
712 */
713 /* kXR_new exclusive-create check against the FINAL path when staging (split out). */
714 {
715 ngx_int_t xrc = brix_open_check_exclusive_create(ctx, c, conf, resolved,
716 stage, options);
717 if (xrc != NGX_DECLINED) {
718 return xrc; /* kXR_ItExists already sent, propagate its rc */
719 }
720 }
721
722 /* Build the POSC/resume staging temp path (split out). */
723 {
724 ngx_int_t strc = brix_open_build_stage_temp(ctx, c, conf, resolved,
725 use_resume, use_posc, posc_temp_path, sizeof(posc_temp_path));
726 if (strc != NGX_DECLINED) {
727 return strc; /* path-too-long error already sent, propagate its rc */
728 }
729 }
730
731 /*
732 * In-place update vs upload: resume staging assumes a writable open is an
733 * UPLOAD — it starts from an empty partial and commits it over the final path
734 * with a rename. But a pure update-in-place open (kXR_open_updt, neither
735 * kXR_delete/truncate nor kXR_new/create) on an ALREADY-COMMITTED file is a
736 * read-modify-write that must preserve the bytes the client does not rewrite.
737 * Staged through an empty partial, those bytes are silently lost on the commit
738 * rename (e.g. a 100-byte write at offset 100 of a 200-byte file zero-fills
739 * [0,100)) — a data-integrity bug, and a divergence from stock xrootd which
740 * always edits such a file in place.
741 *
742 * Only stage such an open when a resume partial ALREADY exists — that is a
743 * genuine reconnect continuing an interrupted upload (the partial holds the
744 * bytes received so far). When no partial exists and the final file is a real
745 * committed regular file, fall through to opening it directly, in place.
746 */
747 if (use_resume && !(options & kXR_delete) && !(options & kXR_new)) {
748 brix_vfs_stat_t fst;
749 int have_partial;
750
751 /* A pure update-in-place open is NOT an upload to stage when no resume
752 * partial is in flight: drop out of resume staging and let the direct
753 * open mapping (O_RDWR, no O_CREAT) decide, matching stock xrootd:
754 * - final is a committed regular file -> edit it in place (preserve the
755 * bytes the client does not rewrite);
756 * - final does not exist -> fail kXR_NotFound, exactly as
757 * O_RDWR-without-O_CREAT would. Staging would otherwise CREATE the
758 * missing file and return kXR_ok, diverging from stock which derives
759 * no O_CREAT for kXR_open_updt alone (XrdXrootdXeq.cc:1524). */
760 /* The partial lives under the upload stage dir when one is configured
761 * (a separate, svc-owned storage domain), else next to the final under
762 * the export root. Probe the export-root partial through the VFS; the
763 * external stage-dir partial is checked as the worker (separate domain,
764 * same reasoning as the open below). */
765 if (conf->upload_stage_dir_canon[0] != '\0') {
766 struct stat sst; /* vfs-seam-allow: separate upload stage-dir domain */
767 have_partial = (stat(posc_temp_path, &sst) == 0); /* vfs-seam-allow: separate upload stage-dir domain */
768 } else {
769 brix_vfs_stat_t pst;
770 have_partial = brix_open_probe(c->log, conf->common.root_canon,
771 posc_temp_path, 0, &pst);
772 }
773 int final_exists = brix_open_probe(c->log, conf->common.root_canon,
774 resolved, 0, &fst);
775 if (!have_partial && (!final_exists || fst.is_regular)) {
776 use_resume = 0;
777 stage = use_posc;
778 }
779 }
780
781 if (!is_write) {
782 brix_vfs_stat_t rst;
783 if (brix_open_probe(c->log, conf->common.root_canon, resolved, 0,
784 &rst) && rst.is_directory) {
785 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
786 resolved, "rd", kXR_isDirectory,
787 "is a directory");
788 }
789 }
790
791 /* The kXR_open option-bit -> POSIX open(2) mapping is the single-sourced
792 * inverse of the client's request builder (protocol/open_flags.h). */
793 brix_open_options_to_posix(options, is_write, &oflags, &is_readable);
794
795 /* Convert XRootD mode bits (Unix permission bits in low 9 bits). */
796 mode_t create_mode = (mode_bits & 0777);
797 if (create_mode == 0) {
798 create_mode = 0644;
799 }
800
801 idx = brix_alloc_fhandle(ctx);
802 if (idx < 0) {
803 BRIX_RETURN_ERR(ctx, c,
804 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
805 "OPEN", resolved, is_write ? "wr" : "rd",
806 kXR_ServerError, "too many open files");
807 }
808
809 /* Layer 3: a non-default storage driver bound to this export (block-striped
810 * or object store) handles its own opens — POSC/resume staging and the
811 * server-managed cache domain remain on the POSIX-fd path. */
812 ngx_int_t driver_backed = 0;
813 ngx_int_t open_failed = 0;
814 ngx_int_t wt_via_stage = 0;
815 brix_sd_instance_t *sd_inst =
816 brix_vfs_backend_resolve(conf->common.root_canon, c->log);
817
818 /* Write-through as ONE mechanism (Option A): route a WRITE through the composed
819 * wt sd_stage decorator (buffer on the export store, flush to the origin on
820 * sync/close) instead of the local backend + close-time run_flush. Falls back to
821 * run_flush (wt_via_stage stays 0) when no sd_stage is composed. */
822 if (is_write && conf->wt.enable && !from_cache && !use_resume) {
823 brix_sd_instance_t *wt = brix_cache_wt_stage_sd_inst(conf);
824
825 if (wt != NULL) {
826 sd_inst = wt;
827 wt_via_stage = 1;
828 }
829 }
830
831 /* §14 (phase-64): the legacy driver-backed read cache (cache_storage_backend)
832 * and the legacy slice decorator (cache_slice_inst) are RETIRED — a driver
833 * cache store and slice/partial serving are the tier grammar's composed
834 * sd_cache, reached through the sd_inst branch below. A POSIX `brix_cache
835 * on` cache keeps the raw-fd from_cache path. */
836 if (sd_inst != NULL && !from_cache && !use_resume) {
837 driver_backed = 1;
838 /* Key the driver namespace on the export-root-relative ("/sub/file")
839 * form — the same convention WebDAV/S3 and the VFS stat/dirlist/unlink
840 * paths use (brix_vfs_export_relative_root, leading slash retained), so
841 * a file written here is found by every other driver-backed op. */
842 if (brix_open_resolved_via_driver(sd_inst,
843 brix_vfs_export_relative_root(resolved,
844 conf->common.root_canon),
845 oflags, is_readable, is_write, create_mode,
846 &ctx->files[idx], &fd, &st) != NGX_OK) {
847 open_failed = 1; /* helper set errno; mapped below */
848 }
849 } else {
850 /* When POSC is active, open the staging temp path instead of the
851 * final path. The O_CREAT flag is forced so the temp file is
852 * always created fresh; O_EXCL is intentionally omitted so that a
853 * previous crash leaving a stale temp file does not block a retry. */
854 const char *open_path = stage ? posc_temp_path : resolved;
855
856 /* Staged opens (POSC or resume) always need O_CREAT on the temp path.
857 * O_TRUNC is inherited from `oflags`: a fresh create/truncate open
858 * starts the partial empty; a resume re-open (kXR_open_updt, no trunc)
859 * preserves the already-written bytes. */
860 /* O_NONBLOCK guarantees the open(2) cannot park the worker in the
861 * kernel FIFO/device "wait_for_partner" rendezvous (a named pipe in the
862 * export would otherwise freeze the worker's event loop and stall every
863 * connection pinned to it). It is harmless for the regular files we
864 * serve and is cleared again on the surviving fd once fstat() confirms
865 * S_ISREG below. Mirrors the central guard in brix_vfs_open_fd_at(). */
866 int effective_oflags = oflags | (stage ? O_CREAT : 0) | O_NONBLOCK;
867
868 /* Resume staging on a configured fast device: the partial lives OUTSIDE
869 * root_canon, so it cannot go through the RESOLVE_BENEATH open. Its
870 * basename is a server-generated hash (no client-controlled component)
871 * inside the operator-trusted, canonicalized stage dir, so a direct open
872 * with O_NOFOLLOW on the final component is safe. */
873 ngx_flag_t stage_external = use_resume
874 && conf->upload_stage_dir_canon[0] != '\0';
875
876 if (stage_external) {
877 /* vfs-seam-allow: separate storage domain. The partial lives under
878 * the operator-trusted upload stage dir (a different root than the
879 * export, server-generated hash basename, svc-owned), so it is opened
880 * as the worker — NOT through the export-confined, impersonation-aware
881 * VFS (which would resolve under the export rootfd / mapped user).
882 * O_NOFOLLOW guards the final component. */
883 fd = open(open_path, effective_oflags | O_NOFOLLOW | O_CLOEXEC, /* vfs-seam-allow: separate svc-owned storage domain (cache/stage), opened as worker */
884 create_mode);
885 } else if (from_cache) {
886 /* vfs-seam-allow: separate storage domain. cache_root files are
887 * server-managed (filled by the cache worker, never client-written,
888 * svc-owned) in a different root than the export, so they are opened
889 * as the worker rather than through the export-confined VFS. O_NOFOLLOW
890 * is defence-in-depth; O_CLOEXEC prevents FD leak into a forked child. */
891 fd = open(open_path, effective_oflags | O_NOFOLLOW | O_CLOEXEC, /* vfs-seam-allow: separate svc-owned storage domain (cache/stage), opened as worker */
892 create_mode);
893 } else {
894 /* The export final/staged path: open beneath the export root through
895 * the VFS (openat2 RESOLVE_BENEATH, impersonation-aware). The VFS
896 * strips the absolute path to its rootfd-relative form. */
897 fd = brix_vfs_open_fd_at(conf->rootfd,
898 brix_open_logical(open_path, conf->common.root_canon),
899 effective_oflags, create_mode);
900 }
901 open_failed = (fd < 0);
902 }
903 if (open_failed) {
904 int err = errno;
905 const char *mode_str = is_write ? "wr" : "rd";
906
907 if (err == ENOENT || err == ENOTDIR) {
908 BRIX_RETURN_ERR(ctx, c,
909 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
910 "OPEN", resolved, mode_str,
911 kXR_NotFound, "file not found");
912 }
913 if (err == EEXIST) {
914 /* O_EXCL (kXR_new without kXR_delete) on an existing file → EEXIST,
915 * which the reference maps to kXR_ItExists (the code raised by the
916 * kXR_new flag), NOT kXR_FileLocked. */
917 BRIX_RETURN_ERR(ctx, c,
918 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
919 "OPEN", resolved, mode_str,
920 kXR_ItExists, "file already exists");
921 }
922 if (err == EACCES) {
923 BRIX_RETURN_ERR(ctx, c,
924 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
925 "OPEN", resolved, mode_str,
926 kXR_NotAuthorized, "permission denied");
927 }
928 if (err == EISDIR) {
929 BRIX_RETURN_ERR(ctx, c,
930 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
931 "OPEN", resolved, mode_str,
932 kXR_isDirectory, "is a directory");
933 }
934 if (err == EAGAIN && !is_write) {
935 /* A nearline (tape) recall is in flight (sd_cache/sd_frm, §9.2). Tell
936 * the client to retry with kXR_wait - the stream equivalent of the
937 * WebDAV 202 "staging": the open "parks" via client retry rather than
938 * blocking the worker for the MSS latency. A later re-open re-polls the
939 * recall and, once the object is online in the cache tier, opens +
940 * serves it. */
941 brix_log_access(ctx, c, "OPEN", resolved, "rd-recall-wait",
942 0, 0, "nearline recall in progress; retry", 0);
943 return brix_send_wait(ctx, c, BRIX_RECALL_WAIT_SECS);
944 }
945 BRIX_RETURN_ERR(ctx, c,
946 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
947 "OPEN", resolved, mode_str,
948 kXR_IOError, strerror(err));
949 }
950
951 /* Stat + validate the POSIX fd (regular file, clear O_NONBLOCK) (split out). */
952 {
953 ngx_int_t vrc = brix_open_validate_fd(ctx, c, resolved, fd, &st,
954 is_write, driver_backed);
955 if (vrc != NGX_OK) {
956 return vrc;
957 }
958 }
959
960 ctx->files[idx].fd = fd;
961 ctx->files[idx].readable = is_readable;
962 ctx->files[idx].writable = is_write;
963 ctx->files[idx].from_cache = from_cache;
964 ctx->files[idx].is_regular = S_ISREG(st.st_mode) ? 1 : 0;
965 ctx->files[idx].device = st.st_dev;
966 ctx->files[idx].inode = st.st_ino;
967 ctx->files[idx].cached_size = (off_t) st.st_size;
968 ctx->files[idx].read_last_end = -1;
969 ctx->files[idx].read_ahead_end = 0;
970
971 /* Attach the CSI integrity engine to the handle (xmeta P3, split out). */
972 {
973 ngx_int_t attach_rc = brix_open_attach_csi(ctx, c, conf, resolved, idx,
974 fd, &st, is_write);
975 if (attach_rc != NGX_OK) {
976 return attach_rc;
977 }
978 }
979
980 /* Enforce the per-user open-files cap (phase-59 W3a, split out). */
981 {
982 ngx_int_t throttle_rc = brix_open_apply_throttle(ctx, c, conf, resolved,
983 idx, fd, is_write);
984 if (throttle_rc != NGX_OK) {
985 return throttle_rc;
986 }
987 }
988
989 /*
990 * Phase-42 W4/W5 — inline compression. `codec` is the codec negotiated from
991 * the open opaque (0 = none). Honour it only for a regular file, and store it
992 * in the direction-appropriate slot: read_codec for a read open (W4, compress
993 * kXR_read responses), write_codec for a write open (W5, decompress kXR_write
994 * payloads). The default (codec==0) leaves both slots 0 / byte-identical.
995 */
996 ctx->files[idx].read_codec = (!is_write && S_ISREG(st.st_mode)
997 && codec != BRIX_CODEC_IDENTITY) ? codec : (uint8_t) BRIX_CODEC_IDENTITY;
998 ctx->files[idx].write_codec = (is_write && S_ISREG(st.st_mode)
999 && codec != BRIX_CODEC_IDENTITY) ? codec : (uint8_t) BRIX_CODEC_IDENTITY;
1000 ctx->files[idx].wt_enabled = 0;
1001 ctx->files[idx].wt_policy = BRIX_WT_DECISION_DENY;
1002 ctx->files[idx].wt_mode_bits = create_mode;
