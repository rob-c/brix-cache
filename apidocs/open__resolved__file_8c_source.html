63/* Build the kXR_retstat metadata string for a retstat open: fstat the real-fd
64 * path (a driver no-fd handle already has st), format "ino size flags mtime"
65 * into statbuf, and clear *want_stat when the stat is unavailable. */
66void
67brix_open_build_retstat(brix_open_args_t *a)
68{
69 brix_ctx_t *ctx = a->ctx;
70 int idx = a->idx;
71 int fd = a->fd;
72 struct stat *st = a->st;
73 ngx_flag_t *want_stat = &a->want_stat;
74 char *statbuf = a->statbuf;
75 size_t statbuf_sz = sizeof(a->statbuf);
76
77 statbuf[0] = '\0';
78 if (*want_stat) {
79 /* A driver-backed no-fd handle (e.g. RADOS) has fd == NGX_INVALID_FILE;
80 * `st` already holds the metadata the driver captured at open (above).
81 * Only the real-fd path needs a fresh fstat. Without this guard,
82 * fstat(-1) fails and we silently drop the retstat the client requested
83 * with kXR_open — which stock clients reject ("invalid response"). */
84 int have_st = (fd != NGX_INVALID_FILE) ? (fstat(fd, st) == 0) : 1;
85
86 if (have_st) {
87 /* Compute the wire `flags` field through the shared StatGen-mirroring
88 * helper (readable/writable/xset + isDir/other, all euid-relative) so
89 * the kXR_open retstat matches stock XRootD exactly — the hand-rolled
90 * subset here dropped kXR_xset (and the type bits), so an executable
91 * (mode-777) file's open-handle stat diverged from stock by the xset
92 * bit. Same predicate as the plain kXR_stat path (brix_make_stat_body). */
93 int stat_flags = brix_stat_flags_from_stat(st, geteuid(), getegid(),
94 ctx->files[idx].from_cache ? kXR_cachersp : 0);
95 snprintf(statbuf, statbuf_sz, "%llu %lld %d %ld",
96 (unsigned long long) st->st_ino,
97 (long long) st->st_size,
98 stat_flags,
99 (long) st->st_mtime);
100 } else {
101 *want_stat = 0;
102 }
103 }
104}

156/* Build the kXR_open response buffer: the 4-byte fhandle by default, the full
157 * 12-byte ServerOpenBody + optional stat tail when the client asked for retstat
158 * or an inline codec was negotiated. Allocates from c->pool; on OOM frees the
159 * handle and returns NGX_ERROR. Writes the out_buf/out_total/out_bodylen slots. */
160static ngx_int_t
161brix_open_build_response(brix_open_args_t *a, u_char **out_buf,
162 size_t *out_total, size_t *out_bodylen)
163{
164 brix_ctx_t *ctx = a->ctx;
165 ngx_connection_t *c = a->c;
166 int idx = a->idx;
167 ngx_flag_t want_stat = a->want_stat;
168 const char *statbuf = a->statbuf;
169 ServerOpenBody body;
170 uint8_t sig_codec = ctx->files[idx].read_codec
171 ? ctx->files[idx].read_codec : ctx->files[idx].write_codec;
172 ngx_flag_t have_codec = (sig_codec != BRIX_CODEC_IDENTITY);
173 ngx_flag_t full_body = want_stat || have_codec;
174 size_t hbytes = full_body ? sizeof(ServerOpenBody)
175 : sizeof(body.fhandle); /* 4 */
176 size_t bodylen, total;
177 u_char *buf;
178
179 ngx_memzero(&body, sizeof(body));
180 body.fhandle[0] = (u_char) idx;
181 body.cpsize = 0;
182 if (have_codec) {
183 body.cpsize = (kXR_int32) htonl(BRIX_INLINE_CMP_MAGIC);
184 body.cptype[0] = sig_codec;
185 }
186
187 bodylen = hbytes;
188 if (want_stat) {
189 bodylen += strlen(statbuf) + 1;
190 }
191
192 total = XRD_RESPONSE_HDR_LEN + bodylen;
193 buf = ngx_palloc(c->pool, total);
194 if (buf == NULL) {
195 brix_free_fhandle(ctx, idx);
196 return NGX_ERROR;
197 }
198
199 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
200 (uint32_t) bodylen,
201 (ServerResponseHdr *) buf);
202
203 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body, hbytes);
204
205 if (want_stat) {
206 size_t slen = strlen(statbuf) + 1;
207 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + sizeof(ServerOpenBody),
208 statbuf, slen);
209 }
210
211 *out_buf = buf;
212 *out_total = total;
213 *out_bodylen = bodylen;
214 return NGX_OK;
215}

217/*
218 *
219 * WHAT: Opens the actual file on disk and allocates a file handle (fhandle). Called after path resolution.
220 * This function performs the POSIX open(2) call with proper security guarantees including:
221 * - POSC mode: staging temp file for persist-on-successful-close writes
222 * - Confined open: brix_open_confined() prevents post-open path escape attacks
223 * - Handle allocation: brix_alloc_fhandle() assigns a slot (0–255) in fd_table.c
224 * - Bookkeeping initialization: readable/writable flags, cache origin, inode/device tracking,
225 * byte counters, timestamps, read-ahead state.
226 *
227 * WHY: This is the bridge between path resolution and data transfer. The resolved file handle
228 * carries all metadata reused by subsequent opcodes (read/pgread/readv/write/close). POSC
229 * protects against crash loss of partial writes; confined open prevents symlink escapes;
230 * handle allocation enforces the 0–255 fd-table limit.
231 *
232 * HOW: Determine POSIX flags from options/mode_bits → build POSC staging path if kXR_posc set →
233 * allocate fhandle slot → open via O_CLOEXEC (cache) or brix_open_confined() (non-cache) →
234 * stat the fd to validate regular file and populate handle metadata → set fhandle path field +
235 * posc_final_path if POSC active → apply parent group policy on write opens → evaluate WT
236 * decision policy at open time → build ServerOpenBody with fhandle + optional retstat → queue response.
237 */

342 * kXR_ok reply (or async-recall asynresp) (split out). */
343 return brix_open_send_response(&a, buf, total, bodylen);
