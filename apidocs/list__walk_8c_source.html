1/*
2 * list_walk.c — S3 ListObjects filesystem walker, comparator, and lazy stat.
3 *
4 * WHAT: Three functions used by the V1/V2 list emitters
5 * (list_objects_v2.c / list_objects_v1.c):
6 * - s3_walk(): recursive directory walker that appends object (is_prefix=0)
7 * and CommonPrefix (is_prefix=1) entries into a growable array, respecting
8 * prefix, delimiter, and sentinel filters.
9 * - entry_cmp(): qsort comparator for lexicographic key ordering.
10 * - s3_entry_fill_stat(): lazily fill size/mtime/ETag for ONE emitted object.
11 *
12 * WHY (phase-45 W1): the previous walker `lstat`'d EVERY entry in the whole
13 * prefix subtree and stored them in a fixed `s3_entry_t[65536]` (~273 MB per
14 * request, key[4096] inline) — both costs grew with the bucket, not the page.
15 * ListObjects requires a globally-sorted key set to paginate, so the *names*
16 * must still be walked; but `size`/`mtime`/`ETag` are only needed for the
17 * ≤max-keys objects actually emitted. So this walker:
18 * 1. classifies dir/file/symlink from readdir's `d_type` with NO stat
19 * (an `lstat` fallback only on `DT_UNKNOWN`, e.g. some NFS mounts), and
20 * 2. collects only a pooled, right-sized key + the is_prefix flag,
21 * and the caller `lstat`s just the emitted page slice via s3_entry_fill_stat().
22 * → `lstat` count drops from O(objects in subtree) to O(page); the entry
23 * store drops from a fixed 273 MB to O(actual key bytes).
24 *
25 * The symlink-skip security property is preserved: a `DT_LNK` entry is skipped
26 * outright, and a `DT_UNKNOWN` entry is `lstat`'d and skipped if `S_ISLNK` — a
27 * symlink is never listed as an object nor recursed into (it could otherwise
28 * leak another tenant's tree or the host filesystem).
29 */

59/*
60 * Push a (pooled, right-sized) key into the growable entry array
61 * */
62
63static int
64s3_walk_push(ngx_array_t *entries, const char *key, unsigned is_prefix)
65{
66 size_t len = strlen(key);
67 char *kdup;
68 s3_entry_t *e;
69
70 BRIX_PNALLOC_OR_RETURN(kdup, entries->pool, len + 1, -1);
71 ngx_memcpy(kdup, key, len + 1);
72
73 e = ngx_array_push(entries);
74 if (e == NULL) {
75 return -1;
76 }
77 e->key = kdup;
78 e->is_prefix = is_prefix;
79 e->size = 0;
80 e->mtime = 0;
81 e->etag[0] = '\0';
82 return 0;
83}

85/*
86 * Classify a directory entry: 1 = dir, 2 = regular file, 0 = skip.
87 * Uses readdir d_type (no syscall); falls back to a confined lstat only on
88 * DT_UNKNOWN. Symlinks (and FIFO/SOCK/CHR/BLK) are skipped.
89 * */
90
91static int
92s3_walk_classify(ngx_pool_t *pool, ngx_log_t *log, const char *root,
93 const char *child_path, brix_vfs_dirent_kind_t dkind)
94{
95 brix_vfs_ctx_t vctx;
96 brix_vfs_stat_t vst;
97
98 switch (dkind) {
99 case BRIX_VFS_DT_DIR:
100 return 1;
101 case BRIX_VFS_DT_REG:
102 return 2;
103 case BRIX_VFS_DT_OTHER:
104 return 0; /* symlink / special: never list or traverse */
105 case BRIX_VFS_DT_UNKNOWN:
106 default:
107 /* The filesystem did not populate d_type (e.g. some NFS mounts): classify
108 * via a confined no-follow probe (non-metered — the ListObjects op already
109 * accounts for the walk). A symlink/special stats as neither dir nor
110 * regular and is skipped, preserving the symlink-skip security property. */
111 brix_vfs_ctx_init(&vctx, pool, log, BRIX_PROTO_S3, root, NULL,
112 0 /* allow_write */, 0 /* is_tls */, NULL, child_path);
113 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) != NGX_OK) {
114 return 0;
115 }
116 if (vst.is_directory) {
117 return 1;
118 }
119 if (vst.is_regular) {
120 return 2;
121 }
122 return 0;
123 }
124}

126/*
127 * Recursive directory walker — appends entries, returns the running total
128 * */
129
130int
131s3_walk(ngx_log_t *log, /* request log (for the access gate) */
132 const char *root, /* filesystem root (== root_canon) */
133 const char *dir_path, /* filesystem path to scan */
134 const char *key_prefix, /* key prefix so far */
135 const char *filter_prefix, /* ListObjects prefix param */
136 const char *delimiter, /* hierarchy delimiter */
137 ngx_array_t *entries, /* growable output array */
138 int max_entries) /* hard cap on entries */
139{
140 brix_vfs_ctx_t wctx;
141 brix_vfs_dir_t *dh;
142 char child_path[PATH_MAX];
143 char child_key[S3_MAX_KEY];
144 /* Cache lengths once — avoids repeated strlen() in the readdir loop. */
145 size_t fp_len = filter_prefix ? strlen(filter_prefix) : 0;
146 size_t del_len = delimiter ? strlen(delimiter) : 0;
147
148 /*
149 * Phase 40 confidentiality gate: under impersonation the worker uid may be
150 * able to readdir() a directory the MAPPED user cannot. Ask the broker to
151 * open it as the mapped user first; on denial skip the whole subtree rather
152 * than enumerate it with the worker's credentials. No-op when off.
153 */
154 if (brix_dirlist_access_ok(log, root, dir_path) != NGX_OK) {
155 return (int) entries->nelts;
156 }
157
158 /* Enumerate through the VFS (broker fdopendir under impersonation), using the
159 * NON-METERED opendir: a recursive ListObjects must not emit one OP_DIRLIST
160 * per visited subdirectory (the enclosing S3 list op accounts for the walk). */
161 brix_vfs_ctx_init(&wctx, entries->pool, log, BRIX_PROTO_S3, root, NULL,
162 0 /* allow_write */, 0 /* is_tls */, NULL, dir_path);
163 dh = brix_vfs_opendir_quiet(&wctx, NULL);
164 if (dh == NULL) {
165 return (int) entries->nelts;
166 }
167
168 for ( ;; ) {
169 ngx_str_t name;
170 brix_vfs_dirent_kind_t dkind;
171 const char *dname;
172 int kind;
173 ngx_int_t rrc;
174
175 if ((int) entries->nelts >= max_entries) {
176 break;
177 }
178 /* "." / ".." are filtered by brix_vfs_readdir_kind; the entry kind
179 * comes from d_type so the fast path needs no per-entry stat. */
180 rrc = brix_vfs_readdir_kind(dh, &name, &dkind);
181 if (rrc != NGX_OK) {
182 break; /* NGX_DONE (end) or error → stop with what we have */
183 }
184 dname = (const char *) name.data;
185
186 /* Build filesystem path */
187 if ((size_t) snprintf(child_path, sizeof(child_path), "%s/%s",
188 dir_path, dname) >= sizeof(child_path)) {
189 continue;
190 }
191
192 /* Build key relative to root */
193 if (key_prefix[0] != '\0') {
194 if ((size_t) snprintf(child_key, sizeof(child_key), "%s/%s",
195 key_prefix, dname) >= sizeof(child_key)) {
196 continue;
197 }
198 } else {
199 if (name.len >= sizeof(child_key)) {
200 continue;
201 }
202 memcpy(child_key, dname, name.len + 1);
203 }
204
205 /*
206 * Classify from the readdir d_type (no stat); a confined no-follow probe
207 * is used only on DT_UNKNOWN. Symlinks/specials are skipped (kind==0).
208 */
209 kind = s3_walk_classify(entries->pool, log, root, child_path, dkind);
210 if (kind == 0) {
211 continue;
212 }
213
214 if (kind == 1) { /* directory */
215 if (del_len > 0) {
216 /* Build "dir_key/" to check against filter_prefix */
217 char prefix_entry[S3_MAX_KEY];
218 if ((size_t) snprintf(prefix_entry, sizeof(prefix_entry), "%s/",
219 child_key) >= sizeof(prefix_entry)) {
220 continue;
221 }
222 size_t pe_len = strlen(prefix_entry);
223
224 /*
225 * Recurse if filter_prefix starts with this dir (i.e. the
226 * user-supplied prefix descends into this directory).
227 */
228 if (fp_len > 0 && fp_len >= pe_len
229 && strncmp(filter_prefix, prefix_entry, pe_len) == 0) {
230 s3_walk(log, root, child_path, child_key,
231 filter_prefix, delimiter, entries, max_entries);
232 continue;
233 }
234
235 /* Emit as CommonPrefix if it matches or starts with filter */
236 if (fp_len > 0
237 && strncmp(prefix_entry, filter_prefix, fp_len) != 0) {
238 continue;
239 }
240 /* CommonPrefix carries no size/mtime/etag (S3 spec) — no stat. */
241 if (s3_walk_push(entries, prefix_entry, 1) != 0) {
242 break;
243 }
244 } else {
245 /* No delimiter: recurse unconditionally */
246 s3_walk(log, root, child_path, child_key,
247 filter_prefix, delimiter, entries, max_entries);
248 }
249 } else { /* kind == 2: regular file */
250 /* Skip directory sentinel */
251 if (strcmp(dname, S3_DIR_SENTINEL) == 0) {
252 continue;
253 }
254 /* Hide internal metadata/staging artifacts (sidecars, upload temps)
255 * — never listable as an S3 object key. */
256 if (brix_is_internal_name(dname)) {
257 continue;
258 }
259 /* Filter by prefix */
260 if (fp_len > 0 && strncmp(child_key, filter_prefix, fp_len) != 0) {
261 continue;
262 }
263 /* With delimiter, skip entries that have a delimiter after prefix
264 * (they belong under a CommonPrefix subtree, not at this level). */
265 if (del_len > 0) {
266 const char *after_prefix = child_key + fp_len;
267 if (strstr(after_prefix, delimiter) != NULL) {
268 continue;
269 }
270 }
271
272 /* phase-45 W1: collect the key only; size/mtime/etag are filled
273 * lazily for the emitted page via s3_entry_fill_stat(). */
274 if (s3_walk_push(entries, child_key, 0) != 0) {
275 break;
276 }
277 }
278 }
279
280 brix_vfs_closedir(dh, log);
281 return (int) entries->nelts;
282}

284/*
285 * Lazy per-object stat — called only for the entries actually emitted
286 * */
287
288ngx_int_t
289s3_entry_fill_stat(ngx_pool_t *pool, ngx_log_t *log, const char *root,
290 s3_entry_t *e)
291{
292 char fs_path[PATH_MAX];
293 brix_vfs_ctx_t vctx;
294 brix_vfs_stat_t vst;
295 struct stat sb;
296
297 if (e->is_prefix) {
298 return NGX_OK; /* CommonPrefix: no size/mtime/etag */
299 }
300
301 /* The walk built keys relative to root, so the filesystem path of an object
302 * is always root + "/" + key (see s3_walk's child_path/child_key). */
303 if ((size_t) snprintf(fs_path, sizeof(fs_path), "%s/%s", root, e->key)
304 >= sizeof(fs_path)) {
305 return NGX_DECLINED;
306 }
307
308 /* Confined no-follow probe (non-metered — the ListObjects op accounts for
309 * the page): if the entry vanished or is no longer a regular file (e.g.
310 * swapped for a symlink after the walk), skip it — matching the eager
311 * walker's stat-failure / symlink skip. */
312 brix_vfs_ctx_init(&vctx, pool, log, BRIX_PROTO_S3, root, NULL,
313 0 /* allow_write */, 0 /* is_tls */, NULL, fs_path);
314 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) != NGX_OK
315 || !vst.is_regular) {
316 return NGX_DECLINED;
317 }
318
319 e->size = vst.size;
320 e->mtime = vst.mtime;
321 ngx_memzero(&sb, sizeof(sb));
322 sb.st_size = vst.size;
323 sb.st_mtime = vst.mtime;
324 s3_etag(&sb, e->etag, sizeof(e->etag));
325 return NGX_OK;
326}
