1/*
2 * list_walk.c — S3 ListObjects filesystem walker, comparator, and lazy stat.
3 *
4 * WHAT: Three functions used by the V1/V2 list emitters
5 * (list_objects_v2.c / list_objects_v1.c):
6 * - s3_walk(): recursive directory walker that appends object (is_prefix=0)
7 * and CommonPrefix (is_prefix=1) entries into a growable array, respecting
8 * prefix, delimiter, and sentinel filters.
9 * - entry_cmp(): qsort comparator for lexicographic key ordering.
10 * - s3_entry_fill_stat(): lazily fill size/mtime/ETag for ONE emitted object.
11 *
12 * WHY (phase-45 W1): the previous walker `lstat`'d EVERY entry in the whole
13 * prefix subtree and stored them in a fixed `s3_entry_t[65536]` (~273 MB per
14 * request, key[4096] inline) — both costs grew with the bucket, not the page.
15 * ListObjects requires a globally-sorted key set to paginate, so the *names*
16 * must still be walked; but `size`/`mtime`/`ETag` are only needed for the
17 * ≤max-keys objects actually emitted. So this walker:
18 * 1. classifies dir/file/symlink from readdir's `d_type` with NO stat
19 * (an `lstat` fallback only on `DT_UNKNOWN`, e.g. some NFS mounts), and
20 * 2. collects only a pooled, right-sized key + the is_prefix flag,
21 * and the caller `lstat`s just the emitted page slice via s3_entry_fill_stat().
22 * → `lstat` count drops from O(objects in subtree) to O(page); the entry
23 * store drops from a fixed 273 MB to O(actual key bytes).
24 *
25 * The symlink-skip security property is preserved: a `DT_LNK` entry is skipped
26 * outright, and a `DT_UNKNOWN` entry is `lstat`'d and skipped if `S_ISLNK` — a
27 * symlink is never listed as an object nor recursed into (it could otherwise
28 * leak another tenant's tree or the host filesystem).
29 */

60 * Push a (pooled, right-sized) key into the growable entry array
61 * */
62
63static int
64s3_walk_push(ngx_array_t *entries, const char *key, unsigned is_prefix)
65{
66 size_t len = strlen(key);
67 char *kdup;
68 s3_entry_t *e;
69
70 BRIX_PNALLOC_OR_RETURN(kdup, entries->pool, len + 1, -1);
71 ngx_memcpy(kdup, key, len + 1);
72
73 e = ngx_array_push(entries);
74 if (e == NULL) {
75 return -1;
76 }
77 e->key = kdup;
78 e->is_prefix = is_prefix;
79 e->size = 0;
80 e->mtime = 0;
81 e->etag[0] = '\0';
82 return 0;
83}

85/*
86 * Classify a directory entry: 1 = dir, 2 = regular file, 0 = skip.
87 * Uses readdir d_type (no syscall); falls back to a confined lstat only on
88 * DT_UNKNOWN. Symlinks (and FIFO/SOCK/CHR/BLK) are skipped.
89 * */
90
91static int
92s3_walk_classify(ngx_pool_t *pool, ngx_log_t *log, const char *root,
93 const char *child_path, brix_vfs_dirent_kind_t dkind)
94{
95 brix_vfs_ctx_t vctx;
96 brix_vfs_stat_t vst;
97
98 switch (dkind) {
99 case BRIX_VFS_DT_DIR:
100 return 1;
101 case BRIX_VFS_DT_REG:
102 return 2;
103 case BRIX_VFS_DT_OTHER:
104 return 0; /* symlink / special: never list or traverse */
105 case BRIX_VFS_DT_UNKNOWN:
106 default:
107 /* The filesystem did not populate d_type (e.g. some NFS mounts): classify
108 * via a confined no-follow probe (non-metered — the ListObjects op already
109 * accounts for the walk). A symlink/special stats as neither dir nor
110 * regular and is skipped, preserving the symlink-skip security property. */
111 brix_vfs_ctx_init(&vctx, pool, log, BRIX_PROTO_S3, root, NULL,
112 0 /* allow_write */, 0 /* is_tls */, NULL, child_path);
113 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) != NGX_OK) {
114 return 0;
115 }
116 if (vst.is_directory) {
117 return 1;
118 }
119 if (vst.is_regular) {
120 return 2;
121 }
122 return 0;
123 }
124}

126/*
127 * s3_walk_ctx_t — the invariant parameters of one recursive ListObjects walk.
128 *
129 * WHAT: bundles the eight fixed inputs of s3_walk() (log, root, the two prefix
130 * filters + their cached lengths, the delimiter, the output array, and the
131 * entry cap) so the recursion and the per-entry helpers pass ONE pointer
132 * instead of an eight-argument list.
133 * WHY: s3_walk()'s public signature is frozen (recursive + an out-of-file
134 * caller in list_common.c), but its internal fan-out — path build, classify,
135 * directory emit, file emit — repeated the same eight values on every call
136 * and inflated cyclomatic complexity. A file-local context lets those helpers
137 * take (ctx, per-entry args) and keeps the walk loop flat.
138 * HOW: s3_walk() populates it once (caching filter_prefix / delimiter lengths,
139 * which are constant across the whole recursion) and hands &ctx to
140 * s3_walk_run(); dir_path / key_prefix stay per-call arguments because they
141 * change at each recursion level.
142 */
143typedef struct {
144 ngx_log_t *log; /* request log (for the access gate) */
145 const char *root; /* filesystem root (== root_canon) */
146 const char *filter_prefix; /* ListObjects prefix param (or NULL) */
147 const char *delimiter; /* hierarchy delimiter (or NULL) */
148 size_t fp_len; /* strlen(filter_prefix), cached */
149 size_t del_len; /* strlen(delimiter), cached */
150 ngx_array_t *entries; /* growable output array */
151 int max_entries; /* hard cap on entries */
152} s3_walk_ctx_t;

159/*
160 * s3_child_paths_t — the two fixed-size buffers built per directory entry.
161 *
162 * WHAT: pairs the filesystem child path (PATH_MAX) with the root-relative S3
163 * key (S3_MAX_KEY) that s3_walk_build_paths() fills for one readdir entry.
164 * WHY: bundling the two output buffers into one out-param keeps
165 * s3_walk_build_paths() within the 5-parameter budget while preserving the
166 * caller-owned, stack-allocated storage (no extra allocation).
167 * HOW: the caller declares one on the stack and passes its address; the buffer
168 * sizes are the struct's array bounds, so no size arguments are needed.
169 */
170typedef struct {
171 char path[PATH_MAX]; /* "<dir_path>/<dname>" */
172 char key[S3_MAX_KEY]; /* root-relative object/prefix key */
173} s3_child_paths_t;

176/*
177 * Build the child filesystem path and S3 key for one directory entry.
178 *
179 * WHAT: writes "<dir_path>/<dname>" into out->path and the root-relative key
180 * into out->key. Returns 0 on success, -1 if either buffer would overflow
181 * (the caller then skips this entry, matching the original inline behaviour).
182 * WHY: hoists two bounds-checked snprintf/memcpy blocks out of the walk loop so
183 * the loop reads as classify → emit without truncation noise.
184 * HOW: out->path is always "<dir_path>/<dname>"; out->key is
185 * "<key_prefix>/<dname>" at depth, or the bare name at the bucket root
186 * (key_prefix == ""). name carries the pre-measured length for the root case.
187 */
188static int
189s3_walk_build_paths(const char *dir_path, const char *key_prefix,
190 const ngx_str_t *name, s3_child_paths_t *out)
191{
192 const char *dname = (const char *) name->data;
193
194 if ((size_t) snprintf(out->path, sizeof(out->path), "%s/%s",
195 dir_path, dname) >= sizeof(out->path)) {
196 return -1;
197 }
198
199 if (key_prefix[0] != '\0') {
200 if ((size_t) snprintf(out->key, sizeof(out->key), "%s/%s",
201 key_prefix, dname) >= sizeof(out->key)) {
202 return -1;
203 }
204 } else {
205 if (name->len >= sizeof(out->key)) {
206 return -1;
207 }
208 memcpy(out->key, dname, name->len + 1);
209 }
210 return 0;
211}

214/*
215 * Handle a directory entry: recurse into it or emit it as a CommonPrefix.
216 *
217 * WHAT: for kind==1 (directory) entries. With no delimiter, recurses
218 * unconditionally. With a delimiter, either recurses (when the user prefix
219 * descends into this dir) or emits "<key>/" as a CommonPrefix if it matches
220 * the filter. Returns 0 to continue the loop, -1 on a push failure (which
221 * stops the walk, as before).
222 * WHY: isolates the directory branch (the deepest nesting in the old loop) into
223 * one named step with early-return over the nested-if pyramid.
224 * HOW: CommonPrefix carries no size/mtime/etag (S3 spec) so no stat is done;
225 * recursion re-uses the shared ctx and only advances dir_path/child_key.
226 */
227static int
228s3_walk_emit_dir(s3_walk_ctx_t *ctx, const char *child_path,
229 const char *child_key)
230{
231 char prefix_entry[S3_MAX_KEY];
232 size_t pe_len;
233
234 if (ctx->del_len == 0) {
235 /* No delimiter: recurse unconditionally. */
236 s3_walk_run(ctx, child_path, child_key);
237 return 0;
238 }
239
240 /* Build "dir_key/" to check against filter_prefix. */
241 if ((size_t) snprintf(prefix_entry, sizeof(prefix_entry), "%s/", child_key)
242 >= sizeof(prefix_entry)) {
243 return 0; /* truncated → skip this entry */
244 }
245 pe_len = strlen(prefix_entry);
246
247 /*
248 * Recurse if filter_prefix starts with this dir (i.e. the user-supplied
249 * prefix descends into this directory).
250 */
251 if (ctx->fp_len > 0 && ctx->fp_len >= pe_len
252 && strncmp(ctx->filter_prefix, prefix_entry, pe_len) == 0) {
253 s3_walk_run(ctx, child_path, child_key);
254 return 0;
255 }
256
257 /* Emit as CommonPrefix if it matches or starts with filter. */
258 if (ctx->fp_len > 0
259 && strncmp(prefix_entry, ctx->filter_prefix, ctx->fp_len) != 0) {
260 return 0;
261 }
262 /* CommonPrefix carries no size/mtime/etag (S3 spec) — no stat. */
263 if (s3_walk_push(ctx->entries, prefix_entry, 1) != 0) {
264 return -1;
265 }
266 return 0;
267}

270/*
271 * Handle a regular-file entry: apply the object filters and push the key.
272 *
273 * WHAT: for kind==2 (regular file) entries. Skips the directory sentinel and
274 * internal sidecar/staging names, applies the prefix filter, and (with a
275 * delimiter) skips keys that carry a delimiter after the prefix (they belong
276 * under a CommonPrefix subtree). Returns 0 to continue, -1 on push failure.
277 * WHY: isolates the file branch so the walk loop stays flat; keeps the sentinel
278 * / internal-name / prefix / delimiter filter order byte-identical to the
279 * original so the emitted key set is unchanged.
280 * HOW: collects the key only (phase-45 W1); size/mtime/etag are filled lazily
281 * for the emitted page via s3_entry_fill_stat().
282 */
283static int
284s3_walk_emit_file(s3_walk_ctx_t *ctx, const char *child_key,
285 const char *dname)
286{
287 /* Skip directory sentinel. */
288 if (strcmp(dname, S3_DIR_SENTINEL) == 0) {
289 return 0;
290 }
291 /* Hide internal metadata/staging artifacts (sidecars, upload temps) —
292 * never listable as an S3 object key. */
293 if (brix_is_internal_name(dname)) {
294 return 0;
295 }
296 /* Filter by prefix. */
297 if (ctx->fp_len > 0
298 && strncmp(child_key, ctx->filter_prefix, ctx->fp_len) != 0) {
299 return 0;
300 }
301 /* With delimiter, skip entries that have a delimiter after prefix
302 * (they belong under a CommonPrefix subtree, not at this level). */
303 if (ctx->del_len > 0) {
304 const char *after_prefix = child_key + ctx->fp_len;
305 if (strstr(after_prefix, ctx->delimiter) != NULL) {
306 return 0;
307 }
308 }
309
310 /* phase-45 W1: collect the key only; size/mtime/etag are filled lazily for
311 * the emitted page via s3_entry_fill_stat(). */
312 if (s3_walk_push(ctx->entries, child_key, 0) != 0) {
313 return -1;
314 }
315 return 0;
316}

319/*
320 * Process one readdir entry: build paths, classify, dispatch to dir/file emit.
321 *
322 * WHAT: given a readdir name+kind, computes the child path/key, classifies it
323 * (dir / regular file / skip), and routes to s3_walk_emit_dir or
324 * s3_walk_emit_file. Returns 0 to continue the loop, -1 to stop it.
325 * WHY: one nameable step per entry keeps s3_walk_run's loop to open → read →
326 * process, removing the branch nesting that drove the CCN warning.
327 * HOW: symlinks/specials classify as kind==0 and are skipped, preserving the
328 * symlink-skip security property.
329 */
330static int
331s3_walk_entry(s3_walk_ctx_t *ctx, const char *dir_path, const char *key_prefix,
332 const ngx_str_t *name, brix_vfs_dirent_kind_t dkind)
333{
334 s3_child_paths_t cp;
335 const char *dname = (const char *) name->data;
336 int kind;
337
338 if (s3_walk_build_paths(dir_path, key_prefix, name, &cp) != 0) {
339 return 0; /* path/key truncation → skip this entry */
340 }
341
342 /*
343 * Classify from the readdir d_type (no stat); a confined no-follow probe is
344 * used only on DT_UNKNOWN. Symlinks/specials are skipped (kind==0).
345 */
346 kind = s3_walk_classify(ctx->entries->pool, ctx->log, ctx->root,
347 cp.path, dkind);
348 if (kind == 0) {
349 return 0;
350 }
351 if (kind == 1) {
352 return s3_walk_emit_dir(ctx, cp.path, cp.key);
353 }
354 return s3_walk_emit_file(ctx, cp.key, dname);
355}

358/*
359 * Recursive walk core — open one directory and process its entries.
360 *
361 * WHAT: enforces the impersonation list gate, opens dir_path through the VFS,
362 * and loops readdir → s3_walk_entry until end-of-dir, the entry cap, or a
363 * push failure. Returns the running entry count.
364 * WHY: carries the recursion (s3_walk_emit_dir calls back into it) while
365 * s3_walk() stays a thin public wrapper with the frozen 8-arg signature.
366 * HOW: uses the NON-METERED opendir so a recursive ListObjects does not emit an
367 * OP_DIRLIST per visited subdirectory (the enclosing S3 list op accounts for
368 * the whole walk).
369 */
370static int
371s3_walk_run(s3_walk_ctx_t *ctx, const char *dir_path, const char *key_prefix)
372{
373 brix_vfs_ctx_t wctx;
374 brix_vfs_dir_t *dh;
375
376 /*
377 * Phase 40 confidentiality gate: under impersonation the worker uid may be
378 * able to readdir() a directory the MAPPED user cannot. Ask the broker to
379 * open it as the mapped user first; on denial skip the whole subtree rather
380 * than enumerate it with the worker's credentials. No-op when off.
381 */
382 if (brix_dirlist_access_ok(ctx->log, ctx->root, dir_path) != NGX_OK) {
383 return (int) ctx->entries->nelts;
384 }
385
386 /* Enumerate through the VFS (broker fdopendir under impersonation), using
387 * the NON-METERED opendir. */
388 brix_vfs_ctx_init(&wctx, ctx->entries->pool, ctx->log, BRIX_PROTO_S3,
389 ctx->root, NULL, 0 /* allow_write */, 0 /* is_tls */, NULL, dir_path);
390 dh = brix_vfs_opendir_quiet(&wctx, NULL);
391 if (dh == NULL) {
392 return (int) ctx->entries->nelts;
393 }
394
395 for ( ;; ) {
396 ngx_str_t name;
397 brix_vfs_dirent_kind_t dkind;
398 ngx_int_t rrc;
399
400 if ((int) ctx->entries->nelts >= ctx->max_entries) {
401 break;
402 }
403 /* "." / ".." are filtered by brix_vfs_readdir_kind; the entry kind
404 * comes from d_type so the fast path needs no per-entry stat. */
405 rrc = brix_vfs_readdir_kind(dh, &name, &dkind);
406 if (rrc != NGX_OK) {
407 break; /* NGX_DONE (end) or error → stop with what we have */
408 }
409
410 if (s3_walk_entry(ctx, dir_path, key_prefix, &name, dkind) != 0) {
411 break; /* push failure → stop with what we have */
412 }
413 }
414
415 brix_vfs_closedir(dh, ctx->log);
416 return (int) ctx->entries->nelts;
417}

420/*
421 * Recursive directory walker — appends entries, returns the running total
422 * */
423
424int
425s3_walk(ngx_log_t *log, /* request log (for the access gate) */
426 const char *root, /* filesystem root (== root_canon) */
427 const char *dir_path, /* filesystem path to scan */
428 const char *key_prefix, /* key prefix so far */
429 const char *filter_prefix, /* ListObjects prefix param */
430 const char *delimiter, /* hierarchy delimiter */
431 ngx_array_t *entries, /* growable output array */
432 int max_entries) /* hard cap on entries */
433{
434 s3_walk_ctx_t ctx;
435
436 ngx_memzero(&ctx, sizeof(ctx));
437 ctx.log = log;
438 ctx.root = root;
439 ctx.filter_prefix = filter_prefix;
440 ctx.delimiter = delimiter;
441 /* Cache lengths once — avoids repeated strlen() in the readdir loop. */
442 ctx.fp_len = filter_prefix ? strlen(filter_prefix) : 0;
443 ctx.del_len = delimiter ? strlen(delimiter) : 0;
444 ctx.entries = entries;
445 ctx.max_entries = max_entries;
446
447 return s3_walk_run(&ctx, dir_path, key_prefix);
448}

450/*
451 * Lazy per-object stat — called only for the entries actually emitted
452 * */
453
454ngx_int_t
455s3_entry_fill_stat(ngx_pool_t *pool, ngx_log_t *log, const char *root,
456 s3_entry_t *e)
457{
458 char fs_path[PATH_MAX];
459 brix_vfs_ctx_t vctx;
460 brix_vfs_stat_t vst;
461 struct stat sb;
462
463 if (e->is_prefix) {
464 return NGX_OK; /* CommonPrefix: no size/mtime/etag */
465 }
466
467 /* The walk built keys relative to root, so the filesystem path of an object
468 * is always root + "/" + key (see s3_walk's child_path/child_key). */
469 if ((size_t) snprintf(fs_path, sizeof(fs_path), "%s/%s", root, e->key)
470 >= sizeof(fs_path)) {
471 return NGX_DECLINED;
472 }
473
474 /* Confined no-follow probe (non-metered — the ListObjects op accounts for
475 * the page): if the entry vanished or is no longer a regular file (e.g.
476 * swapped for a symlink after the walk), skip it — matching the eager
477 * walker's stat-failure / symlink skip. */
478 brix_vfs_ctx_init(&vctx, pool, log, BRIX_PROTO_S3, root, NULL,
479 0 /* allow_write */, 0 /* is_tls */, NULL, fs_path);
480 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) != NGX_OK
481 || !vst.is_regular) {
482 return NGX_DECLINED;
483 }
484
485 e->size = vst.size;
486 e->mtime = vst.mtime;
487 ngx_memzero(&sb, sizeof(sb));
488 sb.st_size = vst.size;
489 sb.st_mtime = vst.mtime;
490 s3_etag(&sb, e->etag, sizeof(e->etag));
491 return NGX_OK;
492}
