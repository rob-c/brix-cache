17/* Build the SSI reply payload (tag + meta + data) into a c->pool buffer and push
18 * it as a deferred response (kXR_ok inner status). 0 on success. */
19static ngx_int_t
20ssi_push_reply(brix_ctx_t *ctx, ngx_connection_t *c, brix_ssi_req_t *rq,
21 char tag, const unsigned char *data, size_t data_len)
22{
23 size_t total = brix_ssi_reply_len(rq->meta_len, data_len);
24 u_char *buf = ngx_palloc(c->pool, total);
25
26 if (buf == NULL) {
27 return NGX_ERROR;
28 }
29 brix_ssi_reply_build(tag, rq->meta, rq->meta_len, data, data_len, buf);
30 return brix_send_attn_asynresp(ctx, c, rq->defer_streamid, kXR_ok,
31 buf, (uint32_t) total);
32}

34/* Push a terminal kXR_error (err_code + err_text) for the request. */
35static ngx_int_t
36ssi_push_error(brix_ctx_t *ctx, ngx_connection_t *c, brix_ssi_req_t *rq)
37{
38 const char *text = rq->err_text[0] ? rq->err_text : "SSI error";
39 size_t tlen = ngx_strlen(text);
40 uint32_t code = htonl((uint32_t) (rq->err_code ? rq->err_code
41 : kXR_ServerError));
42 size_t bodylen = 4 + tlen + 1; /* errnum(BE) + text + NUL */
43 u_char *body = ngx_palloc(c->pool, bodylen);
44
45 if (body == NULL) {
46 return NGX_ERROR;
47 }
48 ngx_memcpy(body, &code, 4);
49 ngx_memcpy(body + 4, text, tlen);
50 body[4 + tlen] = '\0';
51 return brix_send_attn_asynresp(ctx, c, rq->defer_streamid, kXR_error,
52 body, (uint32_t) bodylen);
53}

55void
56brix_ssi_deliver(brix_ctx_t *ctx, ngx_connection_t *c,
57 brix_ssi_session_t *s, uint32_t req_id,
58 brix_ssi_dlv_kind kind)
59{
60 brix_ssi_req_t *rq = brix_ssi_session_req(s, req_id, 0);
61 ngx_int_t rc = NGX_ERROR;
62
63 if (rq == NULL) {
64 return; /* reqId slot gone (cancelled/closed) — drop the late delivery */
65 }
66
67 switch (kind) {
68 case SSI_DLV_RESPONSE:
69 rc = ssi_push_reply(ctx, c, rq, BRIX_SSI_ATTN_FULL,
70 rq->resp.data, rq->resp.len);
71 break;
72 case SSI_DLV_PEND:
73 rc = ssi_push_reply(ctx, c, rq, BRIX_SSI_ATTN_PEND, NULL, 0);
74 break;
75 case SSI_DLV_ERROR:
76 rc = ssi_push_error(ctx, c, rq);
77 BRIX_SRV_METRIC_INC(ctx, ssi_errors_total);
78 break;
79 }
80 if (rc != NGX_OK) {
81 BRIX_SRV_METRIC_INC(ctx, ssi_attn_push_failures_total);
82 }
83}
