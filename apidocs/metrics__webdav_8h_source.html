82/*
83 * Per-process WebDAV metrics. All fields are ngx_atomic_t so they are safe to
84 * increment from any worker without a mutex. Indexed arrays use the
85 * BRIX_WEBDAV_METHOD_*, BRIX_HTTP_STATUS_*, BRIX_WEBDAV_AUTH_RESULT_*,
86 * etc. constants defined above.
87 */
88typedef struct {
89 ngx_atomic_t requests_total[BRIX_WEBDAV_NMETHODS]; /* requests by HTTP method */
90 ngx_atomic_t responses_total[BRIX_WEBDAV_NMETHODS][BRIX_HTTP_NSTATUS]; /* responses by method × status class */
91 ngx_atomic_t auth_total[BRIX_WEBDAV_NAUTH_RESULTS]; /* auth outcomes by result code */
92
93 ngx_atomic_t bytes_rx_total; /* bytes received (PUT body) */
94 ngx_atomic_t bytes_tx_total; /* bytes sent (GET body) */
95
96 /* Per-IP-version bandwidth — avoids high-cardinality label explosion. */
97 ngx_atomic_t bytes_rx_ipv4_total;
98 ngx_atomic_t bytes_tx_ipv4_total;
99 ngx_atomic_t bytes_rx_ipv6_total;
100 ngx_atomic_t bytes_tx_ipv6_total;
101
102 ngx_atomic_t range_total[BRIX_WEBDAV_NRANGE_RESULTS]; /* Range request outcomes */
103 ngx_atomic_t put_body_total[BRIX_WEBDAV_NPUT_MODES]; /* PUT receive mode used */
104 ngx_atomic_t propfind_depth_total[BRIX_WEBDAV_NPROPFIND_DEPTHS]; /* PROPFIND Depth: header value */
105 ngx_atomic_t propfind_entries_total; /* total directory entries listed */
106 ngx_atomic_t tpc_total[BRIX_WEBDAV_NTPC_EVENTS]; /* HTTP-TPC curl pull events */
107 ngx_atomic_t tpc_cred_total[BRIX_WEBDAV_NTPC_CRED_EVENTS]; /* HTTP-TPC cred delegation */
108 ngx_atomic_t cors_total[BRIX_WEBDAV_NCORS_EVENTS]; /* CORS preflight/allowed/denied */
109} ngx_brix_webdav_metrics_t;
