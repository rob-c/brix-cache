1/*
2 * host_format.c — see host_format.h.
3 *
4 * Bracket-on-emit for IPv6 literal hosts. The module stores hosts canonically
5 * bare (ngx_sock_ntop(...,0) for captured peers, ngx_parse_url which strips the
6 * brackets off "[::1]" input, getnameinfo(NI_NUMERICHOST) for resolved pins);
7 * these helpers re-bracket only at the point a host is written into a host:port
8 * wire string, HTTP Host: header, or rebuilt root:// URL. Mirrors the format
9 * already used at src/read/locate.c (AF_INET6 locate branch: "S%c[%s]:%d").
10 */

24size_t
25brix_format_host(const char *host, char *out, size_t sz)
26{
27 int n;
28
29 if (out == NULL || sz == 0) {
30 return 0;
31 }
32 if (host == NULL) {
33 out[0] = '\0';
34 return 0;
35 }
36
37 if (brix_host_is_ipv6_literal(host)) {
38 n = snprintf(out, sz, "[%s]", host);
39 } else {
40 n = snprintf(out, sz, "%s", host);
41 }
42
43 /* Truncation (n >= sz) leaves a partial string — reject it so callers never
44 * emit a malformed half-bracketed host; out stays a valid empty string. */
45 if (n < 0 || (size_t) n >= sz) {
46 out[0] = '\0';
47 return 0;
48 }
49 return (size_t) n;
50}
