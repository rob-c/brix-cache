4/*
5 * sd_xroot.h — remote root:// origin storage driver (read + write data path).
6 *
7 * WHAT: A capability-typed SD driver against a REMOTE XRootD (root:// / roots://)
8 * server. Read slots (open/stat/pread/fstat/close) plus, as of Phase 1 of
9 * the writable-remote-backend work, the WRITE DATA PATH (pwrite/ftruncate/
10 * fsync over kXR_write/_truncate/_sync; caps RANGE_READ|RANDOM_WRITE|
11 * TRUNCATE). It is the root:// sibling of the S3 sd_remote driver: the cache
12 * fill drives the read side driver→driver (open the origin file, pread
13 * ranges into the staged-write sink).
14 *
15 * WHY: It folds the root:// origin onto the SAME SD seam as S3, the local cache,
16 * and the export, so storage is origin-agnostic. The write slots are the
17 * foundation for making a remote root:// a first-class writable backend with
18 * optional local staging (see docs/superpowers/specs/2026-06-29-writable-
19 * remote-root-staged-write-design.md); staged-write and namespace (mkdir/
20 * rename) slots are later phases. Not yet registry-selectable as a primary
21 * export backend (Phase 2 wires that) — today it is cache-constructed only.
22 *
23 * HOW: It wraps the in-process XRootD origin wire client (cache/origin_*.c) —
24 * which needs only a server conf + a logical path — behind the SD vtable.
25 * The per-open object holds a live origin connection + open file handle;
26 * pread issues a kXR_read range into a memory sink. AUTH: anonymous login
27 * (the wire client's native mode). Authenticated root:// origins (bearer
28 * token / GSI X.509 proxy) are filled by the cache's native-client
29 * delegation (brix_cache_origin_proxy / _token), not this in-process
30 * driver — a full in-process root:// token/GSI client is a follow-on (that
31 * auth logic lives in libxrdc, which src/ cannot link). Blocking; runs only
32 * on the cache-fill worker thread. Instance/objects are malloc-owned.
33 *
34 * NOTE: This driver intentionally depends on the cache's origin wire client
35 * (cache/cache_internal.h) — it exists to expose that client as an SD
36 * backend, so the dependency is inherent rather than a layering slip.
37 */

48/* brix_sd_xroot_origin_cfg_t — the EXPLICIT origin parameters for a
49 * registry/cache-built remote root:// instance.
50 *
51 * WHAT: One value carrying every input `brix_sd_xroot_create_origin()` needs to
52 * synthesize the minimal wire-client conf: the endpoint (host/port/tls/
53 * af_policy), the optional §14 credential set (bearer / x509_proxy /
54 * x509_key / sss_keytab), and the CA store source (ca_dir). Every string
55 * field may be NULL/"" (unset); the pointed-at bytes need only outlive the
56 * create call (they are copied onto instance-owned storage).
57 * WHY: It collapses the nine loose create arguments to a single struct pointer,
58 * keeping the create function under the parameter gate and giving each
59 * caller one named-field literal instead of a positional argument stack —
60 * the credential/endpoint fields are easy to transpose positionally.
61 * HOW: `host` (required, non-empty) + `port` (1..65535, required) + `tls` +
62 * `af_policy` (brix_af_policy_t: BRIX_AF_AUTO tries all, _INET/_INET6 force
63 * IPv4/IPv6) name the endpoint. `bearer` is presented via ztn (XrdSecztn),
64 * `x509_proxy` (a PEM path) + optional separate `x509_key` via the
65 * in-process XrdSecgsi handshake, and `sss_keytab` via SSS, when the origin
66 * demands authentication. `ca_dir` (a CA file or hashed dir) builds the
67 * store that VERIFIES the origin's server cert during the GSI handshake
68 * (MITM protection). */
69typedef struct {
70 const char *host; /* required: origin hostname (non-empty) */
71 int port; /* required: origin port (1..65535) */
72 int tls; /* non-zero: connect via roots:// (TLS) */
73 int af_policy; /* brix_af_policy_t: address-family constraint */
74 const char *bearer; /* §14/C-3 ztn bearer token text (NULL/"" = anon) */
75 const char *x509_proxy; /* §14/C-3 GSI proxy (or cert) PEM path */
76 const char *x509_key; /* §14/C-3 separate GSI private key path */
77 const char *ca_dir; /* CA file or hashed dir verifying the origin cert */
78 const char *sss_keytab; /* §14 SSS shared-secret keytab path */
79} brix_sd_xroot_origin_cfg_t;
