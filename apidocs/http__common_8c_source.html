111/*
112 * brix_http_common_create_loc_conf() — allocate the common module's location
113 * conf and seed the embedded preamble with UNSET sentinels so parent->child
114 * inheritance (below) can tell "not configured" from an explicit value.
115 */
116static void *
117brix_http_common_create_loc_conf(ngx_conf_t *cf)
118{
119 ngx_http_brix_common_conf_t *c;
120
121 c = ngx_pcalloc(cf->pool, sizeof(*c));
122 if (c == NULL) {
123 return NULL;
124 }
125 ngx_http_brix_shared_init(&c->common);
126 return c;
127}

129/*
130 * brix_http_common_merge_loc_conf() — inheritance-only merge: propagate parent
131 * values into unset child slots and apply NO defaults. Per-protocol defaults
132 * still come from ngx_http_brix_shared_merge() in each protocol's merge, so a
133 * field left unset here stays UNSET and lets each protocol pick its own
134 * default after adopting the unified value.
135 */
136static char *
137brix_http_common_merge_loc_conf(ngx_conf_t *cf, void *parent, void *child)
138{
139 ngx_http_brix_common_conf_t *prev = parent;
140 ngx_http_brix_common_conf_t *conf = child;
141
142 (void) cf;
143 brix_shared_adopt_unified(&conf->common, &prev->common);
144 return NGX_CONF_OK;
145}
