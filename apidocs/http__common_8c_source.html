37/*
38 * brix_conf_set_mint_ca — setter for "brix_storage_credential_mint_ca <cert>
39 * <key>" (phase-2 T9). Validates both PEM files load-parse at config time
40 * (nginx -t fails loudly on a bad mint CA instead of every mint request
41 * failing at runtime) and stores their paths into the shared preamble's
42 * storage_credential_mint_ca_cert / _key fields. TRUST NOTE: configuring this
43 * directive means the frontend will sign per-user x509 proxies with this CA
44 * key — the ORIGIN must trust this CA for minted credentials to be usable;
45 * see src/fs/backend/cred_mint.h for the full trust-model note.
46 */
47static char *
48brix_conf_set_mint_ca(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
49{
50 ngx_http_brix_common_conf_t *c = conf;
51 ngx_str_t *value = cf->args->elts;
52 FILE *f;
53 X509 *cert;
54 EVP_PKEY *key;
55
56 (void) cmd;
57
58 f = fopen((const char *) value[1].data, "r");
59 cert = (f != NULL) ? PEM_read_X509(f, NULL, NULL, NULL) : NULL;
60 if (f != NULL) {
61 (void) fclose(f); /* read-only stream: close failure cannot lose data */
62 }
63 if (cert == NULL) {
64 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
65 "brix_storage_credential_mint_ca: cannot parse CA cert \"%V\"",
66 &value[1]);
67 return NGX_CONF_ERROR;
68 }
69 X509_free(cert);
70
71 f = fopen((const char *) value[2].data, "r");
72 key = (f != NULL) ? PEM_read_PrivateKey(f, NULL, NULL, NULL) : NULL;
73 if (f != NULL) {
74 (void) fclose(f); /* read-only stream: close failure cannot lose data */
75 }
76 if (key == NULL) {
77 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
78 "brix_storage_credential_mint_ca: cannot parse CA key \"%V\"",
79 &value[2]);
80 return NGX_CONF_ERROR;
81 }
82 EVP_PKEY_free(key);
83
84 c->common.storage_credential_mint_ca_cert = value[1];
85 c->common.storage_credential_mint_ca_key = value[2];
86 return NGX_CONF_OK;
87}

89/*
90 * brix_conf_set_peers — setter for "brix_cache_peers <member> <member> ..."
91 * (phase-85 F8 sibling mesh). Each member is "host:port", with this node's own
92 * ring slot written "self=host:port". The tokens are only COLLECTED here (into
93 * an ngx_str_t array on the shared preamble); shape validation — exactly one
94 * self=, ≥2 members, well-formed authorities — runs at tier registration where
95 * the [emerg] can name the export.
96 */
97static char *
98brix_conf_set_peers(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
99{
100 ngx_http_brix_common_conf_t *c = conf;
101 ngx_str_t *value = cf->args->elts;
102 ngx_str_t *slot;
103 ngx_uint_t i;
104
105 (void) cmd;
106
107 if (c->common.cache_peers != NULL) {
108 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
109 "brix_cache_peers: duplicate directive — list every ring "
110 "member in one declaration");
111 return NGX_CONF_ERROR;
112 }
113 c->common.cache_peers = ngx_array_create(cf->pool, cf->args->nelts - 1,
114 sizeof(ngx_str_t));
115 if (c->common.cache_peers == NULL) {
116 return NGX_CONF_ERROR;
117 }
118 for (i = 1; i < cf->args->nelts; i++) {
119 slot = ngx_array_push(c->common.cache_peers);
120 if (slot == NULL) {
121 return NGX_CONF_ERROR;
122 }
123 *slot = value[i];
124 }
125 return NGX_CONF_OK;
126}
