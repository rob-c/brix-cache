1/*
2 * netpref.c — process-wide address-family preference (IPv6→IPv4 auto-downgrade).
3 *
4 * WHAT: A tiny, lock-free state machine that records whether the current process
5 * (e.g. one FUSE mount session) should still attempt IPv6 or has fallen
6 * back to IPv4-only. brix_tcp_connect consults it for the getaddrinfo
7 * family hint and trips it when it sees IPv6 fail where IPv4 then works.
8 * WHY: On a dual-stack host with a broken IPv6 path, every fresh socket
9 * (pool slot, per-file conn, async stream, each reconnect) would otherwise
10 * pay a full connect timeout on the dead v6 address before falling back to
11 * v4 — repeatedly, for the life of the mount. Once we have positive proof
12 * the v6 path is black-holed AND v4 works, we demote the whole session to
13 * v4-only so the resolver stops returning v6 records entirely, and we say
14 * so exactly once. The downgrade is silent to the data path (transfers
15 * just keep working) but visible to the operator in the log.
16 * HOW: Three relaxed atomics (no ordering dependency between them): a sticky
17 * "demoted" flag, a one-shot "logged" guard, and a lazily-resolved
18 * "enabled" toggle (off via XRDC_NO_IPV6_FALLBACK for IPv6-only sites or
19 * debugging). State is global by design — the preference is a property of
20 * the host's network, not of any one connection — and accessed only through
21 * these functions, never as a bare extern.
22 *
23 * SAFETY: Demotion is never speculative. It happens only on the connect path's
24 * unambiguous evidence (v6 candidate failed, v4 candidate then succeeded),
25 * so an IPv6-only or healthy-IPv6 host never demotes: there is no working
26 * v4 to succeed after a v6 failure, and a healthy v6 never fails.
27 *
28 * Clean-room: depends on the C library and <stdatomic.h> only.
29 */

39/* 0 = still trying both families; 1 = demoted to IPv4-only for this process. */
40static atomic_int g_demoted = 0;
41/* One-shot guard so concurrent pool connections log the downgrade only once. */
42static atomic_int g_logged = 0;
43/* One-shot guard for the self-heal "reverting to dual-stack" message. */
44static atomic_int g_revert_logged = 0;
45/* Lazily resolved: -1 unknown, 0 disabled (opt-out), 1 enabled (default). */
46static atomic_int g_enabled = -1;
47
48/*
49 * Resolve (once) whether auto-downgrade is enabled. Default ON; an operator pins
50 * it OFF with XRDC_NO_IPV6_FALLBACK to keep retrying IPv6 forever — wanted on an
51 * IPv6-only deployment that must never silently shed v6, or when debugging the
52 * v6 path itself. The first caller wins the race; the value never changes after.
53 */
54static int
55fallback_enabled(void)
56{
57 int e = atomic_load_explicit(&g_enabled, memory_order_relaxed);
58 if (e >= 0) {
59 return e;
60 }
61 e = (getenv("XRDC_NO_IPV6_FALLBACK") != NULL) ? 0 : 1;
62 atomic_store_explicit(&g_enabled, e, memory_order_relaxed);
63 return e;
64}

85/*
86 * Commit the sticky IPv4-only demotion and log the cause exactly once for the
87 * life of the process — even if several pool/stream connections trip the
88 * fallback at the same moment. `cause` is the situation-specific lead-in; the
89 * shared tail names the consequence and the opt-out.
90 */
91static void
92do_demote(const char *cause)
93{
94 atomic_store_explicit(&g_demoted, 1, memory_order_relaxed);
95 if (atomic_exchange_explicit(&g_logged, 1, memory_order_relaxed) == 0) {
96 fprintf(stderr,
97 "xrootd: %s; downgrading this session to IPv4-only for all "
98 "further connections (set XRDC_NO_IPV6_FALLBACK=1 to keep "
99 "trying IPv6)\n",
100 cause);
101 }
102}

104void
105brix_netpref_demote_ipv6(const char *host)
106{
107 if (!fallback_enabled()) {
108 return; /* opt-out: keep attempting IPv6 on every connection */
109 }
110 char cause[320];
111 snprintf(cause, sizeof(cause), "IPv6 connect to %s failed but IPv4 succeeded",
112 (host != NULL && host[0] != '\0') ? host : "the server");
113 do_demote(cause);
114}

116void
117brix_netpref_note_wire_error(int family)
118{
119 if (family != AF_INET6 || !fallback_enabled()) {
120 return; /* only an over-the-wire IPv6 failure is evidence of bad v6 */
121 }
122 if (atomic_load_explicit(&g_demoted, memory_order_relaxed)) {
123 return; /* already IPv4-only */
124 }
125 /* An established IPv6 connection failed mid-flight (reset / timeout / EOF):
126 * pin the rest of the session to IPv4 so the reconnect skips v6 entirely and
127 * pays no v6 connect timeout. Self-healing — see brix_netpref_undo_demote:
128 * if IPv4 then turns out to be unavailable (an IPv6-only host), the connect
129 * path reverts so the mount still comes back up. */
130 do_demote("an IPv6 connection to the server failed over the wire");
131}

133void
134brix_netpref_undo_demote(const char *why)
135{
136 if (!atomic_load_explicit(&g_demoted, memory_order_relaxed)) {
137 return;
138 }
139 atomic_store_explicit(&g_demoted, 0, memory_order_relaxed);
140 if (atomic_exchange_explicit(&g_revert_logged, 1, memory_order_relaxed) == 0) {
141 fprintf(stderr,
142 "xrootd: %s; reverting to dual-stack (IPv4+IPv6) connection "
143 "attempts\n",
144 (why != NULL && why[0] != '\0') ? why
145 : "the IPv4-only path did not work");
146 }
