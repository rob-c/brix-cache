1/*
2 * Fattr dispatcher — routes kXR_fattr requests to the appropriate sub-handler.
3 * The fattr operation has four sub-codes: get (read attributes), set (write
4 * attributes), del (delete attributes), and list (enumerate all attributes).
5 * This function validates parameters, resolves path or file handle, checks
6 * authorization, then dispatches to the specific handler based on sub-code.
7 */
8
9#include "ngx_brix_fattr.h"
10#include "fs/vfs/vfs.h" /* confinement check via the VFS seam */
11
12#include <string.h>
13#include "core/compat/alloc_guard.h"
14
15/*
16 *
17 * WHAT: Validates the fattr request header, determines whether it targets an open
18 * file handle or a filesystem path, enforces read-only and auth gating, then
19 * splits the payload into the name-vector (nvec) and value-vector (vvec) and
20 * dispatches to fattr_get/set/del/list.
21 *
22 * WHY: The fattr wire frame is overloaded across four sub-codes whose argument
23 * layout differs (list takes no nvec; set carries an extra vvec) and whose
24 * target can be either an fhandle or an inline path. Centralising the
25 * parsing/validation here keeps each sub-handler free of framing concerns.
26 *
27 * HOW: A three-way branch on the payload shape selects the target (see inline
28 * notes): (a) no payload → fhandle in req->fhandle (list only); (b) payload
29 * begins with a 0 byte → fhandle followed by args; (c) otherwise → leading
30 * NUL-terminated path followed by args. Path targets are auth-gated and
31 * confined-open verified. The args region is then parsed into attrs[]: nvec
32 * first, vvec is whatever follows it. close_fd guards any fd we own. */
33ngx_int_t
34brix_handle_fattr(brix_ctx_t *ctx, ngx_connection_t *c,
35 ngx_stream_brix_srv_conf_t *conf)
36{
37 /* hdr_buf aliases the fixed request header; payload/cur_dlen hold the body. */
38 xrdw_fattr_req_t req;
39 int subcode;
40 int numattr;
41 int options;
42 char full_path[PATH_MAX];
43 char pathbuf[BRIX_MAX_PATH + 1];
44 const char *path = NULL;
45 int fd = -1;
46 int close_fd = 0;
47 u_char *args_buf = NULL;
48 size_t args_len = 0;
49 /* One VFS ctx for the whole request: in path mode it carries the resolved
50 * confined path the xattr ops act on; in fd mode it carries only proto/log
51 * for metric attribution (the fd is the target). Lives for the handler
52 * calls below. */
53 brix_vfs_ctx_t vctx;
54
55 xrdw_fattr_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
56 subcode = req.subcode;
57 numattr = req.numattr;
58 options = req.options;
59
60 if (subcode > kXR_fattrMaxSC) {
61 return brix_send_error(ctx, c, kXR_ArgInvalid,
62 "fattr: invalid subcode");
63 }
64 /* numattr rules differ by sub-code: list enumerates everything (must be 0);
65 * get/set/del operate on an explicit name vector (1..kXR_faMaxVars). */
66 if (subcode == kXR_fattrList) {
67 if (numattr != 0) {
68 return brix_send_error(ctx, c, kXR_ArgInvalid,
69 "fattr list: numattr must be 0");
70 }
71 } else if (numattr == 0 || numattr > kXR_faMaxVars) {
72 return brix_send_error(ctx, c, kXR_ArgInvalid,
73 "fattr: invalid numattr");
74 }
75
76 /* Mutating sub-codes are blocked up-front on read-only servers (invariant:
77 * allow_write is checked globally before any per-path/token scope). */
78 if ((subcode == kXR_fattrSet || subcode == kXR_fattrDel) &&
79 !conf->common.allow_write)
80 {
81 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
82 return brix_send_error(ctx, c, kXR_fsReadOnly,
83 "fattr: server is read-only");
84 }
85
86 /*
87 * Target selection (three mutually exclusive payload shapes):
88 * (a) cur_dlen == 0 → no body: the target is the fhandle in the
89 * request header. Only valid for fattrList.
90 * (b) payload[0] == 0 → leading 0 byte marks an fhandle target; the
91 * file index is still req->fhandle[0] and any
92 * args follow at payload+1.
93 * (c) otherwise → payload starts with a NUL-terminated path;
94 * args (if any) follow the path's terminator.
95 */
96 if (ctx->recv.cur_dlen == 0) {
97 /* (a) Empty body is only meaningful for list; others need an nvec. */
98 if (subcode != kXR_fattrList) {
99 return brix_send_error(ctx, c, kXR_ArgMissing,
100 "fattr: missing arguments");
101 }
102
103 {
104 /* fhandle index is one wire byte; bounds + open-state checked. */
105 int idx = (int) (unsigned char) req.fhandle[0];
106
107 if (idx < 0 || idx >= BRIX_MAX_FILES || ctx->files[idx].fd < 0) {
108 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
109 return brix_send_error(ctx, c, kXR_FileNotOpen,
110 "fattr: invalid file handle");
111 }
112 fd = ctx->files[idx].fd;
113 /* A driver-backed handle (object store: ceph/s3/…) keeps its xattrs on
114 * the BACKEND object, not the local placeholder fd — a raw fsetxattr on
115 * that fd is lost. Route by the handle's resolved path through the driver
116 * (path mode) so set/get/list hit the same store the bytes live in. */
117 if (ctx->files[idx].sd_obj.driver != NULL
118 && ctx->files[idx].path != NULL)
119 {
120 path = ctx->files[idx].path;
121 }
122 }
123 /* fd target: vctx carries proto/log only (path NULL → fd mode); a driver-
124 * backed handle set its resolved path above → driver path mode. */
125 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
126 conf->common.root_canon, NULL, conf->common.allow_write,
127 0 /* is_tls */, NULL, path);
128
129 } else if (ctx->recv.payload != NULL && ctx->recv.payload[0] == 0) {
130 /* (b) fhandle-targeted request with a leading 0 marker byte. */
131 int idx = (int) (unsigned char) req.fhandle[0];
132
133 if (idx < 0 || idx >= BRIX_MAX_FILES || ctx->files[idx].fd < 0) {
134 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
135 return brix_send_error(ctx, c, kXR_FileNotOpen,
136 "fattr: invalid file handle");
137 }
138
139 fd = ctx->files[idx].fd;
140 /* Driver-backed handle: route xattrs by the resolved path through the driver
141 * (path mode), not the local placeholder fd (see branch (a)). */
142 if (ctx->files[idx].sd_obj.driver != NULL
143 && ctx->files[idx].path != NULL)
144 {
145 path = ctx->files[idx].path;
146 }
147 /* fd target: vctx carries proto/log only (path NULL → fd mode); a driver-
148 * backed handle set its resolved path above → driver path mode. */
149 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
150 conf->common.root_canon, NULL, conf->common.allow_write,
151 0 /* is_tls */, NULL, path);
152 /* Everything after the marker byte is the nvec/vvec args region. */
153 if (ctx->recv.cur_dlen > 1) {
154 args_buf = ctx->recv.payload + 1;
155 args_len = ctx->recv.cur_dlen - 1;
156 }
157
158 } else {
159 /* (c) Path-targeted request: payload = "<path>\0[args...]". */
160 size_t path_wire_len;
161 size_t path_payload_len;
162
163 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
164 return brix_send_error(ctx, c, kXR_ArgMissing,
165 "fattr: missing path");
166 }
167
168 /* Bounded scan for the path's NUL; path_payload_len includes it so the
169 * args region begins exactly at payload + path_payload_len below. */
170 path_wire_len = strnlen((char *) ctx->recv.payload, ctx->recv.cur_dlen);
171 path_payload_len = path_wire_len + 1;
172
173 if (!brix_extract_path(c->log, ctx->recv.payload, path_payload_len,
174 pathbuf, sizeof(pathbuf), 1)) {
175 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
176 return brix_send_error(ctx, c, kXR_ArgInvalid,
177 "fattr: invalid path");
178 }
179 /* Resolve the client path beneath the export root → full_path. */
180 brix_beneath_full_path(conf->common.root_canon, pathbuf,
181 full_path, sizeof(full_path));
182 {
183 /* set/del need UPDATE rights; get/list need only READ. */
184 int need_write = (subcode == kXR_fattrSet
185 || subcode == kXR_fattrDel) ? 1 : 0;
186 uint32_t auth_level = need_write ? BRIX_AUTH_UPDATE
187 : BRIX_AUTH_READ;
188 if (brix_auth_gate(ctx, c, BRIX_OP_FATTR, "FATTR",
189 pathbuf, full_path, conf,
190 auth_level, need_write) != NGX_OK) {
191 /* auth_gate already queued the error; ctx->write_rc is its rc. */
192 return ctx->write_rc;
193 }
194 }
195
196 {
197 /* Confinement check via the VFS: a confined no-follow stat verifies
198 * the path is within the export root (and exists) without a raw
199 * open_beneath. The fattr ops then operate by path (through the same
200 * vctx) so a directory list can recurse. */
201 brix_vfs_stat_t vst;
202
203 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,
204 conf->common.root_canon, NULL, conf->common.allow_write,
205 0 /* is_tls */, NULL, full_path);
206 if (brix_vfs_stat(&vctx, &vst) != NGX_OK) {
207 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
208 return brix_send_error(ctx, c, brix_kxr_from_errno(errno),
209 "fattr: cannot open path");
210 }
211 path = full_path;
212 }
213 /* Args (nvec/vvec) are whatever bytes follow the path's NUL. */
214 if (path_payload_len < ctx->recv.cur_dlen) {
215 args_buf = ctx->recv.payload + path_payload_len;
216 args_len = ctx->recv.cur_dlen - path_payload_len;
217 }
218 }
219
220 /* list takes no nvec — dispatch directly. */
221 if (subcode == kXR_fattrList) {
222 ngx_int_t rc = fattr_list(ctx, c, &vctx, path, fd, options);
223 if (close_fd) { close(fd); }
224 return rc;
225 }
226
227 /* get/set/del all require a name vector. */
228 if (args_buf == NULL || args_len == 0) {
229 return brix_send_error(ctx, c, kXR_ArgMissing,
230 "fattr: missing nvec");
231 }
232
233 {
234 u_char *nvec_copy;
235 brix_fattr_entry_t attrs[kXR_faMaxVars];
236 ssize_t nvec_used;
237 size_t nvec_len;
238 u_char *vvec_buf;
239 size_t vvec_len;
240
241 /* Copy the args region so fattr_parse_nvec can record rc_ptr slots that
242 * point into a buffer we own (the wire payload is overwritten in place
243 * with status codes when building the response). */
244 BRIX_PALLOC_OR_RETURN(nvec_copy, c->pool, args_len, brix_send_error(ctx, c, kXR_NoMemory, "out of memory"));
245 ngx_memcpy(nvec_copy, args_buf, args_len);
246
247 /* Parse the name vector; returns the byte count it consumed. */
248 nvec_used = fattr_parse_nvec(c->log, nvec_copy, args_len,
249 numattr, attrs);
250 if (nvec_used < 0) {
251 return brix_send_error(ctx, c, kXR_ArgInvalid,
252 "fattr: malformed nvec");
253 }
254
255 /* Whatever follows the nvec is the value vector (used by set only). */
256 nvec_len = (size_t) nvec_used;
257 vvec_buf = nvec_copy + nvec_len;
258 vvec_len = args_len - nvec_len;
259
260 {
261 ngx_int_t dispatch_rc;
262 switch (subcode) {
263 case kXR_fattrGet:
264 dispatch_rc = fattr_get(ctx, c, &vctx, path, fd,
265 nvec_copy, nvec_len, numattr, attrs);
266 break;
267 case kXR_fattrSet:
268 dispatch_rc = fattr_set(ctx, c, &vctx, path, fd, options,
269 nvec_copy, nvec_len, vvec_buf, vvec_len,
270 numattr, attrs);
271 break;
272 case kXR_fattrDel:
273 dispatch_rc = fattr_del(ctx, c, &vctx, path, fd,
274 nvec_copy, nvec_len, numattr, attrs);
275 break;
276 default:
277 dispatch_rc = brix_send_error(ctx, c, kXR_Unsupported,
278 "fattr: unknown subcode");
279 break;
280 }
281 if (close_fd) { close(fd); }
282 return dispatch_rc;
283 }
284 }
285
286 return brix_send_error(ctx, c, kXR_Unsupported,
287 "fattr: unknown subcode");
288}
