3 * The fattr operation has four sub-codes: get (read attributes), set (write
4 * attributes), del (delete attributes), and list (enumerate all attributes).
5 * This function validates parameters, resolves path or file handle, checks
6 * authorization, then dispatches to the specific handler based on sub-code.

16/*
17 * WHAT: Per-request dispatch state threaded through the fattr helpers: the
18 * decoded sub-code/numattr/options, the resolved target (path or fd),
19 * the raw nvec/vvec args region, the VFS context the xattr ops act
20 * through, and the error rc a failing helper queued.
21 *
22 * WHY: The four sub-codes share one wire frame whose target and argument
23 * layout differ; bundling the resolved pieces in one struct lets each
24 * helper stay under the 5-param gate while keeping data flow explicit.
25 *
26 * HOW: Declared on the dispatcher's stack (full_path/pathbuf live here so
27 * `path` may point into them); helpers fill it in and return
28 * NGX_OK/NGX_ERROR, recording the already-queued send-error rc in
29 * err_rc so the dispatcher can return it verbatim. */
30typedef struct {
31 int subcode;
32 int numattr;
33 int options;
34 const char *path; /* NULL → fd-mode target */
35 int fd;
36 int close_fd; /* dispatcher owns fd → close after use */
37 u_char *args_buf; /* nvec/vvec region within the payload */
38 size_t args_len;
39 ngx_int_t err_rc; /* rc of the error a failing helper queued */
40 char full_path[PATH_MAX];
41 char pathbuf[BRIX_MAX_PATH + 1];
42 /* One VFS ctx for the whole request: in path mode it carries the resolved
43 * confined path the xattr ops act on; in fd mode it carries only proto/log
44 * for metric attribution (the fd is the target). Lives for the handler
45 * calls below. */
46 brix_vfs_ctx_t vctx;
47} fattr_state_t;

85/*
86 * WHAT: Validates the decoded request header fields: sub-code range, the
87 * per-sub-code numattr rules, and the global read-only gate for the
88 * mutating sub-codes.
89 *
90 * WHY: All four sub-codes share these checks; performing them before target
91 * resolution rejects malformed frames without touching the filesystem.
92 *
93 * HOW: Early-returns a queued kXR error (via fattr_fail) on each violation;
94 * list must carry numattr == 0, get/set/del need 1..kXR_faMaxVars, and
95 * set/del are blocked up-front on read-only servers (invariant:
96 * allow_write is checked globally before any per-path/token scope). */
97static ngx_int_t
98fattr_check_request(brix_ctx_t *ctx, ngx_connection_t *c,
99 ngx_stream_brix_srv_conf_t *conf, fattr_state_t *st)
100{
101 if (st->subcode > kXR_fattrMaxSC) {
102 return fattr_fail(st, brix_send_error(ctx, c, kXR_ArgInvalid,
103 "fattr: invalid subcode"));
104 }
105 /* numattr rules differ by sub-code: list enumerates everything (must be 0);
106 * get/set/del operate on an explicit name vector (1..kXR_faMaxVars). */
107 if (st->subcode == kXR_fattrList) {
108 if (st->numattr != 0) {
109 return fattr_fail(st, brix_send_error(ctx, c, kXR_ArgInvalid,
110 "fattr list: numattr must be 0"));
111 }
112 } else if (st->numattr == 0 || st->numattr > kXR_faMaxVars) {
113 return fattr_fail(st, brix_send_error(ctx, c, kXR_ArgInvalid,
114 "fattr: invalid numattr"));
115 }
116
117 if ((st->subcode == kXR_fattrSet || st->subcode == kXR_fattrDel) &&
118 !conf->common.allow_write)
119 {
120 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
121 return fattr_fail(st, brix_send_error(ctx, c, kXR_fsReadOnly,
122 "fattr: server is read-only"));
123 }
124
125 return NGX_OK;
126}

128/*
129 * WHAT: Initialises the request's VFS context from the resolved target and
130 * binds the per-user backend credential and delegated identity.
131 *
132 * WHY: All three target shapes end with this identical init/bind triple;
133 * centralising it keeps the credential threading uniform (a driver-path
134 * fattr on a remote-backed export must present the requesting user's
135 * cred).
136 *

153/*
154 * WHAT: Resolves an fhandle-targeted request: bounds/open-state checks the
155 * wire file-handle index and records the target fd (and, for
156 * driver-backed handles, the resolved path).
157 *
158 * WHY: Payload shapes (a) and (b) share this lookup verbatim; a driver-backed
159 * handle (object store: ceph/s3/…) keeps its xattrs on the BACKEND
160 * object, not the local placeholder fd — a raw fsetxattr on that fd is
161 * lost, so such handles are routed by path through the driver.
162 *
163 * HOW: fh0 is the one wire byte of the fhandle; validates the index against
164 * BRIX_MAX_FILES and the open-fd table, then flips to path mode when the
165 * handle carries a driver + resolved path (set/get/list then hit the
166 * same store the bytes live in). */
167static ngx_int_t
168fattr_fhandle_target(brix_ctx_t *ctx, ngx_connection_t *c,
169 fattr_state_t *st, unsigned char fh0)
170{
171 /* fhandle index is one wire byte; bounds + open-state checked. */
172 int idx = (int) fh0;
173
174 if (idx < 0 || idx >= BRIX_MAX_FILES || ctx->files[idx].fd < 0) {
175 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
176 return fattr_fail(st, brix_send_error(ctx, c, kXR_FileNotOpen,
177 "fattr: invalid file handle"));
178 }
179 st->fd = ctx->files[idx].fd;
180 /* Driver-backed handle: route xattrs by the resolved path through the
181 * driver (path mode), not the local placeholder fd. */
182 if (ctx->files[idx].sd_obj.driver != NULL
183 && ctx->files[idx].path != NULL)
184 {
185 st->path = ctx->files[idx].path;
186 }
187
188 return NGX_OK;
189}

191/*
192 * WHAT: Resolves a path-targeted request (payload = "<path>\0[args...]"):
193 * extracts and confines the client path, auth-gates it, verifies it via
194 * a confined VFS stat, and records the trailing args region.
195 *
196 * WHY: The path shape is the only target that needs auth gating and
197 * confinement here (fhandle targets were gated at open); isolating it
198 * keeps the framing-only dispatcher free of filesystem concerns.
199 *
200 * HOW: Bounded strnlen finds the path's NUL (path_payload_len includes it so
201 * the args region begins exactly at payload + path_payload_len);
202 * brix_extract_path validates, brix_beneath_full_path resolves beneath
203 * the export root, brix_auth_gate enforces READ for get/list and UPDATE
204 * for set/del, and a confined no-follow brix_vfs_stat verifies the path
205 * is within the export root (and exists) without a raw open_beneath.
206 * The fattr ops then operate by path (through the same vctx) so a
207 * directory list can recurse. */
208static ngx_int_t
209fattr_path_target(brix_ctx_t *ctx, ngx_connection_t *c,
210 ngx_stream_brix_srv_conf_t *conf, fattr_state_t *st)
211{
212 size_t path_wire_len;
213 size_t path_payload_len;
214
215 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
216 return fattr_fail(st, brix_send_error(ctx, c, kXR_ArgMissing,
217 "fattr: missing path"));
218 }
219
220 /* Bounded scan for the path's NUL; path_payload_len includes it so the
221 * args region begins exactly at payload + path_payload_len below. */
222 path_wire_len = strnlen((char *) ctx->recv.payload, ctx->recv.cur_dlen);
223 path_payload_len = path_wire_len + 1;
224
225 if (!brix_extract_path(c->log, ctx->recv.payload, path_payload_len,
226 st->pathbuf, sizeof(st->pathbuf), 1)) {
227 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
228 return fattr_fail(st, brix_send_error(ctx, c, kXR_ArgInvalid,
229 "fattr: invalid path"));
230 }
231 /* Resolve the client path beneath the export root → full_path.
232 * phase72-fp: pathbuf is the request path, full_path the output buf. */
233 brix_beneath_full_path(conf->common.root_canon, st->pathbuf, /* NOLINT(readability-suspicious-call-argument) */
234 st->full_path, sizeof(st->full_path));
235 {
236 /* set/del need UPDATE rights; get/list need only READ. */
237 int need_write = (st->subcode == kXR_fattrSet
238 || st->subcode == kXR_fattrDel) ? 1 : 0;
239 uint32_t auth_level = need_write ? BRIX_AUTH_UPDATE
240 : BRIX_AUTH_READ;
241 if (brix_auth_gate(ctx, c, BRIX_OP_FATTR, "FATTR",
242 st->pathbuf, st->full_path, conf,
243 auth_level, need_write) != NGX_OK) {
244 /* auth_gate already queued the error; ctx->write_rc is its rc. */
245 return fattr_fail(st, ctx->write_rc);
246 }
247 }
248
249 {
250 /* Confinement check via the VFS: a confined no-follow stat verifies
251 * the path is within the export root (and exists). */
252 brix_vfs_stat_t vst;
253
254 st->path = st->full_path;
255 fattr_bind_vctx(ctx, c, conf, st);
256 if (brix_vfs_stat(&st->vctx, &vst) != NGX_OK) {
257 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
258 return fattr_fail(st, brix_send_error(ctx, c,
259 brix_kxr_from_errno(errno),
260 "fattr: cannot open path"));
261 }
262 }
263 /* Args (nvec/vvec) are whatever bytes follow the path's NUL. */
264 if (path_payload_len < ctx->recv.cur_dlen) {
265 st->args_buf = ctx->recv.payload + path_payload_len;
266 st->args_len = ctx->recv.cur_dlen - path_payload_len;
267 }
268
269 return NGX_OK;
270}

272/*
273 * WHAT: Selects the request target from the payload shape and fills the
274 * dispatch state with the resolved fd/path, bound vctx, and args region.
275 *
276 * WHY: The fattr wire frame overloads three mutually exclusive payload shapes;
277 * resolving them in one place keeps the sub-handlers free of framing
278 * concerns.
279 *
280 * HOW: Three-way branch: (a) cur_dlen == 0 → no body: the target is the
281 * fhandle in the request header, only valid for fattrList (others need
282 * an nvec); (b) payload[0] == 0 → leading 0 byte marks an fhandle
283 * target, the file index is still req->fhandle[0] and any args follow
284 * at payload+1; (c) otherwise → payload starts with a NUL-terminated
285 * path, args (if any) follow the path's terminator
286 * (fattr_path_target binds the vctx itself, after auth). */
287static ngx_int_t
288fattr_select_target(brix_ctx_t *ctx, ngx_connection_t *c,
289 ngx_stream_brix_srv_conf_t *conf, fattr_state_t *st,
290 const xrdw_fattr_req_t *req)
291{
292 if (ctx->recv.cur_dlen == 0) {
293 /* (a) Empty body is only meaningful for list; others need an nvec. */
294 if (st->subcode != kXR_fattrList) {
295 return fattr_fail(st, brix_send_error(ctx, c, kXR_ArgMissing,
296 "fattr: missing arguments"));
297 }
298 if (fattr_fhandle_target(ctx, c, st, (unsigned char) req->fhandle[0])
299 != NGX_OK)
300 {
301 return NGX_ERROR;
302 }
303 fattr_bind_vctx(ctx, c, conf, st);
304 return NGX_OK;
305 }
306
307 if (ctx->recv.payload != NULL && ctx->recv.payload[0] == 0) {
308 /* (b) fhandle-targeted request with a leading 0 marker byte. */
309 if (fattr_fhandle_target(ctx, c, st, (unsigned char) req->fhandle[0])
310 != NGX_OK)
311 {
312 return NGX_ERROR;
313 }
314 fattr_bind_vctx(ctx, c, conf, st);
315 /* Everything after the marker byte is the nvec/vvec args region. */
316 if (ctx->recv.cur_dlen > 1) {
317 st->args_buf = ctx->recv.payload + 1;
318 st->args_len = ctx->recv.cur_dlen - 1;
319 }
320 return NGX_OK;
321 }
322
323 /* (c) Path-targeted request: payload = "<path>\0[args...]". */
324 return fattr_path_target(ctx, c, conf, st);
325}

327/*
328 * WHAT: Copies the args region and parses it into the name vector (attrs[]),
329 * splitting off whatever follows as the value vector.
330 *
331 * WHY: fattr_parse_nvec records rc_ptr slots that point into the buffer, and
332 * the wire payload is overwritten in place with status codes when
333 * building the response — so the parse must run over a copy we own.
334 *
335 * HOW: ngx_palloc + ngx_memcpy the args region, run fattr_parse_nvec (returns
336 * the byte count it consumed, negative on malformed input), then set
337 * vvec_buf/vvec_len to the remainder (used by set only). */
338static ngx_int_t
339fattr_decode_names(brix_ctx_t *ctx, ngx_connection_t *c,
340 fattr_state_t *st, brix_fattr_entry_t *attrs, fattr_names_t *names)
341{
342 ssize_t nvec_used;
343
344 BRIX_PALLOC_OR_RETURN(names->nvec_copy, c->pool, st->args_len,
345 fattr_fail(st, brix_send_error(ctx, c, kXR_NoMemory, "out of memory")));
346 ngx_memcpy(names->nvec_copy, st->args_buf, st->args_len);
347
348 /* Parse the name vector; returns the byte count it consumed. */
349 nvec_used = fattr_parse_nvec(c->log, names->nvec_copy, st->args_len,
350 st->numattr, attrs);
351 if (nvec_used < 0) {
352 return fattr_fail(st, brix_send_error(ctx, c, kXR_ArgInvalid,
353 "fattr: malformed nvec"));
354 }
355
356 /* Whatever follows the nvec is the value vector (used by set only). */
357 names->nvec_len = (size_t) nvec_used;
358 names->vvec_buf = names->nvec_copy + names->nvec_len;
359 names->vvec_len = st->args_len - names->nvec_len;
360
361 return NGX_OK;
362}

364/*
365 * WHAT: Decodes the name vector and dispatches the named sub-codes
366 * (get/set/del) to their handlers, closing any dispatcher-owned fd.
367 *
368 * WHY: The three named sub-codes share the nvec decode and the fd-cleanup
369 * tail; grouping them leaves the dispatcher a flat validate → resolve →
370 * dispatch sequence.
371 *
372 * HOW: fattr_decode_names produces the nvec/vvec split, then a switch on the
373 * sub-code calls fattr_get/set/del through the VFS xattr seam; an
374 * unknown sub-code (unreachable past fattr_check_request) answers
375 * kXR_Unsupported. close_fd guards any fd we own. */
376static ngx_int_t
377fattr_dispatch_named(brix_ctx_t *ctx, ngx_connection_t *c, fattr_state_t *st)
378{
379 brix_fattr_entry_t attrs[kXR_faMaxVars];
380 fattr_names_t names;
381 ngx_int_t dispatch_rc;
382
383 if (fattr_decode_names(ctx, c, st, attrs, &names) != NGX_OK) {
384 return st->err_rc;
385 }
386
387 switch (st->subcode) {
388 case kXR_fattrGet:
389 dispatch_rc = fattr_get(ctx, c, &st->vctx, st->path, st->fd,
390 names.nvec_copy, names.nvec_len,
391 st->numattr, attrs);
392 break;
393 case kXR_fattrSet:
394 dispatch_rc = fattr_set(ctx, c, &st->vctx, st->path, st->fd,
395 st->options,
396 names.nvec_copy, names.nvec_len,
397 names.vvec_buf, names.vvec_len,
398 st->numattr, attrs);
399 break;
400 case kXR_fattrDel:
401 dispatch_rc = fattr_del(ctx, c, &st->vctx, st->path, st->fd,
402 names.nvec_copy, names.nvec_len,
403 st->numattr, attrs);
404 break;
405 default:
406 dispatch_rc = brix_send_error(ctx, c, kXR_Unsupported,
407 "fattr: unknown subcode");
408 break;
409 }
410 if (st->close_fd) { close(st->fd); }
411 return dispatch_rc;
412}

414/*
415 * WHAT: Validates the fattr request header, determines whether it targets an
416 * open file handle or a filesystem path, enforces read-only and auth
417 * gating, then splits the payload into the name-vector (nvec) and
418 * value-vector (vvec) and dispatches to fattr_get/set/del/list.
419 *
420 * WHY: The fattr wire frame is overloaded across four sub-codes whose argument
421 * layout differs (list takes no nvec; set carries an extra vvec) and
422 * whose target can be either an fhandle or an inline path. Centralising
423 * the parsing/validation here keeps each sub-handler free of framing
424 * concerns.
425 *
426 * HOW: Flat early-return sequence over the fattr_state_t: fattr_check_request
427 * validates the header, fattr_select_target resolves the fhandle/path
428 * target (auth-gating and confined-verifying path targets) and binds
429 * the VFS context, then list dispatches directly (it takes no nvec)
430 * while get/set/del go through fattr_dispatch_named. */
431ngx_int_t
432brix_handle_fattr(brix_ctx_t *ctx, ngx_connection_t *c,
433 ngx_stream_brix_srv_conf_t *conf)
434{
435 /* hdr_buf aliases the fixed request header; payload/cur_dlen hold the body. */
436 xrdw_fattr_req_t req;
437 fattr_state_t st;
438
439 xrdw_fattr_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
440 st.subcode = req.subcode;
441 st.numattr = req.numattr;
442 st.options = req.options;
443 st.path = NULL;
444 st.fd = -1;
445 st.close_fd = 0;
446 st.args_buf = NULL;
447 st.args_len = 0;
448 st.err_rc = NGX_ERROR;
449
450 if (fattr_check_request(ctx, c, conf, &st) != NGX_OK) {
451 return st.err_rc;
452 }
453
454 if (fattr_select_target(ctx, c, conf, &st, &req) != NGX_OK) {
455 return st.err_rc;
456 }
457
458 /* list takes no nvec — dispatch directly. */
459 if (st.subcode == kXR_fattrList) {
460 ngx_int_t rc = fattr_list(ctx, c, &st.vctx, st.path, st.fd,
461 st.options);
462 if (st.close_fd) { close(st.fd); }
463 return rc;
464 }
465
466 /* get/set/del all require a name vector. */
467 if (st.args_buf == NULL || st.args_len == 0) {
468 return brix_send_error(ctx, c, kXR_ArgMissing,
469 "fattr: missing nvec");
470 }
471
472 return fattr_dispatch_named(ctx, c, &st);
473}
