11/* path starts with prefix (plain string compare, not regex). */
12static int
13admit_prefix_match(const char *path, const ngx_str_t *p)
14{
15 if (p->len == 0 || path == NULL) {
16 return 0;
17 }
18 if (strlen(path) < p->len) {
19 return 0;
20 }
21 return ngx_strncmp((u_char *) path, p->data, p->len) == 0;
22}

42brix_cache_admit_e
43brix_cache_admit(const brix_cache_admit_cfg_t *cfg, const char *path,
44 off_t size, int is_new)
45{
46 if (cfg == NULL || path == NULL) {
47 return BRIX_CACHE_DECLINE; /* fail-closed */
48 }
49
50 /* Size cap: an existing file over the limit DECLINEs unless an include regex
51 * matches; a not-yet-existing file (is_new) has no size to test. */
52 if (cfg->size_limit > 0 && !is_new && size > cfg->size_limit) {
53 if (cfg->include_regex == NULL
54 || regexec(cfg->include_regex, path, 0, NULL, 0) != 0)
55 {
56 return BRIX_CACHE_DECLINE;
57 }
58 }
59
60 /* Deny prefixes take precedence (blacklist). */
61 if (admit_any_prefix(path, cfg->deny_prefixes)) {
62 return BRIX_CACHE_DECLINE;
63 }
64
65 /* A configured allow list is a whitelist: a non-match DECLINEs. */
66 if (cfg->allow_prefixes != NULL && cfg->allow_prefixes->nelts > 0
67 && !admit_any_prefix(path, cfg->allow_prefixes))
68 {
69 return BRIX_CACHE_DECLINE;
70 }
71
72 return BRIX_CACHE_ADMIT;
73}
