17static const char *
18ckv_skip_ws(const char *s)
19{
20 while (*s != '\0' && isspace((unsigned char) *s)) {
21 s++;
22 }

26/* Length of s with trailing ASCII whitespace excluded. */
27static size_t
28ckv_trimmed_len(const char *s)
29{
30 size_t len = strlen(s);
31
32 while (len > 0 && isspace((unsigned char) s[len - 1])) {
33 len--;
34 }
35 return len;
36}

38storascan_cks_status
39storascan_cks_compare(const char *computed_hex, const char *stored_hex)
40{
41 const char *want;
42 const char *got;
43 size_t want_len;
44 size_t got_len;
45 size_t i;
46
47 if (stored_hex == NULL) {
48 return STORASCAN_CKS_MISSING;
49 }
50 want = ckv_skip_ws(stored_hex);
51 want_len = ckv_trimmed_len(want);
52 if (want_len == 0) {
53 return STORASCAN_CKS_MISSING;
54 }
55
56 got = (computed_hex != NULL) ? ckv_skip_ws(computed_hex) : "";
57 got_len = ckv_trimmed_len(got);
58
59 if (got_len != want_len) {
60 return STORASCAN_CKS_MISMATCH;
61 }
62 for (i = 0; i < want_len; i++) {
63 if (tolower((unsigned char) got[i]) != tolower((unsigned char) want[i])) {
64 return STORASCAN_CKS_MISMATCH;
65 }
66 }
67 return STORASCAN_CKS_MATCH;
68}

72double
73storascan_percentile_ms(const double *sorted_ms, size_t n, double pct)
74{
75 double rank;
76 size_t idx;
77
78 if (n == 0 || sorted_ms == NULL) {
79 return 0.0;
80 }
81 if (pct < 0.0) {
82 pct = 0.0;
83 }
84 if (pct > 100.0) {
85 pct = 100.0;
86 }
87
88 /* nearest-rank: rank = ceil(pct/100 * n), index = rank-1, clamped */
89 rank = ceil((pct / 100.0) * (double) n);
90 if (rank < 1.0) {
91 idx = 0;
92 } else {
93 idx = (size_t) rank - 1;
94 if (idx >= n) {
95 idx = n - 1;
96 }
97 }
98 return sorted_ms[idx];
99}

116void
117storascan_bench_compute(const double *lat_ms, size_t n,
118 uint64_t bytes, double elapsed_s,
119 storascan_bench_result *out)
120{
121 double *sorted = NULL;
122
123 out->ops = (uint64_t) n;
124 out->bytes = bytes;
125 out->elapsed_s = elapsed_s;
126
127 if (elapsed_s > 0.0) {
128 out->throughput_mibps = ((double) bytes / (double) (1u << 20)) / elapsed_s;
129 out->iops = (double) n / elapsed_s;
130 } else {
131 out->throughput_mibps = 0.0;
132 out->iops = 0.0;
133 }
134
135 out->p50_ms = out->p95_ms = out->p99_ms = 0.0;
136 if (n == 0 || lat_ms == NULL) {
137 return;
138 }
139
140 /* Sort a private copy so the caller's samples are left untouched. */
141 sorted = (double *) malloc(n * sizeof(*sorted));
142 if (sorted == NULL) {
143 return; /* percentiles stay 0 on OOM; totals above are still valid */
144 }
145 memcpy(sorted, lat_ms, n * sizeof(*sorted));
146 qsort(sorted, n, sizeof(*sorted), ckv_cmp_double);
147
148 out->p50_ms = storascan_percentile_ms(sorted, n, 50.0);
149 out->p95_ms = storascan_percentile_ms(sorted, n, 95.0);
150 out->p99_ms = storascan_percentile_ms(sorted, n, 99.0);
151
152 free(sorted);
153}
