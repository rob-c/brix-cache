1/*
2 * brixposix_preload.c — an LD_PRELOAD shim that routes POSIX reads of a configured
3 * path prefix to an XRootD root:// export via libbrix.
4 *
5 * WHAT: Interpose open/read/pread/lseek/close, the stat family (incl. statx and
6 * the LFS *64 variants) and access on libc. Any
7 * path under the prefix named by $BRIX_VMP is served from a remote XRootD
8 * server; every other path passes straight through to the real libc call.
9 * WHY: Legacy tools that only know POSIX paths (cat, md5sum, ls, analysis jobs)
10 * can read remote XRootD data with no recompile and NO libXrdCl/XrdPosix --
11 * just LD_PRELOAD=libbrixposix_preload.so BRIX_VMP=/xrd=root://host:port/.
12 * HOW: $BRIX_VMP = "<localprefix>=root://host[:port][/base]". A path that
13 * starts with <localprefix> is rewritten to the remote logical path and
14 * opened through a single lazily-connected libbrix session (one request in
15 * flight, mutex-guarded). Remote descriptors live in a shadow fd table at
16 * fds >= XFS_FD_BASE so read/lseek/close/fstat can tell them apart from
17 * real fds. Real libc symbols are resolved with dlsym(RTLD_NEXT) via the
18 * __typeof__-based REAL() helper (so each wrapper inherits libc's prototype).
19 *
20 * Scope (first cut): the READ path. Files opened for write under the prefix fall
21 * through to libc (a documented follow-up), as do fopen/mmap and the legacy
22 * __xstat() routing (modern glibc exports stat/lstat/fstat as real symbols, which
23 * we interpose directly).
24 *
25 * Clean-room: composes the public libbrix API + dlsym only; no XrdPosix code.
26 */

58static void
59parse_vmp(void)
60{
61 const char *vmp = getenv("BRIX_VMP");
62 const char *eq;
63 char url[512];
64 brix_status st;
65
66 if (vmp == NULL || vmp[0] == '\0') {
67 return;
68 }
69 eq = strchr(vmp, '=');
70 if (eq == NULL || (size_t) (eq - vmp) >= sizeof(g_prefix)) {
71 return;
72 }
73 memcpy(g_prefix, vmp, (size_t) (eq - vmp));
74 g_prefix[eq - vmp] = '\0';
75 g_prefix_len = strlen(g_prefix);
76 /* strip a trailing slash on the prefix so "/xrd" and "/xrd/" both work */
77 while (g_prefix_len > 1 && g_prefix[g_prefix_len - 1] == '/') {
78 g_prefix[--g_prefix_len] = '\0';
79 }
80
81 snprintf(url, sizeof(url), "%s", eq + 1);
82 brix_status_clear(&st);
83 if (brix_endpoint_parse(url, &g_url, &st) != 0) {
84 return;
85 }
86 if (g_url.path[0] != '\0' && strcmp(g_url.path, "/") != 0) {
87 size_t bl;
88 snprintf(g_base, sizeof(g_base), "%s", g_url.path);
89 bl = strlen(g_base);
90 while (bl > 1 && g_base[bl - 1] == '/') { /* drop trailing slash */
91 g_base[--bl] = '\0';
92 }
93 } else {
94 g_base[0] = '\0';
95 }
96 g_enabled = 1;
97}

99/* Map a local path to the remote logical path; 1 if under the prefix, else 0. */
100static int
101map_path(const char *path, char *out, size_t outsz)
102{
103 const char *rest;
104
105 pthread_once(&g_once, parse_vmp);
106 if (!g_enabled || path == NULL || path[0] != '/') {
107 return 0;
108 }
109 if (strncmp(path, g_prefix, g_prefix_len) != 0) {
110 return 0;
111 }
112 rest = path + g_prefix_len;
113 if (rest[0] != '\0' && rest[0] != '/') {
114 return 0; /* "/xrddata" must not match prefix "/xrd" */
115 }
116 if (rest[0] == '\0') {
117 rest = "/";
118 }
119 {
120 size_t bl = strlen(g_base);
121 size_t rl = strlen(rest);
122 if (bl + rl + 1 > outsz) {
123 return 0; /* too long for the remote path buffer: don't divert */
124 }
125 memcpy(out, g_base, bl);
126 memcpy(out + bl, rest, rl + 1); /* includes the NUL */
127 }
128 return 1;
129}

149#define XFS_FD_BASE 0x40000000
150#define XFS_FD_MAX 1024

152typedef struct {
153 int used;
154 brix_rfile f; /* resilient: reopens + resumes after a sever */
155 int64_t pos;
156 int64_t size;
157} xfs_slot;

159static xfs_slot g_slots[XFS_FD_MAX];
160
161static int
162slot_alloc(void)
163{
164 int i;
165 for (i = 0; i < XFS_FD_MAX; i++) {
166 if (!g_slots[i].used) {
167 memset(&g_slots[i], 0, sizeof(g_slots[i]));
168 g_slots[i].used = 1;
169 return i;
170 }
171 }
172 return -1;
173}

175static xfs_slot *
176slot_of(int fd)
177{
178 int i;
179 if (fd < XFS_FD_BASE) {
180 return NULL;
181 }
182 i = fd - XFS_FD_BASE;
183 if (i < 0 || i >= XFS_FD_MAX || !g_slots[i].used) {
184 return NULL;
185 }
186 return &g_slots[i];
187}

189static void
190fill_stat(const brix_statinfo *si, struct stat *stbuf)
191{
192 memset(stbuf, 0, sizeof(*stbuf));
193 if (si->flags & kXR_isDir) {
194 stbuf->st_mode = S_IFDIR | 0755;
195 stbuf->st_nlink = 2;
196 } else {
197 stbuf->st_mode = S_IFREG | 0644;
198 stbuf->st_nlink = 1;

204/* Resolve (once) the real libc symbol behind the wrapper we're standing in. The
205 * variable inherits libc's exact prototype via __typeof__(name). */
206#define REAL(name) \
207 static __typeof__(name) *real_##name = NULL; \
208 if (real_##name == NULL) { \
209 real_##name = (__typeof__(name) *) dlsym(RTLD_NEXT, #name); \
210 }

214static int
215remote_open(const char *remote)
216{
217 int slot;
218 xfs_slot *s;
219
220 pthread_mutex_lock(&g_lock);
221 if (ensure_conn() != 0) {
222 pthread_mutex_unlock(&g_lock);
223 errno = EIO;
224 return -1;
225 }
226 slot = slot_alloc();
227 if (slot < 0) {
228 pthread_mutex_unlock(&g_lock);
229 errno = EMFILE;
230 return -1;
231 }
232 s = &g_slots[slot];
233 {
234 brix_status st;
235 brix_statinfo si;
236 brix_status_clear(&st);
237 if (brix_rfile_open_read(&g_conn, remote, NULL, 0, -1, &s->f, &st) != 0) {
238 s->used = 0;
239 pthread_mutex_unlock(&g_lock);
240 errno = -brix_kxr_to_errno(&st);
241 return -1;
242 }
243 if (brix_stat(&g_conn, remote, &si, &st) == 0) {
244 s->size = si.size;
245 }
246 }
247 pthread_mutex_unlock(&g_lock);
248 return XFS_FD_BASE + slot;
249}

298ssize_t
299read(int fd, void *buf, size_t count)
300{
301 xfs_slot *s;
302 REAL(read);
303
304 s = slot_of(fd);
305 if (s == NULL) {
306 return real_read(fd, buf, count);
307 }
308 {
309 brix_status st;
310 ssize_t r;
311 pthread_mutex_lock(&g_lock);
312 brix_status_clear(&st);
313 r = brix_rfile_pread(&s->f, s->pos, buf, count, &st);
314 if (r > 0) {
315 s->pos += r;
316 }
317 pthread_mutex_unlock(&g_lock);
318 if (r < 0) {
319 errno = -brix_kxr_to_errno(&st);
320 }
321 return r;
322 }
323}

352off_t
353lseek(int fd, off_t offset, int whence)
354{
355 xfs_slot *s;
356 REAL(lseek);
357
358 s = slot_of(fd);
359 if (s == NULL) {
360 return real_lseek(fd, offset, whence);
361 }
362 switch (whence) {
363 case SEEK_SET: s->pos = offset; break;
364 case SEEK_CUR: s->pos += offset; break;
365 case SEEK_END: s->pos = s->size + offset; break;
366 default: errno = EINVAL; return (off_t) -1;
367 }
368 return (off_t) s->pos;

540/*
541 * statx() is what modern coreutils (ls, stat, find, du) actually call. Without
542 * interposing it, those tools would statx the real (absent) local path and
543 * ENOENT. We fill the common fields (type/mode/nlink/size/mtime); the caller's
544 * requested `mask` is satisfied for what XRootD can report.
545 */
546int
547statx(int dirfd, const char *path, int flags, unsigned int mask,
548 struct statx *stxbuf)
549{
550 char remote[XRDC_PATH_MAX];
551 brix_status st;
552 brix_statinfo si;
553 int rc;
554 REAL(statx);
555
556 (void) mask;
557 if (path[0] != '/' || !map_path(path, remote, sizeof(remote))) {
558 return real_statx(dirfd, path, flags, mask, stxbuf);
559 }
560 pthread_mutex_lock(&g_lock);
561 if (ensure_conn() != 0) {
562 pthread_mutex_unlock(&g_lock);
563 errno = EIO;
564 return -1;
565 }
566 brix_status_clear(&st);
567 rc = brix_stat(&g_conn, remote, &si, &st);
568 pthread_mutex_unlock(&g_lock);
569 if (rc != 0) {
570 errno = -brix_kxr_to_errno(&st);
571 return -1;
572 }
573 memset(stxbuf, 0, sizeof(*stxbuf));
574 stxbuf->stx_mask = STATX_TYPE | STATX_MODE | STATX_NLINK | STATX_SIZE
575 | STATX_MTIME;
576 stxbuf->stx_blksize = 4096;
577 if (si.flags & kXR_isDir) {
578 stxbuf->stx_mode = S_IFDIR | 0755;
579 stxbuf->stx_nlink = 2;
580 } else {
581 stxbuf->stx_mode = S_IFREG | 0644;
582 stxbuf->stx_nlink = 1;
583 stxbuf->stx_size = (uint64_t) si.size;
584 stxbuf->stx_blocks = (uint64_t) ((si.size + 511) / 512);
585 }
586 stxbuf->stx_mtime.tv_sec = si.mtime;
587 stxbuf->stx_atime.tv_sec = si.mtime;
588 stxbuf->stx_ctime.tv_sec = si.mtime;
589 return 0;
590}
