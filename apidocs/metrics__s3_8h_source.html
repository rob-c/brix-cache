65/*
66 * Per-process S3-compatible API metrics. Same atomicity rules as WebDAV.
67 * Key design note: S3 bucket names, object keys, access key IDs, and principals
68 * MUST NOT become Prometheus label values — only the coarse method/status/auth
69 * result codes are safe to export as labels.
70 */
71typedef struct {
72 ngx_atomic_t requests_total[BRIX_S3_NMETHODS]; /* requests by S3 method */
73 ngx_atomic_t responses_total[BRIX_S3_NMETHODS][BRIX_HTTP_NSTATUS]; /* responses by method × status class */
74 ngx_atomic_t auth_total[BRIX_S3_NAUTH_RESULTS]; /* SigV4 auth outcomes */
75
76 ngx_atomic_t bytes_rx_total; /* bytes received (PutObject body) */
77 ngx_atomic_t bytes_tx_total; /* bytes sent (GetObject body) */
78
79 /* Per-IP-version bandwidth — avoids high-cardinality label explosion. */
80 ngx_atomic_t bytes_rx_ipv4_total;
81 ngx_atomic_t bytes_tx_ipv4_total;
82 ngx_atomic_t bytes_rx_ipv6_total;
83 ngx_atomic_t bytes_tx_ipv6_total;
84
85 ngx_atomic_t range_total[BRIX_S3_NRANGE_RESULTS]; /* Range request outcomes */
86 ngx_atomic_t put_body_total[BRIX_S3_NPUT_MODES]; /* PutObject receive mode */
87 ngx_atomic_t events_total[BRIX_S3_NEVENTS]; /* misc error/event counts */
88
89 ngx_atomic_t list_contents_total; /* objects returned by ListObjects */
90 ngx_atomic_t list_common_prefixes_total; /* common-prefix groups returned */
91 ngx_atomic_t list_truncated_total; /* ListObjects responses truncated */
92} ngx_brix_s3_metrics_t;
