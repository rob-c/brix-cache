1/*
2 * sd_stage.c - the generic write-stage decorator (section 12.2). See header.
3 *
4 * The decorator forwards every read / namespace / xattr / dir op to the wrapped
5 * `source` (open returns the source's own object, so read byte-I/O bypasses the
6 * decorator) and implements only the staged-write path: a staged upload lands on
7 * the STAGE STORE (`store`) and is flushed to the source on commit through the one
8 * staging engine (brix_stage_run_inline FLUSH). A posix stage store is
9 * byte-equivalent to phase-63's local-temp promote.
10 */

33/* Write-BACK object state (Option A / §12.2): a random-access write open lands on
34 * the STAGE STORE as a normal writable object; pwrite buffers there, and fsync/close
35 * flush the whole object to the backend through the one staging engine. This lets
36 * root:// kXR_write (direct pwrite at arbitrary offsets) use the SAME stage mechanism
37 * as the staged (HTTP PUT) path, so writethrough_flush's bespoke loop is retired. */
38typedef struct {
39 sd_stage_inst_state *is; /* back-ref: source / store / policy */
40 char key[PATH_MAX]; /* export-relative object key */
41 brix_sd_obj_t store_obj; /* the writable stage-store object */
42 off_t high_water; /* max offset+len written (flush size) */
43 unsigned dirty:1; /* written since the last flush */
44} sd_stage_wb_state;

52/* Open the write-BACK object on the stage store (a normal writable object). The
53 * returned obj carries the stage driver so its pwrite/pread/fsync/close dispatch to
54 * the write-back methods below; the stage-store object is held BY VALUE so the handle
55 * is self-contained (the shell is freed after the caller copies it into fh->sd_obj). */
56static brix_sd_obj_t *
57sd_stage_open_writeback(brix_sd_instance_t *inst, sd_stage_inst_state *is,
58 const char *path, int sd_flags, mode_t mode, int *err_out)
59{
60 brix_sd_obj_t *store_obj;
61 brix_sd_obj_t *obj;
62 sd_stage_wb_state *wb;
63 int e = 0;
64
65 if (is->store->driver->open == NULL || is->store->driver->pwrite == NULL) {
66 if (err_out != NULL) { *err_out = ENOSYS; }
67 return NULL;
68 }
69 store_obj = is->store->driver->open(is->store, path, sd_flags, mode, &e);
70 if (store_obj == NULL) {
71 if (err_out != NULL) { *err_out = e; }
72 return NULL;
73 }
74
75 obj = calloc(1, sizeof(*obj));
76 wb = calloc(1, sizeof(*wb));
77 if (obj == NULL || wb == NULL) {
78 if (store_obj->driver->close != NULL) { store_obj->driver->close(store_obj); }
79 if (store_obj->heap_shell) { free(store_obj); }
80 free(obj);
81 free(wb);
82 if (err_out != NULL) { *err_out = ENOMEM; }
83 return NULL;
84 }
85
86 wb->is = is;
87 wb->high_water = 0;
88 wb->dirty = 0;
89 ngx_cpystrn((u_char *) wb->key, (u_char *) path, sizeof(wb->key));
90 wb->store_obj = *store_obj; /* adopt by value */
91 if (store_obj->heap_shell) { free(store_obj); }
92 wb->store_obj.heap_shell = 0;
93
94 obj->driver = &brix_sd_stage_driver; /* → the write-back methods below */
95 obj->inst = inst;
96 obj->fd = wb->store_obj.fd; /* expose the stage fd (posix sendfile) */
97 obj->snap = wb->store_obj.snap;
98 obj->state = wb;
99 obj->heap_shell = 1; /* malloc'd shell; caller frees post-copy */
100 return obj;
101}

103static brix_sd_obj_t *
104sd_stage_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
105 mode_t mode, int *err_out)
106{
107 sd_stage_inst_state *is = inst->state;
108
109 /* Write open → a write-back object on the stage store (pwrite buffers, fsync/close
110 * flush to the backend). Read open → the source's own object, so read byte-I/O
111 * bypasses the decorator entirely. */
112 if (sd_flags & BRIX_SD_O_WRITE) {
113 return sd_stage_open_writeback(inst, is, path, sd_flags, mode, err_out);
114 }
115 return is->source->driver->open(is->source, path, sd_flags, mode, err_out);
116}

118/* ---- write-back byte-I/O (only reached for objects opened for write above) ------ */

120/* Flush the buffered stage object to the backend through the one staging engine.
121 * Persists the stage buffer first (fsync), then FLUSHes store→source by key. On
122 * success clears the dirty flag; on failure keeps the stage copy for retry. */
123static ngx_int_t
124sd_stage_wb_flush(sd_stage_wb_state *wb)
125{
126 sd_stage_inst_state *is = wb->is;
127 ngx_int_t rc;
128
129 if (wb->store_obj.driver->fsync != NULL) {
130 (void) wb->store_obj.driver->fsync(&wb->store_obj);
131 }
132 if (!wb->dirty) {
133 return NGX_OK;
134 }
135 rc = brix_stage_run_inline(BRIX_STAGE_FLUSH, is->store, wb->key,
136 is->source, wb->key);
137 if (rc == NGX_OK) {
138 wb->dirty = 0;
139 }
140 return rc;
141}

144sd_stage_wb_pwrite(brix_sd_obj_t *obj, const void *buf, size_t len, off_t off)
145{
146 sd_stage_wb_state *wb = obj->state;
147 ssize_t n = wb->store_obj.driver->pwrite(&wb->store_obj, buf, len, off);
148
149 if (n > 0) {
150 wb->dirty = 1;
151 if (off + (off_t) n > wb->high_water) {
152 wb->high_water = off + (off_t) n;
153 }
154 }

373/* Publish the buffered object on the stage store, then FLUSH it to the backend
374 * through the one staging engine. On a successful flush the stage buffer copy is
375 * dropped; on a failed flush it is KEPT (durability preserved for retry, section
376 * 16). Consumes the handle. */
377static ngx_int_t
378sd_stage_staged_commit(brix_sd_staged_t *st, int noreplace)
379{
380 sd_stage_staged_state *ss = st->state;
381 sd_stage_inst_state *is = ss->is;
382 brix_sd_instance_t *store = is->store;
383 brix_sd_instance_t *source = is->source;
384 ngx_int_t rc;
385
386 /* 1. publish the buffered object on the stage store. */
387 if (store->driver->staged_commit(ss->inner, noreplace) != NGX_OK) {
388 int e = errno;
389 if (store->driver->staged_abort != NULL) {
390 store->driver->staged_abort(ss->inner);
391 }
392 free(ss);
393 free(st);
394 errno = e ? e : EIO;
395 return NGX_ERROR;
396 }
397 /* ss->inner is consumed by the commit above. */
398
399 /* 2a. ASYNC write-back (SP4): the object is durable on the stage store now, so
400 * the commit succeeds immediately; the scheduler flushes it to the backend and
401 * drops the stage copy on completion. The export anchor rides on the durable
402 * record so a restart-reconcile can rebuild both tiers and re-flush (§11.3). */
403 if (is->policy.flush_mode == BRIX_WT_MODE_ASYNC) {
404 brix_stage_opts_t o;
405
406 ngx_memzero(&o, sizeof(o));
407 o.async = 1;
408 o.export_root = (is->root_canon[0] != '\0') ? is->root_canon : NULL;
409 (void) brix_stage_submit(BRIX_STAGE_FLUSH, store, ss->key, source,
410 ss->key, &o);
411 free(ss);
412 free(st);
413 return NGX_OK;
414 }
415
416 /* 2b. SYNC write-back: flush inline and reflect the result. */
417 rc = brix_stage_run_inline(BRIX_STAGE_FLUSH, store, ss->key, source,
418 ss->key);
419
420 /* 3. on success drop the stage buffer copy; on failure keep it for retry. */
421 if (rc == NGX_OK && store->driver->unlink != NULL) {
422 (void) store->driver->unlink(store, ss->key, 0);
423 }
424
425 free(ss);
426 free(st);
427 return rc;
428}

479brix_sd_stage_create(brix_sd_instance_t *source, brix_sd_instance_t *store,
480 const brix_stage_policy_t *policy, const char *root_canon, ngx_log_t *log)
481{
482 brix_sd_instance_t *inst;
483 sd_stage_inst_state *is;
484
485 if (source == NULL || store == NULL) {
486 errno = EINVAL;
487 return NULL;
488 }
489 inst = calloc(1, sizeof(*inst));
490 is = calloc(1, sizeof(*is));
491 if (inst == NULL || is == NULL) {
492 free(inst);
493 free(is);
494 errno = ENOMEM;
495 return NULL;
496 }
497 is->source = source;
498 is->store = store;
499 is->log = log;
500 if (root_canon != NULL) {
501 ngx_cpystrn((u_char *) is->root_canon, (u_char *) root_canon,
502 sizeof(is->root_canon));
503 }
504 if (policy != NULL) {
505 is->policy = *policy;
506 } else {
507 ngx_memzero(&is->policy, sizeof(is->policy));
508 is->policy.flush_mode = BRIX_WT_MODE_SYNC;
509 }
510
511 inst->driver = &brix_sd_stage_driver;
512 inst->log = log;
513 inst->pool = NULL;
514 inst->state = is;
515 return inst;

553 * store to the backend (the FLUSH a crash interrupted), dropping the stage copy on
554 * success - exactly the sync staged_commit tail, run again. NGX_OK / NGX_DECLINED

558{
