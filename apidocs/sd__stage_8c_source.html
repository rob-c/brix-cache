1/*
2 * sd_stage.c - the generic write-stage decorator CORE (section 12.2). See header.
3 *
4 * The decorator forwards every read / namespace / xattr / dir op to the wrapped
5 * `source` (open returns the source's own object, so read byte-I/O bypasses the
6 * decorator) and interposes only the staged-write path. This file owns the
7 * decorator core — the open dispatch, the read/namespace/xattr/dir forwarders,
8 * the driver descriptor, and the instance lifecycle (create/destroy/predicates/
9 * reflush). The two interposed WRITE paths (the write-back byte-I/O object and
10 * the staged-upload path) live in sd_stage_write.c; the driver table below
11 * dispatches to them and the shared seam is declared in sd_stage_internal.h.
12 * A posix stage store is byte-equivalent to phase-63's local-temp promote.
13 */

27static brix_sd_obj_t *
28sd_stage_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
29 mode_t mode, int *err_out)
30{
31 sd_stage_inst_state *is = inst->state;
32
33 /* Write open → a write-back object on the stage store (pwrite buffers, fsync/close
34 * flush to the backend). Read open → the source's own object, so read byte-I/O
35 * bypasses the decorator entirely. */
36 if (sd_flags & BRIX_SD_O_WRITE) {
37 return sd_stage_open_writeback(inst, is, path, sd_flags, mode, NULL,
38 err_out);
39 }
40 return is->source->driver->open(is->source, path, sd_flags, mode, err_out);
41}

43/* Credential-scoped open: records the caller's per-user identity in the
44 * write-back state so the eventual flush authenticates as the original user.
45 *
46 * WHAT: Write opens route through sd_stage_open_writeback with the cred so the
47 * owner key/dir/deny are embedded in the durable wb state; read opens
48 * forward to the source via brix_sd_open_maybe_cred.
49 *
50 * WHY: Without this slot the stage decorator drops a caller-supplied cred on
51 * the floor: write opens use the service account for the flush, and read
52 * opens use the service account for the source open on credential-aware
53 * backends.
54 *
55 * HOW: Write → sd_stage_open_writeback(... cred); read →
56 * brix_sd_open_maybe_cred(source, ..., cred). */
57static brix_sd_obj_t *
58sd_stage_open_cred(brix_sd_instance_t *inst, const char *path, int sd_flags,
59 mode_t mode, const brix_sd_cred_t *cred, int *err_out)
60{
61 sd_stage_inst_state *is = inst->state;
62
63 if (sd_flags & BRIX_SD_O_WRITE) {
64 return sd_stage_open_writeback(inst, is, path, sd_flags, mode, cred,
65 err_out);
66 }
67 return brix_sd_open_maybe_cred(is->source, path, sd_flags, mode, cred,
68 err_out);
69}

223brix_sd_instance_t *
224brix_sd_stage_create(brix_sd_instance_t *source, brix_sd_instance_t *store,
225 const brix_stage_policy_t *policy, const char *root_canon, ngx_log_t *log)
226{
227 brix_sd_instance_t *inst;
228 sd_stage_inst_state *is;
229
230 if (source == NULL || store == NULL) {
231 errno = EINVAL;
232 return NULL;
233 }
234 inst = calloc(1, sizeof(*inst));
235 is = calloc(1, sizeof(*is));
236 if (inst == NULL || is == NULL) {
237 free(inst);
238 free(is);
239 errno = ENOMEM;
240 return NULL;
241 }
242 is->source = source;
243 is->store = store;
244 is->log = log;
245 if (root_canon != NULL) {
246 ngx_cpystrn((u_char *) is->root_canon, (u_char *) root_canon,
247 sizeof(is->root_canon));
248 }
249 if (policy != NULL) {
250 is->policy = *policy;
251 } else {
252 ngx_memzero(&is->policy, sizeof(is->policy));
253 is->policy.flush_mode = BRIX_WT_MODE_SYNC;
254 }
255
256 inst->driver = &brix_sd_stage_driver;
257 inst->log = log;
258 inst->pool = NULL;
259 inst->state = is;
260 return inst;
261}

297/* SP4 restart-reconcile: re-flush the durable staged object `key` from the stage
298 * store to the backend (the FLUSH a crash interrupted), dropping the stage copy on
299 * success - exactly the sync staged_commit tail, run again.
300 *
301 * WHAT: Delegates to brix_stage_run_inline_cred so the owner identity (from the
302 * persisted brix_sreq_t.cred) is threaded into the flush and presented to the
303 * backend driver. A NULL cred uses the service credential.
304 *
305 * WHY: A restart-reconcile must authenticate as the original user — not the
306 * service account — for per-user quota / audit / ACL enforcement.
307 *
308 * HOW: Same as the pre-cred path but calls _cred instead of _inline, passing the
309 * caller-supplied cred unchanged. Returns NGX_OK / NGX_DECLINED (not a stage
310 * instance) / NGX_ERROR (errno set; the record is kept for retry). */
311ngx_int_t
312brix_sd_stage_reflush(brix_sd_instance_t *inst, const char *key,
313 const brix_stage_cred_t *cred)
314{
315 sd_stage_inst_state *is;
316 ngx_int_t rc;
317
318 if (!brix_sd_stage_instance_is(inst) || key == NULL) {
319 return NGX_DECLINED;
320 }
321 is = inst->state;
322 rc = brix_stage_run_inline_cred(BRIX_STAGE_FLUSH, is->store, key, is->source,
323 key, cred);
324 if (rc == NGX_OK && is->store->driver->unlink != NULL) {
325 (void) is->store->driver->unlink(is->store, key, 0); /* drop stage copy */
326 }
327 return rc;
328}
