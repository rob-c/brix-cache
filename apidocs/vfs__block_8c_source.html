1/* client/lib/vfs_block.c
2 *
3 * WHAT: Block-device storage backend for brix_vfs.
4 * Handles writes directly to block devices (or plain files used as such)
5 * — no temp file, no rename. commit() = fsync only; abort() = no-op.
6 * Advertises RANDOM_WRITE and FADVISE but NOT TRUNCATE or ATOMIC_TEMP.
7 * WHY: Copying to a block device (/dev/sdb, loop file, etc.) requires writing
8 * in-place. A sibling temp would need to be on the same filesystem as the
9 * device node — which is undefined — and a rename onto a device is wrong.
10 * HOW: open() for READ: O_RDONLY direct open. open() for WRITE: O_WRONLY direct
11 * open (no O_CREAT|O_TRUNC — the device already exists and must not be
12 * re-created or zeroed; the caller supplies XRDC_VFS_FORCE to permit
13 * writing to an existing target). An io_uring ring is optionally wrapped
14 * around the fd using the same AUTO/ON/OFF tri-state as the POSIX backend.
15 * fstat reports device size via BLKGETSIZE64 when S_ISBLK; else st_size.
16 * ngx-free; no goto; functional/modular; one responsibility per function.
17 */

56/* Ring selection */
57/*
58 * block_ring_select — attach an optional io_uring ring to an already-open fd.
59 *
60 * WHAT: implements the AUTO/ON/OFF tri-state, identical to the POSIX backend.
61 * OFF → *ring=NULL, return 0.
62 * ON with unavailable io_uring → *ring=NULL, return -1, st set.
63 * AUTO with unavailable io_uring → *ring=NULL, return 0 (silent fallback).
64 * WHY: mirrors posix_ring_select so the same semantics apply across backends.
65 * HOW: mode from opts->io_uring; guards brix_uring_available(); delegates to
66 * brix_disk_ring_create with a modest window (4 ops, 64 KiB each).
67 */
68static int
69block_ring_select(int fd, int mode, brix_disk_ring **ring, brix_status *st)
70{
71 *ring = NULL;
72
73 if (mode == XRDC_IO_URING_OFF) {
74 return 0;
75 }
76
77 if (!brix_uring_available()) {
78 if (mode == XRDC_IO_URING_ON) {
79 brix_status_set(st, XRDC_EUNSUPPORTED, 0,
80 "vfs block: io_uring=on but io_uring is unavailable "
81 "on this host or this build lacks liburing");
82 return -1;
83 }
84 return 0; /* AUTO: classic path */
85 }
86
87 {
88 brix_status tmp_st;
89 brix_status_clear(&tmp_st);
90 *ring = brix_disk_ring_create(fd, 4, 65536, 0, &tmp_st);
91 if (*ring == NULL && mode == XRDC_IO_URING_ON) {
92 if (st != NULL) {
93 *st = tmp_st;
94 }
95 return -1;
96 }
97 /* AUTO: create failure leaves *ring NULL → classic pread/pwrite */
98 }
99 return 0;
100}

102/* vtable operations */
103/*
104 * block_pread — read n bytes at offset off into buf.
105 *
106 * WHAT: delegates to the ring when present; falls back to plain pread(2).
107 * WHY: single dispatch point — callers never see the ring vs. plain difference.
108 * HOW: ring path: brix_disk_ring_pread; plain path: pread(2) with EINTR retry.
109 */
110static ssize_t
111block_pread(brix_vfs_file *f, int64_t off, void *buf, size_t n, brix_status *st)
112{
113 vfs_block_file *bf = (vfs_block_file *) f;
114
115 if (bf->ring != NULL) {
116 return brix_disk_ring_pread(bf->ring, off, (uint8_t *) buf, n, st);
117 }
118
119 {
120 brix_sd_obj_t obj;
121 ssize_t r;
122
123 brix_sd_posix_wrap(&obj, bf->fd);
124 r = xvfs_pread_once(&obj, buf, n, (off_t) off);
125 if (r < 0) {
126 brix_status_set(st, XRDC_EIO, errno,
127 "vfs block pread: %s", strerror(errno));
128 }
129 return r;
130 }
131}

133/*
134 * block_pwrite — write n bytes from buf at offset off.
135 *
136 * WHAT: delegates to the ring when present; falls back to plain pwrite(2).
137 * WHY: single dispatch point; ring path avoids blocking device writes.
138 * HOW: ring path: split into ring-buffer-sized pieces (brix_disk_ring_pwrite
139 * requires n <= the ring's per-op buffer size; the caller — transfer_pump
140 * — may supply XRDC_COPY_CHUNK = 8 MiB which far exceeds the 64 KiB
141 * ring buffer). Plain path: pwrite(2) retry loop.
142 */
143static int
144block_pwrite(brix_vfs_file *f, int64_t off, const void *buf, size_t n,
145 brix_status *st)
146{
147 vfs_block_file *bf = (vfs_block_file *) f;
148
149 if (bf->ring != NULL) {
150 const uint8_t *p = (const uint8_t *) buf;
151 size_t rem = n;
152 size_t rsz = brix_disk_ring_bufsz(bf->ring);
153
154 while (rem > 0) {
155 size_t chunk = rem < rsz ? rem : rsz;
156 if (brix_disk_ring_pwrite(bf->ring, off, p, chunk, st) != 0) {
157 return -1;
158 }
159 p += chunk;
160 off += (int64_t) chunk;
161 rem -= chunk;
162 }
163 return 0;
164 }
165
166 {
167 brix_sd_obj_t obj;
168
169 brix_sd_posix_wrap(&obj, bf->fd);
170 if (xvfs_pwrite_full(&obj, buf, n, (off_t) off, NULL, NULL) != 0) {
171 brix_status_set(st, XRDC_EIO, errno,
172 "vfs block pwrite: %s", strerror(errno));
173 return -1;
174 }
175 return 0;
176 }
177}

180/*
181 * block_fstat — fill *out with size, mtime, is_dir, exists for the open handle.
182 *
183 * WHAT: calls fstat(2); for block devices (S_ISBLK) queries BLKGETSIZE64 for
184 * the real device size; for other files uses st_size.
185 * WHY: block devices report st_size==0 from fstat; the ioctl gives the real
186 * capacity for copy-progress accounting.
187 * HOW: fstat; if S_ISBLK → BLKGETSIZE64 (with fallback to st_size); populate out.
188 */
189static int
190block_fstat(brix_vfs_file *f, brix_vfs_stat *out, brix_status *st)
191{
192 vfs_block_file *bf = (vfs_block_file *) f;
193 brix_sd_obj_t obj;
194 brix_sd_stat_t sd;
195
196 /* The shared block driver's fstat applies BLKGETSIZE64 for a device's true
197 * size (fd-keyed; the obj wrap's driver is irrelevant for fstat). */
198 brix_sd_posix_wrap(&obj, bf->fd);
199 if (brix_sd_block_driver.fstat(&obj, &sd) != NGX_OK) {
200 brix_status_set(st, XRDC_EIO, errno,
201 "vfs block fstat: %s", strerror(errno));
202 return -1;
203 }
204
205 out->size = (int64_t) sd.size;
206 out->mtime = (int64_t) sd.mtime;
207 out->is_dir = sd.is_dir ? 1 : 0;
208 out->exists = 1;
209 return 0;
210}

232/*
233 * block_sync — flush dirty pages to durable storage.
234 *
235 * WHAT: drains any ring write-behind queue, then calls fsync(2) on bf->fd.
236 * WHY: ensures callers can rely on data being durable after sync() returns.
237 * HOW: brix_disk_ring_flush (no-op on NULL ring); fsync; set st on failure.
238 */
239static int
240block_sync(brix_vfs_file *f, brix_status *st)
241{
242 vfs_block_file *bf = (vfs_block_file *) f;
243
244 if (bf->ring != NULL) {
245 if (brix_disk_ring_flush(bf->ring, st) != 0) {
246 return -1;
247 }
248 }
249
250 {
251 brix_sd_obj_t obj;
252 brix_sd_posix_wrap(&obj, bf->fd);
253 if (xvfs_fsync(&obj) != 0) {
254 brix_status_set(st, XRDC_EIO, errno,
255 "vfs block fsync: %s", strerror(errno));
256 return -1;
257 }
258 }
259 return 0;
260}

309/*
310 * block_close — release the handle: drain ring, close fd, free heap memory.
311 *
312 * WHAT: frees the ring (if any), closes the fd, frees the path string, and
313 * frees the handle struct. Must be called after commit() or abort().
314 * WHY: owns all resources allocated by block_be_open(); one release point.
315 * HOW: brix_disk_ring_destroy (safe on NULL); close(fd); free path; free bf.

317static void
318block_close(brix_vfs_file *f)
319{
320 vfs_block_file *bf = (vfs_block_file *) f;
321
322 brix_disk_ring_destroy(bf->ring);
323
324 if (bf->fd >= 0) {
325 close(bf->fd);
326 }
327
328 free(bf->path);
329 free(bf);
330}

344/* Backend open + stat */
345/*
346 * block_be_open — allocate a vfs_block_file and open the underlying fd.
347 *
348 * WHAT: READ → open(path, O_RDONLY). WRITE → open(path, O_WRONLY) directly,
349 * without O_CREAT|O_TRUNC (an existing device must not be re-created or
350 * zeroed; the FORCE flag permits writing to an existing path but does not
351 * change the open flags for devices). Both paths optionally wrap the fd
352 * in an io_uring ring.
353 * WHY: block devices already exist in /dev/; creating or truncating them via
354 * O_CREAT|O_TRUNC is wrong and potentially destructive.
355 * HOW: allocate bf; open fd; ring_select; set caps.
356 * Caller receives NULL *out on any failure.
357 */
358static int
359block_be_open(const brix_vfs_backend *be, const char *path, int flags,
360 const brix_vfs_open_opts *opts, brix_vfs_file **out,
361 brix_status *st)
362{
363 vfs_block_file *bf;
364 int fd = -1;
365 char *path_copy = NULL;
366 int uring_mode;
367
368 (void) be;
369
370 *out = NULL;
371
372 uring_mode = (opts != NULL) ? opts->io_uring : XRDC_IO_URING_AUTO;
373
374 path_copy = strdup(path);
375 if (path_copy == NULL) {
376 brix_status_set(st, XRDC_EIO, ENOMEM,
377 "vfs block open: out of memory");
378 return -1;
379 }
380
381 if (flags & XRDC_VFS_WRITE) {
382 /* Open the target in place through the shared block driver. No create/
383 * truncate (a device node must pre-exist and must not be zeroed);
384 * O_NOFOLLOW guards against a symlink being swapped in. */
385 fd = brix_sd_block_open_unconfined(path,
386 BRIX_SD_O_WRITE | BRIX_SD_O_NOFOLLOW, 0);
387 if (fd < 0) {
388 brix_status_set(st, XRDC_EIO, errno,
389 "vfs block open write %s: %s",
390 path, strerror(errno));
391 free(path_copy);
392 return -1;
393 }
394 } else {
395 /* READ — in place through the shared block driver. */
396 fd = brix_sd_block_open_unconfined(path, BRIX_SD_O_READ, 0);
397 if (fd < 0) {
398 brix_status_set(st, XRDC_EIO, errno,
399 "vfs block open read %s: %s",
400 path, strerror(errno));
401 free(path_copy);
402 return -1;
403 }
404 }
405
406 bf = calloc(1, sizeof(*bf));
407 if (bf == NULL) {
408 brix_status_set(st, XRDC_EIO, ENOMEM,
409 "vfs block open: out of memory");
410 close(fd);
411 free(path_copy);
412 return -1;
413 }
414
415 bf->fd = fd;
416 bf->path = path_copy;
417 bf->ring = NULL;
418
419 if (block_ring_select(fd, uring_mode, &bf->ring, st) != 0) {
420 close(fd);
421 free(path_copy);
422 free(bf);
423 return -1;
424 }
425
426 bf->base.ops = &s_block_ops;
427 bf->base.caps = (XRDC_VFS_CAP_RANDOM_WRITE | XRDC_VFS_CAP_FADVISE);
428 /* XRDC_VFS_CAP_TRUNCATE and XRDC_VFS_CAP_ATOMIC_TEMP are deliberately absent */
429
430 *out = &bf->base;
431 return 0;
432}

434/*
435 * block_be_stat — stat a path by name without opening a handle.
436 *
437 * WHAT: calls stat(2) on path; fills brix_vfs_stat. For block devices reports
438 * BLKGETSIZE64 size by opening the path briefly for the ioctl.
439 * WHY: allows pre-transfer existence/size checks without a full open.
440 * HOW: stat(2); ENOENT → exists=0, return 0; S_ISBLK → open+BLKGETSIZE64;
441 * other errors → -1, st.
442 */
443static int
444block_be_stat(const brix_vfs_backend *be, const char *path,
445 brix_vfs_stat *out, brix_status *st)
446{
447 struct stat sb;
448
449 (void) be;
450
451 if (stat(path, &sb) != 0) {
452 if (errno == ENOENT) {
453 out->exists = 0;
454 out->size = 0;
455 out->mtime = 0;
456 out->is_dir = 0;
457 return 0;
458 }
459 brix_status_set(st, XRDC_EIO, errno,
460 "vfs block stat %s: %s", path, strerror(errno));
461 return -1;
462 }
463
464 if (S_ISBLK(sb.st_mode)) {
465 int tmp_fd = brix_sd_block_open_unconfined(path,
466 BRIX_SD_O_READ | BRIX_SD_O_NOFOLLOW, 0);
467 if (tmp_fd >= 0) {
468 brix_sd_obj_t obj;
469 brix_sd_stat_t sd;
470 brix_sd_posix_wrap(&obj, tmp_fd);
471 if (brix_sd_block_driver.fstat(&obj, &sd) == NGX_OK
472 && sd.size > 0) {
473 out->size = (int64_t) sd.size;
474 } else {
475 out->size = (int64_t) sb.st_size;
476 }
477 close(tmp_fd);
478 } else {
479 out->size = (int64_t) sb.st_size;
480 }
481 } else {
482 out->size = (int64_t) sb.st_size;
483 }
484
485 out->mtime = (int64_t) sb.st_mtime;
486 out->is_dir = S_ISDIR(sb.st_mode) ? 1 : 0;
487 out->exists = 1;
488 return 0;
489}
