1/* File: tls.c — TPC pull in-protocol TLS upgrade (kXR_gotoTLS)
2 * WHAT: tpc_start_tls() performs a blocking client TLS handshake over the already
3 * connected TPC pull socket after the source answered the kXR_protocol request
4 * with kXR_gotoTLS; it stores the negotiated SSL on t->tls (and its per-pull
5 * SSL_CTX on t->tls_ctx) so the low-level I/O helpers (src/tpc/outbound/io.c) route every
6 * subsequent frame — login, GSI/ztn auth, open/read/close — through SSL_read /
7 * SSL_write. tpc_tls_teardown() releases both.
8 *
9 * WHY: A source that mandates in-protocol TLS (xrd.tls / a TLS-only sec.protbind)
10 * cannot be pulled from over plaintext (phase-57 W1 gap 2). The native client
11 * already does this; the destination's pull path did not. Because the TPC pull
12 * runs on a thread-pool worker (a blocking context, not the nginx event loop),
13 * we use a synchronous SSL_connect over the blocking fd rather than nginx's
14 * async ngx_ssl_handshake.
15 *
16 * HOW: SSL_CTX_new(TLS_client_method) → min TLS 1.2 → if brix_trusted_ca is set,
17 * load it as a CA directory and SSL_VERIFY_PEER (chain verification; on load
18 * failure fall back to no verification so an encrypted-but-unverified transport
19 * still works) → SSL_new + SSL_set_fd(fd) + SNI(src_host) → SSL_connect. On any
20 * failure t->err_msg / t->xrd_error are set and all OpenSSL objects freed.
21 * Gated by the caller on conf->tpc_outbound_tls (bootstrap.c). (phase-57 §F5) */

32/* WHAT: blocking client TLS handshake over the pull fd; stores SSL on t->tls.
33 * Returns 0 on success, -1 with t->err_msg / t->xrd_error set on failure. */
34int
35tpc_start_tls(brix_tpc_pull_t *t, int fd)
36{
37 SSL_CTX *ctx;
38 SSL *ssl;
39
40 ctx = SSL_CTX_new(TLS_client_method());
41 if (ctx == NULL) {
42 snprintf(t->err_msg, sizeof(t->err_msg), "TPC TLS: SSL_CTX_new failed");
43 t->xrd_error = kXR_ServerError;
44 return -1;
45 }
46 SSL_CTX_set_min_proto_version(ctx, TLS1_2_VERSION);
47
48 /* Verify the source's certificate chain against brix_trusted_ca when it is
49 * configured (treated as a CA directory). If the CA store cannot be loaded,
50 * downgrade to an encrypted-but-unverified transport rather than failing the
51 * pull outright. When chain verification IS active we ALSO bind the expected
52 * hostname below (verify_host) — chain-only would accept any CA-valid cert
53 * for any host, a MITM gap. */
54 int verify_host = 0;
55 if (t->conf->trusted_ca.len > 0 && t->conf->trusted_ca.len < PATH_MAX) {
56 char cadir[PATH_MAX];
57
58 if (brix_str_cbuf(cadir, sizeof(cadir), &t->conf->trusted_ca) != NULL
59 && SSL_CTX_load_verify_locations(ctx, NULL, cadir) == 1) {
60 SSL_CTX_set_verify(ctx, SSL_VERIFY_PEER, NULL);
61 verify_host = 1;
62 } else {
63 SSL_CTX_set_verify(ctx, SSL_VERIFY_NONE, NULL);
64 }
65 } else {
66 SSL_CTX_set_verify(ctx, SSL_VERIFY_NONE, NULL);
67 }
68
69 ssl = SSL_new(ctx);
70 if (ssl == NULL) {
71 SSL_CTX_free(ctx);
72 snprintf(t->err_msg, sizeof(t->err_msg), "TPC TLS: SSL_new failed");
73 t->xrd_error = kXR_ServerError;
74 return -1;
75 }
76 SSL_set_fd(ssl, fd);
77 if (t->src_host[0] != '\0') {
78 (void) SSL_set_tlsext_host_name(ssl, t->src_host);
79 /* Only meaningful when the chain is actually being verified. */
80 if (verify_host) {
81 SSL_set_hostflags(ssl, X509_CHECK_FLAG_NO_PARTIAL_WILDCARDS);
82 if (SSL_set1_host(ssl, t->src_host) != 1) {
83 snprintf(t->err_msg, sizeof(t->err_msg),
84 "TPC TLS: host-verify setup failed");
85 t->xrd_error = kXR_ServerError;
86 SSL_free(ssl);
87 SSL_CTX_free(ctx);
88 return -1;
89 }
90 }
91 }
92
93 if (SSL_connect(ssl) != 1) {
94 char ebuf[160];
95
96 ERR_error_string_n(ERR_get_error(), ebuf, sizeof(ebuf));
97 snprintf(t->err_msg, sizeof(t->err_msg),
98 "TPC TLS handshake to source failed: %s", ebuf);
99 t->xrd_error = kXR_NotAuthorized;
100 SSL_free(ssl);
101 SSL_CTX_free(ctx);
102 return -1;
103 }
104
105 t->tls = ssl;
106 t->tls_ctx = ctx;
107 return 0;
108}
