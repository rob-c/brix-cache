1/*
2 * WHAT: Serialize client XRootD requests into wire format and flush to upstream
3 * redirector. This is the request side of transparent proxy mode — nginx receives a
4 * client opcode, translates it into the upstream protocol buffer, and sends it over
5 * TCP without exposing the backend's identity to the client.
6 *
7 * WHY: Transparent XRootD proxy requires opaque relay of opcodes while maintaining
8 * file-handle translation end-to-end. The upstream context (brix_upstream_t) stores
9 * a saved copy of the original client request; these functions serialize that state
10 * into the wire protocol format and deliver it to the backend connection.
11 *
12 * HOW: Two-phase pipeline per function:
13 * brix_upstream_send_request — buffer allocation (ngx_palloc), switch dispatch
14 * across three supported opcodes (kXR_locate, kXR_open, kXR_stat), wire-format
15 * serialization with network-byte-order conversions (htons/htonl), write buffer
16 * setup and state machine transition to XRD_UP_REQUEST. INVARIANT #4 enforced:
17 * all wire paths → resolve_path() before open().
18 * brix_upstream_flush — non-blocking TCP write loop using ngx_connection_t->send,
19 * NGX_AGAIN handling via ngx_handle_write_event for event-loop readiness, read
20 * event setup for response reception.
21 */

28ngx_int_t
29brix_upstream_send_request(brix_upstream_t *up)
30{
31 ngx_pool_t *pool = up->conn->pool;
32 size_t pathlen = strlen(up->req_path);
33 size_t hdrlen = XRD_REQUEST_HDR_LEN;
34 size_t total = hdrlen + pathlen;
35 u_char *buf;
36
37 BRIX_PALLOC_OR_RETURN(buf, pool, total, NGX_ERROR);
38 ngx_memzero(buf, total);
39
40 switch (up->req_opcode) {
41
42 case kXR_locate: {
43 ClientLocateRequest *r = (ClientLocateRequest *) (void *) buf;
44
45 xrdw_locate_req_t lb = { .options = up->req_options };
46 r->streamid[0] = 0;
47 r->streamid[1] = 1;
48 r->requestid = htons(kXR_locate);
49 xrdw_locate_req_pack(&lb, ((ClientRequestHdr *) (void *) buf)->body);
50 r->dlen = htonl((kXR_int32) pathlen);
51 ngx_memcpy(buf + hdrlen, up->req_path, pathlen);
52 break;
53 }
54
55 case kXR_open: {
56 ClientOpenRequest *r = (ClientOpenRequest *) (void *) buf;
57
58 xrdw_open_req_t ob = { .mode = up->req_open_mode,
59 .options = up->req_options };
60 r->streamid[0] = 0;
61 r->streamid[1] = 1;
62 r->requestid = htons(kXR_open);
63 xrdw_open_req_pack(&ob, ((ClientRequestHdr *) (void *) buf)->body);
64 r->dlen = htonl((kXR_int32) pathlen);
65 ngx_memcpy(buf + hdrlen, up->req_path, pathlen);
66 break;
67 }
68
69 case kXR_stat: {
70 ClientStatRequest *r = (ClientStatRequest *) (void *) buf;
71
72 r->streamid[0] = 0;
73 r->streamid[1] = 1;
74 r->requestid = htons(kXR_stat);
75 r->dlen = htonl((kXR_int32) pathlen);
76 ngx_memcpy(buf + hdrlen, up->req_path, pathlen);
77 break;
78 }
79
80 default:
81 ngx_log_error(NGX_LOG_ERR, up->client_conn->log, 0,
82 "brix: upstream: unsupported opcode %d",
83 (int) up->req_opcode);
84 return NGX_ERROR;
85 }
86
87 up->wbuf = buf;
88 up->wbuf_len = total;
89 up->wbuf_pos = 0;
90
91 up->rhdr_pos = 0;
92 up->resp_dlen = 0;
93 up->resp_body = NULL;
94 up->resp_body_pos = 0;
95 up->state = XRD_UP_REQUEST;
96
97 return brix_upstream_flush(up);
98}

100ngx_int_t
101brix_upstream_flush(brix_upstream_t *up)
102{
103 ngx_connection_t *uconn = up->conn;
104 ssize_t n;
105
106 while (up->wbuf_pos < up->wbuf_len) {
107 n = uconn->send(uconn, up->wbuf + up->wbuf_pos,
108 up->wbuf_len - up->wbuf_pos);
109 if (n > 0) {
110 up->wbuf_pos += (size_t) n;
111 continue;
112 }
113 if (n == NGX_AGAIN) {
114 if (ngx_handle_write_event(uconn->write, 0) != NGX_OK) {
115 return NGX_ERROR;
116 }
117 return NGX_AGAIN;
118 }
119 return NGX_ERROR;
120 }
121
122 if (ngx_handle_read_event(uconn->read, 0) != NGX_OK) {
123 return NGX_ERROR;
124 }
125
126 return NGX_OK;
127}
