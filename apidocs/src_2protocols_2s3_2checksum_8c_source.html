1/*
2 * checksum.c — S3 additional-algorithm checksums (phase-43 W1).
3 *
4 * WHAT: Generalizes S3 full-object integrity beyond CRC-64/NVME to the full AWS
5 * set — CRC32, CRC32C, SHA-1, SHA-256, CRC-64/NVME — for both the upload
6 * verify+echo path (PUT / UploadPart) and the GET/HEAD echo path.
7 *
8 * WHY: Modern AWS SDKs negotiate a request checksum by default and send
9 * x-amz-checksum-<algo> (and/or x-amz-sdk-checksum-algorithm). Before this
10 * workstream only crc64nvme was verified; a client that selected crc32 or
11 * sha256 had its integrity header silently ignored — a data-integrity gap.
12 *
13 * HOW: One descriptor table maps each AWS algorithm token to the integrity-layer
14 * algorithm name and the wire header. s3_checksum_b64() computes (or
15 * cache-reads) the checksum via the shared integrity engine — which already
16 * owns every algorithm and the xattr cache — and re-encodes the hex digest as
17 * AWS's base64-of-raw-bytes wire form at the edge (INVARIANT #9: encode at the
18 * edge, never in the kernel). s3_put_checksum_apply() picks the client's
19 * algorithm, rejects conflicting/ambiguous selections (400 InvalidRequest),
20 * verifies a supplied value (400 BadDigest on mismatch — object removed) and
21 * echoes the result; with no client checksum it preserves the historical
22 * crc64nvme default.
23 */

116/*
117 * s3_checksum_b64 — base64 wire value of one algorithm for an open fd.
118 *
119 * Pulls the lowercase hex digest from the shared integrity engine (xattr cache
120 * when present), converts it to raw bytes, and base64-encodes them — the AWS
121 * x-amz-checksum-* wire form. cache_only=1 returns NGX_DECLINED on a cache miss
122 * rather than paying a full-file read (GET/HEAD echo path).
123 *
124 * Returns NGX_OK (out filled), NGX_DECLINED (cache_only miss), or NGX_ERROR.
125 */
126ngx_int_t
127s3_checksum_b64(ngx_http_request_t *r, int fd, const char *path,
128 const char *alg_name, ngx_flag_t cache_only, char *out, size_t outsz)
129{
130 brix_integrity_info_t info;
131 brix_integrity_opts_t iopts;
132 unsigned char bytes[64];
133 size_t hexlen, nbytes, i;
134 ngx_str_t src, dst;
135 ngx_int_t rc;
136
137 ngx_memzero(&iopts, sizeof(iopts));
138 iopts.allow_xattr_cache = 1;
139 iopts.update_xattr_cache = cache_only ? 0 : 1;
140 iopts.require_regular_file = 1;
141 iopts.no_compute = cache_only ? 1 : 0;
142
143 rc = brix_integrity_get_fd(r->connection->log, fd, NULL, path, alg_name,
144 &iopts, &info);
145 if (rc != NGX_OK) {
146 return rc;
147 }
148
149 hexlen = ngx_strlen(info.hex);
150 if (hexlen == 0 || (hexlen % 2) != 0 || (hexlen / 2) > sizeof(bytes)) {
151 return NGX_ERROR;
152 }
153 nbytes = hexlen / 2;
154 for (i = 0; i < nbytes; i++) {
155 int hi = brix_hex_from_char((unsigned char) info.hex[2 * i]);
156 int lo = brix_hex_from_char((unsigned char) info.hex[2 * i + 1]);
157 if (hi < 0 || lo < 0) {
158 return NGX_ERROR;
159 }
160 bytes[i] = (unsigned char) ((hi << 4) | lo);
161 }
162
163 if (outsz < (size_t) ngx_base64_encoded_length(nbytes) + 1) {
164 return NGX_ERROR;
165 }
166 src.data = bytes;
167 src.len = nbytes;
168 dst.data = (u_char *) out;
169 ngx_encode_base64(&dst, &src);
170 out[dst.len] = '\0';
171 return NGX_OK;
172}

174/* Find a descriptor by lowercase/uppercase AWS token, else NULL. */
175static const s3_cksum_desc_t *
176s3_cksum_by_token(const char *tok, size_t len)
177{
178 ngx_uint_t i;
179
180 for (i = 0; i < S3_CKSUM_TABLE_N; i++) {
181 if (ngx_strlen(s3_cksum_table[i].token) == len
182 && ngx_strncasecmp((u_char *) s3_cksum_table[i].token,
183 (u_char *) tok, len) == 0)
184 {
185 return &s3_cksum_table[i];
186 }
187 }
188 return NULL;
189}

191/*
192 * s3_put_select_algo — decide which checksum algorithm a PUT requested.
193 *
194 * Scans the x-amz-checksum-<algo> value headers and the
195 * x-amz-sdk-checksum-algorithm / x-amz-checksum-algorithm declaration. Sets
196 * *want_value to the supplied value header (NULL if only a declaration / none).
197 *
198 * Returns the chosen descriptor, or NULL for "no checksum requested"
199 * (caller applies the crc64nvme default). Sets *conflict=1 when the selection
200 * is ambiguous (two different value headers, or a value header disagreeing with
201 * the declared algorithm) — the caller answers 400 InvalidRequest.
202 */
203static const s3_cksum_desc_t *
204s3_put_select_algo(ngx_http_request_t *r, ngx_table_elt_t **want_value,
205 int *conflict)
206{
207 const s3_cksum_desc_t *chosen = NULL;
208 ngx_table_elt_t *value_hdr = NULL;
209 ngx_table_elt_t *decl;
210 ngx_uint_t i, value_count = 0;
211
212 *conflict = 0;
213 *want_value = NULL;
214
215 for (i = 0; i < S3_CKSUM_TABLE_N; i++) {
216 ngx_table_elt_t *h = brix_http_find_header(
217 r, s3_cksum_table[i].header,
218 ngx_strlen(s3_cksum_table[i].header));
219 if (h != NULL && h->value.len > 0) {
220 value_count++;
221 chosen = &s3_cksum_table[i];
222 value_hdr = h;
223 }
224 }
225
226 if (value_count > 1) {
227 *conflict = 1;
228 return NULL;
229 }
230
231 /* x-amz-sdk-checksum-algorithm (preferred) or x-amz-checksum-algorithm. */
232 decl = brix_http_find_header(r, "x-amz-sdk-checksum-algorithm",
233 sizeof("x-amz-sdk-checksum-algorithm") - 1);
234 if (decl == NULL) {
235 decl = brix_http_find_header(r, "x-amz-checksum-algorithm",
236 sizeof("x-amz-checksum-algorithm") - 1);
237 }
238 if (decl != NULL && decl->value.len > 0) {
239 const s3_cksum_desc_t *d =
240 s3_cksum_by_token((const char *) decl->value.data, decl->value.len);
241 if (d == NULL) {
242 *conflict = 1; /* declared an algorithm we don't support */
243 return NULL;
244 }
245 if (chosen != NULL && chosen != d) {
246 *conflict = 1; /* value header disagrees with declaration */
247 return NULL;
248 }
249 chosen = d; /* declaration alone selects the algorithm */
250 }
251
252 *want_value = value_hdr;
253 return chosen;
254}

256/*
257 * s3_put_checksum_apply — verify a client-supplied full-object checksum (any
258 * supported algorithm) and echo it; replaces the crc64nvme-only path.
259 *
260 * Returns S3_CKSUM_OK (verified or echoed), S3_CKSUM_MISMATCH (supplied value
261 * did not match — object removed; caller answers 400 BadDigest),
262 * S3_CKSUM_CONFLICT (ambiguous selection — caller answers 400 InvalidRequest),
263 * or S3_CKSUM_ERROR (our own compute failed — caller proceeds without a header).
264 */
265s3_cksum_result_t
266s3_put_checksum_apply(ngx_http_request_t *r, const char *fs_path,
267 const char *root_canon)
268{
269 const s3_cksum_desc_t *desc;
270 ngx_table_elt_t *want_value;
271 int conflict;
272 brix_vfs_file_t *fh;
273 char b64[S3_CHECKSUM_B64_MAX];
274 ngx_int_t rc;
275
276 desc = s3_put_select_algo(r, &want_value, &conflict);
277 if (conflict) {
278 /* The object was already committed; remove it — a malformed integrity
279 * request must not leave an object behind (AWS rejects pre-store). */
280 s3_cksum_vfs_unlink(r, fs_path, root_canon);
281 return S3_CKSUM_CONFLICT;
282 }
283 if (desc == NULL) {
284 desc = &s3_cksum_table[S3_CKSUM_DEFAULT_IDX]; /* crc64nvme default */
285 }
286
287 fh = s3_cksum_vfs_open(r, fs_path, root_canon);
288 if (fh == NULL) {
289 return S3_CKSUM_ERROR;
290 }
291 rc = s3_checksum_b64(r, brix_vfs_file_fd(fh), fs_path, desc->alg_name,
292 0 /* compute+cache */, b64, sizeof(b64));
293 brix_vfs_close(fh, r->connection->log);
294 if (rc != NGX_OK) {
295 return S3_CKSUM_ERROR;
296 }
297
298 if (want_value != NULL) {
299 size_t blen = ngx_strlen(b64);
300 if (want_value->value.len != blen
301 || ngx_strncmp(want_value->value.data, (u_char *) b64, blen) != 0)
302 {
303 s3_cksum_vfs_unlink(r, fs_path, root_canon);
304 return S3_CKSUM_MISMATCH;
305 }
306 }
307
308 (void) s3_set_header(r, desc->header, b64);
309 (void) s3_set_header(r, S3_HDR_CHECKSUM_TYPE, "FULL_OBJECT");
310 return S3_CKSUM_OK;
311}

313/*
314 * s3_put_trailer_checksum_apply — verify+echo a checksum carried in an
315 * aws-chunked trailer (phase-43 W0). Unlike the header path, the algorithm and
316 * value come from the decoded stream's trailer line rather than a request
317 * header. Same result contract as s3_put_checksum_apply.

319s3_cksum_result_t
320s3_put_trailer_checksum_apply(ngx_http_request_t *r, const char *fs_path,

341 brix_vfs_close(fh, r->connection->log);
342 if (rc != NGX_OK) {

347 size_t blen = ngx_strlen(b64);
348 if (ngx_strlen(value) != blen

352 return S3_CKSUM_MISMATCH;
353 }
354 }
355
356 (void) s3_set_header(r, desc->header, b64);
357 (void) s3_set_header(r, S3_HDR_CHECKSUM_TYPE, "FULL_OBJECT");

359}

361/*
362 * s3_echo_object_checksums — GET/HEAD response checksum echo.
363 *
364 * Always echoes a cached crc64nvme (historical behavior — never recomputes on
365 * read). When the request carries x-amz-checksum-mode: ENABLED, additionally
366 * echoes every other algorithm that is present in the xattr cache, matching AWS
367 * which returns stored additional checksums only when the client opts in.
368 */
369void
370s3_echo_object_checksums(ngx_http_request_t *r, int fd, const char *path)
371{
372 ngx_table_elt_t *mode;
373 int mode_enabled;
374 int emitted_type = 0;
375 ngx_uint_t i;
376 char b64[S3_CHECKSUM_B64_MAX];
377
378 mode = brix_http_find_header(r, "x-amz-checksum-mode",
379 sizeof("x-amz-checksum-mode") - 1);
380 mode_enabled = (mode != NULL && mode->value.len == sizeof("ENABLED") - 1
381 && ngx_strncasecmp(mode->value.data,
382 (u_char *) "ENABLED",
383 sizeof("ENABLED") - 1) == 0);
384
385 for (i = 0; i < S3_CKSUM_TABLE_N; i++) {
386 /* Without checksum-mode, only the crc64nvme default is echoed. */
387 if (!mode_enabled && i != S3_CKSUM_DEFAULT_IDX) {
388 continue;
389 }
390 if (s3_checksum_b64(r, fd, path, s3_cksum_table[i].alg_name,
391 1 /* cache-only */, b64, sizeof(b64)) == NGX_OK)
392 {
393 (void) s3_set_header(r, s3_cksum_table[i].header, b64);
394 if (!emitted_type) {
395 (void) s3_set_header(r, S3_HDR_CHECKSUM_TYPE, "FULL_OBJECT");
396 emitted_type = 1;
397 }
398 }
399 }
