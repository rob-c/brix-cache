1/*
2 * cred_stage.h — one private staging area for short-lived credential material.
3 *
4 * Several subsystems must hand a live secret (an OAuth2 subject/bearer token, a
5 * delegated X.509 proxy PEM) to a fork/exec'd helper that reads it from a FILE
6 * path rather than a fd — curl -d @file, the GSI blocking client, etc. Writing
7 * that file into world-traversable /tmp opens a co-tenant race (CWE-377): during
8 * the brief window the file exists a shell on the same host / mount namespace can
9 * open() the inode by name and read the secret.
10 *
11 * This module is the SINGLE place that decides where such files live and enforces
12 * the directory's safety. Every credential stager routes through it instead of
13 * open-coding an mkstemp("/tmp/..."). The staging dir is a per-uid 0700 tmpfs
14 * directory (/dev/shm/brix-creds.<euid>); a 0700 parent owned by the worker uid
15 * removes the co-tenant open() entirely, and tmpfs keeps nothing on disk or in
16 * backups. There is deliberately NO /tmp fallback: if a private directory cannot
17 * be guaranteed, callers fail closed rather than stage a secret insecurely.
18 *
19 * Pure libc (no nginx headers) so the path/permission logic is unit-testable
20 * standalone — see tests/c/test_cred_stage.c.
21 */
22

28/*
29 * brix_cred_stage_dir — resolve (creating if absent) the process's private
30 * credential staging directory and copy its NUL-terminated path into out.
31 *
32 * The directory is created 0700 and REQUIRED to be a real directory owned by the
33 * effective uid with no group/other access bits; any deviation (foreign owner, a
34 * loosened mode, a symlink swap, tmpfs absent) is treated as unsafe.
35 *
36 * Returns 0 on success; -1 with errno set when a private directory cannot be
37 * guaranteed (EPERM if a pre-existing path fails the owner/mode check). Callers
38 * MUST fail closed on -1 — never fall back to a world-traversable location.
39 */
40int brix_cred_stage_dir(char *out, size_t outsz);
41
42/*
43 * brix_cred_stage_write — atomically stage len bytes into a fresh 0600 file under
44 * the private staging directory and return its path in path_out.
45 *
46 * The file name begins with prefix (e.g. "tpc_token_body_") followed by mkstemp's
47 * unique suffix. On any failure no file is left behind and -1 is returned with
48 * errno set. The caller owns `bytes`; this function does not retain or scrub them
49 * (the caller cleanses its own buffer) and remains responsible for unlink()ing the
50 * returned path once the helper subprocess has consumed it.
51 *
52 * Returns 0 on success, -1 with errno set on failure.
53 */
54int brix_cred_stage_write(const char *prefix, const void *bytes, size_t len,
55 char *path_out, size_t path_outsz);
