2 * auth_sigv4_verify_crypto.c — SigV4 canonicalisation, signing-key derivation,
3 * signature computation, and the single constant-time signature compare.
4 *
5 * WHAT: Owns the byte-frozen SigV4 crypto carved out of auth_sigv4_verify.c:
6 * the worker-local one-slot signing-key cache + four-round HMAC derive,
7 * the canonical signed-headers block builder, the canonical-request →
8 * string-to-sign → HMAC-SHA256 → hex signature computation, and the single
9 * constant-time key+signature decision.
10 * WHY: auth_sigv4_verify.c exceeded the ~500-line file-size guard (phase-79).
11 * This is the SigV4 cryptographic core — every intermediate byte layout is
12 * load-bearing for interop and security, so it splits into its own unit
13 * away from the timestamp rules and the orchestrator. INVARIANT §6: this
14 * is SigV4-only crypto — it shares no logic with WLCG bearer-token auth.
15 * HOW: s3_sigv4_derive_signing_key_cached caches the last day/region key and
16 * otherwise calls the shared libxrdproto 4-round HMAC chain (byte-identical
17 * to the client sign path). s3_sigv4_compute_signature runs the standard
18 * SigV4 steps 1-7 over canonical URI/query/headers. s3_sigv4_compare folds
19 * the deferred access-key match and the CRYPTO_memcmp into ONE branch so no
20 * timing or message oracle distinguishes an unknown key from a bad
21 * signature. Byte layout, messages, and result codes preserved 1:1 with the
22 * pre-split single-file implementation.
23 */

38/*
39 * SigV4 signing key derivation
40 *
41 * Four-round HMAC chain:
42 * k1 = HMAC("AWS4" + secret, date) k2 = HMAC(k1, region)
43 * k3 = HMAC(k2, "s3") k4 = HMAC(k3, "aws4_request")
44 * */
45
46/* Worker-local one-slot cache: signing key is stable for one calendar day per
47 * region, so cache the last key and avoid four HMAC rounds on every request. */
48static struct {
49 char date[9]; /* YYYYMMDD\0, empty string means invalid */
50 char region[65];
51 u_char key[32];
52} s_signing_key_cache;

54/*
55 * s3_sigv4_derive_signing_key_cached — derive the SigV4 signing key with a
56 * worker-local one-slot cache.
57 *
58 * WHAT: Fill out[0..31] with the day/region signing key, returning 1 on success
59 * or 0 when the shared derive fails.
60 * WHY: The signing key is stable for one calendar day per region, so caching
61 * the last result avoids four HMAC rounds on the hot verify path (and on
62 * the post-policy path that also calls this).
63 * HOW: Return the cached key on a date+region hit; otherwise run the shared
64 * libxrdproto 4-round HMAC chain (byte-identical to the client sign path),
65 * then refresh the cache slot.
66 */
67int
68s3_sigv4_derive_signing_key_cached(const ngx_str_t *secret_key,
69 const char *date, const char *region,
70 u_char out[32])
71{
72 if (s_signing_key_cache.date[0] != '\0'
73 && strcmp(s_signing_key_cache.date, date) == 0
74 && strcmp(s_signing_key_cache.region, region) == 0)
75 {
76 ngx_memcpy(out, s_signing_key_cache.key, 32);
77 return 1;
78 }
79
80 /* Shared 4-round HMAC chain (libxrdproto) — byte-identical to the client's
81 * sign path so client-signs == server-verifies by construction. */
82 if (!brix_sigv4_signing_key((const uint8_t *) secret_key->data,
83 secret_key->len, date, region, "s3", out)) {
84 return 0;
85 }
86
87 ngx_cpystrn((u_char *) s_signing_key_cache.date,
88 (u_char *) date, sizeof(s_signing_key_cache.date));
89 ngx_cpystrn((u_char *) s_signing_key_cache.region,
90 (u_char *) region, sizeof(s_signing_key_cache.region));
91 ngx_memcpy(s_signing_key_cache.key, out, 32);
92 return 1;
93}

95/*
96 * build_canonical_headers — build the SigV4 canonical signed-headers block.
97 *
98 * WHAT: Write the "header-name:value\n" block (lowercased names, trimmed values)
99 * for each entry of the semicolon-separated signed_hdrs list into out,
100 * returning the number of bytes written (out is also NUL-terminated).
101 * WHY: The canonical headers block is a byte-frozen SigV4 input; a dedicated
102 * builder keeps its exact lowering/trim/order rules in one auditable place.
103 * HOW: Tokenise signed_hdrs on ';'; resolve each value (host from the request
104 * line, others via get_header); lowercase the name, trim leading/trailing
105 * SP/TAB from the value, and append name:value\n while bounds-checking out.
106 */
107static size_t
108build_canonical_headers(ngx_http_request_t *r,
109 const char *signed_hdrs,
110 u_char *out, size_t outsz)
111{
112 char hdrs[256];
113 size_t oi = 0;
114
115 ngx_cpystrn((u_char *) hdrs, (u_char *) signed_hdrs, sizeof(hdrs));
116
117 char *save = NULL;
118 char *tok = strtok_r(hdrs, ";", &save);
119
120 while (tok) {
121 ngx_str_t val;
122
123 if (strcmp(tok, "host") == 0) {
124 val = r->headers_in.host ? r->headers_in.host->value
125 : (ngx_str_t) ngx_null_string;
126 } else {
127 val = get_header(r, tok);
128 }
129
130 size_t nlen = strlen(tok);
131 size_t vlen = val.len;
132
133 if (oi + nlen + 1 + vlen + 2 >= outsz) {
134 break;
135 }
136
137 for (size_t i = 0; i < nlen; i++) {
138 out[oi++] = (u_char) tolower((unsigned char) tok[i]);
139 }
140 out[oi++] = ':';
141
142 const u_char *vs = val.data;
143 const u_char *ve = val.data + vlen;
144 while (vs < ve && (*vs == ' ' || *vs == '\t')) { vs++; }
145 while (ve > vs && (ve[-1] == ' ' || ve[-1] == '\t')) { ve--; }
146
147 /* A SignedHeaders entry may name a header the client never sent, so
148 * val.data can be NULL with vlen 0 (empty canonical value). Guard the
149 * copy: memcpy's src is 'nonnull', and passing NULL even with length 0
150 * is undefined behaviour. */
151 if (ve > vs) {
152 ngx_memcpy(out + oi, vs, (size_t)(ve - vs));
153 oi += (size_t)(ve - vs);
154 }
155 out[oi++] = '\n';
156
157 tok = strtok_r(NULL, ";", &save);
158 }
159
160 out[oi] = '\0';
161 return oi;
162}

164/*
165 * s3_sigv4_compute_signature — build the canonical request and compute the
166 * server-side SigV4 signature hex.
167 *
168 * WHAT: Produce the 64-char lowercase-hex HMAC-SHA256 signature that the client
169 * signature must equal, following the standard SigV4 steps 1-7.
170 * WHY: Concentrates the byte-frozen canonicalisation + HMAC chain in one pure-
171 * ish computation so the verifier's decision logic reads cleanly. This is
172 * SigV4-only crypto (INVARIANT §6): it shares no logic with token auth.
173 * HOW: Canonical URI/query/headers → canonical request → string-to-sign →
174 * cached signing-key derive → HMAC-SHA256 → hex. On the two internal
175 * failure modes (derive/HMAC) it records the result and returns
176 * NGX_ERROR; the caller maps that to HTTP 500. out->k4 is exported for
177 * chunk-sig retention. Byte layout of every intermediate is preserved 1:1.
178 *
179 * Returns: NGX_OK with out->computed_hex[0..63] filled and out->k4 populated;
180 * NGX_ERROR (result already recorded) on an internal crypto failure.
181 */
182ngx_int_t
183s3_sigv4_compute_signature(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
184 const sigv4_components_t *comp, const s3_amz_date_out_t *amz,
185 s3_sigv4_sig_out_t *out)
186{
187 u_char canonical[8192];
188 u_char canon_qs[2048];
189 u_char canon_uri[S3_MAX_KEY];
190 u_char canon_hdrs[2048];
191 u_char string_to_sign[4096];
192 u_char hash_hex[65];
193 u_char cr_hash[32];
194 u_char computed[32];
195 size_t n;
196
197 /* 1. Canonical URI */
198 brix_http_urlencode(r->uri.data, r->uri.len,
199 (char *) canon_uri, sizeof(canon_uri), "/");
200
201 /* 2. Canonical query string */
202 build_canonical_qs(r->args.data, r->args.len, comp->presigned,
203 canon_qs, sizeof(canon_qs));
204
205 /* 3. Canonical headers */
206 build_canonical_headers(r, comp->signed_hdrs,
207 canon_hdrs, sizeof(canon_hdrs));
208
209 /* 4. Canonical request */
210 n = (size_t) snprintf((char *) canonical, sizeof(canonical),
211 "%.*s\n" /* method */
212 "%s\n" /* uri */
213 "%s\n" /* qs */
214 "%s\n" /* headers */
215 "%s\n" /* signed header names */
216 "UNSIGNED-PAYLOAD",
217 (int) r->method_name.len, r->method_name.data,
218 (char *) canon_uri,
219 (char *) canon_qs,
220 (char *) canon_hdrs,
221 comp->signed_hdrs);
222
223 brix_sha256(canonical, n, cr_hash);
224 brix_hex_encode(cr_hash, 32, (char *) hash_hex);
225
226 /* 5. String to sign */
227 n = (size_t) snprintf((char *) string_to_sign, sizeof(string_to_sign),
228 "AWS4-HMAC-SHA256\n"
229 "%.*s\n"
230 "%s/%s/s3/aws4_request\n"
231 "%s",
232 (int) amz->len, amz->date,
233 comp->date,
234 comp->region,
235 (char *) hash_hex);
236
237 /* 6. Derive signing key (cached: stable for one calendar day per region) */
238 if (!s3_sigv4_derive_signing_key_cached(&cf->secret_key,
239 comp->date, comp->region, out->k4)) {
240 s3_record_auth_result(BRIX_S3_AUTH_INTERNAL_ERROR);
241 return NGX_ERROR;
242 }
243
244 /* 7. Compute the signature */
245 if (!brix_hmac_sha256(out->k4, 32, string_to_sign, n, computed)) {
246 s3_record_auth_result(BRIX_S3_AUTH_INTERNAL_ERROR);
247 return NGX_ERROR;
248 }
249
250 brix_hex_encode(computed, 32, out->computed_hex);
251 return NGX_OK;
252}

254/*
255 * s3_sigv4_compare — the single constant-time signature decision (W5).
256 *
257 * WHAT: Decide whether the request's access key AND signature are both valid,
258 * emitting the terminal SignatureDoesNotMatch / Signature-length errors.
259 * WHY: An unknown access key and a bad signature MUST take the identical path,
260 * status, message, and HMAC work so no timing or message oracle can
261 * distinguish them. Consolidating the length check + constant-time
262 * compare here keeps that guarantee obvious and unbroken.
263 * HOW: Reject non-64-char signatures first (prevents CRYPTO_memcmp over pad
264 * bytes). Then fold key_ok and CRYPTO_memcmp(computed_hex, signature)
265 * into ONE branch — both use constant-time comparison. Frozen messages,
266 * result codes, and the WARN log line preserved 1:1.
267 *
268 * Returns: NGX_OK when the signature is authentic, otherwise the terminal
269 * ngx_int_t result (response already emitted).
270 */
271ngx_int_t
272s3_sigv4_compare(ngx_http_request_t *r, const sigv4_components_t *comp,
273 const char computed_hex[65], int key_ok)
274{
275 /* Reject signatures that are not exactly 64 hex chars — prevents a short
276 * client string from causing CRYPTO_memcmp to compare against pad bytes. */
277 if (strlen(comp->signature) != 64) {
278 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
279 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
280 "InvalidRequest",
281 "Signature must be 64 hex characters");
282 }
283
284 /*
285 * Single constant-time decision (W5): an unknown access key (!key_ok) and a
286 * bad signature take the identical path, status, and message here, having
287 * both run the full HMAC above — no timing or message oracle distinguishes
288 * them. CRYPTO_memcmp prevents a timing oracle on the HMAC value itself.
289 */
290 if (!key_ok || CRYPTO_memcmp(computed_hex, comp->signature, 64) != 0) {
291 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
292 "brix_s3: SigV4 auth failed for key=%s (key_ok=%d)",
293 comp->akid, key_ok);
294 s3_record_auth_result(key_ok ? BRIX_S3_AUTH_SIG_MISMATCH
295 : BRIX_S3_AUTH_BAD_KEY);
296 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
297 "SignatureDoesNotMatch",
298 "The request signature we calculated does "
299 "not match the signature you provided");
300 }
301
302 return NGX_OK;
303}
