15/*
16 * gsi_recover_peer_signed — recover the client's DH public (peer key) from the
17 * signed-DH round 2. The client sends kXRS_cipher = its Public() blob signed
18 * with the proxy private key (RSA EncryptPrivate) plus kXRS_puk = the proxy
19 * public key (PEM). We verify/recover the blob with DecryptPublic against that
20 * public key, then parse it as a DH peer. Returns the peer EVP_PKEY (caller
21 * frees) or NULL. This authenticates that the sender holds the proxy key.
22 */
23static EVP_PKEY *
24gsi_recover_peer_signed(const u_char *payload, size_t plen, ngx_log_t *log)
25{
26 const u_char *cipher = NULL, *puk = NULL;
27 size_t cipherlen = 0, puklen = 0;
28 BIO *pbio;
29 EVP_PKEY *proxy_pub, *peer;
30 u_char *blob;
31 size_t bloblen;
32
33 if (gsi_find_bucket(payload, plen, (uint32_t) kXRS_cipher,
34 &cipher, &cipherlen) != 0
35 || gsi_find_bucket(payload, plen, (uint32_t) kXRS_puk,
36 &puk, &puklen) != 0)
37 {
38 ngx_log_error(NGX_LOG_WARN, log, 0,
39 "brix: GSI signed-DH: kXRS_cipher/kXRS_puk missing");
40 return NULL;
41 }
42
43 pbio = BIO_new_mem_buf(puk, (int) puklen);
44 proxy_pub = pbio ? PEM_read_bio_PUBKEY(pbio, NULL, NULL, NULL) : NULL;
45 BIO_free(pbio);
46 if (proxy_pub == NULL) {
47 ngx_log_error(NGX_LOG_WARN, log, 0,
48 "brix: GSI signed-DH: cannot read proxy public key");
49 return NULL;
50 }
51
52 bloblen = cipherlen + 2 * (size_t) EVP_PKEY_size(proxy_pub) + 64;
53 blob = malloc(bloblen);
54 bloblen = blob ? brix_gsi_rsa_decrypt_public(proxy_pub, cipher, cipherlen,
55 blob, bloblen) : 0;
56 EVP_PKEY_free(proxy_pub);
57 if (bloblen == 0) {
58 free(blob);
59 ngx_log_error(NGX_LOG_WARN, log, 0,
60 "brix: GSI signed-DH: signature verification failed");
61 return NULL;
62 }
63 peer = brix_gsi_cipher_parse_peer(blob, bloblen);
64 free(blob);
65 return peer;
66}

68/*
69 * gsi_signed_derive_secret — WHAT: agree the padded (HasPad=1) signed-DH shared
70 * secret from our private key and the recovered peer public. WHY: isolates the
71 * multi-step EVP_PKEY_derive ladder (which dominated the signed handler's CCN)
72 * behind one early-return boundary. HOW: two-call derive idiom with
73 * set_dh_pad(1) — the signed-DH AES key is the first key_len bytes of the
74 * prime-sized secret (unlike the unsigned path's pad=0). Allocates `secret` from
75 * c->pool. Frees `peer` on every path (success and failure) so the caller does
76 * not. Returns the pool-allocated secret with *secret_len set, or NULL (logged).
77 */
78static unsigned char *
79gsi_signed_derive_secret(ngx_connection_t *c, EVP_PKEY *dh_key, EVP_PKEY *peer,
80 size_t *secret_len)
81{
82 EVP_PKEY_CTX *pkctx = EVP_PKEY_CTX_new(dh_key, NULL);
83 unsigned char *secret = NULL;
84
85 *secret_len = 0;
86 if (pkctx == NULL
87 || EVP_PKEY_derive_init(pkctx) != 1
88 || EVP_PKEY_CTX_set_dh_pad(pkctx, 1) != 1
89 || EVP_PKEY_derive_set_peer(pkctx, peer) != 1
90 || EVP_PKEY_derive(pkctx, NULL, secret_len) != 1
91 || (secret = ngx_palloc(c->pool, *secret_len)) == NULL
92 || EVP_PKEY_derive(pkctx, secret, secret_len) != 1)
93 {
94 EVP_PKEY_CTX_free(pkctx);
95 EVP_PKEY_free(peer);
96 ngx_log_error(NGX_LOG_WARN, c->log, 0,
97 "brix: GSI signed-DH: DH derive failed");
98 return NULL;
99 }
100 EVP_PKEY_CTX_free(pkctx);
101 EVP_PKEY_free(peer);
102 return secret;
103}

117/*
118 * gsi_signed_select_key — WHAT: resolve the client-selected cipher, verify the
119 * secret is long enough for its key, and copy the AES key into sc. WHY: factors
120 * the Phase-52 cipher-selection + length-check out of the handler. HOW:
121 * kXRS_cipher_alg lookup with an aes-128-cbc fallback (the default every
122 * conformant client offers); rejects (returns 0, after cleansing `secret`) when
123 * secret_len < cipher key_len. On success fills sc->cipher and copies key_len
124 * bytes into sc->aeskey. Returns 1 on success, 0 on reject.
125 */
126static int
127gsi_signed_select_key(const u_char *payload, size_t plen,
128 unsigned char *secret, size_t secret_len,
129 gsi_session_cipher_t *sc)
130{
131 /* Phase 52 (WS-A): honour the cipher the client selected (kXRS_cipher_alg);
132 * fall back to aes-128-cbc, the default every conformant client offers. */
133 brix_gsi_select_cipher_name(payload, plen, sc->name, sizeof(sc->name));
134 if (!brix_gsi_cipher_lookup(sc->name, &sc->cipher)) {
135 (void) brix_gsi_cipher_lookup("aes-128-cbc", &sc->cipher);
136 }
137
138 if (secret_len < (size_t) sc->cipher.key_len) {
139 OPENSSL_cleanse(secret, secret_len);
140 return 0;
141 }
142 ngx_memcpy(sc->aeskey, secret, (size_t) sc->cipher.key_len);
143 return 1;
144}

146/*
147 * brix_gsi_parse_x509_signed — round-2 handler for the signed-DH variant
148 * (mirrors the client's signed path in client/lib/sec/sec_gsi.c). Recovers the
149 * peer DH public from the RSA-signed kXRS_cipher, agrees the padded (HasPad=1)
150 * DH secret, decrypts the IV-prepended kXRS_main, and returns the proxy chain.
151 * Sets ctx->sigver.signing_key = SHA-256(secret) / signing_active for symmetry with
152 * the unsigned path. Returns STACK_OF(X509) * or NULL.
153 */
154STACK_OF(X509) *
155brix_gsi_parse_x509_signed(brix_ctx_t *ctx, ngx_connection_t *c)
156{
157 const u_char *payload = ctx->recv.payload;
158 size_t plen = ctx->recv.cur_dlen;
159 ngx_log_t *log = c->log;
160 const u_char *main_data = NULL;
161 size_t main_len = 0;
162 EVP_PKEY *peer;
163 unsigned char *secret = NULL;
164 size_t secret_len = 0;
165 gsi_session_cipher_t sc = {0};
166 uint8_t *plain;
167 size_t plain_len = 0;
168 STACK_OF(X509) *chain;
169
170 if (gsi_find_bucket(payload, plen, (uint32_t) kXRS_main,
171 &main_data, &main_len) != 0) {
172 ngx_log_error(NGX_LOG_WARN, log, 0,
173 "brix: GSI signed-DH: kXRS_main missing");
174 return NULL;
175 }
176
177 peer = gsi_recover_peer_signed(payload, plen, log);
178 if (peer == NULL) {
179 return NULL;
180 }
181
182 /*
183 * Padded (HasPad=1) DH secret (helper); signing_key = SHA-256(secret)
184 * mirrors the unsigned path; sigver is off by default whenever signed-DH is
185 * selected. gsi_signed_derive_secret frees `peer` on every path.
186 */
187 secret = gsi_signed_derive_secret(c, ctx->gsi.dh_key, peer, &secret_len);
188 if (secret == NULL) {
189 return NULL;
190 }
191
192 if (!gsi_signed_select_key(payload, plen, secret, secret_len, &sc)) {
193 return NULL;
194 }
195
196 /* Persist the session cipher for a possible §F6 delegation round — signed-DH
197 * path always uses an IV-prepended main (use_iv=1). */
198 gsi_persist_session_cipher(ctx, sc.name, sc.aeskey, sc.cipher.key_len, 1);
199
200 (void) gsi_store_signing_key(ctx, secret, secret_len);
201 OPENSSL_cleanse(secret, secret_len);
202
203 /* The negotiated session cipher (above). This is the SIGNED-DH path, which
204 * a peer only enters when its version >= XrdSecgsiVersDHsigned (10400) — and
205 * that is exactly the condition under which stock XrdSecgsi sets useIV=true
206 * (XrdSecProtocolgsi.cc: `useIV = (RemVers >= XrdSecgsiVersDHsigned)`). So
207 * the encrypted main always carries a leading IV of the cipher's own length
208 * (sessionKey->MaxIVLength()); we strip it unconditionally here. The IV is
209 * NOT signalled by a name suffix — the cipher name on the wire is bare, and
210 * select_cipher_name() resolves it. Default aes-128-cbc; the client's cipher
211 * choice is honoured. */
212 plain = brix_gsi_cipher_decrypt(&sc.cipher, sc.aeskey, main_data, main_len,
213 1, &plain_len);
214 OPENSSL_cleanse(sc.aeskey, sizeof(sc.aeskey));
215 if (plain == NULL) {
216 ngx_log_error(NGX_LOG_WARN, log, 0,
217 "brix: GSI signed-DH: main decrypt failed");
218 return NULL;
219 }
220
221 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, log, 0,
222 "brix: GSI signed-DH decrypted kXRS_main: %uz bytes",
223 plain_len);
224 gsi_capture_client_rtag(ctx, plain, plain_len); /* §F6 delegation rtag */
225 gsi_capture_fullproxy(ctx, plain, plain_len); /* §5.1 full-proxy PT */
226 chain = gsi_chain_from_plaintext(plain, (int) plain_len, log);
227 free(plain);
228 return chain;
229}
