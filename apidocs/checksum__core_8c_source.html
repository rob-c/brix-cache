1/*
2 * checksum_core.c — pure (ngx-free) whole-file checksum compute kernels.
3 *
4 * WHAT: brix_cksum_u32_fd() (Adler-32/CRC-32/CRC-32c) and
5 * brix_cksum_digest_fd() (MD5/SHA-1/SHA-256) stream a file descriptor and
6 * return the result; no nginx, no logging.
7 * WHY: Single source for fd→checksum compute, shared by the nginx module
8 * (src/compat/checksum.c delegates here, keeping its ngx logging) and the
9 * native client (client/lib/checksum.c), via libxrdproto build-in-place.
10 * HOW: read the whole file in 64 KiB chunks through the SD driver vtable
11 * (obj.driver->pread, EINTR-retried) so the kernel reads whatever backend
12 * the wrapped object carries; zlib adler32/crc32, brix_crc32c_extend for
13 * crc32c, OpenSSL EVP for digests. No goto (per coding-standards): the EVP
14 * path uses an init-at-edge helper.
15 */

28/* ---- Is a normalized u32 checksum kind supported by the u32 kernel? ----
29 *
30 * WHAT: Returns non-zero when kind is one of the three u32 checksum families
31 * this kernel computes (Adler-32, CRC-32, CRC-32c), zero otherwise.
32 *
33 * WHY: Keeps the argument-validation predicate in brix_cksum_u32_obj() a single
34 * named call instead of an inline chain of inequalities, so the orchestrator
35 * reads as one guard and the accepted-kind set lives in exactly one place.
36 *
37 * HOW: 1. Compare kind against each accepted constant and OR the results.
38 * Callers must fold BRIX_CK_ZCRC32 to BRIX_CK_CRC32 before calling.
39 */
40static int
41u32_kind_ok(int kind)
42{
43 return kind == BRIX_CK_ADLER32 || kind == BRIX_CK_CRC32
44 || kind == BRIX_CK_CRC32C;
45}

47/* ---- Initial running value (seed) for a u32 checksum kind ----
48 *
49 * WHAT: Returns the identity seed for the running accumulator: the zlib
50 * adler32/crc32 identity for those kinds, or 0 for the CRC-32c path.
51 *
52 * WHY: The zlib CRC-32 and Adler-32 identities are obtained by calling the
53 * library with a NULL buffer (their canonical init); folding that choice
54 * into a helper keeps brix_cksum_u32_obj() flat while preserving the exact
55 * seeds. CRC-32c is seeded to 0 because brix_crc32c_extend() owns its init.
56 *
57 * HOW: 1. Adler-32 -> adler32(0L, Z_NULL, 0).
58 * 2. CRC-32 -> crc32(0L, Z_NULL, 0).
59 * 3. otherwise (CRC-32c) -> 0.
60 */
61static uLong
62u32_seed(int kind)
63{
64 if (kind == BRIX_CK_ADLER32) {
65 return adler32(0L, Z_NULL, 0);
66 }
67 if (kind == BRIX_CK_CRC32) {
68 return crc32(0L, Z_NULL, 0);
69 }
70 return 0;
71}

73/* ---- Fold one chunk into the running u32 checksum for the selected kind ----
74 *
75 * WHAT: Extends the running accumulator over n bytes of buf. Adler-32 and CRC-32
76 * update the zlib accumulator via *zcrc; CRC-32c updates *crc32c. No return.
77 *
78 * WHY: Isolates the per-kind arithmetic (the one place bit-exactness matters)
79 * from the read/EINTR loop, so the loop stays a plain drain and the
80 * accumulation is a single-responsibility, independently reviewable step.
81 *
82 * HOW: 1. Adler-32 -> zcrc = adler32(zcrc, buf, (uInt) n).
83 * 2. CRC-32 -> zcrc = crc32(zcrc, buf, (uInt) n).
84 * 3. otherwise (CRC-32c) -> crc32c = brix_crc32c_extend(crc32c, buf, n).
85 * Casts match the by-value original exactly so the folded bytes are identical.
86 */
87static void
88u32_update(int kind, const unsigned char *buf, size_t n,
89 uLong *zcrc, uint32_t *crc32c)
90{
91 if (kind == BRIX_CK_ADLER32) {
92 *zcrc = adler32(*zcrc, buf, (uInt) n);
93 } else if (kind == BRIX_CK_CRC32) {
94 *zcrc = crc32(*zcrc, buf, (uInt) n);
95 } else {
96 *crc32c = brix_crc32c_extend(*crc32c, buf, n);
97 }
98}

100/* ---- Bytes to request this read given the remaining range budget ----
101 *
102 * WHAT: For a bounded range (len >= 0) returns min(bufsz, remaining), clamping
103 * the final read so the sum stops exactly at start+len; for an unbounded
104 * range (len < 0, "to EOF") always returns bufsz.
105 *
106 * WHY: All three ranged kernels (u32/u64/digest) share the same last-read clamp;
107 * naming it keeps each read loop a flat drain and puts the boundary
108 * arithmetic — the one place an off-by-one would corrupt a partial-range
109 * checksum — in a single reviewable spot.
110 */
111static size_t
112range_want(off_t len, off_t remaining, size_t bufsz)
113{
114 if (len < 0) {
115 return bufsz;
116 }
117 if (remaining <= 0) {
118 return 0;
119 }
120 return (remaining < (off_t) bufsz) ? (size_t) remaining : bufsz;
121}

123int
124brix_cksum_u32_obj_range(int kind, brix_sd_obj_t *obj, off_t start, off_t len,
125 uint32_t *out)
126{
127 unsigned char buf[BRIX_CK_BUFSZ];
128 off_t offset;
129 off_t remaining = len;
130 uint32_t crc32c = 0;
131 uLong zcrc;
132
133 /* "zcrc32" is XRootD's name for the zlib CRC-32 (== CRC32/ISO-HDLC); fold it
134 * onto the CRC32 kernel path so there is one implementation. */
135 if (kind == BRIX_CK_ZCRC32) {
136 kind = BRIX_CK_CRC32;
137 }
138
139 if (out == NULL || obj == NULL || obj->driver == NULL
140 || !u32_kind_ok(kind) || start < 0) {
141 return -1;
142 }
143
144 zcrc = u32_seed(kind);
145 offset = start;
146
147 while (len < 0 || remaining > 0) {
148 size_t want = range_want(len, remaining, sizeof(buf));
149 ssize_t n = obj->driver->pread(obj, buf, want, offset);
150 if (n < 0) {
151 if (errno == EINTR) {
152 continue;
153 }
154 return -1;
155 }
156 if (n == 0) {
157 break;
158 }
159 offset += (off_t) n;
160 if (len >= 0) {
161 remaining -= (off_t) n;
162 }
163 u32_update(kind, buf, (size_t) n, &zcrc, &crc32c);
164 }
165
166 *out = (kind == BRIX_CK_CRC32C) ? crc32c : (uint32_t) zcrc;
167 return 0;
168}

257/* Drive an initialised ctx over the byte extent [start, start+len) of the object
258 * (len < 0 ⇒ to EOF); caller owns/frees ctx. 0/-1. */
259static int
260digest_drive(EVP_MD_CTX *ctx, const EVP_MD *md, brix_sd_obj_t *obj,
261 off_t start, off_t len, unsigned char *out, unsigned int *outlen)
262{
263 unsigned char buf[BRIX_CK_BUFSZ];
264 off_t offset = start;
265 off_t remaining = len;
266
267 if (EVP_DigestInit_ex(ctx, md, NULL) != 1) {
268 return -1;
269 }
270 while (len < 0 || remaining > 0) {
271 size_t want = range_want(len, remaining, sizeof(buf));
272 ssize_t n = obj->driver->pread(obj, buf, want, offset);
273 if (n < 0) {
274 if (errno == EINTR) {
275 continue;
276 }
277 return -1;
278 }
279 if (n == 0) {
280 break;
281 }
282 offset += (off_t) n;
283 if (len >= 0) {
284 remaining -= (off_t) n;
285 }
286 if (EVP_DigestUpdate(ctx, buf, (size_t) n) != 1) {
287 return -1;
288 }
289 }
290 return (EVP_DigestFinal_ex(ctx, out, outlen) == 1) ? 0 : -1;
291}

293int
294brix_cksum_digest_obj_range(int kind, brix_sd_obj_t *obj, off_t start,
295 off_t len, unsigned char *out, unsigned int *outlen)
296{
297 const EVP_MD *md = md_for(kind);
298 EVP_MD_CTX *ctx;
299 int rc;
300
301 if (md == NULL || obj == NULL || obj->driver == NULL
302 || out == NULL || outlen == NULL || start < 0) {
303 return -1;
304 }
305 ctx = EVP_MD_CTX_new();
306 if (ctx == NULL) {
307 return -1;
308 }
309 rc = digest_drive(ctx, md, obj, start, len, out, outlen);
310 EVP_MD_CTX_free(ctx);
311 return rc;
312}
