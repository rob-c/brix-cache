1/*
2 * sd_pblock_internal.h — cross-file declarations shared between the translation
3 * units of the pblock ("pseudo-block") storage-driver vtable after the phase-79
4 * size split.
5 *
6 * WHAT: Publishes the one per-open-object state struct that both the driver core
7 * and the byte-I/O file touch (pblock_obj_t), plus the driver-vtable slot
8 * functions that were split out of sd_pblock.c and are now called across a
9 * file boundary — every slot is still named in the descriptor in
10 * sd_pblock.c, so all of them are non-static — together with the two
11 * internal helpers that cross a boundary directly (sd_pblock_ftruncate,
12 * called by the object-open O_TRUNC path in the core, and sd_pblock_drop_dst,
13 * called by both rename and staged-commit).
14 *
15 * WHY: sd_pblock.c was one 1111-line file. It is split by concept into the
16 * driver core — instance + object lifecycle and the descriptor (stays in
17 * sd_pblock.c) — the worker-safe byte I/O (sd_pblock_io.c), the namespace,
18 * directory-iteration and xattr operations (sd_pblock_namespace.c), and the
19 * staged atomic-publish path (sd_pblock_staged.c). The descriptor in the
20 * core references every slot, so the moved functions become non-static and
21 * are declared here; nothing here is exported beyond the pblock backend.
22 *
23 * HOW: Every pblock driver-vtable translation unit includes this header (in
24 * addition to sd.h / sd_pblock_catalog.h / pblock_store.h). It is gated by
25 * BRIX_HAVE_SQLITE exactly like its includers, so a no-sqlite build stays
26 * byte-for-byte unchanged.
27 *
28 * Requires: fs/backend/sd.h (brix_sd_* types, ngx_int_t/ngx_fd_t),
29 * sd_pblock_catalog.h (pblock_meta, PBLOCK_BLOB_ID_CAP),
30 * pblock_store.h (pblock_state_t), <limits.h> (PATH_MAX),
31 * <stdint.h>, <sys/types.h>, <sys/uio.h> before inclusion.
32 */

45/* ---- per-open object state (obj->state) ---------------------------------- *
46 * obj->fd is the block-0 fd (or NGX_INVALID_FILE for directories); higher
47 * blocks are opened transiently per I/O. Shared between the driver core (which
48 * builds and tears it down) and the byte-I/O file (which reads/writes through
49 * it), so its layout lives here. */
50typedef struct {
51 pblock_state_t *st; /* borrowed from inst->state */
52 char path[PATH_MAX]; /* logical path (catalog key) */
53 char blob_id[PBLOCK_BLOB_ID_CAP];
54 int64_t block_size; /* this file's stripe size */
55 pblock_meta meta; /* cached metadata row */
56 void *lab; /* pblock_lab_obj_t* (Phase-83); NULL *
57 * ⇒ no lab rule applies to this handle*/
58 int64_t a_rbytes; /* F17: bytes read through this handle *
59 * (accumulated only when st->audit) */
60 int64_t a_wbytes; /* F17: bytes written through handle */
61 int64_t a_maxblock; /* F17: highest block index touched, +1 */
62 uint32_t *csi_crc; /* F3: at-rest per-block CRC32c snapshot *
63 * loaded at open (0 = unset); NULL when *
64 * csi off. Verify reads this — no DB on *
65 * the hot path. Owned; freed at close. */
66 uint64_t csi_n; /* entries in csi_crc[] */
67 int64_t csi_dlo; /* lowest block index written through *
68 * this handle (verify skips [dlo,dhi): *
69 * its snapshot CRC is stale; flush *
70 * recomputes it). INT64_MAX = none. */
71 int64_t csi_dhi; /* highest written block index + 1 */
72 struct pblock_lock_rng_s *lock_rng; /* F15: foreign range leases snapshotted *
73 * at open (owned; freed at close). NULL *
74 * ⇒ no ranges — pwrite skips the scan. */
75 uint32_t lock_n; /* F15: entries in lock_rng[] */
76 int64_t quota_max; /* F5: largest size this handle may grow *
77 * the file to, snapshotted at open *
78 * (INT64_MAX = no quota) — writes past *
79 * it fail EDQUOT with no DB on the hot *
80 * path */
81 void *wv; /* F10: brix_wverify_t* whole-object CRC *
82 * accumulator — armed only on create / *
83 * O_TRUNC opens when st->refs; feeds *
84 * publish-time dedup. NULL ⇒ no dedup *
85 * candidate hash from this handle. */
86 unsigned dirty:1; /* size/mtime need catalog write-back */
87} pblock_obj_t;

158/* ---- identity resolution + POSIX access checks (sd_pblock_ident.c) ------- *
159 * A request identity resolved to catalog-internal synthetic ids: the
160 * principal's uid plus one gid per VO the credential lists (group = VO).
161 * `service` set (NULL/empty principal) means "no identity" — every check
162 * passes, preserving service (root-like) semantics for identity-less callers. */
163#define PBLOCK_MAX_GIDS 16
164
165typedef struct {
166 int service; /* 1 = bypass all checks */
167 uint32_t uid; /* principal's synthetic uid */
168 uint32_t gid; /* primary gid (first VO, or uid — a
169 * user-private group — when VO-less) */
170 uint32_t gids[PBLOCK_MAX_GIDS];/* every VO gid, gids[0] == gid */
171 int ngids;
172} pblock_ids_t;
