9/*
10 * gsi_mech.c — the mem-BIO GSI GSSAPI accept engine declared in gsi_mech.h.
11 *
12 * A single SSL object drives a TLS 1.2 handshake whose flights arrive/depart as
13 * ADAT tokens through a pair of memory BIOs (rbio = tokens in, wbio = tokens
14 * out). Once the handshake completes we verify the client's proxy chain, then —
15 * if delegating — run the GSI 'D' / proxy-request / signed-proxy sub-exchange as
16 * application-data over the same SSL, reusing the RFC-3820 crypto in
17 * proxy_req.c. After completion the same SSL provides wrap/unwrap for the
18 * RFC 2228 protected control channel.
19 */
20
21enum {
22 ST_TLS = 0, /* driving SSL_accept */
23 ST_WAIT_D, /* handshake done, awaiting the deleg 'D' tok */
24 ST_WAIT_SIGNED, /* CSR sent, awaiting the signed proxy */
25 ST_DONE,
26 ST_FAIL
27};

75/* Drain all pending outbound TLS bytes from wbio into a pool buffer. */
76static ngx_int_t
77gss_drain(brix_gssapi_srv_t *g, ngx_str_t *out)
78{
79 int pend;
80 u_char *buf;
81 int n;
82
83 out->len = 0;
84 out->data = NULL;
85
86 pend = BIO_pending(g->wbio);
87 if (pend <= 0) {
88 return NGX_OK;
89 }
90 buf = ngx_palloc(g->pool, (size_t) pend);
91 if (buf == NULL) {
92 return NGX_ERROR;
93 }
94 n = BIO_read(g->wbio, buf, pend);
95 if (n <= 0) {
96 return NGX_ERROR;
97 }
98 out->data = buf;
99 out->len = (size_t) n;
100 return NGX_OK;
101}

104/* Read all currently-available decrypted application data into a pool buffer
105 * (up to `cap` bytes). Returns NGX_OK with *out set (possibly empty). */
106static ngx_int_t
107gss_read_all(brix_gssapi_srv_t *g, size_t cap, ngx_str_t *out)
108{
109 u_char *buf;
110 size_t off = 0;
111
112 out->len = 0;
113 out->data = NULL;
114
115 buf = ngx_palloc(g->pool, cap);
116 if (buf == NULL) {
117 return NGX_ERROR;
118 }
119 for ( ;; ) {
120 int n = SSL_read(g->ssl, buf + off, (int) (cap - off));
121 if (n > 0) {
122 off += (size_t) n;
123 if (off == cap) {
124 break; /* full — commands/creds fit in cap */
125 }
126 continue;
127 }
128 break; /* WANT_READ / no more data */
129 }
130 out->data = buf;
131 out->len = off;
132 return NGX_OK;
133}

213static ngx_int_t
214gss_verify_peer(brix_gssapi_srv_t *g)
215{
216 X509 *leaf;
217 STACK_OF(X509) *chain;
218 brix_gsi_verify_result_t res;
219 ngx_int_t rc;
220 size_t dnlen;
221
222 leaf = SSL_get_peer_certificate(g->ssl); /* +1 ref */
223 if (leaf == NULL) {
224 ngx_log_error(NGX_LOG_ERR, g->log, 0,
225 "brix gsi_mech: client presented no certificate");
226 return NGX_ERROR;
227 }
228 chain = SSL_get_peer_cert_chain(g->ssl); /* borrowed */
229
230 rc = brix_gsi_verify_chain(g->log, g->ca_store, leaf, chain, 0, &res, 0);
231 if (rc == NGX_OK) {
232 dnlen = ngx_strlen(res.dn_buf);
233 g->dn.data = ngx_pnalloc(g->pool, dnlen);
234 if (g->dn.data == NULL) {
235 X509_free(leaf);
236 return NGX_ERROR;
237 }
238 ngx_memcpy(g->dn.data, res.dn_buf, dnlen);
239 g->dn.len = dnlen;
240 g->verified = 1;
241 }
242 X509_free(leaf);

247/* On 'D', build a proxy-cert request for the client's leaf and send it. */
248static brix_gss_status_e
249gss_send_proxy_req(brix_gssapi_srv_t *g, ngx_str_t *out)
250{
251 X509 *leaf;
252 ngx_str_t parent_pem;
253 brix_gsi_blob_t parent;
254 brix_gsi_buf_t req_der = { 0 };
255 char errbuf[160];
256 brix_gsi_err_t err = { errbuf, sizeof(errbuf) };
257
258 leaf = SSL_get_peer_certificate(g->ssl);
259 if (leaf == NULL || gss_x509_pem(g, leaf, &parent_pem) != NGX_OK) {
260 if (leaf) { X509_free(leaf); }
261 return BRIX_GSS_FAILED;
262 }
263 X509_free(leaf);
264
265 parent.data = parent_pem.data;
266 parent.len = parent_pem.len;
267 if (brix_gsi_build_pxyreq(&parent, &g->req_key, &req_der, &err) != 0) {
268 ngx_log_error(NGX_LOG_ERR, g->log, 0,
269 "brix gsi_mech: build proxy request failed: %s", errbuf);
270 return BRIX_GSS_FAILED;
271 }
272
273 if (SSL_write(g->ssl, req_der.data, (int) req_der.len) <= 0) {
274 OPENSSL_free(req_der.data);
275 return BRIX_GSS_FAILED;
276 }
277 OPENSSL_free(req_der.data);
278
279 if (gss_drain(g, out) != NGX_OK) {
280 return BRIX_GSS_FAILED;
281 }
282 g->state = ST_WAIT_SIGNED;
283 return BRIX_GSS_CONTINUE;
284}

287/* On the signed proxy, assemble the delegated credential. */
288static brix_gss_status_e
289gss_recv_signed(brix_gssapi_srv_t *g, ngx_str_t *out)
290{
291 ngx_str_t signed_der;
292 const u_char *p;
293 X509 *proxy;
294 ngx_str_t proxy_pem, chain_pem;
295 brix_gsi_blob_t proxy_blob, chain_blob;
296 brix_gsi_buf_t cred = { 0 };
297 char errbuf[160];
298 brix_gsi_err_t err = { errbuf, sizeof(errbuf) };
299
300 if (gss_read_all(g, 16384, &signed_der) != NGX_OK || signed_der.len == 0) {
301 return BRIX_GSS_FAILED;
302 }
303 p = signed_der.data;
304 proxy = d2i_X509(NULL, &p, (long) signed_der.len); /* first cert = proxy */
305 if (proxy == NULL) {
306 ngx_log_error(NGX_LOG_ERR, g->log, 0,
307 "brix gsi_mech: signed proxy is not a valid certificate");
308 return BRIX_GSS_FAILED;
309 }
310 if (gss_x509_pem(g, proxy, &proxy_pem) != NGX_OK) {
311 X509_free(proxy);
312 return BRIX_GSS_FAILED;
313 }
314 X509_free(proxy);
315
316 if (gss_chain_pem(g, SSL_get_peer_cert_chain(g->ssl), &chain_pem)
317 != NGX_OK)
318 {
319 return BRIX_GSS_FAILED;
320 }
321
322 proxy_blob.data = proxy_pem.data; proxy_blob.len = proxy_pem.len;
323 chain_blob.data = chain_pem.data; chain_blob.len = chain_pem.len;
324 if (brix_gsi_assemble_proxy(&proxy_blob, g->req_key, &chain_blob, &cred,
325 &err) != 0)
326 {
327 ngx_log_error(NGX_LOG_ERR, g->log, 0,
328 "brix gsi_mech: assemble delegated proxy failed: %s",
329 errbuf);
330 return BRIX_GSS_FAILED;
331 }
332 g->proxy_pem.data = ngx_pnalloc(g->pool, cred.len);
333 if (g->proxy_pem.data == NULL) {
334 OPENSSL_free(cred.data);
335 return BRIX_GSS_FAILED;
336 }
337 ngx_memcpy(g->proxy_pem.data, cred.data, cred.len);
338 g->proxy_pem.len = cred.len;
339 OPENSSL_free(cred.data);
340
341 g->state = ST_DONE;
342 return gss_drain(g, out) == NGX_OK ? BRIX_GSS_COMPLETE : BRIX_GSS_FAILED;
343}

346brix_gssapi_srv_t *
347brix_gssapi_srv_create(ngx_pool_t *pool, ngx_log_t *log, SSL_CTX *ssl_ctx,
348 X509_STORE *ca_store, unsigned accept_deleg)
349{
350 brix_gssapi_srv_t *g;
351 ngx_pool_cleanup_t *cln;
352
353 g = ngx_pcalloc(pool, sizeof(*g));
354 if (g == NULL) {
355 return NULL;
356 }
357 g->pool = pool;
358 g->log = log;
359 g->ca_store = ca_store;
360 g->accept_deleg = accept_deleg;
361 g->state = ST_TLS;
362
363 g->ssl = SSL_new(ssl_ctx);
364 if (g->ssl == NULL) {
365 return NULL;
366 }
367 g->rbio = BIO_new(BIO_s_mem());
368 g->wbio = BIO_new(BIO_s_mem());
369 if (g->rbio == NULL || g->wbio == NULL) {
370 if (g->rbio) { BIO_free(g->rbio); }
371 if (g->wbio) { BIO_free(g->wbio); }
372 SSL_free(g->ssl);
373 g->ssl = NULL;
374 return NULL;
375 }
376 SSL_set_bio(g->ssl, g->rbio, g->wbio); /* ssl takes ownership of BIOs */
377 SSL_set_accept_state(g->ssl);
378 /* Pin TLS 1.2: the GSI ADAT state machine expects the acceptor's
379 * CCS+Finished as the final 335/235 token; TLS 1.3's flight shape breaks it. */
380 SSL_set_max_proto_version(g->ssl, TLS1_2_VERSION);
381 SSL_set_verify(g->ssl, SSL_VERIFY_PEER, gss_verify_cb);
382
383 cln = ngx_pool_cleanup_add(pool, 0);
384 if (cln == NULL) {
385 SSL_free(g->ssl);
386 g->ssl = NULL;
387 return NULL;
388 }
389 cln->handler = gss_cleanup;
390 cln->data = g;
391
392 return g;
393}

396brix_gss_status_e
397brix_gssapi_srv_step(brix_gssapi_srv_t *g, const u_char *in, size_t in_len,
398 ngx_str_t *out)
399{
400 int r, e;
401
402 out->len = 0;
403 out->data = NULL;
404
405 if (g->state == ST_FAIL || g->state == ST_DONE) {
406 return g->state == ST_DONE ? BRIX_GSS_COMPLETE : BRIX_GSS_FAILED;
407 }
408
409 if (in_len > 0 && BIO_write(g->rbio, in, (int) in_len) <= 0) {
410 g->state = ST_FAIL;
411 return BRIX_GSS_FAILED;
412 }
413
414 if (g->state == ST_TLS) {
415 r = SSL_accept(g->ssl);
416 if (r == 1) {
417 if (gss_verify_peer(g) != NGX_OK) {
418 g->state = ST_FAIL;
419 return BRIX_GSS_FAILED;
420 }
421 if (gss_drain(g, out) != NGX_OK) { /* server's final flight */
422 g->state = ST_FAIL;
423 return BRIX_GSS_FAILED;
424 }
425 if (g->accept_deleg) {
426 g->state = ST_WAIT_D;
427 return BRIX_GSS_CONTINUE; /* 335: deliver flight, await 'D' */
428 }
429 g->state = ST_DONE;
430 return BRIX_GSS_COMPLETE;
431 }
432 e = SSL_get_error(g->ssl, r);
433 if (e == SSL_ERROR_WANT_READ || e == SSL_ERROR_WANT_WRITE) {
434 if (gss_drain(g, out) != NGX_OK) {
435 g->state = ST_FAIL;
436 return BRIX_GSS_FAILED;
437 }
438 return BRIX_GSS_CONTINUE;
439 }
440 ngx_log_error(NGX_LOG_ERR, g->log, 0,
441 "brix gsi_mech: TLS handshake failed (SSL_error=%d)", e);
442 g->state = ST_FAIL;
443 return BRIX_GSS_FAILED;
444 }
445
446 if (g->state == ST_WAIT_D) {
447 ngx_str_t dbuf;
448 if (gss_read_all(g, 8, &dbuf) != NGX_OK || dbuf.len == 0
449 || dbuf.data[0] != 'D')
450 {
451 ngx_log_error(NGX_LOG_ERR, g->log, 0,
452 "brix gsi_mech: expected delegation 'D' marker");
453 g->state = ST_FAIL;
454 return BRIX_GSS_FAILED;
455 }
456 if (gss_send_proxy_req(g, out) != BRIX_GSS_CONTINUE) {
457 g->state = ST_FAIL;
458 return BRIX_GSS_FAILED;
459 }
460 return BRIX_GSS_CONTINUE;
461 }
462
463 if (g->state == ST_WAIT_SIGNED) {
464 brix_gss_status_e s = gss_recv_signed(g, out);
465 if (s != BRIX_GSS_COMPLETE) {
466 g->state = ST_FAIL;
467 }
468 return s;
469 }
470
471 g->state = ST_FAIL;
472 return BRIX_GSS_FAILED;
473}
