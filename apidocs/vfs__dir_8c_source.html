1/*
2 * vfs_dir.c — VFS directory enumeration (opendir/readdir/closedir).
3 *
4 * WHAT: Implements brix_vfs_opendir(), brix_vfs_readdir(), and
5 * brix_vfs_closedir() over the opaque brix_vfs_dir_t handle. readdir
6 * yields one entry per call as a pooled, NUL-terminated ngx_str_t, with an
7 * optional lstat of the child filled into an brix_vfs_stat_t.
8 *
9 * WHY: Directory listing (XRootD kXR_dirlist, WebDAV PROPFIND, S3 LIST) needs
10 * confinement, the "." / ".." filter, and a NGX_DONE end-of-stream signal
11 * to be handled once, the same way, for every protocol — rather than each
12 * front end driving opendir/readdir itself.
13 *
14 * HOW: opendir re-verifies confinement, pcalloc's the handle on ctx->pool,
15 * dups the resolved path, and opens the C-library DIR*; the open itself is
16 * observed as BRIX_METRIC_OP_DIRLIST. readdir loops skipping "."/".."
17 * (and entries are distinguished from a real error via errno-cleared
18 * readdir, returning NGX_DONE when the stream ends), copies the name into
19 * the pool, and optionally builds "<dir>/<name>" for an lstat. closedir
20 * calls closedir(3) and nulls the handle so it is idempotent.
21 */
22#include "vfs_internal.h"
23#include "core/compat/log_diag.h"

25/* vfs_opendir_fail — shared error tail for the opendir body.
26 *
27 * WHAT: Reports a failed opendir: copies `err` into *err_out (when the caller
28 * asked for it), meters the failure as OP_DIRLIST when `observe` is set,
29 * and returns NULL for the caller to relay.
30 * WHY: Every early-out of the opendir body ends the same way; folding the
31 * report into one helper keeps the body's control flow flat.
32 * HOW: Pure reporting — errno is deliberately NOT touched here so each call
33 * site preserves exactly the errno state the pre-refactor code left; the
34 * observed path is the ctx's resolved path (what every caller reports). */
35static brix_vfs_dir_t *
36vfs_opendir_fail(brix_vfs_ctx_t *ctx, int err, int observe, uint64_t start,
37 int *err_out)
38{
39 if (err_out != NULL) {
40 *err_out = err;
41 }
42 if (observe) {
43 brix_vfs_observe_ctx_op(ctx, brix_vfs_ctx_path(ctx),
44 BRIX_METRIC_OP_DIRLIST, NULL, 0,
45 NGX_ERROR, err, start);
46 }
47 return NULL;
48}

50/* vfs_dir_route — the SECURITY SEAM: export confinement + plane selection.
51 *
52 * WHAT: Verifies the resolved ctx path is confined to the export root, then
53 * selects the enumeration plane: a non-POSIX driver iterator (*drv_out
54 * set) or the confined POSIX/broker opendir (*drv_out NULL).
55 * WHY: Confinement MUST be re-verified before any directory is opened, and
56 * the driver-vs-broker decision must live in exactly one place so no
57 * future path can enumerate an unconfined directory.
58 * HOW: brix_vfs_require_confined() gates everything; on failure *err carries
59 * its errno for the caller's report (errno itself is left untouched). */
60static ngx_int_t
61vfs_dir_route(brix_vfs_ctx_t *ctx, const brix_sd_driver_t **drv_out, int *err)
62{
63 if (brix_vfs_require_confined(ctx) != NGX_OK) {
64 *err = errno;
65 return NGX_ERROR;
66 }
67 *drv_out = brix_vfs_ctx_driver(ctx);
68 return NGX_OK;
69}

71/* vfs_dir_open_driver — open a non-POSIX backend's directory iterator.
72 *
73 * WHAT: Resolves the export-relative logical path, loads the per-user backend
74 * credential when the export is credential-scoped, and opens the leaf
75 * driver's directory iterator into dh->sd_dir, filling the handle's
76 * driver-plane fields.
77 * WHY: Object/remote backends have no DIR*; enumeration goes through the
78 * driver's opendir/readdir verbs instead of the confined POSIX path.
79 * HOW: Returns NGX_OK, or NGX_ERROR with *err set to the errno to report
80 * (errno itself only set where the pre-refactor code set it: the
81 * credential-load failure). Dispatches on the leaf instance so
82 * brix_sd_opendir_maybe_cred finds the leaf driver's opendir_cred slot
83 * (decorators have only plain relays). */
84static ngx_int_t
85vfs_dir_open_driver(brix_vfs_ctx_t *ctx, brix_vfs_dir_t *dh,
86 const brix_sd_driver_t *drv, const char *path, int *err_out)
87{
88 const char *logical = brix_vfs_export_relative(ctx, path);
89 brix_sd_ucred_t store;
90 brix_sd_cred_t cred;
91 int use_cred = 0, cred_err = 0;
92 int err = 0;
93
94 /* Zero before the gate: it fills only the active credential kind; an
95 * unzeroed cred hands a garbage inactive pointer to the driver's cred
96 * slot (bearer PASSTHROUGH would leave x509_proxy dangling). */
97 ngx_memzero(&cred, sizeof(cred));
98
99 if (brix_vfs_cred_gate_active(ctx)) {
100 if (brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err)
101 != NGX_OK)
102 {
103 *err_out = cred_err ? cred_err : EACCES;
104 errno = *err_out;
105 return NGX_ERROR;
106 }
107 }
108
109 if (drv->opendir == NULL) {
110 errno = ENOTSUP;
111 }
112 dh->sd_dir = (drv->opendir != NULL)
113 ? brix_sd_opendir_maybe_cred(brix_vfs_ns_leaf(ctx->sd),
114 logical, &err, use_cred ? &cred : NULL)
115 : NULL;
116 if (dh->sd_dir == NULL) {
117 *err_out = (err != 0) ? err : errno;
118 return NGX_ERROR;
119 }
120
121 dh->sd = ctx->sd;
122 dh->drv = drv;
123 dh->sd_logical = brix_vfs_copy_path(ctx->pool, logical);
124 dh->pool = ctx->pool;
125 dh->log = ctx->log;
126 return NGX_OK;
127}

129/* vfs_dir_open_confined — the confined POSIX/broker opendir (do NOT weaken).
130 *
131 * WHAT: Opens dh->dir under export-root confinement and fills the handle's
132 * POSIX-plane fields.
133 * WHY: brix_opendir_confined_canon is the openat2 RESOLVE_IN_ROOT path from
134 * the symlink-escape fix — an outward symlink inside an export must stay
135 * invisible — and under impersonation it opens AS THE MAPPED USER
136 * (broker fdopendir) so a 0700 user-owned / 0770 group-restricted dir
137 * the unprivileged worker cannot itself open is enumerable by its
138 * legitimate owner/group-member; off impersonation it is a bare
139 * opendir().
140 * HOW: Returns NGX_OK, or NGX_ERROR with errno left exactly as the confined
141 * open set it. */
142static ngx_int_t
143vfs_dir_open_confined(brix_vfs_ctx_t *ctx, brix_vfs_dir_t *dh,
144 const char *path)
145{
146 dh->dir = brix_opendir_confined_canon(ctx->log, ctx->root_canon, path);
147 if (dh->dir == NULL) {
148 return NGX_ERROR;
149 }
150
151 dh->pool = ctx->pool;
152 dh->log = ctx->log;
153 dh->root_canon = ctx->root_canon;
154 return NGX_OK;
155}

157/* Shared opendir body. When `observe` is set the open is metered as OP_DIRLIST;
158 * the quiet variant (observe=0) skips the metric/access-log entirely — for bulk
159 * recursive walks (S3 ListObjects, WebDAV SEARCH) whose enclosing protocol op
160 * already accounts for the traversal and would otherwise emit one phantom
161 * OP_DIRLIST per visited subdirectory. */
162static brix_vfs_dir_t *
163brix_vfs_opendir_impl(brix_vfs_ctx_t *ctx, int *err_out, int observe)
164{
165 brix_vfs_dir_t *dh;
166 const char *path;
167 const brix_sd_driver_t *drv = NULL;
168 uint64_t start;
169 int err = 0;
170
171 start = brix_vfs_now_ns();
172
173 if (err_out != NULL) {
174 *err_out = 0;
175 }
176
177 if (vfs_dir_route(ctx, &drv, &err) != NGX_OK) {
178 return vfs_opendir_fail(ctx, err, observe, start, err_out);
179 }
180
181 path = brix_vfs_ctx_path(ctx);
182 dh = ngx_pcalloc(ctx->pool, sizeof(*dh));
183 if (dh == NULL) {
184 errno = ENOMEM;
185 return vfs_opendir_fail(ctx, ENOMEM, observe, start, err_out);
186 }
187
188 dh->path = brix_vfs_copy_path(ctx->pool, path);
189 if (dh->path == NULL) {
190 return vfs_opendir_fail(ctx, errno, observe, start, err_out);
191 }
192
193 if (drv != NULL) {
194 /* Non-POSIX backend: enumerate through the driver's iterator. */
195 if (vfs_dir_open_driver(ctx, dh, drv, path, &err) != NGX_OK) {
196 return vfs_opendir_fail(ctx, err, observe, start, err_out);
197 }
198 } else if (vfs_dir_open_confined(ctx, dh, path) != NGX_OK) {
199 return vfs_opendir_fail(ctx, errno, observe, start, err_out);
200 }
201
202 if (observe) {
203 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DIRLIST, NULL, 0,
204 NGX_OK, 0, start);
205 }
206 return dh;
207}

234/* vfs_readdir_fill_entry — copy an entry name into the pooled name_out.
235 *
236 * WHAT: Fills *name_out with a pooled, NUL-terminated copy of `name`.
237 * WHY: Both readdir variants, on both planes, end by handing the caller a
238 * pool-owned copy of the entry name; one helper kills the copy-paste.
239 * HOW: ngx_pnalloc(len+1) + memcpy + explicit NUL; NGX_ERROR with
240 * errno=ENOMEM on allocation failure. */
241static ngx_int_t
242vfs_readdir_fill_entry(ngx_pool_t *pool, const char *name,
243 ngx_str_t *name_out)
244{
245 name_out->len = ngx_strlen(name);
246 name_out->data = ngx_pnalloc(pool, name_out->len + 1);
247 if (name_out->data == NULL) {
248 errno = ENOMEM;
249 return NGX_ERROR;
250 }
251 ngx_memcpy(name_out->data, name, name_out->len);
252 name_out->data[name_out->len] = '\0';
253 return NGX_OK;
254}

256/* vfs_readdir_next_posix — next POSIX dirent, "." / ".." filtered out.
257 *
258 * WHAT: Yields the next non-dot entry of dh->dir into *de_out.
259 * WHY: Both readdir variants share the errno-cleared readdir(3) idiom that
260 * distinguishes end-of-stream from a real error, and the dot filter.
261 * HOW: errno=0 before each readdir(3): NULL with errno still 0 is NGX_DONE,
262 * NULL with errno set is NGX_ERROR; "." and ".." are skipped in-loop. */
263static ngx_int_t
264vfs_readdir_next_posix(brix_vfs_dir_t *dh, struct dirent **de_out)
265{
266 struct dirent *de;
267
268 for ( ;; ) {
269 errno = 0;
270 de = readdir(dh->dir);
271 if (de == NULL) {
272 return errno == 0 ? NGX_DONE : NGX_ERROR;
273 }
274
275 if (de->d_name[0] == '.'
276 && (de->d_name[1] == '\0'
277 || (de->d_name[1] == '.' && de->d_name[2] == '\0')))
278 {
279 continue;
280 }
281
282 *de_out = de;
283 return NGX_OK;
284 }
285}

287/* vfs_readdir_stat_child — confined lstat of one POSIX entry.
288 *
289 * WHAT: Joins "<dir>/<name>", lstats it under export confinement, and fills
290 * *stat_out on success.
291 * WHY: Per-entry stat must SKIP a bad entry rather than truncate the listing;
292 * returning NGX_ERROR here lets the scan loop drop the entry and go on.
293 * HOW: snprintf join (unrepresentable path → NGX_ERROR), then
294 * brix_lstat_confined_canon AS THE MAPPED USER (broker-routed) so
295 * enumerating a group-restricted dir does not EACCES per child as the
296 * worker. lstat (not stat) keeps outward symlinks unfollowed. */
297static ngx_int_t
298vfs_readdir_stat_child(brix_vfs_dir_t *dh, const char *name,
299 brix_vfs_stat_t *stat_out)
300{
301 char child[PATH_MAX];
302 struct stat st;
303 int n;
304
305 n = snprintf(child, sizeof(child), "%s/%s", dh->path, name);
306 if (n < 0 || (size_t) n >= sizeof(child)) {
307 return NGX_ERROR;
308 }
309 if (brix_lstat_confined_canon(dh->log, dh->root_canon, child,
310 &st, 1) != 0) {
311 return NGX_ERROR;
312 }
313 brix_vfs_fill_stat(&st, stat_out);
314 return NGX_OK;
315}

317/* vfs_sd_stat_child — stat one driver-plane entry through the driver.
318 *
319 * WHAT: Joins the handle's logical directory with `name` and stats the child
320 * via dh->drv->stat into *sd_st.
321 * WHY: Both readdir variants build the same "<logical>/<name>" join (with
322 * the "/"-root special case) before the driver stat.
323 * HOW: Fixed PATH_MAX join via ngx_snprintf, then the driver's stat verb;
324 * the caller must have checked dh->drv->stat != NULL. */
325static ngx_int_t
326vfs_sd_stat_child(brix_vfs_dir_t *dh, const char *name, brix_sd_stat_t *sd_st)
327{
328 char child[PATH_MAX];
329
330 ngx_snprintf((u_char *) child, sizeof(child), "%s/%s%Z",
331 (dh->sd_logical[0] == '/' && dh->sd_logical[1] == '\0')
332 ? "" : dh->sd_logical, name);
333 return dh->drv->stat(dh->sd, child, sd_st);
334}

336/* vfs_readdir_sd — driver-plane body of brix_vfs_readdir.
337 *
338 * WHAT: Pulls the next entry from the driver iterator, stats the child
339 * through the same driver when the caller wants stats, and fills the
340 * pooled name.
341 * WHY: Keeps the driver iterator's skip-on-vanished-child scan out of the
342 * public entry point.
343 * HOW: Loops (instead of the former tail-recursion) so a child that vanished
344 * mid-scan is skipped; only NGX_DONE/NGX_ERROR from the iterator stop
345 * the caller's loop. */
346static ngx_int_t
347vfs_readdir_sd(brix_vfs_dir_t *dh, ngx_str_t *name_out,
348 brix_vfs_stat_t *stat_out)
349{
350 brix_sd_dirent_t de_sd;
351 brix_sd_stat_t sd_st;
352 ngx_int_t rc;
353
354 for ( ;; ) {
355 rc = dh->drv->readdir(dh->sd_dir, &de_sd);
356 if (rc != NGX_OK) {
357 return rc; /* NGX_DONE / NGX_ERROR */
358 }
359 if (stat_out == NULL || dh->drv->stat == NULL) {
360 break;
361 }
362 if (vfs_sd_stat_child(dh, de_sd.name, &sd_st) == NGX_OK) {
363 brix_vfs_sd_stat_fill(&sd_st, stat_out);
364 break;
365 }
366 /* Child vanished mid-scan: skip it and keep enumerating. */
367 }
368
369 return vfs_readdir_fill_entry(dh->pool, de_sd.name, name_out);
370}

372/* Return the next entry: name as a pooled NUL-terminated ngx_str_t, plus an
373 * optional lstat of the child. Skips "." and ".."; returns NGX_DONE at
374 * end-of-stream and NGX_ERROR (errno set) on failure. */
375ngx_int_t
376brix_vfs_readdir(brix_vfs_dir_t *dh, ngx_str_t *name_out,
377 brix_vfs_stat_t *stat_out)
378{
379 struct dirent *de;
380 ngx_int_t rc;
381
382 if (dh == NULL || name_out == NULL
383 || (dh->dir == NULL && dh->sd_dir == NULL))
384 {
385 errno = EINVAL;
386 return NGX_ERROR;
387 }
388
389 /* Non-POSIX backend: enumerate through the driver iterator. */
390 if (dh->sd_dir != NULL) {
391 return vfs_readdir_sd(dh, name_out, stat_out);
392 }
393
394 /* Per-entry stat is folded into the scan so a single bad entry SKIPS
395 * rather than truncating the listing: a child that races an unlink
396 * (ENOENT), or whose joined path is unrepresentable, is dropped and the
397 * scan continues. (Only NGX_DONE/NGX_ERROR — true end/stream error —
398 * stop the caller's loop.) */
399 for ( ;; ) {
400 rc = vfs_readdir_next_posix(dh, &de);
401 if (rc != NGX_OK) {
402 return rc;
403 }
404 if (stat_out == NULL
405 || vfs_readdir_stat_child(dh, de->d_name, stat_out) == NGX_OK)
406 {
407 break;
408 }
409 }
410
411 return vfs_readdir_fill_entry(dh->pool, de->d_name, name_out);
412}

414/* Yield the next entry's name plus its KIND from the readdir d_type, with no
415 * per-entry stat — for callers that classify dir-vs-file on the fast path and
416 * only stat (via brix_vfs_probe) on a DT_UNKNOWN filesystem. Skips "."/"..";
417 * NGX_DONE at end-of-stream, NGX_ERROR (errno set) on failure. */
418/* vfs_sd_entry_kind — classify one driver-plane entry as dir/file.
419 *
420 * WHAT: Returns the entry's kind from a driver stat of the joined child.
421 * WHY: Driver dirents carry no d_type; kind comes from the driver's stat
422 * verb, and a driver without one (or a failed stat) yields UNKNOWN so
423 * the caller falls back to brix_vfs_probe.
424 * HOW: Guards dh->drv->stat, joins via vfs_sd_stat_child, and maps
425 * is_dir → DT_DIR / DT_REG. */
426static brix_vfs_dirent_kind_t
427vfs_sd_entry_kind(brix_vfs_dir_t *dh, const char *name)
428{
429 brix_sd_stat_t sd_st;
430
431 if (dh->drv->stat == NULL
432 || vfs_sd_stat_child(dh, name, &sd_st) != NGX_OK)
433 {
434 return BRIX_VFS_DT_UNKNOWN;
435 }
436 return sd_st.is_dir ? BRIX_VFS_DT_DIR : BRIX_VFS_DT_REG;
437}

439/* vfs_posix_dtype_kind — map a readdir d_type onto the VFS entry kind.
440 *
441 * WHAT: DT_DIR/DT_REG/DT_UNKNOWN map onto their VFS kinds; anything else
442 * (symlink, fifo, socket, ...) is DT_OTHER.
443 * WHY: Callers classify dir-vs-file on the fast path and only stat on a
444 * DT_UNKNOWN filesystem; the mapping is a pure table.
445 * HOW: A single switch — no side effects. */
446static brix_vfs_dirent_kind_t

448{
449 switch (d_type) {

451 case DT_REG: return BRIX_VFS_DT_REG;
452 case DT_UNKNOWN: return BRIX_VFS_DT_UNKNOWN;

454 }

496/* Close the directory stream and null the handle (idempotent). Logs and returns
497 * NGX_ERROR if closedir(3) fails. */
498ngx_int_t
499brix_vfs_closedir(brix_vfs_dir_t *dh, ngx_log_t *log)
500{
501 if (dh == NULL || (dh->dir == NULL && dh->sd_dir == NULL)) {
502 return NGX_OK;
503 }
504
505 if (dh->sd_dir != NULL) {
506 ngx_int_t rc = (dh->drv->closedir != NULL)
507 ? dh->drv->closedir(dh->sd_dir) : NGX_OK;
508
509 dh->sd_dir = NULL;
510 return rc;
511 }
512
513 if (closedir(dh->dir) != 0) {
514 BRIX_DIAG_ERR(log != NULL ? log : dh->log, errno,
515 "xrootd[disk]: closedir failed for \"%s\"",
516 "the underlying directory stream returned an error on close — "
517 "usually an I/O error on the backing storage",
518 "check dmesg and the filesystem health for that path; the OS "
519 "reason is appended below",
520 dh->path != NULL ? dh->path : "-");
521 dh->dir = NULL;
522 return NGX_ERROR;
523 }
524
525 dh->dir = NULL;
526 return NGX_OK;
527}
