1/*
2 * vfs_dir.c — VFS directory enumeration (opendir/readdir/closedir).
3 *
4 * WHAT: Implements brix_vfs_opendir(), brix_vfs_readdir(), and
5 * brix_vfs_closedir() over the opaque brix_vfs_dir_t handle. readdir
6 * yields one entry per call as a pooled, NUL-terminated ngx_str_t, with an
7 * optional lstat of the child filled into an brix_vfs_stat_t.
8 *
9 * WHY: Directory listing (XRootD kXR_dirlist, WebDAV PROPFIND, S3 LIST) needs
10 * confinement, the "." / ".." filter, and a NGX_DONE end-of-stream signal
11 * to be handled once, the same way, for every protocol — rather than each
12 * front end driving opendir/readdir itself.
13 *
14 * HOW: opendir re-verifies confinement, pcalloc's the handle on ctx->pool,
15 * dups the resolved path, and opens the C-library DIR*; the open itself is
16 * observed as BRIX_METRIC_OP_DIRLIST. readdir loops skipping "."/".."
17 * (and entries are distinguished from a real error via errno-cleared
18 * readdir, returning NGX_DONE when the stream ends), copies the name into
19 * the pool, and optionally builds "<dir>/<name>" for an lstat. closedir
20 * calls closedir(3) and nulls the handle so it is idempotent.
21 */
22#include "vfs_internal.h"
23#include "core/compat/log_diag.h"

25/* Shared opendir body. When `observe` is set the open is metered as OP_DIRLIST;
26 * the quiet variant (observe=0) skips the metric/access-log entirely — for bulk
27 * recursive walks (S3 ListObjects, WebDAV SEARCH) whose enclosing protocol op
28 * already accounts for the traversal and would otherwise emit one phantom
29 * OP_DIRLIST per visited subdirectory. */
30static brix_vfs_dir_t *
31brix_vfs_opendir_impl(brix_vfs_ctx_t *ctx, int *err_out, int observe)
32{
33 brix_vfs_dir_t *dh;
34 const char *path;
35 uint64_t start;
36 int saved_errno;
37
38 start = brix_vfs_now_ns();
39
40 if (err_out != NULL) {
41 *err_out = 0;
42 }
43
44 if (brix_vfs_require_confined(ctx) != NGX_OK) {
45 saved_errno = errno;
46 if (err_out != NULL) {
47 *err_out = errno;
48 }
49 if (observe) {
50 brix_vfs_observe_ctx_op(ctx, brix_vfs_ctx_path(ctx),
51 BRIX_METRIC_OP_DIRLIST, NULL, 0,
52 NGX_ERROR, saved_errno, start);
53 }
54 return NULL;
55 }
56
57 path = brix_vfs_ctx_path(ctx);
58 dh = ngx_pcalloc(ctx->pool, sizeof(*dh));
59 if (dh == NULL) {
60 errno = ENOMEM;
61 saved_errno = errno;
62 if (err_out != NULL) {
63 *err_out = errno;
64 }
65 if (observe) {
66 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DIRLIST, NULL,
67 0, NGX_ERROR, saved_errno, start);
68 }
69 return NULL;
70 }
71
72 dh->path = brix_vfs_copy_path(ctx->pool, path);
73 if (dh->path == NULL) {
74 saved_errno = errno;
75 if (err_out != NULL) {
76 *err_out = errno;
77 }
78 if (observe) {
79 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DIRLIST, NULL,
80 0, NGX_ERROR, saved_errno, start);
81 }
82 return NULL;
83 }
84
85 /* Non-POSIX backend: enumerate through the driver's directory iterator. */
86 {
87 const brix_sd_driver_t *drv = brix_vfs_ctx_driver(ctx);
88
89 if (drv != NULL) {
90 const char *logical = brix_vfs_export_relative(ctx, path);
91 int err = 0;
92
93 if (drv->opendir == NULL) {
94 errno = ENOTSUP;
95 }
96 dh->sd_dir = (drv->opendir != NULL)
97 ? drv->opendir(ctx->sd, logical, &err) : NULL;
98 if (dh->sd_dir == NULL) {
99 saved_errno = (err != 0) ? err : errno;
100 if (err_out != NULL) { *err_out = saved_errno; }
101 if (observe) {
102 brix_vfs_observe_ctx_op(ctx, path,
103 BRIX_METRIC_OP_DIRLIST, NULL, 0, NGX_ERROR,
104 saved_errno, start);
105 }
106 return NULL;
107 }
108 dh->sd = ctx->sd;
109 dh->drv = drv;
110 dh->sd_logical = brix_vfs_copy_path(ctx->pool, logical);
111 dh->pool = ctx->pool;
112 dh->log = ctx->log;
113 if (observe) {
114 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DIRLIST,
115 NULL, 0, NGX_OK, 0, start);
116 }
117 return dh;
118 }
119 }

121 /* Open the directory AS THE MAPPED USER under impersonation (broker fdopendir)
122 * so a 0700 user-owned / 0770 group-restricted dir the unprivileged worker
123 * cannot itself open is enumerable by its legitimate owner/group-member; off
124 * impersonation this is a bare opendir(). */
125 dh->dir = brix_opendir_confined_canon(ctx->log, ctx->root_canon, path);
126 if (dh->dir == NULL) {
127 saved_errno = errno;
128 if (err_out != NULL) {
129 *err_out = errno;
130 }
131 if (observe) {
132 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DIRLIST, NULL,
133 0, NGX_ERROR, saved_errno, start);
134 }
135 return NULL;
136 }
137
138 dh->pool = ctx->pool;
139 dh->log = ctx->log;
140 dh->root_canon = ctx->root_canon;
141
142 if (observe) {
143 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DIRLIST, NULL, 0,
144 NGX_OK, 0, start);
145 }
146 return dh;
147}

174/* Return the next entry: name as a pooled NUL-terminated ngx_str_t, plus an
175 * optional lstat of the child. Skips "." and ".."; returns NGX_DONE at
176 * end-of-stream and NGX_ERROR (errno set) on failure. */
177ngx_int_t
178brix_vfs_readdir(brix_vfs_dir_t *dh, ngx_str_t *name_out,
179 brix_vfs_stat_t *stat_out)
180{
181 struct dirent *de;
182
183 if (dh == NULL || name_out == NULL
184 || (dh->dir == NULL && dh->sd_dir == NULL))
185 {
186 errno = EINVAL;
187 return NGX_ERROR;
188 }
189
190 /* Non-POSIX backend: pull the next entry from the driver iterator and stat the
191 * child through the same driver (skip a child that vanished mid-scan). */
192 if (dh->sd_dir != NULL) {
193 brix_sd_dirent_t de_sd;
194 ngx_int_t rc = dh->drv->readdir(dh->sd_dir, &de_sd);
195
196 if (rc != NGX_OK) {
197 return rc; /* NGX_DONE / NGX_ERROR */
198 }
199 if (stat_out != NULL && dh->drv->stat != NULL) {
200 char child[PATH_MAX];
201 brix_sd_stat_t sd_st;
202
203 ngx_snprintf((u_char *) child, sizeof(child), "%s/%s%Z",
204 (dh->sd_logical[0] == '/' && dh->sd_logical[1] == '\0')
205 ? "" : dh->sd_logical, de_sd.name);
206 if (dh->drv->stat(dh->sd, child, &sd_st) != NGX_OK) {
207 return brix_vfs_readdir(dh, name_out, stat_out); /* skip */
208 }
209 brix_vfs_sd_stat_fill(&sd_st, stat_out);
210 }
211 name_out->len = ngx_strlen(de_sd.name);
212 name_out->data = ngx_pnalloc(dh->pool, name_out->len + 1);
213 if (name_out->data == NULL) {
214 errno = ENOMEM;
215 return NGX_ERROR;
216 }
217 ngx_memcpy(name_out->data, de_sd.name, name_out->len + 1);
218 return NGX_OK;
219 }
220
221 for ( ;; ) {
222 errno = 0;
223 de = readdir(dh->dir);
224 if (de == NULL) {
225 return errno == 0 ? NGX_DONE : NGX_ERROR;
226 }
227
228 if (de->d_name[0] == '.'
229 && (de->d_name[1] == '\0'
230 || (de->d_name[1] == '.' && de->d_name[2] == '\0')))
231 {
232 continue;
233 }
234
235 /* Per-entry stat is folded into the scan so a single bad entry SKIPS
236 * rather than truncating the listing: a child that races an unlink
237 * (ENOENT), or whose joined path is unrepresentable, is dropped and the
238 * scan continues. (Only NGX_DONE/NGX_ERROR — true end/stream error —
239 * stop the caller's loop.) */
240 if (stat_out != NULL) {
241 char child[PATH_MAX];
242 struct stat st;
243 int n;
244
245 n = snprintf(child, sizeof(child), "%s/%s", dh->path, de->d_name);
246 if (n < 0 || (size_t) n >= sizeof(child)) {
247 continue;
248 }
249 /* lstat the child AS THE MAPPED USER (broker-routed) so enumerating a
250 * group-restricted dir does not EACCES per child as the worker. */
251 if (brix_lstat_confined_canon(dh->log, dh->root_canon, child,
252 &st, 1) != 0) {
253 continue;
254 }
255 brix_vfs_fill_stat(&st, stat_out);
256 }
257
258 break;
259 }
260
261 name_out->len = strlen(de->d_name);
262 name_out->data = ngx_pnalloc(dh->pool, name_out->len + 1);
263 if (name_out->data == NULL) {
264 errno = ENOMEM;
265 return NGX_ERROR;
266 }
267 ngx_memcpy(name_out->data, de->d_name, name_out->len);
268 name_out->data[name_out->len] = '\0';
269
270 return NGX_OK;
271}

273/* Yield the next entry's name plus its KIND from the readdir d_type, with no
274 * per-entry stat — for callers that classify dir-vs-file on the fast path and
275 * only stat (via brix_vfs_probe) on a DT_UNKNOWN filesystem. Skips "."/"..";
276 * NGX_DONE at end-of-stream, NGX_ERROR (errno set) on failure. */
277ngx_int_t
278brix_vfs_readdir_kind(brix_vfs_dir_t *dh, ngx_str_t *name_out,
279 brix_vfs_dirent_kind_t *kind_out)
280{
281 struct dirent *de;
282
283 if (dh == NULL || name_out == NULL
284 || (dh->dir == NULL && dh->sd_dir == NULL))
285 {
286 errno = EINVAL;
287 return NGX_ERROR;
288 }
289
290 if (dh->sd_dir != NULL) {
291 brix_sd_dirent_t de_sd;
292 ngx_int_t rc = dh->drv->readdir(dh->sd_dir, &de_sd);
293
294 if (rc != NGX_OK) {
295 return rc;
296 }
297 if (kind_out != NULL) {
298 char child[PATH_MAX];
299 brix_sd_stat_t sd_st;
300
301 ngx_snprintf((u_char *) child, sizeof(child), "%s/%s%Z",
302 (dh->sd_logical[0] == '/' && dh->sd_logical[1] == '\0')
303 ? "" : dh->sd_logical, de_sd.name);
304 *kind_out = (dh->drv->stat != NULL
305 && dh->drv->stat(dh->sd, child, &sd_st) == NGX_OK)
306 ? (sd_st.is_dir ? BRIX_VFS_DT_DIR : BRIX_VFS_DT_REG)
307 : BRIX_VFS_DT_UNKNOWN;
308 }
309 name_out->len = ngx_strlen(de_sd.name);
310 name_out->data = ngx_pnalloc(dh->pool, name_out->len + 1);
311 if (name_out->data == NULL) {
312 errno = ENOMEM;
313 return NGX_ERROR;
314 }
315 ngx_memcpy(name_out->data, de_sd.name, name_out->len + 1);
316 return NGX_OK;
317 }
318
319 for ( ;; ) {
320 errno = 0;
321 de = readdir(dh->dir);
322 if (de == NULL) {
323 return errno == 0 ? NGX_DONE : NGX_ERROR;
324 }
325
326 if (de->d_name[0] == '.'
327 && (de->d_name[1] == '\0'
328 || (de->d_name[1] == '.' && de->d_name[2] == '\0')))
329 {
330 continue;
331 }
332 break;
333 }
334
335 if (kind_out != NULL) {
336 switch (de->d_type) {
337 case DT_DIR: *kind_out = BRIX_VFS_DT_DIR; break;
338 case DT_REG: *kind_out = BRIX_VFS_DT_REG; break;
339 case DT_UNKNOWN: *kind_out = BRIX_VFS_DT_UNKNOWN; break;
340 default: *kind_out = BRIX_VFS_DT_OTHER; break;
341 }
342 }
343
344 name_out->len = strlen(de->d_name);
345 name_out->data = ngx_pnalloc(dh->pool, name_out->len + 1);
346 if (name_out->data == NULL) {
347 errno = ENOMEM;
348 return NGX_ERROR;
349 }
350 ngx_memcpy(name_out->data, de->d_name, name_out->len);
351 name_out->data[name_out->len] = '\0';
352
353 return NGX_OK;
354}

356/* Close the directory stream and null the handle (idempotent). Logs and returns
357 * NGX_ERROR if closedir(3) fails. */
358ngx_int_t
359brix_vfs_closedir(brix_vfs_dir_t *dh, ngx_log_t *log)
360{
361 if (dh == NULL || (dh->dir == NULL && dh->sd_dir == NULL)) {
362 return NGX_OK;
363 }
364
365 if (dh->sd_dir != NULL) {
366 ngx_int_t rc = (dh->drv->closedir != NULL)
367 ? dh->drv->closedir(dh->sd_dir) : NGX_OK;
368
369 dh->sd_dir = NULL;
370 return rc;
371 }
372
373 if (closedir(dh->dir) != 0) {
374 BRIX_DIAG_ERR(log != NULL ? log : dh->log, errno,
375 "xrootd[disk]: closedir failed for \"%s\"",
376 "the underlying directory stream returned an error on close — "
377 "usually an I/O error on the backing storage",
378 "check dmesg and the filesystem health for that path; the OS "
379 "reason is appended below",
380 dh->path != NULL ? dh->path : "-");
381 dh->dir = NULL;
382 return NGX_ERROR;
383 }
384
385 dh->dir = NULL;
386 return NGX_OK;
387}
