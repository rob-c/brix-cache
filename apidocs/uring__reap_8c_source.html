9/* File: src/core/aio/uring_reap.c — io_uring completion reaper and the UAF-safe
10 * completion-slot table (phase-79 split from uring.c).
11 * WHAT: Owns everything that runs on a completed CQE: the generation-stamped
12 * slot table that maps a CQE cookie back to its in-flight task, the
13 * per-op result translation, and the eventfd-driven harvest loop wired
14 * into the worker's epoll. The ring singleton and bring-up live in
15 * uring.c; detection/gating lives in uring_probe.c.
16 *
17 * WHY: The reaper is the security- and correctness-critical half of the
18 * backend: it must drop stale and orphaned CQEs without ever touching
19 * freed task/connection memory, and post done-callbacks out-of-line so a
20 * re-submitting callback never re-enters the harvest loop.
21 *
22 * HOW: Entirely under #if (BRIX_HAVE_LIBURING) — in a stub build no ring
23 * exists, so none of this is reachable and the file is empty. The slot
24 * acquire/release helpers are also called from the submit path
25 * (uring_submit.c); the eventfd handler is installed by uring.c. */

27/* completion-slot table (UAF-safe task mapping) */
28/* brix_uring_slot_at — bounds-checked slot lookup; NULL if idx is out of
29 * range (a corrupt/forged user_data). */
30static brix_uring_slot_t *
31brix_uring_slot_at(brix_uring_t *u, uint32_t idx)
32{
33 return idx < u->queue_depth ? &u->slots[idx] : NULL;
34}

36/* brix_uring_slot_acquire — claim the first free slot. The selector checks
37 * inflight < queue_depth before calling, so a free slot always exists; the
38 * linear scan over <=4096 entries is negligible. Returns NULL only if the
39 * table is unexpectedly full (caller then falls back to the thread pool).
40 * Non-static: also called from uring_submit.c. */
41brix_uring_slot_t *
42brix_uring_slot_acquire(brix_uring_t *u, uint32_t *idx_out)
43{
44 uint32_t i;
45
46 for (i = 0; i < u->queue_depth; i++) {
47 if (!u->slots[i].in_use) {
48 u->slots[i].in_use = 1;
49 *idx_out = i;
50 return &u->slots[i];
51 }
52 }
53
54 return NULL;
55}

57/* brix_uring_slot_release — free a slot and bump its generation so any later
58 * CQE carrying the old generation is recognised as stale and dropped.
59 * Non-static: also called from uring_submit.c. */
60void
61brix_uring_slot_release(brix_uring_t *u, uint32_t idx)
62{
63 brix_uring_slot_t *s = &u->slots[idx];
64
65 s->in_use = 0;
66 s->task = NULL;
67 s->done_fn = NULL;
68 s->owner = NULL;
69 s->orphaned = 0;
70 s->generation++;
71 (void) u;
72}

74/* brix_uring_orphan_owner — mark every in-flight slot owned by a dying
75 * connection so the reaper drops its late CQE without touching the task (freed
76 * with the connection pool) or posting its completion event. See uring.h. */
77void
78brix_uring_orphan_owner(void *owner)
79{
80 brix_uring_t *u = brix_uring_worker();
81 uint32_t i;
82
83 if (u == NULL || !u->ring_active || owner == NULL) {
84 return;
85 }
86 for (i = 0; i < u->queue_depth; i++) {
87 if (u->slots[i].in_use && u->slots[i].owner == owner) {
88 u->slots[i].orphaned = 1;
89 }
90 }
91}

93/*
94 * brix_uring_apply_cqe — translate cqe->res into the task's OUT fields, the
95 * same fields the worker-thread fn would have set. io_uring reports failures
96 * as a negative -errno in cqe->res (it does NOT set errno).
97 *
98 * Returns 1 when the task is fully complete and its done-callback should be
99 * posted, 0 when more CQEs are still expected for this task (the linked
100 * writev+fsync chain, added in SB-W4). READV/WRITEV cases also land in SB-W4;
101 * READ/WRITE are handled here so the reaper is complete for the hot path.
102 */
103static ngx_int_t
104brix_uring_apply_cqe(brix_uring_slot_t *slot, struct io_uring_cqe *cqe)
105{
106 ngx_thread_task_t *task = slot->task;
107 int32_t res = cqe->res;
108
109 switch (slot->op_kind) {
110
111 case XRD_URING_OP_READ: {
112 brix_read_aio_t *t = task->ctx;
113 if (res < 0) { t->nread = -1; t->io_errno = -res; }
114 else { t->nread = res; t->io_errno = 0; }
115 return 1;
116 }
117
118 case XRD_URING_OP_WRITE: {
119 brix_write_aio_t *t = task->ctx;
120 if (res < 0) { t->nwritten = -1; t->io_errno = -res; }
121 else { t->nwritten = res; t->io_errno = 0; }
122 return 1;
123 }
124
125 case XRD_URING_OP_READV: {
126 brix_readv_aio_t *t = task->ctx;
127 size_t total = 0, i;
128
129 /* Single contiguous group (op_for gated it): one IORING_OP_READV
130 * scattered into the per-segment payload pointers. The wire segment
131 * headers were already written at plan-build, so nothing to fix up here
132 * — only the totals + error state, matching brix_readv_aio_thread. */
133 for (i = 0; i < t->segment_count; i++) {
134 total += t->segments[i].read_length;
135 }
136 if (res < 0) {
137 t->io_error = 1; t->bytes_read_total = 0; t->response_bytes = 0;
138 snprintf(t->err_msg, sizeof(t->err_msg),
139 "readv I/O error: %s", strerror(-res));
140 } else if ((size_t) res != total) {
141 t->io_error = 1; t->bytes_read_total = 0; t->response_bytes = 0;
142 snprintf(t->err_msg, sizeof(t->err_msg), "readv past EOF");
143 } else {
144 t->io_error = 0;
145 t->bytes_read_total = (size_t) res;
146 t->response_bytes = t->segment_count * BRIX_READV_SEGSIZE
147 + (size_t) res;
148 }
149 return 1;
150 }
151
152 case XRD_URING_OP_WRITEV: {
153 brix_writev_aio_t *t = task->ctx;
154 size_t total = 0, i;
155
156 /* Single contiguous same-fd group: one IORING_OP_WRITEV gathering the
157 * segment buffers. io_error encodes the failure kind exactly as
158 * brix_writev_write_aio_thread (1 = hard error, 2 = short write). A
159 * trailing FSYNC (when do_sync) is a separate linked SQE whose CQE the
160 * reaper drops — best-effort, matching the pool path's ignored return. */
161 for (i = 0; i < t->n_segs; i++) {
162 total += t->segs[i].wlen;
163 }
164 if (res < 0) {
165 t->io_error = 1; t->bytes_total = 0;
166 snprintf(t->err_msg, sizeof(t->err_msg),
167 "writev I/O error: %s", strerror(-res));
168 } else if ((size_t) res < total) {
169 t->io_error = 2; t->bytes_total = (size_t) res;
170 snprintf(t->err_msg, sizeof(t->err_msg), "writev short write");
171 } else {
172 t->io_error = 0; t->bytes_total = (size_t) res;
173 }
174 return 1;
175 }
176
177 default:
178 /* pgread/dirlist are never routed to io_uring (op_for -> NONE). */
179 return 1;
180 }
181}

183/*
184 * uring_cqe_retire — acknowledge one CQE and balance the in-flight counter.
185 *
186 * WHAT: Marks the CQE seen (returning its ring slot to the kernel) and
187 * decrements the inflight gauge that the submission selector consults.
188 * WHY: Every CQE — real completion, stale generation, orphan, or trailing
189 * FSYNC — must be retired exactly once; centralising the pair keeps the
190 * accounting impossible to miss on any per-CQE early return.
191 * HOW: io_uring_cqe_seen + a guarded decrement (never underflows).
192 */
193static void
194uring_cqe_retire(brix_uring_t *u, struct io_uring_cqe *cqe)
195{
196 io_uring_cqe_seen(&u->ring, cqe);
197 if (u->inflight > 0) { u->inflight--; }
198}

200/*
201 * uring_complete_one — classify and complete a single harvested CQE.
202 *
203 * WHAT: The per-CQE body of the completion reaper: decode the slot cookie,
204 * run the generation/orphan guards, translate the result into the task
205 * and post its done-callback.
206 * WHY: Extracted from brix_uring_eventfd_handler so the reaper loop stays a
207 * trivial drain and each guard reads as one early return.
208 * HOW: Early-return ladder — trailing-FSYNC drop, stale-generation drop,
209 * orphaned-owner drop (slot recycled, task never touched), then the
210 * normal translate + ngx_post_event path. Every exit retires the CQE
211 * via uring_cqe_retire; done-callbacks are never invoked inline.
212 */
213static void
214uring_complete_one(brix_uring_t *u, struct io_uring_cqe *cqe)
215{
216 uint64_t ud = io_uring_cqe_get_data64(cqe);
217 uint32_t idx;
218 uint32_t gen;
219 brix_uring_slot_t *slot;
220
221 /* 2-pre. trailing FSYNC of a linked writev+fsync chain: carries no slot;
222 * fsync is best-effort (the pool path ignores its return), so drop the
223 * CQE and just balance inflight. */
224 if (ud == BRIX_URING_FSYNC_COOKIE) {
225 uring_cqe_retire(u, cqe);
226 return;
227 }
228
229 idx = (uint32_t) (ud & 0xffffffffULL);
230 gen = (uint32_t) (ud >> 32);
231 slot = brix_uring_slot_at(u, idx);
232
233 /* 2a. generation guard: a stale CQE for a recycled/dead slot is dropped
234 * safely. (The done-callback's own ctx->destroyed check remains the
235 * authoritative guard for the connection.) */
236 if (slot == NULL || !slot->in_use || slot->generation != gen) {
237 uring_cqe_retire(u, cqe);
238 return;
239 }
240
241 /* 2a'. orphaned slot: the owning connection was torn down while this
242 * op was in flight — the task's memory died with the connection pool.
243 * Drop the CQE without dereferencing the task or posting its (equally
244 * dead) completion event, and recycle the slot. */
245 if (slot->orphaned) {
246 brix_uring_slot_release(u, idx);
247 uring_cqe_retire(u, cqe);
248 return;
249 }
250
251 /* 2b. translate + post (the done-callback is task->event.handler, set
252 * at brix_task_bind time). */
253 if (brix_uring_apply_cqe(slot, cqe)) {
254 ngx_thread_task_t *task = slot->task;
255
256 task->event.complete = 1;
257 ngx_post_event(&task->event, &ngx_posted_events);
258 brix_uring_slot_release(u, idx);
259 }
260
261 uring_cqe_retire(u, cqe);
262}

264/*
265 * brix_uring_eventfd_handler — the completion reaper.
266 *
267 * Wired into the worker's epoll as the read handler of the fake connection that
268 * wraps the ring's registered eventfd. This is the public-API analogue of
269 * nginx core's ngx_epoll_eventfd_handler() (the libaio bridge): drain the
270 * eventfd counter, then harvest every ready CQE — validating each slot's
271 * generation, translating the result into the task, and posting the task's
272 * done-callback to ngx_posted_events. Done-callbacks are NEVER invoked inline
273 * here: nginx's posted-events drain runs them after process_events, so a
274 * callback that re-submits the next read window never re-enters the harvest
275 * loop (re-entrancy-safe). Non-static: installed by uring_install_eventfd()
276 * in uring.c.
277 */
278void
279brix_uring_eventfd_handler(ngx_event_t *ev)
280{
281 ngx_connection_t *evc = ev->data;
282 brix_uring_t *u = evc->data;
283 struct io_uring_cqe *cqe;
284 uint64_t counter;
285 ssize_t n;
286
287 /* 1. drain the eventfd counter (mirrors core's first step). */
288 n = read(u->eventfd, &counter, sizeof(counter));
289 if (n != (ssize_t) sizeof(counter)) {
290 if (n == -1 && (ngx_errno == NGX_EAGAIN || ngx_errno == NGX_EINTR)) {
291 return;
292 }
293 ngx_log_error(NGX_LOG_ALERT, evc->log, ngx_errno,
294 "brix: io_uring eventfd read() failed");
295 return;
296 }
297
298 /* 2. harvest all ready completions. */
299 while (io_uring_peek_cqe(&u->ring, &cqe) == 0) {
300 uring_complete_one(u, cqe);
301 }
302}
