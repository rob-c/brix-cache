1/*
2 * file_serve.c — shared HTTP file-body serving pipeline (WebDAV + S3).
3 *
4 * WHAT: Implements brix_http_serve_file_ranged(), the single code path that
5 * turns an already-open, already-stat'd VFS file handle into an HTTP
6 * response body. It parses the Range request header, emits the matching
7 * full-body / 206-partial / 416-unsatisfied response headers, fires the
8 * optional pre-header hook, opens a dashboard transfer record, dups the
9 * fd and streams the requested byte range, then records bytes-sent and
10 * cache-access accounting.
11 *
12 * WHY: WebDAV GET (src/webdav/get.c) and S3 GetObject (src/s3/object.c) need
13 * byte-for-byte identical range handling, ETag/Last-Modified header
14 * emission, dashboard transfer tracking, and cache hit accounting. This
15 * file is the one authoritative implementation so both protocols stay
16 * consistent and bugfixes land once. Callers supply only protocol-specific
17 * knobs (op_name, xfer_proto, identity, ETag flags, pre-header hook) via
18 * brix_http_serve_opts_t and read back metric inputs via
19 * brix_http_serve_result_t.
20 *
21 * HOW: Five sequential phases over the supplied fh:
22 * 1. range parse — brix_http_parse_range() against vst->size; an
23 * unsatisfiable range short-circuits to a 416 (fh closed first).
24 * 2. headers — brix_http_set_file_headers() sets status,
25 * content-length, Content-Range, ETag/Last-Modified.
26 * 3. dashboard — brix_dashboard_http_start_identity() opens a READ
27 * transfer record for live monitoring.
28 * 4. send — the fd is dup()'d so brix_http_send_file_range()
29 * owns its own descriptor; the VFS handle is closed immediately
30 * after the dup so the cache/handle slot is released before the body
31 * (which may block on a slow client) is streamed.
32 * 5. accounting — on success, bytes_sent is reported back and, for
33 * cache-backed files, brix_cache_record_access() updates LRU/hit
34 * stats. header_only (HEAD) requests skip phases 5.
35 * INVARIANT: fh is always closed inside this function (success or error);
36 * callers must never close it afterwards.
37 */

50/* brix_serve_memory_backed — stream [start, start+len) of `fh` to the client by
51 * reading through the storage driver into pool buffers and pushing them through
52 * nginx's output filter. Used when the backend exposes no single sendfile fd —
53 * an object/block backend (e.g. pblock) whose bytes span multiple block files,
54 * which brix_http_send_file_range (a single in_file buffer) cannot serve. The
55 * bytes still flow proto -> VFS -> driver, just memory-backed instead of
56 * zero-copy. The driver preads run on the event loop; a thread-pool/AIO streaming
57 * variant for very large objects is a follow-up. Returns the output-filter rc or
58 * NGX_ERROR. The caller still owns closing `fh`. */
59#define BRIX_SERVE_MEM_CHUNK (256 * 1024)
60
61static ngx_int_t
62brix_serve_memory_backed(ngx_http_request_t *r, brix_vfs_file_t *fh,
63 off_t start, off_t len)
64{
65 off_t done = 0;
66 ngx_int_t hrc;
67
68 /* The response headers were set (set_file_headers) but not yet sent — the
69 * sendfile path sends them inside send_file_range; do the same here. */
70 hrc = ngx_http_send_header(r);
71 if (hrc == NGX_ERROR || hrc > NGX_OK) {
72 return hrc;
73 }
74
75 if (r->header_only || len <= 0) {
76 ngx_buf_t *b = ngx_calloc_buf(r->pool);
77 ngx_chain_t out;
78
79 if (b == NULL) {
80 return NGX_ERROR;
81 }
82 b->last_buf = 1;
83 out.buf = b;
84 out.next = NULL;
85 return ngx_http_output_filter(r, &out);
86 }
87
88 while (done < len) {
89 size_t want = (size_t) (len - done);
90 u_char *buf;
91 ssize_t n;
92 ngx_buf_t *b;
93 ngx_chain_t out;
94 ngx_int_t rc;
95
96 if (want > BRIX_SERVE_MEM_CHUNK) {
97 want = BRIX_SERVE_MEM_CHUNK;
98 }
99 buf = ngx_palloc(r->pool, want);
100 if (buf == NULL) {
101 return NGX_ERROR;
102 }
103 n = brix_vfs_file_pread(fh, buf, want, start + done);
104 if (n <= 0) {
105 return n < 0 ? NGX_ERROR : NGX_OK; /* error / unexpected EOF */
106 }
107
108 b = ngx_calloc_buf(r->pool);
109 if (b == NULL) {
110 return NGX_ERROR;
111 }
112 b->pos = buf;
113 b->last = buf + n;
114 b->memory = 1;
115 b->flush = 1;
116 b->last_buf = (done + n >= len) ? 1 : 0;
117 out.buf = b;
118 out.next = NULL;
119
120 rc = ngx_http_output_filter(r, &out);
121 if (rc == NGX_ERROR || rc > NGX_HTTP_SPECIAL_RESPONSE) {
122 return rc;
123 }
124 done += n;
125 }
126
127 return NGX_OK;
128}

130ngx_int_t
131brix_http_serve_file_ranged(ngx_http_request_t *r,
132 brix_vfs_file_t *fh, const brix_vfs_stat_t *vst,
133 const char *fs_path, const brix_http_serve_opts_t *opts,
134 brix_http_serve_result_t *result)
135{
136 brix_http_range_t rng;
137 off_t range_start, range_end, send_len;
138 ngx_fd_t fd, send_fd;
139 ngx_int_t rc;
140 ngx_uint_t from_cache;
141 const char *cache_path;
142 const char *backend_name;
143
144 ngx_memzero(result, sizeof(*result));
145
146 /* Backend attribution label — captured now because every send path below
147 * releases the vfs handle before the bytes are counted. */
148 backend_name = brix_vfs_file_backend_name(fh);
149
150 /*
151 * Select the configured TCP congestion control on this connection before the
152 * body is streamed. This single site covers BOTH WebDAV GET and S3 GetObject
153 * (both delegate the body send here), so one apply governs every HTTP download
154 * regardless of which protocol handler opened the file. The directive
155 * (brix_tcp_congestion) is owned by the always-present webdav http module;
156 * reading its per-location conf applies the same sender-side policy (e.g.
157 * "bbr") uniformly. Empty value => kernel default (no syscall).
158 */
159 {
160 ngx_http_brix_webdav_loc_conf_t *wconf =
161 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
162
163 if (wconf != NULL && wconf->tcp_congestion.len > 0
164 && r->connection != NULL
165 && r->connection->fd != (ngx_socket_t) -1)
166 {
167 brix_apply_tcp_congestion(r->connection->fd, wconf->tcp_congestion);
168 }
169 }
170
171 /* Phase 1: range parse */
172 brix_http_parse_range(
173 r->headers_in.range ? r->headers_in.range->value.data : NULL,
174 r->headers_in.range ? r->headers_in.range->value.len : 0,
175 vst->size, &rng);
176
177 if (rng.present && !rng.satisfiable) {
178 brix_vfs_close(fh, r->connection->log);
179 result->range_result = BRIX_SERVE_RANGE_UNSATISFIED;
180 r->headers_out.status = NGX_HTTP_RANGE_NOT_SATISFIABLE;
181 r->headers_out.content_length_n = 0;
182 ngx_http_send_header(r);
183 return ngx_http_send_special(r, NGX_HTTP_LAST);
184 }
185
186 range_start = rng.start;
187 range_end = rng.end;
188 send_len = (vst->size > 0) ? (range_end - range_start + 1) : 0;
189 result->range_result = rng.present ? BRIX_SERVE_RANGE_PARTIAL
190 : BRIX_SERVE_RANGE_FULL;
191
192 /* Phase 1b (phase-42): outbound compression. Opt-in per location; only a
193 * whole-object, non-HEAD GET of a compressible type whose Accept-Encoding
194 * names a codec we have takes this path. It bypasses sendfile (read ->
195 * codec -> chunked output filter), so the uncompressed fast path below is
196 * untouched for every other request. */
197 if (opts->compress) {
198 brix_codec_id_t enc;
199 /* Resolve the response content-type from the URI extension BEFORE
200 * negotiating: the negotiator's incompressible-MIME deny list reads
201 * r->headers_out.content_type, which nginx does NOT populate on request
202 * entry (it is otherwise set later, inside set_file_headers). Without
203 * this, content_type is empty at negotiate time and the deny list is dead
204 * code, so already-compressed types (image, video, .gz, ...) get
205 * wastefully re-compressed. Idempotent: set_file_headers' later call
206 * early-returns once it is set. */
207 (void) ngx_http_set_content_type(r);
208 enc = brix_http_compress_negotiate(r, vst->size, rng.present);
209 if (enc != BRIX_CODEC_IDENTITY) {
210 ngx_fd_t cfd, csend;
211 off_t bytes_out = 0;
212
213 cfd = brix_vfs_file_fd(fh);
214 from_cache = brix_vfs_file_from_cache(fh);
215 cache_path = brix_vfs_file_path(fh);
216 csend = dup(cfd);
217 if (csend == NGX_INVALID_FILE) {
218 brix_vfs_close(fh, r->connection->log);
219 return NGX_HTTP_INTERNAL_SERVER_ERROR;
220 }
221 (void) brix_dashboard_http_start_identity(r, fs_path,
222 opts->identity, "", opts->xfer_proto, BRIX_XFER_DIR_READ,
223 opts->op_name, (int64_t) vst->size);
224 brix_vfs_close(fh, r->connection->log);
225
226 rc = brix_http_send_file_compressed(r, csend, fs_path, vst->size,
227 enc, vst->mtime,
228 opts->etag_flags, &bytes_out);
229 if (rc == NGX_ERROR || rc >= NGX_HTTP_SPECIAL_RESPONSE) {
230 brix_dashboard_http_error(r, "serve_file_ranged: compress send failed");
231 brix_dashboard_http_finish(r);
232 return rc;
233 }
234 result->range_result = BRIX_SERVE_RANGE_FULL;
235 result->bytes_sent = bytes_out; /* compressed bytes on wire */
236 brix_metric_backend_bytes(backend_name, BRIX_METRIC_OP_READ,
237 (size_t) bytes_out);
238 brix_dashboard_http_add(r, (ngx_atomic_int_t) bytes_out);
239 if (from_cache && vst->size > 0) {
240 (void) brix_cache_record_access(cache_path, (size_t) vst->size,
241 r->connection->log);
242 }
243 return rc;
244 }
245 }
246
247 /* Phase 2: response headers */
248 if (brix_http_set_file_headers(r, vst->mtime, vst->size, send_len,
249 NULL, opts->etag_flags,
250 rng.present, range_start, range_end)
251 != NGX_OK)
252 {
253 brix_vfs_close(fh, r->connection->log);
254 return NGX_HTTP_INTERNAL_SERVER_ERROR;
255 }
256
257 if (opts->pre_header_send != NULL) {
258 opts->pre_header_send(r, brix_vfs_file_fd(fh), vst->size,
259 opts->pre_header_ud);
260 }
261
262 /* Phase 3: dashboard */
263 (void) brix_dashboard_http_start_identity(r, fs_path,
264 opts->identity, "",
265 opts->xfer_proto, BRIX_XFER_DIR_READ, opts->op_name,
266 (int64_t) send_len);
267
268 /* Phase 4: dup fd, release vfs handle, send. Use the sendfile-gated fd: the
269 * serve path is zero-copy (sendfile for cleartext), so it requires a backend
270 * that can back it. A non-sendfile backend returns NGX_INVALID_FILE here and
271 * fails closed rather than dup'ing a bogus descriptor. */
272 fd = brix_vfs_file_sendfile_fd(fh);
273 from_cache = brix_vfs_file_from_cache(fh);
274 cache_path = brix_vfs_file_path(fh);
275
276 /* A backend with no single sendfile fd (an object/block backend whose bytes
277 * span multiple block files) is served memory-backed via driver reads. */
278 if (fd == NGX_INVALID_FILE) {
279 rc = brix_serve_memory_backed(r, fh, range_start, send_len);
280 brix_vfs_close(fh, r->connection->log);
281 if (rc == NGX_ERROR) {
282 brix_dashboard_http_error(r,
283 "serve_file_ranged: memory-backed send failed");
284 brix_dashboard_http_finish(r);
285 return rc;
286 }
287 if (!r->header_only) {
288 result->bytes_sent = send_len;
289 brix_metric_backend_bytes(backend_name, BRIX_METRIC_OP_READ,
290 (size_t) send_len);
291 brix_dashboard_http_add(r, (ngx_atomic_int_t) send_len);
292 if (from_cache && send_len > 0) {
293 (void) brix_cache_record_access(cache_path, (size_t) send_len,
294 r->connection->log);
295 }
296 }
297 brix_dashboard_http_finish(r);
298 return rc;
299 }
300
301 send_fd = dup(fd);
302 if (send_fd == NGX_INVALID_FILE) {
303 brix_vfs_close(fh, r->connection->log);
304 brix_dashboard_http_error(r, "serve_file_ranged: dup failed");
305 brix_dashboard_http_finish(r);
306 return NGX_HTTP_INTERNAL_SERVER_ERROR;
307 }
308 brix_vfs_close(fh, r->connection->log);
309
310 rc = brix_http_send_file_range(r, send_fd, fs_path,
311 range_start, send_len, 1);
312
313 if (rc == NGX_ERROR) {
314 brix_dashboard_http_error(r, "serve_file_ranged: send failed");
315 brix_dashboard_http_finish(r);
316 return rc;
317 }
318
319 if (r->header_only) {
320 brix_dashboard_http_finish(r);
321 return rc;
322 }
323
324 /* Phase 5: post-send accounting */
325 result->bytes_sent = send_len;
326 brix_metric_backend_bytes(backend_name, BRIX_METRIC_OP_READ,
327 (size_t) send_len);
328 brix_dashboard_http_add(r, (ngx_atomic_int_t) send_len);
329
330 if (from_cache && send_len > 0) {
331 (void) brix_cache_record_access(cache_path, (size_t) send_len,
332 r->connection->log);
333 }
334
335 return rc;
336}
