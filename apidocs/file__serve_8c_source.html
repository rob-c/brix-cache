1/*
2 * file_serve.c — shared HTTP file-body serving pipeline (WebDAV + S3).
3 *
4 * WHAT: Implements brix_http_serve_file_ranged(), the single code path that
5 * turns an already-open, already-stat'd VFS file handle into an HTTP
6 * response body. It parses the Range request header, emits the matching
7 * full-body / 206-partial / 416-unsatisfied response headers, fires the
8 * optional pre-header hook, opens a dashboard transfer record, dups the
9 * fd and streams the requested byte range, then records bytes-sent and
10 * cache-access accounting.
11 *
12 * WHY: WebDAV GET (src/webdav/get.c) and S3 GetObject (src/s3/object.c) need
13 * byte-for-byte identical range handling, ETag/Last-Modified header
14 * emission, dashboard transfer tracking, and cache hit accounting. This
15 * file is the one authoritative implementation so both protocols stay
16 * consistent and bugfixes land once. Callers supply only protocol-specific
17 * knobs (op_name, xfer_proto, identity, ETag flags, pre-header hook) via
18 * brix_http_serve_opts_t and read back metric inputs via
19 * brix_http_serve_result_t.
20 *
21 * HOW: Sequential phases over the supplied fh, orchestrated by
22 * brix_http_serve_file_ranged() and implemented in static helpers so the
23 * orchestrator reads as a flat sequence:
24 * 1. range parse — brix_http_parse_range() against vst->size; an
25 * unsatisfiable range short-circuits to a 416
26 * (serve_range_unsatisfiable, fh closed first).
27 * 2. compress — serve_try_compressed(): opt-in whole-object codec path
28 * that bypasses sendfile; returns SERVE_DONE when it handled the
29 * request, SERVE_CONTINUE otherwise.
30 * 3. headers — brix_http_set_file_headers() sets status,
31 * content-length, Content-Range, ETag/Last-Modified.
32 * 4. dashboard — brix_dashboard_http_start_identity() opens a READ
33 * transfer record for live monitoring.
34 * 5. send — serve_send_sendfile() dup()'s the sendfile-gated fd so
35 * the send owns its own descriptor, or serve_send_memory_backed() drives
36 * driver reads when the backend exposes no single sendfile fd; the VFS
37 * handle is closed immediately after the dup / at the boundary so the
38 * cache/handle slot is released before the (possibly slow) body streams.
39 * 6. accounting — on success serve_record_body_accounting() reports
40 * bytes_sent and, for cache-backed files, updates LRU/hit stats.
41 * header_only (HEAD) requests skip accounting.
42 * INVARIANT: fh is always closed inside this function (success or error);
43 * callers must never close it afterwards.
44 * INVARIANT 2 (TLS memory-backed vs cleartext file-backed/sendfile): the
45 * sendfile path (serve_send_sendfile) is zero-copy; the object/block
46 * backend path (serve_send_memory_backed) is memory-backed. The two never
47 * mix — the backend's sendfile-fd availability picks exactly one.
48 */

70/*
71 * serve_ctx_t — file-local bundle of the immutable serve inputs plus the
72 * per-request byte range and cache attribution.
73 *
74 * WHAT: Threads the request, the vfs handle, the caller's opts/result, the
75 * stat, the backend attribution label, and (once resolved) the byte range
76 * and cache-origin flags through the static send helpers as one pointer.
77 * WHY: The send/accounting phases share the same six-plus inputs; passing them
78 * as a single const context keeps each helper's parameter list small
79 * (§8 explicit-data-flow: pass state, no new globals) instead of a long
80 * positional argument list.
81 * HOW: `r`, `fh`, `vst`, `fs_path`, `opts`, `result`, `backend_name` are set
82 * up-front by the orchestrator; `range_start`, `send_len`, `from_cache`,
83 * `cache_path` are filled once the range and backend fd are resolved.
84 */
85typedef struct {
86 ngx_http_request_t *r;
87 brix_vfs_file_t *fh;
88 const brix_vfs_stat_t *vst;
89 const char *fs_path;
90 const brix_http_serve_opts_t *opts;
91 brix_http_serve_result_t *result;
92 const char *backend_name;
93 off_t range_start;
94 off_t send_len;
95 ngx_uint_t from_cache;
96 const char *cache_path;
97} serve_ctx_t;

99/* brix_serve_memory_backed — stream [start, start+len) of `fh` to the client by
100 * reading through the storage driver into pool buffers and pushing them through
101 * nginx's output filter. Used when the backend exposes no single sendfile fd —
102 * an object/block backend (e.g. pblock) whose bytes span multiple block files,
103 * which brix_http_send_file_range (a single in_file buffer) cannot serve. The
104 * bytes still flow proto -> VFS -> driver, just memory-backed instead of
105 * zero-copy. The driver preads run on the event loop; a thread-pool/AIO streaming
106 * variant for very large objects is a follow-up. Returns the output-filter rc or
107 * NGX_ERROR. The caller still owns closing `fh`. */
108#define BRIX_SERVE_MEM_CHUNK (256 * 1024)
109
110static ngx_int_t
111brix_serve_memory_backed(ngx_http_request_t *r, brix_vfs_file_t *fh,
112 off_t start, off_t len)
113{
114 off_t done = 0;
115 ngx_int_t hrc;
116
117 /* The response headers were set (set_file_headers) but not yet sent — the
118 * sendfile path sends them inside send_file_range; do the same here. */
119 hrc = ngx_http_send_header(r);
120 if (hrc == NGX_ERROR || hrc > NGX_OK) {
121 return hrc;
122 }
123
124 if (r->header_only || len <= 0) {
125 ngx_buf_t *b = ngx_calloc_buf(r->pool);
126 ngx_chain_t out;
127
128 if (b == NULL) {
129 return NGX_ERROR;
130 }
131 b->last_buf = 1;
132 out.buf = b;
133 out.next = NULL;
134 return ngx_http_output_filter(r, &out);
135 }
136
137 while (done < len) {
138 size_t want = (size_t) (len - done);
139 u_char *buf;
140 ssize_t n;
141 ngx_buf_t *b;
142 ngx_chain_t out;
143 ngx_int_t rc;
144
145 if (want > BRIX_SERVE_MEM_CHUNK) {
146 want = BRIX_SERVE_MEM_CHUNK;
147 }
148 buf = ngx_palloc(r->pool, want);
149 if (buf == NULL) {
150 return NGX_ERROR;
151 }
152 n = brix_vfs_file_pread(fh, buf, want, start + done);
153 if (n <= 0) {
154 return n < 0 ? NGX_ERROR : NGX_OK; /* error / unexpected EOF */
155 }
156
157 b = ngx_calloc_buf(r->pool);
158 if (b == NULL) {
159 return NGX_ERROR;
160 }
161 b->pos = buf;
162 b->last = buf + n;
163 b->memory = 1;
164 b->flush = 1;
165 b->last_buf = (done + n >= len) ? 1 : 0;
166 out.buf = b;
167 out.next = NULL;
168
169 rc = ngx_http_output_filter(r, &out);
170 if (rc == NGX_ERROR || rc > NGX_HTTP_SPECIAL_RESPONSE) {
171 return rc;
172 }
173 done += n;
174 }
175
176 return NGX_OK;
177}

179/*
180 * serve_apply_tcp_congestion — select the configured TCP congestion control on
181 * this connection before the body is streamed.
182 *
183 * WHAT: Reads the webdav-owned brix_tcp_congestion directive from the location
184 * conf and, when set, applies it to the connection socket.
185 * WHY: This single site covers BOTH WebDAV GET and S3 GetObject (both delegate
186 * the body send here), so one apply governs every HTTP download regardless
187 * of which protocol handler opened the file. The directive is owned by the
188 * always-present webdav http module; reading its per-location conf applies
189 * the same sender-side policy (e.g. "bbr") uniformly.
190 * HOW: Empty value => kernel default (no syscall). r->connection is known
191 * non-NULL here (the orchestrator's contract guard ran first) but the fd
192 * may still be invalid, so both are re-checked before the setsockopt.
193 */
194static void
195serve_apply_tcp_congestion(ngx_http_request_t *r)
196{
197 ngx_http_brix_webdav_loc_conf_t *wconf =
198 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
199
200 if (wconf != NULL && wconf->tcp_congestion.len > 0
201 && r->connection != NULL
202 && r->connection->fd != (ngx_socket_t) -1)
203 {
204 brix_apply_tcp_congestion(r->connection->fd, wconf->tcp_congestion);
205 }

208/*
209 * serve_range_unsatisfiable — emit a 416 Range Not Satisfiable and release fh.
210 *
211 * WHAT: Closes the vfs handle, records the UNSATISFIED range result, and sends a
212 * zero-length 416 response.
213 * WHY: An unsatisfiable Range short-circuits the whole pipeline: no dashboard
214 * record, no send, no accounting. Factoring it out keeps the orchestrator's
215 * success path flat.
216 * HOW: Mirrors the original inline block byte-for-byte — same status, same
217 * content_length_n=0, same send_header + send_special(NGX_HTTP_LAST).
218 */
219static ngx_int_t
220serve_range_unsatisfiable(ngx_http_request_t *r, brix_vfs_file_t *fh,
221 brix_http_serve_result_t *result)
222{
223 brix_vfs_close(fh, r->connection->log);
224 result->range_result = BRIX_SERVE_RANGE_UNSATISFIED;
225 r->headers_out.status = NGX_HTTP_RANGE_NOT_SATISFIABLE;
226 r->headers_out.content_length_n = 0;
227 ngx_http_send_header(r);
228 return ngx_http_send_special(r, NGX_HTTP_LAST);
229}

231/*
232 * serve_record_body_accounting — post-send bytes/metric/cache accounting.
233 *
234 * WHAT: Reports bytes_sent back to the caller, increments the per-backend read
235 * bytes metric, adds to the live dashboard transfer, and (for cache-backed
236 * files) records a cache access for LRU/hit stats.
237 * WHY: The sendfile, memory-backed and compressed send paths perform identical
238 * body accounting; one helper keeps them in lockstep so a change lands once.
239 * HOW: HEAD (header_only) requests never reach this — accounting is body-only.
240 * from_cache/cache_path are captured by the caller before the vfs handle is
241 * released, since every send path closes fh before the bytes are counted.
242 */
243static void
244serve_record_body_accounting(const serve_ctx_t *sc, off_t bytes,
245 ngx_uint_t from_cache, const char *cache_path)
246{
247 sc->result->bytes_sent = bytes;
248 brix_metric_backend_bytes(sc->backend_name, BRIX_METRIC_OP_READ,
249 (size_t) bytes);
250 brix_dashboard_http_add(sc->r, (ngx_atomic_int_t) bytes);
251
252 if (from_cache && bytes > 0) {
253 (void) brix_cache_record_access(cache_path, (size_t) bytes,
254 sc->r->connection->log);
255 }
256}

258/*
259 * serve_try_compressed — opt-in whole-object outbound compression fast-path.
260 *
261 * WHAT: When compression is enabled for this location and Accept-Encoding names a
262 * codec we have for a whole-object non-HEAD GET of a compressible type,
263 * resolves the content-type, negotiates the codec, and (on a hit) streams
264 * the object read -> codec -> chunked output filter, closing fh and doing
265 * body accounting. On a codec miss it leaves everything untouched.
266 * WHY: This path bypasses sendfile, so isolating it keeps the uncompressed fast
267 * path (headers + sendfile / memory-backed) untouched for every other
268 * request and cuts the orchestrator's branch count.
269 * HOW: Returns SERVE_DONE with *rc set (request fully handled) when it took the
270 * compressed path — including the negotiate-time content-type resolution
271 * that primes the incompressible-MIME deny list — else SERVE_CONTINUE.
272 */
273static serve_disposition_t
274serve_try_compressed(const serve_ctx_t *sc, ngx_uint_t range_present,
275 ngx_int_t *rc)
276{
277 ngx_http_request_t *r = sc->r;
278 brix_vfs_file_t *fh = sc->fh;
279 const brix_vfs_stat_t *vst = sc->vst;
280 const brix_http_serve_opts_t *opts = sc->opts;
281 brix_codec_id_t enc;
282 ngx_fd_t cfd, csend;
283 off_t bytes_out = 0;
284 ngx_uint_t from_cache;
285 const char *cache_path;
286
287 if (!opts->compress) {
288 return SERVE_CONTINUE;
289 }
290
291 /* Resolve the response content-type from the URI extension BEFORE
292 * negotiating: the negotiator's incompressible-MIME deny list reads
293 * r->headers_out.content_type, which nginx does NOT populate on request
294 * entry (it is otherwise set later, inside set_file_headers). Without this,
295 * content_type is empty at negotiate time and the deny list is dead code, so
296 * already-compressed types (image, video, .gz, ...) get wastefully
297 * re-compressed. Idempotent: set_file_headers' later call early-returns once
298 * it is set. */
299 (void) ngx_http_set_content_type(r);
300 enc = brix_http_compress_negotiate(r, vst->size, range_present);
301 if (enc == BRIX_CODEC_IDENTITY) {
302 return SERVE_CONTINUE;
303 }
304
305 cfd = brix_vfs_file_fd(fh);
306 from_cache = brix_vfs_file_from_cache(fh);
307 cache_path = brix_vfs_file_path(fh);
308 csend = dup(cfd);
309 if (csend == NGX_INVALID_FILE) {
310 brix_vfs_close(fh, r->connection->log);
311 *rc = NGX_HTTP_INTERNAL_SERVER_ERROR;
312 return SERVE_DONE;
313 }
314 (void) brix_dashboard_http_start_identity(r, sc->fs_path,
315 opts->identity, "", opts->xfer_proto, BRIX_XFER_DIR_READ,
316 opts->op_name, (int64_t) vst->size);
317 brix_vfs_close(fh, r->connection->log);
318
319 *rc = brix_http_send_file_compressed(r, csend, sc->fs_path, vst->size,
320 enc, vst->mtime,
321 opts->etag_flags, &bytes_out);
322 if (*rc == NGX_ERROR || *rc >= NGX_HTTP_SPECIAL_RESPONSE) {
323 brix_dashboard_http_error(r, "serve_file_ranged: compress send failed");
324 brix_dashboard_http_finish(r);
325 return SERVE_DONE;
326 }
327 sc->result->range_result = BRIX_SERVE_RANGE_FULL;
328 /* compressed bytes on wire */
329 serve_record_body_accounting(sc, bytes_out, from_cache, cache_path);
330 return SERVE_DONE;
331}

333/*
334 * serve_send_memory_backed — memory-backed send path for backends with no single
335 * sendfile fd (INVARIANT 2: memory-backed, never mixed with the sendfile path).
336 *
337 * WHAT: Streams [range_start, range_start+send_len) via driver reads (through
338 * brix_serve_memory_backed), closes fh, and on success does body accounting.
339 * WHY: An object/block backend whose bytes span multiple block files cannot be
340 * served by the single-in_file sendfile buffer; this path drives the driver
341 * preads instead. Splitting it keeps the orchestrator flat.
342 * HOW: sc->from_cache/cache_path are captured by the orchestrator before fh is
343 * released. Returns the send rc; HEAD requests skip accounting.
344 */
345static ngx_int_t
346serve_send_memory_backed(const serve_ctx_t *sc)
347{
348 ngx_http_request_t *r = sc->r;
349 brix_vfs_file_t *fh = sc->fh;
350 ngx_int_t rc;
351
352 rc = brix_serve_memory_backed(r, fh, sc->range_start, sc->send_len);
353 brix_vfs_close(fh, r->connection->log);
354 if (rc == NGX_ERROR) {
355 brix_dashboard_http_error(r,
356 "serve_file_ranged: memory-backed send failed");
357 brix_dashboard_http_finish(r);
358 return rc;
359 }
360 if (!r->header_only) {
361 serve_record_body_accounting(sc, sc->send_len, sc->from_cache,
362 sc->cache_path);
363 }
364 brix_dashboard_http_finish(r);
365 return rc;
366}

368/*
369 * serve_send_sendfile — zero-copy sendfile send path (INVARIANT 2: cleartext
370 * file-backed/sendfile, never mixed with the memory-backed path).
371 *
372 * WHAT: dup()'s the sendfile-gated fd so the send owns its own descriptor,
373 * releases the vfs handle, streams the byte range via
374 * brix_http_send_file_range, and on success does body accounting.
375 * WHY: The serve path is zero-copy; the send must own a private descriptor while
376 * the vfs handle (and its cache/handle slot) is released immediately so a
377 * slow client cannot pin it. Splitting it keeps the orchestrator flat.
378 * HOW: fd is the orchestrator's already-fetched sendfile fd (known !=
379 * NGX_INVALID_FILE). sc->from_cache/cache_path are captured before fh is
380 * released. Returns the send rc; HEAD requests skip accounting.
381 */
382static ngx_int_t
383serve_send_sendfile(const serve_ctx_t *sc, ngx_fd_t fd)
384{
385 ngx_http_request_t *r = sc->r;
386 brix_vfs_file_t *fh = sc->fh;
387 ngx_fd_t send_fd;
388 ngx_int_t rc;
389
390 send_fd = dup(fd);
391 if (send_fd == NGX_INVALID_FILE) {
392 brix_vfs_close(fh, r->connection->log);
393 brix_dashboard_http_error(r, "serve_file_ranged: dup failed");
394 brix_dashboard_http_finish(r);
395 return NGX_HTTP_INTERNAL_SERVER_ERROR;
396 }
397 brix_vfs_close(fh, r->connection->log);
398
399 rc = brix_http_send_file_range(r, send_fd, sc->fs_path,
400 sc->range_start, sc->send_len, 1);
401
402 if (rc == NGX_ERROR) {
403 brix_dashboard_http_error(r, "serve_file_ranged: send failed");
404 brix_dashboard_http_finish(r);
405 return rc;
406 }
407
408 if (r->header_only) {
409 brix_dashboard_http_finish(r);
410 return rc;
411 }
412
413 serve_record_body_accounting(sc, sc->send_len, sc->from_cache,
414 sc->cache_path);
415 return rc;
416}

418ngx_int_t
419brix_http_serve_file_ranged(ngx_http_request_t *r,
420 brix_vfs_file_t *fh, const brix_vfs_stat_t *vst,
421 const char *fs_path, const brix_http_serve_opts_t *opts,
422 brix_http_serve_result_t *result)
423{
424 brix_http_range_t rng;
425 off_t range_end;
426 ngx_fd_t fd;
427 ngx_int_t rc;
428 serve_ctx_t sc;
429
430 ngx_memzero(result, sizeof(*result));
431 ngx_memzero(&sc, sizeof(sc));
432 sc.r = r;
433 sc.fh = fh;
434 sc.vst = vst;
435 sc.fs_path = fs_path;
436 sc.opts = opts;
437 sc.result = result;
438
439 /* The tcp-congestion apply below treats r->connection as possibly NULL,
440 * making that nullability part of this function's contract — so every later
441 * r->connection->log / ->fd use must sit behind one provable guard. A request
442 * with no connection cannot be served; release the handle (NULL log =>
443 * brix_vfs_close falls back to fh->log) and bail. */
444 if (r->connection == NULL) {
445 (void) brix_vfs_close(fh, NULL);
446 return NGX_HTTP_INTERNAL_SERVER_ERROR;
447 }
448
449 /* Backend attribution label — captured now because every send path below
450 * releases the vfs handle before the bytes are counted. */
451 sc.backend_name = brix_vfs_file_backend_name(fh);
452
453 serve_apply_tcp_congestion(r);
454
455 /* Phase 1: range parse */
456 brix_http_parse_range(
457 r->headers_in.range ? r->headers_in.range->value.data : NULL,
458 r->headers_in.range ? r->headers_in.range->value.len : 0,
459 vst->size, &rng);
460
461 if (rng.present && !rng.satisfiable) {
462 return serve_range_unsatisfiable(r, fh, result);
463 }
464
465 sc.range_start = rng.start;
466 range_end = rng.end;
467 sc.send_len = (vst->size > 0) ? (range_end - sc.range_start + 1) : 0;
468 result->range_result = rng.present ? BRIX_SERVE_RANGE_PARTIAL
469 : BRIX_SERVE_RANGE_FULL;
470
471 /* Phase 1b (phase-42): outbound compression. Opt-in per location; only a
472 * whole-object, non-HEAD GET of a compressible type whose Accept-Encoding
473 * names a codec we have takes this path. It bypasses sendfile (read ->
474 * codec -> chunked output filter), so the uncompressed fast path below is
475 * untouched for every other request. */
476 if (serve_try_compressed(&sc, rng.present, &rc) == SERVE_DONE) {
477 return rc;
478 }
479
480 /* Phase 2: response headers */
481 if (brix_http_set_file_headers(r, vst->mtime, vst->size, sc.send_len,
482 NULL, opts->etag_flags,
483 rng.present, sc.range_start, range_end)
484 != NGX_OK)
485 {
486 brix_vfs_close(fh, r->connection->log);
487 return NGX_HTTP_INTERNAL_SERVER_ERROR;
488 }
489
490 if (opts->pre_header_send != NULL) {
491 opts->pre_header_send(r, brix_vfs_file_fd(fh), vst->size,
492 opts->pre_header_ud);
493 }
494
495 /* Phase 3: dashboard */
496 (void) brix_dashboard_http_start_identity(r, fs_path,
497 opts->identity, "",
498 opts->xfer_proto, BRIX_XFER_DIR_READ, opts->op_name,
499 (int64_t) sc.send_len);
500
501 /* Phase 4: dup fd, release vfs handle, send. Use the sendfile-gated fd: the
502 * serve path is zero-copy (sendfile for cleartext), so it requires a backend
503 * that can back it. A non-sendfile backend returns NGX_INVALID_FILE here and
504 * fails closed rather than dup'ing a bogus descriptor. */
505 fd = brix_vfs_file_sendfile_fd(fh);
506 sc.from_cache = brix_vfs_file_from_cache(fh);
507 sc.cache_path = brix_vfs_file_path(fh);
508
509 /* A backend with no single sendfile fd (an object/block backend whose bytes
510 * span multiple block files) is served memory-backed via driver reads. */
511 if (fd == NGX_INVALID_FILE) {
512 return serve_send_memory_backed(&sc);
513 }
514
515 /* Phase 5: post-send accounting is done inside the send helper. */
516 return serve_send_sendfile(&sc, fd);
517}
