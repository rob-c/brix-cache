33static pblock_meta
34file_meta(const char *blob_id, int64_t size)
35{
36 pblock_meta m;
37
38 memset(&m, 0, sizeof(m));
39 m.is_dir = 0;
40 snprintf(m.blob_id, sizeof(m.blob_id), "%s", blob_id);
41 m.size = size;
42 m.block_size = 1 << 20; /* 1 MiB */
43 m.mtime = 1000;
44 m.ctime = 1000;
45 m.mode = S_IFREG | 0644;

62/* put -> lookup round-trips every field; absent paths report not-found. */
63static void
64test_put_lookup(pblock_catalog *cat)
65{
66 pblock_meta in = file_meta("0123456789abcdef0123456789abcdef", 4242);
67 pblock_meta out;
68
69 CHECK(pblock_catalog_put(cat, "/a.txt", &in) == 0, "put failed: %s",
70 strerror(errno));
71 CHECK(pblock_catalog_lookup(cat, "/a.txt", &out) == 0, "lookup miss");
72 CHECK(out.is_dir == 0, "is_dir wrong");
73 CHECK(out.size == 4242, "size %lld", (long long) out.size);
74 CHECK(out.block_size == (1 << 20), "block_size %lld",
75 (long long) out.block_size);
76 CHECK(strcmp(out.blob_id, in.blob_id) == 0, "blob_id %s", out.blob_id);
77 CHECK((out.mode & 0777) == 0644, "mode %o", out.mode);
78
79 CHECK(pblock_catalog_lookup(cat, "/nope.txt", &out) == 1, "ghost found");
80}

82/* touch updates only size+mtime and only for existing rows. */
83static void
84test_parent_gate(pblock_catalog *cat)
85{
86 /* phase-68 orphan fix: every create requires its immediate parent to
87 * exist and be a directory — an orphan row (reachable by key, invisible
88 * to every listing) must be impossible. */
89 pblock_meta in, out;
90
91 memset(&in, 0, sizeof(in));
92 in.mode = S_IFREG | 0644;
93
94 errno = 0;
95 CHECK(pblock_catalog_put(cat, "/ghostdir/file", &in) != 0
96 && errno == ENOENT, "orphan put allowed (errno %d)", errno);
97 CHECK(pblock_catalog_create(cat, "/ghostdir/sub/leaf", &in) != 0
98 && errno == ENOENT, "orphan create allowed (errno %d)", errno);
99 CHECK(pblock_catalog_lookup(cat, "/ghostdir/file", &out) == 1,
100 "orphan row landed anyway");
101
102 /* parent present but NOT a directory -> ENOTDIR */
103 CHECK(pblock_catalog_put(cat, "/plainfile", &in) == 0, "seed file");
104 errno = 0;
105 CHECK(pblock_catalog_put(cat, "/plainfile/child", &in) != 0
106 && errno == ENOTDIR, "file-as-parent allowed (errno %d)", errno);
107
108 /* the chain in order works, and rename into a missing parent fails */
109 memset(&in, 0, sizeof(in));
110 in.is_dir = 1;
111 in.mode = S_IFDIR | 0755;
112 CHECK(pblock_catalog_put(cat, "/ghostdir", &in) == 0, "mkdir parent");
113 memset(&in, 0, sizeof(in));
114 in.mode = S_IFREG | 0644;
115 CHECK(pblock_catalog_put(cat, "/ghostdir/file", &in) == 0,
116 "put after mkdir");
117 errno = 0;
118 CHECK(pblock_catalog_rename(cat, "/ghostdir/file", "/nowhere/f") != 0
119 && errno == ENOENT, "rename into missing parent allowed");
120 CHECK(pblock_catalog_remove(cat, "/ghostdir/file") == 0, "cleanup file");
121 CHECK(pblock_catalog_remove(cat, "/ghostdir") == 0, "cleanup dir");
122 CHECK(pblock_catalog_remove(cat, "/plainfile") == 0, "cleanup plain");
123}

181/* opendir enumerates direct children only (basenames). */
182static void
183test_opendir(pblock_catalog *cat)
184{
185 pblock_meta d = dir_meta();
186 pblock_meta f = file_meta("dddddddddddddddddddddddddddddddd", 1);
187 pblock_catalog_iter *it;
188 char name[256];
189 int seen_a = 0, seen_b = 0, n = 0, rc;
190
191 CHECK(pblock_catalog_put(cat, "/list", &d) == 0, "mkdir list");
192 CHECK(pblock_catalog_put(cat, "/list/a", &f) == 0, "a");
193 CHECK(pblock_catalog_put(cat, "/list/b", &f) == 0, "b");
194 /* "/list/b" is a FILE: since the phase-68 parent gate this insert must
195 * fail ENOTDIR outright (it used to land as an orphan row that readdir
196 * could never reach — the listing-shape CHECKs below still hold). */
197 errno = 0;
198 CHECK(pblock_catalog_put(cat, "/list/b/nested", &f) != 0
199 && errno == ENOTDIR, "nested under a file allowed (errno %d)",
200 errno);
201
202 it = pblock_catalog_opendir(cat, "/list");
203 CHECK(it != NULL, "opendir failed");
204 while ((rc = pblock_catalog_readdir(it, name, sizeof(name))) == 0) {
205 if (strcmp(name, "a") == 0) { seen_a = 1; }
206 if (strcmp(name, "b") == 0) { seen_b = 1; }
207 CHECK(strchr(name, '/') == NULL, "readdir returned a path: %s", name);
208 n++;
209 }
210 CHECK(rc == 1, "readdir ended with error %d", rc);
211 pblock_catalog_closedir(it);
212 CHECK(seen_a && seen_b, "missing children a=%d b=%d", seen_a, seen_b);
213 CHECK(n == 2, "expected 2 direct children, got %d", n);
214}

216/* xattr set/get/list/remove round-trips. */
217static void
218test_xattr(pblock_catalog *cat)
219{
220 pblock_meta f = file_meta("eeeeeeeeeeeeeeeeeeeeeeeeeeeeeeee", 1);
221 char buf[128];
222 ssize_t n;
223
224 CHECK(pblock_catalog_put(cat, "/xf", &f) == 0, "put xf");
225 CHECK(pblock_catalog_setxattr(cat, "/xf", "user.k", "val", 3) == 0,
226 "setxattr");
227 n = pblock_catalog_getxattr(cat, "/xf", "user.k", buf, sizeof(buf));
228 CHECK(n == 3 && memcmp(buf, "val", 3) == 0, "getxattr n=%zd", n);
229
230 n = pblock_catalog_listxattr(cat, "/xf", buf, sizeof(buf));
231 CHECK(n == (ssize_t) sizeof("user.k"), "listxattr n=%zd", n);
232 CHECK(memcmp(buf, "user.k", sizeof("user.k")) == 0, "listxattr name");
233
234 CHECK(pblock_catalog_removexattr(cat, "/xf", "user.k") == 0, "removexattr");
235 errno = 0;
236 n = pblock_catalog_getxattr(cat, "/xf", "user.k", buf, sizeof(buf));
237 CHECK(n == -1 && errno == ENODATA, "get after remove n=%zd errno=%d", n,
238 errno);
239}

241int
242main(void)
243{
244 char dir[] = "/tmp/pb_cat_ut.XXXXXX";
245 char db[512];
246 pblock_catalog *cat;
247
248 CHECK(mkdtemp(dir) != NULL, "mkdtemp failed");
249 snprintf(db, sizeof(db), "%s/catalog.db", dir);
250
251 cat = pblock_catalog_open(db, 2000);
252 CHECK(cat != NULL, "catalog_open failed: %s", strerror(errno));
253 if (cat == NULL) {
254 return 1;
255 }
256
257 test_put_lookup(cat);
258 test_parent_gate(cat);
259 test_touch(cat);
260 test_children_and_remove(cat);
261 test_rename_subtree(cat);
262 test_opendir(cat);
263 test_xattr(cat);
264
265 pblock_catalog_close(cat);
266 unlink(db);
267 rmdir(dir);
268
269 if (failures == 0) {
270 printf("sd_pblock_catalog_unittest: ALL PASS\n");
271 return 0;
272 }
273 fprintf(stderr, "sd_pblock_catalog_unittest: %d FAILURE(S)\n", failures);
274 return 1;
275}
