38

64static void
65test_put_lookup(pblock_catalog *cat)
66{

68 pblock_meta out;
69

73 CHECK(out.is_dir == 0, "is_dir wrong");

83/* touch updates only size+mtime and only for existing rows. */
84static void
85test_parent_gate(pblock_catalog *cat)
86{
87 /* phase-68 orphan fix: every create requires its immediate parent to
88 * exist and be a directory — an orphan row (reachable by key, invisible
89 * to every listing) must be impossible. */
90 pblock_meta in, out;
91
92 memset(&in, 0, sizeof(in));
93 in.mode = S_IFREG | 0644;
94
95 errno = 0;
96 CHECK(pblock_catalog_put(cat, "/ghostdir/file", &in) != 0
97 && errno == ENOENT, "orphan put allowed (errno %d)", errno);
98 CHECK(pblock_catalog_create(cat, "/ghostdir/sub/leaf", &in) != 0
99 && errno == ENOENT, "orphan create allowed (errno %d)", errno);
100 CHECK(pblock_catalog_lookup(cat, "/ghostdir/file", &out) == 1,
101 "orphan row landed anyway");
102
103 /* parent present but NOT a directory -> ENOTDIR */
104 CHECK(pblock_catalog_put(cat, "/plainfile", &in) == 0, "seed file");
105 errno = 0;
106 CHECK(pblock_catalog_put(cat, "/plainfile/child", &in) != 0
107 && errno == ENOTDIR, "file-as-parent allowed (errno %d)", errno);
108
109 /* the chain in order works, and rename into a missing parent fails */
110 memset(&in, 0, sizeof(in));
111 in.is_dir = 1;
112 in.mode = S_IFDIR | 0755;
113 CHECK(pblock_catalog_put(cat, "/ghostdir", &in) == 0, "mkdir parent");
114 memset(&in, 0, sizeof(in));
115 in.mode = S_IFREG | 0644;
116 CHECK(pblock_catalog_put(cat, "/ghostdir/file", &in) == 0,
117 "put after mkdir");
118 errno = 0;
119 CHECK(pblock_catalog_rename(cat, "/ghostdir/file", "/nowhere/f") != 0
120 && errno == ENOENT, "rename into missing parent allowed");
121 CHECK(pblock_catalog_remove(cat, "/ghostdir/file") == 0, "cleanup file");
122 CHECK(pblock_catalog_remove(cat, "/ghostdir") == 0, "cleanup dir");
123 CHECK(pblock_catalog_remove(cat, "/plainfile") == 0, "cleanup plain");
124}

182/* opendir enumerates direct children only (basenames). */
183static void
184test_opendir(pblock_catalog *cat)
185{
186 pblock_meta d = dir_meta();
187 pblock_meta f = file_meta("dddddddddddddddddddddddddddddddd", 1);
188 pblock_catalog_iter *it;
189 char name[256];
190 int seen_a = 0, seen_b = 0, n = 0, rc;
191
192 CHECK(pblock_catalog_put(cat, "/list", &d) == 0, "mkdir list");
193 CHECK(pblock_catalog_put(cat, "/list/a", &f) == 0, "a");
194 CHECK(pblock_catalog_put(cat, "/list/b", &f) == 0, "b");
195 /* "/list/b" is a FILE: since the phase-68 parent gate this insert must
196 * fail ENOTDIR outright (it used to land as an orphan row that readdir
197 * could never reach — the listing-shape CHECKs below still hold). */
198 errno = 0;
199 CHECK(pblock_catalog_put(cat, "/list/b/nested", &f) != 0
200 && errno == ENOTDIR, "nested under a file allowed (errno %d)",
201 errno);
202
203 it = pblock_catalog_opendir(cat, "/list");
204 CHECK(it != NULL, "opendir failed");
205 while ((rc = pblock_catalog_readdir(it, name, sizeof(name))) == 0) {
206 if (strcmp(name, "a") == 0) { seen_a = 1; }
207 if (strcmp(name, "b") == 0) { seen_b = 1; }
208 CHECK(strchr(name, '/') == NULL, "readdir returned a path: %s", name);
209 n++;
210 }
211 CHECK(rc == 1, "readdir ended with error %d", rc);
212 pblock_catalog_closedir(it);
213 CHECK(seen_a && seen_b, "missing children a=%d b=%d", seen_a, seen_b);
214 CHECK(n == 2, "expected 2 direct children, got %d", n);
215}

242/* the identity registry assigns stable ids >= PBLOCK_ID_BASE, never renumbers,
243 * keeps user/group namespaces separate, and rejects empty names. */
244static void
245test_id_map(pblock_catalog *cat)
246{
247 int64_t alice1 = 0, alice2 = 0, bob = 0, atlas = 0, alice_grp = 0;
248
249 CHECK(pblock_catalog_id_map(cat, PBLOCK_ID_USER, "alice", &alice1) == 0,
250 "map alice: %s", strerror(errno));
251 CHECK(alice1 >= PBLOCK_ID_BASE, "alice id below base: %lld",
252 (long long) alice1);
253 CHECK(pblock_catalog_id_map(cat, PBLOCK_ID_USER, "alice", &alice2) == 0
254 && alice2 == alice1, "alice id not stable: %lld vs %lld",
255 (long long) alice1, (long long) alice2);
256
257 CHECK(pblock_catalog_id_map(cat, PBLOCK_ID_USER, "bob", &bob) == 0
258 && bob != alice1, "bob id collides with alice");
259 CHECK(pblock_catalog_id_map(cat, PBLOCK_ID_GROUP, "atlas", &atlas) == 0
260 && atlas != alice1 && atlas != bob, "atlas gid collides");
261 /* the same NAME in the group namespace is a separate identity */
262 CHECK(pblock_catalog_id_map(cat, PBLOCK_ID_GROUP, "alice", &alice_grp) == 0
263 && alice_grp != alice1, "group 'alice' shares the user id");
264
265 errno = 0;
266 CHECK(pblock_catalog_id_map(cat, PBLOCK_ID_USER, "", &bob) == -1
267 && errno == EINVAL, "empty name accepted (errno %d)", errno);

271static void
272test_setattr_owner(pblock_catalog *cat)
273{
274 pblock_meta f = file_meta("ffffffffffffffffffffffffffffffff", 1);
275 pblock_meta out;
276

278 f.gid = 10002;

280 CHECK(pblock_catalog_lookup(cat, "/owned", &out) == 0

282 "put owner not stored: %u/%u", out.uid, out.gid);
283
284 /* chgrp only: uid=-1 keeps the owner */
285 CHECK(pblock_catalog_setattr(cat, "/owned", 0, 0, 1, -1, 10007,
286 0, 0, 2000) == 0, "chgrp failed");
287 CHECK(pblock_catalog_lookup(cat, "/owned", &out) == 0
288 && out.uid == 10001 && out.gid == 10007,
289 "chgrp result %u/%u", out.uid, out.gid);
290 CHECK(out.ctime == 2000, "ctime not bumped: %lld", (long long) out.ctime);
291
292 /* set_owner=0 must not touch the columns even with ids present */
293 CHECK(pblock_catalog_setattr(cat, "/owned", 1, 0600, 0, 10099, 10099,
294 0, 0, 2001) == 0, "chmod failed");
295 CHECK(pblock_catalog_lookup(cat, "/owned", &out) == 0
296 && out.uid == 10001 && out.gid == 10007 && (out.mode & 0777) == 0600,
297 "chmod leaked ownership: %u/%u mode %o", out.uid, out.gid, out.mode);
298
299 errno = 0;
