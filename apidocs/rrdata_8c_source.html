1/*
2 * rrdata.c — typed decode of a CMS request/forwarded-op payload. See rrdata.h.
3 *
4 * Pure C, no nginx dependency, so the byte-exact wire decode is unit-testable
5 * standalone (rrdata_unittest.c). The per-opcode field order reproduces
6 * XrdCmsParser's Pup arg vectors exactly (fwdArgA/B/C, padArgs, pdlArgs,
7 * locArgs); the "Fence" in those vectors is a parser marker, not a wire byte, so
8 * post-fence fields are simply the optional trailing fields here.
9 */

31/*
32 * Read one XrdOucPup string at *p: [2B BE len][bytes][NUL], where len counts the
33 * trailing NUL. A zero length is an absent/empty string (out=NULL). The returned
34 * span EXCLUDES the NUL, so out[out_len] is the in-place NUL terminator. Advances
35 * *p. Returns 0 on success, -1 on a short/overrun buffer.
36 */
37static int
38read_str(const unsigned char **p, const unsigned char *end,
39 const unsigned char **out, size_t *out_len)
40{
41 unsigned len;
42
43 if (end - *p < 2) {
44 return -1;
45 }
46 len = ((unsigned) (*p)[0] << 8) | (unsigned) (*p)[1];
47 *p += 2;
48
49 if (len == 0) {
50 *out = NULL;
51 *out_len = 0;
52 return 0;
53 }
54 if ((size_t) (end - *p) < len) {
55 return -1;
56 }
57 *out = *p;
58 *out_len = len - 1; /* drop the trailing NUL from the content span */
59 *p += len;
60 return 0;
61}

63/*
64 * Read an optional trailing Pup string: absent (no bytes left) is success with
65 * out=NULL; otherwise it must decode cleanly.
66 */
67static int
68read_opt_str(const unsigned char **p, const unsigned char *end,
69 const unsigned char **out, size_t *out_len)
70{
71 if (*p >= end) {
72 *out = NULL;
73 *out_len = 0;
74 return 0;
75 }
76 return read_str(p, end, out, out_len);
77}

79/* Read a tagged Pup int (CMS_PT_INT + 4B BE). Advances *p. 0 / -1. */
80static int
81read_int(const unsigned char **p, const unsigned char *end, uint32_t *v)
82{
83 if (end - *p < 5 || **p != CMS_PT_INT) {
84 return -1;
85 }
86 *v = ((uint32_t) (*p)[1] << 24) | ((uint32_t) (*p)[2] << 16)
87 | ((uint32_t) (*p)[3] << 8) | (uint32_t) (*p)[4];
88 *p += 5;
89 return 0;
90}

93 * Per-arg-group decoders. Each mirrors one XrdCmsParser Pup arg vector
94 * (fwdArgA/B/C, padArgs, pdlArgs, locArgs); field order is wire-frozen.
95 * All take the shared cursor (*p advanced in place, bounded by end) and fill
96 * the matching fields of *out. Return 0 on success, -1 on a short buffer.

99/* fwdArgA: ident, mode, path, [opaque] — kYR_chmod/mkdir/mkpath/trunc. */
100static int
101parse_fwd_a(const unsigned char **p, const unsigned char *end,
102 brix_cms_rrdata_t *out)
103{
104 if (read_str(p, end, &out->ident, &out->ident_len) != 0) return -1;
105 if (read_str(p, end, &out->mode, &out->mode_len) != 0) return -1;
106 if (read_str(p, end, &out->path, &out->path_len) != 0) return -1;
107 return read_opt_str(p, end, &out->opaque, &out->opaque_len);
108}

168int
169brix_cms_rrdata_parse(unsigned char code,
170 const unsigned char *payload, size_t len,
171 brix_cms_rrdata_t *out)
172{
173 const unsigned char *p = payload;
174 const unsigned char *end = payload + len;
175
176 memset(out, 0, sizeof(*out));
177
178 switch (code) {
179
180 case K_CHMOD:
181 case K_MKDIR:
182 case K_MKPATH:
183 case K_TRUNC:
184 return parse_fwd_a(&p, end, out);
185
186 case K_MV:
187 return parse_fwd_b(&p, end, out);
188
189 case K_RM:
190 case K_RMDIR:
191 case K_STATFS:
192 return parse_fwd_c(&p, end, out);
193
194 case K_PREPADD:
195 return parse_pad(&p, end, out);
196
197 case K_PREPDEL:
198 return parse_pdl(&p, end, out);
199
200 case K_LOCATE:
201 case K_SELECT:
202 return parse_loc(&p, end, out);
203
204 default:
205 return -1;
206 }
207}

209/* Encode one Pup string into [*p,end): [2B BE len incl NUL][bytes][NUL]; an empty
210 * string is a bare 2-byte zero. Advances *p. Returns 0 / -1 on overflow. */
211static int
212enc_str(unsigned char **p, unsigned char *end, const char *s)
213{
214 size_t n = (s != NULL) ? strlen(s) : 0;
215
216 if (n == 0) {
217 if (end - *p < 2) {
218 return -1;
219 }
220 (*p)[0] = 0;
221 (*p)[1] = 0;
222 *p += 2;
223 return 0;
224 }
225 if ((size_t) (end - *p) < n + 3) { /* 2 len + n + NUL */
226 return -1;
227 }
228 (*p)[0] = (unsigned char) (((unsigned) (n + 1)) >> 8);
229 (*p)[1] = (unsigned char) (n + 1);
230 memcpy(*p + 2, s, n);
231 (*p)[2 + n] = '\0';
232 *p += n + 3;
233 return 0;
234}

236/*
237 * Encode a fixed sequence of mandatory Pup strings followed by an optional
238 * trailing opaque (omitted entirely when opaque==NULL — the wire form for an
239 * absent optional field). Shared body of every rrdata_encode arg group; the
240 * caller supplies the group's field order. Returns 0 / -1 on overflow.
241 */
242static int
243enc_seq(unsigned char **p, unsigned char *end,
244 const char *const *fields, size_t nfields, const char *opaque)
245{
246 size_t i;
247
248 for (i = 0; i < nfields; i++) {
249 if (enc_str(p, end, fields[i]) != 0) {
250 return -1;
251 }
252 }
253 if (opaque != NULL && enc_str(p, end, opaque) != 0) {
254 return -1;
255 }
256 return 0;
257}

259int
260brix_cms_rrdata_encode(unsigned char code,
261 const brix_cms_fwd_fields_t *fields,
262 unsigned char *buf, size_t buflen)
263{
264 unsigned char *p = buf;
265 unsigned char *end = buf + buflen;
266 const char *seq[3];
267 size_t nseq;
268
269 switch (code) {
270
271 case K_CHMOD:
272 case K_MKDIR:
273 case K_MKPATH:
274 case K_TRUNC:
275 /* fwdArgA: ident, mode, path, [opaque] */
276 seq[0] = fields->ident; seq[1] = fields->mode; seq[2] = fields->path;
277 nseq = 3;
278 break;
279
280 case K_MV:
281 /* fwdArgB: ident, path, path2, [opaque] */
282 seq[0] = fields->ident; seq[1] = fields->path; seq[2] = fields->path2;
283 nseq = 3;
284 break;
285
286 case K_RM:
287 case K_RMDIR:
288 case K_STATFS:
289 /* fwdArgC: ident, path, [opaque] */
290 seq[0] = fields->ident; seq[1] = fields->path;
291 nseq = 2;
292 break;
293
294 default:
295 return -1;
296 }
297
298 if (enc_seq(&p, end, seq, nseq, fields->opaque) != 0) {
299 return -1;
300 }
301 return (int) (p - buf);
302}

304int
305brix_cms_statfs_encode(const brix_cms_statfs_fields_t *space,
306 unsigned char *buf, size_t buflen)
307{
308 int n;
309
310 if (buflen < 5) { /* 4-byte zero prefix + at least a NUL */
311 return -1;
312 }
313 buf[0] = buf[1] = buf[2] = buf[3] = 0;
314 n = snprintf((char *) buf + 4, buflen - 4, "%u %u %u %u %u %u",
315 space->w_num, space->w_free, space->w_util,
316 space->s_num, space->s_free, space->s_util);
317 if (n < 0 || (size_t) n >= buflen - 4) {
318 return -1;
319 }
320 return 4 + n + 1; /* prefix + string + trailing NUL */
321}
