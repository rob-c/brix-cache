1/*
2 * rrdata.c — typed decode of a CMS request/forwarded-op payload. See rrdata.h.
3 *
4 * Pure C, no nginx dependency, so the byte-exact wire decode is unit-testable
5 * standalone (rrdata_unittest.c). The per-opcode field order reproduces
6 * XrdCmsParser's Pup arg vectors exactly (fwdArgA/B/C, padArgs, pdlArgs,
7 * locArgs); the "Fence" in those vectors is a parser marker, not a wire byte, so
8 * post-fence fields are simply the optional trailing fields here.
9 */
10

31/*
32 * Read one XrdOucPup string at *p: [2B BE len][bytes][NUL], where len counts the
33 * trailing NUL. A zero length is an absent/empty string (out=NULL). The returned
34 * span EXCLUDES the NUL, so out[out_len] is the in-place NUL terminator. Advances
35 * *p. Returns 0 on success, -1 on a short/overrun buffer.
36 */
37static int
38read_str(const unsigned char **p, const unsigned char *end,
39 const unsigned char **out, size_t *out_len)
40{
41 unsigned len;
42
43 if (end - *p < 2) {
44 return -1;
45 }
46 len = ((unsigned) (*p)[0] << 8) | (unsigned) (*p)[1];
47 *p += 2;
48
49 if (len == 0) {
50 *out = NULL;
51 *out_len = 0;
52 return 0;
53 }
54 if ((size_t) (end - *p) < len) {
55 return -1;
56 }
57 *out = *p;
58 *out_len = len - 1; /* drop the trailing NUL from the content span */
59 *p += len;
60 return 0;
61}

63/*
64 * Read an optional trailing Pup string: absent (no bytes left) is success with
65 * out=NULL; otherwise it must decode cleanly.
66 */
67static int
68read_opt_str(const unsigned char **p, const unsigned char *end,
69 const unsigned char **out, size_t *out_len)
70{
71 if (*p >= end) {
72 *out = NULL;
73 *out_len = 0;
74 return 0;
75 }
76 return read_str(p, end, out, out_len);
77}

79/* Read a tagged Pup int (CMS_PT_INT + 4B BE). Advances *p. 0 / -1. */
80static int
81read_int(const unsigned char **p, const unsigned char *end, uint32_t *v)
82{
83 if (end - *p < 5 || **p != CMS_PT_INT) {
84 return -1;
85 }
86 *v = ((uint32_t) (*p)[1] << 24) | ((uint32_t) (*p)[2] << 16)
87 | ((uint32_t) (*p)[3] << 8) | (uint32_t) (*p)[4];
88 *p += 5;
89 return 0;
90}

92int
93brix_cms_rrdata_parse(unsigned char code,
94 const unsigned char *payload, size_t len,
95 brix_cms_rrdata_t *out)
96{
97 const unsigned char *p = payload;
98 const unsigned char *end = payload + len;
99
100 memset(out, 0, sizeof(*out));
101
102 switch (code) {
103
104 case K_CHMOD:
105 case K_MKDIR:
106 case K_MKPATH:
107 case K_TRUNC:
108 /* fwdArgA: ident, mode, path, [opaque] */
109 if (read_str(&p, end, &out->ident, &out->ident_len) != 0) return -1;
110 if (read_str(&p, end, &out->mode, &out->mode_len) != 0) return -1;
111 if (read_str(&p, end, &out->path, &out->path_len) != 0) return -1;
112 if (read_opt_str(&p, end, &out->opaque, &out->opaque_len) != 0) return -1;
113 return 0;
114
115 case K_MV:
116 /* fwdArgB: ident, path, path2, [opaque, opaque2] */
117 if (read_str(&p, end, &out->ident, &out->ident_len) != 0) return -1;
118 if (read_str(&p, end, &out->path, &out->path_len) != 0) return -1;
119 if (read_str(&p, end, &out->path2, &out->path2_len) != 0) return -1;
120 if (read_opt_str(&p, end, &out->opaque, &out->opaque_len) != 0) return -1;
121 if (read_opt_str(&p, end, &out->opaque2, &out->opaque2_len) != 0) return -1;
122 return 0;
123
124 case K_RM:
125 case K_RMDIR:
126 case K_STATFS:
127 /* fwdArgC: ident, path, [opaque] */
128 if (read_str(&p, end, &out->ident, &out->ident_len) != 0) return -1;
129 if (read_str(&p, end, &out->path, &out->path_len) != 0) return -1;
130 if (read_opt_str(&p, end, &out->opaque, &out->opaque_len) != 0) return -1;
131 return 0;
132
133 case K_PREPADD:
134 /* padArgs: ident, reqid, notify, prty, mode, path, [opaque] */
135 if (read_str(&p, end, &out->ident, &out->ident_len) != 0) return -1;
136 if (read_str(&p, end, &out->reqid, &out->reqid_len) != 0) return -1;
137 if (read_str(&p, end, &out->notify, &out->notify_len) != 0) return -1;
138 if (read_str(&p, end, &out->prty, &out->prty_len) != 0) return -1;
139 if (read_str(&p, end, &out->mode, &out->mode_len) != 0) return -1;
140 if (read_str(&p, end, &out->path, &out->path_len) != 0) return -1;
141 if (read_opt_str(&p, end, &out->opaque, &out->opaque_len) != 0) return -1;
142 return 0;
143
144 case K_PREPDEL:
145 /* pdlArgs: ident, reqid */
146 if (read_str(&p, end, &out->ident, &out->ident_len) != 0) return -1;
147 if (read_str(&p, end, &out->reqid, &out->reqid_len) != 0) return -1;
148 return 0;
149
150 case K_LOCATE:
151 case K_SELECT:
152 /* locArgs: ident, opts(int), path, [opaque, avoid] */
153 if (read_str(&p, end, &out->ident, &out->ident_len) != 0) return -1;
154 if (read_int(&p, end, &out->opts) != 0) return -1;
155 out->has_opts = 1;
156 if (read_str(&p, end, &out->path, &out->path_len) != 0) return -1;
157 if (read_opt_str(&p, end, &out->opaque, &out->opaque_len) != 0) return -1;
158 if (read_opt_str(&p, end, &out->avoid, &out->avoid_len) != 0) return -1;
159 return 0;
160
161 default:
162 return -1;
163 }
164}

166/* Encode one Pup string into [*p,end): [2B BE len incl NUL][bytes][NUL]; an empty
167 * string is a bare 2-byte zero. Advances *p. Returns 0 / -1 on overflow. */
168static int
169enc_str(unsigned char **p, unsigned char *end, const char *s)
170{
171 size_t n = (s != NULL) ? strlen(s) : 0;
172
173 if (n == 0) {
174 if (end - *p < 2) {
175 return -1;
176 }
177 (*p)[0] = 0;
178 (*p)[1] = 0;
179 *p += 2;
180 return 0;
181 }
182 if ((size_t) (end - *p) < n + 3) { /* 2 len + n + NUL */
183 return -1;
184 }
185 (*p)[0] = (unsigned char) (((unsigned) (n + 1)) >> 8);
186 (*p)[1] = (unsigned char) (n + 1);
187 memcpy(*p + 2, s, n);
188 (*p)[2 + n] = '\0';
189 *p += n + 3;
190 return 0;
191}

193int
194brix_cms_rrdata_encode(unsigned char code, const char *ident,
195 const char *path, const char *path2,
196 const char *mode, const char *opaque,
197 unsigned char *buf, size_t buflen)
198{
199 unsigned char *p = buf;
200 unsigned char *end = buf + buflen;
201
202 switch (code) {
203
204 case K_CHMOD:
205 case K_MKDIR:
206 case K_MKPATH:
207 case K_TRUNC:
208 /* fwdArgA: ident, mode, path, [opaque] */
209 if (enc_str(&p, end, ident) != 0) return -1;
210 if (enc_str(&p, end, mode) != 0) return -1;
211 if (enc_str(&p, end, path) != 0) return -1;
212 if (opaque != NULL && enc_str(&p, end, opaque) != 0) return -1;
213 return (int) (p - buf);
214
215 case K_MV:
216 /* fwdArgB: ident, path, path2, [opaque] */
217 if (enc_str(&p, end, ident) != 0) return -1;
218 if (enc_str(&p, end, path) != 0) return -1;
219 if (enc_str(&p, end, path2) != 0) return -1;
220 if (opaque != NULL && enc_str(&p, end, opaque) != 0) return -1;
221 return (int) (p - buf);
222
223 case K_RM:
224 case K_RMDIR:
225 case K_STATFS:
226 /* fwdArgC: ident, path, [opaque] */
227 if (enc_str(&p, end, ident) != 0) return -1;
228 if (enc_str(&p, end, path) != 0) return -1;
229 if (opaque != NULL && enc_str(&p, end, opaque) != 0) return -1;
230 return (int) (p - buf);
231
232 default:
233 return -1;
234 }
235}

238brix_cms_statfs_encode(uint32_t w_num, uint32_t w_free, uint32_t w_util,
239 uint32_t s_num, uint32_t s_free, uint32_t s_util,
240 unsigned char *buf, size_t buflen)
241{
242 int n;
243
244 if (buflen < 5) { /* 4-byte zero prefix + at least a NUL */
245 return -1;
246 }
247 buf[0] = buf[1] = buf[2] = buf[3] = 0;
248 n = snprintf((char *) buf + 4, buflen - 4, "%u %u %u %u %u %u",
249 w_num, w_free, w_util, s_num, s_free, s_util);
250 if (n < 0 || (size_t) n >= buflen - 4) {
251 return -1;
252 }
253 return 4 + n + 1; /* prefix + string + trailing NUL */
254}
