1/*
2 * stream_mirror_io.h — shared shadow-socket framing primitives for the two
3 * stream mirrors.
4 *
5 * WHAT: The read mirror (stream_mirror.c) and the write mirror
6 * (stream_wmirror.c) drive the same tiny async TCP exchange against a
7 * shadow XRootD server: drain a write buffer non-blocking, then read one
8 * 8-byte ServerResponseHdr plus a bounded body, both resumable across
9 * NGX_AGAIN. Those two primitives were duplicated byte-for-byte; they live
10 * here once and operate on plain field pointers so neither mirror has to
11 * expose its context struct.
12 *
13 * WHY: The framing is security-sensitive (the body length is attacker-supplied
14 * and must be capped) and lifetime-sensitive (cursors must survive partial
15 * reads). One copy means one place to audit and fix.
16 *
17 * HOW: Callers pass their own connection plus pointers to the cursor/buffer
18 * fields embedded in their context. The helpers never touch dispatch,
19 * teardown, or timers — those stay per-mirror.
20 */
21#ifndef BRIX_MIRROR_STREAM_MIRROR_IO_H
22#define BRIX_MIRROR_STREAM_MIRROR_IO_H

24#include "core/ngx_brix_module.h"

26/* Upper bound on a shadow response body, so a hostile/buggy shadow cannot make
27 * us allocate an arbitrary buffer. */
28#define BRIX_MIRROR_MAX_RESP_BODY 65536

30/*
31 * Drain wbuf[*wbuf_pos .. wbuf_len) to the shadow socket without blocking.
32 * Advances *wbuf_pos by the bytes written. Returns NGX_OK when fully drained
33 * (and the read event has been re-armed), NGX_AGAIN when the socket is full (and
34 * the write event has been re-armed), or NGX_ERROR on send/event-arm failure.
35 */
36ngx_int_t brix_mirror_io_flush(ngx_connection_t *c, const u_char *wbuf,
37 size_t wbuf_len, size_t *wbuf_pos);
38
39/*
40 * Read one ServerResponseHdr (XRD_RESPONSE_HDR_LEN bytes) followed by a
41 * dlen-byte body from the shadow. Resumable: the two cursors (*rhdr_pos,
42 * *resp_body_pos) survive NGX_AGAIN so a frame split across recv()s is
43 * reassembled across wakeups. The header is parsed exactly once (when *rhdr_pos
44 * first reaches the full header length) to latch *resp_status / *resp_dlen and
45 * allocate *resp_body from c->pool. Returns NGX_AGAIN until the whole frame is
46 * in hand, NGX_OK when complete, NGX_ERROR on EOF or an oversize body
47 * (> BRIX_MIRROR_MAX_RESP_BODY).
48 *
49 * rhdr must point at a caller-owned buffer of at least XRD_RESPONSE_HDR_LEN
50 * bytes; all cursors must be zeroed before the first frame.
51 */
52ngx_int_t brix_mirror_io_recv_frame(ngx_connection_t *c, u_char *rhdr,
53 size_t *rhdr_pos, uint16_t *resp_status, uint32_t *resp_dlen,
54 u_char **resp_body, size_t *resp_body_pos);
55
56#endif /* BRIX_MIRROR_STREAM_MIRROR_IO_H */
