29static long g_blk_count, g_blk_total, g_blk_max;

31static int
32blk_walk_cb(const char *path, const struct stat *sb, int type, struct FTW *ftw)
33{
34 (void) path;
35 (void) ftw;
36 if (type == FTW_F) {
37 g_blk_count++;
38 g_blk_total += (long) sb->st_size;
39 if ((long) sb->st_size > g_blk_max) {
40 g_blk_max = (long) sb->st_size;
41 }
42 }
43 return 0;
44}

46/* scan_blocks — tally the on-disk block files under <root>/data: count, total
47 * bytes, and the largest single block (which must never exceed block_size). */
48static void
49scan_blocks(const char *root)
50{
51 char data[512];
52
53 snprintf(data, sizeof(data), "%s/data", root);
54 g_blk_count = g_blk_total = g_blk_max = 0;
55 nftw(data, blk_walk_cb, 16, FTW_PHYS);
56}

58/* pblock_open_block_export — fresh export with a small block size so striping is
59 * exercised with a handful of bytes. Returns 0/-1 and fills *inst + *root. */
60int
61open_block_export(brix_sd_instance_t *inst, char *root, int64_t block_size)
62{
63 static brix_sd_pblock_conf_t conf; /* root must outlive the instance */
64
65 conf.root = root;
66 conf.busy_timeout_ms = 2000;
67 conf.block_size = block_size;
68 memset(inst, 0, sizeof(*inst));
69 inst->driver = D;
70 return D->init(inst, &conf) == NGX_OK ? 0 : -1;
71}

73/* A 40-byte file with a 16-byte block size lands as three block files
74 * (16 + 16 + 8); reads (incl. across a block boundary) are byte-exact. */
75void
76test_block_striping(void)
77{
78 char root[] = "/tmp/pb_blk.XXXXXX";
79 brix_sd_instance_t inst;
80 brix_sd_obj_t *o;
81 brix_sd_stat_t st;
82 char data[40], buf[40], part[20];
83 int i, err = 0;
84
85 CHECK(mkdtemp(root) != NULL, "mkdtemp");
86 CHECK(open_block_export(&inst, root, 16) == 0, "init bs=16");
87 for (i = 0; i < 40; i++) {
88 data[i] = (char) ('A' + (i % 26));
89 }
90
91 CHECK(write_file(&inst, "/big", data, 40) == 0, "write 40");
92 CHECK(D->stat(&inst, "/big", &st) == NGX_OK && st.size == 40, "size 40");
93 CHECK(read_file(&inst, "/big", buf, sizeof(buf)) == 40
94 && memcmp(buf, data, 40) == 0, "read back 40 byte-exact");
95
96 o = D->open(&inst, "/big", BRIX_SD_O_READ, 0, &err);
97 CHECK(o != NULL, "reopen read");
98 if (o != NULL) {
99 CHECK(D->pread(o, part, 20, 10) == 20
100 && memcmp(part, data + 10, 20) == 0, "cross-boundary read");
101 pb_close(o);
102 }
103
104 scan_blocks(root);
105 CHECK(g_blk_count == 3, "expected 3 block files, got %ld", g_blk_count);
106 CHECK(g_blk_max <= 16, "a block exceeded block_size: %ld", g_blk_max);
107 CHECK(g_blk_total == 40, "block bytes total %ld != 40", g_blk_total);
108
109 D->cleanup(&inst);
110}

139test_block_sparse(void)

163 for (i = 0; i < 33; i++) {

166 CHECK(buf[33] == 'X' && buf[34] == 'Y', "sparse payload");
167

169 scan_blocks(root);
170 CHECK(g_blk_count == 2, "sparse blocks %ld != 2", g_blk_count);
171 D->cleanup(&inst);
