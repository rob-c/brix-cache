27/* ---- identity enforcement (the *_cred slots) ------------------------------- *
28 * alice + carol share VO "atlas"; bob is VO "cms"; a zeroed principal is the
29 * service (bypasses every check). Ownership is catalog-internal synthetic ids,
30 * so all assertions compare ids read back via stat — never fixed numbers. */

35 .vos = "cms" };

39/* cred_write_file — open_cred-create `path` as `who` with `mode`, write a few
40 * bytes, close. 0 or -1/errno. */
41static int
42cred_write_file(brix_sd_instance_t *inst, const char *path, mode_t mode,
43 const brix_sd_cred_t *who)
44{
45 int err = 0;
46 brix_sd_obj_t *o;
47 ssize_t n;
48
49 o = D->open_cred(inst, path,
50 BRIX_SD_O_WRITE | BRIX_SD_O_READ | BRIX_SD_O_CREATE,
51 mode, who, &err);
52 if (o == NULL) {
53 errno = err;
54 return -1;
55 }
56 n = D->pwrite(o, "data", 4, 0);
57 pb_close(o);
58 return n == 4 ? 0 : -1;
59}

61/* creation records the requester as owner; mode bits gate other users by
62 * class (owner / VO-group / other); the service bypasses everything. */
63static void
64test_ident_ownership(brix_sd_instance_t *inst)
65{
66 brix_sd_stat_t st;
67 brix_sd_obj_t *o;
68 int err = 0;
69
70 /* 0640: owner rw, VO-group r, other none */
71 CHECK(cred_write_file(inst, "/a.dat", 0640, &CRED_ALICE) == 0,
72 "alice create: %s", strerror(errno));
73 CHECK(D->stat(inst, "/a.dat", &st) == NGX_OK, "stat a.dat");
74 CHECK(st.uid >= PBLOCK_ID_BASE && st.gid >= PBLOCK_ID_BASE,
75 "synthetic owner not recorded: %u/%u",
76 (unsigned) st.uid, (unsigned) st.gid);
77
78 /* same user re-opens rw (owner class) */
79 o = D->open_cred(inst, "/a.dat", BRIX_SD_O_WRITE | BRIX_SD_O_READ, 0,
80 &CRED_ALICE, &err);
81 CHECK(o != NULL, "owner reopen rw: %s", strerror(err));
82 if (o != NULL) { pb_close(o); }
83
84 /* carol shares VO atlas: group class grants read, denies write */
85 o = D->open_cred(inst, "/a.dat", BRIX_SD_O_READ, 0, &CRED_CAROL, &err);
86 CHECK(o != NULL, "same-VO read: %s", strerror(err));
87 if (o != NULL) { pb_close(o); }
88 err = 0;
89 o = D->open_cred(inst, "/a.dat", BRIX_SD_O_WRITE, 0, &CRED_CAROL, &err);
90 CHECK(o == NULL && err == EACCES, "same-VO write allowed (err %d)", err);
91
92 /* bob (VO cms) is other class: no bits at all — neither read nor write */
93 err = 0;
94 o = D->open_cred(inst, "/a.dat", BRIX_SD_O_READ, 0, &CRED_BOB, &err);
95 CHECK(o == NULL && err == EACCES, "foreign-VO read allowed (err %d)", err);
96 err = 0;
97 o = D->open_cred(inst, "/a.dat", BRIX_SD_O_WRITE, 0, &CRED_BOB, &err);
98 CHECK(o == NULL && err == EACCES, "foreign-VO write allowed (err %d)",
99 err);
100
101 /* the service (NULL cred) bypasses the gate entirely */
102 o = D->open_cred(inst, "/a.dat", BRIX_SD_O_WRITE | BRIX_SD_O_READ, 0,
103 NULL, &err);
104 CHECK(o != NULL, "service bypass: %s", strerror(err));
105 if (o != NULL) { pb_close(o); }
106
107 /* a group-WRITABLE file (0660) really is shared within the VO: carol may
108 * write to alice's file and read alice's bytes back — bob still may not */
109 CHECK(cred_write_file(inst, "/shared.dat", 0660, &CRED_ALICE) == 0,
110 "alice create shared: %s", strerror(errno));
111 o = D->open_cred(inst, "/shared.dat", BRIX_SD_O_WRITE | BRIX_SD_O_READ,
112 0, &CRED_CAROL, &err);
113 CHECK(o != NULL, "same-VO group write: %s", strerror(err));
114 if (o != NULL) {
115 char buf[8];
116
117 CHECK(D->pread(o, buf, 4, 0) == 4 && memcmp(buf, "data", 4) == 0,
118 "same-VO read of alice's bytes");
119 CHECK(D->pwrite(o, "EDIT", 4, 0) == 4, "same-VO pwrite");
120 pb_close(o);
121 }
122 err = 0;
123 o = D->open_cred(inst, "/shared.dat", BRIX_SD_O_WRITE, 0, &CRED_BOB,
124 &err);
125 CHECK(o == NULL && err == EACCES,
126 "foreign-VO write on group file allowed (err %d)", err);
127}

129/* per-VO shared directories: group-writable dir admits VO members and no one
130 * else, for mkdir, staged publish and opendir alike. */
131static void
132test_ident_vo_dir(brix_sd_instance_t *inst)
133{
134 brix_sd_staged_t *sh;
135 brix_sd_stat_t st, sub;
136 brix_sd_dir_t *dir;
137 int err = 0;
138
139 /* alice's VO-shared dir: 0770 — atlas members rwx, others nothing */
140 CHECK(D->mkdir_cred(inst, "/atlas", 0770, &CRED_ALICE) == NGX_OK,
141 "mkdir /atlas: %s", strerror(errno));
142 CHECK(D->stat(inst, "/atlas", &st) == NGX_OK
143 && st.uid >= PBLOCK_ID_BASE, "dir owner missing");
144
145 CHECK(D->mkdir_cred(inst, "/atlas/carol", 0770, &CRED_CAROL) == NGX_OK,
146 "same-VO mkdir: %s", strerror(errno));
147 CHECK(D->stat(inst, "/atlas/carol", &sub) == NGX_OK
148 && sub.uid != st.uid, "subdir not owned by carol");
149
150 errno = 0;
151 CHECK(D->mkdir_cred(inst, "/atlas/bob", 0770, &CRED_BOB) == NGX_ERROR
152 && errno == EACCES, "foreign-VO mkdir allowed (errno %d)", errno);
153
154 /* staged publish obeys the same parent gate and records the requester */
155 err = 0;
156 sh = D->staged_open_cred(inst, "/atlas/stage.dat", 0640, &CRED_BOB, &err);
157 CHECK(sh == NULL && err == EACCES, "foreign-VO staged (err %d)", err);
158 sh = D->staged_open_cred(inst, "/atlas/stage.dat", 0640, &CRED_CAROL,
159 &err);
160 CHECK(sh != NULL, "same-VO staged: %s", strerror(err));
161 if (sh != NULL) {
162 CHECK(D->staged_write(sh, "zz", 2, 0) == 2, "staged write");
163 CHECK(D->staged_commit(sh, 0) == NGX_OK, "staged commit");
164 CHECK(D->stat(inst, "/atlas/stage.dat", &st) == NGX_OK
165 && st.uid == sub.uid, "staged row not owned by carol");
166 }
167
168 /* listing = reading the directory */
169 errno = 0;
170 dir = D->opendir_cred(inst, "/atlas", &err, &CRED_BOB);
171 CHECK(dir == NULL && err == EACCES, "foreign-VO opendir (err %d)", err);
172 dir = D->opendir_cred(inst, "/atlas", &err, &CRED_CAROL);
173 CHECK(dir != NULL, "same-VO opendir: %s", strerror(err));
174 if (dir != NULL) { D->closedir(dir); }
175}

177/* directory traverse: the parent's X bit gates every access to entries inside
178 * it — a world-readable file in a 0770 group dir stays invisible to
179 * non-members (runs after test_ident_vo_dir; reuses its /atlas dir). */
180static void
181test_ident_traverse(brix_sd_instance_t *inst)
182{
183 brix_sd_obj_t *o;
184 char v[8];
185 int err = 0;
186
187 /* 0644 would grant bob other-read — but /atlas (0770) must block him */
188 CHECK(cred_write_file(inst, "/atlas/pub.dat", 0644, &CRED_ALICE) == 0,
189 "alice create in /atlas: %s", strerror(errno));
190
191 o = D->open_cred(inst, "/atlas/pub.dat", BRIX_SD_O_READ, 0, &CRED_CAROL,
192 &err);
193 CHECK(o != NULL, "same-VO traverse+read: %s", strerror(err));
194 if (o != NULL) { pb_close(o); }
195
196 err = 0;
197 o = D->open_cred(inst, "/atlas/pub.dat", BRIX_SD_O_READ, 0, &CRED_BOB,
198 &err);
199 CHECK(o == NULL && err == EACCES,
200 "no-traverse read allowed (err %d)", err);
201 err = 0;
202 o = D->open_cred(inst, "/atlas/new.dat",
203 BRIX_SD_O_WRITE | BRIX_SD_O_CREATE, 0644, &CRED_BOB,
204 &err);
205 CHECK(o == NULL && err == EACCES,
206 "no-traverse create allowed (err %d)", err);
207 errno = 0;
208 CHECK(D->getxattr_cred(inst, "/atlas/pub.dat", "user.k", v, sizeof(v),
209 &CRED_BOB) < 0 && errno == EACCES,
210 "no-traverse getxattr allowed (errno %d)", errno);
211 errno = 0;
212 CHECK(D->unlink_cred(inst, "/atlas/pub.dat", 0, &CRED_BOB) == NGX_ERROR
213 && errno == EACCES,
214 "no-traverse unlink allowed (errno %d)", errno);
215
216 /* the service still bypasses traverse like every other gate */
217 o = D->open_cred(inst, "/atlas/pub.dat", BRIX_SD_O_READ, 0, NULL, &err);
218 CHECK(o != NULL, "service traverse bypass: %s", strerror(err));
219 if (o != NULL) { pb_close(o); }
220}

222/* chmod/chown policy: owner-only chmod; no giving files away; chgrp only into
223 * a VO the owner belongs to. */
224static void
225test_ident_setattr(brix_sd_instance_t *inst)
226{
227 brix_sd_setattr_t attr;
228 brix_sd_stat_t st, bobst;
229
230 CHECK(cred_write_file(inst, "/chown.me", 0640, &CRED_ALICE) == 0,
231 "seed chown.me");
232 CHECK(cred_write_file(inst, "/bob.dat", 0640, &CRED_BOB) == 0,
233 "seed bob.dat");
234 CHECK(D->stat(inst, "/chown.me", &st) == NGX_OK, "stat chown.me");
235 CHECK(D->stat(inst, "/bob.dat", &bobst) == NGX_OK, "stat bob.dat");
236
237 /* chmod: owner yes, non-owner EPERM */
238 memset(&attr, 0, sizeof(attr));
239 attr.set_mode = 1;
240 attr.mode = 0664;
241 CHECK(D->setattr_cred(inst, "/chown.me", &attr, &CRED_ALICE) == NGX_OK,
242 "owner chmod: %s", strerror(errno));
243 errno = 0;
244 CHECK(D->setattr_cred(inst, "/chown.me", &attr, &CRED_BOB) == NGX_ERROR
245 && errno == EPERM, "non-owner chmod allowed (errno %d)", errno);
246
247 /* chown to another uid is service-only */
248 memset(&attr, 0, sizeof(attr));
249 attr.set_owner = 1;
250 attr.uid = (uid_t) bobst.uid;
251 attr.gid = (gid_t) -1;
252 errno = 0;
253 CHECK(D->setattr_cred(inst, "/chown.me", &attr, &CRED_ALICE) == NGX_ERROR
254 && errno == EPERM, "give-away chown allowed (errno %d)", errno);
255 CHECK(D->setattr_cred(inst, "/chown.me", &attr, NULL) == NGX_OK,
256 "service chown: %s", strerror(errno));
257 CHECK(D->stat(inst, "/chown.me", &st) == NGX_OK && st.uid == bobst.uid,
258 "service chown not applied");
259
260 /* chgrp: owner may move into own VO, not a foreign one */
261 memset(&attr, 0, sizeof(attr));
262 attr.set_owner = 1;
263 attr.uid = (uid_t) -1;
264 attr.gid = (gid_t) bobst.gid; /* cms — bob's VO */
265 CHECK(D->setattr_cred(inst, "/bob.dat", &attr, &CRED_BOB) == NGX_OK,
266 "chgrp into own VO: %s", strerror(errno));
267 CHECK(D->stat(inst, "/atlas", &st) == NGX_OK, "stat /atlas");
268 attr.gid = (gid_t) st.gid; /* atlas — NOT bob's VO */
269 errno = 0;
270 CHECK(D->setattr_cred(inst, "/bob.dat", &attr, &CRED_BOB) == NGX_ERROR
271 && errno == EPERM, "chgrp into foreign VO allowed (errno %d)",
272 errno);
273}

275/* xattr mode gates (R to read, W to write) and the sticky top level: users
276 * cannot delete or rename each other's root entries, owners can. */
277static void
278test_ident_xattr_sticky(brix_sd_instance_t *inst)
279{
280 char buf[16];
281 ssize_t n;
282 int alice_err = 0;
283
284 /* /a.dat is alice's, 0640 (VO atlas may read) */
285 CHECK(D->setxattr_cred(inst, "/a.dat", "user.t", "1", 1, 0,
286 &CRED_ALICE) == NGX_OK,
287 "owner setxattr: %s", strerror(errno));
288 n = D->getxattr_cred(inst, "/a.dat", "user.t", buf, sizeof(buf),
289 &CRED_CAROL);
290 CHECK(n == 1 && buf[0] == '1', "same-VO getxattr n=%zd", n);
291 errno = 0;
292 CHECK(D->setxattr_cred(inst, "/a.dat", "user.t", "2", 1, 0,
293 &CRED_CAROL) == NGX_ERROR && errno == EACCES,
294 "read-only VO member setxattr allowed (errno %d)", errno);
295 errno = 0;
296 n = D->listxattr_cred(inst, "/a.dat", buf, sizeof(buf), &CRED_BOB);
297 CHECK(n == -1 && errno == EACCES, "foreign-VO listxattr n=%zd", n);
298 errno = 0;
299 CHECK(D->removexattr_cred(inst, "/a.dat", "user.t",
300 &CRED_BOB) == NGX_ERROR && errno == EACCES,
301 "foreign-VO removexattr allowed (errno %d)", errno);
302
303 /* the denials are symmetric: alice cannot write to or delete bob's file
304 * either (/bob.dat is 0640, gid cms — alice is other class) */
305 errno = 0;
306 CHECK(D->open_cred(inst, "/bob.dat", BRIX_SD_O_WRITE, 0, &CRED_ALICE,
307 &alice_err) == NULL && alice_err == EACCES,
308 "alice write on bob's file allowed (err %d)", alice_err);
309 errno = 0;
310 CHECK(D->unlink_cred(inst, "/bob.dat", 0, &CRED_ALICE) == NGX_ERROR
311 && errno == EPERM, "alice unlink of bob's file allowed (errno %d)",
312 errno);
313
314 /* sticky synthetic root: only the owner removes/renames a root entry */
315 errno = 0;
316 CHECK(D->unlink_cred(inst, "/a.dat", 0, &CRED_BOB) == NGX_ERROR
317 && errno == EPERM, "cross-user root unlink allowed (errno %d)",
318 errno);
319 errno = 0;
320 CHECK(D->rename_cred(inst, "/a.dat", "/stolen.dat", 0,
321 &CRED_BOB) == NGX_ERROR && errno == EPERM,
322 "cross-user root rename allowed (errno %d)", errno);
323 CHECK(D->rename_cred(inst, "/a.dat", "/a2.dat", 0,
324 &CRED_ALICE) == NGX_OK,
325 "owner root rename: %s", strerror(errno));
326 CHECK(D->unlink_cred(inst, "/a2.dat", 0, &CRED_ALICE) == NGX_OK,
327 "owner root unlink: %s", strerror(errno));
328
329 /* server_copy: source readability is enforced */
330 errno = 0;
331 CHECK(D->server_copy_cred(inst, "/bob.dat", "/copy.dat", NULL,
332 &CRED_CAROL) == NGX_ERROR && errno == EACCES,
333 "unreadable-source copy allowed (errno %d)", errno);
334 CHECK(D->server_copy_cred(inst, "/bob.dat", "/copy.dat", NULL,
335 &CRED_BOB) == NGX_OK,
336 "owner copy: %s", strerror(errno));
337}
