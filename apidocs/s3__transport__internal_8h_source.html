4/*
5 * s3_transport_internal.h — shared internals of the libcurl origin transport,
6 * split across two translation units with NO behavior change:
7 * s3_transport.c — response capture buffers + callbacks, the request
8 * execution body, the vtable entry points/accessors.
9 * s3_transport_setup.c — operator policy (globals + public setters), the
10 * per-thread curl handle lifecycle, per-request curl

12 * Not a public interface; see s3_transport.h for the exported transport vtable.

28/* s3o_request_t — one origin request's inputs, shared by the plain and
29 * cred-scoped transport slots.
30 *
31 * WHAT: The immutable inputs of a single synchronous libcurl request: the
32 * target (host/port/tls), the HTTP call (method/path/headers/body), the
33 * caller timeout, the OPTIONAL mutual-TLS client cert, and the operator CA
34 * context (`tctx`).
35 * WHY: Both vtable entry points (s3o_request / s3o_request_cred) pass the same
36 * dozen-plus arguments straight through to one shared body; collapsing
37 * them into a file-local descriptor keeps the shared body and its extracted
38 * helpers at one argument instead of fourteen, with no behavior change.
39 * HOW: `tctx` is the transport's OPTIONAL context — a NUL-terminated CA
40 * file-or-dir PATH (the operator's trusted CA for origin TLS), or NULL for
41 * libcurl's system bundle; applied via s3o_apply_ca(). `client_cert_pem`
42 * is NULL for the plain slot; on a `tls` request a non-empty path is
43 * presented as the mutual-TLS client cert (cert chain + key in one PEM). */
44typedef struct {
45 void *tctx; /* operator CA path (via s3o_apply_ca), or NULL */
46 const char *host;
47 int port;
48 int tls;
49 const char *method;
50 const char *path_and_query;
51 const char *headers;
52 const void *body;
53 size_t body_len;
54 int timeout_ms;
55 const char *client_cert_pem; /* mutual-TLS client PEM, or NULL (plain slot) */
56} s3o_request_t;

58/* s3o_trace_t — the fields of one upstream-request trace line.
59 *
60 * WHAT: Bundles the request-identity (method/host/port/path) plus the outcome
61 * (status/bytes/duration/error) that s3o_trace() formats into a single
62 * line.
63 * WHY: The trace inputs were passed as 8 loose params; a small file-local
64 * descriptor keeps the emit signature at one argument and lets the two
65 * call sites (transport failure vs HTTP outcome) fill a stack struct in
66 * place with no behavior change.
67 * HOW: `status` < 0 marks a transport-level failure (curl error text in
68 * `err`); otherwise it is the HTTP status and `err` is NULL. `path` is
69 * wire-derived and gets sanitized by s3o_trace(). */
70typedef struct {
71 const char *method;
72 const char *host;
73 int port;
74 const char *path; /* path_and_query, wire-derived → sanitized */
75 int status; /* < 0 = transport failure, else HTTP status */
76 size_t bytes;
77 long dur_ms;
78 const char *err; /* curl error text when status < 0, else NULL */
79 const char *proto; /* negotiated HTTP version token, or NULL */
80} s3o_trace_t;
