1/* client/lib/cred_sss.c
2 *
3 * WHAT: SSS (Simple Shared Secret) credential handler for the unified
4 * credential store (cred.c). Implements brix_cred_sss() returning the
5 * XRDC_CRED_SSS handler with available / acquire operations.
6 * WHY: SSS keytab discovery was only available through sec_sss.c; the unified
7 * store needs a standalone handler so auth pre-flight diagnostics and the
8 * auth handshake both share a single keytab resolution path.
9 * HOW: Discovery precedence:
10 * cfg->keytab_path (CLI override, highest precedence)
11 * > brix_sss_keytab_default() ($XrdSecSSSKT > $XrdSecsssKT >
12 * ~/.xrd/sss.keytab)
13 * available() resolves the path and probes with brix_sss_keytab_read
14 * (must read successfully and n>0).
15 * acquire() fills out->path with a static buffer copy of the resolved path
16 * and sets *not_after=0 (a keytab has no per-use expiry).
17 * refresh=NULL: keytab is long-lived; no automatic refresh mechanism.
18 *
19 * Lifetime contract (cred.h): acquire() writes out->path as a pointer to
20 * a function-local static char[] that remains valid after acquire() returns.
21 * The store's slot_store_view strdup's out->path immediately afterwards —
22 * before any other acquire can overwrite the buffer. The store is
23 * single-threaded per acquire, so one static buffer is safe for this window.
24 *
25 * ngx-free. No goto. Functional/modular: one responsibility per function.
26 */

38/* keytab path resolution */
39/*
40 * resolve_keytab_path — fill out[outsz] with the resolved SSS keytab path.
41 *
42 * WHAT: implements the two-level precedence: cfg->keytab_path (CLI override) >
43 * brix_sss_keytab_default() (env/default discovery).
44 * WHY: mirrors the precedence documented in cred.h's keytab_path comment and
45 * matches the behaviour callers expect from sec_sss.c's discovery.
46 * HOW: two early-return branches; snprintf is always NUL-terminated within outsz.
47 */
48static void
49resolve_keytab_path(const brix_cred_config *cfg, char *out, size_t outsz)
50{
51 if (cfg != NULL && cfg->keytab_path != NULL && cfg->keytab_path[0] != '\0') {
52 snprintf(out, outsz, "%s", cfg->keytab_path);
53 return;
54 }
55 brix_sss_keytab_default(out, outsz);
56}

58/* keytab probe */
59/*
60 * keytab_probe — resolve the keytab path and check that it has at least one key.
61 *
62 * WHAT: resolves the path into path_out[outsz] and calls brix_sss_keytab_read;
63 * returns 1 iff the read succeeds and n>0.
64 * WHY: shared by sss_available() and sss_acquire() so the probe logic lives in
65 * one place (available and acquire must agree on what "usable" means).
66 * HOW: stack-allocated key array (no heap); brix_sss_keytab_read fills *n.
67 * path_out is filled by resolve_keytab_path so the caller learns the path
68 * (acquire uses it to populate out->path and the error message).
69 */
70static int
71keytab_probe(const brix_cred_config *cfg, char *path_out, size_t outsz)
72{
73 brix_sss_key keys[XRDC_SSS_KEYS_MAX];
74 brix_status st;
75 int n = 0;
76
77 resolve_keytab_path(cfg, path_out, outsz);
78 brix_status_clear(&st);
79 return (brix_sss_keytab_read(path_out, keys, XRDC_SSS_KEYS_MAX, &n, &st) == 0
80 && n > 0) ? 1 : 0;
81}

83/* handler callbacks */
84/*
85 * sss_available — 1 if the resolved keytab exists and contains at least one key.
86 *
87 * WHAT: fast probe for auth pre-flight diagnostics and brix_cred_available().
88 * WHY: mirrors sec_sss.c:sss_have() semantics (keytab readable + n>0).
89 * HOW: keytab_probe with a scratch path buffer (not exposed to the caller).
90 */
91static int
92sss_available(const brix_cred_config *cfg)
93{
94 char path[XRDC_PATH_MAX];
95 return keytab_probe(cfg, path, sizeof(path));
96}

98/*
99 * sss_acquire — resolve the keytab path, verify it is usable, fill *out.
100 *
101 * WHAT: the store's acquire call for XRDC_CRED_SSS.
102 * WHY: single canonical acquire path for the SSS keytab credential.
103 * HOW: 1) keytab_probe — resolves path into s_path and reads keys;
104 * 2) on failure → -1 + XRDC_EAUTH with the resolved path in the message;
105 * 3) fill out: kind=SSS, path=s_path, not_after=0 (no per-use expiry).
106 *
107 * Lifetime of s_path: declared static so the buffer remains valid after this
108 * function returns. slot_store_view strdup's out->path before any other acquire
109 * can overwrite the buffer.
110 */
111static int
112sss_acquire(const brix_cred_config *cfg, brix_cred_view *out,
113 int64_t *not_after, brix_status *st)
114{
115 /* static: must outlive this return so slot_store_view can strdup it */
116 static char s_path[XRDC_PATH_MAX];
117
118 if (!keytab_probe(cfg, s_path, sizeof(s_path))) {
119 brix_status_set(st, XRDC_EAUTH, 0,
120 "sss: no usable keytab at %s "
121 "(set XrdSecSSSKT or cfg->keytab_path)", s_path);
122 return -1;
123 }
124
125 out->kind = XRDC_CRED_SSS;
126 out->path = s_path; /* static buffer; store deep-copies via slot_store_view */
127 out->token = NULL;
128 out->s3_access = NULL;
129 out->s3_secret = NULL;
130
131 *not_after = 0; /* keytab has no per-use expiry */
132 return 0;
133}
