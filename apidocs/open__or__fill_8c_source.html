103/* ---- composed-cache (tier grammar) slow-tier miss offload (phase-64 SP2) ----
104 *
105 * A tier config (brix_storage_backend + brix_cache_store, conf->cache == 0)
106 * serves reads through the registry's composed sd_cache: its open() runs a MISS
107 * fill INLINE, which is a blocking wire transfer (s3/http HEAD+GET, root://
108 * login+read) — a stall on the stream event loop, and a self-connect deadlock
109 * when the source is served by this same worker. These three helpers are the
110 * stream twin of src/shared/http_cache_fill.c: probe with the decorator's
111 * non-blocking brix_sd_cache_fill_needs_offload, run the whole-file fill
112 * (brix_sd_cache_fill_key) on the async thread pool with the connection
113 * parked in XRD_ST_AIO, then serve the now-cached object from the done
114 * callback. A COMPLETE hit, slice mode, or an all-local stack never gets here
115 * (needs_offload == 0 → the caller opens inline as before). */
116
117/* Lazily resolve the stream plane's async pool: postconfig may leave
118 * common.thread_pool unset when the config only declares `thread_pool default`
119 * (the http_cache_fill.c idiom). NULL when no pool exists. */
120static ngx_thread_pool_t *
121cache_composed_fill_pool(ngx_stream_brix_srv_conf_t *conf)
122{
123 ngx_thread_pool_t *pool = conf->common.thread_pool;
124
125 if (pool == NULL) {
126 static ngx_str_t default_name = ngx_string("default");
127 ngx_str_t *pname = conf->common.thread_pool_name.len > 0
128 ? &conf->common.thread_pool_name
129 : &default_name;
130
131 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
132 if (pool != NULL) {
133 conf->common.thread_pool = pool;
134 }
135 }
136 return pool;
137}

153/* Event loop: resume the parked open — serve the now-cached object (NGX_OK), or
154 * open straight through the decorator for an admission-declined object
155 * (NGX_DECLINED: sd_cache_open serves from the source), or report the fill
156 * failure with the errno-mapped kXR code (ENOENT ⇒ the standard NotFound). */
157static void
158brix_cache_fill_composed_done(ngx_event_t *ev)
159{
160 ngx_thread_task_t *task = ev->data;
161 brix_cache_fill_t *t = task->ctx;
162 brix_ctx_t *ctx = t->ctx;
163 ngx_connection_t *c = t->c;
164 ngx_int_t rc;
165
166 if (!brix_aio_restore_request(ctx, t->streamid)) {
167 return;
168 }
169
170 if (t->result == NGX_ERROR) {
171 int err = t->sys_errno ? t->sys_errno : EIO;
172 uint16_t kxr = brix_kxr_from_errno(err);
173
174 brix_log_access(ctx, c, "OPEN", t->clean_path, "cache-fill", 0, kxr,
175 "composed cache fill failed", 0);
176 BRIX_OP_ERR(ctx, BRIX_OP_OPEN_RD);
177 brix_send_error(ctx, c, kxr,
178 (err == ENOENT) ? "file not found"
179 : "cache fill from source failed");
180 brix_aio_resume(c);
181 return;
182 }
183
184 if (t->result == NGX_OK) {
185 brix_log_access(ctx, c, "CACHE", t->cache_path, "fill", 1, 0, NULL, 0);
186 }
187
188 brix_open_request_t oreq = {
189 .resolved = t->cache_path,
190 .options = t->options,
191 .mode_bits = t->mode_bits,
192 .is_write = 0,
193 .codec = 0,
194 };
195 rc = brix_open_resolved_file(ctx, c, t->conf, &oreq);
196 if (rc != NGX_OK && ctx->state != XRD_ST_SENDING) {
197 brix_send_error(ctx, c, kXR_ServerError,
198 "open after cache fill failed");
199 }
200
201 brix_aio_resume(c);
202}
