10int
11do_status(const diag_args *a)
12{
13 brix_url u;
14 brix_status st;
15 char *body;
16 int http = 0, lines = 0, shown = 0;
17 char *line, *save;
18
19 brix_status_clear(&st);
20 if (brix_endpoint_parse(a->url, &u, &st) != 0) {
21 fprintf(stderr, "xrddiag: %s\n", st.msg);
22 return 50;
23 }
24 body = malloc(1u << 20);
25 if (body == NULL) {
26 fprintf(stderr, "xrddiag: out of memory\n");
27 return 51;
28 }
29 if (brix_http_get(u.host, a->metrics_port, "/metrics", 5000, &http, body,
30 1u << 20, NULL, &st) != 0) {
31 fprintf(stderr, "xrddiag: GET %s:%d/metrics: %s\n",
32 u.host, a->metrics_port, st.msg);
33 free(body);
34 return 51;
35 }
36 printf("Metrics from %s:%d (HTTP %d)\n", u.host, a->metrics_port, http);
37 for (line = strtok_r(body, "\n", &save); line != NULL;
38 line = strtok_r(NULL, "\n", &save)) {
39 if (line[0] == '#' || line[0] == '\0') {
40 continue;
41 }
42 lines++;
43 /* show the first handful of non-zero counters as a one-screen sample */
44 if (shown < 20 && strstr(line, " 0") != line + (int) strlen(line) - 2) {
45 printf(" %s\n", line);
46 shown++;
47 }
48 }
49 printf(" ... %d metric series total\n", lines);
50 free(body);
51 return lines > 0 ? 0 : 51;
52}

72/* Count whitespace/comma-separated tokens (≈ located replica hosts). We only ever
73 * emit the COUNT, never the locate buffer, so no host/IP leaks. */
74int
75watch_count_tokens(const char *s)
76{
77 int n = 0, in = 0;
78 for (; *s != '\0'; s++) {
79 int delim = (*s == ' ' || *s == '\t' || *s == '\r' || *s == '\n' || *s == ',');
80 if (!delim && !in) { in = 1; n++; }
81 else if (delim) { in = 0; }
82 }
83 return n;
84}

87/* Escape a Prometheus label value: backslash/quote/newline per the exposition
88 * format; any other control byte (\r, \t, …) is dropped so the output is always a
89 * single valid line (endpoint labels are host:port, so this is just defensive). */
90void
91watch_prom_label(const char *s, char *out, size_t osz)
92{
93 size_t j = 0;
94 for (; *s != '\0' && j + 2 < osz; s++) {
95 unsigned char ch = (unsigned char) *s;
96 if (ch == '\\' || ch == '"') { out[j++] = '\\'; out[j++] = (char) ch; }
97 else if (ch == '\n') { out[j++] = '\\'; out[j++] = 'n'; }
98 else if (ch < 0x20) { continue; } /* drop \r/\t/other controls */
99 else { out[j++] = (char) ch; }
100 }
101 out[j] = '\0';
102}

105/* Probe one endpoint once: connect (timed), connect-phase split, a tiny read
106 * (TTFB), and a locate (replica count). Never aborts the loop — a down endpoint
107 * just yields up=0. Always returns 0. */
108int
109watch_probe_once(const diag_args *a, const char *url, watch_sample *out)
110{
111 brix_url u;
112 brix_conn c;
113 brix_status st;
114 brix_netfacts f;
115 char target[XRDC_PATH_MAX];
116 brix_statinfo sti;
117 uint64_t t0;
118
119 memset(out, 0, sizeof(*out));
120 snprintf(out->endpoint, sizeof(out->endpoint), "%s", url);
121 out->read_ms = out->locate_ms = -1.0;
122 out->holders = -1;
123 out->proto = "root";
124
125 brix_status_clear(&st);
126 if (brix_endpoint_parse(url, &u, &st) != 0) {
127 /* PII-free contract: never let a raw URL's path/query reach a label. */
128 snprintf(out->endpoint, sizeof(out->endpoint), "(unparseable)");
129 return 0; /* unparseable → down */
130 }
131 out->proto = (u.scheme == XRDC_SCHEME_ROOTS) ? "roots" : "root";
132 /* Label on host:port only — never the path (PII-free metric/label contract). */
133 snprintf(out->endpoint, sizeof(out->endpoint), "%s:%d", u.host, u.port);
134
135 t0 = brix_mono_ns();
136 if (brix_connect(&c, &u, &a->conn, &st) != 0) {
137 out->connect_ms = (double) (brix_mono_ns() - t0) / 1e6;
138 return 0; /* down — reported, not an error */
139 }
140 out->up = 1;
141 out->connect_ms = (double) (brix_mono_ns() - t0) / 1e6;
142 if (a->probe_timeout_ms > 0) { c.io.timeout_ms = a->probe_timeout_ms; }
143
144 brix_netdiag_facts(&c, &f);
145 out->tcp_ms = f.tcp_ms;
146 out->tls_ms = f.tls_ms;
147 out->auth_ms = f.auth_ms;
148 {
149 const char *ver = NULL, *cipher = NULL;
150 out->tls_active = (brix_tls_info(&c, &ver, &cipher) == 1);
151 }
152
153 /* tiny read (TTFB) against the biggest regular file under the namespace */
154 brix_status_clear(&st);
155 if (resolve_target(&c, &u, target, sizeof(target), &sti, &st) == 0) {
156 brix_file fh;
157 brix_status rst;
158 brix_status_clear(&rst);
159 t0 = brix_mono_ns();
160 if (brix_file_open_read(&c, target, &fh, &rst) == 0) {
161 char b[4096];
162 ssize_t got = brix_file_read(&c, &fh, 0, b, sizeof(b), &rst);
163 if (got >= 0) {
164 out->read_ms = (double) (brix_mono_ns() - t0) / 1e6;
165 }
166 brix_file_close(&c, &fh, &rst);
167 }
168 {
169 char lb[8192];
170 brix_status lst;
171 uint64_t l0 = brix_mono_ns();
172 brix_status_clear(&lst);
173 if (brix_locate(&c, target, lb, sizeof(lb), &lst) == 0) {
174 out->locate_ms = (double) (brix_mono_ns() - l0) / 1e6;
175 out->holders = watch_count_tokens(lb);
176 }
177 }
178 }
179 brix_close(&c);
180 return 0;
181}

211/* One metric line for every sample, with HELP/TYPE printed once per metric. */
212void
213watch_emit_prom(const watch_sample *samples, int n, FILE *out)
214{
215 int i;
216 char ep[576], pr[64];
217
218 fputs("# HELP brix_probe_up Endpoint reachable (1) or down (0).\n"
219 "# TYPE brix_probe_up gauge\n", out);
220 for (i = 0; i < n; i++) {
221 watch_prom_label(samples[i].endpoint, ep, sizeof(ep));
222 watch_prom_label(samples[i].proto, pr, sizeof(pr));
223 fprintf(out, "brix_probe_up{endpoint=\"%s\",proto=\"%s\"} %d\n",
224 ep, pr, samples[i].up);
225 }
226 fputs("# HELP brix_probe_connect_seconds Full connect (TCP+TLS+auth).\n"
227 "# TYPE brix_probe_connect_seconds gauge\n", out);
228 for (i = 0; i < n; i++) {
229 watch_prom_label(samples[i].endpoint, ep, sizeof(ep));
230 watch_prom_label(samples[i].proto, pr, sizeof(pr));
231 fprintf(out, "brix_probe_connect_seconds{endpoint=\"%s\",proto=\"%s\"} %.6f\n",
232 ep, pr, samples[i].connect_ms / 1000.0);
233 }
234 /* phase split + read/locate only for endpoints that came up */
235 fputs("# HELP brix_probe_read_seconds Tiny-read time-to-first-byte.\n"
236 "# TYPE brix_probe_read_seconds gauge\n", out);
237 for (i = 0; i < n; i++) {
238 if (!samples[i].up || samples[i].read_ms < 0) { continue; }
239 watch_prom_label(samples[i].endpoint, ep, sizeof(ep));
240 watch_prom_label(samples[i].proto, pr, sizeof(pr));
241 fprintf(out, "brix_probe_read_seconds{endpoint=\"%s\",proto=\"%s\"} %.6f\n",
242 ep, pr, samples[i].read_ms / 1000.0);
243 }
244 fputs("# HELP brix_probe_locate_holders Located replica count.\n"
245 "# TYPE brix_probe_locate_holders gauge\n", out);
246 for (i = 0; i < n; i++) {
247 if (!samples[i].up || samples[i].holders < 0) { continue; }
248 watch_prom_label(samples[i].endpoint, ep, sizeof(ep));
249 watch_prom_label(samples[i].proto, pr, sizeof(pr));
250 fprintf(out, "brix_probe_locate_holders{endpoint=\"%s\",proto=\"%s\"} %d\n",
251 ep, pr, samples[i].holders);
252 }

256/* Write the Prometheus exposition to PATH atomically (tmp + rename) — the
257 * node_exporter textfile-collector contract (never expose a half-written file). */
258int
259watch_write_prom_atomic(const char *path, const watch_sample *samples, int n,
260 brix_status *st)
261{
262 char tmp[XRDC_PATH_MAX];
263 FILE *f;
264 int fd;
265
266 if ((size_t) snprintf(tmp, sizeof(tmp), "%s.XXXXXX", path) >= sizeof(tmp)) {
267 brix_status_set(st, XRDC_EUSAGE, 0, "watch: prometheus path too long");
268 return -1;
269 }
270 fd = mkstemp(tmp);
271 if (fd < 0) {
272 brix_status_set(st, XRDC_ESOCK, errno, "watch: mkstemp %s: %s",
273 path, strerror(errno));
274 return -1;
275 }
276 f = fdopen(fd, "w");
277 if (f == NULL) {
278 close(fd);
279 (void) unlink(tmp);
280 brix_status_set(st, XRDC_ESOCK, errno, "watch: fdopen: %s", strerror(errno));
281 return -1;
282 }
283 watch_emit_prom(samples, n, f);
284 if (fclose(f) != 0 || rename(tmp, path) != 0) {
285 (void) unlink(tmp);
286 brix_status_set(st, XRDC_ESOCK, errno, "watch: write %s: %s",
287 path, strerror(errno));
288 return -1;
289 }
290 return 0;
291}

294/* Interruptible sleep: ~200ms granularity so SIGINT stops promptly. */
295void
296watch_sleep(int seconds)
297{
298 int i;
299 for (i = 0; i < seconds * 5 && !g_watch_stop; i++) {
300 struct timespec ts = { 0, 200L * 1000L * 1000L };
301 (void) nanosleep(&ts, NULL);
302 }
303}

306int
307do_watch(const diag_args *a)
308{
309 struct sigaction sa;
310 watch_sample samples[8];
311 int interval = a->interval_s > 0 ? a->interval_s : 10;
312 int cycle = 0, i;
313
314 if (interval > 86400) { interval = 86400; } /* bound watch_sleep's seconds*5 */
315
316 memset(&sa, 0, sizeof(sa));
317 sa.sa_handler = watch_on_signal;
318 sigaction(SIGINT, &sa, NULL);
319 sigaction(SIGTERM, &sa, NULL);
320
321 while (!g_watch_stop) {
322 for (i = 0; i < a->nurls; i++) {
323 watch_probe_once(a, a->urls[i], &samples[i]);
324 }
325 if (a->watch_prom) {
326 if (a->prom_path != NULL) {
327 brix_status wst;
328 brix_status_clear(&wst);
329 if (watch_write_prom_atomic(a->prom_path, samples, a->nurls, &wst) != 0) {
330 fprintf(stderr, "xrddiag: %s\n", wst.msg);
331 }
332 } else {
333 watch_emit_prom(samples, a->nurls, stdout);
334 }
335 } else if (a->json) {
336 for (i = 0; i < a->nurls; i++) { watch_emit_json(&samples[i], stdout); }
337 } else {
338 for (i = 0; i < a->nurls; i++) { watch_emit_human(&samples[i], stdout); }
339 }
340 fflush(stdout);
341
342 cycle++;
343 if (a->count > 0 && cycle >= a->count) { break; }
344 if (g_watch_stop) { break; }
345 watch_sleep(interval);
346 }
347 return 0;
348}
