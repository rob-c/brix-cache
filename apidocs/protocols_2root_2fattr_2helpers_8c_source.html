43ssize_t
44fattr_parse_nvec(ngx_log_t *log, u_char *nvec_copy, size_t buflen,
45 int numattr, brix_fattr_entry_t *attrs)
46{
47 u_char *cursor;
48 int attr_index;
49
50 /*
51 * nvec is repeated as:
52 * [2-byte per-attribute result slot][NUL-terminated attribute name]
53 *
54 * The request payload is copied before parsing, so rc_ptr can safely point
55 * back into nvec_copy. Later operations overwrite only the result slot.
56 */
57/* WHY: nvec is the XRootD request payload for set/del/fattrGet — it carries attribute names with embedded 2-byte result slots that handlers overwrite with per-attribute status codes. Parsing into attr[] entries enables subsequent operations to read names, compute xkey prefixes, and write rc values at fixed positions. */
58/* HOW: Initializes cursor = nvec_copy, end = nvec_copy + buflen. Loop over numattr entries: checks cursor+2 <= end (truncation guard), sets attrs[i].rc_ptr=cursor, errcode=0, value=NULL, vlen=0; advances cursor by 2. Scans for NUL terminator to find name length — if cursor >= end logs "name not null-terminated" and returns -1. Validates name_len > 0 && <= kXR_faMaxNlen — otherwise logs invalid length and returns -1. Sets attrs[i].name=name_start, nlen=name_len; snprintf xkey with BRIX_FATTR_XKEY_PFX prefix + %.*s format. Advances cursor past NUL. Returns (ssize_t)(cursor - nvec_copy) = bytes consumed. */
59 cursor = nvec_copy;
60
61 for (attr_index = 0; attr_index < numattr; attr_index++) {
62 const uint8_t *name_p;
63 size_t name_len, off, next;
64
65 /* Shared read-only scan of one [int16 rc][name\0] entry (libxrdproto).
66 * The rc slot lives at cursor; we hold that pointer ourselves so the
67 * set/del handlers can rewrite it in place (fattr_set_rc) — the shared
68 * parser never touches the buffer. */
69 off = (size_t) (cursor - nvec_copy);
70 if (xrdp_fattr_nvec_parse(nvec_copy, buflen, off, NULL,
71 &name_p, &name_len, &next) != 0) {
72 ngx_log_error(NGX_LOG_WARN, log, 0,
73 "brix: fattr nvec truncated/unterminated at entry %d",
74 attr_index);
75 return -1;
76 }
77
78 /* Name-length policy is server-side (the wire parser is policy-free). */
79 if (name_len == 0 || name_len > kXR_faMaxNlen) {
80 ngx_log_error(NGX_LOG_WARN, log, 0,
81 "brix: fattr name length %uz invalid",
82 name_len);
83 return -1;
84 }
85
86 attrs[attr_index].rc_ptr = cursor; /* the 2-byte rc slot */
87 attrs[attr_index].errcode = 0;
88 attrs[attr_index].value = NULL;
89 attrs[attr_index].vlen = 0;
90 attrs[attr_index].name = (char *) name_p;
91 attrs[attr_index].nlen = name_len;
92 snprintf(attrs[attr_index].xkey, sizeof(attrs[attr_index].xkey),
93 BRIX_FATTR_XKEY_PFX "%.*s", (int) name_len, name_p);
94
95 cursor = nvec_copy + next;
96 }
97
98 return (ssize_t) (cursor - nvec_copy);
99}
