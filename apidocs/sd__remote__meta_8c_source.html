14/* ---- xattr surface: x-amz-meta-* as the `user.` namespace --------------
15 *
16 * getxattr("user.<name>") reads x-amz-meta-<name> via a signed HEAD;
17 * listxattr enumerates every x-amz-meta-* header (needs a transport with the
18 * optional resp_headers_raw slot — without it sd_s3_list_meta reports
19 * ENOTSUP). Both open the object read-only just for the HEAD, mirroring the
20 * stat body below. */
21static sd_s3_file *
22sd_remote_meta_open(brix_sd_instance_t *inst, const char *path)
23{
24 const brix_sd_remote_cfg_t *cfg = inst->state;
25 sd_s3_open_params p;
26 char objpath[768];
27 char errbuf[256];
28
29 sd_remote_s3_key(cfg, path, objpath, sizeof(objpath));
30 sd_remote_s3_params(cfg, objpath, &p);
31 return sd_s3_open_read(&p, errbuf, sizeof(errbuf));
32}

34ssize_t
35sd_remote_getxattr(brix_sd_instance_t *inst, const char *path,
36 const char *name, void *buf, size_t cap)
37{
38 char val[2048];
39 char errbuf[256];
40 sd_s3_meta_buf dst = { val, sizeof(val) };
41 sd_s3_file *s3;
42 ssize_t n;
43
44 if (strncmp(name, "user.", 5) != 0 || name[5] == '\0') {
45 errno = ENODATA; /* only the user. namespace maps to x-amz-meta- */
46 return -1;
47 }
48 s3 = sd_remote_meta_open(inst, path);
49 if (s3 == NULL) {
50 errno = ENOMEM;
51 return -1;
52 }
53 errno = 0;
54 n = sd_s3_get_meta(s3, name + 5, &dst, errbuf, sizeof(errbuf));
55 sd_s3_close(s3);
56 if (n < 0) {
57 if (errno == 0) { errno = EIO; }
58 return -1;
59 }
60 if (n == 0) {
61 errno = ENODATA; /* HEAD ok, attribute absent */
62 return -1;
63 }
64 if (buf == NULL || cap == 0) {
65 return n; /* getxattr(2) size probe */
66 }
67 if ((size_t) n > cap) {
68 errno = ERANGE;
69 return -1;
70 }
71 memcpy(buf, val, (size_t) n);
72 return n;
73}

97/* Shared stat body: HEAD the object, optionally signing with a per-user
98 * ak/sk/region override (NULL = the instance's static service credential). */
99static ngx_int_t
100sd_remote_stat_impl(brix_sd_instance_t *inst, const char *path,
101 brix_sd_stat_t *out, const char *ak, const char *sk, const char *region)
102{
103 const brix_sd_remote_cfg_t *cfg = inst->state;
104 sd_s3_open_params p;
105 char objpath[768];
106 char errbuf[256];
107 sd_s3_file *s3;
108 int64_t size = 0;
109
110 sd_remote_s3_key(cfg, path, objpath, sizeof(objpath));
111 sd_remote_s3_params(cfg, objpath, &p);
112 if (ak != NULL) { p.ak = ak; }
113 if (sk != NULL) { p.sk = sk; }
114 if (region != NULL) { p.region = region; }
115
116 s3 = sd_s3_open_read(&p, errbuf, sizeof(errbuf));
117 if (s3 == NULL) {
118 errno = ENOENT;
119 return NGX_ERROR;
120 }
121 if (sd_s3_size(s3, &size, errbuf, sizeof(errbuf)) != 0) {
122 sd_s3_close(s3);
123 errno = EIO;
124 return NGX_ERROR;
125 }
126 sd_s3_close(s3);
127
128 memset(out, 0, sizeof(*out));
129 out->size = (off_t) size;
130 out->mode = S_IFREG | 0444;
131 out->is_reg = 1;
132 return NGX_OK;
133}

142/* Cred-scoped stat (P80.3): the probe/HEAD runs as the requesting user, so a
143 * deny-mode request never reaches the origin under the service credential.
144 * Registering this slot is also the canonical capability gate that turns on
145 * per-user namespace credential resolution in brix_vfs_ns_cred(). */
146ngx_int_t
147sd_remote_stat_cred(brix_sd_instance_t *inst, const char *path,
148 brix_sd_stat_t *out, const brix_sd_cred_t *cred)
149{
150 int gate = sd_remote_cred_gate(cred);
151
152 if (gate > 0) {
153 return sd_remote_stat_impl(inst, path, out,
154 cred->s3_ak, cred->s3_sk, cred->s3_region);
155 }
156 if (gate < 0) {
157 errno = EACCES;
158 return NGX_ERROR;
159 }
160 return sd_remote_stat_impl(inst, path, out, NULL, NULL, NULL);
161}
