2 * path.c — client-side path helpers shared by the front-end tools.
3 *
4 * brix_path_resolve() canonicalises a (possibly relative) argument against a
5 * current working directory, collapsing ".", "..", and duplicate slashes to an
6 * absolute server path — the logic xrdfs's interactive shell and one-shot
7 * subcommands used (build_path) and that xrdcp open-codes for its destinations.
8 */

18void
19brix_path_resolve(const char *cwd, const char *arg, char *out, size_t outsz)
20{
21 char raw[XRDC_PATH_MAX * 2];
22 char *comps[256];
23 int nc = 0;
24 char *tok, *save;
25 size_t pos;
26 int i;
27
28 if (arg[0] == '/') {
29 snprintf(raw, sizeof(raw), "%s", arg);
30 } else {
31 snprintf(raw, sizeof(raw), "%s/%s",
32 (cwd[0] == '/' && cwd[1] == '\0') ? "" : cwd, arg);
33 }
34
35 for (tok = strtok_r(raw, "/", &save); tok != NULL && nc < 256;
36 tok = strtok_r(NULL, "/", &save)) {
37 if (strcmp(tok, ".") == 0) {
38 continue;
39 }
40 if (strcmp(tok, "..") == 0) {
41 if (nc > 0) { nc--; }
42 continue;
43 }
44 comps[nc++] = tok;
45 }
46
47 if (nc == 0) {
48 snprintf(out, outsz, "/");
49 return;
50 }
51 pos = 0;
52 for (i = 0; i < nc && pos + 1 < outsz; i++) {
53 int w = snprintf(out + pos, outsz - pos, "/%s", comps[i]);
54 if (w < 0) { break; }
55 pos += (size_t) w;
56 }
57}

59/*
60 * brix_open_credfile — open a credential file for reading, refusing anything an
61 * attacker could have planted or tampered with.
62 *
63 * WHAT: open `path` read-only and return a validated fd (caller closes), or -1.
64 * WHY: credentials default to predictable, world-writable locations
65 * (/tmp/bt_u<uid>, /tmp/x509up_u<uid>). A plain fopen/BIO_new_file there
66 * follows a symlink an attacker pre-planted under the victim's uid — leaking
67 * a secret the client then transmits — or reads an attacker-owned regular
68 * file, authenticating the victim as the attacker (confused deputy). This is
69 * the same hardening brix_sss_keytab_read already applies to keytabs.
70 * HOW: O_NOFOLLOW (no final-component symlink) | O_CLOEXEC; require a regular
71 * file owned by the effective uid; reject group/other WRITE always, and —
72 * for `secret` files (private keys / GSI proxies) — group/other READ too.
73 * `st` may be NULL for silent probing (returns -1 without setting an error).
74 */
75int
76brix_open_credfile(const char *path, int secret, brix_status *st)
77{
78 struct stat sb;
79 mode_t bad;
80 int fd;
81
82 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
83 if (fd < 0) {
84 brix_status_set(st, XRDC_EAUTH, errno, "open %s: %s",
85 path, strerror(errno));
86 return -1;
87 }
88 if (fstat(fd, &sb) != 0 || !S_ISREG(sb.st_mode)) {
89 close(fd);
90 brix_status_set(st, XRDC_EAUTH, 0, "%s is not a regular file", path);
91 return -1;
92 }
93 if (sb.st_uid != geteuid()) {
94 close(fd);
95 brix_status_set(st, XRDC_EAUTH, 0,
96 "%s not owned by uid %u (refusing untrusted credential)",
97 path, (unsigned) geteuid());
98 return -1;
99 }
100 bad = secret ? (S_IRWXG | S_IRWXO) : (S_IWGRP | S_IWOTH);
101 if (sb.st_mode & bad) {
102 close(fd);
103 brix_status_set(st, XRDC_EAUTH, 0,
104 "%s has unsafe permissions (must be %s)",
105 path, secret ? "0600" : "non-writable by group/other");
106 return -1;
107 }
108 return fd;
109}

111/* WHAT: 1 if a relative path would escape the directory it is joined under.
112 * WHY: server dirlists, manifests, and archives are untrusted inputs; a
113 * "../" component or absolute path must never place files outside the
114 * destination root (moved here from apps/copy so every consumer shares
115 * one guard). NULL is treated as unsafe (fail-closed): a caller that
116 * cannot produce a path has no safe join target.
117 * HOW: reject NULL (fail-closed), absolute paths, and any exact ".."
118 * component. Empty string is safe (degenerate, but not escaping). */
119int
120brix_rel_is_unsafe(const char *rel)
121{
122 size_t len;
123
124 if (rel == NULL) {
125 return 1; /* fail-closed: no path at all is not a safe path */
126 }
127 if (rel[0] == '/') {
128 return 1;
129 }
130 len = strlen(rel);
131 if (strcmp(rel, "..") == 0
132 || strncmp(rel, "../", 3) == 0
133 || strstr(rel, "/../") != NULL
134 || (len >= 3 && strcmp(rel + len - 3, "/..") == 0)) {
135 return 1;
136 }
137 return 0;
138}
