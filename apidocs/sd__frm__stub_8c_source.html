29/* The recall-in-flight marker for an async (delayed) stub recall:
30 * <base>/.recalling/<key>. Its mtime records when the recall began; recall_poll
31 * completes the copy once recall_delay_ms has elapsed. */
32static int
33stub_marker_path(const stub_ctx_t *c, const char *key, char *out, size_t cap)
34{
35 int n = snprintf(out, cap, "%s/.recalling/%s", c->base,
36 (key[0] == '/') ? key + 1 : key);
37
38 return (n > 0 && (size_t) n < cap) ? 0 : -1;
39}

59/* mkdir -p the parent directories of `path`. */
60void
61frm_mkparents(const char *path)
62{
63 char tmp[PATH_MAX];
64 size_t i;
65 size_t n = strlen(path);
66
67 if (n == 0 || n >= sizeof(tmp)) {
68 return;
69 }
70 memcpy(tmp, path, n + 1);
71 for (i = 1; i < n; i++) {
72 if (tmp[i] == '/') {
73 tmp[i] = '\0';
74 (void) mkdir(tmp, 0755); /* EEXIST ok */
75 tmp[i] = '/';
76 }
77 }
78}

80/* Copy src -> dst (creating dst's parents), preserving mode. 0 / -1. */
81int
82stub_copyfile(const char *src, const char *dst, mode_t mode)
83{
84 char buf[1u << 16];
85 int in;
86 int out;
87 ssize_t r;
88
89 in = open(src, O_RDONLY | O_CLOEXEC);
90 if (in < 0) {
91 return -1;
92 }
93 frm_mkparents(dst);
94 out = open(dst, O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC, mode ? mode : 0644);
95 if (out < 0) {
96 (void) close(in);
97 return -1;
98 }
99 while ((r = read(in, buf, sizeof(buf))) > 0) {
100 ssize_t off = 0;
101
102 while (off < r) {
103 ssize_t w = write(out, buf + off, (size_t) (r - off));
104
105 if (w < 0) {
106 if (errno == EINTR) { continue; }
107 (void) close(in);
108 (void) close(out);
109 return -1;
110 }
111 off += w;
112 }
113 }
114 (void) close(in);
115 if (close(out) != 0 || r < 0) {
116 return -1;
117 }
118 return 0;
119}

121static int
122stub_residency(void *mss, const char *key, off_t *size_out, time_t *mtime_out)
123{
124 stub_ctx_t *c = mss;
125 char path[PATH_MAX];
126 struct stat sb;
127
128 if (stub_path(c, key, 1 /* online */, path, sizeof(path)) == 0
129 && stat(path, &sb) == 0)
130 {
131 if (size_out) { *size_out = sb.st_size; }
132 if (mtime_out) { *mtime_out = sb.st_mtime; }
133 return BRIX_RESIDENCY_ONLINE;
134 }
135 if (stub_path(c, key, 0 /* tape */, path, sizeof(path)) == 0
136 && stat(path, &sb) == 0)
137 {
138 if (size_out) { *size_out = sb.st_size; }
139 if (mtime_out) { *mtime_out = sb.st_mtime; }
140 return BRIX_RESIDENCY_OFFLINE;
141 }
142 return BRIX_RESIDENCY_ABSENT;
143}

145static int
146stub_recall_begin(void *mss, const char *key)
147{
148 stub_ctx_t *c = mss;
149 char tape[PATH_MAX];
150 char online[PATH_MAX];
151 char marker[PATH_MAX];
152 struct stat sb;
153
154 if (stub_path(c, key, 0, tape, sizeof(tape)) != 0
155 || stub_path(c, key, 1, online, sizeof(online)) != 0
156 || stat(tape, &sb) != 0)
157 {
158 return -1;
159 }
160 if (access(online, F_OK) == 0) {
161 return 0; /* already online */
162 }
163 if (c->recall_delay_ms <= 0) {
164 /* synchronous recall: a local copy tape -> online buffer (the default). */
165 return stub_copyfile(tape, online, sb.st_mode & 0777);
166 }
167 /* Async: drop a recall marker (its mtime records the start). recall_poll
168 * completes the copy once recall_delay_ms has elapsed - simulating MSS
169 * latency so the cache tier parks the open (SP5 §9.2). Idempotent. */
170 if (stub_marker_path(c, key, marker, sizeof(marker)) != 0) {
171 return -1;
172 }
173 if (access(marker, F_OK) != 0) {
174 int fd;
175
176 frm_mkparents(marker);
177 fd = open(marker, O_WRONLY | O_CREAT | O_CLOEXEC, 0644);
178 if (fd >= 0) {
179 (void) close(fd);
180 }
181 }
182 return 0; /* recall in flight */
183}

185static int
186stub_recall_poll(void *mss, const char *key)
187{
188 stub_ctx_t *c = mss;
189 char online[PATH_MAX];
190 char marker[PATH_MAX];
191 char tape[PATH_MAX];
192 struct stat sb;
193
194 if (stub_path(c, key, 1, online, sizeof(online)) != 0) {
195 return -1;
196 }
197 if (access(online, F_OK) == 0) {
198 return 1; /* online */
199 }
200 if (c->recall_delay_ms <= 0) {
201 return 0; /* sync adapter: begin copies; not begun */
202 }
203 if (stub_marker_path(c, key, marker, sizeof(marker)) != 0
204 || stat(marker, &sb) != 0)
205 {
206 return 0; /* no recall in flight */
207 }
208 if ((time(NULL) - sb.st_mtime) * 1000 < (time_t) c->recall_delay_ms) {
209 return 0; /* still staging */
210 }
211 /* Delay elapsed: complete the recall (copy tape -> online), drop the marker. */
212 if (stub_path(c, key, 0, tape, sizeof(tape)) != 0 || stat(tape, &sb) != 0) {
213 return -1;
214 }
215 if (stub_copyfile(tape, online, sb.st_mode & 0777) != 0) {
216 return -1;
217 }
218 (void) unlink(marker);
219 return 1;
220}
