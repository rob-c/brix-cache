1/*
2 * sd_ceph.c — Ceph/RADOS Storage Driver (phase-60, basic librados backend).
3 *
4 * WHAT: brix_sd_ceph_driver — a backend that maps the VFS's logical paths onto
5 * flat RADOS objects via raw librados (rados_read/write/trunc/stat/remove).
6 * Two layers live here:
7 * 1. The pure LFN->object-key map (sd_ceph_normalize/_key/_ino) — libc
8 * only, always compiled, unit-tested standalone (sd_ceph_unittest.c).
9 * 2. The driver vtable — only when the build found librados
10 * (BRIX_HAVE_CEPH); otherwise this file is just the pure helpers and
11 * the build is byte-for-byte unchanged (the driver row in
12 * sd_registry.c is #if-guarded too).
13 *
14 * The driver body has been split (source-size guard) across four TUs:
15 * sd_ceph.c keeps the pure helpers, the instance lifecycle (init/cleanup +
16 * the cluster-connect primitive) and the driver descriptor; sd_ceph_io.c
17 * holds the raw byte ops, staged write and the shared connection layer;
18 * sd_ceph_object.c the object open/close/stat/unlink + xattr slots; and
19 * sd_ceph_cred.c the per-user cred-conn cache, cred-scoped open, the
20 * oid-keyed byte/xattr layer and catalog enumeration. The shared struct
21 * definitions and cross-TU declarations live in sd_ceph_internal.h.
22 *
23 * WHY: RADOS has no kernel fd, no sendfile, no directory tree and no atomic
24 * rename, so this driver advertises only range-read / random-write /
25 * truncate (see .caps). The VFS already serves a no-CAP_FD backend memory-
26 * backed and degrades the absent namespace/rename/xattr ops — the data
27 * plane (root:// read/write, WebDAV/S3 GET/PUT) rides the same VFS seam as
28 * POSIX once the handle path is de-fd'd (phase-60 W0).
29 *
30 * HOW: One rados_t + ioctx per export instance, connected at init() on the event
31 * loop (worker init); the blocking rados_* calls are meant to run on the
32 * nginx thread pool (phase-60 §8 / ADR-4). Object handles carry the object
33 * id + a cached size; the raw byte ops are worker-safe (no pool/log/metrics).
34 * libradosstriper (large-object striping + stock XrdCeph on-disk interop,
35 * ADR-3) is a deliberate follow-on; this basic backend uses raw librados.
36 * Per-user credential scoping (ceph-peruser item): the driver also
37 * implements .open_cred (sd_ceph_open_cred), which authenticates to
38 * RADOS as a specific CephX user (parsed from a <key>.keyring file by
39 * fs/backend/ucred.c) instead of the export's static service
40 * credential. A bounded per-(user,keyring) connection-cache LRU on the
41 * instance state amortizes the rados_connect cost across repeated
42 * opens by the same user; every raw byte op (pread/pwrite/ftruncate/
43 * fstat) is keyed off the OPEN object's own ioctx (sd_ceph_obj_state_t.
44 * ioctx), not the export's, so a cred-scoped open stays scoped to that
45 * user for its whole lifetime.
46 *
47 * Cred-conn lifetime (pin/refcount, fixes a UAF): a cred-scoped
48 * sd_ceph_conn_t is reference-counted (sd_ceph_conn_t.refs), pinned by
49 * every open object that resolved onto it (sd_ceph_obj_state_t.conn)
50 * and released in sd_ceph_close. The bounded LRU never destroys a
51 * pinned (refs>0) connection: eviction skips pinned slots, and if a
52 * pinned slot is chosen to make room in the cache table it is marked
53 * `doomed` and removed from the table WITHOUT destroying the
54 * connection — the connection is destroyed by whichever sd_ceph_close
55 * drops its refcount to zero. If every cache slot is pinned, a fresh
56 * *uncached* connection is created for that one open (never inserted
57 * into the table, pinned for the object's lifetime, destroyed on
58 * close) so a legitimate concurrent identity beyond the cache bound
59 * still works instead of failing the open.
60 */

71/* sd_ceph_seg_t — one path segment carved out of an LFN during normalization:
72 * the pointer into the source string and its length (not NUL-terminated). */
73typedef struct {
74 const char *start;
75 size_t len;
76} sd_ceph_seg_t;

78/* sd_ceph_next_seg — advance a cursor to the next non-empty path segment.
79 *
80 * WHAT: From `*cursor`, skip the run of '/' separators, then capture the next
81 * run of non-'/' bytes into `*seg`. Returns 1 when a segment was found
82 * (and advances `*cursor` past it), or 0 at end of string.
83 *
84 * WHY: Isolates the pointer-walking tokenizer from the segment-classification
85 * logic in sd_ceph_normalize, so that function reads as a flat loop over
86 * already-carved segments instead of interleaving scan and decision.
87 *
88 * HOW: 1. Skip leading '/' separators. 2. If at end, report no segment.
89 * 3. Mark the segment start, walk to the next '/' or NUL, record the
90 * length, and leave the cursor at the delimiter. */
91static int
92sd_ceph_next_seg(const char **cursor, sd_ceph_seg_t *seg)
93{
94 const char *p = *cursor;
95
96 while (*p == '/') { /* skip run of slashes */
97 p++;
98 }
99 if (*p == '\0') {
100 *cursor = p;
101 return 0;
102 }
103
104 seg->start = p;
105 while (*p != '\0' && *p != '/') {
106 p++;
107 }
108 seg->len = (size_t) (p - seg->start);
109 *cursor = p;
110 return 1;
111}

113/* sd_ceph_norm_pop — remove the last component from the canonical buffer for a
114 * ".." segment. Returns the new write cursor, or (size_t)-1 for an escape above
115 * the root (nothing to pop) so the caller can fail the normalization. */
116static size_t
117sd_ceph_norm_pop(char *out, size_t w)
118{
119 if (w == 0) {
120 return (size_t) -1; /* escape above the root */
121 }
122 while (w > 0 && out[w - 1] != '/') {
123 w--; /* back over the last component */
124 }
125 if (w > 0) {
126 w--; /* drop its leading '/' */
127 }
128 out[w] = '\0';
129 return w;
130}
131
132/* sd_ceph_norm_append — append one ordinary segment (as "/<seg>") to the
133 * canonical buffer at write cursor `w`. Returns the new cursor, or (size_t)-1
134 * when the segment would not fit within `cap` (caller sets ENAMETOOLONG). */
135static size_t
136sd_ceph_norm_append(char *out, size_t cap, size_t w, const sd_ceph_seg_t *seg)
137{
138 if (w + 1 + seg->len + 1 > cap) {
139 return (size_t) -1;
140 }
141 out[w++] = '/';
142 memcpy(out + w, seg->start, seg->len);
143 w += seg->len;
144 out[w] = '\0';
145 return w;
146}

148/* sd_ceph_seg_is_dot / sd_ceph_seg_is_dotdot — classify a carved segment as the
149 * "." (self) or ".." (parent) special components. */
150static int
151sd_ceph_seg_is_dot(const sd_ceph_seg_t *seg)
152{
153 return seg->len == 1 && seg->start[0] == '.';
154}
155
156static int
157sd_ceph_seg_is_dotdot(const sd_ceph_seg_t *seg)
158{
159 return seg->len == 2 && seg->start[0] == '.' && seg->start[1] == '.';
160}

162/* sd_ceph_normalize — see sd_ceph.h. Builds the canonical path in `out` by
163 * walking segments: skip empties and ".", pop on "..", append otherwise. A ".."
164 * with nothing to pop is an escape above the root and is rejected. */
165int
166sd_ceph_normalize(const char *lfn, char *out, size_t cap)
167{
168 const char *cursor = lfn;
169 sd_ceph_seg_t seg = {0};
170 size_t w = 0;
171
172 if (lfn == NULL || out == NULL || cap < 2) {
173 errno = EINVAL;
174 return -1;
175 }
176 out[0] = '\0';
177
178 while (sd_ceph_next_seg(&cursor, &seg)) {
179 if (sd_ceph_seg_is_dot(&seg)) {
180 continue; /* "." — no-op */
181 }
182 if (sd_ceph_seg_is_dotdot(&seg)) {
183 w = sd_ceph_norm_pop(out, w);
184 if (w == (size_t) -1) {
185 errno = EINVAL; /* escape above the root */
186 return -1;
187 }
188 continue;
189 }
190 w = sd_ceph_norm_append(out, cap, w, &seg);
191 if (w == (size_t) -1) {
192 errno = ENAMETOOLONG;
193 return -1;
194 }
195 }
196
197 if (w == 0) { /* everything collapsed -> bare root */
198 out[0] = '/';
199 out[1] = '\0';
200 }
201 return 0;
202}

204/* sd_ceph_key — prefix + sd_ceph_normalize(lfn). See sd_ceph.h. */
205int
206sd_ceph_key(const char *key_prefix, const char *lfn, char *out, size_t cap)
207{
208 char norm[1024];
209 size_t plen = (key_prefix != NULL) ? strlen(key_prefix) : 0;
210 size_t nlen;
211
212 if (sd_ceph_normalize(lfn, norm, sizeof(norm)) != 0) {
213 return -1;
214 }
215 nlen = strlen(norm);
216
217 if (plen + nlen + 1 > cap) {
218 errno = ENAMETOOLONG;
219 return -1;
220 }
221 if (plen > 0) {
222 /* phase74-fp: prefix copy is intentionally unterminated — the next
223 * memcpy appends norm with its NUL (nlen + 1), and the cap check above
224 * guarantees room for plen + nlen + 1. */
225 memcpy(out, key_prefix, plen); /* NOLINT(bugprone-not-null-terminated-result) */
226 }
227 memcpy(out + plen, norm, nlen + 1);
228 return 0;
229}

231/* sd_ceph_ino — FNV-1a/64 over the object id. See sd_ceph.h. */
232uint64_t
233sd_ceph_ino(const char *oid)
234{
235 const unsigned char *p = (const unsigned char *) oid;
236 uint64_t h = 1469598103934665603ULL; /* FNV offset basis */
237
238 while (*p != '\0') {
239 h ^= (uint64_t) *p++;
240 h *= 1099511628211ULL; /* FNV prime */
241 }
242 return h;
243}

301/* sd_ceph_cluster_connect — create + configure + connect a rados cluster handle
302 * and open the pool ioctx. Shared by the flat driver's init and the oid-level
303 * connection (sd_ceph_conn_create). 0 / -1 with errno; on failure nothing leaks.
304 * Non-static: sd_ceph_conn_create lives in sd_ceph_io.c (sd_ceph_internal.h). */
305int
306sd_ceph_cluster_connect(const char *conf_file, const char *user,
307 const char *keyring, const char *pool,
308 rados_t *cluster_out, rados_ioctx_t *ioctx_out)
309{
310 rados_t cluster;
311 rados_ioctx_t ioctx;
312
313 if (sd_ceph_set_errno(rados_create(&cluster, sd_ceph_user_id(user)))) {
314 return -1;
315 }
316 if (sd_ceph_set_errno(rados_conf_read_file(cluster,
317 conf_file ? conf_file : "/etc/ceph/ceph.conf")))
318 {
319 rados_shutdown(cluster);
320 return -1;
321 }
322 if (keyring != NULL) {
323 rados_conf_set(cluster, "keyring", keyring);
324 }
325 if (sd_ceph_set_errno(rados_connect(cluster))) {
326 rados_shutdown(cluster);
327 return -1;
328 }
329 if (sd_ceph_set_errno(rados_ioctx_create(cluster, pool, &ioctx))) {
330 rados_shutdown(cluster);
331 return -1;
332 }
333 *cluster_out = cluster;
334 *ioctx_out = ioctx;
335 return 0;
336}

338/* sd_ceph_init — resolve config onto the pool, create + configure + connect the
339 * cluster handle, and open the pool ioctx. Any failure tears down what was built
340 * and returns NGX_ERROR with errno set so the export fails closed at init. */
341static ngx_int_t
342sd_ceph_init(brix_sd_instance_t *inst, void *driver_conf)
343{
344 brix_sd_ceph_conf_t *dc = driver_conf;
345 sd_ceph_state_t *st;
346
347 if (dc == NULL || dc->pool == NULL) {
348 errno = EINVAL;
349 return NGX_ERROR;
350 }
351
352 st = ngx_pcalloc(inst->pool, sizeof(*st));
353 if (st == NULL) {
354 errno = ENOMEM;
355 return NGX_ERROR;
356 }
357 st->pool = sd_ceph_pstrdup(inst->pool, dc->pool);
358 st->user = sd_ceph_pstrdup(inst->pool, dc->user);
359 st->conf_file = sd_ceph_pstrdup(inst->pool,
360 dc->conf_file ? dc->conf_file : "/etc/ceph/ceph.conf");
361 st->keyring = sd_ceph_pstrdup(inst->pool, dc->keyring);
362 st->key_prefix = sd_ceph_pstrdup(inst->pool,
363 dc->key_prefix ? dc->key_prefix : "");
364 inst->state = st;
365
366 if (sd_ceph_cluster_connect(st->conf_file, st->user, st->keyring, st->pool,
367 &st->cluster, &st->ioctx) != 0)
368 {
369 return NGX_ERROR;
370 }
371 st->connected = 1;
372 return NGX_OK;
373}

375/* sd_ceph_cleanup — destroy the ioctx and shut the cluster handle down (a kernel/
376 * network resource that must not leak across reconfig); the pool reclaims state.
377 * Also destroys every cached per-user cred connection (ceph-peruser item) — each
378 * holds its own rados_t/ioctx that would otherwise leak across a reconfig/reload
379 * just like the export's own cluster handle would.
380 *
381 * A connection in the cache table at cleanup time is expected to have
382 * refs==0: instance teardown only runs once every request/handle on this
383 * export has drained (the VFS/protocol layers close all handles before an
384 * instance is torn down), so nothing should still be pinning a cached
385 * connection here. If that invariant were ever violated (an object somehow
386 * outliving its instance), destroying it anyway is still the right call —
387 * the instance and its pool are going away regardless, so there is no
388 * "later close()" left to do the deferred destroy — but it is flagged with
389 * a WARN so the anomaly is visible rather than silently masked. */
390static void
391sd_ceph_cleanup(brix_sd_instance_t *inst)
392{
393 sd_ceph_state_t *st = inst->state;
394 ngx_uint_t i;
395
396 if (st == NULL) {
397 return;
398 }
399
400 for (i = 0; i < st->cred_conn_count; i++) {
401 if (st->cred_conns[i].conn != NULL) {
402#if !defined(XRDPROTO_NO_NGX)
403 /* ngx_log_error/NGX_LOG_WARN are unavailable in the ngx-free
404 * standalone build (tests/ceph/'s live-test drivers link this
405 * file directly with -DXRDPROTO_NO_NGX); the anomaly this warns
406 * about is a should-never-happen invariant violation, not
407 * something the standalone driver tests need to observe. */
408 if (st->cred_conns[i].conn->refs > 0 && inst->log != NULL) {
409 ngx_log_error(NGX_LOG_WARN, inst->log, 0,
410 "sd_ceph_cleanup: cred connection for user \"%s\" still "
411 "had %ui open handle(s) pinned at instance teardown "
412 "(destroying anyway)",
413 st->cred_conns[i].user,
414 (ngx_uint_t) st->cred_conns[i].conn->refs);
415 }
416#endif
417 sd_ceph_conn_destroy(st->cred_conns[i].conn);
418 st->cred_conns[i].conn = NULL;
419 }
420 }
421 st->cred_conn_count = 0;
422
423 if (st->connected) {
424#if defined(BRIX_HAVE_RADOSSTRIPER)
425 if (st->striper_ready) {
426 sd_ceph_striper_destroy(st->striper);
427 st->striper_ready = 0;
428 }
429#endif
430 rados_ioctx_destroy(st->ioctx);
431 rados_shutdown(st->cluster);
432 st->connected = 0;
433 }
434}
