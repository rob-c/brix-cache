1/*
2 * sd_cache_manifest.c — CVMFS manifest/whitelist signature verify on fill
3 * (phase-85 F1: the verifying proxy).
4 *
5 * WHAT: Before a MANIFEST-class staged fill commits, verify the full CVMFS
6 * trust chain against the operator-configured repo master public key:
7 * .cvmfswhitelist — signature vs master key + expiry;
8 * .cvmfspublished — whitelist sig vs master → whitelist not expired →
9 * manifest cert fingerprint ∈ whitelist → manifest
10 * sig vs cert → repo-name cross-check.
11 * A failed chain never publishes: the caller quarantines the part,
12 * aborts the fill, emits signal=cvmfs_tamper and reports EBADMSG.
13 *
14 * WHY: cvmfs-cas already makes every CAS object self-certifying at the edge,
15 * but the CAS *names* flow from the manifest — an origin (or MITM) that
16 * forges .cvmfspublished redirects every client to attacker-named
17 * content. Verifying the signature chain ONCE here protects every
18 * downstream client of this proxy, exactly as the FUSE client's mount
19 * path (shared/cvmfs/client/client.c load_trust_and_catalog) protects
20 * one mount. The verify primitives are the same shared pure-C code.
21 *
22 * HOW: Runs on the blocking fill THREAD (like the cvmfs-cas verify), so the
23 * sibling fetches (.cvmfswhitelist, certificate CAS object) go straight
24 * through the source tier with plain blocking preads. All buffers are
25 * malloc'd (fill threads carry small stacks). The staged part is read
26 * from its posix staged_path — enforced at config time.
27 */

44/* Signed-metadata artifacts are small (manifest ~400B, whitelist ~1KB, cert
45 * ~2KB); 64KB matches the FUSE client's raw_fetch bound and caps a hostile
46 * origin's memory cost. */
47#define SD_CACHE_META_MAX 65536

49/* Read the whole staged part at `pp` into a malloc'd buffer (cap
50 * SD_CACHE_META_MAX). Returns the byte count, or -1 (oversize/unreadable). */
51static ssize_t
52meta_read_part(const char *pp, unsigned char **out)
53{
54 unsigned char *buf;
55 ssize_t n, off = 0;
56 int fd;
57
58 fd = open(pp, O_RDONLY | O_CLOEXEC);
59 if (fd == -1) {
60 return -1;
61 }
62 buf = malloc(SD_CACHE_META_MAX);
63 if (buf == NULL) {
64 (void) close(fd);
65 return -1;
66 }
67 for ( ;; ) {
68 n = read(fd, buf + off, (size_t) (SD_CACHE_META_MAX - off));
69 if (n < 0) {
70 if (errno == EINTR) {
71 continue;
72 }
73 break;
74 }
75 if (n == 0) { /* clean EOF within the cap */
76 (void) close(fd);
77 *out = buf;
78 return off;
79 }
80 off += n;
81 if (off >= SD_CACHE_META_MAX) { /* oversize for signed meta */
82 break;
83 }
84 }
85 (void) close(fd);
86 free(buf);
87 return -1;
88}

90/* Fetch `key` COMPLETE from the source tier into a malloc'd buffer (cap
91 * SD_CACHE_META_MAX). Blocking — fill-thread only. Returns the byte count,
92 * or -1 on any failure (open, read, oversize). */
93static ssize_t
94meta_fetch_source(sd_cache_inst_state *st, const char *key,
95 unsigned char **out)
96{
97 brix_sd_obj_t *so;
98 unsigned char *buf;
99 ssize_t n, off = 0;
100 int err = 0;
101
102 if (st->source == NULL || st->source->driver->open == NULL) {
103 return -1;
104 }
105 so = brix_sd_open_maybe_cred(st->source, key, BRIX_SD_O_READ, 0, NULL,
106 &err);
107 if (so == NULL || so->driver->pread == NULL) {
108 if (so != NULL) {
109 brix_sd_obj_release(so);
110 }
111 return -1;
112 }
113 buf = malloc(SD_CACHE_META_MAX);
114 if (buf == NULL) {
115 brix_sd_obj_release(so);
116 return -1;
117 }
118 for ( ;; ) {
119 n = so->driver->pread(so, buf + off,
120 (size_t) (SD_CACHE_META_MAX - off), off);
121 if (n < 0) {
122 if (errno == EINTR) {
123 continue;
124 }
125 break;
126 }
127 if (n == 0) {
128 brix_sd_obj_release(so);
129 *out = buf;
130 return off;
131 }
132 off += n;
133 if (off >= SD_CACHE_META_MAX) {
134 break;
135 }
136 }
137 brix_sd_obj_release(so);
138 free(buf);
139 return -1;
140}

142/* Parse `buf` as a whitelist and verify it against the policy master key +
143 * wall-clock expiry. 0 on success; fills *out for the caller's fingerprint
144 * check. */
145static int
146meta_check_whitelist(sd_cache_inst_state *st, const unsigned char *buf,
147 size_t len, const char *key, cvmfs_whitelist_t *out)
148{
149 if (cvmfs_whitelist_parse(buf, len, out) != 0) {
150 ngx_log_error(NGX_LOG_WARN, st->log, 0,
151 "sd_cache: cvmfs whitelist for \"%s\" is malformed", key);
152 return -1;
153 }
154 if (cvmfs_verify_whitelist(out, st->policy.cvmfs_master_pub,
155 st->policy.cvmfs_master_pub_len) != 0)
156 {
157 ngx_log_error(NGX_LOG_WARN, st->log, 0,
158 "sd_cache: cvmfs whitelist for \"%s\" FAILED master-key "
159 "signature verification", key);
160 return -1;
161 }
162 if (cvmfs_whitelist_expired(out, (long) time(NULL))) {
163 ngx_log_error(NGX_LOG_WARN, st->log, 0,
164 "sd_cache: cvmfs whitelist for \"%s\" is EXPIRED", key);
165 return -1;
166 }
167 return 0;
168}

170/* Verify a staged .cvmfspublished: parse, cross-check the repo name against
171 * the fill key, fetch the live whitelist + signing cert through the source,
172 * and run the full chain. 0 = verified, -1 = definitive reject (tamper),
173 * -2 = the chain could not be evaluated (sibling fetch failed — an origin
174 * fault, not evidence of tampering). */
175static int
176meta_check_manifest(sd_cache_inst_state *st, const char *key,
177 const cvmfs_url_info_t *info, const unsigned char *buf, size_t len)
178{
179 cvmfs_manifest_t m;
180 cvmfs_whitelist_t wl;
181 unsigned char *wlbuf = NULL, *cert = NULL, *pem = NULL;
182 ssize_t wln, certn;
183 size_t pemn = 0;
184 char fp[64];
185 char sib[1024];
186 char objp[160];
187 int n, rc = -1;
188
189 if (cvmfs_manifest_parse(buf, len, &m) != 0) {
190 ngx_log_error(NGX_LOG_WARN, st->log, 0,
191 "sd_cache: cvmfs manifest \"%s\" is malformed", key);
192 return -1;
193 }
194 /* The signed 'N' must name the repo the key claims — a valid manifest for
195 * repo A must not publish under repo B's namespace (cross-repo splice). */
196 if (m.repo_name[0] != '\0'
197 && (strlen(m.repo_name) != info->repo_len
198 || strncmp(m.repo_name, info->repo, info->repo_len) != 0))
199 {
200 ngx_log_error(NGX_LOG_WARN, st->log, 0,
201 "sd_cache: cvmfs manifest \"%s\" names repo \"%s\" — "
202 "cross-repo splice rejected", key, m.repo_name);
203 return -1;
204 }
205
206 /* Trust anchor: the repo's live whitelist, fetched through the SAME
207 * source tier (ranked failover applies). */
208 n = snprintf(sib, sizeof(sib), "/cvmfs/%.*s/.cvmfswhitelist",
209 (int) info->repo_len, info->repo);
210 if (n <= 0 || (size_t) n >= sizeof(sib)) {
211 return -1;
212 }
213 wln = meta_fetch_source(st, sib, &wlbuf);
214 if (wln < 0) {
215 ngx_log_error(NGX_LOG_WARN, st->log, 0,
216 "sd_cache: cannot fetch \"%s\" to verify \"%s\"", sib, key);
217 return -2; /* origin fault, not tamper */
218 }
219 if (meta_check_whitelist(st, wlbuf, (size_t) wln, key, &wl) != 0) {
220 free(wlbuf);
221 return -1;
222 }
223
224 /* Signing cert: the manifest's 'X' CAS object, inflated to PEM/DER. */
225 if (cvmfs_hash_to_object_path(&m.certificate, 'X', objp,
226 sizeof(objp)) < 0)
227 {
228 free(wlbuf);
229 return -1;
230 }
231 n = snprintf(sib, sizeof(sib), "/cvmfs/%.*s/data/%s",
232 (int) info->repo_len, info->repo, objp);
233 if (n <= 0 || (size_t) n >= sizeof(sib)) {
234 free(wlbuf);
235 return -1;
236 }
237 certn = meta_fetch_source(st, sib, &cert);
238 if (certn < 0) {
239 ngx_log_error(NGX_LOG_WARN, st->log, 0,
240 "sd_cache: cannot fetch cert \"%s\" to verify \"%s\"", sib, key);
241 free(wlbuf);
242 return -2; /* origin fault, not tamper */
243 }
244 /* Stored cert objects are zlib-deflated; fall back to the raw bytes for
245 * an uncompressed store (cvmfs_cert_fingerprint takes PEM or DER). */
246 pem = malloc(SD_CACHE_META_MAX);
247 if (pem == NULL
248 || cvmfs_object_inflate(cert, (size_t) certn, pem,
249 SD_CACHE_META_MAX, &pemn) != 0)
250 {
251 free(pem);
252 pem = cert;
253 pemn = (size_t) certn;
254 cert = NULL;
255 }
256
257 if (cvmfs_cert_fingerprint(pem, pemn, fp, sizeof(fp)) != 0
258 || !cvmfs_whitelist_lists_fp(&wl, fp))
259 {
260 ngx_log_error(NGX_LOG_WARN, st->log, 0,
261 "sd_cache: cvmfs manifest \"%s\" signing cert is NOT in the "
262 "whitelist — rejected", key);
263 } else if (cvmfs_verify_manifest(&m, pem, pemn) != 0) {
264 ngx_log_error(NGX_LOG_WARN, st->log, 0,
265 "sd_cache: cvmfs manifest \"%s\" FAILED certificate signature "
266 "verification", key);
267 } else {
268 rc = 0;
269 }
270
271 free(cert);
272 free(pem);
273 free(wlbuf);
274 return rc;
275}

277/* Verify a MANIFEST-class staged fill before commit (phase-85 F1). See the
278 * declaration in sd_cache_internal.h for the contract; NGX_OK also covers
279 * "not a signed-metadata shape" (.cvmfsreflog is unsigned) and "verify not
280 * configured". */
281ngx_int_t
282sd_cache_verify_manifest(sd_cache_inst_state *st, const char *key,
283 const char *pp)
284{
285 cvmfs_url_info_t info;
286 cvmfs_whitelist_t wl;
287 unsigned char *buf = NULL;
288 ssize_t len;
289 size_t rel;
290 ngx_int_t rc;
291
292 if (st->policy.cvmfs_master_pub == NULL
293 || st->policy.cvmfs_master_pub_len == 0)
294 {
295 return NGX_OK;
296 }
297 (void) cvmfs_classify_url(key, strlen(key), &info);
298 if (info.cls != CVMFS_URL_MANIFEST) {
299 return NGX_OK; /* CAS/GEO — not this gate */
300 }
301 rel = info.rel_len;
302 if (rel == 0 || info.rel == NULL) {
303 return NGX_OK;
304 }
305
306 if (!((rel == sizeof(".cvmfspublished") - 1
307 && memcmp(info.rel, ".cvmfspublished", rel) == 0)
308 || (rel == sizeof(".cvmfswhitelist") - 1
309 && memcmp(info.rel, ".cvmfswhitelist", rel) == 0)))
310 {
311 return NGX_OK; /* .cvmfsreflog etc. — unsigned */
312 }
313
314 if (pp == NULL) {
315 ngx_log_error(NGX_LOG_ERR, st->log, 0,
316 "sd_cache: cvmfs manifest verify has no staged path for \"%s\" "
317 "- failing the fill closed", key);
318 return NGX_DECLINED;
319 }
320 len = meta_read_part(pp, &buf);
321 if (len < 0) {
322 ngx_log_error(NGX_LOG_ERR, st->log, errno,
323 "sd_cache: cvmfs manifest verify cannot read the staged part "
324 "for \"%s\" - failing the fill closed", key);
325 return NGX_DECLINED;
326 }
327
328 if (info.rel[6] == 'w') { /* ".cvmfsWhitelist" */
329 rc = (meta_check_whitelist(st, buf, (size_t) len, key, &wl) == 0)
330 ? NGX_OK : NGX_ERROR; /* every whitelist fail is
331 * evaluated on the artifact
332 * itself — definitive */
333 } else {
334 switch (meta_check_manifest(st, key, &info, buf, (size_t) len)) {
335 case 0: rc = NGX_OK; break;
336 case -2: rc = NGX_DECLINED; break;
337 default: rc = NGX_ERROR; break;
338 }
339 }
340 free(buf);
341 return rc;
342}
