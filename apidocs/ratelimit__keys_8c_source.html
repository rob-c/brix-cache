1/*
2 * ratelimit_keys.c — Phase 25 key extraction + directive parsing.
3 *
4 * Two planes expose identity through different context structs, so a pair of
5 * extraction functions produce the canonical "<type>:<value>" key string used
6 * for the rbtree lookup. Anonymous principals (no VO / issuer / DN) fall back
7 * to the client IP so unauthenticated bulk clients are always subject to at
8 * least an IP-keyed rule (Phase 25 invariant 5).
9 *
10 * The three directive setters (zone / rate-rule / bandwidth-rule) are shared by
11 * the WebDAV and stream command tables; the rule setters locate the per-conf
12 * rules array via cmd->offset so one implementation serves both planes.
13 */
14#include "ratelimit.h"
15#include "protocols/webdav/webdav.h" /* ngx_http_brix_webdav_req_ctx_t */

18/* key extraction */
19/*
20 * Emit a "dn:<hash>" key for a (potentially long, PII-bearing) GSI subject DN.
21 * WHY hash rather than embed the DN verbatim: the key string is bounded by
22 * BRIX_RL_KEY_LEN and is exposed in dashboard/metrics snapshots, so we never
23 * want the raw DN there. The FNV-1a32 hash is rendered as fixed 8 hex digits
24 * (%08xD = zero-padded uppercase 32-bit), giving a stable, short, collision-
25 * tolerable bucket id for the same principal.
26 */
27static void
28rl_key_dn_hash(const u_char *dn, size_t dn_len, char *out, size_t out_sz)
29{
30 uint32_t h = brix_rl_hash((const char *) dn, dn_len);
31 ngx_snprintf((u_char *) out, out_sz, "dn:%08xD%Z", h);
32}

34/*
35 * Stream-plane key builder: derive the "<type>:<value>" rbtree key for `rule`
36 * from the connection's brix_ctx_t identity. Each branch implements the
37 * Phase 25 invariant-5 fallback: when the principal lacks the requested
38 * dimension (anonymous VO/issuer/DN), fall back to keying on the client IP so
39 * an unauthenticated bulk client is still subject to *some* bucket rather than
40 * escaping the limiter entirely. Returns NGX_DECLINED for a VOLUME rule whose
41 * prefix does not match `path` (rule simply does not apply here).
42 */
43ngx_int_t
44brix_rl_key_stream(brix_rl_rule_t *rule, brix_ctx_t *ctx,
45 const char *path, char *out, size_t out_sz)
46{
47 switch (rule->key_type) {
48
49 case BRIX_RL_KEY_VO:
50 if (ctx->login.primary_vo[0] == '\0') {
51 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
52 } else {
53 ngx_snprintf((u_char *) out, out_sz, "vo:%s%Z", ctx->login.primary_vo);
54 }
55 break;
56
57 case BRIX_RL_KEY_ISSUER:
58 if (ctx->identity == NULL || ctx->identity->issuer.len == 0) {
59 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
60 } else {
61 ngx_snprintf((u_char *) out, out_sz, "iss:%V%Z",
62 &ctx->identity->issuer);
63 }
64 break;
65
66 case BRIX_RL_KEY_IP:
67 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
68 break;
69
70 case BRIX_RL_KEY_DN:
71 if (ctx->login.dn[0] == '\0') {
72 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
73 } else {
74 rl_key_dn_hash((u_char *) ctx->login.dn, ngx_strlen(ctx->login.dn),
75 out, out_sz);
76 }
77 break;
78
79 case BRIX_RL_KEY_VOLUME:
80 /* Prefix match: the rule limits aggregate traffic under one storage
81 * path (e.g. "/store/tape"). All requests below the prefix share the
82 * single "vol:<prefix>" bucket; non-matching paths decline the rule. */
83 if (path == NULL || rule->key_match.len == 0
84 || ngx_strncmp(path, rule->key_match.data, rule->key_match.len) != 0)
85 {
86 return NGX_DECLINED; /* rule does not apply to this path */
87 }
88 ngx_snprintf((u_char *) out, out_sz, "vol:%V%Z", &rule->key_match);
89 break;
90
91 default:
92 return NGX_ERROR;
93 }
94
95 /* ngx_snprintf does not guarantee NUL on truncation; force-terminate so the
96 * key is always a valid C string for the hash/copy in the zone layer. */
97 out[out_sz - 1] = '\0';
98 return NGX_OK;
99}

101/*
102 * HTTP/WebDAV-plane key builder — the same dimension-and-fallback logic as
103 * brix_rl_key_stream(), but reading identity from the WebDAV request ctx and
104 * the nginx connection. DN keying has an extra fallback step: prefer the
105 * structured identity DN, then the cert-derived wctx->dn string, then IP.
106 * wctx_v is an opaque ngx_http_brix_webdav_req_ctx_t* (header avoids the
107 * webdav.h include in ratelimit.h) and may be NULL on early-phase requests.
108 */
109ngx_int_t
110brix_rl_key_http(brix_rl_rule_t *rule, ngx_http_request_t *r,
111 void *wctx_v, const char *path, char *out, size_t out_sz)
112{
113 ngx_http_brix_webdav_req_ctx_t *wctx = wctx_v;
114 brix_identity_t *id = wctx ? wctx->identity : NULL;
115 ngx_str_t *ip = &r->connection->addr_text;
116
117 switch (rule->key_type) {
118
119 case BRIX_RL_KEY_VO:
120 if (id == NULL || id->vo_csv.len == 0) {
121 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", ip);
122 } else {
123 ngx_snprintf((u_char *) out, out_sz, "vo:%V%Z", &id->vo_csv);
124 }
125 break;
126
127 case BRIX_RL_KEY_ISSUER:
128 if (id == NULL || id->issuer.len == 0) {
129 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", ip);
130 } else {
131 ngx_snprintf((u_char *) out, out_sz, "iss:%V%Z", &id->issuer);
132 }
133 break;
134
135 case BRIX_RL_KEY_IP:
136 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", ip);
137 break;
138
139 case BRIX_RL_KEY_DN:
140 /* Two DN sources can carry the subject: the unified identity (token/SSS
141 * bridged) and the raw client-cert DN cached on wctx. Prefer identity,
142 * fall back to the cert string, then to IP if the request is anon. */
143 if (id != NULL && id->dn.len > 0) {
144 rl_key_dn_hash(id->dn.data, id->dn.len, out, out_sz);
145 } else if (wctx != NULL && wctx->dn[0] != '\0') {
146 rl_key_dn_hash((u_char *) wctx->dn, ngx_strlen(wctx->dn),
147 out, out_sz);
148 } else {
149 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", ip);
150 }
151 break;
152
153 case BRIX_RL_KEY_VOLUME:
154 if (path == NULL || rule->key_match.len == 0
155 || ngx_strncmp(path, rule->key_match.data, rule->key_match.len) != 0)
156 {
157 return NGX_DECLINED;
158 }
159 ngx_snprintf((u_char *) out, out_sz, "vol:%V%Z", &rule->key_match);
160 break;
161
162 default:
163 return NGX_ERROR;
164 }
165
166 out[out_sz - 1] = '\0';
167 return NGX_OK;
168}

171/* directive parameter parsing */
172/* Parse "key=<type>[:<prefix>]" into rule->key_type / rule->key_match.
173 * sizeof("key=")-1 == 4 strips the literal prefix; the value tail is then split
174 * on the first ':' into <type> and an optional <prefix> (used only by VOLUME). */
175static ngx_int_t
176rl_parse_key(ngx_conf_t *cf, ngx_str_t *v, brix_rl_rule_t *rule)
177{
178 ngx_str_t val = { v->len - (sizeof("key=") - 1),
179 v->data + sizeof("key=") - 1 };
180 u_char *colon;
181 ngx_str_t type = val, prefix = { 0, NULL };
182
183 /* If a ':' is present, narrow `type` to the part before it and point
184 * `prefix` at the (non-NUL-terminated) remainder after it. */
185 colon = ngx_strlchr(val.data, val.data + val.len, ':');
186 if (colon != NULL) {
187 type.len = colon - val.data;
188 prefix.data = colon + 1;
189 prefix.len = val.data + val.len - (colon + 1);
190 }
191
192 if (type.len == 2 && ngx_strncmp(type.data, "vo", 2) == 0) {
193 rule->key_type = BRIX_RL_KEY_VO;
194 } else if (type.len == 6 && ngx_strncmp(type.data, "issuer", 6) == 0) {
195 rule->key_type = BRIX_RL_KEY_ISSUER;
196 } else if (type.len == 2 && ngx_strncmp(type.data, "ip", 2) == 0) {
197 rule->key_type = BRIX_RL_KEY_IP;
198 } else if (type.len == 2 && ngx_strncmp(type.data, "dn", 2) == 0) {
199 rule->key_type = BRIX_RL_KEY_DN;
200 } else if (type.len == 6 && ngx_strncmp(type.data, "volume", 6) == 0) {
201 rule->key_type = BRIX_RL_KEY_VOLUME;
202 rule->key_match = prefix; /* points into cf->args memory (persists) */
203 } else {
204 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
205 "xrootd rate limit: unknown key \"%V\" (expected"
206 " vo|issuer|ip|dn|volume[:<prefix>])", &type);
207 return NGX_ERROR;
208 }
209 return NGX_OK;
210}

212/* Parse "<N>r/s" → requests/s (returns the integer N, or NGX_ERROR). */
213static ngx_int_t
214rl_parse_req_rate(ngx_str_t *v)
215{
216 if (v->len < 4 || ngx_strncmp(v->data + v->len - 3, "r/s", 3) != 0) {
217 return NGX_ERROR;
218 }
219 return ngx_atoi(v->data, v->len - 3);
220}
221
222/* Parse "<N>[k|m|g]/s" → bytes/s. Returns NGX_ERROR on bad input.
223 * Layout: strip the trailing "/s", then consume an optional binary-unit suffix
224 * (k/m/g = 1024^1..3) off the end so ngx_atoi sees only the digits. */
225static ssize_t
226rl_parse_bw_rate(ngx_str_t *v)
227{
228 ngx_str_t num = *v;
229 off_t mult = 1;
230
231 if (v->len < 3 || ngx_strncmp(v->data + v->len - 2, "/s", 2) != 0) {
232 return NGX_ERROR;
233 }
234 num.len = v->len - 2; /* drop "/s" */
235 switch (num.data[num.len - 1]) { /* optional unit suffix */
236 case 'k': case 'K': mult = 1024; num.len--; break;
237 case 'm': case 'M': mult = 1024 * 1024; num.len--; break;
238 case 'g': case 'G': mult = 1024L * 1024 * 1024; num.len--; break;
239 default: break;
240 }
241 {
242 ngx_int_t n = ngx_atoi(num.data, num.len);
243 if (n == NGX_ERROR) { return NGX_ERROR; }
244 return (ssize_t) ((off_t) n * mult);
245 }
246}

252 return ngx_parse_size(v); /* nginx: handles k/m/g suffixes */

256/* zone directive: brix_rate_limit_zone zone=NAME:SIZE */
257char *
258brix_rl_zone_directive(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
259{
260 ngx_str_t *value = cf->args->elts;
261 ngx_str_t spec, name, sizestr;
262 u_char *colon;
263 ssize_t size;
264 ngx_uint_t i;
265
266 (void) cmd; (void) conf;
267
268 name.len = 0; name.data = NULL;
269 sizestr.len = 0; sizestr.data = NULL;
270
271 for (i = 1; i < cf->args->nelts; i++) {
272 if (value[i].len > 5
273 && ngx_strncmp(value[i].data, "zone=", 5) == 0)
274 {
275 spec.data = value[i].data + 5;
276 spec.len = value[i].len - 5;
277 colon = ngx_strlchr(spec.data, spec.data + spec.len, ':');
278 if (colon == NULL) {
279 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
280 "brix_rate_limit_zone: expected zone=NAME:SIZE");
281 return NGX_CONF_ERROR;
282 }
283 name.data = spec.data;
284 name.len = colon - spec.data;
285 sizestr.data = colon + 1;
286 sizestr.len = spec.data + spec.len - (colon + 1);
287 }
288 }
289
290 if (name.len == 0 || sizestr.len == 0) {
291 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
292 "brix_rate_limit_zone: missing zone=NAME:SIZE");
293 return NGX_CONF_ERROR;
294 }
295 size = rl_parse_size(&sizestr);
296 if (size == NGX_ERROR || size <= 0) {
297 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
298 "brix_rate_limit_zone: bad size \"%V\"", &sizestr);
299 return NGX_CONF_ERROR;
300 }
301
302 if (brix_rl_zone_add(cf, &name, (size_t) size, NULL) != NGX_OK) {
303 return NGX_CONF_ERROR;
304 }
305 return NGX_CONF_OK;
306}

309/* shared rule builder */
310/*
311 * Shared setter behind both brix_rate_limit_rule (is_bw=0) and
312 * brix_bandwidth_limit (is_bw=1). Pushes one brix_rl_rule_t onto the rules
313 * array, parses the zone=/key=/rate=/burst=/nodelay= tokens, applies defaults,
314 * and resolves the named zone handle. is_bw selects whether rate=/burst= are
315 * interpreted as bytes/s + byte burst or req/s + request burst. Using
316 * cmd->offset (below) lets the *same* function serve both the WebDAV loc-conf
317 * and the stream srv-conf command tables without knowing the struct layout.
318 */
319static char *
320rl_add_rule(ngx_conf_t *cf, ngx_command_t *cmd, void *conf, int is_bw)
321{
322 ngx_str_t *value = cf->args->elts;
323 ngx_array_t **rulesp;
324 brix_rl_rule_t *rule;
325 ngx_str_t zone_name = { 0, NULL };
326 ngx_uint_t i;
327 int have_key = 0, have_rate = 0;
328
329 /* The rules array lives at cmd->offset inside the (loc or srv) conf;
330 * create it lazily on the first rule for this conf. */
331 rulesp = (ngx_array_t **) ((char *) conf + cmd->offset);
332 if (*rulesp == NULL) {
333 *rulesp = ngx_array_create(cf->pool, 4, sizeof(brix_rl_rule_t));
334 if (*rulesp == NULL) { return NGX_CONF_ERROR; }
335 }
336 rule = ngx_array_push(*rulesp);
337 if (rule == NULL) { return NGX_CONF_ERROR; }
338 ngx_memzero(rule, sizeof(*rule));
339
340 for (i = 1; i < cf->args->nelts; i++) {
341 ngx_str_t *a = &value[i];
342
343 if (a->len > 5 && ngx_strncmp(a->data, "zone=", 5) == 0) {
344 zone_name.data = a->data + 5;
345 zone_name.len = a->len - 5;
346
347 } else if (a->len > 4 && ngx_strncmp(a->data, "key=", 4) == 0) {
348 if (rl_parse_key(cf, a, rule) != NGX_OK) { return NGX_CONF_ERROR; }
349 have_key = 1;
350
351 } else if (a->len > 5 && ngx_strncmp(a->data, "rate=", 5) == 0) {
352 ngx_str_t r = { a->len - 5, a->data + 5 };
353 if (is_bw) {
354 ssize_t bps = rl_parse_bw_rate(&r);
355 if (bps == NGX_ERROR) {
356 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
357 "brix_bandwidth_limit: bad rate \"%V\""
358 " (expected <N>[k|m|g]/s)", &r);
359 return NGX_CONF_ERROR;
360 }
361 rule->bw_rate = (ngx_uint_t) bps;
362 } else {
363 ngx_int_t rps = rl_parse_req_rate(&r);
364 if (rps == NGX_ERROR || rps <= 0) {
365 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
366 "brix_rate_limit_rule: bad rate \"%V\""
367 " (expected <N>r/s)", &r);
368 return NGX_CONF_ERROR;
369 }
370 /* Stored in milli-requests/s (×1000) so the leaky-bucket core
371 * can do sub-request-per-second fixed-point math without
372 * floats; req_excess in the node is in the same ×1000 unit. */
373 rule->req_rate = (ngx_uint_t) rps * 1000; /* store ×1000 */
374 }
375 have_rate = 1;
376
377 } else if (a->len > 6 && ngx_strncmp(a->data, "burst=", 6) == 0) {
378 ngx_str_t b = { a->len - 6, a->data + 6 };
379 if (is_bw) {
380 ssize_t sz = rl_parse_size(&b);
381 if (sz == NGX_ERROR) {
382 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
383 "brix_bandwidth_limit: bad burst \"%V\"", &b);
384 return NGX_CONF_ERROR;
385 }
386 rule->bw_burst = (ngx_uint_t) sz;
387 } else {
388 ngx_int_t n = ngx_atoi(b.data, b.len);
389 if (n == NGX_ERROR) {
390 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
391 "brix_rate_limit_rule: bad burst \"%V\"", &b);
392 return NGX_CONF_ERROR;
393 }
394 rule->req_burst = (ngx_uint_t) n;
395 }
396
397 } else if (a->len == sizeof("nodelay") - 1
398 && ngx_strncmp(a->data, "nodelay", 7) == 0) {
399 rule->nodelay = 1;
400
401 } else {
402 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
403 "xrootd rate limit: unknown parameter \"%V\"", a);
404 return NGX_CONF_ERROR;
405 }
406 }
407
408 if (!have_key || !have_rate || zone_name.len == 0) {
409 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
410 "xrootd rate limit: zone=, key= and rate= are required");
411 return NGX_CONF_ERROR;
412 }
413 /* Apply burst defaults when the operator omitted burst=. Bandwidth: one
414 * second of rate (smooths bursty transfers); requests: 1 (strict). */
415 if (is_bw && rule->bw_burst == 0) {
416 rule->bw_burst = rule->bw_rate; /* default burst = 1s of rate */
417 }
418 if (!is_bw && rule->req_burst == 0) {
419 rule->req_burst = 1;
420 }
421
422 /* Bind to a zone declared earlier by brix_rate_limit_zone; resolution is
423 * by name so http{} and stream{} rules can share one SHM zone. */
424 rule->zone = brix_rl_zone_get(&zone_name);
425 if (rule->zone == NULL) {
426 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
427 "xrootd rate limit: unknown zone \"%V\" (declare it with"
428 " brix_rate_limit_zone first)", &zone_name);
429 return NGX_CONF_ERROR;
430 }
431 return NGX_CONF_OK;
432}
