1/*
2 * ratelimit_keys.c — Phase 25 key extraction.
3 *
4 * Two planes expose identity through different context structs, so a pair of
5 * extraction functions produce the canonical "<type>:<value>" key string used
6 * for the rbtree lookup. Anonymous principals (no VO / issuer / DN) fall back
7 * to the client IP so unauthenticated bulk clients are always subject to at
8 * least an IP-keyed rule (Phase 25 invariant 5).
9 *
10 * The directive setters (zone / rate-rule / bandwidth-rule / concurrency-rule)
11 * live in the sibling ratelimit_keys_parse.c (value primitives) and
12 * ratelimit_keys_rules.c (shared rule builders).
13 */
14#include "ratelimit.h"
15#include "protocols/webdav/webdav.h" /* ngx_http_brix_webdav_req_ctx_t */
16#include "ratelimit_keys_internal.h"

20/*
21 * Emit a "dn:<hash>" key for a (potentially long, PII-bearing) GSI subject DN.
22 * WHY hash rather than embed the DN verbatim: the key string is bounded by
23 * BRIX_RL_KEY_LEN and is exposed in dashboard/metrics snapshots, so we never
24 * want the raw DN there. The FNV-1a32 hash is rendered as fixed 8 hex digits
25 * (%08xD = zero-padded uppercase 32-bit), giving a stable, short, collision-
26 * tolerable bucket id for the same principal.
27 */
28static void
29rl_key_dn_hash(const u_char *dn, size_t dn_len, char *out, size_t out_sz)
30{
31 uint32_t h = brix_rl_hash((const char *) dn, dn_len);
32 ngx_snprintf((u_char *) out, out_sz, "dn:%08xD%Z", h);
33}
34
35/*
36 * Emit a "sub:<hash>" key for a WLCG/JWT token subject. Same reasoning as the
37 * DN hash: a token `sub` claim can be an opaque id or (for some IdPs) an email,
38 * so it is never embedded verbatim in a key that surfaces in dashboard/metrics
39 * snapshots — it is FNV-1a32-hashed to a fixed 8 hex digits, bounding the key
40 * length and keeping the low-cardinality/no-PII invariant the DN key already
41 * observes.
42 */
43static void
44rl_key_sub_hash(const u_char *sub, size_t sub_len, char *out, size_t out_sz)
45{
46 uint32_t h = brix_rl_hash((const char *) sub, sub_len);
47 ngx_snprintf((u_char *) out, out_sz, "sub:%08xD%Z", h);
48}

50/*
51 * Stream-plane key builder: derive the "<type>:<value>" rbtree key for `rule`
52 * from the connection's brix_ctx_t identity. Each branch implements the
53 * Phase 25 invariant-5 fallback: when the principal lacks the requested
54 * dimension (anonymous VO/issuer/DN), fall back to keying on the client IP so
55 * an unauthenticated bulk client is still subject to *some* bucket rather than
56 * escaping the limiter entirely. Returns NGX_DECLINED for a VOLUME rule whose
57 * prefix does not match `path` (rule simply does not apply here).
58 */
59ngx_int_t
60brix_rl_key_stream(brix_rl_rule_t *rule, brix_ctx_t *ctx,
61 const char *path, char *out, size_t out_sz)
62{
63 switch (rule->key_type) {
64
65 case BRIX_RL_KEY_VO:
66 if (ctx->login.primary_vo[0] == '\0') {
67 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
68 } else {
69 ngx_snprintf((u_char *) out, out_sz, "vo:%s%Z", ctx->login.primary_vo);
70 }
71 break;
72
73 case BRIX_RL_KEY_ISSUER:
74 if (ctx->identity == NULL || ctx->identity->issuer.len == 0) {
75 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
76 } else {
77 ngx_snprintf((u_char *) out, out_sz, "iss:%V%Z",
78 &ctx->identity->issuer);
79 }
80 break;
81
82 case BRIX_RL_KEY_IP:
83 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
84 break;
85
86 case BRIX_RL_KEY_DN:
87 if (ctx->login.dn[0] == '\0') {
88 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
89 } else {
90 rl_key_dn_hash((u_char *) ctx->login.dn, ngx_strlen(ctx->login.dn),
91 out, out_sz);
92 }
93 break;
94
95 case BRIX_RL_KEY_SUBJECT:
96 if (ctx->identity == NULL || ctx->identity->subject.len == 0) {
97 ngx_snprintf((u_char *) out, out_sz, "ip:%s%Z", ctx->login.peer_ip);
98 } else {
99 rl_key_sub_hash(ctx->identity->subject.data,
100 ctx->identity->subject.len, out, out_sz);
101 }
102 break;
103
104 case BRIX_RL_KEY_VOLUME:
105 /* Prefix match: the rule limits aggregate traffic under one storage
106 * path (e.g. "/store/tape"). All requests below the prefix share the
107 * single "vol:<prefix>" bucket; non-matching paths decline the rule. */
108 if (path == NULL || rule->key_match.len == 0
109 || ngx_strncmp(path, rule->key_match.data, rule->key_match.len) != 0)
110 {
111 return NGX_DECLINED; /* rule does not apply to this path */
112 }
113 ngx_snprintf((u_char *) out, out_sz, "vol:%V%Z", &rule->key_match);
114 break;
115
116 default:
117 return NGX_ERROR;
118 }
119
120 /* ngx_snprintf does not guarantee NUL on truncation; force-terminate so the
121 * key is always a valid C string for the hash/copy in the zone layer. */
122 out[out_sz - 1] = '\0';
123 return NGX_OK;
124}

126/* rl_key_req_t (the resolved-identity bundle passed to brix_rl_key_http) is
127 * declared in ratelimit.h so ratelimit_http.c can build the literal directly.
128 * `wctx` is void* there to keep webdav.h out of the header; the branch helpers
129 * below cast it to ngx_http_brix_webdav_req_ctx_t* where they read the cert DN. */

131/* ---- Derive the DN key for an HTTP request, with the two-source fallback ----
132 *
133 * WHAT: Writes a "dn:<hash>" (or IP fallback) key into `out`. Prefers the
134 * unified identity DN, then the cert-derived wctx->dn string, then the client IP
135 * when the request is anonymous.
136 *
137 * WHY: DN keying carries an extra fallback step over the other dimensions; the
138 * unified identity (token/SSS bridged) and the raw client-cert DN cached on wctx
139 * can each independently carry the subject. Splitting it out keeps the main
140 * switch flat.
141 *
142 * HOW: (1) identity DN present → hash it; (2) else cert DN present → hash it;
143 * (3) else fall back to the IP key.
144 */
145static void
146rl_key_http_dn(const rl_key_req_t *req, char *out, size_t out_sz)
147{
148 ngx_http_brix_webdav_req_ctx_t *wctx = req->wctx;
149
150 if (req->id != NULL && req->id->dn.len > 0) {
151 rl_key_dn_hash(req->id->dn.data, req->id->dn.len, out, out_sz);
152 } else if (wctx != NULL && wctx->dn[0] != '\0') {
153 rl_key_dn_hash((u_char *) wctx->dn, ngx_strlen(wctx->dn),
154 out, out_sz);
155 } else {
156 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", req->ip);
157 }
158}

160/* ---- Build the "<type>:<value>" key for one HTTP rule dimension ----
161 *
162 * WHAT: Emits the rbtree key for `rule->key_type` into `out`. Returns NGX_OK on
163 * success, NGX_DECLINED for a VOLUME rule whose prefix does not match the path,
164 * and NGX_ERROR for an unknown key type.
165 *
166 * WHY: Isolates the per-key-type branch logic from brix_rl_key_http()'s frozen
167 * six-argument public shell so the derivation reads as one small switch over an
168 * explicit, already-resolved identity struct. Each anon branch keeps the Phase
169 * 25 invariant-5 IP fallback so an unauthenticated client is still bucketed.
170 *
171 * HOW: switch on key_type — VO/ISSUER fall back to IP when the dimension is
172 * absent; IP is unconditional; DN defers to rl_key_http_dn(); VOLUME prefix-
173 * matches `path` or declines.
174 */
175static ngx_int_t
176rl_key_http_derive(brix_rl_rule_t *rule, const rl_key_req_t *req,
177 char *out, size_t out_sz)
178{
179 switch (rule->key_type) {
180
181 case BRIX_RL_KEY_VO:
182 if (req->id == NULL || req->id->vo_csv.len == 0) {
183 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", req->ip);
184 } else {
185 ngx_snprintf((u_char *) out, out_sz, "vo:%V%Z", &req->id->vo_csv);
186 }
187 return NGX_OK;
188
189 case BRIX_RL_KEY_ISSUER:
190 if (req->id == NULL || req->id->issuer.len == 0) {
191 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", req->ip);
192 } else {
193 ngx_snprintf((u_char *) out, out_sz, "iss:%V%Z", &req->id->issuer);
194 }
195 return NGX_OK;
196
197 case BRIX_RL_KEY_IP:
198 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", req->ip);
199 return NGX_OK;
200
201 case BRIX_RL_KEY_DN:
202 rl_key_http_dn(req, out, out_sz);
203 return NGX_OK;
204
205 case BRIX_RL_KEY_SUBJECT:
206 if (req->id == NULL || req->id->subject.len == 0) {
207 ngx_snprintf((u_char *) out, out_sz, "ip:%V%Z", req->ip);
208 } else {
209 rl_key_sub_hash(req->id->subject.data, req->id->subject.len,
210 out, out_sz);
211 }
212 return NGX_OK;
213
214 case BRIX_RL_KEY_VOLUME:
215 if (req->path == NULL || rule->key_match.len == 0
216 || ngx_strncmp(req->path, rule->key_match.data,
217 rule->key_match.len) != 0)
218 {
219 return NGX_DECLINED;
220 }
221 ngx_snprintf((u_char *) out, out_sz, "vol:%V%Z", &rule->key_match);
222 return NGX_OK;
223
224 default:
225 return NGX_ERROR;
226 }
227}

229/*
230 * HTTP/WebDAV-plane key builder — the same dimension-and-fallback logic as
231 * brix_rl_key_stream(), but reading identity from a caller-resolved rl_key_req_t
232 * bundle instead of the WebDAV ctx and connection directly. DN keying has an
233 * extra fallback step: prefer the structured identity DN, then the cert-derived
234 * wctx->dn string, then IP. The caller (ratelimit_http.c) hoists the
235 * side-effecting lookups (wctx->identity, connection addr) into `req` before the
236 * call; the branch logic lives in rl_key_http_derive().
237 */
238ngx_int_t
239brix_rl_key_http(brix_rl_rule_t *rule, const rl_key_req_t *req,
240 char *out, size_t out_sz)
241{
242 ngx_int_t rc = rl_key_http_derive(rule, req, out, out_sz);
243
244 if (rc != NGX_OK) {
245 return rc;
246 }
247 out[out_sz - 1] = '\0';
248 return NGX_OK;
249}
