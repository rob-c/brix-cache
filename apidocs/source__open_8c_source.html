74/*
75 * tpc_open_wait_resend — WHAT: sleep a short retry-poll interval then resend
76 * the open. WHY: kXR_wait / deferred-wait share this exact sleep+resend step;
77 * factoring it removes the duplicate and keeps the resolver flat. The sleep is
78 * a short poll (NOT the source's full wait hint — that lapses the
79 * briefly-visible TPC grant).
80 * HOW: parse the (bounded) wait seconds from `body`, sleep, resend `oc`; 0 on a
81 * successful resend, -1 (with t->err_msg set) if the resend fails.
82 */
83static int
84tpc_open_wait_resend(brix_tpc_pull_t *t, const tpc_open_ctx_t *oc,
85 const u_char *body, uint32_t dlen, const char *what)
86{
87 uint32_t secs = xrd_wait_secs_parse(body, dlen, 1, TPC_OPEN_WAIT_RETRY_SEC);
88
89 sleep((unsigned) secs);
90 if (tpc_send_all(t, oc->fd, oc->buf, oc->len) != 0) {
91 snprintf(t->err_msg, sizeof(t->err_msg),
92 "TPC kXR_open resend (after %s) failed", what);
93 return -1;
94 }
95 return 0;

98/*
99 * tpc_open_resolve_attn — WHAT: unwrap a kXR_attn frame and either fold it back
100 * into the resolution loop (deferred wait) or hand the caller a terminal
101 * embedded reply. WHY: the asynresp unwrap is the densest branch of the resolver;
102 * isolating it drops the resolver's complexity below the gate.
103 * HOW: body layout is
104 * actnum[4 BE] reserved[4] embedded-ServerResponseHeader[8] body[dlen]
105 * where the embedded header is { streamid[2], status[2 BE], dlen[4 BE] }. Only
106 * kXR_asynresp carries a reply; anything else is not ours. Returns:
107 * 0 → `out` holds a TERMINAL embedded reply (out->body owned by caller)
108 * 1 → keep looping (nothing terminal yet; any embedded body already freed)
109 * -1 → hard failure (t->err_msg / t->xrd_error set)
110 * `attn_body` is always consumed (freed) by this function.
111 */
112static int
113tpc_open_resolve_attn(brix_tpc_pull_t *t, const tpc_open_ctx_t *oc,
114 u_char *attn_body, uint32_t attn_dlen,
115 tpc_open_reply_t *out)
116{
117 uint32_t actnum, edlen;
118 uint16_t est;
119 u_char *ebody = NULL;
120
121 if (attn_body == NULL || attn_dlen < 16) {
122 free(attn_body);
123 snprintf(t->err_msg, sizeof(t->err_msg),
124 "TPC kXR_attn frame too short (%u)", (unsigned) attn_dlen);
125 t->xrd_error = kXR_ServerError;
126 return -1;
127 }
128 actnum = xrd_get_u32_be(attn_body);
129 if (actnum != (uint32_t) kXR_asynresp) {
130 free(attn_body);
131 return 1; /* not ours — keep waiting */
132 }
133 est = (uint16_t) (((uint16_t) attn_body[10] << 8)
134 | (uint16_t) attn_body[11]);
135 edlen = xrd_get_u32_be(attn_body + 12);
136 if ((size_t) edlen > (size_t) attn_dlen - 16) {
137 edlen = attn_dlen - 16; /* never over-read */
138 }
139 if (edlen > 0) {
140 ebody = malloc((size_t) edlen + 1);
141 if (ebody == NULL) {
142 free(attn_body);
143 t->xrd_error = kXR_NoMemory;
144 return -1;
145 }
146 memcpy(ebody, attn_body + 16, edlen);
147 ebody[edlen] = '\0';
148 }
149 free(attn_body);
150
151 /* A deferred reply that is itself a wait folds back into the loop. */
152 if (est == kXR_wait) {
153 int rc = tpc_open_wait_resend(t, oc, ebody, edlen, "asynresp wait");
154
155 free(ebody);
156 return (rc != 0) ? -1 : 1;
157 }
158 if (est == kXR_waitresp) {
159 free(ebody);
160 return 1;
161 }
162
163 out->status = est;
164 out->body = ebody;
165 out->dlen = edlen;
166 return 0; /* terminal embedded reply */
167}

169/*
170 * tpc_open_resolve_round — WHAT: run exactly one resolution round: receive a
171 * frame and classify it. WHY: keeps the driver loop (tpc_open_resolve) a flat
172 * dispatch with the per-status handling extracted here.
173 * HOW: recv one frame → kXR_wait (sleep+resend) / kXR_waitresp (wait) / kXR_attn
174 * (delegate) / terminal (fill `out`). Returns 0 = terminal (out filled),
175 * 1 = keep looping, -1 = failure.
176 */
177static int
178tpc_open_resolve_round(brix_tpc_pull_t *t, const tpc_open_ctx_t *oc,
179 tpc_open_reply_t *out)
180{
181 u_char *b = NULL;
182 uint32_t dl = 0;
183 uint16_t st = 0;
184
185 if (tpc_recv_response(t, oc->fd, &st, &b, &dl) != 0) {
186 snprintf(t->err_msg, sizeof(t->err_msg), "TPC kXR_open recv failed");
187 return -1;
188 }
189
190 /* kXR_wait: sleep the (clamped) retry-after, then resend the open. */
191 if (st == kXR_wait) {
192 int rc = tpc_open_wait_resend(t, oc, b, dl, "kXR_wait");
193
194 free(b);
195 return (rc != 0) ? -1 : 1;
196 }
197
198 /* kXR_waitresp: the seconds are informational; the real reply arrives
199 * unsolicited as a kXR_attn(asynresp). Do NOT resend — loop to receive
200 * it (bounded by the tightened SO_RCVTIMEO + the wall-clock deadline). */
201 if (st == kXR_waitresp) {
202 free(b);
203 return 1;
204 }
205
206 /* kXR_attn: unwrap the deferred response (may fold back into the loop). */
207 if (st == kXR_attn) {
208 return tpc_open_resolve_attn(t, oc, b, dl, out);
209 }
210
211 /* Any other status (kXR_ok / kXR_error / …) is the terminal reply. */
212 out->status = st;
213 out->body = b;
214 out->dlen = dl;
215 return 0;
216}

218/*
219 * tpc_open_resolve — receive the kXR_open reply, transparently resolving the
220 * asynchronous XRootD flow-control statuses a real source may return before the
221 * open settles:
222 *
223 * kXR_wait — "retry in N s": sleep (clamped) and RESEND the open.
224 * kXR_waitresp — "the answer arrives later, unsolicited": do NOT resend; wait
225 * for the deferred kXR_attn(kXR_asynresp).
226 * kXR_attn — unwrap the embedded ServerResponseHeader + body (asynresp)
227 * and treat it as the real reply (which may itself be a wait →
228 * fold back into the loop).
229 *
230 * On return 0 `out` carries a TERMINAL reply (kXR_ok / kXR_error / …) the caller
231 * handles exactly as a synchronous open; the common case (a source that answers
232 * kXR_ok immediately) returns on the very first frame with no change in
233 * behaviour. Bounded on every axis — the caller tightens SO_RCVTIMEO around this
234 * call, and a clamped single wait, a total iteration cap and a wall-clock
235 * deadline guarantee a source that never resolves fails cleanly (-1) rather than
236 * hanging the pull thread. Runs on a thread-pool worker, so the bounded sleeps
237 * here block only this transfer's thread, never the nginx event loop.
238 */
239static int
240tpc_open_resolve(brix_tpc_pull_t *t, const tpc_open_ctx_t *oc,
241 tpc_open_reply_t *out)
242{
243 time_t deadline = time(NULL) + TPC_OPEN_RESOLVE_MAX_SEC;
244 int iters;
245
246 for (iters = 0; iters < TPC_OPEN_RESOLVE_MAX_ITERS; iters++) {
247 int rc;
248
249 if (time(NULL) >= deadline) {
250 snprintf(t->err_msg, sizeof(t->err_msg),
251 "TPC kXR_open async deadline exceeded (%ds)",
252 TPC_OPEN_RESOLVE_MAX_SEC);
253 t->xrd_error = kXR_ServerError;
254 return -1;
255 }
256
257 rc = tpc_open_resolve_round(t, oc, out);
258 if (rc == 0) {
259 return 0; /* terminal reply in `out` */
260 }
261 if (rc < 0) {
262 return -1;
263 }
264 /* rc == 1 → keep looping */
265 }
266
267 snprintf(t->err_msg, sizeof(t->err_msg),
268 "TPC kXR_open did not resolve after %d async rounds",
269 TPC_OPEN_RESOLVE_MAX_ITERS);
270 t->xrd_error = kXR_ServerError;
271 return -1;
272}

275/*
276 * tpc_open_build_opaque — WHAT: render the "?tpc.key=…&tpc.org=…" opaque suffix
277 * for the source open into `opaque` and report its length. WHY: the opaque
278 * selection is a three-way policy branch (delegated / key+org / key-only) that
279 * is cleanly separable from wire framing.
280 * HOW: delegated proxy → empty opaque (authenticated read as the user); else
281 * emit the key (+org) rendezvous params. Returns 0 with *opqlen set, or -1 with
282 * t->err_msg / t->xrd_error set when the rendered opaque overflows the buffer.
283 *
284 * TPC-lite delegation: when we hold the user's delegated proxy we authenticate
285 * to the source AS THE USER, so we open the file directly — no tpc.key opaque.
286 * The tpc.key/tpc.org rendezvous is the *anonymous* TPC model where the pulling
287 * server proves authorization with a key the client pre-registered on the
288 * source; presenting it here makes the source defer with kXR_waitresp forever
289 * (it waits for a client-side authorization that the delegate flow never
290 * issues). A delegated open is a plain authenticated read by the file owner.
291 */
292static int
293tpc_open_build_opaque(brix_tpc_pull_t *t, char *opaque, size_t opaque_sz,
294 size_t *opqlen)
295{
296 size_t n = 0;
297
298 opaque[0] = '\0';
299
300 if (t->deleg_cred_pem != NULL && t->deleg_cred_len > 0) {
301 n = 0;
302 } else if (t->tpc_key[0] != '\0' && t->tpc_org[0] != '\0') {
303 n = (size_t) snprintf(opaque, opaque_sz, "?tpc.key=%s&tpc.org=%s",
304 t->tpc_key, t->tpc_org);
305 } else if (t->tpc_key[0] != '\0') {
306 n = (size_t) snprintf(opaque, opaque_sz, "?tpc.key=%s", t->tpc_key);
307 }
308 if (n >= opaque_sz) {
309 snprintf(t->err_msg, sizeof(t->err_msg), "TPC source opaque too long");
310 t->xrd_error = kXR_ArgTooLong;
311 return -1;
312 }
313
314 *opqlen = n;
315 return 0;
316}

330/*
331 * tpc_open_build_request — WHAT: assemble the kXR_open request bytes for the
332 * source into `dst->buf` and report the total length to send in `dst->len`.
333 * WHY: the wire framing (header pack + payload append) is a distinct concern
334 * from resolving the reply. HOW: see the wire-layout comment below. Returns 0
335 * with dst->len set, or -1 (t->err_msg / t->xrd_error set) if path+opaque
336 * overflow the buffer.
337 */
338static int
339tpc_open_build_request(brix_tpc_pull_t *t, tpc_open_buf_t *dst,
340 const char *opaque, size_t opqlen)
341{
342 ClientOpenRequest *opreq;
343 size_t pathlen = strlen(t->src_path);
344 size_t len = sizeof(ClientOpenRequest) + pathlen + opqlen;
345
346 if (len > dst->cap) {
347 snprintf(t->err_msg, sizeof(t->err_msg), "TPC src path too long");
348 t->xrd_error = kXR_ArgTooLong;
349 return -1;
350 }
351
352 /*
353 * Wire layout of the kXR_open request: fixed ClientOpenRequest header
354 * immediately followed by the variable-length payload (path + opaque).
355 * dlen counts only the payload bytes, NOT the header. All multi-byte
356 * header fields are network byte order (htons/htonl). streamid[1]=2 is a
357 * private tag we reuse across this socket so replies can be correlated.
358 */
359 ngx_memzero(dst->buf, len);
360 opreq = (ClientOpenRequest *) dst->buf;
361 opreq->streamid[1] = 2;
362 opreq->requestid = htons(kXR_open);
363 {
364 xrdw_open_req_t b = { .options = kXR_open_read };
365 xrdw_open_req_pack(&b, ((ClientRequestHdr *) dst->buf)->body);
366 }
367 opreq->dlen = htonl((kXR_int32)(pathlen + opqlen));
368
369 /* Append payload right after the header: path first, then "?tpc..." opaque. */
370 ngx_memcpy(dst->buf + sizeof(ClientOpenRequest), t->src_path, pathlen);
371 if (opqlen > 0) {
372 ngx_memcpy(dst->buf + sizeof(ClientOpenRequest) + pathlen,
373 opaque, opqlen);
374 }
375
376 dst->len = len;
377 return 0;
378}

380/*
381 * tpc_open_extract_fhandle — WHAT: validate a terminal kXR_open reply and copy
382 * out the origin fhandle. WHY: separates reply classification (ok vs error, with
383 * the shared kXR_error decode) from the framing/resolve steps. HOW: on kXR_ok
384 * with a body >= XRD_FHANDLE_LEN, copy the leading fhandle bytes; otherwise
385 * surface the remote's code/message. `reply->body` is always freed here.
386 * Returns 0 (fhandle filled) or -1 (t->err_msg / t->xrd_error set).
387 */
388static int
389tpc_open_extract_fhandle(brix_tpc_pull_t *t, tpc_open_reply_t *reply,
390 u_char fhandle[XRD_FHANDLE_LEN])
391{
392 if (reply->status != kXR_ok || reply->body == NULL
393 || reply->dlen < XRD_FHANDLE_LEN)
394 {
395 /*
396 * kXR_error body = [int32 BE errnum][message]. Surface the remote's
397 * code/message verbatim so the destination's reply mirrors the true
398 * origin failure; the shared decoder bounds the (non-NUL) message slice.
399 */
400 int rerr;
401 const char *rmsg;
402 size_t rmsglen;
403
404 if (reply->status == kXR_error
405 && xrd_error_body_decode(reply->body, reply->dlen,
406 &rerr, &rmsg, &rmsglen) == 0) {
407 snprintf(t->err_msg, sizeof(t->err_msg),
408 "TPC source open failed: %.*s", (int) rmsglen, rmsg);
409 t->xrd_error = rerr;
410 } else {
411 snprintf(t->err_msg, sizeof(t->err_msg),
412 "TPC kXR_open rejected (status=%u dlen=%u)",
413 (unsigned) reply->status, (unsigned) reply->dlen);
414 }
415 free(reply->body);
416 return -1;
417 }
418
419 /* fhandle is the first XRD_FHANDLE_LEN bytes regardless of body size
420 * (minimal 4-byte reply vs full ServerOpenBody both lead with it). */
421 ngx_memcpy(fhandle, reply->body, XRD_FHANDLE_LEN);
422 free(reply->body);
423 return 0;
424}

426/*
427 * tpc_open_source — Phase 1: build and send the kXR_open for the remote source,
428 * resolve the (possibly asynchronous) reply, and extract the origin fhandle.
429 * Returns 0 with `fhandle` filled, or -1 with t->err_msg / t->xrd_error set. On
430 * failure the caller has no origin handle to close.
431 */
432int
433tpc_open_source(brix_tpc_pull_t *t, int fd, u_char fhandle[XRD_FHANDLE_LEN])
434{
435 u_char open_buf[sizeof(ClientOpenRequest) + PATH_MAX + 512];
436 char opaque[512];
437 size_t opqlen = 0;
438 tpc_open_buf_t ob = { .buf = open_buf, .cap = sizeof(open_buf), .len = 0 };
439 tpc_open_ctx_t oc;
440 tpc_open_reply_t reply = { 0 };
441
442 if (tpc_open_build_opaque(t, opaque, sizeof(opaque), &opqlen) != 0) {
443 return -1;
444 }
445 if (tpc_open_build_request(t, &ob, opaque, opqlen) != 0) {
446 return -1;
447 }
448
449 if (tpc_send_all(t, fd, ob.buf, ob.len) != 0) {
450 snprintf(t->err_msg, sizeof(t->err_msg), "TPC kXR_open send failed");
451 return -1;
452 }
453
454 /*
455 * Resolve the open through the async XRootD flow (phase-57 §F8): a real
456 * source (EOS/dCache, or any server still completing the TPC rendezvous) may
457 * answer with kXR_wait / kXR_waitresp → kXR_attn(asynresp) before the open
458 * settles. tpc_open_resolve() honours that and returns a TERMINAL reply. We
459 * tighten SO_RCVTIMEO for the negotiation so a silent source fails fast (the
460 * client's --tpc fallback still has time to run), then restore the full I/O
461 * timeout for the streaming read loop below. A source that answers kXR_ok
462 * immediately (e.g. our own nginx source) resolves on the first frame.
463 *
464 * Some peers return a minimal kXR_ok body with only the 4-byte fhandle;
465 * others send the full ServerOpenBody (12+ bytes) — both lead with fhandle.
466 */
467 oc.fd = fd;
468 oc.buf = ob.buf;
469 oc.len = ob.len;
470 tpc_set_rcvtimeo(fd, TPC_OPEN_WAIT_CAP_SEC);
471 if (tpc_open_resolve(t, &oc, &reply) != 0) {
472 tpc_set_rcvtimeo(fd, TPC_IO_TIMEOUT_SEC);
473 return -1;
474 }
475 tpc_set_rcvtimeo(fd, TPC_IO_TIMEOUT_SEC);
476
477 return tpc_open_extract_fhandle(t, &reply, fhandle);
478}
