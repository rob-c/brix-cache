1/*
2 * http_body_decode.c - shared nginx request-body decompress-to-fd pipeline.
3 *
4 * WHAT: Streams a Content-Encoding codec (gzip/deflate/zstd/xz/brotli/bzip2/lz4)
5 * over an nginx request body chain, writing the decoded plaintext to a
6 * destination fd under a decompression-bomb guard. Owns the codec stream
7 * setup, the per-buffer feed loop (memory + spooled buffers), and the
8 * failed-decode HTTP-status mapping.
9 *
10 * WHY: WebDAV PUT and S3 PUT may carry any supported Content-Encoding; the
11 * stored object must be the decoded bytes, streamed so a large or hostile
12 * body never lands fully in RAM or fills the disk. Split from the plain
13 * chain read/write helpers in http_body.c so each translation unit owns a
14 * single concept and stays under the file-size / complexity gates.
15 *
16 * HOW: brix_http_body_decode_to_fd summarises (validates) the body, sets up one
17 * codec_ctx_t (stream + scratch buffers), hands the per-buffer feed loop to
18 * codec_decode_bufs, then frees buffers and closes the stream once on
19 * return. codec_feed drains produced output through the shared
20 * brix_http_body_pwrite_full helper (http_body_internal.h). inbuf holds the
21 * pread of each spooled buffer before it is fed. brix_http_body_inflate_to_fd
22 * is the legacy zlib-window_bits wrapper.
23*/

35/*
36 * codec_ctx_t — file-local decode-pipeline state, threaded through the codec
37 * feed / decode-bufs / setup helpers.
38 *
39 * WHAT: bundles everything a single request-body decode needs — the open codec
40 * stream, the log/destination it writes to, the scratch in/out buffers,
41 * the running output offset, and the first negative codec rc seen.
42 * WHY: codec_feed / codec_decode_bufs / decode_to_fd previously threaded 7–11
43 * positional parameters (stream, log, fd, path, offset, buffers, worst_rc)
44 * through every call; collapsing them into one explicitly-passed struct
45 * keeps data flow visible while dropping the per-function param counts
46 * under the gate. No hidden globals — the struct is a stack local owned by
47 * decode_to_fd and passed by pointer.
48 * HOW: s/log/dst_fd/log_path/outbuf/inbuf are set once at setup and read-only
49 * thereafter; dst_off and worst_rc are the mutable running state advanced
50 * by the feed loop. inbuf is NULL when the body has no spooled buffers.
51 */
52typedef struct {
53 brix_codec_stream_t *s;
54 ngx_log_t *log;
55 ngx_fd_t dst_fd;
56 const char *log_path;
57 u_char *outbuf;
58 u_char *inbuf;
59 off_t dst_off;
60 brix_codec_rc_t worst_rc;
61} codec_ctx_t;

63/* Feed one input chunk through the codec stream, writing all produced output to
64 * cx->dst_fd. finish!=0 marks the final input. *ended is set when the codec
65 * reports end-of-stream. cx->worst_rc captures the first negative codec rc (for
66 * the caller's HTTP-status mapping: ERR_BOMB -> 413, ERR_DATA -> 400). Returns
67 * NGX_OK/ERROR. */
68static ngx_int_t
69codec_feed(codec_ctx_t *cx, const u_char *in, size_t in_len, int finish,
70 int *ended)
71{
72 size_t ip = 0;
73
74 for (;;) {
75 size_t op = 0;
76 brix_codec_rc_t rc;
77
78 rc = brix_codec_step(cx->s, (const uint8_t *) in, in_len, &ip,
79 (uint8_t *) cx->outbuf, BRIX_INFLATE_OUT_BUFSZ,
80 &op, finish);
81 if (rc < 0) {
82 cx->worst_rc = rc;
83 ngx_log_error(NGX_LOG_ERR, cx->log, 0,
84 "brix_http_body: decode error %d for %s",
85 (int) rc, cx->log_path ? cx->log_path : "-");
86 return NGX_ERROR;
87 }
88 if (op > 0) {
89 if (brix_http_body_pwrite_full(cx->log, cx->dst_fd, cx->outbuf, op,
90 &cx->dst_off, cx->log_path)
91 != NGX_OK)
92 {
93 return NGX_ERROR;
94 }
95 }
96 if (rc == BRIX_CODEC_END) {
97 *ended = 1;
98 return NGX_OK;
99 }
100 if (op == BRIX_INFLATE_OUT_BUFSZ) {
101 continue; /* output buffer was full: drain more */
102 }
103 if (ip < in_len) {
104 continue; /* input remains in this chunk */
105 }
106 return NGX_OK; /* chunk consumed, output drained */
107 }
108}

110/*
111 * codec_decode_spooled_buf — feed one spooled (in_file) body buffer through the
112 * codec stream.
113 *
114 * WHAT: preads buf[file_pos..file_last] in BRIX_INFLATE_IN_BUFSZ chunks into
115 * cx->inbuf and hands each chunk to codec_feed; *ended tracks whether the
116 * codec has reported end-of-stream.
117 * WHY: a spooled buffer has no in-memory payload, so it must be read from its
118 * temp fd before feeding; pulling this out of codec_decode_bufs keeps that
119 * orchestrator under the CCN gate.
120 * HOW: loop while file_off < file_last: one storage-seam pread (a short fill
121 * before file_last is a hard error, matching the old n<=0 check), then
122 * feed the got bytes (finish=0). Returns NGX_OK/ERROR.
123 */
124static ngx_int_t
125codec_decode_spooled_buf(codec_ctx_t *cx, ngx_buf_t *b, int *ended)
126{
127 off_t file_off = b->file_pos;
128
129 while (file_off < b->file_last) {
130 size_t want = (size_t) (b->file_last - file_off);
131 size_t got = 0;
132
133 if (want > BRIX_INFLATE_IN_BUFSZ) {
134 want = BRIX_INFLATE_IN_BUFSZ;
135 }
136 /* One chunk through the storage seam; EOF before file_last (a short
137 * fill) is an error, matching the old n<=0 check. */
138 if (brix_vfs_pread_full(b->file->fd, cx->inbuf, want, file_off,
139 &got) != NGX_OK || got < want)
140 {
141 ngx_log_error(NGX_LOG_ERR, cx->log, errno,
142 "brix_http_body: decode pread failed for %s",
143 cx->log_path ? cx->log_path : "-");
144 return NGX_ERROR;
145 }
146 file_off += (off_t) got;
147 if (codec_feed(cx, cx->inbuf, got, 0, ended) != NGX_OK) {
148 return NGX_ERROR;
149 }
150 }
151
152 return NGX_OK;
153}

155/*
156 * codec_decode_bufs — feed every request-body buffer through an open codec
157 * stream, writing decompressed bytes to dst_fd, then finalise.
158 *
159 * In-memory buffers feed directly; spooled buffers are pread in
160 * BRIX_INFLATE_IN_BUFSZ chunks into inbuf first. After the last buffer a final
161 * flush (empty input, finish=1) drains any tail; if the codec never reports
162 * end-of-stream the input was truncated/corrupt (ERR_DATA -> caller maps 400).
163 * Flat, early-return cleanup; the stream + buffers are owned by the caller.
164 */
165static ngx_int_t
166codec_decode_bufs(codec_ctx_t *cx, ngx_http_request_t *r)
167{
168 ngx_chain_t *cl;
169 int ended = 0;
170
171 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
172 ngx_buf_t *b = cl->buf;
173
174 if (b == NULL) {
175 continue;
176 }
177
178 if (b->in_file) {
179 if (codec_decode_spooled_buf(cx, b, &ended) != NGX_OK) {
180 return NGX_ERROR;
181 }
182
183 } else if (b->pos < b->last) {
184 if (codec_feed(cx, b->pos, (size_t) (b->last - b->pos), 0,
185 &ended) != NGX_OK)
186 {
187 return NGX_ERROR;
188 }
189 }
190 }
191
192 /* Final flush: signal end-of-input and drain the codec's tail. */
193 if (!ended) {
194 if (codec_feed(cx, (const u_char *) "", 0, 1, &ended) != NGX_OK) {
195 return NGX_ERROR;
196 }
197 }
198 if (!ended) {
199 /* All input consumed + finish, but no end-of-stream: truncated input. */
200 cx->worst_rc = BRIX_CODEC_ERR_DATA;
201 ngx_log_error(NGX_LOG_ERR, cx->log, 0,
202 "brix_http_body: truncated compressed body for %s",
203 cx->log_path ? cx->log_path : "-");
204 return NGX_ERROR;
205 }
206 return NGX_OK;
207}

209/*
210 * brix_http_body_decode_ratio - choose the untrusted decode ratio ceiling.
211 *
212 * WHAT: Returns the maximum permitted output:input expansion ratio for the given
213 * Content-Encoding codec on HTTP request-body decode.
214 * WHY: The generic 1000:1 ceiling catches classic bombs for most codecs, but LZ4
215 * frames cap zero-heavy compression below that; a lower LZ4-specific limit
216 * keeps PUT bomb protection effective without changing trusted decode paths.
217 * HOW: Select the codec-specific constant for LZ4, otherwise use the shared
218 * default consumed by the central codec guard.
219 */
220static uint32_t
221brix_http_body_decode_ratio(brix_codec_id_t codec)
222{
223 if (codec == BRIX_CODEC_LZ4) {
224 return BRIX_DECODE_LZ4_MAX_RATIO;
225 }
226 return BRIX_DECODE_MAX_RATIO;
227}

229/*
230 * codec_ctx_setup — open the codec stream and its scratch buffers into *cx.
231 *
232 * WHAT: opens a decompress stream for `codec` (bomb-guarded by max_output +
233 * codec ratio), allocates cx->outbuf and — only when the body has spooled
234 * buffers — cx->inbuf, and fills the read-only fields of *cx.
235 * WHY: isolating the multi-resource acquisition (with its own failure ladder)
236 * from the decode orchestrator keeps that function flat and under the CCN
237 * gate; on any failure this helper frees whatever it already took, so the
238 * caller never partial-owns resources.
239 * HOW: cx must arrive with s/log/dst_fd/log_path/outbuf/inbuf pre-set by the
240 * caller (dst_off/worst_rc zero-init). On open failure -> UNSUPPORTED_
241 * MEDIA_TYPE; on alloc failure -> INTERNAL_SERVER_ERROR, freeing the
242 * stream (and outbuf) first. Returns NGX_OK with cx fully populated, else
243 * NGX_ERROR with *http_status_out set.
244 */
245static ngx_int_t
246codec_ctx_setup(codec_ctx_t *cx, brix_codec_id_t codec, uint64_t max_output,
247 int need_inbuf, ngx_int_t *http_status_out)
248{
249 brix_codec_guard_t guard;
250
251 ngx_memzero(&guard, sizeof(guard));
252 guard.out_cap = max_output; /* 0 = unbounded */
253 guard.max_ratio = brix_http_body_decode_ratio(codec);
254
255 cx->s = brix_codec_open(codec, BRIX_CODEC_DIR_DECOMPRESS, -1, &guard);
256 if (cx->s == NULL) {
257 if (http_status_out) {
258 *http_status_out = NGX_HTTP_UNSUPPORTED_MEDIA_TYPE;
259 }
260 ngx_log_error(NGX_LOG_ERR, cx->log, 0,
261 "brix_http_body: codec %d unavailable for %s",
262 (int) codec, cx->log_path ? cx->log_path : "-");
263 return NGX_ERROR;
264 }
265
266 cx->outbuf = ngx_alloc(BRIX_INFLATE_OUT_BUFSZ, cx->log);
267 if (cx->outbuf == NULL) {
268 brix_codec_close(cx->s);
269 if (http_status_out) { *http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR; }
270 return NGX_ERROR;
271 }
272 if (need_inbuf) {
273 cx->inbuf = ngx_alloc(BRIX_INFLATE_IN_BUFSZ, cx->log);
274 if (cx->inbuf == NULL) {
275 ngx_free(cx->outbuf);
276 brix_codec_close(cx->s);
277 if (http_status_out) { *http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR; }
278 return NGX_ERROR;
279 }
280 }
281 return NGX_OK;
282}

284/*
285 * codec_decode_http_status — map a failed decode's worst codec rc onto an HTTP
286 * status. ERR_BOMB -> 413, ERR_DATA -> 400, anything else -> 500. Kept as a pure
287 * lookup so the orchestrator stays branch-light.
288 */
289static ngx_int_t
290codec_decode_http_status(brix_codec_rc_t worst_rc)
291{
292 if (worst_rc == BRIX_CODEC_ERR_BOMB) {
293 return NGX_HTTP_REQUEST_ENTITY_TOO_LARGE; /* 413 */
294 }
295 if (worst_rc == BRIX_CODEC_ERR_DATA) {
296 return NGX_HTTP_BAD_REQUEST; /* 400 */
297 }
298 return NGX_HTTP_INTERNAL_SERVER_ERROR; /* 500 */
299}

301/*
302 * brix_http_body_decode_to_fd - decompress the request body to dst_fd.
303 *
304 * WHAT: streams the Content-Encoding-selected codec over the request body chain,
305 * writing plaintext to dst_fd. Bounds output via the bomb guard (out_cap =
306 * max_output, ratio default) so a hostile highly-compressible upload cannot
307 * exhaust disk; on any failure sets *http_status_out (413 bomb / 400 bad
308 * data / 500 I-O) and returns NGX_ERROR. On success returns NGX_OK.
309 * HOW: summarise (validate) the body, set up one codec_ctx_t (stream + scratch
310 * buffers via codec_ctx_setup), hand the per-buffer feed loop to
311 * codec_decode_bufs, then free buffers + close the stream once on return.
312 */
313ngx_int_t
314brix_http_body_decode_to_fd(ngx_http_request_t *r, ngx_fd_t dst_fd,
315 const char *log_path, brix_codec_id_t codec, uint64_t max_output,
316 brix_http_body_summary_t *summary_out, ngx_int_t *http_status_out)
317{
318 codec_ctx_t cx;
319 ngx_int_t rc;
320 brix_http_body_summary_t summary;
321
322 if (summary_out == NULL) {
323 summary_out = &summary;
324 }
325 if (brix_http_body_summary(r, summary_out) != NGX_OK) {
326 if (http_status_out) { *http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR; }
327 return NGX_ERROR;
328 }
329 if (r == NULL || r->request_body == NULL
330 || r->request_body->bufs == NULL)
331 {
332 return NGX_OK;
333 }
334
335 ngx_memzero(&cx, sizeof(cx));
336 cx.log = r->connection->log;
337 cx.dst_fd = dst_fd;
338 cx.log_path = log_path;
339 cx.worst_rc = BRIX_CODEC_OK;
340
341 if (codec_ctx_setup(&cx, codec, max_output, summary_out->has_spooled,
342 http_status_out) != NGX_OK)
343 {
344 return NGX_ERROR;
345 }
346
347 rc = codec_decode_bufs(&cx, r);
348
349 ngx_free(cx.inbuf);
350 ngx_free(cx.outbuf);
351 brix_codec_close(cx.s);
352
353 if (rc != NGX_OK && http_status_out) {
354 *http_status_out = codec_decode_http_status(cx.worst_rc);
355 }
356 return rc;
357}
