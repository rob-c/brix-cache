1/*
2 * unified.c — the shared realpath()-based path resolver for stream/WebDAV/S3.
3 *
4 * WHAT: Implements brix_path_resolve_cstr() (declared in unified.h) and its
5 * supporting static helpers. Given a canonical export root and a client
6 * request path, it validates components, builds a candidate filesystem
7 * path, canonicalises it with realpath(3), enforces export-root
8 * containment, stats the target for its type, and copies the confined
9 * result out — populating an optional brix_path_result_t. The
10 * brix_path_opts_t flags select per-protocol semantics: allow_root,
11 * require_directory, allow_missing_tail (write of a new leaf), and
12 * allow_missing_parents (recursive mkdir / HTTP PUT-style suffix).
13 *
14 * WHY: This keeps ONE security-critical resolution+confinement implementation
15 * so stream, WebDAV, and S3 callers cannot drift apart on traversal
16 * defences. realpath() collapses ".."/symlinks to an authoritative path,
17 * and brix_path_within_root() then guarantees the result (and, for
18 * not-yet-existing targets, the deepest EXISTING ancestor) stays under the
19 * export root — defeating "/export" vs "/exportdata" prefix attacks and
20 * symlink escapes before any operation touches the target.
21 *
22 * HOW: brix_validate_components_cstr() rejects oversized/over-deep paths and
23 * forbidden "."/".." components; brix_build_candidate() joins root+req;
24 * realpath() canonicalises. If the full path is missing (ENOENT) the
25 * resolver walks back to the nearest existing parent
26 * (brix_resolve_missing_tail / brix_resolve_missing_parents),
27 * canonicalises THAT, re-checks containment, and rebuilds the absolute
28 * target. Status flows back as brix_path_status_t. NOTE: this is the
29 * config-time / legacy resolver — hot runtime client paths now use the
30 * kernel-confined beneath API (beneath.c) instead (see Phase 8 notes).
31 *
32 * The validation/candidate-build stages live in unified_build.c and the
33 * canonicalisation/confinement/fallback stages in unified_resolve.c; the
34 * cross-file seam (shared structs + non-static helpers) is declared in
35 * unified_internal.h. This file keeps the public entry point and the
36 * shared logging/init helpers.
37 */

72/*
73 * brix_path_resolve_cstr — public entry point: resolve and confine req_path
74 * under the already-canonical root_canon, honouring the opts flags, and write
75 * the confined absolute path into resolved[0..resolvsz). When result != NULL it
76 * is filled with the resolved ngx_str_t, target type, depth, and is_confined.
77 *
78 * Returns an brix_path_status_t:
79 * OK — resolved and confined (resolved/result populated).
80 * INVALID — bad components, traversal attempt, or containment violation.
81 * NOT_FOUND — target (and required ancestors) do not exist.
82 * TOO_LONG — a path/buffer bound was exceeded.
83 * ERROR — an unexpected stat()/realpath() failure.
84 *
85 * Flow: validate → build candidate → realpath(); on ENOENT fall back to
86 * missing-parents / missing-tail resolution per opts, otherwise propagate.
87 */
88brix_path_status_t
89brix_path_resolve_cstr(ngx_log_t *log, const char *root_canon,
90 const char *req_path, brix_path_opts_t opts,
91 char *resolved, size_t resolvsz,
92 brix_path_result_t *result)
93{
94 char candidate[PATH_MAX];
95 char canonical[PATH_MAX];
96 ngx_uint_t depth;
97 brix_path_status_t rc;
98 brix_resolve_ctx_t ctx;
99 brix_candidate_out_t cand_out;
100
101 brix_path_result_init(result);
102
103 rc = brix_validate_components_cstr(log, req_path);
104 if (rc != BRIX_PATH_STATUS_OK) {
105 return rc;
106 }
107
108 if (opts.allow_missing_tail && !opts.allow_missing_parents
109 && brix_has_trailing_slash_cstr(req_path))
110 {
111 return BRIX_PATH_STATUS_INVALID;
112 }
113
114 depth = 0;
115 cand_out.buf = candidate;
116 cand_out.bufsz = sizeof(candidate);
117 cand_out.depth_out = &depth;
118 rc = brix_build_candidate(root_canon, req_path, opts, &cand_out);
119 if (rc != BRIX_PATH_STATUS_OK) {
120 return rc;
121 }
122
123 ctx.log = log;
124 ctx.root_canon = root_canon;
125 ctx.resolved = resolved;
126 ctx.resolvsz = resolvsz;
127 ctx.result = result;
128 ctx.depth = depth;
129
130 if (realpath(candidate, canonical) != NULL) {
131 return brix_finish_resolved(&ctx, canonical, opts);
132 }
133
134 if (errno != ENOENT) {
135 return BRIX_PATH_STATUS_ERROR;
136 }
137
138 if (opts.allow_missing_parents) {
139 return brix_resolve_missing_parents(&ctx, candidate);
140 }
141
142 if (opts.allow_missing_tail) {
143 return brix_resolve_missing_tail(&ctx, candidate);
144 }
145
146 return BRIX_PATH_STATUS_NOT_FOUND;
147}
