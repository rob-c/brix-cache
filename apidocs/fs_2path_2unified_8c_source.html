1/*
2 * unified.c — the shared realpath()-based path resolver for stream/WebDAV/S3.
3 *
4 * WHAT: Implements brix_path_resolve_cstr() (declared in unified.h) and its
5 * supporting static helpers. Given a canonical export root and a client
6 * request path, it validates components, builds a candidate filesystem
7 * path, canonicalises it with realpath(3), enforces export-root
8 * containment, stats the target for its type, and copies the confined
9 * result out — populating an optional brix_path_result_t. The
10 * brix_path_opts_t flags select per-protocol semantics: allow_root,
11 * require_directory, allow_missing_tail (write of a new leaf), and
12 * allow_missing_parents (recursive mkdir / HTTP PUT-style suffix).
13 *
14 * WHY: This keeps ONE security-critical resolution+confinement implementation
15 * so stream, WebDAV, and S3 callers cannot drift apart on traversal
16 * defences. realpath() collapses ".."/symlinks to an authoritative path,
17 * and brix_path_within_root() then guarantees the result (and, for
18 * not-yet-existing targets, the deepest EXISTING ancestor) stays under the
19 * export root — defeating "/export" vs "/exportdata" prefix attacks and
20 * symlink escapes before any operation touches the target.
21 *
22 * HOW: brix_validate_components_cstr() rejects oversized/over-deep paths and
23 * forbidden "."/".." components; brix_build_candidate() joins root+req;
24 * realpath() canonicalises. If the full path is missing (ENOENT) the
25 * resolver walks back to the nearest existing parent
26 * (brix_resolve_missing_tail / brix_resolve_missing_parents),
27 * canonicalises THAT, re-checks containment, and rebuilds the absolute
28 * target. Status flows back as brix_path_status_t. NOTE: this is the
29 * config-time / legacy resolver — hot runtime client paths now use the
30 * kernel-confined beneath API (beneath.c) instead (see Phase 8 notes).
31 */

65static ngx_uint_t
66brix_count_components_cstr(const char *path)
67{
68 const char *p;
69 ngx_uint_t count;
70
71 p = path;
72 while (*p == '/') {
73 p++;
74 }
75
76 if (*p == '\0') {
77 return 0;
78 }
79
80 count = 0;
81 while (*p != '\0') {
82 while (*p == '/') {
83 p++;
84 }
85 if (*p == '\0') {
86 break;
87 }
88
89 count++;
90 while (*p != '\0' && *p != '/') {
91 p++;
92 }
93 }
94
95 return count;
96}

98static brix_path_status_t
99brix_validate_components_cstr(ngx_log_t *log, const char *path)
100{
101 const char *p, *seg_start;
102 size_t seg_len;
103
104 if (path == NULL) {
105 return BRIX_PATH_STATUS_INVALID;
106 }
107
108 if (strlen(path) > BRIX_MAX_PATH) {
109 return BRIX_PATH_STATUS_TOO_LONG;
110 }
111
112 if (brix_count_path_depth(path) != NGX_OK) {
113 if (log != NULL) {
114 ngx_log_error(NGX_LOG_WARN, log, 0,
115 "brix: path depth exceeds limit");
116 }
117 return BRIX_PATH_STATUS_INVALID;
118 }
119
120 p = path;
121 while (*p != '\0') {
122 while (*p == '/') {
123 p++;
124 }
125 if (*p == '\0') {
126 break;
127 }
128
129 seg_start = p;
130 while (*p != '\0' && *p != '/') {
131 p++;
132 }
133
134 seg_len = (size_t) (p - seg_start);
135 if (brix_path_component_forbidden(seg_start, seg_len)) {
136 brix_path_warn(log, "brix: path traversal attempt", path);
137 return BRIX_PATH_STATUS_INVALID;
138 }
139 }
140
141 return BRIX_PATH_STATUS_OK;
142}

175static brix_path_status_t
176brix_append_component(char *path, size_t pathsz, size_t *len,
177 const char *component, size_t component_len)
178{
179 if (*len > 1) {
180 if (*len + 1 >= pathsz) {
181 return BRIX_PATH_STATUS_TOO_LONG;
182 }
183 path[(*len)++] = '/';
184 }
185
186 if (*len + component_len >= pathsz) {
187 return BRIX_PATH_STATUS_TOO_LONG;
188 }
189
190 ngx_memcpy(path + *len, component, component_len);
191 *len += component_len;
192 path[*len] = '\0';
193
194 return BRIX_PATH_STATUS_OK;
195}

197static brix_path_status_t
198brix_build_candidate(const char *root_canon, const char *req_path,
199 brix_path_opts_t opts, char *candidate,
200 size_t candidatesz, ngx_uint_t *depth_out)
201{
202 const char *p, *seg_start;
203 size_t len, root_len, seg_len;
204 ngx_uint_t depth;
205 brix_path_status_t rc;
206
207 if (root_canon == NULL || root_canon[0] == '\0' || req_path == NULL) {
208 return BRIX_PATH_STATUS_INVALID;
209 }
210
211 root_len = strlen(root_canon);
212 if (root_len == 0 || root_len >= candidatesz) {
213 return BRIX_PATH_STATUS_TOO_LONG;
214 }
215
216 ngx_memcpy(candidate, root_canon, root_len);
217 len = root_len;
218 while (len > 1 && candidate[len - 1] == '/') {
219 len--;
220 }
221 candidate[len] = '\0';
222
223 depth = brix_count_components_cstr(req_path);
224 if (depth_out != NULL) {
225 *depth_out = depth;
226 }
227
228 if (depth == 0) {
229 if (!opts.allow_root) {
230 return BRIX_PATH_STATUS_INVALID;
231 }
232 return BRIX_PATH_STATUS_OK;
233 }
234
235 p = req_path;
236 while (*p != '\0') {
237 while (*p == '/') {
238 p++;
239 }
240 if (*p == '\0') {
241 break;
242 }
243
244 seg_start = p;
245 while (*p != '\0' && *p != '/') {
246 p++;
247 }
248
249 seg_len = (size_t) (p - seg_start);
250 if (seg_len == 0) {
251 continue;
252 }
253
254 rc = brix_append_component(candidate, candidatesz, &len,
255 seg_start, seg_len);
256 if (rc != BRIX_PATH_STATUS_OK) {
257 return rc;
258 }
259 }
260
261 return BRIX_PATH_STATUS_OK;
262}

264static ngx_int_t
265brix_stat_type_cstr(const char *path)
266{
267 struct stat st;
268
269 if (stat(path, &st) != 0) {
270 if (errno == ENOENT) {
271 return BRIX_PATH_TYPE_NOT_FOUND;
272 }
273 return NGX_ERROR;
274 }
275
276 return (ngx_int_t) (st.st_mode & S_IFMT);
277}

279static brix_path_status_t
280brix_finish_resolved(ngx_log_t *log, const char *root_canon,
281 const char *resolved_path, brix_path_opts_t opts,
282 char *resolved, size_t resolvsz,
283 brix_path_result_t *result, ngx_uint_t depth)
284{
285 ngx_int_t type;
286 brix_path_status_t rc;
287
288 if (!brix_path_within_root(root_canon, resolved_path)) {
289 brix_path_warn(log, "brix: path traversal attempt", resolved_path);
290 return BRIX_PATH_STATUS_INVALID;
291 }
292
293 type = brix_stat_type_cstr(resolved_path);
294 if (type == NGX_ERROR) {
295 if (opts.allow_missing_tail || opts.allow_missing_parents) {
296 type = BRIX_PATH_TYPE_NOT_FOUND;
297 } else {
298 return BRIX_PATH_STATUS_ERROR;
299 }
300 }
301
302 if (opts.require_directory && type != S_IFDIR) {
303 return BRIX_PATH_STATUS_INVALID;
304 }
305
306 rc = brix_copy_cstr(resolved, resolvsz, resolved_path);
307 if (rc != BRIX_PATH_STATUS_OK) {
308 return rc;
309 }
310
311 if (result != NULL) {
312 result->resolved.data = (u_char *) resolved;
313 result->resolved.len = strlen(resolved);
314 result->type = type;
315 result->depth = depth;
316 result->is_confined = 1;
317 }
318
319 return BRIX_PATH_STATUS_OK;
320}

322static brix_path_status_t
323brix_resolve_missing_tail(ngx_log_t *log, const char *root_canon,
324 const char *candidate, char *resolved,
325 size_t resolvsz, brix_path_result_t *result,
326 ngx_uint_t depth)
327{
328 char parent[PATH_MAX];
329 char parent_canon[PATH_MAX];
330 char rebuilt[PATH_MAX];
331 char base_buf[PATH_MAX];
332 char *slash;
333 const char *base;
334 size_t base_len;
335 int n;
336 brix_path_opts_t final_opts;
337
338 if (strlen(candidate) >= sizeof(parent)) {
339 return BRIX_PATH_STATUS_TOO_LONG;
340 }
341
342 ngx_memcpy(parent, candidate, strlen(candidate) + 1);
343 slash = strrchr(parent, '/');
344 if (slash == NULL) {
345 return BRIX_PATH_STATUS_INVALID;
346 }
347
348 base = slash + 1;
349 if (*base == '\0') {
350 return BRIX_PATH_STATUS_INVALID;
351 }
352
353 base_len = strlen(base);
354 if (base_len >= sizeof(base_buf)) {
355 return BRIX_PATH_STATUS_TOO_LONG;
356 }
357 ngx_memcpy(base_buf, base, base_len + 1);
358
359 if (slash == parent) {
360 slash[1] = '\0';
361 } else {
362 *slash = '\0';
363 }
364
365 if (realpath(parent, parent_canon) == NULL) {
366 if (errno == ENOENT) {
367 return BRIX_PATH_STATUS_NOT_FOUND;
368 }
369 return BRIX_PATH_STATUS_ERROR;
370 }
371
372 if (!brix_path_within_root(root_canon, parent_canon)) {
373 brix_path_warn(log, "brix: path traversal attempt in write",
374 parent_canon);
375 return BRIX_PATH_STATUS_INVALID;
376 }
377
378 n = snprintf(rebuilt, sizeof(rebuilt),
379 (strcmp(parent_canon, "/") == 0) ? "/%s" : "%s/%s",
380 parent_canon, base_buf);
381 if (n < 0 || (size_t) n >= sizeof(rebuilt)) {
382 return BRIX_PATH_STATUS_TOO_LONG;
383 }
384
385 final_opts = (brix_path_opts_t) { 0 };
386 final_opts.allow_missing_tail = 1;
387 final_opts.is_write_operation = 1;
388
389 return brix_finish_resolved(log, root_canon, rebuilt, final_opts,
390 resolved, resolvsz, result, depth);
391}

393static brix_path_status_t
394brix_resolve_missing_parents(ngx_log_t *log, const char *root_canon,
395 const char *candidate, char *resolved,
396 size_t resolvsz, brix_path_result_t *result,
397 ngx_uint_t depth)
398{
399 char ancestor[PATH_MAX];
400 char ancestor_canon[PATH_MAX];
401 char rebuilt[PATH_MAX];
402 char *slash;
403 const char *suffix;
404 size_t ancestor_len;
405 int n;
406 brix_path_opts_t final_opts;
407
408 if (strlen(candidate) >= sizeof(ancestor)) {
409 return BRIX_PATH_STATUS_TOO_LONG;
410 }
411
412 ngx_memcpy(ancestor, candidate, strlen(candidate) + 1);
413
414 for ( ;; ) {
415 if (realpath(ancestor, ancestor_canon) != NULL) {
416 break;
417 }
418
419 if (errno != ENOENT) {
420 return BRIX_PATH_STATUS_ERROR;
421 }
422
423 slash = strrchr(ancestor, '/');
424 if (slash == NULL) {
425 return BRIX_PATH_STATUS_NOT_FOUND;
426 }
427
428 if (slash == ancestor) {
429 slash[1] = '\0';
430 } else {
431 *slash = '\0';
432 }
433
434 if (strcmp(ancestor, "/") == 0) {
435 if (realpath(ancestor, ancestor_canon) == NULL) {
436 return BRIX_PATH_STATUS_NOT_FOUND;
437 }
438 break;
439 }
440 }
441
442 if (!brix_path_within_root(root_canon, ancestor_canon)) {
443 brix_path_warn(log, "brix: path traversal attempt", ancestor_canon);
444 return BRIX_PATH_STATUS_INVALID;
445 }
446
447 ancestor_len = strlen(ancestor);
448 suffix = candidate + ancestor_len;
449 if (ancestor_len == 1 && ancestor[0] == '/') {
450 suffix = candidate + 1;
451 n = snprintf(rebuilt, sizeof(rebuilt),
452 (strcmp(ancestor_canon, "/") == 0) ? "/%s" : "%s/%s",
453 ancestor_canon, suffix);
454 } else {
455 n = snprintf(rebuilt, sizeof(rebuilt), "%s%s", ancestor_canon, suffix);
456 }
457
458 if (n < 0 || (size_t) n >= sizeof(rebuilt)) {
459 return BRIX_PATH_STATUS_TOO_LONG;
460 }
461
462 final_opts = (brix_path_opts_t) { 0 };
463 final_opts.allow_missing_parents = 1;
464
465 return brix_finish_resolved(log, root_canon, rebuilt, final_opts,
466 resolved, resolvsz, result, depth);
467}

469/*
470 * brix_path_resolve_cstr — public entry point: resolve and confine req_path
471 * under the already-canonical root_canon, honouring the opts flags, and write
472 * the confined absolute path into resolved[0..resolvsz). When result != NULL it
473 * is filled with the resolved ngx_str_t, target type, depth, and is_confined.
474 *
475 * Returns an brix_path_status_t:
476 * OK — resolved and confined (resolved/result populated).
477 * INVALID — bad components, traversal attempt, or containment violation.
478 * NOT_FOUND — target (and required ancestors) do not exist.
479 * TOO_LONG — a path/buffer bound was exceeded.
480 * ERROR — an unexpected stat()/realpath() failure.
481 *
482 * Flow: validate → build candidate → realpath(); on ENOENT fall back to
483 * missing-parents / missing-tail resolution per opts, otherwise propagate.
484 */
485brix_path_status_t
486brix_path_resolve_cstr(ngx_log_t *log, const char *root_canon,
487 const char *req_path, brix_path_opts_t opts,
488 char *resolved, size_t resolvsz,
489 brix_path_result_t *result)
490{
491 char candidate[PATH_MAX];
492 char canonical[PATH_MAX];
493 ngx_uint_t depth;
494 brix_path_status_t rc;
495
496 brix_path_result_init(result);
497
498 rc = brix_validate_components_cstr(log, req_path);
499 if (rc != BRIX_PATH_STATUS_OK) {
500 return rc;
501 }
502
503 if (opts.allow_missing_tail && !opts.allow_missing_parents
504 && brix_has_trailing_slash_cstr(req_path))
505 {
506 return BRIX_PATH_STATUS_INVALID;
507 }
508
509 rc = brix_build_candidate(root_canon, req_path, opts, candidate,
510 sizeof(candidate), &depth);
511 if (rc != BRIX_PATH_STATUS_OK) {
512 return rc;
513 }
514
515 if (realpath(candidate, canonical) != NULL) {
516 return brix_finish_resolved(log, root_canon, canonical, opts,
517 resolved, resolvsz, result, depth);
518 }
519
520 if (errno != ENOENT) {
521 return BRIX_PATH_STATUS_ERROR;
522 }
523
524 if (opts.allow_missing_parents) {
525 return brix_resolve_missing_parents(log, root_canon, candidate,
526 resolved, resolvsz, result,
527 depth);
528 }
529
530 if (opts.allow_missing_tail) {
531 return brix_resolve_missing_tail(log, root_canon, candidate,
532 resolved, resolvsz, result, depth);
533 }
534
535 return BRIX_PATH_STATUS_NOT_FOUND;
536}
