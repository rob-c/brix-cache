1/*
2 * sd_pblock_cred.c — identity-enforcing *_cred vtable slots for the pblock
3 * storage driver.
4 *
5 * WHAT: Implements every credential-scoped slot of brix_sd_pblock_driver
6 * (open/staged_open/stat/unlink/mkdir/rename/setattr/xattr CRUD/
7 * server_copy/opendir). Each slot resolves the request identity to
8 * catalog-internal synthetic ids (sd_pblock_ident.c), enforces the POSIX
9 * mode-bit rule for the operation against catalog ownership, then
10 * performs the op through the plain implementation (or its owner-aware
11 * `_as` variant when the op creates rows).
12 *
13 * WHY: The settled "enforce in pblock" decision: pblock is its own multi-user
14 * authority — the VFS passes identity down (BRIX_SD_CRED_IDENTITY) and
15 * the backend answers may/may-not from its own registry. group = VO, so
16 * per-VO sharing is plain unix group permissions.
17 *
18 * HOW: The rule per op (documented on each slot): read ops need R on the
19 * object; write/xattr-mutating ops need W on the object; entry
20 * create/remove/rename need W on the immediate parent (plus the sticky
21 * gate on removal); chmod/chown are owner-only, chown-to-other-uid is
22 * service-only, chgrp only to a VO the requester belongs to. A `service`
23 * identity (no principal) bypasses every check — single-user/tests keep
24 * today's semantics. Enforcement is check-then-act against the catalog:
25 * the small TOCTOU window between the check and the act is accepted (a
26 * concurrent chmod may lag one op), matching the kernel-free design.
27 * ngx-free; gated by BRIX_HAVE_SQLITE like the rest of the backend.
28 */

57/* dst_write_gate — POSIX authority to place an entry at `path`: parent X
58 * (traverse) plus W on the entry itself when it exists (an overwrite), else
59 * W+X on the immediate parent (a create). Shared by staged_open_cred and
60 * server_copy_cred. */
61static ngx_int_t
62dst_write_gate(pblock_state_t *st, const char *path, const pblock_ids_t *ids)
63{
64 pblock_meta meta;
65 int rc;
66
67 rc = pblock_catalog_lookup(st->cat, path, &meta);
68 if (rc < 0) {
69 return NGX_ERROR;
70 }
71 if (rc == 0) {
72 if (pblock_ident_check_parent(st, path, ids, X_OK, NULL) != NGX_OK) {
73 return NGX_ERROR;
74 }
75 return pblock_ident_access(&meta, ids, W_OK);
76 }
77 return pblock_ident_check_parent(st, path, ids, W_OK | X_OK, NULL);
78}

80/* remove_gate — POSIX authority to remove/replace the entry `path`: W+X on
81 * the immediate parent plus the sticky-bit owner rule. ENOENT when absent.
82 * Shared by unlink_cred and both halves of rename_cred. */
83static ngx_int_t
84remove_gate(pblock_state_t *st, const char *path, const pblock_ids_t *ids)
85{
86 pblock_meta parent, entry;
87 int rc;
88
89 rc = pblock_catalog_lookup(st->cat, path, &entry);
90 if (rc < 0) {
91 return NGX_ERROR;
92 }
93 if (rc == 1) {
94 errno = ENOENT;
95 return NGX_ERROR;
96 }
97 if (pblock_ident_check_parent(st, path, ids, W_OK | X_OK,
98 &parent) != NGX_OK)
99 {
100 return NGX_ERROR;
101 }
102 return pblock_ident_sticky_gate(&parent, &entry, ids);
103}

107/* open_cred — parent X (traverse) first; then existing object: R for a read
108 * open, W for a write open (O_TRUNC is a write). Absent + O_CREATE: W on the
109 * parent; the new row is owned by the requester (uid, primary-VO gid). */
110brix_sd_obj_t *
111sd_pblock_open_cred(brix_sd_instance_t *inst, const char *path, int sd_flags,
112 mode_t mode, const brix_sd_cred_t *cred, int *err_out)
113{
114 pblock_state_t *st = inst->state;
115 pblock_ids_t ids;
116 pblock_meta meta;
117 int rc;
118
119 if (resolve_or_fail(inst, cred, &ids) != NGX_OK) {
120 if (err_out != NULL) { *err_out = errno; }
121 return NULL;
122 }
123 if (ids.service) {
124 return sd_pblock_open_as(inst, path, sd_flags, mode, 0, 0, err_out);
125 }
126
127 if (pblock_ident_check_parent(st, path, &ids, X_OK, NULL) != NGX_OK) {
128 if (err_out != NULL) { *err_out = errno; }
129 return NULL;
130 }
131
132 rc = pblock_catalog_lookup(st->cat, path, &meta);
133 if (rc < 0) {
134 if (err_out != NULL) { *err_out = errno; }
135 return NULL;
136 }
137 if (rc == 0) {
138 int want = (sd_flags & BRIX_SD_O_WRITE) ? W_OK : R_OK;
139
140 if (pblock_ident_access(&meta, &ids, want) != NGX_OK) {
141 if (err_out != NULL) { *err_out = errno; }
142 return NULL;
143 }
144 } else if (sd_flags & BRIX_SD_O_CREATE) {
145 if (pblock_ident_check_parent(st, path, &ids, W_OK, NULL) != NGX_OK) {
146 if (err_out != NULL) { *err_out = errno; }
147 return NULL;
148 }
149 }
150 /* absent without O_CREATE falls through: open_as reports ENOENT. */
151 return sd_pblock_open_as(inst, path, sd_flags, mode, ids.uid, ids.gid,
152 err_out);
153}

225/* rename_cred — removal authority on src (parent W + sticky); placement
226 * authority on dst (overwrite: removal authority; create: parent W). */
227ngx_int_t
228sd_pblock_rename_cred(brix_sd_instance_t *inst, const char *src,
229 const char *dst, int noreplace, const brix_sd_cred_t *cred)
230{
231 pblock_state_t *st = inst->state;
232 pblock_ids_t ids;
233 int rc;
234

236 return NGX_ERROR;

238 if (!ids.service) {

240 return NGX_ERROR;

244 return NGX_ERROR;

248 return NGX_ERROR;

251 NULL) != NGX_OK)
252 {
253 return NGX_ERROR;
254 }
255 }

259/* setattr_cred — chmod/chown are owner-only; changing the uid at all is
260 * service-only (no giving files away); the owner may chgrp only to a VO they
261 * belong to (or their private group); times need ownership or W. */
262ngx_int_t
263sd_pblock_setattr_cred(brix_sd_instance_t *inst, const char *path,
264 const brix_sd_setattr_t *attr, const brix_sd_cred_t *cred)
265{
266 pblock_state_t *st = inst->state;
267 pblock_ids_t ids;
268 pblock_meta meta;
269
270 if (resolve_or_fail(inst, cred, &ids) != NGX_OK) {
271 return NGX_ERROR;
272 }
273 if (!ids.service) {
274 if (pblock_ident_check(st, path, &ids, 0, &meta) != NGX_OK) {
275 return NGX_ERROR; /* ENOENT */
276 }
277 if ((attr->set_mode || attr->set_owner) && meta.uid != ids.uid) {
278 errno = EPERM;
279 return NGX_ERROR;
280 }
281 if (attr->set_owner) {
282 if (attr->uid != (uid_t) -1 && attr->uid != meta.uid) {
283 errno = EPERM; /* chown is service-only */
284 return NGX_ERROR;
285 }
286 if (attr->gid != (gid_t) -1) {
287 int i, member = attr->gid == ids.uid; /* private group */
288
289 for (i = 0; i < ids.ngids && !member; i++) {
290 member = attr->gid == ids.gids[i];
291 }
292 if (!member) {
293 errno = EPERM; /* chgrp only into own VOs */
294 return NGX_ERROR;
295 }
296 }
297 }
298 if (attr->set_times && meta.uid != ids.uid
299 && pblock_ident_access(&meta, &ids, W_OK) != NGX_OK)
300 {
301 errno = EPERM;
302 return NGX_ERROR;
303 }
304 }
305 return sd_pblock_setattr(inst, path, attr);
306}
