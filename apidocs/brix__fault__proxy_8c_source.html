1/*
2 * brix_fault_proxy.c — brix-fault-proxy: a root-free TCP fault-injection proxy.
3 *
4 * WHAT: Relays TCP from a local listen port to an upstream host:port and injects
5 * network faults on demand — latency, jitter, partial writes, slow-drip,
6 * bandwidth caps, packet-loss (resets), payload corruption, duplication,
7 * deterministic mid-transfer truncation, connection black-holes, and full
8 * outages — set at startup on the command line or live over a control port
9 * while traffic flows. A "toxiproxy-lite" you can splice in front of any
10 * brix (or stock XRootD/HTTP) endpoint to prove a client survives a bad or
11 * actively hostile network, WITHOUT needing root: tc/netem needs
12 * CAP_NET_ADMIN; this is pure userspace sockets.
13 *
14 * DIRECTIONALITY: every byte-level lever is per-direction. `up` = client ->
15 * upstream (the request/upload path), `down` = upstream -> client (the
16 * response/download path). A control command with no direction token
17 * applies to BOTH; append ` up`, ` down`, or ` both` to target one.
18 *
19 * FAULT LEVERS (each off/0 by default; set at startup via the matching --flag or
20 * live by writing the command to the control port):
21 * latency <ms> [dir] delay every forwarded chunk (a high-RTT link)
22 * jitter <ms> [dir] delay each chunk by a uniform-random 0..<ms> ms —
23 * the application signature of IP reordering (see NOTE)
24 * chunk <bytes> [dir] split each write into <=N-byte segments (egress
25 * TCP_NODELAY) so the peer sees genuine partial reads
26 * drip <bytes> <ms> [dir] forward <bytes>, sleep <ms>, repeat (slow stream)
27 * rate <KB/s> [dir] pace the stream to a bandwidth ceiling (slow link)
28 * lossy <pct> [dir] with prob <pct>%/chunk, sever the stream (loss=reset)
29 * reorder <pct> [ms] [dir] with prob <pct>%/chunk, hold that chunk back <ms> ms
30 * corrupt <pct> [dir] flip a random bit in <pct>% of forwarded BYTES — a
31 * MITM tamper that a checksum-verifying peer must catch
32 * dup <pct> [dir] with prob <pct>%/chunk, deliver that chunk twice
33 * truncate-at <bytes> [dir] sever each connection once <bytes> have flowed in
34 * that direction (deterministic mid-transfer cut)
35 * fail-nth <n> fail exactly the Nth accepted connection, pass others
36 * heal-after <ms> auto-clear every lever after <ms> (a transient fault)
37 * one-shot self-heal (clear) the moment a sever/truncate fires
38 * drop graceful-FIN sever every live connection now
39 * reset abortive-RST sever every live connection now, and
40 * make subsequent lossy/truncate severs use RST too
41 * abortive <0|1> select RST (1) vs graceful FIN (0) for auto-severs
42 * half-close FIN the client->upstream path on live conns while the
43 * upstream->client path keeps flowing (tests half-open)
44 * hang accept new connections but never relay (a black hole)
45 * unhang resume relaying newly accepted connections
46 * block drop live conns AND refuse new ones (an outage window)
47 * unblock resume accepting
48 * clear reset every lever (levers=0, unblock, unhang)
49 * status report lever state and traffic/fault counters
50 *
51 * WHY: A client's resilience (reconnect + file-handle resumption, per-PDU read
52 * deadlines, low-speed aborts, checksum-verify-and-retry) needs the exact
53 * conditions of "bad wifi from a laptop abroad" — or an outright tampering
54 * MITM — injected deterministically and without privilege. The peer
55 * connects THROUGH this proxy; an operator (or a test) pulls the levers.
56 *
57 * HOW: Thread-per-connection relay (so per-chunk latency is a simple usleep); a
58 * global drop-epoch counter that, when bumped, makes every live relay thread
59 * sever and exit; a control thread parsing newline commands; an optional
60 * timeline thread replaying a scripted scenario. All randomness is per-thread
61 * rand_r() seeded from --seed for reproducibility.
62 *
63 * SECURITY: the control port is UNAUTHENTICATED — anyone who can reach it can pull
64 * the fault levers and reset live connections. Both the listen and control
65 * sockets therefore bind to loopback by DEFAULT; widening the bind address
66 * (--bind) exposes the control plane and is refused unless you also pass
67 * --insecure-bind, which prints a warning. Never expose the control port to
68 * an untrusted network.
69 *
70 * NOTE: packet LOSS is intentionally modelled as a connection sever, not byte
71 * drops: dropping bytes from an already-ACKed TCP stream corrupts it rather
72 * than emulating loss (which lives below TCP). Likewise true out-of-order
73 * PACKET delivery cannot be emulated by re-ordering bytes — TCP reassembles
74 * in order below us — so `jitter`/`reorder` inject the variable latency that
75 * IP reordering actually imposes on a TCP application. (True packet
76 * reordering needs `tc qdisc ... netem`, i.e. CAP_NET_ADMIN, which this
77 * avoids.) `corrupt`, by contrast, IS a real in-band byte mutation: it is
78 * what an active on-path attacker (or a flaky NIC past the TCP checksum)
79 * does, and it is precisely what an application-layer checksum exists to
80 * catch.
81 */

222/* Blocking connect to host:port (best-effort, first address that works). */
223static int
224dial(const char *host, int port)
225{
226 struct addrinfo hints, *res = NULL, *ai;
227 char portstr[16];
228 int fd = -1;
229
230 memset(&hints, 0, sizeof(hints));
231 hints.ai_family = AF_UNSPEC;
232 hints.ai_socktype = SOCK_STREAM;
233 snprintf(portstr, sizeof(portstr), "%d", port);
234 if (getaddrinfo(host, portstr, &hints, &res) != 0) {
235 return -1;
236 }
237 for (ai = res; ai != NULL; ai = ai->ai_next) {
238 fd = socket(ai->ai_family, ai->ai_socktype, ai->ai_protocol);
239 if (fd < 0) {
240 continue;
241 }
242 if (connect(fd, ai->ai_addr, ai->ai_addrlen) == 0) {
243 break;
244 }
245 close(fd);
246 fd = -1;
247 }
248 freeaddrinfo(res);
249 return fd;
250}

252/* Dial the next target in round-robin order, failing over to the rest. */
253static int
254dial_any(void)
255{
256 int n = g_ntargets;
257 unsigned start = __atomic_fetch_add(&g_rr, 1, __ATOMIC_RELAXED);
258 for (int i = 0; i < n; i++) {
259 int idx = (int) ((start + (unsigned) i) % (unsigned) n);
260 int fd = dial(g_targets[idx].host, g_targets[idx].port);
261 if (fd >= 0) {
262 return fd;
263 }
264 }
265 return -1;
266}

268static int
269write_all(int fd, const char *buf, ssize_t n)
270{
271 ssize_t off = 0;
272 while (off < n) {
273 ssize_t w = write(fd, buf + off, (size_t) (n - off));
274 if (w < 0) {
275 if (errno == EINTR) {
276 continue;
277 }
278 return -1;
279 }
280 off += w;
281 }
282 return 0;

314/* Clamp the outgoing segment to the chunk `piece` and, if truncation is armed,
315 * to the exact remaining distance to the cut. Returns the clamped length, or -1
316 * when the truncation point is already reached (caller must sever). */
317static ssize_t
318fault_clamp_seg(ssize_t seg, int piece, long trunc, unsigned long conn_ctr)
319{
320 if (seg > piece) {
321 seg = piece;
322 }
323 /* Cap at the truncation boundary so the cut is exact rather than firing
324 * only after the whole (possibly 64 KiB) read is delivered. */
325 if (trunc > 0) {
326 long remaining = trunc - (long) conn_ctr;
327 if (remaining <= 0) {
328 return -1;
329 }
330 if (seg > remaining) {
331 seg = remaining;
332 }
333 }
334 return seg;
335}

337/* Apply direction `L`'s pacing/jitter/reorder delays before a segment write. */
338static void
339fault_delays(ssize_t seg, unsigned *seed, const lever_t *L)
340{
341 if (L->jitter_ms > 0) {
342 usleep((useconds_t) (rand_r(seed) % (unsigned) (L->jitter_ms + 1)) * 1000);
343 }
344 if (L->reorder_ppm > 0 && (int) (rand_r(seed) % 1000000u) < L->reorder_ppm) {
345 usleep((useconds_t) L->reorder_ms * 1000);
346 }
347 if (L->rate_kbps > 0) {
348 usleep((useconds_t) ((long long) seg * 1000000
349 / ((long long) L->rate_kbps * 1024)));
350 }
351}

353/* Flip random bits in buf[off, off+seg) at the configured per-byte rate. */
354static void
355fault_corrupt(char *buf, ssize_t off, ssize_t seg, unsigned *seed, int cor)
356{
357 if (cor <= 0) {
358 return;
359 }
360 for (ssize_t k = 0; k < seg; k++) {
361 if ((int) (rand_r(seed) % 1000000u) < cor) {
362 buf[off + k] ^= (char) (1u << (rand_r(seed) % 8u));
363 CBUMP(corrupt, 1);
364 }
365 }
366}

368/* Deliver one segment starting at buf+off, applying `L`'s active faults. On
369 * success writes the delivered length to *wrote and returns 0; returns -1 when
370 * the stream should be severed (truncate cut, lossy drop, write error) or
371 * silently dropped (block/epoch change). */
372static int
373forward_segment(int to, char *buf, ssize_t off, ssize_t n, unsigned epoch,
374 const lever_t *L, unsigned *seed, unsigned long *conn_ctr,
375 unsigned long *glob_ctr, int piece, ssize_t *wrote)
376{
377 ssize_t seg = fault_clamp_seg(n - off, piece, L->truncate_at, *conn_ctr);
378 if (seg < 0) {
379 return fault_sever();
380 }
381
382 fault_delays(seg, seed, L);
383
384 if (L->lossy_ppm > 0 && (int) (rand_r(seed) % 1000000u) < L->lossy_ppm) {
385 return fault_sever(); /* application-visible "loss" = sever the stream */
386 }
387 if (g_blocked || g_drop_epoch != epoch) {
388 return -1;
389 }
390
391 fault_corrupt(buf, off, seg, seed, L->corrupt_ppm);
392
393 if (write_all(to, buf + off, seg) != 0) {
394 return -1;
395 }
396 if (L->dup_ppm > 0 && (int) (rand_r(seed) % 1000000u) < L->dup_ppm) {
397 (void) write_all(to, buf + off, seg); /* duplicate delivery */
398 CBUMP(dups, 1);
399 }
400
401 CBUMP2(conn_ctr, glob_ctr, seg);
402 *wrote = seg;
403
404 if (L->truncate_at > 0 && (long) *conn_ctr >= L->truncate_at) {
405 return fault_sever(); /* deterministic mid-transfer cut */
406 }
407 return 0;
408}

410/*
411 * Forward n bytes to `to`, applying direction `L`'s active fault levers. Bytes
412 * may be mutated in place (corruption). Returns 0 on success, -1 if the
413 * connection should be severed (write error, a lossy drop, or a truncate cut).
414 * `seed` is this thread's private rand_r() state; `conn_ctr` the per-connection
415 * per-direction running byte total (for truncate-at); `glob_ctr` the process-wide
416 * traffic counter for this direction.
417 */
418static int
419forward_faulted(int to, char *buf, ssize_t n, unsigned epoch, volatile lever_t *L,
420 unsigned *seed, unsigned long *conn_ctr, unsigned long *glob_ctr)
421{
422 /* Snapshot the levers once so a mid-buffer control-plane change can't split
423 * this read across two fault configurations (matches the original). */
424 lever_t snap = *L;
425 int piece = (snap.drip_bytes > 0) ? snap.drip_bytes
426 : (snap.chunk_bytes > 0 ? snap.chunk_bytes : (int) n);
427 ssize_t off = 0;
428
429 if (piece <= 0) {
430 piece = (int) n; /* n >= 1 (read returned >0) */
431 }
432 if (snap.latency_ms > 0) {
433 usleep((useconds_t) snap.latency_ms * 1000);
434 }
435
436 while (off < n) {
437 ssize_t wrote = 0;
438 if (forward_segment(to, buf, off, n, epoch, &snap, seed,
439 conn_ctr, glob_ctr, piece, &wrote) != 0) {
440 return -1;
441 }
442 off += wrote;
443 if (snap.drip_bytes > 0 && off < n) {
444 usleep((useconds_t) snap.drip_ms * 1000);
445 }
446 }
447 return 0;
448}

456/* Pre-dial dispositions that answer the client without ever reaching upstream:
457 * the fail-nth sever and the hang/black-hole hold. Returns 1 if the connection
458 * was handled (closed + CDEC), 0 to proceed with dialling upstream. */
459static int
460relay_predial(int cfd, unsigned epoch, unsigned long conn_id)
461{
462 /* fail-nth: sever exactly the Nth accepted connection, then pass the rest. */
463 if (g_fail_nth > 0 && conn_id == (unsigned long) g_fail_nth) {
464 CBUMP(severs, 1);
465 sever(cfd, -1, g_abortive);
466 CDEC(active);
467 return 1;
468 }
469
470 /* hang / black hole: accept but never relay — hold the client open. */
471 if (g_hang) {
472 struct pollfd hp = { cfd, POLLIN, 0 };
473 while (g_hang && g_drop_epoch == epoch && !g_blocked) {
474 if (poll(&hp, 1, 100) > 0 && (hp.revents & (POLLHUP | POLLERR))) {
475 break; /* client gave up */
476 }
477 }
478 close(cfd);
479 CDEC(active);
480 return 1;
481 }
482 return 0;
483}

486/* Relay one poll-ready direction through the fault engine. Returns 0 to keep
487 * looping, 1 on EOF/read error (caller closes both ends), 2 if a fault severed
488 * the pair (already closed + CDEC, caller just returns). */
489static int
490relay_pump_dir(int i, struct pollfd *pfd, int cfd, int ufd,
491 char *buf, size_t bufsz, unsigned epoch,
492 unsigned *seed, unsigned long *up_ctr, unsigned long *down_ctr)
493{
494 if (!(pfd[i].revents & (POLLIN | POLLHUP | POLLERR))) {
495 return 0;
496 }
497 int from = pfd[i].fd;
498 int to = pfd[i ^ 1].fd;
499 ssize_t nr = read(from, buf, bufsz);
500 if (nr <= 0) {
501 return 1;
502 }
503 volatile lever_t *L = (i == 0) ? &g_up : &g_down;
504 unsigned long *conn_ctr = (i == 0) ? up_ctr : down_ctr;
505 unsigned long *glob_ctr = (i == 0) ? &C.up_bytes : &C.down_bytes;
506 if (forward_faulted(to, buf, nr, epoch, L, seed, conn_ctr, glob_ctr) != 0) {
507 sever(cfd, ufd, g_abortive);
508 CDEC(active);
509 return 2;
510 }
511 return 0;
512}

515/* Bidirectional relay loop: shuttle bytes each way through the fault engine
516 * until EOF, a control-plane sever, or a poll error. Closes both ends + CDEC
517 * before returning. */
518static void
519relay_pump(int cfd, int ufd, unsigned epoch, unsigned seed)
520{
521 struct pollfd pfd[2];
522 pfd[0].fd = cfd; /* client -> upstream (up) */
523 pfd[1].fd = ufd; /* upstream -> client (down) */
524 char buf[65536];
525 unsigned long up_ctr = 0, down_ctr = 0;
526 unsigned hc_epoch = g_halfclose_epoch;
527 int hc_done = 0;
528
529 for (;;) {
530 if (g_blocked || g_drop_epoch != epoch) {
531 sever(cfd, ufd, g_abortive);
532 CDEC(active);
533 return;
534 }
535 /* half-close: FIN the up path, keep the down path flowing. */
536 if (!hc_done && g_halfclose_epoch != hc_epoch) {
537 shutdown(cfd, SHUT_RD);
538 shutdown(ufd, SHUT_WR);
539 hc_done = 1;
540 }
541 pfd[0].events = hc_done ? 0 : POLLIN;
542 pfd[1].events = POLLIN;
543 int pr = poll(pfd, 2, 100);
544 if (pr < 0) {
545 if (errno == EINTR) {
546 continue;
547 }
548 break;
549 }
550 if (pr == 0) {
551 continue; /* re-check fault flags */
552 }
553 for (int i = 0; i < 2; i++) {
554 int r = relay_pump_dir(i, pfd, cfd, ufd, buf, sizeof(buf),
555 epoch, &seed, &up_ctr, &down_ctr);
556 if (r == 2) {
557 return; /* severed + CDEC already done */
558 }
559 if (r == 1) {
560 goto done; /* EOF */
561 }
562 }
563 }
564done:
565 close(cfd);
566 close(ufd);
567 CDEC(active);
568}

571static void *
572relay_thread(void *arg)
573{
574 relay_arg *ra = (relay_arg *) arg;
575 int cfd = ra->client_fd;
576 unsigned epoch = ra->epoch;
577 unsigned long conn_id = ra->conn_id;
578 free(ra);
579
580 unsigned seed = g_seed + (unsigned) conn_id * 2654435761u;
581
582 if (relay_predial(cfd, epoch, conn_id)) {
583 return NULL;
584 }
585
586 int ufd = dial_any();
587 if (ufd < 0) {
588 close(cfd);
589 CDEC(active);
590 return NULL;
591 }
592 /* Egress NODELAY on BOTH ends so chunk/drip pieces are delivered as separate
593 * segments (otherwise the kernel coalesces them and the peer never sees a
594 * partial PDU). The accept side already set NODELAY on cfd. */
595 { int one = 1;
596 setsockopt(ufd, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one));
597 setsockopt(cfd, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one)); }
598
599 relay_pump(cfd, ufd, epoch, seed);
600 return NULL;
601}

603/* Strip a trailing up|down|both direction token from `args`; return which lever
604 * set(s) it names: 0 = both, 1 = up only, 2 = down only. */
605static int
606dir_of(char *args)
607{
608 size_t L = strlen(args);
609 while (L > 0 && isspace((unsigned char) args[L - 1])) {
610 args[--L] = '\0';
611 }
612 char *sp = strrchr(args, ' ');
613 char *tok = sp ? sp + 1 : args;
614 int d = -1;
615 if (strcmp(tok, "up") == 0) d = 1;
616 else if (strcmp(tok, "down") == 0) d = 2;
617 else if (strcmp(tok, "both") == 0) d = 0;
618 if (d < 0) {
619 return 0; /* no direction token -> both, leave args untouched */
620 }
621 if (sp) {
622 *sp = '\0'; /* drop the token so numeric parse ignores it */
623 } else {
624 args[0] = '\0';
625 }
626 return d;
627}

666/* Directional traffic levers (latency/bandwidth/corruption). Each strips an
667 * optional up/down/both token via dir_of() then sets the field on the selected
668 * direction(s). `args` is mutated. Returns 1 if `verb` was a lever, else 0. */
669static int
670cmd_set_lever(const char *verb, char *args)
671{
672 if (strcmp(verb, "latency") == 0) {
673 int d = dir_of(args); SET_DIR(d, latency_ms, atoi(args));
674 } else if (strcmp(verb, "jitter") == 0) {
675 int d = dir_of(args); SET_DIR(d, jitter_ms, atoi(args));
676 } else if (strcmp(verb, "chunk") == 0) {
677 int d = dir_of(args); SET_DIR(d, chunk_bytes, atoi(args));
678 } else if (strcmp(verb, "rate") == 0) {
679 int d = dir_of(args); SET_DIR(d, rate_kbps, atoi(args));
680 } else if (strcmp(verb, "drip") == 0) {
681 int d = dir_of(args); int b = 0, m = 0;
682 sscanf(args, "%d %d", &b, &m);
683 SET_DIR(d, drip_bytes, b); SET_DIR(d, drip_ms, m);
684 } else if (strcmp(verb, "lossy") == 0) {
685 int d = dir_of(args);
686 SET_DIR(d, lossy_ppm, (int) (strtod(args, NULL) * 10000.0 + 0.5));
687 } else if (strcmp(verb, "corrupt") == 0) {
688 int d = dir_of(args);
689 SET_DIR(d, corrupt_ppm, (int) (strtod(args, NULL) * 10000.0 + 0.5));
690 } else if (strcmp(verb, "dup") == 0) {
691 int d = dir_of(args);
692 SET_DIR(d, dup_ppm, (int) (strtod(args, NULL) * 10000.0 + 0.5));
693 } else if (strcmp(verb, "reorder") == 0) {
694 int d = dir_of(args); double p = 0; int m = -1;
695 sscanf(args, "%lf %d", &p, &m);
696 SET_DIR(d, reorder_ppm, (int) (p * 10000.0 + 0.5));
697 if (m >= 0) { SET_DIR(d, reorder_ms, m); }
698 } else if (strcmp(verb, "truncate-at") == 0) {
699 int d = dir_of(args); SET_DIR(d, truncate_at, atol(args));
700 } else {
701 return 0;
702 }
703 return 1;
704}

707 * block) or toggle a persistent outage/hang state. Returns 1 if handled. */

721 } else if (strcmp(verb, "unhang") == 0) {
722 g_hang = 0;
723 } else if (strcmp(verb, "block") == 0) {
724 g_blocked = 1;
725 __atomic_add_fetch(&g_drop_epoch, 1, __ATOMIC_SEQ_CST);
726 } else if (strcmp(verb, "unblock") == 0) {
727 g_blocked = 0;
728 } else {
729 return 0;
730 }
731 return 1;

786/* Parse and apply one control command. `line` is mutated. `reply` (may be NULL)
787 * receives a human-readable response of up to `rsz` bytes. */
788static void
789apply_command(char *line, char *reply, size_t rsz)
790{
791 char verb[32] = "", args[224] = "";
792 sscanf(line, "%31s %223[^\n]", verb, args);
793 if (reply && rsz) {
794 snprintf(reply, rsz, "ok\n");
795 }
796
797 if (cmd_set_lever(verb, args) || cmd_set_epoch(verb)
798 || cmd_set_misc(verb, args)) {
799 return;
800 }
801 if (strcmp(verb, "status") == 0) {
802 cmd_status_report(reply, rsz);
803 } else if (verb[0] != '\0' && reply && rsz) {
804 snprintf(reply, rsz, "err: unknown command\n");
805 }
806}

808static void *
809control_thread(void *arg)
810{
811 int lfd = *(int *) arg;
812 free(arg);
813
814 for (;;) {
815 int cfd = accept(lfd, NULL, NULL);
816 if (cfd < 0) {
817 if (errno == EINTR) {
818 continue;
819 }
820 break;
821 }
822 char line[256];
823 ssize_t n = read(cfd, line, sizeof(line) - 1);
824 if (n > 0) {
825 line[n] = '\0';
826 char reply[768];
827 apply_command(line, reply, sizeof(reply));
828 (void) write_all(cfd, reply, (ssize_t) strlen(reply));
829 }
830 close(cfd);
831 }
832 return NULL;
833}

835/* Replay a '<seconds> <command>' timeline file (relative to start). */
836static void *
837script_thread(void *arg)
838{
839 FILE *fp = fopen((const char *) arg, "r");
840 if (fp == NULL) {
841 fprintf(stderr, "brix-fault-proxy: cannot open --script '%s'\n",
842 (const char *) arg);
843 return NULL;
844 }
845 char line[256];
846 double prev = 0.0;
847 while (fgets(line, sizeof(line), fp) != NULL) {
848 char *p = line;
849 while (*p == ' ' || *p == '\t') p++;
850 if (*p == '#' || *p == '\n' || *p == '\0') {
851 continue;
852 }
853 double t = 0.0;
854 int off = 0;
855 if (sscanf(p, "%lf %n", &t, &off) < 1) {
856 continue;
857 }
858 double dt = t - prev;
859 if (dt > 0) {
860 usleep((useconds_t) (dt * 1e6));
861 }
862 prev = t;
863 apply_command(p + off, NULL, 0);
864 }
865 fclose(fp);
866 return NULL;
867}

869/* Add one "host:port" (or a comma-separated list) to the target pool. */
870static int
871add_target(const char *spec)
872{
873 char tmp[512];
874 snprintf(tmp, sizeof(tmp), "%s", spec);
875 for (char *tok = strtok(tmp, ","); tok != NULL; tok = strtok(NULL, ",")) {
876 if (g_ntargets >= FP_MAX_TARGETS) {
877 fprintf(stderr, "brix-fault-proxy: too many --target (max %d)\n",
878 FP_MAX_TARGETS);
879 return -1;
880 }
881 const char *colon = strrchr(tok, ':');
882 if (colon == NULL || colon == tok || colon[1] == '\0') {
883 fprintf(stderr, "brix-fault-proxy: invalid --target '%s' "
884 "(expected HOST:PORT)\n", tok);
885 return -1;
886 }
887 size_t hlen = (size_t) (colon - tok);
888 if (hlen >= sizeof(g_targets[0].host)) {
889 return -1;
890 }
891 int p = atoi(colon + 1);
892 if (p <= 0 || p > 65535) {
893 fprintf(stderr, "brix-fault-proxy: invalid target port in '%s'\n", tok);
894 return -1;
895 }
896 memcpy(g_targets[g_ntargets].host, tok, hlen);
897 g_targets[g_ntargets].host[hlen] = '\0';
898 g_targets[g_ntargets].port = p;
899 g_ntargets++;
900 }
901 return 0;
902}

904static int
905sa_is_loopback(const struct sockaddr *sa)
906{
907 if (sa->sa_family == AF_INET) {
908 uint32_t a = ntohl(((const struct sockaddr_in *) sa)->sin_addr.s_addr);
909 return (a >> 24) == 127;
910 }
911 if (sa->sa_family == AF_INET6) {
912 return IN6_IS_ADDR_LOOPBACK(
913 &((const struct sockaddr_in6 *) sa)->sin6_addr) ? 1 : 0;
914 }
915 return 0;
916}

918/* Bind+listen on a copy of `tmpl` (family-agnostic) with `port` substituted. */
919static int
920listen_sa(const struct sockaddr_storage *tmpl, socklen_t slen, int port)
921{
922 struct sockaddr_storage ss = *tmpl;
923 if (ss.ss_family == AF_INET) {
924 ((struct sockaddr_in *) &ss)->sin_port = htons((uint16_t) port);
925 } else {
926 ((struct sockaddr_in6 *) &ss)->sin6_port = htons((uint16_t) port);
927 }
928 int fd = socket(ss.ss_family, SOCK_STREAM, 0);
929 if (fd < 0) {
930 return -1;
931 }
932 int on = 1;
933 setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on));
934 if (bind(fd, (struct sockaddr *) &ss, slen) != 0 || listen(fd, 64) != 0) {
935 close(fd);
936 return -1;
937 }
938 return fd;
939}

953/* Route an initial-lever long option (--latency, --block, …) through the same
954 * command parser as the live control port. Returns 1 if `opt` was a lever
955 * option (handled), 0 otherwise so the core option switch can claim it. */
956static int
957fp_apply_lever_opt(int opt, const char *optarg)
958{
959 static const struct { int code; const char *name; } lever[] = {
960 {1000, "latency"}, {1001, "jitter"}, {1002, "chunk"}, {1003, "drip"},
961 {1004, "lossy"}, {1005, "reorder"}, {1007, "corrupt"}, {1008, "dup"},
962 {1009, "rate"}, {1010, "truncate-at"}, {1011, "fail-nth"},
963 {1012, "heal-after"},
964 };
965 char cmd[256];
966 size_t i;
967
968 if (opt == 1006) { apply_command((char[]){"block"}, NULL, 0); return 1; }
969 if (opt == 1013) { apply_command((char[]){"hang"}, NULL, 0); return 1; }
970
971 for (i = 0; i < sizeof(lever) / sizeof(lever[0]); i++) {
972 if (lever[i].code == opt) {
973 snprintf(cmd, sizeof(cmd), "%s %s", lever[i].name, optarg);
974 apply_command(cmd, NULL, 0);
975 return 1;
976 }
977 }
978 return 0;
979}

981/* Apply a single core (non-lever) option to `cfg`. Returns FP_CONTINUE to keep
982 * parsing, or a terminal exit code (FP_OK for --help/--version, FP_USAGE on
983 * an unknown option or a rejected --target). */
984static int
985fp_apply_core_opt(int opt, fp_config *cfg)
986{
987 switch (opt) {
988 case 'l': cfg->listen_port = atoi(optarg); break;
989 case 't':
990 if (add_target(optarg) != 0) {
991 return FP_USAGE;
992 }
993 break;
994 case 'c': cfg->control_port = atoi(optarg); break;
995 case 'b': cfg->bind_str = optarg; break;
996 case 'I': cfg->insecure = 1; break;
997 case 'q': cfg->quiet = 1; break;
998 case 1014: g_seed = (unsigned) strtoul(optarg, NULL, 0); break;
999 case 1015: g_max_conns = atoi(optarg); break;
1000 case 1016: cfg->script_path = optarg; break;
1001 case 'h': usage(stdout); return FP_OK;
1002 case 'V':
1003 printf("brix-fault-proxy (BriX-Cache client) %s\n", brix_client_version());
1004 return FP_OK;
1005 default:
1006 usage(stderr);
1007 return FP_USAGE;
1008 }
1009 return FP_CONTINUE;
1010}

1018fp_finalize_config(int argc, char **argv, fp_config *cfg)

1038 fprintf(stderr, "brix-fault-proxy: --listen, --target and --control "

1040 usage(stderr);

1042 }
1043 return FP_CONTINUE;

1060 {"script", required_argument, NULL, 1016},
1061 {"quiet", no_argument, NULL, 'q'},
1062 {"latency", required_argument, NULL, 1000},
1063 {"jitter", required_argument, NULL, 1001},
1064 {"chunk", required_argument, NULL, 1002},
1065 {"drip", required_argument, NULL, 1003},
1066 {"lossy", required_argument, NULL, 1004},
1067 {"reorder", required_argument, NULL, 1005},
1068 {"block", no_argument, NULL, 1006},
1069 {"corrupt", required_argument, NULL, 1007},
1070 {"dup", required_argument, NULL, 1008},

1072 {"truncate-at", required_argument, NULL, 1010},
1073 {"fail-nth", required_argument, NULL, 1011},

1075 {"hang", no_argument, NULL, 1013},

1093/* Resolve `bind_str` into `bind_ss`/`bind_len` and enforce the loopback gate.
1094 * The control port is unauthenticated, so a non-loopback bind must be a
1095 * deliberate, explicit act (--insecure-bind). Returns FP_CONTINUE or FP_USAGE. */
1096static int
1097fp_setup_bind(const char *bind_str, int insecure,
1098 struct sockaddr_storage *bind_ss, socklen_t *bind_len)
1099{
1100 struct addrinfo hints, *bres = NULL;
1101
1102 memset(&hints, 0, sizeof(hints));
1103 hints.ai_family = AF_UNSPEC;
1104 hints.ai_socktype = SOCK_STREAM;
1105 hints.ai_flags = AI_PASSIVE | AI_NUMERICHOST;
1106 if (getaddrinfo(bind_str, "0", &hints, &bres) != 0 || bres == NULL) {
1107 fprintf(stderr, "brix-fault-proxy: invalid --bind address '%s'\n", bind_str);
1108 return FP_USAGE;
1109 }
1110 *bind_len = bres->ai_addrlen;
1111 memcpy(bind_ss, bres->ai_addr, bres->ai_addrlen);
1112 freeaddrinfo(bres);
1113
1114 /* Fail closed on a non-loopback bind unless the operator opts in. */
1115 if (!sa_is_loopback((struct sockaddr *) bind_ss) && !insecure) {
1116 fprintf(stderr, "brix-fault-proxy: refusing to bind the unauthenticated "
1117 "control port to non-loopback '%s' without "
1118 "--insecure-bind\n", bind_str);
1119 return FP_USAGE;
1120 }
1121 if (!sa_is_loopback((struct sockaddr *) bind_ss)) {
1122 fprintf(stderr, "brix-fault-proxy: WARNING binding control port to %s — "
1123 "the control plane is UNAUTHENTICATED\n", bind_str);
1124 }
1125 return FP_CONTINUE;
1126}

1128/* Accept clients on `lfd` forever, spawning a detached relay thread per
1129 * connection (subject to the outage/connection-cap levers). Returns FP_OK when
1130 * the accept loop terminates on a non-EINTR error. */
1131static int
1132fp_accept_loop(int lfd)
1133{
1134 for (;;) {
1135 int client = accept(lfd, NULL, NULL);
1136 if (client < 0) {
1137 if (errno == EINTR) {
1138 continue;
1139 }
1140 break;
1141 }
1142 if (g_blocked) {
1143 CBUMP(refused, 1);
1144 close(client); /* outage: refuse */
1145 continue;
1146 }
1147 if (g_max_conns > 0
1148 && __atomic_load_n(&C.active, __ATOMIC_RELAXED) >= (unsigned long) g_max_conns) {
1149 CBUMP(refused, 1);
1150 close(client); /* connection cap reached */
1151 continue;
1152 }
1153 int one = 1;
1154 setsockopt(client, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one));
1155 relay_arg *ra = malloc(sizeof(*ra));
1156 ra->client_fd = client;
1157 ra->epoch = __atomic_load_n(&g_drop_epoch, __ATOMIC_SEQ_CST);
