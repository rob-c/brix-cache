1/*
2 * brix_fault_proxy.c — brix-fault-proxy: a root-free TCP fault-injection proxy.
3 *
4 * WHAT: Relays TCP from a local listen port to an upstream host:port and injects
5 * network faults on demand — latency, jitter, partial writes, slow-drip,
6 * bandwidth caps, packet-loss (resets), payload corruption, duplication,
7 * deterministic mid-transfer truncation, connection black-holes, and full
8 * outages — set at startup on the command line or live over a control port
9 * while traffic flows. A "toxiproxy-lite" you can splice in front of any
10 * brix (or stock XRootD/HTTP) endpoint to prove a client survives a bad or
11 * actively hostile network, WITHOUT needing root: tc/netem needs
12 * CAP_NET_ADMIN; this is pure userspace sockets.
13 *
14 * DIRECTIONALITY: every byte-level lever is per-direction. `up` = client ->
15 * upstream (the request/upload path), `down` = upstream -> client (the
16 * response/download path). A control command with no direction token
17 * applies to BOTH; append ` up`, ` down`, or ` both` to target one.
18 *
19 * FAULT LEVERS (each off/0 by default; set at startup via the matching --flag or
20 * live by writing the command to the control port):
21 * latency <ms> [dir] delay every forwarded chunk (a high-RTT link)
22 * jitter <ms> [dir] delay each chunk by a uniform-random 0..<ms> ms —
23 * the application signature of IP reordering (see NOTE)
24 * chunk <bytes> [dir] split each write into <=N-byte segments (egress
25 * TCP_NODELAY) so the peer sees genuine partial reads
26 * drip <bytes> <ms> [dir] forward <bytes>, sleep <ms>, repeat (slow stream)
27 * rate <KB/s> [dir] pace the stream to a bandwidth ceiling (slow link)
28 * lossy <pct> [dir] with prob <pct>%/chunk, sever the stream (loss=reset)
29 * reorder <pct> [ms] [dir] with prob <pct>%/chunk, hold that chunk back <ms> ms
30 * corrupt <pct> [dir] flip a random bit in <pct>% of forwarded BYTES — a
31 * MITM tamper that a checksum-verifying peer must catch
32 * dup <pct> [dir] with prob <pct>%/chunk, deliver that chunk twice
33 * truncate-at <bytes> [dir] sever each connection once <bytes> have flowed in
34 * that direction (deterministic mid-transfer cut)
35 * fail-nth <n> fail exactly the Nth accepted connection, pass others
36 * heal-after <ms> auto-clear every lever after <ms> (a transient fault)
37 * one-shot self-heal (clear) the moment a sever/truncate fires
38 * drop graceful-FIN sever every live connection now
39 * reset abortive-RST sever every live connection now, and
40 * make subsequent lossy/truncate severs use RST too
41 * abortive <0|1> select RST (1) vs graceful FIN (0) for auto-severs
42 * half-close FIN the client->upstream path on live conns while the
43 * upstream->client path keeps flowing (tests half-open)
44 * hang accept new connections but never relay (a black hole)
45 * unhang resume relaying newly accepted connections
46 * block drop live conns AND refuse new ones (an outage window)
47 * unblock resume accepting
48 * clear reset every lever (levers=0, unblock, unhang)
49 * status report lever state and traffic/fault counters
50 *
51 * WHY: A client's resilience (reconnect + file-handle resumption, per-PDU read
52 * deadlines, low-speed aborts, checksum-verify-and-retry) needs the exact
53 * conditions of "bad wifi from a laptop abroad" — or an outright tampering
54 * MITM — injected deterministically and without privilege. The peer
55 * connects THROUGH this proxy; an operator (or a test) pulls the levers.
56 *
57 * HOW: Thread-per-connection relay (so per-chunk latency is a simple usleep); a
58 * global drop-epoch counter that, when bumped, makes every live relay thread
59 * sever and exit; a control thread parsing newline commands; an optional
60 * timeline thread replaying a scripted scenario. All randomness is per-thread
61 * rand_r() seeded from --seed for reproducibility.
62 *
63 * SECURITY: the control port is UNAUTHENTICATED — anyone who can reach it can pull
64 * the fault levers and reset live connections. Both the listen and control
65 * sockets therefore bind to loopback by DEFAULT; widening the bind address
66 * (--bind) exposes the control plane and is refused unless you also pass
67 * --insecure-bind, which prints a warning. Never expose the control port to
68 * an untrusted network.
69 *
70 * NOTE: packet LOSS is intentionally modelled as a connection sever, not byte
71 * drops: dropping bytes from an already-ACKed TCP stream corrupts it rather
72 * than emulating loss (which lives below TCP). Likewise true out-of-order
73 * PACKET delivery cannot be emulated by re-ordering bytes — TCP reassembles
74 * in order below us — so `jitter`/`reorder` inject the variable latency that
75 * IP reordering actually imposes on a TCP application. (True packet
76 * reordering needs `tc qdisc ... netem`, i.e. CAP_NET_ADMIN, which this
77 * avoids.) `corrupt`, by contrast, IS a real in-band byte mutation: it is
78 * what an active on-path attacker (or a flaky NIC past the TCP checksum)
79 * does, and it is precisely what an application-layer checksum exists to
80 * catch.
81 */

348/* Blocking connect to host:port (best-effort, first address that works). */
349static int
350dial(const char *host, int port)
351{
352 struct addrinfo hints, *res = NULL, *ai;
353 char portstr[16];
354 int fd = -1;
355
356 memset(&hints, 0, sizeof(hints));
357 hints.ai_family = AF_UNSPEC;
358 hints.ai_socktype = SOCK_STREAM;
359 snprintf(portstr, sizeof(portstr), "%d", port);
360 if (getaddrinfo(host, portstr, &hints, &res) != 0) {
361 return -1;
362 }
363 for (ai = res; ai != NULL; ai = ai->ai_next) {
364 fd = socket(ai->ai_family, ai->ai_socktype, ai->ai_protocol);
365 if (fd < 0) {
366 continue;
367 }
368 if (connect(fd, ai->ai_addr, ai->ai_addrlen) == 0) {
369 break;
370 }
371 close(fd);
372 fd = -1;
373 }
374 freeaddrinfo(res);
375 return fd;
376}

440/* Clamp the outgoing segment to the chunk `piece` and, if truncation is armed,
441 * to the exact remaining distance to the cut. Returns the clamped length, or -1
442 * when the truncation point is already reached (caller must sever). */
443static ssize_t
444fault_clamp_seg(ssize_t seg, int piece, long trunc, unsigned long conn_ctr)
445{
446 if (seg > piece) {
447 seg = piece;
448 }
449 /* Cap at the truncation boundary so the cut is exact rather than firing
450 * only after the whole (possibly 64 KiB) read is delivered. */
451 if (trunc > 0) {
452 long remaining = trunc - (long) conn_ctr;
453 if (remaining <= 0) {
454 return -1;
455 }
456 if (seg > remaining) {
457 seg = remaining;
458 }
459 }
460 return seg;
461}

510 g_gr_tokens = 0.0;
511 g_gr_init = 1;
512 }
513 double elapsed = (now.tv_sec - g_gr_last.tv_sec)
514 + (now.tv_nsec - g_gr_last.tv_nsec) / 1e9;
515 g_gr_tokens += elapsed * rate;
516 if (g_gr_tokens > rate) {
517 g_gr_tokens = rate; /* cap the burst at ~1s of credit */
518 }
519 g_gr_last = now;
520 g_gr_tokens -= (double) seg;
521 double deficit = g_gr_tokens < 0.0 ? -g_gr_tokens : 0.0;
522 pthread_mutex_unlock(&g_gr_lock);
523 if (deficit > 0.0) {
524 usleep((useconds_t) (deficit / rate * 1e6));
525 }
526}

528/* Deliver one segment starting at buf+off, applying `L`'s active faults. On
529 * success writes the delivered length to *wrote and returns 0; returns -1 when
530 * the stream should be severed (truncate cut, lossy drop, write error) or
531 * silently dropped (block/epoch change). */
532static int
533forward_segment(int to, char *buf, ssize_t off, ssize_t n, unsigned epoch,
534 const lever_t *L, unsigned *seed, unsigned long *conn_ctr,
535 unsigned long *glob_ctr, int piece, ssize_t *wrote)
536{
537 ssize_t seg = fault_clamp_seg(n - off, piece, L->truncate_at, *conn_ctr);
538 if (seg < 0) {
539 return fault_sever();
540 }
541
542 fault_delays(seg, seed, L);
543
544 if (L->lossy_ppm > 0 && (int) (rand_r(seed) % 1000000u) < L->lossy_ppm) {
545 return fault_sever(); /* application-visible "loss" = sever the stream */
546 }
547 if (g_blocked || g_drop_epoch != epoch) {
548 return -1;
549 }
550
551 fault_corrupt(buf, off, seg, seed, L->corrupt_ppm);
552 global_rate_gate(seg);
553
554 if (write_all(to, buf + off, seg) != 0) {
555 return -1;
556 }
557 if (L->dup_ppm > 0 && (int) (rand_r(seed) % 1000000u) < L->dup_ppm) {
558 (void) write_all(to, buf + off, seg); /* duplicate delivery */
559 CBUMP(dups, 1);
560 }
561
562 CBUMP2(conn_ctr, glob_ctr, seg);
563 *wrote = seg;
564
565 if (L->truncate_at > 0 && (long) *conn_ctr >= L->truncate_at) {
566 return fault_sever(); /* deterministic mid-transfer cut */
567 }
568 return 0;
569}

571/*
572 * Forward n bytes to `to`, applying direction `L`'s active fault levers. Bytes
573 * may be mutated in place (corruption). Returns 0 on success, -1 if the
574 * connection should be severed (write error, a lossy drop, or a truncate cut).
575 * `seed` is this thread's private rand_r() state; `conn_ctr` the per-connection
576 * per-direction running byte total (for truncate-at); `glob_ctr` the process-wide
577 * traffic counter for this direction.
578 */
579static int
580forward_faulted(int to, char *buf, ssize_t n, unsigned epoch, volatile lever_t *L,
581 unsigned *seed, unsigned long *conn_ctr, unsigned long *glob_ctr)
582{
583 /* Snapshot the levers once so a mid-buffer control-plane change can't split
584 * this read across two fault configurations (matches the original). */
585 lever_t snap = *L;
586 int piece = (snap.drip_bytes > 0) ? snap.drip_bytes
587 : (snap.chunk_bytes > 0 ? snap.chunk_bytes : (int) n);
588 ssize_t off = 0;
589
590 if (piece <= 0) {
591 piece = (int) n; /* n >= 1 (read returned >0) */
592 }
593 if (snap.latency_ms > 0) {
594 usleep((useconds_t) snap.latency_ms * 1000);
595 }
596
597 while (off < n) {
598 ssize_t wrote = 0;
599 if (forward_segment(to, buf, off, n, epoch, &snap, seed,
600 conn_ctr, glob_ctr, piece, &wrote) != 0) {
601 return -1;
602 }
603 off += wrote;
604 if (snap.drip_bytes > 0 && off < n) {
605 usleep((useconds_t) snap.drip_ms * 1000);
606 }
607 }
608 return 0;
609}

617/* Pre-dial dispositions that answer the client without ever reaching upstream:
618 * the fail-nth sever and the hang/black-hole hold. Returns 1 if the connection
619 * was handled (closed + CDEC), 0 to proceed with dialling upstream. */

622{
623 /* fail-nth: sever exactly the Nth accepted connection, then pass the rest. */
624 if (g_fail_nth > 0 && conn_id == (unsigned long) g_fail_nth) {

638 }
639 close(cfd);
640 CDEC(active);
641 return 1;
642 }
643 return 0;

698/* Length-prefix lie: if the armed big-endian uint32 offset lies wholly within
699 * this buffer, rewrite it (set/add/sub) — a framing attack on binary protocols. */
700static void
701mangle_apply(int is_up, char *buf, ssize_t nr, unsigned long base)
702{
703 struct fp_mangle *M = is_up ? &g_mangle_up : &g_mangle_down;
704 if (!M->active) {
705 return;
706 }
707 long off = M->offset;
708 if (off < (long) base || off + 4 > (long) base + nr) {
709 return; /* the 4-byte field is not fully contained in this read */
710 }
711 unsigned char *p = (unsigned char *) buf + (off - (long) base);
712 unsigned long v = ((unsigned long) p[0] << 24) | ((unsigned long) p[1] << 16)
713 | ((unsigned long) p[2] << 8) | (unsigned long) p[3];
714 long nv = (M->op == 0) ? M->val
715 : (M->op == 1) ? (long) v + M->val
716 : (long) v - M->val;
717 unsigned long u = (unsigned long) nv & 0xFFFFFFFFUL;
718 p[0] = (unsigned char) (u >> 24); p[1] = (unsigned char) (u >> 16);
719 p[2] = (unsigned char) (u >> 8); p[3] = (unsigned char) u;
720 CBUMP(mangled, 1);
721}

776/* Relay one poll-ready direction through the fault engine. Returns 0 to keep
777 * looping, 1 on EOF/read error (caller closes both ends), 2 if a fault severed
778 * the pair (already closed + CDEC, caller just returns). */
779static int
780relay_pump_dir(int i, struct pollfd *pfd, int cfd, int ufd,
781 char *buf, size_t bufsz, unsigned char *scratch,
782 unsigned char *scratch2, unsigned epoch,
783 unsigned *seed, unsigned long *up_ctr, unsigned long *down_ctr,
784 int *firstflag)
785{
786 if (!(pfd[i].revents & (POLLIN | POLLHUP | POLLERR))) {
787 return 0;
788 }
789 int from = pfd[i].fd;
790 int to = pfd[i ^ 1].fd;
791 ssize_t nr = read(from, buf, bufsz);
792 if (nr <= 0) {
793 return 1;
794 }
795 int is_up = (i == 0);
796 volatile lever_t *L = is_up ? &g_up : &g_down;
797 unsigned long *conn_ctr = is_up ? up_ctr : down_ctr;
798
799 /* Stateful, content-addressed faults on the RAW bytes (before any mutation
800 * changes their length/offset). */
801 trig_check(is_up, buf, nr);
802 mangle_apply(is_up, buf, nr, *conn_ctr);
803
804 /* delay-first: hold back only the opening chunk of this direction. */
805 if (L->delayfirst_ms > 0 && *firstflag) {
806 usleep((useconds_t) L->delayfirst_ms * 1000);
807 }
808 *firstflag = 0;
809
810 /* Transform chain: each length-changing stage ping-pongs between scratch A
811 * and B so it never overwrites its own input — TLS record surgery, then HTTP
812 * request smuggling, then byte-level MITM mutation. */
813 const char *cur = buf;
814 ssize_t curn = nr;
815 unsigned char *dst = scratch, *alt = scratch2;
816
817 if (fp_tls_active(is_up ? &g_tls_up : &g_tls_down)) {
818 curn = (ssize_t) apply_tls(is_up, cur, curn, dst);
819 cur = (const char *) dst;
820 FP_PSWAP(dst, alt);
821 }
822 if (fp_http_active(is_up ? &g_http_up : &g_http_down)) {
823 int applied = 0;
824 size_t on = apply_http(is_up, cur, curn, dst, &applied);
825 if (applied) {
826 curn = (ssize_t) on;
827 cur = (const char *) dst;
828 FP_PSWAP(dst, alt);
829 }
830 }
831
832 fp_ext_mut mut;
833 unsigned char fbuf[128], rbuf[256], ibuf[512];
834 if (ext_snapshot(is_up, L, &mut, fbuf, rbuf, ibuf)) {
835 fp_ext_stats st = { 0, 0, 0, 0 };
836 size_t on = fp_ext_mutate((const unsigned char *) cur, (size_t) curn,
837 dst, FP_SCRATCH, &mut, seed, &st);
838 curn = (ssize_t) on;
839 cur = (const char *) dst;
840 FP_PSWAP(dst, alt);
841 if (st.dropped) { CBUMP(dropped, st.dropped); }
842 if (st.repeated) { CBUMP(repeated, st.repeated); }
843 if (st.injected) { CBUMP(injected, st.injected); }
844 if (st.replaced) { CBUMP(replaced, st.replaced); }
845 if (curn == 0) {
846 return 0; /* every byte was dropped — nothing to forward */
847 }
848 }
849
850 /* Session recording: capture exactly what we are about to forward. */
851 if (fp_replay_recording()) {
852 fp_replay_record(is_up, now_ms_since(g_t0),
853 (const unsigned char *) cur, (size_t) curn);
854 CBUMP(recorded, (unsigned long) curn);
855 }
856
857 unsigned long *glob_ctr = is_up ? &C.up_bytes : &C.down_bytes;
858 if (forward_faulted(to, (char *) cur, curn, epoch, L, seed,
859 conn_ctr, glob_ctr) != 0) {
860 sever(cfd, ufd, g_abortive);
861 CDEC(active);
862 return 2;
863 }
864 return 0;
865}

868/* Bidirectional relay loop: shuttle bytes each way through the fault engine
869 * until EOF, a control-plane sever, or a poll error. Closes both ends + CDEC
870 * before returning. */
871static void
872relay_pump(int cfd, int ufd, unsigned epoch, unsigned seed)
873{
874 struct pollfd pfd[2];
875 pfd[0].fd = cfd; /* client -> upstream (up) */
876 pfd[1].fd = ufd; /* upstream -> client (down) */
877 char buf[65536];
878 /* Two ping-pong transform buffers (uninit; only written) so the TLS/HTTP/MITM
879 * stages can chain without a stage overwriting its own input. Plain stack
880 * locals (not _Thread_local — avoids the TLS zero-init latency). */
881 unsigned char scratch[FP_SCRATCH];
882 unsigned char scratch2[FP_SCRATCH];
883 unsigned long up_ctr = 0, down_ctr = 0;
884 unsigned hc_epoch = g_halfclose_epoch;
885 int hc_done = 0;
886 int first[2] = { 1, 1 };
887 struct timespec t0;
888 clock_gettime(CLOCK_MONOTONIC, &t0);
889
890 for (;;) {
891 if (g_blocked || g_drop_epoch != epoch) {
892 sever(cfd, ufd, g_abortive);
893 CDEC(active);
894 return;
895 }
896 /* max-lifetime: guillotine a connection that has lived too long. */
897 if (g_max_life_ms > 0) {
898 struct timespec now;
899 clock_gettime(CLOCK_MONOTONIC, &now);
900 long ms = (now.tv_sec - t0.tv_sec) * 1000
901 + (now.tv_nsec - t0.tv_nsec) / 1000000;
902 if (ms >= g_max_life_ms) {
903 CBUMP(severs, 1);
904 sever(cfd, ufd, g_abortive);
905 CDEC(active);
906 return;
907 }
908 }
909 /* half-close: FIN the up path, keep the down path flowing. */
910 if (!hc_done && g_halfclose_epoch != hc_epoch) {
911 shutdown(cfd, SHUT_RD);
912 shutdown(ufd, SHUT_WR);
913 hc_done = 1;
914 }
915 /* stall: stop reading a direction so the peer's send buffer fills
916 * (real TCP backpressure) without ever severing. */
917 pfd[0].events = (hc_done || g_stall_up) ? 0 : POLLIN;
918 pfd[1].events = g_stall_down ? 0 : POLLIN;
919 int pr = poll(pfd, 2, 100);
920 if (pr < 0) {
921 if (errno == EINTR) {
922 continue;
923 }
924 break;
925 }
926 if (pr == 0) {
927 continue; /* re-check fault flags */
928 }
929 for (int i = 0; i < 2; i++) {
930 int r = relay_pump_dir(i, pfd, cfd, ufd, buf, sizeof(buf), scratch,
931 scratch2, epoch, &seed, &up_ctr, &down_ctr,
932 &first[i]);
933 if (r == 2) {
934 return; /* severed + CDEC already done */
935 }
936 if (r == 1) {
937 goto done; /* EOF */
938 }
939 }
940 }
941done:
942 close(cfd);
943 close(ufd);
944 CDEC(active);
945}

1882/* Binary-search the smallest integer value of `lever` in [lo,hi] for which the
1883 * oracle command FAILS (non-zero exit = the bug reproduces), assuming severity
1884 * is monotonic in the lever. Publishes progress + the boundary to g_bisect_result. */
1885static void *
1886bisect_thread(void *arg)
1887{
1888 struct bisect_arg *b = arg;
1889 long lo = b->lo, hi = b->hi, found = b->hi + 1;
1890 int probes = 0;
1891 while (lo <= hi) {
1892 long mid = lo + (hi - lo) / 2;
1893 char cmd[48];
1894 snprintf(cmd, sizeof(cmd), "%s %ld", b->lever, mid);
1895 apply_command(cmd, NULL, 0);
1896 int rc = fp_oracle_run(b->cmd, b->timeout_ms);
1897 probes++;
1898 res_set(g_bisect_result, "running: %s probe#%d val=%ld rc=%d",
1899 b->lever, probes, mid, rc);
1900 if (rc == -2) {
1901 res_set(g_bisect_result, "error: oracle disabled (--enable-exec)");
1902 break;
1903 }
1904 if (rc != 0) { /* reproduced (fail or inconclusive) → go smaller */
1905 found = mid;
1906 hi = mid - 1;
1907 } else { /* survived → need a harsher value */
1908 lo = mid + 1;
1909 }
1910 }
1911 apply_command((char[]){"clear"}, NULL, 0);
1912 if (found <= b->hi) {
1913 res_set(g_bisect_result, "done: minimal %s=%ld reproduces (%d probes)",
1914 b->lever, found, probes);
1915 } else {
1916 res_set(g_bisect_result, "done: no %s in [%ld,%ld] reproduced (%d probes)",
1917 b->lever, b->lo, b->hi, probes);
1918 }
1919 g_oracle_busy = 0;
1920 free(b);
1921 return NULL;
1922}

1924/* Apply a fault for hold_ms, clear it, then poll a health probe until it passes
1925 * or times out — asserting the service recovers. Publishes to g_recovery_result. */
1926static void *
1927recovery_thread(void *arg)
1928{
1929 struct recov_arg *r = arg;
1930 char fcmd[256];
1931 snprintf(fcmd, sizeof(fcmd), "%s", r->fault);
1932 apply_command(fcmd, NULL, 0);
1933 res_set(g_recovery_result, "running: fault held %dms", r->hold_ms);
1934 usleep((useconds_t) r->hold_ms * 1000);
1935 apply_command((char[]){"clear"}, NULL, 0);
1936 struct timespec t0;
1937 clock_gettime(CLOCK_MONOTONIC, &t0);
1938 int elapsed = 0;
1939 for (;;) {
1940 int rc = fp_oracle_run(r->probe, 2000);
1941 if (rc == -2) {
1942 res_set(g_recovery_result, "error: oracle disabled (--enable-exec)");
1943 break;
1944 }
1945 elapsed = (int) now_ms_since(t0);
1946 if (rc == 0) {
1947 res_set(g_recovery_result, "done: recovered in %dms", elapsed);
1948 break;
1949 }
1950 if (elapsed >= r->timeout_ms) {
1951 res_set(g_recovery_result, "done: STUCK — no recovery in %dms", elapsed);
1952 break;
1953 }
1954 usleep(250 * 1000);
1955 }
1956 g_oracle_busy = 0;
1957 free(r);
1958 return NULL;
1959}

2549/* Accept clients on `lfd` forever, spawning a detached relay thread per
2550 * connection (subject to the outage/connection-cap levers). Returns FP_OK when
2551 * the accept loop terminates on a non-EINTR error. */
2552static int
2553fp_accept_loop(int lfd)
2554{
2555 for (;;) {
2556 /* accept-pause: delay servicing the accept queue so pending SYNs / the
2557 * accept backlog pile up (connect-timeout + backlog-overflow testing). */
2558 if (g_accept_pause_ms > 0) {
2559 usleep((useconds_t) g_accept_pause_ms * 1000);
2560 }
2561 int client = accept(lfd, NULL, NULL);
2562 if (client < 0) {
2563 if (errno == EINTR) {
2564 continue;
2565 }
2566 break;
2567 }
2568 if (g_blocked) {
2569 CBUMP(refused, 1);
2570 close(client); /* outage: refuse */
2571 continue;
2572 }
2573 if (g_max_conns > 0
2574 && __atomic_load_n(&C.active, __ATOMIC_RELAXED) >= (unsigned long) g_max_conns) {
2575 CBUMP(refused, 1);
2576 close(client); /* connection cap reached */
2577 continue;
2578 }
2579 int one = 1;
2580 setsockopt(client, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one));
2581 relay_arg *ra = malloc(sizeof(*ra));
2582 ra->client_fd = client;
2583 ra->epoch = __atomic_load_n(&g_drop_epoch, __ATOMIC_SEQ_CST);
2584 ra->conn_id = CBUMP(conns, 1);
2585 CBUMP(active, 1);
2586 pthread_t t;
2587 if (pthread_create(&t, NULL, relay_thread, ra) != 0) {
2588 close(client);
2589 CDEC(active);
2590 free(ra);
2591 continue;
2592 }
2593 pthread_detach(t);
2594 }
2595 return FP_OK;
2596}
