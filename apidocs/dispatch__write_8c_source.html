15/* manager_redirect_mutation — Plane B manager orchestration.
16 *
17 * In manager_mode, a path-based namespace mutation (mkdir/rm/rmdir/mv/
18 * chmod/truncate) is redirected to the data node that should serve the
19 * path, mirroring the existing open-write redirect. The client re-issues
20 * the op to that node, which executes it against real storage and enforces
21 * authz. Data writes (write/pgwrite/writev/sync/chkpoint) are NOT
22 * redirected here — they ride an open handle already redirected at open.
23 * WHY: without this, a mutation sent to a manager executes against the manager's
24 * (empty) local export instead of the cluster — silently wrong. This is
25 * the nginx-correct orchestration: per-worker node CMS connections make a
26 * cross-worker CMS fan-out impractical, whereas redirect needs no shared
27 * connection state. (CMS forwarding remains the path for multi-replica
28 * fan-out via brix_cms_forward_to_node + the node executor.)
29 * HOW: the session layer guarantees the client is logged in before write
30 * dispatch, so this runs post-auth; it deliberately bypasses the local
31 * write-permission gate (the node enforces it), matching the open redirect.
32 *
33 * Returns BRIX_DISPATCH_CONTINUE to fall through to local execution, or a
34 * send-result (redirect / error) the caller returns directly.
35 */
36static ngx_int_t
37manager_redirect_mutation(brix_ctx_t *ctx, ngx_connection_t *c,
38 ngx_stream_brix_srv_conf_t *conf)
39{
40 char path[BRIX_MAX_PATH + 1];
41 char host[256];
42 uint16_t port;
43
44 if (!conf->manager_mode) {
45 return BRIX_DISPATCH_CONTINUE;
46 }
47
48 switch (ctx->recv.cur_reqid) {
49 case kXR_mkdir:
50 case kXR_rm:
51 case kXR_rmdir:
52 case kXR_mv:
53 case kXR_chmod:
54 case kXR_truncate:
55 break;
56 default:
57 return BRIX_DISPATCH_CONTINUE;
58 }
59
60 /* Handle-based ops (e.g. truncate by fhandle) carry no path payload and must
61 * not be redirected — let them process locally / fail naturally. */
62 if (ctx->recv.cur_dlen == 0
63 || brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
64 path, sizeof(path), 1) <= 0
65 || path[0] != '/')
66 {
67 return BRIX_DISPATCH_CONTINUE;
68 }
69
70 if (brix_srv_select_or_blacklisted(path, 1, host, sizeof(host), &port)) {
71 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
72 "brix: manager redirect mutation \"%s\" -> %s",
73 path, host);
74 return brix_send_redirect(ctx, c, host, port);
75 }
76
77 return brix_send_error(ctx, c, kXR_FSError,
78 "no data server available for path");
79}

81/* brix_dispatch_write_opcode — phase 2 routing (from handshake/dispatch.c) for
82 * the mutating opcodes: a switch on ctx->recv.cur_reqid over data writes (kXR_write,
83 * kXR_pgwrite CRC32c-checked, kXR_writev), kXR_sync, structure changes
84 * (truncate/mkdir/rm/rmdir), mv/chmod, and kXR_chkpoint. Each case passes the
85 * DISPATCH_WR gate — brix_dispatch_require_write, stricter than require_auth
86 * (needs auth AND conf->common.allow_write) — before its handler. In manager_mode,
87 * path-based mutations redirect first (manager_redirect_mutation). Returns the
88 * handler result, or BRIX_DISPATCH_CONTINUE when not a write opcode. */
89ngx_int_t
90brix_dispatch_write_opcode(brix_ctx_t *ctx, ngx_connection_t *c,
91 ngx_stream_brix_srv_conf_t *conf)
92{
93 ngx_int_t rc;
94
95 /* Plane B: in manager mode, redirect path-based mutations to a data node
96 * before the local write gate (which the manager may not satisfy). */
97 rc = manager_redirect_mutation(ctx, c, conf);
98 if (rc != BRIX_DISPATCH_CONTINUE) {
99 return rc;
100 }
101
102 switch (ctx->recv.cur_reqid) {
103
104 case kXR_write: DISPATCH_WR(brix_handle_write);
105 case kXR_pgwrite: DISPATCH_WR(brix_handle_pgwrite);
106 case kXR_sync: DISPATCH_WR(brix_handle_sync);
107 case kXR_truncate: DISPATCH_WR(brix_handle_truncate, conf);
108 case kXR_mkdir: DISPATCH_WR(brix_handle_mkdir, conf);
109 case kXR_rm: DISPATCH_WR(brix_handle_rm, conf);
110 case kXR_writev: DISPATCH_WR(brix_handle_writev);
111 case kXR_rmdir: DISPATCH_WR(brix_handle_rmdir, conf);
112 case kXR_mv: DISPATCH_WR(brix_handle_mv, conf);
113 case kXR_chmod: DISPATCH_WR(brix_handle_chmod, conf);
114 case kXR_chkpoint: DISPATCH_WR(brix_handle_chkpoint, conf);
115
116 /* vendor POSIX-completeness extensions (capability "xrdfs.ext") */
117 case kXR_setattr: DISPATCH_WR(brix_handle_setattr, conf);
118 case kXR_symlink: DISPATCH_WR(brix_handle_symlink, conf);
119 case kXR_link: DISPATCH_WR(brix_handle_link, conf);
120
121 default:
122 return BRIX_DISPATCH_CONTINUE;
123 }
124}
