37/*
38 * Phase 51 (A3): per-source-IP connection accounting (per-worker), so one peer
39 * cannot consume every slot of the global cap. A small open-addressing table
40 * keyed by the peer IP string; linear-probe. If the table is full (an abusive
41 * fan-out of distinct IPs — itself bounded by the global cap) we fail OPEN: the
42 * global cap still bounds the total, we just stop enforcing per-IP for the
43 * overflow. Lock-free: per-worker, event-loop only.
44 */
45#define BRIX_CMS_IP_SLOTS 4096
46
47typedef struct {
48 char ip[64]; /* peer IP string; '\0' = free slot */
49 ngx_uint_t count; /* live connections from this IP in this worker */
50} brix_cms_ip_slot_t;
51
52static brix_cms_ip_slot_t brix_cms_ip_table[BRIX_CMS_IP_SLOTS];
53
54static ngx_uint_t
55brix_cms_ip_hash(const char *ip)
56{
57 ngx_uint_t h = 5381;
58 const u_char *p = (const u_char *) ip;
59
60 while (*p) {
61 h = ((h << 5) + h) ^ (ngx_uint_t) *p++; /* djb2-xor */
62 }
63 return h % BRIX_CMS_IP_SLOTS;
64}

66/* Find the slot for ip; if create, claim a free slot when absent. Returns NULL
67 * when absent and (not create OR table full). Probes a bounded window. */
68static brix_cms_ip_slot_t *
69brix_cms_ip_find(const char *ip, int create)
70{
71 ngx_uint_t start = brix_cms_ip_hash(ip);
72 ngx_uint_t i;
73
74 for (i = 0; i < BRIX_CMS_IP_SLOTS; i++) {
75 brix_cms_ip_slot_t *slot =
76 &brix_cms_ip_table[(start + i) % BRIX_CMS_IP_SLOTS];
77
78 if (slot->ip[0] == '\0') {
79 if (!create) {
80 return NULL; /* absent */
81 }
82 ngx_cpystrn((u_char *) slot->ip, (u_char *) ip, sizeof(slot->ip));
83 slot->count = 0;
84 return slot;
85 }
86 if (ngx_strcmp(slot->ip, ip) == 0) {
87 return slot; /* found */
88 }
89 }
90 return NULL; /* table full */
91}

121/*
122 * server_handler.c — CMS server connection handler
123 *
124 * WHAT: Accepts TCP connections from the XRootD CMS manager and maintains a
125 * persistent heartbeat session. Each accepted connection is assigned a
126 * read/write handler pair that exchanges periodic load reports.
127 *
128 * WHY: The CMS manager needs to know which data servers are alive, how much
129 * free space they have, and their utilisation percentage so it can route
130 * client requests (kXR_locate / kXR_redirect) to the best server.
131 *
132 * HOW: On accept we allocate a cms_srv_ctx_t on the connection pool, set up
133 * ping_timer for periodic heartbeats, assign read/write handlers,
134 * and immediately arm the first read event. The read handler (recv.c)
135 * dispatches incoming frames; the write handler (send.c) fires load
136 * reports at conf->interval_ms intervals.
137 */
138
139void
140brix_cms_srv_handler(ngx_stream_session_t *s)
141/*
142 *
143 * WHAT: Entry point for CMS server connections accepted by the stream module.
144 * Allocates context, sets up handlers and timer, arms first read.
145 *
146 * WHY: The CMS manager connects to this port as a data-server client. We need
147 * a per-connection state object (ctx) that tracks the connection pointer,
148 * how many bytes remain in the current frame header, the peer host string,
149 * and the heartbeat timer.
150 *
151 * HOW: 1. pcalloc ctx on c->pool (auto-cleaned on pool destruction).
152 * 2. Initialise ctx fields: c pointer, hdr-in_need = CMS_HDR_LEN,
153 * host from ngx_sock_ntop, interval_ms from conf.
154 * 3. Set ping_timer log/data, assign c->data = ctx.
155 * 4. Replace default read/write handlers with cms_srv_read/cms_srv_write.
156 * 5. Log debug line and arm first read.
157 */
158
159{
160 ngx_connection_t *c;
161 brix_cms_srv_ctx_t *ctx;
162 ngx_stream_brix_cms_srv_conf_t *conf;
163 ngx_stream_brix_srv_conf_t *root_conf;
164 size_t len;
165
166 c = s->connection;
167
168 ctx = ngx_pcalloc(c->pool, sizeof(brix_cms_srv_ctx_t));
169 if (ctx == NULL) {
170 ngx_stream_finalize_session(s, NGX_STREAM_INTERNAL_SERVER_ERROR);
171 return;
172 }
173
174 ctx->c = c;
175 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
176
177 len = ngx_sock_ntop(c->sockaddr, c->socklen,
178 (u_char *) ctx->host, sizeof(ctx->host) - 1, 0);
179 ctx->host[len] = '\0';
180
181 conf = ngx_stream_get_module_srv_conf(s, ngx_stream_brix_cms_srv_module);
182 ctx->conf = conf;
183 ctx->interval_ms = (ngx_msec_t) conf->interval * 1000;
184 if (ctx->interval_ms < 1000) {
185 /* Never arm a 0/sub-1s self-rearming ping timer: interval 0 → 0ms timer
186 * → epoll_wait(timeout=0) busy-loop pinning the worker. Floor at 1s. */
187 ctx->interval_ms = 1000;
188 }
189 ctx->login_timeout_ms = conf->login_timeout; /* WS3 */
190 ctx->idle_timeout_ms = conf->idle_timeout; /* WS3 */
191
192 /*
193 * WS4 — admission cap. Reject before installing any frame handler so an
194 * abusive peer (inside the often-unset CIDR allowlist) cannot exhaust
195 * memory/fds with idle connections. 0 = unlimited (back-compat).
196 */

199 {

203 ctx->host, conf->max_connections);
204 BRIX_RESIL_METRIC_INC(cms_cap_rejections_total);
205 ngx_stream_finalize_session(s, NGX_STREAM_FORBIDDEN);
206 return;
207 }

211 * the global cap above. 0 = disabled (back-compat).

283 * cms_srv_complete_login. 0 = disabled.
284 */

286 ngx_add_timer(c->read, ctx->login_timeout_ms);
287 }
288

292 brix_cms_srv_read(c->read);
