36/*
37 * Phase 51 (A3): per-source-IP connection accounting (per-worker), so one peer
38 * cannot consume every slot of the global cap. A small open-addressing table
39 * keyed by the peer IP string; linear-probe. If the table is full (an abusive
40 * fan-out of distinct IPs — itself bounded by the global cap) we fail OPEN: the
41 * global cap still bounds the total, we just stop enforcing per-IP for the
42 * overflow. Lock-free: per-worker, event-loop only.
43 */
44#define BRIX_CMS_IP_SLOTS 4096
45
46typedef struct {
47 char ip[64]; /* peer IP string; '\0' = free slot */
48 ngx_uint_t count; /* live connections from this IP in this worker */
49} brix_cms_ip_slot_t;
50
51static brix_cms_ip_slot_t brix_cms_ip_table[BRIX_CMS_IP_SLOTS];
52
53static ngx_uint_t
54brix_cms_ip_hash(const char *ip)
55{
56 ngx_uint_t h = 5381;
57 const u_char *p = (const u_char *) ip;
58
59 while (*p) {
60 h = ((h << 5) + h) ^ (ngx_uint_t) *p++; /* djb2-xor */
61 }
62 return h % BRIX_CMS_IP_SLOTS;
63}

65/* Find the slot for ip; if create, claim a free slot when absent. Returns NULL
66 * when absent and (not create OR table full). Probes a bounded window. */
67static brix_cms_ip_slot_t *
68brix_cms_ip_find(const char *ip, int create)
69{
70 ngx_uint_t start = brix_cms_ip_hash(ip);
71 ngx_uint_t i;
72
73 for (i = 0; i < BRIX_CMS_IP_SLOTS; i++) {
74 brix_cms_ip_slot_t *slot =
75 &brix_cms_ip_table[(start + i) % BRIX_CMS_IP_SLOTS];
76
77 if (slot->ip[0] == '\0') {
78 if (!create) {
79 return NULL; /* absent */
80 }
81 ngx_cpystrn((u_char *) slot->ip, (u_char *) ip, sizeof(slot->ip));
82 slot->count = 0;
83 return slot;
84 }
85 if (ngx_strcmp(slot->ip, ip) == 0) {
86 return slot; /* found */
87 }
88 }
89 return NULL; /* table full */
90}

120/*
121 * server_handler.c — CMS server connection handler
122 *
123 * WHAT: Accepts TCP connections from the XRootD CMS manager and maintains a
124 * persistent heartbeat session. Each accepted connection is assigned a
125 * read/write handler pair that exchanges periodic load reports.
126 *
127 * WHY: The CMS manager needs to know which data servers are alive, how much
128 * free space they have, and their utilisation percentage so it can route
129 * client requests (kXR_locate / kXR_redirect) to the best server.
130 *
131 * HOW: On accept we allocate a cms_srv_ctx_t on the connection pool, set up
132 * ping_timer for periodic heartbeats, assign read/write handlers,
133 * and immediately arm the first read event. The read handler (recv.c)
134 * dispatches incoming frames; the write handler (send.c) fires load
135 * reports at conf->interval_ms intervals.
136 */
137
138void
139brix_cms_srv_handler(ngx_stream_session_t *s)
140/*
141 *
142 * WHAT: Entry point for CMS server connections accepted by the stream module.
143 * Allocates context, sets up handlers and timer, arms first read.
144 *
145 * WHY: The CMS manager connects to this port as a data-server client. We need
146 * a per-connection state object (ctx) that tracks the connection pointer,
147 * how many bytes remain in the current frame header, the peer host string,
148 * and the heartbeat timer.
149 *
150 * HOW: 1. pcalloc ctx on c->pool (auto-cleaned on pool destruction).
151 * 2. Initialise ctx fields: c pointer, hdr-in_need = CMS_HDR_LEN,
152 * host from ngx_sock_ntop, interval_ms from conf.
153 * 3. Set ping_timer log/data, assign c->data = ctx.
154 * 4. Replace default read/write handlers with cms_srv_read/cms_srv_write.
155 * 5. Log debug line and arm first read.
156 */
157
158{
159 ngx_connection_t *c;
160 brix_cms_srv_ctx_t *ctx;
161 ngx_stream_brix_cms_srv_conf_t *conf;
162 size_t len;
163
164 c = s->connection;
165
166 ctx = ngx_pcalloc(c->pool, sizeof(brix_cms_srv_ctx_t));
167 if (ctx == NULL) {
168 ngx_stream_finalize_session(s, NGX_STREAM_INTERNAL_SERVER_ERROR);
169 return;
170 }
171
172 ctx->c = c;
173 ctx->in_need = NGX_BRIX_CMS_HDR_LEN;
174
175 len = ngx_sock_ntop(c->sockaddr, c->socklen,
176 (u_char *) ctx->host, sizeof(ctx->host) - 1, 0);
177 ctx->host[len] = '\0';
178
179 conf = ngx_stream_get_module_srv_conf(s, ngx_stream_brix_cms_srv_module);
180 ctx->conf = conf;
181 ctx->interval_ms = (ngx_msec_t) conf->interval * 1000;
182 if (ctx->interval_ms < 1000) {
183 /* Never arm a 0/sub-1s self-rearming ping timer: interval 0 → 0ms timer
184 * → epoll_wait(timeout=0) busy-loop pinning the worker. Floor at 1s. */
185 ctx->interval_ms = 1000;
186 }
187 ctx->login_timeout_ms = conf->login_timeout; /* WS3 */
188 ctx->idle_timeout_ms = conf->idle_timeout; /* WS3 */
189
190 /*
191 * WS4 — admission cap. Reject before installing any frame handler so an
192 * abusive peer (inside the often-unset CIDR allowlist) cannot exhaust
193 * memory/fds with idle connections. 0 = unlimited (back-compat).
194 */
195 if (conf->max_connections > 0
196 && brix_cms_srv_conn_count() >= (ngx_uint_t) conf->max_connections)
197 {
198 ngx_log_error(NGX_LOG_WARN, c->log, 0,
199 "brix: CMS server: connection from %s refused — "
200 "max_connections (%i) reached",
201 ctx->host, conf->max_connections);
202 BRIX_RESIL_METRIC_INC(cms_cap_rejections_total);
203 ngx_stream_finalize_session(s, NGX_STREAM_FORBIDDEN);
204 return;
205 }
206
207 /*
208 * A3 — per-source-IP admission cap, so one peer cannot consume every slot of
209 * the global cap above. 0 = disabled (back-compat).
210 */

213 >= (ngx_uint_t) conf->max_connections_per_ip)
214 {

216 "brix: CMS server: connection from %s refused — "

219 BRIX_RESIL_METRIC_INC(cms_cap_rejections_total);

221 return;
222 }
223
224 /*

226 * frame handler so an unauthorised peer never reaches the LOGIN/registry
227 * path. When no allowlist is configured this passes (back-compat).
228 */

230 BRIX_DIAG(NGX_LOG_NOTICE, c->log, 0,

233 "allowlist (or the allowlist is wrong)",
234 "if this is a legitimate data server, add its IP/CIDR to "
235 "brix_cms_server_allow; otherwise this is an unauthorised peer "
236 "being correctly refused",
237 ctx->host);
238 ngx_stream_finalize_session(s, NGX_STREAM_FORBIDDEN);
239 return;
240 }
241
242 /* W1a — require the sss handshake before registration iff a keytab is set. */
243 ctx->auth_state = (conf->sss_keys != NULL) ? CMS_AUTH_REQUESTED
244 : CMS_AUTH_NONE;
245
246 ctx->ping_timer.log = c->log;
247 ctx->ping_timer.data = ctx;
248
249 /* WS4 + A3: admitted — count it globally and per-IP (decremented in
250 * brix_cms_srv_close). */

252 brix_cms_srv_ip_inc(ctx->host);
253 ctx->counted = 1;
254
255 c->data = ctx;

257 c->write->handler = brix_cms_srv_write;

261 * dropped data node is torn down by the kernel. Best-effort, non-fatal.
262 */

264 conf->tcp_user_timeout);
265

268 * never completes LOGIN (+sss xauth) — or trickles a partial header — is
269 * closed by the read handler's ev->timedout path instead of squatting a ctx +
270 * fd forever (slowloris). Replaced by the post-login idle watchdog in
271 * cms_srv_complete_login. 0 = disabled.
272 */
273 if (ctx->login_timeout_ms > 0) {
274 ngx_add_timer(c->read, ctx->login_timeout_ms);
275 }
276
277 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
278 "brix: CMS server accepted from %s", ctx->host);
279
280 brix_cms_srv_read(c->read);
