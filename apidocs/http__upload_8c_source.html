8/* Stream `clen` bytes to the connected io as the PUT body, pulling them from the
9 * source by ABSOLUTE offset (base_off + bytes already sent) so the same source
10 * serves a whole-body PUT (base_off 0) and a resumable Content-Range chunk
11 * (base_off = chunk start) without a shared file offset. The source shrinking
12 * mid-stream is a protocol error (the Content-Length already promised the full
13 * size). 0 / -1 (st set on error). */
14int
15httpx_upload_body(brix_io *io, brix_http_body_src_fn src, void *src_ctx,
16 long long base_off, long long clen, brix_status *st)
17{
18 uint8_t buf[XRDC_XFER_BUF];
19 long long remaining = clen;
20
21 while (remaining > 0) {
22 size_t want = (remaining < (long long) sizeof(buf))
23 ? (size_t) remaining : sizeof(buf);
24 ssize_t r = src(src_ctx, buf, base_off + (clen - remaining), want, st);
25 if (r < 0) {
26 if (st->kxr == 0 && st->sys_errno == 0) {
27 brix_status_set(st, XRDC_ESOCK, 0, "upload: source read failed");
28 }
29 return -1;
30 }
31 if (r == 0) {
32 brix_status_set(st, XRDC_EPROTO, 0,
33 "upload: source shrank (%lld bytes short)", remaining);
34 return -1;
35 }
36 if (brix_write_full(io, buf, (size_t) r, st) != 0) { return -1; }
37 remaining -= r;
38 }
39 return 0;
40}

43/* Read the PUT response headers and map a 2xx status to success. Owns its scratch
44 * header buffer (allocated, used, freed on every path) so the orchestrator's
45 * transport teardown stays linear. 0 / -1 (st set on error). */
46int
47httpx_upload_response(brix_io *io, int timeout_ms, int *http_status,
48 brix_status *st)
49{
50 char *hdr;
51 size_t total = 0, body_off = 0;
52 int status = 0, rc;
53
54 hdr = (char *) malloc(XRDC_HDR_CAP);
55 if (hdr == NULL) {
56 brix_status_set(st, XRDC_EPROTO, 0, "http: out of memory");
57 return -1;
58 }
59 if (read_resp_headers(io, hdr, XRDC_HDR_CAP, timeout_ms, &status,
60 &total, &body_off, st) != 0) {
61 free(hdr);
62 return -1;
63 }
64 free(hdr);
65 if (http_status != NULL) { *http_status = status; }
66 if (status >= 200 && status < 300) {
67 rc = 0;
68 } else {
69 brix_status_set(st, XRDC_EPROTO, 0, "upload: server returned status %d", status);
70 rc = -1;
71 }
72 return rc;
73}

76/* Send the PUT request line + headers, stream the body, and read the response —
77 * over an already-connected io. Flat early-return so the orchestrator's transport
78 * teardown needs no shared cleanup label. 0 / -1 (st set on error). */
79int
80httpx_upload_exchange(brix_io *io, const char *host, int port, const char *path,
81 const char *extra_headers, brix_http_body_src_fn src,
82 void *src_ctx, long long clen, int timeout_ms,
83 int *http_status, brix_status *st)
84{
85 char req[2048];
86 int rlen;
87
88 char hp[300];
89 brix_format_host_port(host, (uint16_t) port, hp, sizeof(hp));
90 rlen = snprintf(req, sizeof(req),
91 "PUT %s HTTP/1.1\r\nHost: %s\r\nUser-Agent: xrdcp\r\n"
92 "Connection: close\r\nContent-Length: %lld\r\n%s\r\n",
93 path[0] ? path : "/", hp, clen,
94 extra_headers ? extra_headers : "");
95 if (rlen < 0 || (size_t) rlen >= sizeof(req)) {
96 brix_status_set(st, XRDC_EUSAGE, 0, "http: request too long");
97 return -1;
98 }
99 if (brix_write_full(io, req, (size_t) rlen, st) != 0) { return -1; }
100 if (httpx_upload_body(io, src, src_ctx, 0, clen, st) != 0) { return -1; }
101 return httpx_upload_response(io, timeout_ms, http_status, st);
102}

105/* Case-insensitively scan a header block [hdr, hdr+len) for "X-Upload-Offset"
106 * and return its value, or -1 if absent/unparsable. */
107long long
108httpx_parse_upload_offset(const char *hdr, size_t len)
109{
110 static const char key[] = "x-upload-offset:";
111 size_t klen = sizeof(key) - 1;
112 size_t i;
113
114 for (i = 0; i + klen <= len; i++) {
115 size_t j;
116 for (j = 0; j < klen; j++) {
117 char c = hdr[i + j];
118 if (c >= 'A' && c <= 'Z') { c = (char) (c + 32); }
119 if (c != key[j]) { break; }
120 }
121 if (j == klen) {
122 const char *p = hdr + i + klen;
123 const char *e = hdr + len;
124 long long v = 0;
125 int any = 0;
126 while (p < e && (*p == ' ' || *p == '\t')) { p++; }
127 while (p < e && *p >= '0' && *p <= '9') {
128 v = v * 10 + (*p - '0'); p++; any = 1;
129 }
130 return any ? v : -1;
131 }
132 }
133 return -1;
134}

186/*
187 * Resumable upload: stream the source as a sequence of Content-Range PUT chunks,
188 * each on a fresh connection, so an nginx restart mid-upload is survived — on a
189 * transport sever or a 409 (offset-correction) the loop reconnects and continues
190 * from the server's durable offset, within a bounded stall window. Requires the
191 * server's brix_webdav_upload_resume; against a server without it the first
192 * Content-Range PUT is treated as a whole-body write (commit) and this still
193 * works for a single-shot upload. 0 / -1 (st set).
194 */
195int
196brix_http_upload_resumable(const char *host, int port, int tls, const char *path,
197 const char *extra_headers, brix_http_body_src_fn src,
198 void *src_ctx, long long clen, int verify,
199 const char *ca_dir, int timeout_ms, int max_stall_ms,
200 int *http_status, brix_status *st)
201{
202 long long off = 0;
203 unsigned attempt = 0;
204 uint64_t deadline = brix_mono_ns() + (uint64_t) max_stall_ms * 1000000ULL;
205 const long long CHUNK = 8LL * 1024 * 1024;
206
207 if (http_status != NULL) { *http_status = 0; }
208
209 while (off < clen) {
210 brix_io io;
211 void *tls_ctx = NULL;
212 long long chunk = (clen - off < CHUNK) ? (clen - off) : CHUNK;
213 int status = 0;
214 long long srv_off = -1;
215 int rc;
216
217 if (httpx_connect(&io, host, port, tls, verify, ca_dir, timeout_ms,
218 &tls_ctx, st) != 0) {
219 if (max_stall_ms <= 0 || !brix_status_retryable(st)
220 || brix_mono_ns() >= deadline) {
221 return -1;
222 }
223 brix_backoff_sleep_fast(attempt++);
224 continue;
225 }
226
227 rc = httpx_upload_chunk(&io, host, port, path, extra_headers, src,
228 src_ctx, off, chunk, clen, timeout_ms,
229 &status, &srv_off, st);
230 if (tls) { brix_tls_client_free(&io, tls_ctx); }
231 if (io.fd >= 0) { close(io.fd); }
232
233 if (rc != 0) {
234 /* transport sever: reconnect and resume from the same offset */
235 if (max_stall_ms <= 0 || !brix_status_retryable(st)
236 || brix_mono_ns() >= deadline) {
237 return -1;
238 }
239 brix_backoff_sleep_fast(attempt++);
240 continue;
241 }
242
243 if (http_status != NULL) { *http_status = status; }
244 deadline = brix_mono_ns() + (uint64_t) max_stall_ms * 1000000ULL;
245 attempt = 0;
246
247 if (status == 409 && srv_off >= 0) {
248 off = srv_off; /* server told us the real offset */
249 continue;
250 }
251 if (status == 201 || status == 204) {
252 return 0; /* committed (final chunk) */
253 }
254 if (status >= 200 && status < 300) {
255 off += chunk; /* intermediate chunk accepted */
256 continue;
257 }
258 brix_status_set(st, XRDC_EPROTO, 0,
259 "upload: server returned status %d", status);
260 return -1;
261 }
262 return 0;
263}
