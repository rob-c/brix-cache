8/*
9 * httpx_upload_ctx_t - file-local bundle for the upload family.
10 *
11 * WHAT: One value type carrying the endpoint (host/port/tls/path/headers/TLS
12 * verification), the body pull-source callback, the transfer sizing, and
13 * the shared out-params (http_status, status) that every upload helper
14 * needs. Zero-init at the extern entry points; never escapes this file.
15 * WHY: The public/extern upload signatures are frozen (declared in brix_net.h /
16 * http_internal.h with out-of-file callers), so the 11-14 param bloat on
17 * those entry points cannot be removed. Threading THIS ctx through the new
18 * STATIC helpers keeps every decomposed step at a small, honest param
19 * count and lets the resumable driver split its loop body without passing
20 * a dozen scalars per call. The extern PARAM residuals are documented and
21 * accepted; the internal machinery is param-lean.
22 * HOW: Each extern function copies its frozen args into a stack ctx (designated
23 * zero-init) and delegates. Helpers read the endpoint/source/transfer
24 * fields directly and take only the per-step scalars (offsets, lengths).
25 */
26typedef struct {
27 const char *host;
28 int port;
29 int tls;
30 const char *path;
31 const char *extra_headers;
32 int verify;
33 const char *ca_dir;
34 brix_http_body_src_fn src;
35 void *src_ctx;
36 long long clen;
37 int timeout_ms;
38 int *http_status;
39 brix_status *st;
40} httpx_upload_ctx_t;

43/*
44 * httpx_chunk_range_t - one Content-Range chunk's coordinates + out-params.
45 *
46 * WHAT: The per-chunk values that vary from one resumable PUT to the next:
47 * the byte window [off, off+len) within a `total`-byte object, plus the
48 * two out-slots the exchange writes back (HTTP status, server offset).
49 * WHY: These do NOT belong on the reusable httpx_upload_ctx_t (that describes
50 * the whole upload, not one chunk). Bundling them keeps the chunk helpers
51 * at a lean param count without smuggling chunk state into the ctx.
52 * HOW: The resumable driver fills off/len/total per iteration and reads back
53 * status/srv_off after httpx_chunk_core returns.
54 */
55typedef struct {
56 long long off;
57 long long len;
58 long long total;
59 int *status_out;
60 long long *srv_off_out;
61} httpx_chunk_range_t;

64/* Stream `clen` bytes to the connected io as the PUT body, pulling them from the
65 * source by ABSOLUTE offset (base_off + bytes already sent) so the same source
66 * serves a whole-body PUT (base_off 0) and a resumable Content-Range chunk
67 * (base_off = chunk start) without a shared file offset. The source shrinking
68 * mid-stream is a protocol error (the Content-Length already promised the full
69 * size). 0 / -1 (st set on error). */
70int
71httpx_upload_body(brix_io *io, brix_http_body_src_fn src, void *src_ctx,
72 long long base_off, long long clen, brix_status *st)
73{
74 uint8_t buf[XRDC_XFER_BUF];
75 long long remaining = clen;
76
77 while (remaining > 0) {
78 size_t want = (remaining < (long long) sizeof(buf))
79 ? (size_t) remaining : sizeof(buf);
80 ssize_t r = src(src_ctx, buf, base_off + (clen - remaining), want, st);
81 if (r < 0) {
82 if (st->kxr == 0 && st->sys_errno == 0) {
83 brix_status_set(st, XRDC_ESOCK, 0, "upload: source read failed");
84 }
85 return -1;
86 }
87 if (r == 0) {
88 brix_status_set(st, XRDC_EPROTO, 0,
89 "upload: source shrank (%lld bytes short)", remaining);
90 return -1;
91 }
92 if (brix_write_full(io, buf, (size_t) r, st) != 0) { return -1; }
93 remaining -= r;
94 }
95 return 0;
96}

99/* Read the PUT response headers and map a 2xx status to success. Owns its scratch
100 * header buffer (allocated, used, freed on every path) so the orchestrator's
101 * transport teardown stays linear. 0 / -1 (st set on error). */
102int
103httpx_upload_response(brix_io *io, int timeout_ms, int *http_status,
104 brix_status *st)
105{
106 char *hdr;
107 size_t total = 0, body_off = 0;
108 int status = 0, rc;
109
110 hdr = (char *) malloc(XRDC_HDR_CAP);
111 if (hdr == NULL) {
112 brix_status_set(st, XRDC_EPROTO, 0, "http: out of memory");
113 return -1;
114 }
115 if (read_resp_headers(io, hdr, XRDC_HDR_CAP, timeout_ms, &status,
116 &total, &body_off, st) != 0) {
117 free(hdr);
118 return -1;
119 }
120 free(hdr);
121 if (http_status != NULL) { *http_status = status; }
122 if (status >= 200 && status < 300) {
123 rc = 0;
124 } else {
125 brix_status_set(st, XRDC_EPROTO, 0, "upload: server returned status %d", status);
126 rc = -1;
127 }
128 return rc;

132/* WHAT: Format the PUT request line + headers for a whole-body upload into `req`.
133 * WHY: Isolates the snprintf-and-overflow-check so the exchange helper is flat.
134 * HOW: Content-Length only (no Content-Range) — this is the non-resumable path.
135 * 0 / -1 (st set on overflow). */
136static int
137httpx_fmt_put_request(const httpx_upload_ctx_t *c, char *req, size_t cap,
138 int *rlen_out)
139{
140 char hp[300];
141 int rlen;

158/* WHAT: Whole-body PUT exchange over a connected io, driven by the upload ctx.
159 * WHY: The extern httpx_upload_exchange keeps its frozen 11-param signature; this
160 * static core does the work at a lean param count (ctx + io).
161 * HOW: Format request -> write it -> stream the body from offset 0 -> read the
162 * response. Flat early-return; no shared cleanup label. 0 / -1 (st set). */
163static int
164httpx_exchange_core(const httpx_upload_ctx_t *c, brix_io *io)
165{
166 char req[2048];
167 int rlen;
168
169 if (httpx_fmt_put_request(c, req, sizeof(req), &rlen) != 0) { return -1; }
170 if (brix_write_full(io, req, (size_t) rlen, c->st) != 0) { return -1; }
171 if (httpx_upload_body(io, c->src, c->src_ctx, 0, c->clen, c->st) != 0) {
172 return -1;
173 }
174 return httpx_upload_response(io, c->timeout_ms, c->http_status, c->st);
175}

197/* Case-insensitively scan a header block [hdr, hdr+len) for "X-Upload-Offset"
198 * and return its value, or -1 if absent/unparsable. */
199long long
200httpx_parse_upload_offset(const char *hdr, size_t len)
201{
202 static const char key[] = "x-upload-offset:";
203 size_t klen = sizeof(key) - 1;
204 size_t i;
205
206 for (i = 0; i + klen <= len; i++) {
207 size_t j;
208 for (j = 0; j < klen; j++) {
209 char c = hdr[i + j];
210 if (c >= 'A' && c <= 'Z') { c = (char) (c + 32); }
211 if (c != key[j]) { break; }
212 }
213 if (j == klen) {
214 const char *p = hdr + i + klen;
215 const char *e = hdr + len;
216 long long v = 0;
217 int any = 0;
218 while (p < e && (*p == ' ' || *p == '\t')) { p++; }
219 while (p < e && *p >= '0' && *p <= '9') {
220 v = v * 10 + (*p - '0'); p++; any = 1;
221 }
222 return any ? v : -1;
223 }
224 }
225 return -1;
226}

260 * X-Upload-Offset the server advertised.
261 * WHY: Splits the header-read/parse tail out of the chunk driver; owns and frees
262 * its own scratch header buffer on every path.

264 * (st set on error; *srv_off_out already initialised by the caller). */

337/* WHAT: Per-chunk connect + transfer for the resumable driver.
338 * WHY: Isolates one attempt's transport lifecycle (connect, chunk exchange,
339 * unconditional TLS/fd teardown) so the driver loop reads as pure policy.
340 * HOW: httpx_connect -> httpx_chunk_core -> free TLS + close fd on every path.
341 * Returns 0 with *status / *srv_off set on a completed exchange, or -1 with
342 * st set on a connect/transport failure. `rng` carries the chunk window
343 * and receives the status/server-offset out-params. */
344static int
345httpx_resumable_attempt(const httpx_upload_ctx_t *c,
346 const httpx_chunk_range_t *rng)
347{
348 brix_io io;
349 void *tls_ctx = NULL;
350 int rc;
351
352 if (httpx_connect(&io, c->host, c->port, c->tls, c->verify, c->ca_dir,
353 c->timeout_ms, &tls_ctx, c->st) != 0) {
354 return -1;
355 }
356 rc = httpx_chunk_core(c, &io, rng);
357 if (c->tls) { brix_tls_client_free(&io, tls_ctx); }
358 if (io.fd >= 0) { close(io.fd); }
359 return rc;
360}

363/* WHAT: Is the current retry budget still open for another attempt?
364 * WHY: The resumable driver's connect-fail and transport-sever branches share
365 * the exact same retry gate; factoring it keeps the loop honest and one
366 * CCN branch instead of two duplicated ladders.
367 * HOW: A retry is allowed only when stalls are enabled, the error is retryable,
368 * and the stall deadline has not passed. */
369static int
370httpx_resumable_may_retry(const httpx_upload_ctx_t *c, int max_stall_ms,
371 uint64_t deadline)
372{
373 return max_stall_ms > 0 && brix_status_retryable(c->st)
374 && brix_mono_ns() < deadline;

378/*
379 * Resumable upload: stream the source as a sequence of Content-Range PUT chunks,
380 * each on a fresh connection, so an nginx restart mid-upload is survived — on a
381 * transport sever or a 409 (offset-correction) the loop reconnects and continues
382 * from the server's durable offset, within a bounded stall window. Requires the
383 * server's brix_webdav_upload_resume; against a server without it the first
384 * Content-Range PUT is treated as a whole-body write (commit) and this still
385 * works for a single-shot upload. 0 / -1 (st set).
386 */
387int
388brix_http_upload_resumable(const char *host, int port, int tls, const char *path,
389 const char *extra_headers, brix_http_body_src_fn src,
390 void *src_ctx, long long clen, int verify,
391 const char *ca_dir, int timeout_ms, int max_stall_ms,
392 int *http_status, brix_status *st)
393{
394 httpx_upload_ctx_t c = {
395 .host = host, .port = port, .tls = tls, .path = path,
396 .extra_headers = extra_headers, .verify = verify, .ca_dir = ca_dir,
397 .src = src, .src_ctx = src_ctx, .clen = clen, .timeout_ms = timeout_ms,
398 .http_status = http_status, .st = st,
399 };
400 long long off = 0;
401 unsigned attempt = 0;
402 uint64_t deadline = brix_mono_ns() + (uint64_t) max_stall_ms * 1000000ULL;
403 const long long CHUNK = 8LL * 1024 * 1024;
404
405 if (http_status != NULL) { *http_status = 0; }
406
407 while (off < clen) {
408 long long chunk = (clen - off < CHUNK) ? (clen - off) : CHUNK;
409 int status = 0;
410 long long srv_off = -1;
411 httpx_chunk_range_t rng = {
412 .off = off, .len = chunk, .total = clen,
413 .status_out = &status, .srv_off_out = &srv_off,
414 };
415
416 if (httpx_resumable_attempt(&c, &rng) != 0) {
417 /* connect or transport sever: reconnect and resume from `off` */
418 if (!httpx_resumable_may_retry(&c, max_stall_ms, deadline)) {
419 return -1;
420 }
421 brix_backoff_sleep_fast(attempt++);
422 continue;
423 }
424
425 if (http_status != NULL) { *http_status = status; }
426 deadline = brix_mono_ns() + (uint64_t) max_stall_ms * 1000000ULL;
427 attempt = 0;
428
429 if (status == 409 && srv_off >= 0) {
430 off = srv_off; /* server told us the real offset */
431 continue;
432 }
433 if (status == 201 || status == 204) {
434 return 0; /* committed (final chunk) */
435 }
436 if (status >= 200 && status < 300) {
437 off += chunk; /* intermediate chunk accepted */
438 continue;
439 }
440 brix_status_set(st, XRDC_EPROTO, 0,
441 "upload: server returned status %d", status);
442 return -1;
443 }
444 return 0;
445}
