2 * Directory listing handler — chunked entry streaming.
3 * Enumerates an open VFS directory handle and renders each visible entry into
4 * the newline-delimited kXR_dirlist wire format, flushing full 64KB chunks as
5 * kXR_oksofar continuation frames and finishing with a single kXR_ok frame.
6 * Extracted verbatim from handler.c; shared walk state + the public entry point
7 * live in dirlist_handler_internal.h.

20/*
21 * brix_dirlist_entry_fmt_t — formatted per-entry output produced by
22 * brix_dirlist_entry_meta and consumed by brix_dirlist_append_entry.
23 *
24 * Carries the rendered stat line, the optional checksum token, whether stat
25 * data is valid for this entry, and the total wire bytes the entry needs so
26 * the streaming loop can decide when to flush a kXR_oksofar chunk.
27 */
28typedef struct {
29 char statbuf[256]; /* rendered stat line */
30 char cksum_token[EVP_MAX_MD_SIZE * 2 + 64]; /* "algo:hex" token */
31 ngx_flag_t have_stat; /* statbuf is valid */
32 size_t need; /* wire bytes required */
33} brix_dirlist_entry_fmt_t;

35/*
36 * brix_dirlist_name_is_unsafe — detect control characters in a directory
37 * entry name that would corrupt the newline-delimited kXR_dirlist wire format.
38 *
39 * The XRootD dirlist response separates entries with '\n'. A filename
40 * containing '\n' (or any other byte < 0x20) would be misinterpreted as a
41 * record separator by the client. Such filenames are skipped silently.
42 *
43 * Returns 1 (unsafe, skip this entry), 0 (safe to include).
44 */
45static ngx_flag_t
46brix_dirlist_name_is_unsafe(const char *name)
47{
48 const u_char *p;
49
50 for (p = (const u_char *) name; *p != '\0'; p++) {
51 if (*p < 0x20 || *p == 0x7f) {
52 return 1;
53 }
54 }
55
56 return 0;
57}

59/*
60 * brix_dirlist_entry_skip — decide whether a directory entry must be
61 * excluded from the listing.
62 *
63 * Skips "." and "..", this gateway's internal control artifacts (e.g. the
64 * checkpoint recovery lock — a stock XRootD export shows no such files;
65 * mirror that), internal metadata/staging artifacts (cache sidecars, stage
66 * markers, in-flight upload temps — never enumerable to a client), and names
67 * with control bytes that would corrupt the wire format (logged at WARN).
68 *
69 * Returns 1 (skip this entry), 0 (include it).
70 */
71static ngx_flag_t
72brix_dirlist_entry_skip(ngx_log_t *log, const char *name)
73{
74 char safe_name[256];
75
76 if (name[0] == '.' && (name[1] == '\0' ||
77 (name[1] == '.' && name[2] == '\0')))
78 {
79 return 1;
80 }
81
82 if (ngx_strncmp(name, ".nginx-xrootd",
83 sizeof(".nginx-xrootd") - 1) == 0)
84 {
85 return 1;
86 }
87
88 if (brix_is_internal_name(name)) {
89 return 1;
90 }
91
92 if (brix_dirlist_name_is_unsafe(name)) {
93 brix_sanitize_log_string(name, safe_name, sizeof(safe_name));
94 ngx_log_error(NGX_LOG_WARN, log, 0,
95 "brix: dirlist skipping entry with control bytes \"%s\"",
96 safe_name);
97 return 1;
98 }
99
100 return 0;
101}

103/*
104 * brix_dirlist_entry_meta — render the optional stat line and checksum
105 * token for one entry, accumulating the wire bytes it needs into fmt->need.
106 *
107 * readdir already did the no-follow lstat (and skipped any entry whose stat
108 * failed), so vst is valid whenever want_stat is set. The dcksm variant
109 * uses the dcksm stat-body format; the checksum token falls back to
110 * "algo:none" when the entry path would overflow PATH_MAX.
111 *
112 * fmt->need must be pre-seeded with the bare name length + '\n'.
113 */
114static void
115brix_dirlist_entry_meta(ngx_log_t *log, brix_dirlist_walk_t *walk,
116 const char *name, const brix_vfs_stat_t *vst,
117 brix_dirlist_entry_fmt_t *fmt)
118{
119 struct stat entry_st;
120 char entry_path[PATH_MAX];
121
122 fmt->cksum_token[0] = '\0';
123 fmt->have_stat = 0;
124
125 if (walk->want_stat) {
126 ngx_memzero(&entry_st, sizeof(entry_st));
127 entry_st.st_mode = (mode_t) vst->mode;
128 entry_st.st_size = vst->size;
129 entry_st.st_mtime = vst->mtime;
130 entry_st.st_ctime = vst->ctime;
131 entry_st.st_ino = vst->ino;
132 entry_st.st_dev = vst->dev;
133 entry_st.st_uid = vst->uid;
134 entry_st.st_gid = vst->gid;
135 entry_st.st_blocks = vst->blocks;
136
137 fmt->have_stat = 1;
138 if (walk->want_cksum) {
139 brix_dirlist_make_dcksm_stat_body(&entry_st, fmt->statbuf,
140 sizeof(fmt->statbuf));
141 } else {
142 brix_make_stat_body(&entry_st, 0, 0, fmt->statbuf,
143 sizeof(fmt->statbuf));
144 }
145 fmt->need += strlen(fmt->statbuf) + 1;
146 }
147
148 if (walk->want_cksum && fmt->have_stat) {
149 int n;
150
151 n = snprintf(entry_path, sizeof(entry_path), "%s/%s",
152 walk->full_path, name);
153 if (n < 0 || (size_t) n >= sizeof(entry_path)) {
154 snprintf(fmt->cksum_token, sizeof(fmt->cksum_token),
155 "%s:none", walk->cksum_algo);
156 } else {
157 brix_dirlist_checksum_token(log,
158 brix_vfs_dir_fd(walk->dh), name,
159 entry_path, &entry_st,
160 walk->cksum_algo, fmt->cksum_token,
161 sizeof(fmt->cksum_token));
162 }
163 fmt->need += strlen(fmt->cksum_token) + sizeof(" [ ]") - 1;
164 }
165}

167/*
168 * brix_dirlist_flush_chunk — frame the accumulated chunk as a kXR_oksofar
169 * continuation and queue it, resetting the accumulator for the next chunk.
170 *
171 * Returns the brix_queue_response result (NGX_OK on success).
172 */
173static ngx_int_t
174brix_dirlist_flush_chunk(brix_ctx_t *ctx, ngx_connection_t *c,
175 brix_dirlist_walk_t *walk)
176{
177 ngx_int_t rc;
178
179 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_oksofar,
180 (uint32_t) walk->chunk_pos,
181 (ServerResponseHdr *) walk->chunk);
182
183 rc = brix_queue_response(ctx, c, walk->chunk,
184 XRD_RESPONSE_HDR_LEN + walk->chunk_pos);
185 if (rc != NGX_OK) {
186 return rc;
187 }
188
189 walk->chunk_pos = 0;
190 return NGX_OK;
191}

193/*
194 * brix_dirlist_append_entry — copy one formatted entry into the chunk
195 * accumulator: name '\n' [statline [" [ token ]"] '\n'].
196 *
197 * The caller has already guaranteed fmt->need bytes of headroom by flushing
198 * a full chunk beforehand, so the copies cannot overflow chunk_cap.
199 */
200static void
201brix_dirlist_append_entry(brix_dirlist_walk_t *walk, const char *name,
202 size_t nlen, const brix_dirlist_entry_fmt_t *fmt)
203{
204 u_char *data = walk->chunk + XRD_RESPONSE_HDR_LEN;
205
206 ngx_memcpy(data + walk->chunk_pos, name, nlen);
207 walk->chunk_pos += nlen;
208 data[walk->chunk_pos++] = '\n';
209
210 if (walk->want_stat && fmt->have_stat) {
211 size_t slen;
212
213 slen = strlen(fmt->statbuf);
214 ngx_memcpy(data + walk->chunk_pos, fmt->statbuf, slen);
215 walk->chunk_pos += slen;
216
217 if (walk->want_cksum) {
218 int n;
219
220 n = snprintf((char *) data + walk->chunk_pos,
221 walk->chunk_cap - walk->chunk_pos,
222 " [ %s ]", fmt->cksum_token);
223 if (n > 0) {
224 walk->chunk_pos += (size_t) n;
225 }
226 }
227
228 data[walk->chunk_pos++] = '\n';
229 }
230}

232/*
233 * brix_dirlist_send_final — frame the last accumulated chunk as kXR_ok to
234 * signal end-of-listing, log the access, and bump the success metric.
235 *
236 * The trailing '\n' of the final entry is replaced by a NUL terminator (an
237 * empty listing sends a zero-length body).
238 *
239 * Returns the brix_queue_response result.
240 */
241static ngx_int_t
242brix_dirlist_send_final(brix_ctx_t *ctx, ngx_connection_t *c,
243 brix_dirlist_walk_t *walk)
244{
245 u_char *data = walk->chunk + XRD_RESPONSE_HDR_LEN;
246 size_t final_len;
247
248 if (walk->chunk_pos == 0) {
249 final_len = 0;
250 } else {
251 data[walk->chunk_pos - 1] = '\0';
252 final_len = walk->chunk_pos;
253 }
254
255 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
256 (uint32_t) final_len,
257 (ServerResponseHdr *) walk->chunk);
258
259 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
260 "brix: kXR_dirlist final chunk %uz bytes",
261 walk->chunk_pos);
262
263 brix_log_access(ctx, c, "DIRLIST", walk->reqpath,
264 walk->want_cksum ? "dcksm"
265 : (walk->want_stat ? "stat" : "-"),
266 1, 0, NULL, 0);
267 BRIX_OP_OK(ctx, BRIX_OP_DIRLIST);
268
269 return brix_queue_response(ctx, c, walk->chunk,
270 XRD_RESPONSE_HDR_LEN + final_len);
271}

273/*
274 * brix_dirlist_stream_entries — the chunked streaming loop: enumerate the
275 * open directory, format each visible entry, flush full 64KB chunks as
276 * kXR_oksofar frames, and finish with the kXR_ok final frame.
277 *
278 * Owns walk->dh from here on: the handle is closed on every exit path
279 * (queue failure mid-stream or normal end-of-directory).
280 *
281 * Returns the final queue result (NGX_OK on success).
282 */
283ngx_int_t
284brix_dirlist_stream_entries(brix_ctx_t *ctx, ngx_connection_t *c,
285 brix_dirlist_walk_t *walk)
286{
287 ngx_str_t entry_name;
288 brix_vfs_stat_t entry_vst;
289 ngx_int_t rc;
290
291 while (brix_vfs_readdir(walk->dh, &entry_name,
292 walk->want_stat ? &entry_vst : NULL) == NGX_OK) {
293 const char *name = (char *) entry_name.data;
294 size_t nlen = entry_name.len;
295 brix_dirlist_entry_fmt_t fmt;
296
297 if (brix_dirlist_entry_skip(c->log, name)) {
298 continue;
299 }
300
301 fmt.need = nlen + 1;
302 brix_dirlist_entry_meta(c->log, walk, name, &entry_vst, &fmt);
303
304 if (walk->chunk_pos + fmt.need > walk->chunk_cap) {
305 rc = brix_dirlist_flush_chunk(ctx, c, walk);
306 if (rc != NGX_OK) {
307 brix_vfs_closedir(walk->dh, c->log);
308 return rc;
309 }
310 }
311
312 brix_dirlist_append_entry(walk, name, nlen, &fmt);
313 }
314
315 brix_vfs_closedir(walk->dh, c->log);
316
317 return brix_dirlist_send_final(ctx, c, walk);
318}
