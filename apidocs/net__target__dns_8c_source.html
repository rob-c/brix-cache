23/*
24 * net_default_port — resolve the port to use when the URL omitted one.
25 *
26 * WHAT: returns target->port when explicit, else the policy/scheme default.
27 * WHY: identical fallback rules feed both DNS entry points; one helper keeps
28 * the "https implies TLS port, otherwise root:// port; policy value wins,
29 * else IANA default" decision in a single place.
30 * HOW: pure — scheme "https" (len 5) or a require_https policy selects the
31 * TLS port; the policy value wins over the IANA fallback (443 / 1094).
32 */
33static uint16_t
34net_default_port(const brix_net_target_t *target,
35 const brix_net_target_policy_t *policy)
36{
37 if (target->port != 0) {
38 return target->port;
39 }
40
41 if (policy->require_https || target->scheme.len == 5 /* "https" */) {
42 return policy->default_https_port ? policy->default_https_port : 443;
43 }
44
45 return policy->default_root_port ? policy->default_root_port : 1094;
46}

48/*
49 * net_resolve_host — blocking getaddrinfo for host_buf on the chosen port.
50 *
51 * WHAT: on success stores the addrinfo list in *out_res and returns NGX_OK;
52 * on failure writes "DNS resolution failed for <host>" and returns
53 * NGX_ERROR.
54 * WHY: both DNS checkers need the same SOCK_STREAM / AF_UNSPEC resolution and
55 * the same failure message — sharing it keeps them byte-identical.
56 * HOW: BLOCKING getaddrinfo; caller frees the list with freeaddrinfo().
57 */
58static ngx_int_t
59net_resolve_host(const char *host_buf, uint16_t port,
60 struct addrinfo **out_res, char *err, size_t errsz)
61{
62 struct addrinfo hints;
63 char port_str[8];
64
65 snprintf(port_str, sizeof(port_str), "%u", (unsigned) port);
66
67 ngx_memzero(&hints, sizeof(hints));
68 hints.ai_socktype = SOCK_STREAM;
69 hints.ai_family = AF_UNSPEC;
70
71 if (getaddrinfo(host_buf, port_str, &hints, out_res) != 0) {
72 snprintf(err, errsz, "DNS resolution failed for %s", host_buf);
73 return NGX_ERROR;
74 }
75
76 return NGX_OK;
77}

79/*
80 * net_addr_is_prohibited_msg — policy-check one resolved address and, on a
81 * prohibited result, format the shared rejection message.
82 *
83 * WHAT: returns 1 (writing the "host <h> resolves to a prohibited address …"
84 * message into err) when sa is blocked under policy; 0 otherwise.
85 * WHY: the per-result reject verdict and its exact wire message are shared by
86 * check_dns and check_dns_pin; one helper keeps the bytes identical.
87 * HOW: delegates the range test to net_addr_check(); no I/O beyond err.
88 */
89static int
90net_addr_is_prohibited_msg(const struct sockaddr *sa,
91 const brix_net_target_policy_t *policy, const char *host_buf,
92 char *err, size_t errsz)
93{
94 if (!net_addr_check(sa, policy->allow_local, policy->allow_private)) {
95 return 0;
96 }
97
98 snprintf(err, errsz,
99 "host %s resolves to a prohibited address "
100 "(allow_local=%d allow_private=%d)",
101 host_buf,
102 (int) policy->allow_local,
103 (int) policy->allow_private);
104 return 1;
105}

107/*
108 * brix_net_target_check_dns — resolve target->host and reject if ANY
109 * resolved address falls in a prohibited range.
110 *
111 * WHAT: NGX_OK only when every A/AAAA result passes policy; NGX_ERROR (with
112 * reason in err) on resolution failure or a single prohibited result.
113 * WHY: primary SSRF defence for hostname targets — checking all results,
114 * not just the first, stops a multi-A record from hiding a blocked
115 * address behind a permitted one.
116 * HOW: BLOCKING getaddrinfo — caller MUST invoke this from a thread-pool
117 * worker, never the event loop. Use check_dns_pin instead when the
118 * validated address must also be handed to the connect step.
119 */
120ngx_int_t
121brix_net_target_check_dns(const brix_net_target_t *target,
122 const brix_net_target_policy_t *policy,
123 char *err, size_t errsz)
124{
125 struct addrinfo *res, *rp;
126 char host_buf[256];
127
128 if (target->host.len == 0) {
129 snprintf(err, errsz, "target host is empty");
130 return NGX_ERROR;
131 }
132
133 if (brix_str_cbuf(host_buf, sizeof(host_buf), &target->host) == NULL) {
134 snprintf(err, errsz, "target hostname too long");
135 return NGX_ERROR;
136 }
137
138 if (net_resolve_host(host_buf, net_default_port(target, policy),
139 &res, err, errsz) != NGX_OK)
140 {
141 return NGX_ERROR;
142 }
143
144 /* Reject on the FIRST prohibited result, but only after having scanned up
145 * to it — every address must clear policy, so a single bad one fails all. */
146 for (rp = res; rp != NULL; rp = rp->ai_next) {
147 if (net_addr_is_prohibited_msg(rp->ai_addr, policy, host_buf,
148 err, errsz))
149 {
150 freeaddrinfo(res);
151 return NGX_ERROR;
152 }
153 }
154
155 freeaddrinfo(res);
156 return NGX_OK;
157}

174/*
175 * net_pin_append_addr — append this permitted address to the comma-separated
176 * pin list.
177 *
178 * WHAT: writes the numeric form of rp onto the end of pin->buf as
179 * "addr" (first) or ",addr" (subsequent); returns NGX_OK, or NGX_ERROR
180 * (with err set) if getnameinfo fails.
181 * WHY: the caller hands the WHOLE validated set to curl via CURLOPT_RESOLVE
182 * ("host:port:addr1,addr2"), so curl can fall back ACROSS address families
183 * (e.g. a loopback host whose "localhost" is ::1-only in /etc/hosts but
184 * whose server listens on 127.0.0.1 — ::1 is refused, curl then tries the
185 * pinned IPv4) WITHOUT ever re-resolving. Pinning every permitted address
186 * is exactly as safe as pinning one: each was policy-checked before this
187 * call, and no independent re-resolution can smuggle a new address in.
188 * HOW: getnameinfo(NI_NUMERICHOST) into a scratch buffer, then append with a
189 * leading ',' when the list is non-empty. An address that would overflow
190 * the pin buffer is SKIPPED (not truncated) — the list keeps >= 1 addr.
191 */
192static ngx_int_t
193net_pin_append_addr(const struct addrinfo *rp, const char *host_buf,
194 const net_pin_out_t *pin, char *err, size_t errsz)
195{
196 char addr[64]; /* INET6_ADDRSTRLEN (46) + slack */
197 size_t used = ngx_strlen(pin->buf);
198 size_t alen;
199
200 if (getnameinfo(rp->ai_addr, rp->ai_addrlen, addr, sizeof(addr),
201 NULL, 0, NI_NUMERICHOST) != 0)
202 {
203 snprintf(err, errsz, "could not format resolved address for %s",
204 host_buf);
205 return NGX_ERROR;
206 }
207
208 alen = ngx_strlen(addr);
209 if (used + (used ? 1 : 0) + alen + 1 > pin->size) {
210 return NGX_OK; /* buffer full — keep the addresses already pinned */
211 }
212 if (used) {
213 pin->buf[used++] = ',';
214 }
215 ngx_memcpy(pin->buf + used, addr, alen + 1);
216 return NGX_OK;
217}

219/*
220 * brix_net_target_check_dns_pin — like check_dns, but also returns the numeric
221 * IPs of ALL permitted addresses (comma-separated) so the caller can connect to
222 * exactly that validated set (DNS-rebind defence) while still falling back
223 * across them.
224 *
225 * WHAT: validates every resolved address against policy and writes the whole
226 * permitted set's numeric forms into out_ip as "addr1,addr2"; NGX_OK /
227 * NGX_ERROR.
228 * WHY: validating a hostname then letting a separate component re-resolve it
229 * opens a TOCTOU rebind window (DNS answers differently the 2nd time).
230 * Pinning the validated set closes that window; pinning ALL of it (not
231 * just the first) also lets curl fall back e.g. ::1 -> 127.0.0.1 for a
232 * dual-stack/loopback host without ever re-resolving — see loop comment.
233 * HOW: BLOCKING getaddrinfo + getnameinfo(NI_NUMERICHOST); thread-pool only.
234 */
235ngx_int_t
236brix_net_target_check_dns_pin(const brix_net_target_t *target,
237 const brix_net_target_policy_t *policy,
238 char *out_ip, size_t out_ipsz,
239 char *err, size_t errsz)
240{
241 struct addrinfo *res, *rp;
242 char host_buf[256];
243 net_pin_out_t pin = { out_ip, out_ipsz };
244
245 if (out_ip == NULL || out_ipsz == 0) {
246 snprintf(err, errsz, "no pin buffer");
247 return NGX_ERROR;
248 }
249 out_ip[0] = '\0';
250
251 if (target->host.len == 0
252 || brix_str_cbuf(host_buf, sizeof(host_buf), &target->host) == NULL)
253 {
254 snprintf(err, errsz, "target host is empty or too long");
255 return NGX_ERROR;
256 }
257
258 if (net_resolve_host(host_buf, net_default_port(target, policy),
259 &res, err, errsz) != NGX_OK)
260 {
261 return NGX_ERROR;
262 }
263
264 /*
265 * Validate EVERY resolved address (so a multi-A record can't smuggle a
266 * prohibited address past the check) and pin ALL permitted ones (comma-
267 * joined). Pinning the exact validated set is what closes the rebind window
268 * — a later independent re-resolution by the transfer agent is bypassed —
269 * and handing curl the full set lets it fall back across address families.
270 */
271 for (rp = res; rp != NULL; rp = rp->ai_next) {
272 if (net_addr_is_prohibited_msg(rp->ai_addr, policy, host_buf,
273 err, errsz))
274 {
275 freeaddrinfo(res);
276 return NGX_ERROR;
277 }
278
279 if (net_pin_append_addr(rp, host_buf, &pin, err, errsz) != NGX_OK) {
280 freeaddrinfo(res);
281 return NGX_ERROR;
282 }
283 }
284
285 freeaddrinfo(res);
286
287 if (out_ip[0] == '\0') {
288 snprintf(err, errsz, "no addresses resolved for %s", host_buf);
289 return NGX_ERROR;
290 }
291
292 return NGX_OK;
293}
