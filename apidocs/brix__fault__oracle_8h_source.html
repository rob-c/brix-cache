1/*
2 * brix_fault_oracle.h — gated external-command oracle for brix-fault-proxy.
3 *
4 * The auto-bisection and assert-recovery features need to ask a yes/no question
5 * about the system under test — "did the client survive this fault?", "has the
6 * service recovered?" — by running an operator-supplied probe command and reading
7 * its exit code (0 = pass, non-zero = fail/repro).
8 *
9 * Because that means the (loopback, unauthenticated) control port could spawn
10 * processes, it is DOUBLE-GATED like the privileged netem levers: it does nothing
11 * unless the operator passed --enable-exec at startup. Without the flag every
12 * oracle call refuses, so the control port cannot execute anything by default.
13 */
14#ifndef BRIX_FAULT_ORACLE_H
15#define BRIX_FAULT_ORACLE_H

23/* Run `cmd` via /bin/sh -c in its own process group, waiting up to `timeout_ms`
24 * (0 = no timeout). Returns the child's exit status (0..255) on a clean exit,
25 * -2 if the oracle is not enabled, or -1 on a spawn error / timeout kill
26 * (inconclusive). The child is SIGKILLed (whole group) on timeout. */
27int fp_oracle_run(const char *cmd, int timeout_ms);
