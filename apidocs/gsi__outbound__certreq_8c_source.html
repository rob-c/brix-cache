1/* File: gsi_outbound_certreq.c — GSI certificate request for native TPC pull
2 * WHAT: Initiates the outbound GSI authentication handshake on a TPC pull socket. Reads brix_certificate and brix_certificate_key from config, loads X509 chain + private key via OpenSSL BIO/PEM readers, sends kXGC_certreq wire message (gsi\x00 + opcode + kXRS_none), receives kXR_authmore response containing client cert + CA chain. Validates server expects auth continuation before returning NGX_OK or error code.
3 *
4 * WHY: Native TPC pull connects directly to an xrootd server on a separate socket; GSI authentication requires the outbound side to present its certificate chain and private key, then receive the server's client certificate + CA chain for mutual verification. This function performs only the first round of that handshake — sending certreq and verifying kXR_authmore response — with subsequent rounds handled by gsi_outbound_common.c functions (tpc_send_kxr_auth continuation).
5 *
6 * HOW: Validate config has certificate + key paths → ngx_memcpy into local PATH_MAX buffers with NUL termination → BIO_new_file("r") on both cert and key PEM files → sk_X509_new_null() + loop PEM_read_bio_X509(cbio) pushing to chain (x = NULL after each push) → reject if chain empty → PEM_read_bio_PrivateKey(kbio) → malloc(crlen=4+4+8) for wire message → memcpy "gsi\x00" + tpc_put_u32(kXGC_certreq) + tpc_put_u32(kXRS_none) + tpc_put_u32(0) → tpc_send_kxr_auth(t, fd, 3, certreq, crlen) → recv response via tpc_recv_response checking status == kXR_authmore with body ≥16 bytes → every exit returns through tpc_outbound_gsi_finish: BIO_free(cbio/kbio), sk_X509_free(chain), EVP_PKEY_free(pkey), free(certreq/body). Returns NGX_OK on success or error code. Caller: tpc/bootstrap.c (tpc_pull_start).
7 * */
8

24/*
25 * tpc_outbound_gsi_finish — release the certreq round's OpenSSL/wire resources
26 * and return rc unchanged.
27 *
28 * Called at every exit of tpc_outbound_gsi (success and each early failure).
29 * Every handle is NULL-initialised at the top of that function and each guard is
30 * NULL-safe, so invoking this at any point frees exactly what had been built so
31 * far — which lets the caller exit with a plain `return` instead of a shared
32 * goto/label. sk_X509_free frees the stack and every cert it owns.
33 */
34static int
35tpc_outbound_gsi_finish(int rc, BIO *cbio, BIO *kbio, STACK_OF(X509) *chain,
36 EVP_PKEY *pkey, u_char *certreq, u_char *body)
37{
38 if (cbio) BIO_free(cbio);
39 if (kbio) BIO_free(kbio);
40 if (chain) sk_X509_free(chain);
41 if (pkey) EVP_PKEY_free(pkey);
42 if (certreq) free(certreq);
43 if (body) free(body);
44 return rc;
45}

47/* WHAT: Initiates GSI auth handshake on TPC pull socket — read cert/key PEM, send kXGC_certreq wire message, verify kXR_authmore response. */
48int
49tpc_outbound_gsi(brix_tpc_pull_t *t, int fd,
50 const u_char *login_body, uint32_t login_dlen)
51{
52 ngx_stream_brix_srv_conf_t *conf = t->conf;
53 u_char cert_path[PATH_MAX]; /* NUL-terminated copies of the */
54 u_char key_path[PATH_MAX]; /* ngx_str_t config paths */
55 BIO *cbio = NULL, *kbio = NULL; /* pbio unused in this fragment */
56 X509 *x = NULL; /* scratch cert from each PEM read */
57 STACK_OF(X509) *chain = NULL; /* owns every cert pushed onto it */
58 EVP_PKEY *pkey = NULL;
59 u_char *certreq = NULL; /* malloc'd wire message (freed at done) */
60 u_char *body = NULL; /* malloc'd server reply (freed at done) */
61 uint16_t status;
62 uint32_t dlen;
63 int rc = -1; /* default = failure; set 0 only on success */
64
65 /*
66 * §F6: when proxy delegation captured the user's proxy, authenticate to the
67 * source AS THE USER with that in-memory credential (proxy cert + issuer chain
68 * + key, PEM) instead of the gateway's brix_certificate. Two BIOs over the
69 * same blob: one yields the cert chain (stops at the trailing key block), the
70 * other the private key (skips the cert blocks). The blob is owned by the pull
71 * task (freed in thread.c).
72 */
73 if (t->deleg_cred_pem != NULL && t->deleg_cred_len > 0) {
74 cert_path[0] = '\0'; /* unused on this path; silence later references */
75 key_path[0] = '\0';
76 cbio = BIO_new_mem_buf(t->deleg_cred_pem, (int) t->deleg_cred_len);
77 kbio = BIO_new_mem_buf(t->deleg_cred_pem, (int) t->deleg_cred_len);
78 } else {
79 /* WHY: paths come from config as length-counted ngx_str_t; reject empty
80 * paths and any that would not fit (with room for the NUL) in the local
81 * PATH_MAX buffers — brix_str_cbuf() copies + NUL-terminates and
82 * returns NULL when the result would not fit. */
83 if (conf->certificate.len == 0 || conf->certificate_key.len == 0
84 || brix_str_cbuf((char *) cert_path, sizeof(cert_path),
85 &conf->certificate) == NULL
86 || brix_str_cbuf((char *) key_path, sizeof(key_path),
87 &conf->certificate_key) == NULL)
88 {
89 snprintf(t->err_msg, sizeof(t->err_msg),
90 "TPC GSI outbound needs brix_certificate and "
91 "brix_certificate_key");
92 t->xrd_error = kXR_ArgInvalid;
93 return -1;
94 }
95
96 /* Open read-only PEM streams over the cert and key files. */
97 cbio = BIO_new_file((char *) cert_path, "r");
98 kbio = BIO_new_file((char *) key_path, "r");
99 }
100 if (cbio == NULL || kbio == NULL) {
101 snprintf(t->err_msg, sizeof(t->err_msg),
102 "TPC GSI cannot read certificate or key PEM");
103 t->xrd_error = kXR_IOError;
104 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey, certreq, body);
105 }
106
107 chain = sk_X509_new_null();
108 if (chain == NULL) {
109 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey, certreq, body);
110 }
111
112 /* Read every certificate in the PEM file into the chain (leaf first,
113 * then intermediates). sk_X509_push transfers ownership of x to the
114 * stack, so we clear x to NULL afterward — otherwise an early loop exit
115 * or error path could double-free a cert that the stack already owns. */
116 while ((x = PEM_read_bio_X509(cbio, NULL, NULL, NULL)) != NULL) {
117 sk_X509_push(chain, x);
118 x = NULL;
119 }
120
121 /* A cert file that parsed to zero certs cannot authenticate us. */
122 if (sk_X509_num(chain) == 0) {
123 snprintf(t->err_msg, sizeof(t->err_msg),
124 "TPC GSI certificate file has no X.509 certs");
125 t->xrd_error = kXR_ArgInvalid;
126 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey, certreq, body);
127 }
128
129 pkey = PEM_read_bio_PrivateKey(kbio, NULL, NULL, NULL);
130 if (pkey == NULL) {
131 snprintf(t->err_msg, sizeof(t->err_msg),
132 "TPC GSI cannot read private key PEM");
133 t->xrd_error = kXR_ArgInvalid;
134 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey, certreq, body);
135 }
136
137 /* WHY: security gate — before we put our identity on the wire, validate
138 * the proxy PEM (e.g. not-yet-valid / expired / malformed) so a bad local
139 * credential fails fast here with kXR_AuthFailed rather than mid-handshake.
140 * cred wraps the in-memory cert_path buffer; nothing is sent yet.
141 * Skipped on the §F6 delegated-credential path: that proxy is freshly minted
142 * by the client this handshake and already key-verified (assemble_proxy). */
143 if (t->deleg_cred_pem == NULL) {
144 brix_tpc_credential_t cred;
145
146 ngx_memzero(&cred, sizeof(cred));
147 cred.type = BRIX_TPC_CREDENTIAL_PROXY;
148 cred.proxy_pem.data = cert_path;
149 cred.proxy_pem.len = conf->certificate.len;
150
151 if (brix_tpc_credential_validate(
152 &cred, t->c != NULL ? t->c->log : NULL) != NGX_OK)
153 {
154 snprintf(t->err_msg, sizeof(t->err_msg),
155 "TPC GSI credential validation failed");
156 t->xrd_error = kXR_AuthFailed;
157 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey, certreq, body);
158 }
159 }
160
161 /* round 1: kXGC_certreq * Build the GSI credential payload that opens the handshake. Layout (16B):
162 * [0..3] "gsi\0" 4-byte protocol tag (NUL-padded)
163 * [4..7] kXGC_certreq (=1000) opcode: client requests server cert
164 * [8..11] kXRS_none (=0) bucket type 0 = end-of-message marker
165 * [12..15] 0 that terminator bucket's length (0)
166 * All multi-byte fields are big-endian via tpc_put_u32. crlen = 4+4+8 so
167 * the second tpc_put_u32 below writes into bytes [12..15] of the buffer. */
168 {
169 size_t crlen = 0;
170 uint32_t version = 0;
171 char crypto[64];
172 char ca[256];
173
174 /* Build a REAL certreq (the stock XrdSecgsi round-1: crypto module +
175 * version + issuer-hash + client-opts + random rtag) via the shared
176 * gsi_core kernel — the exact bytes the native client sends. A bare
177 * "gsi\0"+kXGC_certreq stub is rejected by the source with kXR_error.
178 *
179 * Parse the source's advertised gsi parms (v:/c:/ca:) from the login
180 * reply body ("<session id><&P=gsi,...>") so it picks the right crypto
181 * module + CA chain, exactly like client/lib/sec/sec_gsi.c. */
182 crypto[0] = '\0';
183 ca[0] = '\0';
184 if (login_dlen > BRIX_SESSION_ID_LEN) {
185 brix_gsi_parse_parms(
186 (const char *) login_body + BRIX_SESSION_ID_LEN,
187 &version, crypto, sizeof(crypto), ca, sizeof(ca));
188 }
189 if (crypto[0] == '\0') {
190 ngx_memcpy(crypto, "ssl", 4);
191 }
192 /* Advertise a pre-DHsigned version (< XrdSecgsiVersDHsigned=10400) so the
193 * source uses the simpler UNSIGNED DH path — it sends its DH public as a
194 * plain kXRS_puk blob, which gsi_outbound_exchange.c implements (it does
195 * NOT handle the signed-DH kXRS_cipher path the native client supports).
196 * Proof-of-possession is still enforced (we sign the server rtag below). */
197 version = 10300;
198
199 if (!brix_gsi_rand(t->gsi_rtag, sizeof(t->gsi_rtag))) {
200 snprintf(t->err_msg, sizeof(t->err_msg), "TPC GSI RNG failed");
201 t->xrd_error = kXR_ServerError;
202 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey,
203 certreq, body);
204 }
205
206 /* clnt_opts 0x80 matches a stock client (delegated-proxy off). The buffer
207 * is malloc'd and freed by tpc_outbound_gsi_finish. */
208 certreq = brix_gsi_build_certreq(crypto, version,
209 ca[0] ? ca : NULL, 0x80u,
210 t->gsi_rtag, sizeof(t->gsi_rtag),
211 &crlen);
212 if (certreq == NULL) {
213 snprintf(t->err_msg, sizeof(t->err_msg),
214 "TPC GSI cannot build certreq");
215 t->xrd_error = kXR_NoMemory;
216 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey,
217 certreq, body);
218 }
219
220 /* seq=3: third request on this stream (protocol=1, login=2). */
221 if (tpc_send_kxr_auth(t, fd, 3, certreq, (uint32_t) crlen) != 0) {
222 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey,
223 certreq, body);
224 }
225 }
226
227 /* Read the server's reply to certreq. body is malloc'd by the helper and
228 * freed at done:; pre-clearing it keeps that cleanup safe on recv failure. */
229 body = NULL;
230 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
231 snprintf(t->err_msg, sizeof(t->err_msg),
232 "TPC GSI kXGS_cert recv failed");
233 t->xrd_error = kXR_ServerError;
234 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey, certreq, body);
235 }
236
237 /* Security check: the server must answer with kXR_authmore (=4002),
238 * signalling "continue authenticating" — any other status (kXR_ok early,
239 * kXR_error, etc.) means we cannot proceed and is treated as auth failure.
240 * dlen < 16 would be too short to hold the cert + CA bucket the next round
241 * (gsi_outbound_exchange.c) parses, so reject undersized bodies here. */
242 if (status != kXR_authmore || body == NULL || dlen < 16) {
243 snprintf(t->err_msg, sizeof(t->err_msg),
244 "TPC GSI expected kXR_authmore after certreq (status=%u)",
245 (unsigned) status);
246 t->xrd_error = kXR_AuthFailed;
247 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey, certreq, body);
248 }
249
250 /* Round 1 complete: server is mid-handshake. Drive round 2 (the DH key
251 * exchange + encrypted client cert) HERE — this call was previously missing,
252 * so tpc_outbound_gsi returned "success" after only certreq and the source
253 * saw the destination as unauthenticated ("user not authenticated").
254 *
255 * tpc_outbound_gsi_exchange consumes `body` (frees it on every exit — see the
256 * matching change in gsi_outbound_exchange.c), and does NOT free
257 * chain/pkey/certreq/cbio/kbio (we do, below). Clear `body` afterwards so
258 * tpc_outbound_gsi_finish cannot double-free it. */
259 rc = tpc_outbound_gsi_exchange(t, fd, body, dlen, x, chain, pkey,
260 certreq, cbio, kbio);
261 body = NULL;
262
263 /* Success exit: tpc_outbound_gsi_finish releases every resource (each guard
264 * is NULL-safe) and returns rc — the same cleanup every error path uses. */
265 return tpc_outbound_gsi_finish(rc, cbio, kbio, chain, pkey, certreq, body);
266}
