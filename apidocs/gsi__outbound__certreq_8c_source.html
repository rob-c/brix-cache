1/* File: gsi_outbound_certreq.c — GSI certificate request for native TPC pull
2 * WHAT: Initiates the outbound GSI authentication handshake on a TPC pull socket. Reads brix_certificate and brix_certificate_key from config, loads X509 chain + private key via OpenSSL BIO/PEM readers, sends kXGC_certreq wire message (gsi\x00 + opcode + kXRS_none), receives kXR_authmore response containing client cert + CA chain. Validates server expects auth continuation before returning NGX_OK or error code.
3 *
4 * WHY: Native TPC pull connects directly to an xrootd server on a separate socket; GSI authentication requires the outbound side to present its certificate chain and private key, then receive the server's client certificate + CA chain for mutual verification. This function performs only the first round of that handshake — sending certreq and verifying kXR_authmore response — with subsequent rounds handled by gsi_outbound_common.c functions (tpc_send_kxr_auth continuation).
5 *
6 * HOW: tpc_outbound_gsi is a flat orchestrator over four single-purpose helpers, each
7 * owning one handshake step and reporting through a shared tpc_certreq_res_t resource
8 * bundle: (1) tpc_certreq_open_creds opens the cert + key BIOs (delegated in-memory blob
9 * OR config PEM files); (2) tpc_certreq_load_chain_key reads the X509 chain + private key
10 * and validates the local credential (skipped for the freshly-minted delegated proxy);
11 * (3) tpc_certreq_build_send parses the source's gsi parms, builds the stock XrdSecgsi
12 * round-1 certreq and sends it as seq=3; (4) tpc_certreq_recv_authmore reads the reply and
13 * enforces status == kXR_authmore with a ≥16-byte body. On success the orchestrator drives
14 * round 2 (tpc_outbound_gsi_exchange, which consumes `body`). Every exit returns through
15 * tpc_outbound_gsi_finish(rc, res): BIO_free(cbio/kbio), sk_X509_free(chain),
16 * EVP_PKEY_free(pkey), free(certreq/body) — every handle is zero-init and each guard is
17 * NULL-safe, so it frees exactly what had been built. Returns NGX_OK on success or an error
18 * code. Caller: tpc/bootstrap.c (tpc_pull_start).
19 * */

36/*
37 * tpc_certreq_res_t — the OpenSSL + wire resources the certreq round acquires.
38 *
39 * WHAT: bundles every handle tpc_outbound_gsi builds during round 1 so the shared
40 * cleanup takes ONE pointer instead of seven positional args.
41 * WHY: the finish helper was a 7-param cleanup ladder; collapsing the handles into a
42 * struct drops it under the param cap while keeping the exact NULL-safe free order.
43 * HOW: zero-initialised at the top of the orchestrator (every field NULL); each step
44 * helper fills the fields it owns; tpc_outbound_gsi_finish frees them all once.
45 */
46typedef struct {
47 BIO *cbio; /* cert PEM stream (file or in-memory delegated blob) */
48 BIO *kbio; /* key PEM stream (file or in-memory delegated blob) */
49 STACK_OF(X509) *chain; /* owns every cert pushed onto it */
50 EVP_PKEY *pkey; /* private key parsed from kbio */
51 u_char *certreq; /* malloc'd round-1 wire message */
52 u_char *body; /* malloc'd server reply body */
53} tpc_certreq_res_t;
54
55/*
56 * tpc_outbound_gsi_finish — release the certreq round's OpenSSL/wire resources
57 * and return rc unchanged.
58 *
59 * Called at every exit of tpc_outbound_gsi (success and each early failure).
60 * Every handle is NULL-initialised at the top of that function and each guard is
61 * NULL-safe, so invoking this at any point frees exactly what had been built so
62 * far — which lets the caller exit with a plain `return` instead of a shared
63 * goto/label. sk_X509_free frees the stack and every cert it owns.
64 */
65static int
66tpc_outbound_gsi_finish(int rc, tpc_certreq_res_t *res)
67{
68 if (res->cbio) BIO_free(res->cbio);
69 if (res->kbio) BIO_free(res->kbio);
70 if (res->chain) sk_X509_free(res->chain);
71 if (res->pkey) EVP_PKEY_free(res->pkey);
72 if (res->certreq) free(res->certreq);
73 if (res->body) free(res->body);
74 return rc;
75}

77/*
78 * tpc_certreq_open_creds — open the cert + key PEM BIOs for the certreq round.
79 *
80 * WHAT: fills res->cbio/res->kbio with read-only PEM streams over either the
81 * §F6 delegated in-memory credential blob or the config brix_certificate /
82 * brix_certificate_key files, plus NUL-terminated copies of those file paths.
83 * WHY: the outbound side must present an identity; delegation authenticates AS
84 * THE USER with the captured proxy, otherwise as the gateway's configured cert.
85 * HOW: if a delegated blob is present, two BIO_new_mem_buf views over the same
86 * blob (one yields the cert chain, the other the key) and blank the unused path
87 * buffers; else validate + copy the two ngx_str_t config paths into the caller's
88 * PATH_MAX buffers via brix_str_cbuf and BIO_new_file("r") each. Returns 0 on
89 * success, or -1 with t->err_msg / t->xrd_error set on a bad/empty config path.
90 * A NULL BIO (unreadable file) is left for the caller to detect and clean up.
91 */
92static int
93tpc_certreq_open_creds(brix_tpc_pull_t *t, ngx_stream_brix_srv_conf_t *conf,
94 tpc_certreq_res_t *res, u_char *cert_path, u_char *key_path)
95{
96 /*
97 * §F6: when proxy delegation captured the user's proxy, authenticate to the
98 * source AS THE USER with that in-memory credential (proxy cert + issuer chain
99 * + key, PEM) instead of the gateway's brix_certificate. Two BIOs over the
100 * same blob: one yields the cert chain (stops at the trailing key block), the
101 * other the private key (skips the cert blocks). The blob is owned by the pull
102 * task (freed in thread.c).
103 */
104 if (t->deleg_cred_pem != NULL && t->deleg_cred_len > 0) {
105 cert_path[0] = '\0'; /* unused on this path; silence later references */
106 key_path[0] = '\0';
107 res->cbio = BIO_new_mem_buf(t->deleg_cred_pem, (int) t->deleg_cred_len);
108 res->kbio = BIO_new_mem_buf(t->deleg_cred_pem, (int) t->deleg_cred_len);
109 return 0;
110 }
111
112 /* WHY: paths come from config as length-counted ngx_str_t; reject empty
113 * paths and any that would not fit (with room for the NUL) in the local
114 * PATH_MAX buffers — brix_str_cbuf() copies + NUL-terminates and
115 * returns NULL when the result would not fit. */
116 if (conf->certificate.len == 0 || conf->certificate_key.len == 0
117 || brix_str_cbuf((char *) cert_path, PATH_MAX,
118 &conf->certificate) == NULL
119 || brix_str_cbuf((char *) key_path, PATH_MAX,
120 &conf->certificate_key) == NULL)
121 {
122 snprintf(t->err_msg, sizeof(t->err_msg),
123 "TPC GSI outbound needs brix_certificate and "
124 "brix_certificate_key");
125 t->xrd_error = kXR_ArgInvalid;
126 return -1;
127 }
128
129 /* Open read-only PEM streams over the cert and key files. */
130 res->cbio = BIO_new_file((char *) cert_path, "r");
131 res->kbio = BIO_new_file((char *) key_path, "r");
132 return 0;
133}

135/*
136 * tpc_certreq_load_chain_key — read the X509 chain + private key and validate.
137 *
138 * WHAT: parses every cert in res->cbio into res->chain (leaf first), reads the
139 * private key from res->kbio into res->pkey, and — on the non-delegated path —
140 * runs the local proxy PEM through brix_tpc_credential_validate.
141 * WHY: an empty chain or unreadable key cannot authenticate us; the pre-flight
142 * credential check fails a bad local proxy fast (kXR_AuthFailed) rather than
143 * mid-handshake. The freshly-minted delegated proxy is already key-verified
144 * (assemble_proxy), so it skips the extra validation.
145 * HOW: sk_X509_push transfers ownership of each cert to the stack, so x is
146 * cleared after each push to avoid a double-free on error. Returns 0 on success,
147 * or -1 with t->err_msg / t->xrd_error set. cert_path/conf feed the credential
148 * validate; the caller owns cleanup via tpc_outbound_gsi_finish.
149 */
150static int
151tpc_certreq_load_chain_key(brix_tpc_pull_t *t, ngx_stream_brix_srv_conf_t *conf,
152 tpc_certreq_res_t *res, u_char *cert_path)
153{
154 X509 *x = NULL; /* scratch cert from each PEM read */
155
156 res->chain = sk_X509_new_null();
157 if (res->chain == NULL) {
158 return -1;
159 }
160
161 /* Read every certificate in the PEM file into the chain (leaf first,
162 * then intermediates). sk_X509_push transfers ownership of x to the
163 * stack, so we clear x to NULL afterward — otherwise an early loop exit
164 * or error path could double-free a cert that the stack already owns. */
165 while ((x = PEM_read_bio_X509(res->cbio, NULL, NULL, NULL)) != NULL) {
166 sk_X509_push(res->chain, x);
167 x = NULL;
168 }
169
170 /* A cert file that parsed to zero certs cannot authenticate us. */
171 if (sk_X509_num(res->chain) == 0) {
172 snprintf(t->err_msg, sizeof(t->err_msg),
173 "TPC GSI certificate file has no X.509 certs");
174 t->xrd_error = kXR_ArgInvalid;
175 return -1;
176 }
177
178 res->pkey = PEM_read_bio_PrivateKey(res->kbio, NULL, NULL, NULL);
179 if (res->pkey == NULL) {
180 snprintf(t->err_msg, sizeof(t->err_msg),
181 "TPC GSI cannot read private key PEM");
182 t->xrd_error = kXR_ArgInvalid;
183 return -1;
184 }
185
186 /* WHY: security gate — before we put our identity on the wire, validate
187 * the proxy PEM (e.g. not-yet-valid / expired / malformed) so a bad local
188 * credential fails fast here with kXR_AuthFailed rather than mid-handshake.
189 * cred wraps the in-memory cert_path buffer; nothing is sent yet.
190 * Skipped on the §F6 delegated-credential path: that proxy is freshly minted
191 * by the client this handshake and already key-verified (assemble_proxy). */
192 if (t->deleg_cred_pem == NULL) {
193 brix_tpc_credential_t cred;
194
195 ngx_memzero(&cred, sizeof(cred));
196 cred.type = BRIX_TPC_CREDENTIAL_PROXY;
197 cred.proxy_pem.data = cert_path;
198 cred.proxy_pem.len = conf->certificate.len;
199
200 if (brix_tpc_credential_validate(
201 &cred, t->c != NULL ? t->c->log : NULL) != NGX_OK)
202 {
203 snprintf(t->err_msg, sizeof(t->err_msg),
204 "TPC GSI credential validation failed");
205 t->xrd_error = kXR_AuthFailed;
206 return -1;
207 }
208 }
209
210 return 0;
211}

213/*
214 * tpc_certreq_build_send — build the stock XrdSecgsi round-1 certreq and send it.
215 *
216 * WHAT: parses the source's advertised gsi parms from the login reply, builds a
217 * REAL certreq (crypto module + version + issuer-hash + client-opts + random
218 * rtag) into res->certreq, and sends it as the third request on the stream.
219 * WHY: a bare "gsi\0"+kXGC_certreq stub is rejected by the source with kXR_error;
220 * the exact bytes the native client sends (via the shared gsi_core kernel) are
221 * required. Advertising a pre-DHsigned version keeps the source on the simpler
222 * UNSIGNED DH path that gsi_outbound_exchange.c implements.
223 * HOW: brix_gsi_parse_parms picks the crypto module + CA chain (defaulting to
224 * "ssl"); brix_gsi_rand fills the session rtag; brix_gsi_build_certreq allocates
225 * the wire message (freed by tpc_outbound_gsi_finish via res->certreq);
226 * tpc_send_kxr_auth frames + sends it as seq=3. Returns 0 on success, or -1 with
227 * t->err_msg / t->xrd_error set on RNG / alloc / send failure.
228 */
229static int
230tpc_certreq_build_send(brix_tpc_pull_t *t, int fd,
231 const u_char *login_body, uint32_t login_dlen, tpc_certreq_res_t *res)
232{
233 /* round 1: kXGC_certreq * Build the GSI credential payload that opens the handshake. Layout (16B):
234 * [0..3] "gsi\0" 4-byte protocol tag (NUL-padded)
235 * [4..7] kXGC_certreq (=1000) opcode: client requests server cert
236 * [8..11] kXRS_none (=0) bucket type 0 = end-of-message marker
237 * [12..15] 0 that terminator bucket's length (0)
238 * All multi-byte fields are big-endian via tpc_put_u32. crlen = 4+4+8 so
239 * the second tpc_put_u32 below writes into bytes [12..15] of the buffer. */
240 size_t crlen = 0;
241 uint32_t version = 0;
242 char crypto[64];
243 char ca[256];
244
245 /* Build a REAL certreq (the stock XrdSecgsi round-1: crypto module +
246 * version + issuer-hash + client-opts + random rtag) via the shared
247 * gsi_core kernel — the exact bytes the native client sends. A bare
248 * "gsi\0"+kXGC_certreq stub is rejected by the source with kXR_error.
249 *
250 * Parse the source's advertised gsi parms (v:/c:/ca:) from the login
251 * reply body ("<session id><&P=gsi,...>") so it picks the right crypto
252 * module + CA chain, exactly like client/lib/sec/sec_gsi.c. */
253 crypto[0] = '\0';
254 ca[0] = '\0';
255 if (login_dlen > BRIX_SESSION_ID_LEN) {
256 brix_gsi_parse_parms(
257 (const char *) login_body + BRIX_SESSION_ID_LEN,
258 &version, crypto, sizeof(crypto), ca, sizeof(ca));
259 }
260 if (crypto[0] == '\0') {
261 ngx_memcpy(crypto, "ssl", 4);
262 }
263 /* Advertise a pre-DHsigned version (< XrdSecgsiVersDHsigned=10400) so the
264 * source uses the simpler UNSIGNED DH path — it sends its DH public as a
265 * plain kXRS_puk blob, which gsi_outbound_exchange.c implements (it does
266 * NOT handle the signed-DH kXRS_cipher path the native client supports).
267 * Proof-of-possession is still enforced (we sign the server rtag below). */
268 version = 10300;
269
270 if (!brix_gsi_rand(t->gsi_rtag, sizeof(t->gsi_rtag))) {
271 snprintf(t->err_msg, sizeof(t->err_msg), "TPC GSI RNG failed");
272 t->xrd_error = kXR_ServerError;
273 return -1;
274 }
275
276 /* clnt_opts 0x80 matches a stock client (delegated-proxy off). The buffer
277 * is malloc'd and freed by tpc_outbound_gsi_finish. */
278 res->certreq = brix_gsi_build_certreq(crypto, version,
279 ca[0] ? ca : NULL, 0x80u,
280 t->gsi_rtag, sizeof(t->gsi_rtag),
281 &crlen);
282 if (res->certreq == NULL) {
283 snprintf(t->err_msg, sizeof(t->err_msg),
284 "TPC GSI cannot build certreq");
285 t->xrd_error = kXR_NoMemory;
286 return -1;
287 }
288
289 /* seq=3: third request on this stream (protocol=1, login=2). */
290 if (tpc_send_kxr_auth(t, fd, 3, res->certreq, (uint32_t) crlen) != 0) {
291 return -1;
292 }
293
294 return 0;
295}

297/*
298 * tpc_certreq_recv_authmore — read the certreq reply and enforce kXR_authmore.
299 *
300 * WHAT: receives one server reply into res->body and validates the server is
301 * continuing the handshake, returning the body length through *dlen_out.
302 * WHY: the server must answer certreq with kXR_authmore (=4002) — any other
303 * status (early kXR_ok, kXR_error, …) means we cannot proceed; a body < 16 bytes
304 * is too short to hold the cert + CA bucket the next round parses.
305 * HOW: tpc_recv_response malloc's res->body (freed by tpc_outbound_gsi_finish);
306 * reject a recv failure (kXR_ServerError) and a non-authmore / undersized reply
307 * (kXR_AuthFailed). Returns 0 on success, or -1 with t->err_msg / t->xrd_error set.
308 */
309static int
310tpc_certreq_recv_authmore(brix_tpc_pull_t *t, int fd,
311 tpc_certreq_res_t *res, uint32_t *dlen_out)
312{
313 uint16_t status;
314 uint32_t dlen;
315
316 /* Read the server's reply to certreq. body is malloc'd by the helper and
317 * freed at finish; pre-clearing it keeps that cleanup safe on recv failure. */
318 res->body = NULL;
319 if (tpc_recv_response(t, fd, &status, &res->body, &dlen) != 0) {
320 snprintf(t->err_msg, sizeof(t->err_msg),
321 "TPC GSI kXGS_cert recv failed");
322 t->xrd_error = kXR_ServerError;
323 return -1;
324 }
325
326 /* Security check: the server must answer with kXR_authmore (=4002),
327 * signalling "continue authenticating" — any other status (kXR_ok early,
328 * kXR_error, etc.) means we cannot proceed and is treated as auth failure.
329 * dlen < 16 would be too short to hold the cert + CA bucket the next round
330 * (gsi_outbound_exchange.c) parses, so reject undersized bodies here. */
331 if (status != kXR_authmore || res->body == NULL || dlen < 16) {
332 snprintf(t->err_msg, sizeof(t->err_msg),
333 "TPC GSI expected kXR_authmore after certreq (status=%u)",
334 (unsigned) status);
335 t->xrd_error = kXR_AuthFailed;
336 return -1;
337 }
338
339 *dlen_out = dlen;
340 return 0;
341}

343/* WHAT: Initiates GSI auth handshake on TPC pull socket — read cert/key PEM, send kXGC_certreq wire message, verify kXR_authmore response. */
344int
345tpc_outbound_gsi(brix_tpc_pull_t *t, int fd,
346 const u_char *login_body, uint32_t login_dlen)
347{
348 ngx_stream_brix_srv_conf_t *conf = t->conf;
349 u_char cert_path[PATH_MAX]; /* NUL-terminated copies of the */
350 u_char key_path[PATH_MAX]; /* ngx_str_t config paths */
351 tpc_certreq_res_t res = { 0 }; /* all handles zero-init, NULL-safe cleanup */
352 uint32_t dlen = 0;
353 int rc = -1; /* default = failure; set 0 only on success */
354
355 if (tpc_certreq_open_creds(t, conf, &res, cert_path, key_path) != 0) {
356 return tpc_outbound_gsi_finish(rc, &res);
357 }
358 if (res.cbio == NULL || res.kbio == NULL) {
359 snprintf(t->err_msg, sizeof(t->err_msg),
360 "TPC GSI cannot read certificate or key PEM");
361 t->xrd_error = kXR_IOError;
362 return tpc_outbound_gsi_finish(rc, &res);
363 }
364
365 if (tpc_certreq_load_chain_key(t, conf, &res, cert_path) != 0) {
366 return tpc_outbound_gsi_finish(rc, &res);
367 }
368
369 if (tpc_certreq_build_send(t, fd, login_body, login_dlen, &res) != 0) {
370 return tpc_outbound_gsi_finish(rc, &res);
371 }
372
373 if (tpc_certreq_recv_authmore(t, fd, &res, &dlen) != 0) {
374 return tpc_outbound_gsi_finish(rc, &res);
375 }
376
377 /* Round 1 complete: server is mid-handshake. Drive round 2 (the DH key
378 * exchange + encrypted client cert) HERE — this call was previously missing,
379 * so tpc_outbound_gsi returned "success" after only certreq and the source
380 * saw the destination as unauthenticated ("user not authenticated").
381 *
382 * tpc_outbound_gsi_exchange consumes `res.body` (frees it on every exit — see
383 * the matching change in gsi_outbound_exchange.c), and does NOT free
384 * chain/pkey/certreq/cbio/kbio (we do, below). Clear res.body afterwards so
385 * tpc_outbound_gsi_finish cannot double-free it. The leaf-cert arg is NULL:
386 * the chain owns every cert (the round-1 read left the scratch x at NULL). */
387 rc = tpc_outbound_gsi_exchange(t, fd, res.body, dlen, NULL, res.chain,
388 res.pkey, res.certreq, res.cbio, res.kbio);
389 res.body = NULL;
390
391 /* Success exit: tpc_outbound_gsi_finish releases every resource (each guard
392 * is NULL-safe) and returns rc — the same cleanup every error path uses. */
393 return tpc_outbound_gsi_finish(rc, &res);
394}
