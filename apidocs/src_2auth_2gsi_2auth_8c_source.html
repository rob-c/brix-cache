9/*
10 * gsi_promote_fullproxy — validate an OPTIONAL client-pushed full proxy
11 * (kXRS_x509_fullproxy, phase-70 §5.1) captured during kXGC_cert and, if it
12 * passes, promote its bytes to ctx->deleg_proxy_pem for backend PASSTHROUGH.
13 *
14 * WHY: A node can only replay x509 upstream if the user voluntarily hands over
15 * a full proxy (chain + private key). This is the wire-receive + identity
16 * gate for that opt-in: it MUST NOT let a session present a proxy for a
17 * different identity, and MUST NOT accept a key over cleartext.
18 *
19 * HOW: Runs only after the front door proved the GSI identity (ctx->login.dn
20 * is set). Rejects the login when the captured proxy fails any check;
21 * silently no-ops when nothing was captured (the common case).
22 * Checks: (1) transport is TLS (roots://) — a private key never crosses
23 * cleartext; (2) the PEM parses to a cert chain AND a private key;
24 * (3) the supplied leaf/EEC subject is a parent of the authenticated
25 * proxy DN (ctx->login.dn begins with it) — no identity swap. On success
26 * the raw bytes move to ctx->deleg_proxy_pem (borrowed by the binder).
27 * Returns NGX_OK to proceed, NGX_ERROR to reject the login.
28 */
29static ngx_int_t
30gsi_promote_fullproxy(brix_ctx_t *ctx, ngx_connection_t *c)
31{
32 BIO *bio;
33 X509 *leaf;
34 EVP_PKEY *key = NULL;
35 char *subj;
36 int dn_ok;
37
38 if (ctx->gsi.client_fullproxy_pem == NULL) {
39 return NGX_OK; /* nothing supplied — normal */
40 }
41
42 /* (1) TLS-only: a full proxy carries a private key; never over cleartext. */
43 if (c->ssl == NULL) {
44 ngx_log_error(NGX_LOG_WARN, c->log, 0,
45 "brix: GSI full-proxy passthrough rejected: "
46 "supplied over cleartext (roots:// required)");
47 return NGX_ERROR;
48 }
49
50 /* (2) Parse the chain leaf and the private key from the same PEM. */
51 bio = BIO_new_mem_buf(ctx->gsi.client_fullproxy_pem,
52 (int) ctx->gsi.client_fullproxy_len);
53 if (bio == NULL) {
54 return NGX_ERROR;
55 }
56 leaf = PEM_read_bio_X509(bio, NULL, NULL, NULL);
57 if (leaf != NULL) {
58 (void) BIO_reset(bio);
59 key = PEM_read_bio_PrivateKey(bio, NULL, NULL, NULL);
60 }
61 BIO_free(bio);
62 if (leaf == NULL || key == NULL) {
63 X509_free(leaf);
64 EVP_PKEY_free(key);
65 ngx_log_error(NGX_LOG_WARN, c->log, 0,
66 "brix: GSI full-proxy passthrough rejected: "
67 "PEM missing cert chain or private key");
68 return NGX_ERROR;
69 }
70 EVP_PKEY_free(key); /* only presence is checked here */
71
72 /* (3) Identity gate: the authenticated proxy DN must sit beneath the
73 * supplied leaf/EEC subject — the user may only pass through a proxy for
74 * their own identity, never elevate to another. */
75 subj = X509_NAME_oneline(X509_get_subject_name(leaf), NULL, 0);
76 dn_ok = (subj != NULL
77 && ngx_strncmp(ctx->login.dn, subj, ngx_strlen(subj)) == 0);
78 OPENSSL_free(subj);
79 X509_free(leaf);
80
81 if (!dn_ok) {
82 ngx_log_error(NGX_LOG_WARN, c->log, 0,
83 "brix: GSI full-proxy passthrough rejected: "
84 "supplied proxy identity does not match authenticated DN");
85 return NGX_ERROR;
86 }
87
88 /* Passed: hand the bytes to the session ctx for the root:// VFS binder.
89 * Copy into c->pool so the lifetime matches the rest of the session ctx,
90 * then release the heap capture. Never logged. */
91 ctx->deleg_proxy_pem.data = ngx_pnalloc(c->pool,
92 ctx->gsi.client_fullproxy_len);
93 if (ctx->deleg_proxy_pem.data == NULL) {
94 return NGX_ERROR;
95 }
96 ngx_memcpy(ctx->deleg_proxy_pem.data, ctx->gsi.client_fullproxy_pem,
97 ctx->gsi.client_fullproxy_len);
98 ctx->deleg_proxy_pem.len = ctx->gsi.client_fullproxy_len;
99
100 OPENSSL_cleanse(ctx->gsi.client_fullproxy_pem, ctx->gsi.client_fullproxy_len);
101 free(ctx->gsi.client_fullproxy_pem);
102 ctx->gsi.client_fullproxy_pem = NULL;
103 ctx->gsi.client_fullproxy_len = 0;
104
105 ngx_log_error(NGX_LOG_INFO, c->log, 0,
106 "brix: GSI full-proxy passthrough accepted for backend "
107 "delegation (identity verified)");
108 return NGX_OK;
109}

111/*
112 * Phase 51 (E4): per-worker in-flight GSI-handshake gauge. A GSI handshake is
113 * multi-round-trip (certreq → cert), so many can be parked in progress at once,
114 * each holding ephemeral-DH + chain-verify CPU state. Under a flood of
115 * simultaneous handshakes that buries the single-threaded event loop, this caps
116 * the number admitted concurrently per worker and sheds the excess with kXR_wait
117 * (the client waits and retries). Cache hits / already-authed sessions never
118 * touch this. Released EXACTLY ONCE — at auth completion OR at disconnect (the
119 * guaranteed funnel) — gated by ctx->login.gsi_counted, so the gauge can never leak and
120 * wedge auth. Lock-free: per-worker, event-loop only.
121 */
122static ngx_uint_t brix_gsi_inflight;
123
124ngx_int_t
125brix_gsi_inflight_admit(brix_ctx_t *ctx, ngx_int_t cap)
126{
127 if (ctx->login.gsi_counted) {
128 return 1; /* already counted this handshake */
129 }
130 if (cap > 0 && brix_gsi_inflight >= (ngx_uint_t) cap) {
131 return 0; /* over the cap — shed */
132 }
133 brix_gsi_inflight++;
134 ctx->login.gsi_counted = 1;
135 return 1;
136}
137
138void
139brix_gsi_inflight_release(brix_ctx_t *ctx)
140{
141 if (ctx->login.gsi_counted) {
142 if (brix_gsi_inflight > 0) {
143 brix_gsi_inflight--;
144 }
145 ctx->login.gsi_counted = 0;
146 }
147}

149/*
150 * brix_gsi_complete_auth — finalize a successful GSI authentication once the
151 * client's DN is known: per-identity rate limit, auth_done, identity/session
152 * registration, and metrics. Factored so both the normal kXGC_cert path and the
153 * §F6 delegation path (which completes only after kXGC_sigpxy) share one
154 * completion. Returns via BRIX_RETURN_OK / BRIX_RETURN_ERR.
155 */
156static ngx_int_t
157brix_gsi_complete_auth(brix_ctx_t *ctx, ngx_connection_t *c,
158 ngx_stream_brix_srv_conf_t *conf)
159{
160 /* Phase 20: per-identity request rate limit, applied once the DN is known. */
161 if (conf->rate_limit.kv != NULL) {
162 const char *rl_id = conf->rate_limit.key_ip ? ctx->login.peer_ip : ctx->login.dn;
163
164 if (brix_rate_limit_check(&conf->rate_limit, rl_id,
165 ngx_strlen(rl_id)) != NGX_OK)
166 {
167 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "gsi",
168 kXR_NotAuthorized, "rate limit exceeded");
169 }
170 }
171
172 ctx->login.auth_done = 1;
173 brix_gsi_inflight_release(ctx); /* E4: handshake done — free the slot */
174 if (ctx->identity != NULL) {
175 if (brix_identity_set_dn(ctx->identity, c->pool, ctx->login.dn,
176 BRIX_AUTHN_GSI) != NGX_OK
177 || brix_identity_set_vos_csv(ctx->identity, c->pool,
178 ctx->login.vo_list) != NGX_OK)
179 {
180 return brix_send_error(ctx, c, kXR_NoMemory,
181 "identity allocation failed");
182 }
183 }
184 brix_session_register(ctx->login.sessid, ctx->login.dn, ctx->login.vo_list, 0);
185
186 /* Track unique user and VO at auth completion. */
187 {
188 ngx_brix_metrics_t *shm = brix_metrics_shared();
189 if (shm != NULL) {
190 size_t vo_len = strlen(ctx->login.primary_vo);
191 if (vo_len > 0 && vo_len < sizeof(ctx->login.primary_vo)) {
192 brix_track_vo_activity(shm, ctx->login.primary_vo, 0, 0);
193 ngx_uint_t vi;
194 for (vi = 0; vi < BRIX_VO_MAX_TRACKED; vi++) {
195 if (ngx_strncmp(shm->vo_global.slots[vi].name, ctx->login.primary_vo,
196 BRIX_VO_NAME_LEN) == 0)
197 {
198 BRIX_ATOMIC_INC(&shm->vo_global.slots[vi].requests_total);
199 break;
200 }
201 }
202 }
203 brix_track_unique_user(shm, ctx->login.dn, strlen(ctx->login.dn));
204 }
205 }
206
207 {
208 char dn_log[1024];
209
210 brix_sanitize_log_string(ctx->login.dn, dn_log, sizeof(dn_log));
211 ngx_log_error(NGX_LOG_INFO, c->log, 0,
212 "brix: GSI auth OK dn=\"%s\"", dn_log);
213 }
214
215 BRIX_RETURN_OK(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "gsi", 0);
216}

218/*
219 * gsi_auth_cred_route_t — one kXR_auth credtype dispatch row.
220 *
221 * WHY: kXR_auth multiplexes every credential protocol through one opcode;
222 * a descriptor table keeps the routing flat and auditable instead of a
223 * branch ladder per protocol.
224 *
225 * HOW: `name` is the wire credential tag; only the first `cmp_len` of the
226 * 4 wire bytes are significant (3-char tags accept any 4th byte, the
227 * stock-client behavior). `mode`/`mode_alt` are the brix_auth modes
228 * that admit the protocol (equal when only one mode does); `deny_msg`
229 * is the kXR_NotAuthorized text when the mode is off. `handler` is the
230 * protocol's own kXR_auth handler; NULL marks the GSI row, whose step
231 * machine lives in this file (routing falls through to the caller).
232 */
233typedef struct {
234 const char *name; /* credtype tag ("ztn", ...) */
235 size_t cmp_len; /* significant tag bytes (3|4) */
236 ngx_uint_t mode; /* admitting BRIX_AUTH_* mode */
237 ngx_uint_t mode_alt; /* second admitting mode */
238 const char *deny_msg; /* error when mode is off */
239 ngx_int_t (*handler)(brix_ctx_t *ctx, ngx_connection_t *c,
240 ngx_stream_brix_srv_conf_t *conf);
241} gsi_auth_cred_route_t;

261/*
262 * gsi_auth_route_credtype — route a kXR_auth request by its 4-byte credtype.
263 *
264 * WHY: Non-GSI protocols (ztn/sss/unix/krb5/host/pwd) each have their own
265 * handler; only GSI continues into this file's DH step machine.
266 * Isolating the routing keeps the dispatcher a pure step machine.
267 *
268 * HOW: Reads the credtype from the wire, walks gsi_auth_cred_routes in
269 * order, enforces the configured auth mode (deny → kXR_NotAuthorized
270 * with the row's message), and invokes the row handler. Unknown tags
271 * are logged (sanitized) and denied. Returns NGX_DONE with *out_rc set
272 * to the final result, or NGX_OK when the credtype is GSI and GSI auth
273 * is enabled (the caller runs the GSI steps).
274 */
275static ngx_int_t
276gsi_auth_route_credtype(brix_ctx_t *ctx, ngx_connection_t *c,
277 ngx_stream_brix_srv_conf_t *conf, ngx_int_t *out_rc)
278{
279 char credtype[5];
280 char safe_credtype[32];
281 ngx_uint_t i;
282
283 ngx_memcpy(credtype, ctx->recv.cur_body + 12, 4);
284 credtype[4] = '\0';
285 brix_sanitize_log_string(credtype, safe_credtype,
286 sizeof(safe_credtype));
287
288 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
289 "brix: kXR_auth credtype=\"%s\" payloadlen=%d",
290 safe_credtype, (int) ctx->recv.cur_dlen);
291
292 for (i = 0;
293 i < sizeof(gsi_auth_cred_routes) / sizeof(gsi_auth_cred_routes[0]);
294 i++)
295 {
296 const gsi_auth_cred_route_t *rt = &gsi_auth_cred_routes[i];
297
298 if (ngx_memcmp(credtype, rt->name, rt->cmp_len) != 0) {
299 continue;
300 }
301
302 if (conf->auth != rt->mode && conf->auth != rt->mode_alt) {
303 *out_rc = brix_send_error(ctx, c, kXR_NotAuthorized,
304 rt->deny_msg);
305 return NGX_DONE;
306 }
307
308 if (rt->handler == NULL) {
309 return NGX_OK; /* GSI — continue in this file */
310 }
311
312 *out_rc = rt->handler(ctx, c, conf);
313 return NGX_DONE;
314 }
315
316 ngx_log_error(NGX_LOG_WARN, c->log, 0,
317 "brix: kXR_auth unknown credtype=\"%s\"",
318 safe_credtype);
319 *out_rc = brix_send_error(ctx, c, kXR_NotAuthorized,
320 "unsupported credential type");
321 return NGX_DONE;
322}

324/*
325 * gsi_auth_step_certreq — GSI step 1 (kXGC_certreq): send the server cert.
326 *
327 * WHY: Round 1 of the two-round DH exchange; the server answers with its
328 * certificate + DH public so the client never sends its proxy in clear.
329 *
330 * HOW: E4: admit this new handshake under the per-worker in-flight cap; shed
331 * the excess with kXR_wait so a handshake flood cannot bury the loop.
332 */
333static ngx_int_t
334gsi_auth_step_certreq(brix_ctx_t *ctx, ngx_connection_t *c,
335 ngx_stream_brix_srv_conf_t *conf)
336{
337 if (!brix_gsi_inflight_admit(ctx, conf->gsi_max_inflight)) {
338 ngx_log_error(NGX_LOG_WARN, c->log, 0,
339 "brix: GSI handshake shed — %i concurrent in-flight "
340 "(cap reached); asking client to retry",
341 conf->gsi_max_inflight);
342 return brix_send_wait(ctx, c, 3);
343 }
344 return brix_gsi_send_cert(ctx, c);
345}

391/*
392 * gsi_cert_verify — verify the parsed client chain against the CA store.
393 *
394 * WHY: The chain-verify call needs the non-leaf certs split out as the
395 * untrusted stack; isolating the dup/verify/free sequence keeps the
396 * cert step linear.
397 *
398 * HOW: Builds the untrusted stack (chain minus leaf) when present, runs
399 * brix_gsi_verify_chain (RFC 3820 proxies accepted, DN/VOMS material
400 * returned in *verify_res), frees the dup, and returns the verify rc
401 * (brix_gsi_verify_chain already logged any specific error).
402 */
403static ngx_int_t
404gsi_cert_verify(ngx_connection_t *c, ngx_stream_brix_srv_conf_t *conf,
405 STACK_OF(X509) *chain, X509 *leaf,
406 brix_gsi_verify_result_t *verify_res)
407{
408 STACK_OF(X509) *untrusted = NULL;
409 ngx_int_t gsi_rc;
410
411 if (sk_X509_num(chain) > 1) {
412 untrusted = sk_X509_dup(chain);
413 sk_X509_delete(untrusted, 0);
414 }
415
416 gsi_rc = brix_gsi_verify_chain(c->log, conf->gsi_store,
417 leaf, untrusted, 0, verify_res,
418 0 /* GSI: accept RFC 3820 proxies */);
419
420 if (untrusted) {
421 sk_X509_free(untrusted);
422 }
423
424 return gsi_rc;
425}

490/*
491 * gsi_auth_step_cert — GSI step 2 (kXGC_cert): the client's proxy chain.
492 *
493 * WHY: Round 2 of the DH exchange carries the client's encrypted proxy
494 * chain — the security-load-bearing heart of GSI auth: parse, chain
495 * verify (+ optional OCSP), DN capture, optional full-proxy
496 * passthrough, VOMS extraction, then either begin §F6 delegation or
497 * complete the auth.
498 *
499 * HOW: Linear pipeline with early-return denies (gsi_cert_deny frees the
500 * chain on every failure). The ephemeral DH key is freed as soon as
501 * parse_x509 has consumed it. When tpc_delegate is on and a session
502 * cipher was persisted, begin_delegation sends kXGS_pxyreq
503 * (kXR_authmore) and auth completes when kXGC_sigpxy arrives;
504 * otherwise brix_gsi_complete_auth finalizes immediately.
505 */
506static ngx_int_t
507gsi_auth_step_cert(brix_ctx_t *ctx, ngx_connection_t *c,
508 ngx_stream_brix_srv_conf_t *conf)
509{
510 STACK_OF(X509) *chain;
511 X509 *leaf;
512 brix_gsi_verify_result_t verify_res;
513
514 chain = brix_gsi_parse_x509(ctx, c);
515
516 if (ctx->gsi.dh_key) {
517 EVP_PKEY_free(ctx->gsi.dh_key);
518 ctx->gsi.dh_key = NULL;
519 }
520
521 if (chain == NULL) {
522 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "gsi",
523 kXR_NotAuthorized, "cannot parse GSI credential");
524 }
525
526 leaf = sk_X509_value(chain, 0);
527
528 if (gsi_cert_verify(c, conf, chain, leaf, &verify_res) != NGX_OK) {
529 /* brix_gsi_verify_chain already logged the specific error */
530 return gsi_cert_deny(ctx, c, chain,
531 "certificate verification failed",
532 "certificate verification failed");
533 }
534
535 /*
536 * OCSP revocation check (Feature 8e).
537 * The certificate chain is still valid at this point: leaf is chain[0],
538 * its issuer is chain[1] (may be NULL for single-cert chains).
539 */
540 if (conf->ocsp.enable) {
541 X509 *issuer = (sk_X509_num(chain) > 1)
542 ? sk_X509_value(chain, 1) : NULL;
543 if (brix_ocsp_check_cert(c->log, leaf, issuer,
544 (int)conf->ocsp.soft_fail,
545 (int)conf->ocsp.require_nonce) != 0)
546 {
547 return gsi_cert_deny(ctx, c, chain,
548 "OCSP check failed",
549 "OCSP certificate check failed");
550 }
551 }
552
553 gsi_cert_capture_dn(ctx, c, &verify_res);
554
555 /* phase-70 §5.1: validate + promote an OPTIONAL client-pushed full proxy
556 * now that the identity DN is known. A supplied-but-invalid proxy fails the
557 * login (never silently downgrades to service cred). */
558 if (gsi_promote_fullproxy(ctx, c) != NGX_OK) {
559 return gsi_cert_deny(ctx, c, chain,
560 "full-proxy passthrough rejected",
561 "full-proxy passthrough rejected");
562 }
563
564 gsi_cert_extract_voms(ctx, c, conf, leaf, chain);
565
566 /*
567 * §F6 X.509 proxy delegation: when enabled, run an extra handshake round to
568 * capture the client's delegated proxy BEFORE completing auth. The DN/VOMS are
569 * already extracted (above) and the chain/leaf are still valid here; the GSI
570 * session cipher was persisted by parse_x509. begin_delegation sends
571 * kXGS_pxyreq (kXR_authmore) and auth completes when kXGC_sigpxy arrives.
572 */
573 if (conf->tpc_delegate && !ctx->gsi.deleg_await && ctx->gsi.sess_keylen > 0) {
574 ngx_int_t drc = brix_gsi_begin_delegation(ctx, c, conf, leaf, chain);
575
576 sk_X509_pop_free(chain, X509_free);
577 if (drc != NGX_OK) {
578 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "gsi",
579 kXR_NotAuthorized, "GSI proxy delegation failed");
580 }
581 return NGX_OK; /* kXGS_pxyreq sent; auth completes on kXGC_sigpxy */
582 }
583
584 sk_X509_pop_free(chain, X509_free);
585 return brix_gsi_complete_auth(ctx, c, conf);
586}

588/* brix_handle_auth_inner — the kXR_auth dispatcher: route by the 4-byte credtype
589 * (ztn→token, sss→SSS, gsi→GSI, ... — gsi_auth_route_credtype), requiring
590 * ctx->login.logged_in first. The GSI path runs the two-round DH exchange
591 * (kXGC_certreq → server cert via gsi_auth_step_certreq; kXGC_cert →
592 * encrypted proxy chain via gsi_auth_step_cert) so the client's cert is never
593 * sent in clear, verifies the chain against the CA store with
594 * X509_V_FLAG_ALLOW_PROXY_CERTS (+ optional OCSP), extracts the DN and optional
595 * VOMS VO membership, then finalizes via brix_gsi_complete_auth (deferred to
596 * gsi_auth_step_sigpxy when §F6 delegation is in flight). */
597static ngx_int_t
598brix_handle_auth_inner(brix_ctx_t *ctx, ngx_connection_t *c)
599{
600 ngx_stream_brix_srv_conf_t *conf;
601 uint32_t gsi_step;
602 ngx_int_t rc;
603
604 if (!ctx->login.logged_in) {
605 return brix_send_error(ctx, c, kXR_NotAuthorized,
606 "login required before auth");
607 }
608
609 conf = ngx_stream_get_module_srv_conf(ctx->session,
610 ngx_stream_brix_module);
611
612 if (conf->auth == BRIX_AUTH_NONE) {
613 ctx->login.auth_done = 1;
614 return brix_send_ok(ctx, c, NULL, 0);
615 }
616
617 rc = NGX_ERROR;
618 if (gsi_auth_route_credtype(ctx, c, conf, &rc) == NGX_DONE) {
619 return rc;
620 }
621
622 if (conf->gsi_store == NULL) {
623 return brix_send_error(ctx, c, kXR_NotAuthorized,
624 "GSI not configured");
625 }
626
627 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen < 8) {
628 return brix_send_error(ctx, c, kXR_NotAuthorized,
629 "empty GSI credential");
630 }
631
632 ngx_memcpy(&gsi_step, ctx->recv.payload + 4, 4);
633 gsi_step = ntohl(gsi_step);
634
635 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
636 "brix: GSI kXR_auth step=%ud", (unsigned) gsi_step);
637
638 if (gsi_step == (uint32_t) kXGC_certreq) {
639 return gsi_auth_step_certreq(ctx, c, conf);
640 }
641
642 if (gsi_step == (uint32_t) kXGC_sigpxy) {
643 return gsi_auth_step_sigpxy(ctx, c, conf);
644 }
645
646 if (gsi_step != (uint32_t) kXGC_cert) {
647 ngx_log_error(NGX_LOG_WARN, c->log, 0,
648 "brix: unexpected GSI step %ud", (unsigned) gsi_step);
649 return brix_send_error(ctx, c, kXR_NotAuthorized,
650 "unexpected GSI auth step");
651 }
652
653 return gsi_auth_step_cert(ctx, c, conf);
654}

656/* brix_handle_auth — rate-limited public entry point for kXR_auth: reject once
657 * auth_fail_count hits BRIX_MAX_AUTH_ATTEMPTS (brute-force / GSI CPU-amplification
658 * guard), but skip the limit on the GSI certreq round (the server's cert response
659 * is not a credential failure). Delegates to brix_handle_auth_inner, then resets
660 * the counter on success or increments it on failure. */
661ngx_int_t
662brix_handle_auth(brix_ctx_t *ctx, ngx_connection_t *c)
663{
664 ngx_flag_t was_auth_done;
665 ngx_flag_t is_certreq;
666 ngx_int_t rc;
667
668 /*
669 * Phase 33: start each GSI auth round with a clean per-thread OpenSSL error
670 * queue. GSI parsing intentionally provokes benign errors ("invalid key
671 * length", PEM "no start line" at EOF) that the module never clears; a dirty
672 * queue later corrupts nginx's TLS clean-close detection on the shared
673 * worker. Clearing here keeps those benign errors from leaking forward.
674 */
675 ERR_clear_error();
676
677 /* Reject after repeated failures — guards against brute-force attempts
678 * and CPU-amplification via costly GSI/OpenSSL/VOMS operations. */
679 if (ctx->login.auth_fail_count >= BRIX_MAX_AUTH_ATTEMPTS) {
680 ngx_log_error(NGX_LOG_WARN, c->log, 0,
681 "brix: %s: auth attempt limit reached, disconnecting",
682 ctx->login.user);
683 return brix_send_error(ctx, c, kXR_NotAuthorized,
684 "Too many authentication failures");
685 }
686
687 /*
688 * GSI round 1 (kXGC_certreq) and pwd round 1: the server responds with
689 * kXR_authmore (its certificate / its DH public) — a protocol continuation,
690 * not a credential failure, so it must not count toward the attempt limit.
691 */
692 is_certreq = 0;
693 if (ctx->recv.cur_dlen >= 8 && ctx->recv.payload != NULL) {
694 const u_char *ctype = ctx->recv.cur_body + 12;
695 if (ctype[0] == 'g' && ctype[1] == 's' && ctype[2] == 'i') {
696 uint32_t step;
697 ngx_memcpy(&step, ctx->recv.payload + 4, 4);
698 is_certreq = (ntohl(step) == (uint32_t) kXGC_certreq);
699 } else if (ctype[0] == 'p' && ctype[1] == 'w' && ctype[2] == 'd'
700 && ctype[3] == 0) {
701 /* pwd round 1 is the puk-exchange (ctx->pwd.round still 0). */
702 is_certreq = (ctx->pwd.round == 0);
703 }
704 }
705
706 was_auth_done = ctx->login.auth_done;
707 rc = brix_handle_auth_inner(ctx, c);
708
709 if (!is_certreq) {
710 if (!was_auth_done && ctx->login.auth_done) {
711 ctx->login.auth_fail_count = 0; /* successful auth resets the counter */
712 } else if (!ctx->login.auth_done) {
713 ctx->login.auth_fail_count++; /* failed or protocol-level challenge */
714 }
715 }
716
717 return rc;
718}
