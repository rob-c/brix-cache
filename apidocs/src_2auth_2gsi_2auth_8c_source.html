8/*
9 * Phase 51 (E4): per-worker in-flight GSI-handshake gauge. A GSI handshake is
10 * multi-round-trip (certreq → cert), so many can be parked in progress at once,
11 * each holding ephemeral-DH + chain-verify CPU state. Under a flood of
12 * simultaneous handshakes that buries the single-threaded event loop, this caps
13 * the number admitted concurrently per worker and sheds the excess with kXR_wait
14 * (the client waits and retries). Cache hits / already-authed sessions never
15 * touch this. Released EXACTLY ONCE — at auth completion OR at disconnect (the
16 * guaranteed funnel) — gated by ctx->login.gsi_counted, so the gauge can never leak and
17 * wedge auth. Lock-free: per-worker, event-loop only.
18 */
19static ngx_uint_t brix_gsi_inflight;
20
21ngx_int_t
22brix_gsi_inflight_admit(brix_ctx_t *ctx, ngx_int_t cap)
23{
24 if (ctx->login.gsi_counted) {
25 return 1; /* already counted this handshake */
26 }
27 if (cap > 0 && brix_gsi_inflight >= (ngx_uint_t) cap) {
28 return 0; /* over the cap — shed */
29 }
30 brix_gsi_inflight++;
31 ctx->login.gsi_counted = 1;
32 return 1;
33}
34
35void
36brix_gsi_inflight_release(brix_ctx_t *ctx)
37{
38 if (ctx->login.gsi_counted) {
39 if (brix_gsi_inflight > 0) {
40 brix_gsi_inflight--;
41 }
42 ctx->login.gsi_counted = 0;
43 }
44}

389/* brix_handle_auth — rate-limited public entry point for kXR_auth: reject once
390 * auth_fail_count hits BRIX_MAX_AUTH_ATTEMPTS (brute-force / GSI CPU-amplification
391 * guard), but skip the limit on the GSI certreq round (the server's cert response
392 * is not a credential failure). Delegates to brix_handle_auth_inner, then resets
393 * the counter on success or increments it on failure. */
394ngx_int_t
395brix_handle_auth(brix_ctx_t *ctx, ngx_connection_t *c)
396{
397 ngx_flag_t was_auth_done;
398 ngx_flag_t is_certreq;
399 ngx_int_t rc;
400
401 /*
402 * Phase 33: start each GSI auth round with a clean per-thread OpenSSL error
403 * queue. GSI parsing intentionally provokes benign errors ("invalid key
404 * length", PEM "no start line" at EOF) that the module never clears; a dirty
405 * queue later corrupts nginx's TLS clean-close detection on the shared
406 * worker. Clearing here keeps those benign errors from leaking forward.
407 */
408 ERR_clear_error();
409
410 /* Reject after repeated failures — guards against brute-force attempts
411 * and CPU-amplification via costly GSI/OpenSSL/VOMS operations. */
412 if (ctx->login.auth_fail_count >= BRIX_MAX_AUTH_ATTEMPTS) {
413 ngx_log_error(NGX_LOG_WARN, c->log, 0,
414 "brix: %s: auth attempt limit reached, disconnecting",
415 ctx->login.user);
416 return brix_send_error(ctx, c, kXR_NotAuthorized,
417 "Too many authentication failures");
418 }
419
420 /*
421 * GSI round 1 (kXGC_certreq) and pwd round 1: the server responds with
422 * kXR_authmore (its certificate / its DH public) — a protocol continuation,
423 * not a credential failure, so it must not count toward the attempt limit.
424 */
425 is_certreq = 0;
426 if (ctx->recv.cur_dlen >= 8 && ctx->recv.payload != NULL) {
427 const u_char *ctype = ctx->recv.cur_body + 12;
428 if (ctype[0] == 'g' && ctype[1] == 's' && ctype[2] == 'i') {
429 uint32_t step;
430 ngx_memcpy(&step, ctx->recv.payload + 4, 4);
431 is_certreq = (ntohl(step) == (uint32_t) kXGC_certreq);
432 } else if (ctype[0] == 'p' && ctype[1] == 'w' && ctype[2] == 'd'
433 && ctype[3] == 0) {
434 /* pwd round 1 is the puk-exchange (ctx->pwd.round still 0). */
435 is_certreq = (ctx->pwd.round == 0);
436 }
437 }
438
439 was_auth_done = ctx->login.auth_done;
440 rc = brix_handle_auth_inner(ctx, c);
441
442 if (!is_certreq) {
443 if (!was_auth_done && ctx->login.auth_done) {
444 ctx->login.auth_fail_count = 0; /* successful auth resets the counter */
445 } else if (!ctx->login.auth_done) {
446 ctx->login.auth_fail_count++; /* failed or protocol-level challenge */
447 }
448 }
449
450 return rc;
451}
