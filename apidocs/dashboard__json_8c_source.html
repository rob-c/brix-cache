21 /* Two-pass serialise: a NULL-buffer dump returns the exact byte count, then
22 * allocate that and dump for real — sizing the response to the payload.
23 * needed==0 means a dump error. */
24 needed = json_dumpb(root, NULL, 0, JSON_COMPACT);
25 if (needed == 0) {
26 json_decref(root);
27 return NGX_HTTP_INTERNAL_SERVER_ERROR;
28 }
29
30 buf = ngx_palloc(r->pool, needed);
31 if (buf == NULL) {
32 json_decref(root);
33 return NGX_HTTP_INTERNAL_SERVER_ERROR;
34 }
35
36 json_dumpb(root, (char *) buf, needed, JSON_COMPACT);
37 /* root is fully serialised into buf; release it now (we own the ref). */
38 json_decref(root);
39
40 b = ngx_pcalloc(r->pool, sizeof(*b));
41 if (b == NULL) {
42 return NGX_HTTP_INTERNAL_SERVER_ERROR;
43 }
44 b->pos = b->start = buf;
45 b->last = b->end = buf + needed;
46 b->memory = 1;
47 b->last_buf = 1;
48
49 r->headers_out.status = status;
50 r->headers_out.content_length_n = (off_t) needed;
51 r->headers_out.content_type = (ngx_str_t) ngx_string("application/json");
52 r->headers_out.content_type_len = r->headers_out.content_type.len;
53

55 if (cc != NULL) {

59 }

63 return rc;
64 }

67 out.next = NULL;
