2/*
3 * open_resolved_file_internal.h — shared state + cross-file entry points for the
4 * kXR_open resolved-file pipeline (phase-79 file-size split).
5 *
6 * WHAT: Declares the per-open carrier struct `brix_open_args_t` and the handful
7 * of pipeline-stage entry points that are DEFINED in one split translation
8 * unit and CALLED from another. Every open_resolved_file*.c includes this.
9 *
10 * WHY: open_resolved_file.c grew past the 500-line focus limit. It was split
11 * into cohesive stages — staging preflight, the open(2)/driver open ops,
12 * backend/credential dispatch, post-fd handle finalize, and the orchestrator
13 * (which also builds/sends the reply). The stages share exactly one state
14 * object and a small set of boundary functions; those live here so no stage
15 * reaches into another's internals and the linker resolves the cross-calls.
16 *
17 * HOW: The struct is byte-identical to the original file-local definition (pure
18 * relocation). Each declared function is non-static in its defining .c and
19 * carries the same signature it had as a file-local static.
20 *
21 * Requires: open.h (brix_ctx_t / ngx types / brix_open_request_t),
22 * fs/vfs/vfs.h (brix_vfs_ctx_t), fs/backend/sd.h (brix_sd_instance_t),
23 * <sys/stat.h> (struct stat / mode_t), <limits.h> (PATH_MAX) before use.
24 */

33/*
34 * brix_open_args_t — the per-request kXR_open state threaded through the
35 * static open pipeline (phase-72 B5 parameter consolidation).
36 *
37 * WHAT: One carrier for everything brix_open_resolved_file computes once — the
38 * session/config pointers, the decoded kXR_open intent, the staging
39 * decision + POSC/resume temp path, the allocated handle slot, the open
40 * outcome (fd/stat/driver routing) and the kXR_retstat buffer — so the
41 * pipeline stages take this struct instead of many positional parameters.
42 *
43 * WHY: The open path is a linear pipeline (preflight → dispatch → finalize →
44 * response) whose stages share one request's state; passing it positionally
45 * bloated every signature past the complexity gate and made call sites
46 * error-prone (runs of adjacent same-typed flags).
47 *
48 * HOW: brix_open_resolved_file zeroes the struct, fills the request-constant
49 * fields, then each stage reads what it needs and writes only the fields
50 * it owns (preflight: use_resume/stage/posc_temp_path; dispatch:
51 * fd/st/driver_backed/wt_via_stage; finalize: want_stat/statbuf).
52 * Pure signature consolidation — no logic moved.
53 */
