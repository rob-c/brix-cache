1/*
2 * wait41.c — block until an XRootD server accepts connections.
3 *
4 * WHAT: `wait41-brix [--timeout S] [--full] host[:port]` — poll until the server is
5 * reachable (TCP connect; with --full, a complete handshake+login). Exits 0
6 * when ready, non-zero on timeout. The readiness helper the harness wants.
7 * WHY: A tiny front-end over the client transport/session layer. libXrdCl-free.
8 * HOW: Loop brix_tcp_connect (or brix_connect with --full) until a deadline,
9 * sleeping 1s between attempts.
10 */
