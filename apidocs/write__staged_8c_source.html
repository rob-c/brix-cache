1/*
2 * write_staged.c — root:// block-write → whole-object staged-commit adapter.
3 *
4 * WHAT: The write/sync/close hooks for a root:// write handle whose backend LEAF
5 * advertises NO random write (BRIX_SD_CAP_RANDOM_WRITE) and has no pwrite
6 * slot — a whole-object store (sd_http/s3 and any driver whose write is a
7 * single commit-time PUT). brix_open_dispatch_staged put such a handle in
8 * STAGED mode (ctx->files[idx].writer != NULL — a unified brix_vfs_writer
9 * session opened with the per-user credential already resolved). This file
10 * APPENDS each kXR_write/pgwrite block to that session and COMMITS the whole
11 * object on kXR_sync / kXR_close.
12 *
13 * WHY: The block-oriented root:// write model (open-for-write → write/pgwrite at
14 * offsets → sync/close → driver pwrite) requires a random-write driver; a
15 * whole-object backend cannot pwrite, so a root:// upload used to fail EROFS.
16 * GridFTP STOR and (via http_body) WebDAV/S3 PUT stream a block body through
17 * the same unified brix_vfs_writer; this routes the root:// path through it
18 * too, so every filesystem shares one verified-write call to the VFS layer.
19 *
20 * HOW: Uploads are sequential appends, which is the only shape a whole-object
21 * store supports. brix_staged_append refuses an out-of-order offset cleanly
22 * with kXR_Unsupported (comparing against brix_vfs_writer_expected_off before
23 * corrupting the object), then appends via brix_vfs_writer_write. sync/close
24 * call brix_staged_commit_handle → brix_vfs_writer_commit (one whole-object
25 * PUT, plus the optional read-back CRC check when brix_verify_write is on),
26 * which on success consumes the session; brix_free_fhandle aborts an
27 * uncommitted session so no partial object is published. No goto; each stage
28 * is a small single-purpose function with explicit data flow.
29 */

35/*
36 * brix_staged_append — append `len` bytes at `offset` to the write session,
37 * WITHOUT sending a success reply (the caller chooses the reply frame: kXR_ok for
38 * kXR_write, kXR_status for pgwrite).
39 *
40 * Enforces sequential append (offset == the writer's expected offset); an
41 * out-of-order offset is refused with kXR_Unsupported. On any failure the error
42 * reply is sent here and *rc holds its return value. Returns NGX_OK when appended
43 * (no reply sent yet); NGX_ERROR when the caller must return *rc immediately.
44 */
45ngx_int_t
46brix_staged_append(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
47 int64_t offset, const u_char *buf, size_t len, ngx_int_t *rc)
48{
49 brix_file_t *file = &ctx->files[idx];
50 char detail[64];
51
52 snprintf(detail, sizeof(detail), "%lld+%zu", (long long) offset, len);
53
54 /* Sequential-append contract: a whole-object backend cannot honour a random
55 * offset. Refuse an out-of-order block cleanly — checked against the writer's
56 * own cursor so the error is deterministic (kXR_Unsupported) rather than
57 * relying on the writer's EINVAL — before corrupting the object (genuinely
58 * random-offset writes stay unsupported). */
59 if (offset != (int64_t) brix_vfs_writer_expected_off(file->writer)) {
60 brix_log_access(ctx, c, "WRITE", file->path, detail, 0,
61 kXR_Unsupported,
62 "random-offset write to whole-object backend unsupported", 0);
63 BRIX_OP_ERR(ctx, BRIX_OP_WRITE);
64 *rc = brix_send_error(ctx, c, kXR_Unsupported,
65 "random-offset write to whole-object backend unsupported");
66 return NGX_ERROR;
67 }
68
69 if (brix_vfs_writer_write(file->writer, buf, len, offset) != NGX_OK) {
70 const char *ioerr = strerror(errno);
71 brix_log_access(ctx, c, "WRITE", file->path, detail, 0,
72 kXR_IOError, ioerr, 0);
73 BRIX_OP_ERR(ctx, BRIX_OP_WRITE);
74 *rc = brix_send_error(ctx, c, kXR_IOError, ioerr);
75 return NGX_ERROR;
76 }
77
78 file->bytes_written += len;
79 ctx->totals.bytes_written += len;
80 brix_rl_charge_ctx(ctx, len);
81
82 if (file->dashboard_slot >= 0 && ngx_brix_dashboard_shm_zone != NULL) {
83 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
84 file->dashboard_slot,
85 (ngx_atomic_int_t) len,
86 (int64_t) ngx_current_msec);
87 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
88 file->dashboard_slot, "write");
89 }
90 return NGX_OK;
91}

93/*
94 * brix_write_staged_buf — append a decoded block and reply kXR_ok. Used by the
95 * kXR_write path (the pgwrite path appends then sends its own kXR_status frame).
96 */
97ngx_int_t
98brix_write_staged_buf(brix_ctx_t *ctx, ngx_connection_t *c,
99 int idx, int64_t offset, const u_char *buf, size_t len)
100{

126/*
127 * brix_staged_commit_handle — commit the whole staged object (one backend PUT,
128 * plus the read-back CRC check when brix_verify_write is on). The kXR_sync /
129 * kXR_close hook. Idempotent: once committed, a later call is a no-op success (a
130 * client that syncs then closes). On failure sets *err_out and returns NGX_ERROR;
131 * brix_vfs_writer_commit already unlinked any published-then-mismatched object, so
132 * no partial/corrupt object is left behind.
133 */
134ngx_int_t
135brix_staged_commit_handle(brix_ctx_t *ctx, int idx, int *err_out)
136{
137 brix_file_t *file = &ctx->files[idx];
138
139 if (file->writer == NULL || file->staged_committed) {
140 return NGX_OK; /* nothing staged, or already published */
141 }
142
143 if (brix_vfs_writer_commit(file->writer) != NGX_OK) {
144 if (err_out != NULL) {
145 *err_out = errno ? errno : EIO;
146 }
147 return NGX_ERROR;
148 }
149
150 /* A successful commit published (and optionally verified) the object and
151 * consumed the session's staged state; mark committed so brix_free_fhandle's
152 * abort is a no-op and no second commit runs. */
153 file->staged_committed = 1;
154 return NGX_OK;
155}
