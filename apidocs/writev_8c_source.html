6/* brix_writev_body_extra — trailing segment-data length for kXR_writev
7 * WHAT: Validates a dlen-framed write_list descriptor block under the stock
8 * wire contract (dlen frames ONLY the descriptors: non-zero, 16-aligned,
9 * <= BRIX_WRITEV_MAXSEGS entries, descriptors + declared data within
10 * BRIX_MAX_WRITE_PAYLOAD) and reports sum(wlen) — the number of segment-data
11 * bytes the client streams after the request frame.
12 * WHY: Shared by the recv framing (to extend the read obligation so the
13 * streamed data lands in payload_buf before dispatch) and kept next to the
14 * handler that consumes the same layout. Also delegated to by
15 * brix_ckpxeq_body_extra (chkpoint_xeq.c): a kXR_ckpXeq-embedded writev
16 * tunnels this exact contract — its embedded header's dlen frames the
17 * descriptors only, and the segment data streams behind. Returns NGX_OK
18 * with *extra set, or NGX_DECLINED when the block violates the contract (the
19 * caller decides the error; the framing never rejects so the auth gates
20 * still run first). */
21ngx_int_t
22brix_writev_body_extra(const u_char *desc, uint32_t dlen, uint32_t *extra)
23{
24 const write_list *wl = (const write_list *) desc;
25 uint32_t n_segs, i;
26 uint64_t total = 0;
27
28 *extra = 0;
29
30 if (desc == NULL || dlen == 0 || dlen % BRIX_WRITEV_SEGSIZE != 0) {
31 return NGX_DECLINED;
32 }
33
34 n_segs = dlen / BRIX_WRITEV_SEGSIZE;
35 if (n_segs > BRIX_WRITEV_MAXSEGS) {
36 return NGX_DECLINED;
37 }
38
39 for (i = 0; i < n_segs; i++) {
40 total += (uint32_t) ntohl((uint32_t) wl[i].wlen);
41 }
42
43 if ((uint64_t) dlen + total > BRIX_MAX_WRITE_PAYLOAD) {
44 return NGX_DECLINED;
45 }
46
47 *extra = (uint32_t) total;
48 return NGX_OK;
49}

51/* writev_validate_framing — enforce the stock do_WriteV framing contract.
52 * WHAT: Validates the descriptor block that recv left in ctx->recv.payload —
53 * dlen present and a whole number of 16-byte descriptors, segment count within
54 * the vector cap, descriptors + declared data within the aggregate transfer
55 * cap, and the recv-appended trailing byte count matching sum(wlen). On
56 * success it publishes the segment count (*n_segs_out) and total data length
57 * (*total_wlen_out) to the caller.
58 * WHY: This is the FROZEN stock-framing gate — the same contract the recv path
59 * and the ckpXeq-embedded writev depend on. Concentrating it here keeps the
60 * handler flat and the wire-error parity (kXR error code + message + link drop)
61 * byte-identical to the reference server.
62 * HOW: Each violation emits its stock error via brix_send_error and returns
63 * NGX_ERROR (caller drops the link — a doubtful descriptor block makes the
64 * trailing byte count unknowable, so the connection cannot resynchronise). An
65 * all-zero-length vector is the stock immediate-success case: it sends OK and
66 * returns NGX_DONE. A clean, non-empty vector returns NGX_OK. */
67static ngx_int_t
68writev_validate_framing(brix_ctx_t *ctx, ngx_connection_t *c,
69 const write_list *wl, size_t *n_segs_out,
70 uint64_t *total_wlen_out)
71{
72 size_t n_segs, i;
73 uint64_t total_wlen = 0;
74
75 *n_segs_out = 0;
76 *total_wlen_out = 0;
77
78 /* Stock parity: a dlen that is zero or not a whole number of write_list
79 * descriptors is "Write vector is invalid" (kXR_ArgInvalid) + link drop. */
80 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0
81 || ctx->recv.cur_dlen % BRIX_WRITEV_SEGSIZE != 0)
82 {
83 BRIX_OP_ERR(ctx, BRIX_OP_WRITEV);
84 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
85 "Write vector is invalid");
86 return NGX_ERROR;
87 }
88
89 n_segs = ctx->recv.cur_dlen / BRIX_WRITEV_SEGSIZE;
90
91 /* Stock parity: vector count cap (stock maxWvecsz == 1024 == ours). */
92 if (n_segs > BRIX_WRITEV_MAXSEGS) {
93 BRIX_OP_ERR(ctx, BRIX_OP_WRITEV);
94 (void) brix_send_error(ctx, c, kXR_ArgTooLong,
95 "Write vector is too long");
96 return NGX_ERROR;
97 }
98
99 for (i = 0; i < n_segs; i++) {
100 total_wlen += (uint32_t) ntohl((uint32_t) wl[i].wlen);
101 }
102
103 /* Our aggregate transfer cap (stock instead caps each element at its
104 * maxTransz); enforced by the framing BEFORE it reads the data, so a
105 * violation means the trailing bytes were never accepted — drop. */
106 if ((uint64_t) ctx->recv.cur_dlen + total_wlen > BRIX_MAX_WRITE_PAYLOAD) {
107 BRIX_OP_ERR(ctx, BRIX_OP_WRITEV);
108 (void) brix_send_error(ctx, c, kXR_NoMemory,
109 "writev transfer is too large");
110 return NGX_ERROR;
111 }
112
113 /* Defensive: the recv framing extends the body by exactly sum(wlen); a
114 * mismatch means the extension never ran for this frame (unreachable in
115 * normal operation) and the data bytes are not in the buffer. */
116 if (total_wlen != (uint64_t) ctx->recv.cur_body_extra) {
117 BRIX_OP_ERR(ctx, BRIX_OP_WRITEV);
118 (void) brix_send_error(ctx, c, kXR_ArgInvalid,
119 "Write vector is invalid");
120 return NGX_ERROR;
121 }
122
123 /* Stock parity: an all-zero-length vector succeeds immediately, before
124 * any handle validation (do_WriteV returns Send() when maxSZ == 0). */
125 if (total_wlen == 0) {
126 BRIX_OP_OK(ctx, BRIX_OP_WRITEV);
127 (void) brix_send_ok(ctx, c, NULL, 0);
128 return NGX_DONE;
129 }
130
131 *n_segs_out = n_segs;
132 *total_wlen_out = total_wlen;
133 return NGX_OK;
134}

136/* writev_validate_handles — all-or-nothing handle admission for the vector.
137 * WHAT: Verifies every segment's target handle is open and writable before any
138 * byte is written.
139 * WHY: INVARIANT — a bad handle in a later segment must never leave earlier
140 * segments already written; admitting the whole vector up front makes the write
141 * all-or-nothing. Only fhandle[0] is significant (handles are 0..255).
142 * HOW: Returns NGX_OK when every handle passes. On the first failure it emits
143 * the stock error (kXR_FileNotOpen for a stale/closed slot, kXR_NotAuthorized
144 * for a read-only handle) and returns its brix_send_error result for the caller
145 * to propagate unchanged. */
146static ngx_int_t
147writev_validate_handles(brix_ctx_t *ctx, ngx_connection_t *c,
148 const write_list *wl, size_t n_segs)
149{
150 size_t i;
151
152 for (i = 0; i < n_segs; i++) {
153 int idx = (int)(unsigned char) wl[i].fhandle[0];
154
155 /* fd < 0 means the slot is closed/never-opened (stale handle). */
156 if (idx < 0 || idx >= BRIX_MAX_FILES || ctx->files[idx].fd < 0) {
157 BRIX_OP_ERR(ctx, BRIX_OP_WRITEV);
158 return brix_send_error(ctx, c, kXR_FileNotOpen,
159 "invalid file handle in writev");
160 }
161 if (!ctx->files[idx].writable) {
162 BRIX_OP_ERR(ctx, BRIX_OP_WRITEV);
163 return brix_send_error(ctx, c, kXR_NotAuthorized,
164 "file not open for writing");
165 }
166 }
167
168 return NGX_OK;
169}

171/* writev_write_segment — write one segment through the VFS on this thread.
172 * WHAT: Handles a single (fd, offset, data, wlen) segment on the event-loop
173 * thread: replay neutralisation, the VFS write, dirty-extent marking, and the
174 * write-recovery journal update; accumulates the client-visible byte total and
175 * advances *data_ptr past this segment's data block.
176 * WHY: The synchronous fallback body per segment is a self-contained unit; the
177 * loop that drives it stays flat and the replay/short-write parity lives in one
178 * reviewable place.
179 * HOW: Decodes the wire fields, returns NGX_OK for an empty or replayed segment
180 * (already accounted, pointer advanced). On I/O error it emits the stock error
181 * via BRIX_RETURN_ERR (which returns brix_send_error's result) and propagates
182 * it as NGX_ERROR so the caller stops the loop and drops the link; on success
183 * it returns NGX_OK. */
184static ngx_int_t
185writev_write_segment(brix_ctx_t *ctx, ngx_connection_t *c,
186 const write_list *seg, const u_char **data_ptr,
187 size_t *bytes_written_total)
188{
189 int idx = (int)(unsigned char) seg->fhandle[0];
190 int64_t offset = (int64_t) be64toh((uint64_t) seg->offset); /* BE64 */
191 uint32_t wlen = (uint32_t) ntohl((uint32_t) seg->wlen); /* BE32 */
192 ssize_t nw;
193
194 /* Empty segment carries no data block; nothing to advance over. */
195 if (wlen == 0) {
196 return NGX_OK;
197 }
198
199 /* Same replay handling as the AIO path: a range already written on a prior
200 * (dropped) connection is acknowledged as success but not re-issued to
201 * disk. data_ptr must still skip this segment's bytes to stay aligned, and
202 * we count the bytes toward the client-visible total so the reply matches
203 * what the client thinks it sent. */
204 /* phase79-fp: security.ArrayBound — idx (0..255 from fhandle[0]) was already
205 * bounds-checked against BRIX_MAX_FILES by writev_validate_handles before the
206 * sync path runs; the analyzer cannot correlate the separate validation loop. */
207 if (ctx->files[idx].wrts_enabled &&
208 brix_wrts_is_replay(&ctx->files[idx], offset, wlen))
209 {
210 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
211 "brix: writev recovery replay skip offset=%lld len=%u",
212 (long long) offset, wlen);
213 *bytes_written_total += (size_t) wlen;
214 *data_ptr += wlen;
215 return NGX_OK;
216 }
217
218 {
219 brix_vfs_job_t job;
220
221 brix_vfs_job_write_init(&job, ctx->files[idx].fd,
222 (off_t) offset, *data_ptr, (size_t) wlen);
223 brix_vfs_job_set_obj(&job, &ctx->files[idx].sd_obj);
224 brix_vfs_io_execute(&job);
225 nw = job.nio;
226 if (job.io_errno != 0) {
227 errno = job.io_errno;
228 }
229 }
230
231 if (nw < 0) {
232 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITEV, "WRITEV",
233 ctx->files[idx].path, "-",
234 kXR_IOError, strerror(errno));
235 }
236 if ((uint32_t) nw < wlen) {
237 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITEV, "WRITEV",
238 ctx->files[idx].path, "-",
239 kXR_IOError, "short write (disk full?)");
240 }
241
242 ctx->files[idx].bytes_written += (size_t) nw;
243 ctx->totals.bytes_written += (size_t) nw;
244 *bytes_written_total += (size_t) nw;
245 /* write-through cache: flag the just-written extent dirty so it gets
246 * flushed to the backing store (end offset is inclusive: off+nw-1). */
247 if (ctx->files[idx].wt_enabled) {
248 brix_wt_mark_dirty(ctx, idx, offset + (int64_t) nw - 1,
249 (size_t) nw);
250 }
251 /* write-recovery journal: remember this range so a later replay after
252 * reconnect is recognised by brix_wrts_is_replay() above. */
253 if (ctx->files[idx].wrts_enabled) {
254 brix_wrts_record(&ctx->files[idx], offset, (uint32_t) nw);
255 }
256 /* Advance by wlen (== nw here, short writes already errored out). */
257 *data_ptr += wlen;
258 return NGX_OK;
259}

261/* writev_execute_sync — synchronous fallback: write every segment in order.
262 * WHAT: Iterates the vector, writing each segment on the event-loop thread and
263 * accumulating the client-visible byte total.
264 * WHY: Used when no thread pool is configured or the AIO post failed; keeps the
265 * orchestrator flat by owning the segment loop.
266 * HOW: Delegates each segment to writev_write_segment, propagating its NGX_ERROR

268 * the whole vector is written. */
269static ngx_int_t
270writev_execute_sync(const writev_run_t *run, size_t *bytes_written_total)
271{
272 const u_char *data_ptr = run->data_ptr;
273 size_t i;
274
275 for (i = 0; i < run->n_segs; i++) {
276 ngx_int_t rc = writev_write_segment(run->ctx, run->c, &run->wl[i],
277 &data_ptr, bytes_written_total);
278 if (rc != NGX_OK) {
279 return rc;
280 }
281 }
282
283 return NGX_OK;

334/* Handle kXR_writev — scatter the request's data segments to their per-handle
335 * (fd, offset) targets through the VFS write path, then reply with the aggregate
336 * status.
337 *
338 * Wire contract (stock XrdXrootdProtocol::do_WriteV): the header dlen frames
339 * ONLY the N*16-byte write_list descriptor block; the concatenated segment data
340 * (sum(wlen) bytes) streams after the frame. The recv framing has already
341 * appended that trailing data to ctx->recv.payload (ctx->recv.cur_body_extra bytes), so
342 * descriptors and data are contiguous here. Framing violations are rejected
343 * exactly like the reference server — send the error, then drop the link
344 * (return NGX_ERROR): once the descriptor block is in doubt the trailing byte
345 * count is unknowable and the connection cannot be resynchronised.
346 *
347 * The checkpoint-embedded form (kXR_ckpXeq carrying a writev — ckp_xeq_writev
348 * in chkpoint_xeq.c) tunnels this SAME layout one level down: the embedded
349 * sub-header's dlen frames the descriptors and the data streams behind, so
350 * the two paths must be kept contract-identical.
351 *
352 * Orchestrator: validate framing → admit every handle → try the AIO offload
353 * (returns early when posted) → synchronous per-segment write → optional
354 * doSync flush → access-log + reply. */
355ngx_int_t
356brix_handle_writev(brix_ctx_t *ctx, ngx_connection_t *c)
357{
358 xrdw_writev_req_t req;
359 write_list *wl;
360 size_t n_segs = 0;
361 uint64_t total_wlen = 0;
362 const u_char *data_ptr;
363 size_t bytes_written_total = 0;
364 int do_sync;
365 ngx_int_t rc;
366 ngx_flag_t posted = 0;
367 writev_run_t run = { 0 };
368
369 xrdw_writev_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
370 do_sync = (req.options & kXR_wv_doSync) ? 1 : 0;
371
372 wl = (write_list *) ctx->recv.payload;
373
374 rc = writev_validate_framing(ctx, c, wl, &n_segs, &total_wlen);
375 if (rc == NGX_DONE) {
376 return NGX_OK; /* all-zero-length vector: OK already sent. */
377 }
378 if (rc != NGX_OK) {
379 return rc; /* framing violation: error sent, drop the link. */
380 }
381
382 rc = writev_validate_handles(ctx, c, wl, n_segs);
383 if (rc != NGX_OK) {
384 return rc;
385 }
386
387 /* All N descriptors occupy the first n_segs*16 bytes; the per-segment data
388 * blocks follow contiguously in segment order from here. */
389 data_ptr = ctx->recv.payload + n_segs * BRIX_WRITEV_SEGSIZE;
390
391 run.ctx = ctx;
392 run.c = c;
393 run.wl = wl;
394 run.n_segs = n_segs;
395 run.data_ptr = data_ptr;
396
397 rc = writev_try_aio(&run, do_sync, &posted);
398 if (rc != NGX_OK) {
399 return rc; /* allocation failure: drop the link. */
400 }
401 if (posted) {
402 return NGX_OK; /* done callback owns the buffer + reply. */
403 }
404
405 /* Synchronous fallback: write each segment on the event-loop thread. */
406 rc = writev_execute_sync(&run, &bytes_written_total);
407 if (rc != NGX_OK) {
408 return rc;
409 }
410
411 if (do_sync) {
412 writev_sync_handles(ctx, wl, n_segs);
413 }
414
415 return writev_reply(ctx, c, n_segs, bytes_written_total);
416}
