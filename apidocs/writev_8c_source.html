5/* brix_writev_body_extra — trailing segment-data length for kXR_writev
6 * WHAT: Validates a dlen-framed write_list descriptor block under the stock
7 * wire contract (dlen frames ONLY the descriptors: non-zero, 16-aligned,
8 * <= BRIX_WRITEV_MAXSEGS entries, descriptors + declared data within
9 * BRIX_MAX_WRITE_PAYLOAD) and reports sum(wlen) — the number of segment-data
10 * bytes the client streams after the request frame.
11 * WHY: Shared by the recv framing (to extend the read obligation so the
12 * streamed data lands in payload_buf before dispatch) and kept next to the
13 * handler that consumes the same layout. Also delegated to by
14 * brix_ckpxeq_body_extra (chkpoint_xeq.c): a kXR_ckpXeq-embedded writev
15 * tunnels this exact contract — its embedded header's dlen frames the
16 * descriptors only, and the segment data streams behind. Returns NGX_OK
17 * with *extra set, or NGX_DECLINED when the block violates the contract (the
18 * caller decides the error; the framing never rejects so the auth gates
19 * still run first). */
20ngx_int_t
21brix_writev_body_extra(const u_char *desc, uint32_t dlen, uint32_t *extra)
22{
23 const write_list *wl = (const write_list *) desc;
24 uint32_t n_segs, i;
25 uint64_t total = 0;
26
27 *extra = 0;
28
29 if (desc == NULL || dlen == 0 || dlen % BRIX_WRITEV_SEGSIZE != 0) {
30 return NGX_DECLINED;
31 }
32
33 n_segs = dlen / BRIX_WRITEV_SEGSIZE;
34 if (n_segs > BRIX_WRITEV_MAXSEGS) {
35 return NGX_DECLINED;
36 }
37
38 for (i = 0; i < n_segs; i++) {
39 total += (uint32_t) ntohl((uint32_t) wl[i].wlen);
40 }
41
42 if ((uint64_t) dlen + total > BRIX_MAX_WRITE_PAYLOAD) {
43 return NGX_DECLINED;
44 }
45
46 *extra = (uint32_t) total;
47 return NGX_OK;
48}

50/* Handle kXR_writev — scatter the request's data segments to their per-handle
51 * (fd, offset) targets through the VFS write path, then reply with the aggregate
52 * status.
53 *
54 * Wire contract (stock XrdXrootdProtocol::do_WriteV): the header dlen frames
55 * ONLY the N*16-byte write_list descriptor block; the concatenated segment data
56 * (sum(wlen) bytes) streams after the frame. The recv framing has already
57 * appended that trailing data to ctx->recv.payload (ctx->recv.cur_body_extra bytes), so
58 * descriptors and data are contiguous here. Framing violations are rejected
59 * exactly like the reference server — send the error, then drop the link
60 * (return NGX_ERROR): once the descriptor block is in doubt the trailing byte
61 * count is unknowable and the connection cannot be resynchronised.
62 *
63 * The checkpoint-embedded form (kXR_ckpXeq carrying a writev — ckp_xeq_writev
64 * in chkpoint_xeq.c) tunnels this SAME layout one level down: the embedded
65 * sub-header's dlen frames the descriptors and the data streams behind, so
66 * the two paths must be kept contract-identical. */
