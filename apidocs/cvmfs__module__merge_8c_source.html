1/*
2 * cvmfs_module_merge.c — cvmfs:// location config-field merge.
3 *
4 * WHAT: The four per-concern merge-field helpers and the orchestrator that
5 * sequences them:
6 * - cvmfs_merge_preamble — adopt unified directives, merge enable, pre-seed
7 * the CAS verify default, run the shared common.* merge.
8 * - cvmfs_merge_upstreams — manifest/negative TTL, upstream allow/cap,
9 * origin-selection knobs, process-wide trace toggle.
10 * - cvmfs_merge_resilience — origin stall-detection / retry group + its
11 * process-wide transport policy.
12 * - cvmfs_merge_secure — server-side geo-answer group + scvmfs (T22)
13 * authz layer and bearer-registry build.
14 * - ngx_http_brix_cvmfs_merge_loc_conf — the orchestrator wired into the
15 * module context; runs the four helpers then delegates the enabled-branch
16 * export build to cvmfs_merge_cache (cvmfs_module_build.c).
17 *
18 * WHY: Split out of module.c (file-size gate). These helpers are one concern —
19 * resolving and merging config fields main→srv→loc — distinct from the
20 * enabled-branch storage build (cvmfs_module_build.c) that mutates persistent
21 * export state. The concerns carry strict ordering dependencies (enable before
22 * the verify pre-seed; field merges before the process-wide transport setters;
23 * scvmfs checks before the export build); the orchestrator makes every one of
24 * those visible as a flat call sequence. Merge order and defaults are
25 * byte-frozen against the pre-split module.c (nginx -t unchanged).
26 *
27 * HOW: Verbatim move. The four field-merge helpers stay static (called only by
28 * the orchestrator here); the orchestrator is exported via
29 * cvmfs_module_internal.h (the module context in module.c references it) and
30 * imports cvmfs_merge_cache from cvmfs_module_build.c through the same header.
31 */

50/*
51 * cvmfs_merge_preamble() — adopt unified directives, merge enable, pre-seed the
52 * CAS verify default, and run the shared common.* merge.
53 *
54 * WHAT: The first four steps of the location merge, in the one order they must
55 * run: brix_http_common_adopt → merge cvmfs.enable → CAS verify pre-seed →
56 * ngx_http_brix_shared_merge.
57 *
58 * WHY: enable must merge BEFORE shared_merge so the verify pre-seed can tell a
59 * cvmfs location from a non-cvmfs one, and the pre-seed must run before
60 * shared_merge turns NGX_CONF_UNSET_UINT into 0. Grouping the four fixes that
61 * ordering in one place and keeps the orchestrator flat.
62 *
63 * HOW: Verbatim move of the original head; returns NGX_CONF_ERROR if the
64 * shared merge fails, else NGX_CONF_OK.
65 */
66static char *
67cvmfs_merge_preamble(ngx_conf_t *cf, ngx_http_brix_cvmfs_loc_conf_t *prev,
68 ngx_http_brix_cvmfs_loc_conf_t *conf)
69{
70 /* Unified directives (brix_export, brix_cache_store, brix_cache_verify,
71 * ...) live in the common module; pull the merged values for this location
72 * into our embedded preamble before protocol merge applies defaults. */
73 brix_http_common_adopt(cf, &conf->common);
74
75 /* Merge enable first — before shared_merge — so we can distinguish a
76 * cvmfs location from a non-cvmfs one when pre-seeding the verify default
77 * below. enable merges only from prev->cvmfs.enable: it has no dependency
78 * on any common.* field set by shared_merge, so this ordering is safe. */
79 ngx_conf_merge_value(conf->cvmfs.enable, prev->cvmfs.enable, 0);
80
81 /* cvmfs default: fills are verified against their CAS SHA-1 content
82 * address. Pre-seed before shared_merge, which turns NGX_CONF_UNSET_UINT
83 * into 0 (BRIX_CACHE_VERIFY_OFF) and would otherwise suppress this default.
84 *
85 * brix_http_common_adopt above copied any user-set value from the common
86 * module into conf->common.cache_verify_mode. The common module's own
87 * merge is inheritance-only (no defaults), so if the user never touched
88 * brix_cache_verify at ANY scope, the common module's conf remains
89 * NGX_CONF_UNSET_UINT and nothing is copied — the field is still UNSET
90 * here. If the user DID set it, adopt carried the explicit value over and
91 * the guard below will not fire, respecting the operator's choice.
92 *
93 * prev->common.cache_verify_mode is intentionally NOT checked: at location
94 * level, prev is the server-level cvmfs conf, which has already been
95 * through shared_merge (setting it to 0), so checking prev would never
96 * allow the default to apply. The UNSET check on conf is sufficient.
97 *
98 * The guard on cvmfs.enable confines this to cvmfs locations only. */
99 if (conf->cvmfs.enable == 1
100 && conf->common.cache_verify_mode == NGX_CONF_UNSET_UINT)
101 {
102 conf->common.cache_verify_mode = BRIX_CACHE_VERIFY_CVMFS_CAS;
103 }
104
105 /* Shared common.* preamble. Two deliberate gains over the old manual
106 * block: read_only now forces allow_write off (hardening; cvmfs is a
107 * read path anyway), and common.ktls merges to its default — inert here
108 * (nothing under protocols/cvmfs/ reads it). */
109 if (ngx_http_brix_shared_merge(cf, &prev->common, &conf->common, "")
110 != NGX_CONF_OK)
111 {
112 return NGX_CONF_ERROR;
113 }
114 return NGX_CONF_OK;
115}

117/*
118 * cvmfs_merge_upstreams() — merge the manifest/negative TTL, upstream allow-list
119 * and cap, origin-selection knobs, and the process-wide trace toggle.
120 *
121 * WHAT: The block of plain field merges between shared_merge and the origin
122 * resilience group: manifest_ttl, negative_ttl, quarantine_dir,
123 * upstream_allow, upstream_max, origin_select, origin_coords, here,
124 * rtt_interval, client_hold, fill_max_life, trace (+ its trace_set).
125 *
126 * WHY: These are pure ngx_conf_merge_* calls with one process-wide side effect
127 * (brix_origin_trace_set) at the tail; collecting them keeps the orchestrator
128 * short. Defaults are byte-frozen so nginx -t is unchanged.
129 *
130 * HOW: Verbatim merge sequence; trace promotion is applied here because it is
131 * process-wide (the shared origin transport has no per-location handle).
132 */

135 ngx_http_brix_cvmfs_loc_conf_t *conf)

144 prev->cvmfs.quarantine_dir, "");

170/*
171 * cvmfs_merge_resilience() — merge the origin stall-detection / retry group and
172 * apply its process-wide transport policy.
173 *
174 * WHAT: origin_connect_timeout, origin_stall_timeout, origin_stall_bytes,
175 * origin_attempt_timeout, origin_reuse_conn, fill_retry_policy, shared_cache,
176 * unified_origin — plus the unified_origin http(s)-backend validation and the
177 * pre-fork process-wide setters (timeouts / reuse / force-primary) when
178 * enabled.
179 *
180 * WHY: The transport bounds and the force-primary toggle are process-wide
181 * operator policy set pre-fork; keeping the merge and its setters together
182 * preserves the "merge then push to the shared transport" ordering exactly.
183 *
184 * HOW: Verbatim; returns NGX_CONF_ERROR on the unified_origin misconfiguration
185 * (identical EMERG line), else NGX_CONF_OK after applying the setters.
186 */
187static char *
188cvmfs_merge_resilience(ngx_conf_t *cf, ngx_http_brix_cvmfs_loc_conf_t *prev,
189 ngx_http_brix_cvmfs_loc_conf_t *conf)
190{
191 /* Upstream stall detection + force-through retry (2026-07-03). The transport
192 * bounds and the force-primary toggle are process-wide operator policy (the
193 * shared curl transport / sd_http driver have no per-location handle), set
194 * pre-fork here beside the trace flag so every worker inherits them. */
195 ngx_conf_merge_sec_value(conf->cvmfs.origin_connect_timeout,
196 prev->cvmfs.origin_connect_timeout, 2);
197 ngx_conf_merge_sec_value(conf->cvmfs.origin_stall_timeout,
198 prev->cvmfs.origin_stall_timeout, 4);
199 ngx_conf_merge_uint_value(conf->cvmfs.origin_stall_bytes,
200 prev->cvmfs.origin_stall_bytes, 1);
201 ngx_conf_merge_sec_value(conf->cvmfs.origin_attempt_timeout,
202 prev->cvmfs.origin_attempt_timeout, 0);
203 ngx_conf_merge_value(conf->cvmfs.origin_reuse_conn,
204 prev->cvmfs.origin_reuse_conn, 1);
205 ngx_conf_merge_uint_value(conf->cvmfs.fill_retry_policy,
206 prev->cvmfs.fill_retry_policy,
207 BRIX_CVMFS_RETRY_FAILOVER);
208 ngx_conf_merge_value(conf->cvmfs.shared_cache, prev->cvmfs.shared_cache, 0);
209 ngx_conf_merge_value(conf->cvmfs.unified_origin, prev->cvmfs.unified_origin,
210 0);
211 /* unified_origin serves every proxy request from the location's configured
212 * origin backend, so that backend MUST be an http(s) origin set (ideally a
213 * "|"-separated multi-endpoint one for the failover that hides a dead
214 * origin). Fail loudly at config time rather than 500 per request. */
215 if (conf->cvmfs.enable && conf->cvmfs.unified_origin
216 && (conf->common.storage_backend.len < 4
217 || ngx_strncasecmp(conf->common.storage_backend.data,
218 (u_char *) "http", 4) != 0))
219 {
220 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
221 "brix_cvmfs_unified_origin on requires brix_storage_backend "
222 "to name an http(s) origin set, e.g. "
223 "\"http://s1a:8000|http://s1b:8000|http://s1c:8000\" (the '|'-list "
224 "is the ranked failover set that hides a dead Stratum-1)");
225 return NGX_CONF_ERROR;
226 }
227 if (conf->cvmfs.enable) {
228 brix_s3_origin_timeouts_set(
229 (long) conf->cvmfs.origin_connect_timeout * 1000,
230 (long) conf->cvmfs.origin_stall_timeout,
231 (long) conf->cvmfs.origin_stall_bytes,
232 (long) conf->cvmfs.origin_attempt_timeout * 1000);
233 brix_s3_origin_reuse_set(conf->cvmfs.origin_reuse_conn ? 1 : 0);
234 if (conf->cvmfs.fill_retry_policy == BRIX_CVMFS_RETRY_FORCE_PRIMARY) {
235 sd_http_force_primary_set(1);
236 }
237 }
238 return NGX_CONF_OK;
239}

241/*
242 * cvmfs_merge_secure() — merge the server-side geo-answer group and the scvmfs
243 * (T22, experimental) authz layer, validating the bearer registry.
244 *
245 * WHAT: geo_answer / geo_cache_ttl / geo_max_servers, then scvmfs /
246 * scvmfs_authz / scvmfs_token_issuers / scvmfs_registry, plus the structural
247 * checks: scvmfs requires cvmfs on, and bearer mode requires a token-issuer
248 * file whose registry is built once here.
249 *
250 * WHY: These two merge groups both run unconditionally after the resilience
251 * group and before the enable block; grouping them keeps the orchestrator a
252 * flat sequence. The scvmfs checks are structural (config-time) and their
253 * EMERG lines are byte-frozen.
254 *
255 * HOW: Verbatim; returns NGX_CONF_ERROR on any scvmfs misconfiguration or a
256 * failed registry build, else NGX_CONF_OK.
257 */
258static char *
259cvmfs_merge_secure(ngx_conf_t *cf, ngx_http_brix_cvmfs_loc_conf_t *prev,
260 ngx_http_brix_cvmfs_loc_conf_t *conf)
261{
262 /* Server-side geo answering (2026-07-03). */
263 ngx_conf_merge_uint_value(conf->cvmfs.geo_answer, prev->cvmfs.geo_answer,
264 BRIX_CVMFS_GEO_PASSTHROUGH);
265 ngx_conf_merge_sec_value(conf->cvmfs.geo_cache_ttl, prev->cvmfs.geo_cache_ttl,
266 60);
267 ngx_conf_merge_uint_value(conf->cvmfs.geo_max_servers,
268 prev->cvmfs.geo_max_servers, 16);
269
270 ngx_conf_merge_value(conf->scvmfs, prev->scvmfs, 0);
271 ngx_conf_merge_uint_value(conf->scvmfs_authz, prev->scvmfs_authz,
272 BRIX_SCVMFS_AUTHZ_NONE);
273 ngx_conf_merge_str_value(conf->scvmfs_token_issuers,
274 prev->scvmfs_token_issuers, "");
275 if (conf->scvmfs_registry == NULL) {
276 conf->scvmfs_registry = prev->scvmfs_registry;
277 }
278
279 /* scvmfs (T22, EXPERIMENTAL) is a LAYER on cvmfs — structural, checked
280 * at config time. Bearer mode needs the issuer registry to exist. */
281 if (conf->scvmfs) {
282 if (!conf->cvmfs.enable) {
283 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
284 "brix_scvmfs requires brix_cvmfs on");
285 return NGX_CONF_ERROR;
286 }
287 if (conf->scvmfs_authz == BRIX_SCVMFS_AUTHZ_BEARER) {
288 if (conf->scvmfs_token_issuers.len == 0) {
289 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
290 "brix_scvmfs_authz bearer requires "
291 "brix_scvmfs_token_issuers <scitokens.cfg>");
292 return NGX_CONF_ERROR;
293 }
294 if (conf->scvmfs_registry == NULL) {
295 brix_token_registry_t *reg = NULL;
296
297 if (brix_token_registry_build(cf,
298 (const char *) conf->scvmfs_token_issuers.data,
299 BRIX_AUTHZ_CAPABILITY, &reg) != NGX_OK)
300 {
301 return NGX_CONF_ERROR;
302 }
303 conf->scvmfs_registry = reg;
304 }
305 }
306 }
307 return NGX_CONF_OK;
308}

310/*
311 * ngx_http_brix_cvmfs_merge_loc_conf() — location merge orchestrator.
312 *
313 * WHAT: Runs the five per-concern merge helpers in their required order:
314 * preamble (adopt/enable/verify/shared_merge) → upstreams → resilience →
315 * secure (geo-answer + scvmfs) → cache (enabled-only export/backend build).
316 *
317 * WHY: The concerns have strict ordering dependencies (enable before verify
318 * pre-seed; merges before the process-wide transport setters; scvmfs before
319 * the export build). Expressing the orchestrator as a flat call sequence
320 * keeps every dependency visible in one place while each concern stays under
321 * the complexity gate. Merge order + defaults are byte-frozen (nginx -t).
322 *
323 * HOW: Each step returns NGX_CONF_OK/ERROR; the orchestrator early-returns on
324 * the first error. The cache step (cvmfs_merge_cache, cvmfs_module_build.c)
325 * runs only for cvmfs-enabled locations.
326 */
327char *
328ngx_http_brix_cvmfs_merge_loc_conf(ngx_conf_t *cf, void *parent, void *child)
329{
330 ngx_http_brix_cvmfs_loc_conf_t *prev = parent;
331 ngx_http_brix_cvmfs_loc_conf_t *conf = child;
332
333 if (cvmfs_merge_preamble(cf, prev, conf) != NGX_CONF_OK) {
334 return NGX_CONF_ERROR;
335 }
336 cvmfs_merge_upstreams(cf, prev, conf);
337 if (cvmfs_merge_resilience(cf, prev, conf) != NGX_CONF_OK) {
338 return NGX_CONF_ERROR;
339 }
340 if (cvmfs_merge_secure(cf, prev, conf) != NGX_CONF_OK) {
341 return NGX_CONF_ERROR;
342 }
343 if (conf->cvmfs.enable) {
344 if (cvmfs_merge_cache(cf, conf) != NGX_CONF_OK) {
345 return NGX_CONF_ERROR;
346 }
347 }
348
349 return NGX_CONF_OK;
350}
