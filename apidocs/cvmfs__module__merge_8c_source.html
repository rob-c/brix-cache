1/*
2 * cvmfs_module_merge.c — cvmfs:// location config-field merge.
3 *
4 * WHAT: The four per-concern merge-field helpers and the orchestrator that
5 * sequences them:
6 * - cvmfs_merge_preamble — adopt unified directives, merge enable, pre-seed
7 * the CAS verify default, run the shared common.* merge.
8 * - cvmfs_merge_upstreams — manifest/negative TTL, upstream allow/cap,
9 * origin-selection knobs, process-wide trace toggle.
10 * - cvmfs_merge_resilience — origin stall-detection / retry group + its
11 * process-wide transport policy.
12 * - cvmfs_merge_secure — server-side geo-answer group + scvmfs (T22)
13 * authz layer and bearer-registry build.
14 * - ngx_http_brix_cvmfs_merge_loc_conf — the orchestrator wired into the
15 * module context; runs the four helpers then delegates the enabled-branch
16 * export build to cvmfs_merge_cache (cvmfs_module_build.c).
17 *
18 * WHY: Split out of module.c (file-size gate). These helpers are one concern —
19 * resolving and merging config fields main→srv→loc — distinct from the
20 * enabled-branch storage build (cvmfs_module_build.c) that mutates persistent
21 * export state. The concerns carry strict ordering dependencies (enable before
22 * the verify pre-seed; field merges before the process-wide transport setters;
23 * scvmfs checks before the export build); the orchestrator makes every one of
24 * those visible as a flat call sequence. Merge order and defaults are
25 * byte-frozen against the pre-split module.c (nginx -t unchanged).
26 *
27 * HOW: Verbatim move. The four field-merge helpers stay static (called only by
28 * the orchestrator here); the orchestrator is exported via
29 * cvmfs_module_internal.h (the module context in module.c references it) and
30 * imports cvmfs_merge_cache from cvmfs_module_build.c through the same header.
31 */

50/*
51 * cvmfs_merge_preamble() — adopt unified directives, merge enable, pre-seed the
52 * CAS verify default, and run the shared common.* merge.
53 *
54 * WHAT: The first four steps of the location merge, in the one order they must
55 * run: brix_http_common_adopt → merge cvmfs.enable → CAS verify pre-seed →
56 * ngx_http_brix_shared_merge.
57 *
58 * WHY: enable must merge BEFORE shared_merge so the verify pre-seed can tell a
59 * cvmfs location from a non-cvmfs one, and the pre-seed must run before
60 * shared_merge turns NGX_CONF_UNSET_UINT into 0. Grouping the four fixes that
61 * ordering in one place and keeps the orchestrator flat.
62 *
63 * HOW: Verbatim move of the original head; returns NGX_CONF_ERROR if the
64 * shared merge fails, else NGX_CONF_OK.
65 */
66static char *
67cvmfs_merge_preamble(ngx_conf_t *cf, ngx_http_brix_cvmfs_loc_conf_t *prev,
68 ngx_http_brix_cvmfs_loc_conf_t *conf)
69{
70 /* Unified directives (brix_export, brix_cache_store, brix_cache_verify,
71 * ...) live in the common module; pull the merged values for this location
72 * into our embedded preamble before protocol merge applies defaults. */
73 brix_http_common_adopt(cf, &conf->common);
74
75 /* Merge enable first — before shared_merge — so we can distinguish a
76 * cvmfs location from a non-cvmfs one when pre-seeding the verify default
77 * below. enable merges only from prev->cvmfs.enable: it has no dependency
78 * on any common.* field set by shared_merge, so this ordering is safe. */
79 ngx_conf_merge_value(conf->cvmfs.enable, prev->cvmfs.enable, 0);
80
81 /* cvmfs default: fills are verified against their CAS SHA-1 content
82 * address. Pre-seed before shared_merge, which turns NGX_CONF_UNSET_UINT
83 * into 0 (BRIX_CACHE_VERIFY_OFF) and would otherwise suppress this default.
84 *
85 * brix_http_common_adopt above copied any user-set value from the common
86 * module into conf->common.cache_verify_mode. The common module's own
87 * merge is inheritance-only (no defaults), so if the user never touched
88 * brix_cache_verify at ANY scope, the common module's conf remains
89 * NGX_CONF_UNSET_UINT and nothing is copied — the field is still UNSET
90 * here. If the user DID set it, adopt carried the explicit value over and
91 * the guard below will not fire, respecting the operator's choice.
92 *
93 * prev->common.cache_verify_mode is intentionally NOT checked: at location
94 * level, prev is the server-level cvmfs conf, which has already been
95 * through shared_merge (setting it to 0), so checking prev would never
96 * allow the default to apply. The UNSET check on conf is sufficient.
97 *
98 * The guard on cvmfs.enable confines this to cvmfs locations only. */
99 if (conf->cvmfs.enable == 1
100 && conf->common.cache_verify_mode == NGX_CONF_UNSET_UINT)
101 {
102 conf->common.cache_verify_mode = BRIX_CACHE_VERIFY_CVMFS_CAS;
103 }
104
105 /* Shared common.* preamble. Two deliberate gains over the old manual
106 * block: read_only now forces allow_write off (hardening; cvmfs is a
107 * read path anyway), and common.ktls merges to its default — inert here
108 * (nothing under protocols/cvmfs/ reads it). */
109 if (ngx_http_brix_shared_merge(cf, &prev->common, &conf->common, "")
110 != NGX_CONF_OK)
111 {
112 return NGX_CONF_ERROR;
113 }
114 return NGX_CONF_OK;
115}

174/*
175 * cvmfs_merge_resilience() — merge the origin stall-detection / retry group and
176 * apply its process-wide transport policy.
177 *
178 * WHAT: origin_connect_timeout, origin_stall_timeout, origin_stall_bytes,
179 * origin_attempt_timeout, origin_reuse_conn, fill_retry_policy, shared_cache,
180 * unified_origin — plus the unified_origin http(s)-backend validation and the
181 * pre-fork process-wide setters (timeouts / reuse / force-primary) when
182 * enabled.
183 *
184 * WHY: The transport bounds and the force-primary toggle are process-wide
185 * operator policy set pre-fork; keeping the merge and its setters together
186 * preserves the "merge then push to the shared transport" ordering exactly.
187 *
188 * HOW: Verbatim; returns NGX_CONF_ERROR on the unified_origin misconfiguration
189 * (identical EMERG line), else NGX_CONF_OK after applying the setters.
190 */
191static char *
192cvmfs_merge_resilience(ngx_conf_t *cf, ngx_http_brix_cvmfs_loc_conf_t *prev,
193 ngx_http_brix_cvmfs_loc_conf_t *conf)
194{
195 /* Upstream stall detection + force-through retry (2026-07-03). The transport
196 * bounds and the force-primary toggle are process-wide operator policy (the
197 * shared curl transport / sd_http driver have no per-location handle), set
198 * pre-fork here beside the trace flag so every worker inherits them. */
199 ngx_conf_merge_sec_value(conf->cvmfs.origin_connect_timeout,
200 prev->cvmfs.origin_connect_timeout, 2);
201 ngx_conf_merge_sec_value(conf->cvmfs.origin_stall_timeout,
202 prev->cvmfs.origin_stall_timeout, 4);
203 ngx_conf_merge_uint_value(conf->cvmfs.origin_stall_bytes,
204 prev->cvmfs.origin_stall_bytes, 1);
205 ngx_conf_merge_sec_value(conf->cvmfs.origin_attempt_timeout,
206 prev->cvmfs.origin_attempt_timeout, 0);
207 ngx_conf_merge_value(conf->cvmfs.origin_reuse_conn,
208 prev->cvmfs.origin_reuse_conn, 1);
209 ngx_conf_merge_uint_value(conf->cvmfs.fill_retry_policy,
210 prev->cvmfs.fill_retry_policy,
211 BRIX_CVMFS_RETRY_FAILOVER);
212 ngx_conf_merge_uint_value(conf->cvmfs.origin_http_version,
213 prev->cvmfs.origin_http_version,
214 BRIX_CVMFS_ORIGIN_HTTP_UNSET);
215 ngx_conf_merge_value(conf->cvmfs.shared_cache, prev->cvmfs.shared_cache, 0);
216 ngx_conf_merge_value(conf->cvmfs.unified_origin, prev->cvmfs.unified_origin,
217 0);
218 /* unified_origin serves every proxy request from the location's configured
219 * origin backend, so that backend MUST be an http(s) origin set (ideally a
220 * "|"-separated multi-endpoint one for the failover that hides a dead
221 * origin). Fail loudly at config time rather than 500 per request. */
222 if (conf->cvmfs.enable && conf->cvmfs.unified_origin
223 && (conf->common.storage_backend.len < 4
224 || ngx_strncasecmp(conf->common.storage_backend.data,
225 (u_char *) "http", 4) != 0))
226 {
227 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
228 "brix_cvmfs_unified_origin on requires brix_storage_backend "
229 "to name an http(s) origin set, e.g. "
230 "\"http://s1a:8000|http://s1b:8000|http://s1c:8000\" (the '|'-list "
231 "is the ranked failover set that hides a dead Stratum-1)");
232 return NGX_CONF_ERROR;
233 }
234 /* HTTP-version policy (phase-85 F11) is process-wide like the bounds
235 * above. A version the linked libcurl cannot speak (today: HTTP/3 without
236 * an nghttp3/quiche build) is refused at config time rather than failing
237 * every fill at runtime. */
238 if (conf->cvmfs.enable
239 && conf->cvmfs.origin_http_version != BRIX_CVMFS_ORIGIN_HTTP_UNSET
240 && !brix_s3_origin_http_version_supported(
241 (int) conf->cvmfs.origin_http_version))
242 {
243 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
244 "brix_cvmfs_origin_http_version: the linked libcurl lacks "
245 "support for the requested HTTP version (HTTP/3 needs a "
246 "libcurl built with an HTTP/3 backend)");
247 return NGX_CONF_ERROR;
248 }
249 if (conf->cvmfs.enable) {
250 brix_s3_origin_http_version_set((int) conf->cvmfs.origin_http_version);
251 brix_s3_origin_timeouts_set(
252 (long) conf->cvmfs.origin_connect_timeout * 1000,
253 (long) conf->cvmfs.origin_stall_timeout,
254 (long) conf->cvmfs.origin_stall_bytes,
255 (long) conf->cvmfs.origin_attempt_timeout * 1000);
256 brix_s3_origin_reuse_set(conf->cvmfs.origin_reuse_conn ? 1 : 0);
257 if (conf->cvmfs.fill_retry_policy == BRIX_CVMFS_RETRY_FORCE_PRIMARY) {
258 sd_http_force_primary_set(1);
259 }
260 }
261 return NGX_CONF_OK;
262}

264/*
265 * cvmfs_merge_secure() — merge the server-side geo-answer group and the scvmfs
266 * (T22, experimental) authz layer, validating the bearer registry.
267 *
268 * WHAT: geo_answer / geo_cache_ttl / geo_max_servers, then scvmfs /
269 * scvmfs_authz / scvmfs_token_issuers / scvmfs_registry, plus the structural
270 * checks: scvmfs requires cvmfs on, and bearer mode requires a token-issuer
271 * file whose registry is built once here.
272 *
273 * WHY: These two merge groups both run unconditionally after the resilience
274 * group and before the enable block; grouping them keeps the orchestrator a
275 * flat sequence. The scvmfs checks are structural (config-time) and their
276 * EMERG lines are byte-frozen.
277 *
278 * HOW: Verbatim; returns NGX_CONF_ERROR on any scvmfs misconfiguration or a
279 * failed registry build, else NGX_CONF_OK.
280 */
281static char *
282cvmfs_merge_secure(ngx_conf_t *cf, ngx_http_brix_cvmfs_loc_conf_t *prev,
283 ngx_http_brix_cvmfs_loc_conf_t *conf)
284{
285 /* Server-side geo answering (2026-07-03). */
286 ngx_conf_merge_uint_value(conf->cvmfs.geo_answer, prev->cvmfs.geo_answer,
287 BRIX_CVMFS_GEO_PASSTHROUGH);
288 ngx_conf_merge_sec_value(conf->cvmfs.geo_cache_ttl, prev->cvmfs.geo_cache_ttl,
289 60);
290 ngx_conf_merge_uint_value(conf->cvmfs.geo_max_servers,
291 prev->cvmfs.geo_max_servers, 16);
292
293 ngx_conf_merge_value(conf->scvmfs, prev->scvmfs, 0);
294 ngx_conf_merge_uint_value(conf->scvmfs_authz, prev->scvmfs_authz,
295 BRIX_SCVMFS_AUTHZ_NONE);
296 ngx_conf_merge_str_value(conf->scvmfs_token_issuers,
297 prev->scvmfs_token_issuers, "");
298 if (conf->scvmfs_registry == NULL) {
299 conf->scvmfs_registry = prev->scvmfs_registry;
300 }
301
302 /* scvmfs (T22, EXPERIMENTAL) is a LAYER on cvmfs — structural, checked
303 * at config time. Bearer mode needs the issuer registry to exist. */
304 if (conf->scvmfs) {
305 if (!conf->cvmfs.enable) {
306 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
307 "brix_scvmfs requires brix_cvmfs on");
308 return NGX_CONF_ERROR;
309 }
310 if (conf->scvmfs_authz == BRIX_SCVMFS_AUTHZ_BEARER) {
311 if (conf->scvmfs_token_issuers.len == 0) {
312 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
313 "brix_scvmfs_authz bearer requires "
314 "brix_scvmfs_token_issuers <scitokens.cfg>");
315 return NGX_CONF_ERROR;
316 }
317 if (conf->scvmfs_registry == NULL) {
318 brix_token_registry_t *reg = NULL;
319
320 if (brix_token_registry_build(cf,
321 (const char *) conf->scvmfs_token_issuers.data,
322 BRIX_AUTHZ_CAPABILITY, &reg) != NGX_OK)
323 {
324 return NGX_CONF_ERROR;
325 }
326 conf->scvmfs_registry = reg;
327 }
328 }
329 }

331 /* Token-gated repos (phase-85 F3). Inherit the entry list by pointer,
332 * then build each entry's issuer registry once (registry == NULL) on
333 * cvmfs-enabled locations only — a non-cvmfs location inheriting the
334 * list never serves the protocol, so it must not pay (or fail on) the
335 * registry load. Building at merge fails misconfigurations loudly at
336 * config time instead of 401ing every request. */
337 /* Per-VO QoS fill classes (phase-85 F9): plain pointer inheritance —
338 * the class list is pure config + worker-local bucket state, nothing to
339 * build at merge time. */
340 ngx_conf_merge_ptr_value(conf->qos, prev->qos, NULL);
341
342 ngx_conf_merge_ptr_value(conf->repo_authz, prev->repo_authz, NULL);
343 if (conf->cvmfs.enable == 1 && conf->repo_authz != NULL) {
344 brix_cvmfs_repo_authz_t *e = conf->repo_authz->elts;
345 ngx_uint_t i;
346
347 for (i = 0; i < conf->repo_authz->nelts; i++) {
348 brix_token_registry_t *reg = NULL;
349
350 if (e[i].registry != NULL) {
351 continue;
352 }
353 if (brix_token_registry_build(cf,
354 (const char *) e[i].issuers.data,
355 BRIX_AUTHZ_CAPABILITY, &reg) != NGX_OK)
356 {
357 return NGX_CONF_ERROR;
358 }
359 e[i].registry = reg;
360 }
361 }
362 return NGX_CONF_OK;
363}

365/*
366 * ngx_http_brix_cvmfs_merge_loc_conf() — location merge orchestrator.
367 *
368 * WHAT: Runs the five per-concern merge helpers in their required order:
369 * preamble (adopt/enable/verify/shared_merge) → upstreams → resilience →
370 * secure (geo-answer + scvmfs) → cache (enabled-only export/backend build).
371 *
372 * WHY: The concerns have strict ordering dependencies (enable before verify
373 * pre-seed; merges before the process-wide transport setters; scvmfs before
374 * the export build). Expressing the orchestrator as a flat call sequence
375 * keeps every dependency visible in one place while each concern stays under
376 * the complexity gate. Merge order + defaults are byte-frozen (nginx -t).
377 *
378 * HOW: Each step returns NGX_CONF_OK/ERROR; the orchestrator early-returns on
379 * the first error. The cache step (cvmfs_merge_cache, cvmfs_module_build.c)
380 * runs only for cvmfs-enabled locations.
381 */
382char *
383ngx_http_brix_cvmfs_merge_loc_conf(ngx_conf_t *cf, void *parent, void *child)
384{
385 ngx_http_brix_cvmfs_loc_conf_t *prev = parent;
386 ngx_http_brix_cvmfs_loc_conf_t *conf = child;
387
388 if (cvmfs_merge_preamble(cf, prev, conf) != NGX_CONF_OK) {
389 return NGX_CONF_ERROR;
390 }
391 cvmfs_merge_upstreams(cf, prev, conf);
392 if (cvmfs_merge_resilience(cf, prev, conf) != NGX_CONF_OK) {
393 return NGX_CONF_ERROR;
394 }
395 if (cvmfs_merge_secure(cf, prev, conf) != NGX_CONF_OK) {
396 return NGX_CONF_ERROR;
397 }
398 if (conf->cvmfs.enable) {
399 if (cvmfs_merge_cache(cf, conf) != NGX_CONF_OK) {
400 return NGX_CONF_ERROR;
401 }
402 }
403
404 return NGX_CONF_OK;
405}
