6/*
7 * dashboard/dashboard_auth_parse.c — request-parsing primitives for the
8 * dashboard authentication unit (split from auth.c, phase-79).
9 *
10 * WHAT: Pure byte-range scanners over untrusted request input:
11 * - the Cookie: header, pair-by-pair, to find one named cookie value
12 * (dashboard_find_cookie), and
13 * - an application/x-www-form-urlencoded body, to extract and URL-decode
14 * one named field (dashboard_form_value).
15 *
16 * WHY: These parsers have no crypto or credential knowledge and are shared by
17 * both the cookie-verification path (auth.c) and the login flow
18 * (dashboard_auth_login.c). Isolating them keeps the parsing surface —
19 * the part that touches attacker-controlled bytes — small and reviewable.
20 * Behaviour is byte-for-byte identical to the original auth.c.
21 */

23/*
24 * WHAT: Locate the first Cookie: request header, or NULL if the request has none.
25 * WHY: nginx 1.23.0 replaced the `headers_in.cookies` array (of
26 * ngx_table_elt_t *) with a single `headers_in.cookie` linked-list head,
27 * so read whichever the build's nginx provides — keeping the module
28 * buildable on EL stock nginx (1.20.x) as well as current releases.
29 * HOW: Either way we take the first Cookie header; dashboard_find_cookie() then
30 * walks the ';'-separated pairs inside its value.
31 */
32static ngx_table_elt_t *
33dash_cookie_header(ngx_http_request_t *r)
34{
35#if (nginx_version >= 1023000)
36 return r->headers_in.cookie;
37#else
38 if (r->headers_in.cookies.nelts > 0) {
39 return ((ngx_table_elt_t **) r->headers_in.cookies.elts)[0];
40 }
41 return NULL;
42#endif
43}

45/*
46 * WHAT: Consume one ';'-delimited "name=value" pair from a Cookie header value,
47 * advancing *pp past it (including the trailing ';' if present).
48 * WHY: dashboard_find_cookie() walks the whole header pair-by-pair; isolating the scan of
49 * a single pair keeps the caller a flat match loop.
50 * HOW: Skip leading whitespace, then scan to ';' (or header end) remembering
51 * the FIRST '=' so name splits from value. Only the FIRST '=' splits; any
52 * later '=' is part of the value (cookie values may contain '=').
53 * Sets *start to the pair's name start, *eq to the '=' (or NULL for a
54 * valueless token); on return *pp points at the pair terminator before it
55 * is skipped by the caller — the value runs [*eq + 1, pair end).
56 */
57static void
58dash_cookie_next_pair(u_char **pp, u_char *end, u_char **start, u_char **eq)
59{
60 u_char *p = *pp;
61
62 /* skip whitespace */
63 while (p < end && (*p == ' ' || *p == '\t')) { p++; }
64
65 *start = p;
66
67 *eq = NULL;
68 while (p < end && *p != ';') {
69 if (*p == '=' && *eq == NULL) { *eq = p; }
70 p++;
71 }
72
73 *pp = p;
74}

76/*
77 * Find the value of a named cookie in the Cookie: header.
78 * Returns NGX_OK and sets *val / *val_len on success, NGX_DECLINED if not found.
79 */
80ngx_int_t
81dashboard_find_cookie(ngx_http_request_t *r, const char *name, size_t name_len,
82 u_char **val, size_t *val_len)
83{
84 ngx_table_elt_t *cookie_hdr;
85 u_char *p, *end, *eq, *start;
86
87 cookie_hdr = dash_cookie_header(r);
88 if (cookie_hdr == NULL) {
89 return NGX_DECLINED;
90 }
91
92 p = cookie_hdr->value.data;
93 end = p + cookie_hdr->value.len;
94
95 /* Walk the "name1=val1; name2=val2" list one pair at a time. p advances
96 * across the whole header; each iteration consumes one ';'-delimited pair. */
97 while (p < end) {
98 dash_cookie_next_pair(&p, end, &start, &eq);
99
100 /* Name must match exactly (length + bytes); value is everything between
101 * the '=' and the pair terminator p. */
102 if (eq != NULL && (size_t)(eq - start) == name_len &&
103 ngx_memcmp(start, name, name_len) == 0)
104 {
105 *val = eq + 1;
106 *val_len = (size_t)(p - (eq + 1));
107 return NGX_OK;
108 }
109
110 if (p < end) { p++; } /* skip ';' */
111 }
112
113 return NGX_DECLINED;
114}

116/* One "key=value" pair delimited inside an urlencoded body: [key_start,
117 * key_start + key_len) is the raw key, [val_start, val_end) the raw value. */
118typedef struct {
119 size_t key_start;
120 size_t key_len;
121 size_t val_start;
122 size_t val_end;
123} dash_form_pair_t;

125/*
126 * WHAT: Scan the next complete "key=value" pair from an urlencoded body,
127 * advancing *pos past it (and past any valueless tokens before it).
128 * WHY: dashboard_form_value() only cares about pairs that HAVE a value;
129 * centralizing the '&'-walk keeps the caller a flat match loop.
130 * HOW: Scan the key up to its '=' (skipping whole tokens with no '='), then
131 * delimit the value from just after '=' to the next '&' or body end.
132 * Returns NGX_OK with *pair filled, or NGX_DECLINED at body end.
133 */
134static ngx_int_t
135dash_form_next_pair(const u_char *body, size_t body_len, size_t *pos,
136 dash_form_pair_t *pair)
137{
138 size_t i = *pos;
139
140 while (i < body_len) {
141 pair->key_start = i;
142
143 /* Scan the key up to its '=' (or '&'/end if the pair has no value). */
144 while (i < body_len && body[i] != '=' && body[i] != '&') {
145 i++;
146 }
147 pair->key_len = i - pair->key_start;
148
149 /* No '=' for this token: skip the whole valueless pair and continue. */
150 if (i >= body_len || body[i] != '=') {
151 while (i < body_len && body[i] != '&') {
152 i++;
153 }
154 if (i < body_len) {
155 i++;
156 }
157 continue;
158 }
159
160 /* Delimit the value: from just after '=' to the next '&' or body end. */
161 i++;
162 pair->val_start = i;
163 while (i < body_len && body[i] != '&') {
164 i++;
165 }
166 pair->val_end = i;
167
168 if (i < body_len) {
169 i++; /* skip '&' */
170 }
171 *pos = i;
172 return NGX_OK;
173 }
174
175 *pos = i;
176 return NGX_DECLINED;
177}

179/*
180 * WHAT: URL-decode one raw form value [val_start, val_end) into out->buf,
181 * NUL-terminated and truncated to out->size; decoded length -> out->len.
182 * WHY: Application/x-www-form-urlencoded values escape spaces as '+' and
183 * arbitrary bytes as "%HH"; the login fields must be decoded before any
184 * credential comparison.
185 * HOW: '+' -> space, "%HH" -> the byte, anything else verbatim; k+1 < size
186 * reserves space for the trailing NUL.
187 */
188static void
189dash_form_decode_value(const u_char *body, size_t val_start, size_t val_end,
190 dash_form_out_t *out)
191{
192 size_t j, k;
193
194 for (j = val_start, k = 0; j < val_end && k + 1 < out->size; j++) {
195 if (body[j] == '+') {
196 out->buf[k++] = ' ';
197 } else if (body[j] == '%' && j + 2 < val_end
198 && isxdigit(body[j + 1])
199 && isxdigit(body[j + 2]))
200 {
201 /* "%HH" escape: fold each hex nibble (case-insensitive via
202 * |0x20) into the decoded byte (hi<<4 | lo); skip the 2 hex
203 * chars consumed. */
204 unsigned int hi, lo;
205 hi = body[j + 1] <= '9' ? body[j + 1] - '0'
206 : (body[j + 1] | 0x20) - 'a' + 10;
207 lo = body[j + 2] <= '9' ? body[j + 2] - '0'
208 : (body[j + 2] | 0x20) - 'a' + 10;
209 out->buf[k++] = (char) ((hi << 4) | lo);
210 j += 2;
211 } else {
212 out->buf[k++] = (char) body[j];
213 }
214 }
215 out->buf[k] = '\0';
216 out->len = k;
217}

219/*
220 * WHAT: Extract and URL-decode one field from an application/x-www-form-urlencoded
221 * request body, writing the decoded value (NUL-terminated, truncated to
222 * out->size) to out->buf and its length to out->len.
223 * HOW: Iterate "key=value" pairs separated by '&' (dash_form_next_pair); on a
224 * name match, decode the value applying form rules (dash_form_decode_value).
225 * Returns NGX_OK on the first match, NGX_DECLINED if the field is absent.
226 */
227ngx_int_t
228dashboard_form_value(const u_char *body, size_t body_len,
229 const char *name, dash_form_out_t *out)
230{
231 size_t name_len = strlen(name);
232 size_t i = 0;
233 dash_form_pair_t pair;
234
235 if (out == NULL || out->buf == NULL || out->size == 0) {
236 return NGX_DECLINED;
237 }
238
239 out->len = 0;
240 out->buf[0] = '\0';
241
242 while (dash_form_next_pair(body, body_len, &i, &pair) == NGX_OK) {
243 if (pair.key_len == name_len
244 && ngx_memcmp(body + pair.key_start, name, name_len) == 0)
245 {
246 dash_form_decode_value(body, pair.val_start, pair.val_end, out);
247 return NGX_OK;
248 }
249 }
250
251 return NGX_DECLINED;
252}
