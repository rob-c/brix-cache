1/*
2 * vfs_open.c — VFS open/close, handle lifecycle, and shared file helpers.
3 *
4 * WHAT: Implements brix_vfs_open()/brix_vfs_close() and the brix_vfs_file_*
5 * accessors (fd/path/size/mtime/from_cache/file_stat). Also hosts the
6 * cross-unit helpers declared in vfs_internal.h: brix_vfs_fill_stat()
7 * (struct stat -> brix_vfs_stat_t), brix_vfs_copy_path() (pool-dup a
8 * C string) and brix_vfs_adopt_fd() (wrap an already-open fd in a
9 * handle).
10 *
11 * WHY: Open is the one place that has to reconcile three concerns at once:
12 * write permission, the read-through cache, and kernel-enforced
13 * confinement. Concentrating that decision here keeps every other op file
14 * able to assume a valid, confined, already-fstat'd handle.
15 *
16 * HOW: brix_vfs_open() re-verifies confinement, gates writes on
17 * ctx->allow_write, optionally pre-creates the parent dir tree
18 * (BRIX_VFS_O_MKDIRPATH), then tries brix_cache_open() first. On a
19 * cache miss it translates flags via brix_vfs_open_flags() and walks the
20 * confinement cascade strongest-first: persistent rootfd +
21 * brix_open_beneath (openat2 RESOLVE_BENEATH), else root_canon +
22 * brix_open_confined_canon, else a raw open() reachable only for
23 * server-constructed paths with no export root. The resulting fd is
24 * wrapped by brix_vfs_adopt_fd(), which fstat()s it into the handle.
25 */

27#include "vfs_backend_registry.h"
28#include "fs/cache/open.h"

30#include "core/compat/log_diag.h"

32/* Populate a per-request brix_vfs_ctx_t with the fields the HTTP front ends
33 * (WebDAV, S3) set identically: a transient (rootfd = -1) confined open of an
34 * already-resolved canonical path. Zeroes the ctx first, then fills pool/log,
35 * the metrics proto, the export + cache roots (deriving cache_enabled), the
36 * write gate, the TLS flag, the identity, and the resolved path (is_confined).
37 * Callers may still adjust individual fields afterwards (e.g. cache_writethrough
38 * config). Kept HTTP-agnostic on purpose so the header stays stream-includable —
39 * callers pass pool/log/is_tls extracted from their own request object. */
40void
41brix_vfs_ctx_init(brix_vfs_ctx_t *vctx, ngx_pool_t *pool, ngx_log_t *log,
42 brix_proto_t proto, const char *root_canon, const char *cache_root_canon,
43 int allow_write, int is_tls, brix_identity_t *identity,
44 const char *resolved_path)
45{
46 if (vctx == NULL) {
47 return;
48 }
49
50 ngx_memzero(vctx, sizeof(*vctx));
51 vctx->rootfd = -1;
52 vctx->pool = pool;
53 vctx->log = log;
54 vctx->metrics_proto = proto;
55 vctx->root_canon = root_canon;
56 /* Resolve the export's selected storage backend (NULL ⇒ default POSIX) so
57 * every VFS op on this ctx routes through the driver without each handler
58 * threading the instance. Per-worker, lazily created on first use. */
59 vctx->sd = brix_vfs_backend_resolve(root_canon, log);
60 vctx->cache_root_canon = cache_root_canon;
61 vctx->cache_enabled =
62 (cache_root_canon != NULL && cache_root_canon[0] != '\0') ? 1 : 0;
63 vctx->allow_write = allow_write ? 1 : 0;
64 vctx->is_tls = is_tls ? 1 : 0;
65 vctx->identity = identity;
66 if (resolved_path != NULL) {
67 vctx->resolved.resolved.data = (u_char *) resolved_path;
68 vctx->resolved.resolved.len = ngx_strlen(resolved_path);
69 vctx->resolved.is_confined = 1;
70 }
71}

73/* Copy a struct stat into the protocol-neutral brix_vfs_stat_t (zeroes the
74 * output first; sets is_directory/is_regular from the mode). No-op on NULL. */
75void
76brix_vfs_fill_stat(const struct stat *st, brix_vfs_stat_t *out)
77{
78 if (st == NULL || out == NULL) {
79 return;
80 }
81
82 ngx_memzero(out, sizeof(*out));
83 out->size = st->st_size;
84 out->mtime = st->st_mtime;
85 out->ctime = st->st_ctime;
86 out->atime = st->st_atime;
87 out->mode = (ngx_uint_t) st->st_mode;
88 out->ino = st->st_ino;
89 out->dev = st->st_dev;
90 out->uid = st->st_uid;
91 out->gid = st->st_gid;
92 out->blocks = st->st_blocks;
93 out->is_directory = S_ISDIR(st->st_mode) ? 1 : 0;
94 out->is_regular = S_ISREG(st->st_mode) ? 1 : 0;
95}

97char *
98brix_vfs_copy_path(ngx_pool_t *pool, const char *path)
99{
100 size_t len;
101 char *copy;
102
103 if (pool == NULL || path == NULL) {
104 errno = EINVAL;
105 return NULL;
106 }
107
108 len = strlen(path);
109 copy = ngx_pnalloc(pool, len + 1);
110 if (copy == NULL) {
111 errno = ENOMEM;
112 return NULL;
113 }
114
115 ngx_memcpy(copy, path, len);
116 copy[len] = '\0';
117 return copy;
118}

120/* Wrap an already-open fd in a freshly pcalloc'd brix_vfs_file_t: fstat the
121 * fd into the cached metadata, dup the path, and record from_cache/is_tls.
122 * `writable` gates the stat_current fast path (set only for read-only handles).
123 * Used by both brix_vfs_open() and the cache layer's open path. */
124ngx_int_t
125brix_vfs_adopt_fd(brix_vfs_ctx_t *ctx, const char *path, ngx_fd_t fd,
126 unsigned from_cache, unsigned writable, brix_vfs_file_t **out)
127{
128 brix_sd_stat_t st;
129 brix_sd_obj_t obj;
130 brix_vfs_file_t *fh;
131
132 if (out == NULL) {
133 errno = EINVAL;
134 return NGX_ERROR;
135 }
136 *out = NULL;
137
138 if (ctx == NULL || path == NULL || fd == NGX_INVALID_FILE) {
139 errno = EINVAL;
140 return NGX_ERROR;
141 }
142
143 /* Capture the open-time metadata through the backend's fstat slot rather
144 * than a direct fstat(2): the VFS records the cached size/mtime/etc. but
145 * leaves the syscall to the storage driver. */
146 brix_vfs_ctx_sd_obj(ctx, fd, &obj);
147 if (obj.driver->fstat == NULL || obj.driver->fstat(&obj, &st) != NGX_OK) {
148 return NGX_ERROR;
149 }
150
151 fh = ngx_pcalloc(ctx->pool, sizeof(*fh));
152 if (fh == NULL) {
153 errno = ENOMEM;
154 return NGX_ERROR;
155 }
156
157 fh->path = brix_vfs_copy_path(ctx->pool, path);
158 if (fh->path == NULL) {
159 return NGX_ERROR;
160 }
161
162 fh->obj = obj; /* backend object built above (driver + inst + fd) */
163 fh->pool = ctx->pool;
164 fh->log = ctx->log;
165 fh->ctx = ctx;
166 fh->size = st.size;
167 fh->mtime = st.mtime;
168 fh->ctime = st.ctime;
169 fh->ino = st.ino;
170 fh->mode = st.mode;
171 fh->from_cache = from_cache ? 1 : 0;
172 fh->is_tls = ctx->is_tls;
173 /* Trust the open-time metadata for stat() only on a read-only handle: the
174 * file cannot change through it, so the fstat above stays authoritative for
175 * the handle's lifetime. A writable handle leaves this 0 so brix_vfs_file_stat()
176 * always issues a live fstat (its bytes/mtime/size move as it is written). */
177 fh->stat_current = writable ? 0 : 1;
178
179 *out = fh;
180 return NGX_OK;
181}

184 * already produced, PRESERVING its per-open state (e.g. an object backend's
185 * block map / metadata), rather than rebuilding from a bare fd (which would drop
186 * that state). Captures open-time metadata via the object's own fstat slot, then
187 * moves the object into the handle by value. A heap-allocated shell (driver set
188 * obj->heap_shell, e.g. pblock's malloc'd object) is freed once copied; the

190 * Used for the non-POSIX storage drivers; POSIX keeps the adopt_fd path. Also

192 * object — hence public (declared in vfs.h). */

197/* brix_vfs_export_relative — the export-root-relative ("logical") form of a
198 * confined path, which is what an inst-keyed storage-driver op expects (per the
199 * SD seam contract). Some callers (WebDAV/S3) resolve to an absolute path under
200 * root_canon; strip that prefix so a non-POSIX backend keys its namespace on the
201 * logical path. Returns `path` unchanged when it is not under root_canon.
202 * Declared in vfs_internal.h; shared with vfs_staged.c. */
203/* Path-based form: strip `root_canon` (an absolute export root) from `path`. */
204const char *
205brix_vfs_export_relative_root(const char *path, const char *root_canon)
206{
207 size_t rlen;
208
209 if (root_canon == NULL || path == NULL) {
210 return path;
211 }
212 rlen = ngx_strlen(root_canon);
213 if (rlen == 0 || ngx_strncmp(path, root_canon, rlen) != 0) {
214 return path;
215 }
216 if (path[rlen] == '/') {
217 return path + rlen; /* "/sub/file" */
218 }
219 if (path[rlen] == '\0') {
220 return "/"; /* the export root itself */
221 }
222 return path; /* a sibling that merely shares the prefix */
223}

225const char *
226brix_vfs_export_relative(const brix_vfs_ctx_t *ctx, const char *path)
227{
228 return brix_vfs_export_relative_root(path, ctx->root_canon);

231ngx_int_t
232brix_vfs_adopt_obj(brix_vfs_ctx_t *ctx, const char *path,
233 brix_sd_obj_t *o, unsigned writable, brix_vfs_file_t **out)
234{
235 brix_sd_stat_t st;
236 brix_vfs_file_t *fh;
237
238 *out = NULL;
239 if (ctx == NULL || path == NULL || o == NULL) {
240 errno = EINVAL;
241 return NGX_ERROR;
242 }
243 if (o->driver->fstat == NULL || o->driver->fstat(o, &st) != NGX_OK) {
244 return NGX_ERROR;
245 }
246
247 fh = ngx_pcalloc(ctx->pool, sizeof(*fh));
248 if (fh == NULL) {
249 errno = ENOMEM;
250 return NGX_ERROR;
251 }
252 fh->path = brix_vfs_copy_path(ctx->pool, path);
253 if (fh->path == NULL) {
254 return NGX_ERROR;
255 }
256
257 fh->obj = *o; /* driver + instance + fd + per-open state */
258 if (o->heap_shell) {
259 free(o); /* shell copied into fh->obj; release it */
260 }
261 fh->obj.heap_shell = 0; /* fh->obj is embedded, not a heap shell */
262
263 fh->pool = ctx->pool;
264 fh->log = ctx->log;
265 fh->ctx = ctx;
266 fh->size = st.size;
267 fh->mtime = st.mtime;
268 fh->ctime = st.ctime;
269 fh->ino = st.ino;
270 fh->mode = st.mode;
271 fh->from_cache = 0;
272 fh->is_tls = ctx->is_tls;
273 fh->stat_current = writable ? 0 : 1;
274
275 *out = fh;
276 return NGX_OK;
277}

279static int
280brix_vfs_open_flags(ngx_uint_t flags)
281{
282 int oflags;
283
284 if ((flags & BRIX_VFS_O_READ) && (flags & BRIX_VFS_O_WRITE)) {
285 oflags = O_RDWR;
286 } else if (flags & BRIX_VFS_O_WRITE) {
287 oflags = O_WRONLY;
288 } else {
289 oflags = O_RDONLY;
290 }
291
292 if (flags & BRIX_VFS_O_CREATE) {
293 oflags |= O_CREAT;
294 }
295 if (flags & BRIX_VFS_O_EXCL) {
296 oflags |= O_EXCL;
297 }
298 if (flags & BRIX_VFS_O_TRUNC) {
299 oflags |= O_TRUNC;
300 }
301 if (flags & BRIX_VFS_O_APPEND) {
302 oflags |= O_APPEND;
303 }
304
305 return oflags;
306}

308/* Map the VFS open flags to the backend-neutral SD open flags the driver open
309 * slot consumes (the driver maps them to O_* itself). 1:1 with the O_* mapping
310 * above, so a driver-routed open is byte-identical to the legacy beneath open. */
311static int
312brix_vfs_to_sd_flags(ngx_uint_t flags)
313{
314 int sd = 0;
315
316 if (flags & BRIX_VFS_O_READ) { sd |= BRIX_SD_O_READ; }
317 if (flags & BRIX_VFS_O_WRITE) { sd |= BRIX_SD_O_WRITE; }
318 if (flags & BRIX_VFS_O_CREATE) { sd |= BRIX_SD_O_CREATE; }
319 if (flags & BRIX_VFS_O_EXCL) { sd |= BRIX_SD_O_EXCL; }
320 if (flags & BRIX_VFS_O_TRUNC) { sd |= BRIX_SD_O_TRUNC; }
321 if (flags & BRIX_VFS_O_APPEND) { sd |= BRIX_SD_O_APPEND; }
322
323 return sd;
324}

326/* For BRIX_VFS_O_MKDIRPATH: create the target's parent directory chain
327 * (mode 0755) before the file open is attempted. A driver-backed export gets
328 * the chain in the DRIVER's namespace (brix_vfs_backend_mkpath — for e.g.
329 * pblock the catalog IS the namespace; raw host mkdirs would leave the new
330 * file an orphan no readdir can reach); the default posix export keeps the
331 * confined host mkdir. */
332static ngx_int_t
333brix_vfs_mkdir_parent_path(brix_vfs_ctx_t *ctx, const char *path)
334{
335 char *parent;
336 char *slash;
337 ngx_int_t rc;
338
339 if (ctx->root_canon == NULL || path == NULL) {
340 return NGX_OK;
341 }
342
343 parent = brix_vfs_copy_path(ctx->pool, path);
344 if (parent == NULL) {
345 return NGX_ERROR;
346 }
347
348 slash = strrchr(parent, '/');
349 if (slash == NULL || slash == parent) {
350 return NGX_OK;
351 }
352
353 *slash = '\0';
354
355 rc = brix_vfs_backend_mkpath(ctx->root_canon,
356 brix_vfs_export_relative_root(parent, ctx->root_canon),
357 0755, ctx->log);
358 if (rc != NGX_DECLINED) {
359 return (rc == 0) ? NGX_OK : NGX_ERROR;
360 }
361
362 if (brix_mkdir_recursive_confined_canon(ctx->log, ctx->root_canon,
363 parent, 0755, NULL) != 0)
364 {
365 return NGX_ERROR;
366 }
367
368 return NGX_OK;
369}

371/* Open the resolved ctx path under the confinement cascade. Returns a handle
372 * or NULL with the syscall errno in *err_out. Cache hits short-circuit; writes
373 * require ctx->allow_write. See the file header for the full open sequence. */
374brix_vfs_file_t *
375brix_vfs_open(brix_vfs_ctx_t *ctx, ngx_uint_t flags, int *err_out)
376{
377 const char *path;
378 int oflags;
379 ngx_fd_t fd;
380 brix_vfs_file_t *fh;
381 ngx_int_t rc;
382
383 if (err_out != NULL) {
384 *err_out = 0;
385 }
386
387 if (brix_vfs_require_confined(ctx) != NGX_OK) {
388 if (err_out != NULL) {
389 *err_out = errno;
390 }
391 return NULL;
392 }
393
394 if ((flags & BRIX_VFS_O_WRITE) && !ctx->allow_write) {
395 errno = EACCES;
396 if (err_out != NULL) {
397 *err_out = errno;
398 }
399 return NULL;
400 }
401
402 path = brix_vfs_ctx_path(ctx);
403 if ((flags & BRIX_VFS_O_MKDIRPATH)
404 && brix_vfs_mkdir_parent_path(ctx, path) != NGX_OK)
405 {
406 if (err_out != NULL) {
407 *err_out = errno;
408 }
409 return NULL;
410 }
411
412 rc = brix_cache_open(ctx, flags, &fh);
413 if (rc == NGX_OK) {
414 brix_metric_cache_result(brix_vfs_metrics_proto(ctx), 1, 0);
415 return fh;
416 }
417 if (rc == NGX_ERROR) {
418 if (err_out != NULL) {
419 *err_out = errno;
420 }
421 return NULL;
422 }
423 if (ctx->cache_enabled && !(flags & BRIX_VFS_O_WRITE)
424 && !(flags & BRIX_VFS_O_NOCACHE))
425 {
426 brix_metric_cache_result(brix_vfs_metrics_proto(ctx), 0, 0);
427 }
428
429 oflags = brix_vfs_open_flags(flags);
430
431 /*
432 * A non-POSIX storage backend (e.g. pblock) owns its own namespace and
433 * confinement, so route EVERY open through its driver and adopt the returned
434 * object (carrying its per-open state). This must run before the POSIX
435 * confinement cascade below: that cascade opens a bare POSIX fd (via
436 * open_beneath / confined_canon / raw open) which this backend's fstat/read
437 * slots cannot interpret — adopting such an fd under a non-POSIX driver would
438 * dereference a NULL per-open state. POSIX (the default driver, or an unset
439 * backend) falls through to the cascade unchanged.
440 */
441 if (ctx->sd != NULL && ctx->sd->driver != brix_sd_default_driver()
442 && ctx->sd->driver->open != NULL)
443 {
444 brix_sd_obj_t *o;
445 int sderr = 0;
446
447 o = ctx->sd->driver->open(ctx->sd,
448 brix_vfs_export_relative(ctx, path),
449 brix_vfs_to_sd_flags(flags), 0644, &sderr);
450 if (o == NULL) {
451 if (err_out != NULL) {
452 *err_out = sderr;
453 }
454 errno = sderr;
455 return NULL;
456 }
457 if (brix_vfs_adopt_obj(ctx, path, o,
458 (flags & BRIX_VFS_O_WRITE) ? 1u : 0u, &fh) != NGX_OK)
459 {
460 int err = errno;
461
462 o->driver->close(o);
463 if (o->heap_shell) {
464 free(o);
465 }
466 if (err_out != NULL) {
467 *err_out = err;
468 }
469 errno = err;
470 return NULL;
471 }
472 return fh;
473 }
474
475 /*
476 * Confinement cascade, strongest first:
477 * 1. ctx->rootfd >= 0 → brix_open_beneath(): persistent per-worker
478 * O_PATH fd + openat2(RESOLVE_BENEATH). This is the Phase 8 path and
479 * the case every real data-server request takes.
480 * 2. root_canon only → brix_open_confined_canon(): same openat2
481 * RESOLVE_BENEATH semantics but opens the rootfd per call. Reached only
482 * by contexts that have a root string but no persistent fd (legacy
483 * callers being retired); still fully confined.
484 * 3. neither → raw open(). NOT a confinement bypass: this branch
485 * is only taken when the VFS ctx carries no root at all (e.g. an
486 * already-absolute, server-constructed path in a non-export context).
487 * `path` here is never a raw client path — client requests always set a
488 * root and therefore take branch 1 or 2. If a root is present the raw
489 * open is unreachable.
490 */
491 if (ctx->rootfd >= 0) {
492 /*
493 * Hot path: route the confined open through the Storage Driver. A
494 * pool-lived POSIX instance borrows the persistent rootfd (no extra
495 * syscall, no fd ownership transfer); the driver performs the
496 * RESOLVE_BENEATH open. The instance is cached on ctx->sd for reuse
497 * across opens on this ctx. driver->open's fstat/obj are discarded —
498 * adopt_fd re-wraps the returned fd below — keeping one handle-build path.
499 */
500 if (ctx->sd == NULL) {
501 ctx->sd = brix_sd_posix_borrow_instance(ctx->pool, ctx->log,
502 ctx->rootfd,
503 ctx->root_canon);
504 }
505 if (ctx->sd != NULL && ctx->sd->driver->open != NULL) {
506 brix_sd_obj_t *o;
507 int sderr = 0;
508
509 o = ctx->sd->driver->open(ctx->sd, path,
510 brix_vfs_to_sd_flags(flags), 0644,
511 &sderr);
512 if (o == NULL) {
513 if (err_out != NULL) {
514 *err_out = sderr;
515 }
516 errno = sderr;
517 return NULL;
518 }
519 fd = o->fd;
520 } else {
521 fd = brix_open_beneath(ctx->rootfd, path, oflags, 0644);
522 }
523 } else if (ctx->root_canon != NULL) {
524 fd = brix_open_confined_canon(ctx->log, ctx->root_canon, path,
525 oflags, 0644);
526 } else {
527 fd = open(path, oflags, 0644);
528 }
529
530 if (fd == NGX_INVALID_FILE) {
531 if (err_out != NULL) {
532 *err_out = errno;
533 }
534 return NULL;
535 }
536
537 if (brix_vfs_adopt_fd(ctx, path, fd, 0,
538 (flags & BRIX_VFS_O_WRITE) ? 1u : 0u, &fh)
539 != NGX_OK)
540 {
541 int err = errno;
542 ngx_close_file(fd);
543 if (err_out != NULL) {
544 *err_out = err;
545 }
546 errno = err;
547 return NULL;
548 }
549
550 return fh;
551}

553ngx_int_t
554brix_vfs_close(brix_vfs_file_t *fh, ngx_log_t *log)
555{
556 if (fh == NULL || fh->obj.driver == NULL
557 || fh->obj.fd == NGX_INVALID_FILE)
558 {
559 return NGX_OK;
560 }
561
562 /* Release the descriptor through the backend's close slot; the driver
563 * marks obj.fd invalid. The VFS keeps the error-log wrapper. */
564 if (fh->obj.driver->close(&fh->obj) != NGX_OK) {
565 BRIX_DIAG_ERR(log != NULL ? log : fh->log, ngx_errno,
566 "xrootd[disk]: close failed for \"%s\"",
567 "a deferred write error surfaced at close — typically the "
568 "filesystem filled up (ENOSPC) or the device returned an I/O error",
569 "check free space and dmesg for disk errors; the file may be "
570 "incomplete, so the client's write should be treated as failed",
571 fh->path != NULL ? fh->path : "-");
572 return NGX_ERROR;
573 }
574
575 return NGX_OK;
576}

590/* Read up to `len` bytes at `off` through the handle's storage driver — the
591 * backend-neutral read used to serve a backend that exposes no single sendfile
592 * fd (e.g. an object backend whose bytes span multiple block files). Returns the
593 * bytes read (0 = EOF), or -1 with errno. One driver pread; the caller loops. */
594ssize_t
595brix_vfs_file_pread(brix_vfs_file_t *fh, void *buf, size_t len, off_t off)
596{
597 if (fh == NULL || fh->obj.driver == NULL
598 || fh->obj.driver->pread == NULL)
599 {
600 errno = EINVAL;
601 return -1;
602 }
603 return fh->obj.driver->pread(&fh->obj, buf, len, off);
604}

606/* The fd only when the backend elects to back a zero-copy transfer of the whole
607 * object, else NGX_INVALID_FILE. The decision is delegated to the backend's
608 * read_sendfile_fd slot (want_zerocopy=1: the HTTP serve helper applies the
609 * TLS/cleartext choice itself). The contract gate for callers that build a
610 * sendfile / file-backed response. */
611ngx_fd_t
612brix_vfs_file_sendfile_fd(const brix_vfs_file_t *fh)
613{
614 if (fh == NULL) {
615 return NGX_INVALID_FILE;

664ngx_int_t
665brix_vfs_file_stat(const brix_vfs_file_t *fh, brix_vfs_stat_t *stat_out)
666{
667 brix_sd_stat_t st;
668 brix_sd_obj_t obj;
669
670 if (fh == NULL || stat_out == NULL
671 || (fh->obj.fd == NGX_INVALID_FILE && fh->obj.driver == NULL))
672 {
673 /* A driver-backed handle (object/remote backend) has no kernel fd; it
674 * answers from cached metadata or the driver's fstat slot below. */
675 errno = EINVAL;
676 return NGX_ERROR;
677 }
678
679 /*
680 * phase-45 W2/R1: when the metadata cached at adopt time is authoritative,
681 * answer from it and skip a redundant fstat(2). This is the common read
682 * path (S3/WebDAV GET open the fd then immediately stat it). stat_current is
683 * set by adopt_fd only for read-only handles, whose file cannot change
684 * through them; a writable handle has it clear and always takes a live fstat.
685 */
686 if (fh->stat_current) {
687 ngx_memzero(stat_out, sizeof(*stat_out));
688 stat_out->size = fh->size;
689 stat_out->mtime = fh->mtime;
690 stat_out->ctime = fh->ctime;
691 stat_out->mode = (ngx_uint_t) fh->mode;
692 stat_out->ino = fh->ino;
693 stat_out->is_directory = S_ISDIR(fh->mode) ? 1 : 0;
694 stat_out->is_regular = S_ISREG(fh->mode) ? 1 : 0;
695 return NGX_OK;
696 }
697
698 /* Live re-stat through the backend's fstat slot (not a direct fstat(2)). */
699 brix_vfs_handle_sd_obj(fh, &obj);
700 if (obj.driver->fstat == NULL || obj.driver->fstat(&obj, &st) != NGX_OK) {
701 return NGX_ERROR;
702 }
703
704 brix_vfs_sd_stat_to_vfs(&st, stat_out);
705 return NGX_OK;
706}
