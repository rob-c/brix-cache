1/*
2 * sss.h — SSS (Simple Shared Secret) wire constants (single source of truth).
3 *
4 * WHAT: the on-the-wire constants of the XRootD SSS credential format.
5 * WHY: the module's SSS auth (src/sss) and the native client's SSS mint
6 * (client/lib/sss_keytab.c, sec_sss.c) must agree byte-for-byte; previously
7 * each kept its own copy with a "must match the other" comment. This header
8 * is the one place they come from (gsi.h precedent in the same directory).
9 * HOW: header-only macros — no ngx, no includes, no code.
10 *
11 * Clean-room: wire facts cross-checked against XrdSecsss, not copied from it.
12 */
13#ifndef BRIX_PROTOCOL_SSS_H
14#define BRIX_PROTOCOL_SSS_H

16/* Outer SSS packet header: "sss\0" magic + version + options + pad + 8-byte keyid. */
17#define BRIX_SSS_HDR_LEN 16

19#define BRIX_SSS_DATA_HDR_LEN 40
