2 * webdav/webdav_props.h

4 * Dead WebDAV properties (persisted as filesystem extended attributes) —
5 * PROPPATCH set/remove, PROPFIND value/empty/enumerate, the protected-DAV guard,
6 * and COPY/MOVE property carry-over — plus the low-level file I/O helpers
7 * (readahead hint, full write, spooled-file copy) used by the PUT/copy paths.
8 * Split out of webdav.h so these surfaces are grouped by concern and
9 * individually reviewable. Includes webdav.h for the shared request types.
10 */

17/* Dead WebDAV properties persisted as filesystem extended attributes.
18 * `xml` for set() must be already-escaped, well-formed XML — it is stored and
19 * echoed back verbatim. ns/local identify the property (namespace URI + local
20 * name); head/tail are an ngx_chain_t accumulator for response XML. */
21/* PROPPATCH set: store xml[xml_len] as the property's xattr value. NGX_OK;
22 * NGX_ERROR (errno ENAMETOOLONG if the name/value is over the cap). */

24 const char *ns, const char *local, const char *xml, size_t xml_len);

26ngx_int_t webdav_dead_prop_remove(ngx_http_request_t *r, const char *path,
27 const char *ns, const char *local);
28/* PROPFIND: if the named property exists, append its stored XML and set *found=1;
29 * a missing property is NGX_OK with *found=0 (caller reports 404 in-multistatus).
30 * NGX_ERROR only on a real fault. */

38 ngx_chain_t **tail);
