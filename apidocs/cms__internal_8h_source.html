9 * INITIAL_DELAY — ms to wait before the first connection attempt after

11 * BACKOFF_INITIAL — ms for the first reconnect delay after a disconnect.

23/*
24 * Fast cold-start mesh settling (src/cms/connect.c). When a whole cluster boots
25 * together — most acutely on one host — a node's first connect often races ahead of
26 * its manager's listen socket and is refused. Instead of jumping straight to the
27 * multi-second reconnect backoff, a node that has NEVER yet logged in retries the
28 * TCP connect on a short fixed interval for a bounded window, then falls back to the
29 * normal exponential backoff. Two locality profiles; an explicit
30 * brix_cms_initial_delay / brix_cms_connect_retry directive overrides the
31 * matching default. The interval floor and the bounded window are what keep this
32 * from ever becoming a busy-spin (cf. the self-rearming-0ms-timer footgun).
33 */
34#define NGX_BRIX_CMS_INITDELAY_LOOPBACK 0 /* loopback: connect immediately */
35#define NGX_BRIX_CMS_INITDELAY_REMOTE 10 /* remote: tiny settle margin */
36#define NGX_BRIX_CMS_FASTRETRY_LOOPBACK 10 /* loopback retry interval (ms) */
37#define NGX_BRIX_CMS_FASTRETRY_REMOTE 75 /* remote retry interval (ms) */
38#define NGX_BRIX_CMS_FASTRETRY_FLOOR 10 /* never retry faster than this */
39#define NGX_BRIX_CMS_FASTWIN_LOOPBACK 2000 /* loopback fast-retry window (ms)*/
40#define NGX_BRIX_CMS_FASTWIN_REMOTE 3000 /* remote fast-retry window (ms) */
41#define NGX_BRIX_CMS_HDR_LEN 8
42#define NGX_BRIX_CMS_MAX_FRAME 4096
43/*
44 * Phase 51 (A2): max complete frames processed per read-event wakeup before the
45 * handler yields (re-posts its read event) so one peer flooding frames on a
46 * single connection cannot monopolise the worker event loop. Generous — a
47 * conformant manager/node never sends this many frames back-to-back.
48 */
49#define NGX_BRIX_CMS_MAX_FRAMES_PER_WAKEUP 64
50#define NGX_BRIX_CMS_MIN_FREE_MB 100
