1/* fill_retry.h — upstream-outcome classification + deadline'd backoff for
2 * never-drop cache fills (phase-68 T20, convention #7).
3 *
4 * WHAT: classifies one fill attempt's outcome (retry vs definitive) and
5 * sleeps the jittered exponential backoff between attempts, bounded by
6 * the client-hold deadline while a client is parked on the fill and by
7 * the detached-fill max-life once every client has gone.
8 * WHY: a CVMFS client that sees connection errors or fast 5xx storms marks
9 * its proxy failed and escalates (next proxy group, then DIRECT) — the
10 * cache must absorb origin trouble, not re-emit it.
11 * HOW: pure C besides ngx_msec_t; runs on fill worker threads (sleeping
12 * there is correct — blocking I/O already lives on these threads).
13 * The waiter count is the owning fill's atomic, written by the event
14 * loop as clients attach/detach.
15 */
16#ifndef BRIX_CACHE_FILL_RETRY_H
17#define BRIX_CACHE_FILL_RETRY_H

23typedef enum {
24 BRIX_FILL_OK = 0,
25 BRIX_FILL_RETRY, /* transient: backoff, retry the ranked walk */
26 BRIX_FILL_DEFINITIVE /* 404/403: the origin's answer — never retry */
27} brix_fill_class_e;

29typedef struct {
30 ngx_msec_t backoff_ms; /* next delay, starts 250, caps 8000 */

32 time_t client_hold; /* conf: brix_cvmfs_client_hold */
33 time_t max_life; /* conf: brix_cvmfs_fill_max_life */
34 _Atomic int *waiters; /* the owning fill's waiter count */
35 unsigned verify_budget; /* MISMATCH retries left (≈ n endpoints) */
36} brix_fill_retry_t;

41/* Classify one attempt: fill_rc is the fill's NGX_* result, err its errno.
42 * ENOENT/ENOTDIR/EACCES/EPERM are the origin's definitive answers; EBADMSG
43 * is a digest MISMATCH (retried while verify_budget lasts — corruption is
44 * often path-local); everything else transient. */
45brix_fill_class_e brix_fill_classify(ngx_int_t fill_rc, int err,
46 brix_fill_retry_t *rs);
47
48/* 1 = slept the backoff, try again; 0 = deadline passed, give up. The
49 * deadline is client_hold while >=1 waiter is attached, max_life once
50 * detached; zero-configured holds (non-CVMFS exports) fail on the first
51 * call — single-pass, the pre-phase-68 semantics. */
52int brix_fill_retry_wait(brix_fill_retry_t *rs);
