12/*
13 * BRIX_PATH_MAX — maximum filesystem path length accepted from any client.
14 *
15 * Both the XRootD stream protocol and the HTTP (WebDAV/S3) protocol layers
16 * enforce this limit before expensive syscalls (realpath, open). Setting it
17 * equal to the Linux PATH_MAX (4096) ensures buffers declared as
18 * char buf[BRIX_PATH_MAX] can always hold a NUL-terminated kernel path.
19 *
20 * Protocol-specific aliases BRIX_MAX_PATH (stream) and WEBDAV_MAX_PATH
21 * (HTTP) are defined in their respective headers as macros that expand to
22 * this value so a single change here propagates everywhere.
23 */
24#define BRIX_PATH_MAX 4096
25
26/*
27 * BRIX_PATH_MIN — minimum valid path length (inclusive).
28 *
29 * An absolute path must contain at least one character ("/"). The stream
30 * path extractor and HTTP path resolver both reject zero-length paths;
31 * this constant makes that threshold explicit and searchable.
32 */
33#define BRIX_PATH_MIN 1

35/*
36 * WHAT: Resolve a pre-decoded URI path within root_canon into a confined filesystem path.
37 * Returns 0 on success; an HTTP status code on failure (see below).
38 *
39 * WHY: WebDAV and S3 PUT handlers both need canonicalization, dot/dotdot rejection,
40 * root confinement verification, and ENOENT-parent handling for new-file paths.
41 * Centralising this logic ensures consistent behaviour across protocols.
42 *
43 * HOW: Reject "." or ".." components → 403. Build candidate = root_canon + decoded_path.
44 * Call realpath() to canonicalize. If realpath succeeds, verify confinement under
45 * root_canon (strncmp + boundary check). If realpath fails with ENOENT, walk up
46 * the path finding the deepest existing ancestor within root_canon and append the
47 * non-existent suffix. Copy resolved into out buffer after size check.
48 *
49 * Input constraints: decoded_path must be percent-decoded and start with '/'.
50 * Trailing slashes should be stripped by the caller before calling here.
51 *
52 * Return codes on failure:
53 * 400 malformed input (empty filename, no slash in candidate)
54 * 403 path traversal detected or resolved path outside root
55 * 404 parent directory cannot be resolved (target does not exist)
56 * 414 path too long (candidate/resolved exceeds PATH_MAX or outsz)
57 * 500 realpath(3) failed for an unexpected reason (not ENOENT)
58 *
59 * The caller must map 400/404 to the appropriate protocol-level status;
60 * for WebDAV COPY/MOVE destinations, 404 should be returned as 409 Conflict.
61 *
62 * This is a thin wrapper over brix_http_resolve_path_ex() with allow_internal=0
63 * — the default-deny behaviour every normal client surface requires.
64 */
65int brix_http_resolve_path(const char *root_canon, const char *decoded_path,
66 char *out, size_t outsz);
67
68/*
69 * WHAT: Same as brix_http_resolve_path(), plus an allow_internal switch. When
70 * allow_internal is non-zero the internal-name → 404 guard is skipped so a
71 * reserved sidecar/staging name (<key>.cinfo, .meta, stage markers, upload
72 * temps) resolves normally; all other validation (NULL→403, depth/dotdot→403,
73 * overflow→414) is identical and still runs. Returns 0 or an HTTP status.
74 *
75 * WHY: A remote WebDAV/S3 cache-STORE endpoint legitimately PUTs and later GETs a
76 * <key>.cinfo sidecar over the same HTTP surface, so on a location explicitly
77 * configured as a trusted store (brix_cache_store_endpoint on) the reserved
78 * name must be a valid target — while every normal client location keeps
79 * allow_internal=0 and the reserved name stays invisible (404). The switch is
80 * an explicit per-location opt-in, so default-deny is preserved everywhere.
81 *
82 * HOW: Runs the NULL/depth/forbidden-component checks, then the internal-name
83 * check ONLY when allow_internal is zero, then the lexical confined join.
84 */
85int brix_http_resolve_path_ex(const char *root_canon, const char *decoded_path,
86 char *out, size_t outsz, unsigned allow_internal);
