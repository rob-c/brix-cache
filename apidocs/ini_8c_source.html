1/*
2 * token/ini.c — minimal INI parser (see ini.h).
3 *
4 * WHAT: Reads an INI file and dispatches every `key = value` line to a caller
5 * callback, tracking the current `[section]`. Used to parse the upstream
6 * SciTokens `scitokens.cfg` grammar (issuer_registry.c) and the throttle
7 * userconfig file (Phase-59 W3a), so both share one auditable reader.
8 * WHY: XRootD ships these as INI files; parsing the grammar verbatim lets an
9 * operator point us at their existing config unchanged (phase-59 ADR-1).
10 * HOW: Line-buffered fgets; strip comments + surrounding whitespace; `[name]`
11 * updates the section; `key = value` splits on the first `=` and invokes
12 * the callback. Pure C, no nginx runtime — unit-testable standalone.
13 */

81/* brix_ini_parse_file — parse path, dispatch each key line */int
82brix_ini_parse_file(const char *path, brix_ini_cb cb, void *user,
83 char *errbuf, size_t errlen)
84{
85 FILE *f;
86 char line[1024];
87 char section[64] = "";
88 int rc = 0;
89 int lineno = 0;
90
91 f = fopen(path, "re"); /* 'e' = O_CLOEXEC */
92 if (f == NULL) {
93 snprintf(errbuf, errlen, "open %s: %s", path, strerror(errno));
94 return -1;
95 }
96
97 while (fgets(line, sizeof(line), f) != NULL) {
98 char *s;
99 char *eq;
100
101 lineno++;
102 ini_drop_comment(line);
103 s = ini_trim(line);
104 if (*s == '\0') {
105 continue;
106 }
107 if (*s == '[') {
108 if (ini_section(s, section, sizeof(section)) != 0) {
109 snprintf(errbuf, errlen, "%s:%d: bad section header",
110 path, lineno);
111 rc = -1;
112 break;
113 }
114 continue;
115 }
116 eq = strchr(s, '=');
117 if (eq == NULL) {
118 snprintf(errbuf, errlen, "%s:%d: missing '='", path, lineno);
119 rc = -1;
120 break;
121 }
122 *eq = '\0';
123 rc = cb(user, section, ini_trim(s), ini_trim(eq + 1));
124 if (rc != 0) {
125 snprintf(errbuf, errlen, "%s:%d: rejected key", path, lineno);
126 break;
127 }
128 }
129
130 fclose(f);
131 return rc;
132}
