16/* ---- Validate the capture arguments ----
17 *
18 * WHAT: Returns 1 when argv, argv[0], and the output buffer are all usable
19 * (non-NULL argv/argv[0]/out and a non-zero outsz); returns 0 otherwise.
20 *
21 * WHY: Keeps the argument-guard branch ladder out of the orchestrator so the
22 * top-level function stays a flat, low-complexity sequence of steps.
23 *
24 * HOW:
25 * 1. Reject a NULL argv, a NULL argv[0], a NULL out, or a zero outsz.
26 * 2. Otherwise report the arguments as valid.
27 */
28static int
29brix_subprocess_args_ok(char *const argv[], const char *out, size_t outsz)
30{
31 if (argv == NULL || argv[0] == NULL || out == NULL || outsz == 0) {
32 return 0;
33 }
34 return 1;
35}

37/* ---- Run the forked child: wire stdout to the pipe and exec ----
38 *
39 * WHAT: In the child process, redirects stdout to the pipe write end and execs
40 * argv. Never returns to the caller: on any failure it calls _exit(127), and on
41 * success control passes to the executed image.
42 *
43 * WHY: The child half of the fork is a distinct, self-contained responsibility;
44 * isolating it keeps the fd wiring and mask restore linear and auditable.
45 *
46 * HOW:
47 * 1. Close the pipe read end (the child only writes).
48 * 2. dup2 the pipe write end onto STDOUT_FILENO; _exit(127) if that fails.
49 * 3. Close the now-redundant original write-end fd. The dup'd stdout stays
50 * open across execvp — it IS the capture channel; the kernel reclaims it at
51 * _exit. (gcc >= 13 -fanalyzer reports it as an fd leak: it does not model
52 * exec/_exit ending the image — known false positive.)
53 * 4. Restore the inherited signal mask so the mask is not leaked to the exec'd
54 * program.
55 * 5. execvp argv; on failure (command not installed) _exit(127).
56 */
57static void
58brix_subprocess_child(int pfd[2], char *const argv[], const sigset_t *old)
59{
60 close(pfd[0]);
61 if (dup2(pfd[1], STDOUT_FILENO) < 0) {
62 _exit(127);
63 }
64 close(pfd[1]);
65 sigprocmask(SIG_SETMASK, old, NULL); /* don't leak the mask to the child */
66 execvp(argv[0], argv);
67 _exit(127); /* exec failed (command not installed) */
68}

70/* ---- Drain the child's stdout into the output buffer ----
71 *
72 * WHAT: Reads from fd into out until the buffer is full (leaving room for a
73 * trailing NUL), EOF, or a non-EINTR error; NUL-terminates out and returns the
74 * number of bytes captured.
75 *
76 * WHY: The bounded read loop is the parent's only real work and has its own
77 * EINTR/EOF handling; keeping it separate isolates that control flow from the
78 * fork/reap plumbing.
79 *
80 * HOW:
81 * 1. While at least one byte of space remains before the reserved NUL, read
82 * into out at the current offset.
83 * 2. On a negative return, retry on EINTR, else stop.
84 * 3. On a zero return (EOF), stop.
85 * 4. Otherwise advance the captured-byte count.
86 * 5. NUL-terminate at the captured length and return that length.
87 */
88static size_t
89brix_subprocess_drain(int fd, char *out, size_t outsz)
90{
91 size_t got = 0;
92
93 while (got + 1 < outsz) {
94 ssize_t r = read(fd, out + got, outsz - 1 - got);
95 if (r < 0) {
96 if (errno == EINTR) {
97 continue;
98 }
99 break;
100 }
101 if (r == 0) {
102 break;
103 }
104 got += (size_t) r;
105 }
106 out[got] = '\0';
107 return got;
108}

110/* ---- Reap the child and report its exit status ----
111 *
112 * WHAT: waitpid()s for pid (retrying on EINTR), then returns 0 and, when
113 * exit_code is non-NULL, stores the child's exit status; returns -1 if the child
114 * did not exit normally (killed by a signal, etc.).
115 *
116 * WHY: Collapsing the waitpid retry loop and the WIFEXITED decoding into one
117 * helper keeps the status semantics in a single place and out of the
118 * orchestrator's branch count.
119 *
120 * HOW:
121 * 1. waitpid for the child, retrying while it fails with EINTR.
122 * 2. If the child did not exit normally, return -1.
123 * 3. Otherwise store WEXITSTATUS into exit_code when provided and return 0.
124 */
125static int
126brix_subprocess_reap(pid_t pid, int *exit_code)
127{
128 int status = 0;
129
130 while (waitpid(pid, &status, 0) < 0 && errno == EINTR) {
131 /* retry */
132 }
133 if (!WIFEXITED(status)) {
134 return -1; /* killed by a signal, etc. */
135 }
136 if (exit_code != NULL) {
137 *exit_code = WEXITSTATUS(status);
138 }
139 return 0;
140}

142/* ---- Capture a child command's stdout ----
143 *
144 * WHAT: Forks and execs argv, capturing its stdout into out (up to outsz-1 bytes
145 * plus a NUL). Returns 0 on a normal child exit (storing the byte count in
146 * out_len and the exit status in exit_code when non-NULL), or -1 on any setup
147 * failure or abnormal child termination.
148 *
149 * WHY: Gives ngx-free callers a single, side-effect-honest primitive for running
150 * a helper command and reading its output without leaking fds, child processes,
151 * or a modified signal mask.
152 *
153 * HOW:
154 * 1. Zero the caller's out_len/exit_code and validate the arguments.
155 * 2. Create the capture pipe.
156 * 3. Block SIGCHLD so a host reaper cannot steal our child before waitpid();
157 * the mask is restored before every return.
158 * 4. fork; on failure restore the mask, close both pipe ends, and fail.
159 * 5. In the child, hand off to brix_subprocess_child (never returns).
160 * 6. In the parent, close the write end, drain stdout, close the read end.
161 * 7. Reap the child, restore the signal mask, publish out_len, and return the
162 * reap result.
163 */
164int
165brix_subprocess_capture(char *const argv[], char *out, size_t outsz,
166 size_t *out_len, int *exit_code)
167{
168 int pfd[2];
169 pid_t pid;
170 size_t got;
171 int rc;
172 sigset_t block, old;
173
174 if (out_len != NULL) { *out_len = 0; }
175 if (exit_code != NULL) { *exit_code = -1; }
176 if (!brix_subprocess_args_ok(argv, out, outsz)) {
177 return -1;
178 }
179 if (pipe(pfd) != 0) {
180 return -1;
181 }
182 /* Block SIGCHLD so a host SIGCHLD handler (e.g. nginx's child reaper) cannot
183 * reap our child before waitpid(); the child restores the mask before exec so
184 * the executed program is unaffected. Restored before every return. */
185 sigemptyset(&block);
186 sigaddset(&block, SIGCHLD);
187 sigprocmask(SIG_BLOCK, &block, &old);
188
189 pid = fork();
190 if (pid < 0) {
191 sigprocmask(SIG_SETMASK, &old, NULL);
192 close(pfd[0]);
193 close(pfd[1]);
194 return -1;
195 }
196 if (pid == 0) {
197 brix_subprocess_child(pfd, argv, &old);
198 }
199
200 close(pfd[1]);
201 got = brix_subprocess_drain(pfd[0], out, outsz);
202 close(pfd[0]);
203
204 rc = brix_subprocess_reap(pid, exit_code);
205 sigprocmask(SIG_SETMASK, &old, NULL);
206
207 if (out_len != NULL) {
208 *out_len = got;
209 }
210 return rc;
211}

213/* ---- Strip trailing whitespace from a C string ----
214 *
215 * WHAT: Removes trailing newline, carriage-return, space, and tab characters
216 * from s in place and returns the resulting length; returns 0 for a NULL s.
217 *
218 * WHY: Command output captured by brix_subprocess_capture routinely carries a
219 * trailing newline that callers must not treat as part of a token or value.
220 *
221 * HOW:
222 * 1. Return 0 immediately for a NULL string.
223 * 2. Measure the current length.
224 * 3. While the last character is a trailing whitespace byte, overwrite it with
225 * NUL and shrink the length.
226 * 4. Return the trimmed length.
227 */
228size_t
229brix_rstrip(char *s)
230{
231 size_t n;
232
233 if (s == NULL) {
234 return 0;
235 }
236 n = strlen(s);
237 while (n > 0 && (s[n - 1] == '\n' || s[n - 1] == '\r'
238 || s[n - 1] == ' ' || s[n - 1] == '\t')) {
239 s[--n] = '\0';
240 }
241 return n;
242}
