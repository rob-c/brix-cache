1/*
2 * beneath.c — kernel-enforced path-confinement primitives (openat2/RESOLVE_BENEATH).
3 *
4 * WHAT: Implements the "beneath" API declared in beneath.h: a small set of
5 * filesystem primitives (open/stat/unlink/mkdir/rename/link) that operate
6 * on a client-supplied reqpath but are GUARANTEED never to escape a
7 * per-worker O_PATH rootfd anchored on the export root. The core wrapper
8 * do_openat2() drives the openat2(2) syscall with RESOLVE_BENEATH |
9 * RESOLVE_NO_MAGICLINKS; brix_open_beneath()/brix_stat_beneath() use it
10 * directly, while the path-MUTATING ops route through beneath_open_parent().
11 *
12 * WHY: This layer IS the confinement boundary (see op_path.c: the lexical
13 * 'resolved' string is for ACL/logging only — it is NOT the security
14 * boundary). XRootD/WebDAV/S3 clients send arbitrary paths; the kernel,
15 * not nginx code, must refuse any traversal out of the export root.
16 * RESOLVE_BENEATH rejects absolute paths, "..", and symlinks that would
17 * leave the rootfd subtree, atomically and without TOCTOU races.
18 * RESOLVE_NO_MAGICLINKS additionally blocks /proc/<pid>/fd-style escapes.
19 * An escape attempt surfaces to callers as EXDEV/ELOOP, which the error
20 * mapper turns into kXR_NotAuthorized / HTTP 403.
21 *
22 * HOW: openat2(2) is a hard build requirement (the #error guards below fail the
23 * compile on kernels/headers < 5.6). do_openat2() masks O_CREAT mode bits
24 * to 07777 because openat2() — unlike open(2) — rejects S_IFMT type bits.
25 * The crucial subtlety: RESOLVE_BENEATH protects ONLY the openat2() call
26 * itself; the legacy *at() syscalls (mkdirat/unlinkat/renameat/linkat) do
27 * NOT honour it, so a symlink in an intermediate component could be
28 * followed out of the root. beneath_open_parent() closes that hole by
29 * resolving the PARENT directory under RESOLVE_BENEATH and then performing
30 * the *at() op on the final component name only (which those syscalls do
31 * not dereference as a symlink for delete/rename/create).
32 */

51 *
52 * When `brix_impersonation map` is active and a per-request principal is set,
53 * brix_imp_client_active() returns true and the confined helpers below delegate

71/*
72 * Open an O_PATH directory fd on root_canon to anchor every beneath() call.
73 * O_PATH gives a traversal-only handle (no read/write to the dir itself),
74 * O_DIRECTORY enforces it is a directory, O_CLOEXEC stops fd leaks across exec.
75 * Returns fd >= 0 (caller closes) or -1 with errno set. Hot paths should reuse
76 * a persistent per-worker rootfd rather than calling this per request.

80{
81 return open(root_canon, O_PATH | O_DIRECTORY | O_CLOEXEC);

84/*
85 * do_openat2 — the single chokepoint through which every confined open passes.
86 * `rel` is a root-RELATIVE path (no leading '/'); RESOLVE_BENEATH treats rootfd
87 * as the top of the tree and refuses to follow anything — "..", absolute paths,
88 * or symlinks — that would resolve outside it, returning EXDEV/ELOOP instead.
89 * RESOLVE_NO_MAGICLINKS additionally blocks /proc/<pid>/fd magic symlinks.
90 * NOTE: this guarantee applies ONLY to the open performed here; mutating *at()
91 * ops must pre-resolve their parent via beneath_open_parent() (see below).
92 */
93static int
94do_openat2_resolve(int rootfd, const char *rel, int flags, mode_t mode,
95 uint64_t resolve)
96{
97 struct open_how how;
98 /* empty rel means root dir itself; "." opens the rootfd directory */
99 if (rel[0] == '\0') { rel = "."; }
100 ngx_memzero(&how, sizeof(how));
101 /*
102 * Force O_NONBLOCK on real (data-bearing) opens so the open(2) itself can
103 * NEVER block the worker. Opening a FIFO O_RDONLY (or a device with a
104 * blocking open routine) without O_NONBLOCK parks the caller in the kernel
105 * "wait_for_partner" state until a peer appears — which, on a single-threaded
106 * nginx worker, freezes the event loop and stalls EVERY connection pinned to
107 * that worker (a remotely triggerable DoS: a directory entry that is a named
108 * pipe wedges the worker on the kXR_open / GET path). The flag is a no-op
109 * for the regular files we serve (local-fs reads never return EAGAIN), and
110 * callers reject any non-regular object after fstat()ing the fd.
111 *
112 * It must NOT be added to an O_PATH open: openat2() is stricter than open(2)
113 * and rejects O_PATH|O_NONBLOCK with EINVAL. O_PATH opens only reference the
114 * inode (stat/lstat, the persistent root anchor, parent resolution for
115 * mutating ops) and never start file I/O, so they cannot block on a FIFO and
116 * need no guard.
117 */
118 how.flags = (uint64_t)(flags | O_CLOEXEC
119 | ((flags & O_PATH) ? 0 : O_NONBLOCK));
120 how.resolve = resolve;
121 /*
122 * openat2() is stricter than open()/openat(): it rejects (EINVAL) any
123 * how.mode bit outside 07777. Callers legitimately pass a full struct
124 * stat st_mode (e.g. staged_file copying a source's permissions), which
125 * carries the S_IFMT type bits. Mask to the permission bits, exactly as
126 * open(2) does by ignoring the type bits, so a struct-stat mode is accepted.
127 */
128 if (flags & O_CREAT) { how.mode = (uint64_t)(mode & 07777); }
129 return (int)syscall(SYS_openat2, rootfd, rel, &how, sizeof(how));
130}

132/* The default confinement for open/mutate paths: RESOLVE_BENEATH (no symlinks,
133 * no "..", no absolute paths) + RESOLVE_NO_MAGICLINKS (refuse /proc/PID/fd). */
134static int
135do_openat2(int rootfd, const char *rel, int flags, mode_t mode)
136{
137 return do_openat2_resolve(rootfd, rel, flags, mode,
138 RESOLVE_BENEATH | RESOLVE_NO_MAGICLINKS);
139}

150int
151brix_stat_beneath(int rootfd, const char *reqpath, struct stat *st)
152{
153 int fd, rc;
154
155 if (brix_imp_client_active()) {
156 return brix_imp_stat(reqpath, st, 0 /* follow */);
157 }
158 /* O_PATH without O_NOFOLLOW: follow symlinks. RESOLVE_IN_ROOT (not BENEATH)
159 * so an in-export symlink — including one with an ABSOLUTE target that points
160 * back into the export — is followed chroot-style (the kernel resolves "/" and
161 * ".." relative to rootfd) and still CANNOT escape the root. BENEATH rejected
162 * every absolute/symlink target outright (EXDEV), diverging from stock, which
163 * follows in-export links (test_conf_stattypes). */
164 fd = do_openat2_resolve(rootfd, brix_beneath_rel(reqpath), O_PATH, 0,
165 RESOLVE_IN_ROOT | RESOLVE_NO_MAGICLINKS);
166 if (fd < 0) { return -1; }
167 rc = fstat(fd, st);
168 close(fd);
169 return rc;
170}

172int
173brix_lstat_beneath(int rootfd, const char *reqpath, struct stat *st)
174{
175 int fd, rc;
176
177 if (brix_imp_client_active()) {
178 return brix_imp_stat(reqpath, st, 1 /* nofollow */);
179 }
180 /* O_PATH | O_NOFOLLOW: do NOT follow a trailing symlink, so fstat() reports
181 * the link itself (lstat semantics). RESOLVE_IN_ROOT (chroot-style) confines
182 * the path while still permitting in-export symlinks in INTERMEDIATE
183 * components; the trailing link is opened as itself via O_NOFOLLOW. (BENEATH
184 * rejected even a trailing symlink here with EXDEV, so lstat of a link failed.) */
185 fd = do_openat2_resolve(rootfd, brix_beneath_rel(reqpath),
186 O_PATH | O_NOFOLLOW, 0,
187 RESOLVE_IN_ROOT | RESOLVE_NO_MAGICLINKS);
188 if (fd < 0) { return -1; }
189 rc = fstat(fd, st);
190 close(fd);
191 return rc;
192}

194DIR *
195brix_opendir_beneath(int rootfd, const char *reqpath)
196{
197 int fd;
198 DIR *d;
199
200 /* O_RDONLY | O_DIRECTORY under RESOLVE_IN_ROOT (chroot-style): a trailing
201 * in-export directory symlink is followed within the export — matching stock,
202 * which lists through in-export links — while an outward/absolute target is
203 * resolved relative to rootfd and so lands on a non-existent in-root path
204 * (ENOENT) or is refused (EXDEV) rather than escaping the export root. The
205 * legacy bare opendir() followed an outward link straight out of the root. */
206 fd = do_openat2_resolve(rootfd, brix_beneath_rel(reqpath),
207 O_RDONLY | O_DIRECTORY, 0,
208 RESOLVE_IN_ROOT | RESOLVE_NO_MAGICLINKS);
209 if (fd < 0) { return NULL; }
210 d = fdopendir(fd);
211 if (d == NULL) { close(fd); }
212 return d; /* closedir() closes the fd */
213}

215/*
216 * SECURITY: the *at() family (mkdirat / unlinkat / renameat / linkat) does NOT
217 * honour RESOLVE_BENEATH — only openat2() does. Handing them a multi-component
218 * relative path lets a symlink in an INTERMEDIATE component (e.g. /link -> /etc,
219 * then "/link/x") be followed straight out of the export root. openat2()'s
220 * RESOLVE_BENEATH only guards the open itself, so for every path-MUTATING op we
221 * must resolve the PARENT directory ourselves under RESOLVE_BENEATH and then
222 * operate on the final component name only. The final component is safe for
223 * these ops: unlinkat/renameat never follow a trailing symlink, and mkdir/link
224 * create a fresh name.
225 *
226 * beneath_open_parent() returns an O_PATH fd on the confined parent of reqpath
227 * (caller closes) and points *base at the final component (into reqpath).
228 * pbuf must be >= PATH_MAX. Returns -1 (errno: EXDEV/ELOOP on escape).
229 */
230static int
231beneath_open_parent(int rootfd, const char *reqpath,
232 char *pbuf, size_t pbufsz, const char **base)
233{
234 const char *rel = brix_beneath_rel(reqpath); /* strip leading '/'s */
235 const char *slash = strrchr(rel, '/');
236 size_t plen;
237
238 if (slash == NULL || slash == rel) {
239 /* single component, or only a leading-slash remnant → parent is root */
240 *base = (slash == NULL) ? rel : slash + 1;
241 return do_openat2(rootfd, ".", O_PATH | O_DIRECTORY, 0);
242 }
243
244 *base = slash + 1;
245 plen = (size_t) (slash - rel);
246 if (plen >= pbufsz) {
247 errno = ENAMETOOLONG;
248 return -1;
249 }
250 memcpy(pbuf, rel, plen);
251 pbuf[plen] = '\0';
252 return do_openat2(rootfd, pbuf, O_PATH | O_DIRECTORY, 0);
253}

314 e = errno; close(sfd); close(dfd); errno = e;
315 return rc;

318/*
319 * Atomic create-if-absent rename: renameat2(RENAME_NOREPLACE) on the final
320 * components, confined under rootfd exactly like brix_rename_beneath(). On a
321 * kernel/filesystem without RENAME_NOREPLACE (ENOSYS/EINVAL) it falls back to a
322 * plain renameat — logged once — so behaviour degrades to the legacy
323 * last-writer-wins rather than spuriously failing (callers still ran their
324 * stat-based precondition, so this is no worse than before W6b on such hosts).
325 * Returns 0; or -1 with errno (EEXIST when the destination already exists).
326 */
327int
328brix_rename_beneath_excl(int rootfd, const char *src, const char *dst)
329{
330 char sbuf[PATH_MAX], dbuf[PATH_MAX];
331 const char *sbase, *dbase;
332 int sfd, dfd, rc, e;
333
334 if (brix_imp_client_active()) {
335 return brix_imp_rename_noreplace(src, dst);
336 }
337
338 sfd = beneath_open_parent(rootfd, src, sbuf, sizeof(sbuf), &sbase);
339 if (sfd < 0) { return -1; }
340 dfd = beneath_open_parent(rootfd, dst, dbuf, sizeof(dbuf), &dbase);
341 if (dfd < 0) { e = errno; close(sfd); errno = e; return -1; }
342 if (sbase[0] == '\0' || dbase[0] == '\0') {
343 close(sfd); close(dfd); errno = EINVAL; return -1;
344 }
345
346 rc = (int) syscall(SYS_renameat2, sfd, sbase, dfd, dbase,
347 (unsigned int) RENAME_NOREPLACE);
348 if (rc != 0 && (errno == ENOSYS || errno == EINVAL)) {
349 static int warned = 0;
350 if (!warned) {
351 warned = 1;
352 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, errno,
353 "brix: renameat2(RENAME_NOREPLACE) unsupported; "
354 "create-if-absent falls back to non-atomic rename");
355 }
356 /* phase74-fp: prototype order (olddirfd, oldpath, newdirfd, newpath). */
357 rc = renameat(sfd, sbase, dfd, dbase); /* NOLINT(readability-suspicious-call-argument) */
358 }
359 e = errno; close(sfd); close(dfd); errno = e;
360 return rc;
361}
