26/* Confined setattr — utimensat + fchownat under parent confinement.
27 *
28 * Apply timestamps (set_times → utimensat) and/or owner (set_owner →
29 * fchownat) to <resolved>, both *at() syscalls anchored at the confined
30 * parent fd so a symlink/parent swap cannot redirect the change outside the
31 * root. AT_SYMLINK_NOFOLLOW is used so the operation never follows a final
32 * symlink. uid/gid of (uid_t)-1 / (gid_t)-1 leave that id unchanged.
33 * kXR_chmod already covers mode, so mode is intentionally not handled here.
34 * Returns 0 on success, -1 with errno set.
35 */
36int
37brix_setattr_confined_canon(ngx_log_t *log, const char *root_canon,
38 const char *resolved, int set_times, const struct timespec times[2],
39 int set_owner, uid_t uid, gid_t gid)
40{
41 char base[NAME_MAX + 1];
42 int parentfd;
43 int rc = 0;
44 int saved_errno;
45
46 /* Phase 40: route through the broker (as the mapped user) when active. */
47 if (brix_imp_client_active()) {
48 char rel[PATH_MAX];
49 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
50 rel, sizeof(rel)))
51 {
52 return -1;
53 }
54 return brix_imp_setattr(rel, set_times, times, set_owner, uid, gid);
55 }
56
57 parentfd = brix_open_confined_parent_canon(log, root_canon, resolved,
58 base, sizeof(base));
59 if (parentfd < 0) {
60 return -1;
61 }
62 if (set_times && utimensat(parentfd, base, times, AT_SYMLINK_NOFOLLOW) != 0) {
63 rc = -1;
64 }
65 if (rc == 0 && set_owner
66 && fchownat(parentfd, base, uid, gid, AT_SYMLINK_NOFOLLOW) != 0) {
67 rc = -1;
68 }
69 saved_errno = errno;
70 close(parentfd);
71 errno = saved_errno;
72 return rc;
73}

75/* Confined chmod — broker-routed under impersonation.
76 *
77 * Apply <mode> (low 12 bits) to <resolved>. Under impersonation routes
78 * through the broker so the chmod runs AS THE MAPPED USER (a chmod requires
79 * being the file's owner; the unprivileged worker is not, so a worker-local
80 * chmod of a user-owned file would EPERM — even for the file's real owner).
81 * Off impersonation, fchmodat anchored at the confined parent fd (so a
82 * symlink/parent swap cannot redirect the change outside the root).
83 * Returns 0 on success, -1 with errno set.
84 */
85int
86brix_chmod_confined_canon(ngx_log_t *log, const char *root_canon,
87 const char *resolved, mode_t mode)
88{
89 char base[NAME_MAX + 1];
90 int parentfd;
91 int rc;
92 int saved_errno;
93
94 /* Phase 40: route through the broker (as the mapped user) when active. */
95 if (brix_imp_client_active()) {
96 char rel[PATH_MAX];
97 if (!brix_resolved_relative_to_root(log, root_canon, resolved,
98 rel, sizeof(rel)))
99 {
100 return -1;
101 }
102 /* Defense in depth: strip setuid/setgid centrally (not only in the

115 saved_errno = errno;
116 close(parentfd);
117 errno = saved_errno;
