3 * WHAT: haversine great-circle distance and a stable argsort used to turn

16double
17brix_cvmfs_haversine_km(double lat1, double lon1, double lat2, double lon2)
18{
19 double rl1 = lat1 * M_PI / 180.0, rl2 = lat2 * M_PI / 180.0;
20 double dla = (lat2 - lat1) * M_PI / 180.0;
21 double dlo = (lon2 - lon1) * M_PI / 180.0;
22 double a = sin(dla / 2) * sin(dla / 2)
23 + cos(rl1) * cos(rl2) * sin(dlo / 2) * sin(dlo / 2);
24
25 return 2.0 * CVMFS_EARTH_RADIUS_KM * atan2(sqrt(a), sqrt(1.0 - a));
26}

29brix_cvmfs_rank_by_metric(const double *metric, int n, int *ranks)
30{
31 int i, j, better;
32
33 for (i = 0; i < n; i++) {
34 better = 0;
35 for (j = 0; j < n; j++) {
36 if (metric[j] < metric[i]
37 || (metric[j] == metric[i] && j < i))
38 {
39 better++;
40 }
41 }
42 ranks[i] = better;
43 }
44}
