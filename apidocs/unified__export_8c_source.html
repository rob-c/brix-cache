3/*
4 * unified_export.c — scrape-time exporter for the non-io unified families and
5 * the top-level brix_export_unified_metrics orchestrator.
6 *
7 * WHAT: Renders the cred_select, cache (hits/misses/evicted + watermark reaper),
8 * write-back staging, auth, and tpc Prometheus families, and hosts the
9 * brix_export_unified_metrics entry point that fans out over every
10 * unified_emit_<family> helper (the io families live in
11 * unified_export_io.c). Also provides unified_emit_proto_counter, the
12 * generic per-proto counter renderer these families share.
13 * WHY: The exporter half of unified.c exceeded the file-size budget; this file
14 * owns the credential/cache/auth/tpc families and the orchestrator, while
15 * unified_export_io.c owns the byte/op/latency families and the legacy
16 * fold — two cohesive clusters, each in its own file (coding-standards §1).
17 * HOW: Each emitter reads its region of *shm via brix_metric_value and prints
18 * HELP/TYPE + per-label lines; labels stay low-cardinality (INVARIANT #8).
19 * The auth family folds in brix_unified_legacy_auth (root:// only). The
20 * orchestrator is a flat call sequence with frozen emission order/bytes.
21 */

23/*
24 * unified_emit_proto_counter — emit a single HELP/TYPE header followed by one
25 * per-protocol line reading counter values from `field`.
26 *
27 * WHAT: Generic per-proto counter renderer (proto label only, no fold).
28 * WHY: The cred_select and cache hit/miss/evicted families are all identical
29 * "HELP/TYPE + per-proto value" shapes; a shared emitter removes the
30 * copy-paste while keeping exposition bytes frozen.
31 * HOW: Caller passes the pre-formatted HELP/TYPE block and the per-proto
32 * counter array base; we print `metric_name{proto="…"} <value>` per proto.
33 */
34static void
35unified_emit_proto_counter(metrics_writer_t *mw, const char *help_type,
36 const char *metric_name, ngx_atomic_t *field)
37{
38 ngx_uint_t proto;
39
40 mw_printf(mw, "%s", help_type);
41 for (proto = 0; proto < BRIX_PROTO_COUNT; proto++) {
42 mw_printf(mw, "%s{proto=\"%s\"} %llu\n", metric_name,
43 brix_metric_proto_name((brix_proto_t) proto),
44 brix_metric_value(&field[proto]));
45 }
46}

178/*
179 * unified_emit_auth — render the brix_auth_total family
180 * (per-proto/method/status authentication counters).
181 *
182 * WHAT: Emits the HELP/TYPE header + one line per (proto, method, status) cell.
183 * WHY: Isolates the triple-nested loop and the root:// legacy auth fold.
184 * HOW: Each cell folds in brix_unified_legacy_auth (non-zero for root:// only)
185 * before printing; method/status labels come from the shared name tables.
186 */
187static void
188unified_emit_auth(metrics_writer_t *mw, ngx_brix_metrics_t *shm)
189{
190 ngx_uint_t proto, method, status;
191 unsigned long long value;
192
193 mw_printf(mw,
194 "# HELP brix_auth_total Authentication attempts by protocol, method, and status.\n"
195 "# TYPE brix_auth_total counter\n");
196 for (proto = 0; proto < BRIX_PROTO_COUNT; proto++) {
197 for (method = 0; method < BRIX_METRIC_AUTH_COUNT; method++) {
198 for (status = 0; status < BRIX_METRIC_AUTH_STATUS_COUNT; status++) {
199 value = brix_metric_value(
200 &shm->unified.auth_total[proto][method][status]);
201 value += brix_unified_legacy_auth(
202 shm, (brix_proto_t) proto, method, status);
203 mw_printf(mw,
204 "brix_auth_total"
205 "{proto=\"%s\",method=\"%s\",status=\"%s\"} %llu\n",
206 brix_metric_proto_name((brix_proto_t) proto),
207 brix_unified_auth_names[method],
208 status == BRIX_METRIC_AUTH_OK ? "ok" : "fail",
209 value);
210 }
211 }
212 }
213}

215/*
216 * unified_emit_tpc — render the brix_tpc_transfers_total + brix_tpc_bytes_total
217 * families (third-party-copy outcomes and successful bytes).
218 *
219 * WHAT: Emits the per-(proto, direction, status) transfer counters and the
220 * per-(proto, direction) successful-byte counters.
221 * WHY: The two TPC families share the proto × direction iteration, so one
222 * helper keeps them together.
223 * HOW: Two loops: transfers add the status dimension, bytes do not.
224 */
225static void
226unified_emit_tpc(metrics_writer_t *mw, ngx_brix_metrics_t *shm)
227{
228 ngx_uint_t proto, direction, err;
229
230 mw_printf(mw,
231 "# HELP brix_tpc_transfers_total Third-party-copy transfer outcomes.\n"
232 "# TYPE brix_tpc_transfers_total counter\n");
233 for (proto = 0; proto < BRIX_PROTO_COUNT; proto++) {
234 for (direction = 0; direction < BRIX_METRIC_TPC_DIRECTION_COUNT;
235 direction++)
236 {
237 for (err = 0; err < BRIX_ERR_COUNT; err++) {
238 mw_printf(mw,
239 "brix_tpc_transfers_total"
240 "{proto=\"%s\",direction=\"%s\",status=\"%s\"} %llu\n",
241 brix_metric_proto_name((brix_proto_t) proto),
242 brix_unified_tpc_direction_names[direction],
243 brix_metric_err_name((brix_err_class_t) err),
244 brix_metric_value(&shm->unified.tpc_transfers
245 [proto][direction][err]));
246 }
247 }
248 }
249
250 mw_printf(mw,
251 "# HELP brix_tpc_bytes_total Successful third-party-copy bytes.\n"
252 "# TYPE brix_tpc_bytes_total counter\n");
253 for (proto = 0; proto < BRIX_PROTO_COUNT; proto++) {
254 for (direction = 0; direction < BRIX_METRIC_TPC_DIRECTION_COUNT;
255 direction++)
256 {
257 mw_printf(mw,
258 "brix_tpc_bytes_total{proto=\"%s\",direction=\"%s\"} %llu\n",
259 brix_metric_proto_name((brix_proto_t) proto),
260 brix_unified_tpc_direction_names[direction],
261 brix_metric_value(&shm->unified.tpc_bytes[proto][direction]));
262 }
263 }
264}
