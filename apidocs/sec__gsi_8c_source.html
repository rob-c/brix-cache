1/*
2 * sec_gsi.c — GSI (X.509 proxy) auth module, client side.
3 *
4 * WHAT: Drives the XRootD GSI two-round exchange against the nginx dialect:
5 * round 1 client → kXGC_certreq; server → kXGS_cert (kXR_authmore) with its DH
6 * public key + cert + cipher list.
7 * round 2 client → kXGC_cert: our DH pubkey + selected cipher + AES-256-CBC
8 * (kXRS_main) wrapping inner "gsi\0"+kXGC_cert+kXRS_x509(proxy PEM);
9 * server → kXR_ok.
10 * WHY: X.509 proxies are the dominant WLCG auth.
11 * HOW: All the DH/bucket/secret math lives in the shared gsi_core (compiled into
12 * both this client and the nginx module); this file is only the orchestration
13 * plus the two client-specific bits: loading the proxy PEM and AES-encrypting
14 * the inner buffer. signing_key = SHA256(secret) via libxrdproto's brix_sha256.
15 *
16 * Clean-room: every wire fact is the inverse of our own server code under src/auth/gsi.
17 */

37/* Resolve the proxy path: $X509_USER_PROXY, else /tmp/x509up_u<uid>. */
38static void
39proxy_path(char *out, size_t outsz)
40{
41 const char *env = getenv("X509_USER_PROXY");
42 if (env != NULL && env[0] != '\0') {
43 snprintf(out, outsz, "%s", env);
44 } else {
45 snprintf(out, outsz, "/tmp/x509up_u%u", (unsigned) geteuid());
46 }
47}

49static int
50gsi_have(brix_conn *c)
51{
52 char path[512];
53 /* The credential store is a fast "yes" when it can confirm a proxy, but a
54 * store MISS must fall through to env/default discovery — otherwise a tool
55 * that attaches a store with no x509 handler (or an empty store) would skip
56 * GSI entirely even though $X509_USER_PROXY is valid. This keeps gsi_have
57 * symmetric with gsi_more, which already falls back to proxy_path(). */
58 if (c != NULL && c->opts.cred != NULL
59 && brix_cred_available(c->opts.cred, XRDC_CRED_X509_PROXY)) {
60 return 1;
61 }
62 proxy_path(path, sizeof(path));
63 return access(path, R_OK) == 0;
64}

67/* The GSI proxy holds a private key at a predictable /tmp/x509up_u<uid>, so it is
68 * opened through brix_credfile_bio (secret=1: no symlink, owned by euid, 0600) —
69 * the shared safe reader in proxy.c. Probe-quiet: NULL just means "no usable
70 * proxy" and the auth driver moves on. */
71#define proxy_bio(path) brix_credfile_bio((path), 1)

73/* Load the proxy cert chain from disk as concatenated PEM (certs only).
74 * path must be the resolved proxy file path (never NULL). */
75static uint8_t *
76load_proxy_pem(const char *path, size_t *outlen)
77{
78 BIO *in, *out;
79 X509 *cert;
80 BUF_MEM *bm;
81 uint8_t *buf = NULL;
82 int n = 0;
83
84 in = proxy_bio(path);
85 if (in == NULL) {
86 return NULL;
87 }
88 out = BIO_new(BIO_s_mem());
89 if (out == NULL) {
90 BIO_free(in);
91 return NULL;
92 }
93 while ((cert = PEM_read_bio_X509(in, NULL, NULL, NULL)) != NULL) {
94 PEM_write_bio_X509(out, cert);
95 X509_free(cert);
96 n++;
97 }
98 ERR_clear_error(); /* the loop's terminating NULL leaves a benign PEM EOF error */
99 BIO_free(in);
100
101 if (n > 0) {
102 BIO_get_mem_ptr(out, &bm);
103 buf = (uint8_t *) malloc(bm->length);
104 if (buf != NULL) {
105 memcpy(buf, bm->data, bm->length);
106 *outlen = bm->length;
107 }
108 }
109 BIO_free(out);
110 return buf;
111}

113/* module callbacks */
114static int
115gsi_first(brix_conn *c, const char *parms, uint8_t **payload, uint32_t *plen,
116 brix_status *st)
117{
118 uint32_t version = 0;
119 char crypto[16] = { 0 };
120 char ca[256] = { 0 };
121 uint8_t *buf;
122 size_t buflen = 0;
123 /* Round-1 random tag (the server signs it). Its lifetime is entirely within
124 * this call — it is folded into the certreq below and never read again in
125 * gsi_more — so it MUST be a stack local: the async manager (xrootdfs) opens
126 * several connections in parallel, and a file-static would let one thread's
127 * tag clobber another's mid-handshake, corrupting the certreq. */
128 uint8_t client_rtag[8];
129
130 (void) c;
131
132 /* The server advertised v:/c:/ca: in the gsi protocol parms; echo crypto+ca
133 * so it picks the right module + CA chain (a stock client does the same). */
134 brix_gsi_parse_parms(parms, &version, crypto, sizeof(crypto),
135 ca, sizeof(ca));
136 if (crypto[0] == '\0') { memcpy(crypto, "ssl", 4); }
137
138 /* phase-48: advertise a pre-DHsigned version (< XrdSecgsiVersDHsigned=10400)
139 * so the (modern) server uses the simpler UNSIGNED DH path: it sends its DH
140 * public as a plain kXRS_puk Public() blob (no RSA-signed DH params), which
141 * we parse with the shared cipher primitives. Proof-of-possession is still
142 * enforced — we sign the server's rtag with the proxy key in round 2.
143 *
144 * phase-48: advertise the modern version (>= XrdSecgsiVersDHsigned=10400) by
145 * default so the server uses the SIGNED-DH path (the DH params are bound to
146 * the certificates — stronger, and what a stock client does). gsi_more
147 * branches on the server's actual response (kXRS_cipher = signed,
148 * kXRS_puk = unsigned), so this stays compatible with old servers too.
149 * XRDC_GSI_VERSION overrides (e.g. 10300 to force the unsigned path). */
150 {
151 const char *vov = getenv("XRDC_GSI_VERSION");
152 version = vov != NULL ? (uint32_t) strtoul(vov, NULL, 10) : 10600;
153 }
154
155 if (!brix_gsi_rand(client_rtag, sizeof(client_rtag))) {
156 brix_status_set(st, XRDC_EAUTH, 0, "gsi: RNG failed");
157 return -1;
158 }
159
160 /* clnt_opts 0x80 matches a stock client's default (delegated-proxy off). */
161 buf = brix_gsi_build_certreq(crypto, version, ca, 0x80u,
162 client_rtag, sizeof(client_rtag), &buflen);
163 if (buf == NULL) {
164 brix_status_set(st, XRDC_EAUTH, 0, "gsi: cannot build certreq");
165 return -1;
166 }
167 *payload = buf; /* ownership → caller */
168 *plen = (uint32_t) buflen;
169 return 0;
170}

188/* Resolve the proxy path (credential store first, else env/default). Shared by
189 * the round-2 (kXGC_cert) and the delegation (kXGC_sigpxy) rounds. */
190static void
191gsi_resolve_proxy(brix_conn *c, char *proxy, size_t sz, brix_status *st)
192{
193 if (c != NULL && c->opts.cred != NULL) {
194 brix_cred_view v;
195 if (brix_cred_acquire(c->opts.cred, XRDC_CRED_X509_PROXY, 0, &v, st) == 0
196 && v.path != NULL) {
197 snprintf(proxy, sz, "%s", v.path);
198 return;
199 }
200 brix_status_clear(st);
201 }
202 proxy_path(proxy, sz);
203}

205/* The XrdSutBuffer step immediately follows the NUL-terminated proto name
206 * ("gsi\0"); read it as a big-endian u32. 0 on a malformed/short buffer. */
207static uint32_t
208gsi_msg_step(const uint8_t *sbody, uint32_t slen)
209{
210 size_t nl;
211
212 if (sbody == NULL || slen < 8) {
213 return 0;
214 }
215 nl = strnlen((const char *) sbody, slen) + 1;
216 if (nl + 4 > (size_t) slen) {
217 return 0;
218 }
219 return ((uint32_t) sbody[nl] << 24) | ((uint32_t) sbody[nl + 1] << 16)
220 | ((uint32_t) sbody[nl + 2] << 8) | (uint32_t) sbody[nl + 3];
221}

223/* Re-encode a PEM X509_REQ (how the server sends kXRS_x509_req) as DER, which
224 * brix_gsi_sign_pxyreq consumes. 0 / -1. */
225static int
226gsi_req_pem_to_der(const uint8_t *pem, size_t pemlen, uint8_t **der, size_t *derlen)
227{
228 BIO *b = BIO_new_mem_buf(pem, (int) pemlen);
229 X509_REQ *req = b ? PEM_read_bio_X509_REQ(b, NULL, NULL, NULL) : NULL;
230 int n;
231 unsigned char *out = NULL, *pp;
232
233 BIO_free(b);
234 if (req == NULL) {
235 return -1;
236 }
237 n = i2d_X509_REQ(req, NULL);
238 if (n > 0 && (out = malloc((size_t) n)) != NULL) {
239 pp = out;
240 if (i2d_X509_REQ(req, &pp) <= 0) {
241 free(out);
242 out = NULL;
243 }
244 }
245 X509_REQ_free(req);
246 if (out == NULL) {
247 return -1;
248 }
249 *der = out;
250 *derlen = (size_t) n;
251 return 0;
252}

254/*
255 * gsi_sigpxy — the X.509 delegation round (kXGS_pxyreq → kXGC_sigpxy). The
256 * server (a TPC destination / tap proxy with delegation on) asks us to sign a
257 * proxy request so it can act AS US upstream. We decrypt its request with the
258 * session cipher agreed in round 2, sign it with our proxy (a key-bearing
259 * delegated proxy the server can use), and return the encrypted signed proxy.
260 * Gated on XRDC_GSI_DELEGATE — handing our credential to a server is opt-in,
261 * mirroring the stock client's XrdSecGSIDELEGPROXY.
262 */
263static int
264gsi_sigpxy(brix_conn *c, const uint8_t *sbody, uint32_t slen, uint8_t **payload,
265 uint32_t *plen, brix_status *st)
266{
267 brix_gsi_cipher_t cipher;
268 const uint8_t *enc = NULL, *reqpem = NULL;
269 size_t enclen = 0, reqpemlen = 0, plainlen = 0;
270 uint8_t *plain = NULL, *reqder = NULL, *signed_pem = NULL;
271 uint8_t *enc2 = NULL, *proxy_pem = NULL;
272 size_t reqderlen = 0, siglen = 0, enc2len = 0, proxy_pem_len = 0;
273 EVP_PKEY *proxy_key = NULL;
274 char proxy[512], err[160];
275 brix_gbuf inner, outer;
276
277 if (getenv("XRDC_GSI_DELEGATE") == NULL) {
278 brix_status_set(st, XRDC_EAUTH, 0,
279 "gsi: server requested X.509 delegation but "
280 "XRDC_GSI_DELEGATE is not set");
281 return -1;
282 }
283 if (!c->gsi_deleg_ready
284 || !brix_gsi_cipher_lookup(c->gsi_deleg_cipher, &cipher)) {
285 brix_status_set(st, XRDC_EAUTH, 0,
286 "gsi: no session cipher for delegation");
287 return -1;
288 }
289
290 /* 1. Decrypt the server's proxy request, extract the PEM kXRS_x509_req. */
291 if (brix_gsi_find_bucket(sbody, slen, (uint32_t) kXRS_main, &enc, &enclen)
292 != 0) {
293 brix_status_set(st, XRDC_EAUTH, 0, "gsi: pxyreq main missing");
294 return -1;
295 }
296 plain = brix_gsi_cipher_decrypt(&cipher, c->gsi_deleg_key, enc, enclen,
297 c->gsi_deleg_use_iv, &plainlen);
298 if (plain == NULL) {
299 brix_status_set(st, XRDC_EAUTH, 0, "gsi: pxyreq decrypt failed");
300 return -1;
301 }
302 if (brix_gsi_find_bucket(plain, plainlen, (uint32_t) kXRS_x509_req,
303 &reqpem, &reqpemlen) != 0
304 || gsi_req_pem_to_der(reqpem, reqpemlen, &reqder, &reqderlen) != 0) {
305 brix_status_set(st, XRDC_EAUTH, 0, "gsi: pxyreq x509_req missing/bad");
306 free(plain);
307 return -1;
308 }
309 free(plain);
310
311 /* 2. Sign the request with our proxy → a key-bearing delegated proxy. */
312 gsi_resolve_proxy(c, proxy, sizeof(proxy), st);
313 proxy_key = load_proxy_key(proxy);
314 proxy_pem = load_proxy_pem(proxy, &proxy_pem_len);
315 err[0] = '\0';
316 {
317 const brix_gsi_blob_t signer_blob = { proxy_pem, proxy_pem_len };
318 const brix_gsi_blob_t req_blob = { reqder, reqderlen };
319 const brix_gsi_err_t err_sink = { err, sizeof(err) };
320 brix_gsi_buf_t signed_out = { NULL, 0 };
321
322 if (proxy_key == NULL || proxy_pem == NULL
323 || brix_gsi_sign_pxyreq(&signer_blob, proxy_key, &req_blob,
324 &signed_out, &err_sink) != 0) {
325 brix_status_set(st, XRDC_EAUTH, 0,
326 err[0] != '\0' ? err
327 : "gsi: cannot sign proxy request");
328 free(reqder);
329 free(proxy_pem);
330 EVP_PKEY_free(proxy_key);
331 return -1;
332 }
333 signed_pem = signed_out.data;
334 siglen = signed_out.len;
335 }
336 free(reqder);
337 free(proxy_pem);
338 EVP_PKEY_free(proxy_key);
339
340 /* 3. Inner main = { kXGC_sigpxy + kXRS_x509(signed proxy) }, AES-encrypted
341 * under the round-2 session cipher — what the server's handle_sigpxy
342 * finds after decrypting kXRS_main. */
343 brix_gbuf_init(&inner);
344 brix_gbuf_init(&outer);
345 brix_gbuf_start(&inner, (uint32_t) kXGC_sigpxy);
346 brix_gbuf_bucket(&inner, (uint32_t) kXRS_x509, signed_pem, siglen);
347 brix_gbuf_end(&inner);
348 free(signed_pem);
349 if (!inner.err) {
350 enc2 = brix_gsi_cipher_encrypt(&cipher, c->gsi_deleg_key, inner.p,
351 inner.len, c->gsi_deleg_use_iv, &enc2len);
352 }
353 brix_gbuf_free(&inner);
354 if (enc2 == NULL) {
355 brix_status_set(st, XRDC_EAUTH, 0, "gsi: sigpxy main encrypt failed");
356 return -1;
357 }
358
359 /* 4. Outer kXGC_sigpxy = { kXRS_main(enc) + kXRS_cipher_alg }. */
360 brix_gbuf_start(&outer, (uint32_t) kXGC_sigpxy);
361 brix_gbuf_bucket(&outer, (uint32_t) kXRS_main, enc2, enc2len);
362 brix_gbuf_bucket(&outer, (uint32_t) kXRS_cipher_alg, c->gsi_deleg_cipher,
363 strlen(c->gsi_deleg_cipher));
364 brix_gbuf_end(&outer);
365 free(enc2);
366 if (outer.err) {
367 brix_gbuf_free(&outer);
368 brix_status_set(st, XRDC_EAUTH, 0, "gsi: sigpxy assembly failed");
369 return -1;
370 }
371 *payload = outer.p; /* ownership → the auth loop (frees it) */
372 *plen = (uint32_t) outer.len;
373 outer.p = NULL;
374 brix_gbuf_free(&outer);
375 return 0;
376}

378/*
379 * gsi_more — round 2 (kXGC_cert), standard XrdSecgsi, both DH variants. The
380 * full handshake (DH variant detection, session-key agreement, proof-of-
381 * possession, encrypted main + outer kXGC_cert assembly) now lives in the shared
382 * gsi_core kernel (brix_gsi_build_cert_response) so the native client and the
383 * TPC destination (src/tpc/gsi/gsi_outbound_exchange.c) drive ONE implementation.
384 * This wrapper only supplies the client's proxy credential (loaded from the
385 * configured proxy file) and maps the kernel's error string onto brix_status.
386 */
387static int
388gsi_more(brix_conn *c, const uint8_t *sbody, uint32_t slen, uint8_t **payload,
389 uint32_t *plen, brix_status *st)
390{
391 char proxy[512];
392 uint8_t *proxy_pem;
393 size_t proxy_pem_len = 0;
394 EVP_PKEY *proxy_key;
395 char err[160];
396 int rc;
397
398 /* A later authmore may be the X.509 delegation request (kXGS_pxyreq) rather
399 * than the cert round — dispatch on the buffer's step. */
400 if (gsi_msg_step(sbody, slen) == (uint32_t) kXGS_pxyreq) {
401 return gsi_sigpxy(c, sbody, slen, payload, plen, st);
402 }
403
404 gsi_resolve_proxy(c, proxy, sizeof(proxy), st);
405
406 proxy_key = load_proxy_key(proxy);
407 proxy_pem = load_proxy_pem(proxy, &proxy_pem_len);
408 if (proxy_key == NULL || proxy_pem == NULL) {
409 free(proxy_pem);
410 EVP_PKEY_free(proxy_key);
411 brix_status_set(st, XRDC_EAUTH, 0, "gsi: cannot load proxy credential");
412 return -1;
413 }
414
415 /* Retain the agreed session cipher on the connection: if the server follows
416 * with a kXGS_pxyreq (delegation), gsi_sigpxy reuses it to en/decrypt. */
417 err[0] = '\0';
418 rc = brix_gsi_build_cert_response_ex(sbody, slen, proxy_pem, proxy_pem_len,
419 proxy_key, payload, plen,
420 c ? c->gsi_deleg_key : NULL,
421 c ? &c->gsi_deleg_keylen : NULL,
422 c ? c->gsi_deleg_cipher : NULL,
423 c ? sizeof(c->gsi_deleg_cipher) : 0,
424 c ? &c->gsi_deleg_use_iv : NULL,
425 err, sizeof(err));
426 free(proxy_pem);
427 EVP_PKEY_free(proxy_key);
428 if (rc != 0) {
429 brix_status_set(st, XRDC_EAUTH, 0,
430 err[0] != '\0' ? err : "gsi: round-2 failed");
431 return -1;
432 }
433 if (c != NULL) {
434 c->gsi_deleg_ready = 1;
435 }
436 return 0;
437}
