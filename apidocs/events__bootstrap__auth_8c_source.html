22/* auth-frame builder */
23/*
24 * Builds a kXR_auth frame carrying a ztn (bearer token) credential.
25 *
26 * Wire format expected by the nginx-xrootd server (gsi/auth.c):
27 * Header bytes 16-19 (= cur_body[12:15]): "ztn\0" ← credential type
28 * Header bytes 20-23 (dlen): 4 + token_len
29 * Payload bytes 0-3: "ztn\0" ← payload prefix
30 * Payload bytes 4..: JWT token
31 *
32 * Returns the total frame length (24-byte header + 4 + token_len).
33 * buf must be at least 24 + 4 + token_len bytes.
34 */
35static size_t
36proxy_build_auth_ztn(u_char *buf, const char *token, size_t token_len)
37{
38 uint16_t rid = htons(kXR_auth);
39 uint32_t dlen_be = htonl((uint32_t)(4 + token_len));
40
41 ngx_memzero(buf, 24);
42 ngx_memcpy(buf + 2, &rid, 2);
43 /* credtype "ztn\0" at header body offset 12 (= buf offset 16) */
44 buf[16] = 'z'; buf[17] = 't'; buf[18] = 'n'; buf[19] = '\0';
45 ngx_memcpy(buf + 20, &dlen_be, 4);
46 /* payload: "ztn\0" prefix followed by JWT token */
47 buf[24] = 'z'; buf[25] = 't'; buf[26] = 'n'; buf[27] = '\0';
48 ngx_memcpy(buf + 28, token, token_len);
49
50 return 28 + token_len;
51}

53/* upstream auth resolution * WHAT: Resolve the effective upstream auth policy and (for SSS) the key to use
54 * for THIS upstream: a per-upstream `auth`/`sss:<keyname>` override wins,
55 * else the global brix_proxy_auth + first configured key.
56 * WHY: Both the kXR_authmore branch and the kXR_ok login-sec-hint branch need
57 * the same decision; factoring it keeps them in lock-step.
58 * HOW: Writes *eff_auth (BRIX_PROXY_AUTH_*) and *eff_key (NULL unless an SSS
59 * key was selected). */
60static void
61proxy_resolve_upstream_auth(brix_proxy_ctx_t *proxy, ngx_uint_t *eff_auth,
62 const brix_sss_key_t **eff_key)
63{
64 ngx_stream_brix_srv_conf_t *conf = proxy->conf;
65
66 *eff_auth = conf ? conf->proxy.auth : BRIX_PROXY_AUTH_ANONYMOUS;
67 *eff_key = NULL;
68
69 if (proxy->upstream_idx >= 0
70 && conf != NULL
71 && conf->proxy.upstreams != NULL
72 && proxy->upstream_idx < (int) conf->proxy.upstreams->nelts)
73 {
74 brix_proxy_upstream_t *u =
75 (brix_proxy_upstream_t *) conf->proxy.upstreams->elts
76 + proxy->upstream_idx;
77
78 if (u->auth >= 0) {
79 *eff_auth = (ngx_uint_t) u->auth;
80 }
81 if (*eff_auth == BRIX_PROXY_AUTH_SSS
82 && conf->sss_keys != NULL
83 && conf->sss_keys->nelts > 0)
84 {
85 brix_sss_key_t *keys = conf->sss_keys->elts;
86 ngx_uint_t ki;
87 if (u->sss_keyname[0] != '\0') {
88 for (ki = 0; ki < conf->sss_keys->nelts; ki++) {
89 if (ngx_strcmp(keys[ki].name, u->sss_keyname) == 0) {
90 *eff_key = &keys[ki];
91 break;
92 }
93 }
94 }
95 if (*eff_key == NULL) {
96 *eff_key = keys; /* fall back to first key */
97 }
98 }
99 }
100}

102/* auth-send failure/plumbing helpers */
103/*
104 * WHAT: Per-arm abort texts + metric policy for one auth-send attempt.
105 * WHY: The auth-send arms differ ONLY in message wording and in whether the
106 * per-upstream error counter is bumped alongside the global one; a
107 * descriptor keeps the shared send machinery literally identical.
108 * HOW: Each arm owns a function-local `static const` instance and threads it
109 * through proxy_bs_send_ztn / proxy_bs_queue_auth_frame.
110 */
111typedef struct {
112 const char *oom_msg; /* abort text: frame allocation failed */
113 const char *send_msg; /* abort text: flushing the frame failed */
114 const char *arm_msg; /* abort text: arming the write event failed */
115 unsigned up_inc; /* 1 = also bump the per-upstream error counter */
116} proxy_bs_auth_errs_t;
117
118/*
119 * WHAT: Record an upstream-auth failure and abort the proxy connection.
120 * WHY: Every failure exit of the bootstrap auth arms increments the global
121 * upstream_auth_errors metric; only some arms additionally bump the
122 * per-upstream counter (historical behavior, frozen).
123 * HOW: Bumps metrics per `up_inc`, then brix_proxy_abort(msg).
124 */
125void
126proxy_bs_auth_error(brix_proxy_ctx_t *proxy, unsigned up_inc, const char *msg)
127{
128 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_auth_errors);
129 if (up_inc) {
130 BRIX_PROXY_UP_INC(proxy, upstream_auth_errors);
131 }
132 brix_proxy_abort(proxy, msg);
133}

135/*
136 * WHAT: Reset the upstream response accumulator (header pos, dlen, body).
137 * WHY: Done after every consumed bootstrap response — both before sending
138 * the next request frame and in the shared handle_bootstrap tail — so
139 * the accumulator is ready for the next upstream reply.
140 * HOW: Frees resp_body if allocated and zeroes the three cursor fields.
141 */
142void
143proxy_bs_reset_resp(brix_proxy_ctx_t *proxy)
144{
145 if (proxy->resp_body != NULL) {
146 ngx_free(proxy->resp_body);
147 proxy->resp_body = NULL;
148 }
149 proxy->rhdr_pos = 0;
150 proxy->resp_dlen = 0;
151 proxy->resp_body_pos = 0;
152}

154/*
155 * WHAT: Install a built kXR_auth frame as the write buffer, advance the
156 * bootstrap to BS_AUTH, and flush it toward the upstream.
157 * WHY: All auth-send arms (bearer forward, SSS, token-file, login-sec hint)
158 * share this exact tail: reset accumulator -> set wbuf -> flush -> arm
159 * the write event on partial write. Factoring it makes the arms differ
160 * only in credential sourcing.
161 * HOW: flush() may only partially write under backpressure; if so, arm the
162 * write event so the rest goes out later. Either way the caller returns
163 * and waits for the BS_AUTH reply on the next read event. On any
164 * failure the proxy is aborted (per errs) and NGX_ERROR is returned.
165 */
166static ngx_int_t
167proxy_bs_queue_auth_frame(brix_proxy_ctx_t *proxy, u_char *frame,
168 size_t frame_len, const proxy_bs_auth_errs_t *errs)
169{
170 /* Clear the just-consumed response so the accumulator is ready for the
171 * kXR_auth reply, then install the auth frame as the write buffer and
172 * advance to BS_AUTH before flushing. */
173 proxy_bs_reset_resp(proxy);
174
175 proxy->wbuf = frame;
176 proxy->wbuf_len = frame_len;
177 proxy->wbuf_pos = 0;
178 proxy->bs_phase = XRD_PX_BS_AUTH;
179
180 if (brix_proxy_flush(proxy) == NGX_ERROR) {
181 proxy_bs_auth_error(proxy, errs->up_inc, errs->send_msg);
182 return NGX_ERROR;
183 }
184 if (proxy->wbuf_pos < proxy->wbuf_len
185 && ngx_handle_write_event(proxy->conn->write, 0) != NGX_OK)
186 {
187 proxy_bs_auth_error(proxy, errs->up_inc, errs->arm_msg);
188 return NGX_ERROR;
189 }
190 return NGX_OK;
191}

193/*
194 * WHAT: Build a ztn (bearer token) kXR_auth frame for `token` and queue it.
195 * WHY: Three arms send a ztn credential (client bearer forward, token-file
196 * fallback, login-sec hint); only the token source and abort texts
197 * differ, so the alloc/build/queue sequence lives once.
198 * HOW: Allocates the frame from the upstream connection pool, encodes it
199 * via proxy_build_auth_ztn, hands off to proxy_bs_queue_auth_frame.
200 */
201static ngx_int_t
202proxy_bs_send_ztn(brix_proxy_ctx_t *proxy, const char *token,
203 size_t token_len, const proxy_bs_auth_errs_t *errs)
204{
205 size_t frame_len = 28 + token_len; /* hdr24 + "ztn\0" + jwt */
206 u_char *frame = ngx_palloc(proxy->conn->pool, frame_len);
207
208 if (frame == NULL) {
209 proxy_bs_auth_error(proxy, errs->up_inc, errs->oom_msg);
210 return NGX_ERROR;
211 }
212 proxy_build_auth_ztn(frame, token, token_len);
213 return proxy_bs_queue_auth_frame(proxy, frame, frame_len, errs);
214}

216/* send an SSS kXR_auth credential to the upstream * WHAT: Build an SSS credential from `key`, wrap it in a kXR_auth request, and
217 * arm the write side; advance bs_phase to BS_AUTH.
218 * WHY: Used by BOTH the kXR_authmore path and the kXR_ok login-sec-hint path
219 * (our own server advertises SSS via the latter), so the logic lives once.
220 * HOW: Returns NGX_OK once the frame is queued (caller must return), or
221 * NGX_ERROR after aborting the proxy on any failure. */
222static ngx_int_t
223proxy_send_sss_auth(brix_proxy_ctx_t *proxy, const brix_sss_key_t *key)
224{
225 static const proxy_bs_auth_errs_t sss_errs = {
226 "proxy: OOM for SSS frame",
227 "proxy: SSS frame send failed",
228 "proxy: write arm for SSS failed",
229 1
230 };
231 u_char cred[512];
232 size_t cred_len;
233 u_char *frame;
234 size_t frame_len;
235 uint32_t dlen_be;
236
237 if (brix_sss_build_proxy_credential(key, key->user,
238 cred, sizeof(cred), &cred_len) != NGX_OK)
239 {
240 proxy_bs_auth_error(proxy, 1, "proxy: SSS credential build failed");
241 return NGX_ERROR;
242 }
243
244 frame_len = XRD_REQUEST_HDR_LEN + cred_len;
245 frame = ngx_palloc(proxy->conn->pool, frame_len);
246 if (frame == NULL) {
247 proxy_bs_auth_error(proxy, 1, sss_errs.oom_msg);
248 return NGX_ERROR;
249 }
250
251 /* kXR_auth request header: reqid at [2:3], credtype "sss\0" at [16:19]
252 * (the server routes on this field — leaving it zero yields "unknown
253 * credtype"), dlen at [20:23], SSS credential body after the 24-byte hdr. */
254 ngx_memzero(frame, XRD_REQUEST_HDR_LEN);
255 frame[2] = (kXR_auth >> 8) & 0xFF;
256 frame[3] = kXR_auth & 0xFF;
257 frame[16] = 's'; frame[17] = 's'; frame[18] = 's'; frame[19] = '\0';
258 dlen_be = htonl((uint32_t) cred_len);
259 ngx_memcpy(frame + 20, &dlen_be, 4);
260 ngx_memcpy(frame + XRD_REQUEST_HDR_LEN, cred, cred_len);
261
262 return proxy_bs_queue_auth_frame(proxy, frame, frame_len, &sss_errs);
263}

266/*
267 * WHAT: FORWARD mode, preferred path — re-present the connecting client's
268 * WLCG bearer token verbatim to the upstream as a ztn credential.
269 * WHY: The client already authenticated to us with this token; forwarding
270 * it preserves the client identity end-to-end.
271 * HOW: Sources the token from client_ctx (caller verified it is non-empty)
272 * and delegates the frame build/send to proxy_bs_send_ztn.
273 */
274static void
275proxy_bs_auth_forward_bearer(brix_proxy_ctx_t *proxy)
276{
277 static const proxy_bs_auth_errs_t errs = {
278 "proxy: OOM building auth frame",
279 "proxy: auth frame send failed",
280 "proxy: write arm for auth failed",
281 1
282 };
283 const char *token = proxy->client_ctx->bearer_token;
284
285 (void) proxy_bs_send_ztn(proxy, token, ngx_strlen(token), &errs);
286}

324/*
325 * WHAT: Answer an upstream kXR_authmore by the configured auth policy —
326 * FORWARD client bearer, SSS shared secret, FORWARD token-file
327 * fallback, or abort when no mechanism applies.
328 * WHY: This is the per-mechanism dispatch of the bootstrap's do-auth step;
329 * each mechanism lives in its own helper so the selection order (the
330 * frozen behavior) reads as a flat ladder.
331 * HOW: Resolves the effective policy (per-upstream override wins), tries
332 * the arms in the historical order, always returns 1 (a frame was
333 * queued or the proxy was aborted; the caller must return).
334 */
335int
336proxy_bs_do_auth(brix_proxy_ctx_t *proxy)
337{
338 ngx_stream_brix_srv_conf_t *conf = proxy->conf;
339 ngx_uint_t eff_auth;
340 const brix_sss_key_t *eff_key;
341
342 proxy_resolve_upstream_auth(proxy, &eff_auth, &eff_key);
343
344 if (eff_auth == BRIX_PROXY_AUTH_FORWARD
345 && proxy->client_ctx != NULL
346 && proxy->client_ctx->bearer_token[0] != '\0')
347 {
348 proxy_bs_auth_forward_bearer(proxy);
349 return 1;
350 }
351
352 if (eff_auth == BRIX_PROXY_AUTH_SSS && eff_key != NULL) {
353 (void) proxy_send_sss_auth(proxy, eff_key);
354 return 1; /* frame queued, or proxy aborted on failure */
355 }
356
357 if (eff_auth == BRIX_PROXY_AUTH_FORWARD
358 && conf != NULL
359 && conf->upstream_token_file.len > 0)
360 {
361 proxy_bs_auth_token_file(proxy);
362 return 1;
363 }
364
365 proxy_bs_auth_error(proxy, 1,
366 "upstream requires authentication "
367 "(set brix_proxy_auth forward or sss)");
368 return 1;
369}

371/* login-sec hint arms (kXR_ok path) */
372/*
373 * WHAT: Proactive ztn send when the login response advertised "P=ztn" without
374 * a kXR_authmore. Source the token from the client bearer (FORWARD)
375 * first, else from the upstream token file.
376 * WHY: Token-only servers embed the challenge in the login reply; the client
377 * must send kXR_auth unprompted or the forwarded open is rejected.
378 * HOW: Returns 1 when a frame was queued (or the send aborted the proxy);
379 * 0 when no token could be sourced and the caller should fall through.
380 * lftok is function-static (64 KiB) — same rationale as the authmore
381 * token-file arm.
382 */
383static int
384proxy_bs_login_sec_ztn(brix_proxy_ctx_t *proxy)
385{
386 static const proxy_bs_auth_errs_t errs = {
387 "proxy: OOM for login-sec token frame",
388 "proxy: login-sec token frame send failed",
389 "proxy: write arm for login-sec failed",
390 0
391 };
392 static u_char lftok[PROXY_UPSTREAM_BEARER_MAX];
393 ngx_stream_brix_srv_conf_t *lconf = proxy->conf;
394 const char *ltoken = NULL;
395 size_t ltoken_len = 0;
396
397 /* Prefer client's bearer token (FORWARD mode) */
398 if (proxy->client_ctx != NULL
399 && proxy->client_ctx->bearer_token[0] != '\0')
400 {
401 ltoken = proxy->client_ctx->bearer_token;
402 ltoken_len = ngx_strlen(ltoken);
403 }
404 /* Fallback: read from upstream_token_file */
405 else if (lconf != NULL && lconf->upstream_token_file.len > 0) {
406 size_t lflen = 0;
407 if (brix_token_read_file(&lconf->upstream_token_file,
408 lftok, sizeof(lftok), &lflen,
409 proxy->client_conn->log,
410 "proxy login-sec ztn") == NGX_OK
411 && lflen > 0)
412 {
413 ltoken = (const char *) lftok;
414 ltoken_len = lflen;
415 }
416 }
417
418 if (ltoken == NULL || ltoken_len == 0) {
419 return 0;
420 }
421 (void) proxy_bs_send_ztn(proxy, ltoken, ltoken_len, &errs);
422 return 1;
423}

425/*
426 * WHAT: Scan the successful login reply for an embedded security hint and,
427 * if one matches our policy, proactively send the kXR_auth credential.
428 * WHY: Token-only servers embed a security challenge in the login response:
429 * [sessid:16][&P=ztn,v:10000] — and the nginx-xrootd server (and stock
430 * xrootd) advertise SSS the same way ("&P=sss,...") — NOT via
431 * kXR_authmore. Without this, SSS/ztn upstream auth silently never
432 * happens and the forwarded open is rejected NotAuthorized.
433 * HOW: The hint string follows the fixed 16-byte session id in the login
434 * body, so step past it before scanning. resp_body was NUL-padded by
435 * the read handler (alloc dlen+1), making strstr() safe here. Returns
436 * 1 when an auth frame was issued (caller returns and waits for the
437 * BS_AUTH reply), 0 when the login phase is complete.
438 */
439int
440proxy_bs_login_sec_hint(brix_proxy_ctx_t *proxy)
441{
442 const char *parms;
443
444 if (proxy->resp_dlen <= BRIX_SESSION_ID_LEN || proxy->resp_body == NULL) {
445 return 0;
446 }
447 parms = (const char *)(proxy->resp_body + BRIX_SESSION_ID_LEN);
448
449 if (strstr(parms, "P=sss") != NULL) {
450 ngx_uint_t sss_auth;
451 const brix_sss_key_t *sss_key;
452
453 proxy_resolve_upstream_auth(proxy, &sss_auth, &sss_key);
454 if (sss_auth == BRIX_PROXY_AUTH_SSS && sss_key != NULL) {
455 (void) proxy_send_sss_auth(proxy, sss_key);
456 return 1; /* frame queued, or proxy aborted on failure */
457 }
458 }
459
460 if (strstr(parms, "P=ztn") != NULL) {
461 return proxy_bs_login_sec_ztn(proxy);
462 }
463
464 return 0;
465}
