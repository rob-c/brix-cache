3 * minting (phase-2 T9). See cred_mint.h for the full rationale/trust-model and

6 * WHAT: The certificate-construction half of brix_cred_mint(): sanitize the

27/* ---- mint_cn_char_ok -----------------------------------------------------
28 * WHAT: 1 iff byte `c` is safe to copy verbatim into a CN RDN value: the
29 * ASCII alphanumerics plus the punctuation set [@._-/ =] (space and '='
30 * appear in DN-derived principals; '/' is a DN separator X509_NAME
31 * tolerates inside a single RDN value). Everything else — including
32 * control bytes and NUL — is rejected so mint_sanitize_cn maps it to '_'.
33 * WHY: Isolating the charset predicate keeps mint_sanitize_cn's loop trivial
34 * and makes the exact allow-list a single, testable, security-load-
35 * bearing point of truth (the set MUST stay identical byte-for-byte).
36 * HOW: Pure range/equality checks over the unsigned byte value. */
37static int
38mint_cn_char_ok(unsigned char c)
39{
40 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
41 || (c >= '0' && c <= '9') || c == '@' || c == '.'
42 || c == '_' || c == '-' || c == '/' || c == ' ' || c == '=';
43}

45/* ---- mint_sanitize_cn ----------------------------------------------------
46 * WHAT: Copy `principal` into `out` (cap bytes) as a CN-safe string: any byte
47 * rejected by mint_cn_char_ok() is replaced with '_'. Truncates rather
48 * than overflows.
49 * WHY: The principal may be a DN, an S3 access key, or a JWT subject; RFC
50 * 5280 does not forbid most of these characters in a CN, but control
51 * characters or an embedded NUL must never reach X509_NAME_add_entry.
52 * HOW: Single bounded pass; deterministic (same principal -> same CN). */
53static void
54mint_sanitize_cn(const char *principal, char *out, size_t cap)
55{
56 size_t i;
57 size_t n = strlen(principal);
58
59 if (n >= cap) {
60 n = cap - 1;
61 }
62 for (i = 0; i < n; i++) {
63 unsigned char c = (unsigned char) principal[i];
64 out[i] = mint_cn_char_ok(c) ? (char) c : '_';
65 }
66 out[n] = '\0';
67}

69/* ---- mint_make_subject ---------------------------------------------------
70 * WHAT: Build the minted-proxy subject X509_NAME "/O=brix-minted/CN=<cn>";
71 * NULL on any allocation/entry failure (freeing a partial name first).
72 * WHY: Owns the X509_NAME lifecycle in one place so mint_populate_cert()'s
73 * early-return ladder never has to special-case the subject build.
74 * HOW: X509_NAME_new + two X509_NAME_add_entry_by_txt calls; `cn` is already
75 * CN-sanitized by the caller (mint_sanitize_cn). */
76static X509_NAME *
77mint_make_subject(const char *cn, ngx_log_t *log)
78{
79 X509_NAME *subj = X509_NAME_new();
80
81 if (subj == NULL
82 || !X509_NAME_add_entry_by_txt(subj, "O", MBSTRING_ASC,
83 (const unsigned char *) "brix-minted", -1, -1, 0)
84 || !X509_NAME_add_entry_by_txt(subj, "CN", MBSTRING_ASC,
85 (const unsigned char *) cn, -1, -1, 0)) {
86 MINT_LOG(NGX_LOG_ERR, log, 0,
87 "brix: cred_mint: cannot build proxy subject");
88 X509_NAME_free(subj);
89 return NULL;
90 }
91 return subj;
92}

94/* ---- mint_serial_from_rand -----------------------------------------------
95 * WHAT: Pack 8 random bytes into a uint64 serial with the top bit cleared so
96 * the ASN1 INTEGER encoding is unambiguously positive.
97 * WHY: Keeps the bit-shuffle out of mint_populate_cert()'s boolean ladder
98 * and names the "coerce positive" invariant.
99 * HOW: Big-endian assembly of rnd[0..7]; rnd[0]'s top bit is masked off. */
100static uint64_t
101mint_serial_from_rand(const unsigned char rnd[8])
102{
103 return (((uint64_t) (rnd[0] & 0x7f)) << 56) | ((uint64_t) rnd[1] << 48)
104 | ((uint64_t) rnd[2] << 40) | ((uint64_t) rnd[3] << 32)
105 | ((uint64_t) rnd[4] << 24) | ((uint64_t) rnd[5] << 16)
106 | ((uint64_t) rnd[6] << 8) | (uint64_t) rnd[7];
107}

109/* ---- mint_cert_spec_t ----------------------------------------------------
110 * WHAT: The immutable inputs mint_populate_cert() needs to set and sign one
111 * minted leaf certificate: its subject name, the mint CA cert (issuer
112 * source) and key (signer), the leaf's own public key, the ASN1 serial,
113 * and the validity window length in seconds.
114 * WHY: Bundling these related inputs keeps mint_populate_cert() at one job
115 * and ≤5 parameters without introducing any globals or changing what it
116 * does — purely a call-shape change.
117 * HOW: Plain aggregate, filled on the stack by mint_build_cert() and passed
118 * by const pointer; the helper mutates only the caller-owned `cert`. */
119typedef struct {
120 X509_NAME *subj;
121 X509 *ca_cert;
122 EVP_PKEY *key;
123 EVP_PKEY *ca_key;
124 uint64_t serial;
125 int ttl_secs;
126} mint_cert_spec_t;

128/* ---- mint_populate_cert --------------------------------------------------
129 * WHAT: Populate and sign an already-allocated `cert` from `spec`: version 3,
130 * spec->serial, spec->subj, issuer = spec->ca_cert's subject, pubkey =
131 * spec->key, notBefore=now, notAfter=now+spec->ttl_secs,
132 * X509_sign(spec->ca_key, SHA-256). Returns NGX_OK/NGX_ERROR; frees
133 * nothing (caller owns cert).
134 * WHY: Splits the long field-setting boolean chain out of mint_build_cert()
135 * so each function stays under the CCN cap while preserving the exact
136 * single-expression short-circuit ordering (security-load-bearing).
137 * HOW: One boolean AND-chain; any failure logs and returns NGX_ERROR. */
138static int
139mint_populate_cert(X509 *cert, const mint_cert_spec_t *spec, ngx_log_t *log)
140{
141 if (X509_set_version(cert, 2L) != 1
142 || ASN1_INTEGER_set_uint64(X509_get_serialNumber(cert),
143 spec->serial) != 1
144 || X509_set_subject_name(cert, spec->subj) != 1
145 || X509_set_issuer_name(cert,
146 X509_get_subject_name(spec->ca_cert)) != 1
147 || X509_set_pubkey(cert, spec->key) != 1
148 || X509_gmtime_adj(X509_getm_notBefore(cert), 0) == NULL
149 || X509_gmtime_adj(X509_getm_notAfter(cert),
150 (long) spec->ttl_secs) == NULL
151 || X509_sign(cert, spec->ca_key, EVP_sha256()) <= 0) {
152 MINT_LOG(NGX_LOG_ERR, log, 0,
153 "brix: cred_mint: cannot build/sign minted proxy");
154 return NGX_ERROR;
155 }
156 return NGX_OK;
157}

159/* ---- mint_build_cert -----------------------------------------------------
160 * WHAT: Generate a fresh EC P-256 keypair and a signed X509 whose subject is
161 * "/O=brix-minted/CN=<sanitized principal>", issuer = the mint CA's
162 * subject, notBefore=now, notAfter=now+ttl_secs, signed by ca_key with
163 * SHA-256.
164 * WHY: Isolates the OpenSSL object lifecycle for the mint step so
165 * brix_cred_mint() stays a thin orchestrator; every early return here
166 * frees exactly what this call allocated.
167 * HOW: EVP_PKEY_Q_keygen(EC, P-256) -> X509_new -> mint_make_subject ->
168 * mint_populate_cert (which sets/signs every field). Serial is 8 random
169 * bytes coerced positive so it round-trips through ASN1_INTEGER. On
170 * success fills out->cert/out->key (leaves out->ca_cert untouched). */
171int
172mint_build_cert(const mint_ca_t *ca, const char *principal, int ttl_secs,
173 mint_material_t *out, ngx_log_t *log)
174{
175 EVP_PKEY *key = NULL;
176 X509 *cert = NULL;
177 X509_NAME *subj = NULL;
178 unsigned char rnd[8];
179 char cn[256];
180 int ok;
181
182 out->cert = NULL;
183 out->key = NULL;
184
185 key = EVP_PKEY_Q_keygen(NULL, NULL, "EC", "P-256");
186 if (key == NULL) {
187 MINT_LOG(NGX_LOG_ERR, log, 0,
188 "brix: cred_mint: EC P-256 key generation failed");
189 return NGX_ERROR;
190 }
191
192 if (RAND_bytes(rnd, sizeof(rnd)) != 1) {
193 MINT_LOG(NGX_LOG_ERR, log, 0, "brix: cred_mint: RNG failed");
194 EVP_PKEY_free(key);
195 return NGX_ERROR;
196 }
197
198 cert = X509_new();
199 if (cert == NULL) {
200 EVP_PKEY_free(key);
201 return NGX_ERROR;
202 }
203
204 mint_sanitize_cn(principal, cn, sizeof(cn));
205 subj = mint_make_subject(cn, log);
206 if (subj == NULL) {
207 X509_free(cert);
208 EVP_PKEY_free(key);
209 return NGX_ERROR;
210 }
211
212 mint_cert_spec_t spec = {
213 .subj = subj,
214 .ca_cert = ca->cert,
215 .key = key,
216 .ca_key = ca->key,
217 .serial = mint_serial_from_rand(rnd),
218 .ttl_secs = ttl_secs,
219 };
220 ok = mint_populate_cert(cert, &spec, log) == NGX_OK;
221 X509_NAME_free(subj);
222 if (!ok) {
223 X509_free(cert);
224 EVP_PKEY_free(key);
225 return NGX_ERROR;
226 }
227
228 out->cert = cert;
229 out->key = key;
230 return NGX_OK;
231}
