55/*
56 * webdav_move_probe — confined stat of `path` (follow, matching the prior
57 * lstat nofollow=0) via the VFS probe, projected to the struct stat fields MOVE
58 * reads (ino/dev for the self-move guard, mode for the dir branch). Non-metered.
59 * NGX_OK / NGX_DECLINED (errno kept).
60 */
61static ngx_int_t
62webdav_move_probe(ngx_http_request_t *r, const char *path, struct stat *sb)
63{
64 ngx_http_brix_webdav_loc_conf_t *conf =
65 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
66 ngx_http_brix_webdav_req_ctx_t *rx =
67 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
68 brix_vfs_ctx_t vctx;
69 brix_vfs_stat_t vst;
70 int is_tls = 0;
71
72#if (NGX_HTTP_SSL)
73 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
74#endif
75
76 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
77 conf->common.root_canon, conf->cache_root_canon, conf->common.allow_write,
78 is_tls, (rx != NULL) ? rx->identity : NULL, path);
79 /* Bind the export's per-user backend credential policy so a remote-backed
80 * export's probe (and the deny gate it enforces) sees the REQUESTING
81 * USER's credential, not the shared service credential — this probe is
82 * also the pre-rename existence/self-move check, so deny must be
83 * enforced here before MOVE ever reaches the rename. */
84 brix_vfs_ctx_bind_backend_cred(&vctx,
85 &conf->common.storage_credential_dir,
86 conf->common.storage_credential_fallback);
87 webdav_vfs_bind_deleg(r, conf, &vctx);
88 if (brix_vfs_probe(&vctx, 0 /* follow */, &vst) != NGX_OK) {
89 return NGX_DECLINED;
90 }
91 ngx_memzero(sb, sizeof(*sb));
92 sb->st_mode = (mode_t) vst.mode;
93 sb->st_ino = vst.ino;
94 sb->st_dev = vst.dev;
95 return NGX_OK;
96}

98/*
99 * webdav_move_execute_cred — perform the actual rename through the ctx-bound
100 * brix_vfs_rename (rather than the pool-less brix_vfs_rename_path this
101 * replaced) and map the namespace result to an HTTP status:
102 * OK -> 204 (replaced) / 201 (created)
103 * EXISTS -> 412 (Overwrite:F race)
104 * CONFLICT / NOT_FOUND -> 409 (missing parent / non-empty dst dir)
105 * FORBIDDEN -> 403 (per-user backend credential gate denied)
106 * anything else -> 500.
107 * The raw errno is reported via *sys_errno for the caller's log.
108 *
109 * WHY: Building the ctx here and binding brix_vfs_ctx_bind_backend_cred so a
110 * remote-backed export sees the REQUESTING USER's credential (not the
111 * shared service credential) for the rename — matching root:// mv.c.
112 * Both MOVE callers (the synchronous single-file path and the
113 * thread-pool collection-MOVE offload) share this one function.
114 *
115 * HOW: Called from both the event loop (webdav_handle_move) and a
116 * thread-pool worker (webdav_move_collection_thread); the ctx is built
117 * fresh here each call using `r`'s pool/connection/identity, which stay
118 * valid for either caller's lifetime (the collection-offload caller
119 * took r->main->count++ before posting the task). req->overwrite is
120 * threaded into brix_vfs_rename as overwrite_dirs so Overwrite:T can
121 * replace an existing DIRECTORY destination (RFC 4918 §9.9.4 — the
122 * target tree is removed before the rename); Overwrite:F was already
123 * rejected as 412 before either path reaches this function.
124 *
125 * The 7 scalar arguments this once took are now carried on a file-local
126 * webdav_move_req_t (see above) — no behavior change, only the calling
127 * convention; *sys_errno reports the raw errno for the caller's log.
128 */
129static ngx_int_t
130webdav_move_execute_cred(const webdav_move_req_t *req, int *sys_errno)
131{
132 ngx_http_request_t *r = req->r;
133 ngx_http_brix_webdav_loc_conf_t *conf = req->conf;
134 ngx_http_brix_webdav_req_ctx_t *wctx =
135 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
136 brix_vfs_ctx_t vctx;
137 brix_path_result_t dst_result;
138
139 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
140 conf->common.root_canon, conf->cache_root_canon,
141 conf->common.allow_write, 0 /* is_tls: irrelevant off the wire */,
142 (wctx != NULL) ? wctx->identity : NULL, req->src_path);
143 brix_vfs_ctx_bind_backend_cred(&vctx,
144 &conf->common.storage_credential_dir,
145 conf->common.storage_credential_fallback);
146 webdav_vfs_bind_deleg(r, conf, &vctx);
147 vctx.sd = req->sd;
148
149 ngx_memzero(&dst_result, sizeof(dst_result));
150 dst_result.is_confined = 1;
151 dst_result.resolved.data = (u_char *) req->dst_path;
152 dst_result.resolved.len = ngx_strlen(req->dst_path);
153
154 if (brix_vfs_rename(&vctx, &dst_result,
155 req->overwrite ? 1 : 0) == NGX_OK) {
156 if (sys_errno != NULL) {
157 *sys_errno = 0;
158 }
159 return req->dst_existed ? NGX_HTTP_NO_CONTENT : NGX_HTTP_CREATED;
160 }
161
162 if (sys_errno != NULL) {
163 *sys_errno = errno;
164 }
165
166 if (errno == EEXIST) { /* NS_EXISTS */
167 return NGX_HTTP_PRECONDITION_FAILED;
168 }
169
170 if (errno == ENOTDIR || errno == ENOENT) { /* NS_CONFLICT / NS_NOT_FOUND */
171 return NGX_HTTP_CONFLICT;
172 }
173
174 if (errno == EACCES || errno == EPERM) { /* deny-mode credential gate */
175 return NGX_HTTP_FORBIDDEN;
176 }
177
178 return NGX_HTTP_INTERNAL_SERVER_ERROR;
179}

231/*
232 * Try to offload a collection MOVE to the thread pool. Returns NGX_DONE when
233 * queued (request held via r->main->count++), NGX_DECLINED when no pool is
234 * available (caller runs the rename synchronously), NGX_ERROR on failure.
235 */
236static ngx_int_t
237webdav_move_collection_post_task(const webdav_move_req_t *req)
238{
239 ngx_http_request_t *r = req->r;
240 ngx_http_brix_webdav_loc_conf_t *conf = req->conf;
241 ngx_thread_task_t *task;
242 webdav_move_collection_task_t *t;
243 ngx_thread_pool_t *pool;
244
245 /*
246 * Under impersonation the per-worker broker socket is a single fd used by
247 * the event-loop thread; a thread-pool task issuing confined ops would race
248 * it and corrupt the broker framing (and lacks the per-worker principal).
249 * Force the synchronous rename path (NGX_DECLINED). See copy.c for detail.
250 */
251 if (brix_imp_enabled()) {
252 return NGX_DECLINED;
253 }
254
255 /* postconfig only visits server-level loc_conf; resolve lazily for nested
256 * location blocks where conf->common.thread_pool may still be NULL. */
257 pool = conf->common.thread_pool;
258 if (pool == NULL) {
259 static ngx_str_t default_name = ngx_string("default");
260 ngx_str_t *pname = conf->common.thread_pool_name.len > 0
261 ? &conf->common.thread_pool_name : &default_name;
262 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
263 if (pool != NULL) {
264 conf->common.thread_pool = pool;
265 }
266 }
267 if (pool == NULL) {
268 return NGX_DECLINED;
269 }
270
271 task = ngx_thread_task_alloc(r->pool, sizeof(webdav_move_collection_task_t));
272 if (task == NULL) {
273 return NGX_ERROR;
274 }
275
276 t = task->ctx;
277 t->r = r;
278 t->log = r->connection->log;
279 t->dst_existed = req->dst_existed;
280 t->overwrite = req->overwrite;
281 t->http_status = NGX_HTTP_INTERNAL_SERVER_ERROR;
282 t->sys_errno = 0;
283 t->sd = req->sd;
284
285 ngx_cpystrn((u_char *) t->root_canon, (u_char *) conf->common.root_canon,
286 sizeof(t->root_canon));
287 ngx_cpystrn((u_char *) t->src_path, (u_char *) req->src_path,
288 sizeof(t->src_path));
289 ngx_cpystrn((u_char *) t->dst_path, (u_char *) req->dst_path,
290 sizeof(t->dst_path));
291
292 brix_task_bind(task, webdav_move_collection_thread, webdav_move_collection_done);
293 task->event.log = r->connection->log;
294
295 if (ngx_thread_task_post(pool, task) != NGX_OK) {
296 return NGX_ERROR;
297 }
298
299 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
300 "brix_webdav: offloaded collection MOVE to thread pool");
301
302 r->main->count++;
303 return NGX_DONE;
304}

306/*
307 * webdav_move_resolve_dest — stage 1 of MOVE destination handling: read the
308 * mandatory Destination header, strip its scheme://authority, URL-decode it,
309 * and resolve+confine it to `dst_path`.
310 *
311 * WHAT: Turns the raw Destination request header into a confined absolute
312 * filesystem path (dst_path, capacity dst_cap).
313 * WHY: Lifts the four sequential "extract → decode → resolve" steps (each with
314 * its own early-return) out of webdav_handle_move so the top-level MOVE
315 * reads as a linear sequence of stages, dropping its branch count.
316 * HOW: Returns NGX_OK on success; on any failure returns the exact HTTP status
317 * the original inline code returned (400 for a missing Destination header,
318 * otherwise the sub-helper's rc) so the wire behavior is unchanged.
319 */
320static ngx_int_t
321webdav_move_resolve_dest(ngx_http_request_t *r,
322 ngx_http_brix_webdav_loc_conf_t *conf, char *dst_path, size_t dst_cap)
323{
324 ngx_table_elt_t *dest_hdr;
325 char dest_decoded[WEBDAV_MAX_PATH];
326 const u_char *dest_path_start;
327 size_t dest_path_len;
328 ngx_int_t rc;
329
330 /* Require Destination header (RFC 4918 §9.9.4 — missing → 400) */
331 dest_hdr = webdav_tpc_find_header(r, "Destination",
332 sizeof("Destination") - 1);
333 if (dest_hdr == NULL) {
334 return NGX_HTTP_BAD_REQUEST;
335 }
336
337 /* Extract path component from Destination URL, stripping scheme://authority. */
338 dest_path_start = dest_hdr->value.data;
339 dest_path_len = dest_hdr->value.len;
340 rc = webdav_destination_extract_path(dest_path_start, dest_path_len,
341 &dest_path_start, &dest_path_len);
342 if (rc != NGX_OK) {
343 return rc;
344 }
345
346 /* URL-decode the destination path */
347 rc = webdav_urldecode(dest_path_start, dest_path_len,
348 dest_decoded, sizeof(dest_decoded));
349 if (rc != NGX_OK) {
350 return rc;
351 }
352
353 /* Resolve destination */
354 return webdav_resolve_destination_path(r->connection->log, "MOVE",
355 conf->common.root_canon, dest_decoded,
356 dst_path, dst_cap,
357 conf->common.cache_store_endpoint);
358}

360/*
361 * webdav_move_run_sync — stage: perform the rename on the event-loop thread
362 * (used for non-directory moves and directory moves when no thread pool is
363 * available) and map the namespace result to the final HTTP response.
364 *
365 * WHAT: Runs webdav_move_execute_cred synchronously and returns the terminal
366 * value webdav_handle_move should hand back to nginx.
367 * WHY: Isolates the success/forbidden/500-log/412-409 status ladder from the
368 * top-level handler, cutting its cyclomatic complexity without changing
369 * any status code or the 500 rename-failure log line.
370 * HOW: Sends 201/204 via webdav_send_no_body on success; on 500 logs the raw
371 * rename errno against the source path; returns 403/412/409 verbatim and
372 * falls back to 500 for anything else — byte-for-byte the prior inline
373 * block.
374 */
375static ngx_int_t
376webdav_move_run_sync(const webdav_move_req_t *req)
377{
378 ngx_http_request_t *r = req->r;
379 int sys_errno = 0;
380 ngx_int_t rc;
381
382 rc = webdav_move_execute_cred(req, &sys_errno);
383
384 if (rc == NGX_HTTP_CREATED || rc == NGX_HTTP_NO_CONTENT) {
385 return webdav_send_no_body(r, (ngx_uint_t) rc);
386 }
387
388 if (rc == NGX_HTTP_FORBIDDEN) {
389 return rc;
390 }
391
392 if (rc == NGX_HTTP_INTERNAL_SERVER_ERROR) {
393 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, sys_errno,
394 "brix_webdav MOVE: rename() failed for: \"%s\"",
395 req->src_path);
396 }
397
398 if (rc == NGX_HTTP_PRECONDITION_FAILED || rc == NGX_HTTP_CONFLICT) {
399 return rc;
400 }
401
402 return NGX_HTTP_INTERNAL_SERVER_ERROR;
403}

405/*
406 * webdav_move_dispatch — stage: route a fully-resolved MOVE to its executor.
407 *
408 * WHAT: For a directory source, tries the thread-pool offload; for everything
409 * else (and for a directory when no pool is available) runs the rename
410 * synchronously. Returns the terminal value for webdav_handle_move.
411 * WHY: Collapses the S_ISDIR offload branch + its NGX_DONE/NGX_ERROR/NGX_DECLINED
412 * handling and the synchronous fall-through into one helper, keeping the
413 * collection-recursion contract (child-lock checks already done upstream)
414 * intact while removing branches from the top-level handler.
415 * HOW: NGX_DONE from the offload ends the request here (already returned to
416 * nginx); NGX_ERROR maps to 500; NGX_DECLINED (no pool / impersonation)
417 * falls through to webdav_move_run_sync — identical to the prior inline
418 * control flow. `is_dir` is the source's S_ISDIR bit, computed by the
419 * caller from the same stat used for the self-move guard.
420 */
421static ngx_int_t
422webdav_move_dispatch(const webdav_move_req_t *req, int is_dir)
423{
424 ngx_int_t rc;
425
426 /* Directories are offloaded to a thread when one is available (NGX_DONE ends
427 * the request here); NGX_DECLINED (no pool) and all non-directory moves fall
428 * through to the synchronous rename below — rename(2) itself is fast, the
429 * offload mainly isolates the worker from rare slow-fs stalls. */
430 if (is_dir) {
431 rc = webdav_move_collection_post_task(req);
432 if (rc == NGX_DONE) {
433 return NGX_DONE;
434 }
435 if (rc == NGX_ERROR) {
436 return NGX_HTTP_INTERNAL_SERVER_ERROR;
437 }
438 /* NGX_DECLINED: no thread pool, fall through to synchronous rename. */
439 }
440
441 return webdav_move_run_sync(req);
442}

444/*
445 * webdav_handle_move — implement RFC 4918 §9.9 MOVE.
446 *
447 * Key protocol requirements enforced:
448 * - Destination header is mandatory (RFC 4918 §9.9.4).
449 * - Overwrite:F with an existing destination → 412 Precondition Failed.
450 * - Moving a resource onto itself → 403 Forbidden.
451 * - Non-empty destination directory → 409 Conflict (renameat ENOTEMPTY).
452 *
453 * Atomicity: rename(2) is atomic within the same filesystem. Both source
454 * and destination must be within root_canon, so cross-device moves are
455 * impossible (they would be caught by realpath confinement).
456 *
457 * Fd-cache eviction: both source and destination paths are evicted from the
458 * per-connection fd-cache after a successful rename to prevent stale fds.
459 */
460ngx_int_t
461webdav_handle_move(ngx_http_request_t *r)
462{
463 ngx_http_brix_webdav_loc_conf_t *conf;
464 webdav_move_req_t req;
465 char src_path[WEBDAV_MAX_PATH];
466 char dst_path[WEBDAV_MAX_PATH];
467 ngx_int_t rc;
468 struct stat src_sb;
469 struct stat dst_sb;
470
471 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
472
473 /* Resolve source */
474 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
475 src_path, sizeof(src_path));
476 if (rc != NGX_OK) {
477 return rc;
478 }
479
480 if (webdav_move_probe(r, src_path, &src_sb) != NGX_OK) {
481 return NGX_HTTP_NOT_FOUND;
482 }
483
484 /* MOVE on a collection recursively checks child locks (source subtree). */
485 rc = webdav_check_locks_tree(r, src_path);
486 if (rc != NGX_OK) {
487 return rc;
488 }
489
490 /* Read + strip + decode + confine the Destination header. */
491 rc = webdav_move_resolve_dest(r, conf, dst_path, sizeof(dst_path));
492 if (rc != NGX_OK) {
493 return rc;
494 }
495
496 ngx_memzero(&req, sizeof(req));
497 req.r = r;
498 req.conf = conf;
499 req.sd = (brix_sd_instance_t *)
500 brix_webdav_backend_instance(conf, r->connection->log);
501 req.src_path = src_path;
502 req.dst_path = dst_path;
503 req.overwrite = !brix_http_overwrite_forbidden(r);
504 req.dst_existed = (webdav_move_probe(r, dst_path, &dst_sb) == NGX_OK);
505
506 /* MOVE on a collection recursively checks child locks (dest subtree). */
507 rc = webdav_check_locks_tree(r, dst_path);
508 if (rc != NGX_OK) {
509 return rc;
510 }
511
512 /* RFC 4918 §9.9.4 — Overwrite:F and destination exists → 412 */
513 if (req.dst_existed && !req.overwrite) {
514 return NGX_HTTP_PRECONDITION_FAILED;
515 }
516
517 /* Prevent moving a resource onto itself */
518 if (req.dst_existed && src_sb.st_ino == dst_sb.st_ino
519 && src_sb.st_dev == dst_sb.st_dev)
520 {
521 return NGX_HTTP_FORBIDDEN;
522 }
523
524 return webdav_move_dispatch(&req, S_ISDIR(src_sb.st_mode));
525}
