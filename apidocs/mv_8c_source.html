11/*
12 * mv_req_t — per-request path state threaded through the mv helpers.
13 *
14 * WHAT: The two request paths (as extracted from the wire) and their
15 * resolved absolute forms, bundled so each helper takes one pointer
16 * instead of four buffers.
17 * WHY: Keeps every helper under the 5-parameter gate with explicit data
18 * flow (no globals, no hidden state).
19 * HOW: Filled left-to-right: mv_parse_paths sets src_buf/dst_buf, the
20 * resolve steps in brix_handle_mv fill src_resolved/dst_resolved.
21 */
22typedef struct {
23 char src_buf[BRIX_MAX_PATH + 1];
24 char dst_buf[BRIX_MAX_PATH + 1];
25 char src_resolved[PATH_MAX];
26 char dst_resolved[PATH_MAX];
27} mv_req_t;

29/*
30 * mv_fail — error access-log triplet for the MV op, helper-return form.
31 *
32 * WHAT: Emits the mandatory error triplet (brix_log_access + BRIX_OP_ERR +
33 * brix_send_error) for a failed MV, storing the send result in
34 * ctx->write_rc.
35 * WHY: BRIX_RETURN_ERR embeds a `return`, which from inside a helper would
36 * leave the caller unable to forward the nginx rc; this mirrors the
37 * brix_auth_gate_op convention (respond, stash rc in ctx->write_rc,
38 * return non-NGX_OK) so the orchestrator just `return ctx->write_rc`.
39 * HOW: Same three calls in the same order as BRIX_RETURN_ERR, verb "MV",
40 * detail "-"; always returns NGX_ERROR.
41 */
42static ngx_int_t
43mv_fail(brix_ctx_t *ctx, ngx_connection_t *c, const char *path,
44 int kxr, const char *msg)
45{
46 brix_log_access(ctx, c, "MV", path, "-", 0, kxr, msg, 0);
47 BRIX_OP_ERR(ctx, BRIX_OP_MV);
48 ctx->write_rc = brix_send_error(ctx, c, kxr, msg);
49 return NGX_ERROR;
50}

52/*
53 * mv_parse_paths — decode the kXR_mv payload into two request paths.
54 *
55 * WHAT: Validates the wire payload, splits it at the space separator and
56 * extracts source + destination into mv->src_buf / mv->dst_buf.
57 * WHY: The mv wire format (from XrdClFileSystem.cc) is:
58 * arg1len = source.length() (NOT including any terminator)
59 * dlen = src.length() + dst.length() + 1
60 * payload = src[arg1len] + ' ' + dst[...]
61 * The separator between source and destination is a single space
62 * (0x20). arg1len == 0: the reference do_Mv (XrdXrootdXeq.cc) splits
63 * the buffer on the FIRST space itself ("old new") rather than
64 * rejecting, so a well-formed space-separated buffer with arg1len=0
65 * is autosplit the same way.
66 * HOW: Early-return on each malformed shape with the historical (plain,
67 * untripleted) brix_send_error stored in ctx->write_rc; each half is
68 * parsed independently via brix_extract_path so embedded-NUL and
69 * traversal checks apply to both. Returns NGX_OK when both request
70 * paths are extracted; NGX_ERROR after responding otherwise.
71 */
72static ngx_int_t
73mv_parse_paths(brix_ctx_t *ctx, ngx_connection_t *c, mv_req_t *mv)
74{
75 xrdw_twopath_req_t req;
76 int16_t src_len;
77 size_t dst_len;
78
79 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
80 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgMissing,
81 "no paths given");
82 return NGX_ERROR;
83 }
84
85 xrdw_twopath_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
86 src_len = req.arg1len;
87 if (src_len < 0 || (uint32_t) src_len >= ctx->recv.cur_dlen) {
88 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
89 "invalid arg1len for mv");
90 return NGX_ERROR;
91 }
92
93 if (src_len == 0) {
94 u_char *sp = memchr(ctx->recv.payload, ' ', (size_t) ctx->recv.cur_dlen);
95 if (sp == NULL || sp == ctx->recv.payload
96 || (size_t)(sp - ctx->recv.payload) > 0x7fff) {
97 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
98 "invalid path specification");
99 return NGX_ERROR;
100 }
101 src_len = (int16_t)(sp - ctx->recv.payload);
102 }
103
104 /* Separator byte at src_len must be a space, with a non-empty dst after it. */
105 if ((uint32_t)(src_len + 1) >= ctx->recv.cur_dlen
106 || ctx->recv.payload[src_len] != ' ') {
107 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
108 "mv payload separator not a space");
109 return NGX_ERROR;
110 }
111 dst_len = (size_t) ctx->recv.cur_dlen - (size_t) src_len - 1;
112 if (dst_len == 0) {
113 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
114 "missing destination path");
115 return NGX_ERROR;
116 }
117
118 if (!brix_extract_path(c->log, ctx->recv.payload, (size_t) src_len,
119 mv->src_buf, sizeof(mv->src_buf), 1)) {
120 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
121 "invalid source path payload");
122 return NGX_ERROR;
123 }
124
125 if (!brix_extract_path(c->log, ctx->recv.payload + src_len + 1, dst_len,
126 mv->dst_buf, sizeof(mv->dst_buf), 1)) {
127 ctx->write_rc = brix_send_error(ctx, c, kXR_ArgInvalid,
128 "invalid destination path payload");
129 return NGX_ERROR;
130 }
131
132 return NGX_OK;
133}

163/*
164 * mv_dst_slash_precheck — apply POSIX trailing-slash semantics to the dest.
165 *
166 * WHAT: Rejects a FILE source when the destination is "/"-terminated, then
167 * strips the trailing slashes from mv->dst_buf.
168 * WHY: Matches stock do_Mv/rename(2): rename(X, "Y/") requires X to be a
169 * directory. A FILE source with a "/"-terminated dest is ENOENT; a
170 * DIRECTORY source renames to the stripped name. (Without this our
171 * parent-creation step would create "Y" as a dir and the rename would
172 * then report kXR_isDirectory — diverging from stock.)
173 * HOW: Only engages when dst_buf ends in '/': probes the (already
174 * resolved) source no-follow; a non-directory source fails with the
175 * ENOTDIR category ("not a directory"), exactly as stock reports.
176 * Returns NGX_OK to continue (dst_buf stripped in place), NGX_ERROR
177 * after responding via mv_fail.
178 */
179static ngx_int_t
180mv_dst_slash_precheck(brix_ctx_t *ctx, ngx_connection_t *c,
181 ngx_stream_brix_srv_conf_t *conf, mv_req_t *mv)
182{
183 size_t dl = ngx_strlen(mv->dst_buf);
184 brix_vfs_ctx_t sctx;
185 brix_vfs_stat_t svst;
186
187 if (dl <= 1 || mv->dst_buf[dl - 1] != '/') {
188 return NGX_OK;
189 }
190
191 mv_bind_vfs(ctx, c, conf, mv->src_resolved, &sctx);
192 if (brix_vfs_probe(&sctx, 1 /* no-follow */, &svst) == NGX_OK
193 && !svst.is_directory) {
194 /* rename(file, "dst/") → ENOTDIR (a "/"-terminated target must be
195 * a directory). Stock reports the "not a directory" category. */
196 return mv_fail(ctx, c, mv->src_buf,
197 brix_kxr_from_errno(ENOTDIR), strerror(ENOTDIR));
198 }
199 while (dl > 1 && mv->dst_buf[dl - 1] == '/') {
200 mv->dst_buf[--dl] = '\0';
201 }
202 return NGX_OK;
203}

205/*
206 * mv_make_dst_parents — best-effort creation of the destination parent chain.
207 *
208 * WHAT: Creates any missing parent directories of the destination path,
209 * through the backend driver namespace for a non-POSIX backend or via
210 * brix_mkdir_recursive_policy on a real filesystem.
211 * WHY: Matches the reference rename (XrdOss makes the target path) —
212 * verified against stock xrdfs, which lands a move into a
213 * not-yet-existing directory. Confined beneath the export root by
214 * brix_mkdir_recursive_policy; the source RENAME auth in the caller
215 * has already gated the operation. A component that is an existing
216 * non-dir fails here and the WRITE resolve in the caller then reports
217 * the error — so this helper never responds itself.
218 * HOW: Non-POSIX backend: mkpath the export-relative parent (setgid group
219 * policy is a real-FS-only semantic and is intentionally not applied
220 * for a catalog/object backend). POSIX: expand to the full beneath
221 * path and recursive-mkdir the parent with the export's group rules.
222 */
223static void
224mv_make_dst_parents(ngx_connection_t *c, ngx_stream_brix_srv_conf_t *conf,
225 const char *dst_buf)
226{
227 if (brix_vfs_backend_resolve(conf->common.root_canon, c->log) != NULL) {
228 char rel[BRIX_MAX_PATH + 1];
229 char *slash;
230 size_t rl = ngx_strlen(dst_buf);
231 if (rl < sizeof(rel)) {
232 ngx_memcpy(rel, dst_buf, rl + 1);
233 slash = strrchr(rel, '/');
234 if (slash && slash > rel) {
235 *slash = '\0';
236 (void) brix_vfs_backend_mkpath(conf->common.root_canon, rel,
237 0755, c->log);
238 }
239 }
240 } else {
241 char dst_full[PATH_MAX];
242 char *slash;
243 /* phase72-fp: dst_buf is the request path, dst_full the output buf. */
244 brix_beneath_full_path(conf->common.root_canon, dst_buf, /* NOLINT(readability-suspicious-call-argument) */
245 dst_full, sizeof(dst_full));
246 slash = strrchr(dst_full, '/');
247 if (slash && slash > dst_full) {
248 *slash = '\0';
249 brix_mkdir_recursive_policy(dst_full, 0755, c->log,
250 conf->group_rules);
251 }
252 }
253}

255/*
256 * mv_execute — perform the confined rename and map failure to a kXR error.
257 *
258 * WHAT: Renames src_resolved → dst_resolved through the ctx-bound VFS
259 * surface and, on failure, translates errno into the exact historical
260 * kXR code + message and responds.
261 * WHY: Rename goes through brix_vfs_rename (not the pool-less
262 * brix_vfs_rename_path) so ctx->identity + the backend credential
263 * bind make a remote-backed export see the REQUESTING USER's
264 * credential for the rename call, matching every other data/namespace
265 * op in this file. The namespace status is conveyed as errno (1:1
266 * with the old brix_ns_status_t); the switch mirrors
267 * brix_kxr_map_ns_status exactly.
268 * HOW: Builds a fresh VFS ctx (unconditionally in scope, unlike the
269 * trailing-slash pre-check's conditional one) and a confined
270 * dst_result, then calls brix_vfs_rename. brix_vfs_rename does not
271 * compute was_dir itself (no post-rename stat for the driver case),
272 * so on the EEXIST branch the destination is probed inline to
273 * classify kXR_isDirectory vs kXR_ItExists — extra syscall only on
274 * that error path. Returns NGX_OK on success, NGX_ERROR after
275 * responding via mv_fail.
276 */
277static ngx_int_t
278mv_execute(brix_ctx_t *ctx, ngx_connection_t *c,
279 ngx_stream_brix_srv_conf_t *conf, mv_req_t *mv)
280{
281 brix_vfs_ctx_t rvctx;
282 brix_path_result_t dst_result;
283 int e;
284 int kxr;
285 const char *msg;
286
287 mv_bind_vfs(ctx, c, conf, mv->src_resolved, &rvctx);
288
289 ngx_memzero(&dst_result, sizeof(dst_result));
290 dst_result.is_confined = 1;
291 dst_result.resolved.data = (u_char *) mv->dst_resolved;
292 dst_result.resolved.len = ngx_strlen(mv->dst_resolved);
293
294 if (brix_vfs_rename(&rvctx, &dst_result,
295 0 /* kXR_mv: never replace a dir dest */) == NGX_OK) {
296 return NGX_OK;
297 }
298
299 e = errno;
300 if (e == EACCES || e == EPERM) { /* NS_DENIED */
301 kxr = kXR_NotAuthorized;
302 msg = "permission denied";
303 } else if (e == EISDIR) {
304 /* rename(file, existing-dir) → EISDIR straight from the kernel
305 * (no EEXIST detour, so the was_dir probe below never runs).
306 * Stock reports kXR_isDirectory ("… is a directory") — match it. */
307 kxr = kXR_isDirectory;
308 msg = "destination is a directory";
309 } else if (e == EEXIST) { /* NS_EXISTS */
310 /* was_dir source changed: brix_vfs_rename does not report it,
311 * so probe the (still-existing) destination directly to
312 * classify kXR_isDirectory vs kXR_ItExists. */
313 brix_vfs_ctx_t dctx;
314 brix_vfs_stat_t dvst;
315 int was_dir;
316
317 mv_bind_vfs(ctx, c, conf, mv->dst_resolved, &dctx);
318 was_dir = (brix_vfs_probe(&dctx, 1 /* no-follow */, &dvst) == NGX_OK)
319 && dvst.is_directory;
320
321 kxr = was_dir ? kXR_isDirectory : kXR_ItExists;
322 msg = was_dir ? "destination is a directory"
323 : "destination already exists";
324 } else {
325 /* Mirror brix_kxr_map_ns_status exactly via the 1:1 errno. */
326 switch (e) {
327 case ENOENT: kxr = kXR_NotFound; break; /* NOT_FOUND */
328 case ENOTDIR: kxr = kXR_FSError; break; /* CONFLICT */
329 case ENOTEMPTY: kxr = kXR_ItExists; break; /* NOT_EMPTY */
330 case ENOSPC: kxr = kXR_NoSpace; break; /* NO_SPACE */
331#ifdef EDQUOT
332 case EDQUOT: kxr = kXR_NoSpace; break; /* NO_SPACE */
333#endif
334 case ENAMETOOLONG: kxr = kXR_ArgTooLong; break; /* TOO_LONG */
335 default: kxr = brix_kxr_from_errno(e); break;
336 }
337 msg = strerror(e);
338 }
339 return mv_fail(ctx, c, mv->src_resolved, kxr, msg);
340}

342/*
343 * brix_handle_mv â rename a file or directory.
344 *
345 * Wire format: the payload is src + ' ' + dst.
346 * ClientMvRequest.arg1len carries the source path length (not null-terminated).
347 * The byte at payload[arg1len] must be 0x20 (space) — any other separator
348 * is rejected with kXR_ArgInvalid.
349 *
350 * Both source and destination are independently path-extracted, resolved, and
351 * VO/token-scope-checked. The source must exist (brix_resolve_path); the
352 * destination uses brix_resolve_path_write so a non-existent target is
353 * accepted. The actual rename is done via brix_ns_rename (which performs a
354 * confined renameat under the export root) to close the TOCTOU race between
355 * realpath and rename(2).
356 *
357 * Orchestrator only: parse → resolve/auth source → dest trailing-slash
358 * pre-check → dest parent creation → resolve/auth dest → rename. Each
359 * helper responds itself on failure (rc stashed in ctx->write_rc).
360 */
361ngx_int_t
362brix_handle_mv(brix_ctx_t *ctx, ngx_connection_t *c,
363 ngx_stream_brix_srv_conf_t *conf)
364{
365 mv_req_t mv;
366
367 if (mv_parse_paths(ctx, c, &mv) != NGX_OK) {
368 return ctx->write_rc;
369 }
370
371 /* Source must exist (EXISTING). Confinement is the kernel RESOLVE_BENEATH
372 * inside brix_ns_rename below; this gate only reproduces the historical
373 * "source not found" 404 without a realpath() call. */
374 if (brix_path_resolve_beneath(conf, c->log, mv.src_buf, BRIX_PATH_EXISTING,
375 mv.src_resolved, sizeof(mv.src_resolved)) != NGX_OK) {
376 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MV, "MV", mv.src_buf, "-",
377 kXR_NotFound, "no such file or directory");
378 }
379
380 /* XrdAcc: the move SOURCE requires Rename; the destination requires Insert. */
381 if (brix_auth_gate_op(ctx, c, BRIX_OP_MV, "MV",
382 mv.src_buf, mv.src_resolved, conf,
383 BRIX_AUTH_DELETE, 1, BRIX_AOP_RENAME) != NGX_OK) {
384 return ctx->write_rc;
385 }
386
387 if (mv_dst_slash_precheck(ctx, c, conf, &mv) != NGX_OK) {
388 return ctx->write_rc;
389 }
390
391 mv_make_dst_parents(c, conf, mv.dst_buf);
392
393 /* Destination parent must exist; the tail may not (WRITE). */
394 if (brix_path_resolve_beneath(conf, c->log, mv.dst_buf, BRIX_PATH_WRITE,
395 mv.dst_resolved, sizeof(mv.dst_resolved)) != NGX_OK) {
396 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MV, "MV", mv.src_buf, "-",
397 kXR_NotFound, "invalid destination path");
398 }
399
400 if (brix_auth_gate_op(ctx, c, BRIX_OP_MV, "MV",
401 mv.dst_buf, mv.dst_resolved, conf,
402 BRIX_AUTH_UPDATE, 1, BRIX_AOP_INSERT) != NGX_OK) {
403 return ctx->write_rc;
404 }
405
406 if (mv_execute(ctx, c, conf, &mv) != NGX_OK) {
407 return ctx->write_rc;
408 }
409
410 BRIX_RETURN_OK(ctx, c, BRIX_OP_MV, "MV", mv.src_resolved, mv.dst_resolved, 0);
411}
