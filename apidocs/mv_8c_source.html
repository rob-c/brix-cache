11/*
12 * brix_handle_mv â rename a file or directory.
13 *
14 * Wire format: the payload is src + ' ' + dst.
15 * ClientMvRequest.arg1len carries the source path length (not null-terminated).
16 * The byte at payload[arg1len] must be 0x20 (space) — any other separator
17 * is rejected with kXR_ArgInvalid.
18 *
19 * Both source and destination are independently path-extracted, resolved, and
20 * VO/token-scope-checked. The source must exist (brix_resolve_path); the
21 * destination uses brix_resolve_path_write so a non-existent target is
22 * accepted. The actual rename is done via brix_ns_rename (which performs a
23 * confined renameat under the export root) to close the TOCTOU race between
24 * realpath and rename(2).
25 */
26ngx_int_t
27brix_handle_mv(brix_ctx_t *ctx, ngx_connection_t *c,
28 ngx_stream_brix_srv_conf_t *conf)
29{
30 xrdw_twopath_req_t req;
31 char src_resolved[PATH_MAX];
32 char dst_resolved[PATH_MAX];
33 char src_buf[BRIX_MAX_PATH + 1];
34 char dst_buf[BRIX_MAX_PATH + 1];
35 int16_t src_len;
36 size_t dst_len;
37
38 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
39 return brix_send_error(ctx, c, kXR_ArgMissing, "no paths given");
40 }
41
42 /*
43 * Wire format (from XrdClFileSystem.cc):
44 * arg1len = source.length() (NOT including any terminator)
45 * dlen = src.length() + dst.length() + 1
46 * payload = src[arg1len] + ' ' + dst[...]
47 * The separator between source and destination is a single space (0x20).
48 */
49 xrdw_twopath_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
50 src_len = req.arg1len;
51 if (src_len < 0 || (uint32_t) src_len >= ctx->recv.cur_dlen) {
52 return brix_send_error(ctx, c, kXR_ArgInvalid,
53 "invalid arg1len for mv");
54 }
55
56 /* arg1len == 0: the reference do_Mv (XrdXrootdXeq.cc) splits the buffer on
57 * the FIRST space itself ("old new") rather than rejecting. Reproduce that
58 * autosplit so a well-formed space-separated buffer with arg1len=0 works. */
59 if (src_len == 0) {
60 u_char *sp = memchr(ctx->recv.payload, ' ', (size_t) ctx->recv.cur_dlen);
61 if (sp == NULL || sp == ctx->recv.payload
62 || (size_t)(sp - ctx->recv.payload) > 0x7fff) {
63 return brix_send_error(ctx, c, kXR_ArgInvalid,
64 "invalid path specification");
65 }
66 src_len = (int16_t)(sp - ctx->recv.payload);
67 }
68
69 /* Separator byte at src_len must be a space, with a non-empty dst after it. */
70 if ((uint32_t)(src_len + 1) >= ctx->recv.cur_dlen
71 || ctx->recv.payload[src_len] != ' ') {
72 return brix_send_error(ctx, c, kXR_ArgInvalid,
73 "mv payload separator not a space");
74 }
75 dst_len = (size_t) ctx->recv.cur_dlen - (size_t) src_len - 1;
76 if (dst_len == 0) {
77 return brix_send_error(ctx, c, kXR_ArgInvalid,
78 "missing destination path");
79 }
80
81 /* Parse each half independently so embedded-NUL and traversal checks apply to both. */
82 if (!brix_extract_path(c->log, ctx->recv.payload, (size_t) src_len,
83 src_buf, sizeof(src_buf), 1)) {
84 return brix_send_error(ctx, c, kXR_ArgInvalid,
85 "invalid source path payload");
86 }
87
88 if (!brix_extract_path(c->log, ctx->recv.payload + src_len + 1, dst_len,
89 dst_buf, sizeof(dst_buf), 1)) {
90 return brix_send_error(ctx, c, kXR_ArgInvalid,
91 "invalid destination path payload");
92 }
93
94 /* Source must exist (EXISTING). Confinement is the kernel RESOLVE_BENEATH
95 * inside brix_ns_rename below; this gate only reproduces the historical
96 * "source not found" 404 without a realpath() call. */
97 if (brix_path_resolve_beneath(conf, c->log, src_buf, BRIX_PATH_EXISTING,
98 src_resolved, sizeof(src_resolved)) != NGX_OK) {
99 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MV, "MV", src_buf, "-",
100 kXR_NotFound, "no such file or directory");
101 }
102
103 /* XrdAcc: the move SOURCE requires Rename; the destination requires Insert. */
104 if (brix_auth_gate_op(ctx, c, BRIX_OP_MV, "MV",
105 src_buf, src_resolved, conf,
106 BRIX_AUTH_DELETE, 1, BRIX_AOP_RENAME) != NGX_OK) {
107 return ctx->write_rc;
108 }
109
110 /* POSIX trailing-slash on the destination, matching stock do_Mv/rename(2):
111 * rename(X, "Y/") requires X to be a directory. A FILE source with a
112 * "/"-terminated dest is ENOENT; a DIRECTORY source renames to the stripped
113 * name. (Without this our parent-creation below would create "Y" as a dir
114 * and the rename would then report kXR_isDirectory — diverging from stock.) */
115 {
116 size_t dl = ngx_strlen(dst_buf);
117 if (dl > 1 && dst_buf[dl - 1] == '/') {
118 brix_vfs_ctx_t sctx;
119 brix_vfs_stat_t svst;
120
121 brix_vfs_ctx_init(&sctx, c->pool, c->log, BRIX_PROTO_ROOT,
122 conf->common.root_canon, NULL, conf->common.allow_write,
123 0 /* is_tls */, NULL, src_resolved);
124 if (brix_vfs_probe(&sctx, 1 /* no-follow */, &svst) == NGX_OK
125 && !svst.is_directory) {
126 /* rename(file, "dst/") → ENOTDIR (a "/"-terminated target must be
127 * a directory). Stock reports the "not a directory" category. */
128 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MV, "MV", src_buf, "-",
129 brix_kxr_from_errno(ENOTDIR),
130 strerror(ENOTDIR));
131 }
132 while (dl > 1 && dst_buf[dl - 1] == '/') {
133 dst_buf[--dl] = '\0';
134 }
135 }
136 }
137
138 /* Create the destination parent chain if missing, matching the reference
139 * rename (XrdOss makes the target path) — verified against stock xrdfs, which
140 * lands a move into a not-yet-existing directory. Confined beneath the export
141 * root by brix_mkdir_recursive_policy; the source RENAME auth above has
142 * already gated the operation. A component that is an existing non-dir fails
143 * here and the WRITE resolve below then reports the error. */
144 if (brix_vfs_backend_resolve(conf->common.root_canon, c->log) != NULL) {
145 /* Non-POSIX backend: create the destination parent chain in the driver
146 * namespace (export-relative). setgid group policy is a real-FS-only
147 * semantic and is intentionally not applied for a catalog/object backend. */
148 char rel[BRIX_MAX_PATH + 1];
149 char *slash;
150 size_t rl = ngx_strlen(dst_buf);
151 if (rl < sizeof(rel)) {
152 ngx_memcpy(rel, dst_buf, rl + 1);
153 slash = strrchr(rel, '/');
154 if (slash && slash > rel) {
155 *slash = '\0';
156 (void) brix_vfs_backend_mkpath(conf->common.root_canon, rel,
157 0755, c->log);
158 }
159 }
160 } else {
161 char dst_full[PATH_MAX];
162 char *slash;
163 brix_beneath_full_path(conf->common.root_canon, dst_buf,
164 dst_full, sizeof(dst_full));
165 slash = strrchr(dst_full, '/');
166 if (slash && slash > dst_full) {
167 *slash = '\0';
168 brix_mkdir_recursive_policy(dst_full, 0755, c->log,
169 conf->group_rules);
170 }
171 }
172
173 /* Destination parent must exist; the tail may not (WRITE). */
174 if (brix_path_resolve_beneath(conf, c->log, dst_buf, BRIX_PATH_WRITE,
175 dst_resolved, sizeof(dst_resolved)) != NGX_OK) {
176 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MV, "MV", src_buf, "-",
177 kXR_NotFound, "invalid destination path");
178 }
179
180 if (brix_auth_gate_op(ctx, c, BRIX_OP_MV, "MV",
181 dst_buf, dst_resolved, conf,
182 BRIX_AUTH_UPDATE, 1, BRIX_AOP_INSERT) != NGX_OK) {
183 return ctx->write_rc;
184 }
185
186 {
187 int was_dir = 0;
188
189 /* Rename through the thread-safe VFS surface; the namespace status is
190 * conveyed as errno (1:1 with the old brix_ns_status_t) + was_dir, so
191 * the kXR mapping below is identical to the prior brix_ns_rename path. */
192 if (brix_vfs_rename_path(
193 brix_vfs_backend_resolve(conf->common.root_canon, c->log),
194 c->log, conf->common.root_canon,
195 src_resolved, dst_resolved, 0, &was_dir)
196 != NGX_OK)
197 {
198 int e = errno;
199 int kxr;
200 const char *msg;
201
202 if (e == EACCES || e == EPERM) { /* NS_DENIED */
203 kxr = kXR_NotAuthorized;
204 msg = "permission denied";
205 } else if (e == EEXIST) { /* NS_EXISTS */
206 kxr = was_dir ? kXR_isDirectory : kXR_ItExists;
207 msg = was_dir ? "destination is a directory"
208 : "destination already exists";
209 } else {
210 /* Mirror brix_kxr_map_ns_status exactly via the 1:1 errno. */
211 switch (e) {
212 case ENOENT: kxr = kXR_NotFound; break; /* NOT_FOUND */
213 case ENOTDIR: kxr = kXR_FSError; break; /* CONFLICT */
214 case ENOTEMPTY: kxr = kXR_ItExists; break; /* NOT_EMPTY */
215 case ENOSPC: kxr = kXR_NoSpace; break; /* NO_SPACE */
216 case ENAMETOOLONG: kxr = kXR_ArgTooLong; break; /* TOO_LONG */
217 default: kxr = brix_kxr_from_errno(e); break;
218 }
219 msg = strerror(e);
220 }
221 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MV, "MV", src_resolved, "-",
222 kxr, msg);
223 }
224 }
225
226 BRIX_RETURN_OK(ctx, c, BRIX_OP_MV, "MV", src_resolved, dst_resolved, 0);
227}
