10 * brix_handle_sync — sync an open file by handle through the VFS core.
11 *
12 * Wire format (ClientSyncRequest): fhandle[4] identifies the open slot.
13 * No payload is expected. On success, all pending writes on the file
14 * descriptor are flushed to stable storage before kXR_ok is sent.
15 */

28 return rc;
29 }

31 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
