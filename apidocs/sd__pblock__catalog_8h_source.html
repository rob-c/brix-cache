1/*
2 * sd_pblock_catalog.h — SQLite metadata catalog for the pblock backend.
3 *
4 * WHAT: The namespace + metadata half of the pblock ("pseudo-block") storage
5 * driver. Owns one SQLite connection per instance and exposes typed CRUD
6 * over two tables: `objects` (one row per logical path: dir flag, blob id,
7 * size, mtime, ctime, mode) and `xattrs` (one row per (path, name)). The
8 * catalog holds the entire logical namespace and the path->blob mapping;
9 * the opaque blob bytes live in plain POSIX files keyed by blob id.
10 *
11 * WHY: Splitting metadata (here) from bulk content (blob files) is what lets
12 * pblock be a full-capability drop-in for POSIX while keeping the hot
13 * data path (pread/pwrite/sendfile on a blob fd) free of any database
14 * work. Keeping this layer pure libc + sqlite3 — no nginx types — makes it
15 * independently unit-testable (sd_pblock_catalog_unittest.c) and reusable
16 * from the driver vtable above it (sd_pblock.c).
17 *
18 * HOW: An opaque handle wraps the sqlite3* and its prepared statements. All
19 * calls return 0 / a count on success and -1 with errno set on failure
20 * (lookup uses +1 for "no such row"). Directory rename reparents every
21 * descendant in one transaction. Concurrency across nginx worker processes
22 * is handled by WAL journal mode + a busy timeout, set at open.
23 */

31/* Width of a blob id: 128 bits rendered as lowercase hex, plus the NUL. */
32#define PBLOCK_BLOB_ID_HEX 32
33#define PBLOCK_BLOB_ID_CAP (PBLOCK_BLOB_ID_HEX + 1)
34
35/* Default object block size (64 MiB) when an export configures none. A file's
36 * data is striped across fixed-size block files of this size; the value is fixed
37 * per file at creation, so retuning the default only affects newer files. */
38#define PBLOCK_DEFAULT_BLOCK_SIZE (64LL * 1024 * 1024)

40/* Opaque per-instance catalog handle: one sqlite3 connection + prepared stmts. */
41typedef struct pblock_catalog pblock_catalog;
42
43/* A namespace entry's metadata. blob_id is the empty string for directories;
44 * block_size is the per-file stripe size (0 for directories). */
45typedef struct {
46 int is_dir;
47 char blob_id[PBLOCK_BLOB_ID_CAP];
48 int64_t size;
49 int64_t block_size;
50 int64_t mtime;
51 int64_t ctime;
52 uint32_t mode;
53} pblock_meta;

55/* Open (creating the file + schema if needed) the catalog at db_path.
56 * busy_timeout_ms arms SQLite's busy handler so a writer blocked by another
57 * worker process retries rather than failing. Returns a handle, or NULL/errno. */
58pblock_catalog *pblock_catalog_open(const char *db_path, int busy_timeout_ms);

60/* Finalize statements + close the connection. NULL-safe. */
61void pblock_catalog_close(pblock_catalog *cat);

63/* Look up `path`: 0 and fill *out when present, 1 when absent, -1/errno on error.
64 * *out may be NULL to probe existence only. */

78/* Apply a metadata change to an existing row in ONE UPDATE (no read-modify-write):
