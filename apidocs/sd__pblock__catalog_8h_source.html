1/*
2 * sd_pblock_catalog.h — SQLite metadata catalog for the pblock backend.
3 *
4 * WHAT: The namespace + metadata half of the pblock ("pseudo-block") storage
5 * driver. Owns one SQLite connection per instance and exposes typed CRUD
6 * over three tables: `objects` (one row per logical path: dir flag, blob
7 * id, size, mtime, ctime, mode, uid, gid), `xattrs` (one row per
8 * (path, name)) and `ids` (the catalog-internal identity registry). The
9 * catalog holds the entire logical namespace and the path->blob mapping;
10 * the opaque blob bytes live in plain POSIX files keyed by blob id.
11 *
12 * WHY: Splitting metadata (here) from bulk content (blob files) is what lets
13 * pblock be a full-capability drop-in for POSIX while keeping the hot
14 * data path (pread/pwrite/sendfile on a blob fd) free of any database
15 * work. Keeping this layer pure libc + sqlite3 — no nginx types — makes it
16 * independently unit-testable (sd_pblock_catalog_unittest.c) and reusable
17 * from the driver vtable above it (sd_pblock.c).
18 *
19 * HOW: An opaque handle wraps the sqlite3* and its prepared statements. All
20 * calls return 0 / a count on success and -1 with errno set on failure
21 * (lookup uses +1 for "no such row"). Directory rename reparents every
22 * descendant in one transaction. Concurrency across nginx worker processes
23 * is handled by WAL journal mode + a busy timeout, set at open.
24 */

32/* Width of a blob id: 128 bits rendered as lowercase hex, plus the NUL. */
33#define PBLOCK_BLOB_ID_HEX 32
34#define PBLOCK_BLOB_ID_CAP (PBLOCK_BLOB_ID_HEX + 1)

36/* Default object block size (64 MiB) when an export configures none. A file's
37 * data is striped across fixed-size block files of this size; the value is fixed
38 * per file at creation, so retuning the default only affects newer files. */
39#define PBLOCK_DEFAULT_BLOCK_SIZE (64LL * 1024 * 1024)

41/* Opaque per-instance catalog handle: one sqlite3 connection + prepared stmts. */
42typedef struct pblock_catalog pblock_catalog;

44/* A namespace entry's metadata. blob_id is the empty string for directories;
45 * block_size is the per-file stripe size (0 for directories). uid/gid are
46 * catalog-internal synthetic ids from the `ids` registry (0 = the service
47 * itself — rows created before the identity registry existed, or by requests
48 * carrying no identity). */
49typedef struct {
50 int is_dir;
51 char blob_id[PBLOCK_BLOB_ID_CAP];
52 int64_t size;
53 int64_t block_size;
54 int64_t mtime;
55 int64_t ctime;
56 uint32_t mode;
57 uint32_t uid;
58 uint32_t gid;
59 char xform[16]; /* F12/F13: per-file transform kind name ("crypt"/
60 * "zstd"), "" = raw. Recorded at create; the export's
61 * configured transform holds the key/codec. */
62} pblock_meta;
63
64/* Catalog-internal identity registry (`ids` table): stable synthetic numeric
65 * ids auto-assigned per principal (uid namespace) and per VO/group name (gid
66 * namespace), starting at PBLOCK_ID_BASE. Ids are never reused or renumbered —
67 * they are what the objects table's uid/gid columns reference. */
68#define PBLOCK_ID_BASE 10000
69#define PBLOCK_ID_USER 0
70#define PBLOCK_ID_GROUP 1

72/* Open (creating the file + schema if needed) the catalog at db_path.

95/* Apply a metadata change to an existing row in ONE UPDATE (no read-modify-write):

97 * with `perm` (preserving the type bits); when set_mtime, sets mtime; when
98 * set_owner, sets uid/gid — pass -1 for either to keep its current value
99 * (POSIX chown semantics). 0, or -1/errno (ENOENT if the path is absent). */

104/* Map a principal (PBLOCK_ID_USER) or VO/group name (PBLOCK_ID_GROUP) to its
