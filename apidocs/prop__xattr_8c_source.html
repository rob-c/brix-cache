1/*
2 * prop_xattr.c — xattr-based lock persistence for WebDAV.
3 *
4 * A WebDAV lock is encoded as a single xattr on the locked resource:
5 * token=<tok>|owner=<owner>|expires=<msec>|scope=<exclusive|shared>|depth=<infinity|0>
6 *
7 * XATTR_CREATE semantics make lock creation atomic across workers: if two
8 * workers race on the same unlocked path, exactly one setxattr(XATTR_CREATE)
9 * succeeds and the other gets EEXIST → NGX_DECLINED → 423 Locked.
10 */

12#include "fs/path/path.h"
13#include "fs/vfs/vfs.h"

15#include <sys/xattr.h>
16#include <stdio.h>

21/*
22 * Phase 40: lock xattrs must be written/read/removed AS THE MAPPED USER under
23 * impersonation, else the worker (svc) cannot setxattr on the user-owned lock
24 * file (EACCES) and LOCK/UNLOCK break. These helpers take the request so they
25 * can resolve the export root and route through the VFS xattr surface, which
26 * delegates to brix_*xattr_confined_canon (the broker when map mode is active,
27 * the raw path-based syscall otherwise) while adding the OP_XATTR metric +
28 * access-log line — confinement and errno behaviour are unchanged.
29 */
30

32 * Build a transient VFS ctx for a confined xattr op on `path` (mirrors the
33 * canonical construction in get.c). The xattr family is not allow_write-gated,
34 * so the allow_write flag passed here does not affect set/remove behaviour.
35 */

60ngx_int_t
61webdav_lock_xattr_encode(const webdav_lock_xattr_t *e, char *out, size_t outsz)
62{
63 int n;
64
65 /*
66 * Schema v2: `expires` is absolute Unix WALL-CLOCK seconds (not the legacy
67 * v1 monotonic-msec value, which was meaningless after a reboot). The leading
68 * `v=2` lets the decoder reject/expire any pre-upgrade v1 record. `null=1`
69 * marks a lock-null placeholder (RFC 4918 §9.10.1).
70 */
71 n = snprintf(out, outsz,
72 "v=2|token=%s|owner=%s|expires=%lld|scope=%s|depth=%s|null=%d",
73 e->token, e->owner,
74 (long long) e->expires,
75 e->exclusive ? "exclusive" : "shared",
76 e->depth_infinity ? "infinity" : "0",
77 e->is_null ? 1 : 0);
78 return (n > 0 && (size_t) n < outsz) ? NGX_OK : NGX_ERROR;
79}

81ngx_int_t
82webdav_lock_xattr_decode(const char *raw, size_t rawlen, webdav_lock_xattr_t *e)
83{
84 char buf[WEBDAV_LOCK_XATTR_MAXLEN];
85 char *p, *end, *val, *next;
86 int version = 0;
87
88 if (rawlen == 0 || rawlen >= sizeof(buf)) {
89 return NGX_DECLINED;
90 }
91
92 ngx_memcpy(buf, raw, rawlen);
93 buf[rawlen] = '\0';
94 ngx_memzero(e, sizeof(*e));
95
96 p = buf;
97 end = buf + rawlen;
98
99 while (p < end) {
100 next = strchr(p, '|');
101 if (next != NULL) {
102 *next = '\0';
103 }
104
105 val = strchr(p, '=');
106 if (val != NULL) {
107 *val++ = '\0';
108
109 if (strcmp(p, "v") == 0) {
110 version = (int) strtol(val, NULL, 10);
111 } else if (strcmp(p, "token") == 0) {
112 ngx_cpystrn((u_char *) e->token, (u_char *) val,
113 sizeof(e->token));
114 } else if (strcmp(p, "owner") == 0) {
115 ngx_cpystrn((u_char *) e->owner, (u_char *) val,
116 sizeof(e->owner));
117 } else if (strcmp(p, "expires") == 0) {
118 e->expires = (int64_t) strtoll(val, NULL, 10);
119 } else if (strcmp(p, "scope") == 0) {
120 e->exclusive = (strcmp(val, "exclusive") == 0) ? 1 : 0;
121 } else if (strcmp(p, "depth") == 0) {
122 e->depth_infinity = (strcmp(val, "infinity") == 0) ? 1 : 0;
123 } else if (strcmp(p, "null") == 0) {
124 e->is_null = (strcmp(val, "1") == 0) ? 1 : 0;
125 }
126 }
127
128 p = next ? next + 1 : end;
129 }
130
131 if (e->token[0] == '\0') {
132 return NGX_DECLINED;
133 }
134
135 /*
136 * Migration guard: a legacy v1 record (no `v=2`) carries a MONOTONIC `expires`
137 * that is meaningless in this process (especially after a reboot). Force it to
138 * 0 (already-expired) so every caller's existing expired-lock cleanup path
139 * deletes it and proceeds — a downgrade therefore releases stale locks rather
140 * than honouring a bogus deadline. Returning NGX_OK (not NGX_DECLINED) is
141 * deliberate: NGX_DECLINED would leave the physical xattr in place and a fresh
142 * XATTR_CREATE LOCK would then wrongly hit EEXIST -> 423.
143 */
144 if (version != 2) {
145 e->expires = 0;
146 }
147
148 return NGX_OK;
149}

151ngx_int_t
152webdav_lock_xattr_write(ngx_http_request_t *r, const char *path,
153 const webdav_lock_xattr_t *e, int flags)
154{
155 ngx_log_t *log = r->connection->log;
156 brix_vfs_ctx_t vctx;
157 char buf[WEBDAV_LOCK_XATTR_MAXLEN];
158
159 if (webdav_lock_xattr_encode(e, buf, sizeof(buf)) != NGX_OK) {
160 ngx_log_error(NGX_LOG_ERR, log, 0,
161 "brix_webdav: lock xattr encode failed for \"%s\"", path);
162 return NGX_ERROR;
163 }
164
165 webdav_lock_vfs_ctx_init(r, path, &vctx);
166
167 if (brix_vfs_setxattr(&vctx, WEBDAV_LOCK_XATTR_KEY, buf, strlen(buf),
168 flags) != NGX_OK)
169 {
170 if (errno == EEXIST) {
171 return NGX_DECLINED; /* XATTR_CREATE race — another worker won */
172 }
173 ngx_log_error(NGX_LOG_ERR, log, errno,
174 "brix_webdav: setxattr lock on \"%s\" failed", path);
175 return NGX_ERROR;
176 }
177
178 return NGX_OK;
179}

181ngx_int_t
182webdav_lock_xattr_read(ngx_http_request_t *r, const char *path,
183 webdav_lock_xattr_t *e)
184{
185 ngx_log_t *log = r->connection->log;
186 brix_vfs_ctx_t vctx;
187 char buf[WEBDAV_LOCK_XATTR_MAXLEN];
188 ssize_t n;
189
190 webdav_lock_vfs_ctx_init(r, path, &vctx);
191
192 n = brix_vfs_getxattr(&vctx, WEBDAV_LOCK_XATTR_KEY, buf, sizeof(buf) - 1);
193 if (n < 0) {
194 /* No lock present, OR a backend that cannot store xattrs at all (object /
195 * remote root:// stores) — either way the resource carries no WebDAV lock,
196 * so a write may proceed. */
197 if (errno == ENODATA || errno == ENOATTR || errno == ENOENT
198 || errno == ENOTSUP || errno == EOPNOTSUPP)
199 {
200 return NGX_DECLINED;
201 }
202 ngx_log_error(NGX_LOG_ERR, log, errno,
203 "brix_webdav: getxattr lock on \"%s\" failed", path);
204 return NGX_ERROR;
205 }
206
207 return webdav_lock_xattr_decode(buf, (size_t) n, e);
208}

211webdav_lock_xattr_delete(ngx_http_request_t *r, const char *path)
212{
213 ngx_log_t *log = r->connection->log;
214 brix_vfs_ctx_t vctx;
215
216 webdav_lock_vfs_ctx_init(r, path, &vctx);
217
218 if (brix_vfs_removexattr(&vctx, WEBDAV_LOCK_XATTR_KEY) != NGX_OK) {
219 if (errno == ENODATA || errno == ENOATTR || errno == ENOENT
220 || errno == ENOTSUP || errno == EOPNOTSUPP)
221 {
222 return NGX_OK; /* idempotent (incl. backends without xattr) */
223 }
224 ngx_log_error(NGX_LOG_WARN, log, errno,
225 "brix_webdav: removexattr lock on \"%s\" failed", path);
226 return NGX_ERROR;
227 }
228
229 return NGX_OK;
