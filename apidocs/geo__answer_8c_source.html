1/* geo_answer.c — server-side RTT-ranked answer to the CVMFS geo API.
2 *
3 * WHAT: when brix_cvmfs_geo_answer is `rtt`, this replaces the blind upstream
4 * passthrough (geo.c): it parses the client-supplied server list, measures
5 * TCP-connect RTT from THIS proxy to each server, and replies with the
6 * nearest-first 1-based permutation the CVMFS client expects. Because the
7 * proxy sits next to the client, its RTT is the client's effective
8 * proximity — so a buggy/mis-ordered upstream GeoAPI (probing distant
9 * Stratum-1s) is bypassed entirely (replace, not repair).
10 * WHY: a client that receives a bad geo order flails to far servers before
11 * falling back; answering authoritatively from here keeps it on the near
12 * ones without trusting the upstream answer.
13 * HOW: the geo.c async idiom — parse + cache-read on the event loop, blocking
14 * connect probes on a thread-pool task, fold/rank/respond in the done
15 * handler. Ranking uses three order-preserving buckets:
16 * reachable (by RTT) → unreachable → unprobed (guard-skipped / over-cap),
17 * so a server is never dropped and the client always gets a complete,
18 * well-formed permutation. The list is client-supplied, so probing is
19 * guarded: only CVMFS server ports {80,443,8000}, capped at
20 * brix_cvmfs_geo_max_servers — the node cannot be turned into a general
21 * port scanner. A short per-worker EWMA cache (keyed host:port) answers
22 * repeat/remount requests without re-probing. Any parse/setup failure
23 * falls back to brix_cvmfs_geo_passthrough.
24 */

34#define CVMFS_GEO_MAX_ENTRIES 64 /* hard cap on the parsed list */
35#define CVMFS_GEO_PROBE_TIMEOUT_MS 1000 /* per-server connect probe budget */
36#define CVMFS_GEO_CACHE_SLOTS 256u /* per-worker EWMA cache (pow2 mask) */

38/* Metric bucket bases: reachable RTTs (µs, small) sort first; unreachable and
39 * unprobed sort after, each preserving the client's original relative order via
40 * the +index tiebreak. The gaps dwarf any real RTT so buckets never overlap. */
41#define CVMFS_GEO_M_UNREACHABLE 1.0e12
42#define CVMFS_GEO_M_UNPROBED 2.0e12

73static uint64_t
74cvmfs_geo_key(const char *host, int port)
75{
76 uint64_t h = 0xcbf29ce484222325ull;
77 const char *p;
78 char portbuf[8];
79 int i, n;
80
81 for (p = host; *p != '\0'; p++) {
82 h = (h ^ (unsigned char) *p) * 0x100000001b3ull;
83 }
84 n = snprintf(portbuf, sizeof(portbuf), ":%d", port);
85 for (i = 0; i < n; i++) {
86 h = (h ^ (unsigned char) portbuf[i]) * 0x100000001b3ull;
87 }
88 return (h != 0) ? h : 1;
89}

97/* Parse one "host" or "host:port" token into `e`. A token with a non-host
98 * character (a coordinate literal, IPv6, garbage) or a disallowed port is left
99 * !valid — it keeps its response slot but lands in the unprobed bucket. */
100static void
101cvmfs_geo_parse_entry(const u_char *tok, size_t len, cvmfs_geo_entry_t *e)
102{
103 size_t i, hlen = 0;
104 int port = 80;
105
106 e->valid = 0;
107 e->need_probe = 0;
108 e->cached_ewma = 0.0;
109 e->sample_us = -1;
110 e->host[0] = '\0';
111 e->port = 0;
112
113 if (len == 0) {
114 return;
115 }
116 /* host run: [a-zA-Z0-9.-] */
117 for (i = 0; i < len; i++) {
118 u_char c = tok[i];
119
120 if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')
121 || (c >= '0' && c <= '9') || c == '.' || c == '-')
122 {
123 if (hlen + 1 >= sizeof(e->host)) {
124 return; /* host too long → unprobed */
125 }
126 e->host[hlen++] = (char) c;
127 continue;
128 }
129 break;
130 }
131 if (hlen == 0) {
132 return; /* no host → unprobed */
133 }
134 e->host[hlen] = '\0';
135
136 if (i < len) {
137 /* the only accepted separator is ":<digits>" to end-of-token */
138 if (tok[i] != ':') {
139 return;
140 }
141 port = 0;
142 for (i++; i < len; i++) {
143 if (tok[i] < '0' || tok[i] > '9') {
144 return; /* junk after host → unprobed */
145 }
146 port = port * 10 + (tok[i] - '0');
147 if (port > 65535) {
148 return;
149 }
150 }
151 if (port == 0) {
152 return;
153 }
154 }
155 e->port = port;
156 e->valid = cvmfs_geo_port_allowed(port); /* guard: CVMFS server ports */
157}

159/* Split the server list (last '/'-segment of the geo URI) into entries. Returns
160 * the count, or −1 to signal the caller to fall back to passthrough (no list,
161 * or more servers than we can index). */
162static int
163cvmfs_geo_parse(ngx_http_request_t *r, cvmfs_geo_task_t *t, ngx_uint_t max_probe)
164{
165 const u_char *uri = r->uri.data;
166 const u_char *end = uri + r->uri.len;
167 const u_char *slash = NULL, *p, *tok;
168 int n = 0;
169
170 for (p = uri; p < end; p++) {
171 if (*p == '/') {
172 slash = p;
173 }
174 }
175 if (slash == NULL || slash + 1 >= end) {
176 return -1;
177 }
178 tok = slash + 1;
179 for (p = tok; p <= end; p++) {
180 if (p == end || *p == ',') {
181 if (n >= CVMFS_GEO_MAX_ENTRIES) {
182 return -1; /* absurd list → passthrough */
183 }
184 cvmfs_geo_parse_entry(tok, (size_t) (p - tok), &t->e[n]);
185 if ((ngx_uint_t) n >= max_probe) {
186 t->e[n].valid = 0; /* over cap → unprobed */
187 }
188 n++;
189 tok = p + 1;
190 }
191 }
192 t->n = n;
193 return (n > 0) ? 0 : -1;
194}

196/* Event-loop pre-pass: mark each valid entry probe-needed unless a fresh cache
197 * entry answers it. */
198static void
199cvmfs_geo_cache_read(cvmfs_geo_task_t *t, time_t now)
200{
201 int i;
202
203 for (i = 0; i < t->n; i++) {
204 cvmfs_geo_entry_t *e = &t->e[i];
205 cvmfs_geo_slot_t *s;
206 uint64_t k;
207
208 if (!e->valid) {
209 continue;
210 }
211 k = cvmfs_geo_key(e->host, e->port);
212 s = &cvmfs_geo_cache[k & (CVMFS_GEO_CACHE_SLOTS - 1)];
213 if (s->key == k && s->ewma_us > 0.0 && (now - s->updated) < t->ttl) {
214 e->need_probe = 0;
215 e->cached_ewma = s->ewma_us;
216 } else {
217 e->need_probe = 1;
218 }
219 }
220}

222/* Done-side: fold one fresh (reachable) sample into the cache and return the
223 * effective EWMA. Folds onto an existing same-key sample, else seeds it. */
224static double
225cvmfs_geo_cache_fold(const cvmfs_geo_entry_t *e, time_t now)
226{
227 uint64_t k = cvmfs_geo_key(e->host, e->port);
228 cvmfs_geo_slot_t *s = &cvmfs_geo_cache[k & (CVMFS_GEO_CACHE_SLOTS - 1)];
229 double sample = (double) e->sample_us;
230 double ewma;
231
232 ewma = (s->key == k && s->ewma_us > 0.0)
233 ? s->ewma_us * 0.75 + sample * 0.25
234 : sample;
235 s->key = k;
236 s->ewma_us = ewma;
237 s->updated = now;
238 return ewma;
239}

259/* Build the nearest-first 1-based permutation string into t->body. */
260static void
261cvmfs_geo_render(cvmfs_geo_task_t *t, time_t now)
262{
263 double metric[CVMFS_GEO_MAX_ENTRIES];
264 int ranks[CVMFS_GEO_MAX_ENTRIES];
265 int order[CVMFS_GEO_MAX_ENTRIES];
266 int i;
267 size_t off = 0;
268 size_t cap = (size_t) t->n * 8 + 2;
269
270 for (i = 0; i < t->n; i++) {
271 cvmfs_geo_entry_t *e = &t->e[i];
272
273 if (!e->valid) {
274 metric[i] = CVMFS_GEO_M_UNPROBED + i;
275 } else if (!e->need_probe) {
276 metric[i] = e->cached_ewma; /* reachable bucket */
277 } else if (e->sample_us >= 0) {
278 metric[i] = cvmfs_geo_cache_fold(e, now); /* reachable bucket */
279 } else {
280 metric[i] = CVMFS_GEO_M_UNREACHABLE + i; /* unreachable */
281 }
282 }
283
284 brix_cvmfs_rank_by_metric(metric, t->n, ranks);
285 for (i = 0; i < t->n; i++) {
286 order[ranks[i]] = i + 1; /* 1-based index */
287 }
288 for (i = 0; i < t->n && off < cap; i++) {
289 int w = snprintf((char *) t->body + off, cap - off, "%s%d",
290 (i > 0) ? "," : "", order[i]);
291 if (w < 0 || (size_t) w >= cap - off) {
292 break;
293 }
294 off += (size_t) w;
295 }
296 if (off + 1 < cap) {
297 t->body[off++] = '\n';
298 }
299 t->body_len = off;
300}

302/* event-loop side: fold + rank + respond. */
303static void
304cvmfs_geo_done(ngx_event_t *ev)
305{
306 ngx_thread_task_t *task = ev->data;
307 cvmfs_geo_task_t *t = task->ctx;
308 ngx_http_request_t *r = t->r;
309 ngx_connection_t *c = r->connection;
310 ngx_buf_t *b;
311 ngx_chain_t out;
312
313 cvmfs_geo_render(t, ngx_time());
314
315 r->headers_out.status = NGX_HTTP_OK;
316 r->headers_out.content_length_n = (off_t) t->body_len;
317 ngx_str_set(&r->headers_out.content_type, "text/plain");
318 r->headers_out.content_type_len = r->headers_out.content_type.len;
319 if (ngx_http_send_header(r) != NGX_OK || r->header_only || t->body_len == 0) {
320 ngx_http_finalize_request(r, ngx_http_send_special(r, NGX_HTTP_LAST));
321 ngx_http_run_posted_requests(c);
322 return;
323 }
324 b = ngx_pcalloc(r->pool, sizeof(*b));
325 if (b == NULL) {
326 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
327 ngx_http_run_posted_requests(c);
328 return;
329 }
330 b->pos = b->start = t->body;
331 b->last = b->end = t->body + t->body_len;
332 b->memory = 1;
333 b->last_buf = 1;
334 out.buf = b;
335 out.next = NULL;
336 ngx_http_finalize_request(r, ngx_http_output_filter(r, &out));
337 ngx_http_run_posted_requests(c);
338}

340ngx_int_t
341brix_cvmfs_geo_answer(ngx_http_request_t *r,
342 ngx_http_brix_cvmfs_loc_conf_t *lcf)
343{
344 ngx_thread_task_t *task;
345 cvmfs_geo_task_t *t;
346 ngx_thread_pool_t *pool;
347 ngx_uint_t max_probe;
348
349 pool = lcf->common.thread_pool;
350 if (pool == NULL) {
351 static ngx_str_t default_name = ngx_string("default");
352 ngx_str_t *pname = lcf->common.thread_pool_name.len > 0
353 ? &lcf->common.thread_pool_name
354 : &default_name;
355
356 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
357 if (pool != NULL) {
358 lcf->common.thread_pool = pool;
359 }
360 }
361 if (pool == NULL) {
362 return brix_cvmfs_geo_passthrough(r, lcf); /* no pool → relay */
363 }
364
365 task = ngx_thread_task_alloc(r->pool, sizeof(cvmfs_geo_task_t));
366 if (task == NULL) {
367 return brix_cvmfs_geo_passthrough(r, lcf);
368 }
369 t = task->ctx;
370 ngx_memzero(t, sizeof(*t));
371 t->r = r;
372 t->ttl = (lcf->cvmfs.geo_cache_ttl > 0) ? lcf->cvmfs.geo_cache_ttl : 60;
373
374 max_probe = (lcf->cvmfs.geo_max_servers > 0) ? lcf->cvmfs.geo_max_servers
375 : 16;
376 if (cvmfs_geo_parse(r, t, max_probe) != 0) {
377 return brix_cvmfs_geo_passthrough(r, lcf); /* unparseable → relay */
378 }
379
380 t->body = ngx_palloc(r->pool, (size_t) t->n * 8 + 2);
381 if (t->body == NULL) {
382 return brix_cvmfs_geo_passthrough(r, lcf);
383 }
384
385 cvmfs_geo_cache_read(t, ngx_time());
386
387 brix_task_bind(task, cvmfs_geo_thread, cvmfs_geo_done);
388 task->event.log = r->connection->log;
389
390 if (ngx_thread_task_post(pool, task) != NGX_OK) {
391 return brix_cvmfs_geo_passthrough(r, lcf);
392 }
393 r->main->count++;
394 return NGX_DONE;
395}
