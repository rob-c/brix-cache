1/*
2 * zip_dir.c — ZIP central-directory reader (phase-57 W2). See zip_dir.h.
3 *
4 * Pure C, pread-only, fully bounds-checked. No nginx / OpenSSL deps so it can be
5 * unit-tested standalone. The security-critical parsing (EOCD/ZIP64 location,
6 * ZIP64-extra decode, local-header data-offset resolution) lives in the shared
7 * zip_kernel.c — the single audited copy linked into both this module and the
8 * native client. This file is the server-side adapter: it wraps the open fd in
9 * the kernel's pread callback (via the SD backend), walks the central directory

30/* Read exactly n bytes at off into buf; 0 on success, -1 on short read/error. */
31static int pread_full(int fd, void *buf, size_t n, off_t off)
32{
33 unsigned char *p = buf;
34 brix_sd_obj_t obj;
35 brix_sd_posix_wrap(&obj, fd);
36 while (n > 0) {
37 ssize_t r = obj.driver->pread(&obj, p, n, off);
38 if (r < 0) {
39 if (errno == EINTR) {
40 continue;
41 }
42 return -1;
43 }
44 if (r == 0) {
45 return -1; /* unexpected EOF */
46 }
47 p += r;
48 off += r;
49 n -= (size_t) r;
50 }
51 return 0;
52}

54/* Kernel pread adapter: read exactly `len` bytes from the open fd boxed in ctx. */
55static ssize_t zip_fd_pread(void *ctx, uint64_t off, void *buf, size_t len)
56{
57 int fd = *(int *) ctx;
58 if (pread_full(fd, buf, len, (off_t) off) != 0) {
59 return -1;
60 }
61 return (ssize_t) len;

98/*
99 * WHAT: Decode and bounds-check the CDFH record at offset `p` in the central
100 * directory buffer `cd` of length `cd_size`.
101 * WHY: The record header, the variable-length trailer (name/extra/comment),
102 * and the derived `span` are all derived from untrusted directory bytes;
103 * every field must be validated before the walk reads them.
104 * HOW: Verify the fixed header fits and carries the CDFH signature, decode the
105 * fields, then verify the full variable-length record fits in `cd_size`.
106 * Returns 0 on success, -1 on any malformed/overrunning record.
107 */
108static int
109zip_parse_cdfh(const unsigned char *cd, uint64_t p, uint64_t cd_size,
110 brix_zip_cdfh_t *rec)
111{
112 if (p + ZIP_CDFH_BASE > cd_size || zip_rd32le(cd + p) != ZIP_CDFH_SIG) {
113 return -1;
114 }
115 rec->bits = zip_rd16le(cd + p + 8);
116 rec->method = zip_rd16le(cd + p + 10);
117 rec->crc32 = zip_rd32le(cd + p + 16);
118 rec->comp = zip_rd32le(cd + p + 20);
119 rec->uncomp = zip_rd32le(cd + p + 24);
120 rec->fn = zip_rd16le(cd + p + 28);
121 rec->ex = zip_rd16le(cd + p + 30);
122 rec->cm = zip_rd16le(cd + p + 32);
123 rec->lhdr_off = zip_rd32le(cd + p + 42);
124
125 rec->span = (uint64_t) ZIP_CDFH_BASE + rec->fn + rec->ex + rec->cm;
126 if (p + rec->span > cd_size) {
127 return -1;
128 }
129 return 0;
130}

132/*
133 * WHAT: Build the resolved member `last` for a CDFH record whose name matched.
134 * WHY: A matched entry must reject unsupported kinds, apply the ZIP64 extra
135 * overrides, and resolve the local-header data offset before it can be
136 * served — this is the security-critical leaf of the walk.
137 * HOW: Reject encrypted / data-descriptor / non-store-non-deflate entries,
138 * apply the ZIP64 extra to the size/offset triple, populate `last`, then
139 * resolve the data offset via the kernel. Returns a BRIX_ZIP_* code:
140 * BRIX_ZIP_OK on success, an error code otherwise.
141 */
142static int
143zip_build_member(const brix_zip_arc_t *arc, const unsigned char *base,
144 const brix_zip_cdfh_t *rec, brix_zip_member_t *last)
145{
146 uint64_t uncomp = rec->uncomp, comp = rec->comp, lhdr_off = rec->lhdr_off;
147 int fd = arc->fd;
148 int rc;
149
150 /* Unsupported entry kinds → reject (matches plan W2 edge matrix). */
151 if ((rec->bits & ZIP_GPF_ENCRYPTED) || (rec->bits & ZIP_GPF_DATADESC)) {
152 return BRIX_ZIP_ECORRUPT;
153 }
154 if (rec->method != BRIX_ZIP_METHOD_STORE
155 && rec->method != BRIX_ZIP_METHOD_DEFLATE) {
156 return BRIX_ZIP_ECORRUPT;
157 }
158
159 zip_apply_zip64_extra(base + ZIP_CDFH_BASE + rec->fn, rec->ex,
160 &uncomp, &comp, &lhdr_off);
161
162 memset(last, 0, sizeof(*last));
163 memcpy(last->name, base + ZIP_CDFH_BASE, rec->fn);
164 last->name[rec->fn] = '\0';
165 last->method = rec->method;
166 last->comp_size = comp;
167 last->uncomp_size = uncomp;
168 last->crc32 = rec->crc32;
169
170 rc = zip_resolve_data_off(zip_fd_pread, &fd, (uint64_t) arc->size,
171 lhdr_off, comp, &last->data_off);
172 if (rc != ZIP_K_OK) {
173 return zip_map_kernel_rc(rc);
174 }
175 return BRIX_ZIP_OK;
176}

195/*
196 * WHAT: Locate the central directory, allocate a +1-sentinel buffer, and read
197 * the whole directory into `cd`. On success the caller owns `cd->buf`.
198 * WHY: Isolates the untrusted-size overflow guard and the two-step
199 * locate-then-read from the record walk, keeping each concern testable.
200 * HOW: zip_locate_cd → reject empty directory → overflow-checked malloc →
201 * pread the directory. Returns a BRIX_ZIP_* code; on error `cd->buf`=NULL.
202 */
203static int
204zip_load_cd(int fd, off_t archive_size, size_t cd_max, brix_zip_cd_t *cd)
205{
206 uint64_t cd_off;
207 size_t cd_alloc;
208 int rc;
209
210 cd->buf = NULL;
211 rc = zip_locate_cd(zip_fd_pread, &fd, (uint64_t) archive_size,
212 (uint64_t) cd_max, 0, &cd_off, &cd->size, &cd->nrec);
213 if (rc != ZIP_K_OK) {
214 return zip_map_kernel_rc(rc);
215 }
216 if (cd->size == 0) {
217 return BRIX_ZIP_ECORRUPT; /* empty directory — no member to find */
218 }
219
220 /* cd size is read from the (untrusted) EOCD record; guard against a value
221 * so close to SIZE_MAX that the +1 sentinel byte allocation would wrap. */
222 if (brix_size_add((size_t) cd->size, 1, &cd_alloc) != NGX_OK) {
223 return BRIX_ZIP_ECORRUPT;
224 }
225 cd->buf = malloc(cd_alloc);
226 if (cd->buf == NULL) {
227 return BRIX_ZIP_EIO;
228 }
229 if (pread_full(fd, cd->buf, (size_t) cd->size, (off_t) cd_off) != 0) {
230 free(cd->buf);
231 cd->buf = NULL;
232 return BRIX_ZIP_EIO;
233 }
234 return BRIX_ZIP_OK;
235}

237/*
238 * WHAT: Walk the CDFH records in `cd`, resolving the last record whose name
239 * equals `key` into `last`.
240 * WHY: The last-match-wins scan is the core XrdZip semantic; keeping it in its
241 * own function bounds the caller's complexity and localises the walk.
242 * HOW: For each record: parse+bounds-check, compare the name, and on a match
243 * build the member (may overwrite an earlier match). Returns BRIX_ZIP_OK
244 * with `*found` = whether any match was resolved, or an error code.
245 */
246static int
247zip_scan_directory(const brix_zip_arc_t *arc, const brix_zip_cd_t *cd,
248 const brix_zip_key_t *key, brix_zip_member_t *last,
249 int *found)
250{
251 uint64_t p = 0;
252
253 *found = BRIX_ZIP_NOMEMBER;
254 for (uint64_t i = 0; i < cd->nrec; ++i) {
255 brix_zip_cdfh_t rec = {0};
256 int rc;
257
258 if (zip_parse_cdfh(cd->buf, p, cd->size, &rec) != 0) {
259 return BRIX_ZIP_ECORRUPT;
260 }
261 if (rec.fn == key->len
262 && memcmp(cd->buf + p + ZIP_CDFH_BASE, key->name, rec.fn) == 0) {
263 rc = zip_build_member(arc, cd->buf + p, &rec, last);
264 if (rc != BRIX_ZIP_OK) {
265 return rc;
266 }
267 *found = BRIX_ZIP_OK; /* keep scanning: last match wins */
268 }
269 p += rec.span;
270 }
271 return BRIX_ZIP_OK;
272}

274int
275brix_zip_find_member(int fd, off_t archive_size, const char *member,
276 size_t cd_max, brix_zip_member_t *out)
277{
278 brix_zip_arc_t arc = { fd, archive_size };
279 brix_zip_cd_t cd = {0};
280 brix_zip_key_t key;
281 int rc, found;
282 brix_zip_member_t last = {0};
283
284 if (fd < 0 || member == NULL || out == NULL) {
285 return BRIX_ZIP_ECORRUPT;
286 }
287 key.name = member;
288 key.len = strlen(member);
289 if (key.len == 0 || key.len >= PATH_MAX) {
290 return BRIX_ZIP_ECORRUPT;
291 }
292
293 rc = zip_load_cd(fd, archive_size, cd_max, &cd);
294 if (rc != BRIX_ZIP_OK) {
295 return rc;
296 }
297
298 rc = zip_scan_directory(&arc, &cd, &key, &last, &found);
299 free(cd.buf);
300 if (rc != BRIX_ZIP_OK) {
301 return rc;
302 }
303
304 if (found == BRIX_ZIP_OK) {
305 *out = last;
306 }
307 return found;
308}

310ssize_t
311brix_zip_extract_full(int fd, const brix_zip_member_t *m,
312 unsigned char *out, size_t outcap)
313{
314 if (m == NULL || out == NULL || outcap < m->uncomp_size) {
315 return -1;
316 }
317
318 if (m->method == BRIX_ZIP_METHOD_STORE) {
319 if (pread_full(fd, out, (size_t) m->uncomp_size,
320 (off_t) m->data_off) != 0) {
321 return -1;
322 }
323 return (ssize_t) m->uncomp_size;
324 }
325
326 if (m->method == BRIX_ZIP_METHOD_DEFLATE) {
327 unsigned char *comp;
328 z_stream zs;
329 int zr;
330
331 if (m->comp_size == 0) {
332 return (m->uncomp_size == 0) ? 0 : -1;
333 }
334 /* comp_size comes from the (untrusted) central directory; reject a
335 * value that would cause the +1 guard byte allocation to overflow. */
336 size_t comp_alloc;
337 if (brix_size_add((size_t) m->comp_size, 1, &comp_alloc) != NGX_OK) {
338 return -1;
339 }
340 comp = malloc(comp_alloc);
341 if (comp == NULL) {
342 return -1;
343 }
344 if (pread_full(fd, comp, (size_t) m->comp_size,
345 (off_t) m->data_off) != 0) {
346 free(comp);
347 return -1;
348 }
349
350 memset(&zs, 0, sizeof(zs));
351 if (inflateInit2(&zs, -15) != Z_OK) { /* raw deflate */
352 free(comp);
353 return -1;
354 }
355 zs.next_in = comp;
356 zs.avail_in = (uInt) m->comp_size;
357 zs.next_out = out;
358 zs.avail_out = (uInt) m->uncomp_size;
359 zr = inflate(&zs, Z_FINISH);
360 {
361 size_t produced = (size_t) m->uncomp_size - zs.avail_out;
362 inflateEnd(&zs);
363 free(comp);
364 if (zr != Z_STREAM_END || produced != m->uncomp_size) {
365 return -1;
366 }
367 return (ssize_t) produced;
368 }
369 }
370
371 return -1; /* unsupported method (find_member already rejects these) */
372}
