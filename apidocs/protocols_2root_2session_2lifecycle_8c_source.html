19/* kXR_endsess - client wants to end the session gracefully */
20/* Handle kXR_endsess — the client's explicit request to gracefully terminate
21 * its session. */
22ngx_int_t
23brix_handle_endsess(brix_ctx_t *ctx, ngx_connection_t *c)
24{
25 ClientEndsessRequest *req = (ClientEndsessRequest *) ctx->recv.hdr_buf;
26
27 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, c->log, 0,
28 "brix: kXR_endsess received");
29
30 /*
31 * kXR_endsess names the session to terminate (req.sessid) — which is NOT
32 * necessarily the session of the connection it arrives on. The official
33 * client's reconnect recovery, after a dropped link, opens a NEW connection,
34 * logs in afresh, and THEN sends kXR_endsess for its PREVIOUS (now-dead)
35 * session to release it. Per the protocol we must terminate only the NAMED
36 * session; tearing down or de-authenticating THIS freshly-authenticated
37 * connection would make the client's very next request (its recovery
38 * kXR_open) fail with kXR_NotAuthorized — which is exactly what broke
39 * official-client transfer recovery against this server on a lossy link.
40 *
41 * So: if the request names a DIFFERENT session, just unregister that session
42 * (idempotent cross-worker cleanup) and leave this connection's auth state
43 * and open handles untouched. Only an endsess for THIS connection's own
44 * session performs the full teardown + auth clear below.
45 */
46 if (ngx_memcmp(req->sessid, ctx->login.sessid, BRIX_SESSION_ID_LEN) != 0) {
47 brix_session_unregister(req->sessid);
48 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, c->log, 0,
49 "brix: kXR_endsess for a different session — "
50 "released it, this connection stays authenticated");
51 return brix_send_ok(ctx, c, NULL, 0);
52 }
53
54 /*
55 * Proxy mode: fire-and-forget kXR_endsess to the upstream before tearing
56 * down the upstream connection. This allows the upstream to close the
57 * session gracefully (record final stats, release slots) rather than
58 * treating the subsequent TCP close as an abrupt disconnect.
59 * We do not wait for the upstream response — the cleanup that follows is
60 * unconditional and correct regardless of whether the send succeeds.
61 */
62 if (ctx->proxy != NULL) {
63 brix_proxy_ctx_t *proxy = (brix_proxy_ctx_t *) ctx->proxy;
64
65 proxy->no_pool = 1;
66
67 if (proxy->state == XRD_PX_IDLE && proxy->conn != NULL) {
68 u_char frame[XRD_REQUEST_HDR_LEN + 16]; /* hdr + 16-byte sessid */
69 ngx_memzero(frame, sizeof(frame));
70 {
71 uint16_t rid = htons(kXR_endsess);
72 ngx_memcpy(frame + 2, &rid, 2);
73 }
74 proxy->conn->send(proxy->conn, frame, sizeof(frame));
75 }
76 }
77
78 /*
79 * Mirror disconnect cleanup immediately so metrics and open-handle state
80 * are settled before the final response is queued.
81 * This keeps explicit end-of-session requests aligned with the same cleanup
82 * bookkeeping used for timeouts and transport-level disconnects.
83 */
84 brix_on_disconnect(ctx, c);
85 brix_close_all_files(ctx);
86
87 /*
88 * SECURITY: clear session-level auth flags so the dispatcher rejects any
89 * further requests that the client attempts on this TCP connection.
90 * Without this, a client could re-open files and read/write them after
91 * kXR_endsess, bypassing the session-end semantics (and, in GSI deployments,
92 * bypassing proxy-certificate expiry that triggered the endsess).
93 */
94 ctx->login.logged_in = 0;
95 ctx->login.auth_done = 0;
96
97 return brix_send_ok(ctx, c, NULL, 0);
98}
