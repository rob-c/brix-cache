34/*
35 * D-1 — brix_min_sec_level: session-posture floor enforced post-handshake.
36 * A separate axis from brix_security_level (which governs kXR_sigver request
37 * SIGNING): this governs the negotiated SESSION's transport/identity posture so
38 * a client cannot walk the connection below the operator's floor.
39 * none (0) — no floor (default).
40 * compat (1) — the session transport must be TLS-encrypted; a client that
41 * finished login/auth in cleartext is refused every data op.
42 * intense (2) — compat + a non-anonymous authenticated identity (an auth=none
43 * listener has no identity to present, so it is below the floor).
44 * Pairs with A-1 so neither the client nor the upstream leg can be downgraded.
45 */
46#define BRIX_MIN_SEC_NONE 0
47#define BRIX_MIN_SEC_COMPAT 1
48#define BRIX_MIN_SEC_INTENSE 2
