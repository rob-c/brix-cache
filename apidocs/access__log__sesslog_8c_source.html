2 * WHAT: Mirror selected legacy access-log records into the sesslog lifecycle
3 * (AUTH events and namespace-verb ATTEMPT/RESULT pairs). Split out of

33/*
34 * WHAT: Convert the legacy access-log auth detail into a sesslog auth method.
35 * WHY: Existing root auth handlers already log AUTH records; piggybacking those
36 * calls keeps method-specific auth code untouched.
37 * HOW: Prefer the detail token when it names a method, otherwise derive from
38 * the authenticated ctx/config state.
39 */
40static brix_sess_am_t
41brix_access_sess_auth_method(brix_ctx_t *ctx,
42 ngx_stream_brix_srv_conf_t *conf, const char *detail)
43{
44 if (brix_access_streq(detail, "gsi")) {
45 return BRIX_SESS_AM_GSI;
46 }
47 if (brix_access_streq(detail, "ztn") || brix_access_streq(detail, "token")) {
48 return BRIX_SESS_AM_TOKEN;
49 }
50 if (brix_access_streq(detail, "sss")) {
51 return BRIX_SESS_AM_SSS;
52 }
53 if (brix_access_streq(detail, "krb5")) {
54 return BRIX_SESS_AM_KRB5;
55 }
56 if (brix_access_streq(detail, "pwd")) {
57 return BRIX_SESS_AM_PWD;
58 }
59 if (brix_access_streq(detail, "unix")) {
60 return BRIX_SESS_AM_UNIX;
61 }
62 if (brix_access_streq(detail, "host")) {
63 return BRIX_SESS_AM_HOST;
64 }
65 if (ctx != NULL && ctx->token.auth) {
66 return BRIX_SESS_AM_TOKEN;
67 }
68 return brix_sess_am_from_stream_auth(conf != NULL ? conf->auth
69 : BRIX_AUTH_NONE);
70}

72/*
73 * WHAT: Collapse method-specific auth messages into stable sesslog err tokens.
74 * WHY: AUTH failure text is deliberately human-readable and inconsistent; the
75 * session grammar requires a small closed token set.
76 * HOW: Preserve high-signal substrings where available and fall back to kXR
77 * status mapping.
78 */
79static const char *
80brix_access_sess_auth_err(uint16_t errcode, const char *errmsg, char *scratch,
81 size_t scratch_size)
82{
83 if (brix_access_contains(errmsg, "scope")) {
84 return "scope";
85 }
86 if (brix_access_contains(errmsg, "expired")) {
87 return "expired-cert";
88 }
89 if (brix_access_contains(errmsg, "signature")
90 || brix_access_contains(errmsg, "validation")
91 || brix_access_contains(errmsg, "verification")
92 || brix_access_contains(errmsg, "decrypt"))
93 {
94 return "bad-signature";
95 }
96 if (brix_access_contains(errmsg, "rate limit")) {
97 return "busy";
98 }
99 if (brix_access_contains(errmsg, "not authorized")
100 || brix_access_contains(errmsg, "denied"))
101 {
102 return "permission";
103 }
104
105 return brix_sesslog_err_from_kxr((int) errcode, scratch, scratch_size);
106}

109 * WHAT: Static verb→sesslog-mode mapping for the fixed-mode namespace verbs.
110 * WHY: The mode assignment for these verbs is a pure lookup; expressing it as a
111 * data table (rather than an if/else ladder) keeps the classifier flat and makes
112 * adding a verb a one-row edit (coding-standards §8.6, table-driven dispatch).
113 * HOW: One row per verb naming its sesslog mode. OPEN is deliberately absent —
114 * its mode depends on the detail token and is resolved separately.
115 */

141/*
142 * WHAT: Resolve an OPEN verb to its sesslog mode from the detail token.
143 * WHY: OPEN is the one namespace verb whose direction (read vs write) is not
144 * fixed by the verb name — write/staging/tpc-pull opens count as WRITE, every
145 * other open as READ.
146 * HOW: Inspect the detail substring; return WRITE for any write-shaped open,
147 * READ otherwise. Behavior identical to the former inline OPEN branch.
148 */
149static brix_sess_mode_t
150brix_access_open_mode(const char *detail)
151{
152 if (brix_access_contains(detail, "wr")
153 || brix_access_contains(detail, "staging")
154 || brix_access_contains(detail, "tpc-pull"))
155 {
156 return BRIX_SESS_MODE_WRITE;
157 }
158 return BRIX_SESS_MODE_READ;
159}

161/*
162 * WHAT: Map root access-log verbs to sesslog access modes.
163 * WHY: The legacy access logger is already present on namespace-operation exits;
164 * this table lets it produce uniform ATTEMPT/RESULT pairs without touching every
165 * handler in the root plane.
166 * HOW: OPEN resolves via detail (brix_access_open_mode); every other namespace
167 * verb is a row lookup in brix_access_verb_modes. Lifecycle and pure I/O verbs
168 * match nothing and return 0 (they aggregate into AUTH, XFER, or END).
169 */
170static ngx_flag_t
171brix_access_sess_mode(const char *verb, const char *detail,
172 brix_sess_mode_t *mode)
173{
174 size_t i;
175
176 if (verb == NULL || mode == NULL) {
177 return 0;
178 }
179
180 if (brix_access_streq(verb, "OPEN")) {
181 *mode = brix_access_open_mode(detail);
182 return 1;
183 }
184
185 for (i = 0; i < sizeof(brix_access_verb_modes)
186 / sizeof(brix_access_verb_modes[0]); i++)
187 {
188 if (brix_access_streq(verb, brix_access_verb_modes[i].verb)) {
189 *mode = brix_access_verb_modes[i].mode;
190 return 1;
191 }
192 }
193
194 return 0;
195}

197/*
198 * WHAT: Emit the sesslog AUTH event for an AUTH access-log record.
199 * WHY: Keeps the AUTH-specific method/identity/err derivation out of the mirror
200 * dispatcher so each stays single-purpose.
201 * HOW: On success pass DN/VO (or "-"); on failure derive a stable err token.
202 */
203static void
204brix_access_sesslog_auth(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
205 const brix_access_event_t *ev)
206{
207 char errscratch[64];
208 const char *err;
209
210 err = ev->xrd_ok ? NULL
211 : brix_access_sess_auth_err(ev->errcode, ev->errmsg,
212 errscratch, sizeof(errscratch));
213 brix_sess_auth(ctx->sess, ev->xrd_ok,
214 brix_access_sess_auth_method(ctx, conf, ev->detail),
215 ev->xrd_ok && ctx->login.dn[0] != '\0'
216 ? ctx->login.dn : "-",
217 ev->xrd_ok && ctx->login.primary_vo[0] != '\0'
218 ? ctx->login.primary_vo : "-",
219 err);
220}

222/*
223 * WHAT: Emit the sesslog ATTEMPT/RESULT pair for a namespace access-log record.
224 * WHY: Isolates the namespace-verb path (mode classification + attempt/result)
225 * from the AUTH path in the dispatcher.
226 * HOW: Classify the verb into a mode; ignore verbs with no mode; otherwise emit
227 * ATTEMPT then RESULT with a derived err token on failure.
228 */
229static void
230brix_access_sesslog_namespace(brix_ctx_t *ctx, const brix_access_event_t *ev)
231{
232 char errscratch[64];
233 brix_sess_mode_t mode = 0;
234 const char *err;
235 const char *path;
236
237 if (!brix_access_sess_mode(ev->verb, ev->detail, &mode)) {
238 return;
239 }
240
241 err = ev->xrd_ok ? NULL
242 : brix_sesslog_err_from_kxr((int) ev->errcode, errscratch,
243 sizeof(errscratch));
244 path = ev->path != NULL ? ev->path : "-";
245 brix_sess_attempt(ctx->sess, path, mode);
246 brix_sess_result(ctx->sess, ev->xrd_ok, path, mode, err);
247}

249/*
250 * WHAT: Mirror selected legacy access-log calls into the sesslog lifecycle.
251 * WHY: Root handlers already have well-audited success/error exits; using this
252 * chokepoint gives root:// ATTEMPT/RESULT and AUTH coverage with minimal risk.
253 * HOW: AUTH emits auth events; namespace verbs emit immediate ATTEMPT followed
254 * by RESULT; pure data I/O is intentionally ignored and summarized by XFER.
255 */
256void
257brix_access_maybe_sesslog(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
258 const brix_access_event_t *ev)
259{
260 if (ctx == NULL || ctx->sess == NULL || ev->verb == NULL) {
261 return;
262 }
263
264 if (brix_access_streq(ev->verb, "AUTH")) {
265 brix_access_sesslog_auth(ctx, conf, ev);
266 return;
267 }
268
269 brix_access_sesslog_namespace(ctx, ev);
270}
