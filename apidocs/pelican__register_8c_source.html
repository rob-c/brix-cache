1/*
2 * pelican_register.c — advertise this node as a discoverable Pelican cache.
3 *
4 * Wire details (from PelicanPlatform/pelican):
5 * - POST <director>/api/v1.0/director/registerCache, body = OriginAdvertiseV2
6 * JSON, header "Authorization: Bearer <advertise-JWT>".
7 * - OriginAdvertiseV2 (server_structs/director.go): ServerBaseAd {name,
8 * startTime, instanceID (hex UUID), generationID (monotonic), version,
9 * expiry (RFC3339)} + serverId, registry-prefix "/caches/<sitename>",
10 * data-url (required), web-url, capabilities {PublicRead,Read,Write,Listing,
11 * FallBackRead,Copies}, namespaces[] {Caps,path,token-generation,
12 * token-issuer}, storageType, directorTest, status, now.
13 * - Advertise JWT (origin/advertise.go GetAdTokCfg): iss = server issuer,
14 * sub = this cache's data URL, aud = the Director, scope "pelican.advertise",
15 * lifetime 1 minute, ES256.
16 *
17 * The per-worker timer offloads each advertisement to the cache thread pool so
18 * the blocking discovery+POST never runs on the event loop.
19 */

54static void
55brix_pelican_hex_random(char *out, size_t nbytes)
56{
57 unsigned char raw[20];
58 static const char hexd[] = "0123456789abcdef";
59 size_t i;
60
61 if (nbytes > sizeof(raw)) {
62 nbytes = sizeof(raw);
63 }
64 if (RAND_bytes(raw, (int) nbytes) != 1) {
65 for (i = 0; i < nbytes; i++) {
66 raw[i] = (unsigned char) (ngx_random() & 0xff);
67 }
68 }
69 for (i = 0; i < nbytes; i++) {
70 out[i * 2] = hexd[raw[i] >> 4];
71 out[i * 2 + 1] = hexd[raw[i] & 0x0f];
72 }
73 out[nbytes * 2] = '\0';
74}

100/* OriginAdvertiseV2 document → compact JSON string (caller free()s) */static char *
101brix_pelican_build_ad(ngx_stream_brix_srv_conf_t *conf, time_t now)
102{
103 json_t *ad;
104 json_t *nslist;
105 char expiry[40];
106 char nowstr[40];
107 char regpfx[256];
108 const char *site;
109 char *out;
110
111 ad = json_object();
112 if (ad == NULL) {
113 return NULL;
114 }
115
116 site = brix_pelican_cstr(&conf->advertise.sitename, "nginx-xrootd-cache");
117 brix_pelican_rfc3339(now + (time_t) (conf->advertise.interval / 1000)
118 + 30, expiry, sizeof(expiry));
119 brix_pelican_rfc3339(now, nowstr, sizeof(nowstr));
120 (void) snprintf(regpfx, sizeof(regpfx), "/caches/%s", site);
121
122 /* ServerBaseAd (embedded — fields are flattened into the parent object). */
123 json_object_set_new(ad, "name", json_string(site));
124 json_object_set_new(ad, "startTime", json_integer((json_int_t) now));
125 json_object_set_new(ad, "instanceID",
126 json_string(conf->advertise.instance));
127 json_object_set_new(ad, "generationID",
128 json_integer((json_int_t) conf->advertise.gen));
129 json_object_set_new(ad, "version",
130 json_string(BRIX_SERVER_NAME " " BRIX_SERVER_VERSION));
131 json_object_set_new(ad, "expiry", json_string(expiry));
132
133 json_object_set_new(ad, "serverId", json_string(site));
134 json_object_set_new(ad, "registry-prefix", json_string(regpfx));
135 json_object_set_new(ad, "data-url",
136 json_string(brix_pelican_cstr(&conf->advertise.data_url, "")));
137 if (conf->advertise.web_url.len > 0) {
138 json_object_set_new(ad, "web-url",
139 json_string((char *) conf->advertise.web_url.data));
140 }
141 json_object_set_new(ad, "capabilities", brix_pelican_caps_json());
142
143 /* namespaces[]: the configured prefixes, or "/" (cache everything). */
144 nslist = json_array();
145 if (nslist != NULL) {
146 ngx_uint_t i, n = (conf->advertise.ns != NULL)
147 ? conf->advertise.ns->nelts : 0;
148 if (n == 0) {
149 json_t *ns = json_object();
150 json_object_set_new(ns, "Caps", brix_pelican_caps_json());
151 json_object_set_new(ns, "path", json_string("/"));
152 json_object_set_new(ns, "token-generation", json_array());
153 json_object_set_new(ns, "token-issuer", json_array());
154 json_array_append_new(nslist, ns);
155 } else {
156 ngx_str_t *pfx = conf->advertise.ns->elts;
157 for (i = 0; i < n; i++) {
158 json_t *ns = json_object();
159 json_object_set_new(ns, "Caps", brix_pelican_caps_json());
160 json_object_set_new(ns, "path",
161 json_stringn((char *) pfx[i].data, pfx[i].len));
162 json_object_set_new(ns, "token-generation", json_array());
163 json_object_set_new(ns, "token-issuer", json_array());
164 json_array_append_new(nslist, ns);
165 }
166 }
167 }
168 json_object_set_new(ad, "namespaces", nslist);
169
170 json_object_set_new(ad, "storageType", json_string("posix"));
171 json_object_set_new(ad, "directorTest", json_true()); /* disable test */
172 json_object_set_new(ad, "status", json_string("ok"));
173 json_object_set_new(ad, "now", json_string(nowstr));
174
175 out = json_dumps(ad, JSON_COMPACT);
176 json_decref(ad);
177 return out; /* malloc'd; caller free()s */
178}

181/* advertise JWT (ES256, scope pelican.advertise) */static ngx_int_t
182brix_pelican_build_jwt(ngx_stream_brix_srv_conf_t *conf,
183 const char *director, time_t now, char *out, size_t outsz)
184{
185 json_t *payload;
186 char *pstr;
187 char jti[48];
188 time_t exp = now + 60; /* 1-minute lifetime (MinFedTokenTicker) */
189 ngx_int_t rc;
190
191 brix_pelican_hex_random(jti, 16);
192
193 payload = json_object();
194 if (payload == NULL) {
195 return NGX_ERROR;
196 }
197 json_object_set_new(payload, "iss",
198 json_string(brix_pelican_cstr(&conf->advertise.issuer_url,
199 brix_pelican_cstr(&conf->advertise.data_url, ""))));
200 json_object_set_new(payload, "sub",
201 json_string(brix_pelican_cstr(&conf->advertise.data_url, "")));
202 json_object_set_new(payload, "aud", json_string(director));
203 json_object_set_new(payload, "scope", json_string("pelican.advertise"));
204 json_object_set_new(payload, "wlcg.ver", json_string("1.0"));
205 json_object_set_new(payload, "jti", json_string(jti));
206 json_object_set_new(payload, "iat", json_integer((json_int_t) now));
207 json_object_set_new(payload, "nbf", json_integer((json_int_t) now));
208 json_object_set_new(payload, "exp", json_integer((json_int_t) exp));
209
210 pstr = json_dumps(payload, JSON_COMPACT);
211 json_decref(payload);
212 if (pstr == NULL) {
213 return NGX_ERROR;
214 }
215
216 rc = brix_jwt_sign_es256((EVP_PKEY *) conf->advertise.key_pkey,
217 "{\"alg\":\"ES256\",\"typ\":\"JWT\"}", pstr,
218 out, outsz);
219 free(pstr);
220 return rc;
221}

249static ngx_int_t
250brix_pelican_discover_cfg(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
251 char *out, size_t outsz)
252{
253 CURL *curl;
254 CURLcode res;
255 brix_pelican_mem2_t mem;
256 char url[512];
257 char doc[64 * 1024];
258 json_t *root, *ep;
259 json_error_t jerr;
260 const char *director;
261 long code = 0;
262 int n;
263
264 n = snprintf(url, sizeof(url),
265 "https://%s:%u/.well-known/pelican-configuration",
266 (char *) conf->cache_origin_host.data,
267 (unsigned) conf->cache_origin_port);
268 if (n < 0 || (size_t) n >= sizeof(url)) {
269 return NGX_ERROR;
270 }
271
272 curl = curl_easy_init();
273 if (curl == NULL) {
274 return NGX_ERROR;
275 }
276 mem.buf = doc; mem.len = 0; mem.cap = sizeof(doc) - 1;
277 curl_easy_setopt(curl, CURLOPT_URL, url);
278 curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, brix_pelican_mem2_cb);
279 curl_easy_setopt(curl, CURLOPT_WRITEDATA, &mem);
280 curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, 1L);
281 curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
282 curl_easy_setopt(curl, CURLOPT_FAILONERROR, 1L);
283 curl_easy_setopt(curl, CURLOPT_CONNECTTIMEOUT, (long) BRIX_CACHE_IO_TIMEOUT);
284 curl_easy_setopt(curl, CURLOPT_TIMEOUT, (long) BRIX_CACHE_IO_TIMEOUT);
285 brix_pelican_set_ca(conf, curl);
286
287 res = curl_easy_perform(curl);
288 if (res == CURLE_OK) {
289 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &code);
290 }
291 curl_easy_cleanup(curl);
292
293 if (res != CURLE_OK || code < 200 || code >= 300) {
294 ngx_log_error(NGX_LOG_ERR, log, 0,
295 "brix: pelican advertise: discovery failed (%s, http %ld)",
296 curl_easy_strerror(res), code);
297 return NGX_ERROR;
298 }
299
300 doc[mem.len] = '\0';
301 root = json_loads(doc, JSON_REJECT_DUPLICATES, &jerr);
302 if (root == NULL) {
303 return NGX_ERROR;
304 }
305 ep = json_object_get(root, "director_endpoint");
306 director = json_is_string(ep) ? json_string_value(ep) : NULL;
307 if (director == NULL || ngx_strlen(director) >= outsz) {
308 json_decref(root);
309 return NGX_ERROR;
310 }
311 ngx_memcpy(out, director, ngx_strlen(director) + 1);
312 json_decref(root);
313 return NGX_OK;
314}

317/* POST the advertisement */static ngx_int_t
318brix_pelican_post(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log,
319 const char *director, const char *body, const char *jwt)
320{
321 CURL *curl;
322 CURLcode res;
323 struct curl_slist *hdrs = NULL;
324 char url[640];
325 char authz[2200];
326 long code = 0;
327 int n;
328
329 n = snprintf(url, sizeof(url),
330 "%s/api/v1.0/director/registerCache", director);
331 if (n < 0 || (size_t) n >= sizeof(url)) {
332 return NGX_ERROR;
333 }
334
335 curl = curl_easy_init();
336 if (curl == NULL) {
337 return NGX_ERROR;
338 }
339
340 hdrs = curl_slist_append(hdrs, "Content-Type: application/json");
341 n = snprintf(authz, sizeof(authz), "Authorization: Bearer %s", jwt);
342 if (n > 0 && (size_t) n < sizeof(authz)) {
343 hdrs = curl_slist_append(hdrs, authz);
344 }
345
346 curl_easy_setopt(curl, CURLOPT_URL, url);
347 curl_easy_setopt(curl, CURLOPT_POST, 1L);
348 curl_easy_setopt(curl, CURLOPT_POSTFIELDS, body);
349 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, hdrs);
350 curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, 1L);
351 curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
352 curl_easy_setopt(curl, CURLOPT_CONNECTTIMEOUT, (long) BRIX_CACHE_IO_TIMEOUT);
353 curl_easy_setopt(curl, CURLOPT_TIMEOUT, (long) BRIX_CACHE_IO_TIMEOUT);
354 brix_pelican_set_ca(conf, curl);
355
356 res = curl_easy_perform(curl);
357 if (res == CURLE_OK) {
358 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &code);
359 }
360 if (hdrs != NULL) {
361 curl_slist_free_all(hdrs);
362 }
363 curl_easy_cleanup(curl);
364
365 if (res != CURLE_OK || code < 200 || code >= 300) {
366 ngx_log_error(NGX_LOG_WARN, log, 0,
367 "brix: pelican advertise: registerCache POST failed "
368 "(%s, http %ld) to %s", curl_easy_strerror(res), code, url);
369 return NGX_ERROR;
370 }
371 ngx_log_error(NGX_LOG_INFO, log, 0,
372 "brix: pelican advertise: registered cache with director (http %ld)",
373 code);
374 return NGX_OK;
375}

378ngx_int_t
379brix_cache_pelican_advertise_once(ngx_stream_brix_srv_conf_t *conf,
380 ngx_log_t *log)
381{
382 char director[512];
383 char jwt[3072];
384 char *body;
385 time_t now = time(NULL);
386 ngx_int_t rc;
387
388 if (conf->advertise.key_pkey == NULL
389 || conf->advertise.data_url.len == 0)
390 {
391 return NGX_ERROR;
392 }
393
394 if (brix_pelican_discover_cfg(conf, log, director, sizeof(director))
395 != NGX_OK)
396 {
397 return NGX_ERROR;
398 }
399
400 conf->advertise.gen++;
401 body = brix_pelican_build_ad(conf, now);
402 if (body == NULL) {
403 return NGX_ERROR;
404 }
405
406 if (brix_pelican_build_jwt(conf, director, now, jwt, sizeof(jwt))
407 != NGX_OK)
408 {
409 free(body);
410 ngx_log_error(NGX_LOG_ERR, log, 0,
411 "brix: pelican advertise: JWT signing failed");
412 return NGX_ERROR;
413 }
414
415 rc = brix_pelican_post(conf, log, director, body, jwt);
416 free(body);
417 return rc;
418}

436static void
437brix_pelican_adv_timer(ngx_event_t *ev)
438{
439 brix_pelican_adv_t *a = ev->data;
440 ngx_stream_brix_srv_conf_t *conf = a->conf;
441
442 /* Re-arm first so a slow advertisement never stops the cadence. */
443 ngx_add_timer(ev, conf->advertise.interval);
444
445 if (a->in_flight || a->task == NULL
446 || conf->common.thread_pool == NULL)
447 {
448 /* No pool or a still-running advertisement: run inline only when there
449 * is no thread pool (best-effort; bounded by the curl timeouts). */
450 if (conf->common.thread_pool == NULL && !a->in_flight) {
451 (void) brix_cache_pelican_advertise_once(conf, ev->log);
452 }
453 return;
454 }
455
456 a->in_flight = 1;
457 if (ngx_thread_task_post(conf->common.thread_pool, a->task) != NGX_OK) {
458 a->in_flight = 0;
459 }
460}

463void
464brix_cache_pelican_schedule_advertise(ngx_cycle_t *cycle,
465 ngx_stream_brix_srv_conf_t *conf)
466{
467 ngx_event_t *ev;
468 ngx_thread_task_t *task;
469 brix_pelican_adv_t *a;
470
471 if (!conf->advertise.enable || conf->advertise.enable == NGX_CONF_UNSET
472 || conf->advertise.key.len == 0
473 || conf->advertise.data_url.len == 0
474 || conf->cache_origin_host.len == 0)
475 {
476 return;
477 }
478
479 /* Load the EC signing key once per worker. */
480 conf->advertise.key_pkey =
481 brix_jwt_load_ec_key((char *) conf->advertise.key.data);
482 if (conf->advertise.key_pkey == NULL) {
483 ngx_log_error(NGX_LOG_ERR, cycle->log, 0,
484 "brix: pelican advertise disabled — cannot load EC key \"%s\"",
485 conf->advertise.key.data);
486 return;
487 }
488
489 if (conf->advertise.instance[0] == '\0') {
490 brix_pelican_hex_random(conf->advertise.instance, 16);
491 }
492
493 task = ngx_thread_task_alloc(cycle->pool, sizeof(brix_pelican_adv_t));
494 ev = ngx_pcalloc(cycle->pool, sizeof(*ev));
495 if (task == NULL || ev == NULL) {
496 return;
497 }
498 a = task->ctx;
499 a->conf = conf;
500 a->log = cycle->log;
501 a->task = task;
502 a->in_flight = 0;
503 task->handler = brix_pelican_adv_thread;
504 task->event.handler = brix_pelican_adv_done;
505 task->event.data = a;
506
507 ev->handler = brix_pelican_adv_timer;
508 ev->data = a;
509 ev->log = cycle->log;
510 conf->advertise.timer = ev;
511
512 /* First advertisement shortly after startup, then every interval. */
513 ngx_add_timer(ev, 2000);
514
515 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
516 "brix: pelican cache advertise started — federation=%s interval=%Mms "
517 "data-url=%V", conf->cache_origin_host.data,
518 conf->advertise.interval, &conf->advertise.data_url);
519}
