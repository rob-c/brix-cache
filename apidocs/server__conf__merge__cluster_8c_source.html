2 * server_conf_merge_cluster.c — server-block merge for the third-party-copy and
3 * cluster/session configuration areas.
4 *
5 * WHAT: Owns brix_merge_srv_tpc() (TPC allowances, key TTL, transfer caps +
6 * reaper age, SSI/CNS, outbound credentials) and brix_merge_srv_cluster()
7 * (manager/redirector mode, staged uploads, pipeline/registry/session
8 * sizing, health checks, traffic mirror, CMS client, listen port,
9 * checksum-scan limits, VO/group/manager-map rule arrays + redirector
10 * inheritance), together with the file-local per-concern helpers.
11 * WHY: Split (phase-79 file-size cap) out of the former 1249-line
12 * server_conf.c. Both entry points are non-static (declared in
13 * server_conf_internal.h) because the top-level orchestrator in
14 * server_conf.c calls them in linear order; every sub-helper stays
15 * file-local.
16 * HOW: Standard nginx parent->child inheritance via ngx_conf_merge_* and the
17 * BRIX_MERGE_* macros, one helper per concern group, invoked in the
18 * original order so cross-group derivations (pipeline-depth clamp, CMS
19 * read-timeout from the heartbeat interval) still observe their
20 * already-merged inputs. Config-time side effects (CNS collect flag, TPC
21 * registry reaper age) run once inputs settle, exactly as before. No
22 * behaviour change from the split.
23 */

34/* Third-party copy (TPC): local/private allowances, key TTL, transfer caps and
35 * the abandoned-slot reaper age, and the outbound OAuth2/bearer credentials. */
36void
37brix_merge_srv_tpc(ngx_stream_brix_srv_conf_t *conf,
38 ngx_stream_brix_srv_conf_t *prev)
39{
40 ngx_conf_merge_value(conf->tpc_allow_local, prev->tpc_allow_local, 0);
41 ngx_conf_merge_value(conf->tpc_allow_private, prev->tpc_allow_private, 1);
42 ngx_conf_merge_value(conf->ssi_enable, prev->ssi_enable, 0);
43 ngx_conf_merge_value(conf->ssi_cta_enable, prev->ssi_cta_enable, 0);
44 /* defaults mirror BRIX_SSI_MAX_INFLIGHT (8) and the 1 MiB req/resp caps. */
45 ngx_conf_merge_uint_value(conf->ssi_max_inflight, prev->ssi_max_inflight, 8);
46 ngx_conf_merge_size_value(conf->ssi_request_max, prev->ssi_request_max, 1u << 20);
47 ngx_conf_merge_size_value(conf->ssi_response_max, prev->ssi_response_max, 1u << 20);
48 ngx_conf_merge_str_value(conf->ssi_cta_journal, prev->ssi_cta_journal, "");
49 ngx_conf_merge_uint_value(conf->ssi_cta_executor, prev->ssi_cta_executor, 0);
50 ngx_conf_merge_uint_value(conf->cns_mode, prev->cns_mode, BRIX_CNS_OFF);
51 if (conf->cns_mode == BRIX_CNS_COLLECT) {
52 brix_cns_set_collect(1); /* §6: this node maintains the CNS inventory */
53 }
54 ngx_conf_merge_value(conf->tpc_outbound_tls, prev->tpc_outbound_tls, 0);
55 ngx_conf_merge_value(conf->tpc_delegate, prev->tpc_delegate, 0);
56 /* Phase-70/opportunistic: passthrough of the client's own inbound bearer JWT
57 * to the TPC source is ON by default so token-authenticated pulls forward the
58 * end-user identity without any per-server opt-in. The default path is
59 * OPPORTUNISTIC (token_mode "passthrough-opt", set in tpc_init_dst_file): an
60 * inbound token is used when present, but its ABSENCE falls back to GSI proxy
61 * delegation / static bearer file / anonymous exactly as before this default
62 * flip — never a new denial. Only an explicit client tpc.token_mode=passthrough
63 * stays STRICT/fail-closed. Operators can still set the directive to `off`. */
64 ngx_conf_merge_value(conf->tpc_outbound_passthrough,
65 prev->tpc_outbound_passthrough, 1);
66 ngx_conf_merge_msec_value(conf->tpc_key_ttl_ms, prev->tpc_key_ttl_ms,
67 BRIX_TPC_KEY_TTL_MS);
68 /* Phase 51 (B2): default native-TPC absolute wall-clock cap to a generous
69 * 24h so a wedged transfer cannot pin a thread-pool worker forever; 0 still
70 * means unlimited (back-compat). Progress-based stall detection (the curl
71 * low-speed bounds, webdav/tpc_config.c) is the primary guard — this is only
72 * the absolute backstop, large enough never to clip a real transfer. */
73 ngx_conf_merge_uint_value(conf->tpc_max_transfer_secs,
74 prev->tpc_max_transfer_secs, 86400);
75 /* Hostile-network completion/integrity gates for the native TPC pull, both
76 * default off (a size mismatch always fails regardless; these only govern the
77 * "no size" and "verify content checksum" postures). */
78 ngx_conf_merge_value(conf->tpc_require_source_size,
79 prev->tpc_require_source_size, 0);
80 ngx_conf_merge_value(conf->tpc_verify_checksum,
81 prev->tpc_verify_checksum, 0);
82 ngx_conf_merge_value(conf->tpc_transfer_max_age,
83 prev->tpc_transfer_max_age, 0);
84 /* Phase 39 (WS5): publish the abandoned-slot reaper age to the shared TPC
85 * registry (config-time, before fork). Guarded so a 0-default block does not
86 * disable a sibling block that enabled it. */
87 if (conf->tpc_transfer_max_age > 0) {
88 brix_tpc_registry_set_max_age((time_t) conf->tpc_transfer_max_age);
89 }
90 ngx_conf_merge_str_value(conf->tpc_outbound_bearer_file,
91 prev->tpc_outbound_bearer_file, "");
92 ngx_conf_merge_str_value(conf->tpc_outbound_token_endpoint,
93 prev->tpc_outbound_token_endpoint, "");
94 ngx_conf_merge_str_value(conf->tpc_outbound_client_id,
95 prev->tpc_outbound_client_id, "");
96 ngx_conf_merge_str_value(conf->tpc_outbound_client_secret,
97 prev->tpc_outbound_client_secret, "");
98 ngx_conf_merge_str_value(conf->tpc_outbound_scope,
99 prev->tpc_outbound_scope, "storage.read");

102/*
103 * WHAT: merge the node sizing group — manager mode, staged/resumable uploads,
104 * and the pipeline/registry/session/redirector-cache slot counts.
105 * WHY: the pipeline-window clamp is its own decision; isolating it keeps the
106 * cluster orchestrator flat.
107 * HOW: standard child<-parent inheritance, then clamp pipeline_depth into
108 * [MIN, MAX] (the recv-loop ring arithmetic needs a positive bounded
109 * modulus).
110 */
111static void
112brix_merge_srv_cluster_sizing(ngx_stream_brix_srv_conf_t *conf,
113 ngx_stream_brix_srv_conf_t *prev)
114{
115 ngx_conf_merge_value(conf->manager_mode, prev->manager_mode, 0);
116 brix_node_caps_conf_merge(&conf->caps, &prev->caps);
117 /* Uploads are staged + resumable by DEFAULT (atomic commit-on-close, resume
118 * across a restart). Set brix_upload_resume off to opt out. */
119 ngx_conf_merge_value(conf->upload_resume, prev->upload_resume, 1);
120 ngx_conf_merge_str_value(conf->upload_stage_dir, prev->upload_stage_dir, "");
121 ngx_conf_merge_uint_value(conf->pipeline_depth, prev->pipeline_depth,
122 BRIX_PIPELINE_DEPTH_DEFAULT);
123 /* Clamp the in-flight pipeline window to a sane range: >=1 (the recv loop and
124 * ring arithmetic require a positive modulus) and <=MAX (bounds per-connection
125 * out_ring/rd_pool memory). */
126 if (conf->pipeline_depth < BRIX_PIPELINE_DEPTH_MIN) {
127 conf->pipeline_depth = BRIX_PIPELINE_DEPTH_MIN;
128 } else if (conf->pipeline_depth > BRIX_PIPELINE_DEPTH_MAX) {
129 conf->pipeline_depth = BRIX_PIPELINE_DEPTH_MAX;
130 }
131 ngx_conf_merge_uint_value(conf->registry_slots, prev->registry_slots, 128);
132 ngx_conf_merge_uint_value(conf->session_slots, prev->session_slots,
133 BRIX_SESSION_REGISTRY_SLOTS);
134 /* Left UNSET when unconfigured; postconfiguration treats UNSET as
135 * "use the compile-time default" (BRIX_REDIR_CACHE_SLOTS). */
136 ngx_conf_merge_uint_value(conf->redir_cache_slots, prev->redir_cache_slots,
137 NGX_CONF_UNSET_UINT);
138}

159/*
160 * WHAT: merge the traffic-mirror group (Phase 24) — targets, sample rate, the
161 * opcode/method include+exclude masks, auth-strip, divergence logging,
162 * timeout, and write-mirroring; then derive the `enabled` flag.
163 * WHY: the derived `enabled` (targets present ⇒ on) is a distinct rule worth
164 * isolating.
165 * HOW: inherit parent targets when unset, merge the tunables, then set
166 * `enabled` from the presence of at least one target.
167 */
168static void
169brix_merge_srv_mirror(ngx_stream_brix_srv_conf_t *conf,
170 ngx_stream_brix_srv_conf_t *prev)
171{
172 /* Phase 24: traffic mirror — inherit parent targets if none set locally,
173 * then derive `enabled` from the presence of at least one target. */
174 if (conf->mirror.targets == NULL) {
175 conf->mirror.targets = prev->mirror.targets;
176 }
177 ngx_conf_merge_uint_value(conf->mirror.sample_pct, prev->mirror.sample_pct, 100);
178 /* Default: mirror ALL ops; the operator de-selects with
179 * brix_mirror_exclude_opcodes (or restricts with brix_mirror_opcodes). */
180 ngx_conf_merge_uint_value(conf->mirror.opcode_mask, prev->mirror.opcode_mask,
181 BRIX_MIRROR_OP_ALL);
182 ngx_conf_merge_uint_value(conf->mirror.opcode_exclude_mask,
183 prev->mirror.opcode_exclude_mask, 0);
184 ngx_conf_merge_uint_value(conf->mirror.method_mask, prev->mirror.method_mask,
185 BRIX_MIRROR_M_DEFAULT);
186 ngx_conf_merge_value(conf->mirror.strip_auth, prev->mirror.strip_auth, 1);
187 ngx_conf_merge_value(conf->mirror.log_diverge, prev->mirror.log_diverge, 1);
188 ngx_conf_merge_msec_value(conf->mirror.timeout_ms, prev->mirror.timeout_ms, 5000);
189 ngx_conf_merge_value(conf->mirror.mirror_writes,
190 prev->mirror.mirror_writes, 0);
191 conf->mirror.enabled = (conf->mirror.targets != NULL
192 && conf->mirror.targets->nelts > 0) ? 1 : 0;
193}

195/*
196 * WHAT: merge the CMS client group (Phase 50) — locate timeout, paths, the
197 * heartbeat interval, and the resilience deadlines (read/send/keepalive/
198 * tcp_user + the connect-backoff pair left UNSET for worker-time choice).
199 * WHY: the read-timeout auto-derivation depends on the just-merged interval;
200 * isolating it keeps that ordering explicit.
201 * HOW: merge interval (floored at 1s to avoid a busy-loop heartbeat), then
202 * derive read_timeout = max(3×interval, 90s) when unset, and default
203 * tcp_user_timeout to the read timeout.
204 */
205static void
206brix_merge_srv_cms(ngx_stream_brix_srv_conf_t *conf,
207 ngx_stream_brix_srv_conf_t *prev)
208{
209 ngx_conf_merge_msec_value(conf->cms.locate_timeout, prev->cms.locate_timeout,
210 5000);
211 ngx_conf_merge_str_value(conf->cms.paths, prev->cms.paths, "");
212 ngx_conf_merge_str_value(conf->cms.vnid, prev->cms.vnid, "");
213
214 /* Phase-89 W4: load-weighted selection knob (0 = legacy scoring). Values
215 * are clamped, not rejected — the weight is a balancing hint. Applied
216 * process-wide like brix_manager_stale_after (set once before fork). */
217 ngx_conf_merge_value(conf->cms.load_weight, prev->cms.load_weight, 0);
218 if (conf->cms.load_weight < 0) {
219 conf->cms.load_weight = 0;
220 }
221 if (conf->cms.load_weight > 100) {
222 conf->cms.load_weight = 100;
223 }
224 if (conf->cms.load_weight > 0) {
225 brix_srv_set_load_weight((ngx_uint_t) conf->cms.load_weight);
226 }
227
228 /* Phase-89 W5: path-affinity sticky selection (process-wide, like the
229 * load weight — set once before fork) + multi-source locate replies. */
230 ngx_conf_merge_value(conf->cms.affinity, prev->cms.affinity, 0);
231 ngx_conf_merge_value(conf->cms.locate_multi, prev->cms.locate_multi, 0);
232 if (conf->cms.affinity) {
233 brix_srv_set_affinity(1);
234 }
235
236 /* Phase-89 W8: rm/rmdir fan-out to all holder nodes + its reply window.
237 * The window is the success deadline (node executor is silent on success),
238 * so it must be short enough to sit inside any client request timeout. */
239 ngx_conf_merge_value(conf->cms.fanout, prev->cms.fanout, 0);
240 ngx_conf_merge_msec_value(conf->cms.fanout_window,
241 prev->cms.fanout_window, 500);
242 ngx_conf_merge_value(conf->cms.interval, prev->cms.interval, 30);
243 if (conf->cms.interval < 1) {
244 /* 0 would arm a 0ms heartbeat timer AND zero the reconnect backoff
245 * (connect.c) — both busy-loops. Floor the heartbeat at 1s. */
246 conf->cms.interval = 1;
247 }
248
249 /*
250 * Phase 50: CMS client resilience deadlines. Resolve here (after
251 * cms_interval is merged) so an unset directive auto-derives a generous
252 * ON-by-default value from the heartbeat interval; an explicit 0 disables.
253 * - read timeout: max(3 x interval, 90s) — a healthy real cmsd pings well
254 * within its interval, so this never trips a conformant manager.
255 * - send timeout: 10s — bounds a manager that stops draining our writes.
256 * - tcp_user_timeout: defaults to the read-timeout as a kernel backstop.
257 */
258 if (conf->cms.read_timeout == NGX_CONF_UNSET_MSEC) {
259 if (prev->cms.read_timeout != NGX_CONF_UNSET_MSEC) {
260 conf->cms.read_timeout = prev->cms.read_timeout;
261 } else {
262 ngx_msec_t d = (ngx_msec_t) conf->cms.interval * 3 * 1000;
263 conf->cms.read_timeout = (d > 90000) ? d : 90000;
264 }
265 }
266 ngx_conf_merge_msec_value(conf->cms.send_timeout, prev->cms.send_timeout,
267 10000);
268 ngx_conf_merge_value(conf->cms.tcp_keepalive, prev->cms.tcp_keepalive, 1);
269 if (conf->cms.tcp_user_timeout == NGX_CONF_UNSET_MSEC) {
270 conf->cms.tcp_user_timeout =
271 (prev->cms.tcp_user_timeout != NGX_CONF_UNSET_MSEC)
272 ? prev->cms.tcp_user_timeout
273 : conf->cms.read_timeout;
274 }
275
276 /* Leave these UNSET through the merge so connect.c can pick the manager-locality
277 * (loopback vs remote) profile default at worker start; an explicit directive
278 * still wins and is inherited child<-parent. */
279 ngx_conf_merge_msec_value(conf->cms.initial_delay, prev->cms.initial_delay,
280 NGX_CONF_UNSET_MSEC);
281 ngx_conf_merge_msec_value(conf->cms.connect_retry, prev->cms.connect_retry,
282 NGX_CONF_UNSET_MSEC);
283}

319/*
320 * WHAT: merge the VO/group/manager-map rule arrays.
321 * WHY: the three arrays share an identical merge-and-fail-check shape; a helper
322 * keeps the NGX_CONF_ERROR handling out of the orchestrator.
323 * HOW: brix_merge_arrays() each child array over the parent; a NULL result with
324 * a non-empty parent or child signals allocation failure → NGX_CONF_ERROR.
325 */
326static char *
327brix_merge_srv_rules(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *conf,
328 ngx_stream_brix_srv_conf_t *prev)
329{
330 ngx_array_t *child_vo_rules;
331 ngx_array_t *child_group_rules;
332 ngx_array_t *child_manager_map;
333
334 child_vo_rules = conf->vo_rules;
335 conf->vo_rules = brix_merge_arrays(cf, prev->vo_rules, child_vo_rules,
336 sizeof(brix_vo_rule_t));
337 if (conf->vo_rules == NULL && (prev->vo_rules != NULL || child_vo_rules != NULL)) {
338 return NGX_CONF_ERROR;
339 }
340
341 child_group_rules = conf->group_rules;
342 conf->group_rules = brix_merge_arrays(cf, prev->group_rules,
343 child_group_rules,
344 sizeof(brix_group_rule_t));
345 if (conf->group_rules == NULL
346 && (prev->group_rules != NULL || child_group_rules != NULL)) {
347 return NGX_CONF_ERROR;
348 }
349
350 /* Merge manager_map entries (prefix -> backend mappings) */
351 child_manager_map = conf->manager_map;
352 conf->manager_map = brix_merge_arrays(cf, prev->manager_map,
353 child_manager_map,
354 sizeof(brix_manager_map_t));
355 if (conf->manager_map == NULL
356 && (prev->manager_map != NULL || child_manager_map != NULL)) {
357 return NGX_CONF_ERROR;
358 }
359
360 return NGX_CONF_OK;
361}
