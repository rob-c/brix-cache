6/*
7 * Phase 31 W4 — transfer-heap memory budget (SHM-global pool).
8 *
9 * WHAT: a single shared atomic (metrics->xfer_heap_in_use) tracks, across all
10 * worker processes, the bytes currently held in per-connection transfer scratch
11 * buffers (read/write scratch + recv payload). brix_memory_budget caps that
12 * sum; when a read would push it over, the read is deferred with kXR_wait and
13 * the client re-issues it — no server-side suspend/resume needed.
14 *
15 * WHY: per-connection trimming (W1) bounds idle memory, but only a global cap
16 * bounds the *sum* under arbitrary concurrency, turning an OOM-kill risk into
17 * graceful backpressure — the enforcement that makes the 1 GB target a guarantee.
18 *
19 * HOW: charging is done by idempotent reconciliation, not paired inc/dec, so it
20 * can never drift negative or double-count: brix_budget_sync() compares the
21 * connection's *current* scratch footprint to what it has already charged and
22 * applies only the delta. brix_budget_release() (disconnect) zeroes it.
23 */
24
25static ngx_inline size_t
26brix_budget_ctx_footprint(brix_ctx_t *ctx)
27{
28 size_t total;
29 ngx_uint_t i;
30
31 total = ctx->rd.read_scratch_size + ctx->rd.read_hdr_scratch_size
32 + ctx->rd.write_scratch_size + ctx->recv.payload_buf_size
33 + ctx->rd.cmp_scratch_size; /* phase-42 W4 inline-read codec output buf */
34
35 /* Phase 32 WS3: include the concurrent-AIO read-pool buffers (several may be
36 * in flight at once once read pipelining is enabled). */
37 for (i = 0; i < ctx->out.pipeline_depth; i++) {
38 total += ctx->rd.pool[i].size;
39 }
40
41 return total;
42}

44/*
45 * Reconcile this connection's charged bytes with its current scratch footprint.
46 * Idempotent — safe to call as often as wanted (after any scratch grow/trim).
47 */
48static ngx_inline void
49brix_budget_sync(brix_ctx_t *ctx)
50{
51 ngx_brix_srv_metrics_t *m = ctx->metrics;
52 size_t cur;
53
54 if (m == NULL) {
55 return;
56 }
57
58 cur = brix_budget_ctx_footprint(ctx);
59 if (cur == ctx->budget_charged) {
60 return;
61 }
62
63 if (cur > ctx->budget_charged) {
64 ngx_atomic_int_t add = (ngx_atomic_int_t) (cur - ctx->budget_charged);
65 ngx_atomic_uint_t now = ngx_atomic_fetch_add(&m->xfer_heap_in_use, add)
66 + (ngx_atomic_uint_t) add;
67
68 /* Best-effort high-water — a racy monotone hint, not a strict max. */
69 if (now > (ngx_atomic_uint_t) m->xfer_heap_high_water) {
70 m->xfer_heap_high_water = (ngx_atomic_t) now;
71 }
72 } else {
73 ngx_atomic_int_t sub = (ngx_atomic_int_t) (ctx->budget_charged - cur);
74 (void) ngx_atomic_fetch_add(&m->xfer_heap_in_use, -sub);
75 }
76
77 ctx->budget_charged = cur;
78}

93/*
94 * Admission: return 1 if a transfer that will hold `want` heap bytes may
95 * proceed now, 0 if it should be deferred (caller sends kXR_wait).
96 *
97 * budget <= 0 disables the cap. A connection is never blocked by the bytes it
98 * already holds (subtracted), and a lone transfer is never deadlocked when the
99 * pool is otherwise empty. On deferral, budget_waits_total is incremented.
100 */
101static ngx_inline ngx_int_t
102brix_budget_admit(brix_ctx_t *ctx, off_t budget, size_t want)
103{
104 ngx_brix_srv_metrics_t *m = ctx->metrics;
105 ngx_atomic_uint_t in_use, others;
106
107 if (m == NULL || budget <= 0) {
108 return 1;
109 }
110
111 in_use = (ngx_atomic_uint_t) m->xfer_heap_in_use;
112 others = (in_use > ctx->budget_charged)
113 ? in_use - (ngx_atomic_uint_t) ctx->budget_charged : 0;
114
115 if (others == 0 || others + want <= (ngx_atomic_uint_t) budget) {
116 return 1;
117 }
118
119 (void) ngx_atomic_fetch_add(&m->budget_waits_total, 1);
120 return 0;
121}
