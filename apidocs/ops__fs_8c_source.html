1/*
2 * ops_fs.c — namespace / filesystem operations behind the xrdfs subcommands.
3 *
4 * WHAT: mkdir, rm, rmdir, mv, chmod, truncate, query, statvfs, locate, prepare.
5 * WHY: These make the native xrdfs feature-complete (M9) — the same opcode set
6 * the system xrdfs exposes — so the harness can drive every subcommand.
7 * HOW: Each builds its packed Client*Request (wire_write_extended_requests.h /
8 * wire_core_requests.h), big-endian fields, and exchanges one frame via
9 * brix_roundtrip so path-based ops transparently follow a cluster redirect.
10 * Mutating ops expect kXR_ok (no body); query/locate/statvfs return the
11 * server's text reply verbatim for the CLI to print.
12 *
13 * wire: kXR_mv payload is "src ' ' dst" with arg1len=len(src) (src/protocols/root/write/mv.c).
14 * wire: kXR_locate reply is an "S<rw><host>:<port>" token (src/protocols/root/read/locate.c).
15 */

25/* Send one path-style request via the redirect-aware AND resilient roundtrip and
26 * require a kXR_ok with no meaningful body (mkdir/rm/rmdir/mv/chmod/truncate).
27 * cls/benign_errno tune re-issue after a sever (every tool inherits this). */
28static int
29fs_simple(brix_conn *c, void *hdr24, const void *payload, uint32_t plen,
30 brix_op_class cls, int benign_errno, brix_status *st)
31{
32 uint16_t status;
33 uint8_t *body = NULL;
34 uint32_t blen = 0;
35
36 if (brix_roundtrip_resilient(c, hdr24, payload, plen, cls, benign_errno,
37 &status, &body, &blen, st) != 0) {
38 return -1;
39 }
40 free(body);
41 return 0;
42}
43
44/* Send a request whose kXR_ok body is a text reply; copy it (NUL-terminated,
45 * trailing CR/LF trimmed) into out[outsz]. Read-only/idempotent (query / locate /
46 * statvfs / prepare): safe to retry freely on a sever. */
47static int
48fs_text(brix_conn *c, void *hdr24, const void *payload, uint32_t plen,
49 char *out, size_t outsz, brix_status *st)
50{
51 uint16_t status;
52 uint8_t *body = NULL;
53 uint32_t blen = 0;
54 size_t n;
55
56 if (brix_roundtrip_resilient(c, hdr24, payload, plen, XRDC_OP_READONLY, 0,
57 &status, &body, &blen, st) != 0) {
58 return -1;
59 }
60 n = (blen < outsz - 1) ? blen : outsz - 1;
61 memcpy(out, body, n);
62 out[n] = '\0';
63 free(body);
64 while (n > 0 && (out[n - 1] == '\n' || out[n - 1] == '\r')) {
65 out[--n] = '\0';
66 }
67 return 0;
68}

72{
73 ClientMkdirRequest req;

75 req.requestid = htons(kXR_mkdir);

109int
110brix_mv(brix_conn *c, const char *src, const char *dst, brix_status *st)
111{
112 ClientMvRequest req;
113 char *payload;
114 size_t sl = strlen(src), dl = strlen(dst), total = sl + 1 + dl;
115 int rc;
116
117 if (sl == 0 || sl > 0x7fff) {
118 brix_status_set(st, XRDC_EUSAGE, 0, "mv source path length out of range");
119 return -1;
120 }
121 payload = (char *) malloc(total);
122 if (payload == NULL) {
123 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
124 return -1;
125 }
126 /* wire: src + ' ' + dst; arg1len = len(src). */
127 memcpy(payload, src, sl);
128 payload[sl] = ' ';
129 memcpy(payload + sl + 1, dst, dl);
130
131 memset(&req, 0, sizeof(req));
132 req.requestid = htons(kXR_mv);
133 {
134 xrdw_twopath_req_t b = { .arg1len = (int16_t) sl };
135 xrdw_twopath_req_pack(&b, ((ClientRequestHdr *) &req)->body);
136 }
137
138 /* A resumed mv whose first attempt already moved the file reports ENOENT on
139 * the (now-absent) source → success. */
140 rc = fs_simple(c, &req, payload, (uint32_t) total,
141 XRDC_OP_MUTATION_NORMALIZE, ENOENT, st);
142 free(payload);
143 return rc;
144}

219int
220brix_prepare(brix_conn *c, const char *const *paths, int npaths, int options,
221 int optionX, int prty, char *out, size_t outsz, brix_status *st)
222{
223 ClientPrepareRequest req;
224 char *payload;
225 size_t total = 0;
226 int i, rc;
227 char *p;
228
229 for (i = 0; i < npaths; i++) {
230 total += strlen(paths[i]) + 1; /* path + '\n' (or final, see below) */
231 }
232 if (total == 0) {
233 brix_status_set(st, XRDC_EUSAGE, 0, "prepare needs at least one path");
234 return -1;
235 }
236 payload = (char *) malloc(total);
237 if (payload == NULL) {
238 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
239 return -1;
240 }
241 /* Newline-separated paths (no trailing newline). */
242 p = payload;
243 for (i = 0; i < npaths; i++) {
244 size_t L = strlen(paths[i]);
245 memcpy(p, paths[i], L);
246 p += L;
247 if (i + 1 < npaths) {
248 *p++ = '\n';
249 }
250 }
251
252 memset(&req, 0, sizeof(req));
253 req.requestid = htons(kXR_prepare);
254 {
255 xrdw_prepare_req_t b = { .options = (uint8_t) (options ? options : kXR_stage),
256 .prty = (uint8_t) (prty & 0x03),
257 .port = 0,
258 .optionX = (uint16_t) optionX };
259 xrdw_prepare_req_pack(&b, ((ClientRequestHdr *) &req)->body);
260 }
261
262 rc = fs_text(c, &req, payload, (uint32_t) (p - payload), out, outsz, st);
263 free(payload);
264 return rc;
265}
