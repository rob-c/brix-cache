1/*
2 * credrefresh.c — Phase 40 (b): proactive client-side credential (re)acquisition.
3 *
4 * WHAT: Before (and between) transfers, if the bearer token or GSI proxy is
5 * missing / expired / near-expiry, VOLUNTEER to obtain a fresh one — a
6 * token via the local oidc-agent (the `oidc-token` CLI), a GSI proxy via
7 * brix_proxy_create() from the long-lived user cert. Opt-in (xrdcp
8 * --auto-refresh); a silent no-op when the credential is healthy or no
9 * refresh source is available.
10 * WHY: long-running / batch transfers routinely outlive a ~1h access token or a
11 * ~12h proxy. Phase 40 (c) already WARNS ("expired — run X"); doing it FOR
12 * the user is the user-friendly next step the toolkit is positioned to give.
13 * HOW: reuses the SAME expiry introspection as (c) (brix_token_meta_get,
14 * brix_proxy_remaining) and the EXISTING brix_proxy_create primitive — no
15 * new auth plumbing. A refreshed token is installed into the process
16 * environment ($BEARER_TOKEN) so the existing brix_token_discover path picks
17 * it up on the next connection; a refreshed proxy is written to the default
18 * path the GSI code already reads. Every step fails SOFT: a missing
19 * oidc-token, no configured account, or no user cert simply skips.
20 *
21 * Clean-room: standard POSIX fork/exec/pipe + the in-tree proxy/token helpers.
22 */

34/* Refresh anything with this many seconds (or fewer) of remaining validity, so a
35 * credential cannot expire in the middle of a transfer that has just started. */
36#define XRDC_CRED_MARGIN_SECS 300

38/*
39 * Fetch a fresh access token from the local oidc-agent via the `oidc-token` CLI
40 * (`oidc-token <account>`). Returns 0 and a NUL-terminated bare token in out[],
41 * or -1 (st set) if oidc-token is absent, the account is unknown to the agent, or
42 * it produced no token. Pure fork/exec/pipe; the bare token is read from stdout.
43 */
44static int
45run_oidc_token(const char *account, char *out, size_t outsz, brix_status *st)
46{
47 char *const argv[] = { (char *) "oidc-token", (char *) account, NULL };
48 size_t got = 0;
49 int ec = -1;
50
51 if (account == NULL || account[0] == '\0') {
52 brix_status_set(st, XRDC_EAUTH, 0,
53 "no oidc account (set --oidc-account or $OIDC_ACCOUNT)");
54 return -1;
55 }
56 /* Shared fork/exec stdout capture (libxrdproto). */
57 if (brix_subprocess_capture(argv, out, outsz, &got, &ec) != 0) {
58 brix_status_set(st, XRDC_ESOCK, 0, "oidc-token: failed to spawn");
59 return -1;
60 }
61 if (ec != 0) {
62 brix_status_set(st, XRDC_EAUTH, 0,
63 "oidc-token failed for account \"%s\" (status %d)",
64 account, ec);
65 return -1;
66 }
67 if (got + 1 >= outsz) {
68 /* Filled the buffer — the token was (probably) truncated; refuse to
69 * install a half token rather than fail opaquely server-side. */
70 brix_status_set(st, XRDC_EAUTH, 0,
71 "oidc-token output too large (> %zu bytes)", outsz - 1);
72 return -1;
73 }
74 if (brix_rstrip(out) == 0) { /* trim trailing whitespace the CLI appends */
75 brix_status_set(st, XRDC_EAUTH, 0,
76 "oidc-token returned an empty token for \"%s\"", account);
77 return -1;
78 }
79 return 0;
80}

82/* 1 if a long-lived GSI user cert is discoverable (so minting a proxy can work);
83 * avoids attempting brix_proxy_create for token-only users with no cert. */
84static int
85gsi_cert_available(void)
86{
87 const char *cert = getenv("X509_USER_CERT");
88 char def[1024];
89 const char *home;
90
91 if (cert != NULL && cert[0] != '\0') {
92 return access(cert, R_OK) == 0;
93 }
94 home = getenv("HOME");
95 if (home == NULL || home[0] == '\0') {
96 return 0;
97 }
98 snprintf(def, sizeof(def), "%s/.globus/usercert.pem", home);
99 return access(def, R_OK) == 0;
100}

102/* Does the bearer token need refreshing? 1 if absent/expired/near-expiry. */
103static int
104token_needs_refresh(void)
105{
106 char *tok = brix_token_discover();
107 brix_token_meta m;
108
109 if (tok == NULL) {
110 return 1; /* none discoverable — get one if we can */
111 }
112 memset(&m, 0, sizeof(m));
113 brix_token_meta_get(tok, &m);
114 free(tok);
115 if (!m.valid) {
116 return 0; /* opaque/non-JWT token we cannot reason about — leave it */
117 }
118 if (m.expired) {
119 return 1;
120 }
121 if (m.exp > 0 && (m.exp - (long) time(NULL)) < XRDC_CRED_MARGIN_SECS) {
122 return 1;
123 }
124 return 0;
125}

127/*
128 * Refresh the bearer token via oidc-agent when it is stale, best-effort.
129 *
130 * WHAT: If an oidc account is configured AND the current bearer token is
131 * absent/expired/near-expiry, mint a fresh one with oidc-token and install
132 * it into $BEARER_TOKEN. Returns 1 when a fresh token was installed, else 0
133 * (no account, token still healthy, oidc-token failed, or setenv failed).
134 * WHY: isolates the token half of brix_cred_autorefresh so the orchestrator
135 * stays a flat two-step sequence; preserves the exact "only act on a
136 * configured account and a genuinely stale token" gate the caller relied on.
137 * HOW: 1. bail out (return 0) unless account is non-empty and token_needs_refresh().
138 * 2. run oidc-token; on failure narrate (verbose) and return 0 — soft-fail.
139 * 3. setenv $BEARER_TOKEN; on failure return 0 without narrating.
140 * 4. narrate success (verbose) and return 1.
141 */
142static int
143refresh_bearer_token(const char *account, int verbose, FILE *out)
144{
145 char tok[16384];
146 brix_status st;
147
148 if (account == NULL || account[0] == '\0' || !token_needs_refresh()) {
149 return 0;
150 }
151 brix_status_clear(&st);
152 if (run_oidc_token(account, tok, sizeof(tok), &st) != 0) {
153 if (verbose && out != NULL) {
154 fprintf(out, "xrdcp: token auto-refresh skipped: %s\n", st.msg);
155 }
156 return 0;
157 }
158 if (setenv("BEARER_TOKEN", tok, 1) != 0) {
159 return 0;
160 }
161 if (verbose && out != NULL) {
162 fprintf(out, "xrdcp: refreshed bearer token via oidc-agent "
163 "(account %s)\n", account);
164 }
165 return 1;
166}

168/*
169 * Is the GSI proxy at pxp stale (missing / unparseable / expiring soon)?
170 *
171 * WHAT: Returns 1 when a fresh proxy should be minted — the file is unreadable,
172 * its remaining validity cannot be parsed, or fewer than
173 * XRDC_CRED_MARGIN_SECS of validity remain; 0 when the existing proxy is
174 * healthy enough to keep.
175 * WHY: pulls the three-way staleness decision out of the refresh path so the
176 * expiry-margin policy lives in one named predicate, unchanged from the
177 * original inline have/parse/margin ladder.
178 * HOW: 1. no readable file → stale.
179 * 2. brix_proxy_remaining() cannot parse it → stale.
180 * 3. otherwise stale iff remaining seconds < XRDC_CRED_MARGIN_SECS.
181 */
182static int
183proxy_is_stale(const char *pxp)
184{
185 long left = 0;
186
187 if (access(pxp, R_OK) != 0) {
188 return 1; /* none → mint one */
189 }
190 if (brix_proxy_remaining(pxp, &left) != 0) {
191 return 1; /* present but unparseable → renew */
192 }
193 return left < XRDC_CRED_MARGIN_SECS; /* expiring soon → renew */
194}

196/*
197 * Refresh the GSI proxy via brix_proxy_create when it is stale, best-effort.
198 *
199 * WHAT: If a long-lived user cert is discoverable AND the default-path proxy is
200 * stale (proxy_is_stale), mint a fresh proxy from the cert. Returns 1 when a
201 * fresh proxy was created, else 0 (no cert, proxy still healthy, or
202 * brix_proxy_create failed).
203 * WHY: isolates the proxy half of brix_cred_autorefresh; preserves the
204 * "only attempt a mint when a cert exists and the proxy is stale" gate so
205 * token-only users are never troubled by proxy creation.
206 * HOW: 1. bail out (return 0) unless gsi_cert_available().
207 * 2. resolve the default proxy path; bail out unless proxy_is_stale().
208 * 3. brix_proxy_create with default opts; on failure narrate (verbose) and
209 * return 0 — soft-fail.
210 * 4. narrate success (verbose) and return 1.
211 */
212static int
213refresh_gsi_proxy(int verbose, FILE *out)
214{
215 char pxp[1024];
216 brix_proxy_opts po;
217 brix_status st;
218
219 if (!gsi_cert_available()) {
220 return 0;
221 }
222 brix_proxy_default_path(pxp, sizeof(pxp));
223 if (!proxy_is_stale(pxp)) {
224 return 0;
225 }
226 memset(&po, 0, sizeof(po)); /* defaults: ~/.globus cert/key, 12h, 2048 */
227 brix_status_clear(&st);
228 if (brix_proxy_create(&po, &st) != 0) {
229 if (verbose && out != NULL) {
230 fprintf(out, "xrdcp: proxy auto-refresh skipped: %s\n", st.msg);
231 }
232 return 0;
233 }
234 if (verbose && out != NULL) {
235 fprintf(out, "xrdcp: refreshed GSI proxy (%s)\n", pxp);
236 }
237 return 1;
238}

240/*
241 * Refresh whatever local credential is stale, best-effort.
242 *
243 * want_write — reserved for future scope-aware refresh (unused today).
244 * oidc_account — oidc-agent account; NULL ⇒ $OIDC_ACCOUNT.
245 * verbose / out — narrate what was (re)acquired when verbose and out != NULL.
246 * Returns the number of credentials freshly acquired (0 = nothing to do / no
247 * source available). Never fails the caller — a refresh source that is absent or
248 * errors simply leaves the existing credential (the server stays authoritative,
249 * and Phase 40 (c) pre-flight will still warn if it is genuinely unusable).
250 */
251int
252brix_cred_autorefresh(int want_write, const char *oidc_account, int verbose,
253 FILE *out)
254{
255 const char *account = oidc_account;
256 int refreshed = 0;
257
258 (void) want_write;
259 if (account == NULL || account[0] == '\0') {
260 account = getenv("OIDC_ACCOUNT");
261 }
262
263 /* Bearer token via oidc-agent — only when an account is configured. */
264 refreshed += refresh_bearer_token(account, verbose, out);
265
266 /* GSI proxy via brix_proxy_create — only when a user cert exists. */
267 refreshed += refresh_gsi_proxy(verbose, out);
268
269 return refreshed;
270}
