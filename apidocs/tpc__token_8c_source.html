35/* WHAT: Trims trailing whitespace characters (space, tab), newline ('\n'), and carriage return ('\r') from string s by decrementing len and setting each trimmed position to '\0'. Returns early if s is NULL or already empty. Enforces strlen() boundary via while(len > 0) guard to prevent underflow when trimming an all-whitespace string.
36 * WHY: TPC token helper outputs may arrive with trailing whitespace from subprocess execution; this cleanup ensures clean token strings before parsing and comparison operations downstream. Prevents token validation failures caused by invisible trailing characters from subprocess stdout buffering or terminal emulation artifacts.
37 * HOW: NULL/empty check → strlen(s) → while loop decrementing len from end, replacing each whitespace/newline/carriage-return byte with '\0' → stops when first non-trimmable character reached or len reaches 0 (all-whitespace string case). */
38
39static void
40tpc_trim_trailing(char *s)
41{
42 size_t len;
43
44 if (s == NULL || *s == '\0') {
45 return;
46 }
47
48 len = strlen(s);
49 while (len > 0 && (isspace((unsigned char) s[len - 1]) || s[len - 1] == '\n'
50 || s[len - 1] == '\r')) {
51 s[--len] = '\0';
52 }
53}

77/* WHAT: Sanity-check a freshly fetched delegated token before it is used to
78 * authenticate the outbound TPC pull.
79 * WHY: A delegation backend (oidc-agent / token-exchange) can return a string
80 * that is well-formed JSON-wise but not a usable credential; validating here
81 * fails fast with kXR_AuthFailed rather than letting a bad token surface as an
82 * opaque remote-open rejection later. An empty token is treated as "no
83 * delegation requested" and accepted (returns 0).
84 * HOW: Wrap delegated_token in an ngx_str_t, parse+validate it as a
85 * BRIX_TPC_CREDENTIAL_TOKEN; on any failure set err_msg/xrd_error and -1. */
86static int
87tpc_token_validate_delegated(brix_tpc_pull_t *t)
88{
89 ngx_str_t raw;
90 brix_tpc_credential_t cred;
91
92 /* Empty token => caller did not request delegation; nothing to validate. */
93 if (t->delegated_token[0] == '\0') {
94 return 0;
95 }
96
97 raw.data = (u_char *) t->delegated_token;
98 raw.len = ngx_strlen(t->delegated_token);
99
100 if (brix_tpc_credential_parse(&raw, BRIX_TPC_CREDENTIAL_TOKEN,
101 &cred, NULL,
102 t->c != NULL ? t->c->log : NULL)
103 != NGX_OK
104 || brix_tpc_credential_validate(
105 &cred, t->c != NULL ? t->c->log : NULL) != NGX_OK)
106 {
107 snprintf(t->err_msg, sizeof(t->err_msg),
108 "TPC token: delegated token validation failed");
109 t->xrd_error = kXR_AuthFailed;
110 return -1;
111 }
112
113 return 0;
114}

116/* WHAT: Resolve the oidc-token binary to an absolute path — first honours an
117 * explicit override env var, then probes standard install locations.
118 * WHY: The fallback exec path must not be subject to $PATH substitution in the
119 * daemon's environment (a compromised PATH entry could shadow the real binary).
120 * HOW: secure_getenv("BRIX_OIDC_TOKEN_BIN") if access(X_OK) passes; else
121 * walk a fixed candidate list; return NULL if none executable → caller _exit(127). */
122static const char *
123resolve_oidc_token_binary(void)
124{
125 static const char *const candidates[] = {
126 "/usr/bin/oidc-token", "/usr/local/bin/oidc-token", NULL
127 };
128 const char *const *p;
129 const char *override = secure_getenv("BRIX_OIDC_TOKEN_BIN");
130
131 if (override != NULL && access(override, X_OK) == 0) {
132 return override;
133 }
134 for (p = candidates; *p != NULL; p++) {
135 if (access(*p, X_OK) == 0) {
136 return *p;
137 }
138 }
139 return NULL;
140}

142/* WHAT: Fetches OAuth2/OIDC access token via fork/exec to local oidc-agent daemon using UNIX-socket JSON IPC. Creates pipe → forks child → dup2 STDOUT → execve dedicated helper binary (nginx-xrootd-tpc-token) with OIDC_SOCK env or fallback /run/user/1000/oidc/oidc_agent.sock → if execve fails falls back to execve resolved oidc-token binary with sanitized env → parent reads pipe stdout into buffer → waitpid checks WIFEXITED+WEXITSTATUS==0 → tpc_trim_trailing(buf) → if buf[0]=='{' parses JSON via brix_oauth2_parse_access_token else copies plain token with size guard. Returns 0 on success, -1 with err_msg/xrd_error=kXR_AuthFailed on failure.
143 * WHY: CMS/Fermilab environments use oidc-agent daemons for OIDC token management. This function provides a robust fallback chain (dedicated helper binary → generic oidc-token CLI) ensuring token fetch works even when the custom binary is absent. JSON parsing handles agent daemon's structured output; plain-token paths handle oidc-token CLI's raw output. Pipe-based IPC avoids TOCTOU race on executable path (execve without access() pre-check).
144 * HOW: pipe() → fork() → child close(pipefd[0]) dup2(pipefd[1],STDOUT) execve helper binary or resolve_oidc_token_binary → execve with controlled envp → parent close(pipefd[1]) read loop into buffer null-terminate waitpid check trim trailing whitespace if JSON parse access_token else memcpy plain token. */
145static int
146tpc_token_oidc_agent(brix_tpc_pull_t *t, char *token_out, size_t token_out_sz)
147{
148 int pipefd[2];
149 pid_t pid;
150 char buf[TPC_TOKEN_MAX_LEN + 256];
151 ssize_t nread;
152 int wstatus;
153 const char *sock_env;
154 char *argv[3];
155
156 pipefd[0] = -1;
157 pipefd[1] = -1;
158
159 if (pipe(pipefd) == -1) {
160 snprintf(t->err_msg, sizeof(t->err_msg),
161 "TPC token: pipe() failed: %s", strerror(errno));
162 t->xrd_error = kXR_ServerError;
163 return -1;
164 }
165
166 pid = fork();
167 if (pid == -1) {
168 snprintf(t->err_msg, sizeof(t->err_msg),
169 "TPC token: fork() failed: %s", strerror(errno));
170 close(pipefd[0]);
171 close(pipefd[1]);
172 t->xrd_error = kXR_ServerError;
173 return -1;
174 }
175
176 if (pid == 0) {
177 /* child --- : redirect stdout into the write end of the pipe so the
178 * parent can read the helper's token output, then close both raw fds. */
179 close(pipefd[0]);
180 dup2(pipefd[1], STDOUT_FILENO);
181 close(pipefd[1]);
182
183 /* Try the dedicated helper first; fall through to oidc-token if
184 * execve fails (binary absent / not executable). Avoid access()
185 * before execve — that is a TOCTOU race on the executable path. */
186 sock_env = getenv("OIDC_SOCK");
187 argv[0] = (char *) TPC_TOKEN_HELPER_PATH;
188 argv[1] = sock_env ? (char *) sock_env
189 : (char *) "/run/user/1000/oidc/oidc_agent.sock";
190 argv[2] = NULL;
191 execve(argv[0], argv, NULL);
192 /* execve returned — helper not found or not executable; continue */
193
194 /* Fallback: the generic oidc-token CLI, resolved to an absolute path.
195 * Avoids $PATH substitution — the daemon's PATH is untrusted.
196 * Build a minimal controlled environment: PATH + HOME + OIDC_SOCK +
197 * XDG_RUNTIME_DIR (the latter two are needed by oidc-token to locate its
198 * config and agent socket); do not inherit the full environ. */
199 {
200 const char *fb_bin = resolve_oidc_token_binary();
201 if (fb_bin != NULL) {
202 const char *home_val = getenv("HOME");
203 const char *xdg_val = getenv("XDG_RUNTIME_DIR");
204 char home_buf[280], oidc_buf[280], xdg_buf[280];
205 char *fb_argv[4] = { (char *) fb_bin, "-c", "default", NULL };
206 char *fb_envp[5];
207 int ei = 0;
208
209 fb_envp[ei++] = "PATH=/usr/bin:/bin";
210 if (home_val != NULL) {
211 snprintf(home_buf, sizeof(home_buf), "HOME=%s", home_val);
212 fb_envp[ei++] = home_buf;
213 }
214 if (sock_env != NULL) {
215 snprintf(oidc_buf, sizeof(oidc_buf), "OIDC_SOCK=%s", sock_env);
216 fb_envp[ei++] = oidc_buf;
217 }
218 if (xdg_val != NULL) {
219 snprintf(xdg_buf, sizeof(xdg_buf), "XDG_RUNTIME_DIR=%s", xdg_val);
220 fb_envp[ei++] = xdg_buf;
221 }
222 fb_envp[ei] = NULL;
223 execve(fb_bin, fb_argv, fb_envp);
224 }
225 }
226 _exit(127); /* both execs failed: 127 == "command not found" */
227 }
228
229 /* parent --- : close the write end so read() sees EOF when the child
230 * exits, then drain all stdout into buf (loop because a single read() may
231 * return short; stop on EOF/error or when buf is full minus the NUL slot). */
232 close(pipefd[1]);
233
234 nread = 0;
235 while ((ssize_t) nread < (ssize_t) sizeof(buf) - 1) {
236 ssize_t nr = read(pipefd[0], buf + nread, sizeof(buf) - 1 - nread);
237 if (nr <= 0) {
238 break;
239 }
240 nread += nr;
241 }
242 buf[nread] = '\0';
243 close(pipefd[0]);
244
245 waitpid(pid, &wstatus, 0);
246 if (!WIFEXITED(wstatus) || WEXITSTATUS(wstatus) != 0) {
247 snprintf(t->err_msg, sizeof(t->err_msg),
248 "TPC token: oidc-agent helper exited with status %d",
249 WIFEXITED(wstatus) ? WEXITSTATUS(wstatus) : -1);
250 t->xrd_error = kXR_AuthFailed;
251 return -1;
252 }
253
254 tpc_trim_trailing(buf);
255
256 /*
257 * Output discrimination: a leading '{' means the helper returned a JSON
258 * object (oidc-agent daemon style) → parse out access_token. Anything else
259 * is treated as a bare token string (oidc-token CLI style) and copied
260 * verbatim, with a length guard against the destination buffer.
261 */
262 if (buf[0] == '{') {
263 if (tpc_token_parse_access_token(buf, token_out, token_out_sz) != 0) {
264 t->xrd_error = kXR_AuthFailed;
265 return -1;
266 }
267 } else {
268 size_t tok_len = strlen(buf);
269 if (tok_len >= token_out_sz) {
270 snprintf(t->err_msg, sizeof(t->err_msg),
271 "TPC token: token too long");
272 t->xrd_error = kXR_AuthFailed;
273 return -1;
274 }
275 ngx_memcpy(token_out, buf, tok_len + 1);
276 }
277
278 return 0;
279}

281/* WHAT: Fetches delegated OAuth2/OIDC access token via RFC 8693 token-exchange POST to external OAuth2 endpoint using fork/exec curl. Reads subject_token from tpc_outbound_bearer_file → builds POST body with grant_type=urn:ietf:params:oauth:grant-type:token-exchange + subject_token/resource/audience/scope params into mkstemp temp file → constructs curl argv (-s -S -f -X POST -H Content-Type application/x-www-form-urlencoded optional -u client_id/client_secret -d body_file token_endpoint) → pipe fork execvp curl → parent reads pipe stdout → waitpid checks exit status 0 → unlink temp file → parses JSON access_token via brix_oauth2_parse_access_token. Returns 0 on success, -1 with err_msg/xrd_error set on failure.
282 * WHY: When TPC source requires a delegated token (different identity from client), RFC 8693 token-exchange converts the destination's bearer file subject token into an access token scoped for the remote origin. curl subprocess handles HTTP POST with form-urlencoded body; temp file avoids shell quoting issues with long tokens; unlink ensures cleanup on both success and failure paths. Optional client_id/client_secret basic auth supports OAuth2 client credential requirements.
283 * HOW: read bearer file → snprintf POST body (grant_type+subject_token+resource+audience+scope) → mkstemp temp write body_buf → close body_fd → curl_argv build (-s -S -f -X POST -H Content-Type optional -u client_id/client_secret -d body_file token_endpoint) → pipe fork dup2 STDOUT execvp curl → parent read loop null-terminate waitpid check unlink → parse JSON access_token. */
284static int
285tpc_token_rfc8693(brix_tpc_pull_t *t, char *token_out, size_t token_out_sz)
286{
287 char buf[TPC_TOKEN_MAX_LEN + 256];
288 char *curl_argv[20];
289 int argc = 0;
290 char body_buf[4096];
291 char body_file[NGX_MAX_PATH];
292 int body_fd;
293 char subject_token[TPC_TOKEN_MAX_LEN];
294
295 if (brix_token_read_file(&t->conf->tpc_outbound_bearer_file,
296 (u_char *) subject_token, sizeof(subject_token),
297 NULL, NULL, "TPC token")
298 != NGX_OK)
299 {
300 if (errno == EINVAL) {
301 snprintf(t->err_msg, sizeof(t->err_msg),
302 "TPC token: bearer file not configured or empty");
303 t->xrd_error = kXR_ArgInvalid;
304 } else {
305 snprintf(t->err_msg, sizeof(t->err_msg),
306 "TPC token: cannot read bearer file: %s", strerror(errno));
307 t->xrd_error = kXR_IOError;
308 }
309 return -1;
310 }

312 /*
313 * RFC 8693 token-exchange request body (form-urlencoded). The subject_token
314 * is our local bearer (the identity we already hold); resource+audience are
315 * both set to the remote origin host so the issued token is scoped to it;
316 * scope carries the configured outbound scope. NOTE: subject_token is not
317 * URL-encoded here — it is a JWT (URL-safe base64, no reserved chars), so
318 * it is form-safe as-is.
319 */
320 ngx_snprintf((u_char *) body_buf, sizeof(body_buf),
321 "grant_type=urn:ietf:params:oauth:grant-type:"
322 "token-exchange"
323 "&subject_token=%s"
324 "&resource=%s"
325 "&audience=%s"
326 "&scope=%s",
327 subject_token,
328 t->src_host,
329 t->src_host,
330 t->token_scope);
331
332 /*
333 * Stage the form body in a private temp file (mkstemp => unpredictable
334 * name, exclusive create) instead of building it on curl's command line,
335 * keeping the bearer/subject token out of /proc/PID/cmdline and the
336 * process listing. The path is captured in body_file and unlinked on every
337 * exit path below (success and all error returns).
338 */
339 {
340 char tmpl[NGX_MAX_PATH];
341 ngx_snprintf((u_char *) tmpl, sizeof(tmpl),
342 "/tmp/tpc_token_body_XXXXXX");
343 body_fd = mkstemp(tmpl);
344 if (body_fd == -1) {
345 snprintf(t->err_msg, sizeof(t->err_msg),
346 "TPC token: temp file creation failed: %s", strerror(errno));
347 t->xrd_error = kXR_IOError;
348 return -1;
349 }
350 ngx_memcpy(body_file, tmpl, strlen(tmpl) + 1);
351 }
352
353 {
354 ssize_t nw = write(body_fd, body_buf, strlen(body_buf));
355 (void) nw;
356 }
357 close(body_fd);
358
359 /*
360 * Build curl argv (execvp, no shell — so no quoting/injection risk).
361 * -s -S : silent but still print errors; -f : fail (non-zero exit) on
362 * HTTP >= 400 so the waitpid status check below catches auth failures.
363 */
364 curl_argv[argc++] = "curl";
365 curl_argv[argc++] = "-s";
366 curl_argv[argc++] = "-S";
367 curl_argv[argc++] = "-f";
368 curl_argv[argc++] = "-X";
369 curl_argv[argc++] = "POST";
370 curl_argv[argc++] = "-H";
371 curl_argv[argc++] = "Content-Type: application/x-www-form-urlencoded";
372
373 /* Optional OAuth2 client basic auth: curl -u "id:secret" — only added when
374 * both halves are configured. NOTE: -u takes a single "id:secret" arg, so
375 * these two array slots become two separate argv entries that curl joins
376 * via the -u/value convention. */
377 if (t->conf->tpc_outbound_client_id.len > 0
378 && t->conf->tpc_outbound_client_secret.len > 0) {
379 curl_argv[argc++] = "-u";
380 curl_argv[argc++] = (char *) t->conf->tpc_outbound_client_id.data;
381 curl_argv[argc++] = (char *) t->conf->tpc_outbound_client_secret.data;
382 }
383
384 /* -d data: the form body, sourced from the staged temp file path. */
385 curl_argv[argc++] = "-d";
386 curl_argv[argc++] = body_file;
387 /* W3 — end-of-options terminator so the endpoint URL can never be parsed
388 * as a curl option even if a misconfigured endpoint begins with '-'. */
389 curl_argv[argc++] = "--";
390 curl_argv[argc++] = (char *) t->conf->tpc_outbound_token_endpoint.data;
391 curl_argv[argc++] = NULL;
392
393 /*
394 * Run curl synchronously and capture its stdout (the token JSON) via the
395 * shared SIGCHLD-safe fork/exec helper (src/compat/subprocess.c) — the same
396 * pipe->fork->dup2->drain->waitpid skeleton used by the native client and
397 * the oidc-agent path. A non-zero rc means pipe/fork failed or curl was
398 * signal-killed; a non-zero child exit (including curl -f on HTTP >= 400) is
399 * an auth failure. The staged body file is unlinked on every exit path.
400 */
401 {
402 int ec = -1;
403
404 if (brix_subprocess_capture(curl_argv, buf, sizeof(buf), NULL, &ec)
405 != 0)
406 {
407 snprintf(t->err_msg, sizeof(t->err_msg),
408 "TPC token: token-exchange subprocess failed "
409 "(pipe/fork or signal)");
410 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
411 t->xrd_error = kXR_ServerError;
412 return -1;
413 }
414 if (ec != 0) {
415 snprintf(t->err_msg, sizeof(t->err_msg),
416 "TPC token: token exchange failed (curl exit %d)", ec);
417 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
418 t->xrd_error = kXR_AuthFailed;
419 return -1;
420 }
421 }
422
423 /* Success: body file no longer needed; remove before parsing the reply. */
424 unlink(body_file); /* vfs-seam-allow: /tmp credential temp, not export storage */
425
426 if (tpc_token_parse_access_token(buf, token_out, token_out_sz) != 0) {
427 t->xrd_error = kXR_AuthFailed;
428 return -1;
429 }
430
431 return 0;
432}

435/* WHAT: Public entry point that dispatches delegated token fetching based on t->token_mode. Returns 0 immediately for "none" or empty mode; delegates to tpc_token_oidc_agent() for "oidc-agent" mode (UNIX-socket JSON IPC); delegates to tpc_token_rfc8693() for "token-exchange" mode (RFC 8693 POST, validates token_endpoint configured first). Returns -1 with err_msg/xrd_error set on unknown mode or dispatch failure.
436 * WHY: TPC source authentication requires delegated tokens when the destination server authenticates as a different identity to the remote origin. This dispatcher centralizes mode selection — callers pass t->token_mode and receive the fetched token in t->delegated_token without knowing which backend mechanism was used. Prevents callers from duplicating mode-switch logic across launch.c/thread.c.
437 * HOW: token_mode=="none"/empty → return 0; "oidc-agent" → call tpc_token_oidc_agent(t, delegated_token, sizeof); "token-exchange" → validate conf->tpc_outbound_token_endpoint.len>0 else error → call tpc_token_rfc8693(t, delegated_token, sizeof); unknown mode → snprintf err_msg/xrd_error=kXR_ArgInvalid → return -1. */
438int
439tpc_fetch_delegated_token(brix_tpc_pull_t *t)
440{
441 if (t->token_mode[0] == '\0'
442 || ngx_strcmp(t->token_mode, "none") == 0) {
443 return 0;
444 }
445
446 if (ngx_strcmp(t->token_mode, "oidc-agent") == 0) {
447 if (tpc_token_oidc_agent(t, t->delegated_token,
448 sizeof(t->delegated_token)) != 0) {
449 return -1;
450 }
451 return tpc_token_validate_delegated(t);
452 }
453
454 if (ngx_strcmp(t->token_mode, "token-exchange") == 0) {
455 if (t->conf->tpc_outbound_token_endpoint.len == 0) {
456 snprintf(t->err_msg, sizeof(t->err_msg),
457 "TPC token: token_endpoint not configured for "
458 "token-exchange mode");
459 t->xrd_error = kXR_ArgInvalid;
460 return -1;
461 }
462 if (tpc_token_rfc8693(t, t->delegated_token,
463 sizeof(t->delegated_token)) != 0) {
464 return -1;
465 }
466 return tpc_token_validate_delegated(t);
467 }
468
469 snprintf(t->err_msg, sizeof(t->err_msg),
470 "TPC token: unknown delegation mode \"%s\"", t->token_mode);
471 t->xrd_error = kXR_ArgInvalid;
472 return -1;
473}
