36/* WHAT: Trims trailing whitespace characters (space, tab), newline ('\n'), and carriage return ('\r') from string s by decrementing len and setting each trimmed position to '\0'. Returns early if s is NULL or already empty. Enforces strlen() boundary via while(len > 0) guard to prevent underflow when trimming an all-whitespace string.
37 * WHY: TPC token helper outputs may arrive with trailing whitespace from subprocess execution; this cleanup ensures clean token strings before parsing and comparison operations downstream. Prevents token validation failures caused by invisible trailing characters from subprocess stdout buffering or terminal emulation artifacts.
38 * HOW: NULL/empty check → strlen(s) → while loop decrementing len from end, replacing each whitespace/newline/carriage-return byte with '\0' → stops when first non-trimmable character reached or len reaches 0 (all-whitespace string case). */
39
40static void
41tpc_trim_trailing(char *s)
42{
43 size_t len;
44
45 if (s == NULL || *s == '\0') {
46 return;
47 }
48
49 len = strlen(s);
50 while (len > 0 && (isspace((unsigned char) s[len - 1]) || s[len - 1] == '\n'
51 || s[len - 1] == '\r')) {
52 s[--len] = '\0';
53 }
54}

78/* WHAT: Sanity-check a freshly fetched delegated token before it is used to
79 * authenticate the outbound TPC pull.
80 * WHY: A delegation backend (oidc-agent / token-exchange) can return a string
81 * that is well-formed JSON-wise but not a usable credential; validating here
82 * fails fast with kXR_AuthFailed rather than letting a bad token surface as an
83 * opaque remote-open rejection later. An empty token is treated as "no
84 * delegation requested" and accepted (returns 0).
85 * HOW: Wrap delegated_token in an ngx_str_t, parse+validate it as a
86 * BRIX_TPC_CREDENTIAL_TOKEN; on any failure set err_msg/xrd_error and -1. */
87static int
88tpc_token_validate_delegated(brix_tpc_pull_t *t)
89{
90 ngx_str_t raw;
91 brix_tpc_credential_t cred;
92
93 /* Empty token => caller did not request delegation; nothing to validate. */
94 if (t->delegated_token[0] == '\0') {
95 return 0;
96 }
97
98 raw.data = (u_char *) t->delegated_token;
99 raw.len = ngx_strlen(t->delegated_token);
100
101 if (brix_tpc_credential_parse(&raw, BRIX_TPC_CREDENTIAL_TOKEN,
102 &cred, NULL,
103 t->c != NULL ? t->c->log : NULL)
104 != NGX_OK
105 || brix_tpc_credential_validate(
106 &cred, t->c != NULL ? t->c->log : NULL) != NGX_OK)
107 {
108 snprintf(t->err_msg, sizeof(t->err_msg),
109 "TPC token: delegated token validation failed");
110 t->xrd_error = kXR_AuthFailed;
111 return -1;
112 }
113
114 return 0;
115}

117/* WHAT: Resolve the oidc-token binary to an absolute path — first honours an
118 * explicit override env var, then probes standard install locations.
119 * WHY: The fallback exec path must not be subject to $PATH substitution in the
120 * daemon's environment (a compromised PATH entry could shadow the real binary).
121 * HOW: secure_getenv("BRIX_OIDC_TOKEN_BIN") if access(X_OK) passes; else
122 * walk a fixed candidate list; return NULL if none executable → caller _exit(127). */
123static const char *
124resolve_oidc_token_binary(void)
125{
126 static const char *const candidates[] = {
127 "/usr/bin/oidc-token", "/usr/local/bin/oidc-token", NULL
128 };
129 const char *const *p;
130 const char *override = secure_getenv("BRIX_OIDC_TOKEN_BIN");
131
132 if (override != NULL && access(override, X_OK) == 0) {
133 return override;
134 }
135 for (p = candidates; *p != NULL; p++) {
136 if (access(*p, X_OK) == 0) {
137 return *p;
138 }
139 }
140 return NULL;
141}

143/* WHAT: In the forked child, resolve the generic oidc-token CLI to an absolute
144 * path and execve it with a minimal controlled environment; returns only if the
145 * binary is unresolved or execve fails (caller then _exit(127)).
146 * WHY: The dedicated helper binary may be absent, so oidc-token is the fallback
147 * token source. It must run with an absolute path (no $PATH substitution — the
148 * daemon's PATH is untrusted) and a curated environment: PATH + HOME + OIDC_SOCK
149 * + XDG_RUNTIME_DIR (the latter two let oidc-token locate its config and agent
150 * socket); the full environ is deliberately NOT inherited.
151 * HOW: resolve_oidc_token_binary(); if found, read HOME/XDG_RUNTIME_DIR, build a
152 * fixed argv (-c default) and a curated envp (PATH always, HOME/OIDC_SOCK/
153 * XDG_RUNTIME_DIR only when set), then execve. */
154static void
155tpc_oidc_exec_fallback(const char *sock_env)
156{
157 const char *fb_bin = resolve_oidc_token_binary();
158
159 if (fb_bin != NULL) {
160 const char *home_val = getenv("HOME");
161 const char *xdg_val = getenv("XDG_RUNTIME_DIR");
162 char home_buf[280], oidc_buf[280], xdg_buf[280];
163 char *fb_argv[4] = { (char *) fb_bin, "-c", "default", NULL };
164 char *fb_envp[5];
165 int ei = 0;
166
167 fb_envp[ei++] = "PATH=/usr/bin:/bin";
168 if (home_val != NULL) {
169 snprintf(home_buf, sizeof(home_buf), "HOME=%s", home_val);
170 fb_envp[ei++] = home_buf;
171 }
172 if (sock_env != NULL) {
173 snprintf(oidc_buf, sizeof(oidc_buf), "OIDC_SOCK=%s", sock_env);
174 fb_envp[ei++] = oidc_buf;
175 }
176 if (xdg_val != NULL) {
177 snprintf(xdg_buf, sizeof(xdg_buf), "XDG_RUNTIME_DIR=%s", xdg_val);
178 fb_envp[ei++] = xdg_buf;
179 }
180 fb_envp[ei] = NULL;
181 execve(fb_bin, fb_argv, fb_envp);
182 }

185/* WHAT: Child-process body of the oidc-agent fetch — redirects stdout to the
186 * pipe write end and execs the token helper chain; never returns (ends in
187 * _exit(127) if both execs fail).
188 * WHY: Isolating the post-fork child path keeps the orchestrator flat and puts
189 * the exec fallback chain (dedicated helper → generic oidc-token) in one place.
190 * The dedicated helper is tried first via execve with an absolute path (avoiding
191 * a TOCTOU access()-before-exec race), then the CLI fallback.
192 * HOW: close read end; dup2 write end onto STDOUT; close write end; read
193 * OIDC_SOCK (default the standard user socket path); execve the dedicated helper
194 * with a two-arg argv; on return call tpc_oidc_exec_fallback(); _exit(127). */
195static void
196tpc_oidc_child_run(int *pipefd)
197{
198 const char *sock_env;
199 char *argv[3];
200
201 /* redirect stdout into the write end of the pipe so the parent can read the
202 * helper's token output, then close both raw fds. */
203 close(pipefd[0]);
204 dup2(pipefd[1], STDOUT_FILENO);
205 close(pipefd[1]);
206
207 /* Try the dedicated helper first; fall through to oidc-token if execve fails
208 * (binary absent / not executable). Avoid access() before execve — that is a
209 * TOCTOU race on the executable path. */
210 sock_env = getenv("OIDC_SOCK");
211 argv[0] = (char *) TPC_TOKEN_HELPER_PATH;
212 argv[1] = sock_env ? (char *) sock_env
213 : (char *) "/run/user/1000/oidc/oidc_agent.sock";
214 argv[2] = NULL;
215 execve(argv[0], argv, NULL);
216 /* execve returned — helper not found or not executable; try the fallback. */
217
218 tpc_oidc_exec_fallback(sock_env);
219 _exit(127); /* both execs failed: 127 == "command not found" */
220}

222/* WHAT: Parent-side drain of the child's stdout — reads up to bufsz-1 bytes from
223 * fd into buf and NUL-terminates it. Returns the byte count read.
224 * WHY: A single read() may return short, so the token output must be drained in
225 * a loop until EOF/error or the buffer is full (minus the NUL slot). Factoring
226 * this out keeps the orchestrator's control flow linear.
227 * HOW: loop read()ing into buf at the running offset while space remains; stop on
228 * EOF/error (nr <= 0); NUL-terminate at the accumulated length. */
229static ssize_t
230tpc_oidc_read_all(int fd, char *buf, size_t bufsz)
231{
232 ssize_t nread = 0;
233
234 while ((size_t) nread < bufsz - 1) {
235 ssize_t nr = read(fd, buf + nread, bufsz - 1 - nread);
236 if (nr <= 0) {
237 break;
238 }
239 nread += nr;
240 }
241 buf[nread] = '\0';
242 return nread;
243}

245/* WHAT: Extracts the usable token from the (already whitespace-trimmed) helper
246 * output buf into token_out. Returns 0 on success, -1 (with err_msg/xrd_error
247 * set) on parse failure or oversize plain token.
248 * WHY: The two token producers differ in shape — the oidc-agent daemon emits a
249 * JSON object, the oidc-token CLI emits a bare token — and this discrimination
250 * belongs in one pure helper so the orchestrator does not branch on output form.
251 * HOW: a leading '{' means JSON → tpc_token_parse_access_token(); otherwise a
252 * bare token → length-guard against token_out_sz and ngx_memcpy verbatim. */
253static int
254tpc_oidc_extract_token(brix_tpc_pull_t *t, const char *buf,
255 char *token_out, size_t token_out_sz)
256{
257 if (buf[0] == '{') {
258 if (tpc_token_parse_access_token(buf, token_out, token_out_sz) != 0) {
259 t->xrd_error = kXR_AuthFailed;
260 return -1;
261 }
262 return 0;
263 }
264
265 {
266 size_t tok_len = strlen(buf);
267 if (tok_len >= token_out_sz) {
268 snprintf(t->err_msg, sizeof(t->err_msg),
269 "TPC token: token too long");
270 t->xrd_error = kXR_AuthFailed;
271 return -1;
272 }
273 ngx_memcpy(token_out, buf, tok_len + 1);
274 }
275
276 return 0;
277}

279/* WHAT: Fetches an OAuth2/OIDC access token via fork/exec to a local oidc-agent
280 * (dedicated helper binary, falling back to the oidc-token CLI) using pipe IPC.
281 * Returns 0 on success (token in token_out), -1 with err_msg/xrd_error set on
282 * pipe/fork failure, non-zero child exit, parse failure, or oversize token.
283 * WHY: CMS/Fermilab environments use oidc-agent daemons for OIDC token
284 * management. The fallback chain (dedicated helper → generic oidc-token CLI)
285 * ensures token fetch works even when the custom binary is absent. Pipe-based
286 * IPC with execve (no access() pre-check) avoids a TOCTOU race on the exec path.
287 * HOW: pipe() then fork(); child runs tpc_oidc_child_run(); parent closes the
288 * write end, drains stdout via tpc_oidc_read_all(), waitpid()s and checks the
289 * exit status, trims trailing whitespace, then tpc_oidc_extract_token(). */
290static int
291tpc_token_oidc_agent(brix_tpc_pull_t *t, char *token_out, size_t token_out_sz)
292{
293 int pipefd[2];
294 pid_t pid;
295 char buf[TPC_TOKEN_MAX_LEN + 256];
296 int wstatus;
297
298 pipefd[0] = -1;
299 pipefd[1] = -1;
300
301 if (pipe(pipefd) == -1) {
302 snprintf(t->err_msg, sizeof(t->err_msg),
303 "TPC token: pipe() failed: %s", strerror(errno));
304 t->xrd_error = kXR_ServerError;
305 return -1;
306 }
307
308 pid = fork();
309 if (pid == -1) {
310 snprintf(t->err_msg, sizeof(t->err_msg),
311 "TPC token: fork() failed: %s", strerror(errno));
312 close(pipefd[0]);
313 close(pipefd[1]);
314 t->xrd_error = kXR_ServerError;
315 return -1;
316 }
317
318 if (pid == 0) {
319 tpc_oidc_child_run(pipefd);
320 }
321
322 /* parent --- : close the write end so read() sees EOF when the child exits,
323 * then drain all stdout into buf. */
324 close(pipefd[1]);
325 tpc_oidc_read_all(pipefd[0], buf, sizeof(buf));
326 close(pipefd[0]);
327
328 waitpid(pid, &wstatus, 0);
329 if (!WIFEXITED(wstatus) || WEXITSTATUS(wstatus) != 0) {
330 snprintf(t->err_msg, sizeof(t->err_msg),
331 "TPC token: oidc-agent helper exited with status %d",
332 WIFEXITED(wstatus) ? WEXITSTATUS(wstatus) : -1);
333 t->xrd_error = kXR_AuthFailed;
334 return -1;
335 }
336
337 tpc_trim_trailing(buf);
338
339 return tpc_oidc_extract_token(t, buf, token_out, token_out_sz);
340}

342/* WHAT: Public entry point that dispatches delegated token fetching based on t->token_mode. Returns 0 immediately for "none" or empty mode; for "passthrough" (explicit/strict) validates the client's inbound bearer JWT already snapshotted into t->delegated_token (empty → kXR_AuthFailed); for "passthrough-opt" (default/opportunistic) validates the inbound JWT when present but returns 0 with no token when absent (fall back to bearer-file/GSI/anon); delegates to tpc_token_oidc_agent() for "oidc-agent" mode (UNIX-socket JSON IPC); delegates to tpc_token_rfc8693() for "token-exchange" mode (RFC 8693 POST, validates token_endpoint configured first). Returns -1 with err_msg/xrd_error set on unknown mode or dispatch failure.
343 * WHY: TPC source authentication requires delegated tokens when the destination server authenticates as a different identity to the remote origin. This dispatcher centralizes mode selection — callers pass t->token_mode and receive the fetched token in t->delegated_token without knowing which backend mechanism was used. Prevents callers from duplicating mode-switch logic across launch.c/thread.c. The passthrough modes differ: no fetch happens here — the token was captured on the event loop (launch.c) — so they only inspect the snapshot. "passthrough" (client asked for it) fails closed when it is empty; "passthrough-opt" (server default) instead returns 0 so the outbound auth path can fall back, so making passthrough the default never denies a previously-anonymous/GSI-only pull.
344 * HOW: token_mode=="none"/empty → return 0; "passthrough" → require t->delegated_token[0] set (else err_msg/xrd_error=kXR_AuthFailed) → tpc_token_validate_delegated; "passthrough-opt" → empty token → return 0, else tpc_token_validate_delegated; "oidc-agent" → call tpc_token_oidc_agent(t, delegated_token, sizeof); "token-exchange" → validate conf->tpc_outbound_token_endpoint.len>0 else error → call tpc_token_rfc8693(t, delegated_token, sizeof); unknown mode → snprintf err_msg/xrd_error=kXR_ArgInvalid → return -1. */
345int
346tpc_fetch_delegated_token(brix_tpc_pull_t *t)
347{
348 if (t->token_mode[0] == '\0'
349 || ngx_strcmp(t->token_mode, "none") == 0) {
350 return 0;
351 }
352
353 /*
354 * passthrough (STRICT) — an explicit client tpc.token_mode=passthrough. The
355 * client's own inbound bearer JWT was already snapshotted into
356 * t->delegated_token on the event loop (tpc_pull_capture_passthrough_token in
357 * launch.c). There is nothing to fetch; just require the token to be present
358 * and valid so the source authenticates the end user. An empty token (no
359 * inbound bearer / one too large to forward) is a clean kXR_AuthFailed —
360 * never a silent anonymous/service fallback, because the client asked for it.
361 */
362 if (ngx_strcmp(t->token_mode, "passthrough") == 0) {
363 if (t->delegated_token[0] == '\0') {
364 snprintf(t->err_msg, sizeof(t->err_msg),
365 "TPC token: passthrough requested but no inbound bearer "
366 "token was presented by the client");
367 t->xrd_error = kXR_AuthFailed;
368 return -1;
369 }
370 return tpc_token_validate_delegated(t);
371 }
372
373 /*
374 * passthrough-opt (OPPORTUNISTIC) — the default-on brix_tpc_outbound_passthrough
375 * mode. Same event-loop capture as strict passthrough, but the ABSENCE of an
376 * inbound token is NOT an error: return 0 with delegated_token left empty so
377 * the outbound auth path (gsi_outbound_finish) falls back to the static bearer
378 * file, then GSI proxy delegation, then anonymous — exactly as it did before
379 * this mode became the default. This guarantees a flipped default never turns
380 * a previously-succeeding GSI-only or anonymous TPC pull into a new denial.
381 * When an inbound token IS present, use it (validate + forward via ztn).
382 */
383 if (ngx_strcmp(t->token_mode, "passthrough-opt") == 0) {
384 if (t->delegated_token[0] == '\0') {
385 return 0;
386 }
387 return tpc_token_validate_delegated(t);
388 }
389
390 if (ngx_strcmp(t->token_mode, "oidc-agent") == 0) {
391 if (tpc_token_oidc_agent(t, t->delegated_token,
392 sizeof(t->delegated_token)) != 0) {
393 return -1;
394 }
395 return tpc_token_validate_delegated(t);
396 }
397
398 if (ngx_strcmp(t->token_mode, "token-exchange") == 0) {
399 if (t->conf->tpc_outbound_token_endpoint.len == 0) {
400 snprintf(t->err_msg, sizeof(t->err_msg),
401 "TPC token: token_endpoint not configured for "
402 "token-exchange mode");
403 t->xrd_error = kXR_ArgInvalid;
404 return -1;
405 }
406 if (tpc_token_rfc8693(t, t->delegated_token,
407 sizeof(t->delegated_token)) != 0) {
408 return -1;
409 }
410 return tpc_token_validate_delegated(t);
411 }
412
413 snprintf(t->err_msg, sizeof(t->err_msg),
414 "TPC token: unknown delegation mode \"%s\"", t->token_mode);
415 t->xrd_error = kXR_ArgInvalid;
416 return -1;
417}
