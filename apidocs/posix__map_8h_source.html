1/*
2 * posix_map.h — backend-agnostic XRootD↔POSIX translation shared by the FUSE
3 * drivers (xrootdfs.c async + xrootdfs_legacy.c) and the preload shim.
4 *
5 * These are pure functions over libbrix data structures with NO connection model
6 * baked in, so both FUSE backends (pool / mfile / webfile) call one copy:
7 * - statinfo → struct stat (dir/regular/symlink mode, stable st_ino, blocks)
8 * - kXR_Qspace text → byte totals
9 * - server fattr list ("U.<x>\0…") → FUSE "user.<x>\0…" listxattr buffer
10 */
