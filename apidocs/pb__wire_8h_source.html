4/*
5 * pb_wire.h — minimal protobuf wire-format primitives (no external library).
6 *
7 * WHAT: bounds-checked readers and appenders for the protobuf binary encoding —
8 * varints, tags (field<<3 | wiretype), length-delimited fields, and a
9 * field skipper for unknown fields.
10 * WHY: the CTA SSI messages are protobuf; we decode/encode exactly the fields we
11 * use without pulling in protobuf-c or generated code. Pure C, standalone-
12 * testable.
13 * HOW: every reader takes a {p, end} cursor and fails (returns -1) rather than
14 * reading past end — the input is untrusted. Writers append into a
15 * caller-sized buffer and fail on overflow.
16 */
