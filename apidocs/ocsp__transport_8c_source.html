1/*
2 * OCSP transport — responder URL parsing and deadline-bounded BIO connections.
3 *
4 * WHAT: Owns everything between "OCSP responder URL string" and "a connected,
5 * verified BIO ready to POST on". Provides parse_ocsp_url() (URL → ocsp_url_t),
6 * ocsp_open_bio() (build a plain-TCP or verifying-TLS connect BIO),
7 * ocsp_connect_bio() (drive the connect under BRIX_OCSP_TIMEOUT_SECS),
8 * ocsp_set_io_timeouts() (bound the read/write phases), ocsp_tls_handshake()
9 * (complete + verify the HTTPS handshake), and ocsp_conn_free() (fixed-order
10 * teardown).
11 *
12 * WHY: Split out of ocsp.c under the phase-79 file-size guard. Concentrating
13 * the network/transport machinery here keeps ocsp_request.c focused on the OCSP
14 * request/response crypto and ocsp.c on the public revocation/staple entry
15 * points. The per-phase deadlines (E1) are the reason a black-holed responder
16 * cannot freeze the single-threaded worker event loop.
17 *
18 * HOW: The connect phase is bounded with a non-blocking connect + poll()
19 * (SO_SNDTIMEO does not bound connect()); the handshake/read/write phases with
20 * SO_RCVTIMEO/SO_SNDTIMEO. ocsp_conn_t is the single owner of the BIO chain and
21 * (HTTPS only) the SSL_CTX; every failure path cleans up in the original
22 * BIO_free_all() → SSL_CTX_free() order. The ssl handle is owned by the BIO
23 * chain and is never freed separately. Shared structs/decls live in
24 * ocsp_internal.h.
25 */

49/*
50 * Drive BIO_do_connect to completion under a `secs` deadline using non-blocking
51 * mode + poll(). Returns 0 on connect, -1 on hard failure, -2 on TIMEOUT. Leaves
52 * the BIO in blocking mode on success so the subsequent handshake/sendreq (bounded
53 * by SO_*TIMEO) behave normally.
54 */
55#define OCSP_CONNECT_OK 0
56#define OCSP_CONNECT_FAIL (-1)
57#define OCSP_CONNECT_TIMEOUT (-2)
58
59static int
60ocsp_connect_deadline(BIO *cbio, int secs)
61{
62 time_t deadline = time(NULL) + secs;
63
64 BIO_set_nbio(cbio, 1);
65
66 for ( ;; ) {
67 struct pollfd pfd;
68 int fd = -1;
69 int remaining_ms;
70
71 if (BIO_do_connect(cbio) > 0) {
72 BIO_set_nbio(cbio, 0); /* back to blocking for SO_*TIMEO phase */
73 return OCSP_CONNECT_OK;
74 }
75 if (!BIO_should_retry(cbio)) {
76 return OCSP_CONNECT_FAIL; /* hard failure */
77 }
78 if (BIO_get_fd(cbio, &fd) <= 0 || fd < 0) {
79 return OCSP_CONNECT_FAIL;
80 }
81 remaining_ms = (int) ((deadline - time(NULL)) * 1000);
82 if (remaining_ms <= 0) {
83 return OCSP_CONNECT_TIMEOUT; /* connect deadline exceeded */
84 }
85 pfd.fd = fd;
86 pfd.events = POLLOUT | POLLIN;
87 pfd.revents = 0;
88 if (poll(&pfd, 1, remaining_ms) <= 0) {
89 return OCSP_CONNECT_TIMEOUT; /* poll timeout / error */
90 }
91 }
92}

94/*
95 * ocsp_url_scheme_start — advance past the URL scheme and set port/use_ssl.
96 *
97 * WHAT: Recognises the "http://" or "https://" prefix, seeds the default port
98 * (80 / 443) and use_ssl flag on out, and returns the pointer just past the
99 * scheme.
100 * WHY: Isolates the one branch that distinguishes the two supported schemes so
101 * parse_ocsp_url() reads as a straight-line hostname/port/path split.
102 * HOW: Prefix compare with strncmp; NULL return signals an unsupported scheme.
103 */
104static const char *
105ocsp_url_scheme_start(const char *url, ocsp_url_t *out)
106{
107 out->use_ssl = 0;
108 out->port = 80;
109
110 if (strncmp(url, "http://", 7) == 0) {
111 return url + 7;
112 }
113 if (strncmp(url, "https://", 8) == 0) {
114 out->use_ssl = 1;
115 out->port = 443;
116 return url + 8;
117 }
118 return NULL;
119}

121/*
122 * parse_ocsp_url — parse "http://host[:port]/path" into an ocsp_url_t.
123 *
124 * WHAT: Fills out->host / out->path / out->port / out->use_ssl from url.
125 * WHY: Callers need host:port + request path + scheme to open the responder
126 * connection; centralising the split keeps the URL grammar in one place.
127 * HOW: Skip the scheme (ocsp_url_scheme_start), locate the hostname end at the
128 * first ':' (port) or '/' (path) — port defaults to the scheme default, path
129 * defaults to "/". Returns 0 on success, -1 if the scheme is unsupported or
130 * the hostname is empty / does not fit out->host.
131 */
132int
133parse_ocsp_url(const char *url, ocsp_url_t *out)
134{
135 const char *p;
136 const char *colon;
137 const char *slash;
138 size_t hlen;
139
140 p = ocsp_url_scheme_start(url, out);
141 if (p == NULL) {
142 return -1;
143 }
144
145 slash = strchr(p, '/');
146 colon = strchr(p, ':');
147
148 /* Determine the end of the hostname */
149 if (colon != NULL && (slash == NULL || colon < slash)) {
150 hlen = (size_t)(colon - p);
151 out->port = atoi(colon + 1);
152 } else if (slash != NULL) {
153 hlen = (size_t)(slash - p);
154 } else {
155 hlen = strlen(p);
156 }
157
158 if (hlen == 0 || hlen >= sizeof(out->host)) {
159 return -1;
160 }
161 memcpy(out->host, p, hlen);
162 out->host[hlen] = '\0';
163
164 /* Path defaults to "/" */
165 if (slash != NULL && strlen(slash) < sizeof(out->path)) {
166 memcpy(out->path, slash, strlen(slash) + 1);
167 } else {
168 memcpy(out->path, "/", 2);
169 }
170
171 return 0;
172}

174/*
175 * ocsp_conn_free — release a connection's OpenSSL objects in fixed order.
176 *
177 * WHAT: Frees the BIO chain, then the SSL_CTX (if any).
178 * WHY: The BIO must be freed before the SSL_CTX it was built from; this mirrors
179 * the original inline cleanup exactly to avoid leaks or double-frees.
180 * HOW: BIO_free_all() tolerates NULL; ssl_ctx is guarded. ssl is owned by the
181 * BIO chain and must NOT be freed separately.
182 */
183void
184ocsp_conn_free(ocsp_conn_t *c)
185{
186 if (c->cbio != NULL) {
187 BIO_free_all(c->cbio);
188 }
189 if (c->ssl_ctx != NULL) {
190 SSL_CTX_free(c->ssl_ctx);
191 }
192}

194/*
195 * ocsp_bio_new_ssl — build a verifying TLS-client BIO for an HTTPS responder.
196 *
197 * WHAT: Creates an SSL_CTX with peer verification + default trust paths, wraps
198 * it in a connect BIO, and arms SNI + strict hostname verification for `host`.
199 * On success out->cbio / out->ssl_ctx / out->ssl are populated; returns 0.
200 * On any failure out is left cleaned up (freed, fields NULL) and returns -1.
201 * WHY: Concentrates the HTTPS-only setup so ocsp_open_bio() stays a thin
202 * scheme switch; preserves the original security posture (SSL_VERIFY_PEER,
203 * SSL_set1_host) exactly.
204 * HOW: Cleans up in-place on each failure in the same BIO-then-CTX order as the
205 * original inline code. ssl is owned by the BIO chain (never freed here).
206 */
207static int
208ocsp_bio_new_ssl(ngx_log_t *log, const char *host, ocsp_conn_t *out)
209{
210 out->ssl_ctx = SSL_CTX_new(TLS_client_method());
211 if (out->ssl_ctx == NULL) {
212 return -1;
213 }
214
215 SSL_CTX_set_verify(out->ssl_ctx, SSL_VERIFY_PEER, NULL);
216 if (SSL_CTX_set_default_verify_paths(out->ssl_ctx) != 1) {
217 ngx_log_error(NGX_LOG_WARN, log, 0,
218 "brix_ocsp: cannot load default HTTPS trust paths");
219 SSL_CTX_free(out->ssl_ctx);
220 out->ssl_ctx = NULL;
221 return -1;
222 }
223
224 out->cbio = BIO_new_ssl_connect(out->ssl_ctx);
225 if (out->cbio == NULL) {
226 SSL_CTX_free(out->ssl_ctx);
227 out->ssl_ctx = NULL;
228 return -1;
229 }
230
231 BIO_get_ssl(out->cbio, &out->ssl);
232 if (out->ssl != NULL) {
233 (void) SSL_set_tlsext_host_name(out->ssl, host);
234 if (SSL_set1_host(out->ssl, host) != 1) {
235 ngx_log_error(NGX_LOG_WARN, log, 0,
236 "brix_ocsp: cannot set HTTPS hostname "
237 "verification for \"%s\"", host);
238 ocsp_conn_free(out);
239 out->cbio = NULL;
240 out->ssl_ctx = NULL;
241 out->ssl = NULL;
242 return -1;
243 }
244 }
245 return 0;
246}

248/*
249 * ocsp_open_bio — build the responder BIO for a parsed URL (HTTP or HTTPS).
250 *
251 * WHAT: Populates out with a connect BIO for `u` (plus SSL_CTX/SSL for HTTPS)
252 * and sets its connect hostname to `hostport`. Returns 0 on success, -1 on
253 * failure (out fully cleaned up on failure).
254 * WHY: Gives do_ocsp_request() a single scheme-agnostic entry point so its body
255 * is connect → handshake → send with no BIO-construction branching.
256 * HOW: Delegates HTTPS to ocsp_bio_new_ssl(); plain HTTP uses BIO_new_connect().
257 * BIO_set_conn_hostname() is applied uniformly, matching the original.
258 */
259int
260ocsp_open_bio(ngx_log_t *log, const ocsp_url_t *u,
261 const char *hostport, ocsp_conn_t *out)
262{
263 out->cbio = NULL;
264 out->ssl_ctx = NULL;
265 out->ssl = NULL;
266
267 if (u->use_ssl) {
268 if (ocsp_bio_new_ssl(log, u->host, out) != 0) {
269 return -1;
270 }
271 } else {
272 out->cbio = BIO_new_connect(hostport);
273 }
274
275 if (out->cbio == NULL) {
276 ocsp_conn_free(out);
277 out->cbio = NULL;
278 out->ssl_ctx = NULL;
279 return -1;
280 }
281
282 BIO_set_conn_hostname(out->cbio, hostport);
283 return 0;
284}

286/*
287 * ocsp_connect_bio — drive the deadline-bounded connect and log the outcome.
288 *
289 * WHAT: Runs ocsp_connect_deadline() on conn->cbio; returns 0 on connect, -1 on
290 * failure or timeout (bumping the timeout metric and logging on the sad paths).
291 * WHY: Keeps the E1 connect-deadline bookkeeping (metric + WARN) out of the main
292 * flow while leaving cleanup ownership with the caller.
293 * HOW: Pure status translation — no OpenSSL objects are freed here.
294 */
295int
296ocsp_connect_bio(ngx_log_t *log, ocsp_conn_t *conn, const char *hostport)
297{
298 int crc = ocsp_connect_deadline(conn->cbio, BRIX_OCSP_TIMEOUT_SECS);
299
300 if (crc == OCSP_CONNECT_OK) {
301 return 0;
302 }
303 if (crc == OCSP_CONNECT_TIMEOUT) {
304 BRIX_RESIL_METRIC_INC(ocsp_timeouts_total);
305 }
306 ngx_log_error(NGX_LOG_WARN, log, 0,
307 "brix_ocsp: connect to \"%s\" %s (%ds)", hostport,
308 crc == OCSP_CONNECT_TIMEOUT ? "timed out" : "failed",
309 BRIX_OCSP_TIMEOUT_SECS);
310 return -1;
311}

313/*
314 * ocsp_tls_handshake — complete + verify the TLS handshake for an HTTPS BIO.
315 *
316 * WHAT: Runs BIO_do_handshake() and checks SSL_get_verify_result()==X509_V_OK.
317 * Returns 0 on success, -1 on handshake or verification failure.
318 * WHY: Isolates the HTTPS-only handshake/verify pair so do_ocsp_request() only
319 * calls it on the SSL path; preserves the exact verification requirement.
320 * HOW: No objects are freed here — the caller owns teardown on failure.
321 */
322int
323ocsp_tls_handshake(ngx_log_t *log, ocsp_conn_t *conn,
324 const char *host, const char *hostport)
325{
326 if (BIO_do_handshake(conn->cbio) <= 0) {
327 ngx_log_error(NGX_LOG_WARN, log, 0,
328 "brix_ocsp: TLS handshake with \"%s\" failed", hostport);
329 return -1;
330 }
331 if (conn->ssl == NULL
332 || SSL_get_verify_result(conn->ssl) != X509_V_OK)
333 {
334 ngx_log_error(NGX_LOG_WARN, log, 0,
335 "brix_ocsp: TLS verification for \"%s\" failed", host);
336 return -1;
337 }
338 return 0;
339}
