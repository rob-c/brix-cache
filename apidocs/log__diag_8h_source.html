1/*
2 * log_diag.h — admin-friendly diagnostic logging helpers.
3 *
4 * WHAT: turn a terse one-line error into a self-explanatory, three-part
5 * diagnostic that an operator who has never seen this module before can act on:
6 *
7 * xrootd[pki]: no CRLs found in "/etc/grid-security/certificates"
8 * cause: CRL directory is empty or the path is wrong
9 * fix: run fetch-crl (or set brix_crl_path); until then revoked
10 * certificates are ACCEPTED
11 *
12 * The macros wrap ngx_log_error() with a fixed message shape:
13 *
14 * <summary> — what happened, in plain language, with context
15 * cause: <cause> — the most likely reason, named explicitly
16 * fix: <fix> — the concrete next step the admin should take
17 *
18 * WHY: this module fronts grid storage (GSI/x509, CRLs, tokens, SSS keytabs,
19 * CMS clustering, tape/FRM, disk I/O). When something breaks at 3am the error
20 * log is the only thing the on-call admin has. A bare "frm: pread failed" or
21 * "EVP_DecryptFinal failed" tells them nothing; a message that names the cause
22 * and the fix turns a 2-hour outage into a 2-minute one. Centralising the shape
23 * also keeps the whole module's diagnostics uniform and greppable
24 * (`grep 'fix:'` surfaces every actionable line).
25 *
26 * HOW: `summary`, `cause`, and `fix` are string literals, concatenated at
27 * compile time with the indented "cause:/fix:" scaffolding — zero runtime cost,
28 * nothing built on the stack. Dynamic context (paths, addresses, sizes) goes in
29 * the trailing varargs and is consumed by printf-style specifiers in `summary`
30 * (nginx's %V/%s/%d/%ui/etc). `err` is an errno (or 0) exactly as ngx_log_error
31 * expects; when non-zero nginx appends the OS string, so disk/network failures
32 * carry the kernel's own truth ("(28: No space left on device)") after the fix.
33 *
34 * Convention for callers:
35 * - Begin `summary` with the subsystem tag: "xrootd[<sub>]: ...".
36 * - Keep `cause`/`fix` literal; fold any dynamic value into `summary` instead.
37 * - Use the level that matches operator impact:
38 * EMERG — startup/config; nginx -t fails, nothing serves yet.
39 * CRIT — running service degraded for everyone (store reload failed).
40 * ERR — one request/connection failed; service otherwise healthy.
41 * WARN — working but mis-tuned or insecure; admin should look.
42 * - For untrusted, wire-supplied strings sanitise first
43 * (brix_sanitize_log_string) before passing as a vararg.
44 *
45 * For failures with no operator action (internal invariant violations) keep a
46 * plain ngx_log_error — a "fix:" line you can't honour is worse than none.
47 */
48#ifndef BRIX_COMPAT_LOG_DIAG_H
49#define BRIX_COMPAT_LOG_DIAG_H

51#include <ngx_core.h>

53/*
54 * BRIX_DIAG — emit a three-part diagnostic at an explicit level.
55 *
56 * Expands to a single ngx_log_error() call; the message body is the
57 * compile-time concatenation of the summary and the indented cause/fix lines.
58 */
59#define BRIX_DIAG(level, log, err, summary, cause, fix, ...) \
60 ngx_log_error((level), (log), (err), \
61 summary "\n cause: " cause "\n fix: " fix, ##__VA_ARGS__)
62
63/* Level-specific shorthands — pick by operator impact (see header comment). */
64#define BRIX_DIAG_EMERG(log, err, summary, cause, fix, ...) \
65 BRIX_DIAG(NGX_LOG_EMERG, (log), (err), summary, cause, fix, ##__VA_ARGS__)
66
67#define BRIX_DIAG_CRIT(log, err, summary, cause, fix, ...) \
68 BRIX_DIAG(NGX_LOG_CRIT, (log), (err), summary, cause, fix, ##__VA_ARGS__)
69
70#define BRIX_DIAG_ERR(log, err, summary, cause, fix, ...) \
71 BRIX_DIAG(NGX_LOG_ERR, (log), (err), summary, cause, fix, ##__VA_ARGS__)
72
73#define BRIX_DIAG_WARN(log, err, summary, cause, fix, ...) \
74 BRIX_DIAG(NGX_LOG_WARN, (log), (err), summary, cause, fix, ##__VA_ARGS__)
75
76/*
77 * BRIX_DIAG_CONF — same three-part shape for the config-parse phase, where
78 * the available context is an ngx_conf_t* (cf) rather than an ngx_log_t*.
79 * nginx prefixes these with the config file and line number, so a directive
80 * mistake points the admin straight at the offending line. Use NGX_LOG_EMERG
81 * to make `nginx -t` fail.
82 */
83#define BRIX_DIAG_CONF(level, cf, err, summary, cause, fix, ...) \
84 ngx_conf_log_error((level), (cf), (err), \
85 summary "\n cause: " cause "\n fix: " fix, ##__VA_ARGS__)
86
87#endif /* BRIX_COMPAT_LOG_DIAG_H */
