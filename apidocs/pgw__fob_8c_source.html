1/*
2 * src/write/pgw_fob.c — kXR_pgwrite CSE uncorrected-page registry (see pgw_fob.h).
3 *
4 * A fixed-size set of pages that failed CRC32c and await a kXR_pgRetry
5 * correction. Keyed exactly like stock XrdXrootdPgwFob so partial last pages
6 * are distinct from full pages at the same file offset. Linear scan, no heap,
7 * single-threaded per connection.
8 */

15/* Encode (offset,dlen) the way stock does: shift the offset up by the page-bit
16 * count and fold a short (unaligned) page's length into the low bits. A full
17 * page contributes 0 in the low bits. */
18static int64_t
19pgw_fob_key(int64_t off, uint32_t dlen)
20{
21 int64_t key = off << kXR_pgPageBL;
22
23 if (dlen < (uint32_t) kXR_pgPageSZ) {
24 key |= (int64_t) dlen;
25 }
26 return key;
27}

42int
43brix_pgw_fob_add(brix_file_t *f, int64_t off, uint32_t dlen)
44{
45 int64_t key = pgw_fob_key(off, dlen);
46 uint32_t i;
47 int free_slot = -1;
48
49 f->pgw_fob_errs++;
50
51 /* Already registered → idempotent (a page re-failing keeps one entry). */
52 for (i = 0; i < BRIX_PGW_FOB_SLOTS; i++) {
53 if (f->pgw_fob[i].used) {
54 if (f->pgw_fob[i].key == key) {
55 return 1;
56 }
57 } else if (free_slot < 0) {
58 free_slot = (int) i;
59 }
60 }
61
62 if (free_slot < 0) {
63 return 0; /* registry full → caller replies kXR_TooManyErrs */
64 }
65
66 f->pgw_fob[free_slot].key = key;
67 f->pgw_fob[free_slot].used = 1;
68 f->pgw_fob_count++;
69 return 1;
70}

73brix_pgw_fob_del(brix_file_t *f, int64_t off, uint32_t dlen)
74{
75 int64_t key = pgw_fob_key(off, dlen);
76 uint32_t i;
77
78 for (i = 0; i < BRIX_PGW_FOB_SLOTS; i++) {
79 if (f->pgw_fob[i].used && f->pgw_fob[i].key == key) {
80 f->pgw_fob[i].used = 0;
81 f->pgw_fob[i].key = 0;
82 if (f->pgw_fob_count > 0) {
83 f->pgw_fob_count--;
84 }
85 f->pgw_fob_fixes++;
86 return 1;
87 }
88 }
