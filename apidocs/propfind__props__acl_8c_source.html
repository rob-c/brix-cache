7/*
8 * WHAT: Emit owner property (authenticated DN or "anonymous"). Returns NGX_OK or NGX_ERROR.
9 *
10 * WHY: RFC 3744 D:owner identifies the resource owner. We use the authenticated
11 * client DN from the request context, or "anonymous" if unavailable.
12 *
13 * HOW: Check ctx for non-empty DN, XML-escape it, emit as <D:owner><D:href>.
14 */
15static ngx_int_t
16propfind_emit_owner(ngx_http_request_t *r, ngx_chain_t **head,
17 ngx_chain_t **tail, ngx_http_brix_webdav_req_ctx_t *ctx)
18{
19 ngx_pool_t *pool = r->pool;
20 const char *owner = (ctx != NULL && ctx->dn[0] != '\0')
21 ? ctx->dn : "anonymous";
22 char *safe_owner = webdav_escape_xml_text(pool, owner);
23
24 if (safe_owner == NULL
25 || brix_http_chain_appendf(pool, head, tail,
26 "<D:owner><D:href>%s</D:href></D:owner>",
27 safe_owner) == NULL)
28 {
29 return NGX_ERROR;
30 }
31
32 return NGX_OK;
33}

35/*
36 * WHAT: Emit current-user-privilege-set ACL property. Returns NGX_OK or NGX_ERROR.
37 *
38 * WHY: RFC 3744 requires reporting the effective privileges for the current user.
39 * We synthesize read-only or read-write privileges based on allow_write.
40 *
41 * HOW: Emit <D:current-user-privilege-set> with <D:read/>. If allow_write is
42 * enabled, add write/write-content/write-properties/bind/unbind.
43 */
44static ngx_int_t
45propfind_emit_current_privilege(ngx_http_request_t *r, ngx_chain_t **head,
46 ngx_chain_t **tail,
47 ngx_http_brix_webdav_loc_conf_t *conf)
48{
49 ngx_pool_t *pool = r->pool;
50
51 if (brix_http_chain_appendf(pool, head, tail,
52 "<D:current-user-privilege-set>"
53 "<D:privilege><D:read/></D:privilege>") == NULL)
54 {
55 return NGX_ERROR;
56 }
57
58 if (conf->common.allow_write
59 && brix_http_chain_appendf(pool, head, tail,
60 "<D:privilege><D:write/></D:privilege>"
61 "<D:privilege><D:write-content/></D:privilege>"
62 "<D:privilege><D:write-properties/></D:privilege>"
63 "<D:privilege><D:bind/></D:privilege>"
64 "<D:privilege><D:unbind/></D:privilege>") == NULL)
65 {
66 return NGX_ERROR;

68
69 if (brix_http_chain_appendf(pool, head, tail,
70 "</D:current-user-privilege-set>") == NULL)
71 {
72 return NGX_ERROR;
73 }

124/*
125 * WHAT: Emit ACL element with all/read/write privileges. Returns NGX_OK or NGX_ERROR.
126 *
127 * WHY: RFC 3744 D:acl property describes the ACL entries for the resource.
128 * We advertise a simple all-principal grant with read and (optionally) write.
129 *
130 * HOW: Emit <D:acl><D:ace> with all principal, read privilege, and (if allow_write)
131 * write privilege.
132 */
133static ngx_int_t
134propfind_emit_acl(ngx_http_request_t *r, ngx_chain_t **head,
135 ngx_chain_t **tail, ngx_http_brix_webdav_loc_conf_t *conf)
136{

138
139 if (brix_http_chain_appendf(pool, head, tail,
140 "<D:acl><D:ace><D:principal><D:all/></D:principal>"
141 "<D:grant><D:privilege><D:read/></D:privilege>") == NULL)
142 {
143 return NGX_ERROR;
144 }
145
146 if (conf->common.allow_write
147 && brix_http_chain_appendf(pool, head, tail,

149 {
150 return NGX_ERROR;

154 "</D:grant><D:protected/></D:ace></D:acl>") == NULL)
155 {
156 return NGX_ERROR;
157 }

162/*
163 * Append the RFC 3744 ACL-related DAV: properties selected by `mask`.
164 * These are synthesized from the request's authenticated identity and the
165 * location's allow_write flag rather than from a real ACL store: owner is the
166 * client DN (or "anonymous"), and the privilege/ACL sets advertise read plus
167 * (only when writes are enabled) the write family. Each clause is gated on its
168 * own mask bit so PROP requests get exactly the properties they asked for.
169 */
170ngx_int_t
171propfind_append_acl_properties(ngx_http_request_t *r, ngx_chain_t **head,
172 ngx_chain_t **tail, unsigned mask)
173{
174 ngx_http_brix_webdav_loc_conf_t *conf;
175 ngx_http_brix_webdav_req_ctx_t *ctx;
176 ngx_pool_t *pool = r->pool;
177
178 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
179 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
180
181 if (mask & PF_OWNER) {
182 if (propfind_emit_owner(r, head, tail, ctx) != NGX_OK) {
183 return NGX_ERROR;
184 }
185 }
186
187 if ((mask & PF_GROUP)
188 && brix_http_chain_appendf(pool, head, tail, "<D:group/>") == NULL)
189 {
190 return NGX_ERROR;
191 }
192
193 if (mask & PF_CURRENT_PRIVILEGE) {
194 if (propfind_emit_current_privilege(r, head, tail, conf) != NGX_OK) {
195 return NGX_ERROR;
196 }
197 }
198
199 if (mask & PF_SUPPORTED_PRIVILEGE) {
200 if (propfind_emit_supported_privilege(r, head, tail) != NGX_OK) {
201 return NGX_ERROR;
202 }
203 }
204
205 if (mask & PF_ACL) {
206 if (propfind_emit_acl(r, head, tail, conf) != NGX_OK) {
207 return NGX_ERROR;
208 }
209 }
210
211 if ((mask & PF_ACL_RESTRICTIONS)
212 && brix_http_chain_appendf(pool, head, tail,
213 "<D:acl-restrictions><D:grant-only/><D:no-invert/>"
214 "</D:acl-restrictions>") == NULL)
215 {
216 return NGX_ERROR;
217 }
218
219 if ((mask & PF_PRINCIPAL_SET)
220 && brix_http_chain_appendf(pool, head, tail,
221 "<D:principal-collection-set/>") == NULL)
222 {
223 return NGX_ERROR;
224 }
225
226 return NGX_OK;
227}
