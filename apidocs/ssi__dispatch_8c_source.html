1/*
2 * ssi_dispatch.c — SSI request dispatch, async deferral, and the kXR_write
3 * accumulation path (split from ssi.c, phase-79). See ssi.h / ssi_internal.h.
4 *
5 * WHAT: The write side of the byte-exact XrdSsi-over-xroot engine glue. A
6 * kXR_write accumulates a request (ssi_write_accumulate) and, once the
7 * request is complete, dispatches it through a native responder
8 * (brix_ssi_dispatch) to the resolved service. A service may answer inline
9 * or defer; deferred completions are driven off an event-loop timer
10 * (ssi_defer_*) that resolves a LIVE session through the registry before
11 * delivering the pushed response.
12 * WHY: ssi.c held both the client-facing reply side (open/query/read) and this
13 * write/dispatch side in one 725-line file. Isolating the dispatch +
14 * deferral + write machinery here keeps each half focused and under the
15 * 500-line cap without changing a single byte of wire behaviour.
16 * HOW: The responder callbacks fill brix_ssi_req_t; brix_ssi_dispatch runs the
17 * submit phase (defer accepted), ssi_complete_deferred the completion phase
18 * (defer refused, alerts pushed). The heap-allocated ssi_defer_t carries a
19 * session key + generation (never a raw pointer) so a fired timer never
20 * touches freed memory. brix_ssi_write coalesces writes into one request and
21 * arms the deferral, replying kXR_waitresp for deferred services.
22 */

45/*
46 * Responder state. The synchronous/submit phase fills only `rq` (ctx/c NULL, so
47 * alerts drop — there is no push channel yet). The completion phase (driven from
48 * the deferral timer) also carries ctx/c, so alerts are pushed live.
49 */
50typedef struct {
51 brix_ssi_req_t *rq;
52 brix_ctx_t *ctx; /* non-NULL only when a push channel exists */
53 ngx_connection_t *c;
54} ssi_resp_state_t;

75static void
76ssi_resp_set_response(brix_ssi_responder_t *r, const unsigned char *buf,
77 size_t len, int last)
78{
79 brix_ssi_req_t *rq = ((ssi_resp_state_t *) r->state)->rq;
80
81 /* A non-final chunk marks the response as streamed: it is delivered to the
82 * client via kXR_read pulls (pendResp) rather than inline in the reply. */
83 if (!last) {
84 rq->streaming = 1;
85 }
86 {
87 size_t cap = rq->response_max > 0 ? rq->response_max : BRIX_SSI_RESP_MAX;
88 if (brix_ssi_respbuf_append(&rq->resp, buf, len, cap) != 0) {
89 rq->error = 1;
90 rq->err_code = kXR_NoMemory;
91 return;
92 }
93 }
94 rq->responded = 1;
95}

97static void
98ssi_resp_alert(brix_ssi_responder_t *r, const unsigned char *buf, size_t len)
99{
100 ssi_resp_state_t *st = r->state;
101
102 /* Deliverable only when a push channel exists (the deferred completion phase)

104 * out-of-band channel, so it drops alerts (Phase-1 behaviour). */

106 brix_ssi_deliver_alert(st->ctx, st->c, st->rq, buf, len);

161/* Run the resolved service over the accumulated request (submit phase). The
162 * service may answer inline or call r->defer to be completed later. */
163void
164brix_ssi_dispatch(brix_ssi_req_t *rq, brix_ssi_process_fn process)
165{
166 brix_ssi_responder_t r;
167 ssi_resp_state_t st = { rq, NULL, NULL };
168
169 ssi_responder_init(&r, &st, ssi_resp_defer_accept);
170
171 rq->dispatched = 1;
172 if (process(rq->req, rq->req_len, &r) != 0 && !rq->responded) {
173 rq->error = 1;
174 rq->err_code = kXR_ServerError;
175 rq->responded = 1;
176 }
177}

198/* Async deferral: an event-loop timer drives the deferred completion. */

201/*
202 * The timer state is heap-allocated (NOT pool-allocated) so it has an independent
203 * lifetime: handle teardown cancels it (ngx_del_timer + free) before the
204 * connection pool dies, and the timer handler frees it on fire. It carries the
205 * session key + generation (never a raw session pointer) so the handler resolves
206 * a LIVE session through the registry — the use-after-free guard.
207 */
208typedef struct {
209 ngx_event_t timer;
210 brix_ctx_t *ctx;
211 ngx_connection_t *c;
212 uintptr_t session_key;
213 uint64_t generation;
214 uint32_t req_id;
215} ssi_defer_t;

217static void
218ssi_defer_cancel(void *p)
219{
220 ssi_defer_t *d = p;
221
222 if (d == NULL) {
223 return;
224 }
225 if (d->timer.timer_set) {
226 ngx_del_timer(&d->timer);
227 }
228 ngx_free(d);
229}

231static void
232ssi_defer_timer_handler(ngx_event_t *ev)
233{
234 ssi_defer_t *d = ev->data;
235 brix_ssi_session_t *s;
236 brix_ssi_req_t *rq;
237
238 /* Liveness gate: a NULL find means the session/connection is gone (teardown
239 * removed it). Do not touch ctx/c in that case. */
240 s = brix_ssi_registry_find(d->session_key, d->generation);
241 if (s != NULL) {
242 rq = brix_ssi_session_req(s, d->req_id, 0);
243 if (rq != NULL && !rq->responded) {
244 brix_ssi_dlv_kind kind;
245
246 rq->defer_ctx = NULL; /* consume the timer before delivering */
247 ssi_complete_deferred(d->ctx, d->c, s, rq);
248 /* A streamed response is signalled PEND (client pulls the body via
249 * kXR_read); a unary response is pushed inline; an error is terminal. */
250 kind = rq->error ? SSI_DLV_ERROR
251 : rq->streaming ? SSI_DLV_PEND
252 : SSI_DLV_RESPONSE;
253 brix_ssi_deliver(d->ctx, d->c, s, d->req_id, kind);
254 } else if (rq != NULL) {
255 rq->defer_ctx = NULL;
256 }
257 }
258 ngx_free(d);
259}

261static ngx_int_t
262ssi_defer_arm(brix_ctx_t *ctx, ngx_connection_t *c,
263 brix_ssi_session_t *sess, brix_ssi_req_t *rq)
264{
265 ssi_defer_t *d = ngx_alloc(sizeof(*d), c->log);
266
267 if (d == NULL) {
268 return NGX_ERROR;
269 }
270 ngx_memzero(&d->timer, sizeof(d->timer));
271 d->timer.handler = ssi_defer_timer_handler;
272 d->timer.data = d;
273 d->timer.log = c->log;
274 d->ctx = ctx;
275 d->c = c;
276 d->session_key = sess->conn_id;
277 d->generation = sess->generation;
278 d->req_id = rq->req_id;
279 rq->defer_ctx = d;
280 ngx_add_timer(&d->timer, 10); /* 10ms — proves the push without stalling */
281 return NGX_OK;
282}

284/*
285 * Teardown hook (called from brix_free_fhandle before the .ssi slot is cleared,
286 * on both kXR_close and connection disconnect): cancel any armed deferral timers
287 * and unregister the session, so no async completion can run against freed memory.
288 */
289void
290brix_ssi_handle_cleanup(void *ssi_session)
291{
292 brix_ssi_session_t *sess = ssi_session;
293 int i;
294
295 if (sess == NULL) {
296 return;
297 }
298 for (i = 0; i < BRIX_SSI_MAX_INFLIGHT; i++) {
299 if (sess->rr[i].in_use && sess->rr[i].defer_ctx != NULL) {
300 ssi_defer_cancel(sess->rr[i].defer_ctx);
301 sess->rr[i].defer_ctx = NULL;
302 }
303 }
304 brix_ssi_registry_remove(sess->conn_id);
305}

307/* ---- Append this write's payload to the accumulating SSI request ----
308 *
309 * WHAT: Appends n bytes of the current write payload to rq->req, enforcing the
310 * per-request size cap. Returns 1 when accumulation succeeded (including the n==0
311 * no-op case); returns 0 after sending a kXR error (ArgTooLong or NoMemory) and
312 * stashing its status in *out_rc.
313 *
314 * WHY: A single logical SSI request can span several writes; concentrating the
315 * cap check, lazy buffer allocation, and copy in one helper keeps brix_ssi_write
316 * flat and its cyclomatic complexity within budget while preserving the exact
317 * accept/reject behavior clients depend on.
318 *
319 * HOW: 1) Skip when there is nothing to append. 2) Clamp the effective cap to the
320 * configured request_max but never above the compile-time BRIX_SSI_REQ_MAX (the
321 * buffer is sized to that ceiling). 3) Reject when the append would exceed the
322 * cap. 4) Lazily allocate the request buffer. 5) Copy the payload when present
323 * and advance req_len.
324 */
325static int
326ssi_write_accumulate(brix_ctx_t *ctx, ngx_connection_t *c,
327 brix_ssi_session_t *sess, brix_ssi_req_t *rq,
328 size_t n, ngx_int_t *out_rc)
329{
330 size_t reqcap;
331
332 if (n == 0) {
333 return 1;
334 }
335 reqcap = (sess->request_max > 0 && sess->request_max < BRIX_SSI_REQ_MAX)
336 ? sess->request_max : BRIX_SSI_REQ_MAX;
337 if (rq->req_len + n > reqcap) {
338 *out_rc = brix_send_error(ctx, c, kXR_ArgTooLong,
339 "SSI request too large");
340 return 0;
341 }
342 if (rq->req == NULL) {
343 rq->req = ngx_palloc(c->pool, BRIX_SSI_REQ_MAX);
344 if (rq->req == NULL) {
345 *out_rc = brix_send_error(ctx, c, kXR_NoMemory, "ssi req buf");
346 return 0;
347 }
348 }
349 if (ctx->recv.payload != NULL) {
350 ngx_memcpy(rq->req + rq->req_len, ctx->recv.payload, n);
351 rq->req_len += n;
352 }
353 return 1;
354}

356/* ---- Dispatch the SSI request once fully accumulated ----
357 *
358 * WHAT: When the client declared a total request size and that many bytes have
359 * arrived, dispatches the request to the provider and (for deferred services)
360 * arms async completion. Returns 1 when the caller should ack with kXR_ok
361 * (nothing dispatched yet, or a synchronous dispatch); returns 0 when it has
362 * already produced the terminal reply (kXR_waitresp, or a kXR_NoMemory error via
363 * *out_rc) for a deferred dispatch.
364 *
365 * WHY: With no declared size (req_expected == 0, the write-until-read protocol)
366 * dispatch must be deferred to the first read so several writes coalesce into one
367 * request; firing on the first write rejected every following write as "already
368 * dispatched". Isolating this decision keeps the routing exact and the caller
369 * within complexity budget.
370 *
371 * HOW: 1) Return early when fewer than req_expected bytes have accumulated (or no
372 * size was declared). 2) Dispatch and count the request. 3) For a deferred,
373 * not-yet-responded request: capture the submit streamid, mark it waiting, arm
374 * the completion (NoMemory error on failure), and ack with kXR_waitresp — the
375 * response is later pushed via kXR_attn (brix_ssi_deliver). 4) Otherwise let the
376 * caller send the inline kXR_ok.
377 */
378static int
379ssi_write_maybe_dispatch(brix_ctx_t *ctx, ngx_connection_t *c,
380 brix_ssi_session_t *sess, brix_ssi_req_t *rq,
381 ngx_int_t *out_rc)
382{
383 if (!(rq->req_expected > 0 && rq->req_len >= rq->req_expected)) {
384 return 1;
385 }
386 brix_ssi_dispatch(rq, sess->provider.process);
387 BRIX_SRV_METRIC_INC(ctx, ssi_requests_total);
388
389 if (rq->deferred && !rq->responded) {
390 ngx_memcpy(rq->defer_streamid, ctx->recv.cur_streamid,
391 sizeof(rq->defer_streamid));
392 rq->waiting = 1;
393 if (ssi_defer_arm(ctx, c, sess, rq) != NGX_OK) {
394 *out_rc = brix_send_error(ctx, c, kXR_NoMemory, "ssi defer");
395 return 0;
396 }
397 *out_rc = brix_send_waitresp(ctx, c);
398 return 0;
399 }
400 return 1;
401}

403ngx_int_t
404brix_ssi_write(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
405 const unsigned char off8[8])
406{
407 brix_ssi_session_t *sess = ctx->files[idx].ssi;
408 brix_ssi_req_t *rq;
409 int cmd;
410 uint32_t id, size;
411 size_t n = ctx->recv.cur_dlen;
412 ngx_int_t rc = NGX_OK;
413
414 brix_ssi_rrinfo_decode(off8, &cmd, &id, &size);
415
416 if (cmd == BRIX_SSI_CMD_CAN) {
417 brix_ssi_session_drop(sess, id); /* request-phase cancel */
418 return brix_send_ok(ctx, c, NULL, 0);
419 }
420
421 rq = brix_ssi_session_req(sess, id, 1);
422 if (rq == NULL) {
423 return brix_send_error(ctx, c, kXR_Overloaded,
424 "too many concurrent SSI requests");
425 }
426 if (rq->dispatched) {
427 return brix_send_error(ctx, c, kXR_FileLocked,
428 "SSI request already dispatched");
429 }
430
431 if (!rq->have_size) {
432 rq->req_expected = size;
433 rq->have_size = 1;
434 }
435
436 if (!ssi_write_accumulate(ctx, c, sess, rq, n, &rc)) {
437 return rc;
438 }
439
440 if (!ssi_write_maybe_dispatch(ctx, c, sess, rq, &rc)) {
441 return rc;
442 }
443
444 return brix_send_ok(ctx, c, NULL, 0);
445}
