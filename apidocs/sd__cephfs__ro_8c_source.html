1/*
2 * sd_cephfs_ro.c — read-only CephFS-via-RADOS storage driver ("cephfsro").
3 *
4 * WHAT: A Storage Driver that serves a real CephFS filesystem by reading its
5 * on-RADOS structures directly — directory entries from the metadata pool's
6 * omaps, file bytes from the data pool's objects — with no kernel mount, no
7 * MDS, and no libcephfs. It implements the read side of the vtable
8 * (open/pread/fstat/stat/opendir/readdir/getxattr/listxattr); every
9 * mutating slot is absent (the driver advertises no write capability).
10 *
11 * WHY: When a CephFS cannot be mounted but its pools are intact, this is the
12 * rescue path to get the data back. It reuses sd_ceph's cluster/oid layer
13 * for raw RADOS access and the cephfs_layout decoders for the MDS encodings.
14 *
15 * HOW: Path resolution walks dentries from the root inode (1): for each name it
16 * reads the omap key "<name>_head" from the parent directory's fragment
17 * object(s) "<dirino>.<frag>" and decodes the embedded inode. File reads map
18 * (offset,len) through the inode's file_layout_t to "<ino>.<objno>" data
19 * objects. Directory listing enumerates every leaf-fragment omap. xattrs and
20 * symlink targets come straight from the decoded inode.
21 *
22 * SAFETY: a pure-RADOS reader only sees a consistent namespace on a QUIESCED fs
23 * (MDS journal flushed, no live writers). init() refuses to bind unless the
24 * operator asserts this via `assume_quiesced` (no active MDS probing — by
25 * design). See the program design doc referenced in sd_ceph.h.
26 *
27 * LAYOUT: this driver is split across three TUs for file size / auditability —
28 * this file (lifecycle + open/read/stat/xattr + the driver descriptor),
29 * sd_cephfs_ro_resolve.c (path resolution + retry policy) and
30 * sd_cephfs_ro_dir.c (directory iteration). Cross-TU symbols live in
31 * sd_cephfs_ro_internal.h.
32 *
33 * The whole driver compiles only when the build found librados (BRIX_HAVE_CEPH);
34 * otherwise this is an empty translation unit and the registry never references
35 * the symbol.
36 */

79static ngx_int_t
80cephfsro_init(brix_sd_instance_t *inst, void *driver_conf)
81{
82 brix_sd_cephfs_ro_conf_t *dc = driver_conf;
83 cephfsro_state_t *st;
84 int err = 0;
85
86 if (dc == NULL || dc->meta_pool == NULL || dc->data_pool == NULL) {
87 errno = EINVAL;
88 return NGX_ERROR;
89 }
90 /* operator safety assertion — refuse to bind unless the consistency model is
91 * acknowledged: `quiesced` (fs frozen) OR `live` (still mounted; best-effort
92 * eventually-consistent, with optimistic revalidation + retry). */
93 if (!dc->assume_quiesced && !dc->live) {
94 errno = EPERM;
95 return NGX_ERROR;
96 }
97
98 st = ngx_pcalloc(inst->pool, sizeof(*st));
99 if (st == NULL) { errno = ENOMEM; return NGX_ERROR; }
100 st->live = dc->live ? 1 : 0;
101 st->max_retry = CEPHFS_RO_MAX_RETRY;
102 inst->state = st;
103
104 st->meta = cephfsro_connect_pool(dc, dc->meta_pool, inst->pool, &err);
105 if (st->meta == NULL) { errno = err ? err : EIO; return NGX_ERROR; }
106 st->data = cephfsro_connect_pool(dc, dc->data_pool, inst->pool, &err);
107 if (st->data == NULL) {
108 sd_ceph_conn_destroy(st->meta);
109 st->meta = NULL;
110 errno = err ? err : EIO;
111 return NGX_ERROR;
112 }
113 return NGX_OK;
114}

130static brix_sd_obj_t *
131cephfsro_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
132 mode_t mode, int *err_out)
133{
134 cephfsro_state_t *st = inst->state;
135 cephfs_dentry_t dn;
136 brix_sd_obj_t *obj;
137 cephfsro_obj_t *os;
138 int rc;
139
140 (void) mode;
141
142 /* read-only: reject any write/create intent up front */
143 if (sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC)) {
144 if (err_out != NULL) { *err_out = EROFS; }
145 return NULL;
146 }
147
148 rc = cephfsro_resolve_retry(st, path, &dn);
149 if (rc != 0) {
150 if (err_out != NULL) { *err_out = (rc == 1) ? ENOENT : errno; }
151 return NULL;
152 }
153 if (dn.kind != CEPHFS_DENTRY_PRIMARY) {
154 /* a remote (hardlink) target: we have only ino+d_type, not a full inode */
155 if (err_out != NULL) { *err_out = ENOTSUP; }
156 return NULL;
157 }
158 if ((sd_flags & BRIX_SD_O_DIR) && !cephfs_mode_is_dir(dn.inode.mode)) {
159 if (err_out != NULL) { *err_out = ENOTDIR; }
160 return NULL;
161 }
162
163 obj = ngx_pcalloc(inst->pool, sizeof(*obj));
164 os = ngx_pcalloc(inst->pool, sizeof(*os));
165 if (obj == NULL || os == NULL) {
166 if (err_out != NULL) { *err_out = ENOMEM; }
167 return NULL;
168 }
169 os->ino = dn.inode.ino;
170 os->size = dn.inode.size;
171 os->layout = dn.inode.layout;
172 os->is_dir = cephfs_mode_is_dir(dn.inode.mode) ? 1 : 0;
173
174 obj->driver = &brix_sd_cephfs_ro_driver;
175 obj->inst = inst;
176 obj->fd = NGX_INVALID_FILE; /* no kernel fd — memory-backed */
177 obj->state = os;
178 cephfsro_stat_from_inode(&dn.inode, &obj->snap);
179 return obj;
180}

191/* Map a file offset to its data object + in-object offset via the file layout
192 * (the standard Ceph striping math), and return how many contiguous bytes can be
193 * read there (bounded by the stripe unit). su==os, sc==1 is the default layout. */
194static size_t
195cephfsro_map(const cephfs_layout_t *L, off_t off, uint64_t *objno_out,
196 off_t *objoff_out)
197{
198 uint64_t su = L->stripe_unit;
199 uint64_t sc = (L->stripe_count > 0) ? L->stripe_count : 1;
200 uint64_t os = L->object_size;
201 uint64_t blockno, blockoff, stripeno, stripepos, objsetno, opo;
202
203 if (su == 0 || os == 0 || (os % su) != 0) {
204 return 0; /* implausible layout */
205 }
206 blockno = (uint64_t) off / su;
207 blockoff = (uint64_t) off % su;
208 stripeno = blockno / sc;
209 stripepos = blockno % sc;
210 objsetno = stripeno / (os / su);
211 opo = stripeno % (os / su);
212
213 *objno_out = objsetno * sc + stripepos;
214 *objoff_out = (off_t) (opo * su + blockoff);
215 return (size_t) (su - blockoff); /* contiguous run within this stripe unit */
216}

218static ssize_t
219cephfsro_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
220{
221 cephfsro_state_t *st = obj->inst->state;
222 cephfsro_obj_t *os = obj->state;
223 uint64_t objno = 0;
224 off_t objoff = 0;
225 size_t run;
226 char oid[64];
227 ssize_t n;
228
229 if (off < 0) { errno = EINVAL; return -1; }
230 if ((uint64_t) off >= os->size) { return 0; } /* at/after EOF */
231
232 run = cephfsro_map(&os->layout, off, &objno, &objoff);
233 if (run == 0) { errno = EIO; return -1; }
234 if (run > len) { run = len; }
235 if ((uint64_t) off + run > os->size) { /* clamp to file size */
236 run = (size_t) (os->size - (uint64_t) off);
237 }
238
239 snprintf(oid, sizeof(oid), "%llx.%08llx",
240 (unsigned long long) os->ino, (unsigned long long) objno);
241 {
242 int attempt;
243 for (attempt = 0; ; attempt++) {
244 n = sd_ceph_oid_read(st->data, oid, buf, run, objoff);
245 if (n >= 0) { break; }
246 if (errno == ENOENT) { /* sparse hole within EOF → zeros */
247 memset(buf, 0, run);
248 return (ssize_t) run;
249 }
250 if (cephfsro_is_transient(errno) && attempt < st->max_retry) {
251 cephfsro_backoff(attempt);
252 continue;
253 }
254 return -1;
255 }
256 }
257 if (n == 0 && run > 0) { /* object shorter than expected */
258 memset(buf, 0, run);
259 return (ssize_t) run;
260 }
261 return n;
262}

264static ssize_t
265cephfsro_preadv(brix_sd_obj_t *obj, const struct iovec *iov, int iovcnt,
266 off_t off)
267{
268 ssize_t total = 0;
269 int i;
270
271 for (i = 0; i < iovcnt; i++) {
272 ssize_t n = cephfsro_pread(obj, iov[i].iov_base, iov[i].iov_len, off);
273 if (n < 0) { return (total > 0) ? total : -1; }
274 total += n;
275 off += n;
276 if ((size_t) n < iov[i].iov_len) { break; } /* short read → stop */
277 }
278 return total;
279}

318static ssize_t
319cephfsro_getxattr(brix_sd_instance_t *inst, const char *path, const char *name,
320 void *buf, size_t cap)
321{
322 cephfs_dentry_t dn;
323 uint32_t i;
324
325 if (cephfsro_resolve_xattrs(inst, path, &dn) != 0) { return -1; }
326 for (i = 0; i < dn.nxattrs; i++) {
327 if (dn.xattrs[i].name_len == strlen(name)
328 && memcmp(dn.xattrs[i].name, name, dn.xattrs[i].name_len) == 0)
329 {
330 if (cap == 0) { return (ssize_t) dn.xattrs[i].val_len; } /* size probe */
331 if (dn.xattrs[i].val_len > cap) { errno = ERANGE; return -1; }
332 memcpy(buf, dn.xattrs[i].val, dn.xattrs[i].val_len);
333 return (ssize_t) dn.xattrs[i].val_len;
334 }
335 }
336 errno = ENODATA;
337 return -1;
338}

340static ssize_t
341cephfsro_listxattr(brix_sd_instance_t *inst, const char *path, void *buf,
342 size_t cap)
343{
344 cephfs_dentry_t dn;
345 uint32_t i;
346 size_t need = 0;
347 char *out = buf;
348
349 if (cephfsro_resolve_xattrs(inst, path, &dn) != 0) { return -1; }
350 for (i = 0; i < dn.nxattrs; i++) {
351 need += dn.xattrs[i].name_len + 1; /* name + NUL */
352 }
353 if (cap == 0) { return (ssize_t) need; } /* size probe */
354 if (need > cap) { errno = ERANGE; return -1; }
355 for (i = 0; i < dn.nxattrs; i++) {
356 memcpy(out, dn.xattrs[i].name, dn.xattrs[i].name_len);
357 out += dn.xattrs[i].name_len;
358 *out++ = '\0';
359 }
360 return (ssize_t) need;
361}
