1/*
2 * sd_cephfs_ro.c — read-only CephFS-via-RADOS storage driver ("cephfsro").
3 *
4 * WHAT: A Storage Driver that serves a real CephFS filesystem by reading its
5 * on-RADOS structures directly — directory entries from the metadata pool's
6 * omaps, file bytes from the data pool's objects — with no kernel mount, no
7 * MDS, and no libcephfs. It implements the read side of the vtable
8 * (open/pread/fstat/stat/opendir/readdir/getxattr/listxattr); every
9 * mutating slot is absent (the driver advertises no write capability).
10 *
11 * WHY: When a CephFS cannot be mounted but its pools are intact, this is the
12 * rescue path to get the data back. It reuses sd_ceph's cluster/oid layer
13 * for raw RADOS access and the cephfs_layout decoders for the MDS encodings.
14 *
15 * HOW: Path resolution walks dentries from the root inode (1): for each name it
16 * reads the omap key "<name>_head" from the parent directory's fragment
17 * object(s) "<dirino>.<frag>" and decodes the embedded inode. File reads map
18 * (offset,len) through the inode's file_layout_t to "<ino>.<objno>" data
19 * objects. Directory listing enumerates every leaf-fragment omap. xattrs and
20 * symlink targets come straight from the decoded inode.
21 *
22 * SAFETY: a pure-RADOS reader only sees a consistent namespace on a QUIESCED fs
23 * (MDS journal flushed, no live writers). init() refuses to bind unless the
24 * operator asserts this via `assume_quiesced` (no active MDS probing — by
25 * design). See the program design doc referenced in sd_ceph.h.
26 *
27 * The whole driver compiles only when the build found librados (BRIX_HAVE_CEPH);
28 * otherwise this is an empty translation unit and the registry never references
29 * the symbol.
30 */

54 * walk-version revalidation (detect an MDS write that landed mid-walk and retry)

100/* ---- retry policy --------------------------------------------------------- */
101
102/* A transient cluster error is worth retrying regardless of consistency mode
103 * (cluster blip, peering, blocklist churn); a permanent one (ENOENT/EACCES/…) is
104 * fast-failed. EIO is treated as transient because librados surfaces several
105 * recoverable conditions as EIO. */
106static int
107cephfsro_is_transient(int err)
108{
109 switch (err) {
110 case EAGAIN: case ETIMEDOUT: case EBUSY: case EINTR: case EIO:
111 return 1;
112 default:
113 return 0;
114 }
115}

117/* Sleep an exponentially-growing, jittered, capped backoff before a retry. Held
118 * only on the (rare) error/race path; the ceph data path is synchronous anyway. */
119static void
120cephfsro_backoff(int attempt)
121{
122 long us = (long) CEPHFS_RO_BACKOFF_US << attempt;
123 struct timespec ts;
124
125 if (us > 50000) { us = 50000; } /* cap at 50 ms */
126 us += (long) (us / 4) ? (long) (clock() % (us / 4 + 1)) : 0; /* light jitter */
127 ts.tv_sec = us / 1000000;
128 ts.tv_nsec = (us % 1000000) * 1000;
129 nanosleep(&ts, NULL);
130}

134/* Read a single omap value by key from `oid` into `buf`. On success returns the
135 * value length (>=0) via *outlen and 0; returns -1/errno on a RADOS error and 1
136 * when the key is simply absent (so callers distinguish "not found" from error).
137 * When `version` is non-NULL it receives the RADOS object version at read time
138 * (for optimistic revalidation). */
139static int
140cephfsro_omap_get1(rados_ioctx_t ioctx, const char *oid, const char *key,
141 void *buf, size_t cap, size_t *outlen, uint64_t *version)
142{
143 const char *keys[1];
144 rados_omap_iter_t it = NULL;
145 rados_read_op_t op;
146 int prval = 0, rc, found = 0;
147 char *k = NULL, *v = NULL;
148 size_t vlen = 0;
149
150 keys[0] = key;
151 op = rados_create_read_op();
152 rados_read_op_omap_get_vals_by_keys(op, keys, 1, &it, &prval);
153 rc = rados_read_op_operate(op, ioctx, oid, 0);
154 if (rc < 0) {
155 rados_release_read_op(op);
156 errno = -rc;
157 return -1;
158 }
159 if (version != NULL) { *version = rados_get_last_version(ioctx); }
160 if (it != NULL && rados_omap_get_next(it, &k, &v, &vlen) == 0
161 && k != NULL && v != NULL)
162 {
163 size_t n = (vlen < cap) ? vlen : cap;
164 memcpy(buf, v, n);
165 *outlen = n;
166 found = 1;
167 }
168 if (it != NULL) { rados_omap_get_end(it); }
169 rados_release_read_op(op);
170 return found ? 0 : 1;
171}

173/* Current RADOS version of `oid` (a cheap stat op). Returns 0 / -1 (errno). */
174static int
175cephfsro_obj_version(rados_ioctx_t ioctx, const char *oid, uint64_t *version)
176{
177 rados_read_op_t op = rados_create_read_op();
178 uint64_t psize = 0;
179 struct timespec pmt;
180 int prval = 0, rc;
181
182 rados_read_op_stat2(op, &psize, &pmt, &prval);
183 rc = rados_read_op_operate(op, ioctx, oid, 0);
184 rados_release_read_op(op);
185 if (rc < 0) { errno = -rc; return -1; }
186 *version = rados_get_last_version(ioctx);
187 return 0;
188}

190/* Append a (oid, version) record to a walk, if capturing and not full. */
191static void
192cephfsro_walk_add(cephfsro_walk_t *w, const char *oid, uint64_t version)
193{
194 if (w == NULL || w->n >= CEPHFS_RO_MAXDEPTH) { return; }
195 snprintf(w->ent[w->n].oid, sizeof(w->ent[w->n].oid), "%s", oid);
196 w->ent[w->n].version = version;
197 w->n++;
198}

200/* Look up `name` directly beneath directory inode `dirino` whose leaf fragments
201 * are `frags[0..nfrags)`: scan each fragment object for "<name>_head" and decode
202 * the first hit into *out. Scanning every leaf frag is correct for both unsplit
203 * and fragmented dirs (a dentry lives in exactly one frag) and avoids needing the
204 * dentry-hash→frag mapping. When `walk` is non-NULL the dir object the result
205 * depended on (the frag that held the entry, or the first frag searched when
206 * absent) is recorded with its version for later revalidation. Returns 0 (found),
207 * 1 (absent), -1 (error). */
208static int
209cephfsro_lookup(cephfsro_state_t *st, uint64_t dirino, const uint32_t *frags,
210 uint32_t nfrags, const char *name, cephfs_dentry_t *out,
211 cephfsro_walk_t *walk)
212{
213 rados_ioctx_t meta = sd_ceph_conn_ioctx(st->meta);
214 char key[300];
215 char first_oid[64] = "";
216 uint64_t first_ver = 0;
217 int have_first = 0;
218 uint32_t i;
219
220 if (snprintf(key, sizeof(key), "%s_head", name) >= (int) sizeof(key)) {
221 errno = ENAMETOOLONG;
222 return -1;
223 }
224 for (i = 0; i < nfrags; i++) {
225 char oid[64];
226 char val[CEPHFS_DENTRY_VAL_MAX];
227 size_t vlen = 0;
228 uint64_t ver = 0;
229 int rc;
230
231 snprintf(oid, sizeof(oid), "%llx.%08llx",
232 (unsigned long long) dirino, (unsigned long long) frags[i]);
233 rc = cephfsro_omap_get1(meta, oid, key, val, sizeof(val), &vlen, &ver);
234 if (rc < 0) {
235 /* an absent fragment object is not an error — try the next frag */
236 if (errno == ENOENT) { continue; }
237 return -1;
238 }
239 if (!have_first) {
240 snprintf(first_oid, sizeof(first_oid), "%s", oid);
241 first_ver = ver; have_first = 1;
242 }
243 if (rc == 0) {
244 if (cephfs_decode_dentry(val, vlen, out) != 0) {
245 errno = EIO;
246 return -1;
247 }
248 cephfsro_walk_add(walk, oid, ver); /* depended on this frag */
249 return 0;
250 }
251 }
252 /* absent: record the dir we searched so a concurrent change is detectable */
253 if (have_first) { cephfsro_walk_add(walk, first_oid, first_ver); }
254 return 1; /* not found in any fragment */
255}

257/* Fill *out with a synthetic dentry for the CephFS root (inode 1). The root is
258 * assumed unfragmented (the overwhelmingly common case); its single fragment
259 * object is "1.00000000". */
260static void
261cephfsro_fill_root(cephfs_dentry_t *out)
262{
263 memset(out, 0, sizeof(*out));
264 out->kind = CEPHFS_DENTRY_PRIMARY;
265 out->inode.ino = CEPHFS_ROOT_INO;
266 out->inode.mode = CEPHFS_S_IFDIR | 0755;
267 out->inode.nlink = 2;
268 out->nfrags = 1;
269 out->frag_enc[0] = 0;
270}

272/* Resolve an absolute logical path to its dentry by walking from the root. The
273 * leading '/' is optional; "" or "/" resolves to the root. When `walk` is
274 * non-NULL, the dir objects each component depended on are recorded with their
275 * versions for optimistic revalidation. Returns 0 on success, 1 when a component
276 * is absent (ENOENT), -1 on error. */
277static int
278cephfsro_resolve(cephfsro_state_t *st, const char *path, cephfs_dentry_t *out,
279 cephfsro_walk_t *walk)
280{
281 uint64_t parent_ino = CEPHFS_ROOT_INO;
282 uint32_t parent_frags[CEPHFS_MAX_FRAGS];
283 uint32_t parent_nfrags = 1;
284 const char *p = (path != NULL) ? path : "";
285
286 parent_frags[0] = 0;
287 cephfsro_fill_root(out);
288 if (walk != NULL) { walk->n = 0; }
289
290 while (*p != '\0') {
291 char comp[256];
292 size_t len = 0;
293 int rc;
294
295 while (*p == '/') { p++; }
296 if (*p == '\0') { break; }
297 while (*p != '\0' && *p != '/') {
298 if (len + 1 >= sizeof(comp)) { errno = ENAMETOOLONG; return -1; }
299 comp[len++] = *p++;
300 }
301 comp[len] = '\0';
302
303 rc = cephfsro_lookup(st, parent_ino, parent_frags, parent_nfrags, comp,
304 out, walk);
305 if (rc != 0) { return rc; } /* absent (1) or error (-1) */
306
307 if (out->kind == CEPHFS_DENTRY_REMOTE) {
308 /* a hardlink: terminal use is fine, but we cannot descend through it */
309 while (*p == '/') { p++; }
310 if (*p != '\0') { errno = ENOTDIR; return -1; }
311 return 0;
312 }
313
314 parent_ino = out->inode.ino;
315 parent_nfrags = (out->nfrags < CEPHFS_MAX_FRAGS) ? out->nfrags
316 : CEPHFS_MAX_FRAGS;
317 memcpy(parent_frags, out->frag_enc, parent_nfrags * sizeof(uint32_t));
318 }
319 return 0;
320}

322/* Re-check every dir object a resolve depended on: 1 if any changed version
323 * since the walk (an MDS write landed mid-walk → the result may be inconsistent),
324 * 0 if all stable, and 1 (conservatively "changed") if a probe errors. */
325static int
326cephfsro_walk_changed(cephfsro_state_t *st, const cephfsro_walk_t *w)
327{
328 rados_ioctx_t meta = sd_ceph_conn_ioctx(st->meta);
329 int i;
330
331 for (i = 0; i < w->n; i++) {
332 uint64_t now = 0;
333 if (cephfsro_obj_version(meta, w->ent[i].oid, &now) != 0) {
334 return 1; /* can't confirm stable → retry */
335 }
336 if (now != w->ent[i].version) { return 1; }
337 }
338 return 0;
339}

341/* Resolve with the consistency policy applied: bounded retries on a transient
342 * cluster error always, and — in live mode — a retry when optimistic
343 * revalidation shows a dir on the path changed during the walk (so an infrequent
344 * MDS sync that lands mid-resolve is picked up rather than served torn/stale).
345 * A genuine, stable not-found or permanent error fast-fails. Returns 0 / 1 / -1
346 * like cephfsro_resolve, with errno set. */
347static int
348cephfsro_resolve_retry(cephfsro_state_t *st, const char *path,
349 cephfs_dentry_t *out)
350{
351 int attempt;
352
353 for (attempt = 0; ; attempt++) {
354 cephfsro_walk_t walk;
355 cephfsro_walk_t *wp = st->live ? &walk : NULL;
356 int rc;
357
358 walk.n = 0;
359 rc = cephfsro_resolve(st, path, out, wp);
360
361 if (rc == 0) { /* found */
362 if (wp != NULL && cephfsro_walk_changed(st, wp)
363 && attempt < st->max_retry)
364 {
365 cephfsro_backoff(attempt); /* raced — re-resolve */
366 continue;
367 }
368 return 0;
369 }
370 if (rc < 0 && cephfsro_is_transient(errno)
371 && attempt < st->max_retry)
372 {
373 cephfsro_backoff(attempt); /* transient — retry */
374 continue;
375 }
376 if (rc == 1 && wp != NULL && cephfsro_walk_changed(st, wp)
377 && attempt < st->max_retry)
378 {
379 cephfsro_backoff(attempt); /* not-found but path raced — retry */
380 continue;
381 }
382 return rc; /* stable result */
383 }
384}

417static ngx_int_t
418cephfsro_init(brix_sd_instance_t *inst, void *driver_conf)
419{
420 brix_sd_cephfs_ro_conf_t *dc = driver_conf;
421 cephfsro_state_t *st;
422 int err = 0;
423
424 if (dc == NULL || dc->meta_pool == NULL || dc->data_pool == NULL) {
425 errno = EINVAL;
426 return NGX_ERROR;
427 }
428 /* operator safety assertion — refuse to bind unless the consistency model is
429 * acknowledged: `quiesced` (fs frozen) OR `live` (still mounted; best-effort
430 * eventually-consistent, with optimistic revalidation + retry). */
431 if (!dc->assume_quiesced && !dc->live) {
432 errno = EPERM;
433 return NGX_ERROR;
434 }
435
436 st = ngx_pcalloc(inst->pool, sizeof(*st));
437 if (st == NULL) { errno = ENOMEM; return NGX_ERROR; }
438 st->live = dc->live ? 1 : 0;
439 st->max_retry = CEPHFS_RO_MAX_RETRY;
440 inst->state = st;
441
442 st->meta = cephfsro_connect_pool(dc, dc->meta_pool, inst->pool, &err);
443 if (st->meta == NULL) { errno = err ? err : EIO; return NGX_ERROR; }
444 st->data = cephfsro_connect_pool(dc, dc->data_pool, inst->pool, &err);
445 if (st->data == NULL) {
446 sd_ceph_conn_destroy(st->meta);
447 st->meta = NULL;
448 errno = err ? err : EIO;
449 return NGX_ERROR;
450 }
451 return NGX_OK;
452}

468static brix_sd_obj_t *
469cephfsro_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
470 mode_t mode, int *err_out)
471{
472 cephfsro_state_t *st = inst->state;
473 cephfs_dentry_t dn;
474 brix_sd_obj_t *obj;
475 cephfsro_obj_t *os;
476 int rc;
477
478 (void) mode;
479
480 /* read-only: reject any write/create intent up front */
481 if (sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC)) {
482 if (err_out != NULL) { *err_out = EROFS; }
483 return NULL;
484 }
485
486 rc = cephfsro_resolve_retry(st, path, &dn);
487 if (rc != 0) {
488 if (err_out != NULL) { *err_out = (rc == 1) ? ENOENT : errno; }
489 return NULL;
490 }
491 if (dn.kind != CEPHFS_DENTRY_PRIMARY) {
492 /* a remote (hardlink) target: we have only ino+d_type, not a full inode */
493 if (err_out != NULL) { *err_out = ENOTSUP; }
494 return NULL;
495 }
496 if ((sd_flags & BRIX_SD_O_DIR) && !cephfs_mode_is_dir(dn.inode.mode)) {
497 if (err_out != NULL) { *err_out = ENOTDIR; }
498 return NULL;
499 }
500
501 obj = ngx_pcalloc(inst->pool, sizeof(*obj));
502 os = ngx_pcalloc(inst->pool, sizeof(*os));
503 if (obj == NULL || os == NULL) {
504 if (err_out != NULL) { *err_out = ENOMEM; }
505 return NULL;
506 }
507 os->ino = dn.inode.ino;
508 os->size = dn.inode.size;
509 os->layout = dn.inode.layout;
510 os->is_dir = cephfs_mode_is_dir(dn.inode.mode) ? 1 : 0;
511
512 obj->driver = &brix_sd_cephfs_ro_driver;
513 obj->inst = inst;
514 obj->fd = NGX_INVALID_FILE; /* no kernel fd — memory-backed */
515 obj->state = os;
516 cephfsro_stat_from_inode(&dn.inode, &obj->snap);
517 return obj;
518}

529/* Map a file offset to its data object + in-object offset via the file layout
530 * (the standard Ceph striping math), and return how many contiguous bytes can be
531 * read there (bounded by the stripe unit). su==os, sc==1 is the default layout. */
532static size_t
533cephfsro_map(const cephfs_layout_t *L, off_t off, uint64_t *objno_out,
534 off_t *objoff_out)
535{
536 uint64_t su = L->stripe_unit;
537 uint64_t sc = (L->stripe_count > 0) ? L->stripe_count : 1;
538 uint64_t os = L->object_size;
539 uint64_t blockno, blockoff, stripeno, stripepos, objsetno, opo;
540
541 if (su == 0 || os == 0 || (os % su) != 0) {
542 return 0; /* implausible layout */
543 }
544 blockno = (uint64_t) off / su;
545 blockoff = (uint64_t) off % su;
546 stripeno = blockno / sc;
547 stripepos = blockno % sc;
548 objsetno = stripeno / (os / su);
549 opo = stripeno % (os / su);
550
551 *objno_out = objsetno * sc + stripepos;
552 *objoff_out = (off_t) (opo * su + blockoff);
553 return (size_t) (su - blockoff); /* contiguous run within this stripe unit */
554}

556static ssize_t
557cephfsro_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
558{
559 cephfsro_state_t *st = obj->inst->state;
560 cephfsro_obj_t *os = obj->state;
561 uint64_t objno = 0;
562 off_t objoff = 0;
563 size_t run;
564 char oid[64];
565 ssize_t n;
566
567 if (off < 0) { errno = EINVAL; return -1; }
568 if ((uint64_t) off >= os->size) { return 0; } /* at/after EOF */
569
570 run = cephfsro_map(&os->layout, off, &objno, &objoff);
571 if (run == 0) { errno = EIO; return -1; }
572 if (run > len) { run = len; }
573 if ((uint64_t) off + run > os->size) { /* clamp to file size */
574 run = (size_t) (os->size - (uint64_t) off);
575 }
576
577 snprintf(oid, sizeof(oid), "%llx.%08llx",
578 (unsigned long long) os->ino, (unsigned long long) objno);
579 {
580 int attempt;
581 for (attempt = 0; ; attempt++) {
582 n = sd_ceph_oid_read(st->data, oid, buf, run, objoff);
583 if (n >= 0) { break; }
584 if (errno == ENOENT) { /* sparse hole within EOF → zeros */
585 memset(buf, 0, run);
586 return (ssize_t) run;
587 }
588 if (cephfsro_is_transient(errno) && attempt < st->max_retry) {
589 cephfsro_backoff(attempt);
590 continue;
591 }
592 return -1;
593 }
594 }
595 if (n == 0 && run > 0) { /* object shorter than expected */
596 memset(buf, 0, run);

598 }

645static brix_sd_dir_t *
646cephfsro_opendir(brix_sd_instance_t *inst, const char *path, int *err_out)
647{
648 cephfsro_state_t *st = inst->state;
649 cephfs_dentry_t dn;
650 brix_sd_dir_t *d;
651 cephfsro_dir_t *ds;
652 int rc = cephfsro_resolve_retry(st, path, &dn);
653
654 if (rc != 0) {
655 if (err_out != NULL) { *err_out = (rc == 1) ? ENOENT : errno; }
656 return NULL;
657 }
658 if (dn.kind != CEPHFS_DENTRY_PRIMARY || !cephfs_mode_is_dir(dn.inode.mode)) {
659 if (err_out != NULL) { *err_out = ENOTDIR; }
660 return NULL;
661 }
662
663 d = ngx_pcalloc(inst->pool, sizeof(*d));
664 ds = ngx_pcalloc(inst->pool, sizeof(*ds));
665 if (d == NULL || ds == NULL) {
666 if (err_out != NULL) { *err_out = ENOMEM; }
667 return NULL;
668 }
669 ds->ino = dn.inode.ino;
670 ds->nfrags = (dn.nfrags < CEPHFS_MAX_FRAGS) ? dn.nfrags : CEPHFS_MAX_FRAGS;
671 if (ds->nfrags == 0) { ds->nfrags = 1; ds->frags[0] = 0; }
672 else { memcpy(ds->frags, dn.frag_enc, ds->nfrags * sizeof(uint32_t)); }
673
674 /* prime the first fragment's page so readdir starts serving immediately */
675 if (cephfsro_dir_fetch_page(st, ds) != 0) {
676 if (err_out != NULL) { *err_out = errno; }
677 return NULL;
678 }
679
680 d->inst = inst;
681 d->state = ds;
682 return d;
683}

685/* Strip the trailing "_<snapid>" (e.g. "_head") from a dir omap key to recover
686 * the dentry name. Copies up to the last '_' into out. Returns 0, or -1 if the
687 * key has no '_' (not a dentry key — caller should skip it). */
688static int
689cephfsro_key_to_name(const char *key, char *out, size_t cap)
690{
691 const char *us = strrchr(key, '_');
692 size_t n;
693
694 if (us == NULL) { return -1; }
695 n = (size_t) (us - key);
696 if (n == 0 || n + 1 > cap) { return -1; }
697 memcpy(out, key, n);
698 out[n] = '\0';
699 return 0;
700}

702/* Fetch the next page of dentry names for the directory's current fragment into
703 * ds->names[], advancing the omap cursor. Returns 0 on success (npage set, may be
704 * 0), -1/errno on a RADOS error. */
705static int
706cephfsro_dir_fetch_page(cephfsro_state_t *st, cephfsro_dir_t *ds)
707{
708 rados_ioctx_t meta = sd_ceph_conn_ioctx(st->meta);
709 char oid[64];
710 rados_omap_iter_t it = NULL;
711 rados_read_op_t op;
712 int prval = 0, rc;
713 unsigned char pmore = 0;
714 char *k = NULL, *v = NULL;
715 size_t vlen = 0;
716
717 ds->npage = 0;
718 ds->pi = 0;
719
720 snprintf(oid, sizeof(oid), "%llx.%08llx",
721 (unsigned long long) ds->ino,
722 (unsigned long long) ds->frags[ds->fi]);
723
724 {
725 int attempt;
726 for (attempt = 0; ; attempt++) {
727 op = rados_create_read_op();
728 rados_read_op_omap_get_vals2(op, ds->last_key[0] ? ds->last_key : "",
729 "", CEPHFS_READDIR_PAGE, &it, &pmore,
730 &prval);
731 rc = rados_read_op_operate(op, meta, oid, 0);
732 if (rc >= 0) { break; }
733 rados_release_read_op(op);
734 if (-rc == ENOENT) { ds->more_in_frag = 0; return 0; } /* no such frag */
735 if (cephfsro_is_transient(-rc) && attempt < st->max_retry) {
736 cephfsro_backoff(attempt);
737 continue;
738 }
739 errno = -rc;
740 return -1;
741 }
742 }
743 while (it != NULL && rados_omap_get_next(it, &k, &v, &vlen) == 0 && k != NULL) {
744 char name[256];
745 (void) v; (void) vlen;
746 if (ds->npage < CEPHFS_READDIR_PAGE
747 && cephfsro_key_to_name(k, name, sizeof(name)) == 0)
748 {
749 ngx_memcpy(ds->names[ds->npage], name, strlen(name) + 1);
750 ds->npage++;
751 }
752 ngx_memcpy(ds->last_key, k, strlen(k) + 1); /* cursor = last key seen */
753 }
754 if (it != NULL) { rados_omap_get_end(it); }
755 rados_release_read_op(op);
756 ds->more_in_frag = (pmore != 0);
757 return 0;
758}

760static ngx_int_t
761cephfsro_readdir(brix_sd_dir_t *d, brix_sd_dirent_t *out)
762{
763 cephfsro_state_t *st = d->inst->state;
764 cephfsro_dir_t *ds = d->state;
765
766 for (;;) {
767 if (ds->pi < ds->npage) {
768 ngx_memcpy(out->name, ds->names[ds->pi], strlen(ds->names[ds->pi]) + 1);
769 ds->pi++;
770 return NGX_OK;
771 }
772 /* current page exhausted: more in this frag? */
773 if (ds->more_in_frag) {
774 if (cephfsro_dir_fetch_page(st, ds) != 0) { return NGX_ERROR; }
775 continue;
776 }
777 /* advance to the next fragment */
778 ds->fi++;
779 if (ds->fi >= ds->nfrags) { return NGX_DONE; } /* end of directory */
780 ds->last_key[0] = '\0';
781 ds->more_in_frag = 1; /* force a first fetch */
782 if (cephfsro_dir_fetch_page(st, ds) != 0) { return NGX_ERROR; }
783 }
784}
