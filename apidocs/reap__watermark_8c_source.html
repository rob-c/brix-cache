11ngx_uint_t
12brix_cache_watermark_purge(ngx_stream_brix_srv_conf_t *conf, ngx_log_t *log)
13{
14 brix_cache_fs_usage_t usage;
15 char lock_path[PATH_MAX];
16 ngx_uint_t evicted_files = 0;
17 uint64_t evicted_bytes = 0;
18 /* §14a unification: a cache is active under EITHER the legacy activation
19 * (conf->cache, set by brix_cache on) OR the composable tier grammar (a
20 * cache_store, conf->cache==0). The physical dir the reaper walks comes from
21 * brix_cache_state_root, now tier-aware (returns the posix cache_store dir). */
22 int cache_active = (conf->cache != 0)
23 || (conf->common.cache_store.len > 0);
24 const char *phys_root = brix_cache_state_root(conf);
25
26 if (conf == NULL || !cache_active || phys_root == NULL
27 || conf->reaper.high_watermark == 0
28 || conf->reaper.high_watermark >= BRIX_CACHE_PPM_FULL_SCALE)
29 {
30 return 0;
31 }
32
33 /*
34 * Cheap pre-check off the TTL-cached sampler (1s) — avoid taking the
35 * cross-worker lock while the cache is calm. A statvfs failure here is a
36 * monitoring fault, not a cache fault: log-and-skip rather than wedge.
37 */
38 if (brix_cache_fs_usage_sampled((char *) phys_root, 1000, &usage) != NGX_OK)
39 {
40 ngx_log_error(NGX_LOG_WARN, log, errno,
41 "brix: watermark reaper could not stat cache root \"%s\"",
42 phys_root);
43 return 0;
44 }
45
46 /* Publish occupancy as a gauge every tick, whether or not we purge. */
47 brix_metric_cache_usage_ratio(usage.occupancy_ppm);
48
49 if (usage.occupancy_ppm <= conf->reaper.high_watermark) {
50 return 0; /* below the high mark — nothing to do */
51 }
52
53 ngx_memzero(lock_path, sizeof(lock_path));
54 if (brix_cache_try_evict_lock(conf, lock_path, sizeof(lock_path), log)
55 != NGX_OK)
56 {
57 return 0; /* another worker is purging */
58 }
59
60 /* Reap down to the LOW watermark (hysteresis), oldest-first. No connection
61 * context (NULL ctx/c): the dedicated SHM metrics are emitted below. */
62 (void) brix_cache_purge_to_target(conf, NULL, NULL, NULL,
63 conf->reaper.low_watermark, log, &evicted_files, &evicted_bytes);
64
65 brix_cache_evict_unlock(lock_path);
66
67 if (evicted_files > 0) {
68 brix_metric_cache_watermark_purge(evicted_files, evicted_bytes);
69 ngx_log_error(NGX_LOG_NOTICE, log, 0,
70 "brix: watermark reaper purged %ui file(s), %uL bytes "
71 "from \"%s\" (low=0.%06ui)",
72 evicted_files, (uint64_t) evicted_bytes,
73 phys_root, conf->reaper.low_watermark);
74 }
75 return evicted_files;
76}
