4/*
5 * Path resolution strategy for namespace operations (Phase 8: existence is
6 * verified with brix_stat_beneath, not realpath; confinement is the kernel's
7 * RESOLVE_BENEATH at the op).
8 *
9 * EXISTING — target must already exist. Read-side ops (stat, locate,
10 * open-read) and rm/chmod.
11 * WRITE — parent dir must exist, target may not; creating a new file.
12 * NOEXIST — full path may not exist; no existence gate; recursive mkdir.
13 * EITHER — parent-exists (WRITE) OR target-exists (EXISTING); truncate, rmdir.
14 *
15 * NOTE: defined before the module include below so this header stays usable from
16 * the op_table.h / ngx_brix_module.h include cycle (op_table embeds a mode).
17 */
18typedef enum {
19 BRIX_PATH_EXISTING,
20 BRIX_PATH_WRITE,
21 BRIX_PATH_NOEXIST,
22 BRIX_PATH_EITHER,
23} brix_path_mode_t;

27/*
28 * brix_resolve_op_path — extract, depth-check, and resolve a path from the
29 * current request payload in a single call.
30 *
31 * On success: reqpath and resolved are filled; returns NGX_OK.
32 * On failure: sends the appropriate kXR error wire response, stores the nginx
33 * return code in ctx->write_rc, and returns NGX_DONE.
34 * The caller must return ctx->write_rc immediately.
35 *
36 * Parameters:
37 * ctx — connection context (payload + cur_dlen consumed here)
38 * c — nginx connection (for logging and error sending)
39 * op_id — BRIX_OP_* constant for metric tracking
40 * op_name — verb for the access log ("MKDIR", "RMDIR", etc.)
41 * conf — server config (root, depth limit)
42 * mode — path resolution strategy (see brix_path_mode_t above)
43 * reqpath — caller buffer, BRIX_MAX_PATH+1 bytes; filled on NGX_OK
44 * reqpath_sz — sizeof(reqpath)
45 * resolved — caller buffer, PATH_MAX bytes; filled on NGX_OK
46 * resolved_sz — sizeof(resolved)
47 *
48 * Usage:
49 * if (brix_resolve_op_path(ctx, c, BRIX_OP_MKDIR, "MKDIR", conf,
50 * BRIX_PATH_WRITE,
51 * reqpath, sizeof(reqpath),
52 * resolved, sizeof(resolved)) != NGX_OK) {
53 * return ctx->write_rc;
54 * }
55 */
