76static void test_table(void) {
77 static brixautofs_table_t t;
78 brixautofs_table_init(&t);
79
80 CHECK(brixautofs_find_locked(&t, "atlas.cern.ch") == -1, "table: empty find → -1");
81 int i = brixautofs_claim_locked(&t, "atlas.cern.ch");
82 CHECK(i >= 0 && t.slot[i].st == BRIXAUTOFS_MOUNTING, "table: claim → MOUNTING");
83 CHECK(brixautofs_find_locked(&t, "atlas.cern.ch") == i, "table: find claimed slot");
84 brixautofs_commit_locked(&t, i, 4242);
85 CHECK(t.slot[i].st == BRIXAUTOFS_MOUNTED && t.slot[i].pid == 4242,
86 "table: commit → MOUNTED with pid");
87 CHECK(brixautofs_find_pid_locked(&t, 4242) == i, "table: find by pid");
88 CHECK(brixautofs_find_pid_locked(&t, 4243) == -1, "table: unknown pid → -1");
89 brixautofs_release_locked(&t, i);
90 CHECK(brixautofs_find_locked(&t, "atlas.cern.ch") == -1, "table: release frees slot");
91
92 /* error: table full refuses further claims */
93 char name[64];
94 int claimed = 0;
95 for (int k = 0; k < BRIXAUTOFS_MAX_REPOS; k++) {
96 snprintf(name, sizeof(name), "r%d.cern.ch", k);
97 if (brixautofs_claim_locked(&t, name) >= 0) claimed++;
98 }
99 CHECK(claimed == BRIXAUTOFS_MAX_REPOS, "table: fills to capacity");
100 CHECK(brixautofs_claim_locked(&t, "overflow.cern.ch") == -1,
101 "table: full → claim refused");
102}
