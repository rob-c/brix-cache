1/*
2 * zip.h — clean-room ZIP archive member reader (phase-42 W3).
3 *
4 * WHAT: Parse a ZIP archive's End-of-Central-Directory + central directory
5 * (ZIP64-aware), locate a member, and stream its fully-inflated bytes.
6 * Supports the two methods real archives use: STORE (0) and DEFLATE (8) —
7 * matching XRootD's XrdCl ZipArchive reader. The archive is accessed only
8 * through a caller-supplied pread callback, so it works equally over a
9 * remote root:// handle or a local fd.
10 * WHY: XRootD's only at-rest data compression is client-side ZIP member reads
11 * (?xrdcl.unzip=<member> / xrdcp --zip). The server serves the raw .zip
12 * bytes; ALL ZIP logic is here, client-side, zlib-only (no libXrdCl, no
13 * libzip) — keeping libbrix clean-room.
14 * HOW: zlib raw-inflate (windowBits -15) for DEFLATE members, direct copy for
15 * STORE; CRC-32 verified against the central-directory value.
16 *
17 * SECURITY: the archive is untrusted input. Every offset/size is bounds-checked
18 * against the archive size; the entry count and central-directory size are
19 * capped; a member's inflated output is bounded (absolute cap + the declared
20 * uncompressed size) so a zip-bomb cannot exhaust memory/disk. Unsupported
21 * methods are rejected. The caller is responsible for sanitising member names
22 * before using them as local filesystem paths.
23 */

88/* ---- STORE-only ZIP writer (phase-42 W3 write side) ----------------------- *
89 *
90 * Emits a standards-compliant ZIP archive (method 0 / STORE, matching XRootD's
91 * XrdCl write side which never compresses) through a sequential write callback,
92 * so it streams equally to a local fd or a remote root:// handle. Members are
93 * added from a local fd in two passes (CRC-32 + size, then data) so no data
94 * descriptors are needed and the result is readable by stock `unzip` and XrdCl.
95 * ZIP64 records are emitted automatically when a size/offset exceeds 32 bits or
96 * the entry count exceeds 0xFFFF.
97 */
98

109/*
110 * brix_zip_seed - the existing archive's central directory, for append mode.
111 *
112 * WHAT: bundles the three verbatim-CD facts an append writer needs to resume an
113 * archive — the raw central-directory bytes (`cd`), their length
114 * (`cd_len`), and the entry count they describe (`n`).
115 * WHY: these three values are one cohesive unit (a snapshot of the archive's
116 * existing central directory) and are always produced/consumed together;
117 * carrying them as a single struct keeps brix_zip_writer_new_append() at
118 * ≤5 params without touching the frozen ZIP byte layout.
119 * HOW: filled by the caller from brix_zip_read_eocd() + a read of the old CD,
120 * then passed by const pointer to brix_zip_writer_new_append(), which
121 * copies the bytes into its own buffer (the caller may free `cd` after).
122 */
123typedef struct {
124 const uint8_t *cd; /* verbatim existing central-directory bytes */
125 size_t cd_len; /* length of `cd` in bytes */
126 size_t n; /* number of entries `cd` describes */
127} brix_zip_seed;
