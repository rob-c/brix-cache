81 * rados pools and tape MSS endpoints carry no TCP port). */

84{

86 if (ngx_strcmp(driver, "http") == 0) { return tls ? 443 : 80; }

110/* Parse a port digit run into *portnum. WHY: the [v6]:port and host:port
111 * branches share the exact same validation and operator error. HOW: ngx_atoi
112 * + the 1..65535 range check; tier_fail on anything else. */
113static ngx_int_t
114tier_parse_port(brix_tier_parse_t *p, u_char *digits, size_t len, ngx_int_t *portnum)
115{
116 *portnum = ngx_atoi(digits, len);
117 if (*portnum == NGX_ERROR || *portnum <= 0 || *portnum > 65535) {
118 return tier_fail(p->cf, 1, p->err, p->errcap, "invalid store port");
119 }
120 return NGX_OK;
121}

123/* Parse a bracketed "[v6][:port]" authority into out->host/out->port.
124 * WHY: IPv6 literals embed ':', so the bracket form gets its own scanner.
125 * HOW: find ']', take the bracket interior as the host, parse an optional
126 * ":port" suffix after the bracket. */
127static ngx_int_t
128tier_parse_host_v6(brix_tier_parse_t *p, u_char *authority, size_t authlen)
129{
130 brix_tier_cfg_t *out = p->out;
131 u_char *rb = NULL;
132 size_t i;
133 size_t hostlen;
134 ngx_int_t portnum = 0;
135
136 for (i = 1; i < authlen; i++) {
137 if (authority[i] == ']') { rb = authority + i; break; }
138 }
139 if (rb == NULL) {
140 return tier_fail(p->cf, 1, p->err, p->errcap,
141 "unbalanced \"[\" in store host");
142 }
143 hostlen = (size_t) (rb - (authority + 1));
144 if (rb + 1 < authority + authlen && rb[1] == ':') {
145 if (tier_parse_port(p, rb + 2,
146 (size_t) (authority + authlen - (rb + 2)), &portnum) != NGX_OK)
147 {
148 return NGX_ERROR;
149 }
150 }
151 if (hostlen == 0 || hostlen >= sizeof(out->host)) {
152 return tier_fail(p->cf, 1, p->err, p->errcap, "invalid store host");
153 }
154 ngx_memcpy(out->host, authority + 1, hostlen);
155 out->host[hostlen] = '\0';
156 out->port = (portnum > 0) ? (int) portnum : 0;
157 return NGX_OK;
158}

160/* Parse a plain "host[:port]" authority into out->host/out->port.
161 * WHY: the non-bracket form splits on the LAST ':' (matching the historical
162 * scan direction); a trailing "host:" with no digits is accepted as port-less.
163 * HOW: scan for the last colon, validate the port when digits follow it. */
164static ngx_int_t
165tier_parse_host_plain(brix_tier_parse_t *p, u_char *authority, size_t authlen)
166{
167 brix_tier_cfg_t *out = p->out;
168 u_char *colon = NULL;
169 size_t i;
170 size_t hostlen;
171 ngx_int_t portnum = 0;
172
173 for (i = authlen; i > 0; i--) {
174 if (authority[i - 1] == ':') { colon = authority + i - 1; break; }
175 }
176 if (colon != NULL && colon + 1 < authority + authlen) {
177 hostlen = (size_t) (colon - authority);
178 if (tier_parse_port(p, colon + 1,
179 (size_t) (authority + authlen - (colon + 1)), &portnum) != NGX_OK)
180 {
181 return NGX_ERROR;
182 }
183 } else if (colon != NULL) {
184 hostlen = (size_t) (colon - authority); /* "host:" trailing colon, no port */
185 } else {
186 hostlen = authlen;
187 }
188 if (hostlen == 0 || hostlen >= sizeof(out->host)) {
189 return tier_fail(p->cf, 1, p->err, p->errcap, "invalid store host");
190 }
191 ngx_memcpy(out->host, authority, hostlen);
192 out->host[hostlen] = '\0';
193 out->port = (portnum > 0) ? (int) portnum : 0;
194 return NGX_OK;
195}

197/* Copy the remainder after the authority into out->path.
198 * WHY: "host://path" and "host//path" delimiter styles must land on ONE
199 * canonical absolute path. HOW: collapse a leading "//" to one '/', bound the
200 * length against out->path, copy + NUL-terminate. */
201static ngx_int_t
202tier_copy_remote_path(brix_tier_parse_t *p, u_char *path, size_t pathlen)
203{
204 brix_tier_cfg_t *out = p->out;
205
206 /* Collapse a leading "//" (the "host://path" form) to one '/' so the path is a
207 * single canonical absolute path regardless of host/path delimiter style. */
208 while (pathlen >= 2 && path[0] == '/' && path[1] == '/') {
209 path++;
210 pathlen--;
211 }
212
213 if (pathlen >= sizeof(out->path)) {
214 return tier_fail(p->cf, 1, p->err, p->errcap, "store path too long");
215 }
216 if (pathlen > 0) {
217 ngx_memcpy(out->path, path, pathlen);
218 out->path[pathlen] = '\0';
219 }
220 return NGX_OK;
221}

223/* Parse a remote "//host[:port][/path]" authority (IPv6-bracket aware) into
224 * p->out. WHY: the remote grammar has three independent pieces — the "//"
225 * marker, the host[:port] (bracketed or plain) and the trailing path — each
226 * with its own operator errors. HOW: split the authority at the first '/',
227 * dispatch on the '[' bracket form, then copy the path remainder. */
228static ngx_int_t
229tier_parse_authority(brix_tier_parse_t *p, u_char *loc, size_t loclen)
230{
231 u_char *authority;
232 u_char *path;
233 size_t authlen;
234 size_t pathlen;
235 size_t i;
236 ngx_int_t rc;
237
238 if (loclen < 2 || loc[0] != '/' || loc[1] != '/') {
239 return tier_fail(p->cf, 1, p->err, p->errcap,
240 "remote store needs \"//host\"");
241 }
242 authority = loc + 2;
243 authlen = loclen - 2;
244 for (i = 0; i < authlen; i++) {
245 if (authority[i] == '/') {
246 authlen = i;
247 break;
248 }
249 }
250 path = authority + authlen;
251 pathlen = (size_t) ((loc + loclen) - path);
252
253 if (authlen > 0 && authority[0] == '[') { /* [v6]:port */
254 rc = tier_parse_host_v6(p, authority, authlen);
255 } else { /* host[:port] */
256 rc = tier_parse_host_plain(p, authority, authlen);
257 }
258 if (rc != NGX_OK) {
259 return NGX_ERROR;
260 }
261 return tier_copy_remote_path(p, path, pathlen);
262}

265static ngx_int_t
266tier_parse_args(ngx_conf_t *cf, ngx_array_t *args, brix_tier_cfg_t *out,
267 char *err, size_t errcap)
268{
269 ngx_str_t *a;
270 ngx_uint_t i;
271
272 if (args == NULL) {
273 return NGX_OK;
274 }
275 a = args->elts;
276 for (i = 0; i < args->nelts; i++) {
277 static const char cred[] = "credential=";
278 static const char blk[] = "block_size=";
279
280 if (a[i].len > sizeof(cred) - 1
281 && ngx_strncmp(a[i].data, cred, sizeof(cred) - 1) == 0)
282 {
283 const brix_credential_t *c;
284 char name[256];
285 size_t nl = a[i].len - (sizeof(cred) - 1);
286

288 return tier_fail(cf, 1, err, errcap, "invalid credential name");
289 }

291 name[nl] = '\0';

293 if (c == NULL) {
294 return tier_fail(cf, 1, err, errcap,
295 "no brix_credential \"%s\"", name);

298 } else if (a[i].len > sizeof(blk) - 1

313 "unknown store param \"%.*s\"", (int) a[i].len, a[i].data);

319/* Split "<scheme>:<location>" and resolve the scheme via tier_schemes.
320 * WHY: the scheme is the ONLY driver selector; an unknown scheme is an
321 * operator error (Appendix F). HOW: find the first ':', match the prefix
322 * against the generated table, stamp driver/tls/nearline into p->out and
323 * return the location remainder through loc/loclen. */
324static ngx_int_t
325tier_split_scheme(brix_tier_parse_t *p, ngx_str_t *url, u_char **loc, size_t *loclen)
326{
327 brix_tier_cfg_t *out = p->out;
328 const char *driver = NULL;
329 size_t schemelen;
330 size_t i;
331 int tls = 0;
332 int nearline = 0;
333
334 for (i = 0; i < url->len; i++) {
335 if (url->data[i] == ':') { break; }
336 }
337 if (i == 0 || i >= url->len) {
338 return tier_fail(p->cf, 1, p->err, p->errcap,
339 "store url \"%.*s\" has no scheme", (int) url->len, url->data);
340 }
341 schemelen = i;
342 *loc = url->data + i + 1;
343 *loclen = url->len - i - 1;
344
345 for (i = 0; i < sizeof(tier_schemes) / sizeof(tier_schemes[0]); i++) {
346 if (ngx_strlen(tier_schemes[i].scheme) == schemelen
347 && ngx_strncmp(url->data, tier_schemes[i].scheme, schemelen) == 0)
348 {
349 driver = tier_schemes[i].driver;
350 tls = tier_schemes[i].tls;
351 nearline = tier_schemes[i].nearline;
352 break;
353 }
354 }
355 if (driver == NULL) {
356 return tier_fail(p->cf, 1, p->err, p->errcap,
357 "unknown driver scheme \"%.*s\"", (int) schemelen, url->data);
358 }
359 ngx_cpystrn((u_char *) out->driver, (u_char *) driver, sizeof(out->driver));
360 out->tls = tls;
361 out->nearline = (unsigned) nearline;
362 return NGX_OK;
363}

365/* Resolve a LOCAL (posix/pblock) store location into a confined canonical
366 * path. WHY: local stores must go through brix_prepare_export_root so the
367 * store dir is validated/created exactly like an export root. HOW: wrap the
368 * location in an ngx_str_t, build the per-role opts, canonicalize into
369 * p->out->path. */
370static ngx_int_t
371tier_parse_local(brix_tier_parse_t *p, u_char *loc, size_t loclen)
372{
373 brix_tier_cfg_t *out = p->out;
374 brix_export_root_opts_t opts;
375 ngx_str_t loc_str;
376 char canon[PATH_MAX];
377
378 loc_str.len = loclen;
379 loc_str.data = loc;
380 ngx_memzero(&opts, sizeof(opts));
381 opts.directive_name = tier_role_directive(out->role);
382 opts.allow_write = (out->role != BRIX_TIER_BACKEND) ? 1 : 0;
383 opts.required = 1;
384 opts.canon_size = sizeof(canon);
385
386 if (brix_prepare_export_root(p->cf, &loc_str, &opts, canon) != NGX_CONF_OK) {
387 /* prepare_export_root already emitted an [emerg]. */
388 return tier_fail(p->cf, 0, p->err, p->errcap,
389 "invalid local store path for %s", opts.directive_name);
390 }
391 if (ngx_strlen(canon) >= sizeof(out->path)) {
392 return tier_fail(p->cf, 1, p->err, p->errcap,
393 "local store path too long");
394 }
395 ngx_cpystrn((u_char *) out->path, (u_char *) canon, sizeof(out->path));
396 return NGX_OK;
397}

399/* Resolve a REMOTE store location: authority parse + port defaulting.
400 * WHY: TCP-carrying schemes (xroot/http/s3) MUST end up with a port — either
401 * explicit or the scheme default; port-less schemes (rados pools, tape MSS)
402 * legitimately stay at 0. HOW: parse the //host[:port][/path] authority,
403 * apply tier_default_port, then reject a still-missing port for the TCP
404 * drivers (url kept only for the operator error text). */
405static ngx_int_t
406tier_parse_remote(brix_tier_parse_t *p, ngx_str_t *url, u_char *loc, size_t loclen)
407{
408 brix_tier_cfg_t *out = p->out;
409
410 if (tier_parse_authority(p, loc, loclen) != NGX_OK) {
411 return NGX_ERROR;
412 }
413 if (out->port == 0) {
414 out->port = tier_default_port(out->driver, out->tls);
415 }
416 if (out->port == 0
417 && (ngx_strcmp(out->driver, "xroot") == 0
418 || ngx_strcmp(out->driver, "http") == 0
419 || ngx_strcmp(out->driver, "s3") == 0))
420 {
421 return tier_fail(p->cf, 1, p->err, p->errcap,
422 "remote store \"%.*s\" is missing a port", (int) url->len,
423 url->data);
424 }
425 return NGX_OK;
426}

428/* ---- public: parse a store URL -------------------------------------------- */
429
430ngx_int_t
431brix_tier_parse_store(brix_tier_parse_t *p, ngx_str_t *url, ngx_array_t *args,
432 brix_tier_role_t role)
433{
434 brix_tier_cfg_t *out = p->out;
435 u_char *loc = NULL;
436 size_t loclen = 0;
437 int is_local;
438
439 if (out == NULL || url == NULL || url->len == 0) {
440 return tier_fail(p->cf, 1, p->err, p->errcap, "empty store url");
441 }
442 ngx_memzero(out, sizeof(*out));
443 out->role = role;
444
445 if (tier_split_scheme(p, url, &loc, &loclen) != NGX_OK) {
446 return NGX_ERROR;
447 }
448
449 is_local = (ngx_strcmp(out->driver, "posix") == 0
450 || ngx_strcmp(out->driver, "pblock") == 0);
451
452 if (is_local) {
453 if (tier_parse_local(p, loc, loclen) != NGX_OK) {
454 return NGX_ERROR;
455 }
456 } else if (tier_parse_remote(p, url, loc, loclen) != NGX_OK) {
457 return NGX_ERROR;
458 }
459
460 if (tier_parse_args(p->cf, args, out, p->err, p->errcap) != NGX_OK) {
461 return NGX_ERROR;
462 }
463
464 out->configured = 1;
465 return NGX_OK;
466}
