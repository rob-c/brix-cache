1/*
2 * vfs_writer.c — unified verified write session across every storage backend.
3 *
4 * WHAT: brix_vfs_writer_open/write/commit/abort — one write entry point a
5 * protocol path (GridFTP STOR) uses regardless of backend, with an
6 * optional self-computed read-back integrity check folded in.
7 * WHY: The write mechanics differ per backend — a random-write backend (the
8 * POSIX default export, pblock) is patched in place through an O_WRITE
9 * handle, while an object store (S3) has no seekable file and must be
10 * written as an atomic staged upload — and the verify-on-write logic
11 * (accumulate a CRC, re-read the persisted object, compare) was hand-rolled
12 * in the STOR path. Folding both behind one session gives every filesystem
13 * the same verified-write call and lets a single caller serve S3 too.
14 * HOW: writer_open resolves the backend's CAP_RANDOM_WRITE bit from ctx->sd
15 * (NULL ⇒ the POSIX default export ⇒ random-write): a random backend opens
16 * an in-place handle, an object backend opens a staged upload. Each write is
17 * routed through the driver (brix_vfs_file_pwrite / brix_vfs_staged_write)
18 * and, when verifying, folded into a brix_wverify accumulator. commit closes
19 * or atomically publishes, then — when verifying a non-empty object — drives
20 * brix_vfs_wverify_check over a fresh read-only handle and unlinks on any
21 * mismatch. No goto; early-return per coding-standards.
22 */
23#include "vfs_internal.h"
24#include "core/compat/wverify.h"
25#include "core/compat/copy_range.h" /* brix_copy_range — zero-copy fd ingest */

27struct brix_vfs_writer_s {
28 brix_vfs_ctx_t *ctx; /* pool-owned deep clone (self-contained) */
29 ngx_pool_t *pool;
30 ngx_log_t *log;
31 brix_vfs_file_t *fh; /* random-write path (else NULL) */
32 brix_vfs_staged_t *st; /* staged-upload path (else NULL) */
33 brix_wverify_t *wv; /* verify accumulator (NULL when !verify) */
34 off_t staged_cursor; /* next expected offset on the staged path*/
35 off_t written; /* total bytes written */
36 unsigned random:1; /* 1 = in-place handle, 0 = staged */
37 unsigned verify:1;
38 unsigned finished:1; /* commit or abort has run */
39};

41/* Does the backend behind `ctx` accept an in-place random-offset write? The
42 * POSIX default export has no resolved sd instance (ctx->sd == NULL) and is
43 * always random-write; any other backend must advertise CAP_RANDOM_WRITE. */
44static int
45writer_random_backend(const brix_vfs_ctx_t *ctx)
46{
47 if (ctx->sd == NULL) {
48 return 1;
49 }
50 return (brix_sd_caps(ctx->sd) & BRIX_SD_CAP_RANDOM_WRITE) != 0;
51}
52
53/* Release the descriptor/temp a partially-constructed writer holds without
54 * publishing anything (used on an open-time failure and by writer_abort). The
55 * random path leaves any created object in place — a mid-STOR failure is
56 * REST-resumable — while the staged path drops its unpublished temp. */
57static void
58writer_release(brix_vfs_writer_t *w)
59{
60 if (w->fh != NULL) {
61 brix_vfs_close(w->fh, w->log);
62 w->fh = NULL;
63 }
64 if (w->st != NULL) {
65 brix_vfs_staged_abort(w->st, 1 /* remove temp */);
66 w->st = NULL;
67 }
68 if (w->wv != NULL) {
69 brix_wverify_free(w->wv);
70 w->wv = NULL;
71 }
72}

74brix_vfs_writer_t *
75brix_vfs_writer_open(brix_vfs_ctx_t *ctx, unsigned flags, int verify,
76 int *err_out)
77{
78 brix_vfs_writer_t *w;
79 int verr = 0;
80
81 if (err_out != NULL) {
82 *err_out = 0;
83 }
84 if (ctx == NULL) {
85 if (err_out != NULL) {
86 *err_out = EINVAL;
87 }
88 return NULL;
89 }
90
91 w = ngx_pcalloc(ctx->pool, sizeof(*w));
92 if (w == NULL) {
93 if (err_out != NULL) {
94 *err_out = ENOMEM;
95 }
96 return NULL;
97 }
98 /* Self-contain a deep copy of ctx: a write session outlives the request that
99 * opened it (the caller's cred ctx can be a stack frame), and commit re-derives
100 * from w->ctx for the verify read-back + unlink-on-mismatch. */
101 w->ctx = brix_vfs_ctx_pool_clone(ctx, ctx->pool);
102 if (w->ctx == NULL) {
103 if (err_out != NULL) {
104 *err_out = ENOMEM;
105 }
106 return NULL;
107 }
108 w->pool = ctx->pool;
109 w->log = ctx->log;
110 w->verify = verify ? 1 : 0;
111 /* O_ATOMIC forces the staged temp+publish path even for a random-write backend
112 * so a failed write leaves no partial at the final path (WebDAV/S3 PUT). */
113 w->random = (flags & BRIX_VFS_O_ATOMIC)
114 ? 0
115 : (writer_random_backend(w->ctx) ? 1 : 0);
116
117 if (w->random) {
118 unsigned oflags = BRIX_VFS_O_WRITE | BRIX_VFS_O_CREATE
119 | (flags & BRIX_VFS_O_TRUNC);
120 w->fh = brix_vfs_open(w->ctx, oflags, &verr);
121 if (w->fh == NULL) {
122 if (err_out != NULL) {
123 *err_out = verr ? verr : EIO;
124 }
125 return NULL;
126 }
127 } else {
128 w->st = brix_vfs_staged_open(w->ctx, NGX_FILE_DEFAULT_ACCESS,
129 16 /* excl-name attempts */, &verr);
130 if (w->st == NULL) {
131 if (err_out != NULL) {
132 *err_out = verr ? verr : EIO;
133 }
134 return NULL;
135 }
136 }
137
138 if (w->verify) {
139 w->wv = brix_wverify_begin();
140 if (w->wv == NULL) {
141 writer_release(w);
142 if (err_out != NULL) {
143 *err_out = ENOMEM;
144 }
145 return NULL;
146 }
147 }
148 return w;
149}

151/* Push one extent onto the chosen backend path (no verify bookkeeping). */
152static ngx_int_t
153writer_put(brix_vfs_writer_t *w, const void *buf, size_t len, off_t off)
154{
155 if (w->random) {
156 const u_char *p = buf;
157 size_t left = len;
158
159 while (left > 0) {
160 ssize_t n = brix_vfs_file_pwrite(w->fh, p, left, off);
161 if (n <= 0) {
162 return NGX_ERROR;
163 }
164 p += n;
165 left -= (size_t) n;
166 off += n;
167 }
168 return NGX_OK;
169 }
170
171 /* Staged / object store: no in-place patching — extents must land in order
172 * from offset 0 (the object is built sequentially into the temp/upload). */
173 if (off != w->staged_cursor) {
174 errno = EINVAL;
175 return NGX_ERROR;
176 }
177 if (brix_vfs_staged_write(w->st, buf, len, off) != NGX_OK) {
178 return NGX_ERROR;
179 }
180 w->staged_cursor += (off_t) len;
181 return NGX_OK;
182}

184ngx_int_t
185brix_vfs_writer_write(brix_vfs_writer_t *w, const void *buf, size_t len,
186 off_t off)
187{
188 if (w == NULL || w->finished || buf == NULL || off < 0) {
189 errno = EINVAL;
190 return NGX_ERROR;
191 }
192 if (len == 0) {
193 return NGX_OK;
194 }
195 if (writer_put(w, buf, len, off) != NGX_OK) {
196 return NGX_ERROR;
197 }
198 if (w->wv != NULL && brix_wverify_update(w->wv, buf, off, len) != 0) {
199 errno = EIO;
200 return NGX_ERROR;
201 }
202 w->written += (off_t) len;
203 return NGX_OK;
204}

206/* Read the source fd in 64 KiB chunks and push each through the normal write
207 * engine, which dispatches random/staged and folds the verify CRC. Used whenever
208 * the bytes must pass through userspace: verify is on, the destination is a
209 * staged/object upload, or the random backend has no single seekable fd (pblock). */
210static ngx_int_t
211writer_ingest_bounce(brix_vfs_writer_t *w, int src_fd, off_t src_off, size_t len,
212 off_t dst_off)
213{
214 off_t s = src_off;
215 off_t d = dst_off;
216 size_t left = len;
217
218 while (left > 0) {
219 u_char chunk[65536];
220 size_t want = left < sizeof(chunk) ? left : sizeof(chunk);
221 ssize_t n = pread(src_fd, chunk, want, s);
222
223 if (n < 0) {
224 if (errno == EINTR) {
225 continue;
226 }
227 return NGX_ERROR;
228 }
229 if (n == 0) { /* source shorter than the caller promised */
230 errno = EIO;
231 return NGX_ERROR;
232 }
233 if (brix_vfs_writer_write(w, chunk, (size_t) n, d) != NGX_OK) {
234 return NGX_ERROR;
235 }
236 s += n;
237 d += (off_t) n;
238 left -= (size_t) n;
239 }
240 return NGX_OK;
241}

243ngx_int_t
244brix_vfs_writer_write_fd(brix_vfs_writer_t *w, int src_fd, off_t src_off,
245 size_t len, off_t dst_off)
246{
247 if (w == NULL || w->finished || src_fd < 0 || src_off < 0 || dst_off < 0) {
248 errno = EINVAL;
249 return NGX_ERROR;
250 }
251 if (len == 0) {
252 return NGX_OK;
253 }
254
255 /* Zero-copy fast path: a single-fd, sendfile-capable random backend with no
256 * verify obligation can move the bytes kernel-side without a userspace bounce.
257 * sendfile_fd() returns NGX_INVALID_FILE for an object/block backend (no single
258 * seekable destination), which — like verify and the staged path — falls back
259 * to the bounce so block routing / the CRC accumulator are not bypassed. */
260 if (w->random && !w->verify) {
261 ngx_fd_t dfd = brix_vfs_file_sendfile_fd(w->fh);
262
263 if (dfd != NGX_INVALID_FILE) {
264 if (brix_copy_range(w->log, src_fd, src_off, dfd, dst_off,
265 len, NULL, NULL) != NGX_OK) {
266 return NGX_ERROR;
267 }
268 w->written += (off_t) len;
269 return NGX_OK;
270 }
271 }
272
273 return writer_ingest_bounce(w, src_fd, src_off, len, dst_off);
274}

276off_t
277brix_vfs_writer_expected_off(const brix_vfs_writer_t *w)
278{
279 if (w == NULL) {
280 return -1;
281 }
282 /* Staged/object path: the next byte must land at the sequential cursor. The
283 * random path has no ordering constraint; report the high-water byte count. */
284 return w->random ? w->written : w->staged_cursor;
285}

287/* Re-open the just-committed object read-only and confirm the driver persisted
288 * exactly the CRC-checked bytes. An empty object has no extents to expect and is
289 * trivially complete, so it is not read back. */
290static ngx_int_t
291writer_verify(brix_vfs_writer_t *w)
292{
293 brix_vfs_file_t *rfh;
294 int verr = 0;
295 ngx_int_t rc;
296
297 if (w->wv == NULL || w->written == 0) {
298 return NGX_OK;
299 }
300 rfh = brix_vfs_open(w->ctx, BRIX_VFS_O_READ, &verr);
301 if (rfh == NULL) {
302 return NGX_ERROR;
303 }
304 rc = brix_vfs_wverify_check(w->wv, rfh);
305 brix_vfs_close(rfh, w->log);
306 return rc;
307}

309ngx_int_t
310brix_vfs_writer_commit_ex(brix_vfs_writer_t *w, unsigned excl)
311{
312 if (w == NULL || w->finished) {
313 return NGX_ERROR;
314 }
315 w->finished = 1;
316
317 /* Persist: fsync + close the in-place handle, or atomically publish the temp
318 * onto the final path. A staged commit failure leaves nothing published.
319 * `excl` uses RENAME_NOREPLACE on the staged path (S3 If-None-Match → EEXIST);
320 * it is meaningless for the in-place random path (no separate publish). */
321 if (w->random) {
322 (void) brix_vfs_sync(w->fh);
323 brix_vfs_close(w->fh, w->log);
324 w->fh = NULL;
325 } else {
326 ngx_int_t crc = brix_vfs_staged_commit(w->st, excl);
327 int ce = errno; /* preserve EEXIST (excl create lost the race) */
328
329 brix_vfs_staged_abort(w->st, 0 /* already published/closed */);
330 w->st = NULL;
331 if (crc != NGX_OK) {
332 if (w->wv != NULL) {
333 brix_wverify_free(w->wv);
334 w->wv = NULL;
335 }
336 errno = ce;
337 return NGX_ERROR;
338 }
339 }
340
341 /* Read-back integrity check; a mismatch must never leave a corrupt object. */
342 if (writer_verify(w) != NGX_OK) {
343 (void) brix_vfs_unlink(w->ctx);
344 if (w->wv != NULL) {
345 brix_wverify_free(w->wv);
346 w->wv = NULL;
347 }
348 return NGX_ERROR;
349 }
350 if (w->wv != NULL) {
351 brix_wverify_free(w->wv);
352 w->wv = NULL;
353 }
354 return NGX_OK;
355}
