1/*
2 * pb_wire.c — minimal protobuf wire primitives. See pb_wire.h.
3 *
4 * Protobuf binary encoding: a message is a sequence of (tag, value) pairs where
5 * tag = (field_number << 3) | wire_type. Varints are little-endian base-128 with
6 * the high bit as a continuation flag. All readers are bounds-checked against the
7 * cursor end; input is untrusted.
8 */

13int
14pb_read_varint(pb_reader *r, uint64_t *out)
15{
16 uint64_t v = 0;
17 int shift = 0;
18
19 while (r->p < r->end) {
20 unsigned char b = *r->p++;
21 if (shift <= 63) {
22 v |= (uint64_t) (b & 0x7f) << shift;
23 }
24 if ((b & 0x80) == 0) {
25 *out = v;
26 return 0;
27 }
28 shift += 7;
29 if (shift >= 70) {
30 return -1; /* varint too long (> 10 bytes) */
31 }
32 }
33 return -1; /* ran out of input mid-varint */
34}

36int
37pb_read_tag(pb_reader *r, uint32_t *field, int *wiretype)
38{
39 uint64_t tag;
40
41 if (pb_read_varint(r, &tag) != 0) {
42 return -1;
43 }
44 *wiretype = (int) (tag & 0x7);
45 *field = (uint32_t) (tag >> 3);
46 if (*field == 0) {
47 return -1; /* field number 0 is invalid */
48 }
49 return 0;
50}

52int
53pb_read_len_delim(pb_reader *r, const unsigned char **data, size_t *len)
54{
55 uint64_t n;

62 }
63 *data = r->p;
64 *len = (size_t) n;

69int
70pb_skip_field(pb_reader *r, int wiretype)
71{
72 uint64_t v;
73 const unsigned char *d;
74 size_t n;
75
76 switch (wiretype) {
77 case PB_WT_VARINT:
78 return pb_read_varint(r, &v);
79 case PB_WT_I64:
80 if (r->end - r->p < 8) {
81 return -1;
82 }
83 r->p += 8;
84 return 0;
85 case PB_WT_LEN:
86 return pb_read_len_delim(r, &d, &n);
87 case PB_WT_I32:
88 if (r->end - r->p < 4) {
89 return -1;
90 }
91 r->p += 4;
92 return 0;
93 default:
94 return -1; /* groups (3,4) unsupported / unknown wire type */
95 }
96}

98int
99pb_write_varint(pb_writer *w, uint64_t v)
100{
101 do {
102 unsigned char b = (unsigned char) (v & 0x7f);
103 v >>= 7;
104 if (v != 0) {
105 b |= 0x80;
106 }
107 if (w->len >= w->cap) {
108 return -1;
109 }
110 w->p[w->len++] = b;
111 } while (v != 0);
112 return 0;
113}
