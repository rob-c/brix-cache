1/*
2 * connect_upstream.c — upstream TCP connect, XRootD bootstrap buffer,
3 * TLS handshake callback, and lazy-connect lifecycle.
4 *
5 * WHAT: Builds the 68-byte bootstrap payload (client hello + kXR_protocol
6 * + kXR_login), resolves DNS, creates a non-blocking socket, connects
7 * to an upstream XRootD server, optionally performs TLS, then transitions
8 * into the bootstrap-read state.
9 *
10 * WHY: The proxy lazily opens backend connections on the first post-login opcode
11 * rather than at client login. This avoids idle upstream sockets during
12 * periods of low traffic and lets per-request auth decisions determine
13 * whether a backend connection is worth keeping.
14 *
15 * HOW: brix_proxy_connect() selects an endpoint from pool/rr/single,
16 * resolves via getaddrinfo(), creates non-blocking socket, arms write-event
17 * for async connect, optionally performs TLS, builds bootstrap buffer,
18 * then sets state to XRD_PX_BOOTSTRAP so read_handler parses responses.
19 */

27/* bootstrap frame builder */
28/*
29 * Monotonically increasing counter mixed into the upstream login PID.
30 * xrootd treats repeated logins from the same PID as session reconnects and
31 * applies a backoff delay. Using a unique virtual PID per upstream connection
32 * prevents that stall.
33 */
34static ngx_atomic_t proxy_upstream_seq;
35
36/* Round-robin counter for multiple upstream endpoints. */
37static ngx_atomic_t proxy_upstream_rr;

39/*
40 * Builds the 68-byte bootstrap buffer:
41 * 20 XRootD client hello
42 * 24 kXR_protocol request
43 * 24 kXR_login request
44 *
45 * username: NUL-terminated string for kXR_login (max 8 chars).
46 * NULL or empty → default "xrd".
47 */
48static size_t
49brix_proxy_build_bootstrap(u_char *buf, const char *username)
50{
51 u_char *cursor = buf;
52
53 /* Client hello: 12 zero bytes + protocol version + ROOTD_PQ selector */
54 ngx_memzero(cursor, 12);
55 cursor += 12;
56 {
57 uint32_t v = htonl(4);
58 ngx_memcpy(cursor, &v, 4); cursor += 4;
59 v = htonl(ROOTD_PQ);
60 ngx_memcpy(cursor, &v, 4); cursor += 4;
61 }
62
63 /* kXR_protocol */
64 {
65 ClientProtocolRequest *r = (ClientProtocolRequest *)(void *) cursor;
66 xrdw_protocol_req_t b = { .clientpv = kXR_PROTOCOLVERSION,
67 .expect = 0x03 };
68 ngx_memzero(r, sizeof(*r));
69 r->streamid[0] = 0; r->streamid[1] = 1;
70 r->requestid = htons(kXR_protocol);
71 xrdw_protocol_req_pack(&b, ((ClientRequestHdr *) (void *) cursor)->body);
72 cursor += sizeof(*r);
73 }
74
75 /* kXR_login */
76 {
77 ClientLoginRequest *r = (ClientLoginRequest *)(void *) cursor;
78 xrdw_login_req_t b = { .capver = kXR_ver005 };
79 ngx_atomic_uint_t seq = ngx_atomic_fetch_add(&proxy_upstream_seq, 1);
80
81 b.pid = (int32_t) ((ngx_pid << 16) ^ (seq & 0xFFFF));
82 if (username != NULL && username[0] != '\0') {
83 size_t ulen = ngx_strlen(username);
84 if (ulen > sizeof(b.username)) {
85 ulen = sizeof(b.username);
86 }
87 ngx_memcpy(b.username, username, ulen);
88 } else {
89 b.username[0] = 'x';
90 b.username[1] = 'r';
91 b.username[2] = 'd';
92 }
93
94 ngx_memzero(r, sizeof(*r));
95 r->streamid[0] = 0; r->streamid[1] = 1;
96 r->requestid = htons(kXR_login);
97 xrdw_login_req_pack(&b, ((ClientRequestHdr *) (void *) cursor)->body);
98 cursor += sizeof(*r);
99 }
100
101 return (size_t)(cursor - buf);
102}

186brix_proxy_connect(brix_proxy_ctx_t *proxy,
187 ngx_connection_t *client_conn,
188 ngx_stream_brix_srv_conf_t *conf)

192 int fd;
193 ngx_connection_t *uconn;

201 /* Select upstream: redirected host > pool > round-robin array > single host */
202 if (proxy->redirect_host.len > 0) {
203 use_host = &proxy->redirect_host;
204 use_port = (ngx_int_t) proxy->redirect_port;
205 /* upstream_idx stays what it was, or -1 if we started redirected */
206 } else {
207 /* GSI-as-user connections are per-user authenticated — never reuse a
208 * pooled connection (it carries a different identity). */
209 uconn = (conf->proxy.auth == BRIX_PROXY_AUTH_GSI)
210 ? NULL : brix_proxy_pool_get(proxy, conf, &pooled_idx);
211 if (uconn != NULL) {
212 proxy->conn = uconn;
213 proxy->upstream_idx = pooled_idx;
214 proxy->state = XRD_PX_IDLE;
215 proxy->from_pool = 1;
216 uconn->data = proxy;
217 uconn->log = client_conn->log;
218 uconn->read->log = client_conn->log;
219 uconn->write->log = client_conn->log;
220
221 if (proxy->saved_req != NULL) {
222 brix_proxy_dispatch_pending(proxy);
223 } else {
224 proxy->client_ctx->state = XRD_ST_REQ_HEADER;
225 brix_schedule_read_resume(client_conn);
226 }
227 return NGX_OK;
228 }
229
230 if (conf->proxy.upstreams != NULL && conf->proxy.upstreams->nelts > 0) {
231 brix_proxy_upstream_t *ups = conf->proxy.upstreams->elts;
232 ngx_atomic_uint_t idx, i;
233
234 /* Select a healthy upstream. If EVERY upstream is currently marked
235 * down (and none has aged past the retry window) do NOT fall through
236 * to a known-dead endpoint — fail the connect so the caller returns a
237 * clean error instead of hammering a dead upstream in a tight loop. */
238 int found = 0;
239 idx = ngx_atomic_fetch_add(&proxy_upstream_rr, 1) % conf->proxy.upstreams->nelts;
240 /* proxy_up_status is lazily allocated by the health-tracking path and
241 * is NULL until a failure marks an upstream down (the mark_fail/is_down
242 * accessors are all NULL-tolerant no-ops). Treat a NULL table as
243 * "every upstream healthy" so the round-robin pick stands — the same
244 * semantics, without dereferencing a NULL array. */
245 for (i = 0; proxy_up_status != NULL
246 && i < conf->proxy.upstreams->nelts; i++) {
247 ngx_uint_t cur = (idx + i) % conf->proxy.upstreams->nelts;
248 if (!proxy_up_status[cur].down ||
249 ngx_time() - proxy_up_status[cur].checked >= BRIX_PROXY_FAIL_TIMEOUT)
250 {
251 idx = cur;
252 found = 1;
253 break;
254 }
255 }
256 if (proxy_up_status == NULL) {
257 found = 1; /* no health table → RR pick is authoritative */
258 }
259
260 if (!found) {
261 ngx_log_error(NGX_LOG_ERR, client_conn->log, 0,
262 "xrootd proxy: all %ui upstream(s) down — "
263 "failing request",
264 (ngx_uint_t) conf->proxy.upstreams->nelts);
265 return NGX_ERROR;
266 }
267
268 use_host = &ups[idx].host;
269 use_port = (ngx_int_t) ups[idx].port;
270 proxy->upstream_idx = (int) idx;
271 } else {
272 use_host = &conf->proxy.host;
273 use_port = conf->proxy.port;
274 proxy->upstream_idx = -1;
275 }
276 }
277
278 /* GSI delegation: log in to the upstream AS THE USER in a thread (the blocking
279 * in-process GSI client), then hand the authenticated fd to the relay. */
280 if (conf->proxy.auth == BRIX_PROXY_AUTH_GSI) {
281 return brix_proxy_gsi_connect_async(proxy, conf, use_host,
282 (uint16_t) use_port);
283 }
284
285 {
286 brix_resolve_status_t rstatus;
287
288 fd = brix_resolve_connect_socket((const char *) use_host->data,
289 (unsigned) use_port,
290 BRIX_AF_AUTO,
291 &chosen_addr, &chosen_addrlen,
292 &rstatus);
293 if (fd == (int) NGX_INVALID_FILE) {
294 if (rstatus == BRIX_RESOLVE_ERR_DNS) {
295 ngx_log_error(NGX_LOG_ERR, client_conn->log, 0,
296 "xrootd proxy: cannot resolve \"%s\"",
297 use_host->data);
298 return NGX_ERROR;
299 }

301 "xrootd proxy: no usable address for \"%s\"",
302 use_host->data);

304 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
305 return NGX_ERROR;
306 }
307 }
308
309 uconn = ngx_get_connection(fd, client_conn->log);
310 if (uconn == NULL) {
311 ngx_close_socket(fd);
312 return NGX_ERROR;
313 }
314
315 uconn->pool = ngx_create_pool(512, client_conn->log);
316 if (uconn->pool == NULL) {
317 ngx_free_connection(uconn);
318 ngx_close_socket(fd);
319 return NGX_ERROR;
320 }
321
322 /* Build bootstrap buffer in the upstream connection pool */
323 bsbuf = ngx_palloc(uconn->pool,
324 XRD_HANDSHAKE_LEN
325 + sizeof(ClientProtocolRequest)
326 + sizeof(ClientLoginRequest));
327 if (bsbuf == NULL) {
328 ngx_destroy_pool(uconn->pool);
329 ngx_free_connection(uconn);
330 ngx_close_socket(fd);
331 return NGX_ERROR;
332 }
333 /* Resolve the upstream login username from the configured policy. */
334 {
335 char upstream_user[9];
336
337 switch (conf->proxy.login_user) {
338 case BRIX_PROXY_LOGIN_PASSTHROUGH:
339 if (proxy->client_ctx != NULL
340 && proxy->client_ctx->login.user[0] != '\0')
341 {
342 ngx_cpystrn((u_char *) upstream_user,
343 (u_char *) proxy->client_ctx->login.user,
344 sizeof(upstream_user));
345 } else {
346 upstream_user[0] = '\0'; /* no client username → fall back to "xrd" */
347 }

352 sizeof(upstream_user));
353 break;
354 default: /* BRIX_PROXY_LOGIN_ANONYMOUS */
355 upstream_user[0] = '\0';
356 break;
357 }

360 }
361
362 uconn->data = proxy;
363 uconn->recv = ngx_recv;
364 uconn->send = ngx_send;
365 uconn->recv_chain = ngx_recv_chain;
366 uconn->send_chain = ngx_send_chain;
367 uconn->log = client_conn->log;
368 uconn->read->handler = brix_proxy_read_handler;
369 uconn->write->handler = brix_proxy_write_handler;
370 uconn->read->log = client_conn->log;
371 uconn->write->log = client_conn->log;
372
373 proxy->conn = uconn;
374 proxy->wbuf = bsbuf;
375 proxy->wbuf_len = bslen;
376 proxy->wbuf_pos = 0;
377 proxy->wbuf_owned = 0; /* Phase 39: bsbuf is pool-allocated — pool owns it */
378 proxy->state = XRD_PX_CONNECTING;
379
380 ngx_log_debug(NGX_LOG_DEBUG_STREAM, client_conn->log, 0,
381 "xrootd proxy: connect() to %s:%d",
382 use_host->data, (int) use_port);
383
384 rc = connect(fd, (struct sockaddr *)(void *) &chosen_addr,
385 chosen_addrlen);
386 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
387 ngx_log_error(NGX_LOG_ERR, client_conn->log, ngx_socket_errno,
388 "xrootd proxy: connect to %s:%d failed",
389 use_host->data, (int) use_port);
390 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
391 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
392 brix_proxy_cleanup(proxy);
393 return NGX_ERROR;
394 }
395
396 if (ngx_handle_write_event(uconn->write, 0) != NGX_OK) {
397 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
398 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
399 brix_proxy_cleanup(proxy);
400 return NGX_ERROR;
401 }
402
403 /* Arm connect timeout: fires in write_handler as wev->timedout */
404 if (rc == -1 && conf->proxy.connect_timeout > 0) {
405 ngx_add_timer(uconn->write, conf->proxy.connect_timeout);
406 }
407
408 if (rc == 0) {
409 /* Immediate connect (local loopback etc.) */
410#if (NGX_SSL)
411 if (conf->proxy.upstream_tls && conf->proxy.tls_ctx != NULL) {
412 if (ngx_ssl_create_connection(conf->proxy.tls_ctx, uconn,
413 NGX_SSL_BUFFER | NGX_SSL_CLIENT)
414 != NGX_OK)
415 {
416 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
417 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
418 brix_proxy_cleanup(proxy);
419 return NGX_ERROR;
420 }
421 /* SNI: prefer explicit name directive, fall back to configured host */
422 {
423 const char *sni = (conf->proxy.upstream_tls_name.len > 0)
424 ? (const char *) conf->proxy.upstream_tls_name.data
425 : (const char *) conf->proxy.host.data;
426 SSL_set_tlsext_host_name(uconn->ssl->connection, sni);
427 }
428 uconn->ssl->handler = brix_proxy_tls_handshake_done;
429 proxy->state = XRD_PX_TLS_HANDSHAKE;
430 if (ngx_ssl_handshake(uconn) == NGX_AGAIN) {
431 return NGX_OK; /* TLS callback will continue */
432 }
433 brix_proxy_tls_handshake_done(uconn);
434 return NGX_OK;
435 }
436#endif
437 proxy->state = XRD_PX_BOOTSTRAP;
438 proxy->bs_phase = XRD_PX_BS_HANDSHAKE;
439 proxy->rhdr_pos = 0;
440
441 if (brix_proxy_flush(proxy) == NGX_ERROR) {
442 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
443 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
444 brix_proxy_cleanup(proxy);
445 return NGX_ERROR;
446 }
447 }
448
449 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, client_conn->log, 0,
450 "xrootd proxy: connecting to %s:%d",
451 use_host->data, (int) use_port);
452 return NGX_OK;
453}
