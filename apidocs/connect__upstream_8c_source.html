1/*
2 * connect_upstream.c — upstream TCP connect, XRootD bootstrap buffer,
3 * TLS handshake callback, and lazy-connect lifecycle.
4 *
5 * WHAT: Builds the 68-byte bootstrap payload (client hello + kXR_protocol
6 * + kXR_login), resolves DNS, creates a non-blocking socket, connects
7 * to an upstream XRootD server, optionally performs TLS, then transitions
8 * into the bootstrap-read state.
9 *
10 * WHY: The proxy lazily opens backend connections on the first post-login opcode
11 * rather than at client login. This avoids idle upstream sockets during
12 * periods of low traffic and lets per-request auth decisions determine
13 * whether a backend connection is worth keeping.
14 *
15 * HOW: brix_proxy_connect() selects an endpoint from pool/rr/single,
16 * resolves via getaddrinfo(), creates non-blocking socket, arms write-event
17 * for async connect, optionally performs TLS, builds bootstrap buffer,
18 * then sets state to XRD_PX_BOOTSTRAP so read_handler parses responses.
19 */

52/*
53 * WHAT: Picks a healthy upstream index from the round-robin array, honouring
54 * the lazily-allocated proxy_up_status health table.
55 *
56 * WHY: When every upstream is marked down (and none has aged past the retry
57 * window) we must fail the connect rather than fall through to a known-dead
58 * endpoint and hammer it in a tight loop.
59 *
60 * HOW: Advances the shared RR counter, then walks up to nelts entries from that
61 * start looking for one that is up (or stale enough to retry). A NULL
62 * health table means "all healthy" — the RR pick stands. Returns NGX_OK
63 * with *idx_out set, or NGX_ERROR when all are down.
64 */
65static ngx_int_t
66pc_pick_healthy_upstream(ngx_stream_brix_srv_conf_t *conf,
67 ngx_uint_t *idx_out)
68{
69 ngx_uint_t nelts = conf->proxy.upstreams->nelts;
70 ngx_uint_t idx;
71 ngx_uint_t i;
72 int found = 0;
73
74 idx = ngx_atomic_fetch_add(&proxy_upstream_rr, 1) % nelts;
75
76 /* proxy_up_status is lazily allocated by the health-tracking path and is
77 * NULL until a failure marks an upstream down (the mark_fail/is_down
78 * accessors are all NULL-tolerant no-ops). Treat a NULL table as "every
79 * upstream healthy" so the round-robin pick stands — same semantics,
80 * without dereferencing a NULL array. */
81 for (i = 0; proxy_up_status != NULL && i < nelts; i++) {
82 ngx_uint_t cur = (idx + i) % nelts;
83 if (!proxy_up_status[cur].down ||
84 ngx_time() - proxy_up_status[cur].checked >= BRIX_PROXY_FAIL_TIMEOUT)
85 {
86 idx = cur;
87 found = 1;
88 break;
89 }
90 }
91 if (proxy_up_status == NULL) {
92 found = 1; /* no health table → RR pick is authoritative */
93 }
94
95 if (!found) {
96 return NGX_ERROR;
97 }
98
99 *idx_out = idx;
100 return NGX_OK;
101}

103/*
104 * WHAT: Chooses the upstream endpoint into *tgt (host+port) with priority
105 * redirect > pooled connection > round-robin healthy array > single host.
106 *
107 * WHY: A pooled connection short-circuits the whole connect: it is already
108 * authenticated and bootstrapped, so we adopt it and either dispatch the
109 * saved request or resume the client read loop. GSI-as-user connections
110 * are per-user authenticated and must never reuse a pooled (foreign
111 * identity) connection.
112 *
113 * HOW: Returns NGX_OK when a pooled connection was adopted (caller returns OK to
114 * its own caller — connect is complete), NGX_DECLINED when *tgt was filled
115 * and the caller must proceed to resolve/connect, or NGX_ERROR when all
116 * upstreams are down.
117 */
118static ngx_int_t
119pc_select_endpoint(brix_proxy_ctx_t *proxy, ngx_connection_t *client_conn,
120 ngx_stream_brix_srv_conf_t *conf, pc_target_t *tgt)
121{
122 ngx_connection_t *uconn;
123 int pooled_idx = -1;
124 ngx_uint_t idx;
125
126 if (proxy->redirect_host.len > 0) {
127 tgt->host = &proxy->redirect_host;
128 tgt->port = (ngx_int_t) proxy->redirect_port;
129 /* upstream_idx stays what it was, or -1 if we started redirected */
130 return NGX_DECLINED;
131 }
132
133 /* GSI-as-user connections are per-user authenticated — never reuse a
134 * pooled connection (it carries a different identity). */
135 uconn = (conf->proxy.auth == BRIX_PROXY_AUTH_GSI)
136 ? NULL : brix_proxy_pool_get(proxy, conf, &pooled_idx);
137 if (uconn != NULL) {
138 proxy->conn = uconn;
139 proxy->upstream_idx = pooled_idx;
140 proxy->state = XRD_PX_IDLE;
141 proxy->from_pool = 1;
142 uconn->data = proxy;
143 uconn->log = client_conn->log;
144 uconn->read->log = client_conn->log;
145 uconn->write->log = client_conn->log;
146
147 if (proxy->saved_req != NULL) {
148 brix_proxy_dispatch_pending(proxy);
149 } else {
150 proxy->client_ctx->state = XRD_ST_REQ_HEADER;
151 brix_schedule_read_resume(client_conn);
152 }
153 return NGX_OK;
154 }
155
156 if (conf->proxy.upstreams != NULL && conf->proxy.upstreams->nelts > 0) {
157 brix_proxy_upstream_t *ups = conf->proxy.upstreams->elts;
158
159 if (pc_pick_healthy_upstream(conf, &idx) != NGX_OK) {
160 ngx_log_error(NGX_LOG_ERR, client_conn->log, 0,
161 "xrootd proxy: all %ui upstream(s) down — "
162 "failing request",
163 (ngx_uint_t) conf->proxy.upstreams->nelts);
164 return NGX_ERROR;
165 }
166
167 tgt->host = &ups[idx].host;
168 tgt->port = (ngx_int_t) ups[idx].port;
169 proxy->upstream_idx = (int) idx;
170 } else {
171 tgt->host = &conf->proxy.host;
172 tgt->port = conf->proxy.port;
173 proxy->upstream_idx = -1;
174 }
175
176 return NGX_DECLINED;
177}

179/*
180 * WHAT: Resolves tgt->host:port (AF policy) into a non-blocking socket fd and
181 * records the chosen sockaddr in tgt->addr/addrlen.
182 *
183 * WHY: Endpoint selection yields a host string; the async connect() needs a
184 * concrete socket and address. AF selection is BRIX_AF_AUTO (try every
185 * family) here — the proxy does not constrain the upstream family.
186 *
187 * HOW: Delegates to brix_resolve_connect_socket() and maps its status to the
188 * proxy's own log message (DNS failure vs no-usable-socket, the latter
189 * also bumping connect-error metrics). Stores the fd in tgt->fd and
190 * returns NGX_OK, or NGX_ERROR on failure (nothing left open).
191 */
192static ngx_int_t
193pc_resolve_target(brix_proxy_ctx_t *proxy, ngx_connection_t *client_conn,
194 pc_target_t *tgt)
195{
196 brix_resolve_status_t rstatus = BRIX_RESOLVE_OK;
197
198 tgt->fd = brix_resolve_connect_socket((const char *) tgt->host->data,
199 (unsigned) tgt->port,
200 BRIX_AF_AUTO,
201 &tgt->addr, &tgt->addrlen,
202 &rstatus);
203 if (tgt->fd == (int) NGX_INVALID_FILE) {
204 if (rstatus == BRIX_RESOLVE_ERR_DNS) {
205 ngx_log_error(NGX_LOG_ERR, client_conn->log, 0,
206 "xrootd proxy: cannot resolve \"%s\"",
207 tgt->host->data);
208 return NGX_ERROR;
209 }
210 ngx_log_error(NGX_LOG_ERR, client_conn->log, 0,
211 "xrootd proxy: no usable address for \"%s\"",
212 tgt->host->data);
213 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
214 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
215 return NGX_ERROR;
216 }
217
218 return NGX_OK;
219}

221/*
222 * WHAT: Resolves the upstream login username from the configured policy into
223 * user_out (a >= 9-byte buffer).
224 *
225 * WHY: The kXR_login bootstrap request carries a username; policy decides
226 * whether it passes through the client's login name, uses a fixed
227 * configured name, or stays anonymous (empty → "xrd" default downstream).
228 *
229 * HOW: Switches on conf->proxy.login_user; ngx_cpystrn truncates to the buffer
230 * size. Passthrough with no client username falls back to empty.
231 */
232static void
233pc_resolve_login_user(brix_proxy_ctx_t *proxy,
234 ngx_stream_brix_srv_conf_t *conf, char *user_out)
235{
236 switch (conf->proxy.login_user) {
237 case BRIX_PROXY_LOGIN_PASSTHROUGH:
238 if (proxy->client_ctx != NULL
239 && proxy->client_ctx->login.user[0] != '\0')
240 {
241 ngx_cpystrn((u_char *) user_out,
242 (u_char *) proxy->client_ctx->login.user, 9);
243 } else {
244 user_out[0] = '\0'; /* no client username → fall back to "xrd" */
245 }
246 break;
247 case BRIX_PROXY_LOGIN_FIXED:
248 ngx_cpystrn((u_char *) user_out,
249 (u_char *) conf->proxy.login_user_name, 9);
250 break;
251 default: /* BRIX_PROXY_LOGIN_ANONYMOUS */
252 user_out[0] = '\0';
253 break;
254 }
255}

257/*
258 * WHAT: Wraps a connected-ready fd in an nginx connection with its own pool,
259 * builds the bootstrap buffer, and wires up the proxy send/recv handlers.
260 *
261 * WHY: Every upstream connection needs a dedicated pool (for the bootstrap
262 * frames), the standard ngx send/recv vtable, and the proxy read/write
263 * handlers before connect() is armed.
264 *
265 * HOW: ngx_get_connection + ngx_create_pool + ngx_palloc for the 68-byte
266 * bootstrap; fills it via pc_resolve_login_user + brix_proxy_build_bootstrap;
267 * points proxy->wbuf at the pool-owned buffer (wbuf_owned=0). Returns the
268 * connection, or NULL after closing the fd / freeing partial state.
269 */
270static ngx_connection_t *
271pc_open_socket(brix_proxy_ctx_t *proxy, ngx_connection_t *client_conn,
272 ngx_stream_brix_srv_conf_t *conf, int fd)
273{
274 ngx_connection_t *uconn;
275 u_char *bsbuf;
276 char upstream_user[9] = { 0 };
277
278 uconn = ngx_get_connection(fd, client_conn->log);
279 if (uconn == NULL) {
280 ngx_close_socket(fd);
281 return NULL;
282 }
283
284 uconn->pool = ngx_create_pool(512, client_conn->log);
285 if (uconn->pool == NULL) {
286 ngx_free_connection(uconn);
287 ngx_close_socket(fd);
288 return NULL;
289 }
290
291 /* Build bootstrap buffer in the upstream connection pool */
292 bsbuf = ngx_palloc(uconn->pool,
293 XRD_HANDSHAKE_LEN
294 + sizeof(ClientProtocolRequest)
295 + sizeof(ClientLoginRequest));
296 if (bsbuf == NULL) {
297 ngx_destroy_pool(uconn->pool);
298 ngx_free_connection(uconn);
299 ngx_close_socket(fd);
300 return NULL;
301 }
302
303 pc_resolve_login_user(proxy, conf, upstream_user);
304
305 uconn->data = proxy;
306 uconn->recv = ngx_recv;
307 uconn->send = ngx_send;
308 uconn->recv_chain = ngx_recv_chain;
309 uconn->send_chain = ngx_send_chain;
310 uconn->log = client_conn->log;
311 uconn->read->handler = brix_proxy_read_handler;
312 uconn->write->handler = brix_proxy_write_handler;
313 uconn->read->log = client_conn->log;
314 uconn->write->log = client_conn->log;
315
316 proxy->conn = uconn;
317 proxy->wbuf = bsbuf;
318 proxy->wbuf_len = brix_proxy_build_bootstrap(bsbuf, upstream_user);
319 proxy->wbuf_pos = 0;
320 proxy->wbuf_owned = 0; /* Phase 39: bsbuf is pool-allocated — pool owns it */
321 proxy->state = XRD_PX_CONNECTING;
322
323 return uconn;
324}

327/*
328 * WHAT: Starts the upstream TLS handshake on an already-connected socket.
329 *
330 * WHY: On an immediate (rc==0) connect to a TLS upstream we begin the client
331 * TLS handshake right away; the async callback finishes the bootstrap.
332 *
333 * HOW: ngx_ssl_create_connection + SNI (explicit name directive, else host) +
334 * ngx_ssl_handshake. NGX_AGAIN leaves the callback to continue; a synchronous
335 * completion invokes brix_proxy_tls_handshake_done() inline. Returns NGX_OK
336 * on success (connect continues via callback/inline), NGX_ERROR after
337 * cleanup on failure.
338 */
339static ngx_int_t
340pc_start_tls(brix_proxy_ctx_t *proxy, ngx_stream_brix_srv_conf_t *conf,
341 ngx_connection_t *uconn)
342{
343 const char *sni;
344
345 if (ngx_ssl_create_connection(conf->proxy.tls_ctx, uconn,
346 NGX_SSL_BUFFER | NGX_SSL_CLIENT)
347 != NGX_OK)
348 {
349 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
350 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
351 brix_proxy_cleanup(proxy);
352 return NGX_ERROR;
353 }
354
355 /* SNI: prefer explicit name directive, fall back to configured host */

364 return NGX_OK; /* TLS callback will continue */

371/*
372 * WHAT: Fires the async connect() on fd, arms the write event / connect timer,
373 * and — on an immediate connect — either starts TLS or flushes bootstrap.
374 *
375 * WHY: nginx drives outbound connects under the event loop: connect() returns
376 * EINPROGRESS and the write_handler completes it, unless the peer is local
377 * (rc==0) in which case we proceed to TLS or bootstrap flush immediately.
378 *
379 * HOW: connect() (EINPROGRESS is expected/OK), ngx_handle_write_event, optional
380 * connect timer; on rc==0 dispatches to pc_start_tls (TLS upstreams) or sets
381 * BOOTSTRAP state + brix_proxy_flush. Any failure cleans up. Returns NGX_OK
382 * or NGX_ERROR.
383 */
384static ngx_int_t
385pc_arm_events(brix_proxy_ctx_t *proxy, ngx_connection_t *client_conn,
386 ngx_stream_brix_srv_conf_t *conf, ngx_connection_t *uconn,
387 pc_target_t *tgt)
388{
389 ngx_int_t rc;
390
391 ngx_log_debug(NGX_LOG_DEBUG_STREAM, client_conn->log, 0,
392 "xrootd proxy: connect() to %s:%d",
393 tgt->host->data, (int) tgt->port);
394
395 rc = connect(tgt->fd, (struct sockaddr *)(void *) &tgt->addr,
396 tgt->addrlen);
397 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
398 ngx_log_error(NGX_LOG_ERR, client_conn->log, ngx_socket_errno,
399 "xrootd proxy: connect to %s:%d failed",
400 tgt->host->data, (int) tgt->port);
401 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
402 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
403 brix_proxy_cleanup(proxy);
404 return NGX_ERROR;
405 }
406
407 if (ngx_handle_write_event(uconn->write, 0) != NGX_OK) {
408 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
409 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
410 brix_proxy_cleanup(proxy);
411 return NGX_ERROR;
412 }
413
414 /* Arm connect timeout: fires in write_handler as wev->timedout */
415 if (rc == -1 && conf->proxy.connect_timeout > 0) {
416 ngx_add_timer(uconn->write, conf->proxy.connect_timeout);
417 }
418
419 if (rc == 0) {
420 /* Immediate connect (local loopback etc.) */
421#if (NGX_SSL)
422 if (conf->proxy.upstream_tls && conf->proxy.tls_ctx != NULL) {
423 return pc_start_tls(proxy, conf, uconn);
424 }
425#endif
426 proxy->state = XRD_PX_BOOTSTRAP;
427 proxy->bs_phase = XRD_PX_BS_HANDSHAKE;
428 proxy->rhdr_pos = 0;
429
430 if (brix_proxy_flush(proxy) == NGX_ERROR) {
431 BRIX_PROXY_METRIC_INC(proxy->client_ctx, upstream_connect_errors);
432 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
433 brix_proxy_cleanup(proxy);
434 return NGX_ERROR;
435 }
436 }
437
438 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, client_conn->log, 0,
439 "xrootd proxy: connecting to %s:%d",
440 tgt->host->data, (int) tgt->port);
441 return NGX_OK;
442}

445ngx_int_t
446brix_proxy_connect(brix_proxy_ctx_t *proxy,
447 ngx_connection_t *client_conn,
448 ngx_stream_brix_srv_conf_t *conf)
449{
450 pc_target_t tgt;
451 ngx_connection_t *uconn;
452 ngx_int_t rc;
453
454 ngx_memzero(&tgt, sizeof(tgt));
455
456 /* Select upstream: redirected host > pool > round-robin array > single host.
457 * NGX_OK = a pooled connection was adopted (connect complete). */
458 rc = pc_select_endpoint(proxy, client_conn, conf, &tgt);
459 if (rc != NGX_DECLINED) {
460 return rc;
461 }
462
463 /* GSI delegation: log in to the upstream AS THE USER in a thread (the blocking
464 * in-process GSI client), then hand the authenticated fd to the relay. */
465 if (conf->proxy.auth == BRIX_PROXY_AUTH_GSI) {
466 return brix_proxy_gsi_connect_async(proxy, conf, tgt.host,
467 (uint16_t) tgt.port);
468 }
469
470 if (pc_resolve_target(proxy, client_conn, &tgt) != NGX_OK) {
471 return NGX_ERROR;
472 }
473
474 uconn = pc_open_socket(proxy, client_conn, conf, tgt.fd);
475 if (uconn == NULL) {
476 return NGX_ERROR;
477 }
478
479 return pc_arm_events(proxy, client_conn, conf, uconn, &tgt);
480}
