1/*
2 * sd_xroot_staged.c — staged atomic-publish (Mode A passthrough) for the
3 * root:// origin driver.
4 *
5 * WHAT: The staged-write vtable slots — staged_open/staged_open_cred (connect +
6 * bootstrap + kXR_open(write) a fresh writable origin handle),
7 * staged_write (kXR_write at offset), staged_commit (kXR_sync + close,
8 * consuming the handle), and staged_abort (close, leaving whatever bytes
9 * already streamed).
10 *
11 * WHY: Split out of sd_xroot.c (phase-79 file-size split): the staged handle is
12 * a distinct object (its own state struct + connection) from the read/
13 * write object (sd_xroot_io.c), the shared origin machinery + driver
14 * vtable (sd_xroot.c), and the namespace ops (sd_xroot_ns.c).
15 *
16 * HOW: The staged handle is a live origin write handle. With NO local staging
17 * store the write streams straight to the FINAL remote path (transparent
18 * write-through, no local copy), so it is NOT atomic — a failed upload can
19 * leave a partial object on the origin. Configure a local staging directory
20 * (Mode B) for atomic / durable publish (a later phase). The opaque
21 * brix_sd_staged_t (sd.h) carries our state in ->state.
22 */
23

38static void
39sd_xroot_staged_teardown(brix_sd_staged_t *handle)
40{
41 sd_xroot_staged_state *ss;
42
43 if (handle == NULL) {
44 return;
45 }
46 ss = handle->state;
47 if (ss != NULL) {
48 if (ss->file_open) {
49 brix_cache_origin_close_file(&ss->oc, ss->fhandle);
50 }
51 brix_cache_origin_close(&ss->oc);
52 free(ss->t);
53 free(ss);
54 }
55 free(handle);

58/* sd_xroot_staged_connect_open — connect + bootstrap + write-open a staged handle.
59 *
60 * WHAT: Run connect → bootstrap → kXR_open(write) on the staged handle's origin
61 * session for `final_path`. Returns 0 on success (ss->file_open set), or
62 * -1 with *err_out set on any step failure (the origin connection is
63 * closed on failure; the caller frees the shells).
64 * WHY: Isolating the three-step origin bring-up keeps sd_xroot_staged_open_common
65 * a flat allocate-then-open orchestrator and shares the failure logging in
66 * one place.
67 * HOW: 1) connect, 2) bootstrap (using whatever credential the fill task already
68 * carries), 3) open_write with the create mode; on any failure log the
69 * actual origin reason, map errno, and close the connection. */
70static int
71sd_xroot_staged_connect_open(sd_xroot_staged_state *ss, const char *final_path,
72 mode_t mode, int *err_out)
73{
74 brix_cache_fill_t *t = ss->t;
75
76 if (brix_cache_origin_connect(t, &ss->oc) != 0
77 || brix_cache_origin_bootstrap(t, &ss->oc) != 0
78 || brix_cache_origin_open_write(t, &ss->oc, final_path,
79 (uint16_t) ((mode != 0) ? (mode & 0777) : 0644), ss->fhandle) != 0)
80 {
81 /* Surface the ACTUAL origin failure reason (auth/TLS/protocol) — the
82 * caller only sees the mapped errno, which hides why the origin session
83 * failed (e.g. "requires TLS", "no credential set", gsi handshake). */
84 ngx_log_error(NGX_LOG_ERR, ngx_cycle->log, 0,
85 "brix sd_xroot: origin staged-open \"%s\" failed: %s (kXR %d)",
86 final_path,
87 (t->err_msg[0] != '\0') ? t->err_msg : "(no detail)",
88 t->xrd_error);
89 if (err_out) { *err_out = sd_xroot_errno(t); }
90 brix_cache_origin_close(&ss->oc);
91 return -1;
92 }
93 ss->file_open = 1;
94 return 0;
95}

97/* sd_xroot_staged_open_common — shared body for plain and cred-scoped staged opens.
98 *
99 * WHAT: Connect, bootstrap (with optional per-user credential), kXR_open write,
100 * and wire up the staged handle.
101 * WHY: Plain and cred paths share identical connect/bootstrap/open logic; the
102 * only difference is whether the fill task carries a per-user proxy.
103 * HOW: 1) allocate handle + state + fill task (zeroed cred fields ⇒ cred=NULL
104 * behaves like Phase-0). 2) copy the credential via the shared helper. 3)
105 * delegate connect/bootstrap/open-write to sd_xroot_staged_connect_open;
106 * free all three shells on failure. */
107static brix_sd_staged_t *
108sd_xroot_staged_open_common(brix_sd_instance_t *inst, const char *final_path,
109 mode_t mode, const brix_sd_cred_t *cred, int *err_out)
110{
111 sd_xroot_inst_state *is = inst->state;
112 brix_sd_staged_t *handle = calloc(1, sizeof(*handle));
113 sd_xroot_staged_state *ss = calloc(1, sizeof(*ss));
114 brix_cache_fill_t *t = calloc(1, sizeof(*t));
115
116 if (handle == NULL || ss == NULL || t == NULL) {
117 free(handle);
118 free(ss);
119 free(t);
120 if (err_out) { *err_out = ENOMEM; }
121 return NULL;
122 }
123 ss->t = t;
124 ss->oc.fd = -1;
125 t->conf = is->conf;
126 ngx_cpystrn((u_char *) t->clean_path, (u_char *) final_path,
127 sizeof(t->clean_path));
128
129 sd_xroot_copy_cred_into_task(t, cred);
130
131 if (sd_xroot_staged_connect_open(ss, final_path, mode, err_out) != 0) {
132 free(t);
133 free(ss);
134 free(handle);
135 return NULL;
136 }
137 handle->inst = inst;
138 handle->state = ss;
139 return handle;
140}

142/* sd_xroot_staged_open — vtable staged_open slot: service credential / anonymous.
143 *
144 * WHAT: Plain staged open for callers that do not carry a per-user credential.
145 * WHY: Preserves the existing public vtable signature; cred=NULL means the
146 * bootstrap uses the static service credential (or anonymous).
147 * HOW: Delegates to sd_xroot_staged_open_common with cred=NULL. */

169sd_xroot_staged_write(brix_sd_staged_t *handle, const void *buf, size_t len,
170 off_t off)
171{
172 sd_xroot_staged_state *ss = handle->state;

176 }

178 (uint64_t) off, (const u_char *) buf, len) != 0)
179 {
180 errno = EIO;
181 return -1;
182 }
183 return (ssize_t) len;

186/* Publish: sync + close the origin file. On success the handle is consumed (freed);
187 * on failure it stays valid for the caller to staged_abort. `noreplace` cannot be
188 * enforced on a Mode-A direct write (open_write already created the destination) —
189 * use a staging dir (Mode B) for exclusive/atomic publish. */
190ngx_int_t
191sd_xroot_staged_commit(brix_sd_staged_t *handle, int noreplace)
192{
193 sd_xroot_staged_state *ss = handle->state;
194
195 (void) noreplace;
196
197 if (brix_cache_origin_sync(ss->t, &ss->oc, ss->fhandle) != 0) {
198 return NGX_ERROR; /* leave valid; caller aborts */
199 }
200 sd_xroot_staged_teardown(handle); /* close_file + free (consumed) */
201 return NGX_OK;
202}
