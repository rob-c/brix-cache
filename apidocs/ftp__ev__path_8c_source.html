12/*
13 * ftp_ev_path.c — command tokenising, path confinement, and VFS context setup.
14 *
15 * WHAT: split a command line into verb + argument, resolve a (possibly relative)
16 * command argument into a canonical filesystem path confined under the export
17 * root, and build the per-operation brix_vfs_ctx_t.
18 *
19 * WHY: every namespace/transfer verb needs the same two things — a path that
20 * cannot escape the export (INVARIANT 4: resolve_path before open) and a VFS
21 * context carrying the caller's identity + write permission. Centralising them
22 * keeps the per-verb handlers to their own logic. Ported verbatim from the sync
23 * engine so both share identical confinement semantics during the transition.
24 *
25 * HOW: brix_ftp_ev_resolve() joins the session CWD with the argument, then hands
26 * the logical path to brix_http_resolve_path() for canonicalisation + confinement
27 * against conf->root_canon; brix_ftp_ev_vfs_ctx() threads the GSI identity and
28 * TLS state into brix_vfs_ctx_init().
29 */

49/* Combine the session CWD with `arg` into a logical path and confine it under the
50 * export root. On success `abs` holds the canonical filesystem path and the
51 * return is 0; otherwise an FTP failure code (550 not-found/denied, 553 malformed). */
52int
53brix_ftp_ev_resolve(ftp_ev_t *fc, const char *arg, char *abs, size_t abssz)
54{
55 char logical[PATH_MAX];
56 int rc;
57
58 if (arg == NULL || arg[0] == '\0') {
59 ngx_memcpy(logical, fc->cwd, ngx_strlen(fc->cwd) + 1);
60 } else if (arg[0] == '/') {
61 if (ngx_strlen(arg) >= sizeof(logical)) {
62 return 553;
63 }
64 ngx_memcpy(logical, arg, ngx_strlen(arg) + 1);
65 } else {
66 int n = snprintf(logical, sizeof(logical), "%s%s%s",
67 fc->cwd,
68 (fc->cwd[1] == '\0') ? "" : "/", /* cwd == "/" ? */
69 arg);
70 if (n <= 0 || (size_t) n >= sizeof(logical)) {
71 return 553;
72 }
73 }
74
75 rc = brix_http_resolve_path(fc->conf->root_canon, logical, abs, abssz);
76 if (rc != 0) {
77 return 550; /* 400/403/404/414/500 → FTP 550 */
78 }
79 return 0;
80}

83/* Read every PEM certificate in `pem` onto `sink`, de-duplicating by identity so
84 * a cert already present (e.g. the spliced issuer) is not added twice. */
85static void
86fwd_collect_certs(ngx_str_t *pem, STACK_OF(X509) *sink)
87{
88 BIO *bio;
89 X509 *x;
90
91 if (pem == NULL || pem->len == 0) {
92 return;
93 }
94 bio = BIO_new_mem_buf(pem->data, (int) pem->len);
95 if (bio == NULL) {
96 return;
97 }
98 while ((x = PEM_read_bio_X509(bio, NULL, NULL, NULL)) != NULL) {
99 int i, dup = 0;
100 for (i = 0; i < sk_X509_num(sink); i++) {
101 if (X509_cmp(sk_X509_value(sink, i), x) == 0) {
102 dup = 1;
103 break;
104 }
105 }
106 if (dup) {
107 X509_free(x);
108 } else if (sk_X509_push(sink, x) == 0) {
109 X509_free(x);
110 }
111 }
112 BIO_free(bio);
113}

116/* Assemble the RFC 3820 proxy chain to forward to the storage backend.
117 *
118 * The credential captured on the control channel (fc->deleg_proxy) is what the
119 * server-side GSSAPI layer could see of the client's chain — leaf + whatever
120 * SSL_get_peer_cert_chain() exposed — which (a) omits the delegated leaf's DIRECT
121 * issuer (the peer's own control leaf, excluded by OpenSSL server-side), (b) may
122 * carry the certs OUT OF ISSUER ORDER, and (c) includes the self-signed CA. A
123 * GSI upstream (XrdSecgsi) walks the presented chain strictly and rejects any of
124 * those as "chain is inconsistent".
125 *
126 * Rebuild a clean chain: parse every cert (from the capture and the separately
127 * captured control leaf, fc->ctrl_leaf_pem), find the leaf by matching the
128 * private key, then walk issuer→subject links emitting leaf → … → end-entity,
129 * STOPPING before the self-signed trust anchor (the upstream verifies the CA from
130 * its own trust store; a forwarded CA breaks the walk). Serialize the ordered
131 * certs followed by the private key — the layout every GSI consumer expects.
132 *
133 * Returns *deleg verbatim on any parse/alloc failure (defensive: never worse than
134 * forwarding the raw capture). All OpenSSL temporaries are freed before return;
135 * the result bytes are owned by `pool`. */
136static ngx_str_t
137brix_ftp_ev_forward_pem(ngx_pool_t *pool, ngx_str_t *deleg, ngx_str_t *issuer)
138{
139 ngx_str_t out = *deleg; /* default: forward verbatim */
140 STACK_OF(X509) *certs;
141 EVP_PKEY *key = NULL;
142 X509 *leaf = NULL, *cur;
143 BIO *kbio, *obio = NULL;
144 int i;
145
146 if (deleg->len == 0) {
147 return out;
148 }
149
150 certs = sk_X509_new_null();
151 if (certs == NULL) {
152 return out;
153 }
154 fwd_collect_certs(deleg, certs);
155 fwd_collect_certs(issuer, certs);
156
157 /* the private key identifies the leaf (the cert we can actually sign with). */
158 kbio = BIO_new_mem_buf(deleg->data, (int) deleg->len);
159 if (kbio != NULL) {
160 key = PEM_read_bio_PrivateKey(kbio, NULL, NULL, NULL);
161 BIO_free(kbio);
162 }
163 if (key != NULL) {
164 for (i = 0; i < sk_X509_num(certs); i++) {
165 X509 *c = sk_X509_value(certs, i);
166 EVP_PKEY *pk = X509_get_pubkey(c);
167 int hit = (pk != NULL && EVP_PKEY_eq(pk, key) == 1);
168 EVP_PKEY_free(pk);
169 if (hit) {
170 leaf = c;
171 break;
172 }
173 }
174 }
175
176 obio = (leaf != NULL) ? BIO_new(BIO_s_mem()) : NULL;
177 if (obio != NULL) {
178 int ok = 1;
179 int steps = 0;
180 int ncerts = sk_X509_num(certs);
181
182 /* Emit leaf → issuer → … stopping before the self-signed trust anchor.
183 * Bound the walk by the cert count: a hostile client could delegate a
184 * cross-signed cert pair (A issues B, B issues A) with no self-signed
185 * terminus, which would otherwise spin the worker. */
186 cur = leaf;
187 while (cur != NULL && steps++ <= ncerts) {
188 X509 *next = NULL;
189
190 if (!PEM_write_bio_X509(obio, cur)) {
191 ok = 0;
192 break;
193 }
194 if (X509_NAME_cmp(X509_get_subject_name(cur),
195 X509_get_issuer_name(cur)) == 0)
196 {
197 break; /* reached a self-issued cert — stop */
198 }
199 for (i = 0; i < sk_X509_num(certs); i++) {
200 X509 *c = sk_X509_value(certs, i);
201 if (c == cur) {
202 continue;
203 }
204 if (X509_NAME_cmp(X509_get_subject_name(c),
205 X509_get_issuer_name(cur)) == 0)
206 {
207 next = c;
208 break;
209 }
210 }
211 /* drop the trust anchor: if the only remaining issuer is self-signed
212 * (the CA), do not forward it — the upstream trusts it out of band. */
213 if (next != NULL
214 && X509_NAME_cmp(X509_get_subject_name(next),
215 X509_get_issuer_name(next)) == 0)
216 {
217 next = NULL;
218 }
219 cur = next;
220 }
221
222 if (ok && key != NULL
223 && PEM_write_bio_PrivateKey(obio, key, NULL, NULL, 0, NULL, NULL))
224 {
225 u_char *data;
226 long len = BIO_get_mem_data(obio, &data);
227 if (len > 0) {
228 u_char *copy = ngx_pnalloc(pool, (size_t) len);
229 if (copy != NULL) {
230 ngx_memcpy(copy, data, (size_t) len);
231 out.data = copy;
232 out.len = (size_t) len;
233 }
234 }
235 }
236 BIO_free(obio);
237 }
238
239 if (key != NULL) {
240 EVP_PKEY_free(key);
241 }
242 sk_X509_pop_free(certs, X509_free);
243 return out;
244}

247/* Build the per-operation VFS context: export root, write permission, TLS flag,
248 * and the verified GSI principal (NULL for a cleartext session).
249 *
250 * When the client delegated an X.509 proxy on the control channel, forward it to
251 * the storage backend so the upstream authenticates AS the gsiftp user (the
252 * legacy gsiftp → xrootd gateway). A full proxy is a PASSTHROUGH credential —
253 * presented to the upstream unmodified, exactly as the root:// and WebDAV
254 * backends treat a captured proxy. Gated three ways so it never disturbs the
255 * existing paths: only when a proxy was actually delegated, only when the
256 * resolved mode forwards it (not SELECT), and only when the leaf backend consumes
257 * a proxy PEM (xroot/s3) — a posix/pblock export accepts no forwarded proxy, so
258 * the bind is skipped and the request serves with the session's own identity as
259 * before. The bag is bound before any open (INVARIANT 4 ordering unchanged). */
260void
261brix_ftp_ev_vfs_ctx(ftp_ev_t *fc, const char *abs, void *vctx)
262{
263 brix_vfs_ctx_t *ctx = vctx;
264
265 brix_vfs_ctx_init(ctx, fc->c->pool, fc->c->log,
266 BRIX_PROTO_ROOT, fc->conf->root_canon, "",
267 fc->conf->allow_write ? 1 : 0,
268 fc->sec_active ? 1 : 0 /* is_tls */,
269 fc->identity, abs);
270
271 if (fc->deleg_proxy.len > 0
272 && fc->conf->deleg_mode != BRIX_CRED_SELECT
273 && brix_vfs_backend_accepts_proxy(ctx))
274 {
275 ngx_str_t fwd = brix_ftp_ev_forward_pem(fc->c->pool, &fc->deleg_proxy,
276 &fc->ctrl_leaf_pem);
277 (void) brix_vfs_deleg_bind(fc->c->pool, ctx, fc->conf->deleg_mode,
278 NULL, &fwd);
279 }
280}
