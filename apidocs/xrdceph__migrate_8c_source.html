1/*
2 * xrdceph_migrate.c — migrate a flat RADOS pool into a real filesystem tree
3 * (copy-through-mount). Operator utility.
4 *
5 * WHAT: Reads every object from a flat `ceph`-backend pool (object key == logical
6 * path) and writes it as a file at the corresponding path beneath a
7 * destination directory, recreating parent directories and carrying user
8 * xattrs (including `user.XrdCks.*` checksums-at-rest). When the destination
9 * is a mounted CephFS, the MDS allocates inodes and builds the namespace —
10 * i.e. this performs the spike's copy-through-mount migration (the only sound
11 * way: CephFS keys data by MDS-allocated inode, so an in-place "upgrade" is
12 * impossible; the bytes must be rewritten through the fs).
13 *
14 * USAGE:
15 * xrdceph_migrate <pool> <dest_dir> (dest_dir is a CephFS mount point)
16 * (env CEPH_CONF overrides /etc/ceph/ceph.conf)
17 *
18 * The destination is treated as a plain directory, so this is verifiable against a
19 * local folder; pointing it at a CephFS mount is what makes it a migration.
20 *
21 * BUILD: `make -C client ceph-tools` (dep-gated), or by hand where
22 * librados-devel exists:
23 * gcc -DXRDPROTO_NO_NGX -DBRIX_HAVE_CEPH -I src/fs/backend -I src/fs/backend/rados \
24 * -include client/apps/ceph/ngx_shim.h client/apps/ceph/xrdceph_migrate.c \
25 * src/fs/backend/rados/sd_ceph.c src/fs/backend/rados/sd_ceph_compat.c \
26 * -lrados -o xrdceph_migrate
27 */

45/* mkdir -p for every parent directory of `path` (path itself is a file). */
46static int
47mkdirs_for(const char *path)
48{
49 char tmp[4096];
50 char *s;
51
52 snprintf(tmp, sizeof(tmp), "%s", path);
53 for (s = tmp + 1; *s != '\0'; s++) {
54 if (*s == '/') {
55 *s = '\0';
56 if (mkdir(tmp, 0755) != 0 && errno != EEXIST) { return -1; }
57 *s = '/';
58 }
59 }
60 return 0;
61}

63/* Carry every user xattr from the RADOS object `oid` onto the open dest fd. Best
64 * effort: a per-xattr failure is logged but does not fail the file. */
65static void
66carry_xattrs(sd_ceph_conn_t *c, const char *oid, int fd)
67{
68 char names[8192];
69 ssize_t n = sd_ceph_oid_listxattr(c, oid, names, sizeof(names));
70 ssize_t p;
71
72 if (n <= 0) { return; }
73 for (p = 0; p < n; p += (ssize_t) strlen(names + p) + 1) {
74 const char *name = names + p;
75 char val[65536];
76 ssize_t vlen;
77
78 if (name[0] == '\0') { break; }
79 vlen = sd_ceph_oid_getxattr(c, oid, name, val, sizeof(val));
80 if (vlen < 0) { continue; }
81 if (fsetxattr(fd, name, val, (size_t) vlen, 0) != 0) {
82 fprintf(stderr, " warn: xattr %s on %s: %s\n", name, oid, strerror(errno));
83 }
84 }
85}

87/* Stream object `oid` to the already-open dest fd in 1 MiB reads. 0 / -1. */
88static int
89stream_object(sd_ceph_conn_t *c, const char *oid, int fd)
90{
91 char buf[1u << 20];
92 off_t off = 0;
93 ssize_t n;
94
95 while ((n = sd_ceph_oid_read(c, oid, buf, sizeof(buf), off)) > 0) {
96 ssize_t w = 0;
97 while (w < n) {
98 ssize_t k = write(fd, buf + w, (size_t) (n - w));
99 if (k < 0) { if (errno == EINTR) { continue; } return -1; }
100 w += k;
101 }
102 off += n;
103 if ((size_t) n < sizeof(buf)) { break; }
104 }
105 return (n < 0) ? -1 : 0;
106}

108/* ---- Connect to the RADOS pool and prepare the destination + object list ----
109 *
110 * WHAT: Connects to `pool` (honouring CEPH_CONF), ensures `dest` exists as a
111 * directory, and opens a whole-pool object-list cursor. On success returns
112 * 0 with `*conn_out` set to a live connection and `*lc_out` to an open list
113 * cursor; on any failure prints a diagnostic, tears down anything already
114 * acquired, and returns -1 with `*conn_out` left NULL.
115 *
116 * WHY: Keeps main's orchestration flat by confining all one-time acquisition and
117 * its error/cleanup ladder to a single owner, so the migration loop can
118 * assume a ready source and destination.
119 *
120 * HOW:
121 * 1. Zero a conf, point it at `pool` and the CEPH_CONF (or default) conf file.
122 * 2. Create the ceph connection; on failure print and return -1.
123 * 3. mkdir(dest) tolerating EEXIST; on real failure destroy the conn, return -1.
124 * 4. Open the object-list cursor over the connection's ioctx; on failure
125 * destroy the conn, return -1.
126 * 5. Publish the connection and cursor to the out-params and return 0.
127 */
128static int
129open_source_and_dest(const char *pool, const char *dest,
130 sd_ceph_conn_t **conn_out, rados_list_ctx_t *lc_out)
131{
132 brix_sd_ceph_conf_t conf;
133 sd_ceph_conn_t *c;
134 int err = 0;
135
136 *conn_out = NULL;
137
138 memset(&conf, 0, sizeof(conf));
139 conf.pool = pool;
140 conf.conf_file = getenv("CEPH_CONF") ? getenv("CEPH_CONF") : "/etc/ceph/ceph.conf";
141
142 c = sd_ceph_conn_create(&conf, NULL, &err);
143 if (c == NULL) { fprintf(stderr, "connect %s: %s\n", pool, strerror(err)); return -1; }
144
145 if (mkdir(dest, 0755) != 0 && errno != EEXIST) {
146 fprintf(stderr, "mkdir %s: %s\n", dest, strerror(errno));
147 sd_ceph_conn_destroy(c);
148 return -1;
149 }
150 if (rados_nobjects_list_open(sd_ceph_conn_ioctx(c), lc_out) < 0) {
151 fprintf(stderr, "list open: %s\n", strerror(errno));
152 sd_ceph_conn_destroy(c);
153 return -1;
154 }
155
156 *conn_out = c;
157 return 0;
158}

160/* ---- Copy one RADOS object to its file under the destination tree ----
161 *
162 * WHAT: Materialises object `entry` as a file at its logical path beneath `dest`,
163 * recreating parent directories, carrying user xattrs, and adding the
164 * object's byte size to `*bytes`. Returns 0 on success (and prints the
165 * `src -> dst` progress line); returns -1 on any failure after printing a
166 * diagnostic. Best-effort xattr carry never fails the file.
167 *
168 * WHY: Isolates the whole per-object copy pipeline behind one call so the driver
169 * loop only accounts success/failure — one nameable responsibility per unit
170 * and no cleanup ladder leaking into main.
171 *
172 * HOW:
173 * 1. Build the destination path, dropping a leading '/' so the key is relative.
174 * 2. mkdir -p its parents; on failure print and return -1.
175 * 3. Open the destination file for writing; on failure print and return -1.
176 * 4. Stream the object bytes to the fd; on failure print, close, return -1.
177 * 5. Carry user xattrs (best effort), then close the file.
178 * 6. Add the object size to `*bytes` — from the object stat, else the written
179 * file's stat — print the progress line, and return 0.
180 */
181static int
182migrate_one_object(sd_ceph_conn_t *c, const char *dest, const char *entry,
183 unsigned long long *bytes)
184{
185 char path[4096];
186 struct stat sb;
187 FILE *out;
188 int fd;
189 uint64_t sz = 0;
190
191 /* object key is the logical path; drop a leading '/' to make it relative */
192 snprintf(path, sizeof(path), "%s/%s", dest,
193 (entry[0] == '/') ? entry + 1 : entry);
194
195 if (mkdirs_for(path) != 0) {
196 fprintf(stderr, "mkdirs for %s: %s\n", path, strerror(errno));
197 return -1;
198 }
199 out = fopen(path, "wb");
200 if (out == NULL) {
201 fprintf(stderr, "create %s: %s\n", path, strerror(errno));
202 return -1;
203 }
204 fd = fileno(out);
205
206 if (stream_object(c, entry, fd) != 0) {
207 fprintf(stderr, "copy %s failed\n", entry);
208 fclose(out);
209 return -1;
210 }
211 carry_xattrs(c, entry, fd);
212 fclose(out);
213
214 if (sd_ceph_oid_stat(c, entry, &sz, NULL) == 0) { *bytes += sz; }
215 else if (stat(path, &sb) == 0) { *bytes += (uint64_t) sb.st_size; }
216
217 printf(" %s -> %s\n", entry, path);
218 return 0;
219}

221/* ---- Drive the whole-pool copy loop, tallying files/bytes/failures ----
222 *
223 * WHAT: Iterates every object in the open list cursor `lc`, copying each into the
224 * `dest` tree via migrate_one_object() and accumulating the counts into
225 * `*files`, `*bytes`, and `*fails`. Returns when the cursor is exhausted or
226 * a fatal list error occurs (the latter counted as one failure).
227 *
228 * WHY: Separates the iteration/accounting policy from the per-object copy so main
229 * stays a short, linear sequence of named steps.
230 *
231 * HOW:
232 * 1. Advance the cursor; stop cleanly on -ENOENT (exhausted).
233 * 2. On a negative list error, print it, count one failure, and stop.
234 * 3. Skip empty/NULL keys.
235 * 4. Copy the object; bump `*files` on success or `*fails` on failure.
236 */
237static void
238migrate_pool(sd_ceph_conn_t *c, const char *dest, rados_list_ctx_t lc,
239 unsigned long long *files, unsigned long long *bytes,
240 unsigned long long *fails)
241{
242 for (;;) {
243 const char *entry = NULL;
244 int rc;
245
246 rc = rados_nobjects_list_next2(lc, &entry, NULL, NULL, NULL, NULL, NULL);
247 if (rc == -ENOENT) { break; }
248 if (rc < 0) { fprintf(stderr, "list next: %d\n", rc); (*fails)++; break; }
249 if (entry == NULL || entry[0] == '\0') { continue; }
250
251 if (migrate_one_object(c, dest, entry, bytes) == 0) { (*files)++; }
252 else { (*fails)++; }
253 }
254}

256int
257main(int argc, char **argv)
258{
259 sd_ceph_conn_t *c;
260 rados_list_ctx_t lc;
261 const char *dest;
262 unsigned long long files = 0, bytes = 0, fails = 0;
263
264 if (argc < 3) {
265 fprintf(stderr, "usage: %s <pool> <dest_dir>\n", argv[0]);
266 return 2;
267 }
268 dest = argv[2];
269
270 if (open_source_and_dest(argv[1], dest, &c, &lc) != 0) { return 1; }
271
272 migrate_pool(c, dest, lc, &files, &bytes, &fails);
273
274 rados_nobjects_list_close(lc);
275 sd_ceph_conn_destroy(c);
276
277 printf("migrated %llu file(s), %llu byte(s) into %s (%llu failure(s))\n",
278 files, bytes, dest, fails);
279 return (fails == 0) ? 0 : 1;
280}
