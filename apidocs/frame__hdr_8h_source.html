1/*
2 * frame_hdr.h — XRootD response-frame field codecs (single source of truth).
3 *
4 * WHAT: unaligned-safe big-endian accessors + the small pure codecs both the
5 * module and the native client apply to server response frames:
6 * - xrd_resp_hdr_unpack: ServerResponseHeader (streamid/status/dlen)
7 * - xrd_wait_secs_parse: kXR_wait / kXR_waitresp retry-after seconds
8 * - xrd_error_body_decode: kXR_error [errnum][msg] (msg NOT NUL-guaranteed)
9 * WHY: these layouts were hand-decoded ~6× in the client (with unaligned
10 * `ntohl(*(uint32_t*)p)` casts — UB on non-aligned frame buffers — and two
11 * `%s` over-reads on the non-NUL error message) and parsed again on the
12 * server's proxy/upstream relay paths. One header keeps the wire facts in a
13 * single place and the access UB-free.
14 * HOW: header-only static inlines over the documented wire vocabulary — no ngx,
15 * no allocation, no OpenSSL. The memcpy+ntoh accessors are the exact idiom
16 * the safe server sites (compat/vendor_ext.c) already use.
17 *
18 * Clean-room: layouts from src/protocol (cross-checked vs XProtocol.hh).
19 */
20#ifndef BRIX_PROTOCOL_FRAME_HDR_H
21#define BRIX_PROTOCOL_FRAME_HDR_H

28/* ---- unaligned-safe big-endian accessors (memcpy + ntoh) ---- */
29static inline uint16_t
30xrd_get_u16_be(const void *p)
31{
32 uint16_t v;
33 memcpy(&v, p, sizeof(v));
34 return ntohs(v);
35}
36
37static inline uint32_t
38xrd_get_u32_be(const void *p)
39{
40 uint32_t v;
41 memcpy(&v, p, sizeof(v));
42 return ntohl(v);
43}
44
45static inline uint64_t
46xrd_get_u64_be(const void *p)
47{
48 uint32_t hi, lo;
49 memcpy(&hi, p, 4);
50 memcpy(&lo, (const uint8_t *) p + 4, 4);
51 return ((uint64_t) ntohl(hi) << 32) | (uint64_t) ntohl(lo);
52}

54static inline void
55xrd_put_u16_be(void *p, uint16_t v)
56{
57 v = htons(v);
58 memcpy(p, &v, sizeof(v));
59}

61static inline void

68static inline void
69xrd_put_u64_be(void *p, uint64_t v)
70{
71 uint32_t hi = htonl((uint32_t) (v >> 32));
72 uint32_t lo = htonl((uint32_t) (v & 0xffffffffu));
73 memcpy(p, &hi, 4);
74 memcpy((uint8_t *) p + 4, &lo, 4);
75}

77/*
78 * ServerResponseHeader = streamid[2] + status[2 BE] + dlen[4 BE] (8 bytes).
79 * streamid is an opaque 2-byte token echoed back; we read it big-endian so it
80 * compares equal to the value the sender wrote. Any of the out-params may be NULL.
81 */
82static inline void
83xrd_resp_hdr_unpack(const uint8_t h[8], uint16_t *streamid, uint16_t *status,
84 uint32_t *dlen)
85{
86 if (streamid != NULL) { *streamid = xrd_get_u16_be(h); }
87 if (status != NULL) { *status = xrd_get_u16_be(h + 2); }
88 if (dlen != NULL) { *dlen = xrd_get_u32_be(h + 4); }
89}
90
91/*
92 * kXR_wait / kXR_waitresp retry-after body = [int32 BE seconds]. Returns the
93 * advised seconds clamped to [1, cap]; uses `fallback` if the body is too short.
94 * The cap is caller-supplied on purpose — each role wants a different ceiling
95 * (a client honoring an origin's tape-recall vs a proxy absorbing for a client).
96 */
97static inline uint32_t
98xrd_wait_secs_parse(const uint8_t *body, uint32_t blen, uint32_t fallback,
99 uint32_t cap)
100{
101 uint32_t s = (blen >= 4 && body != NULL) ? xrd_get_u32_be(body) : fallback;
102 if (s < 1) { s = 1; }
103 if (s > cap) { s = cap; }
104 return s;
105}
106
107/*
108 * kXR_error body = [int32 BE errnum][message bytes]. The message is NOT
109 * NUL-terminated on the wire, so this returns a BOUNDED slice (*msg + *msglen) —
110 * callers MUST use the length (e.g. "%.*s"), never treat *msg as a C string.
111 * Returns 0 on success, -1 if the body is too short to hold errnum. Out-params
112 * other than the return may be NULL.
113 */
114static inline int
115xrd_error_body_decode(const uint8_t *body, uint32_t dlen, int *errnum,
116 const char **msg, size_t *msglen)
117{
118 if (body == NULL || dlen < 4) {
119 return -1;
120 }
121 if (errnum != NULL) { *errnum = (int) xrd_get_u32_be(body); }
122 if (msg != NULL) { *msg = (const char *) (body + 4); }
123 if (msglen != NULL) { *msglen = (size_t) (dlen - 4); }
124 return 0;
125}
