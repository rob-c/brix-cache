1/* walk.c — generic pre-order remote tree walk over brix_dirlist.
2 * WHAT: visit every entry under `path` (files and directories), parent first.
3 * WHY: xrdcksum tree, sync --delete and future tools all need one bounded,
4 * overflow-checked walker instead of four ad-hoc recursions.
5 * HOW: brix_dirlist(want_stat=1) per directory; recursion capped at
6 * BRIX_WALK_MAXDEPTH; path joins are length-checked. */

21static int
22walk_join(const char *dir, const char *name, char *out, size_t outsz)
23{
24 size_t dl = strlen(dir);
25 const char *sep = (dl > 0 && dir[dl - 1] == '/') ? "" : "/";
26 return ((size_t) snprintf(out, outsz, "%s%s%s", dir, sep, name) >= outsz)
27 ? -1 : 0;
28}

30static int
31tree_walk_depth(brix_conn *c, const char *path, int depth,
32 brix_walk_fn fn, void *u, brix_status *st)
33{
34 brix_dirent *ents = NULL;
35 size_t n = 0, i;
36 int rc = 0;
37
38 if (brix_dirlist(c, path, 1, &ents, &n, st) != 0) {
39 return -1;
40 }
41 for (i = 0; i < n && rc == 0; i++) {
42 char full[XRDC_PATH_MAX];
43 if (walk_is_dot(ents[i].name)) {
44 continue;
45 }
46 if (walk_join(path, ents[i].name, full, sizeof(full)) != 0) {
47 brix_status_set(st, XRDC_EUSAGE, 0, "walk: path too long under %s", path);
48 rc = -1;
49 break;
50 }
51 rc = fn(full, &ents[i], depth, u);
52 /* Normalize any non-zero visitor return to 1 (abort); -1 is reserved for
53 * internal walk errors (dirlist failure, depth cap, path overflow). */
54 if (rc != 0) {
55 rc = 1;
56 }
57 if (rc == 0 && ents[i].have_stat && (ents[i].st.flags & kXR_isDir)) {
58 if (depth + 1 >= BRIX_WALK_MAXDEPTH) {
59 brix_status_set(st, XRDC_EUSAGE, 0, "walk: depth cap at %s", full);
60 rc = -1;
61 } else {
62 rc = tree_walk_depth(c, full, depth + 1, fn, u, st);
63 }
64 }
65 }
66 free(ents);
67 return rc;
68}
