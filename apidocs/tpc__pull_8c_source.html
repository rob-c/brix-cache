1/*
2 * tpc_pull.c - HTTP-TPC COPY staged pull execution for the WebDAV module.
3 *
4 * WHAT: Owns the pull direction of HTTP-TPC once the request has been parsed and
5 * authorized: confined-target probing/preparation, the atomic-commit staged
6 * temp file, the three execution tiers (202-streaming marker, thread-pool
7 * task, synchronous curl), the link-vs-rename commit, and success/error
8 * finalisation with registry, dashboard, metric, and audit-ledger updates.
9 * WHY: Split from tpc.c (which was over the 500-line cap) so the tiered pull
10 * state machine — the most intricate part of the COPY path — is grouped and
11 * individually reviewable, separate from the dispatcher, request parsing,
12 * and push handling.
13 * HOW: webdav_tpc_prepare_pull_target, webdav_tpc_pull_start_accounting, and
14 * webdav_tpc_pull_execute (declared in tpc_internal_split.h) are called by
15 * the COPY dispatcher in tpc.c in the same order as before. All the internal
16 * probe/url/tier/commit/finish helpers stay static here. Shared request
17 * helpers (authorize, register, session-xfer note, dashboard identity) live
18 * in tpc.c and are reached through the split header. No behaviour change: the
19 * first tier that does not return NGX_DECLINED owns the response, and every
20 * error path aborts the staged temp file so a failed pull never leaves a
21 * partial file behind.
22 */

44/* Confined no-follow stat of `path` through the VFS probe, projected into the
45 * struct stat fields the TPC handler reads (mode for the dir check, size for the
46 * dashboard/registry/ledger). Non-metered. NGX_OK / NGX_DECLINED (errno kept). */
47static ngx_int_t
48webdav_tpc_probe(ngx_http_request_t *r,
49 ngx_http_brix_webdav_loc_conf_t *conf, const char *path, struct stat *sb)
50{
51 ngx_http_brix_webdav_req_ctx_t *rx =
52 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
53 brix_vfs_ctx_t vctx;
54 brix_vfs_stat_t vst;
55 int is_tls = 0;
56
57#if (NGX_HTTP_SSL)
58 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
59#endif
60
61 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
62 conf->common.root_canon, conf->cache_root_canon, conf->common.allow_write,
63 is_tls, (rx != NULL) ? rx->identity : NULL, path);
64 if (brix_vfs_probe(&vctx, 1 /* no-follow */, &vst) != NGX_OK) {
65 return NGX_DECLINED;
66 }
67 ngx_memzero(sb, sizeof(*sb));
68 sb->st_mode = (mode_t) vst.mode;
69 sb->st_size = vst.size;
70 sb->st_mtime = vst.mtime;
71 return NGX_OK;
72}

74/*
75 * Extract just the path component of a (possibly scheme-qualified) URL for use
76 * as an authorization scope, e.g. "https://host/a/b" -> "/a/b".
77 * `out` points into the caller's `url` buffer (no copy); defaults to "/" when
78 * the URL has no path. Used to scope-check the remote endpoint of a TPC.
79 */
80static void
81webdav_tpc_url_path(const char *url, ngx_str_t *out)
82{
83 const char *scheme;
84 const char *path;
85
86 out->data = (u_char *) "/"; /* default scope when URL has no path */
87 out->len = 1;
88
89 if (url == NULL) {
90 return;
91 }
92
93 /* If "://" is present, the path is the first '/' after it; otherwise the
94 * first '/' in the whole string. */
95 scheme = strstr(url, "://");
96 path = scheme != NULL ? strchr(scheme + 3, '/') : strchr(url, '/');
97 if (path == NULL || *path == '\0') {
98 return;
99 }
100
101 out->data = (u_char *) path;
102 out->len = ngx_strlen(path);
103}

105ngx_int_t
106webdav_tpc_prepare_pull_target(webdav_tpc_pull_ctx_t *pl)
107{
108 ngx_http_request_t *r = pl->r;
109 ngx_http_brix_webdav_loc_conf_t *conf = pl->conf;
110 const char *source_url = pl->source_url;
111 ngx_flag_t overwrite = pl->overwrite;
112 char *path = pl->path;
113 size_t path_len = pl->path_len;
114 struct stat *sb = pl->sb;
115 brix_staged_file_t *staged = pl->staged;
116 ngx_int_t rc;
117 ngx_str_t src_scope;
118 ngx_str_t dst_scope;
119
120 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon, path,
121 path_len);
122 if (rc != NGX_OK) {
123 return rc;
124 }
125
126 pl->existed = (webdav_tpc_probe(r, conf, path, sb) == NGX_OK) ? 1 : 0;
127 if (pl->existed && S_ISDIR(sb->st_mode)) {
128 return NGX_HTTP_CONFLICT;
129 }
130 if (pl->existed && !overwrite) {
131 return NGX_HTTP_PRECONDITION_FAILED;
132 }
133
134 webdav_tpc_url_path(source_url, &src_scope);
135 dst_scope = r->uri;
136 rc = webdav_tpc_authorize(r, &src_scope, &dst_scope);
137 if (rc != NGX_OK) {
138 return rc;
139 }
140
141 {
142 brix_staged_open_req_t oreq = {
143 .root_canon = conf->common.root_canon,
144 .final_path = path,
145 .open_flags = O_WRONLY,
146 .mode = 0600,
147 .attempts = 16,
148 };
149 if (brix_staged_open(r->connection->log, &oreq, staged) != NGX_OK) {
150 return errno == ENAMETOOLONG ? NGX_HTTP_REQUEST_URI_TOO_LARGE
151 : NGX_HTTP_INTERNAL_SERVER_ERROR;
152 }
153 }
154 ngx_close_file(staged->fd);
155 staged->fd = NGX_INVALID_FILE;
156 return NGX_OK;
157}

172static ngx_int_t
173webdav_tpc_pull_marker_exec(webdav_tpc_pull_ctx_t *pl)
174{
175 ngx_http_request_t *r = pl->r;
176 ngx_http_brix_webdav_loc_conf_t *conf = pl->conf;
177 const char *source_url = pl->source_url;
178 const char *path = pl->path;
179 brix_staged_file_t *staged = pl->staged;
180 ngx_array_t *transfer_headers = pl->transfer_headers;
181 ngx_uint_t n_streams = pl->n_streams;
182 ngx_flag_t overwrite = pl->overwrite;
183 ngx_flag_t existed = pl->existed;
184 ngx_int_t rc;
185 uint64_t transfer_id;
186
187 if (conf->tpc_marker_interval <= 0) {
188 return NGX_DECLINED;
189 }

192 if (transfer_id == 0) {
193 brix_dashboard_http_error(r, "webdav TPC registry full");
194 brix_dashboard_http_finish(r);
195 brix_staged_abort(r->connection->log, conf->common.root_canon,
196 staged, 1);
197 return NGX_HTTP_SERVICE_UNAVAILABLE;
198 }
199 rc = webdav_tpc_marker_start(r, conf, n_streams, source_url,
200 staged->tmp_path, path, 1, overwrite, existed,
201 transfer_headers, transfer_id,
202 pl->user_cert, pl->user_key);
203 if (rc == NGX_DECLINED) {
204 (void) brix_tpc_registry_remove(transfer_id, r->connection->log);
205 return NGX_DECLINED;
206 }
207 if (rc != NGX_DONE) {
208 (void) brix_tpc_registry_remove(transfer_id, r->connection->log);
209 brix_dashboard_http_error(r, "webdav TPC marker start failed");
210 brix_dashboard_http_finish(r);
211 brix_staged_abort(r->connection->log, conf->common.root_canon,
212 staged, 1);
213 return (rc == NGX_ERROR) ? NGX_HTTP_INTERNAL_SERVER_ERROR : rc;
214 }
215 return NGX_DONE;

315 "brix_webdav: HTTP-TPC rename failed for: \"%s\"",

318 }
319
320 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_COMMIT_ERROR]);
321 brix_dashboard_http_error(r, "webdav TPC pull commit failed");
322 brix_dashboard_http_finish(r);
323 brix_staged_abort(r->connection->log, conf->common.root_canon, staged, 1);
324 (void) brix_tpc_registry_update(transfer_id, 0, BRIX_TPC_STATE_ERROR,
325 r->connection->log);
326 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV, BRIX_TPC_DIR_PULL,
327 BRIX_TPC_METRIC_ERROR, 0, r->connection->log);
328 (void) brix_tpc_registry_remove(transfer_id, r->connection->log);
329 return status;

370/* ---- Run the staged pull to completion across the three execution tiers ----
371 *
372 * WHAT: Executes the prepared pull described by pl, then commits and finalises it.
373 * Returns the terminal NGX_HTTP_* / NGX_OK status from whichever tier or
374 * stage produced it.
375 *
376 * WHY: Isolates the tiered pull-execute + commit + finish sequence from the COPY
377 * orchestrator so each is separately reviewable. Behaviour is unchanged: the
378 * first tier that does not return NGX_DECLINED owns the response, and any
379 * commit/finish error path aborts the staged temp file so a failed pull never
380 * leaves a partial file behind.
381 *
382 * HOW: Pull executes via the first available of three tiers:
383 * (1) 202-streaming marker path when tpc_marker_interval > 0: async
384 * transfer that dribbles Performance-Marker lines to the client.
385 * (2) thread-pool task when a thread pool is configured.
386 * (3) synchronous curl (final fallback) that blocks the worker.
387 * Each tier returns NGX_DECLINED to mean "not applicable, try the next".
388 * 1. Try the marker tier; return its status unless NGX_DECLINED.
389 * 2. Try the thread tier; return its status unless NGX_DECLINED.
390 * 3. Run the synchronous tier; return on error.
391 * 4. Commit the staged temp file (link for no-overwrite, rename for
392 * overwrite); return on error.
393 * 5. Finish with success accounting and the final response.
394 */
395ngx_int_t
396webdav_tpc_pull_execute(webdav_tpc_pull_ctx_t *pl)
397{
398 ngx_int_t rc;
399
400 rc = webdav_tpc_pull_marker_exec(pl);
401 if (rc != NGX_DECLINED) {
402 return rc;
403 }
404
405 rc = webdav_tpc_pull_thread_exec(pl);
406 if (rc != NGX_DECLINED) {
407 return rc;
408 }
409
410 rc = webdav_tpc_pull_sync_exec(pl);
411 if (rc != NGX_OK) {
412 return rc;
413 }
414
415 rc = webdav_tpc_commit_pulled_file(pl);
416 if (rc != NGX_OK) {
417 return rc;
418 }
419
420 return webdav_tpc_finish_pull_success(pl);
421}
