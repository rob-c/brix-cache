1/*
2 * gsi_core_cresp_util.c — round-2 XrdSecgsi cert-response leaf helpers.
3 *
4 * WHAT: The self-contained, stateless helpers the round-2 cert-response builder
5 * (gsi_core_cresp.c) leans on: extract a peer certificate's public key
6 * from a PEM bucket (gsi_cresp_cert_pubkey), export an EVP_PKEY public
7 * part as PEM SubjectPublicKeyInfo (gsi_cresp_export_pubkey_pem), choose
8 * the kXRS_md_alg digest to echo back (gsi_cresp_pick_md_alg), free the
9 * whole owned-resource set on the single error/success path
10 * (gsi_cresp_fail, operating on gsi_cresp_ctx), and the opt-in full-proxy
11 * passthrough bucket (gsi_add_fullproxy_bucket).
12 *
13 * WHY: gsi_core.c carried the entire round-2 exchange and exceeded the ~500-line
14 * file-size guard, so the leaf helpers were carved out here and the state
15 * machine into gsi_core_cresp.c (phase-79). These five functions are
16 * leaves — they take inputs and return outputs (or free a passed context)
17 * without touching the gsi_cresp_state_t aggregate — so they belong in
18 * their own translation unit. Splitting moves no logic, changes no OpenSSL
19 * call order and no emitted bytes: the DH exchange, cert/response build,
20 * and signature crypto are byte-for-byte unchanged from the original.
21 *
22 * HOW: All entry points cross a TU boundary into gsi_core_cresp.c and are
23 * declared non-static in gsi_core_internal.h, the shared private contract
24 * both files include.
25 */

29/* Public key of the first X.509 cert in a PEM bucket (the peer's EEC). */
30EVP_PKEY *
31gsi_cresp_cert_pubkey(const uint8_t *pem, size_t len)
32{
33 BIO *bio = BIO_new_mem_buf(pem, (int) len);
34 X509 *cert = bio ? PEM_read_bio_X509(bio, NULL, NULL, NULL) : NULL;
35 EVP_PKEY *pk = cert ? X509_get_pubkey(cert) : NULL;
36
37 X509_free(cert);
38 BIO_free(bio);
39 return pk;
40}

43/* Export an EVP_PKEY public part as PEM SubjectPublicKeyInfo
44 * (XrdCryptosslRSA::ExportPublic = PEM_write_bio_PUBKEY). malloc'd,
45 * NUL-terminated, *outlen = strlen. The signed-DH path sends this as kXRS_puk so
46 * the server can verify our signed DH before it has decrypted our cert chain. */
47char *
48gsi_cresp_export_pubkey_pem(EVP_PKEY *key, size_t *outlen)
49{
50 BIO *bio = BIO_new(BIO_s_mem());
51 char *pem = NULL, *out = NULL;
52 long len;
53
54 if (bio != NULL && PEM_write_bio_PUBKEY(bio, key) == 1) {
55 len = BIO_get_mem_data(bio, &pem);
56 out = malloc((size_t) len + 1);
57 if (out != NULL) {
58 memcpy(out, pem, (size_t) len);
59 out[len] = '\0';
60 *outlen = (size_t) len;
61 }
62 }
63 BIO_free(bio);
64 return out;
65}

68/* Choose the kXRS_md_alg digest to echo back: prefer "sha256" when the server
69 * offers it (it matches our SHA256(secret) key derivation), else the server's
70 * first ':'-separated token, else "sha256". dCache rejects the handshake if the
71 * reply names a digest it did not advertise. */
72size_t
73gsi_cresp_pick_md_alg(const uint8_t *sbody, uint32_t slen, char *out, size_t outcap)
74{
75 const uint8_t *list = NULL;
76 size_t listlen = 0, n = 0;
77
78 if (brix_gsi_find_bucket(sbody, slen, (uint32_t) kXRS_md_alg,
79 &list, &listlen) == 0 && listlen > 0) {
80 if (listlen >= 6) {
81 for (size_t i = 0; i + 6 <= listlen; i++) {
82 if (memcmp(list + i, "sha256", 6) == 0) {
83 memcpy(out, "sha256", 6);
84 out[6] = '\0';
85 return 6;
86 }
87 }
88 }
89 while (n < listlen && n + 1 < outcap && list[n] != ':') {
90 out[n] = (char) list[n];
91 n++;
92 }
93 }
94 if (n == 0) {
95 memcpy(out, "sha256", 6);
96 n = 6;
97 }
98 out[n] = '\0';
99 return n;
100}

103/*

105 * gsi_cresp_cleanup (single NULL-safe destructor, no goto). proxy_pem and the

110{

123 brix_gbuf_free(&x->inner);
124 brix_gbuf_free(&x->outer);

129/*
130 * gsi_add_fullproxy_bucket — OPT-IN client full-proxy passthrough (phase-70
131 * §5.1). When XRD_DELEGATEFULLPROXY is set, load the FULL proxy (cert chain +
132 * private key PEM) from $X509_USER_PROXY and append it as a kXRS_x509_fullproxy
133 * bucket in the encrypted inner buffer, so a delegation-enabled node can present
134 * the user's own proxy upstream. STRICTLY off by default: no env ⇒ no-op, so
135 * the shared kernel's server/TPC callers (which never set the var) are inert.
136 * Best-effort — a load failure silently omits the bucket (login still succeeds
137 * with the normal chain-only kXRS_x509). The bytes carry a private key: the
138 * caller encrypts the whole inner buffer, so they never cross the wire in clear,
139 * and the server accepts them only under TLS.
140 */
141void
142gsi_add_fullproxy_bucket(brix_gbuf *inner)
143{
144 const char *proxy;
145 FILE *fp;
146 uint8_t buf[16384];
147 size_t total;
148
149 if (getenv("XRD_DELEGATEFULLPROXY") == NULL) {
150 return;
151 }
152 proxy = getenv("X509_USER_PROXY");
153 if (proxy == NULL || proxy[0] == '\0') {
154 return;
155 }
156
157 fp = fopen(proxy, "r");
158 if (fp == NULL) {
159 return;
160 }
161 /* A grid proxy file stores the proxy cert chain followed by its private key
162 * in PEM — forward it verbatim as the full proxy (chain + key). Capped at
163 * buf[]; a proxy larger than that omits the bucket rather than truncating. */
164 total = fread(buf, 1, sizeof(buf), fp);
165 if (total > 0 && total < sizeof(buf)) {
166 brix_gbuf_bucket(inner, (uint32_t) kXRS_x509_fullproxy, buf, total);
167 }
168 OPENSSL_cleanse(buf, sizeof(buf));
169 (void) fclose(fp); /* phase74-fp: read-only stream, close failure cannot lose data */
170}
