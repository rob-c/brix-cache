2 * auth_sigv4_verify_time.c — SigV4 request-timestamp parsing and clock-skew /
3 * presigned-expiry enforcement.
4 *
5 * WHAT: Owns the SigV4 timestamp concern carved out of auth_sigv4_verify.c:
6 * the fixed-width YYYYMMDDTHHMMSSZ datetime parser, the header/presigned
7 * clock-skew and expiry rules, and the resolver that turns either source
8 * into the full amz timestamp view (s3_amz_date_out_t) the string-to-sign
9 * consumes. Every rejection edge records the SigV4 auth result and emits
10 * the frozen XML error.
11 * WHY: auth_sigv4_verify.c exceeded the ~500-line file-size guard (phase-79).
12 * Timestamp parsing + skew arithmetic is one cohesive, side-effect-light
13 * concern with two mirrored source paths (presigned query param vs the
14 * x-amz-date header), so it splits cleanly from the verifier orchestrator
15 * and the byte-frozen crypto. INVARIANT §6: SigV4-only — no bearer logic.
16 * HOW: Pure digit/datetime parsers feed timegm(); the two skew checks bound the
17 * parsed time against ngx_time(); s3_sigv4_resolve_request_time forks on
18 * comp->presigned and returns out->date/out->len set to a buffer that
19 * outlives the call (comp's own storage for presigned, out->buf for the
20 * header path). Byte layout and error messages preserved 1:1 with the
21 * pre-split single-file implementation.
22 */

35/*
36 * s3_digit — map an ASCII byte to its decimal value.
37 *
38 * WHAT: Return 0-9 for an ASCII digit byte, or -1 for any non-digit.
39 * WHY: The amz timestamp is a fixed-width all-ASCII-digit field; a branch-free
40 * per-byte validator keeps the datetime parser strict and linear.
41 * HOW: Range-check against '0'..'9' and subtract '0'.
42 */
43static int
44s3_digit(u_char c)
45{
46 return (c >= '0' && c <= '9') ? (int) (c - '0') : -1;
47}

49/*
50 * s3_parse_2digits — parse a 2-character decimal field.
51 *
52 * WHAT: Return the value of s[0..1] as a 0-99 integer, or -1 if either byte is
53 * not a digit.
54 * WHY: Month/day/hour/minute/second fields of the amz timestamp are all
55 * 2-digit; one validated helper avoids repeating the digit math.
56 * HOW: Validate both digits, combine hi*10 + lo.
57 */
58static int
59s3_parse_2digits(const char *s)
60{
61 int hi, lo;
62
63 hi = s3_digit((u_char) s[0]);
64 lo = s3_digit((u_char) s[1]);
65 if (hi < 0 || lo < 0) {
66 return -1;
67 }
68 return hi * 10 + lo;
69}

71/*
72 * s3_parse_4digits — parse a 4-character decimal field.
73 *
74 * WHAT: Return the value of s[0..3] as a 0-9999 integer, or -1 if any byte is
75 * not a digit.
76 * WHY: The amz timestamp year is a 4-digit field; a dedicated validator keeps
77 * the parser uniform.
78 * HOW: Validate all four digits, combine into a base-10 value.
79 */
80static int
81s3_parse_4digits(const char *s)
82{
83 int a, b, c, d;
84
85 a = s3_digit((u_char) s[0]);
86 b = s3_digit((u_char) s[1]);
87 c = s3_digit((u_char) s[2]);
88 d = s3_digit((u_char) s[3]);
89 if (a < 0 || b < 0 || c < 0 || d < 0) {
90 return -1;
91 }
92 return a * 1000 + b * 100 + c * 10 + d;
93}

95/*
96 * s3_parse_amz_datetime — parse a SigV4 YYYYMMDDTHHMMSSZ timestamp to time_t.
97 *
98 * WHAT: Parse the fixed 16-byte amz timestamp into *out (UTC epoch seconds),
99 * returning NGX_OK, or NGX_ERROR when the layout or any field is invalid.
100 * WHY: SigV4 timestamps are always the fixed ISO-8601 basic UTC form; strict
101 * field validation prevents timegm from normalising a malformed value.
102 * HOW: Require length 16 with 'T' at [8] and 'Z' at [15]; parse each field;
103 * range-check; populate a zeroed struct tm and convert via timegm.
104 */
105static ngx_int_t
106s3_parse_amz_datetime(const char *s, time_t *out)
107{
108 struct tm tm;
109 int year, mon, day, hour, min, sec;
110
111 if (strlen(s) != 16 || s[8] != 'T' || s[15] != 'Z') {
112 return NGX_ERROR;
113 }
114
115 year = s3_parse_4digits(s);
116 mon = s3_parse_2digits(s + 4);
117 day = s3_parse_2digits(s + 6);
118 hour = s3_parse_2digits(s + 9);
119 min = s3_parse_2digits(s + 11);
120 sec = s3_parse_2digits(s + 13);
121
122 if (year < 1970 || mon < 1 || mon > 12 || day < 1 || day > 31
123 || hour < 0 || hour > 23 || min < 0 || min > 59
124 || sec < 0 || sec > 60)
125 {
126 return NGX_ERROR;
127 }
128
129 ngx_memzero(&tm, sizeof(tm));
130 tm.tm_year = year - 1900;
131 tm.tm_mon = mon - 1;
132 tm.tm_mday = day;
133 tm.tm_hour = hour;
134 tm.tm_min = min;
135 tm.tm_sec = sec;
136 *out = timegm(&tm);
137 return NGX_OK;
138}

175/*
176 * s3_check_header_clock_skew — bound a header-signed request's timestamp.
177 *
178 * WHAT: Return NGX_OK when request_time is within +/-900s of now, otherwise the
179 * terminal skew rejection.
180 * WHY: Header-signed requests may legitimately be slightly ahead of or behind
181 * server time; a symmetric window blocks replay/forgery outside it.
182 * HOW: Compare the signed magnitude of (request_time - now) against the window.
183 */
184static ngx_int_t
185s3_check_header_clock_skew(ngx_http_request_t *r, time_t request_time)
186{
187 time_t now;
188
189 now = ngx_time();
190 if (request_time > now) {
191 if (request_time - now > S3_SIGV4_MAX_HEADER_SKEW_SEC) {
192 return s3_reject_clock_skew(r);
193 }
194 } else if (now - request_time > S3_SIGV4_MAX_HEADER_SKEW_SEC) {
195 return s3_reject_clock_skew(r);
196 }
197
198 return NGX_OK;
199}

201/*
202 * s3_check_presigned_future_skew — reject a presigned request dated too far ahead.
203 *
204 * WHAT: Return NGX_OK unless request_time is more than 900s in the future, in
205 * which case return the terminal skew rejection.
206 * WHY: Presigned URLs carry their own expiry window, so only the future edge
207 * needs bounding here; the past edge is the separate expiry check.
208 * HOW: Reject only when request_time - now exceeds the future window.
209 */
210static ngx_int_t
211s3_check_presigned_future_skew(ngx_http_request_t *r, time_t request_time)
212{
213 time_t now;
214
215 now = ngx_time();
216 if (request_time > now
217 && request_time - now > S3_SIGV4_MAX_FUTURE_SKEW_SEC)
218 {
219 return s3_reject_clock_skew(r);
220 }
221
222 return NGX_OK;
223}

225/*
226 * s3_sigv4_resolve_request_time — resolve and skew/expiry-check the request time.
227 *
228 * WHAT: Parse the SigV4 timestamp (presigned X-Amz-Date query param or the
229 * x-amz-date header), enforce clock-skew / presigned-expiry limits, and
230 * return the full amz timestamp string used in the string-to-sign.
231 * WHY: The presigned and header paths compute the same downstream inputs
232 * (amz_date + amz_date_len) via different sources and different skew
233 * rules; folding both into one helper keeps the verifier body linear.
234 * HOW: Presigned path: parse query-supplied timestamp, future-skew + expiry
235 * check. Header path: parse the header timestamp, symmetric skew check.
236 * out->date is set to a buffer that outlives this call — either comp's
237 * own storage (presigned) or out->buf (the header path). Frozen error
238 * messages and result codes preserved 1:1.
239 *
240 * Returns: NGX_OK with out->date/out->len set, otherwise the terminal ngx_int_t
241 * result (response already emitted).
242 */
243ngx_int_t
244s3_sigv4_resolve_request_time(ngx_http_request_t *r,
245 const sigv4_components_t *comp, s3_amz_date_out_t *out)
246{
247 time_t request_time = 0;
248 ngx_str_t date_hdr;
249 ngx_int_t skew_rc;
250
251 if (comp->presigned) {
252 if (s3_parse_amz_datetime(comp->amz_date, &request_time) != NGX_OK) {
253 return s3_reject_bad_amz_date(r, "Invalid X-Amz-Date");
254 }
255
256 skew_rc = s3_check_presigned_future_skew(r, request_time);
257 if (skew_rc != NGX_OK) {
258 return skew_rc;
259 }
260
261 if (ngx_time() >= request_time
262 && ngx_time() - request_time > (time_t) comp->amz_expires)
263 {
264 s3_record_auth_result(BRIX_S3_AUTH_BAD_DATE);
265 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
266 "AccessDenied",
267 "Request has expired");
268 }
269
270 out->date = comp->amz_date;
271 out->len = strlen(comp->amz_date);
272 return NGX_OK;
273 }
274
275 date_hdr = get_header(r, "x-amz-date");
276 if (date_hdr.len == 0) {
277 return s3_reject_bad_amz_date(r, "Missing X-Amz-Date header");
278 }
279
280 if (brix_str_cbuf(out->buf, sizeof(out->buf), &date_hdr) == NULL) {
281 return s3_reject_bad_amz_date(r, "Invalid X-Amz-Date");
282 }
283
284 if (s3_parse_amz_datetime(out->buf, &request_time) != NGX_OK) {
285 return s3_reject_bad_amz_date(r, "Invalid X-Amz-Date");
286 }
287
288 skew_rc = s3_check_header_clock_skew(r, request_time);
289 if (skew_rc != NGX_OK) {
290 return skew_rc;
291 }
292
293 out->date = out->buf;
294 out->len = strlen(out->buf);
295 return NGX_OK;
296}
