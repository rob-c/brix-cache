1/* client/lib/cred_s3.c
2 *
3 * WHAT: S3-keys credential handler for the unified credential store (cred.c).
4 * Implements brix_cred_s3keys() returning the XRDC_CRED_S3KEYS handler
5 * with available / acquire operations.
6 * WHY: S3 key discovery was inline in s3.c with no shared path for auth
7 * pre-flight diagnostics. A unified handler lets the store serve both the
8 * S3 transport (s3.c) and the auth pre-flight probe (credinfo.c) from one
9 * place and covers all three discovery levels consistently.
10 * HOW: Discovery precedence (highest to lowest):
11 * 1. cfg->s3_access + cfg->s3_secret (CLI --s3-access/--s3-secret)
12 * 2. $AWS_ACCESS_KEY_ID + $AWS_SECRET_ACCESS_KEY (environment)
13 * 3. ~/.aws/credentials [default] section
14 * (aws_access_key_id / aws_secret_access_key)
15 * Both access key AND secret are required; a partial pair is treated as
16 * "not available".
17 *
18 * available() returns 1 iff a complete pair is obtainable via any level.
19 * acquire() fills out->s3_access + out->s3_secret from function-static
20 * buffers and sets *not_after=0 (static keys have no expiry).
21 * refresh = NULL (static keys are long-lived; rotation is operator-driven).
22 *
23 * Lifetime contract (cred.h §LIFETIME CONTRACT): acquire() writes
24 * out->s3_access / out->s3_secret as pointers to function-local static
25 * char[] buffers that remain valid after acquire() returns. The store's
26 * slot_store_view strdup's them immediately afterward — before any other
27 * acquire can overwrite the buffers. Single-threaded per-acquire, so one
28 * static buffer pair is safe for this window.
29 *
30 * Secret values are NEVER written to log or error messages.
31 *
32 * ngx-free. No goto. Functional/modular: one responsibility per function.
33 */

49/* ~/.aws/credentials INI parser */
50/*
51 * trim_inplace — strip leading and trailing ASCII whitespace from a string.
52 *
53 * WHAT: modifies s in-place; returns a pointer to the first non-space char.
54 * WHY: INI files may have arbitrary whitespace around '=' or at line ends;
55 * normalising both sides before comparison avoids false-negative matches.
56 * HOW: advance start past spaces; NUL-terminate at the last non-space.
57 * The returned pointer may be s or an interior location within s.
58 */
59static char *
60trim_inplace(char *s)
61{
62 char *end;
63
64 while (*s == ' ' || *s == '\t') {
65 s++;
66 }
67 if (*s == '\0') {
68 return s;
69 }
70 end = s + strlen(s) - 1;
71 while (end > s && (*end == ' ' || *end == '\t' || *end == '\r' || *end == '\n')) {
72 *end-- = '\0';
73 }
74 return s;
75}

77/*
78 * parse_ini_key_value — extract the value for a given key from an INI line.
79 *
80 * WHAT: checks whether the trimmed line matches "<key> = <value>" (whitespace
81 * around '=' is tolerated); on match copies the trimmed value into out[outsz].
82 * WHY: shared between the access-key and secret-key extraction so each has one
83 * call site with the same logic.
84 * HOW: look for '='; split; trim both sides; compare the key; copy the value.
85 * Returns 1 on match, 0 otherwise. out is always NUL-terminated within outsz.
86 * Mutates buf in place (NUL-terminates at '=').
87 */
88static int
89parse_ini_key_value(char *buf, const char *key,
90 char *out, size_t outsz)
91{
92 /* mutates buf in place */
93 char *eq = strchr(buf, '=');
94 char *lkey;
95 char *lval;
96
97 if (eq == NULL) {
98 return 0;
99 }
100 *eq = '\0';
101 lkey = trim_inplace(buf);
102 lval = trim_inplace(eq + 1);
103
104 if (strcmp(lkey, key) != 0) {
105 return 0;
106 }
107 snprintf(out, outsz, "%s", lval);
108 return 1;
109}

111/*
112 * parse_aws_credentials_default — read the [default] section of an AWS
113 * credentials file and extract the access key and secret.
114 *
115 * WHAT: opens the credentials file at `path` and scans it line by line;
116 * populates access_out[access_sz] and secret_out[secret_sz] when
117 * the [default] section's two keys are found.
118 * WHY: minimal INI reader — no third-party dependency, handles only the
119 * fields the S3 handler needs.
120 * HOW: track `in_default` flag: set on "[default]", cleared on any other
121 * "[section]" header. Extract aws_access_key_id and aws_secret_access_key
122 * while in_default; stop early once both are found.
123 * Returns 1 iff both access_out and secret_out are non-empty.
124 */
125static int
126parse_aws_credentials_default(const char *path,
127 char *access_out, size_t access_sz,
128 char *secret_out, size_t secret_sz)
129{
130 FILE *f;
131 char line[1024];
132 int in_default = 0;
133 int got_access = 0;
134 int got_secret = 0;
135
136 access_out[0] = '\0';
137 secret_out[0] = '\0';
138
139 f = fopen(path, "r");
140 if (f == NULL) {
141 return 0;
142 }
143
144 while (fgets(line, (int)sizeof(line), f) != NULL) {
145 char *t = trim_inplace(line);
146
147 /* Skip blank lines and comments. */
148 if (t[0] == '\0' || t[0] == '#' || t[0] == ';') {
149 continue;
150 }
151
152 /* Section header. */
153 if (t[0] == '[') {
154 in_default = (strcmp(t, "[default]") == 0) ? 1 : 0;
155 continue;
156 }
157
158 if (!in_default) {
159 continue;
160 }
161
162 if (!got_access) {
163 char tmp[S3KEY_BUFSZ];
164 if (parse_ini_key_value(t, "aws_access_key_id",
165 tmp, sizeof(tmp)) && tmp[0] != '\0') {
166 snprintf(access_out, access_sz, "%s", tmp);
167 got_access = 1;
168 }
169 }
170 if (!got_secret) {
171 char tmp[S3KEY_BUFSZ];
172 if (parse_ini_key_value(t, "aws_secret_access_key",
173 tmp, sizeof(tmp)) && tmp[0] != '\0') {
174 snprintf(secret_out, secret_sz, "%s", tmp);
175 got_secret = 1;
176 }
177 }
178
179 if (got_access && got_secret) {
180 break; /* found both keys; no need to read further */
181 }
182 }
183
184 fclose(f);
185 return (got_access && got_secret) ? 1 : 0;
186}

188/*
189 * try_credentials_file — attempt to read ~/.aws/credentials using $HOME.
190 *
191 * WHAT: builds the credentials path as $HOME/.aws/credentials; delegates to
192 * parse_aws_credentials_default.
193 * WHY: isolated from discover_s3_keys so the path-construction logic has one
194 * home and can be unit-tested independently.
195 * HOW: getenv("HOME"); snprintf the path; call parse_aws_credentials_default.
196 * Returns 1 iff both keys were found. Returns 0 if $HOME is absent.
197 */
198static int
199try_credentials_file(char *access_out, size_t access_sz,
200 char *secret_out, size_t secret_sz)
201{
202 const char *home = getenv("HOME");
203 char path[XRDC_PATH_MAX];
204
205 if (home == NULL || home[0] == '\0') {
206 return 0;
207 }
208 snprintf(path, sizeof(path), "%s/.aws/credentials", home);
209 return parse_aws_credentials_default(path, access_out, access_sz,
210 secret_out, secret_sz);
211}

213/* key discovery */
214/*
215 * discover_s3_keys — probe all three discovery levels; fill access + secret.
216 *
217 * WHAT: tries cfg override → environment → ~/.aws/credentials in order;
218 * returns 1 iff a complete (access + secret) pair was found.
219 * WHY: single authoritative resolution path shared by available() and acquire()
220 * so both always agree on whether keys are present.
221 * HOW: three explicit, ordered checks with early-return on the first complete
222 * pair. Partial pairs (access without secret or vice-versa) are skipped
223 * so a misconfigured override does not shadow a good env / file pair.
224 */
225static int
226discover_s3_keys(const brix_cred_config *cfg,
227 char *access_out, size_t access_sz,
228 char *secret_out, size_t secret_sz)
229{
230 const char *env_access;
231 const char *env_secret;
232
233 /* Level 1: explicit cfg fields (both must be present and non-empty). */
234 if (cfg != NULL
235 && cfg->s3_access != NULL && cfg->s3_access[0] != '\0'
236 && cfg->s3_secret != NULL && cfg->s3_secret[0] != '\0') {
237 snprintf(access_out, access_sz, "%s", cfg->s3_access);
238 snprintf(secret_out, secret_sz, "%s", cfg->s3_secret);
239 return 1;
240 }
241
242 /* Level 2: environment variables (both must be set and non-empty). */
243 env_access = getenv("AWS_ACCESS_KEY_ID");
244 env_secret = getenv("AWS_SECRET_ACCESS_KEY");
245 if (env_access != NULL && env_access[0] != '\0'
246 && env_secret != NULL && env_secret[0] != '\0') {
247 snprintf(access_out, access_sz, "%s", env_access);
248 snprintf(secret_out, secret_sz, "%s", env_secret);
249 return 1;
250 }
251
252 /* Level 3: ~/.aws/credentials [default] section. */
253 return try_credentials_file(access_out, access_sz, secret_out, secret_sz);
254}

272/*
273 * s3keys_acquire — discover the S3 key pair and fill *out.
274 *
275 * WHAT: the store's acquire call for XRDC_CRED_S3KEYS.
276 * WHY: single canonical acquire path for AWS SigV4 credentials.
277 * HOW: 1) discover_s3_keys into function-static buffers (lifetime contract);
278 * 2) on failure → -1 + XRDC_EAUTH (no secret values in the message);
279 * 3) fill out: kind=S3KEYS, s3_access=s_access, s3_secret=s_secret,
280 * not_after=0 (static keys have no per-use expiry).
281 *
282 * Lifetime of s_access / s_secret: declared static so they remain valid after
283 * this function returns. slot_store_view strdup's out->s3_{access,secret}
284 * before any other acquire can overwrite the buffers.
285 *
286 * Secret values are NEVER written into st->msg or any log path.
287 */
288static int
289s3keys_acquire(const brix_cred_config *cfg, brix_cred_view *out,
290 int64_t *not_after, brix_status *st)
291{
292 /* static: must outlive this return so slot_store_view can strdup them */
293 static char s_access[S3KEY_BUFSZ];
294 static char s_secret[S3KEY_BUFSZ];
295
296 if (!discover_s3_keys(cfg, s_access, sizeof(s_access),
297 s_secret, sizeof(s_secret))) {
298 brix_status_set(st, XRDC_EAUTH, 0,
299 "s3keys: no S3 credentials found "
300 "(set AWS_ACCESS_KEY_ID/AWS_SECRET_ACCESS_KEY "
301 "or add [default] to ~/.aws/credentials)");
302 return -1;
303 }
304
305 out->kind = XRDC_CRED_S3KEYS;
306 out->path = NULL;
307 out->token = NULL;
308 out->s3_access = s_access; /* static buffer; store deep-copies via slot_store_view */
309 out->s3_secret = s_secret; /* static buffer; store deep-copies via slot_store_view */
310
311 *not_after = 0; /* static keys have no per-use expiry */
312 return 0;
313}
