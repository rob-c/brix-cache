1/*
2 * idmap_gridmap.c — grid-mapfile (DN -> local username) parse/load/lookup.
3 *
4 * WHAT: Parses the classic grid-mapfile (each line: "<quoted DN>" <username>)
5 * once at init into a small array of (quoted-DN, username) pairs, and answers
6 * exact-DN lookups against it. Split out of idmap.c (file-size cap).
7 *
8 * WHY: This is a pure, self-contained text-parsing concern with no dependency on
9 * the numeric mapping policy or the deny-lists — it only feeds a candidate
10 * username to the resolver.
11 *
12 * HOW: The mapfile table (idmap_gridmap[]) is private to this translation unit;
13 * only idmap_gridmap_load()/idmap_gridmap_lookup() cross the boundary (declared
14 * in idmap_internal.h). No goto; pure helpers, side effects at the edges.
15 */

55/* Skip leading spaces/tabs; returns the first non-blank position. */
56static const char *
57idmap_skip_blanks(const char *p)
58{
59 while (*p == ' ' || *p == '\t') { p++; }

61}

63/*
64 * Scan the leading "<quoted DN>" field of a grid-mapfile line. Comments (#),
65 * blank lines, unquoted and unterminated DNs are all "no field here". Pure
66 * scan: on success sets [*start, *end) to the DN bytes (quotes excluded),
67 * advances *pp past the closing quote, and returns 1; returns 0 to skip.
68 */
69static int
70idmap_scan_quoted_dn(const char **pp, const char **start, const char **end)
71{
72 const char *p = idmap_skip_blanks(*pp);
73
74 if (*p != '"') {
75 return 0; /* comment, blank, or unquoted -> skip */
76 }
77 *start = ++p; /* after the opening quote */
78 while (*p != '"' && *p != '\0' && *p != '\n') { p++; }
79 if (*p != '"') {
80 return 0; /* unterminated quote -> skip */
81 }
82 *end = p; /* at closing quote */
83 *pp = p + 1;
84 return 1;
85}

87/*
88 * Scan the local-username field that follows the DN. Pure scan: sets
89 * [*start, *end) to the token and advances *pp; returns 1 on a non-empty
90 * username, 0 when the field is missing.
91 */
92static int
93idmap_scan_username(const char **pp, const char **start, const char **end)
94{
95 const char *p = idmap_skip_blanks(*pp);
96
97 *start = p;
98 while (*p != '\0' && *p != '\n' && *p != ' ' && *p != '\t') { p++; }
99 *end = p;
100 *pp = p;
101 return *end != *start;
102}

104/*
105 * Parse one grid-mapfile line of the classic form:
106 * "<quoted DN>" <local-username>
107 * Comments (#) and blank lines are skipped. Returns 1 and fills the two out
108 * params (malloc'd) on a usable mapping, 0 to skip the line, -1 on OOM.
109 */
110static int
111idmap_gridmap_parse_line(const char *line, char **dn_out, char **user_out)
112{
113 const char *p = line;
114 const char *dn_start, *dn_end, *u_start, *u_end;
115
116 if (!idmap_scan_quoted_dn(&p, &dn_start, &dn_end)
117 || !idmap_scan_username(&p, &u_start, &u_end)
118 || (size_t) (dn_end - dn_start) >= IDMAP_PRINC_MAX)
119 {
120 return 0; /* unusable line -> skip */
121 }
122
123 *dn_out = strndup(dn_start, (size_t) (dn_end - dn_start));
124 *user_out = strndup(u_start, (size_t) (u_end - u_start));
125 if (*dn_out == NULL || *user_out == NULL) {
126 free(*dn_out);
127 free(*user_out);
128 return -1;
129 }
130 return 1;
131}

133/* Load the grid-mapfile at `path` into idmap_gridmap[]. Returns NGX_OK (incl.
134 * the "no path" case) or NGX_ERROR on IO/OOM. */
135ngx_int_t
136idmap_gridmap_load(const char *path, ngx_log_t *log)
137{
138 FILE *fp;
139 char line[1024];
140 size_t cap = 0;
141
142 idmap_gridmap_free();
143 if (path == NULL || path[0] == '\0') {
144 return NGX_OK; /* no mapfile configured */
145 }
146
147 fp = fopen(path, "re");
148 if (fp == NULL) {
149 if (log != NULL) {
150 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno,
151 "impersonate: cannot open grid-mapfile \"%s\"", path);
152 }
153 return NGX_ERROR;
154 }
155
156 while (fgets(line, sizeof(line), fp) != NULL) {
157 char *dn = NULL, *user = NULL;
158 int r = idmap_gridmap_parse_line(line, &dn, &user);
159
160 if (r < 0) {
161 /* phase72-fp: read-only stream — fclose result carries no data loss */
162 (void) fclose(fp);
163 idmap_gridmap_free();
164 return NGX_ERROR;
165 }
166 if (r == 0) {
167 continue;
168 }
169 if (idmap_gridmap_n == cap) {
170 size_t ncap = cap ? cap * 2 : 16;
171 idmap_gridmap_entry_t *ne =
172 realloc(idmap_gridmap, ncap * sizeof(*ne));
173 if (ne == NULL) {
174 free(dn); free(user);
175 /* phase72-fp: read-only stream — fclose result carries no data loss */
176 (void) fclose(fp);
177 idmap_gridmap_free();
178 return NGX_ERROR;
179 }
180 idmap_gridmap = ne;
181 cap = ncap;
182 }
183 idmap_gridmap[idmap_gridmap_n].dn = dn;
184 idmap_gridmap[idmap_gridmap_n].user = user;
185 idmap_gridmap_n++;
186 }
187 /* phase72-fp: read-only stream — fclose result carries no data loss */
188 (void) fclose(fp);
189
190 if (log != NULL) {
191 ngx_log_error(NGX_LOG_NOTICE, log, 0,
192 "impersonate: loaded %uz grid-mapfile entries from \"%s\"",
193 idmap_gridmap_n, path);
194 }
195 return NGX_OK;
196}

198/* Look up a DN in the grid-mapfile; returns the local username or NULL. */
199const char *
200idmap_gridmap_lookup(const char *dn)
201{
202 size_t i;
203
204 for (i = 0; i < idmap_gridmap_n; i++) {
205 if (strcmp(idmap_gridmap[i].dn, dn) == 0) {
206 return idmap_gridmap[i].user;
207 }
208 }
209 return NULL;
