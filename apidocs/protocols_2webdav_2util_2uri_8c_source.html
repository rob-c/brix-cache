1/*
2 * uri.c — WebDAV URI decoding and path extraction utilities.
3 *
4 * WHAT: Provides two helper functions for processing HTTP URIs in WebDAV requests:
5 * (1) `webdav_urldecode()` — percent-decodes a URI component into a destination
6 * buffer, rejecting embedded null bytes. (2)
7 * `webdav_destination_extract_path()` — strips the scheme://authority prefix
8 * from a WebDAV Destination header value to isolate the path component.
9 *
10 * WHY: WebDAV clients send URIs in percent-encoded form (e.g. `/data/my%20file.root`).
11 * The Destination header for COPY/MOVE may be an absolute URI (`davs://host/path`)
12 * or a path-only form (`/path`). Both forms must be handled per RFC 4918 §8.3.
13 * Null bytes in URIs are forbidden by HTTP semantics and indicate malformed input.
14 *
15 * HOW:
16 * webdav_urldecode():
17 * Wraps `brix_http_urldecode()` from src/compat/uri.h with a WebDAV-specific
18 * error mapping. The compat function performs the actual percent-decoding; this
19 * wrapper converts its return codes into nginx status codes:
20 * BRIX_URLDECODE_OK → NGX_OK
21 * BRIX_URLDECODE_OVERFLOW → NGX_HTTP_REQUEST_URI_TOO_LARGE (414)
22 * BRIX_URLDECODE_NUL_BYTE → NGX_HTTP_BAD_REQUEST (400)
23 * default → NGX_HTTP_INTERNAL_SERVER_ERROR (500)
24 * The compat function is called with `BRIX_URLDECODE_REJECT_NUL` so embedded
25 * null bytes produce a 400 response rather than truncating the decoded string.
26 *
27 * webdav_destination_extract_path():
28 * Scans the Destination header value for a `scheme://authority` prefix by locating
29 * the first ':' and verifying that positions +1/+2 are "//". If found, it advances
30 * past the authority portion (host+port) to the first '/' in the path. If no scheme
31 * is present, the pointer starts at the beginning of the input buffer. Returns 400
32 * if the extracted path would be empty (p >= end). On success, *path_out points
33 * into the original dest_data buffer and *path_len_out gives the remaining length —
34 * no allocation required, zero-copy extraction.
35 *
36 * DEPENDENCIES: src/compat/uri.h (brix_http_urldecode), RFC 4918 §8.3 (Destination header)
37 * SEE ALSO: webdav/copy.c, webdav/move.c (callers), compat/uri.c (urldecode implementation)
38 */

43ngx_int_t
44webdav_urldecode(const u_char *src, size_t src_len, char *dst, size_t dst_sz)
45{
46 switch (brix_http_urldecode(src, src_len, dst, dst_sz,
47 BRIX_URLDECODE_REJECT_NUL))
48 {
49 case BRIX_URLDECODE_OK: return NGX_OK;
50 case BRIX_URLDECODE_OVERFLOW: return NGX_HTTP_REQUEST_URI_TOO_LARGE;
51 case BRIX_URLDECODE_NUL_BYTE: return NGX_HTTP_BAD_REQUEST;
52 default: return NGX_HTTP_INTERNAL_SERVER_ERROR;
53 }
54}

56ngx_int_t
57webdav_destination_extract_path(const u_char *dest_data, size_t dest_len,
58 const u_char **path_out, size_t *path_len_out)
59{
60 const u_char *p = dest_data;
61 const u_char *end = p + dest_len;
62 const u_char *scheme_end;
63
64 /* Strip scheme://authority if present (RFC 4918 §8.3 absolute-URI form). */
65 scheme_end = ngx_strlchr((u_char *) p, (u_char *) end, ':');
66 if (scheme_end != NULL && scheme_end + 2 < end
67 && scheme_end[1] == '/' && scheme_end[2] == '/')
68 {
69 p = scheme_end + 3;
70 while (p < end && *p != '/') {
71 p++;
72 }
73 }
74
75 if (p >= end) {
76 return NGX_HTTP_BAD_REQUEST;
77 }
78
79 *path_out = p;
80 *path_len_out = (size_t) (end - p);
81 return NGX_OK;
82}
