50/*
51 * imp_op_ctx_t — one privileged fs-op invocation, bundled. WHAT: the full
52 * argument set of a broker fs op plus the once-computed root-relative path,
53 * so imp_do_op() and every per-op helper take a single explicit parameter.
54 * WHY: explicit data flow with no globals while keeping each imp_op_<name>()
55 * under the complexity gate. HOW: the caller (imp_serve_one) fills every
56 * field except `rel`, which imp_do_op() derives from req->path.
57 */
58typedef struct {
59 int rootfd; /* export-root O_PATH fd (confinement anchor) */

64 size_t data_max; /* capacity of data_out */
65 const char *data_in; /* SETXATTR inbound value payload */
66 size_t data_in_len; /* byte count of data_in */
67 const char *rel; /* imp_rel(req->path): root-relative path */
68} imp_op_ctx_t;
