1/*
2 * pblock_nearline.h — Phase-83 F4 nearline/tape simulation for the pblock driver.
3 *
4 * WHAT: A driver-owned `nearline(path, res)` catalog table modelling tape
5 * residency (the brix_sd_residency_t states) plus a bounded synchronous
6 * recall — the same contract sd_frm implements over a real MSS adapter,
7 * here as pure simulation: recall latency and failure are runtime `ctl`
8 * rules, no MSS process anywhere.
9 *
10 * WHY: Exercises the dormant nearline lane natively — the `residency`/`recall`
11 * vtable slots, `BRIX_SD_CAP_NEARLINE`, the cache tier's recall-at-fill,
12 * kXR offline/stall classification — with deterministic timing tests
13 * (today only reachable with a scripted frm://exec MSS).
14 *
15 * HOW: Absent row ⇒ ONLINE (the default for everything ever written); tests
16 * demote files by inserting rows via the sqlite3 CLI (the standard ctl
17 * channel). `ctl` knobs: `nearline.recall_ms` (simulated recall latency,
18 * capped) and `nearline.fail.<path>` (that path's recall lands LOST).
19 * Recall is synchronous like sd_frm's (nothing consumes NGX_AGAIN parking
20 * yet — the stage-engine RECALL integration is the deferred async half):
21 * sleep the latency, then flip the row ONLINE (delete) or LOST. Armed by
22 * `nearline=1` in the `?tail` static opts; off ⇒ no table consulted,
23 * byte-for-byte production driver. ngx-free; BRIX_HAVE_SQLITE-gated.
24 *
25 * Requires: pblock_store.h (pblock_state_t) + fs/backend/sd.h
26 * (brix_sd_residency_t) before inclusion.
27 */

40/* Bounded synchronous recall of `path`: ONLINE ⇒ 0 immediately; LOST ⇒
41 * -1/errno=ENOENT; NEARLINE/OFFLINE ⇒ sleep the simulated latency
42 * (`ctl:nearline.recall_ms`, capped at 30s), then either flip the row ONLINE
43 * and return 0, or — when `ctl:nearline.fail.<path>` is set — mark it LOST and
44 * return -1/errno=EIO (the recall failed; the object is gone). */
45int pblock_nearline_recall(const pblock_state_t *st, const char *path);
