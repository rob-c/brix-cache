2 * tpc_copy.c - HTTP-TPC COPY request parsing, credential delegation, and the
3 * per-user proxy/bearer forwarding helpers for the WebDAV module.
4 *
5 * WHAT: Owns the request-level parsing and credential decisions the COPY
6 * dispatcher makes before any data moves: pull-xor-push header validation,
7 * X-Number-Of-Streams / Overwrite parsing, Source-URL validation, the
8 * OAuth2/OIDC Credential token-exchange delegation for the pull source, the
9 * opportunistic user-bearer forwarding, and the per-user x509 proxy
10 * resolution into the pull context.
11 * WHY: Split from tpc.c (which was over the 500-line cap) so the security-load-
12 * bearing credential/authorization decisions are grouped and individually
13 * reviewable, separate from the dispatcher and the staged pull execution.
14 * HOW: Each entry point declared in tpc_internal_split.h is called by the COPY
15 * dispatcher (ngx_http_brix_webdav_tpc_handle_copy in tpc.c) in the same
16 * order as before. The shared request helpers (subject-token extract, bearer
17 * append) live in tpc.c and are reached through the split header. No
18 * behaviour change: the delegation order (Credential parse -> subject-token
19 * -> token-exchange -> inject; then opportunistic user-bearer; then per-user
20 * proxy) and every metric slot are preserved exactly.
21 */

43/* OAuth2/OIDC credential delegation for an HTTP-TPC pull: parse the Credential /
44 * Credentials request header and, unless it is absent or "none", obtain a
45 * delegated token for source_url and inject it as an "Authorization: Bearer"
46 * entry into transfer_headers (consumed by the curl subprocess). Returns NGX_OK
47 * to continue (whether or not delegation happened), or an NGX_HTTP_* status the
48 * caller must return on a parse/obtain/alloc failure. */
49ngx_int_t
50webdav_tpc_apply_credential_delegation(ngx_http_request_t *r,
51 ngx_http_brix_webdav_loc_conf_t *conf, const char *source_url,
52 ngx_array_t *transfer_headers)
53{
54 ngx_table_elt_t *credential_hdr;
55 brix_tpc_cred_mode_e mode;
56 ngx_str_t delegated_token;
57 ngx_table_elt_t *auth_hdr;
58 const char *subject_token = NULL;
59 ngx_int_t rc;
60
61 credential_hdr = webdav_tpc_find_header(r, "Credential",
62 sizeof("Credential") - 1);
63 if (credential_hdr == NULL) {
64 credential_hdr = webdav_tpc_find_header(r, "Credentials",
65 sizeof("Credentials") - 1);
66 }
67 if (credential_hdr == NULL
68 || webdav_tpc_header_value_equals(&credential_hdr->value, "none"))
69 {
70 return NGX_OK;
71 }
72
73 mode = webdav_tpc_cred_parse_mode(
74 (const char *) credential_hdr->value.data,
75 credential_hdr->value.len);
76
77 if (mode == BRIX_TPC_CRED_UNKNOWN) {
78 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
79 "brix_webdav: unsupported HTTP-TPC credential "
80 "delegation mode \"%V\"", &credential_hdr->value);
81 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_BAD_REQUEST]);
82 return NGX_HTTP_BAD_REQUEST;
83 }
84
85 /* Extract the subject token from the request's Authorization header. */
86 auth_hdr = webdav_tpc_find_header(r, "Authorization",
87 sizeof("Authorization") - 1);
88 rc = webdav_tpc_extract_subject_token(r, auth_hdr, &subject_token);
89 if (rc != NGX_OK) {
90 return NGX_HTTP_INTERNAL_SERVER_ERROR;
91 }
92
93 rc = webdav_tpc_cred_obtain_token(r, mode, source_url,
94 subject_token,
95 conf->tpc_cred.token_scope.len > 0
96 ? (const char *) conf->tpc_cred.token_scope.data
97 : "storage.read",
98 &delegated_token);
99 if (rc != NGX_OK) {
100 return NGX_HTTP_BAD_GATEWAY;
101 }
102
103 /* Inject delegated token as Authorization header. */
104 {
105 size_t total_len = sizeof("Authorization: Bearer ") - 1
106 + delegated_token.len;
107 ngx_str_t *dst = ngx_array_push(transfer_headers);
108 if (dst == NULL) {
109 return NGX_HTTP_INTERNAL_SERVER_ERROR;
110 }
111 dst->data = ngx_pnalloc(r->pool, total_len + 1);
112 if (dst->data == NULL) {
113 return NGX_HTTP_INTERNAL_SERVER_ERROR;
114 }
115 ngx_memcpy(dst->data, "Authorization: Bearer ",
116 sizeof("Authorization: Bearer ") - 1);
117 ngx_memcpy(dst->data + sizeof("Authorization: Bearer ") - 1,
118 delegated_token.data, delegated_token.len);
119 dst->len = total_len;
120 dst->data[dst->len] = '\0';
121 }
122
123 return NGX_OK;
124}

140webdav_tpc_parse_stream_count(ngx_http_request_t *r,
141 ngx_http_brix_webdav_loc_conf_t *conf)
142{
143 ngx_table_elt_t *streams_hdr;
144 ngx_uint_t n_streams = 1;
145
146 streams_hdr = webdav_tpc_find_header(r, "X-Number-Of-Streams",
147 sizeof("X-Number-Of-Streams") - 1);
148 if (streams_hdr != NULL && streams_hdr->value.len > 0) {
149 ngx_int_t v = ngx_atoi(streams_hdr->value.data, streams_hdr->value.len);
150 if (v > 1) {
151 n_streams = (ngx_uint_t) v;
152 }
153 }
154 if (n_streams > conf->tpc_max_streams && conf->tpc_max_streams > 0) {
155 n_streams = conf->tpc_max_streams;
156 }
157 return n_streams;

201 return NGX_HTTP_BAD_REQUEST;

204/*
205 * webdav_tpc_transfer_headers_have_authorization — does the outbound header set
206 * already carry an Authorization entry?
207 *
208 * WHAT: scans the collected "Name: value" transfer-header strings for one whose
209 * name (up to the ':') is "Authorization", case-insensitively.
210 * WHY: the opportunistic default bearer-forward must never override a bearer the
211 * client explicitly delegated — an explicit TransferHeaderAuthorization, or
212 * a Credential-mode token-exchange result, already sits in this array and
213 * must win over the requesting user's ambient token.
214 * HOW: matches the fixed prefix "authorization:"; returns 1 on the first hit,
215 * 0 when none is present (or the array is empty/NULL).
216 */
217static int
218webdav_tpc_transfer_headers_have_authorization(const ngx_array_t *headers)
219{
220 static const u_char name[] = "authorization";
221 const size_t name_len = sizeof(name) - 1;
222 const ngx_str_t *elts;
223 ngx_uint_t i;
224
225 if (headers == NULL) {
226 return 0;
227 }
228
229 elts = headers->elts;
230 for (i = 0; i < headers->nelts; i++) {
231 if (elts[i].len > name_len
232 && elts[i].data[name_len] == ':'
233 && ngx_strncasecmp(elts[i].data, (u_char *) name, name_len) == 0)
234 {
235 return 1;
236 }
237 }
238 return 0;
239}

241/*
242 * webdav_tpc_forward_user_bearer — opportunistic default: present the requesting
243 * user's own captured bearer token to the pull source.
244 *
245 * WHAT: when credential forwarding is enabled (the default) and the outbound
246 * header set does not already carry an Authorization entry, appends the raw
247 * JWT the request authenticated with (rctx->bearer_token) as an
248 * "Authorization: Bearer <token>" transfer header.
249 * WHY: a TPC pull should act as the END USER against the source by default —
250 * this is the HTTP equivalent of the native root:// bearer passthrough. It
251 * is OPPORTUNISTIC: the absence of a captured token is not an error (the leg
252 * falls back to the service x509 cert / anonymous exactly as before), and a
253 * client-supplied Authorization is never overridden.
254 * HOW: no-ops (NGX_OK) when the toggle is off, no rctx/token is present, or an
255 * Authorization header already exists; otherwise reuses
256 * webdav_tpc_add_bearer_header. Returns its NGX_HTTP_* only on an allocation
257 * failure — never a denial.
258 */
259ngx_int_t
260webdav_tpc_forward_user_bearer(ngx_http_request_t *r,
261 ngx_http_brix_webdav_loc_conf_t *conf, ngx_array_t *transfer_headers)
262{
263 ngx_http_brix_webdav_req_ctx_t *rctx;
264
265 if (!conf->tpc_credential_forward) {
266 return NGX_OK;
267 }
268
269 rctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
270 if (rctx == NULL || rctx->bearer_token.len == 0
271 || rctx->bearer_token.data == NULL)
272 {
273 return NGX_OK;
274 }
275
276 if (webdav_tpc_transfer_headers_have_authorization(transfer_headers)) {
277 return NGX_OK;
278 }
279
280 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
281 "brix_webdav: TPC pull forwarding requesting user's bearer"
282 " token to source (opportunistic default)");
283 return webdav_tpc_add_bearer_header(r, transfer_headers,
284 &rctx->bearer_token);
285}

287/* ---- Resolve the per-user pull-leg x509 proxy into the pull context ----
288 *
289 * WHAT: When brix_webdav_tpc_credential_forward is on, resolves the requesting
290 * user's delegated x509 proxy and records its cert/key paths in
291 * pl->user_cert / pl->user_key. Returns NGX_OK on success (leaving both
292 * fields NULL when no per-user proxy is available, so the leg falls back to
293 * the static service cert). Returns NGX_HTTP_INTERNAL_SERVER_ERROR only when
294 * an explicitly delegated proxy could not be materialised (up.deny).
295 *
296 * WHY: Presents the END USER's identity to the source (not our service cert) as
297 * the default, independent of brix_backend_delegation which governs the
298 * data-plane backend leg. Kept OPPORTUNISTIC: absence of a per-user proxy
299 * leaves the fields NULL and downgrades to the service cert exactly as
300 * before; only a proxy we were told to use but could not build aborts, so we
301 * never silently transfer under the wrong identity. Forwarding off leaves
302 * the fields untouched (service-cert only).
303 *
304 * HOW: 1. No-op (NGX_OK) when conf->tpc_credential_forward is off — pl fields keep
305 * their memzero'd NULLs.
306 * 2. Resolve the user proxy via webdav_tpc_user_proxy_resolve.
307 * 3. On up.deny, count a bad-request metric and return 500.
308 * 4. Otherwise copy the resolved cert/key paths into pl when up.have, else
309 * leave them NULL.
310 */
311ngx_int_t
312webdav_tpc_apply_user_proxy(ngx_http_request_t *r,
313 ngx_http_brix_webdav_loc_conf_t *conf, webdav_tpc_pull_ctx_t *pl)
314{
315 webdav_tpc_user_proxy_t up;
316
317 if (!conf->tpc_credential_forward) {
318 return NGX_OK;
319 }
320
321 webdav_tpc_user_proxy_resolve(r, conf, &up);
322 if (up.deny) {
323 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_BAD_REQUEST]);
324 return NGX_HTTP_INTERNAL_SERVER_ERROR;
325 }
326
327 pl->user_cert = up.have ? up.cert_path : NULL;
328 pl->user_key = up.have ? up.key_path : NULL;
329 return NGX_OK;
330}
