1/*
2 * checksum_core.h — pure (ngx-free) whole-file checksum compute kernels.
3 *
4 * WHAT: Stream a file descriptor and produce a 32-bit checksum (Adler-32/CRC-32/
5 * CRC-32c) or a cryptographic digest (MD5/SHA-1/SHA-256). No nginx, no
6 * logging — the caller maps its own algorithm enum to the kind codes here
7 * and handles errors/logging at the edge.
8 * WHY: Both the nginx module (src/compat/checksum.c) and the native client
9 * (client/lib/checksum.c) need identical fd→checksum compute; this is the
10 * single source they share via libxrdproto (build-in-place), like crc32c
11 * and gsi_core.
12 * HOW: pread(2) the whole file in 64 KiB chunks; zlib for adler32/crc32,
13 * libxrdproto's brix_crc32c_extend for crc32c, OpenSSL EVP for digests.
14 *
15 * The kind codes match brix_checksum_alg_t's ordinals so the module passes its
16 * enum value directly; the client maps its own (smaller) enum onto them.
17 */
18#ifndef BRIX_CHECKSUM_CORE_H
19#define BRIX_CHECKSUM_CORE_H

36/* Whole-OBJECT (driver pread from offset 0) checksums — the canonical entry for
37 * a backend-bound handle (block-striped/object store): every byte is read
38 * through obj->driver, so a multi-block file is summed in full (not just block
39 * 0). obj->driver must be non-NULL. The _fd wrappers below POSIX-wrap a bare fd
40 * onto this same kernel for the default export. Return 0 / -1. */
41int brix_cksum_u32_obj(int kind, brix_sd_obj_t *obj, uint32_t *out);
42int brix_cksum_u64_obj(int kind, brix_sd_obj_t *obj, uint64_t *out);
43int brix_cksum_digest_obj(int kind, brix_sd_obj_t *obj, unsigned char *out,
44 unsigned int *outlen);
45
46/* Whole-file (pread from offset 0) 32-bit checksum for ADLER32/CRC32/CRC32C into
47 * *out. Returns 0 / -1 (errno set on a read failure). */
48int brix_cksum_u32_fd(int kind, int fd, uint32_t *out);
49
50/* Whole-file (pread from offset 0) 64-bit checksum for CRC64/CRC64NVME into *out.
51 * Returns 0 / -1 (errno set on a read failure; -1 also for an unknown kind). */
52int brix_cksum_u64_fd(int kind, int fd, uint64_t *out);
53
54/* Whole-file digest for MD5/SHA1/SHA256 into out[<=EVP_MAX_MD_SIZE], length in
55 * *outlen. Returns 0 / -1 (errno set on a read failure). */
56int brix_cksum_digest_fd(int kind, int fd, unsigned char *out,
57 unsigned int *outlen);

59#endif /* BRIX_CHECKSUM_CORE_H */
