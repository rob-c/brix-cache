30brix_px_kind_t
31brix_px_classify(X509 *cert)
32{
33 PROXY_CERT_INFO_EXTENSION *pci;
34 brix_px_kind_t kind = BRIX_PX_NONE;
35
36 if (X509_get_extension_flags(cert) & EXFLAG_PROXY) {
37 kind = BRIX_PX_FULL;
38 pci = X509_get_ext_d2i(cert, NID_proxyCertInfo, NULL, NULL);
39 if (pci != NULL) {
40 char oid[128];
41 int n = OBJ_obj2txt(oid, sizeof(oid),
42 pci->proxyPolicy->policyLanguage, 1);
43 if (n > 0 && strcmp(oid, BRIX_PX_LIMITED_OID) == 0) {
44 kind = BRIX_PX_LIMITED;
45 }
46 PROXY_CERT_INFO_EXTENSION_free(pci);
47 }
48 return kind;
49 }
50
51 /* Legacy Globus proxy: last RDN is CN=proxy or CN=limited proxy. */
52 {
53 X509_NAME *nm = X509_get_subject_name(cert);
54 int last = X509_NAME_entry_count(nm) - 1;
55 if (last >= 0) {
56 X509_NAME_ENTRY *e = X509_NAME_get_entry(nm, last);
57 ASN1_STRING *v = X509_NAME_ENTRY_get_data(e);
58 const unsigned char *s = ASN1_STRING_get0_data(v);
59 int len = ASN1_STRING_length(v);
60 if (len == (int) sizeof("limited proxy") - 1
61 && strncasecmp((const char *) s, "limited proxy", len) == 0)
62 {
63 return BRIX_PX_LIMITED;
64 }
65 if (len == (int) sizeof("proxy") - 1
66 && strncasecmp((const char *) s, "proxy", len) == 0)
67 {
68 return BRIX_PX_FULL;
69 }
70 }
71 }
72 return BRIX_PX_NONE;
73}

75int
76brix_proxy_chain_ok(STACK_OF(X509) *chain)
77{
78 int n, i;
79 int seen_limited = 0;
80
81 if (chain == NULL) {
82 return 1;
83 }
84
85 n = sk_X509_num(chain);
86 for (i = n - 1; i >= 0; i--) { /* root .. leaf */
87 brix_px_kind_t kind = brix_px_classify(sk_X509_value(chain, i));
88
89 if (kind == BRIX_PX_LIMITED) {
90 seen_limited = 1;
91 } else if (kind == BRIX_PX_FULL && seen_limited) {
92 return 0; /* full proxy beneath a limited one — escalation */
93 }
94 }
95 return 1;
96}

105static int
106brix_dn_has_control_bytes(X509_NAME *nm)
107{
108 int i, n = X509_NAME_entry_count(nm);
109 for (i = 0; i < n; i++) {
110 X509_NAME_ENTRY *e = X509_NAME_get_entry(nm, i);
111 ASN1_STRING *v = X509_NAME_ENTRY_get_data(e);
112 const unsigned char *s = ASN1_STRING_get0_data(v);
113 int len = ASN1_STRING_length(v);
114 int j;
115 for (j = 0; j < len; j++) {
116 if (s[j] < 0x20 || s[j] == 0x7f) {
117 return 1; /* embedded control or NUL byte */
118 }
119 }
120 }
121 return 0;
122}

124/*
125 * WHAT: reject a public key weaker than the IGTF floor (RSA/DSA >= 2048 bits,
126 * EC >= 256 bits).
127 * WHY: weak keys are trivially forgeable; the floor is the same one XRootD and
128 * the IGTF profile enforce. Unknown key types carry no verdict (0).
129 * HOW: read the key's base id + bit count once and apply the per-family floor;
130 * a cert with no extractable key is not judged here (0).
131 */
132static int
133brix_key_strength_violation(X509 *cert)
134{
135 EVP_PKEY *pk = X509_get0_pubkey(cert);
136 int id;
137 int bits;
138
139 if (pk == NULL) {
140 return 0;
141 }
142 id = EVP_PKEY_base_id(pk);
143 bits = EVP_PKEY_bits(pk);
144 if ((id == EVP_PKEY_RSA || id == EVP_PKEY_RSA2 || id == EVP_PKEY_DSA)
145 && bits < 2048) {
146 return 1;
147 }
148 if (id == EVP_PKEY_EC && bits < 256) {
149 return 1;
150 }
151 return 0;
152}
153
154/*
155 * WHAT: reject a certificate signed with a broken message digest (MD2/MD4/MD5
156 * or SHA-1).
157 * WHY: collision attacks make these digests unsafe for signatures; XRootD's
158 * hardened profile refuses them too. A signature algorithm we cannot
159 * decompose carries no verdict (0).
160 * HOW: map the signature OID to its digest NID and compare against the
161 * forbidden set.
162 */
163static int
164brix_sig_alg_violation(X509 *cert)
165{
166 int sig_nid = X509_get_signature_nid(cert);
167 int md_nid = NID_undef, pk_nid = NID_undef;
168
169 if (!OBJ_find_sigid_algs(sig_nid, &md_nid, &pk_nid)) {
170 return 0;
171 }
172 if (md_nid == NID_md5 || md_nid == NID_sha1 || md_nid == NID_md2
173 || md_nid == NID_md4) {
174 return 1;
175 }
176 return 0;
177}
178
179/*
180 * WHAT: reject a serial number that is zero, negative, or (for a non-proxy)
181 * longer than 20 octets (RFC 5280 §4.1.2.2).
182 * WHY: a non-positive serial is malformed; the 20-octet ceiling is the RFC
183 * cap. Proxies are exempt from the ceiling because grid proxies derive
184 * large serials. A serial we cannot parse carries no verdict (0).
185 * HOW: convert the serial to a BIGNUM, test the three conditions, free.
186 */
187static int
188brix_serial_violation(X509 *cert, int is_proxy)
189{
190 const ASN1_INTEGER *sn = X509_get0_serialNumber(cert);
191 BIGNUM *bn = ASN1_INTEGER_to_BN(sn, NULL);
192 int bad;
193
194 if (bn == NULL) {
195 return 0;
196 }
197 bad = BN_is_zero(bn) || BN_is_negative(bn)
198 || (!is_proxy && BN_num_bytes(bn) > 20);
199 BN_free(bn);
200 return bad ? 1 : 0;
201}

203int
204brix_cert_policy_violation(X509 *cert)
205{
206 int is_proxy = (X509_get_extension_flags(cert) & EXFLAG_PROXY) != 0;
207
208 /* Key strength (IGTF: RSA >= 2048, EC >= 256). */
209 if (brix_key_strength_violation(cert)) {
210 return 1;
211 }
212
213 /* Signature algorithm: reject MD5 and SHA-1 based signatures. */
214 if (brix_sig_alg_violation(cert)) {
215 return 1;
216 }
217
218 /* Serial number: positive and <= 20 octets (RFC 5280 §4.1.2.2). Proxies
219 * are exempt from the ceiling — grid proxies use large derived serials. */
220 if (brix_serial_violation(cert, is_proxy)) {
221 return 1;
222 }
223
224 /* DN sanity: no embedded control/NUL bytes (RFC 5280 §4.1.2.6). */
225 if (brix_dn_has_control_bytes(X509_get_subject_name(cert))
226 || brix_dn_has_control_bytes(X509_get_issuer_name(cert))) {
227 return 1;
228 }
229
230 return 0;
231}

233int
234brix_leaf_purpose_violation(X509 *leaf)
235{
236 /* extendedKeyUsage: if present, must allow client auth. */
237 EXTENDED_KEY_USAGE *eku = X509_get_ext_d2i(leaf, NID_ext_key_usage, NULL,
238 NULL);
239 if (eku != NULL) {
240 int i, ok = 0;
241 for (i = 0; i < sk_ASN1_OBJECT_num(eku); i++) {
242 int nid = OBJ_obj2nid(sk_ASN1_OBJECT_value(eku, i));
243 if (nid == NID_client_auth || nid == NID_anyExtendedKeyUsage) {
244 ok = 1;
245 break;
246 }
247 }
248 sk_ASN1_OBJECT_pop_free(eku, ASN1_OBJECT_free);
249 if (!ok) {
250 return 1;
251 }
252 }
253
254 /* keyUsage: if present, must assert digitalSignature (RSA/ECDSA client
255 * auth) OR keyAgreement (fixed-ECDH client auth). */
256 {
257 uint32_t flags = X509_get_key_usage(leaf); /* UINT32_MAX if absent */
258 if (flags != UINT32_MAX
259 && !(flags & (KU_DIGITAL_SIGNATURE | KU_KEY_AGREEMENT))) {
260 return 1;
261 }
262 }
263 return 0;
264}

266int
267brix_proxy_pci_ok(STACK_OF(X509) *chain)
268{
269 int n, i;
270
271 if (chain == NULL) {
272 return 1;
273 }
274 n = sk_X509_num(chain);
275 for (i = 0; i < n; i++) {
276 X509 *cert = sk_X509_value(chain, i);
277 int loc = X509_get_ext_by_NID(cert, NID_proxyCertInfo, -1);
278 X509_EXTENSION *ext;
279 PROXY_CERT_INFO_EXTENSION *pci;
280 char oid[128];
281 int ok_oid;
282
283 if (loc < 0) {
284 continue; /* no proxyCertInfo — legacy/EEC handled elsewhere */
285 }
286 ext = X509_get_ext(cert, loc);
287 if (!X509_EXTENSION_get_critical(ext)) {
288 return 0; /* proxyCertInfo present but not critical (RFC 3820 §3.1) */
289 }
290 pci = X509_get_ext_d2i(cert, NID_proxyCertInfo, NULL, NULL);
291 if (pci == NULL) {
292 return 0; /* malformed proxyCertInfo */
293 }
294 oid[0] = '\0';
295 OBJ_obj2txt(oid, sizeof(oid), pci->proxyPolicy->policyLanguage, 1);
296 PROXY_CERT_INFO_EXTENSION_free(pci);
297 ok_oid = (strcmp(oid, BRIX_PX_OID_IMPERSONATION) == 0)
298 || (strcmp(oid, BRIX_PX_OID_INDEPENDENT) == 0)
299 || (strcmp(oid, BRIX_PX_LIMITED_OID) == 0);
300 if (!ok_oid) {
301 return 0; /* unrecognised proxy policy language (RFC 3820 §3.2) */
302 }
303 }
304 return 1;
305}
