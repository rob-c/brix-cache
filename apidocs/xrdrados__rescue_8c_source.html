1/*
2 * xrdrados_rescue.c — offline pure-RADOS recovery tool (operator utility).
3 *
4 * WHAT: Enumerates and extracts objects from a flat RADOS pool — the block-only
5 * `ceph` backend's storage, or any pool of opaque objects — with no
6 * namespace service. It lists object keys, stats them, reads their bytes and
7 * xattrs, and bulk-extracts objects under a key prefix to local files. For
8 * pools that hold the `ceph` driver's data (object key == logical path) this
9 * recovers the files directly.
10 *
11 * It is the flat-pool counterpart to xrdcephfs_rescue (which understands the
12 * CephFS layout). Both reuse sd_ceph's connection + oid layer.
13 *
14 * USAGE:
15 * xrdrados_rescue <pool> ls [prefix]
16 * xrdrados_rescue <pool> stat <key>
17 * xrdrados_rescue <pool> get <key> <local_file>
18 * xrdrados_rescue <pool> cp <prefix> <local_dir> (extract all under prefix)
19 * (env CEPH_CONF overrides /etc/ceph/ceph.conf)
20 *
21 * BUILD: `make -C client ceph-tools` (dep-gated), or by hand where
22 * librados-devel exists:
23 * gcc -DXRDPROTO_NO_NGX -DBRIX_HAVE_CEPH -I src/fs/backend -I src/fs/backend/rados \
24 * -include client/apps/ceph/ngx_shim.h client/apps/ceph/xrdrados_rescue.c \
25 * src/fs/backend/rados/sd_ceph.c src/fs/backend/rados/sd_ceph_compat.c \
26 * -lrados -o xrdrados_rescue
27 */

43/* Stream one object's bytes (by oid) to a FILE* in 1 MiB reads. 0 / -1. */
44static int
45extract_object(sd_ceph_conn_t *c, const char *oid, FILE *out)
46{
47 char buf[1u << 20];
48 off_t off = 0;
49 ssize_t n;
50
51 while ((n = sd_ceph_oid_read(c, oid, buf, sizeof(buf), off)) > 0) {
52 if (fwrite(buf, 1, (size_t) n, out) != (size_t) n) { return -1; }
53 off += n;
54 if ((size_t) n < sizeof(buf)) { break; } /* short read ⇒ end of object */
55 }
56 return (n < 0) ? -1 : 0;
57}

59/* Iterate every object in the pool, invoking cb(key, ctx) for those matching the
60 * (optional) prefix. Returns 0, or -1/errno on a RADOS error. */
61typedef int (*obj_cb)(const char *key, void *ctx);
62static int
63for_each_object(sd_ceph_conn_t *c, const char *prefix, obj_cb cb, void *ctx)
64{
65 rados_ioctx_t ioctx = sd_ceph_conn_ioctx(c);
66 rados_list_ctx_t lc;
67 size_t plen = (prefix != NULL) ? strlen(prefix) : 0;
68 int rc;
69
70 if (rados_nobjects_list_open(ioctx, &lc) < 0) { return -1; }
71 for (;;) {
72 const char *entry = NULL, *nspace = NULL;
73 size_t klen = 0;
74
75 rc = rados_nobjects_list_next2(lc, &entry, NULL, &nspace, &klen, NULL, NULL);
76 if (rc == -ENOENT) { rc = 0; break; } /* end of listing */
77 if (rc < 0) { break; }
78 if (entry == NULL) { continue; }
79 if (plen == 0 || strncmp(entry, prefix, plen) == 0) {
80 if (cb(entry, ctx) != 0) { rc = -1; break; }
81 }
82 }
83 rados_nobjects_list_close(lc);
84 if (rc < 0 && errno == 0) { errno = EIO; }
85 return rc < 0 ? -1 : 0;
86}

93/* Replace '/' in a key with '_' so a path-like key becomes one local filename
94 * (flat extraction; the operator can re-layout afterwards). */
95static void
96flatten_key(const char *key, char *out, size_t cap)
97{
98 size_t i;
99 for (i = 0; key[i] != '\0' && i + 1 < cap; i++) {
100 out[i] = (key[i] == '/') ? '_' : key[i];
101 }
102 out[i] = '\0';
103}

105static int
106cp_one(const char *key, void *ctx)
107{
108 cp_ctx_t *cc = ctx;
109 char name[1024], path[2048];
110 FILE *out;
111
112 flatten_key(key, name, sizeof(name));
113 snprintf(path, sizeof(path), "%s/%s", cc->dst, name);
114 out = fopen(path, "wb");
115 if (out == NULL) { fprintf(stderr, "create %s: %s\n", path, strerror(errno));
116 cc->fails++; return 0; }
117 if (extract_object(cc->c, key, out) != 0) {
118 fprintf(stderr, "extract %s failed\n", key); cc->fails++;
119 } else {
120 printf(" %s -> %s\n", key, path);
121 }
122 fclose(out);
123 return 0;
124}

126/* ---- List every object key (optionally prefix-filtered) to stdout ----
127 *
128 * WHAT: Runs the `ls [prefix]` subcommand: prints one matching object key per
129 * line. Returns 0 on success, 1 on a RADOS listing error.
130 *
131 * WHY: Splitting the per-command body out of main() keeps each subcommand a
132 * single-purpose unit and holds main()'s branching under the complexity cap.
133 *
134 * HOW:
135 * 1. Pass argv[3] as the prefix when present, else NULL (list-all).
136 * 2. Walk the pool via for_each_object() with the print_key callback.
137 * 3. On error, report strerror(errno) and normalise the status to 1.
138 */
139static int
140cmd_ls(sd_ceph_conn_t *c, int argc, char **argv)
141{
142 int rc = for_each_object(c, (argc > 3) ? argv[3] : NULL, print_key, NULL);
143 if (rc != 0) { fprintf(stderr, "list failed: %s\n", strerror(errno)); rc = 1; }
144 return rc;
145}

147/* ---- Print size + mtime for one object key ----
148 *
149 * WHAT: Runs the `stat <key>` subcommand: prints key/size/mtime. Returns 0 on
150 * success, 2 on missing argument, 1 on a stat failure.
151 *
152 * WHY: Isolates the stat formatting and its argument/error handling from the
153 * other subcommands so each stays independently reviewable.
154 *
155 * HOW:
156 * 1. Require argv[3] (the key); otherwise emit usage and return 2.
157 * 2. Query sd_ceph_oid_stat(); on failure report strerror(errno), return 1.
158 * 3. On success print key, size (unsigned long long), mtime (long long).
159 */
160static int
161cmd_stat(sd_ceph_conn_t *c, int argc, char **argv)
162{
163 uint64_t sz = 0; time_t mt = 0;
164 if (argc < 4) { fprintf(stderr, "stat needs <key>\n"); return 2; }
165 if (sd_ceph_oid_stat(c, argv[3], &sz, &mt) != 0) {
166 fprintf(stderr, "stat %s: %s\n", argv[3], strerror(errno)); return 1;
167 }
168 printf("key: %s\nsize: %llu\nmtime: %lld\n",
169 argv[3], (unsigned long long) sz, (long long) mt);
170 return 0;
171}

173/* ---- Extract one object's bytes to a named local file ----
174 *
175 * WHAT: Runs the `get <key> <local_file>` subcommand: streams the object into
176 * the file. Returns 0 on success, 2 on missing args, 1 on open/read error.
177 *
178 * WHY: Keeps the single-object extraction path (open target, stream, close)
179 * self-contained and off main()'s branch ladder.
180 *
181 * HOW:
182 * 1. Require both argv[3] (key) and argv[4] (local file); else return 2.
183 * 2. Open the target for binary write; on failure report and return 1.
184 * 3. Stream via extract_object(), map success to 0 / failure to 1, close.
185 */
186static int
187cmd_get(sd_ceph_conn_t *c, int argc, char **argv)
188{
189 FILE *out;
190 int rc;
191 if (argc < 5) { fprintf(stderr, "get needs <key> <local_file>\n"); return 2; }
192 out = fopen(argv[4], "wb");
193 if (out == NULL) { fprintf(stderr, "create %s: %s\n", argv[4], strerror(errno)); return 1; }
194 rc = (extract_object(c, argv[3], out) == 0) ? 0 : 1;
195 fclose(out);
196 return rc;
197}

199/* ---- Bulk-extract every object under a prefix into a local directory ----
200 *
201 * WHAT: Runs the `cp <prefix> <local_dir>` subcommand: creates the directory and
202 * writes each matching object to a flattened filename. Returns 0 when all
203 * objects extracted cleanly, 2 on missing args, 1 on mkdir/extraction fail.
204 *
205 * WHY: Groups the destination setup and the per-object copy fan-out into one
206 * unit, preserving the "0 only if every object succeeded" contract.
207 *
208 * HOW:
209 * 1. Require argv[3] (prefix) and argv[4] (local dir); else return 2.
210 * 2. mkdir the destination, tolerating EEXIST; other errors report + return 1.
211 * 3. Walk matching objects via for_each_object()+cp_one accumulating failures.
212 * 4. Return 0 only when the walk succeeded and cc.fails is zero, else 1.
213 */
214static int
215cmd_cp(sd_ceph_conn_t *c, int argc, char **argv)
216{
217 cp_ctx_t cc;
218 int rc;
219 if (argc < 5) { fprintf(stderr, "cp needs <prefix> <local_dir>\n"); return 2; }
220 cc.c = c; cc.dst = argv[4]; cc.fails = 0;
221 if (mkdir(argv[4], 0755) != 0 && errno != EEXIST) {
222 fprintf(stderr, "mkdir %s: %s\n", argv[4], strerror(errno)); return 1;
223 }
224 rc = for_each_object(c, argv[3], cp_one, &cc);
225 return (rc == 0 && cc.fails == 0) ? 0 : 1;
226}

228/* ---- Dispatch a parsed subcommand to its handler ----
229 *
230 * WHAT: Maps the command word to cmd_ls/cmd_stat/cmd_get/cmd_cp and returns that
231 * handler's status; returns 2 for an unknown command.
232 *
233 * WHY: Keeps the command-name branch ladder in one small function so main() is a
234 * flat connect → dispatch → destroy sequence within the complexity cap.
235 *
236 * HOW:
237 * 1. Compare cmd against each known verb in turn.
238 * 2. Delegate to the matching handler, forwarding argc/argv unchanged.
239 * 3. On no match, report the unknown command and return 2.
240 */
241static int
242run_command(sd_ceph_conn_t *c, const char *cmd, int argc, char **argv)
243{
244 if (strcmp(cmd, "ls") == 0) { return cmd_ls(c, argc, argv); }
245 if (strcmp(cmd, "stat") == 0) { return cmd_stat(c, argc, argv); }
246 if (strcmp(cmd, "get") == 0) { return cmd_get(c, argc, argv); }
247 if (strcmp(cmd, "cp") == 0) { return cmd_cp(c, argc, argv); }
248 fprintf(stderr, "unknown command '%s'\n", cmd);
249 return 2;
250}
