13const brix_http_operation_t *
14brix_http_operation_find(ngx_http_request_t *r,
15 const brix_http_operation_t *ops, ngx_uint_t nops)
16{
17 ngx_uint_t i;
18
19 for (i = 0; i < nops; i++) {
20 if (ops[i].http_method != 0 && ops[i].http_method == r->method) {
21 return &ops[i];
22 }
23
24 if (r->method_name.len == ngx_strlen(ops[i].name)
25 && ngx_strncmp(r->method_name.data, ops[i].name,
26 r->method_name.len) == 0)
27 {
28 return &ops[i];
29 }
30 }
31 return NULL;
32}

34ngx_int_t
35brix_http_operation_allow_header(ngx_pool_t *pool,
36 const brix_http_operation_t *ops, ngx_uint_t nops,
37 ngx_uint_t enabled_flags, ngx_str_t *out)
38{
39 size_t total = 0;
40 ngx_uint_t i;
41 u_char *p;
42 int first;
43
44 /* First pass: compute total length of the comma-separated list. */
45 for (i = 0; i < nops; i++) {
46 if (ops[i].flags & enabled_flags) {
47 total += ngx_strlen(ops[i].name) + 2; /* ", " separator */
48 }
49 }
50
51 if (total == 0) {
52 out->len = 0;
53 out->data = (u_char *) "";
54 return NGX_OK;
55 }
56
57 /* Second pass: write the string. */
58 BRIX_PALLOC_OR_RETURN(p, pool, total + 1, NGX_ERROR);
59
60 out->data = p;
61 first = 1;
62
63 for (i = 0; i < nops; i++) {
64 if (ops[i].flags & enabled_flags) {
65 size_t nlen = ngx_strlen(ops[i].name);
66 if (!first) {
67 *p++ = ',';
68 *p++ = ' ';
69 }
70 ngx_memcpy(p, ops[i].name, nlen);
71 p += nlen;
72 first = 0;
73 }
74 }
75 *p = '\0';
76
77 out->len = (size_t)(p - out->data);
78 return NGX_OK;
79}
