24/* exec functions (one per op) */
25static ngx_int_t
26exec_chmod(const brix_op_exec_t *e, int *out_errno)
27{
28 xrdw_chmod_req_t req;
29 mode_t mode;
30
31 xrdw_chmod_req_unpack(((ClientRequestHdr *) e->ctx->recv.hdr_buf)->body, &req);
32 mode = req.mode & 0777;
33
34 if (mode == 0) {
35 mode = 0644;
36 }
37 /* brix_vfs_chmod delegates to the impersonation-aware confined chmod, so
38 * under impersonation it is performed BY THE BROKER as the mapped user (the
39 * unprivileged worker is not the owner and a worker-local chmod would EPERM). */
40 {
41 brix_vfs_ctx_t vctx;
42
43 op_vfs_ctx(e, &vctx);
44 if (brix_vfs_chmod(&vctx, mode) == NGX_OK) {
45 return NGX_OK;
46 }
47 }
48 *out_errno = errno;
49 return NGX_ERROR;
50}

52static ngx_int_t
53exec_rm(const brix_op_exec_t *e, int *out_errno)
54{
55 brix_vfs_ctx_t vctx;
56
57 /* kXR_rm mirrors osFS->rem: unlink a file, rmdir a directory — NON-recursively
58 * (brix_vfs_unlink → brix_vfs_delete(recursive=0): an empty dir is removed,
59 * a non-empty dir fails ENOTEMPTY). It must NEVER recurse (recursion here would
60 * silently delete whole subtrees on a plain `rm` — data loss). */
61 op_vfs_ctx(e, &vctx);
62 if (brix_vfs_unlink(&vctx) == NGX_OK) {
63 return NGX_OK;
64 }
65 *out_errno = errno ? errno : EISDIR;
66 return NGX_ERROR;
67}

69static ngx_int_t
70exec_rmdir(const brix_op_exec_t *e, int *out_errno)
71{
72 brix_vfs_ctx_t vctx;
73
74 /* rmdir must reject regular files (brix_vfs_rmdir → vfs_delete with
75 * require_directory) and is idempotent for a missing target (stock do_Rmdir
76 * tolerates ENOENT), so a not-found dir is treated as success. */
77 op_vfs_ctx(e, &vctx);
78 if (brix_vfs_rmdir(&vctx, 0 /* non-recursive */) == NGX_OK) {
79 return NGX_OK;
80 }
81 if (errno == ENOENT) {
82 return NGX_OK; /* idempotent_missing */
83 }
84 *out_errno = errno ? errno : ENOTEMPTY;
85 return NGX_ERROR;
86}

99/* interpreter */
100ngx_int_t
101brix_dispatch_op(brix_ctx_t *ctx, ngx_connection_t *c,
102 ngx_stream_brix_srv_conf_t *conf,
103 uint16_t opcode)
104{
105 char reqpath[BRIX_MAX_PATH + 1];
106 char resolved[PATH_MAX];
107 const brix_op_desc_t *d = NULL;
108 brix_op_exec_t ex;
109 int err = 0;
110 size_t i;
111
112 for (i = 0; i < N_OPS; i++) {
113 if (_ops[i].opcode == opcode) {
114 d = &_ops[i];
115 break;
116 }
117 }
118 if (d == NULL) {
119 return brix_send_error(ctx, c, kXR_InvalidRequest, "unknown opcode");
120 }
121
122 if (brix_resolve_op_path(ctx, c, d->op_id, d->name, conf,
123 d->path_mode,
124 reqpath, sizeof(reqpath),
125 resolved, sizeof(resolved)) != NGX_OK) {
126 return ctx->write_rc;
127 }
128
129 if (brix_auth_gate(ctx, c, d->op_id, d->name,
130 reqpath, resolved, conf,
131 d->auth_level, d->need_write) != NGX_OK) {
132 return ctx->write_rc;
133 }
134
135 ex.ctx = ctx;
136 ex.c = c;
137 ex.conf = conf;
138 ex.reqpath = reqpath;
139 ex.resolved = resolved;
140
141 if (d->exec(&ex, &err) != NGX_OK) {
142 BRIX_RETURN_ERR(ctx, c, d->op_id, d->name, resolved, "-",
143 brix_kxr_from_errno(err),
144 brix_kxr_err_string(err));
145 }
146
147 BRIX_RETURN_OK(ctx, c, d->op_id, d->name, resolved, "-", 0);
148}
