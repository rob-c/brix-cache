1/*
2 * sd_posix.c — the POSIX Storage Driver: a behaviour-preserving wrapper.
3 *
4 * WHAT: Implements brix_sd_posix_driver, the default backend. Every vtable
5 * slot delegates to an EXISTING confined helper (brix_open_beneath,
6 * brix_vfs_pread_full/pwrite_full, brix_ns_*, brix_lstat_beneath,
7 * the *xattr_confined_canon family, brix_staged_*), so the POSIX path
8 * is byte-identical to today's VFS once the VFS is wired to it (55.B+).
9 *
10 * WHY: Cutting the SD seam must not change POSIX behaviour. The safest way to
11 * guarantee that is for the POSIX driver to call the same code the VFS
12 * calls now, rather than re-implement any syscall. Confinement stays the
13 * kernel RESOLVE_BENEATH API; an EXDEV still means an escape attempt.
14 *
15 * HOW: The instance holds a persistent O_PATH rootfd + the root_canon string.
16 * Object handles carry the open fd and a metadata snapshot. The raw byte
17 * ops are worker-safe (no pool/metrics/log). Namespace ops translate the
18 * brix_ns_result_t status to errno via brix_vfs_ns_status_errno().
19 */

45/* Driver-private instance state. */
46typedef struct {
47 int rootfd; /* persistent O_PATH fd on root_canon, or -1 */
48 char *root_canon; /* pool-owned copy (or borrowed when borrowed=1) */
49 unsigned borrowed:1; /* rootfd/root_canon are owned by the caller's conf:
50 * cleanup must NOT close the fd or free the string */
51} sd_posix_state_t;

53/* sd_posix_flags — map the backend-neutral BRIX_SD_O_* bits to an open(2) flag
54 * set (the vtable speaks SD flags; only the POSIX driver knows O_*). Always
55 * compiled: the server's confined open and the client's unconfined open
56 * (brix_sd_posix_open_unconfined) both use it, so the mapping is single-sourced. */
57static int
58sd_posix_flags(int sd_flags)
59{
60 int flags = 0;
61
62 if ((sd_flags & BRIX_SD_O_WRITE) && (sd_flags & BRIX_SD_O_READ)) {
63 flags = O_RDWR;
64 } else if (sd_flags & BRIX_SD_O_WRITE) {
65 flags = O_WRONLY;
66 } else {
67 flags = O_RDONLY;
68 }
69
70 if (sd_flags & BRIX_SD_O_CREATE) { flags |= O_CREAT; }
71 if (sd_flags & BRIX_SD_O_EXCL) { flags |= O_EXCL; }
72 if (sd_flags & BRIX_SD_O_TRUNC) { flags |= O_TRUNC; }
73 if (sd_flags & BRIX_SD_O_APPEND) { flags |= O_APPEND; }
74 if (sd_flags & BRIX_SD_O_DIR) { flags |= O_DIRECTORY; }
75 if (sd_flags & BRIX_SD_O_NOFOLLOW) { flags |= O_NOFOLLOW; }
76
77 return flags;
78}

80/* brix_sd_posix_open_unconfined — plain open(2) of `path` (no export-root
81 * confinement), the client-side counterpart of the server's confined
82 * sd_posix_open. ngx-free + always compiled so the userland clients open through
83 * the one driver. Returns an fd, or -1 with errno set. */
84int
85brix_sd_posix_open_unconfined(const char *path, int sd_flags, mode_t mode)

90/* sd_posix_fill_stat — copy the protocol-neutral fields out of a struct stat into
91 * the brix_sd_stat_t the VFS consumes, deriving is_dir/is_reg from the mode. */
92static void
93sd_posix_fill_stat(const struct stat *st, brix_sd_stat_t *out)
94{
95 ngx_memzero(out, sizeof(*out));
96 out->size = st->st_size;
97 out->mtime = st->st_mtime;
98 out->ctime = st->st_ctime;
99 out->mode = st->st_mode;
100 out->ino = st->st_ino;
101 out->is_dir = S_ISDIR(st->st_mode) ? 1 : 0;
102 out->is_reg = S_ISREG(st->st_mode) ? 1 : 0;
103}

106/* sd_posix_init — pcalloc the instance state, copy root_canon, and open its
107 * persistent O_PATH anchor fd (the beneath API needs one per worker for hot
108 * opens; a -1 rootfd is tolerated). */
109static ngx_int_t
110sd_posix_init(brix_sd_instance_t *inst, void *driver_conf)
111{
112 sd_posix_state_t *st;
113 const char *root_canon = driver_conf;
114
115 if (root_canon == NULL) {
116 errno = EINVAL;
117 return NGX_ERROR;
118 }
119
120 st = ngx_pcalloc(inst->pool, sizeof(*st));
121 if (st == NULL) {
122 errno = ENOMEM;
123 return NGX_ERROR;
124 }
125
126 st->root_canon = ngx_pnalloc(inst->pool, ngx_strlen(root_canon) + 1);
127 if (st->root_canon == NULL) {
128 errno = ENOMEM;
129 return NGX_ERROR;
130 }
131 ngx_memcpy(st->root_canon, root_canon, ngx_strlen(root_canon) + 1);
132
133 st->rootfd = brix_beneath_open_root(root_canon); /* -1 tolerated */
134 inst->state = st;
135 return NGX_OK;
136}

138/* sd_posix_cleanup — close the persistent O_PATH rootfd (a kernel resource that
139 * must not leak on reconfig); the pool reclaims the state struct. A borrowed fd
140 * is left alone. */
141static void
142sd_posix_cleanup(brix_sd_instance_t *inst)
143{
144 sd_posix_state_t *st = inst->state;
145
146 if (st != NULL && !st->borrowed && st->rootfd >= 0) {
147 close(st->rootfd);
148 st->rootfd = -1;
149 }
150}

152/* brix_sd_posix_borrow_instance — build a pool-lived POSIX instance whose
153 * confinement anchor BORROWS an already-open persistent rootfd + root_canon owned
154 * by the caller's config (cleanup never closes a borrowed fd). Lets the VFS route
155 * its hot-path confined open through driver->open with no second openat() per
156 * request. NULL (errno set) for rootfd < 0 or OOM. */
157brix_sd_instance_t *
158brix_sd_posix_borrow_instance(ngx_pool_t *pool, ngx_log_t *log, int rootfd,
159 const char *root_canon)
160{
161 brix_sd_instance_t *inst;
162 sd_posix_state_t *st;
163
164 if (rootfd < 0) {
165 errno = EINVAL;
166 return NULL;
167 }
168
169 inst = ngx_pcalloc(pool, sizeof(*inst));
170 st = ngx_pcalloc(pool, sizeof(*st));
171 if (inst == NULL || st == NULL) {
172 errno = ENOMEM;
173 return NULL;
174 }
175
176 st->rootfd = rootfd;
177 st->root_canon = (char *) root_canon;
178 st->borrowed = 1;
179 inst->driver = &brix_sd_posix_driver;
180 inst->log = log;
181 inst->pool = pool;
182 inst->state = st;
183 return inst;
184}

186/* sd_posix_open — the POSIX realization of the VFS open cascade's beneath branch:
187 * brix_open_beneath under RESOLVE_BENEATH into an object carrying the fd. It does
188 * NOT fstat (metadata is the separate fstat slot, so a caller that doesn't need it
189 * pays no syscall; the VFS open path fstats once in adopt_fd). */
190static brix_sd_obj_t *
191sd_posix_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
192 mode_t mode, int *err_out)
193{
194 sd_posix_state_t *st = inst->state;
195 brix_sd_obj_t *obj;
196 int fd;
197
198 fd = brix_open_beneath(st->rootfd, path, sd_posix_flags(sd_flags), mode);
199 if (fd < 0) {
200 if (err_out != NULL) { *err_out = errno; }
201 return NULL;
202 }
203
204 obj = ngx_pcalloc(inst->pool, sizeof(*obj));
205 if (obj == NULL) {
206 close(fd);
207 if (err_out != NULL) { *err_out = ENOMEM; }
208 return NULL;
209 }
210
211 obj->driver = inst->driver;
212 obj->inst = inst;
213 obj->fd = fd;
214 return obj;
215}

217/* sd_posix_close — close the handle fd (idempotent; the obj lives on the pool);
218 * a close(2) error maps to NGX_ERROR. The VFS owns the close via brix_vfs_close. */
219static ngx_int_t
220sd_posix_close(brix_sd_obj_t *obj)
221{
222 if (obj == NULL || obj->fd == NGX_INVALID_FILE) {
223 return NGX_OK;
224 }
225 if (close(obj->fd) != 0) {
226 obj->fd = NGX_INVALID_FILE;
227 return NGX_ERROR;
228 }
229 obj->fd = NGX_INVALID_FILE;
230 return NGX_OK;
231}

368static ngx_int_t
369sd_posix_unlink(brix_sd_instance_t *inst, const char *path, int is_dir)
370{
371 sd_posix_state_t *st = inst->state;
372 brix_ns_delete_opts_t opts;
373 char abspath[PATH_MAX];
374
375 /* The vtable contract is a root-RELATIVE key (leading slash), matching
376 * sd_posix_open/stat and the non-POSIX drivers. brix_ns_delete works in
377 * ABSOLUTE paths under root_canon (strip_root), so build the absolute here. */
378 if ((size_t) snprintf(abspath, sizeof(abspath), "%s%s",
379 st->root_canon, path) >= sizeof(abspath))
380 {
381 errno = ENAMETOOLONG;
382 return NGX_ERROR;
383 }
384 ngx_memzero(&opts, sizeof(opts));
385 opts.require_directory = is_dir ? 1 : 0;
386 return sd_posix_ns_result(
387 brix_ns_delete(inst->log, st->root_canon, abspath, &opts));
388}

449static brix_sd_dir_t *
450sd_posix_opendir(brix_sd_instance_t *inst, const char *path, int *err_out)
451{
452 sd_posix_state_t *st = inst->state;
453 brix_sd_dir_t *dir;
454 sd_posix_dir_t *pd;
455 int fd;
456 DIR *dp;
457
458 fd = brix_open_beneath(st->rootfd, path, O_RDONLY | O_DIRECTORY, 0);
459 if (fd < 0) {
460 if (err_out != NULL) { *err_out = errno; }
461 return NULL;
462 }
463
464 dp = fdopendir(fd);
465 if (dp == NULL) {
466 if (err_out != NULL) { *err_out = errno; }
467 close(fd);
468 return NULL;
469 }
470
471 dir = ngx_pcalloc(inst->pool, sizeof(*dir));
472 pd = ngx_pcalloc(inst->pool, sizeof(*pd));
473 if (dir == NULL || pd == NULL) {
474 closedir(dp);
475 if (err_out != NULL) { *err_out = ENOMEM; }
476 return NULL;
477 }
478
479 pd->dp = dp;
480 dir->inst = inst;
481 dir->state = pd;
482 return dir;
483}

485static ngx_int_t
486sd_posix_readdir(brix_sd_dir_t *d, brix_sd_dirent_t *out)
487{
488 sd_posix_dir_t *pd = d->state;
489 struct dirent *de;
490
491 for ( ;; ) {
492 errno = 0;
493 de = readdir(pd->dp);
494 if (de == NULL) {
495 return errno != 0 ? NGX_ERROR : NGX_DONE;
496 }
497 if (de->d_name[0] == '.'
498 && (de->d_name[1] == '\0'
499 || (de->d_name[1] == '.' && de->d_name[2] == '\0')))
500 {
501 continue; /* skip "." and ".." */
502 }
503 ngx_cpystrn((u_char *) out->name, (u_char *) de->d_name,
504 sizeof(out->name));
505 return NGX_OK;
506 }
507}

602 brix_staged_file_t staged;
603 char final_path[PATH_MAX];
604} sd_posix_staged_t;

606static brix_sd_staged_t *
607sd_posix_staged_open(brix_sd_instance_t *inst, const char *final_path,
608 mode_t mode, int *err_out)
609{
610 sd_posix_state_t *st = inst->state;
611 brix_sd_staged_t *handle;
612 sd_posix_staged_t *ps;
613 char abspath[PATH_MAX];
614
615 handle = ngx_pcalloc(inst->pool, sizeof(*handle));
616 ps = ngx_pcalloc(inst->pool, sizeof(*ps));
617 if (handle == NULL || ps == NULL) {
618 if (err_out != NULL) { *err_out = ENOMEM; }
619 return NULL;
620 }
621
622 /* The vtable contract is a root-RELATIVE key (leading slash), matching
623 * sd_posix_open and the non-POSIX drivers' staged_open. brix_staged_open
624 * (and _commit/_abort) work in ABSOLUTE paths under root_canon, so build the
625 * absolute final path here and store it for commit/abort. */
626 if ((size_t) snprintf(abspath, sizeof(abspath), "%s%s",
627 st->root_canon, final_path) >= sizeof(abspath))
628 {
629 if (err_out != NULL) { *err_out = ENAMETOOLONG; }
630 return NULL;
631 }
632
633 if (brix_staged_open(inst->log, st->root_canon, abspath,
634 O_WRONLY | O_CREAT | O_EXCL, mode, 8, &ps->staged)
635 != NGX_OK)
636 {
637 if (err_out != NULL) { *err_out = errno; }
638 return NULL;
639 }
640
641 ngx_cpystrn((u_char *) ps->final_path, (u_char *) abspath,
642 sizeof(ps->final_path));
643 handle->inst = inst;
644 handle->state = ps;
645 return handle;
646}

648static ssize_t
649sd_posix_staged_write(brix_sd_staged_t *st, const void *buf, size_t len,
650 off_t off)
651{
652 sd_posix_staged_t *ps = st->state;
653
654 if (brix_vfs_pwrite_full(ps->staged.fd, buf, len, off) != NGX_OK) {
655 return -1;
656 }
657 return (ssize_t) len;
658}

660static ngx_int_t
661sd_posix_staged_commit(brix_sd_staged_t *st, int noreplace)
662{
663 sd_posix_staged_t *ps = st->state;
664 sd_posix_state_t *inst_st = st->inst->state;
665
666 return noreplace
667 ? brix_staged_commit_excl(st->inst->log, inst_st->root_canon,
668 &ps->staged, ps->final_path)
669 : brix_staged_commit(st->inst->log, inst_st->root_canon,
670 &ps->staged, ps->final_path);
671}

676 sd_posix_staged_t *ps = st->state;
677 sd_posix_state_t *inst_st = st->inst->state;
678
679 brix_staged_abort(st->inst->log, inst_st->root_canon, &ps->staged, 1);
680}
