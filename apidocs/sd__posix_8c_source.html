1/*
2 * sd_posix.c — the POSIX Storage Driver: a behaviour-preserving wrapper.
3 *
4 * WHAT: Implements brix_sd_posix_driver, the default backend. Every vtable
5 * slot delegates to an EXISTING confined helper (brix_open_beneath,
6 * brix_vfs_pread_full/pwrite_full, brix_ns_*, brix_lstat_beneath,
7 * the *xattr_confined_canon family, brix_staged_*), so the POSIX path
8 * is byte-identical to today's VFS once the VFS is wired to it (55.B+).
9 *
10 * WHY: Cutting the SD seam must not change POSIX behaviour. The safest way to
11 * guarantee that is for the POSIX driver to call the same code the VFS
12 * calls now, rather than re-implement any syscall. Confinement stays the
13 * kernel RESOLVE_BENEATH API; an EXDEV still means an escape attempt.
14 *
15 * HOW: The instance holds a persistent O_PATH rootfd + the root_canon string.
16 * Object handles carry the open fd and a metadata snapshot. The raw byte
17 * ops are worker-safe (no pool/metrics/log). Namespace ops translate the
18 * brix_ns_result_t status to errno via brix_vfs_ns_status_errno().
19 *
20 * NOTE: The worker-safe raw byte ops live in sd_posix_io.c and the nginx-coupled
21 * namespace/dir/xattr/staged ops in sd_posix_ns.c; this file keeps the
22 * instance lifecycle + the brix_sd_posix_driver descriptor, which
23 * references the split-out slots via sd_posix_internal.h.
24 */

52/* sd_posix_flags — map the backend-neutral BRIX_SD_O_* bits to an open(2) flag
53 * set (the vtable speaks SD flags; only the POSIX driver knows O_*). Always
54 * compiled: the server's confined open and the client's unconfined open
55 * (brix_sd_posix_open_unconfined) both use it, so the mapping is single-sourced. */
56static int
57sd_posix_flags(int sd_flags)
58{
59 int flags = 0;
60
61 if ((sd_flags & BRIX_SD_O_WRITE) && (sd_flags & BRIX_SD_O_READ)) {
62 flags = O_RDWR;
63 } else if (sd_flags & BRIX_SD_O_WRITE) {
64 flags = O_WRONLY;
65 } else {
66 flags = O_RDONLY;
67 }
68
69 if (sd_flags & BRIX_SD_O_CREATE) { flags |= O_CREAT; }
70 if (sd_flags & BRIX_SD_O_EXCL) { flags |= O_EXCL; }
71 if (sd_flags & BRIX_SD_O_TRUNC) { flags |= O_TRUNC; }
72 if (sd_flags & BRIX_SD_O_APPEND) { flags |= O_APPEND; }
73 if (sd_flags & BRIX_SD_O_DIR) { flags |= O_DIRECTORY; }
74 if (sd_flags & BRIX_SD_O_NOFOLLOW) { flags |= O_NOFOLLOW; }
75
76 return flags;
77}

79/* brix_sd_posix_open_unconfined — plain open(2) of `path` (no export-root

83int
84brix_sd_posix_open_unconfined(const char *path, int sd_flags, mode_t mode)

89/* sd_posix_fill_stat — copy the protocol-neutral fields out of a struct stat into
90 * the brix_sd_stat_t the VFS consumes, deriving is_dir/is_reg from the mode.
91 * Non-static + declared in sd_posix_internal.h: used by the split-out fstat op
92 * (sd_posix_io.c) and stat op (sd_posix_ns.c). */
93void
94sd_posix_fill_stat(const struct stat *st, brix_sd_stat_t *out)
95{
96 ngx_memzero(out, sizeof(*out));
97 out->size = st->st_size;
98 out->mtime = st->st_mtime;
99 out->ctime = st->st_ctime;
100 out->mode = st->st_mode;
101 out->ino = st->st_ino;
102 out->is_dir = S_ISDIR(st->st_mode) ? 1 : 0;
103 out->is_reg = S_ISREG(st->st_mode) ? 1 : 0;
104}

107/* sd_posix_init — pcalloc the instance state, copy root_canon, and open its
108 * persistent O_PATH anchor fd (the beneath API needs one per worker for hot
109 * opens; a -1 rootfd is tolerated). */
110static ngx_int_t
111sd_posix_init(brix_sd_instance_t *inst, void *driver_conf)
112{
113 sd_posix_state_t *st;
114 const char *root_canon = driver_conf;
115
116 if (root_canon == NULL) {
117 errno = EINVAL;
118 return NGX_ERROR;
119 }
120
121 st = ngx_pcalloc(inst->pool, sizeof(*st));
122 if (st == NULL) {
123 errno = ENOMEM;
124 return NGX_ERROR;
125 }
126
127 st->root_canon = ngx_pnalloc(inst->pool, ngx_strlen(root_canon) + 1);
128 if (st->root_canon == NULL) {
129 errno = ENOMEM;
130 return NGX_ERROR;
131 }
132 ngx_memcpy(st->root_canon, root_canon, ngx_strlen(root_canon) + 1);
133
134 st->rootfd = brix_beneath_open_root(root_canon); /* -1 tolerated */
135 inst->state = st;
136 return NGX_OK;
137}

139/* sd_posix_cleanup — close the persistent O_PATH rootfd (a kernel resource that
140 * must not leak on reconfig); the pool reclaims the state struct. A borrowed fd
141 * is left alone. */
142static void
143sd_posix_cleanup(brix_sd_instance_t *inst)
144{
145 sd_posix_state_t *st = inst->state;
146
147 if (st != NULL && !st->borrowed && st->rootfd >= 0) {
148 close(st->rootfd);
149 st->rootfd = -1;
150 }
151}

153/* brix_sd_posix_borrow_instance — build a pool-lived POSIX instance whose
154 * confinement anchor BORROWS an already-open persistent rootfd + root_canon owned
155 * by the caller's config (cleanup never closes a borrowed fd). Lets the VFS route
156 * its hot-path confined open through driver->open with no second openat() per
157 * request. NULL (errno set) for rootfd < 0 or OOM. */
158brix_sd_instance_t *
159brix_sd_posix_borrow_instance(ngx_pool_t *pool, ngx_log_t *log, int rootfd,
160 const char *root_canon)
161{
162 brix_sd_instance_t *inst;
163 sd_posix_state_t *st;
164
165 if (rootfd < 0) {
166 errno = EINVAL;
167 return NULL;
168 }
169
170 inst = ngx_pcalloc(pool, sizeof(*inst));
171 st = ngx_pcalloc(pool, sizeof(*st));
172 if (inst == NULL || st == NULL) {
173 errno = ENOMEM;
174 return NULL;
175 }
176
177 st->rootfd = rootfd;
178 st->root_canon = (char *) root_canon;
179 st->borrowed = 1;
180 inst->driver = &brix_sd_posix_driver;
181 inst->log = log;
182 inst->pool = pool;
183 inst->state = st;
184 return inst;
185}

187/* sd_posix_open — the POSIX realization of the VFS open cascade's beneath branch:
188 * brix_open_beneath under RESOLVE_BENEATH into an object carrying the fd. It does
189 * NOT fstat (metadata is the separate fstat slot, so a caller that doesn't need it
190 * pays no syscall; the VFS open path fstats once in adopt_fd). */
191static brix_sd_obj_t *
192sd_posix_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
193 mode_t mode, int *err_out)
194{
195 sd_posix_state_t *st = inst->state;
196 brix_sd_obj_t *obj;
197 int fd;
198
199 fd = brix_open_beneath(st->rootfd, path, sd_posix_flags(sd_flags), mode);
200 if (fd < 0) {
201 if (err_out != NULL) { *err_out = errno; }
202 return NULL;
203 }
204
205 /* Heap-allocate the obj shell (ngx_calloc), NOT from inst->pool. inst->pool is
206 * ngx_cycle->pool — thread-UNSAFE and also allocated from by the main event
207 * loop — but sd_posix_open runs in cache-fill worker threads too (a POSIX
208 * cache origin is opened per fill). A pool alloc there races the main thread
209 * and corrupts the pool -> SIGSEGV (confirmed via ThreadSanitizer). heap_shell
210 * makes the adopting layer free this shell: the VFS frees it after copying
211 * *obj by value (vfs_open.c adopt), and a pointer-holder (sd_cache/stage_engine
212 * source objects) frees it at close via brix_sd_obj_release(). */
213 obj = ngx_calloc(sizeof(*obj), inst->log);
214 if (obj == NULL) {
215 close(fd);
216 if (err_out != NULL) { *err_out = ENOMEM; }
217 return NULL;
218 }
219
220 obj->driver = inst->driver;
221 obj->inst = inst;
222 obj->fd = fd;
223 obj->heap_shell = 1;
224 return obj;
225}

227/* sd_posix_close — close the handle fd (idempotent; the obj lives on the pool);
228 * a close(2) error maps to NGX_ERROR. The VFS owns the close via brix_vfs_close. */
229static ngx_int_t
230sd_posix_close(brix_sd_obj_t *obj)
231{
232 if (obj == NULL || obj->fd == NGX_INVALID_FILE) {
233 return NGX_OK;
234 }
235 if (close(obj->fd) != 0) {
236 obj->fd = NGX_INVALID_FILE;
237 return NGX_ERROR;
238 }
239 obj->fd = NGX_INVALID_FILE;
240 return NGX_OK;
241}
