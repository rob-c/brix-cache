1/*
2 * put.c - WebDAV PUT body handling, including optional thread-pool writes.
3 *
4 * Phase 39 (WS8/HTTP-4): the body is written to a staged temp file and atomically
5 * renamed onto the final path on success (brix_staged_open/commit/abort, the
6 * same crash-safe lifecycle S3 PUT uses). So a client that drops mid-upload, a
7 * crash, or a failed inflate never leaves a partial/truncated object at the final
8 * path, and a concurrent GET only ever observes the old or the new object — never
9 * a half-written one.
10 */
11

68/*
69 * webdav_put_persist_checksums — §8.3 checksum-on-ingest.
70 *
71 * After a successful PUT commit, when brix_webdav_checksum_on_write names one or
72 * more algorithms, compute+persist each digest on the freshly committed file via
73 * the integrity service (xattr, or .cks sidecar fallback §8.2). Best-effort and
74 * opt-in (default off): any failure is ignored — the digest simply stays lazy.
75 * Synchronous (the operator opts in knowing the cost); large-file async offload is
76 * a future refinement.
77 */
78static void
79webdav_put_persist_checksums(ngx_http_request_t *r, const char *path)
80{
81 ngx_http_brix_webdav_loc_conf_t *conf;
82 char algs[256];
83 char *save = NULL, *tok;
84 int fd;
85
86 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
87 if (conf->checksum_on_write.len == 0
88 || brix_str_cbuf(algs, sizeof(algs), &conf->checksum_on_write) == NULL)
89 {
90 return;
91 }
92
93 /* Re-open the just-committed export file through the VFS (confined beneath
94 * the export root, impersonation-aware) to compute its digests. */
95 fd = brix_vfs_open_fd(r->connection->log, conf->common.root_canon, path,
96 O_RDONLY, 0);
97 if (fd < 0) {
98 return;
99 }
100 for (tok = strtok_r(algs, ", \t", &save); tok != NULL;
101 tok = strtok_r(NULL, ", \t", &save))
102 {
103 brix_integrity_info_t info;
104 brix_integrity_opts_t o;
105 ngx_memzero(&o, sizeof(o));
106 o.allow_xattr_cache = 1;
107 o.update_xattr_cache = 1;
108 o.require_regular_file = 1;
109 (void) brix_integrity_get_fd(r->connection->log, fd, NULL, path, tok,
110 &o, &info);
111 }
112 (void) close(fd);
113}

115static void
116webdav_put_aio_thread(void *data, ngx_log_t *log)
117{
118 webdav_put_aio_t *t = data;
119
120 (void) log;
121
122 t->io_errno = 0;
123
124 /*
125 * Phase 31 W2: stream the body straight from nginx's own request buffers
126 * to the staged temp fd — no full-body contiguous copy. The body for this
127 * path is all in-memory bufs anchored in r->pool (the caller gates spooled
128 * bodies to the synchronous streaming path), so they are stable for the
129 * lifetime of the request (held alive by r->main->count++). This helper
130 * only does pwrite(2), so it is safe to run on the thread pool — no nginx
131 * pool allocation or event-loop calls.
132 */
133 /* A driver-backed (object) staged target has no kernel fd — stream the body
134 * through the staged-write primitive; otherwise write straight to the temp fd. */
135 if (brix_vfs_staged_is_driver(t->staged)
136 ? brix_http_body_write_to_staged(t->r, t->staged) != NGX_OK
137 : brix_http_body_write_to_fd(t->r, brix_vfs_staged_fd(t->staged),
138 t->path, NULL) != NGX_OK)
139 {
140 t->io_errno = errno;
141 t->nwritten = -1;
142 return;
143 }
144
145 t->nwritten = (ssize_t) t->len;
146}

148static void
149webdav_put_aio_done(ngx_event_t *ev)
150{
151 ngx_thread_task_t *task = ev->data;
152 webdav_put_aio_t *t = task->ctx;
153 ngx_http_request_t *r = t->r;
154 ngx_int_t status;
155
156 /* Balance the r->main->count++ in webdav_handle_put_body that keeps the
157 * request alive across the async thread dispatch. */
158 r->main->count--;
159
160 if (t->nwritten < 0 || (size_t) t->nwritten < t->len) {
161
162 brix_log_safe_path(r->connection->log, NGX_LOG_ERR,
163 (ngx_uint_t) t->io_errno,
164 "brix_webdav: async write() failed for: \"%s\"",
165 t->path);
166 brix_dashboard_http_error(r, "webdav PUT async write failed");
167 brix_dashboard_http_finish(r);
168 /* Abort the staged temp (close + unlink) — the final path is untouched. */
169 brix_vfs_staged_abort(t->staged, 1);
170 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
171 return;
172 }
173
174 /* Atomically publish the completed temp onto the final path. */
175 if (brix_vfs_staged_commit(t->staged, 0) != NGX_OK) {
176 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, ngx_errno,
177 "brix_webdav: async staged commit failed for: "
178 "\"%s\"", t->path);
179 brix_dashboard_http_error(r, "webdav PUT staged commit failed");
180 brix_dashboard_http_finish(r);
181 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
182 return;
183 }
184
185 brix_dashboard_http_add(r, (ngx_atomic_int_t) t->len);
186 brix_dashboard_http_finish(r);
187
188 webdav_put_persist_checksums(r, (const char *) t->path); /* §8.3 */
189
190 status = t->created ? NGX_HTTP_CREATED : NGX_HTTP_NO_CONTENT;
191 webdav_send_status_only(r, (ngx_uint_t) status);
192}

194/*
195 * webdav_handle_put_body — write the request body to the destination file.
196 *
197 * This is the ngx_http_read_client_request_body() completion callback. By
198 * the time it runs, nginx has either buffered the body in memory
199 * (r->request_body->bufs with in_file=0) or spooled it to a temp file
200 * (in_file=1).
201 *
202 * The body is always written to a STAGED temp file (brix_staged_open) and
203 * atomically committed onto the final path on success (brix_staged_commit) or
204 * aborted on any error (brix_staged_abort) — so a partial/failed/aborted PUT
205 * never leaves a readable object at the final path (WS8/HTTP-4).
206 *
207 * The function chooses one of three write paths in order of preference:
208 * 1. Thread-pool async write (NGX_THREADS && memory body only): streams the
209 * body to the staged fd on the pool and commits in webdav_put_aio_done().
210 * 2. Synchronous spooled-file copy.
211 * 3. Synchronous in-memory write (+ optional inflate).
212 *
213 * Preconditions: the caller holds a reference count increment on r->main
214 * (done by ngx_http_read_client_request_body) so nginx will not free the
215 * request while this callback is pending.
216 *
217 * Ownership: the staged temp is opened here. The async path transfers the
218 * staged struct into the task (committed/aborted in the done handler); the
219 * synchronous paths commit or abort before sending the response.
220 *
221 * Pool allocation lifetime: all ngx_palloc calls here use r->pool.
222 */
223static void webdav_put_body_inner(ngx_http_request_t *r);
224
225/*
226 * Phase 40: the PUT body is read asynchronously, so the outer dispatch wrapper
227 * has already cleared the impersonation principal by the time this callback runs.
228 * Re-establish it for the duration of the (synchronous) body write — the staged
229 * open (brix_staged_open) then routes to the broker and the new file is
230 * owned by the mapped user. No-op unless map mode is active.
231 */
232void
233webdav_handle_put_body(ngx_http_request_t *r)
234{
235 ngx_http_brix_webdav_req_ctx_t *rx =
236 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
237
238 brix_imp_request_begin(rx != NULL ? rx->identity : NULL);
239 webdav_put_body_inner(r);
240 brix_imp_request_end();
241}

243/* Add an "X-Upload-Offset: <off>" response header (the resumable-PUT progress
244 * marker the client reads to know where to continue). Best-effort. */
245static void
246webdav_set_upload_offset(ngx_http_request_t *r, off_t off)
247{
248 ngx_table_elt_t *h = ngx_list_push(&r->headers_out.headers);
249 if (h == NULL) {
250 return;
251 }
252 h->hash = 1;
253 ngx_str_set(&h->key, "X-Upload-Offset");
254 h->value.data = ngx_pnalloc(r->pool, NGX_OFF_T_LEN);
255 if (h->value.data == NULL) {
256 h->hash = 0;
257 return;
258 }
259 h->value.len = ngx_sprintf(h->value.data, "%O", off) - h->value.data;
260}

262/*
263 * Resumable Content-Range PUT (brix_webdav_upload_resume on). Writes this
264 * chunk to a persistent, identity-keyed partial at its absolute offset and
265 * commits (atomic rename) only when the upload is complete. Across separate PUT
266 * requests — and across an nginx restart — the partial survives, so the client
267 * resumes from X-Upload-Offset. Append-only: the chunk must start exactly at the
268 * current partial size, else 409 + X-Upload-Offset tells the client the truth.
269 */
270static void
271webdav_put_ranged_resume(ngx_http_request_t *r,
272 ngx_http_brix_webdav_loc_conf_t *conf, const char *path,
273 const brix_http_content_range_t *cr)
274{
275 brix_staged_file_t staged;
276 off_t cur = 0;
277 brix_http_body_summary_t bs;
278 ngx_http_brix_webdav_req_ctx_t *wctx =
279 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
280 const char *principal = (wctx != NULL && wctx->dn[0] != '\0')
281 ? wctx->dn : NULL;
282 const char *stage_dir = conf->upload_stage_dir_canon[0]
283 ? conf->upload_stage_dir_canon : NULL;
284
285 if (brix_staged_open_resume(r->connection->log, conf->common.root_canon,
286 path, principal, stage_dir, NGX_FILE_DEFAULT_ACCESS, &staged, &cur)
287 != NGX_OK)
288 {
289 ngx_int_t st = brix_http_map_errno(ngx_errno);
290 webdav_metrics_finalize_request(r,
291 st >= 500 ? NGX_HTTP_INTERNAL_SERVER_ERROR : st);
292 return;
293 }
294
295 /* Append-only contiguity: the chunk must begin at the current partial size.
296 * If it doesn't, report the real resume offset and let the client re-issue —
297 * keep the partial (close without unlink). */
298 if (cr->start != cur) {
299 brix_staged_abort(r->connection->log, conf->common.root_canon,
300 &staged, 0 /* keep */);
301 webdav_set_upload_offset(r, cur);
302 webdav_send_status_only(r, NGX_HTTP_CONFLICT);
303 return;
304 }
305
306 if (r->request_body != NULL) {
307 if (brix_http_body_write_to_fd_at(r, staged.fd, path, &bs, cr->start)
308 != NGX_OK)
309 {
310 /* keep the partial; the client can re-send this chunk */
311 brix_staged_abort(r->connection->log, conf->common.root_canon,
312 &staged, 0 /* keep */);
313 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
314 return;
315 }
316 BRIX_WEBDAV_METRIC_ADD(bytes_rx_total, bs.bytes);
317 cur = cr->end + 1;
318 /* Flush so the reported resume offset is durable across a restart. */
319 (void) fsync(staged.fd);
320 }
321
322 /* Final chunk (last byte of the declared total) → commit; else keep partial
323 * and report the next expected offset. */
324 if (cr->total >= 0 && cr->end + 1 >= cr->total) {
325 /* Commit the staged partial onto the destination — atomic rename on the
326 * same filesystem, or copy-then-rename when staging on a different device
327 * (brix_webdav_stage_dir). brix_commit_staged owns neither close nor
328 * the active flag, so finalize the staged struct ourselves. On a stage
329 * device, record a durable pending-commit marker first so an interrupted
330 * cross-device move is finished by the reaper across a restart. */
331 ngx_flag_t stage_tracked = (stage_dir != NULL);
332 ngx_int_t crc;
333 if (stage_tracked) {
334 (void) brix_stage_mark_pending(staged.tmp_path, path,
335 r->connection->log);
336 }
337 crc = brix_commit_staged(staged.fd, staged.tmp_path, path,
338 staged.final_mode, r->connection->log);
339 if (staged.fd != NGX_INVALID_FILE) {
340 ngx_close_file(staged.fd);
341 staged.fd = NGX_INVALID_FILE;
342 }
343 staged.active = 0;
344 if (crc != NGX_OK) {
345 /* Keep the marker + partial — the reaper / a client retry completes it. */
346 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
347 return;
348 }
349 if (stage_tracked) {
350 brix_stage_unmark_pending(staged.tmp_path);
351 }
352 webdav_put_persist_checksums(r, (const char *) path); /* §8.3 */
353 webdav_send_status_only(r, NGX_HTTP_CREATED);
354 return;
355 }
356
357 brix_staged_abort(r->connection->log, conf->common.root_canon,
358 &staged, 0 /* keep partial for the next chunk */);
359 webdav_set_upload_offset(r, cur);
360 webdav_send_status_only(r, NGX_HTTP_OK);
361}

363static void
364webdav_put_body_inner(ngx_http_request_t *r)
365{
366 ngx_http_brix_webdav_loc_conf_t *conf;
367 char path[WEBDAV_MAX_PATH];
368 ngx_int_t rc;
369 brix_vfs_ctx_t vctx;
370 brix_vfs_staged_t *staged;
371 int staged_err = 0;
372 int created = 0;
373 brix_codec_id_t put_codec = BRIX_CODEC_IDENTITY;
374 struct stat sb;
375 ngx_int_t status;
376 brix_http_body_summary_t body_summary;
377 ngx_http_brix_webdav_req_ctx_t *wctx;
378 const char *identity;
379
380 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
381
382 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon, path,
383 sizeof(path));
384 if (rc != NGX_OK) {
385 /* RFC 4918 §9.7.1: missing intermediate collection → 409 Conflict */
386 if (rc == NGX_HTTP_NOT_FOUND) {
387 webdav_metrics_finalize_request(r, NGX_HTTP_CONFLICT);
388 return;
389 }
390 webdav_metrics_finalize_request(r, rc);
391 return;
392 }
393
394 /*
395 * Resumable upload: a Content-Range PUT places its chunk at an absolute
396 * offset on a persistent identity-keyed partial and commits only when the
397 * upload completes (brix_webdav_upload_resume on). Handled before the
398 * whole-body staged-write path below. A malformed Content-Range is 400.
399 */
400 if (conf->upload_resume) {
401 ngx_table_elt_t *crh = brix_http_find_header(
402 r, "Content-Range", sizeof("Content-Range") - 1);
403 if (crh != NULL && crh->value.len > 0) {
404 brix_http_content_range_t cr;
405 brix_http_parse_content_range(crh->value.data, crh->value.len, &cr);
406 if (!cr.present) {
407 webdav_metrics_finalize_request(r, NGX_HTTP_BAD_REQUEST);
408 return;
409 }
410 webdav_put_ranged_resume(r, conf, path, &cr);
411 return;
412 }
413 }
414
415 {
416 ngx_http_brix_webdav_req_ctx_t *rx =
417 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
418 brix_vfs_ctx_t pctx;
419 brix_vfs_stat_t pst;
420 int is_tls = 0;
421
422#if (NGX_HTTP_SSL)
423 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
424#endif
425 ngx_memzero(&sb, sizeof(sb));
426 brix_vfs_ctx_init(&pctx, r->pool, r->connection->log,
427 BRIX_PROTO_WEBDAV, conf->common.root_canon, conf->cache_root_canon,
428 conf->common.allow_write, is_tls,
429 (rx != NULL) ? rx->identity : NULL, path);
430 if (brix_vfs_probe(&pctx, 0 /* follow */, &pst) == NGX_OK) {
431 created = 0; /* target exists — sb populated for the ETag check */
432 sb.st_mtime = pst.mtime;
433 sb.st_size = pst.size;
434 sb.st_mode = (mode_t) pst.mode;
435 } else {
436 created = 1; /* new file — sb unused (etag check sees !exists) */
437 }
438 }
439
440 rc = brix_http_check_etag_preconditions(
441 r, !created, &sb, BRIX_ETAG_WEAK, BRIX_HTTP_COND_WEAK_EQUIV);
442 if (rc != NGX_OK) {
443 webdav_metrics_finalize_request(r, rc);
444 return;
445 }
446
447 /*
448 * Open a staged temp file beside the final path (O_EXCL, unique name, up to
449 * 16 attempts) through the metered VFS staged surface. The body is written
450 * here and atomically renamed onto the final path only on success — never
451 * exposing a partial object.
452 */
453 webdav_put_vfs_ctx_init(r, conf, path, &vctx);
454
455 staged = brix_vfs_staged_open(&vctx, NGX_FILE_DEFAULT_ACCESS, 16,
456 &staged_err);
457 if (staged == NULL) {
458 /* Use the errno the VFS open captured (logging may clobber errno). */
459 int open_errno = staged_err;
460
461 if (open_errno == NGX_ENOENT || open_errno == NGX_ENOTDIR) {
462 /* RFC 4918 §9.7.1 — PUT to non-existent parent collection is 409 */
463 webdav_metrics_finalize_request(r, NGX_HTTP_CONFLICT);
464 return;
465 }
466 /*
467 * A DAC-denied staged-temp create (e.g. the accessor lacks write on the
468 * parent collection → EACCES/EPERM, EXDEV/ELOOP on a confinement-blocked
469 * traversal) is a forbidden, bounded request — surface it as a clean 4xx
470 * via the shared errno map (EACCES/EPERM → 403, ENOSPC → 507, …), not a
471 * blanket 500. Only an errno with no clean 4xx contract (e.g. EIO) keeps
472 * the internal-error path. This mirrors S3 PUT's s3_put_finalize_fs_error.
473 */
474 status = brix_http_map_errno(open_errno);
475 brix_log_safe_path(r->connection->log, NGX_LOG_ERR,
476 (ngx_uint_t) open_errno,
477 "brix_webdav: staged open for write failed for: "
478 "\"%s\"", path);
479 webdav_metrics_finalize_request(
480 r, status >= 500 ? NGX_HTTP_INTERNAL_SERVER_ERROR : status);
481 return;
482 }
483
484 if (r->request_body != NULL) {
485 if (brix_http_body_summary(r, &body_summary) != NGX_OK) {
486 brix_vfs_staged_abort(staged, 1);
487 webdav_metrics_finalize_request(r,
488 NGX_HTTP_INTERNAL_SERVER_ERROR);
489 return;
490 }
491
492 wctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
493 identity = (wctx != NULL && wctx->dn[0] != '\0')
494 ? wctx->dn : "anonymous";
495 (void) brix_dashboard_http_start_identity(r, path, identity, "",
496 BRIX_XFER_PROTO_WEBDAV, BRIX_XFER_DIR_WRITE, "PUT",
497 (int64_t) body_summary.bytes);
498
499 BRIX_WEBDAV_METRIC_ADD(bytes_rx_total, body_summary.bytes);
500
501 {
502 ngx_table_elt_t *ce = brix_http_find_header(
503 r, "Content-Encoding", sizeof("Content-Encoding") - 1);
504 if (ce != NULL && ce->value.len > 0) {
505 const brix_codec_desc_t *d = brix_codec_by_http_token(
506 (const char *) ce->value.data, ce->value.len);
507 if (d == NULL || !d->available) {
508 /* Unknown or not-compiled-in Content-Encoding: never store the
509 * undecoded bytes (would silently corrupt the object). 415. */
510 brix_dashboard_http_error(r,
511 "webdav PUT unsupported Content-Encoding");
512 brix_dashboard_http_finish(r);
513 brix_vfs_staged_abort(staged, 1);
514 webdav_metrics_finalize_request(r,
515 NGX_HTTP_UNSUPPORTED_MEDIA_TYPE);
516 return;
517 }
518 put_codec = d->id;
519 }
520 }
521
522 if (!body_summary.has_spooled && body_summary.bytes > 0
523 && put_codec == BRIX_CODEC_IDENTITY
524 && conf->common.thread_pool != NULL)
525 {
526 ngx_thread_task_t *task;
527 webdav_put_aio_t *t;
528
529 task = ngx_thread_task_alloc(r->pool, sizeof(webdav_put_aio_t));
530 if (task == NULL) {
531 brix_dashboard_http_error(r, "webdav PUT task allocation failed");
532 brix_dashboard_http_finish(r);
533 brix_vfs_staged_abort(staged, 1);
534 webdav_metrics_finalize_request(r,
535 NGX_HTTP_INTERNAL_SERVER_ERROR);
536 return;
537 }
538
539 t = task->ctx;
540 t->r = r;
541 t->staged = staged; /* transfer staged ownership to the task */
542 t->len = body_summary.bytes;
543 t->created = created;
544 ngx_cpystrn((u_char *) t->path, (u_char *) path, sizeof(t->path));
545
546 /*
547 * Phase 31 W2: no full-body collection — the thread streams the
548 * in-memory body bufs straight to the staged fd (see
549 * webdav_put_aio_thread); the done handler commits/aborts.
550 */
551
552 brix_task_bind(task, webdav_put_aio_thread, webdav_put_aio_done);
553
554 if (ngx_thread_task_post(conf->common.thread_pool, task) != NGX_OK) {
555 brix_dashboard_http_error(r, "webdav PUT task post failed");
556 brix_dashboard_http_finish(r);
557 brix_vfs_staged_abort(t->staged, 1);
558 webdav_metrics_finalize_request(r,
559 NGX_HTTP_INTERNAL_SERVER_ERROR);
560 return;
561 }
562
563 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_THREADED]);
564 r->main->count++;
565 return;
566 }
567
568 if (body_summary.has_spooled) {
569 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_SPOOLED]);
570 } else if (body_summary.has_memory) {
571 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_MEMORY]);
572 } else {
573 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_EMPTY]);
574 }
575
576 {
577 ngx_int_t wrc;
578 ngx_int_t decode_status = NGX_HTTP_INTERNAL_SERVER_ERROR;
579
580 if (put_codec != BRIX_CODEC_IDENTITY) {
581 if (brix_vfs_staged_is_driver(staged)) {
582 /* Content-Encoding decode targets a kernel fd; an object
583 * backend exposes none. Decode-to-staged is a follow-up. */
584 errno = ENOSYS;
585 wrc = NGX_ERROR;
586 decode_status = NGX_HTTP_NOT_IMPLEMENTED;
587 } else {
588 wrc = brix_http_body_decode_to_fd(r,
589 brix_vfs_staged_fd(staged),
590 path, put_codec,
591 BRIX_DECODE_MAX_OUTPUT,
592 &body_summary,
593 &decode_status);
594 }
595 } else if (brix_vfs_staged_is_driver(staged)) {
596 wrc = brix_http_body_write_to_staged(r, staged);
597 } else {
598 wrc = brix_http_body_write_to_fd(r,
599 brix_vfs_staged_fd(staged),
600 path, &body_summary);
601 }
602 if (wrc != NGX_OK) {
603 brix_dashboard_http_error(r, "webdav PUT body write failed");
604 brix_dashboard_http_finish(r);
605 /* Abort the staged temp — the final path is never touched, so a
606 * failed write (e.g. a corrupt/over-large Content-Encoding that
607 * fails to decode) can never leave a readable partial object. The
608 * decode path reports the precise status (413 bomb / 400 corrupt). */
609 brix_vfs_staged_abort(staged, 1);
610 webdav_metrics_finalize_request(r, decode_status);
611 return;
612 }
613 }
614 brix_dashboard_http_add(r, (ngx_atomic_int_t) body_summary.bytes);
615 } else {
616 wctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
617 identity = (wctx != NULL && wctx->dn[0] != '\0')
618 ? wctx->dn : "anonymous";
619 (void) brix_dashboard_http_start_identity(r, path, identity, "",
620 BRIX_XFER_PROTO_WEBDAV, BRIX_XFER_DIR_WRITE, "PUT", 0);
621 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_EMPTY]);
622 }
623
624 brix_dashboard_http_finish(r);
625
626 /* Atomically publish the staged temp onto the final path (an empty PUT
627 * commits an empty file). excl=0 == replace (the prior brix_staged_commit
628 * non-EXCL semantics). */
629 if (brix_vfs_staged_commit(staged, 0) != NGX_OK) {
630 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, ngx_errno,
631 "brix_webdav: staged commit failed for: \"%s\"",
632 path);
633 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
634 return;
635 }
636
637 webdav_put_persist_checksums(r, (const char *) path); /* §8.3 */
638
639 status = created ? NGX_HTTP_CREATED : NGX_HTTP_NO_CONTENT;
640 r->headers_out.status = status;
641 r->headers_out.content_length_n = 0;
642 ngx_http_send_header(r);
643 /* r->header_only is false for PUT (clients don't send HEAD-like PUT);
644 * no need to check it here, but ngx_http_send_special handles it. */
645 webdav_metrics_finalize_request(r, ngx_http_send_special(r,
646 NGX_HTTP_LAST));
647}
