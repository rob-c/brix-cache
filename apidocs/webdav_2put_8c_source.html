1/*
2 * put.c - WebDAV PUT body handling, including optional thread-pool writes.
3 *
4 * Phase 39 (WS8/HTTP-4): the body is written to a staged temp file and atomically
5 * renamed onto the final path on success (brix_staged_open/commit/abort, the
6 * same crash-safe lifecycle S3 PUT uses). So a client that drops mid-upload, a
7 * crash, or a failed inflate never leaves a partial/truncated object at the final
8 * path, and a concurrent GET only ever observes the old or the new object — never
9 * a half-written one.
10 */

32/*
33 * webdav_handle_put_body — write the request body to the destination file.
34 *
35 * This is the ngx_http_read_client_request_body() completion callback. By
36 * the time it runs, nginx has either buffered the body in memory
37 * (r->request_body->bufs with in_file=0) or spooled it to a temp file
38 * (in_file=1).
39 *
40 * The body is always written to a STAGED temp file (brix_staged_open) and
41 * atomically committed onto the final path on success (brix_staged_commit) or
42 * aborted on any error (brix_staged_abort) — so a partial/failed/aborted PUT
43 * never leaves a readable object at the final path (WS8/HTTP-4).
44 *
45 * The function chooses one of three write paths in order of preference:
46 * 1. Thread-pool async write (NGX_THREADS && memory body only): streams the
47 * body to the staged fd on the pool and commits in webdav_put_aio_done().
48 * 2. Synchronous spooled-file copy.
49 * 3. Synchronous in-memory write (+ optional inflate).
50 *
51 * Preconditions: the caller holds a reference count increment on r->main
52 * (done by ngx_http_read_client_request_body) so nginx will not free the
53 * request while this callback is pending.
54 *
55 * Ownership: the staged temp is opened here. The async path transfers the
56 * staged struct into the task (committed/aborted in the done handler); the
57 * synchronous paths commit or abort before sending the response.
58 *
59 * Pool allocation lifetime: all ngx_palloc calls here use r->pool.
60 */
61static void webdav_put_body_inner(ngx_http_request_t *r);

63/*
64 * Phase 40: the PUT body is read asynchronously, so the outer dispatch wrapper
65 * has already cleared the impersonation principal by the time this callback runs.
66 * Re-establish it for the duration of the (synchronous) body write — the staged
67 * open (brix_staged_open) then routes to the broker and the new file is
68 * owned by the mapped user. No-op unless map mode is active.
69 */
70void
71webdav_handle_put_body(ngx_http_request_t *r)
72{
73 ngx_http_brix_webdav_req_ctx_t *rx =
74 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
75
76 brix_imp_request_begin(rx != NULL ? rx->identity : NULL);
77 webdav_put_body_inner(r);
78 brix_imp_request_end();
79}

81/*
82 * webdav_put_commit — atomically publish the staged temp and reply.
83 *
84 * WHAT: Commits the staged temp onto the final path (excl=0 == replace),
85 * persists any on-write checksums, and sends the terminal 201/204 with a
86 * zero-length body. A commit failure logs the path and finalizes 500.
87 *
88 * WHY: The commit + checksum + status trio is the single success epilogue for
89 * the sync and empty-PUT paths; keeping it here mirrors the async done
90 * handler's tail and keeps the orchestrator flat. Status codes/headers are
91 * byte-identical to the prior inline tail.
92 *
93 * HOW:
94 * 1. brix_vfs_writer_commit; failure → log + finalize 500, return.
95 * 2. Persist checksum-on-write digests (§8.3, best-effort).
96 * 3. Emit 201 Created (new file) or 204 No Content, content-length 0, send
97 * headers, and finalize with ngx_http_send_special(NGX_HTTP_LAST).
98 */
99static void
100webdav_put_commit(ngx_http_request_t *r, const char *path,
101 brix_vfs_writer_t *writer, int created)
102{
103 ngx_int_t status;
104 ngx_int_t vrc;
105 /* A direct WebDAV PUT is a STAGE-class ingest (client stream -> local
106 * object), so it belongs in the unified transfer ledger exactly like a
107 * root:// staged write or a WebDAV TPC — the "publish" record an operator
108 * greps for. bytes = the request body length (0 for an empty PUT). */
109 size_t xfer_bytes = (r->headers_in.content_length_n > 0)
110 ? (size_t) r->headers_in.content_length_n : 0;
111
112 brix_dashboard_http_finish(r);
113
114 /* Client->server body integrity: if the PUT asserted an ingest digest,
115 * verify it over the staged bytes and refuse the commit on mismatch (or, under
116 * require_digest, when none was supplied) — never publish poisoned bytes. */
117 vrc = webdav_put_verify_ingest_digest(r,
118 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module),
119 writer, path);
120 if (vrc != NGX_OK) {
121 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, 0,
122 "brix_webdav: PUT ingest digest rejected for: \"%s\"", path);
123 brix_vfs_writer_abort(writer);
124 webdav_metrics_finalize_request(r, (ngx_uint_t) vrc);
125 return;
126 }
127
128 /* Atomically publish the staged temp onto the final path (an empty PUT
129 * commits an empty file), folding the verify read-back when the export opts
130 * in. Replace semantics (excl=0 — the prior brix_staged_commit non-EXCL
131 * behaviour); a mismatch/short verify unlinks and fails. */
132 if (brix_vfs_writer_commit(writer) != NGX_OK) {
133 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, ngx_errno,
134 "brix_webdav: staged commit failed for: \"%s\"",
135 path);
136 brix_xfer_finish(BRIX_XFER_STAGE, "in", path, NULL, xfer_bytes,
137 BRIX_XFER_COMMIT_ERR, ngx_errno, r->connection->log);
138 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
139 return;
140 }
141
142 webdav_put_persist_checksums(r, path); /* §8.3 */
143
144 brix_xfer_finish(BRIX_XFER_STAGE, "in", path, NULL, xfer_bytes,
145 BRIX_XFER_OK, 0, r->connection->log);
146
147 status = created ? NGX_HTTP_CREATED : NGX_HTTP_NO_CONTENT;
148 r->headers_out.status = status;
149 r->headers_out.content_length_n = 0;
150 ngx_http_send_header(r);
151 /* r->header_only is false for PUT (clients don't send HEAD-like PUT);
152 * no need to check it here, but ngx_http_send_special handles it. */
153 webdav_metrics_finalize_request(r, ngx_http_send_special(r,
154 NGX_HTTP_LAST));
155}

157static void
158webdav_put_body_inner(ngx_http_request_t *r)
159{
160 ngx_http_brix_webdav_loc_conf_t *conf;
161 char path[WEBDAV_MAX_PATH];
162 brix_vfs_ctx_t vctx;
163 brix_vfs_writer_t *writer = NULL;
164 int created = 0;
165
166 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
167
168 if (webdav_put_precheck(r, conf, path, &created) == WEBDAV_PUT_DONE) {
169 return;
170 }
171
172 if (webdav_put_open_target(r, conf, path, &vctx, &writer)
173 == WEBDAV_PUT_DONE)
174 {
175 return;
176 }
177
178 if (webdav_put_stream_body(r, conf, path, writer, created)
179 == WEBDAV_PUT_DONE)
180 {
181 return;
182 }
183
184 webdav_put_commit(r, path, writer, created);
185}
