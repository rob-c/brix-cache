1/*
2 * sd_pblock_catalog.c — connection/cache core of the pblock SQLite catalog.
3 *
4 * WHAT: Owns the catalog's lifecycle and the machinery its two sibling files
5 * build on: opening the sqlite3 connection (schema bootstrap, WAL + sync
6 * tunables), closing it, the direct-mapped namespace lookup cache
7 * (nscache_*), and the shared SQL primitives (cat_prepare/cat_exec/cat_fail,
8 * parent_of/cat_parent_gate). The single-row objects CRUD lives in
9 * sd_pblock_catalog_objects.c and the namespace-tree operations (rename,
10 * directory iteration, xattrs) in sd_pblock_catalog_ns.c; both reach this
11 * core through sd_pblock_catalog_internal.h.
12 *
13 * WHY: See the header. This layer is pure libc + sqlite3 so it is independently
14 * unit-testable and carries none of the data-plane cost — the driver above
15 * it touches it only at metadata boundaries (open, fsync, close, namespace).
16 * Split from a single 1054-line file (phase-79) to hold every catalog file
17 * under the ~500-line, one-concept-per-file cap.
18 *
19 * HOW: One sqlite3 connection per handle, opened SERIALIZED (SQLITE_OPEN_FULLMUTEX)
20 * so a single per-export instance can be shared across a worker's thread
21 * pool, and in WAL mode with a busy timeout so separate worker *processes*
22 * contend safely. Statements are prepared per call (no cached cursors to
23 * race between threads) and finalized immediately. All paths are bound
24 * parameters — never string-formatted into SQL.
25 *
26 * Compiled only when the build found libsqlite3 (BRIX_HAVE_SQLITE), the
27 * same gate as sd_pblock.c — a no-sqlite build must stay byte-for-byte
28 * unchanged (see ./config).
29 */

45/* ---- namespace lookup cache ---------------------------------------------- *
46 * The cache layout (struct pblock_catalog, pblock_nscache_ent, the bucket/path
47 * limits) and the design rationale live in sd_pblock_catalog_internal.h so the
48 * objects/namespace files can consult and invalidate it. The mutators below are
49 * non-static (declared there) except the two purely local helpers. */

51/* FNV-1a of `path`, masked to the bucket count (power of two). */
52static uint32_t
53nscache_hash(const char *p)
54{
55 uint32_t h = BRIX_FNV1A32_OFFSET_BASIS;
56
57 for (; *p != '\0'; p++) {
58 h = (h ^ (unsigned char) *p) * BRIX_FNV1A32_PRIME;
59 }
60 return h & (PBLOCK_NSCACHE_BUCKETS - 1u);
61}

63static int
64nscache_ok(const char *p)
65{
66 return p != NULL && strlen(p) < PBLOCK_NSCACHE_PATHMAX;
67}

69/* Positive hit → 1 and fills *out (when non-NULL); miss → 0. */
70int
71nscache_get(pblock_catalog *cat, const char *path, pblock_meta *out)
72{
73 pblock_nscache_ent *e;
74 int hit = 0;
75
76 if (cat->cache == NULL || !nscache_ok(path)) {
77 return 0;
78 }
79 pthread_mutex_lock(&cat->cache_mtx);
80 e = &cat->cache[nscache_hash(path)];
81 if (e->valid && strcmp(e->path, path) == 0) {
82 if (out != NULL) {
83 *out = e->meta;
84 }
85 hit = 1;
86 }
87 pthread_mutex_unlock(&cat->cache_mtx);
88 return hit;
89}

91/* Snapshot the generation (call before the SQL read whose result may be cached). */
92uint64_t
93nscache_gen(pblock_catalog *cat)
94{
95 uint64_t g;
96
97 if (cat->cache == NULL) {
98 return 0;
99 }
100 pthread_mutex_lock(&cat->cache_mtx);
101 g = cat->cache_gen;
102 pthread_mutex_unlock(&cat->cache_mtx);
103 return g;
104}

106/* Store a positive entry, but only if no invalidation happened since `gen` was
107 * snapshotted (else the row we read may already be stale → drop it). */
108void
109nscache_store(pblock_catalog *cat, const char *path, const pblock_meta *meta,
110 uint64_t gen)
111{
112 pblock_nscache_ent *e;
113
114 if (cat->cache == NULL || !nscache_ok(path)) {
115 return;
116 }
117 pthread_mutex_lock(&cat->cache_mtx);
118 if (cat->cache_gen == gen) {
119 e = &cat->cache[nscache_hash(path)];
120 memcpy(e->path, path, strlen(path) + 1);
121 e->meta = *meta;
122 e->valid = 1;
123 }
124 pthread_mutex_unlock(&cat->cache_mtx);
125}

127/* Authoritative install after a write (put): bump the generation (discarding any
128 * concurrent lookup-fill for this path) and store the just-written meta, so the
129 * common write-then-read sequence (create → chmod → stat) serves the reads from
130 * memory. Assumes a path is not written concurrently from two threads (true on
131 * the worker event loop; the SQL write remains the source of truth regardless). */
132void
133nscache_put(pblock_catalog *cat, const char *path, const pblock_meta *meta)
134{
135 pblock_nscache_ent *e;
136
137 if (cat->cache == NULL) {
138 return;
139 }
140 pthread_mutex_lock(&cat->cache_mtx);
141 cat->cache_gen++;
142 if (nscache_ok(path)) {
143 e = &cat->cache[nscache_hash(path)];
144 memcpy(e->path, path, strlen(path) + 1);
145 e->meta = *meta;
146 e->valid = 1;
147 }
148 pthread_mutex_unlock(&cat->cache_mtx);
149}

151/* Drop the entry for `path` (if present) and bump the generation so any
152 * in-flight fill for a now-stale read is discarded. */
153void
154nscache_inval(pblock_catalog *cat, const char *path)
155{
156 pblock_nscache_ent *e;
157
158 if (cat->cache == NULL || !nscache_ok(path)) {
159 return;
160 }
161 pthread_mutex_lock(&cat->cache_mtx);
162 cat->cache_gen++;
163 e = &cat->cache[nscache_hash(path)];
164 if (e->valid && strcmp(e->path, path) == 0) {
165 e->valid = 0;
166 }
167 pthread_mutex_unlock(&cat->cache_mtx);
168}

170/* Invalidate everything (used by rename, which reparents whole subtrees). */
171void
172nscache_clear(pblock_catalog *cat)
173{
174 if (cat->cache == NULL) {
175 return;
176 }
177 pthread_mutex_lock(&cat->cache_mtx);
178 cat->cache_gen++;
179 memset(cat->cache, 0,
180 (size_t) PBLOCK_NSCACHE_BUCKETS * sizeof(cat->cache[0]));
181 pthread_mutex_unlock(&cat->cache_mtx);
182}

218/* parent_of — write the parent directory path of `path` into out[cap]. An
219 * absolute child of the root ("/x") and the root itself both yield "/". */
220void
221parent_of(const char *path, char *out, size_t cap)
222{
223 const char *slash = strrchr(path, '/');
224 size_t len;
225
226 /* The root has no parent ("" — not "/", which would make it its own child in
227 * a parent= listing). A direct child of the root has parent "/". */
228 if (path[0] == '/' && path[1] == '\0') {
229 out[0] = '\0';
230 return;
231 }
232 if (slash == NULL || slash == path) {
233 snprintf(out, cap, "/");
234 return;
235 }
236 len = (size_t) (slash - path);
237 if (len >= cap) {
238 len = cap - 1;
239 }
240 memcpy(out, path, len);
241 out[len] = '\0';
242}

244/* cat_parent_gate — POSIX parent gate for a NEW namespace row: the immediate
245 * parent must exist and be a directory. "/" exists implicitly (flat exports
246 * never pay a lookup), and the root row itself (parent "") always passes.
247 * Without this gate a create with missing parents inserted an ORPHAN: the key
248 * resolved (GET worked) but no parent= listing could ever reach it — readdir of
249 * an ancestor showed nothing and stat of the missing parent said ENOENT. Costs
250 * one indexed lookup per NEW row — creates only, never the byte path — and the
251 * nscache usually answers it. */
252int
253cat_parent_gate(pblock_catalog *cat, const char *path)
254{
255 char parent[1024];
256 pblock_meta pm;
257
258 parent_of(path, parent, sizeof(parent));
259 if (parent[0] == '\0' /* the root row */
260 || (parent[0] == '/' && parent[1] == '\0')) /* child of "/" */
261 {
262 return 0;
263 }
264 if (pblock_catalog_lookup(cat, parent, &pm) != 0) {
265 return cat_fail(ENOENT);
266 }
267 if (!pm.is_dir) {
268 return cat_fail(ENOTDIR);
269 }
270 return 0;
271}

275pblock_catalog *
276pblock_catalog_open(const char *db_path, int busy_timeout_ms)
277{
278 pblock_catalog *cat;
279 int flags = SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE
280 | SQLITE_OPEN_FULLMUTEX;
281
282 if (db_path == NULL) {
283 errno = EINVAL;
284 return NULL;
285 }
286
287 cat = calloc(1, sizeof(*cat));
288 if (cat == NULL) {
289 errno = ENOMEM;
290 return NULL;
291 }
292
293 if (sqlite3_open_v2(db_path, &cat->db, flags, NULL) != SQLITE_OK) {
294 sqlite3_close(cat->db); /* close is NULL/partial-open safe */
295 free(cat);
296 errno = EIO;
297 return NULL;
298 }
299
300 sqlite3_busy_timeout(cat->db, busy_timeout_ms > 0 ? busy_timeout_ms : 0);
301
302 /* Durability/throughput tunables (env): PBLOCK_SYNC ∈ {OFF,NORMAL,FULL}
303 * (default NORMAL) selects the WAL fsync discipline; PBLOCK_WAL_AUTOCKPT sets
304 * the WAL auto-checkpoint threshold in pages (default SQLite's 1000; 0 disables
305 * auto-checkpoint). Used to characterise the metadata-throughput trade-off. */
306 {
307 const char *sync_env = getenv("PBLOCK_SYNC");
308 const char *ck_env = getenv("PBLOCK_WAL_AUTOCKPT");
309 char pragma[64];
310
311 if (sync_env == NULL
312 || (strcmp(sync_env, "OFF") != 0 && strcmp(sync_env, "NORMAL") != 0
313 && strcmp(sync_env, "FULL") != 0)) {
314 sync_env = "NORMAL";
315 }
316 snprintf(pragma, sizeof(pragma), "PRAGMA synchronous=%s;", sync_env);
317 if (cat_exec(cat, "PRAGMA journal_mode=WAL;") != 0
318 || cat_exec(cat, pragma) != 0) {
319 int err = errno;
320 sqlite3_close(cat->db);
321 free(cat);
322 errno = err;
323 return NULL;
324 }
325 if (ck_env != NULL) {
326 snprintf(pragma, sizeof(pragma),
327 "PRAGMA wal_autocheckpoint=%d;", atoi(ck_env));
328 (void) cat_exec(cat, pragma); /* best-effort */
329 }
330 }
331
332 if (cat_exec(cat,
333 "CREATE TABLE IF NOT EXISTS objects("
334 " path TEXT PRIMARY KEY,"
335 " parent TEXT NOT NULL,"
336 " is_dir INTEGER NOT NULL,"
337 " blob_id TEXT NOT NULL DEFAULT '',"
338 " size INTEGER NOT NULL DEFAULT 0,"
339 " block_size INTEGER NOT NULL DEFAULT 0,"
340 " mtime INTEGER NOT NULL DEFAULT 0,"
341 " ctime INTEGER NOT NULL DEFAULT 0,"
342 " mode INTEGER NOT NULL DEFAULT 0,"
343 " uid INTEGER NOT NULL DEFAULT 0,"
344 " gid INTEGER NOT NULL DEFAULT 0,"
345 " xform TEXT NOT NULL DEFAULT '');") != 0
346 || cat_exec(cat,
347 "CREATE INDEX IF NOT EXISTS objects_parent"
348 " ON objects(parent);") != 0
349 || cat_exec(cat,
350 "CREATE TABLE IF NOT EXISTS xattrs("
351 " path TEXT NOT NULL,"
352 " name TEXT NOT NULL,"
353 " value BLOB NOT NULL,"
354 " PRIMARY KEY(path, name));") != 0
355 || cat_exec(cat,
356 /* Identity registry: catalog-internal synthetic uid/gid
357 * assignment (kind 0 = principal/uid, 1 = VO-group/gid). The
358 * UNIQUE index makes an id collision from a concurrent
359 * assignment a constraint failure, not silent aliasing. */
360 "CREATE TABLE IF NOT EXISTS ids("
361 " kind INTEGER NOT NULL,"
362 " name TEXT NOT NULL,"
363 " id INTEGER NOT NULL,"
364 " PRIMARY KEY(kind, name));") != 0
365 || cat_exec(cat,
366 "CREATE UNIQUE INDEX IF NOT EXISTS ids_id ON ids(id);") != 0)
367 {
368 int err = errno;
369
370 sqlite3_close(cat->db);
371 free(cat);
372 errno = err;
373 return NULL;
374 }
375
376 /* Forward-compat: add columns to an `objects` table created before they
377 * existed. Best-effort — a "duplicate column" error on an up-to-date
378 * schema is expected and ignored. Pre-migration rows keep uid/gid 0
379 * (service-owned), matching their pre-identity behaviour. */
380 sqlite3_exec(cat->db,
381 "ALTER TABLE objects ADD COLUMN block_size INTEGER NOT NULL DEFAULT 0;",
382 NULL, NULL, NULL);
383 sqlite3_exec(cat->db,
384 "ALTER TABLE objects ADD COLUMN uid INTEGER NOT NULL DEFAULT 0;",
385 NULL, NULL, NULL);
386 sqlite3_exec(cat->db,
387 "ALTER TABLE objects ADD COLUMN gid INTEGER NOT NULL DEFAULT 0;",
388 NULL, NULL, NULL);
389 sqlite3_exec(cat->db,
390 "ALTER TABLE objects ADD COLUMN xform TEXT NOT NULL DEFAULT '';",
391 NULL, NULL, NULL);
392
393 /* Namespace lookup cache (best-effort: a calloc failure just disables it —
394 * lookups still work, only slower). */
395 pthread_mutex_init(&cat->cache_mtx, NULL);
396 cat->cache_gen = 0;
397 cat->cache = calloc((size_t) PBLOCK_NSCACHE_BUCKETS, sizeof(*cat->cache));
398
399 return cat;
400}
