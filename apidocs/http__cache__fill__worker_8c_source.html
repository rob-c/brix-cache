27/* Worker thread: run the blocking source->store fill off the event loop,
28 * retrying transient origin failures to the T20 deadlines (single-pass when
29 * no hold is configured — the pre-phase-68 semantics for other protocols). */
30void
31brix_http_cache_fill_thread(void *data, ngx_log_t *log)
32{
33 brix_http_cache_fill_ctx_t *t = data;
34 brix_fill_retry_t rs;
35 unsigned n_eps;
36 char kbuf[256];
37
38 n_eps = (unsigned) sd_http_n_endpoints(
39 brix_sd_cache_source_instance(t->inst));
40 brix_fill_retry_init(&rs, t->client_hold, t->max_life, &t->waiters_n,
41 n_eps);
42 for ( ;; ) {
43 errno = 0;
44 t->attempts++;
45 t->result = brix_sd_cache_fill_key(t->inst, t->key);
46 t->err = errno;
47 switch (brix_fill_classify(t->result, t->err, &rs)) {
48 case BRIX_FILL_OK:
49 if (t->attempts > 1) {
50 ngx_log_error(NGX_LOG_WARN, log, 0,
51 "xrootd-fill: event=recovered key=\"%s\" attempts=%ud "
52 "elapsed_ms=%M — origin answered after transient failures",
53 brix_http_fill_log_key(t->key, kbuf, sizeof(kbuf)),
54 t->attempts,
55 (ngx_msec_t) (ngx_current_msec - t->started_ms));
56 }
57 return;
58 case BRIX_FILL_DEFINITIVE:
59 return;
60 case BRIX_FILL_RETRY:
61 break;
62 }
63 /* One line per failed attempt: WHICH object, WHICH attempt, WHY —
64 * the trail that locates a flapping origin or a lossy WAN window.
65 * Bounded by the backoff schedule (a handful per fill, worst case). */
66 ngx_log_error(NGX_LOG_WARN, log, t->err,
67 "xrootd-fill: event=retry key=\"%s\" attempt=%ud rc=%i "
68 "elapsed_ms=%M endpoints=%ud waiters=%d next_backoff_ms=%M",
69 brix_http_fill_log_key(t->key, kbuf, sizeof(kbuf)),
70 t->attempts, t->result,
71 (ngx_msec_t) (ngx_current_msec - t->started_ms), n_eps,
72 atomic_load_explicit(&t->waiters_n, memory_order_relaxed),
73 rs.backoff_ms);
74 if (!brix_fill_retry_wait(&rs)) {
75 if (t->err == EBADMSG) {
76 return; /* proven-bad data: 502, not "later" */
77 }
78 t->err = ETIMEDOUT; /* deadline exhausted: 504 to waiters */
79 return;
80 }
81 }
82}

84/* Resolve one waiter with the fill outcome (event loop). */
85static void
86brix_http_fill_resolve_waiter(brix_http_cache_fill_ctx_t *t,
87 brix_http_fill_waiter_t *w)
88{
89 ngx_http_request_t *r = w->r;
90 ngx_connection_t *c = r->connection;
91 ngx_int_t rc;
92
93 if (t->result == NGX_OK) {
94 rc = w->reenter(r, w->reenter_data); /* hit -> serve (zero-copy) */
95 } else if (t->result == NGX_DECLINED) {
96 ngx_log_error(NGX_LOG_WARN, c->log, 0,
97 "brix: cache fill declined for \"%s\" (object not cacheable and "
98 "remote-source streaming is not yet supported) - returning 502",
99 t->key);
100 rc = NGX_HTTP_BAD_GATEWAY;
101 } else if (t->err == ENOENT || t->err == ENOTDIR) {
102 /* The origin's definitive answer: the object does not exist. */
103 rc = NGX_HTTP_NOT_FOUND;
104 } else if (t->err == EACCES || t->err == EPERM) {
105 rc = NGX_HTTP_FORBIDDEN;
106 } else if (t->err == ETIMEDOUT) {
107 /* T20: deadline exhausted while this waiter was still attached —
108 * same keep-alive 504 the hold timer sends. */
109 brix_http_fill_send_retry_later(r);
110 ngx_http_run_posted_requests(c);
111 return;
112 } else {
113 /* per-waiter detail at debug; the fill-level ERR line in done()
114 * carries the full story once, not once per parked client */
115 ngx_log_debug2(NGX_LOG_DEBUG_HTTP, c->log, 0,
116 "xrootd-fill: resolving waiter with 502 (key \"%s\", err %d)",
117 t->key, t->err);
118 rc = NGX_HTTP_BAD_GATEWAY;
119 }
120
121 ngx_http_finalize_request(r, rc);
122 ngx_http_run_posted_requests(c);
123}

125/* Event loop: resolve EVERY still-attached waiter with the one fill outcome
126 * (the stampede coalescing contract), then release the fill. Waiter memory
127 * stays with its request's pool cleanup. */
128void
129brix_http_cache_fill_done(ngx_event_t *ev)
130{
131 ngx_thread_task_t *task = ev->data;
132 brix_http_cache_fill_ctx_t *t = task->ctx;
133 brix_http_cache_fill_ctx_t **pp;
134 brix_http_fill_waiter_t *w;
135 char errscratch[BRIX_SESSLOG_ERR_MAX];
136 const char *err;
137 int ok;
138
139 /* Unlink from the in-flight list FIRST so a re-entered handler that
140 * misses again starts a fresh fill rather than attaching to this one. */
141 for (pp = &brix_http_fills; *pp != NULL; pp = &(*pp)->next) {
142 if (*pp == t) {
143 *pp = t->next;
144 break;
145 }
146 }
147
148 /* ONE outcome line per fill — the anchor every other event (retry /
149 * hold-expired / client-gone) correlates with via the key. */
150 {
151 int waiters = atomic_load_explicit(&t->waiters_n,
152 memory_order_relaxed);
153 char kbuf[256];
154 const char *k = brix_http_fill_log_key(t->key, kbuf, sizeof(kbuf));
155 ngx_msec_t elapsed = ngx_current_msec - t->started_ms;
156
157 if (t->result == NGX_OK) {
158 ngx_uint_t lvl = (waiters > 0) ? NGX_LOG_INFO : NGX_LOG_NOTICE;
159
160 ngx_log_error(lvl, ev->log, 0,
161 "xrootd-fill: event=%s key=\"%s\" attempts=%ud "
162 "elapsed_ms=%M waiters=%d",
163 (waiters > 0) ? "done" : "done-detached", k,
164 t->attempts, elapsed, waiters);
165 } else if (t->err == ENOENT || t->err == ENOTDIR) {
166 ngx_log_error(NGX_LOG_INFO, ev->log, 0,
167 "xrootd-fill: event=not-found key=\"%s\" attempts=%ud "
168 "elapsed_ms=%M waiters=%d — the origin's definitive 404",
169 k, t->attempts, elapsed, waiters);
170 } else if (t->err == ETIMEDOUT) {
171 ngx_log_error(NGX_LOG_WARN, ev->log, 0,
172 "xrootd-fill: event=exhausted key=\"%s\" attempts=%ud "
173 "elapsed_ms=%M waiters=%d verdict=\"504 retry-later to every "
174 "waiter; no origin answered within the deadline\"",
175 k, t->attempts, elapsed, waiters);
176 } else {
177 ngx_log_error(NGX_LOG_ERR, ev->log, t->err,
178 "xrootd-fill: event=failed key=\"%s\" attempts=%ud "
179 "elapsed_ms=%M waiters=%d verdict=%s",
180 k, t->attempts, elapsed, waiters,
181 (t->err == EBADMSG)
182 ? "\"502; every endpoint served corrupt data "
183 "(verify quarantined the evidence)\""
184 : "\"502 to every waiter\"");
185 }
186 }
187
188 ok = (t->result == NGX_OK);
189 err = ok ? NULL : brix_sesslog_err_from_errno(t->err, errscratch,
190 sizeof(errscratch));
191 brix_sess_result(t->sess, ok, t->key, BRIX_SESS_MODE_READ, err);
192 brix_sess_xfer_end(t->sess, &t->sess_xfer,
193 ok ? BRIX_SESS_XFER_COMPLETE
194 : BRIX_SESS_XFER_ABORTED);
195 brix_sess_end(t->sess, ok ? BRIX_SESS_END_SERVER
196 : BRIX_SESS_END_ERROR);
197
198 while ((w = t->waiters) != NULL) {
199 brix_http_fill_detach(w); /* unlink + stop the hold */
200 brix_http_fill_resolve_waiter(t, w);
201 }
202 free(t->task); /* the calloc'd task+ctx block (task is first) */
203}
