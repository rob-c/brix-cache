1/* ---- File: tpc_internal.h — Native TPC source-side pull API and shared types ----
2 *
3 * WHAT: Defines all shared types, constants, and function declarations for native XRootD third-party-copy (TPC) destination-side pull. Wire constants → TPC_IO_TIMEOUT_SEC(60s), TPC_CONNECT_TIMEOUT_SEC(5s), TPC_CHUNK_SIZE(1MB per kXR_read), TPC_RESP_MAX_BODY(1MB+256 malloc cap); typedef brix_tpc_params_t — parsed tpc.* opaque fields (key/src/src_host/src_path/dst/lfn/org/stage/token_mode + has_* flags + src_port); typedef brix_tpc_pull_t — per-pull heap-allocated task context containing connection/ctx/conf refs, streamid/options/mode_bits, src info, key/org/delegated_token/token_scope, dst_path/dst_fd/fhandle_idx/reply_kind/result/xrd_error/bytes_written/err_msg; API declarations → brix_tpc_parse_opaque(opaque,out) parses opaque into params struct (parse.c); tpc_send_all(fd,buf,len)/tpc_recv_response(fd,status,body,dlen) low-level socket helpers (io.c); tpc_connect(t) DNS+TCP connect with timeout (connect.c); brix_tpc_check_src_policy(src_host,port,allow_local,allow_private,err_msg,sz) SSRF preflight; tpc_bootstrap(t,fd) anonymous session setup kXR_protocol+kXR_login (bootstrap.c); tpc_outbound_finish_login/tpc_outbound_gsi/tpc_outbound_ztn/GSI DH helpers for source auth; tpc_pull_from_source(t,fd) remote open+read loop+fsync+close (source.c); brix_tpc_pull_thread(data,log) thread-pool orchestrator connect→bootstrap→pull (thread.c); tpc_fetch_delegated_token(t) OAuth2/OIDC token fetch (tpc_token.c); brix_tpc_pull_done(ev) main-thread completion callback sends kXR_open response/error (done.c); brix_tpc_prepare_pull/launch_pull/start_pull event-thread entry points validate+allocate+post to thread pool (launch.c).
4 *
5 * WHY: TPC destination-side pull requires a coordinated sequence across multiple files — parsing opaque params, connecting to remote origin, bootstrapping session, streaming reads, completion callback. This header centralizes all shared types so launch.c/thread.c/source.c/connect.c/bootstrap.c/io.c/done.c/tpc_token.c can reference the same structs and function signatures without duplication. Heap-allocated pull_t struct enables ngx_thread_task_post() lifecycle (allocate in event thread → post to pool → free in done callback). Wire constants ensure consistent timeout/chunk sizes across all TPC files.
6 *
7 * HOW: Constants at top → typedef brix_tpc_params_t with opaque field comments → API declaration for parse_opaque → typedef brix_tpc_pull_t with heap/lifecycle comments → grouped function declarations by file (io.c helpers, connect.c, bootstrap.c auth helpers, source.c pull, thread.c worker, tpc_token.c fetch, done.c callback, launch.c entry points). Each declaration includes brief WHAT describing behavior and return value. */

28/* Async kXR_open resolution (phase-57 §F8): a real source (EOS/dCache, or any
29 * server still completing the TPC rendezvous) may answer the open with kXR_wait
30 * (retry-after) or kXR_waitresp (a deferred kXR_attn asynresp will follow) before
31 * it settles. tpc_open_resolve() honours that flow, bounded on every axis so a
32 * source that never resolves fails cleanly instead of hanging the pull thread. */
33#define TPC_OPEN_RESOLVE_MAX_SEC 120 /* total wall-clock cap for the negotiation */
34#define TPC_OPEN_WAIT_CAP_SEC 15 /* clamp the per-recv idle timeout during open
35 * resolution, so a silent (no-attn) source
36 * fails fast enough for the client's --tpc
37 * fallback to run */
38#define TPC_OPEN_WAIT_RETRY_SEC 1 /* how long to sleep before RESENDing an open
39 * after a kXR_wait. A real xrootd source can
40 * request a multi-second wait while the TPC
41 * grant is only briefly visible to our racing
42 * dest-open; sleeping the server's full hint
43 * (up to CAP_SEC) misses the window and the
44 * grant lapses ("tpc authorization expired").
45 * Stock XrdCl re-polls sub-second, so cap our
46 * retry sleep here (MAX_ITERS/MAX_SEC still
47 * bound the total) and re-poll while valid. */
48#define TPC_OPEN_RESOLVE_MAX_ITERS 16 /* max wait/waitresp/attn rounds */

57/*
58 * Parsed tpc.* parameters from an XRootD kXR_open opaque query string.
59 *
60 * XRootD TPC opaque parameters are appended to the path after '?':
61 * /path/to/file?tpc.src=root://src//path&tpc.key=<token>&tpc.dst=root://dst//path
62 *
63 * src_host, src_port, src_path are parsed from the tpc.src URL.
64 * has_* flags indicate which fields were actually present in the opaque string.
65 */
66typedef struct {
67 char key[128]; /* tpc.key — authorization token */
68 char src[512]; /* tpc.src raw URL */
69 char src_host[256]; /* hostname parsed from tpc.src */
70 char src_path[PATH_MAX];/* path parsed from tpc.src */
71 char dst[512]; /* tpc.dst raw URL */
72 char lfn[PATH_MAX]; /* tpc.lfn source logical path */
73 char org[256]; /* tpc.org origin identity */
74 char stage[64]; /* tpc.stage, usually "copy" */
75 char token_mode[32]; /* tpc.token_mode: "none", "passthrough",
76 "oidc-agent", "token-exchange" */
77 int has_src; /* 1 = tpc.src was present */
78 int has_dst; /* 1 = tpc.dst was present */
79 int has_key; /* 1 = tpc.key was present */
80 int has_lfn; /* 1 = tpc.lfn was present */
81 int has_org; /* 1 = tpc.org was present */
82 int has_stage; /* 1 = tpc.stage was present */
83 int has_token_mode; /* 1 = tpc.token_mode was present */
84 uint16_t src_port; /* TCP port from tpc.src (0 → use 1094) */
85} brix_tpc_params_t;

100/*

102 * and freed in brix_tpc_pull_done() after the result is consumed.
103 *
104 * The thread function (brix_tpc_pull_thread) connects to the XRootD source
105 * server, bootstraps a session (handshake+protocol+login), opens src_path,
106 * streams the file content into dst_fd, and closes the source connection.
107 */

110 brix_ctx_t *ctx;
111 ngx_stream_brix_srv_conf_t *conf;

113 uint16_t options; /* kXR_retstat etc., from the client's kXR_open */
114 uint16_t mode_bits; /* permission bits, from the client's kXR_open */
115 char src_host[256];
116 uint16_t src_port; /* 0 = use default 1094 */
117 char src_path[PATH_MAX];
118 char tpc_key[128]; /* presented to source in ?tpc.key= opaque */
119 char tpc_org[256]; /* presented to source in ?tpc.org= opaque */
120 char token_mode[32]; /* source-auth token mode: none/passthrough (strict)/
121 * passthrough-opt (default, opportunistic)/
122 * oidc-agent/token-exchange */
123 char delegated_token[65536]; /* delegated/forwarded access token: fetched
124 * (oidc-agent/token-exchange), read from the
125 * bearer file, or — for "passthrough" — the
126 * client's own inbound bearer JWT captured on
127 * the event loop (launch.c) */
128 char token_scope[256]; /* scope string for token exchange request */
129 uint8_t gsi_rtag[8]; /* GSI round-1 random tag (sent in certreq) */
130 char dst_path[PATH_MAX]; /* local path being written */
131 int dst_fd; /* open O_RDWR fd on dst_path; caller must close */
132 int fhandle_idx; /* ctx->files[] slot pre-allocated by launcher */
133 int reply_kind; /* BRIX_TPC_REPLY_* controls done callback */
134 int result; /* NGX_OK on success, NGX_ERROR on failure */
135 int xrd_error; /* kXR_* error code when result == NGX_ERROR */
136 uint64_t transfer_id; /* shared TPC registry entry, 0 if unavailable */
137 brix_sess_t *sess; /* outbound lifecycle audit session */
138 brix_sess_xfer_t sess_xfer; /* source-side transfer audit record */
139 size_t bytes_written;/* source bytes copied into dst_fd */
140 char err_msg[512]; /* human-readable error detail for logging */
141 ngx_uint_t pmark_exp; /* SciTags experiment id for the outbound flow, */
142 ngx_uint_t pmark_act; /* and activity id; 0 = not marked (resolved on the */
143 /* event loop in start_pull, applied in connect.c) */
144 void *tls; /* SSL* once the pull upgraded to TLS (kXR_gotoTLS), */
145 /* NULL = plaintext. The I/O helpers route through */
146 /* it transparently. Owned with tls_ctx; freed in */
147 /* thread.c via tpc_tls_teardown(). (phase-57 §F5) */
148 void *tls_ctx; /* SSL_CTX* backing tls (per-pull client ctx), or NULL */
149 u_char *deleg_cred_pem; /* §F6: captured delegated proxy credential (proxy
150 * cert + key + issuer chain, PEM) to authenticate the
151 * pull AS THE USER instead of conf->certificate;
152 * NULL = use the gateway cert. malloc'd; freed in
153 * thread.c. */
154 size_t deleg_cred_len;
155} brix_tpc_pull_t;

162/*
163 * Send the whole buffer over a blocking fd, looping on partial writes (retries
164 * on EINTR). buf is borrowed. Returns 0 once all len bytes are sent, -1 on any
165 * other send() error; a -1 leaves the socket mid-message (caller must abort).
166 */
167int tpc_send_all(brix_tpc_pull_t *t, int fd, const void *buf, size_t len);
168/*
169 * Read one XRootD ServerResponseHdr frame plus its payload from fd.
170 * On success returns 0 and sets *status (host order kXR_* code) and *dlen; *body
171 * is a malloc'd, NUL-terminated copy of the payload that the caller must free()
172 * (NULL when *dlen == 0). dlen is rejected if it exceeds TPC_RESP_MAX_BODY.
173 * Returns -1 on I/O, framing, oversize, or allocation failure (nothing to free).
174 */
175int tpc_recv_response(brix_tpc_pull_t *t, int fd, uint16_t *status,
176 u_char **body, uint32_t *dlen);

225/*
226 * Complete a kXR_login kXR_authmore using ztn and/or GSI credentials from
227 * ngx_stream_brix_srv_conf_t (bearer file, certificate paths). login_body is
228 * the borrowed authmore payload (login_dlen bytes); the auth-method list is read
229 * from it after the session id. When the server offers both, ZTN is tried first
230 * and falls through to GSI only if the server also lists gsi and a cert is

237/*
238 * GSI handshake round 1: load the local cert chain + key, send a kXGC_certreq
239 * kXR_auth frame on fd, and require a kXR_authmore reply (>= 16 bytes). All
240 * locally allocated OpenSSL objects are freed before return. Returns 0 when the
241 * server is mid-handshake (caller proceeds to tpc_outbound_gsi_exchange), or -1
