18/* Cookie for the catalog-enumeration callback: the same growing heap buffer the
19 * walk path uses, plus an OOM flag that aborts the enumeration. */
20typedef struct {
21 u_char **buf;
22 size_t *cap;
23 size_t *used;
24 brix_scan_summary_t *sum;
25 int oom;
26} scan_catalog_ctx_t;

28/* Emit one "object" record per enumerated backend object. A driver-recovered
29 * logical path of NULL marks an orphan (a stored object with no namespace entry).
30 * Returns 0 to continue the enumeration, 1 to abort (only on OOM). */
31static int
32scan_catalog_emit_object(void *ctx, const brix_sd_catalog_ent_t *ent)
33{
34 scan_catalog_ctx_t *cc = ctx;
35 char line[SCAN_LINE_MAX];
36 int n;
37
38 n = brix_scan_record_object(line, sizeof(line), ent->key, ent->path,
39 ent->have_stat ? (int64_t) ent->size : 0,
40 ent->have_stat ? (int64_t) ent->mtime : 0,
41 ent->path == NULL ? 1 : 0);
42 if (n < 0) {
43 return 0; /* unrepresentable key → soft skip */
44 }
45 cc->sum->files++;
46 cc->sum->ok++;
47 if (scan_append(cc->buf, cc->cap, cc->used, line, (size_t) n) < 0) {
48 cc->oom = 1;
49 return 1; /* abort enumeration */
50 }
51 return 0;
52}

102/*
103 * WHAT: per-object context for the catalog-verify action — the enumeration
104 * cookie plus the one OPEN driver object being verified and the caller's
105 * output line buffer.
106 * WHY: mirrors scan_action_ctx_t on the POSIX walk side; keeps the verify
107 * record builder at ≤ 5 params with the shared state passed explicitly.
108 * HOW: scan_catalog_verify_one fills one on the stack after a successful
109 * driver open and hands its address to scan_verify_obj_record.
110 */
111typedef struct {
112 scan_verify_ctx_t *cc;
113 const char *logical; /* recovered logical path */
114 const brix_sd_catalog_ent_t *ent; /* enumeration entry (stat) */
115 brix_sd_obj_t *obj; /* OPEN driver object (fd = -1)*/
116 char *line; /* output record buffer */
117 size_t linesz;
118} scan_verify_obj_ctx_t;

120/* Build one verify "file" record for an OPEN catalog object: read the stored
121 * XrdCks value (xattr, no byte read), recompute over the object's bytes, compare.
122 * Reuses the integrity layer exactly as the POSIX verify path does — the only
123 * difference is the source is a driver object (fd = -1) instead of a POSIX fd, so
124 * the recompute reads through obj->driver->pread (which, for Ceph, reassembles
125 * the libradosstriper layout → byte-identical to stock XrdCeph). */
126static int
127scan_verify_obj_record(scan_verify_obj_ctx_t *v)
128{
129 scan_verify_ctx_t *cc = v->cc;
130 brix_integrity_info_t stored_info, comp_info;
131 brix_integrity_opts_t io;
132 const char *stored = NULL;
133 const char *computed = NULL;
134 const char *status;
135 char stored_copy[129];
136 int64_t size = v->ent->have_stat ? (int64_t) v->ent->size : 0;
137 int64_t mtime = v->ent->have_stat ? (int64_t) v->ent->mtime : 0;
138
139 ngx_memzero(&io, sizeof(io));
140 io.allow_xattr_cache = 1;
141 io.no_compute = 1; /* stored value only — no byte read */
142 if (brix_integrity_get_fd(cc->log, -1, v->obj, v->logical, cc->alg, &io,
143 &stored_info) == NGX_OK)
144 {
145 ngx_memcpy(stored_copy, stored_info.hex, ngx_strlen(stored_info.hex) + 1);
146 stored = stored_copy; /* survive the recompute's info reuse */
147 }
148
149 ngx_memzero(&io, sizeof(io));
150 io.allow_xattr_cache = 0; /* force a fresh compute over the bytes */
151 io.no_compute = 0;
152 if (brix_integrity_get_fd(cc->log, -1, v->obj, v->logical, cc->alg, &io,
153 &comp_info) != NGX_OK)
154 {
155 status = "unreadable";
156 cc->sum->unreadable++;
157 } else {
158 computed = comp_info.hex;
159 cc->sum->bytes += (uint64_t) size;
160 if (stored == NULL) {
161 status = "missing";
162 cc->sum->missing++;
163 } else if (strcasecmp(stored, computed) == 0) {
164 status = "ok";
165 cc->sum->ok++;
166 } else {
167 status = "mismatch";
168 cc->sum->mismatch++;
169 }
170 }
171 return brix_scan_record_file(v->line, v->linesz, v->logical, size, mtime,
172 cc->alg, stored, computed, status);
173}

175/* Verify one enumerated catalog object: open it through the bound driver, build
176 * the record, close. An open failure is reported as "unreadable" (the scan keeps
177 * going). Returns 0 to continue the enumeration, 1 to abort (OOM only). */
178static int
179scan_catalog_verify_one(void *ctx, const brix_sd_catalog_ent_t *ent)
180{
181 scan_verify_ctx_t *cc = ctx;
182 const char *logical = ent->path ? ent->path : ent->key;
183 brix_sd_obj_t *obj;
184 char line[SCAN_LINE_MAX];
185 int err = 0;
186 int n;
187
188 obj = cc->sd->driver->open(cc->sd, logical, BRIX_SD_O_READ, 0, &err);
189 if (obj == NULL) {
190 cc->sum->unreadable++;
191 n = brix_scan_record_file(line, sizeof(line), logical,
192 ent->have_stat ? (int64_t) ent->size : 0,
193 ent->have_stat ? (int64_t) ent->mtime : 0,
194 cc->alg, NULL, NULL, "unreadable");
195 } else {
196 scan_verify_obj_ctx_t v;
197
198 ngx_memzero(&v, sizeof(v));
199 v.cc = cc;
200 v.logical = logical;
201 v.ent = ent;
202 v.obj = obj;
203 v.line = line;
204 v.linesz = sizeof(line);
205 n = scan_verify_obj_record(&v);
206 (void) cc->sd->driver->close(obj);
207 }
208
209 if (n < 0) {
210 return 0; /* unrepresentable → soft skip */
211 }
212 cc->sum->files++;
213 if (scan_append(cc->buf, cc->cap, cc->used, line, (size_t) n) < 0) {
214 cc->oom = 1;
215 return 1;
216 }
217 return 0;
218}
