8/* Parse the {id}[/action] tail after "/admin/proxy/backends/". */
9ngx_int_t
10admin_parse_proxy_uri(ngx_http_request_t *r, uint32_t *id_out,
11 char *action_out, size_t action_size)
12{
13 static const char pfx[] = ADMIN_PREFIX "proxy/backends/";
14 size_t pfxlen = sizeof(pfx) - 1;
15 char tail[64];
16 size_t taillen;
17 char *slash;
18 ngx_int_t id;
19
20 if (r->uri.len <= pfxlen || ngx_strncmp(r->uri.data, pfx, pfxlen) != 0) {
21 return NGX_ERROR;
22 }
23 taillen = r->uri.len - pfxlen;
24 if (taillen == 0 || taillen >= sizeof(tail)) {
25 return NGX_ERROR;
26 }
27 ngx_memcpy(tail, r->uri.data + pfxlen, taillen);
28 tail[taillen] = '\0';
29
30 /* Tail is "{id}" or "{id}/{action}". If a '/' is present, the part after it
31 * is the action (truncated at any further '/'), and the id is the part
32 * before — NUL the '/' so ngx_atoi below sees only the id. */
33 action_out[0] = '\0';
34 slash = ngx_strchr(tail, '/');
35 if (slash != NULL) {
36 char *act = slash + 1;
37 char *p2 = ngx_strchr(act, '/');
38 if (p2 != NULL) { *p2 = '\0'; } /* drop any trailing slash */
39 *slash = '\0';
40 if (ngx_strlen(act) < action_size) {
41 ngx_cpystrn((u_char *) action_out, (u_char *) act, action_size);
42 }
43 }
44
45 id = ngx_atoi((u_char *) tail, ngx_strlen(tail));
46 if (id == NGX_ERROR || id <= 0) {
47 return NGX_ERROR;
48 }
49 *id_out = (uint32_t) id;
50 return NGX_OK;
51}

54/* Serialise one proxy-backend snapshot entry to JSON (mechanical field mapping;
55 * the only logic is the numeric state -> name translation). */
56json_t *
57admin_proxy_backend_json(const brix_proxy_be_snapshot_t *e)
58{
59 json_t *o = json_object();
60 const char *state;
61
62 if (o == NULL) {
63 return NULL;
64 }
65 switch (e->state) {
66 case BRIX_PROXY_BE_DRAINING: state = "draining"; break;
67 case BRIX_PROXY_BE_DEAD: state = "dead"; break;
68 default: state = "active"; break;
69 }
70 json_object_set_new(o, "id", json_integer(e->id));
71 json_object_set_new(o, "host", json_string(e->host));
72 json_object_set_new(o, "port", json_integer(e->port));
73 json_object_set_new(o, "ssl", e->ssl ? json_true() : json_false());
74 json_object_set_new(o, "weight", json_integer(e->weight));
75 json_object_set_new(o, "state", json_string(state));
76 json_object_set_new(o, "in_flight", json_integer(e->in_flight));
77 return o;
78}

81/*
82 * admin_url_host_allowed — W6/E1 backend-target allowlist.
83 *
84 * A compromised admin credential could otherwise point a dynamic proxy backend
85 * at an arbitrary host (data exfiltration / SSRF pivot). When
86 * brix_admin_proxy_allow is configured, the host parsed out of the backend
87 * URL must match one of the listed hostnames exactly (case-insensitive).
88 * When the directive is absent the function returns 1 (back-compat).
89 */
90int
91admin_url_host_allowed(ngx_http_brix_dashboard_loc_conf_t *conf,
92 const char *url)
93{
94 const char *host, *p;
95 size_t host_len, off, i;
96 ngx_str_t *allow;
97
98 if (conf->admin_proxy_allow == NULL) {
99 return 1;
100 }
101
102 /* admin_validate_url has already confirmed an http(s):// prefix + safe
103 * chars; extract the host (up to ':' or '/'). */
104 off = (ngx_strncmp(url, "https://", 8) == 0) ? 8 : 7;
105 host = url + off;
106 p = host;
107 while (*p != '\0' && *p != ':' && *p != '/') {
108 p++;
109 }
110 host_len = (size_t) (p - host);
111
112 allow = conf->admin_proxy_allow->elts;
113 for (i = 0; i < conf->admin_proxy_allow->nelts; i++) {
114 if (allow[i].len == host_len
115 && ngx_strncasecmp((u_char *) host, allow[i].data, host_len) == 0)
116 {
117 return 1;
118 }
119 }
120 return 0;
121}

124/*
125 * POST body handler: add a dynamic proxy backend from a JSON {url, weight}.
126 * Security gauntlet before mutating the SHM pool: (1) url present, (2) syntactic
127 * whitelist (admin_validate_url), (3) SSRF allowlist (admin_url_host_allowed).
128 * weight is clamped to 1..1000. On success returns 201 with the assigned id.
129 */
130ngx_int_t
131admin_proxy_add(ngx_http_request_t *r, json_t *body)
132{
133 ngx_http_brix_dashboard_loc_conf_t *conf =
134 ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
135 const char *url = json_string_value(json_object_get(body, "url"));
136 json_int_t weight = json_integer_value(json_object_get(body, "weight"));
137 uint32_t id = 0;
138 ngx_int_t rc;
139 char target[64];
140
141 if (url == NULL) {
142 admin_audit(r, "proxy/add", url, "bad_request");
143 return admin_send_error(r, NGX_HTTP_BAD_REQUEST, "missing_field");
144 }
145 if (!admin_validate_url(url)) {
146 admin_audit(r, "proxy/add", url, "invalid_field");
147 return admin_send_error(r, NGX_HTTP_BAD_REQUEST, "invalid_field");
148 }
149 if (!admin_url_host_allowed(conf, url)) {
150 admin_audit(r, "proxy/add", url, "host_not_allowed");
151 return admin_send_error(r, NGX_HTTP_FORBIDDEN, "host_not_allowed");
152 }
153 if (weight <= 0) weight = 1;
154 if (weight > 1000) weight = 1000;
155
156 rc = brix_proxy_pool_add(url, (ngx_uint_t) weight, r->pool,
157 r->connection->log, &id);
158 if (rc == NGX_DECLINED) {
159 admin_audit(r, "proxy/add", url, "not_enabled");
160 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "proxy_pool_disabled");
161 }
162 if (rc != NGX_OK) {
163 admin_audit(r, "proxy/add", url, "add_failed");
164 return admin_send_error(r, NGX_HTTP_BAD_REQUEST, "add_failed");
165 }
166
167 (void) ngx_snprintf((u_char *) target, sizeof(target) - 1, "id=%uD%Z", id);
168 brix_dashboard_event_add(BRIX_DASH_EVENT_NAMESPACE, 0, 0,
169 "admin: proxy backend added", url);
170 admin_audit(r, "proxy/add", url, "added");
171 {
172 json_t *root = json_object();
173 if (root == NULL) {
174 return NGX_HTTP_INTERNAL_SERVER_ERROR;
175 }
176 dashboard_json_set_schema(root);
177 json_object_set_new(root, "result", json_string("added"));
178 json_object_set_new(root, "id", json_integer(id));
179 return dashboard_json_send(r, NGX_HTTP_CREATED, root);
180 }
181}

184/* GET /admin/proxy/backends: snapshot and list all dynamic proxy backends. */
185ngx_int_t
186admin_proxy_list(ngx_http_request_t *r)
187{
188 brix_proxy_be_snapshot_t snap[BRIX_PROXY_POOL_SLOTS];
189 ngx_uint_t i, count;
190 json_t *root, *arr;
191
192 count = brix_proxy_pool_snapshot(snap, BRIX_PROXY_POOL_SLOTS);
193
194 root = json_object();
195 arr = json_array();
196 if (root == NULL || arr == NULL) {
197 if (root) json_decref(root);
198 if (arr) json_decref(arr);
199 return NGX_HTTP_INTERNAL_SERVER_ERROR;
200 }
201 for (i = 0; i < count; i++) {
202 json_t *o = admin_proxy_backend_json(&snap[i]);
203 if (o != NULL) {
204 json_array_append_new(arr, o);
205 }
206 }
207 dashboard_json_set_schema(root);
208 json_object_set_new(root, "backends", arr);
209 return dashboard_json_send(r, NGX_HTTP_OK, root);
210}

213/*
214 * Route a single-backend request "/admin/proxy/backends/{id}[/action]" to the
215 * right pool mutation, enforcing the method per (action x verb) matrix:
216 * /drain, /undrain -> POST only
217 * no action -> DELETE removes, GET reads in-flight count
218 * Anything else is 405/404. `id` and `action` are pre-parsed by the caller.
219 */
220ngx_int_t
221admin_proxy_one(ngx_http_request_t *r, const char *action, uint32_t id)
222{
223 char target[32];
224 (void) ngx_snprintf((u_char *) target, sizeof(target) - 1, "id=%uD%Z", id);
225
226 if (ngx_strcmp(action, "drain") == 0) {
227 if (r->method != NGX_HTTP_POST) {
228 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
229 }
230 if (!brix_proxy_pool_drain(id)) {
231 admin_audit(r, "proxy/drain", target, "not_found");
232 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
233 }
234 admin_audit(r, "proxy/drain", target, "drained");
235 return admin_send_ok(r, "drained");
236 }
237 if (ngx_strcmp(action, "undrain") == 0) {
238 if (r->method != NGX_HTTP_POST) {
239 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
240 }
241 if (!brix_proxy_pool_undrain(id)) {
242 admin_audit(r, "proxy/undrain", target, "not_found");
243 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
244 }
245 admin_audit(r, "proxy/undrain", target, "undrained");
246 return admin_send_ok(r, "undrained");
247 }
248 if (action[0] == '\0') {
249 if (r->method == NGX_HTTP_DELETE) {
250 if (!brix_proxy_pool_remove(id)) {
251 admin_audit(r, "proxy/delete", target, "not_found");
252 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
253 }
254 admin_audit(r, "proxy/delete", target, "removed");
255 return admin_send_ok(r, "removed");
256 }
257 if (r->method == NGX_HTTP_GET) {
258 long inflight = brix_proxy_pool_in_flight(id);
259 json_t *root;
260 if (inflight < 0) {
261 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
262 }
263 root = json_object();
264 if (root == NULL) {
265 return NGX_HTTP_INTERNAL_SERVER_ERROR;
266 }
267 dashboard_json_set_schema(root);
268 json_object_set_new(root, "id", json_integer(id));
269 json_object_set_new(root, "in_flight", json_integer(inflight));
270 return dashboard_json_send(r, NGX_HTTP_OK, root);
271 }
272 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
273 }
274 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
275}

278/* Directive setter for `brix_admin_proxy_allow <host>...`: the backend-target
279 * allowlist consulted by admin_url_host_allowed (SSRF guard). Stores raw host
280 * strings (no parsing); matching is case-insensitive at request time. */
281char *
282brix_admin_set_proxy_allow(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
283{
284 ngx_http_brix_dashboard_loc_conf_t *lcf = conf;
285 ngx_str_t *value = cf->args->elts;
286 ngx_uint_t i;
287
288 (void) cmd;
289
290 if (lcf->admin_proxy_allow == NULL) {
291 lcf->admin_proxy_allow = ngx_array_create(cf->pool,
292 cf->args->nelts - 1,
293 sizeof(ngx_str_t));
294 if (lcf->admin_proxy_allow == NULL) {
295 return NGX_CONF_ERROR;
296 }
297 }
298
299 for (i = 1; i < cf->args->nelts; i++) {
300 ngx_str_t *h = ngx_array_push(lcf->admin_proxy_allow);
301 if (h == NULL) {
302 return NGX_CONF_ERROR;
303 }
304 *h = value[i];
305 }
306 return NGX_CONF_OK;
307}
