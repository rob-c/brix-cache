12/*
13 * WHAT: The resolved, validated state of one COPY request: the confined source
14 * and destination paths, their stat records, and the overwrite/depth flags plus
15 * whether the destination already existed.
16 *
17 * WHY: The COPY handler's pre-flight (parse → resolve → validate) produces a
18 * cluster of related values that the file and collection execute paths both
19 * consume. Threading them as one struct keeps the orchestrator flat and each
20 * pre-flight helper single-purpose, instead of a long function with a dozen
21 * loose locals.
22 *
23 * HOW: Filled incrementally — webdav_copy_parse_request sets the flags and the
24 * decoded destination, webdav_copy_resolve_pair fills the paths/stats/existed —
25 * then read by the execute helpers. Zero-initialised at declaration.
26 */
27typedef struct {
28 char src_path[WEBDAV_MAX_PATH];
29 char dst_path[WEBDAV_MAX_PATH];
30 char dest_decoded[WEBDAV_MAX_PATH];
31 struct stat src_sb;
32 struct stat dst_sb;
33 int dst_existed;

35 int depth_infinity;
36} webdav_copy_req_t;
