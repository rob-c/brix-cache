1/*
2 * namespace_ops.c — shared namespace mutation service.
3 *
4 * WHAT: Implements shared orchestration for filesystem mutations (mkdir, rename,
5 * delete, local copy) using the low-level confined primitives.
6 * WHY: Protocol modules (native root, WebDAV, S3) repeat the same orchestration
7 * around missing-target semantics, recursive-vs-empty directory behavior,
8 * and errno mapping.
9 * HOW: These helpers operate on already-resolved, confined paths. Protocol
10 * handlers are responsible for wire parsing, auth, and lock checks.
11 */

27/*
28 * Confinement model for this file (Phase 8 — openat2 RESOLVE_BENEATH).
29 *
30 * Every path that reaches these helpers is a client-derived path under the
31 * export root. We no longer trust an upstream realpath() to have confined it:
32 * each function opens a kernel-confinement rootfd on root_canon and routes ALL
33 * filesystem touches (stat, open, unlink, mkdir, rename) through the beneath
34 * API, so escape is blocked atomically by the kernel even if the caller passed
35 * a non-canonicalised path. ns_open_rel() centralises the rootfd-open +
36 * within-root strip; a NULL rel means the path is not under root_canon and the
37 * operation is refused (EXDEV) rather than falling through to a raw syscall.
38 *
39 * Where beneath is deliberately NOT used (see the inline notes below):
40 * - brix_copy_range(): operates on already-open file descriptors, not paths.
41 * Confinement happened when those fds were obtained via brix_open_beneath;
42 * a path-anchored RESOLVE_BENEATH check is meaningless on an fd.
43 * - brix_ns_copy_fattrs()/listxattr/getxattr/setxattr: the xattr syscalls are
44 * name+path based and have no openat2/RESOLVE_BENEATH-equivalent that takes a
45 * dirfd. They run on src/dst paths that the immediately-preceding beneath
46 * open already proved resident under the root, so no new escape surface is
47 * introduced; switching them to an fd-based fsetxattr/fgetxattr is a possible
48 * future hardening but is not a confinement gap today.
49 */
50
51/*
52 * ns_rel — open a confinement rootfd on root_canon and return the within-root
53 * relative tail of abspath. *rootfd_out receives the fd (caller closes it);
54 * returns NULL with *rootfd_out=-1 and errno set when the rootfd cannot be
55 * opened or abspath escapes root_canon.
56 */
57static const char *
58ns_rel(const char *root_canon, const char *abspath, int *rootfd_out)
59{
60 const char *rel;
61 int fd;
62
63 *rootfd_out = -1;
64
65 rel = brix_beneath_strip_root(root_canon, abspath);
66 if (rel == NULL) {
67 errno = EXDEV; /* path is not under the export root */
68 return NULL;
69 }
70
71 fd = brix_beneath_open_root(root_canon);
72 if (fd < 0) {
73 return NULL; /* errno from open() */
74 }
75
76 *rootfd_out = fd;
77 return rel;
78}

80static brix_ns_status_t
81errno_to_ns_status(int err)
82{
83 switch (err) {
84 case 0: return BRIX_NS_OK;
85 case ENOENT: return BRIX_NS_NOT_FOUND;
86 case EACCES:
87 case EPERM:
88 case EXDEV: /* openat2 RESOLVE_BENEATH ".." escape (mkdir/rm/rename/copy) */
89 case ELOOP: /* RESOLVE_BENEATH/NO_MAGICLINKS escaping-symlink rejection */
90 return BRIX_NS_DENIED;
91 case EEXIST: return BRIX_NS_EXISTS;
92 case ENOTEMPTY: return BRIX_NS_NOT_EMPTY;
93 case ENAMETOOLONG: return BRIX_NS_TOO_LONG;
94 case ENOSPC: return BRIX_NS_NO_SPACE;
95 case EBUSY:
96 case EINVAL: return BRIX_NS_CONFLICT;
97 default: return BRIX_NS_IO_ERROR;
98 }
99}

101void
102brix_xattr_copy_by_prefix(ngx_log_t *log,
103 const char *src, const char *dst,
104 const char *prefix, size_t prefix_len,
105 size_t value_max)
106{
107 ssize_t list_len, vlen;
108 char *list;
109 char *p;
110 u_char *value;
111
112 list_len = listxattr(src, NULL, 0);
113 if (list_len <= 0) {
114 return;
115 }
116
117 list = ngx_alloc((size_t) list_len, log);
118 if (list == NULL) {
119 return;
120 }
121
122 list_len = listxattr(src, list, (size_t) list_len);
123 if (list_len <= 0) {
124 ngx_free(list);
125 return;
126 }
127
128 value = ngx_alloc(value_max, log);
129 if (value == NULL) {
130 ngx_free(list);
131 return;
132 }
133
134 for (p = list; p < list + list_len; p += strlen(p) + 1) {
135 if (strncmp(p, prefix, prefix_len) != 0) {
136 continue;
137 }
138 vlen = getxattr(src, p, value, value_max);
139 if (vlen >= 0) {
140 (void) setxattr(dst, p, value, (size_t) vlen, 0);
141 }
142 }
143
144 ngx_free(value);
145 ngx_free(list);
146}

156brix_ns_result_t
157brix_ns_delete(ngx_log_t *log, const char *root_canon, const char *path,
158 const brix_ns_delete_opts_t *opts)
159{
160 brix_ns_result_t res;
161 struct stat sb;
162 int rootfd;
163 const char *rel;
164
165 ngx_memzero(&res, sizeof(res));
166
167 rel = ns_rel(root_canon, path, &rootfd);
168 if (rel == NULL) {
169 res.sys_errno = errno;
170 res.status = errno_to_ns_status(errno);
171 return res;
172 }
173
174 /* LSTAT, not stat: delete must operate on the final component itself and must
175 * never dereference a trailing symlink (POSIX unlink/rm semantics). Following it
176 * would (a) misclassify a symlink-to-dir as a directory and (b) fail outright when
177 * the link's stored target is an absolute logical path, which RESOLVE_BENEATH
178 * rejects — leaving the link un-removable. brix_unlink_beneath() likewise unlinks
179 * the name in its parent without following it, so the two agree. */
180 if (brix_lstat_beneath(rootfd, rel, &sb) != 0) {
181 if (errno == ENOENT && opts->idempotent_missing) {
182 res.status = BRIX_NS_OK;
183 close(rootfd);
184 return res;
185 }
186 res.sys_errno = errno;
187 res.status = errno_to_ns_status(errno);
188 close(rootfd);
189 return res;
190 }
191
192 res.existed = 1;
193 res.was_dir = S_ISDIR(sb.st_mode) ? 1 : 0; /* a symlink is never a dir → unlink */
194
195 /* kXR_rmdir is directory-only: reject regular files with ENOTDIR rather
196 * than silently unlinking them (matches rmdir(2) semantics). */
197 if (opts->require_directory && !res.was_dir) {
198 res.sys_errno = ENOTDIR;
199 res.status = errno_to_ns_status(ENOTDIR);
200 close(rootfd);
201 return res;
202 }
203
204 if (res.was_dir && opts->require_empty_dir) {
205 /* dir_is_empty() opens `path` directly; it is reached only after the
206 * stat_beneath above proved the target resident under the root, so the
207 * emptiness probe cannot itself be steered out of the export tree. */
208 ngx_flag_t is_empty;
209 if (brix_fs_dir_is_empty(path, &is_empty) == NGX_OK && !is_empty) {
210 res.status = BRIX_NS_NOT_EMPTY;
211 close(rootfd);
212 return res;
213 }
214 }
215
216 if (opts->recursive && res.was_dir) {
217 /* Recursive tree removal is confined internally by fs_walk (it descends
218 * with O_NOFOLLOW dir fds and unlinks via the beneath/confined API). */
219 if (brix_fs_remove_tree_confined(log, root_canon, path) == NGX_OK) {
220 res.status = BRIX_NS_OK;
221 } else {
222 res.sys_errno = errno;
223 res.status = errno_to_ns_status(errno);
224 }
225 } else {
226 if (brix_unlink_beneath(rootfd, rel, res.was_dir) == 0) {
227 res.status = BRIX_NS_OK;
228 } else {
229 res.sys_errno = errno;
230 res.status = errno_to_ns_status(errno);
231 }
232 }
233
234 close(rootfd);
235 return res;
236}

238brix_ns_result_t
239brix_ns_mkdir(ngx_log_t *log, const char *root_canon, const char *path,
240 mode_t mode, ngx_flag_t recursive)
241{
242 brix_ns_result_t res;
243 int rootfd;
244 const char *rel;
245
246 ngx_memzero(&res, sizeof(res));
247
248 rel = ns_rel(root_canon, path, &rootfd);
249 if (rel == NULL) {
250 res.sys_errno = errno;
251 res.status = errno_to_ns_status(errno);
252 return res;
253 }
254
255 if (recursive) {
256 /* mkdir -p, confined: creates each missing component beneath rootfd.
257 * This variant takes the absolute path + root_canon and strips
258 * internally, so pass `path` (not the pre-stripped rel). */
259 if (brix_mkdir_recursive_beneath(log, rootfd, root_canon, path,
260 mode, NULL) == 0) {
261 res.status = BRIX_NS_OK;
262 res.created = 1;
263 } else {
264 res.sys_errno = errno;
265 res.status = errno_to_ns_status(errno);
266 }
267 } else {
268 if (brix_mkdir_beneath(rootfd, rel, mode) == 0) {
269 res.status = BRIX_NS_OK;
270 res.created = 1;
271 } else {
272 res.sys_errno = errno;
273 res.status = errno_to_ns_status(errno);
274 }
275 }
276
277 close(rootfd);
278 return res;
279}

281brix_ns_result_t
282brix_ns_rename(ngx_log_t *log, const char *root_canon, const char *src,
283 const char *dst, ngx_flag_t overwrite_dirs)
284{
285 brix_ns_result_t res;
286 struct stat sb;
287 int rootfd;
288 const char *src_rel, *dst_rel;
289
290 ngx_memzero(&res, sizeof(res));
291
292 /* Both src and dst must be within the SAME export root (renameat across
293 * roots is refused). Open one rootfd and confine both ends through it. */
294 dst_rel = ns_rel(root_canon, dst, &rootfd);
295 if (dst_rel == NULL) {
296 res.sys_errno = errno;
297 res.status = errno_to_ns_status(errno);
298 return res;
299 }
300 src_rel = brix_beneath_strip_root(root_canon, src);
301 if (src_rel == NULL) {
302 close(rootfd);
303 res.sys_errno = EXDEV;
304 res.status = errno_to_ns_status(EXDEV);
305 return res;
306 }
307
308 if (brix_stat_beneath(rootfd, dst_rel, &sb) == 0) {
309 res.existed = 1;
310 if (S_ISDIR(sb.st_mode)) {
311 res.was_dir = 1;
312 if (overwrite_dirs) {
313 /* WebDAV/S3 semantics: delete target directory tree before rename
314 * if overwrite is requested. rename(2) only replaces empty dirs. */
315 if (brix_fs_remove_tree_confined(log, root_canon, dst) != NGX_OK) {
316 res.sys_errno = errno;
317 res.status = errno_to_ns_status(errno);
318 close(rootfd);
319 return res;
320 }
321 } else {
322 res.status = BRIX_NS_EXISTS;
323 close(rootfd);
324 return res;
325 }
326 }
327 }
328
329 if (brix_rename_beneath(rootfd, src_rel, dst_rel) == 0) {
330 res.status = BRIX_NS_OK;
331 } else {
332 res.sys_errno = errno;
333 res.status = errno_to_ns_status(errno);
334 }
335
336 close(rootfd);
337 return res;
338}

340brix_ns_result_t
341brix_ns_local_copy(ngx_log_t *log, const char *root_canon, const char *src,
342 const char *dst, const brix_ns_copy_opts_t *opts)
343{
344 brix_ns_result_t res;
345 struct stat ssb, dsb;
346 int src_fd = -1, dst_fd = -1;
347 int rootfd;
348 const char *src_rel, *dst_rel;
349 brix_staged_file_t staged;
350 ngx_flag_t use_staging = 0;
351
352 ngx_memzero(&res, sizeof(res));
353
354 src_rel = ns_rel(root_canon, src, &rootfd);
355 if (src_rel == NULL) {
356 res.sys_errno = errno;
357 res.status = errno_to_ns_status(errno);
358 return res;
359 }
360 dst_rel = brix_beneath_strip_root(root_canon, dst);
361 if (dst_rel == NULL) {
362 close(rootfd);
363 res.sys_errno = EXDEV;
364 res.status = errno_to_ns_status(EXDEV);
365 return res;
366 }
367
368 if (brix_stat_beneath(rootfd, src_rel, &ssb) != 0) {
369 res.sys_errno = errno;
370 res.status = errno_to_ns_status(errno);
371 close(rootfd);
372 return res;
373 }
374
375 if (S_ISDIR(ssb.st_mode)) {
376 res.status = BRIX_NS_CONFLICT; /* COPY on collection not yet shared */
377 close(rootfd);
378 return res;
379 }
380
381 if (brix_stat_beneath(rootfd, dst_rel, &dsb) == 0) {
382 res.existed = 1;
383 if (!opts->overwrite) {
384 res.status = BRIX_NS_EXISTS;
385 close(rootfd);
386 return res;
387 }
388 if (S_ISDIR(dsb.st_mode)) {
389 res.was_dir = 1;
390 if (!opts->overwrite_dirs) {
391 res.status = BRIX_NS_EXISTS;
392 close(rootfd);
393 return res;
394 }
395 }
396 }
397
398 src_fd = brix_open_beneath(rootfd, src_rel, O_RDONLY, 0);
399 if (src_fd < 0) {
400 res.sys_errno = errno;
401 res.status = errno_to_ns_status(errno);
402 close(rootfd);
403 return res;
404 }
405
406 if (opts->staged_commit) {
407 /* staged_file confines its own temp open/rename internally. */
408 if (brix_staged_open(log, root_canon, dst,
409 O_WRONLY | O_CREAT | O_TRUNC,
410 ssb.st_mode, 16, &staged) != NGX_OK)
411 {
412 close(src_fd);
413 close(rootfd);
414 res.sys_errno = errno;
415 res.status = errno_to_ns_status(errno);
416 return res;
417 }
418 dst_fd = staged.fd;
419 use_staging = 1;
420 } else {
421 dst_fd = brix_open_beneath(rootfd, dst_rel,
422 O_WRONLY | O_CREAT | O_TRUNC,
423 ssb.st_mode);
424 if (dst_fd < 0) {
425 close(src_fd);
426 close(rootfd);
427 res.sys_errno = errno;
428 res.status = errno_to_ns_status(errno);
429 return res;
430 }
431 }
432
433 /*
434 * NOT a beneath site: brix_copy_range operates on the two file
435 * descriptors obtained above (src_fd, dst_fd). Both fds were opened
436 * through brix_open_beneath / staged_file, so confinement is already
437 * established; copy_file_range(2)/pread/pwrite act on fds, not paths, and
438 * have no RESOLVE_BENEATH concept. Re-confining here is neither possible
439 * nor meaningful.
440 */
441 if (brix_copy_range(log, src_fd, 0, dst_fd, 0, ssb.st_size, src,
442 use_staging ? staged.tmp_path : dst) != NGX_OK)
443 {
444 res.sys_errno = errno;
445 res.status = errno_to_ns_status(errno);
446 close(src_fd);
447 if (use_staging) {
448 brix_staged_abort(log, root_canon, &staged, 1);
449 } else {
450 close(dst_fd);
451 }
452 close(rootfd);
453 return res;
454 }
455
456 close(src_fd);
457
458 if (use_staging) {
459 if (brix_staged_commit(log, root_canon, &staged, dst) != NGX_OK) {
460 res.sys_errno = errno;
461 res.status = errno_to_ns_status(errno);
462 close(rootfd);
463 return res;
464 }
465 } else {
466 close(dst_fd);
467 }
468
469 /*
470 * NOT a beneath site: xattr copy is name+path based (listxattr/getxattr/
471 * setxattr on src/dst). There is no openat2/RESOLVE_BENEATH-equivalent for
472 * extended attributes that takes a dirfd, and src/dst were just proven
473 * resident under the root by the beneath opens above, so this introduces no
474 * new escape surface. (A future hardening could switch to fd-based
475 * fgetxattr/fsetxattr; tracked, not a confinement gap.)
476 */
477 if (opts->preserve_xattrs) {
478 brix_ns_copy_fattrs(log, src, dst);
479 }
480
481 close(rootfd);
482 res.status = BRIX_NS_OK;
483 res.created = 1;
484 return res;
485}
