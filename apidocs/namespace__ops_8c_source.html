1/*
2 * namespace_ops.c — shared namespace mutation service.
3 *
4 * WHAT: Implements shared orchestration for filesystem mutations (mkdir, rename,
5 * delete, local copy) using the low-level confined primitives.
6 * WHY: Protocol modules (native root, WebDAV, S3) repeat the same orchestration
7 * around missing-target semantics, recursive-vs-empty directory behavior,
8 * and errno mapping.
9 * HOW: These helpers operate on already-resolved, confined paths. Protocol
10 * handlers are responsible for wire parsing, auth, and lock checks.
11 */

24/*
25 * Confinement model for this file (Phase 8 — openat2 RESOLVE_BENEATH).
26 *
27 * Every path that reaches these helpers is a client-derived path under the
28 * export root. We no longer trust an upstream realpath() to have confined it:
29 * each function opens a kernel-confinement rootfd on root_canon and routes ALL
30 * filesystem touches (stat, open, unlink, mkdir, rename) through the beneath
31 * API, so escape is blocked atomically by the kernel even if the caller passed
32 * a non-canonicalised path. ns_rel() centralises the rootfd-open +
33 * within-root strip; a NULL rel means the path is not under root_canon and the
34 * operation is refused (EXDEV) rather than falling through to a raw syscall.
35 *
36 * The single-file local-copy path (which additionally reasons about fd-based and
37 * xattr-based confinement) lives in namespace_ops_copy.c; it reuses ns_rel() and
38 * ns_set_err() from this file through namespace_ops_internal.h.
39 */
40
41/*
42 * ns_rel — open a confinement rootfd on root_canon and return the within-root
43 * relative tail of abspath. *rootfd_out receives the fd (caller closes it);
44 * returns NULL with *rootfd_out=-1 and errno set when the rootfd cannot be
45 * opened or abspath escapes root_canon.
46 */
47const char *
48ns_rel(const char *root_canon, const char *abspath, int *rootfd_out)
49{
50 const char *rel;
51 int fd;
52
53 *rootfd_out = -1;
54
55 rel = brix_beneath_strip_root(root_canon, abspath);
56 if (rel == NULL) {
57 errno = EXDEV; /* path is not under the export root */
58 return NULL;
59 }
60
61 fd = brix_beneath_open_root(root_canon);
62 if (fd < 0) {
63 return NULL; /* errno from open() */
64 }
65
66 *rootfd_out = fd;
67 return rel;
68}

70static brix_ns_status_t
71errno_to_ns_status(int err)
72{
73 switch (err) {
74 case 0: return BRIX_NS_OK;
75 case ENOENT: return BRIX_NS_NOT_FOUND;
76 case EACCES:
77 case EPERM:
78 case EXDEV: /* openat2 RESOLVE_BENEATH ".." escape (mkdir/rm/rename/copy) */
79 case ELOOP: /* RESOLVE_BENEATH/NO_MAGICLINKS escaping-symlink rejection */
80 return BRIX_NS_DENIED;
81 case EEXIST: return BRIX_NS_EXISTS;
82 case ENOTEMPTY: return BRIX_NS_NOT_EMPTY;
83 case ENAMETOOLONG: return BRIX_NS_TOO_LONG;
84 case ENOSPC: return BRIX_NS_NO_SPACE;
85 case EBUSY:
86 case EINVAL: return BRIX_NS_CONFLICT;
87 default: return BRIX_NS_IO_ERROR;
88 }
89}

91/*
92 * ns_set_err — record a failed syscall's errno on the result.
93 *
94 * WHAT: Stores err in res->sys_errno and the mapped BRIX_NS_* code in
95 * res->status. Returns nothing.
96 * WHY: Every error path in these orchestrators sets exactly this pair; one
97 * helper keeps the mapping in a single place and shortens each error return
98 * to one call while preserving the original two-field assignment behavior.
99 * HOW: 1. Copy err into sys_errno. 2. Translate err via errno_to_ns_status().
100 */
101void
102ns_set_err(brix_ns_result_t *res, int err)
103{
104 res->sys_errno = err;
105 res->status = errno_to_ns_status(err);
106}

108/*
109 * ns_delete_probe — lstat the delete target and classify it.
110 *
111 * WHAT: Confines an lstat on rel beneath rootfd, sets res->existed/was_dir on
112 * success, and enforces the require_directory gate. Returns 1 when the
113 * caller must return res immediately (missing-idempotent success, stat
114 * error, or ENOTDIR rejection), 0 when removal should proceed.
115 * WHY: Splits the classify/reject decision out of the delete orchestrator so the
116 * orchestrator stays a flat sequence and each half stays under the
117 * complexity cap, without changing any syscall or errno mapping.
118 * HOW: 1. lstat_beneath; on failure honor idempotent_missing for ENOENT else
119 * record the errno. 2. Mark existed and derive was_dir from the mode.
120 * 3. Reject a non-directory when require_directory is set (ENOTDIR).
121 */
122static ngx_flag_t
123ns_delete_probe(int rootfd, const char *rel,
124 const brix_ns_delete_opts_t *opts, brix_ns_result_t *res)
125{
126 struct stat sb;
127
128 /* LSTAT, not stat: delete must operate on the final component itself and must
129 * never dereference a trailing symlink (POSIX unlink/rm semantics). Following it
130 * would (a) misclassify a symlink-to-dir as a directory and (b) fail outright when
131 * the link's stored target is an absolute logical path, which RESOLVE_BENEATH
132 * rejects — leaving the link un-removable. brix_unlink_beneath() likewise unlinks
133 * the name in its parent without following it, so the two agree. */
134 if (brix_lstat_beneath(rootfd, rel, &sb) != 0) {
135 if (errno == ENOENT && opts->idempotent_missing) {
136 res->status = BRIX_NS_OK;
137 return 1;
138 }
139 ns_set_err(res, errno);
140 return 1;
141 }
142
143 res->existed = 1;
144 res->was_dir = S_ISDIR(sb.st_mode) ? 1 : 0; /* a symlink is never a dir → unlink */
145
146 /* kXR_rmdir is directory-only: reject regular files with ENOTDIR rather
147 * than silently unlinking them (matches rmdir(2) semantics). */
148 if (opts->require_directory && !res->was_dir) {
149 ns_set_err(res, ENOTDIR);
150 return 1;
151 }
152
153 return 0;
154}

156/*
157 * ns_delete_remove — enforce the emptiness gate and perform the removal.
158 *
159 * WHAT: For an already-classified target, optionally rejects a non-empty
160 * directory (require_empty_dir), then removes it via a confined recursive
161 * tree walk (recursive dir) or a single confined unlink, filling res.
162 * WHY: Isolates the removal decision tree from the orchestrator so both stay
163 * simple; the confinement primitives and their ordering are unchanged.
164 * HOW: 1. If a directory must be empty, probe emptiness and stop with
165 * BRIX_NS_NOT_EMPTY when it is not. 2. Recursive directory → confined
166 * remove_tree; otherwise → confined unlink. 3. Map any errno onto res.
167 */
168static void
169ns_delete_remove(ngx_log_t *log, const char *root_canon, const char *path,
170 int rootfd, const char *rel, const brix_ns_delete_opts_t *opts,
171 brix_ns_result_t *res)
172{
173 if (res->was_dir && opts->require_empty_dir) {
174 /* dir_is_empty() opens `path` directly; it is reached only after the
175 * stat_beneath above proved the target resident under the root, so the
176 * emptiness probe cannot itself be steered out of the export tree. */
177 ngx_flag_t is_empty;
178 if (brix_fs_dir_is_empty(path, &is_empty) == NGX_OK && !is_empty) {
179 res->status = BRIX_NS_NOT_EMPTY;
180 return;
181 }
182 }
183
184 if (opts->recursive && res->was_dir) {
185 /* Recursive tree removal is confined internally by fs_walk (it descends
186 * with O_NOFOLLOW dir fds and unlinks via the beneath/confined API). */
187 if (brix_fs_remove_tree_confined(log, root_canon, path) == NGX_OK) {
188 res->status = BRIX_NS_OK;
189 } else {
190 ns_set_err(res, errno);
191 }
192 } else {
193 if (brix_unlink_beneath(rootfd, rel, res->was_dir) == 0) {
194 res->status = BRIX_NS_OK;
195 } else {
196 ns_set_err(res, errno);
197 }
198 }
199}

201brix_ns_result_t
202brix_ns_delete(ngx_log_t *log, const char *root_canon, const char *path,
203 const brix_ns_delete_opts_t *opts)
204{
205 brix_ns_result_t res;
206 int rootfd;
207 const char *rel;
208
209 ngx_memzero(&res, sizeof(res));
210
211 rel = ns_rel(root_canon, path, &rootfd);
212 if (rel == NULL) {
213 ns_set_err(&res, errno);
214 return res;
215 }
216
217 if (ns_delete_probe(rootfd, rel, opts, &res)) {
218 close(rootfd);
219 return res;
220 }
221
222 ns_delete_remove(log, root_canon, path, rootfd, rel, opts, &res);
223
224 close(rootfd);
225 return res;
226}

228brix_ns_result_t
229brix_ns_mkdir(ngx_log_t *log, const char *root_canon, const char *path,
230 mode_t mode, ngx_flag_t recursive)
231{
232 brix_ns_result_t res;
233 int rootfd;
234 const char *rel;
235
236 ngx_memzero(&res, sizeof(res));
237
238 rel = ns_rel(root_canon, path, &rootfd);
239 if (rel == NULL) {
240 res.sys_errno = errno;
241 res.status = errno_to_ns_status(errno);
242 return res;
243 }
244
245 if (recursive) {
246 /* mkdir -p, confined: creates each missing component beneath rootfd.
247 * This variant takes the absolute path + root_canon and strips
248 * internally, so pass `path` (not the pre-stripped rel). */
249 if (brix_mkdir_recursive_beneath(log, rootfd, root_canon, path,
250 mode, NULL) == 0) {
251 res.status = BRIX_NS_OK;
252 res.created = 1;
253 } else {
254 res.sys_errno = errno;
255 res.status = errno_to_ns_status(errno);
256 }
257 } else {
258 if (brix_mkdir_beneath(rootfd, rel, mode) == 0) {
259 res.status = BRIX_NS_OK;
260 res.created = 1;
261 } else {
262 res.sys_errno = errno;
263 res.status = errno_to_ns_status(errno);
264 }
265 }
266
267 close(rootfd);
268 return res;
269}

271brix_ns_result_t
272brix_ns_rename(ngx_log_t *log, const char *root_canon, const char *src,
273 const char *dst, ngx_flag_t overwrite_dirs)
274{
275 brix_ns_result_t res;
276 struct stat sb;
277 int rootfd;
278 const char *src_rel, *dst_rel;
279
280 ngx_memzero(&res, sizeof(res));
281
282 /* Both src and dst must be within the SAME export root (renameat across
283 * roots is refused). Open one rootfd and confine both ends through it. */
284 dst_rel = ns_rel(root_canon, dst, &rootfd);
285 if (dst_rel == NULL) {
286 res.sys_errno = errno;
287 res.status = errno_to_ns_status(errno);
288 return res;
289 }
290 src_rel = brix_beneath_strip_root(root_canon, src);
291 if (src_rel == NULL) {
292 close(rootfd);
293 res.sys_errno = EXDEV;
294 res.status = errno_to_ns_status(EXDEV);
295 return res;
296 }
297
298 if (brix_stat_beneath(rootfd, dst_rel, &sb) == 0) {
299 res.existed = 1;
300 if (S_ISDIR(sb.st_mode)) {
301 res.was_dir = 1;
302 if (overwrite_dirs) {
303 /* WebDAV/S3 semantics: delete target directory tree before rename
304 * if overwrite is requested. rename(2) only replaces empty dirs. */
305 if (brix_fs_remove_tree_confined(log, root_canon, dst) != NGX_OK) {
306 res.sys_errno = errno;
307 res.status = errno_to_ns_status(errno);
308 close(rootfd);
309 return res;
310 }
311 } else {
312 res.status = BRIX_NS_EXISTS;
313 close(rootfd);
314 return res;
315 }
316 }
317 }
318
319 if (brix_rename_beneath(rootfd, src_rel, dst_rel) == 0) {
320 res.status = BRIX_NS_OK;
321 } else {
322 res.sys_errno = errno;
323 res.status = errno_to_ns_status(errno);
324 }
325
326 close(rootfd);
327 return res;
328}
