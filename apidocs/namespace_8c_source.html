61/*
62 * webdav_handle_delete — handle HTTP DELETE: remove a file or directory.
63 *
64 * RFC 4918 §9.6.1: DELETE on a collection MUST recursively delete all its
65 * members and all their properties.
66 *
67 * The fd-cache entry for the path is evicted before the delete to prevent
68 * use-after-free on cached file descriptors.
69 */
70ngx_int_t
71webdav_handle_delete(ngx_http_request_t *r)
72{
73 char path[WEBDAV_MAX_PATH];
74 struct stat sb;
75 ngx_int_t rc;
76 brix_vfs_ctx_t vctx;
77
78 rc = webdav_resolve_stat(r, path, sizeof(path), &sb);
79 if (rc != NGX_OK) {
80 return rc;
81 }
82
83 rc = webdav_check_locks_tree(r, path);
84 if (rc != NGX_OK) {
85 return rc;
86 }
87
88 /* Route the delete through the metered VFS surface. webdav_resolve_stat
89 * lstat'd the target (vfs_stat does not follow symlinks), so S_ISDIR here
90 * agrees with brix_ns_delete's own lstat dispatch: a directory goes to
91 * rmdir (non-recursive => require-empty, the Standard WebDAV module policy);
92 * a file or symlink goes to unlink. DELETE is already allow_write-gated at
93 * the access phase, so the VFS write-gate never fires here. */
94 webdav_ns_vfs_ctx_init(r, path, &vctx);
95
96 if (S_ISDIR(sb.st_mode)) {
97 rc = brix_vfs_rmdir(&vctx, 0);
98 } else {
99 rc = brix_vfs_unlink(&vctx);
100 }
101
102 if (rc == NGX_OK) {
103 return webdav_send_no_body(r, NGX_HTTP_NO_CONTENT);
104 }
105
106 /* Status mapping: ENOTEMPTY -> 409, ENOENT -> 404, EACCES -> 403 (deny-mode
107 * credential gate rejection from per-user backend cred gate), else 500. */
108 if (errno == ENOTEMPTY) {
109 return NGX_HTTP_CONFLICT;
110 }
111
112 if (errno == ENOENT) {
113 return NGX_HTTP_NOT_FOUND;
114 }
115
116 if (errno == EACCES) {
117 return NGX_HTTP_FORBIDDEN;
118 }
119
120 return NGX_HTTP_INTERNAL_SERVER_ERROR;
121}

123ngx_int_t
124webdav_handle_mkcol(ngx_http_request_t *r)
125{
126 ngx_http_brix_webdav_loc_conf_t *conf;
127 char path[WEBDAV_MAX_PATH];
128 ngx_int_t rc;
129 brix_vfs_ctx_t vctx;
130
131 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
132
133 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon, path,
134 sizeof(path));
135 if (rc == (ngx_int_t) NGX_HTTP_NOT_FOUND) {
136 return NGX_HTTP_CONFLICT;
137 }
138 if (rc != NGX_OK) {
139 return rc;
140 }
141
142 rc = webdav_check_locks(r, path, 1);
143 if (rc != NGX_OK) {
144 return rc;
145 }
146
147 /* Route MKCOL through the metered VFS surface (non-recursive: a missing
148 * parent component is a 409, matching the prior BRIX_NS_NOT_FOUND mapping).
149 * MKCOL is allow_write-gated at the access phase, so the VFS write-gate never
150 * fires here. errno after a failed vfs_mkdir mirrors brix_ns_mkdir:
151 * EEXIST (target present) -> 405, ENOENT (parent missing) -> 409. */
152 webdav_ns_vfs_ctx_init(r, path, &vctx);
153
154 if (brix_vfs_mkdir(&vctx, 0755, 0 /* no parents */) == NGX_OK) {
155 return webdav_send_no_body(r, NGX_HTTP_CREATED);
156 }
157
158 if (errno == EEXIST) {
159 return NGX_HTTP_NOT_ALLOWED;
160 }
161
162 if (errno == ENOENT) {
163 return NGX_HTTP_CONFLICT;
164 }
165
166 if (errno == EACCES) {
167 return NGX_HTTP_FORBIDDEN;
168 }
169
170 return NGX_HTTP_INTERNAL_SERVER_ERROR;
171}
