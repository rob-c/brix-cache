1/*
2 * vfs_copy.c — VFS single-file copy (server-side COPY / CopyObject).
3 *
4 * WHAT: Implements brix_vfs_copy() — copy the resolved ctx (source) regular
5 * file to a destination path within the same export root, the data-mover
6 * behind WebDAV COPY and S3 CopyObject.
7 *
8 * WHY: The copy engines reached brix_ns_local_copy() (copy_file_range with a
9 * read/write fallback) directly, so the bytes moved were confined but
10 * unmetered. Routing through here books one BRIX_METRIC_OP_COPY metric +
11 * access-log line per copy, with the byte count taken from the resulting
12 * destination size, while still delegating the confined data move to the
13 * namespace layer.
14 *
15 * HOW: brix_vfs_copy() write-gates (brix_vfs_require_write — a copy creates
16 * a new object), translates the public brix_vfs_copy_opts_t into an
17 * brix_ns_copy_opts_t, and calls brix_ns_local_copy(ctx->root_canon,
18 * src=ctx path, dst). The namespace status is mapped back to errno and
19 * observed as OP_COPY; bytes are best-effort from a post-copy lstat of the
20 * destination (0 if that stat fails — it never affects the return value).
21 */
22#include "vfs_internal.h"

24/*
25 * brix_vfs_copy_fail — book a failed OP_COPY observation and return NGX_ERROR.
26 *
27 * WHAT: Emit one BRIX_METRIC_OP_COPY error observation (zero bytes) for `src`
28 * with the caller-chosen errno, set errno to that value, and hand back
29 * NGX_ERROR so callers can `return brix_vfs_copy_fail(...)`.
30 * WHY: Every failure exit in brix_vfs_copy repeated the same errno-set +
31 * observe + return triad; centralising it keeps the mapping and the
32 * metric label identical across all error paths.
33 * HOW: Assign errno from `err`, forward it as the observed status errno with
34 * NGX_ERROR, and return NGX_ERROR.
35 */
36static ngx_int_t
37brix_vfs_copy_fail(brix_vfs_ctx_t *ctx, const char *src, int err,
38 uint64_t start)
39{
40 errno = err;
41 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL, 0,
42 NGX_ERROR, err, start);
43 return NGX_ERROR;
44}

46/*
47 * brix_vfs_copy_driver — non-POSIX backend server-side copy path.
48 *
49 * WHAT: Copy the export-relative source `src` to `dst_resolved` through the
50 * driver's server-copy slot, enforcing the overwrite gate via a pre-stat
51 * and forwarding a per-open credential when the credential gate is armed.
52 * WHY: Object/S3-style backends replace bytes in-store, so the confined POSIX
53 * copy_file_range path does not apply; this branch dispatches on the leaf
54 * instance so *_maybe_cred forwarders reach the leaf driver's *_cred
55 * slots (decorators carry only plain relays).
56 * HOW: Re-fetch the leaf driver (non-NULL by the caller's guard), zero the
57 * cred (the gate fills only the active kind), resolve the credential,
58 * reject an existing destination when overwrite is unset, then call the
59 * server_copy slot (ENOTSUP when absent) and book the OP_COPY observation
60 * with the copied byte count.
61 */
62static ngx_int_t
63brix_vfs_copy_driver(brix_vfs_ctx_t *ctx, const char *src,
64 const char *dst_resolved, const brix_vfs_copy_opts_t *opts,
65 uint64_t start)
66{
67 const brix_sd_driver_t *drv = brix_vfs_ctx_driver(ctx);
68 const char *s = brix_vfs_export_relative(ctx, src);
69 const char *d = brix_vfs_export_relative(ctx, dst_resolved);
70 brix_sd_instance_t *leaf = brix_vfs_ns_leaf(ctx->sd);
71 brix_sd_ucred_t store;
72 brix_sd_cred_t cred;
73 brix_sd_stat_t dst_st;
74 off_t copied = 0;
75 ngx_int_t rc;
76 int saved_errno;
77 int use_cred = 0, cred_err = 0;
78
79 /* The sole caller checks the driver before dispatching here, but the
80 * accessor legitimately returns NULL for the default POSIX driver — keep
81 * the invariant local so a future caller cannot hand us a NULL vtable. */
82 if (drv == NULL) {
83 return brix_vfs_copy_fail(ctx, src, ENOTSUP, start);
84 }
85
86 /* Zero before the gate: it fills only the active credential kind; an
87 * unzeroed cred hands a garbage inactive pointer to the driver's
88 * cred slot (bearer PASSTHROUGH would leave x509_proxy dangling). */
89 ngx_memzero(&cred, sizeof(cred));
90
91 if (brix_vfs_cred_gate_active(ctx)) {
92 if (brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err)
93 != NGX_OK)
94 {
95 return brix_vfs_copy_fail(ctx, src, cred_err ? cred_err : EACCES,
96 start);
97 }
98 }
99
100 if (opts != NULL && !opts->overwrite && drv->stat != NULL
101 && brix_sd_stat_maybe_cred(leaf, d, &dst_st,
102 use_cred ? &cred : NULL) == NGX_OK)
103 {
104 brix_sd_ucred_wipe(&store); /* secret consumed; erase (A-4/T4) */
105 return brix_vfs_copy_fail(ctx, src, EEXIST, start);
106 }
107
108 rc = (drv->server_copy != NULL)
109 ? brix_sd_server_copy_maybe_cred(leaf, s, d, &copied,
110 use_cred ? &cred : NULL)
111 : (errno = ENOTSUP, NGX_ERROR);
112 brix_sd_ucred_wipe(&store); /* secret consumed by copy; erase (A-4/T4) */
113 saved_errno = (rc == NGX_OK) ? 0 : errno;
114 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL,
115 rc == NGX_OK ? (size_t) copied : 0, rc,
116 saved_errno, start);
117 return rc;
118}

120/*
121 * brix_vfs_copy_ns_bytes — best-effort destination size for the metric.
122 *
123 * WHAT: Return the size in bytes of the just-copied regular file at
124 * `dst_resolved`, or 0 when it cannot be confined-stat'd or is not
125 * a regular file.
126 * WHY: The OP_COPY metric byte count is taken from the resulting destination;
127 * a stat failure here must never affect the copy's success return value.
128 * HOW: Confined lstat under root_canon; on success for a regular file, return
129 * st_size, else 0.
130 */
131static size_t
132brix_vfs_copy_ns_bytes(brix_vfs_ctx_t *ctx, const char *dst_resolved)
133{
134 struct stat sb;
135
136 if (brix_lstat_confined_canon(ctx->log, ctx->root_canon, dst_resolved,
137 &sb, 1) == 0
138 && S_ISREG(sb.st_mode))
139 {
140 return (size_t) sb.st_size;
141 }
142 return 0;
143}

145/*
146 * brix_vfs_copy_ns — POSIX namespace copy path (copy_file_range + fallback).
147 *
148 * WHAT: Translate the public copy opts into brix_ns_copy_opts_t and run the
149 * confined brix_ns_local_copy (copy_file_range with a pread/pwrite
150 * fallback) from `src` to `dst_resolved`, booking OP_COPY on completion.
151 * WHY: This is the default POSIX-backend data mover; it must preserve the
152 * exact namespace status→errno mapping and the best-effort byte count.
153 * HOW: Zero-init and populate ns_opts, call brix_ns_local_copy, map a non-OK
154 * status to errno (sys_errno when set, else the status mapping) via the
155 * shared failure helper, and on success observe with the post-copy size.
156 */
157static ngx_int_t
158brix_vfs_copy_ns(brix_vfs_ctx_t *ctx, const char *src,
159 const char *dst_resolved, const brix_vfs_copy_opts_t *opts,
160 uint64_t start)
161{
162 brix_ns_copy_opts_t ns_opts;
163 brix_ns_result_t res;
164 size_t bytes;
165
166 ngx_memzero(&ns_opts, sizeof(ns_opts));
167 if (opts != NULL) {
168 ns_opts.recursive = opts->recursive ? 1 : 0;
169 ns_opts.overwrite = opts->overwrite ? 1 : 0;
170 ns_opts.overwrite_dirs = opts->overwrite_dirs ? 1 : 0;
171 ns_opts.preserve_xattrs = opts->preserve_xattrs ? 1 : 0;
172 ns_opts.staged_commit = opts->staged_commit ? 1 : 0;
173 }
174
175 res = brix_ns_local_copy(ctx->log, ctx->root_canon, src, dst_resolved,
176 &ns_opts);
177 if (res.status != BRIX_NS_OK) {
178 return brix_vfs_copy_fail(ctx, src,
179 res.sys_errno != 0 ? res.sys_errno
180 : brix_vfs_ns_status_errno(res.status),
181 start);
182 }
183
184 /* Best-effort byte count for the metric; a stat failure here is non-fatal. */
185 bytes = brix_vfs_copy_ns_bytes(ctx, dst_resolved);
186 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL, bytes,
187 NGX_OK, 0, start);
188 return NGX_OK;
189}

191/* Copy the resolved ctx source file to dst_resolved (both under ctx->root_canon).
192 * Returns NGX_OK, or NGX_ERROR with errno set (mapped from the namespace status:
193 * EEXIST when the destination exists and overwrite is unset, EXDEV on a confine-
194 * ment escape, EISDIR/ECONFLICT on a directory source). Metered as OP_COPY. */
195ngx_int_t
196brix_vfs_copy(brix_vfs_ctx_t *ctx, const char *dst_resolved,
197 const brix_vfs_copy_opts_t *opts)
198{
199 const char *src = brix_vfs_ctx_path(ctx);
200 uint64_t start = brix_vfs_now_ns();
201 const brix_sd_driver_t *drv;
202
203 if (brix_vfs_require_write(ctx) != NGX_OK) {
204 return brix_vfs_copy_fail(ctx, src, errno, start);
205 }
206
207 if (ctx->root_canon == NULL || dst_resolved == NULL) {
208 return brix_vfs_copy_fail(ctx, src, EINVAL, start);
209 }
210
211 /* Non-POSIX backend: copy through the driver's server-copy slot. */
212 drv = brix_vfs_ctx_driver(ctx);
213 if (drv != NULL) {
214 return brix_vfs_copy_driver(ctx, src, dst_resolved, opts, start);
215 }
216
217 return brix_vfs_copy_ns(ctx, src, dst_resolved, opts, start);
218}
