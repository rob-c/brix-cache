1/*
2 * vfs_copy.c — VFS single-file copy (server-side COPY / CopyObject).
3 *
4 * WHAT: Implements brix_vfs_copy() — copy the resolved ctx (source) regular
5 * file to a destination path within the same export root, the data-mover
6 * behind WebDAV COPY and S3 CopyObject.
7 *
8 * WHY: The copy engines reached brix_ns_local_copy() (copy_file_range with a
9 * read/write fallback) directly, so the bytes moved were confined but
10 * unmetered. Routing through here books one BRIX_METRIC_OP_COPY metric +
11 * access-log line per copy, with the byte count taken from the resulting
12 * destination size, while still delegating the confined data move to the
13 * namespace layer.
14 *
15 * HOW: brix_vfs_copy() write-gates (brix_vfs_require_write — a copy creates
16 * a new object), translates the public brix_vfs_copy_opts_t into an
17 * brix_ns_copy_opts_t, and calls brix_ns_local_copy(ctx->root_canon,
18 * src=ctx path, dst). The namespace status is mapped back to errno and
19 * observed as OP_COPY; bytes are best-effort from a post-copy lstat of the
20 * destination (0 if that stat fails — it never affects the return value).
21 */
22#include "vfs_internal.h"
23
24/* Copy the resolved ctx source file to dst_resolved (both under ctx->root_canon).
25 * Returns NGX_OK, or NGX_ERROR with errno set (mapped from the namespace status:
26 * EEXIST when the destination exists and overwrite is unset, EXDEV on a confine-
27 * ment escape, EISDIR/ECONFLICT on a directory source). Metered as OP_COPY. */
28ngx_int_t
29brix_vfs_copy(brix_vfs_ctx_t *ctx, const char *dst_resolved,
30 const brix_vfs_copy_opts_t *opts)
31{
32 brix_ns_copy_opts_t ns_opts;
33 brix_ns_result_t res;
34 const char *src = brix_vfs_ctx_path(ctx);
35 uint64_t start = brix_vfs_now_ns();
36 size_t bytes = 0;
37 struct stat sb;
38 int saved_errno;
39 const brix_sd_driver_t *drv;
40
41 if (brix_vfs_require_write(ctx) != NGX_OK) {
42 saved_errno = errno;
43 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL, 0,
44 NGX_ERROR, saved_errno, start);
45 return NGX_ERROR;
46 }
47
48 if (ctx->root_canon == NULL || dst_resolved == NULL) {
49 errno = EINVAL;
50 saved_errno = errno;
51 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL, 0,
52 NGX_ERROR, saved_errno, start);
53 return NGX_ERROR;
54 }
55
56 /* Non-POSIX backend: copy through the driver's server-copy slot. The opts'
57 * overwrite gate is enforced here via a pre-stat (the slot itself replaces). */
58 drv = brix_vfs_ctx_driver(ctx);
59 if (drv != NULL) {
60 const char *s = brix_vfs_export_relative(ctx, src);
61 const char *d = brix_vfs_export_relative(ctx, dst_resolved);
62 off_t copied = 0;
63 ngx_int_t rc;
64 brix_sd_stat_t dst_st;
65
66 if (opts != NULL && !opts->overwrite && drv->stat != NULL
67 && drv->stat(ctx->sd, d, &dst_st) == NGX_OK)
68 {
69 errno = EEXIST;
70 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL, 0,
71 NGX_ERROR, EEXIST, start);
72 return NGX_ERROR;
73 }
74
75 rc = (drv->server_copy != NULL)
76 ? drv->server_copy(ctx->sd, s, d, &copied)
77 : (errno = ENOSYS, NGX_ERROR);
78 saved_errno = (rc == NGX_OK) ? 0 : errno;
79 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL,
80 rc == NGX_OK ? (size_t) copied : 0, rc,
81 saved_errno, start);
82 return rc;
83 }
84
85 ngx_memzero(&ns_opts, sizeof(ns_opts));
86 if (opts != NULL) {
87 ns_opts.recursive = opts->recursive ? 1 : 0;
88 ns_opts.overwrite = opts->overwrite ? 1 : 0;
89 ns_opts.overwrite_dirs = opts->overwrite_dirs ? 1 : 0;
90 ns_opts.preserve_xattrs = opts->preserve_xattrs ? 1 : 0;
91 ns_opts.staged_commit = opts->staged_commit ? 1 : 0;
92 }
93
94 res = brix_ns_local_copy(ctx->log, ctx->root_canon, src, dst_resolved,
95 &ns_opts);
96 if (res.status != BRIX_NS_OK) {
97 errno = res.sys_errno != 0 ? res.sys_errno
98 : brix_vfs_ns_status_errno(res.status);
99 saved_errno = errno;
100 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL, 0,
101 NGX_ERROR, saved_errno, start);
102 return NGX_ERROR;
103 }
104
105 /* Best-effort byte count for the metric; a stat failure here is non-fatal. */
106 if (brix_lstat_confined_canon(ctx->log, ctx->root_canon, dst_resolved,
107 &sb, 1) == 0
108 && S_ISREG(sb.st_mode))
109 {
110 bytes = (size_t) sb.st_size;
111 }
112
113 brix_vfs_observe_ctx_op(ctx, src, BRIX_METRIC_OP_COPY, NULL, bytes,
114 NGX_OK, 0, start);
115 return NGX_OK;
116}
