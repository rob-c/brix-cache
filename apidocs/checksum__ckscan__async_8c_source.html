10/*
11 * WHAT: Async kXR_Qckscan — walk a file or directory tree off the event loop, compute checksums
12 * (adler32/crc32c/md5/sha1/sha256) for every regular file encountered, and return one line
13 * per file. Runs in an nginx thread pool worker; results delivered via aio_done callback.
14 *
15 * WHY: ckscan walks potentially large directory trees (deep hierarchies with thousands of files).
16 * Doing this on the event loop blocks all other connections. The async path uses ngx_thread_pool_run
17 * to execute the walk in a background worker thread, keeping nginx responsive while scanning completes.
18 * Results are formatted as one checksum line per file matching xrdadler32 output format for client compatibility.
19 *
20 * HOW: brix_ckscan_aio_thread() allocates response buffer (grows via append), stat's the scan target — if regular
21 * file, opens confined/canonical fd and computes single checksum; if directory, walks tree with depth/file limits
22 * calling ckscan_append per file. On error sets t->error_code + t->error_msg. Thread completion triggers
23 * brix_ckscan_aio_done() which restores the request streamid via aio_restore_request, sends error response or
24 * ok+checksum data to client, frees buffer, and resumes the client connection event loop.
25 */

28/* public API: brix_ckscan_aio_thread() — async ckscan thread worker * WHAT: Thread pool worker that computes checksums for a file or directory tree. Stat's the scan target, opens confined fd
29 * for regular files and computes single checksum via brix_checksum_u32_fd; walks directories recursively with depth/file
30 * limits calling ckscan_append per file. Allocates response buffer with dynamic growth. Sets t->error_code on failure. */
31
32void
33brix_ckscan_aio_thread(void *data, ngx_log_t *log)
34{
35 brix_ckscan_aio_t *t = data;
36 /*
37 * Growing-buffer triple, passed by reference into brix_ckscan_run:
38 * buf = heap block (ngx_alloc, NOT pool — this runs on a thread-pool worker
39 * with no access to the request pool); ownership is transferred to
40 * t->resp on success so the done callback frees it, otherwise every
41 * error path below must ngx_free(buf) before returning.
42 * cap = current allocation size; the run reallocs and updates it in place.
43 * used = bytes written so far (excludes the trailing NUL added at the end).
44 */
45 size_t cap = BRIX_CKSCAN_INIT_CAP;
46 size_t used = 0;
47 u_char *buf;
48 uint16_t err_code = 0;
49
50 buf = ngx_alloc(cap, log);
51 if (buf == NULL) {
52 t->error_code = kXR_NoMemory;
53 snprintf(t->error_msg, sizeof(t->error_msg), "out of memory");
54 return;
55 }
56
57 /* All stat/open/checksum/walk runs through the confined VFS walk (thread-safe:
58 * no pool allocation, no metric), so the worker never touches a confined
59 * helper or the request pool directly. */
60 if (brix_ckscan_run(log, t->rootfd, t->scan_logical, t->algo,
61 &buf, &cap, &used, t->max_depth, t->max_files,
62 &err_code, t->error_msg, sizeof(t->error_msg))
63 != NGX_OK)
64 {
65 ngx_free(buf);
66 t->error_code = err_code;
67 return;
68 }
69
70 /*
71 * NUL-terminate in the slot at [used]: both ngx_alloc(INIT_CAP) above and every
72 * ckscan_append/walk grow keep cap strictly greater than used, so [used] is always
73 * a valid in-bounds byte and never overwrites response data. resp_len stays = used
74 * (the NUL is excluded); the +1 sent to the client in _done covers the terminator.
75 * Success handoff: buf ownership now belongs to t->resp — do NOT ngx_free here.
76 */
77 buf[used] = '\0';
78 t->resp = buf;
79 t->resp_len = used;
80 t->error_code = 0;
81}

84 * sends error response (t->error_code + t->error_msg) or ok+checksum data to client, frees the response buffer, and
85 * resumes the client connection event loop via brix_aio_resume(). */

87void
88brix_ckscan_aio_done(ngx_event_t *ev)
89{
90 ngx_thread_task_t *task = ev->data;
91 brix_ckscan_aio_t *t = task->ctx;
92 brix_ctx_t *ctx = t->ctx;
93 ngx_connection_t *c = t->c;
94
95 /*
96 * Cross-thread race guard. The worker ran detached from the event loop, so the
97 * client connection may have closed (or the session been re-keyed) meanwhile.
98 * aio_restore_request re-binds this op to its original streamid; if it fails the
99 * connection/session is gone — we must NOT touch c or send anything, but we still
100 * own t->resp (transferred from the worker) and must free it here to avoid a leak.
101 */
102 if (!brix_aio_restore_request(ctx, t->streamid)) {
103 if (t->resp) {
104 ngx_free(t->resp);
105 }
106 return;
107 }
108
109 if (t->error_code != 0) {
110 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSCAN);
111 brix_send_error(ctx, c, (uint16_t) t->error_code, t->error_msg);
112 } else {
113 BRIX_OP_OK(ctx, BRIX_OP_QUERY_CKSCAN);
114 brix_log_access(ctx, c, "QUERY", t->scan_logical, "ckscan",
115 1, 0, NULL, 0);
116 /* resp_len + 1: include the worker's trailing NUL in the kXR_ok payload,
117 * matching xrdadler32 client expectations for a C-string body. */
118 brix_send_ok(ctx, c, t->resp, (uint32_t) (t->resp_len + 1));
119 }
120
121 if (t->resp) {
122 ngx_free(t->resp);
123 }
124
125 brix_aio_resume(c);
126}
