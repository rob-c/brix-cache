6/*
7 * unified.c — cross-protocol metric label tables and classification helpers.
8 *
9 * WHAT: Owns the shared label vocabulary for the unified metrics API declared in
10 * unified.h — the static name tables (proto/op/err/auth/tpc-direction) and
11 * the latency-bucket bound table — plus the name-lookup and classification
12 * helpers: brix_metric_proto_name / _op_name / _err_name, the auth-slot
13 * mapper (brix_metric_auth_slot) and its name form, and the errno→err /
14 * http-status→err classifiers. The record-side mutators live in
15 * unified_record.c and the Prometheus exporter in unified_export*.c.
16 * WHY: One vocabulary behind one exporter keeps the three protocols' dashboards
17 * directly comparable and keeps labels low-cardinality (INVARIANT #8).
18 * Concentrating the tables and the pure classifiers here — separate from
19 * the SHM mutators and the exporter — keeps every file single-concern and
20 * under the file-size budget (coding-standards §1).
21 * HOW: Static name tables map each enum to its label string; three of them
22 * (auth names, tpc-direction names, latency bounds) are shared with the
23 * exporter/record files and so carry external linkage via
24 * unified_internal.h. The classification helpers are pure functions —
25 * table lookups and small switch/if ladders — with no SHM or I/O side
26 * effects, so they are trivially reusable across protocols.
27 */

112/*
113 * brix_metric_auth_slot — map an identity auth_method bitmask to one
114 * BRIX_METRIC_AUTH_* slot. Tested in priority order (GSI, TOKEN, SSS, S3KEY,
115 * UNIX, KRB5); returns BRIX_METRIC_AUTH_NONE when no known bit is set.
116 */
117ngx_uint_t
118brix_metric_auth_slot(ngx_uint_t auth_method)
119{
120 if (auth_method & BRIX_AUTHN_GSI) {
121 return BRIX_METRIC_AUTH_GSI;
122 }
123 if (auth_method & BRIX_AUTHN_TOKEN) {
124 return BRIX_METRIC_AUTH_TOKEN;
125 }
126 if (auth_method & BRIX_AUTHN_SSS) {
127 return BRIX_METRIC_AUTH_SSS;
128 }
129 if (auth_method & BRIX_AUTHN_S3KEY) {
130 return BRIX_METRIC_AUTH_S3KEY;
131 }
132 if (auth_method & BRIX_AUTHN_UNIX) {
133 return BRIX_METRIC_AUTH_UNIX;
134 }
135 if (auth_method & BRIX_AUTHN_KRB5) {
136 return BRIX_METRIC_AUTH_KRB5;

154/*
155 * brix_metric_err_from_errno — classify a POSIX errno into an
156 * brix_err_class_t bucket (0→NONE, ENOENT/ENOTDIR→NOT_FOUND,
157 * EACCES/EPERM→FORBIDDEN, EIO/ENOMEM/ENOSPC→IO, else OTHER).
158 */
159brix_err_class_t
160brix_metric_err_from_errno(int sys_errno)
161{
162 switch (sys_errno) {
163 case 0:
164 return BRIX_ERR_NONE;
165 case ENOENT:
166 case ENOTDIR:
167 return BRIX_ERR_NOT_FOUND;
168 case EACCES:
169 case EPERM:
170 return BRIX_ERR_FORBIDDEN;
171 case EIO:
172 case ENOMEM:
173 case ENOSPC:
174 return BRIX_ERR_IO;
175 default:
176 return BRIX_ERR_OTHER;
177 }
178}

180/*
181 * brix_metric_err_from_http_status — classify an HTTP status code into an
182 * brix_err_class_t bucket (2xx/3xx→NONE, 404→NOT_FOUND, 401/403→FORBIDDEN,
183 * 5xx→IO, else OTHER), so WebDAV/S3 outcomes share the unified error vocabulary.
184 */
185brix_err_class_t
186brix_metric_err_from_http_status(ngx_uint_t status)
187{
188 if (status >= 200 && status < 400) {
189 return BRIX_ERR_NONE;
190 }
191 if (status == NGX_HTTP_NOT_FOUND) {
192 return BRIX_ERR_NOT_FOUND;
193 }
194 if (status == NGX_HTTP_FORBIDDEN || status == NGX_HTTP_UNAUTHORIZED) {
195 return BRIX_ERR_FORBIDDEN;
196 }
197 if (status >= 500 && status < 600) {
198 return BRIX_ERR_IO;
199 }
200
201 return BRIX_ERR_OTHER;
202}
