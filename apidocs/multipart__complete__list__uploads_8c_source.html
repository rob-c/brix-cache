1/*
2 * multipart_complete_list_uploads.c — S3 ListMultipartUploads handler.
3 *
4 * WHAT: This fragment implements the GET /bucket/?uploads → ListMultipartUploads operation,
5 * which enumerates all in-progress multipart uploads across the entire bucket root directory.
6 * It contains four components:
7 * - mpu_upload_entry_t typedef: holds key (NAME_MAX+1 char), upload_id (128 char), and mtime —
8 * metadata collected per-upload during bucket-root directory scan. Keys are bounded by NAME_MAX
9 * because staging directory names follow the pattern ".<key>.mpu-<upload_id>" and must fit in
10 * a single POSIX directory entry.
11 * - MPU_MAX_LISTED_UPLOADS constant (1000): matches AWS S3's maximum per-page listing limit.
12 * - mpu_upload_entry_cmp(): qsort comparator that sorts entries by key in ascending lexicographic
13 * order, ensuring deterministic pagination across repeated requests.
14 * - s3_handle_list_multipart_uploads(): the main handler — scans the bucket root directory for hidden
15 * staging dirs matching ".<key>.mpu-<upload_id>", extracts key and upload_id from each name, sorts
16 * by key, applies pagination (key-marker + max-uploads), and emits a ListMultipartUploadsResult XML
17 * document with Key, UploadId, StorageClass (STANDARD), and Initiated (ISO 8601 mtime) per entry.
18 * The handler skips "." and ".." entries, non-directory files, and entries lacking the ".mpu-" marker.
19 *
20 * WHY: Split from multipart.c to keep ListMultipartUploads logic alongside its sibling fragments for
21 * CompleteMultipartUpload operations. The bucket-root scan is structurally different from per-upload
22 * part enumeration (ListParts), so separating them keeps each fragment focused on one distinct pattern.
23 */

37 * GET /bucket/?uploads → ListMultipartUploads
38 * */

40/*
41 * One in-progress upload record collected during the bucket scan.
42 *
43 * Key is bounded by NAME_MAX (255 on Linux): the staging directory name is
44 * ".<key>.mpu-<upload_id>" and must fit in a single directory entry.
45 * We use NAME_MAX + 1 for safety.
46 */
47typedef struct {
48 char key[NAME_MAX + 1]; /* object key (without leading '.') */
49 char upload_id[128]; /* upload id token */
50 time_t mtime; /* staging dir mtime (≈ initiation) */
51} mpu_upload_entry_t;
52
53/*
54 * Maximum number of concurrent in-progress uploads we enumerate in a single
55 * ListMultipartUploads response. AWS S3 returns at most 1000 per page.
56 */
57#define MPU_MAX_LISTED_UPLOADS 1000
58
59/*
60 * Recursion-depth cap for the bucket-tree scan — a runaway guard, generous
61 * enough for realistically-nested object keys ("a/b/c/.../obj").
62 */
63#define MPU_LIST_MAX_DEPTH 16
64
65static int
66mpu_upload_entry_cmp(const void *a, const void *b)
67{
68 return strcmp(((const mpu_upload_entry_t *) a)->key,
69 ((const mpu_upload_entry_t *) b)->key);
70}

72/*
73 * mpu_collect — recursively gather in-progress multipart staging dirs under
74 * dir_path, reconstructing each upload's full object key from key_prefix.
75 *
76 * WHAT: Walks the bucket tree (impersonation-confined) collecting staging dirs
77 * named ".<segment>.mpu-<upload_id>". The reported key is
78 * key_prefix + segment, so a path-keyed upload "alice/foo" staged at
79 * <root>/alice/.foo.mpu-<id> is listed with Key "alice/foo". Plain
80 * subdirectories are recursed into as key-prefix components; a staging dir
81 * is recorded as a leaf and never descended.
82 * WHY: The original flat single-level opendir(root) only saw bucket-root-keyed
83 * uploads, and used a bare worker opendir. Under `brix_impersonation
84 * map` a path-keyed upload lives in the mapped user's 0700 subdir, which
85 * the flat bare scan can neither reach (wrong scope) nor — as the
86 * unprivileged worker — open (EACCES). Mirrors s3_walk()'s confined
87 * recursive enumeration in list_walk.c.
88 * HOW: Each directory is opened via brix_vfs_opendir_quiet (non-metered;
89 * broker-opened as the mapped user, a plain opendir off impersonation). A directory
90 * the mapped user cannot traverse (e.g. another tenant's 0700) yields NULL
91 * and is simply skipped, so cross-tenant uploads are never enumerated.
92 * Bounded by *n < max and MPU_LIST_MAX_DEPTH; symlinks are skipped (lstat).
93 */
94static void
95mpu_collect(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
96 const char *dir_path, const char *key_prefix,
97 mpu_upload_entry_t *uploads, int max, int *n, int depth)
98{
99 brix_vfs_ctx_t vctx;
100 brix_vfs_dir_t *dp;
101
102 if (*n >= max || depth > MPU_LIST_MAX_DEPTH) {
103 return;
104 }
105
106 /* Non-metered confined opendir (the ListMultipartUploads op accounts for the
107 * whole recursive walk; broker fdopendir under impersonation). */
108 s3_build_vfs_ctx(r, dir_path, cf, &vctx);
109 dp = brix_vfs_opendir_quiet(&vctx, NULL);
110 if (dp == NULL) {
111 return; /* unreadable as the mapped user (e.g. another tenant's 0700) */
112 }
113
114 for ( ;; ) {
115 ngx_str_t name;
116 const char *dname;
117 char child_path[PATH_MAX];
118 brix_vfs_ctx_t cctx;
119 brix_vfs_stat_t csb;
120 const char *mpu_marker;
121
122 if (*n >= max) {
123 break;
124 }
125 /* "."/".." filtered by readdir_kind. */
126 if (brix_vfs_readdir_kind(dp, &name, NULL) != NGX_OK) {
127 break; /* NGX_DONE (end) or error → stop */
128 }
129 dname = (const char *) name.data;
130
131 if ((size_t) snprintf(child_path, sizeof(child_path), "%s/%s",
132 dir_path, dname) >= sizeof(child_path)) {
133 continue;
134 }
135
136 /* Confined no-follow probe (non-metered): need both the dir check and the
137 * mtime for a staging dir. A symlink/file/special is skipped. */
138 s3_build_vfs_ctx(r, child_path, cf, &cctx);
139 if (brix_vfs_probe(&cctx, 1 /* no-follow */, &csb) != NGX_OK
140 || !csb.is_directory)
141 {
142 continue; /* not a directory (probe also rejects symlinks) */
143 }
144
145 /* A staging dir is hidden and carries the ".mpu-" marker. */
146 mpu_marker = (dname[0] == '.') ? strstr(dname + 1, ".mpu-") : NULL;
147 if (mpu_marker != NULL) {
148 size_t seg_len = (size_t) (mpu_marker - (dname + 1));
149 const char *uid = mpu_marker + 5; /* skip ".mpu-" */
150 char keybuf[NAME_MAX + 1];
151 int kl;
152
153 if (seg_len == 0 || uid[0] == '\0') {
154 continue;
155 }
156 kl = snprintf(keybuf, sizeof(keybuf), "%s%.*s",
157 key_prefix, (int) seg_len, dname + 1);
158 if (kl < 0 || (size_t) kl >= sizeof(keybuf)
159 || strlen(uid) >= sizeof(uploads[*n].upload_id))
160 {
161 continue; /* key / upload-id too long — skip (bounds-consistent) */
162 }
163 memcpy(uploads[*n].key, keybuf, (size_t) kl + 1);
164 ngx_cpystrn((u_char *) uploads[*n].upload_id, (u_char *) uid,
165 sizeof(uploads[*n].upload_id));
166 uploads[*n].mtime = csb.mtime;
167 (*n)++;
168 continue; /* never descend INTO a staging dir */
169 }
170
171 /* A plain key-prefix directory (e.g. "alice/"): recurse with the prefix
172 * extended by this segment + '/'. */
173 {
174 char next_prefix[PATH_MAX];
175 if ((size_t) snprintf(next_prefix, sizeof(next_prefix), "%s%s/",
176 key_prefix, dname) >= sizeof(next_prefix)) {
177 continue;
178 }
179 mpu_collect(r, cf, child_path, next_prefix, uploads, max, n,
180 depth + 1);
181 }
182 }
183
184 brix_vfs_closedir(dp, r->connection->log);
185}

187ngx_int_t
188s3_handle_list_multipart_uploads(ngx_http_request_t *r,
189 ngx_http_s3_loc_conf_t *cf)
190{
191 mpu_upload_entry_t *uploads;
192 int nuploads = 0;
193 int i;
194 size_t xml_capacity;
195 u_char *xml;
196 size_t xml_len = 0;
197 ngx_buf_t *b;
198 char iso_buf[32];
199 char key_marker[NAME_MAX + 1];
200 char max_uploads_str[16];
201 int max_uploads;
202 int start_idx;
203 int end_idx;
204 int truncated;
205
206 if (cf->common.root.data == NULL || cf->common.root.len == 0) {
207 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
208 return NGX_HTTP_INTERNAL_SERVER_ERROR;
209 }
210
211 /* Parse pagination query parameters */
212 key_marker[0] = '\0';
213 s3_get_query_param(r, "key-marker", key_marker, sizeof(key_marker));
214
215 max_uploads = MPU_MAX_LISTED_UPLOADS;
216 if (s3_get_query_param(r, "max-uploads",
217 max_uploads_str, sizeof(max_uploads_str)))
218 {
219 char *endp;
220 long mu = strtol(max_uploads_str, &endp, 10);
221 if (endp != max_uploads_str && mu > 0 && mu < MPU_MAX_LISTED_UPLOADS) {
222 max_uploads = (int) mu;
223 }
224 }
225
226 /* Collect matching entries first so we can size the XML buffer accurately. */
227 uploads = ngx_palloc(r->pool,
228 sizeof(mpu_upload_entry_t) * MPU_MAX_LISTED_UPLOADS);
229 if (uploads == NULL) {
230 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
231 return NGX_HTTP_INTERNAL_SERVER_ERROR;
232 }
233
234 /*
235 * Recursively enumerate staging dirs across the whole bucket tree, opened
236 * impersonation-confined as the mapped user. Replaces the original flat
237 * single-level bare-opendir(root) scan, which (a) only saw bucket-root-keyed
238 * uploads and (b) under `brix_impersonation map` could not open the mapped
239 * user's own 0700 key subdirs as the unprivileged worker. A dir the mapped
240 * user cannot traverse is skipped, so no cross-tenant upload is enumerated.
241 */
242 mpu_collect(r, cf, (const char *) cf->common.root_canon, "",
243 uploads, MPU_MAX_LISTED_UPLOADS, &nuploads, 0);
244
245 /* Sort by key for deterministic pagination */
246 if (nuploads > 1) {
247 qsort(uploads, (size_t) nuploads, sizeof(mpu_upload_entry_t),
248 mpu_upload_entry_cmp);
249 }
250
251 /* Apply key-marker: skip entries whose key <= marker */
252 start_idx = 0;
253 if (key_marker[0] != '\0') {
254 for (start_idx = 0; start_idx < nuploads; start_idx++) {
255 if (strcmp(uploads[start_idx].key, key_marker) > 0) {
256 break;
257 }
258 }
259 }
260
261 end_idx = start_idx + max_uploads;
262 if (end_idx > nuploads) {
263 end_idx = nuploads;
264 }
265 truncated = (end_idx < nuploads);
266
267 /* Size the XML buffer: header + per-upload block */
268 xml_capacity = 512
269 + (size_t) cf->bucket.len * 6 + 32
270 + strlen(key_marker) * 6 + 32
271 + (end_idx - start_idx) * (NAME_MAX * 6 + 256)
272 + (truncated ? NAME_MAX * 6 + 128 : 0);
273
274 xml = ngx_palloc(r->pool, xml_capacity);
275 if (xml == NULL) {
276 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
277 return NGX_HTTP_INTERNAL_SERVER_ERROR;
278 }
279
280 XML_APPEND("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
281 XML_APPEND("<ListMultipartUploadsResult"
282 " xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">\n");
283 XML_APPEND(" ");
284 XML_APPEND_ELEM("Bucket", cf->bucket.data, cf->bucket.len);
285 XML_APPEND("\n ");
286 XML_APPEND_ELEM("KeyMarker", key_marker, strlen(key_marker));
287 XML_APPEND("\n");
288 XML_APPEND(" <MaxUploads>%d</MaxUploads>\n", max_uploads);
289 XML_APPEND(" <IsTruncated>%s</IsTruncated>\n",
290 truncated ? "true" : "false");
291
292 if (truncated) {
293 XML_APPEND(" ");
294 XML_APPEND_ELEM("NextKeyMarker", uploads[end_idx - 1].key,
295 strlen(uploads[end_idx - 1].key));
296 XML_APPEND("\n ");
297 XML_APPEND_ELEM("NextUploadIdMarker", uploads[end_idx - 1].upload_id,
298 strlen(uploads[end_idx - 1].upload_id));
299 XML_APPEND("\n");
300 }
301
302 for (i = start_idx; i < end_idx; i++) {
303 brix_format_iso8601(uploads[i].mtime, iso_buf, sizeof(iso_buf));
304 XML_APPEND(" <Upload>\n");
305 XML_APPEND(" ");
306 XML_APPEND_ELEM("Key", uploads[i].key, strlen(uploads[i].key));
307 XML_APPEND("\n ");
308 XML_APPEND_ELEM("UploadId", uploads[i].upload_id,
309 strlen(uploads[i].upload_id));
310 XML_APPEND("\n");
311 XML_APPEND(" <StorageClass>STANDARD</StorageClass>\n");
312 XML_APPEND(" <Initiated>%s</Initiated>\n", iso_buf);
313 XML_APPEND(" </Upload>\n");
314 }
315
316 XML_APPEND("</ListMultipartUploadsResult>\n");
317
318 b = ngx_create_temp_buf(r->pool, xml_len);
319 if (b == NULL) {
320 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
321 return NGX_HTTP_INTERNAL_SERVER_ERROR;
322 }
323 ngx_memcpy(b->pos, xml, xml_len);
324 b->last = b->pos + xml_len;
325 b->last_buf = 1;
326
327 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
328 (ngx_str_t) ngx_string("application/xml"), b);
329}
