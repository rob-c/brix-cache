2 * multipart_complete_list_uploads.c — S3 ListMultipartUploads handler.
3 *
4 * WHAT: This fragment implements the GET /bucket/?uploads → ListMultipartUploads operation,
5 * which enumerates all in-progress multipart uploads across the entire bucket root directory.
6 * It contains four components:
7 * - mpu_upload_entry_t typedef: holds key (NAME_MAX+1 char), upload_id (128 char), and mtime —
8 * metadata collected per-upload during bucket-root directory scan. Keys are bounded by NAME_MAX
9 * because staging directory names follow the pattern ".<key>.mpu-<upload_id>" and must fit in
10 * a single POSIX directory entry.
11 * - MPU_MAX_LISTED_UPLOADS constant (1000): matches AWS S3's maximum per-page listing limit.
12 * - mpu_upload_entry_cmp(): qsort comparator that sorts entries by key in ascending lexicographic
13 * order, ensuring deterministic pagination across repeated requests.
14 * - s3_handle_list_multipart_uploads(): the main handler — scans the bucket root directory for hidden
15 * staging dirs matching ".<key>.mpu-<upload_id>", extracts key and upload_id from each name, sorts
16 * by key, applies pagination (key-marker + max-uploads), and emits a ListMultipartUploadsResult XML
17 * document with Key, UploadId, StorageClass (STANDARD), and Initiated (ISO 8601 mtime) per entry.
18 * The handler skips "." and ".." entries, non-directory files, and entries lacking the ".mpu-" marker.
19 *
20 * WHY: Split from multipart.c to keep ListMultipartUploads logic alongside its sibling fragments for
21 * CompleteMultipartUpload operations. The bucket-root scan is structurally different from per-upload
22 * part enumeration (ListParts), so separating them keeps each fragment focused on one distinct pattern.
23 */

40/*
41 * One in-progress upload record collected during the bucket scan.
42 *
43 * Key is bounded by NAME_MAX (255 on Linux): the staging directory name is
44 * ".<key>.mpu-<upload_id>" and must fit in a single directory entry.
45 * We use NAME_MAX + 1 for safety.

47typedef struct {
48 char key[NAME_MAX + 1]; /* object key (without leading '.') */
49 char upload_id[128]; /* upload id token */
50 time_t mtime; /* staging dir mtime (≈ initiation) */
51} mpu_upload_entry_t;

65/*
66 * Bucket-scan accumulator threaded through the recursive walk.
67 *
68 * WHAT: Collapses the request/config context plus the collection sink
69 * (uploads array, capacity, running count) into one struct so the
70 * recursive walker stays under the 5-parameter cap.
71 * WHY: mpu_collect() recurses per directory level; carrying r/cf/uploads/max/n

113/*
114 * mpu_record_staging — record a ".<segment>.mpu-<upload_id>" staging dir.
115 *
116 * WHAT: Given a hidden entry name `dname` carrying the ".mpu-" marker, slices
117 * out the key segment and upload-id, reconstructs the full object key
118 * (key_prefix + segment), validates lengths, and appends an entry to
119 * scan->uploads, advancing scan->n. Empty/oversized entries are skipped.
120 * WHY: Isolating the marker-parse + bounds-check + record step keeps the
121 * directory loop in mpu_collect() flat and drops its complexity below the
122 * CCN cap while preserving identical accept/reject behavior.
123 * HOW: The caller guarantees `dname[0] == '.'` and the marker is present.
124 * Empty segment, empty uid, oversized key, or oversized uid → skip.
125 * On accept, key and upload_id are copied and the probed mtime stored.
126 */
127static void
128mpu_record_staging(mpu_scan_t *scan, const char *dname, const char *key_prefix,
129 time_t mtime)
130{
131 const char *mpu_marker = strstr(dname + 1, ".mpu-");
132 size_t seg_len = (size_t) (mpu_marker - (dname + 1));
133 const char *uid = mpu_marker + 5; /* skip ".mpu-" */
134 mpu_upload_entry_t *e;
135 char keybuf[NAME_MAX + 1];
136 int kl;
137
138 if (seg_len == 0 || uid[0] == '\0') {
139 return;
140 }
141
142 kl = snprintf(keybuf, sizeof(keybuf), "%s%.*s",
143 key_prefix, (int) seg_len, dname + 1);
144 if (kl < 0 || (size_t) kl >= sizeof(keybuf)
145 || strlen(uid) >= sizeof(scan->uploads[0].upload_id))
146 {
147 return; /* key / upload-id too long — skip (bounds-consistent) */
148 }
149
150 e = &scan->uploads[scan->n];
151 memcpy(e->key, keybuf, (size_t) kl + 1);
152 ngx_cpystrn((u_char *) e->upload_id, (u_char *) uid,
153 sizeof(e->upload_id));
154 e->mtime = mtime;
155 scan->n++;
156}

158/*
159 * mpu_scan_entry — classify and dispatch one directory entry during the walk.
160 *
161 * WHAT: For a single readdir entry `dname` under `dir_path` (key `key_prefix`):
162 * builds the child path, confined no-follow-probes it, and either records
163 * it (staging dir), recurses into it (plain key-prefix dir), or skips it.
164 * WHY: Pulling the per-entry classification out of the readdir loop keeps
165 * mpu_collect() a thin driver and removes the nested branch ladder that
166 * drove its cyclomatic complexity over the cap.
167 * HOW: Oversized child path → skip. Non-directory / symlink (probe no-follow) →
168 * skip. Hidden name carrying ".mpu-" → record via mpu_record_staging.
169 * Otherwise a plain directory → recurse with the prefix extended by the
170 * segment + '/'. Never descends INTO a staging dir.
171 */
172static void
173mpu_scan_entry(mpu_scan_t *scan, const char *dir_path, const char *key_prefix,
174 const char *dname, int depth)
175{
176 char child_path[PATH_MAX];
177 brix_vfs_ctx_t cctx;
178 brix_vfs_stat_t csb;
179 char next_prefix[PATH_MAX];
180
181 if ((size_t) snprintf(child_path, sizeof(child_path), "%s/%s",
182 dir_path, dname) >= sizeof(child_path)) {
183 return;
184 }
185
186 /* Confined no-follow probe (non-metered): need both the dir check and the
187 * mtime for a staging dir. A symlink/file/special is skipped. */
188 s3_build_vfs_ctx(scan->r, child_path, scan->cf, &cctx);
189 if (brix_vfs_probe(&cctx, 1 /* no-follow */, &csb) != NGX_OK
190 || !csb.is_directory)
191 {
192 return; /* not a directory (probe also rejects symlinks) */
193 }
194
195 /* A staging dir is hidden and carries the ".mpu-" marker. */
196 if (dname[0] == '.' && strstr(dname + 1, ".mpu-") != NULL) {
197 mpu_record_staging(scan, dname, key_prefix, csb.mtime);
198 return; /* never descend INTO a staging dir */
199 }
200
201 /* A plain key-prefix directory (e.g. "alice/"): recurse with the prefix
202 * extended by this segment + '/'. */
203 if ((size_t) snprintf(next_prefix, sizeof(next_prefix), "%s%s/",
204 key_prefix, dname) >= sizeof(next_prefix)) {
205 return;
206 }
207 mpu_collect(scan, child_path, next_prefix, depth + 1);
208}

210/*
211 * mpu_collect — recursively gather in-progress multipart staging dirs under
212 * dir_path, reconstructing each upload's full object key from key_prefix.
213 *
214 * WHAT: Walks the bucket tree (impersonation-confined) collecting staging dirs
215 * named ".<segment>.mpu-<upload_id>". The reported key is
216 * key_prefix + segment, so a path-keyed upload "alice/foo" staged at
217 * <root>/alice/.foo.mpu-<id> is listed with Key "alice/foo". Plain
218 * subdirectories are recursed into as key-prefix components; a staging dir
219 * is recorded as a leaf and never descended.
220 * WHY: The original flat single-level opendir(root) only saw bucket-root-keyed
221 * uploads, and used a bare worker opendir. Under `brix_impersonation
222 * map` a path-keyed upload lives in the mapped user's 0700 subdir, which
223 * the flat bare scan can neither reach (wrong scope) nor — as the
224 * unprivileged worker — open (EACCES). Mirrors s3_walk()'s confined
225 * recursive enumeration in list_walk.c.
226 * HOW: Each directory is opened via brix_vfs_opendir_quiet (non-metered;
227 * broker-opened as the mapped user, a plain opendir off impersonation). A directory
228 * the mapped user cannot traverse (e.g. another tenant's 0700) yields NULL
229 * and is simply skipped, so cross-tenant uploads are never enumerated.
230 * Bounded by scan->n < max and MPU_LIST_MAX_DEPTH; per-entry classification
231 * (record vs. recurse vs. skip) is delegated to mpu_scan_entry().
232 */
233static void
234mpu_collect(mpu_scan_t *scan, const char *dir_path, const char *key_prefix,
235 int depth)
236{
237 brix_vfs_ctx_t vctx;
238 brix_vfs_dir_t *dp;
239
240 if (scan->n >= scan->max || depth > MPU_LIST_MAX_DEPTH) {
241 return;
242 }
243
244 /* Non-metered confined opendir (the ListMultipartUploads op accounts for the
245 * whole recursive walk; broker fdopendir under impersonation). */
246 s3_build_vfs_ctx(scan->r, dir_path, scan->cf, &vctx);
247 dp = brix_vfs_opendir_quiet(&vctx, NULL);
248 if (dp == NULL) {
249 return; /* unreadable as the mapped user (e.g. another tenant's 0700) */
250 }
251
252 for ( ;; ) {
253 ngx_str_t name;
254
255 if (scan->n >= scan->max) {
256 break;
257 }
258 /* "."/".." filtered by readdir_kind. */
259 if (brix_vfs_readdir_kind(dp, &name, NULL) != NGX_OK) {
260 break; /* NGX_DONE (end) or error → stop */
261 }
262
263 mpu_scan_entry(scan, dir_path, key_prefix,
264 (const char *) name.data, depth);
265 }
266
267 brix_vfs_closedir(dp, scan->r->connection->log);
268}

270/*
271 * mpu_parse_max_uploads — resolve the effective max-uploads page size.
272 *
273 * WHAT: Returns the clamped max-uploads value from the query string, defaulting
274 * to MPU_MAX_LISTED_UPLOADS when the parameter is absent or invalid.
275 * WHY: Isolating the strtol-parse + range clamp keeps the handler prologue flat
276 * and drops one of its nested branches.
277 * HOW: A present, fully-numeric value in (0, MPU_MAX_LISTED_UPLOADS) is used;
278 * anything else falls back to the default cap. Behavior is byte-identical
279 * to the original inline parse.
280 */
281static int
282mpu_parse_max_uploads(ngx_http_request_t *r)
283{
284 char max_uploads_str[16];
285
286 if (s3_get_query_param(r, "max-uploads",
287 max_uploads_str, sizeof(max_uploads_str)))
288 {
289 char *endp;
290 long mu = strtol(max_uploads_str, &endp, 10);
291 if (endp != max_uploads_str && mu > 0 && mu < MPU_MAX_LISTED_UPLOADS) {
292 return (int) mu;
293 }
294 }
295 return MPU_MAX_LISTED_UPLOADS;
296}

298/*
299 * mpu_marker_start_idx — first entry index strictly after key_marker.
300 *
301 * WHAT: Given a key-sorted uploads array, returns the index of the first entry
302 * whose key sorts strictly after key_marker (0 when the marker is empty).
303 * WHY: Pagination "skip entries whose key <= marker" is a self-contained scan;
304 * extracting it keeps the handler linear.
305 * HOW: Empty marker → 0. Otherwise a forward scan stopping at the first key
306 * that compares greater than the marker — identical to the inline loop.
307 */
308static int
309mpu_marker_start_idx(const mpu_upload_entry_t *uploads, int nuploads,
310 const char *key_marker)
311{
312 int start_idx;
313
314 if (key_marker[0] == '\0') {
315 return 0;
316 }
317 for (start_idx = 0; start_idx < nuploads; start_idx++) {
318 if (strcmp(uploads[start_idx].key, key_marker) > 0) {
319 break;
320 }
321 }
322 return start_idx;
323}

325/*
326 * mpu_render_and_send — build the ListMultipartUploadsResult document and send it.
327 *
328 * WHAT: Allocates a right-sized buffer, writes the full XML response (header,
329 * Bucket/KeyMarker/MaxUploads/IsTruncated, optional NextKeyMarker /
330 * NextUploadIdMarker when truncated, and one <Upload> block per listed
331 * entry), copies it into an nginx temp buf and sends it as a 200
332 * application/xml response. Returns the send result, or
333 * NGX_HTTP_INTERNAL_SERVER_ERROR on allocation / encode overflow.
334 * WHY: The XML assembly is the bulk of the handler's length; pulling it out
335 * leaves the handler as a short orchestration of parse → collect → sort →
336 * paginate → render+send. Rendering and sending stay together because the
337 * XML_APPEND macros embed a `return NGX_HTTP_INTERNAL_SERVER_ERROR` on
338 * overflow — the caller must be a handler-returning (ngx_int_t) function.
339 * HOW: Buffer sizing, every XML_APPEND, and the temp-buf send are copied
340 * verbatim from the original inline tail, so the produced document and all
341 * status/error paths are byte-for-byte identical.
342 */
343static ngx_int_t
344mpu_render_and_send(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
345 const mpu_page_t *pg)
346{
347 const mpu_upload_entry_t *uploads = pg->uploads;
348 const char *key_marker = pg->key_marker;
349 int max_uploads = pg->max_uploads;
350 int start_idx = pg->start_idx;
351 int end_idx = pg->end_idx;
352 int truncated = pg->truncated;
353 size_t xml_capacity;
354 u_char *xml;
355 size_t xml_len = 0;
356 char iso_buf[32];
357 int i;
358 ngx_buf_t *b;
359
360 /* Size the XML buffer: header + per-upload block */
361 xml_capacity = 512
362 + (size_t) cf->bucket.len * 6 + 32
363 + strlen(key_marker) * 6 + 32
364 + (end_idx - start_idx) * (NAME_MAX * 6 + 256)
365 + (truncated ? NAME_MAX * 6 + 128 : 0);
366
367 xml = ngx_palloc(r->pool, xml_capacity);
368 if (xml == NULL) {
369 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
370 return NGX_HTTP_INTERNAL_SERVER_ERROR;
371 }
372
373 XML_APPEND("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
374 XML_APPEND("<ListMultipartUploadsResult"
375 " xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">\n");
376 XML_APPEND(" ");
377 XML_APPEND_ELEM("Bucket", cf->bucket.data, cf->bucket.len);
378 XML_APPEND("\n ");
379 XML_APPEND_ELEM("KeyMarker", key_marker, strlen(key_marker));
380 XML_APPEND("\n");
381 XML_APPEND(" <MaxUploads>%d</MaxUploads>\n", max_uploads);
382 XML_APPEND(" <IsTruncated>%s</IsTruncated>\n",
383 truncated ? "true" : "false");
384
385 if (truncated) {
386 XML_APPEND(" ");
387 XML_APPEND_ELEM("NextKeyMarker", uploads[end_idx - 1].key,
388 strlen(uploads[end_idx - 1].key));
389 XML_APPEND("\n ");
390 XML_APPEND_ELEM("NextUploadIdMarker", uploads[end_idx - 1].upload_id,
391 strlen(uploads[end_idx - 1].upload_id));
392 XML_APPEND("\n");
393 }
394
395 for (i = start_idx; i < end_idx; i++) {
396 brix_format_iso8601(uploads[i].mtime, iso_buf, sizeof(iso_buf));
397 XML_APPEND(" <Upload>\n");
398 XML_APPEND(" ");
399 XML_APPEND_ELEM("Key", uploads[i].key, strlen(uploads[i].key));
400 XML_APPEND("\n ");
401 XML_APPEND_ELEM("UploadId", uploads[i].upload_id,
402 strlen(uploads[i].upload_id));
403 XML_APPEND("\n");
404 XML_APPEND(" <StorageClass>STANDARD</StorageClass>\n");
405 XML_APPEND(" <Initiated>%s</Initiated>\n", iso_buf);
406 XML_APPEND(" </Upload>\n");
407 }
408
409 XML_APPEND("</ListMultipartUploadsResult>\n");
410
411 b = ngx_create_temp_buf(r->pool, xml_len);
412 if (b == NULL) {
413 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
414 return NGX_HTTP_INTERNAL_SERVER_ERROR;
415 }
416 ngx_memcpy(b->pos, xml, xml_len);
417 b->last = b->pos + xml_len;
418 b->last_buf = 1;
419
420 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
421 (ngx_str_t) ngx_string("application/xml"), b);
422}

424ngx_int_t
425s3_handle_list_multipart_uploads(ngx_http_request_t *r,
426 ngx_http_s3_loc_conf_t *cf)
427{
428 mpu_upload_entry_t *uploads;
429 mpu_scan_t scan;
430 mpu_page_t pg;
431 int nuploads;
432 char key_marker[NAME_MAX + 1];
433 int max_uploads;
434 int start_idx;
435 int end_idx;
436 int truncated;
437
438 if (cf->common.root.data == NULL || cf->common.root.len == 0) {
439 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
440 return NGX_HTTP_INTERNAL_SERVER_ERROR;
441 }
442
443 /* Parse pagination query parameters */
444 key_marker[0] = '\0';
445 s3_get_query_param(r, "key-marker", key_marker, sizeof(key_marker));
446 max_uploads = mpu_parse_max_uploads(r);
447
448 /* Collect matching entries first so we can size the XML buffer accurately. */
449 uploads = ngx_palloc(r->pool,
450 sizeof(mpu_upload_entry_t) * MPU_MAX_LISTED_UPLOADS);
451 if (uploads == NULL) {
452 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
453 return NGX_HTTP_INTERNAL_SERVER_ERROR;
454 }
455
456 /*
457 * Recursively enumerate staging dirs across the whole bucket tree, opened
458 * impersonation-confined as the mapped user. Replaces the original flat
459 * single-level bare-opendir(root) scan, which (a) only saw bucket-root-keyed
460 * uploads and (b) under `brix_impersonation map` could not open the mapped
461 * user's own 0700 key subdirs as the unprivileged worker. A dir the mapped
462 * user cannot traverse is skipped, so no cross-tenant upload is enumerated.
463 */
464 scan.r = r;
465 scan.cf = cf;
466 scan.uploads = uploads;
467 scan.max = MPU_MAX_LISTED_UPLOADS;
468 scan.n = 0;
469 mpu_collect(&scan, (const char *) cf->common.root_canon, "", 0);
470 nuploads = scan.n;
471
472 /* Sort by key for deterministic pagination */
473 if (nuploads > 1) {
474 qsort(uploads, (size_t) nuploads, sizeof(mpu_upload_entry_t),
475 mpu_upload_entry_cmp);
476 }
477
478 /* Apply key-marker: skip entries whose key <= marker */
479 start_idx = mpu_marker_start_idx(uploads, nuploads, key_marker);
480
481 end_idx = start_idx + max_uploads;
482 if (end_idx > nuploads) {
483 end_idx = nuploads;
484 }
485 truncated = (end_idx < nuploads);
486
487 pg.uploads = uploads;
488 pg.key_marker = key_marker;
489 pg.max_uploads = max_uploads;
490 pg.start_idx = start_idx;
491 pg.end_idx = end_idx;
492 pg.truncated = truncated;
493
494 return mpu_render_and_send(r, cf, &pg);
495}
