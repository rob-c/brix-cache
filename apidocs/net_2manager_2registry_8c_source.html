54/* WHAT
55 * Shared-memory zone initialiser callback. Called by nginx when the shared-
56 * memory zone is mapped (first time) or reattached (subsequent workers).
57
58 * WHY
59 * Ensures the registry table is zero-filled and the spinlock is created on
60 * first boot; on restart (data != NULL) just recreates the lock against the
61 * existing table structure.
62
63 * HOW
64 * Delegates fresh-alloc / reload / re-attach to brix_shm_table_alloc(), which
65 * allocates the table FROM the slab pool so nginx's slab-pool header survives at
66 * shm.addr (ngx_unlock_mutexes() force-unlocks that header on every child death;
67 * laying our own struct over it would SIGSEGV the master). The helper zero-fills
68 * the table and creates the worker-local mutex from tbl->lock (the table's first
69 * member). On a brand-new allocation we set the capacity; on reuse we must not,
70 * to preserve the live table state.
71 */
72ngx_int_t
73brix_srv_shm_init_zone(ngx_shm_zone_t *shm_zone, void *data)
74{
75 brix_srv_table_t *tbl;
76 ngx_flag_t fresh;
77 size_t table_bytes;
78
79 table_bytes = sizeof(brix_srv_table_t)
80 + (size_t) brix_srv_registry_nslots
81 * sizeof(brix_srv_entry_t);
82
83 tbl = brix_shm_table_alloc(shm_zone, data, table_bytes,
84 &brix_srv_mutex, &fresh);
85 if (tbl == NULL) {
86 return NGX_ERROR;
87 }
88
89 if (fresh) {
90 tbl->capacity = brix_srv_registry_nslots;
91 }
92
93 return NGX_OK;
94}

98/* WHAT
99 * Allocates the shared-memory zone for the server registry and sets its size
100 * based on the configured slot count.
101
102 * WHY
103 * Called during nginx configuration parsing (brix_registry_slots directive).
104 * Must happen before any traffic so that workers can find the zone at startup.
105
106 * HOW
107 * Sets brix_srv_registry_nslots to the requested value, computes zone size
108 * via brix_shm_zone_size(table_bytes) — the table is allocated FROM the slab
109 * pool, so the zone must hold the table plus slab overhead — adds the zone via
110 * ngx_shared_memory_add(), sets init callback and (void *)1 sentinel data.
111 */
112ngx_int_t
113brix_srv_configure_registry(ngx_conf_t *cf, ngx_uint_t slots)
114{
115 ngx_str_t zone_name = ngx_string("brix_srv_registry");
116 size_t zone_size;
117
118 brix_srv_registry_nslots = slots;
119 zone_size = brix_shm_zone_size(
120 sizeof(brix_srv_table_t)
121 + (size_t) slots * sizeof(brix_srv_entry_t));
122 brix_srv_shm_zone = ngx_shared_memory_add(cf, &zone_name,
123 zone_size,
124 &ngx_stream_brix_module);
125 if (brix_srv_shm_zone == NULL) {
126 return NGX_ERROR;
127 }
128
129 brix_shm_zone_warn_on_resize(cf, brix_srv_shm_zone,
130 "brix_registry_slots");
131
132 brix_srv_shm_zone->init = brix_srv_shm_init_zone;
133 brix_srv_shm_zone->data = (void *) 1;
134
135 return NGX_OK;
136}

140/* WHAT
141 * Registers or updates a data server entry in the shared-memory registry.
142 * Called by CMS server handler when a data server logs in or heartbeats.
143
144 * WHY
145 * The redirector needs to know which servers exist, what paths they serve,
146 * and their current load metrics so that kXR_locate / kXR_open can pick the
147 * best server for each request.
148
149 * HOW
150 * Locks mutex → scans all slots: if host+port match found, update paths/free/
151 * util/last_seen fields. If no match and a free slot exists, allocate it with
152 * host/port/paths/free/util/last_seen/in_use=1. If registry is full, log warn
153 * and increment registry_full_total Prometheus counter.
154 */
155void
156brix_srv_register(const char *host, uint16_t port,
157 const char *paths, uint32_t free_mb, uint32_t util_pct)
158{
159 brix_srv_table_t *tbl;
160 brix_srv_entry_t *e;
161 ngx_uint_t i, free_slot;
162 int found;
163
164 tbl = srv_table();
165 if (tbl == NULL) {
166 return;
167 }
168
169 /*
170 * W1c — reject any host string that is not a clean hostname / IP literal
171 * before it can enter the registry. This is the single store choke point,
172 * so it also protects every redirect-emit path (brix_srv_select /
173 * brix_srv_locate_all) from control-byte or scheme injection into the
174 * "S<r|w>host:port" string a client parses. Registry hosts are normally
175 * the peer IP from ngx_sock_ntop, so a rejection here means a poisoned or
176 * malformed registration attempt.
177 */
178 if (host == NULL || !brix_net_host_chars_valid(host, ngx_strlen(host))) {
179 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
180 "brix: rejected registration with invalid host string");
181 return;
182 }
183
184 ngx_shmtx_lock(&brix_srv_mutex);
185
186 free_slot = tbl->capacity;
187 found = 0;
188
189 for (i = 0; i < tbl->capacity; i++) {
190 e = &tbl->slots[i];
191 if (!e->in_use) {
192 if (free_slot == tbl->capacity) {
193 free_slot = i;
194 }
195 continue;
196 }
197 if (e->port == port && ngx_strcmp(e->host, host) == 0) {
198 /* Update existing entry; clear any prior blacklist on reconnect. */
199 ngx_cpystrn((u_char *) e->paths,
200 (u_char *) (paths ? paths : ""),
201 sizeof(e->paths));
202 e->free_mb = free_mb;
203 e->util_pct = util_pct;
204 e->last_seen = ngx_current_msec;
205 e->blacklisted_until = 0;
206 e->error_count = 0;
207 found = 1;
208 break;
209 }
210 }
211
212 if (!found && free_slot < tbl->capacity) {
213 e = &tbl->slots[free_slot];
214 ngx_cpystrn((u_char *) e->host, (u_char *) host, sizeof(e->host));
215 e->port = port;
216 ngx_cpystrn((u_char *) e->paths,
217 (u_char *) (paths ? paths : ""),
218 sizeof(e->paths));
219 e->free_mb = free_mb;
220 e->util_pct = util_pct;
221 e->last_seen = ngx_current_msec;
222 e->in_use = 1;
223 } else if (!found) {
224 /* Registry is full: log a warning and increment the Prometheus counter. */
225 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
226 "brix: server registry full (%ui slots); "
227 "dropping registration for %s:%ui "
228 "(increase brix_registry_slots)",
229 tbl->capacity, host, (ngx_uint_t) port);
230 {
231 ngx_brix_metrics_t *m = brix_metrics_shared();
232 if (m != NULL) {
233 ngx_atomic_fetch_add(&m->registry_full_total, 1);
234 }
235 }
236 }
237
238 ngx_shmtx_unlock(&brix_srv_mutex);
239}

243/* WHAT
244 * Refreshes free-space and utilisation metrics for an already-registered server.
245 * Called on each CMS heartbeat from the data server.
246
247 * WHY
248 * Selection policy depends on current load: reads pick least-loaded servers,
249 * writes pick most-free-space. Metrics must stay fresh for accurate routing.
250
251 * HOW
252 * Locks mutex → scans slots for host+port match → updates free_mb, util_pct,
253 * last_seen fields only (no path changes). Unlocks and returns.
254 */
255void
256brix_srv_update_load(const char *host, uint16_t port,
257 uint32_t free_mb, uint32_t util_pct)
258{
259 brix_srv_table_t *tbl;
260 brix_srv_entry_t *e;
261 ngx_uint_t i;
262
263 tbl = srv_table();
264 if (tbl == NULL) {
265 return;
266 }
267
268 ngx_shmtx_lock(&brix_srv_mutex);
269
270 for (i = 0; i < tbl->capacity; i++) {
271 e = &tbl->slots[i];
272 if (e->in_use && e->port == port

287/* WHAT
288 * Removes a data server entry from the registry by host+port match.
289 * Called when a data server disconnects or is removed from the cluster.
290
291 * WHY
292 * Prevents stale entries from being selected by locate/open operations. A
293 * disconnected server should not receive client traffic.

303 brix_srv_entry_t *e;
304 ngx_uint_t i;
305
306 tbl = srv_table();
307 if (tbl == NULL) {
308 return;
309 }
310
311 ngx_shmtx_lock(&brix_srv_mutex);
312
313 for (i = 0; i < tbl->capacity; i++) {
314 e = &tbl->slots[i];
315 if (e->in_use && e->port == port
316 && ngx_strcmp(e->host, host) == 0)

320 }
321 }

326/*
327 * srv_find_locked — locate the in-use entry for host:port. Caller MUST hold
328 * brix_srv_mutex; returns NULL when the node is not registered (or the zone
329 * is absent), so every Phase-89 W9 attribute writer degrades to a no-op for
330 * an unknown node instead of creating one.
331 */
332static brix_srv_entry_t *
333srv_find_locked(const char *host, uint16_t port)
334{
335 brix_srv_table_t *tbl;
336 ngx_uint_t i;
337
338 tbl = srv_table();
339 if (tbl == NULL || host == NULL) {
340 return NULL;
341 }
342
343 for (i = 0; i < tbl->capacity; i++) {
344 if (tbl->slots[i].in_use && tbl->slots[i].port == port
345 && ngx_strcmp(tbl->slots[i].host, host) == 0)
346 {
347 return &tbl->slots[i];
348 }
349 }
350
351 return NULL;
352}

354/*
355 * brix_srv_reset — kYR_status(reset): forget the node's cached metrics and
356 * fault state but keep the registration + paths (contract in registry.h).
357 * The stale metrics window this opens closes on the node's next load/avail
358 * heartbeat; clearing the blacklist too mirrors stock cmsd, where reset is
359 * the node saying "my state machine restarted, judge me fresh".
360 */
361int
362brix_srv_reset(const char *host, uint16_t port)
363{
364 brix_srv_entry_t *e;
365 int found = 0;
366
367 if (srv_table() == NULL) {
368 return 0;
369 }
370
371 ngx_shmtx_lock(&brix_srv_mutex);
372 e = srv_find_locked(host, port);
373 if (e != NULL) {
374 e->free_mb = 0;
375 e->util_pct = 0;
376 e->error_count = 0;
377 e->blacklisted_until = 0;
378 e->hc_fail_count = 0;
379 e->hc_in_progress = 0;
380 e->load_pct = 0;
381 e->last_seen = ngx_current_msec;
382 found = 1;
383 }
384 ngx_shmtx_unlock(&brix_srv_mutex);
385
386 return found;
387}
