36/* WHAT
37 * Shared-memory zone initialiser callback. Called by nginx when the shared-
38 * memory zone is mapped (first time) or reattached (subsequent workers).
39
40 * WHY
41 * Ensures the registry table is zero-filled and the spinlock is created on
42 * first boot; on restart (data != NULL) just recreates the lock against the
43 * existing table structure.
44
45 * HOW
46 * Delegates fresh-alloc / reload / re-attach to brix_shm_table_alloc(), which
47 * allocates the table FROM the slab pool so nginx's slab-pool header survives at
48 * shm.addr (ngx_unlock_mutexes() force-unlocks that header on every child death;
49 * laying our own struct over it would SIGSEGV the master). The helper zero-fills
50 * the table and creates the worker-local mutex from tbl->lock (the table's first
51 * member). On a brand-new allocation we set the capacity; on reuse we must not,
52 * to preserve the live table state.
53 */
54ngx_int_t
55brix_srv_shm_init_zone(ngx_shm_zone_t *shm_zone, void *data)
56{
57 brix_srv_table_t *tbl;
58 ngx_flag_t fresh;
59 size_t table_bytes;
60
61 table_bytes = sizeof(brix_srv_table_t)
62 + (size_t) brix_srv_registry_nslots
63 * sizeof(brix_srv_entry_t);
64
65 tbl = brix_shm_table_alloc(shm_zone, data, table_bytes,
66 &brix_srv_mutex, &fresh);
67 if (tbl == NULL) {
68 return NGX_ERROR;
69 }
70
71 if (fresh) {
72 tbl->capacity = brix_srv_registry_nslots;
73 }
74
75 return NGX_OK;
76}

80/* WHAT
81 * Allocates the shared-memory zone for the server registry and sets its size
82 * based on the configured slot count.
83
84 * WHY
85 * Called during nginx configuration parsing (brix_registry_slots directive).
86 * Must happen before any traffic so that workers can find the zone at startup.
87
88 * HOW
89 * Sets brix_srv_registry_nslots to the requested value, computes zone size
90 * via brix_shm_zone_size(table_bytes) — the table is allocated FROM the slab
91 * pool, so the zone must hold the table plus slab overhead — adds the zone via
92 * ngx_shared_memory_add(), sets init callback and (void *)1 sentinel data.
93 */
94ngx_int_t
95brix_srv_configure_registry(ngx_conf_t *cf, ngx_uint_t slots)
96{
97 ngx_str_t zone_name = ngx_string("brix_srv_registry");
98 size_t zone_size;
99
100 brix_srv_registry_nslots = slots;
101 zone_size = brix_shm_zone_size(
102 sizeof(brix_srv_table_t)
103 + (size_t) slots * sizeof(brix_srv_entry_t));
104 brix_srv_shm_zone = ngx_shared_memory_add(cf, &zone_name,
105 zone_size,
106 &ngx_stream_brix_module);
107 if (brix_srv_shm_zone == NULL) {
108 return NGX_ERROR;
109 }
110
111 brix_shm_zone_warn_on_resize(cf, brix_srv_shm_zone,
112 "brix_registry_slots");
113
114 brix_srv_shm_zone->init = brix_srv_shm_init_zone;
115 brix_srv_shm_zone->data = (void *) 1;
116
117 return NGX_OK;
118}

122/* WHAT
123 * Registers or updates a data server entry in the shared-memory registry.
124 * Called by CMS server handler when a data server logs in or heartbeats.
125
126 * WHY
127 * The redirector needs to know which servers exist, what paths they serve,
128 * and their current load metrics so that kXR_locate / kXR_open can pick the
129 * best server for each request.
130
131 * HOW
132 * Locks mutex → scans all slots: if host+port match found, update paths/free/
133 * util/last_seen fields. If no match and a free slot exists, allocate it with
134 * host/port/paths/free/util/last_seen/in_use=1. If registry is full, log warn
135 * and increment registry_full_total Prometheus counter.
136 */
137void
138brix_srv_register(const char *host, uint16_t port,
139 const char *paths, uint32_t free_mb, uint32_t util_pct)
140{
141 brix_srv_table_t *tbl;
142 brix_srv_entry_t *e;
143 ngx_uint_t i, free_slot;
144 int found;
145
146 tbl = srv_table();
147 if (tbl == NULL) {
148 return;
149 }
150
151 /*
152 * W1c — reject any host string that is not a clean hostname / IP literal
153 * before it can enter the registry. This is the single store choke point,
154 * so it also protects every redirect-emit path (brix_srv_select /
155 * brix_srv_locate_all) from control-byte or scheme injection into the
156 * "S<r|w>host:port" string a client parses. Registry hosts are normally
157 * the peer IP from ngx_sock_ntop, so a rejection here means a poisoned or
158 * malformed registration attempt.
159 */
160 if (host == NULL || !brix_net_host_chars_valid(host, ngx_strlen(host))) {
161 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
162 "brix: rejected registration with invalid host string");
163 return;
164 }
165
166 ngx_shmtx_lock(&brix_srv_mutex);
167
168 free_slot = tbl->capacity;
169 found = 0;
170
171 for (i = 0; i < tbl->capacity; i++) {
172 e = &tbl->slots[i];
173 if (!e->in_use) {
174 if (free_slot == tbl->capacity) {
175 free_slot = i;
176 }
177 continue;
178 }
179 if (e->port == port && ngx_strcmp(e->host, host) == 0) {
180 /* Update existing entry; clear any prior blacklist on reconnect. */
181 ngx_cpystrn((u_char *) e->paths,
182 (u_char *) (paths ? paths : ""),
183 sizeof(e->paths));
184 e->free_mb = free_mb;
185 e->util_pct = util_pct;
186 e->last_seen = ngx_current_msec;
187 e->blacklisted_until = 0;
188 e->error_count = 0;
189 found = 1;
190 break;
191 }
192 }
193
194 if (!found && free_slot < tbl->capacity) {
195 e = &tbl->slots[free_slot];
196 ngx_cpystrn((u_char *) e->host, (u_char *) host, sizeof(e->host));
197 e->port = port;
198 ngx_cpystrn((u_char *) e->paths,
199 (u_char *) (paths ? paths : ""),
200 sizeof(e->paths));
201 e->free_mb = free_mb;
202 e->util_pct = util_pct;
203 e->last_seen = ngx_current_msec;
204 e->in_use = 1;
205 } else if (!found) {
206 /* Registry is full: log a warning and increment the Prometheus counter. */
207 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
208 "brix: server registry full (%ui slots); "
209 "dropping registration for %s:%ui "
210 "(increase brix_registry_slots)",
211 tbl->capacity, host, (ngx_uint_t) port);
212 {
213 ngx_brix_metrics_t *m = brix_metrics_shared();
214 if (m != NULL) {
215 ngx_atomic_fetch_add(&m->registry_full_total, 1);
216 }
217 }
218 }
219
220 ngx_shmtx_unlock(&brix_srv_mutex);
221}

225/* WHAT
226 * Refreshes free-space and utilisation metrics for an already-registered server.
227 * Called on each CMS heartbeat from the data server.
228
229 * WHY
230 * Selection policy depends on current load: reads pick least-loaded servers,
231 * writes pick most-free-space. Metrics must stay fresh for accurate routing.
232
233 * HOW
234 * Locks mutex → scans slots for host+port match → updates free_mb, util_pct,
235 * last_seen fields only (no path changes). Unlocks and returns.
236 */
237void
238brix_srv_update_load(const char *host, uint16_t port,
239 uint32_t free_mb, uint32_t util_pct)
240{
241 brix_srv_table_t *tbl;
242 brix_srv_entry_t *e;
243 ngx_uint_t i;
244
245 tbl = srv_table();
246 if (tbl == NULL) {
247 return;
248 }
249
250 ngx_shmtx_lock(&brix_srv_mutex);
251
252 for (i = 0; i < tbl->capacity; i++) {
253 e = &tbl->slots[i];
254 if (e->in_use && e->port == port
255 && ngx_strcmp(e->host, host) == 0)
256 {
257 e->free_mb = free_mb;
258 e->util_pct = util_pct;
259 e->last_seen = ngx_current_msec;
260 break;
261 }
262 }
263
264 ngx_shmtx_unlock(&brix_srv_mutex);
265}

269/* WHAT
270 * Removes a data server entry from the registry by host+port match.
271 * Called when a data server disconnects or is removed from the cluster.
272
273 * WHY
274 * Prevents stale entries from being selected by locate/open operations. A
275 * disconnected server should not receive client traffic.
276
277 * HOW
278 * Locks mutex → scans slots for host+port match → zero-fills the entry (all
279 * fields cleared, in_use=0). Unlocks and returns.
280 */
281void
282brix_srv_unregister(const char *host, uint16_t port)
283{
284 brix_srv_table_t *tbl;
285 brix_srv_entry_t *e;
286 ngx_uint_t i;
287
288 tbl = srv_table();
289 if (tbl == NULL) {
290 return;
291 }
292
293 ngx_shmtx_lock(&brix_srv_mutex);
294
295 for (i = 0; i < tbl->capacity; i++) {
296 e = &tbl->slots[i];
297 if (e->in_use && e->port == port
298 && ngx_strcmp(e->host, host) == 0)
299 {
300 ngx_memzero(e, sizeof(*e));
301 break;
302 }
303 }
304
305 ngx_shmtx_unlock(&brix_srv_mutex);
306}
