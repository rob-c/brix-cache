1/*
2 * cta_service.c — the "cta" SSI service. See cta_service.h.
3 *
4 * Two phases (the defer-or-respond pattern):
5 * submit — decode the cta.xrd.Request, queue it, and defer (so a long
6 * archive/retrieve answers asynchronously with progress alerts).
7 * query/error answer inline.
8 * completion — run the executor on the stashed queue entry (alerts pushed live),
9 * then answer with a cta.xrd.Response.
10 */

78int
79brix_ssi_cta_process(const unsigned char *req, size_t req_len,
80 brix_ssi_responder_t *r)
81{
82 void **slot = r->svc_slot != NULL ? r->svc_slot(r) : NULL;
83 cta_req_t *e = slot != NULL ? (cta_req_t *) *slot : NULL;
84
85 if (e == NULL) {
86 /* ---- submit phase ---- */
87 cta_request_t creq;
88
89 if (cta_pb_decode_request(req, req_len, &creq) != 0) {
90 cta_respond(r, CTA_RSP_ERR_PROTOBUF, "malformed CTA request", 0);
91 return 0;
92 }
93 if (creq.op == CTA_OP_QUERY) {
94 char msg[64];
95 snprintf(msg, sizeof(msg), "%d active request(s)",
96 cta_queue_active_count(cta_queue()));
97 cta_respond(r, CTA_RSP_SUCCESS, msg, 0);
98 return 0;
99 }
100 if (creq.op == CTA_OP_UNKNOWN) {
101 cta_respond(r, CTA_RSP_ERR_USER, "unsupported workflow event", 0);
102 return 0;
103 }
104 e = cta_queue_submit(cta_queue(), &creq, creq.owner_user);
105 if (e == NULL) {
106 cta_respond(r, CTA_RSP_ERR_CTA, "request queue full", 0);
107 return 0;
108 }
109 if (slot != NULL) {
110 *slot = e;
111 }
112 if (r->defer != NULL && r->defer(r) == 0) {
113 return 0; /* deferred — the executor runs at completion */
114 }
115 /* defer unavailable: fall through and run inline */
116 }
117
118 /* ---- completion phase ---- */
119 {
120 cta_progress_t prog = { cta_prog_alert, r };
121 int rc = cta_exec_run(cta_exec_vtbl(), e, &prog);
122 cta_respond(r, rc == 0 ? CTA_RSP_SUCCESS : CTA_RSP_ERR_CTA,
123 rc == 0 ? "request completed" : "request failed",
124 e->req.archive_id);
125 }
126 return 0;
127}
