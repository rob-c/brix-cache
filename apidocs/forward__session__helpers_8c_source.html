1/*
2 * WHAT: Session helper functions for the transparent XRootD proxy — audit logging,
3 * kXR_wait retry timer callback, local file handle allocation, and lazy open
4 * for bound secondary connections. These helpers bridge client session state
5 * with upstream server operations during opaque relay.
6 *
7 * WHY: The proxy must track per-file metadata (path, bytes read/written, duration)
8 * for audit compliance when files are accessed through the backend. kXR_wait
9 * responses from upstream require the proxy to silently re-issue saved requests
10 * without client involvement — a timer callback handles this asynchronously.
11 * Bound secondary connections share file handles with their primary session but
12 * need lazy-open to establish upstream mappings on demand before read/write ops.
13 * Local fh allocation provides the proxy's own handle namespace for each connection.
14 *
15 * HOW: proxy_write_audit() writes a JSON line (path, bytes_read, bytes_written, duration_ms)
16 * to conf->proxy.audit_log_fd when configured; skips if audit fd is invalid or
17 * local_fh slot is out of range. brix_proxy_wait_handler() fires on timer expiry,
18 * restores fwd_reqid/streamid/fh from saved retry data, resets response state,
19 * flushes the saved request to upstream, and arms read event.
20 * brix_proxy_alloc_local_fh() scans fh_map for a free slot (upstream_fh == FREE)
21 * returning the index or -1 if exhausted. brix_proxy_lazy_open() looks up the
22 * canonical path from shared session handle registry, builds a synthetic anonymous
23 * kXR_open with streamid[1]=0xfe, marks the slot pending (255), saves read_req for
24 * later dispatch by relay_to_client after open response arrives, and flushes.
25 */

30/* proxy_write_audit — append a JSON line (path, bytes_read, bytes_written,
31 * duration_ms) to the configured audit log for a local file-handle slot; no-op if
32 * the audit fd is invalid or the slot is out of range. */
33void
34proxy_write_audit(brix_proxy_ctx_t *proxy, int local_fh)
35{
36 ngx_stream_brix_srv_conf_t *conf = proxy->conf;
37 brix_proxy_fh_entry_t *entry;
38 u_char buf[1024];
39 u_char *p;
40 ngx_msec_int_t duration_ms;
41
42 if (conf == NULL || conf->proxy.audit_log_fd == NGX_INVALID_FILE) {
43 return;
44 }
45 if (local_fh < 0 || local_fh >= BRIX_MAX_FILES) {
46 return;
47 }
48 entry = &proxy->fh_map[local_fh];
49
50 duration_ms = (ngx_msec_int_t)(ngx_current_msec - entry->open_msec);
51
52 p = ngx_snprintf(buf, sizeof(buf) - 2,
53 "{\"path\":\"%s\","
54 "\"bytes_read\":%uL,"
55 "\"bytes_written\":%uL,"
56 "\"duration_ms\":%M}\n",
57 entry->path,
58 (uint64_t) entry->bytes_read,
59 (uint64_t) entry->bytes_written,
60 duration_ms);
61
62 ngx_write_fd(conf->proxy.audit_log_fd, buf, (size_t)(p - buf));
63}

65/* brix_proxy_wait_handler — kXR_wait retry timer: restore the saved request
66 * metadata (reqid, streamid, local fh), reset response state, re-flush the saved
67 * request to upstream, and arm the read event; cleans up the proxy if the client
68 * context was destroyed. */
69void
70brix_proxy_wait_handler(ngx_event_t *ev)
71{
72 brix_proxy_ctx_t *proxy = ev->data;
73 brix_ctx_t *ctx;
74 u_char *req;
75 size_t len;
76 ngx_int_t rc;
77
78 if (proxy == NULL) {
79 return;
80 }
81 ctx = proxy->client_ctx;
82 if (ctx == NULL || ctx->destroyed) {
83 brix_proxy_cleanup(proxy);
84 return;
85 }
86
87 req = proxy->wait_retry_req;
88 len = proxy->wait_retry_req_len;
89 if (req == NULL || proxy->conn == NULL) {
90 return;
91 }
92
93 proxy->wait_retry_req = NULL;
94 proxy->wait_retry_req_len = 0;
95
96 /* Re-issue the saved request (any opcode, not only kXR_open) */
97 proxy->fwd_reqid = ntohs(((ClientRequestHdr *)(void *) req)->requestid);
98 proxy->fwd_streamid[0] = req[0];
99 proxy->fwd_streamid[1] = req[1];
100 proxy->fwd_local_fh = proxy->wait_retry_local_fh;
101 proxy->fwd_streaming = 0;
102 proxy->fwd_payload_len = len > XRD_REQUEST_HDR_LEN
103 ? len - XRD_REQUEST_HDR_LEN : 0;
104
105 proxy->wbuf = req;
106 proxy->wbuf_len = len;
107 proxy->wbuf_pos = 0;
108 proxy->state = XRD_PX_FORWARDING;
109
110 proxy->rhdr_pos = 0;
111 proxy->resp_dlen = 0;
112 proxy->resp_body = NULL;
113 proxy->resp_body_pos = 0;
114
115 rc = brix_proxy_flush(proxy);
116 if (rc == NGX_ERROR) {
117 ngx_free(req);
118 proxy->wbuf = NULL;
119 brix_proxy_abort(proxy, "proxy: wait retry send failed");
120 return;
121 }
122 if (proxy->wbuf_pos == proxy->wbuf_len) {
123 ngx_free(req);
124 proxy->wbuf = NULL;
125 if (ngx_handle_read_event(proxy->conn->read, 0) != NGX_OK) {
126 brix_proxy_abort(proxy, "proxy: read arm failed after wait retry");
127 }
128 }
129 /* else: write handler will complete the send */
130}

132/* brix_proxy_alloc_local_fh — return the first free slot (upstream_fh == FREE) in
133 * the proxy fh_map (its own file-handle namespace), or -1 if all are occupied. */
134int
135brix_proxy_alloc_local_fh(brix_proxy_ctx_t *proxy)
136{
137 int i;
138
139 for (i = 0; i < BRIX_MAX_FILES; i++) {
140 if (proxy->fh_map[i].upstream_fh == BRIX_PROXY_FH_FREE) {
141 return i;
142 }
143 }
144 return -1;
145}

149/*
150 * brix_proxy_lazy_open — issue a synthetic kXR_open on the upstream for a
151 * handle that was opened by the primary session.
152 *
153 * Used when a bound secondary connection (ctx->is_bound) sends kXR_read for a
154 * handle that has no upstream mapping yet in this connection's fh_map. The
155 * function looks up the canonical path from the shared session handle registry
156 * (where the primary published it), builds an anonymous kXR_open, and queues it.
157 * read_req (the original kXR_read) is saved as proxy->saved_req; it will be
158 * dispatched by relay_to_client once the open response arrives.
159 *
160 * Ownership of read_req transfers to this function on NGX_OK; caller must not
161 * free it. On error the function frees read_req itself.
162 */
163ngx_int_t
164brix_proxy_lazy_open(brix_proxy_ctx_t *proxy,
165 brix_ctx_t *ctx, ngx_connection_t *c,
166 int local_fh, u_char *read_req, size_t read_req_len)
167{
168 brix_shared_handle_entry_t he;
169 size_t pathlen, frame_len;
170 u_char *frame;
171 ClientOpenRequest *oreq;
172 ngx_int_t rc;
173
174 if (!brix_session_handle_lookup(ctx->bound_sessid, local_fh, &he)) {
175 ngx_free(read_req);
176 return brix_send_error(ctx, c, kXR_InvalidRequest,
177 "proxy: bound handle not published by primary");
178 }
179
180 pathlen = ngx_strlen(he.path);
181 frame_len = sizeof(ClientOpenRequest) + pathlen;
182
183 frame = ngx_alloc(frame_len, c->log);
184 if (frame == NULL) {
185 ngx_free(read_req);
186 return brix_send_error(ctx, c, kXR_IOError,
187 "proxy: OOM for lazy open");
188 }
189
190 oreq = (ClientOpenRequest *)(void *) frame;
191 ngx_memzero(oreq, sizeof(*oreq));
192 oreq->streamid[0] = 0;
193 oreq->streamid[1] = 0xfe; /* synthetic — not echoed to client */
194 oreq->requestid = htons(kXR_open);
195 {
196 xrdw_open_req_t b = { .options = kXR_open_read };
197 xrdw_open_req_pack(&b, ((ClientRequestHdr *) (void *) frame)->body);
198 }
199 oreq->dlen = htonl((kXR_int32) pathlen);
200 ngx_memcpy(frame + sizeof(*oreq), he.path, pathlen);
201
202 /* Mark the slot pending while we wait for the open response */
203 proxy->fh_map[local_fh].upstream_fh = 255;
204
205 /* Save the kXR_read so relay_to_client can dispatch it after open */
206 proxy->saved_req = read_req;
207 proxy->saved_req_len = read_req_len;
208 proxy->saved_local_fh = local_fh;
209
210 proxy->fwd_is_lazy_open = 1;
211 proxy->fwd_local_fh = local_fh;
212 proxy->fwd_reqid = kXR_open;
213 proxy->fwd_streaming = 0;
214 proxy->fwd_payload_len = 0;
215
216 proxy->wbuf = frame;
217 proxy->wbuf_len = frame_len;
218 proxy->wbuf_pos = 0;
219 proxy->state = XRD_PX_FORWARDING;
220
221 proxy->rhdr_pos = 0;
222 proxy->resp_dlen = 0;
223 proxy->resp_body = NULL;
224 proxy->resp_body_pos = 0;
225
226 ctx->state = XRD_ST_PROXY;
227
228 rc = brix_proxy_flush(proxy);
229 if (rc == NGX_ERROR) {
230 ngx_free(frame);
231 proxy->wbuf = NULL;
232 brix_proxy_abort(proxy, "proxy: lazy open send failed");
233 return NGX_ERROR;
234 }
235 if (proxy->wbuf_pos == proxy->wbuf_len) {
236 ngx_free(frame);
237 proxy->wbuf = NULL;
238 if (ngx_handle_read_event(proxy->conn->read, 0) != NGX_OK) {
239 brix_proxy_abort(proxy, "proxy: read arm failed after lazy open");
240 return NGX_ERROR;
241 }
242 }
243
244 return NGX_OK;
245}
