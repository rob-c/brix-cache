10/*
11 * prepare_qprep.c — kXR_QPrep staging-status query handler (split from
12 * prepare.c so each file owns one wire verb and stays under the size cap).
13 *
14 * WHAT: answers "is each of these paths resident?" for paths named in a prior
15 * kXR_prepare (or inline in the query). Emits one "<status> <path>" line
16 * per file: 'A' resident, 'M' missing/unauthorized, or a live FRM queue
17 * state ('q' queued, 's' staging, 'f' failed) when the durable queue has a
18 * record for a not-yet-resident file.
19 * WHY: clients poll QPrep to learn when a staged/recalled file has landed,
20 * across reconnects; the FRM queue-state passthrough lets a poll follow a
21 * real recall to completion instead of flapping on a bare 'M'.
22 * HOW: capture the request-id (first payload line), pick the path list (inline
23 * lines after the id, else the stored prepare list), then for each path
24 * run the SAME auth chain as prepare before reporting residency. A path
25 * that fails any auth tier reports 'M' — identical to xrootd reference
26 * behaviour and never an authorization oracle.
27 */
28
29/* Phase 35: map a durable queue status to the QPrep per-path status letter.
30 * 'A' available/online, 'q' queued, 's' staging, 'f' failed, 'M' missing. */
31static char
32qprep_status_letter(brix_stage_req_status_t s)
33{
34 switch (s) {
35 case BRIX_STAGE_REQ_QUEUED: return 'q';
36 case BRIX_STAGE_REQ_ACTIVE: return 's';
37 case BRIX_STAGE_REQ_DONE: return 'A';
38 case BRIX_STAGE_REQ_FAILED: return 'f';
39 default: return 'M';
40 }
41}

43/*
44 * Authorize the path through the same three tiers prepare uses (authdb VO/ACL,
45 * VO identity ACL, token scope). Boolean: a denial is silent (the caller maps
46 * it to 'M'), so this never sends a wire response.
47 */
48static ngx_flag_t
49qprep_path_authorized(brix_ctx_t *ctx, ngx_connection_t *c,
50 ngx_stream_brix_srv_conf_t *conf, const char *pathbuf, const char *full_path)
51{
52 return brix_authz_check(ctx, c, conf, pathbuf, full_path, "PREPARE",

54 && brix_check_vo_acl_identity(c->log, full_path, conf->vo_rules,
55 ctx->identity) == NGX_OK
56 && brix_check_token_scope(ctx, pathbuf, 0) == NGX_OK;
57}

59/*
60 * Residency status letter for one already-extracted logical path. Unauthorized
61 * or non-resident with no live queue record → 'M'; resident regular file → 'A';
62 * otherwise the FRM queue state when a durable record exists.
63 */
64static char
65qprep_status_for_path(brix_ctx_t *ctx, ngx_connection_t *c,
66 ngx_stream_brix_srv_conf_t *conf, const char *pathbuf, const char *full_path)
67{
68 struct stat st;
69
70 if (!qprep_path_authorized(ctx, c, conf, pathbuf, full_path)) {
71 return 'M';
72 }
73
74 if (brix_stat_beneath(conf->rootfd, pathbuf, &st) == 0 && S_ISREG(st.st_mode)) {
75 return 'A'; /* resident on disk */
76 }
77
78 if (conf->frm.enable && brix_stage_registry_singleton() != NULL) {
79 brix_stage_registry_t *reg = brix_stage_registry_singleton();
80 brix_stage_request_t qrec;
81 char frq[BRIX_STAGE_REQID_LEN];
82
83 if (brix_stage_request_find_by_path(reg, full_path, frq, sizeof(frq),
84 c->log) == NGX_OK
85 && brix_stage_request_get(reg, frq, &qrec, c->log) == NGX_OK)
86 {
87 return qprep_status_letter(qrec.status);
88 }
89 }
90
91 return 'M';
92}

94/*
95 * Capture the request-id from the first payload line (trimmed of trailing
96 * CR/NUL) into `reqid`, and return the pointer just past that line (the start of
97 * the inline path list, if any). `reqid` is left "" when absent/oversized.
98 */
99static const u_char *
100qprep_capture_reqid(const u_char *p, const u_char *end, char *reqid,
101 size_t reqid_sz)
102{
103 const u_char *rid = p;
104 size_t reqid_len;
105
106 while (p < end && *p != '\n') {
107 p++;
108 }
109 reqid_len = (size_t) (p - rid);
110 while (reqid_len > 0 && (rid[reqid_len - 1] == '\r' || rid[reqid_len - 1] == '\0')) {
111 reqid_len--;
112 }
113 if (reqid_len > 0 && reqid_len < reqid_sz) {
114 ngx_memcpy(reqid, rid, reqid_len);
115 reqid[reqid_len] = '\0';
116 } else {
117 reqid[0] = '\0';
118 }
119
120 if (p < end) {
121 p++; /* consume '\n' */
122 }
123 return p;
124}

126/*
127 * Choose the path list to report on: inline lines after the reqid, else the
128 * session's stored prepare list. Returns NGX_DECLINED with src and src_len set
129 * when a list is available (proceed); otherwise it has already sent the terminal
130 * response (an ok with no body, or — for a named id we never issued — a
131 * kXR_ArgInvalid reject like reference do_Prepare(isQuery)) and returns that rc.
132 */
133static ngx_int_t
134qprep_resolve_src(brix_ctx_t *ctx, ngx_connection_t *c,
135 ngx_stream_brix_srv_conf_t *conf, const u_char *p, const u_char *end,
136 const char *reqid, const u_char **src, size_t *src_len)
137{
138 if (p < end) {
139 *src = p;
140 *src_len = (size_t) (end - p);
141 return NGX_DECLINED;
142 }
143
144 if (ctx->prepare.paths != NULL && ctx->prepare.paths_len > 0) {
145 *src = ctx->prepare.paths;
146 *src_len = ctx->prepare.paths_len;
147 return NGX_DECLINED;
148 }
149
150 if (reqid[0] != '\0') {
151 brix_stage_request_t rec;
152 int known = (conf->frm.enable
153 && brix_stage_registry_singleton() != NULL
154 && brix_stage_request_get(
155 brix_stage_registry_singleton(), reqid, &rec,
156 c->log) == NGX_OK);
157 if (!known) {
158 return brix_prepare_send_fail(ctx, c, reqid, kXR_ArgInvalid,
159 "Prepare requestid owned by an unknown server");
160 }
161 }
162 return brix_send_ok(ctx, c, NULL, 0);
163}

165/*
166 * Emit one "<status> <path>\n" record per non-empty path line in [src, src_len)
167 * into resp (capacity resp_cap), skipping malformed paths and truncating on
168 * overflow. Returns the number of bytes written (0 = nothing emitted).
169 */
170static size_t
171qprep_emit_all(brix_ctx_t *ctx, ngx_connection_t *c,
172 ngx_stream_brix_srv_conf_t *conf, const u_char *src, size_t src_len,
173 u_char *resp, size_t resp_cap)
174{
175 const u_char *p = src;
176 const u_char *end = src + src_len;
177 u_char *rp = resp;
178
179 while (p < end) {
180 const u_char *line;
181 size_t line_len;
182 char pathbuf[BRIX_MAX_PATH + 1];
183 char full_path[PATH_MAX];
184 char status_ch;
185
186 line = p;
187 while (p < end && *p != '\n') {
188 p++;
189 }
190 line_len = (size_t) (p - line);
191 if (p < end) {
192 p++;
193 }
194 while (line_len > 0
195 && (line[line_len - 1] == '\r' || line[line_len - 1] == '\0'))
196 {
197 line_len--;
198 }
199 if (line_len == 0) {
200 continue;
201 }
202
203 if (!brix_extract_path(c->log, line, line_len, pathbuf,
204 sizeof(pathbuf), 1)) {
205 continue; /* skip malformed paths */
206 }
207
208 /* Both non-existent and unauthorized paths report 'M', consistent with
209 * xrootd reference behavior; a live FRM queue record reports q/s/f. */
210 /* phase74-fp: pathbuf is the request path, full_path the output buf. */
211 brix_beneath_full_path(conf->common.root_canon, pathbuf, /* NOLINT(readability-suspicious-call-argument) */
212 full_path, sizeof(full_path));
213 status_ch = qprep_status_for_path(ctx, c, conf, pathbuf, full_path);
214
215 *rp++ = (u_char) status_ch;
216 *rp++ = ' ';
217
218 /* Copy logical path into response. */
219 if ((size_t) (rp - resp) + line_len + 1 >= resp_cap) {
220 break; /* safety: truncate on overflow */
221 }
222 ngx_memcpy(rp, pathbuf, strlen(pathbuf));
223 rp += strlen(pathbuf);
224 *rp++ = '\n';
225 }
226
227 return (size_t) (rp - resp);
228}

230/*
231 * kXR_QPrep handler.
232 *
233 * Payload format (newline-separated):
234 * line 0: request ID (from kXR_prepare response)
235 * line 1+: optional paths to check (may be omitted; use stored path list)
236 *
237 * Response: one "A <path>" / "M <path>" (or q/s/f) line per file, NUL-terminated.
238 */
239ngx_int_t
240brix_query_prep_status(brix_ctx_t *ctx, ngx_connection_t *c,
241 ngx_stream_brix_srv_conf_t *conf)
242{
243 const u_char *src = NULL;
244 size_t src_len = 0;
245 const u_char *p;
246 const u_char *end;
247 u_char *resp;
248 size_t resp_cap;
249 size_t nout;
250 char reqid[BRIX_STAGE_REQID_LEN];
251 ngx_int_t rc;
252
253 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
254 return brix_send_ok(ctx, c, NULL, 0);
255 }
256
257 p = ctx->recv.payload;
258 end = ctx->recv.payload + ctx->recv.cur_dlen;
259
260 /* The first payload line is the prepare request-id. Capture it so an id we
261 * have no record of can be rejected the way reference do_Prepare(isQuery)
262 * does. */
263 p = qprep_capture_reqid(p, end, reqid, sizeof(reqid));
264
265 rc = qprep_resolve_src(ctx, c, conf, p, end, reqid, &src, &src_len);
266 if (rc != NGX_DECLINED) {
267 return rc; /* no path list — response (ok/reject) already sent */
268 }
269
270 /* Allocate response buffer: worst case "A " + path + "\n" per line. */
271 resp_cap = src_len * 2 + 64;
272 BRIX_PALLOC_OR_RETURN(resp, c->pool, resp_cap,
273 brix_send_error(ctx, c, kXR_NoMemory, "out of memory"));
274
275 nout = qprep_emit_all(ctx, c, conf, src, src_len, resp, resp_cap);
276 if (nout == 0) {
277 return brix_send_ok(ctx, c, NULL, 0);
278 }
279
280 resp[nout] = '\0';
281 brix_log_access(ctx, c, "QPREP", "-", "-", 1, kXR_ok, NULL, 0);
282 return brix_send_ok(ctx, c, resp, (uint32_t) (nout + 1));
283}
