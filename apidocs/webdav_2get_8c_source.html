18/* GET range/bytes metrics — shared by the inline serve and the off-loop serve
19 * completion (brix_http_serve_offload), so both report identically. */
20static void
21webdav_serve_metrics(ngx_http_request_t *r,
22 const brix_http_serve_result_t *result)
23{
24 if (result->range_result == BRIX_SERVE_RANGE_UNSATISFIED) {
25 BRIX_WEBDAV_METRIC_INC(range_total[BRIX_WEBDAV_RANGE_UNSATISFIED]);
26 } else if (result->range_result == BRIX_SERVE_RANGE_PARTIAL) {
27 BRIX_WEBDAV_METRIC_INC(range_total[BRIX_WEBDAV_RANGE_PARTIAL]);
28 } else {
29 BRIX_WEBDAV_METRIC_INC(range_total[BRIX_WEBDAV_RANGE_FULL]);
30 }
31 if (result->bytes_sent > 0) {
32 BRIX_WEBDAV_METRIC_ADD(bytes_tx_total, (size_t) result->bytes_sent);
33 if (r->connection && r->connection->sockaddr) {
34 if (r->connection->sockaddr->sa_family == AF_INET6) {
35 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv6_total,
36 (size_t) result->bytes_sent);
37 } else {
38 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv4_total,
39 (size_t) result->bytes_sent);
40 }
41 }
42 }
43}

61static ngx_int_t
62webdav_register_send_fd_cleanup(ngx_http_request_t *r, ngx_fd_t fd,
63 const char *path)
64{
65 ngx_pool_cleanup_t *cln;
66 ngx_pool_cleanup_file_t *clnf;
67 size_t path_len;
68 u_char *name;
69
70 cln = ngx_pool_cleanup_add(r->pool, sizeof(ngx_pool_cleanup_file_t));
71 if (cln == NULL) {
72 return NGX_ERROR;
73 }
74
75 path_len = ngx_strlen(path);
76 name = ngx_pnalloc(r->pool, path_len + 1);
77 if (name == NULL) {
78 return NGX_ERROR;
79 }
80 ngx_cpystrn(name, (u_char *) path, path_len + 1);
81
82 cln->handler = ngx_pool_cleanup_file;
83 clnf = cln->data;
84 clnf->fd = fd;
85 clnf->name = name;
86 clnf->log = r->pool->log;
87
88 return NGX_OK;
89}

102 * get_serve_state_t — the serve-ready handle + derived metadata produced by
103 * the resolve/open phase and consumed by the serving phases. It bundles the
104 * open VFS handle with the POSIX-shaped stat (`sb`, used by the multipart and
105 * pre-header paths), the VFS stat (`vst`, used by the ranged serve), and the
106 * fd-cache bookkeeping (`from_cache`/`cache_path`) so the phase helpers take a
107 * single explicit state object instead of a long parameter list. It is filled

113 brix_vfs_stat_t vst; /* VFS stat for the ranged serve */
114 ngx_uint_t from_cache; /* fd came from the read-through cache tier */
115 const char *cache_path; /* cache path for access accounting */
116} get_serve_state_t;

118/*
119 * get_zip_member_serve — Phase-57 W2 ZIP member access over HTTP GET.
120 *
121 * WHAT: when the location is a ZIP-archive export, serve the requested member
122 * of the archive at `path` rather than the whole archive file.
123 * WHY: archive auth already ran in the access phase; a GET on a zip export
124 * selects one member by argument, and this must short-circuit the normal
125 * file-serving pipeline.
126 * HOW: parse the member argument; a negative result is a malformed request
127 * (400), a positive result serves the member and yields a terminal status,
128 * and zero means "no member requested" — signalled by NGX_DECLINED so the
129 * caller falls through to the normal whole-file path.
130 */
131static ngx_int_t
132get_zip_member_serve(ngx_http_request_t *r,
133 ngx_http_brix_webdav_loc_conf_t *conf, const char *path)
134{
135 char member[WEBDAV_MAX_PATH];
136 int zr;
137
138 if (!conf->zip_access) {
139 return NGX_DECLINED;
140 }
141
142 zr = brix_zip_http_member_arg(r, member, sizeof(member));
143 if (zr < 0) {
144 return NGX_HTTP_BAD_REQUEST;
145 }
146 if (zr > 0) {
147 return brix_zip_http_serve(r, conf->common.root_canon,
148 conf->zip_cd_max_bytes, path, member);
149 }
150 return NGX_DECLINED;
151}

153/*
154 * get_init_vfs_ctx — build the VFS context for the GET.
155 *
156 * WHAT: initialise `vctx` for a WebDAV read of `path`, bind the per-user
157 * backend credential / opt-in mint / delegation, and select the export's
158 * storage backend instance.
159 * WHY: every downstream open/offload/fill step reaches storage through this
160 * context; assembling it in one place keeps the credential-binding order
161 * (cred → mint → deleg) explicit and identical to the historical flow.
162 * HOW: detect TLS, init the context, then apply the three credential binders
163 * and the backend instance in sequence. Pure setup, no I/O.
164 */
165static void
166get_init_vfs_ctx(ngx_http_request_t *r,
167 ngx_http_brix_webdav_loc_conf_t *conf,
168 ngx_http_brix_webdav_req_ctx_t *wctx, const char *path,
169 brix_vfs_ctx_t *vctx)
170{
171 int is_tls = 0;
172#if (NGX_HTTP_SSL)
173 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
174#endif
175 brix_vfs_ctx_init(vctx, r->pool, r->connection->log,
176 BRIX_PROTO_WEBDAV, conf->common.root_canon,
177 conf->cache_root_canon, conf->common.allow_write, is_tls,
178 (wctx != NULL) ? wctx->identity : NULL, path);
179 brix_vfs_ctx_bind_backend_cred(vctx,
180 &conf->common.storage_credential_dir,
181 conf->common.storage_credential_fallback);

184 brix_vfs_ctx_bind_backend_mint(vctx,

186 &conf->common.storage_credential_mint_ca_key,
187 conf->common.storage_credential_mint_ttl);
188 webdav_vfs_bind_deleg(r, conf, vctx);
189
190 /* Route through the export's selected storage backend (NULL ⇒ default POSIX). */
191 vctx->sd = brix_webdav_backend_instance(conf, r->connection->log);
192}

194/*
195 * get_offload_or_fill — run the off-loop serve/fill fast paths.
196 *
197 * WHAT: give the socket-wire serve offload (SP3) and the remote cache-fill
198 * offload (SP2) a chance to handle the request off the event loop.
199 * WHY: a root://-backed or remote-store export cannot open/read on the worker
200 * loop, and a remote cache MISS fill would stall it; both are pushed to the
201 * thread pool and completed (or re-entered) asynchronously.
202 * HOW: returns NGX_DONE when a path took over (async in flight), a terminal
203 * HTTP status on error, or NGX_DECLINED when neither applies and the caller
204 * should open inline. EACCES/EPERM from the offload is the per-user backend
205 * credential gate → 403; any other offload error → 500.
206 */
207static ngx_int_t
208get_offload_or_fill(ngx_http_request_t *r,
209 ngx_http_brix_webdav_loc_conf_t *conf,
210 ngx_http_brix_webdav_req_ctx_t *wctx, const char *path,
211 brix_vfs_ctx_t *vctx)
212{
213 const char *identity =
214 (wctx != NULL && wctx->dn[0] != '\0') ? wctx->dn : "anonymous";
215 brix_http_serve_opts_t sopts;
216 ngx_int_t sr;
217 ngx_int_t fr;
218
219 ngx_memzero(&sopts, sizeof(sopts));
220 sopts.xfer_proto = BRIX_XFER_PROTO_WEBDAV;
221 sopts.op_name = "GET";
222 sopts.identity = identity;
223 sopts.etag_flags = BRIX_ETAG_WEAK;
224 sopts.compress = conf->common.compress;
225
226 sr = brix_http_serve_offload_remote(r, vctx->sd,
227 brix_vfs_export_relative(vctx, path), path, &sopts,
228 &conf->common, vctx, webdav_serve_metrics);
229 if (sr == NGX_DONE) {
230 return NGX_DONE;
231 }
232 if (sr == NGX_ERROR) {
233 return (errno == EACCES || errno == EPERM)
234 ? NGX_HTTP_FORBIDDEN : NGX_HTTP_INTERNAL_SERVER_ERROR;
235 }
236
237 fr = brix_http_cache_fill_if_needed(r, vctx->sd,
238 brix_vfs_export_relative(vctx, path), &conf->common,
239 webdav_get_reenter, NULL);
240 if (fr == NGX_DONE) {
241 return NGX_DONE;
242 }
243 if (fr == NGX_ERROR) {
244 return NGX_HTTP_INTERNAL_SERVER_ERROR;
245 }
246 return NGX_DECLINED;
247}

249/*
250 * get_open_map_error — map a failed VFS open to its terminal HTTP status.
251 *
252 * WHAT: translate `vfs_err` from a NULL `brix_vfs_open` into the exact response
253 * the historical inline code produced.
254 * WHY: the open error mapping carries security-load-bearing decisions
255 * (confinement rejections are 403, never 500) and a special-cased 202 tape
256 * recall — keeping it in one helper preserves those byte-for-byte.
257 * HOW: early-return per errno class: ENOENT/ENOTDIR/ENAMETOOLONG → 404 (with
258 * xrdhttp headers), confinement/permission (EACCES/EPERM/EXDEV/ELOOP) → 403,
259 * EAGAIN → 202 + Retry-After, otherwise log and route through the shared
260 * errno→status table.
261 */
262static ngx_int_t
263get_open_map_error(ngx_http_request_t *r, int vfs_err, const char *path)
264{
265 ngx_table_elt_t *ra;
266
267 if (vfs_err == ENOENT || vfs_err == ENOTDIR
268 || vfs_err == ENAMETOOLONG)
269 {
270 xrdhttp_add_response_headers(r, NGX_HTTP_NOT_FOUND);
271 return NGX_HTTP_NOT_FOUND;
272 }
273
274 /* EXDEV (".." escape) / ELOOP (escaping or magic symlink) are the
275 * kernel RESOLVE_BENEATH confinement rejections — forbidden, never a
276 * 500. EACCES/EPERM map the same way. Route the whole errno set
277 * through the shared table so the codes stay consistent with S3. */
278 if (vfs_err == EACCES || vfs_err == EPERM
279 || vfs_err == EXDEV || vfs_err == ELOOP)
280 {
281 return NGX_HTTP_FORBIDDEN;
282 }
283
284 /* EAGAIN ⇒ a nearline (tape) recall is in flight (sd_frm/sd_cache, §9.2).
285 * Answer 202 "staging" with a Retry-After so the client polls until the
286 * object is recalled into the cache tier and served — never block the
287 * worker for a minutes-to-hours MSS recall. */
288 if (vfs_err == EAGAIN) {
289 ra = ngx_list_push(&r->headers_out.headers);
290 if (ra != NULL) {
291 ra->hash = 1;
292 ngx_str_set(&ra->key, "Retry-After");
293 ngx_str_set(&ra->value, "10");
294 }
295 r->headers_out.status = NGX_HTTP_ACCEPTED;
296 r->headers_out.content_length_n = 0;
297 ngx_http_send_header(r);
298 return ngx_http_send_special(r, NGX_HTTP_LAST);
299 }
300
301 ngx_log_error(NGX_LOG_ERR, r->connection->log, vfs_err,
302 ngx_open_file_n " \"%s\" failed", path);
303 return (ngx_int_t) brix_http_errno_to_status(vfs_err);
304}

306/*
307 * get_resolve_and_stat — open the target and produce serve-ready state.
308 *
309 * WHAT: open `path` through the VFS, stat it, reject directories, and populate
310 * `st` (open handle + POSIX stat + VFS stat + fd-cache bookkeeping).
311 * WHY: the serving phases need an open, statted, non-directory regular-file
312 * handle plus the derived metadata; concentrating open+stat+validation here
313 * keeps the orchestrator flat and each error path owning its own close.
314 * HOW: open (mapping any failure via get_open_map_error), stat (500 on
315 * failure), directory guard (403), then fill `st.sb` from the VFS stat and
316 * capture the sendfile fd cache bookkeeping. Returns NGX_OK with `st`
317 * populated and the handle open, or a terminal status with nothing to close.
318 */
319static ngx_int_t
320get_resolve_and_stat(ngx_http_request_t *r, brix_vfs_ctx_t *vctx,
321 const char *path, get_serve_state_t *st)
322{
323 int vfs_err = 0;
324
325 st->fh = brix_vfs_open(vctx, BRIX_VFS_O_READ, &vfs_err);
326 if (st->fh == NULL) {
327 return get_open_map_error(r, vfs_err, path);
328 }
329
330 if (brix_vfs_file_stat(st->fh, &st->vst) != NGX_OK) {
331 brix_vfs_close(st->fh, r->connection->log);
332 return NGX_HTTP_INTERNAL_SERVER_ERROR;
333 }
334
335 if (st->vst.is_directory) {
336 brix_vfs_close(st->fh, r->connection->log);
337 return NGX_HTTP_FORBIDDEN;
338 }
339
340 ngx_memzero(&st->sb, sizeof(st->sb));
341 st->sb.st_size = st->vst.size;
342 st->sb.st_mtime = st->vst.mtime;
343 st->sb.st_ctime = st->vst.ctime;
344 st->sb.st_mode = (mode_t) st->vst.mode;
345 st->sb.st_ino = st->vst.ino;
346
347 /* Zero-copy (sendfile) serve fd, gated on the backend's CAP_SENDFILE; a
348 * non-sendfile backend returns NGX_INVALID_FILE and the dup in the
349 * multirange path fails closed instead of serving a bogus descriptor. */
350 st->from_cache = brix_vfs_file_from_cache(st->fh);
351 st->cache_path = brix_vfs_file_path(st->fh);
352 return NGX_OK;
353}

355/*
356 * get_serve_range — serve a multi-range (multipart/byteranges) GET.
357 *
358 * WHAT: handle the XrdHttp multi-range vector read by duplicating the sendfile
359 * fd, registering its cleanup, and delegating to the multipart handler.
360 * WHY: multi-range is the kXR_readv-over-HTTP path; the TLS-memory vs
361 * cleartext-sendfile split lives inside xrdhttp_handle_multipart_get and is
362 * FROZEN — this helper only owns the dup + cleanup + cache accounting around
363 * it. The dup lets the multipart handler own an independent fd while the VFS
364 * handle is closed here.
365 * HOW: dup the sendfile fd (500 on failure), register a pool cleanup that owns
366 * the dup, close the VFS handle, run the multipart serve, and record cache
367 * access on a successful full body.
368 */
369static ngx_int_t
370get_serve_range(ngx_http_request_t *r, get_serve_state_t *st, const char *path)
371{
372 ngx_fd_t fd;
373 ngx_fd_t send_fd;
374 ngx_int_t rc;
375
376 fd = brix_vfs_file_sendfile_fd(st->fh);
377
378 send_fd = dup(fd);
379 if (send_fd == NGX_INVALID_FILE) {
380 brix_vfs_close(st->fh, r->connection->log);
381 return NGX_HTTP_INTERNAL_SERVER_ERROR;
382 }
383 if (webdav_register_send_fd_cleanup(r, send_fd, path) != NGX_OK) {
384 ngx_close_file(send_fd);
385 brix_vfs_close(st->fh, r->connection->log);
386 return NGX_HTTP_INTERNAL_SERVER_ERROR;
387 }
388 brix_vfs_close(st->fh, r->connection->log);
389
390 rc = xrdhttp_handle_multipart_get(r, send_fd, &st->sb, 1);
391 if (st->from_cache && rc == NGX_OK && !r->header_only) {
392 (void) brix_cache_record_access(st->cache_path,
393 (size_t) st->sb.st_size, r->connection->log);
394 }
395 return rc;
396}

398/*
399 * get_eval_conditionals — apply the If-Modified-Since precondition.
400 *
401 * WHAT: evaluate the conditional GET via the shared eval and, on a 304, emit
402 * the not-modified response.
403 * WHY: a matching precondition must short-circuit the serve with an empty 304
404 * body; the check owns closing the VFS handle on every terminal outcome.
405 * HOW: NGX_HTTP_NOT_MODIFIED → close, send 304 headers + special last; any
406 * other non-OK → close and propagate; NGX_OK → return NGX_OK (handle stays
407 * open for the serve phase).
408 */
409static ngx_int_t
410get_eval_conditionals(ngx_http_request_t *r, get_serve_state_t *st)
411{
412 ngx_int_t rc = brix_http_check_if_modified_since(r, st->sb.st_mtime);
413
414 if (rc == NGX_HTTP_NOT_MODIFIED) {
415 brix_vfs_close(st->fh, r->connection->log);
416 r->headers_out.status = NGX_HTTP_NOT_MODIFIED;
417 r->headers_out.content_length_n = 0;
418 ngx_http_send_header(r);
419 return ngx_http_send_special(r, NGX_HTTP_LAST);
420 }
421 if (rc != NGX_OK) {
422 brix_vfs_close(st->fh, r->connection->log);
423 return rc;
424 }
425 return NGX_OK;
426}

466/*
467 * webdav_handle_get — serve a file via HTTP GET with Range support.
468 *
469 * Fast path: if the fd-cache already holds an open fd for the requested URI
470 * hash, the stat and open system calls are skipped entirely. The cached fd
471 * remains owned by the fd-cache; the cleanup handler registered below uses
472 * NGX_INVALID_FILE so it does not close it a second time.
473 *
474 * Range handling: a single "bytes=start-end" or "bytes=-suffix" range is
475 * parsed and served as 206 Partial Content. Multi-range requests and
476 * overlapping ranges are not supported; clients that send them receive the
477 * full file (200 OK).
478 *
479 * ngx_http_send_header + r->header_only: after calling ngx_http_send_header(),
480 * always check r->header_only. If true, the client sent HEAD — return
481 * immediately without sending a body. The check inside the serve phases
482 * handles this.
483 *
484 * Pool allocation: ngx_pcalloc(r->pool, ...) for ngx_buf_t and ngx_file_t —
485 * both are freed when the request pool is destroyed after the response
486 * is sent.
487 *
488 * Ownership of fd:
489 * - If fd came from the fd-cache (fd_from_table=1), the cleanup handler
490 * stores NGX_INVALID_FILE so the fd-cache retains ownership.
491 * - If fd was opened here (fd_from_table=0), the cleanup handler closes it.
492 *
493 * Flow: resolve path → (zip member?) → build VFS ctx → off-loop serve/fill →
494 * open+stat (get_resolve_and_stat) → multi-range (get_serve_range) or
495 * conditional (get_eval_conditionals) + full/single-range (get_serve_full).
496 */
497ngx_int_t
498webdav_handle_get(ngx_http_request_t *r)
499{
500 ngx_http_brix_webdav_loc_conf_t *conf;
501 char path[WEBDAV_MAX_PATH];
502 ngx_int_t rc;
503 ngx_http_brix_webdav_req_ctx_t *wctx;
504 brix_vfs_ctx_t vctx;
505 get_serve_state_t st = {0};
506
507 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
508
509 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon, path,
510 sizeof(path));
511 if (rc != NGX_OK) {
512 return rc;
513 }
514
515 /* Phase-57 W2: ZIP member access over HTTP GET. Auth on the archive ran in
516 * the access phase; serve the requested member instead of the whole file. */
517 rc = get_zip_member_serve(r, conf, path);
518 if (rc != NGX_DECLINED) {
519 return rc;
520 }
521
522 wctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
523 get_init_vfs_ctx(r, conf, wctx, path, &vctx);
524
525 /* phase-64 SP3/SP2: socket-wire serve offload and remote cache-fill offload
526 * (both off the event loop). NGX_DONE ⇒ async took over; a terminal HTTP
527 * status ⇒ error; NGX_DECLINED ⇒ open inline below. */
528 rc = get_offload_or_fill(r, conf, wctx, path, &vctx);
529 if (rc != NGX_DECLINED) {
530 return rc;
531 }
532
533 rc = get_resolve_and_stat(r, &vctx, path, &st);
534 if (rc != NGX_OK) {
535 return rc;
536 }
537
538 /* XrdHttp: multi-range vector read (kXR_readv equivalent over HTTP).
539 * A comma in the Range: value indicates multiple byte ranges — delegate
540 * to the multipart/byteranges handler rather than the single-range path. */
541 if (xrdhttp_request_is_multirange(r)) {
542 return get_serve_range(r, &st, path);
543 }
