18/* GET range/bytes metrics — shared by the inline serve and the off-loop serve
19 * completion (brix_http_serve_offload), so both report identically. */
20static void
21webdav_serve_metrics(ngx_http_request_t *r,
22 const brix_http_serve_result_t *result)
23{
24 if (result->range_result == BRIX_SERVE_RANGE_UNSATISFIED) {
25 BRIX_WEBDAV_METRIC_INC(range_total[BRIX_WEBDAV_RANGE_UNSATISFIED]);
26 } else if (result->range_result == BRIX_SERVE_RANGE_PARTIAL) {
27 BRIX_WEBDAV_METRIC_INC(range_total[BRIX_WEBDAV_RANGE_PARTIAL]);
28 } else {
29 BRIX_WEBDAV_METRIC_INC(range_total[BRIX_WEBDAV_RANGE_FULL]);
30 }
31 if (result->bytes_sent > 0) {
32 BRIX_WEBDAV_METRIC_ADD(bytes_tx_total, (size_t) result->bytes_sent);
33 if (r->connection && r->connection->sockaddr) {
34 if (r->connection->sockaddr->sa_family == AF_INET6) {
35 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv6_total,
36 (size_t) result->bytes_sent);
37 } else {
38 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv4_total,
39 (size_t) result->bytes_sent);
40 }
41 }
42 }
43}

45/* Re-entry trampoline for the off-event-loop cache fill: after the fill lands the
46 * completion event re-runs the GET handler, which now finds a cache HIT and serves
47 * it zero-copy. The fill helper carries no per-handler state (the request re-
48 * resolves from r), so `data` is unused. */
49static ngx_int_t
50webdav_get_reenter(ngx_http_request_t *r, void *data)
51{
52 (void) data;
53 return webdav_handle_get(r);
54}

61static ngx_int_t
62webdav_register_send_fd_cleanup(ngx_http_request_t *r, ngx_fd_t fd,
63 const char *path)
64{
65 ngx_pool_cleanup_t *cln;
66 ngx_pool_cleanup_file_t *clnf;
67 size_t path_len;
68 u_char *name;
69
70 cln = ngx_pool_cleanup_add(r->pool, sizeof(ngx_pool_cleanup_file_t));
71 if (cln == NULL) {
72 return NGX_ERROR;
73 }
74
75 path_len = ngx_strlen(path);
76 name = ngx_pnalloc(r->pool, path_len + 1);
77 if (name == NULL) {
78 return NGX_ERROR;
79 }
80 ngx_cpystrn(name, (u_char *) path, path_len + 1);
81
82 cln->handler = ngx_pool_cleanup_file;
83 clnf = cln->data;
84 clnf->fd = fd;
85 clnf->name = name;
86 clnf->log = r->pool->log;
87
88 return NGX_OK;
89}

101/*
102 * webdav_handle_get — serve a file via HTTP GET with Range support.
103 *
104 * Fast path: if the fd-cache already holds an open fd for the requested URI
105 * hash, the stat and open system calls are skipped entirely. The cached fd
106 * remains owned by the fd-cache; the cleanup handler registered below uses
107 * NGX_INVALID_FILE so it does not close it a second time.
108 *
109 * Range handling: a single "bytes=start-end" or "bytes=-suffix" range is
110 * parsed and served as 206 Partial Content. Multi-range requests and
111 * overlapping ranges are not supported; clients that send them receive the
112 * full file (200 OK).
113 *
114 * ngx_http_send_header + r->header_only: after calling ngx_http_send_header(),
115 * always check r->header_only. If true, the client sent HEAD — return
116 * immediately without sending a body. The check at line ~244 handles this.
117 *
118 * Pool allocation: ngx_pcalloc(r->pool, ...) for ngx_buf_t and ngx_file_t —
119 * both are freed when the request pool is destroyed after the response
120 * is sent.
121 *
122 * Ownership of fd:
123 * - If fd came from the fd-cache (fd_from_table=1), the cleanup handler
124 * stores NGX_INVALID_FILE so the fd-cache retains ownership.
125 * - If fd was opened here (fd_from_table=0), the cleanup handler closes it.
126 */
127ngx_int_t
128webdav_handle_get(ngx_http_request_t *r)
129{
130 ngx_http_brix_webdav_loc_conf_t *conf;
131 char path[WEBDAV_MAX_PATH];
132 struct stat sb;
133 ngx_int_t rc;
134 ngx_fd_t fd;
135 ngx_fd_t send_fd;
136 ngx_http_brix_webdav_req_ctx_t *wctx;
137 const char *identity;
138 brix_vfs_ctx_t vctx;
139 brix_vfs_file_t *fh;
140 brix_vfs_stat_t vst;
141 int vfs_err;
142 ngx_uint_t from_cache;
143 const char *cache_path;
144
145 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
146
147 rc = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon, path,
148 sizeof(path));
149 if (rc != NGX_OK) {
150 return rc;
151 }
152
153 /* Phase-57 W2: ZIP member access over HTTP GET. Auth on the archive ran in
154 * the access phase; serve the requested member instead of the whole file. */
155 if (conf->zip_access) {
156 char member[WEBDAV_MAX_PATH];
157 int zr = brix_zip_http_member_arg(r, member, sizeof(member));
158 if (zr < 0) {
159 return NGX_HTTP_BAD_REQUEST;
160 }
161 if (zr > 0) {
162 return brix_zip_http_serve(r, conf->common.root_canon,
163 conf->zip_cd_max_bytes, path, member);
164 }
165 }
166
167 wctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
168
169 {
170 int is_tls = 0;
171#if (NGX_HTTP_SSL)
172 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
173#endif
174 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log,
175 BRIX_PROTO_WEBDAV, conf->common.root_canon,
176 conf->cache_root_canon, conf->common.allow_write, is_tls,
177 (wctx != NULL) ? wctx->identity : NULL, path);
178 }
179 /* Route through the export's selected storage backend (NULL ⇒ default POSIX). */
180 vctx.sd = brix_webdav_backend_instance(conf, r->connection->log);
181
182 /* phase-64 SP3: serving from a socket-wire backend (a root:// primary backend
183 * or a cache_store/stage_store served from one) cannot open/read on the event
184 * loop. Run the whole open+read on the thread pool, materialise + sendfile.
185 * NGX_DECLINED ⇒ not a socket serve; fall through to the local fast paths. */
186 {
187 const char *identity =
188 (wctx != NULL && wctx->dn[0] != '\0') ? wctx->dn : "anonymous";
189 brix_http_serve_opts_t sopts;
190 ngx_int_t sr;
191
192 ngx_memzero(&sopts, sizeof(sopts));
193 sopts.xfer_proto = BRIX_XFER_PROTO_WEBDAV;
194 sopts.op_name = "GET";
195 sopts.identity = identity;
196 sopts.etag_flags = BRIX_ETAG_WEAK;
197 sopts.compress = conf->common.compress;
198
199 sr = brix_http_serve_offload_remote(r, vctx.sd,
200 brix_vfs_export_relative(&vctx, path), path, &sopts,
201 &conf->common, webdav_serve_metrics);
202 if (sr == NGX_DONE) {
203 return NGX_DONE;
204 }
205 if (sr == NGX_ERROR) {
206 return NGX_HTTP_INTERNAL_SERVER_ERROR;
207 }
208 }
209
210 /* phase-64 SP2: a cache MISS whose fill reads a remote source or writes a
211 * remote store would stall the worker inside brix_vfs_open's inline fill.
212 * Offload it to the thread pool and re-enter (a cache hit) on completion;
213 * NGX_DECLINED ⇒ no offload needed (hit / local tier) — open inline below. */
214 {
215 ngx_int_t fr = brix_http_cache_fill_if_needed(r, vctx.sd,
216 brix_vfs_export_relative(&vctx, path), &conf->common,
217 webdav_get_reenter, NULL);
218
219 if (fr == NGX_DONE) {
220 return NGX_DONE;
221 }
222 if (fr == NGX_ERROR) {
223 return NGX_HTTP_INTERNAL_SERVER_ERROR;
224 }
225 }
226
227 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
228 if (fh == NULL) {
229 if (vfs_err == ENOENT || vfs_err == ENOTDIR
230 || vfs_err == ENAMETOOLONG)
231 {
232 xrdhttp_add_response_headers(r, NGX_HTTP_NOT_FOUND);
233 return NGX_HTTP_NOT_FOUND;
234 }
235
236 /* EXDEV (".." escape) / ELOOP (escaping or magic symlink) are the
237 * kernel RESOLVE_BENEATH confinement rejections — forbidden, never a
238 * 500. EACCES/EPERM map the same way. Route the whole errno set
239 * through the shared table so the codes stay consistent with S3. */
240 if (vfs_err == EACCES || vfs_err == EPERM
241 || vfs_err == EXDEV || vfs_err == ELOOP)
242 {
243 return NGX_HTTP_FORBIDDEN;
244 }
245
246 /* EAGAIN ⇒ a nearline (tape) recall is in flight (sd_frm/sd_cache, §9.2).
247 * Answer 202 "staging" with a Retry-After so the client polls until the
248 * object is recalled into the cache tier and served — never block the
249 * worker for a minutes-to-hours MSS recall. */
250 if (vfs_err == EAGAIN) {
251 ngx_table_elt_t *ra = ngx_list_push(&r->headers_out.headers);
252
253 if (ra != NULL) {
254 ra->hash = 1;
255 ngx_str_set(&ra->key, "Retry-After");
256 ngx_str_set(&ra->value, "10");
257 }
258 r->headers_out.status = NGX_HTTP_ACCEPTED;
259 r->headers_out.content_length_n = 0;
260 ngx_http_send_header(r);
261 return ngx_http_send_special(r, NGX_HTTP_LAST);
262 }
263
264 ngx_log_error(NGX_LOG_ERR, r->connection->log, vfs_err,
265 ngx_open_file_n " \"%s\" failed", path);
266
267 return (ngx_int_t) brix_http_errno_to_status(vfs_err);
268 }
269
270 if (brix_vfs_file_stat(fh, &vst) != NGX_OK) {
271 brix_vfs_close(fh, r->connection->log);
272 return NGX_HTTP_INTERNAL_SERVER_ERROR;
273 }
274
275 if (vst.is_directory) {
276 brix_vfs_close(fh, r->connection->log);
277 return NGX_HTTP_FORBIDDEN;
278 }
279
280 ngx_memzero(&sb, sizeof(sb));
281 sb.st_size = vst.size;
282 sb.st_mtime = vst.mtime;
283 sb.st_ctime = vst.ctime;
284 sb.st_mode = (mode_t) vst.mode;
285 sb.st_ino = vst.ino;
286
287 /* Zero-copy (sendfile) serve fd, gated on the backend's CAP_SENDFILE; a
288 * non-sendfile backend returns NGX_INVALID_FILE and the dup below fails
289 * closed instead of serving from a bogus descriptor. */
290 fd = brix_vfs_file_sendfile_fd(fh);
291 from_cache = brix_vfs_file_from_cache(fh);
292 cache_path = brix_vfs_file_path(fh);
293
294 /* XrdHttp: multi-range vector read (kXR_readv equivalent over HTTP).
295 * A comma in the Range: value indicates multiple byte ranges — delegate
296 * to the multipart/byteranges handler rather than the single-range path. */
297 if (xrdhttp_request_is_multirange(r)) {
298 send_fd = dup(fd);
299 if (send_fd == NGX_INVALID_FILE) {
300 brix_vfs_close(fh, r->connection->log);
301 return NGX_HTTP_INTERNAL_SERVER_ERROR;
302 }
303 if (webdav_register_send_fd_cleanup(r, send_fd, path) != NGX_OK) {
304 ngx_close_file(send_fd);
305 brix_vfs_close(fh, r->connection->log);
306 return NGX_HTTP_INTERNAL_SERVER_ERROR;
307 }
308 brix_vfs_close(fh, r->connection->log);
309 rc = xrdhttp_handle_multipart_get(r, send_fd, &sb, 1);
310 if (from_cache && rc == NGX_OK && !r->header_only) {
311 (void) brix_cache_record_access(cache_path, (size_t) sb.st_size,
312 r->connection->log);
313 }
314 return rc;
315 }
316
317 rc = brix_http_check_if_modified_since(r, sb.st_mtime);
318 if (rc == NGX_HTTP_NOT_MODIFIED) {
319 brix_vfs_close(fh, r->connection->log);
320 r->headers_out.status = NGX_HTTP_NOT_MODIFIED;
321 r->headers_out.content_length_n = 0;
322 ngx_http_send_header(r);
323 return ngx_http_send_special(r, NGX_HTTP_LAST);
324 }
325 if (rc != NGX_OK) {
326 brix_vfs_close(fh, r->connection->log);
327 return rc;
328 }
329
330 identity = (wctx != NULL && wctx->dn[0] != '\0') ? wctx->dn : "anonymous";
331 r->allow_ranges = 1;
332
333 {
334 brix_http_serve_opts_t opts;
335 brix_http_serve_result_t result;
336
337 ngx_memzero(&opts, sizeof(opts));
338 opts.xfer_proto = BRIX_XFER_PROTO_WEBDAV;
339 opts.op_name = "GET";
340 opts.identity = identity;
341 opts.etag_flags = BRIX_ETAG_WEAK;
342 opts.compress = conf->common.compress;
343 opts.pre_header_send = webdav_get_add_xrdhttp_headers;
344 opts.pre_header_ud = &sb;
345
346 rc = brix_http_serve_file_ranged(r, fh, &vst, path, &opts, &result);
347 webdav_serve_metrics(r, &result);
348 }
349
350 return rc;
351}
