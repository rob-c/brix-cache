1/*
2 * cms/fanout.c — Phase-89 W8: manager-side rm/rmdir fan-out to all holders.
3 *
4 * WHAT: Forwards a manager-mode client kXR_rm/kXR_rmdir to every registered
5 * node whose exports cover the path (brix_cms_forward_to_node over this
6 * worker's accepted CMS connections), parks the client, aggregates the nodes'
7 * replies over a deadline window, and answers the client once — kXR_ok when no
8 * node errored, else kXR_error carrying the first node error's text.
9 *
10 * WHY: with replicated exports the shipped single-node redirect deletes one
11 * replica and strands the rest — stock cmsd fans namespace deletes out to all
12 * holders. See fanout.h for the WHY of the window design (the node executor
13 * is silent on success, so "no error before the deadline" IS the success
14 * signal, matching stock ofs.forward semantics).
15 *
16 * HOW: a small per-worker slot table keyed by the CMS streamid carries the
17 * aggregation state (expected/got_err/worst error + a deadline ngx_event_t);
18 * the client's identity rides the shared pending-locate table exactly like a
19 * parked locate, so the finalizer reuses the same recycle-guarded wake shape
20 * as brix_cms_wake_pending_session (fd + c->number + XRD_ST_WAITING_CMS).
21 * kYR_error replies arrive in server_recv_frame.c (CMS_RSP_ERROR route) and
22 * fold in through brix_cms_fanout_note_error. Per-worker by design: fan-out
23 * only engages when THIS worker owns a CMS connection to EVERY registry
24 * holder of the path, so replies and the parked client share one worker.
25 */

86/*
87 * Answer the parked client from the folded slot state and release the slot.
88 * Mirrors brix_cms_wake_pending_session's recycle-guarded resolution: the fd
89 * and connection generation ride the pending table; a client that timed out
90 * or disconnected meanwhile makes every guard a clean no-op (the deletes on
91 * the nodes have already happened either way — this only settles the reply).
92 */
93static void
94brix_cms_fanout_finalize(brix_cms_fanout_slot_t *slot, ngx_log_t *log)
95{
96 brix_pending_locate_t *pending;
97 ngx_connection_t *client_conn;
98 ngx_stream_session_t *session;
99 brix_ctx_t *xrd_ctx;
100 int conn_fd;
101 ngx_atomic_uint_t conn_number;
102 u_char client_streamid[2];
103
104 ngx_log_error(NGX_LOG_INFO, log, 0,
105 "brix: CMS fan-out: finalize sid=%uD errors %ui/%ui",
106 slot->sid, slot->got_err, slot->expected);
107
108 pending = brix_pending_lookup(slot->sid, ngx_pid);
109 if (pending == NULL) {
110 ngx_log_error(NGX_LOG_INFO, log, 0,
111 "brix: CMS fan-out: sid=%uD client already gone "
112 "(errors %ui/%ui)", slot->sid, slot->got_err,
113 slot->expected);
114 brix_cms_fanout_slot_free(slot);
115 return;
116 }
117
118 conn_fd = pending->conn_fd;
119 conn_number = pending->conn_number;
120 client_streamid[0] = pending->client_streamid[0];
121 client_streamid[1] = pending->client_streamid[1];
122 brix_pending_unlock();
123 brix_pending_remove(slot->sid, ngx_pid);
124
125 client_conn = brix_cms_client_conn_by_fd(conn_fd);
126 if (client_conn == NULL || client_conn->number != conn_number) {
127 brix_cms_fanout_slot_free(slot);
128 return;
129 }
130
131 session = client_conn->data;
132 if (session == NULL) {
133 brix_cms_fanout_slot_free(slot);
134 return;
135 }
136
137 xrd_ctx = ngx_stream_get_module_ctx(session, ngx_stream_brix_module);
138 if (xrd_ctx == NULL || xrd_ctx->state != XRD_ST_WAITING_CMS
139 || xrd_ctx->cms_wait_streamid != slot->sid)
140 {
141 brix_cms_fanout_slot_free(slot);
142 return;
143 }
144
145 if (client_conn->read->timer_set) {
146 ngx_del_timer(client_conn->read);
147 }
148 xrd_ctx->state = XRD_ST_REQ_HEADER;
149 xrd_ctx->recv.cur_streamid[0] = client_streamid[0];
150 xrd_ctx->recv.cur_streamid[1] = client_streamid[1];
151
152 if (slot->worst_err == 0) {
153 (void) brix_send_ok(xrd_ctx, client_conn, NULL, 0);
154 } else {
155 (void) brix_send_error(xrd_ctx, client_conn, kXR_FSError,
156 slot->errtext[0] != '\0'
157 ? slot->errtext
158 : "forwarded operation failed");
159 }
160
161 (void) brix_schedule_read_resume(client_conn);
162 brix_cms_fanout_slot_free(slot);
163}

165ngx_int_t
166brix_cms_fanout_mutation(brix_ctx_t *ctx, ngx_connection_t *c,
167 ngx_stream_brix_srv_conf_t *conf, const char *path)
168{
169 brix_cms_srv_ctx_t *node;
170 brix_cms_srv_ctx_t *elig[BRIX_CMS_FANOUT_NODES];
171 brix_cms_fanout_slot_t *slot;
172 ngx_uint_t i, n_elig, sent;
173 uint32_t sid;
174 u_char code;
175 char ident[64];
176
177 if (!conf->cms.fanout) {
178 return NGX_DECLINED;
179 }
180
181 switch (ctx->recv.cur_reqid) {
182 case kXR_rm:
183 code = CMS_RR_RM;
184 break;
185 case kXR_rmdir:
186 code = CMS_RR_RMDIR;
187 break;
188 default:
189 return NGX_DECLINED; /* v1 scope: single-path deletes only */
190 }
191
192 n_elig = 0;
193 for (i = 0; i < brix_cms_srv_node_count()
194 && n_elig < BRIX_CMS_FANOUT_NODES; i++)
195 {
196 node = brix_cms_srv_node_at(i);
197 if (node == NULL || !node->logged_in || node->c == NULL
198 || !brix_srv_paths_cover(node->paths, path)
199 || brix_srv_is_blacklisted(node->host, node->port))
200 {
201 continue;
202 }
203 elig[n_elig++] = node;
204 }
205
206 /* Engage only for a genuinely replicated path this worker can settle
207 * alone: >=2 eligible local connections covering EVERY registry holder.
208 * Any shortfall (single holder, a holder connected to another worker, a
209 * drained holder) falls back to the shipped single-node redirect. */
210 if (n_elig < 2 || (int) n_elig != brix_srv_count_matching(path)) {
211 return NGX_DECLINED;
212 }
213
214 slot = NULL;
215 for (i = 0; i < BRIX_CMS_FANOUT_SLOTS; i++) {
216 if (!brix_fanout_slots[i].in_use) {
217 slot = &brix_fanout_slots[i];
218 break;
219 }
220 }
221 if (slot == NULL) {
222 return NGX_DECLINED; /* table full — redirect still deletes one */
223 }
224
225 sid = brix_cms_srv_next_streamid();
226 if (brix_pending_insert(sid, ngx_pid, c->fd, c->number,
227 ctx->recv.cur_streamid,
228 conf->cms.fanout_window * 2) != NGX_OK)
229 {
230 return NGX_DECLINED;
231 }
232
233 ngx_snprintf((u_char *) ident, sizeof(ident) - 1, "brix.%d:%d@mgr%Z",
234 (int) ngx_pid, c->fd);
235
236 sent = 0;
237 for (i = 0; i < n_elig; i++) {
238 if (brix_cms_forward_to_node(elig[i]->c, code, sid, ident, path,
239 NULL, NULL, NULL) == NGX_OK)
240 {
241 sent++;
242 } else {
243 ngx_log_error(NGX_LOG_WARN, c->log, 0,
244 "brix: CMS fan-out: forward of \"%s\" to %s:%d "
245 "failed", path, elig[i]->host, (int) elig[i]->port);
246 }
247 }
248
249 if (sent == 0) {
250 brix_pending_remove(sid, ngx_pid);
251 return NGX_DECLINED;
252 }
253
254 ngx_memzero(slot, sizeof(*slot));
255 slot->sid = sid;
256 slot->expected = sent;
257 slot->in_use = 1;
258 slot->deadline.handler = brix_cms_fanout_deadline;
259 slot->deadline.data = slot;
260 slot->deadline.log = ngx_cycle->log;
261 ngx_add_timer(&slot->deadline, conf->cms.fanout_window);
262
263 /* Park the client exactly like a pending locate; the read timer is only
264 * the backstop should the deadline event be lost (its timeout path sends
265 * kXR_wait and drops the pending entry, making finalize a no-op). */
266 ctx->cms_wait_streamid = sid;
267 ctx->state = XRD_ST_WAITING_CMS;
268 ngx_add_timer(c->read, conf->cms.fanout_window * 2);
269
270 ngx_log_error(NGX_LOG_INFO, c->log, 0,
271 "brix: CMS fan-out: op=%ui \"%s\" -> %ui nodes "
272 "(sid=%uD window=%M)",
273 (ngx_uint_t) ctx->recv.cur_reqid, path, sent, sid,
274 conf->cms.fanout_window);
275
276 return NGX_AGAIN;
277}

279void
280brix_cms_fanout_note_error(uint32_t streamid, uint32_t ecode,
281 const char *text, ngx_log_t *log)
282{
283 brix_cms_fanout_slot_t *slot;
284 size_t i;
285
286 slot = brix_cms_fanout_slot_get(streamid);
287 if (slot == NULL) {
288 return; /* late reply — the window already settled this fan-out */
289 }
290
291 slot->got_err++;
292 if (slot->worst_err == 0) {
293 slot->worst_err = (ecode != 0) ? ecode : 1;
294 /* The text is peer-controlled and is relayed to the client verbatim:
295 * keep printable ASCII (spaces included), squash everything else so
296 * a hostile node cannot smuggle control bytes into the reply. */
297 for (i = 0; text != NULL && text[i] != '\0'
298 && i < sizeof(slot->errtext) - 1; i++)
299 {
300 slot->errtext[i] = (text[i] >= 0x20 && text[i] < 0x7f)
301 ? text[i] : '?';
302 }
303 slot->errtext[i] = '\0';
304 }
305
306 ngx_log_error(NGX_LOG_NOTICE, log, 0,
307 "brix: CMS fan-out: node error %ui/%ui for sid=%uD: %s",
308 slot->got_err, slot->expected, streamid,
309 slot->errtext);
310
311 if (slot->got_err >= slot->expected) {
312 brix_cms_fanout_finalize(slot, log); /* every node answered */
313 }
314}
