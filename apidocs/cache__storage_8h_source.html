4/*
5 * cache_storage.h — per-role SD storage instances for the cache.
6 *
7 * WHAT: The cache performs ALL disk byte-I/O through an brix_sd_instance_t per
8 * role — the read cache (cache_root), the sidecar/state tree
9 * (cache_state_root, POSIX), and the write-back staging cache
10 * (cache_wt_stage_root). The POSIX driver is bound to a per-worker O_PATH
11 * rootfd by default; a configured backend (e.g. pblock) is resolved through
12 * the backend registry instead. No raw libc disk calls in the cache data path.
13 *
14 * WHY: Makes every cache storage role independently backend-pluggable and routes
15 * the cache through the same SD seam as the export, so a node can run one
16 * backend for its primary and its cache. See the design spec.
17 *
18 * HOW: brix_cache_storage_init() (per-worker, from process.c) opens the O_PATH
19 * rootfds and builds the instances; brix_cache_storage_cleanup() closes
20 * them at exit. The resolvers return the pre-built instance for a role. The
21 * cache key is the server-controlled export-relative path (no raw client path).
22 */
23
