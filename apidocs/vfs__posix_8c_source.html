1/* client/lib/vfs_posix.c
2 *
3 * WHAT: POSIX storage backend for brix_vfs — the reference implementation.
4 * Handles local file I/O via pread/pwrite on plain fds, with optional
5 * io_uring overlap for read-ahead / write-behind, and atomic temp+rename
6 * commit for safe overwrites.
7 * WHY: copy.c currently hard-codes POSIX fds at the local endpoint. This
8 * backend encapsulates that in the brix_vfs_backend vtable so any caller
9 * using brix_vfs_open() transparently gets atomic commits, io_uring,
10 * and the correct caps without touching copy.c.
11 * HOW: open() for READ opens the final path directly; open() for WRITE creates
12 * a sibling temp (make_posix_temp_path + open O_WRONLY|O_CREAT|O_TRUNC
13 * with O_EXCL|O_NOFOLLOW for symlink safety), honouring XRDC_VFS_FORCE
14 * against an existing final path. An io_uring ring is optionally wrapped
15 * around the fd (AUTO → use if available, ON → require, OFF → plain fd).
16 * pread/pwrite route through the ring when present, else fall back to the
17 * plain POSIX calls. commit() = fsync + rename(tmp, final).
18 * abort() = unlink(tmp). close() frees the ring and fd.
19 * Self-registers via brix_vfs_posix_backend() on first use.
20 * ngx-free; no goto; functional/modular; one responsibility per function.
21 */

43/* Concrete per-handle struct */
44/*
45 * vfs_posix_file — concrete file handle for the POSIX backend.
46 *
47 * WHAT: extends brix_vfs_file with the fd, optional ring, and temp/final
48 * paths needed for atomic temp+rename commit.
49 * HOW: base MUST be first (struct alias cast to brix_vfs_file *).
50 * final_path is the real destination; tmp_path is the sibling temp
51 * (NULL for READ handles, where commit/abort are no-ops).
52 */
53typedef struct {
54 brix_vfs_file base; /* MUST be first — aliased by façade */
55 int fd;
56 brix_disk_ring *ring; /* NULL when io_uring is OFF or unavailable */
57 char *final_path; /* heap-allocated final path (READ or WRITE) */
58 char *tmp_path; /* heap-allocated temp path (WRITE only; NULL for READ) */
59} vfs_posix_file;

61/* Temp-path helper */
62/*
63 * make_posix_temp_path — allocate a sibling temp path for a WRITE destination.
64 *
65 * WHAT: returns a heap string "<dst>.xrdvfs-tmp.<pid>.<seq>", or NULL on OOM/
66 * overflow.
67 * WHY: co-locates the temp on the same filesystem as the final so rename() is
68 * guaranteed atomic, and opens O_EXCL. The pid alone is NOT unique under
69 * `xrdcp -j`, whose batch workers are threads sharing one pid: two same-
70 * basename destinations would collide on an identical temp and fail O_EXCL
71 * (`File exists`). A process-wide atomic sequence makes every concurrent
72 * open's temp distinct; pids already differ across separate processes.
73 * HOW: snprintf into a heap buffer; return NULL on overflow (path too long) or
74 * malloc failure. Caller must free the result.
75 */
76static char *
77make_posix_temp_path(const char *dst)
78{
79 static atomic_ulong seq;
80 unsigned long s = atomic_fetch_add(&seq, 1ul);
81 size_t n = strlen(dst) + 48; /* room for ".xrdvfs-tmp.<pid>.<seq>" */
82 char *buf = malloc(n);
83 if (buf == NULL) {
84 return NULL;
85 }
86 if ((size_t) snprintf(buf, n, "%s.xrdvfs-tmp.%ld.%lu",
87 dst, (long) getpid(), s) >= n) {
88 free(buf);
89 return NULL;
90 }
91 return buf;
92}

94/* Ring selection */
95/*
96 * posix_ring_select — attach an optional io_uring ring to an already-open fd.
97 *
98 * WHAT: implements the AUTO/ON/OFF tri-state. OFF → *ring=NULL, return 0.
99 * ON with unavailable io_uring → *ring=NULL, return -1, st set.
100 * AUTO with unavailable io_uring → *ring=NULL, return 0 (silent fallback).
101 * Otherwise attempts brix_disk_ring_create; ON failure with ON mode → error.
102 * WHY: mirrors copy.c:local_ring_select so the same semantics apply.
103 * HOW: mode from opts->io_uring; guards brix_uring_available(); delegates to
104 * brix_disk_ring_create with a modest window (4 ops, 64 KiB each).
105 */
106static int
107posix_ring_select(int fd, int mode, brix_disk_ring **ring, brix_status *st)
108{
109 *ring = NULL;
110
111 if (mode == XRDC_IO_URING_OFF) {
112 return 0;
113 }
114
115 if (!brix_uring_available()) {
116 if (mode == XRDC_IO_URING_ON) {
117 brix_status_set(st, XRDC_EUNSUPPORTED, 0,
118 "vfs posix: io_uring=on but io_uring is unavailable "
119 "on this host or this build lacks liburing");
120 return -1;
121 }
122 return 0; /* AUTO: classic path */
123 }
124
125 {
126 brix_status tmp_st;
127 brix_status_clear(&tmp_st);
128 *ring = brix_disk_ring_create(fd, 4, 65536, 0, &tmp_st);
129 if (*ring == NULL && mode == XRDC_IO_URING_ON) {
130 if (st != NULL) {
131 *st = tmp_st;
132 }
133 return -1;
134 }
135 /* AUTO: create failure leaves *ring NULL → classic pread/pwrite */
136 }
137 return 0;
138}

141/*
142 * posix_pread — read n bytes at offset off into buf.
143 *
144 * WHAT: delegates to the ring (read-ahead path) when present; falls back to
145 * plain pread(2) otherwise. Returns bytes read (≥0) or -1 with st set.
146 * WHY: single dispatch point — callers never see the ring vs. plain difference.
147 * HOW: ring path: brix_disk_ring_pread; plain path: pread(2) with EINTR retry.
148 */
149static ssize_t
150posix_pread(brix_vfs_file *f, int64_t off, void *buf, size_t n, brix_status *st)
151{
152 vfs_posix_file *pf = (vfs_posix_file *) f;
153
154 if (pf->ring != NULL) {
155 return brix_disk_ring_pread(pf->ring, off, (uint8_t *) buf, n, st);
156 }
157
158 {
159 brix_sd_obj_t obj;
160 ssize_t r;
161
162 brix_sd_posix_wrap(&obj, pf->fd);
163 r = xvfs_pread_once(&obj, buf, n, (off_t) off);
164 if (r < 0) {
165 brix_status_set(st, XRDC_EIO, errno,
166 "vfs posix pread: %s", strerror(errno));
167 }
168 return r;
169 }
170}

172/*
173 * posix_pwrite — write n bytes from buf at offset off.
174 *
175 * WHAT: delegates to the ring (write-behind path) when present; falls back to
176 * plain pwrite(2) otherwise. Returns 0 on success, -1 with st set.
177 * WHY: single dispatch point; ring path avoids blocking disk writes.
178 * HOW: ring path: split into ring-buffer-sized pieces and call
179 * brix_disk_ring_pwrite for each (brix_disk_ring_pwrite requires n <=
180 * bufsz; the caller may supply a chunk much larger than the ring's
181 * per-op buffer). Plain path: pwrite(2) retry loop.
182 */
183static int
184posix_pwrite(brix_vfs_file *f, int64_t off, const void *buf, size_t n,
185 brix_status *st)
186{
187 vfs_posix_file *pf = (vfs_posix_file *) f;
188
189 if (pf->ring != NULL) {
190 const uint8_t *p = (const uint8_t *) buf;
191 size_t rem = n;
192 size_t rsz = brix_disk_ring_bufsz(pf->ring);
193
194 while (rem > 0) {
195 size_t chunk = rem < rsz ? rem : rsz;
196 if (brix_disk_ring_pwrite(pf->ring, off, p, chunk, st) != 0) {
197 return -1;
198 }
199 p += chunk;
200 off += (int64_t) chunk;
201 rem -= chunk;
202 }
203 return 0;
204 }
205
206 {
207 brix_sd_obj_t obj;
208
209 brix_sd_posix_wrap(&obj, pf->fd);
210 if (xvfs_pwrite_full(&obj, buf, n, (off_t) off, NULL, NULL) != 0) {
211 brix_status_set(st, XRDC_EIO, errno,
212 "vfs posix pwrite: %s", strerror(errno));
213 return -1;
214 }
215 return 0;
216 }
217}

272 * WHAT: drains any ring write-behind queue, then calls fsync(2) on pf->fd.
273 * WHY: ensures callers can rely on data being durable after sync() returns.

299/*
300 * posix_commit — finalise a WRITE handle: fsync the temp then rename to final.
301 *
302 * WHAT: drains any ring queue, fsyncs the fd, then atomically renames the sibling
303 * temp to the final path. For READ handles (no tmp_path) this is a no-op.
304 * WHY: atomic rename guarantees readers never see a partial file at the final path.
305 * HOW: ring flush → fsync → rename; on rename failure unlink the temp and return -1.
306 */
307static int
308posix_commit(brix_vfs_file *f, brix_status *st)
309{
310 vfs_posix_file *pf = (vfs_posix_file *) f;
311
312 if (pf->tmp_path == NULL) {
313 return 0; /* READ handle — nothing to commit */
314 }
315
316 if (pf->ring != NULL) {
317 if (brix_disk_ring_flush(pf->ring, st) != 0) {
318 return -1;
319 }
320 }
321
322 if (fsync(pf->fd) != 0) {
323 brix_status_set(st, XRDC_EIO, errno,
324 "vfs posix commit fsync: %s", strerror(errno));
325 return -1;
326 }
327
328 if (rename(pf->tmp_path, pf->final_path) != 0) {
329 brix_status_set(st, XRDC_EIO, errno,
330 "vfs posix commit rename %s -> %s: %s",
331 pf->tmp_path, pf->final_path, strerror(errno));
332 unlink(pf->tmp_path);
333 return -1;
334 }
335 return 0;
336}

339 * posix_abort — discard a partial WRITE by unlinking the sibling temp.
340 *
341 * WHAT: removes the temp file; no-op for READ handles.
342 * WHY: callers abort() on transfer failure so the temp is never left behind.
343 * HOW: unlink(tmp_path) if present; ignore errors (best-effort cleanup).

345static void
346posix_abort(brix_vfs_file *f)

348 vfs_posix_file *pf = (vfs_posix_file *) f;
349
350 if (pf->tmp_path != NULL) {
351 unlink(pf->tmp_path);
352 }
353}

355/*
356 * posix_close — release the handle: drain ring, close fd, free all heap memory.
357 *
358 * WHAT: frees the ring (if any), closes the fd, frees path strings, and frees
359 * the handle struct. Must be called after commit() or abort().
360 * WHY: owns all resources allocated by posix_open(); one release point.
361 * HOW: brix_disk_ring_destroy (safe on NULL); close(fd); free strings; free pf.
362 */
363static void
364posix_close(brix_vfs_file *f)
365{
366 vfs_posix_file *pf = (vfs_posix_file *) f;
367
368 brix_disk_ring_destroy(pf->ring);
369
370 if (pf->fd >= 0) {
371 close(pf->fd);
372 }
373
374 free(pf->final_path);
375 free(pf->tmp_path);
376 free(pf);
377}

391/* Backend open + stat */
392/*
393 * posix_be_open — allocate a vfs_posix_file and open the underlying fd.
394 *
395 * WHAT: READ → open(path, O_RDONLY); WRITE → FORCE-check then open a sibling
396 * temp with O_WRONLY|O_CREAT|O_TRUNC (O_EXCL|O_NOFOLLOW for safety);
397 * both paths optionally wrap the fd in an io_uring ring.
398 * WHY: single open implementation; FORCE/EXCL/NOFOLLOW guard matches copy.c.
399 * HOW: allocate pf; check FORCE vs existing final; open fd; ring_select; set caps.
400 * Caller receives NULL *out on any failure; the handle is not half-initialised.
401 */
402static int
403posix_be_open(const brix_vfs_backend *be, const char *path, int flags,
404 const brix_vfs_open_opts *opts, brix_vfs_file **out,
405 brix_status *st)
406{
407 vfs_posix_file *pf;
408 int fd = -1;
409 char *final_copy = NULL;
410 char *tmp_path = NULL;
411 int uring_mode;
412
413 (void) be;
414
415 *out = NULL;
416
417 uring_mode = (opts != NULL) ? opts->io_uring : XRDC_IO_URING_AUTO;
418
419 final_copy = strdup(path);
420 if (final_copy == NULL) {
421 brix_status_set(st, XRDC_EIO, ENOMEM,
422 "vfs posix open: out of memory");
423 return -1;
424 }
425
426 if (flags & XRDC_VFS_WRITE) {
427 /* FORCE guard: if final exists and FORCE is not set, reject. */
428 if (!(flags & XRDC_VFS_FORCE) && access(path, F_OK) == 0) {
429 brix_status_set(st, XRDC_EUSAGE, EEXIST,
430 "vfs posix open: destination exists (use "
431 "XRDC_VFS_FORCE to overwrite): %s", path);
432 free(final_copy);
433 return -1;
434 }
435
436 tmp_path = make_posix_temp_path(path);
437 if (tmp_path == NULL) {
438 brix_status_set(st, XRDC_EUSAGE, 0,
439 "vfs posix open: destination path too long: %s",
440 path);
441 free(final_copy);
442 return -1;
443 }
444
445 /* Open through the shared POSIX driver (unconfined): O_EXCL refuses a
446 * pre-existing name; O_NOFOLLOW refuses a symlink. Same driver the nginx
447 * server opens through (its variant adds RESOLVE_BENEATH confinement). */
448 fd = brix_sd_posix_open_unconfined(tmp_path,
449 BRIX_SD_O_WRITE | BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC
450 | BRIX_SD_O_EXCL | BRIX_SD_O_NOFOLLOW, 0644);
451 if (fd < 0) {
452 brix_status_set(st, XRDC_EIO, errno,
453 "vfs posix open write temp %s: %s",
454 tmp_path, strerror(errno));
455 free(final_copy);
456 free(tmp_path);
457 return -1;
458 }
459 } else {
460 /* READ — through the shared POSIX driver (unconfined). */
461 fd = brix_sd_posix_open_unconfined(path, BRIX_SD_O_READ, 0);
462 if (fd < 0) {
463 brix_status_set(st, XRDC_EIO, errno,
464 "vfs posix open read %s: %s",
465 path, strerror(errno));
466 free(final_copy);
467 return -1;
468 }
469 }
470
471 pf = calloc(1, sizeof(*pf));
472 if (pf == NULL) {
473 brix_status_set(st, XRDC_EIO, ENOMEM,
474 "vfs posix open: out of memory");
475 close(fd);
476 free(final_copy);
477 free(tmp_path);
478 return -1;
479 }
480
481 pf->fd = fd;
482 pf->final_path = final_copy;
483 pf->tmp_path = tmp_path; /* NULL for READ */
484 pf->ring = NULL;
485
486 if (posix_ring_select(fd, uring_mode, &pf->ring, st) != 0) {
487 close(fd);
488 free(final_copy);
489 free(tmp_path);
490 free(pf);
491 return -1;
492 }
493
494 pf->base.ops = &s_posix_ops;
495 pf->base.caps = (XRDC_VFS_CAP_RANDOM_WRITE | XRDC_VFS_CAP_TRUNCATE |
496 XRDC_VFS_CAP_ATOMIC_TEMP | XRDC_VFS_CAP_FADVISE);
497
498 *out = &pf->base;
499 return 0;
500}
