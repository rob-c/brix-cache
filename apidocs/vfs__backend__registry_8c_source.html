25/* Find a registered entry by exact root_canon, or NULL. */
26brix_vfs_backend_entry_t *
27brix_vfs_backend_entry_find(const char *root_canon)
28{
29 ngx_uint_t i;
30
31 if (root_canon == NULL || root_canon[0] == '\0') {
32 return NULL;
33 }
34 for (i = 0; i < brix_vfs_backend_count; i++) {
35 if (ngx_strcmp(brix_vfs_backends[i].root_canon, root_canon) == 0) {
36 return &brix_vfs_backends[i];
37 }
38 }
39 return NULL;
40}

45brix_vfs_backend_http_endpoint(const char *root_canon,
46 const char **host, int *port, int *tls, const char **base)
47{

60/* Find or create the entry for root_canon. A cache/stage tier may be registered
61 * for a default-POSIX export that has no backend entry yet, so create one (backend
62 * "" = default POSIX source). NULL when the table is full. */
63brix_vfs_backend_entry_t *
64brix_vfs_backend_entry_get_or_create(const char *root_canon)
65{
66 brix_vfs_backend_entry_t *e = brix_vfs_backend_entry_find(root_canon);
67
68 if (e != NULL) {
69 return e;
70 }
71 if (root_canon == NULL || root_canon[0] == '\0'
72 || brix_vfs_backend_count >= BRIX_VFS_BACKEND_MAX)
73 {
74 return NULL;
75 }
76 e = &brix_vfs_backends[brix_vfs_backend_count++];
77 ngx_memzero(e, sizeof(*e));
78 ngx_cpystrn((u_char *) e->root_canon, (u_char *) root_canon,
79 sizeof(e->root_canon));
80 return e;
81}

83/*
84 * Decorator-composition steps for brix_vfs_backend_entry_build.
85 *
86 * WHAT: each helper takes the current top-of-stack instance + the entry + log,
87 * and returns the (possibly) wrapped instance — the legacy staging shim,
88 * the explicit stage tier, and the read-cache tier respectively.
89 * WHY: the composition orchestrator was an 18-CCN function whose branches were
90 * three near-identical decorator blocks; extracting one helper per tier
91 * lets the orchestrator read as a flat wrap sequence (§8 compose small
92 * functions).
93 * HOW: a helper that cannot build/init its decorator returns the input `top`
94 * unchanged (degraded decorator skipped so the export still serves from the
95 * tier below), preserving the pre-split behavior and every log string.
96 */
97
98/* Legacy local-staging shim (brix_storage_staging): buffer on a posix store
99 * rooted at the export root and flush to the source on commit. Superseded by an
100 * explicit stage tier, so skip it when one is configured. */
101static brix_sd_instance_t *
102brix_vbr_wrap_staging_shim(brix_sd_instance_t *top,
103 brix_vfs_backend_entry_t *e, ngx_log_t *log)
104{
105 int sderr = 0;
106 brix_sd_instance_t *store;
107 brix_sd_instance_t *dec;
108
109 if (!e->staging || e->stage_tier.configured
110 || top->driver->staged_open == NULL)
111 {
112 return top;
113 }
114
115 store = brix_sd_instance_create(log, "posix",
116 (void *) e->root_canon, &sderr);
117 dec = (store != NULL)
118 ? brix_sd_stage_create(top, store, NULL, e->root_canon, log) : NULL;
119
120 if (dec != NULL) {
121 ngx_log_error(NGX_LOG_NOTICE, log, 0,
122 "brix: write-back stage decorator composed over \"%s\"",
123 e->root_canon);
124 return dec;
125 }
126 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
127 "brix: stage shim init failed for \"%s\" - source direct",
128 e->root_canon);
129 return top;
130}

132/* Phase-64 explicit stage tier (sd_stage over an explicit stage store). */
133static brix_sd_instance_t *
134brix_vbr_wrap_stage_tier(brix_sd_instance_t *top,
135 brix_vfs_backend_entry_t *e, ngx_log_t *log)
136{
137 brix_sd_instance_t *store;
138 brix_sd_instance_t *dec;
139

141 return top;

143

158/* Phase-64 read-cache tier (sd_cache, outermost). */
159static brix_sd_instance_t *
160brix_vbr_wrap_cache_tier(brix_sd_instance_t *top,
161 brix_vfs_backend_entry_t *e, ngx_log_t *log)
162{
163 brix_sd_instance_t *store;
164 const char *local_root;
165 brix_sd_instance_t *dec;
166
167 if (!e->cache_tier.configured) {
168 return top;
169 }
170
171 store = brix_tier_build(&e->cache_tier, log);
172 local_root = (ngx_strcmp(e->cache_tier.driver, "posix") == 0
173 && e->cache_tier.path[0] != '\0') ? e->cache_tier.path : NULL;
174 dec = (store != NULL)
175 ? brix_sd_cache_create(top, store, &e->cache_policy, local_root, log)
176 : NULL;
177
178 if (dec != NULL) {
179 ngx_log_error(NGX_LOG_NOTICE, log, 0,
180 "brix: read-cache tier (%s) composed over \"%s\"",
181 e->cache_tier.driver, e->root_canon);
182 return dec;
183 }
184 return top;
185}

187/* Lazily build + memoize the entry's COMPOSED storage stack (per worker). The
188 * registry is the single composition point: build the source, then wrap it
189 * bottom-up in the stage decorator and the read-cache decorator (cache outermost,
190 * phase-64 section 5). A degraded decorator (a store that is "needs development"
191 * or fails to init) is skipped so the export still serves from the tier below. */
192static brix_sd_instance_t *
193brix_vfs_backend_entry_build(brix_vfs_backend_entry_t *e, ngx_log_t *log)
194{
195 brix_sd_instance_t *top;
196
197 if (e->inst != NULL) {
198 return e->inst; /* already built in this worker */
199 }
200 /* The built stack is memoized for the worker's whole life, and drivers
201 * keep the log they are built with for later diagnostics (sd_http logs
202 * selection/failover from fill threads). A request-time resolver hands
203 * us its connection log, which dies with the connection — build with
204 * the cycle log instead so stored pointers never go stale. */
205 if (ngx_cycle != NULL && ngx_cycle->log != NULL) {
206 log = ngx_cycle->log;
207 }
208 top = brix_vfs_backend_build_source(e, log);
209 if (top == NULL) {
210 return NULL;
211 }
212
213 top = brix_vbr_wrap_staging_shim(top, e, log);
214 top = brix_vbr_wrap_stage_tier(top, e, log);
215 top = brix_vbr_wrap_cache_tier(top, e, log);
216
217 e->inst = top;
218 return e->inst;
219}

247brix_sd_instance_t *
248brix_vfs_backend_resolve(const char *root_canon, ngx_log_t *log)
249{
250 ngx_uint_t i;
251
252 if (root_canon == NULL || root_canon[0] == '\0'
253 || brix_vfs_backend_count == 0)
254 {
255 return NULL;
256 }
257
258 for (i = 0; i < brix_vfs_backend_count; i++) {
259 brix_vfs_backend_entry_t *e = &brix_vfs_backends[i];
260
261 if (ngx_strcmp(e->root_canon, root_canon) != 0) {
262 continue;
263 }
264 return brix_vfs_backend_entry_build(e, log);
265 }
266 return NULL;

269brix_sd_instance_t *
270brix_vfs_backend_resolve_for_path(const char *abs_path, const char **root_out,
271 ngx_log_t *log)
272{
273 brix_vfs_backend_entry_t *best = NULL;
274 size_t best_len = 0;
275 ngx_uint_t i;
276
277 if (abs_path == NULL || abs_path[0] == '\0'
278 || brix_vfs_backend_count == 0)
279 {
280 return NULL;
281 }
282
283 /* Longest registered export root that is a prefix of abs_path: a match is the
284 * root itself or root + "/..." (so "/exp" never matches "/export/x"). */
285 for (i = 0; i < brix_vfs_backend_count; i++) {
286 brix_vfs_backend_entry_t *e = &brix_vfs_backends[i];
287 size_t rl = ngx_strlen(e->root_canon);
288
289 if (ngx_strncmp(abs_path, e->root_canon, rl) != 0) {
290 continue;
291 }
292 if (abs_path[rl] != '/' && abs_path[rl] != '\0') {
293 continue; /* shares a prefix but a different name */
294 }
295 if (rl > best_len) {
296 best_len = rl;
297 best = e;
298 }
299 }
300 if (best == NULL) {
301 return NULL;
302 }
303 if (root_out != NULL) {
304 *root_out = best->root_canon;
305 }
306 return brix_vfs_backend_entry_build(best, log);
