30/* Endpoint health TRANSITIONS (sd_http EWMA hysteresis) as single-line
31 * operator events: which Stratum-1/origin flapped, and when it came back.
32 * Fires from fill worker threads — the cycle log is the stable target. */
33static void
34brix_vfs_http_health_note(const char *host, int port, int healthy)
35{
36 if (healthy) {
37 ngx_log_error(NGX_LOG_NOTICE, ngx_cycle->log, 0,
38 "xrootd-origin: event=recovered host=%s port=%d — endpoint "
39 "answering again; rank-preferred routing resumes", host, port);
40 } else {
41 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
42 "xrootd-origin: event=degraded host=%s port=%d — transport "

46 }
47}

52/* Find a registered entry by exact root_canon, or NULL. */
53brix_vfs_backend_entry_t *
54brix_vfs_backend_entry_find(const char *root_canon)
55{
56 ngx_uint_t i;
57
58 if (root_canon == NULL || root_canon[0] == '\0') {
59 return NULL;
60 }
61 for (i = 0; i < brix_vfs_backend_count; i++) {
62 if (ngx_strcmp(brix_vfs_backends[i].root_canon, root_canon) == 0) {
63 return &brix_vfs_backends[i];
64 }
65 }
66 return NULL;
67}

87/* Find or create the entry for root_canon. A cache/stage tier may be registered
88 * for a default-POSIX export that has no backend entry yet, so create one (backend
89 * "" = default POSIX source). NULL when the table is full. */
90brix_vfs_backend_entry_t *
91brix_vfs_backend_entry_get_or_create(const char *root_canon)
92{
93 brix_vfs_backend_entry_t *e = brix_vfs_backend_entry_find(root_canon);
94
95 if (e != NULL) {
96 return e;
97 }
98 if (root_canon == NULL || root_canon[0] == '\0'
99 || brix_vfs_backend_count >= BRIX_VFS_BACKEND_MAX)
100 {
101 return NULL;
102 }
103 e = &brix_vfs_backends[brix_vfs_backend_count++];
104 ngx_memzero(e, sizeof(*e));
105 ngx_cpystrn((u_char *) e->root_canon, (u_char *) root_canon,
106 sizeof(e->root_canon));
107 return e;
108}

110/* Build the entry's RAW source driver instance (no decorators, no memoization). The
111 * instance is malloc/pool-owned, worker-safe; SQLite connections are per-worker so
112 * this is only ever called after fork. Returns NULL (logged) on init failure. */
113static brix_sd_instance_t *
114brix_vfs_backend_build_source(brix_vfs_backend_entry_t *e, ngx_log_t *log)
115{
116 brix_sd_instance_t *inst;
117
118 /* Default / explicit POSIX source. Phase-64: an export whose backend is the
119 * default POSIX tree but which carries a cache/stage tier still needs a
120 * buildable source instance to decorate. */
121 if (e->backend[0] == '\0' || ngx_strcmp(e->backend, "posix") == 0) {
122 int err = 0;
123
124 inst = brix_sd_instance_create(ngx_cycle->pool, log, "posix",
125 (void *) e->root_canon, &err);
126 if (inst == NULL) {
127 ngx_log_error(NGX_LOG_ERR, log, err,
128 "brix: posix source init failed for export \"%s\"",
129 e->root_canon);
130 }
131 return inst;
132 }
133
134 /* Remote root:// backend: the in-process origin wire client (read + write +
135 * staged_open, auth via ztn/gsi). */
136 if (ngx_strcmp(e->backend, "xroot") == 0) {
137 inst = brix_sd_xroot_create_origin(e->origin_host, e->origin_port,
138 e->origin_tls, e->origin_family,
139 (e->origin_token[0] != '\0') ? e->origin_token : NULL,
140 (e->origin_x509_proxy[0] != '\0') ? e->origin_x509_proxy : NULL,
141 (e->origin_ca_dir[0] != '\0') ? e->origin_ca_dir : NULL,

143 log);

145 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
146 "brix: remote root:// backend init failed for export \"%s\"",

149 ngx_log_error(NGX_LOG_NOTICE, log, 0,
150 "brix: remote root:// storage backend ready at \"%s\"",
151 e->root_canon);
152 }
153 return inst;

156#if BRIX_HAVE_CEPH

158 * no local directory namespace (the pure-librados reference). */

343/* Lazily build + memoize the entry's COMPOSED storage stack (per worker). The
344 * registry is the single composition point: build the source, then wrap it
345 * bottom-up in the stage decorator and the read-cache decorator (cache outermost,
346 * phase-64 section 5). A degraded decorator (a store that is "needs development"
347 * or fails to init) is skipped so the export still serves from the tier below. */
348static brix_sd_instance_t *
349brix_vfs_backend_entry_build(brix_vfs_backend_entry_t *e, ngx_log_t *log)
350{
351 brix_sd_instance_t *top;
352
353 if (e->inst != NULL) {
354 return e->inst; /* already built in this worker */
355 }
356 /* The built stack is memoized for the worker's whole life, and drivers
357 * keep the log they are built with for later diagnostics (sd_http logs
358 * selection/failover from fill threads). A request-time resolver hands
359 * us its connection log, which dies with the connection — build with
360 * the cycle log instead so stored pointers never go stale. */
361 if (ngx_cycle != NULL && ngx_cycle->log != NULL) {
362 log = ngx_cycle->log;
363 }
364 top = brix_vfs_backend_build_source(e, log);
365 if (top == NULL) {
366 return NULL;
367 }
368
369 /* Legacy local-staging shim (brix_storage_staging): buffer on a posix store
370 * rooted at the export root and flush to the source on commit. Superseded by an
371 * explicit stage tier, so skip it when one is configured. */
372 if (e->staging && !e->stage_tier.configured
373 && top->driver->staged_open != NULL)
374 {
375 int sderr = 0;
376 brix_sd_instance_t *store =
377 brix_sd_instance_create(ngx_cycle->pool, log, "posix",
378 (void *) e->root_canon, &sderr);
379 brix_sd_instance_t *dec = (store != NULL)
380 ? brix_sd_stage_create(top, store, NULL, e->root_canon, log) : NULL;
381
382 if (dec != NULL) {
383 ngx_log_error(NGX_LOG_NOTICE, log, 0,
384 "brix: write-back stage decorator composed over \"%s\"",
385 e->root_canon);
386 top = dec;
387 } else {
388 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
389 "brix: stage shim init failed for \"%s\" - source direct",
390 e->root_canon);
391 }
392 }
393
394 /* Phase-64 explicit stage tier (sd_stage over an explicit stage store). */
395 if (e->stage_tier.configured) {
396 brix_sd_instance_t *store = brix_tier_build(&e->stage_tier, log);
397 brix_sd_instance_t *dec = (store != NULL)
398 ? brix_sd_stage_create(top, store, &e->stage_policy, e->root_canon,
399 log) : NULL;
400
401 if (dec != NULL) {
402 ngx_log_error(NGX_LOG_NOTICE, log, 0,
403 "brix: write-stage tier (%s) composed over \"%s\"",
404 e->stage_tier.driver, e->root_canon);
405 top = dec;
406 }
407 }
408
409 /* Phase-64 read-cache tier (sd_cache, outermost). */
410 if (e->cache_tier.configured) {
411 brix_sd_instance_t *store = brix_tier_build(&e->cache_tier, log);
412 const char *local_root =
413 (ngx_strcmp(e->cache_tier.driver, "posix") == 0
414 && e->cache_tier.path[0] != '\0') ? e->cache_tier.path : NULL;
415 brix_sd_instance_t *dec = (store != NULL)
416 ? brix_sd_cache_create(top, store, &e->cache_policy, local_root, log)
417 : NULL;
418
419 if (dec != NULL) {
420 ngx_log_error(NGX_LOG_NOTICE, log, 0,
421 "brix: read-cache tier (%s) composed over \"%s\"",
422 e->cache_tier.driver, e->root_canon);
423 top = dec;
424 }
425 }
426
427 e->inst = top;
428 return e->inst;
429}

458brix_vfs_backend_resolve(const char *root_canon, ngx_log_t *log)
459{

461
462 if (root_canon == NULL || root_canon[0] == '\0'
463 || brix_vfs_backend_count == 0)
464 {

467
468 for (i = 0; i < brix_vfs_backend_count; i++) {

479brix_sd_instance_t *
480brix_vfs_backend_resolve_for_path(const char *abs_path, const char **root_out,
481 ngx_log_t *log)
482{
483 brix_vfs_backend_entry_t *best = NULL;
484 size_t best_len = 0;
485 ngx_uint_t i;
486
487 if (abs_path == NULL || abs_path[0] == '\0'
488 || brix_vfs_backend_count == 0)
489 {
490 return NULL;
491 }
492
493 /* Longest registered export root that is a prefix of abs_path: a match is the
494 * root itself or root + "/..." (so "/exp" never matches "/export/x"). */
495 for (i = 0; i < brix_vfs_backend_count; i++) {
496 brix_vfs_backend_entry_t *e = &brix_vfs_backends[i];
497 size_t rl = ngx_strlen(e->root_canon);
498
499 if (ngx_strncmp(abs_path, e->root_canon, rl) != 0) {
500 continue;
501 }
502 if (abs_path[rl] != '/' && abs_path[rl] != '\0') {
503 continue; /* shares a prefix but a different name */
504 }
505 if (rl > best_len) {
506 best_len = rl;
507 best = e;
508 }
509 }
510 if (best == NULL) {
511 return NULL;
512 }
513 if (root_out != NULL) {
514 *root_out = best->root_canon;
515 }
516 return brix_vfs_backend_entry_build(best, log);
517}
