3/*
4 * brix_dispatch_session_opcode — phase 2 of request routing (from
5 * handshake/dispatch.c): a switch on ctx->recv.cur_reqid over the seven session
6 * lifecycle opcodes — kXR_protocol, kXR_login, kXR_auth, kXR_ping, kXR_set
7 * (login-gated), kXR_endsess, kXR_bind — each delegating to its handler (defined
8 * in its own module, documented there). These establish the session (protocol
9 * version, identity, auth, sessid) before any file I/O. kXR_bind is special: it
10 * arrives on a secondary connection before login and must be handled first so the
11 * channel knows its parent session. Returns the handler result, or
12 * BRIX_DISPATCH_CONTINUE when the opcode is not a session opcode (the read/write
13 * dispatchers then try it).
14 */
15ngx_int_t
16brix_dispatch_session_opcode(brix_ctx_t *ctx, ngx_connection_t *c,
17 ngx_stream_brix_srv_conf_t *conf)
18{
19 switch (ctx->recv.cur_reqid) {
20
21 case kXR_protocol:
22 return brix_handle_protocol(ctx, c, conf);
23
24 case kXR_login:
25 return brix_handle_login(ctx, c, conf);
26
27 case kXR_auth:
28 return brix_handle_auth(ctx, c);
29
30 case kXR_ping: {
31 /* The reference auth gate rejects EVERY non-auth request before login,
32 * kXR_ping included (a pre-login ping is not a liveness probe a stock
33 * server answers). Match that — ping requires a completed login. */
34 ngx_int_t rc = brix_dispatch_require_login(ctx, c);
35 if (rc != BRIX_DISPATCH_CONTINUE) { return rc; }
36 return brix_handle_ping(ctx, c);
37 }
38
39 case kXR_set: {
40 ngx_int_t rc = brix_dispatch_require_login(ctx, c);
41 if (rc != BRIX_DISPATCH_CONTINUE) { return rc; }
42 return brix_handle_set(ctx, c);
43 }
44
45 case kXR_endsess:
46 return brix_handle_endsess(ctx, c);
47
48 case kXR_bind:
49 /* kXR_bind arrives on secondary connections before kXR_login.
50 * It must be dispatched before the login/auth guard below. */
51 return brix_handle_bind(ctx, c, conf);
52
53 default:
54 return BRIX_DISPATCH_CONTINUE;
55 }
56}
