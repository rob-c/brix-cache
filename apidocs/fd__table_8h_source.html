6/*
7 * XRootD file-handle table for a single TCP session.
8 *
9 * The XRootD wire protocol uses a 4-byte opaque "fhandle" to refer to an
10 * open file across multiple requests (kXR_read, kXR_write, kXR_close, etc.).
11 * Internally we map the client-chosen fhandle byte to a slot index in
12 * ctx->files[BRIX_MAX_FILES].
13 *
14 * IMPORTANT — fhandle scope: primary fhandles are owned by the TCP session
15 * that issued kXR_open. kXR_bind secondaries may read those handles only
16 * while the primary continues publishing the matching slot in shared memory;
17 * they may not open, close, or write their own file handles.
18 *
19 * Slot lifecycle:
20 * alloc → brix_alloc_fhandle (returns slot index, sets fd = -1)
21 * open → caller sets ctx->files[idx].fd and other fields
22 * use → brix_validate_{file,read,write}_handle before each I/O
23 * close → brix_free_fhandle (closes fd, frees path, clears slot)
24 * cleanup→ brix_close_all_files (called on disconnect)
25 */
26
27/*
28 * brix_alloc_fhandle — find the first free slot in ctx->files[].
29 *
30 * Returns: slot index [0, BRIX_MAX_FILES) on success, -1 if table full.
31 */
32int brix_alloc_fhandle(brix_ctx_t *ctx);

62/*
