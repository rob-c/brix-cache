1/*
2 * close.c — kXR_close (3003) opcode handler: finalise and release an open file handle.
3 *
4 * WHAT: Implements brix_handle_close(), the protocol handler for kXR_close.
5 * It validates the client's file handle, emits an access-log line with
6 * transfer throughput, performs POSC commit, optional write-through
7 * mirroring and dashboard slot release, flushes the write-recovery
8 * journal, frees the handle, and replies with an empty kXR_ok.
9 *
10 * WHY: Close is where a transfer is made durable and visible. POSC
11 * (Persist-On-Successful-Close) files are written to a staging temp path
12 * and only promoted to their final name on a clean close, so close must
13 * fsync and rename atomically before the per-handle state is torn down;
14 * a rename failure must surface as kXR_IOError rather than silently
15 * leaving a partial file. It is also the only point where final byte
16 * counters, timing, and the intended (non-temp) path are still available
17 * for accurate logging and metrics.
18 *
19 * HOW: Resolve the handle index from req->fhandle[0] and validate it via
20 * brix_validate_file_handle(). Compute throughput ("%.2fMB/s") from
21 * bytes_written/bytes_read and (ngx_current_msec - open_time) and log
22 * against posc_final_path when set (never the staging temp path). If
23 * posc_final_path is set, fsync(fd) then rename(temp -> final); on
24 * failure free the handle and return kXR_IOError, on success retarget
25 * write-through to the final path. Run brix_wt_flush_on_close() when
26 * write-through is enabled, clear posc_final_path so brix_free_fhandle()
27 * does not unlink the now-renamed file, release any dashboard transfer
28 * slot, flush the write-recovery journal (brix_wrts_flush) so post-
29 * reconnect writes at the same offsets are not skipped as replays, free
30 * the handle, and send an empty kXR_ok.
31 */

50/*
51 * brix_cns_emit_close — §6: report a just-written file to the manager.
52 *
53 * Best-effort, fire-and-forget: only in EMIT mode, only for a writable handle,
54 * only when the worker's manager link is connected + logged in. The logical path
55 * (export-root-relative, what a client would stat) is derived by stripping
56 * root_canon from the handle's final path; size/mtime come from fstat of the fd.
57 */
58static void
59brix_cns_emit_close(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
60 int idx)
61{
62 const char *fpath, *logical, *root;
63 size_t rlen;
64 struct stat st;
65 uint8_t buf[BRIX_CNS_HDR_LEN + BRIX_CNS_PATH_MAX];
66 size_t n;
67
68 if (conf->cns_mode != BRIX_CNS_EMIT || !ctx->files[idx].writable) {
69 return;
70 }
71 if (conf->cms.ctx == NULL || conf->cms.ctx->connection == NULL
72 || !conf->cms.ctx->logged_in)
73 {
74 return;
75 }
76
77 fpath = (ctx->files[idx].posc_final_path != NULL)
78 ? ctx->files[idx].posc_final_path : ctx->files[idx].path;
79 if (fpath == NULL) {
80 return;
81 }
82 if (ctx->files[idx].fd < 0 || fstat(ctx->files[idx].fd, &st) != 0
83 || !S_ISREG(st.st_mode))
84 {
85 return;
86 }
87
88 root = conf->common.root_canon;
89 rlen = ngx_strlen(root);
90 logical = fpath;
91 if (rlen > 0 && ngx_strncmp(fpath, root, rlen) == 0 && fpath[rlen] == '/') {
92 logical = fpath + rlen; /* keep the leading '/' → client-facing path */
93 }
94
95 n = brix_cns_event_encode(BRIX_CNS_ADD, logical, (uint64_t) st.st_size,
96 (uint64_t) st.st_mtime, buf, sizeof(buf));
97 if (n == 0) {
98 return;
99 }
100 (void) brix_cms_send_frame(conf->cms.ctx->connection, 0, CMS_RR_CNS,
101 CMS_MOD_RAW, buf, n);
102}

104/*
105 * WHAT: Finish the sesslog transfer attached to a root file handle.
106 * WHY: kXR_close is the first point where final byte counters and the durable
107 * commit outcome are known while the handle state is still alive.
108 * HOW: Reconcile the intrusive xfer byte counter from the existing per-handle
109 * read/write totals, then emit COMPLETE or ABORTED before brix_free_fhandle().
110 */
111static void
112brix_close_finish_sess_xfer(brix_ctx_t *ctx, int idx,
113 brix_sess_xfer_status_t status)
114{
115 brix_file_t *file;
116 size_t total;
117
118 file = &ctx->files[idx];
119 total = (file->bytes_written > 0) ? file->bytes_written : file->bytes_read;
120 if (file->sess_xfer.active && total > file->sess_xfer.bytes) {
121 brix_sess_xfer_add(&file->sess_xfer,
122 (uint64_t) (total - file->sess_xfer.bytes));
123 }
124 brix_sess_xfer_end(ctx->sess, &file->sess_xfer, status);
125}

127ngx_int_t brix_handle_close(brix_ctx_t *ctx, ngx_connection_t *c) {
128 xrdw_close_req_t req;
129 ngx_stream_brix_srv_conf_t *conf;
130 int idx;
131 ngx_int_t rc;
132
133 xrdw_close_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
134 idx = (int)(unsigned char) req.fhandle[0];
135
136 if (!brix_validate_file_handle(ctx, c, idx, "CLOSE",
137 BRIX_OP_CLOSE, &rc)) {
138 return rc;
139 }
140
141 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
142 "brix: kXR_close handle=%d", idx);
143
144 /* pgwrite CSE close gate * Refuse to close while any page written via kXR_pgwrite failed CRC32c and
145 * was never corrected by a kXR_pgRetry. Otherwise the POSC rename / write-
146 * through flush below would commit known-corrupt bytes. The handle is left
147 * OPEN (no free) so the client can resend the bad pages and close again —
148 * matching stock do_PgClose, which returns before FTab->Del(). */
149 if (ctx->files[idx].pgw_fob_enabled) {
150 uint32_t left = brix_pgw_fob_count(&ctx->files[idx]);
151
152 if (left > 0) {
153 char emsg[64];
154
155 snprintf(emsg, sizeof(emsg), "%u uncorrected checksum error%s",
156 left, left == 1 ? "" : "s");
157 BRIX_OP_ERR(ctx, BRIX_OP_CLOSE);
158 return brix_send_error(ctx, c, kXR_ChkSumErr, emsg);
159 }
160 }
161
162 conf = ngx_stream_get_module_srv_conf(ctx->session,
163 ngx_stream_brix_module);
164
165 /* §6 CNS: a data server reports a just-written file to the manager so the
166 * cluster name space learns its size/mtime. Best-effort, fire-and-forget over
167 * the worker's manager link; only for writable handles being closed. */
168 brix_cns_emit_close(ctx, conf, idx);
169
170 /* Log before freeing so we still have the path and byte counters.
171 * detail = average throughput for the transfer ("%.2fMB/s").
172 * bytes = total data bytes transferred (read or written). */
173 {
174 char close_detail[64];
175 size_t br = ctx->files[idx].bytes_read;
176 size_t bw = ctx->files[idx].bytes_written;
177 size_t btotal = (bw > 0) ? bw : br;
178 ngx_msec_t dur = ngx_current_msec - ctx->files[idx].open_time;
179
180 if (btotal > 0 && dur > 0) {
181 double mbps = (double) btotal / (double) dur / 1000.0;
182 snprintf(close_detail, sizeof(close_detail), "%.2fMB/s", mbps);
183 } else {
184 snprintf(close_detail, sizeof(close_detail), "-");
185 }
186
187 /* Log using the final (intended) path name, not the staging temp path. */
188 const char *log_path = (ctx->files[idx].posc_final_path != NULL)
189 ? ctx->files[idx].posc_final_path
190 : ctx->files[idx].path;
191
192 brix_log_access(ctx, c, "CLOSE", log_path, close_detail,
193 1, 0, NULL, btotal);
194 }
195
196 /*
197 * POSC: on a clean kXR_close atomically rename the staging temp file to
198 * the final target path. We must do this before brix_free_fhandle()
199 * clears the fields. On success, clear posc_final_path so the free
200 * function does NOT unlink the now-renamed (and thus final) file.
201 */
202 if (ctx->files[idx].posc_final_path != NULL) {
203 const char *temp_path = ctx->files[idx].path;
204 const char *final_path = ctx->files[idx].posc_final_path;
205
206 /* When the partial lives on a separate stage device, record a durable
207 * "complete, pending stage-out" marker BEFORE the cross-device move so a
208 * worker death mid-move is recoverable: the reaper finishes the move on
209 * the next startup/sweep. Removed once the commit succeeds. */
210 ngx_flag_t stage_tracked = (conf != NULL
211 && conf->upload_stage_dir_canon[0] != '\0'
212 && ctx->files[idx].is_resume);
213 if (stage_tracked) {
214 (void) brix_stage_mark_pending(temp_path, final_path, c->log);
215 }
216
217 /* Commit the staged temp onto the final path: fsync + atomic rename on
218 * the same filesystem, or copy-then-rename when the staging device
219 * (brix_stage_dir) differs from the storage (cross-device EXDEV). */
220 /* final_mode 0: the root:// POSC temp is driver-created with the client's
221 * create mode already, so leave it as-is (rename preserves it). */
222 if (brix_commit_staged(ctx->files[idx].fd, temp_path, final_path,
223 0, c->log) != NGX_OK) {
224 int err = errno;
225 ngx_log_error(NGX_LOG_ERR, c->log, err,
226 "brix: staged commit \"%s\" -> \"%s\" failed",
227 temp_path, final_path);
228 /* Keep the staged partial (resume) — only the publish failed; the
229 * client can retry the close. Surface an I/O error. */
230 brix_close_finish_sess_xfer(ctx, idx, BRIX_SESS_XFER_ABORTED);
231 brix_free_fhandle(ctx, idx);
232 BRIX_OP_ERR(ctx, BRIX_OP_CLOSE);
233 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path,
234 ctx->login.dn[0] ? ctx->login.dn : NULL, 0,
235 BRIX_XFER_COMMIT_ERR, err, c->log);
236 return brix_send_error(ctx, c, kXR_IOError,
237 "staged commit failed");
238 }
239
240 if (stage_tracked) {
241 brix_stage_unmark_pending(temp_path);
242 }
243
244 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
245 "brix: staged commit \"%s\" -> \"%s\" ok",
246 temp_path, final_path);
247
248 /* Unified ledger: the root:// upload publication — the same audit line
249 * S3/WebDAV PUT and the other kinds emit (this path also covers a
250 * resumed upload's final commit). */
251 {
252 struct stat sb;
253 size_t n = (fstat(ctx->files[idx].fd, &sb) == 0
254 && S_ISREG(sb.st_mode)) ? (size_t) sb.st_size : 0;
255 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path,
256 ctx->login.dn[0] ? ctx->login.dn : NULL, n,
257 BRIX_XFER_OK, 0, c->log);
258 }
259 }
260
261 /* Write-through flush is part of the storage path now (Option A): a wt sd_stage
262 * handle flushes to the origin on close via sd_stage_wb_close (and on kXR_sync via
263 * the fsync job, which surfaces failures). The legacy run_flush loop is retired. */
264
265 /* Clear posc_final_path so brix_free_fhandle() does not unlink the
266 * now-renamed final file. This is intentionally after write-through so
267 * the flush can mirror the final POSC path instead of the temp path. */
268 if (ctx->files[idx].posc_final_path != NULL) {
269 ngx_free(ctx->files[idx].posc_final_path);
270 ctx->files[idx].posc_final_path = NULL;
271 }
272
273 if (ctx->files[idx].dashboard_slot >= 0 &&
274 ngx_brix_dashboard_shm_zone != NULL)
275 {
276 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
277 ctx->files[idx].dashboard_slot,
278 "close");
279 brix_transfer_slot_free(ngx_brix_dashboard_shm_zone->data,
280 ctx->files[idx].dashboard_slot);
281 ctx->files[idx].dashboard_slot = -1;
282 }
283
284 /* Flush the write-recovery journal so future writes with the same
285 * offsets after a reconnect are not mistakenly skipped as replays. */
286 brix_wrts_flush(&ctx->files[idx]);
287
288 brix_close_finish_sess_xfer(ctx, idx, BRIX_SESS_XFER_COMPLETE);
289
290 brix_free_fhandle(ctx, idx);
291
292 /* phase-59 W3a: release this user's throttle open-files slot. */
293 if (ctx->throttle.open_held > 0) {
294 ngx_stream_brix_srv_conf_t *tconf = ngx_stream_get_module_srv_conf(
295 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
296 if (tconf->throttle.zone != NULL) {
297 brix_throttle_open_dec(tconf->throttle.zone,
298 ctx->login.dn[0] ? ctx->login.dn : "anonymous");
299 }
300 ctx->throttle.open_held--;
301 }
302
303 /* A failed SYNC write-through flush means the file did not reach the origin;
304 * report the close as an error so the client does not assume durability. */
305 BRIX_OP_OK(ctx, BRIX_OP_CLOSE);
306
307 return brix_send_ok(ctx, c, NULL, 0);
308}
