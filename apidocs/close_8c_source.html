1/*
2 * close.c — kXR_close (3003) opcode handler: finalise and release an open file handle.
3 *
4 * WHAT: Implements brix_handle_close(), the protocol handler for kXR_close.
5 * It validates the client's file handle, emits an access-log line with
6 * transfer throughput, performs POSC commit, optional write-through
7 * mirroring and dashboard slot release, flushes the write-recovery
8 * journal, frees the handle, and replies with an empty kXR_ok.
9 *
10 * WHY: Close is where a transfer is made durable and visible. POSC
11 * (Persist-On-Successful-Close) files are written to a staging temp path
12 * and only promoted to their final name on a clean close, so close must
13 * fsync and rename atomically before the per-handle state is torn down;
14 * a rename failure must surface as kXR_IOError rather than silently
15 * leaving a partial file. It is also the only point where final byte
16 * counters, timing, and the intended (non-temp) path are still available
17 * for accurate logging and metrics.
18 *
19 * HOW: Resolve the handle index from req->fhandle[0] and validate it via
20 * brix_validate_file_handle(). Compute throughput ("%.2fMB/s") from
21 * bytes_written/bytes_read and (ngx_current_msec - open_time) and log
22 * against posc_final_path when set (never the staging temp path). If
23 * posc_final_path is set, fsync(fd) then rename(temp -> final); on
24 * failure free the handle and return kXR_IOError, on success retarget
25 * write-through to the final path. Run brix_wt_flush_on_close() when
26 * write-through is enabled, clear posc_final_path so brix_free_fhandle()
27 * does not unlink the now-renamed file, release any dashboard transfer
28 * slot, flush the write-recovery journal (brix_wrts_flush) so post-
29 * reconnect writes at the same offsets are not skipped as replays, free
30 * the handle, and send an empty kXR_ok.
31 */

50/*
51 * brix_cns_emit_close — §6: report a just-written file to the manager.
52 *
53 * Best-effort, fire-and-forget: only in EMIT mode, only for a writable handle,
54 * only when the worker's manager link is connected + logged in. The logical path
55 * (export-root-relative, what a client would stat) is derived by stripping
56 * root_canon from the handle's final path; size/mtime come from fstat of the fd.
57 */
58static void
59brix_cns_emit_close(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
60 int idx)
61{
62 const char *fpath, *logical, *root;
63 size_t rlen;
64 struct stat st;
65 uint8_t buf[BRIX_CNS_HDR_LEN + BRIX_CNS_PATH_MAX];
66 size_t n;
67
68 if (conf->cns_mode != BRIX_CNS_EMIT || !ctx->files[idx].writable) {
69 return;
70 }
71 if (conf->cms.ctx == NULL || conf->cms.ctx->connection == NULL
72 || !conf->cms.ctx->logged_in)
73 {
74 return;
75 }
76
77 fpath = (ctx->files[idx].posc_final_path != NULL)
78 ? ctx->files[idx].posc_final_path : ctx->files[idx].path;
79 if (fpath == NULL) {
80 return;
81 }
82 if (ctx->files[idx].fd < 0 || fstat(ctx->files[idx].fd, &st) != 0
83 || !S_ISREG(st.st_mode))
84 {
85 return;
86 }
87
88 root = conf->common.root_canon;
89 rlen = ngx_strlen(root);
90 logical = fpath;
91 if (rlen > 0 && ngx_strncmp(fpath, root, rlen) == 0 && fpath[rlen] == '/') {
92 logical = fpath + rlen; /* keep the leading '/' → client-facing path */
93 }
94
95 n = brix_cns_event_encode(BRIX_CNS_ADD, logical, (uint64_t) st.st_size,
96 (uint64_t) st.st_mtime, buf, sizeof(buf));
97 if (n == 0) {
98 return;
99 }
100 (void) brix_cms_send_frame(conf->cms.ctx->connection, 0, CMS_RR_CNS,
101 CMS_MOD_RAW, buf, n);
102}

104/*
105 * WHAT: Finish the sesslog transfer attached to a root file handle.
106 * WHY: kXR_close is the first point where final byte counters and the durable
107 * commit outcome are known while the handle state is still alive.
108 * HOW: Reconcile the intrusive xfer byte counter from the existing per-handle
109 * read/write totals, then emit COMPLETE or ABORTED before brix_free_fhandle().

112brix_close_finish_sess_xfer(brix_ctx_t *ctx, int idx,
113 brix_sess_xfer_status_t status)
114{
115 brix_file_t *file;
116 size_t total;
117

120 if (file->sess_xfer.active && total > file->sess_xfer.bytes) {
121 brix_sess_xfer_add(&file->sess_xfer,

123 }
124 brix_sess_xfer_end(ctx->sess, &file->sess_xfer, status);

127/*
128 * WHAT: kXR_pgwrite close gate — refuse close while uncorrected CRC32c errors
129 * remain on the handle.
130 * WHY: A clean close below would POSC-commit / write-through known-corrupt bytes;
131 * stock do_PgClose returns before FTab->Del(), so the handle stays OPEN for the
132 * client to resend the bad pages and close again.
133 * HOW: NGX_DECLINED = "no gate, continue". Otherwise stash the error-send's
134 * NGX_OK in *rc and return NGX_DONE = "handled, stop".
135 */
136static ngx_int_t
137brix_close_pgw_gate(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
138 ngx_int_t *rc)
139{
140 uint32_t left;
141 char emsg[64];
142
143 if (!ctx->files[idx].pgw_fob_enabled) {
144 return NGX_DECLINED;
145 }
146 left = brix_pgw_fob_count(&ctx->files[idx]);
147 if (left == 0) {
148 return NGX_DECLINED;
149 }
150 snprintf(emsg, sizeof(emsg), "%u uncorrected checksum error%s",
151 left, left == 1 ? "" : "s");
152 BRIX_OP_ERR(ctx, BRIX_OP_CLOSE);
153 *rc = brix_send_error(ctx, c, kXR_ChkSumErr, emsg);
154 return NGX_DONE;
155}

157/*
158 * WHAT: Whole-object staged-commit adapter (phase-70) — commit a staged write
159 * handle's object at close via a single backend PUT.
160 * WHY: A staged handle publishes only on a clean close, unless kXR_sync already
161 * committed it (idempotent); a commit failure must surface kXR_IOError and abort
162 * the handle so no partial object is published.
163 * HOW: NGX_DECLINED = "nothing to commit, continue". On failure, abort the
164 * session xfer, free the handle, stash the error-send's NGX_OK in *rc and return
165 * NGX_DONE = "handled, stop".
166 */
167static ngx_int_t
168brix_close_staged_commit(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
169 ngx_int_t *rc)
170{
171 int cerr = 0;
172
173 if (ctx->files[idx].writer == NULL || ctx->files[idx].staged_committed) {
174 return NGX_DECLINED;
175 }
176 if (brix_staged_commit_handle(ctx, idx, &cerr) == NGX_OK) {
177 return NGX_DECLINED;
178 }
179 brix_close_finish_sess_xfer(ctx, idx, BRIX_SESS_XFER_ABORTED);
180 brix_free_fhandle(ctx, idx);
181 BRIX_OP_ERR(ctx, BRIX_OP_CLOSE);
182 *rc = brix_send_error(ctx, c, kXR_IOError, "staged commit failed");
183 return NGX_DONE;
184}

186/*
187 * WHAT: Emit the CLOSE access-log line with average throughput.
188 * WHY: Must run before brix_free_fhandle() clears the path and byte counters,
189 * and must log the final (intended) path, never the staging temp path.
190 * HOW: total bytes = written or (else) read; detail = "%.2fMB/s" over the open
191 * duration, or "-" when nothing transferred / zero elapsed.
192 */
193static void
194brix_close_log_access(brix_ctx_t *ctx, ngx_connection_t *c, int idx)
195{
196 char close_detail[64];
197 size_t br = ctx->files[idx].bytes_read;
198 size_t bw = ctx->files[idx].bytes_written;
199 size_t btotal = (bw > 0) ? bw : br;
200 ngx_msec_t dur = ngx_current_msec - ctx->files[idx].open_time;
201 const char *log_path;
202
203 if (btotal > 0 && dur > 0) {
204 double mbps = (double) btotal / (double) dur / 1000.0;
205 snprintf(close_detail, sizeof(close_detail), "%.2fMB/s", mbps);
206 } else {
207 snprintf(close_detail, sizeof(close_detail), "-");
208 }
209
210 log_path = (ctx->files[idx].posc_final_path != NULL)
211 ? ctx->files[idx].posc_final_path
212 : ctx->files[idx].path;
213
214 brix_log_access(ctx, c, "CLOSE", log_path, close_detail,
215 1, 0, NULL, btotal);
216}

218/*
219 * WHAT: POSC commit — atomically publish the staging temp file onto its final
220 * target path on a clean close.
221 * WHY: Must run before brix_free_fhandle() clears the fields; a rename failure
222 * surfaces as kXR_IOError rather than silently leaving a partial file, keeping
223 * the staged partial for a client retry.
224 * HOW: NGX_DECLINED = "no POSC handle, continue". Mark a durable pending marker
225 * before a cross-device move, fsync+rename via brix_commit_staged(), emit the
226 * unified ledger line, and clear posc_final_path on success. On failure, abort
227 * the session xfer, free the handle, stash the error-send's NGX_OK in *rc and
228 * return NGX_DONE.
229 */
230static ngx_int_t
231brix_close_posc_commit(brix_ctx_t *ctx, ngx_connection_t *c,
232 ngx_stream_brix_srv_conf_t *conf, int idx, ngx_int_t *rc)
233{
234 const char *temp_path;
235 const char *final_path;
236 ngx_flag_t stage_tracked;
237
238 if (ctx->files[idx].posc_final_path == NULL) {
239 return NGX_DECLINED;
240 }
241
242 temp_path = ctx->files[idx].path;
243 final_path = ctx->files[idx].posc_final_path;
244
245 /* When the partial lives on a separate stage device, record a durable
246 * "complete, pending stage-out" marker BEFORE the cross-device move so a
247 * worker death mid-move is recoverable: the reaper finishes the move on
248 * the next startup/sweep. Removed once the commit succeeds. */
249 stage_tracked = (conf != NULL
250 && conf->upload_stage_dir_canon[0] != '\0'
251 && ctx->files[idx].is_resume);
252 if (stage_tracked) {
253 (void) brix_stage_mark_pending(temp_path, final_path, c->log);
254 }
255
256 /* Commit the staged temp onto the final path: fsync + atomic rename on
257 * the same filesystem, or copy-then-rename when the staging device
258 * (brix_stage_dir) differs from the storage (cross-device EXDEV). */
259 /* final_mode 0: the root:// POSC temp is driver-created with the client's
260 * create mode already, so leave it as-is (rename preserves it). */
261 if (brix_commit_staged(ctx->files[idx].fd, temp_path, final_path,
262 0, c->log) != NGX_OK) {
263 int err = errno;
264 ngx_log_error(NGX_LOG_ERR, c->log, err,
265 "brix: staged commit \"%s\" -> \"%s\" failed",
266 temp_path, final_path);
267 /* Keep the staged partial (resume) — only the publish failed; the
268 * client can retry the close. Surface an I/O error. */
269 brix_close_finish_sess_xfer(ctx, idx, BRIX_SESS_XFER_ABORTED);
270 brix_free_fhandle(ctx, idx);
271 BRIX_OP_ERR(ctx, BRIX_OP_CLOSE);
272 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path,
273 ctx->login.dn[0] ? ctx->login.dn : NULL, 0,
274 BRIX_XFER_COMMIT_ERR, err, c->log);
275 *rc = brix_send_error(ctx, c, kXR_IOError, "staged commit failed");
276 return NGX_DONE;
277 }
278
279 if (stage_tracked) {
280 brix_stage_unmark_pending(temp_path);
281 }
282
283 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
284 "brix: staged commit \"%s\" -> \"%s\" ok",
285 temp_path, final_path);
286
287 /* Unified ledger: the root:// upload publication — the same audit line
288 * S3/WebDAV PUT and the other kinds emit (this path also covers a
289 * resumed upload's final commit). */
290 {
291 struct stat sb;
292 size_t n = (fstat(ctx->files[idx].fd, &sb) == 0
293 && S_ISREG(sb.st_mode)) ? (size_t) sb.st_size : 0;
294 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path,
295 ctx->login.dn[0] ? ctx->login.dn : NULL, n,
296 BRIX_XFER_OK, 0, c->log);
297 }
298
299 return NGX_DECLINED;
300}

302/*
303 * WHAT: Post-commit teardown — release the dashboard transfer slot and flush the
304 * write-recovery journal.
305 * WHY: The dashboard slot must be counted+freed and the wrts journal flushed so
306 * future writes at the same offsets after a reconnect are not skipped as replays;
307 * both run after a successful commit while the handle is still alive.
308 * HOW: Clear posc_final_path (so brix_free_fhandle does not unlink the renamed
309 * final file), then count/free any dashboard slot and brix_wrts_flush().
310 */
311static void
312brix_close_release_slots(brix_ctx_t *ctx, int idx)
313{
314 /* Clear posc_final_path so brix_free_fhandle() does not unlink the
315 * now-renamed final file. This is intentionally after write-through so
316 * the flush can mirror the final POSC path instead of the temp path. */
317 if (ctx->files[idx].posc_final_path != NULL) {
318 ngx_free(ctx->files[idx].posc_final_path);
319 ctx->files[idx].posc_final_path = NULL;
320 }
321
322 if (ctx->files[idx].dashboard_slot >= 0 &&
323 ngx_brix_dashboard_shm_zone != NULL)
324 {
325 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
326 ctx->files[idx].dashboard_slot,
327 "close");
328 brix_transfer_slot_free(ngx_brix_dashboard_shm_zone->data,
329 ctx->files[idx].dashboard_slot);
330 ctx->files[idx].dashboard_slot = -1;
331 }
332
333 /* Flush the write-recovery journal so future writes with the same
334 * offsets after a reconnect are not mistakenly skipped as replays. */
335 brix_wrts_flush(&ctx->files[idx]);
336}

362ngx_int_t brix_handle_close(brix_ctx_t *ctx, ngx_connection_t *c) {
363 xrdw_close_req_t req;
364 ngx_stream_brix_srv_conf_t *conf;
365 int idx;
366 ngx_int_t rc = NGX_OK;
367
368 xrdw_close_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
369 idx = (int)(unsigned char) req.fhandle[0];
370
371 if (!brix_validate_file_handle(ctx, c, idx, "CLOSE",
372 BRIX_OP_CLOSE, &rc)) {
373 return rc;
374 }
375
376 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
377 "brix: kXR_close handle=%d", idx);
378
379 if (brix_close_pgw_gate(ctx, c, idx, &rc) == NGX_DONE) {
380 return rc;
381 }
382
383 conf = ngx_stream_get_module_srv_conf(ctx->session,
384 ngx_stream_brix_module);
385
386 if (brix_close_staged_commit(ctx, c, idx, &rc) == NGX_DONE) {
387 return rc;
388 }
389
390 /* §6 CNS: a data server reports a just-written file to the manager so the
391 * cluster name space learns its size/mtime. Best-effort, fire-and-forget over
392 * the worker's manager link; only for writable handles being closed. */
393 brix_cns_emit_close(ctx, conf, idx);
394
395 brix_close_log_access(ctx, c, idx);
396
397 if (brix_close_posc_commit(ctx, c, conf, idx, &rc) == NGX_DONE) {
398 return rc;
399 }
400
401 /* Write-through flush is part of the storage path now (Option A): a wt sd_stage
402 * handle flushes to the origin on close via sd_stage_wb_close (and on kXR_sync via
403 * the fsync job, which surfaces failures). The legacy run_flush loop is retired. */
404
405 brix_close_release_slots(ctx, idx);
406
407 brix_close_finish_sess_xfer(ctx, idx, BRIX_SESS_XFER_COMPLETE);
408
409 brix_free_fhandle(ctx, idx);
410
411 brix_close_release_throttle(ctx, c);
412
413 /* A failed SYNC write-through flush means the file did not reach the origin;
414 * report the close as an error so the client does not assume durability. */
415 BRIX_OP_OK(ctx, BRIX_OP_CLOSE);
416
417 return brix_send_ok(ctx, c, NULL, 0);
418}
