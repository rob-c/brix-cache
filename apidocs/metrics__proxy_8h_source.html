4 * Per-server root:// reverse-proxy metrics: the aggregate counter block plus a

6 * brix_proxy on. Split out of metrics.h so each observability domain owns a

21/*

45/*
46 * Per-server proxy counter block. Tracks outbound upstream connections and

49typedef struct {
50 ngx_atomic_t upstream_connects_total; /* successful TCP (or TLS) connects */
51 ngx_atomic_t upstream_connect_errors; /* TCP connect / TLS handshake errors */
52 ngx_atomic_t upstream_auth_errors; /* upstream login/auth rejected */
53 ngx_atomic_t opens_total; /* kXR_open forwarded OK */
54 ngx_atomic_t open_errors; /* kXR_open forwarded, upstream errored */
55 ngx_atomic_t reads_total; /* kXR_read/pgread/readv forwarded OK */
56 ngx_atomic_t read_bytes_total; /* bytes returned to client via proxy */
57 ngx_atomic_t writes_total; /* kXR_write/pgwrite/writev forwarded OK */
58 ngx_atomic_t write_bytes_total; /* bytes written upstream via proxy */
59 ngx_atomic_t closes_total; /* kXR_close forwarded OK */
60 ngx_atomic_t abandoned_handles_total; /* open handles freed on disconnect */
61 ngx_atomic_t reconnects_total; /* idle upstream reconnect attempts */
62 ngx_atomic_t path_ops_total; /* path-based mutations forwarded OK */
63 ngx_atomic_t path_op_errors_total; /* path-based mutations upstream errored */
64 ngx_atomic_t wait_responses_total; /* kXR_wait responses absorbed by proxy */
65
66 /* Per-upstream breakdown (upstream_idx → slice). Populated at first use. */
67 ngx_brix_proxy_upstream_metrics_t upstreams[BRIX_PROXY_MAX_UPSTREAMS];
68} ngx_brix_proxy_metrics_t;
