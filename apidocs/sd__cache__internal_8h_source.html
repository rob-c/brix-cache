69/* ---- CVMFS manifest/whitelist signature verify (sd_cache_manifest.c) ----- */

71/* Phase-85 F1: verify a MANIFEST-class staged fill before commit when the
72 * repo master public key is configured (policy.cvmfs_master_pub):
73 * .cvmfspublished — full chain: whitelist sig vs master key → whitelist not
74 * expired → manifest cert fingerprint ∈ whitelist → manifest sig vs cert
75 * (whitelist + certificate fetched through the fill's source tier);
76 * .cvmfswhitelist — signature vs master key + expiry.
77 * Returns NGX_OK (verified, or key is not a signed-metadata shape / verify not
78 * configured — commit proceeds); NGX_ERROR (verification definitively FAILED —
79 * the caller emits signal=cvmfs_tamper, quarantines, aborts, EBADMSG); or
80 * NGX_DECLINED (the chain could not be evaluated — sibling fetch / part read
81 * failed; the caller fails the fill closed with EIO, NO tamper signal, so an
82 * origin outage never feeds the maxretry=1 tamper jail). `pp` is the staged
83 * part path. */
