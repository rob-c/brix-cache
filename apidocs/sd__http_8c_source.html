1/*
2 * sd_http.c — read-only HTTP(S) source storage driver (phase-63 C-4). See header.
3 *
4 * A thin driver over the injected brix_s3_transport_t (the same vtable the S3
5 * driver uses): `open`/`stat` HEAD the URL for the size, `pread` issues a byte
6 * Range GET. No SigV4, no auth — plain anonymous HTTP. No kernel fd ⇒ memory-served.
7 */

23/* Force-primary read policy (process-global operator toggle; set pre-fork from
24 * the cvmfs merge when brix_cvmfs_fill_retry_policy is force-primary, so all
25 * workers inherit it — the trace/timeouts idiom). When set, a read always
26 * targets the RANK-PREFERRED endpoint and NEVER fails over to an alternate on a
27 * transport failure: the fill loop retries the SAME preferred origin (RAL) with
28 * a fresh connection until it forces through or the client-hold budget expires.
29 * Off (default) keeps the phase-68 T11 alternate-endpoint failover. */
30static int g_sd_http_force_primary;
31
32void
33sd_http_force_primary_set(int on)
34{
35 g_sd_http_force_primary = on ? 1 : 0;
36}

44/* Per-staged-write state: HTTP has no streaming PUT through this transport, so the
45 * object is buffered and PUT whole at commit (a remote stage/cache store of typical
46 * file sizes; very large objects are a multipart follow-up). */
47typedef struct {
48 char path[SD_HTTP_PATH_MAX];
49 u_char *buf;
50 size_t len;
51 size_t cap;
52} sd_http_staged_state;

54/* Compose the WRITE-target URL path: writes (staged PUT, DELETE) always go
55 * to endpoint 0 — failing a write over to another origin would split-brain
56 * the store; read failover (sd_http_request_fo) never applies here. */
57static void
58sd_http_write_path(const sd_http_inst_state *is, const char *key, char *dst,
59 size_t cap)
60{
61 snprintf(dst, cap, "%s%s", is->eps[0].base_path,
62 (key != NULL && key[0]) ? key : "/");
63}

65/* Effective pick score: policy rank first, health inside a rank. */
66static int
67sd_http_ep_score(const sd_http_endpoint *ep)
68{
69 return atomic_load_explicit((_Atomic int *) &ep->rank,
70 memory_order_relaxed)
71 * SD_HTTP_RANK_WEIGHT + ep->fail_score;
72}

74/* Pick the best endpoint (lowest effective score, order-stable ties). */
75static sd_http_endpoint *
76sd_http_pick(sd_http_inst_state *is)
77{
78 sd_http_endpoint *best = &is->eps[0];
79 int i;
80
81 for (i = 1; i < is->n_eps; i++) {
82 if (sd_http_ep_score(&is->eps[i]) < sd_http_ep_score(best)) {
83 best = &is->eps[i];
84 }
85 }
86 return best;
87}

89/* Rank-preferred endpoint ignoring health — the half-open probe target. */
90static sd_http_endpoint *
91sd_http_preferred(sd_http_inst_state *is)
92{
93 sd_http_endpoint *best = &is->eps[0];
94 int i;
95
96 for (i = 1; i < is->n_eps; i++) {
97 if (atomic_load_explicit(&is->eps[i].rank, memory_order_relaxed)
98 < atomic_load_explicit(&best->rank, memory_order_relaxed))
99 {
100 best = &is->eps[i];
101 }
102 }
103 return best;
104}

106/* Record one transport outcome: score = score*7/8 + (ok ? 0 : 256). */
107static void
108sd_http_score(sd_http_endpoint *ep, int ok)
109{
110 ep->fail_score = ep->fail_score * 7 / 8 + (ok ? 0 : 256);
111}
112
113/* Score + health-transition note in one step: crossing the 128 hysteresis
114 * band emits ONE degraded/recovered event (not one per failure) through the
115 * owner-injected hook — the operator's origin-flap trail. */
116static void
117sd_http_score_noted(sd_http_inst_state *is, sd_http_endpoint *ep, int ok)
118{
119 int was = ep->fail_score;
120
121 sd_http_score(ep, ok);
122 if (is->health_note == NULL) {
123 return;
124 }
125 if (!ok && was < 128 && ep->fail_score >= 128) {
126 is->health_note(ep->host, ep->port, 0);
127 } else if (ok && was >= 128 && ep->fail_score < 128) {
128 is->health_note(ep->host, ep->port, 1);
129 }
130}

150/* Selection audit: log (NOTICE) every change of the endpoint that answers —
151 * the "why is RAL being skipped for CERN" record. States why the policy-
152 * preferred endpoint was overridden when it was (health benching). Called
153 * from fill threads; cur_ep is racy-by-design (see the field comment). */
154static void
155sd_http_log_switch(sd_http_inst_state *is, sd_http_endpoint *ep)
156{
157 int idx = (int) (ep - is->eps);
158 sd_http_endpoint *pref;
159 char prev[300];
160
161 if (idx == is->cur_ep) {
162 return;
163 }
164 if (is->cur_ep >= 0 && is->cur_ep < is->n_eps) {
165 snprintf(prev, sizeof(prev), "%s:%d",
166 is->eps[is->cur_ep].host, is->eps[is->cur_ep].port);
167 } else {
168 snprintf(prev, sizeof(prev), "(none)");
169 }
170 is->cur_ep = idx;
171 if (is->log == NULL) {
172 return;
173 }
174 pref = sd_http_preferred(is);
175 if (ep == pref) {
176 ngx_log_error(NGX_LOG_NOTICE, is->log, 0,
177 "brix: http origin switched to %s:%d (endpoint %d, rank %d, "
178 "fail_score %d; the policy-preferred endpoint), was %s",
179 ep->host, ep->port, idx, sd_http_ep_rank(ep), ep->fail_score,
180 prev);
181 } else {
182 ngx_log_error(NGX_LOG_NOTICE, is->log, 0,
183 "brix: http origin switched to %s:%d (endpoint %d, rank %d, "
184 "fail_score %d), was %s; policy-preferred %s:%d SKIPPED "
185 "(rank %d, fail_score %d - benched by recent transport failures, "
186 "recovers via half-open probing as its score decays)",
187 ep->host, ep->port, idx, sd_http_ep_rank(ep), ep->fail_score,
188 prev, pref->host, pref->port, sd_http_ep_rank(pref),
189 pref->fail_score);
190 }
191}

193/* One request with read-failover (phase-68 T11): try the best endpoint, then
194 * ONE alternate on a TRANSPORT failure (an HTTP 4xx is NOT a transport
195 * failure — the object genuinely isn't there; do not mask it by failing
196 * over). `extra_hdrs` is the pre-joined header block (Range and/or auth).
197 * On success *used (when non-NULL) names the endpoint that answered. */
198static int
199sd_http_request_fo(sd_http_inst_state *is, const char *method, const char *key,
200 const char *extra_hdrs, brix_s3_resp_t *resp, sd_http_endpoint **used)
201{
202 int force_primary = g_sd_http_force_primary;
203 /* Force-primary pins the rank-preferred endpoint (ignore health — there is
204 * nowhere to fail over to by policy); otherwise pick best-by-score. The
205 * outer fill loop retries this same origin on a fresh connection. */
206 sd_http_endpoint *ep = force_primary ? sd_http_preferred(is)
207 : sd_http_pick(is);
208 sd_http_endpoint *first;
209 char full[SD_HTTP_PATH_MAX], errbuf[256], klog[160];
210 int attempt, rc, i, score_before;
211
212 /* Half-open recovery: scores only move on outcomes, so a benched origin
213 * would stay benched forever. Every 4th request re-tries the rank-
214 * preferred endpoint; a recovered origin earns its score back (and the
215 * in-loop failover still answers the client if it is still down).
216 * Skipped under force-primary: the preferred endpoint is already pinned. */
217 if (!force_primary && is->n_eps > 1 && (++is->probe_tick & 3u) == 0) {
218 sd_http_endpoint *pref = sd_http_preferred(is);
219
220 if (pref->fail_score > 0) {
221 if (pref != ep && is->log != NULL) {
222 ngx_log_error(NGX_LOG_INFO, is->log, 0,
223 "brix: http origin half-open probe: re-trying benched "
224 "preferred %s:%d (rank %d, fail_score %d) instead of "
225 "%s:%d for %s \"%s\"",
226 pref->host, pref->port, sd_http_ep_rank(pref),
227 pref->fail_score, ep->host, ep->port, method,
228 sd_http_log_key(key, klog, sizeof(klog)));
229 }
230 ep = pref;
231 }
232 }
233 first = ep;
234
235 for (attempt = 0; attempt < 2; attempt++) {
236 snprintf(full, sizeof(full), "%s%s", ep->base_path,
237 (key != NULL && key[0]) ? key : "/");
238 errbuf[0] = '\0';
239 score_before = ep->fail_score;
240 rc = is->transport->request(is->tctx, ep->host, ep->port, ep->tls,
241 method, full, extra_hdrs, NULL, 0,
242 is->timeout_ms, resp,
243 errbuf, sizeof(errbuf));
244 sd_http_score_noted(is, ep, rc == 0);
245 if (rc == 0) {
246 if (used != NULL) {
247 *used = ep;
248 }
249 snprintf(is->last_origin, sizeof(is->last_origin), "%s:%d",
250 ep->host, ep->port);
251 /* "failover" for the per-upstream metric = served by a NON-PRIMARY
252 * endpoint (not the configured endpoint 0). Keyed on eps[0] rather
253 * than `first` so it holds however scoring reordered the picks and
254 * across the HEAD-then-GET sub-requests of one fill. */
255 is->last_failover = (ep != &is->eps[0]);
256 sd_http_log_switch(is, ep);
257 return 0;
258 }
259 /* The transport failure is the evidence an operator needs to explain
260 * a later "origin switched" line — record it with the curl detail. */
261 if (is->log != NULL) {
262 ngx_log_error(NGX_LOG_WARN, is->log, 0,
263 "brix: http origin %s:%d failed %s \"%s\": %s "
264 "(attempt %d/2, rank %d, fail_score %d -> %d)",
265 ep->host, ep->port, method,
266 sd_http_log_key(key, klog, sizeof(klog)),
267 errbuf[0] ? errbuf : "transport error",
268 attempt + 1, sd_http_ep_rank(ep), score_before,
269 ep->fail_score);
270 }
271 if (is->n_eps < 2 || force_primary) {
272 break; /* force-primary: never fail over */
273 }
274 /* next-best endpoint distinct from the first attempt */
275 ep = NULL;
276 for (i = 0; i < is->n_eps; i++) {
277 if (&is->eps[i] == first) {
278 continue;
279 }
280 if (ep == NULL
281 || sd_http_ep_score(&is->eps[i]) < sd_http_ep_score(ep))
282 {
283 ep = &is->eps[i];
284 }
285 }
286 if (ep == NULL) {
287 break;
288 }
289 if (is->failover_note != NULL) {
290 is->failover_note(); /* T16: failover accounting */
291 }
292 if (is->log != NULL) {
293 ngx_log_error(NGX_LOG_NOTICE, is->log, 0,
294 "brix: http origin failover for %s \"%s\": %s:%d -> %s:%d "
295 "(alternate rank %d, fail_score %d)",
296 method, sd_http_log_key(key, klog, sizeof(klog)),
297 first->host, first->port, ep->host, ep->port,
298 sd_http_ep_rank(ep), ep->fail_score);
299 }
300 }
301 if (is->log != NULL) {
302 ngx_log_error(NGX_LOG_ERR, is->log, 0,
303 "brix: http origin request exhausted all endpoints (%d tried) "
304 "for %s \"%s\" - reporting EIO to the fill layer",
305 (is->n_eps < 2) ? 1 : 2, method,
306 sd_http_log_key(key, klog, sizeof(klog)));
307 }
308 errno = EIO;
309 return -1;
310}

312/* HEAD `key` → *size_out (−1 if no Content-Length). 0, or −1 with errno. */
313static int
314sd_http_head_size(sd_http_inst_state *is, const char *key, int64_t *size_out)
315{
316 brix_s3_resp_t resp;
317 char cl[32];
318
319 if (sd_http_request_fo(is, "HEAD", key,
320 is->auth_hdr[0] ? is->auth_hdr : NULL,
321 &resp, NULL) != 0)
322 {
323 return -1; /* errno = EIO */
324 }
325 if (resp.status == 404) {
326 is->transport->resp_free(&resp);
327 errno = ENOENT;
328 return -1;
329 }
330 if (resp.status != 200) {
331 is->transport->resp_free(&resp);
332 errno = (resp.status == 403 || resp.status == 401) ? EACCES : EIO;
333 return -1;
334 }
335 if (is->transport->resp_header(&resp, "Content-Length", cl, sizeof(cl)) == 0) {
336 *size_out = (int64_t) strtoll(cl, NULL, 10);
337 } else {
338 *size_out = -1;
339 }
340 is->transport->resp_free(&resp);
341 return 0;
342}

344static brix_sd_obj_t *
345sd_http_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
346 mode_t mode, int *err_out)
347{
348 sd_http_inst_state *is = inst->state;
349 sd_http_obj_state *st;
350 brix_sd_obj_t *obj;
351 int64_t size = 0;
352
353 (void) mode;
354
355 /* Read-only source: refuse any write/create/trunc intent. */
356 if (sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC
357 | BRIX_SD_O_APPEND))
358 {
359 if (err_out) { *err_out = EROFS; }
360 return NULL;
361 }
362
363 if (sd_http_head_size(is, path, &size) != 0) {
364 if (err_out) { *err_out = errno; }
365 return NULL;
366 }
367
368 st = calloc(1, sizeof(*st));
369 obj = calloc(1, sizeof(*obj));
370 if (st == NULL || obj == NULL) {
371 free(st);
372 free(obj);
373 if (err_out) { *err_out = ENOMEM; }
374 return NULL;
375 }
376 snprintf(st->key, sizeof(st->key), "%s",
377 (path != NULL && path[0]) ? path : "/");
378
379 obj->driver = inst->driver;
380 obj->inst = inst;
381 obj->fd = NGX_INVALID_FILE; /* no kernel fd — memory-served */
382 obj->state = st;
383 obj->heap_shell = 1;
384 obj->snap.size = (off_t) size;
385 obj->snap.mode = S_IFREG | 0444;
386 obj->snap.is_reg = 1;
387 return obj;
388}

400static ssize_t
401sd_http_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
402{
403 sd_http_inst_state *is = obj->inst->state;
404 sd_http_obj_state *st = obj->state;
405 brix_s3_resp_t resp;
406 char hdrs[SD_HTTP_AUTH_MAX + 80];
407 const void *body;
408 size_t blen = 0, n;
409 int64_t end;
410
411 if (len == 0) {
412 return 0;
413 }
414 if (len > (size_t) SD_HTTP_PREAD_MAX) {
415 len = (size_t) SD_HTTP_PREAD_MAX;
416 }
417 end = (int64_t) off + (int64_t) len - 1;
418 snprintf(hdrs, sizeof(hdrs), "Range: bytes=%lld-%lld\r\n%s",
419 (long long) off, (long long) end, is->auth_hdr);
420
421 if (sd_http_request_fo(is, "GET", st->key, hdrs, &resp, NULL) != 0) {
422 return -1; /* errno = EIO */
423 }
424 if (resp.status == 416) {
425 is->transport->resp_free(&resp);
426 return 0; /* range past EOF → EOF (0) */
427 }
428 if (resp.status != 206 && resp.status != 200) {
429 is->transport->resp_free(&resp);
430 errno = (resp.status == 404) ? ENOENT : EIO;
431 return -1;
432 }
433 body = is->transport->resp_body(&resp, &blen);
434 if (body == NULL || blen == 0) {
435 is->transport->resp_free(&resp);
436 return 0; /* EOF / empty range */
437 }
438
439 /* 206 → body is exactly the requested range, starting at `off`. 200 → the
440 * origin ignored the Range header and returned the WHOLE object from byte 0
441 * (stock python http.server, some proxies), so the bytes we want begin at
442 * `off` within `body`; past EOF that is a short read of 0. Slicing here keeps
443 * a correct, terminating fill loop against either kind of origin. */
444 if (resp.status == 200) {
445 if ((size_t) off >= blen) {
446 is->transport->resp_free(&resp);
447 return 0; /* requested range past EOF */
448 }
449 body = (const char *) body + off;
450 blen -= (size_t) off;
451 }
452 n = (blen < len) ? blen : len;
453 memcpy(buf, body, n);
454 is->transport->resp_free(&resp);
455 return (ssize_t) n;
456}

509static ssize_t
510sd_http_staged_write(brix_sd_staged_t *h, const void *buf, size_t len,
511 off_t off)
512{
513 sd_http_staged_state *ss = h->state;
514
515 /* Sequential append only (whole-object PUT has no random write). */
516 if ((size_t) off != ss->len) {
517 errno = ESPIPE;
518 return -1;
519 }
520 if (ss->len + len > ss->cap) {
521 size_t ncap = ss->cap ? ss->cap * 2 : (1u << 20);
522 u_char *nbuf;
523
524 while (ncap < ss->len + len) {
525 ncap *= 2;
526 }
527 nbuf = realloc(ss->buf, ncap);
528 if (nbuf == NULL) {
529 errno = ENOMEM;
530 return -1;
531 }
532 ss->buf = nbuf;
533 ss->cap = ncap;
534 }
535 ngx_memcpy(ss->buf + ss->len, buf, len);
536 ss->len += len;
537 return (ssize_t) len;
538}

540static ngx_int_t
541sd_http_staged_commit(brix_sd_staged_t *h, int noreplace)
542{
543 sd_http_staged_state *ss = h->state;
544 sd_http_inst_state *is = h->inst->state;
545 brix_s3_resp_t resp;
546 char errbuf[256];
547 ngx_int_t rc = NGX_OK;
548
549 (void) noreplace; /* HTTP PUT always replaces */
550 if (is->transport->request(is->tctx, is->eps[0].host, is->eps[0].port,
551 is->eps[0].tls, "PUT",
552 ss->path, is->auth_hdr[0] ? is->auth_hdr : NULL,
553 ss->buf, ss->len, is->timeout_ms, &resp,
554 errbuf, sizeof(errbuf)) != 0)
555 {
556 free(ss->buf);
557 free(ss);
558 free(h);
559 errno = EIO;
560 return NGX_ERROR;
561 }
562 if (resp.status != 200 && resp.status != 201 && resp.status != 204) {
563 errno = (resp.status == 403 || resp.status == 401) ? EACCES : EIO;
564 rc = NGX_ERROR;
565 }
566 is->transport->resp_free(&resp);
567 free(ss->buf);
568 free(ss);
569 free(h);
570 return rc;
571}

583static ngx_int_t
584sd_http_unlink(brix_sd_instance_t *inst, const char *path, int is_dir)
585{
586 sd_http_inst_state *is = inst->state;
587 brix_s3_resp_t resp;
588 char errbuf[256], full[SD_HTTP_PATH_MAX];
589
590 (void) is_dir;
591 sd_http_write_path(is, path, full, sizeof(full));
592 if (is->transport->request(is->tctx, is->eps[0].host, is->eps[0].port,
593 is->eps[0].tls, "DELETE",
594 full, is->auth_hdr[0] ? is->auth_hdr : NULL,
595 NULL, 0, is->timeout_ms, &resp,
596 errbuf, sizeof(errbuf)) != 0)
597 {
598 errno = EIO;
599 return NGX_ERROR;
600 }
601 /* Idempotent: 204/200 ok, 404 already gone. */
602 if (resp.status != 204 && resp.status != 200 && resp.status != 404) {
603 is->transport->resp_free(&resp);
604 errno = EIO;
605 return NGX_ERROR;
606 }
607 is->transport->resp_free(&resp);
608 return NGX_OK;
609}

638brix_sd_instance_t *
639brix_sd_http_create(const brix_sd_http_cfg_t *cfg, ngx_log_t *log)
640{
641 brix_sd_instance_t *inst;
642 sd_http_inst_state *is;
643
644 if (cfg == NULL || cfg->host == NULL || cfg->host[0] == '\0'
645 || cfg->port <= 0 || cfg->port > 65535 || cfg->transport == NULL)
646 {
647 errno = EINVAL;
648 return NULL;
649 }
650 inst = calloc(1, sizeof(*inst));
651 is = calloc(1, sizeof(*is));
652 if (inst == NULL || is == NULL) {
653 free(inst);
654 free(is);
655 errno = ENOMEM;
656 return NULL;
657 }
658 snprintf(is->eps[0].host, sizeof(is->eps[0].host), "%s", cfg->host);
659 is->eps[0].port = cfg->port;
660 is->eps[0].tls = cfg->tls;
661 snprintf(is->eps[0].base_path, sizeof(is->eps[0].base_path), "%s",
662 (cfg->base_path != NULL) ? cfg->base_path : "");
663 is->n_eps = 1;
664 if (cfg->extra != NULL && cfg->n_extra > 0) {
665 int i, n = cfg->n_extra;
666
667 if (n > SD_HTTP_EP_MAX - 1) {
668 n = SD_HTTP_EP_MAX - 1;
669 }
670 for (i = 0; i < n; i++) {
671 const brix_sd_http_ep_cfg_t *ec = &cfg->extra[i];
672
673 if (ec->host == NULL || ec->host[0] == '\0' || ec->port <= 0
674 || ec->port > 65535)
675 {
676 continue;
677 }
678 snprintf(is->eps[is->n_eps].host, sizeof(is->eps[0].host), "%s",
679 ec->host);
680 is->eps[is->n_eps].port = ec->port;
681 is->eps[is->n_eps].tls = ec->tls;
682 snprintf(is->eps[is->n_eps].base_path,
683 sizeof(is->eps[0].base_path), "%s",
684 (ec->base_path != NULL) ? ec->base_path : "");
685 is->n_eps++;
686 }
687 }
688 is->transport = cfg->transport;
689 is->failover_note = cfg->failover_note;
690 is->health_note = cfg->health_note;
691 is->tctx = cfg->tctx;
692 is->timeout_ms = (cfg->timeout_ms > 0) ? cfg->timeout_ms : 60000;
693 is->log = log;
694 is->cur_ep = -1;
695 if (cfg->bearer_token != NULL && cfg->bearer_token[0] != '\0') {
696 snprintf(is->auth_hdr, sizeof(is->auth_hdr),
697 "Authorization: Bearer %s\r\n", cfg->bearer_token);
698 }
699
700 inst->driver = &brix_sd_http_driver;
701 inst->log = log;
702 inst->pool = NULL;
703 inst->state = is;
704 return inst;
705}
