1/*
2 * sd_http.c — read-only-primary HTTP(S) source storage driver: the driver
3 * vtable + instance lifecycle (create/destroy). See sd_http.h.
4 *
5 * A thin driver over the injected brix_s3_transport_t (the same vtable the S3
6 * driver uses): `open`/`stat` HEAD the URL for the size, `pread` issues a byte
7 * Range GET, and a staged whole-object PUT / DELETE make it a writable cache /
8 * stage store. No kernel fd ⇒ memory-served.
9 *
10 * phase-79 file-size split — the driver is four translation units around one
11 * concept each; this file owns only the vtable wiring and create/destroy:
12 * sd_http_select.c — endpoint selection, health scoring, read failover
13 * sd_http_read.c — HEAD/GET read path + per-open credential resolution
14 * sd_http_write.c — staged whole-object PUT + DELETE write path
15 * sd_http_introspect.c — T19/T20 selection + health introspection API
16 * The seams between them are declared in sd_http_internal.h; the driver struct
17 * and sd_http_instance_is stay file-private here (the check is against this
18 * file's driver address).
19 */

61/* sd_http_init_endpoints — fill the endpoint table from the primary host plus
62 * any extra failover origins.
63 *
64 * WHAT: Writes eps[0] from cfg->{host,port,tls,base_path}, then appends up to
65 * SD_HTTP_EP_MAX-1 valid extras, setting is->n_eps.
66 * HOW: Primary is always present (create() validated it); each extra is
67 * skipped when its host/port is missing or out of range. */
68static void
69sd_http_init_endpoints(sd_http_inst_state *is, const brix_sd_http_cfg_t *cfg)
70{
71 snprintf(is->eps[0].host, sizeof(is->eps[0].host), "%s", cfg->host);
72 is->eps[0].port = cfg->port;
73 is->eps[0].tls = cfg->tls;
74 snprintf(is->eps[0].base_path, sizeof(is->eps[0].base_path), "%s",
75 (cfg->base_path != NULL) ? cfg->base_path : "");
76 is->n_eps = 1;
77
78 if (cfg->extra == NULL || cfg->n_extra <= 0) {
79 return;
80 }
81 int i, n = cfg->n_extra;
82
83 if (n > SD_HTTP_EP_MAX - 1) {
84 n = SD_HTTP_EP_MAX - 1;
85 }
86 for (i = 0; i < n; i++) {
87 const brix_sd_http_ep_cfg_t *ec = &cfg->extra[i];
88
89 if (ec->host == NULL || ec->host[0] == '\0' || ec->port <= 0
90 || ec->port > 65535)
91 {
92 continue;
93 }
94 snprintf(is->eps[is->n_eps].host, sizeof(is->eps[0].host), "%s",
95 ec->host);
96 is->eps[is->n_eps].port = ec->port;
97 is->eps[is->n_eps].tls = ec->tls;
98 snprintf(is->eps[is->n_eps].base_path,
99 sizeof(is->eps[0].base_path), "%s",
100 (ec->base_path != NULL) ? ec->base_path : "");
101 is->n_eps++;
102 }
103}

105/* sd_http_init_tctx — resolve the transport TLS context (trust anchor).
106 *
107 * WHAT: An explicit cfg->tctx wins (a caller injecting a custom transport owns
108 * its own context). Otherwise, when an operator CA path is configured,
109 * store a persistent copy and hand it to the curl transport as tctx so
110 * origin TLS verifies against that trust anchor (phase-70 https backend
111 * leg). No CA and no tctx ⇒ NULL ⇒ the transport's system bundle. */
112static void
113sd_http_init_tctx(sd_http_inst_state *is, const brix_sd_http_cfg_t *cfg)
114{
115 if (cfg->tctx != NULL) {
116 is->tctx = cfg->tctx;
117 } else if (cfg->ca_path != NULL && cfg->ca_path[0] != '\0') {
118 snprintf(is->ca_path, sizeof(is->ca_path), "%s", cfg->ca_path);
119 is->tctx = is->ca_path;
120 } else {
121 is->tctx = NULL;
122 }
123}

125brix_sd_instance_t *
126brix_sd_http_create(const brix_sd_http_cfg_t *cfg, ngx_log_t *log)
127{
128 brix_sd_instance_t *inst;
129 sd_http_inst_state *is;
130
131 if (cfg == NULL || cfg->host == NULL || cfg->host[0] == '\0'
132 || cfg->port <= 0 || cfg->port > 65535 || cfg->transport == NULL)
133 {
134 errno = EINVAL;
135 return NULL;
136 }
137 inst = calloc(1, sizeof(*inst));
138 is = calloc(1, sizeof(*is));
139 if (inst == NULL || is == NULL) {
140 free(inst);
141 free(is);
142 errno = ENOMEM;
143 return NULL;
144 }
145 sd_http_init_endpoints(is, cfg);
146 is->transport = cfg->transport;
147 is->failover_note = cfg->failover_note;
148 is->health_note = cfg->health_note;
149 sd_http_init_tctx(is, cfg);
150 is->timeout_ms = (cfg->timeout_ms > 0) ? cfg->timeout_ms
151 : BRIX_SD_HTTP_DEFAULT_TIMEOUT_MS;
152 /* Store only whether selection/failover logging is enabled, NOT a usable log
153 * pointer: this instance is built at postconfiguration and memoized for the
154 * worker's whole life, so a captured log dangles once fork replaces the
155 * config-phase cycle log (its fd is reused — a fill-thread notice would then
156 * land on a live client socket). The fill-thread log sites resolve the CURRENT
157 * process error log via sd_http_live_log() instead. */
158 is->log = log;
159 is->cur_ep = -1;
160 if (cfg->bearer_token != NULL && cfg->bearer_token[0] != '\0') {
161 snprintf(is->auth_hdr, sizeof(is->auth_hdr),
162 "Authorization: Bearer %s\r\n", cfg->bearer_token);
163 }
164
165 inst->driver = &brix_sd_http_driver;
166 inst->log = log;
167 inst->pool = NULL;
168 inst->state = is;
169 return inst;
170}
