10/*
11 * WHAT: kXR_Qckscan dispatch — parse wire payload (algo:path or bare path), resolve path, check auth/VO/token scope,
12 * then execute ckscan via async thread pool (if configured) or synchronous fallback. Returns checksum results as one
13 * "algo hex logical_path" line per file to the client.
14 *
15 * WHY: Qckscan requests arrive with a wire payload that may optionally prefix the path with an algorithm specifier
16 * ("crc32c:/data/atlas/..."). This dispatch layer extracts and validates the algo, resolves the path against the export root,
17 * checks authdb read permission + VO ACL + token scope, then routes to async (ngx_thread_pool_run) or sync execution.
18 * The async path allocates a thread task with all scan parameters; the done callback delivers results back on the client connection.
19 * Sync fallback runs when no thread pool is configured or the queue is full.
20 *
21 * HOW: brix_query_ckscan() validates payload presence, calls ckscan_select_payload() to extract algo + path from wire data (defaulting
22 * to adler32), extracts and resolves path, checks authdb/VO/token scope, then either posts a thread task or runs ckscan_sync().
23 * ckscan_select_payload() scans for ':' or ' ' before the first '/' to detect algo prefix; validates via checksum_parse + algorithm_supported.
24 * ckscan_sync() mirrors the async thread logic on the event loop: stat → open → checksum for files, walk tree for directories.
25 */

31/* ckscan_algorithm_supported — true only for adler32 or crc32c (xrdadler32
32 * compatibility), via brix_checksum_parse; gates the extracted algo prefix. */
33
34static ngx_flag_t
35brix_ckscan_algorithm_supported(const char *algo)
36{
37 brix_checksum_alg_t alg;
38
39 if (brix_checksum_parse(algo, strlen(algo), &alg, NULL, 0) != NGX_OK) {
40 return 0;
41 }
42
43 return alg == BRIX_CHECKSUM_ADLER32
44 || alg == BRIX_CHECKSUM_CRC32C
45 || alg == BRIX_CHECKSUM_CRC64
46 || alg == BRIX_CHECKSUM_CRC64NVME;
47}

62/* ckscan_select_payload — parse the request payload into algo + scan path: bytes
63 * before the first '/' are checked for a ':'/' ' delimiter ("algo:path"), default
64 * adler32, validated by checksum_parse + algorithm_supported. NGX_DONE on an
65 * invalid/unsupported algo, else NGX_OK with path/algo filled. */
66
67static ngx_int_t
68brix_ckscan_select_payload(brix_ctx_t *ctx, ngx_connection_t *c,
69 const u_char **path_payload, size_t *path_payload_len,
70 char *algo, size_t algo_sz)
71{
72 const u_char *payload = ctx->recv.payload;
73 size_t payload_len = (size_t) ctx->recv.cur_dlen;
74 size_t wire_len;
75 size_t i;
76
77 ngx_cpystrn((u_char *) algo, (u_char *) "adler32", algo_sz);
78 *path_payload = payload;
79 *path_payload_len = payload_len;
80
81 wire_len = strnlen((const char *) payload, payload_len);
82 for (i = 0; i < wire_len && payload[i] != '/'; i++) {
83 if (payload[i] == ':' || payload[i] == ' ') {
84 brix_checksum_alg_t alg;
85
86 if (i == 0 || i >= algo_sz || i + 1 >= payload_len) {
87 return brix_ckscan_send_error(ctx, c, kXR_ArgInvalid,
88 "invalid checksum algorithm");
89 }
90
91 if (brix_checksum_parse((const char *) payload, i, &alg, algo,
92 algo_sz) != NGX_OK
93 || !brix_ckscan_algorithm_supported(algo))
94 {
95 return brix_ckscan_send_error(ctx, c, kXR_ArgInvalid,
96 "unknown checksum algorithm");
97 }
98
99 *path_payload = payload + i + 1;
100 *path_payload_len = payload_len - (i + 1);
101 return NGX_OK;
102 }
103 }
104
105 return NGX_OK;
106}

110/* ckscan_sync — the event-loop fallback (no thread pool, or queue full) mirroring
111 * the async worker: stat the target, checksum a regular file via a confined fd or
112 * walk a directory (ckscan_walk, depth/file limits) into a growable buffer, then
113 * send ok+checksum or error. */
114
115static ngx_int_t
116brix_ckscan_sync(brix_ctx_t *ctx, ngx_connection_t *c,
117 ngx_stream_brix_srv_conf_t *conf, int rootfd,
118 const char *logical, const char *algo)
119{
120 u_char *buf;
121 size_t cap = BRIX_CKSCAN_INIT_CAP;
122 size_t used = 0;
123 uint16_t err_code = 0;
124 char err_msg[128] = "";
125 ngx_int_t rc;
126
127 buf = ngx_alloc(cap, c->log);
128 if (buf == NULL) {
129 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSCAN);
130 return brix_send_error(ctx, c, kXR_NoMemory, "out of memory");
131 }
132
133 /* The whole stat/open/checksum/walk now lives in the confined VFS walk
134 * (brix_ckscan_run → brix_vfs_walk); this layer only frames the result. */
135 if (brix_ckscan_run(c->log, rootfd, logical, algo, &buf, &cap, &used,
136 conf->ckscan_max_depth, conf->ckscan_max_files,
137 &err_code, err_msg, sizeof(err_msg)) != NGX_OK)
138 {
139 ngx_free(buf);
140 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSCAN);
141 return brix_send_error(ctx, c, err_code, err_msg);
142 }
143
144 buf[used] = '\0';
145 BRIX_OP_OK(ctx, BRIX_OP_QUERY_CKSCAN);
146 brix_log_access(ctx, c, "QUERY", logical, "ckscan", 1, 0, NULL, 0);
147 rc = brix_send_ok(ctx, c, buf, (uint32_t) (used + 1));
148 ngx_free(buf);
149 return rc;
150}

152/* brix_query_ckscan — kXR_Qckscan entry point: parse algo+path (select_payload),
153 * resolve against the export root, check authdb read + VO ACL + token scope, then
154 * route to the async thread pool (NGX_OK posted) or the sync fallback
155 * (NGX_DONE/NGX_ERROR). */
156
157ngx_int_t
158brix_query_ckscan(brix_ctx_t *ctx, ngx_connection_t *c,
159 ngx_stream_brix_srv_conf_t *conf)
160{
161 char full_path[PATH_MAX];
162 char pathbuf[BRIX_MAX_PATH + 1];
163 char algo[32];
164 const u_char *path_payload;
165 size_t path_payload_len;
166 ngx_int_t rc;
167
168 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
169 BRIX_RETURN_ERR(ctx, c, BRIX_OP_QUERY_CKSCAN, "QUERY",
170 "-", "ckscan", kXR_ArgMissing, "no path given");
171 }
172
173 rc = brix_ckscan_select_payload(ctx, c, &path_payload, &path_payload_len,
174 algo, sizeof(algo));
175 if (rc == NGX_DONE) {
176 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSCAN);
177 return NGX_OK;
178 }
179 if (rc != NGX_OK) {
180 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSCAN);
181 return rc;
182 }
183
184 if (!brix_extract_path(c->log, path_payload, path_payload_len,
185 pathbuf, sizeof(pathbuf), 1)) {
186 BRIX_RETURN_ERR(ctx, c, BRIX_OP_QUERY_CKSCAN, "QUERY",
187 "-", "ckscan", kXR_ArgInvalid, "invalid path payload");
188 }
189
190 brix_beneath_full_path(conf->common.root_canon, pathbuf,
191 full_path, sizeof(full_path));
192
193 if (brix_auth_gate(ctx, c, BRIX_OP_QUERY_CKSCAN, "QUERY",
194 pathbuf, full_path, conf,
195 BRIX_AUTH_READ, 0) != NGX_OK) {
196 return ctx->write_rc;
197 }
198
199 if (conf->common.thread_pool != NULL) {
200 brix_ckscan_aio_t *t;
201 ngx_thread_task_t *task;
202 ngx_flag_t posted;
203
204 task = ngx_thread_task_alloc(c->pool, sizeof(brix_ckscan_aio_t));
205 if (task == NULL) {
206 BRIX_OP_ERR(ctx, BRIX_OP_QUERY_CKSCAN);
207 return brix_send_error(ctx, c, kXR_NoMemory, "out of memory");
208 }
209
210 t = task->ctx;
211
212 t->ctx = ctx;
213 t->c = c;
214 t->conf = conf;
215 t->rootfd = conf->rootfd;
216 ngx_memcpy(t->streamid, ctx->recv.cur_streamid, 2);
217 ngx_cpystrn((u_char *) t->algo, (u_char *) algo, sizeof(t->algo));
218 ngx_cpystrn((u_char *) t->scan_logical, (u_char *) pathbuf,
219 sizeof(t->scan_logical));
220 t->max_depth = conf->ckscan_max_depth;
221 t->max_files = conf->ckscan_max_files;
222 t->resp = NULL;
223 t->resp_len = 0;
224 t->error_code = 0;
225
226 brix_task_bind(task, brix_ckscan_aio_thread, brix_ckscan_aio_done);
227 task->ctx = t;
228
229 if (brix_aio_post_task(ctx, c, conf->common.thread_pool, task,
230 "ckscan thread pool queue full, using sync",
231 &posted) != NGX_OK)
232 {
233 return NGX_ERROR;
234 }
235
236 if (posted) {
237 return NGX_OK;
238 }
239 }
240
241 return brix_ckscan_sync(ctx, c, conf, conf->rootfd, pathbuf, algo);
242}
