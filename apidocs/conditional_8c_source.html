1/*
2 * conditional.c — S3 conditional requests + response-header overrides (phase-43 W3).
3 *
4 * WHAT: Two GET/HEAD request modifiers that AWS S3 clients rely on:
5 * 1. Conditional preconditions — If-Match / If-None-Match / If-Modified-Since
6 * / If-Unmodified-Since, evaluated with S3 semantics (304 Not Modified,
7 * 412 PreconditionFailed).
8 * 2. response-* query overrides — response-content-type / -content-disposition
9 * / -content-encoding / -content-language / -cache-control / -expires, which
10 * let a (pre)signed GET dictate the response headers (browser "save-as"
11 * download flows).
12 *
13 * WHY: nginx's core not-modified filter already covers If-None-Match, If-Match
14 * and If-Unmodified-Since, but its If-Modified-Since uses `exact` semantics
15 * (304 only when the date equals mtime), whereas S3 requires `before`
16 * semantics (304 when the object has NOT been modified since the date). A
17 * single explicit evaluator here owns the full S3 contract and emits proper
18 * S3 XML for 412 (the core filter sends a bodyless 412). It front-runs the
19 * core filter — it never disagrees with it, it only adds the future-date 304
20 * case and the XML body.
21 *
22 * HOW: Precondition evaluation delegates to the shared RFC 9110 §13.2.2
23 * evaluator brix_http_eval_preconditions() (core/http/http_conditionals.c)
24 * — GET/HEAD in READ|TIME mode (If-None-Match match ⇒ 304, S3 `before`
25 * If-Modified-Since semantics), conditional PUT in ETag-only write mode
26 * (match ⇒ 412). This file owns only the S3 protocol edge: the XML 412
27 * body, the bare 304, the pre-body PUT existence probe, and the response-*
28 * overrides. Overrides are applied from the serve pre-header hook so they
29 * win over the headers set_file_headers computed; every override value is
30 * rejected if it carries a control byte (defeats response-splitting), so
31 * they are safe to honor on any request without an open header-injection
32 * risk.
33 */

56ngx_int_t
57s3_send_not_modified(ngx_http_request_t *r, const char *etag, time_t mtime)
58{
59 (void) s3_set_header(r, "ETag", etag);
60 r->headers_out.last_modified_time = mtime;
61
62 r->headers_out.status = NGX_HTTP_NOT_MODIFIED;
63 r->headers_out.content_length_n = 0;
64 /* A bare 304 carries no body; the not-modified filter must not re-touch it. */
65 r->disable_not_modified = 1;
66 ngx_http_send_header(r);
67 return ngx_http_send_special(r, NGX_HTTP_LAST);
68}

78/*
79 * s3_handle_conditional — evaluate preconditions and, when they short-circuit,
80 * send the 304/412 response. Returns:
81 * NGX_DECLINED — caller should proceed to serve the object (no precondition,
82 * or all preconditions passed);
83 * an HTTP rc — a 304/412 response was sent; caller returns it.
84 */
85ngx_int_t
86s3_handle_conditional(ngx_http_request_t *r, time_t mtime, off_t size)
87{
88 char etag[48];
89 ngx_int_t verdict;
90
91 if (r->headers_in.if_match == NULL
92 && r->headers_in.if_none_match == NULL
93 && r->headers_in.if_modified_since == NULL
94 && r->headers_in.if_unmodified_since == NULL)
95 {
96 return NGX_DECLINED; /* no conditional headers — common fast path */
97 }
98
99 brix_http_etag_str(etag, sizeof(etag), mtime, size, 0);
100 verdict = brix_http_eval_preconditions(r, 1 /* exists */, mtime, size,
101 0, BRIX_HTTP_COND_READ | BRIX_HTTP_COND_TIME
102 | BRIX_HTTP_COND_WEAK_EQUIV);
103
104 if (verdict == NGX_HTTP_NOT_MODIFIED) {
105 return s3_send_not_modified(r, etag, mtime);
106 }
107 if (verdict == NGX_HTTP_PRECONDITION_FAILED) {
108 return s3_send_precondition_failed(r);
109 }
110 return NGX_DECLINED;
111}

117/*
118 * s3_put_precondition — evaluate If-None-Match / If-Match on a PutObject before
119 * the body is read.
120 *
121 * WHAT: Implements S3 conditional writes:
122 * - If-None-Match: * → 412 if the object already exists (atomic-ish
123 * create-if-absent; the canonical concurrency-
124 * safe write).
125 * - If-None-Match: "<etag>" → 412 if the existing object carries that etag.
126 * - If-Match: * / "<etag>" → 412 unless the object exists (and matches).
127 * WHY: Lets clients implement compare-and-swap / create-once semantics without a
128 * separate HEAD round-trip; checked here (pre-body) so a doomed write never
129 * consumes the upload bandwidth.
130 * HOW: A single confined stat of the destination yields existence + mtime/size;
131 * the verdict comes from the shared evaluator in ETag-only write mode (the
132 * same engine as the GET path, minus READ|TIME: a match ⇒ 412, never 304,
133 * and the date headers are not part of the S3 conditional-write contract).
134 *
135 * NOTE (atomicity): the existence test and the later staged-rename commit are
136 * not a single atomic step, so two racing If-None-Match:* creates of the same
137 * key can both pass here (last-writer-wins). The window is the staged write
138 * duration; closing it fully needs renameat2(RENAME_NOREPLACE) plumbed through
139 * the confinement + impersonation rename path — a deliberate follow-up.
140 *
141 * Returns NGX_DECLINED to proceed with the write, or a 412 rc already sent.
142 */
143ngx_int_t
144s3_put_precondition(ngx_http_request_t *r, const char *root_canon,
145 const char *fs_path)
146{
147 ngx_table_elt_t *if_none = r->headers_in.if_none_match;
148 ngx_table_elt_t *if_match = r->headers_in.if_match;
149 int exists = 0;
150 brix_vfs_ctx_t vctx;
151 brix_vfs_stat_t vst;
152 ngx_http_s3_req_ctx_t *s3ctx;
153 ngx_http_s3_loc_conf_t *cf =
154 ngx_http_get_module_loc_conf(r, ngx_http_brix_s3_module);
155 int is_tls = 0;
156
157 if (if_none == NULL && if_match == NULL) {
158 return NGX_DECLINED; /* unconditional PUT — fast path */
159 }
160
161 /*
162 * Existence + synthetic-etag probe via the VFS metadata surface (OP_STAT,
163 * metered + access-logged), delegating to the same confined no-follow
164 * lstat underneath. An object is only "exists" when it is a regular file;
165 * a symlink at the key reports as non-regular and is treated as absent,
166 * matching the S3 object model (the lister never emits symlinks).
167 */
168 s3ctx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
169#if (NGX_HTTP_SSL)
170 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
171#endif
172 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_S3,
173 root_canon, cf->cache_root_canon, cf->common.allow_write, is_tls,
174 (s3ctx != NULL) ? s3ctx->identity : NULL, fs_path);
175
176 if (brix_vfs_stat(&vctx, &vst) == NGX_OK && vst.is_regular) {
177 exists = 1;
178 }
179
180 if (brix_http_eval_preconditions(r, exists,
181 exists ? vst.mtime : 0, exists ? vst.size : 0,
182 0, BRIX_HTTP_COND_WEAK_EQUIV) != NGX_OK)
183 {
184 return s3_send_precondition_failed(r);
185 }
186
187 return NGX_DECLINED;
188}

190/*
191 * s3_put_is_exclusive_create — true when a PutObject carries `If-None-Match: *`,
192 * i.e. the client asked for atomic create-if-absent. W6b uses this to commit
193 * with renameat2(RENAME_NOREPLACE) so two racing creates can't both win (the
194 * stat-based precondition above only narrows, it cannot make it atomic).
195 */
196int
197s3_put_is_exclusive_create(ngx_http_request_t *r)
198{
199 ngx_table_elt_t *if_none = r->headers_in.if_none_match;
200
201 return if_none != NULL
202 && if_none->value.len == 1
203 && if_none->value.data[0] == '*';
204}

210/* Copy a response-* override into a simple response header, rejecting control
211 * bytes (response-splitting guard). No-op when the param is absent. */
212static void
213s3_override_header(ngx_http_request_t *r, const char *param, const char *header)
214{
215 char val[1024];
216 size_t len;
217
218 /* query_get returns 1 on success and NUL-terminates val; the decoded
219 * length is strlen(val), NOT the return value. */
220 if (brix_http_query_get(r->args, param, val, sizeof(val),
221 S3_OVERRIDE_QUERY_FLAGS) <= 0)
222 {
223 return;
224 }
225 len = ngx_strlen(val);
226 if (len == 0 || brix_http_str_has_ctl((u_char *) val, len)) {
227 return;
228 }
229 (void) brix_http_set_header(r, header, val, NULL);
230}

232void
233s3_apply_response_overrides(ngx_http_request_t *r)
234{
235 char ct[256];
236 size_t len;
237
238 if (r->args.len == 0) {
239 return;
240 }
241
242 /* response-content-type overrides the computed Content-Type in-place. */
243 if (brix_http_query_get(r->args, "response-content-type", ct, sizeof(ct),
244 S3_OVERRIDE_QUERY_FLAGS) > 0
245 && (len = ngx_strlen(ct)) > 0
246 && !brix_http_str_has_ctl((u_char *) ct, len))
247 {
248 u_char *p = ngx_pnalloc(r->pool, len);
249 if (p != NULL) {
250 ngx_memcpy(p, ct, len);
251 r->headers_out.content_type.data = p;
252 r->headers_out.content_type.len = len;
253 r->headers_out.content_type_len = len;
254 r->headers_out.content_type_lowcase = NULL;
255 }
256 }
257
258 s3_override_header(r, "response-content-disposition", "Content-Disposition");
259 s3_override_header(r, "response-content-encoding", "Content-Encoding");
260 s3_override_header(r, "response-content-language", "Content-Language");
261 s3_override_header(r, "response-cache-control", "Cache-Control");
262 s3_override_header(r, "response-expires", "Expires");
263}
