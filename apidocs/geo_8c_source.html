1/* geo.c — uncached origin passthrough (geo API + T12 manifest stopgap).
2 *
3 * WHAT: forwards a classified request (with query string) to the location's
4 * configured http origin over the shared blocking libcurl transport
5 * and relays status + body. Never cached: the geo answer depends on
6 * the caller, and (until T12) manifests are mutable signed metadata.
7 * WHY: CVMFS clients call the geo API through their site proxy at mount
8 * time to order CVMFS_SERVER_URL; failure is non-fatal for the client
9 * but a correct answer improves Stratum-1 ordering.
10 * HOW: responses are tiny (a comma-separated index list; manifests are a
11 * few hundred bytes), so one bounded in-memory transport request on a
12 * thread-pool task is appropriate — the same aio posting idiom the
13 * tier fills use (brix_task_bind, r->main->count++, finalize in the
14 * done handler). The response buffer is pool-allocated BEFORE the
15 * post: the request pool is never touched off the event loop.
16 */
17#include "cvmfs.h"
18#include "core/aio/aio.h"

20#include "fs/vfs/vfs_backend_registry.h"

22#define CVMFS_PT_RESP_MAX (64 * 1024) /* geo lists + manifests are tiny */
23#define CVMFS_PT_TIMEOUT_MS 5000

25typedef struct {
26 ngx_http_request_t *r;
27 char path[2048]; /* origin path incl. query */
28 char host[256];
29 int port;
30 int tls;
31 int status; /* HTTP status, or -1 transport fail */
32 u_char *body; /* pool-allocated on the event loop */

36/* thread-pool side: one blocking GET over the shared transport */
37static void
38cvmfs_pt_thread(void *data, ngx_log_t *log)
39{
40 cvmfs_pt_task_t *t = data;
41 const brix_s3_transport_t *tr = &brix_s3_origin_curl_transport;
42 brix_s3_resp_t resp;
43 const void *body;
44 size_t blen = 0;
45 char errbuf[256];
46
47 (void) log;
48 t->status = -1;
49 if (tr->request(NULL, t->host, t->port, t->tls, "GET", t->path, NULL,
50 NULL, 0, CVMFS_PT_TIMEOUT_MS, &resp,
51 errbuf, sizeof(errbuf)) != 0)
52 {
53 return;
54 }
55 t->status = resp.status;
56 body = tr->resp_body(&resp, &blen);
57 if (body != NULL && blen > 0 && blen <= CVMFS_PT_RESP_MAX) {
58 ngx_memcpy(t->body, body, blen);
59 t->body_len = blen;
60 }
61 tr->resp_free(&resp);
62}

64/* event-loop side: emit the relayed response */
65static void
66cvmfs_pt_done(ngx_event_t *ev)
67{
68 ngx_thread_task_t *task = ev->data;
69 cvmfs_pt_task_t *t = task->ctx;
70 ngx_http_request_t *r = t->r;
71 ngx_connection_t *c = r->connection;
72 ngx_buf_t *b;
73 ngx_chain_t out;
74 ngx_int_t rc;
75
76 if (t->status < 100) {
77 ngx_http_finalize_request(r, NGX_HTTP_BAD_GATEWAY);
78 ngx_http_run_posted_requests(c);
79 return;
80 }
81
82 r->headers_out.status = (ngx_uint_t) t->status;
83 r->headers_out.content_length_n = (off_t) t->body_len;
84 rc = ngx_http_send_header(r);
85 if (rc != NGX_OK || r->header_only || t->body_len == 0) {
86 ngx_http_finalize_request(r,
87 (rc == NGX_OK) ? ngx_http_send_special(r, NGX_HTTP_LAST) : rc);
88 ngx_http_run_posted_requests(c);
89 return;
90 }
91
92 b = ngx_pcalloc(r->pool, sizeof(*b));
93 if (b == NULL) {
94 ngx_http_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
95 ngx_http_run_posted_requests(c);
96 return;
97 }
98 b->pos = b->start = t->body;
99 b->last = b->end = t->body + t->body_len;
100 b->memory = 1;
101 b->last_buf = 1;
102 out.buf = b;
103 out.next = NULL;
104 ngx_http_finalize_request(r, ngx_http_output_filter(r, &out));
105 ngx_http_run_posted_requests(c);
106}

108/* Route a classified request to the origin, uncached. Returns NGX_DONE
109 * (async completion via cvmfs_pt_done) or an HTTP error status. */
110ngx_int_t
111brix_cvmfs_geo_passthrough(ngx_http_request_t *r,
112 ngx_http_brix_cvmfs_loc_conf_t *lcf)
113{
114 ngx_http_brix_cvmfs_ctx_t *ctx =
115 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
116 cvmfs_pt_task_t *t;
117 const char *host, *base, *root;
118 int port, tls, n;
119 ngx_thread_task_t *task;
120 ngx_thread_pool_t *pool;
121
122 /* proxy mode (T14) resolves against the per-upstream registry root */
123 root = (ctx != NULL && ctx->up_root != NULL) ? ctx->up_root
124 : lcf->common.root_canon;
125 if (brix_vfs_backend_http_endpoint(root, &host, &port, &tls, &base)
126 != 0)
127 {
128 /* passthrough requires an http(s) backend */
129 return NGX_HTTP_NOT_IMPLEMENTED;
130 }
131
132 pool = lcf->common.thread_pool;
133 if (pool == NULL) {
134 static ngx_str_t default_name = ngx_string("default");
135 ngx_str_t *pname = lcf->common.thread_pool_name.len > 0
136 ? &lcf->common.thread_pool_name
137 : &default_name;
138
139 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
140 if (pool != NULL) {
141 lcf->common.thread_pool = pool;
142 }
143 }
144 if (pool == NULL) {
145 return NGX_HTTP_SERVICE_UNAVAILABLE; /* no pool: cannot relay */
146 }
147
148 task = ngx_thread_task_alloc(r->pool, sizeof(cvmfs_pt_task_t));
149 if (task == NULL) {
150 return NGX_HTTP_INTERNAL_SERVER_ERROR;
151 }
152 t = task->ctx;
153 t->r = r;
154 t->port = port;
155 t->tls = tls;
156 (void) ngx_cpystrn((u_char *) t->host, (u_char *) host, sizeof(t->host));
157 t->body = ngx_palloc(r->pool, CVMFS_PT_RESP_MAX);
158 if (t->body == NULL) {
159 return NGX_HTTP_INTERNAL_SERVER_ERROR;
160 }
161
162 /* origin path = <base> + <uri> [+ "?" + args]; reject over-long paths
163 * instead of truncating (a truncated geo path returns a wrong answer). */
164 if (r->args.len > 0) {
165 n = snprintf(t->path, sizeof(t->path), "%s%.*s?%.*s", base,
166 (int) r->uri.len, r->uri.data,
167 (int) r->args.len, r->args.data);
168 } else {
169 n = snprintf(t->path, sizeof(t->path), "%s%.*s", base,
170 (int) r->uri.len, r->uri.data);
171 }
172 if (n < 0 || (size_t) n >= sizeof(t->path)) {
173 return NGX_HTTP_REQUEST_URI_TOO_LARGE;
174 }
175
176 brix_task_bind(task, cvmfs_pt_thread, cvmfs_pt_done);
177 task->event.log = r->connection->log;
178
179 if (ngx_thread_task_post(pool, task) != NGX_OK) {
180 return NGX_HTTP_INTERNAL_SERVER_ERROR;
181 }
182 r->main->count++; /* request survives until cvmfs_pt_done */
183 return NGX_DONE;
184}
