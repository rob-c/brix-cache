151/*
152 * Content handler for the dashboard page route.
153 *
154 * WHAT: gate on the session cookie, then serve the embedded HTML asset above.
155 * WHY: the UI is behind the same cookie auth as the JSON API, so an
156 * unauthenticated GET must be bounced to the login page rather than
157 * leaking the dashboard shell.
158 * HOW: flow is (1) auth check -> on failure emit a 302 to
159 * <cookie_path>/login and return; (2) reject non-GET/HEAD; (3) hand
160 * back the static literal as a single memory-backed buffer.
161 * Returns an NGX_HTTP_* status on error, or the output-filter result.
162 */
163ngx_int_t
164ngx_http_brix_dashboard_page_handler(ngx_http_request_t *r)
165{
166 ngx_http_brix_dashboard_loc_conf_t *conf;
167 ngx_buf_t *b;
168 ngx_chain_t out;
169 size_t html_len;
170 ngx_int_t rc;
171
172 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
173
174 /*
175 * SECURITY: cookie/session gate. Any non-NGX_OK (no/expired/invalid cookie)
176 * means "not logged in". We then either (a) serve the page shell anyway when
177 * brix_dashboard_anonymous is on - its JS will receive PII/secret-redacted
178 * JSON and render the anonymous banner - or (b) redirect to login. The shell
179 * itself contains no data, so serving it anonymously leaks nothing; all
180 * redaction is enforced by the JSON API the JS fetches.
181 */
182 rc = ngx_http_brix_dashboard_check_auth(r, conf, conf->anonymous);
183 if (rc != NGX_OK && !conf->anonymous) {
184 r->headers_out.location = ngx_list_push(&r->headers_out.headers);
185 if (r->headers_out.location == NULL) {
186 return NGX_HTTP_INTERNAL_SERVER_ERROR;
187 }
188 r->headers_out.location->hash = 1;
189 ngx_str_set(&r->headers_out.location->key, "Location");
190 {
191 /*
192 * Build the redirect target "<cookie_path>/login" in the request
193 * pool. Byte math: sizeof("/login") == 7 (includes the NUL), so
194 * the alloc reserves room for cookie_path + "/login" + trailing
195 * NUL. The memcpy of "/login" copies all 7 bytes including the
196 * NUL terminator (defensive null-termination of the buffer).
197 * The emitted ngx_str_t .len, however, is cookie_path.len + 7 - 1,
198 * i.e. excludes that NUL - nginx header values are length-counted,
199 * not C strings, so the NUL must not be on the wire.
200 */
201 u_char *loc = ngx_pnalloc(r->pool, conf->cookie_path.len
202 + sizeof("/login"));
203 if (loc == NULL) {
204 return NGX_HTTP_INTERNAL_SERVER_ERROR;
205 }
206 ngx_memcpy(loc, conf->cookie_path.data, conf->cookie_path.len);
207 ngx_memcpy(loc + conf->cookie_path.len, "/login",
208 sizeof("/login"));
209 r->headers_out.location->value.data = loc;
210 r->headers_out.location->value.len = conf->cookie_path.len
211 + sizeof("/login") - 1;
212 }
213 return NGX_HTTP_MOVED_TEMPORARILY;
214 }
215
216 /* Read-only asset: only GET (full body) and HEAD (headers only). */
217 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
218 return NGX_HTTP_NOT_ALLOWED;
219 }
220
221 /* -1 drops the literal's terminating NUL: send the HTML, not the NUL. */
222 html_len = sizeof(ngx_brix_dashboard_html) - 1;
223 BRIX_PCALLOC_OR_RETURN(b, r->pool, sizeof(*b), NGX_HTTP_INTERNAL_SERVER_ERROR);
224
225 /*
226 * Point the buffer straight at the const .rodata literal - no copy. Safe
227 * because the literal outlives the request and is never mutated. b->memory
228 * marks it read-only/in-memory (not file-backed, so no sendfile); b->last_buf
229 * makes this the sole and final buf of the response.
230 */
231 b->pos = (u_char *) ngx_brix_dashboard_html;
232 b->last = b->pos + html_len;
233 b->memory = 1;
234 b->last_buf = 1;
235
236 r->headers_out.status = NGX_HTTP_OK;
237 r->headers_out.content_length_n = (off_t) html_len;
238 r->headers_out.content_type =
239 (ngx_str_t) ngx_string("text/html; charset=utf-8");
240 r->headers_out.content_type_len = r->headers_out.content_type.len;
241
242 rc = ngx_http_send_header(r);
243 /*
244 * Bail before emitting a body when: send_header failed (NGX_ERROR),
245 * returned a special/error code (rc > NGX_OK), or this is a HEAD request
246 * (r->header_only) where the body must be suppressed. Otherwise fall
247 * through and write the single-buffer chain.
248 */
249 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
250 return rc;
251 }
252
253 out.buf = b;
254 out.next = NULL;
255 return ngx_http_output_filter(r, &out);
256}
