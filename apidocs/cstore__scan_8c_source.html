25/* True when `name` is a cstore metadata sidecar rather than a cached object. */
26static int
27cstore_is_sidecar(const char *name)
28{
29 size_t n = strlen(name);
30
31 return (n >= 6 && strcmp(name + n - 6, ".cinfo") == 0)
32 || (n >= 5 && strcmp(name + n - 5, ".meta") == 0)
33 || (n >= 9 && strcmp(name + n - 9, ".xrdcinfo") == 0);
34}

36/*
37 * WHAT: Decide whether a readdir entry name should be skipped by the scan.
38 * WHY: The scan must never surface the "." / ".." pseudo-entries, empty
39 * names, or metadata sidecars as cached objects.
40 * HOW: Pure name predicate — true when the entry is not a cached object.
41 */
42static int
43cstore_scan_skip(const char *name)
44{
45 return name[0] == '\0' || strcmp(name, ".") == 0
46 || strcmp(name, "..") == 0 || cstore_is_sidecar(name);
47}

49/*
50 * WHAT: Join `dirkey` and a child `name` into out[cap] (the child's store key).
51 * WHY: Nested scan keys are built the same everywhere: exactly one '/' between
52 * the parent key and the entry name, regardless of a trailing slash.
53 * HOW: Snprintf with a conditional separator; returns 0 on success, -1 when
54 * the result is empty or would overflow (caller skips the entry).
55 */
56static int
57cstore_scan_child_key(const char *dirkey, size_t dklen, const char *name,
58 char *out, size_t cap)
59{
60 int n = snprintf(out, cap, "%s%s%s", dirkey,
61 (dklen > 0 && dirkey[dklen - 1] == '/') ? "" : "/", name);
62
63 return (n > 0 && (size_t) n < cap) ? 0 : -1;
64}

70/*
71 * WHAT: Handle one already-keyed child: recurse into dirs, visit regulars.
72 * WHY: Keeps the scan loop flat — the per-entry stat + dispatch (recurse vs
73 * visit) is a single cohesive step with its own early-returns.
74 * HOW: Stat the child through the store driver; a missing stat slot or a
75 * failed stat is a silent skip (NGX_OK). Directories recurse; regular
76 * objects are handed to the visitor with their cinfo when present (NULL
77 * for orphan/partial) and always the store stat, so an eviction policy
78 * sees the same set a raw driver scan would.
79 */
80static ngx_int_t
81cstore_scan_child(brix_cstore_t *cs, const char *childkey,
82 brix_cstore_visit_fn visit, void *ctx)
83{
84 brix_sd_stat_t stx;
85
86 if (cs->store->driver->stat == NULL
87 || cs->store->driver->stat(cs->store, childkey, &stx) != NGX_OK)
88 {
89 return NGX_OK; /* unstattable: skip, keep going */
90 }
91 if (stx.is_dir) {
92 return cstore_scan_dir(cs, childkey, visit, ctx);
93 }
94 if (stx.is_reg) {
95 brix_cache_cinfo_t ci;
96 int loaded = (brix_cstore_cinfo_load(cs, childkey, &ci) == NGX_OK);
97
98 return visit(childkey, loaded ? &ci : NULL, &stx, ctx);
99 }
100 return NGX_OK; /* neither dir nor regular: skip */
101}

103/*
104 * WHAT: Recursively visit cached objects beneath `dirkey` on the store.
105 * WHY: Sole namespace-walk for scan/evict; every cached object under the root
106 * must be presented to the visitor exactly once.
107 * HOW: Opendir (a missing dir is not an error), then for each non-skipped
108 * entry build its child key and dispatch it via cstore_scan_child; stop
109 * early when the visitor or a subdir returns non-OK.
110 */
111static ngx_int_t
112cstore_scan_dir(brix_cstore_t *cs, const char *dirkey,
113 brix_cstore_visit_fn visit, void *ctx)
114{
115 brix_sd_dir_t *d;
116 brix_sd_dirent_t de;
117 ngx_int_t rc = NGX_OK;
118 int err = 0;
119 size_t dklen = strlen(dirkey);
120
121 d = cs->store->driver->opendir(cs->store, dirkey, &err);
122 if (d == NULL) {
123 return (err == ENOENT) ? NGX_OK : NGX_ERROR;
124 }
125
126 while (cs->store->driver->readdir(d, &de) == NGX_OK) {
127 char childkey[PATH_MAX];
128
129 if (cstore_scan_skip(de.name)) {
130 continue;
131 }
132 if (cstore_scan_child_key(dirkey, dklen, de.name, childkey,
133 sizeof(childkey)) != 0)
134 {
135 continue;
136 }
137 rc = cstore_scan_child(cs, childkey, visit, ctx);
138 if (rc != NGX_OK) {
139 break; /* visitor / subdir stopped early */
140 }
141 }
142
143 cs->store->driver->closedir(d);
144 return rc;
145}
