8/*
9 * kXR_protocol - negotiate protocol version, auth protocols, and TLS support.
10 *
11 * This is the first XRootD request after the raw handshake. It does not log a
12 * user in; it only advertises server capabilities and, when configured, arms
13 * the native TLS upgrade for the following login/auth flow.
14 */
15
16/* WHAT: decoded client request + the server-side auth/TLS wants derived from it.
17 * WHY: the negotiation is a flag matrix over (client capability byte, configured
18 * auth mode, TLS availability); collecting the decoded flags in one struct
19 * lets each helper take a single explicit parameter instead of ten ints.
20 * HOW: filled once by protocol_parse_client_req(), then read-only everywhere. */
21typedef struct {
22 u_char client_flags; /* fifth byte of the kXR_protocol body */
23 int gsi; /* offer "gsi " SecurityProtocol entry */
24 int token; /* offer "ztn " SecurityProtocol entry */
25 int sss; /* offer "sss " SecurityProtocol entry */
26 int unx; /* offer "unix" SecurityProtocol entry */
27 int krb5; /* offer "krb5" SecurityProtocol entry */
28 int host; /* offer "host" SecurityProtocol entry */
29 int pwd; /* offer "pwd " SecurityProtocol entry */
30 int client_wants_tls; /* client set kXR_wantTLS (TLS is mandatory) */
31 int offer_tls; /* server can and will offer the TLS upgrade */
32} protocol_want_t;
33
34/* WHAT: decode the kXR_protocol request into a protocol_want_t.
35 * WHY: isolates the request-byte decode and conf->auth fan-out so the handler
36 * body only deals with negotiation and response assembly.
37 * HOW: reads the client capability byte from the received body, maps the
38 * configured auth mode onto the per-protocol offer flags, and computes
39 * the two TLS decisions from the client flags + listener TLS config. */
40static void
41protocol_parse_client_req(brix_ctx_t *ctx,
42 ngx_stream_brix_srv_conf_t *conf, protocol_want_t *want)
43{
44 /* kXR_protocol packs client capability flags into the fifth byte of body[]. */
45 want->client_flags = ctx->recv.cur_body[4];
46 want->gsi = (conf->auth == BRIX_AUTH_GSI
47 || conf->auth == BRIX_AUTH_BOTH);
48 want->token = (conf->auth == BRIX_AUTH_TOKEN
49 || conf->auth == BRIX_AUTH_BOTH);
50 want->sss = (conf->auth == BRIX_AUTH_SSS);
51 want->unx = (conf->auth == BRIX_AUTH_UNIX);
52 want->krb5 = (conf->auth == BRIX_AUTH_KRB5);
53 want->host = (conf->auth == BRIX_AUTH_HOST);
54 want->pwd = (conf->auth == BRIX_AUTH_PWD);
55
56 /* kXR_wantTLS: client requires TLS; kXR_ableTLS: client is TLS-capable. */
57 want->client_wants_tls = (want->client_flags & kXR_wantTLS) ? 1 : 0;
58 want->offer_tls = (conf->tls && conf->tls_ctx != NULL
59 && (want->client_flags & (kXR_ableTLS | kXR_wantTLS)));
60}
61
62/* WHAT: number of SecurityProtocol entries the SecurityInfo trailer will carry.
63 * WHY: needed twice (body-length computation and trailer header) — computing
64 * it in one place keeps the two in lockstep.
65 * HOW: pure count over the offer flags. */
66static int
67protocol_sec_count(const protocol_want_t *want)
68{
69 return (want->gsi ? 1 : 0) + (want->token ? 1 : 0)
70 + (want->sss ? 1 : 0) + (want->unx ? 1 : 0)
71 + (want->krb5 ? 1 : 0) + (want->host ? 1 : 0)
72 + (want->pwd ? 1 : 0);
73}
74
75/* WHAT: the server-role capability bits (manager/supervisor/redirector/meta/
76 * proxy/cache) for the ServerProtocolBody flags word.
77 * WHY: splits the role half of the capability bit-ladder out of the
78 * negotiation so each half stays a small pure expression.
79 * HOW: pure mapping from the merged server conf; no side effects. */
80static uint32_t
81protocol_role_flags(const ngx_stream_brix_srv_conf_t *conf)
82{
83 return ((conf->manager_map || conf->manager_mode) ? kXR_isManager : 0)
84 | (conf->caps.supervisor
85 ? (kXR_isManager | kXR_attrSuper) : 0)
86 | ((conf->caps.virtual_redirector
87 || (conf->manager_map != NULL && conf->cms.addr == NULL))
88 ? (kXR_isManager | kXR_attrVirtRdr) : 0)
89 | (conf->caps.metadata_only ? kXR_attrMeta : 0)
90 | ((conf->proxy.enable > 0 || conf->proxy.upstreams != NULL)
91 ? kXR_attrProxy : 0)
92 | ((conf->cache_root.len > 0 || conf->cache_origin_host.len > 0)
93 ? kXR_attrCache : 0);
94}
95
96/* WHAT: the full ServerProtocolBody flags word (host-order).
97 * WHY: the capability advertisement is the heart of kXR_protocol; keeping it
98 * one pure expression makes the negotiated bit-set reviewable at a glance.
99 * HOW: fixed server bits | role bits | per-conf feature bits | TLS bits. */
100static uint32_t
101protocol_negotiate_flags(const ngx_stream_brix_srv_conf_t *conf,
102 const protocol_want_t *want)
103{
104 return kXR_isServer
105 | kXR_suppgrw /* pgread/pgwrite always implemented */
106 | kXR_supposc /* POSC always implemented */
107 | protocol_role_flags(conf)
108 | (conf->caps.collapse_redir ? kXR_collapseRedir : 0)
109 | (conf->caps.recover_writes ? kXR_recoverWrts : 0)
110 | (want->offer_tls ? (kXR_haveTLS | kXR_gotoTLS | kXR_tlsLogin) : 0);
111}

113/* WHAT: write one 8-byte SecurityProtocol entry (4-char tag + 4 zero bytes).
114 * WHY: the trailer emits up to seven identically-shaped entries; one writer
115 * replaces seven copies of the byte layout.
116 * HOW: copies the fixed 4-byte tag, zeroes the reserved tail, returns the
117 * advanced cursor so callers chain entries linearly. */
118static u_char *
119protocol_sec_entry_write(u_char *pe, const char tag[4])
120{
121 pe[0] = (u_char) tag[0]; pe[1] = (u_char) tag[1];
122 pe[2] = (u_char) tag[2]; pe[3] = (u_char) tag[3];
123 pe[4] = 0; pe[5] = 0; pe[6] = 0; pe[7] = 0;
124 return pe + 8;
125}
126
127/* WHAT: write the SecurityInfo trailer (header + protocol entries + signing
128 * requirements) at si.
129 * WHY: the trailer is only present when the client set kXR_secreqs; hoisting
130 * it keeps the handler's response assembly linear.
131 * HOW: emits the 4-byte SecurityInfo header, one entry per offered auth
132 * protocol in the fixed wire order (sss, unix, krb5, host, pwd, ztn,
133 * gsi), then the 6-byte ServerResponseReqs_Protocol signing block.
134 * The caller sized the buffer with protocol_sec_count(). */
135static void
136protocol_write_sec_trailer(const ngx_stream_brix_srv_conf_t *conf,
137 const protocol_want_t *want, u_char *si)
138{
139 int sec_count = protocol_sec_count(want);
140 u_char *pe;
141
142 /*
143 * SecurityInfo header:
144 * byte 1 advertises whether security is required,
145 * byte 2 is the number of following protocol entries.
146 * bytes 0 and 3 are left zero because this implementation does not use
147 * any of the optional legacy fields encoded there.
148 */
149 si[0] = 0;
150 si[1] = sec_count > 0 ? 0x01 : 0x00;
151 si[2] = (u_char) sec_count;
152 si[3] = 0;
153
154 pe = si + 4;
155 if (want->sss) {
156 pe = protocol_sec_entry_write(pe, "sss ");
157 }
158 if (want->unx) {
159 pe = protocol_sec_entry_write(pe, "unix");
160 }
161 if (want->krb5) {
162 pe = protocol_sec_entry_write(pe, "krb5");
163 }
164 if (want->host) {
165 pe = protocol_sec_entry_write(pe, "host");
166 }
167 if (want->pwd) {
168 pe = protocol_sec_entry_write(pe, "pwd ");
169 }
170 if (want->token) {
171 pe = protocol_sec_entry_write(pe, "ztn ");
172 }
173 if (want->gsi) {
174 pe = protocol_sec_entry_write(pe, "gsi ");
175 }
176
177 /* ServerResponseReqs_Protocol — signing level requirements.
178 * Always present when kXR_secreqs is set; seclvl=0 means none. */
179 pe[0] = 'S'; /* theTag */
180 pe[1] = 0; /* rsvd */
181 pe[2] = 0; /* secver (kXR_secver_0) */
182 pe[3] = (conf->security_level >= 4) ? kXR_secOData : 0; /* secopt */
183 pe[4] = (u_char) conf->security_level; /* seclvl */
184 pe[5] = 0; /* secvsz=0: no secvec */
185}

188/* WHAT: short auth-mode label for the kXR_protocol debug log line.
189 * WHY: keeps the nine-way ternary chain out of the handler's log call.
190 * Debug-build only — its sole caller is ngx_log_debug4, which compiles
191 * out without NGX_DEBUG and would leave this unused under -Werror.
192 * HOW: pure priority chain over the offer flags — identical ordering to the
193 * historical inline expression so the logged string never changes. */
194static const char *
195protocol_auth_label(const protocol_want_t *want)
196{
197 return want->sss ? "sss" :
198 want->unx ? "unix" :
199 want->krb5 ? "krb5" :
200 want->host ? "host" :
201 want->pwd ? "pwd" :
202 want->gsi && want->token ? "both" :
203 want->gsi ? "gsi" :

208/* Handle kXR_protocol — negotiate server capabilities by building the fixed
209 * 8-byte ServerProtocolBody response (protocol version, flags, and the security
210 * requirement advertised to the client). */
211ngx_int_t
212brix_handle_protocol(brix_ctx_t *ctx, ngx_connection_t *c,
213 ngx_stream_brix_srv_conf_t *conf)
214{
215 ServerProtocolBody body;
216 u_char *buf;
217 size_t bodylen, total;
218 protocol_want_t want;
219
220 protocol_parse_client_req(ctx, conf, &want);
221
222 /* Reject if the client demands TLS but this listener has none configured. */
223 if (want.client_wants_tls && (!conf->tls || conf->tls_ctx == NULL)) {
224 return brix_send_error(ctx, c, kXR_TLSRequired,
225 "TLS required by client but not configured on this server");
226 }
227
228 /*
229 * Base kXR_protocol reply is the fixed 8-byte ServerProtocolBody.
230 * If the client advertised security negotiation support, append the small
231 * SecurityInfo trailer describing which auth protocols we offer.
232 */
233 bodylen = sizeof(body);
234 if (want.client_flags & kXR_secreqs) {
235 bodylen += 4; /* SecurityInfo header */
236 bodylen += (size_t) protocol_sec_count(&want) * 8; /* 8 bytes per SecurityProtocol entry */
237 bodylen += 6; /* ServerResponseReqs_Protocol */
238 }
239
240 total = XRD_RESPONSE_HDR_LEN + bodylen;
241 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
242
243 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
244 (uint32_t) bodylen,
245 (ServerResponseHdr *) buf);
246
247 body.pval = htonl(kXR_PROTOCOLVERSION);
248 body.flags = htonl(protocol_negotiate_flags(conf, &want));
249
250 /* Fixed 8-byte prefix every protocol reply starts with after the response header. */
251 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body, sizeof(body));
252
253 if (want.client_flags & kXR_secreqs) {
254 protocol_write_sec_trailer(conf, &want,
255 buf + XRD_RESPONSE_HDR_LEN + sizeof(body));
256 }
257
258 ngx_log_debug4(NGX_LOG_DEBUG_STREAM, c->log, 0,
259 "brix: kXR_protocol ok (client_flags=0x%02xd "
260 "bodylen=%uz auth=%s tls=%d)",
261 (int) want.client_flags, bodylen,
262 protocol_auth_label(&want),
263 want.offer_tls);
264
265 if (want.offer_tls) {
266 ctx->tls_pending = 1;
267 }
268
269 return brix_queue_response(ctx, c, buf, total);
270}
