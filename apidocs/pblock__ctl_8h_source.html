1/*
2 * pblock_ctl.h — Phase-83 lab control plane for the pblock storage driver.
3 *
4 * WHAT: The static + runtime control surface that turns pblock into a
5 * fault-capable lab filesystem. Two tiers:
6 * (1) STATIC opts, a query tail on the backend root
7 * (…/root?lab=1&caps=-sendfile,+memfile&mem=1), stripped at config
8 * parse time and handed to the driver as a one-line sidecar
9 * (<root>/pblock.opts). Consumed once at instance init — it selects
10 * the fail-closed master gate (lab), the F2 capability mask, and the
11 * F16 in-memory catalog mode.
12 * (2) RUNTIME rules, rows in a `ctl(key TEXT PRIMARY KEY, value TEXT,
13 * epoch INTEGER)` table inside the export's catalog.db, driven by
14 * tests via the sqlite3 CLI. The driver re-reads them ONLY when
15 * MAX(epoch) changes and ONLY at a metadata boundary (open); the hot
16 * byte path never touches SQLite — it sees a per-object snapshot
17 * resolved at open (pblock_fault.h).
18 *
19 * WHY: Every behaviour-altering lab feature is fail-closed behind the single
20 * `lab` gate (default OFF): with the gate off pblock is byte-for-byte the
21 * production driver — one cached flag test on the metadata path, zero cost
22 * on the byte path. The ctl table keeps test control out of the config
23 * file (no reload to inject a fault) while the epoch cache keeps the common
24 * case (no rules set) to a single MAX(epoch) probe per open.
25 *
26 * HOW: ngx-free (libc + sqlite3), malloc-owned. Reaches the catalog's sqlite3
27 * connection through the non-static cat_exec/cat_prepare primitives
28 * (sd_pblock_catalog_internal.h). Gated by BRIX_HAVE_SQLITE like the rest
29 * of the backend.
30 *
31 * Requires: sd_pblock_catalog.h (pblock_catalog) before inclusion.
32 */

41/* Static per-export options parsed from the backend root's query tail. */
42typedef struct {
43 int lab; /* master fail-closed gate for all behaviour features */
44 int mem; /* F16: MEMORY-journal catalog (ephemeral lab export) */
45 int audit; /* F17: append-only oplog of metadata-boundary ops */
46 int csi; /* F3: per-block CRC32c integrity (csi catalog table) */
47 int nearline; /* F4: nearline/tape residency simulation */
48 int locks; /* F15: mandatory byte-range/whole-file lease enforce */
49 int dedup; /* F10: refcounted blobs + content-addressed dedup */
50 int snapshots; /* F6: named snapshots / instant fixture reset (⇒ refs) */
51 int versions; /* F11: prior-blob generations to keep on overwrite */
52 int trash; /* F11: unlink → trash instead of free (⇒ refs) */
53 int64_t trash_ttl; /* F11: trash retention seconds (0 = keep until --gc) */
54 int64_t quota_bytes; /* F5: export byte quota (0 = off); suffix k/m/g/t */
55 int64_t quota_inodes;/* F5: export inode (catalog row) quota (0 = off) */
56 uint32_t caps_add; /* F2: capability bits to OR into the driver caps */
57 uint32_t caps_drop; /* F2: capability bits to AND out of the driver caps */
58 int has_caps; /* 1 iff a caps= token was present */
59 char xform[256]; /* F12/F13: transform spec ("crypt:<keyfile>"/"zstd") */
60 size_t xform_len; /* length of xform[] (0 = none) */
61} pblock_opts_t;

63/* Parse a query string ("lab=1&caps=-sendfile,+memfile&mem=1"; may be NULL or
64 * empty) into *out (zeroed first). Unknown keys are ignored (forward-compatible
65 * with later waves); a malformed caps token returns -1 with errno=EINVAL. */
66int pblock_opts_parse(const char *query, pblock_opts_t *out);

68/* Parse a size value ("10G"/"512m"/"1234"; suffix k/m/g/t, case-insensitive)
69 * into a count of bytes. 0 on empty/garbage (⇒ the quota stays off). */

72/* Map one pblock capability name to its BRIX_SD_CAP_* bit, or 0 for an unknown
73 * name. Recognised: fd, sendfile, random_write, range_read, truncate, append,
74 * iouring, server_copy, xattr, xattr_write, hard_rename, dirs, dirs_write,
75 * memfile, nearline, catalog. */
76uint32_t pblock_cap_bit(const char *name, size_t len);

78/* Apply a parsed caps mask over a driver caps word: (caps | add) & ~drop.

82/* Read the static opts sidecar (<root>/pblock.opts, one query-string line — the

105/* F17: op audit log ------------------------------------------------------ *
106 * With `audit=1` set in the static opts, the driver appends one row per
107 * metadata-boundary op (open/close/namespace/staged-commit — NOT per byte-I/O;
108 * per-handle byte totals are folded into the close record's aux) to an
109 * append-only `oplog(seq, ts, op, path, aux, uid, gid, result, errno)` table.
110 * The uid/gid recorded are the catalog-internal synthetic identity (P80 ids),
111 * so multiuser tests can assert attribution. Audit is deliberately best-effort:
112 * a failed oplog write never fails the user op (pblock_audit_log returns void). */
113
114/* Ensure the oplog table exists (idempotent). 0 or -1/errno. */
115int pblock_audit_init(pblock_catalog *cat);
116
117/* Append one op record. Best-effort — errors are swallowed so audit can never
118 * change the outcome of the op it is recording. `aux` may be NULL/"". */
119void pblock_audit_log(pblock_catalog *cat, const char *op, const char *path,
120 const char *aux, uint32_t uid, uint32_t gid, int result, int err);
