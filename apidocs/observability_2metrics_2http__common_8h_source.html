16/*
17 * brix_http_status_class — map an HTTP status code to the
18 * low-cardinality BRIX_HTTP_STATUS_* bucket used in Prometheus labels.
19 *
20 * Returns one of:
21 * BRIX_HTTP_STATUS_1XX — informational (100–199)
22 * BRIX_HTTP_STATUS_2XX — success (200–299)
23 * BRIX_HTTP_STATUS_3XX — redirection (300–399)
24 * BRIX_HTTP_STATUS_4XX — client error (400–499)
25 * BRIX_HTTP_STATUS_5XX — server error (500–599)
26 * BRIX_HTTP_STATUS_OTHER — anything else (0, out-of-range, etc.)
27 *
28 * WHY static inline: eliminates duplicate local definitions in both
29 * webdav/metrics.c and s3/metrics.c with zero link-time overhead.
30 */
31static inline ngx_uint_t
32brix_http_status_class(ngx_uint_t status)
33{
34 if (status >= 100 && status < 200) { return BRIX_HTTP_STATUS_1XX; }
35 if (status >= 200 && status < 300) { return BRIX_HTTP_STATUS_2XX; }
36 if (status >= 300 && status < 400) { return BRIX_HTTP_STATUS_3XX; }
37 if (status >= 400 && status < 500) { return BRIX_HTTP_STATUS_4XX; }
38 if (status >= 500 && status < 600) { return BRIX_HTTP_STATUS_5XX; }
39 return BRIX_HTTP_STATUS_OTHER;
40}
