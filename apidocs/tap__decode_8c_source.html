1/*
2 * tap_decode.c — XRootD frame header decode for the tap core.
3 *
4 * Reuses the single-source BE accessors (frame_hdr.h) and opcode constants
5 * (opcodes.h). A request header is 24B (streamid[2] + requestid[2] + body[16] +
6 * dlen[4]); a response header is 8B (streamid[2] + status[2] + dlen[4]). For a
7 * path-bearing request the path is the data payload at offset XRD_REQUEST_HDR_LEN.
8 */

14/* True for request opcodes whose data payload is (or begins with) a path. */
15int
16brix_tap_opcode_has_path(uint16_t op)
17{
18 switch (op) {
19 case kXR_open:
20 case kXR_stat:
21 case kXR_statx:
22 case kXR_mkdir:
23 case kXR_rm:
24 case kXR_rmdir:
25 case kXR_mv:
26 case kXR_truncate:
27 case kXR_dirlist:
28 case kXR_locate:
29 return 1;
30 default:
31 return 0;
32 }
33}

35size_t
36brix_tap_decode_request(const uint8_t *buf, size_t len,
37 brix_tap_frame_t *out)
38{
39 if (buf == NULL || out == NULL || len < XRD_REQUEST_HDR_LEN) {
40 return 0;
41 }
42
43 out->is_request = 1;
44 out->streamid = xrd_get_u16_be(buf);
45 out->opcode = xrd_get_u16_be(buf + 2);
46 out->status = 0;
47 out->errnum = 0;
48 out->dlen = xrd_get_u32_be(buf + 20);
49 out->path = NULL;
50 out->path_len = 0;
51
52 if (out->dlen > 0 && brix_tap_opcode_has_path(out->opcode)) {
53 size_t avail = len - XRD_REQUEST_HDR_LEN;
54 size_t plen = (out->dlen < avail) ? out->dlen : avail;
55 if (plen > 0) {
56 out->path = buf + XRD_REQUEST_HDR_LEN;
57 out->path_len = plen;
58 }
59 }
60 return XRD_REQUEST_HDR_LEN;
61}

63size_t
64brix_tap_decode_response(const uint8_t *buf, size_t len,
65 brix_tap_frame_t *out)
66{
67 if (buf == NULL || out == NULL || len < 8) {
68 return 0;
69 }
70
71 out->is_request = 0;
72 out->opcode = 0;
73 out->errnum = 0;
74 out->path = NULL;
75 out->path_len = 0;
76 xrd_resp_hdr_unpack(buf, &out->streamid, &out->status, &out->dlen);
77
78 /* A kXR_error body is errnum[4 BE] + errmsg; surface the errnum when the
79 * payload bytes are present so consumers (the bad-actor guard) can map
80 * kXR_NotFound / kXR_NotAuthorized without re-parsing. */
81 if (out->status == kXR_error && out->dlen >= 4 && len >= 8 + 4) {
82 out->errnum = xrd_get_u32_be(buf + 8);
83 }
84 return 8;
85}
