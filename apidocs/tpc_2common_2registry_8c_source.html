3/*
4 * registry.c — shared-memory registry of in-flight third-party copies.
5 *
6 * WHAT: Implements the registry.h lifecycle: reserve the shared zone
7 * (brix_tpc_registry_configure), publish/update/remove/find transfers
8 * by id, and bulk-copy a consistent view out via
9 * brix_tpc_registry_snapshot. Backed by a fixed slot array
10 * (BRIX_TPC_REGISTRY_SLOTS) guarded by a single shared mutex.
11 *
12 * WHY: TPC state must be visible across all nginx worker processes (a transfer
13 * started by one worker may be queried or reported by another, and by the
14 * dashboard/metrics readers). A shared-memory table with a shmtx gives a
15 * lock-protected, cross-process source of truth without per-worker state.
16 *
17 * HOW: brix_tpc_registry_shm_init() zeroes the table and creates the shmtx on
18 * first map (and re-attaches it on reload via the data carry-over path).
19 * Each slot inlines fixed-size src_url/dst_path storage; the registry copies
20 * caller strings into that storage (brix_tpc_registry_copy_str) so the
21 * published brix_tpc_transfer_t never points at caller-owned memory. IDs
22 * are minted from time<<32 ^ pid<<16 ^ a process-local sequence
23 * (brix_tpc_registry_next_id) to stay unique across workers. All slot
24 * mutations take brix_tpc_registry_mutex; brix_tpc_registry_find()
25 * reads without locking and returns a pointer into shared memory for
26 * best-effort lookups.
27 */
28

50/*
51 * Phase 39 (WS5): max age (seconds since updated_at) after which an in-flight
52 * registry slot is considered abandoned and may be reclaimed. 0 = disabled (no
53 * reaping; current behaviour). Set once at config time from
54 * brix_tpc_transfer_max_age (carried into workers by fork). Healthy transfers
55 * refresh updated_at via brix_tpc_registry_update() on every progress emit
56 * (native source.c per 1 MiB chunk; curl per progress callback), so a slot older
57 * than max_age has made no progress at all and is genuinely stuck.
58 *
59 * Reclaim is safe without a separate generation counter: every mutation matches
60 * on the unique 64-bit transfer id under this mutex, so a stale worker that later
61 * calls update/remove for a reaped (or reused-with-a-new-id) slot simply fails to
62 * find its id and no-ops — there is no raw-pointer retention (registry_find is
63 * read-only and unused).
64 */
65static time_t brix_tpc_registry_max_age;

80static ngx_int_t
81brix_tpc_registry_shm_init(ngx_shm_zone_t *shm_zone, void *data)
82{
83 ngx_flag_t fresh;
84 brix_tpc_registry_table_t *tbl;
85
86 /*
87 * Allocate the slot table FROM the slab pool (never over shm.addr) so the
88 * ngx_slab_pool_t header survives; nginx's ngx_unlock_mutexes() force-unlocks
89 * that header's mutex on every child death and would SIGSEGV the master if it
90 * were clobbered. The helper zeroes the table and creates the process-local
91 * mutex from its first member (the ngx_shmtx_sh_t lock).
92 */
93 tbl = brix_shm_table_alloc(shm_zone, data,
94 sizeof(brix_tpc_registry_table_t),
95 &brix_tpc_registry_mutex, &fresh);
96 if (tbl == NULL) {
97 return NGX_ERROR;
98 }
99
100 /*
101 * No fresh-only field inits are required: every slot is reset by the
102 * helper's memzero on first allocation, and the slot capacity is the
103 * compile-time constant BRIX_TPC_REGISTRY_SLOTS. On reuse (fresh == 0) the
104 * live slot contents are deliberately preserved.
105 */
106 (void) fresh;
107
108 return NGX_OK;
109}

136static uint64_t
137brix_tpc_registry_next_id(void)
138{
139 uint64_t id;
140
141 brix_tpc_registry_sequence++;
142 id = (((uint64_t) ngx_time()) << 32)
143 ^ (((uint64_t) ngx_pid) << 16)
144 ^ brix_tpc_registry_sequence;

146 return id == 0 ? 1 : id;

188/*
189 * Reclaim every in-use slot whose updated_at is older than max_age (no progress
190 * for that long ⇒ abandoned). MUST be called with the registry mutex held.
191 * Returns the number of slots reclaimed. No-op (returns 0) when reaping is
192 * disabled. Reclaim is by memzero, identical to a normal remove — safe because
193 * all other access is by unique id under this same lock (see the max_age note).
194 */
195static ngx_uint_t
196brix_tpc_registry_reap_locked(brix_tpc_registry_table_t *tbl, time_t now)
197{
198 ngx_uint_t i;
199 ngx_uint_t reaped = 0;
200
201 if (brix_tpc_registry_max_age <= 0) {
202 return 0;
203 }
204
205 for (i = 0; i < BRIX_TPC_REGISTRY_SLOTS; i++) {
206 if (tbl->slots[i].in_use
207 && (now - tbl->slots[i].transfer.updated_at)
208 > brix_tpc_registry_max_age)
209 {
210 ngx_memzero(&tbl->slots[i], sizeof(tbl->slots[i]));
211 reaped++;
212 }
213 }
214
215 return reaped;
216}

218/*
219 * Public periodic-reaper entry point: reclaim abandoned slots under the lock.
220 * Returns the number reclaimed (0 if disabled / unavailable). Safe to call from
221 * any worker; intended for a coarse timer but also driven inline on a full add.
222 */
223ngx_uint_t
224brix_tpc_registry_reap_stale(ngx_log_t *log)
225{
226 brix_tpc_registry_table_t *tbl;
227 ngx_uint_t reaped;
228
229 if (brix_tpc_registry_max_age <= 0) {
230 return 0;
231 }
232
233 tbl = brix_tpc_registry_table();
234 if (tbl == NULL) {
235 return 0;
236 }
237
238 ngx_shmtx_lock(&brix_tpc_registry_mutex);
239 reaped = brix_tpc_registry_reap_locked(tbl, ngx_time());
240 ngx_shmtx_unlock(&brix_tpc_registry_mutex);
241
242 if (reaped > 0 && log != NULL) {
243 ngx_log_error(NGX_LOG_WARN, log, 0,
244 "brix_tpc: reaped %ui abandoned transfer slot(s) "
245 "(no progress for >%T s)",
246 reaped, brix_tpc_registry_max_age);
247 }
248
249 return reaped;
250}

252/*
253 * Publish a new transfer into a free slot, copying its src_url/dst_path into
254 * slot-owned storage and stamping a fresh id, started/updated time, and a
255 * default PENDING state. Returns the assigned non-zero id, or 0 if the registry
256 * is unavailable or full.
257 */
258uint64_t
259brix_tpc_registry_add(const brix_tpc_transfer_t *transfer, ngx_log_t *log)
260{
261 brix_tpc_registry_table_t *tbl;
262 brix_tpc_registry_entry_t *entry;
263 ngx_uint_t i;
264 ngx_uint_t free_slot;
265 time_t now;
266 uint64_t id;
267
268 tbl = brix_tpc_registry_table();
269 if (tbl == NULL || transfer == NULL) {
270 ngx_log_error(NGX_LOG_WARN, log, 0,
271 "brix_tpc: transfer registry is unavailable");
272 return 0;
273 }
274
275 now = ngx_time();
276 id = brix_tpc_registry_next_id();
277
278 ngx_shmtx_lock(&brix_tpc_registry_mutex);
279
280 free_slot = BRIX_TPC_REGISTRY_SLOTS;
281 for (i = 0; i < BRIX_TPC_REGISTRY_SLOTS; i++) {
282 if (!tbl->slots[i].in_use) {
283 free_slot = i;
284 break;
285 }
286 }
287
288 if (free_slot == BRIX_TPC_REGISTRY_SLOTS) {
289 /*
290 * Phase 39 (WS5): registry full — reclaim abandoned slots (no progress
291 * for > brix_tpc_transfer_max_age) and retry once, so a flood of stalled
292 * transfers self-heals instead of permanently 503-ing every new TPC.
293 * No-op when reaping is disabled. Runs under the lock already held.
294 */
295 if (brix_tpc_registry_reap_locked(tbl, now) > 0) {
296 for (i = 0; i < BRIX_TPC_REGISTRY_SLOTS; i++) {
297 if (!tbl->slots[i].in_use) {
298 free_slot = i;
299 break;
300 }
301 }
302 }
303 }
304
305 if (free_slot == BRIX_TPC_REGISTRY_SLOTS) {
306 ngx_shmtx_unlock(&brix_tpc_registry_mutex);
307 ngx_log_error(NGX_LOG_WARN, log, 0,
308 "brix_tpc: transfer registry is full");
309 return 0;
310 }
311
312 entry = &tbl->slots[free_slot];
313 ngx_memzero(entry, sizeof(*entry));
314
315 entry->in_use = 1;
316 entry->transfer = *transfer;
317 entry->transfer.id = id;
318 entry->transfer.started_at = now;
319 entry->transfer.updated_at = now;
320 if (entry->transfer.state == 0) {
321 entry->transfer.state = BRIX_TPC_STATE_PENDING;
322 }
323
324 brix_tpc_registry_copy_str(&entry->transfer.src_url,
325 entry->src_url_data,
326 sizeof(entry->src_url_data),
327 &transfer->src_url);
328 brix_tpc_registry_copy_str(&entry->transfer.dst_path,
329 entry->dst_path_data,
330 sizeof(entry->dst_path_data),
331 &transfer->dst_path);
332
333 ngx_shmtx_unlock(&brix_tpc_registry_mutex);
334
335 return id;
336}

454 * for a consistent view. Returns the number of transfers written.

459{
