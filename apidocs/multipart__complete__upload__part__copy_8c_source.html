1/*
2 * multipart_complete_upload_part_copy.c — S3 UploadPartCopy handler.
3 *
4 * WHAT: This fragment implements the PUT /bucket/key?partNumber=N&uploadId=<id> + x-amz-copy-source header
5 * → UploadPartCopy operation, which copies a part from a source object into the staging directory of an
6 * active multipart upload. It contains two components:
7 * - s3_find_request_header_value(): helper that locates a named request header via brix_http_find_header(),
8 * extracts its value bytes into a pool-allocated null-terminated string, and returns it for parsing.
9 * - s3_handle_upload_part_copy(): the main handler — parses x-amz-copy-source to extract source object key
10 * (stripping leading "/" then skipping bucket prefix), validates source path against root_canon with ".."
11 * check, opens source file for read and target part file for write, copies via 64KB buffered read/write loop
12 * with EINTR retry, emits CopyPartResult XML with ETag ("mtime-size") and LastModified (ISO 8601). On write
13 * failure: closes both fds, unlinks the partial part file, returns InternalError.
14 *
15 * WHY: Split from multipart.c to keep UploadPartCopy logic alongside its sibling fragments for CompleteMultipartUpload
16 * operations. The copy-source parsing and buffered file-copy loop are distinct enough from part enumeration and XML
17 * completion that separation keeps each fragment focused on one responsibility.
18 */

72/*
73 * s3_copy_source_key() — extract the source object key from x-amz-copy-source.
74 *
75 * WHAT: Strips a single leading '/' then skips the bucket-name prefix component
76 * (everything up to and including the first remaining '/'), returning a pointer
77 * into the header string at the start of the object key.
78 * WHY: x-amz-copy-source names a source as "/bucket/key" (or "bucket/key"); the
79 * handler resolves paths relative to the export root, so only the key survives.
80 * HOW: Pointer arithmetic over the caller-owned header buffer — no allocation,
81 * no mutation; the returned pointer aliases into copy_src_hdr.
82 */
83static const char *
84s3_copy_source_key(const char *copy_src_hdr)
85{
86 const char *src_key = copy_src_hdr;
87 const char *slash;
88
89 if (*src_key == '/') {
90 src_key++;
91 }
92
93 slash = strchr(src_key, '/');
94 if (slash != NULL) {
95 src_key = slash + 1;
96 }
97
98 return src_key;
99}

101/*
102 * s3_resolve_copy_source() — build + validate + probe the copy-source object.
103 *
104 * WHAT: Composes the absolute source path from root_canon + key, applies the
105 * string escape checks ("/../" reject + root_canon prefix), then probes the
106 * source through the confined resolver (no-follow) requiring a regular file.
107 * On success fills *sctx (ready to open) and returns NGX_OK; on any failure it
108 * emits the appropriate S3 XML error and returns NGX_DONE.
109 * WHY: This is the security-load-bearing source-auth stage. The string checks
110 * alone do NOT stop an in-bucket symlink, so the confined probe (openat2
111 * RESOLVE_BENEATH) is mandatory before the byte copy trusts the source.
112 * HOW: snprintf into a caller-owned PATH_MAX buffer, early-return on each guard
113 * with the exact error mapping the original inline code used.
114 */
115static ngx_int_t
116s3_resolve_copy_source(const s3_upcp_req_t *req, const char *src_key,
117 char *src_fs_path, size_t src_fs_path_sz, brix_vfs_ctx_t *sctx)
118{
119 ngx_http_request_t *r = req->r;
120 ngx_http_s3_loc_conf_t *cf = req->cf;
121 brix_vfs_stat_t svst;
122
123 ngx_memzero(&svst, sizeof(svst));
124
125 if ((size_t) snprintf(src_fs_path, src_fs_path_sz, "%s/%s",
126 cf->common.root_canon, src_key) >= src_fs_path_sz
127 || strstr(src_fs_path, "/../") != NULL
128 || strncmp(src_fs_path, cf->common.root_canon,
129 strlen(cf->common.root_canon)) != 0)
130 {
131 (void) s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
132 "InvalidArgument",
133 "Invalid copy source path.");
134 return NGX_DONE;
135 }
136
137 /* Confinement: stat the source through the confined resolver (openat2
138 * RESOLVE_BENEATH, no-follow) — the strstr/strncmp checks above do NOT stop
139 * a planted in-bucket symlink, so a raw stat() here would follow it out of
140 * the export root (the same hole the open() below was hardened against). */
141 s3_build_vfs_ctx(r, src_fs_path, cf, sctx);
142 if (brix_vfs_probe(sctx, 1 /* no-follow */, &svst) != NGX_OK
143 || !svst.is_regular)
144 {
145 (void) s3_send_xml_error(r, NGX_HTTP_NOT_FOUND,
146 "NoSuchKey",
147 "The specified copy source does not exist.");
148 return NGX_DONE;
149 }
150
151 return NGX_OK;
152}

154/*
155 * s3_resolve_part_target() — validate uploadId, probe the MPU dir, build part path.
156 *
157 * WHAT: Validates the upload_id syntax, derives the staging directory, probes it
158 * through the VFS seam for existence, then formats the destination part path.
159 * Returns NGX_OK with part_path filled, or NGX_DONE after emitting the matching
160 * S3 XML error (InvalidArgument uploadId / NoSuchUpload).
161 * WHY: The write target must exist as an active upload before we open the part
162 * file; keeping this validation together mirrors the original ordering exactly.
163 * HOW: Reuses the request bundle's fs_path/upload_id/part_num; probes with a
164 * local vfs ctx so the source ctx is untouched; early-return on each guard.
165 */
166static ngx_int_t
167s3_resolve_part_target(const s3_upcp_req_t *req, char *part_path,
168 size_t part_path_sz)
169{
170 ngx_http_request_t *r = req->r;
171 ngx_http_s3_loc_conf_t *cf = req->cf;
172 char mpu_dir[PATH_MAX];
173 brix_vfs_ctx_t uctx;
174 brix_vfs_stat_t svst;
175 u_char *p;
176
177 ngx_memzero(&svst, sizeof(svst));
178
179 if (!mpu_validate_upload_id(req->upload_id)) {
180 (void) s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
181 "InvalidArgument",
182 "The uploadId is invalid.");
183 return NGX_DONE;
184 }
185
186 /* Confinement + existence: mpu_dir from s3_get_mpu_dir() (validated upload_id +
187 * confined fs_path). Probe it through the VFS seam rather than a raw stat —
188 * absent → 404 NoSuchUpload. */
189 s3_get_mpu_dir(req->fs_path, req->upload_id, mpu_dir, sizeof(mpu_dir));
190 s3_build_vfs_ctx(r, mpu_dir, cf, &uctx);
191 if (brix_vfs_probe(&uctx, 1 /* nofollow */, &svst) != NGX_OK) {
192 (void) s3_send_xml_error(r, NGX_HTTP_NOT_FOUND,
193 "NoSuchUpload",
194 "The specified upload does not exist.");
195 return NGX_DONE;
196 }
197
198 p = ngx_snprintf((u_char *) part_path, part_path_sz - 1,
199 "%s/part.%z", mpu_dir, (size_t) req->part_num);
200 *p = '\0';
201
202 return NGX_OK;
203}

205/*
206 * s3_copy_bytes() — stream the source object into the destination part fd.
207 *
208 * WHAT: Reads the confined source fd and writes the destination fd via the SD
209 * driver pread/pwrite seam in 64KB positional chunks with EINTR retry, until
210 * the source is exhausted. Returns 0 on success, -1 on an unrecoverable write
211 * error (caller handles cleanup + metric + error status).
212 * WHY: Isolating the copy loop keeps the driver-level byte move (backend↔backend,
213 * POSIX today, positional so a non-POSIX backend slots in) in one focused unit.
214 * HOW: Nested read/write loops over a caller-supplied fd pair; no ownership of
215 * the fds and no error emission — purely the transfer with a boolean outcome.
216 */
217static int
218s3_copy_bytes(int src_fd, int dst_fd, char *iobuf, size_t iobuf_sz)
219{
220 off_t src_off = 0, dst_off = 0;
221 ssize_t nr;
222 brix_sd_obj_t src_obj, dst_obj;
223
224 brix_sd_posix_wrap(&src_obj, src_fd);
225 brix_sd_posix_wrap(&dst_obj, dst_fd);
226
227 while ((nr = src_obj.driver->pread(&src_obj, iobuf, iobuf_sz,
228 src_off)) > 0) {
229 char *wbuf = iobuf;
230 ssize_t remaining = nr;
231 while (remaining > 0) {
232 ssize_t nw = dst_obj.driver->pwrite(&dst_obj, wbuf,
233 (size_t) remaining, dst_off);
234 if (nw <= 0) {
235 if (errno == EINTR) {
236 continue;
237 }
238 return -1;
239 }
240 wbuf += nw;
241 remaining -= nw;
242 dst_off += nw;
243 }
244 src_off += nr;
245 }
246
247 return 0;
248}

250/*
251 * s3_send_copy_part_result() — build + send the CopyPartResult XML body.
252 *
253 * WHAT: Derives ETag ("mtime-size") and ISO 8601 LastModified from the written
254 * part's stat, formats the CopyPartResult document into a pool buffer, and
255 * sends it as application/xml with 200 OK.
256 * WHY: The success response is a self-contained rendering stage with no I/O side
257 * effects, so it composes cleanly at the edge after the byte move completes.
258 * HOW: snprintf into a local buffer, copy into an ngx pool temp buf, delegate to
259 * the shared XML sender; returns its status (or InternalError on OOM).
260 */
261static ngx_int_t
262s3_send_copy_part_result(ngx_http_request_t *r, const struct stat *part_sb)
263{
264 char xml_buf[512];
265 char etag_buf[64];
266 char iso_buf[32];
267 ngx_buf_t *b;
268 size_t xml_len;
269
270 s3_etag(part_sb, etag_buf, sizeof(etag_buf));
271 brix_format_iso8601(part_sb->st_mtime, iso_buf, sizeof(iso_buf));
272
273 xml_len = (size_t) snprintf(xml_buf, sizeof(xml_buf),
274 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
275 "<CopyPartResult "
276 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">"
277 "<ETag>%s</ETag>"
278 "<LastModified>%s</LastModified>"
279 "</CopyPartResult>",
280 etag_buf, iso_buf);
281
282 b = ngx_create_temp_buf(r->pool, xml_len);
283 if (b == NULL) {
284 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
285 return NGX_HTTP_INTERNAL_SERVER_ERROR;
286 }
287 ngx_memcpy(b->pos, xml_buf, xml_len);
288 b->last = b->pos + xml_len;
289 b->last_buf = 1;
290
291 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
292 (ngx_str_t) ngx_string("application/xml"), b);
293}

295ngx_int_t
296s3_handle_upload_part_copy(ngx_http_request_t *r,
297 const char *fs_path,
298 ngx_http_s3_loc_conf_t *cf,
299 const char *upload_id,
300 int part_num)
301{
302 const char *copy_src_hdr;
303 const char *src_key;
304 char src_fs_path[PATH_MAX];
305 char part_path[PATH_MAX];
306 char iobuf[65536];
307 struct stat part_sb; /* fstat of the written part fd (ETag size/mtime) */
308 brix_vfs_ctx_t sctx; /* copy SOURCE ctx (probe + open below) */
309 brix_vfs_file_t *fh_src;
310 int dst_fd;
311 ngx_int_t rc;
312 s3_upcp_req_t req;
313
314 ngx_memzero(&part_sb, sizeof(part_sb));
315
316 req.r = r;
317 req.cf = cf;
318 req.fs_path = fs_path;
319 req.upload_id = upload_id;
320 req.part_num = part_num;
321
322 copy_src_hdr = s3_find_request_header_value(
323 r, "x-amz-copy-source", sizeof("x-amz-copy-source") - 1);
324 if (copy_src_hdr == NULL) {
325 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
326 "InvalidArgument",
327 "Missing x-amz-copy-source header.");
328 }
329
330 src_key = s3_copy_source_key(copy_src_hdr);
331
332 /* Source-object auth (security-load-bearing): validate + confined-probe the
333 * copy source BEFORE the upload target is resolved or opened. */
334 rc = s3_resolve_copy_source(&req, src_key, src_fs_path,
335 sizeof(src_fs_path), &sctx);
336 if (rc != NGX_OK) {
337 return NGX_OK; /* error already emitted (returns NGX_OK to nginx) */
338 }
339
340 rc = s3_resolve_part_target(&req, part_path, sizeof(part_path));
341 if (rc != NGX_OK) {
342 return NGX_OK; /* error already emitted */
343 }
344
345 /* Copy source file → part file. Open the source through the confined
346 * resolver (openat2 RESOLVE_BENEATH): this refuses to follow a symlink out
347 * of the export root and rejects any path that escapes it. */
348 fh_src = brix_vfs_open(&sctx, BRIX_VFS_O_READ, NULL);
349 if (fh_src == NULL) {
350 return s3_send_xml_error(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
351 "The specified copy source does not exist.");
352 }
353
354 dst_fd = brix_vfs_open_fd(r->connection->log, cf->common.root_canon,
355 part_path, O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC,
356 0600);
357 if (dst_fd < 0) {
358 brix_vfs_close(fh_src, r->connection->log);
359 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
360 return NGX_HTTP_INTERNAL_SERVER_ERROR;
361 }
362
363 if (s3_copy_bytes(brix_vfs_file_fd(fh_src), dst_fd,
364 iobuf, sizeof(iobuf)) != 0) {
365 brix_vfs_close(fh_src, r->connection->log);
366 close(dst_fd);
367 (void) brix_vfs_unlink_path(r->connection->log,
368 cf->common.root_canon, part_path);
369 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
370 return NGX_HTTP_INTERNAL_SERVER_ERROR;
371 }
372 brix_vfs_close(fh_src, r->connection->log);
373
374 /* ETag/mtime from the part fd we just wrote (metadata on a VFS-opened confined
375 * fd — no raw path stat) before we release it. */
376 if (fstat(dst_fd, &part_sb) != 0) {
377 close(dst_fd);
378 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
379 return NGX_HTTP_INTERNAL_SERVER_ERROR;
380 }
381 close(dst_fd);
382
383 return s3_send_copy_part_result(r, &part_sb);
384}
