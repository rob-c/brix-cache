1/*
2 * multipart_complete_upload_part_copy.c — S3 UploadPartCopy handler.
3 *
4 * WHAT: This fragment implements the PUT /bucket/key?partNumber=N&uploadId=<id> + x-amz-copy-source header
5 * → UploadPartCopy operation, which copies a part from a source object into the staging directory of an
6 * active multipart upload. It contains two components:
7 * - s3_find_request_header_value(): helper that locates a named request header via brix_http_find_header(),
8 * extracts its value bytes into a pool-allocated null-terminated string, and returns it for parsing.
9 * - s3_handle_upload_part_copy(): the main handler — parses x-amz-copy-source to extract source object key
10 * (stripping leading "/" then skipping bucket prefix), validates source path against root_canon with ".."
11 * check, opens source file for read and target part file for write, copies via 64KB buffered read/write loop
12 * with EINTR retry, emits CopyPartResult XML with ETag ("mtime-size") and LastModified (ISO 8601). On write
13 * failure: closes both fds, unlinks the partial part file, returns InternalError.
14 *
15 * WHY: Split from multipart.c to keep UploadPartCopy logic alongside its sibling fragments for CompleteMultipartUpload
16 * operations. The copy-source parsing and buffered file-copy loop are distinct enough from part enumeration and XML
17 * completion that separation keeps each fragment focused on one responsibility.
18 */

22#include "core/http/http_headers.h"
23#include "fs/backend/sd.h" /* route the part-copy byte move through the SD backend */

33static const char *
34s3_find_request_header_value(ngx_http_request_t *r, const char *name,
35 size_t name_len)
36{
37 ngx_table_elt_t *h;
38 u_char *value;
39
40 h = brix_http_find_header(r, name, name_len);
41 if (h == NULL) {
42 return NULL;
43 }
44
45 value = (u_char *) brix_pstrdup_z(r->pool, &h->value);
46 if (value == NULL) {
47 return NULL;
48 }
49
50 return (const char *) value;
51}

53ngx_int_t
54s3_handle_upload_part_copy(ngx_http_request_t *r,
55 const char *fs_path,
56 ngx_http_s3_loc_conf_t *cf,
57 const char *upload_id,
58 int part_num)
59{
60 const char *copy_src_hdr;
61 const char *src_key;
62 char src_fs_path[PATH_MAX];
63 char mpu_dir[PATH_MAX];
64 char part_path[PATH_MAX];
65 char xml_buf[512];
66 char etag_buf[64];
67 char iso_buf[32];
68 struct stat part_sb; /* fstat of the written part fd (ETag size/mtime) */
69 brix_vfs_ctx_t sctx; /* copy SOURCE ctx (probe + open below) */
70 brix_vfs_ctx_t uctx; /* upload staging-dir (mpu_dir) existence probe */
71 brix_vfs_stat_t svst;
72 brix_vfs_file_t *fh_src;
73 int src_fd, dst_fd;
74 ssize_t nr;
75 char iobuf[65536];
76 ngx_buf_t *b;
77 size_t xml_len;
78 u_char *p;
79
80 copy_src_hdr = s3_find_request_header_value(
81 r, "x-amz-copy-source", sizeof("x-amz-copy-source") - 1);
82 if (copy_src_hdr == NULL) {
83 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
84 "InvalidArgument",
85 "Missing x-amz-copy-source header.");
86 }
87
88 /* Strip leading '/' then skip the bucket prefix component */
89 src_key = copy_src_hdr;
90 if (*src_key == '/') {
91 src_key++;
92 }
93 {
94 const char *slash = strchr(src_key, '/');
95 if (slash != NULL) {
96 src_key = slash + 1;
97 }
98 }
99
100 /* Build and validate source path */
101 if ((size_t) snprintf(src_fs_path, sizeof(src_fs_path), "%s/%s",
102 cf->common.root_canon, src_key) >= sizeof(src_fs_path)
103 || strstr(src_fs_path, "/../") != NULL
104 || strncmp(src_fs_path, cf->common.root_canon,
105 strlen(cf->common.root_canon)) != 0)
106 {
107 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
108 "InvalidArgument",
109 "Invalid copy source path.");
110 }
111
112 /* Confinement: stat the source through the confined resolver (openat2
113 * RESOLVE_BENEATH, no-follow) — the strstr/strncmp checks above do NOT stop
114 * a planted in-bucket symlink, so a raw stat() here would follow it out of
115 * the export root (the same hole the open() below was hardened against). */
116 s3_build_vfs_ctx(r, src_fs_path, cf, &sctx);
117 if (brix_vfs_probe(&sctx, 1 /* no-follow */, &svst) != NGX_OK
118 || !svst.is_regular)
119 {
120 return s3_send_xml_error(r, NGX_HTTP_NOT_FOUND,
121 "NoSuchKey",
122 "The specified copy source does not exist.");
123 }
124
125 if (!mpu_validate_upload_id(upload_id)) {
126 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
127 "InvalidArgument",
128 "The uploadId is invalid.");
129 }
130
131 /* Confinement + existence: mpu_dir from s3_get_mpu_dir() (validated upload_id +
132 * confined fs_path). Probe it through the VFS seam (reuse sctx, re-pointed at
133 * mpu_dir) rather than a raw stat — absent → 404 NoSuchUpload. */
134 s3_get_mpu_dir(fs_path, upload_id, mpu_dir, sizeof(mpu_dir));
135 s3_build_vfs_ctx(r, mpu_dir, cf, &uctx);
136 if (brix_vfs_probe(&uctx, 1 /* nofollow */, &svst) != NGX_OK) {
137 return s3_send_xml_error(r, NGX_HTTP_NOT_FOUND,
138 "NoSuchUpload",
139 "The specified upload does not exist.");
140 }
141
142 p = ngx_snprintf((u_char *) part_path, sizeof(part_path) - 1,
143 "%s/part.%z", mpu_dir, (size_t) part_num);
144 *p = '\0';
145
146 /* Copy source file → part file. Open the source through the confined
147 * resolver (openat2 RESOLVE_BENEATH): this refuses to follow a symlink out
148 * of the export root and rejects any path that escapes it — the same
149 * confinement every other read path uses. A raw open() here followed a
150 * planted in-bucket symlink straight to a host file (e.g. /etc/passwd),
151 * which the string checks above (strstr/strncmp) do not catch. */
152 fh_src = brix_vfs_open(&sctx, BRIX_VFS_O_READ, NULL);
153 if (fh_src == NULL) {
154 return s3_send_xml_error(r, NGX_HTTP_NOT_FOUND, "NoSuchKey",
155 "The specified copy source does not exist.");
156 }
157 src_fd = brix_vfs_file_fd(fh_src);
158
159 dst_fd = brix_vfs_open_fd(r->connection->log, cf->common.root_canon,
160 part_path, O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC,
161 0600);
162 if (dst_fd < 0) {
163 brix_vfs_close(fh_src, r->connection->log);
164 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
165 return NGX_HTTP_INTERNAL_SERVER_ERROR;
166 }
167
168 /* VFS↔VFS (backend↔backend) byte move: read the copy source object through
169 * its driver, write the part object through its driver. Both POSIX today;
170 * positional so a non-POSIX backend slots in unchanged. */
171 {
172 off_t src_off = 0, dst_off = 0;
173 brix_sd_obj_t src_obj, dst_obj;
174
175 brix_sd_posix_wrap(&src_obj, src_fd);
176 brix_sd_posix_wrap(&dst_obj, dst_fd);
177
178 while ((nr = src_obj.driver->pread(&src_obj, iobuf, sizeof(iobuf),
179 src_off)) > 0) {
180 char *wbuf = iobuf;
181 ssize_t remaining = nr;
182 while (remaining > 0) {
183 ssize_t nw = dst_obj.driver->pwrite(&dst_obj, wbuf,
184 (size_t) remaining, dst_off);
185 if (nw <= 0) {
186 if (errno == EINTR) {
187 continue;
188 }
189 brix_vfs_close(fh_src, r->connection->log);
190 close(dst_fd);
191 (void) brix_vfs_unlink_path(r->connection->log,
192 cf->common.root_canon, part_path);
193 BRIX_S3_METRIC_INC(
194 events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
195 return NGX_HTTP_INTERNAL_SERVER_ERROR;
196 }
197 wbuf += nw;
198 remaining -= nw;
199 dst_off += nw;
200 }
201 src_off += nr;
202 }
203 }
204 brix_vfs_close(fh_src, r->connection->log);
205
206 /* ETag/mtime from the part fd we just wrote (metadata on a VFS-opened confined
207 * fd — no raw path stat) before we release it. */
208 if (fstat(dst_fd, &part_sb) != 0) {
209 close(dst_fd);
210 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
211 return NGX_HTTP_INTERNAL_SERVER_ERROR;
212 }
213 close(dst_fd);
214
215 s3_etag(&part_sb, etag_buf, sizeof(etag_buf));
216 brix_format_iso8601(part_sb.st_mtime, iso_buf, sizeof(iso_buf));
217
218 xml_len = (size_t) snprintf(xml_buf, sizeof(xml_buf),
219 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
220 "<CopyPartResult "
221 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">"
222 "<ETag>%s</ETag>"
223 "<LastModified>%s</LastModified>"
224 "</CopyPartResult>",
225 etag_buf, iso_buf);
226
227 b = ngx_create_temp_buf(r->pool, xml_len);
228 if (b == NULL) {
229 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
230 return NGX_HTTP_INTERNAL_SERVER_ERROR;
231 }
232 ngx_memcpy(b->pos, xml_buf, xml_len);
233 b->last = b->pos + xml_len;
234 b->last_buf = 1;
235
236 return brix_http_send_xml_buffer(r, NGX_HTTP_OK,
237 (ngx_str_t) ngx_string("application/xml"), b);
238}
