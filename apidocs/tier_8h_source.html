4/*
5 * tier.h — the composable storage-tier model (phase-64 §2).
6 *
7 * WHAT: The POD types that describe a node's storage stack as up to three
8 * orthogonal TIERS — backend, cache_store, stage_store — each a complete
9 * self-describing filesystem addressed by { driver, location, credential }.
10 * Declares the parsed per-tier config (brix_tier_cfg_t), the per-export
11 * stack (brix_tier_stack_t) that the registry memoises, the cache/stage
12 * policy PODs, the capability-gap report, and the four public entry points
13 * (parse_store / status / build / build_stack).
14 *
15 * WHY: Before phase-64 the registry carried flat backend[16] / origin_x / staging
16 * fields and the cache was a separate, scheme-dispatched subsystem. The tier
17 * model unifies them: every tier is one brix_sd_instance_t, the registry
18 * composes them top-down (cache → stage → backend), and the VFS consumes the
19 * composed top BLIND — no handler, VFS op, or cache module branches on a
20 * driver/protocol/tier (P3, G4/G5). The cache and stage tiers are OPTIONAL,
21 * except a nearline (tape) backend REQUIRES a cache tier as its recall
22 * target (P4, §9.4).
23 *
24 * HOW: tier_config.c parses a "<scheme>:<location> [credential=][block_size=]"
25 * store URL into an brix_tier_cfg_t and reports each tier's readiness
26 * against a per-role capability contract (§2.2) — a missing slot is a
27 * tracked "needs development" task, never a blocker (P1). tier_build.c is
28 * the single driver-name dispatch that turns one cfg into a bound SD
29 * instance, and composes the three into the stack's top via the sd_cache /
30 * sd_stage decorators. See docs/refactor/phase-64-fully-tiered-composable-
31 * storage.md (§2, §4, §5, §8, Appendix I/L).
32 */

47typedef enum {

56/* ---- one parsed tier ------------------------------------------------------
57 * Filled by brix_tier_parse_store from a store-URL directive. `driver` is the
58 * resolved SD driver name (root/roots→"xroot", http/https→"http", else literal).
59 * For a local tier (posix/pblock) `path` is the confined export directory and
60 * host is ""; for a remote tier `host`:`port` is the authority and `path` is the
61 * URL/bucket/pool remainder. */
62typedef struct {
63 brix_tier_role_t role;
64 char driver[16]; /* posix|pblock|xroot|http|s3|rados|tape */
65 char host[256]; /* remote authority; "" when local */
66 int port;
67 int tls; /* roots:// / https → 1 */
68 char path[1024]; /* dir | /sub | /bucket | /pool[/ns] */
69 const brix_credential_t *credential; /* §14; NULL = anonymous */
70 size_t block_size; /* pblock object block size (0 = def) */
71 unsigned nearline:1; /* tape:// — async-recall reads (§9) */
72 unsigned configured:1; /* a store URL was parsed for this tier */
73} brix_tier_cfg_t;

82 "" = unlink the failed part */

104/* ---- the per-export stack -------------------------------------------------

143/* Report whether the driver bound for tier `t` satisfies its role's capability
144 * contract (§2.2). `probe` is a built instance of the tier's driver. Returns
