1/*
2 * sd_pblock_io.c — worker-safe byte I/O for the pblock storage driver.
3 *
4 * WHAT: Implements the data-plane vtable slots of brix_sd_pblock_driver:
5 * pread/pwrite, the vectored preadv/preadv2, copy_range, the sendfile-fd
6 * selector, ftruncate, fsync and fstat. These map a byte range across an
7 * object's fixed-size block files (holes read as zeros) and keep the cached
8 * size/mtime coherent, flushing dirty metadata to the catalog on fsync.
9 *
10 * WHY: Split out of sd_pblock.c (phase-79) so every pblock file stays under the
11 * ~500-line, one-concept cap. This is the hot path — block-file reads and
12 * writes with no SQLite work — kept separate from the object lifecycle and
13 * namespace concerns. All functions are non-static because the driver
14 * descriptor in sd_pblock.c names them (and the object-open O_TRUNC path
15 * calls sd_pblock_ftruncate); declarations live in sd_pblock_internal.h.
16 *
17 * HOW: Each object's block 0 is the persistently-open obj->fd; higher blocks are
18 * opened transiently per I/O through the packed-block engine (pblock_store.h).
19 * Reads clamp to the cached size (EOF → 0); writes advance the cached
20 * high-water size and set the dirty flag. ngx-free (libc + the block engine)
21 * so the module and the standalone unit test compile it identically. Gated
22 * by BRIX_HAVE_SQLITE like the rest of the backend.
23 */

46ssize_t
47sd_pblock_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
48{
49 pblock_obj_t *os = obj->state;
50 size_t avail;
51
52 if (off < 0 || off >= os->meta.size) {
53 return 0; /* at/after EOF */
54 }
55 avail = (size_t) (os->meta.size - off);
56 if (len > avail) {
57 len = avail;
58 }
59 if (len == 0) {
60 return 0;
61 }
62 return pblock_read_blocks(os->st, os->blob_id, os->block_size, obj->fd,
63 buf, len, off);
64}

66ssize_t
67sd_pblock_pwrite(brix_sd_obj_t *obj, const void *buf, size_t len, off_t off)
68{
69 pblock_obj_t *os = obj->state;
70 ssize_t n;
71
72 if (len == 0) {
73 return 0;
74 }
75 n = pblock_write_blocks(os->st, os->blob_id, os->block_size, obj->fd,
76 buf, len, off);
77 if (n > 0) {
78 os->meta.mtime = pblock_now();
79 if ((int64_t) off + n > os->meta.size) {
80 os->meta.size = (int64_t) off + n;
81 }
82 os->dirty = 1;
83 }
84 return n;
85}

87/* sd_pblock_preadv — vectored read as a loop of the block-aware pread (stops at
88 * the first short/EOF segment); bytes read or -1. */
89ssize_t
90sd_pblock_preadv(brix_sd_obj_t *obj, const struct iovec *iov, int iovcnt,
91 off_t off)
92{
93 ssize_t total = 0;
94 int i;
95
96 for (i = 0; i < iovcnt; i++) {
97 ssize_t n = sd_pblock_pread(obj, iov[i].iov_base, iov[i].iov_len,
98 off + total);
99
100 if (n < 0) {
101 return total ? total : -1;
102 }
103 total += n;
104 if ((size_t) n < iov[i].iov_len) {
105 break; /* short read / EOF */
106 }
107 }
108 return total;
109}

119/* sd_pblock_copy_range — copy up to `len` bytes src->dst through a bounded
120 * userspace buffer (blocks straddle file boundaries, so copy_file_range across
121 * whole objects does not apply). The VFS owns the outer loop; one call copies up
122 * to the buffer size. Returns bytes copied (0 = src EOF) or -1. */
123ssize_t
124sd_pblock_copy_range(brix_sd_obj_t *src, off_t src_off, brix_sd_obj_t *dst,
125 off_t dst_off, size_t len)
126{
127 pblock_obj_t *sos = src->state;
128 pblock_obj_t *dos = dst->state;
129 char buf[65536];
130 size_t chunk = len < sizeof(buf) ? len : sizeof(buf);
131 ssize_t r, w;
132
133 if (src_off < 0 || src_off >= sos->meta.size) {
134 return 0;
135 }
136 if (chunk > (size_t) (sos->meta.size - src_off)) {
137 chunk = (size_t) (sos->meta.size - src_off);
138 }
139 if (chunk == 0) {
140 return 0;
141 }
142
143 r = pblock_read_blocks(sos->st, sos->blob_id, sos->block_size, src->fd,
144 buf, chunk, src_off);
145 if (r <= 0) {
146 return r;
147 }
148 w = pblock_write_blocks(dos->st, dos->blob_id, dos->block_size, dst->fd,
149 buf, (size_t) r, dst_off);
150 if (w > 0) {
151 dos->meta.mtime = pblock_now();
152 if ((int64_t) dst_off + w > dos->meta.size) {
153 dos->meta.size = (int64_t) dst_off + w;
154 }
155 dos->dirty = 1;
156 }
157 return w;
158}

160/* sd_pblock_read_sendfile_fd — zero-copy only for an offset-0 range that lies
161 * within block 0 (the persistently-open obj->fd): small files and the start of
162 * large ones. Multi-block ranges return NGX_INVALID_FILE (served memory-backed)
163 * until the VFS read path is block-aware. */
164ngx_fd_t
165sd_pblock_read_sendfile_fd(brix_sd_obj_t *obj, off_t off, size_t len,
166 unsigned want_zerocopy)
167{
168 pblock_obj_t *os = obj->state;
169
170 if (!want_zerocopy || obj->fd == NGX_INVALID_FILE) {
171 return NGX_INVALID_FILE;
172 }
173 if (off == 0 && (int64_t) len <= os->block_size) {
174 return obj->fd;
175 }
176 return NGX_INVALID_FILE;
177}

179ngx_int_t
180sd_pblock_ftruncate(brix_sd_obj_t *obj, off_t len)
181{
182 pblock_obj_t *os = obj->state;
183 int64_t bs = os->block_size;
184 int64_t old_last = pblock_last_block(os->meta.size, bs);
185 int64_t keep = pblock_last_block(len, bs);
186 int64_t boundary = (int64_t) len - keep * bs; /* bytes kept in `keep` */
187 int64_t i;
188 int fd, transient = 0;
189
190 /* trim the boundary block to its surviving length */
191 if (keep == 0) {
192 fd = obj->fd;
193 } else {
194 char bp[PATH_MAX];
195
196 if (pblock_block_path(os->st, os->blob_id, keep, bp, sizeof(bp)) != 0) {
197 return NGX_ERROR;
198 }
199 fd = open(bp, O_RDWR | O_CREAT, 0600);
200 if (fd < 0) {
201 return NGX_ERROR;
202 }
203 transient = 1;
204 }
205 if (ftruncate(fd, (off_t) boundary) != 0) {
206 if (transient) {
207 close(fd);
208 }
209 return NGX_ERROR;
210 }
211 if (transient) {
212 close(fd);
213 }
214
215 /* drop whole blocks past the new size */
216 for (i = keep + 1; i <= old_last; i++) {
217 char bp[PATH_MAX];
218
219 if (pblock_block_path(os->st, os->blob_id, i, bp, sizeof(bp)) == 0) {
220 unlink(bp);
221 }
222 }
223
224 os->meta.size = (int64_t) len;
225 os->meta.mtime = pblock_now();
226 os->dirty = 1;
227 return NGX_OK;
228}

230/* sd_pblock_fsync — durability barrier: fsync every block file backing the
231 * object, then flush dirty size/mtime to the catalog. */
232ngx_int_t
233sd_pblock_fsync(brix_sd_obj_t *obj)
234{
235 pblock_obj_t *os = obj->state;
236 int64_t last = pblock_last_block(os->meta.size, os->block_size);
237 int64_t i;
238
239 if (obj->fd != NGX_INVALID_FILE && fsync(obj->fd) != 0) {
240 return NGX_ERROR;
241 }
242 for (i = 1; i <= last; i++) {
243 char bp[PATH_MAX];
244 int fd;
245
246 if (pblock_block_path(os->st, os->blob_id, i, bp, sizeof(bp)) != 0) {
247 continue;
248 }
249 fd = open(bp, O_RDONLY);
250 if (fd >= 0) {
251 if (fsync(fd) != 0) {
252 close(fd);
253 return NGX_ERROR;
254 }
255 close(fd);
256 }
257 }
258
259 if (os->dirty) {
260 if (pblock_catalog_touch(os->st->cat, os->path, os->meta.size,
261 os->meta.mtime) != 0)
262 {
263 return NGX_ERROR;
264 }
265 os->dirty = 0;
266 }
267 return NGX_OK;
268}
