1/* kXR_clone (3032) — server-side range copy, protocol v5.2.0.
2 *
3 * Wire format:
4 * Header body: dst_fhandle[4] + reserved[12]
5 * Payload: array of clone_item (32 bytes each):
6 * src_fhandle[4] + reserved[4] +
7 * src_offset(u64be) + src_len(u64be) + dst_offset(u64be)
8 *
9 * Each item copies src_len bytes from src_offset in the source file to
10 * dst_offset in the destination file. Uses copy_file_range(2) for
11 * same-filesystem copies; falls back to pread/pwrite for cross-filesystem.
12 *
13 * WHAT: Implements kXR_clone (3032), a server-side range-copy operation that copies multiple byte ranges
14 * from one source file into a single destination file in a single protocol round-trip.
15 *
16 * WHY: Clone avoids client-server data transfer — the copy happens entirely on the server using zero-copy
17 * syscalls when possible. The batched wire format lets clients specify arbitrary source ranges and
18 * destination offsets without multiple individual requests, reducing latency and network bandwidth.
19 *
20 * HOW: Parse clone_item array from payload (32 bytes each), validate dst_fhandle for write access and each
21 * src_fhandle for read access via brix_validate_write_handle/brix_validate_read_handle, decode big-endian
22 * uint64 fields (src_offset, src_len, dst_offset) with be64toh, iterate items calling brix_copy_range() for
23 * each (which uses copy_file_range when same filesystem or pread/pwrite fallback otherwise), skip zero-length
24 * items silently, accumulate total_bytes into file.bytes_written and session_bytes counters, return kXR_OK with
25 * byte count via BRIX_RETURN_OK.
26 */

53

55 BRIX_OP_CLONE, &rc)) {
56 return rc;
57 }
58
59 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
60 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CLONE, "CLONE",
61 ctx->files[dst_idx].path, "-",

63 }
64
65 if (ctx->recv.cur_dlen % CLONE_ITEM_LEN != 0) {
66 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CLONE, "CLONE",

68 kXR_ArgInvalid, "malformed clone list");
69 }
70

72 if (n_items > CLONE_MAX_ITEMS) {
73 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CLONE, "CLONE",
74 ctx->files[dst_idx].path, "-",
75 kXR_ArgTooLong, "too many clone items");
76 }
77
78 dst_fd = ctx->files[dst_idx].fd;
79
80 p = ctx->recv.payload;
81 end = ctx->recv.payload + ctx->recv.cur_dlen;
82
83 while (p < end) {

85 int src_idx;

88 size_t copy_len;
89
90 src_idx = (int)(unsigned char) item->src_fhandle[0];
91
92 if (!brix_validate_read_handle(ctx, c, src_idx, "CLONE",

95 }
96
97 /* All fields are big-endian uint64. */
98 ngx_memcpy(&src_off_raw, &item->src_offset, 8);
99 ngx_memcpy(&src_len_raw, &item->src_len, 8);
100 ngx_memcpy(&dst_off_raw, &item->dst_offset, 8);
101
102 src_off_raw = be64toh(src_off_raw);
103 src_len_raw = be64toh(src_len_raw);
104 dst_off_raw = be64toh(dst_off_raw);
105
106 src_off = (off_t) src_off_raw;
107 dst_off = (off_t) dst_off_raw;
108 copy_len = (size_t) src_len_raw;
109
110 if (copy_len == 0) {
111 p += CLONE_ITEM_LEN;
112 continue;
113 }
114
115 if (brix_copy_range(c->log, ctx->files[src_idx].fd, src_off,
116 dst_fd, dst_off, copy_len,
117 ctx->files[src_idx].path,
118 ctx->files[dst_idx].path) != NGX_OK)
119 {
120 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CLONE, "CLONE",
121 ctx->files[src_idx].path,
122 ctx->files[dst_idx].path,
123 kXR_IOError, "clone copy failed");
124 }
125
126 /* Integrity: clone writes bytes with copy_file_range/pread-pwrite,
127 * bypassing the write path's per-block CRC fold. A csi-tracked dst
128 * would otherwise keep its pre-clone block CRCs, and a later read of
129 * the (now different) data fails verification with EIO. Fold the copied
130 * region into the handle's csi engine so the record flushed at close
131 * carries the cloned data's CRCs (edge blocks are recomputed at flush). */
132 if (ctx->files[dst_idx].csi != NULL) {
133 enum { CLONE_FOLD_WIN = 1 << 20 }; /* one csi block */
134 u_char *fb = ngx_alloc(CLONE_FOLD_WIN, c->log);
135 if (fb != NULL) {
136 off_t fo = dst_off;
137 size_t left = copy_len;
138 while (left > 0) {
139 size_t chunk = left < CLONE_FOLD_WIN ? left : CLONE_FOLD_WIN;
140 ssize_t n = pread(dst_fd, fb, chunk, fo);
141 if (n <= 0) {
142 break;
143 }
144 (void) brix_csi_write_update(
145 (brix_csi_t *) ctx->files[dst_idx].csi, fb, fo,
146 (size_t) n);
147 fo += n;
148 left -= (size_t) n;
149 }
150 ngx_free(fb);
151 }
152 }
153
154 total_bytes += copy_len;
155 ctx->files[dst_idx].bytes_written += copy_len;
156 ctx->totals.bytes += copy_len;
157
158 p += CLONE_ITEM_LEN;
159 }
160
161 BRIX_RETURN_OK(ctx, c, BRIX_OP_CLONE, "CLONE",
162 ctx->files[dst_idx].path, "-", (size_t) total_bytes);
163}
