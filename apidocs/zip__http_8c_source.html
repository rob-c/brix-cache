23static int
24zip_http_name_ok(const char *out, size_t n)
25{
26 if (n == 0 || out[0] == '/' || ngx_strcmp(out, "..") == 0) {
27 return 0;
28 }
29 if (n >= 3 && out[0] == '.' && out[1] == '.' && out[2] == '/') {
30 return 0;
31 }
32 if (strstr(out, "/../") != NULL) {
33 return 0;
34 }
35 if (n >= 3 && ngx_strcmp(out + n - 3, "/..") == 0) {
36 return 0;
37 }
38 return 1;
39}

86/*
87 * WHAT: open the archive through the confined VFS read surface and verify it
88 * is a regular file.
89 * WHY: the member fd is used transiently to read the central directory and
90 * extract/dup the member; every failure maps to a distinct HTTP status.
91 * HOW: brix_vfs_open() under a WEBDAV read-only ctx, then fstat/S_ISREG.
92 * Returns NGX_OK with *fhp and *sizep set, or an HTTP status (fh
93 * already closed).
94 */
95static ngx_int_t
96zip_http_open_archive(ngx_http_request_t *r, const char *root_canon,
97 const char *archive_full, brix_vfs_file_t **fhp, off_t *sizep)
98{
99 struct stat ast;
100 brix_vfs_ctx_t vctx;
101 brix_vfs_file_t *fh;
102 int vfs_err = 0;
103 int is_tls = 0;
104
105#if (NGX_HTTP_SSL)
106 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
107#endif
108 /* triage (review-finding fixes, 2026-07-08): this IS a remote-capable
109 * data-plane read (brix_vfs_open below can reach a driver backend), so it
110 * belongs on the per-user-credential list in principle. Left NULL/unbound:
111 * brix_zip_http_serve() takes only `ngx_http_request_t *r` — it has no
112 * identity parameter, and both call sites (src/protocols/s3/object.c,
113 * src/protocols/webdav/get.c) are HTTP-plane and outside this task's
114 * touched-file scope, so threading identity here would require a public
115 * signature change reaching two more files. Deferred, not a same-file fix. */
116 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
117 root_canon, NULL, 0 /* allow_write */, is_tls, NULL, archive_full);
118 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
119 if (fh == NULL) {
120 errno = vfs_err;
121 if (errno == ENOENT || errno == ENOTDIR) return NGX_HTTP_NOT_FOUND;
122 if (errno == EACCES || errno == EPERM) return NGX_HTTP_FORBIDDEN;
123 return NGX_HTTP_INTERNAL_SERVER_ERROR;
124 }
125 if (fstat(brix_vfs_file_fd(fh), &ast) != 0 || !S_ISREG(ast.st_mode)) {
126 (void) brix_vfs_close(fh, r->connection->log);
127 return NGX_HTTP_INTERNAL_SERVER_ERROR;
128 }
129
130 *fhp = fh;
131 *sizep = (off_t) ast.st_size;
132 return NGX_OK;
133}

135/*
136 * WHAT: locate `member` in the archive's central directory.
137 * WHY: isolates the CD walk and the brix_zip_* → HTTP status mapping.
138 * HOW: brix_zip_find_member() over the open archive fd; NOMEMBER → 404,
139 * EIO → 500, anything else (unsupported feature) → 501. Returns NGX_OK
140 * with *m filled; the caller owns the archive handle on failure.

142static ngx_int_t
143zip_http_locate_member(brix_vfs_file_t *fh, off_t archive_size,
144 const char *member, size_t cd_max, brix_zip_member_t *m)
145{
146 int zrc;
147
148 zrc = brix_zip_find_member(brix_vfs_file_fd(fh), archive_size, member,
149 cd_max, m);
150 if (zrc == BRIX_ZIP_OK) {
151 return NGX_OK;
152 }

158/*
159 * WHAT: apply a single client byte Range against the member's UNCOMPRESSED
160 * length.
161 * WHY: ZIP members are ranged on their logical (inflated) content, not the
162 * archive bytes, so the standard range filter cannot be used.
163 * HOW: parses "bytes=a-[b]" / "bytes=-suffix" from the Range header into
164 * sp->start/sp->len (pre-set to the whole span). Returns NGX_OK, or
165 * NGX_ERROR when the range is unsatisfiable (start beyond total).
166 */
167static ngx_int_t
168zip_http_parse_range(ngx_http_request_t *r, zip_http_span_t *sp)
169{
170 ngx_table_elt_t *range = r->headers_in.range;
171 u_char *p, *eov, *dash;
172
173 if (range == NULL || range->value.len <= 7
174 || ngx_strncasecmp(range->value.data, (u_char *) "bytes=", 6) != 0)
175 {
176 return NGX_OK;
177 }
178
179 p = range->value.data + 6;
180 eov = range->value.data + range->value.len;
181 dash = (u_char *) ngx_strlchr(p, eov, '-');
182 if (dash == NULL) {
183 return NGX_OK;
184 }
185
186 {
187 off_t a = (dash > p) ? ngx_atoof(p, dash - p) : -1;
188 off_t b2 = (dash + 1 < eov) ? ngx_atoof(dash + 1, eov - dash - 1)
189 : -1;
190 if (a >= 0) { /* bytes=a-[b] */
191 sp->start = a;
192 sp->len = (b2 >= 0 && b2 >= a) ? (b2 - a + 1) : (sp->total - a);
193 } else if (b2 > 0) { /* bytes=-suffix */
194 sp->start = (b2 < sp->total) ? sp->total - b2 : 0;
195 sp->len = sp->total - sp->start;
196 }
197 }
198 if (sp->start >= sp->total) {
199 return NGX_ERROR;
200 }
201 if (sp->start + sp->len > sp->total) {
202 sp->len = sp->total - sp->start;
203 }
204 return NGX_OK;
205}

207/*
208 * WHAT: stage a STORE member for zero-copy sendfile over the archive range.
209 * WHY: stored members need no inflation — the response body IS a byte range
210 * of the archive file, so a dup'd fd + file-backed buf is cheapest.
211 * HOW: dup the archive fd (kept alive across the async send by a pool
212 * cleanup), close the VFS handle, point b at data_off + span. Always
213 * consumes fh; returns NGX_OK or 500.
214 */
215static ngx_int_t
216zip_http_stage_stored(ngx_http_request_t *r, brix_vfs_file_t *fh,
217 const brix_zip_member_t *m, const zip_http_span_t *sp, ngx_buf_t *b)
218{
219 ngx_file_t *f = ngx_pcalloc(r->pool, sizeof(ngx_file_t));
220 ngx_pool_cleanup_t *cln = ngx_pool_cleanup_add(r->pool, sizeof(ngx_fd_t));
221 int send_fd;
222
223 send_fd = dup(brix_vfs_file_fd(fh));
224 (void) brix_vfs_close(fh, r->connection->log);
225 if (f == NULL || cln == NULL || send_fd == (ngx_fd_t) -1) {
226 if (send_fd != (ngx_fd_t) -1) (void) close(send_fd);
227 return NGX_HTTP_INTERNAL_SERVER_ERROR;
228 }
229 cln->handler = zip_http_close_fd;
230 *(ngx_fd_t *) cln->data = send_fd;
231 f->fd = send_fd;
232 f->log = r->connection->log;
233 b->in_file = 1;
234 b->file = f;
235 b->file_pos = (off_t) m->data_off + sp->start;
236 b->file_last = (off_t) m->data_off + sp->start + sp->len;
237 return NGX_OK;
238}

240/*
241 * WHAT: stage a DEFLATE member by inflating it into an r->pool buffer.
242 * WHY: compressed members cannot be ranged/sent from the archive bytes;
243 * they are fully inflated (capped at ZIP_HTTP_MEM_MAX → 413) and served
244 * from memory.
245 * HOW: ngx_palloc + brix_zip_extract_full(), then point b at the requested
246 * span of the inflated data. Always consumes fh; returns NGX_OK,
247 * 413 (too large) or 500.
248 */
249static ngx_int_t
250zip_http_stage_deflate(ngx_http_request_t *r, brix_vfs_file_t *fh,
251 const brix_zip_member_t *m, const zip_http_span_t *sp, ngx_buf_t *b)
252{
253 u_char *mem;
254
255 if (m->uncomp_size > ZIP_HTTP_MEM_MAX) {
256 (void) brix_vfs_close(fh, r->connection->log);
257 return NGX_HTTP_REQUEST_ENTITY_TOO_LARGE;
258 }
259 mem = ngx_palloc(r->pool, m->uncomp_size ? m->uncomp_size : 1);
260 if (mem == NULL
261 || brix_zip_extract_full(brix_vfs_file_fd(fh), m, mem,
262 m->uncomp_size) < 0)
263 {
264 (void) brix_vfs_close(fh, r->connection->log);
265 return NGX_HTTP_INTERNAL_SERVER_ERROR;
266 }
267 (void) brix_vfs_close(fh, r->connection->log);
268 b->memory = 1;
269 b->pos = mem + sp->start;
270 b->last = mem + sp->start + sp->len;
271 return NGX_OK;
272}

274/*
275 * WHAT: fill the response headers for the (possibly partial) member body.
276 * WHY: 200 vs 206 + Content-Range depend only on the served span, so the
277 * header block is a pure function of (r, span).
278 * HOW: sets status/content_length/content_type; on a partial span appends a
279 * "Content-Range: bytes a-b/total" header (allocation failures leave
280 * the header out, matching the original best-effort behaviour).
281 */
282static void
283zip_http_set_headers(ngx_http_request_t *r, const zip_http_span_t *sp)
284{
285 r->headers_out.status = (sp->len == sp->total)
286 ? NGX_HTTP_OK
287 : NGX_HTTP_PARTIAL_CONTENT;
288 r->headers_out.content_length_n = sp->len;
289 r->allow_ranges = 1;
290 ngx_str_set(&r->headers_out.content_type, "application/octet-stream");
291 r->headers_out.content_type_len = r->headers_out.content_type.len;
292
293 if (sp->len != sp->total) {
294 ngx_table_elt_t *cr = ngx_list_push(&r->headers_out.headers);
295 if (cr != NULL) {
296 cr->hash = 1;
297 ngx_str_set(&cr->key, "Content-Range");
298 cr->value.data = ngx_pnalloc(r->pool,
299 sizeof("bytes -/-") + 3 * NGX_OFF_T_LEN);
300 if (cr->value.data != NULL) {
301 cr->value.len = ngx_sprintf(cr->value.data, "bytes %O-%O/%O",
302 sp->start,
303 sp->start + sp->len - 1,
304 sp->total)
305 - cr->value.data;
306 }
307 }
308 }

311ngx_int_t
312brix_zip_http_serve(ngx_http_request_t *r, const char *root_canon,
313 size_t cd_max, const char *archive_full, const char *member)
314{
315 brix_zip_member_t m;
316 zip_http_span_t sp;
317 ngx_buf_t *b;
318 ngx_chain_t out;
319 ngx_int_t rc;
320 brix_vfs_file_t *fh = NULL;
321 off_t archive_size = 0;
322
323 rc = zip_http_open_archive(r, root_canon, archive_full, &fh,
324 &archive_size);
325 if (rc != NGX_OK) {
326 return rc;
327 }
328
329 if (cd_max == 0) {
330 cd_max = 16 * 1024 * 1024;
331 }
332 rc = zip_http_locate_member(fh, archive_size, member, cd_max, &m);
333 if (rc != NGX_OK) {
334 (void) brix_vfs_close(fh, r->connection->log);
335 return rc;
336 }
337
338 /* Single byte-range against the UNCOMPRESSED length. */
339 sp.total = (off_t) m.uncomp_size;
340 sp.start = 0;
341 sp.len = sp.total;
342 if (zip_http_parse_range(r, &sp) != NGX_OK) {
343 (void) brix_vfs_close(fh, r->connection->log);
344 r->headers_out.status = NGX_HTTP_RANGE_NOT_SATISFIABLE;
345 r->headers_out.content_length_n = 0;
346 ngx_http_send_header(r);
347 return ngx_http_send_special(r, NGX_HTTP_LAST);
348 }
349
350 b = ngx_pcalloc(r->pool, sizeof(ngx_buf_t));
351 if (b == NULL) {
352 (void) brix_vfs_close(fh, r->connection->log);
353 return NGX_HTTP_INTERNAL_SERVER_ERROR;
354 }
355
356 rc = (m.method == BRIX_ZIP_METHOD_STORE)
357 ? zip_http_stage_stored(r, fh, &m, &sp, b)
358 : zip_http_stage_deflate(r, fh, &m, &sp, b);
359 if (rc != NGX_OK) {
360 return rc;
361 }
362 b->last_buf = 1;
363 b->last_in_chain = 1;
364
365 zip_http_set_headers(r, &sp);
366
367 rc = ngx_http_send_header(r);
368 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
369 return rc;
370 }
371 out.buf = b;
372 out.next = NULL;
373 return ngx_http_output_filter(r, &out);
374}
