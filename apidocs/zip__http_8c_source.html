1/*
2 * zip_http.c — shared HTTP ZIP-member serving (phase-57 W2). See zip_http.h.
3 *
4 * Read-only. Stored members are served zero-copy by sendfile over the archive
5 * byte range; deflate members are inflated into an r->pool buffer (capped) and
6 * served from memory. A single byte Range is honoured against the member's
7 * UNCOMPRESSED length. Shared by WebDAV GET and S3 GetObject.

23static int
24zip_http_name_ok(const char *out, size_t n)
25{
26 if (n == 0 || out[0] == '/' || ngx_strcmp(out, "..") == 0) {
27 return 0;
28 }
29 if (n >= 3 && out[0] == '.' && out[1] == '.' && out[2] == '/') {
30 return 0;
31 }
32 if (strstr(out, "/../") != NULL) {
33 return 0;
34 }
35 if (n >= 3 && ngx_strcmp(out + n - 3, "/..") == 0) {
36 return 0;
37 }
38 return 1;
39}

62 *d = '\0';
63 n = (size_t) ((char *) d - out);
64
65 return zip_http_name_ok(out, n) ? 1 : -1;

78ngx_int_t
79brix_zip_http_serve(ngx_http_request_t *r, const char *root_canon,
80 size_t cd_max, const char *archive_full, const char *member)
81{
82 int fd, send_fd, zrc;
83 struct stat ast;
84 brix_zip_member_t m;
85 off_t total, start, len;
86 ngx_table_elt_t *range;
87 ngx_buf_t *b;
88 ngx_chain_t out;
89 ngx_int_t rc;
90 brix_vfs_ctx_t vctx;
91 brix_vfs_file_t *fh;
92 int vfs_err = 0;
93 int is_tls = 0;
94
95 /* Open the archive through the confined VFS read surface (the member fd is
96 * used transiently to read the central directory + extract/dup the member). */
97#if (NGX_HTTP_SSL)
98 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
99#endif
100 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_WEBDAV,
101 root_canon, NULL, 0 /* allow_write */, is_tls, NULL, archive_full);
102 fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &vfs_err);
103 if (fh == NULL) {
104 errno = vfs_err;
105 if (errno == ENOENT || errno == ENOTDIR) return NGX_HTTP_NOT_FOUND;
106 if (errno == EACCES || errno == EPERM) return NGX_HTTP_FORBIDDEN;
107 return NGX_HTTP_INTERNAL_SERVER_ERROR;
108 }
109 fd = brix_vfs_file_fd(fh);
110 if (fstat(fd, &ast) != 0 || !S_ISREG(ast.st_mode)) {
111 (void) brix_vfs_close(fh, r->connection->log);
112 return NGX_HTTP_INTERNAL_SERVER_ERROR;
113 }
114
115 if (cd_max == 0) {
116 cd_max = 16 * 1024 * 1024;
117 }
118 zrc = brix_zip_find_member(fd, (off_t) ast.st_size, member, cd_max, &m);
119 if (zrc != BRIX_ZIP_OK) {
120 (void) brix_vfs_close(fh, r->connection->log);
121 if (zrc == BRIX_ZIP_NOMEMBER) return NGX_HTTP_NOT_FOUND;
122 return (zrc == BRIX_ZIP_EIO) ? NGX_HTTP_INTERNAL_SERVER_ERROR
123 : NGX_HTTP_NOT_IMPLEMENTED;
124 }
125
126 /* Single byte-range against the UNCOMPRESSED length. */
127 total = (off_t) m.uncomp_size;
128 start = 0;
129 len = total;
130 range = r->headers_in.range;
131 if (range != NULL && range->value.len > 7
132 && ngx_strncasecmp(range->value.data, (u_char *) "bytes=", 6) == 0)
133 {
134 u_char *p = range->value.data + 6;
135 u_char *eov = range->value.data + range->value.len;
136 u_char *dash = (u_char *) ngx_strlchr(p, eov, '-');
137 if (dash != NULL) {
138 off_t a = (dash > p) ? ngx_atoof(p, dash - p) : -1;
139 off_t b2 = (dash + 1 < eov) ? ngx_atoof(dash + 1, eov - dash - 1)
140 : -1;
141 if (a >= 0) { /* bytes=a-[b] */
142 start = a;
143 len = (b2 >= 0 && b2 >= a) ? (b2 - a + 1) : (total - a);
144 } else if (b2 > 0) { /* bytes=-suffix */
145 start = (b2 < total) ? total - b2 : 0;
146 len = total - start;
147 }
148 if (start >= total) {
149 (void) brix_vfs_close(fh, r->connection->log);
150 r->headers_out.status = NGX_HTTP_RANGE_NOT_SATISFIABLE;
151 r->headers_out.content_length_n = 0;
152 ngx_http_send_header(r);
153 return ngx_http_send_special(r, NGX_HTTP_LAST);
154 }
155 if (start + len > total) {
156 len = total - start;
157 }
158 }
159 }
160
161 b = ngx_pcalloc(r->pool, sizeof(ngx_buf_t));
162 if (b == NULL) {
163 (void) brix_vfs_close(fh, r->connection->log);
164 return NGX_HTTP_INTERNAL_SERVER_ERROR;
165 }
166
167 if (m.method == BRIX_ZIP_METHOD_STORE) {
168 /* Zero-copy sendfile over the member's byte range; keep the fd alive
169 * (async send) via a pool cleanup. */
170 ngx_file_t *f = ngx_pcalloc(r->pool, sizeof(ngx_file_t));
171 ngx_pool_cleanup_t *cln = ngx_pool_cleanup_add(r->pool, sizeof(ngx_fd_t));
172 send_fd = dup(fd);
173 (void) brix_vfs_close(fh, r->connection->log);
174 if (f == NULL || cln == NULL || send_fd == (ngx_fd_t) -1) {
175 if (send_fd != (ngx_fd_t) -1) (void) close(send_fd);
176 return NGX_HTTP_INTERNAL_SERVER_ERROR;
177 }
178 cln->handler = zip_http_close_fd;
179 *(ngx_fd_t *) cln->data = send_fd;
180 f->fd = send_fd;
181 f->log = r->connection->log;
182 b->in_file = 1;
183 b->file = f;
184 b->file_pos = (off_t) m.data_off + start;
185 b->file_last = (off_t) m.data_off + start + len;
186 } else {
187 u_char *mem;
188 if (m.uncomp_size > ZIP_HTTP_MEM_MAX) {
189 (void) brix_vfs_close(fh, r->connection->log);
190 return NGX_HTTP_REQUEST_ENTITY_TOO_LARGE;
191 }
192 mem = ngx_palloc(r->pool, m.uncomp_size ? m.uncomp_size : 1);
193 if (mem == NULL
194 || brix_zip_extract_full(fd, &m, mem, m.uncomp_size) < 0)
195 {
196 (void) brix_vfs_close(fh, r->connection->log);
197 return NGX_HTTP_INTERNAL_SERVER_ERROR;
198 }
199 (void) brix_vfs_close(fh, r->connection->log);
200 b->memory = 1;
201 b->pos = mem + start;
202 b->last = mem + start + len;
203 }
204 b->last_buf = 1;
205 b->last_in_chain = 1;
206
207 r->headers_out.status = (len == total) ? NGX_HTTP_OK
208 : NGX_HTTP_PARTIAL_CONTENT;
209 r->headers_out.content_length_n = len;
210 r->allow_ranges = 1;
211 ngx_str_set(&r->headers_out.content_type, "application/octet-stream");
212 r->headers_out.content_type_len = r->headers_out.content_type.len;
213
214 if (len != total) {
215 ngx_table_elt_t *cr = ngx_list_push(&r->headers_out.headers);
216 if (cr != NULL) {
217 cr->hash = 1;
218 ngx_str_set(&cr->key, "Content-Range");
219 cr->value.data = ngx_pnalloc(r->pool,
220 sizeof("bytes -/-") + 3 * NGX_OFF_T_LEN);
221 if (cr->value.data != NULL) {
222 cr->value.len = ngx_sprintf(cr->value.data, "bytes %O-%O/%O",
223 start, start + len - 1, total)
224 - cr->value.data;
225 }
226 }
227 }
228
229 rc = ngx_http_send_header(r);
230 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
231 return rc;
232 }
233 out.buf = b;
234 out.next = NULL;
235 return ngx_http_output_filter(r, &out);
236}
