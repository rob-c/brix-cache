1/*
2 * xrdsssadmin.c — manage an SSS (Simple Shared Secret) keytab.
3 *
4 * WHAT: `xrdsssadmin [-k keytab] <add|list|del|install>` — create and maintain
5 * the symmetric-key keytab that both an XRootD server and the native client
6 * use for SSS authentication.
7 * WHY: SSS needs a shared keytab on both ends; this is the clean-room, libXrdCl-
8 * free tool to mint a random key, list entries, and delete one — mode 0600,
9 * self-validated by re-reading after every mutation.
10 * HOW: Thin arg parse → lib/sss_keytab.c read/write. `add`/`install` generate a
11 * 32-byte RAND_bytes key with id = max+1 (or --id), append, and write back;
12 * `list` prints id/user/group/name/keylen/expiry; `del --id N` removes one.
13 *
14 * Clean-room: keytab grammar matches src/auth/sss/config.c; no XrdSecsssAdmin code.
15 */

57{
58 brix_status st;

71static int
72cmd_add(const char *path, const char *user, const char *group, const char *name,
73 int64_t want_id, int lifetime_days, int keylen)
74{
75 brix_sss_key keys[XRDC_SSS_KEYS_MAX];
76 brix_sss_key chk[XRDC_SSS_KEYS_MAX];
77 brix_status st;
78 int n = 0, i, m = 0;
79 int64_t maxid = 0;
80 char hostname[128];
81
82 if (load_or_empty(path, keys, XRDC_SSS_KEYS_MAX, &n) != 0) {
83 return 1;
84 }
85 if (n >= XRDC_SSS_KEYS_MAX) {
86 fprintf(stderr, "xrdsssadmin: keytab full (%d keys)\n", n);
87 return 1;
88 }
89 if (keylen <= 0 || keylen > XRDC_SSS_KEY_MAX) {
90 keylen = 32;
91 }
92 for (i = 0; i < n; i++) {
93 if (keys[i].id > maxid) {
94 maxid = keys[i].id;
95 }
96 }
97
98 {
99 brix_sss_key *k = &keys[n];
100 memset(k, 0, sizeof(*k));
101 k->id = (want_id >= 0) ? want_id : maxid + 1;
102 k->key_len = (size_t) keylen;
103 if (RAND_bytes(k->key, keylen) != 1) {
104 fprintf(stderr, "xrdsssadmin: RAND_bytes failed\n");
105 return 1;
106 }
107 snprintf(k->user, sizeof(k->user), "%s", user ? user : "anybody");
108 snprintf(k->group, sizeof(k->group), "%s", group ? group : "anygroup");
109 if (name != NULL) {
110 snprintf(k->name, sizeof(k->name), "%s", name);
111 } else {
112 if (gethostname(hostname, sizeof(hostname)) != 0) {
113 snprintf(hostname, sizeof(hostname), "localhost");
114 }
115 snprintf(k->name, sizeof(k->name), "%s", hostname);
116 }
117 k->exp = (lifetime_days > 0)
118 ? (int64_t) time(NULL) + (int64_t) lifetime_days * 86400
119 : 0;
120 n++;
121 }
122
123 brix_status_clear(&st);
124 if (brix_sss_keytab_write(path, keys, n, &st) != 0) {
125 fprintf(stderr, "xrdsssadmin: %s\n", st.msg);
126 return 1;
127 }
128 /* Self-validate: re-read and confirm the new id is present. */
129 brix_status_clear(&st);
130 if (brix_sss_keytab_read(path, chk, XRDC_SSS_KEYS_MAX, &m, &st) != 0) {
131 fprintf(stderr, "xrdsssadmin: re-read failed: %s\n", st.msg);
132 return 1;
133 }
134 printf("added key id=%lld (%d-byte) to %s (%d keys)\n",
135 (long long) keys[n - 1].id, keylen, path, m);
136 return 0;
137}

139static int
140cmd_list(const char *path)
141{
142 brix_sss_key keys[XRDC_SSS_KEYS_MAX];
143 brix_status st;
144 int n = 0, i;
145
146 brix_status_clear(&st);
147 if (brix_sss_keytab_read(path, keys, XRDC_SSS_KEYS_MAX, &n, &st) != 0) {
148 fprintf(stderr, "xrdsssadmin: %s\n", st.msg);
149 return 1;
150 }
151 printf("%s: %d key(s)\n", path, n);
152 for (i = 0; i < n; i++) {
153 printf(" id=%-4lld user=%-10s group=%-10s name=%-16s keylen=%zu",

156 if (keys[i].exp != 0) {
157 printf(" exp=%lld", (long long) keys[i].exp);
158 }
159 printf("\n");
160 }
161 return 0;

164static int
165cmd_del(const char *path, int64_t id)
166{
167 brix_sss_key keys[XRDC_SSS_KEYS_MAX];
168 brix_status st;
169 int n = 0, i, out = 0, removed = 0;
170
171 if (id < 0) {
172 fprintf(stderr, "xrdsssadmin: del requires --id N\n");
173 return 1;
174 }
175 brix_status_clear(&st);
176 if (brix_sss_keytab_read(path, keys, XRDC_SSS_KEYS_MAX, &n, &st) != 0) {
177 fprintf(stderr, "xrdsssadmin: %s\n", st.msg);
178 return 1;
179 }
180 for (i = 0; i < n; i++) {
181 if (keys[i].id == id) {
182 removed++;
183 continue;
184 }
185 if (out != i) {
186 keys[out] = keys[i];
187 }
188 out++;
189 }
190 if (removed == 0) {
191 fprintf(stderr, "xrdsssadmin: no key with id=%lld\n", (long long) id);
192 return 1;
193 }
194 brix_status_clear(&st);
195 if (brix_sss_keytab_write(path, keys, out, &st) != 0) {
196 fprintf(stderr, "xrdsssadmin: %s\n", st.msg);
197 return 1;
198 }
199 printf("removed %d key(s) with id=%lld; %d remain in %s\n",
200 removed, (long long) id, out, path);
201 return 0;
202}

204int
205main(int argc, char **argv)
206{
207 char kt[XRDC_PATH_MAX];
208 const char *keytab = NULL;
209 const char *cmd = NULL;
210 const char *user = NULL, *group = NULL, *name = NULL;
211 int64_t id = -1;
212 int lifetime_days = 0, keylen = 32;
213 int i;
214
215 /* --help / --version before loop (not on shared parser). */
216 if (argc >= 2) {
217 if (strcmp(argv[1], "--version") == 0) {
218 printf("xrdsssadmin (BriX-Cache client) %s\n", brix_client_version());
219 return 0;
220 }
221 if (strcmp(argv[1], "--help") == 0) {
222 usage_fp(stdout);
223 return 0;
224 }
225 }
226
227 for (i = 1; i < argc; i++) {
228 const char *a = argv[i];
229 if ((strcmp(a, "-k") == 0 || strcmp(a, "--keytab") == 0) && i + 1 < argc) { keytab = argv[++i]; }
230 else if (strcmp(a, "--user") == 0 && i + 1 < argc) { user = argv[++i]; }
231 else if (strcmp(a, "--group") == 0 && i + 1 < argc) { group = argv[++i]; }
232 else if (strcmp(a, "--name") == 0 && i + 1 < argc) { name = argv[++i]; }
233 else if (strcmp(a, "--id") == 0 && i + 1 < argc) { id = strtoll(argv[++i], NULL, 10); }
234 else if (strcmp(a, "--lifetime") == 0 && i + 1 < argc) { lifetime_days = atoi(argv[++i]); }
235 else if (strcmp(a, "--keylen") == 0 && i + 1 < argc) { keylen = atoi(argv[++i]); }
236 else if (strcmp(a, "-h") == 0) { usage(); return 0; } /* C1 */
237 else if (a[0] != '-' && cmd == NULL) { cmd = a; }
238 else { fprintf(stderr, "xrdsssadmin: unexpected arg '%s'\n", a); usage(); return 2; }
239 }
240
241 if (cmd == NULL) {
242 usage();
243 return 2;
244 }
245 if (keytab == NULL) {
246 brix_sss_keytab_default(kt, sizeof(kt));
247 keytab = kt;
248 }
249
250 if (strcmp(cmd, "add") == 0 || strcmp(cmd, "install") == 0) {
251 return cmd_add(keytab, user, group, name, id, lifetime_days, keylen);
252 }
253 if (strcmp(cmd, "list") == 0) {
254 return cmd_list(keytab);
255 }
256 if (strcmp(cmd, "del") == 0) {
257 return cmd_del(keytab, id);
258 }
259 fprintf(stderr, "xrdsssadmin: unknown command '%s'\n", cmd);
260 usage();
261 return 2;
262}
