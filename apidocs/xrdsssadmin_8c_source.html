1/*
2 * xrdsssadmin.c — manage an SSS (Simple Shared Secret) keytab.
3 *
4 * WHAT: `xrdsssadmin-brix [-k keytab] <add|list|del|install>` — create and maintain
5 * the symmetric-key keytab that both an XRootD server and the native client
6 * use for SSS authentication.
7 * WHY: SSS needs a shared keytab on both ends; this is the clean-room, libXrdCl-
8 * free tool to mint a random key, list entries, and delete one — mode 0600,
9 * self-validated by re-reading after every mutation.
10 * HOW: Thin arg parse → lib/sss_keytab.c read/write. `add`/`install` generate a
11 * 32-byte RAND_bytes key with id = max+1 (or --id), append, and write back;
12 * `list` prints id/user/group/name/keylen/expiry; `del --id N` removes one.
13 *
14 * Clean-room: keytab grammar matches src/auth/sss/config.c; no XrdSecsssAdmin code.
15 */

72/*
73 * add_args_t — the parsed `add`/`install` request.
74 *
75 * WHAT: bundles the six user-supplied fields of a key-add into one value.
76 * WHY: collapses cmd_add's 7-parameter signature (path + six knobs) to two
77 * (path + this struct), matching the coding-standard param-count gate
78 * while keeping every field's meaning and default behavior unchanged.
79 * HOW: populated once in main() from argv, then passed by const pointer to
80 * cmd_add(); NULL/<=0 sentinels still select the documented defaults.

82typedef struct {

85 const char *name; /* NULL → local hostname */

87 int lifetime_days; /* <=0 → no expiry */
88 int keylen; /* <=0 or >max → 32 bytes */

91/*
92 * add_fill_entry — populate one keytab slot from an add request.
93 *
94 * WHAT: fills *k (id/key/user/group/name/exp) for a new key of keylen bytes.
95 * WHY: isolates the pure per-entry construction (defaults + RAND_bytes) from
96 * cmd_add's file I/O, keeping each function single-purpose and cutting
97 * the orchestrator's branch count.
98 * HOW: zero-init, resolve id/defaults, mint a random key; returns 0 on
99 * success or -1 if RAND_bytes fails (message already printed).
100 */
101static int
102add_fill_entry(brix_sss_key *k, const add_args_t *a, int64_t maxid, int keylen)
103{
104 char hostname[128];
105
106 memset(k, 0, sizeof(*k));
107 k->id = (a->want_id >= 0) ? a->want_id : maxid + 1;
108 k->key_len = (size_t) keylen;
109 if (RAND_bytes(k->key, keylen) != 1) {
110 fprintf(stderr, "xrdsssadmin-brix: RAND_bytes failed\n");
111 return -1;
112 }
113 snprintf(k->user, sizeof(k->user), "%s", a->user ? a->user : "anybody");
114 snprintf(k->group, sizeof(k->group), "%s", a->group ? a->group : "anygroup");
115 if (a->name != NULL) {
116 snprintf(k->name, sizeof(k->name), "%s", a->name);
117 } else {
118 if (gethostname(hostname, sizeof(hostname)) != 0) {
119 snprintf(hostname, sizeof(hostname), "localhost");
120 }
121 snprintf(k->name, sizeof(k->name), "%s", hostname);
122 }
123 k->exp = (a->lifetime_days > 0)
124 ? (int64_t) time(NULL) + (int64_t) a->lifetime_days * 86400
125 : 0;
126 return 0;
127}

129/*
130 * cmd_add — mint and append a new random SSS key, then self-validate.
131 *
132 * WHAT: loads the keytab, appends one freshly generated key, writes it back,
133 * and re-reads to confirm the mutation landed.
134 * WHY: the `add`/`install` subcommand; delegates per-entry construction to
135 * add_fill_entry so this function is just the load/append/write/verify
136 * orchestration.
137 * HOW: load-or-empty → capacity/keylen guards → compute max id → fill slot n
138 * → write → re-read; returns 0 on success, 1 on any error.
139 */
140static int
141cmd_add(const char *path, const add_args_t *a)
142{
143 brix_sss_key keys[XRDC_SSS_KEYS_MAX];
144 brix_sss_key chk[XRDC_SSS_KEYS_MAX];
145 brix_status st;
146 int n = 0, i, m = 0;
147 int keylen = a->keylen;
148 int64_t maxid = 0;
149
150 if (load_or_empty(path, keys, XRDC_SSS_KEYS_MAX, &n) != 0) {
151 return 1;
152 }
153 if (n >= XRDC_SSS_KEYS_MAX) {
154 fprintf(stderr, "xrdsssadmin-brix: keytab full (%d keys)\n", n);
155 return 1;
156 }
157 if (keylen <= 0 || keylen > XRDC_SSS_KEY_MAX) {
158 keylen = 32;
159 }
160 for (i = 0; i < n; i++) {
161 if (keys[i].id > maxid) {
162 maxid = keys[i].id;
163 }
164 }
165
166 if (add_fill_entry(&keys[n], a, maxid, keylen) != 0) {
167 return 1;
168 }
169 n++;
170
171 brix_status_clear(&st);
172 if (brix_sss_keytab_write(path, keys, n, &st) != 0) {
173 fprintf(stderr, "xrdsssadmin-brix: %s\n", st.msg);
174 return 1;
175 }
176 /* Self-validate: re-read and confirm the new id is present. */
177 brix_status_clear(&st);
178 if (brix_sss_keytab_read(path, chk, XRDC_SSS_KEYS_MAX, &m, &st) != 0) {
179 fprintf(stderr, "xrdsssadmin-brix: re-read failed: %s\n", st.msg);
180 return 1;
181 }
182 printf("added key id=%lld (%d-byte) to %s (%d keys)\n",
183 (long long) keys[n - 1].id, keylen, path, m);
184 return 0;
185}

212static int
213cmd_del(const char *path, int64_t id)
214{
215 brix_sss_key keys[XRDC_SSS_KEYS_MAX];
216 brix_status st;
217 int n = 0, i, out = 0, removed = 0;
218
219 if (id < 0) {
220 fprintf(stderr, "xrdsssadmin-brix: del requires --id N\n");
221 return 1;
222 }
223 brix_status_clear(&st);
224 if (brix_sss_keytab_read(path, keys, XRDC_SSS_KEYS_MAX, &n, &st) != 0) {
225 fprintf(stderr, "xrdsssadmin-brix: %s\n", st.msg);
226 return 1;
227 }
228 for (i = 0; i < n; i++) {
229 if (keys[i].id == id) {
230 removed++;
231 continue;
232 }
233 if (out != i) {
234 keys[out] = keys[i];
235 }
236 out++;
237 }
238 if (removed == 0) {
239 fprintf(stderr, "xrdsssadmin-brix: no key with id=%lld\n", (long long) id);
240 return 1;
241 }
242 brix_status_clear(&st);
243 if (brix_sss_keytab_write(path, keys, out, &st) != 0) {
244 fprintf(stderr, "xrdsssadmin-brix: %s\n", st.msg);
245 return 1;
246 }
247 printf("removed %d key(s) with id=%lld; %d remain in %s\n",
248 removed, (long long) id, out, path);
249 return 0;
250}

269/* Kind of value a value-taking flag consumes. */
270typedef enum { OPT_STR, OPT_I32, OPT_I64 } opt_kind;

272/*
273 * opt_spec — one value-taking flag: its two accepted spellings and where its
274 * value lands in a cli_opts_t.
275 *
276 * WHAT: descriptor row {long name, short/alias name, value kind, field offset}.
277 * WHY: expresses the flag ladder as data (coding-standard §8.6) so parse_opts
278 * is a single loop over the table instead of a per-flag if-chain.
279 * HOW: apply_opt() writes the parsed value at (char*)o + off according to
280 * kind; alt may be NULL when a flag has only one spelling.
281 */
282typedef struct {
283 const char *name;
284 const char *alt;
285 opt_kind kind;
286 size_t off;
287} opt_spec;

289static const opt_spec OPTS[] = {
290 { "--keytab", "-k", OPT_STR, offsetof(cli_opts_t, keytab) },
291 { "--user", NULL, OPT_STR, offsetof(cli_opts_t, add.user) },
292 { "--group", NULL, OPT_STR, offsetof(cli_opts_t, add.group) },
293 { "--name", NULL, OPT_STR, offsetof(cli_opts_t, add.name) },
294 { "--id", NULL, OPT_I64, offsetof(cli_opts_t, id) },
295 { "--lifetime", NULL, OPT_I32, offsetof(cli_opts_t, add.lifetime_days) },
296 { "--keylen", NULL, OPT_I32, offsetof(cli_opts_t, add.keylen) },

299/*
300 * apply_opt — store one value-taking flag's argument into *o.
301 *
302 * WHAT: parses val per s->kind and writes it at the struct field s->off.
303 * WHY: keeps the string/int/int64 conversion in one place so the parser loop
304 * stays flat and adding a flag is one table row.
305 * HOW: pointer-arithmetic to the target field, then the kind-appropriate
306 * conversion (identical strtoll/atoi calls as the original loop).
307 */
308static void
309apply_opt(cli_opts_t *o, const opt_spec *s, const char *val)
310{
311 char *base = (char *) o;
312
313 if (s->kind == OPT_STR) {
314 *(const char **) (base + s->off) = val;
315 } else if (s->kind == OPT_I64) {
316 *(int64_t *) (base + s->off) = strtoll(val, NULL, 10);
317 } else {
318 *(int *) (base + s->off) = atoi(val);
319 }
320}

322/*
323 * match_opt — find the value-taking flag matching token a, if any.
324 *
325 * WHAT: returns the OPTS row whose name or alt equals a, else NULL.
326 * WHY: isolates the table scan so parse_opts reads as flag/positional/error.
327 * HOW: linear scan comparing a against each row's two spellings.
328 */
329static const opt_spec *
330match_opt(const char *a)
331{
332 size_t i;
333
334 for (i = 0; i < sizeof(OPTS) / sizeof(OPTS[0]); i++) {
335 if (strcmp(a, OPTS[i].name) == 0
336 || (OPTS[i].alt != NULL && strcmp(a, OPTS[i].alt) == 0)) {
337 return &OPTS[i];
338 }
339 }
340 return NULL;
341}

343/*
344 * parse_opts — walk argv into a cli_opts_t.
345 *
346 * WHAT: fills *o from argv; returns 0 to continue, or a process exit code
347 * (0 for -h) when the caller should stop immediately.
348 * WHY: moves the option-scanning ladder out of main(), keeping the
349 * orchestrator flat; behavior (flags, arg-count checks, error text,
350 * exit codes) is identical to the original inline loop.
351 * HOW: for each token: a table-listed value flag with a following arg is
352 * applied via apply_opt; else -h prints usage and stops (exit 0); else
353 * the first bare word is the command; else usage + stop (exit 2). *stop
354 * is set non-zero when the returned code is a process exit code.
355 */
356static int
357parse_opts(int argc, char **argv, cli_opts_t *o, int *stop)
358{
359 int i;
360
361 *stop = 0;
362 for (i = 1; i < argc; i++) {
363 const char *a = argv[i];
364 const opt_spec *s = match_opt(a);
365
366 if (s != NULL && i + 1 < argc) {
367 apply_opt(o, s, argv[++i]);
368 } else if (strcmp(a, "-h") == 0) {
369 usage(); *stop = 1; return 0; /* C1 */
370 } else if (a[0] != '-' && o->cmd == NULL) {
371 o->cmd = a;
372 } else {
373 fprintf(stderr, "xrdsssadmin-brix: unexpected arg '%s'\n", a);
374 usage(); *stop = 1; return 2;
375 }
376 }
377 return 0;
378}

380/*
381 * dispatch_cmd — run the resolved subcommand.
382 *
383 * WHAT: matches o->cmd against the {add,install,list,del} table and calls the
384 * corresponding handler with the resolved keytab path.
385 * WHY: replaces main()'s trailing if-ladder with a table-driven lookup so the
386 * orchestrator stays flat and new subcommands are one row.
387 * HOW: linear scan of a static-const descriptor table; on no match print the
388 * unknown-command error + usage and return 2 (unchanged behavior).
389 */
390static int
391dispatch_cmd(const cli_opts_t *o, const char *keytab)
392{
393 add_args_t add = o->add;
394 add.want_id = o->id;
395
396 if (strcmp(o->cmd, "add") == 0 || strcmp(o->cmd, "install") == 0) {
397 return cmd_add(keytab, &add);
398 }
399 if (strcmp(o->cmd, "list") == 0) {
400 return cmd_list(keytab);
401 }
402 if (strcmp(o->cmd, "del") == 0) {
403 return cmd_del(keytab, o->id);
404 }
405 fprintf(stderr, "xrdsssadmin-brix: unknown command '%s'\n", o->cmd);
406 usage();
407 return 2;
408}

411main(int argc, char **argv)
412{
413 char kt[XRDC_PATH_MAX];
414 cli_opts_t o = { NULL, NULL, { NULL, NULL, NULL, -1, 0, 32 }, -1 };
415 int rc, stop = 0;
416
417 /* --help / --version before loop (not on shared parser). */

422 }
423 if (strcmp(argv[1], "--help") == 0) {

428

441 }
