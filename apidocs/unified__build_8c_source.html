32static ngx_uint_t
33brix_count_components_cstr(const char *path)
34{
35 const char *p;
36 ngx_uint_t count;
37
38 p = path;
39 while (*p == '/') {
40 p++;
41 }
42
43 if (*p == '\0') {
44 return 0;
45 }
46
47 count = 0;
48 while (*p != '\0') {
49 while (*p == '/') {
50 p++;
51 }
52 if (*p == '\0') {
53 break;
54 }
55
56 count++;
57 while (*p != '\0' && *p != '/') {
58 p++;
59 }
60 }
61
62 return count;
63}

65brix_path_status_t
66brix_validate_components_cstr(ngx_log_t *log, const char *path)
67{
68 const char *p, *seg_start;
69 size_t seg_len;
70
71 if (path == NULL) {
72 return BRIX_PATH_STATUS_INVALID;
73 }
74
75 if (strlen(path) > BRIX_MAX_PATH) {
76 return BRIX_PATH_STATUS_TOO_LONG;
77 }
78
79 if (brix_count_path_depth(path) != NGX_OK) {
80 if (log != NULL) {
81 ngx_log_error(NGX_LOG_WARN, log, 0,
82 "brix: path depth exceeds limit");
83 }
84 return BRIX_PATH_STATUS_INVALID;
85 }
86
87 p = path;
88 while (*p != '\0') {
89 while (*p == '/') {
90 p++;
91 }
92 if (*p == '\0') {
93 break;
94 }
95
96 seg_start = p;
97 while (*p != '\0' && *p != '/') {
98 p++;
99 }
100
101 seg_len = (size_t) (p - seg_start);
102 if (brix_path_component_forbidden(seg_start, seg_len)) {
103 brix_path_warn(log, "brix: path traversal attempt", path);
104 return BRIX_PATH_STATUS_INVALID;
105 }
106 }
107
108 return BRIX_PATH_STATUS_OK;

142static brix_path_status_t
143brix_append_component(char *path, size_t pathsz, size_t *len,
144 const char *component, size_t component_len)
145{
146 if (*len > 1) {
147 if (*len + 1 >= pathsz) {
148 return BRIX_PATH_STATUS_TOO_LONG;
149 }
150 path[(*len)++] = '/';
151 }
152
153 if (*len + component_len >= pathsz) {
154 return BRIX_PATH_STATUS_TOO_LONG;
155 }
156
157 ngx_memcpy(path + *len, component, component_len);
158 *len += component_len;
159 path[*len] = '\0';
160
161 return BRIX_PATH_STATUS_OK;
162}

164/*
165 * brix_candidate_init_root — seed the candidate buffer with the export root.
166 *
167 * WHAT: Copies root_canon into candidate, strips trailing slashes (keeping a
168 * lone "/"), null-terminates, and reports the resulting length via *len.
169 *
170 * WHY: The candidate always begins as the normalised export root before any
171 * request component is appended; isolating this keeps brix_build_candidate
172 * focused on component iteration.
173 *
174 * HOW: Bounds-checks root_len against candidatesz, memcpy's the root, then
175 * trims trailing '/' with the same >1 guard as the original inline code so
176 * "/" survives untouched.
177 */
178static brix_path_status_t
179brix_candidate_init_root(const char *root_canon, char *candidate,
180 size_t candidatesz, size_t *len)
181{
182 size_t root_len;
183
184 root_len = strlen(root_canon);
185 if (root_len == 0 || root_len >= candidatesz) {
186 return BRIX_PATH_STATUS_TOO_LONG;
187 }
188
189 ngx_memcpy(candidate, root_canon, root_len);
190 while (root_len > 1 && candidate[root_len - 1] == '/') {
191 root_len--;
192 }
193 candidate[root_len] = '\0';
194
195 *len = root_len;
196 return BRIX_PATH_STATUS_OK;
197}

199/*
200 * brix_candidate_append_req — append each request component onto the candidate.
201 *
202 * WHAT: Walks req_path segment by segment (collapsing runs of '/'), skipping
203 * empties, and appends each via brix_append_component starting at *len.
204 *
205 * WHY: Component joining is the sole variable-length stage of candidate
206 * building; extracting it drops brix_build_candidate below the complexity
207 * cap while preserving the exact join order and bounds behaviour.
208 *
209 * HOW: Early-returns any non-OK append status; *len is updated in place so the
210 * caller need not track buffer position.
211 */
212static brix_path_status_t
213brix_candidate_append_req(const char *req_path, char *candidate,
214 size_t candidatesz, size_t *len)
215{
216 const char *p, *seg_start;
217 size_t seg_len;
218 brix_path_status_t rc;
219
220 p = req_path;
221 while (*p != '\0') {
222 while (*p == '/') {
223 p++;
224 }
225 if (*p == '\0') {
226 break;
227 }
228
229 seg_start = p;
230 while (*p != '\0' && *p != '/') {
231 p++;
232 }
233
234 seg_len = (size_t) (p - seg_start);
235 if (seg_len == 0) {
236 continue;
237 }
238
239 rc = brix_append_component(candidate, candidatesz, len,
240 seg_start, seg_len);
241 if (rc != BRIX_PATH_STATUS_OK) {
242 return rc;
243 }
244 }
245
246 return BRIX_PATH_STATUS_OK;
247}

249/*
250 * brix_build_candidate — join the export root and request path into an
251 * un-canonicalised candidate filesystem path, reporting the request depth.
252 *
253 * WHAT: Validates inputs, seeds the buffer with the normalised root
254 * (brix_candidate_init_root), records the component depth, and — for a
255 * non-empty request — appends every component (brix_candidate_append_req).
256 * A zero-depth request resolves to the root only when opts.allow_root.
257 *
258 * WHY: This is the pre-realpath() assembly step; keeping the root-seed and
259 * component-append stages in dedicated helpers holds this coordinator
260 * under the complexity cap without altering the assembled path.
261 *
262 * HOW: Early-returns on invalid args / bound overflow; out->depth is always set
263 * when out->depth_out is non-NULL, matching the original contract.
264 */
265brix_path_status_t
266brix_build_candidate(const char *root_canon, const char *req_path,
267 brix_path_opts_t opts, brix_candidate_out_t *out)
268{
269 size_t len;
270 ngx_uint_t depth;
271 brix_path_status_t rc;
272
273 if (root_canon == NULL || root_canon[0] == '\0' || req_path == NULL) {
274 return BRIX_PATH_STATUS_INVALID;
275 }
276
277 rc = brix_candidate_init_root(root_canon, out->buf, out->bufsz, &len);
278 if (rc != BRIX_PATH_STATUS_OK) {
279 return rc;
280 }
281
282 depth = brix_count_components_cstr(req_path);
283 if (out->depth_out != NULL) {
284 *out->depth_out = depth;
285 }
286
287 if (depth == 0) {
288 if (!opts.allow_root) {
289 return BRIX_PATH_STATUS_INVALID;
290 }
291 return BRIX_PATH_STATUS_OK;
292 }
293
294 return brix_candidate_append_req(req_path, out->buf, out->bufsz, &len);
295}
