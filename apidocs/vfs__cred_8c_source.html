1/*
2 * vfs_cred.c — VFS per-user backend credential gate
3 * (Phase 1 + Phase 2 T1 + T3 + T9).
4 *
5 * WHAT: Implements the entry points that translate a request identity into a
6 * per-open / per-namespace backend credential:
7 * brix_vfs_ctx_bind_backend_cred() — wires the conf's credential directory
8 * and fallback policy onto a VFS ctx.
9 * brix_vfs_ctx_bind_backend_mint() — wires the conf's opt-in mint-CA
10 * cert/key/ttl onto a VFS ctx (Phase 2 Task 9).
11 * brix_vfs_backend_cred() — data-plane gate (open/staged_open), keyed on
12 * driver->open_cred capability.
13 * brix_vfs_ns_cred() — namespace gate (stat/unlink/mkdir/rename/…), keyed
14 * on driver->stat_cred capability (Phase 2 Task 1).
15 * Both gates share a single static decision body so the policy cannot drift.
16 * Phase 2 Task 3 adds per-proto Prometheus counters for each terminal outcome
17 * (USER / FALLBACK / DENY) via brix_metric_cred_result.
18 *
19 * WHY: The credential gate must execute BEFORE any driver→open or driver→ns
20 * call so a request can never reach a remote origin under the service
21 * credential when the operator has set fallback=deny and no valid user
22 * credential exists. A deny-mode probe (brix_vfs_probe) that runs the
23 * namespace gate closes the gap where the probe stat still reached the
24 * origin under the service credential (Phase-1 gap). Phase 2 Task 9
25 * closes a different gap: identities with no NATIVE x509 (S3 access
26 * keys, WLCG tokens) can never have a pre-provisioned .pem, so without
27 * minting they are permanently stuck on FALLBACK/DENY. Minting is
28 * strictly opt-in (a mint CA must be configured) — see
29 * fs/backend/cred_mint.h for the trust-model note.
30 *
31 * HOW: vfs_cred.c includes vfs_internal.h (for brix_vfs_ctx_t and
32 * brix_sd_cred_t), fs/backend/ucred.h (for brix_sd_ucred_select and
33 * brix_sd_ucred_t), and fs/backend/cred_mint.h (for brix_cred_mint).
34 * The shared vfs_backend_cred_decide() body performs the ucred_select +
35 * deny/fallback decision; both public gates call it with the relevant
36 * capability flag. On a DECLINED select with cap_ok and a mint CA
37 * configured, it makes ONE brix_cred_mint() attempt then re-resolves by
38 * the same key before falling through to deny/fallback — a mint
39 * failure changes nothing (same deny/fallback path as Phase-1's
40 * DECLINED). Outcome counters are emitted via brix_metric_cred_result
41 * (unified.h, pulled in transitively via vfs_internal.h → access_log.h
42 * → unified.h).
43 */
44#include "vfs_internal.h"

46#include "fs/backend/stage/sd_stage.h"
47#include "fs/backend/cred_mint.h"

49/* ---- brix_vfs_ctx_bind_backend_cred ----------------------------------------
50 *
51 * WHAT: Copies the conf's credential-dir pointer + fallback mode onto the ctx.
52 *
53 * WHY: brix_vfs_ctx_init predates the feature and is called from ~30 sites;
54 * a separate bind keeps the signature stable and lets the data-plane
55 * callsites opt in explicitly.
56 *
57 * HOW: nginx conf tokens are NUL-terminated, so the ngx_str_t data pointer is
58 * usable as a C string for the conf's lifetime (outlives every request).
59 * storage_cred_dir is set to NULL (feature off) when cred_dir is NULL or
60 * empty; storage_cred_deny mirrors the operator's fallback policy. */
61void
62brix_vfs_ctx_bind_backend_cred(brix_vfs_ctx_t *vctx,
63 const ngx_str_t *cred_dir, ngx_uint_t fallback_deny)
64{
65 if (vctx == NULL) {
66 return;
67 }
68
69 vctx->storage_cred_dir = (cred_dir != NULL && cred_dir->len > 0)
70 ? (const char *) cred_dir->data : NULL;
71 vctx->storage_cred_deny = (fallback_deny == 1) ? 1u : 0u;
72}

74/* ---- brix_vfs_ctx_bind_backend_mint ----------------------------------------
75 *
76 * WHAT: Copies the conf's mint-CA cert/key paths + TTL onto the ctx.
77 *
78 * WHY: Opt-in (phase-2 T9): only the data-plane sites that meaningfully
79 * benefit from minting (davs/S3 GET/PUT, where a bearer-only identity
80 * like an S3 access key has no native x509) call this; namespace-only
81 * call sites are unaffected and never mint. Kept as a separate bind
82 * (rather than extending brix_vfs_ctx_bind_backend_cred's signature) so
83 * the ~20 existing call sites of that function are untouched.
84 *
85 * HOW: Same NUL-terminated-conf-string borrowing convention as
86 * brix_vfs_ctx_bind_backend_cred; ca_cert unset/empty leaves minting
87 * off for this ctx (the gate's DECLINED path is then identical to
88 * Phase-1). */
89void
90brix_vfs_ctx_bind_backend_mint(brix_vfs_ctx_t *vctx,
91 const ngx_str_t *ca_cert, const ngx_str_t *ca_key, ngx_uint_t ttl_secs)
92{
93 if (vctx == NULL) {
94 return;
95 }
96
97 vctx->storage_cred_mint_ca_cert = (ca_cert != NULL && ca_cert->len > 0)
98 ? (const char *) ca_cert->data : NULL;
99 vctx->storage_cred_mint_ca_key = (ca_key != NULL && ca_key->len > 0)
100 ? (const char *) ca_key->data : NULL;
101 vctx->storage_cred_mint_ttl = ttl_secs;
102}

104/* ---- vfs_cred_live_bag ------------------------------------------------------
105 *
106 * WHAT: If a PASSTHROUGH/EXCHANGE live-cred bag is bound on the ctx, materialise
107 * it via brix_vfs_deleg_live_cred and report handled=1 with the rc it
108 * returned; otherwise leaves *handled=0 (caller runs the SELECT path).
109 *
110 * WHY: Phase-70 §4/§5.1/§5.4: when the front door bound a PASSTHROUGH or
111 * EXCHANGE live-cred bag (raw bearer / user-supplied full proxy), it must
112 * be materialised here instead of the directory SELECT. EXCHANGE trades
113 * the bearer for a backend-audienced token when an exchange endpoint is
114 * configured, else forwards it verbatim (documented §5.4 fallback).
115 * DELEGATE/MINT are left to fall through to select+mint for now.
116 *
117 * HOW: Reads the ctx cred mode once; only the two live-bag modes short-circuit.
118 * *handled tells the caller whether the returned rc is authoritative. */
119static ngx_int_t
120vfs_cred_live_bag(brix_vfs_ctx_t *ctx, brix_sd_cred_t *cred, int *use_cred,
121 int *err_out, int *handled)
122{
123 enum brix_cred_mode m = brix_vfs_backend_mode(ctx);
124
125 if (m == BRIX_CRED_PASSTHROUGH || m == BRIX_CRED_EXCHANGE) {
126 *handled = 1;
127 return brix_vfs_deleg_live_cred(ctx, cred, use_cred, err_out);
128 }
129
130 *handled = 0;
131 return NGX_OK;
132}

134/* ---- vfs_cred_maybe_mint ----------------------------------------------------
135 *
136 * WHAT: Given the initial select rc, performs the opt-in mint+re-resolve and
137 * returns the (possibly updated) rc.
138 *
139 * WHY: Phase-2 T9 opt-in minting: a DECLINED select (no pre-provisioned or
140 * previously-minted credential is currently valid) gets ONE mint attempt
141 * when a mint CA is configured AND the driver can actually use a per-user
142 * credential (cap_ok) — minting for a backend that cannot scope a session
143 * to a user credential would be pure waste. On a successful mint,
144 * re-resolve by the SAME key select() already derived (store->key is set
145 * on DECLINED). A mint failure is NOT a different denial reason: it
146 * simply leaves rc unchanged so the existing deny/fallback policy still
147 * applies.
148 *
149 * HOW: No-op (returns rc unchanged) unless rc != NGX_OK, cap_ok, and a mint CA
150 * is configured; on a successful brix_cred_mint it returns the re-resolve
151 * rc. */
152static ngx_int_t
153vfs_cred_maybe_mint(brix_vfs_ctx_t *ctx, int cap_ok, brix_sd_ucred_t *store,
154 ngx_int_t rc)
155{
156 if (rc != NGX_OK && cap_ok
157 && ctx->storage_cred_mint_ca_cert != NULL
158 && ctx->storage_cred_mint_ca_cert[0] != '\0') {
159 if (brix_cred_mint(ctx->storage_cred_dir,
160 ctx->storage_cred_mint_ca_cert, ctx->storage_cred_mint_ca_key,
161 store->principal, store->key,
162 (int) ctx->storage_cred_mint_ttl, ctx->log) == NGX_OK) {
163 rc = brix_sd_ucred_resolve(ctx->storage_cred_dir, store->key,
164 store);
165 }
166 }
167
168 return rc;
169}

171/* ---- vfs_cred_no_cap --------------------------------------------------------
172 *
173 * WHAT: Terminal outcome for a found credential on a backend that cannot scope
174 * a session to a user credential (cap_ok=0): NGX_ERROR (EACCES) in deny
175 * mode, else NGX_OK using the service credential.
176 *
177 * WHY: A user credential exists but the leaf backend has no per-user slot. In
178 * fallback=deny this must refuse before any origin connection; otherwise
179 * it warns and falls back to the service credential. Identical to the
180 * Phase-1 body — factored out only to bound the caller's complexity.
181 *
182 * HOW: Sets errno/err_out=EACCES and emits the DENY counter on refusal; emits
183 * the FALLBACK counter and returns NGX_OK otherwise. */
184static ngx_int_t
185vfs_cred_no_cap(brix_vfs_ctx_t *ctx, brix_sd_ucred_t *store, int *err_out)
186{
187 if (ctx->storage_cred_deny) {
188 ngx_log_error(NGX_LOG_ERR, ctx->log, 0,
189 "brix: backend \"%s\" cannot scope a session to a user "
190 "credential (fallback=deny) - refusing principal=\"%s\"",
191 brix_sd_backend_name(ctx->sd), store->principal);
192 errno = EACCES;
193 if (err_out != NULL) {
194 *err_out = EACCES;
195 }
196 brix_metric_cred_result(brix_vfs_metrics_proto(ctx),
197 BRIX_CRED_OUTCOME_DENY);
198 return NGX_ERROR;
199 }
200
201 ngx_log_error(NGX_LOG_WARN, ctx->log, 0,
202 "brix: backend \"%s\" cannot scope a session to a user "
203 "credential - using the service credential for \"%s\"",
204 brix_sd_backend_name(ctx->sd), store->principal);
205 brix_metric_cred_result(brix_vfs_metrics_proto(ctx),
206 BRIX_CRED_OUTCOME_FALLBACK);
207 return NGX_OK;
208}

210/* ---- vfs_cred_fill_user -----------------------------------------------------
211 *
212 * WHAT: Populates *cred from the resolved *store as an ordinary per-user
213 * credential and sets *use_cred=1.
214 *
215 * WHY: x509, bearer, s3, and ceph are mutually exclusive: exactly one kind is
216 * ever populated by ucred_select/resolve (store->is_bearer / is_s3 /
217 * is_ceph). Setting the inactive pointers to NULL prevents callers from
218 * accidentally using a stale value from a zeroed-struct default.
219 *
220 * HOW: Copies the active kind's fields, borrows dir + fallback policy from the
221 * ctx, emits the USER counter, and flags use_cred. */
222static void
223vfs_cred_fill_user(brix_vfs_ctx_t *ctx, brix_sd_ucred_t *store,
224 brix_sd_cred_t *cred, int *use_cred)
225{
226 cred->x509_proxy = (!store->is_bearer && !store->is_s3
227 && !store->is_ceph)
228 ? store->path : NULL;
229 cred->bearer = store->is_bearer ? store->bearer : NULL;
230 cred->s3_ak = store->is_s3 ? store->s3_ak : NULL;
231 cred->s3_sk = store->is_s3 ? store->s3_sk : NULL;
232 cred->s3_region = store->is_s3 ? store->s3_region : NULL;
233 cred->ceph_keyring = store->is_ceph ? store->ceph_keyring : NULL;
234 cred->ceph_user = store->is_ceph ? store->ceph_user : NULL;
235 cred->key = store->key;
236 cred->principal = store->principal;
237 cred->cred_dir = ctx->storage_cred_dir;
238 cred->fallback_deny = ctx->storage_cred_deny;
239 *use_cred = 1;
240 brix_metric_cred_result(brix_vfs_metrics_proto(ctx),
241 BRIX_CRED_OUTCOME_USER);
242}

244/* ---- vfs_cred_deny_or_fallback ---------------------------------------------
245 *
246 * WHAT: Terminal outcome for a select that did NOT yield a valid credential:
247 * NGX_ERROR (EACCES) in deny mode, else NGX_OK falling back to the service
248 * credential.
249 *
250 * WHY: No pre-provisioned/minted credential is currently valid. deny mode must
251 * refuse before any origin connection; otherwise the request proceeds on
252 * the service credential. Identical to the Phase-1 tail.
253 *
254 * HOW: Sets errno/err_out=EACCES and emits the DENY counter on refusal; emits
255 * the FALLBACK counter and returns NGX_OK otherwise. */
256static ngx_int_t
257vfs_cred_deny_or_fallback(brix_vfs_ctx_t *ctx, brix_sd_ucred_t *store,
258 int *err_out)
259{
260 if (ctx->storage_cred_deny) {
261 ngx_log_error(NGX_LOG_ERR, ctx->log, 0,
262 "brix: per-user backend credential %s for principal=\"%s\" "
263 "key=%s dir=\"%s\" (fallback=deny) - refusing",
264 store->expired ? "EXPIRED" : "missing", store->principal,
265 store->key, ctx->storage_cred_dir);
266 errno = EACCES;
267 if (err_out != NULL) {
268 *err_out = EACCES;
269 }
270 brix_metric_cred_result(brix_vfs_metrics_proto(ctx),
271 BRIX_CRED_OUTCOME_DENY);
272 return NGX_ERROR;
273 }
274
275 ngx_log_error(NGX_LOG_INFO, ctx->log, 0,
276 "brix: no per-user backend credential for principal=\"%s\" key=%s - "
277 "falling back to the service credential",
278 store->principal, store->key);
279 brix_metric_cred_result(brix_vfs_metrics_proto(ctx),
280 BRIX_CRED_OUTCOME_FALLBACK);
281 return NGX_OK;
282}

284/* ---- vfs_backend_cred_decide (static, shared by both gates) ----------------
285 *
286 * WHAT: Core select+deny+fallback policy executed for both the data-plane and
287 * namespace credential gates. Returns NGX_OK with *use_cred=1+*cred
288 * filled (user cred), NGX_OK with *use_cred=0 (feature-off or fallback),
289 * or NGX_ERROR (errno/err_out = EACCES, deny mode).
290 *
291 * WHY: Both brix_vfs_backend_cred and brix_vfs_ns_cred share IDENTICAL policy:
292 * ucred_select → if-found-but-not-capable → deny?refuse:warn+service;
293 * if-not-found → deny?refuse:info+service. A single body guarantees the
294 * two gates never drift.
295 *
296 * HOW: `cap_ok` is the caller's capability verdict (driver has the relevant
297 * *_cred slot). The live-bag / mint / found / not-found decision branches
298 * are delegated to dedicated helpers so the order and return values are
299 * identical to Phase-1's brix_vfs_backend_cred body. */
300static ngx_int_t
301vfs_backend_cred_decide(brix_vfs_ctx_t *ctx, int cap_ok,
302 brix_sd_ucred_t *store, brix_sd_cred_t *cred, int *use_cred, int *err_out)
303{
304 ngx_int_t rc;
305 int handled = 0;
306
307 *use_cred = 0;
308
309 rc = vfs_cred_live_bag(ctx, cred, use_cred, err_out, &handled);
310 if (handled) {
311 return rc;
312 }
313
314 if (ctx->storage_cred_dir == NULL || ctx->storage_cred_dir[0] == '\0') {
315 return NGX_OK;
316 }
317
318 rc = brix_sd_ucred_select(ctx->storage_cred_dir, ctx->identity, store);
319 rc = vfs_cred_maybe_mint(ctx, cap_ok, store, rc);
320
321 if (rc == NGX_OK) {
322 if (!cap_ok) {
323 return vfs_cred_no_cap(ctx, store, err_out);
324 }
325
326 vfs_cred_fill_user(ctx, store, cred, use_cred);
327 return NGX_OK;
328 }
329
330 return vfs_cred_deny_or_fallback(ctx, store, err_out);
331}

333/* ---- vfs_leaf_per_user_capable ---------------------------------------------
334 *
335 * WHAT: True when `leaf` (a decorator-unwrapped storage driver instance) can
336 * scope ANY operation to a per-user credential — i.e. it advertises at
337 * least one of the data-plane per-user slots (open_cred / staged_open_cred)
338 * OR the namespace per-user slot (stat_cred).
339 *
340 * WHY: "Can this backend authenticate the origin leg AS the inbound user?" is a
341 * single backend property, not a per-op one. A backend that scopes opens
342 * and staged writes to the user (sd_http: open_cred + staged_open_cred, but
343 * NO stat_cred) is fully per-user-capable — the user's bytes and objects
344 * are correctly isolated on the origin. Keying the capability verdict on
345 * ONE slot per gate wrongly declared such a backend "cannot scope a session
346 * to a user credential" the moment a WRITE flow issued an auxiliary
347 * namespace op (the WebDAV PUT lock-state getxattr routes through the
348 * stat_cred-keyed ns gate): in fallback=deny that refused the whole PUT
349 * even though every user-facing open/write on that leaf IS user-scoped.
350 * A namespace op the leaf simply does not support must degrade to
351 * unsupported at dispatch (getxattr on a backend with no xattr → ENOTSUP,
352 * which the lock check reads as "unlocked"), never to a credential-scoping
353 * DENY at the gate. Testing ANY per-user slot keeps the honest deny: a
354 * leaf with NONE of them still yields 0 and the refusal still fires.
355 *
356 * HOW: Pure predicate over the leaf driver's *_cred slot pointers. Shared by
357 * both cap_ok gates so the data-plane and namespace verdicts agree on the
358 * single "is this backend per-user capable" question. */
359static int
360vfs_leaf_per_user_capable(const brix_sd_instance_t *leaf)
361{
362 if (leaf == NULL || leaf->driver == NULL) {
363 return 0;
364 }
365
366 return (leaf->driver->open_cred != NULL
367 || leaf->driver->staged_open_cred != NULL
368 || leaf->driver->stat_cred != NULL) ? 1 : 0;
369}

371/* ---- vfs_backend_cap_ok -----------------------------------------------------
372 *
373 * WHAT: Whether the leaf driver (below any stage/cache decorators) can scope a
374 * data-plane op (open / staged_open) to a per-user credential.
375 *
376 * WHY: The stage and read-cache drivers are transparent decorators: they relay
377 * the origin open to their source driver via the plain slot and do NOT
378 * carry their own *_cred slots. Keying cap_ok on the TOP instance
379 * therefore reads as cap_ok=0 for ANY stage- or cache-wrapped export —
380 * even when the LEAF (e.g. sd_http, sd_xroot) fully implements open_cred +
381 * staged_open_cred. Unwrap to the leaf and test IT. Both the READ open
382 * (driver->open_cred) and the staged WRITE (driver->staged_open_cred) are
383 * per-user-capable slots, so this gate — shared by open_resolved_file.c
384 * (read) and vfs_staged.c (write) — accepts a leaf that has EITHER.
385 *
386 * HOW: Delegates to brix_vfs_ns_leaf() then vfs_leaf_per_user_capable(). */
387static int
388vfs_backend_cap_ok(const brix_vfs_ctx_t *ctx)
389{
390 return vfs_leaf_per_user_capable(brix_vfs_ns_leaf(ctx->sd));
391}

393/* ---- brix_vfs_backend_cred --------------------------------------------------
394 *
395 * WHAT: Data-plane credential gate: NGX_OK + use_cred=1 (user cred, *cred filled
396 * from *store); NGX_OK + use_cred=0 (feature-off or allowed fallback); or
397 * NGX_ERROR (errno/err_out = EACCES, deny mode, no valid cred).
398 *
399 * WHY: The single policy gate for open/staged_open: deny mode must reject
400 * BEFORE any origin connection is attempted.
401 *
402 * HOW: Delegates to vfs_backend_cred_decide with cap_ok = vfs_backend_cap_ok()
403 * — the leaf driver is per-user capable (open_cred OR staged_open_cred).
404 * Unwrapping the stage/cache decorators mirrors the namespace gate so a
405 * stage- or cache-wrapped https/xroot origin is correctly seen as capable —
406 * the actual open/staged_open runs on that leaf. The deny/fallback
407 * semantics are unchanged: a leaf with NO per-user slot still yields
408 * cap_ok=0 and the "cannot scope a session" refusal still fires. */
409ngx_int_t
410brix_vfs_backend_cred(brix_vfs_ctx_t *ctx, brix_sd_ucred_t *store,
411 brix_sd_cred_t *cred, int *use_cred, int *err_out)
412{
413 int cap_ok = vfs_backend_cap_ok(ctx);
414
415 return vfs_backend_cred_decide(ctx, cap_ok, store, cred, use_cred, err_out);
416}

418/* ---- brix_vfs_cred_gate_active ---------------------------------------------
419 *
420 * WHAT: True when a per-user backend credential SOURCE is bound on this ctx —
421 * either the directory-based SELECT policy (storage_cred_dir) or a live
422 * delegation bag (PASSTHROUGH/EXCHANGE). See vfs_internal.h for the full
423 * contract and the asymmetry it closes.
424 *
425 * WHY: The namespace dispatch sites (vfs_xattr/stat/unlink/mkdir/rename/dir/
426 * copy) must run the cred gate whenever EITHER source is present; keying
427 * on storage_cred_dir alone dropped the passthrough bearer/proxy on ns
428 * ops (the data-plane gate already consults the deleg bag first).
429 *
430 * HOW: storage_cred_dir non-empty OR a bound bag (mode != SELECT). Pure
431 * predicate — no I/O, no cred materialisation (the gate does that). */
432int
433brix_vfs_cred_gate_active(brix_vfs_ctx_t *ctx)
434{
435 if (ctx == NULL) {
436 return 0;
437 }
438 if (ctx->storage_cred_dir != NULL && ctx->storage_cred_dir[0] != '\0') {
439 return 1;
440 }
441 return (brix_vfs_backend_mode(ctx) != BRIX_CRED_SELECT) ? 1 : 0;
442}

444/* ---- brix_vfs_ns_leaf -------------------------------------------------------
445 *
446 * WHAT: Walk the stage/cache decorator chain from `top` to the leaf (the first
447 * non-decorator instance) and return it. Returns `top` unchanged when it
448 * is already the leaf, or NULL when `top` is NULL.
449 *
450 * WHY: The stage and read-cache decorators relay namespace ops (stat/unlink/
451 * rename/…) to their source through the PLAIN slot — they do not have
452 * their own *_cred slots. Calling brix_sd_<op>_maybe_cred on the TOP
453 * instance therefore always falls back to the plain relay (because the
454 * decorator's driver->stat_cred is NULL), so the credential is silently
455 * dropped even when use_cred=1. Dispatching on the LEAF instead lets the
456 * forwarder find the leaf driver's *_cred slot (e.g. sd_xroot) and
457 * correctly present the per-user proxy to the remote origin.
458 *
459 * HOW: Peel off stage and cache decorator layers using their published
460 * source-instance accessors (brix_sd_stage_source_instance /
461 * brix_sd_cache_source_instance) until neither returns an inner instance.
462 * The result is the first non-decorator instance — the actual storage
463 * driver (sd_xroot, sd_pblock, sd_ceph, sd_posix, …). */
464brix_sd_instance_t *
465brix_vfs_ns_leaf(brix_sd_instance_t *top)
466{
467 brix_sd_instance_t *inst = top;
468 brix_sd_instance_t *inner;
469
470 if (inst == NULL) {
471 return NULL;
472 }
473
474 for (;;) {
475 inner = brix_sd_stage_source_instance(inst);
476 if (inner == NULL) {
477 inner = brix_sd_cache_source_instance(inst);
478 }
479 if (inner == NULL) {
480 break; /* inst is the leaf */
481 }
482 inst = inner;
483 }
484
485 return inst;
486}

488/* ---- vfs_ns_cap_ok ----------------------------------------------------------
489 *
490 * WHAT: Whether the leaf driver (below any stage/cache decorators) can scope a
491 * session to a per-user credential — the capability verdict for the
492 * namespace-op gate.
493 *
494 * WHY: The stage and read-cache drivers are transparent decorators: they relay
495 * namespace ops (stat/unlink/rename/…) straight to their source driver via
496 * the plain slot. They do NOT have their own *_cred slots — so we unwrap
497 * to the leaf and test IT. The verdict is the SAME single question the
498 * data-plane gate asks (vfs_leaf_per_user_capable): can this backend
499 * authenticate the origin leg as the inbound user? A leaf that scopes
500 * opens/staged writes to the user but does NOT implement a namespace *_cred
501 * slot (sd_http) is still per-user-capable — its user isolation holds — so
502 * an auxiliary namespace op in a WRITE flow (the WebDAV PUT lock-state
503 * getxattr) must NOT be refused here as "cannot scope a session". When a
504 * namespace op is genuinely unsupported by the leaf, dispatch degrades to
505 * ENOTSUP (read as "unlocked" by the lock check), which is the correct
506 * outcome — not a credential DENY. A leaf with NO per-user slot at all
507 * still yields 0 so the honest refusal is preserved.
508 *
