1/*
2 * ratelimit.h — Phase 25 advanced rate limiting & traffic shaping.
3 *
4 * WHAT: A leaky-bucket request-rate and bandwidth limiter keyed on
5 * XRootD-specific identity dimensions (VO group, token issuer, client IP, DN,
6 * or storage-volume path prefix) rather than the coarse per-IP/per-DN bucket of
7 * the Phase 20 `brix_rate_limit` directive. Throttled XRootD stream clients
8 * receive kXR_wait(seconds); throttled HTTP/WebDAV clients receive 429 +
9 * Retry-After.
10 *
11 * WHY: A Tier-1 storage gateway needs policies like "ATLAS VO: 500 req/s, burst
12 * 800" or "/store/tape: 50 MiB/s aggregate" that the stock nginx limiter and the
13 * Phase 20 per-IP limiter cannot express. This module adds a second, richer
14 * limiter that coexists with — and is independent of — Phase 20.
15 *
16 * HOW: One or more named shared-memory zones (`brix_rate_limit_zone`) hold an
17 * rbtree of per-principal leaky-bucket nodes allocated from the zone's nginx
18 * slab pool, with an LRU queue for O(1) eviction when the zone fills (same
19 * structure as ngx_http_limit_req_module). Rules (`brix_rate_limit_rule`,
20 * `brix_bandwidth_limit`) attach to a WebDAV location or a stream server and
21 * are evaluated by an HTTP access-phase handler and a stream dispatch gate.
22 *
23 * Naming note: the Phase 20 directive `brix_rate_limit` (per-IP/DN token
24 * bucket over the KV store) is unrelated and unchanged; Phase 25 uses the
25 * distinct `brix_rate_limit_zone` / `brix_rate_limit_rule` /
26 * `brix_bandwidth_limit` directives.
27 */

124/*
125 * Zero the in-use gauges (in_flight, open_files) on every node of a shared
126 * zone. Called on reload adoption: these gauges self-heal only via a matched
127 * decrement, so a worker SIGKILLed (e.g. at reload's worker_shutdown_timeout)
128 * mid-request leaks its increment permanently, it survives reload in the SHM
129 * zone, and accumulates every restart cycle until the gauge reaches the
130 * configured cap and rejects that key forever. Resetting at each reload bounds
131 * any crash-leak to a single generation. The time-windowed rate/bandwidth
132 * buckets are deliberately preserved (they self-heal and must survive reload).

203/*
204 * Resolved HTTP/WebDAV-plane identity inputs for one key derivation.
205 *
206 * WHAT: A bundle of the already-resolved identity handles that brix_rl_key_http()
207 * reads to build a rate-limit key: the unified identity (`id`, may be NULL), the
208 * raw WebDAV request ctx that carries the cert DN (`wctx`, an opaque
209 * ngx_http_brix_webdav_req_ctx_t* kept as void* so this header need not pull in
210 * webdav.h, may be NULL on early-phase requests), the connection address (`ip`),
211 * and the lazily-resolved request path (`path`, used only by VOLUME rules).
212 *
213 * WHY: Bundling these into one struct keeps brix_rl_key_http() at four arguments
214 * (it previously took six loose ones) and lets the caller resolve the
215 * side-effecting handles — the WebDAV ctx's identity and the connection address —
216 * once, up front, before populating the literal. The per-key-type branch logic
217 * then reads one explicit, already-resolved struct rather than a loose argument
218 * list. Key-derivation semantics and the anon IP-fallback are unchanged.
219 *
220 * HOW: The HTTP access handler fills `id`/`wctx` from the WebDAV ctx and `ip`
221 * from the connection, passes a pointer here, and brix_rl_key_http() switches on
222 * rule->key_type over these fields.
223 */
224typedef struct {
225 brix_identity_t *id;
226 void *wctx; /* ngx_http_brix_webdav_req_ctx_t* (opaque) */
227 ngx_str_t *ip;
228 const char *path;
229} rl_key_req_t;
