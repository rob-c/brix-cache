12{

18 ngx_uint_t i;

22 for (i = 1; i < cf->args->nelts; i++) {
23 if (ngx_strncmp(value[i].data, "zone=", 5) == 0) {

25 zone.len = value[i].len - 5;
26 } else if (ngx_strncmp(value[i].data, "ttl=", 4) == 0) {
27 n = ngx_atoi(value[i].data + 4, value[i].len - 4);
28 if (n == NGX_ERROR || n <= 0) {
29 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
30 "invalid brix_auth_cache ttl \"%V\"", &value[i]);
31 return NGX_CONF_ERROR;
32 }
33 ttl = (ngx_uint_t) n;
34 } else {
35 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
36 "invalid brix_auth_cache parameter \"%V\"", &value[i]);
37 return NGX_CONF_ERROR;
38 }
39 }
40
41 if (zone.len == 0) {
42 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
43 "brix_auth_cache requires zone=<name>");
44 return NGX_CONF_ERROR;
45 }
46
47 kv = brix_kv_find(&zone);
48 if (kv == NULL) {
49 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
50 "brix_auth_cache: unknown zone \"%V\" "
51 "(declare it with brix_kv_zone first)", &zone);
52 return NGX_CONF_ERROR;
53 }
54 if (kv->key_max < 32 || kv->val_max < sizeof(brix_auth_cache_val_t)) {
55 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
56 "brix_auth_cache zone \"%V\" too small: need key>=32 val>=%uz",
57 &zone, sizeof(brix_auth_cache_val_t));
58 return NGX_CONF_ERROR;
59 }
60
61 ac->kv = kv;
62 ac->ttl_secs = ttl;
63 return NGX_CONF_OK;
64}
