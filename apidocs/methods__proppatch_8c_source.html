2 * methods_proppatch.c - WebDAV PROPPATCH handler (RFC 4918 §9.2).

31/*
32 * WHAT: Invariant state threaded through the PROPPATCH property walk — the
33 * request, the resolved+confined target path, and the growing response chain.
34 *
35 * WHY: These four values are constant for the duration of one PROPPATCH and are
36 * needed by every action/property helper. Bundling them keeps the per-property
37 * and per-action helpers under the parameter budget without new globals; state
38 * is still passed explicitly (one pointer, not a hidden singleton).
39 */
40typedef struct {
41 ngx_http_request_t *r; /* the PROPPATCH request */
42 const char *path; /* resolved, confined target path */
43 ngx_chain_t *head; /* multistatus response chain head */
44 ngx_chain_t *tail; /* multistatus response chain tail */
45} webdav_proppatch_walk_t;

47/*
48 * Re-serialize a client property element into the canonical XML fragment we
49 * persist as its dead-property value: "<X:local xmlns:X="ns">text</X:local>"
50 * (namespaced) or "<local>text</local>" (no namespace). Both the namespace URI
51 * and the text content are XML-escaped first, so untrusted property data cannot
52 * inject markup when it is later spliced into a PROPFIND response.
53 * Returns a pool string and its length in *out_len, or NULL on alloc failure.
54 */
55static char *
56webdav_proppatch_serialize_dead_prop(ngx_http_request_t *r, xmlNodePtr prop,
57 size_t *out_len)
58{
59 const char *local;
60 const char *ns;
61 char *safe_ns = NULL;
62 char *safe_text;
63 char *xml;
64 xmlChar *text;
65 size_t len;
66
67 local = (const char *) prop->name;
68 ns = (prop->ns != NULL && prop->ns->href != NULL)
69 ? (const char *) prop->ns->href : "";
70
71 text = xmlNodeGetContent(prop);
72 safe_text = webdav_escape_xml_text(r->pool,
73 text != NULL ? (const char *) text : "");
74 if (text != NULL) {
75 xmlFree(text);
76 }
77 if (safe_text == NULL) {
78 return NULL;
79 }
80
81 if (ns[0] != '\0') {
82 safe_ns = webdav_escape_xml_text(r->pool, ns);
83 if (safe_ns == NULL) {
84 return NULL;
85 }
86
87 /* Buffer sizing: the literal scaffolding + the local name twice (open
88 * and close tag) + the escaped namespace + escaped text + NUL. */
89 len = ngx_strlen("<X: xmlns:X=\"\"></X:>")
90 + strlen(local) * 2 + strlen(safe_ns) + strlen(safe_text) + 1;
91 BRIX_PNALLOC_OR_RETURN(xml, r->pool, len, NULL);
92 snprintf(xml, len, "<X:%s xmlns:X=\"%s\">%s</X:%s>",
93 local, safe_ns, safe_text, local);
94
95 } else {
96 len = ngx_strlen("<></>") + strlen(local) * 2 + strlen(safe_text) + 1;
97 BRIX_PNALLOC_OR_RETURN(xml, r->pool, len, NULL);
98 snprintf(xml, len, "<%s>%s</%s>", local, safe_text, local);
99 }
100
101 *out_len = strlen(xml);
102 return xml;
103}

105/*
106 * Append one <D:propstat> block reporting the per-property outcome: the empty
107 * property element (name only, via the dead-prop empty serializer) wrapped with
108 * the given HTTP status line. RFC 4918 §9.2 requires one propstat per property
109 * carrying its individual result.
110 */
111static ngx_int_t
112webdav_proppatch_append_propstat(webdav_proppatch_walk_t *w, const char *ns,
113 const char *local, const char *status)
114{
115 if (brix_http_chain_appendf(w->r->pool, &w->head, &w->tail,
116 "<D:propstat><D:prop>") == NULL)
117 {
118 return NGX_ERROR;
119 }
120
121 if (webdav_dead_prop_append_empty(w->r, ns, local, &w->head, &w->tail)
122 != NGX_OK)
123 {
124 return NGX_ERROR;
125 }
126
127 if (brix_http_chain_appendf(w->r->pool, &w->head, &w->tail,
128 "</D:prop><D:status>%s</D:status></D:propstat>",
129 status) == NULL)
130 {
131 return NGX_ERROR;
132 }
133
134 return NGX_OK;
135}

137/*
138 * WHAT: Apply one PROPPATCH property (set or remove) to the xattr-backed
139 * dead-property store and return its RFC 4918 §9.2 per-property status line.
140 *
141 * WHY: The per-property set/remove/protected decision is the innermost logic of
142 * the PROPPATCH walk; isolating it as a pure-ish helper keeps the action loop
143 * flat and expresses partial success precisely (each property gets its own
144 * verdict). No markup is emitted here — the caller wraps the result in propstat.
145 *
146 * HOW: A protected DAV: live property is rejected 403 and never touched. A set
147 * re-serializes the element and writes it; a remove deletes it; either store
148 * failure maps to 507 Insufficient Storage. Defaults to "HTTP/1.1 200 OK".
149 * Returns a static status-line string (never NULL).
150 */
151static const char *
152webdav_proppatch_prop_status(webdav_proppatch_walk_t *w, xmlNodePtr prop,
153 const char *ns, const char *local, ngx_flag_t is_set)
154{
155 if (strcmp(ns, "DAV:") == 0
156 && webdav_dead_prop_is_protected_dav(local))
157 {
158 return "HTTP/1.1 403 Forbidden";
159 }
160
161 if (is_set) {
162 char *xml;
163 size_t xml_len;
164
165 xml = webdav_proppatch_serialize_dead_prop(w->r, prop, &xml_len);
166 if (xml == NULL
167 || webdav_dead_prop_set(w->r, w->path, ns, local, xml, xml_len)
168 != NGX_OK)
169 {
170 return "HTTP/1.1 507 Insufficient Storage";
171 }
172 return "HTTP/1.1 200 OK";
173 }
174
175 if (webdav_dead_prop_remove(w->r, w->path, ns, local) != NGX_OK) {
176 return "HTTP/1.1 507 Insufficient Storage";
177 }
178 return "HTTP/1.1 200 OK";
179}

181/*
182 * WHAT: Process one <set>/<remove> action node of a DAV:propertyupdate — apply
183 * every property inside its <prop> wrapper and append a propstat per property,
184 * bumping *processed for each one.
185 *
186 * WHY: Extracting the middle (<prop> locate) and inner (per-property) loops out
187 * of the orchestrator collapses two nesting levels and gives the action a
188 * single-responsibility home. Malformed actions with no <prop> are silently
189 * skipped (spec-tolerant), matching the pre-refactor behavior.
190 *
191 * HOW: Locate the single <prop> element child; for each element child of it,
192 * compute the property status via webdav_proppatch_prop_status and append its
193 * propstat. Returns NGX_OK, or NGX_ERROR if propstat appending fails.
194 */
195static ngx_int_t
196webdav_proppatch_process_action(webdav_proppatch_walk_t *w, xmlNodePtr action,
197 ngx_flag_t is_set, ngx_uint_t *processed)
198{
199 xmlNodePtr prop_container;
200 xmlNodePtr prop;
201
202 /* Middle loop: locate the single <prop> wrapper inside this action. */
203 for (prop_container = action->children; prop_container != NULL;
204 prop_container = prop_container->next)
205 {
206 if (prop_container->type == XML_ELEMENT_NODE
207 && xmlStrcmp(prop_container->name, BAD_CAST "prop") == 0)
208 {
209 break;
210 }
211 }
212 if (prop_container == NULL) {
213 return NGX_OK; /* malformed action with no <prop>: skip */
214 }
215
216 /* Inner loop: apply each property element in this <prop>. */
217 for (prop = prop_container->children; prop != NULL; prop = prop->next) {
218 const char *local;
219 const char *ns;
220 const char *status;
221
222 if (prop->type != XML_ELEMENT_NODE) {
223 continue;
224 }
225
226 local = (const char *) prop->name;
227 ns = (prop->ns != NULL && prop->ns->href != NULL)
228 ? (const char *) prop->ns->href : "";
229
230 status = webdav_proppatch_prop_status(w, prop, ns, local, is_set);
231
232 if (webdav_proppatch_append_propstat(w, ns, local, status) != NGX_OK) {
233 return NGX_ERROR;
234 }
235
236 (*processed)++;
237 }
238
239 return NGX_OK;
240}

242/*
243 * WHAT: Emit the opening bytes of the 207 multistatus document for a PROPPATCH
244 * on this request URI: the XML prolog, <D:multistatus>, and the <D:href>.
245 *
246 * WHY: Href escaping and the fixed preamble are a self-contained concern; pulling
247 * them out keeps the orchestrator focused on the action walk. The href is
248 * XML-escaped so an untrusted URI cannot inject markup into the response.
249 *
250 * HOW: Null-terminate the URI into a bounded buffer, XML-escape it, and append
251 * the preamble chain fragment. Returns NGX_OK, or NGX_HTTP_INTERNAL_SERVER_ERROR
252 * on escape/allocation failure.
253 */
254static ngx_int_t
255webdav_proppatch_start_multistatus(ngx_http_request_t *r, ngx_chain_t **head,
256 ngx_chain_t **tail)
257{
258 char href_buf[WEBDAV_MAX_PATH + 1];
259 char *safe_href;
260 size_t uri_len;
261
262 uri_len = r->uri.len < sizeof(href_buf) - 1
263 ? r->uri.len : sizeof(href_buf) - 1;
264 ngx_memcpy(href_buf, r->uri.data, uri_len);
265 href_buf[uri_len] = '\0';
266
267 safe_href = webdav_escape_xml_text(r->pool, href_buf);
268 if (safe_href == NULL) {
269 return NGX_HTTP_INTERNAL_SERVER_ERROR;
270 }
271
272 if (brix_http_chain_appendf(r->pool, head, tail,
273 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
274 "<D:multistatus xmlns:D=\"DAV:\">"
275 "<D:response>"
276 "<D:href>%s</D:href>",
277 safe_href) == NULL)
278 {
279 return NGX_HTTP_INTERNAL_SERVER_ERROR;
280 }
281
282 return NGX_OK;
283}

285/*
286 * WHAT: Close the multistatus document, set the 207 status and Content-Type
287 * headers, and send the buffered response chain.
288 *
289 * WHY: The response-finalize edge (closing markup, length tally, header push,
290 * output filter) is one concern separate from the property walk; isolating it
291 * keeps the side effects at the orchestrator's tail in one place.
292 *
293 * HOW: Append the closing elements, mark the last buffer, sum the body length,
294 * set status 207 + application/xml Content-Type, send headers, then flush the
295 * chain via ngx_http_output_filter. Returns the HTTP status / NGX_* code.
296 */
297static ngx_int_t
298webdav_proppatch_finish_multistatus(ngx_http_request_t *r, ngx_chain_t *head,
299 ngx_chain_t *tail)
300{
301 ngx_chain_t *lc;
302 ngx_table_elt_t *h;
303 off_t total_len = 0;
304 ngx_int_t rc;
305
306 if (brix_http_chain_appendf(r->pool, &head, &tail,
307 "</D:response></D:multistatus>") == NULL)
308 {
309 return NGX_HTTP_INTERNAL_SERVER_ERROR;
310 }
311
312 if (tail != NULL) {
313 tail->buf->last_buf = 1;
314 tail->buf->last_in_chain = 1;
315 }
316
317 for (lc = head; lc != NULL; lc = lc->next) {
318 total_len += lc->buf->last - lc->buf->pos;
319 }
320
321 r->headers_out.status = 207;
322 r->headers_out.content_length_n = total_len;
323
324 h = ngx_list_push(&r->headers_out.headers);
325 if (h == NULL) {
326 return NGX_HTTP_INTERNAL_SERVER_ERROR;
327 }
328 h->hash = 1;
329 ngx_str_set(&h->key, "Content-Type");
330 ngx_str_set(&h->value, "application/xml; charset=\"utf-8\"");
331
332 rc = ngx_http_send_header(r);
333 if (rc == NGX_ERROR || r->header_only) {
334 return rc;
335 }
336
337 return ngx_http_output_filter(r, head);
338}

340/*
341 * WHAT: Parse the PROPPATCH request body into a validated DAV:propertyupdate
342 * document root, or map the failure to an HTTP status.
343 *
344 * WHY: Body read + XXE-safe XML parse + root-element validation are a single
345 * input-handling concern with several early-out error codes; separating them
346 * keeps the orchestrator's happy path linear. On success the caller owns *doc
347 * and must xmlFreeDoc it.
348 *
349 * HOW: Read up to WEBDAV_PROPPATCH_BODY_MAX bytes (empty body → 400), parse with
350 * network/entity-expansion disabled, and require a <propertyupdate> root.
351 * Returns NGX_OK with *doc and *root set, else NGX_HTTP_BAD_REQUEST.
352 */
353static ngx_int_t
354webdav_proppatch_parse_body(ngx_http_request_t *r, xmlDocPtr *doc,
355 xmlNodePtr *root)
356{
357 u_char *body = NULL;
358 size_t body_len = 0;
359 ngx_int_t rc;
360 int opts = XML_PARSE_NONET | XML_PARSE_NOERROR
361 | XML_PARSE_NOWARNING;
362#if defined(XML_PARSE_NO_XXE)
363 opts |= XML_PARSE_NO_XXE;
364#endif
365
366 rc = brix_http_body_read_all(r, WEBDAV_PROPPATCH_BODY_MAX,
367 &body, &body_len);
368 if (rc != NGX_OK || body_len == 0) {
369 return NGX_HTTP_BAD_REQUEST;
370 }
371
372 *doc = xmlReadMemory((const char *) body, (int) body_len,
373 "proppatch.xml", NULL, opts);
374 if (*doc == NULL) {
375 return NGX_HTTP_BAD_REQUEST;
376 }
377
378 *root = xmlDocGetRootElement(*doc);
379 if (*root == NULL
380 || xmlStrcmp((*root)->name, BAD_CAST "propertyupdate") != 0)
381 {
382 xmlFreeDoc(*doc);
383 *doc = NULL;
384 return NGX_HTTP_BAD_REQUEST;
385 }
386
387 return NGX_OK;
388}

390/*
391 * Apply a PROPPATCH after the body is read: resolve+stat the target, check it
392 * is not locked, parse the DAV:propertyupdate document, then walk its set/remove
393 * actions applying each to the xattr-backed dead-property store and emitting a
394 * per-property 207 multistatus. Returns the HTTP status / NGX_* code; an empty
395 * or unparseable update is 400.
396 */
397static ngx_int_t
398webdav_proppatch_do(ngx_http_request_t *r)
399{
400 webdav_proppatch_walk_t walk = { r, NULL, NULL, NULL };
401 ngx_int_t rc;
402 char path[WEBDAV_MAX_PATH];
403 struct stat sb;
404 xmlDocPtr doc = NULL;
405 xmlNodePtr root = NULL;
406 xmlNodePtr action;
407 ngx_uint_t processed = 0;
408
409 rc = webdav_resolve_stat(r, path, sizeof(path), &sb);
410 if (rc != NGX_OK) {
411 return rc;
412 }
413 walk.path = path;
414
415 rc = webdav_check_locks(r, path, 1);
416 if (rc != NGX_OK) {
417 return rc;
418 }
419
420 rc = webdav_proppatch_parse_body(r, &doc, &root);
421 if (rc != NGX_OK) {
422 return rc;
423 }
424
425 rc = webdav_proppatch_start_multistatus(r, &walk.head, &walk.tail);
426 if (rc != NGX_OK) {
427 xmlFreeDoc(doc);
428 return rc;
429 }
430
431 /* Outer loop: each child of <propertyupdate> is a <set> or <remove> action
432 * (anything else is ignored). is_set distinguishes the two. */
433 for (action = root->children; action != NULL; action = action->next) {
434 ngx_flag_t is_set;
435
436 if (action->type != XML_ELEMENT_NODE) {
437 continue;
438 }
439
440 if (xmlStrcmp(action->name, BAD_CAST "set") == 0) {
441 is_set = 1;
442 } else if (xmlStrcmp(action->name, BAD_CAST "remove") == 0) {
443 is_set = 0;
444 } else {
445 continue;
446 }
447
448 if (webdav_proppatch_process_action(&walk, action, is_set,
449 &processed) != NGX_OK)
450 {
451 xmlFreeDoc(doc);
452 return NGX_HTTP_INTERNAL_SERVER_ERROR;
453 }
454 }
455
456 xmlFreeDoc(doc);
457
458 if (processed == 0) {
459 return NGX_HTTP_BAD_REQUEST;
460 }
461
462 return webdav_proppatch_finish_multistatus(r, walk.head, walk.tail);
463}
