55/* Acquire everything a fill needs: source object, stat snapshot, staged store
56 * object, and the move buffer.
57 *
58 * WHAT: Opens the source for `key`, snapshots its stat, checks admission, and
59 * opens the staged cache-store object (which takes the fill lock — the
60 * O_EXCL + dead-owner-reclaim logic lives in the cstore layer, frozen).
61 *
62 * WHY: The acquire phase concentrates all the "can this fill happen at all?"
63 * decisions so the pump/commit phases run against a fully-provisioned
64 * state and can stay linear.
65 *
66 * HOW: brix_sd_open_maybe_cred so drivers with an open_cred slot use the
67 * per-user proxy. Returns NGX_OK (fs populated), NGX_DONE (stale copy
68 * served — caller reports success without filling), NGX_DECLINED
69 * (admission declined), or NGX_ERROR with errno set; every failure path
70 * releases whatever it had already acquired. */
71static ngx_int_t
72cache_fill_acquire(sd_cache_inst_state *st, const char *key,
73 const brix_sd_cred_t *cred, sd_cache_fill_state_t *fs)
74{
75 brix_sd_instance_t *src = fs->src;
76 mode_t fmode;
77 int err = 0;
78
79 if (src->driver->open == NULL) {
80 errno = ENOSYS;
81 return NGX_ERROR;
82 }
83 fs->so = brix_sd_open_maybe_cred(src, key, BRIX_SD_O_READ, 0, cred, &err);
84 if (fs->so == NULL) {
85 /* A cold-tier or mesh-sibling miss is NOT an origin outage — never
86 * stale-serve it; the orchestrator falls back to the origin fill. */
87 if (!fs->from_cold && !fs->from_peer
88 && sd_cache_stale_serve_ok(st, key))
89 {
90 return NGX_DONE; /* bounded stale-if-error (phase-68) */
91 }
92 errno = err ? err : EIO;
93 return NGX_ERROR;
94 }
95 /* Every op on the OPEN OBJECT dispatches through so->driver, NOT src->driver:
96 * a decorator source (sd_stage) legitimately returns the tier-below's object
97 * from a read open ("read byte-I/O bypasses the decorator"), so the object's
98 * vtable can differ from the instance's — dispatching object ops through the
99 * instance vtable reinterprets a foreign object state (type confusion). */
100 if (fs->so->driver->pread == NULL) {
101 brix_sd_obj_release(fs->so);
102 errno = ENOSYS;
103 return NGX_ERROR;
104 }
105 /* open() does not guarantee a populated snap (the posix driver fstats lazily),
106 * so fstat the object for an accurate size/mtime/mode - the cinfo validity and
107 * the cached file's permission bits both depend on it. */
108 fs->snap = fs->so->snap;
109 if (fs->so->driver->fstat != NULL) {
110 (void) fs->so->driver->fstat(fs->so, &fs->snap);
111 }
112 /* SECURITY + correctness: the physical cache-store object is a svc-owned
113 * artifact that aggregates MANY users' bytes under one tree. Per-user
114 * authorization is enforced at the protocol gate (open_cache.c), and the
115 * CLIENT-FACING mode is carried in the cinfo record and served by
116 * sd_cache_stat() — decoupled from this physical mode. So force 0600:
117 * - owner rw is REQUIRED (XATTR meta_mode stores user.xrd.cinfo ON this
118 * object; Linux refuses user.* xattrs on a non-writable inode — a
119 * read-only 0444 source would otherwise block cinfo persistence, G3);
120 * - NO group/other bits, so a mapped low-priv uid cannot read another
121 * user's cached bytes by direct filesystem access (the source mode's
122 * 0644 previously leaked here). snap.mode still reaches clients via the
123 * cinfo record (ci.mode below, served by sd_cache_stat). */
124 fmode = S_IRUSR | S_IWUSR; /* 0600 — svc-owned cache artifact */
125
126 if (!sd_cache_admit(&st->policy, key, fs->snap.size)) {
127 brix_sd_obj_release(fs->so);
128 return NGX_DECLINED; /* too big / filtered - do not cache */
129 }
130
131 fs->staged = brix_cstore_fill_open(&st->cstore, key, fmode);
132 if (fs->staged == NULL) {
133 ngx_log_error(NGX_LOG_WARN, st->log, errno,
134 "sd_cache: fill_open on the cache store failed for \"%s\" - not cached",
135 key);
136 brix_sd_obj_release(fs->so);
137 return NGX_ERROR;
138 }
139 fs->buf = malloc(SD_CACHE_CHUNK);
140 if (fs->buf == NULL) {
141 brix_cstore_fill_abort(fs->staged);
142 brix_sd_obj_release(fs->so);
143 errno = ENOMEM;
144 return NGX_ERROR;
145 }
146 return NGX_OK;
147}

149/* Pump the source object's bytes into the staged store object.
150 *
151 * WHAT: The origin read loop — pread from the source, fill_write to the staged
152 * object, chunk by chunk, advancing fs->off.
153 *
154 * WHY: The pump is the fill's only long-running phase; isolating it keeps the
155 * read-error / stale-serve / errno-preservation logic in one place.
156 *
157 * HOW: On success frees the move buffer and closes the source (the staged
158 * object stays open for verify/commit) and returns NGX_OK. On a read or
159 * write failure it releases EVERYTHING (buffer, staged fill, source) and
160 * returns NGX_DONE if a bounded stale copy can be served, else NGX_ERROR
161 * with the READ's errno preserved. */
162static ngx_int_t
163cache_fill_pump(sd_cache_inst_state *st, const char *key,
164 sd_cache_fill_state_t *fs)
165{
166 for ( ;; ) {
167 ssize_t r = fs->so->driver->pread(fs->so, fs->buf, SD_CACHE_CHUNK,
168 fs->off);
169
170 if (r < 0) {
171 int read_err = errno; /* capture BEFORE any cleanup call */
172
173 if (read_err == EINTR) {
174 continue;
175 }
176 free(fs->buf);
177 brix_cstore_fill_abort(fs->staged);
178 brix_sd_obj_release(fs->so);
179 if (!fs->from_cold && !fs->from_peer
180 && sd_cache_stale_serve_ok(st, key))
181 {
182 return NGX_DONE; /* bounded stale-if-error (phase-68) */
183 }
184 /* Restore the READ's errno. sd_cache_stale_serve_ok() (and the
185 * cleanup calls) stat the absent cache entry and leave errno
186 * ENOENT — propagating that would make the fill layer classify a
187 * merely-broken origin connection (EIO/ETIMEDOUT) as the origin's
188 * definitive 404, turning a transient network fault into a
189 * cached, client-poisoning "not found" instead of a retry. */
190 errno = read_err;
191 return NGX_ERROR;
192 }
193 if (r == 0) {
194 /* Clean EOF. If the source declared its size, an EOF short of that
195 * size is a TRUNCATED transfer — the origin (or an in-path actor)
196 * cut the fill mid-stream. Committing fs->off bytes as a whole-file
197 * COMPLETE object would poison the cache with a permanently-short
198 * copy served as a valid hit to every subsequent client. Fail
199 * closed exactly like a mid-fill read error: release everything,
200 * serve a bounded stale copy if one exists, else EIO so the next
201 * read refetches. A size-unknown source (snap.size <= 0, e.g. an
202 * HTTP origin with no Content-Length) has no lower bound to check
203 * against, so a clean EOF is the only completion signal available. */
204 if (fs->snap.size > 0 && fs->off < fs->snap.size) {
205 ngx_log_error(NGX_LOG_ERR, st->log, 0,
206 "sd_cache: origin truncated fill for \"%s\" at %O of %O "
207 "bytes - not caching the short object", key,
208 (off_t) fs->off, (off_t) fs->snap.size);
209 free(fs->buf);
210 brix_cstore_fill_abort(fs->staged);
211 brix_sd_obj_release(fs->so);
212 if (!fs->from_cold && !fs->from_peer
213 && sd_cache_stale_serve_ok(st, key))
214 {
215 return NGX_DONE; /* bounded stale-if-error (phase-68) */
216 }
217 errno = EIO;
218 return NGX_ERROR;
219 }
220 break;
221 }
222 if (brix_cstore_fill_write(fs->staged, fs->buf, (size_t) r,
223 fs->off) < 0)
224 {
225 free(fs->buf);
226 brix_cstore_fill_abort(fs->staged);
227 brix_sd_obj_release(fs->so);
228 return NGX_ERROR;
229 }
230 fs->off += r;
231 if (fs->snap.size > 0 && fs->off >= fs->snap.size) {
232 break; /* size known: skip the EOF-probe round-trip */
233 }
234 }
235 free(fs->buf);
236 fs->buf = NULL;
237 /* Capture the origin's advertised content digest BEFORE releasing the source
238 * object — the query needs a live, open object. Only the xroot source offers
239 * an in-band digest (kXR_Qcksum); other backends (http/s3/posix) leave alg/hex
240 * empty and the verify phase decides best-effort/require on that. Skipped
241 * entirely unless a digest-verify policy is in force, so an OFF/CVMFS-CAS fill
242 * pays no round-trip. Mirrors the fetch.c commit-then-verify pattern. */
243 if ((st->policy.verify == BRIX_CACHE_VERIFY_BESTEFFORT
244 || st->policy.verify == BRIX_CACHE_VERIFY_REQUIRE)
245 && ngx_strcmp(brix_sd_backend_name(fs->src), "xroot") == 0)
246 {
247 brix_sd_xroot_query_checksum(fs->so, fs->origin_alg,
248 sizeof(fs->origin_alg), fs->origin_hex, sizeof(fs->origin_hex));
249 }
250 brix_sd_obj_release(fs->so);
251 fs->so = NULL;
252 return NGX_OK;
253}

255/* Publish the staged object and record its cinfo.
256 *
257 * WHAT: Accounts the upstream fill, commits (renames) the staged object into
258 * the store, and stores the whole-file COMPLETE cinfo built from the
259 * source stat snapshot (the fstat mode fix — origin perms in ci.mode).
260 *
261 * WHY: The commit phase is the fill's publication point; everything before it
262 * is invisible to readers, so all bookkeeping that describes a
263 * SUCCESSFUL fill belongs here.
264 *
265 * HOW: Whole-file COMPLETE cinfo (the 1 MiB granule keys validity; the present
266 * bitmap is all-set). A partial/slice fill is section 6.5 / SP2. A failed
267 * cinfo store is only a WARN — the object is cached but unrecorded, a
268 * safe miss (refill) next time. */
269static ngx_int_t
270cache_fill_commit(sd_cache_inst_state *st, const char *key,
271 sd_cache_fill_state_t *fs)
272{
273 brix_cache_cinfo_t ci;
274
275 if (!fs->from_cold && !fs->from_peer) {
276 /* a cold promote or a sibling fetch moved zero origin-WAN bytes */
277 sd_cache_note_origin_bytes(st, key, fs->off); /* WAN in, this attempt */
278 sd_cache_note_upstream(st, 1, fs->off, sd_cache_ms_since(&fs->t0));
279 }
280
281 if (brix_cstore_fill_commit(fs->staged) != NGX_OK) {
282 return NGX_ERROR;
283 }
284
285 ngx_memzero(&ci, sizeof(ci));
286 ci.magic = BRIX_CACHE_CINFO_MAGIC;
287 ci.version = BRIX_CACHE_CINFO_VERSION;
288 ci.block_size = BRIX_CACHE_DIRTY_BLOCK;
289 ci.size = (uint64_t) fs->off;
290 ci.mtime = (uint64_t) fs->snap.mtime;
291 ci.mode = (uint32_t) (fs->snap.mode & 0777); /* origin perms — served back
292 * so a read-only source is not
293 * masked by the owner-writable
294 * physical store object. */
295 ci.nblocks = brix_cache_cinfo_nblocks((uint64_t) fs->off, ci.block_size);
296 ci.flags = BRIX_CINFO_F_COMPLETE
297 | (fs->verified ? BRIX_CINFO_F_VERIFIED : 0);
298 ci.filled_at = (uint64_t) time(NULL);
299 if (fs->verified && fs->cks_alg[0] != '\0' && fs->cks_hex[0] != '\0') {
300 /* The verified digest — the producer side of xrdckverify --cache.
301 * Only a MATCHED digest is recorded; an unverified fill leaves the
302 * fields empty rather than asserting bytes nobody checked. */
303 ngx_cpystrn((u_char *) ci.cks_alg, (u_char *) fs->cks_alg,
304 sizeof(ci.cks_alg));
305 ci.cks_alg_len = (uint8_t) ngx_strlen(ci.cks_alg);
306 ngx_cpystrn((u_char *) ci.cks_hex, (u_char *) fs->cks_hex,
307 sizeof(ci.cks_hex));
308 ci.cks_len = (uint8_t) ngx_strlen(ci.cks_hex);
309 }
310 if (st->policy.cvmfs_manifest_ttl > 0 && sd_cache_is_manifest_key(key)) {
311 brix_cache_cinfo_set_expires(&ci,
312 (time_t) ci.filled_at + st->policy.cvmfs_manifest_ttl);
313 }
314
315 if (brix_cstore_cinfo_store(&st->cstore, key, &ci) != NGX_OK) {
316 /* The object is cached but unrecorded - a safe miss (refill) next time. */
317 ngx_log_error(NGX_LOG_WARN, st->log, errno,
318 "sd_cache: cinfo store failed for \"%s\" - will refill on next read",
319 key);
320 }
321 return NGX_OK;
322}

324/* One fill attempt from the tier `src` (the wrapped source, the cold store on
325 * a phase-85 F7 promote, or a mesh sibling on a phase-85 F8 peer fetch):
326 * acquire (source open + admission + staged fill-open) -> pump (read loop) ->
327 * verify (phase-68 cvmfs-cas) -> commit (publish + cinfo). NGX_DONE from
328 * acquire/pump means a bounded stale copy was served — reported as NGX_OK
329 * without caching (origin attempts only; a cold or peer attempt never
330 * stale-serves so its failure falls back to the origin). */
331static ngx_int_t
332sd_cache_fill_attempt(sd_cache_inst_state *st, const char *key,
333 const brix_sd_cred_t *cred, brix_sd_instance_t *src, int from_cold,
334 int from_peer)
335{
336 sd_cache_fill_state_t fs;
337 ngx_int_t rc;
338
339 ngx_memzero(&fs, sizeof(fs));
340 fs.src = src;
341 fs.from_cold = from_cold;
342 fs.from_peer = from_peer;
343 (void) clock_gettime(CLOCK_MONOTONIC, &fs.t0);
344
345 rc = cache_fill_acquire(st, key, cred, &fs);
346 if (rc == NGX_DONE) {
347 return NGX_OK; /* bounded stale-if-error (phase-68) */
348 }
349 if (rc != NGX_OK) {
350 return rc; /* NGX_DECLINED or NGX_ERROR */
351 }
352
353 rc = cache_fill_pump(st, key, &fs);
354 if (rc == NGX_DONE) {
355 return NGX_OK; /* bounded stale-if-error (phase-68) */
356 }
357 if (rc != NGX_OK) {
358 return NGX_ERROR;
359 }
360
361 if (cache_fill_verify(st, key, &fs) != NGX_OK) {
362 return NGX_ERROR;
363 }
364 return cache_fill_commit(st, key, &fs);
365}

367/* Rendezvous (highest-random-weight) owner of `key` over the peer ring:
368 * FNV-1a 64 over "<label>\n<key>", owner = argmax across members. Every mesh
369 * node carries the IDENTICAL member list with identical labels, so all nodes
370 * agree on the owner with no coordination. The scheme (FNV-1a 64, offset
371 * 14695981039346656037, prime 1099511628211, '\n' separator, ties to the
372 * lower index) is mirrored bit-for-bit by tests/test_cvmfs_peer_mesh.py —
373 * change both together or not at all. */
374static uint64_t
375sd_cache_hrw_fnv1a64(const char *label, const char *key)
376{
377 uint64_t h = BRIX_FNV1A64_OFFSET_BASIS;
378 const char *p;
379
380 for (p = label; *p != '\0'; p++) {
381 h = (h ^ (uint64_t) (unsigned char) *p) * BRIX_FNV1A64_PRIME;
382 }
383 h = (h ^ (uint64_t) '\n') * BRIX_FNV1A64_PRIME;
384 for (p = key; *p != '\0'; p++) {
385 h = (h ^ (uint64_t) (unsigned char) *p) * BRIX_FNV1A64_PRIME;
386 }
387 return h;
388}

390static int
391sd_cache_peer_owner(const sd_cache_inst_state *st, const char *key)
392{
393 int i, owner = 0;
394 uint64_t best = 0, w;
395
396 for (i = 0; i < st->n_peers; i++) {
397 w = sd_cache_hrw_fnv1a64(st->peers[i].label, key);
398 if (i == 0 || w > best) {
399 best = w;
400 owner = i;
401 }
402 }
403 return owner;
404}

406/* Fill `key` into the cache store and record its cinfo.
407 *
408 * WHAT: Pulls the object `key` into the cache store, building a whole-file
409 * cinfo on success. With a cold store tier attached (phase-85 F7) a
410 * verified PROMOTE from the cold copy is attempted first; with a sibling
411 * mesh attached (phase-85 F8) the key's rendezvous-owning peer is tried
412 * next; the origin (st->source) is the fallback and the only tier that
413 * stale-serves.
414 *
415 * WHY: The fill spine is the single path that populates the cache; threading
416 * the caller's per-user `cred` here ensures the source open uses the
417 * correct identity rather than the service credential. Routing the
418 * promote through the SAME spine means a cold object passes the identical
419 * cvmfs-cas / manifest-signature verify gate as an origin fill — a
420 * corrupt cold copy can never publish.
421 *
422 * HOW: Cold attempt (when st->cold): any failure but admission silently falls
423 * back to the origin; success drops the cold copy (move semantics).
424 * Peer attempt (when the ring owner of `key` is a non-self member with a
425 * built instance): one verified fetch from that sibling; any failure but
426 * admission silently falls back to the origin. Only the configured ring
427 * members are ever contacted (allowlist by construction — no request-
428 * derived authority reaches a peer fetch). Returns NGX_OK (object cached
429 * + cinfo stored, or stale-served), NGX_DECLINED (policy declined), or
430 * NGX_ERROR (failure). */
431ngx_int_t
432sd_cache_fill(sd_cache_inst_state *st, const char *key,
433 const brix_sd_cred_t *cred)
434{
435 if (st->cold != NULL) {
436 ngx_int_t rc = sd_cache_fill_attempt(st, key, cred, st->cold, 1, 0);
437
438 if (rc == NGX_OK) {
439 /* Promoted: the hot tier owns the object now — drop the cold copy
440 * (best-effort; a leftover is simply re-evicted or re-demoted). */
441 if (st->cold->driver->unlink != NULL) {
442 (void) st->cold->driver->unlink(st->cold, key, 0);
443 }
444 ngx_log_error(NGX_LOG_INFO, st->log, 0,
445 "sd_cache: promoted \"%s\" from the cold tier", key);
446 return NGX_OK;
447 }
448 if (rc == NGX_DECLINED) {
449 return rc; /* admission policy — identical for the origin */
450 }
451 /* cold miss / outage / corrupt copy: fall back to the origin fill */
452 }
453
454 if (st->n_peers > 0) {
455 int owner = sd_cache_peer_owner(st, key);
456
457 if (owner != st->peer_self && st->peers[owner].inst != NULL) {
458 ngx_int_t rc = sd_cache_fill_attempt(st, key, cred,
459 st->peers[owner].inst, 0, 1);
460
461 if (rc == NGX_OK) {
462 ngx_log_error(NGX_LOG_INFO, st->log, 0,
463 "sd_cache: filled \"%s\" from mesh sibling %s",
464 key, st->peers[owner].label);
465 return NGX_OK;
466 }
467 if (rc == NGX_DECLINED) {
468 return rc; /* admission policy — identical for the origin */
469 }
470 /* sibling miss / outage / corrupt copy: fall back to the origin */
471 }
472 }
473
474 return sd_cache_fill_attempt(st, key, cred, st->source, 0, 0);
475}
