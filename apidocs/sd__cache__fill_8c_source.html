1/*
2 * sd_cache_fill.c — the whole-file fill spine of the read-through cache.
3 *
4 * WHAT: Pulls a COMPLETE object from the wrapped source into the cache store and
5 * records its cinfo. Owns the four fill phases (acquire -> pump -> verify
6 * -> commit) and the sd_cache_fill orchestrator the interposed read-open
7 * and the async-offload entrypoint both drive.
8 *
9 * WHY: Split from sd_cache.c (phase-79) to keep every cache file under the
10 * ~500-line, one-concept-per-file cap. The fill spine is the single path
11 * that populates the cache; isolating it keeps the origin-read / stale-serve
12 * / verify / cinfo-publish logic reviewable apart from the vtable adapters
13 * (sd_cache.c), the slice/partial machinery (sd_cache_partial.c), and the
14 * namespace/staged forwarders (sd_cache_forward.c).
15 *
16 * HOW: cache_fill_acquire (source open + admission + staged fill-open) -> pump
17 * (origin read loop) -> verify (phase-68 cvmfs-cas) -> commit (publish +
18 * cinfo). sd_cache.c reaches sd_cache_fill (miss-serve and the async
19 * fill-key) through sd_cache_internal.h. The admission/policy +
20 * repo-metrics helpers it calls live in sd_cache_policy.c. Phase-85 F7:
21 * the same spine runs the cold-tier PROMOTE (cold store as the fill
22 * source, from_cold=1) and this file also hosts brix_sd_cache_demote,
23 * the eviction-seam hot->cold copy.
24 */

53/* Transient state threaded through the fill phases (acquire -> pump ->
54 * verify -> commit). Owned by sd_cache_fill; each phase helper consumes or
55 * releases the resources it is documented to on its own failure paths. */
56typedef struct {
57 brix_sd_instance_t *src; /* the tier THIS attempt fills from — the
58 * wrapped source, the cold store tier on a
59 * phase-85 F7 promote, or a mesh sibling
60 * on a phase-85 F8 peer attempt */
61 int from_cold; /* 1 = promote attempt: no stale-serve, no
62 * WAN accounting, no origin tamper signal */
63 int from_peer; /* 1 = mesh-sibling attempt: no stale-serve,
64 * no origin WAN accounting; a verify reject
65 * DOES raise signal=cvmfs_tamper naming
66 * the SIBLING as the actor */
67 brix_sd_obj_t *so; /* the open source object */
68 brix_sd_staged_t *staged; /* the staged (uncommitted) store object */
69 brix_sd_stat_t snap; /* source size/mtime/mode snapshot */
70 u_char *buf; /* SD_CACHE_CHUNK move buffer */
71 off_t off; /* bytes pumped so far */
72 int verified; /* cvmfs-cas / origin-digest verified */
73 char origin_alg[16]; /* origin-advertised digest algorithm
74 * (kXR_Qcksum), captured while the
75 * source object is still open */
76 char origin_hex[129]; /* origin-advertised digest hex */
77 struct timespec t0; /* T16: per-upstream fill duration */
78} sd_cache_fill_state_t;

80/* Acquire everything a fill needs: source object, stat snapshot, staged store
81 * object, and the move buffer.
82 *
83 * WHAT: Opens the source for `key`, snapshots its stat, checks admission, and
84 * opens the staged cache-store object (which takes the fill lock — the
85 * O_EXCL + dead-owner-reclaim logic lives in the cstore layer, frozen).
86 *
87 * WHY: The acquire phase concentrates all the "can this fill happen at all?"
88 * decisions so the pump/commit phases run against a fully-provisioned
89 * state and can stay linear.
90 *
91 * HOW: brix_sd_open_maybe_cred so drivers with an open_cred slot use the
92 * per-user proxy. Returns NGX_OK (fs populated), NGX_DONE (stale copy
93 * served — caller reports success without filling), NGX_DECLINED
94 * (admission declined), or NGX_ERROR with errno set; every failure path
95 * releases whatever it had already acquired. */
96static ngx_int_t
97cache_fill_acquire(sd_cache_inst_state *st, const char *key,
98 const brix_sd_cred_t *cred, sd_cache_fill_state_t *fs)
99{
100 brix_sd_instance_t *src = fs->src;
101 mode_t fmode;
102 int err = 0;
103
104 if (src->driver->open == NULL) {
105 errno = ENOSYS;
106 return NGX_ERROR;
107 }
108 fs->so = brix_sd_open_maybe_cred(src, key, BRIX_SD_O_READ, 0, cred, &err);
109 if (fs->so == NULL) {
110 /* A cold-tier or mesh-sibling miss is NOT an origin outage — never
111 * stale-serve it; the orchestrator falls back to the origin fill. */
112 if (!fs->from_cold && !fs->from_peer
113 && sd_cache_stale_serve_ok(st, key))
114 {
115 return NGX_DONE; /* bounded stale-if-error (phase-68) */
116 }
117 errno = err ? err : EIO;
118 return NGX_ERROR;
119 }
120 /* Every op on the OPEN OBJECT dispatches through so->driver, NOT src->driver:
121 * a decorator source (sd_stage) legitimately returns the tier-below's object
122 * from a read open ("read byte-I/O bypasses the decorator"), so the object's
123 * vtable can differ from the instance's — dispatching object ops through the
124 * instance vtable reinterprets a foreign object state (type confusion). */
125 if (fs->so->driver->pread == NULL) {
126 brix_sd_obj_release(fs->so);
127 errno = ENOSYS;
128 return NGX_ERROR;
129 }
130 /* open() does not guarantee a populated snap (the posix driver fstats lazily),
131 * so fstat the object for an accurate size/mtime/mode - the cinfo validity and
132 * the cached file's permission bits both depend on it. */
133 fs->snap = fs->so->snap;
134 if (fs->so->driver->fstat != NULL) {
135 (void) fs->so->driver->fstat(fs->so, &fs->snap);
136 }
137 /* SECURITY + correctness: the physical cache-store object is a svc-owned
138 * artifact that aggregates MANY users' bytes under one tree. Per-user
139 * authorization is enforced at the protocol gate (open_cache.c), and the
140 * CLIENT-FACING mode is carried in the cinfo record and served by
141 * sd_cache_stat() — decoupled from this physical mode. So force 0600:
142 * - owner rw is REQUIRED (XATTR meta_mode stores user.xrd.cinfo ON this
143 * object; Linux refuses user.* xattrs on a non-writable inode — a
144 * read-only 0444 source would otherwise block cinfo persistence, G3);
145 * - NO group/other bits, so a mapped low-priv uid cannot read another
146 * user's cached bytes by direct filesystem access (the source mode's
147 * 0644 previously leaked here). snap.mode still reaches clients via the
148 * cinfo record (ci.mode below, served by sd_cache_stat). */
149 fmode = S_IRUSR | S_IWUSR; /* 0600 — svc-owned cache artifact */
150
151 if (!sd_cache_admit(&st->policy, key, fs->snap.size)) {
152 brix_sd_obj_release(fs->so);
153 return NGX_DECLINED; /* too big / filtered - do not cache */
154 }
155
156 fs->staged = brix_cstore_fill_open(&st->cstore, key, fmode);
157 if (fs->staged == NULL) {
158 ngx_log_error(NGX_LOG_WARN, st->log, errno,
159 "sd_cache: fill_open on the cache store failed for \"%s\" - not cached",
160 key);
161 brix_sd_obj_release(fs->so);
162 return NGX_ERROR;
163 }
164 fs->buf = malloc(SD_CACHE_CHUNK);
165 if (fs->buf == NULL) {
166 brix_cstore_fill_abort(fs->staged);
167 brix_sd_obj_release(fs->so);
168 errno = ENOMEM;
169 return NGX_ERROR;
170 }
171 return NGX_OK;
172}

174/* Pump the source object's bytes into the staged store object.
175 *
176 * WHAT: The origin read loop — pread from the source, fill_write to the staged
177 * object, chunk by chunk, advancing fs->off.
178 *
179 * WHY: The pump is the fill's only long-running phase; isolating it keeps the
180 * read-error / stale-serve / errno-preservation logic in one place.
181 *
182 * HOW: On success frees the move buffer and closes the source (the staged
183 * object stays open for verify/commit) and returns NGX_OK. On a read or
184 * write failure it releases EVERYTHING (buffer, staged fill, source) and
185 * returns NGX_DONE if a bounded stale copy can be served, else NGX_ERROR
186 * with the READ's errno preserved. */
187static ngx_int_t
188cache_fill_pump(sd_cache_inst_state *st, const char *key,
189 sd_cache_fill_state_t *fs)
190{
191 for ( ;; ) {
192 ssize_t r = fs->so->driver->pread(fs->so, fs->buf, SD_CACHE_CHUNK,
193 fs->off);
194
195 if (r < 0) {
196 int read_err = errno; /* capture BEFORE any cleanup call */
197
198 if (read_err == EINTR) {
199 continue;
200 }
201 free(fs->buf);
202 brix_cstore_fill_abort(fs->staged);
203 brix_sd_obj_release(fs->so);
204 if (!fs->from_cold && !fs->from_peer
205 && sd_cache_stale_serve_ok(st, key))
206 {
207 return NGX_DONE; /* bounded stale-if-error (phase-68) */
208 }
209 /* Restore the READ's errno. sd_cache_stale_serve_ok() (and the
210 * cleanup calls) stat the absent cache entry and leave errno
211 * ENOENT — propagating that would make the fill layer classify a
212 * merely-broken origin connection (EIO/ETIMEDOUT) as the origin's
213 * definitive 404, turning a transient network fault into a
214 * cached, client-poisoning "not found" instead of a retry. */
215 errno = read_err;
216 return NGX_ERROR;
217 }
218 if (r == 0) {
219 /* Clean EOF. If the source declared its size, an EOF short of that
220 * size is a TRUNCATED transfer — the origin (or an in-path actor)
221 * cut the fill mid-stream. Committing fs->off bytes as a whole-file
222 * COMPLETE object would poison the cache with a permanently-short
223 * copy served as a valid hit to every subsequent client. Fail
224 * closed exactly like a mid-fill read error: release everything,
225 * serve a bounded stale copy if one exists, else EIO so the next
226 * read refetches. A size-unknown source (snap.size <= 0, e.g. an
227 * HTTP origin with no Content-Length) has no lower bound to check
228 * against, so a clean EOF is the only completion signal available. */
229 if (fs->snap.size > 0 && fs->off < fs->snap.size) {
230 ngx_log_error(NGX_LOG_ERR, st->log, 0,
231 "sd_cache: origin truncated fill for \"%s\" at %O of %O "
232 "bytes - not caching the short object", key,
233 (off_t) fs->off, (off_t) fs->snap.size);
234 free(fs->buf);
235 brix_cstore_fill_abort(fs->staged);
236 brix_sd_obj_release(fs->so);
237 if (!fs->from_cold && !fs->from_peer
238 && sd_cache_stale_serve_ok(st, key))
239 {
240 return NGX_DONE; /* bounded stale-if-error (phase-68) */
241 }
242 errno = EIO;
243 return NGX_ERROR;
244 }
245 break;
246 }
247 if (brix_cstore_fill_write(fs->staged, fs->buf, (size_t) r,
248 fs->off) < 0)
249 {
250 free(fs->buf);
251 brix_cstore_fill_abort(fs->staged);
252 brix_sd_obj_release(fs->so);
253 return NGX_ERROR;
254 }
255 fs->off += r;
256 if (fs->snap.size > 0 && fs->off >= fs->snap.size) {
257 break; /* size known: skip the EOF-probe round-trip */
258 }
259 }
260 free(fs->buf);
261 fs->buf = NULL;
262 /* Capture the origin's advertised content digest BEFORE releasing the source
263 * object — the query needs a live, open object. Only the xroot source offers
264 * an in-band digest (kXR_Qcksum); other backends (http/s3/posix) leave alg/hex
265 * empty and the verify phase decides best-effort/require on that. Skipped
266 * entirely unless a digest-verify policy is in force, so an OFF/CVMFS-CAS fill
267 * pays no round-trip. Mirrors the fetch.c commit-then-verify pattern. */
268 if ((st->policy.verify == BRIX_CACHE_VERIFY_BESTEFFORT
269 || st->policy.verify == BRIX_CACHE_VERIFY_REQUIRE)
270 && ngx_strcmp(brix_sd_backend_name(fs->src), "xroot") == 0)
271 {
272 brix_sd_xroot_query_checksum(fs->so, fs->origin_alg,
273 sizeof(fs->origin_alg), fs->origin_hex, sizeof(fs->origin_hex));
274 }
275 brix_sd_obj_release(fs->so);
276 fs->so = NULL;
277 return NGX_OK;
278}

280/* Emit the unified guard-core audit line (signal=cvmfs_tamper) for a fill
281 * whose bytes failed CVMFS integrity verification — CAS hash or manifest/
282 * whitelist signature mismatch. The tamper actor is UPSTREAM — the origin, or
283 * the mesh sibling a phase-85 F8 peer attempt filled from — never a client
284 * (fills run detached in the thread pool, often coalesced across clients), so
285 * `actor`'s last-answering authority rides the ip field (NULL = st->source)
286 * and the failed object key rides the path. Runs on a fill THREAD: the
287 * timestamp is built with gmtime_r (never the event-loop's cached iso8601),
288 * and ngx_log_error is the same call the surrounding fill code already makes
289 * from this thread. */
290void
291sd_cache_guard_tamper(sd_cache_inst_state *st, brix_sd_instance_t *actor,
292 const char *key)
293{
294 guard_request_t req;
295 char origin[300];
296 char san_key[512];
297 char line[1024];
298 char ts[sizeof("YYYY-MM-DDThh:mm:ss+00:00")];
299 time_t now;
300 struct tm tmv;
301
302 if (actor == NULL) {
303 actor = st->source;
304 }
305 if (actor == NULL
306 || ngx_strcmp(actor->driver->name, "http") != 0
307 || sd_http_last_origin(actor, origin, sizeof(origin)) != 0)
308 {
309 ngx_cpystrn((u_char *) origin, (u_char *) "unknown-origin",
310 sizeof(origin));
311 }
312
313 req.ip = origin;
314 req.proto = "cvmfs";
315 req.op = GUARD_OP_READ;
316 req.path = san_key;
317 req.path_len = brix_sanitize_log_string(key, san_key, sizeof(san_key));
318 req.cred_present = 0;
319 req.outcome = OUTCOME_PENDING;
320 req.status_code = 502; /* the gateway error the waiting client saw */
321
322 now = time(NULL);
323 if (gmtime_r(&now, &tmv) == NULL
324 || strftime(ts, sizeof(ts), "%Y-%m-%dT%H:%M:%S+00:00", &tmv) == 0)
325 {
326 ngx_cpystrn((u_char *) ts, (u_char *) "1970-01-01T00:00:00+00:00",
327 sizeof(ts));
328 }
329
330 if (guard_audit_format(&req, GUARD_R_TAMPER, ts, line, sizeof(line)) > 0) {
331 ngx_log_error(NGX_LOG_WARN, st->log, 0, "%s", line);
332 }
333}

335/* Reject a staged fill whose bytes failed integrity verification: account the
336 * wasted WAN cost + failure metrics, emit signal=cvmfs_tamper, quarantine the
337 * part for evidence and abort the fill. Shared by the CAS digest mismatch and
338 * the manifest signature-chain reject. Always returns NGX_ERROR with errno
339 * EBADMSG (T20 budgets retries on it). */
340static ngx_int_t
341cache_fill_verify_reject(sd_cache_inst_state *st, const char *key,
342 sd_cache_fill_state_t *fs, const char *pp)
343{
344 ngx_brix_cvmfs_repo_metrics_t *rm = sd_cache_repo_metrics(st, key);
345
346 BRIX_CVMFS_METRIC_INC(verify_failures_total);
347 if (rm != NULL) {
348 BRIX_ATOMIC_INC(&rm->verify_failures_total);
349 }
350 if (fs->from_cold) {
351 /* The corrupt bytes came from the LOCAL cold store tier, not the
352 * origin: no WAN accounting, no signal=cvmfs_tamper, no origin
353 * penalty — a false tamper signal would feed the maxretry=1 fail2ban
354 * jail over a local disk fault. Drop the corrupt cold copy so the
355 * origin-fallback refill repopulates both tiers cleanly. */
356 ngx_log_error(NGX_LOG_NOTICE, st->log, 0,
357 "sd_cache: cold-tier object failed verification for \"%s\" - "
358 "removing the cold copy and refilling from the origin", key);
359 if (fs->src != NULL && fs->src->driver->unlink != NULL) {
360 (void) fs->src->driver->unlink(fs->src, key, 0);
361 }
362 } else if (fs->from_peer) {
363 /* The corrupt bytes came from a mesh SIBLING (phase-85 F8): that IS a
364 * remote tamper actor, so raise signal=cvmfs_tamper naming the
365 * sibling's authority — but no origin WAN accounting and no origin
366 * penalty (the origin never served these bytes; the orchestrator
367 * refetches from it next). */
368 ngx_log_error(NGX_LOG_WARN, st->log, 0,
369 "sd_cache: mesh-sibling object failed verification for \"%s\" - "
370 "refetching from the origin", key);
371 sd_cache_guard_tamper(st, fs->src, key);
372 } else {
373 sd_cache_note_origin_bytes(st, key, fs->off); /* WAN cost is WAN cost */
374 /* a verify mismatch IS an upstream fill failure — the bytes came
375 * from the origin but did not publish. */
376 sd_cache_note_upstream(st, 0, fs->off, sd_cache_ms_since(&fs->t0));
377 sd_cache_guard_tamper(st, NULL, key);
378 /* The origin that served these bytes is proven bad for this object —
379 * bench it so the EBADMSG retry (T20 verify budget) rotates to a clean
380 * endpoint instead of re-picking the same corrupt one (no-op for a
381 * non-http source). */
382 sd_http_penalize_last_origin(st->source);
383 }
384 brix_cache_quarantine_part(pp, st->policy.quarantine_dir, st->log);
385 brix_cstore_fill_abort(fs->staged); /* part already renamed away */
386 errno = EBADMSG; /* digest mismatch — T20 budgets retries */
387 return NGX_ERROR;
388}

390/* Case-insensitive equality of two hex digests. Origins vary in hex case
391 * (XRootD lowercases; some HTTP Digest headers upper-case), so a byte-exact
392 * strcmp would false-mismatch identical digests. */
393static int
394sd_cache_hex_ieq(const char *a, const char *b)
395{
396 for (; *a && *b; a++, b++) {
397 if (ngx_tolower((u_char) *a) != ngx_tolower((u_char) *b)) {
398 return 0;
399 }
400 }
401 return *a == '\0' && *b == '\0';
402}

404/* Verify the staged bytes against the origin-advertised digest (best-effort /
405 * require policy). Called only when a digest-verify mode is in force and the
406 * CAS/manifest gates did not apply.
407 *
408 * The origin digest was captured in the pump (fs->origin_alg/hex); here we
409 * digest the staged part with the SAME algorithm and compare.
410 *
411 * - digest matches -> fs->verified = 1, NGX_OK (commit)
412 * - digest MISMATCH -> quarantine + tamper signal, NGX_ERROR
413 * - no origin digest / unknown algorithm:
414 * REQUIRE -> fail closed (EIO) — an unverifiable object must not cache
415 * BEST-EFFORT -> NGX_OK (commit; nothing to check against)
416 *
417 * On every fail-closed path the staged fill is aborted so nothing publishes. */
418static ngx_int_t
419cache_fill_verify_origin(sd_cache_inst_state *st, const char *key,
420 sd_cache_fill_state_t *fs, const char *pp)
421{
422 int require = (st->policy.verify == BRIX_CACHE_VERIFY_REQUIRE);
423 char hex[129];
424 char norm[32];
425 ngx_int_t rc;
426 int fd;
427
428 if (fs->origin_alg[0] == '\0' || fs->origin_hex[0] == '\0') {
429 if (require) {
430 ngx_log_error(NGX_LOG_ERR, st->log, 0,
431 "sd_cache: verify=require but the origin advertised no usable "
432 "digest for \"%s\" - refusing to cache unverifiable bytes", key);
433 brix_cstore_fill_abort(fs->staged);
434 errno = EIO;
435 return NGX_ERROR;
436 }
437 return NGX_OK; /* best-effort: nothing to verify against */
438 }
439 if (pp == NULL) {
440 ngx_log_error(NGX_LOG_ERR, st->log, 0,
441 "sd_cache: cannot locate the staged part to verify \"%s\" - "
442 "failing the fill closed", key);
443 brix_cstore_fill_abort(fs->staged);
444 errno = EIO;
445 return NGX_ERROR;
446 }
447
448 fd = open(pp, O_RDONLY | O_NOFOLLOW | O_CLOEXEC | O_NOCTTY);
449 if (fd < 0) {
450 ngx_log_error(NGX_LOG_ERR, st->log, errno,
451 "sd_cache: verify open of staged part failed for \"%s\"", key);
452 brix_cstore_fill_abort(fs->staged);
453 errno = EIO;
454 return NGX_ERROR;
455 }
456 rc = brix_checksum_hex_name_fd(fs->origin_alg, fd, pp, st->log,
457 hex, sizeof(hex), norm, sizeof(norm));
458 close(fd);
459
460 if (rc == NGX_DECLINED) {
461 /* We do not implement the algorithm the origin advertised. */
462 if (require) {
463 ngx_log_error(NGX_LOG_ERR, st->log, 0,
464 "sd_cache: verify=require but origin algorithm \"%s\" is "
465 "unsupported for \"%s\" - failing closed", fs->origin_alg, key);
466 brix_cstore_fill_abort(fs->staged);
467 errno = EIO;
468 return NGX_ERROR;
469 }
470 return NGX_OK; /* best-effort: cannot check, commit */
471 }
472 if (rc != NGX_OK) {
473 ngx_log_error(NGX_LOG_ERR, st->log, errno,
474 "sd_cache: verify checksum computation failed for \"%s\" - "
475 "failing closed", key);
476 brix_cstore_fill_abort(fs->staged);
477 errno = EIO;
478 return NGX_ERROR;
479 }
480
481 if (!sd_cache_hex_ieq(hex, fs->origin_hex)) {
482 ngx_log_error(NGX_LOG_ERR, st->log, 0,
483 "sd_cache: origin-digest MISMATCH for \"%s\" (%s: origin=%s local=%s)"
484 " - quarantining and failing the fill", key,
485 norm[0] ? norm : fs->origin_alg, fs->origin_hex, hex);
486 return cache_fill_verify_reject(st, key, fs, pp);
487 }
488
489 fs->verified = 1;
490 return NGX_OK;
491}

493/* Digest- and signature-verify the staged bytes before the commit publishes
494 * them.
495 *
496 * WHAT: phase-68 cvmfs-cas verification — the key names its own sha1, so the
497 * staged part can be checked with no origin digest — plus the phase-85
498 * F1 manifest/whitelist signature chain when brix_cvmfs_verify_manifest
499 * configured a master key.
500 *
501 * WHY: A MISMATCH/signature reject is quarantined for evidence, flagged to
502 * the guard (signal=cvmfs_tamper) and the fill fails (the client sees a
503 * gateway error and retries); an ERROR fails closed with no tamper
504 * signal. CAS-unverifiable keys (manifests) get the signature gate
505 * instead; keys under neither gate commit as before.
506 *
507 * HOW: No-op (NGX_OK) unless policy.verify is CVMFS_CAS or a master key is
508 * loaded. Sets fs->verified on a VERIFIED CAS result. On failure aborts
509 * the staged fill and returns NGX_ERROR with errno EBADMSG (mismatch —
510 * T20 budgets retries) or EIO (verify could not run). */
511static ngx_int_t
512cache_fill_verify(sd_cache_inst_state *st, const char *key,
513 sd_cache_fill_state_t *fs)
514{
515 const char *pp;
516 brix_cache_verify_result_e vr;
517
518 fs->verified = 0;
519 if (st->policy.verify == BRIX_CACHE_VERIFY_OFF
520 && st->policy.cvmfs_master_pub == NULL)
521 {
522 return NGX_OK; /* verification disabled — commit as-is */
523 }
524 pp = (fs->staged->inst->driver->staged_path != NULL)
525 ? fs->staged->inst->driver->staged_path(fs->staged) : NULL;
526
527 if (st->policy.verify == BRIX_CACHE_VERIFY_CVMFS_CAS) {
528 vr = (pp != NULL)
529 ? brix_cache_verify_cvmfs_cas(pp, key, st->log, NULL, NULL)
530 : BRIX_CACHE_VERIFY_ERROR;
531 if (vr == BRIX_CACHE_VERIFY_MISMATCH) {
532 return cache_fill_verify_reject(st, key, fs, pp);
533 }
534 if (vr == BRIX_CACHE_VERIFY_ERROR) {
535 ngx_log_error(NGX_LOG_ERR, st->log, errno,
536 "sd_cache: cvmfs-cas verify could not run for \"%s\" - "
537 "failing the fill closed", key);
538 brix_cstore_fill_abort(fs->staged);
539 errno = EIO;
540 return NGX_ERROR;
541 }
542 fs->verified = (vr == BRIX_CACHE_VERIFY_VERIFIED);
543 }
544
545 if (st->policy.cvmfs_master_pub != NULL) {
546 ngx_int_t mv = sd_cache_verify_manifest(st, key, pp);
547
548 if (mv == NGX_ERROR) { /* definitive signature reject */
549 return cache_fill_verify_reject(st, key, fs, pp);
550 }
551 if (mv == NGX_DECLINED) { /* chain not evaluable — no
552 * tamper signal, fail closed */
553 brix_cstore_fill_abort(fs->staged);
554 errno = EIO;
555 return NGX_ERROR;
556 }
557 }

560 * advertised digest (captured in the pump). OFF/CVMFS-CAS never reach here. */
561 if (st->policy.verify == BRIX_CACHE_VERIFY_BESTEFFORT
562 || st->policy.verify == BRIX_CACHE_VERIFY_REQUIRE)
563 {
564 return cache_fill_verify_origin(st, key, fs, pp);
565 }
566 return NGX_OK;
567}

569/* Publish the staged object and record its cinfo.
570 *
571 * WHAT: Accounts the upstream fill, commits (renames) the staged object into
572 * the store, and stores the whole-file COMPLETE cinfo built from the
573 * source stat snapshot (the fstat mode fix — origin perms in ci.mode).
574 *
575 * WHY: The commit phase is the fill's publication point; everything before it
576 * is invisible to readers, so all bookkeeping that describes a
577 * SUCCESSFUL fill belongs here.
578 *
579 * HOW: Whole-file COMPLETE cinfo (the 1 MiB granule keys validity; the present
580 * bitmap is all-set). A partial/slice fill is section 6.5 / SP2. A failed
581 * cinfo store is only a WARN — the object is cached but unrecorded, a
582 * safe miss (refill) next time. */
583static ngx_int_t
584cache_fill_commit(sd_cache_inst_state *st, const char *key,
585 sd_cache_fill_state_t *fs)
586{
587 brix_cache_cinfo_t ci;
588
589 if (!fs->from_cold && !fs->from_peer) {
590 /* a cold promote or a sibling fetch moved zero origin-WAN bytes */
591 sd_cache_note_origin_bytes(st, key, fs->off); /* WAN in, this attempt */
592 sd_cache_note_upstream(st, 1, fs->off, sd_cache_ms_since(&fs->t0));
593 }
594
595 if (brix_cstore_fill_commit(fs->staged) != NGX_OK) {
596 return NGX_ERROR;
597 }
598
599 ngx_memzero(&ci, sizeof(ci));
600 ci.magic = BRIX_CACHE_CINFO_MAGIC;
601 ci.version = BRIX_CACHE_CINFO_VERSION;
602 ci.block_size = BRIX_CACHE_DIRTY_BLOCK;
603 ci.size = (uint64_t) fs->off;
604 ci.mtime = (uint64_t) fs->snap.mtime;
605 ci.mode = (uint32_t) (fs->snap.mode & 0777); /* origin perms — served back
606 * so a read-only source is not
607 * masked by the owner-writable
608 * physical store object. */
609 ci.nblocks = brix_cache_cinfo_nblocks((uint64_t) fs->off, ci.block_size);
610 ci.flags = BRIX_CINFO_F_COMPLETE
611 | (fs->verified ? BRIX_CINFO_F_VERIFIED : 0);
612 ci.filled_at = (uint64_t) time(NULL);
613 if (st->policy.cvmfs_manifest_ttl > 0 && sd_cache_is_manifest_key(key)) {
614 brix_cache_cinfo_set_expires(&ci,
615 (time_t) ci.filled_at + st->policy.cvmfs_manifest_ttl);
616 }
617
618 if (brix_cstore_cinfo_store(&st->cstore, key, &ci) != NGX_OK) {
619 /* The object is cached but unrecorded - a safe miss (refill) next time. */
620 ngx_log_error(NGX_LOG_WARN, st->log, errno,
621 "sd_cache: cinfo store failed for \"%s\" - will refill on next read",
622 key);
623 }
624 return NGX_OK;
625}

627/* One fill attempt from the tier `src` (the wrapped source, the cold store on
628 * a phase-85 F7 promote, or a mesh sibling on a phase-85 F8 peer fetch):
629 * acquire (source open + admission + staged fill-open) -> pump (read loop) ->
630 * verify (phase-68 cvmfs-cas) -> commit (publish + cinfo). NGX_DONE from
631 * acquire/pump means a bounded stale copy was served — reported as NGX_OK
632 * without caching (origin attempts only; a cold or peer attempt never
633 * stale-serves so its failure falls back to the origin). */
634static ngx_int_t
635sd_cache_fill_attempt(sd_cache_inst_state *st, const char *key,
636 const brix_sd_cred_t *cred, brix_sd_instance_t *src, int from_cold,
637 int from_peer)
638{
639 sd_cache_fill_state_t fs;
640 ngx_int_t rc;
641
642 ngx_memzero(&fs, sizeof(fs));
643 fs.src = src;
644 fs.from_cold = from_cold;
645 fs.from_peer = from_peer;
646 (void) clock_gettime(CLOCK_MONOTONIC, &fs.t0);
647
648 rc = cache_fill_acquire(st, key, cred, &fs);
649 if (rc == NGX_DONE) {
650 return NGX_OK; /* bounded stale-if-error (phase-68) */
651 }
652 if (rc != NGX_OK) {
653 return rc; /* NGX_DECLINED or NGX_ERROR */
654 }
655
656 rc = cache_fill_pump(st, key, &fs);
657 if (rc == NGX_DONE) {
658 return NGX_OK; /* bounded stale-if-error (phase-68) */
659 }
660 if (rc != NGX_OK) {
661 return NGX_ERROR;
662 }
663
664 if (cache_fill_verify(st, key, &fs) != NGX_OK) {
665 return NGX_ERROR;
666 }
667 return cache_fill_commit(st, key, &fs);
668}

670/* Rendezvous (highest-random-weight) owner of `key` over the peer ring:
671 * FNV-1a 64 over "<label>\n<key>", owner = argmax across members. Every mesh
672 * node carries the IDENTICAL member list with identical labels, so all nodes
673 * agree on the owner with no coordination. The scheme (FNV-1a 64, offset
674 * 14695981039346656037, prime 1099511628211, '\n' separator, ties to the
675 * lower index) is mirrored bit-for-bit by tests/test_cvmfs_peer_mesh.py —
676 * change both together or not at all. */
677static uint64_t
678sd_cache_hrw_fnv1a64(const char *label, const char *key)
679{
680 uint64_t h = 14695981039346656037ULL;
681 const char *p;
682
683 for (p = label; *p != '\0'; p++) {
684 h = (h ^ (uint64_t) (unsigned char) *p) * 1099511628211ULL;
685 }
686 h = (h ^ (uint64_t) '\n') * 1099511628211ULL;
687 for (p = key; *p != '\0'; p++) {
688 h = (h ^ (uint64_t) (unsigned char) *p) * 1099511628211ULL;
689 }
690 return h;
691}

693static int
694sd_cache_peer_owner(const sd_cache_inst_state *st, const char *key)
695{
696 int i, owner = 0;
697 uint64_t best = 0, w;
698
699 for (i = 0; i < st->n_peers; i++) {
700 w = sd_cache_hrw_fnv1a64(st->peers[i].label, key);
701 if (i == 0 || w > best) {
702 best = w;
703 owner = i;
704 }
705 }
706 return owner;
707}

709/* Fill `key` into the cache store and record its cinfo.
710 *
711 * WHAT: Pulls the object `key` into the cache store, building a whole-file
712 * cinfo on success. With a cold store tier attached (phase-85 F7) a
713 * verified PROMOTE from the cold copy is attempted first; with a sibling
714 * mesh attached (phase-85 F8) the key's rendezvous-owning peer is tried
715 * next; the origin (st->source) is the fallback and the only tier that
716 * stale-serves.
717 *
718 * WHY: The fill spine is the single path that populates the cache; threading
719 * the caller's per-user `cred` here ensures the source open uses the
720 * correct identity rather than the service credential. Routing the
721 * promote through the SAME spine means a cold object passes the identical
722 * cvmfs-cas / manifest-signature verify gate as an origin fill — a
723 * corrupt cold copy can never publish.
724 *
725 * HOW: Cold attempt (when st->cold): any failure but admission silently falls
726 * back to the origin; success drops the cold copy (move semantics).
727 * Peer attempt (when the ring owner of `key` is a non-self member with a
728 * built instance): one verified fetch from that sibling; any failure but
729 * admission silently falls back to the origin. Only the configured ring
730 * members are ever contacted (allowlist by construction — no request-
731 * derived authority reaches a peer fetch). Returns NGX_OK (object cached
732 * + cinfo stored, or stale-served), NGX_DECLINED (policy declined), or
733 * NGX_ERROR (failure). */
734ngx_int_t
735sd_cache_fill(sd_cache_inst_state *st, const char *key,
736 const brix_sd_cred_t *cred)
737{
738 if (st->cold != NULL) {
739 ngx_int_t rc = sd_cache_fill_attempt(st, key, cred, st->cold, 1, 0);
740
741 if (rc == NGX_OK) {
742 /* Promoted: the hot tier owns the object now — drop the cold copy
743 * (best-effort; a leftover is simply re-evicted or re-demoted). */
744 if (st->cold->driver->unlink != NULL) {
745 (void) st->cold->driver->unlink(st->cold, key, 0);
746 }
747 ngx_log_error(NGX_LOG_INFO, st->log, 0,
748 "sd_cache: promoted \"%s\" from the cold tier", key);
749 return NGX_OK;
750 }
751 if (rc == NGX_DECLINED) {
752 return rc; /* admission policy — identical for the origin */
753 }
754 /* cold miss / outage / corrupt copy: fall back to the origin fill */
755 }
756
757 if (st->n_peers > 0) {
758 int owner = sd_cache_peer_owner(st, key);
759
760 if (owner != st->peer_self && st->peers[owner].inst != NULL) {
761 ngx_int_t rc = sd_cache_fill_attempt(st, key, cred,
762 st->peers[owner].inst, 0, 1);
763
764 if (rc == NGX_OK) {
765 ngx_log_error(NGX_LOG_INFO, st->log, 0,
766 "sd_cache: filled \"%s\" from mesh sibling %s",
767 key, st->peers[owner].label);
768 return NGX_OK;
769 }
770 if (rc == NGX_DECLINED) {
771 return rc; /* admission policy — identical for the origin */
772 }
773 /* sibling miss / outage / corrupt copy: fall back to the origin */
774 }
775 }
776
777 return sd_cache_fill_attempt(st, key, cred, st->source, 0, 0);
778}

780/* Demote the HOT cached object `key` into the cold store tier (phase-85 F7):
781 * copy its bytes from the cache store into the cold store via the cold
782 * driver's staged spine (write + commit — never a torn cold object). Called by
783 * the eviction engine on space-pressure victims ONLY, just before the hot copy
784 * is removed; write invalidation never demotes (a written-over object is stale
785 * and must not survive in cold). Defined here (not sd_cache.c) to reuse the
786 * fill spine's move-granule idiom. NGX_OK / NGX_DECLINED (no cold tier — not
787 * an error) / NGX_ERROR with errno set (the caller evicts anyway: space relief
788 * wins and the origin refill preserves correctness). */
789ngx_int_t
790brix_sd_cache_demote(brix_sd_instance_t *inst, const char *key)
791{
792 sd_cache_inst_state *st;
793 brix_sd_instance_t *hot, *cold;
794 brix_sd_obj_t *so;
795 brix_sd_staged_t *sg;
796 u_char *buf;
797 off_t off = 0;
798 int err = 0;
799
800 if (!brix_sd_cache_instance_is(inst) || key == NULL) {
801 return NGX_DECLINED;
802 }
803 st = SD_CACHE_ST(inst);
804 cold = st->cold;
805 if (cold == NULL) {
806 return NGX_DECLINED;
807 }
808 hot = st->cstore.store;
809 if (hot->driver->open == NULL || cold->driver->staged_open == NULL) {
810 errno = ENOSYS;
811 return NGX_ERROR;
812 }
813 so = hot->driver->open(hot, key, BRIX_SD_O_READ, 0, &err);
814 if (so == NULL) {
815 errno = err ? err : EIO;
816 return NGX_ERROR;
817 }
818 if (so->driver->pread == NULL) {
819 brix_sd_obj_release(so);
820 errno = ENOSYS;
821 return NGX_ERROR;
822 }
823 /* 0600 like the hot store object: the cold tier aggregates many users'
824 * bytes under the svc identity too; client-facing perms live in the cinfo
825 * the PROMOTE rebuilds, never on the physical cold object. */
826 sg = cold->driver->staged_open(cold, key, S_IRUSR | S_IWUSR, &err);
827 if (sg == NULL) {
828 brix_sd_obj_release(so);
829 errno = err ? err : EIO;
830 return NGX_ERROR;
831 }
832 buf = malloc(SD_CACHE_CHUNK);
833 if (buf == NULL) {
834 sg->inst->driver->staged_abort(sg);
835 brix_sd_obj_release(so);
836 errno = ENOMEM;
837 return NGX_ERROR;
838 }
839 for ( ;; ) {
840 ssize_t r = so->driver->pread(so, buf, SD_CACHE_CHUNK, off);
841
842 if (r < 0) {
843 if (errno == EINTR) {
844 continue;
845 }
846 break;
847 }
848 if (r == 0) { /* clean EOF */
849 free(buf);
850 brix_sd_obj_release(so);
851 if (sg->inst->driver->staged_commit(sg, 0) != NGX_OK) {
852 return NGX_ERROR;
853 }
854 return NGX_OK;
855 }
856 if (sg->inst->driver->staged_write(sg, buf, (size_t) r, off) < 0) {
857 break;
858 }
859 off += r;
860 }
861 err = errno;
862 free(buf);
863 sg->inst->driver->staged_abort(sg);
864 brix_sd_obj_release(so);
865 errno = err ? err : EIO;
866 return NGX_ERROR;
867}
