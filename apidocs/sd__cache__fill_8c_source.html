1/*
2 * sd_cache_fill.c — the whole-file fill spine of the read-through cache.
3 *
4 * WHAT: Pulls a COMPLETE object from the wrapped source into the cache store and
5 * records its cinfo. Owns the four fill phases (acquire -> pump -> verify
6 * -> commit) and the sd_cache_fill orchestrator the interposed read-open
7 * and the async-offload entrypoint both drive.
8 *
9 * WHY: Split from sd_cache.c (phase-79) to keep every cache file under the
10 * ~500-line, one-concept-per-file cap. The fill spine is the single path
11 * that populates the cache; isolating it keeps the origin-read / stale-serve
12 * / verify / cinfo-publish logic reviewable apart from the vtable adapters
13 * (sd_cache.c), the slice/partial machinery (sd_cache_partial.c), and the
14 * namespace/staged forwarders (sd_cache_forward.c).
15 *
16 * HOW: cache_fill_acquire (source open + admission + staged fill-open) -> pump
17 * (origin read loop) -> verify (phase-68 cvmfs-cas) -> commit (publish +
18 * cinfo). sd_cache_fill is the only non-static symbol here; sd_cache.c
19 * reaches it (miss-serve and the async fill-key) through
20 * sd_cache_internal.h. The admission/policy + repo-metrics helpers it calls
21 * live in sd_cache_policy.c. ZERO behaviour change from the pre-split file.
22 */

47 * verify -> commit). Owned by sd_cache_fill; each phase helper consumes or

49typedef struct {

59/* Acquire everything a fill needs: source object, stat snapshot, staged store
60 * object, and the move buffer.
61 *
62 * WHAT: Opens the source for `key`, snapshots its stat, checks admission, and
63 * opens the staged cache-store object (which takes the fill lock — the
64 * O_EXCL + dead-owner-reclaim logic lives in the cstore layer, frozen).
65 *
66 * WHY: The acquire phase concentrates all the "can this fill happen at all?"
67 * decisions so the pump/commit phases run against a fully-provisioned
68 * state and can stay linear.
69 *
70 * HOW: brix_sd_open_maybe_cred so drivers with an open_cred slot use the
71 * per-user proxy. Returns NGX_OK (fs populated), NGX_DONE (stale copy
72 * served — caller reports success without filling), NGX_DECLINED
73 * (admission declined), or NGX_ERROR with errno set; every failure path
74 * releases whatever it had already acquired. */
75static ngx_int_t
76cache_fill_acquire(sd_cache_inst_state *st, const char *key,
77 const brix_sd_cred_t *cred, sd_cache_fill_state_t *fs)
78{
79 brix_sd_instance_t *src = st->source;
80 mode_t fmode;
81 int err = 0;
82
83 if (src->driver->open == NULL) {
84 errno = ENOSYS;
85 return NGX_ERROR;
86 }
87 fs->so = brix_sd_open_maybe_cred(src, key, BRIX_SD_O_READ, 0, cred, &err);
88 if (fs->so == NULL) {
89 if (sd_cache_stale_serve_ok(st, key)) {
90 return NGX_DONE; /* bounded stale-if-error (phase-68) */
91 }
92 errno = err ? err : EIO;
93 return NGX_ERROR;
94 }
95 /* Every op on the OPEN OBJECT dispatches through so->driver, NOT src->driver:
96 * a decorator source (sd_stage) legitimately returns the tier-below's object
97 * from a read open ("read byte-I/O bypasses the decorator"), so the object's
98 * vtable can differ from the instance's — dispatching object ops through the
99 * instance vtable reinterprets a foreign object state (type confusion). */
100 if (fs->so->driver->pread == NULL) {
101 brix_sd_obj_release(fs->so);
102 errno = ENOSYS;
103 return NGX_ERROR;
104 }
105 /* open() does not guarantee a populated snap (the posix driver fstats lazily),
106 * so fstat the object for an accurate size/mtime/mode - the cinfo validity and
107 * the cached file's permission bits both depend on it. */
108 fs->snap = fs->so->snap;
109 if (fs->so->driver->fstat != NULL) {
110 (void) fs->so->driver->fstat(fs->so, &fs->snap);
111 }
112 /* SECURITY + correctness: the physical cache-store object is a svc-owned
113 * artifact that aggregates MANY users' bytes under one tree. Per-user
114 * authorization is enforced at the protocol gate (open_cache.c), and the
115 * CLIENT-FACING mode is carried in the cinfo record and served by
116 * sd_cache_stat() — decoupled from this physical mode. So force 0600:
117 * - owner rw is REQUIRED (XATTR meta_mode stores user.xrd.cinfo ON this
118 * object; Linux refuses user.* xattrs on a non-writable inode — a
119 * read-only 0444 source would otherwise block cinfo persistence, G3);
120 * - NO group/other bits, so a mapped low-priv uid cannot read another
121 * user's cached bytes by direct filesystem access (the source mode's
122 * 0644 previously leaked here). snap.mode still reaches clients via the
123 * cinfo record (ci.mode below, served by sd_cache_stat). */
124 fmode = S_IRUSR | S_IWUSR; /* 0600 — svc-owned cache artifact */
125
126 if (!sd_cache_admit(&st->policy, key, fs->snap.size)) {
127 brix_sd_obj_release(fs->so);
128 return NGX_DECLINED; /* too big / filtered - do not cache */
129 }
130
131 fs->staged = brix_cstore_fill_open(&st->cstore, key, fmode);
132 if (fs->staged == NULL) {
133 ngx_log_error(NGX_LOG_WARN, st->log, errno,
134 "sd_cache: fill_open on the cache store failed for \"%s\" - not cached",
135 key);
136 brix_sd_obj_release(fs->so);
137 return NGX_ERROR;
138 }
139 fs->buf = malloc(SD_CACHE_CHUNK);
140 if (fs->buf == NULL) {
141 brix_cstore_fill_abort(fs->staged);
142 brix_sd_obj_release(fs->so);
143 errno = ENOMEM;
144 return NGX_ERROR;
145 }
146 return NGX_OK;
147}

149/* Pump the source object's bytes into the staged store object.
150 *
151 * WHAT: The origin read loop — pread from the source, fill_write to the staged
152 * object, chunk by chunk, advancing fs->off.
153 *
154 * WHY: The pump is the fill's only long-running phase; isolating it keeps the
155 * read-error / stale-serve / errno-preservation logic in one place.
156 *
157 * HOW: On success frees the move buffer and closes the source (the staged
158 * object stays open for verify/commit) and returns NGX_OK. On a read or
159 * write failure it releases EVERYTHING (buffer, staged fill, source) and
160 * returns NGX_DONE if a bounded stale copy can be served, else NGX_ERROR
161 * with the READ's errno preserved. */
162static ngx_int_t
163cache_fill_pump(sd_cache_inst_state *st, const char *key,
164 sd_cache_fill_state_t *fs)
165{
166 for ( ;; ) {
167 ssize_t r = fs->so->driver->pread(fs->so, fs->buf, SD_CACHE_CHUNK,
168 fs->off);
169
170 if (r < 0) {
171 int read_err = errno; /* capture BEFORE any cleanup call */
172
173 if (read_err == EINTR) {
174 continue;
175 }
176 free(fs->buf);
177 brix_cstore_fill_abort(fs->staged);
178 brix_sd_obj_release(fs->so);
179 if (sd_cache_stale_serve_ok(st, key)) {
180 return NGX_DONE; /* bounded stale-if-error (phase-68) */
181 }
182 /* Restore the READ's errno. sd_cache_stale_serve_ok() (and the
183 * cleanup calls) stat the absent cache entry and leave errno
184 * ENOENT — propagating that would make the fill layer classify a
185 * merely-broken origin connection (EIO/ETIMEDOUT) as the origin's
186 * definitive 404, turning a transient network fault into a
187 * cached, client-poisoning "not found" instead of a retry. */
188 errno = read_err;
189 return NGX_ERROR;
190 }
191 if (r == 0) {
192 break;
193 }
194 if (brix_cstore_fill_write(fs->staged, fs->buf, (size_t) r,
195 fs->off) < 0)
196 {
197 free(fs->buf);
198 brix_cstore_fill_abort(fs->staged);
199 brix_sd_obj_release(fs->so);
200 return NGX_ERROR;
201 }
202 fs->off += r;
203 if (fs->snap.size > 0 && fs->off >= fs->snap.size) {
204 break; /* size known: skip the EOF-probe round-trip */
205 }
206 }
207 free(fs->buf);
208 fs->buf = NULL;
209 brix_sd_obj_release(fs->so);
210 fs->so = NULL;
211 return NGX_OK;
212}

214/* Digest-verify the staged bytes before the commit publishes them.
215 *
216 * WHAT: phase-68 cvmfs-cas verification — the key names its own sha1, so the
217 * staged part can be checked with no origin digest.
218 *
219 * WHY: A MISMATCH is quarantined for evidence and the fill fails (the client
220 * sees a gateway error and retries); an ERROR fails closed. Non-CAS keys
221 * (manifests) come back UNVERIFIED and commit as before.
222 *
223 * HOW: No-op (NGX_OK) unless policy.verify is CVMFS_CAS. Sets fs->verified on
224 * a VERIFIED result. On failure aborts the staged fill and returns
225 * NGX_ERROR with errno EBADMSG (mismatch — T20 budgets retries) or EIO
226 * (verify could not run). */
227static ngx_int_t
228cache_fill_verify(sd_cache_inst_state *st, const char *key,
229 sd_cache_fill_state_t *fs)
230{
231 const char *pp;
232 brix_cache_verify_result_e vr;
233
234 fs->verified = 0;
235 if (st->policy.verify != BRIX_CACHE_VERIFY_CVMFS_CAS) {
236 return NGX_OK;
237 }
238 pp = (fs->staged->inst->driver->staged_path != NULL)
239 ? fs->staged->inst->driver->staged_path(fs->staged) : NULL;
240 vr = (pp != NULL)
241 ? brix_cache_verify_cvmfs_cas(pp, key, st->log, NULL, NULL)
242 : BRIX_CACHE_VERIFY_ERROR;
243 if (vr == BRIX_CACHE_VERIFY_MISMATCH) {
244 ngx_brix_cvmfs_repo_metrics_t *rm = sd_cache_repo_metrics(st, key);
245
246 BRIX_CVMFS_METRIC_INC(verify_failures_total);
247 if (rm != NULL) {
248 BRIX_ATOMIC_INC(&rm->verify_failures_total);
249 }
250 sd_cache_note_origin_bytes(st, key, fs->off); /* WAN cost is WAN cost */
251 /* a verify mismatch IS an upstream fill failure — the bytes came
252 * from the origin but did not publish. */
253 sd_cache_note_upstream(st, 0, fs->off, sd_cache_ms_since(&fs->t0));
254 brix_cache_quarantine_part(pp, st->policy.quarantine_dir, st->log);
255 brix_cstore_fill_abort(fs->staged); /* part already renamed away */
256 errno = EBADMSG; /* digest mismatch — T20 budgets retries */
257 return NGX_ERROR;
258 }
259 if (vr == BRIX_CACHE_VERIFY_ERROR) {
260 ngx_log_error(NGX_LOG_ERR, st->log, errno,
261 "sd_cache: cvmfs-cas verify could not run for \"%s\" - "
262 "failing the fill closed", key);
263 brix_cstore_fill_abort(fs->staged);
264 errno = EIO;
265 return NGX_ERROR;
266 }
267 fs->verified = (vr == BRIX_CACHE_VERIFY_VERIFIED);
268 return NGX_OK;
269}

271/* Publish the staged object and record its cinfo.
272 *
273 * WHAT: Accounts the upstream fill, commits (renames) the staged object into
274 * the store, and stores the whole-file COMPLETE cinfo built from the
275 * source stat snapshot (the fstat mode fix — origin perms in ci.mode).
276 *
277 * WHY: The commit phase is the fill's publication point; everything before it
278 * is invisible to readers, so all bookkeeping that describes a
279 * SUCCESSFUL fill belongs here.
280 *
281 * HOW: Whole-file COMPLETE cinfo (the 1 MiB granule keys validity; the present
282 * bitmap is all-set). A partial/slice fill is section 6.5 / SP2. A failed
283 * cinfo store is only a WARN — the object is cached but unrecorded, a
284 * safe miss (refill) next time. */
285static ngx_int_t
286cache_fill_commit(sd_cache_inst_state *st, const char *key,
287 sd_cache_fill_state_t *fs)
288{
289 brix_cache_cinfo_t ci;
290
291 sd_cache_note_origin_bytes(st, key, fs->off); /* WAN in, this attempt */
292 sd_cache_note_upstream(st, 1, fs->off, sd_cache_ms_since(&fs->t0));
293
294 if (brix_cstore_fill_commit(fs->staged) != NGX_OK) {
295 return NGX_ERROR;
296 }
297
298 ngx_memzero(&ci, sizeof(ci));
299 ci.magic = BRIX_CACHE_CINFO_MAGIC;
300 ci.version = BRIX_CACHE_CINFO_VERSION;
301 ci.block_size = BRIX_CACHE_DIRTY_BLOCK;
302 ci.size = (uint64_t) fs->off;
303 ci.mtime = (uint64_t) fs->snap.mtime;
304 ci.mode = (uint32_t) (fs->snap.mode & 0777); /* origin perms — served back
305 * so a read-only source is not
306 * masked by the owner-writable
307 * physical store object. */
308 ci.nblocks = brix_cache_cinfo_nblocks((uint64_t) fs->off, ci.block_size);
309 ci.flags = BRIX_CINFO_F_COMPLETE
310 | (fs->verified ? BRIX_CINFO_F_VERIFIED : 0);
311 ci.filled_at = (uint64_t) time(NULL);
312 if (st->policy.cvmfs_manifest_ttl > 0 && sd_cache_is_manifest_key(key)) {
313 brix_cache_cinfo_set_expires(&ci,
314 (time_t) ci.filled_at + st->policy.cvmfs_manifest_ttl);
315 }
316
317 if (brix_cstore_cinfo_store(&st->cstore, key, &ci) != NGX_OK) {
318 /* The object is cached but unrecorded - a safe miss (refill) next time. */
319 ngx_log_error(NGX_LOG_WARN, st->log, errno,
320 "sd_cache: cinfo store failed for \"%s\" - will refill on next read",
321 key);
322 }
323 return NGX_OK;
324}

326/* Fill `key` from the source into the cache store and record its cinfo.
327 *
328 * WHAT: Pulls the object `key` from `st->source` into the cache store,
329 * building a whole-file cinfo on success.
330 *
331 * WHY: The fill spine is the single path that populates the cache; threading
332 * the caller's per-user `cred` here ensures the source open uses the
333 * correct identity rather than the service credential.
334 *
335 * HOW: acquire (source open + admission + staged fill-open) -> pump (origin
336 * read loop) -> verify (phase-68 cvmfs-cas) -> commit (publish + cinfo).
337 * NGX_DONE from acquire/pump means a bounded stale copy was served —
338 * reported as NGX_OK without caching. Returns NGX_OK (object cached +
339 * cinfo stored, or stale-served), NGX_DECLINED (policy declined), or
340 * NGX_ERROR (failure). */
341ngx_int_t
342sd_cache_fill(sd_cache_inst_state *st, const char *key,
343 const brix_sd_cred_t *cred)
344{
345 sd_cache_fill_state_t fs;
346 ngx_int_t rc;
347
348 ngx_memzero(&fs, sizeof(fs));
349 (void) clock_gettime(CLOCK_MONOTONIC, &fs.t0);
350
351 rc = cache_fill_acquire(st, key, cred, &fs);
352 if (rc == NGX_DONE) {
353 return NGX_OK; /* bounded stale-if-error (phase-68) */
354 }
355 if (rc != NGX_OK) {
356 return rc; /* NGX_DECLINED or NGX_ERROR */
357 }
358
359 rc = cache_fill_pump(st, key, &fs);
360 if (rc == NGX_DONE) {
361 return NGX_OK; /* bounded stale-if-error (phase-68) */
362 }
363 if (rc != NGX_OK) {
364 return NGX_ERROR;
365 }
366
367 if (cache_fill_verify(st, key, &fs) != NGX_OK) {
368 return NGX_ERROR;
369 }
370 return cache_fill_commit(st, key, &fs);
371}
