1/*
2 * namespace_ops_copy.c — shared namespace service: single-file local copy path.
3 *
4 * WHAT: Implements the local file-copy orchestration (brix_ns_local_copy) and
5 * its pre-open validation, destination-open, and finalize helpers, plus the
6 * xattr-mirroring helpers (brix_ns_copy_fattrs / brix_xattr_copy_by_prefix).
7 * WHY: Split out of namespace_ops.c (phase-79 file-size cap): the copy path is a
8 * self-contained concern — stat/overwrite policy, staged-vs-direct dst open,
9 * copy_file_range + commit + xattr mirror — that is larger than the
10 * delete/mkdir/rename core it sits beside. Keeping it here holds both files
11 * under the 500-line limit while leaving the confinement primitives it reuses
12 * (ns_rel/ns_set_err) single-sourced in the core.
13 * HOW: These helpers operate on already-resolved paths; every filesystem touch is
14 * confined by the openat2 RESOLVE_BENEATH primitives (ns_rel opens the rootfd
15 * and strips the within-root tail; brix_open_beneath/brix_stat_beneath act
16 * through it). Protocol handlers are responsible for wire parsing, auth, and
17 * lock checks.
18 */

35/*
36 * Confinement model for this file (Phase 8 — openat2 RESOLVE_BENEATH).
37 *
38 * Every path that reaches these helpers is a client-derived path under the
39 * export root. Confinement is established exactly as in namespace_ops.c:
40 * ns_rel() opens a kernel-confinement rootfd on root_canon and returns the
41 * within-root tail; every stat/open on a path routes through the beneath API so
42 * escape is blocked atomically by the kernel.
43 *
44 * Where beneath is deliberately NOT used (see the inline notes below):
45 * - brix_copy_range(): operates on already-open file descriptors, not paths.
46 * Confinement happened when those fds were obtained via brix_open_beneath;
47 * a path-anchored RESOLVE_BENEATH check is meaningless on an fd.
48 * - brix_ns_copy_fattrs()/listxattr/getxattr/setxattr: the xattr syscalls are
49 * name+path based and have no openat2/RESOLVE_BENEATH-equivalent that takes a
50 * dirfd. They run on src/dst paths that the immediately-preceding beneath
51 * open already proved resident under the root, so no new escape surface is
52 * introduced; switching them to an fd-based fsetxattr/fgetxattr is a possible
53 * future hardening but is not a confinement gap today.
54 */

56void
57brix_xattr_copy_by_prefix(ngx_log_t *log,
58 const char *src, const char *dst,
59 const char *prefix, size_t prefix_len,
60 size_t value_max)
61{
62 ssize_t list_len, vlen;
63 char *list;
64 char *p;
65 u_char *value;
66
67 list_len = listxattr(src, NULL, 0);
68 if (list_len <= 0) {
69 return;
70 }
71
72 list = ngx_alloc((size_t) list_len, log);
73 if (list == NULL) {
74 return;
75 }
76
77 list_len = listxattr(src, list, (size_t) list_len);
78 if (list_len <= 0) {
79 ngx_free(list);
80 return;
81 }
82
83 value = ngx_alloc(value_max, log);
84 if (value == NULL) {
85 ngx_free(list);
86 return;
87 }
88
89 for (p = list; p < list + list_len; p += strlen(p) + 1) {
90 if (strncmp(p, prefix, prefix_len) != 0) {
91 continue;
92 }
93 vlen = getxattr(src, p, value, value_max);
94 if (vlen >= 0) {
95 (void) setxattr(dst, p, value, (size_t) vlen, 0);
96 }
97 }
98
99 ngx_free(value);
100 ngx_free(list);
101}

111/*
112 * ns_copy_validate — stat the copy endpoints and enforce pre-open policy.
113 *
114 * WHAT: Confines a stat on src and dst beneath rootfd, rejects a directory
115 * source (BRIX_NS_CONFLICT), and applies the overwrite / overwrite_dirs
116 * gates for an existing destination. Fills *ssb with the source stat and
117 * sets res->existed/was_dir. Returns 1 when the caller must return res
118 * immediately, 0 when the copy should proceed.
119 * WHY: Extracts the entire pre-open decision tree from the orchestrator so it
120 * stays flat and under the complexity cap; the confined stats and their
121 * ordering are unchanged.
122 * HOW: 1. stat_beneath src; error → record errno. 2. Reject a directory source.
123 * 3. If dst exists, mark existed; refuse when overwrite is off; for a
124 * directory dst mark was_dir and refuse when overwrite_dirs is off.
125 */
126static ngx_flag_t
127ns_copy_validate(int rootfd, const char *src_rel, const char *dst_rel,
128 const brix_ns_copy_opts_t *opts, struct stat *ssb, brix_ns_result_t *res)
129{
130 struct stat dsb;
131
132 if (brix_stat_beneath(rootfd, src_rel, ssb) != 0) {
133 ns_set_err(res, errno);
134 return 1;
135 }
136
137 if (S_ISDIR(ssb->st_mode)) {
138 res->status = BRIX_NS_CONFLICT; /* COPY on collection not yet shared */
139 return 1;
140 }
141
142 if (brix_stat_beneath(rootfd, dst_rel, &dsb) == 0) {
143 res->existed = 1;
144 if (!opts->overwrite) {
145 res->status = BRIX_NS_EXISTS;
146 return 1;
147 }
148 if (S_ISDIR(dsb.st_mode)) {
149 res->was_dir = 1;
150 if (!opts->overwrite_dirs) {
151 res->status = BRIX_NS_EXISTS;
152 return 1;
153 }
154 }
155 }
156
157 return 0;
158}

160/*
161 * ns_copy_open_dst — open the copy destination, staged or direct.
162 *
163 * WHAT: Opens dst either through the staging helper (staged_commit) or via a
164 * direct confined create. On success writes *dst_fd_out, sets
165 * *use_staging_out for the staged path, and (staged) fills *staged.
166 * Returns 1 with res mapped from errno on failure, 0 on success.
167 * WHY: The two destination-open strategies plus their error mapping are their
168 * own concern; isolating them keeps the orchestrator's cleanup linear.
169 * HOW: 1. staged_commit → brix_staged_open (temp under root), publish its fd and
170 * mark staging. 2. Otherwise → brix_open_beneath with O_WRONLY|O_CREAT|
171 * O_TRUNC at the source mode. 3. Map any errno onto res.
172 */
173static ngx_flag_t
174ns_copy_open_dst(ngx_log_t *log, int rootfd, const char *root_canon,
175 const char *dst, const char *dst_rel, mode_t mode,
176 const brix_ns_copy_opts_t *opts, brix_staged_file_t *staged,
177 int *dst_fd_out, ngx_flag_t *use_staging_out, brix_ns_result_t *res)
178{
179 if (opts->staged_commit) {
180 /* staged_file confines its own temp open/rename internally. */
181 brix_staged_open_req_t staged_req = {
182 .root_canon = root_canon,
183 .final_path = dst,
184 .mode = mode,
185 .open_flags = O_WRONLY | O_CREAT | O_TRUNC,
186 .attempts = 16,
187 };
188 if (brix_staged_open(log, &staged_req, staged) != NGX_OK) {
189 ns_set_err(res, errno);
190 return 1;
191 }
192 *dst_fd_out = staged->fd;
193 *use_staging_out = 1;
194 return 0;
195 }
196
197 *dst_fd_out = brix_open_beneath(rootfd, dst_rel,
198 O_WRONLY | O_CREAT | O_TRUNC, mode);
199 if (*dst_fd_out < 0) {
200 ns_set_err(res, errno);
201 return 1;
202 }
203
204 return 0;
205}

207/*
208 * ns_copy_finalize — copy the bytes, commit the destination, copy xattrs.
209 *
210 * WHAT: Copies ssb->st_size bytes from src_fd to dst_fd, then commits the
211 * staged file (or closes a direct dst), and optionally mirrors fattr
212 * xattrs. Fills res with BRIX_NS_OK/created on success or the mapped
213 * errno on failure. Owns and closes src_fd/dst_fd; leaves rootfd to the
214 * caller.
215 * WHY: Concentrates the data-plane copy and its resource cleanup so the
216 * orchestrator does not carry the multi-path close/abort logic; all
217 * confinement was established by the fds' beneath/staged opens.
218 * HOW: 1. brix_copy_range on the fds; on failure close src_fd and abort-or-close
219 * dst. 2. Close src_fd. 3. Commit the staged file (error → return) or
220 * close the direct dst_fd. 4. Copy fattr xattrs when requested.
221 */
222static void
223ns_copy_finalize(ngx_log_t *log, const char *root_canon, const char *src,
224 const char *dst, int src_fd, int dst_fd, const struct stat *ssb,
225 const brix_ns_copy_opts_t *opts, ngx_flag_t use_staging,
226 brix_staged_file_t *staged, brix_ns_result_t *res)
227{
228 /*
229 * NOT a beneath site: brix_copy_range operates on the two file
230 * descriptors obtained above (src_fd, dst_fd). Both fds were opened
231 * through brix_open_beneath / staged_file, so confinement is already
232 * established; copy_file_range(2)/pread/pwrite act on fds, not paths, and
233 * have no RESOLVE_BENEATH concept. Re-confining here is neither possible
234 * nor meaningful.
235 */
236 if (brix_copy_range(log, src_fd, 0, dst_fd, 0, ssb->st_size, src,
237 use_staging ? staged->tmp_path : dst) != NGX_OK)
238 {
239 ns_set_err(res, errno);
240 close(src_fd);
241 if (use_staging) {
242 brix_staged_abort(log, root_canon, staged, 1);
243 } else {
244 close(dst_fd);
245 }
246 return;
247 }
248
249 close(src_fd);
250
251 if (use_staging) {
252 if (brix_staged_commit(log, root_canon, staged, dst) != NGX_OK) {
253 ns_set_err(res, errno);
254 return;
255 }
256 } else {
257 close(dst_fd);
258 }
259
260 /*
261 * NOT a beneath site: xattr copy is name+path based (listxattr/getxattr/
262 * setxattr on src/dst). There is no openat2/RESOLVE_BENEATH-equivalent for
263 * extended attributes that takes a dirfd, and src/dst were just proven
264 * resident under the root by the beneath opens above, so this introduces no
265 * new escape surface. (A future hardening could switch to fd-based
266 * fgetxattr/fsetxattr; tracked, not a confinement gap.)
267 */
268 if (opts->preserve_xattrs) {
269 brix_ns_copy_fattrs(log, src, dst);
270 }
271
272 res->status = BRIX_NS_OK;
273 res->created = 1;
274}

276brix_ns_result_t
277brix_ns_local_copy(ngx_log_t *log, const char *root_canon, const char *src,
278 const char *dst, const brix_ns_copy_opts_t *opts)
279{
280 brix_ns_result_t res;
281 struct stat ssb;
282 int src_fd = -1, dst_fd = -1;
283 int rootfd;
284 const char *src_rel, *dst_rel;
285 brix_staged_file_t staged;
286 ngx_flag_t use_staging = 0;
287
288 ngx_memzero(&res, sizeof(res));
289
290 src_rel = ns_rel(root_canon, src, &rootfd);
291 if (src_rel == NULL) {
292 ns_set_err(&res, errno);
293 return res;
294 }
295 dst_rel = brix_beneath_strip_root(root_canon, dst);
296 if (dst_rel == NULL) {
297 close(rootfd);
298 ns_set_err(&res, EXDEV);
299 return res;
300 }
301
302 if (ns_copy_validate(rootfd, src_rel, dst_rel, opts, &ssb, &res)) {
303 close(rootfd);
304 return res;
305 }
306
307 src_fd = brix_open_beneath(rootfd, src_rel, O_RDONLY, 0);
308 if (src_fd < 0) {
309 ns_set_err(&res, errno);
310 close(rootfd);
311 return res;
312 }
313
314 if (ns_copy_open_dst(log, rootfd, root_canon, dst, dst_rel, ssb.st_mode,
315 opts, &staged, &dst_fd, &use_staging, &res))
316 {
317 close(src_fd);
318 close(rootfd);
319 return res;
320 }
321
322 ns_copy_finalize(log, root_canon, src, dst, src_fd, dst_fd, &ssb, opts,
323 use_staging, &staged, &res);
324
325 close(rootfd);
326 return res;
327}
