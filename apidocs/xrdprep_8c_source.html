3 *
4 * WHAT: `xrdprep [-s|-c|-w|-f|-e] [-p PRTY] host[:port] <path>...` — a scriptable
5 * subset of `xrdfs prepare`. -s stage, -c cancel, -w write-mode, -f fresh,

37/*
38 * xrdprep_flag_t — one boolean prepare flag, matched by short or long spelling.

88/* ---- Handle --version / --help / -h before normal parsing ----
89 *
90 * WHAT: If argv[1] is a recognised meta-flag, prints the requested text and
91 * returns 0 (the process exit code). Returns -1 when no meta-flag applies,
92 * signalling the caller to continue with normal argument parsing.
93 *
94 * WHY: These flags short-circuit the tool and are checked positionally at
95 * argv[1] only — deliberately not routed through the shared flag table,
96 * matching the original behaviour exactly.
97 *
98 * HOW: 1. Require at least one user argument; otherwise report "not handled".
99 * 2. On --version, print the client version banner and return 0.
100 * 3. On --help or -h, print the usage block and return 0.
101 * 4. Otherwise return -1.
102 */
103static int
104xrdprep_handle_meta_flags(int argc, char **argv)
105{
106 if (argc < 2) {
107 return -1;
108 }
109 if (strcmp(argv[1], "--version") == 0) {
110 printf("%s (BriX-Cache client) %s\n", brix_prog_base(argv[0]),
111 brix_client_version());
112 return 0;
113 }
114 if (strcmp(argv[1], "--help") == 0 || strcmp(argv[1], "-h") == 0) {
115 xrdprep_print_help(argv[0]);
116 return 0;
117 }
118 return -1;
119}

121/* ---- Match one argument against the boolean prepare-flag table ----
122 *
123 * WHAT: If `a` equals the short or long spelling of a table entry, OR-s that
124 * entry's bit into *options or *optionX and returns 1. Returns 0 when `a`
125 * is not a boolean flag.
126 *
127 * WHY: Replaces the original per-flag else-if chain with a single table walk,
128 * keeping the wire split (kXR_evict → optionX, others → options) as data.
129 *
130 * HOW: 1. Scan every table row.
131 * 2. On a short- or long-name hit, set the bit in the selected flag word
132 * and return 1.
133 * 3. If no row matches, return 0.
134 */
135static int
136xrdprep_match_flag(const char *a, int *options, int *optionX)
137{
138 size_t k;
139
140 for (k = 0; k < sizeof(xrdprep_flags) / sizeof(xrdprep_flags[0]); k++) {
141 const xrdprep_flag_t *f = &xrdprep_flags[k];
142 if (strcmp(a, f->short_opt) == 0 || strcmp(a, f->long_opt) == 0) {
143 if (f->is_optionX) {
144 *optionX |= f->bit;
145 } else {
146 *options |= f->bit;
147 }
148 return 1;
149 }
150 }
151 return 0;
152}

154/* ---- Parse the whole command line into an xrdprep_args ----
155 *
156 * WHAT: Walks argv[1..argc-1], filling *out with flag words, priority, endpoint,
157 * and up to XRDPREP_MAX_PATHS target paths. No return value; *out must be
158 * zero-initialised by the caller.
159 *
160 * WHY: Isolates the argument grammar from main so main is a flat orchestrator.
161 *
162 * HOW: 1. For each argument, first try the boolean-flag table; on a match, skip
163 * to the next argument.
164 * 2. Else, on -p/--priority with a following token, consume it via atoi.
165 * 3. Else, the first non-flag token is the endpoint.
166 * 4. Else, remaining tokens are paths, until the path array is full.
167 */
168static void
169xrdprep_parse_args(int argc, char **argv, xrdprep_args *out)
170{
171 int i;
172
173 for (i = 1; i < argc; i++) {
174 const char *a = argv[i];
175 if (xrdprep_match_flag(a, &out->options, &out->optionX)) {
176 continue;
177 }
178 if ((strcmp(a, "-p") == 0 || strcmp(a, "--priority") == 0) && i + 1 < argc) {
179 out->prty = atoi(argv[++i]);
180 }
181 else if (out->endpoint == NULL) { out->endpoint = a; }
182 else if (out->np < XRDPREP_MAX_PATHS) { out->paths[out->np++] = a; }
183 }
184}

186/* ---- Connect, send the prepare request, print any reply ----
187 *
188 * WHAT: Connects to args->endpoint, issues brix_prepare for the collected paths,
189 * prints a non-empty server reply, and returns the process exit code
190 * (0 on success, the connect rc or brix_shellcode(&st) on failure).
191 *
192 * WHY: Groups the network side effects behind one call so main reads as parse →
193 * validate → run.
194 *
195 * HOW: 1. Open the connection; propagate a non-zero connect rc unchanged.
196 * 2. On brix_prepare failure, print the error, close, and map st to a
197 * shell exit code.
198 * 3. On success, close, print the reply if any, and return 0.
199 */
200static int
201xrdprep_run(char **argv, const xrdprep_args *args)
202{
203 brix_conn c;
204 brix_status st;
205 char reply[1024];
206
207 int rc = brix_cli_connect(args->endpoint, NULL, &c, argv[0], &st);
208 if (rc != 0) {
209 return rc;
210 }
211 if (brix_prepare(&c, args->paths, args->np, args->options, args->optionX,
212 args->prty, reply, sizeof(reply), &st) != 0) {
213 fprintf(stderr, "%s: %s\n", argv[0], st.msg);
214 brix_close(&c);
215 return brix_shellcode(&st);
216 }
217 brix_close(&c);
218 if (reply[0] != '\0') {
219 printf("%s\n", reply);
220 }
221 return 0;
222}

224/* ---- Program entry point ----
225 *
226 * WHAT: Parses the command line and issues one kXR_prepare, returning the process
227 * exit code (0 success, 50 on usage error, else the connect/prepare code).
228 *
229 * WHY: Thin orchestrator over the parse and run helpers; libXrdCl-free.
230 *
231 * HOW: 1. Handle --version/--help/-h and return early if one applied.
232 * 2. Parse the remaining arguments into an xrdprep_args.
233 * 3. Require an endpoint and at least one path, else print usage, return 50.
234 * 4. Delegate to xrdprep_run and return its exit code.
235 */
236int
237main(int argc, char **argv)
238{
239 xrdprep_args args;
240
241 int meta = xrdprep_handle_meta_flags(argc, argv);
242 if (meta >= 0) {
243 return meta;
244 }
245
246 memset(&args, 0, sizeof(args));
247 xrdprep_parse_args(argc, argv, &args);
248
249 if (args.endpoint == NULL || args.np == 0) {
250 fprintf(stderr,
251 "usage: %s [-s|-c|-w|-f|-e] [-p prty] host[:port] <path>...\n",
252 brix_prog_base(argv[0]));
253 return 50;
254 }
255
256 return xrdprep_run(argv, &args);
257}
