28/*
29 * Locate a CGI opaque key on a token boundary (start / '&' / '?') and return its
30 * value span [*val, *val+*vlen). Shared by the compression and ZIP-member
31 * negotiations, which both scan the same path?opaque carrier. Returns 1 when the
32 * key is present on a boundary, 0 otherwise.
33 */
34static int
35open_cgi_find_value(const char *opaque, const char *key,
36 const char **val, size_t *vlen)
37{
38 const char *p = strstr(opaque, key);
39 const char *v, *end;
40
41 if (p == NULL) {
42 return 0;
43 }
44 if (p != opaque && p[-1] != '&' && p[-1] != '?') {
45 return 0; /* not on a key boundary */
46 }
47
48 v = p + strlen(key);
49 end = v;
50 while (*end != '\0' && *end != '&') {
51 end++;
52 }
53 *val = v;
54 *vlen = (size_t) (end - v);
55 return 1;
56}

58/* Layer 3 — read-open existence + directory probe.
59 *
60 * A driver-backed export (block-striped / object store) keeps its namespace in
61 * the driver: the physical tree under conf->rootfd holds no data files, so a
62 * POSIX confined stat would spuriously report ENOENT for a file that the driver
63 * actually has. Probe the driver on the export-root-relative key when a driver
64 * is bound; otherwise fall back to the confined fd-relative POSIX stat (the
65 * unchanged default-export path). Returns 1 if the path exists (and sets
66 * *is_dir accordingly), 0 if absent.
67 *
68 * WHY (phase-70) `ctx`/`pool`: on a REMOTE driver-backed export (root://
69 * origin) the driver stat is not a local syscall — it opens the origin, which
70 * demands authentication. Without forwarding the session's delegated credential
71 * (bearer JWT / full proxy) this existence probe fails with "backend has NO
72 * credential" and reports the file absent, turning a legitimate authorized read
73 * into a spurious kXR_NotFound (the data open at brix_open_dispatch_open DOES
74 * forward the credential, so writes — which skip this read probe — succeeded).
75 * The probe binds the same deleg the data open uses and stats through
76 * brix_sd_stat_maybe_cred so the origin sees the inbound user. A NULL ctx/pool
77 * keeps the legacy service-credential behaviour (no passthrough configured). */
78static int
79brix_open_read_probe(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
80 ngx_connection_t *c, const char *clean_path, const char *full_path,
81 int *is_dir)
82{
83 ngx_log_t *log = c->log;
84 brix_sd_instance_t *sd =
85 brix_vfs_backend_resolve(conf->common.root_canon, log);
86
87 *is_dir = 0;
88
89 if (sd != NULL && sd->driver->stat != NULL) {
90 brix_sd_stat_t sst;
91 const char *key =
92 brix_vfs_export_relative_root(full_path, conf->common.root_canon);
93
94 /* Slow-tier composed-cache MISS (phase-64 SP2): a source stat here would
95 * be a blocking wire round-trip on the event loop. Report the path as
96 * provisionally existing (a regular file) and let the OFFLOADED fill
97 * resolve the truth — a missing origin object surfaces as kXR_NotFound
98 * from the parked open's done callback. A COMPLETE hit answers below via
99 * the decorator's cinfo-backed stat (no source touch). */
100 if (brix_sd_cache_fill_needs_offload(sd, key)) {
101 *is_dir = 0;
102 return 1;
103 }
104
105 /* Resolve the session's forwarded backend credential (deleg passthrough
106 * bearer / full proxy, or a per-user credential-dir selection) onto a
107 * transient VFS ctx and present it to the driver stat — the same
108 * credential the data-plane open forwards (open_resolved_file.c) — so a
109 * remote origin authorises this existence probe AS the inbound user. */
110 brix_vfs_ctx_t pvctx;
111 brix_sd_ucred_t ustore;
112 brix_sd_cred_t ucred;
113 int use_cred = 0;
114 int cerr = 0;
115
116 ngx_memzero(&ucred, sizeof(ucred));
117 brix_vfs_ctx_init(&pvctx, c->pool, log, BRIX_PROTO_ROOT,
118 conf->common.root_canon, NULL, conf->common.allow_write,
119 0 /* is_tls */, ctx != NULL ? ctx->identity : NULL, full_path);
120 brix_vfs_ctx_bind_backend_cred(&pvctx,
121 &conf->common.storage_credential_dir,
122 conf->common.storage_credential_fallback);
123 if (ctx != NULL) {
124 brix_root_vfs_bind_deleg(ctx, conf, &pvctx);
125 }
126 if (brix_vfs_backend_cred(&pvctx, &ustore, &ucred, &use_cred, &cerr)
127 != NGX_OK)
128 {
129 /* A deny-mode credential gate refused this principal — treat as a
130 * miss (never distinguish kXR_NotFound from kXR_NotAuthorized; the
131 * read auth gate above already authorised the logical path). */
132 return 0;
133 }
134
135 if (brix_sd_stat_maybe_cred(sd, key, &sst,
136 use_cred ? &ucred : NULL) != NGX_OK) {
137 return 0;
138 }
139 *is_dir = sst.is_dir ? 1 : 0;
140 return 1;
141 }
142
143 {
144 struct stat est;
145
146 if (brix_stat_beneath(conf->rootfd, clean_path, &est) != 0) {
147 return 0;
148 }
149 *is_dir = S_ISDIR(est.st_mode) ? 1 : 0;
150 return 1;
151 }
152}

154/*
155 *
156 * WHAT: Phase-42 W4/W5 — negotiate an inline-compression codec from the kXR_open
157 * opaque. Returns the codec ordinal (brix_codec_id_t) to use for this
158 * handle, or BRIX_CODEC_IDENTITY (0) when compression is disabled for the
159 * direction, no "?xrootd.compress=" opaque is present, or the requested
160 * codec is unknown/unavailable. Direction is chosen by is_write: a READ
161 * open gates on brix_read_compress (W4, compress kXR_read responses); a
162 * WRITE open gates on brix_write_compress (W5, decompress kXR_write
163 * payloads on ingest).
164 *
165 * WHY: Inline compression must be strictly opt-in and invisible to stock peers.
166 * The negotiation rides the existing path?opaque carrier (stock servers
167 * ignore unknown CGI keys; stock clients never send them), gated behind the
168 * direction's directive (both off by default). Fail-soft: an unknown or
169 * unbuilt codec degrades to plaintext rather than failing the open.
170 *
171 * HOW: Extract the opaque, scan the '&'-separated CGI list for the
172 * "xrootd.compress=" key on a token boundary, look the value up by
173 * canonical name (then HTTP token, so "br" works) and require it built in.
174 */
175static uint8_t
176open_negotiate_compress_codec(brix_ctx_t *ctx,
177 ngx_stream_brix_srv_conf_t *conf,
178 ngx_flag_t is_write)
179{
180 char opaque[BRIX_MAX_PATH + 1];
181 const char *val;
182 size_t vlen;
183 ngx_flag_t enabled;
184 const brix_codec_desc_t *d;
185
186 enabled = is_write ? conf->write_compress : conf->read_compress;
187 if (!enabled || ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0)
188 {
189 return BRIX_CODEC_IDENTITY;
190 }
191
192 if (!open_extract_opaque(ctx->recv.payload, ctx->recv.cur_dlen,
193 opaque, sizeof(opaque)))
194 {
195 return BRIX_CODEC_IDENTITY;
196 }
197
198 if (!open_cgi_find_value(opaque, "xrootd.compress=", &val, &vlen)
199 || vlen == 0) {
200 return BRIX_CODEC_IDENTITY;
201 }
202
203 d = brix_codec_by_name(val, vlen);
204 if (d == NULL) {
205 d = brix_codec_by_http_token(val, vlen);
206 }
207 if (d == NULL || !d->available || d->id == BRIX_CODEC_IDENTITY) {
208 return BRIX_CODEC_IDENTITY;
209 }
210 return (uint8_t) d->id;
211}

213/*
214 *
215 * WHAT: Phase-57 W2 — pull a validated ZIP member name out of the kXR_open opaque
216 * "?xrdcl.unzip=<member>". Returns 1 with out[] filled (NUL-terminated)
217 * for a usable member, 0 when no "xrdcl.unzip=" key is present (open the
218 * archive normally), or -1 when the key IS present but its value is invalid
219 * (empty / too long / traversal) — the caller rejects with kXR_ArgInvalid.
220 * WHY: ZIP member access serves one file inside an archive. The member name is
221 * intra-archive, but a hostile name must never be trusted, so reject empty,
222 * absolute ("/..."), and any ".." path component here, before it reaches the
223 * central-directory lookup.
224 * HOW: Extract the opaque (same carrier as the compression negotiation), scan for
225 * the "xrdcl.unzip=" key on a token boundary (start / '&' / '?'), copy the
226 * value up to the next '&', then reject leading '/', a bare "..", a leading
227 * "../", any embedded "/../", or a trailing "/..".
228 */
229/* Intra-archive traversal / absolute-path guard for a ZIP member name: reject a
230 * leading '/', a bare "..", a leading "../", any embedded "/../", or a trailing
231 * "/..". Returns 1 (unsafe → caller rejects with kXR_ArgInvalid), else 0. */
232static int
233open_zip_member_path_unsafe(const char *out, size_t vlen)
234{
235 if (out[0] == '/') {
236 return 1;
237 }
238 if (ngx_strcmp(out, "..") == 0) {
239 return 1;
240 }
241 if (vlen >= 3 && out[0] == '.' && out[1] == '.' && out[2] == '/') {
242 return 1; /* leading "../" */
243 }
244 if (strstr(out, "/../") != NULL) {
245 return 1;
246 }
247 if (vlen >= 3 && ngx_strcmp(out + vlen - 3, "/..") == 0) {
248 return 1; /* trailing "/.." */
249 }
250 return 0;
251}

253static int
254open_extract_zip_member(brix_ctx_t *ctx, char *out, size_t outsz)
255{
256 char opaque[BRIX_MAX_PATH + 1];
257 const char *val;
258 size_t vlen;
259
260 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
261 return 0;
262 }
263 if (!open_extract_opaque(ctx->recv.payload, ctx->recv.cur_dlen, opaque, sizeof(opaque))) {
264 return 0;
265 }
266
267 if (!open_cgi_find_value(opaque, "xrdcl.unzip=", &val, &vlen)) {
268 return 0; /* key absent — open the whole archive normally */
269 }
270
271 /* From here the "xrdcl.unzip=" key IS present: a bad value is an explicit
272 * error (return -1, caller rejects with kXR_ArgInvalid), NOT a fall-through
273 * to opening the whole archive — so a traversal attempt is surfaced, not
274 * silently ignored. */
275 if (vlen == 0 || vlen >= outsz) {
276 return -1;
277 }
278
279 ngx_memcpy(out, val, vlen);
280 out[vlen] = '\0';
281
282 if (open_zip_member_path_unsafe(out, vlen)) {
283 return -1;
284 }
285
286 return 1;
287}

306/*
307 * Server-mode rejection for an open: a write to a read-only server (unless in
308 * manager mode, which forwards writes), or any open on a metadata-only server
309 * without a manager_map. NGX_DECLINED to proceed; otherwise the response rc.
310 */
311static ngx_int_t
312brix_open_mode_guard(brix_ctx_t *ctx, ngx_connection_t *c,
313 ngx_stream_brix_srv_conf_t *conf, int is_write, const char *md)
314{
315 /* In manager_mode writes are forwarded to registered data servers;
316 * the allow_write check applies only to local file serving. */
317 if (is_write && !conf->common.allow_write && !conf->manager_mode) {
318 brix_log_access(ctx, c, "OPEN",

339/*
340 * Pre-resolution guards + path extraction: server-mode rejection (via
341 * brix_open_mode_guard), missing/invalid payload, ".." rejection, depth cap, and
342 * the internal-name hide. Fills clean_path (the CGI-stripped wire path).
343 * NGX_DECLINED to proceed; otherwise the response rc.
344 */
345static ngx_int_t
346brix_open_precheck(brix_ctx_t *ctx, ngx_connection_t *c,
347 ngx_stream_brix_srv_conf_t *conf, int is_write, char *clean_path)
348{
349 int op = is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD;
350 const char *md = is_write ? "wr" : "rd";
351 ngx_int_t rc;
352
353 rc = brix_open_mode_guard(ctx, c, conf, is_write, md);
354 if (rc != NGX_DECLINED) {
355 return rc;
356 }
357
358 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
359 BRIX_RETURN_ERR(ctx, c,
360 op,
361 "OPEN", "-", md,
362 kXR_ArgMissing, "no path given");
363 }
364
365 /* Strip XRootD CGI query string ("?oss.asize=N" etc.) from the path.
366 * xrdcp and other clients append these for metadata; they are not part
367 * of the filesystem path. */
368 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
369 clean_path, PATH_MAX, 1)) {
370 BRIX_RETURN_ERR(ctx, c,
371 op,
372 "OPEN", "-", md,
373 kXR_ArgInvalid, "invalid path payload");
374 }
375
376 /* Reject any ".." component (the reference does not normalize ".."); the
377 * open resolves through the kernel RESOLVE_BENEATH which would otherwise
378 * collapse an in-tree "..". Same op id selection as the surrounding errors. */
379 if (brix_reject_dotdot_path(ctx, c,
380 op,
381 "OPEN", clean_path)) {
382 return ctx->write_rc;
383 }
384
385 if (brix_count_path_depth(clean_path) != NGX_OK) {
386 BRIX_RETURN_ERR(ctx, c,
387 op,
388 "OPEN", clean_path, md,
389 kXR_ArgInvalid, "path exceeds maximum depth");
390 }
391
392 /* Internal metadata/staging artifacts are invisible: a client may neither
393 * read one nor create one. Answer as absent so an internal name is
394 * indistinguishable from a genuinely missing path. */
395 if (brix_is_internal_name(clean_path)) {
396 BRIX_RETURN_ERR(ctx, c,
397 op,
398 "OPEN", clean_path, md,
399 kXR_NotFound, "file not found");
400 }
401
402 return NGX_DECLINED;
403}

405/*
406 * Phase 35 residency gate for a read open of a nearline/offline file (backend
407 * model via the VFS seam, so a tape:// export classifies with no FRM xattr).
408 * Runs AFTER auth so an unauthorized caller never learns residency. NEARLINE
409 * recalls and parks the client (kXR_wait, or kXR_waitresp under async_recall);
410 * OFFLINE/LOST errors out; NGX_DECLINED when resident or FRM disabled.
411 */
412static ngx_int_t
413brix_open_residency_gate(brix_ctx_t *ctx, ngx_connection_t *c,
414 ngx_stream_brix_srv_conf_t *conf, const char *clean_path,
415 char *full_path, uint16_t options)
416{
417 brix_vfs_ctx_t _rvc;
418 brix_sd_residency_t _res;
419
420 if (!(conf->frm.enable && brix_stage_registry_singleton() != NULL)) {
421 return NGX_DECLINED;
422 }
423
424 brix_vfs_ctx_init(&_rvc, c->pool, c->log, BRIX_PROTO_ROOT,
425 conf->common.root_canon, NULL, conf->common.allow_write,
426 0 /* is_tls */, NULL, full_path);
427
428 if (brix_vfs_residency(&_rvc, &_res, NULL) != NGX_OK) {
429 return NGX_DECLINED;
430 }
431
432 if (_res == BRIX_SD_RES_OFFLINE || _res == BRIX_SD_RES_LOST) {
433 /* A failed/unretrievable recall — do not spin; surface an error so the
434 * client re-prepares or gives up. */
435 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
436 clean_path, "rd", kXR_FSError,
437 "file is offline (recall failed)");
438 }
439
440 if (_res == BRIX_SD_RES_NEARLINE) {
441 brix_stage_request_view_t _v;
442 char _rq[BRIX_STAGE_REQID_LEN];
443 ngx_memzero(&_v, sizeof(_v));
444 _v.lfn = full_path;
445 _v.requester_dn = (ctx->login.dn[0] != '\0') ? ctx->login.dn : NULL;
446 _v.tod_expire = (int64_t) ngx_time()
447 + (int64_t) (conf->frm.stage_ttl / 1000);
448 (void) brix_stage_request_add(
449 brix_stage_registry_singleton(),
450 &_v, _rq, sizeof(_rq), c->log);
451 /* recall driving (former frm_stage_kick) → engine step */
452
453 /* When async recall is on, park the open with kXR_waitresp and wake it in
454 * place via kXR_attn(asynresp) on completion. Falls back to the kXR_wait
455 * poll model if async is off or the waiter table is full. */
456 if (conf->frm.async_recall
457 && brix_stage_waiter_add(_rq, options,
458 ctx->recv.cur_streamid, c->fd, c->number,
459 ngx_pid, conf->frm.stage_ttl) == NGX_OK)
460 {
461 brix_log_access(ctx, c, "OPEN", clean_path,
462 "staging-async", 1, 0, NULL, 0);
463 (void) brix_send_waitresp(ctx, c);
464 ctx->state = XRD_ST_WAITING_FRM;
465 ngx_add_timer(c->read, conf->frm.stage_ttl);
466 return NGX_AGAIN;
467 }
468
469 brix_log_access(ctx, c, "OPEN", clean_path, "staging",
470 1, kXR_wait, NULL, 0);
471 return brix_send_wait(ctx, c, conf->frm.stage_wait);
472 }
473
474 return NGX_DECLINED;
475}

477/*
478 * Read-open resolution: cache-aware read returns directly; otherwise build the
479 * absolute path, run the auth gate BEFORE probing existence (no namespace
480 * existence oracle), reject a missing/dir target (or forward upstream), serve a
481 * ZIP member, and apply the residency gate. Fills full_path. NGX_DECLINED to
482 * proceed to brix_open_resolved_file; otherwise the response rc.
483 */
484static ngx_int_t
485brix_open_read_resolve(brix_ctx_t *ctx, ngx_connection_t *c,
486 ngx_stream_brix_srv_conf_t *conf, const char *clean_path,
487 char *full_path, uint16_t options, uint16_t mode_bits)
488{
489 if (conf->cache) {
490 return brix_open_cached_read(ctx, c, conf, clean_path,
491 options, mode_bits);
492 }
493
494 /* Read opens: build the absolute path, then authorize BEFORE probing
495 * existence. A denied principal must never distinguish kXR_NotFound from
496 * kXR_NotAuthorized (a namespace existence oracle) and must not be forwarded
497 * upstream on a miss. The gate keys off (identity, logical path) with no
498 * dependency on the probe, so running it first is behavior-identical for
499 * every authorized principal (mirrors statx.c / locate.c ordering). */
500 brix_beneath_full_path(conf->common.root_canon, clean_path,
501 full_path, PATH_MAX);
502
503 if (brix_auth_gate(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
504 clean_path, full_path, conf,
505 BRIX_AUTH_READ, 0) != NGX_OK) {
506 return ctx->write_rc;
507 }
508
509 {
510 int _exists, _is_dir;
511
512 _exists = brix_open_read_probe(ctx, conf, c, clean_path,
513 full_path, &_is_dir);
514 if (!_exists) {
515 if (conf->upstream_host.len > 0) {
516 brix_log_access(ctx, c, "OPEN", clean_path,
517 "upstream", 1, 0, NULL, 0);
518 BRIX_OP_OK(ctx, BRIX_OP_OPEN_RD);
519 return brix_upstream_start(ctx, c, conf);
520 }
521 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
522 clean_path, "rd", kXR_NotFound,
523 "file not found");
524 }
525 if (_is_dir) {
526 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
527 clean_path, "rd", kXR_isDirectory,
528 "is a directory");
529 }
530 }
531
532 /* Phase-57 W2: ZIP member access. The archive existence check and READ auth
533 * above gate access to the member; serve the requested member instead of the
534 * whole file. (The read-through cache path returned earlier, so this is the
535 * direct-serve case.) */
536 if (conf->zip_access) {
537 char zipmember[PATH_MAX];
538 int zr = open_extract_zip_member(ctx, zipmember,
539 sizeof(zipmember));
540 if (zr < 0) {
541 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_RD, "OPEN",
542 clean_path, "zip", kXR_ArgInvalid,
543 "invalid zip member name");
544 }
545 if (zr > 0) {
546 return brix_zip_open_member(ctx, c, conf, clean_path,
547 full_path, zipmember, options);
548 }
549 }
550
551 return brix_open_residency_gate(ctx, c, conf, clean_path, full_path,
552 options);
553}

555/*
556 * Write-open resolution: build the absolute path, run the create/update auth
557 * gate, and (kXR_mkpath | kXR_async) make the parent directory chain confined
558 * beneath the export root. Fills full_path. NGX_DECLINED to proceed.
559 */
560static ngx_int_t
561brix_open_write_resolve(brix_ctx_t *ctx, ngx_connection_t *c,
562 ngx_stream_brix_srv_conf_t *conf, const char *clean_path,
563 char *full_path, uint16_t options)
564{
565 brix_beneath_full_path(conf->common.root_canon, clean_path,
566 full_path, PATH_MAX);
567
568 /* XrdAcc distinguishes creating a NEW file (needs Insert) from updating an
569 * existing one; kXR_new is the create intent. */
570 if (brix_auth_gate_op(ctx, c, BRIX_OP_OPEN_WR, "OPEN",
571 clean_path, full_path, conf,
572 BRIX_AUTH_UPDATE, 1,
573 (options & kXR_new) ? BRIX_AOP_CREATE
574 : BRIX_AOP_UPDATE) != NGX_OK) {
575 return ctx->write_rc;
576 }
577
578 /* Create the parent directory chain when the client set kXR_mkpath OR
579 * kXR_async — the EXACT condition the reference do_Open uses to add
580 * SFS_O_MKPTH. xrdcp sends kXR_async on every upload, so this is what makes
581 * uploads to a missing parent succeed; a create-open with NEITHER flag that
582 * names a missing parent must fail NotFound, matching stock. */
583 if (options & (kXR_mkpath | kXR_async)) {
584 char parent[PATH_MAX];
585 char *slash;
586 ngx_cpystrn((u_char *) parent, (u_char *) full_path, sizeof(parent));
587 slash = strrchr(parent, '/');
588 if (slash && slash > parent) {
589 *slash = '\0';
590 /* mode 0755 for new directories; propagate group policy */
591 brix_mkdir_recursive_policy(parent, 0755, c->log,
592 conf->group_rules);
593 }
594 }
595
596 return NGX_DECLINED;
597}

599/*
600 * SciTags packet marking (phase-34): begin a flow on the FIRST local data open
601 * of this connection and arm the periodic "ongoing" firefly echo. Fail-open: a
602 * NULL flow just means the connection is not marked. Redirect/manager/cached
603 * paths returned earlier and are intentionally not marked here.
604 */
605static void
606brix_open_begin_pmark(brix_ctx_t *ctx, ngx_connection_t *c,
607 ngx_stream_brix_srv_conf_t *conf, int is_write, const char *clean_path)
608{
609 char opq[BRIX_MAX_PATH + 1];
610 const char *cgi = NULL;
611
612 if (ctx->pmark.flow != NULL || !conf->common.pmark.enable) {
613 return;
614 }
615
616 if (ctx->recv.payload != NULL && ctx->recv.cur_dlen > 0
617 && open_extract_opaque(ctx->recv.payload, ctx->recv.cur_dlen, opq, sizeof(opq)))
618 {
619 cgi = opq;
620 }
621 ctx->pmark.flow = brix_pmark_flow_begin(&conf->common.pmark, c->pool, c,
622 is_write,
623 ctx->identity ? brix_identity_vo_csv_cstr(ctx->identity) : "",
624 ctx->identity ? brix_identity_dn_cstr(ctx->identity) : "",
625 clean_path, cgi, c->log);
626
627 /* Arm the periodic "ongoing" firefly echo for this connection's flow
628 * (phase-34), if configured. Cancelled in brix_on_disconnect. */
629 if (ctx->pmark.flow != NULL && conf->common.pmark.echo > 0
630 && !ctx->pmark.echo_ev.timer_set)
631 {
632 ctx->pmark.echo_ms = conf->common.pmark.echo;
633 ctx->pmark.echo_ev.handler = brix_pmark_echo_timer;
634 ctx->pmark.echo_ev.data = ctx;
635 ctx->pmark.echo_ev.log = c->log;
636 ngx_add_timer(&ctx->pmark.echo_ev, ctx->pmark.echo_ms);
637 }
638}

640/*
641 * Slow-tier composed-cache MISS (phase-64 SP2): the decorator's inline open would
642 * run the whole-file fill as a blocking wire transfer on the event loop (and
643 * self-connect deadlock when this worker also serves the source). Park the open
644 * in XRD_ST_AIO and fill on the async thread pool instead. A hit / slice /
645 * all-local stack has needs_offload == 0 and opens inline (NGX_DECLINED).
646 */
647static ngx_int_t
648brix_open_try_cache_offload(brix_ctx_t *ctx, ngx_connection_t *c,
649 ngx_stream_brix_srv_conf_t *conf, const char *clean_path,
650 char *full_path, uint16_t options, uint16_t mode_bits)
651{
652 brix_sd_instance_t *gsd =
653 brix_vfs_backend_resolve(conf->common.root_canon, c->log);
654 const char *gkey =
655 brix_vfs_export_relative_root(full_path, conf->common.root_canon);
656
657 if (gsd != NULL && brix_sd_cache_fill_needs_offload(gsd, gkey)) {
658 ngx_int_t grc = brix_cache_open_fill_offload(ctx, c, conf,
659 clean_path, full_path, gsd, options, mode_bits);
660 if (grc != NGX_DECLINED) {
661 return grc; /* parked (async) or a queued-error rc */
662 }
663 /* NGX_DECLINED: no thread pool — inline open below (may stall). */
664 }
665 return NGX_DECLINED;
666}

668/*
669 *
670 * WHAT: Protocol-level entry point for kXR_open. Parses ClientOpenRequest from wire,
671 * validates permissions through security gates (manager mode redirect, authdb, VO ACL,
672 * token scope), resolves paths to canonical filesystem location based on read/write mode,
673 * detects TPC transfer context embedded as opaque parameters in the path string,
674 * and dispatches to either cached-read handler or brix_open_resolved_file().
675 *
676 * WHY: kXR_open is the densest request in the protocol — it bridges four layers:
677 * 1. Protocol semantics (kXR_new/create, kXR_mkpath/make-parent-dirs, kXR_retstat)
678 * 2. POSIX open(2) with confined path validation and fd allocation
679 * 3. Per-handle bookkeeping (readable/writable flags, byte counters, timestamps)
680 * 4. TPC context detection for third-party server-to-server transfers.
681 * All bookkeeping is reused by subsequent read/pgread/readv/write/close opcodes.
682 *
683 * HOW: Parse options/mode from wire → detect write-mode (kXR_new/delete/updt/wrto/apnd) →
684 * parse opaque query string for TPC params → if TPC destination (write + tpc.src): validate,
685 * resolve, check authdb/VO ACL/token scope, mkpath dirs, delegate to brix_tpc_prepare_pull;
686 * if TPC source (read + tpc.key): register/consume key → return NGX_AGAIN on consume failure.
687 * Normal path: strip CGI query from path → validate depth → manager_mode redirect or CMS locate
688 * → static map prefix redirect → read resolve (cache-aware or realpath) → write resolver →
689 * authdb/VO ACL/token scope gates → reject directories → delegate to brix_open_resolved_file().
690 */
691
692ngx_int_t
693brix_handle_open(brix_ctx_t *ctx, ngx_connection_t *c,
694 ngx_stream_brix_srv_conf_t *conf)
695{
696 xrdw_open_req_t req;
697 uint16_t options;
698 uint16_t mode_bits;
699 char full_path[PATH_MAX];
700 char clean_path[PATH_MAX]; /* path with CGI query stripped */
701 int is_write;
702 ngx_int_t rc;
703
704 xrdw_open_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
705 options = req.options;
706 mode_bits = req.mode;
707
708 /*
709 * open is the densest request in the read-side path because it bridges
710 * protocol semantics (flags, mkpath, retstat) with POSIX open(2) details
711 * including confined-path validation and fd allocation, and seeds per-handle
712 * bookkeeping (readable/writable flags, byte counters, timestamps) that all
713 * subsequent opcodes (read, pgread, readv, write, close) reuse.
714 */
715
716 /* Determine whether this is a write-mode open (shared write-bit set). */
717 is_write = brix_open_options_is_write(options);
718
719 /*
720 * §7 XrdSsi: an open of a "/.ssi/<service>" resource is a unary RPC channel,
721 * not a file — intercept before path resolution/open. Clean early-return:
722 * non-SSI opens are byte-for-byte unchanged.
723 */
724 if (conf->ssi_enable) {
725 char ssi_path[PATH_MAX];
726 const char *svc;
727 size_t svclen;
728
729 if (brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
730 ssi_path, sizeof(ssi_path), 1)
731 && brix_ssi_match(conf, ssi_path, &svc, &svclen))
732 {
733 return brix_ssi_open(ctx, c, svc, svclen, options);
734 }
735 }
736
737 /* TPC (third-party-copy) context — destination pull / source rendezvous —
738 * must act before normal path resolution. */
739 rc = brix_open_handle_tpc(ctx, c, conf, is_write, options, mode_bits);
740 if (rc != NGX_DECLINED) {
741 return rc;
742 }
743
744 /* Mode/metadata guards + path extraction/validation (fills clean_path). */
745 rc = brix_open_precheck(ctx, c, conf, is_write, clean_path);
746 if (rc != NGX_DECLINED) {
747 return rc;
748 }
749
750 /* Manager-mode redirect (dynamic CMS/registry + static manager_map). */
751 rc = brix_open_manager_redirect(ctx, c, conf, is_write, clean_path);
752 if (rc != NGX_DECLINED) {
753 return rc;
754 }
755
756 /* Resolve to the canonical filesystem target and run the auth gate (the sole
757 * checkpoint) before any existence probe. Fills full_path. */
758 if (!is_write) {
759 rc = brix_open_read_resolve(ctx, c, conf, clean_path, full_path,
760 options, mode_bits);
761 } else {
762 rc = brix_open_write_resolve(ctx, c, conf, clean_path, full_path,
763 options);
764 }
765 if (rc != NGX_DECLINED) {
766 return rc;
767 }
768
769 /* SciTags packet marking begins on the first local data open. */
770 brix_open_begin_pmark(ctx, c, conf, is_write, clean_path);
771
772 /* Slow-tier composed-cache MISS: offload the whole-file fill to the pool. */
773 if (!is_write) {
774 rc = brix_open_try_cache_offload(ctx, c, conf, clean_path, full_path,
775 options, mode_bits);
776 if (rc != NGX_DECLINED) {
777 return rc;
778 }
779 }
780
781 brix_open_request_t oreq = {
782 .resolved = full_path,
783 .options = options,
784 .mode_bits = mode_bits,
785 .is_write = is_write,
786 .codec = open_negotiate_compress_codec(ctx, conf, is_write),
787 };
788 return brix_open_resolved_file(ctx, c, conf, &oreq);
789}
