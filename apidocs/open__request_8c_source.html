31/* SciTags echo timer (phase-34): periodically emit an "ongoing" firefly for a

36{

48/*
49 * Server-mode rejection for an open: a write to a read-only server (unless in
50 * manager mode, which forwards writes), or any open on a metadata-only server
51 * without a manager_map. NGX_DECLINED to proceed; otherwise the response rc.
52 */
53static ngx_int_t
54brix_open_mode_guard(brix_ctx_t *ctx, ngx_connection_t *c,
55 ngx_stream_brix_srv_conf_t *conf, int is_write, const char *md)
56{
57 /* In manager_mode writes are forwarded to registered data servers;
58 * the allow_write check applies only to local file serving. */
59 if (is_write && !conf->common.allow_write && !conf->manager_mode) {
60 brix_log_access(ctx, c, "OPEN",
61 ctx->recv.payload ? (char *) ctx->recv.payload : "-", "wr",
62 0, kXR_fsReadOnly, "read-only server", 0);
63 BRIX_OP_ERR(ctx, BRIX_OP_OPEN_WR);
64 return brix_send_error(ctx, c, kXR_fsReadOnly,
65 "this is a read-only server");
66 }
67
68 /* metadata_only without a manager_map: serve namespace only, no file I/O. */
69 if (conf->caps.metadata_only && conf->manager_map == NULL) {
70 brix_log_access(ctx, c, "OPEN",
71 ctx->recv.payload ? (char *) ctx->recv.payload : "-",
72 md,
73 0, kXR_Unsupported, "metadata-only server", 0);
74 return brix_send_error(ctx, c, kXR_Unsupported,
75 "open not available on metadata-only server");
76 }
77
78 return NGX_DECLINED;
79}

81/*
82 * Pre-resolution guards + path extraction: server-mode rejection (via
83 * brix_open_mode_guard), missing/invalid payload, ".." rejection, depth cap, and
84 * the internal-name hide. Fills clean_path (the CGI-stripped wire path).
85 * NGX_DECLINED to proceed; otherwise the response rc.
86 */
87static ngx_int_t
88brix_open_precheck(brix_ctx_t *ctx, ngx_connection_t *c,
89 ngx_stream_brix_srv_conf_t *conf, int is_write, char *clean_path)
90{
91 int op = is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD;
92 const char *md = is_write ? "wr" : "rd";
93 ngx_int_t rc;
94
95 rc = brix_open_mode_guard(ctx, c, conf, is_write, md);
96 if (rc != NGX_DECLINED) {
97 return rc;
98 }
99
100 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0) {
101 BRIX_RETURN_ERR(ctx, c,
102 op,
103 "OPEN", "-", md,
104 kXR_ArgMissing, "no path given");
105 }
106
107 /* D-2 byte-hygiene: reject an opaque carrying a control / non-ASCII /
108 * shell-metacharacter byte before any handler (TPC src/dst URL, delegated-
109 * token mode, compression, ZIP member) parses, logs, or forwards it. A
110 * well-formed opaque percent-encodes anything outside the unreserved/
111 * structural set, so this is a zero-false-positive gate at the parse edge. */
112 {
113 char opq_raw[BRIX_MAX_PATH + 1];
114 unsigned char bad_byte;
115
116 if (open_extract_opaque(ctx->recv.payload, ctx->recv.cur_dlen,
117 opq_raw, sizeof(opq_raw))) {
118 if (brix_opaque_illegal_byte(opq_raw, &bad_byte)) {
119 ngx_log_error(NGX_LOG_INFO, c->log, 0,
120 "brix: rejecting kXR_open — illegal opaque byte 0x%02xd",
121 (int) bad_byte);
122 BRIX_RETURN_ERR(ctx, c,
123 op,
124 "OPEN", "-", md,
125 kXR_ArgInvalid, "illegal byte in opaque");
126 }
127
128 /* D-2 strict half (opt-in): once the bytes are clean, enforce the
129 * schema — oss.asize must be an unsigned integer and every key must
130 * fall in a recognized namespace. Off by default (stock accepts
131 * both unchecked); on, a typed-wrong or unknown key is refused with
132 * kXR_ArgInvalid, named in the log, before any handler parses it. */
133 if (conf->opaque_strict) {
134 char badkey[64];
135 int verdict = brix_opaque_schema_check(opq_raw, badkey,
136 sizeof(badkey));
137
138 if (verdict != BRIX_OPAQUE_SCHEMA_OK) {
139 ngx_log_error(NGX_LOG_INFO, c->log, 0,
140 "brix: rejecting kXR_open — opaque schema %s for key \"%s\"",
141 verdict == BRIX_OPAQUE_SCHEMA_BAD_TYPE
142 ? "type mismatch" : "unknown key",
143 badkey);
144 BRIX_RETURN_ERR(ctx, c,
145 op,
146 "OPEN", "-", md,
147 kXR_ArgInvalid,
148 verdict == BRIX_OPAQUE_SCHEMA_BAD_TYPE
149 ? "opaque parameter type mismatch"
150 : "unknown opaque parameter");
151 }
152 }
153 }
154 }
155
156 /* Strip XRootD CGI query string ("?oss.asize=N" etc.) from the path.
157 * xrdcp and other clients append these for metadata; they are not part
158 * of the filesystem path. */
159 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
160 clean_path, PATH_MAX, 1)) {
161 BRIX_RETURN_ERR(ctx, c,
162 op,
163 "OPEN", "-", md,
164 kXR_ArgInvalid, "invalid path payload");
165 }
166
167 /* Reject any ".." component (the reference does not normalize ".."); the
168 * open resolves through the kernel RESOLVE_BENEATH which would otherwise
169 * collapse an in-tree "..". Same op id selection as the surrounding errors. */
170 if (brix_reject_dotdot_path(ctx, c,
171 op,
172 "OPEN", clean_path)) {
173 return ctx->write_rc;
174 }
175
176 if (brix_count_path_depth(clean_path) != NGX_OK) {
177 BRIX_RETURN_ERR(ctx, c,
178 op,
179 "OPEN", clean_path, md,
180 kXR_ArgInvalid, "path exceeds maximum depth");
181 }
182
183 /* Internal metadata/staging artifacts are invisible: a client may neither
184 * read one nor create one. Answer as absent so an internal name is
185 * indistinguishable from a genuinely missing path. */
186 if (brix_is_internal_name(clean_path)) {
187 BRIX_RETURN_ERR(ctx, c,
188 op,
189 "OPEN", clean_path, md,
190 kXR_NotFound, "file not found");
191 }
192
193 return NGX_DECLINED;
194}

196/*
197 * SciTags packet marking (phase-34): begin a flow on the FIRST local data open
198 * of this connection and arm the periodic "ongoing" firefly echo. Fail-open: a
199 * NULL flow just means the connection is not marked. Redirect/manager/cached
200 * paths returned earlier and are intentionally not marked here.
201 */
202static void
203brix_open_begin_pmark(brix_ctx_t *ctx, ngx_connection_t *c,
204 ngx_stream_brix_srv_conf_t *conf, int is_write, const char *clean_path)
205{
206 char opq[BRIX_MAX_PATH + 1];
207 const char *cgi = NULL;
208
209 if (ctx->pmark.flow != NULL || !conf->common.pmark.enable) {
210 return;
211 }
212
213 if (ctx->recv.payload != NULL && ctx->recv.cur_dlen > 0
214 && open_extract_opaque(ctx->recv.payload, ctx->recv.cur_dlen, opq, sizeof(opq)))
215 {
216 cgi = opq;
217 }
218 ctx->pmark.flow = brix_pmark_flow_begin(&conf->common.pmark, c->pool, c,
219 is_write,
220 ctx->identity ? brix_identity_vo_csv_cstr(ctx->identity) : "",
221 ctx->identity ? brix_identity_dn_cstr(ctx->identity) : "",
222 clean_path, cgi, c->log);
223
224 /* Arm the periodic "ongoing" firefly echo for this connection's flow
225 * (phase-34), if configured. Cancelled in brix_on_disconnect. */
226 if (ctx->pmark.flow != NULL && conf->common.pmark.echo > 0
227 && !ctx->pmark.echo_ev.timer_set)
228 {
229 ctx->pmark.echo_ms = conf->common.pmark.echo;
230 ctx->pmark.echo_ev.handler = brix_pmark_echo_timer;
231 ctx->pmark.echo_ev.data = ctx;
232 ctx->pmark.echo_ev.log = c->log;
233 ngx_add_timer(&ctx->pmark.echo_ev, ctx->pmark.echo_ms);
234 }
235}

237/*
238 * Slow-tier composed-cache MISS (phase-64 SP2): the decorator's inline open would
239 * run the whole-file fill as a blocking wire transfer on the event loop (and
240 * self-connect deadlock when this worker also serves the source). Park the open
241 * in XRD_ST_AIO and fill on the async thread pool instead. A hit / slice /
242 * all-local stack has needs_offload == 0 and opens inline (NGX_DECLINED).
243 */
244static ngx_int_t
245brix_open_try_cache_offload(brix_ctx_t *ctx, ngx_connection_t *c,
246 ngx_stream_brix_srv_conf_t *conf, const char *clean_path,
247 char *full_path, uint16_t options, uint16_t mode_bits)
248{
249 brix_sd_instance_t *gsd =
250 brix_vfs_backend_resolve(conf->common.root_canon, c->log);
251 const char *gkey =
252 brix_vfs_export_relative_root(full_path, conf->common.root_canon);
253
254 if (gsd != NULL && brix_sd_cache_fill_needs_offload(gsd, gkey)) {
255 ngx_int_t grc = brix_cache_open_fill_offload(ctx, c, conf,
256 clean_path, full_path, gsd, options, mode_bits);
257 if (grc != NGX_DECLINED) {
258 return grc; /* parked (async) or a queued-error rc */
259 }
260 /* NGX_DECLINED: no thread pool — inline open below (may stall). */
261 }
262 return NGX_DECLINED;
263}

265/*
266 *
267 * WHAT: Protocol-level entry point for kXR_open. Parses ClientOpenRequest from wire,
268 * validates permissions through security gates (manager mode redirect, authdb, VO ACL,
269 * token scope), resolves paths to canonical filesystem location based on read/write mode,
270 * detects TPC transfer context embedded as opaque parameters in the path string,
271 * and dispatches to either cached-read handler or brix_open_resolved_file().
272 *
273 * WHY: kXR_open is the densest request in the protocol — it bridges four layers:
274 * 1. Protocol semantics (kXR_new/create, kXR_mkpath/make-parent-dirs, kXR_retstat)
275 * 2. POSIX open(2) with confined path validation and fd allocation
276 * 3. Per-handle bookkeeping (readable/writable flags, byte counters, timestamps)
277 * 4. TPC context detection for third-party server-to-server transfers.
278 * All bookkeeping is reused by subsequent read/pgread/readv/write/close opcodes.
279 *
280 * HOW: Parse options/mode from wire → detect write-mode (kXR_new/delete/updt/wrto/apnd) →
281 * parse opaque query string for TPC params → if TPC destination (write + tpc.src): validate,
282 * resolve, check authdb/VO ACL/token scope, mkpath dirs, delegate to brix_tpc_prepare_pull;
283 * if TPC source (read + tpc.key): register/consume key → return NGX_AGAIN on consume failure.
284 * Normal path: strip CGI query from path → validate depth → manager_mode redirect or CMS locate
285 * → static map prefix redirect → read resolve (cache-aware or realpath) → write resolver →
286 * authdb/VO ACL/token scope gates → reject directories → delegate to brix_open_resolved_file().
287 */
288
289ngx_int_t
290brix_handle_open(brix_ctx_t *ctx, ngx_connection_t *c,
291 ngx_stream_brix_srv_conf_t *conf)
292{
293 xrdw_open_req_t req;
294 uint16_t options;
295 uint16_t mode_bits;
296 char full_path[PATH_MAX];
297 char clean_path[PATH_MAX]; /* path with CGI query stripped */
298 int is_write;
299 ngx_int_t rc;
300
301 xrdw_open_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
302 options = req.options;
303 mode_bits = req.mode;
304
305 /*
306 * open is the densest request in the read-side path because it bridges
307 * protocol semantics (flags, mkpath, retstat) with POSIX open(2) details
308 * including confined-path validation and fd allocation, and seeds per-handle
309 * bookkeeping (readable/writable flags, byte counters, timestamps) that all
310 * subsequent opcodes (read, pgread, readv, write, close) reuse.
311 */
312
313 /* Determine whether this is a write-mode open (shared write-bit set). */
314 is_write = brix_open_options_is_write(options);
315
316 /*
317 * §7 XrdSsi: an open of a "/.ssi/<service>" resource is a unary RPC channel,
318 * not a file — intercept before path resolution/open. Clean early-return:
319 * non-SSI opens are byte-for-byte unchanged.
320 */
321 if (conf->ssi_enable) {
322 char ssi_path[PATH_MAX];
323 const char *svc;
324 size_t svclen;
325
326 if (brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
327 ssi_path, sizeof(ssi_path), 1)
328 && brix_ssi_match(conf, ssi_path, &svc, &svclen))
329 {
330 return brix_ssi_open(ctx, c, svc, svclen, options);
331 }
332 }
333
334 /* TPC (third-party-copy) context — destination pull / source rendezvous —
335 * must act before normal path resolution. */
336 rc = brix_open_handle_tpc(ctx, c, conf, is_write, options, mode_bits);
337 if (rc != NGX_DECLINED) {
338 return rc;
339 }
340
341 /* Mode/metadata guards + path extraction/validation (fills clean_path). */
342 rc = brix_open_precheck(ctx, c, conf, is_write, clean_path);
343 if (rc != NGX_DECLINED) {
344 return rc;
345 }
346
347 /* Manager-mode redirect (dynamic CMS/registry + static manager_map). */
348 rc = brix_open_manager_redirect(ctx, c, conf, is_write, clean_path);
349 if (rc != NGX_DECLINED) {
350 return rc;
351 }
352
353 /* Resolve to the canonical filesystem target and run the auth gate (the sole
354 * checkpoint) before any existence probe. Fills full_path. */
355 if (!is_write) {
356 rc = brix_open_read_resolve(ctx, c, conf, clean_path, full_path,
357 options, mode_bits);
358 } else {
359 rc = brix_open_write_resolve(ctx, c, conf, clean_path, full_path,
360 options);
361 }
362 if (rc != NGX_DECLINED) {
363 return rc;
364 }
365
366 /* SciTags packet marking begins on the first local data open. */
367 brix_open_begin_pmark(ctx, c, conf, is_write, clean_path);
368
369 /* Slow-tier composed-cache MISS: offload the whole-file fill to the pool. */
370 if (!is_write) {
371 rc = brix_open_try_cache_offload(ctx, c, conf, clean_path, full_path,
372 options, mode_bits);
373 if (rc != NGX_DECLINED) {
374 return rc;
375 }
376 }
377
378 brix_open_request_t oreq = {
379 .resolved = full_path,
380 .options = options,
381 .mode_bits = mode_bits,
382 .is_write = is_write,
383 .codec = open_negotiate_compress_codec(ctx, conf, is_write),
384 };
385 return brix_open_resolved_file(ctx, c, conf, &oreq);
386}
