4/* sigver.c — request signature verification (kXR_sigver HMAC-SHA256) and security-level enforcement
5 * WHAT: Owns the "verify" half of XRootD request signing. Three entry points:
6 * brix_verify_pending_sigver() consumes the sigver state recorded when a
7 * kXR_sigver request arrived and validates that the immediately-following
8 * covered request matches the expected opcode and HMAC; the static
9 * brix_verify_sigver_hmac() recomputes the HMAC-SHA256 over the seqno,
10 * request header, and (unless sigver_nodata) the payload, comparing it in
11 * constant time against the client-supplied ctx->sigver.hmac; and
12 * brix_signing_enforce_level() rejects opcodes that the configured
13 * brix_security_level requires to be signed but were not.
14 *
15 * WHY: XRootD signs requests so a passive observer cannot inject or tamper with
16 * mutating operations once a session is authenticated. This file enforces
17 * that contract on the server side: a mismatched/absent signature must fail
18 * closed (kXR_NotAuthorized) rather than silently passing the request to the
19 * read/write dispatchers. Splitting verify (here) from the kXR_sigver request
20 * handler (src/session/signing.c, which records the pending state) keeps the
21 * "record intent" and "check intent against next request" responsibilities
22 * in separate, single-purpose files.
23 *
24 * HOW: kXR_sigver records pending state (expected reqid, supplied HMAC, nodata
25 * flag, seqno) elsewhere; dispatch then calls brix_verify_pending_sigver()
26 * on the next request, which gates on ctx->sigver.signing_active, matches
27 * sigver_expectrid against cur_reqid, and delegates the cryptographic check
28 * to brix_verify_sigver_hmac() using ctx->sigver.signing_key via OpenSSL EVP_MAC
29 * (HMAC/SHA256, MAC + ctx cached on ctx for reuse). On success it sets
30 * ctx->sigver.verified; brix_signing_enforce_level() later consults that
31 * flag plus brix_sigver_opcode_requires() (a level 0-4 policy table) to
32 * decide whether an unsigned opcode is permitted. */
33
34/*
35 * Recompute and compare the HMAC-SHA256 over the covered request. Lazily fetches
36 * the OpenSSL "HMAC" EVP_MAC and allocates an EVP_MAC_CTX (both cached on ctx for
37 * reuse across requests), keys it with the 32-byte ctx->sigver.signing_key, and updates
38 * it with the big-endian seqno, the request header (ctx->recv.hdr_buf,
39 * XRD_REQUEST_HDR_LEN), and — unless ctx->sigver.nodata or there is no payload —
40 * the request payload. The digest is compared against ctx->sigver.hmac with
41 * CRYPTO_memcmp (constant time). Returns BRIX_DISPATCH_CONTINUE on a match;
42 * otherwise sends kXR_ServerError on a computation failure or kXR_NotAuthorized
43 * on a digest mismatch and returns that send's result.
44 */
45static ngx_int_t
46brix_verify_sigver_hmac(brix_ctx_t *ctx, ngx_connection_t *c)
47{
48 u_char computed[32];
49
50 /* Shared HMAC (libxrdproto gsi_core) — recomputes over the SAME covered bytes
51 * the client signs: seqno_be(8) || request header(24) || (payload unless the
52 * nodata flag). Single source of the covered-byte layout. */
53 if (!brix_gsi_sigver_hmac(ctx->sigver.signing_key, ctx->sigver.seqno,
54 ctx->recv.hdr_buf, ctx->recv.payload, ctx->recv.cur_dlen,
55 ctx->sigver.nodata, computed))
56 {
57 ngx_log_error(NGX_LOG_ERR, c->log, 0,
58 "brix: sigver HMAC calculation failed for reqid=%d",
59 (int) ctx->recv.cur_reqid);
60 return brix_send_error(ctx, c, kXR_ServerError,
61 "signature verification failed");
62 }
63
64 if (CRYPTO_memcmp(computed, ctx->sigver.hmac, 32) != 0) {
65 ngx_log_error(NGX_LOG_WARN, c->log, 0,
66 "brix: sigver HMAC mismatch for reqid=%d",
67 (int) ctx->recv.cur_reqid);
68 return brix_send_error(ctx, c, kXR_NotAuthorized,
69 "signature verification failed");
70 }
71
72 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
73 "brix: sigver verified reqid=%d",
74 (int) ctx->recv.cur_reqid);
75
76 return BRIX_DISPATCH_CONTINUE;
77}

79/*
80 * Verify the pending kXR_sigver signature against the next request.
81 *
82 * kXR_sigver itself lives in src/session/signing.c because that request records
83 * pending state. This file owns the other half: checking that pending state
84 * before the covered request is routed.
85 */
86ngx_int_t
87brix_verify_pending_sigver(brix_ctx_t *ctx, ngx_connection_t *c)
88{
89 ngx_int_t rc;
90
91 ctx->sigver.verified = 0;
92
93 if (ctx->sigver.pending && ctx->recv.cur_reqid != kXR_sigver) {
94 ctx->sigver.pending = 0;
95
96 if (ctx->sigver.signing_active) {
97 if (ctx->sigver.expectrid != ctx->recv.cur_reqid) {
98 ngx_log_error(NGX_LOG_WARN, c->log, 0,
99 "brix: sigver expectrid=%d but got reqid=%d",
100 (int) ctx->sigver.expectrid,
101 (int) ctx->recv.cur_reqid);
102 /* kXR_InvalidRequest: the request is malformed or not allowed now */
103 return brix_send_error(ctx, c, kXR_InvalidRequest,
104 "signed request opcode mismatch");
105 }
106
107 rc = brix_verify_sigver_hmac(ctx, c);
108 if (rc != BRIX_DISPATCH_CONTINUE) {
109 return rc;
110 }
111
112 ctx->sigver.verified = 1;
113 }
114 } else if (ctx->recv.cur_reqid == kXR_sigver) {
115 ctx->sigver.pending = 0;
116 }
117
118 return BRIX_DISPATCH_CONTINUE;
119}

121/*
122 * Policy table mapping (opcode, security_level) → whether a signature is required.
123 * Levels mirror XRootD's brix_security_level: 0=none, 1=compatible (nothing
124 * required), 2=standard (mutations + kXR_open), 3=intense (everything post-login),
125 * 4=pedantic (everything). Session/auth state-machine opcodes (login, protocol,
126 * auth, endsess, ping, sigver, bind) are always exempt. Returns non-zero when the
127 * opcode must be signed at the given level.
128 */
129static int
130brix_sigver_opcode_requires(uint16_t opcode, ngx_uint_t level)
131{
132 /* Policy table now lives in the shared gsi_core.c (single source with the
133 * native client's signer). Level 4 (pedantic) folds into "everything". */
134 return brix_gsi_sigver_required(opcode, (int) level);
135}

137/*
138 * brix_signing_enforce_level — enforce the configured brix_security_level.
139 *
140 * Checks whether the current opcode requires a signature at the configured
141 * security level. If it does and the request was not signed (verified_signing=0),
142 * rejects the request with kXR_NotAuthorized.
143 */
144ngx_int_t
145brix_signing_enforce_level(brix_ctx_t *ctx, ngx_connection_t *c,
146 ngx_stream_brix_srv_conf_t *conf)
147{
148 if (!ctx->sigver.signing_active || conf->security_level == 0) {
149 return BRIX_DISPATCH_CONTINUE;
150 }
151
152 if (brix_sigver_opcode_requires(ctx->recv.cur_reqid, conf->security_level)) {
153 if (!ctx->sigver.verified) {
154 ngx_log_error(NGX_LOG_WARN, c->log, 0,
155 "brix: unsigned request %d rejected by security_level=%d",
156 (int) ctx->recv.cur_reqid, (int) conf->security_level);
157 return brix_send_error(ctx, c, kXR_NotAuthorized,
158 "request signing required for this opcode");
159 }
160
161 /* Pedantic mode: also enforce that the signature covered the payload. */
162 if (conf->security_level >= 4 && ctx->sigver.nodata && ctx->recv.cur_dlen > 0) {
163 ngx_log_error(NGX_LOG_WARN, c->log, 0,
164 "brix: pedantic signing rejection: sigver nodata flag set but payload present");
165 return brix_send_error(ctx, c, kXR_NotAuthorized,
166 "payload signing required in pedantic mode");
167 }
168 }
169
170 return BRIX_DISPATCH_CONTINUE;
171}
