1/*
2 * lifecycle_timing.h — one-line phase timing for process lifecycle events.
3 *
4 * WHAT: a tiny monotonic stopwatch that accumulates a sequence of named phase
5 * deltas into a single fixed-size buffer and emits ONE summary log line at the
6 * end of a lifecycle event (master postconfiguration, per-worker init_process,
7 * worker exit). Example output:
8 *
9 * xrootd init_process[w0]: uring=812us servers=43120us keypool=15003us total=59102us
10 *
11 * WHY: startup/shutdown cost was previously un-instrumented — there was no way
12 * to tell whether boot latency lived in cert/CA parsing, SHM init, uring bring-up,
13 * or per-server-block preparation without an ad-hoc strace. A permanent, greppable
14 * one-liner (`grep 'init_process\[' error.log`) gives operators and developers an
15 * honest per-phase breakdown on every boot and reload, and is the parse target for
16 * tests/profile_lifecycle.sh.
17 *
18 * HOW: CLOCK_MONOTONIC via clock_gettime() — vDSO-backed (~20 ns/call), so the two
19 * reads per marked phase are lost in the syscalls each phase already makes. Deltas
20 * are reported in MICROseconds (integer; no floats, matching the VFS latency
21 * convention) so sub-millisecond phases are not quantised to 0. The accumulator is
22 * a stack struct owned by the caller — no allocation, no globals, nothing on the
23 * request hot path: a lifecycle event fires this once, never per connection.
24 *
25 * USAGE:
26 * brix_phase_timer_t t;
27 * brix_phase_timer_start(&t);
28 * ... phase 1 ... brix_phase_mark(&t, "uring");
29 * ... phase 2 ... brix_phase_mark(&t, "servers");
30 * brix_phase_timer_log(&t, cycle->log, "xrootd init_process[w0]");
31 *
32 * Each mark records the delta since the PREVIOUS mark (or since _start for the
33 * first). _log appends a final "total=<since start>" and emits at NOTICE.
34 */

42 * sized so marks are never silently dropped; overflow is clamped, not an error. */
