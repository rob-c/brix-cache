21/* Render the walk's internal logical path (anchored at "." or a relative
22 * subdir) as a clean, root-relative display path: "." → "/", "./a/b" → "/a/b",
23 * "sub/x" → "/sub/x". The integrity layer keys on the fd, not this string. */
24static void
25scan_display_path(const char *logical, char *out, size_t outsz)
26{
27 const char *p = logical;
28
29 if (p[0] == '.' && p[1] == '/') {
30 p += 2;
31 } else if (p[0] == '.' && p[1] == '\0') {
32 p += 1;
33 }
34 snprintf(out, outsz, "/%s", p);
35}

50/* Append a ready NDJSON line (llen bytes) plus a newline to the growing heap
51 * buffer. Returns 1 ok, -1 OOM (caller owns *buf on failure). Mirrors
52 * brix_ckscan_append's exponential growth. */
53static int
54scan_append(u_char **buf, size_t *cap, size_t *used, const char *line, size_t llen)
55{
56 if (*used + llen + 2 > *cap) {
57 size_t new_cap = *cap * 2 + llen + 2;
58 u_char *nb = ngx_alloc(new_cap, ngx_cycle->log);
59
60 if (nb == NULL) {
61 return -1;
62 }
63 if (*buf != NULL) {
64 ngx_memcpy(nb, *buf, *used);
65 ngx_free(*buf);
66 }
67 *buf = nb;
68 *cap = new_cap;
69 }
70 ngx_memcpy(*buf + *used, line, llen);
71 *used += llen;
72 (*buf)[(*used)++] = '\n';
73 return 0;
74}

86/* Look up the stored (cached) checksum without computing. Returns the hex on a
87 * cache hit (into `info`), or NULL on miss. */
88static const char *
89scan_stored(scan_walk_ctx_t *cc, int fd, const char *logical,
90 brix_integrity_info_t *info)
91{
92 brix_integrity_opts_t io;
93
94 ngx_memzero(&io, sizeof(io));
95 io.allow_xattr_cache = 1;
96 io.no_compute = 1;
97 if (brix_integrity_get_fd(cc->log, fd, NULL, logical, cc->opts->alg,
98 &io, info) == NGX_OK)

120/* verify: recompute and compare to stored. */
121static int
122scan_action_verify(scan_walk_ctx_t *cc, const char *logical, off_t size,
123 time_t mtime, int fd, char *line, size_t linesz)
124{
125 brix_integrity_info_t stored_info, comp_info;
126 const char *stored;
127 brix_integrity_opts_t io;
128 const char *status;
129 const char *computed = NULL;
130 char stored_copy[129];
131
132 stored = scan_stored(cc, fd, logical, &stored_info);
133 if (stored != NULL) {
134 ngx_memcpy(stored_copy, stored, ngx_strlen(stored) + 1);
135 stored = stored_copy; /* survive the second get_fd's info reuse */
136 }
137
138 ngx_memzero(&io, sizeof(io));
139 io.allow_xattr_cache = 0; /* force a fresh compute over the bytes */
140 io.no_compute = 0;
141 if (brix_integrity_get_fd(cc->log, fd, NULL, logical, cc->opts->alg,
142 &io, &comp_info) != NGX_OK)
143 {
144 status = "unreadable";
145 cc->sum->unreadable++;
146 } else {
147 computed = comp_info.hex;
148 cc->sum->bytes += (uint64_t) size;
149 if (stored == NULL) {
150 status = "missing";
151 cc->sum->missing++;
152 } else if (strcasecmp(stored, computed) == 0) {
153 status = "ok";
154 cc->sum->ok++;
155 } else {
156 status = "mismatch";
157 cc->sum->mismatch++;
158 }
159 }
160 return brix_scan_record_file(line, linesz, logical, (int64_t) size,
161 (int64_t) mtime, cc->opts->alg, stored,
162 computed, status);
163}

165/* fill: persist a checksum only when none is stored. */
166static int
167scan_action_fill(scan_walk_ctx_t *cc, const char *logical, off_t size,
168 time_t mtime, int fd, char *line, size_t linesz)
169{
170 brix_integrity_info_t info;
171 brix_integrity_opts_t io;
172 const char *status;
173 const char *stored;
174
175 stored = scan_stored(cc, fd, logical, &info);
176 if (stored != NULL) {
177 status = "already";
178 cc->sum->already++;
179 return brix_scan_record_file(line, linesz, logical, (int64_t) size,
180 (int64_t) mtime, cc->opts->alg, stored,
181 NULL, status);
182 }
183
184 ngx_memzero(&io, sizeof(io));
185 io.allow_xattr_cache = 1; /* required alongside update for the persist path */
186 io.update_xattr_cache = 1;
187 io.no_compute = 0;
188 if (brix_integrity_get_fd(cc->log, fd, NULL, logical, cc->opts->alg,
189 &io, &info) != NGX_OK)
190 {
191 status = "unreadable";
192 stored = NULL;
193 cc->sum->unreadable++;
194 } else {
195 status = "filled";
196 stored = info.hex;
197 cc->sum->filled++;
198 cc->sum->bytes += (uint64_t) size;
199 }
200 return brix_scan_record_file(line, linesz, logical, (int64_t) size,
201 (int64_t) mtime, cc->opts->alg, stored,
202 NULL, status);

223/* inventory (E1): one "object" record per stored object. The scan endpoint walks

225 * key and every entry is backed (orphan=false). A catalog-native backend's

252static ngx_int_t
253scan_walk_file(void *cookie, const char *logical, const brix_vfs_stat_t *st,
254 int fd)
255{
256 scan_walk_ctx_t *cc = cookie;
257 char line[SCAN_LINE_MAX];
258 char disp[PATH_MAX];
259 int n;
260
261 scan_display_path(logical, disp, sizeof(disp));
262
263 switch (cc->opts->mode) {
264 case BRIX_SCAN_VERIFY:
265 n = scan_action_verify(cc, disp, st->size, st->mtime, fd, line,
266 sizeof(line));
267 break;
268 case BRIX_SCAN_FILL:
269 n = scan_action_fill(cc, disp, st->size, st->mtime, fd, line,
270 sizeof(line));
271 break;
272 case BRIX_SCAN_INSPECT:
273 n = scan_action_inspect(cc, disp, st->size, st->mtime, fd, line,
274 sizeof(line));
275 break;
276 case BRIX_SCAN_INVENTORY:
277 n = scan_action_inventory(cc, disp, st->size, st->mtime, line,
278 sizeof(line));
279 break;
280 case BRIX_SCAN_DRIFT:
281 n = scan_action_drift(cc, disp, st->size, st->mtime, line,
282 sizeof(line));
283 break;
284 case BRIX_SCAN_DUMP:
285 case BRIX_SCAN_COMPARE:
286 default:
287 n = scan_action_dump(cc, disp, st->size, st->mtime, fd, line,
288 sizeof(line));
289 break;
290 }
291
292 if (n < 0) {
293 return NGX_OK; /* path too long to format — soft skip */
294 }
295 cc->sum->files++;
296 if (scan_append(cc->buf, cc->cap, cc->used, line, (size_t) n) < 0) {
297 cc->oom = 1;
298 return NGX_ERROR;
299 }
300 return NGX_OK;
301}

303ngx_int_t
304brix_scan_run(ngx_log_t *log, int rootfd, const char *logical,
305 const brix_scan_opts_t *opts, u_char **buf, size_t *cap, size_t *used,
306 brix_scan_summary_t *summary, uint16_t *err_code, char *err_msg,
307 size_t err_sz)
308{
309 scan_walk_ctx_t cc;
310 brix_vfs_walk_opts_t wopts;
311 brix_vfs_walk_target_t target = BRIX_VFS_WALK_NONE;
312 char walkmsg[128] = "";
313 ngx_int_t rc;
314
315 ngx_memzero(&cc, sizeof(cc));
316 cc.log = log;
317 cc.opts = opts;
318 cc.buf = buf;
319 cc.cap = cap;
320 cc.used = used;
321 cc.sum = summary;
322
323 ngx_memzero(&wopts, sizeof(wopts));
324 wopts.max_depth = opts->max_depth;
325 wopts.max_files = opts->max_files;
326 wopts.open_files = 1; /* the walk hands each regular file a read-only fd */
327
328 rc = brix_vfs_walk(log, rootfd, logical, &wopts, scan_walk_file, &cc,
329 &target, walkmsg, sizeof(walkmsg));
330
331 if (rc == NGX_DECLINED) {
332 *err_code = kXR_NotFound;
333 snprintf(err_msg, err_sz, "target not found");
334 return NGX_ERROR;
335 }
336 if (rc != NGX_OK) {
337 if (cc.oom) {
338 *err_code = kXR_NoMemory;
339 snprintf(err_msg, err_sz, "out of memory");
340 } else {
341 *err_code = kXR_IOError;
342 snprintf(err_msg, err_sz, "%s", walkmsg[0] ? walkmsg : "I/O error");
343 }
344 return NGX_ERROR;
345 }
346 if (target == BRIX_VFS_WALK_OTHER) {
347 *err_code = kXR_ArgInvalid;
348 snprintf(err_msg, err_sz, "not a file or directory");
349 return NGX_ERROR;
350 }
351 return NGX_OK;
352}

364/* Emit one "object" record per enumerated backend object. A driver-recovered
365 * logical path of NULL marks an orphan (a stored object with no namespace entry).
366 * Returns 0 to continue the enumeration, 1 to abort (only on OOM). */
367static int
368scan_catalog_emit_object(void *ctx, const brix_sd_catalog_ent_t *ent)
369{
370 scan_catalog_ctx_t *cc = ctx;
371 char line[SCAN_LINE_MAX];
372 int n;
373
374 n = brix_scan_record_object(line, sizeof(line), ent->key, ent->path,
375 ent->have_stat ? (int64_t) ent->size : 0,
376 ent->have_stat ? (int64_t) ent->mtime : 0,
377 ent->path == NULL ? 1 : 0);
378 if (n < 0) {
379 return 0; /* unrepresentable key → soft skip */
380 }
381 cc->sum->files++;
382 cc->sum->ok++;
383 if (scan_append(cc->buf, cc->cap, cc->used, line, (size_t) n) < 0) {
384 cc->oom = 1;
385 return 1; /* abort enumeration */
386 }
387 return 0;
388}

445scan_verify_obj_record(scan_verify_ctx_t *cc, const char *logical,

494/* Verify one enumerated catalog object: open it through the bound driver, build
495 * the record, close. An open failure is reported as "unreadable" (the scan keeps
496 * going). Returns 0 to continue the enumeration, 1 to abort (OOM only). */
497static int
498scan_catalog_verify_one(void *ctx, const brix_sd_catalog_ent_t *ent)
499{
500 scan_verify_ctx_t *cc = ctx;
501 const char *logical = ent->path ? ent->path : ent->key;
502 brix_sd_obj_t *obj;
503 char line[SCAN_LINE_MAX];
504 int err = 0;
505 int n;
506
507 obj = cc->sd->driver->open(cc->sd, logical, BRIX_SD_O_READ, 0, &err);
508 if (obj == NULL) {
509 cc->sum->unreadable++;
510 n = brix_scan_record_file(line, sizeof(line), logical,
511 ent->have_stat ? (int64_t) ent->size : 0,
512 ent->have_stat ? (int64_t) ent->mtime : 0,
513 cc->alg, NULL, NULL, "unreadable");
514 } else {
515 n = scan_verify_obj_record(cc, logical, ent, obj, line, sizeof(line));
516 (void) cc->sd->driver->close(obj);
517 }
518
519 if (n < 0) {
520 return 0; /* unrepresentable → soft skip */
521 }
522 cc->sum->files++;
523 if (scan_append(cc->buf, cc->cap, cc->used, line, (size_t) n) < 0) {
524 cc->oom = 1;
525 return 1;
526 }
527 return 0;
528}
