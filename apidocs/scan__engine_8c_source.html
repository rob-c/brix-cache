19/* Render the walk's internal logical path (anchored at "." or a relative
20 * subdir) as a clean, root-relative display path: "." → "/", "./a/b" → "/a/b",
21 * "sub/x" → "/sub/x". The integrity layer keys on the fd, not this string. */
22static void
23scan_display_path(const char *logical, char *out, size_t outsz)
24{
25 const char *p = logical;
26
27 if (p[0] == '.' && p[1] == '/') {
28 p += 2;
29 } else if (p[0] == '.' && p[1] == '\0') {
30 p += 1;
31 }
32 snprintf(out, outsz, "/%s", p);
33}

48/* Append a ready NDJSON line (llen bytes) plus a newline to the growing heap
49 * buffer. Returns 1 ok, -1 OOM (caller owns *buf on failure). Mirrors
50 * brix_ckscan_append's exponential growth. Declared in scan_engine_internal.h
51 * (shared with the catalog-enumeration split). */
52int
53scan_append(u_char **buf, size_t *cap, size_t *used, const char *line, size_t llen)
54{
55 if (*used + llen + 2 > *cap) {
56 size_t new_cap = *cap * 2 + llen + 2;
57 u_char *nb = ngx_alloc(new_cap, ngx_cycle->log);
58
59 if (nb == NULL) {
60 return -1;
61 }
62 if (*buf != NULL) {
63 ngx_memcpy(nb, *buf, *used);
64 ngx_free(*buf);
65 }
66 *buf = nb;
67 *cap = new_cap;
68 }
69 ngx_memcpy(*buf + *used, line, llen);
70 *used += llen;
71 (*buf)[(*used)++] = '\n';
72 return 0;
73}

85/*
86 * WHAT: per-file action context — the walk cookie plus the one file the action
87 * is currently building a record for (its display path, stat facts, and
88 * read-only fd) and the caller's output line buffer.
89 * WHY: every mode action threads the identical (walk-state, this-file, output)
90 * triple; bundling it keeps each action at ≤ 2 params (ctx + info out)
91 * and makes the shared data flow explicit instead of a wide argument list.
92 * HOW: scan_walk_file fills one on the stack per file and hands its address to
93 * the selected action; the action never mutates it, only reads.
94 */
95typedef struct {
96 scan_walk_ctx_t *cc;
97 const char *logical; /* clean display path for this file */
98 off_t size;
99 time_t mtime;
100 int fd; /* read-only fd from the walk */
101 char *line; /* output record buffer */
102 size_t linesz;
103} scan_action_ctx_t;

105/* Look up the stored (cached) checksum without computing. Returns the hex on a
106 * cache hit (into `info`), or NULL on miss. */
107static const char *
108scan_stored(scan_action_ctx_t *a, brix_integrity_info_t *info)
109{
110 brix_integrity_opts_t io;
111
112 ngx_memzero(&io, sizeof(io));
113 io.allow_xattr_cache = 1;
114 io.no_compute = 1;
115 if (brix_integrity_get_fd(a->cc->log, a->fd, NULL, a->logical,
116 a->cc->opts->alg, &io, info) == NGX_OK)
117 {
118 return info->hex;
119 }
120 return NULL;
121}

123/* dump / compare: emit stored checksum (or null), never read bytes. */
124static int
125scan_action_dump(scan_action_ctx_t *a)
126{
127 brix_integrity_info_t info;
128 const char *stored = scan_stored(a, &info);
129 int n;
130
131 n = brix_scan_record_file(a->line, a->linesz, a->logical, (int64_t) a->size,
132 (int64_t) a->mtime, a->cc->opts->alg, stored,
133 NULL, "ok");
134 a->cc->sum->ok++;
135 return n;
136}

138/* verify: recompute and compare to stored. */
139static int
140scan_action_verify(scan_action_ctx_t *a)
141{
142 brix_integrity_info_t stored_info, comp_info;
143 const char *stored;
144 brix_integrity_opts_t io;
145 const char *status;
146 const char *computed = NULL;
147 char stored_copy[129];
148
149 stored = scan_stored(a, &stored_info);
150 if (stored != NULL) {
151 ngx_memcpy(stored_copy, stored, ngx_strlen(stored) + 1);
152 stored = stored_copy; /* survive the second get_fd's info reuse */
153 }
154
155 ngx_memzero(&io, sizeof(io));
156 io.allow_xattr_cache = 0; /* force a fresh compute over the bytes */
157 io.no_compute = 0;
158 if (brix_integrity_get_fd(a->cc->log, a->fd, NULL, a->logical,
159 a->cc->opts->alg, &io, &comp_info) != NGX_OK)
160 {
161 status = "unreadable";
162 a->cc->sum->unreadable++;
163 } else {
164 computed = comp_info.hex;
165 a->cc->sum->bytes += (uint64_t) a->size;
166 if (stored == NULL) {
167 status = "missing";
168 a->cc->sum->missing++;
169 } else if (strcasecmp(stored, computed) == 0) {
170 status = "ok";
171 a->cc->sum->ok++;
172 } else {
173 status = "mismatch";
174 a->cc->sum->mismatch++;
175 }
176 }
177 return brix_scan_record_file(a->line, a->linesz, a->logical,
178 (int64_t) a->size, (int64_t) a->mtime,
179 a->cc->opts->alg, stored, computed, status);
180}

182/* fill: persist a checksum only when none is stored. */
183static int
184scan_action_fill(scan_action_ctx_t *a)
185{
186 brix_integrity_info_t info;
187 brix_integrity_opts_t io;
188 const char *status;
189 const char *stored;
190
191 stored = scan_stored(a, &info);
192 if (stored != NULL) {
193 status = "already";
194 a->cc->sum->already++;
195 return brix_scan_record_file(a->line, a->linesz, a->logical,
196 (int64_t) a->size, (int64_t) a->mtime,
197 a->cc->opts->alg, stored, NULL, status);
198 }
199
200 ngx_memzero(&io, sizeof(io));
201 io.allow_xattr_cache = 1; /* required alongside update for the persist path */
202 io.update_xattr_cache = 1;
203 io.no_compute = 0;
204 if (brix_integrity_get_fd(a->cc->log, a->fd, NULL, a->logical,
205 a->cc->opts->alg, &io, &info) != NGX_OK)
206 {
207 status = "unreadable";
208 stored = NULL;
209 a->cc->sum->unreadable++;
210 } else {
211 status = "filled";
212 stored = info.hex;
213 a->cc->sum->filled++;
214 a->cc->sum->bytes += (uint64_t) a->size;
215 }
216 return brix_scan_record_file(a->line, a->linesz, a->logical,
217 (int64_t) a->size, (int64_t) a->mtime,
218 a->cc->opts->alg, stored, NULL, status);
219}

221/* inspect (A2): per-file backend introspection. The scan endpoint walks the
222 * export root via POSIX openat2 (not the SD-driver seam), so the backend is
223 * "posix" and namespace==backend by construction; the driver-bound view
224 * (Ceph object key, cluster facts) is a Phase-4 prerequisite. */
225static int
226scan_action_inspect(scan_action_ctx_t *a)
227{
228 brix_integrity_info_t info;
229 const char *stored = scan_stored(a, &info);
230
231 a->cc->sum->ok++;
232 return brix_scan_record_inspect(a->line, a->linesz, a->logical, "posix",
233 (int64_t) a->size, (int64_t) a->mtime,
234 stored != NULL ? "xattr" : "none",
235 1 /* posix: namespace == backend */);
236}

238/* inventory (E1): one "object" record per stored object. The scan endpoint walks
239 * the export via POSIX (namespace == catalog), so the logical path IS the backend
240 * key and every entry is backed (orphan=false). A catalog-native backend's
241 * inventory runs through brix_vfs_enumerate_catalog (the SD enumerate verb)
242 * instead — the seam is in place; threading a bound instance into the endpoint is
243 * the Ceph follow-on. */
244static int
245scan_action_inventory(scan_action_ctx_t *a)
246{
247 a->cc->sum->ok++;
248 return brix_scan_record_object(a->line, a->linesz, a->logical, a->logical,
249 (int64_t) a->size, (int64_t) a->mtime, 0);
250}

252/* drift (D2): one "drift" record per entry. Over a namespace walk the catalog and
253 * namespace coincide, so every entry is "in_both"; orphan_object / namespace_only
254 * arise only when a native catalog verb (brix_vfs_enumerate_catalog) supplies a
255 * backend-object set to reconcile against (scan_drift) — the Ceph follow-on. */
256static int
257scan_action_drift(scan_action_ctx_t *a)
258{
259 a->cc->sum->ok++;
260 return brix_scan_record_drift(a->line, a->linesz, a->logical, a->logical,
261 "in_both", (int64_t) a->size);
262}

264/* Run the mode's per-file action against the assembled action context. Split out
265 * so scan_walk_file stays a thin fill-and-append shell around the dispatch. */
266static int
267scan_dispatch_action(scan_action_ctx_t *a)
268{
269 switch (a->cc->opts->mode) {
270 case BRIX_SCAN_VERIFY:
271 return scan_action_verify(a);
272 case BRIX_SCAN_FILL:
273 return scan_action_fill(a);
274 case BRIX_SCAN_INSPECT:
275 return scan_action_inspect(a);
276 case BRIX_SCAN_INVENTORY:
277 return scan_action_inventory(a);
278 case BRIX_SCAN_DRIFT:
279 return scan_action_drift(a);
280 case BRIX_SCAN_DUMP:
281 case BRIX_SCAN_COMPARE:
282 default:
283 return scan_action_dump(a);
284 }
285}

287static ngx_int_t
288scan_walk_file(void *cookie, const char *logical, const brix_vfs_stat_t *st,
289 int fd)
290{
291 scan_walk_ctx_t *cc = cookie;
292 scan_action_ctx_t a;
293 char line[SCAN_LINE_MAX];
294 char disp[PATH_MAX];
295 int n;
296
297 scan_display_path(logical, disp, sizeof(disp));
298
299 ngx_memzero(&a, sizeof(a));
300 a.cc = cc;
301 a.logical = disp;
302 a.size = st->size;
303 a.mtime = st->mtime;
304 a.fd = fd;
305 a.line = line;
306 a.linesz = sizeof(line);
307
308 n = scan_dispatch_action(&a);
309
310 if (n < 0) {
311 return NGX_OK; /* path too long to format — soft skip */
312 }
313 cc->sum->files++;
314 if (scan_append(cc->buf, cc->cap, cc->used, line, (size_t) n) < 0) {
315 cc->oom = 1;
316 return NGX_ERROR;
317 }
318 return NGX_OK;
319}

321ngx_int_t
322brix_scan_run(ngx_log_t *log, int rootfd, const char *logical,
323 const brix_scan_opts_t *opts, u_char **buf, size_t *cap, size_t *used,
324 brix_scan_summary_t *summary, uint16_t *err_code, char *err_msg,
325 size_t err_sz)
326{
327 scan_walk_ctx_t cc;
328 brix_vfs_walk_opts_t wopts;
329 brix_vfs_walk_target_t target = BRIX_VFS_WALK_NONE;
330 char walkmsg[128] = "";
331 ngx_int_t rc;
332
333 ngx_memzero(&cc, sizeof(cc));
334 cc.log = log;
335 cc.opts = opts;
336 cc.buf = buf;
337 cc.cap = cap;
338 cc.used = used;
339 cc.sum = summary;
340
341 ngx_memzero(&wopts, sizeof(wopts));
342 wopts.max_depth = opts->max_depth;
343 wopts.max_files = opts->max_files;
344 wopts.open_files = 1; /* the walk hands each regular file a read-only fd */
345
346 rc = brix_vfs_walk(log, rootfd, logical, &wopts, scan_walk_file, &cc,
347 &target, walkmsg, sizeof(walkmsg));
348
349 if (rc == NGX_DECLINED) {
350 *err_code = kXR_NotFound;
351 snprintf(err_msg, err_sz, "target not found");
352 return NGX_ERROR;
353 }
354 if (rc != NGX_OK) {
355 if (cc.oom) {
356 *err_code = kXR_NoMemory;
357 snprintf(err_msg, err_sz, "out of memory");
358 } else {
359 *err_code = kXR_IOError;
360 snprintf(err_msg, err_sz, "%s", walkmsg[0] ? walkmsg : "I/O error");
361 }
362 return NGX_ERROR;
363 }
364 if (target == BRIX_VFS_WALK_OTHER) {
365 *err_code = kXR_ArgInvalid;
366 snprintf(err_msg, err_sz, "not a file or directory");
367 return NGX_ERROR;
368 }
369 return NGX_OK;
