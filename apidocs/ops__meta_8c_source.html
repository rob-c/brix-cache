1/*
2 * ops_meta.c — metadata operations: kXR_stat and kXR_dirlist.
3 *
4 * WHAT: stat() returns id/size/flags/mtime; dirlist() returns entries, optionally
5 * with per-entry stat (kXR_dstat).
6 * WHY: These back `xrdfs stat` and `xrdfs ls`, and stat also serves xrdcp size
7 * checks later.
8 * HOW: The path is sent as the request payload with dlen=strlen (no trailing NUL
9 * — the server reads exactly dlen bytes). dirlist accumulates kXR_oksofar
10 * chunks until the final kXR_ok; with dstat the body begins with the
11 * ".\n0 0 0 0\n" lead-in followed by name/stat line pairs.
12 *
13 * wire: XProtocol.hh kXR_stat response — ASCII "<id> <size> <flags> <mtime>".
14 * wire: XProtocol.hh kXR_dirlist dstat — ".\n0 0 0 0" prefix, then name\nstat\n pairs.
15 */

25static int
26parse_statinfo(const char *s, brix_statinfo *out)
27{
28 brix_statline_ext ext;
29
30 /* The stat-line grammar (mandatory "<id> <size> <flags> <mtime>" + optional
31 * EOS "<ctime> <atime> <mode> <owner> <group>" tail) is owned by the shared
32 * protocol/stat_line.h so this decoder stays in lockstep with the server's
33 * encoder. */
34 if (brix_statline_parse(s, &out->id, &out->size, &out->flags, &out->mtime,
35 &ext) != 0) {
36 return -1;
37 }
38
39 out->have_ext = ext.have_ext;
40 out->ctime = ext.ctime;
41 out->atime = ext.atime;
42 out->mode = ext.mode;
43 snprintf(out->owner, sizeof(out->owner), "%s", ext.owner);
44 snprintf(out->group, sizeof(out->group), "%s", ext.group);
45 return 0;
46}

50static int
51stat_opt(brix_conn *c, const char *path, uint8_t options, brix_statinfo *out,
52 brix_status *st)
53{
54 ClientStatRequest req;
55 uint16_t status;
56 uint8_t *body = NULL;
57 uint32_t blen = 0;
58 char tmp[256];
59 uint32_t n;
60
61 memset(&req, 0, sizeof(req));
62 req.requestid = htons(kXR_stat);
63 {
64 xrdw_stat_req_t b = { .options = options, .wants = 0 };
65 xrdw_stat_req_pack(&b, ((ClientRequestHdr *) &req)->body);
66 }
67
68 if (brix_roundtrip_resilient(c, &req, path, (uint32_t) strlen(path),
69 XRDC_OP_READONLY, 0,
70 &status, &body, &blen, st) != 0) {
71 return -1;
72 }
73
74 n = (blen < sizeof(tmp) - 1) ? blen : (uint32_t) (sizeof(tmp) - 1);
75 memcpy(tmp, body, n);
76 tmp[n] = '\0';
77 free(body);
78
79 if (parse_statinfo(tmp, out) != 0) {
80 brix_status_set(st, XRDC_EPROTO, 0, "unparseable stat body: \"%s\"", tmp);
81 return -1;
82 }
83 return 0;
84}

101/* Append n bytes to a growing heap buffer; returns 0 / -1. */
102static int
103buf_append(uint8_t **buf, size_t *len, size_t *cap, const uint8_t *src, size_t n)
104{
105 if (*len + n > *cap) {
106 size_t newcap = (*cap == 0) ? 65536 : *cap;
107 uint8_t *nb;
108 while (*len + n > newcap) {
109 newcap *= 2;
110 }
111 nb = (uint8_t *) realloc(*buf, newcap);
112 if (nb == NULL) {
113 return -1;
114 }
115 *buf = nb;
116 *cap = newcap;
117 }
118 memcpy(*buf + *len, src, n);
119 *len += n;
120 return 0;
121}

123/* Find the next line within [i, len): a run ending at '\n' or '\0'. Sets *start
124 * and *linelen, returns the index just past the terminator, or len if none. The
125 * caller stops when it encounters a '\0' terminator (returned via *was_nul). */
126static size_t
127next_line(const uint8_t *b, size_t len, size_t i, size_t *start, size_t *linelen,
128 int *was_nul)
129{
130 size_t j = i;
131 *was_nul = 0;
132 while (j < len && b[j] != '\n' && b[j] != '\0') {
133 j++;
134 }
135 *start = i;
136 *linelen = j - i;
137 if (j < len && b[j] == '\0') {
138 *was_nul = 1;
139 }
140 return (j < len) ? j + 1 : len;
141}

143static int
144dirent_push(brix_dirent **ents, size_t *count, size_t *cap, const brix_dirent *e)
145{
146 if (*count == *cap) {
147 size_t newcap = (*cap == 0) ? 64 : *cap * 2;
148 brix_dirent *na = (brix_dirent *) realloc(*ents, newcap * sizeof(**ents));
149 if (na == NULL) {
150 return -1;
151 }
152 *ents = na;
153 *cap = newcap;
154 }
155 (*ents)[*count] = *e;
156 (*count)++;
157 return 0;
158}

160/* ---- Accumulate every dirlist response chunk into one heap buffer ----
161 *
162 * WHAT: Sends the prepared kXR_dirlist request on c and drains the response,
163 * concatenating the first frame and every subsequent kXR_oksofar chunk
164 * into a single freshly-allocated buffer returned via the acc_out and acclen_out out-params.
165 * Returns 0 on success; -1 on any transport or OOM error, having freed all
166 * working buffers so nothing leaks and the caller starts clean.
167 *
168 * WHY: A directory listing arrives split across an arbitrary number of
169 * kXR_oksofar frames terminated by a final kXR_ok. Isolating the
170 * receive-and-concatenate loop keeps dirlist_once a flat orchestrator and
171 * lets the parse step work over one contiguous buffer.
172 *
173 * HOW: 1. Send the first frame via brix_roundtrip (which follows a cluster
174 * redirect to the data server); read every later chunk from that same
175 * post-redirect connection with brix_recv.
176 * 2. Append each non-empty body to acc via buf_append; free each per-frame
177 * body immediately after copying.
178 * 3. Stop once a frame's status is no longer kXR_oksofar.
179 */
180static int
181dirlist_accumulate(brix_conn *c, ClientDirlistRequest *req, const char *path,
182 uint8_t **acc_out, size_t *acclen_out, brix_status *st)
183{
184 uint16_t status;
185 uint8_t *acc = NULL;
186 size_t acclen = 0, acccap = 0;
187 uint8_t *body = NULL;
188 uint32_t blen = 0;
189 int first = 1;
190 brix_payload pl = { path, (uint32_t) strlen(path) };
191 brix_resp_out out = { &status, &body, &blen };
192
193 for (;;) {
194 if (first) {
195 if (brix_roundtrip(c, req, &pl, &out, st) != 0) {
196 free(acc);
197 return -1;
198 }
199 first = 0;
200 } else if (brix_recv(c, 0xffff, &out, st) != 0) {
201 free(acc);
202 return -1;
203 }
204 if (blen > 0
205 && buf_append(&acc, &acclen, &acccap, body, blen) != 0) {
206 free(body);
207 free(acc);
208 brix_status_set(st, XRDC_EPROTO, 0, "out of memory in dirlist");
209 return -1;
210 }
211 free(body);
212 body = NULL;
213 blen = 0;
214 if (status != kXR_oksofar) {
215 break;
216 }
217 }
218
219 *acc_out = acc;
220 *acclen_out = acclen;
221 return 0;
222}

224/* ---- Skip the dstat lead-in prefix, returning the first entry offset ----
225 *
226 * WHAT: Returns the byte offset within acc at which directory entries begin. For
227 * a kXR_dstat listing that starts with the ".\n0 0 0 0" sentinel, that is
228 * the offset just past the sentinel's terminating '\n'; otherwise 0.
229 *
230 * WHY: With kXR_dstat the body opens with a fixed ".\n0 0 0 0\n" lead-in (the
231 * "." pseudo-entry plus its zeroed stat line) that must be consumed before
232 * real name/stat pairs. The sentinel and its match length are shared with
233 * the server via dirlist_fmt.h so both sides agree byte-for-byte.
234 *
235 * HOW: 1. Return 0 unless want_stat is set and acc begins with the shared
236 * BRIX_DSTAT_LEADIN sentinel of BRIX_DSTAT_PREFIX_LEN bytes.
237 * 2. Advance past the first '\n' (end of "."), then past the second '\n'
238 * (end of the "0 0 0 0" stat line), guarding each step against acclen.
239 */
240static size_t
241dirlist_skip_dstat_leadin(const uint8_t *acc, size_t acclen, int want_stat)
242{
243 size_t i = 0;
244
245 if (want_stat && acclen >= BRIX_DSTAT_PREFIX_LEN
246 && memcmp(acc, BRIX_DSTAT_LEADIN, BRIX_DSTAT_PREFIX_LEN) == 0) {
247 while (i < acclen && acc[i] != '\n') { i++; } /* ".\n..." first \n */
248 if (i < acclen) { i++; } /* now at "0 0 0 0\n" */
249 while (i < acclen && acc[i] != '\n') { i++; } /* end of "0 0 0 0" */
250 if (i < acclen) { i++; }
251 }
252 return i;
253}

255/* ---- Parse name (and optional per-entry stat) lines into a dirent array ----
256 *
257 * WHAT: Walks acc from offset i, building a heap array of brix_dirent handed off
258 * via the ents_out and count_out out-params. Returns 0 on success (ownership transferred);
259 * -1 on OOM, having freed its partial array and set a status on st.
260 *
261 * WHY: The entry region is a sequence of name lines, each optionally followed
262 * (under kXR_dstat) by a stat line, terminated by an empty line or a '\0'.
263 * Keeping the parse pure over the already-accumulated buffer makes the
264 * grammar auditable and independent of the transport loop.
265 *
266 * HOW: 1. For each line via next_line: stop on an empty name (trailing line).
267 * 2. Copy the name into the fixed dirent field, truncating to capacity.
268 * 3. When want_stat and the name line was not itself terminated, read the
269 * following stat line and decode it with parse_statinfo (set have_stat
270 * only on a clean parse).
271 * 4. Push the entry with dirent_push; stop when a '\0' terminator was seen.
272 */
273static int
274dirlist_parse_entries(const uint8_t *acc, size_t acclen, size_t i, int want_stat,
275 brix_dirent **ents_out, size_t *count_out, brix_status *st)
276{
277 brix_dirent *ents = NULL;
278 size_t count = 0, entcap = 0;
279
280 while (i < acclen) {
281 size_t ns, nl, ss, sl;
282 int term = 0;
283 brix_dirent e;
284
285 i = next_line(acc, acclen, i, &ns, &nl, &term);
286 if (nl == 0) {
287 break; /* empty trailing line */
288 }
289
290 memset(&e, 0, sizeof(e));
291 {
292 size_t cp = (nl < sizeof(e.name) - 1) ? nl : sizeof(e.name) - 1;
293 memcpy(e.name, acc + ns, cp);
294 e.name[cp] = '\0';
295 }
296
297 if (want_stat && !term) {
298 char statbuf[256];
299 size_t cp;
300 i = next_line(acc, acclen, i, &ss, &sl, &term);
301 cp = (sl < sizeof(statbuf) - 1) ? sl : sizeof(statbuf) - 1;
302 memcpy(statbuf, acc + ss, cp);
303 statbuf[cp] = '\0';
304 if (parse_statinfo(statbuf, &e.st) == 0) {
305 e.have_stat = 1;
306 }
307 }
308
309 if (dirent_push(&ents, &count, &entcap, &e) != 0) {
310 brix_status_set(st, XRDC_EPROTO, 0, "out of memory in dirlist");
311 free(ents);
312 return -1;
313 }
314 if (term) {
315 break;
316 }
317 }
318
319 *ents_out = ents;
320 *count_out = count;
321 return 0;
322}

324/* One full listing pass (a kXR_dirlist roundtrip plus the kXR_oksofar chunk
325 * accumulation) on the current connection. On any failure it leaves *ents_out
326 * NULL and frees its working buffers, so a resilient retry starts clean. */
327static int
328dirlist_once(brix_conn *c, const char *path, int want_stat,
329 brix_dirent **ents_out, size_t *count_out, brix_status *st)
330{
331 ClientDirlistRequest req;
332 uint8_t *acc = NULL;
333 size_t acclen = 0;
334 size_t i;
335 int rc;
336
337 *ents_out = NULL;
338 *count_out = 0;
339
340 memset(&req, 0, sizeof(req));
341 req.requestid = htons(kXR_dirlist);
342 {
343 xrdw_dirlist_req_t b = { .options = (uint8_t) (want_stat ? kXR_dstat : 0) };
344 xrdw_dirlist_req_pack(&b, ((ClientRequestHdr *) &req)->body);
345 }
346
347 /* Drain the whole listing into one buffer (helper frees acc on failure). */
348 if (dirlist_accumulate(c, &req, path, &acc, &acclen, st) != 0) {
349 return -1;
350 }
351
352 /* Skip the dstat lead-in, then parse the name/stat pairs. On a parse OOM the
353 * parse helper frees its partial array; the accumulation buffer we still own
354 * is freed once below, on both the success and error paths. */
355 i = dirlist_skip_dstat_leadin(acc, acclen, want_stat);
356 rc = dirlist_parse_entries(acc, acclen, i, want_stat, ents_out, count_out, st);
357 free(acc);
358 return rc;
359}
