19/* Push selection ranks (rank 0 = most preferred; order = endpoint order).
20 * Written on the event loop, read by fill threads — relaxed atomics. */
21void
22sd_http_set_ranks(brix_sd_instance_t *inst, const int *ranks, int n)
23{
24 sd_http_inst_state *is;
25 int i;
26
27 if (!sd_http_instance_is(inst)) {
28 return;
29 }
30 is = inst->state;
31 for (i = 0; i < is->n_eps && i < n; i++) {
32 atomic_store_explicit(&is->eps[i].rank, ranks[i],
33 memory_order_relaxed);
34 }
35}

37/* Endpoint inventory for the RTT prober (copies, no ngx types). */
38int
39sd_http_endpoint_list(brix_sd_instance_t *inst, char hosts[][256],
40 int *ports, int max)
41{
42 sd_http_inst_state *is;
43 int i, n;
44
45 if (!sd_http_instance_is(inst)) {
46 return 0;
47 }
48 is = inst->state;
49 n = (is->n_eps < max) ? is->n_eps : max;
50 for (i = 0; i < n; i++) {
51 memcpy(hosts[i], is->eps[i].host, sizeof(is->eps[i].host));

65/* "host:port" of the endpoint that answered the most recent read (display
66 * only; racy-by-design). 0 with buf filled, or -1 (non-http / none yet). */
67int
68sd_http_last_origin(brix_sd_instance_t *inst, char *buf, size_t cap)
69{
70 sd_http_inst_state *is;
71
72 if (!sd_http_instance_is(inst)) {
73 return -1;
74 }
75 is = inst->state;
76 if (is->last_origin[0] == '\0') {
77 return -1;
78 }
79 snprintf(buf, cap, "%s", is->last_origin);
80 return 0;
81}

92/* Health snapshot for /healthz: copies up to `max` (host, port, fail_score)

97{
98 sd_http_inst_state *is;
99 int i, n;
100

102 return 0;

105 n = (is->n_eps < max) ? is->n_eps : max;
106 for (i = 0; i < n; i++) {
107 memcpy(hosts[i], is->eps[i].host, sizeof(is->eps[i].host));
108 ports[i] = is->eps[i].port;
