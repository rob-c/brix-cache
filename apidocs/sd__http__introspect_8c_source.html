19/* Push selection ranks (rank 0 = most preferred; order = endpoint order).
20 * Written on the event loop, read by fill threads — relaxed atomics. */
21void
22sd_http_set_ranks(brix_sd_instance_t *inst, const int *ranks, int n)
23{
24 sd_http_inst_state *is;
25 int i;
26
27 if (!sd_http_instance_is(inst)) {
28 return;
29 }
30 is = inst->state;
31 for (i = 0; i < is->n_eps && i < n; i++) {
32 atomic_store_explicit(&is->eps[i].rank, ranks[i],
33 memory_order_relaxed);
34 }
35}

37/* Bench the last-answering endpoint after a verify (cvmfs-cas / manifest)
38 * failure. A digest mismatch is not a transport error, so the failover scorer
39 * never touched this endpoint's health — bump the SAME decaying fail_score a
40 * transport failure would raise, hard enough to lose the next pick to any
41 * clean-scored sibling. Two units (2*256) clears the 128 hysteresis band in
42 * one shot and outlives the half-open probe's single re-selection; it decays
43 * back on subsequent good fills, so a transient corruption does not
44 * permanently demote a healthy origin (a rank change would). Written from a
45 * fill thread; fail_score is racy-by-design like the rest of the score set. */
46#define SD_HTTP_VERIFY_PENALTY 512
47void
48sd_http_penalize_last_origin(brix_sd_instance_t *inst)
49{
50 sd_http_inst_state *is;
51
52 if (!sd_http_instance_is(inst)) {
53 return;
54 }
55 is = inst->state;
56 if (is->cur_ep < 0 || is->cur_ep >= is->n_eps) {
57 return; /* no origin has answered yet */
58 }
59 is->eps[is->cur_ep].fail_score += SD_HTTP_VERIFY_PENALTY;
60}

90/* "host:port" of the endpoint that answered the most recent read (display
91 * only; racy-by-design). 0 with buf filled, or -1 (non-http / none yet). */
92int
93sd_http_last_origin(brix_sd_instance_t *inst, char *buf, size_t cap)
94{
95 sd_http_inst_state *is;
96
97 if (!sd_http_instance_is(inst)) {
98 return -1;
99 }
100 is = inst->state;
101 if (is->last_origin[0] == '\0') {
102 return -1;
103 }
