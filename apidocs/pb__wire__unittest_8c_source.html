18static int bytes_eq(const pb_writer *w, const char *hex)
19{
20 size_t n = strlen(hex) / 2, i;
21 if (w->len != n) {
22 return 0;
23 }
24 for (i = 0; i < n; i++) {
25 unsigned v;
26 sscanf(hex + 2 * i, "%2x", &v);
27 if (w->p[i] != (unsigned char) v) {
28 return 0;
29 }
30 }
31 return 1;
32}
