1/*
2 * sd_http_read.c — read path for the HTTP-origin storage driver.
3 *
4 * WHAT: The read-facing vtable slots — open/open_cred (HEAD the URL for its
5 * size, build a memory-served object shell), close, pread (byte-Range
6 * GET, slicing a whole-object 200 response so the fill loop terminates
7 * against a Range-ignoring origin), fstat, stat/stat_cred — plus the
8 * per-open credential resolution (bearer header + x509 client-cert path)
9 * the write leg (sd_http_write.c) shares.
10 *
11 * WHY: Split out of sd_http.c (phase-79 file-size split): the read/credential
12 * path is one concept, distinct from selection/failover (sd_http_select.c),
13 * the write path (sd_http_write.c), and the driver vtable/lifecycle
14 * (sd_http.c). Every request goes through sd_http_request_fo (select.c),
15 * which owns endpoint choice + failover.
16 *
17 * HOW: cred_gate + resolve_open_cred compute the presented identity once, so
18 * the HEAD size probe and the following reads authenticate identically;
19 * the identity travels as per-object state (no kernel fd / session on an
20 * HTTP origin). sd_http_cred_gate / sd_http_resolve_open_cred are exported
21 * (sd_http_internal.h) because the staged-write leg presents the SAME
22 * identity at commit.
23 */

36/* Per-open object state: an HTTP origin has no kernel fd, so the export key and
37 * the per-user credential resolved at open time ride in the object itself. */
38typedef struct {
39 char key[SD_HTTP_PATH_MAX]; /* export-relative key (leading '/'); the
40 full URL path is composed per endpoint */
41 char auth_hdr[SD_HTTP_AUTH_MAX]; /* per-open "Authorization: Bearer <tok>\r\n"
42 (Phase 2 T7); "" when the object should
43 fall back to the instance's static
44 is->auth_hdr (plain open, or a cred with
45 no usable bearer). A COPY of the bearer
46 bytes — cred->bearer is only borrowed for
47 the duration of the open() call. */
48 char cert_pem[SD_HTTP_PATH_MAX]; /* per-open TLS client-cert PATH (phase-70
49 §5.1 GSI-over-https): the user's proxy PEM
50 (chain+key) presented via mutual-TLS on
51 each read. "" when the open carries no
52 x509 cred. A COPY of cred->x509_proxy,
53 which is only borrowed for the open call. */
54} sd_http_obj_state;

56/* HEAD `key` → *size_out (−1 if no Content-Length). 0, or −1 with errno.
57 * `auth_hdr`/`cert_pem` carry the per-open credential (bearer header and/or
58 * GSI proxy client-cert path) so the size probe authenticates as the same
59 * identity the following reads use; NULL/"" fall back to the instance static. */
60static int
61sd_http_head_size(sd_http_inst_state *is, const char *key,
62 const char *auth_hdr, const char *cert_pem, int64_t *size_out)
63{
64 brix_s3_resp_t resp;
65 char cl[32];
66 sd_http_req_t rq = { is, "HEAD", key, auth_hdr, cert_pem, &resp,
67 g_sd_http_force_primary };
68
69 if (sd_http_request_fo(&rq, NULL) != 0)
70 {
71 return -1; /* errno = EIO */
72 }
73 if (resp.status == 404) {
74 is->transport->resp_free(&resp);
75 errno = ENOENT;
76 return -1;
77 }
78 if (resp.status != 200) {
79 is->transport->resp_free(&resp);
80 errno = (resp.status == 403 || resp.status == 401) ? EACCES : EIO;
81 return -1;
82 }
83 if (is->transport->resp_header(&resp, "Content-Length", cl, sizeof(cl)) == 0) {
84 *size_out = (int64_t) strtoll(cl, NULL, 10);
85 } else {
86 *size_out = -1;
87 }
88 is->transport->resp_free(&resp);
89 return 0;
90}

92/* sd_http_cred_gate — decide whether a proxy-only x509 credential is presentable
93 * to this HTTP origin, and refuse in deny mode when it is not.
94 *
95 * WHAT: An x509-proxy cred over an https backend leg is presented as a mutual-TLS
96 * client cert (phase-70 §5.1) — but ONLY if the injected transport can do
97 * so (it implements request_cred). Returns 0 when the open may proceed,
98 * -1 (errno=EACCES) when a deny-mode proxy cred cannot be presented.
99 * WHY: fallback_deny forbids silently serving a per-user request on the
100 * anonymous/service credential. If the user presented ONLY an x509 proxy
101 * (no bearer) and the transport cannot mutual-TLS, presenting it is
102 * impossible; deny rather than leak onto anonymous access.
103 * HOW: Only fires for a cred whose sole credential is x509_proxy (no bearer).
104 * When the transport lacks request_cred and deny is set → EACCES. A usable
105 * transport, a bearer-carrying cred, or allow-mode all return 0 (proceed;
106 * the open then wires whatever it can). */
107int
108sd_http_cred_gate(sd_http_inst_state *is, const brix_sd_cred_t *cred)
109{
110 int has_bearer;
111 int has_proxy;
112
113 if (cred == NULL) {
114 return 0;
115 }
116 has_bearer = (cred->bearer != NULL && cred->bearer[0] != '\0');
117 has_proxy = (cred->x509_proxy != NULL && cred->x509_proxy[0] != '\0');
118 if (has_bearer || !has_proxy) {
119 return 0; /* bearer path, or no per-user cred */
120 }
121 if (is->transport->request_cred == NULL && cred->fallback_deny) {
122 errno = EACCES; /* proxy-only + can't mutual-TLS + deny */
123 return -1;
124 }
125 return 0;
126}

128/* sd_http_resolve_open_cred — resolve a `cred` into the per-open bearer header
129 * and x509 client-cert path used for both the HEAD probe and later reads.
130 *
131 * WHAT: Writes the "Authorization: Bearer <tok>\r\n" line into `open_auth`
132 * (empty when no usable bearer) and returns the x509 proxy PATH (NULL
133 * when none / the transport cannot present a client cert).
134 * WHY: The size probe and the object's reads must present the SAME identity;
135 * resolving once here keeps them consistent for an origin that authorizes
136 * per-object. cred==NULL leaves auth empty → fall back to the static.
137 * HOW: Bearer → snprintf into open_auth; x509 proxy → return cred->x509_proxy
138 * only when request_cred is available (both are borrowed, copied later). */
139const char *
140sd_http_resolve_open_cred(sd_http_inst_state *is, const brix_sd_cred_t *cred,
141 char *open_auth, size_t auth_cap)
142{
143 open_auth[0] = '\0';
144 if (cred == NULL) {
145 return NULL;
146 }
147 if (cred->bearer != NULL && cred->bearer[0] != '\0') {
148 snprintf(open_auth, auth_cap, "Authorization: Bearer %s\r\n",
149 cred->bearer);
150 }
151 if (cred->x509_proxy != NULL && cred->x509_proxy[0] != '\0'
152 && is->transport->request_cred != NULL)
153 {
154 return cred->x509_proxy;
155 }
156 return NULL;
157}

159/* Resolved per-open result threaded into sd_http_build_obj: the credential the

161typedef struct {

164 int64_t size; /* HEAD Content-Length (−1 if unknown) */
165} sd_http_open_result_t;

167/* sd_http_build_obj — allocate + populate the memory-served object shell.
168 *
169 * WHAT: Allocates the per-open object state and shell, copies the key and the
170 * resolved per-open credential (bearer header / cert path) into the
171 * object's own buffers, and fills the stat snapshot. NULL + *err_out on
172 * allocation failure.
173 * HOW: calloc both, COPY key/open_auth/open_cert into st (cred fields were only
174 * borrowed), wire the read-only regular-file snapshot. No kernel fd. */
175static brix_sd_obj_t *
176sd_http_build_obj(brix_sd_instance_t *inst, const char *path,
177 const sd_http_open_result_t *res, int *err_out)
178{
179 sd_http_obj_state *st = calloc(1, sizeof(*st));
180 brix_sd_obj_t *obj = calloc(1, sizeof(*obj));
181
182 if (st == NULL || obj == NULL) {
183 free(st);
184 free(obj);
185 if (err_out) { *err_out = ENOMEM; }
186 return NULL;
187 }
188 snprintf(st->key, sizeof(st->key), "%s",
189 (path != NULL && path[0]) ? path : "/");
190 if (res->open_auth[0] != '\0') {
191 snprintf(st->auth_hdr, sizeof(st->auth_hdr), "%s", res->open_auth);
192 }
193 if (res->open_cert != NULL) {
194 snprintf(st->cert_pem, sizeof(st->cert_pem), "%s", res->open_cert);
195 }
196
197 obj->driver = inst->driver;
198 obj->inst = inst;
199 obj->fd = NGX_INVALID_FILE; /* no kernel fd — memory-served */
200 obj->state = st;
201 obj->heap_shell = 1;
202 obj->snap.size = (off_t) res->size;
203 obj->snap.mode = S_IFREG | 0444;
204 obj->snap.is_reg = 1;
205 return obj;
206}

218/* sd_http_open_common — shared open path for the plain and credential-scoped
219 * open slots.
220 *
221 * WHAT: HEADs `req->path` for its size and builds the per-open object. A `cred`
222 * with a usable bearer token sets the object's auth_hdr; a `cred` with an
223 * x509 proxy (a PEM chain+key path) sets the object's cert_pem so
224 * subsequent reads present that identity — the bearer via an Authorization
225 * header, the proxy via a mutual-TLS client cert on the origin handshake.
226 * WHY: Phase 2 T7 (bearer) + phase-70 §5.1 (x509) per-user backend credentials
227 * — an HTTP-origin driver has no kernel fd / session to re-scope per user,
228 * so the per-user identity travels as per-object state copied at open time.
229 * HOW: Refuse write intent; gate the cred; resolve it once (helper); HEAD the
230 * size with the SAME identity; then build the object (helper). Exactly one
231 * cred kind is ever set (the VFS gate populates one of bearer/x509_proxy). */
232static brix_sd_obj_t *
233sd_http_open_common(brix_sd_instance_t *inst, const sd_http_open_req_t *req,
234 int *err_out)
235{
236 sd_http_inst_state *is = inst->state;
237 int64_t size = 0;
238 char open_auth[SD_HTTP_AUTH_MAX];
239 const char *open_cert;
240
241 (void) req->mode;
242
243 /* Read-only source: refuse any write/create/trunc intent. */
244 if (req->sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE | BRIX_SD_O_TRUNC
245 | BRIX_SD_O_APPEND))
246 {
247 if (err_out) { *err_out = EROFS; }
248 return NULL;
249 }
250
251 if (sd_http_cred_gate(is, req->cred) != 0) {
252 if (err_out) { *err_out = errno; }
253 return NULL;
254 }
255
256 open_cert = sd_http_resolve_open_cred(is, req->cred, open_auth,
257 sizeof(open_auth));
258
259 if (sd_http_head_size(is, req->path,
260 open_auth[0] ? open_auth
261 : (is->auth_hdr[0] ? is->auth_hdr : NULL),
262 open_cert, &size) != 0)
263 {
264 if (err_out) { *err_out = errno; }
265 return NULL;
266 }
267
268 sd_http_open_result_t res = { open_auth, open_cert, size };
269 return sd_http_build_obj(inst, req->path, &res, err_out);
270}

314ssize_t
315sd_http_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
316{
317 sd_http_inst_state *is = obj->inst->state;
318 sd_http_obj_state *st = obj->state;
319 brix_s3_resp_t resp;
320 char hdrs[SD_HTTP_AUTH_MAX + 80];
321 const void *body;
322 const char *auth_hdr;
323 size_t blen = 0, n;
324 int64_t end;
325
326 if (len == 0) {
327 return 0;
328 }
329 if (len > (size_t) SD_HTTP_PREAD_MAX) {
330 len = (size_t) SD_HTTP_PREAD_MAX;
331 }
332 end = (int64_t) off + (int64_t) len - 1;
333 /* Phase 2 T7: a per-open bearer (open_cred) wins over the instance's
334 * static bearer_token; "" (plain open, or no usable cred) falls back.
335 * Phase-70 §5.1: a per-open x509 proxy path (st->cert_pem) is presented as
336 * the mutual-TLS client cert on the read — orthogonal to the bearer header
337 * (the VFS gate sets exactly one of the two). */
338 auth_hdr = st->auth_hdr[0] ? st->auth_hdr : is->auth_hdr;
339 snprintf(hdrs, sizeof(hdrs), "Range: bytes=%lld-%lld\r\n%s",
340 (long long) off, (long long) end, auth_hdr);
341
342 sd_http_req_t rq = { is, "GET", st->key, hdrs,
343 st->cert_pem[0] ? st->cert_pem : NULL, &resp,
344 g_sd_http_force_primary };
345 if (sd_http_request_fo(&rq, NULL) != 0) {
346 return -1; /* errno = EIO */
347 }
348 if (resp.status == 416) {
349 is->transport->resp_free(&resp);
350 return 0; /* range past EOF → EOF (0) */
351 }
352 if (resp.status != 206 && resp.status != 200) {
353 is->transport->resp_free(&resp);
354 errno = (resp.status == 404) ? ENOENT : EIO;
355 return -1;
356 }
357 body = is->transport->resp_body(&resp, &blen);
358 if (body == NULL || blen == 0) {
359 is->transport->resp_free(&resp);
360 return 0; /* EOF / empty range */
361 }
362
363 /* 206 → body is exactly the requested range, starting at `off`. 200 → the
364 * origin ignored the Range header and returned the WHOLE object from byte 0
365 * (stock python http.server, some proxies), so the bytes we want begin at
366 * `off` within `body`; past EOF that is a short read of 0. Slicing here keeps
367 * a correct, terminating fill loop against either kind of origin. */
368 if (resp.status == 200) {
369 if ((size_t) off >= blen) {
370 is->transport->resp_free(&resp);
371 return 0; /* requested range past EOF */
372 }
373 body = (const char *) body + off;
374 blen -= (size_t) off;
375 }
376 n = (blen < len) ? blen : len;
377 memcpy(buf, body, n);
378 is->transport->resp_free(&resp);
379 return (ssize_t) n;
380}

423/* sd_http_stat_cred — vtable stat_cred slot: per-user credential-scoped stat.
424 *
425 * WHAT: Namespace stat (a HEAD size probe) that presents the requesting user's
426 * credential — a WLCG bearer as the Authorization header, or an x509 proxy
427 * as the mutual-TLS client cert — to the origin instead of the static
428 * service credential.
429 * WHY: Phase-70 §5.1 — an https backend leg authorizes EVERY request on the
430 * presented credential, so a namespace stat issued under a per-user policy
431 * (e.g. the root:// write pre-flight existence check, or a WebDAV lock-state
432 * probe on a remote origin) must carry the same forwarded identity the
433 * open/staged-open legs use. Without this slot the stat dispatched through
434 * the plain .stat and hit the auth-required origin ANONYMOUSLY, which the
435 * backend rejected — aborting the whole two-hop PUT even though the user's
436 * credential was fully resolved. sd_http has no kernel fd / session to
437 * re-scope, so — exactly like sd_http_open_cred — the credential is applied
438 * per request via the HEAD headers + client cert.
439 * HOW: Runs sd_http_cred_gate (deny a proxy-only cred the transport cannot
440 * mutual-TLS present), resolves the cred into a bearer header line + x509
441 * cert path with sd_http_resolve_open_cred (the SAME resolver the read open
442 * uses, so stat and open present one identity), then HEAD-probes with those
443 * — falling back to the instance static header only when the cred yields no
444 * bearer. cred==NULL degrades to the plain-stat behaviour. */
445ngx_int_t
446sd_http_stat_cred(brix_sd_instance_t *inst, const char *path,
447 brix_sd_stat_t *out, const brix_sd_cred_t *cred)
448{
449 sd_http_inst_state *is = inst->state;
450 int64_t size = 0;
451 char open_auth[SD_HTTP_AUTH_MAX];
452 const char *open_cert;
453
454 if (sd_http_cred_gate(is, cred) != 0) {
455 return NGX_ERROR; /* errno = EACCES (set by gate) */
456 }
457
458 open_cert = sd_http_resolve_open_cred(is, cred, open_auth,
459 sizeof(open_auth));
460
461 if (sd_http_head_size(is, path,
462 open_auth[0] ? open_auth

464 open_cert, &size) != 0) {
465 return NGX_ERROR; /* errno set by head_size */
466 }
467 sd_http_stat_fill(out, size);
468 return NGX_OK;
