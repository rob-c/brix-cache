30/* Read the member's raw bytes at data_off and decompress (store=copy,
31 * deflate=raw inflate); compare to want/want_len. Returns 1 on match. */
32static int verify_content(int fd, const brix_zip_member_t *m,
33 const unsigned char *want, size_t want_len)
34{
35 unsigned char *comp = malloc(m->comp_size ? m->comp_size : 1);
36 unsigned char *out = malloc(m->uncomp_size ? m->uncomp_size : 1);
37 int ok = 0;
38
39 if (pread(fd, comp, m->comp_size, (off_t) m->data_off)
40 != (ssize_t) m->comp_size) {
41 goto done;
42 }
43 if (m->uncomp_size != want_len) {
44 goto done;
45 }
46 if (m->method == BRIX_ZIP_METHOD_STORE) {
47 ok = (memcmp(comp, want, want_len) == 0);
48 } else {
49 z_stream zs;
50 memset(&zs, 0, sizeof(zs));
51 /* -15 = raw deflate, no zlib header (ZIP member framing). */
52 if (inflateInit2(&zs, -15) != Z_OK) {
53 goto done;
54 }
55 zs.next_in = comp; zs.avail_in = (uInt) m->comp_size;
56 zs.next_out = out; zs.avail_out = (uInt) m->uncomp_size;
57 int zr = inflate(&zs, Z_FINISH);
58 inflateEnd(&zs);
59 ok = (zr == Z_STREAM_END
60 && zs.total_out == want_len
61 && memcmp(out, want, want_len) == 0);
62 }
63done:
64 free(comp);
65 free(out);
66 return ok;
67}

70{
71 if (argc < 2) {

74 }

80 brix_zip_member_t m;
81 int rc;

83 /* 1. stored member resolves + content matches. */
84 rc = brix_zip_find_member(fd, st.st_size, "stored.txt", 16u << 20, &m);
85 CHECK(rc == BRIX_ZIP_OK, "stored.txt found");
86 if (rc == BRIX_ZIP_OK) {
87 const char *want = "hello stored member\n";
88 CHECK(m.method == BRIX_ZIP_METHOD_STORE, "stored.txt method=store");
89 CHECK(m.uncomp_size == strlen(want), "stored.txt uncomp size");
90 CHECK(verify_content(fd, &m, (const unsigned char *) want, strlen(want)),
91 "stored.txt bytes at data_off match");
92 }
93
94 /* 2. deflate member resolves + inflates to expected content. */
95 rc = brix_zip_find_member(fd, st.st_size, "sub/defl.bin", 16u << 20, &m);
96 CHECK(rc == BRIX_ZIP_OK, "sub/defl.bin found");
97 if (rc == BRIX_ZIP_OK) {
98 size_t n = 100000;
99 unsigned char *want = malloc(n);
100 for (size_t i = 0; i < n; i++) want[i] = (unsigned char) ((i * 31 + 7) & 0xff);
101 CHECK(m.method == BRIX_ZIP_METHOD_DEFLATE, "sub/defl.bin method=deflate");
102 CHECK(m.uncomp_size == n, "sub/defl.bin uncomp size");
103 CHECK(m.comp_size < n, "sub/defl.bin actually compressed");
104 CHECK(verify_content(fd, &m, want, n),
105 "sub/defl.bin inflates to expected content");
106 free(want);
107 }
108
109 /* 2b. ZIP64-forced member: exercises the saturated-size → zip64-extra path. */
110 rc = brix_zip_find_member(fd, st.st_size, "big64.txt", 16u << 20, &m);
111 CHECK(rc == BRIX_ZIP_OK, "big64.txt (zip64 extra) found");
112 if (rc == BRIX_ZIP_OK) {
113 const char *want = "zip64 forced member\n";
114 CHECK(m.uncomp_size == strlen(want), "big64.txt zip64 uncomp size");
115 CHECK(verify_content(fd, &m, (const unsigned char *) want, strlen(want)),
116 "big64.txt bytes at data_off match");
117 }
118
119 /* 3. missing member → NOMEMBER. */
120 rc = brix_zip_find_member(fd, st.st_size, "missing.xyz", 16u << 20, &m);
121 CHECK(rc == BRIX_ZIP_NOMEMBER, "missing.xyz → NOMEMBER");
122
123 /* 4. traversal / bad names rejected. */
124 rc = brix_zip_find_member(fd, st.st_size, "", 16u << 20, &m);
125 CHECK(rc == BRIX_ZIP_ECORRUPT, "empty name rejected");
126
127 /* 5. tiny cd_max bomb guard trips on a real (larger) central directory. */
128 rc = brix_zip_find_member(fd, st.st_size, "stored.txt", 4, &m);
129 CHECK(rc == BRIX_ZIP_ECORRUPT, "cd_max bomb guard trips");
130
131 close(fd);
132 printf("\n%s\n", failures ? "UNITTEST FAILED" : "UNITTEST PASSED");
133 return failures ? 1 : 0;
134}
