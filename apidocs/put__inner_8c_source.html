10/* Directory-sentinel fast path: if fs_path names an S3 directory sentinel
11 * (S3_DIR_SENTINEL), create the parent directory plus the zero-byte sentinel,
12 * finalize the request, and return 1 (the caller must return). Returns 0 when
13 * fs_path is a normal object, so the caller proceeds with the body write. */
14static int
15s3_put_try_sentinel(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
16 const u_char *fs_path)
17{
18 size_t plen = ngx_strlen(fs_path);
19 size_t slen = sizeof(S3_DIR_SENTINEL) - 1;
20 char parent[PATH_MAX];
21 char *slash;
22 int fd;
23 char identity[128];
24
25 if (!(plen >= slen
26 && ngx_strncmp(fs_path + plen - slen,
27 (u_char *) S3_DIR_SENTINEL, slen) == 0))
28 {
29 return 0;
30 }
31
32 /* Create the directory that the sentinel lives in. */
33 if (plen >= sizeof(parent)) {
34 s3_put_finalize_error(r);
35 return 1;
36 }
37 memcpy(parent, fs_path, plen + 1);
38 slash = strrchr(parent, '/');
39 if (slash != NULL) {
40 *slash = '\0';
41 }
42 {
43 brix_vfs_ctx_t pctx;
44
45 s3_build_vfs_ctx(r, parent, cf, &pctx);
46 if (brix_vfs_mkdir(&pctx, 0755, 0 /* no parents */) != NGX_OK
47 && errno != EEXIST)
48 {
49 int mk_errno = errno; /* capture before logging clobbers it */
50 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, mk_errno,
51 "s3: mkdir(\"%s\") failed", parent);
52 s3_put_finalize_fs_error(r, mk_errno);
53 return 1;
54 }
55 }
56
57 /* Write the zero-byte sentinel (confined create via the VFS seam). */
58 fd = brix_vfs_open_fd(r->connection->log, cf->common.root_canon,
59 (const char *) fs_path,
60 O_WRONLY | O_CREAT | O_TRUNC, 0644);
61 if (fd < 0) {
62 int open_errno = errno; /* capture before logging clobbers it */
63 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, open_errno,
64 "s3: open(\"%s\") for sentinel failed",
65 (const char *) fs_path);
66 s3_put_finalize_fs_error(r, open_errno);
67 return 1;
68 }
69 close(fd);
70
71 s3_dashboard_identity(r, cf, identity, sizeof(identity));
72 (void) brix_dashboard_http_start_identity(r, (const char *) fs_path,
73 identity, "", BRIX_XFER_PROTO_S3, BRIX_XFER_DIR_WRITE,
74 "PutObjectSentinel", 0);
75 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_DIR_SENTINEL]);
76 BRIX_S3_METRIC_INC(put_body_total[BRIX_S3_PUT_EMPTY]);
77 s3_put_finalize_empty_ok(r);
78 return 1;
79}

82/*
83 * s3put_ensure_parent_dirs — create the object's parent prefix if absent.
84 *
85 * WHAT: derives the parent directory of fs_path and ensures it exists via the
86 * confined VFS mkdir (with a probe fast-path for the hot "many objects into
87 * one prefix" case).
88 * WHY: keeps the phase-46 W2a probe-then-mkdir optimization and its DAC-denied
89 * → 403 mapping in one focused helper, out of the top-level flow.
90 * HOW: returns NGX_OK to continue; returns NGX_DONE after finalizing the
91 * request with an fs error (caller must return).
92 */
93static ngx_int_t
94s3put_ensure_parent_dirs(s3put_state_t *st)
95{
96 char parent[PATH_MAX];
97 char *last_slash;
98 size_t flen = strlen((const char *) st->fs_path);
99
100 if (flen >= sizeof(parent)) {
101 return NGX_OK;
102 }
103
104 memcpy(parent, st->fs_path, flen + 1);
105 last_slash = strrchr(parent, '/');
106 if (last_slash == NULL || last_slash == parent) {
107 return NGX_OK;
108 }
109
110 *last_slash = '\0';
111
112 {
113 /* Object-store leaf (phase-71 caps / P80.2): a namespace without
114 * CAP_DIRS_WRITE has no physical catalog to mutate — parent prefixes
115 * are virtual, so there is nothing to create here (and the VFS mkdir
116 * caps gate would reject with EPERM, turning every first PUT into a
117 * 403). Capability-driven, never scheme-driven. */
118 brix_sd_instance_t *leaf = brix_vfs_ns_leaf(
119 brix_vfs_backend_resolve(st->cf->common.root_canon,
120 st->r->connection->log));
121
122 if (leaf != NULL && !(brix_sd_caps(leaf) & BRIX_SD_CAP_DIRS_WRITE)) {
123 return NGX_OK;
124 }
125 }
126
127 {
128 brix_vfs_ctx_t pctx;
129 brix_vfs_stat_t pst;
130
131 /*
132 * phase-46 W2a: fast-path the common "many objects into one prefix"
133 * pattern. A single confined probe that finds an existing directory
134 * replaces the per-path-component mkdirat(EEXIST) storm of the recursive
135 * mkdir. A symlink or a miss falls through to the confined vfs_mkdir,
136 * which re-enforces confinement.
137 */
138 s3_build_vfs_ctx(st->r, parent, st->cf, &pctx);
139 if (brix_vfs_probe(&pctx, 1 /* no-follow */, &pst) == NGX_OK
140 && pst.is_directory)
141 {
142 /* parent prefix already exists — nothing to create */
143 return NGX_OK;
144 }
145 }
146
147 {
148 brix_vfs_ctx_t pvctx;
149
150 s3_build_vfs_ctx(st->r, parent, st->cf, &pvctx);
151 if (brix_vfs_mkdir(&pvctx, 0755, 1 /* parents */) != NGX_OK
152 && errno != EEXIST)
153 {
154 int mk_errno = errno; /* capture before logging clobbers it */
155 brix_log_safe_path(st->r->connection->log, NGX_LOG_ERR, mk_errno,
156 "s3: mkdirs_for(\"%s\") failed",
157 (const char *) st->fs_path);
158 /* DAC-denied create of a parent dir → 403, not 500. */
159 s3_put_finalize_fs_error(st->r, mk_errno);
160 return NGX_DONE;
161 }
162 }
163
164 return NGX_OK;
165}

168/*
169 * s3put_precondition — resolve the object, run the sentinel fast-path, and
170 * ensure parent directories.
171 *
172 * WHAT: the pre-write phase — validates fs_path, handles the directory-sentinel
173 * create-and-return path, then creates any missing parent prefix.
174 * WHY: groups the checks that must pass before we open a staged file, each with
175 * its own finalize-and-return path, into one place.
176 * HOW: returns NGX_OK to continue to the open phase; returns NGX_DONE when the
177 * request has already been finalized (missing path, sentinel handled, or a
178 * parent-dir error).
179 */
180static ngx_int_t
181s3put_precondition(s3put_state_t *st)
182{
183 if (st->fs_path == NULL) {
184 s3_put_finalize_error(st->r);
185 return NGX_DONE;
186 }
187
188 /* Directory-sentinel objects take a separate create-dir-and-return path. */
189 if (s3_put_try_sentinel(st->r, st->cf, st->fs_path)) {
190 return NGX_DONE;
191 }
192
193 /* Create parent directories if needed using the shared confined mkdir helper. */
194 return s3put_ensure_parent_dirs(st);
195}

198/*
199 * s3put_open_target — open the unified write session (atomic temp+publish) for
200 * the object.
201 *
202 * WHAT: opens a brix_vfs_writer session into which the body is written before the
203 * atomic commit.
204 * WHY: the phase-74 exclusive-create semantics are frozen here — this helper is
205 * the single place that opens the target, so the staged mode/perms live in
206 * exactly one spot. BRIX_VFS_O_ATOMIC forces the temp+publish path even for
207 * random-capable backends, preserving S3's no-partial-at-final invariant.
208 * HOW: on success stores the session in st->writer and returns NGX_OK; on failure
209 * finalizes with the confinement-aware fs error (403 not 500) and returns
210 * NGX_DONE.
211 */
212static ngx_int_t
213s3put_open_target(s3put_state_t *st)
214{
215 brix_vfs_ctx_t svctx;
216 int staged_err = 0;
217
218 /* A transient stack ctx is fine: brix_vfs_writer_open deep-copies the ctx
219 * (and the strings it points at) onto r->pool, so the session survives the async
220 * body-write completion (put_aio/put_chunk commit after this function returns). */
221 s3_build_vfs_ctx(st->r, (const char *) st->fs_path, st->cf, &svctx);
222 st->writer = brix_vfs_writer_open(&svctx, BRIX_VFS_O_ATOMIC,
223 st->cf->common.verify_write, &staged_err);
224 if (st->writer == NULL) {
225 int open_errno = staged_err; /* the VFS open captured errno */
226 brix_log_safe_path(st->r->connection->log, NGX_LOG_ERR, open_errno,
227 "s3: staged open for \"%s\" failed",
228 (const char *) st->fs_path);
229 /* A DAC-denied / confinement-blocked create is a 403, not a 500. */
230 s3_put_finalize_fs_error(st->r, open_errno);
231 return NGX_DONE;
232 }
233
234 return NGX_OK;
235}
