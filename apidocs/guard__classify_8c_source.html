2 * guard_classify.c — bad-actor request classification.
3 *
4 * WHAT: the classifier half of the guard core: signature-blocklist matching,
5 * namespace-grammar checks, and the pre-backend / post-response verdicts
6 * built on them.
7 * WHY: every adapter (ARC HTTP, XrdHttp/WebDAV, root:// relay) needs the same
8 * junk-vs-legitimate decision; keeping it pure C makes the decision
9 * testable without a server.
10 * HOW: pure functions over a caller-built guard_ruleset_t and a normalized

16/* ---- Test one signature pattern against a path ----
17 *
18 * WHAT: returns 1 if `path[0..len)` matches signature `s` (case-sensitive
19 * suffix, prefix, or substring per s->kind), else 0.
20 *
21 * WHY: the three pattern kinds cover the junk-scanner corpus (extension
22 * probes, well-known directory probes, embedded traversal/artifact tokens)
23 * without regex machinery in a pure-C core.
24 *
25 * HOW: 1. Reject empty or over-long patterns (can never match).
26 * 2. SUFFIX: memcmp against the path tail.
27 * 3. PREFIX: memcmp against the path head.
28 * 4. SUBSTR: sliding-window memcmp across the path.
29 */
30static int
31sig_hit(const guard_sig_t *s, const char *path, size_t len)
32{
33 if (s->pat_len == 0 || s->pat_len > len) {
34 return 0;
35 }
36 switch (s->kind) {
37 case GUARD_SIG_SUFFIX:
38 return memcmp(path + len - s->pat_len, s->pat, s->pat_len) == 0;
39 case GUARD_SIG_PREFIX:
40 return memcmp(path, s->pat, s->pat_len) == 0;
41 case GUARD_SIG_SUBSTR:
42 default: {
43 size_t window_start;
44 for (window_start = 0; window_start + s->pat_len <= len;
45 window_start++)
46 {
47 if (memcmp(path + window_start, s->pat, s->pat_len) == 0) {
48 return 1;
49 }
50 }
51 return 0;
52 }
53 }
54}

56/* ---- Match a path against the ruleset's signature blocklist ----
57 *
58 * WHAT: returns 1 if `path[0..len)` matches any configured signature, else 0.
59 *
60 * WHY: signatures are the highest-confidence bad-actor tell — a single hit on
61 * a grid data path (".php", "/wp-", "/.env") identifies a scanner, so this
62 * is the first check every adapter runs.
63 *
64 * HOW: 1. Linearly test each configured signature via sig_hit().
65 * 2. First hit wins; order is irrelevant to the verdict.
66 */
67int
68guard_signature_match(const guard_ruleset_t *rs, const char *path, size_t len)
69{
70 int sig_index;
71
72 for (sig_index = 0; sig_index < rs->n_sigs; sig_index++) {
73 if (sig_hit(&rs->sigs[sig_index], path, len)) {
74 return 1;
75 }
76 }
77 return 0;
78}

80/* ---- Check (op, path) against the namespace grammar ----
81 *
82 * WHAT: returns 1 if the op-class is permitted and the path starts with one
83 * of the configured namespace prefixes (or no prefixes are configured),
84 * else 0.
85 *
86 * WHY: each fronted service has a tiny legitimate namespace (ARC REST roots,
87 * an export root); anything outside it is scanner traffic even when no
88 * signature fires.
89 *
90 * HOW: 1. Reject ops the profile does not allow.
91 * 2. With no prefixes configured, any path is in-grammar (op-only mode).
92 * 3. Otherwise accept on the first prefix that heads the path.
93 */
94int
95guard_grammar_ok(const guard_ruleset_t *rs, guard_op_class_t op,
96 const char *path, size_t len)
97{
98 int prefix_index;
99
100 if (!rs->op_allowed[op]) {
101 return 0;
102 }
103 if (rs->n_prefixes == 0) {
104 return 1; /* no prefixes configured = any path */
105 }
106 for (prefix_index = 0; prefix_index < rs->n_prefixes; prefix_index++) {
107 if (len >= rs->prefix_len[prefix_index]
108 && memcmp(path, rs->prefixes[prefix_index],
109 rs->prefix_len[prefix_index]) == 0)
110 {
111 return 1;
112 }
113 }
114 return 0;
115}

117/* ---- Does buf open with an HTTP request line? ----
118 *
119 * WHAT: returns 1 if buf[0..len) begins with a recognized HTTP method token
120 * followed by a space (GET/POST/HEAD/PUT/OPTIONS/DELETE/CONNECT/TRACE/PATCH).
121 *
122 * WHY: a web scanner knocking on a root:// port opens with an HTTP request
123 * line; naming it in the audit tells the operator exactly what probed them.
124 *
125 * HOW: 1. Match each method's leading bytes plus the trailing space so a
126 * binary frame that merely starts with those letters is not misread.
127 */
128static int
129is_http_line(const unsigned char *buf, size_t len)
130{
131 static const char *const verbs[] = {
132 "GET ", "POST ", "HEAD ", "PUT ", "OPTIONS ", "DELETE ",
133 "CONNECT ", "TRACE ", "PATCH "
134 };
135 size_t v;
136
137 for (v = 0; v < sizeof(verbs) / sizeof(verbs[0]); v++) {
138 size_t vlen = strlen(verbs[v]);
139 if (len >= vlen && memcmp(buf, verbs[v], vlen) == 0) {
140 return 1;
141 }
142 }
143 return 0;
144}

146/* ---- Wire-level handshake classifier ----
147 *
148 * WHAT: inspects the first bytes of a fresh connection on a root:// port and
149 * decides whether they are the kXR client handshake. Returns GUARD_WIRE_ROOT
150 * for a genuine (or still-incomplete-but-consistent) kXR opening, otherwise
151 * the best-effort identity of the non-root client (TLS/HTTP/SSH/empty/junk).
152 *
153 * WHY: the tap-based guard only sees DECODED kXR frames; a client that never
154 * speaks root — a TLS ClientHello, an HTTP scanner, an SSH bannergrab, raw
155 * junk — produces no frame and would otherwise sail through to the backend.
156 * This is the "who is knocking on the root port" classifier operators want.
157 *
158 * HOW: 1. A real kXR client opens with ClientInitHandShake: 12 zero bytes,
159 * then fourth == htonl(4) (bytes 12..15 == 00 00 00 04); fifth is left
160 * unchecked so odd-but-real clients still forward.
161 * 2. Reject the instant an available leading byte breaks the zero-prefix;
162 * defer (*need_more) only while every byte seen stays consistent and
163 * the 20-byte signature is not yet complete.
164 * 3. A non-root opening is identified for the audit line: TLS record
165 * (0x16 0x03), HTTP method line, "SSH-" banner, empty, else junk.
166 */
167guard_wire_t
168guard_classify_handshake(const unsigned char *buf, size_t len, int *need_more)
169{
170 size_t lead, i;
171
172 *need_more = 0;
173
174 if (len == 0) {
175 return GUARD_WIRE_EMPTY;
176 }
177
178 /* Step 1+2: is the opening consistent with the kXR zero-prefix? */
179 lead = len < 12 ? len : 12;
180 for (i = 0; i < lead; i++) {
181 if (buf[i] != 0) {
182 break; /* zero-prefix broken -> not root */
183 }
184 }
185 if (i == lead) { /* every leading byte zero so far */
186 if (len < 16) {
187 *need_more = 1; /* zero-prefix, fourth not yet in */
188 return GUARD_WIRE_ROOT;
189 }
190 /* fourth is now present; fifth is deliberately unchecked, so the
191 * 16-byte prefix is the whole verdict — final either way. */
192 if (buf[12] == 0 && buf[13] == 0 && buf[14] == 0 && buf[15] == 4) {
193 return GUARD_WIRE_ROOT; /* genuine kXR ClientInitHandShake */
194 }
195 return GUARD_WIRE_JUNK; /* 16+ zero-led bytes, fourth != 4 */
196 }
197
198 /* Step 3: name the non-root client for the operator. */
199 if (buf[0] == 0x16 && len >= 2 && buf[1] == 0x03) {
200 return GUARD_WIRE_TLS; /* TLS handshake record, version 3.x */
201 }
202 if (is_http_line(buf, len)) {
203 return GUARD_WIRE_HTTP;
204 }
205 if (len >= 4 && memcmp(buf, "SSH-", 4) == 0) {
206 return GUARD_WIRE_SSH;
207 }
208 return GUARD_WIRE_JUNK;
209}

211/* ---- Pre-backend verdict: bounce or allow ----
212 *
213 * WHAT: classifies a request before it reaches the backend. Returns
214 * GUARD_BOUNCE with *why set to the firing signal (signature > grammar),
215 * or GUARD_ALLOW with *why = GUARD_R_NONE.
216 *
217 * WHY: bouncing junk pre-backend keeps scanner noise out of ARC/XRootD logs
218 * and off their worker threads, and gives fail2ban a single high-signal
219 * line to ban on.
220 *
221 * HOW: 1. Signature blocklist first — highest confidence, wins over grammar.
222 * 2. Grammar check only when the ruleset enforces it (advisory mode
223 * still allows, adapters may log separately).
224 * 3. Otherwise allow.
225 */
226guard_verdict_t
227guard_classify_pre(const guard_ruleset_t *rs, const guard_request_t *req,
228 guard_reason_t *why)
229{
230 if (guard_signature_match(rs, req->path, req->path_len)) {
231 *why = GUARD_R_SIGNATURE; /* signatures win over grammar */
232 return GUARD_BOUNCE;
233 }
234 if (rs->enforce_grammar
235 && !guard_grammar_ok(rs, req->op, req->path, req->path_len))
236 {
237 *why = GUARD_R_GRAMMAR;
238 return GUARD_BOUNCE;
239 }
240 *why = GUARD_R_NONE;
241 return GUARD_ALLOW;
242}

244/* ---- Post-response signal classification ----
245 *
246 * WHAT: maps a completed request's outcome to a loggable bad-actor reason
247 * (GUARD_R_NOTFOUND / GUARD_R_AUTHFAIL) when the ruleset flags that
248 * outcome, else GUARD_R_NONE. Never bounces — the response already went
249 * out.
250 *
251 * WHY: not-found storms and repeated auth failures are per-request weak
252 * signals that fail2ban aggregates into a ban via jail thresholds; the
253 * guard's job is only to emit one clean line each.
254 *
255 * HOW: 1. NOTFOUND outcome + flag_notfound -> GUARD_R_NOTFOUND.
256 * 2. AUTHFAIL outcome + flag_authfail -> GUARD_R_AUTHFAIL.
257 * 3. Everything else is not a signal.
258 */
259guard_reason_t
260guard_classify_post(const guard_ruleset_t *rs, const guard_request_t *req)
261{
262 if (req->outcome == OUTCOME_NOTFOUND && rs->flag_notfound) {
263 return GUARD_R_NOTFOUND;
264 }
265 if (req->outcome == OUTCOME_AUTHFAIL && rs->flag_authfail) {
266 return GUARD_R_AUTHFAIL;
267 }
268 return GUARD_R_NONE;
269}
