23/*
24 * WHAT: The immutable parameter set of one collection (directory) COPY — the
25 * confinement root, the source and destination paths, the source directory
26 * mode, and the overwrite/depth flags derived from the request.
27 *
28 * WHY: The collection copy is executed either inline on the event loop or on a
29 * thread-pool worker, and it is described by eight loosely-related values. A
30 * struct threads them as a single unit through the execute / post-task /
31 * per-phase helpers — one job, one owner — instead of an eight-argument
32 * signature that trips the complexity gate and is easy to mis-order at a call
33 * site. It is thread-safe by construction: it holds copied stack-buffer paths
34 * (no ngx_http_request_t handle), so a worker thread reads only its own copy.
35 *
36 * HOW: Populated once by the caller from resolved paths and header-derived
37 * flags, then passed by const pointer to the execute helpers. The async task
38 * struct embeds a job plus the fields the event loop needs to reply.
39 */
40typedef struct {
41 ngx_log_t *log;
42 char root_canon[WEBDAV_MAX_PATH];
43 char src_path[WEBDAV_MAX_PATH];
44 char dst_path[WEBDAV_MAX_PATH];
45 mode_t src_mode;
46 ngx_flag_t dst_existed;
47 ngx_flag_t dst_was_dir;
48 ngx_flag_t depth_infinity;
49} webdav_copy_job_t;
50
51typedef struct {
52 webdav_copy_job_t job;
53 ngx_http_request_t *r;
54 ngx_int_t http_status;
55} webdav_copy_collection_task_t;

57/*
58 * WHAT: Populate a webdav_copy_job_t from a resolved COPY request and the
59 * export root, deep-copying the stack-buffer paths into the job's own storage.
60 *
61 * WHY: The collection-copy job derives entirely from the resolved request; a
62 * single builder keeps the inline and threaded paths in lock step and
63 * guarantees the job never aliases a caller stack buffer that could go out of
64 * scope before a worker thread reads it.
65 *
66 * HOW: Assigns the scalar flags/mode from req (dst_was_dir = existed && dir),
67 * and uses ngx_cpystrn (bounded, always NUL-terminating) for each path,
68 * matching the prior per-field copies.
69 */
70static void
71webdav_copy_job_init(webdav_copy_job_t *job, ngx_log_t *log,
72 const char *root_canon, const webdav_copy_req_t *req)
73{
74 job->log = log;
75 job->src_mode = req->src_sb.st_mode;
76 job->dst_existed = req->dst_existed;
77 job->dst_was_dir = req->dst_existed && S_ISDIR(req->dst_sb.st_mode);
78 job->depth_infinity = req->depth_infinity;
79
80 ngx_cpystrn((u_char *) job->root_canon, (u_char *) root_canon,
81 sizeof(job->root_canon));
82 ngx_cpystrn((u_char *) job->src_path, (u_char *) req->src_path,
83 sizeof(job->src_path));
84 ngx_cpystrn((u_char *) job->dst_path, (u_char *) req->dst_path,
85 sizeof(job->dst_path));
86}

88/*
89 * WHAT: Stage the copied tree into a sibling temp directory: make the temp dir,
90 * clone dead props + fattrs, and (depth:infinity) recurse the contents.
91 *
92 * WHY: The atomic-rename strategy needs the whole copy built out of band before
93 * the final rename; isolating the build keeps webdav_copy_collection_execute a
94 * flat sequence and gives one place that reports the build-phase HTTP status.
95 *
96 * HOW: mkdir maps ENOENT-on-parent to 409 Conflict (any other failure to 500);
97 * on a recursion failure it removes the temp tree and returns 500. On success
98 * it returns NGX_OK with tmp_path filled for the commit phase.
99 */
100static ngx_int_t
101webdav_copy_collection_stage(const webdav_copy_job_t *job, char *tmp_path,
102 size_t tmp_path_size)
103{
104 if (brix_make_tmp_path(job->dst_path, tmp_path, tmp_path_size) != NGX_OK) {
105 return NGX_HTTP_REQUEST_URI_TOO_LARGE;
106 }
107
108 if (brix_vfs_mkdir_path(job->log, job->root_canon, tmp_path,
109 job->src_mode & 0777) != 0)
110 {
111 if (errno == ENOENT) {
112 return NGX_HTTP_CONFLICT;
113 }
114 return NGX_HTTP_INTERNAL_SERVER_ERROR;
115 }
116
117 webdav_dead_props_copy(job->log, job->src_path, tmp_path);
118 brix_ns_copy_fattrs(job->log, job->src_path, tmp_path);
119
120 if (job->depth_infinity
121 && webdav_copy_dir_recursive(job->log, job->root_canon, job->src_path,
122 tmp_path) != NGX_OK)
123 {
124 (void) webdav_delete_path_recursive(job->log, job->root_canon, tmp_path);
125 return NGX_HTTP_INTERNAL_SERVER_ERROR;
126 }
127
128 return NGX_OK;
129}

131/*
132 * WHAT: Commit a fully-staged temp tree onto the destination: drop a pre-existing
133 * destination directory (rename cannot replace a non-empty dir) then rename the
134 * temp tree into place.
135 *
136 * WHY: Splitting the commit from the stage keeps each phase single-purpose and
137 * confines the "remove temp on any failure" cleanup to one linear path.
138 *
139 * HOW: If the destination existed as a directory it is removed first; a failure
140 * there or in the rename removes the temp tree and returns 500. On success it
141 * returns 204 (replaced) or 201 (created) matching the prior end-of-function
142 * status.
143 */
144static ngx_int_t
145webdav_copy_collection_commit(const webdav_copy_job_t *job, const char *tmp_path)
146{
147 if (job->dst_existed && job->dst_was_dir) {
148 if (webdav_delete_path_recursive(job->log, job->root_canon,
149 job->dst_path) != NGX_OK)
150 {
151 (void) webdav_delete_path_recursive(job->log, job->root_canon,
152 tmp_path);
153 return NGX_HTTP_INTERNAL_SERVER_ERROR;
154 }
155 }
156
157 if (brix_rename_confined_canon(job->log, job->root_canon, tmp_path,
158 job->dst_path) != 0)
159 {
160 brix_log_safe_path(job->log, NGX_LOG_ERR, ngx_errno,
161 "brix_webdav COPY: dir rename failed: \"%s\"",
162 job->dst_path);
163 (void) webdav_delete_path_recursive(job->log, job->root_canon, tmp_path);
164 return NGX_HTTP_INTERNAL_SERVER_ERROR;
165 }
166
167 return job->dst_existed ? NGX_HTTP_NO_CONTENT : NGX_HTTP_CREATED;
168}

170/*
171 * Copy a collection (directory) atomically.
172 * Strategy: build the whole copy in a sibling temp directory, then rename it
173 * into place — so a concurrent reader never sees a half-copied tree
174 * (stage → commit). Every failure after the temp dir exists removes it, so no
175 * orphan is left. Returns 201/204 on success or an HTTP error status. May run
176 * on a worker thread (see webdav_copy_collection_thread), so it uses only its
177 * job parameters and thread-safe confined-FS helpers — no ngx_http_request_t
178 * access.
179 */
180static ngx_int_t
181webdav_copy_collection_execute(const webdav_copy_job_t *job)
182{
183 char tmp_path[WEBDAV_MAX_PATH];
184 ngx_int_t rc;
185
186 rc = webdav_copy_collection_stage(job, tmp_path, sizeof(tmp_path));
187 if (rc != NGX_OK) {
188 return rc;
189 }
190
191 return webdav_copy_collection_commit(job, tmp_path);
192}

209/*

215webdav_copy_collection_done(ngx_event_t *ev)
216{
217 ngx_thread_task_t *task = ev->data;

230/*
231 * WHAT: Resolve the WebDAV location's collection-copy thread pool, caching the
232 * lookup on conf->common.thread_pool for reuse.
233 *
234 * WHY: postconfig only wires the pool for server-level loc_conf; nested location
235 * blocks may still have a NULL pool, so it must be resolved lazily. Isolating
236 * the lookup keeps the offload path a flat sequence.
237 *
238 * HOW: Returns the cached pool if present; otherwise looks it up by the
239 * configured name (defaulting to "default"), caches a hit, and returns it (NULL
240 * when no pool is available, i.e. the caller must run synchronously).
241 */
242static ngx_thread_pool_t *
243webdav_copy_thread_pool(ngx_http_brix_webdav_loc_conf_t *conf)
244{
245 ngx_thread_pool_t *pool = conf->common.thread_pool;
246
247 if (pool == NULL) {
248 static ngx_str_t default_name = ngx_string("default");
249 ngx_str_t *pname = conf->common.thread_pool_name.len > 0
250 ? &conf->common.thread_pool_name : &default_name;
251 pool = ngx_thread_pool_get((ngx_cycle_t *) ngx_cycle, pname);
252 if (pool != NULL) {
253 conf->common.thread_pool = pool;
254 }
255 }
256
257 return pool;
258}

260/*
261 * Try to offload a collection COPY to the thread pool.
262 * Returns NGX_DONE if queued (request kept alive via r->main->count++),
263 * NGX_DECLINED if no thread pool is available (caller runs it synchronously),
264 * NGX_ERROR on allocation/post failure. The job's stack-buffer paths are
265 * copied into the task context so they remain valid after this returns.
266 */
267static ngx_int_t
268webdav_copy_collection_post_task(ngx_http_request_t *r,
269 ngx_http_brix_webdav_loc_conf_t *conf, const webdav_copy_job_t *job)
270{
271 ngx_thread_task_t *task;
272 webdav_copy_collection_task_t *t;
273 ngx_thread_pool_t *pool;
274
275 /*
276 * Under impersonation the per-worker broker socket is a single fd shared by
277 * the event-loop thread; a thread-pool task issuing confined ops would use
278 * it concurrently and corrupt the request/reply framing, wedging the whole
279 * worker's broker channel. The thread also lacks the per-worker principal.
280 * Force the synchronous (NGX_DECLINED) path so the recursive copy runs on
281 * the event-loop thread where the principal is set and the broker socket
282 * has exactly one user.
283 */
284 if (brix_imp_enabled()) {
285 return NGX_DECLINED;
286 }
287
288 pool = webdav_copy_thread_pool(conf);
289 if (pool == NULL) {
290 return NGX_DECLINED;
291 }
292
293 task = ngx_thread_task_alloc(r->pool, sizeof(webdav_copy_collection_task_t));
294 if (task == NULL) {
295 return NGX_ERROR;
296 }
297
298 t = task->ctx;
299 t->r = r;
300 t->job = *job;
301 t->job.log = r->connection->log;
302 t->http_status = NGX_HTTP_INTERNAL_SERVER_ERROR;
303
304 brix_task_bind(task, webdav_copy_collection_thread, webdav_copy_collection_done);
305 task->event.log = r->connection->log;
306
307 if (ngx_thread_task_post(pool, task) != NGX_OK) {
308 return NGX_ERROR;
309 }
310
311 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
312 "brix_webdav: offloaded collection COPY to thread pool");
313
314 r->main->count++;
315 return NGX_DONE;
316}

318/*
319 * WHAT: Execute a collection (directory) COPY for a resolved request — offload
320 * to the thread pool when possible, else run it inline.
321 *
322 * WHY: Isolates the S_ISDIR branch of the handler, keeping the depth:infinity
323 * offload / fall-through / inline-run decision in one place.
324 *
325 * HOW: Builds a webdav_copy_job_t from the resolved request. For depth:infinity
326 * it tries the thread pool: NGX_DONE means queued (returned as-is so the handler
327 * finishes), NGX_ERROR maps to 500, NGX_DECLINED (no pool) falls through. It
328 * then runs the copy inline; a non-201/204 status is returned as the failure,
329 * NGX_OK means success (the handler emits the final status). Depth:0 always runs
330 * inline (it only creates the top directory).
331 */
332ngx_int_t
333webdav_copy_do_collection(ngx_http_request_t *r,
334 ngx_http_brix_webdav_loc_conf_t *conf, const webdav_copy_req_t *req)
335{
336 webdav_copy_job_t job;
337 ngx_int_t rc;
338
339 webdav_copy_job_init(&job, r->connection->log, conf->common.root_canon,
340 req);
341
342 if (req->depth_infinity) {
343 rc = webdav_copy_collection_post_task(r, conf, &job);
344 if (rc == NGX_DONE) {
345 return NGX_DONE;
346 }
347 if (rc == NGX_ERROR) {
348 return NGX_HTTP_INTERNAL_SERVER_ERROR;
349 }
350 /* rc == NGX_DECLINED: no thread pool, fall through to run inline. */
351 }
352
353 rc = webdav_copy_collection_execute(&job);
354 if (rc != NGX_HTTP_CREATED && rc != NGX_HTTP_NO_CONTENT) {
355 return rc;
356 }
357
358 return NGX_OK;
359}
