5/*
6 * identity_attrs.c — VOMS-FQAN → xrdacc (vorg, role, group) attribute derivation.
7 *
8 * WHAT: Implements brix_identity_derive_attrs and its two per-field helpers
9 * (brix_identity_trim_ws, brix_identity_parse_fqan). Given the flat
10 * comma-separated VO/group CSV, it produces the three index-aligned
11 * attribute CSVs (acc_vorg_csv / acc_role_csv / acc_group_csv) the XrdAcc
12 * engine pairs positionally.
13 *
14 * WHY: Turning "/cms/Role=production/Capability=NULL" into vorg="cms",
15 * role="production", group="/cms" is the single most intricate piece of
16 * identity building — a self-contained substring parser with no
17 * dependencies on the setters/accessors in identity.c. Splitting it out
18 * keeps both files focused and under the 500-line file-size cap while
19 * leaving the parsing/attribute semantics untouched in one place.
20 *
21 * HOW: brix_identity_derive_attrs owns only tokenisation and index-aligned CSV
22 * assembly; each field is trimmed by brix_identity_trim_ws and decomposed
23 * by brix_identity_parse_fqan. Every output is a prefix/substring of the
24 * input, so an strlen(vo_csv)-sized buffer always suffices. identity.c
25 * reaches this via the brix_identity_derive_attrs declaration in
26 * identity_internal.h.
27 */

29/*
30 * WHAT: Strip leading and trailing ASCII spaces/tabs from the [*tok, *tok+*tl)
31 * slice in place, advancing *tok and shrinking *tl to the trimmed span.
32 * WHY: A VOMS FQAN / group CSV field may carry incidental whitespace around
33 * each comma-separated entry; the xrdacc views must store the bare token.
34 * HOW: Advance the start pointer past leading blanks, then pull the length in
35 * past trailing blanks — both guarded by a nonzero remaining length so an
36 * all-blank or empty field collapses to length 0.
37 */
38static void
39brix_identity_trim_ws(const char **tok, size_t *tl)
40{
41 const char *t = *tok;
42 size_t n = *tl;
43
44 while (n > 0 && (*t == ' ' || *t == '\t')) { t++; n--; }
45 while (n > 0 && (t[n - 1] == ' ' || t[n - 1] == '\t')) { n--; }
46
47 *tok = t;
48 *tl = n;
49}

51/*
52 * WHAT: Parse one trimmed FQAN/group token [tok, tok+tl) into three substring
53 * spans — group (*grp,*glen), role (*role,*rlen), vorg (*vorg,*vlen) —
54 * each pointing into the original token buffer.
55 * WHY: An FQAN like "/cms/Role=production/Capability=NULL" must decompose into
56 * vorg="cms", role="production", group="/cms"; "Role=NULL" and a plain
57 * group name ("cms") must yield an empty role. Centralising this keeps
58 * the CSV-assembly loop simple and the field semantics in one place.
59 * HOW: The group defaults to the whole token; a "/Role=" marker truncates the
60 * group there and captures the role up to the next '/' (dropping a literal
61 * "NULL"), else a "/Capability=" marker just truncates the trailing
62 * capability. The vorg is the group with any leading '/' removed and cut
63 * at its first interior '/'.
64 */
65static void
66brix_identity_parse_fqan(const char *tok, size_t tl,
67 const char **grp, size_t *glen,
68 const char **role, size_t *rlen,
69 const char **vorg, size_t *vlen)
70{
71 const char *rp, *cp, *vstart, *vend;
72 size_t g = tl, r = 0, v;
73
74 *role = NULL;
75 *grp = tok;
76
77 rp = memmem(tok, tl, "/Role=", 6);
78 if (rp != NULL) {
79 const char *rs = rp + 6, *re;
80 g = (size_t) (rp - tok); /* group = before /Role= */
81 re = memchr(rs, '/', (size_t) ((tok + tl) - rs));
82 if (re == NULL) { re = tok + tl; }
83 *role = rs;
84 r = (size_t) (re - rs);
85 if (r == 4 && ngx_strncmp(rs, "NULL", 4) == 0) { r = 0; }
86 } else if ((cp = memmem(tok, tl, "/Capability=", 12)) != NULL) {
87 g = (size_t) (cp - tok); /* strip trailing capability */
88 }
89
90 vstart = tok;
91 v = g;
92 if (v > 0 && *vstart == '/') { vstart++; v--; }
93 vend = memchr(vstart, '/', v);
94 if (vend != NULL) { v = (size_t) (vend - vstart); }
95
96 *glen = g;
97 *rlen = r;
98 *vorg = vstart;
99 *vlen = v;
100}

102/*

106 * strlen(vo_csv)-sized buffer always suffices. Per-field parsing is delegated
107 * to brix_identity_trim_ws + brix_identity_parse_fqan; this function owns only
108 * the tokenisation and index-aligned CSV assembly.
