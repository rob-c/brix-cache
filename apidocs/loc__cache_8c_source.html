33/* loc_hash — fnv1a over the NUL-terminated path (the design-of-record hash;
34 * cheap, decent avalanche for path-shaped keys). */
35static uint32_t
36loc_hash(const char *path)
37{
38 uint32_t h = BRIX_FNV1A32_OFFSET_BASIS;
39
40 while (*path != '\0') {
41 h ^= (uint32_t) (u_char) *path++;
42 h *= BRIX_FNV1A32_PRIME;
43 }
44 return h;
45}

84int
85brix_loc_cache_lookup(const char *path, char *host, size_t host_sz,
86 uint16_t *port)
87{
88 brix_loc_table_t *tbl;
89 brix_loc_entry_t *e;
90 uint32_t h;
91 ngx_uint_t i, idx;
92 ngx_msec_t now = ngx_current_msec;
93 int hit = 0;
94
95 tbl = loc_table();
96 if (tbl == NULL || path == NULL || path[0] == '\0'
97 || ngx_strlen(path) >= BRIX_LOC_CACHE_PATH_MAX)
98 {
99 return 0;
100 }
101
102 h = loc_hash(path);
103
104 ngx_shmtx_lock(&brix_loc_cache_mutex);
105
106 for (i = 0; i < BRIX_LOC_CACHE_SLOTS; i++) {
107 idx = (h + i) & (BRIX_LOC_CACHE_SLOTS - 1);
108 e = &tbl->slots[idx];
109
110 if (!e->in_use) {
111 break; /* probe chain ends at the first never-used slot */
112 }
113 if (brix_shm_slot_expired(now, e->expires)) {
114 continue; /* stale — invisible, but keep probing past it */
115 }
116 if (e->path_hash == h && ngx_strcmp(e->path, path) == 0) {
117 ngx_cpystrn((u_char *) host, (u_char *) e->host, host_sz);
118 *port = e->port;
119 hit = 1;
120 break;
121 }
122 }
123
124 ngx_shmtx_unlock(&brix_loc_cache_mutex);
125 return hit;
126}

128void
129brix_loc_cache_insert(const char *path, const char *host, uint16_t port)
130{
131 brix_loc_table_t *tbl;
132 brix_loc_entry_t *e;
133 uint32_t h;
134 ngx_uint_t i, idx, victim;
135 ngx_msec_t now = ngx_current_msec;
136
137 tbl = loc_table();
138 if (tbl == NULL || path == NULL || host == NULL
139 || ngx_strlen(path) >= BRIX_LOC_CACHE_PATH_MAX
140 || ngx_strlen(host) >= sizeof(e->host))
141 {
142 return;
143 }
144
145 h = loc_hash(path);
146 victim = BRIX_LOC_CACHE_SLOTS; /* sentinel: none found yet */
147
148 ngx_shmtx_lock(&brix_loc_cache_mutex);
149
150 for (i = 0; i < BRIX_LOC_CACHE_SLOTS; i++) {
151 idx = (h + i) & (BRIX_LOC_CACHE_SLOTS - 1);
152 e = &tbl->slots[idx];
153
154 if (e->in_use && !brix_shm_slot_expired(now, e->expires)
155 && (e->path_hash != h || ngx_strcmp(e->path, path) != 0))
156 {
157 continue; /* live entry for another path — keep probing */
158 }
159
160 /* Free, expired, or the same path: claim this slot. */
161 victim = idx;
162 break;
163 }
164
165 if (victim == BRIX_LOC_CACHE_SLOTS) {
166 /* Every slot is live for other paths: overwrite the home slot —
167 * bounded eviction (loc_cache.h contract). */
168 victim = h & (BRIX_LOC_CACHE_SLOTS - 1);
169 }
170
171 e = &tbl->slots[victim];
172 e->path_hash = h;
173 ngx_cpystrn((u_char *) e->path, (u_char *) path, sizeof(e->path));
174 ngx_cpystrn((u_char *) e->host, (u_char *) host, sizeof(e->host));
175 e->port = port;
176 e->expires = now + BRIX_LOC_CACHE_TTL_MS;
177 e->in_use = 1;
178
179 ngx_shmtx_unlock(&brix_loc_cache_mutex);
180}
