8/* Gather every endpoint fact doctor reports into *p: a normal (authenticated) connect
9 * for liveness/TLS/auth/caps, plus a separate no-login insecure probe for the server
10 * cert, plus a clock-skew measurement. Never fails hard — unreachable slices are left
11 * zeroed (p->connected / p->cert.have / p->clock_have signal availability). */
12void
13xrd_doctor_probe(const char *endpoint, xrd_probe *p)
14{
15 brix_url u;
16 brix_opts o;
17 brix_conn c;
18 brix_status st;
19 char errbuf[XRDC_MSG_MAX + 64]; /* room for a prefixed status msg */
20
21 memset(p, 0, sizeof(*p));
22 memset(&o, 0, sizeof(o));
23 o.verify_host = 1;
24 brix_status_clear(&st);
25 if (brix_endpoint_parse(endpoint, &u, &st) != 0) {
26 snprintf(p->err, sizeof(p->err), "%s", st.msg);
27 return;
28 }
29 snprintf(p->host, sizeof(p->host), "%s", u.host);
30 p->port = u.port;
31
32 /* (1) authenticated connect: liveness, role, negotiated TLS + auth, caps. */
33 if (brix_connect(&c, &u, &o, &st) != 0) {
34 snprintf(p->err, sizeof(p->err), "%s", st.msg);
35 } else {
36 const char *ver = NULL, *cipher = NULL;
37 p->connected = 1;
38 p->server_flags = c.server_flags;
39 snprintf(p->auth, sizeof(p->auth), "%s",
40 c.diag.chosen_auth ? c.diag.chosen_auth : "anonymous");
41 snprintf(p->sec_list, sizeof(p->sec_list), "%s", c.sec_list);
42 if (brix_tls_info(&c, &ver, &cipher)) {
43 p->tls_active = 1;
44 p->tls_ver = ver; p->tls_cipher = cipher;
45 }
46 xrd_probe_caps(&c, p);
47 brix_close(&c);
48 }
49
50 /* (2) no-login insecure probe for the server certificate (roots:// / gotoTLS). */
51 {
52 brix_conn cc;
53 brix_opts co;
54 brix_status cs;
55 memset(&co, 0, sizeof(co));
56 co.insecure_tls = 1; co.verify_host = 0;
57 brix_status_clear(&cs);
58 if (brix_connect_no_login(&cc, &u, &co, &cs) == 0) {
59 brix_tls_peer_cert_info(&cc, &p->cert);
60 brix_close(&cc);
61 }
62 }
63
64 /* (3) clock skew (HTTP Date for web URLs, touch+stat for root://). */
65 (void) xrd_measure_clock_skew(endpoint, &o, p, errbuf, sizeof(errbuf));
66}

69/* Emit a JSON string literal for `s` (escaped, NULL → ""). */
70void
71xrd_json_str(FILE *f, const char *s)
72{
73 fputc('"', f);
74 for (; s != NULL && *s != '\0'; s++) {
75 unsigned char ch = (unsigned char) *s;
76 if (ch == '"' || ch == '\\') { fputc('\\', f); fputc((int) ch, f); }
77 else if (ch == '\n') { fputs("\\n", f); }
78 else if (ch == '\t') { fputs("\\t", f); }
79 else if (ch == '\r') { fputs("\\r", f); }
80 else if (ch < 0x20) { fprintf(f, "\\u%04x", (unsigned) ch); }
81 else { fputc((int) ch, f); }
82 }
83 fputc('"', f);
84}

300/*
301 * WHAT: the full state of one `xrd doctor` run — parsed CLI options, discovered
302 * local credentials, functional-battery results, and the running fatal flag.
303 * WHY: the run was one 134-NLOC function (phase-72 H4); passing this struct
304 * lets each stage (parse / creds / batteries / render) be a small helper
305 * with explicit data flow instead of a dozen shared locals.
306 * HOW: filled top-down: doctor_parse_args → doctor_discover_creds →
307 * doctor_run_batteries; then either the JSON or text renderer walks the
308 * SAME results in the SAME fixed check order.
309 */
310typedef struct {
311 const char *endpoint; /* primary endpoint (or NULL) */
312 const char *also[XRD_DOCTOR_MAX_EP]; /* --also protocol faces */
313 int nalso;
314 int want_json; /* --json */
315 int do_rw; /* --rw / --write */
316 int verify; /* 0 after --insecure */
317 char *tok; /* discovered bearer token (owned) */
318 int token_present;
319 char pxp[1024]; /* default GSI proxy path */
320 int proxy_present;
321 xrd_battery bats[XRD_DOCTOR_MAX_EP]; /* functional battery results */
322 int nbats;
323 int fatal; /* accumulates → exit code */
324} xrd_doctor_run;

327/*
328 * WHAT: parse `xrd doctor` CLI flags into *d (endpoint, --also list, modes).
329 * WHY: stage 1 of the doctor pipeline, split from the old monolith (H4).
330 * HOW: same flag ladder as before: first non-flag arg is the endpoint,
331 * --also collects up to XRD_DOCTOR_MAX_EP-1 extra faces (excess URLs
332 * are consumed and dropped), unknown flags are ignored.
333 */
334static void
335doctor_parse_args(int argc, char **argv, xrd_doctor_run *d)
336{
337 int i;
338
339 for (i = 2; i < argc; i++) {
340 if (strcmp(argv[i], "--json") == 0) { d->want_json = 1; }
341 else if (strcmp(argv[i], "--rw") == 0
342 || strcmp(argv[i], "--write") == 0) { d->do_rw = 1; }
343 else if (strcmp(argv[i], "--insecure") == 0) { d->verify = 0; }
344 else if (strcmp(argv[i], "--also") == 0 && i + 1 < argc) {
345 if (d->nalso < XRD_DOCTOR_MAX_EP - 1) { d->also[d->nalso++] = argv[++i]; }
346 else { i++; }
347 }
348 else if (d->endpoint == NULL && argv[i][0] != '-') { d->endpoint = argv[i]; }
349 }
350}

354 * WHAT: discover local credentials (bearer token + default GSI proxy) into *d.
355 * WHY: stage 2 of the doctor pipeline; both renderers report these facts.
356 * HOW: token via the standard discovery chain (caller frees d->tok), proxy
357 * presence via an access(R_OK) probe of the conventional path.

359static void
360doctor_discover_creds(xrd_doctor_run *d)
361{
362 d->tok = brix_token_discover();
363 d->token_present = (d->tok != NULL);
364 brix_proxy_default_path(d->pxp, sizeof(d->pxp));
365 d->proxy_present = (access(d->pxp, R_OK) == 0);

369/*
370 * WHAT: run the functional method battery on the primary endpoint + each
371 * --also face; any failed check marks the run fatal.
372 * WHY: stage 3 of the doctor pipeline; results feed both output modes.
373 * HOW: batteries run in CLI order (primary first) so the report order —
374 * text and JSON alike — matches the invocation.
375 */
376static void
377doctor_run_batteries(xrd_doctor_run *d)
378{
379 int i;
380
381 if (d->endpoint != NULL) {
382 xrd_run_battery(d->endpoint, d->do_rw, d->verify, &d->bats[d->nbats++]);
383 }
384 for (i = 0; i < d->nalso; i++) {
385 xrd_run_battery(d->also[i], d->do_rw, d->verify, &d->bats[d->nbats++]);
386 }
387 for (i = 0; i < d->nbats; i++) {
388 if (d->bats[i].nfail > 0) { d->fatal = 1; }
389 }
390}

393/*
394 * WHAT: --json mode — probe the endpoint, emit the whole report as one JSON
395 * object, apply the fatal verdicts, and return the exit code.
396 * WHY: terminal stage of the JSON path, split from the old monolith (H4).
397 * HOW: same verdict rules as the text report: failed connect, expired /
398 * not-yet-valid cert, or >300 s clock skew are fatal.
399 */
400static int
401doctor_emit_json(xrd_doctor_run *d)
402{
403 xrd_probe p;
404 xrd_cred_facts cf;
405
406 memset(&p, 0, sizeof(p));
407 if (d->endpoint != NULL) { xrd_doctor_probe(d->endpoint, &p); }
408 cf.token_present = d->token_present;
409 cf.token_path = d->token_present ? "(discovered)" : "";
410 cf.proxy_present = d->proxy_present;
411 cf.proxy_path = d->pxp;
412 xrd_doctor_json(&p, &cf, d->bats, d->nbats);
413 if (d->tok != NULL) { free(d->tok); }
414 if (d->endpoint != NULL && !p.connected) { d->fatal = 1; }
415 if (p.cert.have && (p.cert.expired || p.cert.not_yet_valid)) { d->fatal = 1; }
416 if (p.clock_have && (p.offset_s > 300.0 || p.offset_s < -300.0)) { d->fatal = 1; }
417 return d->fatal ? 1 : 0;
418}

466 * WHY: per-check renderer mirroring doctor_json_cert on the same probe data.
467 * HOW: expired / not-yet-valid certs are fatal; a valid cert reports days

471doctor_report_cert(const xrd_probe *p, int *fatal)

497 * WHAT: text report — the clock-skew line of the endpoint section.

499 * HOW: >300 s skew is fatal, >60 s warns (breaks token exp/nbf + GSI windows);

563/*
564 * WHAT: text report — one "== <proto> tests: <url> ==" functional-battery
565 * section per probed protocol face.
566 * WHY: final text stage split from the old monolith (H4); walks the same
567 * battery results doctor_json_tests renders.
568 * HOW: unreachable faces report the error and move on; otherwise each check
569 * prints a PASS/FAIL/SKIP row followed by the pass-count summary line.
570 */
571static void
572doctor_report_batteries(const xrd_doctor_run *d)
573{
574 int i;
575
576 for (i = 0; i < d->nbats; i++) {
577 const xrd_battery *bt = &d->bats[i];
578 int j;
579 printf("== %s tests: %s ==\n", bt->protocol, bt->endpoint);
580 if (!bt->reachable) {
581 printf(" unreachable (%s)\n", bt->err[0] ? bt->err : "?");
582 continue;
583 }
584 for (j = 0; j < bt->n; j++) {
585 const xrd_check *ck = &bt->checks[j];

596/* `xrd doctor [endpoint] [--rw] [--also URL]... [--insecure] [--json]` — one-stop
597 * endpoint health: local credentials, and for each endpoint the connect/auth/TLS
598 * posture, server host-cert validity, clock skew, the kXR_Qconfig capability matrix,
599 * AND a functional method battery (read-only by default; --rw adds a full
600 * write/read/verify/checksum/metadata cycle). --also adds protocol faces (root:// /
601 * davs:// / s3://) so one run can cover every method on every protocol. --json emits
602 * the whole report as one object. Exit nonzero on a fatal local cred problem, a failed
603 * connect, an expired/skewed server, or any failed functional check. */
604int
605xrd_doctor(int argc, char **argv)
606{
607 xrd_doctor_run d;
608
609 memset(&d, 0, sizeof(d));
610 d.verify = 1;
611 doctor_parse_args(argc, argv, &d);
612 brix_crypto_init(); /* arm SHA-256/HMAC for token/proxy inspection */
613 doctor_discover_creds(&d);
614
615 /* Run the functional battery on the primary endpoint + each --also face. */
616 doctor_run_batteries(&d);
617
618 if (d.want_json) {
619 return doctor_emit_json(&d);
620 }
621
622 /* human report */
623 doctor_report_creds(&d);
624
625 if (d.endpoint == NULL && d.nbats == 0) {
626 printf("(pass an endpoint to also test connect + TLS + cert + clock + caps;\n"
627 " add --rw for write tests and --also <url> for more protocols)\n");
628 return d.fatal ? 1 : 0;
629 }
630
631 if (d.endpoint != NULL) {
632 doctor_report_endpoint(&d);
633 }
634
635 /* functional method batteries (per protocol face) */
636 doctor_report_batteries(&d);
637 return d.fatal ? 1 : 0;
638}

642void
643xrd_probe_caps(brix_conn *c, xrd_probe *p)
644{
645 int i;
646 p->ncaps = 0;
647 for (i = 0; XRD_CAP_KEYS[i] != NULL && p->ncaps < XRD_CAPS_MAX; i++) {
648 char reply[256], *nl, *eq;
649 const char *val;
650 brix_status st;
651 brix_status_clear(&st);
652 if (brix_query(c, kXR_Qconfig, XRD_CAP_KEYS[i], reply, sizeof(reply), &st) != 0) {
653 continue;
654 }
655 if ((nl = strchr(reply, '\n')) != NULL) { *nl = '\0'; }
656 eq = strchr(reply, '='); /* "key=val" → val; else whole reply */
657 val = (eq != NULL) ? eq + 1 : reply;
658 snprintf(p->caps[p->ncaps].key, sizeof(p->caps[p->ncaps].key), "%s",
659 XRD_CAP_KEYS[i]);
660 snprintf(p->caps[p->ncaps].val, sizeof(p->caps[p->ncaps].val), "%s", val);
661 p->ncaps++;
662 }
663}

666/* `xrd certinfo <endpoint>` — connect (requesting TLS) and report the server's host
667 * certificate: subject/issuer/SAN, validity window, days-until-expiry, host match.
668 * Exit nonzero if the cert is expired or not yet valid. */
669int
670xrd_certinfo(int argc, char **argv)
671{
672 const char *endpoint = (argc >= 3 && argv[2][0] != '-') ? argv[2] : NULL;
673 brix_url u;
674 brix_opts o;
675 brix_conn c;
676 brix_status st;
677 brix_cert_info ci;
678 char nb[32], na[32];
679
680 if (endpoint == NULL) { fprintf(stderr, "usage: xrd certinfo <endpoint>\n"); return 50; }
681 memset(&o, 0, sizeof(o));
682 /* Inspect, don't gate: skip chain + host verification so an expired/untrusted/
683 * self-signed cert is still reportable. TLS happens per the scheme (roots://,
684 * or a server that requires it); a cleartext root:// endpoint reports "no cert". */
685 o.insecure_tls = 1; o.verify_host = 0;
686 brix_crypto_init();
687 brix_status_clear(&st);
688 if (brix_endpoint_parse(endpoint, &u, &st) != 0) {
689 fprintf(stderr, "xrd certinfo: %s\n", st.msg); return 50;
690 }
691 if (brix_connect_no_login(&c, &u, &o, &st) != 0) {
692 fprintf(stderr, "xrd certinfo: connect %s:%d: %s\n", u.host, u.port, st.msg);
693 return brix_shellcode(&st);
694 }
695 if (brix_tls_peer_cert_info(&c, &ci) != 0 || !ci.have) {
696 printf("%s:%d — no server certificate (session is cleartext)\n", u.host, u.port);
697 brix_close(&c);
698 return 0;
699 }
700 xrd_fmt_epoch(ci.not_before, nb, sizeof(nb));
701 xrd_fmt_epoch(ci.not_after, na, sizeof(na));
702 printf("server certificate for %s:%d\n", u.host, u.port);
703 printf(" subject: %s\n", ci.subject);
704 printf(" issuer: %s\n", ci.issuer);
705 if (ci.sans[0] != '\0') { printf(" SAN: %s\n", ci.sans); }
706 printf(" validity: %s .. %s\n", nb, na);
707 if (ci.expired) {
708 printf(" status: EXPIRED %ld day(s) ago\n", -ci.days_left);
709 } else if (ci.not_yet_valid) {
710 printf(" status: NOT YET VALID\n");
711 } else {
712 printf(" status: valid, %ld day(s) left\n", ci.days_left);
713 }
714 printf(" host match: %s%s\n", ci.host_match ? "yes" : "no",
715 ci.self_signed ? " (self-signed)" : "");
716 brix_close(&c);
717 return (ci.expired || ci.not_yet_valid) ? 1 : 0;
718}
