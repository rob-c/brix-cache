8/* Gather every endpoint fact doctor reports into *p: a normal (authenticated) connect
9 * for liveness/TLS/auth/caps, plus a separate no-login insecure probe for the server
10 * cert, plus a clock-skew measurement. Never fails hard — unreachable slices are left
11 * zeroed (p->connected / p->cert.have / p->clock_have signal availability). */
12void
13xrd_doctor_probe(const char *endpoint, xrd_probe *p)
14{
15 brix_url u;
16 brix_opts o;
17 brix_conn c;
18 brix_status st;
19 char errbuf[XRDC_MSG_MAX + 64]; /* room for a prefixed status msg */
20
21 memset(p, 0, sizeof(*p));
22 memset(&o, 0, sizeof(o));
23 o.verify_host = 1;
24 brix_status_clear(&st);
25 if (brix_endpoint_parse(endpoint, &u, &st) != 0) {
26 snprintf(p->err, sizeof(p->err), "%s", st.msg);
27 return;
28 }
29 snprintf(p->host, sizeof(p->host), "%s", u.host);
30 p->port = u.port;
31
32 /* (1) authenticated connect: liveness, role, negotiated TLS + auth, caps. */
33 if (brix_connect(&c, &u, &o, &st) != 0) {
34 snprintf(p->err, sizeof(p->err), "%s", st.msg);
35 } else {
36 const char *ver = NULL, *cipher = NULL;
37 p->connected = 1;
38 p->server_flags = c.server_flags;
39 snprintf(p->auth, sizeof(p->auth), "%s",
40 c.diag.chosen_auth ? c.diag.chosen_auth : "anonymous");
41 snprintf(p->sec_list, sizeof(p->sec_list), "%s", c.sec_list);
42 if (brix_tls_info(&c, &ver, &cipher)) {
43 p->tls_active = 1;
44 p->tls_ver = ver; p->tls_cipher = cipher;
45 }
46 xrd_probe_caps(&c, p);
47 brix_close(&c);
48 }
49
50 /* (2) no-login insecure probe for the server certificate (roots:// / gotoTLS). */
51 {
52 brix_conn cc;
53 brix_opts co;
54 brix_status cs;
55 memset(&co, 0, sizeof(co));
56 co.insecure_tls = 1; co.verify_host = 0;
57 brix_status_clear(&cs);
58 if (brix_connect_no_login(&cc, &u, &co, &cs) == 0) {
59 brix_tls_peer_cert_info(&cc, &p->cert);
60 brix_close(&cc);
61 }
62 }
63
64 /* (3) clock skew (HTTP Date for web URLs, touch+stat for root://). */
65 (void) xrd_measure_clock_skew(endpoint, &o, p, errbuf, sizeof(errbuf));
66}

69/* Emit a JSON string literal for `s` (escaped, NULL → ""). */
70void
71xrd_json_str(FILE *f, const char *s)
72{
73 fputc('"', f);
74 for (; s != NULL && *s != '\0'; s++) {
75 unsigned char ch = (unsigned char) *s;
76 if (ch == '"' || ch == '\\') { fputc('\\', f); fputc((int) ch, f); }
77 else if (ch == '\n') { fputs("\\n", f); }
78 else if (ch == '\t') { fputs("\\t", f); }
79 else if (ch == '\r') { fputs("\\r", f); }
80 else if (ch < 0x20) { fprintf(f, "\\u%04x", (unsigned) ch); }
81 else { fputc((int) ch, f); }
82 }
83 fputc('"', f);
84}

192/* `xrd doctor [endpoint] [--rw] [--also URL]... [--insecure] [--json]` — one-stop
193 * endpoint health: local credentials, and for each endpoint the connect/auth/TLS
194 * posture, server host-cert validity, clock skew, the kXR_Qconfig capability matrix,
195 * AND a functional method battery (read-only by default; --rw adds a full
196 * write/read/verify/checksum/metadata cycle). --also adds protocol faces (root:// /
197 * davs:// / s3://) so one run can cover every method on every protocol. --json emits
198 * the whole report as one object. Exit nonzero on a fatal local cred problem, a failed
199 * connect, an expired/skewed server, or any failed functional check. */
200int
201xrd_doctor(int argc, char **argv)
202{
203 const char *endpoint = NULL;
204 const char *also[XRD_DOCTOR_MAX_EP];
205 int nalso = 0;
206 int want_json = 0, do_rw = 0, verify = 1, fatal = 0, i;
207 char *tok;
208 char pxp[1024];
209 int token_present, proxy_present;
210 xrd_battery bats[XRD_DOCTOR_MAX_EP];
211 int nbats = 0;
212
213 for (i = 2; i < argc; i++) {
214 if (strcmp(argv[i], "--json") == 0) { want_json = 1; }
215 else if (strcmp(argv[i], "--rw") == 0
216 || strcmp(argv[i], "--write") == 0) { do_rw = 1; }
217 else if (strcmp(argv[i], "--insecure") == 0) { verify = 0; }
218 else if (strcmp(argv[i], "--also") == 0 && i + 1 < argc) {
219 if (nalso < XRD_DOCTOR_MAX_EP - 1) { also[nalso++] = argv[++i]; }
220 else { i++; }
221 }
222 else if (endpoint == NULL && argv[i][0] != '-') { endpoint = argv[i]; }
223 }
224 brix_crypto_init(); /* arm SHA-256/HMAC for token/proxy inspection */
225
226 tok = brix_token_discover();
227 token_present = (tok != NULL);
228 brix_proxy_default_path(pxp, sizeof(pxp));
229 proxy_present = (access(pxp, R_OK) == 0);
230
231 /* Run the functional battery on the primary endpoint + each --also face. */
232 if (endpoint != NULL) {
233 xrd_run_battery(endpoint, do_rw, verify, &bats[nbats++]);
234 }
235 for (i = 0; i < nalso; i++) {
236 xrd_run_battery(also[i], do_rw, verify, &bats[nbats++]);
237 }
238 for (i = 0; i < nbats; i++) {
239 if (bats[i].nfail > 0) { fatal = 1; }
240 }
241
242 if (want_json) {
243 xrd_probe p;
244 memset(&p, 0, sizeof(p));
245 if (endpoint != NULL) { xrd_doctor_probe(endpoint, &p); }
246 xrd_doctor_json(&p, token_present, token_present ? "(discovered)" : "",
247 proxy_present, pxp, bats, nbats);
248 if (tok != NULL) { free(tok); }
249 if (endpoint != NULL && !p.connected) { fatal = 1; }
250 if (p.cert.have && (p.cert.expired || p.cert.not_yet_valid)) { fatal = 1; }
251 if (p.clock_have && (p.offset_s > 300.0 || p.offset_s < -300.0)) { fatal = 1; }
252 return fatal ? 1 : 0;
253 }
254
255 /* human report */
256 printf("== credentials ==\n");
257 if (tok != NULL) { brix_token_explain(tok, stdout); free(tok); }
258 else { printf(" bearer token: none discovered (BEARER_TOKEN / *_FILE / XDG / /tmp)\n"); }
259 if (proxy_present) { brix_gsi_cert_explain(pxp, stdout); }
260 else { printf(" GSI proxy: none at %s\n", pxp); }
261 if (brix_cred_diagnose(0, " hint: ", stdout)) { fatal = 1; }
262
263 if (endpoint == NULL && nbats == 0) {
264 printf("(pass an endpoint to also test connect + TLS + cert + clock + caps;\n"
265 " add --rw for write tests and --also <url> for more protocols)\n");
266 return fatal ? 1 : 0;
267 }
268
269 if (endpoint != NULL) {
270 xrd_probe p;
271 char nb[32], na[32];
272 int j;
273 xrd_doctor_probe(endpoint, &p);
274 printf("== endpoint %s:%d ==\n", p.host, p.port);
275 if (!p.connected) {
276 printf(" connect: FAILED (%s)\n", p.err[0] ? p.err : "?");
277 fatal = 1;
278 } else {
279 printf(" connect: OK role=%s auth=%s\n",
280 xrd_role_str(p.server_flags), p.auth);
281 if (p.sec_list[0] != '\0') { printf(" sec: %s\n", p.sec_list); }
282 if (p.tls_active) {
283 printf(" TLS: active (%s %s)\n",
284 p.tls_ver ? p.tls_ver : "?", p.tls_cipher ? p.tls_cipher : "?");
285 } else {
286 printf(" TLS: cleartext\n");
287 }
288 }
289 if (p.cert.have) {
290 xrd_fmt_epoch(p.cert.not_before, nb, sizeof(nb));
291 xrd_fmt_epoch(p.cert.not_after, na, sizeof(na));
292 printf(" cert: %s\n", p.cert.subject);
293 printf(" issuer %s\n", p.cert.issuer);
294 if (p.cert.expired) {
295 printf(" EXPIRED %ld day(s) ago (%s)\n", -p.cert.days_left, na);
296 fatal = 1;
297 } else if (p.cert.not_yet_valid) {
298 printf(" NOT YET VALID (from %s)\n", nb);
299 fatal = 1;
300 } else {
301 printf(" valid, %ld day(s) left (until %s) host-match=%s\n",
302 p.cert.days_left, na, p.cert.host_match ? "yes" : "no");
303 }
304 } else {
305 printf(" cert: none (cleartext session)\n");
306 }
307 if (p.clock_have) {
308 double ao = xrd_fabs(p.offset_s);
309 printf(" clock: offset %+.1f s, rtt %.1f ms (%s)\n",
310 p.offset_s, p.rtt_ms, p.clock_method);
311 if (ao > 300.0) { fatal = 1; }
312 if (ao > 60.0) {
313 printf(" WARNING: skew may break token exp/nbf + GSI validity\n");
314 }
315 } else {
316 printf(" clock: not measured (need an HTTP endpoint or write access)\n");
317 }
318 if (p.ncaps > 0) {
319 printf(" caps: ");
320 for (j = 0; j < p.ncaps; j++) {
321 printf(" %s=%s", p.caps[j].key, p.caps[j].val);
322 }
323 printf("\n");
324 }
325 }
326
327 /* functional method batteries (per protocol face) */
328 for (i = 0; i < nbats; i++) {
329 const xrd_battery *bt = &bats[i];
330 int j;
331 printf("== %s tests: %s ==\n", bt->protocol, bt->endpoint);
332 if (!bt->reachable) {
333 printf(" unreachable (%s)\n", bt->err[0] ? bt->err : "?");
334 continue;
335 }
336 for (j = 0; j < bt->n; j++) {
337 const xrd_check *ck = &bt->checks[j];
338 const char *tag = ck->skipped ? "SKIP" : (ck->ok ? "PASS" : "FAIL");
339 printf(" [%s] %-18s %s\n", tag, ck->name, ck->detail);
340 }
341 printf(" -> %d passed, %d failed, %d skipped%s\n",
342 bt->npass, bt->nfail, bt->nskip, do_rw ? "" : " (read-only; --rw for writes)");
343 }
344 return fatal ? 1 : 0;
345}

348/* Probe the kXR_Qconfig capability keys on a live connection into p->caps. */
349void
350xrd_probe_caps(brix_conn *c, xrd_probe *p)
351{
352 int i;
353 p->ncaps = 0;
354 for (i = 0; XRD_CAP_KEYS[i] != NULL && p->ncaps < XRD_CAPS_MAX; i++) {
355 char reply[256], *nl, *eq;
356 const char *val;
357 brix_status st;
358 brix_status_clear(&st);
359 if (brix_query(c, kXR_Qconfig, XRD_CAP_KEYS[i], reply, sizeof(reply), &st) != 0) {
360 continue;
361 }
362 if ((nl = strchr(reply, '\n')) != NULL) { *nl = '\0'; }
363 eq = strchr(reply, '='); /* "key=val" → val; else whole reply */
364 val = (eq != NULL) ? eq + 1 : reply;
365 snprintf(p->caps[p->ncaps].key, sizeof(p->caps[p->ncaps].key), "%s",
366 XRD_CAP_KEYS[i]);
367 snprintf(p->caps[p->ncaps].val, sizeof(p->caps[p->ncaps].val), "%s", val);
368 p->ncaps++;
369 }
370}

373/* `xrd certinfo <endpoint>` — connect (requesting TLS) and report the server's host
374 * certificate: subject/issuer/SAN, validity window, days-until-expiry, host match.
375 * Exit nonzero if the cert is expired or not yet valid. */
376int
377xrd_certinfo(int argc, char **argv)
378{
379 const char *endpoint = (argc >= 3 && argv[2][0] != '-') ? argv[2] : NULL;
380 brix_url u;
381 brix_opts o;
382 brix_conn c;
383 brix_status st;
384 brix_cert_info ci;
385 char nb[32], na[32];
386
387 if (endpoint == NULL) { fprintf(stderr, "usage: xrd certinfo <endpoint>\n"); return 50; }
388 memset(&o, 0, sizeof(o));
389 /* Inspect, don't gate: skip chain + host verification so an expired/untrusted/
390 * self-signed cert is still reportable. TLS happens per the scheme (roots://,
391 * or a server that requires it); a cleartext root:// endpoint reports "no cert". */
392 o.insecure_tls = 1; o.verify_host = 0;
393 brix_crypto_init();
394 brix_status_clear(&st);
395 if (brix_endpoint_parse(endpoint, &u, &st) != 0) {
396 fprintf(stderr, "xrd certinfo: %s\n", st.msg); return 50;
397 }
398 if (brix_connect_no_login(&c, &u, &o, &st) != 0) {
399 fprintf(stderr, "xrd certinfo: connect %s:%d: %s\n", u.host, u.port, st.msg);
400 return brix_shellcode(&st);
401 }
402 if (brix_tls_peer_cert_info(&c, &ci) != 0 || !ci.have) {
403 printf("%s:%d — no server certificate (session is cleartext)\n", u.host, u.port);
404 brix_close(&c);
405 return 0;
406 }
407 xrd_fmt_epoch(ci.not_before, nb, sizeof(nb));
408 xrd_fmt_epoch(ci.not_after, na, sizeof(na));
409 printf("server certificate for %s:%d\n", u.host, u.port);
410 printf(" subject: %s\n", ci.subject);
411 printf(" issuer: %s\n", ci.issuer);
412 if (ci.sans[0] != '\0') { printf(" SAN: %s\n", ci.sans); }
413 printf(" validity: %s .. %s\n", nb, na);
414 if (ci.expired) {
415 printf(" status: EXPIRED %ld day(s) ago\n", -ci.days_left);
416 } else if (ci.not_yet_valid) {
417 printf(" status: NOT YET VALID\n");
418 } else {
419 printf(" status: valid, %ld day(s) left\n", ci.days_left);
420 }
421 printf(" host match: %s%s\n", ci.host_match ? "yes" : "no",
422 ci.self_signed ? " (self-signed)" : "");
423 brix_close(&c);
424 return (ci.expired || ci.not_yet_valid) ? 1 : 0;
425}
