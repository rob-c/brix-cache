1/*
2 * sd_xroot.c — read-only remote root:// origin storage driver. See the header.
3 *
4 * Wraps the in-process XRootD origin wire client (cache/origin_*.c) behind the SD
5 * vtable. The per-open object holds a live origin connection + open file handle;
6 * pread issues a kXR_read range into a memory sink. Anonymous login only (the
7 * client's native mode); authenticated origins use the cache's native-client
8 * delegation. Only the read slots are populated → CAP_RANGE_READ.
9 */

24/* Per-open state: a live origin connection + open file handle + the synthetic
25 * fill-task the origin functions need (conf + clean_path + error scratch). */
26typedef struct {
27 brix_cache_origin_conn_t oc;
28 u_char fhandle[XRD_FHANDLE_LEN];
29 brix_cache_fill_t *t;
30 int file_open; /* 1 once kXR_open succeeded */
31 int is_write; /* 1 = opened for write (open_write) */
32} sd_xroot_obj_state;

34/* Map a fill-task XRootD error (kXR_*) to an errno-style fact for *err_out. */
35int
36sd_xroot_errno(const brix_cache_fill_t *t)
37{
38 switch (t->xrd_error) {
39 case kXR_NotFound: return ENOENT;
40 case kXR_NotAuthorized: return EACCES;
41 case kXR_isDirectory: return EISDIR;
42 default: return EIO;
43 }

60/* Open the origin file: connect → bootstrap (handshake + anon login) → kXR_open.
61 * `want_write` selects kXR_open(update+delete+mkpath) (a fresh writable handle,
62 * size 0) over the read open (read|retstat for the size). Returns the populated
63 * object state, or NULL with *err_out set. */
64static sd_xroot_obj_state *
65sd_xroot_origin_open(ngx_stream_brix_srv_conf_t *conf, const char *path,
66 int want_write, mode_t mode, off_t *size_out, int *err_out)
67{
68 sd_xroot_obj_state *st = calloc(1, sizeof(*st));
69 brix_cache_fill_t *t = calloc(1, sizeof(*t));
70
71 if (st == NULL || t == NULL) {
72 free(st);
73 free(t);
74 if (err_out) { *err_out = ENOMEM; }
75 return NULL;
76 }
77 st->t = t;
78 st->oc.fd = -1;
79 t->conf = conf;
80 ngx_cpystrn((u_char *) t->clean_path, (u_char *) path,
81 sizeof(t->clean_path));
82
83 if (brix_cache_origin_connect(t, &st->oc) != 0
84 || brix_cache_origin_bootstrap(t, &st->oc) != 0)
85 {
86 if (err_out) { *err_out = sd_xroot_errno(t); }
87 sd_xroot_obj_teardown(st);
88 return NULL;
89 }
90
91 if (want_write) {
92 uint16_t mode_bits = (uint16_t) ((mode != 0) ? (mode & 0777) : 0644);
93
94 if (brix_cache_origin_open_write(t, &st->oc, path, mode_bits,
95 st->fhandle) != 0)
96 {
97 if (err_out) { *err_out = sd_xroot_errno(t); }
98 sd_xroot_obj_teardown(st);
99 return NULL;
100 }
101 st->is_write = 1;
102 if (size_out) { *size_out = 0; } /* open_write truncates to empty */
103 } else {
104 if (brix_cache_origin_open(t, &st->oc, st->fhandle) != 0) {
105 if (err_out) { *err_out = sd_xroot_errno(t); }
106 sd_xroot_obj_teardown(st);
107 return NULL;
108 }
109 if (size_out) { *size_out = t->file_size; }
110 }
111 st->file_open = 1;
112 return st;
113}

161sd_xroot_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
162{
163 sd_xroot_obj_state *st = obj->state;
164 brix_cache_sink_t sink;
165 size_t got = 0;
166
167 if (len == 0) {
168 return 0;
169 }
170 ngx_memzero(&sink, sizeof(sink));
171 sink.fd = -1;
172 sink.mem = buf;
173 sink.mem_cap = len;
174
175 if (brix_cache_origin_read_chunk(st->t, &st->oc, st->fhandle, &sink,

178 errno = EIO;
179 return -1;
180 }
181 return (ssize_t) got;

273/* ---- staged atomic publish (Mode A passthrough) --------------------------- *
274 * The staged handle is a live origin write handle. With NO local staging store
275 * the write streams straight to the FINAL remote path (transparent write-through,
276 * no local copy), so it is NOT atomic — a failed upload can leave a partial object
277 * on the origin. Configure a local staging directory (Mode B) for atomic / durable
278 * publish (a later phase). The opaque brix_sd_staged_t (sd.h) carries our state
279 * in ->state. */
280
281typedef struct {
282 brix_cache_origin_conn_t oc;
283 u_char fhandle[XRD_FHANDLE_LEN];
284 brix_cache_fill_t *t;
285 int file_open;
286} sd_xroot_staged_state;

367/* Publish: sync + close the origin file. On success the handle is consumed (freed);
368 * on failure it stays valid for the caller to staged_abort. `noreplace` cannot be
369 * enforced on a Mode-A direct write (open_write already created the destination) —
370 * use a staging dir (Mode B) for exclusive/atomic publish. */
371static ngx_int_t
372sd_xroot_staged_commit(brix_sd_staged_t *handle, int noreplace)
373{
374 sd_xroot_staged_state *ss = handle->state;
375
376 (void) noreplace;
377
378 if (brix_cache_origin_sync(ss->t, &ss->oc, ss->fhandle) != 0) {
379 return NGX_ERROR; /* leave valid; caller aborts */
380 }
381 sd_xroot_staged_teardown(handle); /* close_file + free (consumed) */
382 return NGX_OK;
383}

393/* Vectored read: serve each iov segment from the origin (contiguous from `off`).
394 * Reuses sd_xroot_pread per segment — short read on any segment ends the read. */
395static ssize_t
396sd_xroot_preadv(brix_sd_obj_t *obj, const struct iovec *iov, int iovcnt,
397 off_t off)
398{
399 ssize_t total = 0;
400 int i;
401
402 for (i = 0; i < iovcnt; i++) {
403 ssize_t n = sd_xroot_pread(obj, iov[i].iov_base, iov[i].iov_len,
404 off + total);
405 if (n < 0) {
406 return (total > 0) ? total : -1;
407 }
408 total += n;
409 if ((size_t) n < iov[i].iov_len) {
410 break; /* short read = EOF */
411 }
412 }
413 return total;
414}

491brix_sd_instance_t *
492brix_sd_xroot_create_origin(const char *host, int port, int tls,
493 int af_policy, const char *bearer, const char *x509_proxy,
494 const char *ca_dir, const char *sss_keytab, ngx_log_t *log)
495{

497 sd_xroot_inst_state *is;
498 ngx_stream_brix_srv_conf_t *synth;
499
500 if (host == NULL || host[0] == '\0' || port <= 0 || port > 65535) {
501 errno = EINVAL;
502 return NULL;
503 }
504 inst = calloc(1, sizeof(*inst));
505 is = calloc(1, sizeof(*is));
506 synth = calloc(1, sizeof(*synth)); /* minimal: just the origin params */
507 if (inst == NULL || is == NULL || synth == NULL) {
508 free(inst);
509 free(is);

516 synth->cache_origin_host.len = ngx_strlen(is->host);
517 synth->cache_origin_port = (uint16_t) port;

519 synth->cache_origin_family = (ngx_uint_t) af_policy;
520

522 * Store the bytes on the instance so synth's ngx_str_t stays valid for life. */
523 if (bearer != NULL && bearer[0] != '\0') {
524 ngx_cpystrn((u_char *) is->bearer, (u_char *) bearer, sizeof(is->bearer));
525 synth->cache_origin_bearer.data = (u_char *) is->bearer;
526 synth->cache_origin_bearer.len = ngx_strlen(is->bearer);
527 }
528 if (x509_proxy != NULL && x509_proxy[0] != '\0') {
529 ngx_cpystrn((u_char *) is->x509_proxy, (u_char *) x509_proxy,
530 sizeof(is->x509_proxy));

532 synth->cache_origin_x509_proxy.len = ngx_strlen(is->x509_proxy);
533 }
534 /* §14 SSS: present a shared-secret credential minted from this keytab when the
535 * origin advertises &P=sss. Bytes on the instance so synth's str stays valid. */
536 if (sss_keytab != NULL && sss_keytab[0] != '\0') {
537 ngx_cpystrn((u_char *) is->sss_keytab, (u_char *) sss_keytab,
538 sizeof(is->sss_keytab));

540 synth->cache_origin_sss_keytab.len = ngx_strlen(is->sss_keytab);
541 }
542 if (ca_dir != NULL && ca_dir[0] != '\0') {
543 struct stat ca_st;
544 int is_dir;
545 int crl_count = 0;
546
547 ngx_cpystrn((u_char *) is->ca_dir, (u_char *) ca_dir, sizeof(is->ca_dir));
548 synth->cache_origin_ca_dir.data = (u_char *) is->ca_dir;

552 * file, with proxy certs allowed (GSI). Failure leaves gsi_store NULL ⇒ the

555 /* Outbound cache→origin client: keep the pre-existing trust behaviour
556 * (no signing_policy namespace enforcement, best-effort CRL) so this
557 * change does not alter which origins a cache node will talk to. */
558 synth->gsi_store = brix_build_ca_store(log,
559 is_dir ? is->ca_dir : NULL, is_dir ? NULL : is->ca_dir, NULL,
560 X509_V_FLAG_ALLOW_PROXY_CERTS, &crl_count,
561 BRIX_SP_MODE_OFF, BRIX_CRL_MODE_TRY);
562 if (synth->gsi_store == NULL) {
563 ngx_log_error(NGX_LOG_ERR, log, 0,

566 }
567 }

570 is->synth = synth; /* owned: free on destroy */
571 inst->driver = &brix_sd_xroot_driver;
572 inst->log = log;
573 inst->pool = NULL;
574 inst->state = is;
575 return inst;
576}
