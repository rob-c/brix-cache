1/*
2 * sd_xroot.c — remote root:// origin storage driver: core.
3 *
4 * Wraps the in-process XRootD origin wire client (cache/origin_*.c) behind the SD
5 * vtable. This translation unit owns the shared origin-open machinery (the
6 * per-open object state, the connect → bootstrap → kXR_open orchestrator, and the
7 * credential-copy / teardown helpers), the driver vtable, the origin-checksum
8 * query, and instance lifecycle (create / create_origin / destroy). The object
9 * I/O + open + stat slots live in sd_xroot_io.c, the staged atomic-publish slots
10 * in sd_xroot_staged.c, and the namespace + metadata slots in sd_xroot_ns.c
11 * (phase-79 file-size split); all share driver-private state via
12 * sd_xroot_internal.h. Anonymous login is the wire client's native mode;
13 * authenticated origins use the cache's native-client delegation or a per-user
14 * credential copied into the fill task.
15 */

30/* Map a fill-task XRootD error (kXR_*) to an errno-style fact for *err_out. */
31int
32sd_xroot_errno(const brix_cache_fill_t *t)
33{
34 switch (t->xrd_error) {
35 case kXR_NotFound: return ENOENT;
36 case kXR_NotAuthorized: return EACCES;
37 case kXR_isDirectory: return EISDIR;
38 default: return EIO;
39 }
40}

42/* sd_xroot_copy_cred_into_task — copy a per-user credential into the fill task.
43 *
44 * WHAT: When `cred` is non-NULL, copy its x509 proxy path, bearer token, and
45 * principal into the fill task's cred_* fields (each only when non-empty).
46 * No-op when cred is NULL.
47 * WHY: The origin bootstrap is the sole auth decision point; a task-level copy
48 * lets each open carry independent identity without touching the shared
49 * conf. Both the plain-open and staged-open paths need the identical copy,
50 * so it lives in one place to prevent divergence.
51 * HOW: 1) return early on NULL cred (the fill task was calloc-zeroed, so the
52 * bootstrap sees the service credential / anonymous). 2) Exactly one of
53 * {x509_proxy, bearer} is non-NULL for a credential-scoped open (mutually
54 * exclusive); copy whichever is present. 3) Copy principal when present. */
55void
56sd_xroot_copy_cred_into_task(brix_cache_fill_t *t, const brix_sd_cred_t *cred)
57{
58 if (cred == NULL) {
59 return;
60 }
61 if (cred->x509_proxy != NULL && cred->x509_proxy[0] != '\0') {
62 ngx_cpystrn((u_char *) t->cred_x509_proxy, (u_char *) cred->x509_proxy,
63 sizeof(t->cred_x509_proxy));
64 }
65 if (cred->bearer != NULL && cred->bearer[0] != '\0') {
66 ngx_cpystrn((u_char *) t->cred_bearer, (u_char *) cred->bearer,
67 sizeof(t->cred_bearer));
68 }
69 if (cred->principal != NULL) {
70 ngx_cpystrn((u_char *) t->cred_principal, (u_char *) cred->principal,
71 sizeof(t->cred_principal));
72 }
73}

91/* sd_xroot_origin_do_open — issue the kXR_open for a bootstrapped connection.
92 *
93 * WHAT: Perform the write- or read-open on an already connected+bootstrapped
94 * origin session, recording the fhandle and file size. Returns 0 on
95 * success (st->file_open set), or -1 with *req->err_out set on failure.
96 * WHY: Splitting the open branch off the connect/bootstrap sequence keeps
97 * sd_xroot_origin_open a flat orchestrator and isolates the write-vs-read
98 * distinction (mode bits, size source) in one place.
99 * HOW: 1) want_write → kXR_open(update) with mode bits, size 0 (truncated). 2)
100 * else → kXR_open(read|retstat), size from the returned stat. 3) on either
101 * failure map the fill-task kXR error to errno. */
102static int
103sd_xroot_origin_do_open(sd_xroot_obj_state *st,
104 const sd_xroot_origin_open_req_t *req)
105{
106 brix_cache_fill_t *t = st->t;
107
108 if (req->want_write) {
109 uint16_t mode_bits =
110 (uint16_t) ((req->mode != 0) ? (req->mode & 0777) : 0644);
111
112 if (brix_cache_origin_open_write(t, &st->oc, req->path, mode_bits,
113 st->fhandle) != 0)
114 {
115 if (req->err_out) { *req->err_out = sd_xroot_errno(t); }
116 return -1;
117 }
118 st->is_write = 1;
119 if (req->size_out) { *req->size_out = 0; } /* open_write truncates to empty */
120 } else {
121 if (brix_cache_origin_open(t, &st->oc, st->fhandle) != 0) {
122 if (req->err_out) { *req->err_out = sd_xroot_errno(t); }
123 return -1;
124 }
125 if (req->size_out) { *req->size_out = t->file_size; }
126 }
127 st->file_open = 1;
128 return 0;
129}

131/* Open the origin file: connect → bootstrap (handshake + anon or per-user login)
132 * → kXR_open. `req->want_write` selects kXR_open(update+delete+mkpath) (a fresh
133 * writable handle, size 0) over the read open (read|retstat for the size).
134 * `req->cred` carries a per-user x509 proxy path; when non-NULL and non-empty the
135 * bootstrap uses it instead of the conf's static service credential. Namespace
136 * callers (stat, sd_xroot_session) pass cred=NULL — they stay on the service
137 * credential in Phase 1. Returns the populated object state, or NULL with
138 * *req->err_out set.
139 *
140 * WHAT: Allocate and wire up the per-open fill task, copy the per-user proxy into
141 * it, then run connect → bootstrap → open as a flat sequence.
142 * WHY: The bootstrap is the sole auth decision point for every origin session; a
143 * task-level field lets each open carry independent identity without touching
144 * the shared conf or the connection pool.
145 * HOW: 1) calloc the state + fill task (zeroed cred fields ⇒ cred=NULL behaves
146 * like Phase-0). 2) copy the credential via the shared helper. 3) connect +
147 * bootstrap; on failure log the actual origin reason and map errno. 4)
148 * delegate the kXR_open branch to sd_xroot_origin_do_open. */
149sd_xroot_obj_state *
150sd_xroot_origin_open(const sd_xroot_origin_open_req_t *req)
151{
152 sd_xroot_obj_state *st = calloc(1, sizeof(*st));
153 brix_cache_fill_t *t = calloc(1, sizeof(*t));
154
155 if (st == NULL || t == NULL) {
156 free(st);
157 free(t);
158 if (req->err_out) { *req->err_out = ENOMEM; }
159 return NULL;
160 }
161 st->t = t;
162 st->oc.fd = -1;
163 t->conf = req->conf;
164 ngx_cpystrn((u_char *) t->clean_path, (u_char *) req->path,
165 sizeof(t->clean_path));
166
167 sd_xroot_copy_cred_into_task(t, req->cred);
168
169 if (brix_cache_origin_connect(t, &st->oc) != 0
170 || brix_cache_origin_bootstrap(t, &st->oc) != 0)
171 {
172 ngx_log_error(NGX_LOG_ERR, ngx_cycle->log, 0,
173 "brix sd_xroot: origin open \"%s\" failed: %s (kXR %d)",
174 req->path, (t->err_msg[0] != '\0') ? t->err_msg : "(no detail)",
175 t->xrd_error);
176 if (req->err_out) { *req->err_out = sd_xroot_errno(t); }
177 sd_xroot_obj_teardown(st);
178 return NULL;
179 }
180
181 if (sd_xroot_origin_do_open(st, req) != 0) {
182 sd_xroot_obj_teardown(st);
183 return NULL;
184 }
185 return st;
186}

326/* sd_xroot_origin_build_ca_store — build the GSI verify store for the origin.
327 *
328 * WHAT: When `ca_dir` is non-empty, record it on synth and build the per-worker
329 * X509 verify store (a hashed CA dir vs a bundle file) into synth->gsi_store;
330 * a build failure is logged and leaves gsi_store NULL. No-op on empty ca_dir.
331 * WHY: A NULL store makes the GSI handshake refuse the origin rather than trust
332 * an unverifiable one; isolating the stat-and-build keeps the create
333 * function flat.
334 * HOW: 1) copy ca_dir onto the instance and point synth at it. 2) stat to decide
335 * hashed-dir vs bundle-file. 3) call brix_build_ca_store with proxy certs
336 * allowed, signing-policy OFF and best-effort CRL (pre-existing outbound
337 * cache→origin trust behaviour — unchanged). 4) log on failure. */
338static void
339sd_xroot_origin_build_ca_store(sd_xroot_inst_state *is,
340 ngx_stream_brix_srv_conf_t *synth, const char *ca_dir, ngx_log_t *log)
341{
342 struct stat ca_st;
343 int is_dir;
344 int crl_count = 0;
345
346 if (ca_dir == NULL || ca_dir[0] == '\0') {
347 return;
348 }
349 ngx_cpystrn((u_char *) is->ca_dir, (u_char *) ca_dir, sizeof(is->ca_dir));
350 synth->cache_origin_ca_dir.data = (u_char *) is->ca_dir;
351 synth->cache_origin_ca_dir.len = ngx_strlen(is->ca_dir);
352 /* Also expose the origin CA as the generic trust anchor so the root:// TLS
353 * upgrade (brix_cache_origin_tls_upgrade → origin_tls_load_verify) verifies
354 * the origin's TLS cert against THIS store (CAfile or hashed CApath) instead
355 * of falling back to the system CAs. Same backing buffer as ca_dir. */
356 synth->trusted_ca.data = (u_char *) is->ca_dir;
357 synth->trusted_ca.len = ngx_strlen(is->ca_dir);
358
359 is_dir = (stat(is->ca_dir, &ca_st) == 0 && S_ISDIR(ca_st.st_mode));
360 synth->gsi_store = brix_build_ca_store(log,
361 is_dir ? is->ca_dir : NULL, is_dir ? NULL : is->ca_dir, NULL,
362 X509_V_FLAG_ALLOW_PROXY_CERTS, &crl_count,
363 BRIX_SP_MODE_OFF, BRIX_CRL_MODE_TRY);
364 if (synth->gsi_store == NULL) {
365 ngx_log_error(NGX_LOG_ERR, log, 0,
366 "brix: gsi origin CA store build failed for \"%s\" — GSI to this "
367 "origin will be refused", ca_dir);
368 }
369}

372brix_sd_xroot_create_origin(const brix_sd_xroot_origin_cfg_t *cfg,

374{

383 return NULL;
384 }
385 inst = calloc(1, sizeof(*inst));
386 is = calloc(1, sizeof(*is));
387 synth = calloc(1, sizeof(*synth)); /* minimal: just the origin params */
388 if (inst == NULL || is == NULL || synth == NULL) {
389 free(inst);
390 free(is);
391 free(synth);
392 errno = ENOMEM;
393 return NULL;
394 }
395 ngx_cpystrn((u_char *) is->host, (u_char *) cfg->host, sizeof(is->host));
396 synth->cache_origin_host.data = (u_char *) is->host;
397 synth->cache_origin_host.len = ngx_strlen(is->host);
398 synth->cache_origin_port = (uint16_t) cfg->port;
399 synth->cache_origin_tls = cfg->tls ? 1 : 0;
400 synth->cache_origin_family = (ngx_uint_t) cfg->af_policy;
401
402 /* Credentials + CA store: each stores bytes on the instance so synth's
403 * ngx_str_t members reference storage valid for the instance lifetime. */
404 sd_xroot_origin_apply_creds(is, synth, cfg);
405 sd_xroot_origin_build_ca_store(is, synth, cfg->ca_dir, log);
406
407 is->conf = synth;
408 is->synth = synth; /* owned: free on destroy */
409 inst->driver = &brix_sd_xroot_driver;
410 inst->log = log;
411 inst->pool = NULL;
412 inst->state = is;
413 return inst;
414}
