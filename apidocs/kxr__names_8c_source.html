106/*
107 * WHAT: look a code up in a kXR name table.
108 * WHY: shared by the three vocabulary name functions so each stays a
109 * one-line wrapper over its table (replaces three switch ladders).
110 * HOW: linear scan (tables are small and lookups are log-path only);
111 * returns the caller's fallback string when the code is absent.
112 */
113static const char *
114kxr_lookup(const kxr_name_entry_t *table, size_t n, int code,
115 const char *fallback)
116{
117 size_t i;
118
119 for (i = 0; i < n; i++) {
120 if (table[i].code == code) {
121 return table[i].name;
122 }
123 }
124
125 return fallback;
126}
127
128#define KXR_TABLE_N(t) (sizeof(t) / sizeof((t)[0]))
