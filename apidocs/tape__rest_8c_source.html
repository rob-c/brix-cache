1/*
2 * tape_rest.c — WLCG HTTP Tape REST API (Phase 35 / Phase 2).
3 *
4 * WHAT: Implements the standard WLCG Tape REST surface under /api/v1/ so FTS and
5 * gfal2 drive tape staging over davs:// against the durable stage request registry that
6 * root:// uses:
7 * POST /api/v1/stage submit a bulk stage request
8 * GET /api/v1/stage/{id} poll its status
9 * DELETE /api/v1/stage/{id} delete the request
10 * POST /api/v1/stage/{id}/cancel cancel it
11 * POST /api/v1/release[/{id}] release disk pins (alias: /unpin)
12 * POST /api/v1/archiveinfo synchronous locality (alias: /fileinfo)
13 * GET /api/v1/stage list active requests
14 *
15 * WHY: This is blocker B2's HTTP face. The durable store (Phase 0) and residency
16 * (Phase 1) already exist; this unit is the endpoint router + the WLCG JSON
17 * schema marshalling on top of the stage registry + residency seam. It mirrors macaroon_endpoint.c
18 * (POST body read → NGX_DONE, jansson build, send_json).
19 *
20 * HOW: every wire path is resolved + confined under root_canon BEFORE any frm_*
21 * call (INVARIANT 4), and authorised per-path (storage.stage for mutating
22 * verbs, storage.read for archiveinfo) with NO partial side effects — all paths
23 * are resolved+authorised first, then the queue is mutated, so a single 403
24 * never leaves orphan records. Anonymous → 401; cert-auth writes are gated on
25 * allow_write (INVARIANT 11). JSON is built with jansson (already a dep).
26 */

55/* Send a JSON body (clone of macaroon_endpoint.c send_json — kept local so we
56 * never extern a static across translation units). */
57static ngx_int_t
58tape_send_json(ngx_http_request_t *r, ngx_int_t status, const char *json)
59{
60 ngx_buf_t *b;
61 ngx_chain_t out;
62 u_char *buf;
63 size_t len = json ? ngx_strlen(json) : 0;
64
65 BRIX_PNALLOC_OR_RETURN(buf, r->pool, len ? len : 1, NGX_HTTP_INTERNAL_SERVER_ERROR);
66 if (len) {
67 ngx_memcpy(buf, json, len);
68 }
69
70 BRIX_PCALLOC_OR_RETURN(b, r->pool, sizeof(*b), NGX_HTTP_INTERNAL_SERVER_ERROR);
71 b->pos = buf;
72 b->last = buf + len;
73 b->memory = 1;
74 b->last_buf = 1;
75 out.buf = b;
76 out.next = NULL;
77
78 r->headers_out.status = status;
79 r->headers_out.content_length_n = (off_t) len;
80 ngx_str_set(&r->headers_out.content_type, "application/json");
81 r->headers_out.content_type_len = r->headers_out.content_type.len;
82
83 {
84 ngx_int_t rc = ngx_http_send_header(r);
85 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
86 return rc;
87 }
88 }
89 return ngx_http_output_filter(r, &out);
90}

92/* Build a {"detail":"..."} error body at `status`. Always returns `status`. */
93static ngx_int_t
94tape_error(ngx_http_request_t *r, ngx_int_t status, const char *detail)
95{
96 json_t *o = json_object();
97 char *s;
98 ngx_int_t rc;
99
100 if (o == NULL) {
101 return NGX_HTTP_INTERNAL_SERVER_ERROR;
102 }
103 json_object_set_new(o, "detail", json_string(detail ? detail : ""));
104 s = json_dumps(o, JSON_COMPACT);
105 json_decref(o);
106 if (s == NULL) {
107 return NGX_HTTP_INTERNAL_SERVER_ERROR;
108 }
109 rc = tape_send_json(r, status, s);
110 free(s);
111 return rc; /* NGX_OK once the body is sent (already set
112 * r->headers_out.status for the metrics) */
113}

135static void
136tape_mint_id(char *buf, size_t sz)
137{
138 static const char hex[] = "0123456789abcdef";
139 u_char rnd[16];
140 size_t i;
141
142 if (sz < TAPE_ID_LEN) {
143 if (sz) { buf[0] = '\0'; }
144 return;
145 }
146 if (RAND_bytes(rnd, (int) sizeof(rnd)) != 1) {
147 /* deterministic, non-crypto fallback — ids are opaque handles, not keys */
148 for (i = 0; i < sizeof(rnd); i++) {
149 rnd[i] = (u_char) ((ngx_pid << 3) ^ (i * 131));
150 }
151 }
152 for (i = 0; i < sizeof(rnd); i++) {
153 buf[i * 2] = hex[rnd[i] >> 4];
154 buf[i * 2 + 1] = hex[rnd[i] & 0xf];
155 }
156 buf[sizeof(rnd) * 2] = '\0';
157}

159/* Map a WLCG checksumType name to the stage-registry checksum enum (F5). */
160static brix_stage_cstype_t
161tape_cstype_from_name(const char *name)
162{
163 if (name == NULL) { return BRIX_STAGE_CS_NONE; }
164 if (ngx_strcasecmp((u_char *) name, (u_char *) "adler32") == 0)
165 { return BRIX_STAGE_CS_ADLER32; }
166 if (ngx_strcasecmp((u_char *) name, (u_char *) "md5") == 0)
167 { return BRIX_STAGE_CS_MD5; }
168 if (ngx_strcasecmp((u_char *) name, (u_char *) "crc32") == 0)
169 { return BRIX_STAGE_CS_CRC32; }
170 if (ngx_strcasecmp((u_char *) name, (u_char *) "sha1") == 0)
171 { return BRIX_STAGE_CS_SHA1; }
172 if (ngx_strcasecmp((u_char *) name, (u_char *) "sha256") == 0
173 || ngx_strcasecmp((u_char *) name, (u_char *) "sha2") == 0)
174 { return BRIX_STAGE_CS_SHA2; }
175 return BRIX_STAGE_CS_NONE;
176}

209/* Map the sd residency + nearline flag to the WLCG locality vocabulary. On a
210 * nearline (tape) export an online object is ONLINE_AND_NEARLINE (resident AND on
211 * the backend); a plain export online object is ONLINE. */
212static const char *
213tape_locality_name(brix_sd_residency_t state, int nearline)
214{
215 switch (state) {
216 case BRIX_SD_RES_ONLINE:
217 return nearline ? "ONLINE_AND_NEARLINE" : "ONLINE";
218 case BRIX_SD_RES_NEARLINE:
219 case BRIX_SD_RES_OFFLINE:
220 return "NEARLINE";
221 case BRIX_SD_RES_LOST:
222 return "LOST";
223 default:
224 return "NONE";
225 }
226}

228/*
229 * Resolve + authorise one logical path. need_write selects storage.stage vs
230 * storage.read scope and the cert-auth allow_write gate. On success fills
231 * abs[abssz] with the confined absolute path and returns NGX_OK; otherwise
232 * returns an NGX_HTTP_* status the caller must surface for the WHOLE request.
233 */
234static ngx_int_t
235tape_authz_path(ngx_http_request_t *r, ngx_http_brix_webdav_loc_conf_t *conf,
236 ngx_http_brix_webdav_req_ctx_t *ctx, const char *logical,
237 int need_write, char *abs, size_t abssz)
238{
239 ngx_int_t rc;
240
241 if (logical == NULL || logical[0] != '/') {
242 return NGX_HTTP_BAD_REQUEST;
243 }
244 rc = webdav_resolve_destination_path(r->connection->log, "tape",
245 conf->common.root_canon, logical,
246 abs, abssz);
247 if (rc != NGX_OK) {
248 return rc; /* 403 confine / 404 / 409 / 400 */
249 }
250 /* cert/anonymous-but-verified writes require the server to allow writes */
251 if (need_write && !ctx->token_auth && !conf->common.allow_write) {
252 return NGX_HTTP_FORBIDDEN;
253 }
254 /* token principals must carry the right scope for this path (a non-token or
255 * anonymous identity is allowed unconditionally — writes were already gated
256 * on allow_write above). */
257 if (ctx->identity != NULL
258 && brix_identity_check_token_scope(ctx->identity, logical, need_write)
259 != NGX_OK)
260 {
261 return NGX_HTTP_FORBIDDEN;
262 }
263 return NGX_OK;
264}

267/* POST /api/v1/stage*/
268static ngx_int_t
269tape_stage_post(ngx_http_request_t *r, ngx_http_brix_webdav_loc_conf_t *conf,
270 ngx_http_brix_webdav_req_ctx_t *ctx, json_t *root)
271{
272 json_t *files, *elem, *resp, *jfiles;
273 size_t i, n;
274 brix_stage_registry_t *q = tape_queue();
275 char id[TAPE_ID_LEN];
276 char **abs; /* resolved paths (pass 1) */
277 const char **logical;
278
279 if (q == NULL) {
280 return tape_error(r, NGX_HTTP_SERVICE_UNAVAILABLE,
281 "tape staging is not configured");
282 }
283 files = json_object_get(root, "files");
284 if (!json_is_array(files) || json_array_size(files) == 0) {
285 return tape_error(r, NGX_HTTP_BAD_REQUEST,
286 "body must contain a non-empty \"files\" array");
287 }
288 n = json_array_size(files);
289 if (n > TAPE_MAX_FILES) {
290 return tape_error(r, NGX_HTTP_BAD_REQUEST, "too many files");
291 }
292
293 abs = brix_palloc_array(r->pool, n, sizeof(*abs));
294 logical = brix_palloc_array(r->pool, n, sizeof(*logical));
295 if (abs == NULL || logical == NULL) {
296 return NGX_HTTP_INTERNAL_SERVER_ERROR;
297 }
298
299 /* Pass 1: resolve + authorise EVERY path before any queue mutation, so a
300 * single denial cannot leave orphan records (no partial side effects). */
301 for (i = 0; i < n; i++) {
302 json_t *p;
303 const char *lp;
304 char *buf;
305 ngx_int_t rc;
306
307 elem = json_array_get(files, i);
308 p = json_is_object(elem) ? json_object_get(elem, "path") : NULL;
309 lp = json_is_string(p) ? json_string_value(p) : NULL;
310 if (lp == NULL) {
311 return tape_error(r, NGX_HTTP_BAD_REQUEST,
312 "each file needs a string \"path\"");
313 }
314 BRIX_PNALLOC_OR_RETURN(buf, r->pool, TAPE_PATH_MAX, NGX_HTTP_INTERNAL_SERVER_ERROR);
315 rc = tape_authz_path(r, conf, ctx, lp, 1, buf, TAPE_PATH_MAX);
316 if (rc != NGX_OK) {
317 return tape_error(r, rc, "path not permitted");
318 }
319 abs[i] = buf;
320 logical[i] = lp;
321 }
322
323 /* Pass 2: enqueue. The bulk request id is opaque; Phase 0/1 tracks one lfn
324 * per FRM reqid, so we use the first file's durable reqid as the externally
325 * visible request id (full bulk grouping is deferred per §3.3). */
326 tape_mint_id(id, sizeof(id));
327 jfiles = json_array();
328 for (i = 0; i < n; i++) {
329 brix_stage_request_view_t v;
330 char reqid[BRIX_STAGE_REQID_LEN];
331 json_t *jf = json_object();
332 ngx_int_t rc;
333
334 ngx_memzero(&v, sizeof(v));
335 v.lfn = abs[i];
336 v.requester_dn = brix_identity_dn_cstr(ctx->identity);
337 /* F5: optional per-file integrity request — the stage worker verifies the
338 * recalled file against this checksum and fails the recall on mismatch. */
339 {
340 json_t *jcs = json_object_get(json_array_get(files, i),
341 "checksum");
342 json_t *jcst = json_object_get(json_array_get(files, i),
343 "checksumType");
344 if (json_is_string(jcs)) {
345 v.cs_value = json_string_value(jcs);
346 v.cs_type = tape_cstype_from_name(
347 json_is_string(jcst) ? json_string_value(jcst) : "adler32");
348 }
349 }
350 rc = brix_stage_request_add(q, &v, reqid, sizeof(reqid),
351 r->connection->log);
352 if (rc == NGX_OK || rc == NGX_DECLINED) {
353 if (i == 0) {
354 ngx_memcpy(id, reqid, ngx_min(sizeof(reqid), sizeof(id)));
355 id[sizeof(id) - 1] = '\0';
356 }
357 /* recall driving (former frm_stage_kick) → engine-integration step */
358 json_object_set_new(jf, "path", json_string(logical[i]));
359 json_object_set_new(jf, "state",
360 json_string(rc == NGX_DECLINED ? "STARTED" : "SUBMITTED"));
361 } else {
362 json_object_set_new(jf, "path", json_string(logical[i]));
363 json_object_set_new(jf, "error", json_string("could not enqueue"));
364 }
365 json_array_append_new(jfiles, jf);
366 }
367
368 resp = json_object();
369 json_object_set_new(resp, "requestId", json_string(id));
370 json_object_set_new(resp, "id", json_string(id));
371 json_object_set_new(resp, "files", jfiles);
372
373 {
374 char buf[64 + TAPE_ID_LEN];
375 ngx_snprintf((u_char *) buf, sizeof(buf), "%s%s%Z",
376 "/api/v1/stage/", id);
377 (void) brix_http_set_header(r, "Location", buf, NULL);
378 }
379 return tape_send_object(r, NGX_HTTP_CREATED, resp);
380}

383/* GET /api/v1/stage/{id}*/
384static ngx_int_t
385tape_stage_get(ngx_http_request_t *r, const char *id)
386{
387 brix_stage_registry_t *q = tape_queue();
388 brix_stage_request_t rec;
389 json_t *o, *jfiles, *jf;
390 brix_sd_residency_t res;
391 int nearline = 0;
392 int on_disk = 0;
393
394 if (q == NULL) {
395 return tape_error(r, NGX_HTTP_SERVICE_UNAVAILABLE, "not configured");
396 }
397 if (brix_stage_request_get(q, id, &rec, r->connection->log) != NGX_OK) {
398 return tape_error(r, NGX_HTTP_NOT_FOUND, "no such request");
399 }
400 if (tape_residency(r, rec.lfn, &res, &nearline) == NGX_OK) {
401 on_disk = (res == BRIX_SD_RES_ONLINE);
402 }
403
404 jf = json_object();
405 json_object_set_new(jf, "path", json_string(rec.lfn));
406 json_object_set_new(jf, "state", json_string(tape_state_name(rec.status)));
407 json_object_set_new(jf, "onDisk", json_boolean(on_disk));
408 if (rec.status == BRIX_STAGE_REQ_FAILED) {
409 json_object_set_new(jf, "error", json_string("stage failed"));
410 }
411 jfiles = json_array();
412 json_array_append_new(jfiles, jf);
413
414 o = json_object();
415 json_object_set_new(o, "id", json_string(id));
416 json_object_set_new(o, "createdAt", json_integer((json_int_t) rec.tod_added));
417 json_object_set_new(o, "files", jfiles);
418 return tape_send_object(r, NGX_HTTP_OK, o);
419}

471 r->header_only = 1;

473 return ngx_http_send_header(r);

502/* POST /api/v1/release (alias /unpin)*/
503static ngx_int_t
504tape_release(ngx_http_request_t *r, ngx_http_brix_webdav_loc_conf_t *conf,
505 ngx_http_brix_webdav_req_ctx_t *ctx, json_t *root)
506{
507 json_t *paths;
508 size_t i, n;
509 json_t *unpinned, *not_unpinned, *o;
510 char *abs;
511
512 paths = json_object_get(root, "paths");
513 if (!json_is_array(paths)) {
514 return tape_error(r, NGX_HTTP_BAD_REQUEST,
515 "body must contain a \"paths\" array");
516 }
517 n = json_array_size(paths);
518 BRIX_PNALLOC_OR_RETURN(abs, r->pool, TAPE_PATH_MAX, NGX_HTTP_INTERNAL_SERVER_ERROR);
519
520 /* authorise all paths first (no partial side effects) */
521 for (i = 0; i < n; i++) {
522 json_t *p = json_array_get(paths, i);
523 const char *lp = json_is_string(p) ? json_string_value(p) : NULL;
524 ngx_int_t rc;
525 if (lp == NULL) {
526 return tape_error(r, NGX_HTTP_BAD_REQUEST, "path must be a string");
527 }
528 rc = tape_authz_path(r, conf, ctx, lp, 1, abs, TAPE_PATH_MAX);
529 if (rc != NGX_OK) {
530 return tape_error(r, rc, "path not permitted");
531 }
532 }
533
534 unpinned = json_array();
535 not_unpinned = json_array();
536 for (i = 0; i < n; i++) {
537 const char *lp = json_string_value(json_array_get(paths, i));
538 if (tape_authz_path(r, conf, ctx, lp, 1, abs,
539 TAPE_PATH_MAX) == NGX_OK
540 && brix_stage_request_pin_release(tape_queue(), abs,
541 r->connection->log) == NGX_OK)
542 {
543 json_array_append_new(unpinned, json_string(lp));
544 } else {
545 json_array_append_new(not_unpinned, json_string(lp));
546 }
547 }
548 o = json_object();
549 json_object_set_new(o, "unpinnedFiles", unpinned);
550 json_object_set_new(o, "nonUnpinnedFiles", not_unpinned);
551 return tape_send_object(r, NGX_HTTP_OK, o);
552}

567 return tape_error(r, NGX_HTTP_BAD_REQUEST,
568 "body must contain a non-empty \"paths\" array");
569 }
570 n = json_array_size(paths);
571 BRIX_PNALLOC_OR_RETURN(abs, r->pool, TAPE_PATH_MAX, NGX_HTTP_INTERNAL_SERVER_ERROR);
572
573 /* read scope for all paths first (no partial disclosure on a later 403) */
574 for (i = 0; i < n; i++) {
575 json_t *p = json_array_get(paths, i);
576 const char *lp = json_is_string(p) ? json_string_value(p) : NULL;
577 ngx_int_t rc;
578 if (lp == NULL) {
579 return tape_error(r, NGX_HTTP_BAD_REQUEST, "path must be a string");
580 }
581 rc = tape_authz_path(r, conf, ctx, lp, 0, abs, TAPE_PATH_MAX);
582 if (rc != NGX_OK) {
583 return tape_error(r, rc, "path not permitted");
584 }
585 }
586
587 arr = json_array();
588 for (i = 0; i < n; i++) {
589 const char *lp = json_string_value(json_array_get(paths, i));
590 json_t *e = json_object();
591 brix_sd_residency_t res;
592 int nearline = 0;
593 ngx_int_t rc;
594
595 json_object_set_new(e, "path", json_string(lp));
596 if (tape_authz_path(r, conf, ctx, lp, 0, abs,
597 TAPE_PATH_MAX) != NGX_OK)
598 {
599 json_object_set_new(e, "error", json_string("denied"));
600 json_array_append_new(arr, e);
601 continue;
602 }
603 rc = tape_residency(r, abs, &res, &nearline);
604 if (rc == NGX_DECLINED || rc == NGX_ERROR) {
605 json_object_set_new(e, "exists", json_false());
606 json_object_set_new(e, "locality", json_string("NONE"));
607 } else {
608 const char *loc = tape_locality_name(res, nearline);
609 json_object_set_new(e, "exists", json_true());
610 json_object_set_new(e, "onDisk",
611 json_boolean(res == BRIX_SD_RES_ONLINE));
612 json_object_set_new(e, "onTape", json_boolean(nearline
613 || res == BRIX_SD_RES_NEARLINE
614 || res == BRIX_SD_RES_OFFLINE));
615 json_object_set_new(e, "locality", json_string(loc));
616 }
617 json_array_append_new(arr, e);
618 }
619 {
620 json_t *o = json_object();
621 json_object_set_new(o, "files", arr);
622 return tape_send_object(r, NGX_HTTP_OK, o);
623 }
624}

627/* routing*/
628/*
629 * Split the path after /api/v1/ into up to three NUL-terminated segments in a
630 * caller buffer. Returns the segment count. e.g. "stage/abc/cancel" → 3.
631 */
632static ngx_uint_t
633tape_split(ngx_http_request_t *r, char *buf, size_t bufsz, char *seg[3])
634{
635 size_t plen = sizeof(TAPE_API_PREFIX) - 1;
636 size_t tail, i;
637 ngx_uint_t nseg = 0;
638
639 seg[0] = seg[1] = seg[2] = NULL;
640 if (r->uri.len <= plen) {
641 return 0;
642 }
643 tail = r->uri.len - plen;
644 if (tail >= bufsz) {
645 tail = bufsz - 1;
646 }
647 ngx_memcpy(buf, r->uri.data + plen, tail);
648 buf[tail] = '\0';
649
650 for (i = 0; i < tail && nseg < 3; ) {
651 seg[nseg++] = &buf[i];
652 while (i < tail && buf[i] != '/') { i++; }
653 if (i < tail) { buf[i++] = '\0'; }
654 }
655 /* trim a trailing empty segment from a path ending in '/' */
656 while (nseg > 0 && seg[nseg - 1][0] == '\0') {
657 nseg--;
658 }
659 return nseg;
660}

662/* Parse the JSON body (already read) into a root object. Caller decrefs. */
663static json_t *
664tape_parse_body(ngx_http_request_t *r)
665{
666 u_char *body = NULL;
667 size_t blen = 0;
668 json_error_t err;
669

678/* POST dispatch (after the body has been read). */
679static ngx_int_t
680tape_dispatch_post(ngx_http_request_t *r)
681{
682 ngx_http_brix_webdav_loc_conf_t *conf =
683 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
684 ngx_http_brix_webdav_req_ctx_t *ctx =
685 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
686 char sbuf[TAPE_PATH_MAX];
687 char *seg[3];
688 ngx_uint_t nseg;
689 json_t *root;
690 ngx_int_t rc;
691
692 nseg = tape_split(r, sbuf, sizeof(sbuf), seg);
693 if (nseg == 0) {
694 return tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
695 }
696
697 /* POST /stage/{id}/cancel acts on the URL id and consumes NO request body —
698 * route it before the mandatory body parse so the body-less cancel that real
699 * FTS/gfal2 clients send is not rejected with 400. */
700 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 3
701 && ngx_strcmp(seg[2], "cancel") == 0) {
702 return tape_stage_cancel(r, ctx, seg[1]);
703 }
704
705 root = tape_parse_body(r);
706 if (root == NULL) {
707 return tape_error(r, NGX_HTTP_BAD_REQUEST, "invalid or empty JSON body");
708 }
709
710 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 1) {
711 rc = tape_stage_post(r, conf, ctx, root);
712 } else if ((ngx_strcmp(seg[0], "release") == 0
713 || ngx_strcmp(seg[0], "unpin") == 0)) {
714 rc = tape_release(r, conf, ctx, root);
715 } else if (ngx_strcmp(seg[0], "archiveinfo") == 0
716 || ngx_strcmp(seg[0], "fileinfo") == 0) {
717 rc = tape_archiveinfo(r, conf, ctx, root);
718 } else {
719 rc = tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
720 }
721
722 json_decref(root);
723 return rc;
724}

734ngx_int_t
735webdav_tape_handle(ngx_http_request_t *r)
736{
737 ngx_http_brix_webdav_loc_conf_t *conf =
738 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
739 ngx_http_brix_webdav_req_ctx_t *ctx =
740 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
741 char sbuf[TAPE_PATH_MAX];
742 char *seg[3];
743 ngx_uint_t nseg;
744
745 if (ctx == NULL) {
746 return tape_error(r, NGX_HTTP_INTERNAL_SERVER_ERROR, "no auth context");
747 }
748 /* On a server that requires auth, an unauthenticated caller is rejected.
749 * On an auth=none / auth=optional server, anonymous is allowed through to
750 * the per-verb checks below (writes still gated on allow_write; token
751 * scopes still enforced when a token is present). */
752 if (!ctx->verified && conf->auth == WEBDAV_AUTH_REQUIRED) {
753 return tape_error(r, NGX_HTTP_UNAUTHORIZED, "authentication required");
754 }
755
756 if (r->method == NGX_HTTP_POST) {
757 ngx_int_t rc = brix_http_read_body(r, tape_post_body_handler);
758 return (rc == NGX_DONE) ? NGX_DONE : rc;
759 }
760
761 nseg = tape_split(r, sbuf, sizeof(sbuf), seg);
762 if (nseg == 0) {
763 return tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
764 }
765
766 if (r->method == NGX_HTTP_GET) {
767 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 2) {
768 return tape_stage_get(r, seg[1]);
769 }
770 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 1) {
771 return tape_stage_list(r);
772 }
773 return tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
774 }
775
776 if (r->method == NGX_HTTP_DELETE) {
777 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 2) {
778 return tape_stage_delete(r, ctx, seg[1]);
779 }
780 return tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
781 }
782
783 return tape_error(r, NGX_HTTP_NOT_ALLOWED, "method not allowed");
784}
