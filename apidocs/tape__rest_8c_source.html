1/*
2 * tape_rest.c — WLCG HTTP Tape REST API router (Phase 35 / Phase 2).
3 *
4 * WHAT: The endpoint router + response marshalling + resolve/authz/residency
5 * helpers for the standard WLCG Tape REST surface under /api/v1/ so FTS and
6 * gfal2 drive tape staging over davs:// against the durable stage request registry that
7 * root:// uses:
8 * POST /api/v1/stage submit a bulk stage request
9 * GET /api/v1/stage/{id} poll its status
10 * DELETE /api/v1/stage/{id} delete the request
11 * POST /api/v1/stage/{id}/cancel cancel it
12 * POST /api/v1/release[/{id}] release disk pins (alias: /unpin)
13 * POST /api/v1/archiveinfo synchronous locality (alias: /fileinfo)
14 * GET /api/v1/stage list active requests
15 *
16 * WHY: This is blocker B2's HTTP face. The durable store (Phase 0) and residency
17 * (Phase 1) already exist; this unit is the endpoint router + the WLCG JSON
18 * schema marshalling on top of the stage registry + residency seam. It mirrors macaroon_endpoint.c
19 * (POST body read → NGX_DONE, jansson build, send_json). The per-endpoint bodies
20 * live in tape_rest_ops.c (phase-79 file-size split); the shared seam is
21 * tape_rest_internal.h.
22 *
23 * HOW: every wire path is resolved + confined under root_canon BEFORE any frm_*
24 * call (INVARIANT 4), and authorised per-path (storage.stage for mutating
25 * verbs, storage.read for archiveinfo) with NO partial side effects — all paths
26 * are resolved+authorised first, then the queue is mutated, so a single 403
27 * never leaves orphan records. Anonymous → 401; cert-auth writes are gated on
28 * allow_write (INVARIANT 11). JSON is built with jansson (already a dep).
29 */

54/* Send a JSON body (clone of macaroon_endpoint.c send_json — kept local so we
55 * never extern a static across translation units). */
56static ngx_int_t
57tape_send_json(ngx_http_request_t *r, ngx_int_t status, const char *json)
58{
59 ngx_buf_t *b;
60 ngx_chain_t out;
61 u_char *buf;
62 size_t len = json ? ngx_strlen(json) : 0;
63
64 BRIX_PNALLOC_OR_RETURN(buf, r->pool, len ? len : 1, NGX_HTTP_INTERNAL_SERVER_ERROR);
65 if (len) {
66 ngx_memcpy(buf, json, len);
67 }
68
69 BRIX_PCALLOC_OR_RETURN(b, r->pool, sizeof(*b), NGX_HTTP_INTERNAL_SERVER_ERROR);
70 b->pos = buf;
71 b->last = buf + len;
72 b->memory = 1;
73 b->last_buf = 1;
74 out.buf = b;
75 out.next = NULL;
76
77 r->headers_out.status = status;
78 r->headers_out.content_length_n = (off_t) len;
79 ngx_str_set(&r->headers_out.content_type, "application/json");
80 r->headers_out.content_type_len = r->headers_out.content_type.len;
81
82 {
83 ngx_int_t rc = ngx_http_send_header(r);
84 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
85 return rc;
86 }
87 }
88 return ngx_http_output_filter(r, &out);
89}

91/* Build a {"detail":"..."} error body at `status`. Always returns `status`. */
92ngx_int_t
93tape_error(ngx_http_request_t *r, ngx_int_t status, const char *detail)
94{
95 json_t *o = json_object();
96 char *s;
97 ngx_int_t rc;
98
99 if (o == NULL) {
100 return NGX_HTTP_INTERNAL_SERVER_ERROR;
101 }
102 json_object_set_new(o, "detail", json_string(detail ? detail : ""));
103 s = json_dumps(o, JSON_COMPACT);
104 json_decref(o);
105 if (s == NULL) {
106 return NGX_HTTP_INTERNAL_SERVER_ERROR;
107 }
108 rc = tape_send_json(r, status, s);
109 free(s);
110 return rc; /* NGX_OK once the body is sent (already set
111 * r->headers_out.status for the metrics) */
112}

134void
135tape_mint_id(char *buf, size_t sz)
136{
137 static const char hex[] = "0123456789abcdef";
138 u_char rnd[16];
139 size_t i;
140
141 if (sz < TAPE_ID_LEN) {
142 if (sz) { buf[0] = '\0'; }
143 return;
144 }
145 if (RAND_bytes(rnd, (int) sizeof(rnd)) != 1) {
146 /* deterministic, non-crypto fallback — ids are opaque handles, not keys */
147 for (i = 0; i < sizeof(rnd); i++) {
148 rnd[i] = (u_char) ((ngx_pid << 3) ^ (i * 131));
149 }
150 }
151 for (i = 0; i < sizeof(rnd); i++) {
152 buf[i * 2] = hex[rnd[i] >> 4];
153 buf[i * 2 + 1] = hex[rnd[i] & 0xf];
154 }
155 buf[sizeof(rnd) * 2] = '\0';
156}

158/* Map a WLCG checksumType name to the stage-registry checksum enum (F5). */
159brix_stage_cstype_t
160tape_cstype_from_name(const char *name)
161{
162 if (name == NULL) { return BRIX_STAGE_CS_NONE; }
163 if (ngx_strcasecmp((u_char *) name, (u_char *) "adler32") == 0)
164 { return BRIX_STAGE_CS_ADLER32; }
165 if (ngx_strcasecmp((u_char *) name, (u_char *) "md5") == 0)
166 { return BRIX_STAGE_CS_MD5; }
167 if (ngx_strcasecmp((u_char *) name, (u_char *) "crc32") == 0)
168 { return BRIX_STAGE_CS_CRC32; }
169 if (ngx_strcasecmp((u_char *) name, (u_char *) "sha1") == 0)
170 { return BRIX_STAGE_CS_SHA1; }
171 if (ngx_strcasecmp((u_char *) name, (u_char *) "sha256") == 0
172 || ngx_strcasecmp((u_char *) name, (u_char *) "sha2") == 0)
173 { return BRIX_STAGE_CS_SHA2; }
174 return BRIX_STAGE_CS_NONE;
175}

227/*
228 * Resolve + authorise one logical path. need_write selects storage.stage vs
229 * storage.read scope and the cert-auth allow_write gate. On success fills
230 * abs[abssz] with the confined absolute path and returns NGX_OK; otherwise
231 * returns an NGX_HTTP_* status the caller must surface for the WHOLE request.
232 */
233ngx_int_t
234tape_authz_path(ngx_http_request_t *r, ngx_http_brix_webdav_loc_conf_t *conf,
235 ngx_http_brix_webdav_req_ctx_t *ctx, const char *logical,
236 int need_write, char *abs, size_t abssz)
237{
238 ngx_int_t rc;
239
240 if (logical == NULL || logical[0] != '/') {
241 return NGX_HTTP_BAD_REQUEST;
242 }
243 rc = webdav_resolve_destination_path(r->connection->log, "tape",
244 conf->common.root_canon, logical,
245 abs, abssz,
246 conf->common.cache_store_endpoint);
247 if (rc != NGX_OK) {
248 return rc; /* 403 confine / 404 / 409 / 400 */
249 }
250 /* cert/anonymous-but-verified writes require the server to allow writes */
251 if (need_write && !ctx->token_auth && !conf->common.allow_write) {
252 return NGX_HTTP_FORBIDDEN;
253 }
254 /* token principals must carry the right scope for this path (a non-token or
255 * anonymous identity is allowed unconditionally — writes were already gated
256 * on allow_write above). */
257 if (ctx->identity != NULL
258 && brix_identity_check_token_scope(ctx->identity, logical, need_write)
259 != NGX_OK)
260 {
261 return NGX_HTTP_FORBIDDEN;
262 }
263 return NGX_OK;
264}

268/*
269 * Split the path after /api/v1/ into up to three NUL-terminated segments in a
270 * caller buffer. Returns the segment count. e.g. "stage/abc/cancel" → 3.
271 */
272static ngx_uint_t
273tape_split(ngx_http_request_t *r, char *buf, size_t bufsz, char *seg[3])
274{
275 size_t plen = sizeof(TAPE_API_PREFIX) - 1;
276 size_t tail, i;
277 ngx_uint_t nseg = 0;
278
279 seg[0] = seg[1] = seg[2] = NULL;
280 if (r->uri.len <= plen) {
281 return 0;
282 }
283 tail = r->uri.len - plen;
284 if (tail >= bufsz) {
285 tail = bufsz - 1;
286 }
287 ngx_memcpy(buf, r->uri.data + plen, tail);
288 buf[tail] = '\0';
289
290 for (i = 0; i < tail && nseg < 3; ) {
291 seg[nseg++] = &buf[i];
292 while (i < tail && buf[i] != '/') { i++; }
293 if (i < tail) { buf[i++] = '\0'; }
294 }
295 /* trim a trailing empty segment from a path ending in '/' */
296 while (nseg > 0 && seg[nseg - 1][0] == '\0') {
297 nseg--;
298 }
299 return nseg;
300}

318/* POST dispatch (after the body has been read). */
319static ngx_int_t
320tape_dispatch_post(ngx_http_request_t *r)
321{
322 ngx_http_brix_webdav_loc_conf_t *conf =
323 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
324 ngx_http_brix_webdav_req_ctx_t *ctx =
325 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
326 char sbuf[TAPE_PATH_MAX];
327 char *seg[3];
328 ngx_uint_t nseg;
329 json_t *root;
330 ngx_int_t rc;
331
332 nseg = tape_split(r, sbuf, sizeof(sbuf), seg);
333 if (nseg == 0) {
334 return tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
335 }
336
337 /* POST /stage/{id}/cancel acts on the URL id and consumes NO request body —
338 * route it before the mandatory body parse so the body-less cancel that real
339 * FTS/gfal2 clients send is not rejected with 400. */
340 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 3
341 && ngx_strcmp(seg[2], "cancel") == 0) {
342 return tape_stage_cancel(r, ctx, seg[1]);
343 }
344
345 root = tape_parse_body(r);
346 if (root == NULL) {
347 return tape_error(r, NGX_HTTP_BAD_REQUEST, "invalid or empty JSON body");
348 }
349
350 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 1) {
351 rc = tape_stage_post(r, conf, ctx, root);
352 } else if ((ngx_strcmp(seg[0], "release") == 0
353 || ngx_strcmp(seg[0], "unpin") == 0)) {
354 rc = tape_release(r, conf, ctx, root);
355 } else if (ngx_strcmp(seg[0], "archiveinfo") == 0
356 || ngx_strcmp(seg[0], "fileinfo") == 0) {
357 rc = tape_archiveinfo(r, conf, ctx, root);
358 } else {
359 rc = tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
360 }
361
362 json_decref(root);
363 return rc;
364}

374ngx_int_t
375webdav_tape_handle(ngx_http_request_t *r)
376{
377 ngx_http_brix_webdav_loc_conf_t *conf =
378 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
379 ngx_http_brix_webdav_req_ctx_t *ctx =
380 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
381 char sbuf[TAPE_PATH_MAX];
382 char *seg[3];
383 ngx_uint_t nseg;
384
385 if (ctx == NULL) {
386 return tape_error(r, NGX_HTTP_INTERNAL_SERVER_ERROR, "no auth context");
387 }
388 /* On a server that requires auth, an unauthenticated caller is rejected.
389 * On an auth=none / auth=optional server, anonymous is allowed through to
390 * the per-verb checks below (writes still gated on allow_write; token
391 * scopes still enforced when a token is present). */
392 if (!ctx->verified && conf->auth == WEBDAV_AUTH_REQUIRED) {
393 return tape_error(r, NGX_HTTP_UNAUTHORIZED, "authentication required");
394 }
395
396 if (r->method == NGX_HTTP_POST) {
397 ngx_int_t rc = brix_http_read_body(r, tape_post_body_handler);
398 return (rc == NGX_DONE) ? NGX_DONE : rc;
399 }
400
401 nseg = tape_split(r, sbuf, sizeof(sbuf), seg);
402 if (nseg == 0) {
403 return tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
404 }
405
406 if (r->method == NGX_HTTP_GET) {
407 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 2) {
408 return tape_stage_get(r, seg[1]);
409 }
410 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 1) {
411 return tape_stage_list(r);
412 }
413 return tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
414 }
415
416 if (r->method == NGX_HTTP_DELETE) {
417 if (ngx_strcmp(seg[0], "stage") == 0 && nseg == 2) {
418 return tape_stage_delete(r, ctx, seg[1]);
419 }
420 return tape_error(r, NGX_HTTP_NOT_FOUND, "unknown endpoint");
421 }
422
423 return tape_error(r, NGX_HTTP_NOT_ALLOWED, "method not allowed");
424}
