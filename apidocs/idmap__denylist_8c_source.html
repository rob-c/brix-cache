1/*
2 * idmap_denylist.c — local-user resolution + forbidden-id / squash policy.
3 *
4 * WHAT: Resolves a local username to {uid, gid, supplementary gids}, and enforces
5 * the safety floor + deny-lists (forbidden accounts/groups) that decide whether
6 * a resolved credential set may be impersonated. Split out of idmap.c (file-
7 * size cap).
8 *
9 * WHY: The numeric-id safety machinery (reserved-id floor, forbidden uid/gid
10 * sets, group-overflow fail-closed handling) is one cohesive defence-in-depth
11 * concern, separate from the grid-mapfile parsing and the cache/resolve entry.
12 *
13 * HOW: The forbidden-id sets are private to this translation unit; the mapping
14 * policy globals (idmap_min_uid / idmap_primary_only) are owned by idmap.c and
15 * consulted through idmap_internal.h. No goto; pure helpers, side effects at
16 * the edges.
17 */

30/* Deny-lists: account/group NAMES resolved to numeric ids once at init. A
31 * resolved target uid in idmap_forbidden_uids, or a user gid (primary or
32 * supplementary) in idmap_forbidden_gids, is refused outright — defence in depth
33 * on top of the numeric min_uid floor (catches privileged accounts/groups whose
34 * ids are >= the floor, e.g. a site nginx uid 1500 or a docker gid 1500). */
35#define IDMAP_FORBID_MAX 256
36static uid_t idmap_forbidden_uids[IDMAP_FORBID_MAX];
37static int idmap_forbidden_uids_n;
38static gid_t idmap_forbidden_gids[IDMAP_FORBID_MAX];
39static int idmap_forbidden_gids_n;

42/* True if gid is reserved (0 / below the effective floor) or on the forbidden
43 * privileged-group list. Used to scan the FULL group set on overflow. */
44static int
45idmap_gid_reserved_or_forbidden(gid_t gid)
46{
47 int k;
48
49 if (gid == 0 || gid < (gid_t) idmap_min_uid) {
50 return 1;
51 }
52 for (k = 0; k < idmap_forbidden_gids_n; k++) {
53 if (gid == idmap_forbidden_gids[k]) {
54 return 1;
55 }
56 }
57 return 0;
58}

60/*
61 * Resolve a local username to {uid, gid, supplementary gids}. Returns 0 and
62 * fills *out on success, -1 if the user does not exist, or -2 if the user is a
63 * member of a RESERVED or FORBIDDEN group anywhere in their (possibly >32) group
64 * set — caller must treat both negatives as deny. Mirrors the getpwnam/
65 * getgrouplist usage in src/acc/groups.c, but keeps the numeric ids and, crucially,
66 * fails CLOSED when the group set overflows the 32-slot buffer.
67 */
68int
69idmap_resolve_user(const char *user, brix_idmap_creds_t *out)
70{
71 struct passwd *pw;
72 gid_t gids[BRIX_IDMAP_MAXGROUPS];
73 int ng = BRIX_IDMAP_MAXGROUPS;
74 int i;
75
76 pw = getpwnam(user);
77 if (pw == NULL) {
78 return -1;
79 }
80
81 out->uid = pw->pw_uid;
82 out->gid = pw->pw_gid;
83
84 if (idmap_primary_only) {
85 out->groups[0] = pw->pw_gid;
86 out->ngroups = 1;
87 return 0;
88 }
89
90 if (getgrouplist(user, pw->pw_gid, gids, &ng) < 0) {
91 /*
92 * OVERFLOW: the user belongs to MORE than BRIX_IDMAP_MAXGROUPS groups.
93 * `ng` now holds the TRUE total; `gids` holds only a 32-entry PREFIX in
94 * /etc/group scan order (NOT sorted). A reserved or forbidden group could
95 * sit past the cap, so checking only the prefix would fail OPEN — a
96 * dropped group is a MISSED deny. Re-resolve the FULL set and REFUSE if
97 * ANY group is reserved (0 / < floor) or forbidden. Common on multi-VO
98 * HEP hosts where users exceed 32 group memberships.
99 */
100 int total = ng;
101 gid_t *full = malloc((size_t) total * sizeof(gid_t));
102 int j;
103
104 if (full == NULL) {
105 return -1; /* OOM -> deny (fail closed) */
106 }
107 if (getgrouplist(user, pw->pw_gid, full, &total) < 0) {
108 free(full);
109 return -1; /* grew again / still failing -> deny */
110 }
111 for (j = 0; j < total; j++) {
112 if (idmap_gid_reserved_or_forbidden(full[j])) {
113 free(full);
114 return -2; /* reserved/forbidden group -> DENY */
115 }
116 }
117 /* Whole set is clean; keep a capped subset for the impersonation setgroups
118 * (a subset only ever GRANTS LESS, so it is fail-safe for the op). */
119 for (i = 0; i < total && i < BRIX_IDMAP_MAXGROUPS; i++) {
120 out->groups[i] = full[i];
121 }
122 out->ngroups = i;
123 free(full);
124 return 0;
125 }
126 if (ng < 1) {
127 gids[0] = pw->pw_gid;
128 ng = 1;
129 }
130 for (i = 0; i < ng && i < BRIX_IDMAP_MAXGROUPS; i++) {
131 out->groups[i] = gids[i];
132 }
133 out->ngroups = i;
134 return 0;
135}

153/*
154 * The single authoritative reserved-id test (declared in impersonate.h, used by
155 * BOTH this mapping layer and the broker's syscall-edge guard). A credential set
156 * is "privileged" — and must be REFUSED — if its primary uid, primary gid, or ANY
157 * supplementary gid is 0 or strictly below `floor`, or if ngroups is out of
158 * range. Pure: no syscalls, no globals.
159 */
160int
161brix_imp_creds_privileged(const brix_idmap_creds_t *cr, uid_t floor,
162 uint32_t *out_id, char *out_kind)
163{
164 int i;
165
166 if (cr == NULL) {
167 imp_priv_report(0, 'n', out_id, out_kind);
168 return 1;
169 }
170 if (cr->ngroups < 0 || cr->ngroups > BRIX_IDMAP_MAXGROUPS) {
171 imp_priv_report((uint32_t) cr->ngroups, 'n', out_id, out_kind);
172 return 1;
173 }
174 if (imp_id_reserved((uint32_t) cr->uid, floor)) {
175 imp_priv_report((uint32_t) cr->uid, 'u', out_id, out_kind);
176 return 1;
177 }
178 if (imp_id_reserved((uint32_t) cr->gid, floor)) {
179 imp_priv_report((uint32_t) cr->gid, 'g', out_id, out_kind);
180 return 1;
181 }
182 for (i = 0; i < cr->ngroups; i++) {
183 if (imp_id_reserved((uint32_t) cr->groups[i], floor)) {
184 imp_priv_report((uint32_t) cr->groups[i], 's', out_id, out_kind);
185 return 1;
186 }
187 }
188 return 0;
189}

193static void
194idmap_forbid_uid(uid_t uid)
195{
196 int i;
197
198 for (i = 0; i < idmap_forbidden_uids_n; i++) {
199 if (idmap_forbidden_uids[i] == uid) {
200 return;
201 }
202 }
203 if (idmap_forbidden_uids_n < IDMAP_FORBID_MAX) {
204 idmap_forbidden_uids[idmap_forbidden_uids_n++] = uid;
205 }

224/*
225 * Pull the NEXT comma/whitespace-separated name from *pp into name[] (NUL
226 * terminated, truncated to cap-1; any overlong remainder of the token is
227 * consumed). Advances *pp; returns the copied length, 0 when the list is
228 * exhausted. Pure scan — no lookups.
229 */
230static size_t
231idmap_forbid_next_name(const char **pp, char *name, size_t cap)
232{
233 const char *p = *pp;
234 size_t n = 0;
235
236 while (*p == ',' || *p == ' ' || *p == '\t') { p++; }
237 while (*p != '\0' && *p != ',' && *p != ' ' && *p != '\t' && n < cap - 1) {
238 name[n++] = *p++;
239 }
240 name[n] = '\0';
241 while (*p != '\0' && *p != ',' && *p != ' ' && *p != '\t') { p++; }
242 *pp = p;
243 return n;
244}

246/* Resolve one account/group NAME to its numeric id (getpwnam vs getgrnam per
247 * `is_user`) and append it to the matching forbidden set. Names that do not
248 * resolve on this host are skipped (an absent account cannot be a target). */
249static void
250idmap_forbid_resolve_name(const char *name, int is_user)
251{
252 if (is_user) {
253 struct passwd *pw = getpwnam(name);
254 if (pw != NULL) {
255 idmap_forbid_uid(pw->pw_uid);
256 }
257 return;
258 }
259 {
260 struct group *gr = getgrnam(name);
261 if (gr != NULL) {
262 idmap_forbid_gid(gr->gr_gid);
263 }
264 }
265}

267/*
268 * Parse a comma/whitespace-separated NAME list and resolve each name to a numeric
269 * id via the supplied resolver, adding it to the matching forbidden set. `is_user`
270 * selects getpwnam (uid) vs getgrnam (gid). Names that do not resolve on this host
271 * are skipped (a privileged group absent here cannot be a target anyway). `list`
272 * is "" => use the built-in default.
273 */
274static void
275idmap_forbid_load(const char *list, const char *def, int is_user, ngx_log_t *log)
276{
277 const char *p = (list != NULL && list[0] != '\0') ? list : def;
278 char name[256];
279
280 while (*p != '\0') {
281 if (idmap_forbid_next_name(&p, name, sizeof(name)) == 0) {
282 continue; /* trailing separators -> re-test end */
283 }
284 idmap_forbid_resolve_name(name, is_user);
285 }
286 (void) log;
287}

303/* True if any of the creds' gids (primary or supplementary) is a forbidden
304 * (privileged) group — even when its numeric gid is >= the floor. */
305static int
306idmap_creds_have_forbidden_group(const brix_idmap_creds_t *cr)
307{
308 int i, j;
309
310 for (j = 0; j < idmap_forbidden_gids_n; j++) {
311 if (cr->gid == idmap_forbidden_gids[j]) {
312 return 1;
313 }
314 for (i = 0; i < cr->ngroups; i++) {
315 if (cr->groups[i] == idmap_forbidden_gids[j]) {
316 return 1;
317 }
318 }
319 }
320 return 0;
321}

323/*
324 * Accept these creds as an impersonation target only if NONE of:
325 * - any reserved/floor id (uid/primary gid/supplementary gid < floor or 0),
326 * - the target uid is an explicitly forbidden account (service/system),
327 * - the user is a member of a forbidden privileged group (sudo/wheel/...).
328 */
329int
330idmap_creds_allowed(const brix_idmap_creds_t *cr)
331{
332 return !brix_imp_creds_privileged(cr, idmap_min_uid, NULL, NULL)
333 && !idmap_uid_forbidden(cr->uid)
334 && !idmap_creds_have_forbidden_group(cr);
335}

337/*
338 * Resolve the deny-lists (account names -> uids, privileged group names ->
339 * gids) once, at init. Reset first so a hot-reload re-reads them. The nginx
340 * worker uid is ALWAYS forbidden as a target so the gateway cannot be
341 * impersonated as itself. Names are taken from config or the built-in
342 * defaults; unresolved names are skipped.
343 */
344void
345idmap_init_denylists(const brix_idmap_conf_t *conf, ngx_log_t *log)
346{
347 char ubuf[1024], gbuf[1024];
348 const char *users = NULL;
349 const char *groups = NULL;
350
351 idmap_forbidden_uids_n = 0;
352 idmap_forbidden_gids_n = 0;
353
354 if (conf->forbidden_users.len > 0
355 && brix_str_cbuf(ubuf, sizeof(ubuf), &conf->forbidden_users) != NULL) {
356 users = ubuf;
357 }
358 idmap_forbid_load(users, BRIX_IMP_DEFAULT_FORBIDDEN_USERS, 1, log);
359
360 if (conf->forbidden_groups.len > 0
361 && brix_str_cbuf(gbuf, sizeof(gbuf), &conf->forbidden_groups) != NULL) {
362 groups = gbuf;
363 }
364 idmap_forbid_load(groups, BRIX_IMP_DEFAULT_FORBIDDEN_GROUPS, 0, log);
365
366 if (conf->worker_uid != 0 && conf->worker_uid != (uid_t) NGX_CONF_UNSET_UINT) {
367 idmap_forbid_uid(conf->worker_uid);
368 }
369 if (log != NULL) {
370 ngx_log_error(NGX_LOG_NOTICE, log, 0,
371 "impersonate: deny-lists loaded (%d forbidden uids, %d "
372 "forbidden gids)", idmap_forbidden_uids_n,
373 idmap_forbidden_gids_n);
374 if (idmap_forbidden_uids_n >= IDMAP_FORBID_MAX
375 || idmap_forbidden_gids_n >= IDMAP_FORBID_MAX) {
376 ngx_log_error(NGX_LOG_WARN, log, 0,
377 "impersonate: forbidden deny-list hit the %d-entry cap; "
378 "names beyond it were NOT enforced — reduce the list or "
379 "raise IDMAP_FORBID_MAX", IDMAP_FORBID_MAX);
380 }
381 }
