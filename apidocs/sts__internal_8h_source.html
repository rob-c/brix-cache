1/*
2 * sts_internal.h — cross-file contract for the S3 STS exchange (phase-79 split).
3 *
4 * WHAT: the shared structs, constants and entry points that the three STS
5 * translation units (`sts.c` orchestrator, `sts_sign.c` SigV4 request
6 * builder, `sts_http.c` transport + XML parser) exchange across their
7 * file boundaries. Nothing here is part of the public API — callers use
8 * only `sts.h`.
9 * WHY: the STS client was one 815-line file; splitting it by concern (build,
10 * transport, orchestrate) keeps each file under the 500-line focus limit,
11 * but the three pieces still share the request/response state structs and
12 * a handful of pure entry points. Declaring those once here — instead of
13 * duplicating the typedefs — is what keeps the split behaviour-identical:
14 * every file sees the exact same struct layout and function signature.
15 * HOW: include after "sts.h" (for brix_s3_sts_conf_t / brix_identity_t). It
16 * declares sts_req_t / sts_resp_t / sts_creds_buf_t, the response size
17 * cap, and the four non-static seam functions the orchestrator calls into.
18 *
19 * Requires: sts.h (brix_s3_sts_conf_t, brix_identity_t) before inclusion.
20 */
21#ifndef BRIX_AUTH_S3_STS_INTERNAL_H
22#define BRIX_AUTH_S3_STS_INTERNAL_H

31/*
32 * sts_req_t — request-construction state for one STS exchange.
33 *
34 * WHAT: bundles the immutable inputs (config, timestamps, derived identity) and
35 * the derived host/credential strings that the query-building and signing
36 * helpers all read, so those helpers take one struct pointer instead of a
37 * loose 5-7 argument tail.
38 * WHY: sts_build_action_qs() and sts_sign_query() previously each carried the
39 * same {cf, host, amzdate, datestamp, ...} argument cluster; promoting it
40 * removes the parameter bloat with zero behaviour change (identical values,
41 * identical evaluation order — the caller fills the struct once). It is
42 * shared across files so the orchestrator (sts.c) can fill it and the
43 * builder (sts_sign.c) can read it.
44 * HOW: brix_s3_sts_assume() zero-initialises one on its stack, populates it in
45 * the prepare phase, and threads a const pointer through the pure builders.
46 *
47 * cf — validated STS target config (borrowed, not owned)
48 * id — inbound identity mapped to RoleSessionName (may be NULL)
49 * host — NUL-terminated authority for the SigV4 "host" header
50 * amzdate — "YYYYMMDDTHHMMSSZ"
51 * datestamp — "YYYYMMDD" (amzdate prefix)
52 * rsn — sanitised RoleSessionName
53 * credential — pre-encoded X-Amz-Credential value ("AKID%2F...%2Faws4_request")
54 * ttl — clamped DurationSeconds
55 */
56typedef struct {
57 const brix_s3_sts_conf_t *cf;
58 const brix_identity_t *id;
59 const char *host;
60 char amzdate[17];
61 char datestamp[9];
62 char rsn[65];
63 char credential[512];
64 int ttl;
65} sts_req_t;

68/*
69 * sts_resp_t — growable capture buffer for the STS HTTP response body.
70 *
71 * WHAT: a pool-backed buffer plus its current length and capacity, filled by the
72 * libcurl write callback and consumed by the XML parser.
73 * WHY: shared because the orchestrator (sts.c) allocates and sizes it before
74 * the transfer while the transport (sts_http.c) grows it during the fetch.
75 * HOW: sts_perform pre-sizes buf to BRIX_STS_RESP_MAX; sts_write_cb appends
76 * bounded by cap; sts_parse_response reads [buf, buf+len).
77 */
78typedef struct {
79 u_char *buf;
80 size_t len;
81 size_t cap;
82} sts_resp_t;

85/*
86 * sts_creds_buf_t — the three fixed output buffers sts_parse_response fills.
87 *
88 * WHAT: bundles each destination buffer with its capacity — AccessKeyId,
89 * SecretAccessKey and (optional) SessionToken — so the parser takes one
90 * struct pointer instead of three loose (char *, size_t) argument pairs.
91 * WHY: keeps sts_parse_response within the ≤5-argument budget with zero
92 * behaviour change; the buffers are the caller's stack storage (borrowed,
93 * written NUL-terminated by the parser). Shared so the orchestrator's
94 * finish phase (sts.c) can hand its stack buffers to the parser (sts_http.c).
95 * HOW: sts_finish declares r_ak/r_sk/r_session, points each field+size at them,
96 * and reads the extracted credentials back after NGX_OK.
97 *
98 * ak / aksz — AccessKeyId buffer and its capacity
99 * sk / sksz — SecretAccessKey buffer and its capacity
100 * session / sesssz — SessionToken buffer and its capacity (may stay empty)
101 */
102typedef struct {
103 char *ak;
104 size_t aksz;
105 char *sk;
106 size_t sksz;
107 char *session;
108 size_t sesssz;
109} sts_creds_buf_t;
