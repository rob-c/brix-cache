1/*
2 * http_body.c - shared nginx request-body chain helpers.
3 *
4 * WHAT: Summarises, writes to fd, and reads from nginx request body chains
5 * (ngx_chain_t of ngx_buf_t). Handles both memory-backed and file-backed
6 * buffers uniformly across WebDAV PUT and S3 PUT operations.
7 *
8 * WHY: HTTP PUT requests carry body content in nginx's request_body->bufs chain,
9 * which mixes spooled-to-file and in-memory buffers. WebDAV and S3 handlers
10 * need to write, read, or summarise this chain without knowing buffer layout.
11 *
12 * HOW: summary walks chain counting file_last-file_pos + last-pos bytes;
13 * write_buf delegates brix_copy_range() for file-backed, pwrite_full loop
14 * for memory-backed; read_all allocates ngx_pnalloc(r->pool) and reads via
15 * pread/ngx_memcpy with EINTR retry. The codec-based decompress-to-fd
16 * pipeline lives in the sibling http_body_decode.c; the shared output-drain
17 * helper below (brix_http_body_pwrite_full) is exported to it via
18 * http_body_internal.h.
19*/

30/*
31 * brix_http_body_pwrite_full - loop pwrite to drain memory buffer into fd.
32 *
33 * WHAT: Writes data[0..len] into fd at position *off via repeated pwrite calls,
34 * advancing off and shrinking len until all bytes written. Retries on EINTR,
35 * errors on zero-byte write or non-EINTR failure.
36 *
37 * WHY: Memory-backed nginx buffers (b->memory=1) must be written via pwrite since
38 * they have no file fd. This function handles partial writes and EINTR retries
39 * uniformly for all memory buffer paths in http_body.c.
40 *
41 * HOW: while(len>0): pwrite(fd, data, len, off). On n<0: retry EINTR, error otherwise.
42 * On n==0: errno=EIO + error. Advances data+=n, len-=n, off+=n until done.
43 */
44
45ngx_int_t
46brix_http_body_pwrite_full(ngx_log_t *log, ngx_fd_t fd, const u_char *data,
47 size_t len, off_t *off, const char *path)
48{
49 /* Delegate the EINTR/short-write loop to the VFS primitive (routes the byte
50 * syscall through the storage seam); keep the path-tagged error log and the
51 * caller's running offset. */
52 if (brix_vfs_pwrite_full(fd, data, len, *off) != NGX_OK) {
53 ngx_log_error(NGX_LOG_ERR, log, errno,
54 "brix_http_body: pwrite failed for %s",
55 path ? path : "-");
56 return NGX_ERROR;
57 }
58 *off += (off_t) len;
59 return NGX_OK;
60}

62/*
63 * brix_http_body_summary - measure the request body without copying it.
64 *
65 * WHAT: walks request_body->bufs once, summing payload bytes and flagging
66 * whether memory-backed and/or spooled-to-file buffers are present.
67 * WHY: callers need the exact size up front (to allocate or to bound input)
68 * and need to know if a file path exists so they can pick copy_range vs
69 * pwrite, before touching any data.
70 * HOW: file buffers contribute file_last-file_pos; memory buffers contribute
71 * last-pos. A malformed file buffer (missing fd, or last<pos) is a hard
72 * NGX_ERROR rather than a silent skip.
73 */
74ngx_int_t
75brix_http_body_summary(ngx_http_request_t *r,
76 brix_http_body_summary_t *out)
77{
78 ngx_chain_t *cl;
79
80 if (out == NULL) {
81 return NGX_ERROR;
82 }
83
84 ngx_memzero(out, sizeof(*out));
85
86 if (r == NULL || r->request_body == NULL) {
87 return NGX_OK;
88 }
89
90 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
91 ngx_buf_t *b = cl->buf;
92
93 if (b == NULL) {
94 continue;
95 }
96
97 if (b->in_file) {
98 if (b->file == NULL || b->file->fd == NGX_INVALID_FILE
99 || b->file_last < b->file_pos)
100 {
101 return NGX_ERROR;
102 }
103 out->has_spooled = 1;
104 out->bytes += (size_t) (b->file_last - b->file_pos);
105
106 } else if (b->pos < b->last) {
107 out->has_memory = 1;
108 out->bytes += (size_t) (b->last - b->pos);
109 }
110 }
111
112 return NGX_OK;
113}

115/*
116 * brix_http_body_write_buf - write one body buffer to dst_fd at *dst_off.
117 *
118 * WHAT: appends a single ngx_buf_t's payload at *dst_off and advances *dst_off
119 * by the number of bytes written.
120 * WHY: the two buffer kinds need different syscalls — a spooled buffer is
121 * fd-to-fd so it can use the zero-copy copy_range path, while a memory
122 * buffer must be pushed out with pwrite.
123 * HOW: in_file -> brix_copy_range(src fd@file_pos -> dst_fd@*dst_off);
124 * memory -> pwrite_full(pos..last). Empty buffers are a no-op success.
125 * errno is set on the EINVAL guard paths so callers can map it.
126 */
127ngx_int_t
128brix_http_body_write_buf(ngx_http_request_t *r, ngx_fd_t dst_fd,
129 ngx_buf_t *buf, off_t *dst_off, const char *log_path)
130{
131 size_t len;
132
133 if (r == NULL || buf == NULL || dst_off == NULL) {
134 errno = EINVAL;
135 return NGX_ERROR;
136 }
137
138 if (buf->in_file) {
139 if (buf->file == NULL || buf->file->fd == NGX_INVALID_FILE
140 || buf->file_last < buf->file_pos)
141 {
142 errno = EINVAL;
143 return NGX_ERROR;
144 }
145
146 len = (size_t) (buf->file_last - buf->file_pos);
147 if (len == 0) {
148 return NGX_OK;
149 }
150
151 if (brix_copy_range(r->connection->log, buf->file->fd,
152 buf->file_pos, dst_fd, *dst_off, len,
153 log_path, log_path) != NGX_OK)
154 {
155 return NGX_ERROR;
156 }
157
158 *dst_off += (off_t) len;
159 return NGX_OK;
160 }
161
162 if (buf->pos >= buf->last) {
163 return NGX_OK;
164 }
165
166 return brix_http_body_pwrite_full(r->connection->log, dst_fd, buf->pos,
167 (size_t) (buf->last - buf->pos),
168 dst_off, log_path);
169}

248/*
249 * brix_http_body_staged_spooled_buf — stage one spooled (in_file) body buffer.
250 *
251 * WHAT: reads buf[file_pos..file_last] from its temp fd in 64 KiB chunks and
252 * forwards each chunk to brix_vfs_staged_write at the running *off,
253 * advancing *off by the bytes staged.
254 * WHY: a spooled nginx buffer has no in-memory payload, so it must be pread out
255 * of its temp fd before the object driver can accept it; factoring this
256 * out keeps the write_to_staged orchestrator flat and under the CCN gate.
257 * HOW: validate the file handle (missing fd / last<pos -> EINVAL); loop pread
258 * (EINTR-retry, n==0 -> EIO) into a stack chunk, stage each read, advance
259 * both the source position and the destination *off. Behaviour is byte-
260 * identical to the previous inline loop.
261 */
262static ngx_int_t
263brix_http_body_staged_spooled_buf(brix_vfs_staged_t *st, ngx_buf_t *b,
264 off_t *off)
265{
266 off_t fpos;
267
268 if (b->file == NULL || b->file->fd == NGX_INVALID_FILE
269 || b->file_last < b->file_pos)
270 {
271 errno = EINVAL;
272 return NGX_ERROR;
273 }
274
275 for (fpos = b->file_pos; fpos < b->file_last; ) {
276 u_char chunk[65536];
277 size_t want = (size_t) (b->file_last - fpos);
278 ssize_t n;
279
280 if (want > sizeof(chunk)) {
281 want = sizeof(chunk);
282 }
283 n = pread(b->file->fd, chunk, want, fpos);
284 if (n < 0) {
285 if (errno == EINTR) {
286 continue;
287 }
288 return NGX_ERROR;
289 }
290 if (n == 0) {
291 errno = EIO;
292 return NGX_ERROR;
293 }
294 if (brix_vfs_staged_write(st, chunk, (size_t) n, *off) != NGX_OK) {
295 return NGX_ERROR;
296 }
297 *off += n;
298 fpos += n;
299 }
300
301 return NGX_OK;
302}

403/*
404 * brix_http_body_read_all - copy the entire body into one pool buffer.
405 *
406 * WHAT: allocates a single NUL-terminated buffer from r->pool and fills it
407 * with the whole body; returns it via *out / *out_len.
408 * WHY: small structured bodies (S3 multi-object delete, lockinfo, etc.) are
409 * easier to parse as one contiguous string than as a buffer chain.
410 * HOW: size via summary; refuse bodies larger than max_bytes with
411 * NGX_DECLINED (the buffer is never allocated in that case); then read
412 * file buffers via pread (EINTR-retry) and memcpy memory buffers, both
413 * appending at a running pos. Buffer is pool-owned (no free needed);
414 * the extra +1 byte holds the trailing '\0' (out_len excludes it).
415 */
416ngx_int_t
417brix_http_body_read_all(ngx_http_request_t *r, size_t max_bytes,
418 u_char **out, size_t *out_len)
419{
420 ngx_chain_t *cl;
421 brix_http_body_summary_t summary;
422 u_char *buf;
423 size_t pos;
424
425 if (out == NULL || out_len == NULL) {
426 return NGX_ERROR;
427 }
428
429 *out = NULL;
430 *out_len = 0;
431
432 if (brix_http_body_summary(r, &summary) != NGX_OK) {
433 return NGX_ERROR;
434 }
435
436 if (summary.bytes > max_bytes) {
437 return NGX_DECLINED;
438 }
439
440 BRIX_PNALLOC_OR_RETURN(buf, r->pool, summary.bytes + 1, NGX_ERROR);
441
442 pos = 0;
443 if (r->request_body != NULL) {
444 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
445 ngx_buf_t *b = cl->buf;
446
447 if (b == NULL) {
448 continue;
449 }
450
451 if (b->in_file) {
452 size_t want = (size_t) (b->file_last - b->file_pos);
453 size_t got = 0;
454
455 /* One full read through the storage seam (EINTR/short-read handled
456 * by the primitive); a short fill is a premature EOF. */
457 if (brix_vfs_pread_full(b->file->fd, buf + pos, want,
458 b->file_pos, &got) != NGX_OK) {
459 return NGX_ERROR;
460 }
461 if (got < want) {
462 errno = EIO;
463 return NGX_ERROR;
464 }
465 pos += got;
466
467 } else if (b->pos < b->last) {
468 size_t n = (size_t) (b->last - b->pos);
469
470 ngx_memcpy(buf + pos, b->pos, n);
471 pos += n;
472 }
473 }
474 }
475
476 buf[pos] = '\0';
477 *out = buf;
478 *out_len = pos;
479 return NGX_OK;
480}
