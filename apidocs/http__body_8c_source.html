1/*
2 * http_body.c - shared nginx request-body chain helpers.
3 *
4 * WHAT: Summarises, writes to fd, and reads from nginx request body chains
5 * (ngx_chain_t of ngx_buf_t). Handles both memory-backed and file-backed
6 * buffers uniformly across WebDAV PUT and S3 PUT operations.
7 *
8 * WHY: HTTP PUT requests carry body content in nginx's request_body->bufs chain,
9 * which mixes spooled-to-file and in-memory buffers. WebDAV and S3 handlers
10 * need to write, read, or summarise this chain without knowing buffer layout.
11 *
12 * HOW: summary walks chain counting file_last-file_pos + last-pos bytes;
13 * write_buf delegates brix_copy_range() for file-backed, pwrite_full loop
14 * for memory-backed; read_all allocates ngx_pnalloc(r->pool) and reads via
15 * pread/ngx_memcpy with EINTR retry.
16*/
17

30/*
31 * brix_http_body_pwrite_full - loop pwrite to drain memory buffer into fd.
32 *
33 * WHAT: Writes data[0..len] into fd at position *off via repeated pwrite calls,
34 * advancing off and shrinking len until all bytes written. Retries on EINTR,
35 * errors on zero-byte write or non-EINTR failure.
36 *
37 * WHY: Memory-backed nginx buffers (b->memory=1) must be written via pwrite since
38 * they have no file fd. This function handles partial writes and EINTR retries
39 * uniformly for all memory buffer paths in http_body.c.
40 *
41 * HOW: while(len>0): pwrite(fd, data, len, off). On n<0: retry EINTR, error otherwise.
42 * On n==0: errno=EIO + error. Advances data+=n, len-=n, off+=n until done.
43 */
44
45static ngx_int_t
46brix_http_body_pwrite_full(ngx_log_t *log, ngx_fd_t fd, const u_char *data,
47 size_t len, off_t *off, const char *path)
48{
49 /* Delegate the EINTR/short-write loop to the VFS primitive (routes the byte
50 * syscall through the storage seam); keep the path-tagged error log and the
51 * caller's running offset. */
52 if (brix_vfs_pwrite_full(fd, data, len, *off) != NGX_OK) {
53 ngx_log_error(NGX_LOG_ERR, log, errno,
54 "brix_http_body: pwrite failed for %s",
55 path ? path : "-");
56 return NGX_ERROR;
57 }
58 *off += (off_t) len;
59 return NGX_OK;
60}

62/*
63 * brix_http_body_summary - measure the request body without copying it.
64 *
65 * WHAT: walks request_body->bufs once, summing payload bytes and flagging
66 * whether memory-backed and/or spooled-to-file buffers are present.
67 * WHY: callers need the exact size up front (to allocate or to bound input)
68 * and need to know if a file path exists so they can pick copy_range vs
69 * pwrite, before touching any data.
70 * HOW: file buffers contribute file_last-file_pos; memory buffers contribute
71 * last-pos. A malformed file buffer (missing fd, or last<pos) is a hard
72 * NGX_ERROR rather than a silent skip.
73 */
74ngx_int_t
75brix_http_body_summary(ngx_http_request_t *r,
76 brix_http_body_summary_t *out)
77{
78 ngx_chain_t *cl;
79
80 if (out == NULL) {
81 return NGX_ERROR;
82 }
83
84 ngx_memzero(out, sizeof(*out));
85
86 if (r == NULL || r->request_body == NULL) {
87 return NGX_OK;
88 }
89
90 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
91 ngx_buf_t *b = cl->buf;
92
93 if (b == NULL) {
94 continue;
95 }
96
97 if (b->in_file) {
98 if (b->file == NULL || b->file->fd == NGX_INVALID_FILE
99 || b->file_last < b->file_pos)
100 {
101 return NGX_ERROR;
102 }
103 out->has_spooled = 1;
104 out->bytes += (size_t) (b->file_last - b->file_pos);
105
106 } else if (b->pos < b->last) {
107 out->has_memory = 1;
108 out->bytes += (size_t) (b->last - b->pos);
109 }
110 }
111
112 return NGX_OK;
113}

115/*
116 * brix_http_body_write_buf - write one body buffer to dst_fd at *dst_off.
117 *
118 * WHAT: appends a single ngx_buf_t's payload at *dst_off and advances *dst_off
119 * by the number of bytes written.
120 * WHY: the two buffer kinds need different syscalls — a spooled buffer is
121 * fd-to-fd so it can use the zero-copy copy_range path, while a memory
122 * buffer must be pushed out with pwrite.
123 * HOW: in_file -> brix_copy_range(src fd@file_pos -> dst_fd@*dst_off);
124 * memory -> pwrite_full(pos..last). Empty buffers are a no-op success.
125 * errno is set on the EINVAL guard paths so callers can map it.
126 */
127ngx_int_t
128brix_http_body_write_buf(ngx_http_request_t *r, ngx_fd_t dst_fd,
129 ngx_buf_t *buf, off_t *dst_off, const char *log_path)
130{
131 size_t len;
132
133 if (r == NULL || buf == NULL || dst_off == NULL) {
134 errno = EINVAL;
135 return NGX_ERROR;
136 }
137
138 if (buf->in_file) {
139 if (buf->file == NULL || buf->file->fd == NGX_INVALID_FILE
140 || buf->file_last < buf->file_pos)
141 {
142 errno = EINVAL;
143 return NGX_ERROR;
144 }
145
146 len = (size_t) (buf->file_last - buf->file_pos);
147 if (len == 0) {
148 return NGX_OK;
149 }
150
151 if (brix_copy_range(r->connection->log, buf->file->fd,
152 buf->file_pos, dst_fd, *dst_off, len,
153 log_path, log_path) != NGX_OK)
154 {
155 return NGX_ERROR;
156 }
157
158 *dst_off += (off_t) len;
159 return NGX_OK;
160 }
161
162 if (buf->pos >= buf->last) {
163 return NGX_OK;
164 }
165
166 return brix_http_body_pwrite_full(r->connection->log, dst_fd, buf->pos,
167 (size_t) (buf->last - buf->pos),
168 dst_off, log_path);
169}

248/*
249 * brix_http_body_write_to_staged — stream the whole request body into a staged
250 * object via brix_vfs_staged_write (a driver-backed/object export has no kernel
251 * fd, so write_to_fd does not apply). Memory buffers are forwarded directly;
252 * spooled (in_file) buffers are read from their temp fd in 64 KiB chunks. The
253 * destination offset runs from 0 so the body lands contiguously.
254 */
255ngx_int_t
256brix_http_body_write_to_staged(ngx_http_request_t *r,
257 brix_vfs_staged_t *st)
258{
259 ngx_chain_t *cl;
260 off_t off = 0;
261
262 if (r == NULL || st == NULL) {
263 errno = EINVAL;
264 return NGX_ERROR;
265 }
266 if (r->request_body == NULL) {
267 return NGX_OK;
268 }
269
270 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
271 ngx_buf_t *b = cl->buf;
272
273 if (b == NULL) {
274 continue;
275 }
276
277 if (b->in_file) {
278 off_t fpos;
279
280 if (b->file == NULL || b->file->fd == NGX_INVALID_FILE
281 || b->file_last < b->file_pos)
282 {
283 errno = EINVAL;
284 return NGX_ERROR;
285 }
286 for (fpos = b->file_pos; fpos < b->file_last; ) {
287 u_char chunk[65536];
288 size_t want = (size_t) (b->file_last - fpos);
289 ssize_t n;
290
291 if (want > sizeof(chunk)) {
292 want = sizeof(chunk);
293 }
294 n = pread(b->file->fd, chunk, want, fpos);
295 if (n < 0) {
296 if (errno == EINTR) {
297 continue;
298 }
299 return NGX_ERROR;
300 }
301 if (n == 0) {
302 errno = EIO;
303 return NGX_ERROR;
304 }
305 if (brix_vfs_staged_write(st, chunk, (size_t) n, off)
306 != NGX_OK)
307 {
308 return NGX_ERROR;
309 }
310 off += n;
311 fpos += n;
312 }
313
314 } else if (b->pos < b->last) {
315 size_t len = (size_t) (b->last - b->pos);
316
317 if (brix_vfs_staged_write(st, b->pos, len, off) != NGX_OK) {
318 return NGX_ERROR;
319 }
320 off += (off_t) len;
321 }
322 }
323
324 return NGX_OK;
325}

327/*
328 * brix_http_body_read_all - copy the entire body into one pool buffer.
329 *
330 * WHAT: allocates a single NUL-terminated buffer from r->pool and fills it
331 * with the whole body; returns it via *out / *out_len.
332 * WHY: small structured bodies (S3 multi-object delete, lockinfo, etc.) are
333 * easier to parse as one contiguous string than as a buffer chain.
334 * HOW: size via summary; refuse bodies larger than max_bytes with
335 * NGX_DECLINED (the buffer is never allocated in that case); then read
336 * file buffers via pread (EINTR-retry) and memcpy memory buffers, both
337 * appending at a running pos. Buffer is pool-owned (no free needed);
338 * the extra +1 byte holds the trailing '\0' (out_len excludes it).
339 */
340ngx_int_t
341brix_http_body_read_all(ngx_http_request_t *r, size_t max_bytes,
342 u_char **out, size_t *out_len)
343{
344 ngx_chain_t *cl;
345 brix_http_body_summary_t summary;
346 u_char *buf;
347 size_t pos;
348
349 if (out == NULL || out_len == NULL) {
350 return NGX_ERROR;
351 }
352
353 *out = NULL;
354 *out_len = 0;
355
356 if (brix_http_body_summary(r, &summary) != NGX_OK) {
357 return NGX_ERROR;
358 }
359
360 if (summary.bytes > max_bytes) {
361 return NGX_DECLINED;
362 }
363
364 BRIX_PNALLOC_OR_RETURN(buf, r->pool, summary.bytes + 1, NGX_ERROR);
365
366 pos = 0;
367 if (r->request_body != NULL) {
368 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
369 ngx_buf_t *b = cl->buf;
370
371 if (b == NULL) {
372 continue;
373 }
374
375 if (b->in_file) {
376 size_t want = (size_t) (b->file_last - b->file_pos);
377 size_t got = 0;
378
379 /* One full read through the storage seam (EINTR/short-read handled
380 * by the primitive); a short fill is a premature EOF. */
381 if (brix_vfs_pread_full(b->file->fd, buf + pos, want,
382 b->file_pos, &got) != NGX_OK) {
383 return NGX_ERROR;
384 }
385 if (got < want) {
386 errno = EIO;
387 return NGX_ERROR;
388 }
389 pos += got;
390
391 } else if (b->pos < b->last) {
392 size_t n = (size_t) (b->last - b->pos);
393
394 ngx_memcpy(buf + pos, b->pos, n);
395 pos += n;
396 }
397 }
398 }
399
400 buf[pos] = '\0';
401 *out = buf;
402 *out_len = pos;
403 return NGX_OK;
404}

407 * brix_http_read_body - kick off async body reading and normalise the rc.
408 *
409 * WHAT: thin wrapper over ngx_http_read_client_request_body that maps nginx's
410 * return convention onto this module's.
411 * WHY: the body usually is not fully buffered yet; nginx will invoke handler

414 * HOW: an error rc (>= NGX_HTTP_SPECIAL_RESPONSE) is returned verbatim as the

418ngx_int_t

421{

423
424 rc = ngx_http_read_client_request_body(r, handler);
425 if (rc >= NGX_HTTP_SPECIAL_RESPONSE) {

431/* Feed one input chunk through the codec stream, writing all produced output to
432 * dst_fd. finish!=0 marks the final input. *ended is set when the codec reports
433 * end-of-stream. *worst_rc captures the first negative codec rc (for the caller's
434 * HTTP-status mapping: ERR_BOMB -> 413, ERR_DATA -> 400). Returns NGX_OK/ERROR. */
435static ngx_int_t
436codec_feed(brix_codec_stream_t *s, ngx_log_t *log, ngx_fd_t dst_fd,
437 const char *log_path, off_t *dst_off, u_char *outbuf,
438 const u_char *in, size_t in_len, int finish, int *ended,
439 brix_codec_rc_t *worst_rc)
440{
441 size_t ip = 0;
442
443 for (;;) {
444 size_t op = 0;
445 brix_codec_rc_t rc;
446
447 rc = brix_codec_step(s, (const uint8_t *) in, in_len, &ip,
448 (uint8_t *) outbuf, BRIX_INFLATE_OUT_BUFSZ,
449 &op, finish);
450 if (rc < 0) {
451 *worst_rc = rc;
452 ngx_log_error(NGX_LOG_ERR, log, 0,
453 "brix_http_body: decode error %d for %s",
454 (int) rc, log_path ? log_path : "-");
455 return NGX_ERROR;
456 }
457 if (op > 0) {
458 if (brix_http_body_pwrite_full(log, dst_fd, outbuf, op,
459 dst_off, log_path) != NGX_OK)
460 {
461 return NGX_ERROR;
462 }
463 }
464 if (rc == BRIX_CODEC_END) {
465 *ended = 1;
466 return NGX_OK;
467 }
468 if (op == BRIX_INFLATE_OUT_BUFSZ) {
469 continue; /* output buffer was full: drain more */
470 }
471 if (ip < in_len) {
472 continue; /* input remains in this chunk */
473 }
474 return NGX_OK; /* chunk consumed, output drained */
475 }
476}

478/*
479 * codec_decode_bufs — feed every request-body buffer through an open codec
480 * stream, writing decompressed bytes to dst_fd, then finalise.
481 *
482 * In-memory buffers feed directly; spooled buffers are pread in
483 * BRIX_INFLATE_IN_BUFSZ chunks into inbuf first. After the last buffer a final
484 * flush (empty input, finish=1) drains any tail; if the codec never reports
485 * end-of-stream the input was truncated/corrupt (ERR_DATA -> caller maps 400).
486 * Flat, early-return cleanup; the stream + buffers are owned by the caller.
487 */
488static ngx_int_t
489codec_decode_bufs(brix_codec_stream_t *s, ngx_http_request_t *r,
490 ngx_fd_t dst_fd, const char *log_path, u_char *outbuf, u_char *inbuf,
491 ngx_log_t *log, brix_codec_rc_t *worst_rc)
492{
493 ngx_chain_t *cl;
494 off_t dst_off = 0;
495 int ended = 0;
496
497 for (cl = r->request_body->bufs; cl != NULL; cl = cl->next) {
498 ngx_buf_t *b = cl->buf;
499
500 if (b == NULL) {
501 continue;
502 }
503
504 if (b->in_file) {
505 off_t file_off = b->file_pos;
506
507 while (file_off < b->file_last) {
508 size_t want = (size_t) (b->file_last - file_off);
509 size_t got = 0;
510
511 if (want > BRIX_INFLATE_IN_BUFSZ) {
512 want = BRIX_INFLATE_IN_BUFSZ;
513 }
514 /* One chunk through the storage seam; EOF before file_last (a
515 * short fill) is an error, matching the old n<=0 check. */
516 if (brix_vfs_pread_full(b->file->fd, inbuf, want, file_off,
517 &got) != NGX_OK || got < want)
518 {
519 ngx_log_error(NGX_LOG_ERR, log, errno,
520 "brix_http_body: decode pread failed for %s",
521 log_path ? log_path : "-");
522 return NGX_ERROR;
523 }
524 file_off += (off_t) got;
525 if (codec_feed(s, log, dst_fd, log_path, &dst_off, outbuf,
526 inbuf, got, 0, &ended, worst_rc) != NGX_OK)
527 {
528 return NGX_ERROR;
529 }
530 }
531
532 } else if (b->pos < b->last) {
533 if (codec_feed(s, log, dst_fd, log_path, &dst_off, outbuf,
534 b->pos, (size_t) (b->last - b->pos), 0, &ended,
535 worst_rc) != NGX_OK)
536 {
537 return NGX_ERROR;
538 }
539 }
540 }
541
542 /* Final flush: signal end-of-input and drain the codec's tail. */
543 if (!ended) {
544 if (codec_feed(s, log, dst_fd, log_path, &dst_off, outbuf,
545 (const u_char *) "", 0, 1, &ended, worst_rc) != NGX_OK)
546 {
547 return NGX_ERROR;
548 }
549 }
550 if (!ended) {
551 /* All input consumed + finish, but no end-of-stream: truncated input. */
552 *worst_rc = BRIX_CODEC_ERR_DATA;
553 ngx_log_error(NGX_LOG_ERR, log, 0,
554 "brix_http_body: truncated compressed body for %s",
555 log_path ? log_path : "-");
556 return NGX_ERROR;
557 }
558 return NGX_OK;
559}

562 * brix_http_body_decode_ratio - choose the untrusted decode ratio ceiling.
563 *
564 * WHAT: Returns the maximum permitted output:input expansion ratio for the given
565 * Content-Encoding codec on HTTP request-body decode.
566 * WHY: The generic 1000:1 ceiling catches classic bombs for most codecs, but LZ4
567 * frames cap zero-heavy compression below that; a lower LZ4-specific limit

571 */
572static uint32_t

576 return BRIX_DECODE_LZ4_MAX_RATIO;
577 }
578 return BRIX_DECODE_MAX_RATIO;
579}

581/*
582 * brix_http_body_decode_to_fd - decompress the request body to dst_fd.
583 *
584 * WHAT: streams the Content-Encoding-selected codec over the request body chain,
585 * writing plaintext to dst_fd. Bounds output via the bomb guard (out_cap =
586 * max_output, ratio default) so a hostile highly-compressible upload cannot
587 * exhaust disk; on any failure sets *http_status_out (413 bomb / 400 bad
588 * data / 500 I-O) and returns NGX_ERROR. On success returns NGX_OK.
589 * HOW: open one brix_codec stream, hand the per-buffer feed loop to
590 * codec_decode_bufs, close once on return. Buffers freed here on all paths.
591 */
592ngx_int_t
593brix_http_body_decode_to_fd(ngx_http_request_t *r, ngx_fd_t dst_fd,
594 const char *log_path, brix_codec_id_t codec, uint64_t max_output,
595 brix_http_body_summary_t *summary_out, ngx_int_t *http_status_out)
596{
597 brix_codec_stream_t *s;
598 brix_codec_guard_t guard;
599 brix_codec_rc_t worst_rc = BRIX_CODEC_OK;
600 ngx_int_t rc;
601 u_char *outbuf = NULL;
602 u_char *inbuf = NULL;
603 ngx_log_t *log;
604 brix_http_body_summary_t summary;
605
606 if (summary_out == NULL) {
607 summary_out = &summary;
608 }
609 if (brix_http_body_summary(r, summary_out) != NGX_OK) {
610 if (http_status_out) { *http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR; }
611 return NGX_ERROR;
612 }
613 if (r == NULL || r->request_body == NULL
614 || r->request_body->bufs == NULL)
615 {
616 return NGX_OK;
617 }
618
619 log = r->connection->log;
620
621 ngx_memzero(&guard, sizeof(guard));
622 guard.out_cap = max_output; /* 0 = unbounded */
623 guard.max_ratio = brix_http_body_decode_ratio(codec);
624
625 s = brix_codec_open(codec, BRIX_CODEC_DIR_DECOMPRESS, -1, &guard);
626 if (s == NULL) {
627 if (http_status_out) {
628 *http_status_out = NGX_HTTP_UNSUPPORTED_MEDIA_TYPE;
629 }
630 ngx_log_error(NGX_LOG_ERR, log, 0,
631 "brix_http_body: codec %d unavailable for %s",
632 (int) codec, log_path ? log_path : "-");
633 return NGX_ERROR;
634 }
635
636 outbuf = ngx_alloc(BRIX_INFLATE_OUT_BUFSZ, log);
637 if (outbuf == NULL) {
638 brix_codec_close(s);
639 if (http_status_out) { *http_status_out = NGX_HTTP_INTERNAL_SERVER_ERROR; }
640 return NGX_ERROR;
641 }
642 if (summary_out->has_spooled) {
