134/* Build the round-2 kXGC_cert response to a server's kXGS_cert, the single shared
135 * XrdSecgsi client/dest round-2 (both DH variants: signed-DH via kXRS_cipher when
136 * the server sends one, else unsigned kXRS_puk). Agrees the AES session key, signs
137 * the server's random tag with proxy_key (proof-of-possession), and emits an
138 * encrypted main carrying proxy_pem (the proxy cert chain PEM) + the signed tag +
139 * a fresh tag, plus our DH public (RSA-signed in the signed-DH case).
140 *
141 * proxy_pem (proxy_pem_len) and proxy_key are BORROWED — the caller still owns and
142 * frees them. On success payload+plen receive a malloc'd kXGC_cert buffer the
143 * caller must free(); returns 0. On failure returns -1 and, if err!=NULL, writes a
144 * short reason into err[errcap]. Both the native client (client/lib/sec/sec_gsi.c)
145 * and the TPC destination (src/tpc/gsi/gsi_outbound_exchange.c) call this. */
146int brix_gsi_build_cert_response(const uint8_t *sbody, uint32_t slen,
