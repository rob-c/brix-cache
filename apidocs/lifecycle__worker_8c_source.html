38/*
39 * Hyper-harden the worker: shed the privilege-escalation / DAC-bypass / identity
40 * capabilities so a compromised worker can never setfsuid/setgid or override DAC,
41 * even if nginx was granted them via file/ambient capabilities to feed the broker.
42 * Best-effort: clearing effective+permitted always succeeds; bounding-set drops
43 * need CAP_SETPCAP and are skipped (with no error) when already unprivileged. The
44 * worker is a pure broker CLIENT and needs none of these.
45 */
46/* True iff this process currently holds `cap` in its effective OR permitted set.
47 * cap < 32 for every cap we test (SETUID=7, SETGID=6, DAC_OVERRIDE=1, SYS_ADMIN=21),
48 * so word 0 suffices. Best-effort: a capget failure reports "not held". */
49int
50brix_imp_cap_held(int cap)
51{
52 struct __user_cap_header_struct hdr;
53 struct __user_cap_data_struct data[2];
54
55 ngx_memzero(&hdr, sizeof(hdr));
56 ngx_memzero(data, sizeof(data));
57 hdr.version = _LINUX_CAPABILITY_VERSION_3;
58 hdr.pid = 0;
59 if (syscall(SYS_capget, &hdr, data) != 0) {
60 return 0;
61 }
62 return (data[0].effective & (1u << cap)) != 0
63 || (data[0].permitted & (1u << cap)) != 0;
64}

66static void
67imp_worker_drop_caps(ngx_log_t *log)
68{
69 struct __user_cap_header_struct hdr;
70 struct __user_cap_data_struct data[2];
71 unsigned keep;
72 /* The dangerous set — always shed from the bounding set (best-effort) and
73 * from permitted+effective. SETUID/SETGID are deliberately NOT here: they are
74 * retained IFF currently held, so a legitimate later identity drop still works
75 * — the end-of-init worker de-escalation (brix_imp_worker_deescalate) uses them
76 * to setuid the worker down to brix_worker_user/nobody, and the pblock backend's
77 * off-root drop uses them too. Keeping SETUID adds no escalation a root (or
78 * already-CAP_SETUID) worker does not already have, and it is stripped again by
79 * the re-harden right after the de-escalation setuid. */
80 static const int kill_caps[] = {
81 CAP_SETPCAP, CAP_DAC_OVERRIDE, CAP_DAC_READ_SEARCH, CAP_FOWNER,
82 CAP_CHOWN, CAP_FSETID, CAP_SYS_ADMIN, CAP_SYS_PTRACE, CAP_MKNOD,
83 CAP_SETFCAP,
84 };
85 unsigned i;
86
87 (void) prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0);
88
89 ngx_memzero(&hdr, sizeof(hdr));
90 ngx_memzero(data, sizeof(data));
91 hdr.version = _LINUX_CAPABILITY_VERSION_3;
92 hdr.pid = 0;
93
94 /* Retain {SETUID,SETGID} only if actually held now (a root worker has all
95 * caps; a non-root worker granted CAP_SETUID via file/ambient caps keeps just
96 * that) — everything else (DAC_OVERRIDE/MKNOD/SYS_ADMIN/PTRACE/CHOWN/...) is
97 * dropped, so uid 0 without CAP_DAC_OVERRIDE now respects on-disk DAC. A plain
98 * unprivileged worker holds nothing and keeps nothing. */
99 keep = ((1u << CAP_SETUID) | (1u << CAP_SETGID));
100 if (!brix_imp_cap_held(CAP_SETUID)) { keep &= ~(1u << CAP_SETUID); }
101 if (!brix_imp_cap_held(CAP_SETGID)) { keep &= ~(1u << CAP_SETGID); }
102 data[0].permitted = data[0].effective = keep;
103
104 if (syscall(SYS_capset, &hdr, data) != 0) {
105 if (log) ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
106 "impersonate: worker capset failed "
107 "(continuing — worker is an unprivileged client)");
108 }
109 /* Best-effort bounding-set drop of the dangerous caps (needs CAP_SETPCAP). */
110 for (i = 0; i < sizeof(kill_caps) / sizeof(kill_caps[0]); i++) {
111 (void) prctl(PR_CAPBSET_DROP, kill_caps[i], 0, 0, 0);
112 }
113}

115/* ---- brix_imp_worker_harden ----
116 *
117 * WHAT: Shed ALL worker capabilities + set NO_NEW_PRIVS, in EVERY impersonation
118 * mode (off/single/map). Idempotent.
119 * WHY: A worker never needs capabilities; a worker (mis)configured to run as
120 * root must not keep CAP_DAC_OVERRIDE/CAP_MKNOD/CAP_SETUID/… — a worker-code
121 * exploit would otherwise be root. The old code only dropped caps in `map`
122 * mode, leaving the default posture (and HTTP-only workers) fully privileged
123 * when the worker ran as root.
124 * HOW: Delegates to imp_worker_drop_caps(). Called from the per-worker init
125 * BEFORE the stream-config early-return so WebDAV/S3 (HTTP-only) workers are
126 * hardened too, not just stream/root:// workers.
127 */
128void
129brix_imp_worker_harden(ngx_log_t *log)
130{
131 imp_worker_drop_caps(log);
132}

134/* ---- brix_worker_user: the confined account the worker is forced down to ----
135 *
136 * Process-global (there is one worker uid per process; the strictest posture is
137 * shared across every brix server, stream + http). Empty => unset => default to
138 * "nobody" with a startup warning. Set by the `brix_worker_user` directive. */
139char brix_worker_user[64] = "";

141char *
142brix_conf_set_worker_user(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
143{
144 ngx_str_t *value = cf->args->elts;
145
146 (void) cmd;
147 (void) conf;
148
149 if (value[1].len == 0 || value[1].len >= sizeof(brix_worker_user)) {
150 return "invalid brix_worker_user account name";
151 }
152 /* Last non-empty setting wins; the account is resolved at worker init. */
153 ngx_memcpy(brix_worker_user, value[1].data, value[1].len);
154 brix_worker_user[value[1].len] = '\0';
155 return NGX_CONF_OK;
156}

158/* ---- brix_imp_worker_runtime_ids ----
159 *
160 * WHAT: Resolve the uid/gid the workers actually serve as (see lifecycle.h).
161 * WHY: Config-time provisioning of worker-writable directories (the default
162 * credential store, the default gateway stage spool) must chown to the
163 * POST-de-escalation identity: with the always-on drop, `user root` (or
164 * no `user` at all under a root master) yields workers running as
165 * brix_worker_user/nobody, and a root:0700 directory would EACCES them.
166 * HOW: A real unprivileged `user` account wins (the drop no-ops for it);
167 * otherwise resolve brix_worker_user (default "nobody") exactly as
168 * brix_imp_worker_deescalate will at worker init. */
169ngx_int_t
170brix_imp_worker_runtime_ids(ngx_uid_t conf_uid, ngx_gid_t conf_gid,
171 uid_t *uid_out, gid_t *gid_out)
172{
173 const char *acct;
174 struct passwd *pw;
175
176 if (conf_uid != (ngx_uid_t) NGX_CONF_UNSET_UINT && conf_uid != 0) {
177 *uid_out = (uid_t) conf_uid;
178 *gid_out = (gid_t) conf_gid;
179 return NGX_OK;
180 }
181 acct = brix_worker_user[0] ? brix_worker_user : "nobody";
182 pw = getpwnam(acct);
183 if (pw == NULL || pw->pw_uid == 0) {
184 return NGX_ERROR;
185 }
186 *uid_out = pw->pw_uid;
187 *gid_out = pw->pw_gid;
188 return NGX_OK;
189}

191/* ---- brix_imp_worker_deescalate ----
192 *
193 * WHAT: Force a worker that can trivially become root — running as root, OR
194 * holding CAP_SETUID (even as a non-root service account) — down to a
195 * confined account (brix_worker_user, else "nobody"), fail-closed, then
196 * re-harden to strip the retained SETUID/SETGID. Runs in EVERY mode and
197 * protocol (stream + http-only), once per worker, right after the cap-harden
198 * and BEFORE backend init / broker connect / the seccomp install.
199 * WHY: The pre-authentication credential parsers (JWT/macaroon/X.509/VOMS/krb5)
200 * run in the worker on fully attacker-controlled bytes; a parse bug must not
201 * land as a root-capable identity. euid==0 is NOT the right test — a non-root
202 * account with CAP_SETUID (or passwordless sudo) is root-equivalent. This
203 * neutralizes the CAP_SETUID case by dropping it; NO_NEW_PRIVS (set by the
204 * harden above, sticky + inherited) neutralizes the setuid-binary/sudo
205 * escalation path for the worker's own exec chain regardless of uid.
206 * HOW: Resolve the target account (fail-closed on missing / uid|gid 0). If already
207 * that uid, done. If we can change uid (root or CAP_SETUID): setgroups/setgid/
208 * setuid, verify it stuck, re-harden. If we CANNOT (plain non-root, no cap):
209 * refuse to serve if any escalation capability somehow survived; else warn.
210 * Returns NGX_OK to continue, NGX_ERROR to fail the worker closed.
211 */
212ngx_int_t
213brix_imp_worker_deescalate(ngx_log_t *log)
214{
215 const char *acct;
216 struct passwd *pw;
217 uid_t tgt_uid;
218 gid_t tgt_gid;
219 int defaulted;
220
221 /* Only real workers de-escalate. The master must stay privileged (bind
222 * privileged ports, fork the map-mode broker); single-process (dev) mode
223 * doubles as the master, so leave it alone too. */
224 if (ngx_process != NGX_PROCESS_WORKER) {
225 return NGX_OK;
226 }
227
228 defaulted = (brix_worker_user[0] == '\0');
229 acct = defaulted ? "nobody" : brix_worker_user;
230
231 errno = 0;
232 pw = getpwnam(acct);
233 if (pw == NULL) {
234 ngx_log_error(NGX_LOG_EMERG, log, errno,
235 "brix_worker_user: account \"%s\" not found — refusing to serve "
236 "(set brix_worker_user to an existing unprivileged account)", acct);
237 return NGX_ERROR;
238 }
239 if (pw->pw_uid == 0 || pw->pw_gid == 0) {
240 ngx_log_error(NGX_LOG_EMERG, log, 0,
241 "brix_worker_user: \"%s\" is a uid/gid 0 account — refusing to serve "
242 "(the worker must run as a confined, non-root account)", acct);
243 return NGX_ERROR;
244 }
245 tgt_uid = pw->pw_uid;
246 tgt_gid = pw->pw_gid;
247
248 /* Already the confined target (e.g. nginx `user` already dropped us): done. */
249 if (getuid() == tgt_uid && geteuid() == tgt_uid) {
250 return NGX_OK;
251 }
252
253 if (geteuid() == 0 || brix_imp_cap_held(CAP_SETUID)) {
254 /* We have the means to change uid — force the drop, fail closed. */
255 if (setgroups(1, &tgt_gid) != 0
256 || setgid(tgt_gid) != 0
257 || setuid(tgt_uid) != 0)
258 {
259 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno,
260 "brix_worker_user: FAILED to drop worker to \"%s\" — refusing to "
261 "serve (pre-auth parsing must never run as a root-capable "
262 "identity)", acct);
263 return NGX_ERROR;
264 }
265 if (getuid() != tgt_uid || geteuid() != tgt_uid) {
266 ngx_log_error(NGX_LOG_EMERG, log, 0,
267 "brix_worker_user: privilege drop to \"%s\" did not stick — "
268 "refusing to serve", acct);
269 return NGX_ERROR;
270 }
271 /* Now unprivileged — strip the SETUID/SETGID we retained for the drop. */
272 brix_imp_worker_harden(log);
273 if (defaulted) {
274 ngx_log_error(NGX_LOG_WARN, log, 0,
275 "brix_worker_user not set: dropped a root-capable worker to "
276 "\"nobody\" (uid=%d) so pre-auth credential parsing never runs as "
277 "a root-capable identity — set brix_worker_user to a dedicated "
278 "confined account instead", (int) tgt_uid);
279 } else {
280 ngx_log_error(NGX_LOG_NOTICE, log, 0,
281 "brix_worker_user: worker dropped to \"%s\" (uid=%d); pre-auth "
282 "parsing runs unprivileged", acct, (int) tgt_uid);
283 }
284 return NGX_OK;
285 }
286
287 /* Cannot change uid (plain non-root, no CAP_SETUID). NO_NEW_PRIVS already
288 * blocks the setuid-binary/sudo escalation path for this worker. Refuse only
289 * if an escalation capability somehow survived the harden; otherwise serve,
290 * warning that the account itself must be sudo-free. */
291 if (brix_imp_cap_held(CAP_SETUID) || brix_imp_cap_held(CAP_SETGID)
292 || brix_imp_cap_held(CAP_DAC_OVERRIDE) || brix_imp_cap_held(CAP_SYS_ADMIN))
293 {
294 ngx_log_error(NGX_LOG_EMERG, log, 0,
295 "worker runs as uid=%d and still holds escalation capabilities that "
296 "could not be dropped — refusing to serve", (int) geteuid());
297 return NGX_ERROR;
298 }
299 if (!defaulted) {
300 /* Operator explicitly named a confined account but the worker is neither
301 * it nor able to drop to it — a real misconfiguration worth a warning. */
302 ngx_log_error(NGX_LOG_WARN, log, 0,
303 "brix_worker_user \"%s\" requested but the worker runs as uid=%d and "
304 "holds no CAP_SETUID to drop — run nginx as root (brix will drop) or "
305 "start it directly as \"%s\"", brix_worker_user, (int) geteuid(),
306 brix_worker_user);
307 return NGX_OK;
308 }
309 /* Plain unprivileged worker (the normal rootless posture): non-root, no caps,
310 * NO_NEW_PRIVS set — not root-capable via its own exec chain, so serve quietly.
311 * INFO (not WARN) records it without cluttering a steady-state log; the residual
312 * to mind is only the account's own sudo/home persistence, a deploy-hygiene note
313 * brix cannot act on from here. */
314 ngx_log_error(NGX_LOG_INFO, log, 0,
315 "worker runs unprivileged as uid=%d (no CAP_SETUID; NO_NEW_PRIVS set); "
316 "pre-auth parsing is unprivileged. For defense in depth ensure this account "
317 "has no passwordless sudo", (int) geteuid());
318 return NGX_OK;
319}

321ngx_int_t
322brix_imp_init_worker(ngx_cycle_t *cycle)
323{
324 char sockbuf[256];
325
326 /* Cap-drop is now done earlier + unconditionally via brix_imp_worker_harden()
327 * (called before the stream-config early-return in ngx_stream_brix_init_process
328 * so HTTP-only workers are covered too). */
329 if (imp_settings.mode != BRIX_IMP_MAP) {
330 return NGX_OK; /* off/single: no broker client */
331 }
332
333 ngx_snprintf((u_char *) sockbuf, sizeof(sockbuf), "%V%Z",
334 &imp_settings.socket);
335
336 if (brix_imp_client_connect(sockbuf, cycle->log) != NGX_OK) {
337 /* Broker may still be starting; the client retries lazily per op. */
338 ngx_log_error(NGX_LOG_WARN, cycle->log, 0,
339 "impersonate: broker not reachable yet at \"%s\" "
340 "(will reconnect on first op)", sockbuf);
341 }
342 return NGX_OK;
343}
