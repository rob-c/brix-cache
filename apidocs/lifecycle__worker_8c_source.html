37/*
38 * Hyper-harden the worker: shed the privilege-escalation / DAC-bypass / identity
39 * capabilities so a compromised worker can never setfsuid/setgid or override DAC,
40 * even if nginx was granted them via file/ambient capabilities to feed the broker.
41 * Best-effort: clearing effective+permitted always succeeds; bounding-set drops
42 * need CAP_SETPCAP and are skipped (with no error) when already unprivileged. The
43 * worker is a pure broker CLIENT and needs none of these.
44 */
45static void
46imp_worker_drop_caps(ngx_log_t *log)
47{
48 struct __user_cap_header_struct hdr;
49 struct __user_cap_data_struct data[2];
50 static const int kill_caps[] = {
51 CAP_SETUID, CAP_SETGID, CAP_SETPCAP, CAP_DAC_OVERRIDE,
52 CAP_DAC_READ_SEARCH, CAP_FOWNER, CAP_CHOWN, CAP_FSETID,
53 CAP_SYS_ADMIN, CAP_SYS_PTRACE, CAP_MKNOD, CAP_SETFCAP,
54 };
55 unsigned i;
56
57 (void) prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0);
58
59 /* Clear ALL effective + permitted + inheritable capabilities. */
60 ngx_memzero(&hdr, sizeof(hdr));
61 ngx_memzero(data, sizeof(data));
62 hdr.version = _LINUX_CAPABILITY_VERSION_3;
63 hdr.pid = 0;
64 if (syscall(SYS_capset, &hdr, data) != 0) {
65 if (log) ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
66 "impersonate: worker capset(clear) failed "
67 "(continuing — worker is an unprivileged client)");
68 }
69 /* Best-effort bounding-set drop of the dangerous caps (needs CAP_SETPCAP). */
70 for (i = 0; i < sizeof(kill_caps) / sizeof(kill_caps[0]); i++) {
71 (void) prctl(PR_CAPBSET_DROP, kill_caps[i], 0, 0, 0);
72 }
73}

75ngx_int_t
76brix_imp_init_worker(ngx_cycle_t *cycle)
77{
78 char sockbuf[256];
79
80 if (imp_settings.mode != BRIX_IMP_MAP) {
81 return NGX_OK; /* off/single: no client */
82 }
83
84 /* Shed privileged caps first — the worker is a pure broker client. */
85 imp_worker_drop_caps(cycle->log);
86
87 ngx_snprintf((u_char *) sockbuf, sizeof(sockbuf), "%V%Z",
88 &imp_settings.socket);
89
90 if (brix_imp_client_connect(sockbuf, cycle->log) != NGX_OK) {
91 /* Broker may still be starting; the client retries lazily per op. */
92 ngx_log_error(NGX_LOG_WARN, cycle->log, 0,
93 "impersonate: broker not reachable yet at \"%s\" "
94 "(will reconnect on first op)", sockbuf);
95 }
96 return NGX_OK;
97}

113void
114brix_imp_request_begin(const brix_identity_t *id)
115{
116 if (imp_settings.mode != BRIX_IMP_MAP) {
117 return;
118 }
119 /* Mark the request active BEFORE setting the principal: even if the identity
120 * yields no mappable principal (empty subject), confined ops must route to the
121 * broker (which denies) rather than fall back to the worker. */
122 brix_imp_mark_in_request(1);
123 brix_imp_set_principal(id != NULL ? imp_principal_of(id) : NULL);
124}
