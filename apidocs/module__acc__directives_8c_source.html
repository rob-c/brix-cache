1/*
2 * module_acc_directives.c — shared XrdAcc HTTP directive setters.
3 *
4 * Moved out of module.c: these setters back the brix_acc_* HTTP directives,
5 * a concern distinct from the command table + module object that remain there.
6 * Each is registered once but populates BOTH the WebDAV and S3 loc-confs (see
7 * the block comment below). Public setters are declared in the matching .h.
8 */

14/*
15 * The XrdAcc directives are valid in any http location and must configure BOTH
16 * the WebDAV and S3 loc-confs (an S3-only location still needs them), but nginx
17 * applies just one module's setter per directive. These shared setters fetch
18 * both loc-confs and populate each, so the directive is registered only once.
19 */
20/*
21 * brix_ktls on|off — registered once (WebDAV) but sets common.ktls on BOTH the
22 * WebDAV and S3 loc-confs, so an S3-only location honours the flag too. Default
23 * ON is applied by the shared merge; this is the explicit per-location override.
24 */
25char *
26brix_http_set_ktls(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
27{
28 ngx_http_brix_webdav_loc_conf_t *wc =
29 ngx_http_conf_get_module_loc_conf(cf, ngx_http_brix_webdav_module);
30 ngx_http_s3_loc_conf_t *sc =
31 ngx_http_conf_get_module_loc_conf(cf, ngx_http_brix_s3_module);
32 ngx_str_t *value = cf->args->elts;
33 ngx_flag_t flag;
34
35 if (ngx_strcasecmp(value[1].data, (u_char *) "on") == 0) {
36 flag = 1;
37 } else if (ngx_strcasecmp(value[1].data, (u_char *) "off") == 0) {
38 flag = 0;
39 } else {
40 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
41 "invalid value \"%V\" in \"%V\" directive, must be \"on\" or \"off\"",
42 &value[1], &cmd->name);
43 return NGX_CONF_ERROR;
44 }
45 wc->common.ktls = flag;
46 sc->common.ktls = flag;
47 return NGX_CONF_OK;
48}

50/*
51 * brix_cache_store_endpoint on|off — registered once (WebDAV) but sets
52 * common.cache_store_endpoint on BOTH the WebDAV and S3 loc-confs, so an S3-only
53 * store location honours the flag too. Marks the location as a trusted remote
54 * cache-STORE surface where internal sidecar names (.cinfo etc.) are legitimate
55 * request targets; default OFF (applied by the shared merge) keeps the
56 * reserved-name 404 guard in force on every normal client location.

97static char *
98brix_acc_http_set_enum(ngx_conf_t *cf, ngx_conf_enum_t *e, ngx_uint_t *wp,
99 ngx_uint_t *sp)
100{
101 ngx_str_t *value = cf->args->elts;
102 ngx_uint_t i;
103 for (i = 0; e[i].name.len != 0; i++) {
104 if (e[i].name.len == value[1].len
105 && ngx_strcmp(e[i].name.data, value[1].data) == 0)
106 {
107 *wp = *sp = e[i].value;
108 return NGX_CONF_OK;
109 }
110 }
111 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0, "invalid value \"%V\"", &value[1]);
112 return NGX_CONF_ERROR;
113}

137/*
138 * Shared scalar setters for the XrdAcc HTTP tunables — registered once but
139 * populate BOTH loc-confs (see the authdb setters above for why). Each grabs
140 * the WebDAV + S3 acc blocks and applies the value to the same field in each.
141 */
142static char *
143brix_acc_http_both(ngx_conf_t *cf, brix_acc_http_t **wc, brix_acc_http_t **sc)
144{
145 ngx_http_brix_webdav_loc_conf_t *w =
146 ngx_http_conf_get_module_loc_conf(cf, ngx_http_brix_webdav_module);
147 ngx_http_s3_loc_conf_t *s =
148 ngx_http_conf_get_module_loc_conf(cf, ngx_http_brix_s3_module);
149 *wc = &w->acc;
150 *sc = &s->acc;
151 return NULL;
152}
