3/* Internal descriptors and cross-file entry points shared by the OCSP split
4 * (ocsp.c / ocsp_transport.c / ocsp_request.c).
5 *
6 * WHAT: Declares the two file-local structs threaded through one OCSP fetch (the
7 * parsed responder URL and the live OpenSSL connection objects), the shared
8 * per-phase network timeout bound, and the seven functions that cross a
9 * translation-unit boundary after the phase-79 file-size split: the URL parse,
10 * the four BIO connection helpers (open / connect / io-timeouts / tls-handshake /
11 * free), and the request/response pair (do_ocsp_request / check_ocsp_response).
12 *
13 * WHY: ocsp.c exceeded the ~500-line file-size guard, so it was carved into
14 * three cohesive units. The URL/BIO transport (ocsp_transport.c) is called by
15 * the request/response crypto (ocsp_request.c), which is in turn called by the
16 * public revocation-check + staple-fetch entry points (ocsp.c). Grouping the
17 * shared structs, the timeout constant, and the crossing entry points here keeps
18 * every definition in exactly one place while preserving the identical request
19 * nonce, signature verification, connect/handshake deadlines, and revocation
20 * decisions of the original single-file implementation.
21 *
22 * HOW: Requires ocsp.h (ngx_log_t / ngx types / the public OCSP declarations)
23 * before inclusion. ocsp_url_t is a fixed-capacity home for the host/path/port/
24 * scheme of a parsed "http[s]://host[:port]/path" string; ocsp_conn_t bundles
25 * the BIO chain and (HTTPS only) the SSL_CTX/SSL so a single owner tears them
26 * down in the required BIO-then-CTX order. The transport helpers build/connect a
27 * responder BIO; do_ocsp_request drives build → connect → send; and
28 * check_ocsp_response verifies the response (status / signature / nonce) and
29 * maps it to GOOD(0) / REVOKED(-1) / UNKNOWN(1). */

38/*
39 * Phase 51 (E1): hard wall-clock bound (seconds) on every network phase of an
40 * OCSP fetch — TCP connect, TLS handshake, and request/response. Without this a
41 * black-holed/slow responder blocks the single-threaded worker event loop for the
42 * kernel TCP timeout (~60-120 s), freezing ALL connections. The connect phase is
43 * bounded with a non-blocking connect + poll() (SO_SNDTIMEO does not bound
44 * connect()); the read/write phases with SO_RCVTIMEO/SO_SNDTIMEO. On timeout the
45 * fetch returns NULL and the caller applies its ocsp.soft_fail policy.
46 */

49/*
50 * ocsp_url_t — parsed components of an OCSP responder URL.
51 *
52 * WHAT: Fixed-capacity home for the host, path, port, and scheme extracted
53 * from an "http[s]://host[:port]/path" string, replacing a 7-parameter
54 * out-argument signature with a single caller-owned struct.
55 * WHY: Keeps parse_ocsp_url() at/under the 5-parameter budget and lets every
56 * consumer carry one value instead of six loose locals — no behaviour change,
57 * identical buffer capacities to the previous char[256]/char[512] locals.
58 * HOW: The struct is stack-allocated by do_ocsp_request(); parse_ocsp_url()
59 * fills it or reports failure. host/path are NUL-terminated C strings.
60 */
61typedef struct {
62 char host[256];
63 char path[512];
64 int port;
65 int use_ssl;
66} ocsp_url_t;

68/*
69 * ocsp_conn_t — live OpenSSL objects for one responder connection.
70 *
71 * WHAT: Holds the BIO chain and (for HTTPS) the SSL_CTX and its SSL handle so a
72 * single owner can tear them all down in the exact required order.
73 * WHY: The connect / handshake / send stages each need cbio (and ssl_ctx for
74 * HTTPS); grouping them lets the builder return one value and lets
75 * do_ocsp_request() keep sole ownership of cleanup — preserving the original
76 * BIO_free_all() → SSL_CTX_free() ordering on every path.
77 * HOW: ssl_ctx / ssl are NULL for plain HTTP. ocsp_conn_free() frees cbio
78 * first, then ssl_ctx, matching the pre-refactor teardown byte-for-byte.
79 */
80typedef struct {
81 BIO *cbio;
82 SSL_CTX *ssl_ctx;
83 SSL *ssl;
84} ocsp_conn_t;
