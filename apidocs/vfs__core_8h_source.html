1/*
2 * vfs_core.h — shared, ngx-free VFS I/O verbs over the storage backend driver.
3 *
4 * WHAT: The storage-neutral byte-I/O verbs (full read, single read, full write,
5 * sync, truncate, fstat) that both the nginx server data plane and the
6 * userland clients run. They operate on an already-opened backend object
7 * (brix_sd_obj_t) — the OPEN, with its per-side policy (server: export-
8 * confined RESOLVE_BENEATH; client: unconfined URL path), stays in the
9 * caller's layer (vfs_server / client adapter). These verbs own only the
10 * EINTR / short-I/O loop policy; the raw syscalls live in the backend.
11 * WHY: The verb loops were maintained twice (src/fs/vfs/vfs_read.c +
12 * vfs_io_core.c on the server; client/lib/vfs_posix.c + vfs_block.c on the
13 * client). One copy removes the drift risk and is the `vfs` layer of the
14 * module->vfs_server->vfs->backend / client->vfs->backend topology.
15 * HOW: Pure C, ngx-free (dual-build via sd.h's XRDPROTO_NO_NGX fallback), in
16 * libxrdproto. Backend-neutral: every op dispatches through obj->driver, so
17 * a non-POSIX backend (block/object) works unchanged.
18 *
19 * See docs/superpowers/specs/2026-06-27-unified-vfs-layering-design.md.
20 */

30 * Convention: all int-returning verbs return 0 on success and -1 on error with

34/*
35 * Full read: loop (EINTR-retried) until `len` bytes are read or EOF. A short
36 * read at EOF is success with *nread < len. *nread (if non-NULL) reports the
37 * byte count even on error. Returns 0 / -1 (errno).
38 */
39int xvfs_pread_full(brix_sd_obj_t *obj, void *buf, size_t len, off_t off,
40 size_t *nread);
41
42/*
43 * Single read with EINTR retry: returns the byte count (>= 0, may be short,
44 * 0 = EOF) or -1 (errno). The caller owns any short-read loop (transfer pump).
45 */
46ssize_t xvfs_pread_once(brix_sd_obj_t *obj, void *buf, size_t len, off_t off);
47
48/*
49 * Full write: loop (EINTR-retried) until all `len` bytes are written. *written
50 * (if non-NULL) reports bytes written; *short_io (if non-NULL) is set when a
51 * 0-byte write or a partial-then-error truncates the write. Returns 0 / -1.
52 */
53int xvfs_pwrite_full(brix_sd_obj_t *obj, const void *buf, size_t len,
54 off_t off, size_t *written, int *short_io);
55
56/* Durability / metadata verbs (single backend op). Return 0 / -1 (errno). */
57int xvfs_fsync(brix_sd_obj_t *obj);
58int xvfs_ftruncate(brix_sd_obj_t *obj, off_t len);
59int xvfs_fstat(brix_sd_obj_t *obj, brix_sd_stat_t *out);
60
61/*
62 * Drain: copy the whole of `src` into `dst` through the driver, chunked via the
63 * caller-provided scratch `buf` (`bufsz` bytes). Positional pread->pwrite reusing
64 * xvfs_pread_once / xvfs_pwrite_full, so EINTR and short writes are handled and
65 * the bytes never leave the backend. *total (if non-NULL) reports bytes copied.
66 * The single primitive every auxiliary-storage scratch copy shares (serve
67 * offload, xvfs_stage_fd). Returns 0 / -1 (errno; EINVAL on a NULL/zero buffer).
68 */
69int xvfs_drain(brix_sd_obj_t *src, brix_sd_obj_t *dst, void *buf,
70 size_t bufsz, off_t *total);
71
72/*
73 * Materialize an already-open source fd into a LOCAL anonymous scratch fd under
74 * `stage_dir`: mkstemp + immediate unlink (the bytes live only behind the fd),
75 * copy src->scratch through the driver (xvfs_drain), then reopen the unlinked
76 * inode O_RDONLY at offset 0 for the consumer. For readers that need a real
77 * kernel fd (random access / sendfile) over a backend object that has none.
78 * Returns the read fd, or -1 (errno). The caller still owns/closes `src_fd`.
79 */
80int xvfs_stage_fd(int src_fd, const char *stage_dir);
