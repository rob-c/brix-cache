7/* File: src/aio/uring_submit.c — io_uring SQE submission (Phase 44)
8 * WHAT: Maps a bound AIO thread task to an io_uring opcode and submits its SQE
9 * (SB-W3: READ/WRITE; SB-W4 extends this with READV and WRITEV+linked
10 * FSYNC). The completion side (slot table + reaper + cqe->res -> OUT
11 * translation) lives in uring.c; this TU owns only submission.
12 *
13 * WHY: Submission is the only place that differs from the thread-pool tier —
14 * the six *_aio_t structs, the *_aio_done callbacks, and every call site
15 * are reused verbatim. Keeping submission in its own TU keeps the seam
16 * small and the reaper independent of opcode prep.
17 *
18 * HOW: brix_uring_op_for() identifies the op by the task's bound worker-fn
19 * pointer. brix_uring_submit() reserves a completion slot, builds the
20 * SQE from the task's IN fields, encodes (generation<<32|slot) into
21 * user_data, and submits; any failure releases the slot and returns
22 * *posted = 0 so the caller falls through to the thread pool. Whole TU
23 * is compiled only when BRIX_HAVE_LIBURING is defined (the config
24 * script lists it in the stream module srcs only in that build). */

28/* Max iovecs in a single readv/writev SQE. Matches the thread-pool coalescer's
29 * preadv cap (BRIX_READV_PREADV_MAXIOV) and stays well under UIO_MAXIOV. */
30#define BRIX_URING_IOV_MAX 64

32/*
33 * brix_readv_is_single_group — true iff the whole readv is one contiguous
34 * same-fd region of 1..IOV_MAX non-empty segments. Only then can one
35 * IORING_OP_READV scatter the contiguous file region into the per-segment
36 * payload buffers; any gap, fd change, zero-length segment, or overflow routes
37 * to the thread pool (which coalesces run-by-run).
38 */
39static ngx_int_t
40brix_readv_is_single_group(brix_readv_aio_t *t)
41{
42 size_t i;
43
44 if (t->segment_count == 0 || t->segment_count > BRIX_URING_IOV_MAX) {
45 return 0;
46 }
47 for (i = 0; i < t->segment_count; i++) {
48 if (t->segments[i].read_length == 0 || t->segments[i].offset < 0) {
49 return 0;
50 }
51 if (t->segments[i].fd != t->segments[0].fd) {
52 return 0;
53 }
54 if (i > 0
55 && t->segments[i].offset != t->segments[i - 1].offset
56 + (off_t) t->segments[i - 1].read_length)
57 {
58 return 0;
59 }
60 }
61 return 1;
62}

65 * brix_writev_is_single_contig_fd — true iff the whole writev is one
66 * contiguous same-fd region of 1..IOV_MAX non-empty segments, so one
67 * IORING_OP_WRITEV can gather the buffers into the contiguous file region.

94/*
95 * io_uring submits raw IORING_OP_READ/WRITE/READV/WRITEV against a real local
96 * file descriptor. A driver-backed handle (obj.driver != NULL — the remote
97 * root:// / HTTP / Ceph / pblock backends) has no such fd: its I/O must run
98 * through the driver's own read/write methods, which only the thread-pool path
99 * reaches (brix_vfs_io_execute). Routing a driver-backed op onto io_uring
100 * would issue a raw syscall on whatever placeholder fd the handle carries —
101 * silently writing bytes nowhere (a remote write-through then lands 0 bytes on
102 * the origin). These predicates keep io_uring to plain POSIX handles; anything
103 * driver-backed returns 0 here and the selector falls through to the pool. */
104static ngx_flag_t
105brix_uring_read_is_posix(void *ctx)
106{
107 return ((brix_read_aio_t *) ctx)->obj.driver == NULL;
108}
109
110static ngx_flag_t
111brix_uring_write_is_posix(void *ctx)
112{
113 return ((brix_write_aio_t *) ctx)->obj.driver == NULL;
114}

144/*
145 * brix_uring_op_for — see uring.h. The worker-fn the task was bound to via
146 * brix_task_bind() is the op identity. readv/writev map to io_uring only for
147 * a single contiguous group; multi-fd/multi-group fall to the pool, as do
148 * pgread (per-page CRC interleave) and dirlist (opendir/readdir). Any
149 * driver-backed handle (non-POSIX backend) also falls to the pool — io_uring
150 * can only touch a real local fd (see the predicates above).
151 */
152brix_uring_op_e
153brix_uring_op_for(ngx_thread_task_t *task)
154{
155 void (*fn)(void *, ngx_log_t *) = task->handler;
156
157 if (fn == brix_read_aio_thread) {
158 return brix_uring_read_is_posix(task->ctx)
159 ? XRD_URING_OP_READ : XRD_URING_OP_NONE;
160 }
161 if (fn == brix_write_aio_thread) {
162 return brix_uring_write_is_posix(task->ctx)
163 ? XRD_URING_OP_WRITE : XRD_URING_OP_NONE;
164 }
165 if (fn == brix_readv_aio_thread) {
166 return (brix_readv_is_single_group(task->ctx)
167 && brix_uring_readv_is_posix(task->ctx))
168 ? XRD_URING_OP_READV : XRD_URING_OP_NONE;
169 }
170 if (fn == brix_writev_write_aio_thread) {
171 return (brix_writev_is_single_contig_fd(task->ctx)
172 && brix_uring_writev_is_posix(task->ctx))
173 ? XRD_URING_OP_WRITEV : XRD_URING_OP_NONE;
174 }
175
176 return XRD_URING_OP_NONE;
177}

193/*
194 * brix_uring_prep_primary — fill the primary SQE from the task's IN fields
195 * and report the fd (for a possible trailing FSYNC). readv/writev build their
196 * iovec from the segment descriptors, allocated from the connection pool so it
197 * stays valid until the CQE (io_uring reads the iovec asynchronously). Returns
198 * NGX_ERROR on an unmapped op or OOM (caller falls back).
199 */
200static ngx_int_t
201brix_uring_prep_primary(struct io_uring_sqe *sqe, brix_uring_op_e op,
202 ngx_thread_task_t *task, int *fd_out)
203{
204 switch (op) {
205
206 case XRD_URING_OP_READ: {
207 brix_read_aio_t *t = task->ctx;
208 io_uring_prep_read(sqe, t->fd, t->databuf, (unsigned) t->rlen,
209 (__u64) t->offset);
210 *fd_out = t->fd;
211 return NGX_OK;
212 }
213
214 case XRD_URING_OP_WRITE: {
215 brix_write_aio_t *t = task->ctx;
216 io_uring_prep_write(sqe, t->fd, t->data, (unsigned) t->len,
217 (__u64) t->offset);
218 *fd_out = t->fd;
219 return NGX_OK;
220 }
221
222 case XRD_URING_OP_READV: {
223 brix_readv_aio_t *t = task->ctx;
224 struct iovec *iov;
225 size_t i;
226
227 iov = brix_palloc_array(t->c->pool, t->segment_count,
228 sizeof(struct iovec));
229 if (iov == NULL) {
230 return NGX_ERROR;
231 }
232 for (i = 0; i < t->segment_count; i++) {
233 iov[i].iov_base = t->segments[i].payload_ptr;
234 iov[i].iov_len = (size_t) t->segments[i].read_length;
235 }
236 io_uring_prep_readv(sqe, t->segments[0].fd, iov,
237 (unsigned) t->segment_count,
238 (__u64) t->segments[0].offset);
239 *fd_out = t->segments[0].fd;
240 return NGX_OK;
241 }
242
243 case XRD_URING_OP_WRITEV: {
244 brix_writev_aio_t *t = task->ctx;
245 struct iovec *iov;
246 size_t i;
247
248 iov = brix_palloc_array(t->c->pool, t->n_segs, sizeof(struct iovec));
249 if (iov == NULL) {
250 return NGX_ERROR;
251 }
252 for (i = 0; i < t->n_segs; i++) {
253 iov[i].iov_base = (void *) t->segs[i].data;
254 iov[i].iov_len = (size_t) t->segs[i].wlen;
255 }
256 io_uring_prep_writev(sqe, t->segs[0].fd, iov, (unsigned) t->n_segs,
257 (__u64) t->segs[0].offset);
258 *fd_out = t->segs[0].fd;
259 return NGX_OK;
260 }
261
262 default:
263 return NGX_ERROR;
264 }
265}

267/*
268 * brix_uring_submit — see uring.h. Reserve a completion slot, build the
269 * primary SQE (plus a linked FSYNC for a do_sync writev), stash the task
270 * identity, encode the UAF-safe cookie, and submit. For the 2-SQE chain the SQ
271 * free space is checked up front so the second get_sqe can never fail mid-build
272 * (no un-prep needed). Every failure path releases the slot and leaves
273 * *posted = 0 so the caller uses the next tier; no op is ever dropped.
274 */
275ngx_int_t
276brix_uring_submit(brix_ctx_t *ctx, ngx_connection_t *c,
277 ngx_thread_task_t *task, brix_uring_op_e op, ngx_flag_t *posted)
278{
279 brix_uring_t *u = brix_uring_worker();
280 brix_uring_slot_t *slot;
281 struct io_uring_sqe *sqe;
282 uint32_t idx;
283 uint32_t nsqe;
284 uint64_t cookie;
285 int fd = -1;
286
287 (void) ctx;
288 (void) c;
289
290 *posted = 0;
291
292 if (u == NULL) {
293 return NGX_OK; /* defensive: selector already checked */
294 }
295
296 nsqe = brix_uring_nsqe(op, task);
297 if (io_uring_sq_space_left(&u->ring) < nsqe) {
298 return NGX_OK; /* SQ too full for the chain -> pool */
299 }
300
301 slot = brix_uring_slot_acquire(u, &idx);
302 if (slot == NULL) {
303 return NGX_OK; /* slot table full -> thread pool */
304 }
305
306 sqe = io_uring_get_sqe(&u->ring); /* non-NULL: space checked above */
307 if (brix_uring_prep_primary(sqe, op, task, &fd) != NGX_OK) {
308 brix_uring_slot_release(u, idx);
309 return NGX_OK;
310 }
311
312 slot->task = task;
313 slot->done_fn = task->event.handler;
314 slot->owner = c; /* orphan key: task lives in c->pool */
315 slot->orphaned = 0;
316 slot->op_kind = (uint8_t) op;
317 /* in_use was set by slot_acquire; generation is current. */
318
319 cookie = ((uint64_t) slot->generation << 32) | idx;
320 io_uring_sqe_set_data64(sqe, cookie);
321
322 if (nsqe == 2) {
323 /* Linked, in-order FSYNC barrier (kXR_writev sync flag). IOSQE_IO_LINK
324 * makes the kernel run the fsync only after the writev succeeds; on a
325 * writev failure the fsync is auto-cancelled. Its CQE carries the
326 * sentinel cookie so the reaper drops it (best-effort, ignored return,
327 * matching the thread-pool path). */
328 struct io_uring_sqe *fsqe;
329
330 io_uring_sqe_set_flags(sqe, IOSQE_IO_LINK);
331 fsqe = io_uring_get_sqe(&u->ring); /* non-NULL: space checked */
332 io_uring_prep_fsync(fsqe, fd, 0);
333 io_uring_sqe_set_data64(fsqe, BRIX_URING_FSYNC_COOKIE);
334 }
335
336 if (io_uring_submit(&u->ring) < 0) {
337 brix_uring_slot_release(u, idx);
338 return NGX_OK;
339 }
340
341 u->inflight += nsqe;
342 *posted = 1;
343 return NGX_OK;
344}
