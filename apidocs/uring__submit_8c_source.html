7/* File: src/aio/uring_submit.c — io_uring SQE submission (Phase 44)
8 * WHAT: Maps a bound AIO thread task to an io_uring opcode and submits its SQE
9 * (SB-W3: READ/WRITE; SB-W4 extends this with READV and WRITEV+linked
10 * FSYNC). The completion side (slot table + reaper + cqe->res -> OUT
11 * translation) lives in uring.c; this TU owns only submission.
12 *
13 * WHY: Submission is the only place that differs from the thread-pool tier —
14 * the six *_aio_t structs, the *_aio_done callbacks, and every call site
15 * are reused verbatim. Keeping submission in its own TU keeps the seam
16 * small and the reaper independent of opcode prep.
17 *
18 * HOW: brix_uring_op_for() identifies the op by the task's bound worker-fn
19 * pointer. brix_uring_submit() reserves a completion slot, builds the
20 * SQE from the task's IN fields, encodes (generation<<32|slot) into
21 * user_data, and submits; any failure releases the slot and returns
22 * *posted = 0 so the caller falls through to the thread pool. Whole TU
23 * is compiled only when BRIX_HAVE_LIBURING is defined (the config
24 * script lists it in the stream module srcs only in that build). */

28/* Max iovecs in a single readv/writev SQE. Matches the thread-pool coalescer's
29 * preadv cap (BRIX_READV_PREADV_MAXIOV) and stays well under UIO_MAXIOV. */
30#define BRIX_URING_IOV_MAX 64
31
32/*
33 * brix_readv_is_single_group — true iff the whole readv is one contiguous
34 * same-fd region of 1..IOV_MAX non-empty segments. Only then can one
35 * IORING_OP_READV scatter the contiguous file region into the per-segment
36 * payload buffers; any gap, fd change, zero-length segment, or overflow routes
37 * to the thread pool (which coalesces run-by-run).
38 */
39static ngx_int_t
40brix_readv_is_single_group(brix_readv_aio_t *t)
41{
42 size_t i;
43
44 if (t->segment_count == 0 || t->segment_count > BRIX_URING_IOV_MAX) {
45 return 0;
46 }
47 for (i = 0; i < t->segment_count; i++) {
48 if (t->segments[i].read_length == 0 || t->segments[i].offset < 0) {
49 return 0;
50 }
51 if (t->segments[i].fd != t->segments[0].fd) {
52 return 0;
53 }
54 if (i > 0
55 && t->segments[i].offset != t->segments[i - 1].offset
56 + (off_t) t->segments[i - 1].read_length)
57 {
58 return 0;
59 }
60 }
61 return 1;
62}

64/*
65 * brix_writev_is_single_contig_fd — true iff the whole writev is one
66 * contiguous same-fd region of 1..IOV_MAX non-empty segments, so one
67 * IORING_OP_WRITEV can gather the buffers into the contiguous file region.
68 */
69static ngx_int_t
70brix_writev_is_single_contig_fd(brix_writev_aio_t *t)

72 size_t i;
73
74 if (t->n_segs == 0 || t->n_segs > BRIX_URING_IOV_MAX) {
75 return 0;
76 }
77 for (i = 0; i < t->n_segs; i++) {
78 if (t->segs[i].wlen == 0 || t->segs[i].offset < 0) {
79 return 0;
80 }
81 if (t->segs[i].fd != t->segs[0].fd) {
82 return 0;
83 }
84 if (i > 0
85 && t->segs[i].offset != t->segs[i - 1].offset
86 + (off_t) t->segs[i - 1].wlen)
87 {
88 return 0;
89 }
90 }
91 return 1;

94/*
95 * brix_uring_op_for — see uring.h. The worker-fn the task was bound to via
96 * brix_task_bind() is the op identity. readv/writev map to io_uring only for
97 * a single contiguous group; multi-fd/multi-group fall to the pool, as do
98 * pgread (per-page CRC interleave) and dirlist (opendir/readdir).
99 */
100brix_uring_op_e
101brix_uring_op_for(ngx_thread_task_t *task)
102{
103 void (*fn)(void *, ngx_log_t *) = task->handler;
104
105 if (fn == brix_read_aio_thread) {
106 return XRD_URING_OP_READ;
107 }
108 if (fn == brix_write_aio_thread) {
109 return XRD_URING_OP_WRITE;
110 }
111 if (fn == brix_readv_aio_thread) {
112 return brix_readv_is_single_group(task->ctx)
113 ? XRD_URING_OP_READV : XRD_URING_OP_NONE;
114 }
115 if (fn == brix_writev_write_aio_thread) {
116 return brix_writev_is_single_contig_fd(task->ctx)
117 ? XRD_URING_OP_WRITEV : XRD_URING_OP_NONE;
118 }
119
120 return XRD_URING_OP_NONE;
121}

124 * brix_uring_nsqe — how many SQEs (and in-flight CQEs) the op consumes: 2 for
125 * a WRITEV whose do_sync links a trailing FSYNC, 1 otherwise.

129{

137/*
138 * brix_uring_prep_primary — fill the primary SQE from the task's IN fields
139 * and report the fd (for a possible trailing FSYNC). readv/writev build their
140 * iovec from the segment descriptors, allocated from the connection pool so it
141 * stays valid until the CQE (io_uring reads the iovec asynchronously). Returns
142 * NGX_ERROR on an unmapped op or OOM (caller falls back).
143 */
144static ngx_int_t
145brix_uring_prep_primary(struct io_uring_sqe *sqe, brix_uring_op_e op,
146 ngx_thread_task_t *task, int *fd_out)
147{
148 switch (op) {
149
150 case XRD_URING_OP_READ: {
151 brix_read_aio_t *t = task->ctx;
152 io_uring_prep_read(sqe, t->fd, t->databuf, (unsigned) t->rlen,
153 (__u64) t->offset);
154 *fd_out = t->fd;
155 return NGX_OK;
156 }
157
158 case XRD_URING_OP_WRITE: {
159 brix_write_aio_t *t = task->ctx;
160 io_uring_prep_write(sqe, t->fd, t->data, (unsigned) t->len,
161 (__u64) t->offset);
162 *fd_out = t->fd;
163 return NGX_OK;
164 }
165
166 case XRD_URING_OP_READV: {
167 brix_readv_aio_t *t = task->ctx;
168 struct iovec *iov;
169 size_t i;
170
171 iov = brix_palloc_array(t->c->pool, t->segment_count,
172 sizeof(struct iovec));
173 if (iov == NULL) {
174 return NGX_ERROR;
175 }
176 for (i = 0; i < t->segment_count; i++) {
177 iov[i].iov_base = t->segments[i].payload_ptr;
178 iov[i].iov_len = (size_t) t->segments[i].read_length;
179 }
180 io_uring_prep_readv(sqe, t->segments[0].fd, iov,
181 (unsigned) t->segment_count,
182 (__u64) t->segments[0].offset);
183 *fd_out = t->segments[0].fd;
184 return NGX_OK;
185 }
186
187 case XRD_URING_OP_WRITEV: {
188 brix_writev_aio_t *t = task->ctx;
189 struct iovec *iov;
190 size_t i;
191
192 iov = brix_palloc_array(t->c->pool, t->n_segs, sizeof(struct iovec));
193 if (iov == NULL) {
194 return NGX_ERROR;
195 }
196 for (i = 0; i < t->n_segs; i++) {
197 iov[i].iov_base = (void *) t->segs[i].data;
198 iov[i].iov_len = (size_t) t->segs[i].wlen;
199 }
200 io_uring_prep_writev(sqe, t->segs[0].fd, iov, (unsigned) t->n_segs,
201 (__u64) t->segs[0].offset);
202 *fd_out = t->segs[0].fd;
203 return NGX_OK;
204 }
205
206 default:
207 return NGX_ERROR;
208 }

211/*
212 * brix_uring_submit — see uring.h. Reserve a completion slot, build the
213 * primary SQE (plus a linked FSYNC for a do_sync writev), stash the task
214 * identity, encode the UAF-safe cookie, and submit. For the 2-SQE chain the SQ
215 * free space is checked up front so the second get_sqe can never fail mid-build
216 * (no un-prep needed). Every failure path releases the slot and leaves
217 * *posted = 0 so the caller uses the next tier; no op is ever dropped.
218 */
219ngx_int_t
220brix_uring_submit(brix_ctx_t *ctx, ngx_connection_t *c,
221 ngx_thread_task_t *task, brix_uring_op_e op, ngx_flag_t *posted)
222{
223 brix_uring_t *u = brix_uring_worker();
224 brix_uring_slot_t *slot;
225 struct io_uring_sqe *sqe;
226 uint32_t idx;
227 uint32_t nsqe;
228 uint64_t cookie;
229 int fd = -1;
230
231 (void) ctx;
232 (void) c;
233
234 *posted = 0;
235
236 if (u == NULL) {
237 return NGX_OK; /* defensive: selector already checked */
238 }
239
240 nsqe = brix_uring_nsqe(op, task);
241 if (io_uring_sq_space_left(&u->ring) < nsqe) {
242 return NGX_OK; /* SQ too full for the chain -> pool */
243 }
244
245 slot = brix_uring_slot_acquire(u, &idx);
246 if (slot == NULL) {
247 return NGX_OK; /* slot table full -> thread pool */
248 }
249
250 sqe = io_uring_get_sqe(&u->ring); /* non-NULL: space checked above */
251 if (brix_uring_prep_primary(sqe, op, task, &fd) != NGX_OK) {
252 brix_uring_slot_release(u, idx);
253 return NGX_OK;
254 }
255
256 slot->task = task;
257 slot->done_fn = task->event.handler;
258 slot->op_kind = (uint8_t) op;
259 /* in_use was set by slot_acquire; generation is current. */
260
261 cookie = ((uint64_t) slot->generation << 32) | idx;
262 io_uring_sqe_set_data64(sqe, cookie);
263
264 if (nsqe == 2) {
265 /* Linked, in-order FSYNC barrier (kXR_writev sync flag). IOSQE_IO_LINK
266 * makes the kernel run the fsync only after the writev succeeds; on a
267 * writev failure the fsync is auto-cancelled. Its CQE carries the
268 * sentinel cookie so the reaper drops it (best-effort, ignored return,
269 * matching the thread-pool path). */
270 struct io_uring_sqe *fsqe;
271
272 io_uring_sqe_set_flags(sqe, IOSQE_IO_LINK);
273 fsqe = io_uring_get_sqe(&u->ring); /* non-NULL: space checked */
274 io_uring_prep_fsync(fsqe, fd, 0);
275 io_uring_sqe_set_data64(fsqe, BRIX_URING_FSYNC_COOKIE);
276 }
277
278 if (io_uring_submit(&u->ring) < 0) {
279 brix_uring_slot_release(u, idx);
280 return NGX_OK;
281 }
282
283 u->inflight += nsqe;
284 *posted = 1;
285 return NGX_OK;
286}
