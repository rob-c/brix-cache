1/*
2 * codec_core.c — codec descriptor table + stream lifecycle + bomb guard.
3 *
4 * WHAT: Implements brix_codec_open/step/close and the by_id/by_name/
5 * by_http_token lookups over a static descriptor table, plus the always-on
6 * IDENTITY (passthrough) backend. Per-codec backends live in codec_<name>.c
7 * and are referenced here by extern descriptor.
8 * WHY: Single dispatch + single place to enforce the decompression-bomb guard,
9 * so every untrusted decode path is protected identically (see codec_core.h).
10 * HOW: Backends always define their descriptor (available=0 when their lib is
11 * absent), so the table is dense and dispatch is a bounds-checked index.
12 * brix_codec_step() wraps the backend step and updates/enforces the guard.
13 */

36static brix_codec_rc_t
37identity_step(void *state, const uint8_t *in, size_t in_len, size_t *in_pos,
38 uint8_t *out, size_t out_size, size_t *out_pos, int finish)
39{
40 size_t avail = in_len - *in_pos;
41 size_t room = out_size - *out_pos;
42 size_t n = (avail < room) ? avail : room;
43
44 (void) state;
45 if (n) {
46 memcpy(out + *out_pos, in + *in_pos, n);
47 *in_pos += n;
48 *out_pos += n;
49 }
50 if (*in_pos == in_len && finish) {
51 return BRIX_CODEC_END;
52 }
53 return BRIX_CODEC_OK;
54}

59 (void) state;

100static int
101codec_ci_eq(const char *a, const char *b, size_t n)
102{
103 size_t i;
104
105 for (i = 0; i < n; i++) {
106 unsigned char ca = (unsigned char) a[i];
107 unsigned char cb = (unsigned char) b[i];
108 if (ca >= 'A' && ca <= 'Z') { ca = (unsigned char) (ca + 32); }
109 if (cb >= 'A' && cb <= 'Z') { cb = (unsigned char) (cb + 32); }
110 if (ca != cb) {
111 return 0;
112 }
113 }
114 return 1;
115}

117static const brix_codec_desc_t *
118codec_match(const char *s, size_t len, int by_token)
119{
120 size_t i;
121
122 if (s == NULL || len == 0) {
123 return NULL;
124 }
125 /* HTTP Content-Encoding tokens are case-insensitive; canonical names match
126 * exactly (config values). */
127 for (i = 0; i < BRIX_CODEC_MAX; i++) {
128 const brix_codec_desc_t *d = codec_table[i];
129 const char *key = by_token ? d->http_token : d->name;
130 if (key != NULL && strlen(key) == len
131 && (by_token ? codec_ci_eq(key, s, len) : memcmp(key, s, len) == 0)) {
132 return d;
133 }
134 }
135 return NULL;
136}

157/* stream lifecycle */
158brix_codec_stream_t *
159brix_codec_open(brix_codec_id_t id, brix_codec_dir_t dir, int level,
160 const brix_codec_guard_t *guard)
161{
162 const brix_codec_desc_t *d = brix_codec_by_id(id);
163 brix_codec_stream_t *s;
164 int lvl;
165
166 if (d == NULL || !d->available || d->backend == NULL) {
167 return NULL;
168 }
169
170 lvl = (level < 0) ? d->level_default : level;
171 if (lvl < d->level_min) { lvl = d->level_min; }
172 if (lvl > d->level_max) { lvl = d->level_max; }
173
174 s = calloc(1, sizeof(*s));
175 if (s == NULL) {
176 return NULL;
177 }
178 s->desc = d;
179 s->dir = dir;
180 if (guard != NULL) {
181 s->guard = *guard;
182 }
183 if (d->backend->init(&s->state, id, dir, lvl) != 0) {
184 free(s);
185 return NULL;
186 }
187 return s;
188}

190brix_codec_rc_t
191brix_codec_step(brix_codec_stream_t *s,
192 const uint8_t *in, size_t in_len, size_t *in_pos,
193 uint8_t *out, size_t out_size, size_t *out_pos, int finish)
194{
195 size_t in0, out0;
196 brix_codec_rc_t rc;
197
198 if (s == NULL || in_pos == NULL || out_pos == NULL
199 || *in_pos > in_len || *out_pos > out_size) {
200 return BRIX_CODEC_ERR_PARAM;
201 }
202
203 in0 = *in_pos;
204 out0 = *out_pos;
205 rc = s->desc->backend->step(s->state, in, in_len, in_pos,
206 out, out_size, out_pos, finish);
207 if (rc < 0) {
208 return rc;
209 }
210
211 s->guard.total_in += (uint64_t) (*in_pos - in0);
212 s->guard.total_out += (uint64_t) (*out_pos - out0);
213
214 if (s->dir == BRIX_CODEC_DIR_DECOMPRESS) {
215 if (s->guard.out_cap != 0 && s->guard.total_out > s->guard.out_cap) {
216 return BRIX_CODEC_ERR_BOMB;
217 }
218 if (s->guard.max_ratio != 0
219 && s->guard.total_in > 0
220 && s->guard.total_out >= BRIX_CODEC_RATIO_FLOOR
221 && s->guard.total_out / s->guard.total_in > (uint64_t) s->guard.max_ratio) {
222 return BRIX_CODEC_ERR_BOMB;
223 }
224 }
225 return rc;
226}
