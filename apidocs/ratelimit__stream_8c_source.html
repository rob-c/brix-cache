1/*
2 * ratelimit_stream.c — Phase 25 XRootD stream enforcement.
3 *
4 * The dispatch gate runs before each data-plane opcode handler. Stat/ping/
5 * close/sync and the session opcodes are never rate-limited (so keepalive and
6 * health checks are unaffected). A throttled request is answered with
7 * kXR_wait(seconds) and the connection stays open for the client to retry;
8 * the gate returns the send result so the dispatcher skips the handler.
9 *
10 * Returns NGX_DECLINED to proceed with normal dispatch.
11 */

24static int
25rl_op_rate_limited(uint16_t reqid)
26{
27 switch (reqid) {
28 case kXR_open: case kXR_read: case kXR_readv: case kXR_pgread:
29 case kXR_write: case kXR_writev: case kXR_pgwrite:
30 case kXR_dirlist: case kXR_locate:
31 return 1;
32 default:
33 return 0; /* stat/statx/ping/close/sync/login/auth: never limited */
34 }
35}

50/* Copy the request path (payload up to '?' / NUL) into buf for a path-bearing
51 * opcode; sets buf[0]='\0' otherwise. */
52static void
53rl_request_path(brix_ctx_t *ctx, char *buf, size_t bufsz)
54{
55 size_t n, i;
56
57 buf[0] = '\0';
58 if (!rl_op_path_bearing(ctx->recv.cur_reqid)
59 || ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0)
60 {
61 return;
62 }
63 n = (ctx->recv.cur_dlen < bufsz - 1) ? ctx->recv.cur_dlen : bufsz - 1;
64 for (i = 0; i < n; i++) {
65 u_char ch = ctx->recv.payload[i];
66 if (ch == '\0' || ch == '?' || ch == '\n') { break; }
67 buf[i] = (char) ch;
68 }
69 buf[i] = '\0';
70}

72/* ---- Resolve (and cache) the rate-limit key for one rule ----
73 *
74 * WHAT: Returns the enforcement key for rule i — either a cached
75 * connection-constant key or a freshly computed one written into key_str.
76 * Returns NULL when the rule does not match this request (VOLUME prefix miss
77 * or key computation error), signalling the caller to skip the rule.
78 *
79 * WHY: Phase 33 C4 hot-path optimisation. Identity-stable rules (VO/ISSUER/
80 * IP/DN) produce a connection-constant key; caching the first
81 * BRIX_RL_RULE_CACHE_MAX of them on the ctx removes a per-read re-hash. VOLUME
82 * rules are path-dependent and rules beyond the cache bound always recompute.
83 * Isolating this keeps the gate loop free of the cache bookkeeping while
84 * preserving the exact match/skip decision.
85 *
86 * HOW:
87 * 1. A rule is cacheable when it is not a VOLUME rule and its index is below
88 * the cache bound.
89 * 2. If cacheable and the cache slot is valid, return the cached key.
90 * 3. Otherwise compute the key; on non-NGX_OK return NULL (rule skipped).
91 * 4. If cacheable, populate and validate the cache slot, then return the key.
92 */
93static const char *
94rl_resolve_key(brix_ctx_t *ctx, brix_rl_rule_t *rule, ngx_uint_t i,
95 char *path, char *key_str, size_t key_str_sz)
96{
97 int cacheable;
98
99 cacheable = (rule->key_type != BRIX_RL_KEY_VOLUME
100 && i < BRIX_RL_RULE_CACHE_MAX);
101
102 if (cacheable && (ctx->rl.key_cache_valid & (1u << i))) {
103 return ctx->rl.key_cache[i];
104 }
105
106 if (brix_rl_key_stream(rule, ctx, path, key_str, key_str_sz) != NGX_OK) {
107 return NULL; /* VOLUME prefix miss or error */
108 }
109
110 if (cacheable) {
111 ngx_cpystrn((u_char *) ctx->rl.key_cache[i], (u_char *) key_str,
112 sizeof(ctx->rl.key_cache[i]));
113 ctx->rl.key_cache_valid |= (1u << i);
114 }
115 return key_str;
116}

118/* ---- Enforce the request-rate dimension of one rule ----
119 *
120 * WHAT: If the rule caps request rate and this request exceeds it, emits the
121 * throttle metric+log, stores the kXR_wait send result in *out_rc and returns
122 * 1 (caller must return *out_rc). Returns 0 to proceed otherwise.
123 *
124 * WHY: Preserves the admit/reject decision of the original inline block — the
125 * rate check runs only when req_rate > 0, and a throttled request answers with
126 * kXR_wait(seconds) and stops dispatch.
127 *
128 * HOW:
129 * 1. If req_rate > 0 and brix_rl_check reports NGX_AGAIN, throttle.
130 * 2. Increment rl_throttled_stream_total, log the wait at INFO.
131 * 3. Send kXR_wait(wait_sec) into *out_rc and return 1.
132 * 4. Otherwise return 0.
133 */
134static int
135rl_enforce_req_rate(brix_ctx_t *ctx, ngx_connection_t *c,
136 brix_rl_rule_t *rule, const char *key, ngx_int_t *out_rc)
137{
138 uint32_t wait_sec;
139
140 if (rule->req_rate > 0
141 && brix_rl_check(rule, key, &wait_sec) == NGX_AGAIN)
142 {
143 BRIX_RL_METRIC_INC(rl_throttled_stream_total);
144 ngx_log_error(NGX_LOG_INFO, c->log, 0,
145 "xrootd rate limit: kXR_wait %uD for op 0x%04xd (key=%s)",
146 wait_sec, (int) ctx->recv.cur_reqid, key);
147 *out_rc = brix_send_wait(ctx, c, wait_sec);
148 return 1;
149 }
150 return 0;
151}

153/* ---- Enforce the bandwidth dimension of one rule ----
154 *
155 * WHAT: If the rule caps bandwidth and this request is over budget, throttles
156 * (metric+log, *out_rc = kXR_wait, return 1). If within budget, records the
157 * matched rule+key as the post-send charge target and returns 0. Rules with no
158 * bandwidth cap return 0.
159 *
160 * WHY: Preserves the original decision AND its side effect: a bandwidth match
161 * that is not throttled must remember the rule so brix_rl_charge_ctx() charges
162 * the transferred bytes after the send.
163 *
164 * HOW:
165 * 1. If the rule has no bandwidth cap, return 0.
166 * 2. If brix_rl_bw_check reports NGX_AGAIN, throttle and return 1.
167 * 3. Otherwise store bw_key/bw_rule as the charge target and return 0.
168 */
169static int
170rl_enforce_bw_rate(brix_ctx_t *ctx, ngx_connection_t *c,
171 brix_rl_rule_t *rule, const char *key, ngx_int_t *out_rc)
172{
173 uint32_t wait_sec;
174
175 if (rule->bw_rate <= 0) {
176 return 0;
177 }
178
179 if (brix_rl_bw_check(rule, key, &wait_sec) == NGX_AGAIN) {
180 BRIX_RL_METRIC_INC(rl_throttled_stream_total);
181 ngx_log_error(NGX_LOG_INFO, c->log, 0,
182 "xrootd rate limit: kXR_wait %uD (bandwidth, key=%s)",
183 wait_sec, key);
184 *out_rc = brix_send_wait(ctx, c, wait_sec);
185 return 1;
186 }
187
188 /* Remember the matched bandwidth rule for the post-send charge. */
189 ngx_cpystrn((u_char *) ctx->rl.bw_key, (u_char *) key,
190 sizeof(ctx->rl.bw_key));
191 ctx->rl.bw_rule = rule;
192 return 0;
193}

195/* ---- Enforce the concurrency dimension of one rule ----
196 *
197 * WHAT: Reserves one in-flight connection slot for this principal on the first
198 * matching concurrency rule. Over-cap throttles (*out_rc = kXR_wait(1),
199 * return 1); on success records the slot for release and returns 0. Returns 0
200 * when the rule has no concurrency cap or a slot is already held.
201 *
202 * WHY: Concurrency dimension (W7, stream). The stream plane has no per-request
203 * teardown, so the slot is held for the CONNECTION's lifetime — acquired once
204 * on the first matching rule and released exactly once in brix_on_disconnect()
205 * via brix_rl_release_ctx(). This caps concurrent connections per principal;
206 * over-cap answers kXR_wait so the client retries when a slot frees. The
207 * once-per-connection guard (conc_rule == NULL) must be preserved exactly.
208 *
209 * HOW:
210 * 1. If no concurrency cap or a slot is already held, return 0.
211 * 2. If brix_rl_conc_acquire reports NGX_AGAIN, throttle and return 1.
212 * 3. Otherwise record conc_key/conc_rule for later release and return 0.
213 */
214static int
215rl_enforce_conc(brix_ctx_t *ctx, ngx_connection_t *c,
216 brix_rl_rule_t *rule, const char *key, ngx_int_t *out_rc)
217{
218 if (rule->req_conc <= 0 || ctx->rl.conc_rule != NULL) {
219 return 0;
220 }
221
222 if (brix_rl_conc_acquire(rule, key) == NGX_AGAIN) {
223 BRIX_RL_METRIC_INC(rl_throttled_stream_total);
224 ngx_log_error(NGX_LOG_INFO, c->log, 0,
225 "xrootd rate limit: kXR_wait (concurrency cap, key=%s)",
226 key);
227 *out_rc = brix_send_wait(ctx, c, 1);
228 return 1;
229 }
230
231 ngx_cpystrn((u_char *) ctx->rl.conc_key, (u_char *) key,
232 sizeof(ctx->rl.conc_key));
233 ctx->rl.conc_rule = rule;
234 return 0;
235}

237/* ---- Determine whether any VOLUME rule needs the request path ----
238 *
239 * WHAT: Copies the request path into path (via rl_request_path) when at least
240 * one configured rule is a VOLUME (path-prefix) rule; otherwise leaves
241 * path[0] = '\0'.
242 *
243 * WHY: Phase 33 C4: the request path is only needed to match VOLUME rules.
244 * Reads — the hot path — are not path-bearing and have no VOLUME match, so the
245 * payload copy is skipped entirely unless a VOLUME rule exists.
246 *
247 * HOW:
248 * 1. Initialise path to empty.
249 * 2. Scan rules; on the first VOLUME rule copy the request path and stop.
250 */
251static void
252rl_maybe_load_path(brix_ctx_t *ctx, brix_rl_rule_t *rules, ngx_uint_t nelts,
253 char *path, size_t path_sz)
254{
255 ngx_uint_t i;
256
257 path[0] = '\0';
258 for (i = 0; i < nelts; i++) {
259 if (rules[i].key_type == BRIX_RL_KEY_VOLUME) {
260 rl_request_path(ctx, path, path_sz);
261 break;
262 }
263 }

266ngx_int_t
267brix_rl_stream_gate(brix_ctx_t *ctx, ngx_connection_t *c,
268 ngx_stream_brix_srv_conf_t *conf)
269{
270 brix_rl_rule_t *rules;
271 ngx_uint_t i;
272 char key_str[BRIX_RL_KEY_LEN];
273 char path[1024];
274 ngx_int_t rc;
275
276 if (conf->rl_rules == NULL || conf->rl_rules->nelts == 0) {
277 return NGX_DECLINED;
278 }
279 if (!rl_op_rate_limited(ctx->recv.cur_reqid)) {
280 return NGX_DECLINED;
281 }
282
283 /* Reset any stale bandwidth charge target from the previous request. */
284 ctx->rl.bw_rule = NULL;
285
286 rules = conf->rl_rules->elts;
287 rl_maybe_load_path(ctx, rules, conf->rl_rules->nelts, path, sizeof(path));
288
289 for (i = 0; i < conf->rl_rules->nelts; i++) {
290 const char *key;
291
292 key = rl_resolve_key(ctx, &rules[i], i, path, key_str,
293 sizeof(key_str));
294 if (key == NULL) { continue; } /* rule does not match this request */
295
296 if (rl_enforce_req_rate(ctx, c, &rules[i], key, &rc)) { return rc; }
297 if (rl_enforce_bw_rate(ctx, c, &rules[i], key, &rc)) { return rc; }
298 if (rl_enforce_conc(ctx, c, &rules[i], key, &rc)) { return rc; }
299 }
300
301 return NGX_DECLINED;
302}
