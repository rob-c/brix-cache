1/*
2 * ratelimit_stream.c — Phase 25 XRootD stream enforcement.
3 *
4 * The dispatch gate runs before each data-plane opcode handler. Stat/ping/
5 * close/sync and the session opcodes are never rate-limited (so keepalive and
6 * health checks are unaffected). A throttled request is answered with
7 * kXR_wait(seconds) and the connection stays open for the client to retry;
8 * the gate returns the send result so the dispatcher skips the handler.
9 *
10 * Returns NGX_DECLINED to proceed with normal dispatch.
11 */

23/* Which opcodes carry request-count rate limiting. */
24static int
25rl_op_rate_limited(uint16_t reqid)
26{
27 switch (reqid) {
28 case kXR_open: case kXR_read: case kXR_readv: case kXR_pgread:
29 case kXR_write: case kXR_writev: case kXR_pgwrite:
30 case kXR_dirlist: case kXR_locate:
31 return 1;
32 default:
33 return 0; /* stat/statx/ping/close/sync/login/auth: never limited */
34 }
35}

37/* True for opcodes whose request payload is the (raw) path — usable for
38 * VOLUME-prefix matching. Handle-bearing read/write ops are excluded. */
39static int
40rl_op_path_bearing(uint16_t reqid)
41{
42 switch (reqid) {
43 case kXR_open: case kXR_dirlist: case kXR_locate:
44 return 1;
45 default:
46 return 0;
47 }
48}

50/* Copy the request path (payload up to '?' / NUL) into buf for a path-bearing
51 * opcode; sets buf[0]='\0' otherwise. */
52static void
53rl_request_path(brix_ctx_t *ctx, char *buf, size_t bufsz)
54{
55 size_t n, i;
56
57 buf[0] = '\0';
58 if (!rl_op_path_bearing(ctx->recv.cur_reqid)
59 || ctx->recv.payload == NULL || ctx->recv.cur_dlen == 0)
60 {
61 return;
62 }
63 n = (ctx->recv.cur_dlen < bufsz - 1) ? ctx->recv.cur_dlen : bufsz - 1;
64 for (i = 0; i < n; i++) {
65 u_char ch = ctx->recv.payload[i];
66 if (ch == '\0' || ch == '?' || ch == '\n') { break; }
67 buf[i] = (char) ch;
68 }
69 buf[i] = '\0';
70}

72ngx_int_t
73brix_rl_stream_gate(brix_ctx_t *ctx, ngx_connection_t *c,
74 ngx_stream_brix_srv_conf_t *conf)
75{
76 brix_rl_rule_t *rules;
77 ngx_uint_t i;
78 char key_str[BRIX_RL_KEY_LEN];
79 char path[1024];
80 uint32_t wait_sec;
81 ngx_int_t rc;
82
83 if (conf->rl_rules == NULL || conf->rl_rules->nelts == 0) {
84 return NGX_DECLINED;
85 }
86 if (!rl_op_rate_limited(ctx->recv.cur_reqid)) {
87 return NGX_DECLINED;
88 }
89
90 /* Reset any stale bandwidth charge target from the previous request. */
91 ctx->rl.bw_rule = NULL;
92
93 rules = conf->rl_rules->elts;
94
95 /*
96 * Phase 33 C4: the request path is only needed to match VOLUME (path-prefix)
97 * rules. Reads — the hot path — are not path-bearing and have no VOLUME
98 * match, so skip the payload copy entirely unless a VOLUME rule exists.
99 */
100 path[0] = '\0';
101 for (i = 0; i < conf->rl_rules->nelts; i++) {
102 if (rules[i].key_type == BRIX_RL_KEY_VOLUME) {
103 rl_request_path(ctx, path, sizeof(path));
104 break;
105 }
106 }
107
108 for (i = 0; i < conf->rl_rules->nelts; i++) {
109 const char *key;
110 int cacheable;
111
112 /*
113 * Phase 33 C4: identity-stable rules (VO/ISSUER/IP/DN) produce a
114 * connection-constant key. Cache the first BRIX_RL_RULE_CACHE_MAX such
115 * keys on the ctx and reuse them, removing the per-read re-hash. VOLUME
116 * rules are path-dependent and rules beyond the cache bound recompute.
117 */
118 cacheable = (rules[i].key_type != BRIX_RL_KEY_VOLUME
119 && i < BRIX_RL_RULE_CACHE_MAX);
120
121 if (cacheable && (ctx->rl.key_cache_valid & (1u << i))) {
122 key = ctx->rl.key_cache[i];
123 } else {
124 rc = brix_rl_key_stream(&rules[i], ctx, path,
125 key_str, sizeof(key_str));
126 if (rc != NGX_OK) { continue; } /* VOLUME prefix miss or error */
127 key = key_str;
128 if (cacheable) {
129 ngx_cpystrn((u_char *) ctx->rl.key_cache[i], (u_char *) key_str,
130 sizeof(ctx->rl.key_cache[i]));
131 ctx->rl.key_cache_valid |= (1u << i);
132 }
133 }
134
135 if (rules[i].req_rate > 0) {
136 if (brix_rl_check(&rules[i], key, &wait_sec) == NGX_AGAIN) {
137 BRIX_RL_METRIC_INC(rl_throttled_stream_total);
138 ngx_log_error(NGX_LOG_INFO, c->log, 0,
139 "xrootd rate limit: kXR_wait %uD for op 0x%04xd (key=%s)",
140 wait_sec, (int) ctx->recv.cur_reqid, key);
141 return brix_send_wait(ctx, c, wait_sec);
142 }
143 }
144
145 if (rules[i].bw_rate > 0) {
146 if (brix_rl_bw_check(&rules[i], key, &wait_sec) == NGX_AGAIN) {
147 BRIX_RL_METRIC_INC(rl_throttled_stream_total);
148 ngx_log_error(NGX_LOG_INFO, c->log, 0,
149 "xrootd rate limit: kXR_wait %uD (bandwidth, key=%s)",
150 wait_sec, key);
151 return brix_send_wait(ctx, c, wait_sec);
152 }
153 /* Remember the matched bandwidth rule for the post-send charge. */
154 ngx_cpystrn((u_char *) ctx->rl.bw_key, (u_char *) key,
155 sizeof(ctx->rl.bw_key));
156 ctx->rl.bw_rule = &rules[i];
157 }
158
159 /*
160 * Concurrency dimension (W7, stream): reserve one in-flight slot for
161 * this connection's principal. Unlike the HTTP plane (one slot per
162 * request, released in the LOG phase) the stream plane has no per-request
163 * teardown, so the slot is held for the CONNECTION's lifetime — acquired
164 * once on the first matching rule and released exactly once in
165 * brix_on_disconnect() via brix_rl_release_ctx(). This caps the
166 * number of concurrent connections per principal; over-cap → kXR_wait so
167 * the client retries when a slot frees.
168 */
169 if (rules[i].req_conc > 0 && ctx->rl.conc_rule == NULL) {
170 if (brix_rl_conc_acquire(&rules[i], key) == NGX_AGAIN) {
171 BRIX_RL_METRIC_INC(rl_throttled_stream_total);
172 ngx_log_error(NGX_LOG_INFO, c->log, 0,
173 "xrootd rate limit: kXR_wait (concurrency cap, key=%s)",
174 key);
175 return brix_send_wait(ctx, c, 1);
176 }
177 ngx_cpystrn((u_char *) ctx->rl.conc_key, (u_char *) key,
178 sizeof(ctx->rl.conc_key));
179 ctx->rl.conc_rule = &rules[i];
180 }
181 }
182
183 return NGX_DECLINED;
184}
