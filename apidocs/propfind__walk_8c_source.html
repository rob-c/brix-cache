44/* Hard ceiling on entries emitted for a Depth: infinity PROPFIND so a deep or

47/*
48 * WHAT: Immutable per-walk context — the request, output chain, request-props,
49 * entry cap/counter and recursion flag that every level of the walk shares.
50 * WHY: propfind_walk's signature is frozen (declared extern in
51 * propfind_internal.h with 9 params), but its helpers must not each re-take
52 * that many arguments; bundling the shared state into one file-local struct
53 * keeps each helper at =5 params and makes the data flow explicit.
54 * HOW: Populated once at the top of propfind_walk and passed by pointer to the
55 * per-entry helpers; entry_count is a live pointer so nested levels update a
56 * single shared counter.
57 */
58typedef struct {
59 ngx_http_request_t *r;
60 ngx_chain_t **head;
61 ngx_chain_t **tail;
62 const propfind_req_t *req;
63 ngx_uint_t *entry_count;
64 ngx_uint_t max_entries;
65 ngx_flag_t recurse;
66} propfind_walk_ctx_t;

68/*
69 * WHAT: Decide whether a readdir entry is emitted at all.
70 * WHY: readdir already drops "."/".."; PROPFIND additionally hides every other
71 * dotfile and every internal metadata/staging artifact (sidecars like
72 * "f.dat.cinfo" or upload temps are not dotfiles, so the leading-'.' test
73 * misses them and the suffix/infix predicate catches them).
74 * HOW: Pure predicate over the entry name; returns 1 to keep, 0 to skip.
75 */
76static int
77propfind_child_visible(const ngx_str_t *name)
78{
79 if (name->data[0] == '.') {
80 return 0;
81 }
82 if (brix_is_internal_name((const char *) name->data)) {
83 return 0;
84 }
85 return 1;
86}

88/*
89 * WHAT: Build the child href = base_href + name into out (size out_sz).
90 * WHY: A single '/' separator is inserted only when base_href is not already
91 * slash-terminated; a truncated result must be discarded rather than emitted
92 * as a corrupted href.
93 * HOW: snprintf returns the would-be length; >= out_sz means truncation, in
94 * which case NGX_ERROR is returned so the caller skips the entry (NGX_OK on
95 * success — bytes identical to the original inline logic).
96 */
97static ngx_int_t
98propfind_build_child_href(const char *base_href, const char *name,
99 char *out, size_t out_sz)
100{
101 size_t blen = strlen(base_href);
102 size_t n;
103
104 if (blen == 0 || base_href[blen - 1] != '/') {
105 n = (size_t) snprintf(out, out_sz, "%s/%s", base_href, name);
106 } else {
107 n = (size_t) snprintf(out, out_sz, "%s%s", base_href, name);
108 }
109 return (n >= out_sz) ? NGX_ERROR : NGX_OK;
110}

112/*
113 * WHAT: Recurse into a subdirectory child during a Depth: infinity walk.
114 * WHY: Only directories are descended, only when recursion is enabled, and only
115 * while the cap has not been hit (re-checked here so a balanced tree cannot
116 * blow past max_entries via nested calls before the per-entry check fires).
117 * HOW: Builds the trailing-slash subdir href; a truncated href is skipped
118 * (NGX_OK, matching the original). Delegates to propfind_walk and propagates
119 * its NGX_ERROR so the caller can close the open DIR without an fd leak.
120 */
121static ngx_int_t
122propfind_recurse_child(const propfind_walk_ctx_t *w,
123 const char *child_path, const char *child_href, const struct stat *csb)
124{
125 char subdir_href[WEBDAV_MAX_PATH + 3];
126
127 if (!w->recurse || !S_ISDIR(csb->st_mode)
128 || *w->entry_count >= w->max_entries)
129 {
130 return NGX_OK;
131 }
132 if ((size_t) snprintf(subdir_href, sizeof(subdir_href), "%s/", child_href)
133 >= sizeof(subdir_href))
134 {
135 return NGX_OK;
136 }
137 return propfind_walk(w->r, w->head, w->tail, child_path, subdir_href,
138 w->entry_count, w->max_entries, w->req, w->recurse);
139}

141/*
142 * WHAT: Emit the D:response for one visible readdir entry and recurse into it
143 * if it is a subdirectory.
144 * WHY: Splits the per-entry work (join path, adapt stat, build href, emit,
145 * count, recurse) out of the enumeration loop so neither exceeds the CCN cap;
146 * the visibility/cap checks stay in the loop, this handles a kept entry.
147 * HOW: A path/href that would truncate is skipped (NGX_OK, no entry emitted),
148 * exactly as before. propfind_entry increments the shared counter on success;
149 * a propfind_entry / recursion failure returns NGX_ERROR so the caller closes
150 * the DIR before propagating it.
151 */
152static ngx_int_t
153propfind_walk_emit(const propfind_walk_ctx_t *w, const char *dir_path,
154 const char *base_href, const ngx_str_t *name, const brix_vfs_stat_t *vst)
155{
156 char child_path[WEBDAV_MAX_PATH];
157 char child_href[WEBDAV_MAX_PATH + 2];
158 struct stat csb;
159
160 if (brix_fs_join_path(dir_path, (char *) name->data, child_path,
161 sizeof(child_path)) != NGX_OK)
162 {
163 return NGX_OK;
164 }
165
166 propfind_vfs_to_stat(vst, &csb); /* no-follow lstat from readdir */
167
168 if (propfind_build_child_href(base_href, (char *) name->data,
169 child_href, sizeof(child_href)) != NGX_OK)
170 {
171 return NGX_OK; /* truncated href: skip this entry, do not corrupt it */
172 }
173
174 if (propfind_entry(w->r, w->head, w->tail, child_href, child_path, &csb,
175 w->req) != NGX_OK)
176 {
177 return NGX_ERROR;
178 }
179 (*w->entry_count)++;
180
181 return propfind_recurse_child(w, child_path, child_href, &csb);
182}

184/*
185 * Emit D:response elements for the children of dir_path — one level when
186 * recurse is 0 (Depth: 1), every descendant when recurse is 1 (Depth:
187 * infinity). entry_count is shared across the whole walk and checked against
188 * max_entries before each entry; once the cap is hit the walk stops and logs a
189 * warning (the response is still well-formed, just truncated). An unreadable
190 * directory is skipped, not fatal. On any propfind_entry error the open DIR
191 * is closed before returning NGX_ERROR (no fd leak on the error path).
192 */
193ngx_int_t
194propfind_walk(ngx_http_request_t *r,
195 ngx_chain_t **head, ngx_chain_t **tail,
196 const char *dir_path, const char *base_href,
197 ngx_uint_t *entry_count, ngx_uint_t max_entries,
198 const propfind_req_t *req, ngx_flag_t recurse)
199{
200 ngx_http_brix_webdav_loc_conf_t *wdcf =
201 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
202 const char *root_canon = wdcf->common.root_canon;
203 brix_vfs_ctx_t vctx;
204 brix_vfs_dir_t *dh;
205 ngx_str_t name;
206 brix_vfs_stat_t vst;
207 propfind_walk_ctx_t w;
208
209 /*
210 * Impersonation read-gate: verify the MAPPED user may READ this directory
211 * (kept as an explicit pre-check; brix_vfs_opendir below also opens AS that
212 * user via the broker, so an unreadable dir is skipped either way). No-op
213 * when impersonation is off; a denied dir is skipped, request not failed.
214 */
215 if (brix_dirlist_access_ok(r->connection->log, root_canon, dir_path)
216 != NGX_OK)
217 {
218 return NGX_OK;
219 }
220
221 /* Enumerate through the VFS: brix_vfs_opendir is impersonation-aware
222 * (broker fdopendir as the mapped user) and brix_vfs_readdir returns each
223 * name with a no-follow lstat — so a symlink lists as a plain resource and is
224 * never recursed, exactly as before. */
225 propfind_dir_ctx(r, root_canon, dir_path, &vctx);
226 dh = brix_vfs_opendir(&vctx, NULL);
227 if (dh == NULL) {
228 return NGX_OK; /* unreadable subtree: skip, do not fail the request */
229 }
230
231 w.r = r;
232 w.head = head;
233 w.tail = tail;
234 w.req = req;
235 w.entry_count = entry_count;
236 w.max_entries = max_entries;
237 w.recurse = recurse;
238
239 while (brix_vfs_readdir(dh, &name, &vst) == NGX_OK) {
240 if (!propfind_child_visible(&name)) {
241 continue;
242 }
243
244 if (*entry_count >= max_entries) {
245 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
246 "brix: PROPFIND Depth:infinity capped at %ui entries "
247 "for \"%s\"", max_entries, dir_path);
248 break;
249 }
250
251 if (propfind_walk_emit(&w, dir_path, base_href, &name, &vst)
252 != NGX_OK)
253 {
254 brix_vfs_closedir(dh, r->connection->log);
255 return NGX_ERROR;
256 }
257 }
258
259 brix_vfs_closedir(dh, r->connection->log);
260 return NGX_OK;
261}

287/*
288 * WHAT: Record the propfind_depth_total metric for the requested Depth.
289 * WHY: Depth 0/1/infinity map to three fixed low-cardinality slots; isolating
290 * the slot selection keeps the ternary out of propfind_do's CCN.
291 * HOW: depth==0→0, depth==1→1, else (infinity)→INF, then increment.
292 */
293static void
294propfind_count_depth(int depth)
295{
296 ngx_uint_t depth_slot = (depth == 0) ? BRIX_WEBDAV_PROPFIND_DEPTH_0
297 : (depth == 1) ? BRIX_WEBDAV_PROPFIND_DEPTH_1
298 : BRIX_WEBDAV_PROPFIND_DEPTH_INF;
299 BRIX_WEBDAV_METRIC_INC(propfind_depth_total[depth_slot]);
300}

302/*
303 * WHAT: Copy the (non-NUL-terminated) request URI into a bounded href buffer.
304 * WHY: The first D:response is always the request target, and its bounded href
305 * then serves as the walk's base; r->uri.data has no terminator so it must be
306 * length-clamped and NUL-terminated by hand.
307 * HOW: Clamps to out_sz-1, memcpy, terminate — identical bytes to the original
308 * inline copy.
309 */
310static void
311propfind_target_href(ngx_http_request_t *r, char *out, size_t out_sz)
312{
313 size_t uri_len = r->uri.len;
314
315 if (uri_len >= out_sz - 1) {
316 uri_len = out_sz - 2;
317 }
318 ngx_memcpy(out, r->uri.data, uri_len);
319 out[uri_len] = '\0';
320}

322/*
323 * WHAT: Emit the multistatus prologue, the target's D:response, and (for Depth
324 * 1/infinity on a collection) the child responses.
325 * WHY: Groups the body-building stage so its several NULL/NGX_OK checks do not
326 * inflate propfind_do's complexity; entry_count/href are passed by pointer as
327 * they are consumed by the finalize stage.
328 * HOW: Prologue → target entry (always) → propfind_walk children when the
329 * target is a directory and Depth is 1 (uncapped) or infinity (capped).
330 * Returns NGX_HTTP_INTERNAL_SERVER_ERROR on any allocation/emit failure,
331 * else NGX_OK — the walk itself is behavior-identical to the prior inline
332 * version.
333 */
334static ngx_int_t
335propfind_build_body(const propfind_resp_t *rb, const char *path,
336 struct stat *sb, const propfind_req_t *req, int depth)
337{
338 ngx_http_request_t *r = rb->r;
339
340 if (brix_http_chain_appendf(r->pool, rb->head, rb->tail,
341 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
342 "<D:multistatus xmlns:D=\"DAV:\""
343 " xmlns:xrd=\"http://xrootd.org/2010/ns/dav\">") == NULL)
344 {
345 return NGX_HTTP_INTERNAL_SERVER_ERROR;
346 }
347
348 propfind_target_href(r, rb->href, rb->href_sz);
349 if (propfind_entry(r, rb->head, rb->tail, rb->href, path, sb, req)
350 != NGX_OK)
351 {
352 return NGX_HTTP_INTERNAL_SERVER_ERROR;
353 }
354 (*rb->entry_count)++;
355
356 if ((depth == 1 || depth == -1) && S_ISDIR(sb->st_mode)) {
357 ngx_uint_t cap = (depth == -1) ? PROPFIND_INFINITY_MAX_ENTRIES
358 : (ngx_uint_t) -1;
359 if (propfind_walk(r, rb->head, rb->tail, path, rb->href,
360 rb->entry_count, cap, req, depth == -1) != NGX_OK)
361 {
362 return NGX_HTTP_INTERNAL_SERVER_ERROR;
363 }
364 }
365
366 if (brix_http_chain_appendf(r->pool, rb->head, rb->tail,
367 "</D:multistatus>") == NULL)
368 {
369 return NGX_HTTP_INTERNAL_SERVER_ERROR;
370 }
371 return NGX_OK;
372}

374/*
375 * WHAT: Mark end-of-response on the last buffer and sum the chain's payload.
376 * WHY: The output filter needs last_buf/last_in_chain on the final buffer, and
377 * Content-Length is the sum of every buffer's [pos,last).
378 * HOW: Sets the flags on tail (if any) then accumulates the byte total.
379 */
380static off_t
381propfind_finalize_chain(ngx_chain_t *head, ngx_chain_t *tail)
382{
383 off_t total_len = 0;
384 ngx_chain_t *lc;
385
386 if (tail != NULL) {
387 tail->buf->last_buf = 1;
388 tail->buf->last_in_chain = 1;
389 }
390 for (lc = head; lc != NULL; lc = lc->next) {
391 total_len += lc->buf->last - lc->buf->pos;
392 }
393 return total_len;
394}

396/*
397 * WHAT: Set the 207 status + Content-Type/Content-Length headers and send them.
398 * WHY: Header assembly is a distinct stage from body building; splitting it out
399 * keeps propfind_do below the CCN cap.
400 * HOW: Pushes the Content-Type header, writing NGX_HTTP_INTERNAL_SERVER_ERROR
401 * to *out_rc and returning NGX_ERROR on list failure so the caller aborts
402 * before sending; otherwise sets the send-header result in *out_rc and
403 * returns NGX_OK, leaving the NGX_ERROR / header_only decision to the caller
404 * exactly as the original inline code.
405 */
406static ngx_int_t
407propfind_send_headers(ngx_http_request_t *r, off_t total_len, ngx_int_t *out_rc)
408{
409 ngx_table_elt_t *h;
410
411 r->headers_out.status = 207;
412 r->headers_out.content_length_n = total_len;
413
414 h = ngx_list_push(&r->headers_out.headers);
415 if (h == NULL) {
416 *out_rc = NGX_HTTP_INTERNAL_SERVER_ERROR;
417 return NGX_ERROR;
418 }
419 h->hash = 1;
420 ngx_str_set(&h->key, "Content-Type");
421 ngx_str_set(&h->value, "application/xml; charset=\"utf-8\"");
422
423 *out_rc = ngx_http_send_header(r);
424 return NGX_OK;
425}

447/*
448 * Build and send the complete 207 Multi-Status response.
449 * Runs after the request body has been read (see propfind_body_handler), so it
450 * may resolve/stat the target and parse the body synchronously. Assembles the
451 * XML as an ngx_chain_t, sums the body length for Content-Length, marks the last
452 * buffer, then sends headers + body. Returns an HTTP status / NGX_* code.
453 */
454ngx_int_t
455propfind_do(ngx_http_request_t *r)
456{
457 char path[WEBDAV_MAX_PATH];
458 char href[WEBDAV_MAX_PATH + 2];
459 struct stat sb;
460 ngx_int_t rc;
461 int depth;
462 ngx_chain_t *head = NULL;
463 ngx_chain_t *tail = NULL;
464 off_t total_len;
465 ngx_uint_t entry_count = 0;
466 propfind_req_t req;
467
468 rc = webdav_resolve_stat(r, path, sizeof(path), &sb);
469 if (rc != NGX_OK) {
470 return rc;
471 }
472
473 propfind_parse_request(r, &req);
474
475 depth = propfind_parse_depth(r);
476 propfind_count_depth(depth);
477
478 {
479 propfind_resp_t rb = {
480 .r = r, .head = &head, .tail = &tail,
481 .entry_count = &entry_count,
482 .href = href, .href_sz = sizeof(href)
483 };
484 rc = propfind_build_body(&rb, path, &sb, &req, depth);
485 }
486 if (rc != NGX_OK) {
487 return rc;
488 }
489
490 total_len = propfind_finalize_chain(head, tail);
491 BRIX_WEBDAV_METRIC_ADD(propfind_entries_total, entry_count);
492
493 if (propfind_send_headers(r, total_len, &rc) != NGX_OK) {
494 return rc; /* header list allocation failed → 500 */
495 }
496 if (rc == NGX_ERROR || r->header_only) {
497 return rc;
498 }
499
500 propfind_count_tx(r, total_len);
501 return ngx_http_output_filter(r, head);
502}
