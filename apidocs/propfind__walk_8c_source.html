44/* Hard ceiling on entries emitted for a Depth: infinity PROPFIND so a deep or
45 * wide tree cannot generate an unbounded response / runaway recursion. */

47/*
48 * Emit D:response elements for the children of dir_path — one level when
49 * recurse is 0 (Depth: 1), every descendant when recurse is 1 (Depth:
50 * infinity). entry_count is shared across the whole walk and checked against
51 * max_entries before each entry; once the cap is hit the walk stops and logs a
52 * warning (the response is still well-formed, just truncated). An unreadable
53 * directory is skipped, not fatal. On any propfind_entry error the open DIR
54 * is closed before returning NGX_ERROR (no fd leak on the error path).
55 */
56ngx_int_t
57propfind_walk(ngx_http_request_t *r,
58 ngx_chain_t **head, ngx_chain_t **tail,
59 const char *dir_path, const char *base_href,
60 ngx_uint_t *entry_count, ngx_uint_t max_entries,
61 const propfind_req_t *req, ngx_flag_t recurse)
62{
63 ngx_http_brix_webdav_loc_conf_t *wdcf =
64 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
65 const char *root_canon = wdcf->common.root_canon;
66 brix_vfs_ctx_t vctx;
67 brix_vfs_dir_t *dh;
68 ngx_str_t name;
69 brix_vfs_stat_t vst;
70
71 /*
72 * Impersonation read-gate: verify the MAPPED user may READ this directory
73 * (kept as an explicit pre-check; brix_vfs_opendir below also opens AS that
74 * user via the broker, so an unreadable dir is skipped either way). No-op
75 * when impersonation is off; a denied dir is skipped, request not failed.
76 */
77 if (brix_dirlist_access_ok(r->connection->log, root_canon, dir_path)
78 != NGX_OK)
79 {
80 return NGX_OK;
81 }
82
83 /* Enumerate through the VFS: brix_vfs_opendir is impersonation-aware
84 * (broker fdopendir as the mapped user) and brix_vfs_readdir returns each
85 * name with a no-follow lstat — so a symlink lists as a plain resource and is
86 * never recursed, exactly as before. */
87 propfind_dir_ctx(r, root_canon, dir_path, &vctx);
88 dh = brix_vfs_opendir(&vctx, NULL);
89 if (dh == NULL) {
90 return NGX_OK; /* unreadable subtree: skip, do not fail the request */
91 }
92
93 while (brix_vfs_readdir(dh, &name, &vst) == NGX_OK) {
94 char child_path[WEBDAV_MAX_PATH];
95 char child_href[WEBDAV_MAX_PATH + 2];
96 struct stat csb;
97
98 /* readdir already skips "."/".."; drop any other dotfile (not listed). */
99 if (name.data[0] == '.') {
100 continue;
101 }
102 /* Hide internal metadata/staging artifacts. A sidecar like
103 * "f.dat.cinfo" or an upload temp is NOT a dotfile, so the check above
104 * misses it — this predicate catches them by suffix/infix. */
105 if (brix_is_internal_name((const char *) name.data)) {
106 continue;
107 }
108
109 if (*entry_count >= max_entries) {
110 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
111 "brix: PROPFIND Depth:infinity capped at %ui entries "
112 "for \"%s\"", max_entries, dir_path);
113 break;
114 }
115
116 if (brix_fs_join_path(dir_path, (char *) name.data, child_path,
117 sizeof(child_path)) != NGX_OK)
118 {
119 continue;
120 }
121
122 propfind_vfs_to_stat(&vst, &csb); /* no-follow lstat from readdir */
123
124 /* Build child href = base_href + name, inserting a single '/' only if
125 * base_href is not already slash-terminated. snprintf returns the
126 * would-be length; >= buffer size means it was truncated, so skip the
127 * entry rather than emit a corrupted href. */
128 {
129 size_t blen = strlen(base_href);
130 if (blen == 0 || base_href[blen - 1] != '/') {
131 if ((size_t) snprintf(child_href, sizeof(child_href),
132 "%s/%s", base_href, (char *) name.data)
133 >= sizeof(child_href))
134 continue;
135 } else if ((size_t) snprintf(child_href, sizeof(child_href),
136 "%s%s", base_href, (char *) name.data)
137 >= sizeof(child_href))
138 {
139 continue;
140 }
141 }
142
143 if (propfind_entry(r, head, tail, child_href, child_path, &csb, req)
144 != NGX_OK)
145 {
146 brix_vfs_closedir(dh, r->connection->log);
147 return NGX_ERROR;
148 }
149 (*entry_count)++;
150
151 /* Recurse into subdirectories (with a trailing-slash href). Re-check
152 * the cap first so a balanced tree cannot blow past max_entries via
153 * nested calls before the per-entry check fires. */
154 if (recurse && S_ISDIR(csb.st_mode) && *entry_count < max_entries) {
155 char subdir_href[WEBDAV_MAX_PATH + 3];
156 if ((size_t) snprintf(subdir_href, sizeof(subdir_href),
157 "%s/", child_href)
158 < sizeof(subdir_href))
159 {
160 if (propfind_walk(r, head, tail, child_path, subdir_href,
161 entry_count, max_entries, req, recurse)
162 != NGX_OK)
163 {
164 brix_vfs_closedir(dh, r->connection->log);
165 return NGX_ERROR;
166 }
167 }
168 }
169 }
170
171 brix_vfs_closedir(dh, r->connection->log);
172 return NGX_OK;
173}

180/*
181 * Build and send the complete 207 Multi-Status response.
182 * Runs after the request body has been read (see propfind_body_handler), so it
183 * may resolve/stat the target and parse the body synchronously. Assembles the
184 * XML as an ngx_chain_t, sums the body length for Content-Length, marks the last
185 * buffer, then sends headers + body. Returns an HTTP status / NGX_* code.
186 */
187ngx_int_t
188propfind_do(ngx_http_request_t *r)
189{
190 char path[WEBDAV_MAX_PATH];
191 struct stat sb;
192 ngx_int_t rc;
193 int depth;
194 ngx_chain_t *head = NULL;
195 ngx_chain_t *tail = NULL;
196 off_t total_len = 0;
197 ngx_chain_t *lc;
198 ngx_table_elt_t *h;
199 ngx_uint_t entry_count = 0;
200 propfind_req_t req;
201
202 rc = webdav_resolve_stat(r, path, sizeof(path), &sb);
203 if (rc != NGX_OK) {
204 return rc;
205 }
206
207 propfind_parse_request(r, &req);
208
209 depth = propfind_parse_depth(r);
210 {
211 ngx_uint_t depth_slot = (depth == 0) ? BRIX_WEBDAV_PROPFIND_DEPTH_0
212 : (depth == 1) ? BRIX_WEBDAV_PROPFIND_DEPTH_1
213 : BRIX_WEBDAV_PROPFIND_DEPTH_INF;
214 BRIX_WEBDAV_METRIC_INC(propfind_depth_total[depth_slot]);
215 }
216
217 if (brix_http_chain_appendf(r->pool, &head, &tail,
218 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
219 "<D:multistatus xmlns:D=\"DAV:\""
220 " xmlns:xrd=\"http://xrootd.org/2010/ns/dav\">") == NULL)
221 {
222 return NGX_HTTP_INTERNAL_SERVER_ERROR;
223 }
224
225 /* First D:response is always the request target itself (Depth 0/1/inf);
226 * its bounded href then serves as the walk's base. */
227 char href[WEBDAV_MAX_PATH + 2];
228 {
229 size_t uri_len = r->uri.len;
230
231 /* r->uri is not NUL-terminated; copy a bounded, terminated href. */
232 if (uri_len >= sizeof(href) - 1) {
233 uri_len = sizeof(href) - 2;
234 }
235 ngx_memcpy(href, r->uri.data, uri_len);
236 href[uri_len] = '\0';
237
238 if (propfind_entry(r, &head, &tail, href, path, &sb, &req) != NGX_OK) {
239 return NGX_HTTP_INTERNAL_SERVER_ERROR;
240 }
241 entry_count++;
242 }
243
244 /* Depth 1 or infinity on a collection: the children come from
245 * propfind_walk — one level for Depth: 1 (uncapped, as before this loop
246 * was unrolled inline), recursive with the entry cap for infinity. */
247 if ((depth == 1 || depth == -1) && S_ISDIR(sb.st_mode)) {
248 ngx_uint_t cap = (depth == -1) ? PROPFIND_INFINITY_MAX_ENTRIES
249 : (ngx_uint_t) -1;
250
251 if (propfind_walk(r, &head, &tail, path, href, &entry_count, cap,
252 &req, depth == -1) != NGX_OK)
253 {
254 return NGX_HTTP_INTERNAL_SERVER_ERROR;
255 }
256 }
257
258 if (brix_http_chain_appendf(r->pool, &head, &tail,
259 "</D:multistatus>") == NULL)
260 {
261 return NGX_HTTP_INTERNAL_SERVER_ERROR;
262 }
263
264 /* Mark end-of-response on the final buffer so the output filter flushes. */
265 if (tail != NULL) {
266 tail->buf->last_buf = 1;
267 tail->buf->last_in_chain = 1;
268 }
269
270 /* Content-Length = sum of all buffer payloads in the chain. */
271 for (lc = head; lc != NULL; lc = lc->next) {
272 total_len += lc->buf->last - lc->buf->pos;
273 }
274
275 BRIX_WEBDAV_METRIC_ADD(propfind_entries_total, entry_count);
276
277 r->headers_out.status = 207;
278 r->headers_out.content_length_n = total_len;
279
280 h = ngx_list_push(&r->headers_out.headers);
281 if (h == NULL) {
282 return NGX_HTTP_INTERNAL_SERVER_ERROR;
283 }
284 h->hash = 1;
285 ngx_str_set(&h->key, "Content-Type");
286 ngx_str_set(&h->value, "application/xml; charset=\"utf-8\"");
287
288 rc = ngx_http_send_header(r);
289 if (rc == NGX_ERROR || r->header_only) {
290 return rc;
291 }
292
293 BRIX_WEBDAV_METRIC_ADD(bytes_tx_total, (size_t) total_len);
294 if (r->connection && r->connection->sockaddr
295 && r->connection->sockaddr->sa_family == AF_INET6) {
296 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv6_total, (size_t) total_len);
297 } else {
298 BRIX_WEBDAV_METRIC_ADD(bytes_tx_ipv4_total, (size_t) total_len);
299 }
300
301 return ngx_http_output_filter(r, head);
302}
