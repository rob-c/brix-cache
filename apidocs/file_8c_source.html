1/*
2 * file.c - shared bounded bearer-token file reader.
3 *
4 * WHAT: Reads a bearer token from a file with configurable size limits. Opens the
5 * specified path, reads up to buf_sz-1 bytes into caller-provided buffer,
6 * strips trailing whitespace (CR/LF/space), and NUL-terminates the result.
7 * Returns token length via out_len pointer when provided.
8 *
9 * WHY: Token endpoints often write tokens to files for credential caching or
10 * service-account integration. This helper provides a safe bounded read that
11 * handles common file-format issues (trailing newlines, oversized content)
12 * without requiring callers to implement size limits and whitespace stripping.
13 * Used by token/validate.c for bearer-token validation from files and s3/auth.c
14 * for S3 credential retrieval. The label parameter enables callers to prefix
15 * error log messages with context (e.g., "token" vs "s3").
16 *
17 * HOW: Validates path non-empty, fits in NGX_MAX_PATH buffer, buf non-null with
18 * capacity >= 2 bytes. Copies path string via ngx_memcpy + NUL terminator. Opens
19 * file with fopen("rb") — sets FD_CLOEXEC to prevent inheritance across exec().
20 * Reads up to buf_sz-1 bytes via fread() (leaves room for NUL). Checks ferror()
21 * after read, saves errno before close. Strips trailing whitespace via isspace() loop.
22 * Rejects empty files (n==0) with EINVAL. Sets out_len on success if pointer non-null.
23 * Returns NGX_OK on success, NGX_ERROR on any failure with log message populated.
24 */

62/*
63 * brix_token_validate_args - reject malformed arguments and build a C path.
64 *
65 * WHAT: Validates path/buffer arguments and copies path into a NUL-terminated
66 * pathz buffer (size NGX_MAX_PATH) on success.
67 * WHY: Bounds must be checked before any file access; concentrating the guard
68 * in one helper keeps the reader loop free of validation branches while
69 * preserving the original EINVAL semantics and log message.
70 * HOW: Rejects NULL/empty/over-long path, NULL buf, or capacity < 2 bytes.
71 * On rejection logs (when log != NULL), sets errno=EINVAL, returns
72 * NGX_ERROR. On success copies path->data + NUL terminator into pathz
73 * (a fixed NGX_MAX_PATH-byte buffer owned by the caller).
74 */
75static ngx_int_t
76brix_token_validate_args(const ngx_str_t *path, const u_char *buf,
77 size_t buf_sz, char *pathz, const brix_token_diag_t *diag)
78{
79 if (path == NULL || path->len == 0 || path->len >= NGX_MAX_PATH
80 || buf == NULL || buf_sz < 2)
81 {
82 if (diag->log != NULL) {
83 ngx_log_error(NGX_LOG_ERR, diag->log, 0,
84 "%s token file path missing or too long",
85 brix_token_label(diag));
86 }
87 errno = EINVAL;
88 return NGX_ERROR;
89 }
90
91 ngx_memcpy(pathz, path->data, path->len);
92 pathz[path->len] = '\0';
93 return NGX_OK;
94}

96/*
97 * brix_token_slurp - bounded read of a token file into buf.
98 *
99 * WHAT: Opens pathz, reads up to buf_sz-1 bytes into buf, and returns the byte
100 * count via *n_out. Does not NUL-terminate or trim (caller's job).
101 * WHY: Isolates all stdio + errno handling (open failure, read error with
102 * errno preservation across fclose) so the top-level flow stays linear.
103 * HOW: fopen("rb"), set FD_CLOEXEC, fread. On ferror, saves errno across
104 * fclose, logs, returns NGX_ERROR. On success closes and reports count.
105 * Read-only stream: fclose status cannot affect already-consumed data.
106 */
107static ngx_int_t
108brix_token_slurp(const char *pathz, u_char *buf, size_t buf_sz, size_t *n_out,
109 const brix_token_diag_t *diag)
110{
111 FILE *fp;
112 size_t n;
113
114 fp = fopen(pathz, "rb");
115 if (fp == NULL) {
116 if (diag->log != NULL) {
117 ngx_log_error(NGX_LOG_ERR, diag->log, ngx_errno,
118 "%s cannot open token file \"%s\"",
119 brix_token_label(diag), pathz);
120 }
121 return NGX_ERROR;
122 }
123 (void) fcntl(fileno(fp), F_SETFD, FD_CLOEXEC);
124
125 n = fread(buf, 1, buf_sz - 1, fp);
126 if (ferror(fp)) {
127 int saved = errno;
128 (void) fclose(fp); /* phase74-fp: read-only stream, ferror already latched the read failure being reported */
129 errno = saved;
130 if (diag->log != NULL) {
131 ngx_log_error(NGX_LOG_ERR, diag->log, ngx_errno,
132 "%s cannot read token file \"%s\"",
133 brix_token_label(diag), pathz);
134 }
135 return NGX_ERROR;
136 }
137 (void) fclose(fp); /* phase74-fp: read-only stream fully consumed, ferror checked above — close status cannot affect the data */
138
139 *n_out = n;
140 return NGX_OK;
141}

143/*
144 * brix_token_trim - NUL-terminate and strip trailing whitespace in place.
145 *
146 * WHAT: NUL-terminates buf at n, then removes trailing isspace() bytes,
147 * returning the trimmed length.
148 * WHY: Token files commonly carry trailing CR/LF/space; trimming is a pure
149 * string transform best kept out of the I/O and validation stages.
150 * HOW: Terminates at n, then walks back over isspace() bytes, NUL-terminating
151 * as it shrinks. Returns the resulting length.
152 */
153static size_t
154brix_token_trim(u_char *buf, size_t n)
155{
156 buf[n] = '\0';
157 while (n > 0 && isspace((unsigned char) buf[n - 1])) {
158 buf[--n] = '\0';
159 }
160 return n;
161}

164brix_token_read_file(const ngx_str_t *path, u_char *buf, size_t buf_sz,
165 size_t *out_len, ngx_log_t *log, const char *label)
166{
167 char pathz[NGX_MAX_PATH];

170
171 diag.log = log;
172 diag.label = label;
173
174 if (out_len != NULL) {
175 *out_len = 0;
176 }
177
178 if (brix_token_validate_args(path, buf, buf_sz, pathz, &diag) != NGX_OK) {
179 return NGX_ERROR;
180 }
181
182 if (brix_token_slurp(pathz, buf, buf_sz, &n, &diag) != NGX_OK) {
183 return NGX_ERROR;
184 }
185
186 n = brix_token_trim(buf, n);
187
188 if (n == 0) {
189 if (diag.log != NULL) {
190 ngx_log_error(NGX_LOG_ERR, diag.log, 0,
191 "%s token file \"%s\" is empty",
192 brix_token_label(&diag), pathz);
193 }
194 errno = EINVAL;
195 return NGX_ERROR;
196 }
197
198 if (out_len != NULL) {
199 *out_len = n;
200 }
201
202 return NGX_OK;
203}
