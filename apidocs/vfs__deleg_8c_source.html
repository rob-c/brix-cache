1/*
2 * vfs_deleg.c — VFS delegation live-cred bag + PASSTHROUGH materialiser
3 * (phase-70 §4, §5.1, §5.4).
4 *
5 * WHAT: Implements the per-request delegation seam that turns the raw
6 * forwardable credential the front door captured (a bearer JWT, or a
7 * user-supplied full x509 proxy PEM) into the exact brix_sd_cred_t form
8 * the backend GSI/ZTN presenter already consumes:
9 * brix_vfs_ctx_bind_backend_deleg() — hangs a live bag on a VFS ctx.
10 * brix_vfs_backend_mode() — reports the ctx's resolved mode.
11 * brix_vfs_deleg_live_cred() — validates + materialises the bag's
12 * bytes into *cred (bearer straight through; proxy PEM → 0600 temp path
13 * with an unlink+zero pool cleanup), honouring fallback-deny.
14 *
15 * WHY: The whole point of phase-70 is to authenticate the backend leg AS the
16 * inbound user with zero admin provisioning. PASSTHROUGH is the only path
17 * that reuses the existing origin-leg code unchanged: a full proxy PEM at
18 * a 0600 path is precisely what brix_cache_origin_auth_gsi() loads, and a
19 * raw JWT is precisely what brix_cache_origin_auth_ztn() presents. This
20 * file is the single seam where captured BYTES become that cred form, so
21 * the decision (validate → materialise → deny-or-fallback) lives in one
22 * auditable place rather than smeared across the protocol handlers.
23 *
24 * HOW: The bag is bound by reference (bytes owned by the request pool). The
25 * materialiser copies a bearer directly, or for a proxy writes the PEM to
26 * an owner-only temp via brix_proxy_gsi_write_pem_temp() (net/proxy) and
27 * registers a pool cleanup that unlink()s the file and zeroes the path
28 * string, so the private key never outlives the request. PEM_read_bio_X509
29 * rejects non-PEM bytes before materialising; the full RFC-3820 chain-trust
30 * + DN-match gate is a documented TODO (§5.1) — the capture agent supplies
31 * validated bytes for now.
32 */

36#include "auth/krb5/forward.h" /* brix_krb5_deleg_to_origin (§5.7) */

42/* Cleanup payload: the pool-allocated 0600 temp path materialised for a
43 * PASSTHROUGH proxy. On pool destruction the file is removed and the path
44 * string is zeroed so it cannot linger in freed-but-reused pool memory. */
45typedef struct {
46 char *path; /* NUL-terminated temp path, owned by the request pool */
47} brix_deleg_temp_t;
48
49/* ---- brix_vfs_deleg_temp_cleanup -------------------------------------------
50 *
51 * WHAT: Pool-cleanup handler: unlink the materialised proxy temp and zero its
52 * path string.
53 *
54 * WHY: §6 secret hygiene — the private key lives only in a 0600 tmpfs file for
55 * the op's duration; it must be unlinked (and the path scrubbed) the
56 * moment the request pool is torn down, success or failure.
57 *
58 * HOW: data is a brix_deleg_temp_t*. unlink() ignores ENOENT (a driver may
59 * already have consumed+removed it); the path bytes are then zeroed. */
60static void
61brix_vfs_deleg_temp_cleanup(void *data)
62{
63 brix_deleg_temp_t *t = data;
64
65 if (t == NULL || t->path == NULL) {
66 return;
67 }
68
69 (void) unlink(t->path); /* vfs-seam-allow: config-domain PASSTHROUGH proxy credential temp (not export storage) */
70 ngx_memzero(t->path, ngx_strlen(t->path));
71 t->path = NULL;
72}

95/* ---- brix_vfs_deleg_set_exchange -------------------------------------------
96 *
97 * WHAT: Populate the EXCHANGE conf (endpoint + client creds + audience) on the
98 * ctx's already-bound live-cred bag.
99 *
100 * WHY: The cred gate needs the RFC-8693 endpoint to trade a live bearer for a
101 * backend-audienced token (§5.4). The bag layout is private to the VFS, so
102 * this setter is the one place the borrowed conf strings are stamped onto
103 * it — kept separate from brix_vfs_deleg_bind so the capture site stays a
104 * single call plus an optional exchange-conf call (no signature churn on
105 * the bind path shared by PASSTHROUGH).
106 *
107 * HOW: A no-op when no bag is bound (nothing forwardable was captured) or the
108 * endpoint is empty (EXCHANGE then degrades to verbatim passthrough in the
109 * gate). All strings are borrowed (conf-owned, NUL-terminated). */
110void
111brix_vfs_deleg_set_exchange(brix_vfs_ctx_t *vctx,
112 const ngx_str_t *endpoint, const ngx_str_t *client_id,
113 const ngx_str_t *client_secret, const ngx_str_t *audience)
114{
115 brix_deleg_live_t *live;
116
117 if (vctx == NULL || vctx->deleg_live == NULL) {
118 return;
119 }
120 if (endpoint == NULL || endpoint->len == 0 || endpoint->data == NULL) {
121 return;
122 }
123
124 live = vctx->deleg_live;
125 live->tx.endpoint = *endpoint;
126 if (client_id != NULL) {
127 live->tx.client_id = *client_id;
128 }
129 if (client_secret != NULL) {
130 live->tx.client_secret = *client_secret;
131 }
132 if (audience != NULL) {
133 live->tx_audience = *audience;
134 }
135}

137/* ---- brix_vfs_deleg_bind ---------------------------------------------------
138 *
139 * WHAT: Allocate a live-cred bag from `pool`, fill it with the captured
140 * forwardable credential bytes (bearer text and/or full proxy PEM) plus
141 * the resolved mode, and bind it onto `vctx`. A no-op (SELECT path) when
142 * `mode` is BRIX_CRED_SELECT or nothing forwardable was captured.
143 *
144 * WHY: This is the single constructor for the bag, kept here (not in a protocol
145 * handler) because brix_deleg_live_t's layout is private to the VFS. Every
146 * front-door bind site calls this uniformly right after
147 * brix_vfs_ctx_bind_backend_cred, so the "conf mode + captured bytes →
148 * bound bag" step is stated once rather than copy-pasted per protocol.
149 *
150 * HOW: Borrows the byte ranges (owned by the caller's request pool). When no
151 * credential bytes are present the bag is not bound and the ctx stays on
152 * the dir-based SELECT path — so an operator enabling passthrough without a
153 * forwardable credential in-hand degrades to SELECT rather than deny here
154 * (the deny decision belongs to the cred gate, not the capture). */
155ngx_int_t
156brix_vfs_deleg_bind(ngx_pool_t *pool, brix_vfs_ctx_t *vctx,
157 enum brix_cred_mode mode, const ngx_str_t *bearer,
158 const ngx_str_t *proxy_pem)
159{
160 brix_deleg_live_t *live;
161 int have_proxy;
162 int have_bearer;
163
164 if (pool == NULL || vctx == NULL || mode == BRIX_CRED_SELECT) {
165 return NGX_OK;
166 }
167
168 have_proxy = (proxy_pem != NULL && proxy_pem->len > 0
169 && proxy_pem->data != NULL);
170 have_bearer = (bearer != NULL && bearer->len > 0 && bearer->data != NULL);
171
172 if (!have_proxy && !have_bearer) {
173 return NGX_OK; /* nothing forwardable captured — stay on SELECT */
174 }
175
176 live = ngx_pcalloc(pool, sizeof(*live));
177 if (live == NULL) {
178 return NGX_ERROR;
179 }
180
181 live->mode = mode;
182 if (have_proxy) {
183 live->have_proxy_pem = 1;
184 live->proxy_pem = *proxy_pem;
185 }
186 if (have_bearer) {
187 live->bearer = *bearer;
188 }
189
190 brix_vfs_ctx_bind_backend_deleg(vctx, live);
191 return NGX_OK;
192}

213/* ---- brix_vfs_backend_accepts_proxy ----------------------------------------
214 *
215 * WHAT: Report whether the ctx's resolved leaf backend consumes a forwarded
216 * X.509 proxy PEM.
217 *
218 * WHY: A protocol that forwards a captured proxy by default (the gsiftp→xrootd
219 * gateway) must not bind a proxy bag on a backend that cannot use one — the
220 * cred gate would then deny (EACCES) a request that should have served on
221 * the service credential. Gating the bind on this predicate keeps the
222 * default-on delegation scoped to proxy-capable backends (xroot, s3).
223 *
224 * HOW: brix_sd_cred_accept on the resolved leaf; NULL-safe (default-POSIX
225 * resolves to a NULL sd whose accept mask is 0). */
226int
227brix_vfs_backend_accepts_proxy(brix_vfs_ctx_t *vctx)
228{
229 if (vctx == NULL || vctx->sd == NULL) {
230 return 0;
231 }
232 return (brix_sd_cred_accept(brix_vfs_ns_leaf(vctx->sd))
233 & BRIX_SD_CRED_PROXY_PEM) ? 1 : 0;
234}

272/* ---- brix_vfs_deleg_pem_is_valid -------------------------------------------
273 *
274 * WHAT: True iff `pem`/`len` parses as at least one PEM X509 certificate.
275 *
276 * WHY: Reject garbage / non-PEM bytes before writing them to a temp and
277 * handing the path to the GSI presenter (§5.1 gate, minimal form). The
278 * full RFC-3820 chain-trust + DN-match validation is a documented TODO
279 * below.
280 *
281 * HOW: BIO over the bytes → PEM_read_bio_X509; a single successful parse is
282 * enough to prove the bytes are a PEM certificate. Frees the cert + BIO. */
283static int
284brix_vfs_deleg_pem_is_valid(const u_char *pem, size_t len)
285{
286 BIO *bio;
287 X509 *cert;
288
289 if (pem == NULL || len == 0) {
290 return 0;
291 }
292
293 bio = BIO_new_mem_buf(pem, (int) len);
294 if (bio == NULL) {
295 return 0;
296 }
297
298 cert = PEM_read_bio_X509(bio, NULL, NULL, NULL);
299 BIO_free(bio);
300
301 if (cert == NULL) {
302 return 0;
303 }
304
305 X509_free(cert);
306 return 1;
307}

309/* ---- brix_vfs_deleg_deny ---------------------------------------------------
310 *
311 * WHAT: Terminal decision for a missing/invalid live cred: EACCES→deny in
312 * fallback-deny mode, else service-credential fallback.
313 *
314 * WHY: §6 no-wrong-identity-fallback — a passthrough failure must never reach
315 * the origin under the service cred when the operator set deny.
316 *
317 * HOW: Deny mode: errno/err_out=EACCES, use_cred=0, NGX_ERROR. Otherwise leave
318 * use_cred=0 and return NGX_OK so the caller falls to the service cred. */
319static ngx_int_t
320brix_vfs_deleg_deny(brix_vfs_ctx_t *ctx, int *use_cred, int *err_out)
321{
322 *use_cred = 0;
323
324 /* A NULL ctx (no VFS context bound — brix_vfs_deleg_live_cred forwards it
325 * here verbatim) means the operator's deny/fallback choice is unknowable:
326 * fail closed rather than silently fall back to the service credential. */
327 if (ctx == NULL || ctx->storage_cred_deny) {
328 errno = EACCES;
329 if (err_out != NULL) {
330 *err_out = EACCES;
331 }
332 return NGX_ERROR;
333 }
334
335 return NGX_OK;
336}

359/* ---- brix_vfs_deleg_exchange -----------------------------------------------
360 *
361 * WHAT: Materialise an EXCHANGE bearer: trade the live subject JWT for a
362 * backend-audienced token via brix_token_exchange() and stamp the result
363 * onto cred->bearer (mode EXCHANGE).
364 *
365 * WHY: §5.4 — when a backend audience is node-bound the cache must mint an
366 * origin-specific token rather than replay the client's verbatim. The
367 * minted token is then presented to the origin ZTN leg exactly like a
368 * passthrough bearer.
369 *
370 * HOW: Requires a configured exchange endpoint (live->tx.endpoint) — the caller
371 * (brix_vfs_deleg_live_cred) has already applied the "endpoint unset ⇒
372 * verbatim passthrough" fallback, so reaching here means the endpoint is
373 * set. brix_token_exchange() POSTs the RFC-8693 grant using the first
374 * backend audience (live->tx_audience); on NGX_OK the pool-copied token is
375 * borrowed into cred->bearer. On failure → deny (never the service cred in
376 * fallback-deny mode). The subject/minted tokens are never logged. */
377static ngx_int_t
378brix_vfs_deleg_exchange(brix_vfs_ctx_t *ctx, brix_sd_cred_t *cred,
379 int *use_cred, int *err_out)
380{
381 brix_deleg_live_t *live = ctx->deleg_live;
382 ngx_str_t minted = ngx_null_string;
383 const ngx_str_t *aud;
384
385 aud = (live->tx_audience.len > 0) ? &live->tx_audience : NULL;
386
387 if (brix_token_exchange(ctx->pool, &live->bearer, aud, NULL,
388 &live->tx, &minted, ctx->log) != NGX_OK) {
389 ngx_log_error(NGX_LOG_ERR, ctx->log, 0,
390 "brix: backend token-exchange failed - denying (no service-cred "
391 "fallback for EXCHANGE)");
392 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
393 }
394
395 cred->bearer = (const char *) minted.data;
396 cred->mode = BRIX_CRED_EXCHANGE;
397 *use_cred = 1;
398
399 return NGX_OK;
400}

402/* ---- brix_vfs_deleg_proxy --------------------------------------------------
403 *
404 * WHAT: Materialise a PASSTHROUGH full x509 proxy: validate the PEM, write it to
405 * a 0600 temp, register the unlink+zero cleanup, and point cred->x509_proxy
406 * at the temp path.
407 *
408 * WHY: brix_cache_origin_auth_gsi() authenticates from a proxy FILE path, so
409 * any strategy that materialises a proxy PEM at a 0600 path reuses the
410 * origin leg unchanged (§5.1). The private key must never be logged and
411 * must be unlinked on pool teardown.
412 *
413 * HOW: brix_vfs_deleg_pem_is_valid() rejects non-PEM bytes → deny. Otherwise
414 * brix_proxy_gsi_write_pem_temp() creates the owner-only temp; the path is
415 * copied onto the pool and a cleanup registered to unlink+zero it.
416 *
417 * RFC-3820 chain-trust (phase-70 §5.1): the full gate is (1) chain parses AND is
418 * unexpired; (2) leaf DN EQUALS the front-door authenticated DN (no
419 * privilege swap); (3) chain is RFC-3820-valid AND trusted by the export's
420 * CA store via brix_gsi_verify_chain(..., client_purpose=0); (4) TLS-only
421 * transport. Steps (2)-(4) are enforced at CAPTURE (webdav auth_cert.c /
422 * delegation.c already run brix_gsi_verify_chain against conf->ca_store and
423 * match the DN before stashing the bytes), so only already-validated PEM
424 * reaches this seam. The trust check cannot be RE-run here because
425 * brix_gsi_verify_chain requires an X509_STORE* and brix_vfs_ctx_t carries
426 * no CA store — the store lives on the protocol conf (webdav conf->ca_store)
427 * and reaching it needs a new ctx-bind at the capture sites (owned by other
428 * agents). DEFERRED: add a `const void *ca_store` field to brix_vfs_ctx_t +
429 * a brix_vfs_ctx_bind_ca_store() call at each deleg capture site, then
430 * re-verify the chain here before materialising. Until then this seam
431 * enforces PEM well-formedness and relies on the capture-side gate. */
432static ngx_int_t
433brix_vfs_deleg_proxy(brix_vfs_ctx_t *ctx, brix_sd_cred_t *cred,
434 int *use_cred, int *err_out)
435{
436 brix_deleg_live_t *live = ctx->deleg_live;
437 char tmp[NGX_MAX_PATH];
438 char *path;
439 size_t path_len;
440 brix_deleg_temp_t *payload;
441 ngx_pool_cleanup_t *cln;
442 ngx_str_t princ = ngx_string("");
443
444 if (!brix_vfs_deleg_pem_is_valid(live->proxy_pem.data,
445 live->proxy_pem.len)) {
446 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
447 }
448
449 if (brix_proxy_gsi_write_pem_temp(live->proxy_pem.data,
450 live->proxy_pem.len, tmp, sizeof(tmp)) != 0) {
451 if (ctx->identity != NULL) {
452 princ = ctx->identity->subject;
453 }
454 ngx_log_error(NGX_LOG_ERR, ctx->log, ngx_errno,
455 "brix: failed to materialise PASSTHROUGH proxy temp for "
456 "principal=\"%V\"", &princ);
457 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
458 }
459
460 path_len = ngx_strlen(tmp);
461 path = ngx_pnalloc(ctx->pool, path_len + 1);
462 if (path == NULL) {
463 (void) unlink(tmp); /* vfs-seam-allow: config-domain PASSTHROUGH proxy credential temp (not export storage) */
464 errno = ENOMEM;
465 if (err_out != NULL) {
466 *err_out = ENOMEM;
467 }
468 *use_cred = 0;
469 return NGX_ERROR;
470 }
471 ngx_memcpy(path, tmp, path_len);
472 path[path_len] = '\0';
473
474 cln = ngx_pool_cleanup_add(ctx->pool, sizeof(*payload));
475 if (cln == NULL) {
476 (void) unlink(path); /* vfs-seam-allow: config-domain PASSTHROUGH proxy credential temp (not export storage) */
477 ngx_memzero(path, path_len);
478 errno = ENOMEM;
479 if (err_out != NULL) {
480 *err_out = ENOMEM;
481 }
482 *use_cred = 0;
483 return NGX_ERROR;
484 }
485 payload = cln->data;
486 payload->path = path;
487 cln->handler = brix_vfs_deleg_temp_cleanup;
488
489 cred->x509_proxy = path;
490 cred->mode = BRIX_CRED_PASSTHROUGH;
491 *use_cred = 1;
492
493 return NGX_OK;
494}

496/* ---- brix_vfs_deleg_live_cred ----------------------------------------------
497 *
498 * WHAT: Validate + materialise the ctx's bound PASSTHROUGH live cred into *cred.
499 * See the vfs_internal.h doc block for the full contract.
500 *
501 * WHY: The cred gate delegates here when brix_vfs_backend_mode() is PASSTHROUGH,
502 * so all the "bytes → cred form (or deny/fallback)" logic is in one place.
503 *
504 * HOW: Dispatch on which byte field the front door filled: a full proxy PEM
505 * (have_proxy_pem) routes to brix_vfs_deleg_proxy; a bearer routes to
506 * brix_vfs_deleg_bearer; neither present is a missing cred → deny/fallback. */
507ngx_int_t
508brix_vfs_deleg_live_cred(brix_vfs_ctx_t *ctx, brix_sd_cred_t *cred,
509 int *use_cred, int *err_out)
510{
511 brix_deleg_live_t *live;
512
513 *use_cred = 0;
514
515 if (ctx == NULL || ctx->deleg_live == NULL) {
516 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
517 }
518 live = ctx->deleg_live;
519
520 {
521 /* phase-71: credential-kind accept gate. Deny (EACCES, before any origin
522 * contact) when the live credential kind is not one the leaf backend can
523 * consume — e.g. a bearer-only backend handed a full x509 proxy, or vice
524 * versa. cred_accept==0 (no delegation support) rejects both kinds. */
525 uint32_t accept = brix_sd_cred_accept(brix_vfs_ns_leaf(ctx->sd));
526
527 if (live->have_proxy_pem && live->proxy_pem.len > 0
528 && live->proxy_pem.data != NULL) {
529 if (!(accept & BRIX_SD_CRED_PROXY_PEM)) {
530 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
531 }
532 return brix_vfs_deleg_proxy(ctx, cred, use_cred, err_out);
533 }
534
535 if (live->bearer.len > 0 && live->bearer.data != NULL) {
536 if (!(accept & BRIX_SD_CRED_BEARER)) {
537 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
538 }
539 /* EXCHANGE with a configured endpoint trades the subject token for a
540 * backend-audienced one; EXCHANGE with no endpoint (documented §5.4
541 * fallback) and plain PASSTHROUGH forward the bearer verbatim. */
542 if (live->mode == BRIX_CRED_EXCHANGE
543 && live->tx.endpoint.len > 0) {
544 return brix_vfs_deleg_exchange(ctx, cred, use_cred, err_out);
545 }
546 return brix_vfs_deleg_bearer(ctx, cred, use_cred);
547 }
548 }
549
550 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
551}

553/* ---- brix_vfs_deleg_sts_cred (call-ready hook, §5.5) -----------------------
554 *
555 * WHAT: Exchange the node's S3 service credential for temporary credentials
556 * scoped to the inbound identity via brix_s3_sts_assume(), and stamp the
557 * result onto cred->s3_ak/s3_sk/s3_region (mode EXCHANGE).
558 *
559 * WHY: An S3 SigV4 secret is never transmitted, so the origin cannot be handed
560 * the caller's key; STS is the EXCHANGE path (§5.5). This helper is the
561 * single seam where the STS result becomes the sd_remote-consumable cred
562 * form, mirroring brix_vfs_deleg_exchange for bearers.
563 *
564 * HOW: Calls brix_s3_sts_assume() with the ctx identity and the caller-supplied
565 * conf; on NGX_OK borrows the pool-copied ak/sk/region onto *cred. On
566 * failure → deny (never the service cred under fallback-deny). Secrets are
567 * never logged.
568 *
569 * DEFERRED (full origin-leg invocation): this compiles and is call-ready, but is
570 * NOT yet driven from brix_vfs_deleg_live_cred because (a) the STS conf
571 * (brix_s3_sts_conf_t: endpoint/region/role_arn + the node's svc_ak/svc_sk)
572 * is not reachable from brix_vfs_ctx_t — conf->common.backend_sts_endpoint/
573 * _role are set, but the SigV4 SERVICE key pair still needs a config source
574 * + a brix_vfs_ctx_bind_backend_sts() at the S3 capture site (owned by
575 * another agent); and (b) sd_remote's open_cred must map s3_ak/sk/session
576 * through to the origin keys. Wire (a)+(b), then call this from the
577 * EXCHANGE branch when the leaf backend accepts S3 creds. */
578ngx_int_t
579brix_vfs_deleg_sts_cred(brix_vfs_ctx_t *ctx, const brix_s3_sts_conf_t *cf,
580 brix_sd_cred_t *cred, int *use_cred, int *err_out)
581{
582 ngx_str_t ak = ngx_null_string;
583 ngx_str_t sk = ngx_null_string;
584 ngx_str_t session = ngx_null_string;
585 brix_s3_sts_out_t out = { &ak, &sk, &session };
586
587 *use_cred = 0;
588
589 if (ctx == NULL || cf == NULL) {
590 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
591 }
592
593 if (brix_s3_sts_assume(ctx->pool, ctx->identity, cf, &out, ctx->log)
594 != NGX_OK) {
595 ngx_log_error(NGX_LOG_ERR, ctx->log, 0,
596 "brix: S3 STS exchange failed - denying (no service-cred fallback "
597 "for EXCHANGE)");
598 return brix_vfs_deleg_deny(ctx, use_cred, err_out);
599 }
600
601 cred->s3_ak = (const char *) ak.data;
602 cred->s3_sk = (const char *) sk.data;
603 cred->s3_region = (cf->region.len > 0) ? (const char *) cf->region.data
604 : NULL;
605 cred->mode = BRIX_CRED_EXCHANGE;
606 *use_cred = 1;
607
608 return NGX_OK;
609}

612 *
613 * WHAT: Initiate the first GSSAPI leg to the origin AS the inbound user, using a

628 * captures them in the session ctx — reaching them needs a bind at the
629 * stream capture site, owned by another agent); and (b) the multi-leg GSS
630 * negotiation belongs to origin_auth.c, which must feed origin replies back
631 * through gss_init_sec_context. Thread the delegated cred onto the ctx and
632 * add the origin-auth krb5-forward leg, then call this from the gate. */
633ngx_int_t
634brix_vfs_deleg_krb5_token(brix_vfs_ctx_t *ctx, void *deleg_gss_cred,

639 }
640

642 ngx_log_error(NGX_LOG_INFO, ctx->log, 0,
643 "brix: krb5 credential forwarding unavailable - falling back to "
644 "SELECT");
645 return NGX_ERROR;
646 }
647
648 return brix_krb5_deleg_to_origin(ctx->pool, deleg_gss_cred,
649 origin_service_princ, out_token, ctx->log);
650}
