4/*
5 * WHAT: XRootD upstream bootstrap sequence — handshake, protocol negotiation, TLS upgrade detection,
6 * and login authentication for transparent proxy mode.
7 * WHY: When nginx connects to a backend XRootD server, it must complete the initial bootstrap protocol
8 * before sending client requests. This involves four phases in order: handshake (12 zero bytes +
9 * version markers), protocol request (streamid/kXR_protocol/kXR_PROTOCOLVERSION), TLS upgrade check
10 * (kXR_gotoTLS flag detection → ngx_ssl handshake callback), and login (pid/username/capver).
11 * kXR_authmore responses trigger token credential exchange via upstream_token_file.
12 * HOW: brix_upstream_build_bootstrap() concatenates handshake zeros, protocol request struct, and
13 * login request struct into a single wire buffer. brix_upstream_handle_bootstrap_response()
14 * implements the state machine across XRD_UP_BS_HANDSHAKE → BS_PROTOCOL → BS_TLS (optional) →
15 * BS_LOGIN → BS_DONE, checking resp_status at each phase transition. bs_phase enum tracks current
16 * stage; on completion, brix_upstream_send_request() is called to forward the saved client request.
17 */
18

23/*
24 * A server-internal connector owns a fixed streamid {0,1} and presents the
25 * anonymous "xrd" identity with no TLS-capability flags in the protocol request
26 * (TLS, when required, is driven by the server's kXR_gotoTLS response, not by a
27 * client-side wantTLS flag). The wire layout itself lives in bootstrap_pack.h.
28 */
29static const uint8_t brix_upstream_streamid[2] = { 0, 1 };

61/*
62 * brix_upstream_continue_auth — respond to a kXR_authmore challenge during the
63 * outbound bootstrap (cache-fill origin), bounded by XRD_OBA_MAX_ROUNDS.
64 *
65 * WHAT: increments the per-connection authmore counter, aborts once it reaches the
66 * bound (hostile/misconfigured origin → no unbounded loop), else sends the ztn
67 * token credential and moves to the AUTH wait phase.
68 * WHY: the origin's sec layer may issue more than one kXR_authmore; the former
69 * hard "second authmore → abort" both blocked any legitimate multi-round
70 * continuation and duplicated the LOGIN/AUTH handling. One bounded helper closes
71 * the hostile-loop security case and is the single seam a future GSI cache-fill
72 * continuation hooks into.
73 * HOW: bound check → counter++ → require brix_upstream_token_file → delegate to
74 * brix_upstream_send_token_auth (which arms the next read in XRD_UP_BS_AUTH).
75 * Every failure path routes through brix_upstream_abort.
76 */
77static void
78brix_upstream_continue_auth(brix_upstream_t *up)
79{
80 ngx_stream_brix_srv_conf_t *conf;
81
82 if (up->authmore_count >= XRD_OBA_MAX_ROUNDS) {
83 brix_upstream_abort(up,
84 "upstream: too many kXR_authmore rounds (bounded)");
85 return;
86 }
87 up->authmore_count++;
88
89 conf = ngx_stream_get_module_srv_conf(
90 up->client_ctx->session, ngx_stream_brix_module);
91
92 if (conf->upstream_token_file.len == 0) {
93 brix_upstream_abort(up,
94 "upstream requires auth; set brix_upstream_token_file");
95 return;
96 }
97 if (brix_upstream_send_token_auth(up, conf) != NGX_OK) {
98 brix_upstream_abort(up, "upstream: token auth exchange failed");
99 }
100}

102void
103brix_upstream_handle_bootstrap_response(brix_upstream_t *up)
104{
105 switch (up->bs_phase) {
106
107 case XRD_UP_BS_HANDSHAKE:
108 /*
109 * Phase 1 consumes the server's fixed handshake response. No request
110 * payload is interpreted until the upstream confirms the handshake.
111 */
112 if (up->resp_status != kXR_ok) {
113 brix_upstream_abort(up, "upstream: bad handshake response");
114 return;
115 }
116 up->bs_phase = XRD_UP_BS_PROTOCOL;
117 break;
118
119 case XRD_UP_BS_PROTOCOL: {
120 ngx_stream_brix_srv_conf_t *conf;
121
122 if (up->resp_status != kXR_ok) {
123 brix_upstream_abort(up, "upstream: protocol response not ok");
124 return;
125 }
126 if (up->resp_dlen >= 8) {
127 uint32_t flags_be;
128
129 ngx_memcpy(&flags_be, up->resp_body + 4, sizeof(flags_be));
130
131 if (ntohl(flags_be) & kXR_gotoTLS) {
132 conf = ngx_stream_get_module_srv_conf(
133 up->client_ctx->session, ngx_stream_brix_module);
134
135#if (NGX_SSL)
136 if (!conf->upstream_tls || conf->upstream_tls_ctx == NULL) {
137 brix_upstream_abort(up,
138 "upstream requires TLS; set brix_upstream_tls on");
139 return;
140 }
141 /*
142 * Upgrade the existing plaintext connection to TLS.
143 * The kXR_login frame already in the socket buffer is
144 * discarded by the server upon receiving gotoTLS; we resend
145 * it over TLS from the handshake callback.
146 */
147 if (brix_upstream_start_tls(up, conf) != NGX_OK) {
148 brix_upstream_abort(up,
149 "upstream: TLS upgrade failed");
150 }
151#else
152 (void) conf;
153 brix_upstream_abort(up,
154 "upstream requires TLS but nginx was built without SSL");
155#endif
156 return; /* resume in TLS handshake callback */
157 }
158 }
159 up->bs_phase = XRD_UP_BS_LOGIN;
160 break;
161 }
162
163 case XRD_UP_BS_TLS:
164 /*
165 * Should not arrive here: the TLS handshake callback transitions
166 * directly to XRD_UP_BS_LOGIN before arming the read event.
167 */
168 brix_upstream_abort(up, "upstream: unexpected read in TLS phase");
169 return;
170
171 case XRD_UP_BS_LOGIN:
172 if (up->resp_status == kXR_authmore) {
173 brix_upstream_continue_auth(up);
174 return; /* resume after write + read cycle */
175 }
176 if (up->resp_status != kXR_ok) {
177 brix_upstream_abort(up, "upstream: login failed");
178 return;
179 }
180 up->bs_phase = XRD_UP_BS_DONE;
181 break;
182
183 case XRD_UP_BS_AUTH:
184 /*
185 * kXR_auth response after we sent a ztn token credential.
186 * kXR_ok → authenticated, proceed to send the request.
187 * kXR_authmore → the origin wants another round; continue the bounded
188 * exchange (XRD_OBA_MAX_ROUNDS) rather than fail outright.
189 * anything else → reject.
190 */
191 if (up->resp_status == kXR_authmore) {
192 brix_upstream_continue_auth(up);
193 return;
194 }
195 if (up->resp_status != kXR_ok) {
196 brix_upstream_abort(up, "upstream: token auth rejected by server");
197 return;
198 }
199 up->bs_phase = XRD_UP_BS_DONE;
200 break;
201
202 default:
203 brix_upstream_abort(up, "upstream: invalid bootstrap phase");
204 return;
205 }
206
207 up->rhdr_pos = 0;
208 up->resp_dlen = 0;
209 up->resp_body = NULL;
210 up->resp_body_pos = 0;
211
212 if (up->bs_phase == XRD_UP_BS_DONE) {
213 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, up->client_conn->log, 0,
214 "brix: upstream bootstrap done; sending request");
215 if (brix_upstream_send_request(up) == NGX_ERROR) {
216 brix_upstream_abort(up, "upstream: send request failed");
217 }
218 return;
219 }
220
221 if (ngx_handle_read_event(up->conn->read, 0) != NGX_OK) {
222 brix_upstream_abort(up, "upstream: read event arm failed");
223 return;
224 }
225 /* In epoll ET mode, if multiple bootstrap responses arrived in the same
226 * TCP segment, the remaining data in the socket buffer won't re-trigger
227 * EPOLLIN. Post a synthetic read event so the next phase is processed
228 * in the current event-loop cycle rather than waiting for new data. */
229 ngx_post_event(up->conn->read, &ngx_posted_events);
230}
