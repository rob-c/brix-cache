4/*
5 * WHAT: XRootD upstream bootstrap sequence — handshake, protocol negotiation, TLS upgrade detection,
6 * and login authentication for transparent proxy mode.
7 * WHY: When nginx connects to a backend XRootD server, it must complete the initial bootstrap protocol
8 * before sending client requests. This involves four phases in order: handshake (12 zero bytes +
9 * version markers), protocol request (streamid/kXR_protocol/kXR_PROTOCOLVERSION), TLS upgrade check
10 * (kXR_gotoTLS flag detection → ngx_ssl handshake callback), and login (pid/username/capver).
11 * kXR_authmore responses trigger token credential exchange via upstream_token_file.
12 * HOW: brix_upstream_build_bootstrap() concatenates handshake zeros, protocol request struct, and
13 * login request struct into a single wire buffer. brix_upstream_handle_bootstrap_response()
14 * implements the state machine across XRD_UP_BS_HANDSHAKE → BS_PROTOCOL → BS_TLS (optional) →
15 * BS_LOGIN → BS_DONE, checking resp_status at each phase transition. bs_phase enum tracks current
16 * stage; on completion, brix_upstream_send_request() is called to forward the saved client request.
17 */
18

23/*
24 * A server-internal connector owns a fixed streamid {0,1} and presents the
25 * anonymous "xrd" identity. By default the protocol request carries no
26 * TLS-capability flags (TLS, when required, is driven by the server's
27 * kXR_gotoTLS response, not by a client-side wantTLS flag); a caller that CAN
28 * upgrade — one holding an outbound TLS ctx, e.g. the Step-F health probe —
29 * uses the _flags variant to advertise kXR_ableTLS, since a brix server only
30 * answers kXR_gotoTLS to TLS-capable clients. The wire layout itself lives in
31 * bootstrap_pack.h.
32 */
33static const uint8_t brix_upstream_streamid[2] = { 0, 1 };

71/*
72 * brix_upstream_continue_auth — respond to a kXR_authmore challenge during the
73 * outbound bootstrap (cache-fill origin), bounded by XRD_OBA_MAX_ROUNDS.
74 *
75 * WHAT: increments the per-connection authmore counter, aborts once it reaches the
76 * bound (hostile/misconfigured origin → no unbounded loop), else sends the ztn
77 * token credential and moves to the AUTH wait phase.
78 * WHY: the origin's sec layer may issue more than one kXR_authmore; the former
79 * hard "second authmore → abort" both blocked any legitimate multi-round
80 * continuation and duplicated the LOGIN/AUTH handling. One bounded helper closes
81 * the hostile-loop security case and is the single seam a future GSI cache-fill
82 * continuation hooks into.
83 * HOW: bound check → counter++ → require brix_upstream_token_file → delegate to
84 * brix_upstream_send_token_auth (which arms the next read in XRD_UP_BS_AUTH).
85 * Every failure path routes through brix_upstream_abort.
86 */
87static void
88brix_upstream_continue_auth(brix_upstream_t *up)
89{
90 ngx_stream_brix_srv_conf_t *conf;
91
92 if (up->authmore_count >= XRD_OBA_MAX_ROUNDS) {
93 brix_upstream_abort(up,
94 "upstream: too many kXR_authmore rounds (bounded)");
95 return;
96 }
97 up->authmore_count++;
98
99 conf = ngx_stream_get_module_srv_conf(
100 up->client_ctx->session, ngx_stream_brix_module);
101
102 if (conf->upstream_token_file.len == 0) {
103 brix_upstream_abort(up,
104 "upstream requires auth; set brix_upstream_token_file");
105 return;
106 }
107 if (brix_upstream_send_token_auth(up, conf) != NGX_OK) {
108 brix_upstream_abort(up, "upstream: token auth exchange failed");
109 }
110}

112/*
113 * Per-phase step contract: a phase handler returns NGX_OK when it has advanced
114 * up->bs_phase and the shared tail (brix_upstream_bootstrap_advance) should run,
115 * or NGX_DONE when it has already terminated the cycle itself — either by
116 * aborting the connection or by deferring to a callback (TLS handshake, token
117 * auth write/read) — in which case the caller must return without touching the
118 * response buffers or read event.
119 */
120
121/*
122 * brix_upstream_bs_handshake — phase 1, consume the server's fixed handshake
123 * response.
124 *
125 * WHAT: fail the connection unless resp_status is kXR_ok, else advance to the
126 * protocol-negotiation phase.
127 * WHY: no request payload is interpreted until the upstream confirms the
128 * handshake, so a bad status here is fatal.
129 * HOW: status gate → set XRD_UP_BS_PROTOCOL → NGX_OK.
130 */
131static ngx_int_t
132brix_upstream_bs_handshake(brix_upstream_t *up)
133{
134 if (up->resp_status != kXR_ok) {
135 brix_upstream_abort(up, "upstream: bad handshake response");
136 return NGX_DONE;
137 }
138 up->bs_phase = XRD_UP_BS_PROTOCOL;
139 return NGX_OK;
140}

142/*
143 * brix_upstream_bs_goto_tls — handle a kXR_gotoTLS flag seen in the protocol
144 * response by upgrading the plaintext connection to TLS.
145 *
146 * WHAT: resolve the srv conf, then (when built with SSL) require an upstream TLS
147 * context and drive brix_upstream_start_tls; without SSL support the request
148 * is fatal.
149 * WHY: TLS is server-driven (kXR_gotoTLS), not client-requested; the kXR_login
150 * frame already pre-sent in the socket buffer is discarded by the server and
151 * is resent over TLS from the handshake callback.
152 * HOW: get conf → SSL-gated capability check → start_tls; always returns
153 * NGX_DONE because the cycle either aborts or resumes in the TLS callback.
154 */
155static ngx_int_t
156brix_upstream_bs_goto_tls(brix_upstream_t *up)
157{
158 ngx_stream_brix_srv_conf_t *conf;
159
160 conf = ngx_stream_get_module_srv_conf(
161 up->client_ctx->session, ngx_stream_brix_module);
162
163#if (NGX_SSL)
164 if (!conf->upstream_tls || conf->upstream_tls_ctx == NULL) {
165 brix_upstream_abort(up,
166 "upstream requires TLS; set brix_upstream_tls on");
167 return NGX_DONE;
168 }
169 if (brix_upstream_start_tls(up, conf) != NGX_OK) {
170 brix_upstream_abort(up, "upstream: TLS upgrade failed");
171 }
172#else
173 (void) conf;
174 brix_upstream_abort(up,
175 "upstream requires TLS but nginx was built without SSL");
176#endif
177 return NGX_DONE; /* resume in TLS handshake callback (or aborted) */
178}

180/*
181 * brix_upstream_bs_protocol — phase 2, protocol negotiation and TLS-upgrade
182 * detection.
183 *
184 * WHAT: fail unless resp_status is kXR_ok; if the response carries the flags
185 * word (dlen >= 8) and it sets kXR_gotoTLS, hand off to the TLS upgrade path,
186 * otherwise advance to the login phase.
187 * WHY: the server signals a mandatory TLS upgrade via a flag in the protocol
188 * response; only after ruling that out is a plaintext login valid.
189 * HOW: status gate → extract big-endian flags at offset 4 → gotoTLS branch →
190 * else set XRD_UP_BS_LOGIN → NGX_OK.
191 */
192static ngx_int_t
193brix_upstream_bs_protocol(brix_upstream_t *up)
194{
195 if (up->resp_status != kXR_ok) {
196 brix_upstream_abort(up, "upstream: protocol response not ok");
197 return NGX_DONE;
198 }
199 if (up->resp_dlen >= 8) {
200 uint32_t flags_be;
201
202 ngx_memcpy(&flags_be, up->resp_body + 4, sizeof(flags_be));
203
204 if (ntohl(flags_be) & kXR_gotoTLS) {
205 return brix_upstream_bs_goto_tls(up);
206 }
207 }
208 up->bs_phase = XRD_UP_BS_LOGIN;
209 return NGX_OK;
210}

212/*
213 * brix_upstream_bs_login — phase 3, evaluate the login response.
214 *
215 * WHAT: a kXR_authmore status continues the bounded credential exchange; a
216 * non-ok status is fatal; kXR_ok advances to the done phase.
217 * WHY: the origin's sec layer may demand a token round before granting login;
218 * only a clean kXR_ok means the session is ready to carry client requests.
219 * HOW: authmore → brix_upstream_continue_auth (defer) → status gate → set
220 * XRD_UP_BS_DONE → NGX_OK.
221 */
222static ngx_int_t
223brix_upstream_bs_login(brix_upstream_t *up)
224{
225 if (up->resp_status == kXR_authmore) {
226 brix_upstream_continue_auth(up);
227 return NGX_DONE; /* resume after write + read cycle */
228 }
229 if (up->resp_status != kXR_ok) {
230 brix_upstream_abort(up, "upstream: login failed");
231 return NGX_DONE;
232 }
233 up->bs_phase = XRD_UP_BS_DONE;
234 return NGX_OK;
235}

237/*
238 * brix_upstream_bs_auth — phase 4, evaluate the kXR_auth response after a ztn
239 * token credential was sent.
240 *
241 * WHAT: kXR_ok → authenticated, advance to done; kXR_authmore → the origin wants
242 * another bounded round (XRD_OBA_MAX_ROUNDS) rather than an outright failure;
243 * anything else → reject.
244 * WHY: multi-round token exchanges are legitimate but must stay bounded so a
245 * hostile/misconfigured origin cannot loop us forever.
246 * HOW: authmore → brix_upstream_continue_auth (defer) → status gate → set
247 * XRD_UP_BS_DONE → NGX_OK.
248 */
249static ngx_int_t
250brix_upstream_bs_auth(brix_upstream_t *up)
251{
252 if (up->resp_status == kXR_authmore) {
253 brix_upstream_continue_auth(up);
254 return NGX_DONE;
255 }
256 if (up->resp_status != kXR_ok) {
257 brix_upstream_abort(up, "upstream: token auth rejected by server");
258 return NGX_DONE;
259 }
260 up->bs_phase = XRD_UP_BS_DONE;
261 return NGX_OK;
262}

264/*
265 * brix_upstream_bootstrap_advance — shared tail after a phase advances.
266 *
267 * WHAT: reset the response-frame cursors, then either send the saved client
268 * request (bootstrap complete) or re-arm the read event for the next phase.
269 * WHY: every successful phase transition needs a fresh frame buffer, and epoll
270 * ET mode can leave already-buffered bytes without a new EPOLLIN — a synthetic
271 * posted read event drains them in the current loop cycle.
272 * HOW: zero the cursors → XRD_UP_BS_DONE branch sends the request → else arm the
273 * read event and post a synthetic read.
274 */
275static void
276brix_upstream_bootstrap_advance(brix_upstream_t *up)
277{
278 up->rhdr_pos = 0;
279 up->resp_dlen = 0;
280 up->resp_body = NULL;
281 up->resp_body_pos = 0;
282
283 if (up->bs_phase == XRD_UP_BS_DONE) {
284 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, up->client_conn->log, 0,
285 "brix: upstream bootstrap done; sending request");
286 if (brix_upstream_send_request(up) == NGX_ERROR) {
287 brix_upstream_abort(up, "upstream: send request failed");
288 }
289 return;
290 }
291
292 if (ngx_handle_read_event(up->conn->read, 0) != NGX_OK) {
293 brix_upstream_abort(up, "upstream: read event arm failed");
294 return;
295 }
296 /* In epoll ET mode, if multiple bootstrap responses arrived in the same
297 * TCP segment, the remaining data in the socket buffer won't re-trigger
298 * EPOLLIN. Post a synthetic read event so the next phase is processed
299 * in the current event-loop cycle rather than waiting for new data. */
300 ngx_post_event(up->conn->read, &ngx_posted_events);
301}

303/*
304 * brix_upstream_handle_bootstrap_response — bootstrap state-machine dispatcher.
305 *
306 * WHAT: route the current bs_phase to its handler; on a clean advance run the
307 * shared tail, otherwise let the handler's own termination stand.
308 * WHY: keeps the four wire phases (handshake → protocol/TLS → login → auth) as
309 * independent, testable stages while preserving the exact original transitions
310 * and error paths.
311 * HOW: switch on bs_phase → per-phase helper returns NGX_OK (advanced) or
312 * NGX_DONE (already terminated) → NGX_OK runs brix_upstream_bootstrap_advance.
313 */
314void
315brix_upstream_handle_bootstrap_response(brix_upstream_t *up)
316{
317 ngx_int_t rc;
318
319 switch (up->bs_phase) {
320
321 case XRD_UP_BS_HANDSHAKE:
322 rc = brix_upstream_bs_handshake(up);
323 break;
324
325 case XRD_UP_BS_PROTOCOL:
326 rc = brix_upstream_bs_protocol(up);
327 break;
328
329 case XRD_UP_BS_TLS:
330 /*
331 * Should not arrive here: the TLS handshake callback transitions
332 * directly to XRD_UP_BS_LOGIN before arming the read event.
333 */
334 brix_upstream_abort(up, "upstream: unexpected read in TLS phase");
335 return;
336
337 case XRD_UP_BS_LOGIN:
338 rc = brix_upstream_bs_login(up);
339 break;
340
341 case XRD_UP_BS_AUTH:
342 rc = brix_upstream_bs_auth(up);
343 break;
344
345 default:
346 brix_upstream_abort(up, "upstream: invalid bootstrap phase");
347 return;
348 }
349
350 if (rc != NGX_OK) {
351 return; /* handler aborted or deferred to a callback */
352 }
353
354 brix_upstream_bootstrap_advance(up);
355}
