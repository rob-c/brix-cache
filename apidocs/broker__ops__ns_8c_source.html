1/*
2 * broker_ops_ns.c — parent-relative namespace-mutation broker ops.
3 *
4 * WHAT: The impersonation-broker file operations that act on a final path
5 * component under its confined PARENT directory: mkdir, unlink/rmdir, chmod,
6 * chown, rename/rename-noreplace/link, setattr (utimens + chown), symlink and
7 * readlink. Every handler returns 0 or -errno and is dispatched by the opcode
8 * table (imp_op_table) that lives in broker_ops.c.
9 *
10 * WHY: Split out of broker_ops.c (phase-79) to keep both files under the
11 * 500-line cap. These ops share one shape — open the parent via
12 * imp_open_parent (RESOLVE_BENEATH), operate on the final component as the
13 * mapped user — so they and their private imp_with_parent / imp_step_*
14 * machinery form a single cohesive concern. The fd-based ops (open/stat/
15 * truncate/xattr), the confined primitives, and the dispatch entry stay in
16 * broker_ops.c; only the eight table-referenced handlers are non-static here.
17 *
18 * HOW: imp_with_parent wraps the open-parent / run-step / close boilerplate for
19 * the single-path ops (mkdir/unlink/chmod/chown), each expressed as a one-line
20 * imp_step_* syscall; the multi-path ops (rename/link) and the payload ops
21 * (setattr/symlink/readlink) open their parent(s) directly. Behaviour is
22 * identical to the pre-split broker_ops.c.
23 */
24#include "broker_internal.h"
25#include "broker_ops_internal.h"

37/* ---- imp_with_parent — run one syscall step under the confined parent ----
38 *
39 * WHAT: Opens the parent directory of c->rel beneath c->rootfd via
40 * imp_open_parent, runs `step` on (pfd, base, c) and returns its result
41 * (0 or -errno) after closing the parent fd; a failed parent open returns
42 * its -errno without calling `step`.
43 *
44 * WHY: Every single-path parent-relative op (mkdir/unlink/rmdir/chmod/chown)
45 * repeated the identical open-parent / early-return / close boilerplate; one
46 * shared wrapper keeps the confinement and fd-lifetime rules in exactly one
47 * place so a per-op handler is only its syscall.
48 *
49 * HOW: 1. imp_open_parent into a local scratch buffer; 2. early-return the
50 * negative errno on open failure; 3. run `step`; 4. close(pfd); 5. return
51 * the step's rc.
52 */
53static int
54imp_with_parent(const imp_op_ctx_t *c, imp_parent_fn_t step)
55{
56 char scratch[IMP_PATH_MAX];
57 const char *base;
58 int pfd, rc;
59
60 pfd = imp_open_parent(c->rootfd, c->rel, scratch, &base);
61 if (pfd < 0) {
62 return pfd;
63 }
64 rc = step(pfd, base, c);
65 close(pfd);
66 return rc;
67}

150/*
151 * imp_op_rename_link — IMP_OP_RENAME / IMP_OP_RENAME_NOREPLACE / IMP_OP_LINK:
152 * two-path ops. HOW: confine BOTH parents via imp_open_parent, then linkat or
153 * imp_do_rename (which handles the RENAME_NOREPLACE renameat2 fallback).
154 */
155int
156imp_op_rename_link(const imp_op_ctx_t *c)
157{
158 const char *rel2 = imp_rel(c->req->path2);
159 char scratch[IMP_PATH_MAX], scratch2[IMP_PATH_MAX];
160 const char *base, *base2;
161 int pfd, pfd2, rc;
162
163 pfd = imp_open_parent(c->rootfd, c->rel, scratch, &base);
164 if (pfd < 0) {
165 return pfd;
166 }
167 pfd2 = imp_open_parent(c->rootfd, rel2, scratch2, &base2);
168 if (pfd2 < 0) {
169 close(pfd);
170 return pfd2;
171 }
172 if (c->req->op == IMP_OP_LINK) {
173 rc = linkat(pfd, base, pfd2, base2, 0) == 0 ? 0 : -errno;
174 } else {
175 rc = imp_do_rename(pfd, base, pfd2, base2,
176 c->req->op == IMP_OP_RENAME_NOREPLACE) == 0
177 ? 0 : -errno;
178 }
179 close(pfd);
180 close(pfd2);
181 return rc;
182}

185/*
186 * imp_op_setattr — IMP_OP_SETATTR: utimensat (+ optional fchownat) on the
187 * final component, NOFOLLOW. As the mapped user: chgrp/utimens on a file
188 * they own succeeds; a chown to a different owner correctly fails (the broker
189 * holds no CAP_CHOWN).
190 */
191int
192imp_op_setattr(const imp_op_ctx_t *c)
193{
194 char scratch[IMP_PATH_MAX];
195 const char *base;
196 int pfd, rc;
197
198 pfd = imp_open_parent(c->rootfd, c->rel, scratch, &base);
199 if (pfd < 0) {
200 return pfd;
201 }
202 rc = 0;
203 if (c->req->attr_flags & IMP_ATTR_TIMES) {
204 struct timespec ts[2];
205 ts[0].tv_sec = (time_t) c->req->atime_sec;
206 ts[0].tv_nsec = (long) c->req->atime_nsec;
207 ts[1].tv_sec = (time_t) c->req->mtime_sec;
208 ts[1].tv_nsec = (long) c->req->mtime_nsec;
209 if (utimensat(pfd, base, ts, AT_SYMLINK_NOFOLLOW) != 0) {
210 rc = -errno;
211 }
212 }
213 if (rc == 0 && (c->req->attr_flags & IMP_ATTR_OWNER)) {
214 uid_t u = (c->req->set_uid == (uint32_t) -1) ? (uid_t) -1
215 : (uid_t) c->req->set_uid;
216 gid_t g = (c->req->set_gid == (uint32_t) -1) ? (gid_t) -1
217 : (gid_t) c->req->set_gid;
218 if (fchownat(pfd, base, u, g, AT_SYMLINK_NOFOLLOW) != 0) {
219 rc = -errno;
220 }
221 }
222 close(pfd);
223 return rc;
224}
