1/*
2 * gsi_verify.c — shared x.509 proxy chain verification core.
3 *
4 * Both the XRootD stream (GSI kXR_auth) and WebDAV (DAVS client certificate)
5 * authentication paths perform identical X509_STORE_CTX setup and chain
6 * verification. Centralising here means any bug fix or security improvement
7 * (flag addition, depth handling, error logging) automatically covers all
8 * callers.
9 *
10 * WHAT: Wraps OpenSSL X509_STORE_CTX lifecycle — allocate, initialise with
11 * leaf + untrusted chain + CA store, set ALLOW_PROXY_CERTS flag,
12 * optionally set depth, call X509_verify_cert, extract leaf subject DN.
13 *
14 * WHY: The stream and WebDAV paths previously duplicated this ~25-line block
15 * verbatim. A single defect (wrong flag, missing depth, wrong error
16 * logging) would affect only one path, causing inconsistent security
17 * behaviour between the two protocols.
18 *
19 * HOW: See brix_gsi_verify_chain() below. The caller retains ownership of
20 * all passed-in objects; none are freed here.
21 */
22

32/*
33 * WHAT: enforce limited-proxy monotonicity over the verified chain.
34 * WHY: RFC 3820 §3.8 — a limited proxy must not be used to issue a full
35 * (unrestricted) proxy; doing so escalates privilege.
36 * HOW: Delegate to the ngx-free brix_proxy_chain_ok() so the pure logic is
37 * unit-testable; log and map its verdict to NGX_OK/NGX_ERROR here.
38 */
39static ngx_int_t
40brix_gsi_enforce_proxy_monotonicity(X509_STORE_CTX *vctx, ngx_log_t *log)
41{
42 if (!brix_proxy_chain_ok(X509_STORE_CTX_get0_chain(vctx))) {
43 ngx_log_error(NGX_LOG_WARN, log, 0,
44 "brix: GSI proxy chain rejected: a full proxy is issued beneath a "
45 "limited proxy (RFC 3820 delegation escalation)");
46 return NGX_ERROR;
47 }
48 return NGX_OK;
49}

51/*
52 * WHAT: enforce Globus signing_policy over a verified chain.
53 * WHY: PKIX validation proves each cert was signed by its issuer, but the
54 * WLCG trust model additionally requires a CA sign only within its
55 * delegated namespace. A trusted CA that signs outside its namespace
56 * passes X509_verify_cert yet must be rejected here.
57 * HOW: Walk the ordered chain leaf..root. For each non-proxy subject, ask
58 * the store's attached policy table whether its issuer (the next cert)
59 * is permitted to sign it. Proxy subjects are exempt — their issuer is
60 * the EEC/parent proxy, never a trust anchor. Returns NGX_OK when every
61 * link is permitted, NGX_ERROR (logged) on the first violation.
62 */
63static ngx_int_t
64brix_gsi_enforce_signing_policy(X509_STORE_CTX *vctx, ngx_log_t *log)
65{
66 brix_sp_table_t *table = brix_store_policy_table(vctx);
67 brix_sp_mode_t mode = brix_store_policy_mode(vctx);
68 STACK_OF(X509) *chain;
69 int n, i;
70
71 if (mode == BRIX_SP_MODE_OFF) {
72 return NGX_OK;
73 }
74
75 chain = X509_STORE_CTX_get0_chain(vctx); /* borrowed; do not free */
76 if (chain == NULL) {
77 return NGX_OK;
78 }
79
80 n = sk_X509_num(chain);
81 for (i = 0; i + 1 < n; i++) {
82 X509 *subject = sk_X509_value(chain, i);
83 X509 *issuer = sk_X509_value(chain, i + 1);
84
85 if (X509_get_extension_flags(subject) & EXFLAG_PROXY) {
86 continue; /* proxy: issuer is the EEC, not a trust anchor */
87 }
88
89 if (!brix_sp_table_check(table, mode, issuer, subject)) {
90 char subj_dn[1024];
91 char ca_dn[1024];
92
93 brix_x509_oneline(X509_get_subject_name(subject),
94 subj_dn, sizeof(subj_dn));
95 brix_x509_oneline(X509_get_subject_name(issuer),
96 ca_dn, sizeof(ca_dn));
97 ngx_log_error(NGX_LOG_WARN, log, 0,
98 "brix: GSI signing_policy: CA \"%s\" may not sign subject "
99 "\"%s\" (namespace violation or missing policy)",
100 ca_dn, subj_dn);
101 return NGX_ERROR;
102 }
103 }
104
105 return NGX_OK;
106}

108/*
109 * WHAT: hold every cert in the verified chain to the per-cert conformance
110 * policy, reject invalid proxyCertInfo, and (client_purpose) reject a
111 * leaf that is not usable for client authentication.
112 * HOW: delegate to the ngx-free brix_cert_policy_violation / brix_proxy_pci_ok
113 * / brix_leaf_purpose_violation; log and map to NGX_OK/NGX_ERROR.
114 */
115static ngx_int_t
116brix_gsi_enforce_cert_policy(X509_STORE_CTX *vctx, X509 *leaf, ngx_log_t *log,
117 int client_purpose)
118{
119 STACK_OF(X509) *chain = X509_STORE_CTX_get0_chain(vctx);
120 int n, i;
121
122 n = chain ? sk_X509_num(chain) : 0;
123 for (i = 0; i < n; i++) {
124 if (brix_cert_policy_violation(sk_X509_value(chain, i))) {
125 ngx_log_error(NGX_LOG_WARN, log, 0,
126 "brix: GSI cert rejected: weak key/algorithm, malformed serial, "
127 "or invalid DN in the chain (RFC 5280 / IGTF profile)");
128 return NGX_ERROR;
129 }
130 }
131 if (!brix_proxy_pci_ok(chain)) {
132 ngx_log_error(NGX_LOG_WARN, log, 0,
133 "brix: GSI proxy rejected: proxyCertInfo not critical or unknown "
134 "policy language (RFC 3820 §3.1/§3.2)");
135 return NGX_ERROR;
136 }
137 if (client_purpose && brix_leaf_purpose_violation(leaf)) {
138 ngx_log_error(NGX_LOG_WARN, log, 0,
139 "brix: client cert rejected: extendedKeyUsage lacks clientAuth or "
140 "keyUsage lacks digitalSignature (RFC 5280 §4.2.1.12/§4.2.1.3)");
141 return NGX_ERROR;
142 }
143 return NGX_OK;
144}

146/*
147 * WHAT: Verify an x.509 proxy certificate chain against a CA trust store.
148 *
149 * HOW (step by step):
150 * 1. Allocate a fresh X509_STORE_CTX (never reuse across requests).
151 * 2. Initialise it: store = CA trust store, subject = leaf, untrusted = intermediates.
152 * 3. Set X509_V_FLAG_ALLOW_PROXY_CERTS so proxy certificates (RFC 3820) in
153 * the chain are accepted — required for WLCG proxy credentials.
154 * 4. Optionally set chain depth via X509_STORE_CTX_set_depth when the
155 * caller has a configured limit (verify_depth > 0).
156 * 5. Call X509_verify_cert(); on failure extract the error code, log it, and
157 * return NGX_ERROR.
158 * 6. On success extract the subject DN from the leaf via X509_NAME_oneline,
159 * copy to res->dn_buf (truncating safely if oversized), free the
160 * OpenSSL-allocated string, and return NGX_OK.
161 */
162ngx_int_t
163brix_gsi_verify_chain(ngx_log_t *log, X509_STORE *store,
164 X509 *leaf, STACK_OF(X509) *untrusted,
165 ngx_uint_t verify_depth,
166 brix_gsi_verify_result_t *res,
167 int client_purpose)
168{
169 X509_STORE_CTX *vctx;
170 char *dn_str;
171 int ok;
172
173 ngx_memzero(res, sizeof(*res));
174
175 vctx = X509_STORE_CTX_new();
176 if (vctx == NULL) {
177 ngx_log_error(NGX_LOG_ERR, log, 0,
178 "brix: GSI: X509_STORE_CTX_new() failed");
179 return NGX_ERROR;
180 }
181
182 if (!X509_STORE_CTX_init(vctx, store, leaf, untrusted)) {
183 ngx_log_error(NGX_LOG_ERR, log, 0,
184 "brix: GSI: X509_STORE_CTX_init() failed");
185 X509_STORE_CTX_free(vctx);
186 return NGX_ERROR;
187 }
188
189 /* RFC 3820 proxy chains are accepted on BOTH the GSI (root://) and the
190 * WebDAV davs:// paths: WLCG/grid clients present voms-proxy-init proxy
191 * credentials over davs:// (gfal2 / davix / FTS) exactly as over root://,
192 * and the gsi-8444 WebDAV suite authenticates with an RFC 3820 proxy. The
193 * client_purpose flag still drives the extra cert-conformance/purpose checks
194 * below (brix_gsi_enforce_cert_policy) — it must NOT gate proxy acceptance,
195 * which regressed davs:// GSI auth to 403 (X509_V_ERR_PROXY_CERTIFICATES_
196 * NOT_ALLOWED at X509_verify_cert). Malformed/untrusted/expired proxies are
197 * still rejected by normal chain verification and the proxy-PCI/monotonicity
198 * enforcement that follows. */
199 X509_STORE_CTX_set_flags(vctx, X509_V_FLAG_ALLOW_PROXY_CERTS);
200
201 if (verify_depth > 0) {
202 X509_STORE_CTX_set_depth(vctx, (int) verify_depth);
203 }
204
205 ok = X509_verify_cert(vctx);
206 if (!ok) {
207 int verr = X509_STORE_CTX_get_error(vctx);
208 const char *verr_str = X509_verify_cert_error_string(verr);
209
210 BRIX_DIAG_WARN(log, 0,
211 "brix: GSI client cert rejected: %s",
212 "the client proxy/cert is expired, or its issuing CA is not in "
213 "this server's trust store",
214 "if \"expired\": the user must renew their proxy "
215 "(voms-proxy-init / grid-proxy-init). If \"unable to get local "
216 "issuer certificate\": add the issuing CA to the trust store and "
217 "reload",
218 verr_str);
219 X509_STORE_CTX_free(vctx);
220 return NGX_ERROR;
221 }
222
223 if (brix_gsi_enforce_signing_policy(vctx, log) != NGX_OK
224 || brix_gsi_enforce_proxy_monotonicity(vctx, log) != NGX_OK
225 || brix_gsi_enforce_cert_policy(vctx, leaf, log, client_purpose)
226 != NGX_OK)
227 {
228 X509_STORE_CTX_free(vctx);
229 return NGX_ERROR;
230 }
231
232 X509_STORE_CTX_free(vctx);
233
234 dn_str = X509_NAME_oneline(X509_get_subject_name(leaf), NULL, 0);
235 if (dn_str != NULL) {
236 ngx_cpystrn((u_char *) res->dn_buf, (u_char *) dn_str,
237 sizeof(res->dn_buf));
238 OPENSSL_free(dn_str);
239 }
240
241 return NGX_OK;
242}
