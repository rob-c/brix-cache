1/*
2 * xrdrc.c — ~/.xrdrc endpoint aliases (the start of the "just works" UX engine).
3 *
4 * WHAT: brix_alias_resolve() turns "name:suffix" into the full URL configured for
5 * that alias, so users can write `xrdcp lab:/data/f.root .` instead of the
6 * long root://host:port//data/... form.
7 * WHY: A swiss-army-knife toolkit should let staff/students name their endpoints
8 * once and never retype host/port/scheme. This is additive: anything that is
9 * not a known alias passes through verbatim, so existing usage is unchanged.
10 * HOW: An optional ini-style file at $XRDRC (else ~/.xrdrc), parsed once and cached:
11 * [alias NAME]
12 * url = root://se.lab.example:1094//data
13 * Resolution: split the arg at the first ':'; if the prefix names an alias and
14 * the suffix is not "//..." (which would be a scheme), the result is the alias
15 * URL joined to the suffix with exactly one '/'.
16 *
17 * Clean-room; no external config library.
18 */

63/*
64 * Known-key descriptor tables (the "table-driven dispatch" for rc settings).
65 *
66 * WHAT: two static const tables mapping a config key to where its value is
67 * stored: alias string fields (offset+size within xrdrc_alias) and the
68 * four [defaults] positive-integer timeouts (address of the static int).
69 * WHY: the parser used to be an if/else ladder per key; a descriptor table
70 * makes each key one data row and keeps rc_apply_setting flat — the same
71 * key names, the same destinations, the same precedence, no new keys.
72 * HOW: rc_apply_setting() scans the table matching on the section, so behavior
73 * is byte-identical to the former ladder (order of rows is immaterial —
74 * keys are unique, last-writer-wins was never possible within one line).
75 */

77 const char *key;
78 size_t off; /* offset of the char[] field inside xrdrc_alias */

105/*
106 * rc_next_line — trim one raw fgets() line to its significant content.
107 *
108 * WHAT: strips leading blanks and trailing whitespace/newline in place and
109 * returns a pointer to the first significant char, or NULL when the line
110 * should be skipped (empty or a '#' comment).
111 * WHY: every parse path first needs the trimmed, comment-filtered text; pulling
112 * it out keeps xrdrc_load() a flat section/line loop.
113 * HOW: advance past spaces/tabs, rstrip(), then reject '\0'/'#' — identical to
114 * the former inline preamble, so ignored lines stay ignored.
115 */
116static char *
117rc_next_line(char *line)
118{
119 char *s = line;
120 while (*s == ' ' || *s == '\t') { s++; }
121 rstrip(s);
122 if (*s == '\0' || *s == '#') {
123 return NULL;
124 }
125 return s;
126}

128/*
129 * rc_open_section — react to a "[...]" header line, updating the parser cursor.
130 *
131 * WHAT: given a header line, sets *in_defaults and *cur to reflect the section
132 * just entered; registers a new alias slot on "[alias NAME]".
133 * WHY: section switching was inlined in the loop; isolating it keeps the loop
134 * linear and the [alias …] registration (name trim + bounds + slot reset)
135 * in one testable place.
136 * HOW: matches "[defaults]" and the "[alias " prefix exactly as before; unknown
137 * or overflow sections clear the cursor (cur=-1) so their keys are dropped.
138 */
139static void
140rc_open_section(char *s, int *cur, int *in_defaults)
141{
142 *cur = -1;
143 *in_defaults = 0;
144 if (strcmp(s, "[defaults]") == 0) {
145 *in_defaults = 1;
146 return;
147 }
148 if (strncmp(s, "[alias ", 7) != 0) {
149 return;
150 }
151 char *nm = s + 7;
152 char *rb = strchr(nm, ']');
153 if (rb != NULL) { *rb = '\0'; }
154 while (*nm == ' ') { nm++; }
155 if (*nm != '\0' && g_nalias < XRDRC_MAX_ALIASES) {
156 *cur = g_nalias++;
157 snprintf(g_aliases[*cur].name, sizeof(g_aliases[*cur].name), "%s", nm);
158 g_aliases[*cur].url[0] = '\0';
159 }
160}

162/*
163 * rc_parse_kv — split a "key = value" line into trimmed key/value in place.
164 *
165 * WHAT: finds the first '=', null-terminates the key, right-trims the key and
166 * left-trims the value, writing both back through *out_k / *out_v; returns
167 * 0 on success or -1 when the line has no '=' (nothing to apply).
168 * WHY: key/value tokenisation is shared by the [defaults] and [alias] paths;
169 * one pure splitter removes the duplicated pointer arithmetic.
170 * HOW: mutates the caller's buffer exactly as the former inline code did — same
171 * trimming rules, so keys and values are byte-identical.
172 */
173static int
174rc_parse_kv(char *s, char **out_k, char **out_v)
175{
176 char *eq = strchr(s, '=');
177 if (eq == NULL) {
178 return -1;
179 }
180 char *k = s, *v = eq + 1, *ke = eq;
181 *eq = '\0';
182 while (ke > k && (ke[-1] == ' ' || ke[-1] == '\t')) { *--ke = '\0'; }
183 while (*v == ' ' || *v == '\t') { v++; }
184 *out_k = k;
185 *out_v = v;
186 return 0;
187}

189/*
190 * rc_apply_default — store one [defaults] timeout if it names a known key.
191 *
192 * WHAT: looks k up in rc_default_fields and, when the value is a valid positive
193 * integer, writes it to the matching static int; ignores everything else.
194 * WHY: keeps the "positive-int only" rule in exactly one spot so garbage can
195 * never become a giant unsigned timeout downstream.
196 * HOW: strtol() with full-consume + n>0 validation identical to the old ladder;
197 * unknown keys and invalid values leave the target untouched.
198 */
199static void
200rc_apply_default(const char *k, const char *v)
201{
202 char *end;
203 long n = strtol(v, &end, 10);
204 int valid = (*v != '\0' && *end == '\0' && n > 0);
205 size_t i;
206
207 if (!valid) {
208 return;
209 }
210 for (i = 0; i < sizeof(rc_default_fields) / sizeof(rc_default_fields[0]); i++) {
211 if (strcmp(k, rc_default_fields[i].key) == 0) {
212 *rc_default_fields[i].dst = (int) n;
213 return;
214 }
215 }
216}

218/*
219 * rc_apply_alias — store one [alias] string field if it names a known key.
220 *
221 * WHAT: looks k up in rc_alias_fields and snprintf's v into the matching char[]
222 * field of g_aliases[cur]; unknown keys are ignored.
223 * WHY: replaces the per-key strcmp/snprintf ladder with a single table walk so
224 * adding a field is a data row, not a new branch.
225 * HOW: offset+size from the descriptor drive a bounded snprintf into the same
226 * destination and with the same truncation bound as before.
227 */
228static void
229rc_apply_alias(int cur, const char *k, const char *v)
230{
231 size_t i;
232 for (i = 0; i < sizeof(rc_alias_fields) / sizeof(rc_alias_fields[0]); i++) {
233 if (strcmp(k, rc_alias_fields[i].key) == 0) {
234 char *dst = (char *) &g_aliases[cur] + rc_alias_fields[i].off;
235 snprintf(dst, rc_alias_fields[i].size, "%s", v);
236 return;
237 }
238 }
239}

241/*
242 * rc_apply_setting — route one parsed key/value to the active section's target.
243 *
244 * WHAT: dispatches to rc_apply_default (in [defaults]) or rc_apply_alias (in a
245 * registered [alias]); no-op when outside any section.
246 * WHY: a single seam between "which section" and "how to store" keeps
247 * precedence — [defaults] never touches aliases and vice versa — explicit.
248 * HOW: pure branch on the two cursor flags; preserves the original mutual
249 * exclusivity (in_defaults handled before cur>=0).
250 */
251static void
252rc_apply_setting(int in_defaults, int cur, const char *k, const char *v)
253{
254 if (in_defaults) {
255 rc_apply_default(k, v);
256 } else if (cur >= 0) {
257 rc_apply_alias(cur, k, v);
258 }
259}

261/* Load $XRDRC (else ~/.xrdrc) once into g_aliases. Missing file → no aliases. */
262static void
263xrdrc_load(void)
264{
265 char path[XRDC_PATH_MAX];
266 const char *p = getenv("XRDRC");
267 FILE *f;
268 /* Big enough to hold an inline `token = <jwt>` line whole — WLCG SciTokens
269 * routinely exceed 2 KB, and the token field is 8 KB; a smaller line buffer
270 * would silently truncate the credential. */
271 char line[8192 + 256];
272 int cur = -1; /* current [alias N] index, or -1 */
273 int in_defaults = 0; /* 1 while parsing the [defaults] section */
274
275 g_loaded = 1;
276 if (p != NULL && p[0] != '\0') {
277 snprintf(path, sizeof(path), "%s", p);
278 } else {
279 const char *home = getenv("HOME");
280 if (home == NULL || home[0] == '\0') {
281 return;
282 }
283 snprintf(path, sizeof(path), "%s/.xrdrc", home);
284 }
285 f = fopen(path, "r");
286 if (f == NULL) {
287 return;
288 }
289 while (fgets(line, sizeof(line), f) != NULL) {
290 char *k = NULL;
291 char *v = NULL;
292 char *s = rc_next_line(line);
293 if (s == NULL) {
294 continue;
295 }
296 if (*s == '[') {
297 rc_open_section(s, &cur, &in_defaults);
298 continue;
299 }
300 if (rc_parse_kv(s, &k, &v) != 0) {
301 continue;
302 }
303 rc_apply_setting(in_defaults, cur, k, v);
304 }
305 fclose(f);
306}

308int
309brix_alias_resolve(const char *arg, char *out, size_t outsz)
310{
311 const char *colon;
312 size_t nlen;
313 char name[XRDRC_NAME_MAX];
314 int i;
315
316 if (!g_loaded) {
317 xrdrc_load();
318 }
319 snprintf(out, outsz, "%s", arg); /* default: pass through verbatim */
320
321 colon = strchr(arg, ':');
322 if (colon == NULL) {
323 return 0;
324 }
325 nlen = (size_t) (colon - arg);
326 if (nlen == 0 || nlen >= sizeof(name)) {
327 return 0;
328 }
329 if (colon[1] == '/' && colon[2] == '/') {
330 return 0; /* "scheme://…" — a URL, not an alias */
331 }
332 memcpy(name, arg, nlen);
333 name[nlen] = '\0';
334
335 for (i = 0; i < g_nalias; i++) {
336 if (strcmp(g_aliases[i].name, name) == 0 && g_aliases[i].url[0] != '\0') {
337 const char *base = g_aliases[i].url;
338 const char *suffix = colon + 1;
339 size_t bl = strlen(base);
340 int base_slash = (bl > 0 && base[bl - 1] == '/');
341 int suf_slash = (suffix[0] == '/');
342 if (base_slash && suf_slash) {
343 snprintf(out, outsz, "%s%s", base, suffix + 1); /* drop one '/' */
344 } else if (!base_slash && !suf_slash) {
345 snprintf(out, outsz, "%s/%s", base, suffix); /* add the '/' */
346 } else {
347 snprintf(out, outsz, "%s%s", base, suffix); /* exactly one */
348 }
349 return 1;
350 }
351 }
352 return 0;
353}

356static int
357read_first_line(const char *path, char *out, size_t outsz)
358{
359 FILE *f = fopen(path, "r");
360 char line[8192];

362 if (f == NULL) {
363 return -1;
364 }

366 char *s = line;

368 rstrip(s);
369 if (*s != '\0') {

373 }
374 }
375 fclose(f);

381{

386 memset(info, 0, sizeof(*info));

414 * brix_xrdrc_default_ms — look up a [defaults] timeout key from ~/.xrdrc.
415 *
416 * WHAT: Returns 1 and sets *out_ms when the [defaults] section of the user's
417 * .xrdrc file carries `key` with a valid positive integer; returns 0
418 * otherwise (key absent, value non-numeric, or value <= 0).
419 * WHY: Allows users and operators to tune network timeouts once in ~/.xrdrc
420 * rather than through environment variables in every shell profile.
421 * Sits below the env-var layer so $XRDC_* and CLI flags always win.
422 * HOW: Ensures the file is loaded via the same lazy-load gate the alias
423 * lookups use, then maps `key` to one of the four parsed static ints.

431 xrdrc_load();

434 *out_ms = g_def_connect_ms;
