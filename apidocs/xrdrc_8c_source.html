1/*
2 * xrdrc.c — ~/.xrdrc endpoint aliases (the start of the "just works" UX engine).
3 *
4 * WHAT: brix_alias_resolve() turns "name:suffix" into the full URL configured for
5 * that alias, so users can write `xrdcp lab:/data/f.root .` instead of the
6 * long root://host:port//data/... form.
7 * WHY: A swiss-army-knife toolkit should let staff/students name their endpoints
8 * once and never retype host/port/scheme. This is additive: anything that is
9 * not a known alias passes through verbatim, so existing usage is unchanged.
10 * HOW: An optional ini-style file at $XRDRC (else ~/.xrdrc), parsed once and cached:
11 * [alias NAME]
12 * url = root://se.lab.example:1094//data
13 * Resolution: split the arg at the first ':'; if the prefix names an alias and
14 * the suffix is not "//..." (which would be a scheme), the result is the alias
15 * URL joined to the suffix with exactly one '/'.
16 *
17 * Clean-room; no external config library.
18 */
19#include "brix.h"

52/* Trim trailing whitespace/newline in place. */
53static void
54rstrip(char *s)
55{
56 char *e = s + strlen(s);
57 while (e > s && (e[-1] == '\n' || e[-1] == '\r' || e[-1] == ' ' || e[-1] == '\t')) {
58 *--e = '\0';
59 }
60}

62/* Load $XRDRC (else ~/.xrdrc) once into g_aliases. Missing file → no aliases. */
63static void
64xrdrc_load(void)
65{
66 char path[XRDC_PATH_MAX];
67 const char *p = getenv("XRDRC");
68 FILE *f;
69 /* Big enough to hold an inline `token = <jwt>` line whole — WLCG SciTokens
70 * routinely exceed 2 KB, and the token field is 8 KB; a smaller line buffer
71 * would silently truncate the credential. */
72 char line[8192 + 256];
73 int cur = -1; /* current [alias N] index, or -1 */
74 int in_defaults = 0; /* 1 while parsing the [defaults] section */
75
76 g_loaded = 1;
77 if (p != NULL && p[0] != '\0') {
78 snprintf(path, sizeof(path), "%s", p);
79 } else {
80 const char *home = getenv("HOME");
81 if (home == NULL || home[0] == '\0') {
82 return;
83 }
84 snprintf(path, sizeof(path), "%s/.xrdrc", home);
85 }
86 f = fopen(path, "r");
87 if (f == NULL) {
88 return;
89 }
90 while (fgets(line, sizeof(line), f) != NULL) {
91 char *s = line;
92 while (*s == ' ' || *s == '\t') { s++; }
93 rstrip(s);
94 if (*s == '\0' || *s == '#') {
95 continue;
96 }
97 if (*s == '[') {
98 cur = -1;
99 in_defaults = 0;
100 if (strcmp(s, "[defaults]") == 0) {
101 in_defaults = 1;
102 } else if (strncmp(s, "[alias ", 7) == 0) {
103 char *nm = s + 7;
104 char *rb = strchr(nm, ']');
105 if (rb != NULL) { *rb = '\0'; }
106 while (*nm == ' ') { nm++; }
107 if (*nm != '\0' && g_nalias < XRDRC_MAX_ALIASES) {
108 cur = g_nalias++;
109 snprintf(g_aliases[cur].name, sizeof(g_aliases[cur].name), "%s", nm);
110 g_aliases[cur].url[0] = '\0';
111 }
112 }
113 continue;
114 }
115 /* key = value lines */
116 char *eq = strchr(s, '=');
117 if (eq == NULL) {
118 continue;
119 }
120 char *k = s, *v = eq + 1, *ke = eq;
121 *eq = '\0';
122 while (ke > k && (ke[-1] == ' ' || ke[-1] == '\t')) { *--ke = '\0'; }
123 while (*v == ' ' || *v == '\t') { v++; }
124
125 if (in_defaults) {
126 /* Parse positive integers only — negative/non-numeric are silently
127 * ignored so they cannot become giant unsigned timeouts. */
128 char *end;
129 long n = strtol(v, &end, 10);
130 int valid = (*v != '\0' && *end == '\0' && n > 0);
131 if (strcmp(k, "connect_timeout_ms") == 0) {
132 if (valid) { g_def_connect_ms = (int) n; }
133 } else if (strcmp(k, "io_timeout_ms") == 0) {
134 if (valid) { g_def_io_ms = (int) n; }
135 } else if (strcmp(k, "max_stall_ms") == 0) {
136 if (valid) { g_def_stall_ms = (int) n; }
137 } else if (strcmp(k, "backoff_base_ms") == 0) {
138 if (valid) { g_def_backoff_ms = (int) n; }
139 }
140 } else if (cur >= 0) {
141 if (strcmp(k, "url") == 0) {
142 snprintf(g_aliases[cur].url, sizeof(g_aliases[cur].url), "%s", v);
143 } else if (strcmp(k, "token") == 0) {
144 snprintf(g_aliases[cur].token, sizeof(g_aliases[cur].token), "%s", v);
145 } else if (strcmp(k, "token_file") == 0 || strcmp(k, "bearer_token_file") == 0) {
146 snprintf(g_aliases[cur].token_file, sizeof(g_aliases[cur].token_file), "%s", v);
147 } else if (strcmp(k, "s3_access") == 0) {
148 snprintf(g_aliases[cur].s3_access, sizeof(g_aliases[cur].s3_access), "%s", v);
149 } else if (strcmp(k, "s3_secret") == 0) {
150 snprintf(g_aliases[cur].s3_secret, sizeof(g_aliases[cur].s3_secret), "%s", v);
151 } else if (strcmp(k, "s3_region") == 0) {
152 snprintf(g_aliases[cur].s3_region, sizeof(g_aliases[cur].s3_region), "%s", v);
153 } else if (strcmp(k, "proxy") == 0) {
154 snprintf(g_aliases[cur].proxy, sizeof(g_aliases[cur].proxy), "%s", v);
155 }
156 }
157 }
158 fclose(f);
159}

161int
162brix_alias_resolve(const char *arg, char *out, size_t outsz)
163{
164 const char *colon;
165 size_t nlen;
166 char name[XRDRC_NAME_MAX];
167 int i;
168
169 if (!g_loaded) {
170 xrdrc_load();
171 }
172 snprintf(out, outsz, "%s", arg); /* default: pass through verbatim */
173
174 colon = strchr(arg, ':');
175 if (colon == NULL) {
176 return 0;
177 }
178 nlen = (size_t) (colon - arg);
179 if (nlen == 0 || nlen >= sizeof(name)) {
180 return 0;
181 }
182 if (colon[1] == '/' && colon[2] == '/') {
183 return 0; /* "scheme://…" — a URL, not an alias */
184 }
185 memcpy(name, arg, nlen);
186 name[nlen] = '\0';
187
188 for (i = 0; i < g_nalias; i++) {
189 if (strcmp(g_aliases[i].name, name) == 0 && g_aliases[i].url[0] != '\0') {
190 const char *base = g_aliases[i].url;
191 const char *suffix = colon + 1;
192 size_t bl = strlen(base);
193 int base_slash = (bl > 0 && base[bl - 1] == '/');
194 int suf_slash = (suffix[0] == '/');
195 if (base_slash && suf_slash) {
196 snprintf(out, outsz, "%s%s", base, suffix + 1); /* drop one '/' */
197 } else if (!base_slash && !suf_slash) {
198 snprintf(out, outsz, "%s/%s", base, suffix); /* add the '/' */
199 } else {
200 snprintf(out, outsz, "%s%s", base, suffix); /* exactly one */
201 }
202 return 1;
203 }
204 }
205 return 0;
206}

208/* Read the first non-empty line of `path` into out[outsz] (trimmed). 0 / -1. */
209static int
210read_first_line(const char *path, char *out, size_t outsz)
211{
212 FILE *f = fopen(path, "r");
213 char line[8192];
214 out[0] = '\0';
215 if (f == NULL) {
216 return -1;
217 }
218 while (fgets(line, sizeof(line), f) != NULL) {
219 char *s = line;
220 while (*s == ' ' || *s == '\t') { s++; }
221 rstrip(s);
222 if (*s != '\0') {
223 snprintf(out, outsz, "%s", s);
224 fclose(f);
225 return 0;
226 }
227 }
228 fclose(f);
229 return -1;
230}

232int
233brix_alias_lookup(const char *name, brix_alias_info *info)
234{
235 int i;
236 if (info == NULL) {
237 return 0;
238 }
239 memset(info, 0, sizeof(*info));
240 if (!g_loaded) {
241 xrdrc_load();
242 }
243 for (i = 0; i < g_nalias; i++) {
244 if (strcmp(g_aliases[i].name, name) == 0) {
245 const xrdrc_alias *a = &g_aliases[i];
246 info->found = 1;
247 snprintf(info->s3_access, sizeof(info->s3_access), "%s", a->s3_access);
248 snprintf(info->s3_secret, sizeof(info->s3_secret), "%s", a->s3_secret);
249 snprintf(info->s3_region, sizeof(info->s3_region), "%s", a->s3_region);
250 snprintf(info->proxy, sizeof(info->proxy), "%s", a->proxy);
251 if (a->token[0] != '\0') {
252 snprintf(info->bearer, sizeof(info->bearer), "%s", a->token);
253 } else if (a->token_file[0] != '\0') {
254 snprintf(info->token_file, sizeof(info->token_file), "%s", a->token_file);
255 if (read_first_line(a->token_file, info->bearer, sizeof(info->bearer)) != 0
256 || info->bearer[0] == '\0') {
257 info->token_file_failed = 1; /* unreadable or empty — caller warns */
258 }
259 }
260 return 1;
261 }
262 }
263 return 0;
264}

266/*
267 * brix_xrdrc_default_ms — look up a [defaults] timeout key from ~/.xrdrc.
268 *
269 * WHAT: Returns 1 and sets *out_ms when the [defaults] section of the user's
270 * .xrdrc file carries `key` with a valid positive integer; returns 0
271 * otherwise (key absent, value non-numeric, or value <= 0).
272 * WHY: Allows users and operators to tune network timeouts once in ~/.xrdrc
273 * rather than through environment variables in every shell profile.
274 * Sits below the env-var layer so $XRDC_* and CLI flags always win.
275 * HOW: Ensures the file is loaded via the same lazy-load gate the alias
276 * lookups use, then maps `key` to one of the four parsed static ints.
277 * Negative and non-numeric values are intentionally rejected at parse
278 * time so they are never stored and can never be returned here.
279 */
280int
281brix_xrdrc_default_ms(const char *key, int *out_ms)
282{
283 if (!g_loaded) {
284 xrdrc_load();
285 }
286 if (strcmp(key, "connect_timeout_ms") == 0 && g_def_connect_ms > 0) {
287 *out_ms = g_def_connect_ms;
288 return 1;
289 }
290 if (strcmp(key, "io_timeout_ms") == 0 && g_def_io_ms > 0) {
291 *out_ms = g_def_io_ms;
292 return 1;
293 }
294 if (strcmp(key, "max_stall_ms") == 0 && g_def_stall_ms > 0) {
295 *out_ms = g_def_stall_ms;
296 return 1;
297 }
298 if (strcmp(key, "backoff_base_ms") == 0 && g_def_backoff_ms > 0) {
299 *out_ms = g_def_backoff_ms;
300 return 1;
301 }
302 return 0;
