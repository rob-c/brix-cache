1/*
2 * sd_registry.c — Storage Driver registry + capability-gated accessors.
3 *
4 * WHAT: Owns the static table of registered drivers (POSIX today; block/object
5 * later), the name->driver lookup, per-export instance construction and
6 * teardown, and the small accessor helpers (caps/fd/supports/name) that
7 * keep callers off the raw vtable.
8 *
9 * WHY: The VFS must select a backend by config name and obtain a bound instance
10 * without knowing any driver's internals. Centralizing registration here
11 * means adding a backend is a one-line table edit plus its sd_<name>.c.
12 *
13 * HOW: brix_sd_instance_create() looks up the driver, pcalloc's an instance,
14 * and runs the driver's init(). The accessors read the driver caps/fd
15 * behind NULL-tolerant guards. Phase 55.A registers only "posix".
16 */

65/* brix_sd_driver_find — resolve a backend name to its driver (linear scan of the
66 * small static table; NULL name → NULL). Used by config parsing and instance
67 * creation to map a string to a driver. */
68const brix_sd_driver_t *
69brix_sd_driver_find(const char *name)
70{
71 ngx_uint_t i;
72
73 if (name == NULL) {
74 return NULL;
75 }
76 for (i = 0; i < sizeof(sd_drivers) / sizeof(sd_drivers[0]); i++) {
77 if (ngx_strcmp(sd_drivers[i]->name, name) == 0) {
78 return sd_drivers[i];
79 }
80 }
81 return NULL;
82}

84/*
85 * brix_sd_instance_pool — private process-lifetime pool for driver instances.
86 *
87 * WHY: instances are process-lifetime singletons, but composition can run
88 * during configuration parse, when ngx_cycle still points at the TRANSIENT
89 * init cycle whose pool is destroyed as startup completes. Instances (and
90 * driver state pcalloc'd from inst->pool by driver->init) allocated there
91 * dangle, and the first request through a tier dereferences freed memory
92 * (SIGSEGV in cstore_make_parents via a garbage store->driver). A pool the
93 * registry owns and never destroys removes the dependence on WHEN composition
94 * runs. Cost: instances are never freed on reload — which was already the
95 * case, since draining workers may still be using them.
96 */
97static ngx_pool_t *
98brix_sd_instance_pool(ngx_log_t *log)
99{
100 static ngx_pool_t *pool; /* per-process; copied to workers by fork */
101
102 if (pool == NULL) {
103 pool = ngx_create_pool(4096, log);
104 }
105 return pool;
106}

108/* brix_sd_instance_create — build the per-export instance bound at config/worker
109 * init: find the named driver, pcalloc the instance from the registry's
110 * process-lifetime pool, call init(driver_conf); on an unknown driver or init
111 * failure, set *err_out and return NULL. */
112brix_sd_instance_t *
113brix_sd_instance_create(ngx_log_t *log, const char *name,
114 void *driver_conf, int *err_out)
115{
116 const brix_sd_driver_t *driver;
117 brix_sd_instance_t *inst;
118 ngx_pool_t *pool;
119
120 driver = brix_sd_driver_find(name);
121 if (driver == NULL) {
122 if (err_out != NULL) { *err_out = ENOENT; }
123 return NULL;
124 }
125
126 pool = brix_sd_instance_pool(log);
127 if (pool == NULL) {
128 if (err_out != NULL) { *err_out = ENOMEM; }
129 return NULL;
130 }
131
132 inst = ngx_pcalloc(pool, sizeof(*inst));
133 if (inst == NULL) {
134 if (err_out != NULL) { *err_out = ENOMEM; }
135 return NULL;
136 }
137
138 inst->driver = driver;
139 inst->log = log;
140 inst->pool = pool;
141 inst->caps = driver->caps; /* effective caps default = descriptor caps;
142 * init may narrow/extend (Phase-83 pblock mask) */
143
144 if (driver->init != NULL && driver->init(inst, driver_conf) != NGX_OK) {
145 if (err_out != NULL) { *err_out = errno != 0 ? errno : EINVAL; }
146 return NULL;
147 }
148 return inst;
149}
