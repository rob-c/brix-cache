1/*
2 * tpc_user_proxy.c — resolve the requesting user's delegated x509 proxy for the
3 * HTTP-TPC pull leg so the SOURCE authenticates the END USER, not the
4 * destination's static service cert (phase-70 credential-forwarding closure).
5 *
6 * See tpc_user_proxy.h for the WHAT/WHY/HOW contract. This file reuses the
7 * existing Phase-70 delegation plumbing end-to-end:
8 * - the live full-proxy passthrough bag captured on the request
9 * (rctx->deleg_proxy_pem, from X-Brix-Delegate-Proxy) — same field the VFS
10 * PASSTHROUGH seam reads;
11 * - the per-user delegation store written by delegation.c
12 * (<storage_credential_dir>/<key>.pem, keyed by DN) — resolved via the same
13 * brix_sd_ucred_select() the davs/S3 origin legs use;
14 * - the same 0600 temp materialiser + unlink/zero cleanup pattern the VFS
15 * PASSTHROUGH seam uses (brix_proxy_gsi_write_pem_temp).
16 */

26/*
27 * Cleanup payload for a materialised passthrough proxy temp: the pool-allocated
28 * 0600 temp path. Mirrors vfs_deleg.c's brix_deleg_temp_t — on request-pool
29 * teardown the file is unlinked and its path string zeroed so the private key
30 * never lingers in freed-but-reused pool memory.
31 */
32typedef struct {
33 char *path; /* NUL-terminated temp path, owned by the request pool */
34} webdav_tpc_user_proxy_temp_t;
35
36/*
37 * webdav_tpc_user_proxy_temp_cleanup — request-pool cleanup handler: unlink the
38 * materialised proxy temp and zero its path string.
39 *
40 * WHAT: unlink() the temp (ignoring ENOENT) then scrub the path bytes.
41 * WHY: secret hygiene — the delegated proxy's private key lives only in the
42 * 0600 temp for the transfer's duration and must vanish the moment the
43 * request pool is torn down, success or failure.
44 * HOW: data is a webdav_tpc_user_proxy_temp_t*; a NULL/consumed entry is a
45 * no-op.
46 */
47static void
48webdav_tpc_user_proxy_temp_cleanup(void *data)
49{
50 webdav_tpc_user_proxy_temp_t *t = data;
51
52 if (t == NULL || t->path == NULL) {
53 return;
54 }
55 (void) unlink(t->path); /* vfs-seam-allow: config-domain delegated proxy credential temp (not export storage) */
56 ngx_memzero(t->path, ngx_strlen(t->path));
57 t->path = NULL;
58}

60/*
61 * webdav_tpc_user_proxy_materialise — write PEM bytes to an owner-only temp and
62 * register the unlink+zero cleanup, pointing *out at the temp path.
63 *
64 * WHAT: 0600 temp via brix_proxy_gsi_write_pem_temp(), pool-copied path,
65 * cleanup registered, out->cert_path == out->key_path == temp path.
66 * WHY: curl reads the client cert/key from a FILE path; a live passthrough
67 * proxy arrives as in-memory PEM (chain + key), so it must be staged to a
68 * 0600 file the same way the VFS PASSTHROUGH seam stages it. cert and key
69 * share one combined PEM, so both curl options point at the same file.
70 * HOW: NGX_OK with out filled on success; NGX_ERROR (out left have=0) on
71 * materialise / alloc / cleanup-registration failure — the temp is
72 * unlinked before returning so no key is left on disk.
73 */
74static ngx_int_t
75webdav_tpc_user_proxy_materialise(ngx_http_request_t *r,
76 const u_char *pem, size_t pem_len, webdav_tpc_user_proxy_t *out)
77{
78 char tmp[NGX_MAX_PATH];
79 char *path;
80 size_t path_len;
81 ngx_pool_cleanup_t *cln;
82 webdav_tpc_user_proxy_temp_t *payload;
83
84 if (brix_proxy_gsi_write_pem_temp(pem, pem_len, tmp, sizeof(tmp)) != 0) {
85 ngx_log_error(NGX_LOG_ERR, r->connection->log, ngx_errno,
86 "brix_webdav: TPC pull could not materialise delegated"
87 " proxy temp");
88 return NGX_ERROR;
89 }
90
91 path_len = ngx_strlen(tmp);
92 path = ngx_pnalloc(r->pool, path_len + 1);
93 if (path == NULL) {
94 (void) unlink(tmp); /* vfs-seam-allow: config-domain delegated proxy credential temp (not export storage) */
95 return NGX_ERROR;
96 }
97 ngx_memcpy(path, tmp, path_len);
98 path[path_len] = '\0';
99
100 cln = ngx_pool_cleanup_add(r->pool, sizeof(*payload));
101 if (cln == NULL) {
102 (void) unlink(path); /* vfs-seam-allow: config-domain delegated proxy credential temp (not export storage) */
103 ngx_memzero(path, path_len);
104 return NGX_ERROR;
105 }
106 payload = cln->data;
107 payload->path = path;
108 cln->handler = webdav_tpc_user_proxy_temp_cleanup;
109
110 out->have = 1;
111 out->cert_path = path;
112 out->key_path = path; /* stored PEM carries cert + key in one file */
113 return NGX_OK;
114}

116/*
117 * webdav_tpc_user_proxy_from_passthrough — resolution step 1: the live full
118 * proxy the front door captured on this request.
119 *
120 * WHAT: if rctx carries a non-empty deleg_proxy_pem, materialise it and mark
121 * out->have=1.
122 * WHY: a user who explicitly delegated a full proxy over the wire
123 * (X-Brix-Delegate-Proxy) must have THAT proxy presented to the source —
124 * it is the most specific, request-scoped credential and takes precedence
125 * over any pre-provisioned store entry.
126 * HOW: returns 1 when the step produced a decision (materialise succeeded, or
127 * failed and was logged — either way the passthrough was the intended
128 * credential and we do not silently fall through to the service cert),
129 * 0 when no passthrough proxy is present so the next step should run.
130 */
131static int
132webdav_tpc_user_proxy_from_passthrough(ngx_http_request_t *r,
133 const ngx_http_brix_webdav_req_ctx_t *rctx, webdav_tpc_user_proxy_t *out)
134{
135 if (rctx == NULL || rctx->deleg_proxy_pem.len == 0
136 || rctx->deleg_proxy_pem.data == NULL)
137 {
138 return 0;
139 }
140
141 if (webdav_tpc_user_proxy_materialise(r, rctx->deleg_proxy_pem.data,
142 rctx->deleg_proxy_pem.len, out) != NGX_OK)
143 {
144 /* Deny-consistent: the user delegated a proxy but we could not stage
145 * it — do NOT quietly present the service identity in its place. Flag
146 * deny so the caller aborts rather than falling back. */
147 out->have = 0;
148 out->deny = 1;
149 return 1;
150 }
151 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
152 "brix_webdav: TPC pull presenting delegated proxy"
153 " (passthrough) to source");
154 return 1;
155}

157/*
158 * webdav_tpc_user_proxy_from_store — resolution step 2: the per-user delegation
159 * store keyed by the authenticated identity.
160 *
161 * WHAT: brix_sd_ucred_select() over common.storage_credential_dir; on an x509
162 * .pem hit point cert_path/key_path at the stored file.
163 * WHY: a proxy uploaded to <cred_dir>/<key>.pem via the GridSite delegation
164 * endpoint (delegation.c) is exactly the user's credential — the same
165 * store the davs/S3 origin legs select from — and must be presented to
166 * the source when no live passthrough proxy is present.
167 * HOW: no credential dir configured, or select returned non-x509 (bearer/s3/
168 * ceph) or DECLINED (absent / expired .pem) → leave have=0 so the caller
169 * falls back to the service cert. A stored .pem holds cert + key in one
170 * file, so cert_path == key_path (the store path itself; no temp).
171 */
172static void
173webdav_tpc_user_proxy_from_store(ngx_http_request_t *r,
174 ngx_http_brix_webdav_loc_conf_t *conf, const brix_identity_t *id,
175 webdav_tpc_user_proxy_t *out)
176{
177 const ngx_str_t *dir = &conf->common.storage_credential_dir;
178 char dirz[BRIX_UCRED_PATH_MAX];
179 brix_sd_ucred_t cred;
180 char *path;
181 size_t path_len;
182
183 if (dir->len == 0 || id == NULL || dir->len >= sizeof(dirz)) {
184 return;
185 }
186 ngx_memcpy(dirz, dir->data, dir->len);
187 dirz[dir->len] = '\0';
188
189 if (brix_sd_ucred_select(dirz, id, &cred) != NGX_OK) {
190 /* Absent, or an expired .pem (cred.expired) — no usable per-user proxy;
191 * fall back to the service cert. */
192 return;
193 }
194 if (cred.is_bearer || cred.is_s3 || cred.is_ceph) {
195 /* A non-x509 stored credential is not a curl client cert for the pull
196 * leg (the bearer path is applied as an Authorization header, s3/ceph
197 * are backend-only). Fall back to the service cert. */
198 return;
199 }
200
201 path_len = ngx_strlen(cred.path);
202 path = ngx_pnalloc(r->pool, path_len + 1);
203 if (path == NULL) {
204 return;
205 }
206 ngx_memcpy(path, cred.path, path_len);
207 path[path_len] = '\0';
208
209 out->have = 1;
210 out->cert_path = path;
211 out->key_path = path; /* stored .pem carries cert + key in one file */
212 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
213 "brix_webdav: TPC pull presenting delegated proxy"
214 " (store) to source");
215}

217void
218webdav_tpc_user_proxy_resolve(ngx_http_request_t *r,
219 ngx_http_brix_webdav_loc_conf_t *conf, webdav_tpc_user_proxy_t *out)
220{
221 ngx_http_brix_webdav_req_ctx_t *rctx =
222 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
223
224 out->have = 0;
225 out->deny = 0;
226 out->cert_path = NULL;
227 out->key_path = NULL;
228
229 /* Step 1: the live full-proxy passthrough the front door captured. */
230 if (webdav_tpc_user_proxy_from_passthrough(r, rctx, out)) {
231 return;
232 }
233
234 /* Step 2: the per-user delegation store keyed by the authenticated DN. */
235 webdav_tpc_user_proxy_from_store(r, conf,
236 (rctx != NULL) ? rctx->identity : NULL, out);
237
238 /* Step 3 (implicit): out->have stays 0 → caller falls back to
239 * conf->tpc_cert, preserving current behaviour for non-delegated setups. */
240}
