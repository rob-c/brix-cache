17/* HTTP methods. Unless noted, each fully handles its method and returns NGX_OK
18 * or an NGX_HTTP_* status for the dispatcher to finalise; the `void` ones own
19 * the response and finalise the request themselves (async/body paths). */
20/* OPTIONS: emit DAV/DASL/Allow headers (Allow derived from the live op table). */
21ngx_int_t webdav_handle_options(ngx_http_request_t *r);

25/* GET a file body (range/multipart aware, cache-backed); 403 on a directory. */
26ngx_int_t webdav_handle_get(ngx_http_request_t *r);
27/* PUT: body read callback — streams the request body to the destination file
28 * and finalises the request (it self-finalises; do not also return its rc). */

30/* DELETE: lock-checked recursive removal (204, or 404/409/500). */

36/* MKCOL: create one collection (201; 409 if parent missing, 405 if exists). */

42/* SEARCH (RFC 5323 DAV:basicsearch) over the request-URI scope. */
43ngx_int_t webdav_handle_search(ngx_http_request_t *r);
44/* ACL: read-only export — always 403 with a cannot-modify-protected-property

48/* MOVE: lock-checked rename within the export (201/204; may finalise -> NGX_DONE). */

67/* §3 XrdDig: GET/HEAD /.well-known/dig/<export>/<rel> — read-only, RESOLVE_BENEATH-
