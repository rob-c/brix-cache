1/*
2 * delegation_gridsite_req.c - Phase-3 T4 step 1: the GridSite getProxyReq
3 * half of the two-step delegation handshake (GET
4 * /.well-known/brix-delegation/request). Split out of delegation.c; the
5 * putProxy half lives in delegation_gridsite_put.c, the pending-delegation
6 * store in delegation_store.c, and the shared cert-chain / storage helpers
7 * in delegation.c (declared in delegation_internal.h).
8 *
9 * ======================= Phase-3 T4: two-step GridSite ====================
10 * WHAT: The standard GridSite getProxyReq/putProxy handshake, adapted to
11 * plain HTTP: GET /.well-known/brix-delegation/request returns a
12 * fresh CSR + a delegation-id; the client signs the CSR with its own
13 * EEC key and PUTs the signed proxy back to
14 * /.well-known/brix-delegation/<id>, which assembles and stores the
15 * full delegated proxy exactly like the T8 upload path.
16 *
17 * WHY: T8's proxy-UPLOAD form requires the client to already hold a
18 * complete, signed proxy chain it is willing to hand over wholesale —
19 * fine for scripted/xrdcp-style clients but not how the reference
20 * GridSite delegation portType works, and not what some interop
21 * partners expect. The two-step form lets the SERVER generate the
22 * keypair (never touching disk, never leaving the server) so the
23 * client only ever signs a request — the standard, narrower trust
24 * model this specification exists to support alongside T8, not to
25 * replace it.
26 *
27 * HOW: 1. brix_deleg_store — a bounded (256-entry), per-worker, in-memory
28 * table of pending {id, fresh EVP_PKEY*, client_dn, expires_at}
29 * entries, TTL 600s, swept lazily on every get/put (no background
30 * thread — see brix_deleg_store_sweep). Per-worker, NOT SHM: the
31 * fresh private key must never leave the process that generated
32 * it, and this is a short interactive two-request handshake, not
33 * durable state — a multi-worker deployment needs sticky routing
34 * (an L7 proxy or client retry-on-same-worker) so both requests of
35 * one handshake land on the same worker. Documented on
36 * brix_deleg_store_t below; NOT a design flaw for THIS store, since
37 * SHM would additionally require serialising an EVP_PKEY private
38 * key into shared memory, which trades one small operational
39 * constraint (stickiness) for a much larger one (private key
40 * material outside process-local heap).
41 * 2. webdav_delegation_request_handle (GET .../request): reuse the
42 * SAME peer-cert access pattern as auth_cert.c's
43 * webdav_verify_proxy_cert() — SSL_get_peer_certificate(ssl) off
44 * r->connection->ssl->connection — PEM-export it as `parent_pem`,
45 * and call brix_gsi_build_pxyreq(parent_pem, ...) to get a fresh
46 * key + CSR (re-encoded request DER→PEM, mirroring
47 * src/auth/gsi/delegation.c's gsi_req_der_to_pem, since the CSR
48 * travels over HTTP as PEM here too — DER has no natural HTTP
49 * content-type and PEM is what a human/openssl workflow expects).
50 * Store the entry keyed by a CSPRNG id (RAND_bytes, 16 bytes / 32
51 * hex chars via brix_hex_encode); respond with the CSR PEM as the
52 * body and the id in X-Brix-Delegation-Id. UNLIKE the other two
53 * entry points, this one runs SYNCHRONOUSLY inside the content
54 * phase (a GET has no body to read asynchronously) and therefore
55 * returns an ngx_int_t instead of self-finalizing — see
56 * delegation.h's doc comment on this function for why
57 * self-finalizing here would double-finalize the request and
58 * crash the worker (caught by testing: the two-step e2e test
59 * initially crashed a worker with SIGSEGV in
60 * ngx_http_set_lingering_close via a double
61 * ngx_http_finalize_request, fixed by returning the status through
62 * webdav_metrics_return() like every other synchronous handler
63 * instead of calling webdav_metrics_finalize_request).
64 * 3. webdav_delegation_put_handle (PUT .../<id>): look up id (checked
65 * BEFORE sweeping the rest of the table, so an expired-but-present
66 * entry is reported as 410, distinct from a never-existed/already-
67 * consumed 404 — see brix_deleg_store_take); 404 if absent, 410 if
68 * expired, 403 (and DO NOT touch any file) if entry.client_dn !=
69 * ctx->dn. Otherwise split the
70 * uploaded body into its first PEM block (the signed proxy —
71 * brix_gsi_assemble_proxy's `proxy_pem` parses only the first
72 * block via PEM_read_bio_X509) and the remainder (the client's EEC
73 * chain — passed verbatim as `chain_pem`, appended after the proxy
74 * with no further parsing, exactly as brix_gsi_assemble_proxy's
75 * implementation does). Call brix_gsi_assemble_proxy(proxy_pem,
76 * reqkey, chain_pem, ...) — it verifies the signed proxy's pubkey
77 * matches the stored fresh key (this IS the delegation's proof of
78 * possession: only a client holding the matching CSR could produce
79 * a proxy whose key matches) and returns the assembled
80 * <proxy><chain><private key> PEM — the reqkey is serialized into
81 * the blob so the stored credential is complete and can
82 * authenticate downstream (proxy_ssl_certificate_key, TPC pull,
83 * origin auth all load key+chain from the one stored file).
84 * Validate every cert's notAfter is still in
85 * the future and the chain's end-entity DN matches ctx->dn (reuse
86 * delegation_chain_expired / delegation_eec_dn_matches, the SAME T8
87 * helpers — a two-step-assembled credential is validated exactly
88 * as strictly as an uploaded one). On success, atomically store
89 * via delegation_store_pem (the SAME T8 helper) and drop the store
90 * entry. One-shot policy (documented on brix_deleg_store_drop
91 * below): the entry is dropped and its EVP_PKEY freed on ANY
92 * terminal outcome of this call — success, DN mismatch, expired,
93 * malformed body, or assemble failure — never left pending for a
94 * retry. This is the simplest policy that guarantees no EVP_PKEY
95 * leak on any path and closes the id after first use either way,
96 * matching the brief's documented default.
97 */

112/*
113 * delegation_get_peer_pem - PEM-export the TLS peer certificate off this
114 * request's SSL connection. Reuses the exact acquisition pattern
115 * auth_cert.c's webdav_verify_proxy_cert() uses (SSL_get_peer_certificate
116 * off r->connection->ssl->connection) — this endpoint runs strictly after
117 * that same verification has already populated ctx->verified/ctx->dn, so a
118 * peer cert is guaranteed present; this just re-fetches the X509* (the
119 * verified req_ctx stores only the derived DN string, not the cert object)
120 * and serialises it to PEM in memory for brix_gsi_build_pxyreq's
121 * `parent_pem` argument. Caller frees *pem via free() (malloc'd, matching
122 * the ngx-free proxy_req.c/gsi delegation.c convention for PEM buffers).
123 */
124static ngx_int_t
125delegation_get_peer_pem(ngx_http_request_t *r, u_char **pem, size_t *pem_len)
126{
127 SSL *ssl;
128 X509 *leaf;
129 BIO *bio;
130 char *data;
131 long n;
132
133 if (r->connection->ssl == NULL) {
134 return NGX_ERROR;
135 }
136 ssl = r->connection->ssl->connection;
137 leaf = SSL_get_peer_certificate(ssl);
138 if (leaf == NULL) {
139 return NGX_ERROR;
140 }
141
142 bio = BIO_new(BIO_s_mem());
143 if (bio == NULL || PEM_write_bio_X509(bio, leaf) != 1) {
144 BIO_free(bio);
145 X509_free(leaf);
146 return NGX_ERROR;
147 }
148 X509_free(leaf);
149
150 n = BIO_get_mem_data(bio, &data);
151 *pem = malloc((size_t) n + 1);
152 if (*pem == NULL) {
153 BIO_free(bio);
154 return NGX_ERROR;
155 }
156 memcpy(*pem, data, (size_t) n);
157 (*pem)[n] = '\0';
158 *pem_len = (size_t) n;
159 BIO_free(bio);
160 return NGX_OK;
161}

163/*
164 * delegation_req_der_to_pem - re-encode brix_gsi_build_pxyreq's DER
165 * X509_REQ output as PEM. The two-step flow travels the CSR over HTTP as
166 * a body a human or `openssl req` workflow can consume directly, and
167 * mirrors src/auth/gsi/delegation.c's gsi_req_der_to_pem (the wire-protocol
168 * delegation path re-encodes for the exact same reason — the crypto core
169 * emits DER, the transport wants PEM). Returns a malloc'd NUL-terminated
170 * PEM buffer (caller frees), or NULL on parse/encode failure.
171 */
172static u_char *
173delegation_req_der_to_pem(const uint8_t *der, size_t der_len, size_t *pem_len)
174{
175 const unsigned char *p = der;
176 X509_REQ *req = d2i_X509_REQ(NULL, &p, (long) der_len);
177 BIO *bio;
178 char *data;
179 long n;
180 u_char *out = NULL;
181
182 if (req == NULL) {
183 return NULL;
184 }
185 bio = BIO_new(BIO_s_mem());
186 if (bio != NULL && PEM_write_bio_X509_REQ(bio, req) == 1) {
187 n = BIO_get_mem_data(bio, &data);
188 out = malloc((size_t) n + 1);
189 if (out != NULL) {
190 memcpy(out, data, (size_t) n);
191 out[n] = '\0';
192 *pem_len = (size_t) n;
193 }
194 }
195 BIO_free(bio);
196 X509_REQ_free(req);
197 return out;
198}

200/*
201 * delegation_send_csr - build a 200 response with the CSR PEM as the body
202 * and the delegation-id in X-Brix-Delegation-Id. Content-Type: application/
203 * x-pem-file (RFC-ish convention for a bare PEM payload; chosen over
204 * text/plain so a client can distinguish "this response IS a PEM
205 * document" from the plain-text error bodies delegation_reject sends).
206 *
207 * Returns the ngx_int_t status the CALLER must return up through
208 * webdav_metrics_return() — this function does NOT call
209 * ngx_http_send_header/ngx_http_output_filter's result into
210 * webdav_metrics_finalize_request itself, unlike delegation_reject.
211 * webdav_delegation_request_handle runs synchronously inside the content
212 * phase (no async body read is in flight), so self-finalizing here would
213 * double-finalize the request once nginx's content phase ALSO finalizes
214 * whatever this handler returns — see delegation.h's doc comment on
215 * webdav_delegation_request_handle for why that crashes the worker.
216 */
217static ngx_int_t
218delegation_send_csr(ngx_http_request_t *r, const char *id,
219 const u_char *pem, size_t pem_len)
220{
221 ngx_buf_t *b;
222 ngx_chain_t out;
223 ngx_table_elt_t *h;
224 u_char *buf;
225 ngx_str_t id_str;
226 u_char *id_copy;
227
228 buf = ngx_pnalloc(r->pool, pem_len);
229 if (buf == NULL) {
230 return NGX_HTTP_INTERNAL_SERVER_ERROR;
231 }
232 ngx_memcpy(buf, pem, pem_len);
233
234 b = ngx_pcalloc(r->pool, sizeof(*b));
235 if (b == NULL) {
236 return NGX_HTTP_INTERNAL_SERVER_ERROR;
237 }
238 b->pos = buf;
239 b->last = buf + pem_len;
240 b->memory = 1;
241 b->last_buf = 1;
242 out.buf = b;
243 out.next = NULL;
244
245 id_str.len = strlen(id);
246 id_str.data = (u_char *) id;
247 id_copy = ngx_pstrdup(r->pool, &id_str);
248 if (id_copy == NULL) {
249 return NGX_HTTP_INTERNAL_SERVER_ERROR;
250 }
251
252 h = ngx_list_push(&r->headers_out.headers);
253 if (h == NULL) {
254 return NGX_HTTP_INTERNAL_SERVER_ERROR;
255 }
256 h->hash = 1;
257 ngx_str_set(&h->key, "X-Brix-Delegation-Id");
258 h->value.data = id_copy;
259 h->value.len = id_str.len;
260
261 r->headers_out.status = NGX_HTTP_OK;
262 r->headers_out.content_length_n = (off_t) pem_len;
263 r->headers_out.content_type.data = (u_char *) "application/x-pem-file";
264 r->headers_out.content_type.len = sizeof("application/x-pem-file") - 1;
265 r->headers_out.content_type_len = r->headers_out.content_type.len;
266
267 if (ngx_http_send_header(r) == NGX_ERROR || r->header_only) {
268 return NGX_HTTP_OK;
269 }
270 return ngx_http_output_filter(r, &out);
271}

273/*
274 * delegation_request_reject - build a small plain-text error response
275 * WITHOUT finalizing (the synchronous-handler counterpart to
276 * delegation_reject, which self-finalizes for the async PUT paths). Returns

279static ngx_int_t
280delegation_request_reject(ngx_http_request_t *r, ngx_uint_t status,
281 const char *msg)
282{
283 ngx_buf_t *b;
284 ngx_chain_t out;
285 size_t len = strlen(msg);
286 u_char *buf = ngx_pnalloc(r->pool, len);
287
288 if (buf == NULL) {
289 return NGX_HTTP_INTERNAL_SERVER_ERROR;
290 }
291 ngx_memcpy(buf, msg, len);
292
293 b = ngx_pcalloc(r->pool, sizeof(*b));
294 if (b == NULL) {
295 return NGX_HTTP_INTERNAL_SERVER_ERROR;
296 }
297 b->pos = buf;
298 b->last = buf + len;
299 b->memory = 1;
300 b->last_buf = 1;
301 out.buf = b;
302 out.next = NULL;
303
304 r->headers_out.status = status;
305 r->headers_out.content_length_n = (off_t) len;
306 r->headers_out.content_type.data = (u_char *) "text/plain";

308 r->headers_out.content_type_len = r->headers_out.content_type.len;
309
310 if (ngx_http_send_header(r) == NGX_ERROR || r->header_only) {
311 return status;
312 }
313 return ngx_http_output_filter(r, &out);
314}

316ngx_int_t
317webdav_delegation_request_handle(ngx_http_request_t *r)
318{
319 ngx_http_brix_webdav_req_ctx_t *ctx =
320 ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
321 u_char *parent_pem = NULL;
322 size_t parent_pem_len = 0;
323 EVP_PKEY *newkey = NULL;
324 uint8_t *req_der = NULL;
325 size_t req_der_len = 0;
326 u_char *req_pem = NULL;
327 size_t req_pem_len = 0;
328 char err[160];
329 char id[BRIX_DELEG_ID_HEXLEN + 1];
330 ngx_int_t rc;
331
332 if (!delegation_client_authenticated(ctx)) {
333 return delegation_request_reject(r, NGX_HTTP_UNAUTHORIZED,
334 "GSI client-certificate authentication"
335 " required\n");
336 }
337
338 if (delegation_get_peer_pem(r, &parent_pem, &parent_pem_len) != NGX_OK) {
339 return delegation_request_reject(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
340 "cannot access client certificate\n");
341 }
342
343 err[0] = '\0';
344 {
345 const brix_gsi_blob_t parent_blob = { parent_pem, parent_pem_len };
346 const brix_gsi_err_t err_sink = { err, sizeof(err) };
347 brix_gsi_buf_t req_out = { NULL, 0 };
348
349 if (brix_gsi_build_pxyreq(&parent_blob, &newkey, &req_out,
350 &err_sink) != 0)
351 {
352 free(parent_pem);
353 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
354 "brix_delegation: build_pxyreq failed: %s", err);
355 return delegation_request_reject(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
356 "cannot build proxy request\n");
357 }
358 req_der = req_out.data;
359 req_der_len = req_out.len;
360 }
361 free(parent_pem);
362
363 req_pem = delegation_req_der_to_pem(req_der, req_der_len, &req_pem_len);
364 free(req_der);
365 if (req_pem == NULL) {
366 EVP_PKEY_free(newkey);
367 return delegation_request_reject(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
368 "cannot encode proxy request\n");
369 }
370
371 if (brix_deleg_store_put(brix_deleg_store(), newkey, ctx->dn,
372 id, sizeof(id)) != NGX_OK)
373 {
374 /* newkey ownership did NOT transfer on a failed put — free it here. */
375 EVP_PKEY_free(newkey);
376 free(req_pem);
377 return delegation_request_reject(r, NGX_HTTP_SERVICE_UNAVAILABLE,
378 "too many pending delegations, try again later\n");
379 }
380
381 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
382 "brix_delegation: issued getProxyReq id=%s", id);
383 rc = delegation_send_csr(r, id, req_pem, req_pem_len);
384 free(req_pem);
385 return rc;
386}
