4/*
5 * session/registry.h — shared-memory registries for kXR_bind.
6 *
7 * When a primary connection completes authentication the server inserts an
8 * entry mapping sessid → {dn, vo_list, token_auth}. A secondary connection
9 * presenting kXR_bind can then look up the sessid and inherit the same
10 * identity without re-authenticating.
11 *
12 * Primary connections also publish their readable file handles in a separate
13 * shared-memory table. Bound secondary connections never inherit raw file
14 * descriptor numbers because nginx workers do not share descriptors opened
15 * after fork. Instead they look up {sessid, handle_index} and reopen the
16 * same canonical path inside their own worker, validating dev/inode so a
17 * replaced path cannot be mistaken for the primary's open file.
18 *
19 * The table lives in a dedicated ngx_shm_zone_t so all worker processes share
20 * one consistent view. Concurrent access is serialised by a single
21 * ngx_shmtx_t spinlock embedded at the start of the shared region.
22 *
23 * Capacity: BRIX_SESSION_REGISTRY_SLOTS entries (default 1024). When the
24 * table is full, new inserts fail gracefully — the primary session continues
25 * normally and secondaries fall back to single-stream I/O.
26 */
27
28#include "core/ngx_brix_module.h"
29
30#define BRIX_SESSION_REGISTRY_SLOTS 1024

32/*
33 * Published-handle table capacity (Phase 31).
34 *
35 * The handle table is the single largest fixed shared-memory allocation in the
36 * module: every slot carries a full path (~4 KB), so sizing it at
37 * registry_slots (1024) x BRIX_MAX_FILES (16) = 16384 slots cost ~68 MB of
38 * RAM at idle — paid by every deployment regardless of load.
39 *
40 * Handles are published here only so that kXR_bind secondary streams (and proxy
41 * mode) can reach a primary session's open files across worker boundaries; the
42 * common single-stream case never publishes. We therefore size this table
43 * independently of the per-connection open-file limit (BRIX_MAX_FILES, which
44 * must stay 16) using a deliberately smaller pair of factors: a modest number of
45 * sessions that use bound streams, each publishing a handful of handles. When
46 * the table is full, publishing fails gracefully (the primary keeps working;
47 * secondaries fall back to single-stream I/O) — see brix_session_handle_*.
48 *
49 * Default 512 x 8 = 4096 slots ~= 17 MB, a ~50 MB saving. Raise these factors
50 * if a deployment genuinely runs many concurrent multi-file bound-stream
51 * transfers and observes handle-publish failures in the access log.
52 */
53#define BRIX_SESSION_HANDLE_SESSIONS 512
54#define BRIX_SESSION_HANDLES_PER_SESSION 8
55#define BRIX_SESSION_HANDLE_SLOTS \
56 (BRIX_SESSION_HANDLE_SESSIONS * BRIX_SESSION_HANDLES_PER_SESSION)

58/* ---- Struct: brix_session_entry_t ----
59 *
60 * WHAT: Single entry in the shared-memory session registry mapping sessid → {dn, vo_list, token_auth}. sessid[16] uniquely identifies an XRootD session; dn (distinguished name from GSI certificate) and vo_list (virtual organization authorization list) determine client identity and eligibility for operations. token_auth flag indicates whether JWT bearer token was used instead of GSI certificate. in_use marks slot occupancy — 0 means free, 1 means registered. Capacity: BRIX_SESSION_REGISTRY_SLOTS entries (default 1024).
61 */
62
63typedef struct {
64 u_char sessid[BRIX_SESSION_ID_LEN]; /* unique session identifier (16 bytes) */
65 char dn[512]; /* distinguished name from GSI certificate */
66 char vo_list[512]; /* virtual organization authorization list */
67 ngx_uint_t token_auth; /* 1 if JWT bearer token used, 0 for GSI */
68 ngx_uint_t in_use; /* slot occupancy: 0=free, 1=registered */
69 ngx_msec_t last_seen; /* Phase 27 F4: register/lookup time (ms);
70 LRU key for reap-on-full anti-exhaustion */
71} brix_session_entry_t;

73/* Phase 27 F4: a slot that is the global-LRU AND older than this minimum age is
74 * eligible for eviction when the table is full, so a slot-exhaustion attacker
75 * cannot permanently deny new logins. Sessions younger than this are never
76 * reaped (avoids thrashing freshly-registered legitimate sessions). */
77#define BRIX_SESSION_REAP_MIN_AGE_MS 60000

79/* ---- Struct: brix_shared_handle_entry_t ----
80 *
81 * WHAT: Single entry in the shared-memory handle table publishing file metadata for bound stream secondary connections. sessid+handle_index form a unique key identifying which primary session and which open handle this entry describes. readable/writable flags indicate channel direction; from_cache marks if the handle was opened via read-through cache. is_regular distinguishes files from directories. device/inode provide file identity verification so bound streams can detect path replacement after publish (stale reference attack prevention). cached_size stores file size at publish time for bound stream read length validation.
82 */
83
84typedef struct {
85 u_char sessid[BRIX_SESSION_ID_LEN]; /* primary session identifier — key component */
86 uint8_t handle_index; /* open handle index (0–255) — key component */
87 uint8_t readable; /* 1 if channel supports reading */
88 uint8_t writable; /* 1 if channel supports writing */
89 uint8_t from_cache; /* 1 if handle opened via read-through cache */
90 uint8_t is_regular; /* 1 for regular files, 0 for directories/collections */
91 ngx_uint_t in_use; /* slot occupancy: 0=free, 1=published */
92 dev_t device; /* file device number — stale reference verification */
93 ino_t inode; /* file inode number — stale reference verification */
94 off_t cached_size; /* file size at publish time — read length validation */
95 char path[BRIX_MAX_PATH + 1]; /* canonical confined path for reopen by secondary */
96} brix_shared_handle_entry_t;
