1/*
2 * proxy.c — X.509 RFC-3820 GSI proxy create / info / destroy (xrdgsiproxy).
3 *
4 * WHAT: Build a short-lived proxy certificate signed by the user's end-entity
5 * cert/key, carrying the proxyCertInfo extension (OID 1.3.6.1.5.5.7.1.14,
6 * id-ppl-inheritAll) so the GSI verifier accepts it; plus info/destroy.
7 * WHY: Users need a proxy before any GSI xrdcp/xrdfs. This is the one genuinely
8 * new crypto surface in the client; it is pure local OpenSSL (no wire, no
9 * libXrdCl/XrdCrypto).
10 * HOW: Mirror utils/make_proxy.py byte-for-byte on the proxyCertInfo DER and the
11 * subject scheme (userDN + CN=<serial>) so output is identical to the
12 * harness's reference proxy. Output chain = proxycert + usercert + proxykey,
13 * written atomically at mode 0400.
14 *
15 * Clean-room: the proxy/extension format is RFC 3820 (a published interface);
16 * OpenSSL X509 is a public API. No upstream implementation was consulted.
17 */

45/* Open a credential file as a BIO with brix_open_credfile's safety checks (no
46 * symlink, owned by euid, secret=1 → 0600), so the predictable /tmp/x509up_u<uid>
47 * proxy can't be hijacked via a planted symlink/file. BIO_CLOSE makes BIO_free
48 * close the underlying fd. Quiet on failure (returns NULL) — callers report. */
49struct bio_st *
50brix_credfile_bio(const char *path, int secret)
51{
52 int fd = brix_open_credfile(path, secret, NULL);
53 FILE *fp;
54 BIO *bio;
55
56 if (fd < 0) {
57 return NULL;
58 }
59 fp = fdopen(fd, "r");
60 if (fp == NULL) {
61 close(fd);
62 return NULL;
63 }
64 bio = BIO_new_fp(fp, BIO_CLOSE);
65 if (bio == NULL) {
66 fclose(fp);
67 }
68 return bio;
69}

107/* All OpenSSL/OS resources owned by brix_proxy_create. Every member starts NULL
108 * (fd = -1) so the cleanup helper can free exactly what was acquired so far, in
109 * the original single-exit ladder order, regardless of which step failed. */
110typedef struct {
111 int fd;
112 BIO *bio;
113 X509_EXTENSION *ext;
114 ASN1_OCTET_STRING *octet;
115 ASN1_OBJECT *pci_obj;
116 X509_NAME *subj;
117 X509 *proxy;
118 EVP_PKEY *pkey;
119 EVP_PKEY *ukey;
120 X509 *user;
121} proxy_build_ctx;
122
123/*
124 * proxy_build_cleanup — release every resource the build owns and return rc.
125 *
126 * WHAT: NULL-safe teardown of the proxy_build_ctx in the exact order the former
127 * single `done:` ladder used (fd, bio, ext, octet, pci_obj, subj, proxy,
128 * pkey, ukey, user).
129 * WHY: Lets brix_proxy_create become a flat early-return sequence with no goto,
130 * while preserving byte-identical free ordering and partial-init safety.
131 * HOW: Free each non-NULL member once, then return the caller's status code so
132 * callsites can `return proxy_build_cleanup(&c, rc)` on every exit.
133 */
134static int
135proxy_build_cleanup(proxy_build_ctx *c, int rc)
136{
137 if (c->fd >= 0) { close(c->fd); }
138 if (c->bio) { BIO_free(c->bio); }
139 if (c->ext) { X509_EXTENSION_free(c->ext); }
140 if (c->octet) { ASN1_OCTET_STRING_free(c->octet); }
141 if (c->pci_obj) { ASN1_OBJECT_free(c->pci_obj); }
142 if (c->subj) { X509_NAME_free(c->subj); }
143 if (c->proxy) { X509_free(c->proxy); }
144 if (c->pkey) { EVP_PKEY_free(c->pkey); }
145 if (c->ukey) { EVP_PKEY_free(c->ukey); }
146 if (c->user) { X509_free(c->user); }
147 return rc;
148}

150/*
151 * pxy_load_issuer — load the user end-entity cert + key (the proxy issuer).
152 *
153 * WHAT: Read `cert_path` into c->user and `key_path` into c->ukey via a scratch
154 * c->bio that is freed (and nulled) after each read, matching the former
155 * inline ladder exactly.
156 * WHY: The proxy is signed by, and inherits its subject/issuer from, the user
157 * credential; isolating the two PEM loads keeps brix_proxy_create flat.
158 * HOW: On any failure set *st and return -1 with whatever was acquired parked in
159 * *c so the caller's single proxy_build_cleanup frees it in ladder order.
160 */
161static int
162pxy_load_issuer(proxy_build_ctx *c, const char *cert_path, const char *key_path,
163 brix_status *st)
164{
165 c->bio = BIO_new_file(cert_path, "r");
166 if (c->bio == NULL) {
167 brix_status_set(st, XRDC_EUSAGE, 0, "cannot open user cert %s", cert_path);
168 return -1;
169 }
170 c->user = PEM_read_bio_X509(c->bio, NULL, NULL, NULL);
171 BIO_free(c->bio); c->bio = NULL;
172 if (c->user == NULL) { return ssl_fail(st, "parse user cert"); }
173
174 c->bio = BIO_new_file(key_path, "r");
175 if (c->bio == NULL) {
176 brix_status_set(st, XRDC_EUSAGE, 0, "cannot open user key %s", key_path);
177 return -1;
178 }
179 c->ukey = PEM_read_bio_PrivateKey(c->bio, NULL, NULL, NULL);
180 BIO_free(c->bio); c->bio = NULL;
181 if (c->ukey == NULL) { return ssl_fail(st, "parse user key"); }
182 return 0;
183}

185/*
186 * pxy_build_request — assemble the unsigned proxy certificate (RFC 3820).
187 *
188 * WHAT: Generate the ephemeral proxy keypair (c->pkey), a v3 X509 (c->proxy) with
189 * serial, subject = user DN + CN=<serial>, issuer = user subject, validity
190 * window, and the two critical extensions (proxyCertInfo + KeyUsage).
191 * WHY: This is the whole certificate-shape step; keeping the exact OpenSSL call
192 * order and DER-frozen proxyCertInfo here preserves byte-identical output.
193 * HOW: Build into *c so partial state is freed by proxy_build_cleanup; the two
194 * transient extension handles are freed-and-nulled after X509_add_ext just
195 * as the original code did. On failure set *st and return -1.
196 */
197static int
198pxy_build_request(proxy_build_ctx *c, int bits, int valid_hours, long serial,
199 brix_status *st)
200{
201 char serial_str[32];
202
203 /* Ephemeral proxy keypair. */
204 c->pkey = EVP_RSA_gen((unsigned) bits);
205 if (c->pkey == NULL) { return ssl_fail(st, "RSA keygen"); }
206
207 c->proxy = X509_new();
208 if (c->proxy == NULL) { return ssl_fail(st, "X509_new"); }
209 X509_set_version(c->proxy, 2); /* v3 */
210 ASN1_INTEGER_set(X509_get_serialNumber(c->proxy), serial);
211
212 /* Subject = user DN + CN=<serial>; issuer = user subject (RFC 3820). */
213 c->subj = X509_NAME_dup(X509_get_subject_name(c->user));
214 snprintf(serial_str, sizeof(serial_str), "%ld", serial);
215 if (c->subj == NULL
216 || X509_NAME_add_entry_by_NID(c->subj, NID_commonName, MBSTRING_ASC,
217 (unsigned char *) serial_str, -1, -1, 0) != 1
218 || X509_set_subject_name(c->proxy, c->subj) != 1
219 || X509_set_issuer_name(c->proxy, X509_get_subject_name(c->user)) != 1
220 || X509_set_pubkey(c->proxy, c->pkey) != 1) {
221 return ssl_fail(st, "build proxy subject");
222 }
223 X509_gmtime_adj(X509_getm_notBefore(c->proxy), -300);
224 X509_gmtime_adj(X509_getm_notAfter(c->proxy), (long) valid_hours * 3600);
225
226 /* proxyCertInfo (critical) — exact DER from make_proxy.py. */
227 c->octet = ASN1_OCTET_STRING_new();
228 c->pci_obj = OBJ_txt2obj(PROXY_CERT_INFO_OID, 1);
229 if (c->octet == NULL || c->pci_obj == NULL
230 || ASN1_OCTET_STRING_set(c->octet, PROXY_CERT_INFO_DER,
231 (int) sizeof(PROXY_CERT_INFO_DER)) != 1) {
232 return ssl_fail(st, "build proxyCertInfo");
233 }
234 c->ext = X509_EXTENSION_create_by_OBJ(NULL, c->pci_obj, 1, c->octet);
235 if (c->ext == NULL || X509_add_ext(c->proxy, c->ext, -1) != 1) {
236 return ssl_fail(st, "add proxyCertInfo");
237 }
238 X509_EXTENSION_free(c->ext); c->ext = NULL;
239
240 /* KeyUsage critical = digitalSignature. */
241 c->ext = X509V3_EXT_conf_nid(NULL, NULL, NID_key_usage, "critical,digitalSignature");
242 if (c->ext == NULL || X509_add_ext(c->proxy, c->ext, -1) != 1) {
243 return ssl_fail(st, "add KeyUsage");
244 }
245 X509_EXTENSION_free(c->ext); c->ext = NULL;
246 return 0;
247}

249/*
250 * pxy_sign — sign the assembled proxy with the user key (SHA-256).
251 *
252 * WHAT: X509_sign(c->proxy, c->ukey, EVP_sha256()); the single signing step.
253 * WHY: Named so the orchestrator reads as load → build → sign → assemble; the
254 * digest algorithm is frozen (matches the reference proxy).
255 * HOW: On failure set *st and return -1; nothing new is acquired here.
256 */
257static int
258pxy_sign(proxy_build_ctx *c, brix_status *st)
259{
260 if (X509_sign(c->proxy, c->ukey, EVP_sha256()) == 0) {
261 return ssl_fail(st, "sign proxy");
262 }
263 return 0;
264}

266/*
267 * pxy_assemble_pem — write proxycert + usercert + proxykey and finalize atomically.
268 *
269 * WHAT: Emit the three PEM objects to a 0600 temp file, then chmod 0400 + rename
270 * onto out_path, exactly as the former inline tail.
271 * WHY: The output-chain order and atomic-install semantics are the contract with
272 * every GSI consumer; keeping them in one helper freezes them.
273 * HOW: The temp fd is handed to a BIO (BIO_CLOSE) parked in *c; on any write or
274 * finalize failure the temp is unlinked here (path is local) and *c is left
275 * for proxy_build_cleanup. Returns 0 on success, -1 with *st set otherwise.
276 */
277static int
278pxy_assemble_pem(proxy_build_ctx *c, const char *out_path, brix_status *st)
279{
280 char tmp_path[1100];
281
282 snprintf(tmp_path, sizeof(tmp_path), "%s.tmp.%u", out_path, (unsigned) getpid());
283 c->fd = open(tmp_path, O_WRONLY | O_CREAT | O_TRUNC, 0600);
284 if (c->fd < 0) {
285 brix_status_set(st, XRDC_EUSAGE, 0, "cannot create %s", tmp_path);
286 return -1;
287 }
288 c->bio = BIO_new_fp(fdopen(c->fd, "w"), BIO_CLOSE);
289 c->fd = -1; /* owned by the BIO now */
290 if (c->bio == NULL
291 || PEM_write_bio_X509(c->bio, c->proxy) != 1
292 || PEM_write_bio_X509(c->bio, c->user) != 1
293 || PEM_write_bio_PrivateKey(c->bio, c->pkey, NULL, NULL, 0, NULL, NULL) != 1) {
294 ssl_fail(st, "write proxy chain");
295 BIO_free(c->bio); c->bio = NULL;
296 unlink(tmp_path);
297 return -1;
298 }
299 BIO_free(c->bio); c->bio = NULL;
300 if (chmod(tmp_path, 0400) != 0 || rename(tmp_path, out_path) != 0) {
301 brix_status_set(st, XRDC_EUSAGE, 0, "cannot finalize %s", out_path);
302 unlink(tmp_path);
303 return -1;
304 }
305 return 0;
306}

308int
309brix_proxy_create(const brix_proxy_opts *o, brix_status *st)
310{
311 char cert_path[1024], key_path[1024], out_path[1024];
312 proxy_build_ctx c = { .fd = -1 };
313 int valid_hours = (o && o->valid_hours > 0) ? o->valid_hours : 12;
314 int bits = (o && o->bits > 0) ? o->bits : 2048;
315 long serial = (long) time(NULL); /* arbitrary; verifier keys on PCI+chain */
316
317 if (o && o->user_cert) { snprintf(cert_path, sizeof(cert_path), "%s", o->user_cert); }
318 else { default_cred("X509_USER_CERT", "usercert.pem", cert_path, sizeof(cert_path)); }
319 if (o && o->user_key) { snprintf(key_path, sizeof(key_path), "%s", o->user_key); }
320 else { default_cred("X509_USER_KEY", "userkey.pem", key_path, sizeof(key_path)); }
321 if (o && o->out_path) { snprintf(out_path, sizeof(out_path), "%s", o->out_path); }
322 else { brix_proxy_default_path(out_path, sizeof(out_path)); }
323
324 if (pxy_load_issuer(&c, cert_path, key_path, st) != 0) {
325 return proxy_build_cleanup(&c, -1);
326 }
327 if (pxy_build_request(&c, bits, valid_hours, serial, st) != 0) {
328 return proxy_build_cleanup(&c, -1);
329 }
330 if (pxy_sign(&c, st) != 0) {
331 return proxy_build_cleanup(&c, -1);
332 }
333 if (pxy_assemble_pem(&c, out_path, st) != 0) {
334 return proxy_build_cleanup(&c, -1);
335 }
336 return proxy_build_cleanup(&c, 0);
337}

339int
340brix_proxy_info(const char *path, FILE *out, brix_status *st)
341{
342 char defp[1024];
343 BIO *bio;
344 X509 *cert;
345 char subj[512], issuer[512];
346
347 if (path == NULL) {
348 brix_proxy_default_path(defp, sizeof(defp));
349 path = defp;
350 }
351 bio = brix_credfile_bio(path, 1);
352 if (bio == NULL) {
353 /* A missing/unreadable proxy is a not-found condition, NOT a CLI
354 * usage error: the arguments were well-formed, the file just isn't
355 * there. Callers (xrdgsiproxy info) lean on XRDC_ENOENT to stay
356 * tolerant of an absent proxy the way stock xrdgsiproxy does. */
357 brix_status_set(st, XRDC_ENOENT, 0, "proxy file: %s not found", path);
358 return -1;
359 }
360 cert = PEM_read_bio_X509(bio, NULL, NULL, NULL);
361 BIO_free(bio);
362 if (cert == NULL) {
363 return ssl_fail(st, "parse proxy");
364 }
365
366 X509_NAME_oneline(X509_get_subject_name(cert), subj, sizeof(subj));
367 X509_NAME_oneline(X509_get_issuer_name(cert), issuer, sizeof(issuer));
368 fprintf(out, "path : %s\n", path);
369 fprintf(out, "subject : %s\n", subj);
370 fprintf(out, "issuer : %s\n", issuer);
371 {
372 int days = 0, secs = 0;
373 const ASN1_TIME *na = X509_get0_notAfter(cert);
374 if (ASN1_TIME_diff(&days, &secs, NULL, na)) {
375 long left = (long) days * 86400 + secs;
376 fprintf(out, "validity : %s (%ld:%02ld:%02ld remaining)\n",
377 left > 0 ? "valid" : "EXPIRED",
378 left > 0 ? left / 3600 : 0,
379 left > 0 ? (left % 3600) / 60 : 0,
380 left > 0 ? left % 60 : 0);
381 }
382 }
383 X509_free(cert);
384 return 0;
385}

387/*
388 * Phase 40 (c): programmatic GSI-proxy lifetime — store the seconds of validity
389 * remaining (negative if already expired) in *secs_left. Returns 0 on success,
390 * -1 if there is no proxy at `path` (default path when NULL) or it cannot be
391 * parsed. Pure local inspection (no network); used by the client-side pre-flight
392 * / auth-failure diagnostics so the user instantly sees an expired proxy.
393 */
394int
395brix_proxy_remaining(const char *path, long *secs_left)
396{
397 char defp[1024];
398 BIO *bio;
399 X509 *cert;
400 int days = 0, secs = 0;
401
402 if (secs_left == NULL) {
403 return -1;
404 }
405 if (path == NULL) {
406 brix_proxy_default_path(defp, sizeof(defp));
407 path = defp;
408 }
409 bio = brix_credfile_bio(path, 1);
410 if (bio == NULL) {
411 return -1;
412 }
413 cert = PEM_read_bio_X509(bio, NULL, NULL, NULL);
414 BIO_free(bio);
415 if (cert == NULL) {
416 return -1;
417 }
418 if (!ASN1_TIME_diff(&days, &secs, NULL, X509_get0_notAfter(cert))) {
419 X509_free(cert);
420 return -1;
421 }
422 X509_free(cert);
423 *secs_left = (long) days * 86400 + secs;
424 return 0;
425}

427int
428brix_proxy_destroy(const char *path, brix_status *st)
429{
430 char defp[1024];
431 struct stat sb;
432 int fd;
433
434 if (path == NULL) {
435 brix_proxy_default_path(defp, sizeof(defp));
436 path = defp;
437 }
438 if (stat(path, &sb) != 0) {
439 brix_status_set(st, XRDC_EUSAGE, 0, "no proxy at %s", path);
440 return -1;
441 }
442 /* Best-effort shred: overwrite the bytes before unlinking. */
443 fd = open(path, O_WRONLY);
444 if (fd >= 0) {
445 char zero[4096];
446 off_t left = sb.st_size;
447 memset(zero, 0, sizeof(zero));
448 while (left > 0) {
449 size_t n = (left < (off_t) sizeof(zero)) ? (size_t) left : sizeof(zero);
450 ssize_t w = write(fd, zero, n);
451 if (w <= 0) { break; }
452 left -= w;
453 }
454 (void) fsync(fd);
455 close(fd);
456 }
457 if (unlink(path) != 0) {
458 brix_status_set(st, XRDC_EUSAGE, 0, "cannot remove %s", path);
459 return -1;
460 }
461 return 0;
462}
