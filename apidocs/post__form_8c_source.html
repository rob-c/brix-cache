8/*
9 * WHAT: Resolve a policy-condition field name to its submitted form value.
10 * WHY: POST-policy conditions reference fields the same way the form submits
11 * them; AWS writes some condition keys with a leading '$' (e.g. "$key"),
12 * which we strip so "$key" and "key" map to the same value. The special
13 * "key" and "Content-Type" fields live in dedicated struct members rather
14 * than the generic fields[] table, so they are checked explicitly.
15 * Returns the value string, or NULL if no such field was submitted.
16 */
17const char *
18s3_post_field_value(const s3_post_form_t *form, const char *name)
19{
20 ngx_uint_t i;
21
22 /* AWS policy syntax writes condition keys as "$name"; normalise to "name". */
23 if (name[0] == '$') {
24 name++;
25 }
26
27 if (strcmp(name, "key") == 0) {
28 return form->key;
29 }
30
31 for (i = 0; i < form->nfields; i++) {
32 if (strcmp(form->fields[i].name, name) == 0) {
33 return form->fields[i].value;
34 }
35 }
36
37 if (strcmp(name, "Content-Type") == 0) {
38 return form->content_type;
39 }
40
41 return NULL;
42}

47 * WHY: Every field is appended to the generic fields[] table (so policy
48 * conditions can later look it up by name), AND the auth-relevant fields

50 * direct, fast access during signature verification. HOW: validate+copy
51 * once into the table (subject to the 64-field cap), then dispatch on the

117/*
118 * WHAT: Copy the value of a "boundary=" parameter (whose value begins at `p`,
119 * just past the '=') into `boundary`, handling quoted and token forms.
120 * WHY: Isolating the value-extraction keeps the parameter-walk in
121 * s3_post_boundary() flat and under the complexity gate; the two value
122 * encodings (quoted-string / bare token) and the length cap live here.
123 * HOW: A leading '"' selects the quoted form (value up to the closing '"');
124 * otherwise the token runs to the next separator/whitespace. The length
125 * is capped (RFC 2046 limits boundaries to 70 chars, we allow up to 200)
126 * to bound the caller's stack buffer, then copied NUL-terminated.
127 * Returns NGX_OK with `boundary` filled; NGX_ERROR if the value is unterminated,
128 * empty, or too long.
129 */
130static ngx_int_t
131pf_scan_boundary(u_char *p, u_char *end, char *boundary, size_t boundary_sz)
132{
133 u_char *q;
134 size_t len;
135
136 if (p < end && *p == '"') {
137 /* Quoted form: value is everything up to the closing '"'. */
138 p++;
139 q = p;
140 while (q < end && *q != '"') {
141 q++;
142 }
143 if (q == end) {
144 return NGX_ERROR; /* unterminated quoted boundary */
145 }
146 len = (size_t) (q - p);
147 } else {
148 /* Token form: value ends at the next separator/whitespace. */
149 q = p;
150 while (q < end && *q != ';' && *q != ' ' && *q != '\t') {
151 q++;
152 }
153 len = (size_t) (q - p);
154 }
155
156 if (len == 0 || len >= boundary_sz || len > 200) {
157 return NGX_ERROR;
158 }
159
160 ngx_memcpy(boundary, p, len);
161 boundary[len] = '\0';
162 return NGX_OK;
163}

166/*
167 * WHAT: Extract the multipart boundary from the request Content-Type header.
168 * WHY: POST Object only accepts "multipart/form-data; boundary=...". The
169 * boundary may be quoted ("...") or a bare token; both forms must be
170 * handled, and length is capped (RFC 2046 limits boundaries to 70 chars,
171 * we allow up to 200) to bound the stack buffer.
172 * HOW: Verify the media type, then walk the ';'-separated parameter list to
173 * find "boundary="; delegate the quoted/token value copy to
174 * pf_scan_boundary().
175 * Returns NGX_OK with `boundary` filled; NGX_DECLINED if not multipart / no
176 * boundary param; NGX_ERROR if the boundary is malformed or too long.
177 */
178ngx_int_t
179s3_post_boundary(ngx_http_request_t *r, char *boundary, size_t boundary_sz)
180{
181 ngx_table_elt_t *ct;
182 u_char *p, *end;
183
184 ct = brix_http_find_header(r, "Content-Type",
185 sizeof("Content-Type") - 1);
186 if (ct == NULL || ct->value.len == 0) {
187 return NGX_DECLINED;
188 }
189
190 if (ngx_strncasecmp(ct->value.data, (u_char *) "multipart/form-data",
191 sizeof("multipart/form-data") - 1) != 0)
192 {
193 return NGX_DECLINED;
194 }
195
196 p = ct->value.data;
197 end = ct->value.data + ct->value.len;
198
199 /* Walk the ';'-separated parameters looking for "boundary=". */
200 while (p < end) {
201 /* Skip leading separators/whitespace before a parameter token. */
202 while (p < end && (*p == ';' || *p == ' ' || *p == '\t')) {
203 p++;
204 }
205 if ((size_t) (end - p) >= sizeof("boundary=") - 1
206 && ngx_strncasecmp(p, (u_char *) "boundary=",
207 sizeof("boundary=") - 1) == 0)
208 {
209 p += sizeof("boundary=") - 1;
210 return pf_scan_boundary(p, end, boundary, boundary_sz);
211 }
212
213 /* Not this parameter — advance to the next ';' and retry. */
214 while (p < end && *p != ';') {
215 p++;
216 }
217 }
218
219 return NGX_DECLINED;
220}

223/*
224 * WHAT: Extract a ';'-delimited parameter (e.g. name= / filename=) from a
225 * Content-Disposition header line into `out`.
226 * WHY: Each multipart part header carries its field name and optional filename
227 * as Content-Disposition parameters; we need them to route the part. Value
228 * may be quoted or a bare token. HOW: scan ';'-separated params for one
229 * whose token equals `name` followed by '='; copy the quoted/token value.
230 * Returns NGX_OK if found (out filled), NGX_DECLINED if absent, NGX_ERROR if
231 * the value does not fit (out is pre-emptied so callers can ignore DECLINED).
232 */
233ngx_int_t
234s3_post_extract_param(const char *line, const char *name,
235 char *out, size_t outsz)
236{
237 const char *p;
238 size_t nlen;
239
240 out[0] = '\0';
241 nlen = strlen(name);
242 p = line;
243
244 /* Each iteration positions p at a ';' beginning the next parameter. */
245 while ((p = strchr(p, ';')) != NULL) {
246 p++;
247 while (*p == ' ' || *p == '\t') {
248 p++;
249 }
250
251 if (ngx_strncasecmp((u_char *) p, (u_char *) name, nlen) == 0
252 && p[nlen] == '=')
253 {
254 const char *v = p + nlen + 1; /* +1 steps over the '=' */
255 const char *e;
256 size_t len;
257
258 if (*v == '"') {
259 /* Quoted value: spans to the closing '"'. */
260 v++;
261 e = v;
262 while (*e != '\0' && *e != '"') {
263 e++;
264 }
265 } else {
266 e = v;
267 while (*e != '\0' && *e != ';' && *e != ' '
268 && *e != '\t')
269 {
270 e++;
271 }
272 }
273
274 len = (size_t) (e - v);
275 if (len >= outsz) {
276 return NGX_ERROR;
277 }
278 ngx_memcpy(out, v, len);
279 out[len] = '\0';
280 return NGX_OK;
281 }
282 }
283
284 return NGX_DECLINED;
285}

288/*
289 * WHAT: Substitute every "${filename}" token in form->key with the uploaded
290 * part's (already basename-reduced) filename, writing back into form->key.
291 * WHY: The S3 POST API lets the key template reference the upload's filename.
292 * Expansion must be bounded (S3_MAX_KEY) and the result must not contain
293 * control characters, since the key becomes a filesystem path.
294 * HOW: Walk the template; copy literal spans, and at each token emit the
295 * filename instead. Every append is length-checked against `left` so a
296 * too-long result fails cleanly rather than overflowing `expanded`.
297 * Returns NGX_OK (form->key updated), or NGX_ERROR on overflow / control chars.
298 */
299ngx_int_t
300s3_post_expand_filename(ngx_http_request_t *r, s3_post_form_t *form)
301{
302 static const char needle[] = "${filename}";
303 char expanded[S3_MAX_KEY];
304 const char *src;
305 char *dst;
306 size_t left;
307
308 if (strstr(form->key, needle) == NULL) {
309 return NGX_OK;
310 }
311
312 src = form->key;
313 dst = expanded;
314 left = sizeof(expanded);
315
316 while (*src != '\0') {
317 const char *m = strstr(src, needle);
318 size_t n;
319
320 if (m == NULL) {
321 /* No further token: copy the remaining literal tail (incl. NUL). */
322 n = strlen(src);
323 if (n >= left) {
324 return NGX_ERROR;
325 }
326 ngx_memcpy(dst, src, n + 1);
327 break;
328 }
329
330 /* Copy the literal span before the token, then the filename in its place. */
331 n = (size_t) (m - src);
332 if (n + strlen(form->filename) >= left) {
333 return NGX_ERROR;
334 }
335 ngx_memcpy(dst, src, n);
336 dst += n;
337 left -= n;
338 n = strlen(form->filename);
339 ngx_memcpy(dst, form->filename, n);
340 dst += n;
341 left -= n;
342 src = m + sizeof(needle) - 1;
343 }
344
345 if (brix_http_str_has_ctl((u_char *) expanded, strlen(expanded))) {
346 return NGX_ERROR;
347 }
348
349 ngx_cpystrn((u_char *) form->key, (u_char *) expanded,
350 sizeof(form->key));
351 (void) r;
352 return NGX_OK;
353}

373/*
374 * WHAT: Parse one multipart part header line (already NUL-terminated in `line`)
375 * into the part's Content-Disposition name/filename and Content-Type.
376 * WHY: Splitting the per-line header dispatch out of the part loop keeps
377 * pf_parse_part_headers() flat; the two header types we recognise and
378 * their bounds checks live in one place.
379 * HOW: Content-Disposition supplies the field `name` and optional `filename`
380 * (basename-reduced); Content-Type fills `content_type` after skipping
381 * leading whitespace. Unrecognised headers are ignored.
382 * Returns NGX_OK on success (hdr possibly updated), NGX_ERROR on an over-long
383 * or malformed recognised header value.
384 */
385static ngx_int_t
386pf_parse_header_line(const char *line, pf_part_headers_t *hdr)
387{
388 if (ngx_strncasecmp((u_char *) line,
389 (u_char *) "Content-Disposition:",
390 sizeof("Content-Disposition:") - 1) == 0)
391 {
392 if (s3_post_extract_param(line, "name", hdr->name, sizeof(hdr->name))
393 == NGX_ERROR
394 || s3_post_extract_param(line, "filename", hdr->filename,
395 sizeof(hdr->filename)) == NGX_ERROR)
396 {
397 return NGX_ERROR;
398 }
399 if (hdr->filename[0] != '\0') {
400 s3_post_basename(hdr->filename);
401 }
402 return NGX_OK;
403 }
404
405 if (ngx_strncasecmp((u_char *) line, (u_char *) "Content-Type:",
406 sizeof("Content-Type:") - 1) == 0)
407 {
408 const char *v = line + sizeof("Content-Type:") - 1;
409 while (*v == ' ' || *v == '\t') {
410 v++;
411 }
412 if (ngx_strlen(v) >= sizeof(hdr->content_type)) {
413 return NGX_ERROR;
414 }
415 ngx_cpystrn((u_char *) hdr->content_type, (u_char *) v,
416 sizeof(hdr->content_type));
417 }
418
419 return NGX_OK;
420}

423/*
424 * WHAT: Read one multipart part's header block (from *pos up to the blank line
425 * separating headers from body) into `hdr`.
426 * WHY: The header block is a bounded CRLF-terminated line loop; keeping it in
427 * its own helper flattens the part loop in s3_post_parse_form().
428 * HOW: Each CRLF-terminated line is copied into a bounded stack buffer and
429 * dispatched via pf_parse_header_line(); a zero-length line ends the
430 * block. On success *pos is advanced past the blank line's CRLF to the
431 * first body byte.
432 * Returns NGX_OK (*pos advanced), or NGX_ERROR on a missing CRLF or over-long
433 * line/header.
434 */
435static ngx_int_t
436pf_parse_part_headers(u_char **pos, u_char *end, pf_part_headers_t *hdr)
437{
438 u_char *p = *pos;
439
440 for (;;) {
441 u_char *line_end;
442 size_t line_len;
443 char line[1024];
444
445 line_end = s3_memmem(p, (size_t) (end - p), (u_char *) "\r\n", 2);
446 if (line_end == NULL) {
447 return NGX_ERROR;
448 }
449
450 line_len = (size_t) (line_end - p);
451 if (line_len == 0) {
452 /* Blank line: headers end here; part body starts after CRLF. */
453 *pos = line_end + 2;
454 return NGX_OK;
455 }
456 if (line_len >= sizeof(line)) {
457 return NGX_ERROR;
458 }
459
460 ngx_memcpy(line, p, line_len);
461 line[line_len] = '\0';
462 p = line_end + 2;
463
464 if (pf_parse_header_line(line, hdr) != NGX_OK) {
465 return NGX_ERROR;
466 }
467 }
468}

495/*
496 * WHAT: Consume one complete multipart part from *pos: its header block, its
497 * body (up to `delim`), and the delimiter that follows.
498 * WHY: Isolating a single part's framing keeps s3_post_parse_form()'s outer
499 * loop short and under the complexity gate; the "another part follows"
500 * vs "closing delimiter" decision lives here in one place.
501 * HOW: Parse headers via pf_parse_part_headers(); the body runs to the next
502 * "\r\n--BOUNDARY"; route the file part via pf_capture_file() and other
503 * fields via s3_post_store_field(); then advance *pos past the delimiter.
504 * Returns NGX_OK when a normal inter-part CRLF follows (another part expected),
505 * NGX_DONE at the closing "--BOUNDARY--", or NGX_ERROR on any framing violation.
506 */
507static ngx_int_t
508pf_parse_one_part(u_char **pos, u_char *end, const char *delim,
509 size_t delim_len, s3_post_form_t *form)
510{
511 pf_part_headers_t hdr = { "", "", "" };
512 u_char *content;
513 u_char *next;
514 size_t content_len;
515
516 if (pf_parse_part_headers(pos, end, &hdr) != NGX_OK) {
517 return NGX_ERROR;
518 }
519
520 /* Every part must have declared a field name in its headers. */
521 if (hdr.name[0] == '\0') {
522 return NGX_ERROR;
523 }
524
525 /* Part body runs from *pos up to the next "\r\n--BOUNDARY" delimiter. */
526 content = *pos;
527 next = s3_memmem(content, (size_t) (end - content),
528 (u_char *) delim, delim_len);
529 if (next == NULL) {
530 return NGX_ERROR;
531 }
532
533 content_len = (size_t) (next - content);
534 if (strcmp(hdr.name, "file") == 0) {
535 pf_capture_file(form, content, content_len, hdr.filename,
536 hdr.content_type);
537 } else if (s3_post_store_field(form, hdr.name, content, content_len)
538 != NGX_OK)
539 {
540 return NGX_ERROR;
541 }
542
543 /* Advance past the delimiter and decide what follows it. */
544 *pos = next + delim_len;
545 if ((size_t) (end - *pos) >= 2 && (*pos)[0] == '-' && (*pos)[1] == '-') {
546 return NGX_DONE; /* "--BOUNDARY--": end of the multipart body */
547 }
548 if ((size_t) (end - *pos) < 2 || (*pos)[0] != '\r' || (*pos)[1] != '\n') {
549 return NGX_ERROR; /* delimiter must be followed by "--" or CRLF */
550 }
551 *pos += 2; /* CRLF: another part follows */
552 return NGX_OK;
553}

556/*
557 * WHAT: Parse a multipart/form-data body into `form` — both the named form
558 * fields and the single "file" part (whose bytes are referenced in place).
559 * WHY: This is the wire parser for browser POST uploads; it must handle binary
560 * file content and is hostile-input facing, so every span is bounds-checked.
561 * HOW (RFC 2046 framing):
562 * - "--BOUNDARY" CRLF opens the body (marker).
563 * - Each part is a CRLF-terminated header block, a blank line, then the body,
564 * followed by "\r\n--BOUNDARY" (delim) introducing the next part.
565 * - A trailing "--" after a delimiter ("\r\n--BOUNDARY--") closes the body.
566 * Each part is consumed by pf_parse_one_part(); NGX_DONE from it signals the
567 * closing delimiter.
568 * Returns NGX_OK when the closing delimiter is reached, NGX_ERROR on any
569 * framing violation or oversized header.
570 */
571ngx_int_t
572s3_post_parse_form(ngx_http_request_t *r, u_char *body, size_t body_len,
573 const char *boundary, s3_post_form_t *form)
574{
575 char marker[256];
576 char delim[258];
577 size_t marker_len, delim_len;
578 u_char *pos, *end;
579 ngx_int_t rc;
580
581 /* marker = opening "--BOUNDARY"; delim = inter-part "\r\n--BOUNDARY". */
582 marker_len = (size_t) snprintf(marker, sizeof(marker), "--%s", boundary);
583 delim_len = (size_t) snprintf(delim, sizeof(delim), "\r\n--%s", boundary);
584 if (marker_len >= sizeof(marker) || delim_len >= sizeof(delim)) {
585 return NGX_ERROR;
586 }
587
588 pos = body;
589 end = body + body_len;
590
591 /* Body must begin with the opening marker followed by CRLF. */
592 if ((size_t) (end - pos) < marker_len
593 || ngx_memcmp(pos, marker, marker_len) != 0)
594 {
595 return NGX_ERROR;
596 }
597 pos += marker_len;
598 if ((size_t) (end - pos) < 2 || pos[0] != '\r' || pos[1] != '\n') {
599 return NGX_ERROR;
600 }
601 pos += 2;
602
603 /* Outer loop: one iteration consumes one complete part. */
604 while (pos < end) {
605 rc = pf_parse_one_part(&pos, end, delim, delim_len, form);
606 if (rc == NGX_DONE) {
607 (void) r;
608 return NGX_OK; /* closing "--BOUNDARY--" reached */
609 }
610 if (rc != NGX_OK) {
611 return NGX_ERROR;
612 }
613 }
614
615 /* Ran off the end without seeing the closing "--BOUNDARY--": malformed. */
616 (void) r;
617 return NGX_ERROR;
618}
