1/* client/lib/vfs.c
2 *
3 * WHAT: Façade dispatch for brix_vfs_* API + a small backend registry.
4 * Routes URL→scheme→backend; thin wrappers call the per-handle vtable.
5 * WHY: Copy endpoints (local POSIX, block device, S3/object) need a single
6 * open-file abstraction. The registry lets backends (A3/A4/A5) plug in
7 * without touching this file.
8 * HOW: g_backends[] static array; lazy one-time init via pthread_once calls
9 * the three backend accessors (defined in tasks A3/A4/A5). URL→scheme:
10 * s3://…/s3s://… if brix_is_web_url && is_s3; block:// or /dev/... →
11 * "block"; else "file" (strip file:// prefix). Façade calls ops vtable.
12 * ngx-free; no goto; functional/modular design.
13 */

21/* Backend registry */
22#define VFS_MAX_BACKENDS 8
23
24static const brix_vfs_backend *g_backends[VFS_MAX_BACKENDS];
25static int g_n_backends;
26static pthread_once_t g_init_once = PTHREAD_ONCE_INIT;

28/*
29 * brix_vfs_register_backend — add *be to the registry.
30 *
31 * WHAT: appends be to g_backends[]; silently ignores NULL and overflow.
32 * WHY: called by each backend accessor during lazy init; callers never
33 * need a return value (double-register is benign).
34 * HOW: guard NULL and capacity, then store.
35 */
36void
37brix_vfs_register_backend(const brix_vfs_backend *be)
38{
39 if (be == NULL || g_n_backends >= VFS_MAX_BACKENDS) {
40 return;
41 }
42 g_backends[g_n_backends++] = be;
43}

57/*
58 * vfs_init_backends — one-time registration of all known backends.
59 *
60 * WHAT: called via pthread_once on the first brix_vfs_open or stat_url.
61 * WHY: defers backend init until needed; avoids ordering issues at startup.
62 * HOW: guards each call on a non-NULL weak accessor before invoking it.
63 */
64static void
65vfs_init_backends(void)
66{
67 if (brix_vfs_posix_backend != NULL) {
68 brix_vfs_register_backend(brix_vfs_posix_backend());
69 }
70 if (brix_vfs_block_backend != NULL) {
71 brix_vfs_register_backend(brix_vfs_block_backend());
72 }
73 if (brix_vfs_s3_backend != NULL) {
74 brix_vfs_register_backend(brix_vfs_s3_backend());

78/* URL→scheme routing */
79/*
80 * vfs_url_to_scheme — classify a URL string into a backend scheme + effective path.
81 *
82 * WHAT: fills *scheme_out and *path_out from the URL grammar.
83 * WHY: one place owns the URL grammar for the VFS layer; keeps open/stat DRY.
84 * HOW:
85 * s3://… / s3s://… → scheme "s3" / "s3s"; path = original URL (backend parses it).
86 * block://… → scheme "block"; path = URL + 8 (strip "block://").
87 * /dev/… → scheme "block"; path = URL (raw device path).
88 * file:///… → scheme "file"; path = URL + 7 (strip "file://").
89 * bare path / other → scheme "file"; path = URL.
90 * http/dav (non-s3) → *scheme_out = NULL (no VFS backend; caller errors out).
91 */
92static void
93vfs_url_to_scheme(const char *url, const char **scheme_out,
94 const char **path_out)
95{
96 if (brix_is_web_url(url)) {
97 brix_weburl wu;
98 if (brix_weburl_parse(url, &wu) == 0 && wu.is_s3) {
99 *scheme_out = (wu.proto == XRDC_WEB_S3S) ? "s3s" : "s3";
100 *path_out = url;
101 return;
102 }
103 /* Non-s3 web URL (http/https/dav/davs) — no VFS backend. */
104 *scheme_out = NULL;
105 *path_out = url;
106 return;
107 }
108
109 if (strncmp(url, "block://", 8) == 0) {
110 *scheme_out = "block";
111 *path_out = url + 8;
112 return;
113 }
114
115 if (strncmp(url, "/dev/", 5) == 0) {
116 *scheme_out = "block";
117 *path_out = url;
118 return;
119 }
120
121 if (strncmp(url, "file://", 7) == 0) {
122 *scheme_out = "file";
123 *path_out = url + 7;
124 return;
125 }
126
127 /* Bare path or any other local-looking string → POSIX file backend. */
128 *scheme_out = "file";
129 *path_out = url;
130}

132/*
133 * vfs_find_backend — look up a registered backend by scheme name.
134 *
135 * WHAT: linear scan of g_backends[0..g_n_backends); returns NULL on miss.
136 * WHY: registry is small (≤8 entries); a linear scan is fast and obvious.
137 * HOW: strcmp on be->scheme; returns first match.
138 */
139static const brix_vfs_backend *
140vfs_find_backend(const char *scheme)
141{
142 int i;
143 for (i = 0; i < g_n_backends; i++) {
144 if (strcmp(g_backends[i]->scheme, scheme) == 0) {
145 return g_backends[i];
146 }
147 }
148 return NULL;
149}

151/* Façade: URL-level operations */
152/*
153 * brix_vfs_open — open a storage URL and return a VFS file handle.
154 *
155 * WHAT: routes URL to the right backend; calls be->open; stores result in *out.
156 * WHY: single entry point for copy.c/tools; hides backend selection.
157 * HOW: lazy-init backends; classify URL; look up backend; delegate to be->open.
158 * Returns 0 on success, -1 with *st filled on error.
159 */
160int
161brix_vfs_open(const char *url, int flags, const brix_vfs_open_opts *opts,
162 brix_vfs_file **out, brix_status *st)
163{
164 const char *scheme;
165 const char *path;
166 const brix_vfs_backend *be;
167
168 pthread_once(&g_init_once, vfs_init_backends);
169 vfs_url_to_scheme(url, &scheme, &path);
170
171 if (scheme == NULL) {
172 brix_status_set(st, XRDC_EUSAGE, 0,
173 "vfs: no backend for non-s3 web URL '%s'", url);
174 return -1;
175 }
176
177 be = vfs_find_backend(scheme);
178 if (be == NULL) {
179 brix_status_set(st, XRDC_EUSAGE, 0,
180 "vfs: no registered backend for scheme '%s'", scheme);
181 return -1;
182 }
183
184 return be->open(be, path, flags, opts, out, st);
185}
