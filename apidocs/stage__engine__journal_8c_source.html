1/*
2 * stage_engine_journal.c — the durable journal + dead-letter substrate of the
3 * async-staging engine (split from stage_engine.c, phase-79).
4 *
5 * WHAT: Owns the per-worker journal directory and the crash-visibility records
6 * that back an async submit: minting a request id, persisting a QUEUED
7 * record, removing a completed one, bumping the retry counter, and moving a
8 * permanently-denied flush to the dead-letter directory. Also the engine's
9 * one-time init that sets the journal directory.
10 *
11 * WHY: stage_engine.c owned three concerns and exceeded the file-size cap. The
12 * durable-record substrate is a self-contained concern: the byte mover and
13 * the scheduler both reach it only through the seam in
14 * stage_engine_internal.h. The exact on-disk record I/O, the O_TRUNC crash-
15 * safe update, and the attempt/age dead-letter caps are unchanged by the
16 * split — this is a re-home, not a behavior change.
17 *
18 * HOW: Every record write opens O_WRONLY|O_CREAT|O_TRUNC|O_CLOEXEC, writes the
19 * full fixed-size brix_sreq_t, and fsyncs. stage_deny_terminal (public,
20 * shared by the scheduler and reconcile halves) bumps attempts, re-persists,
21 * and on cap dead-letters + emits a loud tombstone. The two dead-letter
22 * helpers stay file-local.
23 */

36/* ---- SP4: the durable async queue -----------------------------------------
37 * An async submit is DEFERRED rather than run inline: the request is appended to
38 * a per-worker in-memory pending list (holding the live src/dst instances, which
39 * are the memoised per-worker tier instances - they outlive the request) and,
40 * when a journal dir is configured, persisted as a small record for crash
41 * visibility/recovery. brix_stage_scheduler_tick() (a per-worker timer) drains
42 * the list, runs each mover, and drops the stage copy of a completed FLUSH. This
43 * generalises the FRM queue model to SD instances (section 11); the full physical
44 * extraction of src/frm/ is the remaining SP4/SP5 migration. */
45
46char stage_journal_dir[1024]; /* "" = in-memory only */
47static uint64_t stage_reqid_seq;

67/* Persist (best-effort) a QUEUED request record so a crash leaves a recoverable
68 * row; removed on completion. Skipped when no journal dir is configured. */
69void
70stage_journal_write(const stage_pending_t *p)
71{
72 brix_sreq_t rec;
73 char path[1200];
74 int fd;
75
76 if (stage_journal_dir[0] == '\0') {
77 return;
78 }
79 if ((size_t) snprintf(path, sizeof(path), "%s/%s.req",
80 stage_journal_dir, p->reqid) >= sizeof(path))
81 {
82 return;
83 }
84 ngx_memzero(&rec, sizeof(rec));
85 snprintf(rec.reqid, sizeof(rec.reqid), "%s", p->reqid);
86 rec.kind = p->kind;
87 rec.state = BRIX_SREQ_QUEUED;
88 snprintf(rec.src_driver, sizeof(rec.src_driver), "%s",
89 (p->src->driver && p->src->driver->name) ? p->src->driver->name : "");
90 snprintf(rec.src_key, sizeof(rec.src_key), "%s", p->src_key);
91 snprintf(rec.dst_driver, sizeof(rec.dst_driver), "%s",
92 (p->dst->driver && p->dst->driver->name) ? p->dst->driver->name : "");
93 snprintf(rec.dst_key, sizeof(rec.dst_key), "%s", p->dst_key);
94 snprintf(rec.export_root, sizeof(rec.export_root), "%s", p->export_root);
95 rec.cred = p->cred; /* copy the owner identity into the durable record */
96 rec.enqueued_at = (int64_t) time(NULL);
97
98 fd = open(path, O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC, 0600);
99 if (fd < 0) {
100 return;
101 }
102 if (write(fd, &rec, sizeof(rec)) == (ssize_t) sizeof(rec)) {
103 (void) fsync(fd);
104 }
105 (void) close(fd);
106}

126/* Write the updated rec (with bumped attempts) back to the active journal slot.
127 *
128 * WHAT: Overwrites <journal_dir>/<reqid>.req atomically enough for a crash-safe
129 * update — O_TRUNC on the existing file is sufficient (the record is
130 * readable even if we crash mid-write; the next drive will re-bump).
131 *
132 * WHY: Persisting attempts across restarts is what prevents the unbounded retry
133 * loop: a restart would otherwise reset the in-memory state to zero while
134 * the disk record still carries the accumulated count.
135 *
136 * HOW: Open for write with O_TRUNC, write the full record, fsync. */
137static void
138stage_journal_update_rec(const char *journal_dir, const brix_sreq_t *rec)
139{
140 char path[1200];
141 int fd;
142
143 if (journal_dir == NULL || journal_dir[0] == '\0') {
144 return;
145 }
146 if ((size_t) snprintf(path, sizeof(path), "%s/%s.req",
147 journal_dir, rec->reqid) >= sizeof(path))
148 {
149 return;
150 }
151 fd = open(path, O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC, 0600);
152 if (fd < 0) {
153 return;
154 }
155 if (write(fd, rec, sizeof(*rec)) == (ssize_t) sizeof(*rec)) {
156 (void) fsync(fd);
157 }
158 (void) close(fd);
159}

161/* Stamp a decoded record FAILED (last_errno + finished_at) and re-persist it to
162 * the active journal slot. Shared tail of the two public FAILED helpers below;
163 * the record stays in the active journal so the scheduler / reconcile retry it. */
164static void
165stage_journal_persist_failed(const char *journal_dir, brix_sreq_t *rec,
166 int last_errno)
167{
168 rec->state = BRIX_SREQ_FAILED;
169 rec->last_errno = last_errno;
170 rec->finished_at = (int64_t) time(NULL);
171 stage_journal_update_rec(journal_dir, rec);
172}

174void
175stage_journal_mark_failed(const char *journal_dir, const char *reqid,
176 int last_errno)
177{
178 char path[1200];
179 char rbuf[sizeof(brix_sreq_t)];
180 brix_sreq_t rec;
181 int fd;
182 ssize_t n;
183
184 if (journal_dir == NULL || journal_dir[0] == '\0') {
185 return;
186 }
187 if ((size_t) snprintf(path, sizeof(path), "%s/%s.req",
188 journal_dir, reqid) >= sizeof(path))
189 {
190 return;
191 }
192 fd = open(path, O_RDONLY | O_CLOEXEC);
193 if (fd < 0) {
194 return; /* record already removed / never written */
195 }
196 n = read(fd, rbuf, sizeof(rbuf));
197 (void) close(fd);
198 if (brix_sreq_decode(rbuf, (size_t) n, &rec) != NGX_OK) {
199 return; /* corrupt slot — leave it for reconcile */
200 }
201 stage_journal_persist_failed(journal_dir, &rec, last_errno);
202}

215/* Move the active journal record to <journal_dir>/deadletter/<reqid>.req,
216 * creating the deadletter directory on demand (0700).
217 *
218 * WHAT: rename(2) from the active slot to the deadletter slot; creates the
219 * subdirectory on the first call. If rename fails (e.g. cross-device),
220 * falls back to copy + unlink.
221 *
222 * WHY: Moving the file out of the active journal directory ensures the
223 * scheduler and reconcile stop picking it up while preserving the bytes
224 * and the stage copy for operator recovery.
225 *
226 * HOW: Build the two paths, mkdir the deadletter dir (EEXIST is OK), rename. */
227static void
228stage_journal_move_to_deadletter(const char *journal_dir, const char *reqid,
229 ngx_log_t *log)
230{
231 char src_path[1200];
232 char dl_dir[1200];
233 char dst_path[1300];
234 int fd;
235 char buf[sizeof(brix_sreq_t)];
236 ssize_t n;
237
238 if (journal_dir == NULL || journal_dir[0] == '\0') {
239 return;
240 }
241 if ((size_t) snprintf(src_path, sizeof(src_path), "%s/%s.req",
242 journal_dir, reqid) >= sizeof(src_path))
243 {
244 return;
245 }
246 if ((size_t) snprintf(dl_dir, sizeof(dl_dir), "%s/deadletter",
247 journal_dir) >= sizeof(dl_dir))
248 {
249 return;
250 }
251 if ((size_t) snprintf(dst_path, sizeof(dst_path), "%s/%s.req",
252 dl_dir, reqid) >= sizeof(dst_path))
253 {
254 return;
255 }
256
257 if (mkdir(dl_dir, 0700) != 0 && errno != EEXIST) {
258 ngx_log_error(NGX_LOG_ERR, log, errno,
259 "xrootd stage: dead-letter mkdir \"%s\" failed", dl_dir);
260 return;
261 }
262
263 if (rename(src_path, dst_path) == 0) {
264 return;
265 }
266 /* rename failed (cross-device or other): copy + unlink as fallback */
267 fd = open(src_path, O_RDONLY | O_CLOEXEC);
268 if (fd < 0) {
269 return;
270 }
271 n = read(fd, buf, sizeof(buf));
272 (void) close(fd);
273 if (n > 0) {
274 int dst_fd = open(dst_path, O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC,
275 0600);
276 if (dst_fd >= 0) {
277 ssize_t wn = write(dst_fd, buf, (size_t) n);
278 (void) wn; /* best-effort copy; -Werror=unused-result */
279 (void) fsync(dst_fd);
280 (void) close(dst_fd);
281 }
282 }
283 (void) unlink(src_path);
284}

286/* Dead-letter terminal for a permanently denied flush.
287 *
288 * WHAT: Increments rec->attempts, re-persists the updated record to the active
289 * journal slot, then checks whether the attempt cap
290 * (BRIX_STAGE_DENY_MAX_ATTEMPTS) or the age cap
291 * (BRIX_STAGE_DENY_MAX_AGE_SEC) has been reached. When either cap fires,
292 * moves the active record to <journal_dir>/deadletter/<reqid>.req, emits a
293 * loud NGX_LOG_ERR naming the reqid, principal, key and dst_key, and
294 * returns 1 (stop re-driving). Returns 0 when below both caps (keep
295 * retrying later).
296 *
297 * WHY: A BRIX_XFER_DENIED result means the per-user credential is permanently
298 * missing or expired; the deny mode opted out of service-cred fallback.
299 * Without a cap the scheduler tick and restart-reconcile re-drive the
300 * same record forever. The cap bounds the loop while preserving the
301 * stage copy for operator recovery: a dead-lettered write is NEVER flushed
302 * on the wrong identity — dead-letter means STOP, not "flush as service".
303 *
304 * HOW: Called by stage_complete (BRIX_XFER_DENIED) and stage_reconcile_one
305 * (errno==EACCES from brix_sd_stage_reflush). The rec pointer is the
306 * decoded brix_sreq_t from the on-disk record (already bumped by the
307 * caller's re-read, so we bump it here before persist). journal_dir is
308 * passed explicitly so the function is unit-testable without the module
309 * global. */
310int
311stage_deny_terminal(const char *journal_dir, const char *reqid,
312 brix_sreq_t *rec, ngx_log_t *log)
313{
314 int64_t age_sec;
315
316 /* Bump and persist: always update attempts so the count survives a restart.
317 * Even if we are about to dead-letter, the persisted count is in the
318 * deadletter copy (which survives after the move). */
319 rec->attempts++;
320 stage_journal_update_rec(journal_dir, rec);
321
322 /* Check both caps: attempt count OR age triggers dead-letter. */
323 age_sec = (int64_t) time(NULL) - rec->enqueued_at;
324
325 if (rec->attempts < BRIX_STAGE_DENY_MAX_ATTEMPTS
326 && age_sec < (int64_t) BRIX_STAGE_DENY_MAX_AGE_SEC)
327 {
328 return 0; /* below both caps — keep record in active journal for retry */
329 }
330
331 /* Both-or-either cap reached: move to deadletter and emit a loud tombstone. */
332 ngx_log_error(NGX_LOG_ERR, log, 0,
333 "xrootd stage: flush DEAD-LETTERED (reqid=%s principal=\"%s\" "
334 "key=%s dst=\"%s\" attempts=%uD age=%lds) - "
335 "credential permanently missing/expired in deny mode; "
336 "stage copy retained in deadletter/ for operator recovery",
337 reqid,
338 rec->cred.principal[0] ? rec->cred.principal : "-",
339 rec->cred.key[0] ? rec->cred.key : "-",
340 rec->dst_key,
341 rec->attempts,
342 (long) age_sec);
343
344 stage_journal_move_to_deadletter(journal_dir, reqid, log);
345 return 1;
346}
