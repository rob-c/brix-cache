22/* Parse a hex string into bytes. */
23static void
24hex(const char *s, unsigned char *out, size_t n)
25{
26 for (size_t i = 0; i < n; i++) {
27 unsigned v;
28 sscanf(s + 2 * i, "%2x", &v);
29 out[i] = (unsigned char) v;
30 }
31}

41/* golden RRInfo wire values — the real XrdSsiRRInfo layout: byte0 = reqCmd,
42 * bytes1-3 = reqId big-endian, bytes4-7 = reqSize little-endian. Verified against
43 * live libXrdSsi traffic (e.g. rwt id=0 on the wire = 0100000000000000). */
44static void
45test_decode_golden(void)
46{
47 unsigned char off[8];
48 int cmd; uint32_t id, sz;
49
50 /* rxq id=5 sz=100 -> [00][00 00 05][64 00 00 00] */
51 hex("0000000564000000", off, 8);
52 brix_ssi_rrinfo_decode(off, &cmd, &id, &sz);
53 CHECK(cmd == BRIX_SSI_CMD_RXQ);
54 CHECK(id == 5);
55 CHECK(sz == 100);
56
57 /* can id=0x123456 sz=4096 -> [02][12 34 56][00 10 00 00] */
58 hex("0212345600100000", off, 8);
59 brix_ssi_rrinfo_decode(off, &cmd, &id, &sz);
60 CHECK(cmd == BRIX_SSI_CMD_CAN);
61 CHECK(id == 0x123456);
62 CHECK(sz == 4096);
63
64 /* rwt id=1 sz=0 -> [01][00 00 01][00 00 00 00] */
65 hex("0100000100000000", off, 8);
66 brix_ssi_rrinfo_decode(off, &cmd, &id, &sz);
67 CHECK(cmd == BRIX_SSI_CMD_RWT);
68 CHECK(id == 1);
69 CHECK(sz == 0);
70
71 /* rwt id=0 sz=0 — the exact bytes a real libXrdSsi client sends on the
72 * response-wait, which the old [id_lo..][cmd] layout mis-decoded as id=1. */
73 hex("0100000000000000", off, 8);
74 brix_ssi_rrinfo_decode(off, &cmd, &id, &sz);
75 CHECK(cmd == BRIX_SSI_CMD_RWT);
76 CHECK(id == 0);
77 CHECK(sz == 0);
78}
