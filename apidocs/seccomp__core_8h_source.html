1/* ---- File: src/core/seccomp/seccomp_core.h — ngx-free seccomp filter core ----
2 *
3 * WHAT: The pure, nginx-free half of the D-3 worker syscall filter. Owns the
4 * allowlist / named-dangerous tables and the libseccomp build+load, with
5 * zero dependency on ngx_* types so it links into a standalone C unit test
6 * (tests/c/test_seccomp.c) that forks children and proves execve/ptrace are
7 * killed under enforce, an allowlisted syscall survives, and audit never
8 * kills. seccomp.c is the thin ngx wrapper that logs via the error sink.
9 *
10 * WHY: A duplicated mini-filter in the test would not exercise the shipped
11 * allowlist, so the very table that ships must be the table under test —
12 * hence the split (same ngx-free-core convention as wverify / opaque).
13 *
14 * HOW: brix_seccomp_core_apply() builds and loads the filter on the CALLING
15 * thread group, matching the shape documented in seccomp.h. Modes mirror
16 * BRIX_SECCOMP_* (0/1/2). Failures are reported through an error sink
17 * callback so the core need not know about ngx logging.
18 */
19#ifndef BRIX_CORE_SECCOMP_CORE_H
20#define BRIX_CORE_SECCOMP_CORE_H

22/* Modes — value-identical to BRIX_SECCOMP_OFF/AUDIT/ENFORCE in tunables.h. */
23#define BRIX_SECCOMP_CORE_OFF 0
24#define BRIX_SECCOMP_CORE_AUDIT 1
25#define BRIX_SECCOMP_CORE_ENFORCE 2

27/* Return codes for brix_seccomp_core_apply(). */
28#define BRIX_SECCOMP_CORE_OK 0 /* filter in force (or mode==off) */
29#define BRIX_SECCOMP_CORE_ERR -1 /* build/load failed (sink called) */
30#define BRIX_SECCOMP_CORE_UNAVAIL -2 /* built without libseccomp */

32/*
33 * Error sink: invoked with the failing syscall name (NULL for whole-filter

35 * May be NULL.
36 */
37typedef void (*brix_seccomp_err_fn)(void *ud, const char *name, int rc);

39/*
40 * Build and load the worker filter for `mode` on the calling thread group.
41 * On success returns BRIX_SECCOMP_CORE_OK and, when non-NULL, sets *out_allow
42 * and *out_deny to the number of allow/deny rules actually installed. Returns
43 * BRIX_SECCOMP_CORE_ERR (sink invoked) if the filter could not be constructed
44 * or loaded, or BRIX_SECCOMP_CORE_UNAVAIL if this binary was built without
45 * libseccomp and a non-off mode was requested. mode==off is always OK.
46 */

50#endif /* BRIX_CORE_SECCOMP_CORE_H */
