1/*
2 * tpc.c - HTTP-TPC COPY dispatcher and shared request helpers for the WebDAV
3 * module.
4 *
5 * WHAT: Owns the COPY dispatcher (ngx_http_brix_webdav_tpc_handle_copy) that
6 * routes a request to the pull or push direction, plus the small request
7 * helpers shared across the TPC files: the dashboard display identity, the
8 * session-xfer note, the authorization gate, the live-transfer registry
9 * add, the OAuth2 subject-token extract, and the Authorization: Bearer
10 * header append.
11 * WHY: This file was one 1324-line module — far over the 500-line cap. The
12 * phase-79 split moves the push handler to tpc_push.c, the COPY request
13 * parsing + credential/user-proxy helpers to tpc_copy.c, and the staged
14 * pull execution to tpc_pull.c. The dispatcher and the helpers that every
15 * direction shares stay here; the cross-file entry points are declared in
16 * tpc_internal_split.h.
17 * HOW: The dispatcher parses the pull-xor-push headers, delegates push to
18 * tpc_push.c, and for a pull collects/credential-decorates the transfer
19 * headers (tpc_copy.c), prepares the staged target and runs the tiered pull
20 * (tpc_pull.c). Every helper here is a small side-effect-honest primitive
21 * reused by those files. No behaviour change from the split.
22 */

88/*
89 * Authorize a TPC against the request identity: src_path is the read scope,
90 * dst_path the write scope (NULL when the operation only reads, i.e. push).
91 * Returns NGX_OK if permitted, else NGX_HTTP_FORBIDDEN (and bumps the bad-request
92 * metric). This is the access-control gate before any data movement starts.
93 */
94ngx_int_t
95webdav_tpc_authorize(ngx_http_request_t *r, const ngx_str_t *src_path,
96 const ngx_str_t *dst_path)
97{
98 if (brix_tpc_check_authz(webdav_tpc_request_identity(r), src_path,
99 dst_path, r->connection->log)
100 != NGX_OK)
101 {
102 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_BAD_REQUEST]);
103 return NGX_HTTP_FORBIDDEN;
104 }
105
106 return NGX_OK;
107}

109/*
110 * Register a new in-flight transfer in the cross-process TPC registry so it is
111 * visible to /metrics and the dashboard. Returns a non-zero transfer id used to
112 * update/remove the entry as the transfer progresses; returns 0 if the registry
113 * is full (caller maps that to 503).
114 */
115uint64_t
116webdav_tpc_register_transfer(ngx_http_request_t *r, ngx_uint_t direction,
117 const char *src, const char *dst, off_t bytes_total)
118{
119 brix_tpc_transfer_t transfer;
120 ngx_str_t src_str;
121 ngx_str_t dst_str;
122
123 if (src == NULL || dst == NULL) {
124 return 0;
125 }
126
127 src_str.data = (u_char *) src;
128 src_str.len = ngx_strlen(src);
129 dst_str.data = (u_char *) dst;
130 dst_str.len = ngx_strlen(dst);
131
132 ngx_memzero(&transfer, sizeof(transfer));
133 transfer.protocol = BRIX_TPC_PROTO_WEBDAV;
134 transfer.direction = direction;
135 transfer.src_url = src_str;
136 transfer.dst_path = dst_str;
137 transfer.bytes_total = bytes_total > 0 ? bytes_total : 0;
138 transfer.state = BRIX_TPC_STATE_PENDING;
139
140 return brix_tpc_registry_add(&transfer, r->connection->log);
141}

143/*
144 * Pull the bearer token out of the request's Authorization header for use as the
145 * "subject token" in OAuth2 token-exchange delegation. A missing/non-bearer
146 * header is not an error: *subject_token stays NULL and NGX_OK is returned
147 * (oidc-agent mode does not need it; token-exchange mode will fail later).
148 * NGX_ERROR only on allocation failure. The token is copied (NUL-terminated)
149 * into the request pool so it outlives the header table.
150 */
151ngx_int_t
152webdav_tpc_extract_subject_token(ngx_http_request_t *r,
153 ngx_table_elt_t *auth_hdr,
154 const char **subject_token)
155{
156 ngx_str_t bearer;
157 ngx_int_t rc;
158 char *token;
159
160 *subject_token = NULL;
161
162 if (auth_hdr == NULL) {
163 return NGX_OK;
164 }
165
166 rc = brix_http_extract_bearer(&auth_hdr->value, &bearer);
167 if (rc != NGX_OK) {
168 return NGX_OK;
169 }
170
171 token = webdav_tpc_pstrndup0(r->pool, bearer.data, bearer.len);
172 if (token == NULL) {
173 return NGX_ERROR;
174 }
175
176 *subject_token = token;
177 return NGX_OK;
178}

180ngx_int_t
181webdav_tpc_add_bearer_header(ngx_http_request_t *r, ngx_array_t *headers,
182 ngx_str_t *delegated_token)
183{
184 size_t total_len;
185 ngx_str_t *dst;
186
187 total_len = sizeof("Authorization: Bearer ") - 1 + delegated_token->len;
188 dst = ngx_array_push(headers);
189 if (dst == NULL) {
190 return NGX_HTTP_INTERNAL_SERVER_ERROR;
191 }
192
193 dst->data = ngx_pnalloc(r->pool, total_len + 1);
194 if (dst->data == NULL) {
195 return NGX_HTTP_INTERNAL_SERVER_ERROR;
196 }
197
198 ngx_memcpy(dst->data, "Authorization: Bearer ",
199 sizeof("Authorization: Bearer ") - 1);
200 ngx_memcpy(dst->data + sizeof("Authorization: Bearer ") - 1,
201 delegated_token->data, delegated_token->len);
202 dst->len = total_len;
203 dst->data[dst->len] = '\0';
204 return NGX_OK;
205}

207/**
208 * WHAT: Handle HTTP-TPC (Third-Party Copy) COPY requests — implements GridFTP-style
209 * cross-server file transfer layered on WebDAV. Supports two modes: pull (fetch remote → local)
210 * and push (local → upload remote). Both modes use external curl process to perform the actual
211 * data transfer while nginx handles authentication, path confinement, and HTTP header management.
212 * Returns 201 Created for new files or 204 No Content for overwrites on pull; 201 Created on
213 * push success. Rejects requests with ambiguous headers (both Source + Destination present) or
214 * missing required headers via 400 Bad Request. Uses temporary file staging (<path>.nginx-xrootd-tpc.<pid>.<time>)
215 * for pull mode to ensure atomic commit — rename/link on success, unlink on failure.
216 */
217ngx_int_t
218ngx_http_brix_webdav_tpc_handle_copy(ngx_http_request_t *r)
219{
220 ngx_http_brix_webdav_loc_conf_t *conf;
221 ngx_table_elt_t *source_hdr;
222 ngx_table_elt_t *dest_hdr;
223 ngx_array_t *transfer_headers = NULL;
224 char *source_url;
225 char path[WEBDAV_MAX_PATH];
226 struct stat sb;
227 ngx_int_t rc;
228 ngx_flag_t overwrite;
229 brix_staged_file_t staged;
230 ngx_uint_t n_streams;
231 webdav_tpc_pull_ctx_t pl;
232
233 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
234
235 source_hdr = webdav_tpc_find_header(r, "Source", sizeof("Source") - 1);
236 dest_hdr = webdav_tpc_find_header(r, "Destination",
237 sizeof("Destination") - 1);
238
239 rc = webdav_tpc_validate_copy_headers(source_hdr, dest_hdr);
240 if (rc != NGX_OK) {
241 return rc;
242 }
243
244 /* X-Number-Of-Streams: N — capped at tpc_max_streams (default 1; multi-stream
245 * disabled unless brix_webdav_tpc_max_streams > 1 in the config). */
246 n_streams = webdav_tpc_parse_stream_count(r, conf);
247
248 if (source_hdr == NULL) {
249 /* Push mode: Destination present, no Source. */
250 return webdav_tpc_handle_push(r, conf, dest_hdr);
251 }
252
253 rc = webdav_tpc_source_url(r, source_hdr, &source_url);
254 if (rc != NGX_OK) {
255 return rc;
256 }
257
258 rc = webdav_tpc_collect_transfer_headers(r, &transfer_headers);
259 if (rc != NGX_OK) {
260 return rc;
261 }
262
263 /* OAuth2/OIDC credential delegation: obtain a delegated token for the source
264 * and inject it into transfer_headers before the curl subprocess runs. */
265 rc = webdav_tpc_apply_credential_delegation(r, conf, source_url,
266 transfer_headers);
267 if (rc != NGX_OK) {
268 return rc;
269 }
270
271 /* Opportunistic default: absent an explicit client delegation, forward the
272 * requesting user's own captured token so the pull acts as the END USER.
273 * A no-op when forwarding is off or no token is present — never a denial. */
274 rc = webdav_tpc_forward_user_bearer(r, conf, transfer_headers);
275 if (rc != NGX_OK) {
276 return rc;
277 }
278
279 rc = webdav_tpc_parse_overwrite(r, &overwrite);
280 if (rc != NGX_OK) {
281 return rc;
282 }
283
284 ngx_memzero(&pl, sizeof(pl));
285
286 /* Present the requesting user's delegated x509 proxy to the source (END USER
287 * identity, not our service cert) when available; otherwise fall back to
288 * conf->tpc_cert. See webdav_tpc_apply_user_proxy for the full contract. */
289 rc = webdav_tpc_apply_user_proxy(r, conf, &pl);
290 if (rc != NGX_OK) {
291 return rc;
292 }
293
294 pl.r = r;
295 pl.conf = conf;
296 pl.source_url = source_url;
297 pl.path = path;
298 pl.path_len = sizeof(path);
299 pl.sb = &sb;
300 pl.transfer_headers = transfer_headers;
301 pl.staged = &staged;
302 pl.n_streams = n_streams;
303 pl.overwrite = overwrite;
304
305 rc = webdav_tpc_prepare_pull_target(&pl);
306 if (rc != NGX_OK) {
307 return rc;
308 }
309
310 webdav_tpc_pull_start_accounting(r, path, source_url);
311
312 /* Run the staged pull across the three execution tiers (marker/thread/sync),
313 * then commit and finalise. The staged temp file is aborted (unlinked) on
314 * every error exit so a failed pull never leaves a partial file behind. */
315 return webdav_tpc_pull_execute(&pl);
316}
