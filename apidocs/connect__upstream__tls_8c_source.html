17void
18brix_proxy_tls_handshake_done(ngx_connection_t *uconn)
19{
20 brix_proxy_ctx_t *proxy = uconn->data;
21 brix_ctx_t *ctx = proxy->client_ctx;
22
23 if (ctx == NULL || ctx->destroyed) {
24 brix_proxy_cleanup(proxy);
25 return;
26 }
27
28 if (!uconn->ssl->handshaked) {
29 BRIX_PROXY_METRIC_INC(ctx, upstream_connect_errors);
30 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
31 brix_proxy_abort(proxy, "proxy: upstream TLS handshake failed");
32 return;
33 }
34
35 /* Belt-and-braces: when verification is enabled on the CTX a bad chain/host
36 * already fails the handshake above; this makes the trust decision explicit
37 * and survives refactors. Harmless (X509_V_OK) when verification is off. */
38 if (SSL_get_verify_result(uconn->ssl->connection) != X509_V_OK) {
39 BRIX_PROXY_METRIC_INC(ctx, upstream_connect_errors);
40 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
41 brix_proxy_abort(proxy, "proxy: upstream TLS peer verification failed");
42 return;
43 }
44
45 /* Restore normal event handlers and transition to bootstrap */
46 uconn->read->handler = brix_proxy_read_handler;
47 uconn->write->handler = brix_proxy_write_handler;
48 proxy->state = XRD_PX_BOOTSTRAP;
49 proxy->bs_phase = XRD_PX_BS_HANDSHAKE;
50 proxy->rhdr_pos = 0;
51
52 if (brix_proxy_flush(proxy) == NGX_ERROR) {
53 BRIX_PROXY_METRIC_INC(ctx, upstream_connect_errors);
54 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
55 brix_proxy_abort(proxy, "proxy: upstream write after TLS failed");
56 return;
57 }
58
59 if (proxy->wbuf_pos < proxy->wbuf_len) {
60 if (ngx_handle_write_event(uconn->write, 0) != NGX_OK) {
61 BRIX_PROXY_METRIC_INC(ctx, upstream_connect_errors);
62 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
63 brix_proxy_abort(proxy, "proxy: write arm after TLS failed");
64 }
65 return;
66 }
67
68 if (ngx_handle_read_event(uconn->read, 0) != NGX_OK) {
69 BRIX_PROXY_METRIC_INC(ctx, upstream_connect_errors);
70 BRIX_PROXY_UP_INC(proxy, upstream_connect_errors);
71 brix_proxy_abort(proxy, "proxy: read arm after TLS failed");
72 }
73}

75 * WHAT: nginx SSL callback invoked after upstream TLS handshake completes.
76 * Restores normal read/write handlers, transitions state to BOOTSTRAP,
77 * flushes the bootstrap buffer, and arms both read and write events.
78 *
79 * WHY: TLS handshake runs asynchronously; when it finishes we must switch back
