60/* Print one labelled "%Y-%m-%d %H:%M:%S" time line in UTC (matching official
61 * xrdfs, which formats stat times with gmtime); falls back to the raw epoch. */
62void
63print_stat_time(const char *label, long epoch)
64{
65 time_t t = (time_t) epoch;
66 struct tm tmv;
67 char tb[32];
68
69 if (gmtime_r(&t, &tmv) != NULL
70 && strftime(tb, sizeof(tb), "%Y-%m-%d %H:%M:%S", &tmv) > 0) {
71 printf("%s%s\n", label, tb);
72 } else {
73 printf("%s%ld\n", label, epoch);
74 }
75}

103/* WHAT: emit one stat result as a single-line JSON object for xrdfs stat -j.
104 * WHY: separates JSON formatting from command logic; JSON path is cleanly
105 * isolated so any change to the wire fields is confined here.
106 * HOW: jsonout helpers handle all string escaping so hostile path components
107 * (quotes, control bytes) cannot produce malformed JSON. Extended fields
108 * (mode/owner/group) are guarded by have_ext to match the human output. */
109void
110json_statinfo(const char *path, const brix_statinfo *si)
111{
112 fputc('{', stdout);
113 brix_json_kv_str(stdout, "path", path, 1);
114 brix_json_kv_ll(stdout, "size", (long long) si->size, 1);
115 brix_json_kv_ll(stdout, "mtime", (long long) si->mtime, 1);
116 brix_json_kv_ll(stdout, "flags", (long long) si->flags, 1);
117 brix_json_kv_bool(stdout, "is_dir", (si->flags & kXR_isDir) != 0,
118 si->have_ext);
119 if (si->have_ext) {
120 /* mode as a leading-zero octal string ("0755") rather than a decimal
121 * integer: octal is the conventional POSIX representation and avoids
122 * ambiguity (493 vs "0755"). Consumers compare strings, not integers. */
123 char modebuf[8];
124 snprintf(modebuf, sizeof(modebuf), "0%03o", si->mode & 07777);
125 brix_json_kv_str(stdout, "mode", modebuf, 1);
126 brix_json_kv_str(stdout, "owner", si->owner, 1);
127 brix_json_kv_str(stdout, "group", si->group, 0);
128 }
129 fputs("}\n", stdout);
130}

133/* parse_chmod_mode — accept the stock xrdfs 9-char symbolic form ("rwxr-xr-x",
134 * XrdClFS.cc ConvertMode) AND an octal absolute mode ("755") as a local
135 * extension. Returns the permission bits, or -1 on a malformed mode. The stock
136 * client takes ONLY the symbolic 9-char form, so users/tools that pass it (and
137 * our own conformance suite) must get the right bits — previously strtol(…,8)
138 * turned "rwxr-xr-x" into 0, silently setting mode 000. */
139int
140parse_chmod_mode(const char *s)
141{
142 size_t n = strlen(s);
143
144 if (n == 9) {
145 int i, m = 0, ok = 1;
146 for (i = 0; i < 9; i++) {
147 char want = "rwx"[i % 3];
148 if (s[i] == want) {
149 m |= (1 << (8 - i)); /* pos 0 -> 0400 … pos 8 -> 0001 */
150 } else if (s[i] != '-') {
151 ok = 0;
152 break;
153 }
154 }
155 if (ok) {
156 return m;
157 }
158 }
159
160 {
161 char *endp = NULL;
162 long v = strtol(s, &endp, 8);
163 if (endp != s && *endp == '\0' && v >= 0) {
164 return (int) (v & 07777);
165 }
166 }
167 return -1;
168}

203/* Parse a POSIX touch -t stamp: [[CC]YY]MMDDhhmm[.ss] (local time) into *out
204 * (tv_nsec = 0). Returns 0 / -1 on a malformed stamp. */
205int
206touch_parse_time(const char *s, struct timespec *out)
207{
208 char d[16];
209 size_t dn = 0, i;
210 int sec = 0, year, mon, day, hour, min;
211 const char *dot = strchr(s, '.');
212 const char *p;
213 struct tm tmv;
214 time_t now = time(NULL), t;
215
216 for (i = 0; s[i] != '\0' && s[i] != '.'; i++) {
217 if (!isdigit((unsigned char) s[i]) || dn >= sizeof(d) - 1) { return -1; }
218 d[dn++] = s[i];
219 }
220 d[dn] = '\0';
221 if (dot != NULL) {
222 if (strlen(dot + 1) != 2 || !isdigit((unsigned char) dot[1])
223 || !isdigit((unsigned char) dot[2])) {
224 return -1;
225 }
226 sec = two_digits(dot + 1);
227 }
228 if (dn != 8 && dn != 10 && dn != 12) { return -1; }
229
230 localtime_r(&now, &tmv); /* default year from "now" */
231 year = tmv.tm_year + 1900;
232 p = d;
233 if (dn == 12) { /* CCYYMMDDhhmm */
234 year = two_digits(p) * 100 + two_digits(p + 2);
235 p += 4;
236 } else if (dn == 10) { /* YYMMDDhhmm (POSIX pivot at 69) */
237 int yy = two_digits(p);
238 year = (yy >= 69) ? 1900 + yy : 2000 + yy;
239 p += 2;
240 }
241 mon = two_digits(p); day = two_digits(p + 2);
242 hour = two_digits(p + 4); min = two_digits(p + 6);
243
244 memset(&tmv, 0, sizeof(tmv));
245 tmv.tm_year = year - 1900;
246 tmv.tm_mon = mon - 1;
247 tmv.tm_mday = day;
248 tmv.tm_hour = hour;
249 tmv.tm_min = min;
250 tmv.tm_sec = sec;
251 tmv.tm_isdst = -1;
252 t = mktime(&tmv);
253 if (t == (time_t) -1) { return -1; }
254 out->tv_sec = t;
255 out->tv_nsec = 0;
256 return 0;
257}

278/* Strict non-negative decimal parse: rejects empty, trailing garbage, sign, and
279 * overflow. Fills *out and returns 0; on any error returns -1. */
280int
281parse_u64_strict(const char *s, unsigned long long *out)
282{
283 char *end;
284 unsigned long long v;
285 if (s == NULL || *s == '\0' || *s == '-') {
286 return -1;
287 }
288 errno = 0;
289 v = strtoull(s, &end, 10);
290 if (errno != 0 || end == s || *end != '\0') {
291 return -1;
292 }
293 *out = v;
294 return 0;
295}
