28/* ---- F10 refcounted blobs + dedup ----------------------------------------- *
29 * Introspect the catalog directly (a second SQLite connection, as a pytest
30 * would) to prove sharing: objects.blob_id tells us which physical blob backs a
31 * path, blobs.refcount tells us how many rows share it. */

33/* q_blob_id — the blob_id backing `path` (or "" if the row is missing). */
34static void
35q_blob_id(const char *root, const char *path, char *out, size_t cap)
36{
37 char db[PATH_MAX];
38 sqlite3 *h = NULL;
39 sqlite3_stmt *q = NULL;
40
41 out[0] = '\0';
42 snprintf(db, sizeof(db), "%s/catalog.db", root);
43 CHECK(sqlite3_open(db, &h) == SQLITE_OK, "blobid db open");
44 if (sqlite3_prepare_v2(h,
45 "SELECT blob_id FROM objects WHERE path = ?1;", -1, &q, NULL)
46 == SQLITE_OK)
47 {
48 sqlite3_bind_text(q, 1, path, -1, SQLITE_STATIC);
49 if (sqlite3_step(q) == SQLITE_ROW) {
50 const unsigned char *b = sqlite3_column_text(q, 0);
51
52 snprintf(out, cap, "%s", b ? (const char *) b : "");
53 }
54 }
55 sqlite3_finalize(q);
56 sqlite3_close(h);
57}

59/* q_refcount — a blob's tracked refcount, or -1 if there is no row (which the
60 * driver reads as the implicit single reference). */

62q_refcount(const char *root, const char *blob_id)

64 char db[PATH_MAX];

67 int n = -1;

70 CHECK(sqlite3_open(db, &h) == SQLITE_OK, "refcount db open");
71 if (sqlite3_prepare_v2(h,
72 "SELECT refcount FROM blobs WHERE blob_id = ?1;", -1, &q, NULL)
73 == SQLITE_OK)
74 {
75 sqlite3_bind_text(q, 1, blob_id, -1, SQLITE_STATIC);
76 if (sqlite3_step(q) == SQLITE_ROW) {
77 n = sqlite3_column_int(q, 0);
78 }
79 }
80 sqlite3_finalize(q);

107/* staged_put — publish `data` at `path` through the atomic staged path (the same
108 * path a wire PUT takes), so an overwrite fires F11 version capture / F10 dedup.
109 * 0 or -1. */
110static int
111staged_put(brix_sd_instance_t *inst, const char *path, const char *data,
112 size_t len)
113{
114 int err = 0;
115 brix_sd_staged_t *s = D->staged_open(inst, path, 0644, &err);
116
117 if (s == NULL) {
118 errno = err;
119 return -1;
120 }
121 if (D->staged_write(s, data, len, 0) != (ssize_t) len) {
122 D->staged_abort(s);
123 return -1;
124 }
125 return D->staged_commit(s, 0) == NGX_OK ? 0 : -1;
126}

128/* trunc_write — O_TRUNC-create `path` (breaks any share), write `data`, close. */
129static int
130trunc_write(brix_sd_instance_t *inst, const char *path, const char *data,
131 size_t len)
132{
133 int err = 0;
134 brix_sd_obj_t *o;
135 ssize_t n;
136
137 o = D->open(inst, path,
138 BRIX_SD_O_WRITE | BRIX_SD_O_READ | BRIX_SD_O_CREATE
139 | BRIX_SD_O_TRUNC,
140 0644, &err);
141 if (o == NULL) {
142 return -1;
143 }
144 n = D->pwrite(o, data, len, 0);
145 pb_close(o);
146 return (n == (ssize_t) len) ? 0 : -1;
147}

166void
167test_dedup_refs(void)
168{
169 char root[] = "/tmp/pb_dedup.XXXXXX";
170 brix_sd_instance_t inst = {0};
171 brix_sd_pblock_conf_t conf = {0};
172 char ba[PBLOCK_BLOB_ID_CAP], bb[PBLOCK_BLOB_ID_CAP];
173 char buf[64];
174 /* 10 bytes over a 4-byte stripe ⇒ 3 blocks: byte-verify walks real blocks. */
175 const char *DATA = "abcdefghij";
176 off_t bytes = 0;
177
178 CHECK(mkdtemp(root) != NULL, "mkdtemp");
179 lab_write_sidecar(root, "dedup=1");
180 conf.root = root;
181 conf.busy_timeout_ms = 2000;
182 conf.block_size = 4;
183 inst.driver = D;
184 inst.caps = D->caps;
185 CHECK(D->init(&inst, &conf) == NGX_OK, "dedup init");
186
187 /* SUCCESS: two identical PUTs fold onto one blob (refcount 2). */
188 CHECK(write_file(&inst, "/d1", DATA, 10) == 0, "seed d1");
189 CHECK(write_file(&inst, "/d2", DATA, 10) == 0, "seed d2");
190 q_blob_id(root, "/d1", ba, sizeof(ba));
191 q_blob_id(root, "/d2", bb, sizeof(bb));
192 CHECK(ba[0] && strcmp(ba, bb) == 0, "identical content shares blob");
193 CHECK(q_refcount(root, ba) == 2, "shared blob refcount 2 (got %d)",
194 q_refcount(root, ba));
195 CHECK(read_file(&inst, "/d1", buf, sizeof(buf)) == 10
196 && memcmp(buf, DATA, 10) == 0, "d1 reads back");
197 CHECK(read_file(&inst, "/d2", buf, sizeof(buf)) == 10
198 && memcmp(buf, DATA, 10) == 0, "d2 reads back");
199
200 /* Unlinking one sharer decrements, never removes the shared blocks. */
201 CHECK(D->unlink(&inst, "/d1", 0) == NGX_OK, "unlink d1");
202 CHECK(q_refcount(root, bb) == 1, "refcount 1 after one unlink (got %d)",
203 q_refcount(root, bb));
204 CHECK(read_file(&inst, "/d2", buf, sizeof(buf)) == 10
205 && memcmp(buf, DATA, 10) == 0, "survivor d2 intact");
206
207 /* server_copy = O(metadata) CoW: dst shares the src blob, refcount 2. */
208 CHECK(D->server_copy(&inst, "/d2", "/d3", &bytes) == NGX_OK, "cow copy");
209 q_blob_id(root, "/d3", ba, sizeof(ba));
210 CHECK(strcmp(ba, bb) == 0, "cow copy shares src blob");
211 CHECK(q_refcount(root, bb) == 2, "cow refcount 2 (got %d)",
212 q_refcount(root, bb));
213
214 /* ERROR/CoW break: overwriting a shared path forks a private blob; the
215 * sibling is untouched and the two blob_ids diverge. */
216 CHECK(trunc_write(&inst, "/d3", "ZZZ", 3) == 0, "overwrite d3");
217 q_blob_id(root, "/d3", ba, sizeof(ba));
218 CHECK(strcmp(ba, bb) != 0, "overwrite broke the share");
219 CHECK(read_file(&inst, "/d2", buf, sizeof(buf)) == 10
220 && memcmp(buf, DATA, 10) == 0, "d2 unchanged after d3 overwrite");
221 CHECK(read_file(&inst, "/d3", buf, sizeof(buf)) == 3
222 && memcmp(buf, "ZZZ", 3) == 0, "d3 has new content");
223 CHECK(q_refcount(root, bb) == 1, "src refcount back to 1 (got %d)",
224 q_refcount(root, bb));
225
226 /* CONCURRENCY (the plan's critical-correctness item): two threads break the
227 * share on the same source blob at once — each must end with its own private
228 * content, and neither may see the other's bytes. */
229 CHECK(write_file(&inst, "/c1", DATA, 10) == 0, "seed c1");
230 CHECK(D->server_copy(&inst, "/c1", "/c2", &bytes) == NGX_OK, "share c1→c2");
231 {
232 pthread_t t1, t2;
233 struct break_arg a1 = { &inst, "/c1", "11111", 5, -1 };
234 struct break_arg a2 = { &inst, "/c2", "222222", 6, -1 };
235
236 CHECK(pthread_create(&t1, NULL, break_body, &a1) == 0, "spawn t1");
237 CHECK(pthread_create(&t2, NULL, break_body, &a2) == 0, "spawn t2");
238 pthread_join(t1, NULL);
239 pthread_join(t2, NULL);
240 CHECK(a1.rc == 0 && a2.rc == 0, "both concurrent breaks succeeded");
241 }
242 CHECK(read_file(&inst, "/c1", buf, sizeof(buf)) == 5
243 && memcmp(buf, "11111", 5) == 0, "c1 has its own content");
244 CHECK(read_file(&inst, "/c2", buf, sizeof(buf)) == 6
245 && memcmp(buf, "222222", 6) == 0, "c2 has its own content");
246 q_blob_id(root, "/c1", ba, sizeof(ba));
247 q_blob_id(root, "/c2", bb, sizeof(bb));
248 CHECK(strcmp(ba, bb) != 0, "concurrent breaks produced distinct blobs");
249
250 D->cleanup(&inst);
251}

253/* SECURITY-NEG: a forged blobs.content_hash must NOT let differing content
254 * alias — dedup byte-verifies the candidate and rejects the mismatch. */
255void
256test_dedup_forged_hash(void)
257{
258 char root[] = "/tmp/pb_forge.XXXXXX";
259 brix_sd_instance_t inst = {0};
260 brix_sd_pblock_conf_t conf = {0};
261 char bvictim[PBLOCK_BLOB_ID_CAP];
262 char battack[PBLOCK_BLOB_ID_CAP];
263 char hash[64];
264 char db[PATH_MAX], buf[64];
265 sqlite3 *h = NULL;
266 sqlite3_stmt *q = NULL;
267
268 CHECK(mkdtemp(root) != NULL, "mkdtemp");
269 lab_write_sidecar(root, "dedup=1");
270 conf.root = root;
271 conf.busy_timeout_ms = 2000;
272 conf.block_size = 4;
273 inst.driver = D;
274 inst.caps = D->caps;
275 CHECK(D->init(&inst, &conf) == NGX_OK, "forge init");
276
277 /* Victim blob with a genuine hash; attacker file of the SAME size but
278 * DIFFERENT bytes. */
279 CHECK(write_file(&inst, "/victim", "AAAAAAAA", 8) == 0, "seed victim");
280 CHECK(write_file(&inst, "/attack", "BBBBBBBB", 8) == 0, "seed attack");
281 q_blob_id(root, "/victim", bvictim, sizeof(bvictim));
282 q_blob_id(root, "/attack", battack, sizeof(battack));
283 CHECK(strcmp(bvictim, battack) != 0, "different content keeps blobs apart");
284
285 /* Forge the attacker row's hash to equal the victim's — a same-size,
286 * same-block-size candidate the SELECT will now return. */
287 hash[0] = '\0';
288 snprintf(db, sizeof(db), "%s/catalog.db", root);
289 CHECK(sqlite3_open(db, &h) == SQLITE_OK, "forge db open");
290 if (sqlite3_prepare_v2(h,
291 "SELECT content_hash FROM blobs WHERE blob_id = ?1;", -1, &q, NULL)
292 == SQLITE_OK)
293 {
294 sqlite3_bind_text(q, 1, bvictim, -1, SQLITE_STATIC);
295 if (sqlite3_step(q) == SQLITE_ROW) {
296 const unsigned char *b = sqlite3_column_text(q, 0);
297
298 snprintf(hash, sizeof(hash), "%s", b ? (const char *) b : "");
299 }
300 }
301 sqlite3_finalize(q);
302 q = NULL;
303 if (sqlite3_prepare_v2(h,
304 "UPDATE blobs SET content_hash = ?2 WHERE blob_id = ?1;", -1, &q,
305 NULL) == SQLITE_OK)
306 {
307 sqlite3_bind_text(q, 1, battack, -1, SQLITE_STATIC);
308 sqlite3_bind_text(q, 2, hash, -1, SQLITE_STATIC);
309 CHECK(sqlite3_step(q) == SQLITE_DONE, "forge update");
310 }
311 sqlite3_finalize(q);
312 sqlite3_close(h);
313
314 /* Re-publish a THIRD identical-to-victim file: its only forged-hash
315 * candidate is the attacker blob, whose bytes differ — byte-verify must
316 * reject it, so the new file shares the genuine victim blob, never the
317 * attacker's. Content stays correct regardless. */
318 CHECK(write_file(&inst, "/probe", "AAAAAAAA", 8) == 0, "seed probe");
319 CHECK(read_file(&inst, "/probe", buf, sizeof(buf)) == 8
320 && memcmp(buf, "AAAAAAAA", 8) == 0, "probe content honest");
321 {
322 char bprobe[PBLOCK_BLOB_ID_CAP];
323
324 q_blob_id(root, "/probe", bprobe, sizeof(bprobe));
325 CHECK(strcmp(bprobe, battack) != 0, "probe did NOT alias attacker blob");
326 }
327 /* The attacker blob is unchanged — no reference was ever redirected to it. */
328 CHECK(read_file(&inst, "/attack", buf, sizeof(buf)) == 8
329 && memcmp(buf, "BBBBBBBB", 8) == 0, "attacker content untouched");
330
331 D->cleanup(&inst);
332}

365/* F6 snapshots — take/restore via the reserved-namespace control paths, exactly
366 * as the wire reaches them: mkdir /.pblock/snap/<n> takes, mkdir
367 * /.pblock/restore/<n> restores, rmdir /.pblock/snap/<n> drops. snap=1 auto-arms
368 * F10 refs (a snapshot pins shared blobs so a delete-between-take-and-restore
369 * decrements rather than physically removes). Three legs:
370 * SUCCESS — snapshot, delete everything, restore, byte-identical reads.
371 * ERROR — restore refused (EBUSY, not corruption) while a handle is open.
372 * SECURITY-NEG — a hostile snapshot name (SQL/traversal metachars) is rejected;
373 * the name is only ever a bound column, so injection is out. */
374void
375test_snapshot(void)
376{
377 char root[] = "/tmp/pb_snap.XXXXXX";
378 brix_sd_instance_t inst = {0};
379 brix_sd_pblock_conf_t conf = {0};
380 char buf[64];
381
382 CHECK(mkdtemp(root) != NULL, "mkdtemp");
383 lab_write_sidecar(root, "snap=1");
384 conf.root = root;
385 conf.busy_timeout_ms = 2000;
386 conf.block_size = 4;
387 inst.driver = D;
388 inst.caps = D->caps;
389 CHECK(D->init(&inst, &conf) == NGX_OK, "snap init");
390
391 /* SUCCESS: seed two files (b spans >1 block), snapshot, delete everything,
392 * restore, and read back byte-identical content. */
393 CHECK(write_file(&inst, "/a", "alpha", 5) == 0, "seed a");
394 CHECK(write_file(&inst, "/b", "bravodata", 9) == 0, "seed b");
395 CHECK(D->mkdir(&inst, "/.pblock/snap/fix", 0755) == NGX_OK, "take snapshot");
396
397 CHECK(D->unlink(&inst, "/a", 0) == NGX_OK, "delete a");
398 CHECK(D->unlink(&inst, "/b", 0) == NGX_OK, "delete b");
399 errno = 0;
400 CHECK(read_file(&inst, "/a", buf, sizeof(buf)) == -1 && errno == ENOENT,
401 "a gone before restore");
402
403 CHECK(D->mkdir(&inst, "/.pblock/restore/fix", 0755) == NGX_OK, "restore");
404 CHECK(read_file(&inst, "/a", buf, sizeof(buf)) == 5
405 && memcmp(buf, "alpha", 5) == 0, "a restored byte-identical");
406 CHECK(read_file(&inst, "/b", buf, sizeof(buf)) == 9
407 && memcmp(buf, "bravodata", 9) == 0, "b restored byte-identical");
408
409 /* ERROR: a live regular-file handle bumps open_files; restore must refuse
410 * with EBUSY (never swap the namespace out from under an open fd) and leave
411 * the namespace untouched. */
412 {
413 int err = 0;
414 brix_sd_obj_t *o = D->open(&inst, "/a", BRIX_SD_O_READ, 0, &err);
415
416 CHECK(o != NULL, "open handle for EBUSY: %s", strerror(err));
417 errno = 0;
418 CHECK(D->mkdir(&inst, "/.pblock/restore/fix", 0755) == NGX_ERROR
419 && errno == EBUSY, "restore refused while a handle is open");
420 CHECK(read_file(&inst, "/a", buf, sizeof(buf)) == 5
421 && memcmp(buf, "alpha", 5) == 0, "a intact after refused restore");
422 if (o != NULL) { pb_close(o); }
423 }
424
425 /* SECURITY-NEG: a snapshot name carrying SQL/traversal metacharacters is
426 * rejected at the charset gate (EINVAL); the snapshots table is unharmed, so
427 * a subsequent legitimate snapshot still succeeds. */
428 errno = 0;
429 CHECK(D->mkdir(&inst, "/.pblock/snap/x';DROP TABLE snapshots;--", 0755)
430 == NGX_ERROR && errno == EINVAL,
431 "hostile snapshot name rejected");
432 CHECK(D->mkdir(&inst, "/.pblock/snap/second", 0755) == NGX_OK,
433 "legit snapshot still works after the injection attempt");
434 /* drop it again via rmdir on the reserved control path. */
435 CHECK(D->unlink(&inst, "/.pblock/snap/second", 1) == NGX_OK,
436 "drop snapshot via rmdir");
437
438 D->cleanup(&inst);
439}

441/* F11 versioning + trash/undelete — versions=N holds the prior blob on each
442 * overwrite-publish (trimmed to N generations); trash=1 moves an unlink into the
443 * trash (blob held) and mkdir /.pblock/undelete/<path> pops it back. Both build
444 * ON F10 refs (a held blob is a live reference, decremented not freed). Legs:
445 * SUCCESS — three overwrites keep exactly N=2 versions (the oldest is
446 * trimmed and its blob freed); a trashed file undeletes
447 * byte-identical through the reserved control path.
448 * ERROR — undelete of a never-trashed name is ENOENT; undelete over a
449 * live object is EEXIST.
450 * SECURITY-NEG — a hostile undelete path (SQL metachars) is a bound column, so
451 * it can only miss (ENOENT); the trash table is unharmed and a
452 * legitimate undelete still works. */
453void
454test_versioning(void)
455{
456 char root[] = "/tmp/pb_ver.XXXXXX";
457 brix_sd_instance_t inst = {0};
458 brix_sd_pblock_conf_t conf = {0};
459 char a0[PBLOCK_BLOB_ID_CAP], b1[PBLOCK_BLOB_ID_CAP];
460 char c2[PBLOCK_BLOB_ID_CAP], g0[PBLOCK_BLOB_ID_CAP];
461 char buf[64];
462
463 CHECK(mkdtemp(root) != NULL, "mkdtemp");
464 lab_write_sidecar(root, "versions=2&trash=1");
465 conf.root = root;
466 conf.busy_timeout_ms = 2000;
467 conf.block_size = 4;
468 inst.driver = D;
469 inst.caps = D->caps;
470 CHECK(D->init(&inst, &conf) == NGX_OK, "versioning init");
471
472 /* SUCCESS (versions + trim). Each overwrite captures the prior blob as a new
473 * generation; the blob is held (refcount 1), not freed. */
474 CHECK(staged_put(&inst, "/f", "AAAA", 4) == 0, "seed v1");
475 q_blob_id(root, "/f", a0, sizeof(a0));
476 CHECK(staged_put(&inst, "/f", "BBBBBB", 6) == 0, "overwrite v2");
477 q_blob_id(root, "/f", b1, sizeof(b1));
478 CHECK(q_count(root, "SELECT COUNT(*) FROM versions WHERE path='/f'") == 1,
479 "one version after first overwrite");
480 CHECK(q_refcount(root, a0) == 1, "prior blob held by the version (got %d)",
481 q_refcount(root, a0));
482
483 CHECK(staged_put(&inst, "/f", "CCCCCCCC", 8) == 0, "overwrite v3");
484 q_blob_id(root, "/f", c2, sizeof(c2));
485 CHECK(q_count(root, "SELECT COUNT(*) FROM versions WHERE path='/f'") == 2,
486 "two versions after second overwrite");
487
488 /* The fourth publish trims to N=2: the oldest generation (a0) is dropped and
489 * its blob — referenced nowhere else — is freed. */
490 CHECK(staged_put(&inst, "/f", "DD", 2) == 0, "overwrite v4 (trims oldest)");
491 CHECK(q_count(root, "SELECT COUNT(*) FROM versions WHERE path='/f'") == 2,
492 "still exactly N=2 versions after trim");
493 CHECK(q_count(root, "SELECT COUNT(*) FROM versions WHERE blob_id='' ") == 0,
494 "no version row lost its blob reference");
495 CHECK(q_refcount(root, a0) == -1, "trimmed blob freed (got %d)",
496 q_refcount(root, a0));
497 CHECK(q_refcount(root, b1) == 1, "retained version b1 still held");
498 CHECK(q_refcount(root, c2) == 1, "retained version c2 still held");
499 CHECK(read_file(&inst, "/f", buf, sizeof(buf)) == 2
500 && memcmp(buf, "DD", 2) == 0, "live /f is the newest content");
501
502 /* SUCCESS (trash + undelete): unlink moves /g into the trash (blob held);
503 * mkdir /.pblock/undelete/g pops it back byte-identical. */
504 CHECK(staged_put(&inst, "/g", "hello", 5) == 0, "seed /g");
505 q_blob_id(root, "/g", g0, sizeof(g0));
506 CHECK(D->unlink(&inst, "/g", 0) == NGX_OK, "unlink /g to trash");
507 errno = 0;
508 CHECK(read_file(&inst, "/g", buf, sizeof(buf)) == -1 && errno == ENOENT,
509 "/g gone from the namespace");
510 CHECK(q_refcount(root, g0) == 1, "trashed blob held (got %d)",
511 q_refcount(root, g0));
512 CHECK(D->mkdir(&inst, "/.pblock/undelete/g", 0755) == NGX_OK, "undelete /g");
513 CHECK(read_file(&inst, "/g", buf, sizeof(buf)) == 5
514 && memcmp(buf, "hello", 5) == 0, "/g undeleted byte-identical");
515
516 /* ERROR: undelete of a never-trashed name is ENOENT; undelete over the now
