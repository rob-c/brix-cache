1/*
2 * dashboard/config_download.c - authenticated, secret-safe config export.
3 *
4 * Serves GET /xrootd/api/v1/config: the running nginx configuration
5 * (ngx_cycle->conf_file) as a text/plain attachment, with EVERY secret removed.
6 *
7 * SECURITY MODEL (fail-closed — the whole point of this feature):
8 * The on-disk config is read verbatim, then passed through a line redactor
9 * that masks the VALUE of every directive UNLESS the directive name is
10 * recognised as safe. "Recognised safe" = a project brix_* directive that is
11 * not itself a secret, OR a name on a curated stock-nginx allowlist. Anything
12 * else - unknown directives, stock directives that may carry credentials
13 * (proxy_set_header, set, auth_basic_user_file, ...), `include`, secret
14 * directives - has its value replaced with "[redacted]". Because the default
15 * is to redact, a directive we have never heard of CANNOT leak its value: the
16 * only way a value survives is if we explicitly classified its name as safe.
17 *
18 * On top of that fail-closed core:
19 * - an explicit secret denylist + a secret-name substring net always win
20 * (so even a future brix_*_secret is masked before the brix_* "safe"
21 * rule can pass it through);
22 * - `include` targets are never inlined (separate files are not read), so a
23 * secret living in an included file is never reachable here;
24 * - surviving values are additionally scrubbed of embedded URL credentials
25 * (scheme://user:pass@host and ?token=/secret=/sig= query params), the one
26 * place a secret can hide inside an otherwise-safe URL directive.
27 *
28 * The whole file is read into memory and redacted BEFORE any byte is sent; a
29 * read/stat error or an oversize file yields a 5xx with NO body, never a
30 * partial or unredacted dump.
31 *
32 * AUTH: always required. This handler is never reached anonymously - it is a
33 * separate route from the api_handler and does not consult conf->anonymous.
34 */

48/*
49 * Append a directive line to *out with its value masked:
50 * "<indent><name> [redacted]<terminator>"
51 * where terminator preserves a block '{' or statement ';' and a trailing '\r'.
52 * If the directive has no value (e.g. "sendfile;") the line is emitted intact.
53 * `line`/`line_end` exclude the '\n'.
54 */
55static u_char *
56dashboard_emit_masked(u_char *out, const u_char *line, const u_char *line_end,
57 const u_char *name, const u_char *name_end)
58{
59 const u_char *vs = name_end;
60 const u_char *t;
61 ngx_uint_t has_value;
62 ngx_uint_t is_block = 0;
63 ngx_uint_t has_semi = 0;
64 ngx_uint_t has_cr = (line_end > line && line_end[-1] == '\r');
65
66 while (vs < line_end && (*vs == ' ' || *vs == '\t')) { vs++; }
67 for (t = vs; t < line_end; t++) {
68 if (*t == '{') { is_block = 1; }
69 if (*t == ';') { has_semi = 1; }
70 }
71 has_value = (vs < line_end && *vs != '{' && *vs != ';' && *vs != '\r');
72
73 /* indent + name */
74 ngx_memcpy(out, line, (size_t) (name_end - line));
75 out += (name_end - line);
76
77 if (has_value) {
78 *out++ = ' ';
79 ngx_memcpy(out, DASHBOARD_REDACTED, ngx_strlen(DASHBOARD_REDACTED));
80 out += ngx_strlen(DASHBOARD_REDACTED);
81 }
82 if (is_block) { *out++ = ' '; *out++ = '{'; }
83 else if (has_semi) { *out++ = ';'; }
84 if (has_cr) { *out++ = '\r'; }
85 *out++ = '\n';
86 return out;
87}

89/*
90 * Emit one line verbatim (plus '\n'), then scrub embedded URL credentials in
91 * the copied bytes. Used for pass-through lines: blank/comment/brace lines
92 * and directives whose value passed the fail-closed keep test
93 * (belt-and-suspenders — a secret hiding inside a "safe" URL is still masked).
94 * Returns the new write pointer.
95 */
96static u_char *
97dashboard_emit_scrubbed(u_char *w, const u_char *line, const u_char *line_end)
98{
99 u_char *seg = w;
100
101 ngx_memcpy(w, line, (size_t) (line_end - line));
102 w += (line_end - line);
103 w = dashboard_scrub_value_creds(seg, w);
104 *w++ = '\n';
105 return w;
106}

108/* End of the directive-name token starting at q (space/tab/';'/'{'/'#'). */
109static const u_char *
110dashboard_name_token_end(const u_char *q, const u_char *line_end)
111{
112 while (q < line_end && *q != ' ' && *q != '\t'
113 && *q != ';' && *q != '{' && *q != '#') {
114 q++;
115 }
116 return q;
117}

119/*
120 * Redact a single config line [line,line_end) (excluding the '\n') into w:
121 * blank/comment/brace-only lines pass through cred-scrubbed; directives on
122 * the fail-closed allowlist keep their value (also cred-scrubbed); everything
123 * else has its value masked. Returns the new write pointer.
124 */
125static u_char *
126dashboard_redact_line(u_char *w, const u_char *line, const u_char *line_end)
127{
128 const u_char *q = line;
129 const u_char *name, *name_end;
130
131 /* leading whitespace */
132 while (q < line_end && (*q == ' ' || *q == '\t')) { q++; }
133
134 /* blank / comment / brace-only -> pass through, but scrub creds. */
135 if (q >= line_end || *q == '#' || *q == '{' || *q == '}') {
136 return dashboard_emit_scrubbed(w, line, line_end);
137 }
138
139 /* directive name token */
140 name = q;
141 name_end = dashboard_name_token_end(q, line_end);
142
143 if (dashboard_keep_value(name, (size_t) (name_end - name))) {
144 return dashboard_emit_scrubbed(w, line, line_end);
145 }
146 return dashboard_emit_masked(w, line, line_end, name, name_end);
147}

149/*
150 * Redact a whole config buffer. Allocates and fills out_buf / out_len from
151 * r->pool. Returns NGX_OK or NGX_ERROR (alloc failure).
152 *
153 * Output sizing (must be a hard upper bound — a masked line writes a CONSTANT
154 * " [redacted]" regardless of how short the original value was, so growth is
155 * per-LINE, not proportional to input): a masked line emits at most
156 * indent+name (<= line length) + " [redacted]" (11) + " {"/";" (2) + "\r\n" (2)
157 * = line_len + 15. Kept/blank/comment lines never grow (dashboard_scrub_value_
158 * creds is non-growing). So the body is bounded by in_len + 15*nlines; we
159 * reserve 16*nlines + slack.
160 */
161static ngx_int_t
162dashboard_redact_config(ngx_http_request_t *r, const u_char *in, size_t in_len,
163 u_char **out_buf, size_t *out_len)
164{
165 static const char banner[] =
166 "# nginx-xrootd running configuration (secrets redacted)\n"
167 "# Values of directives not on the dashboard safe-allowlist are shown as\n"
168 "# [redacted]; 'include' targets are NOT inlined. Secrets never appear.\n";
169 size_t nlines = 1, k;
170 size_t cap;
171 u_char *out;
172 u_char *w;
173 const u_char *p = in;
174 const u_char *end = in + in_len;
175
176 for (k = 0; k < in_len; k++) {
177 if (in[k] == '\n') { nlines++; }
178 }
179 cap = in_len + nlines * 16 + sizeof(banner) + 256;
180 out = ngx_palloc(r->pool, cap);
181
182 if (out == NULL) {
183 return NGX_ERROR;
184 }
185 w = out;
186 ngx_memcpy(w, banner, sizeof(banner) - 1);
187 w += sizeof(banner) - 1;
188
189 while (p < end) {
190 const u_char *line = p;
191 const u_char *nl = ngx_strlchr((u_char *) p, (u_char *) end, '\n');
192 const u_char *line_end = (nl != NULL) ? nl : end;
193
194 p = (nl != NULL) ? nl + 1 : end;
195
196 /* Defence-in-depth: the cap above is a proven upper bound, but never let
197 * a write cross out+cap regardless. Worst growth for this line is
198 * (line_len) + " [redacted]" (11) + " {"/";" (2) + "\r\n" (2); reserve
199 * generously and fail closed (5xx, no body) rather than overflow. */
200 if ((size_t) ((out + cap) - w) < (size_t) (line_end - line) + 32) {
201 return NGX_ERROR;
202 }
203
204 w = dashboard_redact_line(w, line, line_end);
205 }
206
207 *out_buf = out;
208 *out_len = (size_t) (w - out);
209 return NGX_OK;
210}

212/*
213 * Read the running config file fully into a pool buffer. Returns NGX_OK with
214 * buf / len set, or an NGX_HTTP_* status on failure (file missing, too large,
215 * read error). Closes the fd on every path. No goto: the fd is owned here and
216 * released before each return.
217 */
218static ngx_int_t
219dashboard_read_config_file(ngx_http_request_t *r, u_char **buf, size_t *len)
220{
221 ngx_str_t path = ngx_cycle->conf_file;
222 char cpath[NGX_MAX_PATH];
223 int fd;
224 struct stat st;
225 u_char *b;
226 ssize_t n;
227 size_t got;
228
229 if (path.len == 0
230 || brix_str_cbuf(cpath, sizeof(cpath), &path) == NULL) {
231 return NGX_HTTP_INTERNAL_SERVER_ERROR;
232 }
233
234 fd = open(cpath, O_RDONLY | O_CLOEXEC);
235 if (fd < 0) {
236 return NGX_HTTP_INTERNAL_SERVER_ERROR;
237 }
238 if (fstat(fd, &st) != 0 || !S_ISREG(st.st_mode)) {
239 close(fd);
240 return NGX_HTTP_INTERNAL_SERVER_ERROR;
241 }
242 if (st.st_size <= 0 || st.st_size > DASHBOARD_CONFIG_MAX_BYTES) {
243 close(fd);
244 return NGX_HTTP_INSUFFICIENT_STORAGE; /* 507: fail-safe, no body */
245 }
246
247 b = ngx_palloc(r->pool, (size_t) st.st_size);
248 if (b == NULL) {
249 close(fd);
250 return NGX_HTTP_INTERNAL_SERVER_ERROR;
251 }
252
253 got = 0;
254 while (got < (size_t) st.st_size) {
255 n = read(fd, b + got, (size_t) st.st_size - got);
256 if (n < 0) {
257 if (errno == EINTR) { continue; }
258 close(fd);
259 return NGX_HTTP_INTERNAL_SERVER_ERROR;
260 }
261 if (n == 0) { break; } /* truncated since fstat — use what we have */
262 got += (size_t) n;
263 }
264 close(fd);
265
266 *buf = b;
267 *len = got;
268 return NGX_OK;
269}

271/* Emit the redacted config as a text/plain download. */
272static ngx_int_t
273dashboard_send_config(ngx_http_request_t *r, u_char *body, size_t len)
274{
275 ngx_buf_t *b;
276 ngx_chain_t out;
277 ngx_table_elt_t *h;
278 ngx_int_t rc;
279
280 b = ngx_pcalloc(r->pool, sizeof(*b));
281 if (b == NULL) { return NGX_HTTP_INTERNAL_SERVER_ERROR; }
282 b->pos = b->start = body;
283 b->last = b->end = body + len;
284 b->memory = 1;
285 b->last_buf = 1;
286
287 r->headers_out.status = NGX_HTTP_OK;
288 r->headers_out.content_length_n = (off_t) len;
289 r->headers_out.content_type =
290 (ngx_str_t) ngx_string("text/plain; charset=utf-8");
291 r->headers_out.content_type_len = r->headers_out.content_type.len;
292
293 h = ngx_list_push(&r->headers_out.headers);
294 if (h != NULL) {
295 h->hash = 1;
296 ngx_str_set(&h->key, "Content-Disposition");
297 ngx_str_set(&h->value, "attachment; filename=\"nginx-xrootd.conf\"");
298 }
299 h = ngx_list_push(&r->headers_out.headers);
300 if (h != NULL) {
301 h->hash = 1;
302 ngx_str_set(&h->key, "Cache-Control");
303 ngx_str_set(&h->value, "no-store");
304 }
305
306 rc = ngx_http_send_header(r);
307 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) { return rc; }
308
309 out.buf = b;
310 out.next = NULL;
311 return ngx_http_output_filter(r, &out);
312}
