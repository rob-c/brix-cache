18/*
19 * brix_make_vfs_body — kXR_stat(kXR_vfs) / `xrdfs statvfs` response body.
20 *
21 * The reference format (XrdCl StatInfoVFS::ParseServerResponse) is SIX
22 * space-separated integers, each a bare number:
23 * "<nodesRW> <freeRW_MB> <utilRW%> <nodesStaging> <freeStaging_MB> <utilStaging%>"
24 * (We previously emitted the 4-field "id size flags mtime" stat line here, which
25 * the stock client rejects as "Invalid response".) Free space comes from
26 * statvfs(2) on the export root; staging is reported as 0 (no tape staging tier).
27 */
28static void
29brix_make_vfs_body(ngx_stream_brix_srv_conf_t *conf, char *out, size_t outsz)
30{
31 struct statvfs vfs;
32 const char *root = conf->common.root_canon[0]
33 ? conf->common.root_canon : "/";
34 long long free_mb = 0, total_mb = 0;
35 int util = 0;
36 int nrw = conf->common.allow_write ? 1 : 0;
37
38 if (statvfs(root, &vfs) == 0) {
39 unsigned long bs = vfs.f_frsize ? vfs.f_frsize : vfs.f_bsize;
40 free_mb = (long long) ((double) vfs.f_bavail * (double) bs / 1048576.0);
41 total_mb = (long long) ((double) vfs.f_blocks * (double) bs / 1048576.0);
42 if (total_mb > 0) {
43 util = (int) (100.0 * (double) (total_mb - free_mb)
44 / (double) total_mb);
45 }
46 }
47
48 snprintf(out, outsz, "%d %lld %d 0 0 0", nrw, free_mb, util);
49}

58/*
59 * brix_vfs_to_struct_stat — project a VFS stat result back into the struct
60 * stat fields the kXR_stat response builder reads: the unique id (st_ino<<32 |
61 * st_dev), st_size, the permission triplet (st_mode/st_uid/st_gid →
62 * readable/writable flags), st_mtime, and st_blocks (statvfs-style body). The
63 * VFS is the only thing that touched the namespace; this is a pure field copy.
64 */
65void
66brix_vfs_to_struct_stat(const brix_vfs_stat_t *v, struct stat *st)
67{
68 ngx_memzero(st, sizeof(*st));
69 st->st_mode = (mode_t) v->mode;
70 st->st_size = v->size;
71 st->st_mtime = v->mtime;
72 st->st_ctime = v->ctime;
73 st->st_ino = v->ino;
74 st->st_dev = v->dev;
75 st->st_uid = v->uid;
76 st->st_gid = v->gid;
77 st->st_blocks = v->blocks;
78 st->st_nlink = 1;
79}

81/* Return kXR_cachersp if reqpath (client's clean path) exists in cache_root. */
82int
83brix_cache_path_flag(const ngx_stream_brix_srv_conf_t *conf, const char *reqpath)
84{
85 char cache_path[PATH_MAX];
86 struct stat cst;
87 int n;
88
89 if (!conf->cache || conf->cache_root.len == 0 || reqpath == NULL) {
90 return 0;
91 }
92
93 n = snprintf(cache_path, sizeof(cache_path), "%s%s",
94 (char *) conf->cache_root.data, reqpath);
95 if (n < 0 || (size_t) n >= sizeof(cache_path)) {
96 return 0;
97 }
98
99 /* vfs-seam-allow: separate storage domain. cache_path is under the
100 * server-managed cache root (svc-owned, distinct from the export root); the
101 * cachersp existence probe runs as the worker, not via the export VFS. */
102 return (stat(cache_path, &cst) == 0 && S_ISREG(cst.st_mode)) /* vfs-seam-allow: separate cache-root domain */
103 ? kXR_cachersp : 0;
104}
