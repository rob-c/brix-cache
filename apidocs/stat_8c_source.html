20/*
21 * brix_make_vfs_body — kXR_stat(kXR_vfs) / `xrdfs statvfs` response body.
22 *
23 * The reference format (XrdCl StatInfoVFS::ParseServerResponse) is SIX
24 * space-separated integers, each a bare number:
25 * "<nodesRW> <freeRW_MB> <utilRW%> <nodesStaging> <freeStaging_MB> <utilStaging%>"
26 * (We previously emitted the 4-field "id size flags mtime" stat line here, which
27 * the stock client rejects as "Invalid response".) Free space comes from
28 * statvfs(2) on the export root; staging is reported as 0 (no tape staging tier).
29 */
30static void
31brix_make_vfs_body(ngx_stream_brix_srv_conf_t *conf, char *out, size_t outsz)
32{
33 struct statvfs vfs;
34 const char *root = conf->common.root_canon[0]
35 ? conf->common.root_canon : "/";
36 long long free_mb = 0, total_mb = 0;
37 int util = 0;
38 int nrw = conf->common.allow_write ? 1 : 0;
39
40 /* Phase-83 F5: a backend implementing the optional `space` slot reports its
41 * own quota-aware logical space (pblock: quota total minus rollup usage);
42 * only exports without a space model fall back to raw statvfs(2). */
43 {
44 brix_sd_instance_t *inst = brix_vfs_backend_resolve(root, ngx_cycle->log);
45
46 if (inst != NULL && inst->driver->space != NULL) {
47 brix_sd_space_t sp;
48
49 if (inst->driver->space(inst, &sp) == NGX_OK) {
50 free_mb = (long long) (sp.free_bytes / 1048576u);
51 total_mb = (long long) (sp.total_bytes / 1048576u);
52 if (total_mb > 0) {
53 util = (int) (100.0 * (double) (total_mb - free_mb)
54 / (double) total_mb);
55 }
56 snprintf(out, outsz, "%d %lld %d 0 0 0", nrw, free_mb, util);
57 return;
58 }
59 }
60 }
61
62 if (statvfs(root, &vfs) == 0) {
63 unsigned long bs = vfs.f_frsize ? vfs.f_frsize : vfs.f_bsize;
64 free_mb = (long long) ((double) vfs.f_bavail * (double) bs / 1048576.0);
65 total_mb = (long long) ((double) vfs.f_blocks * (double) bs / 1048576.0);
66 if (total_mb > 0) {
67 util = (int) (100.0 * (double) (total_mb - free_mb)
68 / (double) total_mb);
69 }
70 }
71
72 snprintf(out, outsz, "%d %lld %d 0 0 0", nrw, free_mb, util);
73}

82/*
83 * brix_vfs_to_struct_stat — project a VFS stat result back into the struct
84 * stat fields the kXR_stat response builder reads: the unique id (st_ino<<32 |
85 * st_dev), st_size, the permission triplet (st_mode/st_uid/st_gid →
86 * readable/writable flags), st_mtime, and st_blocks (statvfs-style body). The
87 * VFS is the only thing that touched the namespace; this is a pure field copy.
88 */
89void
90brix_vfs_to_struct_stat(const brix_vfs_stat_t *v, struct stat *st)
91{
92 ngx_memzero(st, sizeof(*st));
93 st->st_mode = (mode_t) v->mode;
94 st->st_size = v->size;
95 st->st_mtime = v->mtime;
96 st->st_ctime = v->ctime;
97 st->st_ino = v->ino;
98 st->st_dev = v->dev;
99 st->st_uid = v->uid;
100 st->st_gid = v->gid;
101 st->st_blocks = v->blocks;
102 st->st_nlink = 1;
103}

105/* Return kXR_cachersp if reqpath (client's clean path) exists in cache_root. */
106int
107brix_cache_path_flag(const ngx_stream_brix_srv_conf_t *conf, const char *reqpath)
108{
109 char cache_path[PATH_MAX];
110 struct stat cst;
111 int n;
112
113 if (!conf->cache || conf->cache_root.len == 0 || reqpath == NULL) {
114 return 0;
115 }
116
117 n = snprintf(cache_path, sizeof(cache_path), "%s%s",
118 (char *) conf->cache_root.data, reqpath);
119 if (n < 0 || (size_t) n >= sizeof(cache_path)) {
120 return 0;
121 }
122
123 /* vfs-seam-allow: separate storage domain. cache_path is under the
124 * server-managed cache root (svc-owned, distinct from the export root); the
125 * cachersp existence probe runs as the worker, not via the export VFS. */
126 return (stat(cache_path, &cst) == 0 && S_ISREG(cst.st_mode)) /* vfs-seam-allow: separate cache-root domain */
127 ? kXR_cachersp : 0;
128}

172/*
173 * stat_manager_route — redirector-side answers for a path-mode stat.
174 *
175 * A redirector must answer stat without local storage — stock and go-hep
176 * clients stat a path before they open it. In precedence order: a static
177 * manager_map prefix redirects outright (mirrors the open/locate paths); in
178 * manager mode the §6 CNS inventory can answer size/mtime directly (a true
179 * global-namespace stat), tried/triedrc exhaustion answers not-found to stop
180 * redirect loops, the registry redirects to a live data server (tolerating a
181 * just-dropped heartbeat, like open), and a registry miss is forwarded to
182 * the CMS parent, parking the session (NGX_AGAIN). Returns 0 when the
183 * request was answered (*rc holds the value to return, possibly NGX_AGAIN)
184 * and 1 when the caller should continue with a local stat.
185 */
186static int
187stat_manager_route(brix_ctx_t *ctx, ngx_connection_t *c,
188 ngx_stream_brix_srv_conf_t *conf, const char *reqpath, ngx_int_t *rc)
189{
190 /* Static manager_map: an explicit prefix→backend redirect (mirrors the
191 * open/locate paths) so a static-map redirector also serves stat — stock
192 * and go-hep clients stat a path before they open it, and without this a
193 * map-only redirector answered stat locally (IOError, no root). */
194 if (conf->manager_map != NULL) {
195 const brix_manager_map_t *m =
196 brix_find_manager_map(reqpath, conf->manager_map);
197 if (m != NULL) {
198 brix_log_access(ctx, c, "STAT", reqpath, "manager_map",
199 1, kXR_ok, NULL, 0);
200 BRIX_OP_OK(ctx, BRIX_OP_STAT);
201 *rc = brix_send_redirect(ctx, c, (const char *) m->host.data,
202 m->port);
203 return 0;
204 }
205 }
206
207 /* Manager mode: redirect to a registered data server. */
208 if (!conf->manager_mode) {
209 return 1;
210 }
211
212 /* §6 CNS: if the cluster name space inventory has this path, answer
213 * stat directly (size/mtime) instead of redirecting — a true global
214 * namespace stat at the redirector. */
215 if (conf->cns_mode == BRIX_CNS_COLLECT) {
216 struct stat cst;
217 if (brix_cns_stat(reqpath, &cst) == NGX_OK) {
218 char cbody[256];
219 brix_make_stat_body(&cst, 0, 0, cbody, sizeof(cbody));
220 brix_log_access(ctx, c, "STAT", reqpath, "cns",
221 1, 0, NULL, 0);
222 BRIX_OP_OK(ctx, BRIX_OP_STAT);
223 *rc = brix_send_ok(ctx, c, cbody,
224 (uint32_t) (strlen(cbody) + 1));
225 return 0;
226 }
227 }
228
229 /* tried/triedrc: if the client has already visited every server
230 * that holds this path and they returned enoent, stop redirecting
231 * and answer not-found — otherwise the client redirect-loops. */
232 if (brix_manager_tried_exhausted(ctx->recv.payload, ctx->recv.cur_dlen,
233 reqpath)) {
234 BRIX_BAIL_ERR(ctx, c, BRIX_OP_STAT, "STAT", reqpath, "-",
235 kXR_NotFound, "file not found on any data server", rc);
236 }
237
238 /* Like open: tolerate a server whose CMS heartbeat just dropped (it
239 * is almost certainly still serving) rather than a false NotFound. */
240 {
241 char redir_host[256];
242 uint16_t redir_port;
243
244 if (brix_srv_select_or_blacklisted(reqpath, 0, redir_host,
245 sizeof(redir_host), &redir_port)) {
246 brix_log_access(ctx, c, "STAT", reqpath, "registry",
247 1, kXR_ok, NULL, 0);
248 BRIX_OP_OK(ctx, BRIX_OP_STAT);
249 *rc = brix_send_redirect(ctx, c, redir_host, redir_port);
250 return 0;
251 }
252 }
253
254 /* Registry miss — ask CMS parent if configured. */
255 if (conf->cms.ctx != NULL) {
256 uint32_t streamid;
257
258 streamid = ngx_brix_cms_next_streamid(conf->cms.ctx);
259 if (brix_pending_insert(streamid, ngx_pid, c->fd,
260 c->number,
261 ctx->recv.cur_streamid,
262 conf->cms.locate_timeout) == NGX_OK)
263 {
264 ctx->cms_wait_streamid = streamid;
265 ctx->state = XRD_ST_WAITING_CMS;
266 ngx_add_timer(c->read, conf->cms.locate_timeout);
267 if (ngx_brix_cms_send_locate(conf->cms.ctx, streamid,
268 reqpath) == NGX_OK)
269 {
270 *rc = NGX_AGAIN;
271 return 0;
272 }
273 ngx_del_timer(c->read);
274 ctx->state = XRD_ST_REQ_HEADER;
275 brix_pending_remove(streamid, ngx_pid);
276 }
277 }
278
279 return 1;
280}

282/*
283 * stat_symlink_follow_fallback — confined follow of a host-absolute symlink.
284 *
285 * Follow fallback for an in-export symlink with a host-ABSOLUTE target:
286 * RESOLVE_IN_ROOT chroots the absolute target and lands on ENOENT, where
287 * stock follows it on the real fs. Match stock, but CONFINE via realpath —
288 * accept only when the canonical target is within the export root (an
289 * escaping link is rejected). Read-only, so the realpath/stat TOCTOU window
290 * is benign. Returns 0 with *tgt->st filled from the confirmed in-root
291 * target, or -1 (errno describes the last failing probe) to keep the
292 * original miss.
293 */
294static int
295stat_symlink_follow_fallback(brix_ctx_t *ctx, ngx_connection_t *c,
296 ngx_stream_brix_srv_conf_t *conf, stat_target_t *tgt)
297{
298 char real[PATH_MAX];
299 size_t rl = ngx_strlen(conf->common.root_canon);
300 brix_vfs_ctx_t rvctx;
301 brix_vfs_stat_t rvst;
302
303 if (realpath(tgt->full_path, real) == NULL
304 || ngx_strncmp(real, conf->common.root_canon, rl) != 0
305 || (real[rl] != '/' && real[rl] != '\0'))
306 {
307 return -1;
308 }
309
310 /* The canonical target is confirmed within the export root;
311 * read its metadata through the VFS (the symlink chain is
312 * already resolved, so a non-follow probe is exact). */
313 stat_vfs_ctx_prepare(ctx, c, conf, real, &rvctx);
314 if (brix_vfs_probe(&rvctx, 0 /* follow */, &rvst) != NGX_OK) {
315 return -1;
316 }
317
318 brix_vfs_to_struct_stat(&rvst, tgt->st);
319 return 0;
320}

322/*
323 * stat_vfs_query — path-mode metadata through the VFS seam.
324 *
325 * kXR_statNoFollow (vendor): lstat the final component so a symlink reports
326 * as itself (kXR_other + target-length size) for FUSE getattr; default
327 * follows symlinks exactly as before. Both go through the VFS seam
328 * (impersonation-aware, RESOLVE_IN_ROOT-confined); the result is projected
329 * back into struct stat for the response code. An ENOENT on a followable
330 * path gets one confined symlink-follow fallback so a host-absolute in-root
331 * link still stats like stock. Returns 0 with *tgt->st filled, or -1 with
332 * errno describing the failure for the caller's error mapping.
333 */
334static int
335stat_vfs_query(brix_ctx_t *ctx, ngx_connection_t *c,
336 ngx_stream_brix_srv_conf_t *conf, stat_target_t *tgt)
337{
338 brix_vfs_ctx_t vctx;
339 brix_vfs_stat_t vst;
340 int src;
341
342 stat_vfs_ctx_prepare(ctx, c, conf, tgt->full_path, &vctx);
343 src = (((tgt->req->options & kXR_statNoFollow)
344 ? brix_vfs_stat(&vctx, &vst)
345 : brix_vfs_statf(&vctx, &vst)) == NGX_OK) ? 0 : -1;
346 if (src == 0) {
347 brix_vfs_to_struct_stat(&vst, tgt->st);
348 }
349
350 if (src != 0 && errno == ENOENT
351 && !(tgt->req->options & kXR_statNoFollow)
352 && conf->common.root_canon[0] != '\0')
353 {
354 src = stat_symlink_follow_fallback(ctx, c, conf, tgt);
355 }
356
357 return src;
358}

390/*
391 * stat_query_path — path-mode stat: resolve, authorize, query metadata.
392 *
393 * dlen > 0 means the payload names a path to resolve and stat(2). Extract
394 * and sanity-check the client path (".." rejected outright, internal
395 * sidecars reported absent), give redirector routing first refusal, then
396 * confine the path beneath the export root, run the auth gate, and read the
397 * metadata through the VFS seam. On success the cache-residency and
398 * nearline flag bits land in tgt->extra_flags. Returns 1 to continue with
399 * the common response tail; 0 when a response was already arranged (*rc
400 * carries the handler's return value).
401 */
402static int
403stat_query_path(brix_ctx_t *ctx, ngx_connection_t *c,
404 ngx_stream_brix_srv_conf_t *conf, stat_target_t *tgt, ngx_int_t *rc)
405{
406 if (!brix_extract_path(c->log, ctx->recv.payload, ctx->recv.cur_dlen,
407 tgt->reqpath_buf, tgt->reqpath_size, 1)) {
408 BRIX_BAIL_ERR(ctx, c, BRIX_OP_STAT, "STAT", "-", "-",
409 kXR_ArgInvalid, "invalid path payload", rc);
410 }
411 tgt->reqpath = tgt->reqpath_buf;
412
413 /* Reject any ".." component outright (the reference does not normalize
414 * ".."). This op resolves through the kernel RESOLVE_BENEATH, which
415 * would silently collapse an in-tree "..", so the guard is explicit. */
416 if (brix_reject_dotdot_path(ctx, c, BRIX_OP_STAT, "STAT", tgt->reqpath)) {
417 *rc = ctx->write_rc;
418 return 0;
419 }
420
421 /* Internal artifacts (sidecars, upload temps) are invisible → report as
422 * absent, never leaking their size/mtime/existence. */
423 if (brix_is_internal_name(tgt->reqpath)) {
424 BRIX_BAIL_ERR(ctx, c, BRIX_OP_STAT, "STAT", tgt->reqpath, "-",
425 kXR_NotFound, "file not found", rc);
426 }
427
428 if (!stat_manager_route(ctx, c, conf, tgt->reqpath, rc)) {
429 return 0;
430 }
431
432 brix_beneath_full_path(conf->common.root_canon, tgt->reqpath,
433 tgt->full_path, tgt->full_path_size);
434
435 if (brix_auth_gate(ctx, c, BRIX_OP_STAT, "STAT",
436 tgt->reqpath, tgt->full_path, conf,
437 BRIX_AUTH_LOOKUP, 0) != NGX_OK) {
438 *rc = ctx->write_rc;
439 return 0;
440 }
441
442 if (stat_vfs_query(ctx, c, conf, tgt) != 0) {
443 /* E-4: a missing path from a principal already over its miss budget is
444 * answered with kXR_wait instead of NotFound, throttling a stat-harvest
445 * loop while a legitimate client (which rarely misses) sails through. */
446 if (errno == ENOENT && conf->negcache.threshold > 0) {
447 unsigned w = brix_negcache_note_miss(ctx, conf->negcache.threshold,
448 conf->negcache.window_ms,
449 conf->negcache.backoff_s);
450 if (w > 0) {
451 ngx_log_error(NGX_LOG_INFO, c->log, 0,
452 "xrootd negcache: kXR_wait %ud for STAT miss %s",
453 w, tgt->reqpath);
454 BRIX_OP_ERR(ctx, BRIX_OP_STAT);
455 *rc = brix_send_wait(ctx, c, w);
456 return 0;
457 }
458 }
459 BRIX_BAIL_ERR(ctx, c, BRIX_OP_STAT, "STAT", tgt->reqpath,
460 "-", brix_kxr_from_errno(errno),
461 strerror(errno), rc);
462 }
463
464 tgt->extra_flags = brix_cache_path_flag(conf, tgt->reqpath);
465 tgt->extra_flags |= stat_residency_flags(c, conf, tgt->full_path);
466
467 return 1;
468}

470/*
471 * stat_query_handle — handle-based stat: fhandle[0] is our slot index.
472 *
473 * The cached path is only for logging; the real metadata comes from fstat().
474 * Two handle shapes need a size correction: a ZIP member's fd is the ARCHIVE
475 * (take the archive metadata, override size with the member's UNCOMPRESSED
476 * length) and a driver-backed handle's bare fd is only block 0 (ask the
477 * storage driver's worker-safe fstat slot for the catalog-backed metadata).
478 * Returns 1 to continue with the common response tail; 0 when an error
479 * response was sent (*rc carries the handler's return value).
480 */
481static int
482stat_query_handle(brix_ctx_t *ctx, ngx_connection_t *c,
483 stat_target_t *tgt, ngx_int_t *rc)
484{
485 int idx = (int)(unsigned char) tgt->req->fhandle[0];
486 struct stat *st = tgt->st;
487
488 if (!brix_validate_file_handle(ctx, c, idx, "STAT",
489 BRIX_OP_STAT, rc)) {
490 return 0;
491 }
492
493 tgt->full_path[0] = '\0';
494 ngx_cpystrn((u_char *) tgt->full_path,
495 (u_char *) (ctx->files[idx].path != NULL
496 ? ctx->files[idx].path : "-"),
497 tgt->full_path_size);
498
499 if (ctx->files[idx].zip_mode) {
500 /*
501 * ZIP member (phase-57 W2): the handle's fd is the ARCHIVE, so a
502 * plain fstat would report the archive's size. Take the archive's
503 * metadata (mode/mtime/ino) but override the size with the member's
504 * UNCOMPRESSED length (cached_size) so the client sees the logical
505 * file size.
506 */
507 if (fstat(ctx->files[idx].fd, st) != 0) {
508 BRIX_BAIL_ERR(ctx, c, BRIX_OP_STAT, "STAT", tgt->full_path, "-",
509 kXR_IOError, strerror(errno), rc);
510 }
511 st->st_size = (off_t) ctx->files[idx].cached_size;
512 } else if (ctx->files[idx].sd_obj.driver != NULL
513 && ctx->files[idx].sd_obj.driver
514 != brix_sd_default_driver()) {
515 /*
516 * Driver-backed handle (e.g. pblock): the bare fd is only block 0,
517 * so a plain fstat would report the block size, not the logical
518 * object size. Ask the storage driver for the object's
519 * (catalog-backed) metadata via its worker-safe fstat slot.
520 */
521 brix_sd_stat_t sdst;
522
523 if (ctx->files[idx].sd_obj.driver->fstat == NULL
524 || ctx->files[idx].sd_obj.driver->fstat(
525 &ctx->files[idx].sd_obj, &sdst) != NGX_OK)
526 {
527 BRIX_BAIL_ERR(ctx, c, BRIX_OP_STAT, "STAT", tgt->full_path,
528 "-", kXR_IOError, strerror(errno), rc);
529 }
530 ngx_memzero(st, sizeof(*st));
531 st->st_size = sdst.size;
532 st->st_mtime = sdst.mtime;
533 st->st_ctime = sdst.ctime;
534 st->st_ino = sdst.ino;
535 st->st_nlink = 1;
536 st->st_mode = sdst.mode ? (mode_t) sdst.mode
537 : (sdst.is_dir ? (S_IFDIR | 0755) : (S_IFREG | 0644));
538 } else if (fstat(ctx->files[idx].fd, st) != 0) {
539 BRIX_BAIL_ERR(ctx, c, BRIX_OP_STAT, "STAT", tgt->full_path, "-",
540 kXR_IOError, strerror(errno), rc);
541 }
542
543 tgt->extra_flags = ctx->files[idx].from_cache ? kXR_cachersp : 0;
544
545 return 1;
546}

548/*
549 * brix_handle_stat — kXR_stat entry: dual-mode like upstream XRootD.
550 *
551 * - dlen > 0 means the payload names a path to resolve and stat(2)
552 * - dlen == 0 means the opaque handle identifies an already-open fd
553 *
554 * The logging path and the syscall target are deliberately separated in the
555 * handle case: logs use the cached canonical path, while fstat() uses the fd.
556 * Both query helpers fill one stat_target_t; the shared tail encodes it and
557 * sends the reply.
558 */
559ngx_int_t brix_handle_stat(brix_ctx_t *ctx, ngx_connection_t *c, ngx_stream_brix_srv_conf_t *conf)
560{
561 xrdw_stat_req_t req;
562 struct stat st;
563 char full_path[PATH_MAX];
564 char reqpath_buf[BRIX_MAX_PATH + 1];
565 char body[256];
566 ngx_flag_t is_vfs;
567 ngx_int_t rc;
568 stat_target_t tgt;
569
570 xrdw_stat_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
571 is_vfs = (req.options & kXR_vfs) ? 1 : 0;
572
573 ngx_memzero(&tgt, sizeof(tgt));
574 tgt.req = &req;
575 tgt.reqpath_buf = reqpath_buf;
576 tgt.reqpath_size = sizeof(reqpath_buf);
577 tgt.full_path = full_path;
578 tgt.full_path_size = sizeof(full_path);
579 tgt.st = &st;
580
581 if (ctx->recv.cur_dlen > 0 && ctx->recv.payload != NULL) {
582 if (!stat_query_path(ctx, c, conf, &tgt, &rc)) {
583 return rc;
584 }
585 } else {
586 if (!stat_query_handle(ctx, c, &tgt, &rc)) {
587 return rc;
588 }
589 }
590
591 /* Convert into the exact ASCII body the client expects. statvfs has its own
592 * 6-field RW/staging-space format (xrdfs statvfs); a plain stat is the
593 * 4-field "id size flags mtime" line. */
594 if (is_vfs) {
595 brix_make_vfs_body(conf, body, sizeof(body));
596 } else {
597 brix_make_stat_body(&st, 0, tgt.extra_flags, body, sizeof(body));
598 }
599
600 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
601 "brix: kXR_stat ok: %s", body);
602
603 brix_log_access(ctx, c, "STAT",
604 (tgt.reqpath && tgt.reqpath[0]) ? tgt.reqpath : full_path,
605 is_vfs ? "vfs" : "-",
606 1, 0, NULL, 0);
607 BRIX_OP_OK(ctx, BRIX_OP_STAT);
608
609 return brix_send_ok(ctx, c, body, (uint32_t)(strlen(body) + 1));
610}
