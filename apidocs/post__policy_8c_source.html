8/*
9 * WHAT: Convert a proleptic-Gregorian civil date (y, m, d) to a day count
10 * relative to the Unix epoch (1970-01-01 == 0).
11 * WHY: Policy expiry must be compared against ngx_time() without depending on
12 * the host timezone, so we compute a UTC epoch directly rather than via
13 * mktime() (which is local-time and not async-signal-safe).
14 * HOW: Howard Hinnant's well-known days_from_civil algorithm. It shifts the
15 * year so March is month 1 (Feb's leap day lands at year-end), groups
16 * years into 400-year "eras" (each exactly 146097 days), then offsets by
17 * 719468 to re-anchor era day 0 onto the Unix epoch. Constants are exact,
18 * not approximations. Reference: howardhinnant.github.io/date_algorithms.html
19 * Returns signed days since 1970-01-01 (negative for pre-epoch dates).
20 */
21int64_t
22s3_post_days_from_civil(int y, unsigned m, unsigned d)
23{
24 int64_t era;
25 unsigned yoe, doy, doe;
26 int mp;
27
28 y -= m <= 2; /* Jan/Feb count to prior year */
29 era = (y >= 0 ? y : y - 399) / 400; /* 400-year era index */
30 yoe = (unsigned) (y - era * 400); /* year-of-era [0, 399] */
31 mp = (int) m + (m > 2 ? -3 : 9); /* month, March=0 .. Feb=11 */
32 doy = (153 * (unsigned) mp + 2) / 5 + d - 1;/* day-of-year, March 1 == 0 */
33 doe = yoe * 365 + yoe / 4 - yoe / 100 + doy;/* day-of-era [0, 146096] */
34
35 return era * 146097 + (int64_t) doe - 719468;
36}

39/*
40 * WHAT: Parse an ISO 8601 UTC timestamp "YYYY-MM-DDTHH:MM:SSZ" to a time_t.
41 * WHY: POST policy "expiration" is given in this exact UTC form; we need it as
42 * an epoch seconds value to compare against ngx_time(). HOW: strict
43 * sscanf with field widths, range-check each component (se <= 60 allows a
44 * leap second), then combine the civil-day count with the time-of-day.
45 * Returns NGX_OK with *out set, or NGX_ERROR on a malformed/out-of-range value.
46 */
47ngx_int_t
48s3_post_parse_iso8601(const char *s, time_t *out)
49{
50 int y, mo, d, h, mi, se;
51
52 if (sscanf(s, "%4d-%2d-%2dT%2d:%2d:%2dZ",
53 &y, &mo, &d, &h, &mi, &se) != 6)
54 {
55 return NGX_ERROR;
56 }
57
58 if (y < 1970 || mo < 1 || mo > 12 || d < 1 || d > 31
59 || h < 0 || h > 23 || mi < 0 || mi > 59 || se < 0 || se > 60)
60 {
61 return NGX_ERROR;
62 }
63
64 /* epoch = (whole days since 1970) * 86400 + seconds within the day. */
65 *out = (time_t) (s3_post_days_from_civil(y, (unsigned) mo, (unsigned) d)
66 * 86400 + h * 3600 + mi * 60 + se);
67 return NGX_OK;
68}

85/*
86 * WHAT: Evaluate the object-form of a POST-policy condition
87 * {"field":"value", ...} — implicit "equals" for every key/value pair.
88 * WHY: The AWS object shape means every listed field must match exactly; the
89 * special "bucket" key is not a form field but the configured bucket name.
90 * HOW: Iterate each member, reject non-string values as malformed, compare
91 * "bucket" against cf->bucket and all other keys against the form value.
92 * Returns NGX_OK if every pair matches, NGX_DECLINED on the first mismatch,
93 * NGX_ERROR if a value is not a string (malformed condition).
94 */
95static ngx_int_t
96s3_post_cond_object(ngx_http_s3_loc_conf_t *cf, const s3_post_form_t *form,
97 json_t *cond)
98{
99 const char *key;
100 json_t *value;
101
102 json_object_foreach(cond, key, value) {
103 const char *expected;
104 const char *actual;
105
106 if (!json_is_string(value)) {
107 return NGX_ERROR;
108 }
109
110 expected = json_string_value(value);
111 /* "bucket" is not a form field — match the configured bucket name. */
112 if (strcmp(key, "bucket") == 0) {
113 if (cf->bucket.len != strlen(expected)
114 || ngx_strncmp(cf->bucket.data, (u_char *) expected,
115 cf->bucket.len) != 0)
116 {
117 return NGX_DECLINED;
118 }
119 continue;
120 }
121
122 actual = s3_post_field_value(form, key);
123 if (s3_post_check_field_eq(actual, expected) != NGX_OK) {
124 return NGX_DECLINED;
125 }
126 }
127 return NGX_OK;
128}

131/*
132 * WHAT: Evaluate the content-length-range array condition
133 * ["content-length-range", min, max] against the uploaded file size.
134 * WHY: This is the only array operator whose arguments are integers; it bounds
135 * how large the uploaded body may be.
136 * HOW: Require exactly three integer-typed elements, then range-check
137 * form->file_len inclusively.
138 * Returns NGX_OK if the size is within [min, max], NGX_DECLINED if outside,
139 * NGX_ERROR if the argument shapes are malformed.
140 */
141static ngx_int_t
142s3_post_cond_length_range(const s3_post_form_t *form, json_t *cond)
143{
144 json_int_t minv, maxv;
145
146 if (json_array_size(cond) != 3
147 || !json_is_integer(json_array_get(cond, 1))
148 || !json_is_integer(json_array_get(cond, 2)))
149 {
150 return NGX_ERROR;
151 }
152
153 minv = json_integer_value(json_array_get(cond, 1));
154 maxv = json_integer_value(json_array_get(cond, 2));
155 if ((json_int_t) form->file_len < minv
156 || (json_int_t) form->file_len > maxv)
157 {
158 return NGX_DECLINED;
159 }
160 return NGX_OK;
161}

164/*
165 * WHAT: Evaluate a string-valued array condition ["op", field, value] where op
166 * is "eq" (exact match) or "starts-with" (prefix match).
167 * WHY: These two operators share identical argument shapes and differ only in
168 * the final comparison, so one helper covers both.
169 * HOW: Require three string-typed elements; resolve the form value; an absent
170 * field is a violation. For "eq" compare the whole string, for
171 * "starts-with" compare the expected-length prefix (empty value => any).
172 * Returns NGX_OK if satisfied, NGX_DECLINED if violated, NGX_ERROR if the
173 * argument shapes are malformed.
174 */
175static ngx_int_t
176s3_post_cond_string_op(const s3_post_form_t *form, json_t *cond, const char *op)
177{
178 const char *field;
179 const char *expected;
180 const char *actual;
181
182 if (json_array_size(cond) != 3
183 || !json_is_string(json_array_get(cond, 1))
184 || !json_is_string(json_array_get(cond, 2)))
185 {
186 return NGX_ERROR;
187 }
188
189 field = json_string_value(json_array_get(cond, 1));
190 expected = json_string_value(json_array_get(cond, 2));
191 actual = s3_post_field_value(form, field);
192
193 if (actual == NULL) {
194 return NGX_DECLINED;
195 }
196
197 if (strcmp(op, "eq") == 0) {
198 return strcmp(actual, expected) == 0 ? NGX_OK : NGX_DECLINED;
199 }
200
201 return strncmp(actual, expected, strlen(expected)) == 0
202 ? NGX_OK : NGX_DECLINED;
203}

247/*
248 * WHAT: Evaluate the array-form of a condition ["op", arg1, arg2].
249 * WHY: Element 0 names the comparison; the remaining elements are its operands.
250 * HOW: Require a string operator, then look it up in s3_post_ops and delegate
251 * to the matching evaluator.
252 * Returns the evaluator's verdict, or NGX_ERROR if the operator is not a string
253 * or is unknown.
254 */
255static ngx_int_t
256s3_post_cond_array(const s3_post_form_t *form, json_t *cond)
257{
258 const char *op;
259 ngx_uint_t i;
260
261 if (!json_is_string(json_array_get(cond, 0))) {
262 return NGX_ERROR;
263 }
264
265 op = json_string_value(json_array_get(cond, 0));
266
267 for (i = 0; i < sizeof(s3_post_ops) / sizeof(s3_post_ops[0]); i++) {
268 if (strcmp(op, s3_post_ops[i].op) == 0) {
269 return s3_post_ops[i].eval(form, cond, op);
270 }
271 }
272
273 return NGX_ERROR;
274}

277/*
278 * WHAT: Evaluate a single POST-policy condition against the submitted form.
279 * WHY: This is the access-control core: a signed policy lists conditions the
280 * upload must satisfy; if any fails the upload is rejected even though the
281 * signature is valid. Two JSON shapes occur (AWS spec):
282 * - Object {"field":"value", ...} : every field must equal exactly
283 * (the special "bucket" key is matched against the configured bucket).
284 * - Array ["op", "field|$field", "value"] : op is "eq", "starts-with",
285 * or "content-length-range" (which bounds the file size).
286 * HOW: Dispatch on JSON type to the object-form or array-form evaluator.
287 * Returns NGX_OK if the condition is satisfied, NGX_DECLINED if it is violated,
288 * NGX_ERROR if the condition itself is malformed (unknown shape/op/types).
289 */
290ngx_int_t
291s3_post_policy_condition(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
292 const s3_post_form_t *form, json_t *cond)
293{
294 (void) r;
295
296 /* Object form: implicit "equals" for every key/value pair. */
297 if (json_is_object(cond)) {
298 return s3_post_cond_object(cf, form, cond);
299 }
300
301 /* Array form: ["op", arg1, arg2] — element 0 selects the comparison. */
302 if (json_is_array(cond) && json_array_size(cond) >= 3) {
303 return s3_post_cond_array(form, cond);
304 }
305
306 return NGX_ERROR;
307}

310/*
311 * WHAT: Parse the base64-decoded POST policy JSON and enforce all of it.
312 * WHY: The policy document is what the signature actually covers; once the
313 * signature is verified we must (a) confirm the policy has not expired and
314 * (b) confirm every listed condition holds for this upload. The upload is
315 * only authorised if BOTH hold.
316 * HOW: Load JSON, check "expiration" against ngx_time(), then evaluate each
317 * entry of "conditions" via s3_post_policy_condition().
318 * Ownership: `root` is jansson-refcounted; every early-return path json_decref()s
319 * it exactly once to avoid leaking the parse tree.
320 * Returns NGX_OK (allow), NGX_DECLINED (expired or a condition failed -> 403),
321 * NGX_ERROR (document malformed -> treated as invalid policy).
322 */
323ngx_int_t
324s3_post_validate_policy_json(ngx_http_request_t *r,
325 ngx_http_s3_loc_conf_t *cf, const s3_post_form_t *form,
326 u_char *policy_json, size_t policy_len)
327{
328 json_error_t jerr;
329 json_t *root;
330 json_t *expiration;
331 json_t *conditions;
332 size_t i;
333 time_t exp;
334
335 root = json_loadb((const char *) policy_json, policy_len, 0, &jerr);
336 if (root == NULL || !json_is_object(root)) {
337 if (root != NULL) {
338 json_decref(root); /* decref only if a tree was built */
339 }
340 return NGX_ERROR;
341 }
342
343 /* "expiration" must be present, parseable, and not in the past. */
344 expiration = json_object_get(root, "expiration");
345 if (!json_is_string(expiration)
346 || s3_post_parse_iso8601(json_string_value(expiration), &exp)
347 != NGX_OK
348 || ngx_time() > exp)
349 {
350 json_decref(root);
351 return NGX_DECLINED;
352 }
353
354 conditions = json_object_get(root, "conditions");
355 if (!json_is_array(conditions)) {
356 json_decref(root);
357 return NGX_ERROR;
358 }
359
360 /* Every condition must pass; the first non-OK result short-circuits and is
361 * propagated (DECLINED = violated, ERROR = malformed condition). */
362 for (i = 0; i < json_array_size(conditions); i++) {
363 ngx_int_t rc;
364
365 rc = s3_post_policy_condition(r, cf, form,
366 json_array_get(conditions, i));
367 if (rc != NGX_OK) {
368 json_decref(root);
369 return rc;
370 }
371 }
372
373 json_decref(root);
374 return NGX_OK;
375}

378/*
379 * WHAT: Validate the x-amz-credential scope and extract its date and region.
380 * WHY: The credential must be exactly "AKID/YYYYMMDD/REGION/s3/aws4_request"
381 * (full 5-part scope, unlike the parser in auth_sigv4_parse.c which only
382 * needs the first three). We enforce an 8-char date and the literal
383 * "s3/aws4_request" suffix so a forged/short scope is rejected before any
384 * crypto. HOW: locate the four '/' boundaries; *akid is set to point at
385 * the start (the caller measures its length up to p1).
386 * Returns NGX_OK (date/region filled, *akid set), or NGX_ERROR if the scope is
387 * malformed, the date is not 8 chars, or the suffix is wrong.
388 */
389ngx_int_t
390s3_post_parse_credential(const char *credential, char *date, size_t date_sz,
391 char *region, size_t region_sz, const char **akid)
392{
393 const char *p1, *p2, *p3, *p4;
394 size_t len;
395
396 p1 = strchr(credential, '/');
397 if (p1 == NULL) {
398 return NGX_ERROR;
399 }
400 *akid = credential; /* AKID = [credential, p1) */
401
402 /* Locate the remaining three slashes; require exactly the 5-part scope and
403 * the fixed "s3/aws4_request" trailer (text after the third slash). */
404 p2 = strchr(p1 + 1, '/');
405 p3 = p2 ? strchr(p2 + 1, '/') : NULL;
406 p4 = p3 ? strchr(p3 + 1, '/') : NULL;
407 if (p2 == NULL || p3 == NULL || p4 == NULL
408 || strcmp(p3 + 1, "s3/aws4_request") != 0)
409 {
410 return NGX_ERROR;
411 }
412
413 /* DATE = (p1, p2) — must be exactly 8 chars (YYYYMMDD). */
414 len = (size_t) (p2 - (p1 + 1));
415 if (len != 8 || len >= date_sz) {
416 return NGX_ERROR;
417 }
418 ngx_memcpy(date, p1 + 1, len);
419 date[len] = '\0';
420
421 /* REGION = (p2, p3). */
422 len = (size_t) (p3 - (p2 + 1));
423 if (len == 0 || len >= region_sz) {
424 return NGX_ERROR;
425 }
426 ngx_memcpy(region, p2 + 1, len);
427 region[len] = '\0';
428
429 return NGX_OK;
430}

448/*
449 * WHAT: Validate the signature-field shape and parse the credential scope.
450 * WHY: Before any crypto we reject uploads missing a signature field, and we
451 * require the fixed algorithm, a 64-hex signature, and a well-formed
452 * credential — cheap syntactic gates that fail-closed (INVARIANT 6: this
453 * is S3 SigV4, never shared with token auth). HOW: presence checks, then
454 * exact algorithm/length checks, then s3_post_parse_credential().
455 * On success fills sc->date/region/akid. On failure sets sc->status to the
456 * already-sent S3 error status and returns NGX_ERROR.
457 * Returns NGX_OK when the fields are well-formed, NGX_ERROR otherwise.
458 */
459static ngx_int_t
460s3_post_verify_fields(ngx_http_request_t *r, const s3_post_form_t *form,
461 s3_post_scope_t *sc)
462{
463 /* Every signature-bearing field must be present. */
464 if (form->policy[0] == '\0' || form->algorithm[0] == '\0'
465 || form->credential[0] == '\0' || form->amz_date[0] == '\0'
466 || form->signature[0] == '\0')
467 {
468 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
469 "Missing POST policy signature fields.");
470 return NGX_ERROR;
471 }
472
473 /* Algorithm fixed, signature is 64 hex chars, credential well-formed. */
474 if (strcmp(form->algorithm, "AWS4-HMAC-SHA256") != 0
475 || strlen(form->signature) != 64
476 || s3_post_parse_credential(form->credential, sc->date, sizeof(sc->date),
477 sc->region, sizeof(sc->region), &sc->akid)
478 != NGX_OK)
479 {
480 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidRequest",
481 "Malformed POST policy signature fields.");
482 return NGX_ERROR;
483 }
484
485 return NGX_OK;
486}

489/*
490 * WHAT: Match the credential's AKID, region, and date against this endpoint.
491 * WHY: A well-formed credential is not enough — its access key, region, and
492 * scope date must be the ones this location is configured for, else the
493 * upload is for a different key/endpoint. HOW: compare the AKID span
494 * ([akid, first '/')) to cf->access_key, then region to cf->region, then
495 * the scope date to the first 8 chars (YYYYMMDD) of x-amz-date.
496 * On mismatch sets sc->status to the already-sent S3 error and returns
497 * NGX_ERROR. Returns NGX_OK when all three agree, NGX_ERROR otherwise.
498 */
499static ngx_int_t
500s3_post_verify_scope(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
501 const s3_post_form_t *form, s3_post_scope_t *sc)
502{
503 /* AKID = bytes from `akid` up to the first '/'; its length must equal the
504 * configured key length AND the bytes must match exactly. */
505 if ((size_t) (strchr(form->credential, '/') - sc->akid) != cf->access_key.len
506 || ngx_strncmp(cf->access_key.data, (u_char *) sc->akid,
507 cf->access_key.len) != 0)
508 {
509 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidAccessKeyId",
510 "The access key ID does not exist.");
511 return NGX_ERROR;
512 }
513
514 /* Region must match the endpoint, and the credential-scope date must agree
515 * with the first 8 chars (YYYYMMDD) of x-amz-date. */
516 if (cf->region.len != strlen(sc->region)
517 || ngx_strncmp(cf->region.data, (u_char *) sc->region, cf->region.len)
518 != 0
519 || ngx_strncmp((u_char *) form->amz_date, (u_char *) sc->date, 8) != 0)
520 {
521 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
522 "Credential scope does not match this "
523 "endpoint.");
524 return NGX_ERROR;
525 }
526
527 return NGX_OK;
528}

531/*
532 * WHAT: Derive the SigV4 signing key, HMAC the policy string, and compare it
533 * constant-time against the submitted signature.
534 * WHY: This proves the base64 policy string *as submitted* was signed by the
535 * secret-key holder; we HMAC the raw form->policy bytes, NOT the decoded
536 * JSON. HOW: derive the date/region-scoped key, HMAC-SHA256 the policy,
537 * hex-encode, then CRYPTO_memcmp (so a mismatch position cannot be
538 * timing-probed).
539 * On a signature mismatch sets sc->status to the already-sent S3 error and
540 * returns NGX_ERROR; on a crypto failure sets sc->status to 500 and returns
541 * NGX_ERROR. Returns NGX_OK when the signature matches.
542 */
543static ngx_int_t
544s3_post_verify_signature(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
545 const s3_post_form_t *form, s3_post_scope_t *sc)
546{
547 u_char k4[32], computed[32];
548 char computed_hex[65];
549
550 if (!s3_sigv4_derive_signing_key_cached(&cf->secret_key,
551 sc->date, sc->region, k4)
552 || !brix_hmac_sha256(k4, 32, (u_char *) form->policy,
553 strlen(form->policy), computed))
554 {
555 sc->status = NGX_HTTP_INTERNAL_SERVER_ERROR;
556 return NGX_ERROR;
557 }
558
559 brix_hex_encode(computed, 32, computed_hex);
560 /* Constant-time compare to avoid leaking how many bytes matched. */
561 if (CRYPTO_memcmp(computed_hex, form->signature, 64) != 0) {
562 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN,
563 "SignatureDoesNotMatch",
564 "The request signature we calculated does not "
565 "match the signature you provided.");
566 return NGX_ERROR;
567 }
568
569 return NGX_OK;
570}

573/*
574 * WHAT: Base64-decode the (already signature-verified) policy and enforce it.
575 * WHY: Only after the signature matches may we trust the policy string; we then
576 * decode it to JSON and check expiration + conditions. HOW: allocate a
577 * decode buffer (base64 expands ~4:3, plus slack and a NUL), decode, then
578 * s3_post_validate_policy_json(); map its verdict to an S3 error.
579 * Returns NGX_OK if the policy authorises the upload; otherwise an already-sent
580 * S3 error status (suitable for the metrics finalize call).
581 */
582static ngx_int_t
583s3_post_enforce_policy(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
584 const s3_post_form_t *form)
585{
586 ngx_str_t src, decoded;
587 ngx_int_t rc;
588
589 src.data = (u_char *) form->policy;
590 src.len = ngx_strlen(form->policy);
591 decoded.len = src.len / 4 * 3 + 4;
592 decoded.data = ngx_pnalloc(r->pool, decoded.len + 1);
593 if (decoded.data == NULL) {
594 return NGX_HTTP_INTERNAL_SERVER_ERROR;
595 }

597 if (ngx_decode_base64(&decoded, &src) != NGX_OK) {

618/*
619 * WHAT: Verify the SigV4 signature over the POST policy, then enforce the policy.
620 * WHY: This is the authentication+authorization gate for a browser upload. The
621 * signature proves the policy was issued by the holder of the secret key;
622 * the policy then constrains what may be uploaded (INVARIANT 6: S3 SigV4,
623 * never shared with token auth).
624 * HOW (in order, fail-closed at each step):
625 * 1. If no access key is configured, S3 auth is disabled -> allow.
626 * 2-3. s3_post_verify_fields: fields present, algorithm/length/scope shape.
627 * 4. s3_post_verify_scope: AKID, region, and date match this endpoint.
628 * 5. s3_post_verify_signature: derive key, HMAC, constant-time compare.
629 * 6. s3_post_enforce_policy: decode and enforce expiration + conditions.
630 * Returns NGX_OK if authorised; otherwise an already-sent S3 error status (the
631 * return value is the HTTP status, suitable for the metrics finalize call).
632 */
633ngx_int_t
634s3_post_verify_policy(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
635 const s3_post_form_t *form)
636{
637 s3_post_scope_t sc = {0};
638
639 /* Step 1: no configured key => S3 authentication disabled for this location. */
640 if (cf->access_key.len == 0) {
641 return NGX_OK;
642 }
643
644 /* Steps 2-3: signature fields present and well-formed; parse credential. */
645 if (s3_post_verify_fields(r, form, &sc) != NGX_OK) {
646 return sc.status;
647 }
648
649 /* Step 4: AKID, region, and scope date match this endpoint. */
650 if (s3_post_verify_scope(r, cf, form, &sc) != NGX_OK) {
651 return sc.status;
652 }
653
654 /* Step 5: derive signing key, HMAC the policy string, constant-time compare. */
655 if (s3_post_verify_signature(r, cf, form, &sc) != NGX_OK) {
656 return sc.status;
657 }
658
659 /* Step 6: signature is valid -> decode the policy and enforce its contents. */
660 return s3_post_enforce_policy(r, cf, form);
661}
