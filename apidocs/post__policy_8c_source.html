10/*
11 * WHAT: Validate the x-amz-credential scope and extract its date and region.
12 * WHY: The credential must be exactly "AKID/YYYYMMDD/REGION/s3/aws4_request"
13 * (full 5-part scope, unlike the parser in auth_sigv4_parse.c which only
14 * needs the first three). We enforce an 8-char date and the literal
15 * "s3/aws4_request" suffix so a forged/short scope is rejected before any
16 * crypto. HOW: locate the four '/' boundaries; *akid is set to point at
17 * the start (the caller measures its length up to p1).
18 * Returns NGX_OK (date/region filled, *akid set), or NGX_ERROR if the scope is
19 * malformed, the date is not 8 chars, or the suffix is wrong.
20 */
21ngx_int_t
22s3_post_parse_credential(const char *credential, char *date, size_t date_sz,
23 char *region, size_t region_sz, const char **akid)
24{
25 const char *p1, *p2, *p3, *p4;
26 size_t len;
27
28 p1 = strchr(credential, '/');
29 if (p1 == NULL) {
30 return NGX_ERROR;
31 }
32 *akid = credential; /* AKID = [credential, p1) */
33
34 /* Locate the remaining three slashes; require exactly the 5-part scope and
35 * the fixed "s3/aws4_request" trailer (text after the third slash). */
36 p2 = strchr(p1 + 1, '/');
37 p3 = p2 ? strchr(p2 + 1, '/') : NULL;
38 p4 = p3 ? strchr(p3 + 1, '/') : NULL;
39 if (p2 == NULL || p3 == NULL || p4 == NULL
40 || strcmp(p3 + 1, "s3/aws4_request") != 0)
41 {
42 return NGX_ERROR;
43 }
44
45 /* DATE = (p1, p2) — must be exactly 8 chars (YYYYMMDD). */
46 len = (size_t) (p2 - (p1 + 1));
47 if (len != 8 || len >= date_sz) {
48 return NGX_ERROR;
49 }
50 ngx_memcpy(date, p1 + 1, len);
51 date[len] = '\0';
52
53 /* REGION = (p2, p3). */
54 len = (size_t) (p3 - (p2 + 1));
55 if (len == 0 || len >= region_sz) {
56 return NGX_ERROR;
57 }
58 ngx_memcpy(region, p2 + 1, len);
59 region[len] = '\0';
60
61 return NGX_OK;
62}

65/*
66 * Parsed SigV4 credential scope for one POST-policy verification, plus the
67 * error-status out slot. Threaded (by pointer) through the verify helpers so the
68 * per-step signatures stay narrow: `date`/`region` are the NUL-terminated
69 * YYYYMMDD and region strings, `akid` points into form->credential at the AKID
70 * start, and `status` receives the already-sent S3 error status on any failure.
71 */
72typedef struct {
73 char date[16];
74 char region[64];
75 const char *akid;
76 ngx_int_t status;
77} s3_post_scope_t;

80/*
81 * WHAT: Validate the signature-field shape and parse the credential scope.
82 * WHY: Before any crypto we reject uploads missing a signature field, and we
83 * require the fixed algorithm, a 64-hex signature, and a well-formed
84 * credential — cheap syntactic gates that fail-closed (INVARIANT 6: this
85 * is S3 SigV4, never shared with token auth). HOW: presence checks, then
86 * exact algorithm/length checks, then s3_post_parse_credential().
87 * On success fills sc->date/region/akid. On failure sets sc->status to the
88 * already-sent S3 error status and returns NGX_ERROR.
89 * Returns NGX_OK when the fields are well-formed, NGX_ERROR otherwise.
90 */
91static ngx_int_t
92s3_post_verify_fields(ngx_http_request_t *r, const s3_post_form_t *form,
93 s3_post_scope_t *sc)
94{
95 /* Every signature-bearing field must be present. */
96 if (form->policy[0] == '\0' || form->algorithm[0] == '\0'
97 || form->credential[0] == '\0' || form->amz_date[0] == '\0'
98 || form->signature[0] == '\0')
99 {
100 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
101 "Missing POST policy signature fields.");
102 return NGX_ERROR;
103 }
104
105 /* Algorithm fixed, signature is 64 hex chars, credential well-formed. */
106 if (strcmp(form->algorithm, "AWS4-HMAC-SHA256") != 0
107 || strlen(form->signature) != 64
108 || s3_post_parse_credential(form->credential, sc->date, sizeof(sc->date),
109 sc->region, sizeof(sc->region), &sc->akid)
110 != NGX_OK)
111 {
112 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidRequest",
113 "Malformed POST policy signature fields.");
114 return NGX_ERROR;
115 }
116
117 return NGX_OK;
118}

121/*
122 * WHAT: Match the credential's AKID, region, and date against this endpoint.
123 * WHY: A well-formed credential is not enough — its access key, region, and
124 * scope date must be the ones this location is configured for, else the
125 * upload is for a different key/endpoint. HOW: compare the AKID span
126 * ([akid, first '/')) to cf->access_key, then region to cf->region, then
127 * the scope date to the first 8 chars (YYYYMMDD) of x-amz-date.
128 * On mismatch sets sc->status to the already-sent S3 error and returns
129 * NGX_ERROR. Returns NGX_OK when all three agree, NGX_ERROR otherwise.
130 */
131static ngx_int_t
132s3_post_verify_scope(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
133 const s3_post_form_t *form, s3_post_scope_t *sc)
134{
135 /* AKID = bytes from `akid` up to the first '/'; its length must equal the
136 * configured key length AND the bytes must match exactly. */
137 if ((size_t) (strchr(form->credential, '/') - sc->akid) != cf->access_key.len
138 || ngx_strncmp(cf->access_key.data, (u_char *) sc->akid,
139 cf->access_key.len) != 0)
140 {
141 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidAccessKeyId",
142 "The access key ID does not exist.");
143 return NGX_ERROR;
144 }
145
146 /* Region must match the endpoint, and the credential-scope date must agree
147 * with the first 8 chars (YYYYMMDD) of x-amz-date. */
148 if (cf->region.len != strlen(sc->region)
149 || ngx_strncmp(cf->region.data, (u_char *) sc->region, cf->region.len)
150 != 0
151 || ngx_strncmp((u_char *) form->amz_date, (u_char *) sc->date, 8) != 0)
152 {
153 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
154 "Credential scope does not match this "
155 "endpoint.");
156 return NGX_ERROR;
157 }
158
159 return NGX_OK;
160}

163/*
164 * WHAT: Derive the SigV4 signing key, HMAC the policy string, and compare it
165 * constant-time against the submitted signature.
166 * WHY: This proves the base64 policy string *as submitted* was signed by the
167 * secret-key holder; we HMAC the raw form->policy bytes, NOT the decoded
168 * JSON. HOW: derive the date/region-scoped key, HMAC-SHA256 the policy,
169 * hex-encode, then CRYPTO_memcmp (so a mismatch position cannot be
170 * timing-probed).
171 * On a signature mismatch sets sc->status to the already-sent S3 error and
172 * returns NGX_ERROR; on a crypto failure sets sc->status to 500 and returns
173 * NGX_ERROR. Returns NGX_OK when the signature matches.
174 */
175static ngx_int_t
176s3_post_verify_signature(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
177 const s3_post_form_t *form, s3_post_scope_t *sc)
178{
179 u_char k4[32], computed[32];
180 char computed_hex[65];
181
182 if (!s3_sigv4_derive_signing_key_cached(&cf->secret_key,
183 sc->date, sc->region, k4)
184 || !brix_hmac_sha256(k4, 32, (u_char *) form->policy,
185 strlen(form->policy), computed))
186 {
187 sc->status = NGX_HTTP_INTERNAL_SERVER_ERROR;
188 return NGX_ERROR;
189 }
190
191 brix_hex_encode(computed, 32, computed_hex);
192 /* Constant-time compare to avoid leaking how many bytes matched. */
193 if (CRYPTO_memcmp(computed_hex, form->signature, 64) != 0) {
194 sc->status = s3_post_error(r, NGX_HTTP_FORBIDDEN,
195 "SignatureDoesNotMatch",
196 "The request signature we calculated does not "
197 "match the signature you provided.");
198 return NGX_ERROR;
199 }
200
201 return NGX_OK;
202}

205/*
206 * WHAT: Base64-decode the (already signature-verified) policy and enforce it.
207 * WHY: Only after the signature matches may we trust the policy string; we then
208 * decode it to JSON and check expiration + conditions. HOW: allocate a
209 * decode buffer (base64 expands ~4:3, plus slack and a NUL), decode, then
210 * s3_post_validate_policy_json(); map its verdict to an S3 error.
211 * Returns NGX_OK if the policy authorises the upload; otherwise an already-sent
212 * S3 error status (suitable for the metrics finalize call).
213 */
214static ngx_int_t
215s3_post_enforce_policy(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
216 const s3_post_form_t *form)
217{
218 ngx_str_t src, decoded;
219 ngx_int_t rc;
220
221 src.data = (u_char *) form->policy;
222 src.len = ngx_strlen(form->policy);
223 decoded.len = src.len / 4 * 3 + 4;
224 decoded.data = ngx_pnalloc(r->pool, decoded.len + 1);
225 if (decoded.data == NULL) {
226 return NGX_HTTP_INTERNAL_SERVER_ERROR;
227 }
228
229 if (ngx_decode_base64(&decoded, &src) != NGX_OK) {
230 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidPolicyDocument",
231 "The POST policy document is invalid.");
232 }
233 decoded.data[decoded.len] = '\0';
234
235 rc = s3_post_validate_policy_json(r, cf, form, decoded.data, decoded.len);
236 /* DECLINED = a condition/expiry failed (403); other non-OK = bad document. */
237 if (rc == NGX_DECLINED) {
238 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
239 "POST policy conditions were not satisfied.");
240 }
241 if (rc != NGX_OK) {
242 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidPolicyDocument",
243 "The POST policy document is invalid.");
244 }
245
246 return NGX_OK;
247}

250/*
251 * WHAT: Verify the SigV4 signature over the POST policy, then enforce the policy.
252 * WHY: This is the authentication+authorization gate for a browser upload. The
253 * signature proves the policy was issued by the holder of the secret key;
254 * the policy then constrains what may be uploaded (INVARIANT 6: S3 SigV4,
255 * never shared with token auth).
256 * HOW (in order, fail-closed at each step):
257 * 1. If no access key is configured, S3 auth is disabled -> allow.
258 * 2-3. s3_post_verify_fields: fields present, algorithm/length/scope shape.
259 * 4. s3_post_verify_scope: AKID, region, and date match this endpoint.
260 * 5. s3_post_verify_signature: derive key, HMAC, constant-time compare.
261 * 6. s3_post_enforce_policy: decode and enforce expiration + conditions.
262 * Returns NGX_OK if authorised; otherwise an already-sent S3 error status (the
263 * return value is the HTTP status, suitable for the metrics finalize call).
264 */
265ngx_int_t
266s3_post_verify_policy(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
267 const s3_post_form_t *form)
268{
269 s3_post_scope_t sc = {0};
270
271 /* Step 1: no configured key => S3 authentication disabled for this location. */
272 if (cf->access_key.len == 0) {
273 return NGX_OK;
274 }
275
276 /* Steps 2-3: signature fields present and well-formed; parse credential. */
277 if (s3_post_verify_fields(r, form, &sc) != NGX_OK) {
278 return sc.status;
279 }
280
281 /* Step 4: AKID, region, and scope date match this endpoint. */
282 if (s3_post_verify_scope(r, cf, form, &sc) != NGX_OK) {
283 return sc.status;
284 }
285
286 /* Step 5: derive signing key, HMAC the policy string, constant-time compare. */
287 if (s3_post_verify_signature(r, cf, form, &sc) != NGX_OK) {
288 return sc.status;
289 }
290
291 /* Step 6: signature is valid -> decode the policy and enforce its contents. */
292 return s3_post_enforce_policy(r, cf, form);
293}
