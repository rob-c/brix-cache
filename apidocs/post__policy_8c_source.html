8/*
9 * WHAT: Convert a proleptic-Gregorian civil date (y, m, d) to a day count
10 * relative to the Unix epoch (1970-01-01 == 0).
11 * WHY: Policy expiry must be compared against ngx_time() without depending on
12 * the host timezone, so we compute a UTC epoch directly rather than via
13 * mktime() (which is local-time and not async-signal-safe).
14 * HOW: Howard Hinnant's well-known days_from_civil algorithm. It shifts the
15 * year so March is month 1 (Feb's leap day lands at year-end), groups
16 * years into 400-year "eras" (each exactly 146097 days), then offsets by
17 * 719468 to re-anchor era day 0 onto the Unix epoch. Constants are exact,
18 * not approximations. Reference: howardhinnant.github.io/date_algorithms.html
19 * Returns signed days since 1970-01-01 (negative for pre-epoch dates).
20 */
21int64_t
22s3_post_days_from_civil(int y, unsigned m, unsigned d)
23{
24 int64_t era;
25 unsigned yoe, doy, doe;
26 int mp;
27
28 y -= m <= 2; /* Jan/Feb count to prior year */
29 era = (y >= 0 ? y : y - 399) / 400; /* 400-year era index */
30 yoe = (unsigned) (y - era * 400); /* year-of-era [0, 399] */
31 mp = (int) m + (m > 2 ? -3 : 9); /* month, March=0 .. Feb=11 */
32 doy = (153 * (unsigned) mp + 2) / 5 + d - 1;/* day-of-year, March 1 == 0 */
33 doe = yoe * 365 + yoe / 4 - yoe / 100 + doy;/* day-of-era [0, 146096] */
34
35 return era * 146097 + (int64_t) doe - 719468;
36}

39/*
40 * WHAT: Parse an ISO 8601 UTC timestamp "YYYY-MM-DDTHH:MM:SSZ" to a time_t.
41 * WHY: POST policy "expiration" is given in this exact UTC form; we need it as
42 * an epoch seconds value to compare against ngx_time(). HOW: strict
43 * sscanf with field widths, range-check each component (se <= 60 allows a
44 * leap second), then combine the civil-day count with the time-of-day.
45 * Returns NGX_OK with *out set, or NGX_ERROR on a malformed/out-of-range value.
46 */
47ngx_int_t
48s3_post_parse_iso8601(const char *s, time_t *out)
49{
50 int y, mo, d, h, mi, se;
51
52 if (sscanf(s, "%4d-%2d-%2dT%2d:%2d:%2dZ",
53 &y, &mo, &d, &h, &mi, &se) != 6)
54 {
55 return NGX_ERROR;
56 }
57
58 if (y < 1970 || mo < 1 || mo > 12 || d < 1 || d > 31
59 || h < 0 || h > 23 || mi < 0 || mi > 59 || se < 0 || se > 60)
60 {
61 return NGX_ERROR;
62 }
63
64 /* epoch = (whole days since 1970) * 86400 + seconds within the day. */
65 *out = (time_t) (s3_post_days_from_civil(y, (unsigned) mo, (unsigned) d)
66 * 86400 + h * 3600 + mi * 60 + se);
67 return NGX_OK;
68}

85/*
86 * WHAT: Evaluate a single POST-policy condition against the submitted form.
87 * WHY: This is the access-control core: a signed policy lists conditions the
88 * upload must satisfy; if any fails the upload is rejected even though the
89 * signature is valid. Two JSON shapes occur (AWS spec):
90 * - Object {"field":"value", ...} : every field must equal exactly
91 * (the special "bucket" key is matched against the configured bucket).
92 * - Array ["op", "field|$field", "value"] : op is "eq", "starts-with",
93 * or "content-length-range" (which bounds the file size).
94 * HOW: Dispatch on JSON type, then on operator; compare against the form
95 * value resolved via s3_post_field_value().
96 * Returns NGX_OK if the condition is satisfied, NGX_DECLINED if it is violated,
97 * NGX_ERROR if the condition itself is malformed (unknown shape/op/types).
98 */
99ngx_int_t
100s3_post_policy_condition(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
101 const s3_post_form_t *form, json_t *cond)
102{
103 const char *key;
104 json_t *value;
105
106 /* Object form: implicit "equals" for every key/value pair. */
107 if (json_is_object(cond)) {
108 json_object_foreach(cond, key, value) {
109 const char *expected;
110 const char *actual;
111
112 if (!json_is_string(value)) {
113 return NGX_ERROR;
114 }
115
116 expected = json_string_value(value);
117 /* "bucket" is not a form field — match the configured bucket name. */
118 if (strcmp(key, "bucket") == 0) {
119 if (cf->bucket.len != strlen(expected)
120 || ngx_strncmp(cf->bucket.data, (u_char *) expected,
121 cf->bucket.len) != 0)
122 {
123 return NGX_DECLINED;
124 }
125 continue;
126 }
127
128 actual = s3_post_field_value(form, key);
129 if (s3_post_check_field_eq(actual, expected) != NGX_OK) {
130 return NGX_DECLINED;
131 }
132 }
133 return NGX_OK;
134 }
135
136 /* Array form: ["op", arg1, arg2] — element 0 selects the comparison. */
137 if (json_is_array(cond) && json_array_size(cond) >= 3) {
138 const char *op;
139
140 if (!json_is_string(json_array_get(cond, 0))) {
141 return NGX_ERROR;
142 }
143
144 op = json_string_value(json_array_get(cond, 0));
145
146 /* content-length-range: [op, min, max] bounds the uploaded file size. */
147 if (strcmp(op, "content-length-range") == 0) {
148 json_int_t minv, maxv;
149
150 if (json_array_size(cond) != 3
151 || !json_is_integer(json_array_get(cond, 1))
152 || !json_is_integer(json_array_get(cond, 2)))
153 {
154 return NGX_ERROR;
155 }
156
157 minv = json_integer_value(json_array_get(cond, 1));
158 maxv = json_integer_value(json_array_get(cond, 2));
159 if ((json_int_t) form->file_len < minv
160 || (json_int_t) form->file_len > maxv)
161 {
162 return NGX_DECLINED;
163 }
164 return NGX_OK;
165 }
166
167 /* eq: exact match; starts-with: prefix match (empty value => any). */
168 if ((strcmp(op, "eq") == 0 || strcmp(op, "starts-with") == 0)
169 && json_array_size(cond) == 3
170 && json_is_string(json_array_get(cond, 1))
171 && json_is_string(json_array_get(cond, 2)))
172 {
173 const char *field = json_string_value(json_array_get(cond, 1));
174 const char *expected = json_string_value(json_array_get(cond, 2));
175 const char *actual = s3_post_field_value(form, field);
176
177 if (actual == NULL) {
178 return NGX_DECLINED;
179 }
180
181 if (strcmp(op, "eq") == 0) {
182 return strcmp(actual, expected) == 0 ? NGX_OK : NGX_DECLINED;
183 }
184
185 return strncmp(actual, expected, strlen(expected)) == 0
186 ? NGX_OK : NGX_DECLINED;
187 }
188 }
189
190 (void) r;
191 return NGX_ERROR;
192}

195/*
196 * WHAT: Parse the base64-decoded POST policy JSON and enforce all of it.
197 * WHY: The policy document is what the signature actually covers; once the
198 * signature is verified we must (a) confirm the policy has not expired and
199 * (b) confirm every listed condition holds for this upload. The upload is
200 * only authorised if BOTH hold.
201 * HOW: Load JSON, check "expiration" against ngx_time(), then evaluate each
202 * entry of "conditions" via s3_post_policy_condition().
203 * Ownership: `root` is jansson-refcounted; every early-return path json_decref()s
204 * it exactly once to avoid leaking the parse tree.
205 * Returns NGX_OK (allow), NGX_DECLINED (expired or a condition failed -> 403),
206 * NGX_ERROR (document malformed -> treated as invalid policy).
207 */
208ngx_int_t
209s3_post_validate_policy_json(ngx_http_request_t *r,
210 ngx_http_s3_loc_conf_t *cf, const s3_post_form_t *form,
211 u_char *policy_json, size_t policy_len)
212{
213 json_error_t jerr;
214 json_t *root;
215 json_t *expiration;
216 json_t *conditions;
217 size_t i;
218 time_t exp;
219
220 root = json_loadb((const char *) policy_json, policy_len, 0, &jerr);
221 if (root == NULL || !json_is_object(root)) {
222 if (root != NULL) {
223 json_decref(root); /* decref only if a tree was built */
224 }
225 return NGX_ERROR;
226 }
227
228 /* "expiration" must be present, parseable, and not in the past. */
229 expiration = json_object_get(root, "expiration");
230 if (!json_is_string(expiration)
231 || s3_post_parse_iso8601(json_string_value(expiration), &exp)
232 != NGX_OK
233 || ngx_time() > exp)
234 {
235 json_decref(root);
236 return NGX_DECLINED;
237 }
238
239 conditions = json_object_get(root, "conditions");
240 if (!json_is_array(conditions)) {
241 json_decref(root);
242 return NGX_ERROR;
243 }
244
245 /* Every condition must pass; the first non-OK result short-circuits and is
246 * propagated (DECLINED = violated, ERROR = malformed condition). */
247 for (i = 0; i < json_array_size(conditions); i++) {
248 ngx_int_t rc;
249
250 rc = s3_post_policy_condition(r, cf, form,
251 json_array_get(conditions, i));
252 if (rc != NGX_OK) {
253 json_decref(root);
254 return rc;
255 }
256 }
257
258 json_decref(root);
259 return NGX_OK;
260}

263/*
264 * WHAT: Validate the x-amz-credential scope and extract its date and region.
265 * WHY: The credential must be exactly "AKID/YYYYMMDD/REGION/s3/aws4_request"
266 * (full 5-part scope, unlike the parser in auth_sigv4_parse.c which only
267 * needs the first three). We enforce an 8-char date and the literal
268 * "s3/aws4_request" suffix so a forged/short scope is rejected before any
269 * crypto. HOW: locate the four '/' boundaries; *akid is set to point at
270 * the start (the caller measures its length up to p1).
271 * Returns NGX_OK (date/region filled, *akid set), or NGX_ERROR if the scope is
272 * malformed, the date is not 8 chars, or the suffix is wrong.
273 */
274ngx_int_t
275s3_post_parse_credential(const char *credential, char *date, size_t date_sz,
276 char *region, size_t region_sz, const char **akid)
277{
278 const char *p1, *p2, *p3, *p4;
279 size_t len;
280
281 p1 = strchr(credential, '/');
282 if (p1 == NULL) {
283 return NGX_ERROR;
284 }
285 *akid = credential; /* AKID = [credential, p1) */
286
287 /* Locate the remaining three slashes; require exactly the 5-part scope and
288 * the fixed "s3/aws4_request" trailer (text after the third slash). */
289 p2 = strchr(p1 + 1, '/');
290 p3 = p2 ? strchr(p2 + 1, '/') : NULL;
291 p4 = p3 ? strchr(p3 + 1, '/') : NULL;
292 if (p2 == NULL || p3 == NULL || p4 == NULL
293 || strcmp(p3 + 1, "s3/aws4_request") != 0)
294 {
295 return NGX_ERROR;
296 }
297
298 /* DATE = (p1, p2) — must be exactly 8 chars (YYYYMMDD). */
299 len = (size_t) (p2 - (p1 + 1));
300 if (len != 8 || len >= date_sz) {
301 return NGX_ERROR;
302 }
303 ngx_memcpy(date, p1 + 1, len);
304 date[len] = '\0';
305
306 /* REGION = (p2, p3). */
307 len = (size_t) (p3 - (p2 + 1));
308 if (len == 0 || len >= region_sz) {
309 return NGX_ERROR;
310 }
311 ngx_memcpy(region, p2 + 1, len);
312 region[len] = '\0';
313
314 return NGX_OK;
315}

318/*
319 * WHAT: Verify the SigV4 signature over the POST policy, then enforce the policy.
320 * WHY: This is the authentication+authorization gate for a browser upload. The
321 * signature in the form proves the policy was issued by the holder of the
322 * secret key; the policy then constrains what may be uploaded. Note the
323 * signed message is the base64 policy string *as submitted* — we HMAC the
324 * raw form->policy bytes, NOT the decoded JSON.
325 * HOW (in order, fail-closed at each step):
326 * 1. If no access key is configured, S3 auth is disabled -> allow.
327 * 2. Require all signature fields to be present.
328 * 3. Validate algorithm, signature length (hex64), and credential scope shape.
329 * 4. Match the credential's AKID, region, and date against this endpoint.
330 * 5. Derive the SigV4 signing key and HMAC-SHA256 the policy string; compare
331 * against the provided signature with CRYPTO_memcmp (constant-time, so a
332 * mismatch position cannot be timing-probed).
333 * 6. Only after the signature matches, base64-decode the policy and enforce
334 * its expiration + conditions.
335 * Returns NGX_OK if authorised; otherwise an already-sent S3 error status (the
336 * return value is the HTTP status, suitable for the metrics finalize call).
337 */
338ngx_int_t
339s3_post_verify_policy(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
340 const s3_post_form_t *form)
341{
342 char date[16];
343 char region[64];
344 const char *akid;
345 u_char k4[32], computed[32];
346 char computed_hex[65];
347 ngx_str_t src, decoded;
348 ngx_int_t rc;
349
350 /* Step 1: no configured key => S3 authentication disabled for this location. */
351 if (cf->access_key.len == 0) {
352 return NGX_OK;
353 }
354
355 /* Step 2: every signature-bearing field must be present. */
356 if (form->policy[0] == '\0' || form->algorithm[0] == '\0'
357 || form->credential[0] == '\0' || form->amz_date[0] == '\0'
358 || form->signature[0] == '\0')
359 {
360 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
361 "Missing POST policy signature fields.");
362 }
363
364 /* Step 3: algorithm fixed, signature is 64 hex chars, credential well-formed. */
365 if (strcmp(form->algorithm, "AWS4-HMAC-SHA256") != 0
366 || strlen(form->signature) != 64
367 || s3_post_parse_credential(form->credential, date, sizeof(date),
368 region, sizeof(region), &akid) != NGX_OK)
369 {
370 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidRequest",
371 "Malformed POST policy signature fields.");
372 }
373
374 /* Step 4a: AKID = bytes from `akid` up to the first '/'; its length must
375 * equal the configured key length AND the bytes must match exactly. */
376 if ((size_t) (strchr(form->credential, '/') - akid) != cf->access_key.len
377 || ngx_strncmp(cf->access_key.data, (u_char *) akid,
378 cf->access_key.len) != 0)
379 {
380 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidAccessKeyId",
381 "The access key ID does not exist.");
382 }
383
384 /* Step 4b: region must match the endpoint, and the credential-scope date
385 * must agree with the first 8 chars (YYYYMMDD) of x-amz-date. */
386 if (cf->region.len != strlen(region)
387 || ngx_strncmp(cf->region.data, (u_char *) region, cf->region.len) != 0
388 || ngx_strncmp((u_char *) form->amz_date, (u_char *) date, 8) != 0)
389 {
390 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
391 "Credential scope does not match this endpoint.");
392 }
393
394 /* Step 5: derive the date/region-scoped signing key, HMAC the policy string
395 * (the base64 text, not the JSON), hex-encode, and compare. */
396 if (!s3_sigv4_derive_signing_key_cached(&cf->secret_key,
397 date, region, k4)
398 || !brix_hmac_sha256(k4, 32, (u_char *) form->policy,
399 strlen(form->policy), computed))
400 {
401 return NGX_HTTP_INTERNAL_SERVER_ERROR;
402 }
403
404 brix_hex_encode(computed, 32, computed_hex);
405 /* Constant-time compare to avoid leaking how many bytes matched. */
406 if (CRYPTO_memcmp(computed_hex, form->signature, 64) != 0) {
407 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "SignatureDoesNotMatch",
408 "The request signature we calculated does not "
409 "match the signature you provided.");
410 }
411
412 /* Step 6: signature is valid -> decode the policy and enforce its contents.
413 * base64 expands ~4:3, so allocate len/4*3 plus slack and a NUL terminator. */
414 src.data = (u_char *) form->policy;
415 src.len = ngx_strlen(form->policy);
416 decoded.len = src.len / 4 * 3 + 4;
417 decoded.data = ngx_pnalloc(r->pool, decoded.len + 1);
418 if (decoded.data == NULL) {
419 return NGX_HTTP_INTERNAL_SERVER_ERROR;
420 }
421
422 if (ngx_decode_base64(&decoded, &src) != NGX_OK) {
423 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidPolicyDocument",
424 "The POST policy document is invalid.");
425 }
426 decoded.data[decoded.len] = '\0';
427
428 rc = s3_post_validate_policy_json(r, cf, form, decoded.data, decoded.len);
429 /* DECLINED = a condition/expiry failed (403); other non-OK = bad document. */
430 if (rc == NGX_DECLINED) {
431 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "AccessDenied",
432 "POST policy conditions were not satisfied.");
433 }
434 if (rc != NGX_OK) {
435 return s3_post_error(r, NGX_HTTP_FORBIDDEN, "InvalidPolicyDocument",
436 "The POST policy document is invalid.");
437 }
438
439 return NGX_OK;
440}
