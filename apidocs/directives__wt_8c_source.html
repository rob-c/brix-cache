23/* brix_conf_set_wt_enable — parse brix_write_through on|off. When on, dirty
24 * write handles are mirrored to origin on kXR_sync/kXR_close, making client
25 * uploads eligible for write-through. */
26
27char *
28brix_conf_set_wt_enable(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
29{
30 ngx_stream_brix_srv_conf_t *xcf = conf;
31 ngx_str_t *value;
32 int flag;
33
34 value = cf->args->elts;
35 (void) cmd;
36
37 if (ngx_strcmp(value[1].data, "on") == 0) {
38 flag = 1;
39 } else if (ngx_strcmp(value[1].data, "off") == 0) {
40 flag = 0;
41 } else {
42 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
43 "brix_write_through: invalid value \"%V\", must be on or off",
44 &value[1]);
45 return NGX_CONF_ERROR;
46 }
47
48 xcf->wt.enable = flag;
49 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
50 "brix: write-through %s", (flag ? "on" : "off"));
51 return NGX_CONF_OK;
52}

83/* brix_conf_set_wt_origin — parse brix_wt_origin (same format as cache_origin:
84 * plain host:port or root://host:port, IPv6 brackets supported), the destination
85 * for write-through propagation when wt_enable=on. */
86
87char *
88brix_conf_set_wt_origin(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
89{
90 ngx_stream_brix_srv_conf_t *xcf = conf;
91 ngx_str_t *value;
92 char *addr_copy, *colon, *endp;
93 const u_char *addr_data;
94 size_t addr_len;
95 long pnum;
96
97 value = cf->args->elts;
98 (void) cmd;
99
100 if (xcf->wt.origin_host.len > 0) {
101 return "is duplicate";
102 }
103
104 /* Accept plain host:port or root://host:port (same as cache_origin). */
105 addr_data = value[1].data;
106 addr_len = value[1].len;
107
108 if (addr_len > sizeof("root://") - 1
109 && ngx_strncmp(addr_data, "root://", sizeof("root://") - 1) == 0)
110 {
111 addr_data += sizeof("root://") - 1;
112 addr_len -= sizeof("root://") - 1;
113 }
114
115 BRIX_PNALLOC_OR_RETURN(addr_copy, cf->pool, addr_len + 1, NGX_CONF_ERROR);
116 ngx_memcpy(addr_copy, addr_data, addr_len);
117 addr_copy[addr_len] = '\0';
118
119 /* Parse host:port — supports IPv6 bracket notation. */
120 if (addr_copy[0] == '[') {
121 char *rb = strchr(addr_copy, ']');
122 if (rb == NULL || *(rb + 1) != ':') {
123 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
124 "brix_wt_origin: invalid address \"%V\"", &value[1]);
125 return NGX_CONF_ERROR;
126 }
127 size_t hostlen = (size_t)(rb - addr_copy - 1);
128 if (brix_pstrdupz(cf->pool, &xcf->wt.origin_host,
129 (u_char *) addr_copy + 1, hostlen) != NGX_OK) {
130 return NGX_CONF_ERROR;
131 }
132 pnum = strtol(rb + 2, &endp, 10);
133 } else {
134 colon = strrchr(addr_copy, ':');
135 if (colon == NULL) {
136 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
137 "brix_wt_origin: missing port in \"%V\"", &value[1]);
138 return NGX_CONF_ERROR;
139 }
140 size_t hostlen = (size_t)(colon - addr_copy);
141 if (brix_pstrdupz(cf->pool, &xcf->wt.origin_host,
142 (u_char *) addr_copy, hostlen) != NGX_OK) {
143 return NGX_CONF_ERROR;
144 }
145 pnum = strtol(colon + 1, &endp, 10);
146 }
147
148 if (*endp != '\0' || pnum <= 0 || pnum > 65535) {
149 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
150 "brix_wt_origin: invalid port in \"%V\"", &value[1]);
151 return NGX_CONF_ERROR;
152 }
153 xcf->wt.origin_port = (uint16_t) pnum;
154
155 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
156 "brix: write-through origin: %s:%d",
157 (char *) xcf->wt.origin_host.data, (int) xcf->wt.origin_port);
158
159 return NGX_CONF_OK;
160}

172{
173 ngx_stream_brix_srv_conf_t *xcf = conf;
174 (void) cmd;
