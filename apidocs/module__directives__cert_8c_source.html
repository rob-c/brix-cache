19/*
20 * Load the FIRST PEM certificate from a file (leaf-first chain convention:
21 * for a full-chain ssl_certificate the server's own cert comes first, so
22 * this always yields the entity cert whose issuer we must trust).
23 * Logs an EMERG and returns NULL on open/parse failure; caller owns the X509.
24 */
25static X509 *
26webdav_conf_load_leaf_cert(ngx_conf_t *cf, ngx_str_t *path)
27{
28 BIO *bio;
29 X509 *cert;
30
31 bio = BIO_new_file((const char *) path->data, "r");
32 if (bio == NULL) {
33 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
34 "brix_client_certificate_folder: cannot open "
35 "ssl_certificate \"%V\"", path);
36 return NULL;
37 }
38
39 cert = PEM_read_bio_X509(bio, NULL, NULL, NULL);
40 BIO_free(bio);
41
42 if (cert == NULL) {
43 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
44 "brix_client_certificate_folder: no PEM certificate "
45 "found in ssl_certificate \"%V\"", path);
46 }
47
48 return cert;
49}

52/*
53 * Probe <folder>/<issuer_subject_hash>.0 … .9 (the OpenSSL hashed-directory
54 * layout the IGTF distribution installs) for a certificate whose SUBJECT
55 * equals `issuer` — hash collisions make the suffix scan and the name
56 * comparison both necessary. On a match *found receives a pool-allocated,
57 * NUL-terminated path and NGX_OK is returned; NGX_DECLINED means no file in
58 * the folder matched (the caller reports it — the directive is the trust
59 * perimeter, so a miss is fatal); NGX_ERROR is an allocation failure.
60 */
61static ngx_int_t
62webdav_conf_probe_issuer_file(ngx_conf_t *cf, ngx_str_t *folder,
63 X509_NAME *issuer, ngx_str_t *found)
64{
65 u_char *buf, *p;
66 BIO *bio;
67 X509 *cand;
68 size_t buflen;
69 ngx_int_t match;
70 ngx_uint_t i;
71 unsigned long hash;
72
73#if OPENSSL_VERSION_NUMBER >= 0x30000000L
74 hash = X509_NAME_hash_ex(issuer, NULL, NULL, NULL);
75#else
76 hash = X509_NAME_hash(issuer);
77#endif
78
79 buflen = folder->len + sizeof("/xxxxxxxx.9");
80 buf = ngx_pnalloc(cf->pool, buflen);
81 if (buf == NULL) {
82 return NGX_ERROR;
83 }
84
85 for (i = 0; i < 10; i++) {
86 p = ngx_snprintf(buf, buflen - 1, "%V/%08xl.%ui", folder, hash, i);
87 *p = '\0';
88
89 bio = BIO_new_file((const char *) buf, "r");
90 if (bio == NULL) {
91 continue; /* suffix not present — normal */
92 }
93
94 cand = PEM_read_bio_X509(bio, NULL, NULL, NULL);
95 BIO_free(bio);
96 if (cand == NULL) {
97 continue; /* not a PEM cert — skip, keep scanning suffixes */
98 }
99
100 match = (X509_NAME_cmp(X509_get_subject_name(cand), issuer) == 0);
101 X509_free(cand);
102
103 if (match) {
104 found->data = buf;
105 found->len = p - buf;
106 return NGX_OK;
107 }
108 }
109
110 return NGX_DECLINED;
111}

114/*
115 * brix_client_certificate_folder <dir> — auto-pick ssl_client_certificate.
116 *
117 * WHAT: resolves, at config-parse time, the CA file in an OpenSSL hashed
118 * directory (/etc/grid-security/certificates layout) whose subject equals
119 * the ISSUER of this server's own ssl_certificate leaf, and assigns it as
120 * the stock ssl_client_certificate value.
121 *
122 * WHY: grid hosts ship only the hashed IGTF dir — no bundle file exists —
123 * yet nginx's `ssl_verify_client on` hard-requires a non-empty
124 * ssl_client_certificate, checked in the core ssl module's merge (BEFORE
125 * postconfiguration). Resolving here, in the directive handler, and handing
126 * the result to the stock machinery keeps every core semantic (trust load,
127 * advertised-CA list, the verify-on check) untouched.
128 *
129 * HOW: needs the server's own cert to hash its issuer, so it must appear
130 * AFTER ssl_certificate in the same server{} — enforced with a directed
131 * error. Explicit ssl_client_certificate coexistence is a config error
132 * (one source of truth). An unresolvable folder or a missing issuer file
133 * is FATAL: this feeds the client-verify trust store.
134 */
135char *
136webdav_conf_client_cert_folder(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
137{
138 ngx_http_ssl_srv_conf_t *sslcf;
139 ngx_str_t *value, folder, certpath, found;
140 ngx_file_info_t fi;
141 X509 *leaf;
142 X509_NAME *issuer;
143 ngx_int_t rc;
144 char dn[512];
145
146 (void) cmd;
147 (void) conf;
148
149 value = cf->args->elts; /* value[1] = hashed CA directory */
150
151 folder = value[1];
152 if (ngx_conf_full_name(cf->cycle, &folder, 1) != NGX_OK) {
153 return NGX_CONF_ERROR;
154 }
155
156 if (ngx_file_info(folder.data, &fi) != 0 || !ngx_is_dir(&fi)) {
157 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
158 "brix_client_certificate_folder \"%V\" is not an "
159 "accessible directory", &folder);
160 return NGX_CONF_ERROR;
161 }
162
163 sslcf = ngx_http_conf_get_module_srv_conf(cf, ngx_http_ssl_module);
164
165 if (sslcf == NULL
166 || sslcf->certificates == NGX_CONF_UNSET_PTR
167 || sslcf->certificates == NULL
168 || sslcf->certificates->nelts == 0)
169 {
170 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
171 "brix_client_certificate_folder needs the server's "
172 "own certificate to resolve its issuer: place it "
173 "AFTER ssl_certificate in the same server block");
174 return NGX_CONF_ERROR;
175 }
176
177 if (sslcf->client_certificate.len != 0) {
178 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
179 "brix_client_certificate_folder conflicts with an "
180 "ssl_client_certificate already set for this "
181 "server (or is itself a duplicate) — configure "
182 "exactly one of the two");
183 return NGX_CONF_ERROR;
184 }
185
186 certpath = ((ngx_str_t *) sslcf->certificates->elts)[0];
187
188 if (ngx_strlchr(certpath.data, certpath.data + certpath.len, '$') != NULL) {
189 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
190 "brix_client_certificate_folder cannot resolve a "
191 "variable ssl_certificate \"%V\" at config time",
192 &certpath);
193 return NGX_CONF_ERROR;
194 }
195
196 if (ngx_conf_full_name(cf->cycle, &certpath, 1) != NGX_OK) {
197 return NGX_CONF_ERROR;
198 }
199
200 leaf = webdav_conf_load_leaf_cert(cf, &certpath);
201 if (leaf == NULL) {
202 return NGX_CONF_ERROR;
203 }
204
205 issuer = X509_get_issuer_name(leaf);
206
207 rc = webdav_conf_probe_issuer_file(cf, &folder, issuer, &found);
208
209 if (rc == NGX_DECLINED) {
210 X509_NAME_oneline(issuer, dn, sizeof(dn));
211 X509_free(leaf);
212 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
213 "brix_client_certificate_folder: no <hash>.N file "
214 "in \"%V\" matches the issuer of \"%V\" (%s)",
215 &folder, &certpath, dn);
216 return NGX_CONF_ERROR;
217 }
218
219 X509_free(leaf);
220
221 if (rc != NGX_OK) {
222 return NGX_CONF_ERROR;
223 }
224
225 sslcf->client_certificate = found;
226
227 ngx_conf_log_error(NGX_LOG_INFO, cf, 0,
228 "brix_client_certificate_folder: using \"%V\" as "
229 "ssl_client_certificate", &found);
230
231 return NGX_CONF_OK;
232}

241/*
242 * Pick one CA certificate file (<8-hex-hash>.<digits> — the OpenSSL hashed
243 * layout; CRLs are <hash>.r<N> and never match) out of `folder`, returning
244 * the lexicographically smallest match as a pool-allocated, NUL-terminated
245 * full path in *picked. NGX_DECLINED = no CA file in the folder,
246 * NGX_ERROR = unreadable folder or allocation failure (logged).
247 */
248static ngx_int_t
249webdav_conf_pick_ca_file(ngx_conf_t *cf, ngx_str_t *folder, ngx_str_t *picked)
250{
251 ngx_dir_t dir;
252 ngx_str_t best;
253 u_char *name, *p;
254 size_t len, i;
255
256 if (ngx_open_dir(folder, &dir) != NGX_OK) {
257 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
258 "brix_proxy_ssl_capath: cannot open \"%V\"", folder);
259 return NGX_ERROR;
260 }
261
262 ngx_str_null(&best);
263
264 for ( ;; ) {
265 ngx_set_errno(0);
266 if (ngx_read_dir(&dir) != NGX_OK) {
267 break;
268 }
269
270 name = ngx_de_name(&dir);
271 len = ngx_de_namelen(&dir);
272
273 if (len < sizeof("00000000.0") - 1 || name[8] != '.') {
274 continue;
275 }
276 for (i = 0; i < 8; i++) {
277 if (!((name[i] >= '0' && name[i] <= '9')
278 || (name[i] >= 'a' && name[i] <= 'f')))
279 {
280 break;
281 }
282 }
283 if (i < 8) {
284 continue;
285 }
286 for (i = 9; i < len; i++) {
287 if (name[i] < '0' || name[i] > '9') {
288 break;
289 }
290 }
291 if (i < len) {
292 continue; /* .r0 CRLs, .signing_policy, ... */
293 }
294
295 if (best.data == NULL
296 || ngx_memn2cmp(name, best.data, len, best.len) < 0)
297 {
298 best.data = ngx_pnalloc(cf->pool, len);
299 if (best.data == NULL) {
300 ngx_close_dir(&dir);
301 return NGX_ERROR;
302 }
303 ngx_memcpy(best.data, name, len);
304 best.len = len;
305 }
306 }
307
308 ngx_close_dir(&dir);
309
310 if (best.data == NULL) {
311 return NGX_DECLINED;
312 }
313
314 picked->len = folder->len + 1 + best.len;
315 picked->data = ngx_pnalloc(cf->pool, picked->len + 1);
316 if (picked->data == NULL) {
317 return NGX_ERROR;
318 }
319 p = ngx_cpymem(picked->data, folder->data, folder->len);
320 *p++ = '/';
321 p = ngx_cpymem(p, best.data, best.len);
322 *p = '\0';
323
324 return NGX_OK;
325}

328/*
329 * brix_proxy_ssl_capath <dir> — hashed CA directory for the proxy back leg.
330 *
331 * WHAT: makes proxy_ssl_verify consume an OpenSSL hashed CA directory
332 * (/etc/grid-security/certificates) instead of the file-only
333 * proxy_ssl_trusted_certificate.
334 *
335 * WHY: stock nginx demands a non-empty proxy_ssl_trusted_certificate when
336 * proxy_ssl_verify is on (checked in the proxy module's merge, before
337 * postconfiguration) and cannot load a directory — the same gap
338 * brix_ssl_client_capath closes on the front leg.
339 *
340 * HOW: two halves. Here, at parse time: validate the folder, remember it in
341 * wlcf->proxy_ssl_capath (location-exact, never merged), and satisfy the
342 * mandatory-file check by injecting one CA file from the folder through the
343 * proxy module's OWN proxy_ssl_trusted_certificate command entry — the
344 * loc-conf struct is private, but the command's setter+offset are not, so
345 * this reuses stock machinery with no struct-layout assumptions. (Any file
346 * from the folder is sound: the trust it adds is a subset of the folder's.)
347 * The stock slot also makes an explicit proxy_ssl_trusted_certificate in the
348 * same location an automatic "is duplicate" conflict. At postconfiguration
349 * (postconfig.c) the folder itself is added to the location's upstream
350 * SSL_CTX as a hashed lookup.
351 */
352char *
353webdav_conf_proxy_ssl_capath(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
354{
355 ngx_http_brix_webdav_loc_conf_t *wlcf = conf;
356 ngx_str_t *value, folder, picked, args_buf[2];
357 ngx_command_t *pcmd;
358 ngx_array_t args, *saved;
359 ngx_file_info_t fi;
360 ngx_int_t rc;
361 void *plcf;
362 char *rv;
363
364 (void) cmd;
365
366 if (wlcf->proxy_ssl_capath.len > 0) {
367 return "is duplicate";
368 }
369
370 value = cf->args->elts; /* value[1] = hashed CA directory */
371
372 folder = value[1];
373 if (ngx_conf_full_name(cf->cycle, &folder, 1) != NGX_OK) {
374 return NGX_CONF_ERROR;
375 }
376
377 if (ngx_file_info(folder.data, &fi) != 0 || !ngx_is_dir(&fi)) {
378 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
379 "brix_proxy_ssl_capath \"%V\" is not an "
380 "accessible directory", &folder);
381 return NGX_CONF_ERROR;
382 }
383
384 rc = webdav_conf_pick_ca_file(cf, &folder, &picked);
385 if (rc == NGX_DECLINED) {
386 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
387 "brix_proxy_ssl_capath \"%V\" contains no "
388 "<hash>.N CA files", &folder);
389 return NGX_CONF_ERROR;
390 }
391 if (rc != NGX_OK) {
392 return NGX_CONF_ERROR;
393 }
394
395 for (pcmd = ngx_http_proxy_module.commands;
396 pcmd != NULL && pcmd->name.len != 0;
397 pcmd++)
398 {
399 if (ngx_strcmp(pcmd->name.data, "proxy_ssl_trusted_certificate") == 0) {
400 break;
401 }
402 }
403 if (pcmd == NULL || pcmd->name.len == 0) {
404 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
405 "brix_proxy_ssl_capath requires the http proxy "
406 "module with SSL support");
407 return NGX_CONF_ERROR;
408 }
409
410 plcf = ngx_http_conf_get_module_loc_conf(cf, ngx_http_proxy_module);
411
412 args_buf[0] = pcmd->name;
413 args_buf[1] = picked;
414 args.elts = args_buf;
415 args.nelts = 2;
416 args.size = sizeof(ngx_str_t);
417 args.nalloc = 2;
418 args.pool = cf->pool;
419
420 saved = cf->args;
421 cf->args = &args;
422 rv = pcmd->set(cf, pcmd, plcf);
423 cf->args = saved;
424
425 if (rv != NGX_CONF_OK) {
426 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
427 "brix_proxy_ssl_capath conflicts with an explicit "
428 "proxy_ssl_trusted_certificate in the same "
429 "location (proxy_ssl_trusted_certificate %s) — "
430 "configure exactly one of the two", rv);
431 return NGX_CONF_ERROR;
432 }
433
434 wlcf->proxy_ssl_capath = folder;
435
436 ngx_conf_log_error(NGX_LOG_INFO, cf, 0,
437 "brix_proxy_ssl_capath: trusted-certificate seed "
438 "\"%V\"; hashed dir added at postconfiguration",
439 &picked);
440
441 return NGX_CONF_OK;
442}
