2 * ucred_parse.c — credential-file format parsers for the ucred helper.
3 *
4 * WHAT: Implements the four credential-file readers declared in
5 * ucred_internal.h — x509 PEM expiry check, bearer .token reader,
6 * .s3 SigV4 triple parser, and CephX .keyring section-header parser —

31/* ---- Classify one candidate credential file ----
32 * WHAT: NGX_OK valid; NGX_DECLINED absent/unreadable/unparseable;
33 * NGX_DECLINED with *expired=1 when parseable but past notAfter.
34 * WHY: Expiry must be checked at the moment of use — a proxy can lapse
35 * between the request open and a deferred flush.
36 * HOW: 1. fopen; 2. PEM_read_X509 (first block = the proxy cert);
37 * 3. X509_cmp_current_time(notAfter) must be > 0. */
38ngx_int_t
39ucred_check_pem(const char *path, int *expired)
40{
41 FILE *f;
42 X509 *cert;
43 int cmp;
44
45 *expired = 0;
46 f = fopen(path, "r");
47 if (f == NULL) {
48 return NGX_DECLINED;
49 }
50 cert = PEM_read_X509(f, NULL, NULL, NULL);
51 (void) fclose(f); /* read-only stream — nothing buffered to lose */
52 if (cert == NULL) {
53 return NGX_DECLINED; /* unparseable = treated missing */
54 }
55 cmp = X509_cmp_current_time(X509_get0_notAfter(cert));
56 X509_free(cert);
57 if (cmp <= 0) {
58 *expired = 1;
59 return NGX_DECLINED;
60 }
61 return NGX_OK;
62}

64/* ---- Read a bearer token from a .token file --------------------------------
65 * WHAT: Opens `path` (O_RDONLY|O_NOFOLLOW|O_CLOEXEC), reads at most cap-1
66 * bytes, trims at the first CR, LF, space, or NUL, and copies the
67 * result into out_bearer. Returns NGX_OK on success; NGX_DECLINED when
68 * the file is absent/unreadable or the trimmed value is empty.
69 * WHY: Bearer tokens have no X.509 notAfter field — the token's own exp claim
70 * is enforced by the origin server at authentication time. Expiry checking
71 * is therefore the origin's responsibility; we only validate that the file
72 * contains a non-empty string. O_NOFOLLOW prevents a symlink from
73 * escaping the credential directory.
74 * HOW: 1. open with O_RDONLY|O_NOFOLLOW|O_CLOEXEC; absent → DECLINED.
75 * 2. read up to cap-1 bytes into a stack buffer.
76 * 3. Trim at the first whitespace/NUL character.
77 * 4. Reject an empty result → DECLINED; copy into out_bearer → OK. */
78ngx_int_t
79ucred_read_token(const char *path, char *out_bearer, size_t cap)
80{
81 char buf[BRIX_UCRED_BEARER_MAX];
82 ssize_t n;
83 size_t end;
84 int fd;
85
86 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
87 if (fd < 0) {
88 return NGX_DECLINED;
89 }
90 n = read(fd, buf, cap - 1);
91 close(fd);
92 if (n <= 0) {
93 return NGX_DECLINED;
94 }
95 buf[n] = '\0';
96
97 /* Trim at the first whitespace or embedded NUL — tokens must be a single
98 * word with no embedded spaces, CRs, or LFs. */
99 for (end = 0; end < (size_t) n; end++) {
100 unsigned char c = (unsigned char) buf[end];
101 if (c == '\r' || c == '\n' || c == ' ' || c == '\t' || c == '\0') {
102 break;
103 }
104 }
105 if (end == 0) {
106 OPENSSL_cleanse(buf, (size_t) n); /* scratch held a bearer token */
107 return NGX_DECLINED; /* empty after trimming */
108 }
109 memcpy(out_bearer, buf, end);
110 out_bearer[end] = '\0';
111 OPENSSL_cleanse(buf, (size_t) n); /* wipe the secret from the stack */
112 return NGX_OK;
113}

115/* ---- Trim one line from a read buffer at CR/LF, return its length --------
116 * WHAT: Scans buf[start..end) for the first CR or LF and returns the offset
117 * of that terminator (or `end` if none found before the buffer end).
118 * WHY: Shared by ucred_read_s3's 3-line parse; keeps the line-splitting
119 * logic in one place instead of duplicating a scan-for-newline loop
120 * three times.
121 * HOW: Linear scan; no allocation. */
122static size_t
123ucred_line_end(const char *buf, size_t start, size_t end)
124{
125 size_t i;
126
127 for (i = start; i < end; i++) {
128 if (buf[i] == '\r' || buf[i] == '\n') {
129 return i;
130 }
131 }
132 return end;
133}

135/* ---- Advance past exactly ONE line terminator (\r\n, \n, or \r) ----------
136 * WHAT: Given `pos` pointing at a CR/LF (as returned by ucred_line_end) or at
137 * `end` (no terminator found), returns the offset of the next line's
138 * first byte: pos+2 for "\r\n", pos+1 for a lone "\r" or "\n", or `end`
139 * unchanged when pos already equals end.
140 * WHY: A single-terminator skip (as opposed to skipping ALL consecutive
141 * CR/LF bytes) is required so an intentionally blank line — e.g. an
142 * empty secret-key field in a malformed .s3 file — is preserved as a
143 * zero-length field for the caller to reject, instead of being
144 * silently swallowed together with the next line's terminator.
145 * HOW: Bounds-checked lookahead for the 2-byte CRLF case; else advance 1. */
146static size_t
147ucred_skip_eol(const char *buf, size_t pos, size_t end)
148{
149 if (pos >= end) {
150 return end;
151 }
152 if (buf[pos] == '\r' && pos + 1 < end && buf[pos + 1] == '\n') {
153 return pos + 2;
154 }
155 return pos + 1;
156}

158/* ---- Read an S3 access-key/secret-key/region triple from a .s3 file ------
159 * WHAT: Opens `path` (O_RDONLY|O_NOFOLLOW|O_CLOEXEC), reads a bounded chunk,
160 * and parses up to 3 CR/LF-terminated lines: line 1 = access key id
161 * (required, non-empty), line 2 = secret key (required, non-empty),
162 * line 3 = region (optional; "us-east-1" when absent or empty).
163 * Fills out_ak/out_sk/out_region. Returns NGX_OK on a well-formed
164 * file, NGX_DECLINED when the file is absent/unreadable, either
165 * required line is missing/empty, or any field would overflow its
166 * output buffer.
167 * WHY: S3-backed exports need a per-user SigV4 credential (phase-3 T3),
168 * mirroring the .pem/.token precedence chain with a third, simpler
169 * file format (no ASN.1/JWT parsing — just delimited plaintext).
170 * O_NOFOLLOW prevents a symlink from escaping the credential
171 * directory, matching ucred_read_token's hardening.
172 * HOW: 1. open + bounded read into a stack buffer.
173 * 2. ucred_line_end three times to locate line boundaries.
174 * 3. Reject empty ak or sk → DECLINED.
175 * 4. memcpy each field with an explicit NUL; overflow → DECLINED.
176 * 5. Empty/absent region line → default "us-east-1". */
177ngx_int_t
178ucred_read_s3(const char *path, char *out_ak, size_t ak_cap,
179 char *out_sk, size_t sk_cap, char *out_region, size_t region_cap)
180{
181 char buf[BRIX_UCRED_S3_AK_MAX + BRIX_UCRED_S3_SK_MAX
182 + BRIX_UCRED_S3_REGION_MAX + 8];
183 ssize_t n;
184 size_t ak_start, ak_end, sk_start, sk_end, rg_start, rg_end, rg_len;
185 int fd;
186
187 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
188 if (fd < 0) {
189 return NGX_DECLINED;
190 }
191 n = read(fd, buf, sizeof(buf) - 1);
192 close(fd);
193 if (n <= 0) {
194 return NGX_DECLINED;
195 }
196 buf[n] = '\0';
197
198 /* Line 1: access key id. */
199 ak_start = 0;
200 ak_end = ucred_line_end(buf, ak_start, (size_t) n);
201 if (ak_end == ak_start || (ak_end - ak_start) >= ak_cap) {
202 OPENSSL_cleanse(buf, sizeof(buf)); /* scratch held the secret key */
203 return NGX_DECLINED;
204 }
205
206 /* Line 2: secret key. Skip exactly the ONE line terminator after line 1
207 * (not all consecutive CR/LF bytes — a blank line must stay a zero-length
208 * field, not be merged into the following line). */
209 sk_start = ucred_skip_eol(buf, ak_end, (size_t) n);
210 sk_end = ucred_line_end(buf, sk_start, (size_t) n);
211 if (sk_end == sk_start || (sk_end - sk_start) >= sk_cap) {
212 OPENSSL_cleanse(buf, sizeof(buf));
213 return NGX_DECLINED;
214 }
215
216 /* Line 3: region (optional). */
217 rg_start = ucred_skip_eol(buf, sk_end, (size_t) n);
218 rg_end = ucred_line_end(buf, rg_start, (size_t) n);
219 rg_len = rg_end - rg_start;
220 if (rg_len >= region_cap) {
221 OPENSSL_cleanse(buf, sizeof(buf));
222 return NGX_DECLINED;
223 }
224
225 memcpy(out_ak, buf + ak_start, ak_end - ak_start);
226 out_ak[ak_end - ak_start] = '\0';
227 memcpy(out_sk, buf + sk_start, sk_end - sk_start);
228 out_sk[sk_end - sk_start] = '\0';
229 if (rg_len == 0) {
230 snprintf(out_region, region_cap, "us-east-1");
231 } else {
232 memcpy(out_region, buf + rg_start, rg_len);
233 out_region[rg_len] = '\0';
234 }
235 OPENSSL_cleanse(buf, sizeof(buf)); /* wipe the secret key from the stack */
236 return NGX_OK;
237}

239/* ---- Read a CephX user id from a `[client.NAME]` keyring section header --
240 * WHAT: Opens `path` (O_RDONLY|O_NOFOLLOW|O_CLOEXEC), reads a bounded chunk,
241 * and scans for the FIRST line of the form "[client.NAME]" (leading/
242 * trailing whitespace around the brackets tolerated). Fills out_path
243 * with `path` itself (the keyring PATH — librados reads the file, we
244 * never parse the key material) and out_user with the bare id NAME
245 * (no "client." prefix). Returns NGX_OK on a well-formed keyring,
246 * NGX_DECLINED when the file is absent/unreadable or no
247 * "[client.NAME]" section header is found.
248 * WHY: A `<key>.keyring` credential (ceph-peruser item) lets the driver
249 * authenticate to RADOS as a specific CephX user instead of the export's
250 * static service credential. librados's rados_create() wants the bare
251 * id (it re-adds "client." itself — see sd_ceph_user_id in sd_ceph.c),
252 * so the bare id is what this parser stores. O_NOFOLLOW prevents a
253 * symlink from escaping the credential directory, matching
254 * ucred_read_token/ucred_read_s3's hardening.
255 * HOW: 1. open + bounded read into a stack buffer.
256 * 2. Line-scan for "[client." ... "]"; extract the id between them.
257 * 3. Reject an empty id or overflow → DECLINED; else fill out_user and
258 * out_path (verbatim `path`) → OK. */
259/* ---- Advance `i` to the start of the next line (past the next '\n') -------
260 * WHAT: Returns the offset of the first byte after the next '\n' at or after
261 * `i`, or `end` when no '\n' remains.
262 * WHY: ucred_keyring_parse_line uses this to resume scanning after a line
263 * that is not a usable "[client.NAME]" header; factoring it out removes
264 * the three duplicated scan-to-newline loops that inflated the caller's
265 * cyclomatic complexity.
266 * HOW: Linear scan to the terminator; return terminator+1 (bounded by end). */
267static size_t
268ucred_keyring_next_line(const char *buf, size_t i, size_t end)
269{
270 while (i < end && buf[i] != '\n') {
271 i++;
272 }
273 if (i < end) {
274 i++; /* step past the '\n' */
275 }
276 return i;
277}

279/* ---- Parse one keyring line for a "[client.NAME]" section header ----------
280 * WHAT: Examines the line beginning at `line`. On a well-formed
281 * "[client.NAME]" header, sets out->id_start/out->id_len to the bare
282 * NAME span and returns NGX_OK. On any other line returns NGX_DECLINED
283 * and sets out->next to the start of the following line so the caller
284 * keeps scanning.
285 * WHY: Isolates the header-recognition branching (whitespace skip, bracket,
286 * "client." prefix, closing bracket) from ucred_read_keyring's loop,
287 * collapsing that loop to a single dispatch on this helper's result.
288 * HOW: 1. skip leading spaces/tabs; 2. require '['; 3. require "client.";
289 * 4. locate the closing ']' before EOL; on any miss, compute out->next
290 * via ucred_keyring_next_line and return NGX_DECLINED. */
291typedef struct {
292 size_t id_start; /* offset of the bare NAME span (valid iff NGX_OK) */
293 size_t id_len; /* length of the bare NAME span (valid iff NGX_OK) */
294 size_t next; /* start of the next line (valid iff NGX_DECLINED) */
295} ucred_keyring_line_t;

297static ngx_int_t
298ucred_keyring_parse_line(const char *buf, size_t line, size_t end,
299 ucred_keyring_line_t *out)
300{
301 size_t i = line;
302 size_t j, start, stop;
303
304 out->id_start = 0;
305 out->id_len = 0;
306 out->next = end;
307
308 while (i < end && (buf[i] == ' ' || buf[i] == '\t')) {
309 i++;
310 }
311 if (i >= end || buf[i] != '[') {
312 out->next = ucred_keyring_next_line(buf, i, end);
313 return NGX_DECLINED;
314 }
315 j = i + 1;
316 if (j + 7 > end || strncmp(buf + j, "client.", 7) != 0) {
317 out->next = ucred_keyring_next_line(buf, i, end);
318 return NGX_DECLINED;
319 }
320 start = j + 7;
321 stop = start;
322 while (stop < end && buf[stop] != ']' && buf[stop] != '\n') {
323 stop++;
324 }
325 if (stop >= end || buf[stop] != ']') {
326 out->next = ucred_keyring_next_line(buf, i, end);
327 return NGX_DECLINED;
328 }
329 out->id_start = start;
330 out->id_len = stop - start;
331 return NGX_OK;
332}

334ngx_int_t
335ucred_read_keyring(const char *path, char *out_path, size_t path_cap,
336 char *out_user, size_t user_cap)
337{
338 char buf[4096];
339 ssize_t n;
340 size_t i, end;
341 int fd;
342
343 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
344 if (fd < 0) {
345 return NGX_DECLINED;
346 }
347 n = read(fd, buf, sizeof(buf) - 1);
348 close(fd);
349 if (n <= 0) {
350 return NGX_DECLINED;
351 }
352 buf[n] = '\0';
353 end = (size_t) n;
354
355 i = 0;
356 while (i < end) {
357 ucred_keyring_line_t line = { 0, 0, end };
358
359 if (ucred_keyring_parse_line(buf, i, end, &line) != NGX_OK) {
360 i = line.next;
361 continue;
362 }
363
364 if (line.id_len == 0 || line.id_len >= user_cap) {
365 OPENSSL_cleanse(buf, sizeof(buf)); /* scratch held CephX key material */
366 return NGX_DECLINED; /* empty or oversized id: malformed */
367 }
368 memcpy(out_user, buf + line.id_start, line.id_len);
369 out_user[line.id_len] = '\0';
370
371 if (snprintf(out_path, path_cap, "%s", path) >= (int) path_cap) {
372 OPENSSL_cleanse(buf, sizeof(buf));
373 return NGX_DECLINED;
374 }
375 OPENSSL_cleanse(buf, sizeof(buf)); /* wipe key material from the stack */
376 return NGX_OK;
377 }
378
379 OPENSSL_cleanse(buf, sizeof(buf));
380 return NGX_DECLINED; /* no "[client.NAME]" section header found */
381}
