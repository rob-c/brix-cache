1/*
2 * server_conf_merge_security.c — server-block merge for the identity/crypto and
3 * storage-plane configuration areas.
4 *
5 * WHAT: Owns brix_merge_srv_security() (auth scheme + GSI/pwd, XrdAcc engine,
6 * X.509/CRL, tokens + L1/L2 caches, sss/krb5/unix/host, TLS toggles) and
7 * brix_merge_srv_storage() (compression, ZIP, the read-through cache
8 * origin/sizing/eviction/verify, memory budget, readv sizing, io_uring),
9 * together with the file-local per-concern helpers each delegates to.
10 * WHY: Split (phase-79 file-size cap) out of the former 1249-line
11 * server_conf.c. Grouping the identity/crypto and storage merges keeps
12 * their sentinel-vs-configured contracts reviewable in isolation. The two
13 * entry points are non-static (declared in server_conf_internal.h) because
14 * the top-level orchestrator in server_conf.c calls them in linear order;
15 * every sub-helper stays file-local.
16 * HOW: Standard nginx parent->child inheritance via ngx_conf_merge_* and the
17 * BRIX_MERGE_* macros, one helper per concern group, invoked in the
18 * original order so cross-group derivations (staging LOW watermark from
19 * HIGH, reaper watermarks from the eviction threshold) still observe their
20 * already-merged inputs. No behaviour change from the split.
21 */
22

68 /*
69 * The native authdb engine matches by DN/VO and so needs an authenticating
70 * scheme; the xrdacc engine also authorizes anonymous `u *` rules, so it is
71 * exempt. Validated here, where both directives have settled.
72 */
73 if (conf->authdb.len > 0
74 && conf->acc.format == BRIX_AUTHDB_FORMAT_NATIVE
75 && conf->auth != BRIX_AUTH_GSI && conf->auth != BRIX_AUTH_TOKEN
76 && conf->auth != (BRIX_AUTH_GSI | BRIX_AUTH_TOKEN))
77 {
78 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
79 "brix_authdb (native format) requires brix_auth gsi, token "
80 "or both; use `brix_authdb_format xrdacc` for anonymous rules");
81 return NGX_CONF_ERROR;
82 }
83
84 return NGX_CONF_OK;
85}

121/*
122 * WHAT: merge the token group (JWKS, issuer/audience, config/registry, macaroon
123 * secrets) and the throttle group (limits + named rate-limit zone), and
124 * validate the clock-skew bound and throttle-zone reference.
125 * WHY: both carry config-time validation (clock-skew range, zone existence);
126 * keeping them together isolates the two failure paths.
127 * HOW: inherit the token/throttle scalars, clamp-check clock skew to [0,300],
128 * then resolve the named rate-limit zone, failing if it was not declared.
129 */
130static char *

132 ngx_stream_brix_srv_conf_t *prev)

143 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,

170 prev->token_macaroon_secret, "");
171 ngx_conf_merge_str_value(conf->token_macaroon_secret_old,
172 prev->token_macaroon_secret_old, "");
173 return NGX_CONF_OK;

176/*
177 * WHAT: merge the CSI record cache, the Phase-20 L1/L2 caches (token/auth/rate),
178 * the sss/krb5/unix/host schemes, the security level, and the TLS toggles.
179 * WHY: a plain inheritance tail with no validation; a void helper trims the
180 * orchestrator without splitting a decision.
181 * HOW: whole-config inherit for the kv-backed caches (NULL == disabled), then
182 * child<-parent for the remaining scheme/TLS scalars.
183 */
184static void
185brix_merge_srv_authtail(ngx_stream_brix_srv_conf_t *conf,
186 ngx_stream_brix_srv_conf_t *prev)
187{
188 brix_csi_conf_merge(&conf->csi, &prev->csi);
189
190 /* Phase 20 caches/limits: inherit the parent's whole config when this
191 * block did not declare its own (kv still NULL). */
192 if (conf->token_cache_kv == NULL) {
193 conf->token_cache_kv = prev->token_cache_kv;
194 }
195 if (conf->auth_cache.kv == NULL) {
196 conf->auth_cache = prev->auth_cache;
197 }
198 if (conf->rate_limit.kv == NULL) {
199 conf->rate_limit = prev->rate_limit;
200 }
201 ngx_conf_merge_str_value(conf->sss_keytab, prev->sss_keytab, "");
202 ngx_conf_merge_value(conf->sss_lifetime, prev->sss_lifetime, 13);
203 ngx_conf_merge_str_value(conf->krb5.principal, prev->krb5.principal, "");
204 ngx_conf_merge_str_value(conf->krb5.keytab, prev->krb5.keytab, "");
205 ngx_conf_merge_value(conf->krb5.ip_check, prev->krb5.ip_check, 0);
206 ngx_conf_merge_value(conf->unix_trust_remote, prev->unix_trust_remote, 0);
207 ngx_conf_merge_ptr_value(conf->host_allow, prev->host_allow, NULL);
208 ngx_conf_merge_uint_value(conf->security_level, prev->security_level, 0);
209 ngx_conf_merge_uint_value(conf->min_sec_level, prev->min_sec_level, 0);
210 ngx_conf_merge_value(conf->opaque_strict, prev->opaque_strict, 0);
211 ngx_conf_merge_value(conf->tls, prev->tls, 0);
212 /* kTLS default ON (unified with the HTTP plane); SSL_OP_ENABLE_KTLS is a
213 * transparent no-op when the negotiated cipher/kernel cannot offload. */
214 ngx_conf_merge_value(conf->tls_ktls, prev->tls_ktls, 1);

217/* Identity & crypto: auth scheme + GSI/pwd, XrdAcc engine (+ native-authdb
218 * validation), SciTags/FRM, X.509 material + CRL, access log, tokens + L1/L2
219 * caches, sss/krb5/unix/host, security level, and TLS toggles. */
220char *
221brix_merge_srv_security(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *conf,
222 ngx_stream_brix_srv_conf_t *prev)
223{
224 /*
225 * Standard nginx inheritance rules: values set on the current server
226 * override the parent, otherwise we fall back to the parent or the hard
227 * coded module default. Each concern group is delegated to a helper below,
228 * invoked in the original linear order so cross-group derivations still see
229 * their already-merged inputs.
230 */
231 /* Shared common.* preamble (root defaults to "/": a pure cache node may
232 * omit brix_root and serve the whole namespace). Also covers the tier
233 * grammar + pmark + hard read-only enforcement — do not re-merge those. */
234 if (ngx_http_brix_shared_merge(cf, &prev->common, &conf->common, "/")
235 != NGX_CONF_OK)
236 {
237 return NGX_CONF_ERROR;
238 }
239 if (brix_merge_srv_gsi_acc(cf, conf, prev) != NGX_CONF_OK) {
240 return NGX_CONF_ERROR;
241 }
242 if (brix_merge_srv_x509(cf, conf, prev) != NGX_CONF_OK) {
243 return NGX_CONF_ERROR;
244 }
245 if (brix_merge_srv_tokens(cf, conf, prev) != NGX_CONF_OK) {
246 return NGX_CONF_ERROR;
247 }
248 brix_merge_srv_authtail(conf, prev);
249
250 return NGX_CONF_OK;
251}

276 ngx_conf_merge_value(conf->cache, prev->cache, 0);
277 ngx_conf_merge_str_value(conf->cache_root, prev->cache_root, "");

306/*
307 * WHAT: merge the read-through cache origin + eviction group — origin address/
308 * TLS/family, lock timeout, the on-fill eviction threshold, the watermark
309 * reaper, the file-size/memory budgets, and the readv segment size.
310 * WHY: the reaper HIGH/LOW watermarks derive from the just-merged eviction
311 * threshold; grouping keeps that dependency chain visible.
312 * HOW: inherit the origin scalars, then the reaper watermarks (HIGH ← eviction
313 * threshold, LOW ← HIGH − 5%) and the budget/segment defaults.
314 */
315static void
316brix_merge_srv_cache_origin(ngx_stream_brix_srv_conf_t *conf,
317 ngx_stream_brix_srv_conf_t *prev)
318{
319 ngx_conf_merge_str_value(conf->cache_origin, prev->cache_origin, "");
320 ngx_conf_merge_value(conf->cache_origin_tls, prev->cache_origin_tls, 0);
321 ngx_conf_merge_uint_value(conf->cache_origin_family,
322 prev->cache_origin_family, BRIX_AF_AUTO);
323 ngx_conf_merge_value(conf->cache_lock_timeout,
324 prev->cache_lock_timeout, 300);
325 ngx_conf_merge_uint_value(conf->cache_eviction_threshold,
326 prev->cache_eviction_threshold, 900000);
327
328 /* Watermark reaper: HIGH defaults to the on-fill eviction threshold so an
329 * existing config keeps its bound; LOW defaults 50000 ppm (5%) below HIGH for
330 * hysteresis; the timer runs every 60s by default. The ordering invariant
331 * (0 < low < high < 1e6) is enforced in runtime_server.c. */
332 ngx_conf_merge_uint_value(conf->reaper.high_watermark,
333 prev->reaper.high_watermark,
334 conf->cache_eviction_threshold);
335 ngx_conf_merge_uint_value(conf->reaper.low_watermark,
336 prev->reaper.low_watermark,
337 conf->reaper.high_watermark > 50000
338 ? conf->reaper.high_watermark - 50000
339 : conf->reaper.high_watermark / 2);
340 ngx_conf_merge_sec_value(conf->reaper.reap_interval,
341 prev->reaper.reap_interval, 60);
342 ngx_conf_merge_off_value(conf->cache_max_file_size,
343 prev->cache_max_file_size, 0);
344 ngx_conf_merge_off_value(conf->memory_budget,
345 prev->memory_budget, 768 * 1024 * 1024);
346 /* Default = stock XRootD maxReadv_ior = maxBuffsz(2 MiB) - sizeof(readahead_list). */
347 ngx_conf_merge_size_value(conf->readv_segment_size,
348 prev->readv_segment_size,
349 (size_t) (2 * 1024 * 1024) - BRIX_READV_SEGSIZE);

352/*
353 * WHAT: merge the io_uring backend group, the checksum-on-fill verify mode, the
354 * Pelican advertisement group, and inherit the compiled include-regex.
355 * WHY: the advertisement interval is clamped to the federation minimum and the
356 * regex inheritance is a struct-copy neither expressible via
357 * ngx_conf_merge_*; grouping keeps both special cases together.
358 * HOW: inherit the io_uring/verify/advertise scalars, floor the advertise
359 * interval at 60s, and copy the parent's compiled regex when unset.
360 */
361static void
362brix_merge_srv_iouring_advertise(ngx_stream_brix_srv_conf_t *conf,
363 ngx_stream_brix_srv_conf_t *prev)
364{
365 /* Phase 44: optional io_uring backend — default OFF (strictly opt-in).
366 *
367 * WHY not AUTO: the startup probe can only prove the ring accepts opcodes
368 * and (SB-W hardening) that the registered eventfd delivers a NOP
369 * completion — neither of which proves real buffered file WRITES complete
370 * on THIS kernel + filesystem. On at least one production host (EL9 elrepo
371 * 6.15, plain local fs) NOPs drained fine yet io_uring writes never
372 * completed: transfers wedged after exactly queue_depth in-flight ops
373 * (queue_depth x 32 KiB = 8 MiB) with a worker spinning, and a torn-down
374 * connection's still-in-flight ops became a late-CQE use-after-free. The
375 * thread pool is both correct there and FASTER (50 vs stall). io_uring is
376 * a performance option, not a correctness feature, so it must be an
377 * explicit, operator-verified `brix_io_uring on` — never silently engaged.
378 * `on` still fail-fasts if the backend is unavailable; `auto` remains for
379 * anyone who wants best-effort enable. */
380 ngx_conf_merge_uint_value(conf->io_uring,
381 prev->io_uring, BRIX_IO_URING_OFF);
382 /* D-3: seccomp filter default OFF — strictly opt-in (audit-first rollout). */
383 ngx_conf_merge_uint_value(conf->seccomp,
384 prev->seccomp, BRIX_SECCOMP_OFF);
385 /* E-4: negative-path backoff default OFF (threshold 0) — availability-first,
386 * strictly opt-in. window_ms/backoff_s only matter when threshold > 0. */
387 ngx_conf_merge_uint_value(conf->negcache.threshold,
388 prev->negcache.threshold, 0);
389 ngx_conf_merge_uint_value(conf->negcache.window_ms,
390 prev->negcache.window_ms, 0);
391 ngx_conf_merge_uint_value(conf->negcache.backoff_s,
392 prev->negcache.backoff_s, 0);
393 ngx_conf_merge_value(conf->io_uring_queue_depth,
394 prev->io_uring_queue_depth,
395 BRIX_IO_URING_QUEUE_DEPTH);
396 ngx_conf_merge_str_value(conf->io_uring_panic_file,
397 prev->io_uring_panic_file, "");
398 ngx_conf_merge_value(conf->io_uring_admin, prev->io_uring_admin, 0);
399 ngx_conf_merge_value(conf->io_uring_restrict, prev->io_uring_restrict, 1);
400
401
402 /* Checksum-on-fill: default best-effort (verify when a digest is available,
403 * fail-closed on mismatch). Operators opt down to off or up to require. */
404 ngx_conf_merge_uint_value(conf->cache_verify, prev->cache_verify,
405 BRIX_CACHE_VERIFY_BESTEFFORT);
406 ngx_conf_merge_str_value(conf->cache_verify_digest,
407 prev->cache_verify_digest, "");
408
409 /* Pelican cache advertisement (default off; interval clamped to the
410 * federation minimum of 60s = MinFedTokenTickerRate). */
411 ngx_conf_merge_value(conf->advertise.enable, prev->advertise.enable, 0);
412 ngx_conf_merge_msec_value(conf->advertise.interval,
413 prev->advertise.interval, 60000);
414 if (conf->advertise.interval < 60000) {
415 conf->advertise.interval = 60000;
416 }
417 ngx_conf_merge_str_value(conf->advertise.key,
418 prev->advertise.key, "");
419 ngx_conf_merge_str_value(conf->advertise.data_url, prev->advertise.data_url, "");
420 ngx_conf_merge_str_value(conf->advertise.web_url, prev->advertise.web_url, "");
421 ngx_conf_merge_str_value(conf->advertise.sitename, prev->advertise.sitename, "");
422 ngx_conf_merge_str_value(conf->advertise.issuer_url, prev->advertise.issuer_url, "");
423 if (conf->advertise.ns == NULL) {
424 conf->advertise.ns = prev->advertise.ns;
425 }
426
427 /* Inherit compiled regex from parent if the child didn't set one */
428 if (!conf->include_regex.set && prev->include_regex.set) {
429 conf->include_regex.str = prev->include_regex.str;
430 conf->include_regex.re = prev->include_regex.re;
431 conf->include_regex.set = 1;
432 }
