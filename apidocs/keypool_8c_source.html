17/*
18 * Per-worker pool state. The nginx event thread is the ONLY mutator: it pops on
19 * kXGC_certreq and pushes from the refill done-callback (which also runs on the
20 * event thread). The refill THREAD function touches only its own task-local
21 * storage, never this state — so no locking is required.
22 *
23 * `target` is the runtime warm ceiling (brix_gsi_keypool_size); `pool` is the
24 * thread pool captured at init so the done-callback can chain further off-thread
25 * refills until the target is reached without blocking the event loop.
26 */
27static EVP_PKEY *brix_kp_ring[BRIX_GSI_KEYPOOL_CAP];
28static ngx_uint_t brix_kp_count;
29static ngx_uint_t brix_kp_target = BRIX_GSI_KEYPOOL_SIZE_DEFAULT;
30static ngx_uint_t brix_kp_refill_pending;
31static ngx_uint_t brix_kp_warmed_logged; /* one-shot "warmed" notice */
32static ngx_thread_pool_t *brix_kp_pool;
33static ngx_log_t *brix_kp_log;
34
35/* Refill once the pool drops to half the target (was a fixed REFILL_LOW). */
36static ngx_inline ngx_uint_t
37brix_kp_low_water(void)
38{
39 return brix_kp_target / 2;
40}

52/* Worker-thread half of a refill: pure keygen into task-local storage. */
53static void
54brix_kp_refill_thread(void *data, ngx_log_t *log)
55{
56 brix_gsi_refill_t *r = data;
57 ngx_uint_t i;
58
59 (void) log;
60 for (i = 0; i < BRIX_GSI_KEYPOOL_REFILL_BATCH; i++) {
61 EVP_PKEY *k = brix_gsi_dh_keygen();
62 if (k == NULL) {
63 break; /* keep whatever we managed to generate */
64 }
65 r->keys[i] = k;
66 }
67 r->n = i;
68}

74static void
75brix_kp_refill_done(ngx_event_t *ev)
76{
77 ngx_thread_task_t *task = ev->data;
78 brix_gsi_refill_t *r = task->ctx;
79 ngx_uint_t i;
80
81 for (i = 0; i < r->n; i++) {
82 if (brix_kp_count < brix_kp_target) {
83 brix_kp_ring[brix_kp_count++] = r->keys[i];
84 } else {
85 EVP_PKEY_free(r->keys[i]); /* pool refilled meanwhile — drop extra */
86 }
87 }
88 brix_kp_refill_pending = 0;
89 ngx_free(task);
90
91 /* Chain another batch while still below target (initial off-thread warm-up
92 * to the configured size, one batch at a time, all off the event thread). */
93 if (brix_kp_pool != NULL && brix_kp_count < brix_kp_target) {
94 brix_kp_schedule_refill(brix_kp_pool);
95 return;
96 }
97
98 /* Log once when the background warm-up first reaches the configured target,
99 * so an operator can confirm the off-thread fill completed (and as the test
100 * hook that the lazy seed path really did finish filling the pool). */
101 if (!brix_kp_warmed_logged && brix_kp_count >= brix_kp_target) {
102 brix_kp_warmed_logged = 1;
103 ngx_log_error(NGX_LOG_NOTICE, brix_kp_log, 0,
104 "brix: GSI DH key pool warmed to %ui/%ui keys "
105 "(off-thread)", brix_kp_count, brix_kp_target);
106 }
107}

110static void
111brix_kp_schedule_refill(ngx_thread_pool_t *pool)
112{
113 ngx_thread_task_t *task;
114
115 task = ngx_calloc(sizeof(ngx_thread_task_t) + sizeof(brix_gsi_refill_t),
116 brix_kp_log);
117 if (task == NULL) {
118 return; /* leave pending=0; retried on the next low pop */
119 }
120 task->ctx = task + 1;
121 brix_task_bind(task, brix_kp_refill_thread, brix_kp_refill_done);
122
123 if (ngx_thread_task_post(pool, task) != NGX_OK) {
124 ngx_free(task);
125 return;
126 }
127 brix_kp_refill_pending = 1;
128}

131void
132brix_gsi_keypool_init(ngx_cycle_t *cycle, ngx_thread_pool_t *pool,
133 ngx_uint_t target, ngx_uint_t seed)
134{
135 ngx_uint_t warm;
136
137 brix_kp_log = cycle->log;
138
139 /* Clamp to the static ring ceiling and a sane seed window. */
140 if (target == 0 || target > BRIX_GSI_KEYPOOL_CAP) {
141 target = BRIX_GSI_KEYPOOL_CAP;
142 }
143 if (seed == 0) {
144 seed = 1;
145 }
146 if (seed > target) {
147 seed = target;
148 }
149 brix_kp_target = target;
150 brix_kp_pool = pool;
151
152 /*
153 * With a thread pool, generate only `seed` keys synchronously (cheap, keeps
154 * the worker accepting connections fast) and fill the rest OFF the event
155 * thread. Without a pool there is nowhere to offload, so fall back to the
156 * old behaviour and warm the full target synchronously — correctness over
157 * boot latency.
158 */
159 warm = (pool != NULL) ? seed : target;
160
161 while (brix_kp_count < warm) {
162 EVP_PKEY *k = brix_gsi_dh_keygen();
163 if (k == NULL) {
164 break;
165 }
166 brix_kp_ring[brix_kp_count++] = k;
167 }
168
169 if (pool != NULL && brix_kp_count < brix_kp_target) {
170 brix_kp_schedule_refill(pool); /* background warm-up to target */
171 }
172
173 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
174 "brix: GSI DH key pool seeded %ui/%ui keys (%s)",
175 brix_kp_count, brix_kp_target,
176 pool != NULL ? "rest filling off-thread"
177 : "synchronous, no thread pool");
178}

181ngx_int_t
182brix_gsi_keypool_pop(ngx_thread_pool_t *pool, ngx_log_t *log, EVP_PKEY **out)
183{
184 ngx_int_t hit = 0;
185
186 if (brix_kp_log == NULL) {
187 brix_kp_log = log; /* lazy capture for refill task logging */
188 }
189
190 if (brix_kp_count > 0) {
191 *out = brix_kp_ring[--brix_kp_count];
192 hit = 1;
193 }
194
195 /* Top the pool back up off-thread once it runs low (and a pool exists). */
196 if (pool != NULL
197 && !brix_kp_refill_pending
198 && brix_kp_count <= brix_kp_low_water())
199 {
200 brix_kp_schedule_refill(pool);
201 }
202
203 return hit;
204}
