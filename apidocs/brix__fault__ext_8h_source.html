1/*
2 * brix_fault_ext.h — pure, root-free stream-mutation + spoofing helpers that
3 * extend brix-fault-proxy into a full on-path MITM for wild-internet hardening.
4 *
5 * These are the byte-surgery and forgery primitives an adversary between a public
6 * client and a public service can apply WITHOUT touching the kernel: deleting or
7 * duplicating individual bytes (framing desync / length-mismatch bugs), rewriting
8 * a substring on the wire (mangle a Content-Length, flip a status line, poison a
9 * checksum, inject a header), splicing bytes in (request smuggling / protocol
10 * confusion), and prepending a forged PROXY-protocol header to make a service
11 * trust an attacker-chosen source IP.
12 *
13 * Everything here is a PURE function over caller-owned buffers — no globals, no
14 * sockets, no locks — so the relay can snapshot its (locked) config and call in
15 * from the hot path, and the unit tests can exercise the byte math directly.
16 */

22/* A snapshot of the payload-level mutation config for one direction. Byte
23 * pointers are borrowed (the caller owns the storage for the call's duration).
24 * A zero-length field disables that mutation. */
25typedef struct {
26 const unsigned char *find; size_t find_len; /* replace: needle */
27 const unsigned char *repl; size_t repl_len; /* replace: replacement */
28 const unsigned char *inject; size_t inject_len; /* one-shot prefix splice */
29 int drop_ppm; /* per-byte delete probability (ppm) */
30 int repeat_ppm; /* per-byte duplicate probability (ppm) */
31} fp_ext_mut;

33/* Event tallies accumulated by fp_ext_mutate() for the status oracle. */
34typedef struct {
35 unsigned long dropped; /* bytes deleted */
36 unsigned long repeated; /* bytes duplicated */
37 unsigned long injected; /* bytes spliced in */
38 unsigned long replaced; /* substring matches rewritten */
39} fp_ext_stats;

41/* True if `m` has any active mutation (so the caller can skip the copy). */
42int fp_ext_mut_active(const fp_ext_mut *m);

44/* Transform `in[0..n)` into `out` (capacity `outcap`), applying — in order — the
45 * one-shot inject prefix, then per byte: a non-overlapping `find`->`repl` rewrite,
46 * else a probabilistic delete, else emit-and-maybe-duplicate. `seed` is the
47 * caller's private rand_r() state. Never writes past `outcap` (a full buffer just
48 * truncates the mutated output, which is itself a valid fault). Returns the
49 * produced length. Tallies land in *st (added, not reset). */
50size_t fp_ext_mutate(const unsigned char *in, size_t n,
51 unsigned char *out, size_t outcap,
52 const fp_ext_mut *m, unsigned *seed, fp_ext_stats *st);
53
54/* Decode a control-plane payload token into raw bytes. "hex:deadbeef" decodes
55 * hex; "str:foo\r\n" (or a bare token) is taken literally with C escapes
56 * \r \n \t \0 \xNN \\ honoured. Returns the byte length, or -1 on a malformed
57 * token. Writes at most `cap` bytes. */
58int fp_ext_parse_payload(const char *tok, unsigned char *out, size_t cap);
59
60/* Build a PROXY-protocol v1 (human-readable) header for TCP4/TCP6 into `out`.
61 * `src`/`dst` are "ip:port"; if `dst` is NULL the proxy's own endpoint is assumed
62 * by the caller and should be passed in. Returns the header length or -1. */
63int fp_ext_proxy_v1(char *out, size_t cap, const char *src, const char *dst);
64
65/* Build a PROXY-protocol v2 (binary) header. Same argument contract; returns the
66 * total header length or -1 (e.g. mixed/invalid families). */
67int fp_ext_proxy_v2(unsigned char *out, size_t cap, const char *src, const char *dst);
