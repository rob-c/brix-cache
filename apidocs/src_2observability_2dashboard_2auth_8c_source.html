8/*
9 * dashboard/auth.c — session-cookie verification for the live monitor.
10 *
11 * This file owns the "verify a session" half of dashboard authentication; the
12 * "issue a session" login flow is in dashboard_auth_login.c, the credential
13 * store + session HMAC in dashboard_auth_creds.c, and the untrusted-input
14 * parsing in dashboard_auth_parse.c (phase-79 split of the former 1104-line
15 * auth.c — see dashboard_auth_internal.h for the cross-file seam).
16 *
17 * COOKIE FORMAT: xrd_dashboard=<hex32_hmac>.<timestamp_s>
18 * - timestamp_s is Unix time (decimal, seconds) when the cookie was issued.
19 * - hmac is HMAC-SHA256(password, timestamp_s) as 32 lowercase hex bytes.
20 *
21 * VERIFICATION:
22 * 1. Split on '.'; parse timestamp_s.
23 * 2. Reject if now - timestamp_s > session_ttl.
24 * 3. Recompute HMAC(password, timestamp_s); constant-time compare with CRYPTO_memcmp.
25 *
26 * SECURITY NOTE:
27 * This is a minimal, firewalled-admin-only mechanism. The password is in
28 * nginx.conf; the cookie carries no user identity. The module is NOT
29 * suitable for Internet-facing deployments without additional firewall rules.
30 */
31
32/* Fields recovered from a "<hex>.<ts>[.<username>]" session cookie: the

34 * selected for verification (per-user hash or the single configured password). */

42/*
43 * WHAT: Validate the cookie's leading HMAC field and copy it out NUL-terminated.
44 * WHY: The HMAC field must be exactly HMAC_HEX_LEN bytes ending at the first
45 * '.'; anything else (no dot, leading dot, wrong-length hex) is a
46 * malformed cookie and is rejected (fail closed) with an audit event.
47 * HOW: Locate the first '.', check the field length, copy into given_hex
48 * (HMAC_HEX_LEN + 1 bytes) and return the dot via *dot_out.
49 * Returns NGX_OK or NGX_HTTP_UNAUTHORIZED.
50 */
51static ngx_int_t
52dash_auth_extract_hex(u_char *cookie_val, size_t cookie_val_len,
53 char given_hex[HMAC_HEX_LEN + 1], u_char **dot_out)
54{
55 u_char *dot = NULL;
56 size_t i;
57
58 /* Find the '.' separator between hex and timestamp. */
59 for (i = 0; i < cookie_val_len; i++) {
60 if (cookie_val[i] == '.') { dot = cookie_val + i; break; }
61 }
62
63 if (dot == NULL || dot == cookie_val ||
64 (size_t)(dot - cookie_val) != HMAC_HEX_LEN)
65 {
66 brix_dashboard_event_add(BRIX_DASH_EVENT_AUTH, 0,
67 NGX_HTTP_UNAUTHORIZED,
68 "dashboard auth malformed cookie", NULL);
69 return NGX_HTTP_UNAUTHORIZED;
70 }
71
72 /* Extract and null-terminate the HMAC hex. */
73 ngx_memcpy(given_hex, cookie_val, HMAC_HEX_LEN);
74 given_hex[HMAC_HEX_LEN] = '\0';
75
76 *dot_out = dot;
77 return NGX_OK;
78}

80/*
81 * WHAT: Split the cookie's post-HMAC fields and select the HMAC key.
82 * WHY: In multi-user mode a SECOND '.' separates the timestamp from the
83 * username: "<hex>.<ts>.<username>"; the per-user password hash becomes
84 * the HMAC key, so the cookie is verifiable only against the same user
85 * it was issued for. In single-user mode everything after "<hex>." is
86 * the timestamp and the configured password is the key.
87 * HOW: Locate dot2 (multi-user), bound-check and copy the username, resolve
88 * the user; fill ps->{username,username_len,ts_len,key}. Structural
89 * anomalies and unknown users return NGX_HTTP_UNAUTHORIZED (audited
90 * where the original behavior audited).
91 */
92static ngx_int_t
93dash_auth_resolve_key(const ngx_http_brix_dashboard_loc_conf_t *conf,
94 u_char *cookie_val, size_t cookie_val_len, const u_char *dot,
95 dash_cookie_parse_t *ps)
96{
97 u_char *dot2 = NULL;
98 size_t i;

100
101 if (!dashboard_users_enabled(conf)) {
102 ps->ts_len = cookie_val_len - HMAC_HEX_LEN - 1;
103 ps->key = conf->password;
104 return NGX_OK;
105 }
106
107 for (i = (size_t) (dot - cookie_val) + 1; i < cookie_val_len; i++) {
108 if (cookie_val[i] == '.') {
109 dot2 = cookie_val + i;
110 break;
111 }
112 }
113
114 if (dot2 == NULL) {
115 brix_dashboard_event_add(BRIX_DASH_EVENT_AUTH, 0,

118 return NGX_HTTP_UNAUTHORIZED;

120

130 if (user == NULL) {

132 NGX_HTTP_UNAUTHORIZED,
133 "dashboard auth unknown user", NULL);
134 return NGX_HTTP_UNAUTHORIZED;
135 }
136 ps->key = user->password_hash;
137 return NGX_OK;

140/*
141 * WHAT: Extract, parse, and freshness-check the cookie timestamp field.
142 * WHY: Cookies older than session_ttl are expired; cookies dated more than
143 * 60s in the future (clock skew tolerance) are rejected to bound replay
144 * of a forged future timestamp.
145 * HOW: Copy [dot+1, dot+1+ts_len) into ts_str (NUL-terminated), strtol it,
146 * reject non-positive or trailing-garbage values, then apply the TTL
147 * window. Each rejection emits the same audit event as before.
148 */
149static ngx_int_t
150dash_auth_check_timestamp(const u_char *dot, size_t ts_len,
151 ngx_uint_t session_ttl, char ts_str[TIMESTAMP_MAX + 1])
152{
153 char *endptr;
154 long ts;
155 time_t now;
156
157 /* Extract and null-terminate the timestamp. */
158 if (ts_len == 0 || ts_len > TIMESTAMP_MAX) {
159 brix_dashboard_event_add(BRIX_DASH_EVENT_AUTH, 0,
160 NGX_HTTP_UNAUTHORIZED,
161 "dashboard auth bad timestamp", NULL);
162 return NGX_HTTP_UNAUTHORIZED;
163 }
164 ngx_memcpy(ts_str, dot + 1, ts_len);
165 ts_str[ts_len] = '\0';
166
167 endptr = NULL;
168 ts = strtol(ts_str, &endptr, 10);
169 if (ts <= 0 || endptr == NULL || *endptr != '\0') {
170 brix_dashboard_event_add(BRIX_DASH_EVENT_AUTH, 0,
171 NGX_HTTP_UNAUTHORIZED,
172 "dashboard auth bad timestamp", NULL);
173 return NGX_HTTP_UNAUTHORIZED;
174 }
175
176 now = time(NULL);
177 if ((long) now - ts > (long) session_ttl || ts > (long) now + 60) {
178 brix_dashboard_event_add(BRIX_DASH_EVENT_AUTH, 0,
179 NGX_HTTP_UNAUTHORIZED,
180 "dashboard auth expired cookie", NULL);
181 return NGX_HTTP_UNAUTHORIZED;
182 }
183
184 return NGX_OK;
185}

187/*
188 * WHAT: Recompute the session HMAC over the parsed cookie fields and compare
189 * it against the presented signature.
190 * WHY: This is the actual proof of possession: only a holder of the HMAC key
191 * (configured password / per-user hash) can mint a valid signature.
192 * HOW: dashboard_cookie_hmac() re-derives expected_hex for the active auth
193 * mode; CRYPTO_memcmp never short-circuits on the first differing byte,
194 * which would leak how much of the signature an attacker has guessed.
195 */
196static ngx_int_t
197dash_auth_verify_hmac(const dash_cookie_parse_t *ps, const char *ts_str,
198 const char given_hex[HMAC_HEX_LEN + 1], ngx_uint_t multi_user)
199{
200 char expected_hex[HMAC_HEX_LEN + 1];
201
202 if (dashboard_cookie_hmac(&ps->key, ts_str, ps->ts_len,
203 multi_user ? ps->username : NULL,
204 ps->username_len, expected_hex) != NGX_OK)
205 {
206 return NGX_HTTP_UNAUTHORIZED;
207 }
208
209 if (CRYPTO_memcmp(expected_hex, given_hex, HMAC_HEX_LEN) != 0) {
210 brix_dashboard_event_add(BRIX_DASH_EVENT_AUTH, 0,
211 NGX_HTTP_UNAUTHORIZED,
212 "dashboard auth bad signature", NULL);
213 return NGX_HTTP_UNAUTHORIZED;
214 }
215
216 return NGX_OK;
217}

233 if (conf->password.len == 0 && !dashboard_users_enabled(conf)) {

235 return NGX_OK;
236 }
237
238 /* Cookie layout (see file header): "<64-hex HMAC>.<ts>" in single-user mode,
239 * "<64-hex HMAC>.<ts>.<username>" in multi-user mode. The parse below
240 * locates the dot(s), validates each field, then recomputes and compares the
241 * HMAC. Any structural anomaly is treated as unauthorized (fail closed). */
242
243 if (dashboard_find_cookie(r, "xrd_dashboard", 13, &cookie_val,
244 &cookie_val_len) != NGX_OK)
245 {
246 /* A missing cookie is the normal case for an anonymous viewer; the
247 * read-API caller passes suppress_missing_cookie=1 so anonymous polls do
248 * not spam the audit ring. The config-download and other sensitive
249 * callers pass 0, so unauthenticated attempts there are always audited.
250 * Present-but-invalid cookies (malformed/expired/forged, above helpers)
251 * are always logged regardless. */
252 if (!suppress_missing_cookie) {
253 brix_dashboard_event_add(BRIX_DASH_EVENT_AUTH, 0,
254 NGX_HTTP_UNAUTHORIZED,
255 "dashboard auth missing cookie", NULL);
256 }
257 return NGX_HTTP_UNAUTHORIZED;
258 }
259

261 if (rc != NGX_OK) {
262 return rc;
263 }
264

266 rc = dash_auth_resolve_key(conf, cookie_val, cookie_val_len, dot, &ps);
267 if (rc != NGX_OK) {
268 return rc;
269 }
270
271 rc = dash_auth_check_timestamp(dot, ps.ts_len, conf->session_ttl, ts_str);
272 if (rc != NGX_OK) {
273 return rc;
274 }
275
