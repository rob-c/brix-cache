42typedef enum {

48/* RFC-3230 Digest tokens / Content-MD5 we can recompute. `b64` = the value is
49 * base64 (md5/sha per RFC 3230 + RFC 1864); otherwise it is lowercase hex (the
50 * WLCG/dCache convention for adler32/crc32). */
51static const struct {
52 const char *tok; size_t toklen; const char *alg; int b64;
53} webdav_digest_tokens[] = {
54 { "md5", 3, "md5", 1 },
55 { "sha-256", 7, "sha256", 1 },
56 { "sha256", 6, "sha256", 1 },
57 { "adler32", 7, "adler32", 0 },
58 { "crc32c", 6, "crc32c", 0 },
59 { "crc32", 5, "crc32", 0 },
60};

62/* Two hex checksums are equal ignoring case and leading-zero padding (a client
63 * may send an un-padded adler32 while our compute zero-pads to the alg width). */
64static int
65webdav_hex_norm_equal(const char *a, const char *b)
66{
67 while (*a == '0') { a++; }
68 while (*b == '0') { b++; }
69 for (; *a != '\0' && *b != '\0'; a++, b++) {
70 if (ngx_tolower((u_char) *a) != ngx_tolower((u_char) *b)) {
71 return 0;
72 }
73 }
74 return *a == '\0' && *b == '\0';
75}

77/* Normalise a digest header value into lowercase hex in out[]. A base64 value is
78 * decoded then hex-encoded; a hex value is copied lowercased. */
79static ngx_int_t
80webdav_digest_value_hex(ngx_http_request_t *r, const u_char *val, size_t vlen,
81 int is_b64, char *out, size_t outsz)
82{
83 if (vlen == 0) {
84 return NGX_ERROR;
85 }
86 if (is_b64) {
87 ngx_str_t src, dst;
88 src.data = (u_char *) val;
89 src.len = vlen;
90 dst.len = ngx_base64_decoded_length(vlen);
91 dst.data = ngx_pnalloc(r->pool, dst.len ? dst.len : 1);
92 if (dst.data == NULL || ngx_decode_base64(&dst, &src) != NGX_OK) {
93 return NGX_ERROR;
94 }
95 if (dst.len == 0 || dst.len * 2 + 1 > outsz) {
96 return NGX_ERROR;
97 }
98 ngx_hex_dump((u_char *) out, dst.data, dst.len);
99 out[dst.len * 2] = '\0';
100 return NGX_OK;
101 }
102 if (vlen + 1 > outsz) {
103 return NGX_ERROR;
104 }
105 {
106 size_t i;
107 for (i = 0; i < vlen; i++) {
108 u_char c = val[i];
109 if (!((c >= '0' && c <= '9') || (c >= 'a' && c <= 'f')
110 || (c >= 'A' && c <= 'F')))
111 {
112 return NGX_ERROR; /* not a hex digit — malformed */
113 }
114 out[i] = (char) ngx_tolower(c);
115 }
116 out[vlen] = '\0';
117 }
118 return NGX_OK;
119}

121/* Strip leading and trailing linear whitespace from a [*s, *s+*len) slice. */
122static void
123webdav_tok_trim(u_char **s, size_t *len)
124{
125 u_char *p = *s;
126 size_t n = *len;
127 while (n > 0 && (*p == ' ' || *p == '\t')) { p++; n--; }
128 while (n > 0 && (p[n - 1] == ' ' || p[n - 1] == '\t')) { n--; }
129 *s = p;
130 *len = n;
131}

133/* Match a trimmed Digest token=value pair against the supported-alg table.
134 * Returns 1 if the token names a supported alg — *out is then FOUND (with *alg
135 * and exp_hex filled) or BAD (value not valid hex) — else 0 to keep scanning. */
136static int
137webdav_digest_match(ngx_http_request_t *r, u_char *tok, size_t tlen,
138 u_char *v, size_t vlen, const char **alg, char *exp_hex, size_t exp_sz,
139 webdav_digest_kind_t *out)
140{
141 size_t ntok = sizeof(webdav_digest_tokens)
142 / sizeof(webdav_digest_tokens[0]);
143 size_t i;
144
145 for (i = 0; i < ntok; i++) {
146 if (tlen == webdav_digest_tokens[i].toklen
147 && ngx_strncasecmp(tok,
148 (u_char *) webdav_digest_tokens[i].tok, tlen) == 0)
149 {
150 if (webdav_digest_value_hex(r, v, vlen,
151 webdav_digest_tokens[i].b64, exp_hex, exp_sz) != NGX_OK)
152 {
153 *out = WEBDAV_DIGEST_BAD;
154 } else {
155 *alg = webdav_digest_tokens[i].alg;
156 *out = WEBDAV_DIGEST_FOUND;
157 }
158 return 1;
159 }
160 }
161 return 0;
162}

164/* Scan a Digest: header value (comma-separated alg=value pairs) for the first
165 * supported alg. Returns 1 with *out set (FOUND/BAD) once a token matches, or 0
166 * if no listed token is supported (caller falls through to Content-MD5). */
167static int
168webdav_digest_scan(ngx_http_request_t *r, ngx_table_elt_t *h,
169 const char **alg, char *exp_hex, size_t exp_sz, webdav_digest_kind_t *out)
170{
171 u_char *p = h->value.data, *end = p + h->value.len;
172
173 while (p < end) {
174 u_char *comma = ngx_strlchr(p, end, ',');
175 u_char *iend = comma ? comma : end;
176 u_char *eq = ngx_strlchr(p, iend, '=');
177 if (eq != NULL) {
178 u_char *tok = p, *v = eq + 1;
179 size_t tlen = eq - p, vlen = iend - (eq + 1);
180 webdav_tok_trim(&tok, &tlen);
181 webdav_tok_trim(&v, &vlen);
182 if (webdav_digest_match(r, tok, tlen, v, vlen, alg,
183 exp_hex, exp_sz, out))
184 {
185 return 1;
186 }
187 }
188 p = comma ? comma + 1 : end;
189 }
190 return 0;
191}

193/* Pick the first supported digest from Digest: (RFC 3230) then Content-MD5:. On
194 * WEBDAV_DIGEST_FOUND, *alg points at a static alg name and exp_hex holds the
195 * client-asserted value as lowercase hex. Unsupported Digest tokens are skipped
196 * (best-effort interop), so a Digest that names only algs we cannot compute reads
197 * as NONE — require_digest then decides whether that is acceptable. */
198static webdav_digest_kind_t
199webdav_digest_select(ngx_http_request_t *r, const char **alg,
200 char *exp_hex, size_t exp_sz)
201{
202 ngx_table_elt_t *h;
203 webdav_digest_kind_t kind;
204
205 h = brix_http_find_header(r, "Digest", sizeof("Digest") - 1);
206 if (h != NULL && h->value.len > 0
207 && webdav_digest_scan(r, h, alg, exp_hex, exp_sz, &kind))
208 {
209 return kind;
210 }
211
212 h = brix_http_find_header(r, "Content-MD5", sizeof("Content-MD5") - 1);
213 if (h != NULL && h->value.len > 0) {
214 if (webdav_digest_value_hex(r, h->value.data, h->value.len, 1,
215 exp_hex, exp_sz) != NGX_OK)
216 {
217 return WEBDAV_DIGEST_BAD;
218 }
219 *alg = "md5";
220 return WEBDAV_DIGEST_FOUND;
221 }
222
223 return WEBDAV_DIGEST_NONE;
224}

226/*
227 * webdav_put_verify_ingest_digest — verify a client-asserted body digest over
228 * the fully-staged bytes, BEFORE the commit publishes them.
229 *
230 * A PUT may carry an end-to-end digest the client computed over what it sent
231 * (RFC-3230 `Digest:` — the WLCG/XrdHttp convention — or legacy `Content-MD5:`).
232 * The writer otherwise streams and commits whatever lands, so a byte flipped in
233 * flight past the TCP checksum is stored silently. When a usable digest is
234 * present we recompute it over the staged temp fd and refuse the commit (400) on
235 * mismatch — the client told us what it sent, so we can and must check it. With
236 * brix_webdav_require_digest on, a PUT that carries no usable digest is also
237 * refused, for deployments that decline writes they cannot verify.
238 *
239 * Returns NGX_OK to proceed to commit, or an HTTP status (>=400) to reject.
240 *
241 * Scope: the staged POSIX temp fd (brix_vfs_writer_fd). Coded bodies
242 * (Content-Encoding) are skipped — the stored plaintext would not match a digest
243 * over the encoded stream, matching the verify-on-write contract. A driver
244 * object target (S3) exposes no fd here and is verified on its own ingest path.
245 * The recompute is synchronous on the event thread (like §8.3 checksum-on-write);
246 * it only runs when a digest is asserted or required.
247 */
248ngx_int_t
249webdav_put_verify_ingest_digest(ngx_http_request_t *r,
250 ngx_http_brix_webdav_loc_conf_t *conf, brix_vfs_writer_t *writer,
251 const char *path)
252{
253 char exp_hex[129];
254 const char *alg = NULL;
255 webdav_digest_kind_t kind;
256 ngx_fd_t fd;
257 ngx_table_elt_t *ce;
258 brix_integrity_info_t info;
259 brix_integrity_opts_t o;
260
261 /* A coded body is stored decoded; a digest over the encoded stream cannot be
262 * checked against it — skip (same limitation as verify-on-write). */
263 ce = brix_http_find_header(r, "Content-Encoding", sizeof("Content-Encoding") - 1);
264 if (ce != NULL && ce->value.len > 0) {
265 return NGX_OK;
266 }
267
268 kind = webdav_digest_select(r, &alg, exp_hex, sizeof(exp_hex));
269 if (kind == WEBDAV_DIGEST_BAD) {
270 return NGX_HTTP_BAD_REQUEST; /* known alg, bad value */
271 }
272 if (kind == WEBDAV_DIGEST_NONE) {
273 return conf->require_digest ? NGX_HTTP_BAD_REQUEST : NGX_OK;
274 }
275
276 fd = brix_vfs_writer_fd(writer);
277 if (fd == NGX_INVALID_FILE) {
278 /* Driver-backed object target — no local fd to hash here. */
279 return conf->require_digest ? NGX_HTTP_BAD_REQUEST : NGX_OK;
280 }
281
282 ngx_memzero(&o, sizeof(o));
283 o.require_regular_file = 1; /* hash the staged bytes fresh (no xattr cache) */
284 {
285 char procpath[32];
286 ngx_int_t irc;
287 int rfd;
288
289 /* The staged temp is opened write-only (staged_open_posix, O_WRONLY), and
290 * has no final path yet (pre-commit), so it cannot be pread directly nor
291 * reopened by name. Re-open the same open file description read-only via
292 * /proc/self/fd/<fd> — valid on Linux for an unlinked/O_TMPFILE staged
293 * temp — and hash the already-written bytes over the read handle. */
294 (void) ngx_snprintf((u_char *) procpath, sizeof(procpath),
295 "/proc/self/fd/%d%Z", (int) fd);
296 rfd = open(procpath, O_RDONLY | O_CLOEXEC); /* vfs-seam-allow: read-back of staged PUT bytes for pre-commit ingest-digest verify (no final path yet) */
297 if (rfd < 0) {
298 /* Cannot obtain a readable view of what we must verify: fail closed. */
299 return NGX_HTTP_BAD_REQUEST;
300 }
301 irc = brix_integrity_get_fd(r->connection->log, rfd, NULL, path,
302 alg, &o, &info);
303 (void) close(rfd);
304 if (irc != NGX_OK) {
305 /* Cannot compute the digest we were asked to verify: fail closed. */
306 return NGX_HTTP_BAD_REQUEST;
307 }
308 }
309
310 if (!webdav_hex_norm_equal(exp_hex, info.hex)) {
311 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
312 "brix_webdav: PUT ingest %s digest mismatch (asserted=%s got=%s)",
313 alg, exp_hex, info.hex);
314 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, 0,
315 "brix_webdav: ingest digest mismatch rejects PUT for: \"%s\"", path);
316 return NGX_HTTP_BAD_REQUEST;
317 }
318 return NGX_OK;
319}

321/*
322 * webdav_put_persist_checksums — §8.3 checksum-on-ingest.
323 *
324 * After a successful PUT commit, when brix_webdav_checksum_on_write names one or
325 * more algorithms, compute+persist each digest on the freshly committed file via
326 * the integrity service (xattr, or .cks sidecar fallback §8.2). Best-effort and
327 * opt-in (default off): any failure is ignored — the digest simply stays lazy.
328 * Synchronous (the operator opts in knowing the cost); large-file async offload is
329 * a future refinement.
330 */
331void
332webdav_put_persist_checksums(ngx_http_request_t *r, const char *path)
333{
334 ngx_http_brix_webdav_loc_conf_t *conf;
335 char algs[256];
336 char *save = NULL, *tok;
337 int fd;
338
339 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
340 if (conf->checksum_on_write.len == 0
341 || brix_str_cbuf(algs, sizeof(algs), &conf->checksum_on_write) == NULL)
342 {
343 return;
344 }
345
346 /* Re-open the just-committed export file through the VFS (confined beneath
347 * the export root, impersonation-aware) to compute its digests. */
348 fd = brix_vfs_open_fd(r->connection->log, conf->common.root_canon, path,
349 O_RDONLY, 0);
350 if (fd < 0) {
351 return;
352 }
353 for (tok = strtok_r(algs, ", \t", &save); tok != NULL;
354 tok = strtok_r(NULL, ", \t", &save))
355 {
356 brix_integrity_info_t info;
357 brix_integrity_opts_t o;
358 ngx_memzero(&o, sizeof(o));
359 o.allow_xattr_cache = 1;
360 o.update_xattr_cache = 1;
361 o.require_regular_file = 1;
362 (void) brix_integrity_get_fd(r->connection->log, fd, NULL, path, tok,
363 &o, &info);
364 }
365 (void) close(fd);
366}

368static void
369webdav_put_aio_thread(void *data, ngx_log_t *log)
370{
371 webdav_put_aio_t *t = data;
372
373 (void) log;
374
375 t->io_errno = 0;
376
377 /*
378 * Phase 31 W2: stream the body straight from nginx's own request buffers
379 * into the write session — no full-body contiguous copy. The body for this
380 * path is all in-memory bufs anchored in r->pool (the caller gates spooled
381 * bodies to the synchronous streaming path), so they are stable for the
382 * lifetime of the request (held alive by r->main->count++). The writer does
383 * only pwrite(2)/staged-write here (the commit + verify read-back are deferred
384 * to the done handler on the event thread), so it is safe on the thread pool —
385 * no nginx pool allocation or event-loop calls. The writer dispatches the
386 * POSIX-temp-fd vs driver-object path internally.
387 */
388 if (brix_http_body_write_to_writer(t->r, t->writer) != NGX_OK) {
389 t->io_errno = errno;
390 t->nwritten = -1;
391 return;
392 }
393
394 t->nwritten = (ssize_t) t->len;
395}

397static void
398webdav_put_aio_done(ngx_event_t *ev)
399{
400 ngx_thread_task_t *task = ev->data;
401 webdav_put_aio_t *t = task->ctx;
402 ngx_http_request_t *r = t->r;
403 ngx_int_t status;
404
405 /* Balance the r->main->count++ in webdav_handle_put_body that keeps the
406 * request alive across the async thread dispatch. */
407 r->main->count--;
408
409 if (t->nwritten < 0 || (size_t) t->nwritten < t->len) {
410
411 brix_log_safe_path(r->connection->log, NGX_LOG_ERR,
412 (ngx_uint_t) t->io_errno,
413 "brix_webdav: async write() failed for: \"%s\"",
414 t->path);
415 brix_dashboard_http_error(r, "webdav PUT async write failed");
416 brix_dashboard_http_finish(r);
417 /* Abort the session (close + unlink temp) — the final path is untouched. */
418 brix_vfs_writer_abort(t->writer);
419 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
420 return;
421 }
422
423 /* Client->server body integrity: verify any asserted ingest digest over the
424 * fully-staged bytes before publishing (or refuse a digest-less PUT under
425 * require_digest). A rejection aborts the session — the final path is never
426 * touched, so poisoned bytes never land. */
427 {
428 ngx_int_t vrc = webdav_put_verify_ingest_digest(r,
429 ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module),
430 t->writer, (const char *) t->path);
431 if (vrc != NGX_OK) {
432 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, 0,
433 "brix_webdav: async PUT ingest digest rejected for: \"%s\"",
434 t->path);
435 brix_dashboard_http_error(r, "webdav PUT ingest digest mismatch");
436 brix_dashboard_http_finish(r);
437 brix_vfs_writer_abort(t->writer);
438 webdav_metrics_finalize_request(r, (ngx_uint_t) vrc);
439 return;
440 }
441 }
442
443 /* Atomically publish the completed temp onto the final path (folding the
444 * verify read-back when the export opts in; a mismatch unlinks + fails). */
445 if (brix_vfs_writer_commit(t->writer) != NGX_OK) {
446 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, ngx_errno,
447 "brix_webdav: async staged commit failed for: "
448 "\"%s\"", t->path);
449 /* A direct WebDAV PUT is a STAGE-class ingest — record the failed
450 * publish in the unified ledger (mirrors webdav_put_commit). */
451 brix_xfer_finish(BRIX_XFER_STAGE, "in", (const char *) t->path, NULL,
452 t->len, BRIX_XFER_COMMIT_ERR, ngx_errno,
453 r->connection->log);
454 brix_dashboard_http_error(r, "webdav PUT staged commit failed");
455 brix_dashboard_http_finish(r);
456 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
457 return;
458 }
459
460 brix_dashboard_http_add(r, (ngx_atomic_int_t) t->len);
461 brix_dashboard_http_finish(r);
462
463 /* Unified transfer ledger: the async PUT publish (mirrors the sync path). */
464 brix_xfer_finish(BRIX_XFER_STAGE, "in", (const char *) t->path, NULL,
465 t->len, BRIX_XFER_OK, 0, r->connection->log);
466
467 webdav_put_persist_checksums(r, (const char *) t->path); /* §8.3 */
468
469 status = t->created ? NGX_HTTP_CREATED : NGX_HTTP_NO_CONTENT;
470 webdav_send_status_only(r, (ngx_uint_t) status);
471}

473/*
474 * Shared state for the body-write phase (thread offload, codec select, and the

478 * write parameters.

480typedef struct {

482 brix_vfs_writer_t *writer; /* open write session */
483 brix_http_body_summary_t body_summary; /* body shape/byte count */
484 brix_codec_id_t put_codec; /* Content-Encoding codec (or id) */
485 int created; /* 1 = new file (→ 201 on commit) */
486} webdav_put_body_ctx_t;

488/*
489 * webdav_put_try_threaded — offload an in-memory body write to the thread pool.
490 *
491 * WHAT: When the body is entirely in memory, non-empty, identity-coded, and a
492 * thread pool is configured, allocates the AIO task, transfers staged
493 * ownership into it, and posts it — returning WEBDAV_PUT_DONE (the done
494 * handler commits/aborts and sends the response). Returns
495 * WEBDAV_PUT_CONTINUE only when the threaded path does not apply (caller falls
496 * through to the synchronous write); a task alloc/post failure aborts the
497 * staged temp, sends 500, and returns WEBDAV_PUT_DONE.
498 *
499 * WHY: Isolates the async offload decision + task lifecycle so the synchronous
500 * fallback in the orchestrator stays linear. The r->main->count++ that keeps
501 * the request alive across the dispatch and the ownership transfer are
502 * preserved exactly.
503 *
504 * HOW:
505 * 1. If the threaded preconditions do not hold, return CONTINUE.
506 * 2. Allocate the task; on failure abort staged, finalize 500, return DONE.
507 * 3. Populate the task ctx (r, staged, len, created, path).
508 * 4. Bind + post; on post failure abort staged, finalize 500, return DONE.
509 * 5. On success bump the threaded metric, r->main->count++, return DONE.
510 */
511static webdav_put_step_t
512webdav_put_try_threaded(ngx_http_request_t *r,
513 ngx_http_brix_webdav_loc_conf_t *conf, const webdav_put_body_ctx_t *bctx)
514{
515 ngx_thread_task_t *task;
516 webdav_put_aio_t *t;
517
518 if (bctx->body_summary.has_spooled || bctx->body_summary.bytes == 0
519 || bctx->put_codec != BRIX_CODEC_IDENTITY
520 || conf->common.thread_pool == NULL)
521 {
522 return WEBDAV_PUT_CONTINUE;
523 }
524
525 task = ngx_thread_task_alloc(r->pool, sizeof(webdav_put_aio_t));
526 if (task == NULL) {
527 brix_dashboard_http_error(r, "webdav PUT task allocation failed");
528 brix_dashboard_http_finish(r);
529 brix_vfs_writer_abort(bctx->writer);
530 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
531 return WEBDAV_PUT_DONE;
532 }
533
534 t = task->ctx;
535 t->r = r;
536 t->writer = bctx->writer; /* transfer session ownership to the task */
537 t->len = bctx->body_summary.bytes;
538 t->created = bctx->created;
539 ngx_cpystrn((u_char *) t->path, (u_char *) bctx->path, sizeof(t->path));
540
541 /*
542 * Phase 31 W2: no full-body collection — the thread streams the
543 * in-memory body bufs straight to the staged fd (see
544 * webdav_put_aio_thread); the done handler commits/aborts.
545 */
546
547 brix_task_bind(task, webdav_put_aio_thread, webdav_put_aio_done);
548
549 if (ngx_thread_task_post(conf->common.thread_pool, task) != NGX_OK) {
550 brix_dashboard_http_error(r, "webdav PUT task post failed");
551 brix_dashboard_http_finish(r);
552 brix_vfs_writer_abort(t->writer);
553 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
554 return WEBDAV_PUT_DONE;
555 }

557 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_THREADED]);
558 r->main->count++;
559 return WEBDAV_PUT_DONE;

562/*
563 * webdav_put_select_codec — pick the body's Content-Encoding codec.
564 *
565 * WHAT: Reads the Content-Encoding header; sets `bctx->put_codec` to the decoded
566 * codec id and returns WEBDAV_PUT_CONTINUE, or returns WEBDAV_PUT_DONE after
567 * aborting the staged temp and sending 415 for an unknown/not-compiled-in
568 * coding. A missing header leaves `bctx->put_codec` as identity.
569 *
570 * WHY: Never store undecoded bytes for a coding we cannot reverse (would
571 * silently corrupt the object) — this is the single gate enforcing that, kept
572 * out of the write-dispatch body.
573 *
574 * HOW:
575 * 1. Find Content-Encoding; absent/empty → CONTINUE (identity).
576 * 2. Resolve the codec descriptor; NULL or unavailable → abort staged,
577 * finalize 415, return DONE.
578 * 3. Otherwise record the codec id and return CONTINUE.
579 */
580static webdav_put_step_t
581webdav_put_select_codec(ngx_http_request_t *r, webdav_put_body_ctx_t *bctx)
582{
583 ngx_table_elt_t *ce = brix_http_find_header(
584 r, "Content-Encoding", sizeof("Content-Encoding") - 1);
585
586 bctx->put_codec = BRIX_CODEC_IDENTITY;
587
588 if (ce != NULL && ce->value.len > 0) {
589 const brix_codec_desc_t *d = brix_codec_by_http_token(
590 (const char *) ce->value.data, ce->value.len);
591 if (d == NULL || !d->available) {
592 /* Unknown or not-compiled-in Content-Encoding: never store the
593 * undecoded bytes (would silently corrupt the object). 415. */
594 brix_dashboard_http_error(r,
595 "webdav PUT unsupported Content-Encoding");
596 brix_dashboard_http_finish(r);
597 brix_vfs_writer_abort(bctx->writer);
598 webdav_metrics_finalize_request(r,
599 NGX_HTTP_UNSUPPORTED_MEDIA_TYPE);
600 return WEBDAV_PUT_DONE;
601 }
602 bctx->put_codec = d->id;
603
604 /* A non-identity Content-Encoding declares the body IS an encoded
605 * stream, but no codec emits a zero-byte stream (gzip/zstd/… all
606 * carry a header/trailer). An empty body with a real codec is
607 * therefore malformed — reject it (same contract as a truncated
608 * stream) instead of storing a spurious empty object. */
609 if (bctx->put_codec != BRIX_CODEC_IDENTITY
610 && !bctx->body_summary.has_spooled && bctx->body_summary.bytes == 0)
611 {
612 brix_dashboard_http_error(r,
613 "webdav PUT empty body with Content-Encoding");
614 brix_dashboard_http_finish(r);
615 brix_vfs_writer_abort(bctx->writer);
616 webdav_metrics_finalize_request(r,
617 NGX_HTTP_UNSUPPORTED_MEDIA_TYPE);
618 return WEBDAV_PUT_DONE;
619 }
620 }
621
622 return WEBDAV_PUT_CONTINUE;
623}

625/*
626 * webdav_put_write_sync — synchronously stream/decode the body to the staged fd.
627 *
628 * WHAT: Bumps the per-body-shape metric, then writes the body to the staged
629 * target — decode-to-fd for a coded body, staged-driver stream for object
630 * backends, or write-to-fd for a POSIX temp — and accounts the bytes.
631 * Returns WEBDAV_PUT_CONTINUE on success, or WEBDAV_PUT_DONE after aborting
632 * the staged temp and finalizing the precise failure status (415/413/400/501/
633 * 500) on a write error.
634 *
635 * WHY: Confines the three-way write dispatch and its abort-on-failure invariant
636 * (a failed write must never leave a readable partial at the final path) to
637 * one place, mirroring the thread path's error handling.
638 *
639 * HOW:
640 * 1. Increment the SPOOLED/MEMORY/EMPTY body-shape metric.
641 * 2. Coded body: NOSYS/501 on a driver target (no kernel fd), else
642 * brix_http_body_decode_to_fd (reports its own precise status).
643 * 3. Identity body: staged-driver stream or write-to-fd.
644 * 4. On write failure abort staged, finalize the reported status, return DONE.
645 * 5. On success account the bytes via brix_dashboard_http_add, return CONTINUE.
646 */
647static webdav_put_step_t
648webdav_put_write_sync(ngx_http_request_t *r, webdav_put_body_ctx_t *bctx)
649{
650 ngx_int_t wrc;
651 ngx_int_t decode_status = NGX_HTTP_INTERNAL_SERVER_ERROR;
652
653 if (bctx->body_summary.has_spooled) {
654 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_SPOOLED]);
655 } else if (bctx->body_summary.has_memory) {
656 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_MEMORY]);
657 } else {
658 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_EMPTY]);
659 }
660
661 if (bctx->put_codec != BRIX_CODEC_IDENTITY) {
662 /* A Content-Encoding decode writes plaintext straight to the session's
663 * kernel temp fd (the decode engine owns the streaming + bomb guard);
664 * a driver-backed object exposes no fd (NGX_INVALID_FILE) →
665 * decode-to-object is a follow-up. These bytes bypass the writer's CRC
666 * accumulator, so a verifying session leaves them unverified (its commit
667 * read-back is a no-op when nothing went through the accumulator) —
668 * verify-on-write does not cover coded bodies. */
669 ngx_fd_t wfd = brix_vfs_writer_fd(bctx->writer);
670 if (wfd == NGX_INVALID_FILE) {
671 errno = ENOSYS;
672 wrc = NGX_ERROR;
673 decode_status = NGX_HTTP_NOT_IMPLEMENTED;
674 } else {
675 wrc = brix_http_body_decode_to_fd(r, wfd, bctx->path,
676 bctx->put_codec,
677 BRIX_DECODE_MAX_OUTPUT,
678 &bctx->body_summary,
679 &decode_status);
680 }
681 } else {
682 /* Identity body: one common verified-write call — the writer dispatches
683 * memory bufs, spooled temp-fd bufs, and driver-object targets. */
684 wrc = brix_http_body_write_to_writer(r, bctx->writer);
685 }
686 if (wrc != NGX_OK) {
687 brix_dashboard_http_error(r, "webdav PUT body write failed");
688 brix_dashboard_http_finish(r);
689 /* Abort the session — the final path is never touched, so a failed
690 * write (e.g. a corrupt/over-large Content-Encoding that fails to
691 * decode) can never leave a readable partial object. The decode path
692 * reports the precise status (413 bomb / 400 corrupt). */
693 brix_vfs_writer_abort(bctx->writer);
694 webdav_metrics_finalize_request(r, decode_status);
695 return WEBDAV_PUT_DONE;
696 }
697
698 brix_dashboard_http_add(r, (ngx_atomic_int_t) bctx->body_summary.bytes);
699 return WEBDAV_PUT_CONTINUE;
700}

702/*
703 * webdav_put_stream_body — write the request body (async or sync) to staged.
704 *
705 * WHAT: Summarizes the body, starts dashboard tracking, and drives the body to
706 * the staged temp: for a request with a body it tries the thread-pool offload
707 * then falls through the codec-select + synchronous write; an empty PUT just
708 * records the EMPTY metric. Returns WEBDAV_PUT_CONTINUE when the caller
709 * should proceed to commit synchronously, or WEBDAV_PUT_DONE when the body
710 * phase has already answered the request (threaded dispatch, or a handled
711 * error).
712 *
713 * WHY: Groups the whole "get the bytes onto the staged fd" concern — including
714 * the async/sync/empty split — behind one call so the orchestrator commits
715 * only what the sync/empty paths leave staged. Behavior (metrics, statuses,
716 * ownership) is unchanged from the prior inline block.
717 *
718 * HOW:
719 * 1. No request body → start dashboard(0), EMPTY metric, CONTINUE.
720 * 2. Body present → summarize (failure aborts + 500 → DONE), start dashboard,
721 * add bytes_rx metric.
722 * 3. Try the threaded offload; if it took over or failed, return DONE.
723 * 4. Select the codec (415 → DONE), then synchronously write (error → DONE).
724 * 5. CONTINUE — the staged temp holds the full body, ready to commit.
725 */
726webdav_put_step_t
727webdav_put_stream_body(ngx_http_request_t *r,
728 ngx_http_brix_webdav_loc_conf_t *conf, const char *path,
729 brix_vfs_writer_t *writer, int created)
730{
731 webdav_put_body_ctx_t bctx;
732
733 if (r->request_body == NULL) {
734 webdav_put_start_dashboard(r, path, 0);
735 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_EMPTY]);
736 return WEBDAV_PUT_CONTINUE;
737 }
738
739 ngx_memzero(&bctx, sizeof(bctx));
740 bctx.path = path;
741 bctx.writer = writer;
742 bctx.created = created;
743 bctx.put_codec = BRIX_CODEC_IDENTITY;
744
745 if (brix_http_body_summary(r, &bctx.body_summary) != NGX_OK) {
746 brix_vfs_writer_abort(writer);
747 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
748 return WEBDAV_PUT_DONE;
749 }
750
751 webdav_put_start_dashboard(r, path, (int64_t) bctx.body_summary.bytes);
752 BRIX_WEBDAV_METRIC_ADD(bytes_rx_total, bctx.body_summary.bytes);
753
754 if (webdav_put_select_codec(r, &bctx) == WEBDAV_PUT_DONE) {
755 return WEBDAV_PUT_DONE;
756 }
757
758 if (webdav_put_try_threaded(r, conf, &bctx) == WEBDAV_PUT_DONE) {
759 return WEBDAV_PUT_DONE;
760 }
761
762 return webdav_put_write_sync(r, &bctx);
763}
