39/*
40 * webdav_put_persist_checksums — §8.3 checksum-on-ingest.
41 *
42 * After a successful PUT commit, when brix_webdav_checksum_on_write names one or
43 * more algorithms, compute+persist each digest on the freshly committed file via
44 * the integrity service (xattr, or .cks sidecar fallback §8.2). Best-effort and
45 * opt-in (default off): any failure is ignored — the digest simply stays lazy.
46 * Synchronous (the operator opts in knowing the cost); large-file async offload is
47 * a future refinement.
48 */
49void
50webdav_put_persist_checksums(ngx_http_request_t *r, const char *path)
51{
52 ngx_http_brix_webdav_loc_conf_t *conf;
53 char algs[256];
54 char *save = NULL, *tok;
55 int fd;
56
57 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
58 if (conf->checksum_on_write.len == 0
59 || brix_str_cbuf(algs, sizeof(algs), &conf->checksum_on_write) == NULL)
60 {
61 return;
62 }
63
64 /* Re-open the just-committed export file through the VFS (confined beneath
65 * the export root, impersonation-aware) to compute its digests. */
66 fd = brix_vfs_open_fd(r->connection->log, conf->common.root_canon, path,
67 O_RDONLY, 0);
68 if (fd < 0) {
69 return;
70 }
71 for (tok = strtok_r(algs, ", \t", &save); tok != NULL;
72 tok = strtok_r(NULL, ", \t", &save))
73 {
74 brix_integrity_info_t info;
75 brix_integrity_opts_t o;
76 ngx_memzero(&o, sizeof(o));
77 o.allow_xattr_cache = 1;
78 o.update_xattr_cache = 1;
79 o.require_regular_file = 1;
80 (void) brix_integrity_get_fd(r->connection->log, fd, NULL, path, tok,
81 &o, &info);
82 }
83 (void) close(fd);
84}

86static void
87webdav_put_aio_thread(void *data, ngx_log_t *log)
88{
89 webdav_put_aio_t *t = data;
90
91 (void) log;
92
93 t->io_errno = 0;
94
95 /*
96 * Phase 31 W2: stream the body straight from nginx's own request buffers
97 * into the write session — no full-body contiguous copy. The body for this
98 * path is all in-memory bufs anchored in r->pool (the caller gates spooled
99 * bodies to the synchronous streaming path), so they are stable for the
100 * lifetime of the request (held alive by r->main->count++). The writer does
101 * only pwrite(2)/staged-write here (the commit + verify read-back are deferred
102 * to the done handler on the event thread), so it is safe on the thread pool —
103 * no nginx pool allocation or event-loop calls. The writer dispatches the
104 * POSIX-temp-fd vs driver-object path internally.
105 */
106 if (brix_http_body_write_to_writer(t->r, t->writer) != NGX_OK) {
107 t->io_errno = errno;
108 t->nwritten = -1;
109 return;
110 }
111
112 t->nwritten = (ssize_t) t->len;
113}

115static void
116webdav_put_aio_done(ngx_event_t *ev)
117{
118 ngx_thread_task_t *task = ev->data;
119 webdav_put_aio_t *t = task->ctx;
120 ngx_http_request_t *r = t->r;
121 ngx_int_t status;
122
123 /* Balance the r->main->count++ in webdav_handle_put_body that keeps the
124 * request alive across the async thread dispatch. */
125 r->main->count--;
126
127 if (t->nwritten < 0 || (size_t) t->nwritten < t->len) {
128
129 brix_log_safe_path(r->connection->log, NGX_LOG_ERR,
130 (ngx_uint_t) t->io_errno,
131 "brix_webdav: async write() failed for: \"%s\"",
132 t->path);
133 brix_dashboard_http_error(r, "webdav PUT async write failed");
134 brix_dashboard_http_finish(r);
135 /* Abort the session (close + unlink temp) — the final path is untouched. */
136 brix_vfs_writer_abort(t->writer);
137 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
138 return;
139 }
140
141 /* Atomically publish the completed temp onto the final path (folding the
142 * verify read-back when the export opts in; a mismatch unlinks + fails). */
143 if (brix_vfs_writer_commit(t->writer) != NGX_OK) {
144 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, ngx_errno,
145 "brix_webdav: async staged commit failed for: "
146 "\"%s\"", t->path);
147 brix_dashboard_http_error(r, "webdav PUT staged commit failed");
148 brix_dashboard_http_finish(r);
149 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
150 return;
151 }
152
153 brix_dashboard_http_add(r, (ngx_atomic_int_t) t->len);
154 brix_dashboard_http_finish(r);
155
156 webdav_put_persist_checksums(r, (const char *) t->path); /* §8.3 */
157
158 status = t->created ? NGX_HTTP_CREATED : NGX_HTTP_NO_CONTENT;
159 webdav_send_status_only(r, (ngx_uint_t) status);
160}

177/*
178 * webdav_put_try_threaded — offload an in-memory body write to the thread pool.
179 *
180 * WHAT: When the body is entirely in memory, non-empty, identity-coded, and a
181 * thread pool is configured, allocates the AIO task, transfers staged
182 * ownership into it, and posts it — returning WEBDAV_PUT_DONE (the done
183 * handler commits/aborts and sends the response). Returns
184 * WEBDAV_PUT_CONTINUE only when the threaded path does not apply (caller falls
185 * through to the synchronous write); a task alloc/post failure aborts the
186 * staged temp, sends 500, and returns WEBDAV_PUT_DONE.
187 *
188 * WHY: Isolates the async offload decision + task lifecycle so the synchronous
189 * fallback in the orchestrator stays linear. The r->main->count++ that keeps
190 * the request alive across the dispatch and the ownership transfer are
191 * preserved exactly.
192 *
193 * HOW:
194 * 1. If the threaded preconditions do not hold, return CONTINUE.
195 * 2. Allocate the task; on failure abort staged, finalize 500, return DONE.
196 * 3. Populate the task ctx (r, staged, len, created, path).
197 * 4. Bind + post; on post failure abort staged, finalize 500, return DONE.
198 * 5. On success bump the threaded metric, r->main->count++, return DONE.
199 */
200static webdav_put_step_t
201webdav_put_try_threaded(ngx_http_request_t *r,
202 ngx_http_brix_webdav_loc_conf_t *conf, const webdav_put_body_ctx_t *bctx)
203{
204 ngx_thread_task_t *task;
205 webdav_put_aio_t *t;
206
207 if (bctx->body_summary.has_spooled || bctx->body_summary.bytes == 0
208 || bctx->put_codec != BRIX_CODEC_IDENTITY
209 || conf->common.thread_pool == NULL)
210 {
211 return WEBDAV_PUT_CONTINUE;
212 }
213
214 task = ngx_thread_task_alloc(r->pool, sizeof(webdav_put_aio_t));
215 if (task == NULL) {
216 brix_dashboard_http_error(r, "webdav PUT task allocation failed");
217 brix_dashboard_http_finish(r);
218 brix_vfs_writer_abort(bctx->writer);
219 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
220 return WEBDAV_PUT_DONE;
221 }
222
223 t = task->ctx;
224 t->r = r;
225 t->writer = bctx->writer; /* transfer session ownership to the task */
226 t->len = bctx->body_summary.bytes;
227 t->created = bctx->created;
228 ngx_cpystrn((u_char *) t->path, (u_char *) bctx->path, sizeof(t->path));
229
230 /*
231 * Phase 31 W2: no full-body collection — the thread streams the
232 * in-memory body bufs straight to the staged fd (see
233 * webdav_put_aio_thread); the done handler commits/aborts.
234 */
235
236 brix_task_bind(task, webdav_put_aio_thread, webdav_put_aio_done);
237
238 if (ngx_thread_task_post(conf->common.thread_pool, task) != NGX_OK) {
239 brix_dashboard_http_error(r, "webdav PUT task post failed");
240 brix_dashboard_http_finish(r);
241 brix_vfs_writer_abort(t->writer);
242 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
243 return WEBDAV_PUT_DONE;
244 }
245
246 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_THREADED]);
247 r->main->count++;
248 return WEBDAV_PUT_DONE;
249}

251/*
252 * webdav_put_select_codec — pick the body's Content-Encoding codec.
253 *
254 * WHAT: Reads the Content-Encoding header; sets `bctx->put_codec` to the decoded
255 * codec id and returns WEBDAV_PUT_CONTINUE, or returns WEBDAV_PUT_DONE after
256 * aborting the staged temp and sending 415 for an unknown/not-compiled-in
257 * coding. A missing header leaves `bctx->put_codec` as identity.
258 *
259 * WHY: Never store undecoded bytes for a coding we cannot reverse (would
260 * silently corrupt the object) — this is the single gate enforcing that, kept
261 * out of the write-dispatch body.
262 *
263 * HOW:
264 * 1. Find Content-Encoding; absent/empty → CONTINUE (identity).
265 * 2. Resolve the codec descriptor; NULL or unavailable → abort staged,
266 * finalize 415, return DONE.
267 * 3. Otherwise record the codec id and return CONTINUE.
268 */
269static webdav_put_step_t
270webdav_put_select_codec(ngx_http_request_t *r, webdav_put_body_ctx_t *bctx)
271{
272 ngx_table_elt_t *ce = brix_http_find_header(
273 r, "Content-Encoding", sizeof("Content-Encoding") - 1);
274
275 bctx->put_codec = BRIX_CODEC_IDENTITY;
276
277 if (ce != NULL && ce->value.len > 0) {
278 const brix_codec_desc_t *d = brix_codec_by_http_token(
279 (const char *) ce->value.data, ce->value.len);
280 if (d == NULL || !d->available) {
281 /* Unknown or not-compiled-in Content-Encoding: never store the
282 * undecoded bytes (would silently corrupt the object). 415. */
283 brix_dashboard_http_error(r,
284 "webdav PUT unsupported Content-Encoding");
285 brix_dashboard_http_finish(r);
286 brix_vfs_writer_abort(bctx->writer);
287 webdav_metrics_finalize_request(r,
288 NGX_HTTP_UNSUPPORTED_MEDIA_TYPE);
289 return WEBDAV_PUT_DONE;
290 }
291 bctx->put_codec = d->id;
292
293 /* A non-identity Content-Encoding declares the body IS an encoded
294 * stream, but no codec emits a zero-byte stream (gzip/zstd/… all
295 * carry a header/trailer). An empty body with a real codec is
296 * therefore malformed — reject it (same contract as a truncated
297 * stream) instead of storing a spurious empty object. */
298 if (bctx->put_codec != BRIX_CODEC_IDENTITY
299 && !bctx->body_summary.has_spooled && bctx->body_summary.bytes == 0)
300 {
301 brix_dashboard_http_error(r,
302 "webdav PUT empty body with Content-Encoding");
303 brix_dashboard_http_finish(r);
304 brix_vfs_writer_abort(bctx->writer);
305 webdav_metrics_finalize_request(r,
306 NGX_HTTP_UNSUPPORTED_MEDIA_TYPE);
307 return WEBDAV_PUT_DONE;
308 }
309 }
310
311 return WEBDAV_PUT_CONTINUE;
312}

314/*
315 * webdav_put_write_sync — synchronously stream/decode the body to the staged fd.
316 *
317 * WHAT: Bumps the per-body-shape metric, then writes the body to the staged
318 * target — decode-to-fd for a coded body, staged-driver stream for object
319 * backends, or write-to-fd for a POSIX temp — and accounts the bytes.
320 * Returns WEBDAV_PUT_CONTINUE on success, or WEBDAV_PUT_DONE after aborting
321 * the staged temp and finalizing the precise failure status (415/413/400/501/
322 * 500) on a write error.
323 *
324 * WHY: Confines the three-way write dispatch and its abort-on-failure invariant
325 * (a failed write must never leave a readable partial at the final path) to
326 * one place, mirroring the thread path's error handling.
327 *
328 * HOW:
329 * 1. Increment the SPOOLED/MEMORY/EMPTY body-shape metric.
330 * 2. Coded body: NOSYS/501 on a driver target (no kernel fd), else
331 * brix_http_body_decode_to_fd (reports its own precise status).
332 * 3. Identity body: staged-driver stream or write-to-fd.
333 * 4. On write failure abort staged, finalize the reported status, return DONE.
334 * 5. On success account the bytes via brix_dashboard_http_add, return CONTINUE.
335 */
336static webdav_put_step_t
337webdav_put_write_sync(ngx_http_request_t *r, webdav_put_body_ctx_t *bctx)
338{
339 ngx_int_t wrc;
340 ngx_int_t decode_status = NGX_HTTP_INTERNAL_SERVER_ERROR;
341
342 if (bctx->body_summary.has_spooled) {
343 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_SPOOLED]);
344 } else if (bctx->body_summary.has_memory) {
345 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_MEMORY]);
346 } else {
347 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_EMPTY]);
348 }
349
350 if (bctx->put_codec != BRIX_CODEC_IDENTITY) {
351 /* A Content-Encoding decode writes plaintext straight to the session's
352 * kernel temp fd (the decode engine owns the streaming + bomb guard);
353 * a driver-backed object exposes no fd (NGX_INVALID_FILE) →
354 * decode-to-object is a follow-up. These bytes bypass the writer's CRC
355 * accumulator, so a verifying session leaves them unverified (its commit
356 * read-back is a no-op when nothing went through the accumulator) —
357 * verify-on-write does not cover coded bodies. */
358 ngx_fd_t wfd = brix_vfs_writer_fd(bctx->writer);
359 if (wfd == NGX_INVALID_FILE) {
360 errno = ENOSYS;
361 wrc = NGX_ERROR;
362 decode_status = NGX_HTTP_NOT_IMPLEMENTED;
363 } else {
364 wrc = brix_http_body_decode_to_fd(r, wfd, bctx->path,
365 bctx->put_codec,
366 BRIX_DECODE_MAX_OUTPUT,
367 &bctx->body_summary,
368 &decode_status);
369 }
370 } else {
371 /* Identity body: one common verified-write call — the writer dispatches
372 * memory bufs, spooled temp-fd bufs, and driver-object targets. */
373 wrc = brix_http_body_write_to_writer(r, bctx->writer);
374 }
375 if (wrc != NGX_OK) {
376 brix_dashboard_http_error(r, "webdav PUT body write failed");
377 brix_dashboard_http_finish(r);
378 /* Abort the session — the final path is never touched, so a failed
379 * write (e.g. a corrupt/over-large Content-Encoding that fails to
380 * decode) can never leave a readable partial object. The decode path
381 * reports the precise status (413 bomb / 400 corrupt). */
382 brix_vfs_writer_abort(bctx->writer);
383 webdav_metrics_finalize_request(r, decode_status);
384 return WEBDAV_PUT_DONE;
385 }
386
387 brix_dashboard_http_add(r, (ngx_atomic_int_t) bctx->body_summary.bytes);
388 return WEBDAV_PUT_CONTINUE;
389}

391/*
392 * webdav_put_stream_body — write the request body (async or sync) to staged.
393 *
394 * WHAT: Summarizes the body, starts dashboard tracking, and drives the body to
395 * the staged temp: for a request with a body it tries the thread-pool offload
396 * then falls through the codec-select + synchronous write; an empty PUT just
397 * records the EMPTY metric. Returns WEBDAV_PUT_CONTINUE when the caller
398 * should proceed to commit synchronously, or WEBDAV_PUT_DONE when the body
399 * phase has already answered the request (threaded dispatch, or a handled
400 * error).
401 *
402 * WHY: Groups the whole "get the bytes onto the staged fd" concern — including
403 * the async/sync/empty split — behind one call so the orchestrator commits
404 * only what the sync/empty paths leave staged. Behavior (metrics, statuses,
405 * ownership) is unchanged from the prior inline block.
406 *
407 * HOW:
408 * 1. No request body → start dashboard(0), EMPTY metric, CONTINUE.
409 * 2. Body present → summarize (failure aborts + 500 → DONE), start dashboard,
410 * add bytes_rx metric.
411 * 3. Try the threaded offload; if it took over or failed, return DONE.
412 * 4. Select the codec (415 → DONE), then synchronously write (error → DONE).
413 * 5. CONTINUE — the staged temp holds the full body, ready to commit.
414 */
415webdav_put_step_t
416webdav_put_stream_body(ngx_http_request_t *r,
417 ngx_http_brix_webdav_loc_conf_t *conf, const char *path,
418 brix_vfs_writer_t *writer, int created)
419{
420 webdav_put_body_ctx_t bctx;
421
422 if (r->request_body == NULL) {
423 webdav_put_start_dashboard(r, path, 0);
424 BRIX_WEBDAV_METRIC_INC(put_body_total[BRIX_WEBDAV_PUT_EMPTY]);
425 return WEBDAV_PUT_CONTINUE;
426 }
427
428 ngx_memzero(&bctx, sizeof(bctx));
429 bctx.path = path;
430 bctx.writer = writer;
431 bctx.created = created;
432 bctx.put_codec = BRIX_CODEC_IDENTITY;
433
434 if (brix_http_body_summary(r, &bctx.body_summary) != NGX_OK) {
435 brix_vfs_writer_abort(writer);
436 webdav_metrics_finalize_request(r, NGX_HTTP_INTERNAL_SERVER_ERROR);
437 return WEBDAV_PUT_DONE;
438 }
439
440 webdav_put_start_dashboard(r, path, (int64_t) bctx.body_summary.bytes);
441 BRIX_WEBDAV_METRIC_ADD(bytes_rx_total, bctx.body_summary.bytes);
442
443 if (webdav_put_select_codec(r, &bctx) == WEBDAV_PUT_DONE) {
444 return WEBDAV_PUT_DONE;
445 }
446
447 if (webdav_put_try_threaded(r, conf, &bctx) == WEBDAV_PUT_DONE) {
448 return WEBDAV_PUT_DONE;
449 }
450
451 return webdav_put_write_sync(r, &bctx);
452}
