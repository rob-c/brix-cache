114/* WHAT: Build kXR_ok open response body (fhandle + optional statbuf from fstat) → brix_build_resp_hdr → brix_queue_response. Returns NGX_OK or NGX_ERROR on alloc failure. Caller: brix_tpc_prepare_pull (end of pull prep pipeline). */
115static ngx_int_t
116tpc_send_open_response(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
117 int fd, uint16_t options)
118{
119 ServerOpenBody body;
120 struct stat st;
121 char statbuf[256];
122 size_t bodylen;
123 size_t total;
124 u_char *buf;
125 ngx_flag_t want_stat;
126
127 want_stat = (options & kXR_retstat) ? 1 : 0;
128 statbuf[0] = '\0';
129 bodylen = sizeof(ServerOpenBody);
130
131 if (want_stat && fstat(fd, &st) == 0) {
132 int stat_flags = 0;
133
134 if (st.st_mode & (S_IRUSR | S_IRGRP | S_IROTH)) {
135 stat_flags |= kXR_readable;
136 }
137 if (st.st_mode & (S_IWUSR | S_IWGRP | S_IWOTH)) {
138 stat_flags |= kXR_writable;
139 }
140
141 snprintf(statbuf, sizeof(statbuf), "%llu %lld %d %ld",
142 (unsigned long long) st.st_ino,
143 (long long) st.st_size,
144 stat_flags,
145 (long) st.st_mtime);
146 bodylen += strlen(statbuf) + 1;
147 }
148
149 total = XRD_RESPONSE_HDR_LEN + bodylen;
150 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
151
152 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok, (uint32_t) bodylen,
153 (ServerResponseHdr *) buf);
154
155 ngx_memzero(&body, sizeof(body));
156 body.fhandle[0] = (u_char) idx;
157 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body, sizeof(body));
158
159 if (statbuf[0] != '\0') {
160 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + sizeof(ServerOpenBody),
161 statbuf, strlen(statbuf) + 1);
162 }
163
164 return brix_queue_response(ctx, c, buf, total);
165}

167/* WHAT: Construct origin ID string from ctx->login.user+ngx_pid+getnameinfo host via snprintf("%s.%u@host") — falls back to "xrd" for empty user, ngx_pid for zero pid, addr_text.len then "unknown" for unresolved host. Caller: brix_tpc_prepare_pull (origin ID storage step). */

169static void
170tpc_build_origin_id(brix_ctx_t *ctx, ngx_connection_t *c, char *dst,
171 size_t dst_size)
172{
173 char host[NI_MAXHOST];
174 const char *user;
175 uint32_t pid;
176
177 user = ctx->login.user[0] != '\0' ? ctx->login.user : "xrd";
178 pid = ctx->login.pid != 0 ? ctx->login.pid : (uint32_t) ngx_pid;
179
180 host[0] = '\0';
181 if (c->sockaddr != NULL
182 && getnameinfo(c->sockaddr, c->socklen, host, sizeof(host),
183 NULL, 0, NI_NAMEREQD) != 0) {
184 host[0] = '\0';
185 }
186
187 if (host[0] == '\0' && c->addr_text.len > 0) {
188 size_t host_len = c->addr_text.len;
189
190 if (host_len >= sizeof(host)) {
191 host_len = sizeof(host) - 1;
192 }
193 ngx_memcpy(host, c->addr_text.data, host_len);
194 host[host_len] = '\0';
195 }
196
197 if (host[0] == '\0') {
198 ngx_cpystrn((u_char *) host, (u_char *) "unknown", sizeof(host));
199 }
200
201 {
202 int prefix_len;
203
204 prefix_len = snprintf(dst, dst_size, "%s.%u@", user, (unsigned) pid);
205 if (prefix_len < 0 || (size_t) prefix_len >= dst_size) {
206 dst[0] = '\0';
207 return;
208 }
209
210 ngx_cpystrn((u_char *) dst + prefix_len, (u_char *) host,
211 dst_size - (size_t) prefix_len);
212 }
213}

215/* WHAT: Translate the kXR_open options bitmask into POSIX open(2) flags for the
216 * TPC destination file.
217 * WHY: The wire-level create semantics differ from POSIX, so the mapping is
218 * explicit: kXR_new alone => create, fail if it already exists (O_EXCL);
219 * kXR_new + kXR_delete or kXR_delete alone => create-or-truncate; neither flag
220 * (the common "just receive the copy") => create-or-truncate as well. The
221 * O_EXCL is deliberately dropped whenever kXR_delete is present, since delete
222 * means "overwrite is intended". Always O_RDWR (we read back for stat) and
223 * O_NOCTTY (never acquire a controlling terminal). */
224static int
225tpc_destination_open_flags(uint16_t options)
226{
227 int oflags;
228
229 oflags = O_RDWR | O_NOCTTY;
230 if (options & kXR_new) {
231 oflags |= O_CREAT;
232 if (!(options & kXR_delete)) {
233 oflags |= O_EXCL; /* create-new only: refuse existing target */
234 }
235 }
236 if (options & kXR_delete) {
237 oflags |= O_CREAT | O_TRUNC; /* overwrite intended */
238 }
239 if (!(options & (kXR_new | kXR_delete))) {
240 oflags |= O_CREAT | O_TRUNC; /* default: create or replace */
241 }
242
243 return oflags;
244}

246/* WHAT: Register this pull in the shared TPC transfer registry and return its
247 * transfer id (0 on failure / registry full).
248 * WHY: Active transfers are tracked centrally for the dashboard, metrics, and
249 * progress reporting; the id is stored on the file handle and later used by the
250 * worker/done callback to update state. Reconstructs a canonical source URL
251 * from the handle's stored host/port/path for display.
252 * HOW: default port to 1094 if unset, format "root://host:port/path" into a
253 * stack buffer, fill an brix_tpc_transfer_t (PROTO_STREAM/DIR_PULL/PENDING),
254 * and hand it to brix_tpc_registry_add. */
255static uint64_t
256tpc_register_stream_transfer(ngx_connection_t *c, brix_file_t *file)
257{
258 brix_tpc_transfer_t transfer;
259 ngx_str_t src_url;
260 ngx_str_t dst_path;
261 u_char src_buf[PATH_MAX + 320];
262 u_char *last;
263 uint16_t sport;
264
265 if (file == NULL || file->path == NULL) {
266 return 0;
267 }
268
269 /* Reconstruct the transfer URL. Bracket an IPv6 literal source host —
270 * tpc_src_host is stored bare (parse.c strips the brackets off "[::1]"), so
271 * "root://[::1]:1094/path" must be re-bracketed here or the URL is
272 * unparseable. No NUL is written; src_url.len is the returned end pointer. */
273 sport = file->tpc_src_port ? file->tpc_src_port : 1094;
274 {
275 char hostport[288];
276 brix_format_host_port(file->tpc_src_host, sport,
277 hostport, sizeof(hostport));
278 last = ngx_snprintf(src_buf, sizeof(src_buf), "root://%s%s",
279 hostport, file->tpc_src_path);
280 }
281
282 src_url.data = src_buf;
283 src_url.len = (size_t) (last - src_buf);
284 dst_path.data = (u_char *) file->path;
285 dst_path.len = ngx_strlen(file->path);
286
287 ngx_memzero(&transfer, sizeof(transfer));
288 transfer.protocol = BRIX_TPC_PROTO_STREAM;
289 transfer.direction = BRIX_TPC_DIR_PULL;
290 transfer.src_url = src_url;
291 transfer.dst_path = dst_path;
292 transfer.state = BRIX_TPC_STATE_PENDING;
293
294 return brix_tpc_registry_add(&transfer, c->log);
295}

297/* WHAT: Derive O_CREAT/O_EXCL/O_TRUNC flags from options bitmask — kXR_new → O_CREAT+O_EXCL, kXR_delete → O_CREAT+O_TRUNC, neither → O_CREAT+O_TRUNC (default create-new). Always includes O_RDWR|O_NOCTTY. Caller: brix_tpc_prepare_pull (open flags step). */

299brix_tpc_prepare_pull(brix_ctx_t *ctx, ngx_connection_t *c,
300 ngx_stream_brix_srv_conf_t *conf, const brix_tpc_params_t *tpc,
301 const char *dst_path, uint16_t options, uint16_t mode_bits)
302{
303 brix_file_t *file;
304 struct stat st;
305 mode_t create_mode;
306 int idx;
307 int fd;
308
309 if (conf->common.thread_pool == NULL) {
310 brix_log_access(ctx, c, "OPEN", dst_path, "tpc-pull",
311 0, kXR_ServerError, "TPC requires brix_thread_pool",
312 0);
313 BRIX_OP_ERR(ctx, BRIX_OP_OPEN_WR);
314 return brix_send_error(ctx, c, kXR_ServerError,
315 "TPC pull requires brix_thread_pool "
316 "to be configured");
317 }
318
319 if (tpc->src_host[0] == '\0' || tpc->src_path[0] == '\0') {
320 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN", dst_path,
321 "tpc-pull", kXR_ArgInvalid,
322 "invalid or incomplete TPC source");
323 }

325 /*
326 * Source policy gate (SSRF defence): before we ever connect outbound, the
327 * resolved source host/port is checked against the loopback/private-range
328 * allow flags. A destination server must not be coercible into pulling from
329 * internal addresses unless the operator explicitly permits it.
330 */
331 {
332 char policy_err[512];
333 uint16_t sport;
334
335 sport = tpc->src_port ? tpc->src_port : 1094;
336 if (brix_tpc_check_src_policy(tpc->src_host, sport,
337 conf->tpc_allow_local, conf->tpc_allow_private,
338 policy_err, sizeof(policy_err))
339 != 0)
340 {
341 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN", dst_path,
342 "tpc-pull", kXR_NotAuthorized, policy_err);
343 }
344 }
345
346 idx = brix_alloc_fhandle(ctx);
347 if (idx < 0) {
348 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN", dst_path,
349 "tpc-pull", kXR_ServerError, "too many open files");
350 }
351
352 create_mode = (mode_bits & 0777) ? (mode_t) (mode_bits & 0777) : 0644;
353
354 /*
355 * brix_vfs_open_fd_at() resolves its path relative to conf->rootfd (it
356 * strips the leading '/' via brix_beneath_rel), so it must receive the
357 * LOGICAL export path — not the root_canon-prefixed absolute path that
358 * dst_path carries for authz/logging/fhandle metadata. Passing the
359 * absolute path here doubles the root prefix and openat2() fails with
360 * ENOENT. Recover the logical path by stripping the root_canon prefix.

365
366 if (root_len > 1
367 && ngx_strncmp(dst_path, conf->common.root_canon, root_len) == 0
368 && (dst_path[root_len] == '/' || dst_path[root_len] == '\0'))
369 {
370 dst_logical = dst_path + root_len;
371 }
372
373 fd = brix_vfs_open_fd_at(conf->rootfd, dst_logical,
374 tpc_destination_open_flags(options),
375 create_mode);
376 }
377 if (fd < 0) {
378 int err = errno;
379
380 BRIX_RETURN_ERR(ctx, c, BRIX_OP_OPEN_WR, "OPEN", dst_path,
381 "tpc-pull", kXR_IOError, strerror(err));
382 }
383
384 if (fstat(fd, &st) != 0) {
385 int err = errno;
386
387 close(fd);
388 BRIX_OP_ERR(ctx, BRIX_OP_OPEN_WR);

393 file->fd = fd;
394 file->writable = 1;
395 file->readable = 0;
396 file->from_cache = 0;
397 file->is_regular = S_ISREG(st.st_mode) ? 1 : 0;
398 file->device = st.st_dev;
399 file->inode = st.st_ino;
400 file->cached_size = (off_t) st.st_size;
401 file->read_last_end = -1;
402 file->read_ahead_end = 0;
403 file->bytes_read = 0;
404 file->bytes_written = 0;
405 file->open_time = ngx_current_msec;
406 file->tpc_destination = 1;
407 file->tpc_armed = 0;
408 file->tpc_started = 0;
409 file->tpc_done = 0;
410 file->tpc_src_port = tpc->src_port;
411 file->tpc_transfer_id = 0;
412
413 /* TPC rendezvous key: echo the client-supplied key if present (the source
414 * side already knows it), otherwise mint a fresh random one for this leg. */
415 if (tpc->key[0] != '\0') {
416 ngx_cpystrn((u_char *) file->tpc_key, (u_char *) tpc->key,
417 sizeof(file->tpc_key));
418 } else {
419 brix_tpc_generate_key(file->tpc_key, sizeof(file->tpc_key));
420 }
421
422 tpc_build_origin_id(ctx, c, file->tpc_org, sizeof(file->tpc_org));
423 ngx_cpystrn((u_char *) file->tpc_src_host, (u_char *) tpc->src_host,
424 sizeof(file->tpc_src_host));
425 ngx_cpystrn((u_char *) file->tpc_src_path, (u_char *) tpc->src_path,
426 sizeof(file->tpc_src_path));
427
428 if (brix_set_fhandle_path(ctx, c, idx, dst_path) != NGX_OK) {
429 brix_free_fhandle(ctx, idx);
430 return NGX_ERROR;
431 }
432
433 if (!ctx->is_bound) {
434 brix_session_handle_publish(ctx->login.sessid, idx, file);
435 }
436
437 /* Store token_mode for use during pull task execution. */
438 if (tpc->has_token_mode && tpc->token_mode[0] != '\0') {
439 ngx_cpystrn((u_char *) file->tpc_token_mode,
440 (u_char *) tpc->token_mode, sizeof(file->tpc_token_mode));
441 } else {
442 file->tpc_token_mode[0] = '\0';
443 }
444
445 brix_log_access(ctx, c, "OPEN", dst_path, "tpc-pull", 1, 0, NULL, 0);
446 BRIX_OP_OK(ctx, BRIX_OP_OPEN_WR);
447
448 return tpc_send_open_response(ctx, c, idx, fd, options);
449}

451/* WHAT: Allocate ngx_thread_task(sizeof(brix_tpc_pull_t)) → populate struct from file fields (src_host/path/key/org/token_mode/dst_path) → set handler=brix_tpc_pull_thread, event.handler=brix_tpc_pull_done → post to thread pool. 452ngx_int_t
453brix_tpc_start_pull(brix_ctx_t *ctx, ngx_connection_t *c,
454 ngx_stream_brix_srv_conf_t *conf, int fhandle_idx)
455{
456 ngx_thread_task_t *task;
457 brix_tpc_pull_t *t;
458 brix_file_t *file;
459
460 if (fhandle_idx < 0 || fhandle_idx >= BRIX_MAX_FILES) {
461 return NGX_ERROR;
462 }
463
464 /* The handle must be a live TPC destination opened by prepare_pull. */
465 file = &ctx->files[fhandle_idx];
466 if (!file->tpc_destination || file->fd < 0) {
467 return brix_send_error(ctx, c, kXR_FileNotOpen,
468 "invalid TPC destination handle");
469 }
470
471 /* Idempotent re-trigger: a sync arriving while the worker is already
472 * running gets a kXR_wait, not a second thread post. */
473 if (file->tpc_started) {
474 return brix_send_wait(ctx, c, 1);
475 }
476
477 if (file->tpc_transfer_id == 0) {
478 file->tpc_transfer_id = tpc_register_stream_transfer(c, file);
479 if (file->tpc_transfer_id == 0) {
480 BRIX_RETURN_ERR(ctx, c, BRIX_OP_SYNC, "SYNC", file->path,
481 "tpc-pull", kXR_Overloaded,
482 "TPC transfer registry full");
483 }
484 }
485
486 task = ngx_thread_task_alloc(c->pool, sizeof(brix_tpc_pull_t));
487 if (task == NULL) {
488 (void) brix_tpc_registry_remove(file->tpc_transfer_id, c->log);
489 file->tpc_transfer_id = 0;
490 return NGX_ERROR;
491 }
492
493 /*
494 * Snapshot everything the worker thread needs into the task ctx. The worker
495 * runs off-thread and must NOT touch ctx->files or the connection's mutable
496 * state, so all source/dest fields are copied by value here. The streamid
497 * is captured so done.c can re-bind the deferred request when it completes.
498 */
499 t = task->ctx;
500 ngx_memzero(t, sizeof(*t));
501 t->c = c;
502 t->ctx = ctx;
503 t->conf = conf;
504 t->streamid[0] = ctx->recv.cur_streamid[0];
505 t->streamid[1] = ctx->recv.cur_streamid[1];
506 t->dst_fd = file->fd;
507 t->fhandle_idx = fhandle_idx;
508 t->reply_kind = BRIX_TPC_REPLY_SYNC;
509 t->src_port = file->tpc_src_port;
510 t->transfer_id = file->tpc_transfer_id;
511
512 ngx_cpystrn((u_char *) t->src_host, (u_char *) file->tpc_src_host,
513 sizeof(t->src_host));
514 ngx_cpystrn((u_char *) t->src_path, (u_char *) file->tpc_src_path,
515 sizeof(t->src_path));
516 ngx_cpystrn((u_char *) t->tpc_key, (u_char *) file->tpc_key,
517 sizeof(t->tpc_key));
518 ngx_cpystrn((u_char *) t->tpc_org, (u_char *) file->tpc_org,
519 sizeof(t->tpc_org));
520 ngx_cpystrn((u_char *) t->token_mode, (u_char *) file->tpc_token_mode,
521 sizeof(t->token_mode));

549 * this pull HERE on the event loop — where identity + path are available and
550 * pmark runtime init is single-threaded — and stash on the task. The thread

552 * label. Fail-open: 0/0 means the outbound socket is not labelled.
553 */

557 && brix_pmark_runtime_ensure(&conf->common.pmark, ngx_cycle->pool,

562 ctx->identity ? brix_identity_vo_csv_cstr(ctx->identity) : "",

565 {
566 t->pmark_exp = e;
567 t->pmark_act = a;
568 }
569 }
570

572

581 "TPC pull thread post failed");

585 * the event loop defers further request processing until done.c resumes it. */
586 file->tpc_started = 1;
587 ctx->state = XRD_ST_AIO;
588 return NGX_OK;
589}
