34static brix_sess_am_t
35tpc_sess_auth_method(const brix_tpc_pull_t *t)
36{
37 if (t == NULL || t->conf == NULL) {
38 return BRIX_SESS_AM_ANON;
39 }
40
41 if ((t->token_mode[0] != '\0' && ngx_strcmp(t->token_mode, "none") != 0)
42 || t->conf->tpc_outbound_bearer_file.len > 0)
43 {
44 return BRIX_SESS_AM_TOKEN;
45 }
46
47 if (t->deleg_cred_len > 0
48 || (t->conf->certificate.len > 0 && t->conf->certificate_key.len > 0))

50 return BRIX_SESS_AM_GSI;
51 }
52
53 return BRIX_SESS_AM_ANON;

125/* WHAT: Register this pull in the shared TPC transfer registry and return its
126 * transfer id (0 on failure / registry full).
127 * WHY: Active transfers are tracked centrally for the dashboard, metrics, and
128 * progress reporting; the id is stored on the file handle and later used by the
129 * worker/done callback to update state. Reconstructs a canonical source URL
130 * from the handle's stored host/port/path for display.
131 * HOW: default port to 1094 if unset, format "root://host:port/path" into a
132 * stack buffer, fill an brix_tpc_transfer_t (PROTO_STREAM/DIR_PULL/PENDING),
133 * and hand it to brix_tpc_registry_add. */
134static uint64_t
135tpc_register_stream_transfer(ngx_connection_t *c, brix_file_t *file)
136{
137 brix_tpc_transfer_t transfer;
138 ngx_str_t src_url;
139 ngx_str_t dst_path;
140 u_char src_buf[PATH_MAX + 320];
141 u_char *last;
142 uint16_t sport;
143
144 if (file == NULL || file->path == NULL) {
145 return 0;
146 }
147
148 /* Reconstruct the transfer URL. Bracket an IPv6 literal source host —
149 * tpc_src_host is stored bare (parse.c strips the brackets off "[::1]"), so
150 * "root://[::1]:1094/path" must be re-bracketed here or the URL is
151 * unparseable. No NUL is written; src_url.len is the returned end pointer. */
152 sport = file->tpc_src_port ? file->tpc_src_port : 1094;
153 {
154 char hostport[288];
155 brix_format_host_port(file->tpc_src_host, sport,
156 hostport, sizeof(hostport));
157 last = ngx_snprintf(src_buf, sizeof(src_buf), "root://%s%s",
158 hostport, file->tpc_src_path);
159 }
160
161 src_url.data = src_buf;
162 src_url.len = (size_t) (last - src_buf);
163 dst_path.data = (u_char *) file->path;
164 dst_path.len = ngx_strlen(file->path);
165
166 ngx_memzero(&transfer, sizeof(transfer));
167 transfer.protocol = BRIX_TPC_PROTO_STREAM;
168 transfer.direction = BRIX_TPC_DIR_PULL;
169 transfer.src_url = src_url;
170 transfer.dst_path = dst_path;
171 transfer.state = BRIX_TPC_STATE_PENDING;
172
173 return brix_tpc_registry_add(&transfer, c->log);
174}

176/* WHAT: Copy every field the off-thread worker needs out of the connection and
177 * the ctx->files[] slot into the freshly-allocated task context t (by value).
178 * WHY: The worker runs on a thread-pool thread and must NOT touch ctx->files or
179 * the connection's mutable state, so all source/dest info is snapshotted here on
180 * the event loop. The streamid is captured so done.c can re-bind the deferred
181 * request on completion. Isolating the copy keeps start_pull's orchestrator flat
182 * and the field set (and behaviour) unchanged.
183 * HOW: memzero t → set connection/ctx/conf refs, streamid, dst_fd, reply_kind,
184 * src_port, transfer_id scalars → cpystrn src_host/src_path/tpc_key/tpc_org/
185 * token_mode/dst_path from the file's stored fields. The caller sets
186 * t->fhandle_idx (the slot index it already holds) after this returns. */

188tpc_populate_pull_task(brix_tpc_pull_t *t, brix_ctx_t *ctx,
189 ngx_connection_t *c, ngx_stream_brix_srv_conf_t *conf,
190 brix_file_t *file)
191{
192 ngx_memzero(t, sizeof(*t));
193 t->c = c;
194 t->ctx = ctx;
195 t->conf = conf;
196 t->streamid[0] = ctx->recv.cur_streamid[0];
197 t->streamid[1] = ctx->recv.cur_streamid[1];
198 t->dst_fd = file->fd;
199 t->reply_kind = BRIX_TPC_REPLY_SYNC;
200 t->src_port = file->tpc_src_port;
201 t->transfer_id = file->tpc_transfer_id;
202
203 ngx_cpystrn((u_char *) t->src_host, (u_char *) file->tpc_src_host,
204 sizeof(t->src_host));
205 ngx_cpystrn((u_char *) t->src_path, (u_char *) file->tpc_src_path,
206 sizeof(t->src_path));
207 ngx_cpystrn((u_char *) t->tpc_key, (u_char *) file->tpc_key,
208 sizeof(t->tpc_key));
209 ngx_cpystrn((u_char *) t->tpc_org, (u_char *) file->tpc_org,
210 sizeof(t->tpc_org));
211 ngx_cpystrn((u_char *) t->token_mode, (u_char *) file->tpc_token_mode,
212 sizeof(t->token_mode));
213 ngx_cpystrn((u_char *) t->dst_path, (u_char *) file->path,
214 sizeof(t->dst_path));

217/* WHAT: For either passthrough token_mode ("passthrough" strict or
218 * "passthrough-opt" opportunistic), snapshot the client's own inbound bearer JWT
219 * (ctx->bearer_token, captured at login by the token auth path) into
220 * t->delegated_token so the outbound ztn leg forwards it verbatim to the source.
221 * WHY (phase-70): passthrough makes the SOURCE authenticate the END USER, not a
222 * static service credential. The capture MUST happen here on the event loop — the
223 * off-thread worker must never touch ctx — and the token must be snapshotted by
224 * value into the task before hand-off. An empty (or over-length) inbound token is
225 * left empty: the downstream fetch/dispatch then denies cleanly for the strict
226 * "passthrough" mode (kXR_AuthFailed) but opportunistically falls back to
227 * GSI/bearer-file/anonymous for the default "passthrough-opt" mode. Capture is
228 * identical for both — the divergence lives entirely in tpc_fetch_delegated_token.
229 * HOW: only when t->token_mode is "passthrough" or "passthrough-opt"; measure
230 * ctx->bearer_token; on a non-empty token that fits (with its NUL) memcpy it into
231 * t->delegated_token, else leave t->delegated_token empty. Pure side-effect on *t. */
232static void
233tpc_pull_capture_passthrough_token(brix_tpc_pull_t *t, brix_ctx_t *ctx)
234{
235 size_t token_len;
236
237 if (ngx_strcmp(t->token_mode, "passthrough") != 0
238 && ngx_strcmp(t->token_mode, "passthrough-opt") != 0) {
239 return;
240 }
241
242 token_len = ngx_strlen(ctx->bearer_token);
243 if (token_len == 0 || token_len >= sizeof(t->delegated_token)) {
244 /* No inbound token (or one too large): leave delegated_token empty so
245 * have_ztn_cred is false and the pull is denied cleanly. */
246 t->delegated_token[0] = '\0';
247 return;
248 }
249
250 ngx_memcpy(t->delegated_token, ctx->bearer_token, token_len + 1);
251}

253/* WHAT: Attach outbound source-auth credentials to the pull task — the captured
254 * delegated proxy (§F6) when delegation is enabled and a proxy was captured, the
255 * passthrough inbound bearer JWT when token_mode is "passthrough" or
256 * "passthrough-opt", and the configured token exchange scope.
257 * WHY: These auth inputs are conditional and were inline branch blocks in
258 * start_pull; grouping them keeps the orchestrator flat. Behaviour is unchanged
259 * for GSI/scope: the delegated proxy is malloc'd (thread-owned, freed in
260 * thread.c) only when brix_tpc_delegate is set AND a proxy was actually captured
261 * during inbound GSI login; the scope is copied only when configured. The
262 * passthrough capture (phase-70) snapshots ctx->bearer_token into the task on the
263 * event loop so the worker never touches ctx.
264 * HOW: on delegate+captured-proxy, malloc + memcpy the PEM and set deleg_cred_len
265 * (malloc failure leaves deleg_cred_pem NULL → fall back to the gateway cert);
266 * tpc_pull_capture_passthrough_token for the inbound-token snapshot; token_scope
267 * defaults to empty then filled from conf->tpc_outbound_scope. */
268static void
269tpc_pull_attach_creds(brix_tpc_pull_t *t, brix_ctx_t *ctx,
270 ngx_stream_brix_srv_conf_t *conf)
271{
272 /*
273 * §F6: if proxy delegation captured the user's proxy during the inbound GSI
274 * login, hand the full credential (proxy cert + key + chain) to the pull so it
275 * authenticates to the source AS THE USER. malloc (thread-owned, freed in
276 * thread.c). Off unless brix_tpc_delegate + a proxy was actually captured.
277 */
278 if (conf->tpc_delegate && ctx->gsi.deleg_proxy_pem != NULL
279 && ctx->gsi.deleg_proxy_len > 0) {
280 t->deleg_cred_pem = malloc(ctx->gsi.deleg_proxy_len);
281 if (t->deleg_cred_pem != NULL) {
282 ngx_memcpy(t->deleg_cred_pem, ctx->gsi.deleg_proxy_pem,
283 ctx->gsi.deleg_proxy_len);
284 t->deleg_cred_len = ctx->gsi.deleg_proxy_len;
285 }
286 }
287
288 tpc_pull_capture_passthrough_token(t, ctx);
289
290 t->token_scope[0] = '\0';
291 if (conf->tpc_outbound_scope.len > 0) {
292 (void) brix_str_cbuf(t->token_scope, sizeof(t->token_scope),
293 &conf->tpc_outbound_scope);
294 }
295}

297/* WHAT: Resolve the SciTags (experiment, activity) codes for this pull on the
298 * event loop and stash them on the task (0/0 when not applicable).
299 * WHY (phase-34): identity + path are available here and pmark runtime init is
300 * single-threaded, so the resolve must happen on the event loop; the thread
301 * (connect.c) only reads the codes and stamps the outbound socket's IPv6 flow
302 * label. Fail-open: 0/0 means the outbound socket is not labelled. Extracting it
303 * removes the pull's most branch-dense block from the orchestrator unchanged.
304 * HOW: default 0/0 → only when pmark.enable && pmark.flowlabel && runtime ensure
305 * succeeds, build a flow_id from the identity VO/DN + dst_path and map codes;

313

315 t->pmark_act = 0;
316

338/* WHAT: Allocate ngx_thread_task(sizeof(brix_tpc_pull_t)) → populate struct from file fields (src_host/path/key/org/token_mode/dst_path) → set handler=brix_tpc_pull_thread, event.handler=brix_tpc_pull_done → post to thread pool. Returns NGX_OK or error on alloc/post failure. Caller: dispatch.c (kXR_sync TPC launch path). */
339ngx_int_t
340brix_tpc_start_pull(brix_ctx_t *ctx, ngx_connection_t *c,
341 ngx_stream_brix_srv_conf_t *conf, int fhandle_idx)
342{
343 ngx_thread_task_t *task;
344 brix_tpc_pull_t *t;
345 brix_file_t *file;
346
347 if (fhandle_idx < 0 || fhandle_idx >= BRIX_MAX_FILES) {
348 return NGX_ERROR;
349 }
350
351 /* The handle must be a live TPC destination opened by prepare_pull. */
352 file = &ctx->files[fhandle_idx];
353 if (!file->tpc_destination || file->fd < 0) {
354 return brix_send_error(ctx, c, kXR_FileNotOpen,
355 "invalid TPC destination handle");
356 }
357
358 /* Idempotent re-trigger: a sync arriving while the worker is already
359 * running gets a kXR_wait, not a second thread post. */
360 if (file->tpc_started) {
361 return brix_send_wait(ctx, c, 1);
362 }
363
364 if (file->tpc_transfer_id == 0) {
365 file->tpc_transfer_id = tpc_register_stream_transfer(c, file);
366 if (file->tpc_transfer_id == 0) {
367 BRIX_RETURN_ERR(ctx, c, BRIX_OP_SYNC, "SYNC", file->path,
368 "tpc-pull", kXR_Overloaded,
369 "TPC transfer registry full");
370 }
371 }
372
373 task = ngx_thread_task_alloc(c->pool, sizeof(brix_tpc_pull_t));
374 if (task == NULL) {
375 (void) brix_tpc_registry_remove(file->tpc_transfer_id, c->log);
376 file->tpc_transfer_id = 0;
377 return NGX_ERROR;
378 }
379
380 t = task->ctx;
381 tpc_populate_pull_task(t, ctx, c, conf, file);
382 t->fhandle_idx = fhandle_idx;
383 tpc_pull_attach_creds(t, ctx, conf);
384 tpc_sess_begin_pull(t);
385 tpc_pull_resolve_pmark(t, ctx, c, conf);
386
387 brix_task_bind(task, brix_tpc_pull_thread, brix_tpc_pull_done);
388
389 if (ngx_thread_task_post(conf->common.thread_pool, task) != NGX_OK) {
390 (void) brix_tpc_registry_remove(file->tpc_transfer_id, c->log);
391 file->tpc_transfer_id = 0;
392 tpc_sess_abort_pull(t, "io");
393 brix_log_access(ctx, c, "SYNC", file->path, "tpc-pull",
394 0, kXR_ServerError, "thread post failed", 0);
395 BRIX_OP_ERR(ctx, BRIX_OP_SYNC);
396 return brix_send_error(ctx, c, kXR_ServerError,
397 "TPC pull thread post failed");
398 }
399
400 /* Hand-off committed: mark started and park the connection in AIO state so
401 * the event loop defers further request processing until done.c resumes it. */
402 file->tpc_started = 1;
403 ctx->state = XRD_ST_AIO;
404 return NGX_OK;
405}
