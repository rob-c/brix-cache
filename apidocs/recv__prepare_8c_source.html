1/*
2 * cms/recv_prepare.c — forwarded staging ops (kYR_prepadd / kYR_prepdel) for
3 * the CMS client-side (manager-connection) receive path. Phase-89 W2 / PR-6.
4 *
5 * WHAT: A manager forwards a client's stage-in request down to this data node
6 * as kYR_prepadd (Pup padArgs: ident, reqid, notify, prty, mode, path) and a
7 * cancellation as kYR_prepdel (pdlArgs: ident, reqid). prepadd admits the
8 * path into the durable stage-request registry (the same store kXR_prepare
9 * and the Tape REST API use — the recall itself is faulted by the sd_frm
10 * backend on read); prepdel deletes that admission. Replies match stock cmsd
11 * XrdCmsNode::do_PrepAdd/do_PrepDel: silent on success AND on an idempotent
12 * no-op delete; kYR_error only for a refused/undecodable prepadd.
13 *
14 * WHY: Plane-B staging parity (phase-61 W2). The manager's reqid is NOT the
15 * registry's reqid — the registry mints its own "<seq>.<pid>@<host>" id — so
16 * the ADR-2b sidecar (reqid_map) carries cms_reqid → engine_reqid plus the
17 * notify/prty fields the registry view has no columns for.
18 *
19 * HOW: cms_prep_path_ok is the pure path gate (absolute, bounded, no "..";
20 * the registry records logical paths, so a lexical gate is the confinement
21 * seam here — no filesystem I/O happens at admission). cms_node_exec_prepare
22 * decodes via the shared rrdata/node_ops planner and routes on plan->action.
23 */

34/* cms_prep_path_ok — lexical confinement gate for a forwarded stage path:
35 * absolute, shorter than the registry's LFN cap, and free of ".." segments
36 * (a hostile manager must not park an escape path in the durable store that
37 * a later recall would resolve). Returns 1 = acceptable. */
38static int
39cms_prep_path_ok(const char *path)
40{
41 if (path == NULL || path[0] != '/'
42 || ngx_strlen(path) >= BRIX_STAGE_LFN_LEN)
43 {
44 return 0;
45 }
46 return ngx_strstr(path, "..") == NULL;
47}

49/* cms_prep_add — admit a forwarded prepadd into the stage-request registry
50 * and record the manager's reqid in the sidecar map. kYR_error on refusal,
51 * silent on success (stock cmsd contract). */
52static ngx_int_t
53cms_prep_add(ngx_brix_cms_ctx_t *ctx, uint32_t streamid,
54 const brix_cms_rrdata_t *d, const brix_cms_node_plan_t *plan)
55{
56 brix_stage_registry_t *reg = brix_stage_registry_singleton();
57 brix_stage_request_view_t v;
58 char engine_reqid[BRIX_STAGE_REQID_LEN];
59 char ident[BRIX_STAGE_DN_LEN];
60
61 if (reg == NULL) {
62 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
63 "no staging engine");
64 }
65 if (!cms_prep_path_ok(plan->path)) {
66 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
67 "prep path denied");
68 }
69
70 ngx_memzero(&v, sizeof(v));
71 v.lfn = plan->path;
72
73 /* The padArgs ident is the manager-forwarded requester identity; it
74 * becomes the owner key so a later owner-checked cancel matches. */
75 if (d->ident != NULL && d->ident_len > 0
76 && d->ident_len < sizeof(ident))
77 {
78 ngx_memcpy(ident, d->ident, d->ident_len);
79 ident[d->ident_len] = '\0';
80 v.requester_dn = ident;
81 }
82
83 if (ctx->conf->frm.stage_ttl > 0) {
84 v.tod_expire = (int64_t) time(NULL)
85 + (int64_t) (ctx->conf->frm.stage_ttl / 1000);
86 }
87
88 if (brix_stage_request_add(reg, &v, engine_reqid,
89 sizeof(engine_reqid),
90 ctx->cycle->log) != NGX_OK)
91 {
92 return ngx_brix_cms_send_error(ctx, streamid, CMS_ERR_EINVAL,
93 "stage admission refused");
94 }
95
96 brix_cms_reqid_map_put(plan->reqid, engine_reqid,
97 plan->notify, plan->prty);
98
99 ngx_log_error(NGX_LOG_INFO, ctx->cycle->log, 0,
100 "brix: CMS prepadd \"%s\" admitted (cms reqid \"%s\" -> "
101 "engine \"%s\")", plan->path, plan->reqid, engine_reqid);
102 return NGX_OK; /* silent success */
103}

105/* cms_prep_del — consume the sidecar mapping and delete the registry row.
106 * Idempotent and silent either way (an unknown reqid is stock-cmsd silence,
107 * not an error — the request may have been reaped already). */
108static ngx_int_t
109cms_prep_del(ngx_brix_cms_ctx_t *ctx, const brix_cms_node_plan_t *plan)
110{
111 brix_stage_registry_t *reg = brix_stage_registry_singleton();
112 char engine_reqid[BRIX_STAGE_REQID_LEN];
113
114 if (reg != NULL
115 && brix_cms_reqid_map_take(plan->reqid, engine_reqid,
116 sizeof(engine_reqid)))
117 {
118 (void) brix_stage_request_delete(reg, engine_reqid, ctx->cycle->log);
119 ngx_log_error(NGX_LOG_INFO, ctx->cycle->log, 0,
120 "brix: CMS prepdel cms reqid \"%s\" -> engine \"%s\" "
121 "deleted", plan->reqid, engine_reqid);
122 }
123 return NGX_OK; /* idempotent, silent */
124}

128ngx_int_t
129cms_node_exec_prepare(ngx_brix_cms_ctx_t *ctx, u_char code,
