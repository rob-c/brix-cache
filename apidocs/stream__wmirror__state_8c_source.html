1/*
2 * stream_wmirror_state.c — XRootD stream data-write mirror: replay protocol
3 * state machine (see stream_wmirror_internal.h).
4 *
5 * WHAT: The per-phase half of the detached replay. Builds and sends each
6 * protocol frame (open -> write -> close) to the shadow, decides what a
7 * just-received response means, and drives the linear phase progression.
8 *
9 * WHY: Split out of stream_wmirror.c under the phase-79 500-line cap. The frame
10 * construction and the six per-phase decisions form one cohesive concern —
11 * the shadow-side protocol — that reads cleanly apart from the socket
12 * lifecycle and event plumbing (stream_wmirror_replay.c).
13 *
14 * HOW: wmir_read_handler (replay.c) parses one frame and calls wmir_dispatch,
15 * which indexes the wmir_step_table by phase. A request phase queues its
16 * next frame via wmir_send_* (each of which flushes and re-arms the read);
17 * a bootstrap phase advances the phase and re-posts the read to drain
18 * pipelined frames. All buffer draining goes through wmir_flush and every
19 * failure through wmir_finish — both owned by stream_wmirror_replay.c.
20 */
21#include "stream_wmirror.h"
22#include "stream_wmirror_internal.h"

27/* Clear the response accumulator before awaiting the next frame: zero the
28 * header read cursor and the parsed status/dlen/body so a stale value from the
29 * previous phase cannot leak into the next response. */
30static void
31wmir_reset_frame(brix_wmirror_replay_t *r)
32{
33 r->rhdr_pos = 0;
34 r->resp_status = 0;
35 r->resp_dlen = 0;
36 r->resp_body = NULL;
37 r->resp_body_pos = 0;
38}

40/*
41 * Replay the client's original open frame (header[24] + path/cgi payload) to the
42 * shadow. WHAT: re-sends the captured open verbatim so the shadow recreates the
43 * same path under its own namespace. HOW: copy into a fresh pool buffer (the
44 * saved open_frame is reused per launch and must stay intact), overwrite the
45 * streamid to 0x0002 (marks this as a mirror request, matching the read mirror),
46 * then flush. The original open's create/truncate flags are preserved as-is.
47 */
48static void
49wmir_send_open(brix_wmirror_replay_t *r)
50{
51 u_char *p = ngx_palloc(r->pool, r->open_frame_len);
52
53 if (p == NULL) { wmir_finish(r, 0); return; }
54 ngx_memcpy(p, r->open_frame, r->open_frame_len);
55 p[0] = 0; p[1] = 2; /* streamid 0x0002 */
56 r->wbuf = p; r->wbuf_len = r->open_frame_len; r->wbuf_pos = 0;
57 wmir_reset_frame(r);
58 r->phase = WMIR_OPEN;
59 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
60}

62/*
63 * Send the whole accumulated file as a SINGLE kXR_write to the shadow.
64 *
65 * WHAT/WHY: the accumulator only ever captured sequential writes starting at
66 * offset 0 (wmir_observe enforces contiguity), so the entire file collapses into
67 * one write at offset 0 — we do NOT need to replay the client's original chunk
68 * boundaries. HOW: hand-build the 24-byte ClientWriteRequest header:
69 * [0..1] streamid = 0x0002 (mirror marker)
70 * [2..3] requestid = kXR_write, big-endian
71 * [4..7] fhandle = the shadow's handle from the open response
72 * [8..15] offset = 0 (whole file from start; off_be is already zero)
73 * [16..19] reserved (left zero by ngx_memzero)
74 * [20..23] dlen = data_len, big-endian
75 * followed by data_len payload bytes. All multi-byte fields are network order.
76 */
77static void
78wmir_send_write(brix_wmirror_replay_t *r)
79{
80 size_t total = (size_t) 24 + r->data_len;
81 u_char *p = ngx_palloc(r->pool, total);
82 uint32_t dlen_be;
83 uint64_t off_be = 0;
84
85 if (p == NULL) { wmir_finish(r, 0); return; }
86 ngx_memzero(p, 24);
87 p[0] = 0; p[1] = 2;
88 p[2] = (u_char) (kXR_write >> 8); p[3] = (u_char) (kXR_write & 0xff);
89 ngx_memcpy(p + 4, r->shadow_fhandle, 4);
90 ngx_memcpy(p + 8, &off_be, 8); /* whole file at offset 0 */
91 dlen_be = htonl((uint32_t) r->data_len);
92 ngx_memcpy(p + 20, &dlen_be, 4);
93 if (r->data_len) { ngx_memcpy(p + 24, r->data, r->data_len); }
94 r->wbuf = p; r->wbuf_len = total; r->wbuf_pos = 0;
95 wmir_reset_frame(r);
96 r->phase = WMIR_WRITE;
97 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
98}

100/*
101 * Send kXR_close for the shadow handle — finalizes the replayed file. Same
102 * 24-byte header layout as the write: streamid 0x0002, requestid kXR_close at
103 * [2..3], the shadow fhandle at [4..7]; all other fields stay zero.
104 */
105static void
106wmir_send_close(brix_wmirror_replay_t *r)
107{
108 u_char *p = ngx_palloc(r->pool, 24);
109
110 if (p == NULL) { wmir_finish(r, 0); return; }
111 ngx_memzero(p, 24);
112 p[0] = 0; p[1] = 2;
113 p[2] = (u_char) (kXR_close >> 8); p[3] = (u_char) (kXR_close & 0xff);
114 ngx_memcpy(p + 4, r->shadow_fhandle, 4);
115 r->wbuf = p; r->wbuf_len = 24; r->wbuf_pos = 0;
116 wmir_reset_frame(r);
117 r->phase = WMIR_CLOSE;
118 if (wmir_flush(r) == NGX_ERROR) { wmir_finish(r, 0); }
119}

121/* A shadow op succeeded if it returned kXR_ok or kXR_oksofar (the latter is a
122 * partial-but-fine status the server may use for streamed writes). */
123static int
124wmir_status_ok(uint16_t st)
125{
126 return st == kXR_ok || st == kXR_oksofar;
127}

129/*
130 * Outcome of a single per-phase step, telling wmir_dispatch what to do next.
131 * WHAT: decouples the phase decision from the loop plumbing so each phase handler
132 * stays a small pure decision. WHY: a request phase queues its own next frame
133 * (whose flush re-arms the read) and is DONE; a bootstrap phase only advanced the
134 * state and needs the read re-posted to drain pipelined frames; a failure tears
135 * down. HOW: handlers return one of these and never touch the loop directly.
136 */
137typedef enum {
138 WMIR_STEP_FAIL = 0, /* terminal failure — finish(r, 0) */
139 WMIR_STEP_SENT, /* request phase queued its next frame — done */
140 WMIR_STEP_REPOST, /* bootstrap phase advanced — re-post the read */
141} wmir_step_t;
142
143/*
144 * Bootstrap-handshake phase: the shadow's handshake response must be kXR_ok.
145 * WHAT: validate the handshake and advance to PROTOCOL. HOW: any non-ok status
146 * fails; otherwise move to the next bootstrap phase and re-post the read.
147 */
148static wmir_step_t
149wmir_step_handshake(brix_wmirror_replay_t *r)
150{
151 if (r->resp_status != kXR_ok) { return WMIR_STEP_FAIL; }
152 r->phase = WMIR_PROTOCOL;
153 return WMIR_STEP_REPOST;
154}
155
156/*
157 * Bootstrap-protocol phase: validate the ClientProtocol response. WHAT: require
158 * kXR_ok and reject a shadow that demands TLS. WHY: a gotoTLS shadow cannot
159 * continue the cleartext replay. HOW: the flag lives in the response body at
160 * byte offset 4 (a 4-byte big-endian word); if set, fail; else advance to LOGIN.
161 */
162static wmir_step_t
163wmir_step_protocol(brix_wmirror_replay_t *r)
164{
165 if (r->resp_status != kXR_ok) { return WMIR_STEP_FAIL; }
166 if (r->resp_dlen >= 8 && r->resp_body != NULL) {
167 uint32_t flags_be;
168 ngx_memcpy(&flags_be, r->resp_body + 4, sizeof(flags_be));
169 if (ntohl(flags_be) & kXR_gotoTLS) { return WMIR_STEP_FAIL; }
170 }
171 r->phase = WMIR_LOGIN;
172 return WMIR_STEP_REPOST;
173}
174
175/*
176 * Bootstrap-login phase: on a clean login, send the shadow open. WHAT: require a
177 * plain kXR_ok. WHY: kXR_authmore means the shadow wants credentials we cannot
178 * supply on this anonymous replay (non-fatal stop, counted as error). HOW: on
179 * success queue the open frame (which arms the read itself) and report SENT.
180 */
181static wmir_step_t
182wmir_step_login(brix_wmirror_replay_t *r)
183{
184 if (r->resp_status != kXR_ok) { return WMIR_STEP_FAIL; }
185 wmir_send_open(r); /* resets accumulator + sends + arms read */
186 return WMIR_STEP_SENT;
187}
188
189/*
190 * Open-response phase: capture the shadow file handle, then send the write.
191 * WHAT: the open response body begins with the shadow's 4-byte fhandle that all
192 * downstream ops address. HOW: a bad status or a short/missing body fails; else
193 * copy the handle and queue the single whole-file write.
194 */
195static wmir_step_t
196wmir_step_open(brix_wmirror_replay_t *r)
197{
198 if (!wmir_status_ok(r->resp_status) || r->resp_dlen < 4
199 || r->resp_body == NULL)
200 {
201 return WMIR_STEP_FAIL; /* shadow open failed/diverged */
202 }
203 ngx_memcpy(r->shadow_fhandle, r->resp_body, 4);
204 wmir_send_write(r);
205 return WMIR_STEP_SENT;
206}
207
208/*
209 * Write-response phase: on an accepted write, send the close. WHAT: require an
210 * ok/oksofar status. HOW: on success queue the close frame and report SENT.
211 */
212static wmir_step_t
213wmir_step_write(brix_wmirror_replay_t *r)
214{
215 if (!wmir_status_ok(r->resp_status)) { return WMIR_STEP_FAIL; }
216 wmir_send_close(r);
217 return WMIR_STEP_SENT;
218}
219
220/*
221 * Close-response phase: terminal success. WHAT: this is the only path that
222 * reports the replay fully succeeded (ok=1) — iff the close was accepted.
223 */
224static wmir_step_t
225wmir_step_close(brix_wmirror_replay_t *r)
226{
227 wmir_finish(r, wmir_status_ok(r->resp_status) ? 1 : 0);
228 return WMIR_STEP_SENT; /* already finished; do not re-post */
229}

231/*
232 * Re-post the read after a bootstrap phase advanced. WHAT: clear the frame
233 * accumulator and re-arm+post the read event. WHY: the pipelined bootstrap
234 * responses may already be sitting in the socket buffer, so re-posting drains
235 * them within the same loop cycle rather than waiting for a fresh readable event.
236 */
237static void
238wmir_repost_read(brix_wmirror_replay_t *r)
239{
240 wmir_reset_frame(r);
241 if (ngx_handle_read_event(r->conn->read, 0) != NGX_OK) {
242 wmir_finish(r, 0);
243 return;
244 }
245 ngx_post_event(r->conn->read, &ngx_posted_events);
246}

248/* Per-phase step handlers, indexed by wmir_phase_t. Table-driven dispatch keeps
249 * the flow linear and each phase in one focused function. */
250static wmir_step_t (*const wmir_step_table[])(brix_wmirror_replay_t *) = {
251 [WMIR_HANDSHAKE] = wmir_step_handshake,
252 [WMIR_PROTOCOL] = wmir_step_protocol,
253 [WMIR_LOGIN] = wmir_step_login,
254 [WMIR_OPEN] = wmir_step_open,
255 [WMIR_WRITE] = wmir_step_write,
256 [WMIR_CLOSE] = wmir_step_close,
257};
258
259/* Advance one phase; bootstrap phases re-post the read so pipelined frames are
260 * processed; request phases send the next frame (which arms the read itself). */
261void
262wmir_dispatch(brix_wmirror_replay_t *r)
263{
264 wmir_step_t step = wmir_step_table[r->phase](r);
265
266 if (step == WMIR_STEP_FAIL) { wmir_finish(r, 0); return; }
267 if (step == WMIR_STEP_REPOST) { wmir_repost_read(r); }
268 /* WMIR_STEP_SENT: the phase already queued its next frame (or finished). */
269}
