184/* Append a strdup'd copy of `s` to a growable string array. 0 / -1. */
185int
186str_append(char ***list, size_t *n, size_t *cap, const char *s)
187{
188 if (*n == *cap) {
189 size_t nc = *cap ? *cap * 2 : 16;
190 char **na = (char **) realloc(*list, nc * sizeof(char *));
191 if (na == NULL) { return -1; }
192 *list = na;
193 *cap = nc;
194 }
195 (*list)[*n] = strdup(s);
196 if ((*list)[*n] == NULL) { return -1; }
197 (*n)++;
198 return 0;
199}

211/* Extract the alias name from `arg` (the part before the first ':') into name[sz].
212 * Returns 1 if `arg` looks like an alias reference, 0 otherwise (no colon, empty or
213 * oversized name, or a scheme:// URL). */
214static int
215alias_name_of(const char *arg, char *name, size_t sz)
216{
217 const char *colon = strchr(arg, ':');
218 size_t nlen;
219
220 if (colon == NULL) { return 0; }
221 nlen = (size_t) (colon - arg);
222 if (nlen == 0 || nlen >= sz) { return 0; }
223 if (colon[1] == '/' && colon[2] == '/') { return 0; } /* scheme:// — not an alias */
224 memcpy(name, arg, nlen);
225 name[nlen] = '\0';
226 return 1;
227}

230/* Fold the alias `info`'s per-endpoint credentials into `o` — a value already set
231 * (CLI flag or earlier alias) always wins. The opt pointers are backed by static
232 * storage for the process lifetime. PII: creds are never logged. */
233static void
234fold_alias_creds(const brix_alias_info *info, brix_copy_opts *o)
235{
236 static char s_bearer[8192], s_access[256], s_secret[256], s_region[64];
237
238 if (o->bearer == NULL && info->bearer[0] != '\0') {
239 snprintf(s_bearer, sizeof(s_bearer), "%s", info->bearer);
240 o->bearer = s_bearer;
241 }
242 /* Fold the S3 access/secret as ONE unit so a mismatched key pair can never be
243 * assembled from two different aliases. */
244 if (o->s3_access == NULL && o->s3_secret == NULL
245 && info->s3_access[0] != '\0' && info->s3_secret[0] != '\0') {
246 snprintf(s_access, sizeof(s_access), "%s", info->s3_access);
247 snprintf(s_secret, sizeof(s_secret), "%s", info->s3_secret);
248 o->s3_access = s_access;
249 o->s3_secret = s_secret;
250 }
251 if (o->s3_region == NULL && info->s3_region[0] != '\0') {
252 snprintf(s_region, sizeof(s_region), "%s", info->s3_region);
253 o->s3_region = s_region;
254 }
255 if (info->proxy[0] != '\0') {
256 setenv("X509_USER_PROXY", info->proxy, 0); /* 0 = don't clobber an existing env */
257 }
258}

261/* If `arg` names a ~/.xrdrc alias, fold its per-endpoint credentials into `o` — a
262 * value already set (CLI flag or earlier alias) always wins. The opt pointers are
263 * backed by static storage for the process lifetime. PII: creds are never logged. */
264void
265merge_alias_auth(const char *arg, brix_copy_opts *o)
266{
267 char name[256];
268 brix_alias_info info;
269
270 if (!alias_name_of(arg, name, sizeof(name))) { return; }
271 if (!brix_alias_lookup(name, &info)) { return; }
272
273 if (info.token_file_failed) {
274 fprintf(stderr, "xrdcp: alias %s: token_file %s missing or empty\n",
275 name, info.token_file);
276 }
277 fold_alias_creds(&info, o);
278}

281/* Copy the basename of a path/URL (after the last '/', ignoring trailing slashes)
282 * into out[sz]. out is empty if the input is all slashes. */
283void
284path_basename(const char *p, char *out, size_t sz)
285{
286 size_t len = strlen(p), start = 0, i, bl;
287 while (len > 0 && p[len - 1] == '/') { len--; } /* ignore trailing slashes */
288 for (i = 0; i < len; i++) {
289 if (p[i] == '/') { start = i + 1; }
290 }
291 bl = len - start;
292 if (bl >= sz) { bl = sz - 1; }
293 memcpy(out, p + start, bl);
294 out[bl] = '\0';
295}

298/* Trim leading/trailing whitespace (and CR/LF) from `line` in place and return a
299 * pointer to the first non-blank char. The returned string is "" for a blank line. */
300static char *
301manifest_trim(char *line)
302{
303 char *s = line, *e;
304 while (*s == ' ' || *s == '\t') { s++; }
305 e = s + strlen(s);
306 while (e > s && (e[-1] == '\n' || e[-1] == '\r' || e[-1] == ' ' || e[-1] == '\t')) {
307 *--e = '\0';
308 }
309 return s;
310}

313/* Read a manifest (one source per line; '#' comments + blank lines skipped) and
314 * append each entry to the source list. 0 / -1. */
315int
316read_manifest(const char *file, char ***list, size_t *n, size_t *cap)
317{
318 FILE *f = (strcmp(file, "-") == 0) ? stdin : fopen(file, "r");
319 char line[XRDC_PATH_MAX];
320
321 if (f == NULL) {
322 fprintf(stderr, "xrdcp: cannot open manifest %s: %s\n", file, strerror(errno));
323 return -1;
324 }
325 while (fgets(line, sizeof(line), f) != NULL) {
326 char *s = manifest_trim(line);
327 if (*s == '\0' || *s == '#') { continue; }
328 if (str_append(list, n, cap, s) != 0) {
329 if (f != stdin) { fclose(f); }
330 return -1;
331 }
332 }
333 if (f != stdin) { fclose(f); }
334 return 0;
335}

375/* Is `dst` an existing directory? 1=yes, 0=no, -1=can't determine (web/parse). */
376int
377dest_is_dir(const char *dst, const brix_opts *co)
378{
379 brix_url u;
380 brix_status st;
381 if (brix_is_web_url(dst)) { return -1; }
382 brix_status_clear(&st);
383 if (brix_url_parse(dst, &u, &st) != 0) { return -1; }
384 if (u.scheme == XRDC_SCHEME_LOCAL) {
385 struct stat sb;
386 return (stat(u.path, &sb) == 0 && S_ISDIR(sb.st_mode)) ? 1 : 0;
387 }
388 if (u.scheme == XRDC_SCHEME_ROOT || u.scheme == XRDC_SCHEME_ROOTS) {
389 brix_conn c;
390 brix_statinfo si;
391 int isdir;
392 if (brix_connect(&c, &u, co, &st) != 0) { return -1; }
393 isdir = (brix_stat(&c, u.path, &si, &st) == 0 && (si.flags & kXR_isDir)) ? 1 : 0;
394 brix_close(&c);
395 return isdir;
396 }
397 return -1;

448
449/* brix_copy progress callback: a throttled \r-updated stderr bar with rate + ETA. */
450void
451xrdcp_progress(void *arg, long long done, long long total)
452{
453 xrdcp_prog *p = (xrdcp_prog *) arg;
454 uint64_t now = brix_mono_ns();
455 int final = (total >= 0 && done >= total);
456 double secs, mb, rate;
457
458 if (!final && (now - p->last_ns) < 200000000ULL) {
459 return; /* throttle intermediate updates to ~5 Hz */
460 }
461 p->last_ns = now;
462 secs = (double) (now - p->start_ns) / 1e9;
463 mb = (double) done / 1048576.0;
464 rate = (secs > 0.01) ? mb / secs : 0.0;
465 if (total > 0) {
466 int pct = (int) ((done * 100) / total);
467 double tmb = (double) total / 1048576.0;
468 double eta = (rate > 0.01) ? (tmb - mb) / rate : 0.0;
469 fprintf(stderr, "\r%-28s %3d%% %.1f/%.1f MiB %.1f MiB/s ETA %3.0fs ",
470 p->label, pct, mb, tmb, rate, eta);
471 } else {
472 fprintf(stderr, "\r%-28s %.1f MiB %.1f MiB/s ", p->label, mb, rate);
473 }
474 if (final) {
475 fprintf(stderr, "\n");
476 }
477 fflush(stderr);
478}

481/* ========================================================================
482 * WHAT: Validate argument combinations and build source/destination lists
483 * WHY: Many flag combinations are invalid (--delete requires -r+--sync,
484 * --delete conflicts with --remove-source, --verify implies --cksum, etc.)
485 * HOW: Check constraints, derive implicit settings, build srcs list from pos
486 * and --from manifest, derive journal path from --resume+--from
487 * PARAMS: All the parsed values from parse_and_validate_args
488 * RETURNS: 0 on success, 50 on usage error, 51 on OOM
489 * ======================================================================== */
490static int
491xrdcp_validate_flag_matrix(brix_copy_opts *opts, int sync_mode, int verify)
492{
493 /* --sync replaces destinations that differ, so the files it does copy must be
494 * allowed to overwrite (skipped ones are left untouched by the size check). */
495 if (sync_mode) {
496 opts->force = 1;
497 }
498 opts->sync = sync_mode; /* recursive walkers read o->sync (+ sync_cmp/algo) */
499
500 /* --delete (mirror: make the destination match the source) and
501 * --remove-source (move: delete each source once its transfer succeeds) are
502 * contradictory. Run together they destroy BOTH trees: on an upload the
503 * per-file source unlink runs before the mirror-delete pass, which then sees
504 * the now-missing local files and purges the freshly-uploaded remote copies.
505 * Reject the pair before any bytes (or unlinks) move. */
506 if (opts->sync_delete && opts->remove_source) {
507 fprintf(stderr, "xrdcp: --delete and --remove-source are contradictory "
508 "(mirror vs move)\n");
509 return 50;
510 }
511
512 /* --delete requires -r and --sync: without a recursive pass there is no
513 * listing to diff against; without --sync the extra-deletion semantics are
514 * ill-defined (we might delete a destination the caller wanted to keep). */
515 if (opts->sync_delete && !(opts->recursive && sync_mode)) {
516 fprintf(stderr, "xrdcp: --delete requires -r and --sync\n");
517 return 50;
518 }
519
520 /* --verify: post-transfer checksum against the server. An explicit --cksum wins. */
521 if (verify && opts->cksum == NULL) {
522 opts->cksum = "adler32:source";
523 }
524
525 return 0;
526}

909/*
910 * WHAT: Parse and validate command-line arguments.
911 *
912 * WHY: The main() function is CCN 187 (527 lines). Extracting the argument
913 * parsing and validation logic reduces complexity and improves testability.
914 *
915 * HOW: Parse all CLI flags into the provided structures, validate flag
916 * interactions (--sync implies --force, --delete requires -r + --sync,
917 * etc.), build positional/exclusion/inclusion lists, and derive defaults.
918 * Returns 0 on success, 50 on usage error, 51 on OOM.
919 */
920static int
921parse_and_validate_args(int argc, char **argv, xrdcp_opts_t *o, xrdcp_lists_t *l)
922{
923 size_t i;
924 xrdcp_cli_state state;
925
926 memset(&state, 0, sizeof(state));
927 state.o = o;
928 state.l = l;
929
930 /* Phase 44: XRDC_IO_URING env is the default (auto) for the local-disk
931 * overlap ring; --io-uring overrides it below. auto = 0 = memset default. */
932 {
933 const char *e = getenv("XRDC_IO_URING");
934 if (e != NULL) {
935 if (strcmp(e, "on") == 0) { o->copt->io_uring = XRDC_IO_URING_ON; }
936 else if (strcmp(e, "off") == 0) { o->copt->io_uring = XRDC_IO_URING_OFF; }
937 else { o->copt->io_uring = XRDC_IO_URING_AUTO; }
938 }
939 }
940
941 for (i = 1; i < (size_t) argc; i++) {
942 const char *a = argv[i];
943 if (a[0] == '-' && a[1] != '\0' && strcmp(a, "-") != 0) {
944 int parsed = xrdcp_parse_option(&state, argc, argv, &i);
945 if (parsed == 2) { return XRDCP_PARSE_EXIT_OK; }
946 if (parsed == 50) { return 50; }
947 } else if (str_append(&l->pos.items, &l->pos.n, &l->pos.cap, a) != 0) {
948 fprintf(stderr, "xrdcp: out of memory\n");
949 return 51;
950 }
951 }
952 if (state.oom) {
953 fprintf(stderr, "xrdcp: out of memory\n");
954 return 51;
955 }
956
957 o->copt->excludes = (const char *const *) l->excl.items;
958 o->copt->n_excludes = l->excl.n;
959 o->copt->includes = (const char *const *) l->incl.items;
960 o->copt->n_includes = l->incl.n;
961
962 return validate_and_finalize_args(o, l);
963}

966/*
967 * WHAT: Build credential store after alias resolution, glob expansion, and pre-flight.
968 *
969 * WHY: Credential store construction requires expanded sources to determine auth
970 * needs, plus pre-flight validation to warn about expired/read-only credentials.
971 *
972 * HOW: Merge ~/.xrdrc alias credentials, expand globs, validate --remove-source
973 * compatibility with web sources, run credential pre-flight (auto-refresh +
974 * diagnose), then build the credential store. Returns store on success,
975 * NULL on error (with cleanup of passed-in arrays).
976 */
977/* Free every strdup-owned vector main threads through the pipeline. Called on
978 * the credential-build error paths and (via main) on normal exit. */
979static void
980xrdcp_lists_free(xrdcp_lists_t *l)
981{
982 str_free(l->pos.items, l->pos.n);
983 str_free(l->srcs.items, l->srcs.n);
984 str_free(l->exp.items, l->exp.n);
985 str_free(l->excl.items, l->excl.n);
986 str_free(l->incl.items, l->incl.n);
987}

989
990static struct brix_cred_store *
991build_and_preflight_credentials(xrdcp_opts_t *o, xrdcp_lists_t *l)
992{
993 brix_copy_opts *opts = o->copt;
994 size_t i;
995
996 /* Fold any ~/.xrdrc per-endpoint credentials (the dst + every source alias) into
997 * opts so `xrdcp s3lab:/obj .` authenticates with no flags. */
998 merge_alias_auth(l->pos.items[l->pos.n - 1], opts);
999 for (i = 0; i < l->srcs.n; i++) {
1000 merge_alias_auth(l->srcs.items[i], opts);
1001 }
1002
1003 /* Expand globs (root:// + local) into the final source list. */
1004 for (i = 0; i < l->srcs.n; i++) {
1005 if (expand_source(l->srcs.items[i], o->conn,
1006 &l->exp.items, &l->exp.n, &l->exp.cap) != 0) {
1007 fprintf(stderr, "xrdcp: out of memory\n");
1008 xrdcp_lists_free(l);
1009 return NULL;
1010 }
1011 }
1012 if (l->exp.n == 0) {
1013 fprintf(stderr, "xrdcp: no sources after expansion\n");
1014 xrdcp_lists_free(l);
1015 return NULL;
1016 }
1017 /* --remove-source supports local and root:// sources only: web/S3 sources
1018 * have no cheap post-transfer delete path and cannot be safely removed. */
1019 if (opts->remove_source) {
1020 for (i = 0; i < l->exp.n; i++) {
1021 if (brix_is_web_url(l->exp.items[i])) {
1022 fprintf(stderr, "xrdcp: --remove-source supports local and "
1023 "root:// sources only\n");
1024 xrdcp_lists_free(l);
1025 return NULL;
1026 }
1027 }
1028 }
1029
1030 /*
1031 * Phase 40 (c): credential pre-flight. If a server endpoint is involved, warn
1032 * the user INSTANTLY about a locally-detectable auth problem (expired/near-
1033 * expiry bearer token or GSI proxy, or a read-only token on an upload) before
1034 * the transfer fails with a bare "permission denied". Silent when creds look
1035 * fine; never aborts (the server stays authoritative).
1036 */
1037 {
1038 /* Only the GSI/token credential family is diagnosable here — an s3://
1039 * endpoint authenticates with AWS SigV4 keys, so it must not trip the
1040 * "GSI proxy expired" / "bearer token" pre-flight. */
1041 int dst_cred = uses_cred_auth(o->dst);
1042 int any_cred = dst_cred;
1043 for (i = 0; i < l->exp.n && !any_cred; i++) {
1044 if (uses_cred_auth(l->exp.items[i])) { any_cred = 1; }
1045 }
1046 if (any_cred) {
1047 /* Phase 40 (b): if asked, proactively (re)acquire a stale token/proxy
1048 * BEFORE diagnosing — so a healthy refresh leaves nothing to warn. */
1049 if (o->auto_refresh) {
1050 (void) brix_cred_autorefresh(dst_cred, o->oidc_account,
1051 !opts->silent, stderr);
1052 }
1053 (void) brix_cred_diagnose(dst_cred, "xrdcp: ", stderr);
1054 }
1055 }
1056
1057 /* C1: build the credential store from CLI values and attach it to the
1058 * connection options. The store is INERT here — nothing reads conn.cred
1059 * yet; C2 will thread it through the auth/token handshake path. Building
1060 * it now (before the transfer) means C2 only needs to consume conn.cred,
1061 * not rebuild it. NULL/empty args fall back to per-handler env discovery. */
1062 return brix_cli_cred_store_build(o->proxy, opts->bearer, NULL,
1063 opts->s3_access, opts->s3_secret,
1064 o->oidc_account, o->auto_refresh);
1065}

1164xrdcp_enable_progress_if_needed(xrdcp_transfer_ctx *ctx, xrdcp_prog *ps,
