84/* Append a strdup'd copy of `s` to a growable string array. 0 / -1. */
85int
86str_append(char ***list, size_t *n, size_t *cap, const char *s)
87{
88 if (*n == *cap) {
89 size_t nc = *cap ? *cap * 2 : 16;
90 char **na = (char **) realloc(*list, nc * sizeof(char *));
91 if (na == NULL) { return -1; }
92 *list = na;
93 *cap = nc;
94 }
95 (*list)[*n] = strdup(s);
96 if ((*list)[*n] == NULL) { return -1; }
97 (*n)++;
98 return 0;
99}

111/* If `arg` names a ~/.xrdrc alias, fold its per-endpoint credentials into `o` — a
112 * value already set (CLI flag or earlier alias) always wins. The opt pointers are
113 * backed by static storage for the process lifetime. PII: creds are never logged. */
114void
115merge_alias_auth(const char *arg, brix_copy_opts *o)
116{
117 const char *colon = strchr(arg, ':');
118 char name[256];
119 size_t nlen;
120 brix_alias_info info;
121 static char s_bearer[8192], s_access[256], s_secret[256], s_region[64];
122
123 if (colon == NULL) { return; }
124 nlen = (size_t) (colon - arg);
125 if (nlen == 0 || nlen >= sizeof(name)) { return; }
126 if (colon[1] == '/' && colon[2] == '/') { return; } /* scheme:// — not an alias */
127 memcpy(name, arg, nlen);
128 name[nlen] = '\0';
129 if (!brix_alias_lookup(name, &info)) { return; }
130
131 if (info.token_file_failed) {
132 fprintf(stderr, "xrdcp: alias %s: token_file %s missing or empty\n",
133 name, info.token_file);
134 }
135 if (o->bearer == NULL && info.bearer[0] != '\0') {
136 snprintf(s_bearer, sizeof(s_bearer), "%s", info.bearer);
137 o->bearer = s_bearer;
138 }
139 /* Fold the S3 access/secret as ONE unit so a mismatched key pair can never be
140 * assembled from two different aliases. */
141 if (o->s3_access == NULL && o->s3_secret == NULL
142 && info.s3_access[0] != '\0' && info.s3_secret[0] != '\0') {
143 snprintf(s_access, sizeof(s_access), "%s", info.s3_access);
144 snprintf(s_secret, sizeof(s_secret), "%s", info.s3_secret);
145 o->s3_access = s_access;
146 o->s3_secret = s_secret;
147 }
148 if (o->s3_region == NULL && info.s3_region[0] != '\0') {
149 snprintf(s_region, sizeof(s_region), "%s", info.s3_region);
150 o->s3_region = s_region;
151 }
152 if (info.proxy[0] != '\0') {
153 setenv("X509_USER_PROXY", info.proxy, 0); /* 0 = don't clobber an existing env */
154 }
155}

158/* Copy the basename of a path/URL (after the last '/', ignoring trailing slashes)
159 * into out[sz]. out is empty if the input is all slashes. */
160void
161path_basename(const char *p, char *out, size_t sz)
162{
163 size_t len = strlen(p), start = 0, i, bl;
164 while (len > 0 && p[len - 1] == '/') { len--; } /* ignore trailing slashes */
165 for (i = 0; i < len; i++) {
166 if (p[i] == '/') { start = i + 1; }
167 }
168 bl = len - start;
169 if (bl >= sz) { bl = sz - 1; }
170 memcpy(out, p + start, bl);
171 out[bl] = '\0';
172}

175/* Read a manifest (one source per line; '#' comments + blank lines skipped) and
176 * append each entry to the source list. 0 / -1. */
177int
178read_manifest(const char *file, char ***list, size_t *n, size_t *cap)
179{
180 FILE *f = (strcmp(file, "-") == 0) ? stdin : fopen(file, "r");
181 char line[XRDC_PATH_MAX];
182
183 if (f == NULL) {
184 fprintf(stderr, "xrdcp: cannot open manifest %s: %s\n", file, strerror(errno));
185 return -1;
186 }
187 while (fgets(line, sizeof(line), f) != NULL) {
188 char *s = line, *e;
189 while (*s == ' ' || *s == '\t') { s++; }
190 e = s + strlen(s);
191 while (e > s && (e[-1] == '\n' || e[-1] == '\r' || e[-1] == ' ' || e[-1] == '\t')) {
192 *--e = '\0';
193 }
194 if (*s == '\0' || *s == '#') { continue; }
195 if (str_append(list, n, cap, s) != 0) {
196 if (f != stdin) { fclose(f); }
197 return -1;
198 }
199 }
200 if (f != stdin) { fclose(f); }
201 return 0;
202}

242/* Is `dst` an existing directory? 1=yes, 0=no, -1=can't determine (web/parse). */
243int
244dest_is_dir(const char *dst, const brix_opts *co)
245{
246 brix_url u;
247 brix_status st;
248 if (brix_is_web_url(dst)) { return -1; }
249 brix_status_clear(&st);
250 if (brix_url_parse(dst, &u, &st) != 0) { return -1; }
251 if (u.scheme == XRDC_SCHEME_LOCAL) {
252 struct stat sb;
253 return (stat(u.path, &sb) == 0 && S_ISDIR(sb.st_mode)) ? 1 : 0;
254 }
255 if (u.scheme == XRDC_SCHEME_ROOT || u.scheme == XRDC_SCHEME_ROOTS) {
256 brix_conn c;
257 brix_statinfo si;
258 int isdir;
259 if (brix_connect(&c, &u, co, &st) != 0) { return -1; }
260 isdir = (brix_stat(&c, u.path, &si, &st) == 0 && (si.flags & kXR_isDir)) ? 1 : 0;
261 brix_close(&c);
262 return isdir;
263 }
264 return -1;
265}

314/* Live progress state for a single transfer (label + timing). */
315
316/* brix_copy progress callback: a throttled \r-updated stderr bar with rate + ETA. */
317void
318xrdcp_progress(void *arg, long long done, long long total)
319{
320 xrdcp_prog *p = (xrdcp_prog *) arg;
321 uint64_t now = brix_mono_ns();
322 int final = (total >= 0 && done >= total);
323 double secs, mb, rate;
324
325 if (!final && (now - p->last_ns) < 200000000ULL) {
326 return; /* throttle intermediate updates to ~5 Hz */
327 }
328 p->last_ns = now;
329 secs = (double) (now - p->start_ns) / 1e9;
330 mb = (double) done / 1048576.0;
331 rate = (secs > 0.01) ? mb / secs : 0.0;
332 if (total > 0) {
333 int pct = (int) ((done * 100) / total);
334 double tmb = (double) total / 1048576.0;
335 double eta = (rate > 0.01) ? (tmb - mb) / rate : 0.0;
336 fprintf(stderr, "\r%-28s %3d%% %.1f/%.1f MiB %.1f MiB/s ETA %3.0fs ",
337 p->label, pct, mb, tmb, rate, eta);
338 } else {
339 fprintf(stderr, "\r%-28s %.1f MiB %.1f MiB/s ", p->label, mb, rate);
340 }
341 if (final) {
342 fprintf(stderr, "\n");
343 }
344 fflush(stderr);
345}

486 else if (strcmp(a, "--tpc-token-mode") == 0 && i + 1 < (size_t) argc) { opts.tpc_token_mode = argv[++i]; }
487 else if ((strcmp(a, "-T") == 0 || strcmp(a, "--token") == 0) && i + 1 < (size_t) argc) { opts.bearer = argv[++i]; }
488 else if (strcmp(a, "--s3-access") == 0 && i + 1 < (size_t) argc) { opts.s3_access = argv[++i]; }
489 else if (strcmp(a, "--s3-secret") == 0 && i + 1 < (size_t) argc) { opts.s3_secret = argv[++i]; }
490 else if (strcmp(a, "--s3-region") == 0 && i + 1 < (size_t) argc) { opts.s3_region = argv[++i]; }
491 else if (strcmp(a, "-V") == 0) {
492 /* -V is xrdcp's legacy version flag; --version is handled by the
493 * shared parser above (exits inside brix_opts_parse_arg). */
494 printf("xrdcp (BriX-Cache client) %s\n", brix_client_version());
495 return 0;
496 }
497 else if (strcmp(a, "-h") == 0) { usage(); return 0; } /* -h → stderr (C1) */
498 else {
499 fprintf(stderr, "xrdcp: unknown option '%s'\n", a);
500 usage();
501 return 50;
502 }
503 } else if (str_append(&pos, &npos, &poscap, a) != 0) {
504 fprintf(stderr, "xrdcp: out of memory\n");
505 return 51;
506 }
507 }
508
509 opts.excludes = (const char *const *) excl;
510 opts.n_excludes = nexcl;
511 opts.includes = (const char *const *) incl;
512 opts.n_includes = nincl;

514 /* --sync replaces destinations that differ, so the files it does copy must be
515 * allowed to overwrite (skipped ones are left untouched by the size check). */
516 if (sync_mode) {
517 opts.force = 1;
518 }
519 opts.sync = sync_mode; /* recursive walkers read o->sync (+ sync_cmp/algo) */
520 /* --delete (mirror: make the destination match the source) and
521 * --remove-source (move: delete each source once its transfer succeeds) are
522 * contradictory. Run together they destroy BOTH trees: on an upload the
523 * per-file source unlink runs before the mirror-delete pass, which then sees
524 * the now-missing local files and purges the freshly-uploaded remote copies.
525 * Reject the pair before any bytes (or unlinks) move. */
526 if (opts.sync_delete && opts.remove_source) {
527 fprintf(stderr, "xrdcp: --delete and --remove-source are contradictory "
528 "(mirror vs move)\n");
529 str_free(pos, npos); str_free(excl, nexcl); str_free(incl, nincl);
530 return 50;
531 }
532 /* --delete requires -r and --sync: without a recursive pass there is no
533 * listing to diff against; without --sync the extra-deletion semantics are
534 * ill-defined (we might delete a destination the caller wanted to keep). */
535 if (opts.sync_delete && !(opts.recursive && sync_mode)) {
536 fprintf(stderr, "xrdcp: --delete requires -r and --sync\n");
537 str_free(pos, npos); str_free(excl, nexcl); str_free(incl, nincl);
538 return 50;
539 }
540 /* --verify: post-transfer checksum against the server. An explicit --cksum wins. */
541 if (verify && opts.cksum == NULL) {
542 opts.cksum = "adler32:source";
543 }
544
545 /* Need a destination (the last positional) and at least one source. */
546 if (npos < 1) {
547 usage();
548 str_free(pos, npos); str_free(excl, nexcl); str_free(incl, nincl);
549 return 50;
550 }
551 {
552 static char dstbuf[XRDC_PATH_MAX];
553 brix_alias_resolve(pos[npos - 1], dstbuf, sizeof(dstbuf)); /* ~/.xrdrc */
554 dst = dstbuf;
555 }
556 for (i = 0; i + 1 < npos; i++) {
557 if (str_append(&srcs, &nsrc, &srccap, pos[i]) != 0) { oom = 1; }
558 }
559 if (from != NULL && read_manifest(from, &srcs, &nsrc, &srccap) != 0) {
560 str_free(pos, npos); str_free(srcs, nsrc);
561 str_free(excl, nexcl); str_free(incl, nincl);
562 return 51;
563 }
564 if (oom) {
565 fprintf(stderr, "xrdcp: out of memory\n");
566 str_free(pos, npos); str_free(srcs, nsrc);
567 str_free(excl, nexcl); str_free(incl, nincl);
568 return 51;
569 }
570 /* --resume shorthand: derive journal path from the manifest path. Must come
571 * before nsrc==0 so the specific error fires even when there are no sources. */
572 if (resume && journal_path == NULL) {
573 if (from == NULL || strcmp(from, "-") == 0) {
574 fprintf(stderr, "xrdcp: --resume needs --from <file> (not stdin) "
575 "or an explicit --journal <path>\n");
576 str_free(pos, npos); str_free(srcs, nsrc);
577 str_free(excl, nexcl); str_free(incl, nincl);
578 return 50;
579 }
580 {
581 static char jbuf[XRDC_PATH_MAX];
582 if ((size_t) snprintf(jbuf, sizeof(jbuf), "%s.journal", from)
583 >= sizeof(jbuf)) {
584 fprintf(stderr, "xrdcp: journal path too long\n");
585 str_free(pos, npos); str_free(srcs, nsrc);
586 str_free(excl, nexcl); str_free(incl, nincl);
587 return 50;
588 }
589 journal_path = jbuf;
590 }
591 }
592 if (nsrc == 0) {
593 fprintf(stderr, "xrdcp: no source given\n");
594 usage();
595 str_free(pos, npos); str_free(excl, nexcl); str_free(incl, nincl);
596 return 50;
597 }

599 /* Fold any ~/.xrdrc per-endpoint credentials (the dst + every source alias) into
600 * opts so `xrdcp s3lab:/obj .` authenticates with no flags. */
601 merge_alias_auth(pos[npos - 1], &opts);
602 for (i = 0; i < nsrc; i++) {
603 merge_alias_auth(srcs[i], &opts);
604 }
605
606 /* Expand globs (root:// + local) into the final source list. */
607 for (i = 0; i < nsrc; i++) {
608 if (expand_source(srcs[i], &conn, &exp, &nexp, &expcap) != 0) {
609 fprintf(stderr, "xrdcp: out of memory\n");
610 str_free(pos, npos); str_free(srcs, nsrc); str_free(exp, nexp);
611 str_free(excl, nexcl); str_free(incl, nincl);
612 return 51;
613 }
614 }
615 if (nexp == 0) {
616 fprintf(stderr, "xrdcp: no sources after expansion\n");
617 str_free(pos, npos); str_free(srcs, nsrc); str_free(exp, nexp);
618 str_free(excl, nexcl); str_free(incl, nincl);
619 return 50;
620 }
621 /* --remove-source supports local and root:// sources only: web/S3 sources
622 * have no cheap post-transfer delete path and cannot be safely removed. */
623 if (opts.remove_source) {
624 for (i = 0; i < nexp; i++) {
625 if (brix_is_web_url(exp[i])) {
626 fprintf(stderr, "xrdcp: --remove-source supports local and "
627 "root:// sources only\n");
628 str_free(pos, npos); str_free(srcs, nsrc); str_free(exp, nexp);
629 str_free(excl, nexcl); str_free(incl, nincl);
630 return 50;
631 }
632 }

635 /*
636 * Phase 40 (c): credential pre-flight. If a server endpoint is involved, warn
637 * the user INSTANTLY about a locally-detectable auth problem (expired/near-
638 * expiry bearer token or GSI proxy, or a read-only token on an upload) before
639 * the transfer fails with a bare "permission denied". Silent when creds look
640 * fine; never aborts (the server stays authoritative).
641 */
642 {
643 /* Only the GSI/token credential family is diagnosable here — an s3://
644 * endpoint authenticates with AWS SigV4 keys, so it must not trip the
645 * "GSI proxy expired" / "bearer token" pre-flight. */
646 int dst_cred = uses_cred_auth(dst);
647 int any_cred = dst_cred;
648 for (i = 0; i < nexp && !any_cred; i++) {
649 if (uses_cred_auth(exp[i])) { any_cred = 1; }
650 }
651 if (any_cred) {
652 /* Phase 40 (b): if asked, proactively (re)acquire a stale token/proxy
653 * BEFORE diagnosing — so a healthy refresh leaves nothing to warn. */
654 if (auto_refresh) {
655 (void) brix_cred_autorefresh(dst_cred, oidc_account,
656 !opts.silent, stderr);
657 }
658 (void) brix_cred_diagnose(dst_cred, "xrdcp: ", stderr);
659 }
660 }
661
662 /* C1: build the credential store from CLI values and attach it to the
663 * connection options. The store is INERT here — nothing reads conn.cred
664 * yet; C2 will thread it through the auth/token handshake path. Building
665 * it now (before the transfer) means C2 only needs to consume conn.cred,
666 * not rebuild it. NULL/empty args fall back to per-handler env discovery. */
667 cred_store = brix_cli_cred_store_build(proxy, opts.bearer, NULL,
668 opts.s3_access, opts.s3_secret,
669 oidc_account, auto_refresh);
670 conn.cred = cred_store;
671
672 brix_status_clear(&st);
673 /* Recursive copy of a web (davs/http) collection: list it client-side and copy
674 * each file (the wire has no recursive transfer op). root:// and local -r are
675 * handled inside brix_copy, so only intercept web sources here. */
676 if (opts.recursive) {
677 int any_web = 0;
678 for (i = 0; i < nexp; i++) {
679 if (brix_is_web_url(exp[i])) { any_web = 1; }
680 }
681 if (any_web) {
682 size_t bad = 0;
683 for (i = 0; i < nexp; i++) {
684 if (brix_is_web_url(exp[i])) {
685 if (recursive_web_download(exp[i], dst, &opts, &conn, retries) != 0) {
686 bad++;
687 }
688 } else if (copy_one_with_retry(exp[i], dst, &opts, &conn, retries, &st) != 0) {
689 bad++;
690 fprintf(stderr, "xrdcp: %s: %s\n", exp[i], st.msg);
691 brix_cred_hint_for_status_url(&st, is_root_url(dst)
692 || brix_is_web_url(dst),
693 stderr, exp[i]); /* WS-7 */
694 { /* WS-3: double-slash hint on single-slash root:// source */
695 brix_url u_src;
696 brix_status ps;
697 brix_status_clear(&ps);
698 if (brix_url_parse(exp[i], &u_src, &ps) == 0) {
699 brix_hint_url_double_slash(&st, &u_src);
700 }
701 }
702 }
703 }
704 brix_cred_store_free(cred_store);
705 str_free(pos, npos); str_free(srcs, nsrc); str_free(exp, nexp);
706 str_free(excl, nexcl); str_free(incl, nincl);
707 return (bad == 0) ? 0 : 1;
708 }
709 /* Symmetric case: local directory tree(s) → a web (davs/http/s3)
710 * collection. Walk each local dir, MKCOL + PUT; a plain local file with
711 * -r to a web dst is just a single copy. */
712 if (brix_is_web_url(dst)) {
713 int has_dir = 0;
714 for (i = 0; i < nexp; i++) {
715 if (is_local_dir(exp[i])) { has_dir = 1; }
716 }
717 if (has_dir) {
718 size_t bad = 0;
719 /* --delete mirrors a listing back onto the destination, which the
720 * web (davs/http/s3) upload path does not implement. Warn once
721 * so the caller is not misled into thinking stale remote objects
722 * were pruned; the flag is otherwise ignored here. */
723 if (opts.sync_delete) {
724 fprintf(stderr, "xrdcp: --delete is not supported for web "
725 "destinations (ignored)\n");
726 }
727 for (i = 0; i < nexp; i++) {
728 if (is_local_dir(exp[i])) {
729 if (recursive_web_upload(exp[i], dst, &opts, &conn, retries) != 0) {
730 bad++;
731 }
732 } else if (copy_one_with_retry(exp[i], dst, &opts, &conn,
733 retries, &st) != 0) {
734 bad++;
735 fprintf(stderr, "xrdcp: %s: %s\n", exp[i], st.msg);
736 }
737 }
738 brix_cred_store_free(cred_store);
739 str_free(pos, npos); str_free(srcs, nsrc); str_free(exp, nexp);
740 str_free(excl, nexcl); str_free(incl, nincl);
741 return (bad == 0) ? 0 : 1;
742 }
743 }
744 }
745 if (nexp == 1 && from == NULL) {
746 /* Classic single copy — dst may be a file, directory, or '-'. */
747 char label[XRDC_NAME_MAX];
748 xrdcp_prog ps;
749 int one;
750 /* Show a progress bar when asked (--progress) or on an interactive stderr
751 * (unless -N/--no-progress), but never to a piped stdout transfer ('-')
752 * and never when silent. --progress wins over -N. */
753 if ((force_progress || (isatty(STDERR_FILENO) && !no_progress)) && !opts.silent
754 && !(exp[0][0] == '-' && exp[0][1] == '\0')) {
755 path_basename(exp[0], label, sizeof(label));
756 ps.label = (label[0] != '\0') ? label : "transfer";
757 ps.start_ns = brix_mono_ns();
758 ps.last_ns = 0;
759 opts.progress = xrdcp_progress;
760 opts.progress_arg = &ps;
761 }
762 one = transfer_one(exp[0], dst, &opts, &conn, retries, sync_mode, &st);
763 if (one == 1 && !opts.silent) {
764 fprintf(stderr, "xrdcp: %s up-to-date, skipped\n", dst);
765 } else if (one < 0 && !opts.silent) {
766 fprintf(stderr, "xrdcp: %s\n", st.msg);
767 brix_cred_hint_for_status_url(&st, is_root_url(dst)
768 || brix_is_web_url(dst),
769 stderr, exp[0]); /* WS-7 */
770 { /* WS-3: double-slash hint on single-slash root:// source */
771 brix_url u_src;
772 brix_status ps;
773 brix_status_clear(&ps);
774 if (brix_url_parse(exp[0], &u_src, &ps) == 0) {
775 brix_hint_url_double_slash(&st, &u_src);

781 /* Batch copy into a destination directory. */
782 size_t ok = 0, skip = 0, fail = 0;
783 if (dest_is_dir(dst, &conn) != 1) {

785 "multi-source copy: %s\n", dst);

788 str_free(pos, npos); str_free(srcs, nsrc); str_free(exp, nexp);
789 str_free(excl, nexcl); str_free(incl, nincl);
790 return 50;
791 }
792 /* Open the completion journal for the batch path only. A journal on a
793 * single-file copy is meaningless (no resumable item list), and a
794 * dry-run completes nothing so opening the file would create an empty
795 * journal and mislead subsequent runs. Journal writes are serialized
796 * internally so both the sequential and parallel paths share one handle. */
797 if (journal_path != NULL && !opts.dry_run) {
798 brix_status_clear(&st);
799 jrn = brix_journal_open(journal_path, &st);
800 if (jrn == NULL) {
801 fprintf(stderr, "xrdcp: %s\n", st.msg);
802 brix_cred_store_free(cred_store);
803 str_free(pos, npos); str_free(srcs, nsrc); str_free(exp, nexp);
804 str_free(excl, nexcl); str_free(incl, nincl);
805 return 51;
806 }
807 }
808 if (jobs > (int) nexp) { jobs = (int) nexp; }
809 if (jobs > 1) {
810 batch_parallel(exp, nexp, dst, &opts, &conn, retries, sync_mode,
811 jobs, jrn, &ok, &skip, &fail);
812 } else {
813 for (i = 0; i < nexp; i++) {
814 char dpath[XRDC_PATH_MAX];
815 int one;
816 /* journal skip: already-completed sources need no reprocessing */
817 if (jrn != NULL && brix_journal_has(jrn, exp[i])) {
818 skip++;
819 if (!opts.silent) {
820 fprintf(stderr, "[%zu/%zu] %s (already transferred)\n",
821 ok + skip + fail, nexp, exp[i]);
822 }
823 continue;
824 }
825 one = batch_copy_one(exp[i], dst, &opts, &conn, retries,
826 sync_mode, dpath, sizeof(dpath), &st);
827 if (one == 0) {
828 ok++;
829 if (!opts.silent) {
830 fprintf(stderr, "[%zu/%zu] %s -> %s\n",
831 ok + skip + fail, nexp, exp[i], dpath);
832 }
833 /* mark only on real success; dry-run completes nothing */
834 if (jrn != NULL && !opts.dry_run) {
835 (void) brix_journal_mark(jrn, exp[i]);
836 }
837 } else if (one == 1) {
838 skip++;
839 if (!opts.silent) {
840 fprintf(stderr, "[%zu/%zu] %s (up-to-date)\n",
841 ok + skip + fail, nexp, exp[i]);
842 }
843 } else {
844 fail++;
845 fprintf(stderr, "xrdcp: %s: %s\n", exp[i], st.msg);
846 brix_cred_hint_for_status_url(&st, is_root_url(dst)
847 || brix_is_web_url(dst),
848 stderr, exp[i]); /* WS-7 */
849 { /* WS-3: double-slash hint on single-slash root:// source */
850 brix_url u_src;
851 brix_status ps;
852 brix_status_clear(&ps);
853 if (brix_url_parse(exp[i], &u_src, &ps) == 0) {
854 brix_hint_url_double_slash(&st, &u_src);
855 }
856 }
857 }
858 }
859 }
860 if (!opts.silent) {
861 fprintf(stderr, "xrdcp: %zu copied, %zu skipped, %zu failed\n",
862 ok, skip, fail);
863 }
864 rc = (fail == 0) ? 0 : 1;
865 }
866
867 brix_journal_close(jrn);
868 brix_cred_store_free(cred_store);
869 str_free(pos, npos);
870 str_free(srcs, nsrc);
871 str_free(exp, nexp);
872 str_free(excl, nexcl);
873 str_free(incl, nincl);
874 return rc;
