8/*
9 * WHAT: True (1) if the request URI has ADMIN_PREFIX + `sub` as a strict prefix
10 * (i.e. a sub-resource path, not the exact collection), else 0.
11 * WHY: Both the cluster and proxy route groups gate their sub-resource block on
12 * the same len>prefix + ngx_strncmp idiom; naming it removes a duplicated
13 * sizeof/strncmp pair from the router and reads as one predicate.
14 * HOW: Compare against the compile-time length of ADMIN_PREFIX concatenated
15 * with `sub` via a caller-supplied precomputed length (plen), matching the
16 * original sizeof(literal)-1 bounds exactly.
17 */
18static int
19admin_uri_under(ngx_http_request_t *r, const char *pfx, size_t plen)
20{
21 return r->uri.len > plen && ngx_strncmp(r->uri.data, pfx, plen) == 0;
22}

25/*
26 * WHAT: Route the io_uring runtime kill-switch endpoint. Sets *matched=1 and
27 * returns the response status when the URI is this endpoint; leaves
28 * *matched=0 (return value unused) otherwise.
29 * WHY: Splitting each endpoint group out of brix_admin_dispatch keeps the
30 * router a flat sequence of match-or-fall-through steps under the gate.
31 * HOW: Exact-match the URI; when enabled, dispatch by method (POST reads body
32 * into admin_io_uring_set, GET serves admin_io_uring_get), else 404/405.
33 * Behavior byte-frozen.
34 */
35static ngx_int_t
36admin_route_io_uring(ngx_http_request_t *r, int *matched)
37{
38 if (!admin_uri_eq(r, ADMIN_PREFIX "io_uring")) {
39 *matched = 0;
40 return NGX_OK;
41 }
42 *matched = 1;
43
44 /* Phase 44: io_uring runtime kill switch (only when enabled) */
45 if (!brix_uring_admin_enabled()) {
46 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
47 }
48 if (r->method == NGX_HTTP_POST) {
49 return brix_admin_read_body(r, admin_io_uring_set);
50 }
51 if (r->method == NGX_HTTP_GET) {
52 return admin_io_uring_get(r);
53 }
54 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
55}

58/*
59 * WHAT: Route the "/cluster/servers/{host}/{port}[/action]" sub-resource. Sets
60 * *matched per whether the URI is under that prefix.
61 * WHY: Isolates the per-server sub-action ladder (drain/undrain/delete/upsert)
62 * from the router so each stays independently reviewable.
63 * HOW: Prefix-match, then branch on the trailing action or HTTP method.
64 * Behavior byte-frozen.
65 */
66static ngx_int_t
67admin_route_cluster_server(ngx_http_request_t *r, int *matched)
68{
69 if (!admin_uri_under(r, ADMIN_PREFIX "cluster/servers/",
70 sizeof(ADMIN_PREFIX "cluster/servers/") - 1))
71 {
72 *matched = 0;
73 return NGX_OK;
74 }
75 *matched = 1;
76
77 if (admin_uri_has_action(r, "/drain")) {
78 if (r->method != NGX_HTTP_POST) {
79 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
80 }
81 return brix_admin_read_body(r, admin_cluster_drain);
82 }
83 if (admin_uri_has_action(r, "/undrain")) {
84 if (r->method != NGX_HTTP_POST) {
85 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
86 }
87 return admin_cluster_undrain(r);
88 }
89 if (r->method == NGX_HTTP_DELETE) {
90 return admin_cluster_delete(r);
91 }
92 /* PUT to a specific server path is an upsert — same body handler as
93 * POST to the collection; brix_srv_register replaces if it exists. */
94 if (r->method == NGX_HTTP_PUT) {
95 return brix_admin_read_body(r, admin_cluster_register); /* upsert */
96 }
97 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
98}

101/*
102 * WHAT: Route the cluster registry endpoints (collection + sub-resource). Sets
103 * *matched per whether a cluster route handled the request.
104 * WHY: Groups the two cluster URIs behind one match so the top-level router
105 * reads as one line per endpoint family.
106 * HOW: Try the exact collection URI first (POST registers, else 405), then
107 * delegate the "{host}/{port}" sub-resource. Behavior byte-frozen.
108 */
109static ngx_int_t
110admin_route_cluster(ngx_http_request_t *r, int *matched)
111{
112 if (admin_uri_eq(r, ADMIN_PREFIX "cluster/servers")) {
113 *matched = 1;
114 if (r->method == NGX_HTTP_POST) {
115 return brix_admin_read_body(r, admin_cluster_register);
116 }
117 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
118 }
119 return admin_route_cluster_server(r, matched);
120}

123/*
124 * WHAT: Route the dynamic proxy pool endpoints (collection + "{id}[/action]").
125 * Sets *matched per whether a proxy route handled the request.
126 * WHY: Groups the two proxy URIs behind one match, mirroring admin_route_cluster.
127 * HOW: Exact collection URI (POST adds, GET lists, else 405), then the id
128 * sub-resource parsed via admin_parse_proxy_uri. Behavior byte-frozen.
129 */
130static ngx_int_t
131admin_route_proxy(ngx_http_request_t *r, int *matched)
132{
133 uint32_t id = 0;
134 char action[16];
135
136 if (admin_uri_eq(r, ADMIN_PREFIX "proxy/backends")) {
137 *matched = 1;
138 if (r->method == NGX_HTTP_POST) {
139 return brix_admin_read_body(r, admin_proxy_add);
140 }
141 if (r->method == NGX_HTTP_GET) {
142 return admin_proxy_list(r);
143 }
144 return admin_send_error(r, NGX_HTTP_NOT_ALLOWED, "method_not_allowed");
145 }
146 if (admin_uri_under(r, ADMIN_PREFIX "proxy/backends/",
147 sizeof(ADMIN_PREFIX "proxy/backends/") - 1))
148 {
149 *matched = 1;
150 if (admin_parse_proxy_uri(r, &id, action, sizeof(action)) != NGX_OK) {
151 return admin_send_error(r, NGX_HTTP_BAD_REQUEST, "bad_uri");
152 }
153 return admin_proxy_one(r, action, id);
154 }
155
156 *matched = 0;
157 return NGX_OK;
158}

171/*
172 * ---- Per-IP admin API rate gate (phase-23 "Rate limiting" item) ----
173 *
174 * WHAT: Charge this request against the source IP's read or write leaky bucket.
175 * Sets *throttled and returns the finalized 429 response status when the
176 * bucket overflows; leaves *throttled 0 and returns NGX_OK to proceed.
177 *
178 * WHY: An authorized-but-runaway client (looping script, stuck orchestrator)
179 * must not be able to monopolize the admin surface or churn the registry
180 * SHM. Reads and writes get separate generous buckets so extensive
181 * legitimate querying under load never starves — or is starved by — the
182 * mutating traffic the phase-23 design caps. Runs AFTER auth, so an
183 * unauthenticated flood (already answered 403) can neither fill the zone
184 * nor lock out the real operator.
185 *
186 * HOW: 1. Pass through when the throttle is off or was never finalized
187 * (write.zone == NULL) — brix_rl_check itself fails open on SHM
188 * trouble, so the gate can only ever deny by policy.
189 * 2. Classify the method: GET/HEAD -> read rule, else write rule, with
190 * distinct key prefixes so the two buckets never share a node.
191 * 3. brix_rl_check the "adm[rw]:<ip>" key; NGX_OK proceeds.
192 * 4. On NGX_AGAIN set Retry-After (HTTP throttle convention), audit the
193 * event, and send the 429 JSON error.
194 */
195static ngx_int_t
196admin_rate_gate(ngx_http_request_t *r,
197 ngx_http_brix_dashboard_loc_conf_t *conf, int *throttled)
198{
199 brix_rl_rule_t *rule;
200 int is_read;
201 char key[BRIX_RL_KEY_LEN];
202 uint32_t wait_sec = 0;
203
204 *throttled = 0;
205 if (conf == NULL || !conf->admin_rl_enable
206 || conf->admin_rl_write.zone == NULL)
207 {
208 return NGX_OK;
209 }
210
211 is_read = (r->method & (NGX_HTTP_GET | NGX_HTTP_HEAD)) != 0;
212 rule = is_read ? &conf->admin_rl_read : &conf->admin_rl_write;
213
214 ngx_snprintf((u_char *) key, sizeof(key), "%s:%V%Z",
215 is_read ? "admr" : "admw", &r->connection->addr_text);
216 key[sizeof(key) - 1] = '\0';
217
218 if (brix_rl_check(rule, key, &wait_sec) != NGX_AGAIN) {
219 return NGX_OK;
220 }
221
222 *throttled = 1;
223 if (wait_sec > 0) {
224 char retry[NGX_INT_T_LEN + 1];
225 ngx_snprintf((u_char *) retry, sizeof(retry), "%uD%Z", wait_sec);
226 (void) brix_http_set_header(r, "Retry-After", retry, NULL);
227 }
228 admin_audit(r, "ratelimit", is_read ? "read" : "write", "throttled");
229 return admin_send_error(r, NGX_HTTP_TOO_MANY_REQUESTS, "rate_limited");
230}

233ngx_int_t
234brix_admin_dispatch(ngx_http_request_t *r)
235{
236 static const admin_route_fn routes[] = {
237 admin_route_io_uring,
238 admin_route_cluster,
239 admin_route_proxy,
240 };
241 ngx_http_brix_dashboard_loc_conf_t *conf;
242 ngx_uint_t i;
243
244 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
245
246 if (brix_admin_check_auth(r, conf) != BRIX_ADMIN_AUTH_OK) {
247 admin_audit(r, "auth", NULL, "forbidden");
248 return admin_send_error(r, NGX_HTTP_FORBIDDEN, "forbidden");
249 }
250
251 /* Per-IP throttle sits between auth and routing: it therefore runs before
252 * any body read or SHM mutation, and only authorized traffic can consume
253 * bucket capacity. */
254 {
255 int throttled = 0;
256 ngx_int_t rc = admin_rate_gate(r, conf, &throttled);
257 if (throttled) {
258 return rc;
259 }
260 }
261
262 for (i = 0; i < sizeof(routes) / sizeof(routes[0]); i++) {
263 int matched = 0;
264 ngx_int_t rc = routes[i](r, &matched);
265 if (matched) {
266 return rc;
267 }
268 }
269
270 return admin_send_error(r, NGX_HTTP_NOT_FOUND, "not_found");
271}
