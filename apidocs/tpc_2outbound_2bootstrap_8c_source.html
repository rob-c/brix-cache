1/* File: bootstrap.c — Anonymous XRootD source session setup for TPC pull
2 * WHAT: Establishes an anonymous XRootD session on the remote origin by executing the three-step handshake pipeline: ClientInitHandShake → kXR_protocol version negotiation → kXR_login with username "xrd" and capver kXR_ver005. Supports OAuth2/OIDC token delegation — when t->token_mode is set, fetches a delegated token before login so authenticated source fetch replaces anonymous login.
3 *
4 * WHY: Native TPC pull requires nginx to connect to the remote root:// source as an XRootD client before it can read the file and write it locally. This bootstrap establishes the session layer (handshake + protocol version check + login) that every subsequent kXR_open/read/close operation depends on. Token delegation enables authenticated source fetches when the source site requires auth but nginx has a delegated token from the destination's OIDC provider.
5 *
6 * HOW: ClientInitHandShake with fourth=4 and fifth=ROOTD_PQ → send/recv response → kXR_protocol request with streamid[1]=1, requestid=kXR_protocol, clientpv=kXR_PROTOCOLVERSION, expect=0x03 → send/recv response → OAuth2 token fetch if t->token_mode is non-empty/non-"none" → kXR_login with pid=ngx_pid, username="xrd", capver=kXR_ver005 → send/recv response → handle status: kXR_ok returns immediately (anonymous login complete), kXR_authmore delegates to tpc_outbound_finish_login() for multi-step auth, other status rejects.
7 * */

23/*
24 * WHAT: Execute the initial ClientInitHandShake exchange with the TPC source.
25 * WHY: Every XRootD session begins with the fixed handshake preamble; a non-ok
26 * reply here means the peer is not a usable root:// origin and the whole
27 * bootstrap must abort. Isolating it keeps tpc_bootstrap's control flow flat.
28 * HOW: Pack the handshake, send it, receive the response, always free the (unused)
29 * body, then reject any non-kXR_ok status. Returns 0 on success, -1 on failure
30 * (with t->err_msg / t->xrd_error set exactly as the original inline path did).
31 */
32static int
33tpc_bootstrap_handshake(brix_tpc_pull_t *t, int fd)
34{
35 ClientInitHandShake hs;
36 uint16_t status;
37 uint32_t dlen = 0;
38 u_char *body = NULL;
39
40 xrd_pack_handshake(&hs);
41
42 if (tpc_send_all(t, fd, &hs, sizeof(hs)) != 0) {
43 snprintf(t->err_msg, sizeof(t->err_msg), "TPC handshake send failed");
44 t->xrd_error = kXR_ServerError;
45 return -1;
46 }
47 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
48 snprintf(t->err_msg, sizeof(t->err_msg), "TPC handshake recv failed");
49 t->xrd_error = kXR_ServerError;
50 return -1;
51 }
52 free(body);
53 if (status != kXR_ok) {
54 snprintf(t->err_msg, sizeof(t->err_msg),
55 "TPC handshake rejected (status=%u)", (unsigned) status);
56 t->xrd_error = kXR_ServerError;
57 return -1;
58 }
59 return 0;
60}

63/*
64 * WHAT: Complete the in-protocol TLS upgrade if the source requested kXR_gotoTLS.
65 * WHY: A TLS-requiring source answers the kXR_protocol request with kXR_gotoTLS in
66 * the ServerProtocolBody flags; login and all subsequent frames must then ride
67 * the TLS socket. With outbound TLS off the source never offers gotoTLS, so this
68 * is a no-op and behaviour is identical to before.
69 * HOW: Parse the 8-byte ServerProtocolBody { pval[4], flags[4] } from the reply body,
70 * free the body, and if kXR_gotoTLS is set either reject (TLS disabled) or drive
71 * tpc_start_tls(). Returns 0 on success, -1 on failure with error state set.
72 */
73static int
74tpc_bootstrap_maybe_tls(brix_tpc_pull_t *t, int fd, u_char *body, uint32_t dlen)
75{
76 uint32_t flags = 0;
77
78 if (body != NULL && dlen >= 8) {
79 uint32_t flags_be;
80
81 ngx_memcpy(&flags_be, body + 4, sizeof(flags_be));
82 flags = ntohl(flags_be);
83 }
84 free(body);
85
86 if (flags & kXR_gotoTLS) {
87 if (!t->conf->tpc_outbound_tls) {
88 snprintf(t->err_msg, sizeof(t->err_msg),
89 "TPC source requires TLS; set brix_tpc_outbound_tls on");
90 t->xrd_error = kXR_NotAuthorized;
91 return -1;
92 }
93 if (tpc_start_tls(t, fd) != 0) {
94 return -1; /* tpc_start_tls sets t->err_msg / t->xrd_error */
95 }
96 }
97 return 0;
98}

101/*
102 * WHAT: Execute the kXR_protocol version negotiation and any resulting TLS upgrade.
103 * WHY: The source must accept our protocol version before login; its reply also
104 * carries the gotoTLS decision. Combining request/response/TLS keeps the reply
105 * body's lifetime local and out of tpc_bootstrap.
106 * HOW: Pack the request (advertising kXR_ableTLS only when outbound TLS is enabled),
107 * send, receive, reject non-ok status (freeing the body first), then hand the
108 * body to tpc_bootstrap_maybe_tls() which owns the free. Returns 0 / -1.
109 */
110static int
111tpc_bootstrap_protocol(brix_tpc_pull_t *t, int fd)
112{
113 ClientProtocolRequest pr;
114 uint16_t status;
115 uint32_t dlen = 0;
116 u_char *body = NULL;
117
118 /* kXR_protocol — advertise kXR_ableTLS when outbound TLS is enabled so a
119 * TLS-requiring source answers with kXR_gotoTLS (phase-57 §F5). With the
120 * directive off we send no TLS flag, so the source never offers gotoTLS and
121 * behaviour is identical to before. */
122 xrd_pack_protocol_request(&pr, tpc_bootstrap_streamid,
123 t->conf->tpc_outbound_tls ? kXR_ableTLS : 0);
124
125 if (tpc_send_all(t, fd, &pr, sizeof(pr)) != 0) {
126 snprintf(t->err_msg, sizeof(t->err_msg), "TPC kXR_protocol send failed");
127 t->xrd_error = kXR_ServerError;
128 return -1;
129 }
130 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
131 snprintf(t->err_msg, sizeof(t->err_msg), "TPC kXR_protocol recv failed");
132 t->xrd_error = kXR_ServerError;
133 return -1;
134 }
135 if (status != kXR_ok) {
136 free(body);
137 snprintf(t->err_msg, sizeof(t->err_msg),
138 "TPC kXR_protocol rejected (status=%u)", (unsigned) status);
139 t->xrd_error = kXR_ServerError;
140 return -1;
141 }
142
143 /*
144 * TLS upgrade (phase-57 §F5): the kXR_protocol reply body is the 8-byte
145 * ServerProtocolBody { pval[4], flags[4] }. If the source set kXR_gotoTLS we
146 * must complete an in-protocol TLS handshake before login; every subsequent
147 * frame (login + GSI/ztn auth + open/read/close) then rides the TLS socket
148 * via the I/O helpers (t->tls). Mirrors src/upstream/bootstrap.c.
149 */
150 return tpc_bootstrap_maybe_tls(t, fd, body, dlen);
151}

154/*
155 * WHAT: Perform the kXR_login exchange (anonymous or delegated-token-aware) and
156 * complete any follow-on authentication the reply demands.
157 * WHY: This is the session-establishing step; a kXR_ok login can still REQUIRE auth
158 * when the reply carries a security-protocol list, so we must inspect the body,
159 * not just the status. Fetching the delegated token first lets an authenticated
160 * source fetch replace the anonymous login.
161 * HOW: Fetch a delegated token when token_mode is set and not "none"; pack + send
162 * kXR_login; receive the reply; for kXR_ok/kXR_authmore run
163 * tpc_outbound_finish_login() only when the body exceeds the bare session id,
164 * else treat as genuinely anonymous. Returns 0 / -1 with error state set.
165 */
166static int
167tpc_bootstrap_login(brix_tpc_pull_t *t, int fd)
168{
169 ClientLoginRequest lr;
170 uint16_t status;
171 uint32_t dlen = 0;
172 u_char *body = NULL;
173
174 /*
175 * OAuth2/OIDC token delegation: fetch a delegated token before the
176 * initial kXR_login when the client requested a delegation mode.
177 * The fetched token replaces the anonymous login when the source
178 * advertises token auth.
179 */
180 if (t->token_mode[0] != '\0'
181 && ngx_strcmp(t->token_mode, "none") != 0) {
182 if (tpc_fetch_delegated_token(t) != 0) {
183 return -1;
184 }
185 }
186
187 /* kXR_login — anonymous or delegated-token-aware */
188 xrd_pack_login_request(&lr, tpc_bootstrap_streamid, (int32_t) ngx_pid,
189 "xrd", kXR_ver005);
190
191 if (tpc_send_all(t, fd, &lr, sizeof(lr)) != 0) {
192 snprintf(t->err_msg, sizeof(t->err_msg), "TPC kXR_login send failed");
193 t->xrd_error = kXR_AuthFailed;
194 return -1;
195 }
196 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
197 snprintf(t->err_msg, sizeof(t->err_msg), "TPC kXR_login recv failed");
198 t->xrd_error = kXR_AuthFailed;
199 return -1;
200 }
201
202 /*
203 * Authenticate whenever the login reply carries a security token, whether the
204 * status is kXR_authmore OR kXR_ok. In XRootD a kXR_ok login can still
205 * REQUIRE auth: the reply body is the session id (BRIX_SESSION_ID_LEN) plus
206 * the "&P=<proto>,..." security-protocol list, and the client must send
207 * kXR_auth before any operation (exactly what stock XrdCl does — its own login
208 * returns ok with a non-empty body, then it sends kXR_auth). Treating kXR_ok
209 * as "anonymous, done" here made the destination open the source without
210 * authenticating, so a GSI source rejected the pull with "user not
211 * authenticated". A reply body no larger than the bare session id (or an
212 * empty body) means no security token → genuinely anonymous.
213 */
214 if (status == kXR_ok || status == kXR_authmore) {
215 int frc = 0;
216
217 if (dlen > BRIX_SESSION_ID_LEN) {
218 frc = tpc_outbound_finish_login(t, fd, body, dlen);
219 }
220 free(body);
221 return (frc == 0) ? 0 : -1;
222 }
223
224 free(body);
225 snprintf(t->err_msg, sizeof(t->err_msg),
226 "TPC kXR_login rejected (status=%u)", (unsigned) status);
227 t->xrd_error = kXR_AuthFailed;
228 return -1;
229}
