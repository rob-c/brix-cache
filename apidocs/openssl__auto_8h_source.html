1/*
2 * openssl_auto.h — scope-based automatic cleanup for OpenSSL objects.
3 *
4 * WHAT: typed `__attribute__((cleanup))` handlers + the XRD_AUTO(type) macro,
5 * so an OpenSSL handle declared with XRD_AUTO is freed automatically when
6 * it goes out of scope — on EVERY return path, with no manual free.
7 *
8 * XRD_AUTO(EVP_CIPHER_CTX) *ctx = EVP_CIPHER_CTX_new();
9 * if (ctx == NULL) return NULL; // freed (NULL-safe)
10 * if (bad) return NULL; // freed
11 * ... // freed on normal return too
12 *
13 * WHY: this module forbids `goto`, so the classic single-exit cleanup ladder is
14 * replaced by early-return — which means the SAME free has to be repeated at
15 * every exit (gsi_cipher.c freed one EVP_CIPHER_CTX at six different returns;
16 * parse_x509.c freed EVP_PKEY at eight). Each new early-return is a chance to
17 * forget one and leak. A scope-cleanup handler collapses all of them to a
18 * single declaration AND makes the free structurally unforgettable: the
19 * compiler emits it on every path, including ones added later. This turns the
20 * no-goto rule from a liability into an advantage over goto-cleanup.
21 *
22 * HOW: each XRD_OSSL_AUTO_DEFINE(type, freefn) emits a static-inline handler that
23 * takes the address of the variable and frees the pointed-to object if
24 * non-NULL (all the OpenSSL *_free functions below already tolerate NULL, but
25 * the guard keeps the intent explicit and the contract local). The handler is
26 * `static inline`, so there is no new translation unit and no ./configure
27 * change — include the header and declare with XRD_AUTO.
28 *
29 * OWNERSHIP: XRD_AUTO is for objects OWNED by the current scope. If a function
30 * RETURNS the handle (transfers ownership to the caller), do NOT use XRD_AUTO
31 * on it — keep it a plain pointer, or null it out before the return so the
32 * handler becomes a no-op. The malloc'd payload buffers that these functions
33 * return are the textbook example and stay manually managed.
34 */

41/* Define a cleanup handler `xrd_ossl_cleanup_<type>` for an OpenSSL pointer type
42 * whose single-argument destructor is `freefn`. */
43#define XRD_OSSL_AUTO_DEFINE(type, freefn) \
44 static inline void xrd_ossl_cleanup_##type(type **p) \
45 { \
46 if (*p != NULL) { \
47 freefn(*p); \
48 } \
49 }

58 * XRD_AUTO(EVP_CIPHER_CTX) *ctx = EVP_CIPHER_CTX_new();
59 */
60#define XRD_AUTO(type) __attribute__((cleanup(xrd_ossl_cleanup_##type))) type
