19static int
20uc_kv(void *u, const char *section, const char *key, const char *val)
21{
22 brix_throttle_uc_t *uc = u;
23 (void) section; /* each [name] section is one rule */
24
25 if (strcasecmp(key, "name") == 0) {
26 if (uc->count < BRIX_THROTTLE_MAX_UC_RULES) {
27 snprintf(uc->rules[uc->count].pat,
28 sizeof(uc->rules[uc->count].pat), "%s", val);
29 }
30 } else if (strcasecmp(key, "maxconn") == 0) {
31 if (uc->count < BRIX_THROTTLE_MAX_UC_RULES) {
32 uc->rules[uc->count].maxconn = (ngx_uint_t) atoi(val);
33 if (strcmp(uc->rules[uc->count].pat, "*") == 0) {
34 uc->global = uc->rules[uc->count].maxconn;
35 }
36 uc->count++;
37 }
38 }
39 return 0;
40}

50ngx_uint_t
51brix_throttle_userconfig_match(const brix_throttle_uc_t *uc,
52 const char *user)
53{
54 int exact = -1;
55 size_t best_len = 0;
56 ngx_uint_t best = 0;
57 int i;
58
59 for (i = 0; i < uc->count; i++) {
60 const char *pat = uc->rules[i].pat;
61 size_t plen = strlen(pat);
62
63 if (strcmp(pat, user) == 0) {
64 exact = i;
65 break;
66 }
67 if (plen > 1 && pat[plen - 1] == '*') {
68 if (strncmp(pat, user, plen - 1) == 0 && (plen - 1) >= best_len) {
69 best_len = plen - 1;
70 best = uc->rules[i].maxconn;
71 }
72 }
73 }
74 if (exact >= 0) {
75 return uc->rules[exact].maxconn;
76 }
77 if (best_len > 0) {
78 return best;
79 }
80 return uc->global;
81}

96void
97brix_throttle_charge_io(brix_rl_zone_t *zone, ngx_msec_t interval_ms,
98 const char *user, uint64_t service_us)
99{
100 brix_rl_node_t *n;
101
102 if (zone == NULL || zone->sh == NULL) {
103 return;
104 }
105 ngx_shmtx_lock(&zone->shpool->mutex);
106 n = throttle_node_locked(zone, user);
107 if (n != NULL) {
108 ngx_msec_t now = ngx_current_msec;
109
110 if ((ngx_msec_int_t) (now - n->io_window) >= (ngx_msec_int_t) interval_ms) {
111 n->io_time_us = 0;
112 n->io_window = now;
113 }
114 n->io_time_us += service_us;
115 }
116 ngx_shmtx_unlock(&zone->shpool->mutex);
117}

119int
120brix_throttle_ioload_over(brix_rl_zone_t *zone, ngx_msec_t interval_ms,
121 const char *user, double concurrency)
122{
123 brix_rl_node_t *n;
124 int over = 0;
125
126 if (zone == NULL || zone->sh == NULL || concurrency <= 0.0
127 || interval_ms == 0)
128 {
129 return 0;
130 }
131 ngx_shmtx_lock(&zone->shpool->mutex);
132 n = brix_rl_lookup_locked(zone, brix_rl_hash(user, strlen(user)),
133 user, strlen(user));
134 if (n != NULL) {
135 double load = (double) n->io_time_us
136 / ((double) interval_ms * 1000.0);
137 over = (load >= concurrency);
138 }
139 ngx_shmtx_unlock(&zone->shpool->mutex);
140 return over;
141}

143int
144brix_throttle_open_inc(brix_rl_zone_t *zone, const char *user,
145 ngx_uint_t cap)
146{
147 brix_rl_node_t *n;
148 int ok = 1;
149
150 if (zone == NULL || zone->sh == NULL || cap == 0) {
151 return 1; /* unlimited */
152 }
153 ngx_shmtx_lock(&zone->shpool->mutex);
154 n = throttle_node_locked(zone, user);
155 if (n != NULL) {
156 if (n->open_files < cap) {
157 n->open_files++;
158 ok = 1;
159 } else {
160 ok = 0; /* over the per-user cap */
161 }
162 } /* n==NULL (slab OOM) ⇒ fail-open */
163 ngx_shmtx_unlock(&zone->shpool->mutex);
164 return ok;
165}
