1/*
2 * cred_mint.c — opt-in x509 credential minting (phase-2 T9). See cred_mint.h.
3 *
4 * WHAT: Implements brix_cred_mint(): cache-check an existing <key>.pem, and
5 * when it is missing/expired/within-refresh, generate a fresh EC P-256
6 * keypair + X509 signed by the operator's mint CA and write it
7 * atomically to <cred_dir>/<key>.pem.
8 *
9 * WHY: See cred_mint.h for the full rationale and trust-model note. In
10 * short: gives bearer-only identities (S3 access keys, WLCG tokens
11 * with no matching pre-provisioned proxy) a per-user x509 origin
12 * identity without requiring the operator to pre-provision one file
13 * per user.
14 *
15 * HOW: Four static helpers, each owning and freeing its own OpenSSL
16 * objects on every return path (no goto, mirrors the pxr_ctx/pxr_fail
17 * pattern in src/auth/gsi/proxy_req.c):
18 * mint_load_ca() — PEM_read_bio_X509 + PEM_read_bio_PrivateKey.
19 * mint_cached_pem_ok() — parse an existing <key>.pem, compare notAfter
20 * against now + refresh window.
21 * mint_build_cert() — EVP_PKEY_Q_keygen(EC P-256), build+sign X509.
22 * mint_write_pem() — serialize cert+key+CA cert, write via a
23 * same-directory O_EXCL 0600 temp file + fsync
24 * + rename (atomic, never a half-written file).
25 */

55/* ---- mint_load_ca ------------------------------------------------------
56 * WHAT: Load the mint CA's certificate and private key from PEM files.
57 * WHY: Single place that owns the "bad CA material" failure mode so
58 * brix_cred_mint() never has to special-case it.
59 * HOW: fopen + PEM_read_X509 / PEM_read_PrivateKey; NULL on any failure,
60 * with partial allocations freed before returning. */
61static int
62mint_load_ca(const char *ca_cert_path, const char *ca_key_path,
63 X509 **out_cert, EVP_PKEY **out_key, ngx_log_t *log)
64{
65 FILE *f;
66 X509 *cert = NULL;
67 EVP_PKEY *key = NULL;
68
69 *out_cert = NULL;
70 *out_key = NULL;
71
72 f = fopen(ca_cert_path, "r");
73 if (f != NULL) {
74 cert = PEM_read_X509(f, NULL, NULL, NULL);
75 (void) fclose(f); /* read-only stream — nothing buffered to lose */
76 }
77 if (cert == NULL) {
78 MINT_LOG(NGX_LOG_ERR, log, 0,
79 "brix: cred_mint: cannot load mint CA cert \"%s\"", ca_cert_path);
80 return NGX_ERROR;
81 }
82
83 f = fopen(ca_key_path, "r");
84 if (f != NULL) {
85 key = PEM_read_PrivateKey(f, NULL, NULL, NULL);
86 (void) fclose(f); /* read-only stream — nothing buffered to lose */
87 }
88 if (key == NULL) {
89 MINT_LOG(NGX_LOG_ERR, log, 0,
90 "brix: cred_mint: cannot load mint CA key \"%s\"", ca_key_path);
91 X509_free(cert);
92 return NGX_ERROR;
93 }
94
95 *out_cert = cert;
96 *out_key = key;
97 return NGX_OK;
98}

100/* ---- mint_cached_pem_ok -------------------------------------------------
101 * WHAT: 1 iff <path> parses as an X509 whose notAfter is more than
102 * BRIX_CRED_MINT_REFRESH_WINDOW seconds in the future; 0 otherwise
103 * (absent, unparseable, expired, or within the refresh window).
104 * WHY: Avoids re-minting (and re-signing with the CA key) on every request
105 * once a valid proxy is cached; still re-mints proactively before the
106 * cached proxy would lapse mid-use.
107 * HOW: fopen + PEM_read_X509 (reads the FIRST PEM block — the leaf cert,
108 * matching ucred.c's ucred_check_pem convention) + X509_cmp_time
109 * against now + refresh window. */
110static int
111mint_cached_pem_ok(const char *path)
112{
113 FILE *f;
114 X509 *cert;
115 time_t cutoff;
116 int cmp;
117
118 f = fopen(path, "r");
119 if (f == NULL) {
120 return 0;
121 }
122 cert = PEM_read_X509(f, NULL, NULL, NULL);
123 (void) fclose(f); /* read-only stream — nothing buffered to lose */
124 if (cert == NULL) {
125 return 0;
126 }
127
128 cutoff = time(NULL) + BRIX_CRED_MINT_REFRESH_WINDOW;
129 cmp = X509_cmp_time(X509_get0_notAfter(cert), &cutoff);
130 X509_free(cert);
131
132 /* X509_cmp_time returns -1 if notAfter < cutoff (not enough life left),
133 * 0 on a parse error (treat as not-ok), 1 if notAfter > cutoff (ok). */
134 return (cmp > 0) ? 1 : 0;
135}

137/* ---- mint_cn_char_ok -----------------------------------------------------
138 * WHAT: 1 iff byte `c` is safe to copy verbatim into a CN RDN value: the
139 * ASCII alphanumerics plus the punctuation set [@._-/ =] (space and '='
140 * appear in DN-derived principals; '/' is a DN separator X509_NAME
141 * tolerates inside a single RDN value). Everything else — including
142 * control bytes and NUL — is rejected so mint_sanitize_cn maps it to '_'.
143 * WHY: Isolating the charset predicate keeps mint_sanitize_cn's loop trivial
144 * and makes the exact allow-list a single, testable, security-load-
145 * bearing point of truth (the set MUST stay identical byte-for-byte).
146 * HOW: Pure range/equality checks over the unsigned byte value. */
147static int
148mint_cn_char_ok(unsigned char c)
149{
150 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
151 || (c >= '0' && c <= '9') || c == '@' || c == '.'
152 || c == '_' || c == '-' || c == '/' || c == ' ' || c == '=';
153}

155/* ---- mint_sanitize_cn ----------------------------------------------------
156 * WHAT: Copy `principal` into `out` (cap bytes) as a CN-safe string: any byte
157 * rejected by mint_cn_char_ok() is replaced with '_'. Truncates rather
158 * than overflows.
159 * WHY: The principal may be a DN, an S3 access key, or a JWT subject; RFC
160 * 5280 does not forbid most of these characters in a CN, but control
161 * characters or an embedded NUL must never reach X509_NAME_add_entry.
162 * HOW: Single bounded pass; deterministic (same principal -> same CN). */
163static void
164mint_sanitize_cn(const char *principal, char *out, size_t cap)
165{
166 size_t i;
167 size_t n = strlen(principal);
168
169 if (n >= cap) {
170 n = cap - 1;
171 }
172 for (i = 0; i < n; i++) {
173 unsigned char c = (unsigned char) principal[i];
174 out[i] = mint_cn_char_ok(c) ? (char) c : '_';
175 }
176 out[n] = '\0';
177}

179/* ---- mint_make_subject ---------------------------------------------------
180 * WHAT: Build the minted-proxy subject X509_NAME "/O=brix-minted/CN=<cn>";
181 * NULL on any allocation/entry failure (freeing a partial name first).
182 * WHY: Owns the X509_NAME lifecycle in one place so mint_populate_cert()'s
183 * early-return ladder never has to special-case the subject build.
184 * HOW: X509_NAME_new + two X509_NAME_add_entry_by_txt calls; `cn` is already
185 * CN-sanitized by the caller (mint_sanitize_cn). */
186static X509_NAME *
187mint_make_subject(const char *cn, ngx_log_t *log)
188{
189 X509_NAME *subj = X509_NAME_new();
190
191 if (subj == NULL
192 || !X509_NAME_add_entry_by_txt(subj, "O", MBSTRING_ASC,
193 (const unsigned char *) "brix-minted", -1, -1, 0)
194 || !X509_NAME_add_entry_by_txt(subj, "CN", MBSTRING_ASC,
195 (const unsigned char *) cn, -1, -1, 0)) {
196 MINT_LOG(NGX_LOG_ERR, log, 0,
197 "brix: cred_mint: cannot build proxy subject");
198 X509_NAME_free(subj);
199 return NULL;
200 }
201 return subj;
202}

204/* ---- mint_serial_from_rand -----------------------------------------------
205 * WHAT: Pack 8 random bytes into a uint64 serial with the top bit cleared so
206 * the ASN1 INTEGER encoding is unambiguously positive.
207 * WHY: Keeps the bit-shuffle out of mint_populate_cert()'s boolean ladder
208 * and names the "coerce positive" invariant.
209 * HOW: Big-endian assembly of rnd[0..7]; rnd[0]'s top bit is masked off. */
210static uint64_t
211mint_serial_from_rand(const unsigned char rnd[8])
212{
213 return (((uint64_t) (rnd[0] & 0x7f)) << 56) | ((uint64_t) rnd[1] << 48)
214 | ((uint64_t) rnd[2] << 40) | ((uint64_t) rnd[3] << 32)
215 | ((uint64_t) rnd[4] << 24) | ((uint64_t) rnd[5] << 16)
216 | ((uint64_t) rnd[6] << 8) | (uint64_t) rnd[7];
217}

219/* ---- mint_cert_spec_t ----------------------------------------------------
220 * WHAT: The immutable inputs mint_populate_cert() needs to set and sign one
221 * minted leaf certificate: its subject name, the mint CA cert (issuer
222 * source) and key (signer), the leaf's own public key, the ASN1 serial,
223 * and the validity window length in seconds.
224 * WHY: Bundling these related inputs keeps mint_populate_cert() at one job
225 * and ≤5 parameters without introducing any globals or changing what it
226 * does — purely a call-shape change.
227 * HOW: Plain aggregate, filled on the stack by mint_build_cert() and passed
228 * by const pointer; the helper mutates only the caller-owned `cert`. */
229typedef struct {
230 X509_NAME *subj;
231 X509 *ca_cert;
232 EVP_PKEY *key;
233 EVP_PKEY *ca_key;
234 uint64_t serial;
235 int ttl_secs;
236} mint_cert_spec_t;

238/* ---- mint_ca_t / mint_material_t -----------------------------------------
239 * WHAT: mint_ca_t bundles the loaded mint-CA cert+key (the signer);
240 * mint_material_t bundles the minted leaf cert+key plus the CA cert that
241 * forms the trust-chain tail written into the PEM.
242 * WHY: Groups the OpenSSL objects that always travel together so the mint
243 * helpers stay at ≤5 parameters without adding globals — the pointers
244 * and their free ordering are unchanged, only their call shape.
245 * HOW: Plain aggregates filled on the stack; ownership stays with the caller
246 * (each helper documents which objects it allocates or borrows). */
247typedef struct {
248 X509 *cert;
249 EVP_PKEY *key;
250} mint_ca_t;

252typedef struct {
253 X509 *cert; /* minted leaf (owned) */
254 EVP_PKEY *key; /* minted leaf key (owned) */
255 X509 *ca_cert; /* borrowed: mint CA cert, PEM trust-chain tail */
256} mint_material_t;

258/* ---- mint_populate_cert --------------------------------------------------
259 * WHAT: Populate and sign an already-allocated `cert` from `spec`: version 3,
260 * spec->serial, spec->subj, issuer = spec->ca_cert's subject, pubkey =
261 * spec->key, notBefore=now, notAfter=now+spec->ttl_secs,
262 * X509_sign(spec->ca_key, SHA-256). Returns NGX_OK/NGX_ERROR; frees
263 * nothing (caller owns cert).
264 * WHY: Splits the long field-setting boolean chain out of mint_build_cert()
265 * so each function stays under the CCN cap while preserving the exact
266 * single-expression short-circuit ordering (security-load-bearing).
267 * HOW: One boolean AND-chain; any failure logs and returns NGX_ERROR. */
268static int
269mint_populate_cert(X509 *cert, const mint_cert_spec_t *spec, ngx_log_t *log)
270{
271 if (X509_set_version(cert, 2L) != 1
272 || ASN1_INTEGER_set_uint64(X509_get_serialNumber(cert),
273 spec->serial) != 1
274 || X509_set_subject_name(cert, spec->subj) != 1
275 || X509_set_issuer_name(cert,
276 X509_get_subject_name(spec->ca_cert)) != 1
277 || X509_set_pubkey(cert, spec->key) != 1
278 || X509_gmtime_adj(X509_getm_notBefore(cert), 0) == NULL
279 || X509_gmtime_adj(X509_getm_notAfter(cert),
280 (long) spec->ttl_secs) == NULL
281 || X509_sign(cert, spec->ca_key, EVP_sha256()) <= 0) {
282 MINT_LOG(NGX_LOG_ERR, log, 0,
283 "brix: cred_mint: cannot build/sign minted proxy");
284 return NGX_ERROR;
285 }
286 return NGX_OK;
287}

289/* ---- mint_build_cert -----------------------------------------------------
290 * WHAT: Generate a fresh EC P-256 keypair and a signed X509 whose subject is
291 * "/O=brix-minted/CN=<sanitized principal>", issuer = the mint CA's
292 * subject, notBefore=now, notAfter=now+ttl_secs, signed by ca_key with
293 * SHA-256.
294 * WHY: Isolates the OpenSSL object lifecycle for the mint step so
295 * brix_cred_mint() stays a thin orchestrator; every early return here
296 * frees exactly what this call allocated.
297 * HOW: EVP_PKEY_Q_keygen(EC, P-256) -> X509_new -> mint_make_subject ->
298 * mint_populate_cert (which sets/signs every field). Serial is 8 random
299 * bytes coerced positive so it round-trips through ASN1_INTEGER. On
300 * success fills out->cert/out->key (leaves out->ca_cert untouched). */
301static int
302mint_build_cert(const mint_ca_t *ca, const char *principal, int ttl_secs,
303 mint_material_t *out, ngx_log_t *log)
304{
305 EVP_PKEY *key = NULL;
306 X509 *cert = NULL;
307 X509_NAME *subj = NULL;
308 unsigned char rnd[8];
309 char cn[256];
310 int ok;
311
312 out->cert = NULL;
313 out->key = NULL;
314
315 key = EVP_PKEY_Q_keygen(NULL, NULL, "EC", "P-256");
316 if (key == NULL) {
317 MINT_LOG(NGX_LOG_ERR, log, 0,
318 "brix: cred_mint: EC P-256 key generation failed");
319 return NGX_ERROR;
320 }
321
322 if (RAND_bytes(rnd, sizeof(rnd)) != 1) {
323 MINT_LOG(NGX_LOG_ERR, log, 0, "brix: cred_mint: RNG failed");
324 EVP_PKEY_free(key);
325 return NGX_ERROR;
326 }
327
328 cert = X509_new();
329 if (cert == NULL) {
330 EVP_PKEY_free(key);
331 return NGX_ERROR;
332 }
333
334 mint_sanitize_cn(principal, cn, sizeof(cn));
335 subj = mint_make_subject(cn, log);
336 if (subj == NULL) {
337 X509_free(cert);
338 EVP_PKEY_free(key);
339 return NGX_ERROR;
340 }
341
342 mint_cert_spec_t spec = {
343 .subj = subj,
344 .ca_cert = ca->cert,
345 .key = key,
346 .ca_key = ca->key,
347 .serial = mint_serial_from_rand(rnd),
348 .ttl_secs = ttl_secs,
349 };
350 ok = mint_populate_cert(cert, &spec, log) == NGX_OK;
351 X509_NAME_free(subj);
352 if (!ok) {
353 X509_free(cert);
354 EVP_PKEY_free(key);
355 return NGX_ERROR;
356 }
357
358 out->cert = cert;
359 out->key = key;
360 return NGX_OK;
361}

363/* ---- mint_serialize_pem --------------------------------------------------
364 * WHAT: Serialize `cert` + `key` + `ca_cert` as concatenated PEM blocks (leaf
365 * cert, then private key, then the mint CA cert as the trust-chain tail)
366 * into a fresh mem BIO, returned on success. NULL on any failure, with
367 * the (partial) BIO freed.
368 * WHY: Owns the PEM-serialization failure mode so mint_write_pem() only has
369 * to deal with the file side; keeps the exact block ORDER (cert, key,
370 * CA) as the single point of truth — a reader relies on it.
371 * HOW: BIO_s_mem + PEM_write_bio_X509/PrivateKey/X509 + a non-empty check via
372 * BIO_get_mem_data. The caller owns the returned BIO and must BIO_free
373 * it (and reads its bytes via BIO_get_mem_data while it lives). */
374static BIO *
375mint_serialize_pem(X509 *cert, EVP_PKEY *key, X509 *ca_cert, ngx_log_t *log)
376{
377 BIO *mem;
378 char *data = NULL;
379
380 mem = BIO_new(BIO_s_mem());
381 if (mem == NULL
382 || PEM_write_bio_X509(mem, cert) != 1
383 || PEM_write_bio_PrivateKey(mem, key, NULL, NULL, 0, NULL, NULL) != 1
384 || PEM_write_bio_X509(mem, ca_cert) != 1) {
385 MINT_LOG(NGX_LOG_ERR, log, 0,
386 "brix: cred_mint: PEM serialization failed");
387 BIO_free(mem);
388 return NULL;
389 }
390 if (BIO_get_mem_data(mem, &data) <= 0) {
391 MINT_LOG(NGX_LOG_ERR, log, 0,
392 "brix: cred_mint: empty PEM output");
393 BIO_free(mem);
394 return NULL;
395 }
396
397 return mem;
398}

400/* ---- mint_write_tmp ------------------------------------------------------
401 * WHAT: Write `len` bytes of `data` to a freshly-created, O_EXCL, 0600,
402 * O_NOFOLLOW temp file at `tmp_path`, then fsync it. On any failure the
403 * temp file is unlinked before returning NGX_ERROR; on success the file
404 * persists (owned by the caller, which renames it into place).
405 * WHY: Isolates the durable-write failure ladder (open/write/fsync) so
406 * mint_write_pem() reads as serialize -> write-tmp -> rename.
407 * HOW: open(O_CREAT|O_EXCL|O_WRONLY|O_NOFOLLOW,0600), full-length write,
408 * fsync, close; unlink on the error paths. */
409static int
410mint_write_tmp(const char *tmp_path, const char *data, long len,
411 ngx_log_t *log)
412{
413 int fd;
414 int rc = NGX_ERROR;
415 ssize_t w;
416
417 fd = open(tmp_path, O_CREAT | O_EXCL | O_WRONLY | O_NOFOLLOW, 0600);
418 if (fd < 0) {
419 MINT_LOG(NGX_LOG_ERR, log, errno,
420 "brix: cred_mint: cannot create temp file \"%s\"", tmp_path);
421 return NGX_ERROR;
422 }
423
424 w = write(fd, data, (size_t) len);
425 if (w < 0 || w != len) {
426 MINT_LOG(NGX_LOG_ERR, log, errno,
427 "brix: cred_mint: short/failed write to \"%s\"", tmp_path);
428 } else if (fsync(fd) != 0) {
429 MINT_LOG(NGX_LOG_ERR, log, errno,
430 "brix: cred_mint: fsync failed on \"%s\"", tmp_path);
431 } else {
432 rc = NGX_OK;
433 }
434 close(fd);
435
436 if (rc != NGX_OK) {
437 unlink(tmp_path);
438 }
439 return rc;
440}

442/* ---- mint_build_paths ----------------------------------------------------
443 * WHAT: Fill `final_path` (<dir>/<key_stem>.pem) and `tmp_path`
444 * (<dir>/.mint-<key_stem>.<pid>.tmp) — the PID keeps concurrent mints
445 * from colliding on the temp name. NGX_ERROR if either overflows its
446 * MINT_PATH_MAX buffer.
447 * WHY: Removes two length-check branches from mint_write_pem() and names the
448 * temp-file naming convention in one place.
449 * HOW: Bounded snprintf into caller-owned MINT_PATH_MAX buffers. */
450static int
451mint_build_paths(const char *dir, const char *key_stem,
452 char final_path[MINT_PATH_MAX], char tmp_path[MINT_PATH_MAX],
453 ngx_log_t *log)
454{
455 int n;
456
457 n = snprintf(final_path, MINT_PATH_MAX, "%s/%s.pem", dir, key_stem);
458 if (n < 0 || (size_t) n >= (size_t) MINT_PATH_MAX) {
459 MINT_LOG(NGX_LOG_ERR, log, 0,
460 "brix: cred_mint: credential path too long");
461 return NGX_ERROR;
462 }
463 n = snprintf(tmp_path, MINT_PATH_MAX, "%s/.mint-%s.%d.tmp",
464 dir, key_stem, (int) getpid());
465 if (n < 0 || (size_t) n >= (size_t) MINT_PATH_MAX) {
466 MINT_LOG(NGX_LOG_ERR, log, 0,
467 "brix: cred_mint: temp credential path too long");
468 return NGX_ERROR;
469 }
470 return NGX_OK;
471}

473/* ---- mint_write_pem -------------------------------------------------------
474 * WHAT: Serialize mat->cert + mat->key + mat->ca_cert (leaf, then key, then
475 * the mint CA cert as the trust-chain tail) as concatenated PEM blocks
476 * and write them atomically to <dir>/<key_stem>.pem.
477 * WHY: A reader (brix_sd_ucred_resolve, or any other process racing this
478 * mint) must never observe a partially-written file; a crash between
479 * open() and the final bytes must not leave a corrupt .pem where a
480 * valid one previously existed.
481 * HOW: mint_build_paths -> mint_serialize_pem (in-memory) -> mint_write_tmp
482 * (same-directory O_EXCL 0600 temp + fsync) -> rename() over the final
483 * path (POSIX rename is atomic within one filesystem). The temp name
484 * embeds the PID so concurrent mints cannot collide.
485 * vfs-seam-allow: svc-owned credential dir, not an export — this is the
486 * shared per-user backend-credential cache directory (brix_storage_
487 * credential_dir), not an fs/ export root; raw POSIX I/O here is the
488 * existing Phase-1 seam convention (see ucred.c's O_NOFOLLOW token read). */
489static int
490mint_write_pem(const char *dir, const char *key_stem,
491 const mint_material_t *mat, ngx_log_t *log)
492{
493 char final_path[MINT_PATH_MAX];
494 char tmp_path[MINT_PATH_MAX];
495 BIO *mem = NULL;
496 char *data = NULL;
497 long len = 0;
498
499 if (mint_build_paths(dir, key_stem, final_path, tmp_path, log) != NGX_OK) {
500 return NGX_ERROR;
501 }
502
503 mem = mint_serialize_pem(mat->cert, mat->key, mat->ca_cert, log);
504 if (mem == NULL) {
505 return NGX_ERROR;
506 }
507 len = BIO_get_mem_data(mem, &data); /* aliases into mem; valid till free */
508
509 if (mint_write_tmp(tmp_path, data, len, log) != NGX_OK) {
510 BIO_free(mem);
511 return NGX_ERROR;
512 }
513 BIO_free(mem);
514
515 if (rename(tmp_path, final_path) != 0) {
516 MINT_LOG(NGX_LOG_ERR, log, errno,
517 "brix: cred_mint: rename \"%s\" -> \"%s\" failed",
518 tmp_path, final_path);
519 unlink(tmp_path);
520 return NGX_ERROR;
521 }
522
523 return NGX_OK;
524}

544/* ---- mint_generate -------------------------------------------------------
545 * WHAT: Load the mint CA, build+sign a fresh proxy for `principal`, and write
546 * it atomically to <cred_dir>/<key>.pem. Assumes the cache-miss path
547 * (caller has already ruled out a still-valid cached PEM).
548 * WHY: Owns the full OpenSSL object lifecycle (CA cert/key + minted cert/key)
549 * and its exact free ordering — CA material and minted material are each
550 * freed on every return path — so brix_cred_mint() stays a thin
551 * cache-check + delegate + log wrapper under the CCN cap.
552 * HOW: mint_load_ca -> mint_build_cert -> mint_write_pem; frees the minted
553 * cert/key then the CA cert/key (order preserved from the original
554 * single-function form). `principal`, `key`, `ttl_secs` reach the log
555 * line and cert build; the CA-file paths reach only mint_load_ca. */
556static ngx_int_t
557mint_generate(const char *cred_dir, const char *ca_cert_path,
558 const char *ca_key_path, const char *principal, const char *key,
559 int ttl_secs, ngx_log_t *log)
560{
561 mint_ca_t ca = { NULL, NULL };
562 mint_material_t mat = { NULL, NULL, NULL };
563 ngx_int_t rc;
564
565 if (mint_load_ca(ca_cert_path, ca_key_path, &ca.cert, &ca.key, log)
566 != NGX_OK) {
567 return NGX_ERROR;
568 }
569
570 if (mint_build_cert(&ca, principal, ttl_secs, &mat, log) != NGX_OK) {
571 X509_free(ca.cert);
572 EVP_PKEY_free(ca.key);
573 return NGX_ERROR;
574 }
575 mat.ca_cert = ca.cert; /* trust-chain tail written into the PEM */
576
577 rc = mint_write_pem(cred_dir, key, &mat, log) == NGX_OK
578 ? NGX_OK : NGX_ERROR;
579
580 if (rc == NGX_OK) {
581 MINT_LOG(NGX_LOG_NOTICE, log, 0,
582 "brix: cred_mint: minted proxy for principal=\"%s\" key=%s "
583 "ttl=%ds", principal, key, ttl_secs);
584 }
585
586 X509_free(mat.cert);
587 EVP_PKEY_free(mat.key);
588 X509_free(ca.cert);
589 EVP_PKEY_free(ca.key);
590 return rc;
591}

594 * brix_cred_mint — see cred_mint.h.
595 */
596ngx_int_t
597brix_cred_mint(const char *cred_dir, const char *ca_cert_path,
598 const char *ca_key_path, const char *principal, const char *key,
599 int ttl_secs, ngx_log_t *log)
600{
601 char existing_path[MINT_PATH_MAX];
602 int n;
603
604 if (!mint_args_valid(cred_dir, ca_cert_path, ca_key_path, principal,
605 key)) {
606 return NGX_ERROR;
607 }
608
609 n = snprintf(existing_path, sizeof(existing_path), "%s/%s.pem",
610 cred_dir, key);
611 if (n < 0 || (size_t) n >= sizeof(existing_path)) {
612 MINT_LOG(NGX_LOG_ERR, log, 0,
613 "brix: cred_mint: credential path too long for key \"%s\"", key);
614 return NGX_ERROR;
615 }
616
617 /* Cache hit: an existing PEM with enough life left is reused verbatim —
618 * no CA material is even loaded, so a steady-state deployment pays the
619 * mint cost only once per refresh window per principal. */
620 if (mint_cached_pem_ok(existing_path)) {
621 return NGX_OK;
622 }
623
624 return mint_generate(cred_dir, ca_cert_path, ca_key_path, principal, key,
625 ttl_secs, log);
626}
