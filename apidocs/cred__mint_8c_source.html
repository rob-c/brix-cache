1/*
2 * cred_mint.c — opt-in x509 credential minting (phase-2 T9). See cred_mint.h.
3 *
4 * WHAT: Implements brix_cred_mint(): cache-check an existing <key>.pem, and
5 * when it is missing/expired/within-refresh, generate a fresh EC P-256
6 * keypair + X509 signed by the operator's mint CA and write it
7 * atomically to <cred_dir>/<key>.pem.
8 *
9 * WHY: See cred_mint.h for the full rationale and trust-model note. In
10 * short: gives bearer-only identities (S3 access keys, WLCG tokens
11 * with no matching pre-provisioned proxy) a per-user x509 origin
12 * identity without requiring the operator to pre-provision one file
13 * per user.
14 *
15 * HOW: Four static helpers, each owning and freeing its own OpenSSL
16 * objects on every return path (no goto, mirrors the pxr_ctx/pxr_fail
17 * pattern in src/auth/gsi/proxy_req.c):
18 * mint_load_ca() — PEM_read_bio_X509 + PEM_read_bio_PrivateKey.
19 * mint_cached_pem_ok() — parse an existing <key>.pem, compare notAfter
20 * against now + refresh window.
21 * mint_build_cert() — EVP_PKEY_Q_keygen(EC P-256), build+sign X509.
22 * mint_write_pem() — serialize cert+key+CA cert, write via a
23 * same-directory O_EXCL 0600 temp file + fsync
24 * + rename (atomic, never a half-written file).
25 */

49/* ---- mint_load_ca ------------------------------------------------------
50 * WHAT: Load the mint CA's certificate and private key from PEM files.
51 * WHY: Single place that owns the "bad CA material" failure mode so
52 * brix_cred_mint() never has to special-case it.
53 * HOW: fopen + PEM_read_X509 / PEM_read_PrivateKey; NULL on any failure,
54 * with partial allocations freed before returning. */
55static int
56mint_load_ca(const char *ca_cert_path, const char *ca_key_path,
57 X509 **out_cert, EVP_PKEY **out_key, ngx_log_t *log)
58{
59 FILE *f;
60 X509 *cert = NULL;
61 EVP_PKEY *key = NULL;
62
63 *out_cert = NULL;
64 *out_key = NULL;
65
66 f = fopen(ca_cert_path, "r");
67 if (f != NULL) {
68 cert = PEM_read_X509(f, NULL, NULL, NULL);
69 (void) fclose(f); /* read-only stream — nothing buffered to lose */
70 }
71 if (cert == NULL) {
72 MINT_LOG(NGX_LOG_ERR, log, 0,
73 "brix: cred_mint: cannot load mint CA cert \"%s\"", ca_cert_path);
74 return NGX_ERROR;
75 }
76
77 f = fopen(ca_key_path, "r");
78 if (f != NULL) {
79 key = PEM_read_PrivateKey(f, NULL, NULL, NULL);
80 (void) fclose(f); /* read-only stream — nothing buffered to lose */
81 }
82 if (key == NULL) {
83 MINT_LOG(NGX_LOG_ERR, log, 0,
84 "brix: cred_mint: cannot load mint CA key \"%s\"", ca_key_path);
85 X509_free(cert);
86 return NGX_ERROR;
87 }
88
89 *out_cert = cert;
90 *out_key = key;
91 return NGX_OK;
92}

94/* ---- mint_cached_pem_ok -------------------------------------------------
95 * WHAT: 1 iff <path> parses as an X509 whose notAfter is more than
96 * BRIX_CRED_MINT_REFRESH_WINDOW seconds in the future; 0 otherwise
97 * (absent, unparseable, expired, or within the refresh window).
98 * WHY: Avoids re-minting (and re-signing with the CA key) on every request
99 * once a valid proxy is cached; still re-mints proactively before the
100 * cached proxy would lapse mid-use.
101 * HOW: fopen + PEM_read_X509 (reads the FIRST PEM block — the leaf cert,
102 * matching ucred.c's ucred_check_pem convention) + X509_cmp_time
103 * against now + refresh window. */
104static int
105mint_cached_pem_ok(const char *path)
106{
107 FILE *f;
108 X509 *cert;
109 time_t cutoff;
110 int cmp;
111
112 f = fopen(path, "r");
113 if (f == NULL) {
114 return 0;
115 }
116 cert = PEM_read_X509(f, NULL, NULL, NULL);
117 (void) fclose(f); /* read-only stream — nothing buffered to lose */
118 if (cert == NULL) {
119 return 0;
120 }
121
122 cutoff = time(NULL) + BRIX_CRED_MINT_REFRESH_WINDOW;
123 cmp = X509_cmp_time(X509_get0_notAfter(cert), &cutoff);
124 X509_free(cert);
125
126 /* X509_cmp_time returns -1 if notAfter < cutoff (not enough life left),
127 * 0 on a parse error (treat as not-ok), 1 if notAfter > cutoff (ok). */
128 return (cmp > 0) ? 1 : 0;
129}

131/* ---- mint_serialize_pem --------------------------------------------------
132 * WHAT: Serialize `cert` + `key` + `ca_cert` as concatenated PEM blocks (leaf
133 * cert, then private key, then the mint CA cert as the trust-chain tail)
134 * into a fresh mem BIO, returned on success. NULL on any failure, with
135 * the (partial) BIO freed.
136 * WHY: Owns the PEM-serialization failure mode so mint_write_pem() only has
137 * to deal with the file side; keeps the exact block ORDER (cert, key,
138 * CA) as the single point of truth — a reader relies on it.
139 * HOW: BIO_s_mem + PEM_write_bio_X509/PrivateKey/X509 + a non-empty check via
140 * BIO_get_mem_data. The caller owns the returned BIO and must BIO_free
141 * it (and reads its bytes via BIO_get_mem_data while it lives). */
142static BIO *
143mint_serialize_pem(X509 *cert, EVP_PKEY *key, X509 *ca_cert, ngx_log_t *log)
144{
145 BIO *mem;
146 char *data = NULL;
147
148 mem = BIO_new(BIO_s_mem());
149 if (mem == NULL
150 || PEM_write_bio_X509(mem, cert) != 1
151 || PEM_write_bio_PrivateKey(mem, key, NULL, NULL, 0, NULL, NULL) != 1
152 || PEM_write_bio_X509(mem, ca_cert) != 1) {
153 MINT_LOG(NGX_LOG_ERR, log, 0,
154 "brix: cred_mint: PEM serialization failed");
155 BIO_free(mem);
156 return NULL;
157 }
158 if (BIO_get_mem_data(mem, &data) <= 0) {
159 MINT_LOG(NGX_LOG_ERR, log, 0,
160 "brix: cred_mint: empty PEM output");
161 BIO_free(mem);
162 return NULL;
163 }
164
165 return mem;
166}

168/* ---- mint_write_tmp ------------------------------------------------------
169 * WHAT: Write `len` bytes of `data` to a freshly-created, O_EXCL, 0600,
170 * O_NOFOLLOW temp file at `tmp_path`, then fsync it. On any failure the
171 * temp file is unlinked before returning NGX_ERROR; on success the file
172 * persists (owned by the caller, which renames it into place).
173 * WHY: Isolates the durable-write failure ladder (open/write/fsync) so
174 * mint_write_pem() reads as serialize -> write-tmp -> rename.
175 * HOW: open(O_CREAT|O_EXCL|O_WRONLY|O_NOFOLLOW,0600), full-length write,
176 * fsync, close; unlink on the error paths. */
177static int
178mint_write_tmp(const char *tmp_path, const char *data, long len,
179 ngx_log_t *log)
180{
181 int fd;
182 int rc = NGX_ERROR;
183 ssize_t w;
184
185 fd = open(tmp_path, O_CREAT | O_EXCL | O_WRONLY | O_NOFOLLOW, 0600);
186 if (fd < 0) {
187 MINT_LOG(NGX_LOG_ERR, log, errno,
188 "brix: cred_mint: cannot create temp file \"%s\"", tmp_path);
189 return NGX_ERROR;
190 }
191
192 w = write(fd, data, (size_t) len);
193 if (w < 0 || w != len) {
194 MINT_LOG(NGX_LOG_ERR, log, errno,
195 "brix: cred_mint: short/failed write to \"%s\"", tmp_path);
196 } else if (fsync(fd) != 0) {
197 MINT_LOG(NGX_LOG_ERR, log, errno,
198 "brix: cred_mint: fsync failed on \"%s\"", tmp_path);
199 } else {
200 rc = NGX_OK;
201 }
202 close(fd);
203
204 if (rc != NGX_OK) {
205 unlink(tmp_path);
206 }
207 return rc;
208}

210/* ---- mint_build_paths ----------------------------------------------------
211 * WHAT: Fill `final_path` (<dir>/<key_stem>.pem) and `tmp_path`
212 * (<dir>/.mint-<key_stem>.<pid>.tmp) — the PID keeps concurrent mints
213 * from colliding on the temp name. NGX_ERROR if either overflows its
214 * MINT_PATH_MAX buffer.
215 * WHY: Removes two length-check branches from mint_write_pem() and names the
216 * temp-file naming convention in one place.
217 * HOW: Bounded snprintf into caller-owned MINT_PATH_MAX buffers. */
218static int
219mint_build_paths(const char *dir, const char *key_stem,
220 char final_path[MINT_PATH_MAX], char tmp_path[MINT_PATH_MAX],
221 ngx_log_t *log)
222{
223 int n;
224
225 n = snprintf(final_path, MINT_PATH_MAX, "%s/%s.pem", dir, key_stem);
226 if (n < 0 || (size_t) n >= (size_t) MINT_PATH_MAX) {
227 MINT_LOG(NGX_LOG_ERR, log, 0,
228 "brix: cred_mint: credential path too long");
229 return NGX_ERROR;
230 }
231 n = snprintf(tmp_path, MINT_PATH_MAX, "%s/.mint-%s.%d.tmp",
232 dir, key_stem, (int) getpid());
233 if (n < 0 || (size_t) n >= (size_t) MINT_PATH_MAX) {
234 MINT_LOG(NGX_LOG_ERR, log, 0,
235 "brix: cred_mint: temp credential path too long");
236 return NGX_ERROR;
237 }
238 return NGX_OK;
239}

241/* ---- mint_write_pem -------------------------------------------------------
242 * WHAT: Serialize mat->cert + mat->key + mat->ca_cert (leaf, then key, then
243 * the mint CA cert as the trust-chain tail) as concatenated PEM blocks
244 * and write them atomically to <dir>/<key_stem>.pem.
245 * WHY: A reader (brix_sd_ucred_resolve, or any other process racing this
246 * mint) must never observe a partially-written file; a crash between
247 * open() and the final bytes must not leave a corrupt .pem where a
248 * valid one previously existed.
249 * HOW: mint_build_paths -> mint_serialize_pem (in-memory) -> mint_write_tmp
250 * (same-directory O_EXCL 0600 temp + fsync) -> rename() over the final
251 * path (POSIX rename is atomic within one filesystem). The temp name
252 * embeds the PID so concurrent mints cannot collide.
253 * vfs-seam-allow: svc-owned credential dir, not an export — this is the
254 * shared per-user backend-credential cache directory (brix_storage_
255 * credential_dir), not an fs/ export root; raw POSIX I/O here is the
256 * existing Phase-1 seam convention (see ucred.c's O_NOFOLLOW token read). */
257static int
258mint_write_pem(const char *dir, const char *key_stem,
259 const mint_material_t *mat, ngx_log_t *log)
260{
261 char final_path[MINT_PATH_MAX];
262 char tmp_path[MINT_PATH_MAX];
263 BIO *mem = NULL;
264 char *data = NULL;
265 long len = 0;
266
267 if (mint_build_paths(dir, key_stem, final_path, tmp_path, log) != NGX_OK) {
268 return NGX_ERROR;
269 }
270
271 mem = mint_serialize_pem(mat->cert, mat->key, mat->ca_cert, log);
272 if (mem == NULL) {
273 return NGX_ERROR;
274 }
275 len = BIO_get_mem_data(mem, &data); /* aliases into mem; valid till free */
276
277 if (mint_write_tmp(tmp_path, data, len, log) != NGX_OK) {
278 BIO_free(mem);
279 return NGX_ERROR;
280 }
281 BIO_free(mem);
282
283 if (rename(tmp_path, final_path) != 0) {
284 MINT_LOG(NGX_LOG_ERR, log, errno,
285 "brix: cred_mint: rename \"%s\" -> \"%s\" failed",
286 tmp_path, final_path);
287 unlink(tmp_path);
288 return NGX_ERROR;
289 }
290
291 return NGX_OK;
292}

312/* ---- mint_generate -------------------------------------------------------
313 * WHAT: Load the mint CA, build+sign a fresh proxy for `principal`, and write
314 * it atomically to <cred_dir>/<key>.pem. Assumes the cache-miss path
315 * (caller has already ruled out a still-valid cached PEM).
316 * WHY: Owns the full OpenSSL object lifecycle (CA cert/key + minted cert/key)
317 * and its exact free ordering — CA material and minted material are each
318 * freed on every return path — so brix_cred_mint() stays a thin
319 * cache-check + delegate + log wrapper under the CCN cap.
320 * HOW: mint_load_ca -> mint_build_cert -> mint_write_pem; frees the minted
321 * cert/key then the CA cert/key (order preserved from the original
322 * single-function form). `principal`, `key`, `ttl_secs` reach the log
323 * line and cert build; the CA-file paths reach only mint_load_ca. */
324static ngx_int_t
325mint_generate(const char *cred_dir, const char *ca_cert_path,
326 const char *ca_key_path, const char *principal, const char *key,
327 int ttl_secs, ngx_log_t *log)
328{
329 mint_ca_t ca = { NULL, NULL };
330 mint_material_t mat = { NULL, NULL, NULL };
331 ngx_int_t rc;
332
333 if (mint_load_ca(ca_cert_path, ca_key_path, &ca.cert, &ca.key, log)
334 != NGX_OK) {
335 return NGX_ERROR;
336 }
337
338 if (mint_build_cert(&ca, principal, ttl_secs, &mat, log) != NGX_OK) {
339 X509_free(ca.cert);
340 EVP_PKEY_free(ca.key);
341 return NGX_ERROR;
342 }
343 mat.ca_cert = ca.cert; /* trust-chain tail written into the PEM */
344
345 rc = mint_write_pem(cred_dir, key, &mat, log) == NGX_OK
346 ? NGX_OK : NGX_ERROR;
347
348 if (rc == NGX_OK) {
349 MINT_LOG(NGX_LOG_NOTICE, log, 0,
350 "brix: cred_mint: minted proxy for principal=\"%s\" key=%s "
351 "ttl=%ds", principal, key, ttl_secs);
352 }
353
354 X509_free(mat.cert);
355 EVP_PKEY_free(mat.key);
356 X509_free(ca.cert);
357 EVP_PKEY_free(ca.key);
358 return rc;
359}

361/*
362 * brix_cred_mint — see cred_mint.h.
363 */
364ngx_int_t
365brix_cred_mint(const char *cred_dir, const char *ca_cert_path,
366 const char *ca_key_path, const char *principal, const char *key,
367 int ttl_secs, ngx_log_t *log)
368{
369 char existing_path[MINT_PATH_MAX];
370 int n;
371
372 if (!mint_args_valid(cred_dir, ca_cert_path, ca_key_path, principal,
373 key)) {
374 return NGX_ERROR;
375 }
376
377 n = snprintf(existing_path, sizeof(existing_path), "%s/%s.pem",
378 cred_dir, key);
379 if (n < 0 || (size_t) n >= sizeof(existing_path)) {
380 MINT_LOG(NGX_LOG_ERR, log, 0,
381 "brix: cred_mint: credential path too long for key \"%s\"", key);
382 return NGX_ERROR;
383 }
384
385 /* Cache hit: an existing PEM with enough life left is reused verbatim —
386 * no CA material is even loaded, so a steady-state deployment pays the
387 * mint cost only once per refresh window per principal. */
388 if (mint_cached_pem_ok(existing_path)) {
389 return NGX_OK;
390 }
391
392 return mint_generate(cred_dir, ca_cert_path, ca_key_path, principal, key,
393 ttl_secs, log);
394}
