1/*
2 * checksum.c - shared file checksum and digest algorithms.
3 *
4 * WHAT: Implements 32-bit checksums (Adler-32, CRC-32, CRC-32c) and variable-length
5 * digests (MD5, SHA1, SHA256) for XRootD wire protocol and WebDAV/S3 responses.
6 *
7 * WHY: XRootD fattr/PROPFIND return checksums as hex strings. pgread/pgwrite require
8 * per-page CRC32c validation. All callers share this single implementation to
9 * avoid duplication between native XRootD, WebDAV, and S3 modules.
10 *
11 * HOW: 32-bit paths use zlib (adler32/crc32) or crc32c table lookup; digest paths
12 * use OpenSSL EVP API. All fd-based functions loop pread in 64KB chunks with
13 * EINTR retry. Hex encoding converts binary digests to lowercase ASCII.
14*/

32/*
33 * brix_checksum_name — algorithm enum to human-readable string.
34 *
35 * WHAT: Maps checksum algorithm enum values to their canonical lowercase names for
36 * logging and diagnostic messages. WHY: Access logs, error messages, and Prometheus
37 * labels need readable algorithm identifiers; callers don't want to switch-case the
38 * enum themselves. HOW: Direct switch-case returning "adler32", "crc32", "crc32c",
39 * "md5", "sha1", or "sha256"; returns NULL for unknown values. */
40
41const char *
42brix_checksum_name(brix_checksum_alg_t alg)
43{
44 switch (alg) {
45 case BRIX_CHECKSUM_ADLER32:
46 return "adler32";
47 case BRIX_CHECKSUM_CRC32:
48 return "crc32";
49 case BRIX_CHECKSUM_CRC32C:
50 return "crc32c";
51 case BRIX_CHECKSUM_MD5:
52 return "md5";
53 case BRIX_CHECKSUM_SHA1:
54 return "sha1";
55 case BRIX_CHECKSUM_SHA256:
56 return "sha256";
57 case BRIX_CHECKSUM_CRC64:
58 return "crc64";
59 case BRIX_CHECKSUM_CRC64NVME:
60 return "crc64nvme";
61 case BRIX_CHECKSUM_ZCRC32:
62 return "zcrc32";
63 default:
64 return NULL;

102/*
103 * brix_checksum_normalize_name — validate and lowercase an algorithm name.
104 *
105 * WHAT: Copies name[0..len) into buf as a NUL-terminated lowercase string,
106 * rejecting an empty name, one that overflows buf, or one containing a
107 * non-alphanumeric byte. Returns NGX_OK on success, NGX_ERROR on any
108 * rejection. WHY: brix_checksum_parse must sanitize wire/config input to
109 * the exact same rules regardless of which algorithm it later matches;
110 * isolating the byte handling keeps the parser flat and testable. HOW:
111 * 1) Reject NULL/empty/oversized → 2) Loop each byte, reject non-alnum,
112 * tolower into buf → 3) NUL-terminate at len. */
113
114static ngx_int_t
115brix_checksum_normalize_name(const char *name, size_t len,
116 char *buf, size_t bufsz)
117{
118 size_t i;
119
120 if (name == NULL || len == 0 || len >= bufsz) {
121 return NGX_ERROR;
122 }
123
124 for (i = 0; i < len; i++) {
125 if (!isalnum((unsigned char) name[i])) {
126 return NGX_ERROR;
127 }
128 buf[i] = (char) tolower((unsigned char) name[i]);
129 }
130 buf[len] = '\0';
131
132 return NGX_OK;
133}

135/*
136 * brix_checksum_lookup_alg — map a normalized name to its algorithm enum.
137 *
138 * WHAT: Matches an already-lowercased name against the canonical algorithm
139 * names (plus the "crc64xz" alias for CRC-64/XZ), writing the enum to
140 * *out and returning NGX_OK; returns NGX_DECLINED for an unknown name.
141 * WHY: expressing the name→enum mapping as a table instead of a strcmp ladder
142 * keeps brix_checksum_parse's complexity low and makes the accepted set
143 * obvious in one place. HOW: linear strcmp scan over the name/enum table;
144 * first match wins, matching the prior if-else ordering exactly. */
145
146static ngx_int_t
147brix_checksum_lookup_alg(const char *lname, brix_checksum_alg_t *out)
148{
149 static const struct {
150 const char *name;
151 brix_checksum_alg_t alg;
152 } table[] = {
153 { "adler32", BRIX_CHECKSUM_ADLER32 },
154 { "crc32", BRIX_CHECKSUM_CRC32 },
155 { "crc32c", BRIX_CHECKSUM_CRC32C },
156 { "md5", BRIX_CHECKSUM_MD5 },
157 { "sha1", BRIX_CHECKSUM_SHA1 },
158 { "sha256", BRIX_CHECKSUM_SHA256 },
159 { "crc64", BRIX_CHECKSUM_CRC64 },
160 { "crc64xz", BRIX_CHECKSUM_CRC64 }, /* alias of crc64 (CRC-64/XZ) */
161 { "crc64nvme", BRIX_CHECKSUM_CRC64NVME },
162 { "zcrc32", BRIX_CHECKSUM_ZCRC32 }, /* XRootD zlib-CRC32 name */
163 };
164 size_t i;
165
166 for (i = 0; i < sizeof(table) / sizeof(table[0]); i++) {
167 if (strcmp(lname, table[i].name) == 0) {
168 *out = table[i].alg;
169 return NGX_OK;
170 }
171 }
172
173 return NGX_DECLINED;
174}

176/*
177 * brix_checksum_parse — parse algorithm name string into enum value.
178 *
179 * WHAT: Converts a checksum algorithm name (e.g. "crc32c", "sha256") from config or wire
180 * protocol into the corresponding enum value, with optional normalized lowercase output.
181 * WHY: Config directives and fattr responses carry algorithm names as strings; callers need
182 * the enum to dispatch to the right code path (u32 vs digest). HOW: 1) Normalize name
183 * (validate + lowercase) into buf → 2) Table-lookup buf → enum → 3) Set *alg if not NULL
184 * → 4) Copy normalized buf to output buffer → 5) Return NGX_OK/NGX_ERROR/NGX_DECLINED. */
185
186ngx_int_t
187brix_checksum_parse(const char *name, size_t len, brix_checksum_alg_t *alg,
188 char *normalized, size_t normalized_sz)
189{
190 char buf[32];
191 brix_checksum_alg_t parsed;
192
193 if (brix_checksum_normalize_name(name, len, buf, sizeof(buf)) != NGX_OK) {
194 return NGX_ERROR;
195 }
196
197 if (brix_checksum_lookup_alg(buf, &parsed) != NGX_OK) {
198 return NGX_DECLINED;
199 }
200
201 if (alg != NULL) {
202 *alg = parsed;
203 }
204
205 /* Emit the CANONICAL name (not the raw input) so aliases normalize — e.g.
206 * "crc64xz" → "crc64" — keeping xattr cache keys and wire names consistent.
207 * For every non-aliased algorithm the canonical name equals buf, so this is
208 * a no-op there. parsed is always valid here, so name() is never NULL. */
209 if (normalized != NULL && normalized_sz > 0) {
210 ngx_cpystrn((u_char *) normalized,
211 (u_char *) brix_checksum_name(parsed), normalized_sz);
212 }
213
214 return NGX_OK;
215}

242/*
243 * brix_checksum_u32_fd — compute Adler-32 / CRC-32 / CRC-32c on file descriptor.
244 *
245 * WHAT: Reads an entire file via pread in 64KB chunks, accumulating a 32-bit checksum
246 * (Adler-32 via zlib adler32(), CRC-32 via crc32(), or CRC-32c via brix_crc32c_extend()).
247 * WHY: pgread/pgwrite require per-page CRC32c validation; fattr/PROPFIND return checksums.
248 * This shared function serves both XRootD and WebDAV/S3 callers without duplication.
249 * HOW: 1) Validate alg is u32 type → 2) Initialise accumulator (adler32/crc32 or zero for crc32c)
250 * → 3) Loop pread(buf, 64KB, offset) with EINTR retry → 4) Update accumulator per chunk
251 * → 5) Return result via *out. Returns NGX_ERROR on read failure or invalid alg. */
252
253ngx_int_t
254brix_checksum_u32_fd(brix_checksum_alg_t alg, int fd, const char *path,
255 ngx_log_t *log, uint32_t *out)
256{
257 if (out == NULL || !brix_checksum_is_u32(alg)) {
258 return NGX_ERROR;
259 }
260 /* Compute via the shared (ngx-free) kernel; errno carries any read error so
261 * we keep the module's specific read-error log line. (alg ordinals match.) */
262 if (brix_cksum_u32_fd((int) alg, fd, out) != 0) {
263 brix_checksum_log_read_error(log, errno, brix_checksum_name(alg), path);
264 return NGX_ERROR;
265 }
266 return NGX_OK;
267}

326/*
327 * brix_checksum_hex_fd — compute checksum and return as hex string.
328 *
329 * WHAT: Computes a checksum on file descriptor and writes the result as a hex-encoded string,
330 * branching between 32-bit (snprintf "%08x") and digest (EVP_DigestFinal_ex + hex_encode)
331 * paths based on algorithm type. WHY: This is the unified entry point for all callers that
332 * need a hex checksum — XRootD fattr, WebDAV PROPFIND, S3 metadata — without requiring
333 * callers to know whether their algorithm is u32 or digest. HOW: 1) Branch on is_u32() →
334 * 2) For u32: validate hexsz >= 9 bytes, call u32_fd(), snprintf("%08x") → 3) For digest:
335 * validate hexsz >= EVP_MAX_MD_SIZE*2+1, call digest_fd(), hex_encode(). Returns NGX_ERROR
336 * on buffer overflow or computation failure. */
337
338ngx_int_t
339brix_checksum_hex_fd(brix_checksum_alg_t alg, int fd, const char *path,
340 ngx_log_t *log, char *hex, size_t hexsz)
341{
342 if (brix_checksum_is_u32(alg)) {
343 uint32_t value;
344
345 if (hexsz < 9) {
346 return NGX_ERROR;
347 }
348
349 if (brix_checksum_u32_fd(alg, fd, path, log, &value) != NGX_OK) {
350 return NGX_ERROR;
351 }
352
353 snprintf(hex, hexsz, "%08x", (unsigned int) value);
354 return NGX_OK;
355 }
356
357 if (brix_checksum_is_u64(alg)) {
358 uint64_t value;
359
360 if (hexsz < 17) { /* 16 hex digits + NUL */
361 return NGX_ERROR;
362 }
363
364 if (brix_checksum_u64_fd(alg, fd, path, log, &value) != NGX_OK) {
365 return NGX_ERROR;
366 }
367
368 snprintf(hex, hexsz, "%016llx", (unsigned long long) value);
369 return NGX_OK;
370 }
371
372 {
373 unsigned char digest[EVP_MAX_MD_SIZE];
374 unsigned int digest_len;
375
376 if (hexsz < (EVP_MAX_MD_SIZE * 2 + 1)) {
377 return NGX_ERROR;
378 }
379
380 if (brix_checksum_digest_fd(alg, fd, path, log, digest, &digest_len)
381 != NGX_OK)
382 {
383 return NGX_ERROR;
384 }
385
386 brix_checksum_hex_encode(digest, digest_len, hex);
387 return NGX_OK;
388 }
389}

391/*
392 * brix_checksum_hex_obj — hex checksum of a driver-bound OBJECT.
393 *
394 * WHAT: Same contract as brix_checksum_hex_fd, but reads every byte through
395 * obj->driver (block-striped / object store) so a multi-block file is
396 * summed in full rather than just its first backing block. WHY: a Layer-3
397 * backend's open handle exposes only block 0 as a bare fd; checksum-at-rest
398 * must traverse the whole logical object. HOW: branch on alg type and call
399 * the obj-aware kernel (brix_cksum_u32_obj / _u64_obj / _digest_obj),
400 * formatting identically to hex_fd. */
401ngx_int_t
402brix_checksum_hex_obj(brix_checksum_alg_t alg, brix_sd_obj_t *obj,
403 const char *path, ngx_log_t *log, char *hex, size_t hexsz)
404{
405 if (obj == NULL || obj->driver == NULL) {
406 return NGX_ERROR;
407 }
408
409 if (brix_checksum_is_u32(alg)) {
410 uint32_t value;
411
412 if (hexsz < 9 || brix_cksum_u32_obj((int) alg, obj, &value) != 0) {
413 brix_checksum_log_read_error(log, errno,
414 brix_checksum_name(alg), path);
415 return NGX_ERROR;
416 }
417 snprintf(hex, hexsz, "%08x", (unsigned int) value);
418 return NGX_OK;
419 }
420
421 if (brix_checksum_is_u64(alg)) {
422 uint64_t value;
423
424 if (hexsz < 17 || brix_cksum_u64_obj((int) alg, obj, &value) != 0) {
425 brix_checksum_log_read_error(log, errno,
426 brix_checksum_name(alg), path);
427 return NGX_ERROR;
428 }
429 snprintf(hex, hexsz, "%016llx", (unsigned long long) value);

431 }

442 return NGX_ERROR;

445 return NGX_OK;
446 }
447}
