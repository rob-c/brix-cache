39/* brix_cache_ensure_parent — extract path's dirname and create it recursively
40 * (brix_mkdir_recursive, 0755) so a fill worker writing a newly-discovered origin
41 * file doesn't ENOENT; NGX_OK if it exists or was created, -1 on ENAMETOOLONG. */
42
43int
44brix_cache_ensure_parent(const char *path)
45{
46 char parent[PATH_MAX];
47 char *slash;
48 int n;
49
50 n = snprintf(parent, sizeof(parent), "%s", path);
51 if (n < 0 || (size_t) n >= sizeof(parent)) {
52 errno = ENAMETOOLONG;
53 return -1;
54 }
55
56 slash = strrchr(parent, '/');
57 if (slash == NULL || slash == parent) {
58 return 0;
59 }
60
61 *slash = '\0';
62 /* 0700: the cache tree is svc-owned and never client-listable; a mapped
63 * low-priv uid must not be able to traverse/enumerate another user's cached
64 * paths on a shared filesystem. */
65 return brix_mkdir_recursive(parent, 0700);
66}

68/* brix_cache_file_ready — three-state readiness for the open-or-fill decision:
69 * 1 = an existing regular file (cache hit), 0 = ENOENT (cache miss → schedule fill),
70 * -1 = stat failure or non-regular type (S_ISREG rejects dirs/symlinks, errno set). */
71
72int
73brix_cache_file_ready(const char *path)
74{
75 struct stat st;
76
77 if (stat(path, &st) != 0) {
78 return (errno == ENOENT) ? 0 : -1;
79 }
80
81 if (!S_ISREG(st.st_mode)) {
82 errno = S_ISDIR(st.st_mode) ? EISDIR : EINVAL;
83 return -1;
84 }
85
86 return 1;
87}

94{
95 if (conf == NULL) {

101 /* Tier grammar (§14a): the durable cache state + the evictable objects live in
102 * the PHYSICAL cache STORE, never the advertised (logical) cache_root. The
103 * authoritative location is the store instance's OWN local dir — ask the
104 * resolved cstore, not the URL string, so this is driver-truthful for every
105 * scheme. A non-local store (s3/rados/remote) returns NULL: the reaper/state
106 * layer MUST then decline filesystem reaping (store-native eviction handles it)

108 * physical root ONLY in the legacy (no-cache_store) config below. */

111 }
112 if (conf->cache_root.len > 0) {
