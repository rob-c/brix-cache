172{

183

185 CHECK(CLOSE(brix_scan_tb_wait_ns(&tb, 1.0e6, 500000000ull), 5.0e8, 1.0e6),

189 brix_scan_tb_init(&tb, 0.0, 0.0, 0);
190 CHECK(brix_scan_tb_wait_ns(&tb, 1.0e9, 12345) == 0, "rate 0 = unlimited");

309static void
310test_drift(void)
311{
312 brix_scan_driftset_t *s = brix_scan_driftset_create(4);
313 int64_t cs = -1;
314
315 CHECK(s != NULL, "driftset created");
316 CHECK(brix_scan_driftset_add(s, "a", 10) == 0, "add a");
317 CHECK(brix_scan_driftset_add(s, "b", 20) == 0, "add b");
318 CHECK(brix_scan_driftset_add(s, "orphan", 5) == 0, "add orphan");
319
320 /* namespace stream vs catalog */
321 CHECK(brix_scan_driftset_match(s, "a", 10, &cs) == BRIX_DRIFT_IN_BOTH,
322 "a in both"); CHECK(cs == 10, "a cat size");
323 CHECK(brix_scan_driftset_match(s, "b", 99, &cs) == BRIX_DRIFT_SIZE_MISMATCH,
324 "b size mismatch"); CHECK(cs == 20, "b cat size");
325 CHECK(brix_scan_driftset_match(s, "c", 1, NULL) == BRIX_DRIFT_NAMESPACE_ONLY,
326 "c namespace-only");
327
328 /* growth past initial capacity must preserve entries */
329 CHECK(brix_scan_driftset_add(s, "d", 40) == 0, "add d (grow)");
330 CHECK(brix_scan_driftset_add(s, "e", 50) == 0, "add e (grow)");
331 CHECK(brix_scan_driftset_match(s, "d", 40, NULL) == BRIX_DRIFT_IN_BOTH,
332 "d survives grow");
333
334 /* unmatched catalog keys are orphans: "orphan" and "e" */
335 g_orphan_n = 0;
336 brix_scan_driftset_orphans(s, orphan_collect, NULL);
337 CHECK(g_orphan_n == 2, "two orphans");
338 {
339 int seen_orphan = 0, seen_e = 0, i;
340 for (i = 0; i < g_orphan_n; i++) {
341 if (strcmp(g_orphan_keys[i], "orphan") == 0
342 && g_orphan_sizes[i] == 5) {
343 seen_orphan = 1;
344 }
345 if (strcmp(g_orphan_keys[i], "e") == 0) {
346 seen_e = 1;
347 }
348 }
349 CHECK(seen_orphan && seen_e, "orphans are exactly the unmatched keys");
350 }
351
352 /* duplicate add updates size, not a second entry */
353 brix_scan_driftset_add(s, "a", 11);
354 CHECK(brix_scan_driftset_match(s, "a", 11, &cs) == BRIX_DRIFT_IN_BOTH
355 && cs == 11, "duplicate add updates size");
356
357 brix_scan_driftset_free(s);
358}
