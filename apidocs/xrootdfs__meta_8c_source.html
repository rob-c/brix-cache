10/* Report a file we currently hold open for write whose committed final path the
11 * server cannot yet stat (writes are staged to a temp object and renamed into
12 * place only on close). Without this, the getattr the kernel issues right after
13 * CREATE — to finish the open — fails ENOENT and the create()/open() call fails.
14 * Returns 1 (handled, stbuf filled) when fi is a writable handle, else 0. */
15static int
16getattr_from_open_writer(struct fuse_file_info *fi, struct stat *stbuf)
17{
18 afh *h;
19
20 if (fi == NULL || fi->fh == 0) {
21 return 0;
22 }
23 h = (afh *) (uintptr_t) fi->fh;
24 if (h == NULL || !h->writable) {
25 return 0;
26 }
27 memset(stbuf, 0, sizeof(*stbuf));
28 stbuf->st_mode = S_IFREG | 0644;
29 stbuf->st_nlink = 1;
30 stbuf->st_size = (off_t) h->wsize;
31 return 1;
32}

34int
35xfs_getattr(const char *path, struct stat *stbuf, struct fuse_file_info *fi)
36{
37 brix_statinfo si;
38 brix_status st;
39 if (strcmp(path, "/") == 0) {
40 memset(stbuf, 0, sizeof(*stbuf));
41 stbuf->st_mode = S_IFDIR | 0755;
42 stbuf->st_nlink = 2;
43 return 0;
44 }
45 brix_status_clear(&st);
46 if (g_web) {
47 char pbuf[XRDC_PATH_MAX];
48 if (brix_web_stat(&g_weburl, srv_path(path, pbuf, sizeof(pbuf)), g_bearer,
49 g_web_verify, g_web_ca, &si, &st) != 0) {
50 return xfs_err(&st);
51 }
52 xfs_fill_stat(&si, stbuf);
53 return 0;
54 }
55 char pbuf[XRDC_PATH_MAX];
56 struct brix_fuse_ctx_stat a = { srv_path(path, pbuf, sizeof(pbuf)), &si };
57 int rc = xfs_meta(brix_fuse_op_lstat, &a, &st); /* lstat: symlinks present as S_IFLNK */
58 if (rc != 0) {
59 /* A newly created file we still hold open is staged server-side until
60 * close, so its final path is not yet visible to stat (ENOENT). Report
61 * it from the open write handle instead of failing the open. */
62 if (rc == -ENOENT && getattr_from_open_writer(fi, stbuf)) {
63 return 0;
64 }
65 return rc;
66 }
67 xfs_fill_stat(&si, stbuf);
68 return 0;
69}

248int
249xfs_readlink(const char *path, char *buf, size_t size)
250{
251 if (g_web) return -EINVAL;
252 if (!g_ext_readlink || size == 0) {
253 return -ENOTSUP;
254 }
255 brix_status st; brix_status_clear(&st);
256 char pbuf[XRDC_PATH_MAX];
257 const char *sp = srv_path(path, pbuf, sizeof(pbuf));
258 /* readlink is read-only + idempotent → retry-safe. Deadline-bounded like the
259 * rest of the metadata path (ride a lossy link out for g_max_stall rather than
260 * a fixed count); g_max_stall <= 0 falls back to the g_max_retries count. */
261 uint64_t deadline = (g_max_stall > 0)
262 ? brix_mono_ns() + (uint64_t) g_max_stall * 1000000ULL : 0;
263 unsigned max = g_max_retries > 0 ? (unsigned) g_max_retries : 0;
264 unsigned attempt;
265 for (attempt = 0; ; attempt++) {
266 int done = deadline ? (brix_mono_ns() >= deadline) : (attempt >= max);
267 if (attempt > 0) { brix_backoff_sleep_fast(attempt - 1); }
268 brix_conn *c = brix_pool_checkout(g_pool, &st);
269 if (c == NULL) {
270 if (brix_status_retryable(&st) && !done) { continue; }
271 return xfs_err(&st);
272 }
273 ssize_t n = brix_readlink(c, sp, buf, size, &st);
274 brix_pool_checkin(g_pool, c, n >= 0 ? 1 : xfs_conn_healthy(&st));
275 if (n >= 0) {
276 return 0; /* FUSE: buf is NUL-terminated, return 0 on success */
277 }
278 if (!brix_status_retryable(&st) || done) {
279 return xfs_err(&st);
280 }
281 }
