12/* Read up to n bytes; branches on TLS. Returns bytes (>0), 0 on EOF, -1 on error. */
13ssize_t
14httpx_read_some(brix_io *io, void *buf, size_t n, int timeout_ms, brix_status *st)
15{
16 if (io->ssl != NULL) {
17 size_t got = 0;
18 if (brix_tls_read_some(io, buf, n, &got, st) != 0) {
19 return -1;
20 }
21 return (ssize_t) got; /* 0 = EOF */
22 }
23 {
24 struct pollfd pfd;
25 ssize_t r;
26 int pr;
27 pfd.fd = io->fd; pfd.events = POLLIN; pfd.revents = 0;
28 do { pr = poll(&pfd, 1, timeout_ms); } while (pr < 0 && errno == EINTR);
29 if (pr <= 0) {
30 brix_status_set(st, XRDC_ESOCK, pr == 0 ? ETIMEDOUT : errno,
31 "http read %s", pr == 0 ? "timed out" : "poll failed");
32 return -1;
33 }
34 do { r = read(io->fd, buf, n); } while (r < 0 && errno == EINTR);
35 if (r < 0) {
36 brix_status_set(st, XRDC_ESOCK, errno, "http read: %s", strerror(errno));
37 return -1;
38 }
39 return r;
40 }
41}

44/* Decide whether the full response body has arrived in buf[0,total) given the
45 * header block end (body_off) and the framing parsed from the headers.
46 *
47 * WHY: Some servers (notably XrdHttp) advertise "Connection: Close" yet keep the
48 * TLS socket open after the body, so reading to EOF blocks until timeout. We
49 * must stop exactly at the framed body length instead. Returns 1 when the
50 * body is complete; 0 means "keep reading" (genuine EOF still terminates). */
51int
52httpx_body_complete(const char *buf, size_t total, size_t body_off,
53 long long clen, int chunked)
54{
55 if (clen >= 0) {
56 return (total - body_off) >= (size_t) clen;
57 }
58 if (chunked) {
59 const char *b = buf + body_off;
60 size_t bl = total - body_off;
61 /* terminating zero-length chunk: "0\r\n\r\n" (optionally CRLF-prefixed) */
62 if (bl >= 5 && memcmp(b, "0\r\n\r\n", 5) == 0) {
63 return 1;
64 }
65 return memmem(b, bl, "\r\n0\r\n\r\n", 7) != NULL;
66 }
67 return 0; /* EOF-delimited: only the real EOF (read == 0) completes it */
68}

71/* ---- File-local request descriptor for one HTTP client exchange ----
72 *
73 * WHAT: Bundles the immutable inputs of a single request/response exchange
74 * (target host/port, request line + headers, optional body, timeout) so
75 * the exchange helpers take one pointer instead of an 11-argument list.
76 *
77 * WHY: httpx_exchange's frozen extern signature carries 11 parameters; threading
78 * them one-by-one through the build/send/read split would re-inflate every
79 * helper's parameter count. A file-local descriptor keeps the split helpers
80 * small and their data flow explicit without touching the public header.
81 *
82 * HOW: httpx_exchange fills one on the stack from its own parameters, then hands
83 * a const pointer to each phase helper; no field is mutated after fill. */
84typedef struct {
85 const char *host;
86 int port;
87 const char *method;
88 const char *path;
89 const char *extra_headers;
90 const void *body;
91 size_t blen;
92 int timeout_ms;
93} httpx_exchange_t;

96/* ---- Rolling parse state while reading a framed HTTP response ----
97 *
98 * WHAT: Tracks whether the header block has been seen and, once it has, the
99 * body offset and the framing (Content-Length / chunked) parsed from it.
100 *
101 * WHY: The read loop must decide, after each read, whether the framed body is
102 * complete. Grouping the framing state keeps the loop body flat and lets
103 * the header-parse step return its result in one place.
104 *
105 * HOW: Initialised to "no headers yet" (clen -1); httpx_read_frame_headers sets
106 * it once the CRLFCRLF terminator arrives. */
107typedef struct {
108 int have_hdrs;
109 int chunked;
110 size_t body_off;
111 long long clen;
112} httpx_frame_t;

115/* ---- Build the HTTP/1.1 request bytes into a caller buffer ----
116 *
117 * WHAT: Formats the request line, fixed headers, caller-supplied extra headers,
118 * and (when a body is present) the Content-Length header into req[0,cap).
119 * Returns the byte length on success, or -1 (with st set) if it overflows.
120 *
121 * WHY: Isolating the fixed request-formatting from the send/read phases keeps
122 * httpx_exchange linear and makes the "request too long" guards testable
123 * in one place.
124 *
125 * HOW: 1. Format the request line + headers, terminating the header block with a
126 * bare CRLF only when there is no body. 2. When a body is present, append
127 * the Content-Length header (which closes the header block). 3. Guard each
128 * snprintf against truncation. */
129static int
130httpx_build_request(const httpx_exchange_t *ex, char *req, size_t cap,
131 brix_status *st)
132{
133 char hp[300];
134 int rlen;
135
136 brix_format_host_port(ex->host, (uint16_t) ex->port, hp, sizeof(hp));
137 rlen = snprintf(req, cap,
138 "%s %s HTTP/1.1\r\nHost: %s\r\nUser-Agent: xrddiag\r\n"
139 "Accept: */*\r\nConnection: close\r\n%s%s",
140 ex->method, ex->path[0] ? ex->path : "/", hp,
141 ex->extra_headers ? ex->extra_headers : "",
142 (ex->body != NULL && ex->blen > 0) ? "" : "\r\n");
143 if (rlen < 0 || (size_t) rlen >= cap) {
144 brix_status_set(st, XRDC_EUSAGE, 0, "http_req: request too long");
145 return -1;
146 }
147 if (ex->body != NULL && ex->blen > 0) {
148 char cl[64];
149 int cn = snprintf(cl, sizeof(cl), "Content-Length: %zu\r\n\r\n", ex->blen);
150 if (cn < 0 || (size_t) (rlen + cn) >= cap) {
151 brix_status_set(st, XRDC_EUSAGE, 0, "http_req: request too long");
152 return -1;
153 }
154 memcpy(req + rlen, cl, (size_t) cn);
155 rlen += cn;
156 }
157 return rlen;
158}

161/* ---- Send the request head and optional body over io ----
162 *
163 * WHAT: Writes req[0,rlen) then, if the exchange carries a body, the body bytes.
164 * Returns 0 on success, -1 (with st set) on a write error.
165 *
166 * WHY: Separates the side-effecting write from request formatting so the
167 * orchestrator reads as a flat build → send → read sequence.
168 *
169 * HOW: 1. brix_write_full the header block. 2. brix_write_full the body when
170 * one is present. */
171static int
172httpx_send_request(brix_io *io, const httpx_exchange_t *ex, const char *req,
173 size_t rlen, brix_status *st)
174{
175 if (brix_write_full(io, req, rlen, st) != 0) {
176 return -1;
177 }
178 if (ex->body != NULL && ex->blen > 0 &&
179 brix_write_full(io, ex->body, ex->blen, st) != 0) {
180 return -1;
181 }
182 return 0;
183}

186/* ---- Parse framing out of a just-completed header block ----
187 *
188 * WHAT: When buf[0,total) now contains the CRLFCRLF header terminator, records
189 * the body offset and the response framing (chunked vs Content-Length)
190 * into *fr. Leaves *fr untouched if the header block has not fully arrived.
191 *
192 * WHY: Header framing is parsed exactly once, the first time the terminator is
193 * seen; pulling it out of the read loop keeps the loop body flat and the
194 * temporary NUL-termination of the header block confined to one helper.
195 *
196 * HOW: 1. Locate CRLFCRLF; return if absent. 2. Temporarily NUL-terminate the
197 * header block so the string-based header readers stay in bounds.
198 * 3. Prefer a chunked Transfer-Encoding, else read Content-Length.
199 * 4. Restore the clobbered byte. */
200static void
201httpx_read_frame_headers(char *buf, size_t total, httpx_frame_t *fr)
202{
203 char *eoh = memmem(buf, total, "\r\n\r\n", 4);
204 char cl[64];
205 char *save;
206 char saved;
207
208 if (eoh == NULL) {
209 return;
210 }
211 save = eoh + 4;
212 saved = *save;
213 *save = '\0'; /* NUL-terminate header block */
214 fr->have_hdrs = 1;
215 fr->body_off = (size_t) (eoh + 4 - buf);
216 if (raw_header(buf, "Transfer-Encoding", cl, sizeof(cl)) &&
217 ci_contains(cl, "chunked")) {
218 fr->chunked = 1;
219 } else if (raw_header(buf, "Content-Length", cl, sizeof(cl))) {
220 fr->clen = strtoll(cl, NULL, 10);
221 }
222 *save = saved; /* restore the byte */
223}

226/* ---- Read the framed response into a freshly allocated buffer and parse it ----
227 *
228 * WHAT: Allocates an 8 MiB scratch buffer, reads until the framed body is
229 * complete (Content-Length / chunked / EOF), parses it into *resp, and
230 * frees the buffer. Returns 0 on success, -1 (with st set) on error.
231 *
232 * WHY: Owns its own scratch buffer so the caller's teardown stays linear (no
233 * shared cleanup label). Splitting the read/parse phase out of the
234 * build/send phases keeps each function single-purpose.
235 *
236 * HOW: 1. Allocate the ceiling buffer. 2. Loop: stop once the framed body is
237 * complete; at the ceiling do one extra read to tell EOF from truncation;
238 * otherwise read more, parsing framing the first time the header block
239 * completes. 3. NUL-terminate, parse, free. */
240static int
241httpx_read_response(brix_io *io, const httpx_exchange_t *ex,
242 brix_http_resp *resp, brix_status *st)
243{
244 char *buf;
245 size_t total = 0, cap = XRDC_HTTPX_MAX;
246 httpx_frame_t fr = { 0, 0, 0, -1 };
247
248 buf = (char *) malloc(cap);
249 if (buf == NULL) {
250 brix_status_set(st, XRDC_EPROTO, 0, "http_req: out of memory");
251 return -1;
252 }
253 for (;;) {
254 ssize_t r;
255 /* Stop as soon as the framed body is complete — do NOT wait for EOF:
256 * XrdHttp advertises "Connection: Close" but keeps the socket open. */
257 if (fr.have_hdrs &&
258 httpx_body_complete(buf, total, fr.body_off, fr.clen, fr.chunked)) {
259 break;
260 }
261 if (total >= cap - 1) {
262 /* At the ceiling: one more read distinguishes a genuine EOF from a
263 * truncated body — a silent truncation would corrupt a checksum probe. */
264 char sink[1];
265 ssize_t extra = httpx_read_some(io, sink, 1, ex->timeout_ms, st);
266 if (extra < 0) { free(buf); return -1; }
267 if (extra > 0) {
268 free(buf);
269 brix_status_set(st, XRDC_EPROTO, 0,
270 "http_req: response body exceeds the 8 MiB diagnostic limit");
271 return -1;
272 }
273 break; /* extra == 0: EOF exactly at the ceiling */
274 }
275 r = httpx_read_some(io, buf + total, cap - 1 - total, ex->timeout_ms, st);
276 if (r < 0) { free(buf); return -1; }
277 if (r == 0) { break; } /* genuine EOF */
278 total += (size_t) r;
279 if (!fr.have_hdrs) {
280 httpx_read_frame_headers(buf, total, &fr);
281 }
282 }
283 buf[total] = '\0';
284 httpx_parse(buf, total, resp);
285 free(buf);
286 return 0;
287}

290/* The protocol exchange over an established (and TLS-wrapped, if any) io: build +
291 * send the request, read the framed response (Content-Length / chunked / EOF),
292 * parse it. Owns its own scratch buffer
293 * so the caller's teardown stays linear (no shared cleanup label). 0 / -1. */
294int
295httpx_exchange(brix_io *io, const char *host, int port, const char *method,
296 const char *path, const char *extra_headers, const void *body,
297 size_t blen, int timeout_ms, brix_http_resp *resp, brix_status *st)
298{
299 httpx_exchange_t ex = { host, port, method, path, extra_headers,
300 body, blen, timeout_ms };
301 char req[2048];
302 int rlen;
303
304 rlen = httpx_build_request(&ex, req, sizeof(req), st);
305 if (rlen < 0) {
306 return -1;
307 }
308 if (httpx_send_request(io, &ex, req, (size_t) rlen, st) != 0) {
309 return -1;
310 }
311 return httpx_read_response(io, &ex, resp, st);
312}

315int
316brix_http_req(const char *host, int port, int tls, const char *method,
317 const char *path, const char *extra_headers,
318 const void *body, size_t blen, int timeout_ms, int verify,
319 const char *ca_dir, brix_http_resp *resp, brix_status *st)
320{
321 brix_io io;

324
325 memset(resp, 0, sizeof(*resp));
326 if (httpx_connect(&io, host, port, tls, verify, ca_dir, timeout_ms,
327 &tls_ctx, st) != 0) {
328 return -1;
329 }
330 resp->tls = tls;
331 if (tls) {
332 const char *v = NULL, *c = NULL;
333 brix_tls_client_info(&io, &v, &c);
334 snprintf(resp->tls_ver, sizeof(resp->tls_ver), "%s", v ? v : "?");
335 snprintf(resp->tls_cipher, sizeof(resp->tls_cipher), "%s", c ? c : "?");
336 }
337
338 rc = httpx_exchange(&io, host, port, method, path, extra_headers, body, blen,
339 timeout_ms, resp, st);
340
341 if (tls) {
342 brix_tls_client_free(&io, tls_ctx);
343 }
344 close(io.fd);
345 if (rc != 0) {
346 brix_http_resp_free(resp);
347 }
348 return rc;
