12/* Read up to n bytes; branches on TLS. Returns bytes (>0), 0 on EOF, -1 on error. */
13ssize_t
14httpx_read_some(brix_io *io, void *buf, size_t n, int timeout_ms, brix_status *st)
15{
16 if (io->ssl != NULL) {
17 size_t got = 0;
18 if (brix_tls_read_some(io, buf, n, &got, st) != 0) {
19 return -1;
20 }
21 return (ssize_t) got; /* 0 = EOF */
22 }
23 {
24 struct pollfd pfd;
25 ssize_t r;
26 int pr;
27 pfd.fd = io->fd; pfd.events = POLLIN; pfd.revents = 0;
28 do { pr = poll(&pfd, 1, timeout_ms); } while (pr < 0 && errno == EINTR);
29 if (pr <= 0) {
30 brix_status_set(st, XRDC_ESOCK, pr == 0 ? ETIMEDOUT : errno,
31 "http read %s", pr == 0 ? "timed out" : "poll failed");
32 return -1;
33 }
34 do { r = read(io->fd, buf, n); } while (r < 0 && errno == EINTR);
35 if (r < 0) {
36 brix_status_set(st, XRDC_ESOCK, errno, "http read: %s", strerror(errno));
37 return -1;
38 }
39 return r;
40 }
41}

44/* Decide whether the full response body has arrived in buf[0,total) given the
45 * header block end (body_off) and the framing parsed from the headers.
46 *
47 * WHY: Some servers (notably XrdHttp) advertise "Connection: Close" yet keep the
48 * TLS socket open after the body, so reading to EOF blocks until timeout. We
49 * must stop exactly at the framed body length instead. Returns 1 when the
50 * body is complete; 0 means "keep reading" (genuine EOF still terminates). */
51int
52httpx_body_complete(const char *buf, size_t total, size_t body_off,
53 long long clen, int chunked)
54{
55 if (clen >= 0) {
56 return (total - body_off) >= (size_t) clen;
57 }
58 if (chunked) {
59 const char *b = buf + body_off;
60 size_t bl = total - body_off;
61 /* terminating zero-length chunk: "0\r\n\r\n" (optionally CRLF-prefixed) */
62 if (bl >= 5 && memcmp(b, "0\r\n\r\n", 5) == 0) {
63 return 1;
64 }
65 return memmem(b, bl, "\r\n0\r\n\r\n", 7) != NULL;
66 }
67 return 0; /* EOF-delimited: only the real EOF (read == 0) completes it */
68}

71/* The protocol exchange over an established (and TLS-wrapped, if any) io: build +
72 * send the request, read the framed response (Content-Length / chunked / EOF),
73 * parse it. Owns its own scratch buffer
74 * so the caller's teardown stays linear (no shared cleanup label). 0 / -1. */
75int
76httpx_exchange(brix_io *io, const char *host, int port, const char *method,
77 const char *path, const char *extra_headers, const void *body,
78 size_t blen, int timeout_ms, brix_http_resp *resp, brix_status *st)
79{
80 char req[2048];
81 char *buf;
82 size_t total = 0, cap = XRDC_HTTPX_MAX;
83 int rlen;
84
85 char hp[300];
86 brix_format_host_port(host, (uint16_t) port, hp, sizeof(hp));
87 rlen = snprintf(req, sizeof(req),
88 "%s %s HTTP/1.1\r\nHost: %s\r\nUser-Agent: xrddiag\r\n"
89 "Accept: */*\r\nConnection: close\r\n%s%s",
90 method, path[0] ? path : "/", hp,
91 extra_headers ? extra_headers : "",
92 (body != NULL && blen > 0) ? "" : "\r\n");
93 if (rlen < 0 || (size_t) rlen >= sizeof(req)) {
94 brix_status_set(st, XRDC_EUSAGE, 0, "http_req: request too long");
95 return -1;
96 }
97 if (body != NULL && blen > 0) {
98 char cl[64];
99 int cn = snprintf(cl, sizeof(cl), "Content-Length: %zu\r\n\r\n", blen);
100 if (cn < 0 || (size_t) (rlen + cn) >= sizeof(req)) {
101 brix_status_set(st, XRDC_EUSAGE, 0, "http_req: request too long");
102 return -1;
103 }
104 memcpy(req + rlen, cl, (size_t) cn);
105 rlen += cn;
106 }
107 if (brix_write_full(io, req, (size_t) rlen, st) != 0) {
108 return -1;
109 }
110 if (body != NULL && blen > 0 && brix_write_full(io, body, blen, st) != 0) {
111 return -1;
112 }
113
114 buf = (char *) malloc(cap);
115 if (buf == NULL) {
116 brix_status_set(st, XRDC_EPROTO, 0, "http_req: out of memory");
117 return -1;
118 }
119 {
120 int have_hdrs = 0, chunked = 0;
121 size_t body_off = 0;
122 long long clen = -1;
123 for (;;) {
124 ssize_t r;
125 /* Stop as soon as the framed body is complete — do NOT wait for EOF:
126 * XrdHttp advertises "Connection: Close" but keeps the socket open. */
127 if (have_hdrs &&
128 httpx_body_complete(buf, total, body_off, clen, chunked)) {
129 break;
130 }
131 if (total >= cap - 1) {
132 /* At the ceiling: one more read distinguishes a genuine EOF from a
133 * truncated body — a silent truncation would corrupt a checksum probe. */
134 char sink[1];
135 ssize_t extra = httpx_read_some(io, sink, 1, timeout_ms, st);
136 if (extra < 0) { free(buf); return -1; }
137 if (extra > 0) {
138 free(buf);
139 brix_status_set(st, XRDC_EPROTO, 0,
140 "http_req: response body exceeds the 8 MiB diagnostic limit");
141 return -1;
142 }
143 break; /* extra == 0: EOF exactly at the ceiling */
144 }
145 r = httpx_read_some(io, buf + total, cap - 1 - total, timeout_ms, st);
146 if (r < 0) { free(buf); return -1; }
147 if (r == 0) { break; } /* genuine EOF */
148 total += (size_t) r;
149 if (!have_hdrs) {
150 char *eoh = memmem(buf, total, "\r\n\r\n", 4);
151 if (eoh != NULL) {
152 char cl[64];
153 char *save = eoh + 4;
154 char saved = *save;
155 *save = '\0'; /* NUL-terminate header block */
156 have_hdrs = 1;
157 body_off = (size_t) (eoh + 4 - buf);
158 if (raw_header(buf, "Transfer-Encoding", cl, sizeof(cl)) &&
159 ci_contains(cl, "chunked")) {
160 chunked = 1;
161 } else if (raw_header(buf, "Content-Length", cl, sizeof(cl))) {
162 clen = strtoll(cl, NULL, 10);
163 }
164 *save = saved; /* restore the byte */
165 }
166 }
167 }
168 }
169 buf[total] = '\0';
170 httpx_parse(buf, total, resp);
171 free(buf);
172 return 0;
173}

176int
177brix_http_req(const char *host, int port, int tls, const char *method,
178 const char *path, const char *extra_headers,
179 const void *body, size_t blen, int timeout_ms, int verify,
180 const char *ca_dir, brix_http_resp *resp, brix_status *st)
181{
182 brix_io io;
183 void *tls_ctx = NULL;
184 int rc;
185
186 memset(resp, 0, sizeof(*resp));
187 if (httpx_connect(&io, host, port, tls, verify, ca_dir, timeout_ms,
188 &tls_ctx, st) != 0) {
189 return -1;
190 }
191 resp->tls = tls;
192 if (tls) {
193 const char *v = NULL, *c = NULL;
194 brix_tls_client_info(&io, &v, &c);
195 snprintf(resp->tls_ver, sizeof(resp->tls_ver), "%s", v ? v : "?");
196 snprintf(resp->tls_cipher, sizeof(resp->tls_cipher), "%s", c ? c : "?");
197 }
198
199 rc = httpx_exchange(&io, host, port, method, path, extra_headers, body, blen,
200 timeout_ms, resp, st);
201
202 if (tls) {
203 brix_tls_client_free(&io, tls_ctx);
204 }
205 close(io.fd);
206 if (rc != 0) {
207 brix_http_resp_free(resp);
208 }
209 return rc;
210}
