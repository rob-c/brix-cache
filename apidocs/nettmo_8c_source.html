1/*
2 * nettmo.c — network timeout tunables + retry backoff for the native client.
3 *
4 * WHAT: Two process-wide timeout knobs and one backoff helper, the controls that
5 * keep a client usable on a flaky / hostile network (a misbehaving inline
6 * firewall that hangs, RST-injects, or intermittently drops connections):
7 * - connect timeout: caps the whole bring-up phase (TCP connect +
8 * protocol handshake + TLS + login). A firewall that completes the TCP
9 * handshake then black-holes the protocol bytes would otherwise hang
10 * the caller for the full data timeout; bounding bring-up turns that
11 * into a prompt, retryable failure.
12 * - io timeout: the steady-state per-operation read/write cap once a
13 * session is up (unchanged default).
14 * - backoff sleep: exponential + jitter between retries, so intermittent
15 * failures do not become a tight reconnect storm and many threads do
16 * not re-hammer a flapping firewall in lockstep.
17 * WHY: bring-up and steady-state want very different patience: a stalled
18 * handshake should fail in seconds (the reconnect machinery then rides over
19 * it), while a legitimate large read may take much longer. One blanket
20 * timeout cannot serve both.
21 * HOW: Each knob resolves once, lazily: an explicit setter (a CLI flag) wins,
22 * else the env var, else the compiled default. State is process-wide (a
23 * client process talks to one endpoint) and reached only through these
24 * accessors — same encapsulated-global pattern as netpref.c.
25 *
26 * Env vars (EXPANDED — client-only extensions, not vanilla XRootD variables):
27 * XRDC_CONNECT_TIMEOUT_MS, XRDC_IO_TIMEOUT_MS.
28 *
29 * Clean-room: C library + <stdatomic.h> only.
30 */

44/*
45 * resolve_ms — layered timeout resolver for connect and io timeouts.
46 *
47 * WHAT: Resolves `slot` once: setter (already in slot) > env var > xrdrc
48 * [defaults] > compiled default. Caches the result back in `slot`
49 * so the resolution is a single atomic load on every subsequent call.
50 * WHY: Connect and io timeouts should be consistent across all sessions in
51 * a process, and the resolution order must be deterministic.
52 * HOW: Checks the cached slot first (set by an earlier call or by a CLI
53 * setter via brix_tmo_set_*), then falls through env → xrdrc → dflt.
54 */
55static int
56resolve_ms(atomic_int *slot, const char *env, const char *xrdrc_key, int dflt)
57{
58 int v = atomic_load_explicit(slot, memory_order_relaxed);
59 if (v > 0) {
60 return v;
61 }
62 const char *e = getenv(env);
63 int parsed = (e != NULL && e[0] != '\0') ? atoi(e) : 0;
64 if (parsed > 0) {
65 atomic_store_explicit(slot, parsed, memory_order_relaxed);
66 return parsed;
67 }
68 int xv;
69 if (brix_xrdrc_default_ms(xrdrc_key, &xv)) {
70 atomic_store_explicit(slot, xv, memory_order_relaxed);
71 return xv;
72 }
73 atomic_store_explicit(slot, dflt, memory_order_relaxed);
74 return dflt;
75}

107unsigned
108brix_backoff_delay_ms(unsigned attempt, uint64_t *seed)
109{
110 /* Exponential base 100ms<<attempt, capped at 5s; attempt exponent clamped so
111 * a long-running retry loop cannot overflow the shift. */
112 unsigned shift = (attempt < 6) ? attempt : 6;
113 uint64_t base = 100ULL << shift;
114 if (base > 5000) {
115 base = 5000;
116 }
117 /* xorshift64 jitter in [0, base/2] decorrelates concurrent retriers so they
118 * do not all wake and reconnect on the same tick. */
119 uint64_t s = (*seed != 0) ? *seed : 0x9e3779b97f4a7c15ULL;
120 s ^= s << 13;
121 s ^= s >> 7;
122 s ^= s << 17;
123 *seed = s;
124 unsigned jitter = (unsigned) (s % (base / 2 + 1));
125 return (unsigned) base + jitter;
126}

128/* Transport-fault backoff base (ms), tunable via $XRDC_BACKOFF_BASE_MS (default
129 * 25). On a lossy-but-connected link the reconnect itself is sub-millisecond, so
130 * the backoff sleep dominates recovery latency; lowering it (e.g. =1) maximises
131 * throughput under heavy reset-style loss at the cost of a tighter retry loop
132 * against a genuinely-down peer. Read once and cached. */
133/*
134 * backoff_base_ms — resolve the transport-fault backoff base once and cache it.
135 *
136 * WHAT: Returns the number of milliseconds to use as the backoff base for
137 * transport-fault retries. Resolution order: env XRDC_BACKOFF_BASE_MS >
138 * xrdrc [defaults] backoff_base_ms > compiled default (25 ms).
139 * WHY: On a lossy link the default may be tuned site-wide via .xrdrc without
140 * requiring every operator to set an env var in their shell profile.
141 * HOW: Uses a sentinel-initialized atomic (−1 = not yet resolved) so the
142 * resolution happens only once per process lifetime.
143 */
144static unsigned
145backoff_base_ms(void)
146{
147 static atomic_int cached = -1;
148 int v = atomic_load_explicit(&cached, memory_order_relaxed);
149 if (v < 0) {
150 const char *e = getenv("XRDC_BACKOFF_BASE_MS");
151 long parsed;
152 if (e != NULL && *e != '\0') {
153 parsed = strtol(e, NULL, 10);
154 } else {
155 int xv;
156 parsed = brix_xrdrc_default_ms("backoff_base_ms", &xv) ? (long) xv : 25L;
157 }
158 if (parsed < 0) {
159 parsed = 0;
160 }
161 if (parsed > 10000) {
162 parsed = 10000;
163 }
164 v = (int) parsed;
165 atomic_store_explicit(&cached, v, memory_order_relaxed);
166 }
167 return (unsigned) v;
168}

170unsigned
171brix_backoff_delay_fast_ms(unsigned attempt, uint64_t *seed)
172{
173 /* Fast backoff for TRANSPORT faults (a connection reset/EOF is instant and
174 * not a sign of server overload): BASE<<attempt capped at 10x BASE (>=250ms
175 * for the 25ms default), plus jitter. A short cap lets many retries fit inside
176 * the patience window, which is what rides out a high packet-loss link — where
177 * each large read/reconnect is frequently severed and must simply be
178 * re-attempted promptly. BASE is $XRDC_BACKOFF_BASE_MS (default 25). */
179 unsigned base_ms = backoff_base_ms();
180 unsigned shift = (attempt < 4) ? attempt : 4;
181 uint64_t base = (uint64_t) base_ms << shift; /* e.g. 25,50,100,200,400→cap */
182 uint64_t cap = (base_ms >= 25) ? 250ULL : (uint64_t) base_ms * 10ULL;
183 if (base > cap) {
184 base = cap;
185 }
186 uint64_t s = (*seed != 0) ? *seed : 0x2545f4914f6cdd1dULL;
187 s ^= s << 13;
188 s ^= s >> 7;
189 s ^= s << 17;
190 *seed = s;
191 unsigned jitter = (unsigned) (s % (base / 2 + 1));
192 return (unsigned) base + jitter;
193}

195/* Sleep `ms` in <=50ms slices, observing a cooperative cancel between slices. */
196static void
197sleep_ms_cancelable(unsigned ms)
198{
199 while (ms > 0) {
200 if (brix_copy_quit_requested()) {
201 return;
202 }
203 unsigned slice = (ms > 50) ? 50 : ms;
204 struct timespec ts = { (time_t) (slice / 1000),
205 (long) (slice % 1000) * 1000000L };
206 nanosleep(&ts, NULL);
207 ms -= slice;
208 }
209}

211void
212brix_backoff_sleep(unsigned attempt)
213{
214 static _Thread_local uint64_t seed = 0;
215 if (seed == 0) {
216 seed = (uint64_t) brix_mono_ns() ^ (uint64_t) (uintptr_t) &seed;
217 }
218 sleep_ms_cancelable(brix_backoff_delay_ms(attempt, &seed));
219}
