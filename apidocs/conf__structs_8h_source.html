19/* OCSP certificate revocation checking (Feature 8e) — see auth/crypto/ocsp.c.
20 * Grouped as one sub-struct so the per-server config block stays navigable and
21 * every OCSP field is reached as conf->ocsp.<field>. */
22typedef struct {
23 ngx_flag_t enable; /* [brix_ocsp_enable on|off]
24 Query OCSP responder for each client certificate
25 after GSI chain verification. Default off. */
26 ngx_flag_t soft_fail; /* [brix_ocsp_soft_fail on|off]
27 If on (default), network errors and UNKNOWN status
28 are treated as GOOD (non-blocking).
29 REVOKED always fails regardless. */
30 ngx_flag_t require_nonce;/* [brix_ocsp_require_nonce on|off]
31 If on, an OCSP response that omits the nonce our
32 request carried is a hard failure (replay guard,
33 A-6 item 2). Default off — most CA responders
34 serve pre-signed, nonce-less responses. */
35 ngx_flag_t stapling; /* [brix_ocsp_stapling on|off]
36 Fetch an OCSP staple for the server certificate
37 at init time and serve it via the TLS status_request
38 extension (RFC 6066). Default off. */
39 u_char *staple_data; /* Cached DER-encoded OCSP response for stapling;
40 NULL if not yet fetched or stapling is disabled. */
41 size_t staple_len; /* Byte length of staple_data. */
42} brix_ocsp_conf_t;
