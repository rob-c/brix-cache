1/*
2 * core/types/conf_structs.h
3 *
4 * Per-server config sub-struct helper types, grouped out of the main
5 * ngx_stream_brix_srv_conf_t definition so config.h reads as a set of named
6 * concept groups rather than a flat wall of fields. Every field is reached as
7 * conf-><group>.<field> (e.g. conf->proxy.audit_log, conf->acc.tables).
8 *
9 * NOT self-contained: included by config.h at the point AFTER its prerequisite
10 * includes (writethrough_decision.h, shared_conf.h, kv.h, auth_cache.h,
11 * rate_limit.h) and after the translation unit has pulled in ngx core, OpenSSL
12 * (NGX_SSL), krb5 (BRIX_HAVE_KRB5), and <regex.h>, exactly where these typedefs
13 * previously lived inline. Do not include it directly — include config.h.
14 */

19/* OCSP certificate revocation checking (Feature 8e) — see auth/crypto/ocsp.c.
20 * Grouped as one sub-struct so the per-server config block stays navigable and
21 * every OCSP field is reached as conf->ocsp.<field>. */
22typedef struct {
23 ngx_flag_t enable; /* [brix_ocsp_enable on|off]
24 Query OCSP responder for each client certificate
25 after GSI chain verification. Default off. */
26 ngx_flag_t soft_fail; /* [brix_ocsp_soft_fail on|off]
27 If on (default), network errors and UNKNOWN status
28 are treated as GOOD (non-blocking).
29 REVOKED always fails regardless. */
30 ngx_flag_t stapling; /* [brix_ocsp_stapling on|off]
31 Fetch an OCSP staple for the server certificate
32 at init time and serve it via the TLS status_request
33 extension (RFC 6066). Default off. */
34 u_char *staple_data; /* Cached DER-encoded OCSP response for stapling;
35 NULL if not yet fetched or stapling is disabled. */
36 size_t staple_len; /* Byte length of staple_data. */
37} brix_ocsp_conf_t;

356/* ---- merge_srv_conf() helpers (literal-default groups) ------------------
357 * Apply parent->child inheritance + defaults, one per sub-struct, so a reviewer
358 * audits a concern's merge in one place. Only groups whose defaults are literals
359 * live here (this header is widely included and cannot see feature constants such
360 * as BRIX_HC_TYPE_*); groups with constant/computed defaults merge in
361 * server_conf.c. */

365{

368 ngx_conf_merge_value(c->stapling, p->stapling, 0);
