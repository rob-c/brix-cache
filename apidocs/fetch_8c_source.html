26/* The cache key (suffix under cache_root) for this fill's cache_path — what the
27 * cache STORAGE driver keys its namespace on. NULL if cache_path is not under
28 * cache_root (should never happen). */
29static const char *
30brix_cache_fill_key(const brix_cache_fill_t *t)
31{
32 size_t crlen = t->conf->cache_root.len;
33
34 if (crlen == 0
35 || ngx_strncmp(t->cache_path, t->conf->cache_root.data, crlen) != 0)
36 {
37 return NULL;
38 }
39 return t->cache_path + crlen; /* "/suffix" (leading slash) */

42/* Commit-then-verify for the staged (driver-backed) built-in fill: the bytes are
43 * already published by staged_commit. Stat for the size, verify the committed
44 * entry THROUGH the driver against the origin's advertised digest (evict on a
45 * mismatch — never serve proven-bad data), then write the .meta sidecar. */
46static int
47brix_cache_commit_staged(brix_cache_fill_t *t, brix_sd_instance_t *inst,
48 const char *key)
49{
50 ngx_log_t *log = (t->c != NULL) ? t->c->log : NULL;
51 brix_sd_stat_t sst;
52
53 if (inst->driver->stat(inst, key, &sst) != NGX_OK) {
54 brix_cache_set_syserror(t, kXR_IOError, "cache commit stat failed");
55 return -1;
56 }
57 t->file_size = (uint64_t) sst.size;
58
59 if (t->conf->cache_verify != BRIX_CACHE_VERIFY_OFF
60 && t->origin_cks_alg[0] != '\0')
61 {
62 brix_checksum_alg_t alg;
63 char alg_name[16];
64
65 if (brix_checksum_parse(t->origin_cks_alg,
66 ngx_strlen(t->origin_cks_alg),
67 &alg, alg_name, sizeof(alg_name)) == NGX_OK)
68 {
69 int e = 0;
70 brix_sd_obj_t *o = inst->driver->open(inst, key, BRIX_SD_O_READ,
71 0, &e);
72 if (o != NULL) {
73 char hex[EVP_MAX_MD_SIZE * 2 + 1];
74 int ok = (brix_checksum_hex_obj(alg, o, key, log, hex,
75 sizeof(hex)) == NGX_OK)
76 && ngx_strcmp(hex, t->origin_cks_hex) == 0;
77 (void) inst->driver->close(o);
78 if (o->heap_shell) {
79 free(o);
80 }
81 if (!ok) {
82 (void) inst->driver->unlink(inst, key, 0);
83 brix_cache_set_error(t, kXR_ServerError, 0,
84 "cache fill checksum mismatch (entry evicted)");
85 return -1;
86 }
87 }
88 }

158/* brix_cache_fill_from_source — THE single cache-fill spine (phase-63): fill from
159 * any SD source instance generically — `source->driver->open` → `pread` loop →
160 * staged sink → commit-then-verify. The caller owns `source`'s lifecycle (a
161 * registry-owned backend, or a per-fill sd_xroot/sd_http/sd_remote built from the
162 * cache_origin config). Checksum-on-fill reuses the xroot source's kXR_Qcksum.
163 * Returns 1 (admission decline) / -1 (error) / 0 (success). */
164static int
165brix_cache_fill_from_source(brix_cache_fill_t *t,
166 brix_sd_instance_t *source)
167{
168 ngx_stream_brix_srv_conf_t *conf = t->conf;
169 brix_sd_instance_t *cache_inst = brix_cache_storage(conf);
170 const char *key = brix_cache_fill_key(t);
171 brix_sd_obj_t *src;
172 brix_sd_staged_t *staged;
173 brix_cache_sink_t sink;
174 u_char *buf;
175 off_t off = 0;
176 int e = 0;
177
178 if (source == NULL || cache_inst == NULL || key == NULL) {
179 brix_cache_set_error(t, kXR_ServerError, 0,
180 "cache source/storage unavailable");
181 return -1;
182 }
183
184 src = source->driver->open(source, t->clean_path, BRIX_SD_O_READ, 0, &e);
185 if (src == NULL) {
186 brix_cache_set_error(t, (e == ENOENT) ? kXR_NotFound : kXR_IOError, e,
187 "cache source open failed");
188 return -1;
189 }
190 t->file_size = (uint64_t) src->snap.size;
191
192 {
193 brix_cache_admit_cfg_t admit = {
194 .deny_prefixes = conf->cache_deny_prefixes,
195 .allow_prefixes = conf->cache_allow_prefixes,
196 .size_limit = conf->cache_max_file_size,
197 .include_regex = conf->include_regex.set
198 ? &conf->include_regex.re : NULL,
199 };
200 if (brix_cache_admit(&admit, t->clean_path, (off_t) t->file_size, 0)
201 == BRIX_CACHE_DECLINE)
202 {
203 source->driver->close(src);
204 if (src->heap_shell) { free(src); }
205 t->result = NGX_DECLINED;
206 return 1;
207 }
208 }

210 /* SECURITY: cache-store files hold many users' bytes under one service-owned
211 * tree and are created + served AS THE WORKER (open_resolved_file.c from_cache
212 * branch, not the impersonation VFS). 0600 (not 0644) so a mapped low-priv uid
213 * cannot read another user's cached bytes by direct filesystem access; the
214 * per-user protocol gate (open_cache.c) already fronts the served path. */
215 staged = cache_inst->driver->staged_open(cache_inst, key, 0600, &e);
216 if (staged == NULL) {
217 source->driver->close(src);
218 if (src->heap_shell) { free(src); }
219 brix_cache_set_error(t, kXR_IOError, e, "cache staged open failed");
220 return -1;
221 }
222 sink.fd = -1;
223 sink.staged = staged;
224 sink.mem = NULL;
225 sink.mem_cap = 0;
226
227 buf = malloc(BRIX_CACHE_FETCH_CHUNK);
228 if (buf == NULL) {
229 cache_inst->driver->staged_abort(staged);
230 source->driver->close(src);
231 if (src->heap_shell) { free(src); }
232 brix_cache_set_error(t, kXR_NoMemory, 0,
233 "cache fill buffer alloc failed");
234 return -1;
235 }
236
237 for (;;) {
238 ssize_t n = src->driver->pread(src, buf, BRIX_CACHE_FETCH_CHUNK, off);
239
240 if (n < 0
241 || (n > 0
242 && brix_cache_sink_pwrite(&sink, buf, (size_t) n, off) != 0))
243 {
244 free(buf);
245 cache_inst->driver->staged_abort(staged);
246 source->driver->close(src);
247 if (src->heap_shell) { free(src); }
248 brix_cache_set_error(t, kXR_IOError, errno,
249 "cache source read / cache write failed");
250 return -1;
251 }
252 off += n;
253 if ((size_t) n < BRIX_CACHE_FETCH_CHUNK) {
254 break; /* short read = EOF */
255 }
256 }
257 free(buf);
258
259 /* Checksum-on-fill is the xroot source's kXR_Qcksum; other sources (http) offer
260 * no in-band digest here, so the verify policy decides on the local bytes. */
261 if (conf->cache_verify != BRIX_CACHE_VERIFY_OFF
262 && ngx_strcmp(brix_sd_backend_name(source), "xroot") == 0)
263 {
264 brix_sd_xroot_query_checksum(src, t->origin_cks_alg,
265 sizeof(t->origin_cks_alg), t->origin_cks_hex,
266 sizeof(t->origin_cks_hex));
267 }
268
269 source->driver->close(src);
270 if (src->heap_shell) { free(src); }
271
272 if (cache_inst->driver->staged_commit(staged, 0) != NGX_OK) {
273 brix_cache_set_error(t, kXR_IOError, 0, "cache staged commit failed");
274 return -1;
275 }
276 return brix_cache_commit_staged(t, cache_inst, key);
277}
