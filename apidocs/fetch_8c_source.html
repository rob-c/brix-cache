43 * already published by staged_commit. Stat for the size, verify the committed
44 * entry THROUGH the driver against the origin's advertised digest (evict on a
45 * mismatch — never serve proven-bad data), then write the .meta sidecar. */

49{

54 brix_cache_set_syserror(t, kXR_IOError, "cache commit stat failed");
55 return -1;
56 }

58

196/* brix_cache_fill_copy — WHAT: stream the source object into the staged sink.
197 * WHY: the sequential pread→pwrite copy loop is the single hot data-moving stage
198 * and its exact fill-range math (off advances by n; short read == EOF) must be
199 * preserved verbatim; extracting it drops the loop's error branches out of the
200 * spine's cyclomatic count. HOW: chunked pread from `src` at `off`, pwrite the
201 * same bytes at the same `off` into `sink`; 0 on success, -1 on read/write error
202 * (errno set by the failing call, as before). Caller owns buf lifetime. */
203static int
204brix_cache_fill_copy(brix_sd_obj_t *src, brix_cache_sink_t *sink, u_char *buf)
205{
206 off_t off = 0;
207
208 for (;;) {
209 ssize_t n = src->driver->pread(src, buf, BRIX_CACHE_FETCH_CHUNK, off);
210
211 if (n < 0
212 || (n > 0
213 && brix_cache_sink_pwrite(sink, buf, (size_t) n, off) != 0))
214 {
215 return -1;
216 }
217 off += n;
218 if ((size_t) n < BRIX_CACHE_FETCH_CHUNK) {
219 break; /* short read = EOF */
220 }
221 }
222 return 0;
223}

225/* brix_cache_fill_body — WHAT: the staged-write body of a fill: open the cache
226 * staged sink, allocate the copy buffer, run the copy loop, capture the
227 * on-fill checksum, then commit the staged bytes. WHY: pulling the staged
228 * lifecycle out of the spine keeps each function under the CCN cap while holding
229 * the sink/buf ownership contract (abort on any failure) in one place; the src
230 * object is closed by this function on every exit (the caller must not). HOW:
231 * staged_open (0600) → malloc → copy → xroot checksum query → src close →
232 * staged_commit. Returns 0 (bytes committed) / -1 (error; t error fields set,
233 * staged aborted). */
234static int
235brix_cache_fill_body(brix_cache_fill_t *t, brix_sd_instance_t *source,
236 brix_sd_obj_t *src, brix_sd_instance_t *cache_inst, const char *key)
237{
238 ngx_stream_brix_srv_conf_t *conf = t->conf;
239 brix_sd_staged_t *staged;
240 brix_cache_sink_t sink;
241 u_char *buf;
242 int e = 0;
243
244 /* SECURITY: cache-store files hold many users' bytes under one service-owned
245 * tree and are created + served AS THE WORKER (open_resolved_file.c from_cache
246 * branch, not the impersonation VFS). 0600 (not 0644) so a mapped low-priv uid
247 * cannot read another user's cached bytes by direct filesystem access; the
248 * per-user protocol gate (open_cache.c) already fronts the served path. */
249 staged = cache_inst->driver->staged_open(cache_inst, key, 0600, &e);
250 if (staged == NULL) {
251 brix_cache_src_close(source, src);
252 brix_cache_set_error(t, kXR_IOError, e, "cache staged open failed");
253 return -1;
254 }
255 sink.fd = -1;
256 sink.staged = staged;
257 sink.mem = NULL;
258 sink.mem_cap = 0;
259
260 buf = malloc(BRIX_CACHE_FETCH_CHUNK);
261 if (buf == NULL) {
262 cache_inst->driver->staged_abort(staged);
263 brix_cache_src_close(source, src);
264 brix_cache_set_error(t, kXR_NoMemory, 0,
265 "cache fill buffer alloc failed");
266 return -1;
267 }
268
269 if (brix_cache_fill_copy(src, &sink, buf) != 0) {
270 free(buf);
271 cache_inst->driver->staged_abort(staged);
272 brix_cache_src_close(source, src);
273 brix_cache_set_error(t, kXR_IOError, errno,
274 "cache source read / cache write failed");
275 return -1;
276 }
277 free(buf);
278
279 /* Checksum-on-fill is the xroot source's kXR_Qcksum; other sources (http) offer
280 * no in-band digest here, so the verify policy decides on the local bytes. */
281 if (conf->cache_verify != BRIX_CACHE_VERIFY_OFF
282 && ngx_strcmp(brix_sd_backend_name(source), "xroot") == 0)
283 {
284 brix_sd_xroot_query_checksum(src, t->origin_cks_alg,
285 sizeof(t->origin_cks_alg), t->origin_cks_hex,
286 sizeof(t->origin_cks_hex));
287 }
288
289 brix_cache_src_close(source, src);
290
291 if (cache_inst->driver->staged_commit(staged, 0) != NGX_OK) {
292 brix_cache_set_error(t, kXR_IOError, 0, "cache staged commit failed");
293 return -1;
294 }
295 return 0;
296}

298/* brix_cache_fill_from_source — THE single cache-fill spine (phase-63): fill from
299 * any SD source instance generically — `source->driver->open` → `pread` loop →
300 * staged sink → commit-then-verify. The caller owns `source`'s lifecycle (a
301 * registry-owned backend, or a per-fill sd_xroot/sd_http/sd_remote built from the
302 * cache_origin config). Checksum-on-fill reuses the xroot source's kXR_Qcksum.
303 * Returns 1 (admission decline) / -1 (error) / 0 (success). */
304static int
305brix_cache_fill_from_source(brix_cache_fill_t *t,
306 brix_sd_instance_t *source)
307{
308 ngx_stream_brix_srv_conf_t *conf = t->conf;
309 brix_sd_instance_t *cache_inst = brix_cache_storage(conf);
310 const char *key = brix_cache_fill_key(t);
311 brix_sd_obj_t *src;
312 int e = 0;
313
314 if (source == NULL || cache_inst == NULL || key == NULL) {
315 brix_cache_set_error(t, kXR_ServerError, 0,
316 "cache source/storage unavailable");
317 return -1;
318 }
319
320 src = source->driver->open(source, t->clean_path, BRIX_SD_O_READ, 0, &e);
321 if (src == NULL) {
322 brix_cache_set_error(t, (e == ENOENT) ? kXR_NotFound : kXR_IOError, e,
323 "cache source open failed");
324 return -1;
325 }
326 t->file_size = (uint64_t) src->snap.size;
327
328 if (brix_cache_fill_admitted(conf, t->clean_path, (off_t) t->file_size)) {
329 brix_cache_src_close(source, src);
330 t->result = NGX_DECLINED;
331 return 1;
332 }
333
334 if (brix_cache_fill_body(t, source, src, cache_inst, key) != 0) {
335 return -1;
336 }
337 return brix_cache_commit_staged(t, cache_inst, key);
338}
