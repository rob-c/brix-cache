1/*
2 * proxy_config.c - WebDAV upstream URL parsing and proxy backend resolution.
3 *
4 * Phase 21 Step D extends the original single-upstream parser into a
5 * multi-backend builder: every URL in conf->upstream_urls (or the legacy
6 * single conf->upstream_url) is resolved at configuration time, and each
7 * resolved address becomes one brix_webdav_backend_t carrying its own Host:
8 * header value, request-line URL base, and TLS flag. proxy.c round-robins
9 * across the resulting backend array with passive health tracking; the legacy
10 * conf->upstream_{host,url_base,resolved,ssl} fields are aliased to backend[0]
11 * so any older reader still sees a valid single backend.
12 */

18/* ---- Split "http://"/"https://" scheme off an upstream URL ----
19 *
20 * WHAT: Inspects url's scheme prefix and sets *ssl (1 for https, 0 for http),
21 * *scheme_len (8 or 7), and *default_port (443 or 80). Returns NGX_OK on a
22 * recognised scheme, or NGX_ERROR (after logging) for an empty URL or an
23 * unrecognised scheme.
24 *
25 * WHY: Isolates the scheme-classification branch ladder so the resolver's
26 * control flow stays flat and the accepted-scheme table lives in one place.
27 *
28 * HOW:
29 * 1. Reject a zero-length URL with the "no upstream URL given" diagnostic.
30 * 2. Case-insensitively match "https://" → ssl=1, len=8, port=443.
31 * 3. Otherwise match "http://" → ssl=0, len=7, port=80.
32 * 4. Otherwise log the "must start with http:// or https://" error and fail.
33 */
34static ngx_int_t
35webdav_proxy_split_scheme(ngx_conf_t *cf, ngx_str_t *url, ngx_flag_t *ssl,
36 size_t *scheme_len, in_port_t *default_port)
37{
38 if (url->len == 0) {
39 BRIX_DIAG_CONF(NGX_LOG_EMERG, cf, 0,
40 "brix_webdav_proxy: no upstream URL given",
41 "the directive was used without a backend URL argument",
42 "supply the backend, e.g. brix_webdav_proxy https://backend:1094;");
43 return NGX_ERROR;
44 }
45
46 if (ngx_strncasecmp(url->data, (u_char *) "https://", 8) == 0) {
47 *ssl = 1; *scheme_len = 8; *default_port = 443;
48 return NGX_OK;
49 }
50 if (ngx_strncasecmp(url->data, (u_char *) "http://", 7) == 0) {
51 *ssl = 0; *scheme_len = 7; *default_port = 80;
52 return NGX_OK;
53 }
54
55 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
56 "brix_webdav_proxy: upstream URL \"%V\" must start with"
57 " http:// or https://", url);
58 return NGX_ERROR;
59}

61/* ---- Parse the authority of an upstream URL into an ngx_url_t ----
62 *
63 * WHAT: Runs ngx_parse_url over the post-scheme portion of url with the given
64 * default_port and fills *u. Returns NGX_OK when at least one address is
65 * resolved, otherwise NGX_ERROR (after logging the parse error or the
66 * zero-address condition).
67 *
68 * WHY: Keeps the ngx_parse_url invocation plus its two failure diagnostics
69 * together, out of the main resolver body.
70 *
71 * HOW:
72 * 1. Zero *u and point it at url->data+scheme_len for uri_part parsing.
73 * 2. On ngx_parse_url failure, emit u->err (if any) and fail.
74 * 3. On zero resolved addresses, emit the "zero addresses" error and fail.
75 */
76static ngx_int_t
77webdav_proxy_parse_authority(ngx_conf_t *cf, ngx_str_t *url, size_t scheme_len,
78 in_port_t default_port, ngx_url_t *u)
79{
80 ngx_memzero(u, sizeof(ngx_url_t));
81 u->url.data = url->data + scheme_len;
82 u->url.len = url->len - scheme_len;
83 u->uri_part = 1;
84 u->default_port = default_port;
85
86 if (ngx_parse_url(cf->pool, u) != NGX_OK) {
87 if (u->err) {
88 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
89 "brix_webdav_proxy: \"%V\" in upstream URL \"%V\"",
90 &u->err, url);
91 }
92 return NGX_ERROR;
93 }
94 if (u->naddrs == 0) {
95 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
96 "brix_webdav_proxy: upstream URL \"%V\" resolved to zero"
97 " addresses", url);
98 return NGX_ERROR;
99 }
100 return NGX_OK;
101}

103/* ---- Build the pool-owned Host: header value for a resolved upstream ----
104 *
105 * WHAT: Formats "host" or "host:port" (IPv6 literals re-bracketed) into a
106 * fresh cf->pool allocation and stores it in *host. Returns NGX_OK, or
107 * NGX_ERROR on allocation failure.
108 *
109 * WHY: ngx_parse_url strips the brackets off "[::1]", so u->host arrives bare
110 * and must be re-bracketed before it is re-emitted in a Host header; the port
111 * is elided when it equals the scheme default.
112 *
113 * HOW:
114 * 1. NUL-terminate a bounded copy of u->host into a stack buffer.
115 * 2. Format host-only when u->port == default_port, else host:port.
116 * 3. Copy the formatted bytes into a cf->pool allocation and record .len.
117 */
118static ngx_int_t
119webdav_proxy_build_host(ngx_conf_t *cf, ngx_url_t *u, in_port_t default_port,
120 ngx_str_t *host)
121{
122 char hostz[256], fmt[288];
123 size_t hn, fn;
124 u_char *p;
125
126 hn = ngx_min(u->host.len, sizeof(hostz) - 1);
127 ngx_memcpy(hostz, u->host.data, hn);
128 hostz[hn] = '\0';
129
130 if (u->port == default_port) {
131 fn = brix_format_host(hostz, fmt, sizeof(fmt));
132 } else {
133 fn = brix_format_host_port(hostz, (uint16_t) u->port,
134 fmt, sizeof(fmt));
135 }
136
137 p = ngx_pnalloc(cf->pool, fn + 1);
138 if (p == NULL) return NGX_ERROR;
139 ngx_memcpy(p, fmt, fn);
140 p[fn] = '\0';
141 host->data = p;
142 host->len = fn;
143 return NGX_OK;
144}

146/* ---- Build the pool-owned request-line base "scheme://host[:port]" ----
147 *
148 * WHAT: Concatenates url's scheme prefix with the already-formatted host into
149 * a fresh cf->pool allocation and stores it in *url_base. Returns NGX_OK, or
150 * NGX_ERROR on allocation failure.
151 *
152 * WHY: Backends emit absolute request-line URLs; the scheme+authority base is
153 * computed once here and reused for every address of this URL.
154 *
155 * HOW:
156 * 1. Allocate scheme_len + host->len + 1 bytes from cf->pool.
157 * 2. Copy the scheme prefix, then the formatted host, then NUL-terminate.
158 */
159static ngx_int_t
160webdav_proxy_build_url_base(ngx_conf_t *cf, ngx_str_t *url, size_t scheme_len,
161 ngx_str_t *host, ngx_str_t *url_base)
162{
163 u_char *p;
164 size_t base_len = scheme_len + host->len;
165
166 p = ngx_pnalloc(cf->pool, base_len + 1);
167 if (p == NULL) return NGX_ERROR;
168 ngx_memcpy(p, url->data, scheme_len);
169 ngx_memcpy(p + scheme_len, host->data, host->len);
170 p[base_len] = '\0';
171 url_base->data = p;
172 url_base->len = base_len;
173 return NGX_OK;
174}

176/* ---- Lazily create the TLS context shared by all https backends ----
177 *
178 * WHAT: When ssl is set and conf has no upstream_ssl_ctx yet, creates one
179 * cf->pool ngx_ssl_t covering TLSv1 through TLSv1.3 and stores it on conf.
180 * Returns
181 * NGX_OK (including the no-op non-ssl / already-present cases), or NGX_ERROR
182 * on allocation or ngx_ssl_create failure.
183 *
184 * WHY: One shared client TLS context serves every https backend; creating it
185 * once keeps the resolver flat and avoids per-backend contexts. On builds
186 * without NGX_HTTP_SSL this is a no-op.
187 *
188 * HOW:
189 * 1. Do nothing unless ssl is set and no context exists yet.
190 * 2. Allocate and initialise an ngx_ssl_t, then publish it on conf.
191 */
192static ngx_int_t
193webdav_proxy_ensure_ssl_ctx(ngx_conf_t *cf,
194 ngx_http_brix_webdav_loc_conf_t *conf, ngx_flag_t ssl)
195{
196#if (NGX_HTTP_SSL)
197 if (ssl && conf->upstream_ssl_ctx == NULL) {
198 ngx_ssl_t *s = ngx_pcalloc(cf->pool, sizeof(ngx_ssl_t));
199 if (s == NULL) return NGX_ERROR;
200 s->log = cf->log;
201 if (ngx_ssl_create(s,
202 NGX_SSL_TLSv1 | NGX_SSL_TLSv1_1 | NGX_SSL_TLSv1_2
203 | NGX_SSL_TLSv1_3, NULL) != NGX_OK)
204 {
205 return NGX_ERROR;
206 }
207 conf->upstream_ssl_ctx = s; /* shared across https backends */
208 }
209#else
210 (void) cf; (void) conf; (void) ssl;
211#endif
212 return NGX_OK;
213}

215/* ---- Append one backend per resolved address of a single URL ----
216 *
217 * WHAT: Pushes u->naddrs entries onto conf->upstream_backends, each carrying
218 * one resolved address plus the shared host/url_base/ssl (and ssl_ctx).
219 * Returns NGX_OK, or NGX_ERROR if ngx_array_push fails.
220 *
221 * WHY: All addresses of a single URL share that URL's host/url_base/ssl; this
222 * loop materialises one round-robin-able backend per address.
223 *
224 * HOW:
225 * 1. For each address, push and zero a backend slot.
226 * 2. Copy the address into a single-address resolved view.
227 * 3. Alias the shared host/url_base/ssl (and, when built with TLS, ssl_ctx).
228 */
229static ngx_int_t
230webdav_proxy_push_backends(ngx_http_brix_webdav_loc_conf_t *conf,
231 ngx_url_t *u, ngx_str_t *host, ngx_str_t *url_base, ngx_flag_t ssl)
232{
233 ngx_uint_t a;
234 brix_webdav_backend_t *be;
235
236 for (a = 0; a < u->naddrs; a++) {
237 be = ngx_array_push(conf->upstream_backends);
238 if (be == NULL) return NGX_ERROR;
239 ngx_memzero(be, sizeof(*be));
240
241 be->resolved.sockaddr = u->addrs[a].sockaddr;
242 be->resolved.socklen = u->addrs[a].socklen;
243 be->resolved.naddrs = 1;
244 be->resolved.host = u->host;
245 be->resolved.port = u->port;
246 be->host = *host;
247 be->url_base = *url_base;
248 be->ssl = ssl;
249#if (NGX_HTTP_SSL)
250 be->ssl_ctx = ssl ? conf->upstream_ssl_ctx : NULL;
251#endif
252 }
253
254 return NGX_OK;
255}

257/* ---- Resolve one upstream URL and append its backends ----
258 *
259 * WHAT: Parses url (scheme + authority), formats its Host header and
260 * request-line base, ensures the shared https TLS context exists, and appends
261 * one brix_webdav_backend_t per resolved address. Returns NGX_OK, or
262 * NGX_ERROR (after logging) on any parse or allocation failure.
263 *
264 * WHY: All addresses of a single URL share that URL's host/url_base/ssl; this
265 * orchestrator runs the resolution steps in order, keeping each concern in its
266 * own testable helper.
267 *
268 * HOW:
269 * 1. Split the http/https scheme.
270 * 2. Parse the authority into an ngx_url_t (>=1 address).
271 * 3. Build the Host header value and the request-line base.
272 * 4. Ensure the shared TLS context for https backends.
273 * 5. Push one backend per resolved address.
274 */
275static ngx_int_t
276webdav_proxy_add_url(ngx_conf_t *cf, ngx_http_brix_webdav_loc_conf_t *conf,
277 ngx_str_t *url)
278{
279 ngx_url_t u;
280 in_port_t default_port = 0;
281 size_t scheme_len = 0;
282 ngx_flag_t ssl = 0;
283 ngx_str_t host;
284 ngx_str_t url_base;
285
286 if (webdav_proxy_split_scheme(cf, url, &ssl, &scheme_len,
287 &default_port) != NGX_OK)
288 {
289 return NGX_ERROR;
290 }
291
292 if (webdav_proxy_parse_authority(cf, url, scheme_len, default_port, &u)
293 != NGX_OK)
294 {
295 return NGX_ERROR;
296 }
297
298 if (webdav_proxy_build_host(cf, &u, default_port, &host) != NGX_OK) {
299 return NGX_ERROR;
300 }
301
302 if (webdav_proxy_build_url_base(cf, url, scheme_len, &host, &url_base)
303 != NGX_OK)
304 {
305 return NGX_ERROR;
306 }
307
308 if (webdav_proxy_ensure_ssl_ctx(cf, conf, ssl) != NGX_OK) {
309 return NGX_ERROR;
310 }
311
312 return webdav_proxy_push_backends(conf, &u, &host, &url_base, ssl);
313}

315ngx_int_t
316webdav_proxy_build_backends(ngx_conf_t *cf,
317 ngx_http_brix_webdav_loc_conf_t *conf)
318{
319 brix_webdav_backend_t *be0;
320
321 conf->upstream_backends = ngx_array_create(cf->pool, 4,
322 sizeof(brix_webdav_backend_t));
323 if (conf->upstream_backends == NULL) {
324 return NGX_ERROR;
325 }
326
327 if (conf->upstream_urls != NULL && conf->upstream_urls->nelts > 0) {
328 ngx_str_t *urls = conf->upstream_urls->elts;
329 ngx_uint_t i;
330 for (i = 0; i < conf->upstream_urls->nelts; i++) {
331 if (webdav_proxy_add_url(cf, conf, &urls[i]) != NGX_OK) {
332 return NGX_ERROR;
333 }
334 }
335 } else if (conf->upstream_url.len > 0) {
336 if (webdav_proxy_add_url(cf, conf, &conf->upstream_url) != NGX_OK) {
337 return NGX_ERROR;
338 }
339 } else {
340 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
341 "brix_webdav_proxy: missing upstream URL");
342 return NGX_ERROR;
343 }
344
345 if (conf->upstream_backends->nelts == 0) {
346 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
347 "brix_webdav_proxy: no usable backends");
348 return NGX_ERROR;
349 }
350
351 /* upstream_conf defaults (shared by all backends). */
352 if (conf->upstream_conf.connect_timeout == 0) {
353 conf->upstream_conf.connect_timeout = 60000;
354 }
355 if (conf->upstream_conf.send_timeout == 0) {
356 conf->upstream_conf.send_timeout = 60000;
357 }
358 if (conf->upstream_conf.read_timeout == 0) {
359 conf->upstream_conf.read_timeout = 60000;
360 }
361 if (conf->upstream_conf.buffer_size == 0) {
362 conf->upstream_conf.buffer_size = (size_t) ngx_pagesize;
363 }
364 conf->upstream_conf.buffering = 0;
365 conf->upstream_conf.bufs.num = 8;
366 conf->upstream_conf.bufs.size = (size_t) ngx_pagesize;
367 conf->upstream_conf.busy_buffers_size = 2 * ngx_pagesize;
368 conf->upstream_conf.max_temp_file_size = 0;
369 conf->upstream_conf.temp_file_write_size = 0;
370#if (NGX_HTTP_SSL)
371 if (conf->upstream_ssl_ctx != NULL) {
372 conf->upstream_conf.ssl = conf->upstream_ssl_ctx;
373 conf->upstream_conf.ssl_verify = 0; /* internal backend; no CA */
374 }
375#endif
376
377 /* Legacy aliases point at the first backend. */
378 be0 = (brix_webdav_backend_t *) conf->upstream_backends->elts;
379 conf->upstream_host = be0->host;
380 conf->upstream_url_base = be0->url_base;
381 conf->upstream_ssl = be0->ssl;
382 conf->upstream_resolved = &be0->resolved;
383
384 ngx_log_error(NGX_LOG_NOTICE, cf->log, 0,
385 "brix_webdav_proxy: %ui backend(s); primary %V (ssl=%d)",
386 conf->upstream_backends->nelts,
387 &conf->upstream_url_base, (int) conf->upstream_ssl);
388
389 return NGX_OK;
390}

400/*
