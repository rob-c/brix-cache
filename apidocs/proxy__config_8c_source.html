1/*
2 * proxy_config.c - WebDAV upstream URL parsing and proxy backend resolution.
3 *
4 * Phase 21 Step D extends the original single-upstream parser into a
5 * multi-backend builder: every URL in conf->upstream_urls (or the legacy
6 * single conf->upstream_url) is resolved at configuration time, and each
7 * resolved address becomes one brix_webdav_backend_t carrying its own Host:
8 * header value, request-line URL base, and TLS flag. proxy.c round-robins
9 * across the resulting backend array with passive health tracking; the legacy
10 * conf->upstream_{host,url_base,resolved,ssl} fields are aliased to backend[0]
11 * so any older reader still sees a valid single backend.
12 */
13
14#include "proxy_internal.h"
15#include "core/compat/host_format.h" /* brix_format_host[_port] — IPv6 bracketing */

18/*
19 * Resolve one upstream URL and append one backend per resolved address.
20 * All addresses of a single URL share that URL's host/url_base/ssl.
21 */
22static ngx_int_t
23webdav_proxy_add_url(ngx_conf_t *cf, ngx_http_brix_webdav_loc_conf_t *conf,
24 ngx_str_t *url)
25{
26 ngx_url_t u;
27 u_char *p;
28 in_port_t default_port;
29 size_t scheme_len;
30 ngx_flag_t ssl;
31 ngx_str_t host;
32 ngx_str_t url_base;
33 ngx_uint_t a;
34 brix_webdav_backend_t *be;
35
36 if (url->len == 0) {
37 BRIX_DIAG_CONF(NGX_LOG_EMERG, cf, 0,
38 "brix_webdav_proxy: no upstream URL given",
39 "the directive was used without a backend URL argument",
40 "supply the backend, e.g. brix_webdav_proxy https://backend:1094;");
41 return NGX_ERROR;
42 }
43
44 if (ngx_strncasecmp(url->data, (u_char *) "https://", 8) == 0) {
45 ssl = 1; scheme_len = 8; default_port = 443;
46 } else if (ngx_strncasecmp(url->data, (u_char *) "http://", 7) == 0) {
47 ssl = 0; scheme_len = 7; default_port = 80;
48 } else {
49 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
50 "brix_webdav_proxy: upstream URL \"%V\" must start with"
51 " http:// or https://", url);
52 return NGX_ERROR;
53 }
54
55 ngx_memzero(&u, sizeof(ngx_url_t));
56 u.url.data = url->data + scheme_len;
57 u.url.len = url->len - scheme_len;
58 u.uri_part = 1;
59 u.default_port = default_port;
60
61 if (ngx_parse_url(cf->pool, &u) != NGX_OK) {
62 if (u.err) {
63 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
64 "brix_webdav_proxy: \"%V\" in upstream URL \"%V\"",
65 &u.err, url);
66 }
67 return NGX_ERROR;
68 }
69 if (u.naddrs == 0) {
70 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
71 "brix_webdav_proxy: upstream URL \"%V\" resolved to zero"
72 " addresses", url);
73 return NGX_ERROR;
74 }
75
76 /* Host: header value — "host" or "host:port", with IPv6 literals bracketed.
77 * ngx_parse_url strips the brackets off "[::1]", so u.host arrives bare and
78 * must be re-bracketed ("[::1]") before it is re-emitted in a Host header. */
79 {
80 char hostz[256], fmt[288];
81 size_t hn, fn;
82
83 hn = ngx_min(u.host.len, sizeof(hostz) - 1);
84 ngx_memcpy(hostz, u.host.data, hn);
85 hostz[hn] = '\0';
86
87 if (u.port == default_port) {
88 fn = brix_format_host(hostz, fmt, sizeof(fmt));
89 } else {
90 fn = brix_format_host_port(hostz, (uint16_t) u.port,
91 fmt, sizeof(fmt));
92 }
93
94 p = ngx_pnalloc(cf->pool, fn + 1);
95 if (p == NULL) return NGX_ERROR;
96 ngx_memcpy(p, fmt, fn);
97 p[fn] = '\0';
98 host.data = p;
99 host.len = fn;
100 }
101
102 /* Request-line base — "scheme://host[:port]". */
103 {
104 size_t base_len = scheme_len + host.len;
105 p = ngx_pnalloc(cf->pool, base_len + 1);
106 if (p == NULL) return NGX_ERROR;
107 ngx_memcpy(p, url->data, scheme_len);
108 ngx_memcpy(p + scheme_len, host.data, host.len);
109 p[base_len] = '\0';
110 url_base.data = p;
111 url_base.len = base_len;
112 }
113
114#if (NGX_HTTP_SSL)
115 if (ssl && conf->upstream_ssl_ctx == NULL) {
116 ngx_ssl_t *s = ngx_pcalloc(cf->pool, sizeof(ngx_ssl_t));
117 if (s == NULL) return NGX_ERROR;
118 s->log = cf->log;
119 if (ngx_ssl_create(s,
120 NGX_SSL_TLSv1 | NGX_SSL_TLSv1_1 | NGX_SSL_TLSv1_2
121 | NGX_SSL_TLSv1_3, NULL) != NGX_OK)
122 {
123 return NGX_ERROR;
124 }
125 conf->upstream_ssl_ctx = s; /* shared across https backends */
126 }
127#endif
128
129 for (a = 0; a < u.naddrs; a++) {
130 be = ngx_array_push(conf->upstream_backends);
131 if (be == NULL) return NGX_ERROR;
132 ngx_memzero(be, sizeof(*be));
133
134 be->resolved.sockaddr = u.addrs[a].sockaddr;
135 be->resolved.socklen = u.addrs[a].socklen;
136 be->resolved.naddrs = 1;
137 be->resolved.host = u.host;
138 be->resolved.port = u.port;
139 be->host = host;
140 be->url_base = url_base;
141 be->ssl = ssl;
142#if (NGX_HTTP_SSL)
143 be->ssl_ctx = ssl ? conf->upstream_ssl_ctx : NULL;
144#endif
145 }
146
147 return NGX_OK;
148}

150ngx_int_t
151webdav_proxy_build_backends(ngx_conf_t *cf,
152 ngx_http_brix_webdav_loc_conf_t *conf)
153{
154 brix_webdav_backend_t *be0;
155
156 conf->upstream_backends = ngx_array_create(cf->pool, 4,
157 sizeof(brix_webdav_backend_t));
158 if (conf->upstream_backends == NULL) {
159 return NGX_ERROR;
160 }
161
162 if (conf->upstream_urls != NULL && conf->upstream_urls->nelts > 0) {
163 ngx_str_t *urls = conf->upstream_urls->elts;
164 ngx_uint_t i;
165 for (i = 0; i < conf->upstream_urls->nelts; i++) {
166 if (webdav_proxy_add_url(cf, conf, &urls[i]) != NGX_OK) {
167 return NGX_ERROR;
168 }
169 }
170 } else if (conf->upstream_url.len > 0) {
171 if (webdav_proxy_add_url(cf, conf, &conf->upstream_url) != NGX_OK) {
172 return NGX_ERROR;
173 }
174 } else {
175 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
176 "brix_webdav_proxy: missing upstream URL");
177 return NGX_ERROR;
178 }
179
180 if (conf->upstream_backends->nelts == 0) {
181 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
182 "brix_webdav_proxy: no usable backends");
183 return NGX_ERROR;
184 }
185
186 /* upstream_conf defaults (shared by all backends). */
187 if (conf->upstream_conf.connect_timeout == 0) {
188 conf->upstream_conf.connect_timeout = 60000;
189 }
190 if (conf->upstream_conf.send_timeout == 0) {
191 conf->upstream_conf.send_timeout = 60000;
192 }
193 if (conf->upstream_conf.read_timeout == 0) {
194 conf->upstream_conf.read_timeout = 60000;
195 }
196 if (conf->upstream_conf.buffer_size == 0) {
197 conf->upstream_conf.buffer_size = (size_t) ngx_pagesize;
198 }
199 conf->upstream_conf.buffering = 0;
200 conf->upstream_conf.bufs.num = 8;
201 conf->upstream_conf.bufs.size = (size_t) ngx_pagesize;
202 conf->upstream_conf.busy_buffers_size = 2 * ngx_pagesize;
203 conf->upstream_conf.max_temp_file_size = 0;
204 conf->upstream_conf.temp_file_write_size = 0;
205#if (NGX_HTTP_SSL)
206 if (conf->upstream_ssl_ctx != NULL) {
207 conf->upstream_conf.ssl = conf->upstream_ssl_ctx;
208 conf->upstream_conf.ssl_verify = 0; /* internal backend; no CA */
209 }
210#endif
211
212 /* Legacy aliases point at the first backend. */
213 be0 = (brix_webdav_backend_t *) conf->upstream_backends->elts;
214 conf->upstream_host = be0->host;
215 conf->upstream_url_base = be0->url_base;
216 conf->upstream_ssl = be0->ssl;
217 conf->upstream_resolved = &be0->resolved;
218
219 ngx_log_error(NGX_LOG_NOTICE, cf->log, 0,
220 "brix_webdav_proxy: %ui backend(s); primary %V (ssl=%d)",
221 conf->upstream_backends->nelts,
222 &conf->upstream_url_base, (int) conf->upstream_ssl);
223
224 return NGX_OK;
225}

235/*
236 * Phase 23 — configure proxy mode for the dynamic SHM pool. Unlike
237 * build_backends, no static URL is parsed here: the pool lives in shared
238 * memory and is populated at runtime via the admin REST API (it may
239 * legitimately start empty). We still must (1) create the SHM zone, (2)
240 * establish upstream_conf defaults shared by every backend, (3) create a TLS
241 * context up front because runtime-added backends may be https, and (4) build
242 * the hide-headers hash so nginx's upstream header processing has a non-empty
243 * hash to probe (an empty hash divides by zero on the first response).
244 */
245ngx_int_t
246webdav_proxy_pool_setup(ngx_conf_t *cf,
247 ngx_http_brix_webdav_loc_conf_t *conf,
248 ngx_http_brix_webdav_loc_conf_t *prev)
249{
250 static ngx_str_t webdav_proxy_hide_headers[] = { ngx_null_string };
251 ngx_hash_init_t hh;
252
253 if (brix_proxy_pool_configure(cf) != NGX_OK) {
254 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
255 "brix_webdav_proxy_dynamic: shared zone setup"
256 " failed");
257 return NGX_ERROR;
258 }
259
260 /* upstream_conf defaults (shared by all backends). */
261 if (conf->upstream_conf.connect_timeout == 0) {
262 conf->upstream_conf.connect_timeout = 60000;
263 }
264 if (conf->upstream_conf.send_timeout == 0) {
265 conf->upstream_conf.send_timeout = 60000;
266 }
267 if (conf->upstream_conf.read_timeout == 0) {
268 conf->upstream_conf.read_timeout = 60000;
269 }
270 if (conf->upstream_conf.buffer_size == 0) {
271 conf->upstream_conf.buffer_size = (size_t) ngx_pagesize;
272 }
273 conf->upstream_conf.buffering = 0;
274 conf->upstream_conf.bufs.num = 8;
275 conf->upstream_conf.bufs.size = (size_t) ngx_pagesize;
276 conf->upstream_conf.busy_buffers_size = 2 * ngx_pagesize;
277 conf->upstream_conf.max_temp_file_size = 0;
278 conf->upstream_conf.temp_file_write_size = 0;
279
280#if (NGX_HTTP_SSL)
281 /* Runtime-added backends may be https — create the shared TLS context now. */
282 if (conf->upstream_ssl_ctx == NULL) {
283 ngx_ssl_t *s = ngx_pcalloc(cf->pool, sizeof(ngx_ssl_t));
284 if (s == NULL) return NGX_ERROR;
285 s->log = cf->log;
286 if (ngx_ssl_create(s,
287 NGX_SSL_TLSv1 | NGX_SSL_TLSv1_1 | NGX_SSL_TLSv1_2
288 | NGX_SSL_TLSv1_3, NULL) != NGX_OK)
289 {
290 return NGX_ERROR;
291 }
292 conf->upstream_ssl_ctx = s;
293 }
294 conf->upstream_conf.ssl = conf->upstream_ssl_ctx;
295 conf->upstream_conf.ssl_verify = 0; /* internal backend; no CA */
296#endif
297
298 hh.max_size = 512;
299 hh.bucket_size = ngx_align(64, ngx_cacheline_size);
300 hh.name = "webdav_proxy_hide_headers_hash";
301 hh.pool = cf->pool;
302 hh.temp_pool = NULL;
303 if (ngx_http_upstream_hide_headers_hash(cf, &conf->upstream_conf,
304 &prev->upstream_conf, webdav_proxy_hide_headers, &hh) != NGX_OK)
305 {
306 return NGX_ERROR;
307 }
308
309 ngx_log_error(NGX_LOG_NOTICE, cf->log, 0,
310 "brix_webdav_proxy_dynamic: SHM pool configured"
311 " (backends added at runtime via admin API)");
312 return NGX_OK;
313}
