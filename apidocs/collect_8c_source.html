70/*
71 *
72 * WHAT: brix_fqan_to_vo() — extracts the VO name from a Fully-Qualified Attribute
73 * Name (FQAN). FQAN format: "/VO/Role=X/Capability=Y" — the VO is the first path
74 * component after the leading slash. Returns ngx_flag_t: 1 on success, 0 when fqan
75 * is NULL, malformed (no second '/'), or VO name exceeds vo_sz. Uses pointer arithmetic
76 * (end - start) for length calculation to avoid strlen on the full string, then
77 * ngx_memcpy + manual NUL termination. Caller must provide vo_sz >= 128 bytes
78 * (VO names are typically short: "cms", "atlas", "alice"). */

80static ngx_flag_t
81brix_fqan_to_vo(const char *fqan, char *vo, size_t vo_sz)
82{
83 const char *start;
84 const char *end;
85 size_t len;
86
87 if (fqan == NULL || fqan[0] != '/') {
88 return 0;
89 }
90
91 start = fqan + 1;
92 end = strchr(start, '/');
93 if (end == NULL || end == start) {
94 return 0;
95 }
96
97 len = (size_t) (end - start);
98 if (len + 1 > vo_sz || !brix_vo_token_safe(start, len)) {
99 return 0;
100 }
101
102 ngx_memcpy(vo, start, len);
103 vo[len] = '\0';
104 return 1;
105}
