1/*
2 * http_mirror_request.c — shadow HTTP(S) request/response construction for the
3 * Phase 24 HTTP/WebDAV traffic mirror (see http_mirror.h, http_mirror_internal.h).
4 *
5 * WHAT: Owns the nginx upstream callbacks that turn a mirror subrequest into a
6 * wire request to a shadow target and parse (then discard) the shadow response:
7 * the Destination rewrite + PUT-body clone helpers, the two-pass header build
8 * (measure then write) driven by a file-local plan struct, and the
9 * create/reinit/process/abort/finalize callback set.
10 * WHY: split out of http_mirror.c (phase-79 file-size cap) so the "build the
11 * shadow request" concern is one focused file; the subrequest driver in
12 * http_mirror.c wires the five callbacks declared in http_mirror_internal.h.
13 * HOW: brix_http_mirror_proxy() (http_mirror.c) assigns mirror_create_request /
14 * mirror_reinit_request / mirror_process_status_line / mirror_abort_request /
15 * mirror_finalize_request onto r->upstream; nginx invokes them across the
16 * connect→send→read→finalize lifecycle of each background shadow subrequest.
17 */

23/*
24 * Rewrite a WebDAV Destination header URL so its host points at the shadow.
25 * "https://public.example.org/dav/foo" + url_base "http://shadow:8443"
26 * => "http://shadow:8443/dav/foo". Mirrors webdav_proxy_rewrite_destination()
27 * (which is static to the proxy module); a relative/odd value is passed through.
28 */
29static ngx_str_t
30brix_http_mirror_rewrite_dest(ngx_pool_t *pool, ngx_str_t *dest,
31 ngx_str_t *url_base)
32{
33 ngx_str_t result = *dest;
34 u_char *p, *end, *path_start;
35 size_t path_len;
36
37 if (dest->len < 7
38 || (ngx_strncasecmp(dest->data, (u_char *) "http://", 7) != 0
39 && ngx_strncasecmp(dest->data, (u_char *) "https://", 8) != 0))
40 {
41 return result; /* not absolute — forward unchanged */
42 }
43 p = ngx_strlchr(dest->data, dest->data + dest->len, '/');
44 if (p == NULL) { return result; }
45 p++; /* skip first '/' of "://" */
46 p = ngx_strlchr(p, dest->data + dest->len, '/');
47 if (p == NULL) { return *url_base; } /* scheme://host only */
48
49 path_start = p;
50 end = dest->data + dest->len;
51 path_len = (size_t) (end - path_start);
52
53 result.len = url_base->len + path_len;
54 result.data = ngx_pnalloc(pool, result.len);
55 if (result.data == NULL) { return *dest; }
56 p = ngx_copy(result.data, url_base->data, url_base->len);
57 (void) ngx_copy(p, path_start, path_len);
58 return result;
59}

61/*
62 * Clone a request-body buf chain into r->pool with INDEPENDENT buf structs so the
63 * shadow send (which advances buf->pos / file_pos) never disturbs the primary's
64 * own consumption of the same body. The underlying memory / file handle is shared
65 * read-only; only the small ngx_buf_t bookkeeping is duplicated.
66 */
67static ngx_chain_t *
68brix_http_mirror_clone_body(ngx_http_request_t *r,
69 ngx_http_request_t *parent, off_t *len_out)
70{
71 ngx_chain_t *in, *head = NULL, **tail = &head;
72 off_t total = 0;
73
74 *len_out = 0;
75 if (parent->request_body == NULL || parent->request_body->bufs == NULL) {
76 return NULL;
77 }
78 for (in = parent->request_body->bufs; in != NULL; in = in->next) {
79 ngx_buf_t *b;
80 ngx_chain_t *cl;
81
82 if (ngx_buf_size(in->buf) == 0) { continue; }
83 cl = ngx_alloc_chain_link(r->pool);
84 b = ngx_palloc(r->pool, sizeof(ngx_buf_t));
85 if (cl == NULL || b == NULL) { *len_out = -1; return NULL; }
86 *b = *in->buf; /* independent pos/last/file_pos/file_last */
87 b->last_buf = b->last_in_chain = 0;
88 cl->buf = b;
89 cl->next = NULL;
90 *tail = cl;
91 tail = &cl->next;
92 total += ngx_buf_size(in->buf);
93 }
94 *len_out = total;
95 return head;
96}

101/*
102 * Derived per-subrequest inputs for building the shadow request line + headers.
103 *
104 * WHAT: A file-local plan struct that captures everything the header writer needs
105 * — target host, auth policy, rewritten Destination, forwarded Depth/Overwrite,
106 * and the cloned PUT body + its Content-Length rendering.
107 * WHY: keeps mirror_create_request a short orchestrator and lets the length pass
108 * and the write pass share exactly the same inputs (no drift between the two).
109 * HOW: filled once by mirror_build_subrequest(), then consumed read-only by
110 * mirror_headers_len() and mirror_copy_headers(); the body chain is attached by
111 * mirror_set_body(). All fields zero-initialised at declaration.
112 */
113typedef struct {
114 brix_mirror_target_t *t; /* selected shadow target */
115 ngx_str_t host; /* Host: header value */
116 int inject_token; /* send "Authorization: Bearer <tok>" */
117 ngx_uint_t has_body; /* method carries a body (PUT) */
118 ngx_table_elt_t *depth_h; /* forwarded Depth (MOVE/COPY) */
119 ngx_table_elt_t *over_h; /* forwarded Overwrite (MOVE/COPY) */
120 ngx_str_t dest; /* rewritten Destination (MOVE/COPY) */
121 ngx_chain_t *body; /* cloned PUT body chain */
122 u_char clbuf[NGX_OFF_T_LEN]; /* rendered Content-Length */
123 size_t cl_digits; /* digits written into clbuf */
124} mirror_req_plan_t;
125
126/*
127 * WHAT: resolve the shadow target and gather every header input into *plan.
128 * WHY: isolates the "figure out what to send" decisions (target lookup, auth
129 * policy, Destination rewrite, Depth/Overwrite forwarding, PUT body clone) from
130 * the byte-emitting passes, so the two passes cannot disagree.
131 * HOW: returns NGX_OK on success, NGX_ERROR if the target is out of range or a
132 * PUT body clone allocation failed (never send a truncated PUT). *plan is
133 * zero-initialised by the caller.
134 */
135static ngx_int_t
136mirror_build_subrequest(ngx_http_request_t *r,
137 ngx_http_brix_webdav_loc_conf_t *conf,
138 ngx_http_brix_webdav_req_ctx_t *ctx, mirror_req_plan_t *plan)
139{
140 ngx_uint_t method;
141 off_t body_len = 0;
142
143 if (ctx == NULL || conf->mirror.targets == NULL
144 || ctx->mirror_target_idx >= conf->mirror.targets->nelts)
145 {
146 return NGX_ERROR;
147 }
148 plan->t = (brix_mirror_target_t *) conf->mirror.targets->elts
149 + ctx->mirror_target_idx;
150 plan->host = plan->t->host;
151 method = r->method;
152 plan->has_body = brix_http_mirror_method_has_body(method);
153
154 /* Auth policy on the shadow request:
155 * token configured => inject "Authorization: Bearer <token>"
156 * strip_auth on => send no Authorization at all (default)
157 * strip_auth off => forward the client's Authorization header. */
158 plan->inject_token = (conf->mirror.token.len > 0);
159
160 /* MOVE/COPY: rewrite the client's Destination to point at the shadow, and
161 * forward Depth/Overwrite verbatim so the shadow performs the same op. */
162 if (method == NGX_HTTP_MOVE || method == NGX_HTTP_COPY) {
163 ngx_table_elt_t *dest_h = brix_http_find_header(r,
164 "Destination", sizeof("Destination") - 1);
165 if (dest_h != NULL) {
166 plan->dest = brix_http_mirror_rewrite_dest(r->pool, &dest_h->value,
167 &plan->t->url_base);
168 }
169 plan->depth_h = brix_http_find_header(r, "Depth",
170 sizeof("Depth") - 1);
171 plan->over_h = brix_http_find_header(r, "Overwrite",
172 sizeof("Overwrite") - 1);
173 }
174
175 /* PUT: forward the (preserved) request body cloned from the parent request. */
176 if (plan->has_body && r->parent != NULL) {
177 plan->body = brix_http_mirror_clone_body(r, r->parent, &body_len);
178 if (body_len < 0) {
179 return NGX_ERROR; /* clone alloc failed — don't send a truncated PUT */
180 }
181 plan->cl_digits =
182 (size_t) (ngx_sprintf(plan->clbuf, "%O", body_len) - plan->clbuf);
183 }
184 return NGX_OK;
185}

187/*
188 * WHAT: compute the exact byte length of the request line + header block.
189 * WHY: mirror_copy_headers() writes into a fixed-size temp buf; this pure sizing
190 * pass must mirror it field-for-field or the write would overrun/underrun.
191 * HOW: sums the request line and every conditionally-present header from *plan;
192 * no side effects.
193 */
194static size_t
195mirror_headers_len(ngx_http_request_t *r, ngx_http_brix_webdav_loc_conf_t *conf,
196 const mirror_req_plan_t *plan)
197{
198 /* "METHOD uri[?args] HTTP/1.1\r\nHost: <host>\r\nConnection: close\r\n" */
199 size_t len = r->method_name.len + 1
200 + r->uri.len
201 + (r->args.len ? 1 + r->args.len : 0)
202 + sizeof(" HTTP/1.1" CRLF) - 1
203 + sizeof("Host: ") - 1 + plan->host.len + sizeof(CRLF) - 1
204 + sizeof("Connection: close" CRLF) - 1
205 + sizeof("X-Xrootd-Mirror: 1" CRLF) - 1
206 + sizeof(CRLF) - 1;
207
208 if (plan->inject_token) {
209 len += sizeof("Authorization: Bearer ") - 1
210 + conf->mirror.token.len + sizeof(CRLF) - 1;
211 } else if (!conf->mirror.strip_auth
212 && r->headers_in.authorization != NULL) {
213 len += sizeof("Authorization: ") - 1
214 + r->headers_in.authorization->value.len + sizeof(CRLF) - 1;
215 }
216 if (plan->dest.len) {
217 len += sizeof("Destination: ") - 1 + plan->dest.len + sizeof(CRLF) - 1;
218 }
219 if (plan->depth_h != NULL) {
220 len += sizeof("Depth: ") - 1 + plan->depth_h->value.len
221 + sizeof(CRLF) - 1;
222 }
223 if (plan->over_h != NULL) {
224 len += sizeof("Overwrite: ") - 1 + plan->over_h->value.len
225 + sizeof(CRLF) - 1;
226 }
227 if (plan->has_body) {
228 len += sizeof("Content-Length: ") - 1 + plan->cl_digits
229 + sizeof(CRLF) - 1;
230 }
231 return len;
232}

234/*
235 * WHAT: write the request line + header block into buffer b per *plan.
236 * WHY: the byte-emitting half of the two-pass build; keeps the exact wire order
237 * (method line, Host, Connection, loop-guard, auth, Destination/Depth/Overwrite,
238 * Content-Length, blank line) in one place — frozen for AF-shadow behaviour.
239 * HOW: appends with ngx_copy from b->last; advances b->last to the end. Emits
240 * only the headers *plan marked present, matching mirror_headers_len() exactly.
241 */
242static void
243mirror_copy_headers(ngx_http_request_t *r,
244 ngx_http_brix_webdav_loc_conf_t *conf, const mirror_req_plan_t *plan,
245 ngx_buf_t *b)
246{
247 u_char *p = b->last;
248
249 p = ngx_copy(p, r->method_name.data, r->method_name.len);
250 *p++ = ' ';
251 p = ngx_copy(p, r->uri.data, r->uri.len);
252 if (r->args.len) {
253 *p++ = '?';
254 p = ngx_copy(p, r->args.data, r->args.len);
255 }
256 p = ngx_copy(p, " HTTP/1.1" CRLF, sizeof(" HTTP/1.1" CRLF) - 1);
257
258 p = ngx_copy(p, "Host: ", sizeof("Host: ") - 1);
259 p = ngx_copy(p, plan->host.data, plan->host.len);
260 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
261
262 p = ngx_copy(p, "Connection: close" CRLF,
263 sizeof("Connection: close" CRLF) - 1);
264
265 /* Loop guard: a shadow that happens to be this same server will see this
266 * marker and decline to mirror again (brix_http_mirror_precontent_handler). */
267 p = ngx_copy(p, "X-Xrootd-Mirror: 1" CRLF,
268 sizeof("X-Xrootd-Mirror: 1" CRLF) - 1);
269
270 if (plan->inject_token) {
271 p = ngx_copy(p, "Authorization: Bearer ",
272 sizeof("Authorization: Bearer ") - 1);
273 p = ngx_copy(p, conf->mirror.token.data, conf->mirror.token.len);
274 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
275 } else if (!conf->mirror.strip_auth
276 && r->headers_in.authorization != NULL) {
277 p = ngx_copy(p, "Authorization: ", sizeof("Authorization: ") - 1);
278 p = ngx_copy(p, r->headers_in.authorization->value.data,
279 r->headers_in.authorization->value.len);
280 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
281 }
282 if (plan->dest.len) {
283 p = ngx_copy(p, "Destination: ", sizeof("Destination: ") - 1);
284 p = ngx_copy(p, plan->dest.data, plan->dest.len);
285 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
286 }
287 if (plan->depth_h != NULL) {
288 p = ngx_copy(p, "Depth: ", sizeof("Depth: ") - 1);
289 p = ngx_copy(p, plan->depth_h->value.data, plan->depth_h->value.len);
290 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
291 }
292 if (plan->over_h != NULL) {
293 p = ngx_copy(p, "Overwrite: ", sizeof("Overwrite: ") - 1);
294 p = ngx_copy(p, plan->over_h->value.data, plan->over_h->value.len);
295 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
296 }
297 if (plan->has_body) {
298 p = ngx_copy(p, "Content-Length: ", sizeof("Content-Length: ") - 1);
299 p = ngx_copy(p, plan->clbuf, plan->cl_digits);
300 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
301 }
302
303 p = ngx_copy(p, CRLF, sizeof(CRLF) - 1);
304 b->last = p;
305}

307/*
308 * WHAT: chain the header buffer and (for PUT) the cloned body into request_bufs.
309 * WHY: assembles the final send chain — header block first, cloned PUT body next
310 * — and hands it to the upstream, keeping that wiring out of the orchestrator.
311 * HOW: allocates one chain link for b, points its next at the cloned body chain
312 * (NULL for bodyless methods), and stores it in r->upstream->request_bufs.
313 * Returns NGX_OK, or NGX_ERROR on chain-link allocation failure.
314 */
315static ngx_int_t
316mirror_set_body(ngx_http_request_t *r, const mirror_req_plan_t *plan,
317 ngx_buf_t *b)
318{
319 ngx_chain_t *cl = ngx_alloc_chain_link(r->pool);
320
321 if (cl == NULL) {
322 return NGX_ERROR;
323 }
324 cl->buf = b;
325 cl->next = plan->body; /* PUT body (cloned) follows the header block */
326 r->upstream->request_bufs = cl;
327 return NGX_OK;
328}

330/*
331 * WHAT: nginx upstream create_request callback — build the shadow HTTP request.
332 * WHY: orchestrates the frozen two-pass build (measure, then write) plus body
333 * attachment as a short linear sequence; the logic lives in the named helpers.
334 * HOW: gather inputs into a plan, size + allocate the header buf, write it, then
335 * chain the body; any step failure returns NGX_ERROR.
336 */
337ngx_int_t
338mirror_create_request(ngx_http_request_t *r)
339{
340 ngx_http_brix_webdav_loc_conf_t *conf;
341 ngx_http_brix_webdav_req_ctx_t *ctx;
342 mirror_req_plan_t plan;
343 ngx_buf_t *b;
344 size_t len;
345
346 ngx_memzero(&plan, sizeof(plan));
347
348 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
349 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
350
351 if (mirror_build_subrequest(r, conf, ctx, &plan) != NGX_OK) {
352 return NGX_ERROR;
353 }
354
355 len = mirror_headers_len(r, conf, &plan);
356 b = ngx_create_temp_buf(r->pool, len);
357 if (b == NULL) {
358 return NGX_ERROR;
359 }
360
361 mirror_copy_headers(r, conf, &plan, b);
362
363 return mirror_set_body(r, &plan, b);
364}

430void
431mirror_finalize_request(ngx_http_request_t *r, ngx_int_t rc)
432{
433 ngx_http_brix_webdav_loc_conf_t *conf;
434 ngx_http_request_t *parent = r->parent;
435 ngx_uint_t shadow_status;
436
437 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
438
439 shadow_status = (r->upstream != NULL) ? r->upstream->headers_in.status_n : 0;
440
441 if (shadow_status != 0) {
442 MIR_HTTP_INC(mirror_http_total);
443 } else {
444 MIR_HTTP_INC(mirror_http_errors_total);
445 ngx_log_debug2(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
446 "xrootd mirror: shadow request failed (rc=%i) uri=%V",
447 rc, &r->uri);
448 }
449
450 if (parent != NULL && shadow_status != 0) {
451 ngx_http_brix_webdav_req_ctx_t *pctx =
452 ngx_http_get_module_ctx(parent, ngx_http_brix_webdav_module);
453
454 if (pctx != NULL && pctx->primary_status != 0
455 && brix_mirror_status_class(shadow_status)
456 != brix_mirror_status_class(pctx->primary_status))
457 {
458 MIR_HTTP_INC(mirror_http_divergence_total);
459 if (conf->mirror.log_diverge) {
460 ngx_log_error(NGX_LOG_NOTICE, r->connection->log, 0,
461 "xrootd mirror divergence: primary=%ui shadow=%ui uri=%V",
462 pctx->primary_status, shadow_status, &parent->uri);
463 }
464 }
465 }
