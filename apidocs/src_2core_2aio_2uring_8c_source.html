10/* File: src/aio/uring.c — optional io_uring disk-I/O backend (Phase 44)
11 * WHAT: Implements the runtime capability probe, the §32 startup fail-fast
12 * validator, and the per-worker ring lifecycle for the optional io_uring
13 * disk-I/O backend. Submission and completion live in uring_submit.c
14 * (SB-W3/W4); this TU owns detection, gating, and the ring singleton.
15 *
16 * WHY: The backend must be invisible unless explicitly built (pkg-config
17 * liburing) and runtime-available (opcode probe). `auto`/`off` always
18 * start and silently fall back; `on` is a hard requirement that fails
19 * startup when it cannot be provided — caught at config time so even a
20 * stub build flags it under `nginx -t` with no kernel needed.
21 *
22 * HOW: All liburing-specific code is under #if (BRIX_HAVE_LIBURING). When
23 * the macro is undefined the file compiles to inert stubs that report the
24 * backend as unavailable and hard-fail any `brix_io_uring on` block. */

70/*
71 * brix_uring_runtime_available — authoritative, memoized opcode probe.
72 *
73 * Stands up a throwaway 8-entry ring, asks io_uring_get_probe whether every
74 * required server data opcode (READ/WRITE/READV/WRITEV/FSYNC) is supported,
75 * tears it down, and caches the verdict for the lifetime of the process. Never
76 * parses a kernel version: containers/seccomp routinely block io_uring_setup
77 * even on new kernels, so only an actual setup+probe is trustworthy.
78 *
79 * Returns 1 iff a ring can be created and all required opcodes are supported.
80 */
81ngx_int_t
82brix_uring_runtime_available(void)
83{
84 static const int required[] = {
85 IORING_OP_READ, IORING_OP_WRITE, IORING_OP_READV,
86 IORING_OP_WRITEV, IORING_OP_FSYNC
87 };
88 static ngx_int_t cached = -1; /* -1 = not yet probed */
89 struct io_uring ring;
90 struct io_uring_probe *probe;
91 ngx_int_t ok = 1;
92 ngx_uint_t i;
93
94 if (cached != -1) {
95 return cached;
96 }
97
98 if (io_uring_queue_init(8, &ring, 0) < 0) {
99 cached = 0; /* seccomp / no syscall / old kernel */
100 return 0;
101 }
102
103 probe = io_uring_get_probe_ring(&ring);
104 if (probe == NULL) {
105 io_uring_queue_exit(&ring);
106 cached = 0;
107 return 0;
108 }
109
110 for (i = 0; i < sizeof(required) / sizeof(required[0]); i++) {
111 if (!io_uring_opcode_supported(probe, required[i])) {
112 ok = 0;
113 break;
114 }
115 }
116
117 io_uring_free_probe(probe);
118 io_uring_queue_exit(&ring);
119
120 cached = ok;
121 return cached;
122}

146/* brix_uring_slot_acquire — claim the first free slot. The selector checks
147 * inflight < queue_depth before calling, so a free slot always exists; the
148 * linear scan over <=4096 entries is negligible. Returns NULL only if the
149 * table is unexpectedly full (caller then falls back to the thread pool).
150 * Non-static: also called from uring_submit.c. */
151brix_uring_slot_t *
152brix_uring_slot_acquire(brix_uring_t *u, uint32_t *idx_out)
153{
154 uint32_t i;
155
156 for (i = 0; i < u->queue_depth; i++) {
157 if (!u->slots[i].in_use) {
158 u->slots[i].in_use = 1;
159 *idx_out = i;
160 return &u->slots[i];
161 }
162 }
163
164 return NULL;
165}

167/* brix_uring_slot_release — free a slot and bump its generation so any later

182/*
183 * brix_uring_apply_cqe — translate cqe->res into the task's OUT fields, the
184 * same fields the worker-thread fn would have set. io_uring reports failures
185 * as a negative -errno in cqe->res (it does NOT set errno).
186 *
187 * Returns 1 when the task is fully complete and its done-callback should be
188 * posted, 0 when more CQEs are still expected for this task (the linked
189 * writev+fsync chain, added in SB-W4). READV/WRITEV cases also land in SB-W4;
190 * READ/WRITE are handled here so the reaper is complete for the hot path.
191 */
192static ngx_int_t
193brix_uring_apply_cqe(brix_uring_slot_t *slot, struct io_uring_cqe *cqe)
194{
195 ngx_thread_task_t *task = slot->task;
196 int32_t res = cqe->res;
197
198 switch (slot->op_kind) {
199
200 case XRD_URING_OP_READ: {
201 brix_read_aio_t *t = task->ctx;
202 if (res < 0) { t->nread = -1; t->io_errno = -res; }
203 else { t->nread = res; t->io_errno = 0; }
204 return 1;
205 }
206
207 case XRD_URING_OP_WRITE: {
208 brix_write_aio_t *t = task->ctx;
209 if (res < 0) { t->nwritten = -1; t->io_errno = -res; }
210 else { t->nwritten = res; t->io_errno = 0; }
211 return 1;
212 }
213
214 case XRD_URING_OP_READV: {
215 brix_readv_aio_t *t = task->ctx;
216 size_t total = 0, i;
217
218 /* Single contiguous group (op_for gated it): one IORING_OP_READV
219 * scattered into the per-segment payload pointers. The wire segment
220 * headers were already written at plan-build, so nothing to fix up here
221 * — only the totals + error state, matching brix_readv_aio_thread. */
222 for (i = 0; i < t->segment_count; i++) {
223 total += t->segments[i].read_length;
224 }
225 if (res < 0) {
226 t->io_error = 1; t->bytes_read_total = 0; t->response_bytes = 0;
227 snprintf(t->err_msg, sizeof(t->err_msg),
228 "readv I/O error: %s", strerror(-res));
229 } else if ((size_t) res != total) {
230 t->io_error = 1; t->bytes_read_total = 0; t->response_bytes = 0;
231 snprintf(t->err_msg, sizeof(t->err_msg), "readv past EOF");
232 } else {
233 t->io_error = 0;
234 t->bytes_read_total = (size_t) res;
235 t->response_bytes = t->segment_count * BRIX_READV_SEGSIZE
236 + (size_t) res;
237 }
238 return 1;
239 }
240
241 case XRD_URING_OP_WRITEV: {
242 brix_writev_aio_t *t = task->ctx;
243 size_t total = 0, i;
244
245 /* Single contiguous same-fd group: one IORING_OP_WRITEV gathering the
246 * segment buffers. io_error encodes the failure kind exactly as
247 * brix_writev_write_aio_thread (1 = hard error, 2 = short write). A
248 * trailing FSYNC (when do_sync) is a separate linked SQE whose CQE the
249 * reaper drops — best-effort, matching the pool path's ignored return. */
250 for (i = 0; i < t->n_segs; i++) {
251 total += t->segs[i].wlen;
252 }
253 if (res < 0) {
254 t->io_error = 1; t->bytes_total = 0;
255 snprintf(t->err_msg, sizeof(t->err_msg),
256 "writev I/O error: %s", strerror(-res));
257 } else if ((size_t) res < total) {
258 t->io_error = 2; t->bytes_total = (size_t) res;
259 snprintf(t->err_msg, sizeof(t->err_msg), "writev short write");
260 } else {
261 t->io_error = 0; t->bytes_total = (size_t) res;
262 }
263 return 1;
264 }
265
266 default:
267 /* pgread/dirlist are never routed to io_uring (op_for -> NONE). */
268 return 1;
269 }
270}

272/*
273 * brix_uring_eventfd_handler — the completion reaper.
274 *
275 * Wired into the worker's epoll as the read handler of the fake connection that
276 * wraps the ring's registered eventfd. This is the public-API analogue of
277 * nginx core's ngx_epoll_eventfd_handler() (the libaio bridge): drain the
278 * eventfd counter, then harvest every ready CQE — validating each slot's
279 * generation, translating the result into the task, and posting the task's
280 * done-callback to ngx_posted_events. Done-callbacks are NEVER invoked inline
281 * here: nginx's posted-events drain runs them after process_events, so a
282 * callback that re-submits the next read window never re-enters the harvest
283 * loop (re-entrancy-safe).
284 */
285static void
286brix_uring_eventfd_handler(ngx_event_t *ev)
287{
288 ngx_connection_t *evc = ev->data;
289 brix_uring_t *u = evc->data;
290 struct io_uring_cqe *cqe;
291 uint64_t counter;
292 ssize_t n;
293
294 /* 1. drain the eventfd counter (mirrors core's first step). */
295 n = read(u->eventfd, &counter, sizeof(counter));
296 if (n != (ssize_t) sizeof(counter)) {
297 if (n == -1 && (ngx_errno == NGX_EAGAIN || ngx_errno == NGX_EINTR)) {
298 return;
299 }
300 ngx_log_error(NGX_LOG_ALERT, evc->log, ngx_errno,
301 "brix: io_uring eventfd read() failed");
302 return;
303 }
304
305 /* 2. harvest all ready completions. */
306 while (io_uring_peek_cqe(&u->ring, &cqe) == 0) {
307 uint64_t ud = io_uring_cqe_get_data64(cqe);
308 uint32_t idx;
309 uint32_t gen;
310 brix_uring_slot_t *slot;
311
312 /* 2-pre. trailing FSYNC of a linked writev+fsync chain: carries no slot;
313 * fsync is best-effort (the pool path ignores its return), so drop the
314 * CQE and just balance inflight. */
315 if (ud == BRIX_URING_FSYNC_COOKIE) {
316 io_uring_cqe_seen(&u->ring, cqe);
317 if (u->inflight > 0) { u->inflight--; }
318 continue;
319 }
320
321 idx = (uint32_t) (ud & 0xffffffffULL);
322 gen = (uint32_t) (ud >> 32);
323 slot = brix_uring_slot_at(u, idx);
324
325 /* 2a. generation guard: a stale CQE for a recycled/dead slot is dropped
326 * safely. (The done-callback's own ctx->destroyed check remains the
327 * authoritative guard for the connection.) */
328 if (slot == NULL || !slot->in_use || slot->generation != gen) {
329 io_uring_cqe_seen(&u->ring, cqe);
330 if (u->inflight > 0) { u->inflight--; }
331 continue;
332 }
333
334 /* 2b. translate + post (the done-callback is task->event.handler, set
335 * at brix_task_bind time). */
336 if (brix_uring_apply_cqe(slot, cqe)) {
337 ngx_thread_task_t *task = slot->task;
338
339 task->event.complete = 1;
340 ngx_post_event(&task->event, &ngx_posted_events);
341 brix_uring_slot_release(u, idx);
342 }
343
344 io_uring_cqe_seen(&u->ring, cqe);
345 if (u->inflight > 0) { u->inflight--; }
346 }
347}

351/*
352 * brix_uring_apply_restrictions — lock the ring to the fd-only data opcodes
353 * the backend actually submits (NOP for the self-test + READ/WRITE/READV/
354 * WRITEV/FSYNC). With no OPENAT/STATX/UNLINKAT allowed, the ring can neither
355 * open nor traverse a path — which is what makes the unprivileged-worker
356 * containment sound. Requires the ring to have been created R_DISABLED;
357 * caller enables it afterwards. Returns NGX_OK iff restrictions were applied.
358 */
359static ngx_int_t
360brix_uring_apply_restrictions(struct io_uring *ring)
361{
362 struct io_uring_restriction res[6];
363
364 ngx_memzero(res, sizeof(res));
365 res[0].opcode = IORING_RESTRICTION_SQE_OP; res[0].sqe_op = IORING_OP_NOP;
366 res[1].opcode = IORING_RESTRICTION_SQE_OP; res[1].sqe_op = IORING_OP_READ;
367 res[2].opcode = IORING_RESTRICTION_SQE_OP; res[2].sqe_op = IORING_OP_WRITE;
368 res[3].opcode = IORING_RESTRICTION_SQE_OP; res[3].sqe_op = IORING_OP_READV;
369 res[4].opcode = IORING_RESTRICTION_SQE_OP; res[4].sqe_op = IORING_OP_WRITEV;
370 res[5].opcode = IORING_RESTRICTION_SQE_OP; res[5].sqe_op = IORING_OP_FSYNC;
371
372 return io_uring_register_restrictions(ring, res, 6) == 0
373 ? NGX_OK : NGX_ERROR;
374}
375#endif

377/*
378 * brix_uring_teardown — release every ring resource that has been brought up,
379 * in reverse order, idempotently (driven by which fields are set). Mirrors
380 * nginx core's eventfd cleanup: ngx_free_connection + close(fd) rather than
381 * ngx_close_connection, to avoid double-closing the eventfd. No goto.
382 */
383static void
384brix_uring_teardown(brix_uring_t *u)
385{
386 if (u->ring_active && u->eventfd >= 0) {
387 io_uring_unregister_eventfd(&u->ring);
388 }
389
390 if (u->evc != NULL) {
391 if (u->evc->read->active) {
392 (void) ngx_del_event(u->evc->read, NGX_READ_EVENT, 0);
393 }
394 ngx_free_connection(u->evc);
395 u->evc->fd = (ngx_socket_t) -1;
396 u->evc = NULL;
397 }
398
399 if (u->eventfd >= 0) {
400 close(u->eventfd);
401 u->eventfd = -1;
402 }
403
404 if (u->ring_active) {
405 io_uring_queue_exit(&u->ring);
406 u->ring_active = 0;
407 }
408
409 u->enabled = 0;
410 u->slots = NULL; /* pool-owned; freed with the worker cycle pool */
411}

413/*
414 * brix_uring_init_fail — log the bring-up failure, tear down whatever was set
415 * up, and return the right verdict: NGX_ERROR under `on` (the worker refuses to
416 * run on the thread pool — master respawns; the §32.7 backstop) or NGX_OK under
417 * `auto` (silent degrade to the thread pool).

419static ngx_int_t
420brix_uring_init_fail(brix_uring_t *u, ngx_cycle_t *cycle, ngx_uint_t mode_on,
421 const char *what)
422{
423 /* Hoist the level into a variable: the ngx_log_error macro does not
424 * parenthesize its level argument, so a raw ternary would bind under >=. */
425 ngx_uint_t level = mode_on ? NGX_LOG_EMERG : NGX_LOG_NOTICE;
426
427 ngx_log_error(level, cycle->log, ngx_errno,

437/*
438 * brix_uring_init_worker — create this worker's ring after fork.
439 *
440 * Scans every enabled server block: the ring is created if any block wants
441 * io_uring (mode on/auto); queue depth is the max requested; restrictions are
442 * applied unless any wanting block turned them off. Bring-up sequence:
443 * queue_init [R_DISABLED if restricting] -> register restrictions + enable ->
444 * NOP self-test -> eventfd + register_eventfd -> fake-connection epoll bridge ->
445 * slot table. Any failure routes through brix_uring_init_fail (NGX_OK under
446 * auto, NGX_ERROR under on). When no block wants io_uring this is a no-op.
447 */
448ngx_int_t
449brix_uring_init_worker(ngx_cycle_t *cycle)
450{
451 ngx_stream_core_main_conf_t *cmcf;
452 ngx_stream_core_srv_conf_t **cscfp;
453 ngx_stream_brix_srv_conf_t *xcf;
454 brix_uring_t *u = &brix_uring_worker_ring;
455 struct io_uring_params params;
456 struct io_uring_sqe *sqe;
457 struct io_uring_cqe *cqe;
458 ngx_connection_t *evc;
459 ngx_uint_t i;
460 ngx_uint_t want = 0, mode_on = 0, want_restrict = 1;
461 ngx_int_t depth = BRIX_IO_URING_QUEUE_DEPTH;
462 ngx_str_t panic_file = ngx_null_string;
463
464 cmcf = ngx_stream_cycle_get_module_main_conf(cycle, ngx_stream_core_module);
465 if (cmcf == NULL) {
466 return NGX_OK;
467 }
468
469 cscfp = cmcf->servers.elts;
470 for (i = 0; i < cmcf->servers.nelts; i++) {
471 xcf = ngx_stream_conf_get_module_srv_conf(cscfp[i],
472 ngx_stream_brix_module);
473 if (!xcf->common.enable || xcf->io_uring == BRIX_IO_URING_OFF) {
474 continue;
475 }
476 want = 1;
477 if (xcf->io_uring == BRIX_IO_URING_ON) {
478 mode_on = 1;
479 }
480 if (xcf->io_uring_queue_depth > depth) {
481 depth = xcf->io_uring_queue_depth;
482 }
483 if (!xcf->io_uring_restrict) {
484 want_restrict = 0;
485 }
486 if (panic_file.len == 0 && xcf->io_uring_panic_file.len > 0) {
487 panic_file = xcf->io_uring_panic_file;
488 }
489 }
490
491 if (!want) {
492 return NGX_OK; /* every enabled block has io_uring off */
493 }
494
495#if !(BRIX_URING_HAVE_RESTRICTIONS)
496 (void) want_restrict; /* no restrictions API in this liburing/kernel */
497#endif
498
499 /* AUTO blocks need the per-process probe (seccomp-accurate); ON blocks
500 * already passed it at config time but re-checking is cheap + memoized. */
501 if (!mode_on && !brix_uring_runtime_available()) {
502 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
503 "brix: io_uring unavailable on this host; using the thread pool");
504 return NGX_OK;
505 }
506
507 if (depth < 8) { depth = 8; }
508 if (depth > 4096) { depth = 4096; }
509
510 u->log = cycle->log;
511 u->eventfd = -1;
512 u->evc = NULL;
513 u->slots = NULL;
514 u->inflight = 0;
515 u->ring_active = 0;
516 u->enabled = 0;
517 u->restrict_ops = 0;
518 u->queue_depth = (uint32_t) depth;
519
520 /* 1. create the ring (R_DISABLED first when we will register restrictions). */
521 ngx_memzero(&params, sizeof(params));
522#if (BRIX_URING_HAVE_RESTRICTIONS)
523 if (want_restrict) {
524 params.flags |= IORING_SETUP_R_DISABLED;
525 }
526#endif
527 if (io_uring_queue_init_params((unsigned) depth, &u->ring, &params) < 0) {
528 return brix_uring_init_fail(u, cycle, mode_on,
529 "io_uring_queue_init_params");
530 }
531 u->ring_active = 1;
532
533 /*
534 * 2. Register the CQE-notifier eventfd BEFORE applying restrictions /
535 * enabling the ring. io_uring_register_eventfd is a *register* op: once the
536 * ring is restricted+enabled, register ops are denied (EINVAL) unless
537 * explicitly allowed, and while the ring is R_DISABLED register ops are
538 * permitted — so this must happen here, not after enable.
539 */
540 u->eventfd = eventfd(0, EFD_NONBLOCK | EFD_CLOEXEC);
541 if (u->eventfd < 0) {
542 return brix_uring_init_fail(u, cycle, mode_on, "eventfd");
543 }
544 if (io_uring_register_eventfd(&u->ring, u->eventfd) < 0) {
545 return brix_uring_init_fail(u, cycle, mode_on,
546 "io_uring_register_eventfd");
547 }
548
549 /* 3. restrictions (best-effort) then enable the ring. */
550#if (BRIX_URING_HAVE_RESTRICTIONS)
551 if (want_restrict) {
552 if (brix_uring_apply_restrictions(&u->ring) == NGX_OK) {
553 u->restrict_ops = 1;
554 } else {
555 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
556 "brix: io_uring_register_restrictions unavailable "
557 "(kernel < 5.10?); ring runs unrestricted — containment still "
558 "holds via the unprivileged worker + confined fd");
559 }
560 if (io_uring_enable_rings(&u->ring) < 0) {
561 return brix_uring_init_fail(u, cycle, mode_on,
562 "io_uring_enable_rings");
563 }
564 }
565#endif
566
567 /* 4. NOP self-test: prove submit -> complete works before any real op.
568 * (Bumps the registered eventfd's counter; the first reaper call after the
569 * loop starts drains it and finds no CQE — harmless.) */
570 sqe = io_uring_get_sqe(&u->ring);
571 if (sqe == NULL) {
572 return brix_uring_init_fail(u, cycle, mode_on, "io_uring_get_sqe(NOP)");
573 }
574 io_uring_prep_nop(sqe);
575 io_uring_sqe_set_data64(sqe, BRIX_URING_NOP_COOKIE);
576 if (io_uring_submit(&u->ring) < 0
577 || io_uring_wait_cqe(&u->ring, &cqe) < 0)
578 {
579 return brix_uring_init_fail(u, cycle, mode_on, "io_uring NOP self-test");
580 }
581 if (io_uring_cqe_get_data64(cqe) != BRIX_URING_NOP_COOKIE
582 || cqe->res != 0)
583 {
584 io_uring_cqe_seen(&u->ring, cqe);
585 return brix_uring_init_fail(u, cycle, mode_on,
586 "io_uring NOP self-test (unexpected CQE)");
587 }
588 io_uring_cqe_seen(&u->ring, cqe);
589
590 /* 5. wire the eventfd into the worker's epoll via a fake connection. */
591 evc = ngx_get_connection(u->eventfd, cycle->log);
592 if (evc == NULL) {
593 return brix_uring_init_fail(u, cycle, mode_on, "ngx_get_connection");
594 }
595 evc->read->handler = brix_uring_eventfd_handler;
596 evc->read->log = cycle->log;
597 evc->data = u;
598 u->evc = evc;
599 if (ngx_add_event(evc->read, NGX_READ_EVENT, 0) != NGX_OK) {
600 return brix_uring_init_fail(u, cycle, mode_on, "ngx_add_event");
601 }
602
603 /* 6. completion-slot table (pool-owned; freed with the cycle). */
604 u->slots = ngx_pcalloc(cycle->pool,
605 u->queue_depth * sizeof(brix_uring_slot_t));
606 if (u->slots == NULL) {
607 return brix_uring_init_fail(u, cycle, mode_on, "slot table alloc");
608 }
609
610 /* SB-W5b: attach the cross-worker kill-switch flag (NULL if the zone was
611 * not registered — the selector then reads "enabled") and arm the
612 * panic-file poll timer if a path was configured. */
613 u->disabled_flag = brix_uring_killswitch_ptr();
614 (void) brix_uring_panicfile_arm(cycle, &panic_file);
615
616 u->enabled = 1;
617 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
618 "brix: io_uring disk-I/O backend active (queue_depth=%ui%s)",
619 (ngx_uint_t) u->queue_depth,
620 u->restrict_ops ? ", restricted" : "");
621
622 return NGX_OK;
623}

667/*
668 * brix_uring_validate_conf — §32 startup fail-fast (ADR-16).
669 *
670 * If any enabled server block requests `brix_io_uring on`, io_uring MUST be
671 * provided or startup fails (NGX_ERROR -> nginx -t exits non-zero, master
672 * refuses to start). Two independent gates:
673 * (1) compile-time: a stub build can never satisfy `on` — caught with no
674 * kernel needed, so CI / `nginx -t` flags it anywhere;
675 * (2) runtime: a liburing build whose probe fails (old kernel / seccomp).
676 * `off` and `auto` always pass (auto degrades silently per worker).
677 *
678 * Returns NGX_OK to allow startup, NGX_ERROR to abort it.
679 */
680ngx_int_t
681brix_uring_validate_conf(ngx_conf_t *cf)
682{
683 if (!brix_uring_any_block_on(cf)) {
684 return NGX_OK;
685 }
686
687#if !(BRIX_HAVE_LIBURING)
688 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
689 "\"brix_io_uring on\" requires a build with liburing, but this binary "
690 "was compiled WITHOUT it. Rebuild with BRIX_ENABLE_IO_URING=1 and "
691 "liburing-devel installed, or set \"brix_io_uring auto\" to allow "
692 "silent fallback to the thread pool.");
693 return NGX_ERROR;
694#else
695 if (!brix_uring_runtime_available()) {
696 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
697 "\"brix_io_uring on\" requested but io_uring is unavailable on "
698 "this host (io_uring_setup/opcode probe failed). This is typically a "
699 "seccomp policy (Docker/containerd default profiles block io_uring) "
700 "or a kernel older than %d.%d. Set \"brix_io_uring auto\" to fall "
701 "back to the thread pool, or enable io_uring at the host/container "
702 "level.",
703 BRIX_IO_URING_MIN_KERNEL_MAJOR, BRIX_IO_URING_MIN_KERNEL_MINOR);
704 return NGX_ERROR;
705 }
706 return NGX_OK;
707#endif
708}
