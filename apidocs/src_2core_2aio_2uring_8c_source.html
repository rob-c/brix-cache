10/* File: src/core/aio/uring.c — the io_uring per-worker ring singleton and the
11 * init/exit lifecycle orchestrator (phase-79 split from the original 1033-line
12 * uring.c).
13 * WHAT: Owns the file-static per-worker ring singleton + its accessor, and the
14 * two lifecycle entry points: brix_uring_init_worker (drive the ordered
15 * bring-up steps then publish the ring) and brix_uring_exit_worker (tear
16 * it down). The ordered register-phase steps live in uring_bringup.c;
17 * detection/gating in uring_probe.c; the CQE reaper + slot table in
18 * uring_reap.c; submission in uring_submit.c.
19 *
20 * WHY: The backend must be invisible unless explicitly built (pkg-config
21 * liburing) and runtime-available (opcode probe). `auto`/`off` always
22 * start and silently fall back; `on` is a hard requirement — this
23 * orchestrator is the §32.7 backstop that refuses to run a worker that
24 * cannot honour `on` (brix_uring_init_fail returns NGX_ERROR under `on`).
25 *
26 * HOW: All liburing-specific code is under #if (BRIX_HAVE_LIBURING). When the
27 * macro is undefined the lifecycle entry points compile to inert stubs. */

42/*
43 * brix_uring_teardown — release every ring resource that has been brought up,
44 * in reverse order, idempotently (driven by which fields are set). Mirrors
45 * nginx core's eventfd cleanup: ngx_free_connection + close(fd) rather than
46 * ngx_close_connection, to avoid double-closing the eventfd. No goto.
47 */
48static void
49brix_uring_teardown(brix_uring_t *u)
50{
51 if (u->ring_active && u->eventfd >= 0) {
52 io_uring_unregister_eventfd(&u->ring);
53 }
54
55 if (u->evc != NULL) {
56 if (u->evc->read->active) {
57 (void) ngx_del_event(u->evc->read, NGX_READ_EVENT, 0);
58 }
59 ngx_free_connection(u->evc);
60 u->evc->fd = (ngx_socket_t) -1;
61 u->evc = NULL;
62 }
63
64 if (u->eventfd >= 0) {
65 close(u->eventfd);
66 u->eventfd = -1;
67 }
68
69 if (u->ring_active) {
70 io_uring_queue_exit(&u->ring);
71 u->ring_active = 0;
72 }
73
74 u->enabled = 0;
75 u->slots = NULL; /* pool-owned; freed with the worker cycle pool */
76}

78/*
79 * brix_uring_init_fail — log the bring-up failure, tear down whatever was set
80 * up, and return the right verdict: NGX_ERROR under `on` (the worker refuses to
81 * run on the thread pool — master respawns; the §32.7 backstop) or NGX_OK under
82 * `auto` (silent degrade to the thread pool).
83 */
84static ngx_int_t
85brix_uring_init_fail(brix_uring_t *u, ngx_cycle_t *cycle, ngx_uint_t mode_on,
86 const char *what)
87{
88 /* Hoist the level into a variable: the ngx_log_error macro does not
89 * parenthesize its level argument, so a raw ternary would bind under >=. */
90 ngx_uint_t level = mode_on ? NGX_LOG_EMERG : NGX_LOG_NOTICE;
91
92 ngx_log_error(level, cycle->log, ngx_errno,
93 "brix: io_uring bring-up failed at %s%s", what,
94 mode_on
95 ? " — \"brix_io_uring on\" requires it; this worker refuses to run"
96 : "; falling back to the thread pool");
97
98 brix_uring_teardown(u);
99 return mode_on ? NGX_ERROR : NGX_OK;
100}

102/*
103 * brix_uring_init_worker — create this worker's ring after fork.
104 *
105 * Scans every enabled server block: the ring is created if any block wants
106 * io_uring (mode on/auto); queue depth is the max requested; restrictions are
107 * applied unless any wanting block turned them off. Bring-up sequence:
108 * queue_init [R_DISABLED if restricting] -> register restrictions + enable ->
109 * NOP self-test -> eventfd + register_eventfd -> fake-connection epoll bridge ->
110 * slot table. Any failure routes through brix_uring_init_fail (NGX_OK under
111 * auto, NGX_ERROR under on). When no block wants io_uring this is a no-op.
112 */
113ngx_int_t
114brix_uring_init_worker(ngx_cycle_t *cycle)
115{
116 brix_uring_t *u = &brix_uring_worker_ring;
117 brix_uring_scan_t scan;
118 const char *what;
119
120 if (uring_probe_features(cycle, &scan) != NGX_OK) {
121 return NGX_OK; /* nobody wants a ring / auto blocks on a bare host */
122 }
123
124 u->log = cycle->log;
125 u->eventfd = -1;
126 u->evc = NULL;
127 u->slots = NULL;
128 u->inflight = 0;
129 u->ring_active = 0;
130 u->enabled = 0;
131 u->restrict_ops = 0;
132 u->queue_depth = (uint32_t) scan.depth;
133
134 /* 1-3. create + notify-wire + restrict + enable the ring. */
135 what = uring_setup_rings(u, scan.want_restrict);
136 if (what != NULL) {
137 return brix_uring_init_fail(u, cycle, scan.mode_on, what);
138 }
139
140 /* 4. NOP self-test: prove submit -> complete works AND that the registered
141 * eventfd actually delivers the completion notification. */
142 what = uring_selftest_nop(u);
143 if (what != NULL) {
144 return brix_uring_init_fail(u, cycle, scan.mode_on, what);
145 }
146
147 /* 4b. UNDER-LOAD delivery self-test: fill the ring with queue_depth ops and
148 * require EVERY completion to arrive via the eventfd within the deadline. */
149 what = uring_selftest_burst(u);
150 if (what != NULL) {
151 return brix_uring_init_fail(u, cycle, scan.mode_on, what);
152 }
153
154 /* 5. wire the eventfd into the worker's epoll via a fake connection. */
155 what = uring_install_eventfd(u, cycle);
156 if (what != NULL) {
157 return brix_uring_init_fail(u, cycle, scan.mode_on, what);
158 }
159
160 /* 6. completion-slot table (pool-owned; freed with the cycle). */
161 what = uring_register_buffers(u, cycle);
162 if (what != NULL) {
163 return brix_uring_init_fail(u, cycle, scan.mode_on, what);
164 }
165
166 /* SB-W5b: attach the cross-worker kill-switch flag (NULL if the zone was
167 * not registered — the selector then reads "enabled") and arm the
168 * panic-file poll timer if a path was configured. */
169 u->disabled_flag = brix_uring_killswitch_ptr();
170 (void) brix_uring_panicfile_arm(cycle, &scan.panic_file);
171
172 u->enabled = 1;
173 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
174 "brix: io_uring disk-I/O backend active (queue_depth=%ui%s)",
175 (ngx_uint_t) u->queue_depth,
176 u->restrict_ops ? ", restricted" : "");
177
178 return NGX_OK;
179}
