1/*
2 * sd_http_select.c — endpoint selection, health scoring, and read failover for
3 * the HTTP-origin storage driver.
4 *
5 * WHAT: Owns the ranked, health-scored endpoint set's decision logic — pick the
6 * best endpoint, run one request with one-alternate transport failover
7 * (phase-68 T11), record the outcome against the endpoint's EWMA health
8 * score, and emit the operator's origin-flap / origin-switch audit trail.
9 * sd_http_request_fo() is the single request primitive the read path
10 * (HEAD/GET) drives; sd_http_write_path() composes the endpoint-0 write
11 * URL for the write path (writes never fail over — that would split-brain
12 * the store).
13 *
14 * WHY: Split out of sd_http.c (phase-79 file-size split): selection + failover
15 * is one cohesive concept, distinct from the read path (sd_http_read.c),
16 * the write path (sd_http_write.c), and the driver vtable/lifecycle
17 * (sd_http.c). The shared instance/endpoint layout comes from
18 * sd_http_internal.h; the per-request state (sd_http_req_t) is declared
19 * there because the read path constructs it too.
20 *
21 * HOW: Pure health-scoring helpers (score/pick/preferred) feed the failover
22 * orchestrator sd_http_request_fo, which is a flat early-return loop of
23 * named steps (select → perform → note-success / note-fail → next). All
24 * logging is at the edges; the scoring math is side-effect-honest.
25 */

60/* Effective pick score: policy rank first, health inside a rank. */
61static int
62sd_http_ep_score(const sd_http_endpoint *ep)
63{
64 return atomic_load_explicit((_Atomic int *) &ep->rank,
65 memory_order_relaxed)
66 * SD_HTTP_RANK_WEIGHT + ep->fail_score;
67}

69/* Pick the best endpoint (lowest effective score, order-stable ties). */
70static sd_http_endpoint *
71sd_http_pick(sd_http_inst_state *is)
72{
73 sd_http_endpoint *best = &is->eps[0];
74 int i;
75
76 for (i = 1; i < is->n_eps; i++) {
77 if (sd_http_ep_score(&is->eps[i]) < sd_http_ep_score(best)) {
78 best = &is->eps[i];
79 }
80 }
81 return best;
82}

84/* Rank-preferred endpoint ignoring health — the half-open probe target. */
85static sd_http_endpoint *
86sd_http_preferred(sd_http_inst_state *is)
87{
88 sd_http_endpoint *best = &is->eps[0];
89 int i;
90
91 for (i = 1; i < is->n_eps; i++) {
92 if (atomic_load_explicit(&is->eps[i].rank, memory_order_relaxed)
93 < atomic_load_explicit(&best->rank, memory_order_relaxed))
94 {
95 best = &is->eps[i];
96 }
97 }
98 return best;
99}

101/* Record one transport outcome: score = score*7/8 + (ok ? 0 : 256). */
102static void
103sd_http_score(sd_http_endpoint *ep, int ok)
104{
105 ep->fail_score = ep->fail_score * 7 / 8 + (ok ? 0 : 256);
106}

108/* Score + health-transition note in one step: crossing the 128 hysteresis
109 * band emits ONE degraded/recovered event (not one per failure) through the
110 * owner-injected hook — the operator's origin-flap trail. */
111static void
112sd_http_score_noted(sd_http_inst_state *is, sd_http_endpoint *ep, int ok)
113{
114 int was = ep->fail_score;
115
116 sd_http_score(ep, ok);
117 if (is->health_note == NULL) {
118 return;
119 }
120 if (!ok && was < 128 && ep->fail_score >= 128) {
121 is->health_note(ep->host, ep->port, 0);
122 } else if (ok && was >= 128 && ep->fail_score < 128) {
123 is->health_note(ep->host, ep->port, 1);
124 }
125}

146 * disabled for this instance (is->log unset at build).

164/* Selection audit: log (NOTICE) every change of the endpoint that answers —
165 * the "why is RAL being skipped for CERN" record. States why the policy-
166 * preferred endpoint was overridden when it was (health benching). Called
167 * from fill threads; cur_ep is racy-by-design (see the field comment). */
168static void
169sd_http_log_switch(sd_http_inst_state *is, sd_http_endpoint *ep)
170{
171 ngx_log_t *lg;
172 int idx = (int) (ep - is->eps);
173 sd_http_endpoint *pref;
174 char prev[300];
175
176 if (idx == is->cur_ep) {
177 return;
178 }
179 if (is->cur_ep >= 0 && is->cur_ep < is->n_eps) {
180 snprintf(prev, sizeof(prev), "%s:%d",
181 is->eps[is->cur_ep].host, is->eps[is->cur_ep].port);
182 } else {
183 snprintf(prev, sizeof(prev), "(none)");
184 }
185 is->cur_ep = idx;
186 lg = sd_http_live_log(is);
187 if (lg == NULL) {
188 return;
189 }
190 pref = sd_http_preferred(is);
191 if (ep == pref) {
192 ngx_log_error(NGX_LOG_NOTICE, lg, 0,
193 "brix: http origin switched to %s:%d (endpoint %d, rank %d, "
194 "fail_score %d; the policy-preferred endpoint), was %s",
195 ep->host, ep->port, idx, sd_http_ep_rank(ep), ep->fail_score,
196 prev);
197 } else {
198 ngx_log_error(NGX_LOG_NOTICE, lg, 0,
199 "brix: http origin switched to %s:%d (endpoint %d, rank %d, "
200 "fail_score %d), was %s; policy-preferred %s:%d SKIPPED "
201 "(rank %d, fail_score %d - benched by recent transport failures, "
202 "recovers via half-open probing as its score decays)",
203 ep->host, ep->port, idx, sd_http_ep_rank(ep), ep->fail_score,
204 prev, pref->host, pref->port, sd_http_ep_rank(pref),
205 pref->fail_score);
206 }
207}

209/* sd_http_fo_select — choose the endpoint the FIRST attempt targets.
210 *
211 * WHAT: Returns the starting endpoint: rank-preferred under force-primary,
212 * else best-by-score, with a periodic half-open probe of a benched
213 * preferred origin folded in.
214 * WHY: Force-primary pins the preferred endpoint (nowhere to fail over by
215 * policy); otherwise a benched origin would stay benched forever unless
216 * scores could move, so every 4th request re-tries the preferred one.
217 * HOW: Pick base endpoint by policy; when not force-primary and multiple
218 * endpoints exist, on each 4th tick swap in the rank-preferred endpoint
219 * if it carries any fail_score (a recovered origin earns its score back;
220 * the in-loop failover still answers if it is still down). */
221static sd_http_endpoint *
222sd_http_fo_select(const sd_http_req_t *rq)
223{
224 sd_http_inst_state *is = rq->is;
225 sd_http_endpoint *ep = rq->force_primary ? sd_http_preferred(is)
226 : sd_http_pick(is);
227 char klog[160];
228
229 if (rq->force_primary || is->n_eps <= 1 || (++is->probe_tick & 3u) != 0) {
230 return ep;
231 }
232 sd_http_endpoint *pref = sd_http_preferred(is);
233
234 if (pref->fail_score <= 0) {
235 return ep;
236 }
237 ngx_log_t *lg = sd_http_live_log(is);
238 if (pref != ep && lg != NULL) {
239 ngx_log_error(NGX_LOG_INFO, lg, 0,
240 "brix: http origin half-open probe: re-trying benched "
241 "preferred %s:%d (rank %d, fail_score %d) instead of "
242 "%s:%d for %s \"%s\"",
243 pref->host, pref->port, sd_http_ep_rank(pref),
244 pref->fail_score, ep->host, ep->port, rq->method,
245 sd_http_log_key(rq->key, klog, sizeof(klog)));
246 }
247 return pref;
248}

250/* sd_http_fo_perform — issue ONE request to `ep`, score the outcome.
251 *
252 * WHAT: Composes the URL, dispatches through the cred or plain transport slot,
253 * records the transport outcome against the endpoint's health score, and
254 * returns the transport rc (0 = ok). *errbuf carries the curl detail.
255 * WHY: GSI-over-https routes a per-open proxy PEM through request_cred as the
256 * mutual-TLS client cert; the common anonymous/bearer path uses the plain
257 * slot. A cert with no request_cred slot means allow-mode (open already
258 * refused deny-mode) — degrade to the plain request.
259 * HOW: snprintf the endpoint base_path + key, then request_cred vs request by
260 * cert presence and slot availability, then sd_http_score_noted. */
261static int
262sd_http_fo_perform(const sd_http_req_t *rq, sd_http_endpoint *ep,
263 char *errbuf, size_t errcap)
264{
265 sd_http_inst_state *is = rq->is;
266 char full[SD_HTTP_PATH_MAX];
267 int rc;
268
269 snprintf(full, sizeof(full), "%s%s", ep->base_path,
270 (rq->key != NULL && rq->key[0]) ? rq->key : "/");
271 errbuf[0] = '\0';
272 if (rq->cert_pem != NULL && rq->cert_pem[0] != '\0'
273 && is->transport->request_cred != NULL)
274 {
275 rc = is->transport->request_cred(is->tctx, ep->host, ep->port,
276 ep->tls, rq->method, full, rq->extra_hdrs,
277 NULL, 0, is->timeout_ms, rq->cert_pem,
278 rq->resp, errbuf, errcap);
279 } else {
280 rc = is->transport->request(is->tctx, ep->host, ep->port, ep->tls,
281 rq->method, full, rq->extra_hdrs, NULL, 0,
282 is->timeout_ms, rq->resp, errbuf, errcap);
283 }
284 sd_http_score_noted(is, ep, rc == 0);
285 return rc;
286}

288/* sd_http_fo_note_success — record the answering endpoint on a served request.
289 *
290 * WHAT: Stashes last_origin, the per-upstream failover flag, and emits the
291 * selection-change audit line. *used (when non-NULL) names the endpoint.
292 * WHY: The metric "failover" = served by a NON-PRIMARY endpoint; keyed on
293 * eps[0] (not `first`) so it holds however scoring reordered the picks
294 * and across the HEAD-then-GET sub-requests of one fill. */
295static void
296sd_http_fo_note_success(const sd_http_req_t *rq, sd_http_endpoint *ep,
297 sd_http_endpoint **used)
298{
299 sd_http_inst_state *is = rq->is;
300
301 if (used != NULL) {
302 *used = ep;
303 }
304 snprintf(is->last_origin, sizeof(is->last_origin), "%s:%d",
305 ep->host, ep->port);
306 is->last_failover = (ep != &is->eps[0]);
307 sd_http_log_switch(is, ep);
308}

310/* sd_http_fo_next — pick the alternate endpoint for a failover attempt.
311 *
312 * WHAT: Returns the best-by-score endpoint distinct from `first`, or NULL when
313 * there is none. On a hit, fires the failover-accounting hook and the
314 * NOTICE audit line.
315 * WHY: A transport failure on the first pick earns ONE alternate attempt
316 * (phase-68 T11); the alternate must not be the endpoint that just failed.
317 * HOW: Linear best-score scan skipping `first`; failover_note + log on success. */
318static sd_http_endpoint *
319sd_http_fo_next(const sd_http_req_t *rq, sd_http_endpoint *first)
320{
321 sd_http_inst_state *is = rq->is;
322 sd_http_endpoint *ep = NULL;
323 char klog[160];
324 int i;
325
326 for (i = 0; i < is->n_eps; i++) {
327 if (&is->eps[i] == first) {
328 continue;
329 }
330 if (ep == NULL
331 || sd_http_ep_score(&is->eps[i]) < sd_http_ep_score(ep))
332 {
333 ep = &is->eps[i];
334 }
335 }
336 if (ep == NULL) {
337 return NULL;
338 }
339 if (is->failover_note != NULL) {
340 is->failover_note(); /* T16: failover accounting */
341 }
342 ngx_log_t *lg = sd_http_live_log(is);
343 if (lg != NULL) {
344 ngx_log_error(NGX_LOG_NOTICE, lg, 0,
345 "brix: http origin failover for %s \"%s\": %s:%d -> %s:%d "
346 "(alternate rank %d, fail_score %d)",
347 rq->method, sd_http_log_key(rq->key, klog, sizeof(klog)),
348 first->host, first->port, ep->host, ep->port,
349 sd_http_ep_rank(ep), ep->fail_score);
350 }
351 return ep;
352}

374/* One request with read-failover (phase-68 T11): try the best endpoint, then
375 * ONE alternate on a TRANSPORT failure (an HTTP 4xx is NOT a transport
376 * failure — the object genuinely isn't there; do not mask it by failing
377 * over). The request identity (method/key/extra_hdrs/cert_pem/resp) rides in
378 * the caller-populated `rq`; `rq->extra_hdrs` is the pre-joined header block
379 * (Range and/or auth); `rq->cert_pem`, when non-NULL/non-empty, is a per-open
380 * TLS client-cert PATH (a user proxy PEM) presented via mutual-TLS through the
381 * transport's request_cred slot — the GSI-over-https backend leg (phase-70
382 * §5.1); NULL keeps the plain request slot (anonymous / header-borne auth).
383 * On success *used (when non-NULL) names the endpoint that answered. */
384int
385sd_http_request_fo(const sd_http_req_t *rq, sd_http_endpoint **used)
386{
387 sd_http_inst_state *is = rq->is;
388 sd_http_endpoint *ep = sd_http_fo_select(rq);
389 sd_http_endpoint *first = ep;
390 char errbuf[256], klog[160];
391 int attempt, rc, score_before;
392
393 for (attempt = 0; attempt < 2; attempt++) {
394 score_before = ep->fail_score;
395 rc = sd_http_fo_perform(rq, ep, errbuf, sizeof(errbuf));
396 if (rc == 0) {
397 sd_http_fo_note_success(rq, ep, used);
398 return 0;
399 }
400 /* The transport failure is the evidence an operator needs to explain
401 * a later "origin switched" line — record it with the curl detail. */
402 sd_http_fo_note_fail(rq, ep, errbuf, attempt, score_before);
403 if (is->n_eps < 2 || rq->force_primary) {
404 break; /* force-primary: never fail over */
405 }
406 ep = sd_http_fo_next(rq, first);
407 if (ep == NULL) {
408 break;
409 }
410 }
411 ngx_log_t *lg = sd_http_live_log(is);
412 if (lg != NULL) {
413 ngx_log_error(NGX_LOG_ERR, lg, 0,
414 "brix: http origin request exhausted all endpoints (%d tried) "
415 "for %s \"%s\" - reporting EIO to the fill layer",
416 (is->n_eps < 2) ? 1 : 2, rq->method,
417 sd_http_log_key(rq->key, klog, sizeof(klog)));
418 }
419 errno = EIO;
420 return -1;
421}
