1/*
2 * vfs_staged.c — VFS atomic staged-write lifecycle (open temp → commit/abort).
3 *
4 * WHAT: Implements the brix_vfs_staged_* family — a thin VFS-owned wrapper
5 * over the compat staged-file primitive used by every crash-safe upload
6 * (S3 PutObject, WebDAV PUT, multipart assembly): create a unique O_EXCL
7 * temp inside the export root, let the caller write the staged fd, then
8 * atomically publish it onto the final (resolved ctx) path, or abort.
9 *
10 * WHY: The upload paths called brix_staged_open/commit/abort directly, so the
11 * publish of a finished object — a real namespace mutation — produced no
12 * metric or access-log line. Funnelling the lifecycle through here books
13 * one BRIX_METRIC_OP_WRITE on commit (byte count = the committed object
14 * size) and inherits the write gate, while still delegating the temp-file
15 * and rename mechanics to compat/staged_file.
16 *
17 * HOW: brix_vfs_staged_open() write-gates, allocates the handle on ctx->pool,
18 * and opens the temp via brix_staged_open() with the final path taken
19 * from the resolved ctx. Callers write through the raw fd accessor
20 * (brix_vfs_staged_fd) — the same fd they used before. commit publishes
21 * onto the ctx path (RENAME_NOREPLACE when excl) and meters OP_WRITE;
22 * abort closes and optionally unlinks the temp. The handle struct lives in
23 * vfs_internal.h; only the opaque type is exposed in vfs.h.
24 */
25#include "vfs_internal.h"

27#include "core/compat/staged_file.h"
28#include "fs/xfer/xfer.h" /* unified transfer audit ledger (one line per publish) */

32/* Duplicate a C string onto `pool` (NUL-terminated). Returns NULL on a NULL input
33 * (with *lenp = 0) or on allocation failure — the caller distinguishes the two by
34 * checking the input. Used to make a staged handle's ctx self-contained. */
35static char *
36staged_pool_strdup(ngx_pool_t *pool, const char *s, size_t *lenp)
37{
38 size_t n;
39 u_char *p;
40
41 if (s == NULL) {
42 if (lenp != NULL) {
43 *lenp = 0;
44 }
45 return NULL;
46 }
47 n = ngx_strlen(s);
48 p = ngx_pnalloc(pool, n + 1);
49 if (p == NULL) {
50 return NULL;
51 }
52 ngx_memcpy(p, s, n);
53 p[n] = '\0';
54 if (lenp != NULL) {
55 *lenp = n;
56 }
57 return (char *) p;
58}

60/* Open a staged temp file for the resolved ctx (final) path. Write-gated.
61 * `mode` is the final object's permission bits; `attempts` bounds the O_EXCL
62 * unique-name retries. Returns a handle on ctx->pool, or NULL with the errno in
63 * *err_out (if non-NULL). Release via brix_vfs_staged_commit/abort. */
64brix_vfs_staged_t *
65brix_vfs_staged_open(brix_vfs_ctx_t *ctx, mode_t mode, ngx_uint_t attempts,
66 int *err_out)
67{
68 brix_vfs_staged_t *st;
69 const char *final_path = brix_vfs_ctx_path(ctx);
70
71 if (brix_vfs_require_write(ctx) != NGX_OK) {
72 if (err_out != NULL) {
73 *err_out = errno;
74 }
75 return NULL;
76 }
77
78 if (ctx->root_canon == NULL) {
79 errno = EINVAL;
80 if (err_out != NULL) {
81 *err_out = errno;
82 }
83 return NULL;
84 }
85
86 st = ngx_pcalloc(ctx->pool, sizeof(*st));
87 if (st == NULL) {
88 errno = ENOMEM;
89 if (err_out != NULL) {
90 *err_out = errno;
91 }
92 return NULL;
93 }
94
95 /*
96 * Self-contain the ctx. The staged handle outlives the caller's stack frame:
97 * S3 and WebDAV PUT dispatch the body write to a thread pool and RETURN before
98 * brix_vfs_staged_commit runs, so a ctx — and the resolved-path / root_canon
99 * buffers it POINTS at (WebDAV's resolved path lives in a stack `char[]`) —
100 * that was stack-allocated by the caller would be a use-after-free at commit.
101 * Deep-copy the ctx struct and those two strings onto ctx->pool (the request
102 * pool, which lives until the response completes) so the handle is stable
103 * across the async hop regardless of how the caller allocated its ctx.
104 */
105 st->ctx = ngx_palloc(ctx->pool, sizeof(*st->ctx));
106 if (st->ctx == NULL) {
107 errno = ENOMEM;
108 if (err_out != NULL) {
109 *err_out = errno;
110 }
111 return NULL;
112 }
113 *st->ctx = *ctx;
114 {
115 size_t rlen = 0;
116 const char *rpath = brix_vfs_ctx_path(ctx);
117 char *rpath_dup = staged_pool_strdup(ctx->pool, rpath, &rlen);
118 char *root_dup = staged_pool_strdup(ctx->pool, ctx->root_canon, NULL);
119
120 if ((rpath != NULL && rpath_dup == NULL)
121 || (ctx->root_canon != NULL && root_dup == NULL))
122 {
123 errno = ENOMEM;
124 if (err_out != NULL) {
125 *err_out = errno;
126 }
127 return NULL;
128 }
129 if (rpath_dup != NULL) {
130 st->ctx->resolved.resolved.data = (u_char *) rpath_dup;
131 st->ctx->resolved.resolved.len = rlen;
132 }
133 if (root_dup != NULL) {
134 st->ctx->root_canon = root_dup;
135 }
136 }
137 st->pool = ctx->pool;
138 st->log = ctx->log;
139 st->staged.fd = NGX_INVALID_FILE;
140
141 /* A non-POSIX backend owns the staged lifecycle: delegate straight to the
142 * resolved instance's staged_open slot (Mode A). When staging is configured the
143 * registry composes the sd_stage write-back DECORATOR (C-2/C-6) as that instance
144 * — so a local-temp-then-promote upload is the decorator's staged_open, NOT a
145 * second copy here. A bare source streams the body to the remote final path. */
146 if (ctx->sd != NULL && ctx->sd->driver != brix_sd_default_driver()
147 && ctx->sd->driver->staged_open != NULL)
148 {
149 int sderr = 0;
150
151 st->driver_staged = ctx->sd->driver->staged_open(ctx->sd,
152 brix_vfs_export_relative(ctx, final_path), mode, &sderr);
153 if (st->driver_staged == NULL) {
154 if (err_out != NULL) {
155 *err_out = sderr;
156 }
157 errno = sderr;
158 return NULL;
159 }
160 return st;
161 }
162
163 /* Pure-POSIX export: the local confined temp IS the storage. */
164 if (brix_staged_open(ctx->log, ctx->root_canon, final_path, O_WRONLY,
165 mode, attempts, &st->staged) != NGX_OK)
166 {
167 if (err_out != NULL) {
168 *err_out = errno;
169 }
170 return NULL;
171 }
172
173 return st;
174}

193/* Backend-neutral staged write: POSIX → pwrite the temp fd; driver-backed →
194 * driver->staged_write (tracking the high-water mark for the commit metric). */
195ngx_int_t
196brix_vfs_staged_write(brix_vfs_staged_t *st, const void *buf, size_t len,
197 off_t off)
198{
199 if (st == NULL) {
200 errno = EINVAL;
201 return NGX_ERROR;
202 }
203
204 if (st->driver_staged != NULL) {
205 ssize_t n = st->ctx->sd->driver->staged_write(st->driver_staged, buf,
206 len, off);
207 if (n < 0 || (size_t) n != len) {
208 if (n >= 0) {
209 errno = EIO;
210 }
211 return NGX_ERROR;
212 }
213 if (off + (off_t) len > st->driver_total) {
214 st->driver_total = off + (off_t) len;
215 }
216 return NGX_OK;
217 }
218
219 return brix_vfs_pwrite_full(st->staged.fd, buf, len, off);
220}

231/* Atomically publish the staged temp onto the resolved ctx (final) path. When
232 * `excl`, uses RENAME_NOREPLACE and fails with errno==EEXIST if the final path
233 * already exists (caller maps to 412). Metered as OP_WRITE with the committed
234 * object size. Returns NGX_OK or NGX_ERROR with errno set. */
235ngx_int_t
236brix_vfs_staged_commit(brix_vfs_staged_t *st, unsigned excl)
237{
238 const char *final_path;
239 uint64_t start = brix_vfs_now_ns();
240 size_t bytes = 0;
241 struct stat sb;
242 ngx_int_t rc;
243 int saved_errno;
244
245 if (st == NULL) {
246 errno = EINVAL;
247 return NGX_ERROR;
248 }
249
250 final_path = brix_vfs_ctx_path(st->ctx);
251
252 /* Driver-backed: the driver publishes the object atomically. On success it
253 * consumes its staged handle (NULL it out so abort is not double-applied);
254 * on failure the handle stays valid for the caller's abort. The byte count
255 * is the high-water mark tracked across staged writes. */
256 if (st->driver_staged != NULL) {
257 rc = st->ctx->sd->driver->staged_commit(st->driver_staged, excl);
258 if (rc != NGX_OK) {
259 saved_errno = errno;
260 brix_vfs_observe_ctx_op(st->ctx, final_path,
261 BRIX_METRIC_OP_WRITE, NULL, 0, NGX_ERROR,
262 saved_errno, start);
263 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path, NULL, 0,
264 BRIX_XFER_COMMIT_ERR, saved_errno, st->log);
265 errno = saved_errno;
266 return NGX_ERROR;
267 }
268 bytes = (size_t) st->driver_total;
269 st->driver_staged = NULL; /* consumed by a successful commit */
270 brix_vfs_observe_ctx_op(st->ctx, final_path, BRIX_METRIC_OP_WRITE,
271 NULL, bytes, NGX_OK, 0, start);
272 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path, NULL, bytes,
273 BRIX_XFER_OK, 0, st->log);
274 return NGX_OK;
275 }
276
277 rc = excl
278 ? brix_staged_commit_excl(st->log, st->ctx->root_canon, &st->staged,
279 final_path)
280 : brix_staged_commit(st->log, st->ctx->root_canon, &st->staged,
281 final_path);
282 if (rc != NGX_OK) {
283 saved_errno = errno;
284 brix_vfs_observe_ctx_op(st->ctx, final_path, BRIX_METRIC_OP_WRITE,
285 NULL, 0, NGX_ERROR, saved_errno, start);
286 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path, NULL, 0,
287 BRIX_XFER_COMMIT_ERR, saved_errno, st->log);
288 errno = saved_errno;
289 return NGX_ERROR;
290 }
291
292 if (brix_lstat_confined_canon(st->log, st->ctx->root_canon, final_path,
293 &sb, 1) == 0
294 && S_ISREG(sb.st_mode))
295 {
296 bytes = (size_t) sb.st_size;
297 }
298
299 brix_vfs_observe_ctx_op(st->ctx, final_path, BRIX_METRIC_OP_WRITE, NULL,
300 bytes, NGX_OK, 0, start);
301 /* The publication record — the single place this committed object is
302 * accounted for across all transfer kinds. */
303 brix_xfer_finish(BRIX_XFER_STAGE, "in", final_path, NULL, bytes,
304 BRIX_XFER_OK, 0, st->log);
305 return NGX_OK;
306}
