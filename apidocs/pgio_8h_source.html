1/*
2 * pgio.h — kXR page-mode (kXR_pgread / kXR_pgwrite) per-page CRC32c framing.
3 *
4 * WHAT: encode/decode the wire layout [CRC32c-BE(4)][data ≤ kXR_pgPageSZ] repeated,
5 * with each page aligned to the FILE offset (the first/last fragment may be
6 * short on an unaligned offset).
7 * WHY: the module (pgread encode / pgwrite decode) and the native client (pgwrite
8 * encode / pgread decode) implement the IDENTICAL framing + per-page CRC; one
9 * shared kernel makes producer and consumer agree by construction.
10 * HOW: pure ptr+len over caller buffers, using the shared CRC32c kernels
11 * (libxrdproto). No ngx, no allocation, no I/O. (libxrdproto)
12 */

20/*
21 * Encode src[0..len) into page units at dst. Pages align to file_off, so the
22 * first fragment is shortened to land later pages on a kXR_pgPageSZ boundary.
23 * dst must hold at least len + 4*(number-of-pages) bytes. Returns encoded length.
24 */
25size_t xrdp_pg_encode(const uint8_t *src, size_t len, int64_t file_off,
26 uint8_t *dst);

28/*
29 * Decode page units in pg[0..pglen) (aligned at file_off) into dst[0..dstcap),
30 * verifying each page's CRC32c. Returns:
31 * >= 0 decoded data byte count
32 * -1 a page's CRC did not match (sets *bad_off to that page's file offset)
33 * -2 malformed framing, dst overflow, or offset overflow
34 * bad_off may be NULL.
35 */
36ssize_t xrdp_pg_decode(const uint8_t *pg, size_t pglen, int64_t file_off,
37 uint8_t *dst, size_t dstcap, int64_t *bad_off);

39/*
40 * One corrupt page in a CSE (checksum-error) retransmit list: the FILE offset
41 * of the page and the byte length of the page fragment (short for an unaligned
42 * first/last page).
43 */
44typedef struct {
45 int64_t off; /* file offset of the page */
46 uint32_t dlen; /* fragment length in bytes */
47} xrdp_pg_bad_t;

49/*
50 * Decode like xrdp_pg_decode(), but DO NOT abort on a CRC mismatch: verify
51 * every page and copy ALL pages (good and bad) into dst, recording each failed
52 * page in bad_out[0..max_bad). This implements the server's "accept-then-
53 * correct" (CSE) path — corrupt bytes still land in dst so the caller can write
54 * them, and the bad list drives the kXR_pgWrCSE retransmit reply.
55 *
56 * Returns:
57 * >= 0 decoded data byte count (*bad_count = number of corrupt pages, may be 0)
58 * -2 malformed framing, dst overflow, or offset overflow
59 * -3 more than max_bad corrupt pages in this request (→ kXR_TooManyErrs);
60 * *bad_count is set to max_bad and decoding stops
61 * bad_out / bad_count must be non-NULL.
62 */
63ssize_t xrdp_pg_decode_collect(const uint8_t *pg, size_t pglen, int64_t file_off,
64 uint8_t *dst, size_t dstcap,
65 xrdp_pg_bad_t *bad_out, size_t max_bad,
66 size_t *bad_count);
