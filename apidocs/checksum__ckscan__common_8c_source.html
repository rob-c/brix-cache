1/*
2 * checksum_ckscan_common.c — shared helpers for kXR_Qckscan directory/file checksum scans.
3 *
4 * WHAT: Provides three functions used by checksum_ckscan_async.c and checksum_ckscan_dispatch.c to walk filesystem trees and compute per-file checksums. ckscan_append() grows a response buffer with "algo hex logical_path" lines; brix_ckscan_join_logical() constructs canonical logical paths from parent+child components; brix_ckscan_walk() recursively scans directories up to max_depth/max_files limits, computing checksums for regular files.
5 *
6 * WHY: kXR_Qckscan walks a directory tree off the event loop (async via thread pool) and returns one checksum line per file. These shared helpers eliminate code duplication between async and dispatch implementations while providing consistent path construction, buffer growth logic, and recursive walk semantics with depth/capacity limits.
7 */

18/* kXR_Qckscan — directory / file checksum scan */
19/* Append one "algo hex logical_path\n" line, growing the buffer if needed. */
20int
21brix_ckscan_append(u_char **buf, size_t *cap, size_t *used,
22 const char *algo, const char *hex, const char *logical)
23{
24 char line[BRIX_MAX_PATH + 64];
25 size_t llen;
26 u_char *nb;
27
28 /*
29 * snprintf returns the length it WOULD have written, so llen can exceed
30 * sizeof(line) on truncation. The >= test below catches that (and the
31 * mandatory NUL means equality is already overflow), so `line` is never
32 * read past a valid terminator. `hex` is the already-formatted lowercase
33 * checksum — its width is algorithm-dependent (8 chars for adler32/crc32c,
34 * 16 for crc64/crc64nvme), carried in the string so this formatter stays
35 * width-agnostic; the two spaces between hex and path are literal, part of
36 * the on-wire line format the client parses.
37 */
38 llen = (size_t) snprintf(line, sizeof(line), "%s %s %s\n",
39 algo, hex, logical);
40 if (llen >= sizeof(line)) {
41 return 0; /* path too long: skip */
42 }
43
44 /*
45 * Grow trigger reserves llen + 2 (line bytes + a trailing NUL slot + 1
46 * slack) so the caller's final buf[used]='\0' always lands in-bounds.
47 * Growth is exponential (*cap*2) plus the line size, so a single oversized
48 * line still fits even when doubling alone would not.
49 */
50 if (*used + llen + 2 > *cap) {
51 size_t new_cap = *cap * 2 + llen + 2;
52 /*
53 * Manual realloc: this runs on a thread-pool worker, so the buffer is
54 * heap-backed (ngx_alloc, not pool). On OOM we return -1 WITHOUT freeing
55 * *buf — ownership stays with the caller, which frees it on its error
56 * path. Only *used bytes are copied (the live prefix); the old block is
57 * freed only after a successful copy. *buf==NULL on the very first call
58 * is tolerated (skip the copy/free).
59 */
60 nb = ngx_alloc(new_cap, ngx_cycle->log);
61 if (nb == NULL) {
62 return -1;
63 }
64 if (*buf) {
65 ngx_memcpy(nb, *buf, *used);
66 ngx_free(*buf);
67 }
68 *buf = nb;
69 *cap = new_cap;
70 }
71
72 ngx_memcpy(*buf + *used, line, llen);
73 *used += llen;
74 return 1;
75}
76/* WHY: Qckscan responses are newline-delimited "algo hex logical_path" lines growing into a dynamically-sized buffer. The client expects all checksum results in one response, so the buffer must handle arbitrary tree sizes via exponential growth. Returns 1 on success (line appended), 0 if output line would overflow local snprintf buffer (path too long → skip silently), -1 if OOM during buffer reallocation. */
77/* HOW: snprintf(line) formats "%s %08x %s\n" with algo, cksum as hex, logical path — returns llen. If llen >= sizeof(line) returns 0 (skip). Checks *used+llen+2 <= *cap — if insufficient allocates new_cap=*cap*2+llen+2 via ngx_alloc(), copies existing *buf content, frees old buffer, updates *buf and *cap. memcpy line into *buf at offset *used; increments *used by llen. Returns 1. */
78
79/*
80 * ckscan_walk_ctx_t + ckscan_walk_file — the per-file callback driven by
81 * brix_vfs_walk: compute the file's checksum (algorithm-appropriate hex width)
82 * and append one "algo hex logical" line. An unreadable file or unknown algorithm
83 * is a soft skip (return NGX_OK so the scan continues); only an append OOM aborts
84 * the walk (records cc->oom so the caller maps it to kXR_NoMemory). The confined
85 * open + traversal happen INSIDE the VFS walk — this layer never touches a fd
86 * except the read-only one the walk hands it.
87 */
88typedef struct {
89 ngx_log_t *log;
90 const char *algo;
91 u_char **buf;
92 size_t *cap;
93 size_t *used;
94 int oom;
95} ckscan_walk_ctx_t;

97static ngx_int_t
98ckscan_walk_file(void *cookie, const char *logical,
99 const brix_vfs_stat_t *st, int fd)
100{
101 ckscan_walk_ctx_t *cc = cookie;
102 char hex[129]; /* 8 (adler32/crc32c) … EVP_MAX_MD_SIZE*2 */
103
104 (void) st;
105
106 if (brix_checksum_hex_name_fd(cc->algo, fd, logical, cc->log,
107 hex, sizeof(hex), NULL, 0) != NGX_OK)
108 {
109 return NGX_OK; /* skip unreadable / bad-algorithm file (continue scan) */
110 }
111
112 /* append returns 1=ok, 0=line too long (soft skip), <0=OOM (abort). */
113 if (brix_ckscan_append(cc->buf, cc->cap, cc->used, cc->algo, hex,
114 logical) < 0)
115 {
116 cc->oom = 1;
117 return NGX_ERROR;
118 }
119 return NGX_OK;
120}

122/* Run a kXR_Qckscan over a file/dir target through the confined VFS walk and map
123 * the outcome to a kXR error code. See query_internal.h for the full contract. */
124ngx_int_t
125brix_ckscan_run(ngx_log_t *log, int rootfd, const char *logical,
126 const char *algo, u_char **buf, size_t *cap, size_t *used,
127 ngx_uint_t max_depth, ngx_uint_t max_files,
128 uint16_t *err_code, char *err_msg, size_t err_sz)
129{
130 ckscan_walk_ctx_t cc;
131 brix_vfs_walk_opts_t opts;
132 brix_vfs_walk_target_t target = BRIX_VFS_WALK_NONE;
133 char walkmsg[128] = "";
134 ngx_int_t rc;
135
136 ngx_memzero(&cc, sizeof(cc));
137 cc.log = log;
138 cc.algo = algo;
139 cc.buf = buf;
140 cc.cap = cap;
141 cc.used = used;
142
143 ngx_memzero(&opts, sizeof(opts));
144 opts.max_depth = max_depth;
145 opts.max_files = max_files;
146 opts.open_files = 1; /* the walk hands each regular file a read-only fd */
147
148 rc = brix_vfs_walk(log, rootfd, logical, &opts, ckscan_walk_file, &cc,
149 &target, walkmsg, sizeof(walkmsg));
150
151 if (rc == NGX_DECLINED) {
152 *err_code = kXR_NotFound;
153 snprintf(err_msg, err_sz, "stat failed: %s", strerror(errno));
154 return NGX_ERROR;
155 }
156 if (rc != NGX_OK) {
157 if (cc.oom) {
158 *err_code = kXR_NoMemory;
159 snprintf(err_msg, err_sz, "out of memory");
160 } else {
161 *err_code = kXR_IOError;
162 snprintf(err_msg, err_sz, "%s", walkmsg[0] ? walkmsg : "I/O error");
163 }
164 return NGX_ERROR;
165 }
166
167 if (target == BRIX_VFS_WALK_OTHER) {
168 *err_code = kXR_ArgInvalid;
169 snprintf(err_msg, err_sz, "not a file or directory");
170 return NGX_ERROR;
171 }
172 /* A regular-file target that produced no line means its open succeeded but
173 * the checksum failed (the callback skipped it) — a hard error for a single
174 * file, unlike an (acceptably) empty directory listing. */
175 if (target == BRIX_VFS_WALK_FILE && *used == 0) {
176 *err_code = kXR_IOError;
177 snprintf(err_msg, err_sz, "checksum computation failed");
178 return NGX_ERROR;
179 }
180
181 return NGX_OK;
182}
