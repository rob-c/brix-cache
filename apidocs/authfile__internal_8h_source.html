1/*
2 * authfile_internal.h — internal contract shared across the XrdAcc authdb
3 * parser translation units (authfile.c orchestrator + authfile_tokenize.c lexer
4 * + authfile_record.c record builders).
5 *
6 * WHAT: declares the tokenizer scanner state (acc_tok_t) and its word/record/EOF
7 * token codes, the parse-wide dispatch context (acc_parse_ctx_t), and the four
8 * cross-file entry points — the lexer's acc_tok_next() consumed by the
9 * orchestrator, and the three per-record handlers (iddef/rule/named) invoked
10 * from the orchestrator's record dispatch.
11 *
12 * WHY: authfile.c was split into three focused units (file-size + single-concern
13 * standard) — the lexer, the record parser/matcher, and the load/dispatch
14 * orchestrator. A symbol defined in one unit and referenced from another must
15 * be non-static and declared once, in a single include-guarded header, so the
16 * split stays link-clean with zero behaviour change from the former monolith.
17 *
18 * HOW: every authfile*.c includes this after acc.h. Structs are plain PODs
19 * carried on the parse stack; the extern functions keep the exact signatures
20 * they had as file-local statics. Requires: acc.h (brix_acc_* types, ngx core)
21 * before inclusion.
22 */

53/*
54 * acc_parse_ctx_t — the parse-wide state threaded through the record dispatch
55 * chain.
56 *
57 * WHAT: bundles the destination tables, the running id-definition tail, the
58 * exclusive-rule sequence counter and the log so that the record helpers take
59 * one context pointer instead of five loose scalars.
60 * WHY: the XrdAcc record handlers all read `tabs`/`log` and mutate the same
61 * `id_tail`/`excl_seq` cursors; passing them as a unit keeps signatures small
62 * (≤5 params) and makes the shared mutable cursors explicit at every call.
63 * HOW: created once on the parse stack in brix_acc_authfile_parse() and passed
64 * by address down through acc_dispatch_record() and the per-type helpers; the
65 * pool/spacechar/uridecode inputs live on `tabs` and are read via it.
66 */
67typedef struct {
68 brix_acc_tables_t *tabs;
69 brix_acc_idrule_t *id_tail; /* tail of id_defs, appended in file order */
70 int excl_seq; /* next exclusive-rule order number */
71 ngx_log_t *log;
72} acc_parse_ctx_t;
