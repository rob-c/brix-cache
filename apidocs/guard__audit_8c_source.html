1/*
2 * guard_audit.c — fail2ban audit line formatting.
3 *
4 * WHAT: renders one flagged request as a single stable key=value line, plus
5 * the reason/op-class token maps the line (and the fail2ban filters) use.
6 * WHY: the audit line IS the fail2ban contract — field order and token
7 * spelling are load-bearing; keeping the formatter pure and in one place
8 * guarantees every adapter emits the identical shape.
9 * HOW: snprintf into a caller buffer; adapters own the clock (ts string) and
10 * path sanitization, mirroring src/net/tap/tap_audit.c.
11 */
12#include "guard.h"
13#include <stdio.h>
14
15/* ---- Reason -> stable lowercase token ----
16 *
17 * WHAT: returns the audit-line token for a guard_reason_t ("signature",
18 * "grammar", "notfound", "authfail", "none").
19 *
20 * WHY: fail2ban filters match `signal=<token>` literally — the spelling here
21 * and in deploy/fail2ban/filter.d/ must never drift apart.
22 *
23 * HOW: 1. Switch over the enum; unknown values map to "none".
24 */
25const char *
26guard_reason_str(guard_reason_t r)
27{
28 switch (r) {
29 case GUARD_R_SIGNATURE: return "signature";
30 case GUARD_R_GRAMMAR: return "grammar";
31 case GUARD_R_NOTFOUND: return "notfound";
32 case GUARD_R_AUTHFAIL: return "authfail";
33 case GUARD_R_NONE: default: return "none";
34 }
35}
36
37/* ---- Op-class -> stable lowercase token ----
38 *
39 * WHAT: returns the audit-line token for a guard_op_class_t.
40 *
41 * WHY: gives operators grep-able op context in the audit log without
42 * exposing protocol-specific opcode numbers.
43 *
44 * HOW: 1. Switch over the enum; unknown values map to "unknown".
45 */
46const char *
47guard_op_str(guard_op_class_t op)
48{
49 switch (op) {
50 case GUARD_OP_READ: return "read";
51 case GUARD_OP_WRITE: return "write";
52 case GUARD_OP_LIST: return "list";
53 case GUARD_OP_DELETE: return "delete";
54 case GUARD_OP_JOBCTL: return "jobctl";
55 case GUARD_OP_STAGE: return "stage";
56 case GUARD_OP_INFO: return "info";
57 case GUARD_OP_DELEG: return "deleg";
58 case GUARD_OP_HANDSHAKE: return "handshake";
59 case GUARD_OP_UNKNOWN: default: return "unknown";
60 }
61}

63/* ---- Format one flagged request as a key=value audit line ----
64 *
65 * WHAT: writes `<ts> ip=<ip> proto=<proto> signal=<reason> op=<op>
66 * path="<path>" status=<code>` into out[0..outsz). Returns bytes written
67 * (excluding the NUL), or 0 if the line would not fit (nothing partial is
68 * ever emitted).
69 *
70 * WHY: single-line, fixed field order is what the fail2ban failregex and the
71 * operator's grep both key on; a truncated line could split an attacker's
72 * path across the status field, so it is all-or-nothing.
73 *
74 * HOW: 1. One snprintf with the caller's pre-sanitized fields (the adapter
75 * escaped control/quote bytes; the formatter only wraps in quotes).
76 * 2. Treat encoding errors and truncation as "does not fit" -> 0.
77 */
78size_t
79guard_audit_format(const guard_request_t *req, guard_reason_t reason,
80 const char *ts, char *out, size_t outsz)
81{
82 int written;
83
84 written = snprintf(out, outsz,
85 "%s ip=%s proto=%s signal=%s op=%s path=\"%.*s\" status=%d",
86 ts, req->ip, req->proto, guard_reason_str(reason),
87 guard_op_str(req->op), (int) req->path_len, req->path,
88 req->status_code);
89 if (written < 0 || (size_t) written >= outsz) {
90 return 0; /* truncated -> emit nothing */
91 }
92 return (size_t) written;
93}
