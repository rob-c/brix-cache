8/*
9 * kXR_protocol - negotiate protocol version, auth protocols, and TLS support.
10 *
11 * This is the first XRootD request after the raw handshake. It does not log a
12 * user in; it only advertises server capabilities and, when configured, arms
13 * the native TLS upgrade for the following login/auth flow.
14 */
15/* Handle kXR_protocol — negotiate server capabilities by building the fixed
16 * 8-byte ServerProtocolBody response (protocol version, flags, and the security
17 * requirement advertised to the client). */
18ngx_int_t
19brix_handle_protocol(brix_ctx_t *ctx, ngx_connection_t *c,
20 ngx_stream_brix_srv_conf_t *conf)
21{
22 ServerProtocolBody body;
23 u_char *buf;
24 size_t bodylen, total;
25 u_char client_flags;
26 int want_gsi;
27 int want_token;
28 int want_sss;
29 int want_unix;
30 int want_krb5;
31 int want_host;
32 int want_pwd;
33 int client_wants_tls;
34 int offer_tls;
35
36 /* kXR_protocol packs client capability flags into the fifth byte of body[]. */
37 client_flags = ctx->recv.cur_body[4];
38 want_gsi = (conf->auth == BRIX_AUTH_GSI
39 || conf->auth == BRIX_AUTH_BOTH);
40 want_token = (conf->auth == BRIX_AUTH_TOKEN
41 || conf->auth == BRIX_AUTH_BOTH);
42 want_sss = (conf->auth == BRIX_AUTH_SSS);
43 want_unix = (conf->auth == BRIX_AUTH_UNIX);
44 want_krb5 = (conf->auth == BRIX_AUTH_KRB5);
45 want_host = (conf->auth == BRIX_AUTH_HOST);
46 want_pwd = (conf->auth == BRIX_AUTH_PWD);
47
48 /* kXR_wantTLS: client requires TLS; kXR_ableTLS: client is TLS-capable. */
49 client_wants_tls = (client_flags & kXR_wantTLS) ? 1 : 0;
50 offer_tls = (conf->tls && conf->tls_ctx != NULL
51 && (client_flags & (kXR_ableTLS | kXR_wantTLS)));
52
53 /* Reject if the client demands TLS but this listener has none configured. */
54 if (client_wants_tls && (!conf->tls || conf->tls_ctx == NULL)) {
55 return brix_send_error(ctx, c, kXR_TLSRequired,
56 "TLS required by client but not configured on this server");
57 }
58
59 /*
60 * Base kXR_protocol reply is the fixed 8-byte ServerProtocolBody.
61 * If the client advertised security negotiation support, append the small
62 * SecurityInfo trailer describing which auth protocols we offer.
63 */
64 bodylen = sizeof(body);
65 if (client_flags & kXR_secreqs) {
66 int sec_count = (want_gsi ? 1 : 0) + (want_token ? 1 : 0)
67 + (want_sss ? 1 : 0) + (want_unix ? 1 : 0)
68 + (want_krb5 ? 1 : 0) + (want_host ? 1 : 0)
69 + (want_pwd ? 1 : 0);
70 bodylen += 4; /* SecurityInfo header */
71 bodylen += (size_t) sec_count * 8; /* 8 bytes per SecurityProtocol entry */
72 bodylen += 6; /* ServerResponseReqs_Protocol */
73 }
74
75 total = XRD_RESPONSE_HDR_LEN + bodylen;
76 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
77
78 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
79 (uint32_t) bodylen,
80 (ServerResponseHdr *) buf);
81
82 body.pval = htonl(kXR_PROTOCOLVERSION);
83 {
84 uint32_t caps = kXR_isServer
85 | kXR_suppgrw /* pgread/pgwrite always implemented */
86 | kXR_supposc /* POSC always implemented */
87 | ((conf->manager_map || conf->manager_mode) ? kXR_isManager : 0)
88 | (conf->caps.supervisor
89 ? (kXR_isManager | kXR_attrSuper) : 0)
90 | ((conf->caps.virtual_redirector
91 || (conf->manager_map != NULL && conf->cms.addr == NULL))
92 ? (kXR_isManager | kXR_attrVirtRdr) : 0)
93 | (conf->caps.metadata_only ? kXR_attrMeta : 0)
94 | ((conf->proxy.enable > 0 || conf->proxy.upstreams != NULL)
95 ? kXR_attrProxy : 0)
96 | ((conf->cache_root.len > 0 || conf->cache_origin_host.len > 0)
97 ? kXR_attrCache : 0)
98 | (conf->caps.collapse_redir ? kXR_collapseRedir : 0)
99 | (conf->caps.recover_writes ? kXR_recoverWrts : 0)
100 | (offer_tls ? (kXR_haveTLS | kXR_gotoTLS | kXR_tlsLogin) : 0);
101 body.flags = htonl(caps);
102 }
103
104 /* Fixed 8-byte prefix every protocol reply starts with after the response header. */
105 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, &body, sizeof(body));
106
107 if (client_flags & kXR_secreqs) {
108 /*
109 * SecurityInfo header:
110 * byte 1 advertises whether security is required,
111 * byte 2 is the number of following protocol entries.
112 * bytes 0 and 3 are left zero because this implementation does not use
113 * any of the optional legacy fields encoded there.
114 */
115 u_char *si = buf + XRD_RESPONSE_HDR_LEN + sizeof(body);
116 int sec_count = (want_gsi ? 1 : 0) + (want_token ? 1 : 0)
117 + (want_sss ? 1 : 0) + (want_unix ? 1 : 0)
118 + (want_krb5 ? 1 : 0) + (want_host ? 1 : 0)
119 + (want_pwd ? 1 : 0);
120 si[0] = 0;
121 si[1] = sec_count > 0 ? 0x01 : 0x00;
122 si[2] = (u_char) sec_count;
123 si[3] = 0;
124 {
125 u_char *pe = si + 4;
126 if (want_sss) {
127 pe[0] = 's'; pe[1] = 's'; pe[2] = 's'; pe[3] = ' ';
128 pe[4] = 0; pe[5] = 0; pe[6] = 0; pe[7] = 0;
129 pe += 8;
130 }
131 if (want_unix) {
132 pe[0] = 'u'; pe[1] = 'n'; pe[2] = 'i'; pe[3] = 'x';
133 pe[4] = 0; pe[5] = 0; pe[6] = 0; pe[7] = 0;
134 pe += 8;
135 }
136 if (want_krb5) {
137 pe[0] = 'k'; pe[1] = 'r'; pe[2] = 'b'; pe[3] = '5';
138 pe[4] = 0; pe[5] = 0; pe[6] = 0; pe[7] = 0;
139 pe += 8;
140 }
141 if (want_host) {
142 pe[0] = 'h'; pe[1] = 'o'; pe[2] = 's'; pe[3] = 't';
143 pe[4] = 0; pe[5] = 0; pe[6] = 0; pe[7] = 0;
144 pe += 8;
145 }
146 if (want_pwd) {
147 pe[0] = 'p'; pe[1] = 'w'; pe[2] = 'd'; pe[3] = ' ';
148 pe[4] = 0; pe[5] = 0; pe[6] = 0; pe[7] = 0;
149 pe += 8;
150 }
151 if (want_token) {
152 pe[0] = 'z'; pe[1] = 't'; pe[2] = 'n'; pe[3] = ' ';
153 pe[4] = 0; pe[5] = 0; pe[6] = 0; pe[7] = 0;
154 pe += 8;
155 }
156 if (want_gsi) {
157 pe[0] = 'g'; pe[1] = 's'; pe[2] = 'i'; pe[3] = ' ';
158 pe[4] = 0; pe[5] = 0; pe[6] = 0; pe[7] = 0;
159 pe += 8;
160 }
161
162 /* ServerResponseReqs_Protocol — signing level requirements.
163 * Always present when kXR_secreqs is set; seclvl=0 means none. */
164 pe[0] = 'S'; /* theTag */
165 pe[1] = 0; /* rsvd */
166 pe[2] = 0; /* secver (kXR_secver_0) */
167 pe[3] = (conf->security_level >= 4) ? kXR_secOData : 0; /* secopt */
168 pe[4] = (u_char) conf->security_level; /* seclvl */
169 pe[5] = 0; /* secvsz=0: no secvec */
170 }
171 }
172
173 ngx_log_debug4(NGX_LOG_DEBUG_STREAM, c->log, 0,
174 "brix: kXR_protocol ok (client_flags=0x%02xd "
175 "bodylen=%uz auth=%s tls=%d)",
176 (int) client_flags, bodylen,
177 want_sss ? "sss" :
178 want_unix ? "unix" :
179 want_krb5 ? "krb5" :
180 want_host ? "host" :
181 want_pwd ? "pwd" :
182 want_gsi && want_token ? "both" :
183 want_gsi ? "gsi" :
184 want_token ? "token" : "none",
185 offer_tls);
186
187 if (offer_tls) {
188 ctx->tls_pending = 1;
189 }
190
191 return brix_queue_response(ctx, c, buf, total);
192}
