19/* Codec-vs-protocol drift guard: the wire codec (shared libxrdproto, deliberately
20 * XProtocol-free) hard-codes the request body as XRDW_BODY_LEN bytes. This is the
21 * one translation unit that sees both that constant and the real XProtocol
22 * ClientRequestHdr, so it ties them together at compile time — if XRootD ever
23 * resized the body region, every xrdw_*_unpack() call here would read the wrong
24 * offsets, and this assert fails the build instead of corrupting requests. */
25_Static_assert(sizeof(((ClientRequestHdr *) 0)->body) == XRDW_BODY_LEN,
26 "wire codec body length must match XProtocol ClientRequestHdr.body");

28/*
29 * read_validate_req — decode the wire request and run the early-return checks.
30 *
31 * WHAT: unpacks the kXR_read body into *io, validates the file handle, serves
32 * the trivial rlen==0 case, clamps oversized requests and rejects negative
33 * offsets.
34 * WHY: every serve strategy needs the same validated (idx, fd, offset, rlen)
35 * tuple; hoisting the checks keeps the dispatcher a pure strategy selector.
36 * HOW: returns 1 when the request is valid and *io is filled (databuf NULL);
37 * returns 0 when the request was fully handled here (ok/error response already
38 * queued) with *rc set to the value the opcode handler must return.
39 */
40static ngx_flag_t
41read_validate_req(brix_ctx_t *ctx, ngx_connection_t *c,
42 brix_read_io_t *io, ngx_int_t *rc)
43{
44 xrdw_read_req_t req;
45
46 /*
47 * The shared codec decodes the big-endian wire body into host order; the file
48 * handle is a 4-byte blob but only byte 0 indexes our slot table
49 * (BRIX_MAX_FILES <= 256); the (unsigned char) cast prevents sign-extension
50 * of a high-bit handle byte into a negative idx.
51 */
52 xrdw_read_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
53 io->idx = (int) (unsigned char) req.fhandle[0];
54 io->offset = req.offset;
55 io->rlen = (size_t) (uint32_t) req.rlen;
56 io->databuf = NULL;
57
58 if (!brix_validate_read_handle(ctx, c, io->idx, "READ",
59 BRIX_OP_READ, rc)) {
60 return 0;
61 }
62
63 if (io->rlen == 0) {
64 BRIX_OP_OK(ctx, BRIX_OP_READ);
65 *rc = brix_send_ok(ctx, c, NULL, 0);
66 return 0;
67 }
68
69 if (io->rlen > BRIX_READ_REQUEST_MAX) {
70 io->rlen = BRIX_READ_REQUEST_MAX;
71 }
72
73 io->fd = ctx->files[io->idx].fd;
74
75 if (io->offset < 0) {
76 brix_log_access(ctx, c, "READ", ctx->files[io->idx].path, "-",
77 0, kXR_IOError, "negative read offset", 0);
78 BRIX_OP_ERR(ctx, BRIX_OP_READ);
79 *rc = brix_send_error(ctx, c, kXR_IOError, "negative read offset");
80 return 0;
81 }
82
83 return 1;
84}

86/*
87 * brix_handle_read — kXR_read dispatcher: validate, then pick a serve strategy.
88 *
89 * WHAT: routes a validated read to one of the serve paths — SSI/ZIP/codec
90 * early dispatch, zero-copy sendfile, windowed streaming, or the buffered
91 * memory path (warm-probe → AIO → synchronous fallback).
92 * WHY: each strategy has its own invariants (TLS => memory-backed buffers,
93 * cleartext/kTLS => sendfile; heap bounded by the streaming window); keeping
94 * the handler a flat early-return ladder makes the strategy choice auditable.
95 * HOW: read_validate_req() supplies the decoded request; every branch
96 * tail-calls its serve helper.
97 */
98ngx_int_t
99brix_handle_read(brix_ctx_t *ctx, ngx_connection_t *c)
100{
101 brix_read_io_t io;
102 ngx_stream_brix_srv_conf_t *rconf;
103 ngx_int_t rc;
104
105 if (!read_validate_req(ctx, c, &io, &rc)) {
106 return rc;
107 }
108
109 rconf = ngx_stream_get_module_srv_conf(
110 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
111
112 /* §7 XrdSsi: an SSI handle has no backing file — the first read dispatches the
113 * accumulated request to the service and serves the response. Early dispatch
114 * off the normal fd read path, like zip/slice below. */
115 if (ctx->files[io.idx].ssi != NULL) {
116 BRIX_OP_OK(ctx, BRIX_OP_READ);
117 return brix_ssi_read(ctx, c, io.idx, (uint64_t) io.offset,
118 (uint32_t) io.rlen);
119 }
120
121 /* Phase-57 W2: ZIP member handles translate the read into the archive's
122 * byte range (stored = offset add; deflate = stream inflate) — an early
123 * dispatch off the normal fd read path. */
124 if (ctx->files[io.idx].zip_mode) {
125 return brix_zip_read(ctx, c, io.idx, io.offset, io.rlen);
126 }
127
128 /*
129 * Phase-42 W4: inline read compression (opt-in, off by default). Routed to
130 * its own isolated synchronous handler so EVERYTHING below — the sendfile
131 * fast path, windowed streaming and AIO pipeline — stays byte-identical for
132 * the default (read_codec == 0 / BRIX_CODEC_IDENTITY) case. pgread/readv
133 * have their own handlers and never reach here, so their plaintext + CRC32c
134 * invariant is preserved.
135 */
136 if (ctx->files[io.idx].read_codec != 0) {
137 return brix_read_compressed(ctx, c, rconf, io.idx, (off_t) io.offset,
138 io.rlen);
139 }
140
141 /*
142 * Zero-copy sendfile fast path (gate in read_sendfile_eligible — the
143 * TLS-vs-cleartext INVARIANT lives there). Anything that fails the gate
144 * drops to the memory/window path below.
145 */
146 if (read_sendfile_eligible(ctx, c, io.idx)) {
147 return brix_read_serve_sendfile(ctx, c, rconf, &io);
148 }
149
150 /*
151 * Phase 31 W2.1: bound resident heap for large memory-backed reads. This
152 * is the memory path (TLS / non-regular file) — unlike the cleartext
153 * sendfile branch above it must buffer data in heap. Clamp the request to
154 * what the file actually holds (read-only handles have a cached size); if
155 * that exceeds one streaming window, serve the read as a sequence of
156 * window-sized kXR_oksofar chunks ending in kXR_ok, holding only ~one window
157 * in read_scratch at a time instead of the whole request. Writable handles
158 * (size unknown) use rlen and let a short read at EOF terminate early.
159 */
160 {
161 size_t total = read_clamped_total(ctx, &io);
162
163 if (total > (size_t) BRIX_READ_WINDOW) {
164 return read_serve_windowed(ctx, c, rconf, &io, total);
165 }
166 }
167
168 /*
169 * Small memory read (<= one window): single-shot. Admit the full rlen and
170 * buffer it in read_scratch — bounded by the window, so no streaming needed.
171 */
172 return read_serve_buffered(ctx, c, rconf, &io);
173}
