17/* Codec-vs-protocol drift guard: the wire codec (shared libxrdproto, deliberately
18 * XProtocol-free) hard-codes the request body as XRDW_BODY_LEN bytes. This is the
19 * one translation unit that sees both that constant and the real XProtocol
20 * ClientRequestHdr, so it ties them together at compile time — if XRootD ever
21 * resized the body region, every xrdw_*_unpack() call here would read the wrong
22 * offsets, and this assert fails the build instead of corrupting requests. */
23_Static_assert(sizeof(((ClientRequestHdr *) 0)->body) == XRDW_BODY_LEN,
24 "wire codec body length must match XProtocol ClientRequestHdr.body");

26/*
27 * brix_ktls_send_active — true when kernel-TLS transmit is active on this
28 * connection (Phase 29 kTLS).
29 *
30 * Without kTLS, a TLS data stream must encrypt in userspace and therefore cannot
31 * use sendfile(2) — the historical reason the read path gates the zero-copy
32 * sendfile branch on !c->ssl. When the kernel TLS ULP is negotiated for the send
33 * side (OpenSSL SSL_OP_ENABLE_KTLS + a kTLS-offloadable cipher), the kernel does
34 * the record encryption inside sendfile, so a file-backed chain is legal over TLS
35 * and the read inherits the cleartext sendfile fast path (and its Phase-2
36 * pipelining). Returns 0 whenever kTLS is unavailable or not negotiated, so the
37 * caller transparently falls back to the memory/window path — the relaxation is
38 * always safe.
39 */
40static ngx_flag_t
41brix_ktls_send_active(ngx_connection_t *c)
42{
43#ifdef BIO_get_ktls_send
44 if (c->ssl != NULL && c->ssl->connection != NULL) {
45 return BIO_get_ktls_send(SSL_get_wbio(c->ssl->connection)) > 0 ? 1 : 0;
46 }
47#endif
48 return 0;
49}

51/* Zero-copy sendfile serve path for a regular-file cleartext (or kTLS) read:
52 * clamps the chunk to EOF, charges bytes/bandwidth/dashboard + access log,
53 * builds the sendfile chain and queues it. Always completes the request --
54 * the caller tail-calls this under the is_regular && (!ssl || kTLS) gate. */
55static ngx_int_t
56brix_read_serve_sendfile(brix_ctx_t *ctx, ngx_connection_t *c,
57 ngx_stream_brix_srv_conf_t *rconf, int idx, ngx_fd_t fd,
58 int64_t offset, size_t rlen)
59{
60 size_t data_total;
61 u_char *send_base = NULL;
62 ngx_chain_t *rsp_chain;
63
64 off_t file_size;
65 off_t avail;
66
67 /*
68 * Read-only handles: file size is stable, use the value cached at open
69 * time to skip the fstat(2) syscall on every chunk request.
70 * Writable handles (kXR_open_updt): re-stat so a write on the same
71 * session is visible to subsequent reads.
72 */
73 if (!ctx->files[idx].writable) {
74 file_size = ctx->files[idx].cached_size;
75 } else {
76 struct stat st;
77 if (fstat(fd, &st) != 0) {
78 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
79 ctx->files[idx].path, "-",
80 kXR_IOError, strerror(errno));
81 }
82 file_size = st.st_size;
83 }
84
85 /*
86 * Clamp the chunk to bytes actually present: sendfile would otherwise be
87 * asked for data past EOF. offset at/after EOF yields a zero-length OK
88 * (legal short read); otherwise serve min(rlen, remaining-to-EOF). This
89 * is also why data_total — not the client's requested rlen — drives every
90 * accounting counter below.
91 */
92 if ((off_t) offset >= file_size) {
93 data_total = 0;
94 } else {
95 avail = file_size - (off_t) offset;
96 data_total = (avail < (off_t) rlen) ? (size_t) avail : rlen;
97 }
98
99 brix_prefetch_read_file(c->log, &ctx->files[idx], (off_t) offset,
100 data_total, file_size);
101
102 ctx->files[idx].bytes_read += data_total;
103 ctx->totals.bytes += data_total;
104 brix_rl_charge_ctx(ctx, data_total); /* Phase 25 bandwidth */
105
106 /* Per-backend storage byte totals: this zero-copy branch never reaches
107 * brix_vfs_io_execute (the kernel moves the bytes), so attribute here.
108 * The branch gate (sd_obj.driver == NULL) means the backend is always the
109 * default POSIX driver. Buffered/driver-backed reads attribute at the
110 * io_execute seam instead — no double count. */
111 brix_metric_backend_bytes("posix", BRIX_METRIC_OP_READ, data_total);
112
113 if (ctx->files[idx].dashboard_slot >= 0 &&
114 ngx_brix_dashboard_shm_zone != NULL)
115 {
116 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
117 ctx->files[idx].dashboard_slot,
118 (ngx_atomic_int_t) data_total,
119 (int64_t) ngx_current_msec);
120 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
121 ctx->files[idx].dashboard_slot,
122 "read");
123 }
124
125 if (rconf->access_log_fd != NGX_INVALID_FILE) {
126 char read_detail[64];
127
128 snprintf(read_detail, sizeof(read_detail), "%lld+%zu",
129 (long long) offset, rlen);
130 brix_log_access(ctx, c, "READ", ctx->files[idx].path,
131 read_detail, 1, 0, NULL, data_total);
132 }
133 BRIX_OP_OK(ctx, BRIX_OP_READ);
134
135 rsp_chain = brix_build_sendfile_chain(ctx, c, fd,
136 ctx->files[idx].path,
137 (off_t) offset, data_total,
138 &send_base);
139 if (rsp_chain == NULL) {
140 brix_release_read_buffer(ctx, c, send_base);
141 return NGX_ERROR;
142 }
143
144 {
145 ngx_int_t rc = brix_queue_response_chain(ctx, c, rsp_chain,
146 send_base);
147
148 if (rc != NGX_OK || ctx->state != XRD_ST_SENDING) {
149 brix_release_read_buffer(ctx, c, send_base);
150 }
151 return rc;
152 }
153}

159 int idx;

161 size_t rlen;
162 u_char *databuf;
163 ssize_t nread;
164 size_t data_total;

170 /*
171 * The shared codec decodes the big-endian wire body into host order; the file
172 * handle is a 4-byte blob but only byte 0 indexes our slot table
173 * (BRIX_MAX_FILES <= 256); the (unsigned char) cast prevents sign-extension
174 * of a high-bit handle byte into a negative idx.
175 */
176 xrdw_read_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
177 idx = (int) (unsigned char) req.fhandle[0];
178 offset = req.offset;
179 rlen = (size_t) (uint32_t) req.rlen;
180
181 if (!brix_validate_read_handle(ctx, c, idx, "READ",
182 BRIX_OP_READ, &rc)) {
183 return rc;
184 }
185
186 if (rlen == 0) {
187 BRIX_OP_OK(ctx, BRIX_OP_READ);
188 return brix_send_ok(ctx, c, NULL, 0);
189 }
190
191 if (rlen > BRIX_READ_REQUEST_MAX) {
192 rlen = BRIX_READ_REQUEST_MAX;
193 }
194
195 fd = ctx->files[idx].fd;
196
197 if (offset < 0) {
198 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
199 ctx->files[idx].path, "-",
200 kXR_IOError, "negative read offset");
201 }
202
203 rconf = ngx_stream_get_module_srv_conf(
204 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
205
206 /* §7 XrdSsi: an SSI handle has no backing file — the first read dispatches the
207 * accumulated request to the service and serves the response. Early dispatch
208 * off the normal fd read path, like zip/slice below. */
209 if (ctx->files[idx].ssi != NULL) {
210 BRIX_OP_OK(ctx, BRIX_OP_READ);
211 return brix_ssi_read(ctx, c, idx, (uint64_t) offset, (uint32_t) rlen);
212 }
213
214 /* Phase-57 W2: ZIP member handles translate the read into the archive's
215 * byte range (stored = offset add; deflate = stream inflate) — an early
216 * dispatch off the normal fd read path. */
217 if (ctx->files[idx].zip_mode) {
218 return brix_zip_read(ctx, c, idx, offset, rlen);
219 }
220
221 /*
222 * Phase-42 W4: inline read compression (opt-in, off by default). Routed to
223 * its own isolated synchronous handler so EVERYTHING below — the sendfile
224 * fast path, windowed streaming and AIO pipeline — stays byte-identical for
225 * the default (read_codec == 0 / BRIX_CODEC_IDENTITY) case. pgread/readv
226 * have their own handlers and never reach here, so their plaintext + CRC32c
227 * invariant is preserved.
228 */
229 if (ctx->files[idx].read_codec != 0) {
230 return brix_read_compressed(ctx, c, rconf, idx, (off_t) offset, rlen);
231 }
232
233 /*
234 * Zero-copy sendfile fast path. Two conditions must both hold:
235 * - is_regular: sendfile(2) only works against a real file, not a pipe/dir.
236 * - !c->ssl OR kTLS active: a userspace-TLS stream cannot sendfile because
237 * nginx must encrypt each record in user memory (INVARIANT: TLS =>
238 * memory-backed buffers). kTLS lifts that — the kernel encrypts inside
239 * sendfile — so a TLS connection with kTLS negotiated rejoins this branch.
240 * Anything that fails the gate (TLS without kTLS, irregular file) drops to the
241 * memory/window path below.
242 */
243 if (ctx->files[idx].is_regular
244 && (!c->ssl || brix_ktls_send_active(c))
245 && ctx->files[idx].csi == NULL /* phase-59 W2/ADR-6: CSI needs the
246 * bytes in memory to verify, so an
247 * integrity-checked handle takes the
248 * buffered path, not zero-copy sendfile */
249 && ctx->files[idx].sd_obj.driver == NULL) /* Layer 3: a driver-backed
250 * handle's bare fd is only block 0 — a
251 * sendfile over it cannot span striped
252 * blocks, so serve via the buffered
253 * io_core path (driver preadv) instead */
254 {
255 return brix_read_serve_sendfile(ctx, c, rconf, idx, fd,
256 offset, rlen);
257 }

259 /*
260 * Phase 31 W2.1: bound resident heap for large memory-backed reads. This
261 * is the memory path (TLS / non-regular file) — unlike the cleartext
262 * sendfile branch above it must buffer data in heap. Clamp the request to
263 * what the file actually holds (read-only handles have a cached size); if
264 * that exceeds one streaming window, serve the read as a sequence of
265 * window-sized kXR_oksofar chunks ending in kXR_ok, holding only ~one window
266 * in read_scratch at a time instead of the whole request. Writable handles
267 * (size unknown) use rlen and let a short read at EOF terminate early.
268 */
269 {
270 size_t total = rlen;
271
272 if (!ctx->files[idx].writable && ctx->files[idx].cached_size > 0) {
273 off_t avail = ctx->files[idx].cached_size - (off_t) offset;
274 total = (avail <= 0) ? 0
275 : ((off_t) total > avail ? (size_t) avail : total);
276 }
277
278 if (total > (size_t) BRIX_READ_WINDOW) {
279 /* Admit one window's worth — a windowed stream holds ~2 MiB, not
280 * the full request, so many more fit under the budget. */
281 if (!brix_budget_admit(ctx, rconf->memory_budget,
282 (size_t) BRIX_READ_WINDOW)) {
283 return brix_send_wait(ctx, c, 1);
284 }
285
286 /*
287 * Arm the windowed-read state machine: the pump below (and any
288 * resumption after a partial flush) reads from rd_win_* rather than
289 * this request's locals, so it survives across event-loop returns.
290 * cur_streamid is snapshotted into rd_win_streamid because each
291 * kXR_oksofar/kXR_ok chunk must echo the originating request's stream
292 * id, but cur_streamid will be overwritten by the next inbound header
293 * before this stream finishes draining.
294 */
295 ctx->rd.win_active = 1;
296 ctx->rd.win_fd = fd;
297 ctx->rd.win_idx = idx;
298 ctx->rd.win_offset = (off_t) offset;
299 ctx->rd.win_remaining = total;
300 ctx->rd.win_streamid[0] = ctx->recv.cur_streamid[0];
301 ctx->rd.win_streamid[1] = ctx->recv.cur_streamid[1];
302
303 brix_prefetch_read_file(c->log, &ctx->files[idx], (off_t) offset,
304 total,
305 ctx->files[idx].writable
306 ? 0 : ctx->files[idx].cached_size);
307
308 if (rconf->access_log_fd != NGX_INVALID_FILE) {
309 char read_detail[64];
310 snprintf(read_detail, sizeof(read_detail), "%lld+%zu",
311 (long long) offset, rlen);
312 brix_log_access(ctx, c, "READ", ctx->files[idx].path,
313 read_detail, 1, 0, NULL, total);
314 }
315
316 brix_read_window_pump(ctx, c, rconf);
317 return NGX_OK;
318 }
319 }
320
321 /*
322 * Small memory read (<= one window): single-shot. Admit the full rlen and
323 * buffer it in read_scratch — bounded by the window, so no streaming needed.
324 */
325 if (!brix_budget_admit(ctx, rconf->memory_budget, rlen)) {
326 return brix_send_wait(ctx, c, 1);
327 }
328
329 /*
330 * Per-in-flight read buffer (read pipelining): each outstanding memory read
331 * gets its OWN buffer from rd_pool rather than the single shared read_scratch,
332 * so this response can keep draining the (possibly jittered) socket while the
333 * recv loop already issues the next read into a different buffer. Released
334 * back to the pool when this response's out_ring slot drains.
335 */
336 databuf = brix_acquire_read_buffer(ctx, c, rlen);
337 if (databuf == NULL) {
338 return NGX_ERROR;
339 }
340
341 /* Charge the (possibly grown) read-pool footprint to the budget now so a
342 * concurrent connection's admission check sees this allocation promptly. */
343 brix_budget_sync(ctx);
344
345 if (ctx->files[idx].is_regular) {
346 off_t file_size;
347 size_t hint_len;
348
349 file_size = ctx->files[idx].writable ? 0
350 : ctx->files[idx].cached_size;
351 hint_len = rlen;
352
353 if (file_size > 0) {
354 if ((off_t) offset >= file_size) {
355 hint_len = 0;
356 } else if ((off_t) hint_len > file_size - (off_t) offset) {
357 hint_len = (size_t) (file_size - (off_t) offset);
358 }
359 }
360
361 brix_prefetch_read_file(c->log, &ctx->files[idx], (off_t) offset,
362 hint_len, file_size);
363 }

365 /*
366 * Phase 32 WS4: warm-cache fast path. Probe the page cache with a
367 * non-blocking preadv2(RWF_NOWAIT). If the whole request is resident it
368 * returns rlen bytes immediately and we complete inline below — skipping the
369 * thread-pool round-trip (hundreds of µs) that otherwise dominates a
370 * cache-hot read. A short or EAGAIN result is a (partial) cache miss: fall
371 * through to the AIO thread (or a synchronous pread if no pool), which reads
372 * the full data, blocking off the event loop. Only attempted for regular
373 * files (RWF_NOWAIT is meaningful against the page cache).
374 */
375 {
376 ssize_t warm = -1;
377#if defined(RWF_NOWAIT)
378 if (rconf->common.thread_pool != NULL && ctx->files[idx].is_regular) {
379 struct iovec iov;
380 brix_sd_obj_t obj;
381 iov.iov_base = databuf;
382 iov.iov_len = rlen;
383 brix_sd_posix_wrap(&obj, fd); /* phase-55: SD seam */
384 warm = obj.driver->preadv2(&obj, &iov, 1,
385 (off_t) offset, RWF_NOWAIT);
386 }
387#endif
388
389 /*
390 * Only an exact rlen match counts as a hit: a short warm result means
391 * part of the range was not resident (or EOF), and re-issuing a blocking
392 * read for the missing tail from the event loop would stall it — so any
393 * non-exact result falls through to the thread pool / sync path, which
394 * re-reads the full range from offset (the partial warm bytes are simply
395 * overwritten).
396 */
397 if (warm == (ssize_t) rlen) {
398 nread = warm; /* full page-cache hit — databuf is filled; complete inline */
399
400 /* phase-59 W2: the warm fast path bypasses the VFS job, so verify
401 * the page CRCs here too; a mismatch fails the read (EIO). */
402 if (ctx->files[idx].csi != NULL && nread > 0
403 && brix_csi_verify_read(
404 (brix_csi_t *) ctx->files[idx].csi, databuf,
405 (off_t) offset, (size_t) nread) == BRIX_CSI_MISMATCH)
406 {
407 nread = -1;
408 errno = EIO;
409 }
410
411 /* The warm fast path bypasses brix_vfs_io_execute (where the other
412 * read paths attribute), so charge the per-backend read total here. */
413 if (nread > 0) {
414 brix_metric_backend_bytes(
415 ctx->files[idx].sd_obj.driver != NULL
416 ? ctx->files[idx].sd_obj.driver->name : "posix",
417 BRIX_METRIC_OP_READ, (size_t) nread);
418 }
419
420 } else if (rconf->common.thread_pool != NULL) {
421 ngx_thread_task_t *task;
422 brix_read_aio_t *t;
423 ngx_flag_t posted;
424

427 * first time, otherwise reset the two fields ngx reuse requires —
428 * unlink from any prior queue (next) and clear the completion flag so
429 * the event loop will fire the done-callback again.

436 return NGX_ERROR;
437 }
438 ctx->rd.read_aio_task = task;
439 } else {

441 task->event.complete = 0;
442 }
443

445 t->c = c;
446 t->ctx = ctx;
447 t->fd = fd;
448 t->handle_idx = idx;
449 t->offset = (off_t) offset;
450 t->rlen = rlen;
451 t->databuf = databuf;
452 t->streamid[0] = ctx->recv.cur_streamid[0];
453 t->streamid[1] = ctx->recv.cur_streamid[1];
454 t->nread = 0;
455 t->io_errno = 0;
456 t->csi = ctx->files[idx].csi; /* phase-59 W2: verify on read */
457 t->obj = ctx->files[idx].sd_obj; /* Layer 3: driver obj (or zeroed) */
458
459 brix_task_bind(task, brix_read_aio_thread, brix_read_aio_done);
460
461 (void) brix_aio_post_task(ctx, c, rconf->common.thread_pool, task,
462 "brix: thread_task_post failed, sync read fallback",
463 &posted);
464 /*
465 * Posted: the read now completes off-thread; the done-callback owns
466 * databuf and finishes the response, so return early — nothing more to
467 * do on the event loop. Not posted (queue full / post error): fall
468 * through to a blocking pread here so the read never silently drops.
469 */
470 if (posted) {
471 return NGX_OK;
472 }
473 {
474 brix_vfs_job_t job;
475
476 brix_vfs_job_read_init(&job, fd, (off_t) offset, rlen,
477 databuf, rlen, 0);
478 job.csi = ctx->files[idx].csi; /* phase-59 W2: verify on read */
479 brix_vfs_job_set_obj(&job, &ctx->files[idx].sd_obj);
480 brix_vfs_io_execute(&job);
481 nread = job.nio;
482 if (job.io_errno != 0) {
483 /* A CSI page-checksum mismatch surfaces here as EIO
484 * (job.csi_mismatch set); the existing nread<0 path fails
485 * the read so corrupt data is never served. */
486 errno = job.io_errno;
487 }
488 }

490 } else {
491 brix_vfs_job_t job;
492
493 /* No thread pool configured: read inline on the event loop. */
494 brix_vfs_job_read_init(&job, fd, (off_t) offset, rlen,
495 databuf, rlen, 0);
496 job.csi = ctx->files[idx].csi; /* phase-59 W2: verify on read */
497 brix_vfs_job_set_obj(&job, &ctx->files[idx].sd_obj);
498 brix_vfs_io_execute(&job);
499 nread = job.nio;
500 if (job.io_errno != 0) {
501 /* CSI mismatch surfaces as EIO here; the nread<0 path below
502 * fails the read so corrupt data is never served. */
503 errno = job.io_errno;
504 }
505 }
506 }
507 if (nread < 0) {
508 brix_release_read_buffer(ctx, c, databuf);
509 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
510 ctx->files[idx].path, "-",
511 kXR_IOError, strerror(errno));
512 }
513
514 data_total = (size_t) nread;
515
516 ctx->files[idx].bytes_read += data_total;
517 ctx->totals.bytes += data_total;
518
519 if (ctx->files[idx].dashboard_slot >= 0 &&
520 ngx_brix_dashboard_shm_zone != NULL)
521 {
522 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
523 ctx->files[idx].dashboard_slot,
524 (ngx_atomic_int_t) data_total,
525 (int64_t) ngx_current_msec);
526 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
527 ctx->files[idx].dashboard_slot, "read");
528 }

534 (long long) offset, rlen);

544 }

550 brix_release_read_buffer(ctx, c, databuf);
551 } else {

555 * queue the next read behind it instead of idling while it drains a
556 * jittered socket. (A single-chunk response only: the non-windowed
557 * path is bounded by BRIX_READ_WINDOW < BRIX_READ_CHUNK_MAX.)
558 */
559 ctx->out.resp_pipelinable = 1;
560 }
561 return rc;
