6/*
7 * protocol_caps.h — shared HTTP operation capability registry.
8 *
9 * Each protocol module (WebDAV, S3) keeps its own static operation table.
10 * This header defines the shared descriptor shape and the two helper functions
11 * that query and format those tables.
12 *
13 * Protocol-specific concerns (metric slots, auth decisions, dispatch routing)
14 * stay in the protocol modules. These helpers only answer:
15 * - which operation does this request map to?
16 * - what is the correct Allow / Access-Control-Allow-Methods string for the
17 * enabled set of operations?
18 */
19
20/* Capability flags — bitfield stored in brix_http_operation_t.flags. */
21typedef enum {
22 BRIX_PROTO_OP_READ = 0x0001, /* safe, idempotent reads */
23 BRIX_PROTO_OP_WRITE = 0x0002, /* mutating operations */
24 BRIX_PROTO_OP_LIST = 0x0004, /* directory/bucket listing */
25 BRIX_PROTO_OP_TPC = 0x0008, /* third-party copy */
26 BRIX_PROTO_OP_LOCK = 0x0010, /* locking (LOCK/UNLOCK) */
27 BRIX_PROTO_OP_ASYNC_BODY = 0x0020 /* reads body asynchronously */
28} brix_proto_op_flags_t;

57/*
58 * brix_http_operation_allow_header — build an Allow header value string.
59 *
60 * Collects the names of all operations in ops[0..nops) whose flags field has
61 * at least one bit in common with enabled_flags, then writes a
62 * comma-separated list into *out using pool allocation.
63 *
64 * Pass enabled_flags = ~0u to include every operation in the table.
65 * Pass enabled_flags = BRIX_PROTO_OP_READ|BRIX_PROTO_OP_LIST for a
66 * read-only subset.
67 *
68 * Returns NGX_OK on success, NGX_ERROR on pool allocation failure.
69 */
70ngx_int_t brix_http_operation_allow_header(ngx_pool_t *pool,
71 const brix_http_operation_t *ops, ngx_uint_t nops,
72 ngx_uint_t enabled_flags, ngx_str_t *out);
