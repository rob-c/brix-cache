1/*
2 * stream.c — Prometheus exporter for native XRootD (stream-layer) counters.
3 *
4 * WHAT: Maps numeric operation slots to human-readable label strings
5 * (`brix_op_names[]`) and iterates the shared-memory metrics zone
6 * to emit Prometheus exposition-format lines for all stream-protocol
7 * counters — connections, bytes, wire frames, request/reject stats,
8 * per-operation ok/error counts, path-depth violations, registry full.
9 *
10 * WHY: The stream module writes counters into `ngx_brix_metrics_t` shared
11 * memory using atomic fields. This file is the HTTP-side exporter that
12 * reads those counters and formats them as Prometheus scrape output.
13 * The op-name mapping is a binary ABI between stream and HTTP modules:
14 * slot indices must stay aligned with `BRIX_OP_*` constants in metrics.h.
15 *
16 * HOW: `brix_export_prometheus_metrics()` iterates all server slots, reads
17 * each atomic counter via `ngx_atomic_fetch_add(..., 0)` for an
18 * eventually-consistent snapshot, and writes HELP/TYPE/value lines via
19 * the `metrics_writer_t` interface. Per-server families are described by
20 * `srv_family_desc_t` tables (one table per metric family) fed through a
21 * single slot-scan emitter, so the exporter itself is a flat call
22 * sequence. Calls protocol-specific exporters (webdav, s3, proxy,
23 * tracking) at the end.
24 */

77/*
78 * metrics_emit_registry_health() — registry/session anti-exhaustion scalars.
79 *
80 * WHAT: Server registrations dropped at registry capacity, plus the
81 * Phase 27 F4 session-registry pressure counters (logins rejected with
82 * a full table, idle sessions LRU-reaped to admit a new login).
83 * WHY: Non-zero values mean the fixed-slot tables are undersized for the
84 * deployment.
85 * HOW: Zone-global unlabelled counters, verbatim exposition text.
86 */
87static void
88metrics_emit_registry_health(metrics_writer_t *mw, ngx_brix_metrics_t *shm)
89{
90 mw_printf(mw,
91 "# HELP brix_registry_full_total "
92 "Server registrations dropped because the registry was at capacity.\n"
93 "# TYPE brix_registry_full_total counter\n"
94 "brix_registry_full_total %lu\n",
95 (unsigned long) ngx_atomic_fetch_add(&shm->registry_full_total, 0));
96
97 mw_printf(mw,
98 "# HELP brix_session_registry_full_total "
99 "Logins rejected because the session table was full and nothing was reapable.\n"
100 "# TYPE brix_session_registry_full_total counter\n"
101 "brix_session_registry_full_total %lu\n"
102 "# HELP brix_session_evict_total "
103 "Idle sessions reaped (LRU) to admit a new login under table pressure.\n"
104 "# TYPE brix_session_evict_total counter\n"
105 "brix_session_evict_total %lu\n",
106 (unsigned long) ngx_atomic_fetch_add(&shm->session_registry_full_total, 0),
107 (unsigned long) ngx_atomic_fetch_add(&shm->session_evict_total, 0));

110/*
111 * metrics_emit_request_ops() — per-operation request outcome family.
112 *
113 * WHAT: `brix_requests_total{port,auth,op,status}` — ok counts always, error
114 * counts only for op slots that have recorded at least one error.
115 * WHY: The primary per-operation traffic/error breakdown; suppressing
116 * zero-error rows keeps the scrape small.
117 * HOW: Outer loop over BRIX_NOPS name slots (NULL = unused trailing slot),
118 * inner loop over in-use server slots.
119 */
120static void
121metrics_emit_request_ops(metrics_writer_t *mw, ngx_brix_metrics_t *shm)
122{
123 ngx_brix_srv_metrics_t *srv;
124 ngx_uint_t i, op;
125 char port_str[16];
126
127 mw_printf(mw,
128 "# HELP brix_requests_total "
129 "XRootD requests completed, by operation and status.\n"
130 "# TYPE brix_requests_total counter\n");
131 for (op = 0; op < BRIX_NOPS; op++) {
132 if (brix_op_names[op] == NULL) {
133 continue; /* unused trailing slot (NOPS > distinct op count) */
134 }
135 for (i = 0; i < BRIX_METRICS_MAX_SERVERS; i++) {
136 srv = &shm->servers[i];
137 if (!srv->in_use) { continue; }
138 ngx_snprintf((u_char *) port_str, sizeof(port_str),
139 "%ui%Z", srv->port);
140
141 mw_printf(mw,
142 "brix_requests_total"
143 "{port=\"%s\",auth=\"%s\",op=\"%s\",status=\"ok\"}"
144 " %lu\n",
145 port_str, srv->auth, brix_op_names[op],
146 (unsigned long) ngx_atomic_fetch_add(&srv->op_ok[op], 0));
147
148 {
149 ngx_atomic_t errs = ngx_atomic_fetch_add(&srv->op_err[op], 0);
150 if (errs > 0) {
151 mw_printf(mw,
152 "brix_requests_total"
153 "{port=\"%s\",auth=\"%s\",op=\"%s\",status=\"error\"}"
154 " %lu\n",
155 port_str, srv->auth, brix_op_names[op],
156 (unsigned long) errs);
157 }
158 }
159 }
160 }
161}

163/*
164 * metrics_emit_mirror() — traffic-mirror counters.
165 *
166 * WHAT: Phase 24 — mirror request/error/dropped/divergence totals for both
167 * the http and stream surfaces.
168 * WHY: Always exported (independent of the cluster registry, unlike the
169 * health-check block in cluster.c) so shadow behaviour is graphable
170 * even before any node registers.
171 * HOW: Zone-global counters, one verbatim exposition block.
