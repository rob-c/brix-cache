1/*
2 * storascan_core.h — pure, side-effect-free core for xrdstorascan.
3 *
4 * WHAT: the protocol-free arithmetic + decision logic behind the client-side
5 * `bench` (B1) and `verify --wire` (A1) modes — latency percentiles,
6 * throughput/IOPS, and the checksum-comparison verdict.
7 * WHY: isolating the math from the network I/O makes both modes unit-testable
8 * standalone (no server, no libbrix), mirroring the repo's
9 * *_unittest.c pattern.
10 * HOW: pure functions over caller-owned buffers; no allocation that outlives a
11 * call, no globals.
12 */
