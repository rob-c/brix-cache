1/*
2 * frame.c — request/response framing on the wire.
3 *
4 * WHAT: Finalize+send a 24-byte ClientRequestHdr (+ optional payload), and read
5 * one 8-byte ServerResponseHdr (+ body), interpreting the status field.
6 * WHY: Every opcode shares this framing (wire_core_requests.h); centralising it
7 * keeps each op in ops_*.c tiny and the byte-twiddling in one audited place.
8 * HOW: streamid is any 2 bytes echoed back by the server; we assign a per-conn
9 * counter so future parallel/pipelined requests can be matched by it. All
10 * multi-byte header fields are big-endian.
11 *
12 * wire: XProtocol.hh ClientRequestHdr — streamid[2] reqid[2] body[16] dlen[4];
13 * wire: XProtocol.hh ServerResponseHdr — streamid[2] status[2] dlen[4].
14 */

26int
27brix_send_ext(brix_conn *c, void *hdr24, const brix_payload_ext *pl,
28 uint16_t *out_sid, brix_status *st)
29{
30 const void *payload = (pl != NULL) ? pl->data : NULL;
31 uint32_t send_len = (pl != NULL) ? pl->len : 0;
32 uint32_t dlen = (pl != NULL) ? pl->dlen : 0;
33
34 uint8_t *h = (uint8_t *) hdr24;
35 uint16_t sid = c->next_sid++;
36 uint32_t be = htonl(dlen);
37
38 h[0] = (uint8_t) (sid >> 8);
39 h[1] = (uint8_t) (sid & 0xff);
40 memcpy(h + 20, &be, 4); /* dlen at offset 20 */
41
42 if (out_sid != NULL) {
43 *out_sid = sid;
44 }
45
46 /* §15: remember the in-flight requestid (for trace + timing), trace the
47 * request, and stamp the send time. All inert unless armed. */
48 c->diag.inflight_reqid = xrd_get_u16_be(h + 2);
49 if (c->diag.wire_trace) {
50 brix_trace_frame(c, '>', sid, c->diag.inflight_reqid, 1, dlen,
51 payload, send_len);
52 }
53 if (c->diag.cap != NULL) { /* §15.1: record the full request wire bytes */
54 brix_capture_frame(c->diag.cap, '>', sid, c->diag.inflight_reqid, 1,
55 h, XRD_REQUEST_HDR_LEN, payload, send_len);
56 }
57 if (c->diag.timing) {
58 c->diag.t_send_ns = brix_mono_ns();
59 }
60
61 /* When GSI signing is active and the server's security level requires it,
62 * prepend a kXR_sigver frame covering this request (no-op otherwise).
63 * The signature covers the dlen-framed payload — for kXR_writev that is
64 * the descriptor block only, matching what the server hashes. */
65 if (brix_sigver_maybe(c, h, payload, dlen, st) != 0) {
66 return -1;
67 }
68
69 if (brix_write_full(&c->io, h, XRD_REQUEST_HDR_LEN, st) != 0) {
70 return -1;
71 }
72 if (send_len > 0 && payload != NULL) {
73 if (brix_write_full(&c->io, payload, send_len, st) != 0) {
74 return -1;
75 }
76 }
77 return 0;
78}

90/* One decoded server frame: unpacked ServerResponseHdr fields + the malloc'd
91 * body (NULL when dlen == 0; the consumer frees or hands it off). Bundles the
92 * four values every receiver threads around so helpers stay under the
93 * 5-parameter gate without changing any wire semantics. */
94typedef struct {
95 uint16_t sid; /* outer streamid */
96 uint16_t stat; /* outer status */
97 uint32_t dlen; /* body length */
98 uint8_t *buf; /* malloc'd body, or NULL */
99} rx_frame_t;
100
101/* The caller-provided response out-params travel as a brix_resp_out (promoted
102 * to brix_net.h in phase-73) so the delivery helpers shared by brix_recv /
103 * recv_after_waitresp take one arg. */
104
105/*
106 * WHAT: read + unpack one 8-byte ServerResponseHdr into f, with the standard
107 * size cap; the raw header bytes land in hdr (for wire capture).
108 * WHY: shared by every receiver (brix_recv + the raw waitresp reader) so the
109 * cap check and unaligned-safe unpack live in exactly one place.
110 * HOW: brix_read_full for the 8 bytes, xrd_resp_hdr_unpack, cap against
111 * XRDC_DLEN_MAX. f->buf is reset to NULL. 0 / -1 (st set).
112 */
113static int
114frame_read_header(brix_conn *c, uint8_t *hdr, rx_frame_t *f, brix_status *st)
115{
116 f->buf = NULL;
117 if (brix_read_full(&c->io, hdr, XRD_RESPONSE_HDR_LEN, st) != 0) {
118 return -1;
119 }
120 xrd_resp_hdr_unpack(hdr, &f->sid, &f->stat, &f->dlen); /* unaligned-safe */
121 if (f->dlen > XRDC_DLEN_MAX) {
122 brix_status_set(st, XRDC_EPROTO, 0,
123 "response body too large (%u bytes)", f->dlen);
124 return -1;
125 }
126 return 0;
127}

129/*
130 * WHAT: allocate + read the dlen-byte frame body into f->buf.
131 * WHY: the malloc/read/cleanup triple was duplicated verbatim in brix_recv
132 * and recv_raw_frame; one copy keeps the OOM + short-read handling audited.
133 * HOW: no-op for dlen == 0 (buf stays NULL); on a short read the buffer is
134 * freed and NULLed so callers never see a half-filled body. 0 / -1.
135 */
136static int
137frame_read_body(brix_conn *c, rx_frame_t *f, brix_status *st)
138{
139 f->buf = NULL;
140 if (f->dlen == 0) {
141 return 0;
142 }
143 f->buf = (uint8_t *) malloc(f->dlen);
144 if (f->buf == NULL) {
145 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (%u bytes)", f->dlen);
146 return -1;
147 }
148 if (brix_read_full(&c->io, f->buf, f->dlen, st) != 0) {
149 free(f->buf);
150 f->buf = NULL;
151 return -1;
152 }
153 return 0;
154}

156/*
157 * WHAT: format a kXR_error body (errnum[4 BE] + message) into st.
158 * WHY: identical decoding in the synchronous (brix_recv) and asynresp
159 * (recv_after_waitresp) error paths; sharing it keeps the %.*s bound
160 * (wire message is NOT NUL-terminated) in one audited spot.
161 * HOW: tolerates short bodies (errnum 0 / empty message) exactly as before;
162 * the caller still owns + frees the frame buffer AFTER this returns
163 * (the message is formatted out of the live buffer).
164 */
165static void
166set_kxr_error_status(const uint8_t *data, uint32_t dlen, brix_status *st)
167{
168 int errnum = (dlen >= 4) ? (int) xrd_get_u32_be(data) : 0;
169 int mlen = (dlen > 4) ? (int) (dlen - 4) : 0;
170
171 /* The wire message is NOT NUL-terminated; bound %s with %.*s so a
172 * hostile server can't drive a heap over-read past the frame. */
173 brix_status_set(st, errnum, 0, "%.*s (%s)", mlen,
174 (const char *) (data + 4), brix_kxr_name(errnum));
175}

177/* Read one full server frame (header + body) with the standard size cap; fills
178 * f (caller frees f->buf). No streamid matching — used by the kXR_waitresp
179 * async path, where the deferred reply arrives as an unsolicited frame whose
180 * outer streamid may differ from the request's. 0 / -1. */
181static int
182recv_raw_frame(brix_conn *c, rx_frame_t *f, brix_status *st)
183{
184 uint8_t hdr[XRD_RESPONSE_HDR_LEN];
185
186 if (frame_read_header(c, hdr, f, st) != 0) {
187 return -1;
188 }
189 if (frame_read_body(c, f, st) != 0) {
190 return -1;
191 }
192 if (c->diag.wire_trace) {
193 brix_trace_frame(c, '<', f->sid, f->stat, 0, f->dlen, f->buf, f->dlen);
194 }
195 return 0;
196}

198/*
199 * WHAT: handle a re-deferral (another kXR_waitresp while awaiting an asynresp):
200 * consume the frame and extend the read window to the new advertised delay.
201 * WHY: keeps the waitresp wait-loop a flat sequence of early-return guards.
202 * HOW: seconds clamp to 570 before *1000 (no int overflow UB), +30s margin,
203 * and the window only ever grows — same arithmetic as the initial arm.
204 */
205static void
206waitresp_extend_window(brix_conn *c, rx_frame_t *f)
207{
208 unsigned more = (f->dlen >= 4) ? xrd_get_u32_be(f->buf) : 0;
209 int w;
210
211 free(f->buf);
212 if (more > 570) { more = 570; } /* clamp before *1000 (no UB) */
213 w = (int) more * 1000 + 30000;
214 if (w > c->io.timeout_ms) { c->io.timeout_ms = w; }
215}

217/*
218 * WHAT: unwrap a validated kXR_attn(asynresp) envelope and surface the inner
219 * status+data exactly as a synchronous reply would.
220 * WHY: the envelope decode is the bulky tail of the waitresp wait-loop;
221 * isolating it leaves the loop as the retry state machine only.
222 * HOW: nested ServerResponseHdr at buf+8, data at buf+16, inner dlen clamped
223 * to the frame; inner-streamid mismatch and inner kXR_error fail exactly
224 * like the synchronous path. Consumes f->buf on every exit. 0 / -1.
225 */
226static int
227asynresp_deliver(uint16_t want_sid, rx_frame_t *f, brix_resp_out *out,
228 brix_status *st)
229{
230 uint16_t esid, estat;
231 uint32_t edlen;
232 uint8_t *edata = f->buf + 16;
233
234 xrd_resp_hdr_unpack(f->buf + 8, &esid, &estat, &edlen); /* nested hdr */
235 if ((size_t) edlen + 16 > f->dlen) { edlen = f->dlen - 16; } /* clamp to frame */
236
237 if (want_sid != 0xffff && esid != want_sid) {
238 free(f->buf);
239 brix_status_set(st, XRDC_EPROTO, 0,
240 "asynresp stream mismatch (got %u, want %u)",
241 esid, want_sid);
242 return -1;
243 }
244 if (estat == kXR_error) {
245 set_kxr_error_status(edata, edlen, st);
246 free(f->buf);
247 return -1;
248 }
249 if (out->status != NULL) { *out->status = estat; }
250 if (out->body != NULL) {
251 *out->body = NULL;
252 if (edlen > 0) {
253 uint8_t *copy = (uint8_t *) malloc(edlen);
254 if (copy == NULL) {
255 free(f->buf);
256 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (%u)", edlen);
257 return -1;
258 }
259 memcpy(copy, edata, edlen);
260 *out->body = copy;
261 }
262 }
263 if (out->blen != NULL) { *out->blen = edlen; }
264 free(f->buf);
265 return 0;
266}

268/* Handle a kXR_waitresp acknowledgement: the real reply for `want_sid` arrives
269 * later, unsolicited, as a kXR_attn carrying an asynresp envelope
270 * [actnum=kXR_asynresp 4][reserved 4][inner ServerResponseHdr 8][data dlen].
271 * Wait for it (extending the read window to the server's advertised delay) and
272 * surface the inner status+data exactly as a synchronous reply would. Another
273 * kXR_waitresp simply re-arms the wait. 0 / -1.
274 *
275 * The server answers synchronously in this codebase, so this path is exercised
276 * against a real (deferring) XRootD or the mock in test_client_async_tpc.py. */
277static int
278recv_after_waitresp(brix_conn *c, uint16_t want_sid, unsigned secs,
279 brix_resp_out *out, brix_status *st)
280{
281 int saved_to = c->io.timeout_ms;
282 int rounds = 0;
283 unsigned s = (secs > 570) ? 570 : secs; /* clamp before *1000 (no UB) */
284 int want_ms = (int) s * 1000 + 30000; /* delay + margin, <= 600000 */
285
286 if (want_ms > c->io.timeout_ms) { c->io.timeout_ms = want_ms; }
287
288 for (;;) {
289 rx_frame_t f;
290
291 if (++rounds > XRDC_REDIR_MAX) {
292 c->io.timeout_ms = saved_to;
293 brix_status_set(st, XRDC_EPROTO, 0,
294 "waitresp: no async response after %d frames", rounds);
295 return -1;
296 }
297 if (recv_raw_frame(c, &f, st) != 0) {
298 c->io.timeout_ms = saved_to;
299 return -1;
300 }
301 if (f.stat == kXR_waitresp) { /* server re-deferred — keep waiting */
302 waitresp_extend_window(c, &f);
303 continue;
304 }
305 if (f.stat != kXR_attn) {
306 free(f.buf);
307 c->io.timeout_ms = saved_to;
308 brix_status_set(st, XRDC_EPROTO, 0,
309 "waitresp: expected attn(asynresp), got status %u", f.stat);
310 return -1;
311 }
312 if (f.dlen < 16 || xrd_get_u32_be(f.buf) != (uint32_t) kXR_asynresp) {
313 free(f.buf);
314 c->io.timeout_ms = saved_to;
315 brix_status_set(st, XRDC_EPROTO, 0, "waitresp: malformed asynresp envelope");
316 return -1;
317 }
318 c->io.timeout_ms = saved_to;
319 return asynresp_deliver(want_sid, &f, out, st);
320 }
321}

323/*
324 * WHAT: §15 response-side diagnostics — wire trace, full-frame capture, and
325 * per-opcode RTT accumulation. Inert unless armed.
326 * WHY: pure observability side-band; hoisting it out of brix_recv leaves the
327 * receive path as read/validate/dispatch only.
328 * HOW: hdr is the raw 8-byte ServerResponseHdr (capture records the exact
329 * wire bytes); RTT matches the request stamped by brix_send_ext via
330 * diag.inflight_reqid / diag.t_send_ns, then clears the stamp.
331 */
332static void
333recv_note_diag(brix_conn *c, const uint8_t *hdr, const rx_frame_t *f)
334{
335 if (c->diag.wire_trace) {
336 brix_trace_frame(c, '<', f->sid, f->stat, 0, f->dlen, f->buf, f->dlen);
337 }
338 if (c->diag.cap != NULL) { /* §15.1: record the full response wire bytes */
339 brix_capture_frame(c->diag.cap, '<', f->sid, f->stat, 0, hdr,
340 XRD_RESPONSE_HDR_LEN, f->buf, f->dlen);
341 }
342 if (c->diag.timing && c->diag.t_send_ns != 0) {
343 uint64_t dt = brix_mono_ns() - c->diag.t_send_ns;
344 int idx = (int) c->diag.inflight_reqid - kXR_1stRequest;
345 if (idx >= 0 && idx < XRDC_NOP) {
346 uint64_t n = c->diag.rtt[idx].n;
347 c->diag.rtt[idx].n++;
348 c->diag.rtt[idx].tot_ns += dt;
349 if (n == 0 || dt < c->diag.rtt[idx].min_ns) { c->diag.rtt[idx].min_ns = dt; }
350 if (dt > c->diag.rtt[idx].max_ns) { c->diag.rtt[idx].max_ns = dt; }
351 }
352 c->diag.t_send_ns = 0;
353 }
354}

356/*
357 * WHAT: hand a successful frame's status/body/length to the caller's out-params.
358 * WHY: every success exit (ok/oksofar/authmore/redirect/wait, the waitresp
359 * TPC-defer surface) performs the same NULL-tolerant delivery.
360 * HOW: when the caller declined the body, free it here so ownership is never
361 * ambiguous. Always returns 0 so dispatch sites can `return` it.
362 */
363static int
364resp_deliver(brix_resp_out *out, uint16_t stat, uint8_t *buf, uint32_t dlen)
365{
366 if (out->status != NULL) { *out->status = stat; }
367 if (out->body != NULL) { *out->body = buf; } else { free(buf); }
368 if (out->blen != NULL) { *out->blen = dlen; }
369 return 0;
370}

372/*
373 * WHAT: interpret a validated frame's status field — the brix_recv switch.
374 * WHY: isolates the per-status policy (which statuses surface, which defer,
375 * which fail) from the read/validate mechanics.
376 * HOW: passthrough statuses deliver the body as-is; kXR_waitresp either
377 * surfaces the deferral (TPC coordinator open — blocking would deadlock
378 * the rendezvous) or blocks in recv_after_waitresp; kXR_error and
379 * unknown statuses consume the body and fail. 0 / -1 (st set).
380 */
381static int
382recv_dispatch(brix_conn *c, uint16_t want_sid, rx_frame_t *f, brix_resp_out *out,
383 brix_status *st)
384{
385 switch (f->stat) {
386 case kXR_ok:
387 case kXR_oksofar:
388 case kXR_authmore: /* auth driver consumes the challenge body */
389 case kXR_redirect: /* brix_roundtrip follows it */
390 case kXR_wait: /* brix_roundtrip honors the backoff */
391 return resp_deliver(out, f->stat, f->buf, f->dlen);
392
393 case kXR_waitresp: {
394 /* Server acknowledged the request but the real reply comes later as an
395 * unsolicited kXR_attn(asynresp). Transparent to every caller... */
396 unsigned secs = (f->dlen >= 4) ? xrd_get_u32_be(f->buf) : 0;
397 free(f->buf);
398 /* ...EXCEPT a TPC coordinator open: the source registers the rendezvous key
399 * and defers its open reply until the copy completes — but that copy can only
400 * happen once the orchestrator opens the DESTINATION and triggers the pull.
401 * Blocking here for the deferred reply would deadlock (source waits for the
402 * pull; the pull waits for this call to return). Surface the deferral so the
403 * caller proceeds; the deferred reply is drained after the dest sync. */
404 if (c->tpc_coord_defer) {
405 return resp_deliver(out, kXR_waitresp, NULL, 0);
406 }
407 return recv_after_waitresp(c, want_sid, secs, out, st);
408 }
409
410 case kXR_error:
411 /* errmsg is wire data with no guaranteed NUL — bound %s with %.*s. */
412 set_kxr_error_status(f->buf, f->dlen, st);
413 free(f->buf);
414 return -1;
415
416 default:
417 brix_status_set(st, XRDC_EPROTO, 0,
418 "unexpected response status %u", f->stat);
419 free(f->buf);
420 return -1;
421 }
422}

424int
425brix_recv(brix_conn *c, uint16_t want_sid, brix_resp_out *out, brix_status *st)
426{
427 uint8_t hdr[XRD_RESPONSE_HDR_LEN];
428 rx_frame_t f;
429
430 if (out->body != NULL) { *out->body = NULL; }
431 if (out->blen != NULL) { *out->blen = 0; }
432
433 if (frame_read_header(c, hdr, &f, st) != 0) {
434 return -1;
435 }
436 if (want_sid != 0xffff && f.sid != want_sid) {
437 brix_status_set(st, XRDC_EPROTO, 0,
438 "stream id mismatch (got %u, want %u)", f.sid, want_sid);
439 return -1;
440 }
441 if (frame_read_body(c, &f, st) != 0) {
442 return -1;
443 }
444
445 recv_note_diag(c, hdr, &f);
446 return recv_dispatch(c, want_sid, &f, out, st);
447}

451/* One parsed ServerRedirectBody: connectable host, port, and any capability
452 * opaque split off the host field (see parse_redirect). Sized so a long EOS
453 * cap.sym/cap.msg opaque is never truncated by the host buffer. */
454typedef struct {
455 char host[256];
456 int port;
457 char opaque[4096];
458} redir_tgt_t;
459
460/* ServerRedirectBody = port[4 BE] + host[NUL] (host already IPv6-bracketed).
461 * The host field may carry a "?<opaque>" tail: a redirector (notably EOS/cmsd)
462 * appends the open CAPABILITY (cap.sym/cap.msg) that the open MUST replay to the
463 * chosen data server, else the DS cannot authorize it and bounces the open back
464 * (an endless manager↔DS redirect loop). We split it off so `host` is connectable
465 * and `opaque` can be re-attached to the open's path. */
466static int
467parse_redirect(const uint8_t *body, uint32_t blen, redir_tgt_t *t)
468{
469 const char *field, *qmark;
470 uint32_t flen, hlen;
471 if (blen < 5) {
472 return -1;
473 }
474 t->port = (int) xrd_get_u32_be(body);
475
476 /* The host field runs from body+4 to the first NUL/CR/LF (or end of body). We
477 * scan it IN PLACE and split host vs opaque straight into their own buffers, so
478 * a long capability opaque (EOS cap.sym/cap.msg, often >256B) is NOT truncated
479 * by the host buffer — the bug that made the DS reject the capability. */
480 field = (const char *) body + 4;
481 flen = blen - 4;
482 {
483 const char *nl = memchr(field, '\0', flen);
484 if (nl == NULL) { nl = memchr(field, '\r', flen); }
485 if (nl == NULL) { nl = memchr(field, '\n', flen); }
486 if (nl != NULL) {
487 flen = (uint32_t) (nl - field);
488 }
489 }
490
491 qmark = memchr(field, '?', flen);
492 hlen = qmark != NULL ? (uint32_t) (qmark - field) : flen;
493 if (hlen >= sizeof(t->host)) {
494 hlen = (uint32_t) sizeof(t->host) - 1;
495 }
496 memcpy(t->host, field, hlen);
497 t->host[hlen] = '\0';
498
499 t->opaque[0] = '\0';
500 if (qmark != NULL) {
501 const char *o = qmark + 1;
502 uint32_t olen = (uint32_t) (field + flen - o);
503 while (olen > 0 && (*o == '&' || *o == '?')) { /* EOS sends "?&cap.sym=" */
504 o++;
505 olen--;
506 }
507 if (olen >= sizeof(t->opaque)) {
508 olen = (uint32_t) sizeof(t->opaque) - 1;
509 }
510 memcpy(t->opaque, o, olen);
511 t->opaque[olen] = '\0';
512 }
513 return 0;
514}

528/*
529 * Phase 40 (a): reconnect to a redirect target, surviving a DEAD target.
530 *
531 * WHAT: bring up a session against rhost:rport; if that target is unreachable,
532 * fall back ONCE to the home manager (c->home_host/home_port) so it can
533 * re-select a live data server.
534 * WHY: official-parity hard-fail (`brix_reconnect != 0 → give up`) turns one
535 * dead replica into a failed op even when other replicas are healthy, and
536 * surfaces a confusing connect error against a host the user never typed.
537 * HOW: the dead target is already recorded in tried[] by the caller, so the
538 * loop guard prevents the manager bouncing us straight back to it. Returns
539 * 0 if a session is up (target or manager), -1 with a clear combined error.
540 * NEVER calls brix_close between attempts — brix_reconnect/bringup own their
541 * own teardown-on-failure, and a close on a torn-down socket would misfire.
542 */
543static int
544follow_redirect(brix_conn *c, const char *rhost, int rport, brix_status *st)
545{
546 char tgt_msg[XRDC_MSG_MAX];
547
548 if (brix_reconnect(c, rhost, rport, st) == 0) {
549 return 0;
550 }
551 snprintf(tgt_msg, sizeof(tgt_msg), "%s", st->msg); /* keep target error */
552
553 if (c->home_host[0] == '\0'
554 || (strcmp(rhost, c->home_host) == 0 && rport == c->home_port)) {
555 brix_status_set(st, XRDC_ESOCK, 0,
556 "redirect target %s:%d unreachable: %s",
557 rhost, rport, tgt_msg);
558 return -1;
559 }
560 if (brix_reconnect(c, c->home_host, c->home_port, st) == 0) {
561 return 0; /* manager re-selects; the dead target is in tried[] */
562 }
563 brix_status_set(st, XRDC_ESOCK, 0,
564 "redirect target %s:%d unreachable (%s); manager %s:%d "
565 "fallback also failed", rhost, rport, tgt_msg,
566 c->home_host, c->home_port);
567 return -1;
568}

570/* One in-flight roundtrip request: the immutable original payload (opaque
571 * rebuilds always start from it so successive redirects swap rather than
572 * accumulate), the payload actually sent this iteration, and the rebuilt
573 * buffer (owned + freed once by brix_roundtrip — so the in-loop early
574 * returns need no per-exit cleanup). */
575typedef struct {
576 uint16_t reqid; /* request opcode (kXR_open gets opaque replay) */
577 const void *orig_pl; /* original payload, never rewritten */
578 uint32_t orig_len;
579 const void *cur_pl; /* payload actually sent (may be rebuilt) */
580 uint32_t cur_len;
581 char *rebuilt; /* opaque-carrying open payload, if any */
582} rt_req_t;
583
584/* Rebuild a kXR_open payload as "<original-path><sep><opaque>" so a redirected
585 * open replays the redirector's capability to the data server. Always built from
586 * the ORIGINAL path (never an earlier rebuild) so successive redirects swap rather
587 * than accumulate opaques. rq->rebuilt is freed once by the wrapper. On success
588 * points rq->cur_pl and rq->cur_len at the new buffer. 0 / -1 (st set). */
589static int
590open_payload_with_opaque(const char *opaque, rt_req_t *rq, brix_status *st)
591{
592 const char *p = (const char *) rq->orig_pl;
593 int hasq = (rq->orig_len > 0 && memchr(p, '?', rq->orig_len) != NULL);
594 size_t ol = strlen(opaque);
595 size_t need = (size_t) rq->orig_len + 1 + ol + 1;
596 char *nb = (char *) malloc(need);
597 if (nb == NULL) {
598 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (redirect opaque)");
599 return -1;
600 }
601 memcpy(nb, p, rq->orig_len);
602 nb[rq->orig_len] = hasq ? '&' : '?';
603 memcpy(nb + rq->orig_len + 1, opaque, ol + 1); /* includes the NUL */
604 free(rq->rebuilt);
605 rq->rebuilt = nb;
606 rq->cur_pl = nb;
607 rq->cur_len = (uint32_t) (rq->orig_len + 1 + ol);
608 return 0;
609}

611/*
612 * WHAT: process one kXR_redirect frame — parse the target, run the loop guards
613 * (budget / self-loop / already-tried), carry an open capability opaque,
614 * and reconnect to the target (with manager fallback).
615 * WHY: this is the entire redirect policy; hoisting it leaves roundtrip_loop
616 * as the bare retry state machine.
617 * HOW: consumes bd on every path. Records the target in tried[] BEFORE
618 * reconnecting so a manager fallback can't bounce us straight back.
619 * Returns 0 = session up, replay the request; -1 = fail (st set).
620 */
621static int
622rt_handle_redirect(brix_conn *c, rt_req_t *rq, uint8_t *bd, uint32_t bl,
623 brix_status *st)
624{
625 redir_tgt_t t;
626 char hp[XRDC_HOSTPORT_MAX];
627
628 if (parse_redirect(bd, bl, &t) != 0) {
629 free(bd);
630 brix_status_set(st, XRDC_EPROTO, 0, "malformed redirect");
631 return -1;
632 }
633 free(bd);
634 if (c->diag.redir_trace) { /* §15.4: surface each hop on stderr */
635 fprintf(stderr, "redirect[%d] -> %s:%d%s\n", c->redir_depth + 1,
636 t.host, t.port, t.opaque[0] ? " (+opaque)" : "");
637 }
638 if (++c->redir_depth > XRDC_REDIR_MAX) {
639 if (c->diag.redir_trace) {
640 fprintf(stderr, "redirect: budget exhausted (>%d hops)\n",
641 XRDC_REDIR_MAX);
642 }
643 brix_status_set(st, XRDC_EREDIRECT, 0,
644 "too many redirects (>%d)", XRDC_REDIR_MAX);
645 return -1;
646 }
647 brix_format_host_port(t.host, (uint16_t) t.port, hp, sizeof(hp));
648 /* Immediate self-redirect: the server we are talking to right now
649 * bounced us straight back to itself. That is an unambiguous loop;
650 * fail fast here rather than reconnecting to it (which would chase
651 * the loop until the connect timeout — up to 15s per hop). */
652 if (t.port == c->port && strcmp(t.host, c->host) == 0) {
653 if (c->diag.redir_trace) {
654 fprintf(stderr, "redirect: self-loop to %s\n", hp);
655 }
656 brix_status_set(st, XRDC_EREDIRECT, 0,
657 "redirect loop: server redirected to itself (%s)",
658 hp);
659 return -1;
660 }
661 if (tried_seen(c, hp)) {
662 if (c->diag.redir_trace) {
663 fprintf(stderr, "redirect: LOOP to already-tried %s\n", hp);
664 }
665 brix_status_set(st, XRDC_EREDIRECT, 0,
666 "redirect loop to already-tried %s", hp);
667 return -1;
668 }
669 if (c->tried_n < XRDC_REDIR_MAX) {
670 snprintf(c->tried[c->tried_n++], sizeof(c->tried[0]), "%s", hp);
671 }
672 /* EOS/cmsd redirects an open to a data server with a one-shot capability
673 * opaque; carry it onto the open's path so the DS authorizes the open
674 * (else it bounces back → the endless loop the guard above trips on). */
675 if (rq->reqid == kXR_open && t.opaque[0] != '\0'
676 && open_payload_with_opaque(t.opaque, rq, st) != 0) {
677 return -1;
678 }
679 return follow_redirect(c, t.host, t.port, st); /* 0 = target or manager up */
680}

682/*
683 * WHAT: honor one kXR_wait frame — sleep for the server's advised delay
684 * (clamped [1,30]s) plus additive sub-second jitter.
685 * WHY: keeps the backoff policy out of the retry state machine.
686 * HOW: consumes bd; bumps *waits against XRDC_REDIR_MAX so a server that
687 * stalls forever fails cleanly. Phase 40 (a): the jitter is ADDITIVE so
688 * a fleet handed the same advised wait doesn't resend in lockstep —
689 * never shorten the server's requested delay. 0 = resend / -1 (st set).
690 */
691static int
692rt_handle_wait(brix_conn *c, uint8_t *bd, uint32_t bl, int *waits,
693 brix_status *st)
694{
695 unsigned secs = xrd_wait_secs_parse(bd, bl, 1, 30); /* clamp [1,30] */
696 unsigned jms;
697
698 (void) c;
699 free(bd);
700 if (++(*waits) > XRDC_REDIR_MAX) {
701 brix_status_set(st, XRDC_EPROTO, 0, "server kept asking to wait");
702 return -1;
703 }
704 sleep(secs);
705 jms = brix_jitter_ms(secs >= 1 ? 1000u : 250u);
706 if (jms > 0) {
707 struct timespec ts;
708 ts.tv_sec = 0;
709 ts.tv_nsec = (long) jms * 1000000L;
710 (void) nanosleep(&ts, NULL);
711 }
712 return 0;
713}

715/* The redirect/wait-following request loop: send, receive, and either deliver
716 * the reply or hand kXR_redirect / kXR_wait to their policy helpers and replay.
717 * rq->rebuilt is owned + freed once by brix_roundtrip, so the in-loop early
718 * returns need no per-exit cleanup. */
719static int
720roundtrip_loop(brix_conn *c, void *hdr24, rt_req_t *rq, brix_resp_out *out,
721 brix_status *st)
722{
723 int waits = 0;
724
725 /* Each top-level op gets a fresh redirect budget + loop-guard. */
726 c->redir_depth = 0;
727 c->tried_n = 0;
728
729 for (;;) {
730 uint16_t sid, stt;
731 uint8_t *bd = NULL;
732 uint32_t bl = 0;
733 brix_payload pl = { rq->cur_pl, rq->cur_len };
734 brix_resp_out rr = { &stt, &bd, &bl };
735
736 if (brix_send(c, hdr24, &pl, &sid, st) != 0) {
737 return -1;
738 }
739 if (brix_recv(c, sid, &rr, st) != 0) {
740 return -1; /* kXR_error / transport → st already set */
741 }
742
743 if (stt == kXR_redirect) {
744 if (rt_handle_redirect(c, rq, bd, bl, st) != 0) {
745 return -1;
746 }
747 continue; /* replay against the redirect target (or manager) */
748 }
749 if (stt == kXR_wait) {
750 if (rt_handle_wait(c, bd, bl, &waits, st) != 0) {
751 return -1;
752 }
753 continue; /* resend the same request to the same server */
754 }
755
756 /* kXR_ok / kXR_oksofar */
757 if (out->status != NULL) { *out->status = stt; }
758 *out->body = bd;
759 *out->blen = bl;
760 return 0;
761 }
762}

764int
765brix_roundtrip(brix_conn *c, void *hdr24, const brix_payload *pl,
766 brix_resp_out *out, brix_status *st)
767{
768 rt_req_t rq;
769 int rc;
770
771 rq.reqid = xrd_get_u16_be((uint8_t *) hdr24 + 2);
772 rq.orig_pl = (pl != NULL) ? pl->data : NULL;
773 rq.orig_len = (pl != NULL) ? pl->len : 0;
774 rq.cur_pl = rq.orig_pl;
775 rq.cur_len = rq.orig_len;
776 rq.rebuilt = NULL; /* opaque-carrying open payload, if a redirect needs it */
777
778 rc = roundtrip_loop(c, hdr24, &rq, out, st);
779 free(rq.rebuilt);
780 return rc;
781}
