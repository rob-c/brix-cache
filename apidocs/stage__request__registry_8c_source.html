1/*
2 * stage_request_registry.c — durable tape/prepare request registry (Task 4):
3 * the durable-store SUBSTRATE + lifecycle.
4 *
5 * See stage_request_registry.h for the contract. This is the FRM-dissolution
6 * re-home of the FRM queue (former src/frm/queue.c + reqfile.c + reqid.c): a
7 * fixed-slot, crash-durable request file keyed by reqid, backing kXR_prepare and
8 * the WebDAV Tape REST API. The recall/stage TRANSFER is the stage_engine's job;
9 * this store owns only the client-facing request METADATA + lifecycle.
10 *
11 * This translation unit owns the low-level substrate (durable file I/O, whole-
12 * store locking, the fixed-slot scan/alloc, reqid/status/view translation) and
13 * the lifecycle (singleton + init). The client-facing ops were split out at
14 * phase-79 to hold every file under the 500-line cap and are reached back into
15 * this substrate through stage_request_registry_internal.h:
16 * - stage_request_registry_mutate.c — add / set_status / cancel / delete /
17 * reap / reqid_generate (the write side)
18 * - stage_request_registry_query.c — get / find_by_path / owner_check /
19 * list_active / list_files / pin_release (the read side)
20 * The substrate helpers those files call were formerly static and are now extern
21 * (declared in the internal header); nothing else changed — the lock/unlock
22 * boundaries, WAL ordering, and slot math are identical.
23 *
24 * DESIGN (inherited verbatim from the proven FRM reqfile, so the on-disk math and
25 * crash semantics are unchanged):
26 * - The file is a flat array of SRQ_REC_SIZE slots; slot 0 is the header, record
27 * N is at (N+1)*SRQ_REC_SIZE. A fixed size => O(1) pwrite by offset.
28 * - Per-record CRC32c + a self-offset field make a torn write DETECTABLE.
29 * - COMMIT POINT is the header write: a record body is pwritten+fdatasync'd, then
30 * the header (seq) is pwritten+fsync'd to publish it (WAL ordering).
31 * - Serialization: a per-worker pthread mutex (threads of this worker) then a
32 * whole-file fcntl lock on a <path>.lock sidecar (cross-worker). Order is
33 * always mutex -> fcntl.
34 * - Lookups are a linear slot scan (the FRM SHM index was a rebuildable cache;
35 * it is intentionally NOT relocated — admissions/cancels are rare and the file
36 * is bounded to the high-water-mark of concurrent requests).
37 */

119/* full-buffer pread / pwrite (plain fd — this journal file is svc-owned, not an
120 * export object, so it is a vfs-seam-allow raw site like the FRM journal was) */
121
122static ngx_int_t
123srq_pread_all(int fd, void *buf, size_t len, off_t off, ngx_log_t *log)
124{
125 u_char *p = buf;
126 brix_sd_obj_t obj;
127
128 /* Route the journal byte read through the Storage Driver seam so the syscall
129 * stays in the backend (src/fs/backend/), same as the former FRM reqfile. */
130 brix_sd_posix_wrap(&obj, fd);
131 while (len > 0) {
132 ssize_t n = obj.driver->pread(&obj, p, len, off);
133 if (n < 0) {
134 if (errno == EINTR) { continue; }
135 ngx_log_error(NGX_LOG_ERR, log, ngx_errno, "stage-req: pread failed");
136 return NGX_ERROR;
137 }
138 if (n == 0) {
139 ngx_log_error(NGX_LOG_ERR, log, 0, "stage-req: short pread");
140 return NGX_ERROR;
141 }
142 p += n; len -= (size_t) n; off += n;
143 }
144 return NGX_OK;
145}

147static ngx_int_t
148srq_pwrite_all(int fd, const void *buf, size_t len, off_t off, ngx_log_t *log)
149{
150 const u_char *p = buf;
151 brix_sd_obj_t obj;
152
153 brix_sd_posix_wrap(&obj, fd);
154 while (len > 0) {
155 ssize_t n = obj.driver->pwrite(&obj, p, len, off);
156 if (n < 0) {
157 if (errno == EINTR) { continue; }
158 ngx_log_error(NGX_LOG_ERR, log, ngx_errno, "stage-req: pwrite failed");
159 return NGX_ERROR;
160 }
161 p += n; len -= (size_t) n; off += n;
162 }
163 return NGX_OK;
164}

188srq_hdr_write(brix_stage_registry_t *reg, srq_hdr_t *hdr, ngx_log_t *log)

214ngx_int_t
215srq_rec_write(brix_stage_registry_t *reg, int64_t off, srq_rec_t *rec,
216 ngx_log_t *log)
217{
218 if (off < (int64_t) SRQ_REC_SIZE) {
219 return NGX_ERROR;
220 }
221 rec->self = off;
222 srq_rec_stamp_crc(rec);
223 if (srq_pwrite_all(reg->fd, rec, sizeof(*rec), (off_t) off, log) != NGX_OK) {
224 return NGX_ERROR;
225 }
226 if (fdatasync(reg->fd) != 0) { /* body durable before the header (WAL) */
227 ngx_log_error(NGX_LOG_ERR, log, ngx_errno, "stage-req: record fdatasync");
228 return NGX_ERROR;
229 }
230 return NGX_OK;

244/* locking */
245
246ngx_int_t
247srq_lock(brix_stage_registry_t *reg, srq_lock_t mode)
248{
249 struct flock fl;
250 int rc;
251
252 pthread_mutex_lock(&srq_file_mtx);
253 if (reg->lock_fd < 0) {
254 return NGX_OK; /* no file lock; mutex still held */
255 }
256 ngx_memzero(&fl, sizeof(fl));
257 fl.l_type = (mode == SRQ_LK_EXCL) ? F_WRLCK : F_RDLCK;
258 fl.l_whence = SEEK_SET;
259 fl.l_start = 0;
260 fl.l_len = 0;
261 do {
262 rc = fcntl(reg->lock_fd, F_SETLKW, &fl);
263 } while (rc < 0 && errno == EINTR);
264 if (rc != 0) {
265 pthread_mutex_unlock(&srq_file_mtx);
266 return NGX_ERROR;
267 }
268 return NGX_OK;
269}
270
271void
272srq_unlock(brix_stage_registry_t *reg)
273{
274 struct flock fl;
275
276 if (reg->lock_fd >= 0) {
277 ngx_memzero(&fl, sizeof(fl));
278 fl.l_type = F_UNLCK;
279 fl.l_whence = SEEK_SET;
280 fl.l_start = 0;
281 fl.l_len = 0;
282 (void) fcntl(reg->lock_fd, F_SETLK, &fl);
283 }
284 pthread_mutex_unlock(&srq_file_mtx);
285}

290void
291srq_reqid_format(const char *host, uint64_t seq, char *buf, size_t buf_sz)
292{
293 u_char *p;
294
295 if (buf == NULL || buf_sz == 0) {
296 return;
297 }
298 /* ngx_snprintf (not libc) truncates safely; the host is deliberately clipped
299 * to fit — uniqueness rests on the durable seq, not the host string. */
300 p = ngx_snprintf((u_char *) buf, buf_sz - 1, "%uL.%P@%s",
301 seq, ngx_pid, (host && host[0]) ? host : "localhost");
302 *p = '\0';
303}

346/* Resolve a reqid -> file offset by a linear slot scan. -1 on miss/error. */
347int64_t
348srq_offset_by_reqid(brix_stage_registry_t *reg, const char *reqid,
349 ngx_log_t *log)
350{
351 srq_rec_t rec;
352 int64_t off, size = srq_file_size(reg);
353
354 for (off = SRQ_REC_OFF(0);
355 size > 0 && off + (int64_t) SRQ_REC_SIZE <= size;
356 off += (int64_t) SRQ_REC_SIZE)
357 {
358 if (srq_rec_read(reg, off, &rec, log) != NGX_OK) {
359 return -1;
360 }
361 if (srq_rec_valid(&rec, off) && rec.status != SRQ_ST_FREE
362 && ngx_strcmp(rec.reqid, reqid) == 0)
363 {
364 return off;
365 }
366 }
367 return -1;
368}

370/* First FREE/torn slot, else EOF (grow). Caller holds the excl lock. */
371int64_t
372srq_alloc_slot(brix_stage_registry_t *reg, ngx_log_t *log)
373{
374 srq_rec_t rec;
375 int64_t size = srq_file_size(reg);
376 int64_t off;
377
378 if (size < 0) {
379 return -1;
380 }
381 for (off = SRQ_REC_OFF(0);
382 off + (int64_t) SRQ_REC_SIZE <= size;
383 off += (int64_t) SRQ_REC_SIZE)
384 {
385 if (srq_rec_read(reg, off, &rec, log) != NGX_OK) {
386 return -1;
387 }
388 if (!srq_rec_valid(&rec, off) || rec.status == SRQ_ST_FREE) {
389 return off;
390 }
391 }
392 return (size < SRQ_REC_OFF(0)) ? SRQ_REC_OFF(0) : size;
393}

404ngx_int_t
405brix_stage_registry_init(const char *journal_dir, ngx_log_t *log)
406{
407 brix_stage_registry_t *reg = &srq_singleton;
408 char lockpath[NGX_MAX_PATH];
409 struct stat st;
410 size_t dlen;
411
412 if (reg->inited) {
413 return NGX_OK; /* idempotent */
414 }
415 if (journal_dir == NULL) {
416 return NGX_ERROR;
417 }
418 dlen = ngx_strlen(journal_dir);
419 if (dlen == 0 || dlen + sizeof("/stage_requests.dat") >= sizeof(reg->path)) {
420 ngx_log_error(NGX_LOG_EMERG, log, 0, "stage-req: journal dir too long");
421 return NGX_ERROR;
422 }
423 *ngx_snprintf((u_char *) reg->path, sizeof(reg->path) - 1,
424 "%s/stage_requests.dat", journal_dir) = '\0';
425 reg->fd = -1;
426 reg->lock_fd = -1;
427 if (gethostname(reg->host, sizeof(reg->host)) != 0) {
428 ngx_memcpy(reg->host, "localhost", sizeof("localhost"));
429 }
430 reg->host[sizeof(reg->host) - 1] = '\0';
431
432 reg->fd = open(reg->path, O_RDWR | O_CREAT | O_CLOEXEC, 0600);
433 if (reg->fd < 0) {
434 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno,
435 "stage-req: cannot open \"%s\"", reg->path);
436 return NGX_ERROR;
437 }
438 *ngx_snprintf((u_char *) lockpath, sizeof(lockpath) - 1,
439 "%s.lock", reg->path) = '\0';
440 reg->lock_fd = open(lockpath, O_RDWR | O_CREAT | O_CLOEXEC, 0600);
441 if (reg->lock_fd < 0) {
442 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno,
443 "stage-req: cannot open \"%s\"", lockpath);
444 close(reg->fd);
445 reg->fd = -1;
446 return NGX_ERROR;
447 }
448 if (fstat(reg->fd, &st) != 0) {
449 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno, "stage-req: fstat");
450 close(reg->fd); close(reg->lock_fd); reg->fd = reg->lock_fd = -1;
451 return NGX_ERROR;
452 }
453 if (st.st_size == 0) {
454 srq_hdr_t hdr;
455 srq_hdr_init_blank(&hdr, time(NULL));
456 if (srq_hdr_write(reg, &hdr, log) != NGX_OK) {
457 close(reg->fd); close(reg->lock_fd); reg->fd = reg->lock_fd = -1;
458 return NGX_ERROR;
459 }
460 }
461 reg->inited = 1;
462 srq_singleton_used = 1;
463 return NGX_OK;
464}
