1/*
2 * stage_request_registry.c — durable tape/prepare request registry (Task 4).
3 *
4 * See stage_request_registry.h for the contract. This is the FRM-dissolution
5 * re-home of the FRM queue (former src/frm/queue.c + reqfile.c + reqid.c): a
6 * fixed-slot, crash-durable request file keyed by reqid, backing kXR_prepare and
7 * the WebDAV Tape REST API. The recall/stage TRANSFER is the stage_engine's job;
8 * this store owns only the client-facing request METADATA + lifecycle.
9 *
10 * DESIGN (inherited verbatim from the proven FRM reqfile, so the on-disk math and
11 * crash semantics are unchanged):
12 * - The file is a flat array of SRQ_REC_SIZE slots; slot 0 is the header, record
13 * N is at (N+1)*SRQ_REC_SIZE. A fixed size => O(1) pwrite by offset.
14 * - Per-record CRC32c + a self-offset field make a torn write DETECTABLE.
15 * - COMMIT POINT is the header write: a record body is pwritten+fdatasync'd, then
16 * the header (seq) is pwritten+fsync'd to publish it (WAL ordering).
17 * - Serialization: a per-worker pthread mutex (threads of this worker) then a
18 * whole-file fcntl lock on a <path>.lock sidecar (cross-worker). Order is
19 * always mutex -> fcntl.
20 * - Lookups are a linear slot scan (the FRM SHM index was a rebuildable cache;
21 * it is intentionally NOT relocated — admissions/cancels are rare and the file
22 * is bounded to the high-water-mark of concurrent requests).
23 */

180/* full-buffer pread / pwrite (plain fd — this journal file is svc-owned, not an
181 * export object, so it is a vfs-seam-allow raw site like the FRM journal was) */
182
183static ngx_int_t
184srq_pread_all(int fd, void *buf, size_t len, off_t off, ngx_log_t *log)
185{
186 u_char *p = buf;
187 brix_sd_obj_t obj;
188
189 /* Route the journal byte read through the Storage Driver seam so the syscall
190 * stays in the backend (src/fs/backend/), same as the former FRM reqfile. */
191 brix_sd_posix_wrap(&obj, fd);
192 while (len > 0) {
193 ssize_t n = obj.driver->pread(&obj, p, len, off);
194 if (n < 0) {
195 if (errno == EINTR) { continue; }
196 ngx_log_error(NGX_LOG_ERR, log, ngx_errno, "stage-req: pread failed");
197 return NGX_ERROR;
198 }
199 if (n == 0) {
200 ngx_log_error(NGX_LOG_ERR, log, 0, "stage-req: short pread");
201 return NGX_ERROR;
202 }
203 p += n; len -= (size_t) n; off += n;
204 }
205 return NGX_OK;
206}
207
208static ngx_int_t
209srq_pwrite_all(int fd, const void *buf, size_t len, off_t off, ngx_log_t *log)
210{
211 const u_char *p = buf;
212 brix_sd_obj_t obj;
213
214 brix_sd_posix_wrap(&obj, fd);
215 while (len > 0) {
216 ssize_t n = obj.driver->pwrite(&obj, p, len, off);
217 if (n < 0) {
218 if (errno == EINTR) { continue; }
219 ngx_log_error(NGX_LOG_ERR, log, ngx_errno, "stage-req: pwrite failed");
220 return NGX_ERROR;
221 }
222 p += n; len -= (size_t) n; off += n;
223 }
224 return NGX_OK;
225}

228/* header + record I/O */
229
230static ngx_int_t
231srq_hdr_read(brix_stage_registry_t *reg, srq_hdr_t *hdr, ngx_log_t *log)
232{
233 if (srq_pread_all(reg->fd, hdr, sizeof(*hdr), 0, log) != NGX_OK) {
234 return NGX_ERROR;
235 }
236 if (hdr->magic != SRQ_MAGIC || hdr->version > SRQ_VERSION
237 || hdr->rec_size != SRQ_REC_SIZE
238 || hdr->hdr_crc32c != srq_hdr_compute_crc(hdr))
239 {
240 ngx_log_error(NGX_LOG_EMERG, log, 0,
241 "stage-req: \"%s\" is not a valid request store "
242 "(bad magic/version/crc)", reg->path);
243 return NGX_ERROR;
244 }
245 return NGX_OK;
246}
247
248static ngx_int_t
249srq_hdr_write(brix_stage_registry_t *reg, srq_hdr_t *hdr, ngx_log_t *log)
250{
251 hdr->magic = SRQ_MAGIC;
252 hdr->version = SRQ_VERSION;
253 hdr->rec_size = SRQ_REC_SIZE;
254 hdr->hdr_crc32c = srq_hdr_compute_crc(hdr);
255 if (srq_pwrite_all(reg->fd, hdr, sizeof(*hdr), 0, log) != NGX_OK) {
256 return NGX_ERROR;
257 }
258 if (fsync(reg->fd) != 0) { /* the commit point — make it durable */
259 ngx_log_error(NGX_LOG_ERR, log, ngx_errno, "stage-req: header fsync");
260 return NGX_ERROR;
261 }
262 return NGX_OK;
263}

265static ngx_int_t
266srq_rec_read(brix_stage_registry_t *reg, int64_t off, srq_rec_t *rec,
267 ngx_log_t *log)
268{
269 if (off < (int64_t) SRQ_REC_SIZE) {
270 return NGX_ERROR;
271 }
272 return srq_pread_all(reg->fd, rec, sizeof(*rec), (off_t) off, log);
273}
274
275static ngx_int_t
276srq_rec_write(brix_stage_registry_t *reg, int64_t off, srq_rec_t *rec,
277 ngx_log_t *log)
278{
279 if (off < (int64_t) SRQ_REC_SIZE) {
280 return NGX_ERROR;
281 }
282 rec->self = off;
283 srq_rec_stamp_crc(rec);
284 if (srq_pwrite_all(reg->fd, rec, sizeof(*rec), (off_t) off, log) != NGX_OK) {
285 return NGX_ERROR;
286 }
287 if (fdatasync(reg->fd) != 0) { /* body durable before the header (WAL) */
288 ngx_log_error(NGX_LOG_ERR, log, ngx_errno, "stage-req: record fdatasync");
289 return NGX_ERROR;
290 }
291 return NGX_OK;
292}

305/* locking */
306
307static ngx_int_t
308srq_lock(brix_stage_registry_t *reg, srq_lock_t mode)
309{
310 struct flock fl;
311 int rc;
312
313 pthread_mutex_lock(&srq_file_mtx);
314 if (reg->lock_fd < 0) {
315 return NGX_OK; /* no file lock; mutex still held */
316 }
317 ngx_memzero(&fl, sizeof(fl));
318 fl.l_type = (mode == SRQ_LK_EXCL) ? F_WRLCK : F_RDLCK;
319 fl.l_whence = SEEK_SET;
320 fl.l_start = 0;
321 fl.l_len = 0;
322 do {
323 rc = fcntl(reg->lock_fd, F_SETLKW, &fl);
324 } while (rc < 0 && errno == EINTR);
325 if (rc != 0) {
326 pthread_mutex_unlock(&srq_file_mtx);
327 return NGX_ERROR;
328 }
329 return NGX_OK;
330}
331
332static void
333srq_unlock(brix_stage_registry_t *reg)
334{
335 struct flock fl;
336
337 if (reg->lock_fd >= 0) {
338 ngx_memzero(&fl, sizeof(fl));
339 fl.l_type = F_UNLCK;
340 fl.l_whence = SEEK_SET;
341 fl.l_start = 0;
342 fl.l_len = 0;
343 (void) fcntl(reg->lock_fd, F_SETLK, &fl);
344 }
345 pthread_mutex_unlock(&srq_file_mtx);
346}

349/* reqid + record <-> view translation */
350
351static void
352srq_reqid_format(const char *host, uint64_t seq, char *buf, size_t buf_sz)
353{
354 u_char *p;
355
356 if (buf == NULL || buf_sz == 0) {
357 return;
358 }
359 /* ngx_snprintf (not libc) truncates safely; the host is deliberately clipped
360 * to fit — uniqueness rests on the durable seq, not the host string. */
361 p = ngx_snprintf((u_char *) buf, buf_sz - 1, "%uL.%P@%s",
362 seq, ngx_pid, (host && host[0]) ? host : "localhost");
363 *p = '\0';
364}

407/* Resolve a reqid -> file offset by a linear slot scan. -1 on miss/error. */
408static int64_t
409srq_offset_by_reqid(brix_stage_registry_t *reg, const char *reqid,
410 ngx_log_t *log)
411{
412 srq_rec_t rec;
413 int64_t off, size = srq_file_size(reg);
414
415 for (off = SRQ_REC_OFF(0);
416 size > 0 && off + (int64_t) SRQ_REC_SIZE <= size;
417 off += (int64_t) SRQ_REC_SIZE)
418 {
419 if (srq_rec_read(reg, off, &rec, log) != NGX_OK) {
420 return -1;
421 }
422 if (srq_rec_valid(&rec, off) && rec.status != SRQ_ST_FREE
423 && ngx_strcmp(rec.reqid, reqid) == 0)
424 {
425 return off;
426 }
427 }
428 return -1;
429}

431/* First FREE/torn slot, else EOF (grow). Caller holds the excl lock. */
432static int64_t
433srq_alloc_slot(brix_stage_registry_t *reg, ngx_log_t *log)
434{
435 srq_rec_t rec;
436 int64_t size = srq_file_size(reg);
437 int64_t off;
438
439 if (size < 0) {
440 return -1;
441 }
442 for (off = SRQ_REC_OFF(0);
443 off + (int64_t) SRQ_REC_SIZE <= size;
444 off += (int64_t) SRQ_REC_SIZE)
445 {
446 if (srq_rec_read(reg, off, &rec, log) != NGX_OK) {
447 return -1;
448 }
449 if (!srq_rec_valid(&rec, off) || rec.status == SRQ_ST_FREE) {
450 return off;
451 }
452 }
453 return (size < SRQ_REC_OFF(0)) ? SRQ_REC_OFF(0) : size;
454}

465ngx_int_t
466brix_stage_registry_init(const char *journal_dir, ngx_log_t *log)
467{
468 brix_stage_registry_t *reg = &srq_singleton;
469 char lockpath[NGX_MAX_PATH];
470 struct stat st;
471 size_t dlen;
472
473 if (reg->inited) {
474 return NGX_OK; /* idempotent */
475 }
476 if (journal_dir == NULL) {
477 return NGX_ERROR;
478 }
479 dlen = ngx_strlen(journal_dir);
480 if (dlen == 0 || dlen + sizeof("/stage_requests.dat") >= sizeof(reg->path)) {
481 ngx_log_error(NGX_LOG_EMERG, log, 0, "stage-req: journal dir too long");
482 return NGX_ERROR;
483 }
484 *ngx_snprintf((u_char *) reg->path, sizeof(reg->path) - 1,
485 "%s/stage_requests.dat", journal_dir) = '\0';
486 reg->fd = -1;
487 reg->lock_fd = -1;
488 if (gethostname(reg->host, sizeof(reg->host)) != 0) {
489 ngx_memcpy(reg->host, "localhost", sizeof("localhost"));
490 }
491 reg->host[sizeof(reg->host) - 1] = '\0';
492
493 reg->fd = open(reg->path, O_RDWR | O_CREAT | O_CLOEXEC, 0600);
494 if (reg->fd < 0) {
495 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno,
496 "stage-req: cannot open \"%s\"", reg->path);
497 return NGX_ERROR;
498 }
499 *ngx_snprintf((u_char *) lockpath, sizeof(lockpath) - 1,
500 "%s.lock", reg->path) = '\0';
501 reg->lock_fd = open(lockpath, O_RDWR | O_CREAT | O_CLOEXEC, 0600);
502 if (reg->lock_fd < 0) {
503 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno,
504 "stage-req: cannot open \"%s\"", lockpath);
505 close(reg->fd);
506 reg->fd = -1;
507 return NGX_ERROR;
508 }
509 if (fstat(reg->fd, &st) != 0) {
510 ngx_log_error(NGX_LOG_EMERG, log, ngx_errno, "stage-req: fstat");
511 close(reg->fd); close(reg->lock_fd); reg->fd = reg->lock_fd = -1;
512 return NGX_ERROR;
513 }
514 if (st.st_size == 0) {
515 srq_hdr_t hdr;
516 srq_hdr_init_blank(&hdr, time(NULL));
517 if (srq_hdr_write(reg, &hdr, log) != NGX_OK) {
518 close(reg->fd); close(reg->lock_fd); reg->fd = reg->lock_fd = -1;
519 return NGX_ERROR;
520 }
521 }
522 reg->inited = 1;
523 srq_singleton_used = 1;
524 return NGX_OK;
525}

528/* reqid generation */
529
530ngx_int_t
531brix_stage_request_reqid_generate(brix_stage_registry_t *reg,
532 char *reqid_out, size_t reqid_out_sz, ngx_log_t *log)
533{
534 srq_hdr_t hdr;
535 uint64_t seq;
536
537 if (reg == NULL || reg->fd < 0 || reqid_out == NULL
538 || reqid_out_sz < SRQ_REQID_LEN)
539 {
540 return NGX_ERROR;
541 }
542 if (srq_lock(reg, SRQ_LK_EXCL) != NGX_OK) {
543 return NGX_ERROR;
544 }
545 if (srq_hdr_read(reg, &hdr, log) != NGX_OK) {
546 srq_unlock(reg);
547 return NGX_ERROR;
548 }
549 seq = ++hdr.seq;
550 if (srq_hdr_write(reg, &hdr, log) != NGX_OK) {
551 srq_unlock(reg);
552 return NGX_ERROR;
553 }
554 srq_unlock(reg);
555 srq_reqid_format(reg->host, seq, reqid_out, reqid_out_sz);
556 return NGX_OK;
557}

560/* mutating ops */
561
562ngx_int_t
563brix_stage_request_add(brix_stage_registry_t *reg,
564 const brix_stage_request_view_t *view, char *reqid_out,
565 size_t reqid_out_sz, ngx_log_t *log)
566{
567 srq_hdr_t hdr;
568 srq_rec_t rec;
569 int64_t off;
570 uint64_t seq;
571 time_t now;
572
573 if (reg == NULL || reg->fd < 0 || view == NULL || view->lfn == NULL
574 || reqid_out == NULL || reqid_out_sz < SRQ_REQID_LEN)
575 {
576 return NGX_ERROR;
577 }
578 if (ngx_strlen(view->lfn) >= SRQ_LFN_LEN) {
579 ngx_log_error(NGX_LOG_ERR, log, 0, "stage-req: lfn exceeds %d bytes",
580 SRQ_LFN_LEN);
581 return NGX_ERROR;
582 }
583 now = time(NULL);
584
585 if (srq_lock(reg, SRQ_LK_EXCL) != NGX_OK) {
586 return NGX_ERROR;
587 }
588 if (srq_hdr_read(reg, &hdr, log) != NGX_OK) {
589 srq_unlock(reg);
590 return NGX_ERROR;
591 }
592 off = srq_alloc_slot(reg, log);
593 if (off < 0) {
594 srq_unlock(reg);
595 return NGX_ERROR;
596 }
597 seq = ++hdr.seq;
598
599 ngx_memzero(&rec, sizeof(rec));
600 rec.self = off;
601 rec.status = SRQ_ST_QUEUED;
602 rec.cs_type = (uint8_t) view->cs_type;
603 rec.tod_added = (int64_t) now;
604 rec.tod_status = (int64_t) now;
605 rec.tod_expire = view->tod_expire;
606 rec.opaque_off = -1;
607 rec.lfn_url_off = -1;
608 srq_reqid_format(reg->host, seq, rec.reqid, sizeof(rec.reqid));
609 ngx_cpystrn((u_char *) rec.lfn, (u_char *) view->lfn, sizeof(rec.lfn));
610 if (view->requester_dn) {
611 ngx_cpystrn((u_char *) rec.requester_dn,
612 (u_char *) view->requester_dn, sizeof(rec.requester_dn));
613 }
614 if (view->user) {

732ngx_int_t
733brix_stage_request_find_by_path(brix_stage_registry_t *reg, const char *lfn,
734 char *reqid_out, size_t reqid_out_sz, ngx_log_t *log)
735{
736 srq_rec_t rec;
737 int64_t off, size, best_off = -1, best_tod = -1;
738 char best_reqid[SRQ_REQID_LEN];
739
740 if (reg == NULL || reg->fd < 0 || lfn == NULL || reqid_out == NULL
741 || reqid_out_sz < SRQ_REQID_LEN)
742 {
743 return NGX_ERROR;
744 }
745 if (srq_lock(reg, SRQ_LK_SHARE) != NGX_OK) {
746 return NGX_ERROR;
747 }
748 size = srq_file_size(reg);
749 for (off = SRQ_REC_OFF(0);
750 size > 0 && off + (int64_t) SRQ_REC_SIZE <= size;
751 off += (int64_t) SRQ_REC_SIZE)
752 {
753 if (srq_rec_read(reg, off, &rec, log) != NGX_OK) {
754 srq_unlock(reg);
755 return NGX_ERROR;
756 }
757 if (!srq_rec_valid(&rec, off) || rec.status == SRQ_ST_FREE) {
758 continue;
759 }
760 if (ngx_strcmp(rec.lfn, lfn) == 0 && rec.tod_added >= best_tod) {
761 best_tod = rec.tod_added;
762 best_off = off;
763 ngx_cpystrn((u_char *) best_reqid, (u_char *) rec.reqid,
764 sizeof(best_reqid));
765 }
766 }
767 srq_unlock(reg);
768 if (best_off < 0) {
769 return NGX_DECLINED;
770 }
771 ngx_cpystrn((u_char *) reqid_out, (u_char *) best_reqid, reqid_out_sz);
772 return NGX_OK;
773}
