1/*
2 * wire_codec_unittest.c — standalone unit test for the xrdwire codec.
3 *
4 * Built and run OUTSIDE the nginx tree (plain gcc), so it verifies the pure
5 * pack/unpack contract without the module or client build. Two kinds of check
6 * per opcode:
7 * 1. round-trip: populate → pack → unpack → assert field equality;
8 * 2. golden offsets: assert specific bytes land at the wire-spec offsets (the
9 * contract both sides depend on).
10 *
11 * Usage:
12 * cd src/protocol/codec
13 * gcc -O2 -Wall -Wextra -Werror -o /tmp/wire_codec_unittest \
14 * wire_codec_meta.c wire_codec_unittest.c
15 * /tmp/wire_codec_unittest
16 */

29/* ---- kXR_stat ---------------------------------------------------------- */
30static void test_stat(void)
31{
32 xrdw_stat_req_t in = { .options = 1, .wants = 0x01020304,
33 .fhandle = { 9, 8, 7, 6 } };
34 uint8_t body[XRDW_BODY_LEN];
35 xrdw_stat_req_t out;
36
37 CHECK(xrdw_stat_req_pack(&in, body) == XRDW_BODY_LEN, "stat pack returns 16");
38 /* golden offsets: options@0, wants@8 (BE), fhandle@12 */
39 CHECK(body[0] == 1, "stat options @0");
40 CHECK(body[8] == 0x01 && body[9] == 0x02 && body[10] == 0x03 && body[11] == 0x04,
41 "stat wants @8 big-endian");
42 CHECK(body[12] == 9 && body[15] == 6, "stat fhandle @12");
43 CHECK(xrdw_stat_req_unpack(body, &out) == XRDW_OK, "stat unpack ok");
44 CHECK(out.options == in.options && out.wants == in.wants
45 && memcmp(out.fhandle, in.fhandle, 4) == 0, "stat round-trip identity");
46}
