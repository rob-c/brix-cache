1/*
2 * access.c — the authorization decision engine (XrdAccAccess::Access).
3 *
4 * WHAT: brix_acc_access() computes the privileges an entity holds for a path
5 * and tests them against the requested operation. It walks the identity
6 * tables in XRootD's exact precedence order, OR-ing every matching rule's
7 * positive and negative privileges, then grants iff (pprivs & ~nprivs)
8 * satisfies the operation.
9 *
10 * WHY: faithful port of XrdAccAccess::Access()/Access2()/Applies(). The order
11 * and the additive accumulation are the defining XrdAcc semantics: an exclusive
12 * (`x`) rule short-circuits to a single decision, otherwise privileges add up
13 * across default, domain, host, netgroup, fungible, user, group, org, role and
14 * inclusive-rule matches.
15 *
16 * HOW: exclusive list first (first applicable rule wins) → default (`u *`) →
17 * host domain → host exact → netgroups → fungible (`u =`) → user → then, per
18 * attribute tuple, group/org/role/inclusive rules; finally pprivs & ~nprivs and
19 * the operation test (privs.c). Auditing is layered on in M5.
20 */
21

35/*
36 * acc_applies — does a compound rule's selector set match this (attr, name,
37 * host)? A NULL selector is unconstrained. Host beginning with '.' is a domain
38 * suffix. Ports XrdAccAccess_ID::Applies().
39 */
40static int
41acc_applies(const brix_acc_idrule_t *r, const brix_acc_attr_t *attr,
42 const char *name, const char *host)
43{
44 if (r->org && (attr->vorg == NULL || ngx_strcmp(r->org, attr->vorg))) {
45 return 0;
46 }
47 if (r->role && (attr->role == NULL || ngx_strcmp(r->role, attr->role))) {
48 return 0;
49 }
50 if (r->grp && (attr->grup == NULL || ngx_strcmp(r->grp, attr->grup))) {
51 return 0;
52 }
53 if (r->user && (name == NULL || ngx_strcmp(r->user, name))) {
54 return 0;
55 }
56 if (r->host) {
57 const char *hn;
58 if (r->host[0] == '.') {
59 int elen = (host != NULL) ? (int) ngx_strlen(host) : 0;
60 if (elen <= r->hlen) {
61 return 0;
62 }
63 hn = host + (elen - r->hlen);
64 } else {
65 hn = host;
66 }
67 if (hn == NULL || ngx_strcmp(r->host, hn)) {
68 return 0;
69 }
70 }
71 return 1;
72}

74/* Final composite + operation test (XrdAccAccess::Access2). */
75static brix_acc_privs_t
76acc_access2(const brix_acc_priv_caps_t *caps, brix_acc_op_t op)
77{
78 brix_acc_privs_t eff = (brix_acc_privs_t) (caps->pprivs & ~caps->nprivs);
79
80 if (op == BRIX_AOP_ANY) {
81 return eff;
82 }
83 return brix_acc_test(eff, op) ? eff : BRIX_ACC_PRIV_NONE;
84}

86brix_acc_privs_t
87brix_acc_access(brix_acc_tables_t *tabs, const brix_acc_entity_t *ent,
88 const char *path, brix_acc_op_t op)
89{
90 brix_acc_priv_caps_t caps = { BRIX_ACC_PRIV_NONE, BRIX_ACC_PRIV_NONE };
91 brix_acc_cap_t *cp;
92 int plen = (int) ngx_strlen(path);
93 const char *vorg_prev = NULL, *role_prev = NULL;
94 ngx_uint_t i;
95 brix_acc_attr_t *tuples;
96
97 if (tabs == NULL || ent == NULL) {
98 return BRIX_ACC_PRIV_NONE;
99 }
100 tuples = ent->tuples->elts;
101
102 /* (1) Exclusive rules — only the first applicable rule applies. */
103 {
104 brix_acc_idrule_t *r;
105 for (r = tabs->sx_list; r != NULL; r = r->next) {
106 for (i = 0; i < ent->tuples->nelts; i++) {
107 if (acc_applies(r, &tuples[i], ent->name, ent->host)) {
108 brix_acc_cap_privs(r->caps, &caps, path, plen, NULL);
109 return acc_access2(&caps, op);
110 }
111 }
112 }
113 }
114
115 /* (2) Default privileges (u *). */
116 if (tabs->z_list) {
117 brix_acc_cap_privs(tabs->z_list, &caps, path, plen, NULL);
118 }
119
120 /* (3) Host domain (h .suffix). */
121 if ((cp = brix_acc_domain_find(tabs->d_list, ent->host))) {
122 brix_acc_cap_privs(cp, &caps, path, plen, NULL);
123 }
124
125 /* (4) Host exact (h <name>). */
126 if ((cp = brix_acc_named_find(tabs->h_list, ent->host))) {
127 brix_acc_cap_privs(cp, &caps, path, plen, NULL);
128 }
129
130 /* (5) Netgroups (n) — probe each record for (user, host) membership. */
131 if (tabs->n_list && acc_netgrp_member != NULL
132 && ent->host && ent->host[0] != '?')
133 {
134 brix_acc_named_t *nn;
135 for (nn = tabs->n_list; nn != NULL; nn = nn->next) {
136 if (acc_netgrp_member(nn->name, ent->name, ent->host)) {
137 brix_acc_cap_privs(nn->caps, &caps, path, plen, NULL);
138 }
139 }
140 }
141
142 /* (5b) Unix groups — expand the user's OS gidlist and match `g` records,
143 * in addition to the entity's own supplied groups handled in step (8). */
144 if (ent->isuser && tabs->g_list && acc_unixgrp_resolver != NULL) {
145 ngx_array_t *ugs = acc_unixgrp_resolver(ent->pool, ent->name);
146 if (ugs != NULL) {
147 char **gn = ugs->elts;
148 for (i = 0; i < ugs->nelts; i++) {
149 if ((cp = brix_acc_named_find(tabs->g_list, gn[i]))) {
150 brix_acc_cap_privs(cp, &caps, path, plen, NULL);
151 }
152 }
153 }
154 }
155
156 /* (6) Fungible user (u =), with the user name substituted into @=. */
157 if (ent->isuser && tabs->x_list) {
158 brix_acc_cap_privs(tabs->x_list, &caps, path, plen, ent->name);
159 }
160
161 /* (7) Specific user (u <name>). */
162 if (ent->isuser && (cp = brix_acc_named_find(tabs->u_list, ent->name))) {
163 brix_acc_cap_privs(cp, &caps, path, plen, NULL);
164 }
165
166 /* (8) Per attribute tuple: group, org, role, then inclusive rules. */
167 for (i = 0; i < ent->tuples->nelts; i++) {
168 brix_acc_attr_t *a = &tuples[i];
169 brix_acc_idrule_t *r;
170
171 if (a->grup && (cp = brix_acc_named_find(tabs->g_list, a->grup))) {
172 brix_acc_cap_privs(cp, &caps, path, plen, NULL);
173 }
174 if (a->vorg && a->vorg != vorg_prev) {
175 vorg_prev = a->vorg;
176 if ((cp = brix_acc_named_find(tabs->o_list, a->vorg))) {
177 brix_acc_cap_privs(cp, &caps, path, plen, NULL);
178 }
179 }
180 if (a->role && a->role != role_prev) {
181 role_prev = a->role;
182 if ((cp = brix_acc_named_find(tabs->r_list, a->role))) {
183 brix_acc_cap_privs(cp, &caps, path, plen, NULL);
184 }
185 }
186 for (r = tabs->sy_list; r != NULL; r = r->next) {
187 if (acc_applies(r, a, ent->name, ent->host)) {
188 brix_acc_cap_privs(r->caps, &caps, path, plen, NULL);
189 }
190 }
191 }
192
193 return acc_access2(&caps, op);
194}
