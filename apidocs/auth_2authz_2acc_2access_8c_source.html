1/*
2 * access.c — the authorization decision engine (XrdAccAccess::Access).
3 *
4 * WHAT: brix_acc_access() computes the privileges an entity holds for a path
5 * and tests them against the requested operation. It walks the identity
6 * tables in XRootD's exact precedence order, OR-ing every matching rule's
7 * positive and negative privileges, then grants iff (pprivs & ~nprivs)
8 * satisfies the operation.
9 *
10 * WHY: faithful port of XrdAccAccess::Access()/Access2()/Applies(). The order
11 * and the additive accumulation are the defining XrdAcc semantics: an exclusive
12 * (`x`) rule short-circuits to a single decision, otherwise privileges add up
13 * across default, domain, host, netgroup, fungible, user, group, org, role and
14 * inclusive-rule matches.
15 *
16 * HOW: exclusive list first (first applicable rule wins) → default (`u *`) →
17 * host domain → host exact → netgroups → fungible (`u =`) → user → then, per
18 * attribute tuple, group/org/role/inclusive rules; finally pprivs & ~nprivs and
19 * the operation test (privs.c). Auditing is layered on in M5.
20 */
21

35/*
36 * acc_sel_ctx_t — the inputs a single selector predicate tests: the rule whose
37 * selectors are being checked, plus the request identity (one attribute tuple,
38 * scalar user name, scalar host). Bundled so every predicate shares one
39 * signature and the predicate table below can be uniform.
40 */
41typedef struct {
42 const brix_acc_idrule_t *rule;
43 const brix_acc_attr_t *attr;
44 const char *name;
45 const char *host;
46} acc_sel_ctx_t;

48/*
49 * WHAT: predicate — the rule's `o` (org) selector is satisfied by this tuple.
50 * WHY: an unset selector is unconstrained (always satisfied); a set selector
51 * requires the tuple to carry that exact org.
52 * HOW: NULL selector ⇒ pass; else the tuple must have a vorg equal to it.
53 */
54static int
55acc_sel_org(const acc_sel_ctx_t *s)
56{
57 return s->rule->org == NULL
58 || (s->attr->vorg != NULL && ngx_strcmp(s->rule->org, s->attr->vorg) == 0);
59}

94/*
95 * WHAT: predicate — the rule's host selector is satisfied by this identity.
96 * WHY: a host beginning with '.' is a domain suffix that must match the tail of
97 * the entity host (and be strictly shorter than it); otherwise it is an exact
98 * host name. A NULL selector is unconstrained.
99 * HOW: NULL ⇒ pass; suffix form ⇒ compare against host tail of length hlen (host
100 * must be longer than the suffix); exact form ⇒ full-string compare.
101 */
102static int
103acc_sel_host(const acc_sel_ctx_t *s)
104{
105 const char *hn;
106
107 if (s->rule->host == NULL) {
108 return 1;
109 }
110 if (s->rule->host[0] == '.') {
111 int elen = (s->host != NULL) ? (int) ngx_strlen(s->host) : 0;
112 if (elen <= s->rule->hlen) {
113 return 0;
114 }
115 hn = s->host + (elen - s->rule->hlen);
116 } else {
117 hn = s->host;
118 }
119 return hn != NULL && ngx_strcmp(s->rule->host, hn) == 0;
120}

122/*
123 * acc_sel_predicates — the selector-predicate table. A compound rule matches
124 * iff every predicate passes (logical AND); ordering is immaterial to the
125 * result and mirrors XrdAccAccess_ID::Applies()'s selector set.
126 */
127typedef int (*acc_sel_fn)(const acc_sel_ctx_t *s);

129static const acc_sel_fn acc_sel_predicates[] = {
130 acc_sel_org,
131 acc_sel_role,
132 acc_sel_group,
133 acc_sel_user,
134 acc_sel_host,
135};

137/*
138 * WHAT: acc_applies — does a compound rule's selector set match this (attr,
139 * name, host)?
140 * WHY: faithful port of XrdAccAccess_ID::Applies(); a rule applies only when all
141 * of its selectors are satisfied.
142 * HOW: run every predicate in acc_sel_predicates; short-circuit to 0 on the
143 * first failure, else 1.
144 */
145static int
146acc_applies(const brix_acc_idrule_t *r, const brix_acc_attr_t *attr,
147 const char *name, const char *host)
148{
149 acc_sel_ctx_t s = { r, attr, name, host };
150 ngx_uint_t k;
151
152 for (k = 0; k < sizeof(acc_sel_predicates) / sizeof(acc_sel_predicates[0]); k++) {
153 if (!acc_sel_predicates[k](&s)) {
154 return 0;
155 }
156 }
157 return 1;
158}

161static brix_acc_privs_t
162acc_access2(const brix_acc_priv_caps_t *caps, brix_acc_op_t op)
163{
164 brix_acc_privs_t eff = (brix_acc_privs_t) (caps->pprivs & ~caps->nprivs);
165
166 if (op == BRIX_AOP_ANY) {
167 return eff;
168 }
169 return brix_acc_test(eff, op) ? eff : BRIX_ACC_PRIV_NONE;
170}

172/*
173 * acc_eval_t — the shared, read-mostly state threaded through the grant phases
174 * of one brix_acc_access() call: the table set, the request identity (with its
175 * pre-fetched tuple array), the path and its length, and the accumulating
176 * privilege caps. Bundled so each grant helper takes one context pointer rather
177 * than a six-parameter list.
178 */
179typedef struct {
180 brix_acc_tables_t *tabs;
181 const brix_acc_entity_t *ent;
182 const brix_acc_attr_t *tuples;
183 const char *path;
184 int plen;
185 brix_acc_priv_caps_t *caps;
186} acc_eval_t;

188/*
189 * WHAT: acc_match_exclusive — apply the exclusive-rule list (`x`), whose first
190 * applicable rule is the sole decision.
191 * WHY: XrdAcc semantics: an exclusive match short-circuits the whole engine to a
192 * single rule's privileges (no additive accumulation).
193 * HOW: walk sx_list in order; for the first rule that applies to any of the
194 * entity's tuples, fold its caps into e->caps and return 1. Return 0 if none
195 * apply (caps is left untouched, the caller falls through to the additive
196 * pipeline).
197 */
198static int
199acc_match_exclusive(const acc_eval_t *e)
200{
201 brix_acc_idrule_t *r;
202 ngx_uint_t i;
203
204 for (r = e->tabs->sx_list; r != NULL; r = r->next) {
205 for (i = 0; i < e->ent->tuples->nelts; i++) {
206 if (acc_applies(r, &e->tuples[i], e->ent->name, e->ent->host)) {
207 brix_acc_cap_privs(r->caps, e->caps, e->path, e->plen, NULL);
208 return 1;
209 }
210 }
211 }
212 return 0;
213}

215/*
216 * WHAT: acc_grant_host — fold the host-scoped grants (default, domain, exact
217 * host, netgroups) into e->caps.
218 * WHY: steps (2)–(5) of XrdAccAccess::Access() — host/site identity, independent
219 * of whether the entity is a real user.
220 * HOW: default (`u *`), domain (`h .suffix`), exact host (`h <name>`), then each
221 * netgroup record whose (user, host) membership resolves — in that order.
222 */
223static void
224acc_grant_host(const acc_eval_t *e)
225{
226 brix_acc_tables_t *tabs = e->tabs;
227 const brix_acc_entity_t *ent = e->ent;
228 brix_acc_cap_t *cp;
229 brix_acc_named_t *nn;
230
231 if (tabs->z_list) { /* (2) default */
232 brix_acc_cap_privs(tabs->z_list, e->caps, e->path, e->plen, NULL);
233 }
234 if ((cp = brix_acc_domain_find(tabs->d_list, ent->host))) { /* (3) domain */
235 brix_acc_cap_privs(cp, e->caps, e->path, e->plen, NULL);
236 }
237 if ((cp = brix_acc_named_find(tabs->h_list, ent->host))) { /* (4) exact host */
238 brix_acc_cap_privs(cp, e->caps, e->path, e->plen, NULL);
239 }
240 if (tabs->n_list && acc_netgrp_member != NULL /* (5) netgroups */
241 && ent->host && ent->host[0] != '?')
242 {
243 for (nn = tabs->n_list; nn != NULL; nn = nn->next) {
244 if (acc_netgrp_member(nn->name, ent->name, ent->host)) {
245 brix_acc_cap_privs(nn->caps, e->caps, e->path, e->plen, NULL);
246 }
247 }
248 }
249}

251/*
252 * WHAT: acc_grant_user — fold the user-scoped grants (OS gidlist, fungible user,
253 * specific user) into e->caps.
254 * WHY: steps (5b)–(7) of XrdAccAccess::Access() — grants keyed on the entity
255 * being a real, named user.
256 * HOW: expand the user's OS group list against `g` records, then the fungible
257 * user (`u =`, with @= substitution), then the specific user (`u <name>`).
258 * No-op when the entity is not a user.
259 */
260static void
261acc_grant_user(const acc_eval_t *e)
262{
263 brix_acc_tables_t *tabs = e->tabs;
264 const brix_acc_entity_t *ent = e->ent;
265 brix_acc_cap_t *cp;
266 ngx_array_t *ugs;
267 ngx_uint_t i;
268
269 if (!ent->isuser) {
270 return;
271 }
272 if (tabs->g_list && acc_unixgrp_resolver != NULL /* (5b) OS gidlist */
273 && (ugs = acc_unixgrp_resolver(ent->pool, ent->name)) != NULL)
274 {
275 char **gn = ugs->elts;
276 for (i = 0; i < ugs->nelts; i++) {
277 if ((cp = brix_acc_named_find(tabs->g_list, gn[i]))) {
278 brix_acc_cap_privs(cp, e->caps, e->path, e->plen, NULL);
279 }
280 }
281 }
282 if (tabs->x_list) { /* (6) fungible user */
283 brix_acc_cap_privs(tabs->x_list, e->caps, e->path, e->plen, ent->name);
284 }
285 if ((cp = brix_acc_named_find(tabs->u_list, ent->name))) { /* (7) specific user */
286 brix_acc_cap_privs(cp, e->caps, e->path, e->plen, NULL);
287 }
288}

290/*
291 * WHAT: acc_grant_tuples — fold the per-attribute-tuple grants (group, org,
292 * role, inclusive rules) into e->caps.
293 * WHY: step (8) of XrdAccAccess::Access() — the attribute-driven grants, one
294 * pass per (group, org, role) tuple the entity presents.
295 * HOW: for each tuple add its `g`/`o`/`r` record (org and role deduped against
296 * the previous tuple to mirror the reference), then every inclusive rule
297 * (`s`) that applies.
298 */
299static void
300acc_grant_tuples(const acc_eval_t *e)
301{
302 brix_acc_tables_t *tabs = e->tabs;
303 const char *vorg_prev = NULL, *role_prev = NULL;
304 brix_acc_cap_t *cp;
305 brix_acc_idrule_t *r;
306 ngx_uint_t i;
307
308 for (i = 0; i < e->ent->tuples->nelts; i++) {
309 const brix_acc_attr_t *a = &e->tuples[i];
310
311 if (a->grup && (cp = brix_acc_named_find(tabs->g_list, a->grup))) {
312 brix_acc_cap_privs(cp, e->caps, e->path, e->plen, NULL);
313 }
314 if (a->vorg && a->vorg != vorg_prev) {
315 vorg_prev = a->vorg;
316 if ((cp = brix_acc_named_find(tabs->o_list, a->vorg))) {
317 brix_acc_cap_privs(cp, e->caps, e->path, e->plen, NULL);
318 }
319 }
320 if (a->role && a->role != role_prev) {
321 role_prev = a->role;
322 if ((cp = brix_acc_named_find(tabs->r_list, a->role))) {
323 brix_acc_cap_privs(cp, e->caps, e->path, e->plen, NULL);
324 }
325 }
326 for (r = tabs->sy_list; r != NULL; r = r->next) {
327 if (acc_applies(r, a, e->ent->name, e->ent->host)) {
328 brix_acc_cap_privs(r->caps, e->caps, e->path, e->plen, NULL);
329 }
330 }
331 }
332}

334/*
335 * WHAT: brix_acc_access — compute the entity's effective privileges for a path
336 * and test them against the requested operation.
337 * WHY: faithful port of XrdAccAccess::Access(); default-deny — an unknown or
338 * empty identity yields no privileges.
339 * HOW: exclusive rules first (first match is the whole answer), else fold the
340 * host, user and per-tuple grants additively into caps and return the
341 * composite-then-operation verdict (acc_access2). Behaviour is unchanged; the
342 * phases are just factored into named helpers.
343 */
344brix_acc_privs_t
345brix_acc_access(brix_acc_tables_t *tabs, const brix_acc_entity_t *ent,
346 const char *path, brix_acc_op_t op)
347{
348 brix_acc_priv_caps_t caps = { BRIX_ACC_PRIV_NONE, BRIX_ACC_PRIV_NONE };
349 acc_eval_t e;
350
351 if (tabs == NULL || ent == NULL) {
352 return BRIX_ACC_PRIV_NONE;
353 }
354
355 e.tabs = tabs;
356 e.ent = ent;
357 e.tuples = ent->tuples->elts;
358 e.path = path;
359 e.plen = (int) ngx_strlen(path);
360 e.caps = &caps;
361
362 if (acc_match_exclusive(&e)) {
363 return acc_access2(&caps, op);
364 }
365
366 acc_grant_host(&e);
367 acc_grant_user(&e);
368 acc_grant_tuples(&e);
369
370 return acc_access2(&caps, op);
371}
