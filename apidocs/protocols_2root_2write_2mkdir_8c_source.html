10/*
11 * brix_handle_mkdir â create a directory within the export root.
12 *
13 * Wire format (ClientMkdirRequest):
14 * options[0]: bitfield; kXR_mkdirpath (0x01) requests recursive mkdir.
15 * mode: big-endian uint16_t; low 9 bits set the permission mask.
16 * Defaults to 0755 if the client sends 0.
17 *
18 * Path resolution strategy:
19 * kXR_mkdirpath set: brix_resolve_path_noexist â no realpath(3) call
20 * because intermediate directories do not yet exist.
21 * kXR_mkdirpath clear: brix_resolve_path_write â parent must exist,
22 * target may not.
23 *
24 * EEXIST is treated as success for both paths (idempotent mkdir).
25 */
26ngx_int_t
27brix_handle_mkdir(brix_ctx_t *ctx, ngx_connection_t *c,
28 ngx_stream_brix_srv_conf_t *conf)
29{

31 char reqpath[BRIX_MAX_PATH + 1];
32 char resolved[PATH_MAX];
33 mode_t mode;
34 int recursive;
35
36 xrdw_mkdir_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
37 recursive = (req.options & kXR_mkdirpath) ? 1 : 0;
38 mode = req.mode & 0777;
39 if (mode == 0) {
40 mode = 0755;
41 }
42
43 {
44 brix_path_mode_t pmode = recursive ? BRIX_PATH_NOEXIST
45 : BRIX_PATH_WRITE;

47 pmode,
48 reqpath, sizeof(reqpath),
49 resolved, sizeof(resolved)) != NGX_OK) {
50 return ctx->write_rc;
51 }
52 }
53
54 if (brix_auth_gate(ctx, c, BRIX_OP_MKDIR, "MKDIR",
55 reqpath, resolved, conf,
56 BRIX_AUTH_MKDIR, 1) != NGX_OK) {
57 return ctx->write_rc;
58 }
59
60 {
61 brix_vfs_ctx_t vctx;

63
64 brix_vfs_ctx_init(&vctx, c->pool, c->log, BRIX_PROTO_ROOT,

70
71 /* Idempotency follows the reference do_Mkdir: only kXR_mkdirpath
72 * (mkdir -p) tolerates an existing target — and the recursive helper
73 * is itself idempotent (returns NGX_OK), so EEXIST only arises on a
74 * plain single-level mkdir, which must fail kXR_ItExists. */
75 if (err == EEXIST && !recursive) {
76 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MKDIR, "MKDIR", resolved, "-",
77 kXR_ItExists, "file exists");
78 }
79 BRIX_RETURN_ERR(ctx, c, BRIX_OP_MKDIR, "MKDIR", resolved, "-",
80 brix_kxr_from_errno(err),
81 (err == EACCES || err == EPERM)
82 ? "permission denied" : strerror(err));
83 }
84 /* NGX_OK ⟺ the directory was created (not pre-existing); apply parent
85 * group policy after a successful single-level mkdir. setgid group
86 * inheritance is a real-filesystem (POSIX) semantic enforced with raw
87 * chmod/chown — meaningless for a non-POSIX backend whose namespace lives
88 * in a catalog, so it is skipped there (the driver owns that mode). */
89 if (!recursive && conf->group_rules != NULL
90 && brix_vfs_backend_resolve(conf->common.root_canon, c->log) == NULL) {
91 brix_apply_parent_group_policy_path(c->log, resolved,
92 conf->group_rules);
93 }
94 }
95
96 BRIX_RETURN_OK(ctx, c, BRIX_OP_MKDIR, "MKDIR", resolved, "-", 0);
97}
