1/*
2 * sd_ceph.h — Ceph/RADOS storage driver (phase-60, basic librados backend).
3 *
4 * WHAT: Declares (a) the always-available, dependency-free LFN->object-key
5 * helpers (lexical path normalization + key composition + a stable inode
6 * hash) that map a confined logical path to a flat RADOS object id, and
7 * (b) — only when the build found librados (BRIX_HAVE_CEPH) — the
8 * per-export config struct the config layer fills and the driver vtable
9 * symbol the registry registers.
10 *
11 * WHY: RADOS is a flat object store: there is no kernel fd, no directory tree,
12 * no atomic rename. The driver therefore needs a deterministic, injective,
13 * escape-proof LFN->oid map (two logical paths must never alias one object,
14 * and no `..` may address an object outside the export's key prefix). That
15 * map is pure string logic with no librados/nginx dependency, so it is
16 * split out here and unit-tested standalone (sd_ceph_unittest.c) without a
17 * live cluster.
18 *
19 * HOW: The helpers are plain C (libc only). The driver body (sd_ceph.c, under
20 * #if BRIX_HAVE_CEPH) implements the worker-safe raw byte ops and the
21 * minimal namespace ops against raw librados (rados_read/write/trunc/stat/
22 * remove). libradosstriper interop with stock XrdCeph (ADR-3) is a follow-on.
23 */

31/*
32 * sd_ceph_normalize — lexically canonicalize a logical path into `out` (cap
33 * bytes): collapse repeated/leading/trailing '/', drop "." components, and apply
34 * ".." by popping the previous component. A ".." that would climb above the root
35 * is rejected (escape attempt). The result always begins with '/', has no "."/
36 * ".." and no "//", and never ends in '/' (except the bare root "/"). Returns 0,
37 * or -1 with errno set (EINVAL on bad args / escape, ENAMETOOLONG if it won't
38 * fit). This is the single point that guarantees the LFN->oid map is injective
39 * and prefix-confined.
40 */
41int sd_ceph_normalize(const char *lfn, char *out, size_t cap);
42
43/*
44 * sd_ceph_key — compose the RADOS object id for a logical path:
45 * `key_prefix` followed by sd_ceph_normalize(lfn). Returns 0, or -1 (errno) when
46 * normalization fails or the result won't fit in `out` (cap bytes).
47 */
48int sd_ceph_key(const char *key_prefix, const char *lfn, char *out, size_t cap);
49
50/*
51 * sd_ceph_ino — a stable 64-bit inode number synthesized from an object id
52 * (FNV-1a). RADOS objects have no inode; protocol stat needs a stable, distinct
53 * value per object, and a content-independent hash of the (already-confined) oid
54 * provides one.
55 */
56uint64_t sd_ceph_ino(const char *oid);

111/* ---- cephfsro: read-only CephFS-via-RADOS driver (sd_cephfs_ro.c) ----------
112 * Serves a real CephFS by reading its metadata-pool omaps + data-pool objects
113 * directly, for when CephFS cannot be mounted. READ-ONLY: every mutating slot is
114 * absent. The filesystem MUST be quiesced (MDS down / fs failed, journal flushed)
115 * — the namespace a pure-RADOS reader sees is only consistent then — so init
116 * refuses to bind unless `assume_quiesced` is set (operator assertion; there is
117 * no active MDS probing). See docs/superpowers/specs/2026-06-30-cephfs-rados-
118 * program-design.md. */
119typedef struct {
120 const char *meta_pool; /* CephFS metadata pool (REQUIRED) */
121 const char *data_pool; /* CephFS data pool (REQUIRED) */
122 const char *conf_file; /* ceph.conf (default /etc/ceph/ceph.conf) */
123 const char *user; /* ceph user (default client.admin) */
124 const char *keyring; /* optional keyring path override */
125 int assume_quiesced; /* operator assertion: fs is frozen */
126 int live; /* operator assertion: fs still mounted —
127 * best-effort eventually-consistent reads with
128 * optimistic revalidation + retry. One of
129 * assume_quiesced / live MUST be set. */
130} brix_sd_cephfs_ro_conf_t;
