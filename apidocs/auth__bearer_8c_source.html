1/*
2 * auth_bearer.c — WLCG bearer-token validation for the S3 endpoint.
3 *
4 * WHAT: Implements Bearer-token detection and JWT validation for S3 requests.
5 * The two entry points (s3_bearer_present, s3_verify_bearer) are called by
6 * the s3_verify_sigv4 gate when brix_s3_token is enabled.
7 *
8 * WHY: INVARIANT §6: S3 SigV4 and WLCG bearer are mutually exclusive per request
9 * — the auth logic for each scheme must live in separate compilation units
10 * so neither can accidentally depend on the other's internals. The JWKS
11 * keys are loaded once at config time into cf->jwks_keys and reused across
12 * all requests in a worker process.
13 *
14 * HOW:
15 * s3_bearer_present — purely reads the Authorization header; no allocations.
16 * s3_verify_bearer — extracts the token string, validates it with
17 * brix_token_validate(), copies the resulting claims into *identity, and
18 * returns NGX_OK on success. On failure, sends an XML AccessDenied error
19 * and returns an HTTP error code; the caller propagates this directly.
20 * No goto; early-return style throughout.
21 */
22

31/* get_header() is defined (non-static) in auth_sigv4_parse.c; declared extern
32 * here rather than adding a new shared header for a single helper — the
33 * function has external linkage at the object-file level and the linker resolves
34 * it within the same module. */
35extern ngx_str_t get_header(ngx_http_request_t *r, const char *name);

37/* Maximum JWT length we accept. Real WLCG tokens are 300-2000 bytes; 8 KiB
38 * is a generous guard against oversized-token DoS without pathological RAM use. */
39#define S3_BEARER_MAX_TOKEN_LEN 8192
40
41/* Length of the "Bearer " prefix (seven bytes, including the space). */
42#define S3_BEARER_PREFIX_LEN 7
43
44/*
45 * s3_bearer_present — detect a Bearer Authorization header.
46 *
47 * WHAT: Pure predicate — reads the Authorization header and returns 1 if the
48 * value starts with "Bearer " (case-insensitive).
49 * WHY: The caller (s3_verify_sigv4) needs to decide whether to route to the
50 * bearer path or the SigV4/anonymous path before any token work happens.
51 * HOW: brix_http_get_header returns the raw Authorization value; we compare
52 * the first 7 bytes case-insensitively. Short strings are rejected (< 7).
53 */
54int
55s3_bearer_present(ngx_http_request_t *r)
56{
57 ngx_str_t authz;
58
59 authz = get_header(r, "authorization");
60 if (authz.len < S3_BEARER_PREFIX_LEN) {
61 return 0;
62 }
63
64 return ngx_strncasecmp(authz.data,
65 (u_char *) "Bearer ",
66 S3_BEARER_PREFIX_LEN) == 0;
67}

69/*
70 * s3_verify_bearer — validate a WLCG JWT bearer token from Authorization header.
71 *
72 * WHAT: Extracts the token from Authorization: Bearer <token>, validates the JWT
73 * signature and claims against the JWKS keys in *cf, and on success
74 * populates *identity with the subject, scopes, and groups from the claims.
75 *
76 * WHY: brix_token_validate is the canonical JWT verifier shared across all
77 * protocols; this wrapper adapts its inputs/outputs to the S3 handler's
78 * calling convention (returns NGX_OK / HTTP-error-code, response already
79 * sent on error).
80 *
81 * HOW: Extracts token bytes from the Authorization header (after "Bearer "),
82 * guards the length, calls brix_token_validate, on failure sends XML
83 * AccessDenied and returns the HTTP code, on success calls
84 * brix_identity_set_token_claims and records the auth metric.
85 * No goto; early-return on every failure.
86 */
87ngx_int_t
88s3_verify_bearer(ngx_http_request_t *r,
89 ngx_http_s3_loc_conf_t *cf, brix_identity_t *identity)
90{
91 ngx_str_t authz;
92 const char *token;
93 size_t token_len;
94 brix_token_claims_t claims;
95 int rc;
96
97 authz = get_header(r, "authorization");
98
99 /* Caller guarantees s3_bearer_present() returned 1, so the prefix is there;
100 * advance past "Bearer " to the token proper. */
101 if (authz.len <= S3_BEARER_PREFIX_LEN) {
102 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
103 "AccessDenied",
104 "Bearer token missing after Authorization prefix");
105 }
106
107 token = (const char *) authz.data + S3_BEARER_PREFIX_LEN;
108 token_len = authz.len - S3_BEARER_PREFIX_LEN;
109
110 if (token_len > S3_BEARER_MAX_TOKEN_LEN) {
111 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
112 "brix_s3: bearer token too large (%uz bytes)", token_len);
113 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
114 "AccessDenied",
115 "bearer token exceeds maximum permitted length");
116 }
117
118 ngx_memzero(&claims, sizeof(claims));
119
120 {
121 brix_token_validate_args_t va;
122
123 va.log = r->connection->log;
124 va.token = token;
125 va.token_len = token_len;
126 va.keys = cf->jwks_keys;
127 va.key_count = cf->jwks_key_count;
128 va.expected_issuer = (const char *) cf->token_issuer.data;
129 va.expected_audience = (const char *) cf->token_audience.data;
130 va.macaroon_secret = NULL;
131 va.secret_len = 0;
132 va.clock_skew = (int) cf->token_clock_skew;
133 va.claims = &claims;
134
135 rc = brix_token_validate(&va);
136 }
137 if (rc != 0) {
138 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
139 "brix_s3: bearer token validation failed");
140 brix_metric_auth(BRIX_PROTO_S3, BRIX_AUTHN_TOKEN, 0);
141 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
142 "AccessDenied",
143 "bearer token validation failed");
144 }
145
146 if (identity != NULL
147 && brix_identity_set_token_claims(identity, r->pool, &claims) != NGX_OK)
148 {
149 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
150 "brix_s3: failed to set token identity claims (OOM?)");
151 return NGX_HTTP_INTERNAL_SERVER_ERROR;
152 }
153
154 /* Phase-70 §5.4: retain the raw JWT bytes for backend PASSTHROUGH. Copied
155 * onto r->pool (the wire `token` points into the Authorization header value)
156 * and never logged; stored on the S3 req ctx the VFS bind sites read. */
157 {
158 ngx_http_s3_req_ctx_t *s3ctx =
159 ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
160
161 if (s3ctx != NULL) {
162 s3ctx->bearer_token.data = ngx_pnalloc(r->pool, token_len);
163 if (s3ctx->bearer_token.data == NULL) {
164 return NGX_HTTP_INTERNAL_SERVER_ERROR;
165 }
166 ngx_memcpy(s3ctx->bearer_token.data, token, token_len);
167 s3ctx->bearer_token.len = token_len;
168 }
169 }
170
171 /* Record a successful token auth in the unified protocol metrics. */
172 brix_metric_auth(BRIX_PROTO_S3, BRIX_AUTHN_TOKEN, 1);
173 return NGX_OK;
174}
