1/*
2 * staged_file_reap.c — Durable stage-out markers + crash-safe commit reaper.
3 *
4 * WHAT: The pending-commit marker family (register stage dirs, write/remove a
5 * "<stage_partial>.commit" marker) and the reaper that finishes any
6 * interrupted stage-out on the next worker startup / periodic sweep.
7 *
8 * WHY: Split from staged_file.c (phase-79 file-size split). With a synchronous
9 * commit the client waits for the stage→storage move, but if the worker
10 * dies mid-move the COMPLETE file is left in the cache with nothing
11 * recording where it belongs. A durable marker (content = the final
12 * absolute path) written+fsync'd before the move and removed after lets the
13 * reaper always drive complete-but-uncommitted files to storage across
14 * restarts. Keeping this recovery machinery in its own file leaves the
15 * hot-path open/commit lifecycle uncluttered.
16 *
17 * HOW: brix_stage_dir_register/_count track the dirs to sweep. mark/unmark write
18 * and drop the marker around a move. brix_stage_reap_dir snapshots every
19 * marker basename FIRST (reap_collect_markers — it unlinks while iterating,
20 * which would otherwise make readdir skip entries), then reap_one_marker
21 * reads each target (reap_read_marker_target) and republishes via
22 * brix_commit_staged (declared in staged_file.h). Idempotent: a marker whose
23 * partial is already gone is simply dropped. Zero behavior change from the
24 * pre-split code.
25 */

36/*
37 * When an upload completes, the (complete) staged file must be moved from the
38 * stage device to the final storage. With a synchronous commit the client waits
39 * for that move, but if the worker dies mid-move the COMPLETE file is left in the
40 * cache with nothing recording where it should go. A marker file
41 * "<stage_partial>.commit" (content = the final absolute path) is written +
42 * fsync'd just before the move and removed after it; if the move is interrupted
43 * the marker survives, and brix_stage_reap_dir() finishes the move on the next
44 * worker startup / periodic sweep — so complete-but-uncommitted files are tracked
45 * across restarts and always reach storage.
46 */
47#define BRIX_STAGE_COMMIT_SUFFIX ".commit"

53void
54brix_stage_dir_register(const char *canon)
55{
56 ngx_uint_t i;
57
58 if (canon == NULL || canon[0] == '\0' || strlen(canon) >= PATH_MAX) {
59 return;
60 }
61 for (i = 0; i < s_stage_dir_count; i++) {
62 if (strcmp(s_stage_dirs[i], canon) == 0) {
63 return; /* already registered (dedup across server blocks) */
64 }
65 }
66 if (s_stage_dir_count >= BRIX_STAGE_MAX_DIRS) {
67 return;
68 }
69 ngx_memcpy(s_stage_dirs[s_stage_dir_count], canon, strlen(canon) + 1);
70 s_stage_dir_count++;
71}

96 return NGX_ERROR;

99 if ((size_t) write(fd, final_path, flen) != flen || fsync(fd) != 0) {
100 int e = errno; close(fd); (void) unlink(marker); errno = e;
101 return NGX_ERROR;
102 }
103 close(fd);
104 return NGX_OK;

118/* Maximum stage-out markers snapshotted per reap pass, and the per-basename cap. */
119#define BRIX_STAGE_REAP_MAX_MARKERS 256
120#define BRIX_STAGE_REAP_NAME_MAX 256
121
122/*
123 * WHAT: Snapshot the basenames of every ".commit" marker in stage_dir into names[]
124 * (returns the count captured, capped at max).
125 *
126 * WHY: The reap loop unlinks markers while it works, which would make a live
127 * readdir skip entries — so the directory is scanned into a snapshot FIRST,
128 * then the snapshot is processed.
129 *
130 * HOW: opendir + readdir, keeping only names that end in BRIX_STAGE_COMMIT_SUFFIX
131 * and fit the basename buffer; copy each (with its NUL) into names[]. A
132 * missing / unopenable dir yields 0.
133 *
134 * Parameters:
135 * stage_dir — directory to scan
136 * names — output array of NUL-terminated basenames
137 * max — capacity of names[] (entries beyond it are ignored)
138 */
139static ngx_uint_t
140reap_collect_markers(const char *stage_dir,
141 char names[][BRIX_STAGE_REAP_NAME_MAX], ngx_uint_t max)
142{
143 DIR *d;
144 struct dirent *de;
145 size_t slen = sizeof(BRIX_STAGE_COMMIT_SUFFIX) - 1;
146 ngx_uint_t ncount = 0;
147
148 d = opendir(stage_dir);
149 if (d == NULL) {
150 return 0;
151 }
152 while ((de = readdir(d)) != NULL && ncount < max) {
153 size_t nlen = strlen(de->d_name);
154 if (nlen > slen && nlen < BRIX_STAGE_REAP_NAME_MAX
155 && strcmp(de->d_name + nlen - slen, BRIX_STAGE_COMMIT_SUFFIX) == 0)
156 {
157 ngx_memcpy(names[ncount], de->d_name, nlen + 1);
158 ncount++;
159 }
160 }
161 closedir(d);
162 return ncount;
163}

165/*
166 * WHAT: Read a marker's recorded final path into *final (NUL-terminated, trailing
167 * whitespace trimmed), validating it is a sane absolute path.
168 *
169 * WHY: A marker's content is the final absolute path to publish the partial to;
170 * an unreadable, empty, or non-absolute marker is corrupt and must not drive
171 * a commit.
172 *
173 * HOW: open O_RDONLY|O_NOFOLLOW, read up to cap-1 bytes, trim trailing CR/LF/space,
174 * and require a leading '/'. Returns NGX_OK with *final populated, or
175 * NGX_ERROR (the caller drops the marker).
176 *
177 * Parameters:
178 * marker — absolute path to the ".commit" marker file
179 * final — output buffer for the recorded path
180 * cap — size of final
181 */
182static ngx_int_t
183reap_read_marker_target(const char *marker, char *final, size_t cap)
184{
185 int mfd;
186 ssize_t r;
187
188 mfd = open(marker, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
189 if (mfd < 0) {
190 return NGX_ERROR;
191 }
192 r = read(mfd, final, cap - 1);
193 close(mfd);
194 if (r <= 0) {
195 return NGX_ERROR;
196 }
197 final[r] = '\0';
198 while (r > 0 && (final[r - 1] == '\n' || final[r - 1] == '\r'
199 || final[r - 1] == ' ')) {
200 final[--r] = '\0';
201 }
202 if (final[0] != '/') { /* sanity: must be absolute */
203 return NGX_ERROR;
204 }
205 return NGX_OK;
206}

208/*
209 * WHAT: Reap one snapshotted marker basename — finish its pending stage-out and
210 * remove the marker. Returns 1 if a commit completed this call, else 0.
211 *
212 * WHY: Isolating the per-marker decision (build paths, read target, check the
213 * partial still exists, commit) keeps brix_stage_reap_dir a flat scan+loop.
214 * Idempotent: a marker whose partial is gone (committed by a racing pass or
215 * a client retry) is just dropped.
216 *
217 * HOW: Build the marker + partial paths under stage_dir; read the target; if the
218 * partial is missing drop the marker; else brix_commit_staged (no fd →
219 * final_mode unused, fchmod is fd-guarded, partial keeps its on-disk mode).
220 * On success unlink the marker + NOTICE-log; on failure WARN and keep it for
221 * retry.
222 *
223 * Parameters:
224 * stage_dir — directory holding the marker + partial
225 * name — ".commit" marker basename to reap
226 * log — nginx log for the NOTICE / WARN result line
227 */
228static ngx_uint_t
229reap_one_marker(const char *stage_dir, const char *name, ngx_log_t *log)
230{
231 char marker[PATH_MAX], partial[PATH_MAX], final[PATH_MAX];
232 size_t slen = sizeof(BRIX_STAGE_COMMIT_SUFFIX) - 1;
233 size_t nlen = strlen(name);
234 int n;
235 struct stat sb;
236
237 n = snprintf(marker, sizeof(marker), "%s/%s", stage_dir, name);
238 if (n < 0 || (size_t) n >= sizeof(marker)) { return 0; }
239 n = snprintf(partial, sizeof(partial), "%s/%.*s", stage_dir,
240 (int) (nlen - slen), name);
241 if (n < 0 || (size_t) n >= sizeof(partial)) { return 0; }
242
243 if (reap_read_marker_target(marker, final, sizeof(final)) != NGX_OK) {
244 (void) unlink(marker);
245 return 0;
246 }
247
248 if (stat(partial, &sb) != 0) {
249 /* Partial already gone (committed elsewhere) — drop the marker. */
250 (void) unlink(marker);
251 return 0;
252 }
253 /* Crash-recovery: no open fd, so final_mode is unused (fchmod is fd-guarded);
254 * the recovered partial keeps its on-disk mode. */
255 if (brix_commit_staged(NGX_INVALID_FILE, partial, final, 0, log) == NGX_OK) {
256 (void) unlink(marker);
257 ngx_log_error(NGX_LOG_NOTICE, log, 0,
258 "brix: completed pending stage-out \"%s\" -> \"%s\"",
259 partial, final);
260 return 1;
261 }
262 ngx_log_error(NGX_LOG_WARN, log, errno,
263 "brix: pending stage-out \"%s\" -> \"%s\" failed; will retry",
264 partial, final);
265 return 0;
266}

268/* Finish every pending stage-out recorded in stage_dir. Returns the count of
269 * commits completed this pass. Idempotent and crash-safe: a marker whose partial
270 * is already gone (committed by a racing pass / a client retry) is just dropped. */
271ngx_uint_t
272brix_stage_reap_dir(const char *stage_dir, ngx_log_t *log)
273{
274 /* Snapshot marker basenames first — we unlink while iterating, which would
275 * otherwise make readdir skip entries. */
276 char names[BRIX_STAGE_REAP_MAX_MARKERS][BRIX_STAGE_REAP_NAME_MAX];
277 ngx_uint_t done = 0, ncount, i;
278
279 if (stage_dir == NULL || stage_dir[0] == '\0') {
280 return 0;
281 }
282 ncount = reap_collect_markers(stage_dir, names,
283 BRIX_STAGE_REAP_MAX_MARKERS);
284
285 for (i = 0; i < ncount; i++) {
286 done += reap_one_marker(stage_dir, names[i], log);
287 }
288 return done;
289}
