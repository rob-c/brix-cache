1/*
2 * client/lib/cli_cred.c — CLI-to-credential-store builder.
3 *
4 * WHAT: maps per-tool CLI values (proxy path, bearer token, S3 keys, OIDC
5 * account, auto-refresh flag) into an brix_cred_config and returns a
6 * live brix_cred_store.
7 * WHY: every connecting tool (xrdcp, xrdfs, xrootdfs) needs to perform this
8 * same mapping — NULL/empty field → fall back to env/default discovery —
9 * without duplicating the logic or pulling cred.h into every app.
10 * HOW: each field is only populated when the caller supplies a non-NULL,
11 * non-empty string, so empty string == "not set" and the per-handler
12 * env/default discovery path runs unchanged. The function is a pure
13 * builder: it allocates nothing itself — brix_cred_store_new does — and
14 * it has no side effects beyond that allocation.
15 */

21/*
22 * Credential-handler linkage anchor.
23 *
24 * brix_cred_store_new() discovers handlers through WEAK accessors (cred.c). A
25 * weak *undefined* reference does NOT pull the defining object out of the static
26 * libbrix.a, so a statically-linked tool would otherwise build a store with NO
27 * handlers — brix_cred_available() then always returns 0 and GSI/token/sss are
28 * silently skipped (this was the cause of an "no usable auth protocol"
29 * regression in xrdcp). Referencing the accessors STRONGLY here — from the one
30 * TU every cred-store tool already links (brix_cli_cred_store_build) — forces
31 * cred_x509.o / cred_bearer.o / cred_sss.o / cred_krb5.o / cred_s3.o into the
32 * link so the store is actually populated. The accessors stay weak elsewhere,
33 * so a tool that does NOT use the CLI cred store is unaffected.
34 */

47/*
48 * brix_cli_cred_store_build — build a credential store from CLI/env values.
49 *
50 * WHAT: constructs an brix_cred_config from the caller's parsed CLI values and
51 * delegates to brix_cred_store_new.
52 * WHY: centralises the NULL/empty-guard logic so each tool front-end provides
53 * its parsed values without needing to know the config field names.
54 * HOW: each field is copied into the config only when the source string is
55 * non-NULL and non-empty; otherwise the field stays NULL so the per-kind
56 * handler falls back to its env/default discovery path. auto_refresh is
57 * forwarded directly (it's an int flag, not a string).
58 *
59 * Returns a new store on success, or NULL on OOM (brix_cred_store_new failure).
60 * The caller frees with brix_cred_store_free.
61 */
62struct brix_cred_store *
63brix_cli_cred_store_build(const char *proxy,
64 const char *bearer,
65 const char *bearer_file,
66 const char *s3_access,
67 const char *s3_secret,
68 const char *oidc_account,
69 int auto_refresh)
70{
71 brix_cred_config cfg;
72 memset(&cfg, 0, sizeof(cfg));
73
74 /* Only propagate non-NULL, non-empty strings so that an absent/empty CLI
75 * flag leaves the field NULL — letting the per-kind handler discover the
76 * credential from its env/default path instead. */
77 if (proxy != NULL && *proxy != '\0') cfg.proxy_path = proxy;
78 if (bearer != NULL && *bearer != '\0') cfg.bearer_literal = bearer;
79 if (bearer_file != NULL && *bearer_file != '\0') cfg.bearer_path = bearer_file;
80 if (s3_access != NULL && *s3_access != '\0') cfg.s3_access = s3_access;
81 if (s3_secret != NULL && *s3_secret != '\0') cfg.s3_secret = s3_secret;
82 if (oidc_account != NULL && *oidc_account != '\0') cfg.oidc_account = oidc_account;
83 cfg.auto_refresh = auto_refresh;
84
85 return brix_cred_store_new(&cfg);
86}
