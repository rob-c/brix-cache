1/*
2 * pblock_quota.h — Phase-83 F5 quotas + space accounting for the pblock driver.
3 *
4 * WHAT: A `usage(scope, id, bytes, inodes)` rollup table maintained by SQLite
5 * triggers on `objects` — transactional by construction (every catalog
6 * insert/update/delete adjusts the rollup inside the same statement
7 * window), so no driver call site can forget it. Scopes: 'total' (id 0),
8 * 'uid', 'gid'. On top of it: admission checks (EDQUOT before bytes are
9 * accepted) at the catalog boundaries — open-create, staged open/commit,
10 * mkdir, server_copy, and the close/fsync size write-back.
11 *
12 * WHY: Deterministic ENOSPC/EDQUOT/507/kXR_NoSpace tests without filling a
13 * disk, per-uid quota tests for the P80 multiuser model, and honest
14 * numbers for the `space` seam slot (kXR_statvfs / SRR).
15 *
16 * HOW: All work happens at metadata boundaries (the hot byte path is
17 * untouched). Static limits ride the `?tail` opts (`quota=`,
18 * `quota_inodes=`); per-uid limits are runtime `ctl` rows
19 * (`quota.uid.<n>` = bytes, size suffixes allowed). Enabled only when a
20 * quota opt is present — otherwise no table, no triggers, byte-for-byte
21 * production catalog writes. ngx-free; BRIX_HAVE_SQLITE-gated like the
22 * rest of the backend.
23 *
24 * Requires: pblock_store.h (pblock_state_t) before inclusion.
25 */

31/* Create the usage table + maintenance triggers and rebuild the rollup from
32 * the current objects rows (covers enabling quota on an existing export).
33 * 0 on success, -1 on a sqlite failure (quota stays off — fail-safe). */
34int pblock_quota_init(pblock_state_t *st);

36/* Admission: would adding (add_bytes, add_inodes) attributed to `uid` exceed
37 * the export quota (static opts) or the uid's `quota.uid.<n>` ctl limit?
38 * 0 = admitted; -1 with errno=EDQUOT when a limit would be exceeded. Negative
39 * deltas (shrink/replace) are always admitted. */
40int pblock_quota_admit(const pblock_state_t *st, uint32_t uid,
41 int64_t add_bytes, int64_t add_inodes);
42
43/* Largest size a handle opened on a `cur_size`-byte file owned by `uid` may
44 * grow it to — the export byte quota and the uid's ctl limit, whichever leaves
45 * less room. INT64_MAX when no byte limit applies. Snapshotted into the handle
46 * at open so the write hot path enforces quota without touching the DB. */
47int64_t pblock_quota_max_size(const pblock_state_t *st, uint32_t uid,
48 int64_t cur_size);
49
50/* Admission for the close/fsync size write-back on `path`: computes the growth
51 * vs the currently-stored row size and admits it. 0 or -1/errno=EDQUOT. */
52int pblock_quota_touch_admit(const pblock_state_t *st, const char *path,
53 uint32_t uid, int64_t newsize);
54
55/* Read one rollup row (scope 'total'/'uid'/'gid'); absent row reads as zeros.
56 * 0 on success, -1 on a sqlite failure. */
57int pblock_quota_usage(const pblock_state_t *st, const char *scope, int64_t id,
58 int64_t *bytes, int64_t *inodes);
