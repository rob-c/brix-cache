1/*
2 * vfs_unlink.c — VFS delete family (unlink / rmdir).
3 *
4 * WHAT: Implements brix_vfs_unlink() (remove a regular file) and
5 * brix_vfs_rmdir() (remove a directory, recursively or only when empty),
6 * both thin wrappers over the shared brix_vfs_delete() helper.
7 *
8 * WHY: Deletes are write-gated and must be applied through the namespace layer
9 * (../compat/namespace_ops) so confinement and the recursive / require-
10 * empty semantics match the rest of the namespace mutators, with one
11 * metric/access-log emission per call.
12 *
13 * HOW: brix_vfs_delete() enforces brix_vfs_require_write() and a non-NULL
14 * root_canon, builds an brix_ns_delete_opts_t (recursive,
15 * require_empty_dir), and calls brix_ns_delete(); the namespace status
16 * is mapped back to errno (sys_errno or EIO) and observed as
17 * BRIX_METRIC_OP_DELETE. rmdir requests require_empty_dir only when not
18 * recursive.
19 */
20#include "vfs_internal.h"

22/* brix_vfs_driver_rmtree — depth-first delete of `logical` through the storage
23 * driver: a file is unlinked directly; a directory has its children removed
24 * (opendir/readdir recursion) before the now-empty directory itself. Generic over
25 * any driver that implements stat/opendir/readdir/unlink. NGX_OK or NGX_ERROR. */
26static ngx_int_t
27brix_vfs_driver_rmtree(brix_vfs_ctx_t *ctx, const brix_sd_driver_t *drv,
28 const char *logical)
29{
30 brix_sd_stat_t st;
31
32 if (drv->stat == NULL || drv->unlink == NULL) {
33 errno = ENOSYS;
34 return NGX_ERROR;
35 }
36 if (drv->stat(ctx->sd, logical, &st) != NGX_OK) {
37 return NGX_ERROR; /* ENOENT etc. — errno set by the driver */
38 }
39
40 if (st.is_dir && drv->opendir != NULL) {
41 brix_sd_dir_t *dir;
42 int err = 0;
43
44 dir = drv->opendir(ctx->sd, logical, &err);
45 if (dir != NULL) {
46 brix_sd_dirent_t de;
47 ngx_int_t drc;
48
49 while ((drc = drv->readdir(dir, &de)) == NGX_OK) {
50 char child[PATH_MAX];
51
52 ngx_snprintf((u_char *) child, sizeof(child), "%s/%s%Z",
53 (logical[0] == '/' && logical[1] == '\0')
54 ? "" : logical,
55 de.name);
56 if (brix_vfs_driver_rmtree(ctx, drv, child) != NGX_OK) {
57 drv->closedir(dir);
58 return NGX_ERROR;
59 }
60 }
61 drv->closedir(dir);
62 if (drc == NGX_ERROR) {
63 return NGX_ERROR;
64 }
65 }
66 return drv->unlink(ctx->sd, logical, 1);
67 }
68
69 return drv->unlink(ctx->sd, logical, 0);
70}

72/* Shared delete body for unlink/rmdir: write-gate, then brix_ns_delete with
73 * the given recursive / require_empty_dir options; metered as OP_DELETE. */
74static ngx_int_t
75brix_vfs_delete(brix_vfs_ctx_t *ctx, unsigned recursive,
76 unsigned require_empty_dir)
77{
78 brix_ns_delete_opts_t opts;
79 brix_ns_result_t res;
80 const char *path;
81 uint64_t start;
82 int saved_errno;
83 const brix_sd_driver_t *drv;
84
85 start = brix_vfs_now_ns();
86 path = brix_vfs_ctx_path(ctx);
87
88 if (brix_vfs_require_write(ctx) != NGX_OK) {
89 saved_errno = errno;
90 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
91 NGX_ERROR, saved_errno, start);
92 return NGX_ERROR;
93 }
94
95 if (ctx->root_canon == NULL) {
96 errno = EINVAL;
97 saved_errno = errno;
98 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
99 NGX_ERROR, saved_errno, start);
100 return NGX_ERROR;
101 }
102
103 /* Non-POSIX backend: delete through the driver namespace. A recursive delete
104 * (WebDAV DELETE of a collection) walks the tree; a non-recursive delete is a
105 * file unlink or empty-rmdir (require_directory selects is_dir). */
106 drv = brix_vfs_ctx_driver(ctx);
107 if (drv != NULL) {
108 const char *logical = brix_vfs_export_relative(ctx, path);
109 ngx_int_t rc;
110
111 if (recursive) {
112 rc = brix_vfs_driver_rmtree(ctx, drv, logical);
113 } else if (drv->unlink != NULL) {
114 rc = drv->unlink(ctx->sd, logical, require_empty_dir ? 1 : 0);
115 } else {
116 errno = ENOSYS;
117 rc = NGX_ERROR;
118 }
119 saved_errno = errno;
120 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
121 rc, saved_errno, start);
122 return rc;
123 }
124
125 ngx_memzero(&opts, sizeof(opts));
126 opts.recursive = recursive ? 1 : 0;
127 opts.require_empty_dir = require_empty_dir ? 1 : 0;
128 /* A non-recursive "empty dir" delete is rmdir, which must reject a regular
129 * file (ENOTDIR) — kXR_rmdir parity. Recursive deletes (require_empty_dir=0)
130 * remove a file directly, so require_directory stays off there. */
131 opts.require_directory = require_empty_dir ? 1 : 0;
132
133 res = brix_ns_delete(ctx->log, ctx->root_canon,
134 brix_vfs_ctx_path(ctx), &opts);
135 if (res.status == BRIX_NS_OK) {
136 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
137 NGX_OK, 0, start);
138 return NGX_OK;
139 }
140
141 errno = res.sys_errno != 0 ? res.sys_errno
142 : brix_vfs_ns_status_errno(res.status);
143 saved_errno = errno;
144 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
145 NGX_ERROR, saved_errno, start);
146 return NGX_ERROR;
147}
