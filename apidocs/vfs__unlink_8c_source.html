1/*
2 * vfs_unlink.c — VFS delete family (unlink / rmdir).
3 *
4 * WHAT: Implements brix_vfs_unlink() (remove a regular file) and
5 * brix_vfs_rmdir() (remove a directory, recursively or only when empty),
6 * both thin wrappers over the shared brix_vfs_delete() helper.
7 *
8 * WHY: Deletes are write-gated and must be applied through the namespace layer
9 * (../compat/namespace_ops) so confinement and the recursive / require-
10 * empty semantics match the rest of the namespace mutators, with one
11 * metric/access-log emission per call.
12 *
13 * HOW: brix_vfs_delete() enforces brix_vfs_require_write() and a non-NULL
14 * root_canon, builds an brix_ns_delete_opts_t (recursive,
15 * require_empty_dir), and calls brix_ns_delete(); the namespace status
16 * is mapped back to errno (sys_errno or EIO) and observed as
17 * BRIX_METRIC_OP_DELETE. rmdir requests require_empty_dir only when not
18 * recursive.
19 */
20#include "vfs_internal.h"

22/* brix_vfs_driver_rmtree — depth-first delete of `logical` through the storage
23 * driver: a file is unlinked directly; a directory has its children removed
24 * (opendir/readdir recursion) before the now-empty directory itself.
25 *
26 * WHAT: Recursive WebDAV DELETE of a collection on a non-POSIX backend:
27 * dispatches stat/opendir/readdir/unlink through the LEAF driver so that
28 * per-user credentials (stat_cred / unlink_cred / opendir_cred) are
29 * presented at every level of the tree, not just the top-level call.
30 *
31 * WHY: Calling drv->stat(ctx->sd, …) / drv->unlink(ctx->sd, …) bypasses the
32 * cred entirely because ctx->sd may be a decorator without *_cred slots.
33 * Dispatching through brix_sd_*_maybe_cred on the leaf ensures that a
34 * user credential, when present (use_cred=1), is threaded through the
35 * entire recursive walk — not just the outermost call.
36 *
37 * HOW: Accepts the pre-resolved leaf `leaf` and pre-computed `cred` pointer
38 * (NULL when use_cred=0) from the caller, uses brix_sd_*_maybe_cred on
39 * `leaf` for every driver call, and recurses with the same arguments. */
40static ngx_int_t
41brix_vfs_driver_rmtree(brix_sd_instance_t *leaf, const brix_sd_driver_t *drv,
42 const char *logical, const brix_sd_cred_t *cred)
43{
44 brix_sd_stat_t st;
45
46 if (drv->stat == NULL || drv->unlink == NULL) {
47 errno = ENOSYS;
48 return NGX_ERROR;
49 }
50 if (brix_sd_stat_maybe_cred(leaf, logical, &st, cred) != NGX_OK) {
51 return NGX_ERROR; /* ENOENT etc. — errno set by the driver */
52 }
53
54 if (st.is_dir && drv->opendir != NULL) {
55 brix_sd_dir_t *dir;
56 int err = 0;
57
58 dir = brix_sd_opendir_maybe_cred(leaf, logical, &err, cred);
59 if (dir != NULL) {
60 brix_sd_dirent_t de;
61 ngx_int_t drc;
62
63 while ((drc = drv->readdir(dir, &de)) == NGX_OK) {
64 char child[PATH_MAX];
65
66 ngx_snprintf((u_char *) child, sizeof(child), "%s/%s%Z",
67 (logical[0] == '/' && logical[1] == '\0')
68 ? "" : logical,
69 de.name);
70 if (brix_vfs_driver_rmtree(leaf, drv, child, cred) != NGX_OK) {
71 drv->closedir(dir);
72 return NGX_ERROR;
73 }
74 }
75 drv->closedir(dir);
76 if (drc == NGX_ERROR) {
77 return NGX_ERROR;
78 }
79 }
80 return brix_sd_unlink_maybe_cred(leaf, logical, 1, cred);
81 }
82
83 return brix_sd_unlink_maybe_cred(leaf, logical, 0, cred);
84}

86/* brix_vfs_delete_via_driver — perform a delete through a non-POSIX storage
87 * driver's namespace, presenting per-user credentials at every level.
88 *
89 * WHAT: Deletes `path` through `drv` (already confirmed non-NULL by the
90 * caller): a recursive request walks the collection tree, a
91 * non-recursive request unlinks a file or empty directory. Returns
92 * NGX_OK on success and NGX_ERROR (with errno set) on any failure,
93 * emitting exactly one BRIX_METRIC_OP_DELETE observation for `start`.
94 *
95 * WHY: Object/S3 backends have no POSIX namespace, so the delete must be
96 * dispatched on the LEAF instance where the driver's *_cred slots live
97 * (decorators carry only plain relays). Threading the resolved user
98 * credential through the entire walk keeps confinement identical to the
99 * top-level call — a partial-credential walk would delete children as
100 * the wrong identity.
101 *
102 * HOW: 1. Resolve the export-relative logical path and the leaf instance.
103 * 2. Zero the credential struct BEFORE the gate so an inactive kind is
104 * never handed to the driver as a dangling pointer.
105 * 3. When the cred gate is active, resolve the user credential; on
106 * failure observe EACCES (or the gate's errno) and return NGX_ERROR.
107 * 4. Recursive → brix_vfs_driver_rmtree; else unlink via the driver
108 * (require_empty_dir selects directory semantics), or ENOSYS when the
109 * driver has no unlink.
110 * 5. Observe the result and return it. */
111static ngx_int_t
112brix_vfs_delete_via_driver(brix_vfs_ctx_t *ctx, const brix_sd_driver_t *drv,
113 unsigned recursive, unsigned require_empty_dir, const char *path,
114 uint64_t start)
115{
116 const char *logical = brix_vfs_export_relative(ctx, path);
117 brix_sd_instance_t *leaf = brix_vfs_ns_leaf(ctx->sd);
118 brix_sd_ucred_t store;
119 brix_sd_cred_t cred;
120 ngx_int_t rc;
121 int saved_errno;
122 int use_cred = 0, cred_err = 0;
123
124 /* Zero before the gate: it fills only the active credential kind; an
125 * unzeroed cred hands a garbage inactive pointer to the driver's
126 * cred slot (bearer PASSTHROUGH would leave x509_proxy dangling). */
127 ngx_memzero(&cred, sizeof(cred));
128
129 if (brix_vfs_cred_gate_active(ctx)) {
130 if (brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err)
131 != NGX_OK)
132 {
133 saved_errno = cred_err ? cred_err : EACCES;
134 errno = saved_errno;
135 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE,
136 NULL, 0, NGX_ERROR, saved_errno,
137 start);
138 return NGX_ERROR;
139 }
140 }
141
142 if (recursive) {
143 rc = brix_vfs_driver_rmtree(leaf, drv, logical,
144 use_cred ? &cred : NULL);
145 } else if (drv->unlink != NULL) {
146 rc = brix_sd_unlink_maybe_cred(leaf, logical,
147 require_empty_dir ? 1 : 0, use_cred ? &cred : NULL);
148 } else {
149 errno = ENOSYS;
150 rc = NGX_ERROR;
151 }
152 saved_errno = errno;
153 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
154 rc, saved_errno, start);
155 return rc;
156}

158/* brix_vfs_delete_via_namespace — perform a delete through the POSIX namespace
159 * layer (brix_ns_delete) under the confined root_canon.
160 *
161 * WHAT: Builds a brix_ns_delete_opts_t from `recursive` / `require_empty_dir`
162 * and calls brix_ns_delete on the confined root. Returns NGX_OK on
163 * success and NGX_ERROR (with errno mapped from the namespace status) on
164 * failure, emitting exactly one BRIX_METRIC_OP_DELETE observation.
165 *
166 * WHY: This is the default (POSIX) backend path; it must apply the same
167 * recursive / require-empty semantics as the driver path while keeping
168 * confinement in brix_ns_delete's hands (root_canon + logical path).
169 *
170 * HOW: 1. Zero opts and set recursive / require_empty_dir.
171 * 2. require_directory tracks require_empty_dir: a non-recursive
172 * empty-dir delete is rmdir and must reject a regular file (ENOTDIR),
173 * while a recursive delete removes a file directly.
174 * 3. Call brix_ns_delete; on BRIX_NS_OK observe success and return
175 * NGX_OK.
176 * 4. Otherwise map sys_errno (or the status) to errno, observe the
177 * error, and return NGX_ERROR. */
178static ngx_int_t
179brix_vfs_delete_via_namespace(brix_vfs_ctx_t *ctx, unsigned recursive,
180 unsigned require_empty_dir, const char *path, uint64_t start)
181{
182 brix_ns_delete_opts_t opts;
183 brix_ns_result_t res;
184 int saved_errno;
185
186 ngx_memzero(&opts, sizeof(opts));
187 opts.recursive = recursive ? 1 : 0;
188 opts.require_empty_dir = require_empty_dir ? 1 : 0;
189 /* A non-recursive "empty dir" delete is rmdir, which must reject a regular
190 * file (ENOTDIR) — kXR_rmdir parity. Recursive deletes (require_empty_dir=0)
191 * remove a file directly, so require_directory stays off there. */
192 opts.require_directory = require_empty_dir ? 1 : 0;
193
194 res = brix_ns_delete(ctx->log, ctx->root_canon,
195 brix_vfs_ctx_path(ctx), &opts);
196 if (res.status == BRIX_NS_OK) {
197 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
198 NGX_OK, 0, start);
199 return NGX_OK;
200 }
201
202 errno = res.sys_errno != 0 ? res.sys_errno
203 : brix_vfs_ns_status_errno(res.status);
204 saved_errno = errno;
205 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
206 NGX_ERROR, saved_errno, start);
207 return NGX_ERROR;
208}

210/* Shared delete body for unlink/rmdir: write-gate, then dispatch to the driver
211 * or POSIX namespace delete with the given recursive / require_empty_dir
212 * options; metered as OP_DELETE.
213 *
214 * WHAT: Enforces the write gate and a non-NULL root_canon, then routes the
215 * delete to brix_vfs_delete_via_driver (non-POSIX backend present) or
216 * brix_vfs_delete_via_namespace (POSIX default). Returns NGX_OK on
217 * success and NGX_ERROR (with errno set) on any failure; each failure
218 * gate emits its own BRIX_METRIC_OP_DELETE observation.
219 *
220 * WHY: unlink and rmdir share identical gating and observation; centralising
221 * it here keeps confinement (write gate + root_canon) uniform and leaves
222 * only the backend-specific mechanics in the two dispatch helpers.
223 *
224 * HOW: 1. Snapshot start time and the logical path.
225 * 2. Fail closed on write-gate denial or a NULL root_canon, observing
226 * each error.
227 * 3. If a storage driver is bound, delegate to the driver helper.
228 * 4. Otherwise delegate to the POSIX namespace helper. */
229static ngx_int_t
230brix_vfs_delete(brix_vfs_ctx_t *ctx, unsigned recursive,
231 unsigned require_empty_dir)
232{
233 const char *path;
234 uint64_t start;
235 int saved_errno;
236 const brix_sd_driver_t *drv;
237
238 start = brix_vfs_now_ns();
239 path = brix_vfs_ctx_path(ctx);
240
241 if (brix_vfs_require_write(ctx) != NGX_OK) {
242 saved_errno = errno;
243 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
244 NGX_ERROR, saved_errno, start);
245 return NGX_ERROR;
246 }
247
248 if (ctx->root_canon == NULL) {
249 errno = EINVAL;
250 saved_errno = errno;
251 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_DELETE, NULL, 0,
252 NGX_ERROR, saved_errno, start);
253 return NGX_ERROR;
254 }
255
256 /* Non-POSIX backend: delete through the driver namespace. A recursive delete
257 * (WebDAV DELETE of a collection) walks the tree; a non-recursive delete is a
258 * file unlink or empty-rmdir (require_directory selects is_dir).
259 * Dispatch on the leaf instance so *_maybe_cred forwarders find the leaf
260 * driver's *_cred slots (decorators have only plain relays). */
261 drv = brix_vfs_ctx_driver(ctx);
262 if (drv != NULL) {
263 return brix_vfs_delete_via_driver(ctx, drv, recursive,
264 require_empty_dir, path, start);
265 }
266
267 return brix_vfs_delete_via_namespace(ctx, recursive, require_empty_dir,
268 path, start);
269}
