1/*
2 * list_objects_v1.c — ListObjects (V1) XML response builder.
3 *
4 * WHAT: Builds the ListBucketResult XML for the original S3 ListObjects (V1)
5 * operation — GET /<bucket> with no `list-type=2`. Rucio's S3 RSE and older
6 * tooling (some boto2-era clients, a few mirror utilities) still issue V1.
7 *
8 * WHY: V1 and V2 walk and paginate the same filesystem the same way; they
9 * differ ONLY in their request pagination parameter and their response
10 * element names:
11 * - V1 paginates with `marker` (the exclusive start key) and reports
12 * `<Marker>` / `<NextMarker>`; there is no `<KeyCount>` and no
13 * continuation token.
14 * - V2 paginates with `continuation-token` / `start-after` and reports
15 * `<KeyCount>` + `<NextContinuationToken>` (see list_objects_v2.c).
16 * The directory walk (s3_walk) and the lexicographic comparator (entry_cmp)
17 * are already factored into list_walk.c and are shared verbatim — this file
18 * only adds the V1 pagination semantics and the V1 XML dialect.
19 *
20 * HOW:
21 * 1. Parse query params: prefix, delimiter, marker, max-keys, encoding-type.
22 * 2. Walk the bucket root via s3_walk(), qsort() by key.
23 * 3. Skip entries <= marker (V1 marker is exclusive), slice at max-keys.
24 * 4. Emit ListBucketResult with Marker / NextMarker (when truncated) and the
25 * Contents / CommonPrefixes loop (identical shape to V2).
26 */
27

55 brix_http_query_get(r->args, "marker", (char *) marker_buf, sizeof(marker_buf), S3_LIST_QUERY_FLAGS);
56 brix_http_query_get(r->args, "encoding-type", (char *) encoding_type_buf, sizeof(encoding_type_buf), S3_LIST_QUERY_FLAGS);

59 url_encode = 1;
60 }
61
62 max_keys = s3_list_parse_max_keys(r, (int) cf->max_keys);
63
64 if (s3_list_collect_sorted(r, cf, (const char *) prefix_buf,
65 (const char *) delimiter_buf,
66 &items, &total) != NGX_OK)
67 {
68 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
69 return NGX_HTTP_INTERNAL_SERVER_ERROR;
70 }
71
72 /* V1 marker is the exclusive start key — same skip semantics as V2. */
73 truncated = s3_list_paginate(items, total, (const char *) marker_buf,
74 max_keys, &start_idx, &end_idx);
75
76 xml_capacity = 512
77 + (size_t) cf->bucket.len + 32
78 + strlen((const char *) prefix_buf) * 6 + 32
79 + strlen((const char *) delimiter_buf) * 6 + 32
80 + strlen((const char *) marker_buf) * 6 + 32
81 + (size_t) (end_idx - start_idx)
82 * (S3_MAX_KEY * 6 + 512);
83
84 xml = ngx_palloc(r->pool, xml_capacity);
85 if (xml == NULL) {

87 return NGX_HTTP_INTERNAL_SERVER_ERROR;
88 }
89

91 XML_APPEND("<ListBucketResult "
92 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">");
93 XML_APPEND_ELEM("Name", cf->bucket.data, cf->bucket.len);
94 XML_APPEND_ELEM("Prefix", prefix_buf, strlen((const char *) prefix_buf));
95 XML_APPEND_ELEM("Marker", marker_buf, strlen((const char *) marker_buf));
96 XML_APPEND("<MaxKeys>%d</MaxKeys>", max_keys);
97
98 if (delimiter_buf[0] != '\0') {
99 XML_APPEND_ELEM("Delimiter", delimiter_buf,
100 strlen((const char *) delimiter_buf));
101 }
102 if (url_encode) {
103 XML_APPEND("<EncodingType>url</EncodingType>");
104 }
105 XML_APPEND("<IsTruncated>%s</IsTruncated>", truncated ? "true" : "false");
106
107 /*
108 * NextMarker: AWS returns it only when a delimiter is set, but clients that
109 * paginate without a delimiter fall back to the last key — so emitting it
110 * whenever the response is truncated is a safe superset and lets every V1
111 * client page correctly. The value is the last key in this slice.
112 */
113 if (truncated && end_idx > 0) {
114 XML_APPEND_ELEM("NextMarker", items[end_idx - 1].key,
115 strlen(items[end_idx - 1].key));
116 }
117
118 rc = s3_list_emit_entries(r, cf, items, start_idx, end_idx,
119 url_encode, 0 /* fetch_owner: V1 has no Owner */,
120 xml, &xml_len, xml_capacity,
121 &contents, &prefixes);
122 if (rc != NGX_OK) {
123 return rc;
124 }
125
126 XML_APPEND("</ListBucketResult>");
127
128 return s3_list_finalize(r, xml, xml_len, contents, prefixes, truncated);
129}
