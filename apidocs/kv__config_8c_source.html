1/*
2 * kv_config.c — configuration, registry, and zone initialisation for the
3 * generic cross-worker key/value store (data-plane ops live in kv.c).
4 *
5 * WHAT: Owns everything that runs at configuration/init time rather than on the
6 * request hot path — the `brix_kv_zone` directive setter, the
7 * brix_kv_configure() geometry computation, the slab-backed zone init
8 * callback, and the process-wide zone registry (brix_kv_find /
9 * brix_kv_zone_count / brix_kv_zone_get).
10 * WHY: kv.c held both the request-time table operations (get/set/delete/stats)
11 * and this configuration/registry machinery, pushing it past the 500-line
12 * cap. The two planes are independent — they share only the private
13 * on-slab layout types (kv_internal.h) and no functions call across the
14 * split — so separating them keeps each file focused and under the cap.
15 * HOW: brix_kv_configure() derives the power-of-two bucket count from the
16 * requested size, sizes the slab-backed table, and registers the zone;
17 * brix_kv_init_zone() (the zone `init` callback) allocates the table from
18 * the slab pool via brix_shm_table_alloc() and initialises layout fields
19 * only on a fresh zone; the registry array records every configured zone
20 * so consumer directives and the Prometheus exporter can resolve them.
21 */

37/* ---- Largest power of two not exceeding n ----
38 *
39 * WHAT: Returns the greatest power of two that is <= n, or 0 when n is 0.
40 *
41 * WHY: The bucket count must be a power of two so the home bucket can be
42 * selected with (hash & (capacity-1)) instead of a modulo. Deriving it
43 * from the entry budget once, at configure time, keeps the hot path
44 * branch-free.
45 *
46 * HOW:
47 * 1. Return 0 immediately for n == 0.
48 * 2. Double a uint32 accumulator while it stays below 2^31 and the next
49 * doubling (widened to size_t before the shift, so it cannot wrap) still
50 * fits within n; the 2^31 gate preserves the historical cap.
51 */
52static uint32_t
53brix_kv_floor_pow2(size_t n)
54{
55 uint32_t p = 1;
56
57 if (n == 0) {
58 return 0;
59 }
60 /* Widen BEFORE shifting so the doubling cannot wrap in uint32; the
61 * p < 2^31 gate preserves the old wrap-guard's cap of 2^31. */
62 while (p < 0x80000000u && ((size_t) p << 1) <= n) {
63 p <<= 1;
64 }
65 return p;
66}

69/* ---- Zone init callback: allocate the table from the slab pool ----
70 *
71 * WHAT: nginx `init` callback for a brix_kv zone. Allocates (or re-attaches to)
72 * the slab-backed table, builds the process-local mutex handle, and on a
73 * fresh table initialises the immutable layout fields. Returns NGX_OK on
74 * success, NGX_ERROR when the slab allocation fails.
75 *
76 * WHY: The table MUST be allocated FROM the slab pool (not laid over shm.addr),
77 * so nginx's ngx_slab_pool_t header survives ngx_unlock_mutexes() on child
78 * death. Live counters (count/hits/misses/evictions) must persist across a
79 * reload/re-attach, so they are only reset on a genuinely fresh table.
80 *
81 * HOW:
82 * 1. Delegate fresh-alloc / reload / re-attach and mutex creation to
83 * brix_shm_table_alloc(), which publishes the table via shm_zone->data.
84 * 2. On failure return NGX_ERROR.
85 * 3. When *fresh, set capacity/key_max/val_max from the config handle; leave
86 * the already-zeroed live counters untouched.
87 */
88static ngx_int_t
89brix_kv_init_zone(ngx_shm_zone_t *shm_zone, void *data)
90{
91 brix_kv_t *kv = shm_zone->data; /* set by configure */
92 brix_kv_header_t *h;
93 ngx_flag_t fresh;
94
95 /*
96 * Allocate the table FROM the slab pool (not over shm.addr), so nginx's
97 * ngx_slab_pool_t header survives ngx_unlock_mutexes() on child death.
98 * The helper handles fresh-alloc, reload (data != NULL), and re-attach
99 * (shm.exists), publishes the table via shm_zone->data, and creates the
100 * process-local mutex handle from the table's first member (the lock).
101 */
102 h = brix_shm_table_alloc(shm_zone, data, kv->table_bytes,
103 &kv->mutex, &fresh);
104 if (h == NULL) {
105 return NGX_ERROR;
106 }
107
108 if (fresh) {
109 /* Brand-new table: initialise the layout fields. The helper already
110 * zeroed the region and created the mutex, so live counters
111 * (count/hits/misses/evictions) start at 0 and must NOT be reset on
112 * reuse. */
113 h->capacity = kv->capacity;
114 h->key_max = (uint32_t) kv->key_max;
115 h->val_max = (uint32_t) kv->val_max;
116 }
117 return NGX_OK;
118}

120/* ---- Register a brix_kv zone during configuration ----
121 *
122 * WHAT: Computes the zone geometry (bucket count, table byte size, padded zone
123 * size), fills the caller-allocated brix_kv_t handle, adds the shared
124 * memory zone, and records the zone in the registry. Returns NGX_OK, or
125 * NGX_ERROR on registry overflow, too-small size, or a duplicate zone.
126 *
127 * WHY: The bucket count is derived from the requested size exactly as before
128 * (size buys capacity), but the table now lives in slab memory, so the
129 * zone actually requested is padded by brix_shm_zone_size() to cover the
130 * ngx_slab_pool_t header, the page-management array, and slab rounding.
131 *
132 * HOW:
133 * 1. Reject when the registry is full; clamp size up to BRIX_KV_MIN_SIZE.
134 * 2. Derive stride, the entry budget, and the power-of-two capacity; reject
135 * when not even one entry fits.
136 * 3. Compute table_bytes and the padded zone_size; populate the handle.
137 * 4. ngx_shared_memory_add(); reject a duplicate (zone->data already set).
138 * 5. Wire the init callback + handle, append to the registry, return NGX_OK.
139 */
140ngx_int_t
141brix_kv_configure(ngx_conf_t *cf, brix_kv_t *kv, ngx_str_t *name,
142 size_t size, size_t key_max, size_t val_max, void *module)
143{
144 size_t avail, stride, table_bytes, zone_size;
145 uint32_t cap;
146
147 if (brix_kv_nzones >= BRIX_KV_MAX_ZONES) {
148 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
149 "too many brix_kv zones (max %d)",
150 BRIX_KV_MAX_ZONES);
151 return NGX_ERROR;
152 }
153 if (size < BRIX_KV_MIN_SIZE) {
154 size = BRIX_KV_MIN_SIZE;
155 }
156
157 /*
158 * Derive the bucket count from the requested size exactly as before
159 * (size buys capacity), then size the slab-backed table region from it.
160 * The table now lives in slab memory, so the zone we actually request is
161 * padded by brix_shm_zone_size() to cover the ngx_slab_pool_t header,
162 * the page-management array, and slab rounding.
163 */
164 stride = sizeof(brix_kv_entry_t) + key_max + val_max;
165 avail = size - sizeof(brix_kv_header_t);
166 cap = brix_kv_floor_pow2(avail / stride);
167 if (cap < 1) {
168 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
169 "brix_kv_zone \"%V\": size too small for one entry",
170 name);
171 return NGX_ERROR;
172 }
173 table_bytes = sizeof(brix_kv_header_t) + (size_t) cap * stride;
174 zone_size = brix_shm_zone_size(table_bytes);
175
176 kv->name = *name;
177 kv->size = size;
178 kv->key_max = key_max;
179 kv->val_max = val_max;
180 kv->capacity = cap;
181 kv->table_bytes = table_bytes;
182
183 kv->zone = ngx_shared_memory_add(cf, name, zone_size, module);
184 if (kv->zone == NULL) {
185 return NGX_ERROR;
186 }
187 if (kv->zone->data != NULL) {
188 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
189 "duplicate brix_kv_zone \"%V\"", name);
190 return NGX_ERROR;
191 }
192
193 kv->zone->init = brix_kv_init_zone;
194 kv->zone->data = kv;
195
196 brix_kv_zones[brix_kv_nzones++] = kv;
197 return NGX_OK;
198}

201/* ---- Resolve a configured zone by name ----
202 *
203 * WHAT: Returns the registered brix_kv_t whose name matches `name`, or NULL
204 * when no such zone exists.
205 *
206 * WHY: Consumer directives (token cache, auth cache, rate limiter) name a zone
207 * by string and must resolve it to the shared handle at configuration time.
208 *
209 * HOW: Linear scan of the registry comparing length then bytes; return the
210 * first match or NULL.
211 */
212brix_kv_t *
213brix_kv_find(const ngx_str_t *name)
214{
215 ngx_uint_t i;
216
217 for (i = 0; i < brix_kv_nzones; i++) {
218 if (brix_kv_zones[i]->name.len == name->len
219 && ngx_strncmp(brix_kv_zones[i]->name.data,
220 name->data, name->len) == 0)
221 {
222 return brix_kv_zones[i];
223 }
224 }
225 return NULL;
226}

258/* ---- `brix_kv_zone` directive setter ----
259 *
260 * WHAT: Parses `brix_kv_zone <name> <size> key=<bytes> val=<bytes>` and
261 * registers the zone. Returns NGX_CONF_OK, or NGX_CONF_ERROR on an
262 * invalid size, unknown/missing key=/val= parameters, or a duplicate name.
263 *
264 * WHY: The directive is valid in both the http{} and stream{} main blocks; the
265 * owning module tag is selected from cf->cmd_type so the zone is attached
266 * to the correct nginx module for shared-memory lifecycle.
267 *
268 * HOW:
269 * 1. Parse the size argument; reject non-positive sizes.
270 * 2. Walk the remaining args collecting key=/val= byte maxima; reject unknown.
271 * 3. Require both key_max and val_max to be present and valid.
272 * 4. Reject a duplicate name; allocate the handle from cf->pool.
273 * 5. Select the owning module by cf->cmd_type and call brix_kv_configure().
274 */
275char *
276brix_kv_zone_directive(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
277{
278 ngx_str_t *value = cf->args->elts;
279 ngx_str_t name = value[1];
280 ssize_t size;
281 size_t key_max = 0;
282 size_t val_max = 0;
283 ngx_uint_t i;
284 brix_kv_t *kv;
285 void *module;
286
287 size = ngx_parse_size(&value[2]);
288 if (size == NGX_ERROR || size <= 0) {
289 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
290 "invalid brix_kv_zone size \"%V\"", &value[2]);
291 return NGX_CONF_ERROR;
292 }
293
294 for (i = 3; i < cf->args->nelts; i++) {
295 if (ngx_strncmp(value[i].data, "key=", 4) == 0) {
296 key_max = ngx_atoi(value[i].data + 4, value[i].len - 4);
297 } else if (ngx_strncmp(value[i].data, "val=", 4) == 0) {
298 val_max = ngx_atoi(value[i].data + 4, value[i].len - 4);
299 } else {
300 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
301 "invalid brix_kv_zone parameter \"%V\"",
302 &value[i]);
303 return NGX_CONF_ERROR;
304 }
305 }
306
307 if (key_max == 0 || key_max == (size_t) NGX_ERROR
308 || val_max == 0 || val_max == (size_t) NGX_ERROR)
309 {
310 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
311 "brix_kv_zone \"%V\" requires key=<bytes> and val=<bytes>",
312 &name);
313 return NGX_CONF_ERROR;
314 }
315
316 if (brix_kv_find(&name) != NULL) {
317 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
318 "duplicate brix_kv_zone \"%V\"", &name);
319 return NGX_CONF_ERROR;
320 }
321
322 BRIX_PCALLOC_OR_RETURN(kv, cf->pool, sizeof(brix_kv_t), NGX_CONF_ERROR);
323
324 module = (cf->cmd_type & NGX_STREAM_MAIN_CONF)
325 ? (void *) &ngx_stream_brix_module
326 : (void *) &ngx_http_brix_webdav_module;
327
328 if (brix_kv_configure(cf, kv, &name, (size_t) size,
329 key_max, val_max, module) != NGX_OK)
330 {
331 return NGX_CONF_ERROR;
332 }
333
334 return NGX_CONF_OK;
335}
