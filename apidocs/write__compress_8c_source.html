1/*
2 * WHAT: brix_write_compressed() serves a kXR_write for a handle opened for
3 * write with the opaque "?xrootd.compress=<codec>" while the server has
4 * brix_write_compress on. The request payload is ONE self-contained
5 * codec frame (the symmetric inverse of W4's per-request read frames); the
6 * server decompresses it under a decompression-bomb guard and streams the
7 * recovered plaintext through the VFS write core at the request offset,
8 * storing the file plaintext on disk. The client thus trades CPU for
9 * upload-wire bytes.
10 *
11 * WHY: This is the write-direction counterpart of W4 (read decompression) and
12 * the last plan workstream (W5). It is deliberately ISOLATED from the
13 * proven write path: brix_handle_write() routes here only when
14 * write_codec != IDENTITY, so the default (uncompressed) write — its AIO
15 * fast path and the write-recovery journal — is untouched and byte-
16 * identical. pgwrite/writev never call this, so the pgwrite per-page
17 * CRC32c invariant is preserved. The decode is on UNTRUSTED client input,
18 * so the bomb guard (absolute cap + expansion ratio) is mandatory.
19 *
20 * HOW: Each kXR_write is an independent whole-frame, so writes stay offset-
21 * addressable (re-writing the same offset reproduces the same plaintext).
22 * The plaintext is produced into a fixed window (rd.write_scratch) and
23 * written chunk-by-chunk at advancing offsets through the VFS core, so
24 * memory stays bounded regardless of the decompressed size. A
25 * synchronous VFS write keeps the opt-in path simple; the default path
26 * keeps its AIO offload.
27 */

40/* Fixed plaintext window streamed through VFS; bounds memory for any output size. */
41#define BRIX_WCMP_WINDOW (1u << 20) /* 1 MiB */
42/* Absolute ceiling on the plaintext produced from ONE write frame — a bomb that
43 * tries to exceed this is rejected (the ratio guard catches the classic case). */
44#define BRIX_WCMP_OUT_CAP (256ULL * 1024 * 1024) /* 256 MiB */

47brix_write_compressed(brix_ctx_t *ctx, ngx_connection_t *c,
48 int idx, int64_t offset, size_t wlen)
49{
50 int fd = ctx->files[idx].fd;
51 uint8_t codec = ctx->files[idx].write_codec;
52 ngx_stream_brix_srv_conf_t *rconf;
53 u_char *win;
54 brix_codec_stream_t *s;
55 brix_codec_guard_t guard;
56 size_t in_pos = 0, produced = 0;
57 off_t cur = (off_t) offset;
58 int io_err = 0, data_err = 0;
59 int saved_errno = 0;
60 char detail[80];
61
62 rconf = ngx_stream_get_module_srv_conf(
63 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
64
65 win = BRIX_GET_SCRATCH(ctx, c, rd.write_scratch, rd.write_scratch_size,
66 BRIX_WCMP_WINDOW);
67 if (win == NULL) {
68 return NGX_ERROR;
69 }
70
71 ngx_memzero(&guard, sizeof(guard));
72 /* Bomb protection here is the ABSOLUTE per-frame output ceiling, not an
73 * expansion ratio: legitimate writes are often highly compressible (logs,
74 * sparse/zero regions, repetitive scientific data exceed 1000:1), so a ratio
75 * guard would false-reject real data. Memory stays bounded by the streaming
76 * window regardless, and the bytes land in the client's OWN quota-limited
77 * file, so the absolute cap is the right and sufficient guard. */
78 guard.out_cap = BRIX_WCMP_OUT_CAP;
79 guard.max_ratio = 0;
80
81 s = brix_codec_open((brix_codec_id_t) codec,
82 BRIX_CODEC_DIR_DECOMPRESS, -1, &guard);
83 if (s == NULL) {
84 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
85 ctx->files[idx].path, "-",
86 kXR_IOError, "inline write decompression unavailable");
87 }
88
89 for ( ;; ) {
90 size_t out_pos = 0, prev_in = in_pos;
91 brix_codec_rc_t rc = brix_codec_step(s, ctx->recv.payload, wlen, &in_pos,
92 win, BRIX_WCMP_WINDOW,
93 &out_pos, 1 /* finish */);
94 if (out_pos > 0) {
95 brix_vfs_job_t job;
96
97 brix_vfs_job_write_init(&job, fd, cur, win, out_pos);
98 brix_vfs_io_execute(&job);
99 if (job.io_errno != 0 || job.nio < 0
100 || (size_t) job.nio != out_pos)
101 {
102 /* Capture the write errno NOW: brix_codec_close() below runs
103 * backend teardown (free/library cleanup) that may clobber errno
104 * before the io_err branch formats strerror(). */
105 saved_errno = job.io_errno != 0 ? job.io_errno : EIO;
106 io_err = 1;
107 break;
108 }
109 cur += (off_t) job.nio;
110 produced += (size_t) job.nio;
111 }
112 if (rc == BRIX_CODEC_END) {
113 break;
114 }
115 if (rc != BRIX_CODEC_OK || (in_pos == prev_in && out_pos == 0)) {
116 data_err = 1; /* corrupt / bomb / no-progress */
117 break;
118 }
119 }
120 brix_codec_close(s);
121
122 snprintf(detail, sizeof(detail), "%lld+%zu z=%zu",
123 (long long) offset, produced, wlen);
124
125 if (io_err) {
126 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
127 ctx->files[idx].path, detail,
128 kXR_IOError, strerror(saved_errno));
129 }
130 if (data_err) {
131 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
132 ctx->files[idx].path, detail,
133 kXR_IOError, "corrupt or oversized compressed write");
134 }
135
136 /* Accounting: counters reflect PLAINTEXT bytes stored; bandwidth is charged
137 * the actual wire (compressed) bytes received. */
138 ctx->files[idx].bytes_written += produced;
139 ctx->totals.bytes_written += produced;
140 brix_rl_charge_ctx(ctx, wlen);
141
142 if (ctx->files[idx].dashboard_slot >= 0 &&
143 ngx_brix_dashboard_shm_zone != NULL)
144 {
145 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
146 ctx->files[idx].dashboard_slot,
147 (ngx_atomic_int_t) produced,
148 (int64_t) ngx_current_msec);
149 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
150 ctx->files[idx].dashboard_slot, "write");
151 }
152
153 if (rconf->access_log_fd != NGX_INVALID_FILE) {
154 brix_log_access(ctx, c, "WRITE", ctx->files[idx].path,
155 detail, 1, 0, NULL, produced);
156 }
157
158 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
159 return brix_send_ok(ctx, c, NULL, 0);
