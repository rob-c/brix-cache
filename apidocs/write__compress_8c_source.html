1/*
2 * WHAT: brix_write_compressed() serves a kXR_write for a handle opened for
3 * write with the opaque "?xrootd.compress=<codec>" while the server has
4 * brix_write_compress on. The request payload is ONE self-contained
5 * codec frame (the symmetric inverse of W4's per-request read frames); the
6 * server decompresses it under a decompression-bomb guard and streams the
7 * recovered plaintext through the VFS write core at the request offset,
8 * storing the file plaintext on disk. The client thus trades CPU for
9 * upload-wire bytes.
10 *
11 * WHY: This is the write-direction counterpart of W4 (read decompression) and
12 * the last plan workstream (W5). It is deliberately ISOLATED from the
13 * proven write path: brix_handle_write() routes here only when
14 * write_codec != IDENTITY, so the default (uncompressed) write — its AIO
15 * fast path and the write-recovery journal — is untouched and byte-
16 * identical. pgwrite/writev never call this, so the pgwrite per-page
17 * CRC32c invariant is preserved. The decode is on UNTRUSTED client input,
18 * so the bomb guard (absolute cap + expansion ratio) is mandatory.
19 *
20 * HOW: Each kXR_write is an independent whole-frame, so writes stay offset-
21 * addressable (re-writing the same offset reproduces the same plaintext).
22 * The plaintext is produced into a fixed window (rd.write_scratch) and
23 * written chunk-by-chunk at advancing offsets through the VFS core, so
24 * memory stays bounded regardless of the decompressed size. A
25 * synchronous VFS write keeps the opt-in path simple; the default path
26 * keeps its AIO offload.
27 */

40/* Fixed plaintext window streamed through VFS; bounds memory for any output size. */
41#define BRIX_WCMP_WINDOW (1u << 20) /* 1 MiB */
42/* Absolute ceiling on the plaintext produced from ONE write frame — a bomb that
43 * tries to exceed this is rejected (the ratio guard catches the classic case). */
44#define BRIX_WCMP_OUT_CAP (256ULL * 1024 * 1024) /* 256 MiB */
45
46/* ---- Write one decompressed window chunk through the VFS core ----
47 *
48 * WHAT: Writes out_pos plaintext bytes from win at the offset held in *cur via
49 * one synchronous VFS write job, advancing *cur and *produced by the bytes
50 * stored. Returns 0 on a fully-satisfied write; on any I/O fault (job
51 * errno, negative nio, or a short write) it records the errno into
52 * *saved_errno and returns 1 without advancing the cursor or counters.
53 *
54 * WHY: The plaintext window is flushed chunk-by-chunk at advancing offsets so
55 * memory stays bounded regardless of the decompressed size. Isolating the
56 * single write keeps the streaming loop flat and its complexity in check,
57 * and gives the errno-capture rule (below) one honest home.
58 *
59 * HOW: 1. Initialise a whole-frame VFS write job at *cur for out_pos bytes.
60 * 2. Execute it synchronously through the VFS I/O core.
61 * 3. On job errno, nio < 0, or nio != out_pos, capture the write errno and
62 * return 1 (failure).
63 * 4. Otherwise advance *cur and *produced by nio and return 0.
64 */
65static int
66brix_wcmp_flush_window(int fd, off_t *cur, u_char *win, size_t out_pos,
67 size_t *produced, int *saved_errno)
68{
69 brix_vfs_job_t job;
70
71 brix_vfs_job_write_init(&job, fd, *cur, win, out_pos);
72 brix_vfs_io_execute(&job);
73 if (job.io_errno != 0 || job.nio < 0 || (size_t) job.nio != out_pos) {
74 /* Capture the write errno NOW: brix_codec_close() in the caller runs
75 * backend teardown (free/library cleanup) that may clobber errno before
76 * the io_err branch formats strerror(). */
77 *saved_errno = job.io_errno != 0 ? job.io_errno : EIO;
78 return 1;
79 }
80 *cur += (off_t) job.nio;
81 *produced += (size_t) job.nio;
82 return 0;
83}

85/* ---- Decompress one write frame, streaming plaintext through the VFS ----
86 *
87 * WHAT: Drives the codec stream s over the single self-contained request frame
88 * in ctx->recv.payload (wlen bytes), flushing each produced window at
89 * advancing offsets starting from offset. Reports the total plaintext
90 * stored in *produced and sets exactly one outcome flag: *io_err on a
91 * backend write fault (with *saved_errno set), *data_err on a corrupt /
92 * oversized / no-progress frame, or neither on a clean end-of-stream.
93 *
94 * WHY: This is the whole-frame decode-and-store engine; extracting it keeps
95 * brix_write_compressed a flat orchestrator and concentrates the loop's
96 * branch complexity in one reviewable place. Byte-for-byte equivalence
97 * with the inline loop it replaces preserves the W5 write framing.
98 *
99 * HOW: 1. Zero the outputs and seed the input/output cursors.
100 * 2. Loop: step the codec with finish=1 into the fixed window.
101 * 3. If bytes were produced, flush them; on write fault set *io_err, stop.
102 * 4. On BRIX_CODEC_END stop cleanly.
103 * 5. On a non-OK rc or zero forward progress set *data_err, stop.
104 */
105static void
106brix_wcmp_decompress_stream(brix_codec_stream_t *s, brix_ctx_t *ctx,
107 size_t wlen, u_char *win, int fd, int64_t offset,
108 size_t *produced, int *io_err, int *data_err,
109 int *saved_errno)
110{
111 size_t in_pos = 0;
112 off_t cur = (off_t) offset;
113
114 *produced = 0;
115 *io_err = 0;
116 *data_err = 0;
117 *saved_errno = 0;
118
119 for ( ;; ) {
120 size_t out_pos = 0, prev_in = in_pos;
121 brix_codec_rc_t rc = brix_codec_step(s, ctx->recv.payload, wlen, &in_pos,
122 win, BRIX_WCMP_WINDOW,
123 &out_pos, 1 /* finish */);
124 if (out_pos > 0) {
125 if (brix_wcmp_flush_window(fd, &cur, win, out_pos,
126 produced, saved_errno))
127 {
128 *io_err = 1;
129 break;
130 }
131 }
132 if (rc == BRIX_CODEC_END) {
133 break;
134 }
135 if (rc != BRIX_CODEC_OK || (in_pos == prev_in && out_pos == 0)) {
136 *data_err = 1; /* corrupt / bomb / no-progress */
137 break;
138 }
139 }
140}

143brix_write_compressed(brix_ctx_t *ctx, ngx_connection_t *c,
144 int idx, int64_t offset, size_t wlen)
145{
146 int fd = ctx->files[idx].fd;
147 uint8_t codec = ctx->files[idx].write_codec;
148 ngx_stream_brix_srv_conf_t *rconf;
149 u_char *win;

151 brix_codec_guard_t guard;
152 size_t produced = 0;
153 int io_err = 0, data_err = 0;
154 int saved_errno = 0;
155 char detail[80];
156
157 rconf = ngx_stream_get_module_srv_conf(
158 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
159
160 win = BRIX_GET_SCRATCH(ctx, c, rd.write_scratch, rd.write_scratch_size,
161 BRIX_WCMP_WINDOW);
162 if (win == NULL) {
163 return NGX_ERROR;
164 }
165
166 ngx_memzero(&guard, sizeof(guard));
167 /* Bomb protection here is the ABSOLUTE per-frame output ceiling, not an
168 * expansion ratio: legitimate writes are often highly compressible (logs,
169 * sparse/zero regions, repetitive scientific data exceed 1000:1), so a ratio
170 * guard would false-reject real data. Memory stays bounded by the streaming
171 * window regardless, and the bytes land in the client's OWN quota-limited
172 * file, so the absolute cap is the right and sufficient guard. */
173 guard.out_cap = BRIX_WCMP_OUT_CAP;
174 guard.max_ratio = 0;
175
176 s = brix_codec_open((brix_codec_id_t) codec,
177 BRIX_CODEC_DIR_DECOMPRESS, -1, &guard);
178 if (s == NULL) {
179 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
180 ctx->files[idx].path, "-",
181 kXR_IOError, "inline write decompression unavailable");
182 }
183
184 brix_wcmp_decompress_stream(s, ctx, wlen, win, fd, offset,
185 &produced, &io_err, &data_err, &saved_errno);
186 brix_codec_close(s);
187
188 snprintf(detail, sizeof(detail), "%lld+%zu z=%zu",
189 (long long) offset, produced, wlen);
190
191 if (io_err) {
192 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
193 ctx->files[idx].path, detail,
194 kXR_IOError, strerror(saved_errno));
195 }
196 if (data_err) {
197 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
198 ctx->files[idx].path, detail,
199 kXR_IOError, "corrupt or oversized compressed write");
200 }
201
202 /* Accounting: counters reflect PLAINTEXT bytes stored; bandwidth is charged
203 * the actual wire (compressed) bytes received. */
204 ctx->files[idx].bytes_written += produced;
205 ctx->totals.bytes_written += produced;
206 brix_rl_charge_ctx(ctx, wlen);
207
208 if (ctx->files[idx].dashboard_slot >= 0 &&
209 ngx_brix_dashboard_shm_zone != NULL)
210 {
211 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
212 ctx->files[idx].dashboard_slot,
213 (ngx_atomic_int_t) produced,
214 (int64_t) ngx_current_msec);
215 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
216 ctx->files[idx].dashboard_slot, "write");
217 }
218
219 if (rconf->access_log_fd != NGX_INVALID_FILE) {
220 brix_log_access(ctx, c, "WRITE", ctx->files[idx].path,
221 detail, 1, 0, NULL, produced);
222 }
223
224 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
225 return brix_send_ok(ctx, c, NULL, 0);
226}
