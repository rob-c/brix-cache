7/*
8 * auth.c — XRootD `unix` (client-asserted UNIX-name) authentication handler
9 *
10 * WHAT: Implements the kXR_auth handler for the XRootD `unix` security
11 * protocol. The client merely *declares* a user name (and optional
12 * group) in the credential blob; there is no cryptographic proof.
13 * brix_handle_unix_auth() parses and character-validates those names,
14 * populates the per-connection identity (ctx->login.dn, ctx->login.vo_list,
15 * ctx->login.primary_vo, and the canonical ctx->identity), registers the
16 * session, emits auth metrics, writes a sanitised audit line, and
17 * replies kXR_ok — or rejects with kXR_NotAuthorized / kXR_NoMemory.
18 *
19 * WHY: Grid sites need an unauthenticated path for trusted local clients,
20 * sidecars, and intra-host tooling that should not have to mint a proxy
21 * cert or token. Because `unix` is unverified, this handler is
22 * deliberately fail-closed: by default it is honoured only for loopback
23 * peers (unless conf->unix_trust_remote is set), and it must be
24 * explicitly selected (conf->auth == BRIX_AUTH_UNIX) before the GSI
25 * auth dispatcher (../gsi/auth.c) will ever route a "unix" credtype here.
26 * Isolating it in one file keeps the "the client names itself" attack
27 * surface auditable in a single place.
28 *
29 * HOW: This is the stream / root:// path, downstream of kXR_login and inside
30 * the kXR_auth dispatcher. brix_handle_unix_auth() first gates on
31 * brix_unix_peer_is_loopback() (over the connection sockaddr), then
32 * validates the leading "unix\0" tag in ctx->recv.payload, splits the
33 * remaining space/NUL-delimited bytes into user and optional group, runs
34 * each through brix_unix_copy_name() (which enforces an allow-list of
35 * safe bytes via brix_unix_name_byte_ok() and a bounded destination).
36 * Validated names are copied into the identity fields, the session is
37 * registered, and VO/unique-user metrics are bumped via
38 * brix_unix_track_identity(). Every exit path increments the
39 * BRIX_AUTHN_UNIX auth metric (fail=0 / ok=1) and returns through the
40 * BRIX_RETURN_ERR / BRIX_RETURN_OK framing macros.
41 */

43/*
44 * Return non-zero when the peer connection originates from the local host:
45 * an IPv4 127.0.0.0/8 address, the IPv6 ::1 loopback, or any AF_UNIX socket.
46 * This is the default trust gate for `unix` auth (bypassed only when
47 * conf->unix_trust_remote is set), so a NULL connection/sockaddr is treated
48 * as untrusted (returns 0).
49 */
50static ngx_flag_t
51brix_unix_peer_is_loopback(ngx_connection_t *c)
52{
53 if (c == NULL || c->sockaddr == NULL) {
54 return 0;
55 }
56
57 if (c->sockaddr->sa_family == AF_INET) {
58 struct sockaddr_in *sin = (struct sockaddr_in *) c->sockaddr;
59 return ((ntohl(sin->sin_addr.s_addr) >> 24) == 127);
60 }
61
62 if (c->sockaddr->sa_family == AF_INET6) {
63 struct sockaddr_in6 *sin6 = (struct sockaddr_in6 *) c->sockaddr;
64 return IN6_IS_ADDR_LOOPBACK(&sin6->sin6_addr);
65 }
66
67 return c->sockaddr->sa_family == AF_UNIX;
68}

70/*
71 * Allow-list predicate for a single byte of an asserted user/group name.
72 * Accepts only [A-Za-z0-9] plus a small set of identity-safe punctuation
73 * ('_', '-', '.', '@', '+'). Restricting the alphabet up front keeps the
74 * unverified, attacker-controlled name out of log lines, metric labels, and
75 * downstream ACL comparisons as anything other than plain printable text.
76 */
77static ngx_flag_t
78brix_unix_name_byte_ok(u_char ch)
79{
80 return ((ch >= 'A' && ch <= 'Z')
81 || (ch >= 'a' && ch <= 'z')
82 || (ch >= '0' && ch <= '9')
83 || ch == '_' || ch == '-' || ch == '.' || ch == '@' || ch == '+');
84}

86/*
87 * Validate and NUL-terminate one wire name into a fixed-size buffer.
88 * Rejects (NGX_ERROR) an empty source, a length that would not leave room for
89 * the terminator (len >= dst_len), or any byte failing
90 * brix_unix_name_byte_ok(); otherwise copies `len` bytes and appends '\0',
91 * returning NGX_OK. This is the single choke point that bounds and sanitises
92 * both the user and the optional group strings.
93 */
94static ngx_int_t
95brix_unix_copy_name(char *dst, size_t dst_len, const u_char *src,
96 size_t len)
97{
98 size_t i;
99
100 if (dst == NULL || dst_len == 0 || src == NULL || len == 0
101 || len >= dst_len)
102 {
103 return NGX_ERROR;
104 }
105
106 for (i = 0; i < len; i++) {
107 if (!brix_unix_name_byte_ok(src[i])) {
108 return NGX_ERROR;
109 }
110 }
111
112 ngx_memcpy(dst, src, len);
113 dst[len] = '\0';
114 return NGX_OK;
115}

141/*
142 * Parsed result of an asserted `unix` credential: the validated, NUL-terminated
143 * user name and the optional group (empty string when the client sent none).
144 * Populated by brix_unix_parse_cred(); consumed by brix_unix_apply_identity().
145 */
146typedef struct {
147 char user[BRIX_SSS_USER_MAX];
148 char group[BRIX_SSS_GROUP_MAX];
149} brix_unix_names_t;

151/*
152 * WHAT: Confirm the credential blob opens with the expected "unix\0" tag.
153 * WHY: The `unix` scheme is unverified, so the one structural guarantee we
154 * insist on is the leading protocol tag; anything shorter or mistagged is
155 * a malformed credential and must be rejected before we parse names.
156 * HOW: Returns NGX_OK when payload is non-NULL, at least 6 bytes, begins with
157 * "unix", and byte 4 is NUL; otherwise NGX_DECLINED. No side effects.
158 */
159static ngx_int_t
160brix_unix_tag_ok(const brix_ctx_t *ctx)
161{
162 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen < 6
163 || ngx_strncmp(ctx->recv.payload, "unix", 4) != 0
164 || ctx->recv.payload[4] != '\0')
165 {
166 return NGX_DECLINED;
167 }
168 return NGX_OK;
169}

171/*
172 * WHAT: Advance `*p` over any run of ASCII spaces, then return the extent of the
173 * next space/NUL-delimited token as [start, *p).
174 * WHY: Both the user and the optional group are parsed with identical
175 * skip-spaces-then-take-a-token logic; centralising it keeps the parser
176 * linear and removes duplicated pointer walks.
177 * HOW: Pure cursor arithmetic over the half-open [*p, end) range: skip spaces,
178 * record the token start, advance to the first space/NUL/end, and report
179 * the token start plus its length via the out-params.
180 */
181static void
182brix_unix_next_token(const u_char **p, const u_char *end,
183 const u_char **tok_start, size_t *tok_len)
184{
185 const u_char *cur = *p;
186
187 while (cur < end && *cur == ' ') {
188 cur++;
189 }
190 *tok_start = cur;
191 while (cur < end && *cur != '\0' && *cur != ' ') {
192 cur++;
193 }
194 *tok_len = (size_t) (cur - *tok_start);
195 *p = cur;
196}

198/*
199 * WHAT: Parse the space/NUL-delimited user (mandatory) and group (optional) out
200 * of the credential payload and character-validate both into `names`.
201 * WHY: Isolating the untrusted-name parse keeps the single choke point for the
202 * safe-byte allow-list (brix_unix_copy_name) in one auditable place and
203 * out of the top-level control flow.
204 * HOW: Skips the "unix\0" tag, reads the user token, then (only if a further
205 * space-separated token follows) the group token. Each non-empty token is
206 * copied through brix_unix_copy_name(); on any validation failure returns
207 * NGX_DECLINED and writes the reject reason to *msg. Returns NGX_OK with
208 * names->group left "" when no group was asserted.
209 */
210static ngx_int_t
211brix_unix_parse_cred(const brix_ctx_t *ctx, brix_unix_names_t *names,
212 const char **msg)
213{
214 const u_char *p, *end, *user_start, *group_start;
215 size_t user_len = 0, group_len = 0;
216
217 names->user[0] = '\0';
218 names->group[0] = '\0';
219
220 p = ctx->recv.payload + 5;
221 end = ctx->recv.payload + ctx->recv.cur_dlen;
222
223 brix_unix_next_token(&p, end, &user_start, &user_len);
224
225 if (p < end && *p == ' ') {
226 brix_unix_next_token(&p, end, &group_start, &group_len);
227 if (group_len > 0
228 && brix_unix_copy_name(names->group, sizeof(names->group),
229 group_start, group_len) != NGX_OK)
230 {
231 *msg = "invalid unix group";
232 return NGX_DECLINED;
233 }
234 }
235
236 if (brix_unix_copy_name(names->user, sizeof(names->user), user_start,
237 user_len) != NGX_OK)
238 {
239 *msg = "invalid unix user";
240 return NGX_DECLINED;
241 }
242
243 return NGX_OK;
244}

246/*
247 * WHAT: Stamp the validated user/group onto the per-connection identity and
248 * mark the session authenticated (unverified, non-token).
249 * WHY: Populating ctx->login and the canonical ctx->identity is the load-
250 * bearing state transition; keeping it in one helper preserves the exact
251 * field-fill ordering that downstream ACL/session code depends on.
252 * HOW: Sets auth_done/token flags, copies user→login.dn and (when present)
253 * group→login.vo_list + login.primary_vo via ngx_cpystrn, then mirrors dn
254 * and the VO CSV into ctx->identity. Returns NGX_OK on success, or
255 * NGX_ABORT (with *code = kXR_NoMemory) if identity allocation fails.
256 */
257static ngx_int_t
258brix_unix_apply_identity(brix_ctx_t *ctx, ngx_connection_t *c,
259 const brix_unix_names_t *names, int *code)
260{
261 ctx->login.auth_done = 1;
262 ctx->token.auth = 0;
263 ngx_cpystrn((u_char *) ctx->login.dn, (u_char *) names->user,
264 sizeof(ctx->login.dn));
265 if (names->group[0] != '\0') {
266 ngx_cpystrn((u_char *) ctx->login.vo_list, (u_char *) names->group,
267 sizeof(ctx->login.vo_list));
268 ngx_cpystrn((u_char *) ctx->login.primary_vo, (u_char *) names->group,
269 sizeof(ctx->login.primary_vo));
270 }
271
272 if (ctx->identity != NULL) {
273 if (brix_identity_set_dn(ctx->identity, c->pool, ctx->login.dn,
274 BRIX_AUTHN_UNIX) != NGX_OK
275 || brix_identity_set_vos_csv(ctx->identity, c->pool,
276 ctx->login.vo_list) != NGX_OK)
277 {
278 *code = kXR_NoMemory;
279 return NGX_ABORT;
280 }
281 }
282
283 return NGX_OK;
284}

286/*
287 * WHAT: Finalise a successful `unix` auth: register the session, bump identity
288 * metrics, and emit the sanitised audit line.
289 * WHY: The success epilogue is a fixed, side-effect-only sequence; separating
290 * it keeps the top-level function focused on validation control flow.
291 * HOW: Registers the session under login.sessid, calls brix_unix_track_identity
292 * for VO/unique-user metrics, then logs user/group through
293 * brix_sanitize_log_string (group shown as "-" when empty). No return.
294 */
295static void
296brix_unix_finalize(brix_ctx_t *ctx, ngx_connection_t *c,
297 const brix_unix_names_t *names)
298{
299 char safe_user[BRIX_SSS_USER_MAX * 4];
300 char safe_group[BRIX_SSS_GROUP_MAX * 4];
301
302 brix_session_register(ctx->login.sessid, ctx->login.dn,
303 ctx->login.vo_list, 0);
304 brix_unix_track_identity(ctx);
305
306 brix_sanitize_log_string(names->user, safe_user, sizeof(safe_user));
307 brix_sanitize_log_string(names->group[0] ? names->group : "-", safe_group,
308 sizeof(safe_group));
309 ngx_log_error(NGX_LOG_INFO, c->log, 0,
310 "brix: unix auth OK user=\"%s\" group=\"%s\"",
311 safe_user, safe_group);
312}

314/*
315 * Public entry point for the XRootD `unix` auth scheme, called from the GSI
316 * auth dispatcher (../gsi/auth.c) once it has matched the "unix" credtype and
317 * confirmed conf->auth == BRIX_AUTH_UNIX.
318 *
319 * Enforces the loopback trust gate (unless conf->unix_trust_remote), validates
320 * the "unix\0" tag, parses the space/NUL-delimited user and optional group out
321 * of ctx->recv.payload, and character-validates both via brix_unix_copy_name().
322 * On success it marks the session authenticated (ctx->login.auth_done = 1,
323 * token_auth = 0), fills ctx->login.dn / ctx->login.vo_list / ctx->login.primary_vo and the
324 * canonical ctx->identity, registers the session, bumps metrics, logs a
325 * sanitised audit line, and returns kXR_ok via BRIX_RETURN_OK. Any failure
326 * increments the failed-auth metric and returns kXR_NotAuthorized (bad peer,
327 * malformed credential, invalid user/group) or kXR_NoMemory (identity alloc).
328 */
329ngx_int_t
330brix_handle_unix_auth(brix_ctx_t *ctx, ngx_connection_t *c,
331 ngx_stream_brix_srv_conf_t *conf)
332{
333 brix_unix_names_t names;
334 const char *reject = NULL;
335 int err_code = kXR_NoMemory;
336
337 if (!conf->unix_trust_remote && !brix_unix_peer_is_loopback(c)) {
338 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_UNIX, 0);
339 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "unix",
340 kXR_NotAuthorized,
341 "unix auth is restricted to loopback peers");
342 }
343
344 if (brix_unix_tag_ok(ctx) != NGX_OK) {
345 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_UNIX, 0);
346 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "unix",
347 kXR_NotAuthorized, "malformed unix credential");
348 }
349
350 if (brix_unix_parse_cred(ctx, &names, &reject) != NGX_OK) {
351 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_UNIX, 0);
352 BRIX_RETURN_ERR(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "unix",
353 kXR_NotAuthorized, reject);
354 }
355
356 if (brix_unix_apply_identity(ctx, c, &names, &err_code) != NGX_OK) {
357 return brix_send_error(ctx, c, err_code,
358 "identity allocation failed");
359 }
360
361 brix_unix_finalize(ctx, c, &names);
362
363 brix_metric_auth(BRIX_PROTO_ROOT, BRIX_AUTHN_UNIX, 1);
364 BRIX_RETURN_OK(ctx, c, BRIX_OP_AUTH, "AUTH", "-", "unix", 0);
365}
