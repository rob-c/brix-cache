16/*
17 * ftp_ev_data.c — non-blocking data-channel lifecycle for the event engine.
18 *
19 * WHAT: the passive/active setup verbs (PASV/EPSV/PORT/EPRT), the event-driven
20 * bring-up of a data connection (accept a passive client, or connect out in
21 * active mode), and idempotent transfer teardown.
22 *
23 * WHY: the sync engine blocks in accept()/connect() and then in the read/write
24 * loop, monopolising the worker for one transfer. Here the listener and the
25 * connect socket are wrapped in nginx connections and driven by their own
26 * read/write events, so the worker keeps serving other sessions while a transfer
27 * proceeds. The confinement and anti-bounce/SSRF rules are ported verbatim from
28 * the sync engine (ftp_do_pasv/ftp_do_port) so both engines expose the same
29 * data-channel security boundary during the transition.
30 *
31 * HOW: PASV/EPSV open a non-blocking listener bound to the control channel's own
32 * IP and stash its fd in fc->pasv_fd; PORT/EPRT pin+screen a client-nominated
33 * target into fc->active_sa. brix_ftp_ev_data_open() is called once a transfer
34 * verb has sent its 150: it wraps the listener (passive) or a fresh connect()
35 * socket (active) in an ngx_connection_t and arms the accept/connect event. When
36 * the data socket is ready the handler hands off to brix_ftp_ev_data_ready()
37 * (ftp_ev_xfer.c), which opens the VFS side and starts the pump.
38 */

43/* PASV/EPSV: open a non-blocking listener bound to the control connection's local
44 * IP and an ephemeral port; `extended` selects the RFC 2428 (229) vs RFC 959
45 * (227) reply. Mirrors the sync ftp_do_pasv, but the socket is non-blocking so
46 * the later accept runs under the event loop. */
47static ngx_int_t
48ev_do_pasv(ftp_ev_t *fc, int extended)
49{
50 struct sockaddr_in local;
51 socklen_t llen = sizeof(local);
52 struct sockaddr_in bindaddr;
53 socklen_t blen = sizeof(bindaddr);
54 int fd;
55 unsigned char ip[4];
56 unsigned port;
57
58 fc->active = 0; /* PASV/EPSV overrides PORT */
59
60 if (fc->pasv_fd >= 0) {
61 (void) close(fc->pasv_fd);
62 fc->pasv_fd = -1;
63 }
64
65 if (getsockname(fc->c->fd, (struct sockaddr *) &local, &llen) != 0
66 || local.sin_family != AF_INET)
67 {
68 return brix_ftp_ev_reply(fc, "425 Cannot open passive connection\r\n");
69 }
70
71 fd = socket(AF_INET, SOCK_STREAM, 0);
72 if (fd < 0) {
73 return brix_ftp_ev_reply(fc, "425 Cannot open passive connection\r\n");
74 }
75 if (ngx_nonblocking(fd) == -1) {
76 (void) close(fd);
77 return brix_ftp_ev_reply(fc, "425 Cannot open passive connection\r\n");
78 }
79
80 ngx_memzero(&bindaddr, sizeof(bindaddr));
81 bindaddr.sin_family = AF_INET;
82 bindaddr.sin_addr = local.sin_addr; /* same host as control */
83
84 /* Bind the listener. With a configured passive range (brix_gridftp_pasv_port_range)
85 * we walk it and take the first free port so the peer only ever has to reach a
86 * firewall-opened window; unset (lo == 0) keeps the kernel-ephemeral bind. A
87 * fully-occupied range fails the transfer (rather than falling back to a
88 * random, un-firewalled port) so a locked-down deployment stays predictable. */
89 if (fc->conf->pasv_port_lo > 0) {
90 ngx_int_t p;
91 int bound = 0;
92
93 for (p = fc->conf->pasv_port_lo; p <= fc->conf->pasv_port_hi; p++) {
94 bindaddr.sin_port = htons((unsigned short) p);
95 if (bind(fd, (struct sockaddr *) &bindaddr, sizeof(bindaddr)) == 0) {
96 bound = 1;
97 break;
98 }
99 if (errno != EADDRINUSE && errno != EADDRNOTAVAIL) {
100 break; /* a non-contention error won't clear on the next port */
101 }
102 }
103 if (!bound) {
104 (void) close(fd);
105 return brix_ftp_ev_reply(fc,
106 "425 No free passive port in configured range\r\n");
107 }
108 if (listen(fd, BRIX_FTP_EV_DATA_BACKLOG) != 0
109 || getsockname(fd, (struct sockaddr *) &bindaddr, &blen) != 0)
110 {
111 (void) close(fd);
112 return brix_ftp_ev_reply(fc, "425 Cannot open passive connection\r\n");
113 }
114 } else {
115 bindaddr.sin_port = 0; /* ephemeral */
116
117 /* Backlog spans a full MODE E stream fan-out (globus opens all parallel
118 * data connections at once), matching the sync engine. */
119 if (bind(fd, (struct sockaddr *) &bindaddr, sizeof(bindaddr)) != 0
120 || listen(fd, BRIX_FTP_EV_DATA_BACKLOG) != 0
121 || getsockname(fd, (struct sockaddr *) &bindaddr, &blen) != 0)
122 {
123 (void) close(fd);
124 return brix_ftp_ev_reply(fc, "425 Cannot open passive connection\r\n");
125 }
126 }
127
128 fc->pasv_fd = fd;
129 ngx_memcpy(ip, &bindaddr.sin_addr.s_addr, 4);
130 port = ntohs(bindaddr.sin_port);
131
132 if (extended) {
133 return brix_ftp_ev_reply(fc,
134 "229 Entering Extended Passive Mode (|||%ud|)\r\n",
135 (ngx_uint_t) port);
136 }
137 return brix_ftp_ev_reply(fc,
138 "227 Entering Passive Mode (%ud,%ud,%ud,%ud,%ud,%ud)\r\n",
139 (ngx_uint_t) ip[0], (ngx_uint_t) ip[1],
140 (ngx_uint_t) ip[2], (ngx_uint_t) ip[3],
141 (ngx_uint_t) (port >> 8), (ngx_uint_t) (port & 0xff));
142}

145/* PORT/EPRT: arm an active-mode target. The nominated address is pinned to the
146 * control peer (anti FTP-bounce; relaxed only for a DCAU-A TPC leg) and screened
147 * through the SSRF policy — identical to the sync ftp_do_port. */
148static ngx_int_t
149ev_do_port(ftp_ev_t *fc, const char *arg, int extended)
150{
151 struct sockaddr_in peer;
152 socklen_t plen = sizeof(peer);
153 struct sockaddr_in tgt;
154 brix_net_target_policy_t pol;
155 char err[128];
156 unsigned h[4], p[2];
157 in_addr_t addr;
158 unsigned port;
159
160 ngx_memzero(&tgt, sizeof(tgt));
161 tgt.sin_family = AF_INET;
162
163 if (!extended) {
164 if (sscanf(arg, "%u,%u,%u,%u,%u,%u",
165 &h[0], &h[1], &h[2], &h[3], &p[0], &p[1]) != 6
166 || (h[0] | h[1] | h[2] | h[3]) > 255 || (p[0] | p[1]) > 255)
167 {
168 return brix_ftp_ev_reply(fc, "501 Bad PORT argument\r\n");
169 }
170 {
171 unsigned char b[4] = { (unsigned char) h[0], (unsigned char) h[1],
172 (unsigned char) h[2], (unsigned char) h[3] };
173 ngx_memcpy(&addr, b, 4);
174 }
175 port = (p[0] << 8) | p[1];
176 } else {
177 char d = arg[0];
178 const char *fam, *ip, *pt, *end;
179 char ipbuf[64];
180 size_t iplen;
181
182 if (d == '\0') { return brix_ftp_ev_reply(fc, "501 Bad EPRT argument\r\n"); }
183 fam = arg + 1;
184 ip = strchr(fam, d);
185 if (ip == NULL) { return brix_ftp_ev_reply(fc, "501 Bad EPRT argument\r\n"); }
186 ip++;
187 pt = strchr(ip, d);
188 if (pt == NULL) { return brix_ftp_ev_reply(fc, "501 Bad EPRT argument\r\n"); }
189 end = strchr(pt + 1, d);
190 if (end == NULL) { return brix_ftp_ev_reply(fc, "501 Bad EPRT argument\r\n"); }
191 if (fam[0] != '1') {
192 return brix_ftp_ev_reply(fc, "522 Only IPv4 (|1|) supported\r\n");
193 }
194 iplen = (size_t) (pt - ip);
195 if (iplen == 0 || iplen >= sizeof(ipbuf)) {
196 return brix_ftp_ev_reply(fc, "501 Bad EPRT address\r\n");
197 }
198 ngx_memcpy(ipbuf, ip, iplen);
199 ipbuf[iplen] = '\0';
200 if (inet_pton(AF_INET, ipbuf, &addr) != 1) {
201 return brix_ftp_ev_reply(fc, "501 Bad EPRT address\r\n");
202 }
203 port = (unsigned) strtoul(pt + 1, NULL, 10);
204 }
205
206 if (port == 0 || port > 65535) {
207 return brix_ftp_ev_reply(fc, "501 Bad data port\r\n");
208 }
209 tgt.sin_addr.s_addr = addr;
210 tgt.sin_port = htons((unsigned short) port);
211
212 /* Anti-bounce pin: a plain active transfer must target the control peer; an
213 * off-peer target is only allowed as a GSI-authenticated (DCAU A) TPC leg. */
214 if (getpeername(fc->c->fd, (struct sockaddr *) &peer, &plen) != 0
215 || peer.sin_family != AF_INET)
216 {
217 return brix_ftp_ev_reply(fc, "500 Cannot determine control peer\r\n");
218 }
219 fc->active_offpeer = (peer.sin_addr.s_addr != tgt.sin_addr.s_addr);
220 if (fc->active_offpeer && !(fc->sec_active && fc->dcau_a)) {
221 ngx_log_error(NGX_LOG_WARN, fc->c->log, 0,
222 "brix: gsiftp(ev) rejected active-mode target != control "
223 "peer (no DCAU A; possible FTP-bounce)");
224 return brix_ftp_ev_reply(fc, "500 Data address must match control peer\r\n");
225 }
226
227 ngx_memzero(&pol, sizeof(pol));
228 pol.allow_local = 1;
229 pol.allow_private = 1;
230 if (brix_net_target_check_addr((struct sockaddr *) &tgt, &pol,
231 err, sizeof(err)) != NGX_OK)
232 {
233 ngx_log_error(NGX_LOG_WARN, fc->c->log, 0,
234 "brix: gsiftp(ev) active-mode target blocked: %s", err);
235 return brix_ftp_ev_reply(fc, "500 Data address not permitted\r\n");
236 }
237
238 if (fc->pasv_fd >= 0) { (void) close(fc->pasv_fd); fc->pasv_fd = -1; }
239 fc->active_sa = tgt;
240 fc->active = 1;
241 return brix_ftp_ev_reply(fc, "200 %s command successful\r\n",
242 extended ? "EPRT" : "PORT");
243}

283/* Passive listener read event: a client has connected — accept it, retire the
284 * listener, and hand the data socket to the pump. */
285static void
286ev_accept_handler(ngx_event_t *rev)
287{
288 ngx_connection_t *lc = rev->data;
289 ftp_ev_dc_t *dc = lc->data;
290 ftp_ev_t *fc = dc->fc;
291 int dfd;
292
293 if (rev->timedout) {
294 ngx_log_error(NGX_LOG_INFO, fc->c->log, NGX_ETIMEDOUT,
295 "brix: GridFTP(ev) passive data accept timeout");
296 brix_ftp_ev_data_finish(dc, NGX_ERROR);
297 return;
298 }
299
300 dfd = accept(lc->fd, NULL, NULL);
301 if (dfd < 0) {
302 if (ngx_socket_errno == NGX_EAGAIN) {
303 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
304 brix_ftp_ev_data_finish(dc, NGX_ERROR);
305 }
306 return; /* spurious wakeup — wait */
307 }
308 ngx_log_error(NGX_LOG_ERR, fc->c->log, ngx_socket_errno,
309 "brix: GridFTP(ev) passive accept failed");
310 brix_ftp_ev_data_finish(dc, NGX_ERROR);
311 return;
312 }
313
314 /* Retire the listener (closes fc->pasv_fd) — one transfer, one accept. */
315 if (rev->timer_set) { ngx_del_timer(rev); }
316 ngx_close_connection(lc);
317 fc->pasv_fd = -1;
318 dc->lc = NULL;
319
320 if (ngx_nonblocking(dfd) == -1) {
321 (void) close(dfd);
322 brix_ftp_ev_data_finish(dc, NGX_ERROR);
323 return;
324 }
325 dc->dconn = brix_ftp_ev_wrap_conn(fc, dfd);
326 if (dc->dconn == NULL) {
327 (void) close(dfd);
328 brix_ftp_ev_data_finish(dc, NGX_ERROR);
329 return;
330 }
331 dc->dconn->data = dc;
332 if (fc->prot == 'P') {
333 brix_ftp_ev_dc_start_tls(dc); /* GSI-secure the data channel */
334 } else {
335 brix_ftp_ev_data_ready(dc);
336 }
337}

376/* Bring the data connection up under the event loop. Passive: wrap the pending
377 * listener and arm its accept event. Active: open a non-blocking socket bound to
378 * the control channel's local IP, start the connect, and arm the write event.
379 * Returns NGX_OK once the relevant event is armed (completion is asynchronous)
380 * or NGX_ERROR if the socket could not be set up. */
381ngx_int_t
382brix_ftp_ev_data_open(ftp_ev_dc_t *dc)
383{
384 ftp_ev_t *fc = dc->fc;
385
386 if (!fc->active) {
387 /* Passive: the listener fd is already open (PASV/EPSV) and non-blocking. */
388 if (fc->pasv_fd < 0) {
389 return NGX_ERROR;
390 }
391 dc->lc = ngx_get_connection(fc->pasv_fd, fc->c->log);
392 if (dc->lc == NULL) {
393 (void) close(fc->pasv_fd);
394 fc->pasv_fd = -1;
395 return NGX_ERROR;
396 }
397 dc->lc->log = fc->c->log;
398 dc->lc->read->log = fc->c->log;
399 dc->lc->data = dc;
400 /* MODE E STOR keeps the listener open to accept every parallel stream; all
401 * other transfers take a single connection and retire the listener. */
402 dc->lc->read->handler = (dc->mode_e && dc->writing)
403 ? brix_ftp_ev_eb_accept
404 : ev_accept_handler;
405 if (ngx_handle_read_event(dc->lc->read, 0) != NGX_OK) {
406 ngx_close_connection(dc->lc); /* also closes pasv_fd */
407 fc->pasv_fd = -1;
408 dc->lc = NULL;
409 return NGX_ERROR;
410 }
411 ngx_add_timer(dc->lc->read, BRIX_FTP_EV_IO_TIMEO);
412 return NGX_OK;
413 }
414
415 /* Active: connect out to the armed, peer-pinned target. */
416 {
417 struct sockaddr_in local;
418 socklen_t llen = sizeof(local);
419 int dfd;
420 int rc;
421
422 fc->active = 0; /* one-shot: consume the arm */
423
424 dfd = socket(AF_INET, SOCK_STREAM, 0);
425 if (dfd < 0) {
426 return NGX_ERROR;
427 }
428 if (ngx_nonblocking(dfd) == -1) {
429 (void) close(dfd);
430 return NGX_ERROR;
431 }
432 if (getsockname(fc->c->fd, (struct sockaddr *) &local, &llen) == 0
433 && local.sin_family == AF_INET)
434 {
435 local.sin_port = 0; /* ephemeral source port */
436 (void) bind(dfd, (struct sockaddr *) &local, sizeof(local));
437 }
438
439 dc->dconn = brix_ftp_ev_wrap_conn(fc, dfd);
440 if (dc->dconn == NULL) {
441 (void) close(dfd);
442 return NGX_ERROR;
443 }
444 dc->dconn->data = dc;
445 dc->dconn->write->handler = ev_connect_handler;
446 dc->connecting = 1;
447 dc->tls_client = 1; /* we dialled out: TLS client role */
448
449 rc = connect(dfd, (struct sockaddr *) &fc->active_sa,
450 sizeof(fc->active_sa));
451 if (rc == -1 && ngx_socket_errno != NGX_EINPROGRESS) {
452 ngx_log_error(NGX_LOG_ERR, fc->c->log, ngx_socket_errno,
453 "brix: GridFTP(ev) active data connect() failed");
454 ngx_close_connection(dc->dconn);
455 dc->dconn = NULL;
456 return NGX_ERROR;
457 }
458 if (ngx_handle_write_event(dc->dconn->write, 0) != NGX_OK) {
459 ngx_close_connection(dc->dconn);
460 dc->dconn = NULL;
461 return NGX_ERROR;
462 }
463 /* Whether connect() returned 0 (immediate, loopback) or EINPROGRESS, let
464 * the write event drive ev_connect_handler — running the hand-off out of
465 * this stack keeps us from re-entering the control loop synchronously. */
466 ngx_add_timer(dc->dconn->write, BRIX_FTP_EV_IO_TIMEO);
467 return NGX_OK;
468 }
469}

472/* ---- teardown -------------------------------------------------------------- */
473
474/* Finish a transfer: release the VFS side, close the data socket and any pending
475 * listener, emit the control-channel result (226 / 550), and resume the control
476 * state machine. Idempotent-safe: the pump only calls it once per transfer. */
477void
478brix_ftp_ev_data_finish(ftp_ev_dc_t *dc, ngx_int_t rc)
479{
480 ftp_ev_t *fc = dc->fc;
481
482 if (dc->eb_conns != NULL) {
483 brix_ftp_ev_eb_teardown(dc); /* close live MODE E streams */
484 }
485 if (dc->fh != NULL) {
486 (void) brix_vfs_close(dc->fh, fc->c->log);
487 dc->fh = NULL;
488 }
489 if (dc->dh != NULL) {
490 (void) brix_vfs_closedir(dc->dh, fc->c->log);
491 dc->dh = NULL;
492 }
493 if (dc->writer != NULL) {
494 /* A writer still open here means the transfer failed before commit. */
495 brix_vfs_writer_abort(dc->writer);
496 dc->writer = NULL;
497 }
498 if (dc->dconn != NULL) {
499 if (dc->dconn->ssl != NULL) {
500 /* Quiet, non-blocking teardown: free the SSL synchronously (no
501 * close_notify round trip) before closing the socket. */
502 dc->dconn->ssl->no_wait_shutdown = 1;
503 dc->dconn->ssl->no_send_shutdown = 1;
504 (void) ngx_ssl_shutdown(dc->dconn);
505 }
506 ngx_close_connection(dc->dconn);
507 dc->dconn = NULL;
508 }
509 if (dc->lc != NULL) {
510 ngx_close_connection(dc->lc); /* also closes fc->pasv_fd */
511 fc->pasv_fd = -1;
512 dc->lc = NULL;
513 }
514 if (dc->dpool != NULL) {
515 ngx_destroy_pool(dc->dpool); /* the TLS conn's pool */
516 dc->dpool = NULL;
517 }
518
519 fc->dc = NULL;
520 fc->state = FTP_EV_ST_CMD;
521
522 (void) brix_ftp_ev_reply(fc, (rc == NGX_OK)
523 ? "226 Transfer complete\r\n"
524 : "550 Transfer failed\r\n");
525
526 /* Resume the control channel: flush the result and frame the next command. */
527 brix_ftp_ev_resume(fc);
528}
