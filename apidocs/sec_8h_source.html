1/*
2 * sec.h — client auth-protocol module contract.
3 *
4 * WHAT: A small interface each auth protocol (token/unix/gsi) implements; the
5 * auth driver (auth.c) parses the server's "&P=..." list, picks a module we
6 * have credentials for, and drives the kXR_auth/kXR_authmore loop.
7 * WHY: Keeps each protocol self-contained and the driver protocol-agnostic.
8 * HOW: The driver builds the ClientAuthRequest (credtype from .credtype) carrying
9 * the payload from first()/more(); a single-round protocol leaves .more NULL.
10 */
11#ifndef XRDC_SEC_H
12#define XRDC_SEC_H
13
14#include "brix.h"
15
16typedef struct {
17 const char *name; /* matches &P=<name> (e.g. "ztn","gsi","unix") */
18 char credtype[4]; /* kXR_auth credtype field, e.g. {'z','t','n',0} */
19
20 /* Do we have usable credentials for this protocol? (env/file probe).
21 * c may be NULL (e.g. explain path); when c->opts.cred is non-NULL the
22 * store's availability probe is used in place of the inline env probe. */
23 int (*have_creds)(brix_conn *c);
24
25 /* Build the first kXR_auth payload. parms = args after "&P=<name>," ("" if
26 * none). Allocates *payload (caller frees); sets *plen. 0 / -1. */
27 int (*first)(brix_conn *c, const char *parms,
28 uint8_t **payload, uint32_t *plen, brix_status *st);
29
30 /* Handle a kXR_authmore challenge; build the next payload. Set *payload=NULL
31 * when no further round is expected. NULL for single-round protocols. */
32 int (*more)(brix_conn *c, const uint8_t *sbody, uint32_t slen,
33 uint8_t **payload, uint32_t *plen, brix_status *st);
34
35 void (*cleanup)(brix_conn *c); /* optional teardown */
36} brix_sec_module;
