14/* webdav_token_op_class — map the HTTP method to a registry op class * Read-ish verbs (GET/HEAD/PROPFIND/OPTIONS) authorize against read scopes;
15 * everything else (PUT/DELETE/MKCOL/MOVE/COPY/PROPPATCH/LOCK/...) is a write. */
16static brix_token_op_e
17webdav_token_op_class(ngx_http_request_t *r)
18{
19 switch (r->method) {
20 case NGX_HTTP_GET:
21 case NGX_HTTP_HEAD:
22 case NGX_HTTP_PROPFIND:
23 case NGX_HTTP_OPTIONS:
24 return BRIX_TOKEN_OP_READ;
25 default:
26 return BRIX_TOKEN_OP_WRITE;
27 }
28}

30/*
31 * webdav_check_token_scope — enforce WLCG/SciToken read or write scope for
32 * any WebDAV data-access method (GET, HEAD, PROPFIND, PUT, DELETE, MKCOL, …).
33 *
34 * If the request was authenticated via a bearer token (rctx->token_auth == 1),
35 * checks whether the token's scope covers the request URI path for the
36 * operation class (read vs write) derived from the HTTP method.
37 * Returns NGX_OK if the scope check passes or if auth was not token-based.
38 * Returns NGX_HTTP_FORBIDDEN if the token lacks scope for the URI.
39 *
40 * NOTE: scope is checked against the raw decoded URI path, not the filesystem
41 * path — the path-prefix invariant is enforced by the scope matching code in
42 * token/scopes.c (must be an exact prefix, not a partial directory name match).
43 */
44/*
45 *
46 * WHAT: Enforces WLCG/SciToken read or write scope authorization for WebDAV
47 * data-access methods. Derives the required op class (read vs write) from the
48 * HTTP method via webdav_token_op_class(), then checks whether the
49 * authenticated bearer token's scopes cover the request URI path. Returns
50 * NGX_OK if the scope check passes or if authentication was not token-based
51 * (e.g., GSI cert auth has no equivalent scope concept). Returns
52 * NGX_HTTP_FORBIDDEN when the token lacks sufficient permission for the URI.
53 *
54 * WHY: WLCG/SciToken grants fine-grained path-based access rights rather than
55 * binary allow/deny. A token might grant read-only access to /data/atlas but
56 * write access to /data/cms — this function prevents both cross-VO file reads
57 * and cross-VO file mutation by enforcing scope on every data-access method,
58 * not just writes. The raw URI path check (not filesystem path) is intentional
59 * because scope granularity must match the client-facing namespace, not the
60 * underlying storage layout.
61 *
62 * HOW: Retrieves request context and verifies token_auth flag is set; derives
63 * need_write from webdav_token_op_class(); copies r->uri into a
64 * null-terminated buffer for scope checking; calls either
65 * brix_token_check_write() or brix_token_check_read() (or the identity
66 * wrapper) to verify the URI path is covered by a matching scope prefix; logs
67 * warning and returns 403 if no matching scope found. */
68ngx_int_t
69webdav_check_token_scope(ngx_http_request_t *r, const char *method_name)
70{
71 ngx_http_brix_webdav_req_ctx_t *rctx;
72 char uri_path[WEBDAV_MAX_PATH];
73 size_t ulen;
74 int need_write;
75
76 rctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
77 if (rctx == NULL || !rctx->token_auth) {
78 return NGX_OK;
79 }
80
81 ulen = r->uri.len < sizeof(uri_path) - 1
82 ? r->uri.len : sizeof(uri_path) - 1;
83 ngx_memcpy(uri_path, r->uri.data, ulen);
84 uri_path[ulen] = '\0';
85
86 need_write = (webdav_token_op_class(r) == BRIX_TOKEN_OP_WRITE);
87
88 if (rctx->identity != NULL) {
89 if (brix_identity_check_token_scope(rctx->identity, uri_path,
90 need_write) == NGX_OK)
91 {
92 return NGX_OK;
93 }
94 } else if (need_write
95 ? brix_token_check_write(rctx->token_scopes,
96 rctx->token_scope_count, uri_path)
97 : brix_token_check_read(rctx->token_scopes,
98 rctx->token_scope_count, uri_path))
99 {
100 return NGX_OK;
101 }
102
103 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
104 "brix_webdav: token scope denies %s %s to \"%s\"",
105 method_name, need_write ? "write" : "read", uri_path);
106
107 return NGX_HTTP_FORBIDDEN;
108}

130/*
131 * webdav_bearer_from_query — extract a bearer token from ?authz= / ?access_token=.
132 *
133 * WHAT: query-string fallback used only when no Authorization header carries a
134 * Bearer token (davix/gfal2/xrdcp redirect + pre-signed-URL flows).
135 * WHY: XrdHttp accepts the token in the URL; matching it is required for WLCG
136 * client interop. The header path stays primary.
137 * HOW: copies the raw arg into r->pool, URL-decodes in place, strips an optional
138 * case-insensitive "Bearer " prefix, and enforces a length cap. On NGX_OK
139 * *out is a NUL-terminated pool slice. NGX_DECLINED when disabled/absent.
140 */
141static ngx_int_t
142webdav_bearer_from_query(ngx_http_request_t *r,
143 ngx_http_brix_webdav_loc_conf_t *conf, ngx_str_t *out)
144{
145 ngx_str_t raw;
146 size_t len;
147
148 if (!conf->http_query_token) {
149 return NGX_DECLINED;
150 }
151 if (brix_http_arg(r, "authz", 5, &raw) != NGX_OK
152 && brix_http_arg(r, "access_token", 12, &raw) != NGX_OK) {
153 return NGX_DECLINED;
154 }
155 len = brix_urldecode_inplace((char *) raw.data);
156 if (len >= 7 && ngx_strncasecmp(raw.data, (u_char *) "Bearer ", 7) == 0) {
157 raw.data += 7;
158 len -= 7;
159 while (len > 0 && raw.data[0] == ' ') { raw.data++; len--; }
160 }
161 if (len == 0 || len > WEBDAV_QUERY_TOKEN_MAX) {
162 return NGX_DECLINED;
163 }
164 out->data = raw.data;
165 out->len = len;
166 return NGX_OK;
167}

169/*
170 * wt_ensure_ctx — get-or-create the WebDAV request context with an identity.
171 *
172 * WHAT: Retrieves the per-request module context, allocating a fresh one (with
173 * an attached identity) on r->pool when absent, and back-fills an identity onto
174 * a pre-existing context that lacks one. Returns the context via *out.
175 * WHY: bearer-token auth runs after other auth phases may (or may not) have
176 * created the context; a NULL identity would crash the later claims-store step,
177 * so both the create and the repair paths must guarantee one. Factored out so
178 * the orchestrator's allocation-failure branches don't inflate its complexity.
179 * HOW: ngx_http_get_module_ctx(); on NULL allocate+attach identity+set_ctx; on a
180 * context with a NULL identity allocate one; map every OOM to
181 * NGX_HTTP_INTERNAL_SERVER_ERROR. Returns NGX_OK with *out populated otherwise.
182 */
183static ngx_int_t
184wt_ensure_ctx(ngx_http_request_t *r, ngx_http_brix_webdav_req_ctx_t **out)
185{
186 ngx_http_brix_webdav_req_ctx_t *ctx;
187
188 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
189 if (ctx == NULL) {
190 ctx = ngx_pcalloc(r->pool, sizeof(*ctx));
191 if (ctx == NULL) {
192 return NGX_HTTP_INTERNAL_SERVER_ERROR;
193 }
194 ctx->identity = brix_identity_alloc(r->pool);
195 if (ctx->identity == NULL) {
196 return NGX_HTTP_INTERNAL_SERVER_ERROR;
197 }
198 ngx_http_set_ctx(r, ctx, ngx_http_brix_webdav_module);
199 } else if (ctx->identity == NULL) {
200 ctx->identity = brix_identity_alloc(r->pool);
201 if (ctx->identity == NULL) {
202 return NGX_HTTP_INTERNAL_SERVER_ERROR;
203 }
204 }
205
206 *out = ctx;
207 return NGX_OK;
208}

210/*
211 * wt_parse_header — obtain the presented bearer token and scrub it from logs.
212 *
213 * WHAT: Resolves the request's bearer token from the Authorization header,
214 * falling back to the ?authz=/?access_token= query parameter, and — once the
215 * token has been consumed for auth — length-preservingly redacts any URL-borne
216 * token from every loggable request field. Returns the token via *bearer.
217 * WHY: WLCG clients present the token either in the header (primary) or the URL
218 * (davix/gfal2 redirect + pre-signed flows); both must be accepted with the same
219 * precedence XrdHttp uses. A URL token must never reach access/error logs, so
220 * the scrub happens here, immediately after extraction, before any early return
221 * that could log the request line.
222 * HOW: no header → query fallback (NGX_DECLINED when absent); header present but
223 * not Bearer → query fallback; a malformed Bearer header → NGX_HTTP_UNAUTHORIZED;
224 * on success redact r->args/unparsed_uri/request_line when a query string exists
225 * (r->uri excludes the query, so routing/scope are unaffected).
226 */
227static ngx_int_t
228wt_parse_header(ngx_http_request_t *r,
229 ngx_http_brix_webdav_loc_conf_t *conf, ngx_str_t *bearer)
230{
231 ngx_str_t auth_hdr;
232 int rc;
233
234 if (r->headers_in.authorization == NULL) {
235 /* No Authorization header — try the ?authz= query fallback (§1). */
236 if (webdav_bearer_from_query(r, conf, bearer) != NGX_OK) {
237 return NGX_DECLINED;
238 }
239 } else {
240 auth_hdr = r->headers_in.authorization->value;
241 rc = brix_http_extract_bearer(&auth_hdr, bearer);
242 if (rc == NGX_DECLINED) {
243 /* Header present but not Bearer — still allow the query fallback. */
244 if (webdav_bearer_from_query(r, conf, bearer) != NGX_OK) {
245 return NGX_DECLINED;
246 }
247 } else if (rc != NGX_OK) {
248 return NGX_HTTP_UNAUTHORIZED;
249 }
250 }
251
252 /* §1: the token has now been consumed for auth. Scrub any ?authz=/?access_token=
253 * value, length-preserving, from every log source (args, unparsed URI, request
254 * line) so a URL-borne bearer token never reaches access/error logs. r->uri (the
255 * decoded path used for routing/scope) excludes the query, so this is safe. */
256 if (r->args.len > 0) {
257 brix_http_redact_query_token(&r->args);
258 brix_http_redact_query_token(&r->unparsed_uri);
259 brix_http_redact_query_token(&r->request_line);
260 }
261
262 return NGX_OK;
263}

265/*
266 * wt_validate_registry — run path+op-scoped registry validation for a token.
267 *
268 * WHAT: Copies the request URI into a bounded NUL-terminated buffer, populates a
269 * brix_token_registry_args_t from the location config + parsed macaroon secret,
270 * and validates the token against the issuer registry, returning the validator
271 * rc (0 = accepted) with verified claims in *claims.
272 * WHY: registry authz (phase-59 W1) is path+op dependent and therefore MUST NOT
273 * consult the token-keyed caches — it is split out so the orchestrator can route
274 * a registry config straight here every request without cache branching.
275 * HOW: bound the URI to sizeof(pathz)-1, build the args struct (secret only when
276 * slen>0), derive the op class from the HTTP method, and call
277 * brix_token_validate_registry() with a throwaway bucket out-param.
278 */
279static int
280wt_validate_registry(const wt_validate_ctx_t *v)
281{
282 ngx_http_request_t *r = v->r;
283 char pathz[2048];
284 size_t plen;
285 int bucket = 0;
286 brix_token_registry_args_t ra;
287
288 plen = (r->uri.len < sizeof(pathz) - 1) ? r->uri.len : sizeof(pathz) - 1;
289 ngx_memcpy(pathz, r->uri.data, plen);
290 pathz[plen] = '\0';
291
292 ra.log = r->connection->log;
293 ra.token = v->token;
294 ra.token_len = v->token_len;
295 ra.reg = v->conf->token_registry;
296 ra.macaroon_secret = v->slen > 0 ? v->secret : NULL;
297 ra.secret_len = (size_t) v->slen;
298 ra.clock_skew = (int) v->conf->token_clock_skew;
299 ra.claims = v->claims;
300
301 return brix_token_validate_registry(&ra, pathz,
302 webdav_token_op_class(r), &bucket);
303}

305/*
306 * wt_validate_jwks — validate a token against the JWKS keys + macaroon secret.
307 *
308 * WHAT: Builds a brix_token_validate_args_t from the location config's JWKS keys,
309 * issuer/audience, clock skew, and the supplied secret, then runs the JWT/macaroon
310 * validator, returning its rc (0 = accepted) with verified claims in *claims.
311 * WHY: both the primary validation and the old-secret grace-period retry issue an
312 * identical validate call differing only in the secret bytes; centralising the
313 * args assembly keeps the two callsites byte-identical and off the orchestrator's
314 * complexity budget.
315 * HOW: populate the args struct (secret NULL when slen<=0), call
316 * brix_token_validate(); no side effects beyond writing *claims on success. The
317 * secret/slen are taken as explicit overrides so the grace-period retry can pass
318 * the old secret while reusing the same immutable context.
319 */
320static int
321wt_validate_jwks(const wt_validate_ctx_t *v,
322 const u_char *secret, ssize_t slen)
323{
324 brix_token_validate_args_t va;
325
326 va.log = v->r->connection->log;
327 va.token = v->token;
328 va.token_len = v->token_len;
329 va.keys = v->conf->jwks_keys;
330 va.key_count = v->conf->jwks_key_count;
331 va.expected_issuer = (const char *) v->conf->token_issuer.data;
332 va.expected_audience = (const char *) v->conf->token_audience.data;
333 va.macaroon_secret = slen > 0 ? secret : NULL;
334 va.secret_len = (size_t) slen;
335 va.clock_skew = (int) v->conf->token_clock_skew;
336 va.claims = v->claims;
337
338 return brix_token_validate(&va);
339}

341/*
342 * wt_grace_retry — retry macaroon validation with the rotated-out old secret.
343 *
344 * WHAT: When primary validation has failed on a non-registry request and an old
345 * macaroon secret is configured, parses it and re-runs JWKS/macaroon validation
346 * with the old key, logging an informational note on success. Returns the retry
347 * rc, or the unchanged incoming rc when the retry is not applicable.
348 * WHY: an nginx -s reload that rotates the macaroon secret would otherwise hard-
349 * break every in-flight macaroon; accepting the old secret until tokens expire
350 * gives graceful migration. Registry validation has no shared-secret concept and
351 * is excluded.
352 * HOW: return prev_rc unchanged unless (prev_rc!=0 && !via_registry && old secret
353 * present); parse the old secret, and only if it parses (>0) call wt_validate_jwks
354 * with it, emitting the grace-period NGX_LOG_INFO line when it now accepts.
355 */
356static int
357wt_grace_retry(const wt_validate_ctx_t *v, int via_registry, int prev_rc)
358{
359 ngx_http_brix_webdav_loc_conf_t *conf = v->conf;
360 u_char old_secret[64];
361 ssize_t old_slen;
362 int rc;
363
364 /* Grace-period fallback: if the primary secret rejected a macaroon token
365 * and an old secret is configured, try validating with the old key.
366 * This lets in-flight tokens survive nginx -s reload during key rotation. */
367 if (prev_rc == 0 || via_registry || conf->token_macaroon_secret_old.len == 0) {
368 return prev_rc;
369 }
370
371 old_slen = brix_macaroon_secret_parse(
372 (const char *) conf->token_macaroon_secret_old.data,
373 conf->token_macaroon_secret_old.len,
374 old_secret, sizeof(old_secret));
375 if (old_slen <= 0) {
376 return prev_rc;
377 }
378
379 rc = wt_validate_jwks(v, old_secret, old_slen);
380 if (rc == 0) {
381 ngx_log_error(NGX_LOG_INFO, v->r->connection->log, 0,
382 "brix_webdav: macaroon accepted via old secret "
383 "(grace-period key rotation)");
384 }
385 return rc;
386}

388/*
389 * wt_check_issuer_keys — resolve token validity via caches, registry or keys.
390 *
391 * WHAT: Determines whether the presented token is valid, filling *claims and
392 * flagging *cache_hit / *via_registry for the caller's caching decision. Returns
393 * 0 when the token is accepted (from L1, L2, registry, JWKS, or the grace-period
394 * old-secret retry), non-zero when rejected.
395 * WHY: this is the crypto/lookup hot path — consulting the cheapest source first
396 * (per-worker L1, then cross-worker L2 SHM) avoids re-running signature checks and
397 * JSON parsing on the event loop under load. Registry authz is path+op dependent
398 * so it MUST bypass the token-keyed caches and re-validate every request.
399 * HOW: lazily create L1; on a non-registry request probe L1 then L2 (promoting an
400 * L2 hit into L1); otherwise validate via registry or JWKS; finally apply the
401 * old-secret grace retry. Only successfully validated claims are ever cached, and
402 * caching itself is left to the caller.
403 */
404static int
405wt_check_issuer_keys(const wt_validate_ctx_t *v,
406 int *cache_hit, int *via_registry)
407{
408 ngx_http_brix_webdav_loc_conf_t *conf = v->conf;
409 const char *token = v->token;
410 size_t token_len = v->token_len;
411 brix_token_claims_t *claims = v->claims;
412 int rc;
413
414 /*
415 * Token-validation caches, consulted cheapest-first so token auth does not
416 * re-run crypto + JSON parsing on the event loop under load:
417 * L1 — always-on, per-worker, lockless (lazily created here). A hit skips
418 * BOTH the signature verification AND the L2 spinlock.
419 * L2 — the optional cross-worker SHM cache. An L2 hit is promoted into L1
420 * so the next presentation to this worker is an L1 hit.
421 * Only successfully validated claims are ever cached.
422 */
423 *cache_hit = 0;
424
425 /* The token-validity caches are keyed on the token alone. Registry authz
426 * (phase-59 W1) is path+op dependent, so a registry config MUST bypass the
427 * caches and re-run the per-request base_path/strategy check every time. */
428 *via_registry = (conf->token_registry != NULL);
429
430 if (conf->token_l1 == NULL) {
431 conf->token_l1 = brix_token_l1_create(ngx_cycle->pool,
432 BRIX_TOKEN_L1_SLOTS);
433 }
434
435 if (!*via_registry
436 && brix_token_l1_lookup(conf->token_l1, token, token_len, claims))
437 {
438 rc = 0;
439 *cache_hit = 1;
440 } else if (!*via_registry && conf->token_cache_kv != NULL
441 && brix_token_cache_lookup(conf->token_cache_kv,
442 token, token_len, claims))
443 {
444 rc = 0;
445 *cache_hit = 1;
446 brix_token_l1_store(conf->token_l1, token, token_len, claims);
447 } else if (*via_registry) {
448 rc = wt_validate_registry(v);
449 } else {
450 rc = wt_validate_jwks(v, v->secret, v->slen);
451 }
452
453 return wt_grace_retry(v, *via_registry, rc);
454}

456/*
457 * wt_check_claims — persist verified token claims onto the request context.
458 *
459 * WHAT: Marks the context token-authenticated, retains the raw JWT bytes for
460 * backend passthrough, records the identity claims (sub, scopes) and the DN, and
461 * copies the scope list. Returns NGX_OK, or NGX_HTTP_INTERNAL_SERVER_ERROR on an
462 * allocation failure.
463 * WHY: downstream scope enforcement (webdav_check_token_scope) and backend
464 * credential passthrough read these fields; centralising the store keeps the
465 * orchestrator a flat success path and matches the phase-70 §5.4 passthrough rule
466 * that the JWT bytes be copied onto r->pool (the wire buffer may be rewritten,
467 * e.g. by query-token redaction) and never logged.
468 * HOW: set verified/token_auth/auth_source; pnalloc+copy the bearer bytes; call
469 * brix_identity_set_token_claims(); cpystrn the subject into ctx->dn; copy up to
470 * BRIX_MAX_TOKEN_SCOPES scopes; emit the token-auth-OK INFO line.
471 */
472static ngx_int_t
473wt_check_claims(ngx_http_request_t *r,
474 ngx_http_brix_webdav_req_ctx_t *ctx,
475 const char *token, size_t token_len,
476 brix_token_claims_t *claims)
477{
478 int i;
479
480 ctx->verified = 1;
481 ctx->token_auth = 1;
482 ctx->auth_source = "token";
483
484 /* Phase-70 §5.4: retain the raw JWT bytes for backend PASSTHROUGH. Copied
485 * onto r->pool (the wire `token`/`bearer.data` may point into a header value
486 * that later rewrites, e.g. query-token redaction) and never logged. */
487 ctx->bearer_token.data = ngx_pnalloc(r->pool, token_len);
488 if (ctx->bearer_token.data == NULL) {
489 return NGX_HTTP_INTERNAL_SERVER_ERROR;
490 }
491 ngx_memcpy(ctx->bearer_token.data, token, token_len);
492 ctx->bearer_token.len = token_len;
493
494 if (brix_identity_set_token_claims(ctx->identity, r->pool, claims)
495 != NGX_OK)
496 {
497 return NGX_HTTP_INTERNAL_SERVER_ERROR;
498 }
499 ngx_cpystrn((u_char *) ctx->dn, (u_char *) claims->sub, sizeof(ctx->dn));
500
501 ctx->token_scope_count = claims->scope_count;
502 for (i = 0; i < claims->scope_count && i < BRIX_MAX_TOKEN_SCOPES; i++) {
503 ctx->token_scopes[i] = claims->scopes[i];
504 }
505
506 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
507 "brix_webdav: token auth OK sub=\"%s\" scopes=%d",
508 claims->sub, claims->scope_count);
509
510 return NGX_OK;
511}

513/*
514 *
515 * WHAT: Validates WLCG/SciToken bearer tokens presented in HTTP Authorization headers using either JWKS-based JWT verification or macaroon secret-key validation. Extracts token claims (subject, scopes, expiration) and stores them in the request context for downstream operations. Supports grace-period key rotation — if a macaroon is rejected by the current secret but accepted by an old secret configured via conf->token_macaroon_secret_old, the token is still accepted with an informational log message indicating graceful migration during nginx -s reload.
516 *
517 * WHY: WebDAV clients authenticate using bearer tokens rather than GSI certificates or anonymous access. This function must handle both JWT (via JWKS key set) and macaroon formats since different WLCG sites use different token types. The grace-period fallback prevents immediate access disruption during secret key rotation — in-flight tokens should be accepted until they naturally expire, avoiding a "hard break" scenario where all active clients are suddenly denied after a config reload.
518 *
519 * HOW: Declines if no keys/secrets configured; parses macaroon secret for validation if present; creates or retrieves request context (declines if already token-authenticated to avoid redundant verification); extracts a Bearer token from Authorization with shared case-insensitive scheme parsing via wt_parse_header(); resolves validity via wt_check_issuer_keys() (caches → registry/JWKS → grace retry); caches freshly verified non-registry claims; on success stores claims (sub, scopes) in ctx via wt_check_claims() for downstream scope checks. */
520ngx_int_t
521webdav_verify_bearer_token(ngx_http_request_t *r,
522 ngx_http_brix_webdav_loc_conf_t *conf)
523{
524 ngx_http_brix_webdav_req_ctx_t *ctx = NULL;
525 brix_token_claims_t claims;
526 ngx_str_t bearer;
527 const char *token;
528 size_t token_len;
529 int rc;
530 int cache_hit = 0;
531 int via_registry = 0;

535 ssize_t slen = 0;
536

538 && conf->token_registry == NULL)
539 {
540 return NGX_DECLINED;
541 }
542

544 slen = brix_macaroon_secret_parse(
545 (const char *) conf->token_macaroon_secret.data,
546 conf->token_macaroon_secret.len, secret, sizeof(secret));
547 }
548

550 if (crc != NGX_OK) {
551 return crc;
552 }
553
554 if (ctx->token_auth) {
555 return NGX_OK;
556 }
557
558 crc = wt_parse_header(r, conf, &bearer);
559 if (crc != NGX_OK) {
560 return crc;
561 }
562

564 token_len = bearer.len;
565
566 {
567 wt_validate_ctx_t v;
568
569 v.r = r;
570 v.conf = conf;
571 v.token = token;
572 v.token_len = token_len;
573 v.secret = secret;
574 v.slen = slen;
575 v.claims = &claims;
576
577 rc = wt_check_issuer_keys(&v, &cache_hit, &via_registry);
578 }
579
580 if (rc != 0) {
581 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
582 "brix_webdav: bearer token validation failed");
583 return NGX_HTTP_UNAUTHORIZED;
584 }
585
586 /* Cache the freshly verified claims for subsequent presentations (L1 always,
587 * L2 when a SHM zone is configured). Registry-authorized tokens are never
588 * cached: the decision is path-dependent and the cache is token-keyed. */
589 if (!cache_hit && !via_registry) {
590 brix_token_l1_store(conf->token_l1, token, token_len, &claims);
591 if (conf->token_cache_kv != NULL) {
592 brix_token_cache_store(conf->token_cache_kv, token, token_len,
593 &claims);
594 }
595 }
596
597 return wt_check_claims(r, ctx, token, token_len, &claims);
598}
