1/*
2 * src/write/wrts_journal.c — kXR_recoverWrts per-handle write-recovery journal.
3 *
4 * WHAT: Fixed-size ring buffer (BRIX_WRTS_JOURNAL_SLOTS entries per handle)
5 * recording committed (offset, length) write ranges. A replayed write
6 * whose range is fully covered by a journal entry is short-circuited:
7 * the pwrite() is skipped and kXR_ok is returned to the client.
8 *
9 * WHY: When kXR_recoverWrts is advertised, XrdCl replays in-flight writes
10 * after a TCP disconnect. Without this guard the server writes the bytes
11 * twice — data corruption. The journal makes the replay idempotent.
12 *
13 * HOW: Ring is indexed by (wrts_head % BRIX_WRTS_JOURNAL_SLOTS).
14 * brix_wrts_is_replay() does a linear O(N) scan over at most
15 * BRIX_WRTS_JOURNAL_SLOTS (64) entries — acceptable for all practical
16 * HEP workloads where write burst sizes are bounded.
17 *
18 * Thread safety: all callers run on the nginx event-loop thread for a single
19 * connection; no locking needed.
20 */

37/* brix_wrts_record */
38void
39brix_wrts_record(brix_file_t *f, int64_t offset, uint32_t length)
40{
41 brix_wrts_entry_t *e;
42 uint32_t slot;
43
44 if (!f->wrts_enabled || length == 0) {
45 return;
46 }
47
48 slot = f->wrts_head % BRIX_WRTS_JOURNAL_SLOTS;
49 e = &f->wrts_journal[slot];
50
51 e->offset = offset;
52 e->length = length;
53 e->gen = f->wrts_gen++;
54
55 f->wrts_head++;
56 if (f->wrts_count < BRIX_WRTS_JOURNAL_SLOTS) {
57 f->wrts_count++;
58 }
59}

61/* brix_wrts_is_replay */
62int
63brix_wrts_is_replay(const brix_file_t *f, int64_t offset, uint32_t length)
64{
65 uint32_t i;
66 uint32_t limit;
67 int64_t req_end;
68
69 if (!f->wrts_enabled || f->wrts_count == 0 || length == 0) {
70 return 0;
71 }
72
73 req_end = offset + (int64_t) length;
74 limit = f->wrts_count < BRIX_WRTS_JOURNAL_SLOTS
75 ? f->wrts_count
76 : BRIX_WRTS_JOURNAL_SLOTS;
77
78 for (i = 0; i < limit; i++) {
79 const brix_wrts_entry_t *e = &f->wrts_journal[i];
80
81 /* A replayed write has exactly the same offset and length.
82 * Range-coverage would incorrectly skip legitimate overwrites of
83 * a sub-range (e.g. writing 512 B after having written 4096 B at
84 * the same offset with different content). */
85 if (e->length > 0
86 && e->offset == offset
87 && e->length == (uint32_t) length)
88 {
89 return 1;
90 }
91 }
92
93 (void) req_end;
94
95 return 0;
96}
