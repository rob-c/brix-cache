28/*
29 * open_resolved_file_dispatch.c — backend/credential routing for kXR_open
30 * (phase-79 split from open_resolved_file.c).
31 *
32 * WHAT: Chooses HOW the resolved file is opened: resolve the export backend,
33 * apply the write-through sd_stage override, build the per-user backend
34 * credential VFS ctx, route a whole-object write through the staged-commit
35 * adapter, and otherwise dispatch to the driver or the POSIX-fd open.
36 *
37 * WHY: Keeps the backend-selection + credential-bind policy (which mirrors the
38 * davs/S3 call sites) apart from the raw open syscalls, so
39 * brix_open_dispatch_open reads as a two-way driver-vs-POSIX selector.
40 *
41 * HOW: brix_open_dispatch_open (the sole cross-file entry) selects the sd
42 * instance then either runs the driver branch (staged adapter or
43 * brix_open_resolved_via_driver) or brix_open_posix_dispatch, mapping any
44 * failure via brix_open_map_open_error. Byte-identical to the original.
45 */
46
47/* Whole-object staged write open (phase-70): put the freshly-allocated handle in
48 * STAGED mode instead of opening a session-writable driver handle a whole-object
49 * backend cannot provide. Opens a VFS staged handle on `vctx` (which the caller
50 * already bound with the per-user credential + deleg + .sd = the composed
51 * instance) via brix_vfs_staged_open — the SAME staged_open_cred + cred gate the
52 * WebDAV/S3 PUT bridge uses. The staged handle self-contains its ctx on the
53 * connection pool, so it survives across the open→writes→sync/close requests.
54 * On success synthesizes a minimal `struct stat` (no bytes yet), sets fd=-1 and
55 * driver_backed=1 (so brix_open_validate_fd skips the POSIX-fd checks), records
56 * the staged handle + zero append offset on the file slot, and returns
57 * NGX_DECLINED (proceed). On failure sends the mapped kXR error and returns its
58 * rc. Sequential-append + commit-on-sync/close are handled by the write/sync/
59 * close opcode handlers keyed on ctx->files[idx].staged != NULL. */
60static ngx_int_t
61brix_open_dispatch_staged(brix_open_args_t *a, brix_vfs_ctx_t *vctx,
62 brix_sd_instance_t *sd)
63{
64 brix_ctx_t *ctx = a->ctx;
65 ngx_connection_t *c = a->c;
66 brix_file_t *fh = &ctx->files[a->idx];
67 brix_vfs_staged_t *st;
68 int serr = 0;
69
70 vctx->sd = sd; /* brix_vfs_staged_open dispatches on ctx->sd */
71
72 st = brix_vfs_staged_open(vctx, a->create_mode, 16 /* excl-name attempts */,
73 &serr);
74 if (st == NULL) {
75 return brix_open_map_open_error(ctx, c, a->resolved,
76 serr != 0 ? serr : EROFS, a->is_write);
77 }
78
79 fh->staged = st;
80 fh->staged_expected_off = 0;
81 fh->staged_committed = 0;
82
83 /* No fd, no driver object: byte I/O routes through the staged handle. Mark
84 * driver_backed so the POSIX-fd validation is skipped and synthesize a stat
85 * (a brand-new whole object — zero bytes, regular file with the create mode). */
86 a->fd = NGX_INVALID_FILE;
87 a->driver_backed = 1;
88 ngx_memzero(a->st, sizeof(*a->st));
89 a->st->st_mode = S_IFREG | (a->create_mode ? a->create_mode : 0644);
90 return NGX_DECLINED;
91}

93/* WHAT: Select the storage-driver instance for this open, applying the write-
94 * through override.
95 * WHY: Isolates the wt sd_stage selection (Option A) from the dispatch control
96 * flow so the caller reads as a single backend-resolve step.
97 * HOW: Resolves the export backend, then — for a fresh (non-cache/non-resume)
98 * WRITE with wt.enable and a composed wt sd_stage — swaps in the wt
99 * instance and sets a->wt_via_stage. Returns the selected instance (may be
100 * NULL); leaves wt_via_stage untouched (caller pre-zeroes it) when no
101 * override applies. */
102static brix_sd_instance_t *
103brix_open_select_sd_inst(brix_open_args_t *a)
104{
105 ngx_stream_brix_srv_conf_t *conf = a->conf;
106 brix_sd_instance_t *sd_inst =
107 brix_vfs_backend_resolve(conf->common.root_canon, a->c->log);
108
109 if (a->is_write && conf->wt.enable && !a->from_cache && !a->use_resume) {
110 brix_sd_instance_t *wt = brix_cache_wt_stage_sd_inst(conf);
111
112 if (wt != NULL) {
113 sd_inst = wt;
114 a->wt_via_stage = 1;
115 }
116 }
117 return sd_inst;
118}

120/* WHAT: Build the transient per-open VFS ctx that scopes a driver open to the
121 * authenticated client identity.
122 * WHY: The cred/mint/deleg bind sequence is identical to the davs/S3 call sites
123 * and is pure setup — pulling it out of the dispatcher keeps the control
124 * flow legible without changing any bind order.
125 * HOW: Initialises *cred_vctx for the root:// export, binds the credential-dir/
126 * fallback policy, opt-in mint CA, and delegation, then pins the selected
127 * sd instance. No-ops (use_cred/mint stay 0) when the corresponding config
128 * is unset. */
129static void
130brix_open_build_cred_ctx(brix_open_args_t *a, brix_sd_instance_t *sd_inst,
131 brix_vfs_ctx_t *cred_vctx)
132{
133 brix_ctx_t *ctx = a->ctx;
134 ngx_connection_t *c = a->c;
135 ngx_stream_brix_srv_conf_t *conf = a->conf;
136
137 brix_vfs_ctx_init(cred_vctx, c->pool, c->log, BRIX_PROTO_ROOT,
138 conf->common.root_canon, NULL, conf->common.allow_write,
139 0 /* is_tls */, ctx->identity, a->resolved);
140 brix_vfs_ctx_bind_backend_cred(cred_vctx,
141 &conf->common.storage_credential_dir,
142 conf->common.storage_credential_fallback);
143 /* Phase-3 T1: opt-in credential minting for GSI/token identities that
144 * have no pre-provisioned proxy — mirrors the davs/S3 mint bind call
145 * sites (Phase-2 T9). No-op unless a mint CA is configured. */
146 brix_vfs_ctx_bind_backend_mint(cred_vctx,
147 &conf->common.storage_credential_mint_ca_cert,
148 &conf->common.storage_credential_mint_ca_key,
149 conf->common.storage_credential_mint_ttl);
150 brix_root_vfs_bind_deleg(ctx, conf, cred_vctx);
151 cred_vctx->sd = sd_inst;
152}

154/* WHAT: Report whether a WRITE open must use the whole-object staged-commit
155 * adapter instead of a session-writable driver open.
156 * WHY: A whole-object staged backend (sd_http and any driver that advertises NO
157 * BRIX_SD_CAP_RANDOM_WRITE and has no .pwrite — writes are a single commit-
158 * time PUT) cannot satisfy the root:// block-oriented write model; the
159 * plain/cred open slot would refuse with EROFS.
160 * HOW: True only for a WRITE whose ns leaf lacks random-write capability and has
161 * no .pwrite; the caller then routes through brix_open_dispatch_staged. */
162static int
163brix_open_write_needs_staged(brix_open_args_t *a, brix_sd_instance_t *sd_inst)
164{
165 brix_sd_instance_t *leaf;
166
167 if (!a->is_write) {
168 return 0;
169 }
170 leaf = brix_vfs_ns_leaf(sd_inst);
171 return leaf != NULL
172 && !(brix_sd_caps(leaf) & BRIX_SD_CAP_RANDOM_WRITE)
173 && (leaf->driver == NULL || leaf->driver->pwrite == NULL);
174}

176/* WHAT: Dispatch a driver-backed open (the sd_inst branch), routing writes that
177 * need the whole-object staged adapter and otherwise opening via the driver.
178 * WHY: Contains the credential-bind + staged-adapter + driver-open decisions so
179 * brix_open_dispatch_open stays a two-way (driver vs POSIX) selector.
180 * HOW: Marks driver_backed, builds the per-open cred ctx, and — for a staged-
181 * write — returns the staged dispatch rc via *out_rc and NGX_DONE. Otherwise
182 * opens through the driver keyed on the export-root-relative name and returns
183 * NGX_OK, writing open_failed (0/1) to *out_failed for the caller to map. */
184static ngx_int_t
185brix_open_dispatch_driver(brix_open_args_t *a, brix_sd_instance_t *sd_inst,
186 ngx_int_t *out_failed, ngx_int_t *out_rc)
187{
188 ngx_stream_brix_srv_conf_t *conf = a->conf;
189 const char *resolved = a->resolved;
190 brix_vfs_ctx_t cred_vctx;
191
192 a->driver_backed = 1;
193 /* Per-user backend credential (Phase 2 Task 6): bind the authenticated
194 * root:// session identity + the export's credential-dir/fallback policy so
195 * the driver open scopes the origin session to the CLIENT rather than the
196 * shared service credential — mirrors the davs/S3 brix_vfs_open call sites
197 * (Phase 1). */
198 brix_open_build_cred_ctx(a, sd_inst, &cred_vctx);
199
200 /* Whole-object staged-commit adapter (phase-70): open a VFS staged handle
201 * (the same block-body → staged_write+staged_commit bridge WebDAV/S3 PUT use)
202 * and put this file handle in STAGED mode; kXR_write/pgwrite then APPEND, and
203 * kXR_sync/close COMMIT the whole object. brix_open_dispatch_staged binds the
204 * per-user credential + deleg via cred_vctx exactly as the driver open does. */
205 if (brix_open_write_needs_staged(a, sd_inst)) {
206 *out_rc = brix_open_dispatch_staged(a, &cred_vctx, sd_inst);
207 return NGX_DONE;
208 }
209
210 /* Key the driver namespace on the export-root-relative ("/sub/file") form —
211 * the same convention WebDAV/S3 and the VFS stat/dirlist/unlink paths use
212 * (brix_vfs_export_relative_root, leading slash retained), so a file written
213 * here is found by every other driver-backed op. */
214 *out_failed = brix_open_resolved_via_driver(a, &cred_vctx, sd_inst,
215 brix_vfs_export_relative_root(resolved, conf->common.root_canon))
216 != NGX_OK ? 1 : 0;
217 return NGX_OK;
218}

220/* Open the resolved file, routing between the export's storage driver (Layer 3)
221 * and the POSIX-fd path (POSC/resume staging + server-managed cache domain).
222 * Resolves the export's backend, selects the composed wt sd_stage for a write-
223 * through write, binds the per-user backend credential for a driver open, then
224 * dispatches. On success writes *out_fd + *st, sets *driver_backed / *wt_via_stage,
225 * and returns NGX_DECLINED (proceed). On open failure sends the mapped kXR error
226 * and returns that send's rc. `idx` is the already-allocated handle slot. */
227ngx_int_t
228brix_open_dispatch_open(brix_open_args_t *a)
229{
230 brix_ctx_t *ctx = a->ctx;
231 ngx_connection_t *c = a->c;
232 const char *resolved = a->resolved;
233 ngx_int_t open_failed = 0;
234 brix_sd_instance_t *sd_inst;
235
236 a->driver_backed = 0;
237 a->wt_via_stage = 0;
238
239 sd_inst = brix_open_select_sd_inst(a);
240
241 /* §14 (phase-64): the legacy driver-backed read cache (cache_storage_backend)
242 * and the legacy slice decorator (cache_slice_inst) are RETIRED — a driver
243 * cache store and slice/partial serving are the tier grammar's composed
244 * sd_cache, reached through the sd_inst branch below. A POSIX `brix_cache
245 * on` cache keeps the raw-fd from_cache path. */
246 if (sd_inst != NULL && !a->from_cache && !a->use_resume) {
247 ngx_int_t staged_rc = NGX_DECLINED;
248
249 if (brix_open_dispatch_driver(a, sd_inst, &open_failed, &staged_rc)
250 == NGX_DONE) {
251 return staged_rc;
252 }
253 } else {
254 open_failed = brix_open_posix_dispatch(a);
255 }
256 if (open_failed) {
257 return brix_open_map_open_error(ctx, c, resolved, errno, a->is_write);
258 }
259 return NGX_DECLINED;
260}
