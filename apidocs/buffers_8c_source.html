19/*
20 * brix_build_single_memory_chain — build a two-link chain for a kXR_ok
21 * response whose data fits in one wire chunk (data_total <= BRIX_READ_CHUNK_MAX).
22 *
23 * WHAT: Constructs an nginx ngx_chain_t with exactly two links: a header buffer
24 * containing the ServerResponseHdr (streamid, status=kXR_ok, length) followed by
25 * a single data buffer pointing into databuf. Both buffers are memory-backed
26 * (memory=1), enabling nginx to write via read()+write() rather than sendfile.
27 *
28 * WHY: The vast majority of xrdcp reads produce responses under 16 MiB that fit
29 * in one wire chunk. Allocating fresh ngx_chain_t and ngx_buf_t structures on
30 * every response would waste pool memory and CPU cycles. This function uses the
31 * pre-allocated slot->read_fast_* structs (hdr_chain, body_chain, hdr_buf, body_buf)
32 * to avoid any pool allocation — zero-cost for the common case.
33 *
34 * HOW: 1) Acquires rd.read_hdr_scratch via brix_get_read_header_scratch.
35 * 2) Calls brix_build_resp_hdr() to populate ServerResponseHdr with
36 * ctx->recv.cur_streamid, kXR_ok status, and data_total length.
37 * 3) Memzeros the fast structs then configures hdr_buf → hdr_chain (header).
38 * 4) If data_total==0 returns header-only chain; otherwise configures
39 * body_buf pointing into databuf → body_chain (data), linking them.
40 * 5) Sets last_buf/last_in_chain on the final buffer to signal end-of-response.
41 *
42 * Precondition: data_total <= BRIX_READ_CHUNK_MAX.
43 */
44static ngx_chain_t *
45brix_build_single_memory_chain(brix_ctx_t *ctx, ngx_connection_t *c,
46 u_char *databuf, size_t data_total)
47{
48 brix_resp_slot_t *slot = &ctx->out.ring[ctx->out.tail];
49 /*
50 * Header lives in THIS slot's private buffer (not the shared
51 * rd.read_hdr_scratch), so pipelining the next memory read into another slot
52 * cannot overwrite a still-draining response's 8-byte header. This is what
53 * makes the memory (userspace-TLS) read path safe to pipeline; the body
54 * buffer is made per-in-flight by brix_acquire_read_buffer() in read.c.
55 */
56 u_char *hdrbuf = slot->hdr_bytes;
57
58 (void) c;
59 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok, (uint32_t) data_total,
60 (ServerResponseHdr *) hdrbuf);
61
62 ngx_memzero(&slot->read_fast_hdr_chain, sizeof(slot->read_fast_hdr_chain));
63 ngx_memzero(&slot->read_fast_body_chain, sizeof(slot->read_fast_body_chain));
64 ngx_memzero(&slot->read_fast_hdr_buf, sizeof(slot->read_fast_hdr_buf));
65 ngx_memzero(&slot->read_fast_body_buf, sizeof(slot->read_fast_body_buf));
66
67 slot->read_fast_hdr_buf.pos = hdrbuf;
68 slot->read_fast_hdr_buf.last = hdrbuf + XRD_RESPONSE_HDR_LEN;
69 slot->read_fast_hdr_buf.memory = 1;
70 slot->read_fast_hdr_buf.temporary = 1;
71
72 slot->read_fast_hdr_chain.buf = &slot->read_fast_hdr_buf;
73 slot->read_fast_hdr_chain.next = NULL;
74
75 if (data_total == 0) {
76 slot->read_fast_hdr_buf.last_buf = 1;
77 slot->read_fast_hdr_buf.last_in_chain = 1;
78 return &slot->read_fast_hdr_chain;
79 }
80
81 slot->read_fast_body_buf.pos = databuf;
82 slot->read_fast_body_buf.last = databuf + data_total;
83 slot->read_fast_body_buf.memory = 1;
84 slot->read_fast_body_buf.temporary = 1;
85 slot->read_fast_body_buf.last_buf = 1;
86 slot->read_fast_body_buf.last_in_chain = 1;
87
88 slot->read_fast_body_chain.buf = &slot->read_fast_body_buf;
89 slot->read_fast_body_chain.next = NULL;
90 slot->read_fast_hdr_chain.next = &slot->read_fast_body_chain;
91
92 return &slot->read_fast_hdr_chain;
93}

95/*
96 * brix_build_window_chain — build a single memory-backed response chunk with
97 * an explicit wire status (Phase 31 W2.1 windowed reads).
98 *
99 * Identical layout to brix_build_single_memory_chain (header + one data buf,
100 * reusing the pre-allocated read_fast_* structs), but the caller chooses the
101 * status: kXR_oksofar for every window except the last, kXR_ok for the final
102 * window. The client accumulates the oksofar frames (same streamid) until the
103 * kXR_ok, reassembling the full read — the same wire sequence the >16 MiB
104 * multi-chunk path already emits, just sourced one window at a time.
105 *
106 * Precondition: data_total <= BRIX_READ_CHUNK_MAX (a window is <= 2 MiB).
107 */

114

116 XRD_RESPONSE_HDR_LEN);

118 return NULL;
119 }
120

123

155 slot->read_fast_hdr_chain.next = &slot->read_fast_body_chain;

160/*
161 * brix_chunk_geometry — compute the wire-frame count and final-frame size for
162 * a multi-chunk read.
163 *
164 * WHAT: Writes *n_chunks_out = ceil(data_total / CHUNK_MAX) and *last_size_out =
165 * the size of the FINAL frame, remapping an exact-multiple remainder of 0 back
166 * to a full CHUNK_MAX. No return value.
167 *
168 * WHY: Both brix_build_chunked_chain and brix_build_sendfile_chain need the same
169 * chunk-count arithmetic; centralising it keeps the two builders in lock-step and
170 * removes the duplicated ceil-divide + remainder-remap decision points from each.
171 * (Non-static: brix_build_sendfile_chain lives in buffers_sendfile.c — see
172 * buffers_internal.h.)
173 *
174 * HOW: 1) n_chunks = (data_total + CHUNK_MAX - 1) / CHUNK_MAX (the +MAX-1 ceil
175 * trick); clamp 0 to 1 as defence-in-depth (unreachable when called after
176 * the data_total > CHUNK_MAX guard, but keeps the loop non-empty).
177 * 2) last_size = data_total % CHUNK_MAX; a zero remainder with data_total > 0
178 * means the last frame must carry a full CHUNK_MAX, not zero bytes, so
179 * remap 0 back to CHUNK_MAX.
180 */
181void
182brix_chunk_geometry(size_t data_total, size_t *n_chunks_out,
183 size_t *last_size_out)
184{
185 size_t n_chunks;
186 size_t last_size;
187
188 n_chunks = (data_total + BRIX_READ_CHUNK_MAX - 1)
189 / BRIX_READ_CHUNK_MAX;
190 if (n_chunks == 0) {
191 n_chunks = 1;
192 }
193
194 last_size = data_total % BRIX_READ_CHUNK_MAX;
195 if (last_size == 0 && data_total > 0) {
196 last_size = BRIX_READ_CHUNK_MAX;
197 }
198
199 *n_chunks_out = n_chunks;
200 *last_size_out = last_size;
201}

203/*
204 * brix_chain_link_push — append one already-populated chain link to a
205 * head/tail-tracked singly linked list.
206 *
207 * WHAT: Terminates cl (cl->next = NULL), then links it as the new tail: sets
208 * *head on the first push, otherwise splices onto (*tail)->next. Updates *tail.
209 * No return value.
210 *
211 * WHY: The chunked and sendfile builders both grow their response chain link by
212 * link; funnelling the head-first vs append-to-tail decision through one helper
213 * keeps the two loop bodies flat and identical in ordering.
214 *
215 * HOW: 1) Null-terminate the new link. 2) If the list is empty (*head == NULL)
216 * it becomes the head; otherwise the current tail's next points at it.
217 * 3) Advance *tail to the new link.
218 */
219static void
220brix_chain_link_push(ngx_chain_t **head, ngx_chain_t **tail, ngx_chain_t *cl)
221{
222 cl->next = NULL;
223
224 if (*head == NULL) {
225 *head = cl;
226 } else {
227 (*tail)->next = cl;
228 }
229
230 *tail = cl;
231}

233/*
234 * brix_chain_append_mem — allocate and append a memory-backed buffer link
235 * spanning [pos, last).
236 *
237 * WHAT: Allocates one ngx_chain_t + ngx_buf_t from pool, points the buffer at
238 * [pos, last) as memory/temporary, and appends it via brix_chain_link_push.
239 * Returns NGX_OK, or NGX_ERROR if either allocation fails (list left unchanged).
240 *
241 * WHY: Every wire header link — and every data link on the memory (chunked) path
242 * — is an identical memory-backed buffer; sharing one builder removes the
243 * duplicated alloc + NULL-check + splice from both callers. (Non-static: the
244 * sendfile builder in buffers_sendfile.c also appends the memory header links.)
245 *
246 * HOW: 1) Allocate the chain link and buffer; return NGX_ERROR on either NULL.
247 * 2) Set pos/last and memory=temporary=1. 3) Attach the buffer and push the
248 * link onto the caller's head/tail list.
249 */
250ngx_int_t
251brix_chain_append_mem(ngx_pool_t *pool, ngx_chain_t **head, ngx_chain_t **tail,
252 u_char *pos, u_char *last)
253{
254 ngx_chain_t *cl;
255 ngx_buf_t *b;
256
257 cl = ngx_alloc_chain_link(pool);
258 b = ngx_calloc_buf(pool);
259 if (cl == NULL || b == NULL) {
260 return NGX_ERROR;
261 }
262
263 b->pos = pos;
264 b->last = last;
265 b->memory = 1;
266 b->temporary = 1;
267
268 cl->buf = b;

323/*
324 * brix_build_chunked_chain — build a multi-chunk memory chain for large
325 * reads (data_total > BRIX_READ_CHUNK_MAX = 16 MiB).
326 *
327 * XRootD responses larger than 16 MiB must be split into multiple wire frames:
328 * [kXR_oksofar frame][data][kXR_oksofar frame][data]...[kXR_ok frame][data]
329 *
330 * Each header is written into a contiguous block in rd.read_hdr_scratch (avoiding
331 * N separate pool allocations). Each data link points into databuf at the
332 * corresponding offset — zero copy from the AIO receive buffer to the wire.
333 *
334 * Falls through to brix_build_single_memory_chain for data_total <= CHUNK_MAX.
335 */
336ngx_chain_t *
337brix_build_chunked_chain(brix_ctx_t *ctx, ngx_connection_t *c,
338 u_char *databuf, size_t data_total)
339{
340 size_t n_chunks, last_size;
341 u_char *hdrbuf;
342 ngx_chain_t *head = NULL, *tail = NULL;
343 size_t di = 0;
344 size_t chunk;
345
346 if (data_total <= BRIX_READ_CHUNK_MAX) {
347 return brix_build_single_memory_chain(ctx, c, databuf, data_total);
348 }
349
350 /*
351 * Chunk-count math (shared with brix_build_sendfile_chain via
352 * brix_chunk_geometry): n_chunks = ceil(data_total / CHUNK_MAX); last_size =
353 * size of the FINAL chunk, with a zero-remainder exact multiple remapped back
354 * to CHUNK_MAX so the last frame carries the trailing bytes, never zero.
355 */
356 brix_chunk_geometry(data_total, &n_chunks, &last_size);
357
358 /*
359 * One contiguous scratch block holds all N wire headers back-to-back
360 * (8 bytes each); each iteration writes into its own slice at
361 * hdrbuf + chunk*XRD_RESPONSE_HDR_LEN, avoiding N separate allocations.
362 */
363 hdrbuf = BRIX_GET_SCRATCH(ctx, c, rd.read_hdr_scratch, rd.read_hdr_scratch_size,
364 n_chunks * XRD_RESPONSE_HDR_LEN);
365 if (hdrbuf == NULL) {
366 return NULL;
367 }
368
369 for (chunk = 0; chunk < n_chunks; chunk++) {
370 size_t chunk_data;
371 uint16_t status;
372 u_char *hptr;
373
374 /*
375 * Every chunk but the last is a full CHUNK_MAX; the last carries
376 * last_size. The final frame's status is kXR_ok (end of response);
377 * all earlier frames are kXR_oksofar — the client accumulates the
378 * oksofar frames under the same streamid until the kXR_ok arrives.
379 */
380 chunk_data = (chunk < n_chunks - 1) ? BRIX_READ_CHUNK_MAX
381 : last_size;
382 status = (chunk == n_chunks - 1) ? kXR_ok : kXR_oksofar;
383 hptr = hdrbuf + chunk * XRD_RESPONSE_HDR_LEN;
384
385 brix_build_resp_hdr(ctx->recv.cur_streamid, status,
386 (uint32_t) chunk_data,
387 (ServerResponseHdr *) hptr);
388
389 if (brix_chain_append_mem(c->pool, &head, &tail, hptr,
390 hptr + XRD_RESPONSE_HDR_LEN) != NGX_OK)
391 {
392 return NULL;
393 }
394
395 if (chunk_data > 0) {
396 /* di walks the data offset; each data link points straight into
397 * databuf — zero copy from the AIO receive buffer to the wire. */
398 if (brix_chain_append_mem(c->pool, &head, &tail, databuf + di,
399 databuf + di + chunk_data) != NGX_OK)
400 {
401 return NULL;
402 }
403 di += chunk_data;
404 }
405 }
406
407 if (tail != NULL) {
408 tail->buf->last_buf = 1;
409 }
410
411 return head;
412}
