111/* ---- Open the audit log file for a location ----
112 *
113 * WHAT: `brix_guard_audit_log <path>` — resolves the path through nginx's
114 * shared open-file list (append mode, reopened on USR1) into
115 * lcf->audit_log. Returns NGX_CONF_OK or NGX_CONF_ERROR.
116 *
117 * WHY: ngx_conf_open_file gives every worker one O_APPEND fd (short writes
118 * stay atomic) and standard log-rotation semantics for free.
119 *
120 * HOW: 1. Reject a duplicate directive.
121 * 2. ngx_conf_open_file(cf->cycle, &value[1]).
122 */
123static char *
124ngx_http_brix_guard_audit_log_slot(ngx_conf_t *cf, ngx_command_t *cmd,
125 void *conf)
126{
127 ngx_http_brix_guard_loc_conf_t *lcf = conf;
128 ngx_str_t *value = cf->args->elts;
129
130 (void) cmd;
131
132 if (lcf->audit_log != NULL) {
133 return "is duplicate";
134 }
135 lcf->audit_log = ngx_conf_open_file(cf->cycle, &value[1]);
136 if (lcf->audit_log == NULL) {
137 return NGX_CONF_ERROR;
138 }
139 return NGX_CONF_OK;
140}

143/* ---- Append directive arguments to an ngx_str_t array ----
144 *
145 * WHAT: generic repeatable-directive setter: pushes every argument of the
146 * directive onto the ngx_array_t at cmd->offset, creating the array on
147 * first use. Serves brix_guard_signature / _valid_prefix (TAKE1) and
148 * _valid_method (1MORE).
149 *
150 * WHY: the three list directives differ only in which array they fill; one
151 * setter keeps the command table declarative.
152 *
153 * HOW: 1. Create the array when the field is still NGX_CONF_UNSET_PTR.
154 * 2. Push each value[1..nelts) (conf strings are NUL-terminated and
155 * pool-owned, so the ruleset can borrow them).
156 */
157static char *
158ngx_http_brix_guard_array_slot(ngx_conf_t *cf, ngx_command_t *cmd,
159 void *conf)
160{
161 char *base = conf;
162 ngx_array_t **list;
163 ngx_str_t *value = cf->args->elts;
164 ngx_str_t *entry;
165 ngx_uint_t arg_index;
166
167 list = (ngx_array_t **) (base + cmd->offset);
168
169 if (*list == NGX_CONF_UNSET_PTR || *list == NULL) {
170 *list = ngx_array_create(cf->pool, 4, sizeof(ngx_str_t));
171 if (*list == NULL) {
172 return NGX_CONF_ERROR;
173 }
174 }
175 for (arg_index = 1; arg_index < cf->args->nelts; arg_index++) {
176 entry = ngx_array_push(*list);
177 if (entry == NULL) {
178 return NGX_CONF_ERROR;
179 }
180 *entry = value[arg_index];
181 }
182 return NGX_CONF_OK;
183}

186/* ---- Map an HTTP method name to a guard op-class ----
187 *
188 * WHAT: returns the guard_op_class_t for a method-name string ("GET",
189 * "PROPFIND", ...); GUARD_OP_UNKNOWN for names outside the table.
190 *
191 * WHY: brix_guard_valid_method narrows the grammar per location; the
192 * mapping must agree with the request-time one in guard_http_req.c.
193 *
194 * HOW: 1. Walk a static name→op descriptor table, case-insensitive compare.
195 */
196static guard_op_class_t
197method_name_to_op(const ngx_str_t *name)
198{
199 static const struct { const char *name; guard_op_class_t op; } table[] = {
200 { "GET", GUARD_OP_READ }, { "HEAD", GUARD_OP_READ },
201 { "PUT", GUARD_OP_WRITE }, { "POST", GUARD_OP_WRITE },
202 { "DELETE", GUARD_OP_DELETE }, { "PROPFIND", GUARD_OP_LIST },
203 { "OPTIONS", GUARD_OP_INFO },
204 };
205 size_t entry_index;
206
207 for (entry_index = 0; entry_index < sizeof(table) / sizeof(table[0]);
208 entry_index++)
209 {
210 if (name->len == ngx_strlen(table[entry_index].name)
211 && ngx_strncasecmp(name->data,
212 (u_char *) table[entry_index].name, name->len) == 0)
213 {
214 return table[entry_index].op;
215 }
216 }
217 return GUARD_OP_UNKNOWN;
218}

221/* ---- Build the location's ruleset from merged config ----
222 *
223 * WHAT: assembles lcf->ruleset: profile grammar defaults, built-in +
224 * operator signatures, operator namespace prefixes, and the optional
225 * method restriction. Returns NGX_CONF_OK.
226 *
227 * WHY: doing this once at merge time means the ACCESS/LOG handlers only ever
228 * read a finished ruleset — no per-request config interpretation.
229 *
230 * HOW: 1. guard_ruleset_init.
231 * 2. Built-in scanner signatures unless disabled.
232 * 3. Profile grammar defaults (conf strings are NUL-terminated).
233 * 4. Operator signatures (substring kind) + prefixes.
234 * 5. If methods were listed: reset op_allowed to exactly those ops.
235 */
236static char *
237ngx_http_brix_guard_build_ruleset(ngx_conf_t *cf,
238 ngx_http_brix_guard_loc_conf_t *lcf)
239{
240 ngx_str_t *entries;
241 ngx_uint_t entry_index;
242
243 guard_ruleset_init(&lcf->ruleset);
244
245 if (lcf->default_sigs) {
246 guard_ruleset_add_default_signatures(&lcf->ruleset);
247 }
248
249 guard_ruleset_load_profile(&lcf->ruleset,
250 lcf->profile.len ? (const char *) lcf->profile.data : "");
251
252 if (lcf->extra_sigs != NULL) {
253 entries = lcf->extra_sigs->elts;
254 for (entry_index = 0; entry_index < lcf->extra_sigs->nelts;
255 entry_index++)
256 {
257 if (!guard_ruleset_add_signature(&lcf->ruleset, GUARD_SIG_SUBSTR,
258 (const char *) entries[entry_index].data,
259 entries[entry_index].len))
260 {
261 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
262 "brix_guard_signature: more than %d signatures",
263 GUARD_MAX_SIGS);
264 return NGX_CONF_ERROR;
265 }
266 }
267 }
268
269 if (lcf->prefixes != NULL) {
270 entries = lcf->prefixes->elts;
271 for (entry_index = 0; entry_index < lcf->prefixes->nelts;
272 entry_index++)
273 {
274 if (!guard_ruleset_add_prefix(&lcf->ruleset,
275 (const char *) entries[entry_index].data,
276 entries[entry_index].len))
277 {
278 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
279 "brix_guard_valid_prefix: more than %d prefixes",
280 GUARD_MAX_PREFIXES);
281 return NGX_CONF_ERROR;
282 }
283 }
284 }
285
286 if (lcf->methods != NULL && lcf->methods->nelts > 0) {
287 int op_index;
288
289 for (op_index = 0; op_index <= GUARD_OP_UNKNOWN; op_index++) {
290 lcf->ruleset.op_allowed[op_index] = 0;
291 }
292 entries = lcf->methods->elts;
293 for (entry_index = 0; entry_index < lcf->methods->nelts;
294 entry_index++)
295 {
296 guard_op_class_t op = method_name_to_op(&entries[entry_index]);
297
298 if (op == GUARD_OP_UNKNOWN) {
299 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
300 "brix_guard_valid_method: unknown method \"%V\"",
301 &entries[entry_index]);
302 return NGX_CONF_ERROR;
303 }
304 lcf->ruleset.op_allowed[op] = 1;
305 }
306 }
307
308 return NGX_CONF_OK;
309}

312/* ---- Create the per-location config ----
313 *
314 * WHAT: allocates ngx_http_brix_guard_loc_conf_t with every field in the
315 * nginx "unset" state so merge can inherit from the parent level.
316 *
317 * WHY: standard nginx conf lifecycle — anything not UNSET here would silently
318 * shadow parent-level directives.
319 *
320 * HOW: 1. ngx_pcalloc (zeroes ruleset + audit_log).
321 * 2. Mark scalars NGX_CONF_UNSET, arrays NGX_CONF_UNSET_PTR.
322 */
323static void *
324ngx_http_brix_guard_create_loc_conf(ngx_conf_t *cf)
325{
326 ngx_http_brix_guard_loc_conf_t *lcf;
327
328 lcf = ngx_pcalloc(cf->pool, sizeof(*lcf));
329 if (lcf == NULL) {
330 return NULL;
331 }
332 lcf->enable = NGX_CONF_UNSET;
333 lcf->default_sigs = NGX_CONF_UNSET;
334 lcf->bounce_status = NGX_CONF_UNSET;
335 lcf->extra_sigs = NGX_CONF_UNSET_PTR;
336 lcf->prefixes = NGX_CONF_UNSET_PTR;
337 lcf->methods = NGX_CONF_UNSET_PTR;
338 /* profile zeroed by pcalloc; audit_log NULL = inherit */
339 return lcf;
340}

343/* ---- Merge parent/child configs + build the ruleset ----
344 *
345 * WHAT: applies inheritance and defaults (bounce 444, default signatures on),
346 * validates bounce_status is 403 or 444, then builds the location's
347 * ruleset. Returns NGX_CONF_OK / NGX_CONF_ERROR.
348 *
349 * WHY: the ruleset must be built after every directive at every level has
350 * been seen — merge time is the single point where that holds.
351 *
352 * HOW: 1. Merge each scalar/array/file field from the parent.
353 * 2. Reject bounce codes other than 403/444 (anything else would turn
354 * the guard into an open redirect for confusing status codes).
355 * 3. Build the ruleset only for enabled locations (disabled locations
356 * keep the zeroed ruleset and never classify).
357 */
358static char *
359ngx_http_brix_guard_merge_loc_conf(ngx_conf_t *cf, void *parent, void *child)
360{
361 ngx_http_brix_guard_loc_conf_t *prev = parent;
362 ngx_http_brix_guard_loc_conf_t *conf = child;
363
364 ngx_conf_merge_value(conf->enable, prev->enable, 0);
365 ngx_conf_merge_str_value(conf->profile, prev->profile, "");
366 ngx_conf_merge_value(conf->default_sigs, prev->default_sigs, 1);
367 ngx_conf_merge_value(conf->bounce_status, prev->bounce_status, 444);
368 ngx_conf_merge_ptr_value(conf->extra_sigs, prev->extra_sigs, NULL);
369 ngx_conf_merge_ptr_value(conf->prefixes, prev->prefixes, NULL);
370 ngx_conf_merge_ptr_value(conf->methods, prev->methods, NULL);
371 if (conf->audit_log == NULL) {
372 conf->audit_log = prev->audit_log;
373 }
374
375 if (conf->bounce_status != 403 && conf->bounce_status != 444) {
376 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
377 "brix_guard_bounce_status must be 403 or 444, got %i",
378 conf->bounce_status);
379 return NGX_CONF_ERROR;
380 }
381
382 if (!conf->enable) {
383 return NGX_CONF_OK;
384 }
385 return ngx_http_brix_guard_build_ruleset(cf, conf);
386}

389/* ---- Register the ACCESS + LOG phase handlers ----
390 *
391 * WHAT: pushes ngx_http_brix_guard_access_handler onto the ACCESS phase
392 * and ngx_http_brix_guard_log_handler onto the LOG phase.
393 *
394 * WHY: ACCESS runs before proxy_pass content — the bounce keeps junk off the
395 * backend entirely; LOG runs after the response status is known — the only
396 * place outcome signals (404/401) exist.
397 *
398 * HOW: 1. ngx_array_push each handler onto the core main-conf phase arrays.
399 * 2. Handlers self-disable per location via lcf->enable (DECLINED/OK).
400 */
401static ngx_int_t
402ngx_http_brix_guard_postconf(ngx_conf_t *cf)
403{
404 ngx_http_core_main_conf_t *cmcf;
405 ngx_http_handler_pt *h;
406
407 cmcf = ngx_http_conf_get_module_main_conf(cf, ngx_http_core_module);
408
409 h = ngx_array_push(&cmcf->phases[NGX_HTTP_ACCESS_PHASE].handlers);
410 if (h == NULL) {
411 return NGX_ERROR;
412 }
413 *h = ngx_http_brix_guard_access_handler;
414
415 h = ngx_array_push(&cmcf->phases[NGX_HTTP_LOG_PHASE].handlers);
416 if (h == NULL) {
417 return NGX_ERROR;
418 }
419 *h = ngx_http_brix_guard_log_handler;
420
421 return NGX_OK;
422}
