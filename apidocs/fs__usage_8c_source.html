1/*
2 * fs_usage.c - filesystem usage arithmetic from statvfs(2).
3 *
4 * WHAT: Converts struct statvfs fields (f_blocks, f_bfree, f_bavail) into computed
5 * total/free/available/used bytes and occupancy ratio in ppm. Provides both a
6 * raw statvfs converter and a path-based convenience wrapper.
7 *
8 * WHY: Prometheus metrics report disk occupancy. WebDAV PROPFIND may include filesystem
9 * usage properties. Query responses need available-space information for clients.
10 * Single shared implementation avoids duplication across modules.
11 *
12 * HOW: Uses f_frsize (preferred) or f_bsize as block size multiplier. Computes total,
13 * free, available, used, and occupancy bytes. Occupancy ppm = occupancy_bytes /
14 * total_bytes × 1000000 using long double for precision.
15 */

19/*
20 * brix_fs_usage_from_statvfs - convert struct statvfs fields into computed usage bytes.
21 *
22 * WHAT: Takes a populated struct statvfs and computes total_bytes, free_bytes,
23 * available_bytes, used_bytes, occupancy_bytes, and occupancy_ppm into out.
24 *
25 * WHY: statvfs returns raw block counts; callers need byte-level values for metrics
26 * reporting and client responses. This function handles the arithmetic uniformly.
27 *
28 * HOW: Uses f_frsize (preferred) or f_bsize as block size. Computes:
29 * total = f_blocks × block_size, free = f_bfree × block_size,
30 * available = f_bavail × block_size, used = total - free,
31 * occupancy = total - available, occupancy_ppm = occupancy/total × 10^6.
32 */
33
34ngx_int_t
35brix_fs_usage_from_statvfs(const struct statvfs *vfs, brix_fs_usage_t *out)
36{
37 uint64_t block_size;
38
39 if (vfs == NULL || out == NULL || vfs->f_blocks == 0) {
40 return NGX_ERROR;
41 }
42
43 block_size = vfs->f_frsize ? (uint64_t) vfs->f_frsize
44 : (uint64_t) vfs->f_bsize;
45
46 out->total_bytes = (uint64_t) vfs->f_blocks * block_size;
47 out->free_bytes = (uint64_t) vfs->f_bfree * block_size;
48 out->available_bytes = (uint64_t) vfs->f_bavail * block_size;
49 out->used_bytes = out->total_bytes - out->free_bytes;
50 out->occupancy_bytes = out->total_bytes - out->available_bytes;
51 out->occupancy_ppm = (ngx_uint_t)
52 (((long double) out->occupancy_bytes * 1000000.0L)
53 / (long double) out->total_bytes);
54
55 return NGX_OK;
56}

58/*
59 * brix_fs_usage_stat - statvfs a path and convert to usage bytes.
60 *
61 * WHAT: Calls statvfs(path) on the given filesystem path, then delegates to
62 * brix_fs_usage_from_statvfs() to compute usage values into out.
63 *
64 * WHY: Convenience wrapper for callers that have a path string but not a populated
65 * struct statvfs. Used by metrics exporters and WebDAV property handlers.
66 *
67 * HOW: Validates path non-empty, calls statvfs(path, &vfs), then delegates to
68 * brix_fs_usage_from_statvfs(&vfs, out). Returns NGX_OK or NGX_ERROR.
69 */
70
71ngx_int_t
72brix_fs_usage_stat(const char *path, brix_fs_usage_t *out)
73{
74 struct statvfs vfs;
75
76 if (path == NULL || path[0] == '\0') {
77 return NGX_ERROR;
78 }
79
80 if (statvfs(path, &vfs) != 0) {
81 return NGX_ERROR;
82 }
83
84 return brix_fs_usage_from_statvfs(&vfs, out);
85}
