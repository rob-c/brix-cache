8/*
9 * tpc_curl_secure — enforce TLS verification and pin the SSRF-validated IP.
10 *
11 * Two defenses applied to every outbound curl handle (Phase 28 W2):
12 * (1) Explicit CURLOPT_SSL_VERIFYPEER/VERIFYHOST — never rely on curl's
13 * compile-time defaults, which a non-standard build could weaken.
14 * (2) Resolve the URL host once here (under SSRF policy) and pin the result
15 * via CURLOPT_RESOLVE. curl then connects only to that exact address
16 * rather than performing its own independent DNS lookup, closing the
17 * DNS-rebind TOCTOU window between the policy check and the connect.
18 * TLS SNI/cert validation still uses the original hostname, so pinning
19 * does not weaken certificate checking.
20 *
21 * Runs in the TPC thread pool, so the blocking getaddrinfo() inside
22 * check_dns_pin is safe here. Returns 0 with *resolve_out set (caller frees
23 * the slist after the transfer) on success; -1 when the host is prohibited or
24 * unresolvable — the transfer MUST abort.
25 */
26int
27tpc_curl_secure(CURL *curl, ngx_http_brix_webdav_loc_conf_t *conf,
28 const char *url, ngx_log_t *log, struct curl_slist **resolve_out)
29{
30 brix_net_target_t tgt;
31 brix_net_target_policy_t pol;
32 ngx_str_t url_str;
33 char pin_ip[128];
34 char err[256];
35 char entry[512]; /* "[host]:port:addr" — room for a
36 * bracketed IPv6/long host + addr */
37 uint16_t port;
38 struct curl_slist *rs;
39
40 *resolve_out = NULL;
41
42 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYPEER, 1L);
43 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYHOST, 2L);
44
45 url_str.data = (u_char *) url;
46 url_str.len = ngx_strlen(url);
47
48 ngx_memzero(&pol, sizeof(pol));
49 pol.require_https = 1;
50 pol.allow_local = conf->tpc_allow_local;
51 pol.allow_private = conf->tpc_allow_private;
52 pol.default_https_port = 443;
53
54 if (brix_net_target_parse(NULL, &url_str, &tgt, err, sizeof(err)) != NGX_OK
55 || brix_net_target_check_dns_pin(&tgt, &pol, pin_ip, sizeof(pin_ip),
56 err, sizeof(err)) != NGX_OK)
57 {
58 ngx_log_error(NGX_LOG_WARN, log, 0,
59 "brix_webdav: TPC egress check blocked \"%s\": %s",
60 url, err);
61 return -1;
62 }
63
64 port = tgt.has_port ? tgt.port : 443;
65 {
66 /* curl's CURLOPT_RESOLVE host field must bracket an IPv6 literal
67 * ("[::1]:443:addr") to match the bracketed host in an IPv6 URL;
68 * tgt.host arrives bare from net_target. The resolved address (pin_ip)
69 * stays bare — curl accepts a bare IPv6 in the address position. */
70 char hostz[256], hostb[288];
71 size_t hl = ngx_min(tgt.host.len, sizeof(hostz) - 1);
72 ngx_memcpy(hostz, tgt.host.data, hl);
73 hostz[hl] = '\0';
74 brix_format_host(hostz, hostb, sizeof(hostb));
75 snprintf(entry, sizeof(entry), "%s:%u:%s",
76 hostb, (unsigned) port, pin_ip);
77 }
78
79 rs = curl_slist_append(NULL, entry);
80 if (rs == NULL) {
81 return -1;
82 }
83 curl_easy_setopt(curl, CURLOPT_RESOLVE, rs);
84 *resolve_out = rs;
85 return 0;
86}

94 * The caller must free *hdrs_out via curl_slist_free_all() after cleanup.

97/*
98 * Phase 39 (WS4): bound a stalled / black-holed TPC remote. Always cap connect
99 * time (a connect to a black hole otherwise hangs the thread-pool worker forever)
100 * and enable TCP keepalive; optionally abort a transfer that stays below a
101 * bytes/sec floor (idle / low-speed) WITHOUT killing a slow-but-progressing one.
102 * Pure additive curl configuration — no data-copy cost; the low-speed bound is
103 * off unless the operator sets both knobs.
104 */
105void
106tpc_curl_apply_stall_bounds(CURL *curl, ngx_http_brix_webdav_loc_conf_t *conf)
107{
108 curl_easy_setopt(curl, CURLOPT_CONNECTTIMEOUT,
109 BRIX_TPC_CONNECT_TIMEOUT_SECS);
110 curl_easy_setopt(curl, CURLOPT_TCP_KEEPALIVE, 1L);
111 if (conf->tpc_low_speed_bytes > 0 && conf->tpc_low_speed_secs > 0) {
112 curl_easy_setopt(curl, CURLOPT_LOW_SPEED_LIMIT,
113 (long) conf->tpc_low_speed_bytes);
114 curl_easy_setopt(curl, CURLOPT_LOW_SPEED_TIME,
115 (long) conf->tpc_low_speed_secs);
116 }
117}

120/*
121 * tpc_curl_apply_client_cred — install the pull leg's client cert + key on the
122 * handle, preferring the requesting user's delegated proxy over the static
123 * service cert.
124 *
125 * WHAT: when user_cert/user_key are non-NULL (a per-user delegated proxy the
126 * resolver materialised or found in the store), point CURLOPT_SSLCERT /
127 * CURLOPT_SSLKEY at THEM so the source authenticates the end user;
128 * otherwise fall back to conf->tpc_cert/_key (the service identity) exactly
129 * as before, for non-delegated setups.
130 * WHY: the whole gap this closes — a TPC PULL must present the user's own x509
131 * to the source, not the destination's service cert (CN=localhost). The
132 * bearer path is untouched (it is an Authorization header, applied
133 * elsewhere).
134 * HOW: user_cert wins the SSLCERT slot; user_key (which equals user_cert for a
135 * combined-PEM proxy) wins the SSLKEY slot; each falls back independently
136 * to the matching conf field only when its override is NULL.
137 */
138static void
139tpc_curl_apply_client_cred(CURL *curl,
140 ngx_http_brix_webdav_loc_conf_t *conf,
141 const char *user_cert, const char *user_key)
142{
143 if (user_cert != NULL) {
144 curl_easy_setopt(curl, CURLOPT_SSLCERT, user_cert);
145 } else if (conf->tpc_cert.len > 0) {
146 curl_easy_setopt(curl, CURLOPT_SSLCERT,
147 (const char *) conf->tpc_cert.data);
148 }
149 if (user_key != NULL) {
150 curl_easy_setopt(curl, CURLOPT_SSLKEY, user_key);
151 } else if (conf->tpc_key.len > 0) {
152 curl_easy_setopt(curl, CURLOPT_SSLKEY,
153 (const char *) conf->tpc_key.data);
154 }
155}

158int
159tpc_curl_apply_conf(CURL *curl,
160 ngx_http_brix_webdav_loc_conf_t *conf,
161 const char *url, ngx_array_t *transfer_headers, ngx_log_t *log,
162 const char *user_cert, const char *user_key,
163 struct curl_slist **hdrs_out, struct curl_slist **resolve_out)
164{
165 ngx_uint_t i;
166 ngx_str_t *headers;
167 struct curl_slist *hdrs = NULL;
168
169 *resolve_out = NULL;
170
171 curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
172 curl_easy_setopt(curl, CURLOPT_FAILONERROR, 1L);
173 curl_easy_setopt(curl, CURLOPT_URL, url);
174
175#ifdef CURLOPT_PROTOCOLS_STR
176 curl_easy_setopt(curl, CURLOPT_PROTOCOLS_STR, "https");
177#else
178 curl_easy_setopt(curl, CURLOPT_PROTOCOLS, (long) CURLPROTO_HTTPS);
179#endif
180
181 /* TLS verification + DNS-rebind pin (W2). */
182 if (tpc_curl_secure(curl, conf, url, log, resolve_out) < 0) {
183 return -1;
184 }
185
186 if (conf->tpc_timeout > 0)
187 curl_easy_setopt(curl, CURLOPT_TIMEOUT, (long) conf->tpc_timeout);
188 tpc_curl_apply_stall_bounds(curl, conf); /* Phase 39 (WS4) */
189 tpc_curl_apply_client_cred(curl, conf, user_cert, user_key);
190 if (conf->tpc_cafile.len > 0)
191 curl_easy_setopt(curl, CURLOPT_CAINFO,
192 (const char *) conf->tpc_cafile.data);
193 if (conf->tpc_cadir.len > 0)
194 curl_easy_setopt(curl, CURLOPT_CAPATH,
195 (const char *) conf->tpc_cadir.data);
196
197 if (transfer_headers != NULL && transfer_headers->nelts > 0) {
198 headers = transfer_headers->elts;
199 for (i = 0; i < transfer_headers->nelts; i++) {
200 struct curl_slist *next = curl_slist_append(
201 hdrs, (const char *) headers[i].data);
202 if (next == NULL) {
203 if (hdrs) curl_slist_free_all(hdrs);
204 *hdrs_out = NULL;
205 return -1;
206 }
207 hdrs = next;
208 }
209 }
210
211 *hdrs_out = hdrs;
212 if (hdrs != NULL) {
213 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, hdrs);
214 }
215 return 0;
216}

219/*
220 * Issue a HEAD request to retrieve Content-Length.
221 * Returns the size in bytes, or -1 if unknown / request failed.
222 */
223off_t
224tpc_curl_head_size(ngx_log_t *log,
225 ngx_http_brix_webdav_loc_conf_t *conf,
226 const char *url, ngx_array_t *transfer_headers,
227 const char *user_cert, const char *user_key)
228{
229 CURL *curl;
230 struct curl_slist *hdrs = NULL;
231 struct curl_slist *resolve = NULL;
232 CURLcode res;
233 off_t content_length = -1;
234
235 curl = curl_easy_init();
236 if (curl == NULL) {
237 ngx_log_error(NGX_LOG_WARN, log, 0,
238 "brix_webdav: curl_easy_init() failed for HEAD");
239 return -1;
240 }
241
242 if (tpc_curl_apply_conf(curl, conf, url, transfer_headers, log,
243 user_cert, user_key, &hdrs, &resolve) < 0) {
244 if (resolve) curl_slist_free_all(resolve);
245 curl_easy_cleanup(curl);
246 return -1;
247 }
248 curl_easy_setopt(curl, CURLOPT_NOBODY, 1L);
249
250 res = curl_easy_perform(curl);
251 if (res == CURLE_OK) {
252#ifdef CURLINFO_CONTENT_LENGTH_DOWNLOAD_T
253 curl_off_t cl;
254 curl_easy_getinfo(curl, CURLINFO_CONTENT_LENGTH_DOWNLOAD_T, &cl);
255 content_length = (cl >= 0) ? (off_t) cl : -1;
256#else
257 double cl;
258 curl_easy_getinfo(curl, CURLINFO_CONTENT_LENGTH_DOWNLOAD, &cl);
259 content_length = (cl >= 1.0) ? (off_t) cl : -1;
260#endif
261 } else {
262 ngx_log_error(NGX_LOG_WARN, log, 0,
263 "brix_webdav: HEAD for multi-stream size failed: %s",
264 curl_easy_strerror(res));
265 }
266
267 if (hdrs) curl_slist_free_all(hdrs);
268 if (resolve) curl_slist_free_all(resolve);
269 curl_easy_cleanup(curl);
270 return content_length;
271}
