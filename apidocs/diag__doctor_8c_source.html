10/* Stream a remote file through the live connection into fd; returns 0 / -1 and
11 * sets *out_bytes to the number of bytes written. Reuses the authenticated conn
12 * (no reconnect), so it measures the data path itself. */
13int
14download_to_fd(brix_conn *c, const char *path, int fd, int64_t *out_bytes,
15 brix_status *st)
16{
17 brix_file f;
18 int64_t off = 0;
19 char *buf;
20
21 if (brix_file_open_read(c, path, &f, st) != 0) {
22 return -1;
23 }
24 buf = (char *) malloc(1u << 20);
25 if (buf == NULL) {
26 brix_file_close(c, &f, st);
27 brix_status_set(st, XRDC_EPROTO, 0, "download: out of memory");
28 return -1;
29 }
30 for (;;) {
31 ssize_t r = brix_file_read(c, &f, off, buf, 1u << 20, st);
32 ssize_t w = 0;
33 if (r < 0) {
34 free(buf);
35 brix_file_close(c, &f, st);
36 return -1;
37 }
38 if (r == 0) {
39 break;
40 }
41 while (w < r) {
42 ssize_t k = write(fd, buf + w, (size_t) (r - w));
43 if (k < 0) {
44 free(buf);
45 brix_file_close(c, &f, st);
46 brix_status_set(st, XRDC_ESOCK, 0, "download: local write failed");
47 return -1;
48 }
49 w += k;
50 }
51 off += r;
52 }
53 free(buf);
54 if (out_bytes != NULL) {
55 *out_bytes = off;
56 }
57 return brix_file_close(c, &f, st);
58}

89/* Throughput probe over an established conn: TTFB (first read) + MB/s (whole file). */
90int
91doctor_xfer(brix_conn *c, const char *path, double *ttfb_ms, double *mbps,
92 int64_t *bytes)
93{
94 brix_file f;
95 brix_status st;
96 uint8_t *buf;
97 int64_t off = 0;
98 uint64_t t0, tf = 0, t1;
99
100 brix_status_clear(&st);
101 if (brix_file_open_read(c, path, &f, &st) != 0) {
102 return -1;
103 }
104 buf = (uint8_t *) malloc(1u << 20);
105 if (buf == NULL) {
106 brix_file_close(c, &f, &st);
107 return -1;
108 }
109 t0 = brix_mono_ns();
110 for (;;) {
111 ssize_t r = brix_file_read(c, &f, off, buf, 1u << 20, &st);
112 if (r < 0) { free(buf); brix_file_close(c, &f, &st); return -1; }
113 if (tf == 0) { tf = brix_mono_ns(); } /* time-to-first-byte */
114 if (r == 0) { break; }
115 off += r;
116 }
117 t1 = brix_mono_ns();
118 free(buf);
119 brix_file_close(c, &f, &st);
120 *ttfb_ms = (double) (tf - t0) / 1e6;
121 *bytes = off;
122 {
123 double secs = (double) (t1 - t0) / 1e9;
124 if (secs <= 0.0) { secs = 1e-9; }
125 *mbps = (double) off / 1e6 / secs;
126 }
127 return 0;
128}

131/* /metrics signal: reachable? + any kXR_wait/budget shedding gauge nonzero. */
132void
133doctor_metrics(const char *host, int port, doctor_ep *e)
134{
135 char *body;
136 brix_status st;
137 int http = 0;
138 char *line, *save;
139
140 body = (char *) malloc(1u << 20);
141 if (body == NULL) {
142 return;
143 }
144 brix_status_clear(&st);
145 if (brix_http_get(host, port, "/metrics", 4000, &http, body, 1u << 20, NULL,
146 &st) != 0) {
147 free(body);
148 return;
149 }
150 e->metrics_http = http;
151 for (line = strtok_r(body, "\n", &save); line != NULL;
152 line = strtok_r(NULL, "\n", &save)) {
153 if (line[0] == '#') {
154 continue;
155 }
156 if (strstr(line, "kXR_wait") != NULL || strstr(line, "_wait_") != NULL
157 || strstr(line, "budget") != NULL || strstr(line, "shed") != NULL) {
158 /* nonzero counter at the end of the line ⇒ active shedding */
159 char *sp = strrchr(line, ' ');
160 if (sp != NULL && strtod(sp + 1, NULL) > 0.0) {
161 e->shedding = 1;
162 }
163 }
164 }
165 free(body);
166}

169/*
170 * The full auth/permissions suite. Opens its own scoped connections (the credential
171 * matrix) — does not reuse the primary. Read-only assertions always run; the
172 * write-scope assertion is gated like the write probe. PII-free: only verdicts +
173 * kXR codes are recorded, never token/cert/scope contents.
174 */
175void
176doctor_auth_suite(const diag_args *a, const brix_url *u, const char *target,
177 int have_target, doctor_ep *e)
178{
179 char sec_list[256];
180 int ztn_adv;
181 char *tok;
182
183 /* 1) anonymous access must be denied on an auth-required server — this also
184 * discovers the server's advertised auth (&P=) from a force_anon session. */
185 dx_authz_anon(a, u, target, have_target, sec_list, sizeof(sec_list), e);
186 ztn_adv = (strstr(sec_list, "ztn") != NULL);
187
188 /* 2,3) forged-credential rejection — only if the server takes bearer tokens. */
189 if (ztn_adv) {
190 char fsig[1024], fnone[1024];
191 /* No kid: a kid the server doesn't know would be rejected at key SELECTION,
192 * short-circuiting the signature check — we want the server to reach
193 * signature verification (and reject the garbage sig) so the test actually
194 * exercises signature verification on typical single-key deployments. */
195 if (dx_make_jwt(
196 "{\"alg\":\"RS256\",\"typ\":\"JWT\"}",
197 "{\"iss\":\"https://xrddiag.invalid\",\"sub\":\"xrddiag-probe\","
198 "\"scope\":\"storage.read:/ storage.modify:/\",\"exp\":4102444800}",
199 "ZHVtbXktc2lnbmF0dXJlLW5vdC12YWxpZA", fsig, sizeof(fsig)) == 0) {
200 dx_authz_forged(a, u, "authz-forgesig", fsig, e);
201 }
202 if (dx_make_jwt(
203 "{\"alg\":\"none\",\"typ\":\"JWT\"}",
204 "{\"iss\":\"https://xrddiag.invalid\",\"sub\":\"xrddiag-probe\","
205 "\"scope\":\"storage.read:/ storage.modify:/\",\"exp\":4102444800}",
206 "", fnone, sizeof(fnone)) == 0) {
207 dx_authz_forged(a, u, "authz-algnone", fnone, e);
208 }
209 } else {
210 dx_record(e, "authz-token", DX_OK, 0,
211 "server does not offer bearer-token auth (forged-token tests N/A)", "");
212 }
213
214 /* 4,5) tests that require the operator's real token. */
215 tok = brix_token_discover();
216 if (tok != NULL) {
217 brix_token_meta m;
218 brix_token_meta_get(tok, &m);
219 if (ztn_adv && m.valid && m.expired) {
220 dx_authz_expired(a, u, tok, e);
221 }
222 if (a->allow_write && ztn_adv && m.valid && m.has_scope
223 && m.has_read && !m.has_write
224 && (dx_is_loopback(e->host) || a->authorized)) {
225 dx_authz_scope(a, u, tok, e);
226 }
227 free(tok);
228 }
229}

232/*
233 * Run the active-diagnosis battery over an already-open connection. Read-only probes
234 * always run; write/stage probes run only when --allow-write is set AND the target is
235 * loopback or the operator passed --i-am-authorized (mutations on a remote server).
236 */
237void
238doctor_diagnose(const diag_args *a, brix_conn *c, const brix_url *u,
239 const char *target, int have_target, doctor_ep *e)
240{
241 dx_probe_auth(c, e);
242 dx_probe_namespace(c, e);
243 if (have_target) {
244 dx_probe_read(c, target, e);
245 dx_probe_checksum(c, target, e);
246 }
247 {
248 char loc[2048];
249 brix_status lst;
250 brix_status_clear(&lst);
251 if (brix_locate(c, u->path[0] ? u->path : "/", loc, sizeof(loc), &lst) != 0) {
252 dx_record_status(e, "locate", &lst);
253 } else {
254 char *t, *save;
255 for (t = strtok_r(loc, " \t\r\n", &save); t != NULL;
256 t = strtok_r(NULL, " \t\r\n", &save)) {
257 if (t[0] == 'S') { e->holders++; }
258 }
259 if (e->holders == 0) {
260 dx_record(e, "locate", DX_WARN, 0, "no replica located for the path",
261 "check data-server health and the CMS/manager registry");
262 } else {
263 dx_record(e, "locate", DX_OK, 0, "replica(s) located", "");
264 }
265 }
266 }
267 if (a->allow_write) {
268 int permitted = dx_is_loopback(e->host) || a->authorized;
269 if (!permitted) {
270 dx_record(e, "write", DX_WARN, 0,
271 "write probe skipped on a non-loopback host without --i-am-authorized",
272 "re-run with --i-am-authorized to actively probe the write path");
273 } else {
274 dx_probe_write(c, e);
275 if (have_target && e->offline_seen) {
276 dx_probe_stage(c, target, e);
277 }
278 }
279 }
280 if (a->auth_suite) {
281 doctor_auth_suite(a, u, target, have_target, e);
282 }
283}

286/*
287 * http / https deep-dive: connect (+TLS cert/cipher), HEAD/GET the path, classify
288 * the HTTP status, byte-range support, the Digest (checksum) header, and 401 auth
289 * posture. For davs, also OPTIONS (WebDAV class) and PROPFIND (listing). Every probe
290 * is bounded by the per-probe timeout. PII-free: only statuses/header-names/sizes.
291 */
292void
293doctor_http(const diag_args *a, dx_proto proto, int tls, const char *host,
294 int port, const char *path, doctor_ep *e)
295{
296 brix_http_resp r;
297 brix_status st;
298 int verify = a->verify_tls;
299 int tmo = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
300 char val[512];
301
302 memset(e, 0, sizeof(*e));
303 e->proto = proto;
304 e->status = DOC_GREEN;
305 snprintf(e->host, sizeof(e->host), "%s", host);
306 e->port = port;
307
308 /* Stage 1: reachability + (TLS handshake/cert). Try HEAD; fall back to a 1-byte
309 * ranged GET if HEAD is refused, so we still measure connect/TLS. */
310 brix_status_clear(&st);
311 if (brix_http_req(host, port, tls, "HEAD", path, NULL, NULL, 0, tmo, verify,
312 NULL, &r, &st) != 0) {
313 brix_status_clear(&st);
314 if (brix_http_req(host, port, tls, "GET", path, "Range: bytes=0-0\r\n",
315 NULL, 0, tmo, verify, NULL, &r, &st) != 0) {
316 dx_http_fail(e, tls, &st);
317 return;
318 }
319 }
320 e->connected = 1;
321 if (tls) {
322 e->tls_active = 1;
323 snprintf(e->tls_ver, sizeof(e->tls_ver), "%s", r.tls_ver);
324 snprintf(e->tls_cipher, sizeof(e->tls_cipher), "%s", r.tls_cipher);
325 dx_record(e, "tls", DX_OK, 0, "TLS handshake completed + certificate accepted", "");
326 }
327
328 /* Stage 2: HTTP status. */
329 dx_http_status(e, "http", r.status);
330
331 /* Stage 3: byte ranges (partial reads / multi-stream transfers depend on this). */
332 if (brix_http_header(&r, "Accept-Ranges", val, sizeof(val))
333 && strstr(val, "bytes") != NULL) {
334 dx_record(e, "ranges", DX_OK, 0, "byte-range reads supported (Accept-Ranges: bytes)", "");
335 } else {
336 dx_record(e, "ranges", DX_WARN, 0,
337 "server did not advertise Accept-Ranges (partial/parallel reads may not work)",
338 "enable range support if clients use partial reads");
339 }
340
341 /* Stage 4: checksum advertisement (RFC-3230 Digest, WLCG transfers rely on it). */
342 if (brix_http_header(&r, "Digest", val, sizeof(val)) && strchr(val, '=') != NULL) {
343 /* RFC-3230 form is "algo=value"; require the '=' so a malformed header
344 * isn't counted as a working checksum. */
345 dx_record(e, "checksum", DX_OK, 0, "server advertises a content Digest (checksum)", "");
346 } else {
347 dx_record(e, "checksum", DX_WARN, 0,
348 "no Digest header (checksum verification unavailable over HTTP)",
349 "enable Want-Digest/Digest if integrity checks are required");
350 }
351
352 /* Stage 5: content-length present (sized transfers). */
353 if (brix_http_header(&r, "Content-Length", val, sizeof(val))) {
354 dx_record(e, "content-length", DX_OK, 0, "response is sized (Content-Length present)", "");
355 }
356 brix_http_resp_free(&r);
357
358 /* davs extras: OPTIONS (WebDAV class) + PROPFIND (collection listing). */
359 if (proto == DXP_DAVS) {
360 brix_status_clear(&st);
361 if (brix_http_req(host, port, tls, "OPTIONS", path, NULL, NULL, 0, tmo,
362 verify, NULL, &r, &st) == 0) {
363 if (brix_http_header(&r, "DAV", val, sizeof(val))) {
364 int class2 = (strstr(val, "2") != NULL);
365 dx_record(e, "davs-class", DX_OK, r.status,
366 class2 ? "WebDAV class 2 advertised (LOCK supported)"
367 : "WebDAV advertised (DAV header present)", "");
368 } else {
369 dx_record(e, "davs-class", DX_WARN, r.status,
370 "OPTIONS returned no DAV header (WebDAV may be disabled)",
371 "confirm brix_webdav is on for this location");
372 }
373 if (brix_http_header(&r, "Allow", val, sizeof(val))
374 && strstr(val, "COPY") != NULL) {
375 dx_record(e, "davs-tpc", DX_OK, 0,
376 "COPY method allowed (third-party-copy capable)", "");
377 }
378 brix_http_resp_free(&r);
379 } else {
380 dx_record(e, "davs-class", DX_WARN, st.kxr,
381 "OPTIONS request failed", "");
382 }
383 brix_status_clear(&st);
384 if (brix_http_req(host, port, tls, "PROPFIND", path, "Depth: 1\r\n",
385 NULL, 0, tmo, verify, NULL, &r, &st) == 0) {
386 if (r.status == 207) {
387 dx_record(e, "davs-listing", DX_OK, 207,
388 "PROPFIND multistatus listing works", "");
389 } else if (r.status == 401 || r.status == 403) {
390 dx_record(e, "davs-listing", DX_WARN, r.status,
391 "PROPFIND requires authentication", "provide a credential");
392 } else {
393 dx_record(e, "davs-listing", DX_WARN, r.status,
394 "PROPFIND did not return 207 multistatus", "");
395 }
396 brix_http_resp_free(&r);
397 }
398 }
399}

402/*
403 * s3 deep-dive: connect (+TLS), an UNAUTHENTICATED GET to confirm reachability +
404 * that the server enforces auth (403/AccessDenied) vs is public (200) vs missing
405 * (404/NoSuchBucket). If AWS_ACCESS_KEY_ID/SECRET are in the environment, also send
406 * a SigV4-signed HEAD and confirm the signature is accepted (catches signer/clock/
407 * region faults). PII-free: never emits the key or signature.
408 */
409void
410doctor_s3(const diag_args *a, int tls, const char *host, int port,
411 const char *uri, doctor_ep *e)
412{
413 brix_http_resp r;
414 brix_status st;
415 int verify = a->verify_tls;
416 int tmo = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
417 const char *ak = getenv("AWS_ACCESS_KEY_ID");
418 const char *sk = getenv("AWS_SECRET_ACCESS_KEY");
419
420 memset(e, 0, sizeof(*e));
421 e->proto = DXP_S3;
422 e->status = DOC_GREEN;
423 snprintf(e->host, sizeof(e->host), "%s", host);
424 e->port = port;
425
426 /* Stage 1: reachability + TLS via an unauthenticated GET. */
427 brix_status_clear(&st);
428 if (brix_http_req(host, port, tls, "GET", uri, NULL, NULL, 0, tmo, verify,
429 NULL, &r, &st) != 0) {
430 dx_http_fail(e, tls, &st);
431 return;
432 }
433 e->connected = 1;
434 if (tls) {
435 e->tls_active = 1;
436 snprintf(e->tls_ver, sizeof(e->tls_ver), "%s", r.tls_ver);
437 snprintf(e->tls_cipher, sizeof(e->tls_cipher), "%s", r.tls_cipher);
438 dx_record(e, "tls", DX_OK, 0, "TLS handshake completed + certificate accepted", "");
439 }
440 if (r.status == 403) {
441 dx_record(e, "s3-auth", DX_OK, 403,
442 "endpoint enforces S3 authentication (anonymous request denied)", "");
443 } else if (r.status == 404) {
444 dx_record(e, "s3-bucket", DX_WARN, 404,
445 "bucket/key not found (NoSuchBucket/NoSuchKey)",
446 "verify the bucket and key path");
447 } else if (r.status >= 200 && r.status < 300) {
448 dx_record(e, "s3-auth", DX_WARN, r.status,
449 "anonymous S3 request SUCCEEDED — the resource is public",
450 "confirm public access is intended; otherwise restrict it");
451 } else {
452 dx_http_status(e, "s3-req", r.status);
453 }
454 brix_http_resp_free(&r);
455
456 /* Stage 2: authenticated SigV4 probe (only if AWS creds are present). */
457 if (ak != NULL && sk != NULL && ak[0] != '\0' && sk[0] != '\0') {
458 char hdrs[1024];
459 char hostport[300];
460 const char *region = getenv("AWS_DEFAULT_REGION");
461 if (region == NULL || region[0] == '\0') { region = "us-east-1"; }
462 /* Sign the exact Host header we send (host:port) — the server canonicalises
463 * the Host value verbatim, so signing the bare host would mismatch. */
464 snprintf(hostport, sizeof(hostport), "%s:%d", host, port);
465 if (s3_sign("GET", hostport, uri, ak, sk, region, hdrs, sizeof(hdrs)) != 0) {
466 dx_record(e, "s3-sigv4", DX_WARN, 0, "could not build a SigV4 signature (client)", "");
467 } else {
468 brix_status_clear(&st);
469 if (brix_http_req(host, port, tls, "GET", uri, hdrs, NULL, 0, tmo,
470 verify, NULL, &r, &st) != 0) {
471 dx_record(e, "s3-sigv4", DX_WARN, st.kxr, "signed request failed to complete", "");
472 } else {
473 if (r.status >= 200 && r.status < 300) {
474 dx_record(e, "s3-sigv4", DX_OK, r.status,
475 "SigV4-signed request accepted (signature/clock/region OK)", "");
476 } else if (r.status == 403) {
477 /* read the S3 <Code> element (not a body-wide substring, which
478 * could false-match) to tell a signature fault from a policy deny. */
479 const char *cs = r.body ? strstr(r.body, "<Code>") : NULL;
480 int sig_fault = (cs != NULL
481 && strncmp(cs + 6, "SignatureDoesNotMatch", 21) == 0);
482 if (sig_fault) {
483 dx_record(e, "s3-sigv4", DX_FAIL, 403,
484 "SigV4 signature rejected (SignatureDoesNotMatch — clock skew / region / key mismatch)",
485 "check client clock vs server, the region, and the access key/secret");
486 } else {
487 dx_record(e, "s3-sigv4", DX_WARN, 403,
488 "signed request denied (access policy, not a signature fault)",
489 "check the bucket/object policy for this identity");
490 }
491 } else {
492 dx_http_status(e, "s3-sigv4", r.status);
493 }
494 brix_http_resp_free(&r);
495 }
496 }
497 } else {
498 dx_record(e, "s3-sigv4", DX_OK, 0,
499 "no AWS credentials in environment — signed-request check skipped (posture only)",
500 "set AWS_ACCESS_KEY_ID/AWS_SECRET_ACCESS_KEY to test SigV4 acceptance");
501 }
502}

505/*
506 * cms deep-dive: a cluster manager (cmsd/redirector) is a root:// endpoint that
507 * answers kXR_locate with the data server(s) holding a path and issues kXR_redirect.
508 * Connect to the manager, locate the path, and confirm the redirect resolution to a
509 * reachable data server; flag no-holder / unreachable-DS (ghost) / redirect issues.
510 * Reuses the libbrix locate + reconnect machinery (the redirect loop-guard applies).
511 */
512void
513doctor_cms(const diag_args *a, const char *host, int port, const char *path,
514 doctor_ep *e)
515{
516 brix_url u;
517 brix_conn c;
518 brix_status st;
519
520 memset(e, 0, sizeof(*e));
521 e->proto = DXP_CMS;
522 e->status = DOC_GREEN;
523 snprintf(e->host, sizeof(e->host), "%s", host);
524 e->port = port;
525
526 memset(&u, 0, sizeof(u));
527 u.scheme = XRDC_SCHEME_ROOT;
528 snprintf(u.host, sizeof(u.host), "%s", host);
529 u.port = port;
530 snprintf(u.path, sizeof(u.path), "%s", path[0] ? path : "/");
531
532 brix_status_clear(&st);
533 if (brix_connect(&c, &u, &a->conn, &st) != 0) {
534 dx_record(e, "cms-connect", DX_FAIL, st.kxr,
535 "could not connect to the cluster manager",
536 "check the manager (cmsd/redirector) is up on this host:port");
537 doc_issue(e, DOC_RED, "manager connect failed");
538 return;
539 }
540 e->connected = 1;
541 c.io.timeout_ms = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
542 dx_record(e, "cms-connect", DX_OK, 0, "manager reachable + login completed", "");
543
544 {
545 char loc[2048];
546 brix_status lst;
547 brix_status_clear(&lst);
548 if (brix_locate(&c, u.path, loc, sizeof(loc), &lst) != 0) {
549 /* a manager returns NotFound when no server holds the path. */
550 dx_record_status(e, "cms-locate", &lst);
551 } else {
552 char *t, *save;
553 for (t = strtok_r(loc, " \t\r\n", &save); t != NULL;
554 t = strtok_r(NULL, " \t\r\n", &save)) {
555 if (t[0] == 'S') { e->holders++; }
556 }
557 if (e->holders == 0) {
558 dx_record(e, "cms-locate", DX_FAIL, 0,
559 "manager located no data server for the path (no holder)",
560 "check data-server registration and the CMS registry");
561 } else {
562 dx_record(e, "cms-locate", DX_OK, 0,
563 "manager resolved the path to data server(s)", "");
564 }
565 }
566 }
567
568 /* Resolution: a stat through the manager must follow the redirect to a live DS.
569 * A redirect loop or dead DS surfaces here (the loop-guard returns an error). */
570 {
571 brix_statinfo si;
572 brix_status rst;
573 brix_status_clear(&rst);
574 if (brix_stat(&c, u.path, &si, &rst) == 0) {
575 dx_record(e, "cms-redirect", DX_OK, 0,
576 "manager→data-server redirect resolved to a live server", "");
577 } else if (rst.kxr == kXR_NotFound) {
578 dx_record(e, "cms-redirect", DX_WARN, rst.kxr,
579 "path not found via the manager (redirect resolved, file absent)",

583 "manager redirect did not resolve to a reachable data server (dead DS / redirect loop)",
584 "check data-server health and the CMS registry for stale entries");

591/* Interrogate ONE endpoint into *e. Bounded by the conn timeout (never hangs). */
592void
593doctor_one(const diag_args *a, const char *url, doctor_ep *e)
594{
595 brix_url u;
596 brix_conn c;
597 brix_status st;
598 const char *ver = NULL, *cipher = NULL;
599 char target[XRDC_PATH_MAX];
600 int have_target = 0;
601
602 target[0] = '\0';
603 memset(e, 0, sizeof(*e));
604 e->status = DOC_GREEN;
605 brix_status_clear(&st);
606 if (brix_endpoint_parse(url, &u, &st) != 0) {
607 snprintf(e->host, sizeof(e->host), "%s", url);
608 doc_issue(e, DOC_RED, "unparseable URL (bad scheme/host/port)");
609 return;
610 }
611 snprintf(e->host, sizeof(e->host), "%s", u.host);
612 e->port = u.port;
613
614 if (brix_connect(&c, &u, &a->conn, &st) != 0) {
615 /* Distinguish "server reachable but auth failed" from "couldn't reach it"
616 * by the client's error CODE: XRDC_EAUTH / kXR_NotAuthorized / kXR_AuthFailed
617 * mean auth was attempted and rejected; everything else is transport. */
618 if (st.kxr == XRDC_EAUTH || st.kxr == kXR_NotAuthorized
619 || st.kxr == kXR_AuthFailed) {
620 doc_issue(e, DOC_RED, "authentication failed");
621 dx_record(e, "auth", DX_FAIL, st.kxr,
622 "could not authenticate (credential rejected, or none usable for the server's auth)",
623 "check the credential's validity/scope and that it matches the server's auth mode");
624 } else {
625 /* reachability: classify *why* the connection could not be set up. */
626 const char *cause = "connection setup failed";
627 const char *remedy = "check the network path and that the server is running";
628 if (st.sys_errno == ECONNREFUSED) {
629 cause = "no listener on host:port (service down or wrong port)";
630 remedy = "start the gateway / verify the port and any firewall";
631 } else if (st.sys_errno == ETIMEDOUT || st.sys_errno == EHOSTUNREACH
632 || st.sys_errno == ENETUNREACH) {
633 cause = "host/network unreachable (routing or firewall drop)";
634 remedy = "check routing/firewall and that the host is up";
635 } else if (st.msg[0] != '\0' && strstr(st.msg, "resolve") != NULL) {
636 cause = "DNS resolution failed";
637 remedy = "check the hostname and DNS resolver";
638 }
639 /* use the classified cause, not st.msg — wire text may carry PII. */
640 doc_issue(e, DOC_RED, "connect failed: %s", cause);
641 dx_record(e, "reachability", DX_FAIL, st.kxr, cause, remedy);
642 }
643 /* the auth-suite is self-contained (its own force_anon session) — run it
644 * even when our credential could not establish the primary connection. */
645 if (a->auth_suite) {
646 doctor_auth_suite(a, &u, target, 0, e);
647 }
648 return;
649 }
650 e->connected = 1;
651 c.io.timeout_ms = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
652
653 /* network + transport facts */
654 brix_netdiag_facts(&c, &e->nf);
655 e->caps = (unsigned) c.server_flags;
656 e->gototls = (c.server_flags & kXR_gotoTLS) != 0;
657 e->tls_active = brix_tls_info(&c, &ver, &cipher);
658 if (e->tls_active) {
659 snprintf(e->tls_ver, sizeof(e->tls_ver), "%s", ver ? ver : "?");
660 snprintf(e->tls_cipher, sizeof(e->tls_cipher), "%s", cipher ? cipher : "?");
661 }
662 snprintf(e->auth, sizeof(e->auth), "%s",
663 c.diag.chosen_auth ? c.diag.chosen_auth : "anon");
664
665 /* no-silent-downgrade: gotoTLS advertised but the session is cleartext */
666 if (e->gototls && !e->tls_active) {
667 doc_issue(e, DOC_RED, "gotoTLS advertised but session is cleartext");
668 }
669
670 /* throughput probe over a resolved file (skip cleanly if the export is empty) */
671 {
672 brix_statinfo sti;
673 brix_status rst;
674 brix_status_clear(&rst);
675 if (resolve_target(&c, &u, target, sizeof(target), &sti, &rst) == 0) {
676 have_target = 1;
677 if (doctor_xfer(&c, target, &e->ttfb_ms, &e->mbps, &e->xfer_bytes)
678 == 0) {
679 e->have_xfer = 1;
680 }
681 }
682 }
683
684 /* active differential diagnosis — exercise subsystems + classify (incl. locate). */
685 doctor_diagnose(a, &c, &u, target, have_target, e);
686
687 brix_close(&c);
688
689 /* server-side load signal (cleartext /metrics; best-effort, 0 = skip) */
690 if (a->metrics_port > 0) {
691 doctor_metrics(e->host, a->metrics_port, e);
692 }
693 if (e->shedding) {
694 doc_issue(e, DOC_YELLOW, "server reports kXR_wait / budget shedding");
695 }
696 /* cwnd/BDP signal — only meaningful once enough bytes moved to time it. */
697 if (e->have_xfer && e->xfer_bytes >= (4 << 20) && e->nf.have_tcpinfo
698 && e->nf.rtt_us > 0 && e->nf.rtt_us < 5000 && e->mbps < 5.0) {
699 doc_issue(e, DOC_YELLOW, "low throughput (%.1f MB/s) at low RTT — cwnd/BDP?",
700 e->mbps);
701 }
702 if (e->nf.have_tcpinfo && e->nf.retrans > 0) {
703 doc_issue(e, DOC_YELLOW, "%u TCP retransmit(s)", e->nf.retrans);
704 }
705}

715/* Cross-endpoint diff engine over the transfer path. Returns #critical diffs. */
716int
717doctor_cross(const doctor_ep *eps, int n, FILE *out)
718{
719 int i, crit = 0, connected = 0;
720 if (n < 2) {
721 return 0;
722 }
723 for (i = 0; i < n; i++) {
724 if (eps[i].connected) { connected++; }
725 }
726 if (connected < 2) {
727 return 0; /* fewer than two reachable hops — nothing to compare */
728 }
729 fprintf(out, "Path analysis (%d hops):\n", n);
730 for (i = 1; i < n; i++) {
731 const doctor_ep *p = &eps[i - 1], *q = &eps[i];
732 if (!p->connected || !q->connected) {
733 continue;
734 }
735 /* TLS-downgrade: a TLS hop followed by a cleartext one */
736 if (p->tls_active && !q->tls_active) {
737 fprintf(out, " %s:%d -> %s:%d TLS DOWNGRADE (encrypted then cleartext)\n",
738 p->host, p->port, q->host, q->port);
739 crit++;
740 }
741 /* auth-fallback: the chosen auth weakens across the hop */
742 if (strcmp(p->auth, q->auth) != 0) {
743 fprintf(out, " %s:%d -> %s:%d auth changed %s -> %s\n",
744 p->host, p->port, q->host, q->port, p->auth, q->auth);
745 }
746 /* v4/v6 asymmetry */
747 if (p->nf.family && q->nf.family && p->nf.family != q->nf.family) {
748 fprintf(out, " %s:%d -> %s:%d address-family asymmetry (%s vs %s)\n",
749 p->host, p->port, q->host, q->port,
750 p->nf.family == 10 ? "IPv6" : "IPv4",
751 q->nf.family == 10 ? "IPv6" : "IPv4");
752 }
753 }
754 return crit;
755}

826/* Route one URL to its protocol battery by scheme. root:// (and any unrecognized
827 * scheme, for back-compat) goes to the full libbrix battery; the rest to their

837 sizeof(path)) != 0 || proto == DXP_ROOT) {

842 case DXP_HTTP: doctor_http(a, DXP_HTTP, 0, host, port, path, e); break;
843 case DXP_HTTPS: doctor_http(a, DXP_HTTPS, 1, host, port, path, e); break;
844 case DXP_DAVS: doctor_http(a, DXP_DAVS, tls, host, port, path, e); break;
845 case DXP_S3: doctor_s3(a, tls, host, port, path, e); break;
846 case DXP_CMS: doctor_cms(a, host, port, path, e); break;
847 default: doctor_one(a, url, e); break;
848 }
