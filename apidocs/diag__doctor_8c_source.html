10/* Stream a remote file through the live connection into fd; returns 0 / -1 and
11 * sets *out_bytes to the number of bytes written. Reuses the authenticated conn
12 * (no reconnect), so it measures the data path itself. */
13int
14download_to_fd(brix_conn *c, const char *path, int fd, int64_t *out_bytes,
15 brix_status *st)
16{
17 brix_file f;
18 int64_t off = 0;
19 char *buf;
20
21 if (brix_file_open_read(c, path, &f, st) != 0) {
22 return -1;
23 }
24 buf = (char *) malloc(1u << 20);
25 if (buf == NULL) {
26 brix_file_close(c, &f, st);
27 brix_status_set(st, XRDC_EPROTO, 0, "download: out of memory");
28 return -1;
29 }
30 for (;;) {
31 ssize_t r = brix_file_read(c, &f, off, buf, 1u << 20, st);
32 ssize_t w = 0;
33 if (r < 0) {
34 free(buf);
35 brix_file_close(c, &f, st);
36 return -1;
37 }
38 if (r == 0) {
39 break;
40 }
41 while (w < r) {
42 ssize_t k = write(fd, buf + w, (size_t) (r - w));
43 if (k < 0) {
44 free(buf);
45 brix_file_close(c, &f, st);
46 brix_status_set(st, XRDC_ESOCK, 0, "download: local write failed");
47 return -1;
48 }
49 w += k;
50 }
51 off += r;
52 }
53 free(buf);
54 if (out_bytes != NULL) {
55 *out_bytes = off;
56 }
57 return brix_file_close(c, &f, st);
58}

89/* Throughput probe over an established conn: TTFB (first read) + MB/s (whole file). */
90int
91doctor_xfer(brix_conn *c, const char *path, double *ttfb_ms, double *mbps,
92 int64_t *bytes)
93{
94 brix_file f;
95 brix_status st;
96 uint8_t *buf;
97 int64_t off = 0;
98 uint64_t t0, tf = 0, t1;
99
100 brix_status_clear(&st);
101 if (brix_file_open_read(c, path, &f, &st) != 0) {
102 return -1;
103 }
104 buf = (uint8_t *) malloc(1u << 20);
105 if (buf == NULL) {
106 brix_file_close(c, &f, &st);
107 return -1;
108 }
109 t0 = brix_mono_ns();
110 for (;;) {
111 ssize_t r = brix_file_read(c, &f, off, buf, 1u << 20, &st);
112 if (r < 0) { free(buf); brix_file_close(c, &f, &st); return -1; }
113 if (tf == 0) { tf = brix_mono_ns(); } /* time-to-first-byte */
114 if (r == 0) { break; }
115 off += r;
116 }
117 t1 = brix_mono_ns();
118 free(buf);
119 brix_file_close(c, &f, &st);
120 *ttfb_ms = (double) (tf - t0) / 1e6;
121 *bytes = off;
122 {
123 double secs = (double) (t1 - t0) / 1e9;
124 if (secs <= 0.0) { secs = 1e-9; }
125 *mbps = (double) off / 1e6 / secs;
126 }
127 return 0;
128}

354/*
355 * WHAT: verdict-render half of the generic HTTP battery — classify the
356 * transfer-relevant headers of the already-received response.
357 * WHY: extracted from doctor_http for the complexity gate (request-build vs
358 * verdict-render split); stage numbering/order preserved.
359 * HOW: each stage inspects one response header and records one finding.
360 */
361static void
362http_check_common_headers(doctor_ep *e, const brix_http_resp *r)
363{
364 char val[512];
365
366 /* Stage 3: byte ranges (partial reads / multi-stream transfers depend on this). */
367 if (brix_http_header(r, "Accept-Ranges", val, sizeof(val))
368 && strstr(val, "bytes") != NULL) {
369 dx_record(e, &(dx_note){ "ranges", DX_OK, 0, "byte-range reads supported (Accept-Ranges: bytes)", "" });
370 } else {
371 dx_record(e, &(dx_note){ "ranges", DX_WARN, 0,
372 "server did not advertise Accept-Ranges (partial/parallel reads may not work)",
373 "enable range support if clients use partial reads" });
374 }
375
376 /* Stage 4: checksum advertisement (RFC-3230 Digest, WLCG transfers rely on it). */
377 if (brix_http_header(r, "Digest", val, sizeof(val)) && strchr(val, '=') != NULL) {
378 /* RFC-3230 form is "algo=value"; require the '=' so a malformed header
379 * isn't counted as a working checksum. */
380 dx_record(e, &(dx_note){ "checksum", DX_OK, 0, "server advertises a content Digest (checksum)", "" });
381 } else {
382 dx_record(e, &(dx_note){ "checksum", DX_WARN, 0,
383 "no Digest header (checksum verification unavailable over HTTP)",
384 "enable Want-Digest/Digest if integrity checks are required" });
385 }
386
387 /* Stage 5: content-length present (sized transfers). */
388 if (brix_http_header(r, "Content-Length", val, sizeof(val))) {
389 dx_record(e, &(dx_note){ "content-length", DX_OK, 0, "response is sized (Content-Length present)", "" });
390 }
391}

449/*
450 * http / https deep-dive: connect (+TLS cert/cipher), HEAD/GET the path, classify
451 * the HTTP status, byte-range support, the Digest (checksum) header, and 401 auth
452 * posture. For davs, also OPTIONS (WebDAV class) and PROPFIND (listing). Every probe
453 * is bounded by the per-probe timeout. PII-free: only statuses/header-names/sizes.
454 */
455void
456doctor_http(const diag_args *a, const dx_url_t *u, doctor_ep *e)
457{
458 brix_http_resp r;
459 brix_status st;
460 http_probe_ctx pc;
461
462 pc.tls = u->tls;
463 pc.host = u->host;
464 pc.port = u->port;
465 pc.path = u->path;
466 pc.tmo = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
467 pc.verify = a->verify_tls;
468
469 memset(e, 0, sizeof(*e));
470 e->proto = u->proto;
471 e->status = DOC_GREEN;
472 snprintf(e->host, sizeof(e->host), "%s", u->host);
473 e->port = u->port;
474
475 /* Stage 1: reachability + (TLS handshake/cert). Try HEAD; fall back to a 1-byte
476 * ranged GET if HEAD is refused, so we still measure connect/TLS. */
477 brix_status_clear(&st);
478 if (brix_http_req(u->host, u->port, u->tls, "HEAD", u->path, NULL, NULL, 0,
479 pc.tmo, pc.verify, NULL, &r, &st) != 0) {
480 brix_status_clear(&st);
481 if (brix_http_req(u->host, u->port, u->tls, "GET", u->path,
482 "Range: bytes=0-0\r\n", NULL, 0, pc.tmo, pc.verify,

486 }
487 }
488 e->connected = 1;
489 http_record_tls(u->tls, &r, e);
490
491 /* Stage 2: HTTP status. */
492 dx_http_status(e, "http", r.status);
493
494 /* Stages 3-5: transfer-relevant response headers. */
495 http_check_common_headers(e, &r);
496 brix_http_resp_free(&r);
497
498 /* davs extras: OPTIONS (WebDAV class) + PROPFIND (collection listing). */
499 if (u->proto == DXP_DAVS) {
500 http_davs_extras(&pc, e);
501 }
502}

533 * request's response.
534 * WHY: extracted from s3_sigv4_probe for the complexity gate.
535 * HOW: 2xx = accepted; 403 splits into signature fault vs policy deny by

539static void
540s3_classify_signed(doctor_ep *e, const brix_http_resp *r)
541{

558 "check the bucket/object policy for this identity" });

566/*
567 * WHAT: request-build half of the authenticated SigV4 probe.
568 * WHY: extracted from doctor_s3 for the complexity gate; only called when
569 * AWS credentials are present in the environment.
570 * HOW: sign the exact Host header we send, replay the GET with the SigV4
571 * Authorization headers, and hand the response to s3_classify_signed.
572 * PII-free: never emits the key or signature.
573 */
574static void
575s3_sigv4_probe(const http_probe_ctx *pc, const char *ak, const char *sk,
576 doctor_ep *e)
577{
578 brix_http_resp r;
579 brix_status st;
580 char hdrs[1024];
581 char hostport[300];
582 const char *region = getenv("AWS_DEFAULT_REGION");
583 s3_sign_req sq;
584
585 if (region == NULL || region[0] == '\0') { region = "us-east-1"; }
586 /* Sign the exact Host header we send (host:port) — the server canonicalises
587 * the Host value verbatim, so signing the bare host would mismatch. */
588 snprintf(hostport, sizeof(hostport), "%s:%d", pc->host, pc->port);
589 sq.method = "GET";
590 sq.host = hostport;
591 sq.uri = pc->path;
592 sq.ak = ak;
593 sq.sk = sk;
594 sq.region = region;
595 if (s3_sign(&sq, hdrs, sizeof(hdrs)) != 0) {
596 dx_record(e, &(dx_note){ "s3-sigv4", DX_WARN, 0, "could not build a SigV4 signature (client)", "" });
597 return;
598 }
599 brix_status_clear(&st);

602 dx_record(e, &(dx_note){ "s3-sigv4", DX_WARN, st.kxr, "signed request failed to complete", "" });
603 return;
604 }
605 s3_classify_signed(e, &r);
606 brix_http_resp_free(&r);

610/*
611 * s3 deep-dive: connect (+TLS), an UNAUTHENTICATED GET to confirm reachability +
612 * that the server enforces auth (403/AccessDenied) vs is public (200) vs missing
613 * (404/NoSuchBucket). If AWS_ACCESS_KEY_ID/SECRET are in the environment, also send
614 * a SigV4-signed HEAD and confirm the signature is accepted (catches signer/clock/
615 * region faults). PII-free: never emits the key or signature.

618doctor_s3(const diag_args *a, const dx_url_t *u, doctor_ep *e)

624 const char *sk = getenv("AWS_SECRET_ACCESS_KEY");

626 pc.tls = u->tls;
627 pc.host = u->host;
628 pc.port = u->port;

632
633 memset(e, 0, sizeof(*e));
634 e->proto = DXP_S3;
635 e->status = DOC_GREEN;
636 snprintf(e->host, sizeof(e->host), "%s", u->host);

640 brix_status_clear(&st);

662/*
663 * cms deep-dive: a cluster manager (cmsd/redirector) is a root:// endpoint that
664 * answers kXR_locate with the data server(s) holding a path and issues kXR_redirect.
665 * Connect to the manager, locate the path, and confirm the redirect resolution to a
666 * reachable data server; flag no-holder / unreachable-DS (ghost) / redirect issues.
667 * Reuses the libbrix locate + reconnect machinery (the redirect loop-guard applies).
668 */
669void
670doctor_cms(const diag_args *a, const char *host, int port, const char *path,
671 doctor_ep *e)
672{
673 brix_url u;
674 brix_conn c;
675 brix_status st;
676
677 memset(e, 0, sizeof(*e));
678 e->proto = DXP_CMS;
679 e->status = DOC_GREEN;
680 snprintf(e->host, sizeof(e->host), "%s", host);
681 e->port = port;
682
683 memset(&u, 0, sizeof(u));
684 u.scheme = XRDC_SCHEME_ROOT;
685 snprintf(u.host, sizeof(u.host), "%s", host);
686 u.port = port;
687 snprintf(u.path, sizeof(u.path), "%s", path[0] ? path : "/");
688
689 brix_status_clear(&st);
690 if (brix_connect(&c, &u, &a->conn, &st) != 0) {
691 dx_record(e, &(dx_note){ "cms-connect", DX_FAIL, st.kxr,
692 "could not connect to the cluster manager",
693 "check the manager (cmsd/redirector) is up on this host:port" });
694 doc_issue(e, DOC_RED, "manager connect failed");
695 return;
696 }
697 e->connected = 1;
698 c.io.timeout_ms = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
699 dx_record(e, &(dx_note){ "cms-connect", DX_OK, 0, "manager reachable + login completed", "" });
700
701 {
702 char loc[2048];
703 brix_status lst;
704 brix_status_clear(&lst);
705 if (brix_locate(&c, u.path, loc, sizeof(loc), &lst) != 0) {
706 /* a manager returns NotFound when no server holds the path. */
707 dx_record_status(e, "cms-locate", &lst);
708 } else {
709 char *t, *save;
710 for (t = strtok_r(loc, " \t\r\n", &save); t != NULL;
711 t = strtok_r(NULL, " \t\r\n", &save)) {
712 if (t[0] == 'S') { e->holders++; }
713 }
714 if (e->holders == 0) {
715 dx_record(e, &(dx_note){ "cms-locate", DX_FAIL, 0,
716 "manager located no data server for the path (no holder)",
717 "check data-server registration and the CMS registry" });
718 } else {
719 dx_record(e, &(dx_note){ "cms-locate", DX_OK, 0,
720 "manager resolved the path to data server(s)", "" });
721 }
722 }
723 }
724
725 /* Resolution: a stat through the manager must follow the redirect to a live DS.
726 * A redirect loop or dead DS surfaces here (the loop-guard returns an error). */
727 {
728 brix_statinfo si;
729 brix_status rst;
730 brix_status_clear(&rst);
731 if (brix_stat(&c, u.path, &si, &rst) == 0) {
732 dx_record(e, &(dx_note){ "cms-redirect", DX_OK, 0,
733 "manager→data-server redirect resolved to a live server", "" });
734 } else if (rst.kxr == kXR_NotFound) {
735 dx_record(e, &(dx_note){ "cms-redirect", DX_WARN, rst.kxr,
736 "path not found via the manager (redirect resolved, file absent)",
737 "verify the path exists on a registered data server" });
738 } else {
739 dx_record(e, &(dx_note){ "cms-redirect", DX_FAIL, rst.kxr,
740 "manager redirect did not resolve to a reachable data server (dead DS / redirect loop)",
741 "check data-server health and the CMS registry for stale entries" });
742 }
743 }
744 brix_close(&c);
745}

748/*
749 * WHAT: classify why the primary connection could not be set up.
750 * WHY: extracted from doctor_one's connect-failure path (complexity gate).
751 * HOW: map errno / status-message onto a fixed cause+remedy pair; the caller
752 * reports the classified cause, not st->msg (wire text may carry PII).
753 */
754static void
755doctor_classify_conn_error(const brix_status *st, const char **cause,
756 const char **remedy)
757{
758 *cause = "connection setup failed";
759 *remedy = "check the network path and that the server is running";
760 if (st->sys_errno == ECONNREFUSED) {
761 *cause = "no listener on host:port (service down or wrong port)";
762 *remedy = "start the gateway / verify the port and any firewall";
763 } else if (st->sys_errno == ETIMEDOUT || st->sys_errno == EHOSTUNREACH
764 || st->sys_errno == ENETUNREACH) {
765 *cause = "host/network unreachable (routing or firewall drop)";
766 *remedy = "check routing/firewall and that the host is up";
767 } else if (st->msg[0] != '\0' && strstr(st->msg, "resolve") != NULL) {
768 *cause = "DNS resolution failed";
769 *remedy = "check the hostname and DNS resolver";
770 }
771}

774/*
775 * WHAT: verdict-render half of a failed primary connect — auth rejection vs
776 * transport unreachability, plus the standalone auth-suite.
777 * WHY: extracted from doctor_one for the complexity gate.
778 * HOW: distinguish "server reachable but auth failed" from "couldn't reach
779 * it" by the client's error CODE: XRDC_EAUTH / kXR_NotAuthorized /
780 * kXR_AuthFailed mean auth was attempted and rejected; everything else
781 * is transport.
782 */
783static void
784doctor_one_connect_fail(const diag_args *a, const brix_url *u,
785 const brix_status *st, const char *target,
786 doctor_ep *e)
787{
788 if (st->kxr == XRDC_EAUTH || st->kxr == kXR_NotAuthorized
789 || st->kxr == kXR_AuthFailed) {
790 doc_issue(e, DOC_RED, "authentication failed");
791 dx_record(e, &(dx_note){ "auth", DX_FAIL, st->kxr,
792 "could not authenticate (credential rejected, or none usable for the server's auth)",
793 "check the credential's validity/scope and that it matches the server's auth mode" });
794 } else {
795 /* reachability: classify *why* the connection could not be set up. */
796 const char *cause, *remedy;
797 doctor_classify_conn_error(st, &cause, &remedy);
798 /* use the classified cause, not st.msg — wire text may carry PII. */
799 doc_issue(e, DOC_RED, "connect failed: %s", cause);
800 dx_record(e, &(dx_note){ "reachability", DX_FAIL, st->kxr, cause, remedy });
801 }
802 /* the auth-suite is self-contained (its own force_anon session) — run it
803 * even when our credential could not establish the primary connection. */
804 if (a->auth_suite) {
805 doctor_auth_suite(a, u, target, 0, e);
806 }
807}

895/* Interrogate ONE endpoint into *e. Bounded by the conn timeout (never hangs). */
896void
897doctor_one(const diag_args *a, const char *url, doctor_ep *e)
898{
899 brix_url u;
900 brix_conn c;
901 brix_status st;
902 char target[XRDC_PATH_MAX];
903 int have_target = 0;
904
905 target[0] = '\0';
906 memset(e, 0, sizeof(*e));
907 e->status = DOC_GREEN;
908 brix_status_clear(&st);
909 if (brix_endpoint_parse(url, &u, &st) != 0) {
910 snprintf(e->host, sizeof(e->host), "%s", url);
911 doc_issue(e, DOC_RED, "unparseable URL (bad scheme/host/port)");
912 return;
913 }
914 snprintf(e->host, sizeof(e->host), "%s", u.host);
915 e->port = u.port;
916
917 if (brix_connect(&c, &u, &a->conn, &st) != 0) {
918 doctor_one_connect_fail(a, &u, &st, target, e);
919 return;
920 }
921 e->connected = 1;
922 c.io.timeout_ms = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
923
924 doctor_one_session_facts(&c, e);
925
926 /* throughput probe over a resolved file (skip cleanly if the export is empty) */
927 have_target = doctor_one_xfer_probe(&c, &u, target, sizeof(target), e);
928
929 /* active differential diagnosis — exercise subsystems + classify (incl. locate). */
930 doctor_diagnose(a, &c, &u,
931 &(dx_target){ .path = target, .have = have_target }, e);
932
933 brix_close(&c);
934
935 doctor_one_load_signals(a, e);
936}

946/*
947 * WHAT: transfer-path diff of one adjacent endpoint pair.
948 * WHY: extracted from doctor_cross for the complexity gate.
949 * HOW: emit TLS-downgrade (critical), auth-fallback, and address-family
950 * asymmetry lines in the original order; returns the number of
951 * critical diffs found (0 or 1).
952 */
953static int
954cross_diff_pair(const doctor_ep *p, const doctor_ep *q, FILE *out)
955{
956 int crit = 0;
957
958 if (!p->connected || !q->connected) {
959 return 0;
960 }
961 /* TLS-downgrade: a TLS hop followed by a cleartext one */
962 if (p->tls_active && !q->tls_active) {
963 fprintf(out, " %s:%d -> %s:%d TLS DOWNGRADE (encrypted then cleartext)\n",
964 p->host, p->port, q->host, q->port);
965 crit++;
966 }
967 /* auth-fallback: the chosen auth weakens across the hop */
968 if (strcmp(p->auth, q->auth) != 0) {
969 fprintf(out, " %s:%d -> %s:%d auth changed %s -> %s\n",
970 p->host, p->port, q->host, q->port, p->auth, q->auth);
971 }
972 /* v4/v6 asymmetry */
973 if (p->nf.family && q->nf.family && p->nf.family != q->nf.family) {
974 fprintf(out, " %s:%d -> %s:%d address-family asymmetry (%s vs %s)\n",
975 p->host, p->port, q->host, q->port,
976 p->nf.family == 10 ? "IPv6" : "IPv4",
977 q->nf.family == 10 ? "IPv6" : "IPv4");
978 }
979 return crit;
980}

983/* Cross-endpoint diff engine over the transfer path. Returns #critical diffs. */
984int
985doctor_cross(const doctor_ep *eps, int n, FILE *out)
986{
987 int i, crit = 0, connected = 0;
988 if (n < 2) {
989 return 0;
990 }
991 for (i = 0; i < n; i++) {
992 if (eps[i].connected) { connected++; }
993 }
994 if (connected < 2) {
995 return 0; /* fewer than two reachable hops — nothing to compare */
996 }
997 fprintf(out, "Path analysis (%d hops):\n", n);
998 for (i = 1; i < n; i++) {
999 crit += cross_diff_pair(&eps[i - 1], &eps[i], out);
1000 }
1001 return crit;
1002}
