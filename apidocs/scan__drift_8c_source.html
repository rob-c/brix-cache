25static size_t
26drift_hash(const char *s)
27{
28 size_t h = 5381;
29 for (; *s != '\0'; s++) {
30 h = ((h << 5) + h) ^ (unsigned char) *s; /* djb2-xor */
31 }
32 return h;
33}

35static size_t
36drift_roundup_pow2(size_t v)
37{
38 size_t p = 16;
39 while (p < v) {
40 p <<= 1;

77/* Locate the slot for `key`: either the slot holding it, or the first empty slot
78 * on its probe chain. Returns the index. */
79static size_t
80drift_probe(const drift_slot *tab, size_t cap, const char *key)
81{
82 size_t i = drift_hash(key) & (cap - 1);
83
84 while (tab[i].key != NULL && strcmp(tab[i].key, key) != 0) {
85 i = (i + 1) & (cap - 1);
86 }
87 return i;
88}

90static int
91drift_grow(brix_scan_driftset_t *s)
92{
93 size_t newcap = s->cap << 1;
94 drift_slot *newtab = calloc(newcap, sizeof(*newtab));
95 size_t i;
96
97 if (newtab == NULL) {
98 return -1;
99 }
100 for (i = 0; i < s->cap; i++) {
101 if (s->tab[i].key != NULL) {
102 size_t j = drift_probe(newtab, newcap, s->tab[i].key);
103 newtab[j] = s->tab[i]; /* move the key pointer + size + seen */
104 }
105 }
106 free(s->tab);
107 s->tab = newtab;
108 s->cap = newcap;
109 return 0;
110}

112int
113brix_scan_driftset_add(brix_scan_driftset_t *s, const char *key, int64_t size)
114{
115 size_t i;
116
117 if ((s->n + 1) * 10 >= s->cap * 7) { /* > 70% load ⇒ grow */
118 if (drift_grow(s) != 0) {
119 return -1;
120 }
121 }
122 i = drift_probe(s->tab, s->cap, key);
123 if (s->tab[i].key != NULL) {
124 s->tab[i].size = size; /* update existing */
125 return 0;
126 }
127 s->tab[i].key = strdup(key);
128 if (s->tab[i].key == NULL) {
129 return -1;
130 }
131 s->tab[i].size = size;
132 s->tab[i].seen = 0;
133 s->n++;
134 return 0;
135}
