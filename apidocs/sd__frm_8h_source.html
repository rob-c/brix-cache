4/*
5 * sd_frm.h - the nearline (tape/MSS) backend FS driver (phase-64 SP5, P4/P6).
6 *
7 * WHAT: A capability-typed SD driver whose objects are NEARLINE: an object is
8 * either ONLINE (present in the MSS disk buffer, directly readable) or
9 * OFFLINE (on tape, minutes-to-hours away). It advertises CAP_NEARLINE and
10 * implements the `recall` slot; the composing registry therefore REQUIRES a
11 * cache tier in front of it (G8, the recall target). A read of an offline
12 * object is a RECALL into the online buffer - exactly the cache miss-fill,
13 * but sourced from tape.
14 *
15 * WHY: P6 makes tape "just another backend tier": instead of a parallel FRM
16 * subsystem that drivers call, tape residency becomes an SD driver behind
17 * the same seam (stat/open/pread/recall/staged_*). A node composes
18 * cache(frm-backend) and serves tape transparently - no special FRM code
19 * path above the seam. (This SP5 increment lands the driver + an in-tree
20 * stub MSS adapter and a SYNCHRONOUS recall; the async park/wake of a slow
21 * recall via the stage_engine waiter, and the dissolution of the legacy
22 * src/frm/ subsystem, are the remaining P6 migration.)
23 *
24 * HOW: Tape access is abstracted behind a pluggable MSS adapter
25 * (brix_mss_adapter_t): residency (online/offline/absent), recall
26 * begin/poll (tape -> online buffer), and migrate (online -> tape). The
27 * `tape://` store-URL selects the adapter; "stub" is the built-in
28 * local-directory simulation used for tests. recall ensures the object is
29 * online (driving the adapter), then sd_cache fills from the online buffer
30 * through this driver's open/pread; staged_* writes the online buffer then
31 * migrates to tape. See docs/refactor/phase-64-fully-tiered-composable-
32 * storage.md (section 9, 13b, 26, Appendix I).
33 */

43/*
44 * The pluggable MSS adapter - how the frm driver talks to the tape system. One
45 * vtable; the tape:// store-URL selects it (exec | hpss | cta | stub). All calls
46 * are blocking and run on the cache-fill / scheduler path (off the event loop).
47 */
48typedef struct {
49 const char *name; /* "stub" | "exec" | ... */
50 /* Is `key` online, nearline, or offline/absent? Fills size/mtime when known. */
51 int (*residency)(void *mss, const char *key, off_t *size_out,
52 time_t *mtime_out);
53 /* Begin a recall of `key` into the MSS online buffer; 0 started / -1 error. */
54 int (*recall_begin)(void *mss, const char *key);
55 /* Poll a recall: 1 online (ready), 0 in-flight, -1 error. */
56 int (*recall_poll)(void *mss, const char *key);
57 /* Migrate `key` from the online buffer to tape (after a staged write). */
58 int (*migrate)(void *mss, const char *key);
59 /* Drop the online copy of `key` (free buffer space). */
60 int (*purge)(void *mss, const char *key);
61 /* Open the online-buffer file of an ONLINE `key` for reading; fd or -1. */
62 int (*open_online)(void *mss, const char *key);
63 /* Create+open the online-buffer file of `key` for a staged write; fd or -1. */
64 int (*create_online)(void *mss, const char *key, mode_t mode);
65 void (*destroy)(void *mss);
66} brix_mss_adapter_t;
