1/*
2 * aio_mgr.c — connection manager + transparent file-handle resumption (M3).
3 *
4 * WHAT: Two layers on top of the async loop (aio.c):
5 * - brix_mgr: the loop plus a small pool of attached connections. Metadata
6 * requests round-robin across them; idempotent ones survive a reconnect
7 * transparently (retry_safe at the transport layer, M2).
8 * - brix_mfile: an open file that survives a connection drop. Because an
9 * XRootD file handle is valid only on the session that opened it, a
10 * reconnect invalidates it; so on a transport failure or a stale-handle
11 * error this layer REOPENS the file (fresh handle, NON-destructively — no
12 * re-truncate, no create-excl) and RE-ISSUES the read/write at the same
13 * absolute offset. Re-issuing the identical offset is idempotent, so a
14 * mid-transfer cat/dd survives a server restart with no data loss and no
15 * EIO reaching the caller.
16 * WHY: M2 makes the transport reconnect; M3 makes *open files* recover, which is
17 * what "survives a server bounce mid-transfer" actually requires.
18 * HOW: mfile is a blocking façade over the async loop: each pread/pwrite is one
19 * brix_aio_call, and the (many) FUSE worker threads calling concurrently
20 * pipeline over the shared connection. A per-file mutex + generation counter
21 * serialises reopen so concurrent callers reopen at most once and then reuse
22 * the fresh handle.
23 *
24 * Clean-room: the existing wire structs (ClientOpen/Read/Write/Close/SyncRequest)
25 * + the async loop. No XrdCl.
26 */

40/* mgr */
41struct brix_mgr {
42 brix_loop *loop;
43 int n; /* total stream slots */
44 brix_conn *conns; /* array[n], owned */
45 brix_aconn **acs; /* array[n]; acs[i]==NULL ⇒ slot i not yet connected */
46 int rr; /* round-robin cursor */
47 int max_stall_ms;
48 int keepalive_ms;
49 int max_retries;
50 /* Retained so lazily-opened streams (eager < n) can connect on first use. */
51 brix_url url;
52 brix_opts opts;
53 int have_opts; /* opts captured (vs caller passed NULL) */
54 pthread_mutex_t lazy_lock; /* serialises first-use connect of a lazy slot */
55};

57/* One parallel-connect job: a worker thread runs brix_connect on its own slot so
58 * the eager streams' connect+TLS+login+auth round-trips overlap (mount-time wall
59 * collapses from eager×RTT to ~1×RTT). Threads are joined before mgr_create
60 * returns, so they never cross a later fuse daemonize fork. */
61typedef struct {
62 const brix_url *u;
63 const brix_opts *o;
64 brix_conn *conn; /* slot to fill */
65 brix_status st; /* per-thread status (no shared writes) */
66 int rc; /* 0 ok, -1 fail */
67} mgr_connect_job;

69static void *
70mgr_connect_worker(void *arg)
71{
72 mgr_connect_job *j = (mgr_connect_job *) arg;
73 j->rc = brix_connect(j->conn, j->u, j->o, &j->st);
74 return NULL;
75}

77/* Run `count` connect jobs concurrently and wait for all to finish. A thread that
78 * fails to spawn runs its job inline (degrades to serial for that one), so the
79 * caller always sees every job's rc/st populated on return. */
80static void
81mgr_connect_parallel(mgr_connect_job *jobs, int count)
82{
83 pthread_t *tids = (pthread_t *) calloc((size_t) count, sizeof(*tids));
84 int i;
85
86 for (i = 0; i < count; i++) {
87 if (tids == NULL
88 || pthread_create(&tids[i], NULL, mgr_connect_worker, &jobs[i]) != 0) {
89 if (tids != NULL) { tids[i] = (pthread_t) 0; }
90 mgr_connect_worker(&jobs[i]); /* inline fallback */
91 }
92 }
93 if (tids != NULL) {
94 for (i = 0; i < count; i++) {
95 if (tids[i] != (pthread_t) 0) { pthread_join(tids[i], NULL); }
96 }
97 free(tids);
98 }
99}

101/* Tear down a partially-built manager (used on any create-time failure). */
102static void
103mgr_free(brix_mgr *m)
104{
105 int i;
106 for (i = 0; i < m->n; i++) {
107 if (m->acs[i] != NULL) { brix_aconn_close(m->acs[i]); }
108 }
109 if (m->loop != NULL) { brix_loop_destroy(m->loop); }
110 for (i = 0; i < m->n; i++) {
111 if (m->acs[i] != NULL) { brix_close(&m->conns[i]); }
112 }
113 pthread_mutex_destroy(&m->lazy_lock);
114 free(m->conns);
115 free(m->acs);
116 free(m);
117}

119brix_mgr *
120brix_mgr_create(const brix_url *u, const brix_opts *o, int nconns, int eager,
121 int max_stall_ms, int keepalive_ms, int max_retries,
122 brix_status *st)
123{
124 mgr_connect_job *jobs;
125 int i;
126
127 if (nconns < 1) {
128 nconns = 1;
129 }
130 /* At least one stream connects up front so a bad endpoint / auth fails the
131 * mount immediately; the remainder may be eager or lazy per the caller. */
132 if (eager < 1) { eager = 1; }
133 if (eager > nconns) { eager = nconns; }
134
135 brix_mgr *m = (brix_mgr *) calloc(1, sizeof(*m));
136 if (m == NULL) {
137 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (mgr)");
138 return NULL;
139 }
140 m->conns = (brix_conn *) calloc((size_t) nconns, sizeof(*m->conns));
141 m->acs = (brix_aconn **) calloc((size_t) nconns, sizeof(*m->acs));
142 if (m->conns == NULL || m->acs == NULL) {
143 free(m->conns);
144 free(m->acs);
145 free(m);
146 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (mgr arrays)");
147 return NULL;
148 }
149 m->n = nconns;
150 m->max_stall_ms = max_stall_ms;
151 m->keepalive_ms = keepalive_ms;
152 m->max_retries = max_retries;
153 m->url = *u;
154 if (o != NULL) { m->opts = *o; m->have_opts = 1; }
155 pthread_mutex_init(&m->lazy_lock, NULL);
156
157 m->loop = brix_loop_create(st);
158 if (m->loop == NULL) {
159 mgr_free(m);
160 return NULL;
161 }
162
163 /* Connect the eager streams concurrently, then attach them to the loop
164 * serially (attach touches the shared loop and is not the slow part). */
165 jobs = (mgr_connect_job *) calloc((size_t) eager, sizeof(*jobs));
166 if (jobs == NULL) {
167 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (mgr jobs)");
168 mgr_free(m);
169 return NULL;
170 }
171 for (i = 0; i < eager; i++) {
172 jobs[i].u = &m->url; jobs[i].o = m->have_opts ? &m->opts : NULL;
173 jobs[i].conn = &m->conns[i]; jobs[i].rc = -1;
174 }
175 mgr_connect_parallel(jobs, eager);
176
177 for (i = 0; i < eager; i++) {
178 if (jobs[i].rc != 0) {
179 *st = jobs[i].st; /* surface the first real failure */
180 for (int k = 0; k < eager; k++) { /* close any that DID connect */
181 if (k != i && jobs[k].rc == 0) { brix_close(&m->conns[k]); }
182 }
183 free(jobs);
184 mgr_free(m); /* acs all NULL ⇒ frees loop+arrays */
185 return NULL;
186 }
187 }
188 free(jobs);
189
190 for (i = 0; i < eager; i++) {
191 m->acs[i] = brix_aconn_attach(m->loop, &m->conns[i], st);
192 if (m->acs[i] == NULL) {
193 brix_close(&m->conns[i]); /* this one attached failed */
194 for (int k = i + 1; k < eager; k++) { brix_close(&m->conns[k]); }
195 mgr_free(m);
196 return NULL;
197 }
198 brix_aconn_set_resilience(m->acs[i], max_stall_ms, keepalive_ms,
199 max_retries);
200 }
201 return m;
202}

213/* Bring a lazily-deferred slot up on first use: connect + attach + arm
214 * resilience under lazy_lock (double-checked so concurrent pickers connect it
215 * at most once). On failure the slot stays NULL and the caller falls back to an
216 * already-live stream — eager ≥ 1 guarantees one exists. */
217static brix_aconn *
218mgr_ensure_slot(brix_mgr *m, int i)
219{
220 brix_aconn *ac = __atomic_load_n(&m->acs[i], __ATOMIC_ACQUIRE);
221 brix_status st;
222
223 if (ac != NULL) {
224 return ac;
225 }
226 pthread_mutex_lock(&m->lazy_lock);
227 ac = m->acs[i]; /* re-check under the lock */
228 if (ac == NULL) {
229 if (brix_connect(&m->conns[i], &m->url,
230 m->have_opts ? &m->opts : NULL, &st) == 0) {
231 ac = brix_aconn_attach(m->loop, &m->conns[i], &st);
232 if (ac != NULL) {
233 brix_aconn_set_resilience(ac, m->max_stall_ms, m->keepalive_ms,
234 m->max_retries);
235 __atomic_store_n(&m->acs[i], ac, __ATOMIC_RELEASE);
236 } else {
237 brix_close(&m->conns[i]);
238 }
239 }
240 }
241 pthread_mutex_unlock(&m->lazy_lock);
242 return ac;
243}

245brix_aconn *
246brix_mgr_pick(brix_mgr *m)
247{
248 int i = __atomic_fetch_add(&m->rr, 1, __ATOMIC_RELAXED);
249 brix_aconn *ac;
250 int k;
251
252 if (i < 0) {
253 i = -i;
254 }
255 i %= m->n;
256
257 ac = mgr_ensure_slot(m, i);
258 if (ac != NULL) {
259 return ac;
260 }
261 /* Lazy connect of slot i failed (server hiccup): fall back to any live
262 * stream rather than returning NULL — at least the eager slot(s) are up. */
263 for (k = 0; k < m->n; k++) {
264 ac = __atomic_load_n(&m->acs[k], __ATOMIC_ACQUIRE);
265 if (ac != NULL) {
266 return ac;
267 }
268 }
269 return NULL;
270}

283struct brix_mfile {
284 brix_aconn *ac;
285 char path[XRDC_PATH_MAX];
286 char opaque[256];
287 int have_opaque;
288 int writable;
289 int posc;

295 uint64_t gen; /* bumped on each (re)open */
296 /* phase-42 W4: inline read-compression codec learned from the open reply
297 * cptype[0] (0 = plaintext). Re-learned on every (re)open in mfile_do_open,
298 * so reopen-at-offset after a fault transparently re-negotiates; a server
299 * that declines simply yields 0 and reads stay plaintext. */
300 uint8_t read_codec;
301 /* phase-42 W5: inline write-compression codec (write opens only); each
302 * brix_mfile_pwrite compresses its payload as a self-contained frame the
303 * server decompresses on ingest. Re-learned on every (re)open. */
304 uint8_t write_codec;

313/* Build + send a kXR_open with the given force tri-state (see brix_file_open_opaque).
314 * On success copies the fresh handle into mf->fhandle. Caller holds no lock. */
315static int
316mfile_do_open(brix_mfile *mf, int force, brix_status *st)
317{
318 ClientOpenRequest req;
319 uint16_t options;
320 uint16_t kxr = 0;
321 uint8_t *body = NULL;
322 uint32_t blen = 0;
323 char *payload;
324 size_t need;
325 int plen;
326
327 if (mf->writable) {
328 options = kXR_open_updt | kXR_mkpath;
329 if (force == 1) {
330 options |= kXR_delete; /* truncate/overwrite */
331 } else if (force == 0) {
332 options |= kXR_new; /* create-excl */
333 } /* force == 2: update in place */
334 if (mf->posc) {
335 options |= kXR_posc;
336 }
337 } else {
338 options = kXR_open_read;
339 }
340
341 need = strlen(mf->path) + 1 + (mf->have_opaque ? strlen(mf->opaque) + 1 : 0);
342 payload = (char *) malloc(need);
343 if (payload == NULL) {
344 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (open payload)");
345 return -1;
346 }
347 plen = mf->have_opaque
348 ? snprintf(payload, need, "%s?%s", mf->path, mf->opaque)
349 : snprintf(payload, need, "%s", mf->path);
350
351 memset(&req, 0, sizeof(req));
352 req.requestid = htons(kXR_open);
353 {
354 xrdw_open_req_t b = { .mode = (uint16_t) (mf->writable ? 0644 : 0),
355 .options = options, .optiont = 0 };
356 xrdw_open_req_pack(&b, ((ClientRequestHdr *) &req)->body);
357 }
358
359 brix_aio_opts o = { mf->max_stall_ms, mf->max_retries, 1 /*open is idempotent*/ };
360 int rc = brix_aio_call_ex(mf->ac, &req, payload, (uint32_t) plen, &o,
361 &kxr, &body, &blen, st);
362 free(payload);
363 if (rc != 0) {
364 return -1;
365 }
366 if (blen < XRD_FHANDLE_LEN) {
367 brix_status_set(st, XRDC_EPROTO, 0, "open reply too short (%u bytes)", blen);
368 free(body);
369 return -1;
370 }
371 memcpy(mf->fhandle, body, XRD_FHANDLE_LEN);
372
373 /* phase-42 W4/W5: learn the inline-compression codec from the open reply
374 * (ServerOpenBody = fhandle[4] cpsize[4] cptype[4]). Per the dual-check
375 * contract (codec_core.h) adopt cptype[0] ONLY when cpsize == the big-endian
376 * BRIX_INLINE_CMP_MAGIC: cptype is a legacy field a non-cooperating server
377 * may reuse, and trusting it alone would inflate plaintext and corrupt data.
378 * Read opens inflate responses (W4); write opens compress payloads (W5).
379 * Stock servers leave cpsize 0 → plaintext. Re-learned on every (re)open, so
380 * a fault mid-transfer transparently re-negotiates. */
381 mf->read_codec = 0;
382 mf->write_codec = 0;
383 if (blen >= 12) {
384 uint32_t cpsize = ((uint32_t) body[4] << 24) | ((uint32_t) body[5] << 16)
385 | ((uint32_t) body[6] << 8) | (uint32_t) body[7];
386 uint8_t cid = body[8]; /* cptype[0] */
387 if (cpsize == BRIX_INLINE_CMP_MAGIC && cid >= 1 && cid < BRIX_CODEC_MAX) {
388 /* Server confirmed codec `cid`; if this build cannot handle it, fail
389 * the open rather than silently copying compressed bytes as plaintext
390 * (asymmetric-build corruption). */
391 if (!brix_codec_available(cid)) {
392 brix_status_set(st, XRDC_EUNSUPPORTED, 0,
393 "server negotiated inline-compression codec %u that this "
394 "client build cannot decode", (unsigned) cid);
395 free(body);
396 return -1;
397 }
398 if (mf->writable) {
399 mf->write_codec = cid;
400 } else {
401 mf->read_codec = cid;
402 }
403 }
404 }
405
406 free(body);
407 return 0;
408}

410brix_mfile *
411brix_mfile_open(brix_aconn *ac, const char *path, int writable, int force,
412 int posc, const char *opaque, int max_stall_ms, int max_retries,
413 brix_status *st)
414{
415 brix_mfile *mf = (brix_mfile *) calloc(1, sizeof(*mf));
416 if (mf == NULL) {
417 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (mfile)");
418 return NULL;
419 }
420 mf->ac = ac;
421 snprintf(mf->path, sizeof(mf->path), "%s", path);
422 if (opaque != NULL && opaque[0] != '\0') {
423 snprintf(mf->opaque, sizeof(mf->opaque), "%s", opaque);
424 mf->have_opaque = 1;
425 }
426 mf->writable = writable;
427 mf->posc = posc;
428 mf->max_stall_ms = max_stall_ms;
429 mf->max_retries = max_retries;
430 pthread_mutex_init(&mf->lock, NULL);
431
432 if (mfile_do_open(mf, force, st) != 0) {
433 pthread_mutex_destroy(&mf->lock);
434 free(mf);
435 return NULL;
436 }
437 mf->have_handle = 1;
438 mf->gen = 1;
439 return mf;

442/* A failure that a reopen + retry might fix: a transient transport fault, or a

446{
447 return brix_status_retryable(st) || st->kxr == kXR_FileNotOpen;

450/* Reopen the file NON-destructively if no one else already did (generation match).
451 * Returns 0 if the file is open (by us or a racing caller), -1 on failure. */
452static int
453mfile_reopen(brix_mfile *mf, uint64_t expected_gen, brix_status *st)
454{
455 pthread_mutex_lock(&mf->lock);
456 if (mf->gen != expected_gen && mf->have_handle) {
457 pthread_mutex_unlock(&mf->lock);
458 return 0; /* someone else reopened while we waited */
459 }
460 mf->have_handle = 0;
461 int rc = mfile_do_open(mf, mf->writable ? 2 /*update, no truncate*/ : -1, st);
462 if (rc == 0) {
463 mf->have_handle = 1;
464 mf->gen++;
465 }
466 pthread_mutex_unlock(&mf->lock);
467 return rc;
468}

470/* Snapshot the current handle + generation (+ optional read/write codecs) under
471 * the lock. rcodec/wcodec may be NULL for callers that don't need them. */
472static int
473mfile_snapshot(brix_mfile *mf, uint8_t fh[XRD_FHANDLE_LEN], uint64_t *gen,
474 uint8_t *rcodec, uint8_t *wcodec)
475{
476 pthread_mutex_lock(&mf->lock);
477 int ok = mf->have_handle;
478 if (ok) {
479 memcpy(fh, mf->fhandle, XRD_FHANDLE_LEN);
480 }
481 if (rcodec != NULL) {
482 *rcodec = mf->read_codec;
483 }
484 if (wcodec != NULL) {
485 *wcodec = mf->write_codec;
486 }
487 *gen = mf->gen;
488 pthread_mutex_unlock(&mf->lock);
489 return ok;
490}

492ssize_t
493brix_mfile_pread(brix_mfile *mf, int64_t off, void *buf, size_t len,
494 brix_status *st)
495{
496 /* Retry transport faults until the max_stall patience window elapses, with
497 * exponential backoff between attempts — NOT a fixed small count. Under
498 * sustained loss (a flapping firewall) a fixed count is exhausted in a tight
499 * spin long before the patience window, surfacing a spurious EIO; bounding by
500 * the deadline rides the loss out as long as progress is possible. */
501 uint64_t deadline = now_ns() + (uint64_t) mf->max_stall_ms * 1000000ULL;
502 unsigned attempt = 0;
503
504 for (;;) {
505 uint8_t fh[XRD_FHANDLE_LEN];
506 uint64_t gen;
507 uint8_t codec = 0;
508 if (!mfile_snapshot(mf, fh, &gen, &codec, NULL)) {
509 if (mfile_reopen(mf, gen, st) != 0) {
510 if (now_ns() >= deadline) {
511 return -1;
512 }
513 brix_backoff_sleep_fast(attempt++);
514 }
515 continue;
516 }
517
518 ClientReadRequest req;
519 memset(&req, 0, sizeof(req));
520 req.requestid = htons(kXR_read);
521 {
522 xrdw_read_req_t b = { .offset = off, .rlen = (int32_t) len };
523 memcpy(b.fhandle, fh, XRD_FHANDLE_LEN);
524 xrdw_read_req_pack(&b, ((ClientRequestHdr *) &req)->body);
525 }
526
527 uint16_t kxr = 0;
528 uint8_t *body = NULL;
529 uint32_t blen = 0;
530 brix_aio_opts o = { mf->max_stall_ms, 0, 0 /*we own reopen/retry*/ };
531 if (brix_aio_call_ex(mf->ac, &req, NULL, 0, &o, &kxr, &body, &blen, st) == 0) {
532 /* phase-42 W4: a compressed handle returns one self-contained codec
533 * frame per request (aio_call_ex has already accumulated all
534 * oksofar frames into `body`); inflate it into the caller's buffer.
535 * A corrupt frame is not a transport fault — do not reopen/retry. */
536 if (codec != 0) {
537 ssize_t pl = brix_inflate_frame(codec, body, blen, buf, len, st);
538 free(body);
539 return pl;
540 }
541 size_t n = (blen < len) ? blen : len;
542 if (n > 0) {
543 memcpy(buf, body, n);
544 }
545 free(body);
546 return (ssize_t) n;
547 }
548 if (!mfile_should_reopen(st) || now_ns() >= deadline) {
549 return -1;
550 }
551 brix_backoff_sleep_fast(attempt++); /* don't tight-spin under loss */
552 (void) mfile_reopen(mf, gen, st); /* refresh handle, then retry */
553 }
554}

556int
557brix_mfile_pwrite(brix_mfile *mf, int64_t off, const void *buf, size_t len,
558 brix_status *st)
559{
560 /* Deadline-bounded retry with backoff (see brix_mfile_pread). */
561 uint64_t deadline = now_ns() + (uint64_t) mf->max_stall_ms * 1000000ULL;
562 unsigned attempt = 0;
563
564 for (;;) {
565 uint8_t fh[XRD_FHANDLE_LEN];
566 uint64_t gen;
567 uint8_t wcodec = 0;
568 if (!mfile_snapshot(mf, fh, &gen, NULL, &wcodec)) {
569 if (mfile_reopen(mf, gen, st) != 0) {
570 if (now_ns() >= deadline) {
571 return -1;
572 }
573 brix_backoff_sleep_fast(attempt++);
574 }
575 continue;
576 }
577
578 /* phase-42 W5: a compression-negotiated write handle compresses each
579 * payload as a self-contained frame; the server decompresses on ingest.
580 * The request offset stays the PLAINTEXT offset. */
581 const void *payload = buf;
582 size_t plen = len;
583 uint8_t *frame = NULL;
584 if (wcodec != 0 && len > 0) {
585 size_t flen = 0;
586 frame = brix_deflate_frame(wcodec, buf, len, &flen, st);
587 if (frame == NULL) {
588 return -1; /* compress failure is not a transport fault */
589 }
590 payload = frame;
591 plen = flen;
592 }
593
594 ClientWriteRequest req;
595 memset(&req, 0, sizeof(req));
596 req.requestid = htons(kXR_write);
597 {
598 xrdw_write_req_t b = { .offset = off, .pathid = 0 };
599 memcpy(b.fhandle, fh, XRD_FHANDLE_LEN);
600 xrdw_write_req_pack(&b, ((ClientRequestHdr *) &req)->body);
601 }
602
603 uint16_t kxr = 0;
604 brix_aio_opts o = { mf->max_stall_ms, 0, 0 };
605 int rc = brix_aio_call_ex(mf->ac, &req, payload, (uint32_t) plen, &o,
606 &kxr, NULL, NULL, st);
607 free(frame);
608 if (rc == 0) {
609 return 0;
610 }
611 if (!mfile_should_reopen(st) || now_ns() >= deadline) {
612 return -1;
613 }
614 brix_backoff_sleep_fast(attempt++);
615 (void) mfile_reopen(mf, gen, st);
616 }
617}

619int
620brix_mfile_sync(brix_mfile *mf, brix_status *st)
621{
622 uint8_t fh[XRD_FHANDLE_LEN] = { 0 };
623 uint64_t gen;
624 if (!mfile_snapshot(mf, fh, &gen, NULL, NULL)) {
625 if (mfile_reopen(mf, gen, st) != 0) {
626 return -1;
627 }
628 (void) mfile_snapshot(mf, fh, &gen, NULL, NULL);
629 }
630 ClientSyncRequest req;
631 memset(&req, 0, sizeof(req));
632 req.requestid = htons(kXR_sync);
633 {
634 xrdw_sync_req_t b;
635 memcpy(b.fhandle, fh, XRD_FHANDLE_LEN);
636 xrdw_sync_req_pack(&b, ((ClientRequestHdr *) &req)->body);
637 }
638 uint16_t kxr = 0;
639 /* A sync after a reopen is meaningless (fresh handle), but re-issuing is
640 * harmless; do not loop — a failed sync is reported, not retried forever. */
641 return brix_aio_call(mf->ac, &req, NULL, 0, &kxr, NULL, NULL,
642 mf->max_stall_ms, st);
643}

645int
646brix_mfile_close(brix_mfile *mf, brix_status *st)
647{
648 int rc = 0;
649 if (mf->have_handle) {
650 ClientCloseRequest req;
651 memset(&req, 0, sizeof(req));
652 req.requestid = htons(kXR_close);
653 {
654 xrdw_close_req_t b;
655 memcpy(b.fhandle, mf->fhandle, XRD_FHANDLE_LEN);
656 xrdw_close_req_pack(&b, ((ClientRequestHdr *) &req)->body);
657 }
658 uint16_t kxr = 0;
659 /* Best-effort: if the conn is down the server reaps the handle on
660 * disconnect anyway, so a close failure is not fatal. */
661 rc = brix_aio_call(mf->ac, &req, NULL, 0, &kxr, NULL, NULL, 5000, st);
662 }
663 pthread_mutex_destroy(&mf->lock);
664 free(mf);
665 return rc;
666}
