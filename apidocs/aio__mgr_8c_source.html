1/*
2 * aio_mgr.c — connection manager + transparent file-handle resumption (M3).
3 *
4 * WHAT: Two layers on top of the async loop (aio.c):
5 * - brix_mgr: the loop plus a small pool of attached connections. Metadata
6 * requests round-robin across them; idempotent ones survive a reconnect
7 * transparently (retry_safe at the transport layer, M2).
8 * - brix_mfile: an open file that survives a connection drop. Because an
9 * XRootD file handle is valid only on the session that opened it, a
10 * reconnect invalidates it; so on a transport failure or a stale-handle
11 * error this layer REOPENS the file (fresh handle, NON-destructively — no
12 * re-truncate, no create-excl) and RE-ISSUES the read/write at the same
13 * absolute offset. Re-issuing the identical offset is idempotent, so a
14 * mid-transfer cat/dd survives a server restart with no data loss and no
15 * EIO reaching the caller.
16 * WHY: M2 makes the transport reconnect; M3 makes *open files* recover, which is
17 * what "survives a server bounce mid-transfer" actually requires.
18 * HOW: mfile is a blocking façade over the async loop: each pread/pwrite is one
19 * brix_aio_call, and the (many) FUSE worker threads calling concurrently
20 * pipeline over the shared connection. A per-file mutex + generation counter
21 * serialises reopen so concurrent callers reopen at most once and then reuse
22 * the fresh handle.
23 *
24 * Clean-room: the existing wire structs (ClientOpen/Read/Write/Close/SyncRequest)
25 * + the async loop. No XrdCl.
26 */

41struct brix_mgr {

43 int n; /* total stream slots */
44 brix_conn *conns; /* array[n], owned */
45 brix_aconn **acs; /* array[n]; acs[i]==NULL ⇒ slot i not yet connected */
46 int rr; /* round-robin cursor */
47 int max_stall_ms;

57/* One parallel-connect job: a worker thread runs brix_connect on its own slot so
58 * the eager streams' connect+TLS+login+auth round-trips overlap (mount-time wall
59 * collapses from eager×RTT to ~1×RTT). Threads are joined before mgr_create
60 * returns, so they never cross a later fuse daemonize fork. */
61typedef struct {
62 const brix_url *u;
63 const brix_opts *o;
64 brix_conn *conn; /* slot to fill */
65 brix_status st; /* per-thread status (no shared writes) */
66 int rc; /* 0 ok, -1 fail */
67} mgr_connect_job;

69static void *
70mgr_connect_worker(void *arg)
71{
72 mgr_connect_job *j = (mgr_connect_job *) arg;
73 j->rc = brix_connect(j->conn, j->u, j->o, &j->st);
74 return NULL;
75}

77/* Run `count` connect jobs concurrently and wait for all to finish. A thread that
78 * fails to spawn runs its job inline (degrades to serial for that one), so the
79 * caller always sees every job's rc/st populated on return. */
80static void
81mgr_connect_parallel(mgr_connect_job *jobs, int count)
82{
83 pthread_t *tids = (pthread_t *) calloc((size_t) count, sizeof(*tids));
84 int i;
85
86 for (i = 0; i < count; i++) {
87 if (tids == NULL
88 || pthread_create(&tids[i], NULL, mgr_connect_worker, &jobs[i]) != 0) {
89 if (tids != NULL) { tids[i] = (pthread_t) 0; }
90 mgr_connect_worker(&jobs[i]); /* inline fallback */
91 }
92 }
93 if (tids != NULL) {
94 for (i = 0; i < count; i++) {
95 if (tids[i] != (pthread_t) 0) { pthread_join(tids[i], NULL); }
96 }
97 free(tids);
98 }
99}

102static void
103mgr_free(brix_mgr *m)
104{
105 int i;
106 for (i = 0; i < m->n; i++) {
107 if (m->acs[i] != NULL) { brix_aconn_close(m->acs[i]); }
108 }
109 if (m->loop != NULL) { brix_loop_destroy(m->loop); }
110 for (i = 0; i < m->n; i++) {
111 if (m->acs[i] != NULL) { brix_close(&m->conns[i]); }
112 }

119/* Allocate the manager shell and bring the async loop up.
120 *
121 * WHAT: calloc the brix_mgr plus its per-slot conns[nconns]/acs[nconns] arrays,
122 * copy the retained url/opts + resilience knobs, init lazy_lock, and create
123 * the shared loop. Returns the ready (but streamless) manager, or NULL with
124 * *st populated on any allocation/loop-create failure.
125 * WHY: brix_mgr_create's own complexity comes from the eager connect/attach
126 * phases; pulling the fixed setup out keeps that orchestration flat and
127 * keeps every early-return cleanup local to the resource it owns.
128 * HOW: 1) calloc the shell (out-of-memory → status, return NULL).
129 * 2) calloc conns + acs; on either failure free both + shell, return NULL.
130 * 3) copy scalar fields + url; copy opts only when the caller passed some.
131 * 4) init lazy_lock, then create the loop; loop failure → mgr_free (acs all
132 * NULL ⇒ it frees loop+arrays) and return NULL. */
133static brix_mgr *
134mgr_alloc_and_init(const brix_url *u, const brix_opts *o, int nconns,
135 int max_stall_ms, int keepalive_ms, int max_retries,
136 brix_status *st)
137{
138 brix_mgr *m = (brix_mgr *) calloc(1, sizeof(*m));
139 if (m == NULL) {
140 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (mgr)");
141 return NULL;
142 }
143 m->conns = (brix_conn *) calloc((size_t) nconns, sizeof(*m->conns));
144 m->acs = (brix_aconn **) calloc((size_t) nconns, sizeof(*m->acs));
145 if (m->conns == NULL || m->acs == NULL) {
146 free(m->conns);
147 free(m->acs);
148 free(m);
149 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (mgr arrays)");
150 return NULL;
151 }
152 m->n = nconns;
153 m->max_stall_ms = max_stall_ms;
154 m->keepalive_ms = keepalive_ms;
155 m->max_retries = max_retries;
156 m->url = *u;
157 if (o != NULL) { m->opts = *o; m->have_opts = 1; }
158 pthread_mutex_init(&m->lazy_lock, NULL);
159
160 m->loop = brix_loop_create(st);
161 if (m->loop == NULL) {
162 mgr_free(m);
163 return NULL;
164 }
165 return m;
166}

168/* Connect the first `eager` streams concurrently and require all to succeed.
169 *
170 * WHAT: Runs `eager` parallel connect jobs against slots 0..eager-1; returns 0
171 * with every conns[i] connected, or -1 with *st set to the first failure
172 * and any slot that DID connect already closed. The acs[] entries stay NULL
173 * (attach happens later).
174 * WHY: Overlapping the connect+TLS+login+auth round-trips collapses mount-time
175 * wall from eager×RTT to ~1×RTT; failing the whole mount if any eager stream
176 * cannot connect gives a bad endpoint/auth an immediate, clean failure.
177 * HOW: 1) calloc the job array (OOM → status, return -1).
178 * 2) point each job at the retained url/opts and its slot, prime rc=-1.
179 * 3) mgr_connect_parallel spawns+joins the workers.
180 * 4) scan results; on the first rc!=0 copy its status, close the OTHER
181 * successfully-connected conns (this one isn't open), free jobs, -1.
182 * 5) all connected → free jobs, return 0. Caller closes conns via mgr_free
183 * only after acs are populated, so failed-phase conns are closed here. */
184static int
185mgr_connect_eager(brix_mgr *m, int eager, brix_status *st)
186{
187 mgr_connect_job *jobs = (mgr_connect_job *) calloc((size_t) eager,
188 sizeof(*jobs));
189 int i;
190
191 if (jobs == NULL) {
192 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (mgr jobs)");
193 return -1;
194 }
195 for (i = 0; i < eager; i++) {
196 jobs[i].u = &m->url; jobs[i].o = m->have_opts ? &m->opts : NULL;
197 jobs[i].conn = &m->conns[i]; jobs[i].rc = -1;
198 }
199 mgr_connect_parallel(jobs, eager);
200
201 for (i = 0; i < eager; i++) {
202 if (jobs[i].rc != 0) {
203 *st = jobs[i].st; /* surface the first real failure */
204 for (int k = 0; k < eager; k++) { /* close any that DID connect */
205 if (k != i && jobs[k].rc == 0) { brix_close(&m->conns[k]); }
206 }
207 free(jobs);
208 return -1;
209 }
210 }
211 free(jobs);
212 return 0;
213}

215/* Attach the eager (already-connected) streams to the loop and arm resilience.
216 *
217 * WHAT: For slots 0..eager-1, attach conns[i] to the loop into acs[i] and set its
218 * stall/keepalive/retry policy. Returns 0 on full success, or -1 with the
219 * just-failed conn and every still-unattached higher conn closed (their acs
220 * stay NULL, so a following mgr_free frees the rest).
221 * WHY: Attach touches the shared loop and is cheap, so it runs serially after the
222 * parallel connect; keeping it separate keeps brix_mgr_create flat.
223 * HOW: 1) walk i in [0, eager): brix_aconn_attach conns[i] → acs[i].
224 * 2) on NULL, close conns[i] (its attach failed) and every conns[i+1..]
225 * (never attached, so mgr_free won't reach them) and return -1.
226 * 3) otherwise arm resilience and continue; return 0 when all attached. */
227static int
228mgr_attach_eager(brix_mgr *m, int eager, brix_status *st)
229{
230 int i;
231 for (i = 0; i < eager; i++) {
232 m->acs[i] = brix_aconn_attach(m->loop, &m->conns[i], st);
233 if (m->acs[i] == NULL) {
234 brix_close(&m->conns[i]); /* this one attached failed */
235 for (int k = i + 1; k < eager; k++) { brix_close(&m->conns[k]); }
236 return -1;
237 }
238 brix_aconn_set_resilience(m->acs[i], m->max_stall_ms, m->keepalive_ms,
239 m->max_retries);
240 }
241 return 0;
242}

244brix_mgr *
245brix_mgr_create(const brix_url *u, const brix_opts *o, int nconns, int eager,
246 int max_stall_ms, int keepalive_ms, int max_retries,
247 brix_status *st)
248{
249 brix_mgr *m;
250
251 if (nconns < 1) {
252 nconns = 1;
253 }
254 /* At least one stream connects up front so a bad endpoint / auth fails the
255 * mount immediately; the remainder may be eager or lazy per the caller. */
256 if (eager < 1) { eager = 1; }
257 if (eager > nconns) { eager = nconns; }
258
259 m = mgr_alloc_and_init(u, o, nconns, max_stall_ms, keepalive_ms, max_retries,
260 st);
261 if (m == NULL) {
262 return NULL;
263 }
264
265 /* Connect the eager streams concurrently, then attach them to the loop
266 * serially (attach touches the shared loop and is not the slow part). */
267 if (mgr_connect_eager(m, eager, st) != 0) {
268 mgr_free(m); /* acs all NULL ⇒ frees loop+arrays */
269 return NULL;
270 }
271 if (mgr_attach_eager(m, eager, st) != 0) {
272 mgr_free(m);
273 return NULL;
274 }
275 return m;
276}

287/* Bring a lazily-deferred slot up on first use: connect + attach + arm
288 * resilience under lazy_lock (double-checked so concurrent pickers connect it
289 * at most once). On failure the slot stays NULL and the caller falls back to an
290 * already-live stream — eager ≥ 1 guarantees one exists. */
291static brix_aconn *
292mgr_ensure_slot(brix_mgr *m, int i)
293{
294 brix_aconn *ac = __atomic_load_n(&m->acs[i], __ATOMIC_ACQUIRE);
295 brix_status st;
296
297 if (ac != NULL) {
298 return ac;
299 }
300 pthread_mutex_lock(&m->lazy_lock);
301 ac = m->acs[i]; /* re-check under the lock */
302 if (ac == NULL) {
303 if (brix_connect(&m->conns[i], &m->url,
304 m->have_opts ? &m->opts : NULL, &st) == 0) {
305 ac = brix_aconn_attach(m->loop, &m->conns[i], &st);
306 if (ac != NULL) {
307 brix_aconn_set_resilience(ac, m->max_stall_ms, m->keepalive_ms,
308 m->max_retries);
309 __atomic_store_n(&m->acs[i], ac, __ATOMIC_RELEASE);
310 } else {
311 brix_close(&m->conns[i]);
312 }
313 }
314 }
315 pthread_mutex_unlock(&m->lazy_lock);
316 return ac;
317}

319brix_aconn *
320brix_mgr_pick(brix_mgr *m)
321{
322 int i = __atomic_fetch_add(&m->rr, 1, __ATOMIC_RELAXED);
323 brix_aconn *ac;
324 int k;
325
326 if (i < 0) {
327 i = -i;
328 }
329 i %= m->n;
330
331 ac = mgr_ensure_slot(m, i);
332 if (ac != NULL) {
333 return ac;
334 }
335 /* Lazy connect of slot i failed (server hiccup): fall back to any live
336 * stream rather than returning NULL — at least the eager slot(s) are up. */
337 for (k = 0; k < m->n; k++) {
338 ac = __atomic_load_n(&m->acs[k], __ATOMIC_ACQUIRE);
339 if (ac != NULL) {
340 return ac;
341 }
342 }
343 return NULL;
344}

387/* Compute the kXR_open options word for a (re)open with the given force tri-state.
388 *
389 * WHAT: Returns the wire options bitmask — kXR_open_read for a read handle, or
390 * kXR_open_updt|kXR_mkpath plus (force==1) kXR_delete / (force==0) kXR_new /
391 * (force==2) nothing, plus kXR_posc when POSC is set — for a write handle.
392 * WHY: The force tri-state (truncate vs create-excl vs update-in-place) is exactly
393 * what makes reopen NON-destructive: reopen passes force==2 so recovery never
394 * re-truncates or re-creates. Isolating it keeps mfile_do_open flat.
395 * HOW: 1) read handle → kXR_open_read, done.
396 * 2) write handle → base updt|mkpath; OR in delete/new per force (force==2
397 * adds neither = update in place); OR in posc when requested. */
398static uint16_t
399mfile_open_options(const brix_mfile *mf, int force)
400{
401 uint16_t options;
402
403 if (!mf->writable) {
404 return kXR_open_read;
405 }
406 options = kXR_open_updt | kXR_mkpath;
407 if (force == 1) {
408 options |= kXR_delete; /* truncate/overwrite */
409 } else if (force == 0) {
410 options |= kXR_new; /* create-excl */
411 } /* force == 2: update in place */
412 if (mf->posc) {
413 options |= kXR_posc;
414 }
415 return options;
416}

418/* Learn the inline-compression codec from a kXR_open reply body.
419 *
420 * WHAT: Resets mf->read_codec/write_codec to 0, then (write handle → write_codec,
421 * read handle → read_codec) adopts the server's codec id when the reply
422 * confirms it. Returns 0 on success (including "no codec"), or -1 with *st
423 * set when the server negotiated a codec this build cannot decode.
424 * WHY: phase-42 W4/W5. Per the dual-check contract (codec_core.h) cptype is a
425 * legacy field a non-cooperating server may reuse; trusting it alone would
426 * inflate plaintext and corrupt data, so it is adopted ONLY alongside the
427 * cpsize magic. Failing an un-decodable codec avoids asymmetric-build
428 * corruption (copying compressed bytes through as plaintext). Re-learned on
429 * every (re)open, so a fault mid-transfer transparently re-negotiates.
430 * HOW: 1) clear both codecs (stock servers leave cpsize 0 → stays plaintext).
431 * 2) ServerOpenBody is fhandle[4] cpsize[4] cptype[4]; a body under 12 bytes
432 * carries no codec fields → return 0.
433 * 3) parse big-endian cpsize and cptype[0]; unless cpsize == the magic and
434 * the id is in [1, BRIX_CODEC_MAX) → return 0 (plaintext).
435 * 4) if this build cannot decode the id → status + return -1.
436 * 5) store it in the write- or read-side codec per mf->writable, return 0. */
437static int
438mfile_learn_codec(brix_mfile *mf, const uint8_t *body, uint32_t blen,
439 brix_status *st)
440{
441 mf->read_codec = 0;
442 mf->write_codec = 0;
443 if (blen < 12) {
444 return 0;
445 }
446 uint32_t cpsize = ((uint32_t) body[4] << 24) | ((uint32_t) body[5] << 16)
447 | ((uint32_t) body[6] << 8) | (uint32_t) body[7];
448 uint8_t cid = body[8]; /* cptype[0] */
449 if (cpsize != BRIX_INLINE_CMP_MAGIC || cid < 1 || cid >= BRIX_CODEC_MAX) {
450 return 0;
451 }
452 /* Server confirmed codec `cid`; if this build cannot handle it, fail the open
453 * rather than silently copying compressed bytes as plaintext. */
454 if (!brix_codec_available(cid)) {
455 brix_status_set(st, XRDC_EUNSUPPORTED, 0,
456 "server negotiated inline-compression codec %u that this "
457 "client build cannot decode", (unsigned) cid);
458 return -1;
459 }
460 if (mf->writable) {
461 mf->write_codec = cid;
462 } else {
463 mf->read_codec = cid;
464 }
465 return 0;
466}

468/* Build + send a kXR_open with the given force tri-state (see brix_file_open_opaque).
469 * On success copies the fresh handle into mf->fhandle. Caller holds no lock. */
470static int
471mfile_do_open(brix_mfile *mf, int force, brix_status *st)
472{
473 ClientOpenRequest req;
474 uint16_t options = mfile_open_options(mf, force);
475 uint16_t kxr = 0;
476 uint8_t *body = NULL;
477 uint32_t blen = 0;
478 char *payload;
479 size_t need;
480 int plen;
481
482 need = strlen(mf->path) + 1 + (mf->have_opaque ? strlen(mf->opaque) + 1 : 0);
483 payload = (char *) malloc(need);
484 if (payload == NULL) {
485 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (open payload)");
486 return -1;
487 }
488 plen = mf->have_opaque
489 ? snprintf(payload, need, "%s?%s", mf->path, mf->opaque)
490 : snprintf(payload, need, "%s", mf->path);
491
492 memset(&req, 0, sizeof(req));
493 req.requestid = htons(kXR_open);
494 {
495 xrdw_open_req_t b = { .mode = (uint16_t) (mf->writable ? 0644 : 0),
496 .options = options, .optiont = 0 };
497 xrdw_open_req_pack(&b, ((ClientRequestHdr *) &req)->body);
498 }
499
500 brix_aio_opts o = { mf->max_stall_ms, mf->max_retries, 1 /*open is idempotent*/ };
501 int rc = brix_aio_call_ex(mf->ac, &req, payload, (uint32_t) plen, &o,
502 &kxr, &body, &blen, st);
503 free(payload);
504 if (rc != 0) {
505 return -1;
506 }
507 if (blen < XRD_FHANDLE_LEN) {
508 brix_status_set(st, XRDC_EPROTO, 0, "open reply too short (%u bytes)", blen);
509 free(body);
510 return -1;
511 }
512 memcpy(mf->fhandle, body, XRD_FHANDLE_LEN);
513
514 /* phase-42 W4/W5: learn the inline-compression codec from the open reply.
515 * Re-learned on every (re)open, so a fault mid-transfer re-negotiates. */
516 rc = mfile_learn_codec(mf, body, blen, st);
517 free(body);
518 return rc;
519}

521brix_mfile *
522brix_mfile_open(brix_aconn *ac, const char *path, int writable, int force,
523 int posc, const char *opaque, int max_stall_ms, int max_retries,
524 brix_status *st)
525{
526 brix_mfile *mf = (brix_mfile *) calloc(1, sizeof(*mf));
527 if (mf == NULL) {
528 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (mfile)");
529 return NULL;
530 }
531 mf->ac = ac;
532 snprintf(mf->path, sizeof(mf->path), "%s", path);
533 if (opaque != NULL && opaque[0] != '\0') {
534 snprintf(mf->opaque, sizeof(mf->opaque), "%s", opaque);
535 mf->have_opaque = 1;
536 }
537 mf->writable = writable;
538 mf->posc = posc;
539 mf->max_stall_ms = max_stall_ms;
540 mf->max_retries = max_retries;
541 pthread_mutex_init(&mf->lock, NULL);
542
543 if (mfile_do_open(mf, force, st) != 0) {
544 pthread_mutex_destroy(&mf->lock);
545 free(mf);
546 return NULL;
547 }
548 mf->have_handle = 1;
549 mf->gen = 1;
550 return mf;
551}

561/* Reopen the file NON-destructively if no one else already did (generation match).
562 * Returns 0 if the file is open (by us or a racing caller), -1 on failure. */
563static int
564mfile_reopen(brix_mfile *mf, uint64_t expected_gen, brix_status *st)
565{
566 pthread_mutex_lock(&mf->lock);
567 if (mf->gen != expected_gen && mf->have_handle) {
568 pthread_mutex_unlock(&mf->lock);
569 return 0; /* someone else reopened while we waited */
570 }
571 mf->have_handle = 0;
572 int rc = mfile_do_open(mf, mf->writable ? 2 /*update, no truncate*/ : -1, st);
573 if (rc == 0) {
574 mf->have_handle = 1;
575 mf->gen++;
576 }
577 pthread_mutex_unlock(&mf->lock);
578 return rc;
579}

581/* Snapshot the current handle + generation (+ optional read/write codecs) under
582 * the lock. rcodec/wcodec may be NULL for callers that don't need them. */
583static int
584mfile_snapshot(brix_mfile *mf, uint8_t fh[XRD_FHANDLE_LEN], uint64_t *gen,
585 uint8_t *rcodec, uint8_t *wcodec)
586{
587 pthread_mutex_lock(&mf->lock);
588 int ok = mf->have_handle;
589 if (ok) {
590 memcpy(fh, mf->fhandle, XRD_FHANDLE_LEN);
591 }
592 if (rcodec != NULL) {
593 *rcodec = mf->read_codec;
594 }
595 if (wcodec != NULL) {
596 *wcodec = mf->write_codec;
597 }
598 *gen = mf->gen;
599 pthread_mutex_unlock(&mf->lock);
600 return ok;
601}

603ssize_t
604brix_mfile_pread(brix_mfile *mf, int64_t off, void *buf, size_t len,
605 brix_status *st)
606{
607 /* Retry transport faults until the max_stall patience window elapses, with
608 * exponential backoff between attempts — NOT a fixed small count. Under
609 * sustained loss (a flapping firewall) a fixed count is exhausted in a tight
610 * spin long before the patience window, surfacing a spurious EIO; bounding by
611 * the deadline rides the loss out as long as progress is possible. */
612 uint64_t deadline = now_ns() + (uint64_t) mf->max_stall_ms * 1000000ULL;
613 unsigned attempt = 0;
614
615 for (;;) {
616 uint8_t fh[XRD_FHANDLE_LEN];
617 uint64_t gen;
618 uint8_t codec = 0;
619 if (!mfile_snapshot(mf, fh, &gen, &codec, NULL)) {
620 if (mfile_reopen(mf, gen, st) != 0) {
621 if (now_ns() >= deadline) {
622 return -1;
623 }
624 brix_backoff_sleep_fast(attempt++);
625 }
626 continue;
627 }
628
629 ClientReadRequest req;
630 memset(&req, 0, sizeof(req));
631 req.requestid = htons(kXR_read);
632 {
633 xrdw_read_req_t b = { .offset = off, .rlen = (int32_t) len };
634 memcpy(b.fhandle, fh, XRD_FHANDLE_LEN);
635 xrdw_read_req_pack(&b, ((ClientRequestHdr *) &req)->body);
636 }
637
638 uint16_t kxr = 0;
639 uint8_t *body = NULL;
640 uint32_t blen = 0;
641 brix_aio_opts o = { mf->max_stall_ms, 0, 0 /*we own reopen/retry*/ };
642 if (brix_aio_call_ex(mf->ac, &req, NULL, 0, &o, &kxr, &body, &blen, st) == 0) {
643 /* phase-42 W4: a compressed handle returns one self-contained codec
644 * frame per request (aio_call_ex has already accumulated all
645 * oksofar frames into `body`); inflate it into the caller's buffer.
646 * A corrupt frame is not a transport fault — do not reopen/retry. */
647 if (codec != 0) {
648 ssize_t pl = brix_inflate_frame(codec, body, blen, buf, len, st);
649 free(body);
650 return pl;
651 }
652 size_t n = (blen < len) ? blen : len;
653 if (n > 0) {
654 memcpy(buf, body, n);
655 }
656 free(body);
657 return (ssize_t) n;
658 }
659 if (!mfile_should_reopen(st) || now_ns() >= deadline) {
660 return -1;
661 }
662 brix_backoff_sleep_fast(attempt++); /* don't tight-spin under loss */
663 (void) mfile_reopen(mf, gen, st); /* refresh handle, then retry */
664 }
665}

667int
668brix_mfile_pwrite(brix_mfile *mf, int64_t off, const void *buf, size_t len,
669 brix_status *st)
670{
671 /* Deadline-bounded retry with backoff (see brix_mfile_pread). */
672 uint64_t deadline = now_ns() + (uint64_t) mf->max_stall_ms * 1000000ULL;
673 unsigned attempt = 0;
674
675 for (;;) {
676 uint8_t fh[XRD_FHANDLE_LEN];
677 uint64_t gen;
678 uint8_t wcodec = 0;
679 if (!mfile_snapshot(mf, fh, &gen, NULL, &wcodec)) {
680 if (mfile_reopen(mf, gen, st) != 0) {
681 if (now_ns() >= deadline) {
682 return -1;
683 }
684 brix_backoff_sleep_fast(attempt++);
685 }
686 continue;
687 }
688
689 /* phase-42 W5: a compression-negotiated write handle compresses each
690 * payload as a self-contained frame; the server decompresses on ingest.
691 * The request offset stays the PLAINTEXT offset. */
692 const void *payload = buf;
693 size_t plen = len;
694 uint8_t *frame = NULL;
695 if (wcodec != 0 && len > 0) {
696 size_t flen = 0;
697 frame = brix_deflate_frame(wcodec, buf, len, &flen, st);
698 if (frame == NULL) {
699 return -1; /* compress failure is not a transport fault */
700 }
701 payload = frame;
702 plen = flen;
703 }
704
705 ClientWriteRequest req;
706 memset(&req, 0, sizeof(req));
707 req.requestid = htons(kXR_write);
708 {
709 xrdw_write_req_t b = { .offset = off, .pathid = 0 };
710 memcpy(b.fhandle, fh, XRD_FHANDLE_LEN);
711 xrdw_write_req_pack(&b, ((ClientRequestHdr *) &req)->body);
712 }
713
714 uint16_t kxr = 0;
715 brix_aio_opts o = { mf->max_stall_ms, 0, 0 };
716 int rc = brix_aio_call_ex(mf->ac, &req, payload, (uint32_t) plen, &o,
717 &kxr, NULL, NULL, st);
718 free(frame);
719 if (rc == 0) {
720 return 0;
721 }
722 if (!mfile_should_reopen(st) || now_ns() >= deadline) {
723 return -1;
724 }
725 brix_backoff_sleep_fast(attempt++);
726 (void) mfile_reopen(mf, gen, st);
727 }
728}

730int
731brix_mfile_sync(brix_mfile *mf, brix_status *st)
732{
733 uint8_t fh[XRD_FHANDLE_LEN] = { 0 };
734 uint64_t gen;
735 if (!mfile_snapshot(mf, fh, &gen, NULL, NULL)) {
736 if (mfile_reopen(mf, gen, st) != 0) {
737 return -1;
738 }
739 (void) mfile_snapshot(mf, fh, &gen, NULL, NULL);
740 }
741 ClientSyncRequest req;
742 memset(&req, 0, sizeof(req));
743 req.requestid = htons(kXR_sync);
744 {
745 xrdw_sync_req_t b;
746 memcpy(b.fhandle, fh, XRD_FHANDLE_LEN);
747 xrdw_sync_req_pack(&b, ((ClientRequestHdr *) &req)->body);
748 }
749 uint16_t kxr = 0;
750 /* A sync after a reopen is meaningless (fresh handle), but re-issuing is
751 * harmless; do not loop — a failed sync is reported, not retried forever. */
752 return brix_aio_call(mf->ac, &req, NULL, 0, &kxr, NULL, NULL,
753 mf->max_stall_ms, st);
754}

756int
757brix_mfile_close(brix_mfile *mf, brix_status *st)
758{
759 int rc = 0;
760 if (mf->have_handle) {
761 ClientCloseRequest req;
762 memset(&req, 0, sizeof(req));
763 req.requestid = htons(kXR_close);
764 {
765 xrdw_close_req_t b;
766 memcpy(b.fhandle, mf->fhandle, XRD_FHANDLE_LEN);
767 xrdw_close_req_pack(&b, ((ClientRequestHdr *) &req)->body);
768 }
769 uint16_t kxr = 0;
770 /* Best-effort: if the conn is down the server reaps the handle on
771 * disconnect anyway, so a close failure is not fatal. */
772 rc = brix_aio_call(mf->ac, &req, NULL, 0, &kxr, NULL, NULL, 5000, st);
773 }
774 pthread_mutex_destroy(&mf->lock);
775 free(mf);
776 return rc;
777}
