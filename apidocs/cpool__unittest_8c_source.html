88/* N+1 threads contend for N slots: assert no conn pointer is ever held by two
89 * threads at once (unique-issue), and the (N+1)th blocks then proceeds. */
90#define NSLOT 3
91#define NTHR (NSLOT + 1)
92static brix_cpool *g_p;
93static atomic_int g_inflight[64]; /* per-slot-id in-use flag */
94static void *
95worker(void *arg)
96{
97 int rounds = *(int *) arg, r;
98 brix_status st; brix_status_clear(&st);
99 for (r = 0; r < rounds; r++) {
100 fake_conn *c = brix_cpool_checkout(g_p, &st);
101 assert(c != NULL);
102 int prev = atomic_fetch_add(&g_inflight[c->id], 1);
103 assert(prev == 0); /* NOT double-issued */
104 atomic_fetch_sub(&g_inflight[c->id], 1);
105 brix_cpool_checkin(g_p, c, 1);
106 }
107 return NULL;
108}
109static void
110test_contention_unique_issue(void)
111{
112 fake_ctx ctx = {0};
113 brix_status st; brix_status_clear(&st);
114 pthread_t th[NTHR];
115 int rounds = 500, i;
116 memset(g_inflight, 0, sizeof(g_inflight));
117 g_p = brix_cpool_create(&VT, &ctx, NSLOT, &st);
118 assert(g_p != NULL);
119 for (i = 0; i < NTHR; i++) pthread_create(&th[i], NULL, worker, &rounds);
120 for (i = 0; i < NTHR; i++) pthread_join(th[i], NULL);
121 assert(atomic_load(&ctx.connects) <= NSLOT); /* never more than N */
122 brix_cpool_destroy(g_p);
123 printf("ok test_contention_unique_issue\n");
124}
