1/*
2 * sd_ceph_object_rename.c — Ceph/RADOS driver: the rename vtable slot.
3 *
4 * WHAT: The copy+delete rename path (phase-89 §B.2 / ADR-5) split out of
5 * sd_ceph_object.c (file-size guard): the bare-oid existence/layout probe,
6 * the layout-aware object remove, the chunked byte copy, the striped and
7 * flat xattr-copy legs, the absent-source classifier, the destination
8 * prep, the copy+size-verify, and the sd_ceph_rename orchestrator.
9 *
10 * WHY: Rename is the single largest self-contained concern in the object TU —
11 * a non-atomic copy+delete with its own probe/remove/copy machinery that
12 * nothing else in the driver shares. Isolating it keeps every file under
13 * the source-size guard while preserving byte-for-byte behaviour.
14 *
15 * HOW: The private struct definitions and the cross-TU helper/op declarations
16 * come from sd_ceph_internal.h (the child-prefix probe callback it reuses
17 * is defined in sd_ceph_object.c and declared there). The whole body is
18 * gated on BRIX_HAVE_CEPH, exactly like the driver it was split from.
19 */

39/* sd_ceph_path_probe_t / sd_ceph_probe_oid — existence/layout probe for a
40 * bare oid (no open request): striper view first (a striped object must be
41 * copied through the striper to reassemble), then flat. 0 on success (present
42 * or cleanly absent), negative errno on a hard stat error. */
43typedef struct {
44 int present;
45 int striped;
46 uint64_t size;
47} sd_ceph_path_probe_t;

49static int
50sd_ceph_probe_oid(sd_ceph_state_t *st, const char *oid,
51 sd_ceph_path_probe_t *pr)
52{
53 uint64_t size = 0;
54 time_t mtime = 0;
55 int rc;
56
57 ngx_memzero(pr, sizeof(*pr));
58
59#if defined(BRIX_HAVE_RADOSSTRIPER)
60 {
61 rados_striper_t s = sd_ceph_striper(st);
62
63 if (s != NULL && sd_ceph_striper_stat(s, oid, &size, &mtime) == 0) {
64 pr->present = 1;
65 pr->striped = 1;
66 pr->size = size;
67 return 0;
68 }
69 }
70#endif
71 rc = rados_stat(st->ioctx, oid, &size, &mtime);
72 if (rc == 0) {
73 pr->present = 1;
74 pr->size = size;
75 return 0;
76 }
77 return (rc == -ENOENT) ? 0 : rc;
78}

96/* sd_ceph_copy_bytes — chunked byte copy src → dst. A striped source is read
97 * AND written through the striper so the destination keeps the stock-XrdCeph
98 * layout; a flat source copies flat. An empty source is created flat (the
99 * caller forces striped=0 for size 0 — a zero-length striper write would not
100 * materialize the first stripe). 0 or negative errno. */
101static int
102sd_ceph_copy_bytes(sd_ceph_state_t *st, const char *src, int striped,
103 uint64_t size, const char *dst)
104{
105 char *buf;
106 uint64_t off = 0;
107
108#if !defined(BRIX_HAVE_RADOSSTRIPER)
109 (void) striped;
110#endif
111
112 if (size == 0) {
113 return rados_write_full(st->ioctx, dst, "", 0);
114 }
115
116 buf = malloc(SD_CEPH_COPY_CHUNK);
117 if (buf == NULL) {
118 return -ENOMEM;
119 }
120
121 while (off < size) {
122 size_t want = (size - off < SD_CEPH_COPY_CHUNK)
123 ? (size_t) (size - off) : SD_CEPH_COPY_CHUNK;
124 ssize_t n;
125 ssize_t wrc;
126
127#if defined(BRIX_HAVE_RADOSSTRIPER)
128 if (striped) {
129 rados_striper_t s = sd_ceph_striper(st);
130
131 if (s == NULL) {
132 free(buf);
133 return -EIO;
134 }
135 n = sd_ceph_striper_read(s, src, buf, want, off);
136 if (n > 0) {
137 wrc = sd_ceph_striper_write(s, dst, buf, (size_t) n, off);
138 } else {
139 wrc = 0;
140 }
141 } else
142#endif
143 {
144 n = rados_read(st->ioctx, src, buf, want, off);
145 wrc = (n > 0) ? rados_write(st->ioctx, dst, buf, (size_t) n, off)
146 : 0;
147 }
148
149 if (n < 0) {
150 free(buf);
151 return (int) n;
152 }
153 if (n == 0) {
154 free(buf);
155 return -EIO; /* source shrank mid-copy */
156 }
157 if (wrc < 0) {
158 free(buf);
159 return (int) wrc;
160 }
161 off += (uint64_t) n;
162 }
163
164 free(buf);
165 return 0;
166}

169/* sd_ceph_copy_xattrs_striped — the striped-source leg of the xattr copy:
170 * list the striper object's names, then get/set each one. Striper-internal
171 * "striper.*" layout attrs are never copied — the destination's own write path
172 * stamps its layout. 0 or negative errno. */
173static int
174sd_ceph_copy_xattrs_striped(sd_ceph_state_t *st, const char *src,
175 const char *dst, int dst_striped)
176{
177 rados_striper_t s = sd_ceph_striper(st);
178 char *names;
179 char *val;
180 ssize_t total;
181 ssize_t used = 0;
182 int rc = 0;
183
184 if (s == NULL) {
185 return -EIO;
186 }
187 names = malloc(SD_CEPH_XATTR_MAX);
188 val = malloc(SD_CEPH_XATTR_MAX);
189 if (names == NULL || val == NULL) {
190 free(names);
191 free(val);
192 return -ENOMEM;
193 }
194 total = sd_ceph_striper_listxattr(s, src, names, SD_CEPH_XATTR_MAX);
195 if (total < 0) {
196 free(names);
197 free(val);
198 return (int) total;
199 }
200 while (rc == 0 && used < total) {
201 const char *nm = names + used;
202 ssize_t vlen;
203
204 used += (ssize_t) strlen(nm) + 1;
205 if (strncmp(nm, "striper.", 8) == 0) {
206 continue; /* layout attrs: never copied */
207 }
208 vlen = sd_ceph_striper_getxattr(s, src, nm, val,
209 SD_CEPH_XATTR_MAX);
210 if (vlen < 0) {
211 rc = (int) vlen;
212 break;
213 }
214 rc = dst_striped
215 ? sd_ceph_striper_setxattr(s, dst, nm, val, (size_t) vlen)
216 : rados_setxattr(st->ioctx, dst, nm, val, (size_t) vlen);
217 }
218 free(names);
219 free(val);
220 return rc;
221}

224/* sd_ceph_copy_xattrs_plain — the flat-object leg of the xattr copy: walk the
225 * source's xattr iterator and stamp each pair onto the destination (striper
226 * or flat, per dst_striped). 0 or negative errno. */
227static int
228sd_ceph_copy_xattrs_plain(sd_ceph_state_t *st, const char *src,
229 const char *dst, int dst_striped)
230{
231 rados_xattrs_iter_t it;
232 int rc;
233
234 rc = rados_getxattrs(st->ioctx, src, &it);
235 if (rc < 0) {
236 return rc;
237 }
238 for (;;) {
239 const char *nm = NULL;
240 const char *vv = NULL;
241 size_t vlen = 0;
242
243 rc = rados_getxattrs_next(it, &nm, &vv, &vlen);
244 if (rc < 0 || nm == NULL) {
245 break;
246 }
247#if defined(BRIX_HAVE_RADOSSTRIPER)
248 if (dst_striped) {
249 rados_striper_t s = sd_ceph_striper(st);
250
251 rc = (s != NULL)
252 ? sd_ceph_striper_setxattr(s, dst, nm, vv, vlen) : -EIO;
253 } else
254#endif
255 {
256 rc = rados_setxattr(st->ioctx, dst, nm, vv, vlen);
257 }
258 if (rc < 0) {
259 break;
260 }
261 }
262 rados_getxattrs_end(it);
263#if !defined(BRIX_HAVE_RADOSSTRIPER)
264 (void) dst_striped;
265#endif
266 return (rc < 0) ? rc : 0;
267}

269/* sd_ceph_copy_xattrs — carry the object xattrs across the copy (the cache
270 * cinfo/meta/checksum-at-rest records must survive a rename). Dispatches on
271 * the source's layout to the striped or plain leg above. Values are bounded
272 * by SD_CEPH_XATTR_MAX (the driver's existing xattr value bound). 0 or
273 * negative errno. */
274static int
275sd_ceph_copy_xattrs(sd_ceph_state_t *st, const char *src, int src_striped,
276 const char *dst, int dst_striped)
277{
278#if defined(BRIX_HAVE_RADOSSTRIPER)
279 if (src_striped) {
280 return sd_ceph_copy_xattrs_striped(st, src, dst, dst_striped);
281 }
282#else
283 (void) src_striped;
284#endif
285 return sd_ceph_copy_xattrs_plain(st, src, dst, dst_striped);
286}

288/* sd_ceph_rename_missing_src — classify a rename whose source object is
289 * absent. A populated synthetic-directory source is EISDIR (no collection
290 * rename on a flat namespace), a bare miss is ENOENT. Always NGX_ERROR with
291 * errno set. */
292static ngx_int_t
293sd_ceph_rename_missing_src(brix_sd_instance_t *inst, const char *src)
294{
295 char norm[1024];
296 sd_ceph_child_probe_t c;
297
298 errno = ENOENT;
299 if (sd_ceph_normalize(src, norm, sizeof(norm)) == 0) {
300 c.dir = norm;
301 c.found = 0;
302 if (sd_ceph_enumerate(inst, 0, sd_ceph_child_probe_cb, &c)
303 == NGX_OK && c.found)
304 {
305 errno = EISDIR;
306 }
307 }
308 return NGX_ERROR;
309}

311/* sd_ceph_rename_prep_dst — probe the destination oid and clear the way:
312 * honour noreplace (EEXIST) and remove a pre-existing destination object.
313 * NGX_OK or NGX_ERROR with errno set. */
314static ngx_int_t
315sd_ceph_rename_prep_dst(sd_ceph_state_t *st, const char *dm, int noreplace)
316{
317 sd_ceph_path_probe_t dp;
318 int rc;
319
320 rc = sd_ceph_probe_oid(st, dm, &dp);
321 if (rc < 0) {
322 errno = -rc;
323 return NGX_ERROR;
324 }
325 if (dp.present && noreplace) {
326 errno = EEXIST;
327 return NGX_ERROR;
328 }
329 if (dp.present) {
330 rc = sd_ceph_remove_oid(st, dm, dp.striped);
331 if (rc < 0) {
332 errno = -rc;
333 return NGX_ERROR;
334 }
335 }
336 return NGX_OK;
337}

339/* sd_ceph_rename_copy_verify — land the destination's bytes AND xattrs, then
340 * size-verify it against the source probe. 0 or negative errno (the caller
341 * cleans up the partial destination on failure). */
342static int
343sd_ceph_rename_copy_verify(sd_ceph_state_t *st, const char *so,
344 const char *dm, const sd_ceph_path_probe_t *sp, int dst_striped)
345{
346 int rc;
347
348 rc = sd_ceph_copy_bytes(st, so, dst_striped, sp->size, dm);
349 if (rc == 0) {
350 rc = sd_ceph_copy_xattrs(st, so, sp->striped, dm, dst_striped);
351 }
352 if (rc == 0) {
353 sd_ceph_path_probe_t vp;
354
355 rc = sd_ceph_probe_oid(st, dm, &vp);
356 if (rc == 0 && (!vp.present || vp.size != sp->size)) {
357 rc = -EIO; /* verify: dst must hold src's bytes */
358 }
359 }
360 return rc;
361}

363/* sd_ceph_rename — copy + delete on the flat namespace (phase-89 §B.2).
364 * NON-ATOMIC by design (the driver honestly does not advertise
365 * CAP_HARD_RENAME — same posture as the other object backends): the source is
366 * removed only after the destination's bytes AND xattrs landed and a size
367 * verify passed, so a mid-copy failure leaves the source intact (the partial
368 * destination is cleaned up best-effort). `noreplace` is honoured by a probe
369 * (racy by nature on an object store — documented). Directory rename is
370 * refused (EISDIR): a flat namespace has no atomic prefix move, and a
371 * recursive key rewrite is out of scope (ADR-5). */
372ngx_int_t
373sd_ceph_rename(brix_sd_instance_t *inst, const char *src, const char *dst,
374 int noreplace)
375{
376 sd_ceph_state_t *st = inst->state;
377 char so[1024];
378 char dm[1024];
379 sd_ceph_path_probe_t sp;
380 int dst_striped;
381 int rc;
382
383 if (sd_ceph_key(st->key_prefix, src, so, sizeof(so)) != 0
384 || sd_ceph_key(st->key_prefix, dst, dm, sizeof(dm)) != 0)
385 {
386 return NGX_ERROR;
387 }
388 if (strcmp(so, dm) == 0) {
389 return NGX_OK; /* same object — nothing to do */
390 }
391
392 rc = sd_ceph_probe_oid(st, so, &sp);
393 if (rc < 0) {
394 errno = -rc;
395 return NGX_ERROR;
396 }
397 if (!sp.present) {
398 return sd_ceph_rename_missing_src(inst, src);
399 }
400
401 if (sd_ceph_rename_prep_dst(st, dm, noreplace) != NGX_OK) {
402 return NGX_ERROR;
403 }
404
405 dst_striped = (sp.striped && sp.size > 0);
406 rc = sd_ceph_rename_copy_verify(st, so, dm, &sp, dst_striped);
407 if (rc < 0) {
408 (void) sd_ceph_remove_oid(st, dm, dst_striped); /* best-effort cleanup */
409 errno = -rc;
410 return NGX_ERROR;
411 }
412
413 rc = sd_ceph_remove_oid(st, so, sp.striped);
414 if (rc < 0) {
415 errno = -rc; /* dst intact; src survived too */
416 return NGX_ERROR;
417 }
418 return NGX_OK;
419}
