14/* WHAT: Normalise the source path into `path`: collapse a leading run of '/'
15 * down to a single '/', and guarantee the result starts with exactly one '/'.
16 * WHY: xrootd URLs use "root://host//path" (double slash separates authority
17 * from an absolute path), and clients are inconsistent about leading slashes.
18 * Collapsing leading slashes prevents "//" path-traversal ambiguity downstream;
19 * forcing a single leading '/' yields a canonical absolute path for authz.
20 * HOW: empty/NULL -> "" ; skip leading "//" pairs; size-check accounting for a
21 * possibly-prepended '/'; copy verbatim if already rooted, else prepend '/'. */
22static int
23tpc_copy_src_path(char *path, size_t path_size, const char *path_start)
24{
25 size_t path_len;
26
27 if (path_start == NULL || *path_start == '\0') {
28 path[0] = '\0';
29 return 0;
30 }
31
32 /* Collapse a leading run of slashes ("///x" -> "/x") to defeat the
33 * "root://host//path" double-slash and any "//" traversal ambiguity. */
34 while (path_start[0] == '/' && path_start[1] == '/') {
35 path_start++;
36 }
37
38 /* Reserve one extra byte for a '/' we will prepend when not already rooted. */
39 path_len = strlen(path_start);
40 if (path_len + (path_start[0] == '/' ? 0 : 1) >= path_size) {
41 return -1;
42 }
43
44 if (path_start[0] == '/') {
45 memcpy(path, path_start, path_len + 1);
46 } else {
47 path[0] = '/';
48 memcpy(path + 1, path_start, path_len + 1);
49 }
50
51 return 0;
52}

54/* WHAT: Decomposes a TPC source endpoint string (root://host//path, xroot://host/path, or bare host[:port]) into host, port, and path components. Finds "://" scheme separator → extracts authority between scheme and '/' → handles IPv6 bracket notation [...] → strtol validates port range 1-65535 → copies path component after '/' via tpc_copy_src_path (strips leading double-slashes, ensures single '/' prefix). Returns 0 on success, -1 on failure.
55 * WHY: TPC clients send source endpoints in varying formats — full URLs with scheme and path, bare host[:port] with lfn carrying the file name separately, or IPv6 addresses in bracket notation. This function normalizes all variants into a consistent host/port/path triplet for downstream security validation and file resolution. Port validation (1-65535) prevents overflow; full-field-clearing on failure prevents partial-parse bypass.
56 * HOW: strstr("://") scheme separator → authority_start = after "://" → memchr('/') path boundary → handle IPv6 brackets [...] → strtol port range 1-65535 → tpc_copy_src_path for path component with double-slash stripping and single '/' prefix enforcement. */
57static int
58tpc_parse_src_spec(const char *src, char *host, size_t host_size,
59 uint16_t *port, char *path, size_t path_size)
60{
61 const char *authority_start;
62 const char *authority_end;
63 const char *path_start;
64 const char *scheme_separator;
65 const char *src_end;
66 char authority[320];
67 size_t authority_len;
68 int parsed_port = 0;
69
70 if (src == NULL || *src == '\0') {
71 return -1;
72 }
73
74 src_end = src + strlen(src);
75 scheme_separator = strstr(src, "://");
76 if (scheme_separator != NULL) {
77 /* URL form: authority begins after "://" and runs up to the first '/'
78 * (which also marks where the path begins). No '/' => authority spans
79 * to end and there is no path component. */
80 authority_start = scheme_separator + 3;
81 authority_end = memchr(authority_start, '/',
82 (size_t) (src_end - authority_start));
83 path_start = authority_end;
84 if (authority_end == NULL) {
85 authority_end = src_end;
86 path_start = NULL;
87 }
88 } else {
89 /* Bare "host[:port]" form: the whole string is the authority; the file
90 * name will be supplied separately via the lfn parameter. */
91 authority_start = src;
92 authority_end = src_end;
93 path_start = NULL;
94 }
95
96 if (authority_start == authority_end) {
97 return -1;
98 }
99
100 /* Lift the authority slice into a NUL-terminated scratch buffer and split it
101 * with the shared bracketed-IPv6-aware host:port parser — the same leaf the
102 * native client (url.c) uses, so both validate the authority identically.
103 * default_port 0 keeps this caller's "no explicit port -> 0 sentinel"
104 * contract (the launch path applies the real default later); an explicit
105 * port is validated to 1..65535. */
106 authority_len = (size_t) (authority_end - authority_start);
107 if (authority_len >= sizeof(authority)) {
108 return -1;
109 }
110 memcpy(authority, authority_start, authority_len);
111 authority[authority_len] = '\0';
112
113 if (brix_split_host_port(authority, host, host_size, &parsed_port, 0)
114 != 0) {
115 return -1;
116 }
117 *port = (uint16_t) parsed_port;
118
119 return tpc_copy_src_path(path, path_size, path_start);
120}

122/* WHAT: Copy a value_len-byte opaque value into dst as a NUL-terminated string,
123 * rejecting any value that does not fit (with its NUL) — the core overflow
124 * guard for every tpc.* parameter parsed by tpc_parse_token().
125 * WHY: Opaque values come straight off the wire; a fixed-size destination must
126 * never be overrun by an attacker-controlled length.
127 * HOW: bounds-check value_len < dst_size, memcpy, terminate. */
128static int
129tpc_copy_value(char *dst, size_t dst_size, const char *value_start,
130 size_t value_len)
131{
132 if (value_len >= dst_size) {
133 return -1;
134 }
135
136 memcpy(dst, value_start, value_len);
137 dst[value_len] = '\0';
138
139 return 0;
140}

142/* WHAT: Converts a Logical File Name (lfn) into src_path, adding a leading '/' if the lfn lacks one. Only executes when out->src_path is empty AND out->has_lfn is true. Handles both absolute paths (starting with '/') and relative paths by prepending '/'. Enforces sizeof(out->src_path) boundary to prevent overflow.
143 * WHY: TPC source specifications may arrive as LFNs without leading slashes; this helper normalizes path format before src_host/src_port parsing ensures consistent path representation for downstream security validation and file resolution. Prevents cross-filesystem transfer attempts with malformed paths.
144 * HOW: Three cases → if src_path already populated or has_lfn=false, return immediately; if lfn starts with '/', copy directly via ngx_cpystrn; if relative (no leading '/'), prepend '/' then copy remaining characters into buffer with size guard. */
145
146static void
147tpc_fill_src_path_from_lfn(brix_tpc_params_t *out)
148{
149 if (out->src_path[0] != '\0' || !out->has_lfn) {
150 return;
151 }
152
153 if (out->lfn[0] == '/') {
154 ngx_cpystrn((u_char *) out->src_path, (u_char *) out->lfn,
155 sizeof(out->src_path));
156 return;
157 }
158
159 if (strlen(out->lfn) + 1 < sizeof(out->src_path)) {
160 out->src_path[0] = '/';
161 ngx_cpystrn((u_char *) out->src_path + 1, (u_char *) out->lfn,
162 sizeof(out->src_path) - 1);
163 }
164}

166/* WHAT: Parses the Source field from a TPC request into src_host, src_port, and src_path components via tpc_parse_src_spec(). On failure, clears all three fields to prevent partial-parse security bypass. Then delegates path normalization to tpc_fill_src_path_from_lfn() for LFN-to-src_path conversion.
167 * WHY: TPC (Transfer Protocol Client) requires parsing source specification into host/port/path components before initiating cross-server transfer. Complete-field-clearing on failure prevents security bypass where a partially-parsed source could be used with incomplete authentication. Path normalization ensures consistent source path format for downstream validation.
168 * HOW: Two-phase → if has_src is false, return immediately; call tpc_parse_src_spec() to extract host/port/path; on non-zero error result, clear all fields (src_host='\0', src_path='\0', src_port=0); then call tpc_fill_src_path_from_lfn() for LFN normalization. */

170static void
171tpc_parse_src_fields(brix_tpc_params_t *out)
172{
173 if (!out->has_src) {
174 return;

189/* WHAT: Iterates through one "key=value" token in the '&' delimited opaque query string. Finds token boundary (next '&' or end-of-string) → locates '=' separator → verifies "tpc." prefix (4 bytes) → matches remaining key length against known keys (src=3, dst=3, key=3, lfn=3, org=3, stage=5, token_mode=10) → copies value into corresponding buffer with size guard → sets has_* flag on success → returns pointer to next token or NULL when done.
190 * WHY: TPC opaque parameters are '&' delimited key=value pairs prefixed with "tpc.". This function extracts each recognized parameter into the typed brix_tpc_params_t struct while silently ignoring unknown keys (forward compatibility). Size-guarded copies prevent buffer overflow from oversized values. Returns next-token pointer enables iterative loop in tpc_parse_opaque.
191 * HOW: memchr(&) for token boundary → memchr(=) for key/value split → memcmp("tpc.") prefix check → switch on remaining key length against known keys → tpc_copy_value with size guard → set has_* flag → return next-token pointer or NULL. */
192static const char *
193tpc_parse_token(const char *token_start, const char *opaque_end,
194 brix_tpc_params_t *out)
195{
196 const char *token_end;
197 const char *equals;
198 const char *key_start;
199 const char *value_start;
200 size_t key_len;
201 size_t value_len;
202
203 /* Find end of this token. */
204 token_end = memchr(token_start, '&', (size_t) (opaque_end - token_start));
205 if (token_end == NULL) {
206 token_end = opaque_end;
207 }
208
209 equals = memchr(token_start, '=', (size_t) (token_end - token_start));
210 if (equals == NULL) {
211 return (token_end < opaque_end) ? token_end + 1 : NULL;
212 }
213
214 key_start = token_start;
215 value_start = equals + 1;
216 key_len = (size_t) (equals - key_start);
217 value_len = (size_t) (token_end - value_start);
218
219 /* Only handle "tpc." keys. key_len < 5 means there is no name after the
220 * 4-char prefix, so it cannot be a recognised tpc.* parameter. Unknown
221 * keys are skipped (return next token), giving forward compatibility. */
222 if (key_len < 5 || memcmp(key_start, "tpc.", 4) != 0) {
223 return (token_end < opaque_end) ? token_end + 1 : NULL;
224 }
225
226 /* Advance past "tpc." so the comparisons below match the bare name. */
227 key_start += 4;
228 key_len -= 4;
229
230 if (key_len == 3 && memcmp(key_start, "src", 3) == 0) {
231 if (tpc_copy_value(out->src, sizeof(out->src), value_start,
232 value_len) == 0) {
233 out->has_src = 1;
234 }
235 } else if (key_len == 3 && memcmp(key_start, "dst", 3) == 0) {
236 if (tpc_copy_value(out->dst, sizeof(out->dst), value_start,
237 value_len) == 0) {
238 out->has_dst = 1;
239 }
240 } else if (key_len == 3 && memcmp(key_start, "key", 3) == 0) {
241 if (tpc_copy_value(out->key, sizeof(out->key), value_start,
242 value_len) == 0) {
243 out->has_key = 1;
244 }
245 } else if (key_len == 3 && memcmp(key_start, "lfn", 3) == 0) {
246 if (tpc_copy_value(out->lfn, sizeof(out->lfn), value_start,
247 value_len) == 0) {
248 out->has_lfn = 1;
249 }
250 } else if (key_len == 3 && memcmp(key_start, "org", 3) == 0) {
251 if (tpc_copy_value(out->org, sizeof(out->org), value_start,
252 value_len) == 0) {
253 out->has_org = 1;
254 }
255 } else if (key_len == 5 && memcmp(key_start, "stage", 5) == 0) {
256 if (tpc_copy_value(out->stage, sizeof(out->stage), value_start,
257 value_len) == 0) {
258 out->has_stage = 1;
259 }
260 } else if (key_len == 10 && memcmp(key_start, "token_mode", 10) == 0) {
261 if (tpc_copy_value(out->token_mode, sizeof(out->token_mode),
262 value_start, value_len) == 0) {
263 out->has_token_mode = 1;
264 }
265 }
266
267 return (token_end < opaque_end) ? token_end + 1 : NULL;
268}

270/* WHAT: Public entry point that parses the TPC opaque parameter string from a kXR_open request into structured brix_tpc_params_t fields. Zero-initializes out → iterates '&' delimited key=value tokens via tpc_parse_token → validates at least one recognized "tpc." key present (src/dst/key/lfn/org/stage/token_mode) → delegates src URL/host/port/path decomposition to tpc_parse_src_fields → returns 0 on success, -1 on failure.
271 * WHY: TPC requests encode source endpoint information in opaque query parameters appended to the kXR_open path field. This function extracts those parameters into a typed struct for downstream security validation and file resolution. Zero-initialization prevents stale data from previous parses; full-field-clearing on src-parse failure prevents partial-parse security bypass.
272 * HOW: memset(out,0) → iterate tokens via tpc_parse_token(&-delimited loop) → check at least one has_* flag set → call tpc_parse_src_fields if has_src=true → return 0/−1 based on found flags count. */
273
274int
275brix_tpc_parse_opaque(const char *opaque, brix_tpc_params_t *out)
276{
277 const char *token;
278 const char *end;
279 int found = 0;
280
281 memset(out, 0, sizeof(*out));
282
283 if (opaque == NULL || *opaque == '\0') {
284 return -1;
285 }
286
287 token = opaque;
288 end = opaque + strlen(opaque);
289
290 while (token != NULL && token < end) {
291 token = tpc_parse_token(token, end, out);
292 }
293
294 found = (out->has_src || out->has_dst || out->has_key
295 || out->has_lfn || out->has_org || out->has_stage
296 || out->has_token_mode);
297 if (!found) {
298 return -1;
299 }
300
301 tpc_parse_src_fields(out);
302
303 return 0;
304}
