15/* WHAT: Normalise the source path into `path`: collapse a leading run of '/'
16 * down to a single '/', and guarantee the result starts with exactly one '/'.
17 * WHY: xrootd URLs use "root://host//path" (double slash separates authority
18 * from an absolute path), and clients are inconsistent about leading slashes.
19 * Collapsing leading slashes prevents "//" path-traversal ambiguity downstream;
20 * forcing a single leading '/' yields a canonical absolute path for authz.
21 * HOW: empty/NULL -> "" ; skip leading "//" pairs; size-check accounting for a
22 * possibly-prepended '/'; copy verbatim if already rooted, else prepend '/'. */
23static int
24tpc_copy_src_path(char *path, size_t path_size, const char *path_start)
25{
26 size_t path_len;
27
28 if (path_start == NULL || *path_start == '\0') {
29 path[0] = '\0';
30 return 0;
31 }
32
33 /* Collapse a leading run of slashes ("///x" -> "/x") to defeat the
34 * "root://host//path" double-slash and any "//" traversal ambiguity. */
35 while (path_start[0] == '/' && path_start[1] == '/') {
36 path_start++;
37 }
38
39 /* Reserve one extra byte for a '/' we will prepend when not already rooted. */
40 path_len = strlen(path_start);
41 if (path_len + (path_start[0] == '/' ? 0 : 1) >= path_size) {
42 return -1;
43 }
44
45 if (path_start[0] == '/') {
46 memcpy(path, path_start, path_len + 1);
47 } else {
48 path[0] = '/';
49 memcpy(path + 1, path_start, path_len + 1);
50 }
51
52 return 0;
53}

55/* WHAT: Bundle the destination out-params of tpc_parse_src_spec — the caller's

73/* WHAT: Decomposes a TPC source endpoint string (root://host//path, xroot://host/path, or bare host[:port]) into host, port, and path components. Finds "://" scheme separator → extracts authority between scheme and '/' → handles IPv6 bracket notation [...] → strtol validates port range 1-65535 → copies path component after '/' via tpc_copy_src_path (strips leading double-slashes, ensures single '/' prefix). Returns 0 on success, -1 on failure.
74 * WHY: TPC clients send source endpoints in varying formats — full URLs with scheme and path, bare host[:port] with lfn carrying the file name separately, or IPv6 addresses in bracket notation. This function normalizes all variants into a consistent host/port/path triplet for downstream security validation and file resolution. Port validation (1-65535) prevents overflow; full-field-clearing on failure prevents partial-parse bypass.
75 * HOW: strstr("://") scheme separator → authority_start = after "://" → memchr('/') path boundary → handle IPv6 brackets [...] → strtol port range 1-65535 → tpc_copy_src_path for path component with double-slash stripping and single '/' prefix enforcement. */
76static int
77tpc_parse_src_spec(const char *src, tpc_src_spec_out_t *dst)
78{
79 const char *authority_start;
80 const char *authority_end;
81 const char *path_start;
82 const char *scheme_separator;
83 const char *src_end;
84 char authority[320];
85 size_t authority_len;
86 int parsed_port = 0;
87
88 if (src == NULL || *src == '\0') {
89 return -1;
90 }
91
92 src_end = src + strlen(src);
93 scheme_separator = strstr(src, "://");
94 if (scheme_separator != NULL) {
95 /* URL form: authority begins after "://" and runs up to the first '/'
96 * (which also marks where the path begins). No '/' => authority spans
97 * to end and there is no path component. */
98 authority_start = scheme_separator + 3;
99 authority_end = memchr(authority_start, '/',
100 (size_t) (src_end - authority_start));
101 path_start = authority_end;
102 if (authority_end == NULL) {
103 authority_end = src_end;
104 path_start = NULL;
105 }
106 } else {
107 /* Bare "host[:port]" form: the whole string is the authority; the file
108 * name will be supplied separately via the lfn parameter. */
109 authority_start = src;
110 authority_end = src_end;
111 path_start = NULL;
112 }
113
114 if (authority_start == authority_end) {
115 return -1;
116 }
117
118 /* Lift the authority slice into a NUL-terminated scratch buffer and split it
119 * with the shared bracketed-IPv6-aware host:port parser — the same leaf the
120 * native client (url.c) uses, so both validate the authority identically.
121 * default_port 0 keeps this caller's "no explicit port -> 0 sentinel"
122 * contract (the launch path applies the real default later); an explicit
123 * port is validated to 1..65535. */
124 authority_len = (size_t) (authority_end - authority_start);
125 if (authority_len >= sizeof(authority)) {
126 return -1;
127 }
128 memcpy(authority, authority_start, authority_len);
129 authority[authority_len] = '\0';
130
131 if (brix_split_host_port(authority, dst->host, dst->host_size,
132 &parsed_port, 0) != 0) {
133 return -1;
134 }
135 *dst->port = (uint16_t) parsed_port;
136
137 return tpc_copy_src_path(dst->path, dst->path_size, path_start);
138}

140/* WHAT: Copy a value_len-byte opaque value into dst as a NUL-terminated string,
141 * rejecting any value that does not fit (with its NUL) — the core overflow
142 * guard for every tpc.* parameter parsed by tpc_parse_token().
143 * WHY: Opaque values come straight off the wire; a fixed-size destination must
144 * never be overrun by an attacker-controlled length.
145 * HOW: bounds-check value_len < dst_size, memcpy, terminate. */
146static int
147tpc_copy_value(char *dst, size_t dst_size, const char *value_start,
148 size_t value_len)
149{
150 if (value_len >= dst_size) {
151 return -1;
152 }
153
154 memcpy(dst, value_start, value_len);
155 dst[value_len] = '\0';
156
157 return 0;
158}

160/* WHAT: Converts a Logical File Name (lfn) into src_path, adding a leading '/' if the lfn lacks one. Only executes when out->src_path is empty AND out->has_lfn is true. 161 * WHY: TPC source specifications may arrive as LFNs without leading slashes; this helper normalizes path format before src_host/src_port parsing ensures consistent path representation for downstream security validation and file resolution. Prevents cross-filesystem transfer attempts with malformed paths.
162 * HOW: Three cases → if src_path already populated or has_lfn=false, return immediately; if lfn starts with '/', copy directly via ngx_cpystrn; if relative (no leading '/'), prepend '/' then copy remaining characters into buffer with size guard. */
163
164static void
165tpc_fill_src_path_from_lfn(brix_tpc_params_t *out)
166{
167 if (out->src_path[0] != '\0' || !out->has_lfn) {
168 return;
169 }
170
171 if (out->lfn[0] == '/') {
172 ngx_cpystrn((u_char *) out->src_path, (u_char *) out->lfn,
173 sizeof(out->src_path));
174 return;
175 }
176
177 if (strlen(out->lfn) + 1 < sizeof(out->src_path)) {
178 out->src_path[0] = '/';
179 ngx_cpystrn((u_char *) out->src_path + 1, (u_char *) out->lfn,
180 sizeof(out->src_path) - 1);
181 }
182}

184/* WHAT: Parses the Source field from a TPC request into src_host, src_port, and src_path components via tpc_parse_src_spec(). On failure, clears all three fields to prevent partial-parse security bypass. Then delegates path normalization to tpc_fill_src_path_from_lfn() for LFN-to-src_path conversion.
185 * WHY: TPC (Transfer Protocol Client) requires parsing source specification into host/port/path components before initiating cross-server transfer. Complete-field-clearing on failure prevents security bypass where a partially-parsed source could be used with incomplete authentication. Path normalization ensures consistent source path format for downstream validation.
186 * HOW: Two-phase → if has_src is false, return immediately; call tpc_parse_src_spec() to extract host/port/path; on non-zero error result, clear all fields (src_host='\0', src_path='\0', src_port=0); then call tpc_fill_src_path_from_lfn() for LFN normalization. */
187
188static void
189tpc_parse_src_fields(brix_tpc_params_t *out)
190{
191 tpc_src_spec_out_t spec = {
192 .host = out->src_host,
193 .host_size = sizeof(out->src_host),
194 .port = &out->src_port,
195 .path = out->src_path,
196 .path_size = sizeof(out->src_path),
197 };
198
199 if (!out->has_src) {
200 return;
201 }
202

204 out->src_host[0] = '\0';
205 out->src_path[0] = '\0';
206 out->src_port = 0;
207 return;
208 }

210 tpc_fill_src_path_from_lfn(out);

213/* WHAT: One row of the recognised-tpc.*-key table — the bare key name (after
214 * the "tpc." prefix) and its length, plus the byte offset, capacity, and
215 * has_* flag offset of the destination field inside brix_tpc_params_t.
216 * WHY: The seven recognised keys differ only in name and target field, so they
217 * are data, not control flow. A descriptor table replaces the seven-branch
218 * if/else ladder (§8.6) with one match loop, collapsing the parser's CCN while
219 * keeping the exact same set of keys, lengths, and target buffers.
220 * HOW: name/name_len drive the memcmp match; value_off/value_size drive the
221 * size-guarded tpc_copy_value; flag_off addresses the int has_* flag to set. */
222typedef struct {
223 const char *name;
224 size_t name_len;
225 size_t value_off;
226 size_t value_size;
227 size_t flag_off;
228} tpc_key_desc_t;

230/* WHAT: The recognised tpc.* parameters, keyed by bare name. Offsets/sizes
231 * address the matching brix_tpc_params_t buffer and its has_* flag.
232 * WHY: Table-driven dispatch (§8.6) — adding a parameter is a new row, not a
233 * new branch. Rows mirror the former ladder 1:1 (src/dst/key/lfn/org/stage/
234 * token_mode) so parsing is byte-for-byte identical.
235 * HOW: offsetof()/sizeof() computed once at compile time; consumed by the
236 * match loop in tpc_parse_token. */
237#define TPC_KEY_ROW(bare, field) \
238 { (bare), sizeof(bare) - 1, \
239 offsetof(brix_tpc_params_t, field), sizeof(((brix_tpc_params_t *)0)->field), \
240 offsetof(brix_tpc_params_t, has_##field) }
241

248 TPC_KEY_ROW("stage", stage),

254/* WHAT: Match a bare key (name/len, "tpc." already stripped) against the
255 * recognised-key table and, on a hit, copy the value into the target field and
256 * set its has_* flag — the table-driven body that replaced the if/else ladder.
257 * WHY: Isolates the per-token field-store so tpc_parse_token stays a flat token
258 * splitter. A recognised-but-oversized value copies nothing and leaves the flag
259 * clear, exactly as the former ladder did (tpc_copy_value returns non-zero).
260 * HOW: Linear scan of tpc_key_table; on a length+memcmp match, tpc_copy_value
261 * into (out + value_off) and, on success, set *(int *)(out + flag_off) = 1. */
262static void
263tpc_apply_key(brix_tpc_params_t *out, const char *key_start, size_t key_len,
264 const char *value_start, size_t value_len)
265{
266 char *base = (char *) out;
267 size_t i;
268
269 for (i = 0; i < sizeof(tpc_key_table) / sizeof(tpc_key_table[0]); i++) {
270 const tpc_key_desc_t *row = &tpc_key_table[i];
271
272 if (key_len != row->name_len
273 || memcmp(key_start, row->name, key_len) != 0) {
274 continue;
275 }
276
277 if (tpc_copy_value(base + row->value_off, row->value_size,
278 value_start, value_len) == 0) {
279 *(int *) (base + row->flag_off) = 1;
280 }
281 return;
282 }
283}

285/* WHAT: Iterates through one "key=value" token in the '&' delimited opaque query string. Finds token boundary (next '&' or end-of-string) → locates '=' separator → verifies "tpc." prefix (4 bytes) → matches remaining key against the recognised-key table (tpc_key_table) via tpc_apply_key → copies value into corresponding buffer with size guard → sets has_* flag on success → returns pointer to next token or NULL when done.
286 * WHY: TPC opaque parameters are '&' delimited key=value pairs prefixed with "tpc.". This function extracts each recognized parameter into the typed brix_tpc_params_t struct while silently ignoring unknown keys (forward compatibility). Size-guarded copies prevent buffer overflow from oversized values. Returns next-token pointer enables iterative loop in tpc_parse_opaque.
287 * HOW: memchr(&) for token boundary → memchr(=) for key/value split → memcmp("tpc.") prefix check → switch on remaining key length against known keys → tpc_copy_value with size guard → set has_* flag → return next-token pointer or NULL. */
288static const char *
289tpc_parse_token(const char *token_start, const char *opaque_end,
290 brix_tpc_params_t *out)
291{
292 const char *token_end;
293 const char *equals;
294 const char *key_start;
295 const char *value_start;
296 size_t key_len;
297 size_t value_len;
298
299 /* Find end of this token. */
300 token_end = memchr(token_start, '&', (size_t) (opaque_end - token_start));
301 if (token_end == NULL) {
302 token_end = opaque_end;
303 }
304
305 equals = memchr(token_start, '=', (size_t) (token_end - token_start));
306 if (equals == NULL) {
307 return (token_end < opaque_end) ? token_end + 1 : NULL;
308 }
309
310 key_start = token_start;
311 value_start = equals + 1;
312 key_len = (size_t) (equals - key_start);
313 value_len = (size_t) (token_end - value_start);
314
315 /* Only handle "tpc." keys. key_len < 5 means there is no name after the
316 * 4-char prefix, so it cannot be a recognised tpc.* parameter. Unknown
317 * keys are skipped (return next token), giving forward compatibility. */
318 if (key_len < 5 || memcmp(key_start, "tpc.", 4) != 0) {
319 return (token_end < opaque_end) ? token_end + 1 : NULL;
320 }
321
322 /* Advance past "tpc." so the comparisons below match the bare name. */
323 key_start += 4;
324 key_len -= 4;
325
326 tpc_apply_key(out, key_start, key_len, value_start, value_len);
327
328 return (token_end < opaque_end) ? token_end + 1 : NULL;
329}

331/* WHAT: Public entry point that parses the TPC opaque parameter string from a kXR_open request into structured brix_tpc_params_t fields. 337{

341

345 return -1;
