93/*
94 * WHAT: The invariant inputs of one resilient download-body stream — the
95 * caller-owned control connection, the source URL, the source stat info
96 * (its size drives the pump + progress), the transfer options, and the
97 * bound-streams set to attach after the redirect.
98 * WHY: download_stream_body previously took these five stable inputs plus the
99 * per-callsite (sink, sinkctx) pump pair as eight positional parameters,
100 * over the 5-parameter gate. Bundling the invariants that every caller
101 * shares into one struct leaves only the varying pump pair (and the st
102 * out-param) as free arguments, keeping the extern under the gate without
103 * altering what it does.
104 * HOW: Each caller fills one of these (all pointers borrowed for the call's
105 * duration, none owned by the callee) and passes it by const pointer; the
106 * body reads c/su/si/o/ss exactly as it read the former parameters. The
107 * (sink, sinkctx) sink and the brix_status out-param stay separate because
108 * they vary per callsite / follow the out-param convention.
109 */
110typedef struct {
111 brix_conn *c; /* caller-owned control connection */
112 const brix_url *su; /* remote source URL (path + opaque source) */
113 const brix_statinfo *si; /* source stat (si->size drives pump/progress)*/
114 const brix_copy_opts *o; /* transfer options */
115 brix_streamset *ss; /* bound secondary-stream set to attach */
116} download_body_ctx;
