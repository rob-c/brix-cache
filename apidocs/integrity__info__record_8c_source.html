1/*
2 * integrity_info_record.c — record-DIGEST checksum fallback for exports without
3 * user xattrs (split from integrity_info.c, phase-79).
4 *
5 * WHAT: Reads and writes a file's checksum as a DIGEST entry inside the file's
6 * unified xmeta record, keyed and validated against the live file's
7 * mtime+size. integrity_record_read returns 1 with a populated
8 * brix_integrity_info_t on a fresh hit; integrity_record_write persists a
9 * computed value best-effort.
10 * WHY: Filesystems that lack user extended attributes cannot use the primary
11 * "user.XrdCks.<alg>" xattr cache, so the checksum must ride in the same
12 * xmeta record that already carries block metadata (which itself falls back
13 * to a stock-readable "<path>.cinfo" sidecar) — ONE metadata form. Keeping
14 * this fallback path in its own file holds the checksum service under the
15 * 500-line cap and focused: integrity_info.c owns the xattr layer and the
16 * orchestration, this file owns the record-DIGEST layer.
17 * HOW: integrity_alg_id maps an algorithm name to its BRIX_XMETA_ALG_* id;
18 * integrity_record_read loads the xmeta record and scans its DIGEST entries
19 * for a matching, still-current value; integrity_record_write locks, loads
20 * (or initialises a fresh record for a foreign/live export file), sets the
21 * DIGEST entry, and saves.
22 */

32/* Record-DIGEST fallback (§8.2, xmeta P4) — for exports without user xattrs.
33 * Instead of the retired "<path>.cks" sidecar, the checksum rides as a DIGEST
34 * entry in the file's unified xmeta record (which itself falls back to the
35 * stock-readable "<path>.cinfo" sidecar on such filesystems — ONE metadata
36 * form). Entry value = "<hex> <mtime_sec> <mtime_nsec> <size>", keyed and
37 * validated on the live file's mtime+size, mirroring the xattr policy. Stock
38 * interop is untouched: xattr-capable exports keep user.XrdCks.<alg>. */
39
40static uint16_t
41integrity_alg_id(const char *algo)
42{
43 static const struct { const char *name; uint16_t id; } map[] = {
44 { "adler32", BRIX_XMETA_ALG_ADLER32 },
45 { "crc32", BRIX_XMETA_ALG_CRC32 },
46 { "crc32c", BRIX_XMETA_ALG_CRC32C },
47 { "md5", BRIX_XMETA_ALG_MD5 },
48 { "sha1", BRIX_XMETA_ALG_SHA1 },
49 { "sha256", BRIX_XMETA_ALG_SHA256 },
50 { "crc64", BRIX_XMETA_ALG_CRC64XZ },
51 { "crc64nvme", BRIX_XMETA_ALG_CRC64NVME },
52 { "zcrc32", BRIX_XMETA_ALG_ZCRC32 },
53 { NULL, 0 },
54 };
55 int i;
56
57 for (i = 0; map[i].name != NULL; i++) {
58 if (strcmp(map[i].name, algo) == 0) {
59 return map[i].id;
60 }
61 }
62 return 0; /* unmapped: no record caching */
63}

65int
66integrity_record_read(const char *path, const char *algo,
67 brix_integrity_info_t *out)
68{
69 brix_xmeta_t xm;
70 struct stat st;
71 uint16_t want = integrity_alg_id(algo);
72 uint32_t idx;
73 int found = 0;
74
75 if (path == NULL || want == 0 || stat(path, &st) != 0
76 || brix_xmeta_path_load(path, &xm) != BRIX_XMETA_OK)
77 {
78 return 0;
79 }
80 for (idx = 0; ; idx++) {
81 uint16_t alg = 0, vlen = 0;
82 const uint8_t *val = NULL;
83 char rec[INTEGRITY_XATTR_VAL_MAX + 64];
84 char rhex[INTEGRITY_XATTR_VAL_MAX];
85 long ms, mn;
86 long long sz;
87
88 if (brix_xmeta_digest_get(&xm, idx, &alg, &val, &vlen)
89 != BRIX_XMETA_OK)
90 {
91 break;
92 }
93 if (alg != want || vlen == 0 || vlen >= sizeof(rec)) {
94 continue;
95 }
96 memcpy(rec, val, vlen);
97 rec[vlen] = 0;
98 if (sscanf(rec, "%159s %ld %ld %lld", rhex, &ms, &mn, &sz) != 4) {
99 continue;
100 }
101 if ((long) st.st_mtim.tv_sec == ms
102 && (long) st.st_mtim.tv_nsec == mn
103 && (long long) st.st_size == sz)
104 {
105 ngx_cpystrn((u_char *) out->hex, (u_char *) rhex,
106 sizeof(out->hex));
107 out->from_cache = 1;
108 found = 1;
109 }
110 break; /* entry found (fresh or stale) */
111 }
112 brix_xmeta_free(&xm);
113 return found;
114}

116void
117integrity_record_write(const char *path, const char *algo, const char *hexval)
118{
119 brix_xmeta_t xm;
120 struct stat st;
121 char rec[INTEGRITY_XATTR_VAL_MAX + 64];
122 uint16_t id = integrity_alg_id(algo);
123 int n, rc, lockfd;
124
125 if (path == NULL || id == 0 || stat(path, &st) != 0) {
126 return;
127 }
128 n = snprintf(rec, sizeof(rec), "%s %ld %ld %lld", hexval,
129 (long) st.st_mtim.tv_sec, (long) st.st_mtim.tv_nsec,
130 (long long) st.st_size);
131 if (n <= 0 || (size_t) n >= sizeof(rec)) {
132 return;
133 }
134
135 lockfd = brix_xmeta_path_lock(path);
136 if (lockfd < 0) {
137 return;
138 }
139 rc = brix_xmeta_path_load(path, &xm);
140 if (rc == BRIX_XMETA_ERR
141 || (rc == BRIX_XMETA_FOREIGN
142 && brix_xmeta_init(&xm, st.st_size, 1024 * 1024)
143 != BRIX_XMETA_OK))
144 {
145 brix_xmeta_path_unlock(lockfd);
146 return;
147 }
148 if (rc == BRIX_XMETA_FOREIGN) {
149 /* fresh record for a live export file: fully present, no CRC table */
150 xm.origin_mtime = (uint64_t) st.st_mtime;
151 if (xm.nblocks > 0) {
152 memset(xm.bitmap, 0xFF, (size_t) ((xm.nblocks + 7) / 8));
153 }
154 xm.have_blockcrc = 0;
155 }
156 if (brix_xmeta_digest_set(&xm, id, rec, (uint16_t) n)
157 == BRIX_XMETA_OK)
158 {
159 (void) brix_xmeta_path_save(path, &xm); /* best-effort cache */
160 }
161 brix_xmeta_free(&xm);
162 brix_xmeta_path_unlock(lockfd);
163}
