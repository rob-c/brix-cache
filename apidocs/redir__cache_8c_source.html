1/*
2 * redir_cache.c — shared-memory redirect-collapse cache.
3 *
4 * WHAT: Fixed-size open-addressing hash cache in nginx shared memory.
5 * Key: canonical path string (up to 255 bytes).
6 * Value: DS host string + port + expiry timestamp (ngx_current_msec + ttl).
7 * Lookup: FNV-1a hash → bounded linear probe; expired entries are misses.
8 * Insert: same hash window — update in place, else reuse a free/expired slot,
9 * else evict the soonest-to-expire entry in the window.
10 *
11 * WHY: When kXR_collapseRedir is advertised, clients expect that repeated
12 * identical open/locate requests skip the CMS round-trip. The cache stores
13 * the resolved DS address from a prior successful redirect so the event-loop
14 * handler can answer from memory without blocking on CMS. On a busy manager
15 * every open/locate consults this cache under a single spinlock, so the
16 * previous O(capacity) full-table scan (~512 × ~400-byte entries, L3-missing)
17 * was a contention hot spot — hashing bounds the locked work to a small fixed
18 * probe window regardless of table size.
19 *
20 * HOW: One nginx shared-memory zone ("brix_redir_cache") holds the header
21 * (lock + capacity) and the entry array. A single spinlock serialises reads
22 * and writes, but each operation now touches at most BRIX_REDIR_PROBE_MAX
23 * consecutive slots starting at hash(path) % capacity — O(1) under the lock.
24 * An entry always lives within that probe window of its hash bucket or it is
25 * not cached; lookup and insert probe the same window, so they stay
26 * consistent. ngx_current_msec (updated by the nginx timer) provides
27 * millisecond-resolution expiry without syscalls.
28 */

67redir_cache(void)
68{
69 if (brix_redir_shm_zone == NULL
70 || brix_redir_shm_zone->data == NULL
71 || brix_redir_shm_zone->data == (void *) 1)

78/* FNV-1a 32-bit hash of a NUL-terminated path, used to pick the probe window. */
79static uint32_t
80redir_hash(const char *path)
81{
82 uint32_t h = BRIX_FNV1A32_OFFSET_BASIS;
83
84 while (*path) {
85 h ^= (unsigned char) *path++;
86 h *= BRIX_FNV1A32_PRIME;
87 }
88
89 return h;
90}

92static ngx_int_t
93redir_cache_shm_init(ngx_shm_zone_t *shm_zone, void *data)
94{
95 brix_redir_cache_t *c;
96 ngx_flag_t fresh;
97 size_t table_bytes;
98
99 /*
100 * Allocate the table FROM the slab pool so nginx's slab-pool header (and the
101 * mutex ngx_unlock_mutexes() force-unlocks on every child death) survives at
102 * shm.addr. The helper handles fresh alloc, reload (data != NULL), and
103 * re-attach, zeroes the table, creates the process-local mutex from the
104 * table's leading ngx_shmtx_sh_t lock, and publishes it via shm_zone->data.
105 */
106 table_bytes = sizeof(brix_redir_cache_t)
107 + (size_t) brix_redir_cache_nslots
108 * sizeof(brix_redir_cache_entry_t);
109
110 c = brix_shm_table_alloc(shm_zone, data, table_bytes,
111 &brix_redir_mutex, &fresh);
112 if (c == NULL) {
113 return NGX_ERROR;
114 }
115
116 if (fresh) {
117 c->capacity = brix_redir_cache_nslots;
118 }
119
120 return NGX_OK;
121}

151int
152brix_redir_cache_lookup(const char *path,
153 char *host_out, size_t host_size, uint16_t *port_out)
154{
155 brix_redir_cache_t *c;
156 brix_redir_cache_entry_t *e;
157 ngx_uint_t probe, nprobe, start;
158 int found;
159 ngx_msec_t now;
160
161 c = redir_cache();
162 if (c == NULL) {
163 return 0;
164 }
165
166 now = ngx_current_msec;
167 found = 0;
168 start = (ngx_uint_t) redir_hash(path) % c->capacity;
169 nprobe = (c->capacity < BRIX_REDIR_PROBE_MAX)
170 ? c->capacity : BRIX_REDIR_PROBE_MAX;
171
172 ngx_shmtx_lock(&brix_redir_mutex);
173
174 for (probe = 0; probe < nprobe; probe++) {
175 e = &c->entries[(start + probe) % c->capacity];
176 if (!e->in_use || e->expires <= now) {
177 continue;
178 }
179 if (ngx_strcmp(e->path, path) == 0) {
180 ngx_cpystrn((u_char *) host_out, (u_char *) e->host, host_size);
181 *port_out = e->port;
182 found = 1;
183 break;
184 }
185 }
186
187 ngx_shmtx_unlock(&brix_redir_mutex);
188 return found;
189}

191void
192brix_redir_cache_insert(const char *path,
193 const char *host, uint16_t port, ngx_msec_t ttl_ms)
194{
195 brix_redir_cache_t *c;
196 brix_redir_cache_entry_t *e, *free_slot, *lru_slot, *victim;
197 ngx_uint_t probe, nprobe, start;
198 ngx_msec_t now, lru_exp;
199
200 c = redir_cache();
201 if (c == NULL || ttl_ms == 0) {
202 return;
203 }
204
205 now = ngx_current_msec;
206 /* phase79-fp: c NULL-checked above; analyzer loses the guard on c through this path */
207 start = (ngx_uint_t) redir_hash(path) % c->capacity;
208 nprobe = (c->capacity < BRIX_REDIR_PROBE_MAX)
209 ? c->capacity : BRIX_REDIR_PROBE_MAX;
210
211 ngx_shmtx_lock(&brix_redir_mutex);
212
213 /*
214 * Single pass over the probe window: if the path is already cached (live),
215 * update it in place; otherwise remember the best eviction target — the
216 * first free/expired slot if any, else the live entry closest to expiry.
217 * Scanning the whole window before evicting prevents inserting a duplicate
218 * of a path that already lives further along the window.
219 */
220 free_slot = NULL;
221 lru_slot = NULL;
222 lru_exp = 0;
223
224 for (probe = 0; probe < nprobe; probe++) {
225 /* phase79-fp: c NULL-checked at entry; analyzer drops the guard across ngx_shmtx_lock */
226 e = &c->entries[(start + probe) % c->capacity];
227
228 if (e->in_use && e->expires > now) {
229 if (ngx_strcmp(e->path, path) == 0) {
230 ngx_cpystrn((u_char *) e->host, (u_char *) host,
231 sizeof(e->host));
232 e->port = port;
233 e->expires = now + ttl_ms;
234 ngx_shmtx_unlock(&brix_redir_mutex);
235 return;
236 }
237 if (lru_slot == NULL || e->expires < lru_exp) {
238 lru_slot = e;
239 lru_exp = e->expires;
240 }
241 } else if (free_slot == NULL) {
242 free_slot = e; /* free or expired — best reuse target */
243 }
244 }
245
246 victim = (free_slot != NULL) ? free_slot : lru_slot;
247 if (victim == NULL) {
248 ngx_shmtx_unlock(&brix_redir_mutex); /* nprobe >= 1; defensive */
249 return;
250 }
251
252 ngx_cpystrn((u_char *) victim->path, (u_char *) path, sizeof(victim->path));
253 ngx_cpystrn((u_char *) victim->host, (u_char *) host, sizeof(victim->host));
254 victim->port = port;
255 victim->expires = now + ttl_ms;
256 victim->in_use = 1;
257
258 ngx_shmtx_unlock(&brix_redir_mutex);
259}
