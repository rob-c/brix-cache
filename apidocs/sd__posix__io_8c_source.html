1/*
2 * sd_posix_io.c — the POSIX Storage Driver's worker-safe raw byte I/O ops.
3 *
4 * WHAT: The raw fd byte primitives of brix_sd_posix_driver (pread/pwrite/
5 * preadv/preadv2/copy_range/read_sendfile_fd/ftruncate/fsync/fstat),
6 * split VERBATIM out of sd_posix.c. The driver descriptor stays in
7 * sd_posix.c and references these via sd_posix_internal.h.
8 *
9 * WHY: These ops are pure POSIX (no pool/metrics/log) and also build into the
10 * ngx-free shared libxrdproto, so a shared kernel can route its fd reads
11 * through the one driver in both worlds. Splitting them keeps every unit
12 * under the file-size cap with zero behaviour change.
13 *
14 * HOW: Each op is a thin wrapper over the raw syscall; the VFS owns every
15 * EINTR/short-io loop and policy above the seam.
16 */

46/* sd_posix_pread — one pread(2) at off (0 = EOF, -1 = errno). The raw storage
47 * primitive; the VFS owns the EINTR/short-read loop (brix_vfs_pread_full), so
48 * every VFS read funnels through the driver without changing semantics. */
49ssize_t
50sd_posix_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
51{
52 return pread(obj->fd, buf, len, off);
53}
54
55/* sd_posix_pwrite — one pwrite(2) at off (bytes written, or -1). The raw
56 * primitive; the VFS owns the EINTR/short-write loops (brix_vfs_pwrite_full,
57 * brix_vfs_io_write_counted), so every VFS write funnels through the driver. */
58ssize_t
59sd_posix_pwrite(brix_sd_obj_t *obj, const void *buf, size_t len, off_t off)

84/* sd_posix_copy_range — one copy_file_range(2) of up to len bytes src->dst (0 =
85 * EOF, -1/ENOSYS where unavailable): the server-side zero-copy primitive. The VFS
86 * owns the loop and the pread/pwrite fallback. */
87ssize_t
88sd_posix_copy_range(brix_sd_obj_t *src, off_t src_off, brix_sd_obj_t *dst,
89 off_t dst_off, size_t len)
90{
91#if defined(__linux__) && defined(__NR_copy_file_range)
92 loff_t si = (loff_t) src_off;
93 loff_t di = (loff_t) dst_off;
94
95 return (ssize_t) syscall(__NR_copy_file_range, src->fd, &si, dst->fd, &di,
96 len, 0u);
97#else
98 (void) src; (void) src_off; (void) dst; (void) dst_off; (void) len;
99 errno = ENOSYS;
100 return -1;
101#endif
102}

113 (void) len;
