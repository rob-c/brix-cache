34typedef enum {
35 WEBDAV_DIGEST_NONE = 0, /* no usable ingest digest header present */
36 WEBDAV_DIGEST_FOUND, /* a supported alg + value parsed (alg/exp filled)*/
37 WEBDAV_DIGEST_BAD /* a supported alg named, but its value is garbage*/
38} webdav_digest_kind_t;

40/* RFC-3230 Digest tokens / Content-MD5 we can recompute. `b64` = the value is
41 * base64 (md5/sha per RFC 3230 + RFC 1864); otherwise it is lowercase hex (the
42 * WLCG/dCache convention for adler32/crc32). */
43static const struct {
44 const char *tok; size_t toklen; const char *alg; int b64;
45} webdav_digest_tokens[] = {
46 { "md5", 3, "md5", 1 },
47 { "sha-256", 7, "sha256", 1 },
48 { "sha256", 6, "sha256", 1 },
49 { "adler32", 7, "adler32", 0 },
50 { "crc32c", 6, "crc32c", 0 },
51 { "crc32", 5, "crc32", 0 },
52};

54/* Two hex checksums are equal ignoring case and leading-zero padding (a client
55 * may send an un-padded adler32 while our compute zero-pads to the alg width). */
56static int
57webdav_hex_norm_equal(const char *a, const char *b)
58{
59 while (*a == '0') { a++; }
60 while (*b == '0') { b++; }
61 for (; *a != '\0' && *b != '\0'; a++, b++) {
62 if (ngx_tolower((u_char) *a) != ngx_tolower((u_char) *b)) {
63 return 0;
64 }
65 }
66 return *a == '\0' && *b == '\0';
67}

69/* Normalise a digest header value into lowercase hex in out[]. A base64 value is
70 * decoded then hex-encoded; a hex value is copied lowercased. */
71static ngx_int_t
72webdav_digest_value_hex(ngx_http_request_t *r, const u_char *val, size_t vlen,
73 int is_b64, char *out, size_t outsz)
74{
75 if (vlen == 0) {
76 return NGX_ERROR;
77 }
78 if (is_b64) {
79 ngx_str_t src, dst;
80 src.data = (u_char *) val;
81 src.len = vlen;
82 dst.len = ngx_base64_decoded_length(vlen);
83 dst.data = ngx_pnalloc(r->pool, dst.len ? dst.len : 1);
84 if (dst.data == NULL || ngx_decode_base64(&dst, &src) != NGX_OK) {
85 return NGX_ERROR;
86 }
87 if (dst.len == 0 || dst.len * 2 + 1 > outsz) {
88 return NGX_ERROR;
89 }
90 ngx_hex_dump((u_char *) out, dst.data, dst.len);
91 out[dst.len * 2] = '\0';
92 return NGX_OK;
93 }
94 if (vlen + 1 > outsz) {
95 return NGX_ERROR;
96 }
97 {
98 size_t i;
99 for (i = 0; i < vlen; i++) {
100 u_char c = val[i];
101 if (!((c >= '0' && c <= '9') || (c >= 'a' && c <= 'f')
102 || (c >= 'A' && c <= 'F')))
103 {
104 return NGX_ERROR; /* not a hex digit — malformed */
105 }
106 out[i] = (char) ngx_tolower(c);
107 }
108 out[vlen] = '\0';
109 }
110 return NGX_OK;
111}

113/* Strip leading and trailing linear whitespace from a [*s, *s+*len) slice. */
114static void
115webdav_tok_trim(u_char **s, size_t *len)
116{
117 u_char *p = *s;
118 size_t n = *len;
119 while (n > 0 && (*p == ' ' || *p == '\t')) { p++; n--; }
120 while (n > 0 && (p[n - 1] == ' ' || p[n - 1] == '\t')) { n--; }
121 *s = p;
122 *len = n;
123}

125/* Match a trimmed Digest token=value pair against the supported-alg table.
126 * Returns 1 if the token names a supported alg — *out is then FOUND (with *alg
127 * and exp_hex filled) or BAD (value not valid hex) — else 0 to keep scanning. */
128static int
129webdav_digest_match(ngx_http_request_t *r, u_char *tok, size_t tlen,
130 u_char *v, size_t vlen, const char **alg, char *exp_hex, size_t exp_sz,
131 webdav_digest_kind_t *out)
132{
133 size_t ntok = sizeof(webdav_digest_tokens)
134 / sizeof(webdav_digest_tokens[0]);
135 size_t i;
136
137 for (i = 0; i < ntok; i++) {
138 if (tlen == webdav_digest_tokens[i].toklen
139 && ngx_strncasecmp(tok,
140 (u_char *) webdav_digest_tokens[i].tok, tlen) == 0)
141 {
142 if (webdav_digest_value_hex(r, v, vlen,
143 webdav_digest_tokens[i].b64, exp_hex, exp_sz) != NGX_OK)
144 {
145 *out = WEBDAV_DIGEST_BAD;
146 } else {
147 *alg = webdav_digest_tokens[i].alg;
148 *out = WEBDAV_DIGEST_FOUND;
149 }
150 return 1;
151 }
152 }
153 return 0;
154}

156/* Scan a Digest: header value (comma-separated alg=value pairs) for the first
157 * supported alg. Returns 1 with *out set (FOUND/BAD) once a token matches, or 0
158 * if no listed token is supported (caller falls through to Content-MD5). */
159static int
160webdav_digest_scan(ngx_http_request_t *r, ngx_table_elt_t *h,
161 const char **alg, char *exp_hex, size_t exp_sz, webdav_digest_kind_t *out)
162{
163 u_char *p = h->value.data, *end = p + h->value.len;
164
165 while (p < end) {
166 u_char *comma = ngx_strlchr(p, end, ',');
167 u_char *iend = comma ? comma : end;
168 u_char *eq = ngx_strlchr(p, iend, '=');
169 if (eq != NULL) {
170 u_char *tok = p, *v = eq + 1;
171 size_t tlen = eq - p, vlen = iend - (eq + 1);
172 webdav_tok_trim(&tok, &tlen);
173 webdav_tok_trim(&v, &vlen);
174 if (webdav_digest_match(r, tok, tlen, v, vlen, alg,
175 exp_hex, exp_sz, out))
176 {
177 return 1;
178 }
179 }
180 p = comma ? comma + 1 : end;
181 }
182 return 0;
183}

185/* Pick the first supported digest from Digest: (RFC 3230) then Content-MD5:. On
186 * WEBDAV_DIGEST_FOUND, *alg points at a static alg name and exp_hex holds the
187 * client-asserted value as lowercase hex. Unsupported Digest tokens are skipped
188 * (best-effort interop), so a Digest that names only algs we cannot compute reads
189 * as NONE — require_digest then decides whether that is acceptable. */
190static webdav_digest_kind_t
191webdav_digest_select(ngx_http_request_t *r, const char **alg,
192 char *exp_hex, size_t exp_sz)
193{
194 ngx_table_elt_t *h;
195 webdav_digest_kind_t kind;
196
197 h = brix_http_find_header(r, "Digest", sizeof("Digest") - 1);
198 if (h != NULL && h->value.len > 0
199 && webdav_digest_scan(r, h, alg, exp_hex, exp_sz, &kind))
200 {
201 return kind;
202 }
203
204 h = brix_http_find_header(r, "Content-MD5", sizeof("Content-MD5") - 1);
205 if (h != NULL && h->value.len > 0) {
206 if (webdav_digest_value_hex(r, h->value.data, h->value.len, 1,
207 exp_hex, exp_sz) != NGX_OK)
208 {
209 return WEBDAV_DIGEST_BAD;
210 }
211 *alg = "md5";
212 return WEBDAV_DIGEST_FOUND;
213 }
214
215 return WEBDAV_DIGEST_NONE;
216}

218/*
219 * webdav_put_verify_ingest_digest — verify a client-asserted body digest over
220 * the fully-staged bytes, BEFORE the commit publishes them.
221 *
222 * A PUT may carry an end-to-end digest the client computed over what it sent
223 * (RFC-3230 `Digest:` — the WLCG/XrdHttp convention — or legacy `Content-MD5:`).
224 * The writer otherwise streams and commits whatever lands, so a byte flipped in
225 * flight past the TCP checksum is stored silently. When a usable digest is
226 * present we recompute it over the staged temp fd and refuse the commit (400) on
227 * mismatch — the client told us what it sent, so we can and must check it. With
228 * brix_webdav_require_digest on, a PUT that carries no usable digest is also
229 * refused, for deployments that decline writes they cannot verify.
230 *
231 * Returns NGX_OK to proceed to commit, or an HTTP status (>=400) to reject.
232 *
233 * Scope: the staged POSIX temp fd (brix_vfs_writer_fd). Coded bodies
234 * (Content-Encoding) are skipped — the stored plaintext would not match a digest
235 * over the encoded stream, matching the verify-on-write contract. A driver
236 * object target (S3) exposes no fd here and is verified on its own ingest path.
237 * The recompute is synchronous on the event thread (like §8.3 checksum-on-write);
238 * it only runs when a digest is asserted or required.
239 */
240ngx_int_t
241webdav_put_verify_ingest_digest(ngx_http_request_t *r,
242 ngx_http_brix_webdav_loc_conf_t *conf, brix_vfs_writer_t *writer,
243 const char *path)
244{
245 char exp_hex[129];
246 const char *alg = NULL;
247 webdav_digest_kind_t kind;
248 ngx_fd_t fd;
249 ngx_table_elt_t *ce;
250 brix_integrity_info_t info;
251 brix_integrity_opts_t o;
252
253 /* A coded body is stored decoded; a digest over the encoded stream cannot be
254 * checked against it — skip (same limitation as verify-on-write). */
255 ce = brix_http_find_header(r, "Content-Encoding", sizeof("Content-Encoding") - 1);
256 if (ce != NULL && ce->value.len > 0) {
257 return NGX_OK;
258 }
259
260 kind = webdav_digest_select(r, &alg, exp_hex, sizeof(exp_hex));
261 if (kind == WEBDAV_DIGEST_BAD) {
262 return NGX_HTTP_BAD_REQUEST; /* known alg, bad value */
263 }
264 if (kind == WEBDAV_DIGEST_NONE) {
265 return conf->require_digest ? NGX_HTTP_BAD_REQUEST : NGX_OK;
266 }
267
268 fd = brix_vfs_writer_fd(writer);
269 if (fd == NGX_INVALID_FILE) {
270 /* Driver-backed object target — no local fd to hash here. */
271 return conf->require_digest ? NGX_HTTP_BAD_REQUEST : NGX_OK;
272 }
273
274 ngx_memzero(&o, sizeof(o));
275 o.require_regular_file = 1; /* hash the staged bytes fresh (no xattr cache) */
276 {
277 char procpath[32];
278 ngx_int_t irc;
279 int rfd;
280
281 /* The staged temp is opened write-only (staged_open_posix, O_WRONLY), and
282 * has no final path yet (pre-commit), so it cannot be pread directly nor
283 * reopened by name. Re-open the same open file description read-only via
284 * /proc/self/fd/<fd> — valid on Linux for an unlinked/O_TMPFILE staged
285 * temp — and hash the already-written bytes over the read handle. */
286 (void) ngx_snprintf((u_char *) procpath, sizeof(procpath),
287 "/proc/self/fd/%d%Z", (int) fd);
288 rfd = open(procpath, O_RDONLY | O_CLOEXEC); /* vfs-seam-allow: read-back of staged PUT bytes for pre-commit ingest-digest verify (no final path yet) */
289 if (rfd < 0) {
290 /* Cannot obtain a readable view of what we must verify: fail closed. */
291 return NGX_HTTP_BAD_REQUEST;
292 }
293 irc = brix_integrity_get_fd(r->connection->log, rfd, NULL, path,
294 alg, &o, &info);
295 (void) close(rfd);
296 if (irc != NGX_OK) {
297 /* Cannot compute the digest we were asked to verify: fail closed. */
298 return NGX_HTTP_BAD_REQUEST;
299 }
300 }
301
302 if (!webdav_hex_norm_equal(exp_hex, info.hex)) {
303 ngx_log_error(NGX_LOG_ERR, r->connection->log, 0,
304 "brix_webdav: PUT ingest %s digest mismatch (asserted=%s got=%s)",
305 alg, exp_hex, info.hex);
306 brix_log_safe_path(r->connection->log, NGX_LOG_ERR, 0,
307 "brix_webdav: ingest digest mismatch rejects PUT for: \"%s\"", path);
308 return NGX_HTTP_BAD_REQUEST;
309 }
310 return NGX_OK;
311}
