1/*
2 * tpc_push.c - HTTP-TPC COPY push handler for the WebDAV module.
3 *
4 * WHAT: Implements the push direction of HTTP-TPC (Third-Party Copy): read a
5 * local export file and PUT it to a remote HTTPS Destination. Owns the
6 * Destination-URL validation, the OAuth2/OIDC Credential delegation and
7 * Overwrite forwarding for the outbound leg, and the thread-pool /
8 * synchronous-curl execution tiers.
9 * WHY: Split from tpc.c (which was over the 500-line cap) so the push concern
10 * is grouped and individually reviewable, separate from the COPY dispatcher,
11 * the request parsing, and the staged pull execution.
12 * HOW: webdav_tpc_handle_push (the sole cross-file entry, declared in
13 * tpc_internal_split.h) validates the destination, collects transfer
14 * headers, applies credential delegation, stats the source, authorizes,
15 * forwards Overwrite, then hands off to the thread pool, falling back to the
16 * synchronous exec. The shared request helpers (authorize, register,
17 * subject-token extract, bearer append, session-xfer note, dashboard
18 * identity) live in tpc.c and are reached through the split header.
19 */

41static ngx_int_t
42webdav_tpc_push_apply_credential(ngx_http_request_t *r,
43 ngx_http_brix_webdav_loc_conf_t *conf,
44 char *dest_url, ngx_array_t *headers)
45{
46 ngx_table_elt_t *credential_hdr;
47 ngx_table_elt_t *auth_hdr;
48 ngx_str_t delegated_token;
49 ngx_int_t rc;
50 brix_tpc_cred_mode_e mode;
51 const char *subject_token;
52
53 credential_hdr = webdav_tpc_find_header(r, "Credential",
54 sizeof("Credential") - 1);
55 if (credential_hdr == NULL) {
56 credential_hdr = webdav_tpc_find_header(r, "Credentials",
57 sizeof("Credentials") - 1);
58 }
59
60 if (credential_hdr == NULL
61 || webdav_tpc_header_value_equals(&credential_hdr->value, "none"))
62 {
63 return NGX_OK;
64 }
65
66 mode = webdav_tpc_cred_parse_mode(
67 (const char *) credential_hdr->value.data, credential_hdr->value.len);
68 if (mode == BRIX_TPC_CRED_UNKNOWN) {
69 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
70 "brix_webdav: unsupported HTTP-TPC credential "
71 "delegation mode \"%V\"", &credential_hdr->value);
72 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_BAD_REQUEST]);
73 return NGX_HTTP_BAD_REQUEST;
74 }
75
76 auth_hdr = webdav_tpc_find_header(r, "Authorization",
77 sizeof("Authorization") - 1);
78 subject_token = NULL;
79 rc = webdav_tpc_extract_subject_token(r, auth_hdr, &subject_token);
80 if (rc != NGX_OK) {
81 return NGX_HTTP_INTERNAL_SERVER_ERROR;
82 }
83
84 rc = webdav_tpc_cred_obtain_token(r, mode, dest_url, subject_token,
85 conf->tpc_cred.token_scope.len > 0
86 ? (const char *) conf->tpc_cred.token_scope.data
87 : "storage.read",
88 &delegated_token);
89 if (rc != NGX_OK) {
90 return NGX_HTTP_BAD_GATEWAY;
91 }
92
93 return webdav_tpc_add_bearer_header(r, headers, &delegated_token);
94}

96static ngx_int_t
97webdav_tpc_push_dest_url(ngx_http_request_t *r, ngx_table_elt_t *dest_hdr,
98 char **dest_url)
99{
100 if (dest_hdr->value.len < sizeof("https://") - 1
101 || ngx_strncasecmp(dest_hdr->value.data, (u_char *) "https://",
102 sizeof("https://") - 1) != 0
103 || webdav_tpc_str_has_ctl(dest_hdr->value.data, dest_hdr->value.len))
104 {
105 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
106 "brix_webdav: HTTP-TPC push Destination must be"
107 " an https URL");
108 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_BAD_REQUEST]);
109 return NGX_HTTP_BAD_REQUEST;
110 }
111
112 *dest_url = webdav_tpc_pstrndup0(r->pool, dest_hdr->value.data,
113 dest_hdr->value.len);
114 return (*dest_url == NULL) ? NGX_HTTP_INTERNAL_SERVER_ERROR : NGX_OK;
115}

117static ngx_int_t
118webdav_tpc_push_add_overwrite_value(ngx_http_request_t *r,
119 ngx_array_t *headers,
120 ngx_table_elt_t *overwrite_hdr)
121{
122 ngx_str_t *dst;
123
124 dst = ngx_array_push(headers);
125 if (dst == NULL) {
126 return NGX_HTTP_INTERNAL_SERVER_ERROR;
127 }
128
129 if (overwrite_hdr->value.len == 1 && overwrite_hdr->value.data[0] == 'F') {
130 dst->len = sizeof("If-None-Match: *") - 1;
131 dst->data = (u_char *) "If-None-Match: *";
132 return NGX_OK;
133 }
134
135 dst->len = sizeof("Overwrite: ") - 1 + overwrite_hdr->value.len;
136 dst->data = ngx_pnalloc(r->pool, dst->len + 1);
137 if (dst->data == NULL) {
138 return NGX_HTTP_INTERNAL_SERVER_ERROR;
139 }
140 ngx_memcpy(dst->data, "Overwrite: ", sizeof("Overwrite: ") - 1);
141 ngx_memcpy(dst->data + sizeof("Overwrite: ") - 1,
142 overwrite_hdr->value.data, overwrite_hdr->value.len);
143 dst->data[dst->len] = '\0';
144 return NGX_OK;
145}

199/*
200 * WHAT: Bundle of the shared state threaded through the file-local HTTP-TPC push
201 * exec helpers (currently the synchronous fallback tier).
202 * WHY: These helpers otherwise pass the same request/config/endpoint/header set
203 * through many positional parameters; collecting them into one stack-local
204 * context keeps each helper at =5 params and makes the shared data flow
205 * explicit without introducing any global state.
206 * HOW: Zero-initialised with ngx_memzero by the push handler, then populated
207 * with the exact values previously passed positionally and handed to each
208 * helper by pointer. Fields carry identical values and are read in the same
209 * order, so behaviour (curl PUT, credential headers, registry/dashboard
210 * ordering) is unchanged.
211 */
212typedef struct {
213 ngx_http_request_t *r; /* current request */
214 ngx_http_brix_webdav_loc_conf_t *conf; /* location config */
215 char *dest_url; /* remote https destination */
216 char *path; /* local source path */
217 ngx_array_t *headers; /* outbound transfer headers */
218 off_t expected_bytes;/* source size for accounting */
219} webdav_tpc_push_ctx_t;
220
221static ngx_int_t
222webdav_tpc_push_sync_exec(webdav_tpc_push_ctx_t *px)
223{
224 ngx_http_request_t *r = px->r;
225 ngx_http_brix_webdav_loc_conf_t *conf = px->conf;
226 char *dest_url = px->dest_url;
227 char *path = px->path;
228 ngx_array_t *headers = px->headers;
229 off_t expected_bytes = px->expected_bytes;
230 ngx_int_t rc;
231 uint64_t transfer_id;
232
233 transfer_id = webdav_tpc_register_transfer(r, BRIX_TPC_DIR_PUSH, path,
234 dest_url, expected_bytes);
235 if (transfer_id == 0) {
236 brix_dashboard_http_error(r, "webdav TPC registry full");
237 brix_dashboard_http_finish(r);
238 return NGX_HTTP_SERVICE_UNAVAILABLE;
239 }
240
241 rc = webdav_tpc_run_curl_push(r->connection->log, conf, dest_url, path,
242 headers, transfer_id);
243 if (rc != NGX_OK) {
244 return webdav_tpc_push_sync_error(r, transfer_id, path, expected_bytes,
245 rc);
246 }
247
248 (void) brix_tpc_registry_update(transfer_id, expected_bytes,
249 BRIX_TPC_STATE_DONE, r->connection->log);
250 brix_xfer_finish(BRIX_XFER_TPC, "out", path, NULL,
251 (size_t) expected_bytes, BRIX_XFER_OK, 0,
252 r->connection->log);
253 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV, BRIX_TPC_DIR_PUSH,
254 BRIX_TPC_METRIC_SUCCESS,
255 (size_t) expected_bytes, r->connection->log);
256 (void) brix_tpc_registry_remove(transfer_id, r->connection->log);
257 brix_dashboard_http_add(r, (ngx_atomic_int_t) expected_bytes);
258 brix_dashboard_http_finish(r);
259 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_PUSH_SUCCESS]);
260 webdav_tpc_note_client_copy_xfer(r, expected_bytes,
261 (int64_t) expected_bytes);
262 return webdav_send_no_body(r, NGX_HTTP_CREATED);
263}

265/*
266 * webdav_tpc_handle_push — HTTP-TPC push: read a local file and PUT it to a
267 * remote HTTPS destination.
268 *
269 * The request URI identifies the local source file. The Destination: header
270 * carries the remote URL to push to. No temp-file dance is needed: the local
271 * file is only read, never written.
272 *
273 * OAuth2/OIDC delegation: when Credential: header is "oidc-agent" or
274 * "token-exchange", the server obtains a delegated access token for the
275 * remote destination and injects it as an Authorization: Bearer header.
276 */
277ngx_int_t
278webdav_tpc_handle_push(ngx_http_request_t *r,
279 ngx_http_brix_webdav_loc_conf_t *conf,
280 ngx_table_elt_t *dest_hdr)
281{
282 ngx_array_t *transfer_headers = NULL;
283 char *dest_url;
284 char path[WEBDAV_MAX_PATH];
285 struct stat sb;
286 ngx_int_t rc;
287 ngx_str_t src_scope;
288
289 rc = webdav_tpc_push_dest_url(r, dest_hdr, &dest_url);
290 if (rc != NGX_OK) {
291 return rc;
292 }
293
294 rc = webdav_tpc_collect_transfer_headers(r, &transfer_headers);
295 if (rc != NGX_OK) {
296 return rc;
297 }
298
299 /*
300 * Parse the Credential header. "none" (or absent) means no delegation.
301 * "oidc-agent" or "token-exchange" triggers token acquisition for the
302 * remote destination.
303 */
304 rc = webdav_tpc_push_apply_credential(r, conf, dest_url, transfer_headers);
305 if (rc != NGX_OK) {
306 return rc;
307 }
308
309 rc = webdav_resolve_stat(r, path, sizeof(path), &sb);
310 if (rc != NGX_OK) {
311 return rc;
312 }
313 if (S_ISDIR(sb.st_mode)) {
314 return NGX_HTTP_CONFLICT;
315 }
316
317 src_scope = r->uri;
318 rc = webdav_tpc_authorize(r, &src_scope, NULL);
319 if (rc != NGX_OK) {
320 return rc;
321 }
322
323 (void) brix_dashboard_http_start_identity(r, path,
324 webdav_dashboard_identity(r), "", BRIX_XFER_PROTO_WEBDAV,
325 BRIX_XFER_DIR_TPC, "TPC_PUSH", (int64_t) sb.st_size);
326 brix_dashboard_http_tpc_remote(r, dest_url, 0, 0);
327
328 /* Forward Overwrite header if present.
329 * WebDAV COPY uses Overwrite: F, but the outbound TPC push uses PUT.
330 * PUT doesn't support Overwrite: F; it uses If-None-Match: *. */
331 rc = webdav_tpc_push_forward_overwrite(r, transfer_headers);
332 if (rc != NGX_OK) {
333 return rc;
334 }
335
336 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_PUSH_STARTED]);
337 brix_tpc_metric_transfer(BRIX_TPC_PROTO_WEBDAV, BRIX_TPC_DIR_PUSH,
338 BRIX_TPC_METRIC_STARTED, 0,
339 r->connection->log);
340
341 /* Preferred path: hand the push off to the thread pool so the event loop is
342 * never blocked on curl. NGX_DECLINED means "no thread pool configured" and
343 * we fall through to running curl synchronously below; NGX_DONE means the
344 * task was queued (response finalized later); any other rc is terminal. */
345 rc = webdav_tpc_push_post_thread(r, conf, dest_url, path,
346 transfer_headers);
347 if (rc != NGX_DECLINED) {
348 return rc;
349 }
350
351 /* Synchronous fallback (blocks the worker for the duration of the push). */
352 {
353 webdav_tpc_push_ctx_t px;
354
355 ngx_memzero(&px, sizeof(px));
356 px.r = r;
357 px.conf = conf;
358 px.dest_url = dest_url;
359 px.path = path;
360 px.headers = transfer_headers;
361 px.expected_bytes = sb.st_size;
362 return webdav_tpc_push_sync_exec(&px);
363 }
364}
