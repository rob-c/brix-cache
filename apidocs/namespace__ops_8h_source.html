48/*
49 * Shared namespace mutation APIs.
50 *
51 * These operate on already-resolved, confined paths. They do not perform
52 * wire path parsing or token/ACL checks; protocol handlers do those first.
53 *
54 * INVARIANT: All protocol handlers implementing user-visible namespace
55 * mutations (delete, mkdir, rename) MUST route through the VFS layer
56 * (src/fs/vfs/vfs.h: brix_vfs_unlink/rmdir/mkdir/rename/copy and the staged-write
57 * family), which calls these functions underneath while adding the metrics +
58 * access-log layer. These brix_ns_*() entry points remain the seam the VFS
59 * delegates to; calling them directly from an EVENT-LOOP handler path is now a
60 * second choice (it loses the OP metric/log) and direct brix_*_confined*()
61 * calls from such paths are a defect.
62 *
63 * EXEMPT — namespace MUTATION running on a THREAD-POOL worker (off the event
64 * loop): native TPC pull, async/multipart S3 PUT assembly, the collection
65 * copy/move engines. The metered VFS namespace entry points allocate from an
66 * nginx pool and emit metrics/log lines, none of which are thread-safe, so
67 * worker-thread namespace mutation stays on this brix_ns_*() / confined-helper
68 * / staged_file tier. Confinement is identical (RESOLVE_BENEATH); only the VFS
69 * metering is skipped.
70 *
71 * NOTE (phase-54): this exemption is for namespace MUTATION only. Raw byte I/O
72 * (read/write/readv/writev/pgread + the dirlist scan) is no longer exempt —
73 * worker threads now run it through the VFS-owned thread-safe core
74 * brix_vfs_io_execute() (src/fs/vfs/vfs_io_core.c), so that path is unified across
75 * all dispatch tiers. See src/fs/README.md "Two VFS tiers".
76 */

78/*
79 * brix_ns_delete — remove a file or directory, kernel-confined under root_canon.
80 * path is the absolute path (borrowed, not freed); it is stripped to a
81 * within-root tail and all touches go through the openat2 RESOLVE_BENEATH API,
82 * so a path that escapes root_canon is refused (status DENIED, errno EXDEV)
83 * rather than acted on. opts selects: recursive tree removal, idempotent_missing
84 * (ENOENT becomes OK with .existed=0), require_empty_dir (non-empty dir =>
85 * NOT_EMPTY), require_directory (rmdir semantics: a regular file => DENIED,
86 * errno ENOTDIR). Returns a by-value result: .status is the neutral code,
87 * .sys_errno the raw errno on failure, and .existed/.was_dir describe the target
88 * pre-delete. Opens and closes its own transient rootfd; no allocation.
89 */
90brix_ns_result_t brix_ns_delete(ngx_log_t *log,
91 const char *root_canon, const char *path,
92 const brix_ns_delete_opts_t *opts);

94/*
95 * brix_ns_mkdir — create directory path beneath root_canon (RESOLVE_BENEATH).
96 * mode is the raw mkdir(2) mode (subject to the process umask). recursive => mkdir -p
97 * (creates each missing parent component). path borrowed, not freed. Returns a
98 * by-value result: .status OK with .created=1 on success; an existing target
99 * yields status EXISTS (errno EEXIST); a path outside root_canon yields DENIED.
100 * Opens/closes its own rootfd; no heap allocation.
101 */
102brix_ns_result_t brix_ns_mkdir(ngx_log_t *log,
103 const char *root_canon, const char *path, mode_t mode,
104 ngx_flag_t recursive);

106/*
107 * brix_ns_rename — atomically move src to dst; BOTH must be under the SAME
108 * root_canon (cross-root rename is refused with status DENIED, errno EXDEV).
109 * src and dst are borrowed absolute paths. If dst already exists and is a
110 * directory: overwrite_dirs=1 first recursively removes the dst tree (rename(2)
111 * only replaces empty dirs), overwrite_dirs=0 returns status EXISTS. An
112 * existing non-directory dst is replaced atomically by renameat. Returns a
113 * by-value result; .existed/.was_dir describe a pre-existing dst. Confinement
114 * is enforced inside the kernel rename, closing the realpath-to-rename TOCTOU.
115 */
116brix_ns_result_t brix_ns_rename(ngx_log_t *log,
117 const char *root_canon, const char *src, const char *dst,
118 ngx_flag_t overwrite_dirs);

120/*
121 * brix_ns_local_copy — copy a single regular file src to dst within root_canon
122 * via copy_file_range(2) (falls back to read/write inside brix_copy_range).
123 * src/dst borrowed; both must be under root_canon (else DENIED/EXDEV).
124 * Directory sources are rejected (status CONFLICT — recursive copy not handled
125 * here). dst handling: !overwrite + existing dst => EXISTS; existing dst dir
126 * without overwrite_dirs => EXISTS. opts->staged_commit writes a temp file and
127 * atomically renames it into place on success (aborted/unlinked on failure);
128 * opts->preserve_xattrs copies XRootD fattr xattrs after the data copy.
129 * Returns a by-value result: OK with .created=1 on success. Opens/closes the
130 * src/dst fds and rootfd itself.
131 */
