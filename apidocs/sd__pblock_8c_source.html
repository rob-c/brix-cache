1/*
2 * sd_pblock.c — the pblock ("pseudo-block") Storage Driver: a full-capability,
3 * block-based drop-in for POSIX.
4 *
5 * WHAT: Implements brix_sd_pblock_driver, a complete backend that stores each
6 * object's bytes striped across fixed-size POSIX "block" files and the
7 * entire logical namespace + metadata in a SQLite catalog
8 * (sd_pblock_catalog.c). It advertises the same capabilities as the POSIX
9 * driver and implements every vtable slot. This file owns the instance and
10 * object lifecycle (init/cleanup, open/close and their helpers) and the
11 * driver descriptor; the byte I/O (sd_pblock_io.c), namespace/directory/
12 * xattr ops (sd_pblock_namespace.c) and staged atomic publish
13 * (sd_pblock_staged.c) live in siblings reached through
14 * sd_pblock_internal.h.
15 *
16 * WHY: Striping bulk content into fixed-size blocks (default 64 MiB, set per
17 * file at creation and configurable per export) is the defining property
18 * of a block backend, and splitting that content from the namespace/
19 * metadata keeps the hot byte path free of any database work. SQLite is
20 * touched only at metadata boundaries. Split from a single 1111-line file
21 * (phase-79) to hold every pblock file under the ~500-line, one-concept cap.
22 *
23 * HOW: An object's bytes live at <data>/<aa>/<bb>/<blob_id>/<block_index>;
24 * block 0 is opened persistently as obj->fd (so small files keep a real
25 * fd for zero-copy sendfile), and higher blocks are opened transiently per
26 * I/O. Reads/writes map [off,off+len) across blocks (holes read as zeros);
27 * writes update the cached size/mtime in memory and flush to the catalog
28 * on fsync/close. The whole file is ngx-free (libc + sqlite, malloc-owned
29 * state) so it is identical in the module and the standalone unit test.
30 * Compiled only when the build found libsqlite3 (BRIX_HAVE_SQLITE).
31 */

73static ngx_int_t
74sd_pblock_init(brix_sd_instance_t *inst, void *driver_conf)
75{
76 const brix_sd_pblock_conf_t *conf = driver_conf;
77 pblock_state_t *st;
78 char db[PATH_MAX];
79
80 if (conf == NULL || conf->root == NULL) {
81 errno = EINVAL;
82 return NGX_ERROR;
83 }
84
85 st = calloc(1, sizeof(*st));
86 if (st == NULL) {
87 errno = ENOMEM;
88 return NGX_ERROR;
89 }
90 snprintf(st->root, sizeof(st->root), "%s", conf->root);
91 snprintf(st->data_dir, sizeof(st->data_dir), "%s/data", conf->root);
92 st->block_size = conf->block_size > 0 ? conf->block_size
93 : PBLOCK_DEFAULT_BLOCK_SIZE;
94
95 if (pblock_mkdir_p(st->root) != 0 || pblock_mkdir_p(st->data_dir) != 0) {
96 int err = errno;
97
98 free(st);
99 errno = err;
100 return NGX_ERROR;
101 }
102
103 snprintf(db, sizeof(db), "%s/catalog.db", conf->root);
104 st->cat = pblock_catalog_open(db, conf->busy_timeout_ms);
105 if (st->cat == NULL) {
106 int err = errno;
107
108 free(st);
109 errno = err;
110 return NGX_ERROR;
111 }
112
113 /* Phase-83 lab control plane. The static opts sidecar (<root>/pblock.opts,
114 * written by the config finaliser when it strips a `?tail` off a pblock://
115 * root) selects the fail-closed master gate and the caps/mem knobs. With
116 * lab OFF (the default — no sidecar) the driver stays byte-for-byte the
117 * production backend: st->lab is NULL and the hot path never consults it. */
118 {
119 pblock_opts_t opts;
120
121 (void) pblock_opts_load_sidecar(st->root, &opts); /* absent ⇒ all-zero */
122
123 if (opts.mem) { /* F16 */
124 (void) pblock_ctl_mem_pragmas(st->cat);
125 }
126 if (opts.lab) {
127 inst->caps = pblock_caps_apply(inst->caps, &opts); /* F2 */
128 st->lab = pblock_lab_state_create(st->cat); /* F1/F8 */
129 if (st->lab != NULL) {
130 /* F9 rides the lab master gate (an anomaly simulator is a lab
131 * toy by definition). Best-effort: with no `recent` table the
132 * event writers no-op and consultation finds nothing. */
133 (void) pblock_anomaly_init(st);
134 }
135 }
136 if (opts.audit) { /* F17 */
137 /* Independent of the lab gate (its own opt): only turn audit on if
138 * the oplog table is actually present, so a create failure leaves
139 * the driver in the byte-for-byte production path (audit off). */
140 st->audit = (pblock_audit_init(st->cat) == 0);
141 }
142 if (opts.csi) { /* F3 */
143 /* Its own opt (integrity, not a lab toy). Advertise CAP_FSCS only
144 * when the csi table is actually present — an honest per-instance
145 * capability the driver will really honour. */
146 if (pblock_csi_init(st->cat) == 0) {
147 st->csi = 1;
148 inst->caps |= BRIX_SD_CAP_FSCS;
149 }
150 }
151 if (opts.nearline) { /* F4 */
152 /* Its own opt: advertise CAP_NEARLINE only when the residency
153 * table actually installed — an honest per-instance capability
154 * (the residency seam and the cache tier's recall-at-fill key off
155 * it). Init failure ⇒ production driver, no nearline claims. */
156 if (pblock_nearline_init(st) == 0) {
157 st->nearline = 1;
158 inst->caps |= BRIX_SD_CAP_NEARLINE;
159 }
160 }
161 if (opts.locks) { /* F15 */
162 /* Its own opt: mandatory lease enforcement only arms when the
163 * locks table actually installed — an init failure leaves the
164 * byte-for-byte production path (no lease reads anywhere). */
165 st->locks = (pblock_locks_init(st) == 0);
166 }
167 if (opts.dedup) { /* F10 */
168 /* Its own opt: refcounted blobs + dedup only arm when the blobs
169 * table actually installed — an init failure leaves the
170 * byte-for-byte production path (no refcount reads anywhere). */
171 st->refs = (pblock_refs_init(st) == 0);
172 }
173 if (opts.snapshots) { /* F6 */
174 /* Snapshots build ON refcounted blobs (F10): a delete between take
175 * and restore must decrement a shared blob, never physically remove
176 * it. Arm refs first (idempotent), then the snapshot tables — armed
177 * only when both installed, else the production path is untouched. */
178 if (!st->refs) {
179 st->refs = (pblock_refs_init(st) == 0);
180 }
181 if (st->refs && pblock_snap_init(st) == 0) {
182 st->snap = 1;
183 }
184 }
185 if (opts.versions > 0 || opts.trash) { /* F11 */
186 /* Versioning and trash both HOLD prior blobs (an overwrite/unlink
187 * decrements a shared blob instead of freeing it), so they build ON
188 * refcounted blobs (F10) exactly like snapshots. Arm refs first
189 * (idempotent), then the history tables — armed only when both
190 * installed, else the byte-for-byte production path is untouched. */
191 if (!st->refs) {
192 st->refs = (pblock_refs_init(st) == 0);
193 }
194 if (st->refs && pblock_hist_init(st) == 0) {
195 st->versions = opts.versions;
196 st->trash = opts.trash;
197 }
198 }
199 if (opts.xform_len > 0) { /* F12/F13 */
200 /* Per-block transform. Unlike the lab toys this is a hard config
201 * gate: a bad spec (unknown transform, unreadable crypt keyfile,
202 * `zstd` without libzstd) fails instance init fail-closed — the
203 * pblock store is never built for the export (logged "backend init
204 * failed"), rather than silently serving unreadable transformed
205 * bytes as raw. A transformed export cannot hand out a block-0 fd,
206 * so drop the zero-copy caps at build (per-export mask, doc §5). */
207 if (pblock_xform_config(&st->xform, opts.xform, opts.xform_len)
208 != 0)
209 {
210 int err = errno;
211
212 pblock_catalog_close(st->cat);
213 free(st);
214 errno = err;
215 return NGX_ERROR;
216 }
217 if (pblock_xform_active(&st->xform)) {
218 inst->caps &= ~(uint32_t) (BRIX_SD_CAP_SENDFILE
219 | BRIX_SD_CAP_IOURING);
220 }
221 }
222 if (opts.quota_bytes > 0 || opts.quota_inodes > 0) { /* F5 */
223 /* Its own opt like csi/audit: quota only arms when the rollup
224 * table + triggers actually installed, so an init failure leaves
225 * the byte-for-byte production catalog path. */
226 if (pblock_quota_init(st) == 0) {
227 st->quota = 1;
228 st->quota_bytes = opts.quota_bytes;
229 st->quota_inodes = opts.quota_inodes;
230 }
231 }
232 }
233
234 /* The export root "/" always exists (a directory), like a POSIX mount point —
235 * so stat("/")/opendir("/")/PROPFIND on the root succeed before anything is
236 * written. Created once; harmless if a concurrent worker also creates it.
237 * Sticky world-writable (/tmp semantics) so identity-enforced multi-user
238 * exports work out of the box: anyone may create top-level entries, only
239 * owners may remove them (pblock_ident_sticky_gate). Operators wanting a
240 * stricter top level chmod "/" or pre-create VO directories. */
241 if (pblock_catalog_lookup(st->cat, "/", NULL) == 1) {
242 pblock_meta root_meta;
243
244 memset(&root_meta, 0, sizeof(root_meta));
245 root_meta.is_dir = 1;
246 root_meta.mtime = root_meta.ctime = pblock_now();
247 root_meta.mode = S_IFDIR | S_ISVTX | 0777;
248 (void) pblock_catalog_put(st->cat, "/", &root_meta);
249 }
250
251 inst->state = st;
252 return NGX_OK;
253}

270/* pblock_make_obj — wrap a block-0 fd + cached metadata in a heap object. On
271 * failure the fd is closed and NULL/errno returned. */
272static brix_sd_obj_t *
273pblock_make_obj(brix_sd_instance_t *inst, const char *path, int fd,
274 const pblock_meta *meta)
275{
276 brix_sd_obj_t *obj;
277 pblock_obj_t *os;
278
279 obj = calloc(1, sizeof(*obj));
280 os = calloc(1, sizeof(*os));
281 if (obj == NULL || os == NULL) {
282 free(obj);
283 free(os);
284 if (fd >= 0) {
285 close(fd);
286 }
287 errno = ENOMEM;
288 return NULL;
289 }
290
291 os->st = inst->state;
292 os->meta = *meta;
293 os->block_size = meta->block_size;
294 snprintf(os->path, sizeof(os->path), "%s", path);
295 snprintf(os->blob_id, sizeof(os->blob_id), "%s", meta->blob_id);
296
297 /* Phase-83: resolve any lab fault/shape rule for this handle ONCE, at open
298 * (a metadata boundary). NULL when lab is off or no rule applies — the hot
299 * byte path then skips the gate entirely. */
300 os->lab = pblock_lab_snapshot(os->st->lab, path);
301
302 /* F3: snapshot the at-rest per-block CRCs ONCE at open so the hot read path
303 * verifies against memory (no DB). csi_dlo/dhi is the empty written extent
304 * (dlo > dhi) until the first write widens it. */
305 os->csi_dlo = INT64_MAX;
306 os->csi_dhi = 0;
307 if (os->st->csi && !meta->is_dir) {
308 int64_t nblk = meta->size > 0
309 ? pblock_last_block(meta->size, os->block_size) + 1 : 0;
310
311 (void) pblock_csi_load(os->st->cat, os->blob_id, nblk,
312 &os->csi_crc, &os->csi_n);
313 }
314
315 /* F5: snapshot how far this handle may grow the file ONCE at open (quota
316 * off ⇒ INT64_MAX) — pwrite/copy_range enforce it without touching the DB. */
317 os->quota_max = pblock_quota_max_size(os->st, meta->uid, meta->size);
318
319 obj->driver = inst->driver;
320 obj->inst = inst;
321 obj->fd = fd >= 0 ? fd : NGX_INVALID_FILE;
322 obj->state = os;
323 obj->heap_shell = 1; /* malloc'd shell; a VFS adopter frees it after copy */
324
325 /* Capture the open-time metadata snapshot the VFS reports to callers (file
326 * size, mode, mtime) — without this the adopting handle would see a zeroed
327 * snap and treat the file as empty (immediate EOF on read). */
328 ngx_memzero(&obj->snap, sizeof(obj->snap));
329 obj->snap.size = meta->size;
330 obj->snap.mtime = meta->mtime;
331 obj->snap.ctime = meta->ctime;
332 obj->snap.mode = meta->mode;
333 obj->snap.ino = (ino_t) pblock_fnv(path);
334 obj->snap.uid = meta->uid;
335 obj->snap.gid = meta->gid;
336 obj->snap.is_dir = meta->is_dir ? 1 : 0;
337 obj->snap.is_reg = meta->is_dir ? 0 : 1;
338
339 /* F6: count live regular-file handles so a snapshot restore can refuse
340 * (EBUSY) rather than swap the namespace out from under an open file.
341 * Balanced by the matching decrement in sd_pblock_close. */
342 if (os->st->snap && !meta->is_dir) {
343 __atomic_add_fetch(&os->st->open_files, 1, __ATOMIC_RELEASE);
344 }
345 return obj;
346}

348/* pblock_open_create — create a brand-new file: a fresh per-object dir + block 0,
349 * plus its catalog row owned by (uid, gid) — 0/0 for the service itself.
350 * Returns the object or NULL/errno (*err_out set). */
351static brix_sd_obj_t *
352pblock_open_create(brix_sd_instance_t *inst, const char *path, mode_t mode,
353 uint32_t uid, uint32_t gid, int *err_out)
354{
355 pblock_state_t *st = inst->state;
356 pblock_meta meta;
357 brix_sd_obj_t *obj;
358 char block0[PATH_MAX];
359 int fd;
360
361 if (pblock_quota_admit(st, uid, 0, 1) != 0) { /* F5 */
362 if (err_out != NULL) { *err_out = errno; }
363 return NULL;
364 }
365
366 memset(&meta, 0, sizeof(meta));
367 if (pblock_gen_blob_id(meta.blob_id) != 0
368 || pblock_ensure_obj_dir(st, meta.blob_id) != 0
369 || pblock_block_path(st, meta.blob_id, 0, block0, sizeof(block0)) != 0)
370 {
371 if (err_out != NULL) { *err_out = errno; }
372 return NULL;
373 }
374
375 fd = open(block0, O_RDWR | O_CREAT | O_EXCL, 0600);
376 if (fd < 0) {
377 if (err_out != NULL) { *err_out = errno; }
378 return NULL;
379 }
380
381 meta.is_dir = 0;
382 meta.size = 0;
383 meta.block_size = st->block_size;
384 meta.mtime = meta.ctime = pblock_now();
385 meta.mode = S_IFREG | (mode & 0777);
386 meta.uid = uid;
387 meta.gid = gid;
388 snprintf(meta.xform, sizeof(meta.xform), "%s", /* F12/F13: record kind */
389 pblock_xform_name(st->xform.kind));
390
391 if (pblock_catalog_put(st->cat, path, &meta) != 0) {
392 int err = errno;
393
394 close(fd);
395 pblock_remove_blocks(st, meta.blob_id, 0, st->block_size);
396 if (err_out != NULL) { *err_out = err; }
397 return NULL;
398 }
399
400 obj = pblock_make_obj(inst, path, fd, &meta);
401 if (obj == NULL && err_out != NULL) {
402 *err_out = errno;
403 }
404 if (obj != NULL && st->refs) { /* F10 */
405 (void) pblock_refs_track(st, meta.blob_id, 0, meta.block_size, 0, 0);
406 ((pblock_obj_t *) obj->state)->wv = brix_wverify_begin();
407 }
408 return obj;
409}

411/* pblock_open_existing — open an existing file's block 0; honours O_TRUNC on
412 * write via the block-aware truncate. Returns the object or NULL/errno. */
413static brix_sd_obj_t *
414pblock_open_existing(brix_sd_instance_t *inst, const char *path,
415 pblock_meta *meta, int sd_flags, int *err_out)
416{
417 pblock_state_t *st = inst->state;
418 brix_sd_obj_t *obj;
419 char block0[PATH_MAX];
420 int want_write = sd_flags & BRIX_SD_O_WRITE;
421 int fd;
422
423 /* F12/F13: refuse to serve a file whose recorded transform does not match the
424 * export's configured one (wrong key/codec ⇒ garbage). A config error, not a
425 * data error — fail the open rather than hand back undecodable bytes. */
426 if (pblock_xform_kind_from_name(meta->xform) != st->xform.kind) {
427 if (err_out != NULL) { *err_out = EIO; }
428 return NULL;
429 }
430
431 /* F10: never write through a shared blob — a write-intent open first gives
432 * this object a private copy (empty on O_TRUNC: the bytes are doomed). */
433 if (st->refs && want_write
434 && pblock_refs_break_share(st, path, meta,
435 (sd_flags & BRIX_SD_O_TRUNC) != 0) != 0)
436 {
437 if (err_out != NULL) { *err_out = errno; }
438 return NULL;
439 }
440
441 if (pblock_block_path(st, meta->blob_id, 0, block0, sizeof(block0)) != 0) {
442 if (err_out != NULL) { *err_out = errno; }
443 return NULL;
444 }
445 fd = open(block0, want_write ? O_RDWR : O_RDONLY);
446 if (fd < 0) {
447 if (err_out != NULL) { *err_out = errno; }
448 return NULL;
449 }
450
451 obj = pblock_make_obj(inst, path, fd, meta);
452 if (obj == NULL) {
453 if (err_out != NULL) { *err_out = errno; }
454 return NULL;
455 }
456
457 if (want_write && (sd_flags & BRIX_SD_O_TRUNC) && meta->size > 0) {
458 if (sd_pblock_ftruncate(obj, 0) != NGX_OK) {
459 int err = errno;
460
461 obj->driver->close(obj); /* frees state + closes fd */
462 free(obj); /* … and the malloc'd shell */
463 if (err_out != NULL) { *err_out = err; }
464 return NULL;
465 }
466 }
467 if (st->refs && want_write && (sd_flags & BRIX_SD_O_TRUNC)) {
468 /* F10: content is being rewritten from scratch — this handle can grow
469 * an honest whole-object CRC, making the blob a dedup candidate. */
470 ((pblock_obj_t *) obj->state)->wv = brix_wverify_begin();
471 }
472 return obj;
473}

475/* sd_pblock_open_as — the open implementation, parameterized by the owner
476 * (uid, gid) recorded on a CREATE. The plain vtable slot passes 0/0 (the
477 * service); the identity-aware open_cred slot (sd_pblock_cred.c) passes the
478 * requester's resolved catalog ids so new files are owned by their creator. */
479static brix_sd_obj_t *
480pblock_open_as_inner(brix_sd_instance_t *inst, const char *path, int sd_flags,
481 mode_t mode, uint32_t uid, uint32_t gid, int *err_out)
482{
483 pblock_state_t *st = inst->state;
484 pblock_meta meta;
485 int rc;
486
487 rc = pblock_catalog_lookup(st->cat, path, &meta);
488 if (rc < 0) {
489 if (err_out != NULL) { *err_out = errno; }
490 return NULL;
491 }
492
493 /* F15: mandatory lease gate — before both branches, since a whole-file
494 * lease guards the *name* (a create under a foreign lease is as much a
495 * conflicting write as an overwrite). Live foreign 'X' refuses everyone;
496 * a live foreign whole-file 'W' refuses write-intent opens. EBUSY maps
497 * to kXR_FileLocked / HTTP 423 at the protocol edge. */
498 if (st->locks
499 && pblock_locks_open_check(st, path,
500 (sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE
501 | BRIX_SD_O_TRUNC | BRIX_SD_O_APPEND)) != 0,
502 uid) != 0)
503 {
504 if (err_out != NULL) { *err_out = errno; }
505 return NULL;
506 }
507
508 if (rc == 0 && meta.is_dir) {
509 if (sd_flags & BRIX_SD_O_WRITE) {
510 if (err_out != NULL) { *err_out = EISDIR; }
511 return NULL;
512 }
513 return pblock_make_obj(inst, path, NGX_INVALID_FILE, &meta);
514 }
515
516 if (rc == 0) { /* existing file */
517 /* F9: visibility lag — a freshly created path is ENOENT to other
518 * readers for the armed window. Write-intent opens are exempt: the
519 * writer/overwriter lane must never see its own phantom ENOENT (the
520 * S3 session monotonic-read guarantee — a handle that wrote reads
521 * through its own open snapshot and is untouched by design). */
522 if (st->lab != NULL
523 && !(sd_flags & (BRIX_SD_O_WRITE | BRIX_SD_O_CREATE
524 | BRIX_SD_O_TRUNC))
525 && pblock_anomaly_hidden(st, path))
526 {
527 if (err_out != NULL) { *err_out = ENOENT; }
528 return NULL;
529 }
530 if ((sd_flags & BRIX_SD_O_CREATE) && (sd_flags & BRIX_SD_O_EXCL)) {
531 if (err_out != NULL) { *err_out = EEXIST; }
532 return NULL;
533 }
534 /* F4: a demoted (nearline/offline) file must be recalled before it
535 * serves — a bounded synchronous recall like sd_frm's, so a plain
536 * export exercises the lane without a cache tier in front. A LOST
537 * object or failed recall errors out (never silently serves the bytes
538 * a real tape system could not). O_TRUNC replaces the content, so it
539 * skips the recall and comes back ONLINE below via the write path. */
540 if (st->nearline && !(sd_flags & BRIX_SD_O_TRUNC)
541 && pblock_nearline_recall(st, path) != 0)
542 {
543 if (err_out != NULL) { *err_out = errno; }
544 return NULL;
545 }
546 return pblock_open_existing(inst, path, &meta, sd_flags, err_out);
547 }
548
549 /* absent */
550 if (!(sd_flags & BRIX_SD_O_CREATE)) {
551 if (err_out != NULL) { *err_out = ENOENT; }
552 return NULL;
553 }
554 {
555 brix_sd_obj_t *obj;
556
557 obj = pblock_open_create(inst, path, mode, uid, gid, err_out);
558 if (obj != NULL && st->lab != NULL) {
559 pblock_anomaly_created(st, path); /* F9 */
560 }
561 return obj;
562 }
563}

565/* sd_pblock_open_as — the identity-parameterised open, wrapping the real
566 * implementation with the F17 audit record. The wrapper (not the inner) is the
567 * audit boundary so every terminal outcome — dir open, existing file, create,
568 * or an error — is recorded exactly once with the actor identity and result. */
569brix_sd_obj_t *

605static ngx_int_t
606sd_pblock_close(brix_sd_obj_t *obj)
607{
608 pblock_obj_t *os;
609 ngx_int_t rc = NGX_OK;
610
611 if (obj == NULL) {
612 return NGX_OK;
613 }
614 os = obj->state;
615
616 if (os != NULL && os->dirty) {
617 pblock_meta old;
618 int have_old = 0;
619
620 /* F9: the pre-update row is what a stale stat will serve — capture it
621 * before the touch publishes the new size/mtime. */
622 if (os->st->lab != NULL) {
623 have_old = pblock_catalog_lookup(os->st->cat, os->path, &old) == 0;
624 }
625 pblock_lab_crash(os->st->lab, "before_catalog_update"); /* F7 */
626 if (pblock_quota_touch_admit(os->st, os->path, os->meta.uid,
627 os->meta.size) != 0 /* F5 */
628 || pblock_catalog_touch(os->st->cat, os->path, os->meta.size,
629 os->meta.mtime) != 0)
630 {
631 rc = NGX_ERROR;
632 }
633 pblock_lab_crash(os->st->lab, "after_catalog_update"); /* F7 */
634 if (rc == NGX_OK && have_old) {
635 pblock_anomaly_updated(os->st, os->path, old.size,
636 old.mtime); /* F9 */
637 }
638 os->dirty = 0;
639 }
640 if (obj->fd != NGX_INVALID_FILE) {
641 if (close(obj->fd) != 0) {
642 rc = NGX_ERROR;
643 }
644 obj->fd = NGX_INVALID_FILE;
645 }
646 if (os != NULL) {
647 if (os->st->audit) { /* F17: fold per-I/O totals here */
648 char aux[64];
649
650 snprintf(aux, sizeof(aux), "r=%lld w=%lld mb=%lld",
651 (long long) os->a_rbytes, (long long) os->a_wbytes,
652 (long long) os->a_maxblock);
653 pblock_audit_log(os->st->cat, "close", os->path, aux,
654 os->meta.uid, os->meta.gid,
655 rc == NGX_OK ? 0 : -1, 0);
656 }
657 if (os->st->csi) { /* F3: persist CRCs for written blocks */
658 (void) pblock_csi_flush(os->st, os->blob_id, os->meta.size,
659 os->block_size, os->csi_dlo, os->csi_dhi);
660 }
661 if (os->wv != NULL) { /* F10: publish-time dedup fold */
662 if (os->st->refs && rc == NGX_OK) {
663 uint32_t crc = 0;
664 off_t total = 0;
665 int ok;
666
667 ok = brix_wverify_expected(os->wv, &crc, &total) == 0
668 && (int64_t) total == os->meta.size;
669 (void) pblock_refs_dedup_publish(os->st, os->path, &os->meta,
670 crc, ok);
671 }
672 brix_wverify_free(os->wv);
673 }
674 if (os->st->snap && !os->meta.is_dir) { /* F6: release the handle count */
675 __atomic_sub_fetch(&os->st->open_files, 1, __ATOMIC_RELEASE);
676 }
677 free(os->csi_crc);
678 free(os->lock_rng); /* F15 per-open lease snapshot */
679 pblock_lab_obj_free(os->lab); /* Phase-83 per-open snapshot */
680 }
681 free(os);
682 obj->state = NULL;
683 /* The obj shell is NOT freed here: it may be embedded in the adopter's
684 * handle (the VFS copies *obj by value). Its owner — the VFS adopter (via
685 * obj->heap_shell), the internal open-error paths, or the standalone test —
686 * frees the malloc'd shell. */
687 return rc;
688}

742/* driver->space — quota-aware capacity numbers for kXR_statvfs/SRR. Answers
743 * only when a byte quota is armed (then total = the quota, used = the usage
744 * rollup); otherwise NGX_ERROR/ENOTSUP so callers fall back to statvfs(2) on
745 * the backing filesystem — byte-for-byte the pre-F5 behaviour. */
746static ngx_int_t
747sd_pblock_space(brix_sd_instance_t *inst, brix_sd_space_t *out)
748{
749 pblock_state_t *st = inst->state;
750 int64_t bytes, inodes;
751
752 if (st == NULL || !st->quota || st->quota_bytes <= 0) {
753 errno = ENOTSUP;
754 return NGX_ERROR;
755 }
756 if (pblock_quota_usage(st, "total", 0, &bytes, &inodes) != 0) {
757 return NGX_ERROR;

760 out->used_bytes = bytes > 0 ? (uint64_t) bytes : 0;
761 out->free_bytes = bytes < st->quota_bytes
762 ? (uint64_t) (st->quota_bytes - bytes) : 0;
763 return NGX_OK;
