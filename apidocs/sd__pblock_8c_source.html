1/*
2 * sd_pblock.c — the pblock ("pseudo-block") Storage Driver: a full-capability,
3 * block-based drop-in for POSIX.
4 *
5 * WHAT: Implements brix_sd_pblock_driver, a complete backend that stores each
6 * object's bytes striped across fixed-size POSIX "block" files and the
7 * entire logical namespace + metadata in a SQLite catalog
8 * (sd_pblock_catalog.c). It advertises the same capabilities as the POSIX
9 * driver and implements every vtable slot. This file owns the instance and
10 * object lifecycle (init/cleanup, open/close and their helpers) and the
11 * driver descriptor; the byte I/O (sd_pblock_io.c), namespace/directory/
12 * xattr ops (sd_pblock_namespace.c) and staged atomic publish
13 * (sd_pblock_staged.c) live in siblings reached through
14 * sd_pblock_internal.h.
15 *
16 * WHY: Striping bulk content into fixed-size blocks (default 64 MiB, set per
17 * file at creation and configurable per export) is the defining property
18 * of a block backend, and splitting that content from the namespace/
19 * metadata keeps the hot byte path free of any database work. SQLite is
20 * touched only at metadata boundaries. Split from a single 1111-line file
21 * (phase-79) to hold every pblock file under the ~500-line, one-concept cap.
22 *
23 * HOW: An object's bytes live at <data>/<aa>/<bb>/<blob_id>/<block_index>;
24 * block 0 is opened persistently as obj->fd (so small files keep a real
25 * fd for zero-copy sendfile), and higher blocks are opened transiently per
26 * I/O. Reads/writes map [off,off+len) across blocks (holes read as zeros);
27 * writes update the cached size/mtime in memory and flush to the catalog
28 * on fsync/close. The whole file is ngx-free (libc + sqlite, malloc-owned
29 * state) so it is identical in the module and the standalone unit test.
30 * Compiled only when the build found libsqlite3 (BRIX_HAVE_SQLITE).
31 */

76/* ---- catalog enumeration (F14) -------------------------------------------- */

78/* Adapter from the catalog's flat row callback to the driver's enumerate cb. */
79typedef struct {
80 brix_sd_catalog_cb cb;
81 void *ctx;
82 int want_stat;
83} pblock_enum_ctx_t;

85static int
86pblock_enum_thunk(void *vctx, const char *path, int is_dir, int64_t size,
87 int64_t mtime)
88{
89 pblock_enum_ctx_t *ec = vctx;
90 brix_sd_catalog_ent_t ent;
91
92 if (is_dir) {
93 return 0; /* enumerate reports stored objects, not directories */
94 }
95 memset(&ent, 0, sizeof(ent));
96 ent.key = path; /* the catalog IS the namespace: key == logical path */
97 ent.path = path;
98 ent.have_stat = ec->want_stat;
99 ent.size = (off_t) size;
100 ent.mtime = (time_t) mtime;
101 return ec->cb(ec->ctx, &ent);
102}

104/* driver->enumerate — flat scan of every stored object (NGX_OK, or NGX_ERROR
105 * with errno on a catalog error). Advertised via BRIX_SD_CAP_CATALOG. */
106static ngx_int_t
107sd_pblock_enumerate(brix_sd_instance_t *inst, int want_stat,
108 brix_sd_catalog_cb cb, void *ctx)
109{
110 pblock_state_t *st = inst->state;
111 pblock_enum_ctx_t ec;
112
113 if (st == NULL || cb == NULL) {
114 errno = EINVAL;
115 return NGX_ERROR;
116 }
117 ec.cb = cb;
118 ec.ctx = ctx;
119 ec.want_stat = want_stat;
120 if (pblock_catalog_enumerate(st->cat, pblock_enum_thunk, &ec) != 0) {
121 return NGX_ERROR;
122 }
123 return NGX_OK;
124}

126/* ---- space accounting (F5) ------------------------------------------------ */
127
128/* driver->space — quota-aware capacity numbers for kXR_statvfs/SRR. Answers
129 * only when a byte quota is armed (then total = the quota, used = the usage
130 * rollup); otherwise NGX_ERROR/ENOTSUP so callers fall back to statvfs(2) on
131 * the backing filesystem — byte-for-byte the pre-F5 behaviour. */
132static ngx_int_t
133sd_pblock_space(brix_sd_instance_t *inst, brix_sd_space_t *out)
134{
135 pblock_state_t *st = inst->state;
136 int64_t bytes, inodes;
137
138 if (st == NULL || !st->quota || st->quota_bytes <= 0) {
139 errno = ENOTSUP;
140 return NGX_ERROR;
141 }
142 if (pblock_quota_usage(st, "total", 0, &bytes, &inodes) != 0) {
143 return NGX_ERROR;
144 }
145 out->total_bytes = (uint64_t) st->quota_bytes;
146 out->used_bytes = bytes > 0 ? (uint64_t) bytes : 0;
147 out->free_bytes = bytes < st->quota_bytes
148 ? (uint64_t) (st->quota_bytes - bytes) : 0;
149 return NGX_OK;
150}

152/* ---- nearline simulation (F4) --------------------------------------------- */
153
154/* driver->residency — pure read of the simulated tape-residency model. Only
155 * reached when CAP_NEARLINE is advertised (nearline=1 armed); never initiates
156 * or advances a recall — protocol handlers (kXR stat's offline flag, the tape
157 * REST API, S3 storage-class) classify without staging side effects. */
158static ngx_int_t
159sd_pblock_residency(brix_sd_instance_t *inst, const char *key,
160 brix_sd_residency_t *out)
161{
162 pblock_state_t *st = inst->state;
163 int rc;
164
165 if (st == NULL || !st->nearline) {
166 *out = BRIX_SD_RES_ONLINE;
167 return NGX_OK;
168 }
169 rc = pblock_catalog_lookup(st->cat, key, NULL);
170 if (rc != 0) {
171 errno = rc == 1 ? ENOENT : errno;
172 return NGX_ERROR;
173 }
174 return pblock_nearline_res(st, key, out) == 0 ? NGX_OK : NGX_ERROR;
175}

177/* driver->recall — the cache tier's recall-at-fill entry (phase-64 §9.3).
178 * Synchronous like sd_frm's (no parking handle to mint): by return the object
179 * is either ONLINE (NGX_OK — a normal fill follows) or the recall failed
180 * (NGX_ERROR, errno set). The simulated latency/outcome are ctl-driven
181 * (pblock_nearline.h). */
182static ngx_int_t
183sd_pblock_recall(brix_sd_instance_t *inst, const char *key,
184 char reqid_out[40])
185{
186 pblock_state_t *st = inst->state;
187 int rc;
188
189 reqid_out[0] = '\0';
190 if (st == NULL || !st->nearline) {
191 return NGX_OK;
192 }
193 rc = pblock_catalog_lookup(st->cat, key, NULL);
194 if (rc != 0) {
195 errno = rc == 1 ? ENOENT : errno;
196 return NGX_ERROR;
197 }
198 return pblock_nearline_recall(st, key) == 0 ? NGX_OK : NGX_ERROR;
199}
