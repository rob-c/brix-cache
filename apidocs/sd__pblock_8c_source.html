1/*
2 * sd_pblock.c — the pblock ("pseudo-block") Storage Driver: a full-capability,
3 * block-based drop-in for POSIX.
4 *
5 * WHAT: Implements brix_sd_pblock_driver, a complete backend that stores each
6 * object's bytes striped across fixed-size POSIX "block" files and the
7 * entire logical namespace + metadata in a SQLite catalog
8 * (sd_pblock_catalog.c). It advertises the same capabilities as the POSIX
9 * driver and implements every vtable slot.
10 *
11 * WHY: Striping bulk content into fixed-size blocks (default 64 MiB, set per
12 * file at creation and configurable per export) is the defining property
13 * of a block backend, and splitting that content from the namespace/
14 * metadata keeps the hot byte path free of any database work. SQLite is
15 * touched only at metadata boundaries.
16 *
17 * HOW: An object's bytes live at <data>/<aa>/<bb>/<blob_id>/<block_index>;
18 * block 0 is opened persistently as obj->fd (so small files keep a real
19 * fd for zero-copy sendfile), and higher blocks are opened transiently per
20 * I/O. Reads/writes map [off,off+len) across blocks (holes read as zeros);
21 * writes update the cached size/mtime in memory and flush to the catalog
22 * on fsync/close. The whole file is ngx-free (libc + sqlite, malloc-owned
23 * state) so it is identical in the module and the standalone unit test.
24 * Compiled only when the build found libsqlite3 (BRIX_HAVE_SQLITE).
25 */

86static ngx_int_t
87sd_pblock_init(brix_sd_instance_t *inst, void *driver_conf)
88{
89 const brix_sd_pblock_conf_t *conf = driver_conf;
90 pblock_state_t *st;
91 char db[PATH_MAX];
92
93 if (conf == NULL || conf->root == NULL) {
94 errno = EINVAL;
95 return NGX_ERROR;
96 }
97
98 st = calloc(1, sizeof(*st));
99 if (st == NULL) {
100 errno = ENOMEM;
101 return NGX_ERROR;
102 }
103 snprintf(st->root, sizeof(st->root), "%s", conf->root);
104 snprintf(st->data_dir, sizeof(st->data_dir), "%s/data", conf->root);
105 st->block_size = conf->block_size > 0 ? conf->block_size
106 : PBLOCK_DEFAULT_BLOCK_SIZE;
107
108 if (pblock_mkdir_p(st->root) != 0 || pblock_mkdir_p(st->data_dir) != 0) {
109 int err = errno;
110
111 free(st);
112 errno = err;
113 return NGX_ERROR;
114 }
115
116 snprintf(db, sizeof(db), "%s/catalog.db", conf->root);
117 st->cat = pblock_catalog_open(db, conf->busy_timeout_ms);
118 if (st->cat == NULL) {
119 int err = errno;
120
121 free(st);
122 errno = err;
123 return NGX_ERROR;
124 }
125
126 /* The export root "/" always exists (a directory), like a POSIX mount point —
127 * so stat("/")/opendir("/")/PROPFIND on the root succeed before anything is
128 * written. Created once; harmless if a concurrent worker also creates it. */
129 if (pblock_catalog_lookup(st->cat, "/", NULL) == 1) {
130 pblock_meta root_meta;
131
132 memset(&root_meta, 0, sizeof(root_meta));
133 root_meta.is_dir = 1;
134 root_meta.mtime = root_meta.ctime = pblock_now();
135 root_meta.mode = S_IFDIR | 0755;
136 (void) pblock_catalog_put(st->cat, "/", &root_meta);
137 }
138
139 inst->state = st;
140 return NGX_OK;
141}

157/* pblock_make_obj — wrap a block-0 fd + cached metadata in a heap object. On
158 * failure the fd is closed and NULL/errno returned. */
159static brix_sd_obj_t *
160pblock_make_obj(brix_sd_instance_t *inst, const char *path, int fd,
161 const pblock_meta *meta)
162{
163 brix_sd_obj_t *obj;
164 pblock_obj_t *os;
165
166 obj = calloc(1, sizeof(*obj));
167 os = calloc(1, sizeof(*os));
168 if (obj == NULL || os == NULL) {
169 free(obj);
170 free(os);
171 if (fd >= 0) {
172 close(fd);
173 }
174 errno = ENOMEM;
175 return NULL;
176 }
177
178 os->st = inst->state;
179 os->meta = *meta;
180 os->block_size = meta->block_size;
181 snprintf(os->path, sizeof(os->path), "%s", path);
182 snprintf(os->blob_id, sizeof(os->blob_id), "%s", meta->blob_id);
183
184 obj->driver = inst->driver;
185 obj->inst = inst;
186 obj->fd = fd >= 0 ? fd : NGX_INVALID_FILE;
187 obj->state = os;
188 obj->heap_shell = 1; /* malloc'd shell; a VFS adopter frees it after copy */
189
190 /* Capture the open-time metadata snapshot the VFS reports to callers (file
191 * size, mode, mtime) — without this the adopting handle would see a zeroed
192 * snap and treat the file as empty (immediate EOF on read). */
193 ngx_memzero(&obj->snap, sizeof(obj->snap));
194 obj->snap.size = meta->size;
195 obj->snap.mtime = meta->mtime;
196 obj->snap.ctime = meta->ctime;
197 obj->snap.mode = meta->mode;
198 obj->snap.ino = (ino_t) pblock_fnv(path);
199 obj->snap.is_dir = meta->is_dir ? 1 : 0;
200 obj->snap.is_reg = meta->is_dir ? 0 : 1;
201 return obj;
202}

204/* pblock_open_create — create a brand-new file: a fresh per-object dir + block 0,
205 * plus its catalog row. Returns the object or NULL/errno (*err_out set). */
206static brix_sd_obj_t *
207pblock_open_create(brix_sd_instance_t *inst, const char *path, mode_t mode,
208 int *err_out)
209{
210 pblock_state_t *st = inst->state;
211 pblock_meta meta;
212 brix_sd_obj_t *obj;
213 char block0[PATH_MAX];
214 int fd;
215
216 memset(&meta, 0, sizeof(meta));
217 if (pblock_gen_blob_id(meta.blob_id) != 0
218 || pblock_ensure_obj_dir(st, meta.blob_id) != 0
219 || pblock_block_path(st, meta.blob_id, 0, block0, sizeof(block0)) != 0)
220 {
221 if (err_out != NULL) { *err_out = errno; }
222 return NULL;
223 }
224
225 fd = open(block0, O_RDWR | O_CREAT | O_EXCL, 0600);
226 if (fd < 0) {
227 if (err_out != NULL) { *err_out = errno; }
228 return NULL;
229 }
230
231 meta.is_dir = 0;
232 meta.size = 0;
233 meta.block_size = st->block_size;
234 meta.mtime = meta.ctime = pblock_now();
235 meta.mode = S_IFREG | (mode & 0777);
236
237 if (pblock_catalog_put(st->cat, path, &meta) != 0) {
238 int err = errno;
239
240 close(fd);
241 pblock_remove_blocks(st, meta.blob_id, 0, st->block_size);
242 if (err_out != NULL) { *err_out = err; }
243 return NULL;
244 }
245
246 obj = pblock_make_obj(inst, path, fd, &meta);
247 if (obj == NULL && err_out != NULL) {
248 *err_out = errno;
249 }
250 return obj;
251}

253/* pblock_open_existing — open an existing file's block 0; honours O_TRUNC on
254 * write via the block-aware truncate. Returns the object or NULL/errno. */
255static brix_sd_obj_t *
256pblock_open_existing(brix_sd_instance_t *inst, const char *path,
257 pblock_meta *meta, int sd_flags, int *err_out)
258{
259 pblock_state_t *st = inst->state;
260 brix_sd_obj_t *obj;
261 char block0[PATH_MAX];
262 int want_write = sd_flags & BRIX_SD_O_WRITE;
263 int fd;
264
265 if (pblock_block_path(st, meta->blob_id, 0, block0, sizeof(block0)) != 0) {
266 if (err_out != NULL) { *err_out = errno; }
267 return NULL;
268 }
269 fd = open(block0, want_write ? O_RDWR : O_RDONLY);
270 if (fd < 0) {
271 if (err_out != NULL) { *err_out = errno; }
272 return NULL;
273 }
274
275 obj = pblock_make_obj(inst, path, fd, meta);
276 if (obj == NULL) {
277 if (err_out != NULL) { *err_out = errno; }
278 return NULL;
279 }
280
281 if (want_write && (sd_flags & BRIX_SD_O_TRUNC) && meta->size > 0) {
282 if (sd_pblock_ftruncate(obj, 0) != NGX_OK) {
283 int err = errno;
284
285 obj->driver->close(obj); /* frees state + closes fd */
286 free(obj); /* … and the malloc'd shell */
287 if (err_out != NULL) { *err_out = err; }
288 return NULL;
289 }
290 }
291 return obj;
292}

294static brix_sd_obj_t *
295sd_pblock_open(brix_sd_instance_t *inst, const char *path, int sd_flags,
296 mode_t mode, int *err_out)
297{
298 pblock_state_t *st = inst->state;
299 pblock_meta meta;
300 int rc;
301
302 rc = pblock_catalog_lookup(st->cat, path, &meta);
303 if (rc < 0) {
304 if (err_out != NULL) { *err_out = errno; }
305 return NULL;
306 }
307
308 if (rc == 0 && meta.is_dir) {
309 if (sd_flags & BRIX_SD_O_WRITE) {
310 if (err_out != NULL) { *err_out = EISDIR; }
311 return NULL;
312 }
313 return pblock_make_obj(inst, path, NGX_INVALID_FILE, &meta);
314 }
315
316 if (rc == 0) { /* existing file */
317 if ((sd_flags & BRIX_SD_O_CREATE) && (sd_flags & BRIX_SD_O_EXCL)) {
318 if (err_out != NULL) { *err_out = EEXIST; }
319 return NULL;
320 }
321 return pblock_open_existing(inst, path, &meta, sd_flags, err_out);
322 }
323
324 /* absent */
325 if (!(sd_flags & BRIX_SD_O_CREATE)) {
326 if (err_out != NULL) { *err_out = ENOENT; }
327 return NULL;
328 }
329 return pblock_open_create(inst, path, mode, err_out);
330}

332static ngx_int_t
333sd_pblock_close(brix_sd_obj_t *obj)
334{
335 pblock_obj_t *os;
336 ngx_int_t rc = NGX_OK;
337
338 if (obj == NULL) {
339 return NGX_OK;
340 }
341 os = obj->state;
342
343 if (os != NULL && os->dirty) {
344 if (pblock_catalog_touch(os->st->cat, os->path, os->meta.size,
345 os->meta.mtime) != 0)
346 {
347 rc = NGX_ERROR;
348 }
349 os->dirty = 0;
350 }
351 if (obj->fd != NGX_INVALID_FILE) {
352 if (close(obj->fd) != 0) {
353 rc = NGX_ERROR;
354 }
355 obj->fd = NGX_INVALID_FILE;
356 }
357 free(os);
358 obj->state = NULL;
359 /* The obj shell is NOT freed here: it may be embedded in the adopter's
360 * handle (the VFS copies *obj by value). Its owner — the VFS adopter (via
361 * obj->heap_shell), the internal open-error paths, or the standalone test —
362 * frees the malloc'd shell. */
363 return rc;
364}

368static ssize_t
369sd_pblock_pread(brix_sd_obj_t *obj, void *buf, size_t len, off_t off)
370{
371 pblock_obj_t *os = obj->state;
372 size_t avail;
373
374 if (off < 0 || off >= os->meta.size) {
375 return 0; /* at/after EOF */
376 }
377 avail = (size_t) (os->meta.size - off);
378 if (len > avail) {
379 len = avail;
380 }
381 if (len == 0) {
382 return 0;
383 }
384 return pblock_read_blocks(os->st, os->blob_id, os->block_size, obj->fd,
385 buf, len, off);
386}

388static ssize_t
389sd_pblock_pwrite(brix_sd_obj_t *obj, const void *buf, size_t len, off_t off)
390{
391 pblock_obj_t *os = obj->state;
392 ssize_t n;
393
394 if (len == 0) {
395 return 0;
396 }
397 n = pblock_write_blocks(os->st, os->blob_id, os->block_size, obj->fd,
398 buf, len, off);
399 if (n > 0) {
400 os->meta.mtime = pblock_now();
401 if ((int64_t) off + n > os->meta.size) {
402 os->meta.size = (int64_t) off + n;
403 }
404 os->dirty = 1;
405 }
406 return n;
407}

409/* sd_pblock_preadv — vectored read as a loop of the block-aware pread (stops at
410 * the first short/EOF segment); bytes read or -1. */
411static ssize_t
412sd_pblock_preadv(brix_sd_obj_t *obj, const struct iovec *iov, int iovcnt,
413 off_t off)
414{
415 ssize_t total = 0;
416 int i;
417
418 for (i = 0; i < iovcnt; i++) {
419 ssize_t n = sd_pblock_pread(obj, iov[i].iov_base, iov[i].iov_len,
420 off + total);
421
422 if (n < 0) {
423 return total ? total : -1;
424 }
425 total += n;
426 if ((size_t) n < iov[i].iov_len) {
427 break; /* short read / EOF */
428 }
429 }
430 return total;
431}

441/* sd_pblock_copy_range — copy up to `len` bytes src->dst through a bounded
442 * userspace buffer (blocks straddle file boundaries, so copy_file_range across
443 * whole objects does not apply). The VFS owns the outer loop; one call copies up
444 * to the buffer size. Returns bytes copied (0 = src EOF) or -1. */
445static ssize_t
446sd_pblock_copy_range(brix_sd_obj_t *src, off_t src_off, brix_sd_obj_t *dst,
447 off_t dst_off, size_t len)
448{
449 pblock_obj_t *sos = src->state;
450 pblock_obj_t *dos = dst->state;
451 char buf[65536];
452 size_t chunk = len < sizeof(buf) ? len : sizeof(buf);
453 ssize_t r, w;
454
455 if (src_off < 0 || src_off >= sos->meta.size) {
456 return 0;
457 }
458 if (chunk > (size_t) (sos->meta.size - src_off)) {
459 chunk = (size_t) (sos->meta.size - src_off);
460 }
461 if (chunk == 0) {
462 return 0;
463 }
464
465 r = pblock_read_blocks(sos->st, sos->blob_id, sos->block_size, src->fd,
466 buf, chunk, src_off);
467 if (r <= 0) {
468 return r;
469 }
470 w = pblock_write_blocks(dos->st, dos->blob_id, dos->block_size, dst->fd,
471 buf, (size_t) r, dst_off);
472 if (w > 0) {
473 dos->meta.mtime = pblock_now();
474 if ((int64_t) dst_off + w > dos->meta.size) {
475 dos->meta.size = (int64_t) dst_off + w;
476 }
477 dos->dirty = 1;
478 }
479 return w;
480}

482/* sd_pblock_read_sendfile_fd — zero-copy only for an offset-0 range that lies
483 * within block 0 (the persistently-open obj->fd): small files and the start of
484 * large ones. Multi-block ranges return NGX_INVALID_FILE (served memory-backed)
485 * until the VFS read path is block-aware. */
486static ngx_fd_t
487sd_pblock_read_sendfile_fd(brix_sd_obj_t *obj, off_t off, size_t len,
488 unsigned want_zerocopy)
489{
490 pblock_obj_t *os = obj->state;
491
492 if (!want_zerocopy || obj->fd == NGX_INVALID_FILE) {
493 return NGX_INVALID_FILE;
494 }
495 if (off == 0 && (int64_t) len <= os->block_size) {
496 return obj->fd;
497 }
498 return NGX_INVALID_FILE;
499}

501static ngx_int_t
502sd_pblock_ftruncate(brix_sd_obj_t *obj, off_t len)
503{
504 pblock_obj_t *os = obj->state;
505 int64_t bs = os->block_size;
506 int64_t old_last = pblock_last_block(os->meta.size, bs);
507 int64_t keep = pblock_last_block(len, bs);
508 int64_t boundary = (int64_t) len - keep * bs; /* bytes kept in `keep` */
509 int64_t i;
510 int fd, transient = 0;
511
512 /* trim the boundary block to its surviving length */
513 if (keep == 0) {
514 fd = obj->fd;
515 } else {
516 char bp[PATH_MAX];
517
518 if (pblock_block_path(os->st, os->blob_id, keep, bp, sizeof(bp)) != 0) {
519 return NGX_ERROR;
520 }
521 fd = open(bp, O_RDWR | O_CREAT, 0600);
522 if (fd < 0) {
523 return NGX_ERROR;
524 }
525 transient = 1;
526 }
527 if (ftruncate(fd, (off_t) boundary) != 0) {
528 if (transient) {
529 close(fd);
530 }
531 return NGX_ERROR;
532 }
533 if (transient) {
534 close(fd);
535 }
536
537 /* drop whole blocks past the new size */
538 for (i = keep + 1; i <= old_last; i++) {
539 char bp[PATH_MAX];
540
541 if (pblock_block_path(os->st, os->blob_id, i, bp, sizeof(bp)) == 0) {
542 unlink(bp);
543 }
544 }
545
546 os->meta.size = (int64_t) len;
547 os->meta.mtime = pblock_now();
548 os->dirty = 1;
549 return NGX_OK;
550}

552/* sd_pblock_fsync — durability barrier: fsync every block file backing the
553 * object, then flush dirty size/mtime to the catalog. */
554static ngx_int_t
555sd_pblock_fsync(brix_sd_obj_t *obj)
556{
557 pblock_obj_t *os = obj->state;
558 int64_t last = pblock_last_block(os->meta.size, os->block_size);
559 int64_t i;
560
561 if (obj->fd != NGX_INVALID_FILE && fsync(obj->fd) != 0) {
562 return NGX_ERROR;
563 }
564 for (i = 1; i <= last; i++) {
565 char bp[PATH_MAX];
566 int fd;
567
568 if (pblock_block_path(os->st, os->blob_id, i, bp, sizeof(bp)) != 0) {
569 continue;
570 }
571 fd = open(bp, O_RDONLY);
572 if (fd >= 0) {
573 if (fsync(fd) != 0) {
574 close(fd);
575 return NGX_ERROR;
576 }
577 close(fd);
578 }
579 }
580
581 if (os->dirty) {
582 if (pblock_catalog_touch(os->st->cat, os->path, os->meta.size,
583 os->meta.mtime) != 0)
584 {
585 return NGX_ERROR;
586 }
587 os->dirty = 0;
588 }
589 return NGX_OK;
590}

623static ngx_int_t
624sd_pblock_unlink(brix_sd_instance_t *inst, const char *path, int is_dir)
625{
626 pblock_state_t *st = inst->state;
627 pblock_meta meta;
628 int rc;
629
630 rc = pblock_catalog_lookup(st->cat, path, &meta);
631 if (rc < 0) {
632 return NGX_ERROR;
633 }
634 if (rc == 1) {
635 errno = ENOENT;
636 return NGX_ERROR;
637 }
638
639 if (is_dir) {
640 int children;
641
642 if (!meta.is_dir) {
643 errno = ENOTDIR;
644 return NGX_ERROR;
645 }
646 children = pblock_catalog_child_count(st->cat, path);
647 if (children < 0) {
648 return NGX_ERROR;
649 }
650 if (children > 0) {
651 errno = ENOTEMPTY;
652 return NGX_ERROR;
653 }
654 } else if (meta.is_dir) {
655 errno = EISDIR;
656 return NGX_ERROR;
657 }
658
659 if (!meta.is_dir) {
660 pblock_remove_blocks(st, meta.blob_id, meta.size, meta.block_size);
661 }
662 return pblock_catalog_remove(st->cat, path) == 0 ? NGX_OK : NGX_ERROR;
663}

681 * existing catalog row. The catalog records mode + mtime + ctime, so those are
682 * honoured; it has no owner (uid/gid) or atime columns, so set_owner and atime
683 * are accepted-and-ignored (object-store semantics) rather than failing — a chown
684 * an export cannot represent should not break the op. ctime is bumped on any
685 * change, matching POSIX. lookup→modify→put reuses the upsert path. */
686static ngx_int_t
687sd_pblock_setattr(brix_sd_instance_t *inst, const char *path,
688 const brix_sd_setattr_t *attr)
689{
690 pblock_state_t *st = inst->state;
691 int set_mtime = 0;
692 int64_t mtime = 0;
693
694 if (attr->set_times && attr->mtime.tv_nsec != UTIME_OMIT) {
695 set_mtime = 1;
696 mtime = (attr->mtime.tv_nsec == UTIME_NOW)
697 ? pblock_now() : (int64_t) attr->mtime.tv_sec;
698 }
699
700 /* One UPDATE — no read-modify-write. ENOENT (rc<0 with errno) when absent. */
701 return pblock_catalog_setattr(st->cat, path, attr->set_mode ? 1 : 0,
702 attr->mode & 0777, set_mtime, mtime, pblock_now()) == 0
703 ? NGX_OK : NGX_ERROR;
704}

706/* sd_pblock_drop_dst — clear an existing rename/copy destination: reject a
707 * non-empty directory, remove a file's blocks, then drop the catalog row. */
708static ngx_int_t
709sd_pblock_drop_dst(pblock_state_t *st, const char *dst,
710 const pblock_meta *dmeta)
711{
712 if (dmeta->is_dir) {
713 int children = pblock_catalog_child_count(st->cat, dst);
714
715 if (children < 0) {
716 return NGX_ERROR;
717 }
718 if (children > 0) {
719 errno = ENOTEMPTY;
720 return NGX_ERROR;
721 }
722 } else {
723 pblock_remove_blocks(st, dmeta->blob_id, dmeta->size,
724 dmeta->block_size);
725 }
726 return pblock_catalog_remove(st->cat, dst) == 0 ? NGX_OK : NGX_ERROR;
727}

765static ngx_int_t
766sd_pblock_server_copy(brix_sd_instance_t *inst, const char *src,
767 const char *dst, off_t *bytes_out)
768{
769 pblock_state_t *st = inst->state;
770 pblock_meta smeta, dmeta;
771 int64_t last, i;
772 int rc;
773
774 rc = pblock_catalog_lookup(st->cat, src, &smeta);
775 if (rc < 0) {
776 return NGX_ERROR;
777 }
778 if (rc == 1) {
779 errno = ENOENT;
780 return NGX_ERROR;
781 }
782 if (smeta.is_dir) {
783 errno = EISDIR;
784 return NGX_ERROR;
785 }
786
787 memset(&dmeta, 0, sizeof(dmeta));
788 if (pblock_gen_blob_id(dmeta.blob_id) != 0
789 || pblock_ensure_obj_dir(st, dmeta.blob_id) != 0)
790 {
791 return NGX_ERROR;
792 }
793
794 last = pblock_last_block(smeta.size, smeta.block_size);
795 for (i = 0; i <= last; i++) {
796 char sp[PATH_MAX], dp[PATH_MAX];
797
798 if (pblock_block_path(st, smeta.blob_id, i, sp, sizeof(sp)) != 0
799 || pblock_block_path(st, dmeta.blob_id, i, dp, sizeof(dp)) != 0
800 || pblock_copy_one_block(sp, dp) < 0)
801 {
802 int err = errno;
803
804 pblock_remove_blocks(st, dmeta.blob_id, smeta.size,
805 smeta.block_size);
806 errno = err;
807 return NGX_ERROR;
808 }
809 }
810
811 dmeta.is_dir = 0;
812 dmeta.size = smeta.size;
813 dmeta.block_size = smeta.block_size;
814 dmeta.mtime = dmeta.ctime = pblock_now();
815 dmeta.mode = smeta.mode;
816
817 if (pblock_catalog_put(st->cat, dst, &dmeta) != 0) {
818 int err = errno;
819
820 pblock_remove_blocks(st, dmeta.blob_id, smeta.size, smeta.block_size);
821 errno = err;
822 return NGX_ERROR;
823 }
824 if (bytes_out != NULL) {
825 *bytes_out = (off_t) smeta.size;
826 }
827 return NGX_OK;
828}

945static brix_sd_staged_t *
946sd_pblock_staged_open(brix_sd_instance_t *inst, const char *final_path,
947 mode_t mode, int *err_out)
948{
949 pblock_state_t *st = inst->state;
950 brix_sd_staged_t *handle;
951 pblock_staged_t *ps;
952
953 /* POSIX parity with the posix driver's staged temp (O_EXCL in the final
954 * directory): a missing parent collection fails HERE — before any blob
955 * is allocated or a single body byte is accepted — not at commit. */
956 if (pblock_catalog_parent_ok(st->cat, final_path) != 0) {
957 if (err_out != NULL) { *err_out = errno; }
958 return NULL;
959 }
960
961 handle = calloc(1, sizeof(*handle));
962 ps = calloc(1, sizeof(*ps));
963 if (handle == NULL || ps == NULL) {
964 free(handle);
965 free(ps);
966 if (err_out != NULL) { *err_out = ENOMEM; }
967 return NULL;
968 }
969
970 if (pblock_gen_blob_id(ps->blob_id) != 0
971 || pblock_ensure_obj_dir(st, ps->blob_id) != 0)
972 {
973 if (err_out != NULL) { *err_out = errno; }
974 free(handle);
975 free(ps);
976 return NULL;
977 }
978
979 ps->st = st;
980 ps->block_size = st->block_size;
981 ps->size = 0;
982 ps->mode = mode;
983 snprintf(ps->final_path, sizeof(ps->final_path), "%s", final_path);
984 handle->inst = inst;
985 handle->state = ps;
986 return handle;

1004/* sd_pblock_staged_commit — publish the staged blocks atomically by inserting
1005 * the final catalog row pointing at the staged blob id (the blocks simply become
1006 * the final object — no copy or rename). On success the handle is consumed; on
1007 * failure it stays valid and the caller must staged_abort to release it. */
1008static ngx_int_t
1009sd_pblock_staged_commit(brix_sd_staged_t *st, int noreplace)
1010{
1011 pblock_staged_t *ps = st->state;
1012 pblock_state_t *pst = ps->st;
1013 pblock_meta meta, dmeta;
1014 int rc;
1015
1016 rc = pblock_catalog_lookup(pst->cat, ps->final_path, &dmeta);
1017 if (rc < 0) {
1018 return NGX_ERROR;
1019 }
1020 if (rc == 0) {
1021 if (noreplace) {
1022 errno = EEXIST;
1023 return NGX_ERROR;
1024 }
1025 if (sd_pblock_drop_dst(pst, ps->final_path, &dmeta) != NGX_OK) {
1026 return NGX_ERROR;
1027 }
1028 }
1029
1030 memset(&meta, 0, sizeof(meta));
1031 meta.is_dir = 0;
1032 snprintf(meta.blob_id, sizeof(meta.blob_id), "%s", ps->blob_id);
1033 meta.size = ps->size;
1034 meta.block_size = ps->block_size;
1035 meta.mtime = meta.ctime = pblock_now();
1036 meta.mode = S_IFREG | (ps->mode & 0777);
1037
1038 if (pblock_catalog_put(pst->cat, ps->final_path, &meta) != 0) {
1039 return NGX_ERROR;
1040 }
1041
1042 free(ps);
1043 free(st);
1044 return NGX_OK;
1045}
