1/*
2 * validate_registry.c — multi-issuer registry authN/authZ for WLCG bearer tokens.
3 *
4 * WHAT: The issuer-registry entry points that sit above brix_token_validate():
5 * brix_token_peek_iss() reads the "iss" claim WITHOUT trusting the signature so
6 * the registry can pick which issuer's keys to verify against;
7 * brix_token_validate_registry_authn() does the authN half (peek → registry_find
8 * → validate() with that issuer's keys → multi-audience accept);
9 * brix_token_authz_strategy() runs the per-path authorization ladder
10 * (base_path/restricted_path gate → capability / group / mapping strategy); and
11 * brix_token_validate_registry() combines authN+authZ where the request path is
12 * known at validation time (WebDAV/S3).
13 *
14 * WHY: validate.c exceeded the ~500-line file-size guard, so the registry plane
15 * — which layers issuer selection and per-path authorization on top of the core
16 * signature/claims pipeline — was split out under the phase-79 guard. Stream
17 * kXR_auth happens before any path is known, so issuer-keyed authentication and
18 * per-path authorization are deliberately split into the authn half and the
19 * strategy ladder; both are reused by the combined HTTP entry point. The peek →
20 * find → verify → strategy ordering and the capability/group/mapping semantics
21 * are preserved exactly from the original single-file implementation.
22 *
23 * HOW: peek_iss splits + b64url-decodes the payload and reads "iss" (untrusted,
24 * re-derived from verified claims afterwards). The authn half hands a single
25 * declared audience to brix_token_validate() for correct string-or-array
26 * membership and accepts multiple issuer/global audiences best-effort. The
27 * strategy ladder short-circuits ALLOW on the first satisfied strategy bit.
28 * token_sanitize_for_log() (validate.c, via validate_internal.h) guards every
29 * untrusted value that reaches the error log.
30 */

42/* brix_token_peek_iss — read the "iss" claim WITHOUT trusting the signature
43 * (xrdjwt_split + b64url_decode + json_get_string): the registry must pick an
44 * issuer, and thus its verification keys, before it can trust anything, so this
45 * read is explicitly untrusted and re-derived from verified claims afterwards. */
46int
47brix_token_peek_iss(const char *token, size_t token_len,
48 char *out, size_t outsz)
49{
50 xrdjwt_seg seg[3];
51 u_char pay[4096];
52 ssize_t n;
53
54 out[0] = '\0';
55
56 if (xrdjwt_split(token, token_len, seg) != 0) {
57 return -1; /* not a compact JWS */
58 }
59 n = b64url_decode(seg[1].p, seg[1].n, pay, sizeof(pay) - 1);
60 if (n < 0) {
61 return -1;
62 }
63 pay[n] = '\0';
64 if (json_get_string((char *) pay, (size_t) n, "iss", out, outsz) < 0) {
65 return -1;
66 }
67 return 0;
68}

70/* brix_token_validate_registry_authn — registry authN with no path gate: peek
71 * iss → registry_find → validate() with THAT issuer's keys → multi-audience accept.
72 * Stream kXR_auth happens before any path is known, so issuer-keyed authentication
73 * and per-path authorization are split; this is the authN half (also reused by the
74 * combined HTTP entry point). On success returns 0, fills *claims, sets *out_issuer. */
75int
76brix_token_validate_registry_authn(const brix_token_registry_args_t *a,
77 const brix_token_issuer_t **out_issuer)
78{
79 char iss[256];
80 const brix_token_issuer_t *is;
81 const char *expected_aud;
82 int i;
83
84 *out_issuer = NULL;
85
86 if (brix_token_peek_iss(a->token, a->token_len, iss, sizeof(iss)) != 0) {
87 ngx_log_error(NGX_LOG_WARN, a->log, 0,
88 "brix_token: cannot read iss for issuer selection");
89 return -1;
90 }
91 is = brix_token_registry_find(a->reg, iss);
92 if (is == NULL) {
93 char safe[512];
94 token_sanitize_for_log(iss, safe, sizeof(safe));
95 ngx_log_error(NGX_LOG_WARN, a->log, 0,
96 "brix_token: unknown issuer \"%s\"", safe);
97 return -1;
98 }
99
100 /* Verify with THIS issuer's keys; validate() re-checks iss after the
101 * signature is trusted. A single declared audience is handed to validate()
102 * for correct string-or-array membership; multiple audiences are accepted
103 * best-effort below (full multi-audience array membership = W1b). */
104 expected_aud = (is->audience_count == 1) ? is->audiences[0] : NULL;
105
106 {
107 brix_token_validate_args_t va;
108
109 va.log = a->log;
110 va.token = a->token;
111 va.token_len = a->token_len;
112 va.keys = is->jwks_key_count ? is->jwks_keys : NULL;
113 va.key_count = is->jwks_key_count;
114 va.expected_issuer = is->issuer;
115 va.expected_audience = expected_aud;
116 va.macaroon_secret = a->macaroon_secret;
117 va.secret_len = a->secret_len;
118 va.clock_skew = a->clock_skew;
119 va.claims = a->claims;
120
121 if (brix_token_validate(&va) != 0) {
122 return -1;
123 }
124 }
125
126 if (expected_aud == NULL && is->audience_count > 0) {
127 int ok = 0;
128 for (i = 0; i < is->audience_count && !ok; i++) {
129 ok = (strcmp(a->claims->aud, is->audiences[i]) == 0);
130 }
131 for (i = 0; i < a->reg->global_audience_count && !ok; i++) {
132 ok = (strcmp(a->claims->aud, a->reg->global_audiences[i]) == 0);
133 }
134 if (!ok) {
135 ngx_log_error(NGX_LOG_WARN, a->log, 0,
136 "brix_token: audience not accepted for issuer \"%s\"",
137 is->name);
138 return -1;
139 }
140 }
141
142 *out_issuer = is;
143 return 0;
144}

146/* brix_token_authz_strategy — the per-path authorization ladder: enforce the
147 * issuer base_path/restricted_path gate, then run the authorization_strategy
148 * (capability / group / mapping) for (req_path, op). Shared by the HTTP combined
149 * path and the stream per-path identity check. Returns 1 = ALLOW, 0 = DENY. */
150int
151brix_token_authz_strategy(const brix_token_issuer_t *is,
152 const brix_token_claims_t *claims, const char *req_path,
153 brix_token_op_e op)
154{
155 if (!brix_token_issuer_path_ok(is, req_path)) {
156 return 0;
157 }
158
159 /* capability — the token's own WLCG scopes must cover (path, op). */
160 if (is->strategy & BRIX_AUTHZ_CAPABILITY) {
161 int ok = (op == BRIX_TOKEN_OP_WRITE)
162 ? brix_token_check_write(claims->scopes, claims->scope_count,
163 req_path)
164 : brix_token_check_read(claims->scopes, claims->scope_count,
165 req_path);
166 if (ok) {
167 return 1;
168 }
169 }
170
171 /* group — the issuer vouches for its base_path for any token bearing a
172 * group claim; the per-VO/group ACL layer refines the actual access using
173 * the identity's groups (claims->groups → identity vo_list). */
174 if (is->strategy & BRIX_AUTHZ_GROUP) {
175 if (claims->groups[0] != '\0') {
176 return 1;
177 }
178 }
179
180 /* mapping — the subject must resolve to a local user (map_subject +
181 * name_mapfile, with onmissing/default policy); the per-user ACL/POSIX
182 * layer then enforces that user's permissions within base_path. */
183 if (is->strategy & BRIX_AUTHZ_MAPPING) {
184 char user[64];
185
186 if (!is->map_subject) {
187 return 1; /* trust the issuer's subject as-is */
188 }
189 if (is->name_mapfile[0] != '\0'
190 && brix_subject_mapfile_lookup(is->name_mapfile, claims->sub,
191 user, sizeof(user)) == 0)
192 {
193 return 1;
194 }
195 if (!is->onmissing_fail && is->default_user[0] != '\0') {
196 return 1; /* fall back to default_user */
197 }
198 }
199
200 return 0;
201}

203/* brix_token_validate_registry — combined authN+authZ for where the request path
204 * is known at validation time (WebDAV/S3): the authN half then the per-path
205 * strategy ladder. */
206int
207brix_token_validate_registry(const brix_token_registry_args_t *a,
208 const char *req_path, brix_token_op_e op, int *out_issuer_bucket)
209{
210 const brix_token_issuer_t *is;
211
212 *out_issuer_bucket = -1;
213
214 if (brix_token_validate_registry_authn(a, &is) != 0) {
215 return -1;
216 }
217 if (!brix_token_authz_strategy(is, a->claims, req_path, op)) {
218 ngx_log_error(NGX_LOG_WARN, a->log, 0,
219 "brix_token: issuer \"%s\" did not authorize path", is->name);
220 return -1;
221 }
222
223 *out_issuer_bucket = is->metric_bucket;
224 return 0;
225}
