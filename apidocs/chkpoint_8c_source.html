15/*
16 * kXR_chkpoint — checkpoint/transaction write semantics.
17 *
18 * Opcodes:
19 * kXR_ckpBegin (0) — snapshot the file into a temp copy
20 * kXR_ckpCommit (1) — discard the snapshot (writes are permanent)
21 * kXR_ckpRollback (3) — restore the file from the snapshot
22 * kXR_ckpQuery (2) — report checkpoint capacity and usage
23 * kXR_ckpXeq (4) — execute a write sub-operation (write/pgwrite/
24 * truncate/writev) under checkpoint protection
25 *
26 * The checkpoint is stored as a sibling file: <open-path>.ckp.
27 * A non-NULL ckp_path in the brix_file_t slot indicates an active checkpoint.
28 */

40/* WHY: Clean up checkpoint state when no longer active — prevents memory leaks (ckp_path string) and stale size tracking. Called after unlink() in commit/rollback, on copy failure cleanup, and by ckp_xeq.c to reset the file slot between sub-operations. */
41/* HOW: Free f->ckp_path if non-NULL via ngx_free(), set it to NULL; zero out f->ckp_size. No stat() or unlink() — caller handles .ckp file removal before calling this helper. Used exclusively by ckp_commit, ckp_rollback, and ckp_begin failure cleanup paths. */

43/* WHAT: Creates a checkpoint snapshot by copying the entire file to <path>.ckp.
44 * Marks f->ckp_path as non-NULL and stores original size in f->ckp_size.
45 * WHY: Enables transactional write semantics — writes under ckpXeq are "tentative"
46 * until committed; rollback restores the pre-write state from the snapshot.
47 * HOW: 1) Verify no existing checkpoint (f->ckp_path == NULL). 2) Check file size ≤ kXR_ckpMinMax.
48 * 3) Allocate .ckp path string. 4) Create .ckp file with O_CREAT|O_TRUNC. 5) Copy full file via brix_copy_range(). */

57

60 kXR_inProgress, "checkpoint already active");
61 }
62

65 kXR_IOError, strerror(errno));

117 }

121 BRIX_RETURN_OK(ctx, c, BRIX_OP_CHKPOINT, "CHKPOINT", f->path, "begin", 0);

124/* WHAT: Unlinks the .ckp snapshot file and clears f->ckp_path/f->ckp_size. After commit,
125 * all ckpXeq writes become permanent on the original file.
126 * WHY: Transactional write semantics require two-phase commitment — tentative writes (ckpXeq)
127 * are only finalized after explicit commit. Committing means "accept these changes" by
128 * deleting the rollback snapshot.
129 * HOW: Verify f->ckp_path != NULL, unlink .ckp file, call ckp_clear_path() to reset state. */
130
131/* WHAT: Truncates original file to f->ckp_size, then restores content from .ckp snapshot via brix_copy_range().
132 * After rollback, the original file returns to pre-checkpoint state.
133 * WHY: Transactional write semantics provide "undo" capability — if writes under ckpXeq should be rejected,
134 * rollback restores the exact original content and length. This is essential for atomic operations where
135 * partial failures must not leave the file in inconsistent state.
136 * HOW: 1) Verify f->ckp_path != NULL. 2) Run a VFS TRUNCATE job to checkpointed size (may shrink). 3) Copy .ckp→original via brix_copy_range(). 4) unlink + clear_path. */
137
138/* WHAT: Returns ServerResponseBody_ChkPoint with maxCkpSize (kXR_ckpMinMax) and useCkpSize
139 * (size of .ckp file if active, 0 otherwise).
140 * WHY: Clients need to know available checkpoint space before beginning a transaction. Large files
141 * may require more disk space for the snapshot than is available.
142 * HOW: Check f->ckp_path != NULL; stat() .ckp file for current usage; populate response body with max/usage values. */

163ckp_rollback(brix_ctx_t *ctx, ngx_connection_t *c, int idx)
164{
165 brix_file_t *f = &ctx->files[idx];
166 int ckp_fd;
167 brix_vfs_job_t job;
168
169 if (f->ckp_path == NULL) {
170 BRIX_RETURN_ERR(ctx, c, BRIX_OP_CHKPOINT, "CHKPOINT", f->path, "rollback",
171 kXR_InvalidRequest, "no active checkpoint");
172 }
173

183 if (f->ckp_size > 0) {
184 ckp_fd = open(f->ckp_path, O_RDONLY | O_CLOEXEC | O_NOFOLLOW);
185 if (ckp_fd < 0) {

187 kXR_IOError, strerror(errno));
188 }
189

191 f->ckp_path, f->path) != NGX_OK) {
192 close(ckp_fd);
193 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
194 return brix_send_error(ctx, c, kXR_IOError,
195 "checkpoint restore copy failed");
196 }
197 close(ckp_fd);
198 }
199
200 (void) unlink(f->ckp_path);
201 ckp_clear_path(f);
202
203 BRIX_RETURN_OK(ctx, c, BRIX_OP_CHKPOINT, "CHKPOINT", f->path, "rollback", 0);

232brix_handle_chkpoint(brix_ctx_t *ctx, ngx_connection_t *c,
233 ngx_stream_brix_srv_conf_t *conf)
234{
235 xrdw_chkpoint_req_t req;
236 int idx;
237 ngx_int_t validate_rc;
238
239 (void) conf;
240
241 xrdw_chkpoint_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
242 idx = (int)(unsigned char) req.fhandle[0];
243
244 if (!brix_validate_write_handle(ctx, c, idx, "CHKPOINT",
245 BRIX_OP_CHKPOINT, &validate_rc)) {
246 return validate_rc;
247 }
248
249 /* Checkpoint operates on a real file path. Reject handles that have none — a
250 * virtual/path-less handle (e.g. an SSI channel) or any handle opened without
251 * a stored path. Guards ckp_begin's strlen(f->path) from a NULL deref. */
252 if (ctx->files[idx].path == NULL || ctx->files[idx].ssi != NULL) {
253 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
254 return brix_send_error(ctx, c, kXR_ArgInvalid,
255 "checkpoint not supported on this handle");
256 }
257
258 switch ((unsigned char) req.opcode) {
259
260 case kXR_ckpBegin:
261 return ckp_begin(ctx, c, idx);
262
263 case kXR_ckpCommit:
264 return ckp_commit(ctx, c, idx);
265
266 case kXR_ckpQuery:
267 return ckp_query(ctx, c, idx);
268
269 case kXR_ckpRollback:
270 return ckp_rollback(ctx, c, idx);
271
272 case kXR_ckpXeq:
273 return ckp_xeq(ctx, c, idx);
274
275 default:
276 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
277 "brix: kXR_chkpoint unknown opcode=%d",
278 (int)(unsigned char) req.opcode);
279 BRIX_OP_ERR(ctx, BRIX_OP_CHKPOINT);
280 return brix_send_error(ctx, c, kXR_ArgInvalid,
281 "unknown chkpoint opcode");
282 }
283
284/* WHAT: Dispatches kXR_chkpoint sub-operations on req->opcode — routes to ckp_begin, ckp_commit, ckp_query, or ckp_rollback; also handles ckpXeq via ckp_xeq(). Validates the write handle before dispatch. */
285/* WHY: kXR_chkpoint is a compound opcode with 5 sub-codes (begin/commit/query/rollback/Xeq). 286/* HOW: Extracts idx from req->fhandle[0] as unsigned char; calls brix_validate_write_handle() for validation (returns early on failure). switch(req->opcode): kXR_ckpBegin→ckp_begin, kXR_ckpCommit→ckp_commit, kXR_ckpQuery→ckp_query, kXR_ckpRollback→ckp_rollback, kXR_ckpXeq→ckp_xeq. 299ckp_recover_one(ngx_log_t *log, const char *root_canon,
300 const char *ckp_path)
301{
302 char orig_path[PATH_MAX];
303 size_t len;
304 int ckp_fd;
305 brix_staged_file_t staged;
306 struct stat st;
307
308 len = strlen(ckp_path);
309 if (len <= 4 || len >= sizeof(orig_path)) {
310 return NGX_ERROR;
311 }
312
313 ngx_memcpy(orig_path, ckp_path, len - 4);
314 orig_path[len - 4] = '\0';
315
316 ckp_fd = brix_vfs_open_fd(log, root_canon, ckp_path,
317 O_RDONLY | O_CLOEXEC | O_NOFOLLOW, 0);
318 if (ckp_fd < 0) {
319 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
320 "brix: checkpoint recovery cannot open \"%s\"",
321 ckp_path);
322 return NGX_ERROR;
323 }
324
325 if (fstat(ckp_fd, &st) != 0 || !S_ISREG(st.st_mode)) {
326 ngx_close_file(ckp_fd);
327 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
328 "brix: checkpoint recovery invalid snapshot "
329 "\"%s\"", ckp_path);
330 return NGX_ERROR;
331 }
332
333 if (brix_staged_open(log, root_canon, orig_path, O_WRONLY, 0600, 16,
334 &staged) != NGX_OK)
335 {
336 ngx_close_file(ckp_fd);
337 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
338 "brix: checkpoint recovery cannot stage "
339 "\"%s\"", orig_path);
340 return NGX_ERROR;
341 }
342
343 if (st.st_size > 0
344 && brix_copy_range(log, ckp_fd, 0, staged.fd, 0,
345 (size_t) st.st_size, ckp_path,
346 staged.tmp_path) != NGX_OK)
347 {
348 brix_staged_abort(log, root_canon, &staged, 1);
349 ngx_close_file(ckp_fd);
350 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
351 "brix: checkpoint recovery copy failed for "
352 "\"%s\"", orig_path);
353 return NGX_ERROR;
354 }
355
356 {
357 brix_vfs_job_t job;
358
359 brix_vfs_job_sync_init(&job, staged.fd);
360 brix_vfs_io_execute(&job);
361 }
362
363 if (brix_staged_commit(log, root_canon, &staged, orig_path)
364 != NGX_OK)
365 {
366 ngx_close_file(ckp_fd);
367 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
368 "brix: checkpoint recovery commit failed for "
369 "\"%s\"", orig_path);
370 return NGX_ERROR;
371 }
372
373 ngx_close_file(ckp_fd);
374
375 if (brix_vfs_unlink_path(log, root_canon, ckp_path) != 0) {
376 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
377 "brix: checkpoint recovery cannot remove "
378 "\"%s\"", ckp_path);
379 return NGX_ERROR;
380 }
381
382 brix_log_safe_path(log, NGX_LOG_NOTICE, 0,
383 "brix: recovered abandoned checkpoint \"%s\"",
384 ckp_path);
385 return NGX_OK;
386}

388static ngx_int_t
389ckp_recover_scan(ngx_log_t *log, const char *root_canon, const char *dir,
390 ngx_uint_t depth)
391{
392 DIR *dp;
393 int dfd;
394 struct dirent *de;
395
396 if (depth > 128) {
397 brix_log_safe_path(log, NGX_LOG_ERR, 0,
398 "brix: checkpoint recovery depth exceeded at "
399 "\"%s\"", dir);
400 return NGX_ERROR;
401 }
402
403 dfd = brix_vfs_open_fd(log, root_canon, dir,
404 O_RDONLY | O_DIRECTORY | O_CLOEXEC
405 | O_NOFOLLOW, 0);
406 if (dfd < 0) {
407 /*
408 * A SUBDIRECTORY we cannot enter must NOT abort recovery: under per-request
409 * impersonation the export legitimately contains per-user PRIVATE dirs
410 * (e.g. 0700) the worker uid cannot read, and a dir can be removed mid-scan.
411 * Skip those (recovery only concerns this server's own .ckp temp files,
412 * which live in dirs the worker can reach). Only an inaccessible EXPORT
413 * ROOT (depth 0) or an unexpected errno is fatal.
414 */
415 if (depth > 0 && (ngx_errno == EACCES || ngx_errno == ENOENT
416 || ngx_errno == ENOTDIR || ngx_errno == ELOOP))
417 {
418 brix_log_safe_path(log, NGX_LOG_INFO, ngx_errno,
419 "brix: checkpoint recovery skipping "
420 "inaccessible dir \"%s\"", dir);
421 return NGX_OK;
422 }
423 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
424 "brix: checkpoint recovery cannot scan \"%s\"",
425 dir);
426 return NGX_ERROR;
427 }
428
429 dp = fdopendir(dfd);
430 if (dp == NULL) {
431 ngx_close_file(dfd);
432 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
433 "brix: checkpoint recovery cannot scan \"%s\"",
434 dir);
435 return NGX_ERROR;
436 }
437
438 while ((de = readdir(dp)) != NULL) {
439 char path[PATH_MAX];
440 size_t dlen, nlen;
441 struct stat st;
442
443 if (strcmp(de->d_name, ".") == 0 || strcmp(de->d_name, "..") == 0) {
444 continue;
445 }
446
447 dlen = strlen(dir);
448 nlen = strlen(de->d_name);
449 if (dlen + 1 + nlen >= sizeof(path)) {
450 closedir(dp);
451 return NGX_ERROR;
452 }
453
454 ngx_memcpy(path, dir, dlen);
455 path[dlen] = '/';
456 ngx_memcpy(path + dlen + 1, de->d_name, nlen + 1);
457
458 if (fstatat(dirfd(dp), de->d_name, &st, AT_SYMLINK_NOFOLLOW) != 0) {
459 /* A transiently-removed or inaccessible entry: skip it, don't abort
460 * the whole recovery (and thus the worker). */
461 brix_log_safe_path(log, NGX_LOG_INFO, ngx_errno,
462 "brix: checkpoint recovery skipping entry "
463 "\"%s\"", path);
464 continue;
465 }
466
467 if (S_ISDIR(st.st_mode)) {
468 if (ckp_recover_scan(log, root_canon, path, depth + 1) != NGX_OK) {
469 closedir(dp);
470 return NGX_ERROR;
471 }
472 continue;
473 }
474
475 if (S_ISREG(st.st_mode) && ckp_name_has_suffix(de->d_name)) {
476 if (ckp_recover_one(log, root_canon, path) != NGX_OK) {
477 closedir(dp);
478 return NGX_ERROR;
479 }
480 }
481 }
482
483 closedir(dp);
484 return NGX_OK;
485}

488brix_chkpoint_recover_root(ngx_log_t *log, const char *root_canon)
489{
490 char lock_path[PATH_MAX];
491 size_t root_len;
492 int lock_fd;
493 ngx_int_t rc;
494
495 if (root_canon == NULL || root_canon[0] == '\0') {
496 return NGX_OK;
497 }
498 /* A pure cache node (no brix_root) anchors at the "/" namespace: there is no
499 * local export tree, so no checkpoint journal to recover (and "/" is not a
500 * writable place to drop a recovery lock). Nothing to do. */
501 if (root_canon[0] == '/' && root_canon[1] == '\0') {
502 return NGX_OK;
503 }
504
505 root_len = strlen(root_canon);
506 if (root_len + sizeof("/.nginx-xrootd-ckp-recovery.lock")
507 > sizeof(lock_path))
508 {
509 return NGX_ERROR;
510 }
511
512 ngx_memcpy(lock_path, root_canon, root_len);
513 ngx_memcpy(lock_path + root_len, "/.nginx-xrootd-ckp-recovery.lock",
514 sizeof("/.nginx-xrootd-ckp-recovery.lock"));
515
516 lock_fd = open(lock_path, O_CREAT | O_RDWR | O_CLOEXEC | O_NOFOLLOW, 0600);
517 if (lock_fd < 0) {
518 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
519 "brix: checkpoint recovery lock failed "
520 "\"%s\"", lock_path);
521 return NGX_ERROR;
522 }
523
524 if (flock(lock_fd, LOCK_EX) != 0) {
525 ngx_close_file(lock_fd);
526 brix_log_safe_path(log, NGX_LOG_ERR, ngx_errno,
527 "brix: checkpoint recovery cannot lock "
528 "\"%s\"", lock_path);
529 return NGX_ERROR;
530 }
531
532 rc = ckp_recover_scan(log, root_canon, root_canon, 0);
533
534 (void) flock(lock_fd, LOCK_UN);
535 ngx_close_file(lock_fd);
536
537 return rc;
