1/*
2 * xrdhttp.h — XrdHttp protocol extension support for the nginx-xrootd WebDAV module.
3 *
4 * WHAT: Public API for the XrdHttp compatibility layer. Covers:
5 * - Detection of XrdHttp clients via X-Xrootd-Proto header
6 * - Extraction of ?xrd.* and ?tpc.* query parameters per the XrdHttp URL dialect
7 * - Response header injection: X-Xrootd-Requuid, X-Xrootd-Status, X-Xrootd-Wait/Retry
8 * - XrdHttp redirect dialect: X-Xrootd-Redir-Host, X-Xrootd-Redir-Port, opaque passthrough
9 * - Multipart/byteranges GET for efficient vector reads (kXR_readv over HTTP)
10 * - URI-based TPC via ?tpc.src= / ?tpc.dst= query parameters
11 * - ?xrd.stats XML statistics endpoint
12 *
13 * WHY: The XRootD reference implementation (XrdHttp) uses these extensions so that
14 * XRootD-aware HTTP clients (xrdcp --prefer-xrdhttp, ROOT TFile, davix with xrd hints)
15 * get richer error diagnostics, server-side checksum hints, redirect loop prevention,
16 * and high-performance vector reads. Without this layer those clients fall back to
17 * slower code paths or emit spurious errors.
18 *
19 * USAGE: Call xrdhttp_parse_request() once per request (in dispatch, after auth).
20 * Call xrdhttp_add_response_headers() before returning from any method handler.
21 * For multi-range GETs, check xrdhttp_request_is_multirange() and route to
22 * xrdhttp_handle_multipart_get() instead of the standard single-range path.
23 */
24

47/*
48 * Per-request XrdHttp context. Allocated lazily in xrdhttp_parse_request()
49 * from r->pool on first use. Retrieved with xrdhttp_get_ctx().
50 */
51typedef struct {
52 /* Client identification (from X-Xrootd-Proto header). */
53 unsigned is_xrdhttp:1; /* 1 = client sent X-Xrootd-Proto */
54 unsigned headers_injected:1; /* 1 = xrdhttp response headers already added */
55 char proto_version[XRDHTTP_PROTO_MAX]; /* e.g. "5.2", NUL-term */
56
57 /* ?xrd.* query parameters — logged and used for routing/metrics. */
58 char clnt_uuid[XRDHTTP_UUID_MAX]; /* xrd.clnt.uuid */
59 char clnt_app[XRDHTTP_APP_MAX]; /* xrd.clnt.app */
60 char want_cksum[XRDHTTP_CKSUM_MAX]; /* xrd.want.cksum alg */
61 char opaque[XRDHTTP_OPAQUE_MAX]; /* xrd.opaque blob */
62
63 /* ?tpc.* parameters — alternate TPC signaling (besides headers). */
64 char tpc_src[XRDHTTP_TPC_URL_MAX]; /* tpc.src pull URL */
65 char tpc_dst[XRDHTTP_TPC_URL_MAX]; /* tpc.dst push URL */
66 char tpc_key[XRDHTTP_TPC_KEY_MAX]; /* tpc.key security cookie */
67 char tpc_token[XRDHTTP_TPC_KEY_MAX]; /* X-Xrootd-Tpc-Token */
68
69 /* Request UUID echo — sent back in every response as X-Xrootd-Requuid. */
70 char requuid[XRDHTTP_UUID_MAX]; /* from X-Xrootd-Requuid or generated */
71
72 /* Wait/retry — populated by handlers that need back-pressure. */
73 int wait_seconds; /* emit X-Xrootd-Wait: <N> if > 0 */
74 int retry_seconds; /* emit X-Xrootd-Retry: <N> if > 0 */
75
76 /* Phase 21 Step B — streaming Digest over the response body. Set by the
77 * GET path when the client sent "Want-Digest: adler32" and the digest was
78 * not already produced from the open fd. The body filter folds each output
79 * buffer into `adler` and emits "Digest: adler32=<hex>" as a trailer. */
80 unsigned compute_digest:1; /* 1 = accumulate adler32 over the body */
81 unsigned digest_emitted:1; /* 1 = trailer already queued */
82 uint32_t adler; /* running adler32 (seed 1) */
83} xrdhttp_req_ctx_t;

95/*
96 * Parse all XrdHttp signals from the incoming request:
97 * - X-Xrootd-Proto → is_xrdhttp + proto_version
98 * - X-Xrootd-Requuid → requuid (echoed in response)
99 * - X-Xrootd-Tpc-Token → tpc_token
100 * - Query string → clnt_uuid, clnt_app, want_cksum, opaque, tpc_src, tpc_dst, tpc_key
101 * Logs clnt_app and clnt_uuid at debug level for auditing.
102 * Always succeeds (any unparseable value is silently ignored).
103 */

106/* ---- response header injection ---- */

108/*
109 * Inject XrdHttp response headers into r->headers_out before the caller
110 * sends headers:
111 * - X-Xrootd-Requuid: <requuid> (always if is_xrdhttp)
112 * - X-Xrootd-Status: <xrd_code> (for non-2xx status; 0 = success)
113 * - X-Xrootd-Wait: <N> (if ctx->wait_seconds > 0)
114 * - X-Xrootd-Retry: <N> (if ctx->retry_seconds > 0)
115 * Safe to call even when is_xrdhttp==0 — emits nothing in that case.
116 * http_status is the HTTP status code about to be sent (for kXR mapping).
117 */
118ngx_int_t xrdhttp_add_response_headers(ngx_http_request_t *r,
