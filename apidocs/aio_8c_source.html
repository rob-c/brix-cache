8/* Reconnect worker (its own thread). Retries brix_reconnect with exponential
9 * backoff + jitter until success or the reconnect deadline, then signals the loop
10 * via rc_finished + the eventfd. Touches only conn (via brix_reconnect) and the
11 * rc_* handoff fields — never the loop-owned per-conn state. */
12void *
13rc_worker_main(void *arg)
14{
15 brix_aconn *ac = arg;
16 uint64_t seed = (uint64_t) (uintptr_t) ac ^ brix_mono_ns();
17 int attempt = 0;
18 brix_status st;
19 char host[256];
20 int port;
21
22 brix_status_set(&st, XRDC_ESOCK, 0, "reconnect not attempted");
23 snprintf(host, sizeof(host), "%s", ac->conn->host);
24 port = ac->conn->port;
25
26 while (brix_mono_ns() < ac->reconnect_deadline_ns) {
27 if (brix_reconnect(ac->conn, host, port, &st) == 0) {
28 ac->rc_result = 0;
29 __atomic_store_n(&ac->rc_finished, 1, __ATOMIC_RELEASE);
30 uint64_t one = 1;
31 ssize_t w = write(ac->loop->evfd, &one, sizeof(one));
32 (void) w;
33 return NULL;
34 }
35 attempt++;
36 /* Fast reconnect cadence: a transport drop is instant, not server
37 * overload, so retry promptly — 100ms..1s — which is what lets a session
38 * recover on a high packet-loss link where each multi-RTT bring-up is
39 * frequently severed. Total reconnect time is still bounded by max_stall
40 * (reconnect_deadline_ns). */
41 uint64_t base_ms = 100ULL << ((attempt < 4) ? attempt : 4); /* 100ms..1.6s */
42 if (base_ms > 1000) {
43 base_ms = 1000;
44 }
45 seed ^= seed << 13; seed ^= seed >> 7; seed ^= seed << 17; /* xorshift */
46 uint64_t sleep_ms = base_ms + (seed % 100); /* + jitter */
47 uint64_t now = brix_mono_ns();
48 if (now >= ac->reconnect_deadline_ns) {
49 break;
50 }
51 uint64_t remain_ns = ac->reconnect_deadline_ns - now;
52 uint64_t sleep_ns = sleep_ms * 1000000ULL;
53 if (sleep_ns > remain_ns) {
54 sleep_ns = remain_ns;
55 }
56 struct timespec ts = { (time_t) (sleep_ns / 1000000000ULL),
57 (long) (sleep_ns % 1000000000ULL) };
58 nanosleep(&ts, NULL);
59 }
60
61 ac->rc_result = -1;
62 ac->rc_st = st;
63 __atomic_store_n(&ac->rc_finished, 1, __ATOMIC_RELEASE);
64 uint64_t one = 1;
65 ssize_t w = write(ac->loop->evfd, &one, sizeof(one));
66 (void) w;
67 return NULL;
68}

91/* Run a synchronous control command (ADD/CLOSE/STOP) and wait for the loop. */
92void
93loop_run_control(brix_loop *l, cmd_type type, brix_aconn *ac)
94{
95 pthread_mutex_t mx = PTHREAD_MUTEX_INITIALIZER;
96 pthread_cond_t cv = PTHREAD_COND_INITIALIZER;
97 int done = 0;
98
99 cmd *c = (cmd *) calloc(1, sizeof(*c));
100 if (c == NULL) {
101 return; /* best-effort; nothing else we can do */
102 }
103 c->type = type;
104 c->ac = ac;
105 c->dmx = &mx;
106 c->dcv = &cv;
107 c->done = &done;
108
109 loop_push_cmd(l, c);
110
111 pthread_mutex_lock(&mx);
112 while (!done) {
113 pthread_cond_wait(&cv, &mx);
114 }
115 pthread_mutex_unlock(&mx);
116}

119void
120loop_drain_commands(brix_loop *l)
121{
122 pthread_mutex_lock(&l->cq_lock);
123 cmd *head = l->cq_head;
124 l->cq_head = l->cq_tail = NULL;
125 pthread_mutex_unlock(&l->cq_lock);
126
127 while (head != NULL) {
128 cmd *c = head;
129 head = head->next;
130
131 switch (c->type) {
132 case CMD_ADD_ACONN: {
133 brix_aconn *ac = c->ac;
134 if (io_engine_arm(l, ac, EPOLLIN) == 0) {
135 ac->next = l->aconns;
136 l->aconns = ac;
137 } else {
138 ac->dead = 1; /* attach will surface failure via submits */
139 }
140 break;
141 }
142 case CMD_SUBMIT:
143 aconn_submit_cmd(c->ac, c);
144 break;
145 case CMD_CLOSE_ACONN:
146 aconn_destroy(c->ac);
147 break;
148 case CMD_STOP:
149 l->stop = 1;
150 break;
151 }
152
153 if (c->done != NULL) {
154 pthread_mutex_lock(c->dmx);
155 *c->done = 1;
156 pthread_cond_signal(c->dcv);
157 pthread_mutex_unlock(c->dmx);
158 }
159 free(c);
160 }
161}

166/* WHAT: Fold one request's future deadline into the running "nearest deadline"
167 * accumulator *next (ns; 0 = none seen yet).
168 * WHY: Both the reconnecting-parked scan and the alive-inflight scan must track
169 * the soonest un-expired deadline to size the next epoll_wait; sharing one
170 * helper keeps that min-reduction identical across both paths.
171 * HOW: A zero deadline_ns means "no deadline" and is ignored; otherwise the
172 * value replaces *next when it is the first candidate or strictly sooner —
173 * byte-for-byte the original inline `(next == 0 || dl < next)` test. */
174static void
175timeouts_track_next(uint64_t deadline_ns, uint64_t *next)
176{
177 if (deadline_ns != 0 && (*next == 0 || deadline_ns < *next)) {
178 *next = deadline_ns;
179 }
180}
181
182/* WHAT: Expire every parked request on a RECONNECTING conn whose deadline has
183 * passed; fold survivors into *next.
184 * WHY: While a link is rebuilding, parked requests still owe their caller a
185 * deadline; ones that ran out the reconnect budget must fail with the
186 * "(reconnecting)" status, the rest keep the loop's wake timer tight.
187 * HOW: Walks the pending singly-linked list via a pointer-to-pointer so an
188 * expired node is unlinked in place before areq_complete frees it; live
189 * nodes advance the cursor and feed timeouts_track_next. Preserves the
190 * original unlink-then-complete ordering exactly. */
191static void
192timeouts_expire_parked(brix_aconn *ac, uint64_t now, uint64_t *next)
193{
194 brix_areq **pp = &ac->pending;
195 while (*pp != NULL) {
196 brix_areq *r = *pp;
197 if (r->deadline_ns != 0 && now >= r->deadline_ns) {
198 *pp = r->pend_next;
199 brix_status st;
200 brix_status_set(&st, XRDC_ESOCK, ETIMEDOUT,
201 "request deadline exceeded (reconnecting)");
202 areq_complete(r, -1, XRDC_ESOCK, &st);
203 } else {
204 timeouts_track_next(r->deadline_ns, next);
205 pp = &r->pend_next;
206 }
207 }
208}
209
210/* WHAT: Expire every in-flight request on an ALIVE conn whose deadline has
211 * passed; fold survivors into *next. Returns 1 if a timed-out request was
212 * a keepalive ping (the link is presumed dead), else 0.
213 * WHY: Isolating the inflight-slot scan keeps loop_process_timeouts flat and
214 * lets the caller run the transport-error recovery only AFTER the whole
215 * slot array is drained — recursing mid-scan would corrupt the walk.
216 * HOW: Tombstones an expired slot (dropping count, bumping tomb) before
217 * areq_complete frees the request, matching the original bookkeeping order;
218 * live slots feed timeouts_track_next. The ping-timeout flag is only
219 * reported, never acted on here. */
220static int
221timeouts_expire_inflight(brix_aconn *ac, uint64_t now, uint64_t *next)
222{
223 int ping_timed_out = 0;
224 for (uint32_t i = 0; i < ac->inflight.cap; i++) {
225 brix_areq *r = ac->inflight.slots[i];
226 if (r == NULL || r == REQMAP_TOMB || r->deadline_ns == 0) {
227 continue;
228 }
229 if (now >= r->deadline_ns) {
230 ac->inflight.slots[i] = REQMAP_TOMB;
231 ac->inflight.count--;
232 ac->inflight.tomb++;
233 if (r->is_ping) {
234 ping_timed_out = 1;
235 }
236 brix_status st;
237 brix_status_set(&st, XRDC_ESOCK, ETIMEDOUT,
238 "request deadline exceeded");
239 areq_complete(r, -1, XRDC_ESOCK, &st);
240 } else {
241 timeouts_track_next(r->deadline_ns, next);
242 }
243 }
244 return ping_timed_out;
245}

247/* WHAT: Run the deadline scan for one connection, folding its soonest survivor
248 * into *next.
249 * WHY: Each conn state routes to a different expiry path; centralizing the
250 * dispatch keeps the top-level sweep a single clean loop body.
251 * HOW: RECONNECTING → parked-request expiry; ALIVE → inflight expiry followed
252 * by a post-scan transport-error kick when a heartbeat ping timed out
253 * (safe now that the slot array is fully drained); any other state is
254 * skipped, matching the original early-`continue` structure. */
255static void
256timeouts_scan_conn(brix_aconn *ac, uint64_t now, uint64_t *next)
257{
258 if (ac->state == ACONN_RECONNECTING) {
259 timeouts_expire_parked(ac, now, next);
260 return;
261 }
262 if (ac->state != ACONN_ALIVE) {
263 return;
264 }
265
266 if (timeouts_expire_inflight(ac, now, next)) {
267 /* inflight scan for this ac is done — safe to recurse */
268 brix_status st;
269 brix_status_set(&st, XRDC_ESOCK, 0, "keepalive heartbeat timed out");
270 aconn_on_transport_error(ac, &st);
271 }
272}

274/* WHAT: Convert the nearest future deadline into the epoll_wait timeout (ms).
275 * WHY: The sweep produces an absolute ns deadline; the event loop needs a
276 * relative, clamped millisecond budget for its next block.
277 * HOW: No deadline → the idle tick; otherwise round the remaining ns up by 1ms,
278 * clamp to [1, AIO_TICK_MS]. Byte-for-byte the original tail arithmetic. */
279static int
280timeouts_next_wait_ms(uint64_t next, uint64_t now)
281{
282 if (next == 0) {
283 return AIO_TICK_MS;
284 }
285 uint64_t dt_ns = (next > now) ? next - now : 0;
286 int ms = (int) (dt_ns / 1000000ULL) + 1;
287 if (ms > AIO_TICK_MS) {
288 ms = AIO_TICK_MS;
289 }
290 if (ms < 1) {
291 ms = 1;
292 }
293 return ms;
294}
295
296/* Fail any request whose deadline has passed; a timed-out heartbeat ping is
297 * promoted to a transport error (the link is presumed dead -> reconnect). Returns
298 * ms until the next deadline (capped at AIO_TICK_MS) for the next epoll_wait. */
299int
300loop_process_timeouts(brix_loop *l)
301{
302 uint64_t now = brix_mono_ns();
303 uint64_t next = 0; /* nearest future deadline (ns), 0 = none */
304
305 for (brix_aconn *ac = l->aconns; ac != NULL; ac = ac->next) {
306 timeouts_scan_conn(ac, now, &next);
307 }
308
309 return timeouts_next_wait_ms(next, now);
310}

313/* loop */
314void *
315loop_thread(void *arg)
316{
317 brix_loop *l = (brix_loop *) arg;
318 struct epoll_event events[AIO_MAXEV];
319
320 int timeout = AIO_TICK_MS;
321 while (!l->stop) {
322 int n = io_engine_wait(l, events, AIO_MAXEV, timeout);
323
324 loop_drain_commands(l);
325 if (l->stop) {
326 break;
327 }
328
329 for (int i = 0; i < n; i++) {
330 void *ptr = events[i].data.ptr;
331 if (ptr == l) { /* eventfd wakeup */
332 uint64_t v;
333 ssize_t rd = read(l->evfd, &v, sizeof(v));
334 (void) rd;
335 continue;
336 }
337 aconn_handle_io((brix_aconn *) ptr, events[i].events);
338 }
339
340 /* adopt finished reconnects + send heartbeats on idle links */
341 for (brix_aconn *ac = l->aconns; ac != NULL; ac = ac->next) {
342 aconn_poll_reconnect(ac);
343 aconn_maybe_ping(ac);
344 }
345
346 timeout = loop_process_timeouts(l);
347 }
348
349 /* shutdown: tear down every remaining connection (fails their in-flight) */
350 while (l->aconns != NULL) {
351 aconn_destroy(l->aconns);
352 }
353 return NULL;
354}

358/* WHAT: Tear down a partially-initialized loop and return NULL for the caller.
359 * WHY: brix_loop_create acquires epfd, evfd, and an always-initialized cq_lock
360 * in sequence; any step can fail. Centralizing the unwind keeps each error
361 * path a flat `return` (no goto) while freeing exactly what was acquired.
362 * HOW: Both fds start at -1 and are only set once successfully opened, so the
363 * >=0 guards close only real descriptors; cq_lock is initialized before any
364 * failure can reach here, so it is always safe to destroy. Mirrors the
365 * original single-exit `fail:` ladder byte-for-byte. */
366brix_loop *
367brix_loop_create_fail(brix_loop *l)
368{
369 io_engine_teardown(l);
370 pthread_mutex_destroy(&l->cq_lock);
371 free(l);
372 return NULL;
373}

395brix_loop *
396brix_loop_create(brix_status *st)
397{
398 brix_loop *l = (brix_loop *) calloc(1, sizeof(*l));
399 if (l == NULL) {
400 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (loop)");
401 return NULL;
402 }
403 pthread_mutex_init(&l->cq_lock, NULL);
404 l->epfd = -1;
405 l->evfd = -1;
406 l->use_uring = brix_loop_want_uring(); /* phase-44 loop engine (default off) */
407
408 if (io_engine_setup(l, st) != 0) {
409 return brix_loop_create_fail(l);
410 }
411 if (pthread_create(&l->thread, NULL, loop_thread, l) != 0) {
412 brix_status_set(st, XRDC_ESOCK, errno, "pthread_create: %s", strerror(errno));
413 return brix_loop_create_fail(l);
414 }
415 l->thread_ok = 1;
416 return l;
417}

421brix_loop_destroy(brix_loop *l)
422{
423 if (l == NULL) {
424 return;
425 }
426 if (l->thread_ok) {
427 cmd *c = (cmd *) calloc(1, sizeof(*c));
428 if (c != NULL) {

431 } else {

433 uint64_t one = 1;

438 }

440 pthread_mutex_destroy(&l->cq_lock);
441 free(l);

445brix_aconn *
446brix_aconn_attach(brix_loop *l, brix_conn *conn, brix_status *st)
447{
448 if (l == NULL || conn == NULL) {
449 brix_status_set(st, XRDC_EUSAGE, 0, "attach: null argument");
450 return NULL;
451 }
452 if (conn->signing_active) {
453 brix_status_set(st, XRDC_EAUTH, 0,
454 "request signing active — async pipelining not supported; "
455 "use the synchronous path");
456 return NULL;
457 }
458
459 int flags = fcntl(conn->io.fd, F_GETFL, 0);
460 if (flags < 0 || fcntl(conn->io.fd, F_SETFL, flags | O_NONBLOCK) < 0) {
461 brix_status_set(st, XRDC_ESOCK, errno, "set non-blocking: %s", strerror(errno));
462 return NULL;
463 }
464
465 brix_aconn *ac = (brix_aconn *) calloc(1, sizeof(*ac));
466 if (ac == NULL) {
467 brix_status_set(st, XRDC_EPROTO, 0, "out of memory (aconn)");
468 return NULL;
469 }
470 ac->loop = l;
471 ac->conn = conn;
472 ac->fd = conn->io.fd;
473 ac->ssl = conn->io.ssl;
474 ac->next_sid = 1;
475 ac->uring_slot = -1; /* phase-44: no io_uring poll slot yet (0 is valid!) */
476 ac->state = ACONN_ALIVE;
477 ac->last_activity_ns = brix_mono_ns();
478
479 /* resilience defaults (override with brix_aconn_set_resilience) */
480 ac->max_stall_ms = 60000; /* 60 s of reconnect patience */
481 ac->keepalive_ms = 15000; /* heartbeat after 15 s idle */
482 ac->def_retries = 5;
483
484 loop_run_control(l, CMD_ADD_ACONN, ac);
485
486 if (ac->dead) {
487 brix_status_set(st, XRDC_ESOCK, 0, "epoll registration failed");
488 free(ac->inflight.slots);
489 free(ac);
490 return NULL;
491 }
492 return ac;
493}

594int
595brix_aio_call_ex(brix_aconn *ac, const void *hdr24,
596 const void *payload, uint32_t plen,
597 const brix_aio_opts *opts,
598 uint16_t *kxr, uint8_t **body, uint32_t *blen,
599 brix_status *st)
600{
601 call_wait w;
602 memset(&w, 0, sizeof(w));
603 pthread_mutex_init(&w.mx, NULL);
604 pthread_cond_init(&w.cv, NULL);
605
606 if (brix_aio_submit_ex(ac, hdr24, payload, plen, opts, call_cb, &w, st) != 0) {
607 pthread_mutex_destroy(&w.mx);
608 pthread_cond_destroy(&w.cv);
609 return -1;
610 }
611
612 pthread_mutex_lock(&w.mx);
613 while (!w.done) {
614 pthread_cond_wait(&w.cv, &w.mx);
615 }
616 pthread_mutex_unlock(&w.mx);
617 pthread_mutex_destroy(&w.mx);
618 pthread_cond_destroy(&w.cv);
619
620 if (w.rc < 0) {
621 if (st != NULL) {
622 *st = w.st;
623 }
624 free(w.body);
625 return -1;
626 }
627 if (kxr != NULL) {
628 *kxr = w.kxr;
629 }
630 if (body != NULL) {
631 *body = w.body;
632 } else {
633 free(w.body);
634 }
635 if (blen != NULL) {
636 *blen = w.blen;
637 }
638 return 0;
639}
