19/*
20 * WHAT: metrics_writer_t buffer-chain writer — alloc, append, printf, and finish for Prometheus exposition output.
21 * WHY: The /metrics endpoint must emit HELP/TYPE/value lines in the Prometheus text format (0.0.4). A
22 * linked chain of ngx_buf_t buffers provides a growing write surface that can expand when vsnprintf
23 * output exceeds the current buffer capacity — nginx-chain-compatible for direct filter delivery.
24 * HOW: mw_init allocates the first buffer and sets position markers. mw_printf uses vsnprintf with available-space
25 * check; on overflow, mw_append_buffer creates a new chain link and resets positions. mw_select_tail_buffer()
26 * establishes write boundaries (mw->pos / mw->last) within the selected tail buffer. mw_finish marks the last
27 * buffer's last_buf flag so nginx knows this is the complete response chain.
28 */
29
30static ngx_int_t
31mw_alloc_chain_buffer(ngx_pool_t *pool, ngx_chain_t **chain_out)
32{
33 ngx_buf_t *buffer;
34 ngx_chain_t *chain;
35
36 buffer = ngx_create_temp_buf(pool, METRICS_BUF_SIZE);
37 if (buffer == NULL) {
38 return NGX_ERROR;
39 }
40
41 chain = ngx_alloc_chain_link(pool);
42 if (chain == NULL) {
43 return NGX_ERROR;
44 }
45
46 chain->buf = buffer;
47 chain->next = NULL;
48
49 *chain_out = chain;
50 return NGX_OK;
51}

53/* WHAT: Selects the metrics_writer_t's tail ngx_buf_t for appending by initializing mw->pos to buffer's current position (buffer->pos) and setting mw->last = buffer->last + METRICS_BUF_SIZE. This establishes the write boundaries within the tail buffer before subsequent mw_append_buffer calls fill content.
54 * WHY: The metrics writer uses a linked chain of buffers where each new buffer is appended via mw_append_buffer(). Before appending to any buffer, mw_select_tail_buffer() must establish position markers so mw->pos (write start) and mw->last (write end) are correctly set relative to the selected tail buffer. METRICS_BUF_SIZE defines how much beyond buffer->last we consider writable — this accounts for pre-allocated extra space in nginx_buf_t allocation.
55 * HOW: Single three-step assignment → read mw->tail->buf pointer, copy buffer->pos into mw->pos (write start), set mw->last = buffer->last + METRICS_BUF_SIZE (write end boundary). No validation or error handling — assumes tail buffer is valid and pre-allocated with sufficient capacity. */
56
57static void
58mw_select_tail_buffer(metrics_writer_t *mw)
59{
60 ngx_buf_t *buffer;
61
62 buffer = mw->tail->buf;
63 mw->pos = buffer->pos;
64 mw->last = buffer->last + METRICS_BUF_SIZE;
65}

67static ngx_int_t
68mw_append_buffer(metrics_writer_t *mw)
69{
70 ngx_chain_t *chain;
71
72 mw->tail->buf->last = mw->pos;
73
74 if (mw_alloc_chain_buffer(mw->pool, &chain) != NGX_OK) {
75 return NGX_ERROR;
76 }
77
78 mw->tail->next = chain;
79 mw->tail = chain;
80 mw_select_tail_buffer(mw);
81
82 return NGX_OK;
83}

104ngx_int_t
105mw_printf(metrics_writer_t *mw, const char *fmt, ...)
106{
107 va_list args;
108 int formatted_len;
109 size_t available;
110
111 available = (size_t) (mw->last - mw->pos);
112
113 va_start(args, fmt);
114 formatted_len = vsnprintf((char *) mw->pos, available, fmt, args);
115 va_end(args);
116
117 if (formatted_len < 0) {
118 return NGX_ERROR;
119 }
120
121 if ((size_t) formatted_len >= available) {
122 if (mw_append_buffer(mw) != NGX_OK) {
123 return NGX_ERROR;
124 }
125
126 available = METRICS_BUF_SIZE;
127 va_start(args, fmt);
128 formatted_len = vsnprintf((char *) mw->pos, available, fmt, args);
129 va_end(args);
130
131 if (formatted_len < 0 || (size_t) formatted_len >= available) {
132 return NGX_ERROR;
133 }
134 }
135
136 mw->pos += formatted_len;
137 mw->total += formatted_len;
138
139 return NGX_OK;
140}

232 * brix_storage_backend_info_emit — C-7: one info gauge per registered export
233 * describing its composed storage stack: the source backend, its origin (host:port
234 * [+tls]), the auth method threaded through the §14 credential, and whether the
235 * write-back stage decorator (C-2/C-6) is composed. Value is always 1 (the labels
236 * carry the information, Prometheus `_info` convention). Exports are config-fixed
237 * and few, so the export-root label stays low-cardinality.

239static void
240brix_storage_backend_info_emit(metrics_writer_t *mw, ngx_uint_t n)
241{

248

271/* brix_storage_backend_is_local — 1 iff the census backend name is a LOCAL
272 * driver whose export root is a real local filesystem (statvfs-able). Remote/
273 * origin backends (xroot/http/s3/ceph...) have no local volume behind
274 * root_canon — a statvfs there would report the wrong filesystem. */
275static int
276brix_storage_backend_is_local(const char *backend)
277{
278 return strcmp(backend, "posix") == 0 || strcmp(backend, "pblock") == 0;

316 info.root_canon, info.backend,
