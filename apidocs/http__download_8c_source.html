8/* Stream a Content-Length-bounded body (`remaining` bytes) from src to out_fd. */
9int
10stream_clen(body_src *src, long long remaining, int out_fd, long long *written,
11 brix_status *st)
12{
13 char buf[XRDC_XFER_BUF];
14 while (remaining > 0) {
15 size_t want = (remaining < (long long) sizeof(buf))
16 ? (size_t) remaining : sizeof(buf);
17 ssize_t r = bsrc_read(src, buf, want, st);
18 if (r < 0) { return -1; }
19 if (r == 0) {
20 brix_status_set(st, XRDC_EPROTO, 0,
21 "http: body truncated (%lld bytes short)", remaining);
22 return -1;
23 }
24 if (write_all_fd(out_fd, buf, (size_t) r, st) != 0) { return -1; }
25 remaining -= r;
26 *written += r;
27 }
28 return 0;
29}

33int
34stream_eof(body_src *src, int out_fd, long long *written, brix_status *st)
35{
36 char buf[XRDC_XFER_BUF];
37 for (;;) {
38 ssize_t r = bsrc_read(src, buf, sizeof(buf), st);
39 if (r < 0) { return -1; }
40 if (r == 0) { break; }
41 if (write_all_fd(out_fd, buf, (size_t) r, st) != 0) { return -1; }
42 *written += r;
43 }

48/* Stream a Transfer-Encoding: chunked body from src to out_fd. */
49int
50stream_chunked(body_src *src, int out_fd, long long *written, brix_status *st)
51{
52 char buf[XRDC_XFER_BUF];
53 for (;;) {
54 char line[64];
55 long csz;
56 char *endp = NULL;
57 if (bsrc_getline(src, line, sizeof(line), st) != 0) { return -1; }
58 csz = strtol(line, &endp, 16);
59 if (endp == line || csz < 0) {
60 brix_status_set(st, XRDC_EPROTO, 0, "http: bad chunk size");
61 return -1;
62 }
63 if (csz == 0) { break; } /* last chunk */
64 while (csz > 0) {
65 size_t want = (csz < (long) sizeof(buf)) ? (size_t) csz : sizeof(buf);
66 ssize_t r = bsrc_read(src, buf, want, st);
67 if (r <= 0) {
68 brix_status_set(st, XRDC_EPROTO, 0, "http: chunk truncated");
69 return -1;
70 }
71 if (write_all_fd(out_fd, buf, (size_t) r, st) != 0) { return -1; }
72 csz -= r;
73 *written += r;
74 }
75 {
76 char crlf[4]; /* consume the CRLF trailing the chunk data */
77 (void) bsrc_getline(src, crlf, sizeof(crlf), st);
78 }
79 }
80 return 0;
81}

84/* Read the response header block into hdr[]; parse status. Sets *total (bytes in
85 * hdr buffer), *body_off (offset of first body byte in hdr). 0 / -1. */
86int
87read_resp_headers(brix_io *io, char *hdr, size_t hdrcap, int timeout_ms,
88 int *status, size_t *total, size_t *body_off, brix_status *st)
89{
90 size_t n = 0;
91 char *eoh = NULL;
92 while (n < hdrcap - 1) {
93 ssize_t r = httpx_read_some(io, hdr + n, hdrcap - 1 - n, timeout_ms, st);
94 if (r < 0) { return -1; }
95 if (r == 0) { break; }
96 n += (size_t) r;
97 eoh = memmem(hdr, n, "\r\n\r\n", 4);
98 if (eoh != NULL) { break; }
99 }
100 if (eoh == NULL) {
101 brix_status_set(st, XRDC_EPROTO, 0, "http: response headers too large/truncated");
102 return -1;
103 }
104 hdr[n] = '\0';
105 *status = 0;
106 if (strncmp(hdr, "HTTP/", 5) == 0) {
107 char *sp = strchr(hdr, ' ');
108 if (sp != NULL) { *status = atoi(sp + 1); }
109 }
110 *total = n;
111 *body_off = (size_t) (eoh + 4 - hdr);
112 return 0;
113}

116/* Find a header value in a NUL-terminated raw header block (line-anchored,
117 * case-insensitive). 1 if found. */
118int
119raw_header(const char *hdr, const char *name, char *out, size_t outsz)
120{
121 const char *p = hdr;
122 size_t nl = strlen(name);
123 if (outsz == 0) { return 0; }
124 out[0] = '\0';
125 while (*p != '\0') {
126 if (strncasecmp(p, name, nl) == 0 && p[nl] == ':') {
127 const char *v = p + nl + 1, *e;
128 size_t n;
129 while (*v == ' ' || *v == '\t') { v++; }
130 e = v;
131 while (*e != '\0' && *e != '\r' && *e != '\n') { e++; }
132 n = (size_t) (e - v);
133 if (n >= outsz) { n = outsz - 1; }
134 memcpy(out, v, n);
135 out[n] = '\0';
136 return 1;
137 }
138 while (*p != '\0' && *p != '\n') { p++; }
139 if (*p == '\n') { p++; }
140 }
141 return 0;
142}

145/* Stream the 2xx response body to out_fd, picking the framing from the headers.
146 *
147 * WHAT: Given a parsed header block (hdr/total/body_off) and a status, drive the
148 * chunked / Content-Length / EOF body decoder into out_fd and report bytes.
149 * WHY: Keeps the framing decision out of the orchestrator so the latter stays a
150 * flat connect→exchange→teardown sequence (no shared cleanup label).
151 * HOW: Wrap the leftover header bytes + socket in a body_src, then dispatch on
152 * Transfer-Encoding / Content-Length / neither. 0 / -1 (st set on error). */
153int
154httpx_download_body(brix_io *io, char *hdr, size_t total, size_t body_off,
155 int out_fd, int timeout_ms, long long *body_len,
156 brix_status *st)
157{
158 body_src src;
159 char clbuf[32], tebuf[32];
160 long long written = 0;
161 int rc;
162
163 src.io = io; src.lo = hdr; src.lo_len = total; src.lo_off = body_off;
164 src.timeout_ms = timeout_ms;
165 if (raw_header(hdr, "Transfer-Encoding", tebuf, sizeof(tebuf))
166 && strcasecmp(tebuf, "chunked") == 0) {
167 rc = stream_chunked(&src, out_fd, &written, st);
168 } else if (raw_header(hdr, "Content-Length", clbuf, sizeof(clbuf))) {
169 /* Validate the wire value — a negative/garbage length parsed as 0
170 * would silently truncate the body and report success. */
171 char *cend = NULL;
172 long long cl;
173 errno = 0;
174 cl = strtoll(clbuf, &cend, 10);
175 if (cend == clbuf || cl < 0 || errno == ERANGE
176 || (*cend != '\0' && *cend != '\r' && *cend != ' ')) {
177 brix_status_set(st, XRDC_EPROTO, 0,
178 "http: invalid Content-Length \"%s\"", clbuf);
179 rc = -1;
180 } else {
181 rc = stream_clen(&src, cl, out_fd, &written, st);
182 }
183 } else {
184 rc = stream_eof(&src, out_fd, &written, st);
185 }
186 /* Report bytes written even on FAILURE: the resilient download wrapper uses
187 * it as the resume offset (the next Range GET picks up from there). */
188 if (body_len != NULL) { *body_len = written; }
189 return rc;
190}

193/* Send the GET, read the response headers, and stream a 2xx body — over an
194 * already-connected io. Owns its scratch header buffer (allocated, used, freed on
195 * every path) so the orchestrator's transport teardown stays linear. 0 / -1. */
196int
197httpx_download_exchange(brix_io *io, const char *host, int port,
198 const char *path, const char *extra_headers,
199 long long start_off, int out_fd,
200 int timeout_ms, int *http_status, long long *body_len,
201 brix_status *st)
202{
203 char *hdr;
204 char req[2048];
205 char rangeh[64];
206 size_t total = 0, body_off = 0;
207 int status = 0, rlen, rc;
208
209 char hp[300];
210 brix_format_host_port(host, (uint16_t) port, hp, sizeof(hp));
211 /* Resume from start_off via an open-ended byte range (the server replies 206
212 * with the remaining bytes); empty for a fresh download. */
213 rangeh[0] = '\0';
214 if (start_off > 0) {
215 snprintf(rangeh, sizeof(rangeh), "Range: bytes=%lld-\r\n", start_off);
216 }
217 rlen = snprintf(req, sizeof(req),
218 "GET %s HTTP/1.1\r\nHost: %s\r\nUser-Agent: xrdcp\r\n"
219 "Accept: */*\r\nConnection: close\r\n%s%s\r\n",
220 path[0] ? path : "/", hp, rangeh,
221 extra_headers ? extra_headers : "");
222 if (rlen < 0 || (size_t) rlen >= sizeof(req)) {
223 brix_status_set(st, XRDC_EUSAGE, 0, "http: request too long");
224 return -1;
225 }
226 if (brix_write_full(io, req, (size_t) rlen, st) != 0) { return -1; }
227
228 hdr = (char *) malloc(XRDC_HDR_CAP);
229 if (hdr == NULL) {
230 brix_status_set(st, XRDC_EPROTO, 0, "http: out of memory");
231 return -1;
232 }
233 if (read_resp_headers(io, hdr, XRDC_HDR_CAP, timeout_ms, &status,
234 &total, &body_off, st) != 0) {
235 free(hdr);
236 return -1;
237 }
238 if (http_status != NULL) { *http_status = status; }
239
240 /* Only stream the body for a success status (2xx). For anything else the
241 * caller decides; we still drain nothing (Connection: close ends it). */
242 if (status >= 200 && status < 300) {
243 if (start_off > 0 && status != 206) {
244 /* We asked to resume from start_off but the server ignored Range and
245 * is streaming the whole object from 0 — appending it at the resume
246 * offset would corrupt the file, so refuse rather than overwrite. */
247 brix_status_set(st, XRDC_EPROTO, 0,
248 "http: resume Range ignored (status %d)", status);
249 rc = -1;
250 } else {
251 rc = httpx_download_body(io, hdr, total, body_off, out_fd, timeout_ms,
252 body_len, st);
253 }
254 } else {
255 brix_status_set(st, XRDC_EPROTO, 0, "http: server returned status %d", status);
256 rc = -1;
257 }
258 free(hdr);
259 return rc;
260}

263/* ---- Run one connect -> GET -> stream-body attempt over a fresh transport ----
264 *
265 * WHAT: Opens a transport to host:port, seeks the destination to total_got when
266 * resuming, issues the GET, streams a 2xx body into out_fd, and tears the
267 * transport down. Reports the bytes written this attempt via *body_this
268 * (always set, including 0 on a connect failure). Returns 0 on a fully
269 * streamed body, -1 on any connect/exchange fault (st set by the callee).
270 * WHY: Isolates the per-attempt transport lifecycle (connect, optional resume
271 * seek, exchange, TLS/fd teardown) so the retry orchestrator stays a flat
272 * loop with no shared cleanup label and a bounded complexity.
273 * HOW: 1) Zero *body_this so a connect failure contributes no progress.
274 * 2) Connect; bail out -1 if the transport cannot be established.
275 * 3) When resuming (total_got > 0) lseek the output to the resume offset.
276 * 4) Exchange, forwarding http_status only on the first attempt so the
277 * caller records the ORIGINAL server status, not a resume 206.
278 * 5) Free the TLS context and close the socket unconditionally, then
279 * return the exchange result. */
280static int
281http_download_attempt(const char *host, int port, int tls, const char *path,
282 const char *extra_headers, int verify, const char *ca_dir,
283 int out_fd, int timeout_ms, long long total_got,
284 int *http_status, long long *body_this, brix_status *st)
285{
286 brix_io io;
287 void *tls_ctx = NULL;
288 int rc;
289
290 *body_this = 0;
291 if (httpx_connect(&io, host, port, tls, verify, ca_dir, timeout_ms,
292 &tls_ctx, st) != 0) {
293 return -1;
294 }
295 if (total_got > 0) {
296 (void) lseek(out_fd, (off_t) total_got, SEEK_SET);
297 }
298 rc = httpx_download_exchange(&io, host, port, path, extra_headers,
299 total_got, out_fd, timeout_ms,
300 /* report the ORIGINAL status only */
301 total_got == 0 ? http_status : NULL,
302 body_this, st);
303 if (tls) { brix_tls_client_free(&io, tls_ctx); }
304 if (io.fd >= 0) { close(io.fd); }
305 return rc;
306}

309/* ---- Decide whether a faulted download attempt may be retried ----
310 *
311 * WHAT: Returns 1 when another attempt is permitted, 0 when the download must
312 * fail now. Retry is allowed only for a seekable destination with an
313 * active patience window, a transient (retryable) fault, and time left
314 * before the deadline.
315 * WHY: Keeps the four-way retry gate out of the orchestrator so its loop stays
316 * under the cyclomatic-complexity cap and the policy is stated in one place.
317 * HOW: 1) Refuse if the destination is not seekable, the window is disabled
318 * (window_ms <= 0), or the fault is not retryable.
319 * 2) Refuse if the time-since-progress deadline has elapsed.
320 * 3) Otherwise permit a retry. */
321static int
322http_download_should_retry(int seekable, int window_ms, uint64_t deadline,
323 brix_status *st)
324{
325 if (!seekable || window_ms <= 0 || !brix_status_retryable(st)) {
326 return 0;
327 }
328 if (brix_mono_ns() >= deadline) {
329 return 0;
330 }
331 return 1;
332}

335int
336brix_http_download(const char *host, int port, int tls, const char *path,
337 const char *extra_headers, int verify, const char *ca_dir,
338 int out_fd, int timeout_ms, int *http_status,
339 long long *body_len, brix_status *st)
340{
341 long long total_got = 0;
342 int window_ms = httpx_window_ms();
343 unsigned attempt = 0;
344 uint64_t deadline;
345 int seekable;
346 struct stat stt;
347
348 if (http_status != NULL) { *http_status = 0; }
349 if (body_len != NULL) { *body_len = 0; }
350
351 /* Resume requires a seekable destination (a regular file); a pipe/stdout
352 * cannot be rewound, so there we fall back to a single attempt. */
353 seekable = (fstat(out_fd, &stt) == 0 && S_ISREG(stt.st_mode));
354 deadline = brix_mono_ns() + (uint64_t) window_ms * 1000000ULL;
355
356 /*
357 * Deadline-bounded resume (mirrors the root:// resilience window). On a
358 * transport sever mid-download, reconnect and re-issue the GET with
359 * Range: bytes=<bytes-already-written>- (server replies 206), appending the
360 * remainder — so an HTTP download rides out packet loss instead of dying on
361 * the first reset. The deadline measures time-since-progress (it resets when
362 * bytes are written), so a steadily-advancing transfer never times out.
363 * window<=0 ($XRDC_MAX_STALL_MS=0) or a non-seekable dst ⇒ a single attempt.
364 */
365 for (;;) {
366 long long body_this = 0;
367 int rc;
368
369 rc = http_download_attempt(host, port, tls, path, extra_headers, verify,
370 ca_dir, out_fd, timeout_ms, total_got,
371 http_status, &body_this, st);
372 total_got += body_this;
373 if (rc == 0) {
374 if (body_len != NULL) { *body_len = total_got; }
375 return 0;
376 }
377
378 /* Connect or transfer fault. Retry only if we can resume (seekable),
379 * the fault is transient, and the patience window has not elapsed. */
380 if (!http_download_should_retry(seekable, window_ms, deadline, st)) {
381 return -1;
382 }
383 if (body_this > 0) {
384 deadline = brix_mono_ns() + (uint64_t) window_ms * 1000000ULL;
385 }
386 brix_backoff_sleep_fast(attempt++);
387 }
388}
