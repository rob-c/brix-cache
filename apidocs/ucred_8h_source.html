1/*
2 * ucred.h — per-user backend credential selection helper (phase-1 + phase-2 T2).
3 *
4 * WHAT: Declares the brix_sd_ucred_t result descriptor and the four public
5 * functions that map an authenticated brix_identity_t to a candidate
6 * credential file on disk — an x509 proxy PEM (checked for expiry), a
7 * WLCG bearer token (read verbatim, no expiry check here — the origin
8 * enforces the token's own exp claim), an S3 access-key/secret-key/
9 * region triple (phase-3 T3) for backends that authenticate via SigV4,
10 * or a CephX keyring file (per-user Ceph/RADOS auth, the ceph-peruser
11 * item) for backends that authenticate via librados.
12 *
13 * WHY: Storage backends (e.g. remote root://, WebDAV, S3, Ceph/RADOS)
14 * require a per-user credential when acting on behalf of an
15 * authenticated principal. Credential delegation capture is
16 * unreliable (clients may refuse); the sysadmin pre-provisions
17 * credentials under a shared directory keyed by a fs-safe principal
18 * stem or a short SHA-256 hash. This helper encapsulates the
19 * key-derivation and expiry-check logic so callers never reimplement
20 * the search or validation.
21 *
22 * HOW: brix_sd_ucred_principal() extracts the canonical principal string
23 * from brix_identity_t (DN preferred over subject).
24 * brix_sd_ucred_key() derives a filesystem-safe filename stem:
25 * - verbatim if [A-Za-z0-9@._][A-Za-z0-9@._-]{0,63} (S3/JWT subs);
26 * - "x5h-" + first 32 lowercase hex chars of SHA256 otherwise (DNs).
27 * brix_sd_ucred_select() tries literal then hash candidates:
28 * 1. <dir>/<key>.pem (x509 proxy, expiry-checked);
29 * 2. <dir>/<key>.token (bearer token, no expiry check);
30 * 3. <dir>/<key>.s3 (SigV4 access-key/secret-key/region triple);
31 * 4. <dir>/<key>.keyring (CephX keyring: user id + keyring path).
32 * x509 wins when multiple files exist; bearer wins over s3; s3 wins
33 * over ceph; an expired .pem is a hard DECLINED and does NOT silently
34 * fall through to a .token/.s3/.keyring file for the SAME key.
35 * brix_sd_ucred_resolve() looks up a specific key directly (flush path).
36 *
37 * `.s3` FILE FORMAT (phase-3 T3): exactly 3 lines, LF- or CRLF-terminated,
38 * mode 0600 recommended:
39 * line 1: access key id (required, non-empty)
40 * line 2: secret key (required, non-empty)
41 * line 3: region (optional; defaults to "us-east-1" when the
42 * file has only 2 lines or line 3 is empty)
43 * A file missing ak or sk (empty line 1 or 2, or fewer than 2 non-empty
44 * lines) is treated as malformed and the .s3 candidate is DECLINED — it is
45 * NOT a parse error surfaced to the caller, exactly like an unparseable .pem.
46 * The secret key is sensitive: callers MUST NOT log brix_sd_ucred_t.s3_sk.
47 *
48 * `.keyring` FILE FORMAT (ceph-peruser item): a standard CephX keyring file,
49 * e.g.:
50 * [client.bob]
51 * key = AQBv...==
52 * Only the FIRST `[client.NAME]` section header is parsed (the caller does
53 * not need the key material itself — librados reads the keyring file
54 * directly via rados_conf_set(cluster,"keyring",path)). The BARE id
55 * ("bob", not "client.bob") is stored in out->ceph_user — sd_ceph_user_id()
56 * (sd_ceph.c) re-adds the "client." prefix before rados_create(), matching
57 * the export-level `user` config convention. A keyring with no
58 * `[client.X]` section header is malformed and the .keyring candidate is
59 * DECLINED — it is NOT a parse error surfaced to the caller, exactly like an
60 * unparseable .pem. The keyring PATH (not its contents) is stored in
61 * out->ceph_keyring; callers MUST NOT log its contents (it contains the
62 * CephX secret key).
63 */

98/*
99 * Result descriptor filled by brix_sd_ucred_select() and
100 * brix_sd_ucred_resolve(). All fields are NUL-terminated C strings stored
101 * inline — no heap allocation, safe to pass between worker-thread and
102 * event-loop (no nginx pool dependency).
103 *
104 * Exactly one credential kind is populated per successful select:
105 * is_bearer=0/is_s3=0/is_ceph=0 → x509 (path has the PEM path, all other
106 * kind-specific fields empty);
107 * is_bearer=1 → bearer token (bearer has the token text, path has the
108 * .token file path for logging, s3_ak/s3_sk/ceph_keyring/ceph_user empty);
109 * is_s3=1 → S3 SigV4 triple (s3_ak/s3_sk/s3_region populated, path has the
110 * .s3 file path for logging, bearer/ceph_keyring/ceph_user empty);
111 * is_ceph=1 → CephX keyring (ceph_keyring has the keyring PATH, ceph_user
112 * has the parsed bare user id, path has the .keyring file path for logging,
113 * bearer/s3_* empty). The four kinds are mutually exclusive for a given
114 * key — x509 wins when multiple files exist, bearer wins over s3, s3 wins
115 * over ceph (see brix_sd_ucred_resolve).
116 */
117typedef struct {
118 char principal[BRIX_UCRED_PRINC_MAX]; /* canonical principal string */
119 char key[BRIX_UCRED_KEY_MAX]; /* derived filename stem */
120 char path[BRIX_UCRED_PATH_MAX]; /* absolute path (.pem/.token/.s3/ */
121 /* .keyring) */
122 char bearer[BRIX_UCRED_BEARER_MAX]; /* bearer token text (is_bearer=1) */
123 char s3_ak[BRIX_UCRED_S3_AK_MAX]; /* S3 access key id (is_s3=1) */
124 char s3_sk[BRIX_UCRED_S3_SK_MAX]; /* S3 secret key (is_s3=1; never */
125 /* log this field) */
126 char s3_region[BRIX_UCRED_S3_REGION_MAX]; /* S3 region (is_s3=1) */
127 char ceph_keyring[BRIX_UCRED_CEPH_KEYRING_MAX]; /* keyring PATH (is_ceph=1;
128 * librados reads the file itself;
129 * never log its contents) */
130 char ceph_user[BRIX_UCRED_CEPH_USER_MAX]; /* bare CephX user id, e.g. */
131 /* "bob" from "[client.bob]" */
132 /* (is_ceph=1) */
133 int expired; /* 1 if .pem existed but past */
134 /* notAfter (at least one candidate) */
135 unsigned is_bearer:1; /* 1 when bearer[] is populated */
136 unsigned is_s3:1; /* 1 when s3_ak/s3_sk populated */
137 unsigned is_ceph:1; /* 1 when ceph_keyring/ceph_user set */
138} brix_sd_ucred_t;

140/*
141 * brix_sd_ucred_principal — extract the canonical principal from an identity.
142 *
143 * WHAT: Copies id->dn (if non-empty) else id->subject into buf as a
144 * NUL-terminated C string. Returns NGX_ERROR if the identity is
145 * unauthenticated, both fields are empty, or the chosen field does not
146 * fit within cap bytes (including the NUL).
147 *
148 * WHY: DN is the richer, more stable identifier for GSI/SSS principals;
149 * JWT sub / S3 access key is used only when no DN is present.
150 *
151 * HOW: 1. Reject unauthenticated or both-empty.
152 * 2. Pick dn when dn.len > 0, else subject.
153 * 3. memcpy + explicit NUL (ngx_str_t is not NUL-terminated).
154 */
155ngx_int_t brix_sd_ucred_principal(const brix_identity_t *id,
156 char *buf, size_t cap);

158/*
159 * brix_sd_ucred_key — derive a filesystem-safe credential filename stem.
160 *
161 * WHAT: If principal matches [A-Za-z0-9@._][A-Za-z0-9@._-]{0,63} the key
162 * is the principal verbatim. Otherwise the key is
163 * "x5h-" + the first 32 lowercase hex digits of SHA256(principal)
164 * (i.e. the first 16 bytes of the digest encoded as hex).
165 * Returns NGX_ERROR on empty principal or if the key would overflow cap.
166 *
167 * WHY: Human-manageable filenames for token subjects and S3 access keys;
168 * DNs (which contain '/') always fall through to the hash form so
169 * they never corrupt a directory hierarchy.
170 *
171 * HOW: 1. Classify via ucred_principal_fs_safe (static helper in .c).
172 * 2. Literal path: snprintf verbatim.
173 * 3. Hash path: SHA256 via OpenSSL, snprintf 16-byte hex prefix.
174 */
175ngx_int_t brix_sd_ucred_key(const char *principal, char *key, size_t cap);

177/*
178 * brix_sd_ucred_resolve — look up a credential file by its exact key.
179 *
180 * WHAT: Tries <dir>/<key>.pem (x509 expiry-checked); if absent, <dir>/<key>.token
181 * (bearer, read verbatim); if that is also absent, <dir>/<key>.s3
182 * (SigV4 ak/sk/region triple, phase-3 T3); if that is also absent,
183 * <dir>/<key>.keyring (CephX keyring, ceph-peruser item). Returns
184 * NGX_OK (valid — out->is_bearer / out->is_s3 / out->is_ceph distinguish
185 * the four kinds), NGX_DECLINED (all absent / .pem expired / .s3 or
186 * .keyring malformed — out->expired set on PEM expiry), or NGX_ERROR
187 * (path overflow).
188 *
189 * WHY: Used by flush/write paths that already know the key (e.g. fetched
190 * from the open-time select result) and need a fresh expiry check.
191 * An expired .pem is a hard DECLINED — neither .token, .s3, nor
192 * .keyring is tried as a fallback. x509 wins when multiple files
193 * exist; bearer wins over s3; s3 wins over ceph.
194 *
195 * HOW: 1. snprintf .pem path; overflow → NGX_ERROR.
196 * 2. ucred_check_pem (internal): NGX_OK → return; absent → try .token;
197 * expired-or-error → return DECLINED (no further fallback).
198 * 3. ucred_read_token: NGX_OK → set is_bearer=1, fill bearer, return OK;
199 * absent → try .s3; ucred_read_s3: NGX_OK → set is_s3=1, fill
200 * s3_ak/s3_sk/s3_region, return OK; absent → try .keyring;
201 * ucred_read_keyring: NGX_OK → set is_ceph=1, fill ceph_keyring/
202 * ceph_user, return OK; else → return DECLINED.
203 * 4. On x509 NGX_OK fill out->path/key/is_bearer=0/is_s3=0/is_ceph=0; on
204 * bearer NGX_OK fill out->path/bearer/is_bearer=1; on s3 NGX_OK fill
205 * out->path/s3_ak/s3_sk/s3_region/is_s3=1; on ceph NGX_OK fill
206 * out->path/ceph_keyring/ceph_user/is_ceph=1; out->principal
207 * untouched.
208 */
209ngx_int_t brix_sd_ucred_resolve(const char *dir, const char *key,
210 brix_sd_ucred_t *out);

212/*
213 * brix_sd_ucred_select — map an identity to its best available credential.
214 *
215 * WHAT: Zeroes *out, derives principal into out->principal, then tries:
216 * 1. literal-key candidate <dir>/<principal>.{pem,token,s3,keyring}
217 * (only if fs-safe);
218 * 2. hash-key candidate <dir>/x5h-<hex>.{pem,token,s3,keyring} (always).
219 * Within each candidate, .pem is tried before .token before .s3 before
220 * .keyring: x509 wins when multiple exist, bearer wins over s3, s3
221 * wins over ceph. An expired .pem is a hard DECLINED — neither
222 * .token, .s3, nor .keyring for the SAME key is tried as a silent
223 * fallback. The first candidate whose credential is valid wins
224 * (NGX_OK). When no candidate wins, returns NGX_DECLINED with out->key
225 * set to the hash-form key (so callers can log the filename the
226 * administrator must provision) and out->expired OR'd across both
227 * candidates.
228 *
229 * WHY: A single entry point for all per-user credential lookups; caller
230 * need not know about key derivation or credential-kind selection.
231 *
232 * HOW: 1. Zero *out; derive principal.
233 * 2. Build candidate list (up to 2 entries).
234 * 3. brix_sd_ucred_resolve each in order; first NGX_OK → return.
235 * 4. On all-DECLINED: set out->key to hash key, return NGX_DECLINED.
236 */
237ngx_int_t brix_sd_ucred_select(const char *dir, const brix_identity_t *id,
238 brix_sd_ucred_t *out);

240/*
241 * brix_sd_ucred_wipe — cleanse the secret-bearing fields of a resolved
242 * credential once the caller has finished with it.
243 *
244 * WHAT: OPENSSL_cleanse()s the fields that carry live secret material —
245 * bearer[], s3_sk[], ceph_keyring[] — so the token/secret does not
246 * linger on the worker stack or heap after consumption. Non-secret
247 * fields (principal, key, path, s3_ak, s3_region, ceph_user) and the
248 * kind flags are left intact for post-use logging; NULL is a no-op.
249 *
250 * WHY: Defense-in-depth (T4): minimize secret residency so a later
251 * stack/heap info-leak or a core dump cannot read back a credential
252 * that should have been erased on use. Pairs with the reader-side
253 * cleanse in ucred_read_token/_s3/_keyring.
254 *
255 * HOW: Cleanse each secret field unconditionally (the field is zero-length
256 * when its kind is not populated, so cleansing is harmless).
257 */
258void brix_sd_ucred_wipe(brix_sd_ucred_t *cred);
