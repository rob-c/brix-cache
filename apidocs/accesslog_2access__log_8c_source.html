1/*
2 * WHAT: Format and write an access log line for every XRootD request. Produces a structured
3 * one-line record containing client IP, authentication method, user identity (DN), timestamp,
4 * verb/path/detail triple, result status (OK/ERR), byte count, and request duration in milliseconds.
5 * All fields are sanitized through brix_sanitize_log_string() before inclusion in the log line.
6 */
7
8/* WHY: Every request must be observable for security auditing, capacity planning, and anomaly detection.
9 * Access logs provide three critical data streams: (1) Security — authmethod+identity shows who accessed
10 * what resources; (2) Operations — duration_ms enables SLA monitoring and performance trending; (3)
11 * Capacity — bytes sent per op reveals bandwidth utilization patterns across protocols. Sanitization is
12 * mandatory because wire protocol paths, error messages, and client addresses may contain arbitrary byte
13 * sequences that would corrupt downstream log parsers (Prometheus exporters, ELK collectors). The 4096-byte
14 * buffer accommodates the worst-case expanded sanitized output where each dangerous byte becomes a \xNN escape. */
15
16/* HOW: Six-phase formatting pipeline. Phase 1: Extract client IP from c->addr_text or fallback to '-'.
17 * Phase 2: Determine authmethod (gsi/sss/anon) and identity (ctx->login.dn or '-') based on server config + session state.
18 * Phase 3: Format timestamp using ngx_timeofday() + strftime() in Apache-like "%d/%b/%Y:%H:%M:%S %z" format.
19 * Phase 4: Compute request duration from ngx_current_msec - ctx->req_start with floor at zero (clock skew protection).
20 * Phase 5: Sanitize ALL fields through brix_sanitize_log_string() — client_ip, identity, verb, path, detail, errmsg.
21 * If error but no errmsg provided, generate "code:N" placeholder from errcode. Phase 6: Select line template based on
22 * xrd_ok flag (OK or ERR variant), snprintf into 4096-byte buffer, write to conf->access_log_fd if valid. Early return
23 * if access_log_fd == NGX_INVALID_FILE (logging disabled by config). */
24

31/*
32 * Phase 33 C1 — per-worker access-log batch buffer.
33 *
34 * Previously every logged request did its own write(2). Under high op rates
35 * (steady reads) that syscall is the dominant per-request fixed cost when
36 * logging is on. The nginx stream worker is a single-threaded event loop, so a
37 * single static buffer per worker needs no locking: lines accumulate and are
38 * flushed with one write(2) on buffer-full, a target-fd switch, a 1s timer, and
39 * connection close (brix_on_disconnect → brix_access_log_flush). The log fd
40 * is opened O_APPEND (src/config/runtime_server.c), so a batched multi-line
41 * write stays atomic per call and interleaves cleanly with other workers.
42 */
43#define BRIX_ALOG_BUF_SIZE (64 * 1024)
44

51void
52brix_access_log_flush(void)
53{
54 if (brix_alog_len > 0 && brix_alog_fd != NGX_INVALID_FILE) {
55 (void) ngx_write_fd(brix_alog_fd, brix_alog_buf, brix_alog_len);
56 }
57 brix_alog_len = 0;
58}

60static void

67/* Append one formatted line to the per-worker buffer, flushing as needed. */
68static void
69brix_alog_emit(ngx_fd_t fd, const char *line, size_t n)
70{
71 if (fd == NGX_INVALID_FILE || n == 0) {
72 return;
73 }
74
75 /* Never let lines cross fds: if this line targets a different log than what
76 * is buffered, flush the buffered run first, then adopt the new fd. */
77 if (brix_alog_len > 0 && fd != brix_alog_fd) {
78 brix_access_log_flush();
79 }
80 brix_alog_fd = fd;
81
82 /* A line at least as large as the whole buffer can never be batched; flush
83 * any pending run and write it directly so nothing is dropped/truncated. */
84 if (n >= BRIX_ALOG_BUF_SIZE) {
85 brix_access_log_flush();
86 (void) ngx_write_fd(fd, (void *) line, n);
87 return;
88 }
89
90 if (brix_alog_len + n > BRIX_ALOG_BUF_SIZE) {
91 brix_access_log_flush();
92 }
93 ngx_memcpy(brix_alog_buf + brix_alog_len, line, n);
94 brix_alog_len += n;
95
96 /* Bound how long a buffered line waits on a low-rate connection. The timer
97 * fires once (no rearm here); the next append re-arms it. A flush from any
98 * other path before it fires just makes the eventual fire a no-op. */
99 if (!brix_alog_timer_set) {
100 brix_alog_timer.handler = brix_alog_timer_handler;
101 brix_alog_timer.log = ngx_cycle->log;
102 brix_alog_timer.data = NULL;
103 ngx_add_timer(&brix_alog_timer, 1000);
104 brix_alog_timer_set = 1;
105 }
106}

108void
109brix_log_access(brix_ctx_t *ctx, ngx_connection_t *c,
110 const char *verb, const char *path, const char *detail,
111 ngx_uint_t xrd_ok, uint16_t errcode, const char *errmsg,
112 size_t bytes)
113{
114 ngx_stream_brix_srv_conf_t *conf;
115 ngx_msec_int_t duration_ms;
116 char line[4096];
117 int n;
118 const char *authmethod;
119 const char *identity;
120 char client_ip[INET6_ADDRSTRLEN + 8];
121 char safe_client_ip[128];
122 char safe_identity[1024];
123 char safe_verb[64];
124 char safe_path[1024];
125 char safe_detail[512];
126 char safe_errmsg[1024];
127 ngx_time_t *tp;
128 struct tm tm;
129 char timebuf[64];
130 char errbuf[64];
131
132 conf = ngx_stream_get_module_srv_conf(ctx->session, ngx_stream_brix_module);
133
134 if (conf->access_log_fd == NGX_INVALID_FILE) {
135 return;
136 }
137
138 if (c->addr_text.len == 0
139 || brix_str_cbuf(client_ip, sizeof(client_ip), &c->addr_text) == NULL)
140 {
141 client_ip[0] = '-';
142 client_ip[1] = '\0';
143 }
144
145 if (conf->auth == BRIX_AUTH_GSI) {
146 authmethod = "gsi";
147 identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
148 } else if (conf->auth == BRIX_AUTH_SSS) {
149 authmethod = "sss";
150 identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
151 } else if (conf->auth == BRIX_AUTH_UNIX) {
152 authmethod = "unix";
153 identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
154 } else if (conf->auth == BRIX_AUTH_KRB5) {
155 authmethod = "krb5";
156 identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
157 } else {
158 authmethod = "anon";
159 identity = "-";
160 }
161
162 tp = ngx_timeofday();
163 ngx_libc_localtime(tp->sec, &tm);
164 strftime(timebuf, sizeof(timebuf), "%d/%b/%Y:%H:%M:%S %z", &tm);
165
166 duration_ms = (ngx_msec_int_t) (ngx_current_msec - ctx->req_start);
167 if (duration_ms < 0) {
168 duration_ms = 0;
169 }
170
171 if (!xrd_ok && errmsg == NULL) {
172 snprintf(errbuf, sizeof(errbuf), "code:%u", (unsigned) errcode);
173 errmsg = errbuf;
174 }
175
176 brix_sanitize_log_string(client_ip, safe_client_ip, sizeof(safe_client_ip));
177 brix_sanitize_log_string(identity, safe_identity, sizeof(safe_identity));
178 brix_sanitize_log_string(verb ? verb : "-", safe_verb, sizeof(safe_verb));
179 brix_sanitize_log_string(path ? path : "-", safe_path, sizeof(safe_path));
180 brix_sanitize_log_string(detail ? detail : "-", safe_detail, sizeof(safe_detail));
181 brix_sanitize_log_string(errmsg ? errmsg : "-", safe_errmsg,
182 sizeof(safe_errmsg));
183
184 if (xrd_ok) {
185 n = snprintf(line, sizeof(line),
186 "%s %s \"%s\" [%s] \"%s %s %s\" OK %zu %dms\n",
187 safe_client_ip, authmethod, safe_identity, timebuf,
188 safe_verb, safe_path, safe_detail, bytes,
189 (int) duration_ms);
190 } else {
191 n = snprintf(line, sizeof(line),
192 "%s %s \"%s\" [%s] \"%s %s %s\" ERR %zu %dms \"%s\"\n",
193 safe_client_ip, authmethod, safe_identity, timebuf,
194 safe_verb, safe_path, safe_detail, bytes,
195 (int) duration_ms, safe_errmsg);
196 }
197
198 if (n > 0 && (size_t) n < sizeof(line)) {
199 brix_alog_emit(conf->access_log_fd, line, (size_t) n);
200 }
201}
