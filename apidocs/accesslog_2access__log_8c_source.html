1/*
2 * WHAT: Format and write an access log line for every XRootD request. Produces a structured
3 * one-line record containing client IP, authentication method, user identity (DN), timestamp,
4 * verb/path/detail triple, result status (OK/ERR), byte count, and request duration in milliseconds.
5 * All fields are sanitized through brix_sanitize_log_string() before inclusion in the log line.
6 */
7
8/* WHY: Every request must be observable for security auditing, capacity planning, and anomaly detection.
9 * Access logs provide three critical data streams: (1) Security — authmethod+identity shows who accessed
10 * what resources; (2) Operations — duration_ms enables SLA monitoring and performance trending; (3)
11 * Capacity — bytes sent per op reveals bandwidth utilization patterns across protocols. Sanitization is
12 * mandatory because wire protocol paths, error messages, and client addresses may contain arbitrary byte
13 * sequences that would corrupt downstream log parsers (Prometheus exporters, ELK collectors). The 4096-byte
14 * buffer accommodates the worst-case expanded sanitized output where each dangerous byte becomes a \xNN escape. */
15
16/* HOW: Six-phase formatting pipeline. Phase 1: Extract client IP from c->addr_text or fallback to '-'.
17 * Phase 2: Determine authmethod (gsi/sss/anon) and identity (ctx->login.dn or '-') based on server config + session state.
18 * Phase 3: Format timestamp using ngx_timeofday() + strftime() in Apache-like "%d/%b/%Y:%H:%M:%S %z" format.
19 * Phase 4: Compute request duration from ngx_current_msec - ctx->req_start with floor at zero (clock skew protection).
20 * Phase 5: Sanitize ALL fields through brix_sanitize_log_string() — client_ip, identity, verb, path, detail, errmsg.
21 * If error but no errmsg provided, generate "code:N" placeholder from errcode. Phase 6: Select line template based on
22 * xrd_ok flag (OK or ERR variant), snprintf into 4096-byte buffer, write to conf->access_log_fd if valid. Early return
23 * if access_log_fd == NGX_INVALID_FILE (logging disabled by config). */
24

35/*
36 * Phase 33 C1 — per-worker access-log batch buffer.
37 *
38 * Previously every logged request did its own write(2). Under high op rates
39 * (steady reads) that syscall is the dominant per-request fixed cost when
40 * logging is on. The nginx stream worker is a single-threaded event loop, so a
41 * single static buffer per worker needs no locking: lines accumulate and are
42 * flushed with one write(2) on buffer-full, a target-fd switch, a 1s timer, and
43 * connection close (brix_on_disconnect → brix_access_log_flush). The log fd
44 * is opened O_APPEND (src/config/runtime_server.c), so a batched multi-line
45 * write stays atomic per call and interleaves cleanly with other workers.
46 */
47#define BRIX_ALOG_BUF_SIZE (64 * 1024)
48
49static u_char brix_alog_buf[BRIX_ALOG_BUF_SIZE];
50static size_t brix_alog_len;
51static ngx_fd_t brix_alog_fd = NGX_INVALID_FILE;
52static ngx_event_t brix_alog_timer;
53static ngx_uint_t brix_alog_timer_set;

55void
56brix_access_log_flush(void)
57{
58 if (brix_alog_len > 0 && brix_alog_fd != NGX_INVALID_FILE) {
59 (void) ngx_write_fd(brix_alog_fd, brix_alog_buf, brix_alog_len);
60 }
61 brix_alog_len = 0;
62}

71/*
72 * WHAT: Render the access-log timestamp prefix shared by request and session
73 * logs.
74 * WHY: SESS records must be grep-compatible with the existing access-log
75 * convention and must not drift from it over time.
76 * HOW: Use nginx's cached time and the same strftime pattern access_log.c has
77 * used for request records, wrapping it as "[timestamp] ".
78 */
79size_t
80brix_access_log_time_prefix(char *dst, size_t dst_size)
81{
82 ngx_time_t *tp;
83 struct tm tm;
84 char timebuf[64];
85 int n;
86
87 if (dst == NULL || dst_size == 0) {
88 return 0;
89 }
90
91 tp = ngx_timeofday();
92 ngx_libc_localtime(tp->sec, &tm);
93 if (strftime(timebuf, sizeof(timebuf), "%d/%b/%Y:%H:%M:%S %z", &tm) == 0) {
94 /* On failure the buffer contents are unspecified — emit an empty
95 * timestamp rather than garbage (cannot happen with a 64-byte buf). */
96 timebuf[0] = '\0';
97 }
98
99 n = snprintf(dst, dst_size, "[%s] ", timebuf);
100 if (n <= 0 || (size_t) n >= dst_size) {
101 dst[0] = '\0';
102 return 0;
103 }
104
105 return (size_t) n;
106}

108/* Append one formatted line to the per-worker buffer, flushing as needed. */
109void
110brix_alog_emit(ngx_fd_t fd, const char *line, size_t n)
111{
112 if (fd == NGX_INVALID_FILE || n == 0) {
113 return;
114 }
115
116 /* Never let lines cross fds: if this line targets a different log than what
117 * is buffered, flush the buffered run first, then adopt the new fd. */
118 if (brix_alog_len > 0 && fd != brix_alog_fd) {
119 brix_access_log_flush();
120 }
121 brix_alog_fd = fd;
122
123 /* A line at least as large as the whole buffer can never be batched; flush
124 * any pending run and write it directly so nothing is dropped/truncated. */
125 if (n >= BRIX_ALOG_BUF_SIZE) {
126 brix_access_log_flush();
127 (void) ngx_write_fd(fd, (void *) line, n);
128 return;
129 }
130
131 if (brix_alog_len + n > BRIX_ALOG_BUF_SIZE) {
132 brix_access_log_flush();
133 }
134 ngx_memcpy(brix_alog_buf + brix_alog_len, line, n);
135 brix_alog_len += n;
136
137 /* Bound how long a buffered line waits on a low-rate connection. The timer
138 * fires once (no rearm here); the next append re-arms it. A flush from any
139 * other path before it fires just makes the eventual fire a no-op. */
140 if (!brix_alog_timer_set) {
141 /* Shared dummy connection (src/core/config/process.c): nginx's
142 * --with-debug timer-expiry log reads ngx_event_ident(ev->data)->fd, so a
143 * connection-less timer must not leave ev->data NULL (worker SIGSEGV). */
144 extern ngx_connection_t brix_maint_timer_conn;
145 brix_alog_timer.handler = brix_alog_timer_handler;
146 brix_alog_timer.log = ngx_cycle->log;
147 brix_alog_timer.data = &brix_maint_timer_conn;
148 ngx_add_timer(&brix_alog_timer, 1000);
149 brix_alog_timer_set = 1;
150 }
151}

153/*
154 * WHAT: Static auth-mode → access-log method-label mapping.
155 * WHY: The access-log "authmethod" token for each authenticated mode is a fixed
156 * string; a table keeps the resolver flat and colocates the label set.
157 * HOW: One row per authenticated brix_auth mode. Modes absent from the table
158 * (anonymous/none) fall through to the "anon" default in the resolver.

172/*
173 * WHAT: File-local record carrying the fully-prepared fields of one access-log
174 * line between the compute helpers and the formatter.
175 * WHY: brix_log_access has a frozen 9-param public signature; threading the
176 * derived state through an internal struct (rather than a dozen locals) keeps
177 * each helper below the complexity gate without touching the extern.
178 * HOW: Populated by the resolve/sanitize helpers, then consumed read-only by
179 * brix_access_format_line. All char buffers are sized for the sanitized worst
180 * case, matching the original inline buffers exactly.
181 */
182typedef struct {
183 const char *authmethod;
184 char timebuf[64];
185 ngx_msec_int_t duration_ms;
186 size_t bytes;
187 ngx_uint_t xrd_ok;
188 char safe_client_ip[128];
189 char safe_identity[1024];
190 char safe_verb[64];
191 char safe_path[1024];
192 char safe_detail[512];
193 char safe_errmsg[1024];
194} brix_alog_record_t;

196/*
197 * WHAT: Resolve the access-log auth-method label and identity string for a
198 * request from the server auth mode and login state.
199 * WHY: Isolating this lookup keeps the orchestrator flat and expresses the
200 * fixed-label modes as a data table (coding-standards §8.6).
201 * HOW: Table lookup on conf->auth for the label; identity is the login DN when
202 * present. Any mode not in the table is anonymous → "anon"/"-".
203 */
204static void
205brix_access_resolve_identity(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
206 const char **authmethod, const char **identity)
207{
208 size_t i;
209
210 for (i = 0; i < sizeof(brix_access_authlabels)
211 / sizeof(brix_access_authlabels[0]); i++)
212 {
213 if (conf->auth == brix_access_authlabels[i].auth) {
214 *authmethod = brix_access_authlabels[i].label;
215 *identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
216 return;
217 }
218 }
219
220 *authmethod = "anon";
221 *identity = "-";
222}

224/*
225 * WHAT: Render the bare access-log timestamp into dst.
226 * WHY: brix_log_access emits the timestamp inside its own "[...]" template (not
227 * the "[ts] " prefix form), so it needs the unwrapped strftime output.
228 * HOW: Cached nginx time + the shared strftime pattern; empty string on the
229 * (unreachable, 64-byte-buffer) strftime failure, matching prior behavior.
230 */
231static void
232brix_access_format_time(char *dst, size_t dst_size)
233{
234 ngx_time_t *tp;
235 struct tm tm;
236
237 tp = ngx_timeofday();
238 ngx_libc_localtime(tp->sec, &tm);
239 if (strftime(dst, dst_size, "%d/%b/%Y:%H:%M:%S %z", &tm) == 0) {
240 /* On failure the buffer contents are unspecified — emit an empty
241 * timestamp rather than garbage (cannot happen with a 64-byte buf). */
242 dst[0] = '\0';
243 }
244}

246/*
247 * WHAT: File-local bundle of the raw (un-sanitized) string fields of one
248 * access-log record.
249 * WHY: Lets the sanitize helper take one input group and one output record,
250 * keeping it under the parameter gate.
251 * HOW: client_ip/identity are already normalized to non-NULL; verb/path/detail/
252 * errmsg may be NULL and are mapped to "-" during sanitization.
253 */
254typedef struct {
255 const char *client_ip;
256 const char *identity;
257 const char *verb;
258 const char *path;
259 const char *detail;
260 const char *errmsg;
261} brix_access_raw_t;

263/*
264 * WHAT: Sanitize every wire-derived field of one request into the record's safe
265 * buffers.
266 * WHY: Wire paths/messages/addresses may carry arbitrary bytes that would
267 * corrupt downstream log parsers; sanitization is mandatory before formatting.
268 * HOW: Each field passes through brix_sanitize_log_string into its own
269 * fixed-size buffer, NULLs mapped to "-", identical to the former inline block.
270 */
271static void
272brix_access_sanitize_fields(brix_alog_record_t *rec,
273 const brix_access_raw_t *raw)
274{
275 brix_sanitize_log_string(raw->client_ip, rec->safe_client_ip,
276 sizeof(rec->safe_client_ip));
277 brix_sanitize_log_string(raw->identity, rec->safe_identity,
278 sizeof(rec->safe_identity));
279 brix_sanitize_log_string(raw->verb ? raw->verb : "-", rec->safe_verb,
280 sizeof(rec->safe_verb));
281 brix_sanitize_log_string(raw->path ? raw->path : "-", rec->safe_path,
282 sizeof(rec->safe_path));
283 brix_sanitize_log_string(raw->detail ? raw->detail : "-", rec->safe_detail,
284 sizeof(rec->safe_detail));
285 brix_sanitize_log_string(raw->errmsg ? raw->errmsg : "-", rec->safe_errmsg,
286 sizeof(rec->safe_errmsg));
287}

289/*
290 * WHAT: Format the final access-log line from a fully-prepared record.
291 * WHY: The OK/ERR line templates are byte-frozen (tests grep them); keeping the
292 * formatting in one leaf helper isolates the frozen format strings.
293 * HOW: Select the OK or ERR template on rec->xrd_ok and snprintf into line.
294 * Returns snprintf's result (negative on error, >= dst_size on truncation).
295 */
296static int
297brix_access_format_line(const brix_alog_record_t *rec, char *line,
298 size_t line_size)
299{
300 if (rec->xrd_ok) {
301 return snprintf(line, line_size,
302 "%s %s \"%s\" [%s] \"%s %s %s\" OK %zu %dms\n",
303 rec->safe_client_ip, rec->authmethod,
304 rec->safe_identity, rec->timebuf, rec->safe_verb,
305 rec->safe_path, rec->safe_detail, rec->bytes,
306 (int) rec->duration_ms);
307 }
308
309 return snprintf(line, line_size,
310 "%s %s \"%s\" [%s] \"%s %s %s\" ERR %zu %dms \"%s\"\n",
311 rec->safe_client_ip, rec->authmethod, rec->safe_identity,
312 rec->timebuf, rec->safe_verb, rec->safe_path,
313 rec->safe_detail, rec->bytes, (int) rec->duration_ms,
314 rec->safe_errmsg);
315}

317void
318brix_log_access(brix_ctx_t *ctx, ngx_connection_t *c,
319 const char *verb, const char *path, const char *detail,
320 ngx_uint_t xrd_ok, uint16_t errcode, const char *errmsg,
321 size_t bytes)
322{
323 ngx_stream_brix_srv_conf_t *conf;
324 brix_alog_record_t rec = {0};
325 char line[4096];
326 int n;
327 const char *identity;
328 char client_ip[INET6_ADDRSTRLEN + 8];
329 char errbuf[64];
330
331 conf = ngx_stream_get_module_srv_conf(ctx->session, ngx_stream_brix_module);
332
333 if (conf->access_log_fd == NGX_INVALID_FILE) {
334 return;
335 }
336
337 if (c->addr_text.len == 0
338 || brix_str_cbuf(client_ip, sizeof(client_ip), &c->addr_text) == NULL)
339 {
340 client_ip[0] = '-';
341 client_ip[1] = '\0';
342 }
343
344 brix_access_resolve_identity(ctx, conf, &rec.authmethod, &identity);
345 brix_access_format_time(rec.timebuf, sizeof(rec.timebuf));
346
347 rec.duration_ms = (ngx_msec_int_t) (ngx_current_msec - ctx->req_start);
348 if (rec.duration_ms < 0) {
349 rec.duration_ms = 0;
350 }
351 rec.bytes = bytes;
352 rec.xrd_ok = xrd_ok;
353
354 if (!xrd_ok && errmsg == NULL) {
355 snprintf(errbuf, sizeof(errbuf), "code:%u", (unsigned) errcode);
356 errmsg = errbuf;
357 }
358
359 {
360 brix_access_event_t ev = {
361 .verb = verb, .path = path, .detail = detail,
362 .xrd_ok = xrd_ok, .errcode = errcode, .errmsg = errmsg
363 };
364 brix_access_maybe_sesslog(ctx, conf, &ev);
365 }
366
367 {
368 brix_access_raw_t raw = {
369 .client_ip = client_ip, .identity = identity, .verb = verb,
370 .path = path, .detail = detail, .errmsg = errmsg
371 };
372 brix_access_sanitize_fields(&rec, &raw);
373 }
374
375 n = brix_access_format_line(&rec, line, sizeof(line));
376
377 if (n > 0 && (size_t) n < sizeof(line)) {
378 brix_alog_emit(conf->access_log_fd, line, (size_t) n);
379 }
380}
