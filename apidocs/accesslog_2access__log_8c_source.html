1/*
2 * WHAT: Format and write an access log line for every XRootD request. Produces a structured
3 * one-line record containing client IP, authentication method, user identity (DN), timestamp,
4 * verb/path/detail triple, result status (OK/ERR), byte count, and request duration in milliseconds.
5 * All fields are sanitized through brix_sanitize_log_string() before inclusion in the log line.
6 */
7
8/* WHY: Every request must be observable for security auditing, capacity planning, and anomaly detection.
9 * Access logs provide three critical data streams: (1) Security — authmethod+identity shows who accessed
10 * what resources; (2) Operations — duration_ms enables SLA monitoring and performance trending; (3)
11 * Capacity — bytes sent per op reveals bandwidth utilization patterns across protocols. Sanitization is
12 * mandatory because wire protocol paths, error messages, and client addresses may contain arbitrary byte
13 * sequences that would corrupt downstream log parsers (Prometheus exporters, ELK collectors). The 4096-byte
14 * buffer accommodates the worst-case expanded sanitized output where each dangerous byte becomes a \xNN escape. */
15
16/* HOW: Six-phase formatting pipeline. Phase 1: Extract client IP from c->addr_text or fallback to '-'.
17 * Phase 2: Determine authmethod (gsi/sss/anon) and identity (ctx->login.dn or '-') based on server config + session state.
18 * Phase 3: Format timestamp using ngx_timeofday() + strftime() in Apache-like "%d/%b/%Y:%H:%M:%S %z" format.
19 * Phase 4: Compute request duration from ngx_current_msec - ctx->req_start with floor at zero (clock skew protection).
20 * Phase 5: Sanitize ALL fields through brix_sanitize_log_string() — client_ip, identity, verb, path, detail, errmsg.
21 * If error but no errmsg provided, generate "code:N" placeholder from errcode. Phase 6: Select line template based on
22 * xrd_ok flag (OK or ERR variant), snprintf into 4096-byte buffer, write to conf->access_log_fd if valid. Early return
23 * if access_log_fd == NGX_INVALID_FILE (logging disabled by config). */
24

34/*
35 * Phase 33 C1 — per-worker access-log batch buffer.
36 *
37 * Previously every logged request did its own write(2). Under high op rates
38 * (steady reads) that syscall is the dominant per-request fixed cost when
39 * logging is on. The nginx stream worker is a single-threaded event loop, so a
40 * single static buffer per worker needs no locking: lines accumulate and are
41 * flushed with one write(2) on buffer-full, a target-fd switch, a 1s timer, and
42 * connection close (brix_on_disconnect → brix_access_log_flush). The log fd
43 * is opened O_APPEND (src/config/runtime_server.c), so a batched multi-line
44 * write stays atomic per call and interleaves cleanly with other workers.
45 */
46#define BRIX_ALOG_BUF_SIZE (64 * 1024)
47
48static u_char brix_alog_buf[BRIX_ALOG_BUF_SIZE];
49static size_t brix_alog_len;
50static ngx_fd_t brix_alog_fd = NGX_INVALID_FILE;
51static ngx_event_t brix_alog_timer;
52static ngx_uint_t brix_alog_timer_set;

54void
55brix_access_log_flush(void)
56{
57 if (brix_alog_len > 0 && brix_alog_fd != NGX_INVALID_FILE) {
58 (void) ngx_write_fd(brix_alog_fd, brix_alog_buf, brix_alog_len);
59 }
60 brix_alog_len = 0;
61}

63static void
64brix_alog_timer_handler(ngx_event_t *ev)

66 brix_alog_timer_set = 0;
67 brix_access_log_flush();

70/*
71 * WHAT: Render the access-log timestamp prefix shared by request and session
72 * logs.
73 * WHY: SESS records must be grep-compatible with the existing access-log
74 * convention and must not drift from it over time.
75 * HOW: Use nginx's cached time and the same strftime pattern access_log.c has
76 * used for request records, wrapping it as "[timestamp] ".
77 */
78size_t
79brix_access_log_time_prefix(char *dst, size_t dst_size)
80{
81 ngx_time_t *tp;
82 struct tm tm;
83 char timebuf[64];
84 int n;
85
86 if (dst == NULL || dst_size == 0) {
87 return 0;
88 }
89
90 tp = ngx_timeofday();
91 ngx_libc_localtime(tp->sec, &tm);
92 if (strftime(timebuf, sizeof(timebuf), "%d/%b/%Y:%H:%M:%S %z", &tm) == 0) {
93 /* On failure the buffer contents are unspecified — emit an empty
94 * timestamp rather than garbage (cannot happen with a 64-byte buf). */
95 timebuf[0] = '\0';
96 }
97
98 n = snprintf(dst, dst_size, "[%s] ", timebuf);
99 if (n <= 0 || (size_t) n >= dst_size) {
100 dst[0] = '\0';
101 return 0;
102 }
103
104 return (size_t) n;
105}

107/* Append one formatted line to the per-worker buffer, flushing as needed. */
108void
109brix_alog_emit(ngx_fd_t fd, const char *line, size_t n)
110{
111 if (fd == NGX_INVALID_FILE || n == 0) {
112 return;
113 }
114
115 /* Never let lines cross fds: if this line targets a different log than what
116 * is buffered, flush the buffered run first, then adopt the new fd. */
117 if (brix_alog_len > 0 && fd != brix_alog_fd) {
118 brix_access_log_flush();
119 }
120 brix_alog_fd = fd;
121
122 /* A line at least as large as the whole buffer can never be batched; flush
123 * any pending run and write it directly so nothing is dropped/truncated. */
124 if (n >= BRIX_ALOG_BUF_SIZE) {
125 brix_access_log_flush();
126 (void) ngx_write_fd(fd, (void *) line, n);
127 return;
128 }
129
130 if (brix_alog_len + n > BRIX_ALOG_BUF_SIZE) {
131 brix_access_log_flush();
132 }
133 ngx_memcpy(brix_alog_buf + brix_alog_len, line, n);
134 brix_alog_len += n;
135
136 /* Bound how long a buffered line waits on a low-rate connection. The timer
137 * fires once (no rearm here); the next append re-arms it. A flush from any
138 * other path before it fires just makes the eventual fire a no-op. */
139 if (!brix_alog_timer_set) {
140 /* Shared dummy connection (src/core/config/process.c): nginx's
141 * --with-debug timer-expiry log reads ngx_event_ident(ev->data)->fd, so a
142 * connection-less timer must not leave ev->data NULL (worker SIGSEGV). */
143 extern ngx_connection_t brix_maint_timer_conn;
144 brix_alog_timer.handler = brix_alog_timer_handler;
145 brix_alog_timer.log = ngx_cycle->log;
146 brix_alog_timer.data = &brix_maint_timer_conn;
147 ngx_add_timer(&brix_alog_timer, 1000);
148 brix_alog_timer_set = 1;
149 }
150}

164/*
165 * WHAT: Convert the legacy access-log auth detail into a sesslog auth method.
166 * WHY: Existing root auth handlers already log AUTH records; piggybacking those
167 * calls keeps method-specific auth code untouched.
168 * HOW: Prefer the detail token when it names a method, otherwise derive from

171static brix_sess_am_t
172brix_access_sess_auth_method(brix_ctx_t *ctx,

174{
175 if (brix_access_streq(detail, "gsi")) {
176 return BRIX_SESS_AM_GSI;
177 }
178 if (brix_access_streq(detail, "ztn") || brix_access_streq(detail, "token")) {
179 return BRIX_SESS_AM_TOKEN;
180 }
181 if (brix_access_streq(detail, "sss")) {
182 return BRIX_SESS_AM_SSS;
183 }
184 if (brix_access_streq(detail, "krb5")) {
185 return BRIX_SESS_AM_KRB5;
186 }
187 if (brix_access_streq(detail, "pwd")) {
188 return BRIX_SESS_AM_PWD;
189 }
190 if (brix_access_streq(detail, "unix")) {
191 return BRIX_SESS_AM_UNIX;
192 }
193 if (brix_access_streq(detail, "host")) {
194 return BRIX_SESS_AM_HOST;
195 }

203/*
204 * WHAT: Collapse method-specific auth messages into stable sesslog err tokens.
205 * WHY: AUTH failure text is deliberately human-readable and inconsistent; the
206 * session grammar requires a small closed token set.
207 * HOW: Preserve high-signal substrings where available and fall back to kXR
208 * status mapping.
209 */
210static const char *
211brix_access_sess_auth_err(uint16_t errcode, const char *errmsg, char *scratch,
212 size_t scratch_size)
213{
214 if (brix_access_contains(errmsg, "scope")) {
215 return "scope";
216 }
217 if (brix_access_contains(errmsg, "expired")) {
218 return "expired-cert";
219 }
220 if (brix_access_contains(errmsg, "signature")
221 || brix_access_contains(errmsg, "validation")
222 || brix_access_contains(errmsg, "verification")
223 || brix_access_contains(errmsg, "decrypt"))
224 {
225 return "bad-signature";
226 }
227 if (brix_access_contains(errmsg, "rate limit")) {
228 return "busy";
229 }
230 if (brix_access_contains(errmsg, "not authorized")
231 || brix_access_contains(errmsg, "denied"))
232 {
233 return "permission";
234 }
235
236 return brix_sesslog_err_from_kxr((int) errcode, scratch, scratch_size);
237}

272/*
273 * WHAT: Resolve an OPEN verb to its sesslog mode from the detail token.
274 * WHY: OPEN is the one namespace verb whose direction (read vs write) is not
275 * fixed by the verb name — write/staging/tpc-pull opens count as WRITE, every
276 * other open as READ.
277 * HOW: Inspect the detail substring; return WRITE for any write-shaped open,
278 * READ otherwise. Behavior identical to the former inline OPEN branch.
279 */
280static brix_sess_mode_t
281brix_access_open_mode(const char *detail)
282{
283 if (brix_access_contains(detail, "wr")
284 || brix_access_contains(detail, "staging")
285 || brix_access_contains(detail, "tpc-pull"))
286 {
287 return BRIX_SESS_MODE_WRITE;
288 }
289 return BRIX_SESS_MODE_READ;
290}

292/*
293 * WHAT: Map root access-log verbs to sesslog access modes.
294 * WHY: The legacy access logger is already present on namespace-operation exits;
295 * this table lets it produce uniform ATTEMPT/RESULT pairs without touching every
296 * handler in the root plane.
297 * HOW: OPEN resolves via detail (brix_access_open_mode); every other namespace
298 * verb is a row lookup in brix_access_verb_modes. Lifecycle and pure I/O verbs
299 * match nothing and return 0 (they aggregate into AUTH, XFER, or END).
300 */
301static ngx_flag_t
302brix_access_sess_mode(const char *verb, const char *detail,
303 brix_sess_mode_t *mode)
304{
305 size_t i;
306
307 if (verb == NULL || mode == NULL) {
308 return 0;
309 }
310
311 if (brix_access_streq(verb, "OPEN")) {
312 *mode = brix_access_open_mode(detail);
313 return 1;
314 }
315
316 for (i = 0; i < sizeof(brix_access_verb_modes)
317 / sizeof(brix_access_verb_modes[0]); i++)
318 {
319 if (brix_access_streq(verb, brix_access_verb_modes[i].verb)) {
320 *mode = brix_access_verb_modes[i].mode;
321 return 1;
322 }
323 }
324
325 return 0;
326}

328/*
329 * WHAT: File-local descriptor of one access-log event to be mirrored into the
330 * sesslog lifecycle.
331 * WHY: The mirror entry point needs the raw (un-sanitized) request fields;
332 * grouping them into one struct keeps brix_access_maybe_sesslog under the
333 * parameter gate without altering what it consumes.
334 * HOW: Populated once in brix_log_access from its own arguments and passed by
335 * const pointer to the mirror helper and its AUTH/namespace sub-helpers.
336 */
337typedef struct {
338 const char *verb;
339 const char *path;
340 const char *detail;
341 ngx_uint_t xrd_ok;
342 uint16_t errcode;
343 const char *errmsg;
344} brix_access_event_t;

346/*
347 * WHAT: Emit the sesslog AUTH event for an AUTH access-log record.
348 * WHY: Keeps the AUTH-specific method/identity/err derivation out of the mirror
349 * dispatcher so each stays single-purpose.
350 * HOW: On success pass DN/VO (or "-"); on failure derive a stable err token.
351 */
352static void
353brix_access_sesslog_auth(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
354 const brix_access_event_t *ev)
355{
356 char errscratch[64];
357 const char *err;
358
359 err = ev->xrd_ok ? NULL
360 : brix_access_sess_auth_err(ev->errcode, ev->errmsg,
361 errscratch, sizeof(errscratch));
362 brix_sess_auth(ctx->sess, ev->xrd_ok,
363 brix_access_sess_auth_method(ctx, conf, ev->detail),
364 ev->xrd_ok && ctx->login.dn[0] != '\0'
365 ? ctx->login.dn : "-",
366 ev->xrd_ok && ctx->login.primary_vo[0] != '\0'
367 ? ctx->login.primary_vo : "-",
368 err);
369}

371/*
372 * WHAT: Emit the sesslog ATTEMPT/RESULT pair for a namespace access-log record.
373 * WHY: Isolates the namespace-verb path (mode classification + attempt/result)
374 * from the AUTH path in the dispatcher.
375 * HOW: Classify the verb into a mode; ignore verbs with no mode; otherwise emit
376 * ATTEMPT then RESULT with a derived err token on failure.
377 */
378static void
379brix_access_sesslog_namespace(brix_ctx_t *ctx, const brix_access_event_t *ev)
380{
381 char errscratch[64];
382 brix_sess_mode_t mode = 0;
383 const char *err;
384 const char *path;
385
386 if (!brix_access_sess_mode(ev->verb, ev->detail, &mode)) {
387 return;
388 }
389
390 err = ev->xrd_ok ? NULL
391 : brix_sesslog_err_from_kxr((int) ev->errcode, errscratch,
392 sizeof(errscratch));
393 path = ev->path != NULL ? ev->path : "-";
394 brix_sess_attempt(ctx->sess, path, mode);
395 brix_sess_result(ctx->sess, ev->xrd_ok, path, mode, err);
396}

398/*
399 * WHAT: Mirror selected legacy access-log calls into the sesslog lifecycle.
400 * WHY: Root handlers already have well-audited success/error exits; using this
401 * chokepoint gives root:// ATTEMPT/RESULT and AUTH coverage with minimal risk.
402 * HOW: AUTH emits auth events; namespace verbs emit immediate ATTEMPT followed
403 * by RESULT; pure data I/O is intentionally ignored and summarized by XFER.
404 */
405static void
406brix_access_maybe_sesslog(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
407 const brix_access_event_t *ev)
408{
409 if (ctx == NULL || ctx->sess == NULL || ev->verb == NULL) {
410 return;
411 }
412
413 if (brix_access_streq(ev->verb, "AUTH")) {
414 brix_access_sesslog_auth(ctx, conf, ev);
415 return;
416 }
417
418 brix_access_sesslog_namespace(ctx, ev);
419}

440/*
441 * WHAT: File-local record carrying the fully-prepared fields of one access-log
442 * line between the compute helpers and the formatter.
443 * WHY: brix_log_access has a frozen 9-param public signature; threading the
444 * derived state through an internal struct (rather than a dozen locals) keeps
445 * each helper below the complexity gate without touching the extern.
446 * HOW: Populated by the resolve/sanitize helpers, then consumed read-only by
447 * brix_access_format_line. All char buffers are sized for the sanitized worst

459 char safe_path[1024];

464/*
465 * WHAT: Resolve the access-log auth-method label and identity string for a
466 * request from the server auth mode and login state.
467 * WHY: Isolating this lookup keeps the orchestrator flat and expresses the
468 * fixed-label modes as a data table (coding-standards §8.6).
469 * HOW: Table lookup on conf->auth for the label; identity is the login DN when
470 * present. Any mode not in the table is anonymous → "anon"/"-".
471 */
472static void
473brix_access_resolve_identity(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
474 const char **authmethod, const char **identity)
475{
476 size_t i;
477
478 for (i = 0; i < sizeof(brix_access_authlabels)
479 / sizeof(brix_access_authlabels[0]); i++)
480 {
481 if (conf->auth == brix_access_authlabels[i].auth) {
482 *authmethod = brix_access_authlabels[i].label;
483 *identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
484 return;
485 }
486 }
487
488 *authmethod = "anon";
489 *identity = "-";
490}

531/*

533 * buffers.

535 * corrupt downstream log parsers; sanitization is mandatory before formatting.
536 * HOW: Each field passes through brix_sanitize_log_string into its own
537 * fixed-size buffer, NULLs mapped to "-", identical to the former inline block.

539static void

542{
543 brix_sanitize_log_string(raw->client_ip, rec->safe_client_ip,

545 brix_sanitize_log_string(raw->identity, rec->safe_identity,

547 brix_sanitize_log_string(raw->verb ? raw->verb : "-", rec->safe_verb,
548 sizeof(rec->safe_verb));
549 brix_sanitize_log_string(raw->path ? raw->path : "-", rec->safe_path,
550 sizeof(rec->safe_path));
551 brix_sanitize_log_string(raw->detail ? raw->detail : "-", rec->safe_detail,

553 brix_sanitize_log_string(raw->errmsg ? raw->errmsg : "-", rec->safe_errmsg,
554 sizeof(rec->safe_errmsg));

557/*
558 * WHAT: Format the final access-log line from a fully-prepared record.
559 * WHY: The OK/ERR line templates are byte-frozen (tests grep them); keeping the
560 * formatting in one leaf helper isolates the frozen format strings.
561 * HOW: Select the OK or ERR template on rec->xrd_ok and snprintf into line.
562 * Returns snprintf's result (negative on error, >= dst_size on truncation).
563 */
564static int
565brix_access_format_line(const brix_alog_record_t *rec, char *line,
566 size_t line_size)
567{
568 if (rec->xrd_ok) {
569 return snprintf(line, line_size,
570 "%s %s \"%s\" [%s] \"%s %s %s\" OK %zu %dms\n",
571 rec->safe_client_ip, rec->authmethod,
572 rec->safe_identity, rec->timebuf, rec->safe_verb,
573 rec->safe_path, rec->safe_detail, rec->bytes,
574 (int) rec->duration_ms);
575 }
576
577 return snprintf(line, line_size,
578 "%s %s \"%s\" [%s] \"%s %s %s\" ERR %zu %dms \"%s\"\n",
579 rec->safe_client_ip, rec->authmethod, rec->safe_identity,
580 rec->timebuf, rec->safe_verb, rec->safe_path,
581 rec->safe_detail, rec->bytes, (int) rec->duration_ms,
582 rec->safe_errmsg);
583}
