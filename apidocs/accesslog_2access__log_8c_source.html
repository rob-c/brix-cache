1/*
2 * WHAT: Format and write an access log line for every XRootD request. Produces a structured
3 * one-line record containing client IP, authentication method, user identity (DN), timestamp,
4 * verb/path/detail triple, result status (OK/ERR), byte count, and request duration in milliseconds.
5 * All fields are sanitized through brix_sanitize_log_string() before inclusion in the log line.
6 */
7
8/* WHY: Every request must be observable for security auditing, capacity planning, and anomaly detection.
9 * Access logs provide three critical data streams: (1) Security — authmethod+identity shows who accessed
10 * what resources; (2) Operations — duration_ms enables SLA monitoring and performance trending; (3)
11 * Capacity — bytes sent per op reveals bandwidth utilization patterns across protocols. Sanitization is
12 * mandatory because wire protocol paths, error messages, and client addresses may contain arbitrary byte
13 * sequences that would corrupt downstream log parsers (Prometheus exporters, ELK collectors). The 4096-byte
14 * buffer accommodates the worst-case expanded sanitized output where each dangerous byte becomes a \xNN escape. */
15
16/* HOW: Six-phase formatting pipeline. Phase 1: Extract client IP from c->addr_text or fallback to '-'.
17 * Phase 2: Determine authmethod (gsi/sss/anon) and identity (ctx->login.dn or '-') based on server config + session state.
18 * Phase 3: Format timestamp using ngx_timeofday() + strftime() in Apache-like "%d/%b/%Y:%H:%M:%S %z" format.
19 * Phase 4: Compute request duration from ngx_current_msec - ctx->req_start with floor at zero (clock skew protection).
20 * Phase 5: Sanitize ALL fields through brix_sanitize_log_string() — client_ip, identity, verb, path, detail, errmsg.
21 * If error but no errmsg provided, generate "code:N" placeholder from errcode. Phase 6: Select line template based on
22 * xrd_ok flag (OK or ERR variant), snprintf into 4096-byte buffer, write to conf->access_log_fd if valid. Early return
23 * if access_log_fd == NGX_INVALID_FILE (logging disabled by config). */
24

34/*
35 * Phase 33 C1 — per-worker access-log batch buffer.
36 *
37 * Previously every logged request did its own write(2). Under high op rates
38 * (steady reads) that syscall is the dominant per-request fixed cost when
39 * logging is on. The nginx stream worker is a single-threaded event loop, so a
40 * single static buffer per worker needs no locking: lines accumulate and are
41 * flushed with one write(2) on buffer-full, a target-fd switch, a 1s timer, and
42 * connection close (brix_on_disconnect → brix_access_log_flush). The log fd
43 * is opened O_APPEND (src/config/runtime_server.c), so a batched multi-line
44 * write stays atomic per call and interleaves cleanly with other workers.
45 */
46#define BRIX_ALOG_BUF_SIZE (64 * 1024)
47
48static u_char brix_alog_buf[BRIX_ALOG_BUF_SIZE];
49static size_t brix_alog_len;
50static ngx_fd_t brix_alog_fd = NGX_INVALID_FILE;
51static ngx_event_t brix_alog_timer;
52static ngx_uint_t brix_alog_timer_set;

54void
55brix_access_log_flush(void)
56{
57 if (brix_alog_len > 0 && brix_alog_fd != NGX_INVALID_FILE) {
58 (void) ngx_write_fd(brix_alog_fd, brix_alog_buf, brix_alog_len);
59 }
60 brix_alog_len = 0;
61}

70/*
71 * WHAT: Render the access-log timestamp prefix shared by request and session
72 * logs.
73 * WHY: SESS records must be grep-compatible with the existing access-log
74 * convention and must not drift from it over time.
75 * HOW: Use nginx's cached time and the same strftime pattern access_log.c has
76 * used for request records, wrapping it as "[timestamp] ".
77 */
78size_t
79brix_access_log_time_prefix(char *dst, size_t dst_size)
80{
81 ngx_time_t *tp;
82 struct tm tm;
83 char timebuf[64];
84 int n;
85
86 if (dst == NULL || dst_size == 0) {
87 return 0;
88 }
89
90 tp = ngx_timeofday();
91 ngx_libc_localtime(tp->sec, &tm);
92 strftime(timebuf, sizeof(timebuf), "%d/%b/%Y:%H:%M:%S %z", &tm);
93
94 n = snprintf(dst, dst_size, "[%s] ", timebuf);
95 if (n <= 0 || (size_t) n >= dst_size) {
96 dst[0] = '\0';
97 return 0;
98 }
99
100 return (size_t) n;
101}

103/* Append one formatted line to the per-worker buffer, flushing as needed. */
104void
105brix_alog_emit(ngx_fd_t fd, const char *line, size_t n)
106{
107 if (fd == NGX_INVALID_FILE || n == 0) {
108 return;
109 }
110
111 /* Never let lines cross fds: if this line targets a different log than what
112 * is buffered, flush the buffered run first, then adopt the new fd. */
113 if (brix_alog_len > 0 && fd != brix_alog_fd) {
114 brix_access_log_flush();
115 }
116 brix_alog_fd = fd;
117
118 /* A line at least as large as the whole buffer can never be batched; flush
119 * any pending run and write it directly so nothing is dropped/truncated. */
120 if (n >= BRIX_ALOG_BUF_SIZE) {
121 brix_access_log_flush();
122 (void) ngx_write_fd(fd, (void *) line, n);
123 return;
124 }
125
126 if (brix_alog_len + n > BRIX_ALOG_BUF_SIZE) {
127 brix_access_log_flush();
128 }
129 ngx_memcpy(brix_alog_buf + brix_alog_len, line, n);
130 brix_alog_len += n;
131
132 /* Bound how long a buffered line waits on a low-rate connection. The timer
133 * fires once (no rearm here); the next append re-arms it. A flush from any
134 * other path before it fires just makes the eventual fire a no-op. */
135 if (!brix_alog_timer_set) {
136 brix_alog_timer.handler = brix_alog_timer_handler;
137 brix_alog_timer.log = ngx_cycle->log;
138 brix_alog_timer.data = NULL;
139 ngx_add_timer(&brix_alog_timer, 1000);
140 brix_alog_timer_set = 1;
141 }
142}

156/*
157 * WHAT: Convert the legacy access-log auth detail into a sesslog auth method.
158 * WHY: Existing root auth handlers already log AUTH records; piggybacking those
159 * calls keeps method-specific auth code untouched.
160 * HOW: Prefer the detail token when it names a method, otherwise derive from
161 * the authenticated ctx/config state.
162 */
163static brix_sess_am_t
164brix_access_sess_auth_method(brix_ctx_t *ctx,
165 ngx_stream_brix_srv_conf_t *conf, const char *detail)
166{
167 if (brix_access_streq(detail, "gsi")) {
168 return BRIX_SESS_AM_GSI;
169 }
170 if (brix_access_streq(detail, "ztn") || brix_access_streq(detail, "token")) {
171 return BRIX_SESS_AM_TOKEN;
172 }
173 if (brix_access_streq(detail, "sss")) {
174 return BRIX_SESS_AM_SSS;
175 }
176 if (brix_access_streq(detail, "krb5")) {
177 return BRIX_SESS_AM_KRB5;
178 }
179 if (brix_access_streq(detail, "pwd")) {
180 return BRIX_SESS_AM_PWD;
181 }
182 if (brix_access_streq(detail, "unix")) {
183 return BRIX_SESS_AM_UNIX;
184 }
185 if (brix_access_streq(detail, "host")) {
186 return BRIX_SESS_AM_HOST;
187 }
188 if (ctx != NULL && ctx->token.auth) {
189 return BRIX_SESS_AM_TOKEN;
190 }
191 return brix_sess_am_from_stream_auth(conf != NULL ? conf->auth
192 : BRIX_AUTH_NONE);
193}

195/*
196 * WHAT: Collapse method-specific auth messages into stable sesslog err tokens.
197 * WHY: AUTH failure text is deliberately human-readable and inconsistent; the
198 * session grammar requires a small closed token set.
199 * HOW: Preserve high-signal substrings where available and fall back to kXR
200 * status mapping.
201 */
202static const char *
203brix_access_sess_auth_err(uint16_t errcode, const char *errmsg, char *scratch,
204 size_t scratch_size)
205{
206 if (brix_access_contains(errmsg, "scope")) {
207 return "scope";
208 }
209 if (brix_access_contains(errmsg, "expired")) {
210 return "expired-cert";
211 }
212 if (brix_access_contains(errmsg, "signature")
213 || brix_access_contains(errmsg, "validation")
214 || brix_access_contains(errmsg, "verification")
215 || brix_access_contains(errmsg, "decrypt"))
216 {
217 return "bad-signature";
218 }
219 if (brix_access_contains(errmsg, "rate limit")) {
220 return "busy";
221 }
222 if (brix_access_contains(errmsg, "not authorized")
223 || brix_access_contains(errmsg, "denied"))
224 {
225 return "permission";
226 }
227
228 return brix_sesslog_err_from_kxr((int) errcode, scratch, scratch_size);
229}

231/*
232 * WHAT: Map root access-log verbs to sesslog access modes.
233 * WHY: The legacy access logger is already present on namespace-operation exits;
234 * this table lets it produce uniform ATTEMPT/RESULT pairs without touching every
235 * handler in the root plane.
236 * HOW: Return 0 for lifecycle and pure I/O verbs that aggregate into AUTH,
237 * XFER, or END; return 1 with a mode for namespace operations.
238 */
239static ngx_flag_t
240brix_access_sess_mode(const char *verb, const char *detail,
241 brix_sess_mode_t *mode)
242{
243 if (verb == NULL || mode == NULL) {
244 return 0;
245 }
246
247 if (brix_access_streq(verb, "OPEN")) {
248 if (brix_access_contains(detail, "wr")
249 || brix_access_contains(detail, "staging")
250 || brix_access_contains(detail, "tpc-pull"))
251 {
252 *mode = BRIX_SESS_MODE_WRITE;
253 } else {
254 *mode = BRIX_SESS_MODE_READ;
255 }
256 return 1;
257 }
258
259 if (brix_access_streq(verb, "STAT")
260 || brix_access_streq(verb, "STATX")
261 || brix_access_streq(verb, "LOCATE")
262 || brix_access_streq(verb, "QUERY")
263 || brix_access_streq(verb, "SET")
264 || brix_access_streq(verb, "READLINK"))
265 {
266 *mode = BRIX_SESS_MODE_META;
267 return 1;
268 }
269
270 if (brix_access_streq(verb, "DIRLIST")) {
271 *mode = BRIX_SESS_MODE_LIST;
272 return 1;
273 }
274
275 if (brix_access_streq(verb, "RM")
276 || brix_access_streq(verb, "RMDIR")
277 || brix_access_streq(verb, "DELETE"))
278 {
279 *mode = BRIX_SESS_MODE_DELETE;
280 return 1;
281 }
282
283 if (brix_access_streq(verb, "MKDIR")
284 || brix_access_streq(verb, "MV")
285 || brix_access_streq(verb, "TRUNCATE")
286 || brix_access_streq(verb, "CHMOD")
287 || brix_access_streq(verb, "SETATTR")
288 || brix_access_streq(verb, "LINK")
289 || brix_access_streq(verb, "SYMLINK"))
290 {
291 *mode = BRIX_SESS_MODE_WRITE;
292 return 1;
293 }
294
295 return 0;
296}

298/*
299 * WHAT: Mirror selected legacy access-log calls into the sesslog lifecycle.
300 * WHY: Root handlers already have well-audited success/error exits; using this
301 * chokepoint gives root:// ATTEMPT/RESULT and AUTH coverage with minimal risk.
302 * HOW: AUTH emits auth events; namespace verbs emit immediate ATTEMPT followed
303 * by RESULT; pure data I/O is intentionally ignored and summarized by XFER.
304 */
305static void
306brix_access_maybe_sesslog(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *conf,
307 const char *verb, const char *path, const char *detail,
308 ngx_uint_t xrd_ok, uint16_t errcode, const char *errmsg)
309{
310 char errscratch[64];
311 brix_sess_mode_t mode;
312 const char *err;
313
314 if (ctx == NULL || ctx->sess == NULL || verb == NULL) {
315 return;
316 }
317
318 if (brix_access_streq(verb, "AUTH")) {
319 err = xrd_ok ? NULL
320 : brix_access_sess_auth_err(errcode, errmsg, errscratch,
321 sizeof(errscratch));
322 brix_sess_auth(ctx->sess, xrd_ok,
323 brix_access_sess_auth_method(ctx, conf, detail),
324 xrd_ok && ctx->login.dn[0] != '\0'
325 ? ctx->login.dn : "-",
326 xrd_ok && ctx->login.primary_vo[0] != '\0'
327 ? ctx->login.primary_vo : "-",
328 err);
329 return;
330 }
331
332 if (!brix_access_sess_mode(verb, detail, &mode)) {
333 return;
334 }
335
336 err = xrd_ok ? NULL
337 : brix_sesslog_err_from_kxr((int) errcode, errscratch,
338 sizeof(errscratch));
339 brix_sess_attempt(ctx->sess, path != NULL ? path : "-", mode);
340 brix_sess_result(ctx->sess, xrd_ok, path != NULL ? path : "-", mode, err);
341}

343void
344brix_log_access(brix_ctx_t *ctx, ngx_connection_t *c,
345 const char *verb, const char *path, const char *detail,
346 ngx_uint_t xrd_ok, uint16_t errcode, const char *errmsg,
347 size_t bytes)
348{
349 ngx_stream_brix_srv_conf_t *conf;
350 ngx_msec_int_t duration_ms;
351 char line[4096];
352 int n;
353 const char *authmethod;
354 const char *identity;
355 char client_ip[INET6_ADDRSTRLEN + 8];
356 char safe_client_ip[128];
357 char safe_identity[1024];
358 char safe_verb[64];
359 char safe_path[1024];
360 char safe_detail[512];
361 char safe_errmsg[1024];
362 ngx_time_t *tp;
363 struct tm tm;
364 char timebuf[64];
365 char errbuf[64];
366
367 conf = ngx_stream_get_module_srv_conf(ctx->session, ngx_stream_brix_module);
368
369 if (conf->access_log_fd == NGX_INVALID_FILE) {
370 return;
371 }
372
373 if (c->addr_text.len == 0
374 || brix_str_cbuf(client_ip, sizeof(client_ip), &c->addr_text) == NULL)
375 {
376 client_ip[0] = '-';
377 client_ip[1] = '\0';
378 }
379
380 if (conf->auth == BRIX_AUTH_GSI) {
381 authmethod = "gsi";
382 identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
383 } else if (conf->auth == BRIX_AUTH_SSS) {
384 authmethod = "sss";
385 identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
386 } else if (conf->auth == BRIX_AUTH_UNIX) {
387 authmethod = "unix";
388 identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
389 } else if (conf->auth == BRIX_AUTH_KRB5) {
390 authmethod = "krb5";
391 identity = (ctx->login.dn[0] != '\0') ? ctx->login.dn : "-";
392 } else {
393 authmethod = "anon";
394 identity = "-";
395 }
396
397 tp = ngx_timeofday();
398 ngx_libc_localtime(tp->sec, &tm);
399 strftime(timebuf, sizeof(timebuf), "%d/%b/%Y:%H:%M:%S %z", &tm);
400
401 duration_ms = (ngx_msec_int_t) (ngx_current_msec - ctx->req_start);
402 if (duration_ms < 0) {
403 duration_ms = 0;
404 }
405
406 if (!xrd_ok && errmsg == NULL) {
407 snprintf(errbuf, sizeof(errbuf), "code:%u", (unsigned) errcode);
408 errmsg = errbuf;
409 }
410
411 brix_access_maybe_sesslog(ctx, conf, verb, path, detail, xrd_ok,
412 errcode, errmsg);
413
414 brix_sanitize_log_string(client_ip, safe_client_ip, sizeof(safe_client_ip));
415 brix_sanitize_log_string(identity, safe_identity, sizeof(safe_identity));
416 brix_sanitize_log_string(verb ? verb : "-", safe_verb, sizeof(safe_verb));
417 brix_sanitize_log_string(path ? path : "-", safe_path, sizeof(safe_path));
418 brix_sanitize_log_string(detail ? detail : "-", safe_detail, sizeof(safe_detail));
419 brix_sanitize_log_string(errmsg ? errmsg : "-", safe_errmsg,
420 sizeof(safe_errmsg));
421
422 if (xrd_ok) {
423 n = snprintf(line, sizeof(line),
424 "%s %s \"%s\" [%s] \"%s %s %s\" OK %zu %dms\n",
425 safe_client_ip, authmethod, safe_identity, timebuf,
426 safe_verb, safe_path, safe_detail, bytes,
427 (int) duration_ms);
428 } else {
429 n = snprintf(line, sizeof(line),
430 "%s %s \"%s\" [%s] \"%s %s %s\" ERR %zu %dms \"%s\"\n",
431 safe_client_ip, authmethod, safe_identity, timebuf,
432 safe_verb, safe_path, safe_detail, bytes,
433 (int) duration_ms, safe_errmsg);
434 }
435
436 if (n > 0 && (size_t) n < sizeof(line)) {
437 brix_alog_emit(conf->access_log_fd, line, (size_t) n);
438 }
439}
