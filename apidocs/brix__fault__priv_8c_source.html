1/*
2 * brix_fault_priv.c — privileged ("root-ful") fault levers for brix-fault-proxy.
3 *
4 * See brix_fault_priv.h for the rationale. This unit is the ONLY place the proxy
5 * touches host-global network state, and it does so through three subsystems, each
6 * fully owned so teardown is a single clean removal:
7 *
8 * netem — a `tc qdisc ... netem` on the configured NIC's egress. We keep a
9 * per-feature fragment table (delay, loss, corrupt, …) and re-emit the
10 * whole qdisc on every change (`tc qdisc replace`), so features compose
11 * and any one can be cleared independently. This is REAL below-TCP
12 * impairment: loss/corrupt force genuine retransmits or checksum drops,
13 * reorder/duplicate deliver genuine out-of-order / doubled PACKETS —
14 * none of which the userland byte-stream levers can honestly produce.
15 * cut — an `nft` table (inet brix_fault_proxy) whose input-hook rules match
16 * THIS proxy's own 4-tuples (listen port / target ports) and hand the
17 * kernel a verdict: `reject with tcp reset` (a correctly-sequenced RST,
18 * i.e. an on-path reset attack), `reject with icmpx ...` (a forged ICMP
19 * unreachable), or `drop` (a true silent black hole — the peer keeps
20 * retransmitting into the void, unlike the userland `hang` which still
21 * completes the TCP handshake and ACKs).
22 * mtu — shrink the NIC MTU to wedge large transfers behind a next-hop MTU
23 * black hole / forced fragmentation, a below-TCP effect no relay can fake.
24 *
25 * NON-GOAL: forging spoofed-source ICMP "fragmentation needed" via a raw socket.
26 * Hardened stacks (RFC 5927) validate the quoted sequence number, which an off-path
27 * injector cannot know, so it is unreliable; `priv mtu` achieves the same wedged-
28 * transfer outcome deterministically.
29 *
30 * NO SHELL: every external command is fork()+execvp() with an explicit argv; the
31 * nft ruleset is piped to `nft -f -` over a pipe. Interface names and numeric
32 * operands are validated before use.
33 */

71/* fork+execvp `argv` with stdout silenced (stderr inherited for diagnostics).
72 * Returns 0 on a clean exit(0), the child's non-zero exit code otherwise, or -1
73 * if the process could not be created. */
74static int
75priv_run(char *const argv[])
76{
77 pid_t pid = fork();
78 if (pid < 0) {
79 return -1;
80 }
81 if (pid == 0) {
82 int nul = open("/dev/null", O_WRONLY);
83 if (nul >= 0) {
84 dup2(nul, STDOUT_FILENO);
85 close(nul);
86 }
87 execvp(argv[0], argv);
88 _exit(127);
89 }
90 int st = 0;
91 while (waitpid(pid, &st, 0) < 0 && errno == EINTR) {
92 /* retry */
93 }
94 if (WIFEXITED(st)) {
95 return WEXITSTATUS(st);
96 }
97 return -1;
98}

100/* fork+execvp `argv`, piping `input` to the child's stdin (for `nft -f -`).
101 * Same return convention as priv_run(). */
102static int
103priv_run_stdin(char *const argv[], const char *input)
104{
105 int pfd[2];
106 if (pipe(pfd) != 0) {
107 return -1;
108 }
109 pid_t pid = fork();
110 if (pid < 0) {
111 close(pfd[0]);
112 close(pfd[1]);
113 return -1;
114 }
115 if (pid == 0) {
116 dup2(pfd[0], STDIN_FILENO);
117 close(pfd[0]);
118 close(pfd[1]);
119 int nul = open("/dev/null", O_WRONLY);
120 if (nul >= 0) {
121 dup2(nul, STDOUT_FILENO);
122 close(nul);
123 }
124 execvp(argv[0], argv);
125 _exit(127);
126 }
127 close(pfd[0]);
128 size_t len = strlen(input), off = 0;
129 while (off < len) {
130 ssize_t w = write(pfd[1], input + off, len - off);
131 if (w < 0) {
132 if (errno == EINTR) {
133 continue;
134 }
135 break;
136 }
137 off += (size_t) w;
138 }
139 close(pfd[1]);
140 int st = 0;
141 while (waitpid(pid, &st, 0) < 0 && errno == EINTR) {
142 /* retry */
143 }
144 return WIFEXITED(st) ? WEXITSTATUS(st) : -1;
145}

149/* An interface name that both matches a conservative charset and actually exists
150 * under /sys/class/net (so a bad --priv-iface fails loudly, not silently). */
151static int
152valid_iface(const char *s)
153{
154 size_t n = strlen(s);
155 if (n == 0 || n >= IFNAMSIZ) {
156 return 0;
157 }
158 for (size_t i = 0; i < n; i++) {
159 char c = s[i];
160 if (!isalnum((unsigned char) c) && c != '.' && c != '_' &&
161 c != '-' && c != '@') {
162 return 0;
163 }
164 }
165 char path[64 + IFNAMSIZ];
166 snprintf(path, sizeof(path), "/sys/class/net/%s", s);
167 struct stat sb;
168 return stat(path, &sb) == 0;
169}

171/* Format a 0..100 percentage token ("3", "0.1" -> "3%", "0.1%"). Returns 0 ok. */
172static int
173fmt_pct(const char *s, char *out, size_t osz)
174{
175 char *end;
176 double v = strtod(s, &end);
177 if (end == s || v < 0.0 || v > 100.0) {
178 return -1;
179 }
180 snprintf(out, osz, "%g%%", v);
181 return 0;
182}

184/* Format a bounded non-negative integer token. max<=0 means "no upper bound". */
185static int
186fmt_uint(const char *s, char *out, size_t osz, long max)

191 return -1;

194 return 0;

197/* A `tc`-style rate token: digits then an optional bit/byte unit. */
198static int
199valid_rate(const char *s)
200{
201 const char *p = s;
202 if (!isdigit((unsigned char) *p)) {
203 return 0;
204 }
205 while (isdigit((unsigned char) *p)) {
206 p++;
207 }
208 static const char *units[] = { "", "bit", "kbit", "mbit", "gbit", "tbit",
209 "bps", "kbps", "mbps", "gbps", NULL };
210 for (int i = 0; units[i]; i++) {
211 if (strcmp(p, units[i]) == 0) {
212 return 1;
213 }
214 }
215 return 0;
216}

220/* Rebuild and install the whole netem qdisc from the current fragment table, or
221 * delete it when every slot is empty. Caller holds g_lock. */
222static int
223netem_apply(void)
224{
225 if (g_iface[0] == '\0') {
226 return -2; /* no interface configured */
227 }
228
229 int empty = 1;
230 for (int i = 0; i < NE_N; i++) {
231 if (g_ne[i][0]) {
232 empty = 0;
233 break;
234 }
235 }
236 if (empty) {
237 if (!g_netem_on) {
238 return 0; /* nothing was ever installed */
239 }
240 char *argv[] = { "tc", "qdisc", "del", "dev", g_iface, "root", NULL };
241 int rc = priv_run(argv);
242 g_netem_on = 0;
243 /* rc 2 == "no such qdisc" (already clean); treat as success. */
244 return (rc == 0 || rc == 2) ? 0 : -1;
245 }
246
247 /* tc qdisc replace dev IFACE root netem <tokens...> */
248 char *argv[8 + NE_N * 8];
249 int ac = 0;
250 argv[ac++] = "tc";
251 argv[ac++] = "qdisc";
252 argv[ac++] = "replace";
253 argv[ac++] = "dev";
254 argv[ac++] = g_iface;
255 argv[ac++] = "root";
256 argv[ac++] = "netem";
257
258 static char frag[NE_N][96]; /* mutable copies; strtok writes into these */
259 for (int i = 0; i < NE_N; i++) {
260 if (!g_ne[i][0]) {
261 continue;
262 }
263 memcpy(frag[i], g_ne[i], sizeof(frag[i]));
264 for (char *t = strtok(frag[i], " "); t; t = strtok(NULL, " ")) {
265 if (ac < (int) (sizeof(argv) / sizeof(argv[0])) - 1) {
266 argv[ac++] = t;
267 }
268 }
269 }
270 argv[ac] = NULL;
271 if (priv_run(argv) != 0) {
272 return -1;
273 }
274 g_netem_on = 1;
275 return 0;
276}

278/* Set the delay slot: "delay <ms>ms [<jit>ms [<corr>% [distribution <d>]]]". */
279static int
280netem_set_delay(char *args, char *reply, size_t rsz)
281{
282 char *ms = strtok(args, " ");
283 char *jit = strtok(NULL, " ");
284 char *corr = strtok(NULL, " ");
285 char *dist = strtok(NULL, " ");
286 if (!ms || strcmp(ms, "off") == 0) {
287 g_ne[NE_DELAY][0] = '\0';
288 goto done;
289 }
290 char mt[24], jt[24], ct[16];
291 if (fmt_uint(ms, mt, sizeof(mt), 0) != 0) {
292 snprintf(reply, rsz, "err: bad delay ms\n");
293 return -1;
294 }
295 int n = snprintf(g_ne[NE_DELAY], sizeof(g_ne[NE_DELAY]), "delay %sms", mt);
296 if (jit && fmt_uint(jit, jt, sizeof(jt), 0) == 0) {
297 n += snprintf(g_ne[NE_DELAY] + n, sizeof(g_ne[NE_DELAY]) - n, " %sms", jt);
298 if (corr && fmt_pct(corr, ct, sizeof(ct)) == 0) {
299 n += snprintf(g_ne[NE_DELAY] + n, sizeof(g_ne[NE_DELAY]) - n, " %s", ct);
300 }
301 }
302 if (dist && (!strcmp(dist, "normal") || !strcmp(dist, "pareto") ||
303 !strcmp(dist, "paretonormal") || !strcmp(dist, "uniform"))) {
304 snprintf(g_ne[NE_DELAY] + n, sizeof(g_ne[NE_DELAY]) - n,
305 " distribution %s", dist);
306 }
307done:
308 return netem_apply();
309}

313netem_set_pct(int slot, const char *verb, char *args, char *reply, size_t rsz)
314{
315 char *p = strtok(args, " ");
316 char *corr = strtok(NULL, " ");

321 char pt[16], ct[16];

333/* Set bursty Gilbert-Elliott loss: "loss gemodel <p>% [<r>% [<1-h>% [<1-k>%]]]".
334 * p = good->bad, r = bad->good, 1-h = loss-in-bad, 1-k = loss-in-good. */
335static int
336netem_set_gemodel(char *args, char *reply, size_t rsz)
337{
338 char *tok[4] = { 0 };
339 int nt = 0;
340 for (char *t = strtok(args, " "); t && nt < 4; t = strtok(NULL, " ")) {
341 tok[nt++] = t;
342 }
343 if (nt == 0 || strcmp(tok[0], "off") == 0) {
344 g_ne[NE_LOSS][0] = '\0';
345 return netem_apply();

347 char buf[96];
348 int n = snprintf(buf, sizeof(buf), "loss gemodel");
349 for (int i = 0; i < nt; i++) {
350 char pt[16];

352 snprintf(reply, rsz, "err: bad gemodel percentage\n");

361/* Dispatch `priv netem <sub> ...`. Caller holds g_lock. */
362static int
363cmd_netem(char *args, char *reply, size_t rsz)
364{
365 char *sub = strtok(args, " ");
366 char *rest = strtok(NULL, ""); /* remainder after the sub-verb */
367 char empty[1] = "";
368 if (!rest) {
369 rest = empty;
370 }
371 if (!sub || strcmp(sub, "show") == 0) {
372 char *w = reply;
373 int left = (int) rsz;
374 int k = snprintf(w, left, "netem[%s]:", g_iface[0] ? g_iface : "?");
375 w += k; left -= k;
376 for (int i = 0; i < NE_N && left > 1; i++) {
377 if (g_ne[i][0]) {
378 int m = snprintf(w, left, " %s", g_ne[i]);
379 w += m; left -= m;
380 }
381 }
382 snprintf(w, left > 0 ? left : 0, "\n");
383 return 0;
384 }
385 if (strcmp(sub, "clear") == 0 || strcmp(sub, "off") == 0) {
386 for (int i = 0; i < NE_N; i++) {
387 g_ne[i][0] = '\0';
388 }
389 return netem_apply();
390 }
391 int rc;
392 if (strcmp(sub, "delay") == 0) {
393 rc = netem_set_delay(rest, reply, rsz);
394 } else if (strcmp(sub, "loss") == 0) {
395 rc = netem_set_pct(NE_LOSS, "loss random", rest, reply, rsz);
396 } else if (strcmp(sub, "loss-gemodel") == 0) {
397 rc = netem_set_gemodel(rest, reply, rsz);
398 } else if (strcmp(sub, "corrupt") == 0) {
399 rc = netem_set_pct(NE_CORRUPT, "corrupt", rest, reply, rsz);
400 } else if (strcmp(sub, "duplicate") == 0) {
401 rc = netem_set_pct(NE_DUP, "duplicate", rest, reply, rsz);
402 } else if (strcmp(sub, "reorder") == 0) {
403 rc = netem_set_pct(NE_REORDER, "reorder", rest, reply, rsz);
404 } else if (strcmp(sub, "rate") == 0) {
405 char *r = strtok(rest, " ");
406 if (!r || strcmp(r, "off") == 0) {
407 g_ne[NE_RATE][0] = '\0';
408 } else if (!valid_rate(r)) {
409 snprintf(reply, rsz, "err: bad rate (e.g. 1mbit)\n");
410 return -1;
411 } else {
412 snprintf(g_ne[NE_RATE], sizeof(g_ne[NE_RATE]), "rate %s", r);
413 }
414 rc = netem_apply();
415 } else if (strcmp(sub, "limit") == 0) {
416 char *l = strtok(rest, " "), lt[24];
417 if (!l || strcmp(l, "off") == 0) {
418 g_ne[NE_LIMIT][0] = '\0';
419 } else if (fmt_uint(l, lt, sizeof(lt), 0) != 0) {
420 snprintf(reply, rsz, "err: bad limit\n");
421 return -1;
422 } else {
423 snprintf(g_ne[NE_LIMIT], sizeof(g_ne[NE_LIMIT]), "limit %s", lt);
424 }
425 rc = netem_apply();
426 } else {
427 snprintf(reply, rsz, "err: unknown netem sub-command '%s'\n", sub);
428 return -1;
429 }
430 if (rc == -2) {
431 snprintf(reply, rsz, "err: no --priv-iface configured for netem\n");
432 } else if (rc != 0 && strncmp(reply, "err:", 4) != 0) {
433 /* A validation handler already left a specific "err: ..." message; only
434 * the qdisc install itself falls through with reply still == "ok". */
435 snprintf(reply, rsz, "err: tc failed (see proxy stderr)\n");
436 } else if (rc == 0 && g_ne[NE_REORDER][0] && !g_ne[NE_DELAY][0]) {
437 snprintf(reply, rsz, "ok (note: reorder needs a delay to take effect)\n");
438 }
439 return rc == 0 ? 0 : -1;
440}

444/* Map a cut mode to an nft verdict clause, or NULL if the mode is unknown. */
445static const char *
446cut_verdict(const char *mode)
447{
448 if (strcmp(mode, "rst") == 0) return "reject with tcp reset";
449 if (strcmp(mode, "drop") == 0) return "drop";
450 if (strcmp(mode, "icmp-admin") == 0) return "reject with icmpx type admin-prohibited";
451 if (strcmp(mode, "icmp-host") == 0) return "reject with icmpx type host-unreachable";
452 if (strcmp(mode, "icmp-net") == 0) return "reject with icmpx type no-route";
453 if (strcmp(mode, "icmp-port") == 0) return "reject with icmpx type port-unreachable";
454 return NULL;
455}

457/* Install (or replace) the nft table cutting `dir` traffic with `mode`. dir:
458 * 0 both / 1 up (proxy<->upstream) / 2 down (proxy<->client). Caller holds
459 * g_lock. */
460static int
461cut_apply(const char *mode, int dir, char *reply, size_t rsz)
462{
463 const char *v = cut_verdict(mode);
464 if (!v) {
465 snprintf(reply, rsz, "err: unknown cut mode '%s'\n", mode);
466 return -1;
467 }
468
469 /* Build a target-port set "{ p1, p2 }" for the upstream direction. */
470 char pset[128] = "";
471 int po = snprintf(pset, sizeof(pset), "{ ");
472 for (int i = 0; i < g_ntports; i++) {
473 po += snprintf(pset + po, sizeof(pset) - po, "%s%d",
474 i ? ", " : "", g_tports[i]);
475 }
476 snprintf(pset + po, sizeof(pset) - po, " }");
477
478 char rs[768];
479 int n = snprintf(rs, sizeof(rs),
480 "add table inet brix_fault_proxy\n"
481 "delete table inet brix_fault_proxy\n"
482 "add table inet brix_fault_proxy\n"
483 "add chain inet brix_fault_proxy in "
484 "{ type filter hook input priority -150; policy accept; }\n");
485 if (dir != 1) { /* down: packets arriving from the client (dport=listen) */
486 n += snprintf(rs + n, sizeof(rs) - n,
487 "add rule inet brix_fault_proxy in tcp dport %d %s\n",
488 g_listen_port, v);
489 }
490 if (dir != 2 && g_ntports > 0) { /* up: replies from upstream (sport=target) */
491 n += snprintf(rs + n, sizeof(rs) - n,
492 "add rule inet brix_fault_proxy in tcp sport %s %s\n", pset, v);
493 }
494
495 char *argv[] = { "nft", "-f", "-", NULL };
496 if (priv_run_stdin(argv, rs) != 0) {
497 snprintf(reply, rsz, "err: nft failed (see proxy stderr)\n");
498 return -1;
499 }
500 g_nft_on = 1;
501 return 0;
502}

545/* Handle `priv mtu <bytes>|restore`. Caller holds g_lock. */
546static int
547cmd_mtu(char *args, char *reply, size_t rsz)
548{
549 if (g_iface[0] == '\0') {
550 snprintf(reply, rsz, "err: no --priv-iface configured for mtu\n");
551 return -1;
552 }
553 char *tok = strtok(args, " ");
554 if (!tok) {
555 snprintf(reply, rsz, "err: mtu needs <bytes> or 'restore'\n");
556 return -1;
557 }
558 if (strcmp(tok, "restore") == 0) {
559 if (g_mtu_saved <= 0) {
560 snprintf(reply, rsz, "ok (nothing to restore)\n");
561 return 0;
562 }
563 int rc = mtu_set(g_mtu_saved);
564 g_mtu_saved = -1;
565 if (rc != 0) {
566 snprintf(reply, rsz, "err: ip link set mtu failed\n");
567 }
568 return rc;
569 }
570 char *end;
571 long b = strtol(tok, &end, 10);
572 if (end == tok || b < 68 || b > 65535) { /* IPv4 minimum link MTU is 68 */
573 snprintf(reply, rsz, "err: mtu out of range (68..65535)\n");
574 return -1;
575 }
576 if (g_mtu_saved < 0) {
577 g_mtu_saved = mtu_current(); /* remember the real MTU once */
578 }
579 if (mtu_set((int) b) != 0) {
580 snprintf(reply, rsz, "err: ip link set mtu failed\n");
581 return -1;
582 }
583 return 0;
584}

588int
589fp_priv_enable(const char *iface, int listen_port,
590 const int *target_ports, int nports, const char **err)
591{
592 if (geteuid() != 0) {
593 *err = "privileged levers require root (euid 0)";
594 return -1;
595 }
596 if (iface && iface[0] && !valid_iface(iface)) {
597 *err = "invalid or nonexistent --priv-iface";
598 return -1;
599 }
600 pthread_mutex_lock(&g_lock);
601 memset(g_iface, 0, sizeof(g_iface));
602 if (iface) {
603 snprintf(g_iface, sizeof(g_iface), "%s", iface);
604 }
605 g_listen_port = listen_port;
606 g_ntports = 0;
607 for (int i = 0; i < nports && g_ntports < (int) (sizeof(g_tports) / sizeof(g_tports[0])); i++) {
608 g_tports[g_ntports++] = target_ports[i];
609 }
610 g_on = 1;
611 pthread_mutex_unlock(&g_lock);
612 (void) err;
613 return 0;
614}

662 cmd_mtu(rest, reply, rsz);

665 g_ne[i][0] = '\0';

668 cut_clear();

670 mtu_set(g_mtu_saved);
671 g_mtu_saved = -1;
672 }
673 } else if (strcmp(verb, "status") == 0) {
674 snprintf(reply, rsz, "priv on iface=%s nft=%s mtu_saved=%d\n",

676 g_mtu_saved);
677 } else {
