108/* ---- resilient.c — network resilience for the synchronous tools ----
109 *
110 * Brings xrootdfs-style recovery (reconnect + full re-auth + handle reopen +
111 * offset resume + bounded backoff) to one-shot CLI flows, lifted from the proven
112 * xrdcp pump (copy.c) and the async mfile layer (aio_mgr.c). Two seams:
113 * - brix_with_resilience(): wrap any stateless op (stat/ls/query/...) so it is
114 * re-issued after a sever, gated by an idempotency class.
115 * - brix_rfile: a synchronous file handle that reopens + resumes mid-transfer.
116 * Both are no-ops (single attempt) when the window is 0, so --no-retry restores
117 * the exact legacy fail-fast path. Raw ops (and copy.c) are untouched. */
118
119/* Idempotency class for brix_with_resilience — governs re-issue after a sever. */
120typedef enum {
121 XRDC_OP_READONLY, /* stat/ls/locate/query/statvfs: retry freely */
122 XRDC_OP_IDEMPOTENT, /* chmod: re-apply is harmless — retry freely */
123 XRDC_OP_MUTATION_NORMALIZE, /* mkdir/rm/rmdir/mv/prepare: re-issue ONCE, then
124 * treat benign_errno (EEXIST/ENOENT) as success */
125 XRDC_OP_UNSAFE /* never auto-retry */
126} brix_op_class;
