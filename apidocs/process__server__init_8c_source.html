1/*
2 * process_server_init.c — the per-server worker-init ladder.
3 *
4 * WHAT: Owns brix_init_one_server() — every per-server init step for one
5 * enabled brix server block, in the frozen worker-init order (credential
6 * replay, log-fd capture, staging registry, export rootfd, checkpoint
7 * recovery, cache storage, XrdAcc, CMS, health-check, Pelican advertise,
8 * the two cache reaper timers, CRL/JWKS) — together with the file-local
9 * per-step helpers it delegates to.
10 * WHY: Split (phase-79 file-size cap) out of the former 925-line process.c.
11 * The per-server ladder is a self-contained concern: keeping it in one
12 * focused file leaves the init_process orchestrator (process.c) a short
13 * loop while preserving the EXACT initialization order the
14 * reload-semantics contract freezes. Only brix_init_one_server crosses
15 * the file boundary (declared in process_internal.h); every step helper
16 * stays file-local.
17 * HOW: brix_init_one_server runs the step helpers as a flat early-return
18 * sequence; a step returning NGX_ERROR aborts worker startup. The two
19 * reaper-timer arming helpers point their events at the callbacks defined
20 * in process_timers.c (brix_cache_reap_handler and, from
21 * reap_watermark.h, brix_cache_watermark_timer_handler). No behaviour
22 * change from the split.
23 */

39/* ---- Replay the remote-backend credential into this worker's VFS registry ----
40 *
41 * WHAT: Re-applies the server's configured storage credential to the
42 * process-global VFS backend registry entry for its export root. No return
43 * value — a missing credential/backend is a legitimate no-op.
44 *
45 * WHY: brix_vfs_backend_set_credential runs only at config parse, in the
46 * throwaway config-load process; the serving master/workers do not inherit
47 * that entry's credential, so their registry has the backend but an EMPTY
48 * credential and the origin login fails with "no credential set". The srv
49 * conf's fields ARE inherited reliably (nginx config), so replay them here,
50 * per worker, before any request resolves + builds the backend instance.
51 *
52 * HOW:
53 * 1. Skip unless both a storage credential name and an export root are set.
54 * 2. NUL-terminate the credential name and look it up in the credential
55 * block table; a miss is loud (error log) — a worker serving with a
56 * wiped credential must never be silent.
57 * 3. Map the credential through brix_credential_to_backend_cred() — the
58 * ONE shared mapper (P80.1). Its earlier hand-copied twin here dropped
59 * bearer + all three s3 fields, and because set_credential overwrites
60 * all 8 registry slots unconditionally, every worker spawn wiped the
61 * parse-time S3 keys to "" (phase-80 finding 1.1).
62 * 4. Install it on the export root's registry entry; set_credential also
63 * resets e->inst so the instance rebuilds with the credential on first
64 * use.
65 */
66static void
67brix_init_server_backend_credential(ngx_cycle_t *cycle,
68 ngx_stream_brix_srv_conf_t *xcf)
69{
70 char credz[256];
71 char bearer[4096];
72 const brix_credential_t *cred;
73 brix_vfs_backend_cred_t bcred;
74
75 if (xcf->common.storage_credential.len == 0
76 || xcf->common.root_canon[0] == '\0')
77 {
78 return;
79 }
80
81 ngx_cpystrn((u_char *) credz, xcf->common.storage_credential.data,
82 ngx_min(xcf->common.storage_credential.len + 1,
83 sizeof(credz)));
84 cred = brix_credential_lookup(credz);
85 if (cred == NULL) {
86 ngx_log_error(NGX_LOG_ERR, cycle->log, 0,
87 "brix: worker credential replay: no brix_credential \"%s\" "
88 "for export \"%s\" — backend keeps its (empty) worker-side "
89 "credential; upstream auth WILL fail", credz,
90 xcf->common.root_canon);
91 return;
92 }
93
94 if (brix_credential_to_backend_cred(cred, bearer, sizeof(bearer),
95 &bcred, cycle->log) != NGX_OK)
96 {
97 ngx_log_error(NGX_LOG_ERR, cycle->log, 0,
98 "brix: worker credential replay: cannot derive bearer for "
99 "brix_credential \"%s\" (export \"%s\") — credential NOT "
100 "installed; upstream auth WILL fail", credz,
101 xcf->common.root_canon);
102 return;
103 }
104 brix_vfs_backend_set_credential(xcf->common.root_canon, &bcred);
105}

107/* ---- Open this worker's fds onto the staging registry + resume sweep ----
108 *
109 * WHAT: Initialises the composable stage request registry from the server's
110 * (tape) control dir and registers the upload-resume TTL sweep for its stage
111 * dir. No return value — both are best-effort, idempotent no-ops when the
112 * respective dir is not configured.
113 *
114 * WHY: The migrated staging callers (prepare/tape_rest/open) record into the
115 * registry, which needs per-worker fds; abandoned upload-resume partials in
116 * the stage dir would otherwise leak disk. The legacy FRM
117 * queue/scheduler/purge worker-init is retired — the recall is driven by the
118 * sd_frm backend + the client poll model.
119 *
120 * HOW:
121 * 1. When FRM is enabled with a control dir, NUL-terminate the dir for the
122 * C API and init the stage registry (journal dir = the control dir).
123 * 2. When an upload stage dir is canonicalised, register the resume sweep
124 * (worker-0 only internally; the register itself is idempotent and arms
125 * a single timer for the first stage-dir server).
126 */
127static void
128brix_init_server_stage_registry(ngx_cycle_t *cycle,
129 ngx_stream_brix_srv_conf_t *xcf)
130{
131 if (xcf->frm.enable && xcf->frm.control_dir.len > 0) {
132 char _jd[NGX_MAX_PATH];
133 if (brix_str_cbuf(_jd, sizeof(_jd), &xcf->frm.control_dir) != NULL) {
134 (void) brix_stage_registry_init(_jd, cycle->log);
135 }
136 }
137
138 /* Phase 6 housekeeping: TTL-sweep abandoned upload-resume partials from
139 * the stage dir (worker-0 only; the register itself is idempotent and
140 * arms a single timer for the first stage-dir server). */
141 if (xcf->upload_stage_dir_canon[0] != '\0') {
142 brix_xfer_resume_sweep_register(cycle,
143 xcf->upload_stage_dir_canon);
144 }
145}

147/* ---- Open the confined export-root fd for a data server ----
148 *
149 * WHAT: Opens the server's export root as an O_PATH directory fd for
150 * kernel-confined path operations. Returns NGX_OK on success or when no
151 * local export root is configured; NGX_ERROR (with an EMERG log) when the
152 * root cannot be opened.
153 *
154 * WHY: All confined path resolution (openat2 RESOLVE_IN_ROOT / O_NOFOLLOW
155 * walks) anchors on this fd; a data server that cannot open its export root
156 * must refuse to run rather than serve unconfined.
157 *
158 * HOW:
159 * 1. Skip servers with an empty root_canon (proxy/manager/supervisor).
160 * 2. open(root, O_PATH|O_DIRECTORY|O_CLOEXEC) into xcf->rootfd.
161 * 3. On failure log EMERG with errno and return NGX_ERROR.
162 */
163static ngx_int_t
164brix_init_server_rootfd(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *xcf)
165{
166 /* Only open rootfd for data servers with a local export root.
167 * Proxy/manager/supervisor servers leave root_canon empty. */
168 if (xcf->common.root_canon[0] == '\0') {
169 return NGX_OK;
170 }
171
172 xcf->rootfd = open(xcf->common.root_canon,
173 O_PATH | O_DIRECTORY | O_CLOEXEC);
174 if (xcf->rootfd < 0) {
175 ngx_log_error(NGX_LOG_EMERG, cycle->log, errno,
176 "brix: cannot open export root \"%s\" for "
177 "kernel-confined path operations",
178 xcf->common.root_canon);
179 return NGX_ERROR;
180 }
181
182 return NGX_OK;
183}

185/* ---- Arm the per-server stale-dirty cache reaper timer ----
186 *
187 * WHAT: Allocates and arms the hourly stale-dirty reaper timer for a server
188 * whose cache-state engine is active. Returns NGX_OK (including the
189 * not-configured no-op) or NGX_ERROR on allocation failure.
190 *
191 * WHY: The eviction guard protects dirty write-back staging files, so an
192 * abandoned flush would leak disk forever without a reaper that runs on a
193 * maintenance timer independent of occupancy.
194 *
195 * HOW:
196 * 1. Skip unless brix_cache_dirty_max_age > 0 and a state root resolves.
197 * 2. pcalloc the timer event from the cycle pool (NGX_ERROR on failure).
198 * 3. Point it at brix_cache_reap_handler with the srv conf as ev->data,
199 * mark cancelable, arm at BRIX_CACHE_REAP_FIRST_MS.
200 */
201static ngx_int_t
202brix_init_server_cache_reap_timer(ngx_cycle_t *cycle,
203 ngx_stream_brix_srv_conf_t *xcf)
204{
205 if (xcf->cache_dirty_max_age <= 0
206 || brix_cache_state_root(xcf) == NULL)
207 {
208 return NGX_OK;
209 }
210
211 xcf->cache_reap_timer = ngx_pcalloc(cycle->pool, sizeof(ngx_event_t));
212 if (xcf->cache_reap_timer == NULL) {
213 return NGX_ERROR;
214 }
215 xcf->cache_reap_timer->handler = brix_cache_reap_handler;
216 xcf->cache_reap_timer->data = xcf;
217 xcf->cache_reap_timer->log = cycle->log;
218 xcf->cache_reap_timer->cancelable = 1; /* don't delay graceful shutdown */
219 ngx_add_timer(xcf->cache_reap_timer, BRIX_CACHE_REAP_FIRST_MS);
220
221 return NGX_OK;
222}

224/* ---- Arm the per-server watermark-driven LRU reaper timer ----
225 *
226 * WHAT: Allocates and arms the proactive watermark LRU reaper timer for a
227 * server with a cache and a valid HIGH watermark. Returns NGX_OK (including
228 * the not-configured no-op) or NGX_ERROR on allocation failure.
229 *
230 * WHY: Waiting for occupancy-triggered eviction alone lets a busy cache
231 * overshoot its watermark; a periodic proactive reap keeps it bounded. A
232 * small per-worker jitter on the first tick keeps the workers from all
233 * firing together.
234 *
235 * HOW:
236 * 1. Skip unless a cache is configured, a state root resolves, and
237 * 0 < high_watermark < 1000000.
238 * 2. pcalloc the timer event from the cycle pool (NGX_ERROR on failure).
239 * 3. Point it at brix_cache_watermark_timer_handler, mark cancelable, arm
240 * at BRIX_CACHE_REAP_FIRST_MS + (pid % 1000) jitter.
241 */
242static ngx_int_t
243brix_init_server_watermark_timer(ngx_cycle_t *cycle,
244 ngx_stream_brix_srv_conf_t *xcf)
245{
246 if ((!xcf->cache && xcf->common.cache_store.len == 0)
247 || brix_cache_state_root(xcf) == NULL
248 || xcf->reaper.high_watermark <= 0
249 || xcf->reaper.high_watermark >= 1000000)
250 {
251 return NGX_OK;
252 }
253
254 xcf->reaper.timer =
255 ngx_pcalloc(cycle->pool, sizeof(ngx_event_t));
256 if (xcf->reaper.timer == NULL) {
257 return NGX_ERROR;
258 }
259 xcf->reaper.timer->handler =
260 brix_cache_watermark_timer_handler;
261 xcf->reaper.timer->data = xcf;
262 xcf->reaper.timer->log = cycle->log;
263 xcf->reaper.timer->cancelable = 1; /* don't delay graceful shutdown */
264 ngx_add_timer(xcf->reaper.timer,
265 BRIX_CACHE_REAP_FIRST_MS
266 + (ngx_msec_t) (ngx_pid % 1000));
267
268 return NGX_OK;
269}

271/* ---- Arm the per-server CRL reload timer + schedule JWKS refresh ----
272 *
273 * WHAT: Starts the CRL hot-reload timer for GSI servers that configured
274 * brix_crl_reload, and schedules the JWKS refresh in every case. Returns
275 * NGX_OK, or NGX_ERROR when the timer event cannot be allocated.
276 *
277 * WHY: Timers are per-worker because each nginx worker process has its own
278 * event loop and its own copy of the config pointers (but the X509_STORE*
279 * is shared within a worker). JWKS refresh is orthogonal to CRLs and must be
280 * scheduled whether or not a CRL timer is armed.
281 *
282 * HOW:
283 * 1. When the server is not GSI/BOTH, or has no CRL path/interval: just
284 * schedule the JWKS refresh and return NGX_OK.
285 * 2. Otherwise pcalloc the CRL timer event (NGX_ERROR on failure), point
286 * it at brix_crl_reload_handler, mark cancelable, arm at the configured
287 * interval and log a NOTICE.
288 * 3. Schedule the JWKS refresh for the GSI/BOTH server too.
289 */
290static ngx_int_t
291brix_init_server_crl_jwks(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *xcf)
292{
293 if ((xcf->auth != BRIX_AUTH_GSI && xcf->auth != BRIX_AUTH_BOTH)
294 || xcf->crl.len == 0 || xcf->crl_reload == 0)
295 {
296 /* CRL timer not needed — but still check for JWKS refresh */
297 brix_token_jwks_schedule_refresh(cycle, xcf);
298 return NGX_OK;
299 }
300
301 /* Allocate and start the CRL reload timer */
302 xcf->crl_timer = ngx_pcalloc(cycle->pool, sizeof(ngx_event_t));
303 if (xcf->crl_timer == NULL) {
304 return NGX_ERROR;
305 }
306 xcf->crl_timer->handler = brix_crl_reload_handler;
307 xcf->crl_timer->data = xcf;
308 xcf->crl_timer->log = cycle->log;
309 xcf->crl_timer->cancelable = 1; /* don't delay graceful shutdown */
310
311 ngx_add_timer(xcf->crl_timer, (ngx_msec_t) xcf->crl_reload * 1000);
312
313 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
314 "brix: CRL reload timer started - interval=%ds "
315 "path=\"%s\"",
316 (int) xcf->crl_reload, xcf->crl.data);
317
318 /* Also check for JWKS refresh on GSI/BOTH servers */
319 brix_token_jwks_schedule_refresh(cycle, xcf);
320
321 return NGX_OK;
322}

324/* ---- Per-server worker initialization for one enabled brix server block ----
325 *
326 * WHAT: Runs every per-server init step for one enabled server conf, in the
327 * frozen worker-init order: credential replay, log-fd capture, staging
328 * registry, export rootfd, checkpoint recovery, cache storage, XrdAcc, CMS,
329 * health-check, Pelican advertise, cache reaper timers, CRL/JWKS. Returns
330 * NGX_OK, or NGX_ERROR to abort worker startup.
331 *
332 * WHY: Worker init is a flat sequence of independent subsystem
333 * initializations; keeping each server's ladder in one orchestrator keeps
334 * init_process itself a short loop while preserving the exact initialization
335 * order the reload-semantics contract freezes.
336 *
337 * HOW:
338 * 1. Replay the storage credential into the VFS backend registry.
339 * 2. Capture the nginx-managed access/audit log fds.
340 * 3. Init the staging registry + upload-resume sweep.
341 * 4. Open the confined export rootfd (fatal on failure).
342 * 5. Recover interrupted checkpoints, build cache SD storage instances,
343 * build XrdAcc tables — each fatal on failure.
344 * 6. Start CMS handlers (when cms_addr set), health-check timer, Pelican
345 * cache advertisement — all internal no-ops when unconfigured.
346 * 7. Arm the stale-dirty and watermark cache reaper timers (fatal on
347 * allocation failure).
348 * 8. Arm the CRL reload timer / schedule JWKS refresh.
349 */
350ngx_int_t
351brix_init_one_server(ngx_cycle_t *cycle, ngx_stream_brix_srv_conf_t *xcf)
352{
353 brix_init_server_backend_credential(cycle, xcf);
354
355 /*
356 * Capture the nginx-managed log fds the master opened during
357 * ngx_init_cycle. We read file->fd here (not at config time, where it
358 * is still NGX_INVALID_FILE) so the hot access-/audit-log write paths
359 * keep using a plain int fd. nginx dup2()s a reopened log onto the same
360 * fd number on USR1, so this captured value stays valid for the worker's
361 * lifetime. A NULL handle means logging is disabled for this server.
362 */
363 xcf->access_log_fd = xcf->access_log_file != NULL
364 ? xcf->access_log_file->fd : NGX_INVALID_FILE;
365 xcf->proxy.audit_log_fd = xcf->proxy.audit_log_file != NULL
366 ? xcf->proxy.audit_log_file->fd : NGX_INVALID_FILE;
367
368 brix_init_server_stage_registry(cycle, xcf);
369
370 if (brix_init_server_rootfd(cycle, xcf) != NGX_OK) {
371 return NGX_ERROR;
372 }
373
374 if (brix_chkpoint_recover_root(cycle->log, xcf->common.root_canon)
375 != NGX_OK)
376 {
377 return NGX_ERROR;
378 }
379
380 /* The per-export storage backend (e.g. pblock) is registered at config
381 * time and built per worker lazily by the VFS backend registry on first
382 * use (brix_vfs_ctx_init → brix_vfs_backend_resolve); no per-server
383 * init_process creation is needed here. */
384
385 /* The cache performs all disk I/O through SD storage instances (POSIX
386 * driver on a per-worker rootfd by default, or a configured backend) —
387 * build them now. No-op unless a cache is configured. */
388 if (brix_cache_storage_init(xcf, cycle) != NGX_OK) {
389 return NGX_ERROR;
390 }
391
392 /* Build the per-worker XrdAcc tables + hot-reload timer (no-op unless
393 * this server uses `brix_authdb_format xrdacc`). */
394 if (brix_acc_init_server(xcf, cycle) != NGX_OK) {
395 return NGX_ERROR;
396 }
397
398 if (xcf->cms.addr != NULL) {
399 ngx_brix_cms_start(cycle, xcf);
400 }
401
402 /* Phase 22: start the active health-check timer (no-op if disabled). */
403 brix_hc_manager_start(cycle, xcf);
404
405 /* Pelican: start the cache advertisement timer (no-op unless
406 * brix_cache_advertise is on with a key + data-url configured). */
407 brix_cache_pelican_schedule_advertise(cycle, xcf);
408
409 /* Unified cache-state engine: arm the per-worker stale-dirty reaper when
410 * a state root resolves and a max age is set. Independent of occupancy. */
411 if (brix_init_server_cache_reap_timer(cycle, xcf) != NGX_OK) {
412 return NGX_ERROR;
413 }
414
415 /* Watermark-driven LRU reaper: arm the proactive per-worker timer when a
416 * cache is configured with a valid HIGH watermark. A small per-worker
417 * jitter on the first tick keeps the workers from all firing together. */
418 if (brix_init_server_watermark_timer(cycle, xcf) != NGX_OK) {
419 return NGX_ERROR;
420 }
421
422 return brix_init_server_crl_jwks(cycle, xcf);
423}
