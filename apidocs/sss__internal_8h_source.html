49/*
50 * brix_sss_verify_blob — validate a self-contained SSS credential against a
51 * keytab, transport-independent (used by both the XRootD kXR_auth path and the
52 * CMS kYR_xauth handshake).
53 *
54 * Performs the full verification chain: outer "sss\0"+BF32 header, key-id
55 * lookup, Blowfish-CFB64 decrypt, CRC32 wrong-key detection, timestamp replay
56 * window (lifetime seconds), and identity TLV parse. Only the self-contained
57 * USEDATA credential form is accepted; the interactive SNDLID (server-supplies
58 * login-id) form returns NGX_DECLINED so the caller can reject it explicitly.
59 *
60 * Returns NGX_OK with *id_out filled and *key_out set to the matched key on
61 * success; NGX_DECLINED for an unsupported SNDLID credential; NGX_ERROR for any
62 * malformed/forged/expired credential (err filled). Does not allocate.
63 */
64ngx_int_t brix_sss_verify_blob(ngx_array_t *keys, time_t lifetime,
65 const u_char *blob, size_t blob_len,
66 brix_sss_identity_t *id_out, const brix_sss_key_t **key_out,
67 char *err, size_t errsz);
