1/*
2 * sec_token.c — WLCG bearer token (ztn) auth module.
3 *
4 * WHAT: Discover a JWT from the environment and send it as the kXR_auth payload.
5 * WHY: Token auth is the simplest XRootD credential — a single round, no crypto
6 * (TLS provides confidentiality where required).
7 * HOW: Discovery order: $BEARER_TOKEN, $BEARER_TOKEN_FILE,
8 * $XDG_RUNTIME_DIR/bt_u<uid>, /tmp/bt_u<uid>. Payload = "ztn\0" + JWT
9 * (the server skips the 4-byte tag and strips trailing whitespace/NULs).
10 *
11 * wire: XProtocol.hh kXR_auth — credtype "ztn", payload repeats "ztn\0" then JWT.
12 */

21static void
22rstrip(char *s)
23{
24 size_t n = strlen(s);
25 while (n > 0 && (s[n - 1] == '\n' || s[n - 1] == '\r'
26 || s[n - 1] == ' ' || s[n - 1] == '\t')) {
27 s[--n] = '\0';
28 }
29}

31static char *
32slurp(const char *path)
33{
34 FILE *fp;
35 char *buf;
36 long sz;
37 size_t got;
38 int fd;
39
40 /* Discovery probe (st=NULL): a missing OR unsafe file is silently skipped so
41 * the caller falls through to the next location. O_NOFOLLOW + owner check
42 * stop an attacker pre-planting /tmp/bt_u<uid> as a symlink (secret leak) or
43 * a regular file they own (confused-deputy auth). */
44 fd = brix_open_credfile(path, 0, NULL);
45 if (fd < 0) {
46 return NULL;
47 }
48 fp = fdopen(fd, "rb");
49 if (fp == NULL) {
50 close(fd);
51 return NULL;
52 }
53 if (fseek(fp, 0, SEEK_END) != 0 || (sz = ftell(fp)) < 0) {
54 fclose(fp);
55 return NULL;
56 }
57 rewind(fp);
58 if (sz > (1 << 20)) { /* a JWT is never a megabyte */
59 fclose(fp);
60 return NULL;
61 }
62 buf = (char *) malloc((size_t) sz + 1);
63 if (buf == NULL) {
64 fclose(fp);
65 return NULL;
66 }
67 got = fread(buf, 1, (size_t) sz, fp);
68 fclose(fp);
69 buf[got] = '\0';
70 rstrip(buf);
71 if (buf[0] == '\0') {
72 free(buf);
73 return NULL;
74 }
75 return buf;
76}

78/* Return a malloc'd token string, or NULL if none found. Shared with credinfo.c
79 * (the explain/diagnostics path) so token discovery lives in one place. */
80char *
81brix_token_discover(void)
82{
83 const char *env;
84 char path[256];
85 uid_t uid = geteuid();
86
87 env = getenv("BEARER_TOKEN");
88 if (env != NULL && env[0] != '\0') {
89 char *t = strdup(env);
90 if (t != NULL) {
91 rstrip(t);
92 }
93 return t;
94 }
95
96 env = getenv("BEARER_TOKEN_FILE");
97 if (env != NULL && env[0] != '\0') {
98 return slurp(env);
99 }
100
101 env = getenv("XDG_RUNTIME_DIR");
102 if (env != NULL && env[0] != '\0') {
103 char *t;
104 snprintf(path, sizeof(path), "%s/bt_u%u", env, (unsigned) uid);
105 t = slurp(path);
106 if (t != NULL) {
107 return t;
108 }
109 }
110
111 snprintf(path, sizeof(path), "/tmp/bt_u%u", (unsigned) uid);
112 return slurp(path);
113}

115static int
116token_have(brix_conn *c)
117{
118 /* Store is a fast "yes"; a store miss falls through to env discovery so a
119 * tool whose store lacks the bearer handler still finds a $BEARER_TOKEN. */
120 if (c != NULL && c->opts.cred != NULL
121 && brix_cred_available(c->opts.cred, XRDC_CRED_BEARER)) {
122 return 1;
123 }
124 char *t = brix_token_discover();
125 if (t == NULL) {
126 return 0;
127 }
128 free(t);
129 return 1;
130}

132static int
133token_first(brix_conn *c, const char *parms, uint8_t **payload, uint32_t *plen,
134 brix_status *st)
135{
136 char *tok = NULL;
137 size_t tl;
138 uint8_t *p;
139
140 (void) parms;
141
142 /* Try the credential store when present; fall back to env discovery on
143 * failure so env-sourced tokens behave identically to today. */
144 if (c != NULL && c->opts.cred != NULL) {
145 brix_cred_view v;
146 if (brix_cred_acquire(c->opts.cred, XRDC_CRED_BEARER, 0, &v, st) == 0
147 && v.token != NULL) {
148 tok = strdup(v.token);
149 } else {
150 brix_status_clear(st);
151 }
152 }
153 if (tok == NULL) {
154 tok = brix_token_discover();
155 }
156 if (tok == NULL) {
157 brix_status_set(st, XRDC_EAUTH, 0,
158 "no bearer token (set BEARER_TOKEN or BEARER_TOKEN_FILE)");
159 return -1;
160 }
161 tl = strlen(tok);
162 p = (uint8_t *) malloc(4 + tl);
163 if (p == NULL) {
164 free(tok);
165 brix_status_set(st, XRDC_EAUTH, 0, "out of memory");
166 return -1;
167 }
168 memcpy(p, "ztn\0", 4);
169 memcpy(p + 4, tok, tl);
170 free(tok);
171
172 *payload = p;
173 *plen = (uint32_t) (4 + tl);
174 return 0;
175}
