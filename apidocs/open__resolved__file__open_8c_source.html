28/*
29 * open_resolved_file_open.c — the low-level kXR_open file-open operations
30 * (phase-79 split from open_resolved_file.c).
31 *
32 * WHAT: The actual open of the resolved file — the driver-backed (Layer 3)
33 * open that synthesizes a struct stat from the storage driver, the
34 * POSIX-fd open of the staging temp / cache / export-final path, and the
35 * errno→kXR error mapper shared by both.
36 *
37 * WHY: This is the hot syscall core of the open path; isolating it (and its
38 * small POSIX-flag/logical-path helpers) keeps the exact syscall order,
39 * O_NONBLOCK guard, and errno→kXR mapping reviewable in one place, apart
40 * from the backend-routing policy that selects which open to run.
41 *
42 * HOW: brix_open_resolved_via_driver, brix_open_posix_dispatch and
43 * brix_open_map_open_error are the cross-file entry points (called by the
44 * dispatch stage); brix_open_logical/brix_open_oflags_to_sd are file-local
45 * helpers. Behaviour is byte-identical to the original.
46 */

52/* The export-root-relative ("logical") form of an absolute path confined under
53 * `root`: strips the root prefix + leading '/'. Returns the suffix, or the path
54 * unchanged when it is not under root (then the VFS open/probe will reject it).
55 * Centralises the rel-strip the kXR_open path repeats for the export-root open
56 * (the cache/stage domains open as the worker and need no rel form). */
57static const char *
58brix_open_logical(const char *abs, const char *root)
59{
60 size_t root_len = (root != NULL) ? strlen(root) : 0;
61
62 if (root_len > 0
63 && ngx_strncmp((u_char *) abs, (u_char *) root, root_len) == 0
64 && abs[root_len] == '/')
65 {
66 return abs + root_len + 1;
67 }
68 return abs;
69}

71/* Map the POSIX open(2) flags the kXR_open path computed back to the backend-
72 * neutral BRIX_SD_O_* intent the storage driver understands. The driver
73 * re-derives its own native flags from these (the POSIX driver re-expands to
74 * O_*), so a non-POSIX backend never sees Linux-specific bits. */
75static int
76brix_open_oflags_to_sd(int oflags, int is_readable, int is_write)
77{
78 int sd = 0;
79
80 if (is_readable) { sd |= BRIX_SD_O_READ; }
81 if (is_write) { sd |= BRIX_SD_O_WRITE; }
82 if (oflags & O_CREAT) { sd |= BRIX_SD_O_CREATE; }
83 if (oflags & O_EXCL) { sd |= BRIX_SD_O_EXCL; }
84 if (oflags & O_TRUNC) { sd |= BRIX_SD_O_TRUNC; }
85 if (oflags & O_APPEND) { sd |= BRIX_SD_O_APPEND; }
86 return sd;
87}

89/* Driver-backed kXR_open (Layer 3): open `logical` through the export's storage
90 * driver into the handle's sd_obj, then synthesize a struct stat from the
91 * driver's captured open snapshot so the rest of the open path (bookkeeping,
92 * size reporting) is backend-agnostic. The handle's bare `fd` becomes the
93 * driver's representative descriptor (a block-0 fd for CAP_FD backends, or
94 * NGX_INVALID_FILE for a pure object store) and all subsequent byte I/O routes
95 * through fh->sd_obj.driver. Writes *out_fd and *st on success and returns
96 * NGX_OK; on failure sets errno and returns NGX_ERROR (the caller maps errno to
97 * the kXR error exactly as for a POSIX open).
98 *
99 * WHAT: `vctx` carries the per-user backend credential policy (Phase 2 Task 6)
100 * bound by the caller via brix_vfs_ctx_bind_backend_cred(); when the
101 * policy resolves a user credential this open presents it to the
102 * backend driver via brix_sd_open_maybe_cred instead of the static
103 * service credential.
104 *
105 * WHY: A root:// session already carries the authenticated brix_identity_t
106 * on brix_ctx_t; without this gate every remote-backed root:// open
107 * (davs/S3 already had this via brix_vfs_open) silently used the
108 * shared service credential for every user, indistinguishable in the
109 * origin's own auth/session log.
110 *
111 * HOW: Runs brix_vfs_backend_cred() BEFORE the driver open (deny mode must
112 * refuse before any origin connection is attempted, exactly as the
113 * VFS's own brix_vfs_open does); on NGX_ERROR propagates errno (EACCES)
114 * without touching the driver. */
115ngx_int_t
116brix_open_resolved_via_driver(brix_open_args_t *a, brix_vfs_ctx_t *vctx,
117 brix_sd_instance_t *sd, const char *logical)
118{
119 brix_file_t *fh = &a->ctx->files[a->idx];
120 struct stat *st = a->st;
121 int sd_flags = brix_open_oflags_to_sd(a->oflags,
122 a->is_readable,
123 a->is_write);
124 int oerr = 0;
125 brix_sd_obj_t *obj;
126 brix_sd_ucred_t ustore;
127 brix_sd_cred_t ucred;
128 int use_cred = 0;
129
130 ngx_memzero(&ucred, sizeof(ucred));
131 if (brix_vfs_backend_cred(vctx, &ustore, &ucred, &use_cred, &oerr)
132 != NGX_OK)
133 {
134 errno = (oerr != 0) ? oerr : EACCES;
135 return NGX_ERROR;
136 }
137
138 obj = brix_sd_open_maybe_cred(sd, logical, sd_flags, a->create_mode,
139 use_cred ? &ucred : NULL, &oerr);
140 if (obj == NULL) {
141 errno = (oerr != 0) ? oerr : EIO;
142 return NGX_ERROR;
143 }
144
145 /* Adopt the object by value into the handle. A driver that malloc'd the obj
146 * shell (heap_shell) hands ownership of the COPY to us; free the now-
147 * redundant shell. The embedded copy is not itself a heap shell. */
148 fh->sd_obj = *obj;
149 if (obj->heap_shell) {
150 free(obj);
151 }
152 fh->sd_obj.heap_shell = 0;
153
154 /* The driver's open may defer metadata (the POSIX driver deliberately skips
155 * the fstat at open — see sd_posix_open). Populate the snapshot now via the
156 * driver's own fstat so the synthesized `struct stat` (and hence the handle's
157 * cached_size) is correct: a driver-aware fstat reports the LOGICAL object
158 * size (e.g. pblock's whole-object size, not block 0). Without this the
159 * cached_size stays 0 and the buffered read path — inline read compression,
160 * and any non-sendfile serve — sees EOF immediately and returns nothing. */
161 if (sd->driver->fstat != NULL && fh->sd_obj.snap.size == 0) {
162 (void) sd->driver->fstat(&fh->sd_obj, &fh->sd_obj.snap);
163 }
164
165 ngx_memzero(st, sizeof(*st));
166 st->st_size = fh->sd_obj.snap.size;
167 st->st_mtime = fh->sd_obj.snap.mtime;
168 st->st_ctime = fh->sd_obj.snap.ctime;
169 st->st_ino = fh->sd_obj.snap.ino;
170 st->st_mode = (fh->sd_obj.snap.mode != 0)
171 ? fh->sd_obj.snap.mode
172 : (fh->sd_obj.snap.is_dir ? (S_IFDIR | 0755)
173 : (S_IFREG | 0644));
174
175 /* The driver snapshot carries no device id, so st_dev would stay 0 and the
176 * published-handle table would record device 0. A bound secondary reopens
177 * the file for itself (a real fstat, real st_dev) and revalidates device+
178 * inode against the published entry — a 0 vs real-device mismatch would then
179 * wrongly revoke every bound read (kXR_error). For a driver with a real
180 * backing descriptor (the POSIX driver's fd is the file itself), capture the
181 * real device here so the published identity matches the secondary's reopen. */
182 if (fh->sd_obj.fd >= 0) {
183 struct stat rst;
184 if (fstat(fh->sd_obj.fd, &rst) == 0) {
185 st->st_dev = rst.st_dev;
186
187 /* Anti-wedge parity with the non-driver path's S_ISREG gate: a FIFO,
188 * socket, or device is not a servable byte stream. The confined open
189 * forced O_NONBLOCK so the open(2) itself could not park the worker,
190 * but a subsequent read/write would spin on EAGAIN — refuse it here,
191 * where the real st_mode is available (the driver snapshot may not
192 * carry a mode). A directory keeps its own handling upstream; only
193 * special files are cut. */
194 if (!S_ISREG(rst.st_mode) && !S_ISDIR(rst.st_mode)) {
195 (void) sd->driver->close(&fh->sd_obj);
196 fh->sd_obj.driver = NULL;
197 fh->sd_obj.inst = NULL;
198 fh->sd_obj.state = NULL;
199 fh->sd_obj.fd = -1;
200 errno = EINVAL;
201 return NGX_ERROR;
202 }
203 }
204 }
205
206 a->fd = fh->sd_obj.fd; /* block-0 fd, or NGX_INVALID_FILE (-1) */
207 return NGX_OK;
208}

210/* Map a failed open(2)'s errno to the kXR error response the reference raises,
211 * sending it and returning that send's rc; EAGAIN on a read is a nearline recall
212 * (kXR_wait retry) rather than an error. Reached only when open_failed is set, so
213 * it always sends exactly one reply. Returns the value the caller must return. */
214ngx_int_t
215brix_open_map_open_error(brix_ctx_t *ctx, ngx_connection_t *c,
216 const char *resolved, int err, ngx_flag_t is_write)
217{
218 const char *mode_str = is_write ? "wr" : "rd";
219
220 if (err == ENOENT || err == ENOTDIR) {
221 BRIX_RETURN_ERR(ctx, c,
222 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
223 "OPEN", resolved, mode_str,
224 kXR_NotFound, "file not found");
225 }
226 if (err == EEXIST) {
227 /* O_EXCL (kXR_new without kXR_delete) on an existing file → EEXIST,
228 * which the reference maps to kXR_ItExists (the code raised by the
229 * kXR_new flag), NOT kXR_FileLocked. */
230 BRIX_RETURN_ERR(ctx, c,
231 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
232 "OPEN", resolved, mode_str,
233 kXR_ItExists, "file already exists");
234 }
235 if (err == EACCES) {
236 BRIX_RETURN_ERR(ctx, c,
237 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
238 "OPEN", resolved, mode_str,
239 kXR_NotAuthorized, "permission denied");
240 }
241 if (err == EISDIR) {
242 BRIX_RETURN_ERR(ctx, c,
243 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
244 "OPEN", resolved, mode_str,
245 kXR_isDirectory, "is a directory");
246 }
247 if (err == EAGAIN && !is_write) {
248 /* A nearline (tape) recall is in flight (sd_cache/sd_frm, §9.2). Tell
249 * the client to retry with kXR_wait - the stream equivalent of the
250 * WebDAV 202 "staging": the open "parks" via client retry rather than
251 * blocking the worker for the MSS latency. A later re-open re-polls the
252 * recall and, once the object is online in the cache tier, opens +
253 * serves it. */
254 brix_log_access(ctx, c, "OPEN", resolved, "rd-recall-wait",
255 0, 0, "nearline recall in progress; retry", 0);
256 return brix_send_wait(ctx, c, BRIX_RECALL_WAIT_SECS);
257 }
258 BRIX_RETURN_ERR(ctx, c,
259 is_write ? BRIX_OP_OPEN_WR : BRIX_OP_OPEN_RD,
260 "OPEN", resolved, mode_str,
261 kXR_IOError, strerror(err));
262}

264/* The POSIX-fd open dispatch: open the staging temp / cache / export final path
265 * (never a driver-backed export — that is handled by the caller). Forces O_CREAT
266 * for a staged open and O_NONBLOCK so the open cannot park the worker; cache and
267 * external-stage domains open raw as the worker, the export final opens beneath
268 * the export rootfd through the VFS. Writes *out_fd and returns 1 when the open
269 * failed (errno set), else 0. */
270int
271brix_open_posix_dispatch(brix_open_args_t *a)
272{
273 ngx_stream_brix_srv_conf_t *conf = a->conf;
274 const char *resolved = a->resolved;
275 int oflags = a->oflags;
276 mode_t create_mode = a->create_mode;
277 ngx_flag_t stage = a->stage;
278 ngx_flag_t use_resume = a->use_resume;
279 ngx_flag_t from_cache = a->from_cache;
280
281 /* When POSC is active, open the staging temp path instead of the
282 * final path. The O_CREAT flag is forced so the temp file is
283 * always created fresh; O_EXCL is intentionally omitted so that a
284 * previous crash leaving a stale temp file does not block a retry. */
285 const char *open_path = stage ? a->posc_temp_path : resolved;
286 int fd;
287
288 /* Staged opens (POSC or resume) always need O_CREAT on the temp path.
289 * O_TRUNC is inherited from `oflags`: a fresh create/truncate open
290 * starts the partial empty; a resume re-open (kXR_open_updt, no trunc)
291 * preserves the already-written bytes. */
292 /* O_NONBLOCK guarantees the open(2) cannot park the worker in the
293 * kernel FIFO/device "wait_for_partner" rendezvous (a named pipe in the
294 * export would otherwise freeze the worker's event loop and stall every
295 * connection pinned to it). It is harmless for the regular files we
296 * serve and is cleared again on the surviving fd once fstat() confirms
297 * S_ISREG below. Mirrors the central guard in brix_vfs_open_fd_at(). */
298 int effective_oflags = oflags | (stage ? O_CREAT : 0) | O_NONBLOCK;
299
300 /* Resume staging on a configured fast device: the partial lives OUTSIDE
301 * root_canon, so it cannot go through the RESOLVE_BENEATH open. Its
302 * basename is a server-generated hash (no client-controlled component)
303 * inside the operator-trusted, canonicalized stage dir, so a direct open
304 * with O_NOFOLLOW on the final component is safe. */
305 ngx_flag_t stage_external = use_resume
306 && conf->upload_stage_dir_canon[0] != '\0';
307
308 if (stage_external) {
309 /* vfs-seam-allow: separate storage domain. The partial lives under
310 * the operator-trusted upload stage dir (a different root than the
311 * export, server-generated hash basename, svc-owned), so it is opened
312 * as the worker — NOT through the export-confined, impersonation-aware
313 * VFS (which would resolve under the export rootfd / mapped user).
314 * O_NOFOLLOW guards the final component. */
315 fd = open(open_path, effective_oflags | O_NOFOLLOW | O_CLOEXEC, /* vfs-seam-allow: separate svc-owned storage domain (cache/stage), opened as worker */
316 create_mode);
317 } else if (from_cache) {
318 /* vfs-seam-allow: separate storage domain. cache_root files are
319 * server-managed (filled by the cache worker, never client-written,
320 * svc-owned) in a different root than the export, so they are opened
321 * as the worker rather than through the export-confined VFS. O_NOFOLLOW
322 * is defence-in-depth; O_CLOEXEC prevents FD leak into a forked child. */
323 fd = open(open_path, effective_oflags | O_NOFOLLOW | O_CLOEXEC, /* vfs-seam-allow: separate svc-owned storage domain (cache/stage), opened as worker */
324 create_mode);
325 } else {
326 /* The export final/staged path: open beneath the export root through
327 * the VFS (openat2 RESOLVE_BENEATH, impersonation-aware). The VFS
328 * strips the absolute path to its rootfd-relative form. */
329 fd = brix_vfs_open_fd_at(conf->rootfd,
330 brix_open_logical(open_path, conf->common.root_canon),
331 effective_oflags, create_mode);
332 }
333 a->fd = fd;
334 return (fd < 0);
335}
