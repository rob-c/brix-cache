1/*
2 * wverify.c — self-computed write-integrity accumulator (see wverify.h).
3 *
4 * Offset-sorted extent list; each update CRC-32's its buffer and coalesces with
5 * any extent it is now adjacent to, combining CRCs with zlib crc32_combine. A
6 * complete write leaves exactly one extent {0, size, crc}; anything else (gap,
7 * overlap, degraded) makes brix_wverify_expected fail closed.
8 */

11#include <stdlib.h>

13#include <zlib.h>

15/* Cap the live extent count. An in-order stream stays at 1; an out-of-order
16 * writer (MODE E) holds at most one extent per not-yet-adjacent stream, far
17 * below this. Exceeding it marks the accumulator degraded (fails closed) rather
18 * than growing unbounded on a pathological arrival pattern. */
19#define BRIX_WVERIFY_MAX_EXTENTS 4096
20
21typedef struct {
22 off_t lo;
23 off_t hi; /* exclusive */
24 uint32_t crc; /* CRC-32 of bytes [lo, hi) */
25} wv_extent_t;
26
27struct brix_wverify_s {
28 wv_extent_t *ext;
29 size_t n;
30 size_t cap;
31 int degraded; /* an update failed; expected() must fail closed */
32};

51/* Ensure room for one more extent. 0 / -1 (cap or OOM). */
52static int
53wv_reserve(brix_wverify_t *w)
54{
55 size_t want;
56 wv_extent_t *grown;
57
58 if (w->n < w->cap) {
59 return 0;
60 }
61 if (w->n >= BRIX_WVERIFY_MAX_EXTENTS) {
62 return -1;
63 }
64 want = w->cap ? w->cap * 2 : 16;
65 if (want > BRIX_WVERIFY_MAX_EXTENTS) {
66 want = BRIX_WVERIFY_MAX_EXTENTS;
67 }
68 grown = realloc(w->ext, want * sizeof(*grown));
69 if (grown == NULL) {
70 return -1;
71 }
72 w->ext = grown;
73 w->cap = want;
74 return 0;
75}

77/* Merge extent i with i+1 when they abut (ext[i].hi == ext[i+1].lo). */
78static void
79wv_merge_at(brix_wverify_t *w, size_t i)
80{
81 wv_extent_t *a = &w->ext[i];
82 wv_extent_t *b = &w->ext[i + 1];
83
84 a->crc = (uint32_t) crc32_combine((uLong) a->crc, (uLong) b->crc,
85 (z_off_t) (b->hi - b->lo));
86 a->hi = b->hi;
87 memmove(b, b + 1, (w->n - i - 2) * sizeof(*b));
88 w->n--;
89}

91int
92brix_wverify_update(brix_wverify_t *w, const void *buf, off_t off, size_t len)
93{
94 off_t hi;
95 size_t pos;
96
97 if (w == NULL || off < 0 || (buf == NULL && len != 0)) {
98 if (w != NULL) { w->degraded = 1; }
99 return -1;
100 }
101 if (len == 0) {
102 return 0;
103 }
104 if ((off_t) len < 0 || off > (off_t) (((uint64_t) 1 << 62)) - (off_t) len) {
105 w->degraded = 1;
106 return -1; /* off + len would overflow */
107 }
108 hi = off + (off_t) len;
109
110 /* Insertion point: first extent whose lo >= off. Reject any overlap with the
111 * neighbours on either side (a fresh write must not restate committed bytes). */
112 for (pos = 0; pos < w->n && w->ext[pos].lo < off; pos++) {
113 /* nothing */
114 }
115 if (pos > 0 && w->ext[pos - 1].hi > off) {
116 w->degraded = 1;
117 return -1; /* overlaps the extent to the left */
118 }
119 if (pos < w->n && w->ext[pos].lo < hi) {
120 w->degraded = 1;
121 return -1; /* overlaps the extent to the right */
122 }
123
124 if (wv_reserve(w) != 0) {
125 w->degraded = 1;
126 return -1;
127 }
128 memmove(&w->ext[pos + 1], &w->ext[pos], (w->n - pos) * sizeof(w->ext[0]));
129 w->ext[pos].lo = off;
130 w->ext[pos].hi = hi;
131 w->ext[pos].crc = (uint32_t) crc32(crc32(0L, Z_NULL, 0),
132 (const Bytef *) buf, (uInt) len);
133 w->n++;
134
135 /* Coalesce with an abutting right neighbour, then the left, so the list stays
136 * minimal (a contiguous stream collapses to a single extent). */
137 if (pos + 1 < w->n && w->ext[pos].hi == w->ext[pos + 1].lo) {
138 wv_merge_at(w, pos);
139 }
140 if (pos > 0 && w->ext[pos - 1].hi == w->ext[pos].lo) {
141 wv_merge_at(w, pos - 1);
142 }
143
144 return 0;
145}
