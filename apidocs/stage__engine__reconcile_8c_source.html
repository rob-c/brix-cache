1/*
2 * stage_engine_reconcile.c — restart replay of the durable staging journal
3 * (split from stage_engine.c, phase-79).
4 *
5 * WHAT: On worker start, re-drives every persisted staged-flush record so no
6 * write-back transfer is lost across a crash: snapshot the *.req names,
7 * decode each (size-tolerantly), and re-flush the recoverable ones through
8 * the export's rebuilt stage decorator with the persisted owner credential.
9 *
10 * WHY: stage_engine.c owned three concerns and exceeded the file-size cap.
11 * Restart reconcile is a self-contained concern reached only through the
12 * journal seam (stage_journal_dir) and the public dead-letter helper. The
13 * replayable-vs-droppable classification (only a FLUSH with a durable stage
14 * copy replays; an UPLOAD/MULTIPART body is gone so its record is dropped)
15 * and the deny dead-letter cap are unchanged by the split — this is a
16 * re-home, not a behavior change.
17 *
18 * HOW: brix_stage_reconcile snapshots the journal names first (records are
19 * unlinked while driving), then calls stage_reconcile_one per name, which
20 * reads + decodes the record, rebuilds the export stack via
21 * brix_vfs_backend_resolve, unwraps to the stage decorator, and calls
22 * brix_sd_stage_reflush; a permanent deny goes through stage_deny_terminal.
23 */

38/* Re-flush ONE persisted record. Only a staged-write FLUSH is replayable: its
39 * bytes are durable on the stage store and the export anchor lets us rebuild both
40 * tiers. An UPLOAD/MULTIPART (client body -> stage) cannot be replayed - the body
41 * is gone after a crash, so the client retries; such records are dropped, never
42 * silently re-driven wrong. Returns 1 replayed / 0 kept (retry) / -1 dropped. */
43/* Re-flush ONE persisted record.
44 *
45 * WHAT: Reads the journal record (up to sizeof brix_sreq_t bytes), decodes it
46 * with size-tolerance via brix_sreq_decode, then re-drives a FLUSH through
47 * the export's stage decorator with the persisted owner credential.
48 *
49 * WHY: Only a staged-write FLUSH is replayable: its bytes are durable on the
50 * stage store and the export anchor lets us rebuild both tiers. An
51 * UPLOAD/MULTIPART (client body -> stage) cannot be replayed — the body is
52 * gone after a crash, so the client retries; such records are dropped.
53 * The size-tolerant decode allows journals written before the cred field
54 * was appended to be replayed with a zeroed cred (= service-credential flush,
55 * preserving pre-feature semantics).
56 *
57 * HOW: Read into a max-size buffer, call brix_sreq_decode; on decode failure
58 * drop the record. Pass the cred (NULL when key[0]=='\0') to reflush.
59 * Returns 1 replayed / 0 kept (retry) / -1 dropped. */
60static int
61stage_reconcile_one(const char *path, ngx_log_t *log)
62{
63 char rbuf[sizeof(brix_sreq_t)];
64 brix_sreq_t rec;
65 brix_sd_instance_t *inst;
66 const brix_stage_cred_t *credp;
67 int fd;
68 ssize_t n;
69
70 fd = open(path, O_RDONLY | O_CLOEXEC);
71 if (fd < 0) {
72 return 0;
73 }
74 n = read(fd, rbuf, sizeof(rbuf));
75 (void) close(fd);
76
77 if (brix_sreq_decode(rbuf, (size_t) n, &rec) != NGX_OK) {
78 (void) unlink(path); /* corrupt/short/oversized record - drop */
79 return -1;
80 }
81 if (rec.kind != BRIX_STAGE_FLUSH || rec.export_root[0] == '\0') {
82 (void) unlink(path); /* not a recoverable staged write */
83 return -1;
84 }
85
86 credp = (rec.cred.key[0] != '\0') ? &rec.cred : NULL;
87
88 /* Rebuild the export's composed stack and unwrap to its stage decorator.
89 * On success the stage copy is already dropped by reflush. On a permanent
90 * deny (EACCES from reflush + deny cred), apply the dead-letter cap so the
91 * reconcile does not re-drive the same record forever. On any other failure,
92 * keep the record for a later transient retry. */
93 inst = brix_vfs_backend_resolve(rec.export_root, log);
94 if (brix_sd_cache_instance_is(inst)) {
95 inst = brix_sd_cache_source_instance(inst);
96 }
97 if (brix_sd_stage_instance_is(inst)) {
98 ngx_int_t rc;
99 int saved_errno;
100
101 errno = 0;
102 rc = brix_sd_stage_reflush(inst, rec.dst_key, credp);
103 saved_errno = errno;
104
105 if (rc == NGX_OK) {
106 (void) unlink(path); /* re-flushed + stage copy dropped */
107 return 1;
108 }
109
110 /* Distinguish a permanent deny (EACCES, deny cred) from a transient
111 * error so dead-letter logic applies only to the former. */
112 if (saved_errno == EACCES && rec.cred.deny) {
113 /* stage_deny_terminal bumps attempts, persists, moves to deadletter
114 * when the cap is reached. The `path` is the active record; we use
115 * stage_journal_dir + reqid (not path) so the helper can rebuild the
116 * active and deadletter paths itself from the canonical dir. */
117 if (stage_deny_terminal(stage_journal_dir, rec.reqid, &rec, log)) {
118 return -1; /* dead-lettered = dropped from active journal */
119 }
120 /* Below the cap: keep for retry (same as transient). */
121 ngx_log_error(NGX_LOG_WARN, log, 0,
122 "xrootd stage: reconcile denied flush \"%s\" (export \"%s\" "
123 "attempts=%uD) - record kept (below dead-letter cap)",
124 rec.dst_key, rec.export_root, rec.attempts);
125 return 0;
126 }
127
128 /* Transient re-drive failure (the origin is still unreachable): bump
129 * attempts and re-persist the record FAILED. The higher attempt count is
130 * the durable evidence that the restart replay re-drove this transfer —
131 * against a recovered origin the reflush above would instead have
132 * completed and unlinked the record. Kept in the active journal so a
133 * later tick / restart retries it. */
134 stage_journal_bump_failed(stage_journal_dir, &rec, saved_errno);
135 ngx_log_error(NGX_LOG_WARN, log, 0,
136 "xrootd stage: reconcile re-flush of \"%s\" (export \"%s\") failed "
137 "(errno %d attempts=%uD) - record kept FAILED for retry",
138 rec.dst_key, rec.export_root, saved_errno, rec.attempts);
139 return 0;
140 }
141
142 /* Backend unreachable / no stage tier now: keep the record for a later retry. */
143 ngx_log_error(NGX_LOG_WARN, log, 0,
144 "xrootd stage: reconcile could not re-flush \"%s\" (export \"%s\") - "
145 "record kept for retry", rec.dst_key, rec.export_root);
146 return 0;
147}

149/* Snapshot the *.req journal record names from `dir` into `names`.
150 *
151 * WHAT: Opens `dir`, collects up to `cap` names ending in ".req" (each shorter
152 * than the per-slot width) into `names`, and returns the count. Returns 0
153 * when the directory cannot be opened.
154 *
155 * WHY: The driving loop unlinks records as it replays/drops them, so the name
156 * set must be snapshotted before driving rather than iterated live.
157 *
158 * HOW: readdir loop with a suffix + length filter, copying each accepted name
159 * (with its NUL) into the caller's fixed-width array. */
160static ngx_uint_t
161stage_reconcile_snapshot(const char *dir, char names[][256], ngx_uint_t cap)
162{
163 DIR *d;
164 struct dirent *de;
165 ngx_uint_t ncount = 0;
166
167 d = opendir(dir);
168 if (d == NULL) {
169 return 0;
170 }
171 while ((de = readdir(d)) != NULL && ncount < cap) {
172 size_t nlen = strlen(de->d_name);
173
174 if (nlen > 4 && nlen < 256
175 && strcmp(de->d_name + nlen - 4, ".req") == 0)
176 {
177 memcpy(names[ncount], de->d_name, nlen + 1);
178 ncount++;
179 }
180 }
181 closedir(d);
182 return ncount;
183}

185void
186brix_stage_reconcile(brix_stage_queue_t *queue)
187{
188 char names[1024][256];
189 ngx_uint_t ncount, i, replayed = 0, kept = 0, dropped = 0;
190 ngx_log_t *log;
191
192 (void) queue;
193 if (stage_journal_dir[0] == '\0') {
194 return; /* no durable journal configured */
195 }
196 /* Reconcile REPORTS every replay/keep/drop decision through ngx_log_error,
197 * whose macro dereferences the log unconditionally — a NULL log here is a
198 * crash, not a quiet run. Without a cycle (pre-init or a bare harness)
199 * DEFER: return with the journal untouched so a properly-initialised boot
200 * replays it. The in-tree caller (worker-0 init_process) always has one. */
201 if (ngx_cycle == NULL) {
202 return;
203 }
204 log = ngx_cycle->log;
205
206 /* Snapshot the *.req names first (we unlink while driving). */
207 ncount = stage_reconcile_snapshot(stage_journal_dir, names, 1024);
208
209 for (i = 0; i < ncount; i++) {
210 char path[1300];
211 int r;
212
213 if ((size_t) snprintf(path, sizeof(path), "%s/%s",
214 stage_journal_dir, names[i]) >= sizeof(path))
215 {
216 continue;
217 }
218 r = stage_reconcile_one(path, log);
219 if (r > 0) { replayed++; }
220 else if (r < 0) { dropped++; }
221 else { kept++; }
222 }
223
224 if (replayed > 0 || kept > 0 || dropped > 0) {
225 ngx_log_error(NGX_LOG_NOTICE, log, 0,
226 "xrootd stage: restart reconcile - %ui staged flush(es) replayed, "
227 "%ui kept for retry, %ui dropped", replayed, kept, dropped);
228 }
229}
