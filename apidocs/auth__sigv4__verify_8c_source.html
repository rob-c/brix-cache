2 * auth_sigv4_verify.c — AWS Signature Version 4 verification orchestrator.
3 *
4 * WHAT: Owns the SigV4 verification entry point (s3_verify_sigv4) and the
5 * decision-side helpers around it: the auth-result metric sink, the
6 * WLCG bearer-token intercept, the presigned-vs-header credential parse
7 * fork, the STS session-token gate, and the success-side commit.
8 * WHY: This file exceeded the ~500-line file-size guard, so the byte-frozen
9 * timestamp rules and the SigV4 crypto were carved into auth_sigv4_verify_
10 * time.c and auth_sigv4_verify_crypto.c (phase-79). What remains here is
11 * the orchestration — a flat early-return sequence over the split steps —
12 * plus the side-effect-light gates that guard it. INVARIANT §6: SigV4 and
13 * Bearer are mutually exclusive per request; the intercept keeps that
14 * arbitration in one auditable place.
15 * HOW: s3_verify_sigv4 threads the split steps: bearer intercept → anonymous

32/*
33 * s3_record_auth_result — record a SigV4 auth outcome into the metric counters.
34 *
35 * WHAT: Increment the per-result S3 auth counter and the unified auth metric
36 * for the given BRIX_S3_AUTH_* result code.
37 * WHY: Every SigV4 verify unit (this file plus verify_time.c / verify_crypto.c)
38 * records its outcome on both success and rejection edges, so the sink is
39 * exported here and called across the split.
40 * HOW: Bump the raw result counter, then map the result to the unified
41 * authn-method/success pair.
42 */
43void
44s3_record_auth_result(ngx_uint_t result)
45{
46 BRIX_S3_METRIC_INC(auth_total[result]);
47
48 switch (result) {
49 case BRIX_S3_AUTH_ANONYMOUS:
50 brix_metric_auth(BRIX_PROTO_S3, BRIX_AUTHN_NONE, 1);
51 break;
52 case BRIX_S3_AUTH_SIGV4_OK:
53 brix_metric_auth(BRIX_PROTO_S3, BRIX_AUTHN_S3KEY, 1);
54 break;
55 case BRIX_S3_AUTH_MISSING:
56 brix_metric_auth(BRIX_PROTO_S3, BRIX_AUTHN_NONE, 0);
57 break;
58 default:
59 brix_metric_auth(BRIX_PROTO_S3, BRIX_AUTHN_S3KEY, 0);
60 break;
61 }

64/*
65 * s3_signed_headers_contains — test whether a SignedHeaders list names a header.
66 *
67 * WHAT: Return 1 when the semicolon-separated signed_hdrs list contains name
68 * (case-insensitive, whole-token match), 0 otherwise.
69 * WHY: The session-token gate must confirm x-amz-security-token was actually
70 * covered by the signature; a whole-token scan avoids substring matches.
71 * HOW: Walk signed_hdrs token by token on ';', comparing each token's length
72 * and case-insensitive bytes against name.
73 */
74static ngx_flag_t
75s3_signed_headers_contains(const char *signed_hdrs, const char *name)
76{
77 size_t name_len;
78 const char *p;
79 const char *end;
80
81 if (signed_hdrs == NULL || name == NULL) {
82 return 0;
83 }
84
85 name_len = strlen(name);
86 p = signed_hdrs;
87
88 while (*p != '\0') {
89 while (*p == ';') {
90 p++;
91 }
92
93 end = strchr(p, ';');
94 if (end == NULL) {
95 end = p + strlen(p);
96 }
97
98 if ((size_t) (end - p) == name_len
99 && ngx_strncasecmp((u_char *) p, (u_char *) name, name_len) == 0)
100 {
101 return 1;
102 }
103
104 p = end;
105 }
106
107 return 0;
108}

141/*
142 * WLCG bearer-token intercept (INVARIANT §6: SigV4 and Bearer are mutually
143 * exclusive per request — never blend the auth logic).
144 *
145 * WHAT: When brix_s3_token is enabled, decide whether the request should be
146 * handled by the Bearer path, rejected, or allowed to fall through to
147 * SigV4/anonymous.
148 * WHY: Keeps the token-vs-SigV4 arbitration in one place so the SigV4 verifier
149 * body stays a linear sequence and INVARIANT §6 is auditable at a glance.
150 * HOW: Detects Bearer and SigV4 presence; both-present is a 400 client error;
151 * Bearer-only dispatches to s3_verify_bearer; token-only mode with neither
152 * is a 403. Returns NGX_DECLINED to mean "fall through to SigV4"; any
153 * other value is a terminal response already emitted by this helper.
154 *
155 * Returns: NGX_DECLINED to continue into the SigV4/anonymous path, otherwise the
156 * terminal ngx_int_t result (response emitted).
157 */
158static ngx_int_t
159s3_sigv4_bearer_intercept(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
160 brix_identity_t *identity)
161{
162 ngx_str_t authz;
163 int has_bearer;
164 int has_sigv4;
165
166 if (!cf->token_enable) {
167 return NGX_DECLINED;
168 }
169
170 authz = get_header(r, "authorization");
171 has_bearer = s3_bearer_present(r);
172 has_sigv4 = (authz.len >= 4
173 && ngx_strncasecmp(authz.data, (u_char *) "AWS4", 4) == 0);
174
175 if (has_bearer && has_sigv4) {
176 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
177 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
178 "InvalidRequest",
179 "both Bearer and SigV4 credentials present");
180 }
181
182 if (has_bearer) {
183 return s3_verify_bearer(r, cf, identity);
184 }
185
186 /* Enforcing token-only mode: no Bearer, no SigV4 — reject. */
187 if (!has_sigv4) {
188 s3_record_auth_result(BRIX_S3_AUTH_MISSING);
189 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
190 "AccessDenied",
191 "bearer token required");
192 }
193
194 /* SigV4 credentials were presented: fall through to verify them if an
195 * access_key is configured, or to the anonymous path if not. */
196 return NGX_DECLINED;
197}

199/*
200 * s3_sigv4_parse_authz_header — parse the SigV4 credentials into components.
201 *
202 * WHAT: Populate *comp from either a presigned URL query string or the
203 * Authorization header.
204 * WHY: Isolates the presigned-vs-header parse fork (and its two distinct error
205 * messages) from the verifier orchestrator.
206 * HOW: Try presigned first; on NGX_DECLINED fall back to the header parse.
207 * Frozen error messages and result codes preserved 1:1.
208 *
209 * Returns: NGX_OK when *comp is filled, otherwise the terminal ngx_int_t result
210 * (response already emitted).
211 */
212static ngx_int_t
213s3_sigv4_parse_authz_header(ngx_http_request_t *r, sigv4_components_t *comp)
214{
215 ngx_str_t auth;

217

250/*
251 * s3_sigv4_check_session_token — enforce STS session-token policy.
252 *
253 * WHAT: Reject requests that carry an X-Amz-Security-Token when STS session
254 * tokens are disabled, or (for header-signed requests) when the token was
255 * not covered by the signature.
256 * WHY: Keeps the session-token gate out of the verifier's main line.
257 * HOW: Only acts when the request actually carries a session token; frozen
258 * error messages and result codes preserved 1:1.
259 *
260 * Returns: NGX_OK to continue, otherwise the terminal ngx_int_t result.
261 */
262static ngx_int_t
263s3_sigv4_check_session_token(ngx_http_request_t *r,
264 ngx_http_s3_loc_conf_t *cf, const sigv4_components_t *comp)
265{
266 if (!s3_request_has_session_token(r, comp->presigned)) {
267 return NGX_OK;
268 }
269
270 if (!cf->allow_unsigned_session_token) {
271 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
272 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
273 "AccessDenied",
274 "STS session tokens are not enabled");
275 }
276
277 if (!comp->presigned
278 && !s3_signed_headers_contains(comp->signed_hdrs,
279 "x-amz-security-token"))
280 {
281 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
282 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
283 "InvalidRequest",
284 "X-Amz-Security-Token must be signed");
285 }
286
287 return NGX_OK;

290/*
291 * s3_sigv4_finish — commit a successful verification.
292 *
293 * WHAT: Record the OK result, bind the S3-key subject identity, and (when
294 * per-chunk signature verification is enabled) retain the SigV4 material
295 * the streaming decoder needs.
296 * WHY: Keeps the success-side side effects at the edge of the verifier, out of
297 * the decision logic.
298 * HOW: Set identity subject; on the signed (non-presigned) path only, copy the
299 * signing key, seed signature, full timestamp, and scope into the request
300 * module ctx (W6a). Frozen result codes and buffer layout preserved 1:1.
301 *
302 * Returns: NGX_OK on success, otherwise the terminal ngx_int_t result.
303 */
304static ngx_int_t
305s3_sigv4_finish(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
306 const sigv4_components_t *comp, brix_identity_t *identity,
307 const u_char k4[32], const char *amz_date)
308{
309 ngx_http_s3_req_ctx_t *rx;
310
311 s3_record_auth_result(BRIX_S3_AUTH_SIGV4_OK);
312 if (identity != NULL
313 && brix_identity_set_subject(identity, r->pool, comp->akid,
314 BRIX_AUTHN_S3KEY) != NGX_OK)
315 {
316 s3_record_auth_result(BRIX_S3_AUTH_INTERNAL_ERROR);
317 return NGX_HTTP_INTERNAL_SERVER_ERROR;
318 }
319
320 /*
321 * W6a: when per-chunk signature verification is enabled, retain the SigV4
322 * material the streaming decoder needs (seed signature, signing key, scope,
323 * timestamp) — it is otherwise local to this function and discarded. Only
324 * the signed (non-presigned) header path produces a seed signature usable as
325 * the first chunk's previous-signature.
326 */
327 if (cf->verify_chunk_signatures && !comp->presigned) {
328 rx = ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
329 if (rx != NULL) {
330 u_char *e;
331 ngx_memcpy(rx->sigv4_signing_key, k4, 32);
332 ngx_cpystrn((u_char *) rx->sigv4_seed_signature,
333 (u_char *) comp->signature,
334 sizeof(rx->sigv4_seed_signature));
335 ngx_cpystrn((u_char *) rx->sigv4_amz_date,
336 (u_char *) amz_date, /* full timestamp (header path) */
337 sizeof(rx->sigv4_amz_date));
338 e = ngx_snprintf((u_char *) rx->sigv4_scope,
339 sizeof(rx->sigv4_scope) - 1,
340 "%s/%s/s3/aws4_request", comp->date, comp->region);
341 *e = '\0';
342 rx->have_sigv4 = 1;
343 }
344 }
345
346 return NGX_OK;
347}

353/*
354 * s3_verify_sigv4 — verify an AWS Signature Version 4 Authorization header.
355 *
356 * Implements the standard SigV4 verification algorithm:
357 * 1. Parse the Authorization header into AKID, date, region, signed headers
358 * and signature components.
359 * 2. Verify the access key matches the configured key.
360 * 3. Build the canonical request (method, URI, query string, headers).
361 * 4. Build the string-to-sign from the canonical request hash.
362 * 5. Derive the signing key via four rounds of HMAC-SHA256:
363 * k1 = HMAC("AWS4" + secret, date)
364 * k2 = HMAC(k1, region)
365 * k3 = HMAC(k2, "s3")
366 * k4 = HMAC(k3, "aws4_request")
367 * 6. Compute HMAC-SHA256(k4, string-to-sign) and compare with the
368 * client-provided signature.
369 *
370 * XrdClS3 always uses UNSIGNED-PAYLOAD so the request body is not hashed.
371 *
372 * Anonymous mode: when cf->access_key.len == 0, all requests pass without
373 * any verification (useful for read-only public endpoints).
374 *
375 * Returns: NGX_OK if the signature is valid (or anonymous mode), or an
376 * XML S3 error response (via s3_send_xml_error) on failure.
377 */
378ngx_int_t
379s3_verify_sigv4(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
380 brix_identity_t *identity)
381{
382 sigv4_components_t comp;
383 s3_amz_date_out_t amz;
384 s3_sigv4_sig_out_t sig;
385 ngx_int_t rc;
386 int key_ok; /* W5: deferred access-key match flag */
387
388 ngx_memzero(&amz, sizeof(amz));
389 ngx_memzero(&sig, sizeof(sig));
390
391 /* WLCG bearer-token intercept (INVARIANT §6). NGX_DECLINED = fall through. */
392 rc = s3_sigv4_bearer_intercept(r, cf, identity);
393 if (rc != NGX_DECLINED) {
394 return rc;
395 }
396
397 /* Anonymous mode */
398 if (cf->access_key.len == 0) {
399 if (identity != NULL) {
400 identity->auth_method = BRIX_AUTHN_NONE;
401 }
402 s3_record_auth_result(BRIX_S3_AUTH_ANONYMOUS);
403 return NGX_OK;
404 }
405
406 rc = s3_sigv4_parse_authz_header(r, &comp);
407 if (rc != NGX_OK) {
408 return rc;
409 }
410
411 /*
412 * Verify the access key — W5: do NOT early-return on mismatch.
413 *
414 * An early exit here (before the signing-key derive + HMAC compute below)
415 * created a timing AND message oracle: an unknown key returned quickly with
416 * "InvalidAccessKeyId", while a known key with a bad signature ran the full
417 * HMAC and returned "SignatureDoesNotMatch". An attacker could enumerate
418 * valid access keys from the response time and message alone.
419 *
420 * Instead, record the result in key_ok (constant-time compare) and fold it
421 * into the single signature decision at the end, so both an unknown key and
422 * a bad signature traverse the same HMAC work and return the identical
423 * "SignatureDoesNotMatch" error. The length check short-circuits the
424 * CRYPTO_memcmp only to avoid an out-of-bounds read; the akid length is not
425 * a sensitive secret.
426 */
427 key_ok = (cf->access_key.len == strlen(comp.akid))
428 && CRYPTO_memcmp(cf->access_key.data,
429 (u_char *) comp.akid, cf->access_key.len) == 0;
430
431 rc = s3_sigv4_check_session_token(r, cf, &comp);
432 if (rc != NGX_OK) {
433 return rc;
434 }
435
436 rc = s3_sigv4_resolve_request_time(r, &comp, &amz);
437 if (rc != NGX_OK) {
438 return rc;
439 }
440
441 if (s3_sigv4_compute_signature(r, cf, &comp, &amz, &sig) != NGX_OK) {
442 return NGX_HTTP_INTERNAL_SERVER_ERROR;
443 }
444
445 rc = s3_sigv4_compare(r, &comp, sig.computed_hex, key_ok);
446 if (rc != NGX_OK) {
447 return rc;
448 }
449
450 return s3_sigv4_finish(r, cf, &comp, identity, sig.k4, amz.date);
451}
