52static ngx_flag_t
53s3_signed_headers_contains(const char *signed_hdrs, const char *name)
54{
55 size_t name_len;
56 const char *p;
57 const char *end;
58
59 if (signed_hdrs == NULL || name == NULL) {
60 return 0;
61 }
62
63 name_len = strlen(name);
64 p = signed_hdrs;
65
66 while (*p != '\0') {
67 while (*p == ';') {
68 p++;
69 }
70
71 end = strchr(p, ';');
72 if (end == NULL) {
73 end = p + strlen(p);
74 }
75
76 if ((size_t) (end - p) == name_len
77 && ngx_strncasecmp((u_char *) p, (u_char *) name, name_len) == 0)
78 {
79 return 1;
80 }
81
82 p = end;
83 }
84
85 return 0;

115static int
116s3_parse_2digits(const char *s)
117{
118 int hi, lo;
119
120 hi = s3_digit((u_char) s[0]);
121 lo = s3_digit((u_char) s[1]);
122 if (hi < 0 || lo < 0) {
123 return -1;
124 }
125 return hi * 10 + lo;

128static int
129s3_parse_4digits(const char *s)
130{
131 int a, b, c, d;
132
133 a = s3_digit((u_char) s[0]);
134 b = s3_digit((u_char) s[1]);
135 c = s3_digit((u_char) s[2]);
136 d = s3_digit((u_char) s[3]);
137 if (a < 0 || b < 0 || c < 0 || d < 0) {
138 return -1;
139 }
140 return a * 1000 + b * 100 + c * 10 + d;
141}

143static ngx_int_t
144s3_parse_amz_datetime(const char *s, time_t *out)
145{
146 struct tm tm;
147 int year, mon, day, hour, min, sec;
148
149 if (strlen(s) != 16 || s[8] != 'T' || s[15] != 'Z') {
150 return NGX_ERROR;
151 }
152
153 year = s3_parse_4digits(s);
154 mon = s3_parse_2digits(s + 4);
155 day = s3_parse_2digits(s + 6);
156 hour = s3_parse_2digits(s + 9);
157 min = s3_parse_2digits(s + 11);
158 sec = s3_parse_2digits(s + 13);
159
160 if (year < 1970 || mon < 1 || mon > 12 || day < 1 || day > 31
161 || hour < 0 || hour > 23 || min < 0 || min > 59
162 || sec < 0 || sec > 60)
163 {
164 return NGX_ERROR;
165 }
166
167 ngx_memzero(&tm, sizeof(tm));
168 tm.tm_year = year - 1900;
169 tm.tm_mon = mon - 1;
170 tm.tm_mday = day;
171 tm.tm_hour = hour;
172 tm.tm_min = min;
173 tm.tm_sec = sec;
174 *out = timegm(&tm);
175 return NGX_OK;
176}

228/*
229 * SigV4 signing key derivation
230 *
231 * Four-round HMAC chain:
232 * k1 = HMAC("AWS4" + secret, date) k2 = HMAC(k1, region)
233 * k3 = HMAC(k2, "s3") k4 = HMAC(k3, "aws4_request")
234 * */
235
236/* Worker-local one-slot cache: signing key is stable for one calendar day per
237 * region, so cache the last key and avoid four HMAC rounds on every request. */
238static struct {
239 char date[9]; /* YYYYMMDD\0, empty string means invalid */
240 char region[65];
241 u_char key[32];
242} s_signing_key_cache;
243
244int
245s3_sigv4_derive_signing_key_cached(const ngx_str_t *secret_key,
246 const char *date, const char *region,
247 u_char out[32])
248{
249 if (s_signing_key_cache.date[0] != '\0'
250 && strcmp(s_signing_key_cache.date, date) == 0
251 && strcmp(s_signing_key_cache.region, region) == 0)
252 {
253 ngx_memcpy(out, s_signing_key_cache.key, 32);
254 return 1;
255 }
256
257 /* Shared 4-round HMAC chain (libxrdproto) — byte-identical to the client's
258 * sign path so client-signs == server-verifies by construction. */
259 if (!brix_sigv4_signing_key((const uint8_t *) secret_key->data,
260 secret_key->len, date, region, "s3", out)) {
261 return 0;
262 }
263
264 ngx_cpystrn((u_char *) s_signing_key_cache.date,
265 (u_char *) date, sizeof(s_signing_key_cache.date));
266 ngx_cpystrn((u_char *) s_signing_key_cache.region,
267 (u_char *) region, sizeof(s_signing_key_cache.region));
268 ngx_memcpy(s_signing_key_cache.key, out, 32);
269 return 1;
270}

272/*
273 * Canonical signed headers builder (merged from auth_sigv4_headers.c)
274 *
275 * Builds the "header-name:value\n" block required by SigV4. signed_hdrs is
276 * a semicolon-separated list, e.g. "host;x-amz-content-sha256;x-amz-date".
277 * */
278
279static size_t
280build_canonical_headers(ngx_http_request_t *r,
281 const char *signed_hdrs,
282 u_char *out, size_t outsz)
283{
284 char hdrs[256];
285 size_t oi = 0;
286
287 ngx_cpystrn((u_char *) hdrs, (u_char *) signed_hdrs, sizeof(hdrs));
288
289 char *save = NULL;
290 char *tok = strtok_r(hdrs, ";", &save);
291
292 while (tok) {
293 ngx_str_t val;
294
295 if (strcmp(tok, "host") == 0) {
296 val = r->headers_in.host ? r->headers_in.host->value
297 : (ngx_str_t) ngx_null_string;
298 } else {
299 val = get_header(r, tok);
300 }
301
302 size_t nlen = strlen(tok);
303 size_t vlen = val.len;
304
305 if (oi + nlen + 1 + vlen + 2 >= outsz) {
306 break;
307 }
308
309 for (size_t i = 0; i < nlen; i++) {
310 out[oi++] = (u_char) tolower((unsigned char) tok[i]);
311 }
312 out[oi++] = ':';
313
314 const u_char *vs = val.data;
315 const u_char *ve = val.data + vlen;
316 while (vs < ve && (*vs == ' ' || *vs == '\t')) { vs++; }
317 while (ve > vs && (ve[-1] == ' ' || ve[-1] == '\t')) { ve--; }
318
319 ngx_memcpy(out + oi, vs, (size_t)(ve - vs));
320 oi += (size_t)(ve - vs);
321 out[oi++] = '\n';
322
323 tok = strtok_r(NULL, ";", &save);
324 }
325
326 out[oi] = '\0';
327 return oi;
328}

334/*
335 * s3_verify_sigv4 — verify an AWS Signature Version 4 Authorization header.
336 *
337 * Implements the standard SigV4 verification algorithm:
338 * 1. Parse the Authorization header into AKID, date, region, signed headers
339 * and signature components.
340 * 2. Verify the access key matches the configured key.
341 * 3. Build the canonical request (method, URI, query string, headers).
342 * 4. Build the string-to-sign from the canonical request hash.
343 * 5. Derive the signing key via four rounds of HMAC-SHA256:
344 * k1 = HMAC("AWS4" + secret, date)
345 * k2 = HMAC(k1, region)
346 * k3 = HMAC(k2, "s3")
347 * k4 = HMAC(k3, "aws4_request")
348 * 6. Compute HMAC-SHA256(k4, string-to-sign) and compare with the
349 * client-provided signature.
350 *
351 * XrdClS3 always uses UNSIGNED-PAYLOAD so the request body is not hashed.
352 *
353 * Anonymous mode: when cf->access_key.len == 0, all requests pass without
354 * any verification (useful for read-only public endpoints).
355 *
356 * Returns: NGX_OK if the signature is valid (or anonymous mode), or an
357 * XML S3 error response (via s3_send_xml_error) on failure.
358 */
359ngx_int_t
360s3_verify_sigv4(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
361 brix_identity_t *identity)
362{
363 sigv4_components_t comp;
364 u_char canonical[8192];
365 u_char canon_qs[2048];
366 u_char canon_uri[S3_MAX_KEY];
367 u_char canon_hdrs[2048];
368 u_char string_to_sign[4096];
369 u_char hash_hex[65];
370 u_char cr_hash[32];
371 u_char k4[32];
372 u_char computed[32];
373 char computed_hex[65];
374 size_t n;
375 ngx_str_t auth, date_hdr;
376 const char *amz_date;
377 size_t amz_date_len;
378 char header_amz_date[32];
379 time_t request_time;
380 int parse_rc;
381 int key_ok; /* W5: deferred access-key match flag */
382
383 /*
384 * WLCG bearer-token intercept (INVARIANT §6: SigV4 and Bearer are mutually
385 * exclusive per request — never blend the auth logic).
386 *
387 * When brix_s3_token is enabled, detect and validate Bearer tokens BEFORE
388 * the SigV4 / anonymous path. A request that carries BOTH schemes is
389 * rejected immediately (400 InvalidRequest) — that is a client error, not
390 * an auth failure. A request that carries only a Bearer token is validated
391 * and returns; the SigV4 path is not entered. A request that carries
392 * neither, while token_enable is set, falls through to the SigV4 check
393 * (which will reject it as "Missing Authorization" if access_key is set,
394 * or as "AccessDenied: bearer token required" below if it is not).
395 */
396 if (cf->token_enable) {
397 ngx_str_t authz = get_header(r, "authorization");
398 int has_bearer = s3_bearer_present(r);
399 int has_sigv4 = (authz.len >= 4
400 && ngx_strncasecmp(authz.data,
401 (u_char *) "AWS4", 4) == 0);
402
403 if (has_bearer && has_sigv4) {
404 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
405 return s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
406 "InvalidRequest",
407 "both Bearer and SigV4 credentials present");
408 }
409
410 if (has_bearer) {
411 return s3_verify_bearer(r, cf, identity);
412 }
413
414 /* Enforcing token-only mode: no Bearer, no SigV4 — reject. */
415 if (!has_sigv4) {
416 s3_record_auth_result(BRIX_S3_AUTH_MISSING);
417 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
418 "AccessDenied",
419 "bearer token required");
420 }
421
422 /* SigV4 credentials were presented: fall through to verify them if
423 * an access_key is configured, or to the anonymous path if not. */
424 }
425
426 /* Anonymous mode */
427 if (cf->access_key.len == 0) {
428 if (identity != NULL) {
429 identity->auth_method = BRIX_AUTHN_NONE;
430 }
431 s3_record_auth_result(BRIX_S3_AUTH_ANONYMOUS);
432 return NGX_OK;
433 }
434
435 ngx_memzero(&comp, sizeof(comp));
436
437 parse_rc = parse_presigned_authorization(r, &comp);
438 if (parse_rc == NGX_DECLINED) {
439 auth = get_header(r, "authorization");
440 if (auth.len == 0) {
441 s3_record_auth_result(BRIX_S3_AUTH_MISSING);
442 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
443 "InvalidRequest",
444 "Missing Authorization header");
445 }
446
447 if (!parse_authorization(&auth, &comp)) {
448 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
449 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
450 "InvalidRequest",
451 "Malformed Authorization header");
452 }
453 } else if (parse_rc != NGX_OK) {
454 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
455 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
456 "InvalidRequest",
457 "Malformed presigned URL");
458 }
459
460 /*
461 * Verify the access key — W5: do NOT early-return on mismatch.
462 *
463 * An early exit here (before the signing-key derive + HMAC compute below)
464 * created a timing AND message oracle: an unknown key returned quickly with
465 * "InvalidAccessKeyId", while a known key with a bad signature ran the full
466 * HMAC and returned "SignatureDoesNotMatch". An attacker could enumerate
467 * valid access keys from the response time and message alone.
468 *
469 * Instead, record the result in key_ok (constant-time compare) and fold it
470 * into the single signature decision at the end, so both an unknown key and
471 * a bad signature traverse the same HMAC work and return the identical
472 * "SignatureDoesNotMatch" error. The length check short-circuits the
473 * CRYPTO_memcmp only to avoid an out-of-bounds read; the akid length is not
474 * a sensitive secret.
475 */
476 key_ok = (cf->access_key.len == strlen(comp.akid))
477 && CRYPTO_memcmp(cf->access_key.data,
478 (u_char *) comp.akid, cf->access_key.len) == 0;
479
480 if (s3_request_has_session_token(r, comp.presigned)) {
481 if (!cf->allow_unsigned_session_token) {
482 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
483 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
484 "AccessDenied",
485 "STS session tokens are not enabled");
486 }
487
488 if (!comp.presigned
489 && !s3_signed_headers_contains(comp.signed_hdrs,
490 "x-amz-security-token"))
491 {
492 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
493 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
494 "InvalidRequest",
495 "X-Amz-Security-Token must be signed");
496 }
497 }
498
499 if (comp.presigned) {
500 if (s3_parse_amz_datetime(comp.amz_date, &request_time) != NGX_OK) {
501 return s3_reject_bad_amz_date(r, "Invalid X-Amz-Date");
502 }
503
504 {
505 ngx_int_t skew_rc;
506
507 skew_rc = s3_check_presigned_future_skew(r, request_time);
508 if (skew_rc != NGX_OK) {
509 return skew_rc;
510 }
511 }
512
513 if (ngx_time() >= request_time
514 && ngx_time() - request_time > (time_t) comp.amz_expires)
515 {
516 s3_record_auth_result(BRIX_S3_AUTH_BAD_DATE);
517 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
518 "AccessDenied",
519 "Request has expired");
520 }
521
522 amz_date = comp.amz_date;
523 amz_date_len = strlen(comp.amz_date);
524
525 } else {
526 date_hdr = get_header(r, "x-amz-date");
527 if (date_hdr.len == 0) {
528 return s3_reject_bad_amz_date(r, "Missing X-Amz-Date header");
529 }
530
531 if (brix_str_cbuf(header_amz_date, sizeof(header_amz_date), &date_hdr)
532 == NULL)
533 {
534 return s3_reject_bad_amz_date(r, "Invalid X-Amz-Date");
535 }
536
537 if (s3_parse_amz_datetime(header_amz_date, &request_time) != NGX_OK) {
538 return s3_reject_bad_amz_date(r, "Invalid X-Amz-Date");
539 }
540
541 {
542 ngx_int_t skew_rc;
543
544 skew_rc = s3_check_header_clock_skew(r, request_time);
545 if (skew_rc != NGX_OK) {
546 return skew_rc;
547 }
548 }
549
550 amz_date = header_amz_date;
551 amz_date_len = strlen(header_amz_date);
552 }
553
554/*
555 * 1. Canonical URI
556 * */
557 brix_http_urlencode(r->uri.data, r->uri.len,
558 (char *) canon_uri, sizeof(canon_uri), "/");
559
560/*
561 * 2. Canonical query string
562 * */
563 build_canonical_qs(r->args.data, r->args.len, comp.presigned,
564 canon_qs, sizeof(canon_qs));
565
566/*
567 * 3. Canonical headers
568 * */
569 build_canonical_headers(r, comp.signed_hdrs,
570 canon_hdrs, sizeof(canon_hdrs));
571
572/*
573 * 4. Canonical request
574 * */
575 n = (size_t) snprintf((char *) canonical, sizeof(canonical),
576 "%.*s\n" /* method */
577 "%s\n" /* uri */
578 "%s\n" /* qs */
579 "%s\n" /* headers */
580 "%s\n" /* signed header names */
581 "UNSIGNED-PAYLOAD",
582 (int) r->method_name.len, r->method_name.data,
583 (char *) canon_uri,
584 (char *) canon_qs,
585 (char *) canon_hdrs,
586 comp.signed_hdrs);
587
588 brix_sha256(canonical, n, cr_hash);
589 brix_hex_encode(cr_hash, 32, (char *) hash_hex);
590
591/*
592 * 5. String to sign
593 * */
594 n = (size_t) snprintf((char *) string_to_sign, sizeof(string_to_sign),
595 "AWS4-HMAC-SHA256\n"
596 "%.*s\n"
597 "%s/%s/s3/aws4_request\n"
598 "%s",
599 (int) amz_date_len, amz_date,
600 comp.date,
601 comp.region,
602 (char *) hash_hex);
603
604/*
605 * 6. Derive signing key (cached: stable for one calendar day per region)
606 * */
607 if (!s3_sigv4_derive_signing_key_cached(&cf->secret_key,
608 comp.date, comp.region, k4)) {
609 s3_record_auth_result(BRIX_S3_AUTH_INTERNAL_ERROR);
610 return NGX_HTTP_INTERNAL_SERVER_ERROR;
611 }
612
613/*
614 * 7. Compute and compare signatures
615 * */
616 if (!brix_hmac_sha256(k4, 32, string_to_sign, n, computed)) {
617 s3_record_auth_result(BRIX_S3_AUTH_INTERNAL_ERROR);
618 return NGX_HTTP_INTERNAL_SERVER_ERROR;
619 }
620
621 brix_hex_encode(computed, 32, computed_hex);
622
623 /* Reject signatures that are not exactly 64 hex chars — prevents a short
624 * client string from causing CRYPTO_memcmp to compare against pad bytes. */
625 if (strlen(comp.signature) != 64) {
626 s3_record_auth_result(BRIX_S3_AUTH_MALFORMED);
627 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
628 "InvalidRequest",
629 "Signature must be 64 hex characters");
630 }
631
632 /*
633 * Single constant-time decision (W5): an unknown access key (!key_ok) and a
634 * bad signature take the identical path, status, and message here, having
635 * both run the full HMAC above — no timing or message oracle distinguishes
636 * them. CRYPTO_memcmp prevents a timing oracle on the HMAC value itself.
637 */
638 if (!key_ok || CRYPTO_memcmp(computed_hex, comp.signature, 64) != 0) {
639 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
640 "brix_s3: SigV4 auth failed for key=%s (key_ok=%d)",
641 comp.akid, key_ok);
642 s3_record_auth_result(key_ok ? BRIX_S3_AUTH_SIG_MISMATCH
643 : BRIX_S3_AUTH_BAD_KEY);
644 return s3_send_xml_error(r, NGX_HTTP_FORBIDDEN,
645 "SignatureDoesNotMatch",
646 "The request signature we calculated does "
647 "not match the signature you provided");
648 }
649
650 s3_record_auth_result(BRIX_S3_AUTH_SIGV4_OK);
651 if (identity != NULL
652 && brix_identity_set_subject(identity, r->pool, comp.akid,
653 BRIX_AUTHN_S3KEY) != NGX_OK)
654 {

683 }
