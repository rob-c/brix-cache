1/* client/lib/cred.c
2 *
3 * WHAT: Credential store — per-kind handler registry, view cache, expiry gate,
4 * and refresh machinery for all five credential kinds (X.509, bearer,
5 * Kerberos, SSS, S3 keys).
6 * WHY: Discovery and refresh logic was scattered across the per-protocol sec
7 * modules (sec_gsi.c / sec_token.c / …) and duplicated for the HTTP/S3
8 * transports. One store removes the drift and makes auto-refresh uniform.
9 * HOW: Per-kind handler accessors are declared __attribute__((weak)) here so
10 * the library and unit-test binaries compile before any handler (B3-B6)
11 * exists. brix_cred_store_new() calls each non-NULL weak accessor and
12 * stores the returned handler pointer in a fixed-size per-kind slot array.
13 * A per-kind cache slot (loaded flag + owned string copies + not_after)
14 * is checked on every brix_cred_acquire; if auto_refresh is set and the
15 * credential is within min_remaining_s of its expiry, handler->refresh
16 * (if non-NULL) then handler->acquire are called and the cache is updated.
17 * ngx-free; no goto; functional/modular design (one job per function).
18 */

54/* config deep-copy helpers */
55/*
56 * cfg_copy_alloc — deep-copy the caller's brix_cred_config onto the heap.
57 *
58 * WHAT: allocates a new brix_cred_config whose string fields are independent
59 * heap copies of the originals so the store outlives the caller's cfg.
60 * WHY: safe design: store OWNS all strings → no lifetime coupling to caller.
61 * HOW: struct copy + individual strdup for every const char * field.
62 * Returns NULL on any allocation failure; partial copies are freed.
63 */
64static brix_cred_config *
65cfg_copy_alloc(const brix_cred_config *src)
66{
67 brix_cred_config *c = calloc(1, sizeof(*c));
68 if (c == NULL) {
69 return NULL;
70 }
71
72 c->auto_refresh = src->auto_refresh;
73
74/* Helper: strdup one field; on failure free already-allocated fields and c. */
75#define DUP_FIELD(field) \
76 do { \
77 if (src->field != NULL) { \
78 c->field = strdup(src->field); \
79 if (c->field == NULL) { \
80 cfg_copy_free(c); \
81 return NULL; \
82 } \
83 } \
84 } while (0)
85
86 DUP_FIELD(proxy_path);
87 DUP_FIELD(bearer_literal);
88 DUP_FIELD(bearer_path);
89 DUP_FIELD(keytab_path);
90 DUP_FIELD(ccache);
91 DUP_FIELD(s3_access);
92 DUP_FIELD(s3_secret);
93 DUP_FIELD(oidc_account);
94
95#undef DUP_FIELD
96 return c;
97}

99/*
100 * cfg_copy_free — release all heap memory in a deep-copied config.
101 *
102 * WHAT: frees every strdup'd string field, then the config itself.
103 * WHY: companion to cfg_copy_alloc; ensures no leaks on store teardown.
104 * HOW: explicit free for each field (safe on NULL; free(NULL) is a no-op).
105 */
106static void
107cfg_copy_free(brix_cred_config *c)
108{
109 if (c == NULL) {
110 return;
111 }
112 free((char *)c->proxy_path);
113 free((char *)c->bearer_literal);
114 free((char *)c->bearer_path);
115 free((char *)c->keytab_path);
116 free((char *)c->ccache);
117 free((char *)c->s3_access);
118 free((char *)c->s3_secret);
119 free((char *)c->oidc_account);
120 free(c);
121}

123/* per-kind cache slot */
124/*
125 * cred_slot — one per brix_cred_kind; holds the cached view + owned strings.
126 *
127 * The brix_cred_view.s3_access / .s3_secret / .path / .token fields point at
128 * the owned[] array below (valid until the SAME kind's next acquire), exactly
129 * as documented in cred.h: "valid until the next acquire of the SAME kind on
130 * the SAME store."
131 */
132typedef struct {
133 int loaded;
134 brix_cred_view view;
135 int64_t not_after; /* unix epoch expiry; 0 = no expiry known */
136
137 /* Heap-owned copies of view strings — view.* pointers alias into these. */
138 char *owned_path;
139 char *owned_token;
140 char *owned_s3_access;
141 char *owned_s3_secret;
142} cred_slot;
143
144/* store struct */
145struct brix_cred_store {
146 brix_cred_config *cfg; /* deep copy */
147 const brix_cred_handler *handlers[XRDC_CRED_KIND_COUNT]; /* may be NULL */
148 cred_slot slots[XRDC_CRED_KIND_COUNT];
149};

151/* slot helpers */
152/*
153 * slot_clear_owned — free the owned string copies in a cache slot.
154 *
155 * WHAT: releases path/token/s3_access/s3_secret heap copies; zeroes the slot.
156 * WHY: called before writing a fresh acquire result so old strings don't leak.
157 * HOW: individual free + NULL-out; preserves .loaded and .not_after (cleared
158 * by the caller after this, so the order doesn't matter).
159 */
160static void
161slot_clear_owned(cred_slot *sl)
162{
163 free(sl->owned_path);
164 free(sl->owned_token);
165 free(sl->owned_s3_access);
166 free(sl->owned_s3_secret);
167 sl->owned_path = NULL;
168 sl->owned_token = NULL;
169 sl->owned_s3_access = NULL;
170 sl->owned_s3_secret = NULL;
171}

173/*
174 * slot_store_view — deep-copy a transient view into the cache slot.
175 *
176 * WHAT: copies each non-NULL string from *v into slot-owned buffers, then
177 * points sl->view.* at those buffers and records not_after.
178 * WHY: handler->acquire fills a view whose string pointers may be stack or
179 * handler-internal; the store must outlive that call frame.
180 * HOW: strdup each field; on any allocation failure free the partial set and
181 * return -1. The caller marks sl->loaded on success.
182 */
183static int
184slot_store_view(cred_slot *sl, const brix_cred_view *v, int64_t not_after)
185{
186 slot_clear_owned(sl);
187 /* Synchronise view pointers so any early-return (OOM) path leaves the slot
188 * in a valid (not-loaded) state rather than with dangling pointers. */
189 sl->view.path = NULL;
190 sl->view.token = NULL;
191 sl->view.s3_access = NULL;
192 sl->view.s3_secret = NULL;
193 sl->loaded = 0;
194
195 if (v->path != NULL) {
196 sl->owned_path = strdup(v->path);
197 if (sl->owned_path == NULL) {
198 slot_clear_owned(sl);
199 return -1;
200 }
201 }
202 if (v->token != NULL) {
203 sl->owned_token = strdup(v->token);
204 if (sl->owned_token == NULL) {
205 slot_clear_owned(sl);
206 return -1;
207 }
208 }
209 if (v->s3_access != NULL) {
210 sl->owned_s3_access = strdup(v->s3_access);
211 if (sl->owned_s3_access == NULL) {
212 slot_clear_owned(sl);
213 return -1;
214 }
215 }
216 if (v->s3_secret != NULL) {
217 sl->owned_s3_secret = strdup(v->s3_secret);
218 if (sl->owned_s3_secret == NULL) {
219 slot_clear_owned(sl);
220 return -1;
221 }
222 }
223
224 sl->view.kind = v->kind;
225 sl->view.path = sl->owned_path;
226 sl->view.token = sl->owned_token;
227 sl->view.s3_access = sl->owned_s3_access;
228 sl->view.s3_secret = sl->owned_s3_secret;
229 sl->view.not_after = not_after;
230 sl->not_after = not_after;
231 sl->loaded = 1;
232 return 0;
233}

249/* public API */
250/*
251 * brix_cred_store_new — allocate and initialise a credential store.
252 *
253 * WHAT: deep-copies cfg, queries each weak accessor for its handler, and
254 * stores non-NULL results in the per-kind slot array.
255 * WHY: the caller's cfg may be stack-allocated and go out of scope; deep copy
256 * is the safe option (documented contract).
257 * HOW: calloc the store; cfg_copy_alloc; call each weak accessor (guarding
258 * NULL) and store the returned handler pointer. Returns NULL on OOM.
259 */
260brix_cred_store *
261brix_cred_store_new(const brix_cred_config *cfg)
262{
263 brix_cred_store *s = calloc(1, sizeof(*s));
264 if (s == NULL) {
265 return NULL;
266 }
267
268 s->cfg = cfg_copy_alloc(cfg);
269 if (s->cfg == NULL) {
270 free(s);
271 return NULL;
272 }
273
274 /* Populate handler slots from the (possibly weak-NULL) accessors. */
275 if (brix_cred_x509 != NULL) {
276 s->handlers[XRDC_CRED_X509_PROXY] = brix_cred_x509();
277 }
278 if (brix_cred_bearer != NULL) {
279 s->handlers[XRDC_CRED_BEARER] = brix_cred_bearer();
280 }
281 if (brix_cred_krb5 != NULL) {
282 s->handlers[XRDC_CRED_KRB5] = brix_cred_krb5();
283 }
284 if (brix_cred_sss != NULL) {
285 s->handlers[XRDC_CRED_SSS] = brix_cred_sss();
286 }
287 if (brix_cred_s3keys != NULL) {
288 s->handlers[XRDC_CRED_S3KEYS] = brix_cred_s3keys();
289 }
290
291 return s;
292}

294/*
295 * brix_cred_store_free — release all store resources.
296 *
297 * WHAT: frees the per-kind owned strings, the deep-copied config, and the
298 * store itself.
299 * WHY: public teardown; matches brix_cred_store_new.
300 * HOW: iterate slots calling slot_clear_owned; then cfg_copy_free; then free.
301 */
302void
303brix_cred_store_free(brix_cred_store *s)
304{
305 int i;
306
307 if (s == NULL) {
308 return;
309 }
310 for (i = 0; i < XRDC_CRED_KIND_COUNT; i++) {
311 slot_clear_owned(&s->slots[i]);
312 }
313 cfg_copy_free(s->cfg);
314 free(s);
315}

317/*
318 * brix_cred_available — probe whether a credential of `kind` appears usable.
319 *
320 * WHAT: returns 1 if a handler is registered for `kind` AND handler->available
321 * returns non-zero; 0 otherwise. Does NOT load or cache anything.
322 * WHY: auth pre-flight diagnostic: predict server auth without a network call.
323 * HOW: bounds-check kind; guard NULL handler; delegate to available(cfg).

325int
326brix_cred_available(brix_cred_store *s, brix_cred_kind kind)
327{
328 const brix_cred_handler *h;
329
330 if (s == NULL || (unsigned)kind >= XRDC_CRED_KIND_COUNT) {
331 return 0;
332 }
333 h = s->handlers[kind];
334 if (h == NULL || h->available == NULL) {
335 return 0;
336 }
337 return h->available(s->cfg) != 0 ? 1 : 0;
338}

340/*
341 * should_refresh — decide whether the cached credential needs re-acquisition.
342 *
343 * WHAT: returns 1 when auto_refresh is enabled, an expiry is known, and the
344 * credential expires within min_remaining_s seconds.
345 * WHY: centralises the expiry-gate logic so brix_cred_acquire is readable.
346 * HOW: pure predicate; no side effects. min_remaining_s<=0 disables the check.
347 * Called only from the `else if (sl->loaded)` branch in brix_cred_acquire,
348 * so no redundant loaded-check here.
349 */
350static int
351should_refresh(const brix_cred_store *s, const cred_slot *sl, int min_remaining_s)
352{
353 int64_t now;
354
355 if (!s->cfg->auto_refresh) {
356 return 0;
357 }
358 if (sl->not_after == 0) {
359 return 0;
360 }
361 if (min_remaining_s <= 0) {
362 return 0;
363 }
364 now = (int64_t)time(NULL);
365 return (now + (int64_t)min_remaining_s >= sl->not_after) ? 1 : 0;
366}

368/*
369 * do_acquire — call handler->acquire and cache the result in the slot.
370 *
371 * WHAT: invokes handler->acquire(cfg, &raw, &not_after, st); on success
372 * deep-copies the view into the slot via slot_store_view.
373 * WHY: factored out so both the cold-miss and the expiry-triggered paths
374 * share one acquire site.
375 * HOW: calls acquire; on -1 propagates the error. On OOM during
376 * slot_store_view sets XRDC_EAUTH + message. Returns 0 on success.
377 */
378static int
379do_acquire(brix_cred_store *s, brix_cred_kind kind,
380 const brix_cred_handler *h, cred_slot *sl, brix_status *st)
381{
382 brix_cred_view raw = {0};
383 int64_t not_after = 0;
384 int rc;
385
386 rc = h->acquire(s->cfg, &raw, &not_after, st);
387 if (rc != 0) {
388 return -1;
389 }
390 if (slot_store_view(sl, &raw, not_after) != 0) {
391 brix_status_set(st, XRDC_EAUTH, 0,
392 "cred store: OOM caching %s credential",
393 kind_name(kind));
394 return -1;
395 }
396 return 0;
397}

399/*
400 * brix_cred_acquire — discover, cache, and return a credential view.
401 *
402 * WHAT: resolves the handler for `kind`; returns the cached view unless the
403 * slot is empty (cold miss) or auto_refresh is set and the credential
404 * is within min_remaining_s of its expiry. On cache hit returns *view
405 * pointing at the store-owned strings. Returns 0 / -1 (st set).
406 * WHY: single authoritative acquire path for all code that needs a cred.
407 * HOW: 1) guard NULL handler → XRDC_EAUTH.
408 * 2) cold miss → do_acquire.
409 * 3) warm hit: check should_refresh; if so, handler->refresh (if non-
410 * NULL) then do_acquire.
411 * 4) fill *view from the cached slot.
412 */
413int
414brix_cred_acquire(brix_cred_store *s, brix_cred_kind kind,
415 int min_remaining_s, brix_cred_view *view, brix_status *st)
416{
417 const brix_cred_handler *h;
418 cred_slot *sl;
419
420 if (s == NULL || (unsigned)kind >= XRDC_CRED_KIND_COUNT) {
421 brix_status_set(st, XRDC_EUSAGE, 0, "cred_acquire: invalid store or kind");
422 return -1;
423 }
424
425 h = s->handlers[kind];
426 if (h == NULL) {
427 brix_status_set(st, XRDC_EAUTH, 0,
428 "no %s credential handler available",
429 kind_name(kind));
430 return -1;
431 }
432
433 sl = &s->slots[kind];
434
435 /* Cold miss — no cached result yet. */
436 if (!sl->loaded) {
437 if (do_acquire(s, kind, h, sl, st) != 0) {
438 return -1;
439 }
440 } else if (should_refresh(s, sl, min_remaining_s)) {
441 /* Near-expiry refresh: call refresh (if available), then re-acquire. */
442 if (h->refresh != NULL) {
443 (void)h->refresh(s->cfg, st); /* best-effort; non-fatal on -1 */
444 }
445 if (do_acquire(s, kind, h, sl, st) != 0) {
446 return -1;
447 }
448 }
449
450 *view = sl->view;
451 return 0;
452}
