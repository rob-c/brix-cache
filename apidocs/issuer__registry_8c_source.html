1/*
2 * token/issuer_registry.c — SciTokens multi-issuer registry (see header).
3 *
4 * WHAT: Loads the upstream XRootD `scitokens.cfg` INI into a read-only table of
5 * issuers, each with its iss URL, audiences, base_path/restricted_path
6 * namespace scope, authorization strategy, and (optional) per-issuer
7 * JWKS. Provides exact-iss lookup and the base/restricted path gate.
8 * WHY: XRootD delegates authorization for a namespace subtree to an external
9 * issuer via this file; parsing the grammar verbatim lets an operator
10 * reuse their existing config unchanged (phase-59 W1, ADR-1).
11 * HOW: brix_ini_parse_file() drives reg_kv() per line; [Issuer N] keys fill
12 * an brix_token_issuer_t, [Global] keys fill shared audiences. After
13 * parsing, each issuer is validated (issuer + base_path required) and its
14 * JWKS loaded. Path scoping reuses brix_token_scope_path_matches().
15 * Unsupported keys are WARN-logged, never silently dropped (R4).
16 */

44/* brix_token_strategy_parse — "capability group mapping" → bits */uint32_t
45brix_token_strategy_parse(const char *value)
46{
47 char buf[128];
48 char *tok;
49 char *save = NULL;
50 uint32_t bits = 0;
51
52 copy_z(buf, sizeof(buf), value);
53 for (tok = strtok_r(buf, " ,", &save); tok != NULL;
54 tok = strtok_r(NULL, " ,", &save))
55 {
56 if (eqi(tok, "capability")) {
57 bits |= BRIX_AUTHZ_CAPABILITY;
58 } else if (eqi(tok, "group")) {
59 bits |= BRIX_AUTHZ_GROUP;
60 } else if (eqi(tok, "mapping")) {
61 bits |= BRIX_AUTHZ_MAPPING;
62 }
63 }
64 return bits;
65}

100/* reg_issuer_for — find-or-create the issuer for a section name */static brix_token_issuer_t *
101reg_issuer_for(brix_token_registry_t *reg, const char *name)
102{
103 brix_token_issuer_t *is;
104 int i;
105
106 for (i = 0; i < reg->count; i++) {
107 if (eqi(reg->issuers[i].name, name)) {
108 return &reg->issuers[i];
109 }
110 }
111 if (reg->count >= BRIX_TOKEN_MAX_ISSUERS) {
112 return NULL;
113 }
114 is = &reg->issuers[reg->count];
115 copy_z(is->name, sizeof(is->name), name);
116 copy_z(is->username_claim, sizeof(is->username_claim), "sub");
117 is->enabled = 1;
118 is->onmissing_fail = 1;
119 is->strategy = reg->default_strategy;
120 is->metric_bucket = reg->count;
121 reg->count++;
122 return is;

125/* ---- reg_kv_issuer_str — assign a bounded string-valued [Issuer] key ----
126 *
127 * WHAT: If key names one of the fixed-buffer string fields of an issuer
128 * (issuer URL, username/groups claim, default user, name map file, JWKS
129 * path), copies val into that field and returns 1; returns 0 if key is
130 * not one of those string fields (leaving it for another handler).
131 *
132 * WHY: Splits the flat issuer key ladder by value-handling kind so that no
133 * single dispatcher exceeds the complexity cap; each of these keys shares
134 * the identical bounded-copy shape, so grouping them keeps the mapping
135 * between key name and destination field trivially auditable.
136 *
137 * HOW: 1. Compare key case-insensitively against each string field name.
138 * 2. On the first match, copy_z into that field and return 1.
139 * 3. If none match, return 0 so the caller can try other kinds.
140 */
141static int
142reg_kv_issuer_str(brix_token_issuer_t *is, const char *key, const char *val)
143{
144 if (eqi(key, "issuer")) {
145 copy_z(is->issuer, sizeof(is->issuer), val);
146 } else if (eqi(key, "username_claim")) {
147 copy_z(is->username_claim, sizeof(is->username_claim), val);
148 } else if (eqi(key, "groups_claim")) {
149 copy_z(is->groups_claim, sizeof(is->groups_claim), val);
150 } else if (eqi(key, "default_user")) {
151 copy_z(is->default_user, sizeof(is->default_user), val);
152 } else if (eqi(key, "name_mapfile")) {
153 copy_z(is->name_mapfile, sizeof(is->name_mapfile), val);
154 } else if (eqi(key, "jwks_file")) {
155 copy_z(is->jwks_path, sizeof(is->jwks_path), val);
156 } else {
157 return 0;
158 }
159 return 1;
160}

162/* ---- reg_kv_issuer_list — append a list-valued [Issuer] key ----
163 *
164 * WHAT: If key names one of the multi-valued namespace/audience fields
165 * (base_path, restricted_path, audience/audience_json), splits val into
166 * the matching fixed array and returns 1; returns 0 otherwise.
167 *
168 * WHY: These keys share the "split a CSV list into a capped array" shape but
169 * target two different array element widths (path vs audience), so they
170 * are grouped apart from the scalar-string keys to keep each dispatcher
171 * small while preserving the exact per-key target arrays.
172 *
173 * HOW: 1. Compare key against base_path / restricted_path / audience names.
174 * 2. On match, call reg_add_list / reg_add_aud with that key's own
175 * destination array, count, and cap, then return 1.
176 * 3. If none match, return 0.
177 */
178static int
179reg_kv_issuer_list(brix_token_issuer_t *is, const char *key, const char *val)
180{
181 if (eqi(key, "base_path")) {
182 reg_add_list(is->base_paths, &is->base_path_count,
183 BRIX_TOKEN_MAX_BASEPATHS, val);
184 } else if (eqi(key, "restricted_path")) {
185 reg_add_list(is->restricted_paths, &is->restricted_path_count,
186 BRIX_TOKEN_MAX_BASEPATHS, val);
187 } else if (eqi(key, "audience") || eqi(key, "audience_json")) {
188 reg_add_aud(is->audiences, &is->audience_count,
189 BRIX_TOKEN_MAX_AUDIENCES, val);
190 } else {
191 return 0;
192 }
193 return 1;
194}

196/* ---- reg_kv_issuer_flag — set a boolean/strategy [Issuer] key ----
197 *
198 * WHAT: If key names one of the policy toggles (map_subject, onmissing,
199 * enabled, authorization_strategy), interprets val into the matching
200 * field and returns 1; returns 0 otherwise.
201 *
202 * WHY: These keys each parse val through a small predicate (parse_bool, an
203 * exact "fail" test, or brix_token_strategy_parse) rather than copying
204 * it; grouping them isolates the security-sensitive enable/deny and
205 * authorization-strategy decoding into one auditable place.
206 *
207 * HOW: 1. Compare key against each toggle name.
208 * 2. On match, decode val into that field (onmissing_fail is true only
209 * for the literal "fail", exactly as before) and return 1.
210 * 3. If none match, return 0.
211 */
212static int
213reg_kv_issuer_flag(brix_token_issuer_t *is, const char *key, const char *val)
214{
215 if (eqi(key, "map_subject")) {
216 is->map_subject = parse_bool(val);
217 } else if (eqi(key, "onmissing")) {
218 is->onmissing_fail = eqi(val, "fail");
219 } else if (eqi(key, "enabled")) {
220 is->enabled = parse_bool(val);
221 } else if (eqi(key, "authorization_strategy")) {
222 is->strategy = brix_token_strategy_parse(val);
223 } else {
224 return 0;
225 }
226 return 1;
227}

229/* ---- reg_kv_issuer — dispatch one [Issuer N] key line ----
230 *
231 * WHAT: Applies one key/val pair to issuer is by trying the string, list, and
232 * flag handlers in turn; if no handler claims the key, WARN-logs it as
233 * unsupported (never silently dropped, R4).
234 *
235 * WHY: Concentrates the full set of recognised issuer keys behind three
236 * kind-grouped handlers so the recognised-vs-unsupported decision — and
237 * thus which keys can influence issuer trust — stays in one small,
238 * reviewable function.
239 *
240 * HOW: 1. Try reg_kv_issuer_str, then _list, then _flag; the first to return
241 * 1 has fully handled the key, so return immediately.
242 * 2. If all three decline, emit the unsupported-issuer-key WARN.
243 */
244static void
245reg_kv_issuer(brix_token_registry_t *reg, brix_token_issuer_t *is,
246 const char *key, const char *val)
247{
248 if (reg_kv_issuer_str(is, key, val)) {
249 return;
250 }
251 if (reg_kv_issuer_list(is, key, val)) {
252 return;
253 }
254 if (reg_kv_issuer_flag(is, key, val)) {
255 return;
256 }
257 ngx_log_error(NGX_LOG_WARN, reg->log, 0,
258 "scitokens: unsupported issuer key \"%s\" (ignored)", key);
259}

261/* ---- reg_kv_global — dispatch one [Global] key line ----
262 *
263 * WHAT: Applies one [Global] key/val pair: audience/audience_json append to the
264 * shared global audience array; any other key is WARN-logged as
265 * unsupported.
266 *
267 * WHY: Keeps the [Global] section's small key set out of the top-level
268 * callback so the section split reads as a flat two-branch dispatch.
269 *
270 * HOW: 1. If key is audience/audience_json, append val to global_audiences.
271 * 2. Otherwise emit the unsupported-[Global]-key WARN.
272 */
273static void
274reg_kv_global(brix_token_registry_t *reg, const char *key, const char *val)
275{
276 if (eqi(key, "audience") || eqi(key, "audience_json")) {
277 reg_add_aud(reg->global_audiences, &reg->global_audience_count,
278 BRIX_TOKEN_MAX_AUDIENCES, val);
279 } else {
280 ngx_log_error(NGX_LOG_WARN, reg->log, 0,
281 "scitokens: unsupported [Global] key \"%s\" (ignored)", key);
282 }
283}

285/* ---- reg_kv — INI callback: route one key line to its section handler ----
286 *
287 * WHAT: brix_ini_parse_file() callback for each key/val line. Routes [Issuer N]
288 * lines to reg_kv_issuer (returning -1 if the issuer table is full) and
289 * [Global] lines to reg_kv_global; unknown/section-less lines are
290 * ignored. Returns 0 on success, -1 to abort parsing.
291 *
292 * WHY: Section routing is the one place that must fail the whole parse when no
293 * issuer slot is available; isolating it from per-key handling keeps that
294 * fatal path obvious and the callback within the complexity budget.
295 *
296 * HOW: 1. If section starts with "Issuer ", find-or-create the issuer; return
297 * -1 when the table is full, else hand the key to reg_kv_issuer.
298 * 2. If section is [Global], hand the key to reg_kv_global.
299 * 3. Otherwise ignore the line. Return 0 in the non-fatal cases.
300 */
301static int
302reg_kv(void *u, const char *section, const char *key, const char *val)
303{
304 brix_token_registry_t *reg = u;
305
306 if (strncasecmp(section, "Issuer ", 7) == 0) {
307 brix_token_issuer_t *is = reg_issuer_for(reg, section + 7);
308 if (is == NULL) {
309 return -1; /* too many issuers */
310 }
311 reg_kv_issuer(reg, is, key, val);
312 return 0;
313 }
314
315 if (eqi(section, "Global")) {
316 reg_kv_global(reg, key, val);
317 return 0;
318 }
319
320 return 0; /* unknown / no section */
321}

323/* brix_token_registry_load — parse + validate + load JWKS */ngx_int_t
324brix_token_registry_load(brix_token_registry_t *reg, const char *cfg_path,
325 uint32_t default_strategy, char *errbuf, size_t errlen)
326{
327 int i;
328
329 reg->default_strategy = default_strategy;
330
331 if (brix_ini_parse_file(cfg_path, reg_kv, reg, errbuf, errlen) != 0) {
332 return NGX_ERROR;
333 }
334 if (reg->count == 0) {
335 snprintf(errbuf, errlen, "%s: no [Issuer ...] section found", cfg_path);
336 return NGX_ERROR;
337 }
338
339 for (i = 0; i < reg->count; i++) {
340 brix_token_issuer_t *is = &reg->issuers[i];
341
342 if (is->issuer[0] == '\0') {
343 snprintf(errbuf, errlen,
344 "issuer \"%s\": missing issuer= URL", is->name);
345 return NGX_ERROR;
346 }
347 if (is->base_path_count == 0) {
348 snprintf(errbuf, errlen,
349 "issuer \"%s\": missing base_path", is->name);
350 return NGX_ERROR;
351 }
352 if (is->jwks_path[0] != '\0') {
353 struct stat st;
354
355 is->jwks_key_count = brix_jwks_load(reg->log, is->jwks_path,
356 is->jwks_keys, BRIX_MAX_JWKS_KEYS);
357 if (is->jwks_key_count <= 0) {
358 snprintf(errbuf, errlen,
359 "issuer \"%s\": no usable JWKS keys in %s",
360 is->name, is->jwks_path);
361 return NGX_ERROR;
362 }
363 if (stat(is->jwks_path, &st) == 0) {
364 is->jwks_mtime = st.st_mtime;
365 }
366 }
367 }
368 return NGX_OK;
369}

372brix_token_registry_build(ngx_conf_t *cf, const char *cfg_path,
373 uint32_t default_strategy, brix_token_registry_t **out)
374{
375 brix_token_registry_t *reg;
376 char err[256];
377 int i;
378
379 reg = ngx_pcalloc(cf->pool, sizeof(*reg));
380 if (reg == NULL) {
381 return NGX_ERROR;
382 }
383 reg->log = cf->log;
384
385 if (brix_token_registry_load(reg, cfg_path, default_strategy,

390 return NGX_ERROR;
391 }
392
393 /* Free each issuer's loaded EVP_PKEYs when the conf pool is destroyed. */
394 for (i = 0; i < reg->count; i++) {

398 {
399 return NGX_ERROR;
400 }
401 }
402
403 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
404 "brix_token: loaded %d issuer(s) from %s", reg->count, cfg_path);
405 *out = reg;
406 return NGX_OK;
407}
