1/*
2 * token/issuer_registry.c — SciTokens multi-issuer registry (see header).
3 *
4 * WHAT: Loads the upstream XRootD `scitokens.cfg` INI into a read-only table of
5 * issuers, each with its iss URL, audiences, base_path/restricted_path
6 * namespace scope, authorization strategy, and (optional) per-issuer
7 * JWKS. Provides exact-iss lookup and the base/restricted path gate.
8 * WHY: XRootD delegates authorization for a namespace subtree to an external
9 * issuer via this file; parsing the grammar verbatim lets an operator
10 * reuse their existing config unchanged (phase-59 W1, ADR-1).
11 * HOW: brix_ini_parse_file() drives reg_kv() per line; [Issuer N] keys fill
12 * an brix_token_issuer_t, [Global] keys fill shared audiences. After
13 * parsing, each issuer is validated (issuer + base_path required) and its
14 * JWKS loaded. Path scoping reuses brix_token_scope_path_matches().
15 * Unsupported keys are WARN-logged, never silently dropped (R4).
16 */

44/* brix_token_strategy_parse — "capability group mapping" → bits */uint32_t
45brix_token_strategy_parse(const char *value)
46{
47 char buf[128];
48 char *tok;
49 char *save = NULL;
50 uint32_t bits = 0;
51
52 copy_z(buf, sizeof(buf), value);
53 for (tok = strtok_r(buf, " ,", &save); tok != NULL;
54 tok = strtok_r(NULL, " ,", &save))
55 {
56 if (eqi(tok, "capability")) {
57 bits |= BRIX_AUTHZ_CAPABILITY;
58 } else if (eqi(tok, "group")) {
59 bits |= BRIX_AUTHZ_GROUP;
60 } else if (eqi(tok, "mapping")) {
61 bits |= BRIX_AUTHZ_MAPPING;
62 }
63 }
64 return bits;
65}

100/* reg_issuer_for — find-or-create the issuer for a section name */static brix_token_issuer_t *
101reg_issuer_for(brix_token_registry_t *reg, const char *name)
102{
103 brix_token_issuer_t *is;
104 int i;
105
106 for (i = 0; i < reg->count; i++) {
107 if (eqi(reg->issuers[i].name, name)) {
108 return &reg->issuers[i];
109 }
110 }
111 if (reg->count >= BRIX_TOKEN_MAX_ISSUERS) {
112 return NULL;
113 }
114 is = &reg->issuers[reg->count];
115 copy_z(is->name, sizeof(is->name), name);
116 copy_z(is->username_claim, sizeof(is->username_claim), "sub");
117 is->enabled = 1;
118 is->onmissing_fail = 1;
119 is->strategy = reg->default_strategy;
120 is->metric_bucket = reg->count;
121 reg->count++;
122 return is;
123}

125/* reg_kv — INI callback: dispatch one key line */static int
126reg_kv(void *u, const char *section, const char *key, const char *val)
127{
128 brix_token_registry_t *reg = u;
129
130 if (strncasecmp(section, "Issuer ", 7) == 0) {
131 brix_token_issuer_t *is = reg_issuer_for(reg, section + 7);
132 if (is == NULL) {
133 return -1; /* too many issuers */
134 }
135 if (eqi(key, "issuer")) {
136 copy_z(is->issuer, sizeof(is->issuer), val);
137 } else if (eqi(key, "base_path")) {
138 reg_add_list(is->base_paths, &is->base_path_count,
139 BRIX_TOKEN_MAX_BASEPATHS, val);
140 } else if (eqi(key, "restricted_path")) {
141 reg_add_list(is->restricted_paths, &is->restricted_path_count,
142 BRIX_TOKEN_MAX_BASEPATHS, val);
143 } else if (eqi(key, "audience") || eqi(key, "audience_json")) {
144 reg_add_aud(is->audiences, &is->audience_count,
145 BRIX_TOKEN_MAX_AUDIENCES, val);
146 } else if (eqi(key, "map_subject")) {
147 is->map_subject = parse_bool(val);
148 } else if (eqi(key, "username_claim")) {
149 copy_z(is->username_claim, sizeof(is->username_claim), val);
150 } else if (eqi(key, "groups_claim")) {
151 copy_z(is->groups_claim, sizeof(is->groups_claim), val);
152 } else if (eqi(key, "default_user")) {
153 copy_z(is->default_user, sizeof(is->default_user), val);
154 } else if (eqi(key, "name_mapfile")) {
155 copy_z(is->name_mapfile, sizeof(is->name_mapfile), val);
156 } else if (eqi(key, "jwks_file")) {
157 copy_z(is->jwks_path, sizeof(is->jwks_path), val);
158 } else if (eqi(key, "onmissing")) {
159 is->onmissing_fail = eqi(val, "fail");
160 } else if (eqi(key, "enabled")) {
161 is->enabled = parse_bool(val);
162 } else if (eqi(key, "authorization_strategy")) {
163 is->strategy = brix_token_strategy_parse(val);
164 } else {
165 ngx_log_error(NGX_LOG_WARN, reg->log, 0,
166 "scitokens: unsupported issuer key \"%s\" (ignored)", key);
167 }
168 return 0;
169 }
170
171 if (eqi(section, "Global")) {
172 if (eqi(key, "audience") || eqi(key, "audience_json")) {
173 reg_add_aud(reg->global_audiences, &reg->global_audience_count,
174 BRIX_TOKEN_MAX_AUDIENCES, val);
175 } else {
176 ngx_log_error(NGX_LOG_WARN, reg->log, 0,
177 "scitokens: unsupported [Global] key \"%s\" (ignored)", key);
178 }
179 return 0;
180 }
181
182 return 0; /* unknown / no section */
183}

185/* brix_token_registry_load — parse + validate + load JWKS */ngx_int_t
186brix_token_registry_load(brix_token_registry_t *reg, const char *cfg_path,
187 uint32_t default_strategy, char *errbuf, size_t errlen)
188{
189 int i;
190
191 reg->default_strategy = default_strategy;
192
193 if (brix_ini_parse_file(cfg_path, reg_kv, reg, errbuf, errlen) != 0) {
194 return NGX_ERROR;
195 }
196 if (reg->count == 0) {
197 snprintf(errbuf, errlen, "%s: no [Issuer ...] section found", cfg_path);
198 return NGX_ERROR;
199 }
200
201 for (i = 0; i < reg->count; i++) {
202 brix_token_issuer_t *is = &reg->issuers[i];
203
204 if (is->issuer[0] == '\0') {
205 snprintf(errbuf, errlen,
206 "issuer \"%s\": missing issuer= URL", is->name);
207 return NGX_ERROR;
208 }
209 if (is->base_path_count == 0) {
210 snprintf(errbuf, errlen,
211 "issuer \"%s\": missing base_path", is->name);
212 return NGX_ERROR;
213 }
214 if (is->jwks_path[0] != '\0') {
215 struct stat st;
216
217 is->jwks_key_count = brix_jwks_load(reg->log, is->jwks_path,
218 is->jwks_keys, BRIX_MAX_JWKS_KEYS);
219 if (is->jwks_key_count <= 0) {
220 snprintf(errbuf, errlen,
221 "issuer \"%s\": no usable JWKS keys in %s",
222 is->name, is->jwks_path);
223 return NGX_ERROR;
224 }
225 if (stat(is->jwks_path, &st) == 0) {
226 is->jwks_mtime = st.st_mtime;
227 }
228 }
229 }
230 return NGX_OK;
231}

233/* brix_token_registry_build — config-time allocate + load + cleanup */ngx_int_t
234brix_token_registry_build(ngx_conf_t *cf, const char *cfg_path,
235 uint32_t default_strategy, brix_token_registry_t **out)
236{
237 brix_token_registry_t *reg;
238 char err[256];
239 int i;
240
241 reg = ngx_pcalloc(cf->pool, sizeof(*reg));
242 if (reg == NULL) {
243 return NGX_ERROR;
244 }
245 reg->log = cf->log;
246
247 if (brix_token_registry_load(reg, cfg_path, default_strategy,
248 err, sizeof(err)) != NGX_OK)
249 {
250 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
251 "brix_token_config: %s", err);
252 return NGX_ERROR;
253 }
254
255 /* Free each issuer's loaded EVP_PKEYs when the conf pool is destroyed. */
256 for (i = 0; i < reg->count; i++) {
257 if (reg->issuers[i].jwks_key_count > 0
258 && brix_jwks_register_cleanup(cf->pool, reg->issuers[i].jwks_keys,
259 &reg->issuers[i].jwks_key_count) != NGX_OK)
260 {
261 return NGX_ERROR;
262 }
263 }
264
265 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
266 "brix_token: loaded %d issuer(s) from %s", reg->count, cfg_path);
267 *out = reg;
268 return NGX_OK;
269}

286/* brix_token_issuer_path_ok — base_path ∧ ¬restricted_path gate */int
287brix_token_issuer_path_ok(const brix_token_issuer_t *is,
288 const char *req_path)
289{

292

294 if (brix_token_scope_path_matches(is->base_paths[i], req_path)) {
295 under_base = 1;
296 break;
297 }
298 }
299 if (!under_base) {
300 return 0;
301 }
302 for (i = 0; i < is->restricted_path_count; i++) {
303 if (brix_token_scope_path_matches(is->restricted_paths[i], req_path)) {
304 return 0; /* explicitly carved out */
305 }
306 }
307 return 1;
