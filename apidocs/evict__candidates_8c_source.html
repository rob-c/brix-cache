15/* brix_cache_fs_usage — query filesystem occupancy via statvfs.
16 *
17 * Reads filesystem statistics from the cache root directory using
18 * statvfs() and computes total, available, used bytes plus occupancy
19 * as parts-per-million (ppm). The ppm value is compared against
20 * conf->cache_eviction_threshold to decide whether eviction runs.
21 *
22 * WHY: Cache eviction needs an accurate occupancy measurement that accounts
23 * for reserved blocks (root-only space) — f_bavail excludes those,
24 * while f_blocks counts all blocks including reserved. Using f_frsize
25 * as the block size multiplier gives correct byte totals on modern
26 * filesystems.
27 *
28 * HOW: statvfs(root, &vfs) → extract f_frsize/f_bsize/f_blocks/f_bavail
29 * → compute total=blocks*block_size, available=avail_blocks*block_size,
30 * → used = total - available, occupancy_ppm = (used/total)*1000000.
31 * Returns NGX_OK on success, NGX_ERROR if statvfs fails or total is zero
32 * (empty filesystem).
33 */
34static ngx_int_t brix_cache_add_candidate(brix_cache_evict_list_t *list,
35 const char *path, const struct stat *st);
36static int brix_cache_skip_name(const char *name);

38ngx_int_t
39brix_cache_fs_usage(const char *root, brix_cache_fs_usage_t *usage)
40{
41 struct statvfs vfs;
42 uint64_t block_size;
43 uint64_t blocks;
44 uint64_t avail_blocks;
45
46 if (statvfs(root, &vfs) != 0 || vfs.f_blocks == 0) {
47 return NGX_ERROR;
48 }
49
50 block_size = vfs.f_frsize ? (uint64_t) vfs.f_frsize
51 : (uint64_t) vfs.f_bsize;
52 blocks = (uint64_t) vfs.f_blocks;
53 avail_blocks = (uint64_t) vfs.f_bavail;
54
55 usage->total = blocks * block_size;
56 usage->available = avail_blocks * block_size;
57 usage->used = (blocks - avail_blocks) * block_size;
58 usage->occupancy_ppm = (ngx_uint_t)
59 (((long double) usage->used * 1000000.0L) / (long double) usage->total);
60
61 return NGX_OK;
62}

89/*
90 * brix_cache_try_evict_lock — acquire an exclusive eviction lock via
91 * a sentinel file created with O_CREAT | O_EXCL.
92 *
93 * The lock prevents two nginx worker processes from running concurrent cache
94 * evictions, which could lead to both workers evicting the same file and
95 * double-counting bytes.
96 *
97 * Stale lock handling: if the lock file is older than conf->cache_lock_timeout
98 * seconds, it is treated as abandoned (worker crashed) and removed.
99 *
100 * Returns:
101 * NGX_OK — lock acquired (caller must call brix_cache_evict_unlock).
102 * NGX_DECLINED — another worker holds a fresh lock; skip eviction.
103 * NGX_ERROR — unexpected OS error.
104 */
105ngx_int_t
106brix_cache_try_evict_lock(ngx_stream_brix_srv_conf_t *conf,
107 char *lock_path, size_t lock_pathsz, ngx_log_t *log)
108{
109 int n;
110 int fd;
111 struct stat st;
112 time_t now;
113 /* §14a: the lock sentinel lives in the PHYSICAL cache root (tier-aware — the
114 * posix cache_store dir for a tier cache, else cache_root). A pure tier cache
115 * advertises cache_root "/", and creating "/.ngx-xrootd-evict-lock" fails EACCES,
116 * silently disabling the reaper — the lock must land in the writable store dir. */
117 const char *root = brix_cache_state_root(conf);
118
119 if (root == NULL) {
120 errno = EINVAL;
121 return NGX_ERROR;
122 }
123 n = snprintf(lock_path, lock_pathsz, "%s/%s", root,
124 BRIX_CACHE_EVICT_LOCK_NAME);
125 if (n < 0 || (size_t) n >= lock_pathsz) {
126 errno = ENAMETOOLONG;
127 return NGX_ERROR;
128 }
129
130 fd = open(lock_path, O_CREAT | O_EXCL | O_WRONLY | O_NOCTTY | O_CLOEXEC, 0600);
131 if (fd >= 0) {
132 close(fd);
133 return NGX_OK;
134 }
135
136 if (errno != EEXIST) {
137 ngx_log_error(NGX_LOG_WARN, log, errno,
138 "brix: cache eviction lock create failed \"%s\"",
139 lock_path);
140 return NGX_ERROR;
141 }
142
143 if (stat(lock_path, &st) != 0) {
144 return NGX_DECLINED;
145 }
146
147 now = time(NULL);
148 if (now <= st.st_mtime
149 || now - st.st_mtime <= conf->cache_lock_timeout)
150 {
151 return NGX_DECLINED;
152 }
153
154 ngx_log_error(NGX_LOG_WARN, log, 0,
155 "brix: removing stale cache eviction lock \"%s\"",
156 lock_path);
157 if (unlink(lock_path) != 0 && errno != ENOENT) {
158 return NGX_DECLINED;
159 }
160
161 fd = open(lock_path, O_CREAT | O_EXCL | O_WRONLY | O_NOCTTY | O_CLOEXEC, 0600);
162 if (fd < 0) {
163 return NGX_DECLINED;
164 }
165
166 close(fd);
167 return NGX_OK;
168}

190/*
191 * brix_cache_skip_name — filter special names during cache eviction scan.
192 *
193 * WHAT: Returns 1 (skip) for names that must not be evicted or processed:
194 * - "." and ".." (parent directory entries)
195 * - The eviction lock sentinel file name
196 * - Files ending with the partial-cache suffix (*.part)
197 * - Files ending with the lock-file suffix (*.lock)
198 *
199 * WHY: Prevents accidental deletion of critical infrastructure files during
200 * cache eviction. Partial cache files are in-flight transfers that must
201 * not be interrupted; lock files prevent concurrent evictions.
202 *
203 * HOW: Sequential strcmp checks against known special names, then suffix
204 * comparison for *.part and *.lock patterns. Returns 0 (include) if none
205 * match.
206 */
207static int
208brix_cache_skip_name(const char *name)
209{
210 size_t name_len;
211 size_t suffix_len;
212
213 if (name[0] == '\0'
214 || strcmp(name, ".") == 0
215 || strcmp(name, "..") == 0
216 || strcmp(name, BRIX_CACHE_EVICT_LOCK_NAME) == 0)
217 {
218 return 1;
219 }
220
221 name_len = strlen(name);
222
223 suffix_len = sizeof(BRIX_CACHE_PART_SUFFIX) - 1;
224 if (name_len >= suffix_len
225 && strcmp(name + name_len - suffix_len, BRIX_CACHE_PART_SUFFIX) == 0)
226 {
227 return 1;
228 }
229
230 suffix_len = sizeof(BRIX_CACHE_LOCK_SUFFIX) - 1;
231 if (name_len >= suffix_len
232 && strcmp(name + name_len - suffix_len, BRIX_CACHE_LOCK_SUFFIX) == 0)
233 {
234 return 1;
235 }
236
237 suffix_len = sizeof(".meta") - 1;
238 if (name_len >= suffix_len
239 && strcmp(name + name_len - suffix_len, ".meta") == 0)
240 {
241 return 1;
242 }
243
244 /* .cinfo sidecars are cache STATE, not cache data — the dirty/present-bitmap
245 * record. Evicting one orphans its data file's write-back-dirty protection
246 * (the next pass would no longer see the file as dirty and could reap it) and
247 * loses the present-block bitmap. Never a candidate; it is removed only with
248 * its data file by brix_cache_evict_one(). */
249 suffix_len = sizeof(".cinfo") - 1;
250 if (name_len >= suffix_len
251 && strcmp(name + name_len - suffix_len, ".cinfo") == 0)
252 {
253 return 1;
254 }
255
256 return 0;
257}

259static ngx_int_t
260brix_cache_add_candidate(brix_cache_evict_list_t *list, const char *path,
261 const struct stat *st)
262{
263 brix_cache_evict_candidate_t *elts;
264 char *evicted;
265 char *copy;
266 size_t len;
267 size_t new_cap;
268
269 if (list->protect_path != NULL
270 && strcmp(path, list->protect_path) == 0)
271 {
272 return NGX_OK;
273 }
274
275 if (list->nelts == list->cap) {
276 size_t elts_sz;
277
278 new_cap = list->cap ? list->cap * 2 : 128;
279
280 /* Phase 27 F9/W1: bound the candidate set and use overflow-checked size
281 * math so a pathological scan cannot wrap new_cap*sizeof into a tiny
282 * allocation that the loop then overruns. */
283 /* phase79-fp: "leak of copy" — copy's ownership transfers into
284 * list->elts[].path below and is released by brix_cache_free_candidates;
285 * analyzer does not track ownership through the realloc'd container */
286 if (new_cap > BRIX_EVICT_MAX_CANDIDATES
287 || brix_size_mul(new_cap, sizeof(list->elts[0]), &elts_sz)
288 != NGX_OK)
289 {
290 return NGX_ERROR;
291 }
292
293 /*
294 * Grow evicted before elts: if elts realloc succeeds but evicted
295 * fails we would lose the original elts pointer (realloc freed it).
296 * Growing the smaller array first avoids that dangling-pointer hazard.
297 */
298 evicted = realloc(list->evicted, new_cap * sizeof(char));
299 if (evicted == NULL) {
300 return NGX_ERROR;
301 }
302 list->evicted = evicted;
303
304 elts = realloc(list->elts, elts_sz);
305 if (elts == NULL) {
306 /* evicted grew but elts didn't; cap stays at old value so the
307 * next add call triggers a fresh grow attempt. */
308 return NGX_ERROR;
309 }
310 list->elts = elts;
311 list->cap = new_cap;
312 }
313
314 len = strlen(path);
315 if (len == (size_t)-1) {
316 return NGX_ERROR;
317 }
318 copy = malloc(len + 1);
319 if (copy == NULL) {
320 return NGX_ERROR;
321 }
322 memcpy(copy, path, len + 1);
323
324 list->elts[list->nelts].path = copy;
325 list->elts[list->nelts].size = st->st_size;
326 list->elts[list->nelts].atime = st->st_atime;
327 list->elts[list->nelts].mtime = st->st_mtime;
328 list->evicted[list->nelts] = 0;
329 list->nelts++;
330
331 return NGX_OK;
332}
333/* brix_cache_add_candidate — append a candidate to the eviction list.
334 *
335 * Adds a file path and its stat metadata (size, atime, mtime) to the
336 * eviction candidate list. First checks whether this is the protected
337 * path (the lock sentinel itself) and skips it immediately.
338 *
339 * WHY: The candidate list grows dynamically via realloc — each new entry
340 * needs both a strdup'd path string and a struct slot. Growing the
341 * smaller evicted[] array before elts[] avoids the dangling-pointer
342 * hazard where elts realloc frees the original but evicted realloc fails.
343 *
344 * HOW: Check protect_path → if list full, grow arrays (evicted first, then
345 * elts) → strdup path → populate candidate struct fields → increment nelts.
346 * Returns NGX_OK on success, NGX_ERROR if memory allocation fails or
347 * strlen returns -1 (invalid path).
348 */
349/* evict_collect_visit — cstore_scan visitor: one call per cached object with its
350 * store stat (`stx`) and cinfo (`ci`, NULL for an orphan/partial). Applies the
351 * same policy the raw scan did — skip control/sidecar names, never collect a key
352 * with un-flushed local writes (dirty .cinfo at the state path), add the rest as
353 * eviction candidates keyed by cache_root + key. Returns NGX_OK to continue the
354 * scan, NGX_ERROR to stop (candidate-array growth failure). */
355static ngx_int_t
356evict_collect_visit(const char *key, const brix_cache_cinfo_t *ci,
357 const brix_sd_stat_t *stx, void *ctx)
358{
359 brix_cache_evict_list_t *list = ctx;
360 const char *name;
361 char childpath[PATH_MAX];
362 char sidecar[PATH_MAX];
363 const char *base;
364 uint64_t dlo, dhi, dsince;
365 struct stat synth;
366 int n;
367
368 (void) ci; /* dirty guard reads the state-root sidecar for exact parity */
369
370 name = strrchr(key, '/');
371 name = (name != NULL) ? name + 1 : key;
372 if (brix_cache_skip_name(name)) {
373 return NGX_OK;
374 }
375
376 n = snprintf(childpath, sizeof(childpath), "%s%s", list->cache_root, key);
377 if (n < 0 || (size_t) n >= sizeof(childpath)) {
378 return NGX_OK; /* unrepresentable — skip this key */
379 }
380
381 /* Never evict a file with un-flushed local writes (dirty .cinfo). The .cinfo
382 * lives at the state path (== childpath for a co-located cache). */
383 base = childpath;
384 if (brix_cache_sidecar_path(list->cache_root, list->state_root,
385 childpath, sidecar, sizeof(sidecar)) == 0)
386 {
387 base = sidecar;
388 }
389 if (brix_cache_cinfo_dirty_extent(base, &dlo, &dhi, &dsince) == NGX_OK) {
390 return NGX_OK; /* dirty — keep */
391 }
392
393 /* Synthesize a stat for the candidate: driver stat has no atime, so the LRU
394 * score uses mtime (a deterministic, monotone proxy; record_access bumps both
395 * the cache-file mtime and the .cinfo). */
396 ngx_memzero(&synth, sizeof(synth));
397 synth.st_size = stx->size;
398 synth.st_mtime = stx->mtime;
399 synth.st_atime = stx->mtime;
400 return (brix_cache_add_candidate(list, childpath, &synth) == NGX_OK)
401 ? NGX_OK : NGX_ERROR;
402}

430/* brix_cache_candidate_cmp — sort comparator for eviction candidates.
431 *
432 * qsort-compatible comparison function that orders candidates by
433 * oldest access time first, then oldest modification time as tiebreaker,
434 * then alphabetical path name as final tiebreaker.
435 *
436 * WHY: Eviction policy is "oldest-first" — files accessed longest ago are
437 * least likely to be needed again. atime is the primary sort key because
438 * it reflects actual usage recency; mtime breaks ties when two files
439 * have identical access times; path name provides deterministic ordering.
440 *
441 * HOW: Compare atime (ascending) → if equal compare mtime (ascending)
442 * → if equal strcmp on path strings. Returns -1/0/+1 for qsort semantics.
443 */
444
445int
446brix_cache_candidate_cmp(const void *a, const void *b)
447{
448 const brix_cache_evict_candidate_t *ca = a;
449 const brix_cache_evict_candidate_t *cb = b;
450
451 if (ca->atime < cb->atime) { return -1; }
452 if (ca->atime > cb->atime) { return 1; }
453 if (ca->mtime < cb->mtime) { return -1; }
454 if (ca->mtime > cb->mtime) { return 1; }
455 return strcmp(ca->path, cb->path);
456}
