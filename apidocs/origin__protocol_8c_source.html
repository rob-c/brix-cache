29/* Extract the "gsi" protocol's parameter substring from a login advert that may
30 * carry several "&P=<proto>,<parms>" entries (e.g. "&P=ztn,v:10000&P=gsi,v:10600,
31 * c:ssl,ca:HASH"). Returns a pointer INTO `parms` just past "gsi," (the v:/c:/ca:
32 * list brix_gsi_parse_parms wants), or NULL when gsi is not advertised. */
33static const char *
34cache_origin_gsi_parms(const char *parms, size_t plen)
35{
36 static const char needle[] = "gsi,";
37 size_t i;
38
39 if (parms == NULL || plen < sizeof(needle) - 1) {
40 return NULL;
41 }
42 for (i = 0; i + (sizeof(needle) - 1) <= plen; i++) {
43 if (ngx_strncmp(parms + i, needle, sizeof(needle) - 1) == 0) {
44 return parms + i + (sizeof(needle) - 1);
45 }
46 }
47 return NULL;
48}

51 * raw TCP/TLS socket: ClientInitHandShake → kXR_protocol negotiation (a

53 * (user 'xrd', capver kXR_ver005, streamid[1]=1). When the origin demands auth

139 /* A kXR_ok login on an AUTHENTICATED origin still carries an auth advert:
140 * body = sessid(16) + "&P=<proto>,..." (anonymous origins send only the 16-byte
141 * sessid). So a kXR_ok with a "&P=" parameter block means the session is NOT yet
142 * authenticated — present the configured bearer via ztn (§14/C-3). kXR_authmore
143 * is the mid-protocol variant; handle it the same way. */
144 if ((status == kXR_ok || status == kXR_authmore)
145 && dlen > BRIX_SESSION_ID_LEN)

148 size_t plen = dlen - BRIX_SESSION_ID_LEN;

152 int has_sss = (ngx_strnstr((u_char *) parms, "sss", plen) != NULL);
153 const char *gp = cache_origin_gsi_parms((const char *) parms, plen);
154 char gsi_parms[256];

158 * at the next "&P=" entry so a co-advertised ztn block isn't mis-parsed. */

161 size_t end = (size_t) ((const char *) parms + plen - gp);
162 size_t i;

165 if (i >= sizeof(gsi_parms)) { i = sizeof(gsi_parms) - 1; }
166 ngx_memcpy(gsi_parms, gp, i);

169 }

174 return brix_cache_origin_auth_ztn(t, oc,
175 &t->conf->cache_origin_bearer);
176 }
177 if (has_gsi && t->conf->cache_origin_x509_proxy.len > 0) {
178 return brix_cache_origin_auth_gsi(t, oc, gsi_parms,
179 (const char *) t->conf->cache_origin_x509_proxy.data);

182 return brix_cache_origin_auth_sss(t, oc,
183 (const char *) t->conf->cache_origin_sss_keytab.data);

185 brix_cache_set_error(t, kXR_AuthFailed, 0,

191 return -1;
192 }

194 }
195 free(body);
196
197 if (status == kXR_authmore) {
198 if (t->conf->cache_origin_bearer.len > 0) {
199 return brix_cache_origin_auth_ztn(t, oc, &t->conf->cache_origin_bearer);
200 }
201 brix_cache_set_error(t, kXR_AuthFailed, 0,
202 "cache origin requires authentication");
203 return -1;
204 }
205 if (status != kXR_ok) {
206 brix_cache_set_error(t, kXR_ServerError, 0,
207 "cache origin anonymous login failed");
208 return -1;
209 }
210
211 return 0;
212}

315/* brix_cache_origin_query_checksum — ask the origin for its stored digest of
316 * t->clean_path (path-based kXR_query/kXR_Qcksum), returning "<algo> <hex>" split
317 * into the caller buffers. Checksum-on-fill (verify.c) validates downloaded bytes
318 * against this before publishing. BEST-EFFORT: an origin with no checksum or a
319 * wire hiccup must NOT fail an otherwise-complete fill (data is already on disk) —
320 * on ANY failure it restores t's error state and returns 0 with alg_out emptied,
321 * so the caller treats it as "no origin digest" and the verify policy decides. */
322int
323brix_cache_origin_query_checksum(brix_cache_fill_t *t,
324 brix_cache_origin_conn_t *oc, char *alg_out, size_t alg_sz,
325 char *hex_out, size_t hex_sz)
326{
327 size_t pathlen, total;
328 u_char *buf;
329 ClientQueryRequest *req;
330 uint16_t status;
331 uint32_t dlen;
332 u_char *body;
333 char *sp;
334 int saved_result, saved_xrd;
335
336 if (alg_sz > 0) {
337 alg_out[0] = '\0';
338 }
339 if (hex_sz > 0) {
340 hex_out[0] = '\0';
341 }
342
343 /* The download already succeeded; never let a checksum-query failure leak an
344 * error onto the task. Snapshot and restore the error triple. */
345 saved_result = t->result;
346 saved_xrd = t->xrd_error;
347
348 pathlen = strlen(t->clean_path);
349 total = sizeof(ClientQueryRequest) + pathlen;
350
351 buf = malloc(total);
352 if (buf == NULL) {
353 return 0; /* best-effort: skip verification on OOM */
354 }
355
356 ngx_memzero(buf, total);
357 req = (ClientQueryRequest *) buf;
358 req->streamid[1] = 6; /* unused stream slot */
359 req->requestid = htons(kXR_query);
360 {
361 xrdw_query_req_t b = { .infotype = kXR_Qcksum }; /* fhandle 0 ⇒ path-based */
362 xrdw_query_req_pack(&b, ((ClientRequestHdr *) buf)->body);
363 }
364 req->dlen = htonl((kXR_int32) pathlen);
365 ngx_memcpy(buf + sizeof(*req), t->clean_path, pathlen);
366
367 if (brix_cache_io_send(oc, buf, total) != 0) {
368 free(buf);
369 return 0;
370 }
371 free(buf);
372
373 body = NULL;
374 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 512) != 0) {
375 t->result = saved_result;
376 t->xrd_error = saved_xrd;
377 return 0;
378 }
379
380 if (status != kXR_ok || body == NULL || dlen == 0) {
381 free(body); /* origin has no checksum */
382 return 0;
383 }
384
385 /* body is NUL-terminated "<algo> <hexvalue>". */
386 sp = strchr((char *) body, ' ');
387 if (sp != NULL) {
388 size_t an = (size_t) (sp - (char *) body);
389 char *hv = sp + 1;
390 char *end = hv + strlen(hv);
391 size_t hn;
392
393 while (end > hv && (end[-1] == '\n' || end[-1] == '\r'
394 || end[-1] == ' ' || end[-1] == '\t')) {
395 end--;
396 }
397 hn = (size_t) (end - hv);
398
399 if (an > 0 && an < alg_sz && hn > 0 && hn < hex_sz) {
400 ngx_memcpy(alg_out, body, an);
401 alg_out[an] = '\0';
402 ngx_memcpy(hex_out, hv, hn);
403 hex_out[hn] = '\0';
404 }
405 }
406
407 free(body);
408 return 0;
409}

412/* brix_cache_origin_read_chunk — kXR_read at (offset, rlen), writing each reply
413 * payload to the sink via brix_cache_sink_pwrite and looping over kXR_oksofar
414 * until the final kXR_ok. dlen is bounded (<= want, accumulated *got within
415 * request bounds) to prevent overflow. Sets *got; returns 0 / -1. */
416int
417brix_cache_origin_read_chunk(brix_cache_fill_t *t,
418 brix_cache_origin_conn_t *oc, const u_char fhandle[XRD_FHANDLE_LEN],
419 brix_cache_sink_t *sink, uint64_t read_off, uint64_t dst_off,
420 size_t want, size_t *got)
421{
422 ClientReadRequest req;
423 uint16_t status;
424 uint32_t dlen;
425 u_char *body;
426
427 *got = 0;
428
429 ngx_memzero(&req, sizeof(req));
430 req.streamid[1] = 3;
431 req.requestid = htons(kXR_read);
432 ngx_memcpy(req.fhandle, fhandle, XRD_FHANDLE_LEN);
433 req.offset = (kXR_int64) htobe64(read_off);
434 req.rlen = htonl((kXR_int32) want);
435 req.dlen = 0;
436
437 if (brix_cache_io_send(oc, &req, sizeof(req)) != 0) {
438 brix_cache_set_error(t, kXR_ServerError, errno,
439 "cache origin read write failed");
440 return -1;
441 }
442
443 for (;;) {
444 body = NULL;
445 if (brix_cache_read_response(t, oc, &status, &body, &dlen,
446 BRIX_CACHE_FETCH_CHUNK) != 0) {
447 return -1;
448 }
449
450 if (status == kXR_error) {
451 brix_cache_set_origin_error(t, body, dlen,
452 "cache origin read failed");
453 free(body);
454 return -1;
455 }
456
457 if (status != kXR_ok && status != kXR_oksofar) {
458 free(body);
459 brix_cache_set_error(t, kXR_ServerError, 0,
460 "cache origin read returned invalid status");
461 return -1;
462 }
463
464 if ((size_t) dlen > want || *got > want - (size_t) dlen) {
465 free(body);
466 brix_cache_set_error(t, kXR_ServerError, 0,
467 "cache origin read returned too much data");
468 return -1;
469 }
470
471 if (dlen > 0) {
472 /* Write at dst_off + bytes already written this call (*got). dst_off
473 * is the caller's WRITE base, decoupled from the origin READ offset:
474 * the whole-file fetch passes dst_off==read_off (absolute), a slice
475 * fill passes a 0-relative base. Using *got alone restarts at 0 each
476 * 1 MiB chunk, so multi-chunk whole-file fetches overwrote at offset 0
477 * (corrupting any file > BRIX_CACHE_FETCH_CHUNK → adler32 mismatch). */
478 if (brix_cache_sink_pwrite(sink, body, dlen,
479 (off_t) (dst_off + *got)) != 0) {
480 free(body);
481 brix_cache_set_syserror(t, kXR_IOError,
482 "cache file write failed");
483 return -1;
484 }
485 *got += (size_t) dlen;
486 }
487
488 free(body);
489
490 if (status == kXR_ok) {
491 return 0;
492 }
493 }
494}
