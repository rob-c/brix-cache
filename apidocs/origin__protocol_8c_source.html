29/* Extract the "gsi" protocol's parameter substring from a login advert that may
30 * carry several "&P=<proto>,<parms>" entries (e.g. "&P=ztn,v:10000&P=gsi,v:10600,
31 * c:ssl,ca:HASH"). Returns a pointer INTO `parms` just past "gsi," (the v:/c:/ca:
32 * list brix_gsi_parse_parms wants), or NULL when gsi is not advertised. */
33static const char *
34cache_origin_gsi_parms(const char *parms, size_t plen)
35{
36 static const char needle[] = "gsi,";
37 size_t i;
38
39 if (parms == NULL || plen < sizeof(needle) - 1) {
40 return NULL;
41 }
42 for (i = 0; i + (sizeof(needle) - 1) <= plen; i++) {
43 if (ngx_strncmp(parms + i, needle, sizeof(needle) - 1) == 0) {
44 return parms + i + (sizeof(needle) - 1);
45 }
46 }
47 return NULL;
48}

50/* origin_frame_t — one decoded origin reply (status + owned body). WHY: the
51 * bootstrap wire steps each read exactly one frame; bundling the status/body/dlen
52 * triple keeps the step helpers under the 5-parameter cap and makes body
53 * ownership explicit (whoever holds the struct frees fr->body). */
54typedef struct {
55 uint16_t status;
56 uint32_t dlen;
57 u_char *body;
58} origin_frame_t;

60/* origin_auth_advert_t — parsed "&P=<proto>,..." login advert. WHY: the login
61 * reply's protocol list drives the credential dispatch ladder; parsing it into
62 * flags + a copied gsi parameter string decouples the (freed) reply body from
63 * the auth decision. gsi_parms is a stable copy because the advert body is
64 * released before any auth round-trip starts. */
65typedef struct {
66 int needs_auth; /* advert carries "&P=..." ⇒ session not authenticated */
67 int has_ztn;
68 int has_gsi;
69 int has_sss;
70 char gsi_parms[256]; /* gsi v:/c:/ca: list, NUL-terminated */
71} origin_auth_advert_t;

73/* origin_expect_frame — read one origin reply into fr and require kXR_ok. WHAT:
74 * shared reply validator for the bootstrap wire steps. HOW: on transport/timeout
75 * failure brix_cache_read_response has already set the task error; on a non-ok
76 * status the body is freed and fail_msg becomes a kXR_ServerError. On success
77 * fr->body is the caller's to free. Returns 0 / -1. */
78static int
79origin_expect_frame(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
80 uint32_t max_dlen, const char *fail_msg, origin_frame_t *fr)
81{
82 fr->body = NULL;
83 if (brix_cache_read_response(t, oc, &fr->status, &fr->body,
84 &fr->dlen, max_dlen) != 0) {
85 return -1;
86 }
87 if (fr->status != kXR_ok) {
88 free(fr->body);
89 fr->body = NULL;
90 brix_cache_set_error(t, kXR_ServerError, 0, fail_msg);
91 return -1;
92 }
93 return 0;
94}

120/* origin_bs_protocol — bootstrap step 2: kXR_protocol negotiation, with the
121 * cleartext-handshake-then-TLS-upgrade a stock XRootD `roots://` origin demands.
122 *
123 * WHY: a TLS-for-ztn origin answers the CLEARTEXT kXR_protocol request with a
124 * kXR_gotoTLS advert and expects the client to upgrade THIS fd to TLS before
125 * kXR_login/auth — an immediate SSL_connect at byte 0 (the old behaviour) never
126 * reaches this exchange, so the origin was unreachable. When brix_cache_origin_tls
127 * is set we advertise kXR_ableTLS so the origin knows it may request the upgrade.
128 * HOW: send the protocol request on the connector streamid (advertising ableTLS
129 * when configured), inspect the reply flags, and:
130 * - gotoTLS + tls on → brix_cache_origin_tls_upgrade (every later frame rides
131 * TLS via io.c once oc->ssl is set)
132 * - gotoTLS + tls off → refuse (kXR_TLSRequired) rather than a mid-session surprise
133 * - no gotoTLS → stay cleartext
134 * Returns 0 / -1 (task error set). */
135static int
136origin_bs_protocol(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc)
137{
138 ClientProtocolRequest pr;
139 origin_frame_t fr;
140 static const uint8_t sid[2] = { 0, 1 }; /* cache-origin connector streamid */
141 uint8_t cap;
142 uint32_t flags = 0;
143
144 /* Advertise TLS capability so a TLS-requiring origin will answer with
145 * kXR_gotoTLS; with tls off we send no flag and behaviour is unchanged. */
146 cap = t->conf->cache_origin_tls ? (uint8_t) kXR_ableTLS : 0;
147 xrd_pack_protocol_request(&pr, sid, cap);
148
149 if (brix_cache_io_send(oc, &pr, sizeof(pr)) != 0) {
150 brix_cache_set_error(t, kXR_ServerError, errno,
151 "cache origin protocol write failed");
152 return -1;
153 }
154 if (origin_expect_frame(t, oc, sizeof(ServerProtocolBody),
155 "cache origin protocol negotiation failed",
156 &fr) != 0) {
157 return -1;
158 }
159
160 if (fr.dlen >= sizeof(ServerProtocolBody)) {
161 ServerProtocolBody *pb = (ServerProtocolBody *) fr.body;
162
163 flags = (uint32_t) ntohl(pb->flags);
164 }
165 free(fr.body);
166
167 if (flags & kXR_gotoTLS) {
168 if (!t->conf->cache_origin_tls) {
169 brix_cache_set_error(t, kXR_TLSRequired, 0,
170 "cache origin requires TLS; enable brix_cache_origin_tls");
171 return -1;
172 }
173 /* In-place upgrade of the connected fd BEFORE kXR_login/auth: the origin
174 * cert is verified against the configured origin CA (synth->trusted_ca)
175 * with hostname binding inside brix_cache_origin_tls_upgrade. */
176 if (brix_cache_origin_tls_upgrade(t, oc,
177 &t->conf->cache_origin_host) != 0)
178 {
179 return -1; /* upgrade helper set t's error */
180 }
181 }
182 return 0;
183}

185/* origin_bs_login — bootstrap step 3: anonymous kXR_login (user 'xrd', capver
186 * kXR_ver005). WHAT: sends the login and reads the reply into fr WITHOUT
187 * validating the status — kXR_authmore and an auth-advert-bearing kXR_ok are
188 * both legitimate here, so the caller owns the status decision (and fr->body).
189 * Returns 0 with fr populated, -1 on send/read failure (task error set). */

193{

195 static const uint8_t sid[2] = { 0, 1 }; /* cache-origin connector streamid */
196

208 return -1;
209 }
210 return 0;
211}

213/* origin_bs_parse_advert — pure parse of the login auth advert into ad. WHAT:
214 * detects "&P=..." (needs_auth), the ztn/sss protocol names, and copies the gsi
215 * v:/c:/ca: list out of the (about-to-be-freed) body, stopping at the next "&P="
216 * entry so a co-advertised ztn block isn't mis-parsed. No I/O, no task state. */
217static void
218origin_bs_parse_advert(const u_char *parms, size_t plen,
219 origin_auth_advert_t *ad)
220{
221 const char *gp;
222
223 ad->needs_auth = (ngx_strlchr((u_char *) parms,
224 (u_char *) parms + plen, '=') != NULL);
225 ad->has_ztn = (ngx_strnstr((u_char *) parms, "ztn", plen) != NULL);
226 ad->has_sss = (ngx_strnstr((u_char *) parms, "sss", plen) != NULL);
227 ad->has_gsi = 0;
228 ad->gsi_parms[0] = '\0';
229
230 gp = cache_origin_gsi_parms((const char *) parms, plen);
231 if (gp != NULL) {
232 const char *amp = gp;
233 size_t end = (size_t) ((const char *) parms + plen - gp);
234 size_t i;
235
236 for (i = 0; i < end && amp[i] != '&'; i++) { /* find terminator */ }
237 if (i >= sizeof(ad->gsi_parms)) { i = sizeof(ad->gsi_parms) - 1; }
238 ngx_memcpy(ad->gsi_parms, gp, i);
239 ad->gsi_parms[i] = '\0';
240 ad->has_gsi = 1;
241 }
242}

265/* origin_bs_auth_dispatch — credential ladder for an auth-demanding advert.
266 * WHY the ordering: per-user overrides WIN over every static service credential
267 * (the session must carry the user's identity, never the service's), and x509 vs
268 * bearer are mutually exclusive — at most one will be non-empty. A per-user
269 * credential must NEVER fall back to a service credential: the operator
270 * provisioned it for a reason, and silent fallback would change the presented
271 * identity. Returns the auth round-trip result, or -1 with kXR_AuthFailed when
272 * no advertised protocol matches a configured credential. */
273static int
274origin_bs_auth_dispatch(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
275 const origin_auth_advert_t *ad)
276{
277 if (t->cred_x509_proxy[0] != '\0') {
278 if (ad->has_gsi) {
279 return brix_cache_origin_auth_gsi(t, oc, ad->gsi_parms,
280 t->cred_x509_proxy);
281 }
282 brix_cache_set_error(t, kXR_AuthFailed, 0,
283 "origin does not advertise gsi for the per-user credential");
284 return -1;
285 }
286 if (t->cred_bearer[0] != '\0') {
287 if (ad->has_ztn) {
288 ngx_str_t bt = {
289 ngx_strlen(t->cred_bearer),
290 (u_char *) t->cred_bearer
291 };
292 return brix_cache_origin_auth_ztn(t, oc, &bt);
293 }
294 brix_cache_set_error(t, kXR_AuthFailed, 0,
295 "origin does not advertise ztn for the per-user bearer credential");
296 return -1;
297 }
298 if (ad->has_ztn && t->conf->cache_origin_bearer.len > 0) {
299 return brix_cache_origin_auth_ztn(t, oc,
300 &t->conf->cache_origin_bearer);
301 }
302 if (ad->has_gsi && t->conf->cache_origin_x509_proxy.len > 0) {
303 return brix_cache_origin_auth_gsi(t, oc, ad->gsi_parms,
304 (const char *) t->conf->cache_origin_x509_proxy.data);
305 }
306 if (ad->has_sss && t->conf->cache_origin_sss_keytab.len > 0) {
307 return brix_cache_origin_auth_sss(t, oc,
308 (const char *) t->conf->cache_origin_sss_keytab.data);
309 }
310 brix_cache_set_error(t, kXR_AuthFailed, 0, origin_bs_auth_fail_msg(t));
311 return -1;
312}

341/* brix_cache_origin_bootstrap — three-phase XRootD connection bootstrap over a
342 * CLEARTEXT TCP socket: ClientInitHandShake → kXR_protocol negotiation → anonymous
343 * kXR_login (user 'xrd', capver kXR_ver005, streamid[1]=1). A stock XRootD
344 * `roots://` origin answers the cleartext kXR_protocol with a kXR_gotoTLS advert;
345 * origin_bs_protocol then upgrades THIS fd to TLS in place (verified against the
346 * origin CA) BEFORE login/auth, so the ztn/GSI credential exchange rides the
347 * encrypted channel. When the origin demands auth
348 * (kXR_authmore) and a bearer token is configured, a ztn kXR_auth completes the
349 * session. Every cache fill needs a valid session before reading. HOW: one
350 * static helper per wire step (origin_bs_handshake/_protocol/_login), then the
351 * login status decides between the advert-driven credential dispatch and the
352 * advert-less kXR_authmore fallback. Returns 0 on success, -1 on any phase
353 * failure. */
354int
355brix_cache_origin_bootstrap(brix_cache_fill_t *t,
356 brix_cache_origin_conn_t *oc)
357{
358 origin_frame_t fr;
359
360 if (origin_bs_handshake(t, oc) != 0) {
361 return -1;
362 }
363 if (origin_bs_protocol(t, oc) != 0) {
364 return -1;
365 }
366 if (origin_bs_login(t, oc, &fr) != 0) {
367 return -1;
368 }
369
370 /* A kXR_ok login on an AUTHENTICATED origin still carries an auth advert:
371 * body = sessid(16) + "&P=<proto>,..." (anonymous origins send only the 16-byte
372 * sessid). So a kXR_ok with a "&P=" parameter block means the session is NOT yet
373 * authenticated — present the configured bearer via ztn (§14/C-3). kXR_authmore
374 * is the mid-protocol variant; handle it the same way. */
375 if ((fr.status == kXR_ok || fr.status == kXR_authmore)
376 && fr.dlen > BRIX_SESSION_ID_LEN)
377 {
378 origin_auth_advert_t ad;
379
380 origin_bs_parse_advert(fr.body + BRIX_SESSION_ID_LEN,
381 fr.dlen - BRIX_SESSION_ID_LEN, &ad);
382 free(fr.body);
383
384 if (ad.needs_auth) {
385 return origin_bs_auth_dispatch(t, oc, &ad);
386 }
387 return 0;
388 }
389 free(fr.body);
390
391 if (fr.status == kXR_authmore) {
392 return origin_bs_authmore_fallback(t, oc);
393 }
394 if (fr.status != kXR_ok) {
395 brix_cache_set_error(t, kXR_ServerError, 0,
396 "cache origin anonymous login failed");
397 return -1;
398 }
399
400 return 0;
401}

403/* brix_cache_origin_open — kXR_open (read + kXR_retstat) of the source file:
404 * parse ServerOpenBody for the fhandle and the appended stat string, so file_size
405 * is known before a full download (the admission filter can reject oversized files
406 * without fetching them). Returns 0 with fhandle set, -1 on error or redirect. */
407int
408brix_cache_origin_open(brix_cache_fill_t *t,
409 brix_cache_origin_conn_t *oc, u_char fhandle[XRD_FHANDLE_LEN])
410{
411 size_t pathlen, total;
412 u_char *buf;
413 ClientOpenRequest *req;
414 uint16_t status;
415 uint32_t dlen;
416 u_char *body;
417
418 pathlen = strlen(t->clean_path);
419 total = sizeof(ClientOpenRequest) + pathlen;
420
421 buf = malloc(total);
422 if (buf == NULL) {
423 brix_cache_set_error(t, kXR_NoMemory, 0,
424 "cache origin open allocation failed");
425 return -1;
426 }
427
428 ngx_memzero(buf, total);
429 req = (ClientOpenRequest *) buf;
430 req->streamid[1] = 2;
431 req->requestid = htons(kXR_open);
432 /* kXR_retstat requests an ASCII stat string appended after the fhandle so we
433 * can learn the file size before committing to a full download */
434 {
435 xrdw_open_req_t b = { .options = kXR_open_read | kXR_retstat };
436 xrdw_open_req_pack(&b, ((ClientRequestHdr *) buf)->body);
437 }
438 req->dlen = htonl((kXR_int32) pathlen);
439 ngx_memcpy(buf + sizeof(*req), t->clean_path, pathlen);
440
441 if (brix_cache_io_send(oc, buf, total) != 0) {
442 free(buf);
443 brix_cache_set_error(t, kXR_ServerError, errno,
444 "cache origin open write failed");
445 return -1;
446 }
447 free(buf);
448
449 body = NULL;
450 if (brix_cache_read_response(t, oc, &status, &body, &dlen,
451 BRIX_MAX_PATH + 256) != 0) {
452 return -1;
453 }
454
455 if (status == kXR_error) {
456 brix_cache_set_origin_error(t, body, dlen,
457 "cache origin open failed");
458 free(body);
459 return -1;
460 }
461 if (status == kXR_redirect) {
462 free(body);
463 brix_cache_set_error(t, kXR_Unsupported, 0,
464 "cache origin redirected open; direct data "
465 "server origin is required");
466 return -1;
467 }
468 if (status != kXR_ok || dlen < sizeof(ServerOpenBody)) {
469 free(body);
470 brix_cache_set_error(t, kXR_ServerError, 0,
471 "cache origin open returned invalid response");
472 return -1;
473 }
474
475 ngx_memcpy(fhandle, ((ServerOpenBody *) body)->fhandle, XRD_FHANDLE_LEN);
476
477 /*
478 * If kXR_retstat was honored the stat string follows ServerOpenBody.
479 * Format: "<id> <size> <flags> <modtime>" — we only need the size (field 2).
480 * The body is always NUL-terminated by brix_cache_read_response, so
481 * strtoull is safe.
482 */
483 if (dlen > sizeof(ServerOpenBody)) {
484 const char *stat_str = (const char *) body + sizeof(ServerOpenBody);
485 const char *p;
486
487 p = strchr(stat_str, ' ');
488 if (p != NULL) {
489 char *endp;
490 unsigned long long sv;
491
492 errno = 0;
493 sv = strtoull(p + 1, &endp, 10);
494 if (errno == 0 && endp != p + 1) {
495 t->file_size = (off_t) sv;
496 }
497 }
498 }
499
500 free(body);
501 return 0;
502}

543/* origin_cksum_split — pure parse of the NUL-terminated "<algo> <hexvalue>"
544 * reply body into the caller buffers, trimming trailing whitespace from the hex
545 * field. Leaves both buffers untouched (still empty) when the body doesn't fit
546 * the expected shape or exceeds the buffer sizes — the caller then treats it as
547 * "no origin digest". No I/O, no task state. */
548static void
549origin_cksum_split(const u_char *body, char *alg_out, size_t alg_sz,
550 char *hex_out, size_t hex_sz)
551{
552 const char *sp;
553
554 sp = strchr((const char *) body, ' ');
555 if (sp != NULL) {
556 size_t an = (size_t) (sp - (const char *) body);
557 const char *hv = sp + 1;
558 const char *end = hv + strlen(hv);
559 size_t hn;
560
561 while (end > hv && (end[-1] == '\n' || end[-1] == '\r'
562 || end[-1] == ' ' || end[-1] == '\t')) {
563 end--;
564 }
565 hn = (size_t) (end - hv);
566
567 if (an > 0 && an < alg_sz && hn > 0 && hn < hex_sz) {
568 ngx_memcpy(alg_out, body, an);
569 alg_out[an] = '\0';
570 ngx_memcpy(hex_out, hv, hn);
571 hex_out[hn] = '\0';

576/* brix_cache_origin_query_checksum — ask the origin for its stored digest of
577 * t->clean_path (path-based kXR_query/kXR_Qcksum), returning "<algo> <hex>" split
578 * into the caller buffers (out->alg / out->hex). Checksum-on-fill (verify.c)
579 * validates downloaded bytes against this before publishing. BEST-EFFORT: an
580 * origin with no checksum or a wire hiccup must NOT fail an otherwise-complete
581 * fill (data is already on disk) — on ANY failure it restores t's error state
582 * and returns 0 with out->alg emptied, so the caller treats it as "no origin
583 * digest" and the verify policy decides.
584 *
585 * WAITRESP: a real origin usually does NOT have the digest cached at fill time —
586 * it computes it on demand and PARKS the query with kXR_waitresp, then delivers
587 * the answer as an unsolicited kXR_attn(kXR_asynresp) frame (exactly what a stock
588 * XRootD client absorbs transparently). We follow that handshake for a bounded
589 * number of hops so the common lazy-checksum origin still yields a digest for
590 * verify=require, while a hostile origin that streams frames forever cannot wedge
591 * the fill thread — each recv is bounded by the origin socket timeout and the hop
592 * count caps the exchange. Unrelated async pushes (kXR_asyncms) are skipped. */
593int
594brix_cache_origin_query_checksum(brix_cache_fill_t *t,
595 brix_cache_origin_conn_t *oc, const brix_cache_cksum_out_t *out)
596{
597 uint16_t status;
598 uint32_t dlen;
599 u_char *body;
600 int saved_result, saved_xrd;
601 int hops;
602
603 if (out->alg_sz > 0) {
604 out->alg[0] = '\0';
605 }
606 if (out->hex_sz > 0) {
607 out->hex[0] = '\0';
608 }
609
610 /* The download already succeeded; never let a checksum-query failure leak an
611 * error onto the task. Snapshot and restore the error triple. */
612 saved_result = t->result;
613 saved_xrd = t->xrd_error;
614
615 if (origin_cksum_send_query(t, oc) != 0) {
616 return 0;
617 }
618
619 for (hops = 0; hops < 8; hops++) {
620 body = NULL;
621 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 512) != 0) {
622 t->result = saved_result;
623 t->xrd_error = saved_xrd;
624 return 0;
625 }
626
627 if (status == kXR_waitresp) {
628 free(body); /* body = advised seconds; the answer frame follows */
629 continue;
630 }
631
632 if (status == kXR_attn) {
633 /* kXR_attn asynresp body layout (opcodes.h):
634 * actnum[4] reserved[4] ServerResponseHdr[8] response[inner_dlen] */
635 uint32_t actnum;
636 uint16_t inner_status;
637 uint32_t inner_dlen;
638
639 if (body == NULL || dlen < 16) {
640 free(body);
641 return 0;
642 }
643 actnum = xrd_get_u32_be(body);
644 if (actnum != (uint32_t) kXR_asynresp) {
645 free(body); /* asyncms / other push — keep going */
646 continue;
647 }
648 xrd_resp_hdr_unpack(body + 8, NULL, &inner_status, &inner_dlen);
649 if (inner_status != kXR_ok || inner_dlen == 0
650 || (size_t) 16 + inner_dlen > (size_t) dlen)
651 {
652 free(body); /* deferred result was not a digest */
653 return 0;
654 }
655 /* NUL-terminate the inner payload before the split (read_response
656 * over-allocated dlen+1, and 16+inner_dlen <= dlen, so this index is
657 * in bounds). */
658 body[16 + inner_dlen] = '\0';
659 origin_cksum_split(body + 16, out->alg, out->alg_sz,
660 out->hex, out->hex_sz);
661 free(body);
662 return 0;
663 }
664
665 if (status == kXR_ok) {
666 if (body == NULL || dlen == 0) {
667 free(body);
668 return 0;
669 }
670 /* body is NUL-terminated "<algo> <hexvalue>". */
671 origin_cksum_split(body, out->alg, out->alg_sz,
672 out->hex, out->hex_sz);
673 free(body);
674 return 0;
675 }
676
677 /* kXR_error or any other status: origin has no usable checksum. */
678 free(body);
679 return 0;
680 }
681
682 return 0; /* too many hops — treat as no origin digest (verify decides) */
683}

686/* brix_cache_origin_read_chunk — kXR_read at (rng->read_off, rng->want),
687 * writing each reply payload to the sink via brix_cache_sink_pwrite and looping
688 * over kXR_oksofar until the final kXR_ok. dlen is bounded (<= want, accumulated
689 * rng->got within request bounds) to prevent overflow. Sets rng->got;
690 * returns 0 / -1. */
691int
692brix_cache_origin_read_chunk(brix_cache_fill_t *t,
693 brix_cache_origin_conn_t *oc, const u_char fhandle[XRD_FHANDLE_LEN],
694 brix_cache_sink_t *sink, brix_cache_read_range_t *rng)
695{
696 ClientReadRequest req;
697 uint16_t status;
698 uint32_t dlen;
699 u_char *body;
700
701 rng->got = 0;
702
703 ngx_memzero(&req, sizeof(req));
704 req.streamid[1] = 3;
705 req.requestid = htons(kXR_read);
706 ngx_memcpy(req.fhandle, fhandle, XRD_FHANDLE_LEN);
707 req.offset = (kXR_int64) htobe64(rng->read_off);
708 req.rlen = htonl((kXR_int32) rng->want);
709 req.dlen = 0;
710
711 if (brix_cache_io_send(oc, &req, sizeof(req)) != 0) {
712 brix_cache_set_error(t, kXR_ServerError, errno,
713 "cache origin read write failed");
714 return -1;
715 }
716
717 for (;;) {
718 body = NULL;
719 if (brix_cache_read_response(t, oc, &status, &body, &dlen,
720 BRIX_CACHE_FETCH_CHUNK) != 0) {
721 return -1;
722 }
723
724 if (status == kXR_error) {
725 brix_cache_set_origin_error(t, body, dlen,
726 "cache origin read failed");
727 free(body);
728 return -1;
729 }
730
731 if (status != kXR_ok && status != kXR_oksofar) {
732 free(body);
733 brix_cache_set_error(t, kXR_ServerError, 0,
734 "cache origin read returned invalid status");
735 return -1;
736 }
737
738 if ((size_t) dlen > rng->want || rng->got > rng->want - (size_t) dlen) {
739 free(body);
740 brix_cache_set_error(t, kXR_ServerError, 0,
741 "cache origin read returned too much data");
742 return -1;
743 }
744
745 if (dlen > 0) {
746 /* Write at dst_off + bytes already written this call (rng->got).
747 * dst_off is the caller's WRITE base, decoupled from the origin READ
748 * offset: the whole-file fetch passes dst_off==read_off (absolute), a
749 * slice fill passes a 0-relative base. Using got alone restarts at 0
750 * each 1 MiB chunk, so multi-chunk whole-file fetches overwrote at
751 * offset 0 (corrupting any file > BRIX_CACHE_FETCH_CHUNK → adler32
752 * mismatch). */
753 if (brix_cache_sink_pwrite(sink, body, dlen,
754 (off_t) (rng->dst_off + rng->got)) != 0)
755 {
756 free(body);
757 brix_cache_set_syserror(t, kXR_IOError,
758 "cache file write failed");
759 return -1;
760 }
761 rng->got += (size_t) dlen;
762 }
763
764 free(body);
765
766 if (status == kXR_ok) {
767 return 0;
768 }
769 }
770}
