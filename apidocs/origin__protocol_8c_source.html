35/* brix_cache_origin_open — kXR_open (read + kXR_retstat) of the source file:
36 * parse ServerOpenBody for the fhandle and the appended stat string, so file_size
37 * is known before a full download (the admission filter can reject oversized files
38 * without fetching them). Returns 0 with fhandle set, -1 on error or redirect. */
39int
40brix_cache_origin_open(brix_cache_fill_t *t,
41 brix_cache_origin_conn_t *oc, u_char fhandle[XRD_FHANDLE_LEN])
42{

44 u_char *buf;
45 ClientOpenRequest *req;
46 uint16_t status;
47 uint32_t dlen;
48 u_char *body;
49
50 pathlen = strlen(t->clean_path);
51 total = sizeof(ClientOpenRequest) + pathlen;
52
53 buf = malloc(total);
54 if (buf == NULL) {
55 brix_cache_set_error(t, kXR_NoMemory, 0,
56 "cache origin open allocation failed");
57 return -1;
58 }
59
60 ngx_memzero(buf, total);
61 req = (ClientOpenRequest *) buf;
62 req->streamid[1] = 2;
63 req->requestid = htons(kXR_open);

65 * can learn the file size before committing to a full download */

68 xrdw_open_req_pack(&b, ((ClientRequestHdr *) buf)->body);

104 return -1;

111 * Format: "<id> <size> <flags> <modtime>" — we only need the size (field 2).
112 * The body is always NUL-terminated by brix_cache_read_response, so
113 * strtoull is safe.
114 */

116 const char *stat_str = (const char *) body + sizeof(ServerOpenBody);

120 if (p != NULL) {
121 char *endp;
122 unsigned long long sv;
123
124 errno = 0;
125 sv = strtoull(p + 1, &endp, 10);
126 if (errno == 0 && endp != p + 1) {

128 }
129 }
130 }

132 free(body);
133 return 0;
134}

136/* origin_cksum_send_query — build and send the path-based kXR_query/kXR_Qcksum
137 * request for t->clean_path. WHY separate: the request assembly (malloc + pack +
138 * send + free) is the only allocation in the checksum exchange; isolating it
139 * keeps the best-effort orchestrator a flat status sequence. Returns 0 on send,
140 * -1 on OOM or write failure — NO task error is set (best-effort). */
141static int
142origin_cksum_send_query(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc)
143{
144 size_t pathlen, total;
145 u_char *buf;
146 ClientQueryRequest *req;
147
148 pathlen = strlen(t->clean_path);
149 total = sizeof(ClientQueryRequest) + pathlen;
150
151 buf = malloc(total);
152 if (buf == NULL) {
153 return -1; /* best-effort: skip verification on OOM */
154 }
155
156 ngx_memzero(buf, total);
157 req = (ClientQueryRequest *) buf;
158 req->streamid[1] = 6; /* unused stream slot */
159 req->requestid = htons(kXR_query);

161 xrdw_query_req_t b = { .infotype = kXR_Qcksum }; /* fhandle 0 ⇒ path-based */
162 xrdw_query_req_pack(&b, ((ClientRequestHdr *) buf)->body);

165 ngx_memcpy(buf + sizeof(*req), t->clean_path, pathlen);
166

168 free(buf);
169 return -1;
170 }
171 free(buf);
172 return 0;
173}

175/* origin_cksum_split — pure parse of the NUL-terminated "<algo> <hexvalue>"
176 * reply body into the caller buffers, trimming trailing whitespace from the hex
177 * field. Leaves both buffers untouched (still empty) when the body doesn't fit
178 * the expected shape or exceeds the buffer sizes — the caller then treats it as
179 * "no origin digest". No I/O, no task state. */
180static void
181origin_cksum_split(const u_char *body, char *alg_out, size_t alg_sz,
182 char *hex_out, size_t hex_sz)
183{
184 const char *sp;
185
186 sp = strchr((const char *) body, ' ');
187 if (sp != NULL) {
188 size_t an = (size_t) (sp - (const char *) body);
189 const char *hv = sp + 1;
190 const char *end = hv + strlen(hv);
191 size_t hn;
192
193 while (end > hv && (end[-1] == '\n' || end[-1] == '\r'
194 || end[-1] == ' ' || end[-1] == '\t')) {
195 end--;
196 }
197 hn = (size_t) (end - hv);
198
199 if (an > 0 && an < alg_sz && hn > 0 && hn < hex_sz) {
200 ngx_memcpy(alg_out, body, an);
201 alg_out[an] = '\0';
202 ngx_memcpy(hex_out, hv, hn);
203 hex_out[hn] = '\0';
204 }
205 }
206}

208/* brix_cache_origin_query_checksum — ask the origin for its stored digest of
209 * t->clean_path (path-based kXR_query/kXR_Qcksum), returning "<algo> <hex>" split
210 * into the caller buffers (out->alg / out->hex). Checksum-on-fill (verify.c)
211 * validates downloaded bytes against this before publishing. BEST-EFFORT: an
212 * origin with no checksum or a wire hiccup must NOT fail an otherwise-complete
213 * fill (data is already on disk) — on ANY failure it restores t's error state
214 * and returns 0 with out->alg emptied, so the caller treats it as "no origin
215 * digest" and the verify policy decides.
216 *
217 * WAITRESP: a real origin usually does NOT have the digest cached at fill time —
218 * it computes it on demand and PARKS the query with kXR_waitresp, then delivers
219 * the answer as an unsolicited kXR_attn(kXR_asynresp) frame (exactly what a stock
220 * XRootD client absorbs transparently). We follow that handshake for a bounded
221 * number of hops so the common lazy-checksum origin still yields a digest for
222 * verify=require, while a hostile origin that streams frames forever cannot wedge
223 * the fill thread — each recv is bounded by the origin socket timeout and the hop
224 * count caps the exchange. Unrelated async pushes (kXR_asyncms) are skipped. */
225int
226brix_cache_origin_query_checksum(brix_cache_fill_t *t,
227 brix_cache_origin_conn_t *oc, const brix_cache_cksum_out_t *out)
228{
229 uint16_t status;
230 uint32_t dlen;
231 u_char *body;
232 int saved_result, saved_xrd;
233 int hops;
234
235 if (out->alg_sz > 0) {
236 out->alg[0] = '\0';
237 }
238 if (out->hex_sz > 0) {
239 out->hex[0] = '\0';
240 }
241
242 /* The download already succeeded; never let a checksum-query failure leak an
243 * error onto the task. Snapshot and restore the error triple. */
244 saved_result = t->result;
245 saved_xrd = t->xrd_error;
246
247 if (origin_cksum_send_query(t, oc) != 0) {
248 return 0;
249 }
250
251 for (hops = 0; hops < 8; hops++) {
252 body = NULL;
253 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 512) != 0) {
254 t->result = saved_result;
255 t->xrd_error = saved_xrd;
256 return 0;
257 }
258
259 if (status == kXR_waitresp) {
260 free(body); /* body = advised seconds; the answer frame follows */
261 continue;
262 }
263
264 if (status == kXR_attn) {
265 /* kXR_attn asynresp body layout (opcodes.h):
266 * actnum[4] reserved[4] ServerResponseHdr[8] response[inner_dlen] */
267 uint32_t actnum;
268 uint16_t inner_status;
269 uint32_t inner_dlen;
270
271 if (body == NULL || dlen < 16) {
272 free(body);
273 return 0;
274 }
275 actnum = xrd_get_u32_be(body);
276 if (actnum != (uint32_t) kXR_asynresp) {
277 free(body); /* asyncms / other push — keep going */
278 continue;
279 }
280 xrd_resp_hdr_unpack(body + 8, NULL, &inner_status, &inner_dlen);
281 if (inner_status != kXR_ok || inner_dlen == 0
282 || (size_t) 16 + inner_dlen > (size_t) dlen)
283 {
284 free(body); /* deferred result was not a digest */
285 return 0;
286 }
287 /* NUL-terminate the inner payload before the split (read_response
288 * over-allocated dlen+1, and 16+inner_dlen <= dlen, so this index is
289 * in bounds). */
290 body[16 + inner_dlen] = '\0';
291 origin_cksum_split(body + 16, out->alg, out->alg_sz,
292 out->hex, out->hex_sz);
293 free(body);
294 return 0;
295 }
296
297 if (status == kXR_ok) {
298 if (body == NULL || dlen == 0) {
299 free(body);
300 return 0;
301 }
302 /* body is NUL-terminated "<algo> <hexvalue>". */
303 origin_cksum_split(body, out->alg, out->alg_sz,
304 out->hex, out->hex_sz);
305 free(body);
306 return 0;
307 }
308
309 /* kXR_error or any other status: origin has no usable checksum. */
310 free(body);
311 return 0;
312 }
313
314 return 0; /* too many hops — treat as no origin digest (verify decides) */
315}

318/* brix_cache_origin_read_chunk — kXR_read at (rng->read_off, rng->want),
319 * writing each reply payload to the sink via brix_cache_sink_pwrite and looping
320 * over kXR_oksofar until the final kXR_ok. dlen is bounded (<= want, accumulated
321 * rng->got within request bounds) to prevent overflow. Sets rng->got;
322 * returns 0 / -1. */
323int
324brix_cache_origin_read_chunk(brix_cache_fill_t *t,
325 brix_cache_origin_conn_t *oc, const u_char fhandle[XRD_FHANDLE_LEN],
326 brix_cache_sink_t *sink, brix_cache_read_range_t *rng)
327{
328 ClientReadRequest req;
329 uint16_t status;
330 uint32_t dlen;
331 u_char *body;
332
333 rng->got = 0;
334
335 ngx_memzero(&req, sizeof(req));
336 req.streamid[1] = 3;
337 req.requestid = htons(kXR_read);
338 ngx_memcpy(req.fhandle, fhandle, XRD_FHANDLE_LEN);
339 req.offset = (kXR_int64) htobe64(rng->read_off);
340 req.rlen = htonl((kXR_int32) rng->want);
341 req.dlen = 0;
342
343 if (brix_cache_io_send(oc, &req, sizeof(req)) != 0) {
344 brix_cache_set_error(t, kXR_ServerError, errno,
345 "cache origin read write failed");
346 return -1;
347 }
348
349 for (;;) {
350 body = NULL;
351 if (brix_cache_read_response(t, oc, &status, &body, &dlen,
352 BRIX_CACHE_FETCH_CHUNK) != 0) {
353 return -1;
354 }
355
356 if (status == kXR_error) {
357 brix_cache_set_origin_error(t, body, dlen,
358 "cache origin read failed");
359 free(body);
360 return -1;
361 }
362
363 if (status != kXR_ok && status != kXR_oksofar) {
364 free(body);
365 brix_cache_set_error(t, kXR_ServerError, 0,
366 "cache origin read returned invalid status");
367 return -1;
368 }
369
370 if ((size_t) dlen > rng->want || rng->got > rng->want - (size_t) dlen) {
371 free(body);
372 brix_cache_set_error(t, kXR_ServerError, 0,
373 "cache origin read returned too much data");
374 return -1;
375 }
376
377 if (dlen > 0) {
378 /* Write at dst_off + bytes already written this call (rng->got).
379 * dst_off is the caller's WRITE base, decoupled from the origin READ
380 * offset: the whole-file fetch passes dst_off==read_off (absolute), a
381 * slice fill passes a 0-relative base. Using got alone restarts at 0
382 * each 1 MiB chunk, so multi-chunk whole-file fetches overwrote at
383 * offset 0 (corrupting any file > BRIX_CACHE_FETCH_CHUNK → adler32
384 * mismatch). */
385 if (brix_cache_sink_pwrite(sink, body, dlen,
386 (off_t) (rng->dst_off + rng->got)) != 0)
387 {
388 free(body);
389 brix_cache_set_syserror(t, kXR_IOError,
390 "cache file write failed");
391 return -1;
392 }
393 rng->got += (size_t) dlen;
394 }
395
396 free(body);
397
398 if (status == kXR_ok) {
399 return 0;
400 }
401 }
402}
