1/*
2 * credinfo.c — credential introspection for `explain` (§15.2).
3 *
4 * WHAT: Best-effort decoders that narrate the credentials a session is (or could
5 * be) using: a bearer token's claims, and a GSI proxy's subject / issuer /
6 * validity / VOMS FQANs / clock skew. Output is human diagnostic text.
7 * WHY: "Why did auth fail / why am I 'nobody'?" is usually a credential problem
8 * (expired token, wrong aud, near-expiry proxy, clock skew, missing VOMS).
9 * Surfacing the decoded credential turns a silent fallback into an obvious
10 * cause — exactly what the native client is positioned to show.
11 * HOW: JWT: base64url-decode the payload segment and scalar-scan iss/sub/aud/
12 * scope/exp (NO signature verify, NO jansson — a diagnostic, not a gate);
13 * flag EXPIRED against the local clock. GSI: reuse the OpenSSL X509 parse
14 * (as proxy.c) for subject/issuer/notAfter, walk the VOMS AC extension
15 * (OID 1.3.6.1.4.1.8005.100.100.5) for FQANs, and report notBefore skew.
16 *
17 * Every function is fail-soft: malformed input prints a one-line note and
18 * returns; it must never crash `explain`. Clean-room: standard JWT/RFC + OpenSSL.
19 */

39/* Decode a JWT's payload segment into out[outsz] as a NUL-terminated JSON string,
40 * via the SHARED base64url decoder + JWS splitter (libxrdproto — the same code the
41 * module's token path uses). Returns the decoded length (>=0), or -1 if the input
42 * is not a JWT, the payload is too large, or it fails to decode. */
43static ssize_t
44decode_payload_json(const char *jwt, char *out, size_t outsz)
45{
46 xrdjwt_seg seg[3];
47 ssize_t n;
48
49 if (outsz < 2 || xrdjwt_split(jwt, strlen(jwt), seg) != 0) {
50 return -1;
51 }
52 n = b64url_decode(seg[1].p, seg[1].n, (uint8_t *) out, outsz - 1);
53 if (n < 0) {
54 return -1;
55 }
56 out[n] = '\0';
57 return n;
58}

72void
73brix_token_explain(const char *jwt, FILE *out)
74{
75 char json[8192];
76 char buf[512];
77
78 if (jwt == NULL || jwt[0] == '\0') {
79 return;
80 }
81 if (decode_payload_json(jwt, json, sizeof(json)) < 0 || json[0] != '{') {
82 fprintf(out, " token: (present, unparseable — not a JWT payload)\n");
83 return;
84 }
85 fprintf(out, " token:\n");
86 if (json_str(json, "iss", buf, sizeof(buf))) { fprintf(out, " iss: %s\n", buf); }
87 if (json_str(json, "sub", buf, sizeof(buf))) { fprintf(out, " sub: %s\n", buf); }
88 if (json_str(json, "aud", buf, sizeof(buf))) { fprintf(out, " aud: %s\n", buf); }
89 if (json_str(json, "scope", buf, sizeof(buf))) { fprintf(out, " scope: %s\n", buf); }
90 if (json_str(json, "exp", buf, sizeof(buf))) {
91 long expv = strtol(buf, NULL, 10);
92 long now = (long) time(NULL);
93 fprintf(out, " exp: %ld (%s)\n", expv,
94 expv <= now ? "EXPIRED" : "valid");
95 }
96}

98/*
99 * Apply the JWT `exp` claim to *m (m->exp + m->expired).
100 *
101 * WHAT: if `json` carries a top-level `exp`, record its epoch value in m->exp and
102 * set m->expired when that instant is at/before the local clock. No-op when
103 * the claim is absent (m keeps its zeroed exp/expired).
104 * WHY: isolating the expiry classification keeps brix_token_meta_get a flat
105 * sequence of one-claim steps rather than one branchy procedure — the exp
106 * comparison is the single fact the doctor oracle needs to predict a deny.
107 * HOW: (1) extract `exp` with the shared JSON scanner; (2) strtol to epoch;
108 * (3) mark expired only for a positive exp that is <= now (guarding a
109 * missing/zero value from spuriously reading as expired).
110 */
111static void
112token_meta_apply_exp(const char *json, brix_token_meta *m)
113{
114 char expbuf[32];
115
116 if (json_str(json, "exp", expbuf, sizeof(expbuf))) {
117 m->exp = strtol(expbuf, NULL, 10);
118 m->expired = (m->exp > 0 && m->exp <= (long) time(NULL));
119 }
120}

122/*
123 * Set m->has_read / m->has_write from the SHARED parser's WLCG scope flags.
124 *
125 * WHAT: fold the `nsc` parsed WLCG scopes into a path-agnostic "any read-capable
126 * scope?" / "any write-capable scope?" pair on *m.
127 * WHY: the read/write reduction over the parser's exact permission flags is the
128 * authoritative branch — pulling it out keeps token_meta_apply_scope's
129 * parsed-vs-heuristic choice readable and this reduction independently
130 * testable against the server's grant semantics.
131 * HOW: walk each parsed scope; OR its `read` flag into m->has_read and its
132 * write/create/modify flags into m->has_write (any one implies writable).
133 */
134static void
135token_scope_from_parsed(const brix_token_scope_t *sc, int nsc, brix_token_meta *m)
136{
137 int i;
138
139 for (i = 0; i < nsc; i++) {
140 if (sc[i].read) { m->has_read = 1; }
141 if (sc[i].write || sc[i].create || sc[i].modify) { m->has_write = 1; }
142 }
143}

145/*
146 * Fail-soft keyword classification of an opaque (non-WLCG) scope string.
147 *
148 * WHAT: set m->has_read / m->has_write by substring-matching read / write /
149 * create / modify in a scope the WLCG parser yielded nothing for.
150 * WHY: an opaque or non-standard `scope` claim must still produce a best-effort
151 * answer rather than silently reading as no-access; isolating the legacy
152 * heuristic keeps it clearly the fallback path, not the primary one.
153 * HOW: substring-test `read` for has_read; treat any of write/create/modify as
154 * write-capable (deliberately generous — the server stays the real gate).
155 */
156static void
157token_scope_from_heuristic(const char *scope, brix_token_meta *m)
158{
159 m->has_read = (strstr(scope, "read") != NULL);
160 m->has_write = (strstr(scope, "write") != NULL
161 || strstr(scope, "create") != NULL
162 || strstr(scope, "modify") != NULL);
163}

165/*
166 * Apply the JWT `scope` claim to *m (has_scope + has_read/has_write).
167 *
168 * WHAT: if `json` carries a top-level `scope`, mark m->has_scope and derive the
169 * read/write capability, preferring the SHARED WLCG parser and falling back
170 * to the keyword heuristic only when the parser yields no scopes.
171 * WHY: centralising the parsed-vs-heuristic decision in one helper keeps the
172 * parser the single source of truth (matching the server's grant) with a
173 * clearly-secondary fallback, and flattens brix_token_meta_get.
174 * HOW: (1) extract `scope`; (2) set has_scope; (3) parse with
175 * brix_token_parse_scopes; (4) if it produced scopes, reduce them via
176 * token_scope_from_parsed, else classify the raw string via
177 * token_scope_from_heuristic.
178 */
179static void
180token_meta_apply_scope(const char *json, brix_token_meta *m)
181{
182 char scope[1024];
183
184 if (json_str(json, "scope", scope, sizeof(scope))) {
185 brix_token_scope_t sc[16];
186 int nsc = brix_token_parse_scopes(scope, sc, 16);
187 m->has_scope = 1;
188 if (nsc > 0) {
189 /* Path-agnostic "any read/write-capable scope?" derived from the
190 * shared parser's exact WLCG permission flags, matching the server. */
191 token_scope_from_parsed(sc, nsc, m);
192 } else {
193 /* Fail-soft: an opaque/non-WLCG scope the parser yields nothing for
194 * falls back to the legacy keyword heuristic (be generous on write). */
195 token_scope_from_heuristic(scope, m);
196 }
197 }
198}

200/*
201 * WHAT: fill *m with a bearer token's machine-readable facts (validity + WLCG
202 * read/write scope presence). m->valid=0 if the JWT can't be parsed.
203 * WHY: the remote-doctor auth-suite must know, programmatically, whether a token
204 * is expired or read-only so it can predict whether the server SHOULD allow
205 * or deny an operation — turning a guess into an oracle-backed assertion.
206 * HOW: base64url-decode the payload (shared decoder), scalar-scan exp (vs local
207 * clock), and classify the WLCG `scope` claim with the SHARED scope parser
208 * (brix_token_parse_scopes — the exact mapping the server enforces) so the
209 * oracle predicts the server's decision rather than approximating it. No
210 * signature verify — the server is the gate; this only predicts its answer.
211 */
212void
213brix_token_meta_get(const char *jwt, brix_token_meta *m)
214{
215 char json[8192];
216
217 if (m == NULL) {
218 return;
219 }
220 memset(m, 0, sizeof(*m));
221 if (jwt == NULL || jwt[0] == '\0') {
222 return;
223 }
224 if (decode_payload_json(jwt, json, sizeof(json)) < 0 || json[0] != '{') {
225 return;
226 }
227 m->valid = 1;
228 token_meta_apply_exp(json, m);
229 token_meta_apply_scope(json, m);
230}

237/* Scan the DER bytes of the VOMS extension for FQAN-looking ASCII runs
238 * ("/vo/...", "/vo/Role=..."), best-effort (full AC decode is out of scope). */
239static void
240voms_scan(const unsigned char *der, int len, FILE *out)
241{
242 int i = 0, printed = 0;
243 char run[256];
244
245 while (i < len) {
246 if (der[i] == '/') {
247 int j = 0;
248 while (i < len && j < (int) sizeof(run) - 1
249 && der[i] >= 0x20 && der[i] < 0x7f) {
250 run[j++] = (char) der[i++];
251 }
252 run[j] = '\0';
253 if (j >= 4 && (strstr(run, "Role=") != NULL || strchr(run + 1, '/') != NULL)) {
254 fprintf(out, " VOMS: %s\n", run);
255 printed = 1;
256 }
257 } else {
258 i++;
259 }
260 }
261 if (!printed) {
262 fprintf(out, " VOMS: present (no FQAN decoded)\n");
263 }
264}

266void
267brix_gsi_cert_explain(const char *proxy_path, FILE *out)
268{
269 BIO *bio;
270 X509 *cert;
271 char subj[512], issuer[512];
272
273 if (proxy_path == NULL || proxy_path[0] == '\0') {
274 return;
275 }
276 bio = brix_credfile_bio(proxy_path, 1);
277 if (bio == NULL) {
278 return;
279 }
280 cert = PEM_read_bio_X509(bio, NULL, NULL, NULL);
281 BIO_free(bio);
282 if (cert == NULL) {
283 fprintf(out, " proxy: (present at %s, unparseable PEM)\n", proxy_path);
284 return;
285 }
286
287 X509_NAME_oneline(X509_get_subject_name(cert), subj, sizeof(subj));
288 X509_NAME_oneline(X509_get_issuer_name(cert), issuer, sizeof(issuer));
289 fprintf(out, " proxy:\n");
290 fprintf(out, " subject: %s\n", subj);
291 fprintf(out, " issuer: %s\n", issuer);
292
293 /* validity + clock skew */
294 {
295 int dn = 0, ds = 0, bn = 0, bs = 0;
296 const ASN1_TIME *na = X509_get0_notAfter(cert);
297 const ASN1_TIME *nb = X509_get0_notBefore(cert);
298 if (ASN1_TIME_diff(&dn, &ds, NULL, na)) {
299 if (dn < 0 || (dn == 0 && ds < 0)) {
300 fprintf(out, " expiry: EXPIRED\n");
301 } else {
302 fprintf(out, " expiry: %dd %dh left\n", dn, ds / 3600);
303 }
304 }
305 if (ASN1_TIME_diff(&bn, &bs, NULL, nb) && (bn > 0 || bs > 0)) {
306 fprintf(out, " skew: notBefore is in the future "
307 "(local clock behind / cert not yet valid)\n");
308 }
309 }
310
311 /* VOMS attribute certificate */
312 {
313 int loc = -1, n = X509_get_ext_count(cert), i;
314 for (i = 0; i < n; i++) {
315 X509_EXTENSION *ex = X509_get_ext(cert, i);
316 ASN1_OBJECT *obj = X509_EXTENSION_get_object(ex);
317 char oid[128];
318 OBJ_obj2txt(oid, sizeof(oid), obj, 1);
319 if (strcmp(oid, VOMS_AC_OID) == 0) {
320 loc = i;
321 break;
322 }
323 }
324 if (loc < 0) {
325 fprintf(out, " VOMS: none\n");
326 } else {
327 ASN1_OCTET_STRING *data =
328 X509_EXTENSION_get_data(X509_get_ext(cert, loc));
329 voms_scan(ASN1_STRING_get0_data(data), ASN1_STRING_length(data), out);
330 }
331 }
332 X509_free(cert);
333}

337/*
338 * Diagnose the discoverable bearer token; return 1 for a likely-fatal problem.
339 *
340 * WHAT: discover a local bearer token (env / file / XDG / /tmp); if it parses,
341 * print — indented by `prefix` — a note for an EXPIRED token (returns 1), a
342 * read-only token on a `want_write` op (returns 1), or a near-expiry token
343 * (advisory only, returns 0). Returns 0 when no token, an unparseable token,
344 * or no likely-fatal problem.
345 * WHY: splitting the token arm out of brix_cred_diagnose keeps each credential
346 * kind's fail-fast logic self-contained and independently reviewable, and
347 * keeps the orchestrator a flat two-step sequence.
348 * HOW: (1) brix_token_discover; (2) brix_token_meta_get; (3) on a valid token,
349 * branch expired -> read-only-write -> near-expiry (first match wins,
350 * preserving the original else-if precedence); (4) free the token.
351 */
352static int
353diagnose_token(int want_write, const char *prefix, FILE *out)
354{
355 char *tok;
356 int problem = 0;
357 brix_token_meta m;
358
359 tok = brix_token_discover();
360 if (tok == NULL) {
361 return 0;
362 }
363 memset(&m, 0, sizeof(m));
364 brix_token_meta_get(tok, &m);
365 if (m.valid) {
366 if (m.expired) {
367 fprintf(out, "%sbearer token has EXPIRED (exp %ld, %ld s ago) — "
368 "obtain a fresh token and retry\n",
369 prefix, m.exp, (long) time(NULL) - m.exp);
370 problem = 1;
371 } else if (want_write && m.has_scope && !m.has_write) {
372 fprintf(out, "%sbearer token grants READ scope only — this write "
373 "needs a storage.write / create / modify scope\n",
374 prefix);
375 problem = 1;
376 } else if (m.exp > 0) {
377 long secs = m.exp - (long) time(NULL);
378 if (secs >= 0 && secs < 300) {
379 fprintf(out, "%sbearer token expires in %ld s — it may expire "
380 "mid-transfer\n", prefix, secs);
381 }
382 }
383 }
384 free(tok);
385 return problem;
386}

388/*
389 * Diagnose the default-path GSI proxy; return 1 for a likely-fatal problem.
390 *
391 * WHAT: if a readable GSI proxy exists at the default path and its remaining life
392 * is computable, print — indented by `prefix` — a note for an EXPIRED proxy
393 * (returns 1) or a near-expiry proxy (advisory only, returns 0). Returns 0
394 * when no proxy is present or its remaining life cannot be determined.
395 * WHY: mirrors diagnose_token so the proxy arm is self-contained; the two
396 * credential kinds stay decoupled and the orchestrator stays flat.
397 * HOW: (1) resolve the default proxy path; (2) require R_OK and a successful
398 * brix_proxy_remaining; (3) classify remaining <= 0 as expired (fatal) vs
399 * < 300 s as near-expiry (advisory).
400 */
401static int
402diagnose_proxy(const char *prefix, FILE *out)
403{
404 char pxp[1024];
405 long left = 0;
406
407 brix_proxy_default_path(pxp, sizeof(pxp));
408 if (access(pxp, R_OK) != 0 || brix_proxy_remaining(pxp, &left) != 0) {
409 return 0;
410 }
411 if (left <= 0) {
412 fprintf(out, "%sGSI proxy has EXPIRED (%s) — run 'xrdgsiproxy init'\n",
413 prefix, pxp);
414 return 1;
415 }
416 if (left < 300) {
417 fprintf(out, "%sGSI proxy expires in %ld s (%s) — it may expire "
418 "mid-transfer\n", prefix, left, pxp);
419 }
420 return 0;
421}

423/*
424 * WHAT: inspect whatever credential is present locally (a discoverable bearer
425 * token, then a GSI proxy at the default path) and print a specific,
426 * actionable hint to `out` when a likely auth problem is detectable WITHOUT
427 * contacting the server — an expired or near-expiry token/proxy, or (when
428 * want_write) a token that grants read scope only. Returns 1 if a
429 * likely-fatal local problem was found, else 0.
430 * WHY: so the user sees the real cause instantly ("token expired 3m ago",
431 * "proxy expired — run xrdgsiproxy init", "token is read-only for a write")
432 * instead of digging after a bare "permission denied". Pure local
433 * inspection: no network, no signature verification (it is a hint, not a
434 * gate — the server remains authoritative).
435 */
436int
437brix_cred_diagnose(int want_write, const char *prefix, FILE *out)
438{
439 int problem = 0;
440
441 if (out == NULL) {
442 return 0;
443 }
444 if (prefix == NULL) {
445 prefix = "";
446 }
447
448 /* Bearer token, if one is discoverable (env / file / XDG / /tmp). */
449 problem |= diagnose_token(want_write, prefix, out);
450
451 /* GSI proxy, if one is present at the default path. */
452 problem |= diagnose_proxy(prefix, out);
453
454 return problem;
455}

459 * status carries an auth/authz wire error — so an ordinary ENOENT/EIO failure
460 * does not trigger credential noise. Indented under the primary error line.
461 * Also fires the doctor-referral hint (spec WS-7) so users know that
462 * `xrddiag check <endpoint>` can walk through the auth handshake for them —
