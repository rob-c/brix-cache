1/*
2 * credinfo.c — credential introspection for `explain` (§15.2).
3 *
4 * WHAT: Best-effort decoders that narrate the credentials a session is (or could
5 * be) using: a bearer token's claims, and a GSI proxy's subject / issuer /
6 * validity / VOMS FQANs / clock skew. Output is human diagnostic text.
7 * WHY: "Why did auth fail / why am I 'nobody'?" is usually a credential problem
8 * (expired token, wrong aud, near-expiry proxy, clock skew, missing VOMS).
9 * Surfacing the decoded credential turns a silent fallback into an obvious
10 * cause — exactly what the native client is positioned to show.
11 * HOW: JWT: base64url-decode the payload segment and scalar-scan iss/sub/aud/
12 * scope/exp (NO signature verify, NO jansson — a diagnostic, not a gate);
13 * flag EXPIRED against the local clock. GSI: reuse the OpenSSL X509 parse
14 * (as proxy.c) for subject/issuer/notAfter, walk the VOMS AC extension
15 * (OID 1.3.6.1.4.1.8005.100.100.5) for FQANs, and report notBefore skew.
16 *
17 * Every function is fail-soft: malformed input prints a one-line note and
18 * returns; it must never crash `explain`. Clean-room: standard JWT/RFC + OpenSSL.
19 */

39/* Decode a JWT's payload segment into out[outsz] as a NUL-terminated JSON string,
40 * via the SHARED base64url decoder + JWS splitter (libxrdproto — the same code the
41 * module's token path uses). Returns the decoded length (>=0), or -1 if the input
42 * is not a JWT, the payload is too large, or it fails to decode. */
43static ssize_t
44decode_payload_json(const char *jwt, char *out, size_t outsz)
45{
46 xrdjwt_seg seg[3];
47 ssize_t n;
48
49 if (outsz < 2 || xrdjwt_split(jwt, strlen(jwt), seg) != 0) {
50 return -1;
51 }
52 n = b64url_decode(seg[1].p, seg[1].n, (uint8_t *) out, outsz - 1);
53 if (n < 0) {
54 return -1;
55 }
56 out[n] = '\0';
57 return n;
58}

72void
73brix_token_explain(const char *jwt, FILE *out)
74{
75 char json[8192];
76 char buf[512];
77
78 if (jwt == NULL || jwt[0] == '\0') {
79 return;
80 }
81 if (decode_payload_json(jwt, json, sizeof(json)) < 0 || json[0] != '{') {
82 fprintf(out, " token: (present, unparseable — not a JWT payload)\n");
83 return;
84 }
85 fprintf(out, " token:\n");
86 if (json_str(json, "iss", buf, sizeof(buf))) { fprintf(out, " iss: %s\n", buf); }
87 if (json_str(json, "sub", buf, sizeof(buf))) { fprintf(out, " sub: %s\n", buf); }
88 if (json_str(json, "aud", buf, sizeof(buf))) { fprintf(out, " aud: %s\n", buf); }
89 if (json_str(json, "scope", buf, sizeof(buf))) { fprintf(out, " scope: %s\n", buf); }
90 if (json_str(json, "exp", buf, sizeof(buf))) {
91 long expv = strtol(buf, NULL, 10);
92 long now = (long) time(NULL);
93 fprintf(out, " exp: %ld (%s)\n", expv,
94 expv <= now ? "EXPIRED" : "valid");
95 }
96}

98/*
99 * WHAT: fill *m with a bearer token's machine-readable facts (validity + WLCG
100 * read/write scope presence). m->valid=0 if the JWT can't be parsed.
101 * WHY: the remote-doctor auth-suite must know, programmatically, whether a token
102 * is expired or read-only so it can predict whether the server SHOULD allow
103 * or deny an operation — turning a guess into an oracle-backed assertion.
104 * HOW: base64url-decode the payload (shared decoder), scalar-scan exp (vs local
105 * clock), and classify the WLCG `scope` claim with the SHARED scope parser
106 * (brix_token_parse_scopes — the exact mapping the server enforces) so the
107 * oracle predicts the server's decision rather than approximating it. No
108 * signature verify — the server is the gate; this only predicts its answer.
109 */
110void
111brix_token_meta_get(const char *jwt, brix_token_meta *m)
112{
113 char json[8192];
114 char scope[1024];
115
116 if (m == NULL) {
117 return;
118 }
119 memset(m, 0, sizeof(*m));
120 if (jwt == NULL || jwt[0] == '\0') {
121 return;
122 }
123 if (decode_payload_json(jwt, json, sizeof(json)) < 0 || json[0] != '{') {
124 return;
125 }
126 m->valid = 1;
127 {
128 char expbuf[32];
129 if (json_str(json, "exp", expbuf, sizeof(expbuf))) {
130 m->exp = strtol(expbuf, NULL, 10);
131 m->expired = (m->exp > 0 && m->exp <= (long) time(NULL));
132 }
133 }
134 if (json_str(json, "scope", scope, sizeof(scope))) {
135 brix_token_scope_t sc[16];
136 int nsc = brix_token_parse_scopes(scope, sc, 16);
137 m->has_scope = 1;
138 if (nsc > 0) {
139 /* Path-agnostic "any read/write-capable scope?" derived from the
140 * shared parser's exact WLCG permission flags (read vs
141 * write/create/modify), matching what the server would grant. */
142 int i;
143 for (i = 0; i < nsc; i++) {
144 if (sc[i].read) { m->has_read = 1; }
145 if (sc[i].write || sc[i].create || sc[i].modify) { m->has_write = 1; }
146 }
147 } else {
148 /* Fail-soft: an opaque/non-WLCG scope the parser yields nothing for
149 * falls back to the legacy keyword heuristic (be generous on write). */
150 m->has_read = (strstr(scope, "read") != NULL);
151 m->has_write = (strstr(scope, "write") != NULL
152 || strstr(scope, "create") != NULL
153 || strstr(scope, "modify") != NULL);
154 }
155 }
156}

161#define VOMS_AC_OID "1.3.6.1.4.1.8005.100.100.5"

163/* Scan the DER bytes of the VOMS extension for FQAN-looking ASCII runs
164 * ("/vo/...", "/vo/Role=..."), best-effort (full AC decode is out of scope). */
165static void
166voms_scan(const unsigned char *der, int len, FILE *out)
167{
168 int i = 0, printed = 0;
169 char run[256];
170
171 while (i < len) {
172 if (der[i] == '/') {
173 int j = 0;
174 while (i < len && j < (int) sizeof(run) - 1
175 && der[i] >= 0x20 && der[i] < 0x7f) {
176 run[j++] = (char) der[i++];
177 }
178 run[j] = '\0';
179 if (j >= 4 && (strstr(run, "Role=") != NULL || strchr(run + 1, '/') != NULL)) {
180 fprintf(out, " VOMS: %s\n", run);
181 printed = 1;
182 }
183 } else {
184 i++;
185 }
186 }
187 if (!printed) {
188 fprintf(out, " VOMS: present (no FQAN decoded)\n");
189 }
190}

192void
193brix_gsi_cert_explain(const char *proxy_path, FILE *out)
194{
195 BIO *bio;
196 X509 *cert;
197 char subj[512], issuer[512];
198
199 if (proxy_path == NULL || proxy_path[0] == '\0') {
200 return;
201 }
202 bio = brix_credfile_bio(proxy_path, 1);
203 if (bio == NULL) {
204 return;
205 }
206 cert = PEM_read_bio_X509(bio, NULL, NULL, NULL);
207 BIO_free(bio);
208 if (cert == NULL) {
209 fprintf(out, " proxy: (present at %s, unparseable PEM)\n", proxy_path);
210 return;
211 }
212
213 X509_NAME_oneline(X509_get_subject_name(cert), subj, sizeof(subj));
214 X509_NAME_oneline(X509_get_issuer_name(cert), issuer, sizeof(issuer));
215 fprintf(out, " proxy:\n");
216 fprintf(out, " subject: %s\n", subj);
217 fprintf(out, " issuer: %s\n", issuer);
218
219 /* validity + clock skew */
220 {
221 int dn = 0, ds = 0, bn = 0, bs = 0;
222 const ASN1_TIME *na = X509_get0_notAfter(cert);
223 const ASN1_TIME *nb = X509_get0_notBefore(cert);
224 if (ASN1_TIME_diff(&dn, &ds, NULL, na)) {
225 if (dn < 0 || (dn == 0 && ds < 0)) {
226 fprintf(out, " expiry: EXPIRED\n");
227 } else {
228 fprintf(out, " expiry: %dd %dh left\n", dn, ds / 3600);
229 }
230 }
231 if (ASN1_TIME_diff(&bn, &bs, NULL, nb) && (bn > 0 || bs > 0)) {
232 fprintf(out, " skew: notBefore is in the future "
233 "(local clock behind / cert not yet valid)\n");
234 }
235 }
236
237 /* VOMS attribute certificate */
238 {
239 int loc = -1, n = X509_get_ext_count(cert), i;
240 for (i = 0; i < n; i++) {
241 X509_EXTENSION *ex = X509_get_ext(cert, i);
242 ASN1_OBJECT *obj = X509_EXTENSION_get_object(ex);
243 char oid[128];
244 OBJ_obj2txt(oid, sizeof(oid), obj, 1);
245 if (strcmp(oid, VOMS_AC_OID) == 0) {
246 loc = i;
247 break;
248 }
249 }
250 if (loc < 0) {
251 fprintf(out, " VOMS: none\n");
252 } else {
253 ASN1_OCTET_STRING *data =
254 X509_EXTENSION_get_data(X509_get_ext(cert, loc));
255 voms_scan(ASN1_STRING_get0_data(data), ASN1_STRING_length(data), out);
256 }
257 }
258 X509_free(cert);
259}

261/* Phase 40 (c): client-side pre-flight / auth-failure diagnostics */
262/*
263 * WHAT: inspect whatever credential is present locally (a discoverable bearer
264 * token, then a GSI proxy at the default path) and print a specific,
265 * actionable hint to `out` when a likely auth problem is detectable WITHOUT
266 * contacting the server — an expired or near-expiry token/proxy, or (when
267 * want_write) a token that grants read scope only. Returns 1 if a
268 * likely-fatal local problem was found, else 0.
269 * WHY: so the user sees the real cause instantly ("token expired 3m ago",
270 * "proxy expired — run xrdgsiproxy init", "token is read-only for a write")
271 * instead of digging after a bare "permission denied". Pure local
272 * inspection: no network, no signature verification (it is a hint, not a
273 * gate — the server remains authoritative).
274 */
275int
276brix_cred_diagnose(int want_write, const char *prefix, FILE *out)
277{
278 int problem = 0;
279 char *tok;
280 char pxp[1024];
281 long left = 0;
282
283 if (out == NULL) {
284 return 0;
285 }
286 if (prefix == NULL) {
287 prefix = "";
288 }
289
290 /* Bearer token, if one is discoverable (env / file / XDG / /tmp). */
291 tok = brix_token_discover();
292 if (tok != NULL) {
293 brix_token_meta m;
294 memset(&m, 0, sizeof(m));
295 brix_token_meta_get(tok, &m);
296 if (m.valid) {
297 if (m.expired) {
298 fprintf(out, "%sbearer token has EXPIRED (exp %ld, %ld s ago) — "
299 "obtain a fresh token and retry\n",
300 prefix, m.exp, (long) time(NULL) - m.exp);
301 problem = 1;
302 } else if (want_write && m.has_scope && !m.has_write) {
303 fprintf(out, "%sbearer token grants READ scope only — this write "
304 "needs a storage.write / create / modify scope\n",
305 prefix);
306 problem = 1;
307 } else if (m.exp > 0) {
308 long secs = m.exp - (long) time(NULL);
309 if (secs >= 0 && secs < 300) {
310 fprintf(out, "%sbearer token expires in %ld s — it may expire "
311 "mid-transfer\n", prefix, secs);
312 }
313 }
314 }
315 free(tok);
316 }
317
318 /* GSI proxy, if one is present at the default path. */
319 brix_proxy_default_path(pxp, sizeof(pxp));
320 if (access(pxp, R_OK) == 0 && brix_proxy_remaining(pxp, &left) == 0) {
321 if (left <= 0) {
322 fprintf(out, "%sGSI proxy has EXPIRED (%s) — run 'xrdgsiproxy init'\n",
323 prefix, pxp);
324 problem = 1;
325 } else if (left < 300) {
326 fprintf(out, "%sGSI proxy expires in %ld s (%s) — it may expire "
327 "mid-transfer\n", prefix, left, pxp);
328 }
329 }
330
331 return problem;
332}

334/*
335 * Print a credential hint at an app's error-reporting site, but only when the
336 * status carries an auth/authz wire error — so an ordinary ENOENT/EIO failure
337 * does not trigger credential noise. Indented under the primary error line.
338 * Also fires the doctor-referral hint (spec WS-7) so users know that
339 * `xrddiag check <endpoint>` can walk through the auth handshake for them —
340 * with the endpoint string when the caller has one (url_str, nullable).
341 *
342 * This _url variant is the canonical implementation; the historical
343 * brix_cred_hint_for_status signature delegates with url_str=NULL so its
344 * many existing call sites keep working unchanged. brix_hint_doctor_referral
345 * self-gates on the auth class (incl. the local XRDC_EAUTH sentinel, which is
346 * broader than the wire-code gate used for the credential diagnosis) and
347 * dedupes via brix_cli_hint_once("doctor").
348 */
349void
350brix_cred_hint_for_status_url(const brix_status *st, int want_write, FILE *out,
351 const char *url_str)
352{
353 if (st == NULL || out == NULL) {
354 return;
355 }
356 if (st->kxr == kXR_NotAuthorized || st->kxr == kXR_AuthFailed) {
357 (void) brix_cred_diagnose(want_write, " hint: ", out);
358 }
359 /* Spec WS-7: offer the xrddiag referral (self-gated on auth class). */
360 brix_hint_doctor_referral(st, url_str);
361}
