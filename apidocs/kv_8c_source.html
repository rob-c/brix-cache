1/*
2 * kv.c — generic cross-worker key/value store in nginx shared memory.
3 *
4 * The table is allocated FROM the zone's slab pool (via brix_shm_table_alloc)
5 * and published through shm_zone->data — it does NOT overlay shm.addr, which
6 * holds nginx's ngx_slab_pool_t header. That header must stay intact because
7 * ngx_unlock_mutexes() (run on every child death) force-unlocks the slab mutex
8 * at shm.addr; clobbering it SIGSEGVs the master. Layout of the slab block:
9 *
10 * +----------------------+ table base (shm_zone->data)
11 * | brix_kv_header_t | (lock must be first — ngx_shmtx_create target)
12 * +----------------------+ offset sizeof(header)
13 * | entry[0] | stride = sizeof(entry) + key_max + val_max
14 * | entry[1] |
15 * | ... |
16 * | entry[capacity-1] |
17 * +----------------------+
18 *
19 * capacity is the largest power of two that fits the configured zone size
20 * (computed at configure time), so (hash & (capacity-1)) selects the home
21 * bucket without a modulo. The requested zone is padded by
22 * brix_shm_zone_size() to cover the slab-pool overhead.
23 */
24#include "core/ngx_brix_module.h" /* full ngx core + stream (NGX_STREAM_MAIN_CONF) */
25#include "kv.h"
26#include "core/compat/shm_slots.h"
27#include "core/compat/alloc_guard.h"

59static uint64_t
60brix_kv_hash(const void *key, size_t len)
61{
62 const uint8_t *p = key;
63 uint64_t h = 14695981039346656037ULL; /* FNV offset basis */
64 size_t i;
65
66 for (i = 0; i < len; i++) {
67 h ^= p[i];
68 h *= 1099511628211ULL; /* FNV prime */
69 }
70 return h;
71}

73static uint32_t
74brix_kv_floor_pow2(size_t n)
75{
76 uint32_t p = 1;
77
78 if (n == 0) {
79 return 0;
80 }
81 while ((size_t) (p << 1) <= n && (p << 1) != 0) {
82 p <<= 1;
83 }
84 return p;
85}

87static brix_kv_header_t *
88brix_kv_hdr(brix_kv_t *kv)
89{
90 /*
91 * The table (header + entry array) is slab-allocated and published via
92 * shm_zone->data by brix_shm_table_alloc() — it does NOT sit at
93 * shm.addr (that holds nginx's ngx_slab_pool_t header, which must stay
94 * intact so ngx_unlock_mutexes() can force-unlock the slab mutex on child
95 * death). Until init runs, zone->data still holds the (void*)1 pending
96 * sentinel or the kv handle, so guard against a non-table pointer.
97 */
98 if (kv == NULL || kv->zone == NULL || kv->zone->data == NULL
99 || kv->zone->data == kv || kv->zone->data == (void *) 1)
100 {
101 return NULL;
102 }
103 return (brix_kv_header_t *) kv->zone->data;
104}

106static brix_kv_entry_t *
107brix_kv_slot(brix_kv_header_t *h, size_t stride, uint32_t i)
108{
109 return (brix_kv_entry_t *)
110 ((u_char *) h + sizeof(brix_kv_header_t) + (size_t) i * stride);
111}
112
113/*
114 * Backward-shift deletion (Knuth Algorithm R): empty the slot at `hole`, then
115 * pull forward any following entries whose home bucket lets them fill the gap,
116 * preserving the linear-probe invariant so no live entry becomes unreachable.
117 */
118static ngx_uint_t
119brix_kv_should_shift(uint32_t home, uint32_t hole, uint32_t cur)
120{
121 /* Shift iff `home` is NOT in the cyclic interval (hole, cur]. */
122 if (hole <= cur) {
123 return !(home > hole && home <= cur);
124 }
125 return !(home > hole || home <= cur);
126}

128static void
129brix_kv_remove_at(brix_kv_header_t *h, size_t stride, uint32_t mask,
130 uint32_t hole)
131{
132 uint32_t cur = hole;
133
134 for ( ;; ) {
135 brix_kv_slot(h, stride, hole)->key_len = 0;
136
137 for ( ;; ) {
138 brix_kv_entry_t *e;
139 uint32_t home;
140
141 cur = (cur + 1) & mask;
142 e = brix_kv_slot(h, stride, cur);
143 if (e->key_len == 0) {
144 return; /* end of probe chain */
145 }
146 home = (uint32_t) (e->hash & mask);
147 if (brix_kv_should_shift(home, hole, cur)) {
148 break; /* this entry can fill the hole */
149 }
150 }
151
152 ngx_memcpy(brix_kv_slot(h, stride, hole),
153 brix_kv_slot(h, stride, cur), stride);
154 hole = cur;
155 }
156}

159static ngx_int_t
160brix_kv_init_zone(ngx_shm_zone_t *shm_zone, void *data)
161{
162 brix_kv_t *kv = shm_zone->data; /* set by configure */
163 brix_kv_header_t *h;
164 ngx_flag_t fresh;
165
166 /*
167 * Allocate the table FROM the slab pool (not over shm.addr), so nginx's
168 * ngx_slab_pool_t header survives ngx_unlock_mutexes() on child death.
169 * The helper handles fresh-alloc, reload (data != NULL), and re-attach
170 * (shm.exists), publishes the table via shm_zone->data, and creates the
171 * process-local mutex handle from the table's first member (the lock).
172 */
173 h = brix_shm_table_alloc(shm_zone, data, kv->table_bytes,
174 &kv->mutex, &fresh);
175 if (h == NULL) {
176 return NGX_ERROR;
177 }
178
179 if (fresh) {
180 /* Brand-new table: initialise the layout fields. The helper already
181 * zeroed the region and created the mutex, so live counters
182 * (count/hits/misses/evictions) start at 0 and must NOT be reset on
183 * reuse. */
184 h->capacity = kv->capacity;
185 h->key_max = (uint32_t) kv->key_max;
186 h->val_max = (uint32_t) kv->val_max;
187 }
188 return NGX_OK;
189}

191ngx_int_t
192brix_kv_configure(ngx_conf_t *cf, brix_kv_t *kv, ngx_str_t *name,
193 size_t size, size_t key_max, size_t val_max, void *module)
194{
195 size_t avail, stride, table_bytes, zone_size;
196 uint32_t cap;
197
198 if (brix_kv_nzones >= BRIX_KV_MAX_ZONES) {
199 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
200 "too many brix_kv zones (max %d)",
201 BRIX_KV_MAX_ZONES);
202 return NGX_ERROR;
203 }
204 if (size < BRIX_KV_MIN_SIZE) {
205 size = BRIX_KV_MIN_SIZE;
206 }
207
208 /*
209 * Derive the bucket count from the requested size exactly as before
210 * (size buys capacity), then size the slab-backed table region from it.
211 * The table now lives in slab memory, so the zone we actually request is
212 * padded by brix_shm_zone_size() to cover the ngx_slab_pool_t header,
213 * the page-management array, and slab rounding.
214 */
215 stride = sizeof(brix_kv_entry_t) + key_max + val_max;
216 avail = size - sizeof(brix_kv_header_t);
217 cap = brix_kv_floor_pow2(avail / stride);
218 if (cap < 1) {
219 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
220 "brix_kv_zone \"%V\": size too small for one entry",
221 name);
222 return NGX_ERROR;
223 }
224 table_bytes = sizeof(brix_kv_header_t) + (size_t) cap * stride;
225 zone_size = brix_shm_zone_size(table_bytes);
226
227 kv->name = *name;
228 kv->size = size;
229 kv->key_max = key_max;
230 kv->val_max = val_max;
231 kv->capacity = cap;
232 kv->table_bytes = table_bytes;
233
234 kv->zone = ngx_shared_memory_add(cf, name, zone_size, module);
235 if (kv->zone == NULL) {
236 return NGX_ERROR;
237 }
238 if (kv->zone->data != NULL) {
239 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
240 "duplicate brix_kv_zone \"%V\"", name);
241 return NGX_ERROR;
242 }
243
244 kv->zone->init = brix_kv_init_zone;
245 kv->zone->data = kv;
246
247 brix_kv_zones[brix_kv_nzones++] = kv;
248 return NGX_OK;
249}

252int
253brix_kv_get(brix_kv_t *kv, const void *key, size_t key_len,
254 void *out, size_t *out_len)
255{
256 brix_kv_header_t *h = brix_kv_hdr(kv);
257 uint64_t hash;
258 size_t stride;
259 uint32_t mask, maxprobe, idx, p;
260 ngx_msec_t now;
261 int result = 0;
262
263 if (h == NULL || key_len == 0 || key_len > kv->key_max) {
264 return 0;
265 }
266
267 hash = brix_kv_hash(key, key_len);
268 stride = sizeof(brix_kv_entry_t) + kv->key_max + kv->val_max;
269 now = ngx_current_msec;
270
271 ngx_shmtx_lock(&kv->mutex);
272
273 mask = h->capacity - 1;
274 maxprobe = h->capacity / 2;
275 idx = (uint32_t) (hash & mask);
276
277 for (p = 0; p < maxprobe; p++) {
278 brix_kv_entry_t *e = brix_kv_slot(h, stride, idx);
279
280 if (e->key_len == 0) {
281 break; /* probe chain ends — not found */
282 }
283 if (e->hash == hash && e->key_len == key_len
284 && ngx_memcmp((u_char *) e + sizeof(*e), key, key_len) == 0)
285 {
286 if (e->expires != 0 && e->expires <= now) {
287 brix_kv_remove_at(h, stride, mask, idx);
288 if (h->count > 0) { h->count--; }
289 h->evictions++;
290 break; /* expired — treat as miss */
291 }
292 if (out != NULL && out_len != NULL) {
293 size_t vl = e->val_len;
294 if (vl > *out_len) { vl = *out_len; }
295 ngx_memcpy(out, (u_char *) e + sizeof(*e) + kv->key_max, vl);
296 *out_len = vl;
297 }
298 result = 1;
299 break;
300 }
301 idx = (idx + 1) & mask;
302 }
303
304 if (result) { h->hits++; } else { h->misses++; }
305
306 ngx_shmtx_unlock(&kv->mutex);
307 return result;
308}

310ngx_int_t
311brix_kv_set(brix_kv_t *kv, const void *key, size_t key_len,
312 const void *val, size_t val_len, ngx_msec_t ttl_ms)
313{
314 brix_kv_header_t *h = brix_kv_hdr(kv);
315 uint64_t hash;
316 size_t stride;
317 uint32_t mask, maxprobe, idx, p;
318 ngx_msec_t now;
319 ngx_int_t rc = NGX_ERROR;
320
321 if (h == NULL
322 || key_len == 0 || key_len > kv->key_max
323 || val_len > kv->val_max)
324 {
325 return NGX_ERROR;
326 }
327
328 hash = brix_kv_hash(key, key_len);
329 stride = sizeof(brix_kv_entry_t) + kv->key_max + kv->val_max;
330 now = ngx_current_msec;
331
332 ngx_shmtx_lock(&kv->mutex);
333
334 mask = h->capacity - 1;
335 maxprobe = h->capacity / 2;
336 idx = (uint32_t) (hash & mask);
337
338 for (p = 0; p < maxprobe; p++) {
339 brix_kv_entry_t *e = brix_kv_slot(h, stride, idx);
340
341 if (e->key_len == 0) {
342 /* New insert — enforce the 0.5 load-factor cap. */
343 if (h->count >= h->capacity / 2) {
344 break;
345 }
346 e->hash = hash;
347 e->key_len = (uint32_t) key_len;
348 e->val_len = (uint32_t) val_len;
349 e->expires = ttl_ms ? (now + ttl_ms) : 0;
350 ngx_memcpy((u_char *) e + sizeof(*e), key, key_len);
351 if (val_len) {
352 ngx_memcpy((u_char *) e + sizeof(*e) + kv->key_max,
353 val, val_len);
354 }
355 h->count++;
356 rc = NGX_OK;
357 break;
358 }
359 if (e->hash == hash && e->key_len == key_len
360 && ngx_memcmp((u_char *) e + sizeof(*e), key, key_len) == 0)
361 {
362 /* Overwrite existing key. */
363 e->val_len = (uint32_t) val_len;
364 e->expires = ttl_ms ? (now + ttl_ms) : 0;
365 if (val_len) {
366 ngx_memcpy((u_char *) e + sizeof(*e) + kv->key_max,
367 val, val_len);
368 }
369 rc = NGX_OK;
370 break;
371 }
372 idx = (idx + 1) & mask;
373 }
374
375 ngx_shmtx_unlock(&kv->mutex);
376 return rc;
377}
