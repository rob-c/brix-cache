3/*
4 * WHAT: Prometheus HTTP exporter for WebDAV protocol counters.
5 * WHY: WebDAV requests (GET, PUT, COPY, PROPFIND, etc.) need their own counter families
6 * separate from native XRootD stream counters — different methods, auth paths, TPC model,
7 * CORS handling, fd cache, and range request semantics. All exported in Prometheus text
8 * exposition format on /metrics endpoint.
9 * HOW: Static name-table arrays map enum indices to label strings (method, status class, auth result,
10 * range outcome, PUT body mode, TPC event, CORS decision, PROPFIND depth). brix_export_webdav_metrics()
11 * iterates each counter slot via ngx_atomic_fetch_add(..., 0) for lock-free reads, emits HELP/TYPE/value
12 * lines per metric family. Labels are low-cardinality enums only — no paths/bucket-names/DNs as labels.
13 */

85/*
86 * WHAT: Emit every WebDAV counter family to the Prometheus writer.
87 * WHY: WebDAV has its own counter set (distinct from native-stream metrics):
88 * per-method requests/responses, auth outcomes, byte tallies, range/PUT
89 * modes, PROPFIND depth, HTTP-TPC, CORS, and credential delegation.
90 * HOW: Most families are single-label (one axis), so they delegate to the
91 * mw_emit_labeled() helper, which iterates a counter array against a name
92 * table and emits the HELP/TYPE/rows block. The one exception is
93 * responses_total, which is TWO-dimensional (method x status_class) and so
94 * is hand-rolled below with a nested loop. mw_emit_labeled and mw_printf
95 * both read counters atomically (ngx_atomic_fetch_add(&x, 0) == lock-free
96 * load), so this scrape never blocks the request hot path.
97 */
98void
99brix_export_webdav_metrics(metrics_writer_t *mw,
100 ngx_brix_metrics_t *shm)
101{
102 ngx_uint_t method, status;
103
104 mw_emit_labeled(mw,
105 "brix_webdav_requests_total",
106 "WebDAV requests received, by HTTP/WebDAV method.",
107 "method",
108 brix_webdav_method_names, BRIX_WEBDAV_NMETHODS,
109 shm->webdav.requests_total);
110
111 /* responses_total: the only 2-D family (method x status_class), so it cannot
112 * use mw_emit_labeled (single-axis). HELP/TYPE is emitted once, then the
113 * nested loop emits one row per (method, status_class) cell of the SHM
114 * matrix shm->webdav.responses_total[method][status]. */
115 mw_printf(mw,
116 "# HELP brix_webdav_responses_total "
117 "WebDAV responses by method and HTTP status class.\n"
118 "# TYPE brix_webdav_responses_total counter\n");
119 for (method = 0; method < BRIX_WEBDAV_NMETHODS; method++) {
120 for (status = 0; status < BRIX_HTTP_NSTATUS; status++) {
121 mw_printf(mw,
122 "brix_webdav_responses_total"
123 "{method=\"%s\",status_class=\"%s\"} %lu\n",
124 brix_webdav_method_names[method],
125 brix_http_status_names[status],
126 (unsigned long) ngx_atomic_fetch_add(
127 &shm->webdav.responses_total[method][status], 0));
128 }
129 }
130
131 mw_emit_labeled(mw,
132 "brix_webdav_auth_total",
133 "WebDAV authentication outcomes.",
134 "result",
135 brix_webdav_auth_names, BRIX_WEBDAV_NAUTH_RESULTS,
136 shm->webdav.auth_total);
137
