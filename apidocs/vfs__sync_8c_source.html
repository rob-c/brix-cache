1/*
2 * vfs_sync.c — VFS handle-level truncate and durability.
3 *
4 * WHAT: Implements brix_vfs_truncate() (resize an open handle to `length`) and
5 * brix_vfs_sync() (flush an open handle to stable storage).
6 *
7 * WHY: kXR_truncate and the sync/commit step of writes (kXR_sync, WebDAV PUT
8 * finalisation) operate on an already-open handle rather than a path, so
9 * they live with the file-handle ops; truncate must also keep the handle's
10 * cached size in step with the file.
11 *
12 * HOW: truncate validates the handle/fd and a non-negative length, runs a VFS
13 * I/O-core TRUNCATE job, and on success updates fh->size so later reads
14 * see the new length. sync validates the handle and runs a VFS I/O-core
15 * SYNC job. Both are unmetered handle operations (the surrounding write
16 * op records the metric) returning NGX_OK / NGX_ERROR with errno set.
17 */
18#include "vfs_internal.h"
19#include "vfs_io_core.h"

21/* Resize the open handle to `length` (ftruncate) and update the cached
22 * fh->size. NGX_ERROR with errno set on a bad handle or negative length. */
23ngx_int_t
24brix_vfs_truncate(brix_vfs_file_t *fh, off_t length)
25{
26 brix_vfs_job_t job;
27
28 if (fh == NULL || fh->obj.fd == NGX_INVALID_FILE || length < 0) {
29 errno = EINVAL;
30 return NGX_ERROR;
31 }
32
33 /* phase-71: capability gate — a backend without CAP_TRUNCATE rejects resize
34 * uniformly rather than issuing an ftruncate the driver cannot honor. */
35 if (fh->obj.inst != NULL
36 && !(brix_sd_caps(fh->obj.inst) & BRIX_SD_CAP_TRUNCATE))
37 {
38 errno = ENOTSUP;
39 return NGX_ERROR;
40 }
41
42 brix_vfs_job_truncate_init(&job, fh->obj.fd, length);
43 brix_vfs_io_execute(&job);
44 if (job.io_errno != 0) {
45 errno = job.io_errno;
46 return NGX_ERROR;
47 }
48
49 fh->size = length;
50 return NGX_OK;
51}

53/* Flush the open handle to stable storage (fsync). NGX_ERROR with errno set on
54 * a bad handle or fsync failure. */
55ngx_int_t
56brix_vfs_sync(brix_vfs_file_t *fh)
57{
58 brix_vfs_job_t job;
59
60 if (fh == NULL || fh->obj.fd == NGX_INVALID_FILE) {
61 errno = EINVAL;
62 return NGX_ERROR;
63 }
64
65 brix_vfs_job_sync_init(&job, fh->obj.fd);
66 brix_vfs_io_execute(&job);
67 if (job.io_errno != 0) {
68 errno = job.io_errno;
69 return NGX_ERROR;
70 }
71
72 return NGX_OK;
