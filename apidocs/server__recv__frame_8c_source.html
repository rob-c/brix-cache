1/*
2 * cms/server_recv_frame.c — CMS server-side opcode handlers + dispatch table.
3 *
4 * WHAT: The per-opcode frame handlers for an accepted CMS data-server
5 * connection (LOGIN, kYR_xauth, LOAD, AVAIL/SPACE, PONG, PING, DISC, UPDATE,
6 * STATFS, GONE, CNS, and the unknown-opcode fallback), the login-completion
7 * step, the periodic ping timer, and the data-driven opcode → handler route
8 * table consumed by cms_srv_process_frame().
9 *
10 * WHY: Split out of the former monolithic server_recv.c (Phase-79 file-size
11 * split). Grouping "what each opcode does" behind a single dispatch table keeps
12 * the wire-framing state machine (server_recv.c) and the pure decoders
13 * (server_recv_parse.c) each single-purpose. Handlers reach the lifecycle
14 * helpers and the payload parsers through server_recv_internal.h; only
15 * cms_srv_process_frame is exported (declared there and thus non-static).
16 *
17 * HOW: Each handler shares one signature so the route table can dispatch by
18 * opcode; a handler that tears the connection down leaves ctx->c NULL for the
19 * read loop to observe. cms_srv_process_frame scans the table, falling through
20 * to cms_srv_frame_unknown for codes we do not act on.
21 */
22

51/*
52 * cms_srv_complete_login — register the data server and arm heartbeats.
53 *
54 * Called once the LOGIN frame has been parsed AND (when sss is configured) the
55 * kYR_xauth credential has been verified. Splitting this out lets the sss path
56 * defer registration until authentication succeeds without duplicating the
57 * register/log/timer sequence.
58 */
59static void
60cms_srv_complete_login(brix_cms_srv_ctx_t *ctx)
61{
62 brix_srv_register(ctx->host, ctx->port, ctx->paths,
63 ctx->free_mb, ctx->util_pct);
64 ctx->logged_in = 1;
65 cms_srv_log_registration(ctx);
66
67 /* Arm the periodic ping now that the data server is logged in. */
68 ctx->ping_timer.handler = brix_cms_srv_ping_timer;
69 ngx_add_timer(&ctx->ping_timer, ctx->interval_ms);
70
71 /*
72 * WS3: switch the read deadline from the LOGIN handshake bound to the
73 * post-login idle watchdog. Re-armed on every received frame in
74 * brix_cms_srv_read; if the node goes silent for idle_timeout_ms it is
75 * closed + unregistered + blacklisted (reaps a black-holed node the ping
76 * send cannot detect). 0 = disabled.
77 */
78 if (ctx->idle_timeout_ms > 0 && ctx->c != NULL) {
79 ngx_add_timer(ctx->c->read, ctx->idle_timeout_ms);
80 } else if (ctx->c != NULL && ctx->c->read->timer_set) {
81 ngx_del_timer(ctx->c->read); /* idle watchdog disabled — drop login timer */
82 }
83
84 /* ctx->c is non-NULL for a frame being processed (a close nulls it and
85 * returns), but match the ctx->c guards above for consistency. */
86 if (ctx->c != NULL) {
87 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
88 "brix: CMS server: registered %s:%d paths=[%s] "
89 "free_mb=%uD util_pct=%uD",
90 ctx->host, (int) ctx->port, ctx->paths,
91 ctx->free_mb, ctx->util_pct);
92 }
93}

95/* Per-opcode frame handlers. All share one signature so the dispatcher can be
96 * a data-driven route table (mirrors recv.c's client-side shape); handlers
97 * that need no payload/streamid simply ignore those arguments. A handler that
98 * closes the connection leaves ctx->c NULL — the read loop checks that. */
99
100typedef void (*cms_srv_frame_pt)(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
101 const u_char *payload, size_t payload_len);

103/*
104 * cms_srv_frame_login — LOGIN: parse the CmsLoginData payload; on parse
105 * failure log + close. W1a — if sss cluster auth is required, do NOT register
106 * yet. Mirror the real cmsd manager (XrdCmsLogin::Admit): after the LOGIN
107 * frame arrives, send our security parms and wait for the node's kYR_xauth
108 * credential before admitting it into the registry.
109 */
110static void
111cms_srv_frame_login(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
112 const u_char *payload, size_t payload_len)
113{
114 if (!cms_srv_parse_login(ctx, payload, payload_len)) {
115 ngx_log_error(NGX_LOG_WARN, ctx->c->log, 0,
116 "brix: CMS server: malformed LOGIN from %s",
117 ctx->host);
118 cms_srv_log_auth_fail(ctx, "invalid");
119 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
120 return;
121 }
122
123 if (ctx->auth_state == CMS_AUTH_REQUESTED) {
124 static const u_char sss_parms[] = "&P=sss";
125 if (brix_cms_srv_send_xauth(ctx, sss_parms,
126 sizeof(sss_parms)) != NGX_OK)
127 {
128 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
129 return;
130 }
131 ctx->auth_state = CMS_AUTH_CHALLENGED;
132 return;
133 }
134
135 cms_srv_complete_login(ctx);
136}

138/*
139 * cms_srv_frame_xauth — kYR_xauth: verify the node's sss credential and admit
140 * it. Only meaningful while we are waiting for a credential; anything else
141 * (or a bad signature) is an auth failure and closes the connection.
142 */
143static void
144cms_srv_frame_xauth(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
145 const u_char *payload, size_t payload_len)
146{
147 if (ctx->auth_state != CMS_AUTH_CHALLENGED) {
148 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
149 "brix: CMS server: unexpected kYR_xauth from %s",
150 ctx->host);
151 cms_srv_log_auth_fail(ctx, "invalid");
152 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
153 return;
154 }
155 if (brix_cms_srv_verify_xauth(ctx, payload, payload_len) != NGX_OK) {
156 cms_srv_log_auth_fail(ctx, "bad-signature");
157 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
158 return;
159 }
160 ctx->auth_state = CMS_AUTH_DONE;
161 cms_srv_complete_login(ctx);
162}

260/*
261 * cms_srv_frame_statfs — space query for a path (do_StatFS): answer kYR_data
262 * with aggregate cluster space for the servers that hold the path. We report
263 * a single tier (staging == writable) since we model one storage class.
264 */
265static void
266cms_srv_frame_statfs(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
267 const u_char *payload, size_t payload_len)
268{
269 brix_cms_rrdata_t d;
270 brix_cms_statfs_fields_t space;
271 u_char out[64];
272 uint32_t free_mb = 0, util = 0, num;
273 int olen;
274
275 if (!ctx->logged_in
276 || brix_cms_rrdata_parse(CMS_RR_STATFS, payload, payload_len, &d)
277 != 0
278 || d.path == NULL)
279 {
280 return; /* malformed / pre-auth: ignore */
281 }
282
283 num = (uint32_t) brix_srv_count_matching((const char *) d.path);
284 brix_srv_aggregate_space(&free_mb, &util);
285 space.w_num = num; space.w_free = free_mb; space.w_util = util;
286 space.s_num = num; space.s_free = free_mb; space.s_util = util;
287 olen = brix_cms_statfs_encode(&space, out, sizeof(out));
288 if (olen > 0) {
289 (void) brix_cms_srv_send_data(ctx, streamid, out, (size_t) olen);
290 }
291}

293/*
294 * cms_srv_frame_gone — a data server signals that it no longer holds a
295 * specific path. Remove that path token from its registry entry without
296 * touching the rest of the registration (the server still holds other paths).
297 */
298static void
299cms_srv_frame_gone(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
300 const u_char *payload, size_t payload_len)
301{
302 char path[BRIX_SRV_MAX_PATHS];
303 size_t copy;
304
305 if (!ctx->logged_in || payload_len == 0) {
306 return;
307 }
308
309 copy = payload_len < sizeof(path) - 1
310 ? payload_len : sizeof(path) - 1;
311 ngx_memcpy(path, payload, copy);
312 path[copy] = '\0';
313 brix_srv_unregister_path(ctx->host, ctx->port, path);
314 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
315 "brix: CMS server: kYR_gone path=%s from %s:%d",
316 path, ctx->host, (int) ctx->port);
317}

319/*
320 * cms_srv_frame_cns — §6 Composite Cluster Name Space: a data server reports a
321 * namespace mutation. Apply it into the manager inventory (collect mode only).
322 */
323static void
324cms_srv_frame_cns(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
325 const u_char *payload, size_t payload_len)
326{
327 uint8_t op;
328 uint64_t size, mtime;
329 char path[BRIX_CNS_PATH_MAX + 1];
330 uint32_t server_id;
331 const u_char *h = (const u_char *) ctx->host;
332 size_t i;
333
334 if (!ctx->logged_in || !brix_cns_collecting()) {
335 return;
336 }
337
338 if (brix_cns_event_decode(payload, payload_len, &op, &size, &mtime,
339 path, sizeof(path)) != NGX_OK)
340 {
341 ngx_log_error(NGX_LOG_WARN, ctx->c->log, 0,
342 "brix: CMS server: malformed CNS event from %s",
343 ctx->host);
344 return;
345 }
346 server_id = (uint32_t) ctx->port; /* host hash + port = origin id */
347 for (i = 0; h[i] != '\0'; i++) {
348 server_id = server_id * 33 + h[i];
349 }
350 (void) brix_cns_apply(op, path, size, mtime, server_id);
351 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
352 "brix: CNS event op=%ud path=%s from %s",
353 (unsigned) op, path, ctx->host);
354}

358 * log distinguishes a valid manager opcode we have not wired yet (e.g.

364{

372 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,

400/*
401 * cms_srv_process_frame — route one complete frame by opcode through the
402 * table above. A handler that tears the connection down leaves ctx->c NULL;
403 * the read loop checks that after every dispatch.
404 */
405void
406cms_srv_process_frame(brix_cms_srv_ctx_t *ctx, u_char code, uint32_t streamid,
407 const u_char *payload, size_t payload_len)
408{
409 ngx_uint_t i;
410
411 for (i = 0; cms_srv_frame_routes[i].handler != NULL; i++) {
412 if (cms_srv_frame_routes[i].code == code) {
413 cms_srv_frame_routes[i].handler(ctx, streamid,
414 payload, payload_len);
415 return;
416 }
417 }
418
419 cms_srv_frame_unknown(ctx, code);
420}
