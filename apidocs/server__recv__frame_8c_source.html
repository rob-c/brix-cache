1/*
2 * cms/server_recv_frame.c — CMS server-side opcode handlers + dispatch table.
3 *
4 * WHAT: The per-opcode frame handlers for an accepted CMS data-server
5 * connection (LOGIN, kYR_xauth, LOAD, AVAIL/SPACE, PONG, PING, DISC, UPDATE,
6 * STATFS, USAGE, STATS, STATUS, GONE, CNS, and the unknown-opcode fallback), the
7 * login-completion
8 * step, the periodic ping timer, and the data-driven opcode → handler route
9 * table consumed by cms_srv_process_frame().
10 *
11 * WHY: Split out of the former monolithic server_recv.c (Phase-79 file-size
12 * split). Grouping "what each opcode does" behind a single dispatch table keeps
13 * the wire-framing state machine (server_recv.c) and the pure decoders
14 * (server_recv_parse.c) each single-purpose. Handlers reach the lifecycle
15 * helpers and the payload parsers through server_recv_internal.h; only
16 * cms_srv_process_frame is exported (declared there and thus non-static).
17 *
18 * HOW: Each handler shares one signature so the route table can dispatch by
19 * opcode; a handler that tears the connection down leaves ctx->c NULL for the
20 * read loop to observe. cms_srv_process_frame scans the table, falling through
21 * to cms_srv_frame_unknown for codes we do not act on.
22 */
23

36/*
37 * cms_srv_blfile_tick — enforce the operator blacklist file (Phase-89 W6′).
38 *
39 * Piggybacked on the per-connection ping cadence (and called with force=1
40 * right after a registration) instead of a dedicated timer: the file only
41 * matters while data servers are registered, and registered servers are
42 * exactly the ones with a live ping timer. The poll self-rate-limits its
43 * stat() to 1/s, so many nodes sharing one srv block stay cheap. The
44 * blacklist duration is 3x the ping interval — always longer than the gap to
45 * the next re-assert, so a file-listed server stays down (over undrain and
46 * over the register-time blacklist clear) until the line is removed.
47 */
48static void
49cms_srv_blfile_tick(brix_cms_srv_ctx_t *ctx, ngx_uint_t force,
50 ngx_log_t *log)
51{
52 brix_cms_blfile_poll(&ctx->conf->blfile, &ctx->conf->blacklist_file,
53 ctx->interval_ms * 3, force, log);
54}

77/*
78 * cms_srv_complete_login — register the data server and arm heartbeats.
79 *
80 * Called once the LOGIN frame has been parsed AND (when sss is configured) the
81 * kYR_xauth credential has been verified. Splitting this out lets the sss path
82 * defer registration until authentication succeeds without duplicating the
83 * register/log/timer sequence.
84 */
85static void
86cms_srv_complete_login(brix_cms_srv_ctx_t *ctx)
87{
88 brix_srv_register(ctx->host, ctx->port, ctx->paths,
89 ctx->free_mb, ctx->util_pct);
90 if (ctx->vnid[0] != '\0') {
91 brix_srv_set_vnid(ctx->host, ctx->port, ctx->vnid);
92 }
93 ctx->logged_in = 1;
94 cms_srv_log_registration(ctx);
95
96 /* W3: join this worker's fan-out set — locate misses can now probe this
97 * node with kYR_state. Removed again in brix_cms_srv_close. */
98 brix_cms_srv_node_add(ctx);
99
100 /* W6′: brix_srv_register just cleared any prior blacklist — re-assert
101 * the file immediately so a listed node is never selectable, not even
102 * for one poll period. */
103 if (ctx->c != NULL) {
104 cms_srv_blfile_tick(ctx, 1, ctx->c->log);
105 }
106
107 /* Arm the periodic ping now that the data server is logged in. */
108 ctx->ping_timer.handler = brix_cms_srv_ping_timer;
109 ngx_add_timer(&ctx->ping_timer, ctx->interval_ms);
110
111 /*
112 * WS3: switch the read deadline from the LOGIN handshake bound to the
113 * post-login idle watchdog. Re-armed on every received frame in
114 * brix_cms_srv_read; if the node goes silent for idle_timeout_ms it is
115 * closed + unregistered + blacklisted (reaps a black-holed node the ping
116 * send cannot detect). 0 = disabled.
117 */
118 if (ctx->idle_timeout_ms > 0 && ctx->c != NULL) {
119 ngx_add_timer(ctx->c->read, ctx->idle_timeout_ms);
120 } else if (ctx->c != NULL && ctx->c->read->timer_set) {
121 ngx_del_timer(ctx->c->read); /* idle watchdog disabled — drop login timer */
122 }
123
124 /* ctx->c is non-NULL for a frame being processed (a close nulls it and
125 * returns), but match the ctx->c guards above for consistency. */
126 if (ctx->c != NULL) {
127 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
128 "brix: CMS server: registered %s:%d paths=[%s] "
129 "free_mb=%uD util_pct=%uD",
130 ctx->host, (int) ctx->port, ctx->paths,
131 ctx->free_mb, ctx->util_pct);
132 }
133}

135/* Per-opcode frame handlers. All share one signature so the dispatcher can be
136 * a data-driven route table (mirrors recv.c's client-side shape); handlers
137 * that need no payload/streamid simply ignore those arguments. A handler that
138 * closes the connection leaves ctx->c NULL — the read loop checks that. */

143/*
144 * cms_srv_frame_login — LOGIN: parse the CmsLoginData payload; on parse
145 * failure log + close. W1a — if sss cluster auth is required, do NOT register
146 * yet. Mirror the real cmsd manager (XrdCmsLogin::Admit): after the LOGIN
147 * frame arrives, send our security parms and wait for the node's kYR_xauth
148 * credential before admitting it into the registry.
149 */
150static void
151cms_srv_frame_login(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
152 const u_char *payload, size_t payload_len)
153{
154 if (!cms_srv_parse_login(ctx, payload, payload_len)) {
155 ngx_log_error(NGX_LOG_WARN, ctx->c->log, 0,
156 "brix: CMS server: malformed LOGIN from %s",
157 ctx->host);
158 cms_srv_log_auth_fail(ctx, "invalid");
159 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
160 return;
161 }
162
163 if (ctx->auth_state == CMS_AUTH_REQUESTED) {
164 static const u_char sss_parms[] = "&P=sss";
165 if (brix_cms_srv_send_xauth(ctx, sss_parms,
166 sizeof(sss_parms)) != NGX_OK)
167 {
168 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
169 return;
170 }
171 ctx->auth_state = CMS_AUTH_CHALLENGED;
172 return;
173 }
174
175 cms_srv_complete_login(ctx);
176}

178/*
179 * cms_srv_frame_xauth — kYR_xauth: verify the node's sss credential and admit
180 * it. Only meaningful while we are waiting for a credential; anything else
181 * (or a bad signature) is an auth failure and closes the connection.
182 */
183static void
184cms_srv_frame_xauth(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
185 const u_char *payload, size_t payload_len)
186{
187 if (ctx->auth_state != CMS_AUTH_CHALLENGED) {
188 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
189 "brix: CMS server: unexpected kYR_xauth from %s",
190 ctx->host);
191 cms_srv_log_auth_fail(ctx, "invalid");
192 cms_srv_fail_close(ctx, BRIX_SESS_END_ERROR);
193 return;
194 }
195 if (brix_cms_srv_verify_xauth(ctx, payload, payload_len) != NGX_OK) {
196 cms_srv_log_auth_fail(ctx, "bad-signature");

198 return;
199 }
200 ctx->auth_state = CMS_AUTH_DONE;
201 cms_srv_complete_login(ctx);
202}

303/*
304 * cms_srv_frame_statfs — space query for a path (do_StatFS): answer kYR_data
305 * with aggregate cluster space for the servers that hold the path. We report
306 * a single tier (staging == writable) since we model one storage class.
307 */
308static void
309cms_srv_frame_statfs(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
310 const u_char *payload, size_t payload_len)
311{
312 brix_cms_rrdata_t d;
313 brix_cms_statfs_fields_t space;
314 u_char out[64];
315 uint32_t free_mb = 0, util = 0, num;
316 int olen;
317
318 if (!ctx->logged_in
319 || brix_cms_rrdata_parse(CMS_RR_STATFS, payload, payload_len, &d)
320 != 0
321 || d.path == NULL)
322 {
323 return; /* malformed / pre-auth: ignore */
324 }
325
326 num = (uint32_t) brix_srv_count_matching((const char *) d.path);
327 brix_srv_aggregate_space(&free_mb, &util);
328 space.w_num = num; space.w_free = free_mb; space.w_util = util;
329 space.s_num = num; space.s_free = free_mb; space.s_util = util;
330 olen = brix_cms_statfs_encode(&space, out, sizeof(out));
331 if (olen > 0) {
332 (void) brix_cms_srv_send_data(ctx, streamid, out, (size_t) olen);
333 }
334}

336/*
337 * cms_srv_frame_status — kYR_status from a node: the phase-89 W9 state
338 * machine. Modifier bits (read from the frame header at ctx->inbuf[5], the
339 * dispatch table forwards only code/streamid/payload):
340 * suspend → stop selecting the node (drain until it resumes; bounded at
341 * 24 h so a node that dies suspended eventually ages out via the
342 * normal stale-entry path instead of pinning a slot forever);
343 * resume → immediately selectable again (clears the suspend drain);
344 * reset → forget cached metrics/fault state, node re-announces
345 * (brix_srv_reset contract);
346 * stage/nostage → record staging availability for stage-aware selection.
347 * Unknown/empty modifiers are a no-op, matching stock cmsd; pre-login status
348 * frames are ignored.
349 */
350static void
351cms_srv_frame_status(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
352 const u_char *payload, size_t payload_len)
353{
354 u_char mod = ctx->inbuf[5];
355
356 if (!ctx->logged_in) {
357 return; /* pre-auth: ignore */
358 }
359
360 if (mod & CMS_ST_SUSPEND) {
361 brix_srv_blacklist(ctx->host, ctx->port, 24 * 60 * 60 * 1000);
362 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
363 "brix: CMS server: %s:%d suspended",
364 ctx->host, (int) ctx->port);
365 } else if (mod & CMS_ST_RESUME) {
366 (void) brix_srv_undrain(ctx->host, ctx->port);
367 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
368 "brix: CMS server: %s:%d resumed",
369 ctx->host, (int) ctx->port);
370 }
371
372 if (mod & CMS_ST_RESET) {
373 (void) brix_srv_reset(ctx->host, ctx->port);
374 ngx_log_error(NGX_LOG_NOTICE, ctx->c->log, 0,
375 "brix: CMS server: %s:%d state reset",
376 ctx->host, (int) ctx->port);
377 }
378
379 if (mod & CMS_ST_STAGE) {
380 brix_srv_set_stage(ctx->host, ctx->port, 1);
381 } else if (mod & CMS_ST_NOSTAGE) {
382 brix_srv_set_stage(ctx->host, ctx->port, 0);
383 }
384}

387 * cms_srv_frame_usage — load/usage query (do_Usage): answer kYR_load with our
388 * 6-byte load vector + aggregate free MB, echoing the streamid. The five
389 * non-dsk bytes come from the Phase-89 W4 machine meter (state lives on the
390 * srv conf — one meter per CMS server block, sampling the shared machine);
391 * dsk carries the aggregate utilisation percentage.

410/*
411 * cms_srv_frame_stats — statistics query (do_Stats), size form: answer
412 * kYR_data with a raw 4-byte big-endian length = the buffer a full stats
413 * reply would need. Stock cmsd's kYR_size modifier asks exactly this so the
414 * caller can allocate before requesting the text form; we answer the size
415 * form for every modifier until a text renderer exists (the dispatch table
416 * does not forward the modifier byte), which callers handle as "allocate,
417 * then re-ask" — the honest subset of do_Stats.
418 */
419static void
420cms_srv_frame_stats(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
421 const u_char *payload, size_t payload_len)
422{
423 u_char out[4];
424 uint32_t need;
425
426 if (!ctx->logged_in) {
427 return; /* pre-auth: ignore */
428 }
429
430 /* An allocation advertisement, not a promise of exact length: a flat
431 * upper bound covering the stock statistics document shape (<statistics>
432 * shell + per-node <subscriber> rows) at the registry's configured slot
433 * ceiling, so a future text form always fits the advertised buffer. */
434 need = 4096;
435
436 out[0] = (u_char) (need >> 24);
437 out[1] = (u_char) (need >> 16);
438 out[2] = (u_char) (need >> 8);
439 out[3] = (u_char) need;
440 (void) brix_cms_srv_send_data(ctx, streamid, out, sizeof(out));
441}

469/*
470 * cms_srv_frame_have — Phase-89 W3: a node asserts it holds a path (kYR_have,
471 * raw NUL-terminated path), answering one of our kYR_state fan-out probes.
472 *
473 * SECURITY GATE: a node may only assert `have` for paths under its own
474 * login-Paths prefixes. Anything else (including an unsolicited HAVE from a
475 * hostile/buggy node) is logged and dropped — never cached, never woken on.
476 * A currently-blacklisted/draining node is likewise refused, so the dynamic
477 * location plane cannot bypass the operator blacklist (W6′) or a drain.
478 *
479 * On acceptance: record node → path in the SHM loc cache, then wake the
480 * parked client whose pending-locate entry the echoed streamid keys (first
481 * HAVE wins — later answers find no pending entry and are cache-only).
482 */
483static void
484cms_srv_frame_have(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
485 const u_char *payload, size_t payload_len)
486{
487 char path[BRIX_LOC_CACHE_PATH_MAX];
488 size_t plen;
489
490 if (!ctx->logged_in || payload_len == 0 || ctx->c == NULL) {
491 return;
492 }
493
494 /* Bound + shape the path: NUL-terminated, absolute, no ".." traversal.
495 * The node side stats before answering, but this side must not trust the
496 * peer's parser — same reject-not-guess posture as the state ingest. */
497 plen = ngx_strnlen((u_char *) payload, payload_len);
498 if (plen == 0 || plen >= sizeof(path) || payload[0] != '/') {
499 return;
500 }
501 ngx_memcpy(path, payload, plen);
502 path[plen] = '\0';
503 if (strstr(path, "..") != NULL) {
504 return;
505 }
506
507 if (!brix_srv_paths_cover(ctx->paths, path)) {
508 ngx_log_error(NGX_LOG_WARN, ctx->c->log, 0,
509 "brix: CMS server: %s:%d asserted have for \"%s\" "
510 "outside its exported paths [%s] — dropped",
511 ctx->host, (int) ctx->port, path, ctx->paths);
512 return;
513 }
514
515 if (brix_srv_is_blacklisted(ctx->host, ctx->port)) {
516 ngx_log_error(NGX_LOG_INFO, ctx->c->log, 0,
517 "brix: CMS server: ignoring have for \"%s\" from "
518 "drained node %s:%d", path, ctx->host, (int) ctx->port);
519 return;
520 }
521
522 brix_loc_cache_insert(path, ctx->host, ctx->port);
523 (void) brix_cms_wake_pending_session(ctx->c->log, streamid,
524 ctx->host, ctx->port);
525}

527/*
528 * cms_srv_frame_error — Phase-89 W8: a node answered a forwarded namespace op
529 * with kYR_error ([4B big-endian ecode][text + NUL], echoing the fan-out
530 * streamid). Fold it into the fan-out aggregation slot; success stays silent
531 * (the deadline window in fanout.c is the success signal). Payload text is
532 * peer-controlled — it is bounded here and sanitized before it can reach the
533 * client reply.
534 */
535static void
536cms_srv_frame_error(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
537 const u_char *payload, size_t payload_len)
538{
539 uint32_t ecode;
540 char text[128];
541 size_t tlen;
542
543 if (!ctx->logged_in || ctx->c == NULL) {
544 return;
545 }
546
547 ecode = 0;
548 text[0] = '\0';
549 if (payload_len >= 4) {
550 ecode = ((uint32_t) payload[0] << 24) | ((uint32_t) payload[1] << 16)
551 | ((uint32_t) payload[2] << 8) | (uint32_t) payload[3];
552 tlen = ngx_strnlen((u_char *) payload + 4, payload_len - 4);
553 if (tlen >= sizeof(text)) {
554 tlen = sizeof(text) - 1;
555 }
556 ngx_memcpy(text, payload + 4, tlen);
557 text[tlen] = '\0';
558 }
559

563/*
564 * cms_srv_frame_cns — §6 Composite Cluster Name Space: a data server reports a
565 * namespace mutation. Apply it into the manager inventory (collect mode only).
566 */
567static void
568cms_srv_frame_cns(brix_cms_srv_ctx_t *ctx, uint32_t streamid,
569 const u_char *payload, size_t payload_len)
570{
571 uint8_t op;
572 uint64_t size, mtime;
573 char path[BRIX_CNS_PATH_MAX + 1];
574 uint32_t server_id;
575 const u_char *h = (const u_char *) ctx->host;
576 size_t i;
577
578 if (!ctx->logged_in || !brix_cns_collecting()) {
579 return;
580 }
581
582 if (brix_cns_event_decode(payload, payload_len, &op, &size, &mtime,
583 path, sizeof(path)) != NGX_OK)
584 {
585 ngx_log_error(NGX_LOG_WARN, ctx->c->log, 0,
586 "brix: CMS server: malformed CNS event from %s",
587 ctx->host);
588 return;
589 }
590 server_id = (uint32_t) ctx->port; /* host hash + port = origin id */
591 for (i = 0; h[i] != '\0'; i++) {
592 server_id = server_id * 33 + h[i];
593 }
594 (void) brix_cns_apply(op, path, size, mtime, server_id);
595 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
596 "brix: CNS event op=%ud path=%s from %s",
597 (unsigned) op, path, ctx->host);
598}

600/*
601 * cms_srv_frame_unknown — default: consult the manager routing table so the
602 * log distinguishes a valid manager opcode we have not wired yet (e.g.
603 * usage/stats/statfs) from a truly unroutable code. Either way we drop it —
604 * matching cmsd's tolerance of frames it does not act on.
605 */
606static void
607cms_srv_frame_unknown(brix_cms_srv_ctx_t *ctx, u_char code)
608{
609 const brix_cms_route_t *r =
610 brix_cms_route_lookup(XRDCMS_ROLE_MANAGER, code);
611 if (r != NULL) {
612 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
613 "brix: CMS server: unhandled opcode '%s' from %s",
614 r->name, ctx->host);
615 } else {
616 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, ctx->c->log, 0,
617 "brix: CMS server: unknown rrCode=%ui from %s",
618 (ngx_uint_t) code, ctx->host);
619 }
620}

622/* Opcode → handler route table. AVAIL and SPACE share a payload shape and a
623 * handler; the NULL sentinel terminates the scan (unmatched codes fall through
624 * to cms_srv_frame_unknown). */
625static const struct {
626 u_char code;
627 cms_srv_frame_pt handler;
628} cms_srv_frame_routes[] = {
629 { CMS_RR_LOGIN, cms_srv_frame_login },

631 { CMS_RR_LOAD, cms_srv_frame_load },
632 { CMS_RR_AVAIL, cms_srv_frame_avail },
633 { CMS_RR_SPACE, cms_srv_frame_avail },

635 { CMS_RR_PING, cms_srv_frame_ping },
636 { CMS_RR_DISC, cms_srv_frame_disc },
637 { CMS_RR_UPDATE, cms_srv_frame_update },
638 { CMS_RR_STATFS, cms_srv_frame_statfs },
639 { CMS_RR_USAGE, cms_srv_frame_usage },
640 { CMS_RR_STATS, cms_srv_frame_stats },
641 { CMS_RR_STATUS, cms_srv_frame_status },
642 { CMS_RR_GONE, cms_srv_frame_gone },
643 { CMS_RR_HAVE, cms_srv_frame_have },
644 { CMS_RSP_ERROR, cms_srv_frame_error }, /* W8 fan-out reply fold */
645 { CMS_RR_CNS, cms_srv_frame_cns },
646 { 0, NULL },
647};
648
649/*
650 * cms_srv_process_frame — route one complete frame by opcode through the
651 * table above. A handler that tears the connection down leaves ctx->c NULL;
652 * the read loop checks that after every dispatch.
653 */
654void
655cms_srv_process_frame(brix_cms_srv_ctx_t *ctx, u_char code, uint32_t streamid,
656 const u_char *payload, size_t payload_len)
657{
658 ngx_uint_t i;
659
660 for (i = 0; cms_srv_frame_routes[i].handler != NULL; i++) {
661 if (cms_srv_frame_routes[i].code == code) {
662 cms_srv_frame_routes[i].handler(ctx, streamid,
663 payload, payload_len);
664 return;
665 }
666 }
667
668 cms_srv_frame_unknown(ctx, code);
669}
