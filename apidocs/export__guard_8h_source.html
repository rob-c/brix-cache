1/*
2 * export_guard.h — config-time assertion that service trees holding internal
3 * metadata sidecars never live inside a client-visible export namespace.
4 *
5 * WHAT: brix_assert_dir_outside_export(cf, label, export_canon, dir) fails the
6 * config (NGX_ERROR + emerg log) when `dir` is at or beneath `export_canon`.
7 * A cache/state/stage tree nested under an export would place .cinfo/.meta
8 * sidecars and upload temps (.xrd-tmp./.xrdresume./.commit) directly in the
9 * namespace clients browse.
10 *
11 * WHY: the runtime name filter (brix_is_internal_name) hides those artifacts at
12 * every client checkpoint, but the STRUCTURAL guarantee is that the service
13 * trees are simply not in the export at all. This is a hard check — a
14 * misconfiguration is a deploy-blocking error, not a warning — so an operator
15 * cannot ship a topology that depends on the runtime filter as its only line
16 * of defence. Both paths are canonicalized (realpath, when the dir exists)
17 * before the at-or-beneath test so a symlinked or "."/".."-laden dir cannot
18 * sneak inside.
19 *
20 * HOW: realpath the tree (fall back to the given string when it is not yet
21 * created — operators configure absolute paths, and a lexical prefix match
22 * still catches the dangerous /export/... nesting), then reuse
23 * brix_beneath_strip_root() (non-NULL == inside the root). No-op on empty input.
24 */

34static inline ngx_int_t
35brix_assert_dir_outside_export(ngx_conf_t *cf, const char *label,
36 const char *export_canon, const char *dir)
37{
38 char canon[PATH_MAX];
39 const char *cmp;
40
41 if (export_canon == NULL || export_canon[0] == '\0'
42 || dir == NULL || dir[0] == '\0')
43 {
44 return NGX_OK;
45 }
46
47 /* A pure cache node advertises export "/" (the whole namespace maps to a
48 * remote origin); every local path is trivially "beneath /", so the guard
49 * would reject an otherwise-valid topology. Such a node has no local export
50 * tree to leak into and relies on the runtime name filter — skip it. */
51 if (export_canon[0] == '/' && export_canon[1] == '\0') {
52 return NGX_OK;
53 }
54
55 cmp = (realpath(dir, canon) != NULL) ? canon : dir;
56
57 if (brix_beneath_strip_root(export_canon, cmp) != NULL) {
58 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
59 "brix: %s \"%s\" is at or beneath export root \"%s\" — internal "
60 "metadata sidecars (.cinfo/.meta, upload temps) would be exposed in "
61 "the client namespace; configure it outside every brix_export root",
62 label, cmp, export_canon);
63 return NGX_ERROR;
64 }
65 return NGX_OK;
66}
67
68#endif /* BRIX_CORE_CONFIG_EXPORT_GUARD_H */
