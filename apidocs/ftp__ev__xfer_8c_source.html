8/*
9 * ftp_ev_xfer.c — data-transfer verbs and the non-blocking RETR/STOR/LIST pump.
10 *
11 * WHAT: the dispatch entry for every data verb (PASV/EPSV/PORT/EPRT delegate to
12 * ftp_ev_data.c; RETR/STOR/APPE/LIST/NLST/MLSD run here), the transfer set-up
13 * (validate, resolve the write offset, send 150, open the data channel), and the
14 * three event-driven pumps that move bytes between the data socket and the VFS.
15 *
16 * WHY: the sync engine blocks the worker inside one transfer's read/write loop.
17 * Here each transfer is a small state machine hung off the data connection's
18 * read/write events, so the worker interleaves it with other sessions. The VFS
19 * side is opened once the socket is up (matching the sync order: 150 → connect →
20 * open → move bytes → 226) and the socket side never blocks — a full socket
21 * yields NGX_AGAIN and re-arms the event. Local VFS reads/writes are performed
22 * inline (fast POSIX/pblock syscalls); a thread-pool offload for slow object
23 * backends is future work (P82.2 note in the phase-82 doc).
24 *
25 * HOW: brix_ftp_ev_do_transfer() validates and queues the 150, then arms the data
26 * channel and yields; the control loop parks in FTP_EV_ST_XFER. When the socket
27 * becomes ready ftp_ev_data.c calls brix_ftp_ev_data_ready(), which opens the VFS
28 * handle and kicks the matching pump. Each pump drains its buffer to the sink,
29 * refills from the source, and on completion/failure calls
30 * brix_ftp_ev_data_finish(), which emits the 226/550 and resumes the control loop.
31 */

45static void
46ev_retr_write(ngx_event_t *wev)
47{
48 ngx_connection_t *c = wev->data;
49 ftp_ev_dc_t *dc = c->data;
50
51 if (wev->timedout) {
52 brix_ftp_ev_data_finish(dc, NGX_ERROR);
53 return;
54 }
55 if (wev->timer_set) {
56 ngx_del_timer(wev);
57 }
58
59 for ( ;; ) {
60 if (dc->buf_pos < dc->buf_len) {
61 ssize_t n = c->send(c, dc->buf + dc->buf_pos,
62 dc->buf_len - dc->buf_pos);
63 if (n > 0) {
64 dc->buf_pos += (size_t) n;
65 continue;
66 }
67 if (n == NGX_AGAIN) {
68 if (ngx_handle_write_event(wev, 0) != NGX_OK) {
69 brix_ftp_ev_data_finish(dc, NGX_ERROR);
70 return;
71 }
72 ngx_add_timer(wev, BRIX_FTP_EV_IO_TIMEO);
73 return;
74 }
75 brix_ftp_ev_data_finish(dc, NGX_ERROR); /* NGX_ERROR or peer EOF */
76 return;
77 }
78
79 /* Buffer drained — refill from the VFS handle (an object backend
80 * reassembles its block files through the driver; a raw pread would only
81 * see block 0). */
82 if (dc->off >= dc->size) {
83 brix_ftp_ev_data_finish(dc, NGX_OK); /* whole file sent */
84 return;
85 }
86 {
87 size_t want = (dc->size - dc->off > BRIX_FTP_EV_XFER_BUF)
88 ? BRIX_FTP_EV_XFER_BUF
89 : (size_t) (dc->size - dc->off);
90 ssize_t got = brix_vfs_file_pread(dc->fh, dc->buf, want, dc->off);
91 if (got <= 0) {
92 brix_ftp_ev_data_finish(dc, NGX_ERROR);
93 return;
94 }
95 dc->off += got;
96 dc->buf_len = (size_t) got;
97 dc->buf_pos = 0;
98 }
99 }

103/* ---- STOR/APPE: drain the data socket into the file ------------------------ */
104
105static void
106ev_stor_read(ngx_event_t *rev)
107{
108 ngx_connection_t *c = rev->data;
109 ftp_ev_dc_t *dc = c->data;
110
111 if (rev->timedout) {
112 brix_ftp_ev_data_finish(dc, NGX_ERROR);
113 return;
114 }
115 if (rev->timer_set) {
116 ngx_del_timer(rev);
117 }
118
119 for ( ;; ) {
120 ssize_t n = c->recv(c, dc->buf, BRIX_FTP_EV_XFER_BUF);
121
122 if (n > 0) {
123 if (brix_vfs_writer_write(dc->writer, dc->buf, (size_t) n, dc->off)
124 != NGX_OK)
125 {
126 brix_ftp_ev_data_finish(dc, NGX_ERROR);
127 return;
128 }
129 dc->off += n;
130 continue;
131 }
132 if (n == NGX_AGAIN) {
133 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
134 brix_ftp_ev_data_finish(dc, NGX_ERROR);
135 return;
136 }
137 ngx_add_timer(rev, BRIX_FTP_EV_IO_TIMEO);
138 return;
139 }
140 if (n == 0) {
141 /* Client closed the data channel → EOF. In stream mode a bare close
142 * is the *only* completion signal, so it is indistinguishable from a
143 * mid-flight truncation (a hostile middlebox dropping the connection).
144 * When the client declared the size via ALLO and the operator opted
145 * into brix_gridftp_require_allo_size, hold the transfer to exactly
146 * that many bytes: a short (or over-long) delivery fails 550 rather
147 * than committing a truncated object as complete. The clean prefix
148 * is left in place (unlike the MODE E declared-complete-but-holed
149 * case) so a REST-resume can continue from dc->off. */
150 ngx_int_t rc;
151
152 if (dc->fc->conf->require_allo_size && dc->allo_size >= 0
153 && dc->off != dc->allo_size)
154 {
155 brix_ftp_ev_data_finish(dc, NGX_ERROR);
156 return;
157 }
158 /* Commit runs the read-back verify (and unlinks a mismatch); a NULL
159 * writer skips abort in finish. */
160 rc = brix_vfs_writer_commit(dc->writer);
161 dc->writer = NULL;
162 brix_ftp_ev_data_finish(dc, rc);
163 return;
164 }
165 brix_ftp_ev_data_finish(dc, NGX_ERROR); /* NGX_ERROR */
166 return;
167 }
168}

171/* ---- LIST/NLST/MLSD: stream a directory listing to the data socket --------- */
172
173/* Refill the transfer buffer with as many formatted entries as fit whole; sets
174 * src_eof once the directory is exhausted. Each line is bounded by PATH_MAX+128,
175 * so we stop before the last line that could overflow — no entry is ever split. */
176static void
177ev_list_fill(ftp_ev_dc_t *dc)
178{
179 ngx_str_t name;
180 brix_vfs_stat_t st;
181
182 dc->buf_pos = 0;
183 dc->buf_len = 0;
184
185 while (!dc->src_eof
186 && dc->buf_len + PATH_MAX + 128 <= BRIX_FTP_EV_XFER_BUF)
187 {
188 u_char *p = dc->buf + dc->buf_len;
189 u_char *end = dc->buf + BRIX_FTP_EV_XFER_BUF;
190
191 if (brix_vfs_readdir(dc->dh, &name, &st) != NGX_OK) {
192 dc->src_eof = 1;
193 break;
194 }
195 if (name.len == 1 && name.data[0] == '.') { continue; }
196 if (name.len == 2 && name.data[0] == '.' && name.data[1] == '.') {
197 continue;
198 }
199
200 if (dc->ls_mode == FTP_EV_LS_LONG) {
201 p = ngx_snprintf(p, end - p,
202 "%crw%cr--r-- 1 brix brix %O Jan 1 00:00 %V\r\n",
203 st.is_directory ? 'd' : '-',
204 st.is_directory ? 'x' : '-',
205 st.size, &name);
206 } else if (dc->ls_mode == FTP_EV_LS_MLSD) {
207 p = ngx_snprintf(p, end - p,
208 "type=%s;size=%O;perm=%s; %V\r\n",
209 st.is_directory ? "dir" : "file",
210 st.size,
211 st.is_directory ? "el" : "r",
212 &name);
213 } else {
214 p = ngx_snprintf(p, end - p, "%V\r\n", &name);
215 }
216 dc->buf_len = (size_t) (p - dc->buf);
217 }
218}

266/* The data connection is up: open the VFS side and start the matching pump. */
267void
268brix_ftp_ev_data_ready(ftp_ev_dc_t *dc)
269{
270 ftp_ev_t *fc = dc->fc;
271 brix_vfs_ctx_t vctx;
272 int verr = 0;
273
274 brix_ftp_ev_vfs_ctx(fc, dc->abs, &vctx);
275
276 switch (dc->op) {
277 case FTP_EV_OP_RETR:
278 dc->fh = brix_vfs_open(&vctx, BRIX_VFS_O_READ, &verr);
279 if (dc->fh == NULL) {
280 brix_ftp_ev_data_finish(dc, NGX_ERROR);
281 return;
282 }
283 dc->size = brix_vfs_file_size(dc->fh);
284 if (!(dc->off > 0 && dc->off <= dc->size)) { /* clamp a stale REST */
285 dc->off = 0;
286 }
287 if (dc->mode_e) {
288 brix_ftp_ev_retr_mode_e_start(dc); /* extended-block framing */
289 return;
290 }
291 dc->dconn->read->handler = ev_data_idle;
292 dc->dconn->write->handler = ev_retr_write;
293 ev_retr_write(dc->dconn->write);
294 return;
295
296 case FTP_EV_OP_STOR:
297 case FTP_EV_OP_APPE:
298 dc->writer = brix_vfs_writer_open(&vctx, dc->flags, dc->verify, &verr);
299 if (dc->writer == NULL) {
300 brix_ftp_ev_data_finish(dc, NGX_ERROR);
301 return;
302 }
303 dc->dconn->write->handler = ev_data_idle;
304 dc->dconn->read->handler = ev_stor_read;
305 ev_stor_read(dc->dconn->read);
306 return;
307
308 default: /* FTP_EV_OP_LIST / NLST / MLSD */
309 dc->dh = brix_vfs_opendir(&vctx, &verr);
310 if (dc->dh == NULL) {
311 brix_ftp_ev_data_finish(dc, NGX_ERROR);
312 return;
313 }
314 dc->dconn->read->handler = ev_data_idle;
315 dc->dconn->write->handler = ev_list_write;
316 ev_list_write(dc->dconn->write);
317 return;
318 }
319}

322/* Validate a transfer verb, resolve the write offset, send the 150, and arm the
323 * data channel. Returns NGX_OK (transfer running or a queued 5xx), NGX_DONE is
324 * never used, NGX_ERROR only on a fatal reply-buffer failure. */
325static ngx_int_t
326ev_begin_transfer(ftp_ev_t *fc, int op, const char *arg)
327{
328 char abs[PATH_MAX];
329 int code;
330 int writing = (op == FTP_EV_OP_STOR || op == FTP_EV_OP_APPE);
331 off_t start = 0;
332 unsigned flags = 0;
333 int verify = 0;
334 off_t allo = fc->allo_size; /* one-shot, per this command */
335 ftp_ev_dc_t *dc;
336
337 fc->allo_size = -1; /* consume ALLO unconditionally */
338
339 if (writing && !fc->conf->allow_write) {
340 return brix_ftp_ev_reply(fc,
341 "550 Permission denied (read-only export)\r\n");
342 }
343 if (!fc->active && fc->pasv_fd < 0) {
344 return brix_ftp_ev_reply(fc, "425 Use PASV or PORT first\r\n");
345 }
346
347 /* MODE E STOR reassembles up to `Parallelism` inbound data streams, so it
348 * requires a passive listener the peer opens all its streams to; an active
349 * single connect cannot carry the fan-out. */
350 if (fc->mode_e && writing && fc->active) {
351 fc->rest_off = 0;
352 return brix_ftp_ev_reply(fc,
353 "504 MODE E upload requires passive mode\r\n");
354 }
355
356 code = brix_ftp_ev_resolve(fc, arg, abs, sizeof(abs));
357 if (code != 0) {
358 fc->rest_off = 0;
359 return brix_ftp_ev_reply(fc, "%d Failed to resolve path\r\n", code);
360 }
361
362 /* Resolve the write start / validate the source before the 150. */
363 if (op == FTP_EV_OP_STOR) {
364 start = fc->rest_off;
365 flags = (fc->rest_off == 0) ? BRIX_VFS_O_TRUNC : 0;
366 verify = (fc->conf->verify_write && start == 0) ? 1 : 0;
367 } else if (op == FTP_EV_OP_APPE) {
368 brix_vfs_ctx_t vctx;
369 brix_vfs_stat_t st;
370 brix_ftp_ev_vfs_ctx(fc, abs, &vctx);
371 if (brix_vfs_stat(&vctx, &st) == NGX_OK && !st.is_directory) {
372 start = st.size;
373 }
374 } else if (op == FTP_EV_OP_RETR) {
375 brix_vfs_ctx_t vctx;
376 brix_vfs_stat_t st;
377 brix_ftp_ev_vfs_ctx(fc, abs, &vctx);
378 if (brix_vfs_stat(&vctx, &st) != NGX_OK || st.is_directory) {
379 fc->rest_off = 0;
380 return brix_ftp_ev_reply(fc, "550 No such file\r\n");
381 }
382 start = fc->rest_off;
383 }
384 fc->rest_off = 0; /* REST is one-shot */
385
386 dc = ngx_pcalloc(fc->c->pool, sizeof(ftp_ev_dc_t));
387 if (dc == NULL) {
388 return brix_ftp_ev_reply(fc, "425 Cannot open data connection\r\n");
389 }
390 /* +FTP_EB_HDR: MODE E RETR frames each chunk as [17-byte header][payload] in
391 * one buffer, so it must hold a full payload chunk plus the header. */
392 dc->buf = ngx_pnalloc(fc->c->pool, BRIX_FTP_EV_XFER_BUF + FTP_EB_HDR);
393 if (dc->buf == NULL) {
394 return brix_ftp_ev_reply(fc, "425 Cannot open data connection\r\n");
395 }
396 dc->fc = fc;
397 dc->op = op;
398 dc->writing = writing;
399 dc->off = start;
400 dc->flags = flags;
401 dc->verify = verify;
402 /* ALLO completeness enforcement is stream-mode STOR only: MODE E validates
403 * completeness structurally (gapless tiling), and RETR/APPE have no ALLO. */
404 dc->allo_size = (op == FTP_EV_OP_STOR && !fc->mode_e) ? allo : -1;
405 dc->mode_e = fc->mode_e;
406 dc->eb_eof_total = -1; /* set by the EOF block (MODE E STOR) */
407 dc->ls_mode = (op == FTP_EV_OP_LIST) ? FTP_EV_LS_LONG
408 : (op == FTP_EV_OP_MLSD) ? FTP_EV_LS_MLSD
409 : FTP_EV_LS_NLST;
410 memcpy(dc->abs, abs, ngx_strlen(abs) + 1);
411
412 fc->dc = dc;
413 fc->state = FTP_EV_ST_XFER;
414
415 if (brix_ftp_ev_reply(fc, "150 Opening %s mode data connection\r\n",
416 fc->type_binary ? "BINARY" : "ASCII") != NGX_OK)
417 {
418 fc->dc = NULL;
419 fc->state = FTP_EV_ST_CMD;
420 return NGX_ERROR;
421 }
422
423 if (brix_ftp_ev_data_open(dc) != NGX_OK) {
424 /* Synchronous set-up failure (before any async event): the 150 is already
425 * queued — append the 550 and let the control loop flush both. Don't
426 * call data_finish here; it would re-enter the control loop we're inside. */
427 fc->dc = NULL;
428 fc->state = FTP_EV_ST_CMD;
429 return brix_ftp_ev_reply(fc, "425 Cannot open data connection\r\n");
430 }
431 return NGX_OK;
432}
