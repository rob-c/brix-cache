3/*
4 * access_log.c — structured JSON access-log line emitter for VFS operations.
5 *
6 * WHAT: Emits one machine-parsable JSON access record per completed VFS I/O
7 * operation via brix_access_log_emit(). Each line carries timestamp,
8 * protocol, op name, request path, byte count, offset, latency, error
9 * status, cache-hit flag, auth method, and the authenticated subject/DN.
10 * WHY: Prometheus counters are aggregate and low-cardinality (no paths or DNs
11 * as labels — INVARIANT #8), but operators still need per-request audit
12 * detail. A JSON log line carries the high-cardinality fields (path,
13 * subject) that must never become metric labels, while reusing the same
14 * unified vocabulary (brix_metric_proto_name / _op_name / _err_name /
15 * _auth_method_name) so logs and metrics line up.
16 * HOW: The line is written with ngx_log_error at NGX_LOG_INFO and prefixed
17 * "brix_access_json: " so a log pipeline can grep and strip the prefix.
18 * Free-text fields are run through brix_access_json_escape() into fixed
19 * stack buffers before being interpolated, so untrusted wire bytes can
20 * never break out of the JSON string. Subject is taken from the identity
21 * (subject, falling back to DN); path/offset/cache come from the
22 * brix_vfs_io_result_t when present.
23 */
24
25/*
26 * brix_access_json_escape — copy src into dst as a JSON-string-safe sequence.
27 *
28 * Escapes '"' and '\\' with a backslash and renders any byte outside printable
29 * ASCII (< 0x20 or >= 0x7f) as a \u00NN escape, so control bytes and non-ASCII
30 * from the wire cannot terminate the surrounding JSON string. Always writes a
31 * NUL terminator (unless dstsz == 0) and never exceeds dstsz, truncating cleanly
32 * if an escape sequence would not fit.
33 */
34static void
35brix_access_json_escape(const char *src, char *dst, size_t dstsz)
36{
37 static const char hex[] = "0123456789abcdef";
38 size_t used;
39
40 if (dstsz == 0) {
41 return;
42 }
43
44 used = 0;
45 if (src == NULL) {
46 src = "";
47 }
48
49 while (*src != '\0' && used + 1 < dstsz) {
50 unsigned char ch = (unsigned char) *src++;
51
52 if (ch == '"' || ch == '\\') {
53 if (used + 2 >= dstsz) {
54 break;
55 }
56 dst[used++] = '\\';
57 dst[used++] = (char) ch;
58 continue;
59 }
60
61 if (ch >= 0x20 && ch < 0x7f) {
62 dst[used++] = (char) ch;
63 continue;
64 }
65
66 if (used + 6 >= dstsz) {
67 break;
68 }
69 dst[used++] = '\\';
70 dst[used++] = 'u';
71 dst[used++] = '0';
72 dst[used++] = '0';
73 dst[used++] = hex[ch >> 4];
74 dst[used++] = hex[ch & 0x0f];
75 }
76
77 dst[used] = '\0';
78}
