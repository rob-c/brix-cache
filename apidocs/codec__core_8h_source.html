1/*
2 * codec_core.h — pure (ngx-free) streaming compression/decompression abstraction.
3 *
4 * WHAT: One uniform streaming API over every compression codec the project
5 * supports (gzip/deflate via zlib, zstd, xz/lzma, brotli, bzip2). A caller
6 * opens a stream for one codec + direction, pumps input through step() and
7 * drains output, then closes — never touching z_stream / ZSTD_CCtx /
8 * lzma_stream / Brotli* / bz_stream directly. A descriptor table makes
9 * codec selection (by id / canonical name / HTTP Content-Encoding token)
10 * branch-free.
11 * WHY: All four compression surfaces (inbound PUT decode, outbound GET encode,
12 * ZIP member inflate, root:// inline) and BOTH consumers — the nginx module
13 * (src/) and the native client (client/) — must share one implementation.
14 * Like checksum_core.c / crc32c.c / gsi_core.c, this is an ngx-free kernel
15 * carried by libxrdproto (build-in-place), linked by module + libxrdc.
16 * HOW: Each codec is a backend (init/step/end vtable) in its own codec_<name>.c,
17 * compiled only when its library is present (-DBRIX_HAVE_<NAME>); an
18 * absent backend leaves a descriptor with available=0 so the table never
19 * has holes. A shared bomb guard (output cap + expansion-ratio ceiling) is
20 * enforced centrally in brix_codec_step() so every untrusted decode path
21 * is protected uniformly.
22 *
23 * No nginx, no logging, no errno semantics — the caller maps result codes to its
24 * own HTTP/wire errors and logs at the edge.
25 */

62/*
63 * Decompression-bomb guard. Bounds total produced bytes and the output:input
64 * expansion ratio. Pass NULL to brix_codec_open() for compress streams or any
65 * trusted decode; ALWAYS pass a populated guard for untrusted decode (PUT, ZIP).
66 * total_in/total_out are running counters maintained by brix_codec_step().
67 */
68typedef struct {
69 uint64_t out_cap; /* hard ceiling on TOTAL produced bytes; 0 = unbounded */
70 uint32_t max_ratio; /* reject once total_out/total_in exceeds this; 0 = off */
71 uint64_t total_in; /* init 0 */
72 uint64_t total_out; /* init 0 */
73} brix_codec_guard_t;
74
75/* Ratio check only engages once this many OUTPUT bytes have been produced, so a
76 * tiny stream doesn't trip a false positive on its first expansion. It MUST gate
77 * on output, not input: a compression bomb has tiny compressed input, so gating
78 * on input would disable the ratio guard for exactly the most dangerous streams. */
79#define BRIX_CODEC_RATIO_FLOOR (64u * 1024u)

81/*
82 * root:// inline read compression (phase-42 W4) wire signalling.
83 *
84 * When a read handle is opened with the opaque "?xrootd.compress=<codec>" AND
85 * the server has brix_read_compress on, the server sets the (otherwise
86 * vestigial) kXR_open reply fields ServerResponseBody_Open.cpsize =
87 * BRIX_INLINE_CMP_MAGIC and cptype[0] = <codec id ordinal>. Each subsequent
88 * kXR_read response is then a single self-contained codec frame of the requested
89 * plaintext range (offset-addressable + resumable because every request is an
90 * independent whole-range frame). pgread/readv ALWAYS stay plaintext so the
91 * pgread kXR_status + per-page CRC32c invariant is preserved byte-for-byte.
92 *
93 * A stock client never sends the opaque, so cpsize stays 0 and the uncompressed
94 * path is byte-identical; a stock server ignores the opaque, so cpsize stays 0
95 * and our client falls back to plaintext. cptype[0] carries the codec ordinal
96 * (1..BRIX_CODEC_MAX-1) rather than a 4-char name so both ends agree without
97 * string parsing — only our own client interprets it.
98 */
99#define BRIX_INLINE_CMP_MAGIC 0x5A /* 'Z'; nonzero, fits kXR_int32 cpsize */
100
101/* ---- backend-author interface (implemented by codec_<name>.c) ---- */
102
103typedef struct {
104 /* Allocate + initialise native state for (id, dir, level). level is clamped
105 * to [level_min,level_max] by the core before this call (compress only;
106 * ignored for decompress). Returns 0 on success, <0 (brix_codec_rc_t) on
107 * failure (no state leaked). */
108 int (*init)(void **state, brix_codec_id_t id, brix_codec_dir_t dir,
109 int level);
110 /* Consume in[*in_pos..in_len), produce into out[*out_pos..out_size),
111 * advancing both cursors by the amount processed/produced. finish!=0 => the
112 * caller will supply no input after the current buffer; flush/finalise.
113 * Returns OK / END / ERR_*. Must make forward progress when given room. */
114 brix_codec_rc_t (*step)(void *state,
115 const uint8_t *in, size_t in_len, size_t *in_pos,
116 uint8_t *out, size_t out_size, size_t *out_pos,
117 int finish);
118 void (*end)(void *state); /* free native state; tolerate NULL */
119} brix_codec_backend_t;
