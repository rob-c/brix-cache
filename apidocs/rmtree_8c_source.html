1/* rmtree.c — post-order recursive remote delete (rm -r for root://).
2 * WHAT: delete every file, then every directory bottom-up, under `path`,
3 * then `path` itself.
4 * WHY: the wire has no bulk delete; xrdfs rm -r and xrdcp --delete both
5 * need one careful implementation (depth-capped, overflow-checked,
6 * refuses the export root).
7 * HOW: brix_dirlist(want_stat=1); recurse into dirs first, brix_rm files,
8 * brix_rmdir self last. BRIX_RMTREE_DRYRUN reports without deleting.
9 * Before descending a dirlist entry flagged kXR_isDir, call brix_lstat
10 * to detect directory symlinks: if the entry is a symlink (kXR_other),
11 * brix_rm the LINK instead of descending into the target's subtree. */

20/*
21 * rmtree_join — join dir + name into out[outsz] with exactly one separator.
22 *
23 * WHAT: produce "dir/name" (or "dir" + name when dir already ends with '/').
24 * WHY: the same path-join pattern appears in walk.c; centralise it as a
25 * static helper to avoid silent snprintf truncation.
26 * HOW: snprintf returns ≥ outsz on truncation; treat that as an error (-1).
27 */
28static int
29rmtree_join(const char *dir, const char *name, char *out, size_t outsz)
30{
31 size_t dl = strlen(dir);
32 const char *sep = (dl > 0 && dir[dl - 1] == '/') ? "" : "/";
33 return ((size_t) snprintf(out, outsz, "%s%s%s", dir, sep, name) >= outsz)
34 ? -1 : 0;
35}

37/*
38 * rmtree_depth — post-order recursive delete of `path` subtree.
39 *
40 * WHAT: dirlist `path`; for each entry: recurse into real dirs, probe symlinks
41 * with lstat and rm the link, rm plain files. Then rmdir `path` itself.
42 * WHY: dirlist stat does not distinguish a real directory from a dir-symlink
43 * (both carry kXR_isDir); descending a symlink would delete the target's
44 * contents, not the link. brix_lstat (kXR_statNoFollow) resolves this:
45 * a symlink returns kXR_other, not kXR_isDir.
46 * HOW: for entries with kXR_isDir: brix_lstat first; if kXR_other → brix_rm
47 * the link; if still kXR_isDir → descend; if lstat fails → abort (-1).
48 * On report-callback abort: set st with XRDC_EUSAGE and return -1. */
49static int
50rmtree_depth(brix_conn *c, const char *path, int depth, unsigned flags,
51 brix_rmtree_report report, void *u, brix_status *st)
52{
53 brix_dirent *ents = NULL;
54 size_t n = 0, i;
55 int rc = 0;
56
57 if (depth >= BRIX_RMTREE_MAXDEPTH) {
58 brix_status_set(st, XRDC_EUSAGE, 0, "rmtree: depth cap at %s", path);
59 return -1;
60 }
61 if (brix_dirlist(c, path, 1, &ents, &n, st) != 0) {
62 return -1;
63 }
64 for (i = 0; i < n && rc == 0; i++) {
65 char full[XRDC_PATH_MAX];
66 if (ents[i].name[0] == '.'
67 && (ents[i].name[1] == '\0'
68 || (ents[i].name[1] == '.' && ents[i].name[2] == '\0'))) {
69 continue;
70 }
71 if (rmtree_join(path, ents[i].name, full, sizeof(full)) != 0) {
72 brix_status_set(st, XRDC_EUSAGE, 0, "rmtree: path too long under %s", path);
73 rc = -1;
74 break;
75 }
76 if (ents[i].have_stat && (ents[i].st.flags & kXR_isDir)) {
77 /* The dirlist kXR_isDir bit is set for both real directories and
78 * directory symlinks. Probe with lstat (kXR_statNoFollow) to
79 * distinguish them before deciding whether to descend. */
80 brix_statinfo lsi;
81 if (brix_lstat(c, full, &lsi, st) != 0) {
82 /* Conservative: if the probe fails we cannot safely determine
83 * the entry type — abort rather than guess. */
84 rc = -1;
85 break;
86 }
87 if (lsi.flags & kXR_other) {
88 /* kXR_other means "not a regular file or dir" — i.e. a symlink.
89 * Remove the LINK itself; do not descend into the target. */
90 if (report != NULL && report(full, 0, u) != 0) {
91 brix_status_set(st, XRDC_EUSAGE, 0,
92 "rmtree: aborted by report callback at %s", full);
93 rc = -1;
94 break;
95 }
96 if (!(flags & BRIX_RMTREE_DRYRUN) && brix_rm(c, full, st) != 0) {
97 rc = -1;
98 }
99 } else {
100 /* Real directory: recurse post-order. */
101 rc = rmtree_depth(c, full, depth + 1, flags, report, u, st);
102 }
103 } else {
104 if (report != NULL && report(full, 0, u) != 0) {
105 brix_status_set(st, XRDC_EUSAGE, 0,
106 "rmtree: aborted by report callback at %s", full);
107 rc = -1;
108 break;
109 }
110 if (!(flags & BRIX_RMTREE_DRYRUN) && brix_rm(c, full, st) != 0) {
111 rc = -1;
112 }
113 }
114 }
115 free(ents);
116 if (rc != 0) {
117 return rc;
118 }
119 if (report != NULL && report(path, 1, u) != 0) {
120 brix_status_set(st, XRDC_EUSAGE, 0,
121 "rmtree: aborted by report callback at %s", path);
122 return -1;
123 }
124 if (!(flags & BRIX_RMTREE_DRYRUN) && brix_rmdir(c, path, st) != 0) {
125 return -1;
126 }
127 return 0;
128}

130/*
131 * brix_rmtree — post-order recursive delete of a remote tree.
132 *
133 * WHAT: guard the export root and all-slash paths, then recurse.
134 * WHY: "/" and "//" both resolve to the export root on the server; rejecting
135 * any path made solely of '/' characters prevents accidental fleet-wide
136 * deletes from paths that collapse to root.
137 * HOW: scan for at least one non-'/' byte; NULL / empty are also refused. */
138int
139brix_rmtree(brix_conn *c, const char *path, unsigned flags,
140 brix_rmtree_report report, void *u, brix_status *st)
141{
142 const char *p;
143
144 if (path == NULL || path[0] == '\0') {
145 brix_status_set(st, XRDC_EUSAGE, 0,
146 "rmtree: refusing to delete the export root");
147 return -1;
148 }
149 /* Reject paths that consist entirely of '/' characters ("/" "//" etc). */
150 for (p = path; *p == '/'; p++) { }
151 if (*p == '\0') {
152 brix_status_set(st, XRDC_EUSAGE, 0,
153 "rmtree: refusing to delete the export root");
154 return -1;
155 }
156 return rmtree_depth(c, path, 0, flags, report, u, st);
157}
