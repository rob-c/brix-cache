1/* rmtree.c — post-order recursive remote delete (rm -r for root://).
2 * WHAT: delete every file, then every directory bottom-up, under `path`,
3 * then `path` itself.
4 * WHY: the wire has no bulk delete; xrdfs rm -r and xrdcp --delete both
5 * need one careful implementation (depth-capped, overflow-checked,
6 * refuses the export root).
7 * HOW: brix_dirlist(want_stat=1); recurse into dirs first, brix_rm files,
8 * brix_rmdir self last. BRIX_RMTREE_DRYRUN reports without deleting.
9 * Before descending a dirlist entry flagged kXR_isDir, call brix_lstat
10 * to detect directory symlinks: if the entry is a symlink (kXR_other),
11 * brix_rm the LINK instead of descending into the target's subtree. */

26/*
27 * rmtree_join — join dir + name into out[outsz] with exactly one separator.
28 *
29 * WHAT: produce "dir/name" (or "dir" + name when dir already ends with '/');
30 * return 0 on success, -1 if the result would be truncated.
31 * WHY: the same path-join pattern appears in walk.c; centralise it as a
32 * static helper to avoid silent snprintf truncation.
33 * HOW: snprintf returns >= outsz on truncation; treat that as an error (-1).
34 */
35static int
36rmtree_join(const char *dir, const char *name, char *out, size_t outsz)
37{
38 size_t dl = strlen(dir);
39 const char *sep = (dl > 0 && dir[dl - 1] == '/') ? "" : "/";
40 return ((size_t) snprintf(out, outsz, "%s%s%s", dir, sep, name) >= outsz)
41 ? -1 : 0;
42}

44/*
45 * rmtree_is_dot — test whether a dirlist entry name is "." or "..".
46 *
47 * WHAT: return 1 for the self ("." ) and parent ("..") entries, 0 otherwise.
48 * WHY: a post-order walk must skip the two directory-relative entries or it
49 * would recurse into itself / its parent; factoring the compound test
50 * out keeps the traversal loop readable.
51 * HOW: match a leading '.' followed by end-of-string, or a second '.'
52 * followed by end-of-string.
53 */
54static int
55rmtree_is_dot(const char *name)
56{
57 return name[0] == '.'
58 && (name[1] == '\0'
59 || (name[1] == '.' && name[2] == '\0'));
60}

62/*
63 * rmtree_remove_leaf — report then delete a single file or symlink.
64 *
65 * WHAT: invoke the report callback (is_dir=0) and, unless dry-run, brix_rm the
66 * path; return 0 on success, -1 on callback abort or removal failure.
67 * WHY: the identical report -> dry-run gate -> brix_rm sequence is needed for
68 * both plain files and for symlinks (whose LINK is removed rather than
69 * descended); sharing it keeps the two callers byte-for-byte identical.
70 * HOW: 1. if a report callback is set and it returns non-zero, record an
71 * XRDC_EUSAGE abort status and return -1.
72 * 2. unless BRIX_RMTREE_DRYRUN is set, brix_rm the path; propagate its
73 * failure as -1.
74 */
75static int
76rmtree_remove_leaf(brix_conn *c, const char *full, unsigned flags,
77 brix_rmtree_report report, void *u, brix_status *st)
78{
79 if (report != NULL && report(full, 0, u) != 0) {
80 brix_status_set(st, XRDC_EUSAGE, 0,
81 "rmtree: aborted by report callback at %s", full);
82 return -1;
83 }
84 if (!(flags & BRIX_RMTREE_DRYRUN) && brix_rm(c, full, st) != 0) {
85 return -1;
86 }
87 return 0;
88}

90/*
91 * rmtree_handle_dir_entry — resolve a kXR_isDir entry into descend-or-unlink.
92 *
93 * WHAT: lstat `full`; if it is a symlink (kXR_other) remove the link, else
94 * recurse into it as a real directory; return 0 on success, -1 on error.
95 * WHY: the dirlist kXR_isDir bit is set for both real directories and
96 * directory symlinks; descending a symlink would delete the target's
97 * contents, not the link. brix_lstat (kXR_statNoFollow) resolves this:
98 * a symlink returns kXR_other, not kXR_isDir.
99 * HOW: 1. brix_lstat the path; on failure return -1 (conservative — we cannot
100 * safely determine the entry type, so abort rather than guess).
101 * 2. if the result carries kXR_other it is a symlink — remove the LINK
102 * via rmtree_remove_leaf and do not descend.
103 * 3. otherwise it is a real directory — recurse post-order at depth+1.
104 */
105static int
106rmtree_handle_dir_entry(brix_conn *c, const char *full, int depth,
107 unsigned flags, brix_rmtree_report report, void *u,
108 brix_status *st)
109{
110 brix_statinfo lsi;
111
112 if (brix_lstat(c, full, &lsi, st) != 0) {
113 return -1;
114 }
115 if (lsi.flags & kXR_other) {
116 return rmtree_remove_leaf(c, full, flags, report, u, st);
117 }
118 return rmtree_depth(c, full, depth + 1, flags, report, u, st);
119}

121/*
122 * rmtree_one_entry — process a single dirlist entry of the current directory.
123 *
124 * WHAT: skip "."/"..", build the child path, then descend directories or
125 * remove files/symlinks; return 0 on success (including skipped dots),
126 * -1 on any error (path too long, lstat/removal failure, callback abort).
127 * WHY: isolates the per-entry decision (dir-vs-leaf, symlink probing) so the
128 * traversal loop stays a flat iterate-until-error sequence.
129 * HOW: 1. return 0 immediately for the "." and ".." entries.
130 * 2. join the parent path and entry name; on truncation record an
131 * XRDC_EUSAGE status and return -1.
132 * 3. if the entry has stat info flagged kXR_isDir, hand it to
133 * rmtree_handle_dir_entry (symlink probe / descend); otherwise treat
134 * it as a plain file and remove it via rmtree_remove_leaf.
135 */
136static int
137rmtree_one_entry(brix_conn *c, const char *path, const brix_dirent *ent,
138 int depth, unsigned flags, brix_rmtree_report report, void *u,
139 brix_status *st)
140{
141 char full[XRDC_PATH_MAX];
142
143 if (rmtree_is_dot(ent->name)) {
144 return 0;
145 }
146 if (rmtree_join(path, ent->name, full, sizeof(full)) != 0) {
147 brix_status_set(st, XRDC_EUSAGE, 0,
148 "rmtree: path too long under %s", path);
149 return -1;
150 }
151 if (ent->have_stat && (ent->st.flags & kXR_isDir)) {
152 return rmtree_handle_dir_entry(c, full, depth, flags, report, u, st);
153 }
154 return rmtree_remove_leaf(c, full, flags, report, u, st);
155}

157/*
158 * rmtree_depth — post-order recursive delete of `path` subtree.
159 *
160 * WHAT: dirlist `path`, remove every child (files/symlinks directly, real
161 * directories by recursion), then rmdir `path` itself; return 0 on
162 * success, -1 on depth cap, dirlist failure, or any child/self error.
163 * WHY: the server has no bulk delete, so the tree must be walked bottom-up;
164 * the depth cap bounds recursion and per-entry symlink probing (see
165 * rmtree_handle_dir_entry) prevents descending a directory symlink.
166 * HOW: 1. refuse when depth reaches BRIX_RMTREE_MAXDEPTH (XRDC_EUSAGE).
167 * 2. brix_dirlist(want_stat=1); on failure propagate -1.
168 * 3. process each entry via rmtree_one_entry, stopping at the first error.
169 * 4. free the entry array, then (if no error) report and rmdir `path`
170 * itself, honouring BRIX_RMTREE_DRYRUN and callback abort.
171 */
172static int
173rmtree_depth(brix_conn *c, const char *path, int depth, unsigned flags,
174 brix_rmtree_report report, void *u, brix_status *st)
175{
176 brix_dirent *ents = NULL;
177 size_t n = 0, i;
178 int rc = 0;
179
180 if (depth >= BRIX_RMTREE_MAXDEPTH) {
181 brix_status_set(st, XRDC_EUSAGE, 0, "rmtree: depth cap at %s", path);
182 return -1;
183 }
184 if (brix_dirlist(c, path, 1, &ents, &n, st) != 0) {
185 return -1;
186 }
187 for (i = 0; i < n && rc == 0; i++) {
188 rc = rmtree_one_entry(c, path, &ents[i], depth, flags, report, u, st);
189 }
190 free(ents);
191 if (rc != 0) {
192 return rc;
193 }
194 if (report != NULL && report(path, 1, u) != 0) {
195 brix_status_set(st, XRDC_EUSAGE, 0,
196 "rmtree: aborted by report callback at %s", path);
197 return -1;
198 }
199 if (!(flags & BRIX_RMTREE_DRYRUN) && brix_rmdir(c, path, st) != 0) {
200 return -1;
201 }
202 return 0;
203}

205/*
206 * brix_rmtree — post-order recursive delete of a remote tree.
207 *
208 * WHAT: guard the export root and all-slash paths, then recurse.
209 * WHY: "/" and "//" both resolve to the export root on the server; rejecting
210 * any path made solely of '/' characters prevents accidental fleet-wide
211 * deletes from paths that collapse to root.
212 * HOW: scan for at least one non-'/' byte; NULL / empty are also refused. */
213int
214brix_rmtree(brix_conn *c, const char *path, unsigned flags,
215 brix_rmtree_report report, void *u, brix_status *st)
216{
217 const char *p;
218
219 if (path == NULL || path[0] == '\0') {
220 brix_status_set(st, XRDC_EUSAGE, 0,
221 "rmtree: refusing to delete the export root");
222 return -1;
223 }
224 /* Reject paths that consist entirely of '/' characters ("/" "//" etc). */
225 for (p = path; *p == '/'; p++) { }
226 if (*p == '\0') {
227 brix_status_set(st, XRDC_EUSAGE, 0,
228 "rmtree: refusing to delete the export root");
229 return -1;
230 }
231 return rmtree_depth(c, path, 0, flags, report, u, st);
232}
