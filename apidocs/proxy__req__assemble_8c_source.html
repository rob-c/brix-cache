1/* File: proxy_req_assemble.c — GSI delegated-credential assembly (phase-57 §F6)
2 * WHAT: brix_gsi_assemble_proxy() ASSEMBLES the usable delegated credential:
3 * given the client-signed proxy cert (PEM), the request private key we
4 * generated in brix_gsi_build_pxyreq() (reqkey), and the signer's EEC chain
5 * (PEM), it verifies the proxy's public key matches reqkey and concatenates
6 * proxy (leaf) + chain + reqkey into one malloc'd NUL-terminated PEM blob.
7 *
8 * WHY: This is the GSI server / TPC destination's final step after kXGC_sigpxy —
9 * it turns the signed proxy back into a credential the destination can present
10 * as the user. The private key MUST be in the blob: every downstream consumer
11 * (TPC pull, proxy_ssl_certificate_key, ucred/origin-auth) loads cert chain
12 * AND key from the one stored PEM, and reqkey exists only in this process —
13 * omitting it makes the credential unable to authenticate anything. Split out
14 * of proxy_req.c (phase-79 file-size guard) as one cohesive unit; the request
15 * build stays in proxy_req.c and the issuance/signing lives in
16 * proxy_req_sign.c.
17 *
18 * HOW: parse the signed proxy PEM → compare its pubkey to reqkey (a mismatch
19 * would be someone else's credential) → PEM-export reqkey → overflow-guard the
20 * PEM lengths (proxy_pem_len is wire-supplied) → malloc + copy proxy, chain,
21 * key, NUL-terminate. No goto: each helper owns and frees its own temporaries
22 * linearly. The overflow-checked brix_size_add comes from
23 * <core/compat/safe_size.h> via proxy_req_internal.h. */

51/* Parse the signed proxy PEM and verify its public key matches the request
52 * key we generated (a mismatched proxy would be someone else's credential).
53 * 0 on success; -1 with a->err set on parse failure or key mismatch. */
54static int
55asm_check_key_match(const asm_args *a, EVP_PKEY *reqkey)
56{
57 BIO *pbio;
58 X509 *proxy;
59 EVP_PKEY *ppub;
60 int match;
61
62 pbio = BIO_new_mem_buf(a->proxy_pem, (int) a->proxy_pem_len);
63 proxy = pbio ? PEM_read_bio_X509(pbio, NULL, NULL, NULL) : NULL;
64 if (proxy == NULL) {
65 BIO_free(pbio);
66 if (a->err) snprintf(a->err, a->errcap,
67 "gsi assemble: cannot parse signed proxy");
68 return -1;
69 }
70
71 ppub = X509_get_pubkey(proxy);
72 match = (ppub != NULL && EVP_PKEY_eq(ppub, reqkey) == 1);
73 EVP_PKEY_free(ppub);
74 X509_free(proxy);
75 BIO_free(pbio);
76 if (!match) {
77 if (a->err) snprintf(a->err, a->errcap,
78 "gsi assemble: proxy key does not match request key");
79 return -1;
80 }
81 return 0;
82}

84/* PEM-serialize the request private key into a malloc'd buffer so it can be
85 * embedded in the assembled credential. Returns the buffer (caller cleanses
86 * and frees) with *len set, or NULL with a->err set. */
87static uint8_t *
88asm_export_key(const asm_args *a, EVP_PKEY *reqkey, size_t *len)
89{
90 BIO *kb;
91 char *kd = NULL;
92 long kl;
93 uint8_t *out = NULL;
94
95 kb = BIO_new(BIO_s_mem());
96 if (kb == NULL
97 || PEM_write_bio_PrivateKey(kb, reqkey, NULL, NULL, 0, NULL, NULL) != 1)
98 {
99 BIO_free(kb);
100 if (a->err) snprintf(a->err, a->errcap,
101 "gsi assemble: cannot export request key");
102 return NULL;
103 }
104 kl = BIO_get_mem_data(kb, &kd);
105 if (kl > 0) {
106 out = malloc((size_t) kl);
107 }
108 if (out != NULL) {
109 memcpy(out, kd, (size_t) kl);
110 *len = (size_t) kl;
111 } else if (a->err) {
112 snprintf(a->err, a->errcap, "gsi assemble: cannot export request key");
113 }
114 BIO_free(kb);
115 return out;
116}

118/* Concatenate proxy (leaf) + signer chain + request key into one malloc'd
119 * NUL-terminated PEM blob. Order: certs first, key last — a cert-reader stops
120 * cleanly at the trailing key block while a key-reader skips the cert blocks.
121 * proxy_pem_len is wire-supplied (client-sent kXGC_sigpxy) so the length
122 * additions are overflow-guarded. 0 on success; -1 with a->err set. */
123static int
124asm_concat_pem(const asm_args *a, uint8_t **out_pem, size_t *out_len)
125{
126 size_t total, need;
127 uint8_t *out;
128
129 if (brix_size_add(a->proxy_pem_len, a->chain_pem_len, &total) != NGX_OK
130 || brix_size_add(total, a->key_pem_len, &total) != NGX_OK
131 || brix_size_add(total, 1, &need) != NGX_OK) {
132 if (a->err) snprintf(a->err, a->errcap,
133 "gsi assemble: PEM length overflow");
134 return -1;
135 }
136 out = malloc(need);
137 if (out == NULL) {
138 if (a->err) snprintf(a->err, a->errcap, "gsi assemble: out of memory");
139 return -1;
140 }
141 memcpy(out, a->proxy_pem, a->proxy_pem_len);
142 memcpy(out + a->proxy_pem_len, a->chain_pem, a->chain_pem_len);
143 memcpy(out + a->proxy_pem_len + a->chain_pem_len, a->key_pem,
144 a->key_pem_len);
145 out[total] = '\0';
146 *out_pem = out;
147 *out_len = total;
148 return 0;
149}

152brix_gsi_assemble_proxy(const brix_gsi_blob_t *proxy_pem, EVP_PKEY *reqkey,

162 if (ebuf) snprintf(ebuf, ecap, "gsi assemble: bad arguments");
163 return -1;
164 }
165
166 a.proxy_pem = proxy_pem->data;
167 a.proxy_pem_len = proxy_pem->len;
168 a.chain_pem = chain_pem->data;
169 a.chain_pem_len = chain_pem->len;
170 a.key_pem = NULL;
171 a.key_pem_len = 0;
172 a.err = ebuf;
173 a.errcap = ecap;
174
175 /* The signed proxy's public key MUST match the request key we generated. */
176 if (asm_check_key_match(&a, reqkey) != 0) {
177 return -1;
178 }
179
180 /* Delegated credential PEM = proxy (leaf), signer chain, then the request
181 * private key — the complete credential the holder can authenticate with. */
182 {
183 uint8_t *keybuf;
184 size_t keylen = 0;
185 int rc;
186
187 keybuf = asm_export_key(&a, reqkey, &keylen);
188 if (keybuf == NULL) {
189 return -1;
190 }
191 a.key_pem = keybuf;
192 a.key_pem_len = keylen;
193 rc = asm_concat_pem(&a, &out_pem->data, &out_pem->len);
194 OPENSSL_cleanse(keybuf, keylen);
195 free(keybuf);
196 return rc;
197 }
