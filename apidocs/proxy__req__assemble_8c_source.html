1/* File: proxy_req_assemble.c — GSI delegated-credential assembly (phase-57 §F6)
2 * WHAT: brix_gsi_assemble_proxy() ASSEMBLES the usable delegated proxy chain:
3 * given the client-signed proxy cert (PEM), the request private key we
4 * generated in brix_gsi_build_pxyreq() (reqkey), and the signer's EEC chain
5 * (PEM), it verifies the proxy's public key matches reqkey and concatenates
6 * proxy (leaf) + chain into one malloc'd NUL-terminated PEM blob.
7 *
8 * WHY: This is the GSI server / TPC destination's final step after kXGC_sigpxy —
9 * it turns the signed proxy back into a credential the destination can present
10 * as the user. Split out of proxy_req.c (phase-79 file-size guard) as one
11 * cohesive unit; the request build stays in proxy_req.c and the issuance/signing
12 * lives in proxy_req_sign.c.
13 *
14 * HOW: parse the signed proxy PEM → compare its pubkey to reqkey (a mismatch
15 * would be someone else's credential) → overflow-guard the two PEM lengths
16 * (proxy_pem_len is wire-supplied) → malloc proxy_len + chain_len + 1 → copy
17 * proxy then chain, NUL-terminate. No goto: each helper owns and frees its own
18 * temporaries linearly. The overflow-checked brix_size_add comes from
19 * <core/compat/safe_size.h> via proxy_req_internal.h. */

43/* Parse the signed proxy PEM and verify its public key matches the request
44 * key we generated (a mismatched proxy would be someone else's credential).
45 * 0 on success; -1 with a->err set on parse failure or key mismatch. */
46static int
47asm_check_key_match(const asm_args *a, EVP_PKEY *reqkey)
48{
49 BIO *pbio;
50 X509 *proxy;
51 EVP_PKEY *ppub;
52 int match;
53
54 pbio = BIO_new_mem_buf(a->proxy_pem, (int) a->proxy_pem_len);
55 proxy = pbio ? PEM_read_bio_X509(pbio, NULL, NULL, NULL) : NULL;
56 if (proxy == NULL) {
57 BIO_free(pbio);
58 if (a->err) snprintf(a->err, a->errcap,
59 "gsi assemble: cannot parse signed proxy");
60 return -1;
61 }
62
63 ppub = X509_get_pubkey(proxy);
64 match = (ppub != NULL && EVP_PKEY_eq(ppub, reqkey) == 1);
65 EVP_PKEY_free(ppub);
66 X509_free(proxy);
67 BIO_free(pbio);
68 if (!match) {
69 if (a->err) snprintf(a->err, a->errcap,
70 "gsi assemble: proxy key does not match request key");
71 return -1;
72 }
73 return 0;
74}

76/* Concatenate proxy (leaf) + signer chain into one malloc'd NUL-terminated
77 * PEM blob. proxy_pem_len is wire-supplied (client-sent kXGC_sigpxy) so the
78 * length additions are overflow-guarded. 0 on success; -1 with a->err set. */
79static int
80asm_concat_pem(const asm_args *a, uint8_t **out_pem, size_t *out_len)
81{
82 size_t total, need;
83 uint8_t *out;
84
85 if (brix_size_add(a->proxy_pem_len, a->chain_pem_len, &total) != NGX_OK
86 || brix_size_add(total, 1, &need) != NGX_OK) {
87 if (a->err) snprintf(a->err, a->errcap,
88 "gsi assemble: PEM length overflow");
89 return -1;
90 }
91 out = malloc(need);
92 if (out == NULL) {
93 if (a->err) snprintf(a->err, a->errcap, "gsi assemble: out of memory");
94 return -1;
95 }
96 memcpy(out, a->proxy_pem, a->proxy_pem_len);
97 memcpy(out + a->proxy_pem_len, a->chain_pem, a->chain_pem_len);
98 out[total] = '\0';
99 *out_pem = out;
100 *out_len = total;
101 return 0;
102}
