1/*
2 * trace.c — §15 client diagnostics: wire-trace + per-opcode timing.
3 *
4 * WHAT: Decode requestid/status codes to names, emit one human-readable line per
5 * wire frame (optionally with a bounded hexdump), and accumulate/print
6 * per-opcode round-trip times.
7 * WHY: The native client owns both ends of the protocol and `libxrdproto` can
8 * name every kXR_* struct it frames — so `--wire-trace`/`--timing` give a
9 * paste-into-a-bug view the stock terse tools cannot. Off by default; every
10 * hook in frame.c/conn.c is a single `if (c->diag.*)` when disabled.
11 * HOW: Switch tables over our own opcodes.h constants (clean-room: names from the
12 * wire spec, never from XrdCl). All output goes to stderr so stdout stays
13 * clean for the conformance harness.

48/*
49 * Phase 40 (a): bounded backoff jitter for the SYNCHRONOUS retry / kXR_wait
50 * paths. Returns a pseudo-random value in [0, span_ms). Lazily seeded from the
51 * monotonic clock so concurrent clients/processes spread their wake-ups instead
52 * of retrying in lockstep (thundering-herd). Same xorshift the async core uses
53 * (aio.c rc_worker_main); kept here as the single leaf helper both frame.c and
54 * xrdcp.c reuse, with NO dependency on the aio/thread machinery.
55 */
56unsigned
57brix_jitter_ms(unsigned span_ms)
58{
59 static uint64_t state; /* lazily seeded; 0 until first use */
60 if (span_ms == 0) {
61 return 0;
62 }
63 if (state == 0) {
64 state = brix_mono_ns() | 1ull;
65 }
66 state ^= state << 13;
67 state ^= state >> 7;
68 state ^= state << 17;
69 return (unsigned) (state % span_ms);
70}

72/* Bounded hexdump (cap 256 bytes) of buf[0..n) to stderr, 16 bytes per row. */
73static void
74hexdump(const uint8_t *buf, uint32_t n)
75{
76 uint32_t cap = (n < 256) ? n : 256;
77 uint32_t i;
78 for (i = 0; i < cap; i += 16) {
79 uint32_t j;
80 fprintf(stderr, " %04x: ", i);
81 for (j = i; j < i + 16 && j < cap; j++) {
82 fprintf(stderr, "%02x ", buf[j]);
83 }
84 fprintf(stderr, "\n");
85 }
86 if (n > cap) {
87 fprintf(stderr, " … (%u more bytes)\n", n - cap);
88 }
89}

91void
92brix_trace_frame(brix_conn *c, int dir, uint16_t sid, int code,
93 int is_request, uint32_t dlen, const void *body, uint32_t blen)
94{
95 const uint8_t *b = (const uint8_t *) body;
96
97 if (is_request) {
98 fprintf(stderr, "%c sid=%u %s dlen=%u\n",
99 dir, sid, brix_reqid_name(code), dlen);
100 } else {
101 /* Decode the few high-value fields per status, from our own structs. */
102 if (code == kXR_error && b != NULL && blen >= 4) {
103 int errnum = (int) xrd_get_u32_be(b);
104 fprintf(stderr, "%c sid=%u error errnum=%d (%s) \"%.*s\"\n",
105 dir, sid, errnum, brix_kxr_name(errnum),
106 (int) (blen > 4 ? blen - 4 : 0),
107 blen > 4 ? (const char *) (b + 4) : "");
108 } else if (code == kXR_redirect && b != NULL && blen >= 4) {
109 int port = (int) xrd_get_u32_be(b);
110 fprintf(stderr, "%c sid=%u redirect → %.*s:%d\n",
111 dir, sid, (int) (blen > 4 ? blen - 4 : 0),
112 blen > 4 ? (const char *) (b + 4) : "", port);
113 } else if (code == kXR_wait && b != NULL && blen >= 4) {
114 fprintf(stderr, "%c sid=%u wait %us\n",
115 dir, sid, xrd_get_u32_be(b));
116 } else {
117 fprintf(stderr, "%c sid=%u %s dlen=%u\n",
118 dir, sid, brix_status_name(code), dlen);
119 }
120 }
121
122 if (c->diag.wire_trace >= 2 && b != NULL && blen > 0) {
123 hexdump(b, blen);
124 }
125}

127void
128brix_timing_report(const brix_conn *c)
129{
130 int i, any = 0;
131 for (i = 0; i < XRDC_NOP; i++) {
132 uint64_t n = c->diag.rtt[i].n;
133 double tot, avg;
134 if (n == 0) {
135 continue;
136 }
137 if (!any) {
138 fprintf(stderr, "--- per-opcode RTT (n / total / min / avg / max ms) ---\n");
139 any = 1;
140 }
141 tot = (double) c->diag.rtt[i].tot_ns / 1e6;
142 avg = tot / (double) n;
143 fprintf(stderr, " %-12s n=%-4llu tot=%.3f min=%.3f avg=%.3f max=%.3f\n",
144 brix_reqid_name(i + kXR_1stRequest),
145 (unsigned long long) n, tot,
146 (double) c->diag.rtt[i].min_ns / 1e6, avg,
147 (double) c->diag.rtt[i].max_ns / 1e6);
148 }
