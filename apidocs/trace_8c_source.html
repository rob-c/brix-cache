2 * trace.c — §15 client diagnostics: wire-trace + per-opcode timing.
3 *
4 * WHAT: Decode requestid/status codes to names, emit one human-readable line per
5 * wire frame (optionally with a bounded hexdump), and accumulate/print
6 * per-opcode round-trip times.
7 * WHY: The native client owns both ends of the protocol and `libxrdproto` can
8 * name every kXR_* struct it frames — so `--wire-trace`/`--timing` give a

48/*
49 * Phase 40 (a): bounded backoff jitter for the SYNCHRONOUS retry / kXR_wait
50 * paths. Returns a pseudo-random value in [0, span_ms). Lazily seeded from the
51 * monotonic clock so concurrent clients/processes spread their wake-ups instead
52 * of retrying in lockstep (thundering-herd). Same xorshift the async core uses
53 * (aio.c rc_worker_main); kept here as the single leaf helper both frame.c and
54 * xrdcp.c reuse, with NO dependency on the aio/thread machinery.
55 */
56unsigned
57brix_jitter_ms(unsigned span_ms)
58{
59 static uint64_t state; /* lazily seeded; 0 until first use */
60 if (span_ms == 0) {
61 return 0;
62 }
63 if (state == 0) {
64 state = brix_mono_ns() | 1ull;
65 }
66 state ^= state << 13;
67 state ^= state >> 7;
68 state ^= state << 17;
69 return (unsigned) (state % span_ms);
70}

72/* Bounded hexdump (cap 256 bytes) of buf[0..n) to stderr, 16 bytes per row. */
73static void
74hexdump(const uint8_t *buf, uint32_t n)
75{
76 uint32_t cap = (n < 256) ? n : 256;
77 uint32_t i;
78 for (i = 0; i < cap; i += 16) {
79 uint32_t j;
80 fprintf(stderr, " %04x: ", i);
81 for (j = i; j < i + 16 && j < cap; j++) {
82 fprintf(stderr, "%02x ", buf[j]);
83 }
84 fprintf(stderr, "\n");
85 }
86 if (n > cap) {
87 fprintf(stderr, " … (%u more bytes)\n", n - cap);
88 }
89}

91/* ---- Trace one kXR_error response frame ----
92 *
93 * WHAT: Emits a single stderr line decoding an error frame: the 4-byte
94 * big-endian errnum, its symbolic name, and the trailing message text
95 * (bytes past the errnum, printed with an explicit length so the buffer
96 * need not be null-terminated).
97 *
98 * WHY: The errnum-plus-message layout is specific to kXR_error; isolating it
99 * keeps brix_trace_response a flat dispatch and each per-status decoder
100 * independently reviewable. Caller guarantees b != NULL and blen >= 4.
101 *
102 * HOW: 1. Read the leading 32-bit BE errnum via the unaligned-safe accessor.
103 * 2. Print errnum, its brix_kxr_name(), and the message slice
104 * (blen - 4 bytes when present, else an empty string).
105 */
106static void
107brix_trace_error(int dir, uint16_t sid, const uint8_t *b, uint32_t blen)
108{
109 int errnum = (int) xrd_get_u32_be(b);
110 fprintf(stderr, "%c sid=%u error errnum=%d (%s) \"%.*s\"\n",
111 dir, sid, errnum, brix_kxr_name(errnum),
112 (int) (blen > 4 ? blen - 4 : 0),
113 blen > 4 ? (const char *) (b + 4) : "");
114}

116/* ---- Trace one kXR_redirect response frame ----
117 *
118 * WHAT: Emits a single stderr line decoding a redirect frame: the 4-byte
119 * big-endian port followed by the target host slice (bytes past the
120 * port), rendered as host:port.
121 *
122 * WHY: The port-then-host layout is unique to kXR_redirect; a dedicated
123 * decoder mirrors brix_trace_error and keeps the dispatcher branch-free
124 * of field math. Caller guarantees b != NULL and blen >= 4.
125 *
126 * HOW: 1. Read the leading 32-bit BE port via the unaligned-safe accessor.
127 * 2. Print the host slice (blen - 4 bytes when present, else empty)
128 * joined to the port with a colon.
129 */
130static void
131brix_trace_redirect(int dir, uint16_t sid, const uint8_t *b, uint32_t blen)
132{
133 int port = (int) xrd_get_u32_be(b);
134 fprintf(stderr, "%c sid=%u redirect → %.*s:%d\n",
135 dir, sid, (int) (blen > 4 ? blen - 4 : 0),
136 blen > 4 ? (const char *) (b + 4) : "", port);
137}

139/* ---- Trace one kXR_wait response frame ----
140 *
141 * WHAT: Emits a single stderr line decoding a wait frame's 4-byte big-endian
142 * wait interval, in seconds.
143 *
144 * WHY: kXR_wait carries only the retry delay; a leaf decoder keeps it uniform
145 * with the other per-status helpers. Caller guarantees b != NULL and
146 * blen >= 4.
147 *
148 * HOW: 1. Read the 32-bit BE interval via the unaligned-safe accessor.
149 * 2. Print it with a trailing 's' for seconds.
150 */
151static void
152brix_trace_wait(int dir, uint16_t sid, const uint8_t *b)
153{
154 fprintf(stderr, "%c sid=%u wait %us\n",
155 dir, sid, xrd_get_u32_be(b));
156}

158/* ---- Trace a response (non-request) frame ----
159 *
160 * WHAT: Emits one stderr line for a response frame, decoding the few
161 * high-value bodies (error, redirect, wait) when the body is present and
162 * long enough, or falling back to a generic status-name + dlen line.
163 *
164 * WHY: Splitting the response side out of brix_trace_frame collapses the
165 * status branch ladder into a flat early-dispatch and lets each body
166 * decoder be a small single-purpose helper — the same wire fields are
167 * decoded from our own structs, never from XrdCl.
168 *
169 * HOW: 1. If code is kXR_error with a >= 4-byte body, delegate to
170 * brix_trace_error.
171 * 2. Else if kXR_redirect with a >= 4-byte body, delegate to
172 * brix_trace_redirect.
173 * 3. Else if kXR_wait with a >= 4-byte body, delegate to
174 * brix_trace_wait.
175 * 4. Otherwise print the generic status-name + dlen line.
176 */
177static void
178brix_trace_response(int dir, uint16_t sid, int code, uint32_t dlen,
179 const uint8_t *b, uint32_t blen)
180{
181 if (code == kXR_error && b != NULL && blen >= 4) {
182 brix_trace_error(dir, sid, b, blen);
183 } else if (code == kXR_redirect && b != NULL && blen >= 4) {
184 brix_trace_redirect(dir, sid, b, blen);
185 } else if (code == kXR_wait && b != NULL && blen >= 4) {
186 brix_trace_wait(dir, sid, b);
187 } else {
188 fprintf(stderr, "%c sid=%u %s dlen=%u\n",
189 dir, sid, brix_status_name(code), dlen);
190 }
191}

193void
194brix_trace_frame(brix_conn *c, int dir, uint16_t sid, int code,
195 int is_request, uint32_t dlen, const void *body, uint32_t blen)
196{
197 const uint8_t *b = (const uint8_t *) body;
198
199 if (is_request) {
200 fprintf(stderr, "%c sid=%u %s dlen=%u\n",
201 dir, sid, brix_reqid_name(code), dlen);
202 } else {
203 brix_trace_response(dir, sid, code, dlen, b, blen);
204 }
205
206 if (c->diag.wire_trace >= 2 && b != NULL && blen > 0) {
207 hexdump(b, blen);
208 }
209}

211void
212brix_timing_report(const brix_conn *c)
213{
214 int i, any = 0;
215 for (i = 0; i < XRDC_NOP; i++) {
216 uint64_t n = c->diag.rtt[i].n;
217 double tot, avg;
218 if (n == 0) {
219 continue;
220 }
221 if (!any) {
222 fprintf(stderr, "--- per-opcode RTT (n / total / min / avg / max ms) ---\n");
223 any = 1;
224 }
225 tot = (double) c->diag.rtt[i].tot_ns / 1e6;
226 avg = tot / (double) n;
227 fprintf(stderr, " %-12s n=%-4llu tot=%.3f min=%.3f avg=%.3f max=%.3f\n",
228 brix_reqid_name(i + kXR_1stRequest),
229 (unsigned long long) n, tot,
230 (double) c->diag.rtt[i].min_ns / 1e6, avg,
231 (double) c->diag.rtt[i].max_ns / 1e6);
232 }
