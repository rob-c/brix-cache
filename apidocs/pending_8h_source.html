8/*
9 * Shared-memory table of in-flight kYR_locate requests.
10 *
11 * When a worker sends kYR_locate to the CMS manager it inserts an entry here
12 * keyed by the per-worker streamid. When kYR_select arrives on that worker's
13 * CMS socket, recv.c looks up the entry, writes the redirect host/port into
14 * it, and wakes the suspended XRootD session.
15 *
16 * Entries expire after cms_locate_timeout milliseconds so that a lost reply
17 * does not strand a slot forever.
18 */
19
20#define BRIX_PENDING_LOCATE_SLOTS 32 /* concurrent in-flight locates */
