9/*
10 * WHAT: Convert a proleptic-Gregorian civil date (y, m, d) to a day count
11 * relative to the Unix epoch (1970-01-01 == 0).
12 * WHY: Policy expiry must be compared against ngx_time() without depending on
13 * the host timezone, so we compute a UTC epoch directly rather than via
14 * mktime() (which is local-time and not async-signal-safe).
15 * HOW: Howard Hinnant's well-known days_from_civil algorithm. It shifts the
16 * year so March is month 1 (Feb's leap day lands at year-end), groups
17 * years into 400-year "eras" (each exactly 146097 days), then offsets by
18 * 719468 to re-anchor era day 0 onto the Unix epoch. Constants are exact,
19 * not approximations. Reference: howardhinnant.github.io/date_algorithms.html
20 * Returns signed days since 1970-01-01 (negative for pre-epoch dates).
21 */
22int64_t
23s3_post_days_from_civil(int y, unsigned m, unsigned d)
24{
25 int64_t era;
26 unsigned yoe, doy, doe;
27 int mp;
28
29 y -= m <= 2; /* Jan/Feb count to prior year */
30 era = (y >= 0 ? y : y - 399) / 400; /* 400-year era index */
31 yoe = (unsigned) (y - era * 400); /* year-of-era [0, 399] */
32 mp = (int) m + (m > 2 ? -3 : 9); /* month, March=0 .. Feb=11 */
33 doy = (153 * (unsigned) mp + 2) / 5 + d - 1;/* day-of-year, March 1 == 0 */
34 doe = yoe * 365 + yoe / 4 - yoe / 100 + doy;/* day-of-era [0, 146096] */
35
36 return era * 146097 + (int64_t) doe - 719468;
37}

40/*
41 * WHAT: Parse an ISO 8601 UTC timestamp "YYYY-MM-DDTHH:MM:SSZ" to a time_t.
42 * WHY: POST policy "expiration" is given in this exact UTC form; we need it as
43 * an epoch seconds value to compare against ngx_time(). HOW: strict
44 * sscanf with field widths, range-check each component (se <= 60 allows a
45 * leap second), then combine the civil-day count with the time-of-day.
46 * Returns NGX_OK with *out set, or NGX_ERROR on a malformed/out-of-range value.
47 */
48ngx_int_t
49s3_post_parse_iso8601(const char *s, time_t *out)
50{
51 int y, mo, d, h, mi, se;
52
53 if (sscanf(s, "%4d-%2d-%2dT%2d:%2d:%2dZ",
54 &y, &mo, &d, &h, &mi, &se) != 6)
55 {
56 return NGX_ERROR;
57 }
58
59 if (y < 1970 || mo < 1 || mo > 12 || d < 1 || d > 31
60 || h < 0 || h > 23 || mi < 0 || mi > 59 || se < 0 || se > 60)
61 {
62 return NGX_ERROR;
63 }
64
65 /* epoch = (whole days since 1970) * 86400 + seconds within the day. */
66 *out = (time_t) (s3_post_days_from_civil(y, (unsigned) mo, (unsigned) d)
67 * 86400 + h * 3600 + mi * 60 + se);
68 return NGX_OK;
69}

86/*
87 * WHAT: Evaluate the object-form of a POST-policy condition
88 * {"field":"value", ...} — implicit "equals" for every key/value pair.
89 * WHY: The AWS object shape means every listed field must match exactly; the
90 * special "bucket" key is not a form field but the configured bucket name.
91 * HOW: Iterate each member, reject non-string values as malformed, compare
92 * "bucket" against cf->bucket and all other keys against the form value.
93 * Returns NGX_OK if every pair matches, NGX_DECLINED on the first mismatch,
94 * NGX_ERROR if a value is not a string (malformed condition).
95 */
96static ngx_int_t
97s3_post_cond_object(ngx_http_s3_loc_conf_t *cf, const s3_post_form_t *form,
98 json_t *cond)
99{
100 const char *key;
101 json_t *value;
102
103 json_object_foreach(cond, key, value) {
104 const char *expected;
105 const char *actual;
106
107 if (!json_is_string(value)) {
108 return NGX_ERROR;
109 }
110
111 expected = json_string_value(value);
112 /* "bucket" is not a form field — match the configured bucket name. */
113 if (strcmp(key, "bucket") == 0) {
114 if (cf->bucket.len != strlen(expected)
115 || ngx_strncmp(cf->bucket.data, (u_char *) expected,
116 cf->bucket.len) != 0)
117 {
118 return NGX_DECLINED;
119 }
120 continue;
121 }
122
123 actual = s3_post_field_value(form, key);
124 if (s3_post_check_field_eq(actual, expected) != NGX_OK) {
125 return NGX_DECLINED;
126 }
127 }
128 return NGX_OK;
129}

132/*
133 * WHAT: Evaluate the content-length-range array condition
134 * ["content-length-range", min, max] against the uploaded file size.
135 * WHY: This is the only array operator whose arguments are integers; it bounds
136 * how large the uploaded body may be.
137 * HOW: Require exactly three integer-typed elements, then range-check
138 * form->file_len inclusively.
139 * Returns NGX_OK if the size is within [min, max], NGX_DECLINED if outside,
140 * NGX_ERROR if the argument shapes are malformed.
141 */
142static ngx_int_t
143s3_post_cond_length_range(const s3_post_form_t *form, json_t *cond)
144{
145 json_int_t minv, maxv;
146
147 if (json_array_size(cond) != 3
148 || !json_is_integer(json_array_get(cond, 1))
149 || !json_is_integer(json_array_get(cond, 2)))
150 {
151 return NGX_ERROR;
152 }
153
154 minv = json_integer_value(json_array_get(cond, 1));
155 maxv = json_integer_value(json_array_get(cond, 2));
156 if ((json_int_t) form->file_len < minv
157 || (json_int_t) form->file_len > maxv)
158 {
159 return NGX_DECLINED;
160 }
161 return NGX_OK;
162}

165/*
166 * WHAT: Evaluate a string-valued array condition ["op", field, value] where op
167 * is "eq" (exact match) or "starts-with" (prefix match).
168 * WHY: These two operators share identical argument shapes and differ only in
169 * the final comparison, so one helper covers both.
170 * HOW: Require three string-typed elements; resolve the form value; an absent
171 * field is a violation. For "eq" compare the whole string, for
172 * "starts-with" compare the expected-length prefix (empty value => any).
173 * Returns NGX_OK if satisfied, NGX_DECLINED if violated, NGX_ERROR if the
174 * argument shapes are malformed.
175 */
176static ngx_int_t
177s3_post_cond_string_op(const s3_post_form_t *form, json_t *cond, const char *op)
178{
179 const char *field;
180 const char *expected;
181 const char *actual;
182
183 if (json_array_size(cond) != 3
184 || !json_is_string(json_array_get(cond, 1))
185 || !json_is_string(json_array_get(cond, 2)))
186 {
187 return NGX_ERROR;
188 }
189
190 field = json_string_value(json_array_get(cond, 1));
191 expected = json_string_value(json_array_get(cond, 2));
192 actual = s3_post_field_value(form, field);
193
194 if (actual == NULL) {
195 return NGX_DECLINED;
196 }
197
198 if (strcmp(op, "eq") == 0) {
199 return strcmp(actual, expected) == 0 ? NGX_OK : NGX_DECLINED;
200 }
201
202 return strncmp(actual, expected, strlen(expected)) == 0
203 ? NGX_OK : NGX_DECLINED;
204}

207/*
208 * Descriptor for one array-form operator: its literal name and the evaluator
209 * that checks a condition of that operator. `wants_string_args` selects the

217} s3_post_op_desc_t;

248/*
249 * WHAT: Evaluate the array-form of a condition ["op", arg1, arg2].
250 * WHY: Element 0 names the comparison; the remaining elements are its operands.
251 * HOW: Require a string operator, then look it up in s3_post_ops and delegate
252 * to the matching evaluator.
253 * Returns the evaluator's verdict, or NGX_ERROR if the operator is not a string
254 * or is unknown.
255 */
256static ngx_int_t
257s3_post_cond_array(const s3_post_form_t *form, json_t *cond)
258{
259 const char *op;
260 ngx_uint_t i;
261
262 if (!json_is_string(json_array_get(cond, 0))) {
263 return NGX_ERROR;
264 }
265
266 op = json_string_value(json_array_get(cond, 0));
267
268 for (i = 0; i < sizeof(s3_post_ops) / sizeof(s3_post_ops[0]); i++) {
269 if (strcmp(op, s3_post_ops[i].op) == 0) {
270 return s3_post_ops[i].eval(form, cond, op);
271 }
272 }
273
274 return NGX_ERROR;
275}

278/*
279 * WHAT: Evaluate a single POST-policy condition against the submitted form.
280 * WHY: This is the access-control core: a signed policy lists conditions the
281 * upload must satisfy; if any fails the upload is rejected even though the
282 * signature is valid. Two JSON shapes occur (AWS spec):
283 * - Object {"field":"value", ...} : every field must equal exactly
284 * (the special "bucket" key is matched against the configured bucket).
285 * - Array ["op", "field|$field", "value"] : op is "eq", "starts-with",
286 * or "content-length-range" (which bounds the file size).
287 * HOW: Dispatch on JSON type to the object-form or array-form evaluator.
288 * Returns NGX_OK if the condition is satisfied, NGX_DECLINED if it is violated,
289 * NGX_ERROR if the condition itself is malformed (unknown shape/op/types).
290 */
291ngx_int_t
292s3_post_policy_condition(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
293 const s3_post_form_t *form, json_t *cond)
294{
295 (void) r;
296
297 /* Object form: implicit "equals" for every key/value pair. */
298 if (json_is_object(cond)) {
299 return s3_post_cond_object(cf, form, cond);
300 }
301
302 /* Array form: ["op", arg1, arg2] — element 0 selects the comparison. */
303 if (json_is_array(cond) && json_array_size(cond) >= 3) {
304 return s3_post_cond_array(form, cond);
305 }
306
307 return NGX_ERROR;
308}

311/*
312 * WHAT: Parse the base64-decoded POST policy JSON and enforce all of it.
313 * WHY: The policy document is what the signature actually covers; once the
314 * signature is verified we must (a) confirm the policy has not expired and
315 * (b) confirm every listed condition holds for this upload. The upload is
316 * only authorised if BOTH hold.
317 * HOW: Load JSON, check "expiration" against ngx_time(), then evaluate each
318 * entry of "conditions" via s3_post_policy_condition().
319 * Ownership: `root` is jansson-refcounted; every early-return path json_decref()s
320 * it exactly once to avoid leaking the parse tree.
321 * Returns NGX_OK (allow), NGX_DECLINED (expired or a condition failed -> 403),
322 * NGX_ERROR (document malformed -> treated as invalid policy).
323 */
324ngx_int_t
325s3_post_validate_policy_json(ngx_http_request_t *r,
326 ngx_http_s3_loc_conf_t *cf, const s3_post_form_t *form,
327 u_char *policy_json, size_t policy_len)
328{
329 json_error_t jerr;
330 json_t *root;
331 json_t *expiration;
332 json_t *conditions;
333 size_t i;
334 time_t exp;
335
336 root = json_loadb((const char *) policy_json, policy_len, 0, &jerr);
337 if (root == NULL || !json_is_object(root)) {
338 if (root != NULL) {
339 json_decref(root); /* decref only if a tree was built */
340 }
341 return NGX_ERROR;
342 }
343
344 /* "expiration" must be present, parseable, and not in the past. */
345 expiration = json_object_get(root, "expiration");
346 if (!json_is_string(expiration)
347 || s3_post_parse_iso8601(json_string_value(expiration), &exp)
348 != NGX_OK
349 || ngx_time() > exp)
350 {
351 json_decref(root);
352 return NGX_DECLINED;
353 }
354
355 conditions = json_object_get(root, "conditions");
356 if (!json_is_array(conditions)) {
357 json_decref(root);
358 return NGX_ERROR;
359 }
360
361 /* Every condition must pass; the first non-OK result short-circuits and is
362 * propagated (DECLINED = violated, ERROR = malformed condition). */
363 for (i = 0; i < json_array_size(conditions); i++) {
364 ngx_int_t rc;
365
366 rc = s3_post_policy_condition(r, cf, form,
367 json_array_get(conditions, i));
368 if (rc != NGX_OK) {
369 json_decref(root);
370 return rc;
371 }
372 }
373
374 json_decref(root);
375 return NGX_OK;
376}
