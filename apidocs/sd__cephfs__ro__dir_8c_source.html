2 * sd_cephfs_ro_dir.c — directory iteration for the read-only CephFS-via-RADOS
3 * driver ("cephfsro").
4 *
5 * WHAT: The opendir/readdir/closedir vtable ops split out of sd_cephfs_ro.c.

33/* per-opendir: the directory inode + its leaf fragments, plus an omap-paging
34 * cursor and one buffered page of entry names. */
35typedef struct {
36 uint64_t ino;
37 uint32_t frags[CEPHFS_MAX_FRAGS];
38 uint32_t nfrags;
39 uint32_t fi; /* current fragment index */

42 char names[CEPHFS_READDIR_PAGE][256]; /* buffered page */
43 uint32_t npage;

51brix_sd_dir_t *
52cephfsro_opendir(brix_sd_instance_t *inst, const char *path, int *err_out)
53{
54 cephfsro_state_t *st = inst->state;
55 cephfs_dentry_t dn;
56 brix_sd_dir_t *d;
57 cephfsro_dir_t *ds;
58 int rc = cephfsro_resolve_retry(st, path, &dn);
59
60 if (rc != 0) {
61 if (err_out != NULL) { *err_out = (rc == 1) ? ENOENT : errno; }
62 return NULL;
63 }
64 if (dn.kind != CEPHFS_DENTRY_PRIMARY || !cephfs_mode_is_dir(dn.inode.mode)) {
65 if (err_out != NULL) { *err_out = ENOTDIR; }
66 return NULL;
67 }
68
69 d = ngx_pcalloc(inst->pool, sizeof(*d));
70 ds = ngx_pcalloc(inst->pool, sizeof(*ds));
71 if (d == NULL || ds == NULL) {
72 if (err_out != NULL) { *err_out = ENOMEM; }
73 return NULL;
74 }
75 ds->ino = dn.inode.ino;
76 ds->nfrags = (dn.nfrags < CEPHFS_MAX_FRAGS) ? dn.nfrags : CEPHFS_MAX_FRAGS;
77 if (ds->nfrags == 0) { ds->nfrags = 1; ds->frags[0] = 0; }
78 else { memcpy(ds->frags, dn.frag_enc, ds->nfrags * sizeof(uint32_t)); }
79
80 /* prime the first fragment's page so readdir starts serving immediately */
81 if (cephfsro_dir_fetch_page(st, ds) != 0) {
82 if (err_out != NULL) { *err_out = errno; }
83 return NULL;
84 }
85
86 d->inst = inst;
87 d->state = ds;
88 return d;
89}

91/* Strip the trailing "_<snapid>" (e.g. "_head") from a dir omap key to recover
92 * the dentry name. Copies up to the last '_' into out. Returns 0, or -1 if the
93 * key has no '_' (not a dentry key — caller should skip it). */
94static int
95cephfsro_key_to_name(const char *key, char *out, size_t cap)
96{
97 const char *us = strrchr(key, '_');
98 size_t n;
99
100 if (us == NULL) { return -1; }
101 n = (size_t) (us - key);
102 if (n == 0 || n + 1 > cap) { return -1; }
103 memcpy(out, key, n);
104 out[n] = '\0';
105 return 0;
106}

108/* Fetch the next page of dentry names for the directory's current fragment into
109 * ds->names[], advancing the omap cursor. Returns 0 on success (npage set, may be
110 * 0), -1/errno on a RADOS error. */
111static int
112cephfsro_dir_fetch_page(cephfsro_state_t *st, cephfsro_dir_t *ds)
113{
114 rados_ioctx_t meta = sd_ceph_conn_ioctx(st->meta);
115 char oid[64];
116 rados_omap_iter_t it = NULL;
117 rados_read_op_t op;
118 int prval = 0, rc;
119 unsigned char pmore = 0;
120 char *k = NULL, *v = NULL;
121 size_t vlen = 0;
122
123 ds->npage = 0;
124 ds->pi = 0;
125
126 snprintf(oid, sizeof(oid), "%llx.%08llx",
127 (unsigned long long) ds->ino,
128 (unsigned long long) ds->frags[ds->fi]);
129
130 {
131 int attempt;
132 for (attempt = 0; ; attempt++) {
133 op = rados_create_read_op();
134 rados_read_op_omap_get_vals2(op, ds->last_key[0] ? ds->last_key : "",
135 "", CEPHFS_READDIR_PAGE, &it, &pmore,
136 &prval);
137 rc = rados_read_op_operate(op, meta, oid, 0);
138 if (rc >= 0) { break; }
139 rados_release_read_op(op);
140 if (-rc == ENOENT) { ds->more_in_frag = 0; return 0; } /* no such frag */
141 if (cephfsro_is_transient(-rc) && attempt < st->max_retry) {
142 cephfsro_backoff(attempt);
143 continue;
144 }
145 errno = -rc;
146 return -1;
147 }
148 }
149 while (it != NULL && rados_omap_get_next(it, &k, &v, &vlen) == 0 && k != NULL) {
150 char name[256];
151 (void) v; (void) vlen;
152 if (ds->npage < CEPHFS_READDIR_PAGE
153 && cephfsro_key_to_name(k, name, sizeof(name)) == 0)
154 {
155 ngx_memcpy(ds->names[ds->npage], name, strlen(name) + 1);
156 ds->npage++;
157 }
158 ngx_memcpy(ds->last_key, k, strlen(k) + 1); /* cursor = last key seen */
159 }
160 if (it != NULL) { rados_omap_get_end(it); }
161 rados_release_read_op(op);
162 ds->more_in_frag = (pmore != 0);
163 return 0;
164}

166ngx_int_t
167cephfsro_readdir(brix_sd_dir_t *d, brix_sd_dirent_t *out)
168{
169 cephfsro_state_t *st = d->inst->state;
170 cephfsro_dir_t *ds = d->state;
171
172 for (;;) {
173 if (ds->pi < ds->npage) {
174 ngx_memcpy(out->name, ds->names[ds->pi], strlen(ds->names[ds->pi]) + 1);
175 ds->pi++;
176 return NGX_OK;
177 }
178 /* current page exhausted: more in this frag? */
179 if (ds->more_in_frag) {
180 if (cephfsro_dir_fetch_page(st, ds) != 0) { return NGX_ERROR; }
181 continue;
182 }
183 /* advance to the next fragment */
184 ds->fi++;
185 if (ds->fi >= ds->nfrags) { return NGX_DONE; } /* end of directory */
186 ds->last_key[0] = '\0';
187 ds->more_in_frag = 1; /* force a first fetch */
188 if (cephfsro_dir_fetch_page(st, ds) != 0) { return NGX_ERROR; }
189 }
190}
