1/*
2 * codec_lzma.c — xz/lzma backend for the codec abstraction (BRIX_CODEC_XZ).
3 *
4 * WHAT: Streaming .xz (de)compression via liblzma. Compress emits the .xz
5 * container (CRC64-checked); decompress auto-detects .xz.
6 * WHY: xz/lzma is a high-ratio modern codec the user explicitly requested;
7 * optional (compile-gated on -DBRIX_HAVE_LZMA).
8 * HOW: init() arms lzma_easy_encoder(preset) or lzma_stream_decoder(); step()
9 * maps the contract onto lzma_stream next_in/out + avail_in/out + lzma_code(RUN/FINISH).
10 * A zeroed lzma_stream equals LZMA_STREAM_INIT, so calloc suffices.
11 */

26static int
27lzma_be_init(void **state, brix_codec_id_t id, brix_codec_dir_t dir, int level)
28{
29 lzma_state *st = calloc(1, sizeof(*st));
30 lzma_ret r;
31
32 (void) id;
33 if (st == NULL) {
34 return BRIX_CODEC_ERR_MEM;
35 }
36 st->dir = dir;
37 if (dir == BRIX_CODEC_DIR_DECOMPRESS) {
38 r = lzma_stream_decoder(&st->strm, UINT64_MAX, 0);
39 } else {
40 r = lzma_easy_encoder(&st->strm, (uint32_t) level, LZMA_CHECK_CRC64);
41 }
42 if (r != LZMA_OK) {
43 free(st);
44 return (r == LZMA_MEM_ERROR) ? BRIX_CODEC_ERR_MEM : BRIX_CODEC_ERR_PARAM;
45 }
46 st->inited = 1;
47 *state = st;
48 return 0;
49}

51static brix_codec_rc_t
52lzma_step(void *state, const uint8_t *in, size_t in_len, size_t *in_pos,
53 uint8_t *out, size_t out_size, size_t *out_pos, int finish)
54{
55 lzma_state *st = state;
56 size_t in_avail0, out_avail0;
57 lzma_ret r;
58
59 st->strm.next_in = in + *in_pos;
60 st->strm.avail_in = in_len - *in_pos;
61 st->strm.next_out = out + *out_pos;
62 st->strm.avail_out = out_size - *out_pos;
63 in_avail0 = st->strm.avail_in;
64 out_avail0 = st->strm.avail_out;
65
66 r = lzma_code(&st->strm, finish ? LZMA_FINISH : LZMA_RUN);
67
68 *in_pos += (in_avail0 - st->strm.avail_in);
69 *out_pos += (out_avail0 - st->strm.avail_out);
70
71 if (r == LZMA_STREAM_END) {
72 return BRIX_CODEC_END;
73 }
74 if (r == LZMA_OK) {
75 return BRIX_CODEC_OK;
76 }
77 if (r == LZMA_MEM_ERROR) {
78 return BRIX_CODEC_ERR_MEM;
79 }
80 return BRIX_CODEC_ERR_DATA; /* LZMA_DATA_ERROR / LZMA_FORMAT_ERROR / ... */
81}

83static void
84lzma_be_end(void *state)
85{
86 lzma_state *st = state;
87
88 if (st == NULL) {
89 return;
90 }
91 if (st->inited) {
92 lzma_end(&st->strm);
93 }
94 free(st);
95}
