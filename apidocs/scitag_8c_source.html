1/*
2 * scitag.c — SciTags flow-id parsing from a client-supplied opaque/CGI string.
3 *
4 * WHAT: brix_pmark_parse_scitag() extracts the `scitag.flow=<N>` parameter a
5 * client may attach to a root:// open (opaque `?scitag.flow=N`) or send as an
6 * HTTP `SciTag:` header (rendered here as "scitag.flow=N"), validates it
7 * against the 16-bit SciTags range, and splits it into (experiment, activity).
8 * The flow-id encode/decode/validate helpers themselves are tiny inlines in
9 * pmark.h; this file holds the one non-trivial routine (untrusted parsing).
10 *
11 * WHY: A client tag is a top-priority override of the config-driven mapping
12 * (mapping.c), so it must be parsed defensively: out-of-range or malformed
13 * input is rejected (treated as "no tag"), never coerced to a bogus code, and
14 * the raw value never reaches the firefly JSON unescaped (we only ever emit the
15 * parsed integers, never the client's bytes).
16 *
17 * HOW: Scan for the "scitag.flow=" token at a parameter boundary (string start,
18 * or just after '?' / '&' / ';'), parse the following run of decimal digits,
19 * range-check [65,65535], and split via brix_pmark_flow_split().
20 */
21

31 return ch == '?' || ch == '&' || ch == ';';

34/*
35 * Parse the value of a located "scitag.flow=" token: `p` points at the run of
36 * decimal digits just past the '='. Range-checks the integer and splits it
37 * into (experiment, activity). Returns NGX_OK on a valid in-range flow,
38 * NGX_ERROR on malformed / out-of-range input (caller treats as "no tag").
39 */
40static ngx_int_t
41pmark_parse_flow_value(const char *p, ngx_uint_t *exp, ngx_uint_t *act)
42{
43 ngx_uint_t flow = 0;
44 int have_digits = 0;
45
46 /* Parse the run of decimal digits. Bound the accumulator so a long digit
47 * string cannot overflow before the range check rejects it. */
48 while (*p >= '0' && *p <= '9') {
49 have_digits = 1;
50 flow = flow * 10 + (ngx_uint_t) (*p - '0');
51 if (flow > BRIX_PMARK_FLOW_MAX) {
52 return NGX_ERROR; /* out of range -> "no tag" */
53 }
54 p++;
55 }
56
57 /* Must be exactly an integer terminated by end / next param. */
58 if (!have_digits || (*p != '\0' && !pmark_is_param_boundary(*p))) {
59 return NGX_ERROR;
60 }
61
62 if (brix_pmark_flow_split(flow, exp, act) != NGX_OK) {
63 return NGX_ERROR; /* < 65 etc. */
64 }
65
66 /* Defence in depth: the split fields must also be individually in range. */
67 if (brix_pmark_codes_valid(*exp, *act) != NGX_OK) {
68 return NGX_ERROR;
69 }
70
71 return NGX_OK;
72}

74ngx_int_t
75brix_pmark_parse_scitag(const char *cgi, ngx_uint_t *exp, ngx_uint_t *act)
76{
77 const char *p;
78
79 if (cgi == NULL || *cgi == '\0') {
80 return NGX_DECLINED;
81 }
82
83 /* Find "scitag.flow=" at a parameter boundary, then parse its value. */
84 for (p = cgi; *p; p++) {
85 if (p != cgi && !pmark_is_param_boundary(p[-1])) {
86 continue;
87 }
88 if (ngx_strncmp(p, BRIX_PMARK_SCITAG_KEY, BRIX_PMARK_SCITAG_KEY_LEN)
89 == 0)
90 {
91 return pmark_parse_flow_value(p + BRIX_PMARK_SCITAG_KEY_LEN,
92 exp, act);
93 }
94 }
95 return NGX_DECLINED;
96}
