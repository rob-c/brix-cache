1/*
2 * pool.c — brix_conn adapter over the generic brix_cpool engine.
3 *
4 * WHAT: brix_pool_create/checkout/checkin/destroy — the same public API as
5 * before (a pool of N brix_conn to one endpoint), now implemented as a
6 * thin vtable adapter over lib/net/cpool.c so the binary root:// path and
7 * the HTTP keep-alive metadata path share ONE pool implementation.
8 * WHY: the pool's slot/mutex/condvar/health-drop bookkeeping is transport-
9 * agnostic; only the connect/close and the connection type differ. Keeping
10 * brix_pool behavior-identical means the core root:// path is unchanged.
11 * HOW: a pool_ctx {url, opts} is the connect template stored INSIDE the heap
12 * brix_pool (so the &p->ctx handed to brix_cpool is stable for the pool's
13 * lifetime); the vtable wraps brix_connect/brix_close. No goto.
14 *
15 * Clean-room: composes the public libbrix connection API only.
16 */

43brix_pool *
44brix_pool_create(const brix_url *u, const brix_opts *o, int n, brix_status *st)
45{
46 brix_pool *p;
47
48 if (u == NULL || n < 1) {
49 brix_status_set(st, XRDC_EUSAGE, 0, "pool: bad arguments");
50 return NULL;
51 }
52 p = calloc(1, sizeof(*p));
53 if (p == NULL) {
54 brix_status_set(st, XRDC_ESOCK, 0, "pool: out of memory");
55 return NULL;
56 }
57 p->ctx.url = *u;
58 if (o != NULL) {
59 p->ctx.opts = *o;
60 }
61 p->cp = brix_cpool_create(&POOL_VT, &p->ctx, n, st); /* &p->ctx: stable */
62 if (p->cp == NULL) {
63 free(p);
64 return NULL;
65 }
66 return p;
67}
