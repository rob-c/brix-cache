1/*
2 * delete_objects.c — S3 DeleteObjects handler (POST /bucket/?delete).
3 *
4 * Parses a list of <Object><Key>…</Key></Object> entries from the request
5 * body, deletes each key, and returns a <DeleteResult> XML response with
6 * <Deleted> or <Error> entries for each object.
7 *
8 * The body XML parser uses libxml2 with network access disabled, so XML
9 * entities in <Key> values are decoded correctly before filesystem resolution.
10 * Non-regular objects (directories) that fail with EISDIR/EPERM are retried
11 * as rmdir. ENOENT is treated as success per S3 idempotency rules.
12 */
13/* WHY: S3 DeleteObjects supports batch deletion in a single POST request —
14 * far more efficient than individual DELETE calls for pipelines that need to
15 * remove hundreds or thousands of objects. Non-regular-object fallback
16 * (unlink → rmdir) handles directories transparently; ENOENT is treated as
17 * success per S3 idempotency semantics. */

54/*
55 * s3_delete_one — remove one resolved object through the VFS unlink surface
56 * (OP_DELETE metric + access-log + write gate + confinement). S3 DeleteObjects
57 * is idempotent, so a missing key (ENOENT) is reported as success. On a real
58 * failure returns NGX_ERROR and fills *err with the S3 error pair (errno is
59 * mapped the same way the old brix_ns_delete status was: EACCES/EPERM →
60 * AccessDenied, ENOTEMPTY → BucketNotEmpty, else InternalError).
61 */
62static ngx_int_t
63s3_delete_one(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
64 const char *fs_path, s3_del_err_t *err)
65{
66 ngx_http_s3_req_ctx_t *s3ctx =
67 ngx_http_get_module_ctx(r, ngx_http_brix_s3_module);
68 brix_vfs_ctx_t vctx;
69 int is_tls = 0;
70
71#if (NGX_HTTP_SSL)
72 is_tls = (r->connection->ssl != NULL) ? 1 : 0;
73#endif
74
75 brix_vfs_ctx_init(&vctx, r->pool, r->connection->log, BRIX_PROTO_S3,
76 cf->common.root_canon, cf->cache_root_canon, cf->common.allow_write,
77 is_tls, (s3ctx != NULL) ? s3ctx->identity : NULL, fs_path);
78
79 if (brix_vfs_unlink(&vctx) == NGX_OK || errno == ENOENT) {
80 return NGX_OK; /* deleted, or idempotent-missing */
81 }
82
83 if (errno == EACCES || errno == EPERM) {
84 err->code = "AccessDenied";
85 err->message = "Access Denied.";
86 } else if (errno == ENOTEMPTY) {
87 err->code = "BucketNotEmpty";
88 err->message = "The directory is not empty.";
89 } else {
90 err->code = "InternalError";
91 err->message = "Internal server error.";
92 }
93 return NGX_ERROR;
94}

96static ngx_int_t
97s3_delete_xml_append_raw(ngx_buf_t *xml_buf, size_t *xml_len, const char *text)
98{
99 size_t len;
100 size_t cap;
101
102 len = strlen(text);
103 cap = (size_t) (xml_buf->end - xml_buf->start);
104 if (*xml_len > cap || len > cap - *xml_len) {
105 return NGX_ERROR;
106 }
107
108 ngx_memcpy(xml_buf->start + *xml_len, text, len);
109 *xml_len += len;
110 return NGX_OK;
111}

151 *
152 * WHY: S3 DeleteObjects reports per-key failures inline in the batch result

206 * memory-backed buf_t from it, and send the response as application/xml with HTTP 200.
207 *
208 * WHY: The XML buffer was pre-allocated during parsing (xml_buf->start/end). We append
209 * the closing tag in-place if there is room, then wrap the completed XML into a
210 * ngx_create_temp_buf() for the nginx output filter. Metrics are finalised via
211 * s3_metrics_finalize_request_method().
212 *
213 * HOW: memcpy trailer → create temp buf → ngx_memcpy copy → last_buf=1 → send header
214 * + output filter. On allocation failure, metrics finalise with 500 and return.
215 */
216
217static void
218s3_delete_objects_finish(ngx_http_request_t *r,
219 ngx_uint_t method_slot,
220 ngx_buf_t *xml_buf,
221 size_t xml_len)
222{
223 ngx_buf_t *b;
224 const char trailer[] = "</DeleteResult>";
225
226 if (s3_delete_xml_append_raw(xml_buf, &xml_len, trailer) != NGX_OK) {
227 s3_metrics_finalize_request_method(r, method_slot,
228 NGX_HTTP_INTERNAL_SERVER_ERROR);
229 return;
230 }
231
232 b = ngx_create_temp_buf(r->pool, xml_len);
233 if (b == NULL) {
234 s3_metrics_finalize_request_method(r, method_slot,
235 NGX_HTTP_INTERNAL_SERVER_ERROR);
236 return;

239 b->last = b->pos + xml_len;

241
242 s3_metrics_finalize_request_method(r, method_slot,
243 brix_http_send_xml_buffer(r, NGX_HTTP_OK,

247/*
248 * s3_delete_objects_body — body callback invoked by nginx after the POST body
249 * has been fully buffered.
250 */
251/* WHAT: Parse the POST body XML to extract <Object><Key>…</Key></Object> entries,
252 * delete each key from the filesystem, and build a DeleteResult XML response with
253 * <Deleted> or <Error> entries per object.
254 *
255 * WHY: S3 clients send DeleteObjects as a single POST with an XML body listing all
256 * keys to remove. libxml2 is used because real clients may XML-escape key
257 * names; string scanning would delete the wrong key and emit malformed XML.
258 *
259 * Per S3 semantics:
260 * - ENOENT → success (idempotent: deleting something that doesn't exist is fine)
261 * - EISDIR/EPERM on unlink → retry with rmdir flag for directories
262 * - AccessDenied → path escaped outside root_canon via s3_resolve_key()
263 *
264 * HOW:
265 * 1. Read body via brix_http_body_read_all() (cap: S3_DEL_MAX_BODY)
266 * 2. Allocate XML output buffer (S3_DEL_XML_MAX)
267 * 3. Write XML header
268 * 4. Parse <Delete><Object><Key> entries with libxml2, resolve fs_path
269 * 5. Attempt unlink → fallback rmdir on EISDIR/EPERM
270 * 6. Emit <Deleted> (success/ENOENT) or <Error> (AccessDenied/BucketNotEmpty/InternalError)
271 * 7. Finalise response via s3_delete_objects_finish()
272 *
273 * Constraints:
274 * - Max keys: S3_DEL_MAX_KEYS (1000) — rejects XML that exceeds the limit
275 * - Body cap: S3_DEL_MAX_BODY (1 MiB) — rejects oversized XML as MalformedXML
276 * - Key length: < S3_MAX_KEY — reports InvalidArgument per offending object
277 */

295/*
296 * s3_delete_parse_body — read the buffered POST body and parse it into a
297 * validated <Delete> document.
298 *
299 * WHAT: Reads the body (cap S3_DEL_MAX_BODY), parses it with the hardened
300 * libxml2 posture, and confirms the root element is <Delete>; on success
301 * *doc_out owns the parsed document (caller must xmlFreeDoc it).
302 * WHY: Body read and XML validation are one cohesive precondition stage; the
303 * client-facing MalformedXML / body-too-large errors and the 500 paths are
304 * all resolved here so the batch loop only runs on a good document.
305 * HOW: Returns NGX_OK with a document, or NGX_DONE after finalising the
306 * response with the appropriate error (nothing left for the caller to do).
307 */
308static ngx_int_t
309s3_delete_parse_body(ngx_http_request_t *r, ngx_uint_t method_slot,
310 xmlDocPtr *doc_out)
311{
312 u_char *body_buf;
313 size_t body_len;
314 xmlDocPtr doc;
315 xmlNodePtr root;
316 ngx_int_t rc;
317
318 *doc_out = NULL;
319
320 if (r->request_body == NULL) {
321 BRIX_S3_METRIC_INC(events_total[BRIX_S3_EVENT_INTERNAL_ERROR]);
322 s3_metrics_finalize_request_method(r, method_slot,
323 NGX_HTTP_INTERNAL_SERVER_ERROR);
324 return NGX_DONE;
325 }
326
327 rc = brix_http_body_read_all(r, S3_DEL_MAX_BODY, &body_buf, &body_len);
328 if (rc == NGX_DECLINED) {
329 s3_metrics_finalize_request_method(r, method_slot,
330 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
331 "MalformedXML",
332 "Request body too large."));
333 return NGX_DONE;
334 }
335 if (rc != NGX_OK) {
336 s3_metrics_finalize_request_method(r, method_slot,
337 NGX_HTTP_INTERNAL_SERVER_ERROR);
338 return NGX_DONE;
339 }
340
341 {
342 /* Match the hardened libxml2 posture used by the WebDAV parsers
343 * (propfind/proppatch/search/lockinfo): NONET blocks network entity
344 * fetches; NO_XXE (libxml2 >= 2.13, when available) refuses external
345 * entity loading outright. NOENT/DTDLOAD are deliberately NOT set, so
346 * external entities are never substituted (no file:// XXE), and HUGE is
347 * omitted so libxml2 keeps its billion-laughs amplification cap. */
348 int opts = XML_PARSE_NONET | XML_PARSE_NOERROR | XML_PARSE_NOWARNING;
349#if defined(XML_PARSE_NO_XXE)
350 opts |= XML_PARSE_NO_XXE;
351#endif
352 doc = xmlReadMemory((const char *) body_buf, (int) body_len,
353 "delete_objects.xml", NULL, opts);
354 }
355 if (doc == NULL) {
356 s3_metrics_finalize_request_method(r, method_slot,
357 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
358 "MalformedXML",
359 "Request body is not valid DeleteObjects XML."));
360 return NGX_DONE;
361 }
362
363 root = xmlDocGetRootElement(doc);
364 if (!s3_delete_xml_name_is(root, "Delete")) {
365 xmlFreeDoc(doc);
366 s3_metrics_finalize_request_method(r, method_slot,
367 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
368 "MalformedXML",
369 "DeleteObjects root element must be Delete."));
370 return NGX_DONE;
371 }
372
373 *doc_out = doc;
374 return NGX_OK;
375}

377/*
378 * s3_delete_result_init — allocate the output XML buffer and write the
379 * <DeleteResult> preamble.
380 *
381 * WHAT: Allocates S3_DEL_XML_MAX from the request pool, points xml_buf at it,
382 * and seeds *xml_len with the length of the XML header prefix.
383 * WHY: The batch loop appends per-key elements into a fixed, pre-sized buffer
384 * (no reallocation mid-stream); this stage establishes that buffer once.
385 * HOW: Returns NGX_OK on success or NGX_ERROR on allocation failure (the
386 * caller maps that to a 500).
387 */
388static ngx_int_t
389s3_delete_result_init(ngx_http_request_t *r, ngx_buf_t *xml_buf,
390 size_t *xml_len)
391{
392 static const char header[] =
393 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
394 "<DeleteResult "
395 "xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">";
396 u_char *xml_out;
397
398 xml_out = ngx_palloc(r->pool, S3_DEL_XML_MAX);
399 if (xml_out == NULL) {
400 return NGX_ERROR;
401 }
402
403 xml_buf->start = xml_out;
404 xml_buf->end = xml_out + S3_DEL_XML_MAX;
405
406 *xml_len = sizeof(header) - 1;
407 memcpy(xml_out, header, *xml_len);
408 return NGX_OK;

411/*
412 * s3_delete_dispose_key — resolve one already-extracted key and render its
413 * batch result (<Deleted> or <Error>).
414 *
415 * WHAT: Confines the key to root_canon via s3_resolve_key(), and on success
416 * deletes it through s3_delete_one(); appends the matching per-key XML.
417 * WHY: This is the per-key auth+delete stage. The path-escape (AccessDenied)
418 * check runs BEFORE any unlink — a key that resolves outside the export root
419 * is rejected without ever touching the filesystem, which is
420 * security-load-bearing and must precede the delete.
421 * HOW: Returns NGX_OK once a result element has been appended, or NGX_ERROR if
422 * the output buffer overflowed (fatal → caller maps to 500).
423 */
424static ngx_int_t
425s3_delete_dispose_key(s3_del_ctx_t *dc, const xmlChar *key_text, size_t key_len)
426{
427 char fs_path[PATH_MAX];
428 s3_del_err_t err;
429 char key_str[S3_MAX_KEY];
430
431 ngx_memcpy(key_str, key_text, key_len);
432 key_str[key_len] = '\0';
433
434 if (!s3_resolve_key(dc->cf->common.root_canon, key_str, fs_path,
435 sizeof(fs_path), dc->cf->common.cache_store_endpoint))
436 {
437 err.code = "AccessDenied";
438 err.message = "Access Denied.";
439 return s3_delete_xml_append_error(dc->xml_buf, dc->xml_len,
440 key_text, key_len, &err);
441 }
442
443 if (s3_delete_one(dc->r, dc->cf, fs_path, &err) == NGX_OK) {
444 return s3_delete_xml_append_deleted(dc->xml_buf, dc->xml_len,
445 key_text, key_len);
446 }
447
448 return s3_delete_xml_append_error(dc->xml_buf, dc->xml_len,
449 key_text, key_len, &err);
450}

452/*
453 * s3_delete_process_object — process one <Object> node end to end.
454 *
455 * WHAT: Extracts and length-validates the object's <Key>, then either appends
456 * a per-key InvalidArgument <Error> (empty/too-long key) or dispatches the
457 * key to s3_delete_dispose_key() for confinement, deletion and result XML.
458 * WHY: One object is one loop iteration; isolating it flattens the batch loop
459 * and keeps every xmlFree paired with its xmlNodeGetContent on all exits.
460 * HOW: Returns NGX_OK (result appended — continue the batch), NGX_ERROR
461 * (output buffer overflow — fatal), or NGX_ABORT (malformed object: missing
462 * <Key> or unreadable content — whole request is MalformedXML/500). The
463 * *malformed_500 out-flag distinguishes the two NGX_ABORT sub-cases.
464 */
465static ngx_int_t
466s3_delete_process_object(s3_del_ctx_t *dc, xmlNodePtr obj, int *malformed_500)
467{
468 xmlNodePtr key_node;
469 xmlChar *key_text;
470 size_t key_len;
471 ngx_int_t rc;
472
473 *malformed_500 = 0;
474
475 key_node = s3_delete_xml_find_child(obj, "Key");
476 if (key_node == NULL) {
477 return NGX_ABORT; /* missing <Key> → MalformedXML */
478 }
479
480 key_text = xmlNodeGetContent(key_node);
481 if (key_text == NULL) {
482 *malformed_500 = 1;
483 return NGX_ABORT; /* unreadable content → 500 */
484 }
485 key_len = (size_t) xmlStrlen(key_text);
486
487 if (key_len == 0 || key_len >= S3_MAX_KEY) {
488 s3_del_err_t err = { "InvalidArgument",
489 "Object key is empty or too long." };
490 rc = s3_delete_xml_append_error(dc->xml_buf, dc->xml_len,
491 key_text, key_len, &err);
492 } else {
493 rc = s3_delete_dispose_key(dc, key_text, key_len);
494 }
495
496 xmlFree(key_text);
497 return rc;
498}

517/*
518 * s3_delete_run_batch — iterate the <Object> children of <Delete>, rendering a
519 * per-key result for each into the output XML.
520 *
521 * WHAT: Enforces the S3_DEL_MAX_KEYS cap, requires at least one <Object>, and
522 * delegates each object to s3_delete_process_object(); non-<Object> children
523 * are skipped.
524 * WHY: Separating the loop from the surrounding setup/teardown keeps this
525 * function's only concern the batch iteration and its client-facing
526 * MalformedXML edge cases (too many keys / no objects).
527 * HOW: Returns NGX_OK once every object has an appended result (caller sends
528 * the buffered response). On any error it finalises the response here and
529 * returns NGX_DONE — the malformed sub-cases pick the exact message.
530 */
531static ngx_int_t
532s3_delete_run_batch(s3_del_ctx_t *dc, ngx_uint_t method_slot, xmlNodePtr root)
533{
534 ngx_http_request_t *r = dc->r;
535 xmlNodePtr obj;
536 ngx_uint_t nkeys = 0;
537
538 for (obj = root->children; obj != NULL; obj = obj->next) {
539 int malformed_500 = 0;
540 ngx_int_t rc;
541
542 if (!s3_delete_xml_name_is(obj, "Object")) {
543 continue;
544 }
545
546 if (nkeys >= S3_DEL_MAX_KEYS) {
547 s3_metrics_finalize_request_method(r, method_slot,
548 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
549 "MalformedXML",
550 "DeleteObjects request contains too "
551 "many keys."));
552 return NGX_DONE;
553 }
554 nkeys++;
555
556 rc = s3_delete_process_object(dc, obj, &malformed_500);
557 if (rc == NGX_OK) {
558 continue;
559 }
560 if (rc == NGX_ERROR) {
561 s3_metrics_finalize_request_method(r, method_slot,
562 NGX_HTTP_INTERNAL_SERVER_ERROR);
563 return NGX_DONE;
564 }
565 /* rc == NGX_ABORT: malformed object */
566 if (malformed_500) {
567 s3_metrics_finalize_request_method(r, method_slot,
568 NGX_HTTP_INTERNAL_SERVER_ERROR);
569 } else {
570 s3_metrics_finalize_request_method(r, method_slot,
571 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
572 "MalformedXML",
573 "DeleteObjects object is missing Key."));
574 }
575 return NGX_DONE;
576 }
577
578 if (nkeys == 0) {
579 s3_metrics_finalize_request_method(r, method_slot,
580 s3_send_xml_error(r, NGX_HTTP_BAD_REQUEST,
581 "MalformedXML",
582 "DeleteObjects request has no Object entries."));
583 return NGX_DONE;
584 }
585
586 return NGX_OK;
587}

589static void
590s3_delete_objects_body_handler_inner(ngx_http_request_t *r)
