11/*
12 * Kernel-enforced path confinement API.
13 *
14 * Every function takes a persistent rootfd (O_PATH fd opened once per worker
15 * on the export root) and a client-supplied reqpath. The kernel's
16 * RESOLVE_BENEATH flag ensures no path can escape the root directory tree —
17 * symlinks, ".." traversal, and magic-links are all blocked atomically.
18 *
19 * EXDEV from any of these functions means an escape attempt was blocked;
20 * callers must map this to kXR_NotAuthorized (see brix_kxr_from_errno).

79/*
80 * brix_beneath_strip_root — turn an absolute, already-canonical path that
81 * lives under root_canon into the root-relative tail the beneath API expects.
82 *
83 * This is the bridge for migrating the legacy brix_*_confined_canon(root_canon,
84 * abspath) callers to brix_*_beneath(rootfd, rel): those functions already
85 * stripped root_canon internally before openat2, so doing it here is
86 * behaviour-preserving. It ALSO re-implements their within-root guard — a
87 * NULL return means abspath is not under root_canon (the old code's
88 * brix_path_within_root() rejection), and the caller must treat it as an
89 * escape (errno=EXDEV) and NOT fall through to a raw syscall.
90 *
91 * Returns a pointer into abspath (starting at the '/' after root_canon, or the
92 * terminating NUL when abspath == root_canon), or NULL if abspath is not within
93 * root_canon. The trailing-character check rejects a sibling like "/rootX" that
94 * merely shares the "/root" string prefix.
95 */
96static inline const char *
97brix_beneath_strip_root(const char *root_canon, const char *abspath)
98{
99 size_t rlen = strlen(root_canon);
100
101 if (strncmp(abspath, root_canon, rlen) != 0) {
102 return (const char *) 0;
103 }
104 if (abspath[rlen] == '\0' || abspath[rlen] == '/') {
105 return abspath + rlen; /* == root, or a real path boundary under root */
106 }
107 return (const char *) 0; /* prefix match but not a '/' boundary */
108}

110/* Build the full filesystem path for auth_gate: root_canon + "/" + reqpath_rel.
111 * Replaces realpath()-derived 'resolved' for ACL prefix matching.
112 * Returns number of bytes written (excluding NUL); buf is valid when < (int)bufsz. */
113static inline int
114brix_beneath_full_path(const char *root_canon, const char *reqpath,
115 char *buf, size_t bufsz)
116{
117 const char *rel = brix_beneath_rel(reqpath);
118 size_t rlen = strlen(root_canon);
119 size_t plen = (rel[0] != '\0') ? strlen(rel) : 0;
120 size_t need = rlen + (plen ? 1 + plen : 0);
121
122 if (need + 1 > bufsz) {
123 if (bufsz > 0) buf[0] = '\0';
124 return (int) need;
125 }
126
127 memcpy(buf, root_canon, rlen);
128 if (plen) {
129 buf[rlen] = '/';
130 memcpy(buf + rlen + 1, rel, plen + 1);
131 } else {
132 buf[rlen] = '\0';
133 }
134 return (int) need;
135}
