17size_t
18xrdp_pg_encode(const uint8_t *src, size_t len, int64_t file_off, uint8_t *dst)
19{
20 const uint8_t *p = src;
21 uint8_t *out = dst;
22 size_t remaining = len;
23 int64_t cur = file_off;
24
25 while (remaining > 0) {
26 /* kXR_pgPageSZ is a power of two, so (cur & (sz-1)) is the in-page offset;
27 * shorten the first fragment so every later page starts on a boundary. */
28 size_t to_boundary = (size_t) kXR_pgPageSZ
29 - (size_t) (cur & (int64_t) (kXR_pgPageSZ - 1));
30 size_t page_data = (remaining < to_boundary) ? remaining : to_boundary;
31 uint32_t crc_be;
32
33 /* Wire per page: [CRC32c(4)][data] — copy + CRC in one pass. */
34 crc_be = htonl(brix_crc32c_copy_value(p, out + XRDP_PG_CKSZ, page_data));
35 memcpy(out, &crc_be, XRDP_PG_CKSZ);
36 out += XRDP_PG_CKSZ + page_data;
37 p += page_data;
38 cur += page_data;
39 remaining -= page_data;
40 }
41 return (size_t) (out - dst);
42}

45xrdp_pg_decode(const uint8_t *pg, size_t pglen, int64_t file_off,
46 uint8_t *dst, size_t dstcap, int64_t *bad_off)
47{
48 size_t i = 0;
49 size_t out = 0;
50 int64_t cur = file_off;
51
52 if (bad_off != NULL) {
53 *bad_off = file_off;
54 }
55 if (file_off < 0) {
56 return -2;
57 }
58 while (i < pglen) {
59 uint32_t want, actual;
60 size_t to_boundary, data_n;
61
62 /* A lone/short trailing CRC (≤ 4 bytes with no full page) is malformed. */
63 if (pglen - i <= XRDP_PG_CKSZ) {
64 if (bad_off != NULL) { *bad_off = cur; }
65 return -2;
66 }
67 memcpy(&want, pg + i, XRDP_PG_CKSZ); /* unaligned-safe BE read */
68 want = ntohl(want);
69 i += XRDP_PG_CKSZ;
70
71 to_boundary = (size_t) kXR_pgPageSZ
72 - (size_t) (cur & (int64_t) (kXR_pgPageSZ - 1));
73 data_n = (pglen - i < to_boundary) ? (pglen - i) : to_boundary;
74
75 if (out + data_n > dstcap) {
76 if (bad_off != NULL) { *bad_off = cur; }
77 return -2;
78 }
79 if (cur > INT64_MAX - (int64_t) data_n) {
80 if (bad_off != NULL) { *bad_off = cur; }
81 return -2;
82 }
83 actual = brix_crc32c_copy_value(pg + i, dst + out, data_n);
84 if (actual != want) {
85 if (bad_off != NULL) { *bad_off = cur; }
86 return -1;
87 }
88 out += data_n;
89 i += data_n;
90 cur += data_n;
91 }
92 return (ssize_t) out;
93}
