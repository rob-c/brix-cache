2 * copy_range.c — shared copy_file_range(2) + pread/pwrite fallback.
3 *
4 * WHY: kXR_clone (read/clone.c) and kXR_chkpoint (write/chkpoint.c) each
5 * contained an identical private static function implementing the same
6 * two-phase copy algorithm. Centralising it here means a fix or tuning
7 * (e.g. new fallback errno, larger buffer, EINTR handling) applies to
8 * both callers automatically.
9 *
10 * HOW: Two-phase loop.
11 * Phase 1 (Linux only): syscall(__NR_copy_file_range) loop until len == 0

13 * jump to phase 2.
14 * Phase 2 (fallback): pread from src_off, pwrite to dst_off in 256 KB chunks.
15 * 256 KB matches clone.c's CLONE_COPY_BUF - large enough for HEP file

34/* brix_copy_range_fallback — portable pread/pwrite copy of [src_off, +len)
35 * WHAT: Copies `len` bytes from src_fd@src_off to dst_fd@dst_off in 256 KB
36 * chunks using pread/pwrite. Returns NGX_OK on full copy, NGX_ERROR (errno set
37 * and logged) on I/O failure or unexpected EOF/short write.
38 *
39 * WHY: Phase-2 of brix_copy_range — used both when copy_file_range(2) is not
40 * compiled in and when it returns a recoverable error mid-copy. Extracted into
41 * its own function so the two callers share one implementation and the caller
42 * stays a flat, goto-free control flow.
43 *
44 * HOW: Outer loop fills one buffer via brix_vfs_pread_full (EINTR + short-read
45 * handled by the primitive); a partial fill below `want` is a premature EOF and
46 * fails; brix_vfs_pwrite_full writes the buffer fully. Advances
47 * src_off/dst_off/len until len reaches 0.
48 */
49static ngx_int_t
50brix_copy_range_fallback(ngx_log_t *log, int src_fd, off_t src_off,
51 int dst_fd, off_t dst_off, size_t len,
52 const char *src_path, const char *dst_path)
53{
54 u_char buf[BRIX_COPY_RANGE_BUFSZ];
55
56 while (len > 0) {
57 size_t want = (len < sizeof(buf)) ? len : sizeof(buf);
58 size_t got = 0;
59
60 /* pread_full loops over EINTR/short reads through the storage seam,
61 * filling `want` bytes unless the source ends early. */
62 if (brix_vfs_pread_full(src_fd, buf, want, src_off, &got) != NGX_OK) {
63 ngx_log_error(NGX_LOG_ERR, log, errno,
64 "brix: copy_range pread failed %s",
65 src_path ? src_path : "-");
66 return NGX_ERROR;
67 }
68
69 /* A short fill means EOF before `len` bytes — the source is smaller than
70 * the requested range, which is a hard error for a fixed-length copy. */
71 if (got < want) {
72 errno = EIO;
73 ngx_log_error(NGX_LOG_ERR, log, 0,
74 "brix: copy_range pread unexpected EOF %s",
75 src_path ? src_path : "-");
76 return NGX_ERROR;
77 }
78
79 if (brix_vfs_pwrite_full(dst_fd, buf, got, dst_off) != NGX_OK) {
80 ngx_log_error(NGX_LOG_ERR, log, errno,
81 "brix: copy_range pwrite failed %s",
82 dst_path ? dst_path : "-");
83 return NGX_ERROR;
84 }
85
86 src_off += (off_t) got;
87 dst_off += (off_t) got;
88 len -= got;
89 }
90
91 return NGX_OK;
92}

96/* brix_cfr_errno_recoverable — is a copy_file_range errno finishable by fallback
97 * WHAT: Returns 1 when `err` is one of the errnos that mean copy_file_range is
98 * unsupported for this fd pair or range (EXDEV, ENOSYS, EOPNOTSUPP, EINVAL,
99 * EPERM) and the copy can still complete via the portable pread/pwrite path;
100 * returns 0 otherwise (a genuine I/O failure).
101 *
102 * WHY: The recoverable-errno set is the single decision that separates a
103 * fall-back-and-continue from a hard error. Naming it keeps the phase-1 loop's
104 * control flow flat and the exact errno list reviewable in one place.
105 *
106 * HOW: A single boolean over the five sanctioned errnos.
107 */
108static int
109brix_cfr_errno_recoverable(int err)
110{
111 return err == EXDEV || err == ENOSYS || err == EOPNOTSUPP
112 || err == EINVAL || err == EPERM;
113}

115/* brix_copy_range_cfr — phase-1 copy_file_range(2) loop over the SD seam
116 * WHAT: Copies the range via the object's own driver copy_range vtable until
117 * `*len` reaches 0. Returns NGX_OK on full copy, NGX_ERROR (errno set + logged)
118 * on unexpected EOF or a genuine failure, or NGX_AGAIN when copy_file_range is
119 * unsupported mid-copy — in which case the src_off, dst_off and len out-params are left pointing
120 * at the remaining bytes for the caller's fallback.
121 *
122 * WHY: Isolating the zero-copy loop keeps brix_copy_range a flat two-line
123 * dispatch and confines the Linux-only, errno-branching logic to one place that
124 * updates the running offsets/length in step so the fallback can resume exactly
125 * where the fast path stopped.
126 *
127 * HOW: 1) Wrap both bare fds as stack POSIX SD objects. 2) Each iteration clamps
128 * `want` to SSIZE_MAX and calls the driver copy_range. 3) n>0 advances the
129 * offsets/length; n==0 is a premature EOF (EIO) and fails; EINTR retries; a
130 * recoverable errno returns NGX_AGAIN with the remaining range; any other errno
131 * is logged and fails.
132 */
133static ngx_int_t
134brix_copy_range_cfr(ngx_log_t *log, int src_fd, off_t *src_off,
135 int dst_fd, off_t *dst_off, size_t *len,
136 const char *src_path, const char *dst_path)
137{
138 brix_sd_obj_t src_obj, dst_obj;
139
140 /* Route the zero-copy primitive through the Storage Driver seam; dispatch via
141 * the object's own driver vtable (backend-agnostic) rather than the hardcoded
142 * POSIX driver. The pread/pwrite fallback composes the VFS primitives. */
143 brix_sd_posix_wrap(&src_obj, src_fd);
144 brix_sd_posix_wrap(&dst_obj, dst_fd);
145
146 while (*len > 0) {
147 size_t want = (*len > (size_t) SSIZE_MAX) ? (size_t) SSIZE_MAX : *len;
148 ssize_t n = src_obj.driver->copy_range(&src_obj, *src_off,
149 &dst_obj, *dst_off, want);
150 if (n > 0) {
151 *src_off += (off_t) n;
152 *dst_off += (off_t) n;
153 *len -= (size_t) n;
154 continue;
155 }
156
157 if (n == 0) {
158 errno = EIO;
159 ngx_log_error(NGX_LOG_ERR, log, 0,
160 "brix: copy_file_range unexpected EOF %s -> %s",
161 src_path ? src_path : "-",
162 dst_path ? dst_path : "-");
163 return NGX_ERROR;
164 }
165
166 if (errno == EINTR) {
167 continue;
168 }
169
170 if (brix_cfr_errno_recoverable(errno)) {
171 /* copy_file_range unsupported for this fd pair / range — signal the
172 * caller to finish the remaining bytes with the portable path. */
173 return NGX_AGAIN;
174 }
175
176 ngx_log_error(NGX_LOG_ERR, log, errno,
177 "brix: copy_file_range failed %s -> %s",
178 src_path ? src_path : "-",
179 dst_path ? dst_path : "-");
180 return NGX_ERROR;
181 }
182 return NGX_OK;
183}
