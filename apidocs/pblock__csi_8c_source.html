1/*
2 * pblock_csi.c — F3 per-block CRC32c integrity for the pblock driver.
3 *
4 * WHAT: The read-verify / close-flush engine behind `csi=1`. One CRC-32c per
5 * pblock block (the block granule IS the CSI granule, so "fully covered
6 * block" maps exactly onto one block file), stored in the
7 * `csi(blob_id, block_no, crc)` catalog table keyed by the rename-stable
8 * blob id.
9 *
10 * WHY: pblock owns its at-rest integrity in its own store rather than the
11 * posix xmeta tagstore (which carries tags at a real POSIX path that does
12 * not exist for a sharded-blob backend). See pblock_csi.h.
13 *
14 * HOW: Verify is a pure function over the open-time snapshot (no DB, no locks
15 * on the hot path). Flush recomputes the written block extent from disk
16 * and UPSERTs in one transaction at close. Reuses the shared crc32c
17 * helper (INVARIANT 1/9 — never reimplement the checksum). 0 is the
18 * "unset" sentinel (a genuine CRC of 0 is skipped — a ~1/2^32 benign
19 * miss, the same convention the xmeta tagstore uses).
20 */

51int
52pblock_csi_load(pblock_catalog *cat, const char *blob_id, int64_t nblocks,
53 uint32_t **out, uint64_t *out_n)
54{
55 sqlite3_stmt *st;
56 uint32_t *arr;
57 uint64_t cap = nblocks > 0 ? (uint64_t) nblocks : 1;
58
59 *out = NULL;
60 *out_n = 0;
61 arr = calloc(cap, sizeof(*arr));
62 if (arr == NULL) {
63 return -1;
64 }
65 st = cat_prepare(cat, "SELECT block_no, crc FROM csi WHERE blob_id = ?1;");
66 if (st != NULL) {
67 sqlite3_bind_text(st, 1, blob_id, -1, SQLITE_STATIC);
68 while (sqlite3_step(st) == SQLITE_ROW) {
69 int64_t bno = sqlite3_column_int64(st, 0);
70
71 if (bno >= 0 && (uint64_t) bno < cap) {
72 arr[bno] = (uint32_t) sqlite3_column_int64(st, 1);
73 }
74 }
75 sqlite3_finalize(st);
76 }
77 *out = arr;
78 *out_n = cap;
79 return 0;
80}

82int
83pblock_csi_verify(const uint32_t *crc, uint64_t n, int64_t bs, int64_t size,
84 const unsigned char *buf, off_t off, size_t len, int64_t dlo, int64_t dhi)
85{
86 int64_t b, first, last;
87 off_t end = off + (off_t) len;
88
89 if (crc == NULL || n == 0 || bs <= 0 || len == 0) {
90 return 0;
91 }
92 first = off / bs;
93 last = (end - 1) / bs;
94 for (b = first; b <= last; b++) {
95 int64_t bstart = b * bs;
96 int64_t bend = bstart + bs;
97 uint32_t got;
98
99 if (bend > size) {
100 bend = size; /* short last block */
101 }
102 if (bend <= bstart) {
103 continue; /* block at/after EOF */
104 }
105 if (bstart < off || (off_t) bend > end) {
106 continue; /* not fully covered by this buffer */
107 }
108 if ((uint64_t) b >= n || crc[b] == 0) {
109 continue; /* unrecorded / unset slot */
110 }
111 if (b >= dlo && b < dhi) {
112 continue; /* written this handle: snapshot stale */
113 }
114 got = brix_crc32c_value(buf + (bstart - off), (size_t) (bend - bstart));
115 if (got != crc[b]) {
116 errno = EIO;
117 return -1;
118 }
119 }
120 return 0;
121}

123int
124pblock_csi_flush(const pblock_state_t *st, const char *blob_id, int64_t size,
125 int64_t bs, int64_t dlo, int64_t dhi)
126{
127 unsigned char *buf;
128 int64_t b, last;
129 int rc = 0;
130
131 if (bs <= 0 || dlo >= dhi) {
132 return 0; /* nothing written through this handle */
133 }
134 last = size > 0 ? pblock_last_block(size, bs) : -1;
135 if (dhi > last + 1) {
136 dhi = last + 1; /* clamp to EOF (a truncate may shrink) */
137 }
138 if (dlo < 0) {
139 dlo = 0;
140 }
141 if (dlo >= dhi) {
142 return 0;
143 }
144 buf = malloc((size_t) bs);
145 if (buf == NULL) {
146 return -1;
147 }
148 (void) cat_exec(st->cat, "BEGIN;");
149 for (b = dlo; b < dhi; b++) {
150 int64_t bstart = b * bs;
151 int64_t blen = bs;
152 ssize_t r;
153 sqlite3_stmt *ins;
154
155 if (bstart >= size) {
156 continue;
157 }
158 if (bstart + blen > size) {
159 blen = size - bstart; /* short last block */
160 }
161 r = pblock_read_blocks(st, blob_id, bs, -1, buf, (size_t) blen, bstart);
162 if (r != (ssize_t) blen) {
163 rc = -1;
164 continue;
165 }
166 ins = cat_prepare(st->cat,
167 "INSERT OR REPLACE INTO csi(blob_id, block_no, crc) "
168 "VALUES(?1, ?2, ?3);");
169 if (ins != NULL) {
170 sqlite3_bind_text(ins, 1, blob_id, -1, SQLITE_STATIC);
171 sqlite3_bind_int64(ins, 2, b);
172 sqlite3_bind_int64(ins, 3,
173 (sqlite3_int64) brix_crc32c_value(buf, (size_t) blen));
174 (void) sqlite3_step(ins);
175 sqlite3_finalize(ins);
176 } else {
177 rc = -1;
178 }
179 }
180 (void) cat_exec(st->cat, "COMMIT;");
181 free(buf);
182 return rc;
183}
