41/* Forwards to the shared Blowfish-CFB64 kernel (libxrdproto), which owns the
42 * OpenSSL-3 legacy-provider load — one source of truth with the server's SSS path. */
43int
44brix_sss_bf32_encrypt(const uint8_t *key, size_t key_len,
45 const uint8_t *src, size_t src_len,
46 uint8_t *dst, size_t dst_max, size_t *out_len,
47 brix_status *st)
48{
49 if (key_len == 0 || src_len == 0 || dst_max < src_len) {

55 brix_status_set(st, XRDC_EPROTO, 0,

58 }

64void
65brix_sss_keytab_default(char *out, size_t outsz)
66{
67 /*
68 * WHAT: resolve the SSS keytab path from the canonical alias chain.
69 * WHY: XrdSecSSSKT vs XrdSecsssKT differ only by case; the resolver
70 * emits a TTY note if both are set with different values (WS-1).
71 * HOW: brix_env_resolve walks the chain in precedence order; legacy
72 * names remain accepted forever (C2 compat).
73 */
74 static const char *const keytab_chain[] = {
75 "XRDC_SSS_KEYTAB", "XrdSecSSSKT", "XrdSecsssKT", NULL
76 };
77 const char *env = brix_env_resolve(keytab_chain, NULL);
78 const char *home;
79
80 if (env != NULL && env[0] != '\0') {
81 snprintf(out, outsz, "%s", env);
82 return;
83 }
84 home = getenv("HOME");
85 if (home == NULL || home[0] == '\0') {
86 struct passwd *pw = getpwuid(geteuid());
87 home = (pw != NULL) ? pw->pw_dir : "/tmp";
88 }
89 snprintf(out, outsz, "%s/.xrd/sss.keytab", home);
90}

92/* The neutral kernel entry must hold any field this struct can — else the copy
93 * below could truncate a valid key/name (a drift in either would be a real bug). */
94_Static_assert(sizeof(((brix_sss_key *) 0)->key) == SSS_K_KEY_MAX,
95 "SSS key buffer size drift vs shared kernel");
96_Static_assert(sizeof(((brix_sss_key *) 0)->user) == SSS_K_USER_MAX,
97 "SSS user buffer size drift vs shared kernel");
98_Static_assert(sizeof(((brix_sss_key *) 0)->group) == SSS_K_GROUP_MAX,
99 "SSS group buffer size drift vs shared kernel");
100_Static_assert(sizeof(((brix_sss_key *) 0)->name) == SSS_K_NAME_MAX,
101 "SSS name buffer size drift vs shared kernel");

103/* Parse one keytab line (mutated by strtok_r) via the shared grammar kernel.
104 * Returns 1 if a key was filled, 0 for blank/comment/expired, -1 on a malformed
105 * required field. */
106static int
107parse_line(char *line, brix_sss_key *k)
108{
109 sss_keytab_entry_t entry;
110 int rc;
111
112 rc = sss_keytab_parse_line(line, &entry, (int64_t) time(NULL));
113 if (rc <= 0) {
114 return rc; /* 0 = skip, -1 = malformed */
115 }
116
117 memset(k, 0, sizeof(*k));
118 k->id = entry.id;
119 k->exp = entry.exp;
120 k->key_len = entry.key_len;
121 memcpy(k->key, entry.key, entry.key_len);
122 snprintf(k->user, sizeof(k->user), "%s", entry.user);
123 snprintf(k->group, sizeof(k->group), "%s", entry.group);
124 snprintf(k->name, sizeof(k->name), "%s", entry.name);
125 return 1;
126}

128int
129brix_sss_keytab_read(const char *path, brix_sss_key *keys, int max, int *n,
130 brix_status *st)
131{
132 struct stat sb;
133 int fd;
134 FILE *fp;
135 char line[1024];
136 int count = 0, lineno = 0;
137
138 *n = 0;
139 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC);
140 if (fd < 0) {
141 brix_status_set(st, XRDC_EAUTH, errno, "open keytab %s: %s",
142 path, strerror(errno));
143 return -1;
144 }
145 if (fstat(fd, &sb) != 0 || !S_ISREG(sb.st_mode)) {
146 close(fd);
147 brix_status_set(st, XRDC_EAUTH, 0, "keytab %s is not a regular file", path);
148 return -1;
149 }
150 if (sss_keytab_mode_ok(path, sb.st_mode, 0) != 0) {
151 close(fd);
152 brix_status_set(st, XRDC_EAUTH, 0,
153 "keytab %s must be mode 0600 (group/other bits set)", path);
154 return -1;
155 }
156 fp = fdopen(fd, "r");
157 if (fp == NULL) {
158 close(fd);
159 brix_status_set(st, XRDC_EAUTH, errno, "fdopen keytab: %s", strerror(errno));
160 return -1;
161 }
162
163 while (fgets(line, sizeof(line), fp) != NULL && count < max) {
164 int rc;
165 lineno++;
166 rc = parse_line(line, &keys[count]);
167 if (rc < 0) {
168 fclose(fp);
169 brix_status_set(st, XRDC_EAUTH, 0,
170 "keytab %s: malformed key on line %d", path, lineno);
171 return -1;
172 }
173 if (rc == 1) {
174 count++;
175 }
176 }
177 fclose(fp);
178 *n = count;
179 return 0;
180}

182int
183brix_sss_keytab_write(const char *path, const brix_sss_key *keys, int n,
184 brix_status *st)
185{
186 int fd, i;
187 FILE *fp;
188 char hex[XRDC_SSS_KEY_MAX * 2 + 1];
189
190 fd = open(path, O_WRONLY | O_CREAT | O_TRUNC | O_NOFOLLOW | O_CLOEXEC, 0600);
191 if (fd < 0) {
192 brix_status_set(st, XRDC_EAUTH, errno, "create keytab %s: %s",
193 path, strerror(errno));
194 return -1;
195 }
196 /* Enforce 0600 even if the file pre-existed with looser bits. */
197 if (fchmod(fd, 0600) != 0) {
198 close(fd);
199 brix_status_set(st, XRDC_EAUTH, errno, "chmod keytab: %s", strerror(errno));
200 return -1;
201 }
202 fp = fdopen(fd, "w");
203 if (fp == NULL) {
204 close(fd);
205 brix_status_set(st, XRDC_EAUTH, errno, "fdopen keytab: %s", strerror(errno));
206 return -1;
207 }
208 for (i = 0; i < n; i++) {
209 brix_hex_encode(keys[i].key, keys[i].key_len, hex);
210 fprintf(fp, "0 u:%s g:%s N:%lld k:%s n:%s",
211 keys[i].user, keys[i].group, (long long) keys[i].id, hex,
212 keys[i].name);
213 if (keys[i].exp != 0) {
214 fprintf(fp, " e:%lld", (long long) keys[i].exp);
215 }
216 fputc('\n', fp);
217 }
218 if (fclose(fp) != 0) {
219 brix_status_set(st, XRDC_EAUTH, errno, "write keytab: %s", strerror(errno));
220 return -1;
221 }
222 return 0;
223}
