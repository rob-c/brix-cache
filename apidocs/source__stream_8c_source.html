21/* File: source_stream.c — TPC remote source pull, Phase 2/3 (kXR_read stream
22 * loop + fsync) and the best-effort remote close, split from source.c in the

24 * WHAT: tpc_stream_to_dst() streams the whole origin into t->dst_fd one
25 * TPC_CHUNK_SIZE kXR_read window at a time (draining the kXR_oksofar/kXR_ok

65/*
66 * tpc_stream_classify_frame — WHAT: decide what a received read-reply frame
67 * means. WHY: keeps the drain loop flat by lifting the kXR_error / invalid-status
68 * classification out of it. HOW: kXR_error → decode the remote code/message;
69 * anything other than kXR_ok / kXR_oksofar is a protocol violation. Returns 0 if
70 * the frame is a valid data frame (caller writes its body), -1 on failure with
71 * t->err_msg / t->xrd_error set. `body` is NOT freed here (the caller owns it).
72 */
73static int
74tpc_stream_classify_frame(brix_tpc_pull_t *t, uint16_t status,
75 const u_char *body, uint32_t dlen, uint64_t offset)
76{
77 if (status == kXR_error) {
78 if (body != NULL && dlen >= 4) {
79 snprintf(t->err_msg, sizeof(t->err_msg),
80 "TPC source read error: %s", (const char *) body + 4);
81 t->xrd_error = (int) ntohl(*(const uint32_t *) body);
82 } else {
83 snprintf(t->err_msg, sizeof(t->err_msg),
84 "TPC kXR_read error at offset %llu",
85 (unsigned long long) offset);
86 t->xrd_error = kXR_IOError;
87 }
88 return -1;
89 }
90
91 if (status != kXR_ok && status != kXR_oksofar) {
92 snprintf(t->err_msg, sizeof(t->err_msg),
93 "TPC kXR_read returned invalid status %u at offset %llu",
94 (unsigned) status, (unsigned long long) offset);
95 t->xrd_error = kXR_ServerError;
96 return -1;
97 }
98
99 return 0;
100}

146/*
147 * tpc_stream_drain_request — WHAT: receive and write every frame answering a
148 * single kXR_read, reporting the byte count delivered in *got_this_req.
149 * WHY: a single kXR_read may be answered by zero or more kXR_oksofar frames
150 * (more data still coming for THIS request) terminated by exactly one kXR_ok
151 * frame; draining that sequence is one nameable step.
152 * HOW: loop recv → classify → write each valid data frame → stop on the terminal
153 * kXR_ok. Returns 0 with *got_this_req set (0 means EOF for this window), or -1
154 * with t->err_msg / t->xrd_error set.
155 */
156static int
157tpc_stream_drain_request(brix_tpc_pull_t *t, int fd, uint64_t offset,
158 size_t *got_this_req)
159{
160 *got_this_req = 0;
161
162 for (;;) {
163 uint16_t status;
164 uint32_t dlen;
165 u_char *body = NULL;
166 int rc;
167
168 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
169 snprintf(t->err_msg, sizeof(t->err_msg),
170 "TPC kXR_read recv failed at offset %llu",
171 (unsigned long long) offset);
172 return -1;
173 }
174
175 rc = tpc_stream_classify_frame(t, status, body, dlen, offset);
176 if (rc != 0) {
177 free(body);
178 return -1;
179 }

194 }

219/*
220 * tpc_stream_to_dst — Phase 2/3: stream the whole source into t->dst_fd one
221 * kXR_read window at a time, then fsync for durability. Returns 0 (with
222 * t->result=NGX_OK, t->xrd_error=0) once the file is fully written and synced,
223 * or -1 with t->err_msg / t->xrd_error set. The caller still issues the
224 * best-effort remote close on either outcome.
225 *
226 * Outer loop: one kXR_read request per TPC_CHUNK_SIZE window, advancing `offset`
227 * by the bytes actually delivered. We never pipeline reads — each request is
228 * fully drained before the next is issued, so `offset` math stays simple and a
229 * short final read cleanly signals EOF. A request that delivers zero bytes means
230 * we read past EOF and stop.
231 *
232 * Phase 39 (WS4): wall-clock cap on the whole pull, sampled once per 1 MiB chunk
233 * (NOT per frame) so it adds no per-frame syscall cost. Bounds a slow-drip
234 * remote that keeps resetting the per-recv SO_RCVTIMEO idle timer. 0 = no cap
235 * (current behaviour). The per-recv idle timeout still applies.
236 */
237int
238tpc_stream_to_dst(brix_tpc_pull_t *t, int fd, const u_char *fhandle)
239{
240 uint64_t offset = 0;
241 time_t pull_start = time(NULL);
242 time_t pull_max = (t->conf != NULL)
243 ? (time_t) t->conf->tpc_max_transfer_secs : 0;
244
245 for (;;) {
246 size_t got_this_req = 0;
247
248 if (pull_max > 0 && (time(NULL) - pull_start) > pull_max) {
249 snprintf(t->err_msg, sizeof(t->err_msg),
250 "TPC pull exceeded brix_tpc_max_transfer_secs (%lds) "
251 "at offset %llu", (long) pull_max,
252 (unsigned long long) offset);
253 t->xrd_error = kXR_IOError;
254 return -1;
255 }
256
257 if (tpc_stream_send_read(t, fd, fhandle, offset) != 0) {
258 return -1;
259 }
260 if (tpc_stream_drain_request(t, fd, offset, &got_this_req) != 0) {
261 /* Ensure an error code is always set (some paths set only the
262 * message). */
263 if (t->xrd_error == 0) {
264 t->xrd_error = kXR_IOError;
265 }
266 return -1;
267 }
268
269 if (got_this_req == 0) {
270 break; /* EOF */
271 }
272 offset += got_this_req;
273 }

275 /*
276 * Completion gate (hostile-network truncation fix): the loop's only in-band
277 * EOF signal is a zero-byte read reply, which a truncating middlebox or a
278 * source that dies mid-stream can produce (or forge) — so "the reads stopped"
279 * is NOT proof the whole file arrived. Verify the delivered byte count against
280 * the source's authoritative size (captured by tpc_stat_source before the
281 * loop). A mismatch is unambiguous truncation and ALWAYS fails the pull; the
282 * caller unlinks the partial destination so it is never committed as complete.
283 * When the source would not declare a size, brix_tpc_require_source_size
284 * decides whether an unverifiable pull is refused or proceeds (default).
285 */
286 if (t->src_size_known) {
287 if ((uint64_t) t->bytes_written != t->src_size) {
288 snprintf(t->err_msg, sizeof(t->err_msg),
289 "TPC pull truncated: wrote %llu of %llu source bytes",
290 (unsigned long long) t->bytes_written,
291 (unsigned long long) t->src_size);
292 t->xrd_error = kXR_IOError;
293 return -1;
294 }
295 } else if (t->conf != NULL && t->conf->tpc_require_source_size) {
296 snprintf(t->err_msg, sizeof(t->err_msg),
297 "TPC pull: source declared no size and "
298 "brix_tpc_require_source_size is on");
299 t->xrd_error = kXR_ServerError;
300 return -1;
301 }
302
303 if (tpc_stream_sync_dst(t) != 0) {
304 return -1;
305 }
306
307 t->result = NGX_OK;
308 t->xrd_error = 0;
309 return 0;
310}

313/*
314 * tpc_stat_parse_size — WHAT: read the size (2nd whitespace token) out of an
315 * XRootD kXR_stat reply body "id size flags mtime". WHY: the size is the pull's
316 * authoritative completion target; the surrounding tokens are irrelevant here.
317 * HOW: skip token 0 (id), then strtoull the leading digits of token 1. Returns 0
318 * with *size set, or -1 if there is no numeric second token.
319 */
320static int
321tpc_stat_parse_size(const char *body, uint64_t *size)
322{
323 const char *p = body;
324 char *end;
325 unsigned long long v;
326
327 while (*p == ' ') {
328 p++;
329 }
330 while (*p != '\0' && *p != ' ') { /* skip token 0 (id) */
331 p++;
332 }
333 while (*p == ' ') {
334 p++;
335 }
336 if (*p < '0' || *p > '9') {
337 return -1;
338 }
339 errno = 0;
340 v = strtoull(p, &end, 10);
341 if (end == p || errno != 0) {
342 return -1;
343 }
344 *size = (uint64_t) v;
345 return 0;
346}

403/*
404 * tpc_verify_source_checksum — opt-in post-copy integrity for the TPC pull.
405 * kXR_query(kXR_Qcksum) the source (distinct streamid tag 5), parse the
406 * "<alg> <hex>" reply, recompute the same algorithm over the written destination
407 * with brix_checksum_hex_name_fd, and fail closed (kXR_ChkSumErr) on any of:
408 * source cannot supply a checksum, malformed reply, an algorithm brix cannot
409 * compute, a destination read failure, or a digest mismatch. See
410 * source_internal.h.
411 */

413tpc_verify_source_checksum(brix_tpc_pull_t *t, int fd)

441 qreq.dlen = htonl((kXR_int32) pathlen);
442 ngx_memcpy(req, &qreq, sizeof(qreq));
443 ngx_memcpy(req + sizeof(qreq), t->src_path, pathlen);
444
445 if (tpc_send_all(t, fd, req, total) != 0) {
446 snprintf(t->err_msg, sizeof(t->err_msg),
447 "TPC checksum query send failed");
448 t->xrd_error = kXR_IOError;
449 return -1;
450 }
451 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
452 snprintf(t->err_msg, sizeof(t->err_msg),
453 "TPC checksum query recv failed");
454 t->xrd_error = kXR_IOError;
455 return -1;
456 }
457
458 /* Fail closed: the operator required verification and the source gave us no
459 * checksum to verify against. Reply body is "<alg> <hex>". */

461 free(body);
462 snprintf(t->err_msg, sizeof(t->err_msg),
463 "TPC checksum verify: source supplied no checksum");
464 t->xrd_error = kXR_ChkSumErr;
465 return -1;
466 }
467
468 sp = strchr((const char *) body, ' ');
469 if (sp == NULL) {
470 free(body);
471 snprintf(t->err_msg, sizeof(t->err_msg),
472 "TPC checksum verify: malformed source checksum reply");
473 t->xrd_error = kXR_ChkSumErr;
474 return -1;
475 }

477 if (alglen == 0 || alglen >= sizeof(alg)) {
478 free(body);
479 snprintf(t->err_msg, sizeof(t->err_msg),
480 "TPC checksum verify: bad source checksum type");
481 t->xrd_error = kXR_ChkSumErr;
482 return -1;
483 }
484 ngx_memcpy(alg, body, alglen);
485 alg[alglen] = '\0';
486
487 /* value = token after the space, trimmed of trailing whitespace/CRLF. */
488 sp++;
489 while (*sp == ' ') {
490 sp++;
491 }
492 vlen = 0;
493 while (sp[vlen] != '\0' && sp[vlen] != ' '
494 && sp[vlen] != '\n' && sp[vlen] != '\r')
495 {
496 vlen++;
497 }
498 if (vlen == 0 || vlen >= sizeof(src_hex)) {
499 free(body);
500 snprintf(t->err_msg, sizeof(t->err_msg),
501 "TPC checksum verify: bad source checksum value");
502 t->xrd_error = kXR_ChkSumErr;
503 return -1;
504 }
505 ngx_memcpy(src_hex, sp, vlen);
506 src_hex[vlen] = '\0';
507 free(body);
508
509 /* Recompute the SAME algorithm over the destination and compare. */
510 if (brix_checksum_hex_name_fd(alg, t->dst_fd, t->dst_path, log,
511 local_hex, sizeof(local_hex),
512 normalized, sizeof(normalized)) != NGX_OK)
513 {
514 snprintf(t->err_msg, sizeof(t->err_msg),
515 "TPC checksum verify: cannot compute %s on destination", alg);
516 t->xrd_error = kXR_ChkSumErr;
517 return -1;
518 }
519
520 if (strcasecmp(local_hex, src_hex) != 0) {
521 snprintf(t->err_msg, sizeof(t->err_msg),
522 "TPC checksum mismatch: source %s=%s destination=%s",
523 alg, src_hex, local_hex);
524 t->xrd_error = kXR_ChkSumErr;
525 return -1;
526 }
527
528 return 0;
529}
