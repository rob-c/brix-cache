19/* File: source_stream.c — TPC remote source pull, Phase 2/3 (kXR_read stream
20 * loop + fsync) and the best-effort remote close, split from source.c in the
21 * phase-79 file-size burndown.
22 * WHAT: tpc_stream_to_dst() streams the whole origin into t->dst_fd one
23 * TPC_CHUNK_SIZE kXR_read window at a time (draining the kXR_oksofar/kXR_ok
24 * frame sequence per request, writing each frame's body through the VFS core),
25 * then fsyncs for durability; tpc_close_source() issues the best-effort
26 * kXR_close so the origin handle is never leaked. WHY: keeps the byte-streaming
27 * concern separate from the open/resolve (source_open.c) and the driver
28 * (source.c). HOW: send read → drain → advance offset → repeat to EOF → sync.
29 * Every framing helper is file-static; only tpc_stream_to_dst() and
30 * tpc_close_source() cross the file boundary. */

33/*
34 * tpc_stream_send_read — WHAT: build and send one kXR_read request for the
35 * TPC_CHUNK_SIZE window at `offset`. WHY: isolates the read-request framing from
36 * the drain/advance loop. HOW: kXR_read header with an 8-byte big-endian offset
37 * (htobe64, NOT htonl) and a 4-byte requested length; streamid[1]=3 tags read
38 * replies on this socket distinctly from the open/close tag (2). Returns 0 on a
39 * successful send, -1 with t->err_msg set on failure.
40 */
41static int
42tpc_stream_send_read(brix_tpc_pull_t *t, int fd, const u_char *fhandle,
43 uint64_t offset)
44{
45 ClientReadRequest rdreq;
46
47 ngx_memzero(&rdreq, sizeof(rdreq));
48 rdreq.streamid[1] = 3;
49 rdreq.requestid = htons(kXR_read);
50 ngx_memcpy(rdreq.fhandle, fhandle, XRD_FHANDLE_LEN);
51 rdreq.offset = (kXR_int64) htobe64(offset);
52 rdreq.rlen = htonl((kXR_int32) TPC_CHUNK_SIZE);
53
54 if (tpc_send_all(t, fd, &rdreq, sizeof(rdreq)) != 0) {
55 snprintf(t->err_msg, sizeof(t->err_msg),
56 "TPC kXR_read send failed at offset %llu",
57 (unsigned long long) offset);
58 return -1;
59 }
60 return 0;
61}

63/*
64 * tpc_stream_classify_frame — WHAT: decide what a received read-reply frame
65 * means. WHY: keeps the drain loop flat by lifting the kXR_error / invalid-status
66 * classification out of it. HOW: kXR_error → decode the remote code/message;
67 * anything other than kXR_ok / kXR_oksofar is a protocol violation. Returns 0 if
68 * the frame is a valid data frame (caller writes its body), -1 on failure with
69 * t->err_msg / t->xrd_error set. `body` is NOT freed here (the caller owns it).
70 */
71static int
72tpc_stream_classify_frame(brix_tpc_pull_t *t, uint16_t status,
73 const u_char *body, uint32_t dlen, uint64_t offset)
74{
75 if (status == kXR_error) {
76 if (body != NULL && dlen >= 4) {
77 snprintf(t->err_msg, sizeof(t->err_msg),
78 "TPC source read error: %s", (const char *) body + 4);
79 t->xrd_error = (int) ntohl(*(const uint32_t *) body);
80 } else {
81 snprintf(t->err_msg, sizeof(t->err_msg),
82 "TPC kXR_read error at offset %llu",
83 (unsigned long long) offset);
84 t->xrd_error = kXR_IOError;
85 }
86 return -1;
87 }
88
89 if (status != kXR_ok && status != kXR_oksofar) {
90 snprintf(t->err_msg, sizeof(t->err_msg),
91 "TPC kXR_read returned invalid status %u at offset %llu",
92 (unsigned) status, (unsigned long long) offset);
93 t->xrd_error = kXR_ServerError;
94 return -1;
95 }
96
97 return 0;
98}

100/*
101 * tpc_stream_write_frame — WHAT: write one data frame's body to t->dst_fd at
102 * (offset + *got_this_req) through the VFS core and advance the counters.
103 * WHY: isolates the positional-write + progress side effect from frame receipt.
104 * HOW: overflow-guard the destination offset, run a positional VFS WRITE job,
105 * verify the full length was written, then bump *got_this_req / t->bytes_written
106 * and emit a progress sample. Returns 0 on success, -1 with t->err_msg /
107 * t->xrd_error set on overflow or a short/failed write.
108 */
109static int
110tpc_stream_write_frame(brix_tpc_pull_t *t, uint64_t offset,
111 size_t *got_this_req, u_char *body, uint32_t dlen)
112{
113 brix_vfs_job_t job;
114 off_t dst_offset;
115
116 if (offset > (uint64_t) LLONG_MAX - (uint64_t) *got_this_req) {
117 snprintf(t->err_msg, sizeof(t->err_msg),
118 "TPC dst write offset too large");
119 t->xrd_error = kXR_IOError;
120 return -1;
121 }
122
123 dst_offset = (off_t) (offset + (uint64_t) *got_this_req);
124 brix_vfs_job_write_init(&job, t->dst_fd, dst_offset, body, (size_t) dlen);
125 brix_vfs_io_execute(&job);
126 if (job.io_errno != 0 || job.nio < 0 || (uint32_t) job.nio != dlen) {
127 int err = job.io_errno != 0 ? job.io_errno : EIO;
128
129 snprintf(t->err_msg, sizeof(t->err_msg),
130 "TPC dst write failed: %s", strerror(err));
131 t->xrd_error = kXR_IOError;
132 return -1;
133 }
134
135 *got_this_req += (size_t) job.nio;
136 t->bytes_written += (size_t) job.nio;
137 (void) brix_tpc_progress_emit(
138 t->transfer_id, (off_t) t->bytes_written, 0,
139 BRIX_TPC_STATE_ACTIVE,
140 t->c != NULL ? t->c->log : NULL);
141 return 0;
142}

144/*
145 * tpc_stream_drain_request — WHAT: receive and write every frame answering a
146 * single kXR_read, reporting the byte count delivered in *got_this_req.
147 * WHY: a single kXR_read may be answered by zero or more kXR_oksofar frames
148 * (more data still coming for THIS request) terminated by exactly one kXR_ok
149 * frame; draining that sequence is one nameable step.
150 * HOW: loop recv → classify → write each valid data frame → stop on the terminal
151 * kXR_ok. Returns 0 with *got_this_req set (0 means EOF for this window), or -1
152 * with t->err_msg / t->xrd_error set.
153 */
154static int
155tpc_stream_drain_request(brix_tpc_pull_t *t, int fd, uint64_t offset,
156 size_t *got_this_req)
157{
158 *got_this_req = 0;
159
160 for (;;) {
161 uint16_t status;
162 uint32_t dlen;
163 u_char *body = NULL;
164 int rc;
165
166 if (tpc_recv_response(t, fd, &status, &body, &dlen) != 0) {
167 snprintf(t->err_msg, sizeof(t->err_msg),
168 "TPC kXR_read recv failed at offset %llu",
169 (unsigned long long) offset);
170 return -1;
171 }
172
173 rc = tpc_stream_classify_frame(t, status, body, dlen, offset);
174 if (rc != 0) {
175 free(body);
176 return -1;
177 }
178
179 if (dlen > 0 && body != NULL
180 && tpc_stream_write_frame(t, offset, got_this_req, body, dlen) != 0)
181 {
182 free(body);
183 return -1;
184 }
185
186 free(body);
187
188 if (status == kXR_ok) {
189 return 0;
190 }
191 /* kXR_oksofar: loop to receive next frame */
192 }
193}

217/*
218 * tpc_stream_to_dst — Phase 2/3: stream the whole source into t->dst_fd one
219 * kXR_read window at a time, then fsync for durability. Returns 0 (with
220 * t->result=NGX_OK, t->xrd_error=0) once the file is fully written and synced,
221 * or -1 with t->err_msg / t->xrd_error set. The caller still issues the
222 * best-effort remote close on either outcome.
223 *
224 * Outer loop: one kXR_read request per TPC_CHUNK_SIZE window, advancing `offset`
225 * by the bytes actually delivered. We never pipeline reads — each request is
226 * fully drained before the next is issued, so `offset` math stays simple and a
227 * short final read cleanly signals EOF. A request that delivers zero bytes means
228 * we read past EOF and stop.
229 *
230 * Phase 39 (WS4): wall-clock cap on the whole pull, sampled once per 1 MiB chunk
231 * (NOT per frame) so it adds no per-frame syscall cost. Bounds a slow-drip
232 * remote that keeps resetting the per-recv SO_RCVTIMEO idle timer. 0 = no cap
233 * (current behaviour). The per-recv idle timeout still applies.
234 */
235int
236tpc_stream_to_dst(brix_tpc_pull_t *t, int fd, const u_char *fhandle)
237{
238 uint64_t offset = 0;
239 time_t pull_start = time(NULL);
240 time_t pull_max = (t->conf != NULL)
241 ? (time_t) t->conf->tpc_max_transfer_secs : 0;
242
243 for (;;) {
244 size_t got_this_req = 0;
245
246 if (pull_max > 0 && (time(NULL) - pull_start) > pull_max) {
247 snprintf(t->err_msg, sizeof(t->err_msg),
248 "TPC pull exceeded brix_tpc_max_transfer_secs (%lds) "
249 "at offset %llu", (long) pull_max,
250 (unsigned long long) offset);
251 t->xrd_error = kXR_IOError;
252 return -1;
253 }
254
255 if (tpc_stream_send_read(t, fd, fhandle, offset) != 0) {
256 return -1;
257 }
258 if (tpc_stream_drain_request(t, fd, offset, &got_this_req) != 0) {
259 /* Ensure an error code is always set (some paths set only the
260 * message). */
261 if (t->xrd_error == 0) {
262 t->xrd_error = kXR_IOError;
263 }
264 return -1;
265 }
266
267 if (got_this_req == 0) {
268 break; /* EOF */
269 }
270 offset += got_this_req;
271 }
272
273 if (tpc_stream_sync_dst(t) != 0) {
274 return -1;
275 }
276
277 t->result = NGX_OK;
278 t->xrd_error = 0;
279 return 0;
280}
