87/* Map a FUSE path ("/file") to the server path under the export base. With an
88 * empty base the FUSE path is used verbatim; with base "/data" → "/data/file"
89 * and "/" → "/data". Shared by BOTH transports (root:// and http/WebDAV) so a
90 * URL like root://host/data or https://host/data mounts that subtree. */
91const char *
92srv_path(const char *p, char *buf, size_t sz)
93{
94 if (g_base[0] == '\0') {
95 return p;
96 }
97 if (strcmp(p, "/") == 0) {
98 return g_base;
99 }
100 size_t bl = strlen(g_base);
101 size_t pl = strlen(p);
102 if (bl + pl + 1 > sz) { /* impossible for real paths — fail safe */
103 return g_base;
104 }
105 memcpy(buf, g_base, bl);
106 memcpy(buf + bl, p, pl + 1); /* includes the NUL */
107 return buf;
108}

248/* Entry point for the default (async/resilient) driver. Invoked by the unified
249 * xrootdfs front-end (apps/xrootdfs_main.c); see xrootdfs_drivers.h. */
250int
251xrootdfs_aio_main(int argc, char **argv)
252{
253 brix_status st;
254 const char *endpoint = NULL;
255 char *fuse_argv[64];
256 int fuse_argc = 0;
257 int i, rc;
258
259 /* Check --version / --help before the argc < 3 guard so they work standalone. */
260 if (argc >= 2) {
261 if (strcmp(argv[1], "--version") == 0) {
262 printf("xrootdfs (BriX-Cache client) %s\n", brix_client_version());
263 return 0;
264 }
265 if (strcmp(argv[1], "--help") == 0) { usage_fp(stdout); return 0; }
266 if (strcmp(argv[1], "-h") == 0) { usage(); return 0; }
267 }
268
269 if (argc < 3) {
270 usage();
271 return 2;
272 }
273 signal(SIGPIPE, SIG_IGN); /* a dropped peer must never kill the mount */
274 memset(&g_opts, 0, sizeof(g_opts));
275 g_opts.verify_host = 1;
276 brix_crypto_init();
277
278 fuse_argv[fuse_argc++] = argv[0];
279
280 for (i = 1; i < argc; i++) {
281 const char *a = argv[i];
282 /* Parse our known options ANYWHERE on the line (before OR after the
283 * endpoint), so a resilience flag placed after the URL is honored rather
284 * than silently leaking to libfuse. Unknown dash-args fall through to the
285 * fuse passthrough (so -f/-d/-s/-o still work); the first bare word is the
286 * endpoint, the next the mountpoint. */
287 if (a[0] == '-') {
288 if (strcmp(a, "--tls") == 0) { g_opts.want_tls = 1; }
289 else if (strcmp(a, "--notlsok") == 0) { g_opts.notlsok = 1; }
290 else if (strcmp(a, "--noverifyhost") == 0) { g_opts.verify_host = 0; }
291 else if (strcmp(a, "--auth") == 0 && i + 1 < argc) { g_opts.auth_force = argv[++i]; }
292 else if (strcmp(a, "--max-conns") == 0 && i + 1 < argc) {
293 g_max_conns = atoi(argv[++i]);
294 if (g_max_conns < 1) { g_max_conns = 1; }
295 }
296 else if (strcmp(a, "--streams") == 0 && i + 1 < argc) {
297 g_streams = atoi(argv[++i]);
298 if (g_streams < 1) { g_streams = 1; }
299 }
300 else if (strcmp(a, "--lazy-streams") == 0) { g_lazy_streams = 1; }
301 else if (strcmp(a, "--max-stall") == 0 && i + 1 < argc) {
302 g_max_stall = atoi(argv[++i]);
303 if (g_max_stall < 0) { g_max_stall = 0; }
304 }
305 else if (strcmp(a, "--keepalive") == 0 && i + 1 < argc) {
306 g_keepalive = atoi(argv[++i]);
307 if (g_keepalive < 0) { g_keepalive = 0; }
308 }
309 else if (strcmp(a, "--max-retries") == 0 && i + 1 < argc) {
310 g_max_retries = atoi(argv[++i]);
311 if (g_max_retries < 0) { g_max_retries = 0; }
312 }
313 else if (strcmp(a, "--connect-timeout") == 0 && i + 1 < argc) {
314 brix_tmo_set_connect_ms(atoi(argv[++i]));
315 }
316 else if (strcmp(a, "--io-timeout") == 0 && i + 1 < argc) {
317 brix_tmo_set_io_ms(atoi(argv[++i]));
318 }
319 else if (strcmp(a, "--attr-timeout") == 0 && i + 1 < argc) {
320 g_attr_timeout = atof(argv[++i]);
321 }
322 else if (strcmp(a, "--entry-timeout") == 0 && i + 1 < argc) {
323 g_entry_timeout = atof(argv[++i]);
324 }
325 else if (strcmp(a, "--kernel-cache") == 0) { g_kernel_cache = 1; }
326 else if (strcmp(a, "--readahead") == 0 && i + 1 < argc) {
327 long v = atol(argv[++i]);
328 g_readahead = (v > 0) ? (size_t) v : 0;
329 }
330 else if (strcmp(a, "--writeback") == 0 && i + 1 < argc) {
331 long v = atol(argv[++i]);
332 g_writeback = (v > 0) ? (size_t) v : 0;
333 }
334 else if (strcmp(a, "--xattr") == 0) { g_xattr = 1; }
335 else if (strcmp(a, "--compress") == 0 && i + 1 < argc) {
336 snprintf(g_compress, sizeof(g_compress), "%s", argv[++i]);
337 }
338 else if (strcmp(a, "--token") == 0 && i + 1 < argc) { g_bearer = argv[++i]; }
339 else if (strcmp(a, "--version") == 0) {
340 printf("xrootdfs (BriX-Cache client) %s\n", brix_client_version());
341 return 0;
342 }
343 else if (strcmp(a, "--help") == 0) { usage_fp(stdout); return 0; } /* WS-2 */
344 else if (strcmp(a, "-h") == 0) { usage(); return 0; } /* C1 */
345 else if (fuse_argc < 61) { fuse_argv[fuse_argc++] = argv[i]; } /* fuse opt */
346 } else if (endpoint == NULL) {
347 endpoint = a;
348 } else if (fuse_argc < 61) {
349 fuse_argv[fuse_argc++] = argv[i];
350 }
351 }
352
353 if (endpoint == NULL || fuse_argc < 2) {
354 usage();
355 return 2;
356 }
357 fuse_argv[fuse_argc] = NULL;
358
359 brix_status_clear(&st);
360
361 /* HTTP(S)/WebDAV read-only mount when the endpoint is a web URL. */
362 if (brix_is_web_url(endpoint)) {
363 brix_statinfo si;
364 size_t bl;
365 if (brix_weburl_parse(endpoint, &g_weburl) != 0) {
366 fprintf(stderr, "xrootdfs: bad web URL: %s\n", endpoint);
367 return 2;
368 }
369 if (g_weburl.is_s3) {
370 fprintf(stderr, "xrootdfs: s3:// is not supported as a FUSE mount "
371 "(use http/https/dav/davs)\n");
372 return 2;
373 }
374 g_web = 1;
375 if (g_bearer == NULL) {
376 g_bearer = getenv("BEARER_TOKEN");
377 }
378 g_web_verify = g_opts.verify_host;
379 g_web_ca = brix_resolve_ca_dir(g_opts.ca_dir);
380 /* export base = the URL path, trailing '/' trimmed; "/" → "" (verbatim). */
381 snprintf(g_base, sizeof(g_base), "%s", g_weburl.path);
382 bl = strlen(g_base);
383 while (bl > 0 && g_base[bl - 1] == '/') {
384 g_base[--bl] = '\0';
385 }
386 /* fail the mount up front if the export root is unreachable/denied. */
387 if (brix_web_stat(&g_weburl, g_base[0] ? g_base : "/", g_bearer,
388 g_web_verify, g_web_ca, &si, &st) != 0) {
389 fprintf(stderr, "xrootdfs: %s://%s:%d%s: %s\n",
390 g_weburl.tls ? "https" : "http", g_weburl.host, g_weburl.port,
391 g_weburl.path, st.msg);
392 return brix_shellcode(&st);
393 }
394 fprintf(stderr,
395 "xrootdfs: mounted %s:%d via %s%s (read-only WebDAV; "
396 "verify=%d, auth=%s)\n",
397 g_weburl.host, g_weburl.port, g_weburl.tls ? "HTTPS" : "HTTP",
398 g_base, g_web_verify, g_bearer ? "bearer" : "anon");
399 rc = fuse_main(fuse_argc, fuse_argv, &xfs_ops, NULL);
400 return rc;
401 }
402
403 if (brix_endpoint_parse(endpoint, &g_url, &st) != 0) {
404 fprintf(stderr, "xrootdfs: %s\n", st.msg);
405 return 2;
406 }
407
408 /* Export base = the URL path component (root://host/data → "/data"), so the
409 * mount roots at that subtree. Trailing '/' trimmed; a bare host (path "/" or
410 * empty) → verbatim FUSE paths. Shared with the web transport via srv_path(). */
411 {
412 size_t bl;
413 snprintf(g_base, sizeof(g_base), "%s",
414 (g_url.path[0] == '/') ? g_url.path : "");
415 bl = strlen(g_base);
416 while (bl > 0 && g_base[bl - 1] == '/') {
417 g_base[--bl] = '\0';
418 }
419 }
420
421 g_pool = brix_pool_create(&g_url, &g_opts, g_max_conns, &st);
422 if (g_pool == NULL) {
423 fprintf(stderr, "xrootdfs: connect %s:%d: %s\n",
424 g_url.host, g_url.port, st.msg);
425 return brix_shellcode(&st);
426 }
427 /* Default: connect all data streams up front (in parallel — ~1×RTT mount).
428 * --lazy-streams trades first-read warm-up for the lowest possible mount
429 * latency by bringing up just one stream now and the rest on demand. */
430 g_mgr = brix_mgr_create(&g_url, &g_opts, g_streams,
431 g_lazy_streams ? 1 : g_streams,
432 g_max_stall, g_keepalive, g_max_retries, &st);
433 if (g_mgr == NULL) {
434 fprintf(stderr, "xrootdfs: async manager: %s\n", st.msg);
435 brix_pool_destroy(g_pool);
436 return brix_shellcode(&st);
437 }
438
439 /* Probe the server's vendor POSIX extensions (kXR_Qconfig "xrdfs.ext") once;
440 * utimens/chown/symlink/readlink/link adapt to what is advertised. */
441 {
442 brix_conn *pc = brix_pool_checkout(g_pool, &st);
443 if (pc != NULL) {
444 int ok = (brix_ext_probe(pc, &g_ext_setattr, &g_ext_symlink,
445 &g_ext_readlink, &g_ext_link, &st) == 0);
446 brix_pool_checkin(g_pool, pc, ok ? 1 : xfs_conn_healthy(&st));
447 }
448 }
449
450 fprintf(stderr,
451 "xrootdfs: mounted %s:%d (meta-pool=%d, data-streams=%d, "
452 "max-stall=%dms; network-resilient; ext: setattr=%d symlink=%d "
453 "readlink=%d link=%d)\n",
454 g_url.host, g_url.port, g_max_conns, g_streams, g_max_stall,
455 g_ext_setattr, g_ext_symlink, g_ext_readlink, g_ext_link);
456
457 rc = fuse_main(fuse_argc, fuse_argv, &xfs_ops, NULL);
458
459 brix_mgr_destroy(g_mgr);
460 brix_pool_destroy(g_pool);
461 return rc;
462}
