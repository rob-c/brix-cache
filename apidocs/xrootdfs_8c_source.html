115/* Map a FUSE path ("/file") to the server path under the export base. With an
116 * empty base the FUSE path is used verbatim; with base "/data" → "/data/file"
117 * and "/" → "/data". Shared by BOTH transports (root:// and http/WebDAV) so a
118 * URL like root://host/data or https://host/data mounts that subtree. */
119const char *
120srv_path(const char *p, char *buf, size_t sz)
121{
122 if (g_base[0] == '\0') {
123 return p;
124 }
125 if (strcmp(p, "/") == 0) {
126 return g_base;
127 }
128 size_t bl = strlen(g_base);
129 size_t pl = strlen(p);
130 if (bl + pl + 1 > sz) { /* impossible for real paths — fail safe */
131 return g_base;
132 }
133 memcpy(buf, g_base, bl);
134 memcpy(buf + bl, p, pl + 1); /* includes the NUL */
135 return buf;
136}

276/* WHAT: consume one value-less command-line flag (boolean toggles).
277 * WHY: keeps the arg loop flat — each option family is one small matcher
278 * instead of a single branch ladder (phase-72 H3 decomposition).
279 * HOW: strcmp over the flag names, setting the matching global; returns 1
280 * when `a` was one of ours, 0 to let the caller keep matching. */
281static int
282aio_opt_novalue(const char *a)
283{
284 if (strcmp(a, "--tls") == 0) { g_opts.want_tls = 1; return 1; }
285 if (strcmp(a, "--notlsok") == 0) { g_opts.notlsok = 1; return 1; }
286 if (strcmp(a, "--noverifyhost") == 0) { g_opts.verify_host = 0; return 1; }
287 if (strcmp(a, "--lazy-streams") == 0) { g_lazy_streams = 1; return 1; }
288 if (strcmp(a, "--kernel-cache") == 0) { g_kernel_cache = 1; return 1; }
289 if (strcmp(a, "--xattr") == 0) { g_xattr = 1; return 1; }
290 return 0;
291}

294/* WHAT: consume one connection/resilience option that takes a value.
295 * WHY: splits the value-option ladder in two so each stays under the
296 * complexity gate; clamping (floors) is preserved verbatim.
297 * HOW: `v` is the next argv word (caller guarantees it exists and advances
298 * past it on a hit); returns 1 when `a` was consumed. */
299static int
300aio_opt_conn_value(const char *a, char *v)
301{
302 if (strcmp(a, "--auth") == 0) { g_opts.auth_force = v; return 1; }
303 if (strcmp(a, "--max-conns") == 0) {
304 g_max_conns = atoi(v);
305 if (g_max_conns < 1) { g_max_conns = 1; }
306 return 1;
307 }
308 if (strcmp(a, "--streams") == 0) {
309 g_streams = atoi(v);
310 if (g_streams < 1) { g_streams = 1; }
311 return 1;
312 }
313 if (strcmp(a, "--max-stall") == 0) {
314 g_max_stall = atoi(v);
315 if (g_max_stall < 0) { g_max_stall = 0; }
316 return 1;
317 }
318 if (strcmp(a, "--keepalive") == 0) {
319 g_keepalive = atoi(v);
320 if (g_keepalive < 0) { g_keepalive = 0; }
321 return 1;
322 }
323 if (strcmp(a, "--max-retries") == 0) {
324 g_max_retries = atoi(v);
325 if (g_max_retries < 0) { g_max_retries = 0; }
326 return 1;
327 }
328 if (strcmp(a, "--connect-timeout") == 0) { brix_tmo_set_connect_ms(atoi(v)); return 1; }
329 if (strcmp(a, "--io-timeout") == 0) { brix_tmo_set_io_ms(atoi(v)); return 1; }
330 return 0;
331}

334/* WHAT: consume one caching/auth option that takes a value.
335 * WHY: second half of the value-option split (see aio_opt_conn_value).
336 * HOW: same contract — `v` is the next argv word; returns 1 on a hit. */
337static int
338aio_opt_cache_value(const char *a, char *v)

342 if (strcmp(a, "--readahead") == 0) {
343 long n = atol(v);
344 g_readahead = (n > 0) ? (size_t) n : 0;
345 return 1;

348 long n = atol(v);

353 snprintf(g_compress, sizeof(g_compress), "%s", v);
354 return 1;
355 }
356 if (strcmp(a, "--token") == 0) { g_bearer = v; return 1; }
357 return 0;

361/* WHAT: split the command line into our options, the endpoint, and the
362 * libfuse passthrough vector.
363 * WHY: our known options are honored ANYWHERE on the line (before OR after
364 * the endpoint), so a resilience flag placed after the URL is honored
365 * rather than silently leaking to libfuse. Unknown dash-args fall
366 * through to the fuse passthrough (so -f/-d/-s/-o still work); the
367 * first bare word is the endpoint, the next the mountpoint.
368 * HOW: value options are only matched when a next word exists (a trailing
369 * `--token` passes through to fuse, exactly as before). Returns -1 to
370 * proceed with the mount, or a process exit code (--version/--help/-h
371 * inside the line exit immediately with 0). */
372static int
373aio_parse_args(int argc, char **argv, char **fuse_argv, int *fuse_argc,
374 const char **endpoint)
375{
376 int i;
377
378 for (i = 1; i < argc; i++) {
379 char *a = argv[i];
380 if (a[0] == '-') {
381 if (aio_opt_novalue(a)) { continue; }
382 if (i + 1 < argc && aio_opt_conn_value(a, argv[i + 1])) { i++; continue; }
383 if (i + 1 < argc && aio_opt_cache_value(a, argv[i + 1])) { i++; continue; }
384 if (strcmp(a, "--version") == 0) {
385 printf("xrootdfs (BriX-Cache client) %s\n", brix_client_version());
386 return 0;
387 }
388 if (strcmp(a, "--help") == 0) { usage_fp(stdout); return 0; } /* WS-2 */
389 if (strcmp(a, "-h") == 0) { usage(); return 0; } /* C1 */
390 if (*fuse_argc < 61) { fuse_argv[(*fuse_argc)++] = argv[i]; } /* fuse opt */
391 } else if (*endpoint == NULL) {
392 *endpoint = a;
393 } else if (*fuse_argc < 61) {
394 fuse_argv[(*fuse_argc)++] = argv[i];
395 }
396 }
397 return -1;
398}

401/* WHAT: set the export base (g_base) from a URL path component.
402 * WHY: shared by BOTH transports — the mount roots at the URL's /base
403 * subtree; srv_path() prepends it to every FUSE path.
404 * HOW: copy then trim trailing '/'; "/" or "" → empty base (verbatim FUSE
405 * paths). */
406static void
407aio_set_base(const char *path)
408{
409 size_t bl;
410
411 snprintf(g_base, sizeof(g_base), "%s", path);
412 bl = strlen(g_base);
413 while (bl > 0 && g_base[bl - 1] == '/') {
414 g_base[--bl] = '\0';
415 }
416}

419/* WHAT: mount an HTTP(S)/WebDAV endpoint (read-only, ranged GET).
420 * WHY: isolates the whole web transport bring-up (URL parse, bearer/TLS
421 * policy, reachability probe) from the root:// path in main.
422 * HOW: parse + validate the web URL, derive the export base, stat the
423 * export root up front (fail the mount early if unreachable/denied),
424 * then hand off to fuse_main. Returns the process exit code. */
425static int
426aio_web_mount(int fuse_argc, char **fuse_argv, const char *endpoint)
427{
428 brix_status st;
429 brix_statinfo si;
430
431 brix_status_clear(&st);
432 if (brix_weburl_parse(endpoint, &g_weburl) != 0) {
433 fprintf(stderr, "xrootdfs: bad web URL: %s\n", endpoint);
434 return 2;
435 }
436 if (g_weburl.is_s3) {
437 fprintf(stderr, "xrootdfs: s3:// is not supported as a FUSE mount "
438 "(use http/https/dav/davs)\n");
439 return 2;
440 }
441 g_web = 1;
442 if (g_bearer == NULL) {
443 g_bearer = getenv("BEARER_TOKEN");
444 }
445 g_web_verify = g_opts.verify_host;
446 g_web_ca = brix_resolve_ca_dir(g_opts.ca_dir);
447 /* export base = the URL path, trailing '/' trimmed; "/" → "" (verbatim). */
448 aio_set_base(g_weburl.path);
449 /* fail the mount up front if the export root is unreachable/denied. */
450 if (brix_web_stat(&g_weburl, g_base[0] ? g_base : "/", g_bearer,
451 g_web_verify, g_web_ca, &si, &st) != 0) {
452 fprintf(stderr, "xrootdfs: %s://%s:%d%s: %s\n",
453 g_weburl.tls ? "https" : "http", g_weburl.host, g_weburl.port,
454 g_weburl.path, st.msg);
455 return brix_shellcode(&st);
456 }
457 fprintf(stderr,
458 "xrootdfs: mounted %s:%d via %s%s (read-only WebDAV; "
459 "verify=%d, auth=%s, meta-pool=%d)\n",
460 g_weburl.host, g_weburl.port, g_weburl.tls ? "HTTPS" : "HTTP",
461 g_base, g_web_verify, g_bearer ? "bearer" : "anon", g_max_conns);
462
463 /* Pool the metadata path: the probe above validated endpoint/auth/TLS, so
464 * slot-0's eager connect will not be the first failure point. */
465 brix_webmeta_init(&g_web_tmpl, g_weburl.host, g_weburl.port, g_weburl.tls,
466 g_web_verify, g_web_ca, g_bearer, 0 /* → default 30 s */);
467 brix_status_clear(&st);
468 g_web_pool = brix_cpool_create(&WEB_VT, &g_web_tmpl, g_max_conns, &st);
469 if (g_web_pool == NULL) {
470 fprintf(stderr, "xrootdfs: web pool: %s\n", st.msg);
471 return brix_shellcode(&st);
472 }
473 int rc = fuse_main(fuse_argc, fuse_argv, &xfs_ops, NULL);
474 brix_cpool_destroy(g_web_pool);
475 g_web_pool = NULL;
476 return rc;
477}

481 * once at mount.
482 * WHY: utimens/chown/symlink/readlink/link adapt to what is advertised;
483 * absent capabilities keep the honest fallbacks.
484 * HOW: checkout a pooled metadata connection, probe, and check it back in
485 * (unhealthy on a probe failure that severed the link). */
486static void
487aio_probe_ext(brix_status *st)
488{
489 brix_conn *pc = brix_pool_checkout(g_pool, st);
490
491 if (pc != NULL) {

493 &g_ext_readlink, &g_ext_link, st) == 0);
494 brix_pool_checkin(g_pool, pc, ok ? 1 : xfs_conn_healthy(st));
495 }

499/* WHAT: mount a root[s]:// endpoint (binary XRootD; read-write, resilient).
500 * WHY: isolates the async-driver bring-up (metadata pool + data-stream
501 * manager + extension probe) and its teardown ordering from main.
502 * HOW: parse the endpoint, derive the export base, create the pool then the
503 * manager (destroyed in reverse order after fuse_main returns), probe
504 * extensions, run fuse. Returns the process exit code. */
505static int
506aio_root_mount(int fuse_argc, char **fuse_argv, const char *endpoint)
507{
508 brix_status st;
509 int rc;
510
511 brix_status_clear(&st);
512 if (brix_endpoint_parse(endpoint, &g_url, &st) != 0) {
513 fprintf(stderr, "xrootdfs: %s\n", st.msg);
514 return 2;
515 }
516
517 /* Export base = the URL path component (root://host/data → "/data"), so the
518 * mount roots at that subtree. Trailing '/' trimmed; a bare host (path "/" or
519 * empty) → verbatim FUSE paths. Shared with the web transport via srv_path(). */
520 aio_set_base((g_url.path[0] == '/') ? g_url.path : "");
521
522 g_pool = brix_pool_create(&g_url, &g_opts, g_max_conns, &st);
523 if (g_pool == NULL) {
524 fprintf(stderr, "xrootdfs: connect %s:%d: %s\n",
525 g_url.host, g_url.port, st.msg);
526 return brix_shellcode(&st);
527 }
528 /* Default: connect all data streams up front (in parallel — ~1×RTT mount).
529 * --lazy-streams trades first-read warm-up for the lowest possible mount
530 * latency by bringing up just one stream now and the rest on demand. */
531 g_mgr = brix_mgr_create(&g_url, &g_opts, g_streams,
532 g_lazy_streams ? 1 : g_streams,
533 g_max_stall, g_keepalive, g_max_retries, &st);
534 if (g_mgr == NULL) {
535 fprintf(stderr, "xrootdfs: async manager: %s\n", st.msg);
536 brix_pool_destroy(g_pool);
537 return brix_shellcode(&st);
538 }
539
540 aio_probe_ext(&st);
541
542 fprintf(stderr,
543 "xrootdfs: mounted %s:%d (meta-pool=%d, data-streams=%d, "
544 "max-stall=%dms; network-resilient; ext: setattr=%d symlink=%d "
545 "readlink=%d link=%d)\n",
546 g_url.host, g_url.port, g_max_conns, g_streams, g_max_stall,
547 g_ext_setattr, g_ext_symlink, g_ext_readlink, g_ext_link);
548
549 rc = fuse_main(fuse_argc, fuse_argv, &xfs_ops, NULL);
550
551 brix_mgr_destroy(g_mgr);
552 brix_pool_destroy(g_pool);
553 return rc;
554}

557/* Entry point for the default (async/resilient) driver. Invoked by the unified
558 * xrootdfs front-end (apps/xrootdfs_main.c); see xrootdfs_drivers.h. */
559int
560xrootdfs_aio_main(int argc, char **argv)
561{
562 const char *endpoint = NULL;
563 char *fuse_argv[64];
564 int fuse_argc = 0;
565 int rc;
566
