1/*
2 * tpc_user_proxy.h — per-user delegated x509 proxy resolution for the HTTP-TPC
3 * pull leg (phase-70 credential-forwarding closure).
4 *
5 * WHAT: Declares webdav_tpc_user_proxy_resolve(), which maps the requesting
6 * user's authenticated identity to the client certificate + key the pull
7 * leg must present to the SOURCE so the source authenticates the END USER
8 * (not the destination's static service cert), plus the small descriptor
9 * it fills.
10 *
11 * WHY: Without this, a WebDAV/HTTP TPC PULL presents conf->tpc_cert (a static
12 * service identity) to the source — the source logs the destination's DN,
13 * not the user's. The Phase-70 delegation infrastructure already captures
14 * the user's proxy two ways: (1) a live full-proxy passthrough on the
15 * request (rctx->deleg_proxy_pem, from X-Brix-Delegate-Proxy), and (2) a
16 * proxy uploaded to the per-user delegation store at
17 * <storage_credential_dir>/<key>.pem (keyed by the authenticated DN via
18 * brix_sd_ucred_*). This resolver reuses BOTH so the pull leg presents
19 * the user's own credential, closing the gap.
20 *
21 * HOW: Resolution order (first hit wins):
22 * 1. rctx->deleg_proxy_pem (live passthrough) → materialise to a 0600
23 * temp via brix_proxy_gsi_write_pem_temp(), register an unlink+zero
24 * request-pool cleanup, point cert_path/key_path at the temp.
25 * 2. per-user delegation store: brix_sd_ucred_select() over
26 * common.storage_credential_dir keyed by the request identity; on an
27 * x509 .pem hit (cert + key combined in one PEM) point cert_path/
28 * key_path at the stored file directly (no temp, no cleanup).
29 * 3. neither present → out->have=0: the caller keeps the current
30 * behaviour and falls back to conf->tpc_cert (non-delegated setups).
31 *
32 * Bearer-token identities are ignored here (the bearer path is applied as
33 * an Authorization header elsewhere and already works). A present-but-
34 * unusable delegation (malformed PEM, expired stored .pem) yields have=0
35 * — the resolver never fabricates or downgrades a credential silently.
36 *
37 * This runs on the EVENT LOOP (it needs the request ctx + pool); the resolved
38 * path strings are request-pool-owned and remain valid for the whole request,
39 * so the single-stream/multi-stream/thread-pool curl runners can all consume
40 * them by borrowed pointer.
41 */

47/*
48 * webdav_tpc_user_proxy_t — resolved per-user pull-leg client credential.
49 *
50 * have=0 → no per-user proxy for this request; cert_path/key_path are
51 * NULL and the caller falls back to conf->tpc_cert/_key.
52 * have=1 → present the user proxy: cert_path and key_path are
53 * NUL-terminated, request-pool-owned C strings a curl handle
54 * can pass to CURLOPT_SSLCERT / CURLOPT_SSLKEY. For both a
55 * materialised passthrough temp and a stored .pem the cert
56 * and key live in one combined PEM, so cert_path == key_path.
57 */
58typedef struct {
59 int have; /* 1 → present cert_path/key_path, 0 → fall back */
60 int deny; /* 1 → the user delegated a proxy we could NOT */
61 /* materialise: the caller MUST abort rather */
62 /* than silently present the service cert */
63 const char *cert_path; /* CURLOPT_SSLCERT path (NULL when have=0) */
64 const char *key_path; /* CURLOPT_SSLKEY path (NULL when have=0) */
65} webdav_tpc_user_proxy_t;

67/*
68 * webdav_tpc_user_proxy_resolve — fill *out with the requesting user's pull-leg
69 * client credential, or mark it absent so the caller falls back to the service
70 * cert.
71 *
72 * Never fails the request: on any resolution error (materialise failure, cred
73 * lookup error) it logs and leaves out->have=0, letting the caller decide (the
74 * pull still proceeds with the static cert exactly as before this change). The
75 * private key of a materialised temp is unlinked and zeroed by a request-pool
76 * cleanup registered here, so it never outlives the transfer.
77 */
78void webdav_tpc_user_proxy_resolve(ngx_http_request_t *r,
