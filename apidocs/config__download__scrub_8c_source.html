21/*
22 * Query-string credential keys whose value is scrubbed inside surviving URLs
23 * ("?token=..." / "&sig=..."). Matched case-insensitively at a '?'/'&'
24 * boundary; first hit wins (behavior-frozen table — do not reorder).

26static const char *const dashboard_cred_query_keys[] = {
27 "token=", "access_token=", "secret=", "client_secret=", "password=",
28 "passwd=", "apikey=", "api_key=", "key=", "sig=", "signature=",
29 "x-amz-credential=", "x-amz-signature=", "x-amz-security-token=", NULL

32/*
33 * Overwrite the secret region [vs,ve) with the redaction marker. NEVER grow
34 * the buffer: write min(marker,old) bytes (a sub-marker-length secret is
35 * fully covered by a marker prefix), and only shrink (memmove the tail left)
36 * when the secret is longer than the marker. Returns the new end pointer.
37 */
38static u_char *
39dashboard_redact_region(u_char *vs, u_char *ve, u_char *end)
40{
41 size_t rep = ngx_strlen(DASHBOARD_REDACTED);
42 size_t old = (size_t) (ve - vs);
43 size_t n = (rep < old) ? rep : old;
44
45 ngx_memcpy(vs, DASHBOARD_REDACTED, n);
46 if (old > rep) {
47 ngx_memmove(vs + rep, ve, (size_t) (end - ve));
48 end -= (old - rep);
49 }
50 return end;
51}

53/*
54 * Scan the URL authority starting at `auth` (just past "://") for a
55 * "user:pass@" userinfo block. Returns the '@' pointer when one is found
56 * before the authority ends (path/query/fragment/whitespace/quote/';'),
57 * else NULL. *colon receives the first ':' seen inside the scanned span
58 * (may be set even when NULL is returned).
59 */
60static u_char *
61dashboard_userinfo_at(u_char *auth, u_char *end, u_char **colon)
62{
63 u_char *q = auth;
64
65 *colon = NULL;
66 while (q < end && *q != '/' && *q != '?' && *q != '#'
67 && *q != ' ' && *q != '\t' && *q != '"' && *q != '\''
68 && *q != ';')
69 {
70 if (*q == ':' && *colon == NULL) { *colon = q; }
71 if (*q == '@') { return q; }
72 q++;
73 }
74 return NULL;
75}

77/*
78 * Redact "scheme://user:pass@host" userinfo when `s` points at "://": both a
79 * ':' and a later '@' must occur before the authority ends. On a rewrite,
80 * *endp is updated (the buffer may shrink) and the resume pointer (start of
81 * the rewritten authority) is returned so the caller re-scans from there;
82 * otherwise NULL (no rewrite, caller advances normally).
83 */
84static u_char *
85dashboard_scrub_userinfo(u_char *s, u_char **endp)
86{
87 u_char *end = *endp;
88 u_char *auth, *at, *colon;
89
90 if (s + 3 > end || s[0] != ':' || s[1] != '/' || s[2] != '/') {
91 return NULL;
92 }
93 auth = s + 3;
94 at = dashboard_userinfo_at(auth, end, &colon);
95 if (at == NULL || colon == NULL || colon >= at) {
96 return NULL;
97 }
98 *endp = dashboard_redact_region(auth, at, end);
99 return auth;
100}

115/*
116 * Scrub one query parameter starting at `kv` (the byte after '?'/'&'): if it
117 * matches a credential key from the table, overwrite its value with the
118 * redaction marker (never growing — see dashboard_redact_region). Returns the
119 * (possibly shrunk) new end pointer; first table hit wins.
120 */
121static u_char *
122dashboard_scrub_query_param(u_char *kv, u_char *end)
123{
124 size_t i;
125
126 for (i = 0; dashboard_cred_query_keys[i] != NULL; i++) {
127 size_t kl = ngx_strlen(dashboard_cred_query_keys[i]);
128
129 if (kv + kl <= end
130 && dashboard_name_eq(kv, kl, dashboard_cred_query_keys[i]))
131 {
132 u_char *vs = kv + kl;
133 u_char *ve = dashboard_query_value_end(vs, end);
134
135 if (ve > vs) {
136 end = dashboard_redact_region(vs, ve, end);
137 }
138 return end;
139 }
140 }
141 return end;
142}

144/*
145 * In-place scrub of embedded credentials in a surviving value region [p,end):
146 * scheme://user:pass@host -> scheme://[redacted]@host
147 * ...?token=v / &secret=v / &sig=v -> ...?token=[redacted]&...
148 * Returns the new end pointer (the region may shrink). Defence-in-depth for the
149 * brix_*_url / upstream / origin directives that pass the keep test.
150 */
151u_char *
152dashboard_scrub_value_creds(u_char *p, u_char *end)
153{
154 u_char *s = p;
155
156 while (s < end) {
157 /* userinfo: "://" ... ':' ... '@' before the authority ends */
158 u_char *resume = dashboard_scrub_userinfo(s, &end);
159
160 if (resume != NULL) {
161 s = resume; /* continue scanning after the rewrite */
162 continue;
163 }
164
165 /* query credential params: <key>=<value> up to & / delimiter */
166 if (*s == '?' || *s == '&') {
167 end = dashboard_scrub_query_param(s + 1, end);
168 }
169 s++;
170 }
171 return end;
172}
