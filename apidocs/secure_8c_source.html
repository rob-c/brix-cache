1/* secure.c — scvmfs:// security preamble (EXPERIMENTAL, phase-68 T22).
2 *
3 * WHAT: transport + client-authz gate that runs before the cvmfs gate on
4 * locations with `brix_scvmfs on`.
5 * WHY: "secure CVMFS" repositories are credential-protected; the site
6 * cache must enforce the same boundary or it becomes a leak. Layering
7 * it as a preamble keeps ONE protocol core — scvmfs can never drift
8 * behaviorally from cvmfs because it IS cvmfs after this function.
9 * HOW: TLS presence comes from the connection (r->connection->ssl); bearer
10 * mode delegates to the shared SciTokens issuer registry
11 * (brix_token_validate_registry — the same engine the WebDAV and
12 * stream token paths use; READ scope suffices for a read-only
13 * protocol). This file contains POLICY GLUE ONLY — zero crypto.
14 * VOMS/GSI client-cert mode is future work (the WebDAV auth_cert
15 * machinery needs a conf-independent seam first); the directive
16 * accepts none|bearer today.
17 */

25/* Extract "Authorization: Bearer <token>" into a NUL-terminated pool copy. */
26static ngx_int_t
27scvmfs_bearer_token(ngx_http_request_t *r, const char **token, size_t *len)
28{
29 ngx_str_t *v;
30 u_char *p;
31 size_t n;
32
33 if (r->headers_in.authorization == NULL) {
34 return NGX_DECLINED;
35 }
36 v = &r->headers_in.authorization->value;
37 if (v->len <= sizeof("Bearer ") - 1
38 || ngx_strncasecmp(v->data, (u_char *) "Bearer ",
39 sizeof("Bearer ") - 1) != 0)
40 {
41 return NGX_DECLINED;
42 }
43 n = v->len - (sizeof("Bearer ") - 1);
44 p = ngx_pnalloc(r->pool, n + 1);
45 if (p == NULL) {
46 return NGX_ERROR;
47 }
48 ngx_memcpy(p, v->data + sizeof("Bearer ") - 1, n);
49 p[n] = '\0';
50 *token = (const char *) p;
51 *len = n;
52 return NGX_OK;
53}

55static ngx_int_t
56scvmfs_check_bearer(ngx_http_request_t *r,
57 ngx_http_brix_cvmfs_loc_conf_t *lcf)
58{
59 const char *token;
60 size_t token_len;
61 char uri_path[PATH_MAX];
62 brix_token_claims_t claims;
63 int bucket = 0;
64 ngx_int_t rc;
65
66 if (lcf->scvmfs_registry == NULL) {
67 /* bearer mode without a loaded issuer registry fails CLOSED —
68 * merge-time validation makes this unreachable, but never open up */
69 return NGX_HTTP_UNAUTHORIZED;
70 }
71
72 rc = scvmfs_bearer_token(r, &token, &token_len);
73 if (rc == NGX_DECLINED) {
74 return NGX_HTTP_UNAUTHORIZED; /* no Bearer credential */
75 }
76 if (rc != NGX_OK) {
77 return NGX_HTTP_INTERNAL_SERVER_ERROR;
78 }
79
80 if (brix_str_cbuf(uri_path, sizeof(uri_path), &r->uri) == NULL) {
81 return NGX_HTTP_REQUEST_URI_TOO_LARGE;
82 }
83
84 /* read scope suffices for a read-only protocol */
85 {
86 brix_token_registry_args_t ra;
87
88 ra.log = r->connection->log;
89 ra.token = token;
90 ra.token_len = token_len;
91 ra.reg = (const brix_token_registry_t *) lcf->scvmfs_registry;
92 ra.macaroon_secret = NULL;
93 ra.secret_len = 0;
94 ra.clock_skew = BRIX_TOKEN_CLOCK_SKEW_SECS;
95 ra.claims = &claims;
96
97 if (brix_token_validate_registry(&ra, uri_path, BRIX_TOKEN_OP_READ,
98 &bucket) != 0)
99 {
100 return NGX_HTTP_UNAUTHORIZED; /* invalid/expired/out-of-scope */
101 }
102 }
103 return NGX_DECLINED; /* authenticated: proceed */
104}

106ngx_int_t
107brix_scvmfs_preamble(ngx_http_request_t *r,
108 ngx_http_brix_cvmfs_loc_conf_t *lcf)
109{
110 ngx_http_brix_cvmfs_ctx_t *ctx =
111 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
112 ngx_int_t rc;
113
114#if (NGX_HTTP_SSL)
115 if (r->connection->ssl == NULL)
116#endif
117 {
118 /* nginx core already 400s plain-HTTP-on-ssl-port before we run;
119 * this guards mixed listeners and future non-TLS plumbing. */
120 return NGX_HTTP_BAD_REQUEST;
121 }
122
123 switch (lcf->scvmfs_authz) {
124 case BRIX_SCVMFS_AUTHZ_BEARER:
125 rc = scvmfs_check_bearer(r, lcf);
126 break;
127 case BRIX_SCVMFS_AUTHZ_NONE:
128 default:
129 rc = NGX_DECLINED;
130 break;
131 }
132 if (rc != NGX_DECLINED) {
133 BRIX_CVMFS_METRIC_INC(requests_total[BRIX_CVMFS_CLASS_REJECT]);
134 return rc;
135 }
136 ctx->secure = 1; /* unlocks https upstream */
137 BRIX_CVMFS_METRIC_INC(secure_requests_total);
138 return NGX_DECLINED;
139}
