1/* secure.c — scvmfs:// security preamble (EXPERIMENTAL, phase-68 T22).
2 *
3 * WHAT: transport + client-authz gate that runs before the cvmfs gate on
4 * locations with `brix_scvmfs on`.
5 * WHY: "secure CVMFS" repositories are credential-protected; the site
6 * cache must enforce the same boundary or it becomes a leak. Layering
7 * it as a preamble keeps ONE protocol core — scvmfs can never drift
8 * behaviorally from cvmfs because it IS cvmfs after this function.
9 * HOW: TLS presence comes from the connection (r->connection->ssl); bearer
10 * mode delegates to the shared SciTokens issuer registry
11 * (brix_token_validate_registry — the same engine the WebDAV and
12 * stream token paths use; READ scope suffices for a read-only
13 * protocol). This file contains POLICY GLUE ONLY — zero crypto.
14 * VOMS/GSI client-cert mode is future work (the WebDAV auth_cert
15 * machinery needs a conf-independent seam first); the directive
16 * accepts none|bearer today.
17 */

26/* Extract "Authorization: Bearer <token>" into a NUL-terminated pool copy. */
27static ngx_int_t
28scvmfs_bearer_token(ngx_http_request_t *r, const char **token, size_t *len)
29{
30 ngx_str_t *v;
31 u_char *p;
32 size_t n;
33
34 if (r->headers_in.authorization == NULL) {
35 return NGX_DECLINED;
36 }
37 v = &r->headers_in.authorization->value;
38 if (v->len <= sizeof("Bearer ") - 1
39 || ngx_strncasecmp(v->data, (u_char *) "Bearer ",
40 sizeof("Bearer ") - 1) != 0)
41 {
42 return NGX_DECLINED;
43 }
44 n = v->len - (sizeof("Bearer ") - 1);
45 p = ngx_pnalloc(r->pool, n + 1);
46 if (p == NULL) {
47 return NGX_ERROR;
48 }
49 ngx_memcpy(p, v->data + sizeof("Bearer ") - 1, n);
50 p[n] = '\0';
51 *token = (const char *) p;
52 *len = n;
53 return NGX_OK;
54}

56static ngx_int_t
57scvmfs_check_bearer(ngx_http_request_t *r,
58 ngx_http_brix_cvmfs_loc_conf_t *lcf)
59{
60 const char *token;
61 size_t token_len;
62 char uri_path[PATH_MAX];
63 brix_token_claims_t claims;
64 int bucket = 0;
65 ngx_int_t rc;
66
67 if (lcf->scvmfs_registry == NULL) {
68 /* bearer mode without a loaded issuer registry fails CLOSED —
69 * merge-time validation makes this unreachable, but never open up */
70 return NGX_HTTP_UNAUTHORIZED;
71 }
72
73 rc = scvmfs_bearer_token(r, &token, &token_len);
74 if (rc == NGX_DECLINED) {
75 return NGX_HTTP_UNAUTHORIZED; /* no Bearer credential */
76 }
77 if (rc != NGX_OK) {
78 return NGX_HTTP_INTERNAL_SERVER_ERROR;
79 }
80
81 if (brix_str_cbuf(uri_path, sizeof(uri_path), &r->uri) == NULL) {
82 return NGX_HTTP_REQUEST_URI_TOO_LARGE;
83 }
84
85 /* read scope suffices for a read-only protocol */
86 {
87 brix_token_registry_args_t ra;
88
89 ra.log = r->connection->log;
90 ra.token = token;
91 ra.token_len = token_len;
92 ra.reg = (const brix_token_registry_t *) lcf->scvmfs_registry;
93 ra.macaroon_secret = NULL;
94 ra.secret_len = 0;
95 ra.clock_skew = BRIX_TOKEN_CLOCK_SKEW_SECS;
96 ra.claims = &claims;
97
98 if (brix_token_validate_registry(&ra, uri_path, BRIX_TOKEN_OP_READ,
99 &bucket) != 0)
100 {
101 return NGX_HTTP_UNAUTHORIZED; /* invalid/expired/out-of-scope */
102 }
103 }
104
105 /* the VALIDATED subject is the F9 QoS classification key */
106 {
107 ngx_http_brix_cvmfs_ctx_t *ctx =
108 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
109
110 if (ctx != NULL) {
111 ngx_cpystrn((u_char *) ctx->token_sub, (u_char *) claims.sub,
112 sizeof(ctx->token_sub));
113 }
114 }
115 return NGX_DECLINED; /* authenticated: proceed */
116}

157/* The gate entry matching this repo, or NULL = repo not gated. First match
158 * wins; "*" is a catch-all so an exact entry listed before it can pin a
159 * specific registry while "*" sweeps the rest. */
160static brix_cvmfs_repo_authz_t *
161cvmfs_repo_authz_match(ngx_http_brix_cvmfs_loc_conf_t *lcf,
162 const char *repo, size_t repo_len)
163{
164 brix_cvmfs_repo_authz_t *e = lcf->repo_authz->elts;
165 ngx_uint_t i;
166
167 for (i = 0; i < lcf->repo_authz->nelts; i++) {
168 if (e[i].repo.len == 1 && e[i].repo.data[0] == '*') {
169 return &e[i];
170 }
171 if (e[i].repo.len == repo_len
172 && ngx_strncmp(e[i].repo.data, repo, repo_len) == 0)
173 {
174 return &e[i];
175 }
176 }
177 return NULL;
178}

256brix_scvmfs_preamble(ngx_http_request_t *r,
257 ngx_http_brix_cvmfs_loc_conf_t *lcf)
258{
259 ngx_http_brix_cvmfs_ctx_t *ctx =

261 ngx_int_t rc;
262
263#if (NGX_HTTP_SSL)
264 if (r->connection->ssl == NULL)

270 }
271
272 switch (lcf->scvmfs_authz) {

283 return rc;
284 }
285 ctx->secure = 1; /* unlocks https upstream */
286 BRIX_CVMFS_METRIC_INC(secure_requests_total);
287 return NGX_DECLINED;

290/* ---- per-VO/per-job QoS fill throttling (phase-85 F9) ---------------------
291 * brix_cvmfs_qos <class> sub=<subject>|default fills=<n> — multi-occurrence;
292 * each entry maps ONE validated token subject (or `default` = everything
293 * unclassified) to a fills-per-second budget. Only ORIGIN FILLS are charged
294 * (the caller gates on the remote-miss predicate) — cache hits always flow,
295 * so a throttled class is bounded at the shared WAN/origin resource, which
296 * is exactly the noisy-neighbor surface. fills=0 = unlimited (parity). */
297
298char *
299cvmfs_conf_qos(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
300{
301 ngx_http_brix_cvmfs_loc_conf_t *c = conf;
302 ngx_str_t *value = cf->args->elts;
303 brix_cvmfs_qos_t *entry;
304 ngx_int_t fills;
305
306 (void) cmd;
307
308 if (value[3].len <= sizeof("fills=") - 1
309 || ngx_strncmp(value[3].data, "fills=", sizeof("fills=") - 1) != 0)
310 {
311 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
312 "brix_cvmfs_qos needs <class> sub=<subject>|default fills=<n>");
313 return NGX_CONF_ERROR;
314 }
315 fills = ngx_atoi(value[3].data + sizeof("fills=") - 1,
316 value[3].len - (sizeof("fills=") - 1));
317 if (fills == NGX_ERROR) {
318 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
319 "brix_cvmfs_qos: \"%V\" is not a fill rate", &value[3]);
320 return NGX_CONF_ERROR;
321 }
322
323 if (c->qos == NGX_CONF_UNSET_PTR) {
324 c->qos = ngx_array_create(cf->pool, 4, sizeof(brix_cvmfs_qos_t));
325 if (c->qos == NULL) {
326 return NGX_CONF_ERROR;
327 }
328 }
329 entry = ngx_array_push(c->qos);
330 if (entry == NULL) {
331 return NGX_CONF_ERROR;
332 }
333 ngx_memzero(entry, sizeof(*entry));
334 entry->name = value[1];
335 entry->fills = (ngx_uint_t) fills;
336
337 if (value[2].len == sizeof("default") - 1
338 && ngx_strncmp(value[2].data, "default", value[2].len) == 0)
339 {
340 /* sub stays empty = the unclassified catch-all */
341 return NGX_CONF_OK;
342 }
343 if (value[2].len <= sizeof("sub=") - 1
344 || ngx_strncmp(value[2].data, "sub=", sizeof("sub=") - 1) != 0)
345 {
346 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
347 "brix_cvmfs_qos: match must be sub=<subject> or default");
348 return NGX_CONF_ERROR;
349 }
350 entry->sub.data = value[2].data + (sizeof("sub=") - 1);
351 entry->sub.len = value[2].len - (sizeof("sub=") - 1);
352 return NGX_CONF_OK;
353}

355/* The class for `sub` ("" = anonymous): first sub= match wins, else the
356 * first `default` entry, else NULL (identity unthrottled). */
357static brix_cvmfs_qos_t *
358cvmfs_qos_class(ngx_http_brix_cvmfs_loc_conf_t *lcf, const char *sub)
359{
360 brix_cvmfs_qos_t *e = lcf->qos->elts;
361 brix_cvmfs_qos_t *def = NULL;
362 size_t sub_len = ngx_strlen(sub);
363 ngx_uint_t i;
364
365 for (i = 0; i < lcf->qos->nelts; i++) {
366 if (e[i].sub.len == 0) {
367 if (def == NULL) {
368 def = &e[i];
369 }
370 continue;
371 }
372 if (e[i].sub.len == sub_len
373 && ngx_strncmp(e[i].sub.data, sub, sub_len) == 0)
374 {
375 return &e[i];
376 }
377 }
378 return def;
379}

381ngx_int_t
382brix_cvmfs_qos_check(ngx_http_request_t *r,
383 ngx_http_brix_cvmfs_loc_conf_t *lcf)
384{
385 ngx_http_brix_cvmfs_ctx_t *ctx =
386 ngx_http_get_module_ctx(r, ngx_http_brix_cvmfs_module);
387 brix_cvmfs_qos_t *cls;
388 ngx_msec_t now;
389
390 if (lcf->qos == NULL || ctx == NULL) {
391 return NGX_DECLINED;
392 }
393 cls = cvmfs_qos_class(lcf, ctx->token_sub);
394 if (cls == NULL || cls->fills == 0) {
395 return NGX_DECLINED; /* unthrottled / 0 = parity */
396 }
397
398 /* Token bucket in milli-fills: capacity fills*1000, refill fills/ms-
399 * scaled, one fill costs 1000. Worker-local (COW conf memory), event-
400 * loop only — each worker bounds its own share. */
401 now = ngx_current_msec;
402 if (cls->last == 0) {
403 cls->tokens = (ngx_int_t) (cls->fills * 1000); /* first sight: full */
404 } else if (now != cls->last) {
405 ngx_int_t cap = (ngx_int_t) (cls->fills * 1000);
406
407 cls->tokens += (ngx_int_t) ((now - cls->last) * cls->fills);
408 if (cls->tokens > cap) {
409 cls->tokens = cap;
410 }
411 }
412 cls->last = now;
413
414 if (cls->tokens < 1000) {
415 ngx_log_error(NGX_LOG_NOTICE, r->connection->log, 0,
416 "cvmfs: qos class \"%V\" fill budget exhausted "
417 "(sub \"%s\", %ui fills/s) - 429", &cls->name,
418 ctx->token_sub[0] != '\0' ? ctx->token_sub : "anonymous",
419 cls->fills);
420 return NGX_HTTP_TOO_MANY_REQUESTS;
421 }
422 cls->tokens -= 1000;
423 return NGX_DECLINED;
424}
