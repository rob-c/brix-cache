22/* brix_pblock_write_opts_sidecar — persist a pblock `?tail` query string as the
23 * one-line <root>/pblock.opts sidecar the pblock driver parses at instance init
24 * (Phase-83 static opts). Best-effort at config finalise: the root dir already
25 * exists (mkdir ran first). Kept here (not in the sqlite-gated pblock driver) so
26 * it links regardless of BRIX_HAVE_SQLITE. */
27static void
28brix_pblock_write_opts_sidecar(const char *root, const char *tail)
29{
30 char path[4096];
31 FILE *f;
32
33 (void) snprintf(path, sizeof(path), "%s/pblock.opts", root);
34 f = fopen(path, "we");
35 if (f == NULL) {
36 return;
37 }
38 (void) fputs(tail != NULL ? tail : "", f);
39 (void) fputc('\n', f);
40 (void) fclose(f);
41}

43/* Directive setter for a tier store-URL directive: arg[1] = the store URL (into the
44 * ngx_str_t at cmd->offset); args[2..] = trailing credential=/block_size= params
45 * (into the ngx_array_t* at the field offset carried in cmd->post). See header. */
46char *
47brix_conf_set_store_slot(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
48{
49 char *p = conf;
50 ngx_str_t *url = (ngx_str_t *) (p + cmd->offset);
51 ngx_array_t **args = (ngx_array_t **) (p + (uintptr_t) cmd->post);
52 ngx_str_t *value = cf->args->elts;
53 ngx_uint_t i;
54
55 if (url->data != NULL) {
56 return "is duplicate";
57 }
58 *url = value[1]; /* the store URL token */
59
60 if (cf->args->nelts > 2) {
61 *args = ngx_array_create(cf->pool, cf->args->nelts - 2, sizeof(ngx_str_t));
62 if (*args == NULL) {
63 return NGX_CONF_ERROR;
64 }
65 for (i = 2; i < cf->args->nelts; i++) {
66 ngx_str_t *a = ngx_array_push(*args);
67
68 if (a == NULL) {
69 return NGX_CONF_ERROR;
70 }
71 *a = value[i]; /* "credential=..." / "block_size=..." */
72 }
73 }
74 return NGX_CONF_OK;
75}

77/* A LOCAL storage backend NAMES THE EXPORT TREE — the fully composable replacement
78 * for brix_root. Rewrites common->root from the backend URL and anchors root_canon
79 * there. No-op for a remote/non-local backend (root://, tape://, http://, a cache
80 * origin, or none). Shared by all three protocol finalisers; called BEFORE the
81 * export-root prep. Two forms:
82 * posix:<path> → root = <path>; CLEAR the backend (default POSIX driver).
83 * pblock://<path> → root = /<path> (one leading '/' guaranteed: "pblock://x"→
84 * "/x", "pblock:///abs"→"//abs"→/abs); KEEP the "pblock"
85 * driver; create the block-store directory on init if needed. */
86void
87brix_storage_backend_posix_root(ngx_http_brix_shared_conf_t *common)
88{
89 ngx_str_t *sb = &common->storage_backend;
90
91 if (sb->len > sizeof("posix:") - 1
92 && ngx_strncmp(sb->data, "posix:", sizeof("posix:") - 1) == 0)
93 {
94 u_char *p = sb->data + (sizeof("posix:") - 1);
95 size_t plen = sb->len - (sizeof("posix:") - 1);
96
97 /* Accept both posix:<path> and posix://<path>: collapse leading "//" runs to
98 * one '/' so the root is a single canonical path (a stray double slash
99 * propagates into common->root and breaks raw prefix-strip path uses).
100 * "posix://abs" with an absolute path yields "///abs" → "/abs". */
101 while (plen >= 2 && p[0] == '/' && p[1] == '/') {
102 p++;
103 plen--;
104 }
105 common->root.data = p;
106 common->root.len = plen;
107 sb->len = 0; /* default POSIX driver */
108 return;
109 }
110
111 if (sb->len > sizeof("pblock://") - 1
112 && ngx_strncmp(sb->data, "pblock://", sizeof("pblock://") - 1) == 0)
113 {
114 size_t base = sizeof("pblock://") - 1; /* past "pblock://" */
115
116 /* Yield EXACTLY one leading '/': "pblock:///abs" already has it (offset
117 * `base`); "pblock://rel" gains it by keeping the prior '/' (offset base-1).
118 * A double slash would break the write-through's prefix-strip path derivation. */
119 if (sb->data[base] == '/') {
120 common->root.data = sb->data + base;
121 common->root.len = sb->len - base;
122 } else {
123 common->root.data = sb->data + base - 1;
124 common->root.len = sb->len - base + 1;
125 }
126 sb->len = sizeof("pblock") - 1; /* the bare "pblock" driver */
127
128 /* Phase-83 static opts: a `?tail` query string on the root
129 * (pblock:///srv/x?lab=1&caps=-sendfile) selects the lab gate + caps/mem
130 * knobs. Strip it BEFORE mkdir/root_canon (which would otherwise create a
131 * literal '?' directory / canonicalise it away) by NUL-terminating the
132 * root at '?', then stash the tail as the <root>/pblock.opts sidecar the
133 * driver reads at init. No tail ⇒ no sidecar ⇒ lab OFF (production path). */
134 {
135 u_char *q = memchr(common->root.data, '?', common->root.len);
136
137 if (q != NULL) {
138 const char *tail = (const char *) (q + 1);
139
140 *q = '\0'; /* truncate root at '?' */
141 common->root.len = (size_t) (q - common->root.data);
142 (void) brix_mkdir_recursive((const char *) common->root.data,
143 0755);
144 brix_pblock_write_opts_sidecar(
145 (const char *) common->root.data, tail);
146 } else {
147 (void) brix_mkdir_recursive((const char *) common->root.data,
148 0755);
149 }
150 }
151 }
152}

154/* 1 iff the storage backend is REMOTE (the export's bytes live off-box: a root://
155 * origin, http(s)://, s3://, tape://, ceph). For such an export the LOCAL root_canon
156 * is only a namespace anchor — never written — so its export-root prep must not
157 * demand W_OK (a pure remote-backed node defaults root_canon to "/", which is not
158 * writable). Local backends (posix/pblock, or none) return 0 and keep the W_OK
159 * check. Called after brix_storage_backend_posix_root (posix:/pblock:// already
160 * rewritten away). */
161int
162brix_storage_backend_is_remote(const ngx_http_brix_shared_conf_t *common)
163{
164 static const char *const schemes[] = {
165 "root://", "roots://", "http://", "https://", "s3://",
166 "tape://", "frm://", "rados://", "ceph:", "cephfsro:", NULL
167 };
168 const ngx_str_t *sb = &common->storage_backend;
169 int i;
170
171 for (i = 0; schemes[i] != NULL; i++) {
172 size_t n = ngx_strlen(schemes[i]);
173
174 if (sb->len >= n && ngx_strncmp(sb->data, schemes[i], n) == 0) {
175 return 1;
176 }
177 }
178 return 0;
179}

192 int i;

196

199 }

206/* 1 iff this process's /tmp is a systemd PrivateTmp mount — the /tmp
207 * mountpoint's root inside its filesystem is a /tmp/systemd-private-<id>/tmp
208 * subtree. Parsed from /proc/self/mountinfo (fields: id parent major:minor
209 * ROOT MOUNTPOINT ...). Config-domain read, best-effort (0 on any failure). */
210static int
211brix_tmp_is_systemd_private(void)
212{
213 FILE *f;
214 char line[1024];
215 int private_tmp = 0;
216
217 f = fopen("/proc/self/mountinfo", "re");
218 if (f == NULL) {
219 return 0;
220 }
221 while (fgets(line, sizeof(line), f) != NULL) {
222 char root[512], mnt[512];
223
224 if (sscanf(line, "%*d %*d %*31s %511s %511s", root, mnt) == 2
225 && strcmp(mnt, "/tmp") == 0
226 && strstr(root, "/systemd-private-") != NULL)
227 {
228 private_tmp = 1;
229 break;
230 }
231 }
232 (void) fclose(f);
233 return private_tmp;
234}

236/* Shout — loudly, per store — when a stage/cache store lives under a systemd
237 * PrivateTmp /tmp: that mount is PRIVATE to the service and DELETED on every
238 * service stop/restart, so staged-but-unflushed writes and cached objects do
239 * not survive. The shipped unit sets PrivateTmp=true, making this the default
240 * deployment posture — the operator must be told, not left to find out from a
241 * post-restart empty spool. */
242static void
243brix_tier_warn_private_tmp(ngx_conf_t *cf,
244 const ngx_http_brix_shared_conf_t *common)
245{
246 const ngx_str_t *stores[2];
247 int i, shouted = 0;
248
249 if (!brix_tmp_is_systemd_private()) {
250 return;
251 }
252 stores[0] = &common->stage_store;
253 stores[1] = &common->cache_store;
254
255 for (i = 0; i < 2; i++) {
256 if (stores[i]->len == 0
257 || ngx_strnstr(stores[i]->data, ":/tmp/", stores[i]->len) == NULL)
258 {
259 continue;
260 }
261 if (!shouted) {
262 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
263 "brix: ==================== PrivateTmp DETECTED "
264 "====================");
265 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
266 "brix: this service runs with systemd PrivateTmp: /tmp is "
267 "PRIVATE to the service and is DELETED on every service "
268 "stop/restart.");
269 shouted = 1;
270 }
271 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
272 "brix: the store \"%V\" therefore does NOT survive a restart — "
273 "staged-but-unflushed writes / cached objects are LOST. Point "
274 "it at a durable path or set PrivateTmp=no in the unit.",
275 stores[i]);
276 }
277 if (shouted) {
278 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
279 "brix: ==========================================================="
280 "====");
281 }
282}

284/* Build "<base>/<sanitised backend URL>" for the brix-managed default stage
285 * store: every backend-URL byte outside [A-Za-z0-9._-] becomes '_', capped at
286 * 160 bytes, so distinct backends get distinct filesystem-safe spool
287 * directories under the one managed base. */
288static void
289brix_tier_default_stage_dir(const ngx_str_t *backend, char *dir, size_t cap)
290{
291 size_t pos = sizeof(BRIX_TIER_DEFAULT_STAGE_BASE "/") - 1;
292 size_t i;
293
294 ngx_cpystrn((u_char *) dir, (u_char *) BRIX_TIER_DEFAULT_STAGE_BASE "/",
295 cap);
296 for (i = 0; i < backend->len && i < 160 && pos + 1 < cap; i++) {
297 u_char c = backend->data[i];
298 int keep = (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')
299 || (c >= '0' && c <= '9') || c == '.' || c == '_'
300 || c == '-';
301
302 dir[pos++] = keep ? (char) c : '_';
303 }
304 dir[pos] = '\0';
305}

307/*
308 * brix_tier_default_stage_store() — zero-config write staging for a
309 * whole-object remote gateway (WebDAV/S3 backend).
310 *
311 * WHAT: When a writable export forwards to a whole-object remote backend and
312 * the operator configured NO stage tier at all, provision the stock
313 * stage decorator over a brix-managed posix store at
314 * /tmp/staging/<sanitised-backend-url>: create the directory (0711
315 * base, 0700 leaf chown'd to the worker user, exactly as
316 * brix_shared_credential_dir_ensure hands over /dev/shm/brix-creds),
317 * point common->stage_store at "posix:<dir>", flip stage_enable on,
318 * and shout a multi-line [warn] banner naming the directory, the tmpfs
319 * option and the opt-outs.
320 *
321 * WHY: Without a stage tier the remote driver rejects random-offset writes
322 * at the cap (kXR_Unsupported / EINVAL) — correct but hostile as the
323 * default posture of a POSIX-speaking gateway. Routing the default
324 * through the SAME tier grammar the operator would write by hand
325 * (brix_stage on + brix_stage_store posix:…) keeps one code path:
326 * tier_parse_local validates the directory like an export root and the
327 * stock sd_stage decorator does all buffering and flushing — no bespoke
328 * staging machinery.
329 *
330 * HOW: Build the sanitised leaf path; mkdir base 0711 (root-owned, traverse
331 * only) and leaf 0700; chown the leaf to ccf->user when the master runs
332 * as root (the workers write the spool). On ANY failure warn and return
333 * with staging left off — writes then require strictly sequential
334 * offsets, exactly as before — never fatal. On success rewrite
335 * stage_store (cf->pool) + stage_enable and emit the banner.
336 */
337static void
338brix_tier_default_stage_store(ngx_conf_t *cf,
339 ngx_http_brix_shared_conf_t *common)
340{
341 char dir[PATH_MAX];
342 u_char *url;
343 size_t url_len;
344 struct stat st;
345 uid_t want_uid;
346 gid_t want_gid;
347
348 brix_tier_default_stage_dir(&common->storage_backend, dir, sizeof(dir));
349
350 if (brix_mkdir_recursive(BRIX_TIER_DEFAULT_STAGE_BASE, 0711) != 0
351 || (mkdir(dir, 0700) != 0 && errno != EEXIST))
352 {
353 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
354 "brix: cannot create the default stage store \"%s\" — "
355 "write staging stays OFF; random writes to the remote backend "
356 "will be rejected (set brix_stage_store, or \"brix_stage off\" "
357 "to silence this warning)", dir);
358 return;
359 }
360 if (stat(dir, &st) != 0 || !S_ISDIR(st.st_mode)) {
361 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
362 "brix: default stage store \"%s\" is not a usable directory — "
363 "write staging stays OFF", dir);
364 return;
365 }
366
367 /* The master parses config as root but the workers write the spool as
368 * the RUNTIME worker identity — the `user` account, or the always-on
369 * de-escalation target (brix_worker_user/nobody) for a root-capable
370 * worker — hand the directory to them, exactly as ngx_create_paths does
371 * for the temp paths. */
372 brix_shared_worker_dir_ids(cf, &want_uid, &want_gid);
373 if (geteuid() == 0 && st.st_uid != want_uid
374 && chown(dir, want_uid, want_gid) != 0)
375 {
376 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
377 "brix: cannot chown the default stage store \"%s\" to the "
378 "worker user — write staging stays OFF", dir);
379 return;
380 }
381
382 url_len = sizeof("posix:") - 1 + ngx_strlen(dir);
383 url = ngx_pnalloc(cf->pool, url_len + 1);
384 if (url == NULL) {
385 return;
386 }
387 ngx_sprintf(url, "posix:%s%Z", dir);
388 common->stage_store.data = url;
389 common->stage_store.len = url_len;
390 common->stage_enable = 1;
391
392 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
393 "brix: ==================== DEFAULT WRITE-STAGING ===================");
394 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
395 "brix: no stage tier is configured for the whole-object remote "
396 "backend \"%V\";", &common->storage_backend);
397 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
398 "brix: random writes will be buffered in \"%s\",", dir);
399 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
400 "brix: a directory MANAGED BY BRIX (created, filled and cleaned by "
401 "brix — do not place files there; mount it tmpfs if you want "
402 "RAM-tier performance).");
403 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
404 "brix: override with \"brix_stage_store <store-url>\", or opt out "
405 "with \"brix_stage off\" (writes must then be strictly sequential).");
406 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
407 "brix: ===============================================================");
408}

410/* Load the CVMFS repo master public key PEM named by `path` into the cache
411 * policy (phase-85 F1). Config-time, cf->pool-owned (cycle lifetime — workers
412 * inherit the pointer through the registered policy copy). The file may hold
413 * several concatenated PEM keys (CVMFS key rotation); content is validated by
414 * the OpenSSL PEM parser at verify time, here only shape-checked. [emerg] on
415 * any failure — a verifying proxy with an unloadable trust anchor must not
416 * start. */
417static ngx_int_t
418brix_tier_load_master_key(ngx_conf_t *cf, const ngx_str_t *path,
419 brix_cache_policy_t *pol)
420{
421 char pathz[1024];
422 u_char *buf;
423 ssize_t n;
424 off_t size;
425 ngx_fd_t fd;
426 ngx_file_info_t fi;
427
428 if (path->len >= sizeof(pathz)) {
429 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
430 "brix_cvmfs_verify_manifest: key path too long");
431 return NGX_ERROR;
432 }
433 ngx_cpystrn((u_char *) pathz, path->data, path->len + 1);
434
435 fd = open(pathz, O_RDONLY | O_CLOEXEC); /* vfs-seam-allow: config-domain trust-anchor PEM (not export storage) */
436 if (fd == -1) {
437 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
438 "brix_cvmfs_verify_manifest: cannot open \"%s\"", pathz);
439 return NGX_ERROR;
440 }
441 if (ngx_fd_info(fd, &fi) == -1
442 || (size = ngx_file_size(&fi)) <= 0 || size > 65536)
443 {
444 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
445 "brix_cvmfs_verify_manifest: \"%s\" is empty, unreadable "
446 "or larger than 64KB", pathz);
447 (void) close(fd);
448 return NGX_ERROR;
449 }
450 buf = ngx_palloc(cf->pool, (size_t) size + 1);
451 if (buf == NULL) {
452 (void) close(fd);
453 return NGX_ERROR;
454 }
455 n = read(fd, buf, (size_t) size); /* vfs-seam-allow: config-domain trust-anchor PEM (not export storage) */
456 (void) close(fd);
457 if (n != (ssize_t) size) {
458 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
459 "brix_cvmfs_verify_manifest: short read on \"%s\"", pathz);
460 return NGX_ERROR;
461 }
462 buf[size] = '\0';
463 if (ngx_strstr(buf, "BEGIN PUBLIC KEY") == NULL) {
464 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
465 "brix_cvmfs_verify_manifest: \"%s\" holds no PEM public key",
466 pathz);
467 return NGX_ERROR;
468 }
469 pol->cvmfs_master_pub = buf;
470 pol->cvmfs_master_pub_len = (size_t) size;
471 return NGX_OK;
472}

474/* Parse + validate the brix_cache_peers ring (phase-85 F8) and record it on
475 * the backend registry. Each member token is "host:port"; this node's own slot
476 * is "self=host:port" (the mesh needs every node to carry the IDENTICAL list so
477 * rendezvous ownership agrees, so self is marked, never omitted). Operator
478 * errors — malformed authority, no/duplicate self, fewer than 2 members, more
479 * than 16 — are [emerg], failing nginx -t. */
480static ngx_int_t
481brix_tier_register_cache_peers(ngx_conf_t *cf,
482 ngx_http_brix_shared_conf_t *common)
483{
484 char hosts[16][256];
485 int ports[16];
486 int self = -1;
487 ngx_str_t *tok = common->cache_peers->elts;
488 ngx_uint_t i;
489
490 if (common->cache_peers->nelts < 2) {
491 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
492 "brix_cache_peers: a mesh needs at least 2 ring members "
493 "(self=host:port plus one sibling)");
494 return NGX_ERROR;
495 }
496 if (common->cache_peers->nelts > 16) {
497 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
498 "brix_cache_peers: at most 16 ring members supported");
499 return NGX_ERROR;
500 }
501
502 for (i = 0; i < common->cache_peers->nelts; i++) {
503 char buf[300];
504 char *auth = buf;
505 char *colon;
506 long port;
507
508 if (tok[i].len >= sizeof(buf)) {
509 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
510 "brix_cache_peers: member \"%V\" is too long", &tok[i]);
511 return NGX_ERROR;
512 }
513 ngx_memcpy(buf, tok[i].data, tok[i].len);
514 buf[tok[i].len] = '\0';
515
516 if (ngx_strncmp(auth, "self=", sizeof("self=") - 1) == 0) {
517 if (self >= 0) {
518 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
519 "brix_cache_peers: more than one self= member");
520 return NGX_ERROR;
521 }
522 self = (int) i;
523 auth += sizeof("self=") - 1;
524 }
525 colon = strrchr(auth, ':');
526 if (colon == NULL || colon == auth || colon[1] == '\0') {
527 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
528 "brix_cache_peers: member \"%V\" is not host:port", &tok[i]);
529 return NGX_ERROR;
530 }
531 port = strtol(colon + 1, NULL, 10);
532 if (port < 1 || port > 65535) {
533 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
534 "brix_cache_peers: member \"%V\" has an invalid port",
535 &tok[i]);
536 return NGX_ERROR;
537 }
538 *colon = '\0';
539 if (ngx_strlen(auth) >= sizeof(hosts[0])) {
540 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
541 "brix_cache_peers: member \"%V\" host is too long", &tok[i]);
542 return NGX_ERROR;
543 }
544 ngx_cpystrn((u_char *) hosts[i], (u_char *) auth, sizeof(hosts[i]));
545 ports[i] = (int) port;
546 }
547
548 if (self < 0) {
549 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
550 "brix_cache_peers: mark this node's own ring slot with "
551 "self=host:port");
552 return NGX_ERROR;
553 }

561/* Parse the cache_store URL and record its tier cfg + read-through policy on the
562 * backend registry. Split out of brix_tier_register_stores so each function's
563 * branching stays within the readability gate. Operator errors are [emerg]. */
564static ngx_int_t
565brix_tier_register_cache_store(ngx_conf_t *cf,
566 ngx_http_brix_shared_conf_t *common)
567{
568 char err[256];
569 brix_tier_cfg_t cfg;
570 brix_cache_policy_t pol;
571 brix_tier_parse_t tp = { cf, &cfg, err, sizeof(err) };
572
573 if (brix_tier_parse_store(&tp, &common->cache_store,
574 common->cache_store_args, BRIX_TIER_CACHE) != NGX_OK)
575 {
576 return NGX_ERROR; /* [emerg] already logged */
577 }
578 ngx_memzero(&pol, sizeof(pol));
579 pol.enabled = 1;
580 pol.max_file_size = common->cache_max_object;
581 /* Read-fill admission (bridged from the srv conf at finalisation): deny/allow
582 * prefixes + include regex gate the composable sd_cache fill for parity with
583 * write-through and the legacy cache_origin admit. NULL => no filter. */
584 pol.deny_prefixes = common->cache_deny_prefixes;
585 pol.allow_prefixes = common->cache_allow_prefixes;
586 pol.include_regex = common->cache_include_re;
587 pol.evict_at = common->cache_evict_at;
588 pol.evict_to = common->cache_evict_to;
589 pol.meta_mode = (int) common->cache_meta_mode;
590 pol.batch_cinfo = (common->cache_batch_cinfo == 2)
591 ? -1 : (int) common->cache_batch_cinfo;
592 pol.l1_entries = common->cache_index_cache;
593 pol.slice_size = common->cache_slice_size;
594 /* phase-68: digest verification on fill (cvmfs-cas today). The verify
595 * runs on the staged temp BEFORE commit, which needs the store's
596 * staged_path — a local posix store; reject other stores loudly. */
597 pol.verify = (common->cache_verify_mode == NGX_CONF_UNSET_UINT)
598 ? BRIX_CACHE_VERIFY_OFF
599 : (brix_cache_verify_mode_e) common->cache_verify_mode;
600 if (pol.verify == BRIX_CACHE_VERIFY_CVMFS_CAS
601 && ngx_strcmp(cfg.driver, "posix") != 0)
602 {
603 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
604 "brix_cache_verify cvmfs-cas requires a local posix "
605 "cache store (got \"%s\")", cfg.driver);
606 return NGX_ERROR;
607 }
608 pol.cvmfs_manifest_ttl = common->cache_manifest_ttl;
609 pol.cvmfs_offline_ttl = common->cache_offline_ttl;
610 /* phase-85 F1: brix_cvmfs_verify_manifest — load the repo master public
611 * key(s) once at config time; the fill spine verifies every MANIFEST-class
612 * fill's signature chain against it before publish. Same posix-store
613 * constraint as cvmfs-cas (the verify reads the staged part path). */
614 if (common->cache_cvmfs_master_key.len > 0) {
615 if (ngx_strcmp(cfg.driver, "posix") != 0) {
616 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
617 "brix_cvmfs_verify_manifest requires a local posix "
618 "cache store (got \"%s\")", cfg.driver);
619 return NGX_ERROR;
620 }
621 if (brix_tier_load_master_key(cf, &common->cache_cvmfs_master_key,
622 &pol) != NGX_OK)
623 {
624 return NGX_ERROR; /* [emerg] already logged */
625 }
626 }
627 if (common->cache_quarantine_dir.len > 0) {
628 ngx_cpystrn((u_char *) pol.quarantine_dir,
629 common->cache_quarantine_dir.data,
630 ngx_min(common->cache_quarantine_dir.len + 1,
631 sizeof(pol.quarantine_dir)));
632 }
633 brix_vfs_backend_config_cache_store(common->root_canon, &cfg, &pol);
634
635 /* Phase-85 F7: the optional cold tier under the cache — its own store URL,
636 * governed by the hot cache's policy (no separate knobs). */
637 if (common->cache_cold_store.len > 0) {
638 brix_tier_cfg_t ccfg;
639 brix_tier_parse_t ctp = { cf, &ccfg, err, sizeof(err) };
640
641 if (brix_tier_parse_store(&ctp, &common->cache_cold_store,
642 common->cache_cold_store_args, BRIX_TIER_CACHE) != NGX_OK)
643 {
644 return NGX_ERROR; /* [emerg] already logged */
645 }
646 brix_vfs_backend_config_cache_cold_store(common->root_canon, &ccfg);
647 }
648
649 /* Phase-85 F8: the sibling-mesh ring under the cache tier. */
650 if (common->cache_peers != NULL
651 && brix_tier_register_cache_peers(cf, common) != NGX_OK)
652 {
653 return NGX_ERROR; /* [emerg] already logged */
654 }
655 return NGX_OK;
656}

658/* Register the export's phase-64 composable cache/stage tiers (additive over the
659 * storage backend). Parses the cache_store / stage_store URLs (operator errors are
660 * [emerg], failing nginx -t) and records the tier cfg + policy on the backend
661 * registry, which composes the sd_cache / sd_stage decorators per worker. Shared by
662 * all three protocol finalisers (§4.4) - it reads only the common preamble. */
663ngx_int_t
664brix_tier_register_stores(ngx_conf_t *cf, ngx_http_brix_shared_conf_t *common)
665{
666 char err[256];
667
668 /* G8 (P4/§9.4): a nearline (tape) backend is unservable without a cache tier
669 * as the recall target - reject at config time. "frm://" is the tape:// alias. */
670 {
671 const ngx_str_t *sb = &common->storage_backend;
672 int is_nearline =
673 (sb->len > sizeof("tape://") - 1
674 && ngx_strncmp(sb->data, "tape://", sizeof("tape://") - 1) == 0)
675 || (sb->len > sizeof("frm://") - 1
676 && ngx_strncmp(sb->data, "frm://", sizeof("frm://") - 1) == 0);
677
678 if (is_nearline && common->cache_store.len == 0) {
679 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
680 "brix: a \"tape://\"/\"frm://\" backend is nearline and requires "
681 "brix_cache_store (the recall target); add a cache tier");
682 return NGX_ERROR;
683 }
684 }
685
686 /* Phase-85 F7: a cold tier is meaningless without the hot cache it sits
687 * under — reject at config time rather than silently ignoring it. */
688 if (common->cache_cold_store.len > 0 && common->cache_store.len == 0) {
689 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
690 "brix_cache_cold_store requires brix_cache_store (the hot tier)");
691 return NGX_ERROR;
692 }
693
694 /* Phase-85 F8: the sibling mesh fills INTO the cache tier — meaningless
695 * without one. */
696 if (common->cache_peers != NULL && common->cache_store.len == 0) {
697 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
698 "brix_cache_peers requires brix_cache_store (the mesh fills "
699 "the cache tier)");
700 return NGX_ERROR;
701 }
702
703 if (common->cache_store.len > 0
704 && brix_tier_register_cache_store(cf, common) != NGX_OK)
705 {
706 return NGX_ERROR;
707 }
708
709 /* Zero-config gateway staging: a whole-object remote backend (WebDAV/S3)
710 * rejects random-offset writes at the cap, so a WRITABLE export with no
711 * stage tier configured at all gets a brix-managed default store under
712 * /tmp/staging (loud [warn] banner; opt out with "brix_stage off"). An
713 * explicit "brix_stage off" merges to 0, not UNSET, and skips this. */
714 if (common->stage_enable == NGX_CONF_UNSET
715 && common->stage_store.len == 0
716 && common->allow_write == 1
717 && brix_storage_backend_is_whole_object(common))
718 {
719 brix_tier_default_stage_store(cf, common);
720 }
721
722 /* PrivateTmp posture check: any stage/cache store under a systemd-private
723 * /tmp (including the default just provisioned) is wiped on restart. */
724 brix_tier_warn_private_tmp(cf, common);
725
726 if (common->stage_enable == 1) {
727 brix_tier_cfg_t cfg;
728 brix_stage_policy_t spol;
729 brix_tier_parse_t tp = { cf, &cfg, err, sizeof(err) };
730
731 if (common->stage_store.len == 0) {
732 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
733 "brix_stage on requires brix_stage_store");
734 return NGX_ERROR;
735 }
736 if (brix_tier_parse_store(&tp, &common->stage_store,
737 common->stage_store_args, BRIX_TIER_STAGE) != NGX_OK)
738 {
739 return NGX_ERROR;
740 }
741 ngx_memzero(&spol, sizeof(spol));
742 spol.enabled = 1;
743 spol.flush_mode = common->stage_flush_async ? BRIX_WT_MODE_ASYNC
744 : BRIX_WT_MODE_SYNC;
745 brix_vfs_backend_config_stage_store(common->root_canon, &cfg, &spol);
746 }
747
748 return NGX_OK;
749}
