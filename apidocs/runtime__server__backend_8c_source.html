19/* Directive setter for a tier store-URL directive: arg[1] = the store URL (into the
20 * ngx_str_t at cmd->offset); args[2..] = trailing credential=/block_size= params
21 * (into the ngx_array_t* at the field offset carried in cmd->post). See header. */
22char *
23brix_conf_set_store_slot(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
24{
25 char *p = conf;
26 ngx_str_t *url = (ngx_str_t *) (p + cmd->offset);
27 ngx_array_t **args = (ngx_array_t **) (p + (uintptr_t) cmd->post);
28 ngx_str_t *value = cf->args->elts;
29 ngx_uint_t i;
30
31 if (url->data != NULL) {
32 return "is duplicate";
33 }
34 *url = value[1]; /* the store URL token */
35
36 if (cf->args->nelts > 2) {
37 *args = ngx_array_create(cf->pool, cf->args->nelts - 2, sizeof(ngx_str_t));
38 if (*args == NULL) {
39 return NGX_CONF_ERROR;
40 }
41 for (i = 2; i < cf->args->nelts; i++) {
42 ngx_str_t *a = ngx_array_push(*args);
43
44 if (a == NULL) {
45 return NGX_CONF_ERROR;
46 }
47 *a = value[i]; /* "credential=..." / "block_size=..." */
48 }
49 }
50 return NGX_CONF_OK;
51}

53/* A LOCAL storage backend NAMES THE EXPORT TREE — the fully composable replacement
54 * for brix_root. Rewrites common->root from the backend URL and anchors root_canon
55 * there. No-op for a remote/non-local backend (root://, tape://, http://, a cache
56 * origin, or none). Shared by all three protocol finalisers; called BEFORE the
57 * export-root prep. Two forms:
58 * posix:<path> → root = <path>; CLEAR the backend (default POSIX driver).
59 * pblock://<path> → root = /<path> (one leading '/' guaranteed: "pblock://x"→
60 * "/x", "pblock:///abs"→"//abs"→/abs); KEEP the "pblock"
61 * driver; create the block-store directory on init if needed. */
62void
63brix_storage_backend_posix_root(ngx_http_brix_shared_conf_t *common)
64{
65 ngx_str_t *sb = &common->storage_backend;
66
67 if (sb->len > sizeof("posix:") - 1
68 && ngx_strncmp(sb->data, "posix:", sizeof("posix:") - 1) == 0)
69 {
70 u_char *p = sb->data + (sizeof("posix:") - 1);
71 size_t plen = sb->len - (sizeof("posix:") - 1);
72
73 /* Accept both posix:<path> and posix://<path>: collapse leading "//" runs to
74 * one '/' so the root is a single canonical path (a stray double slash
75 * propagates into common->root and breaks raw prefix-strip path uses).
76 * "posix://abs" with an absolute path yields "///abs" → "/abs". */
77 while (plen >= 2 && p[0] == '/' && p[1] == '/') {
78 p++;
79 plen--;
80 }
81 common->root.data = p;
82 common->root.len = plen;
83 sb->len = 0; /* default POSIX driver */
84 return;
85 }
86
87 if (sb->len > sizeof("pblock://") - 1
88 && ngx_strncmp(sb->data, "pblock://", sizeof("pblock://") - 1) == 0)
89 {
90 size_t base = sizeof("pblock://") - 1; /* past "pblock://" */
91
92 /* Yield EXACTLY one leading '/': "pblock:///abs" already has it (offset
93 * `base`); "pblock://rel" gains it by keeping the prior '/' (offset base-1).
94 * A double slash would break the write-through's prefix-strip path derivation. */
95 if (sb->data[base] == '/') {
96 common->root.data = sb->data + base;
97 common->root.len = sb->len - base;
98 } else {
99 common->root.data = sb->data + base - 1;
100 common->root.len = sb->len - base + 1;
101 }
102 sb->len = sizeof("pblock") - 1; /* the bare "pblock" driver */
103 (void) brix_mkdir_recursive((const char *) common->root.data, 0755);
104 }
105}

107/* 1 iff the storage backend is REMOTE (the export's bytes live off-box: a root://
108 * origin, http(s)://, s3://, tape://, ceph). For such an export the LOCAL root_canon
109 * is only a namespace anchor — never written — so its export-root prep must not
110 * demand W_OK (a pure remote-backed node defaults root_canon to "/", which is not
111 * writable). Local backends (posix/pblock, or none) return 0 and keep the W_OK
112 * check. Called after brix_storage_backend_posix_root (posix:/pblock:// already
113 * rewritten away). */
114int
115brix_storage_backend_is_remote(const ngx_http_brix_shared_conf_t *common)
116{
117 static const char *const schemes[] = {
118 "root://", "roots://", "http://", "https://", "s3://",
119 "tape://", "frm://", "rados://", "ceph:", "cephfsro:", NULL
120 };
121 const ngx_str_t *sb = &common->storage_backend;
122 int i;
123
124 for (i = 0; schemes[i] != NULL; i++) {
125 size_t n = ngx_strlen(schemes[i]);
126
127 if (sb->len >= n && ngx_strncmp(sb->data, schemes[i], n) == 0) {
128 return 1;
129 }
130 }
131 return 0;
132}

134/* Parse the cache_store URL and record its tier cfg + read-through policy on the
135 * backend registry. Split out of brix_tier_register_stores so each function's
136 * branching stays within the readability gate. Operator errors are [emerg]. */
137static ngx_int_t
138brix_tier_register_cache_store(ngx_conf_t *cf,
139 ngx_http_brix_shared_conf_t *common)
140{
141 char err[256];
142 brix_tier_cfg_t cfg;
143 brix_cache_policy_t pol;
144 brix_tier_parse_t tp = { cf, &cfg, err, sizeof(err) };
145
146 if (brix_tier_parse_store(&tp, &common->cache_store,
147 common->cache_store_args, BRIX_TIER_CACHE) != NGX_OK)
148 {
149 return NGX_ERROR; /* [emerg] already logged */
150 }
151 ngx_memzero(&pol, sizeof(pol));
152 pol.enabled = 1;
153 pol.max_file_size = common->cache_max_object;
154 pol.evict_at = common->cache_evict_at;
155 pol.evict_to = common->cache_evict_to;
156 pol.meta_mode = (int) common->cache_meta_mode;
157 pol.batch_cinfo = (common->cache_batch_cinfo == 2)
158 ? -1 : (int) common->cache_batch_cinfo;
159 pol.l1_entries = common->cache_index_cache;
160 pol.slice_size = common->cache_slice_size;
161 /* phase-68: digest verification on fill (cvmfs-cas today). The verify
162 * runs on the staged temp BEFORE commit, which needs the store's
163 * staged_path — a local posix store; reject other stores loudly. */
164 pol.verify = (common->cache_verify_mode == NGX_CONF_UNSET_UINT)
165 ? BRIX_CACHE_VERIFY_OFF
166 : (brix_cache_verify_mode_e) common->cache_verify_mode;
167 if (pol.verify == BRIX_CACHE_VERIFY_CVMFS_CAS
168 && ngx_strcmp(cfg.driver, "posix") != 0)
169 {
170 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
171 "brix_cache_verify cvmfs-cas requires a local posix "
172 "cache store (got \"%s\")", cfg.driver);
173 return NGX_ERROR;
174 }
175 pol.cvmfs_manifest_ttl = common->cache_manifest_ttl;
176 if (common->cache_quarantine_dir.len > 0) {
177 ngx_cpystrn((u_char *) pol.quarantine_dir,
178 common->cache_quarantine_dir.data,
179 ngx_min(common->cache_quarantine_dir.len + 1,
180 sizeof(pol.quarantine_dir)));
181 }
182 brix_vfs_backend_config_cache_store(common->root_canon, &cfg, &pol);
183 return NGX_OK;
184}

186/* Register the export's phase-64 composable cache/stage tiers (additive over the
187 * storage backend). Parses the cache_store / stage_store URLs (operator errors are
188 * [emerg], failing nginx -t) and records the tier cfg + policy on the backend
189 * registry, which composes the sd_cache / sd_stage decorators per worker. Shared by
190 * all three protocol finalisers (§4.4) - it reads only the common preamble. */
191ngx_int_t
192brix_tier_register_stores(ngx_conf_t *cf, ngx_http_brix_shared_conf_t *common)
193{
194 char err[256];
195
196 /* G8 (P4/§9.4): a nearline (tape) backend is unservable without a cache tier
197 * as the recall target - reject at config time. "frm://" is the tape:// alias. */
198 {
199 const ngx_str_t *sb = &common->storage_backend;
200 int is_nearline =
201 (sb->len > sizeof("tape://") - 1
202 && ngx_strncmp(sb->data, "tape://", sizeof("tape://") - 1) == 0)
203 || (sb->len > sizeof("frm://") - 1
204 && ngx_strncmp(sb->data, "frm://", sizeof("frm://") - 1) == 0);
205
206 if (is_nearline && common->cache_store.len == 0) {
207 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
208 "brix: a \"tape://\"/\"frm://\" backend is nearline and requires "
209 "brix_cache_store (the recall target); add a cache tier");
210 return NGX_ERROR;
211 }
212 }
213
214 if (common->cache_store.len > 0
215 && brix_tier_register_cache_store(cf, common) != NGX_OK)
216 {
217 return NGX_ERROR;
218 }
219
220 if (common->stage_enable == 1) {
221 brix_tier_cfg_t cfg;
222 brix_stage_policy_t spol;
223 brix_tier_parse_t tp = { cf, &cfg, err, sizeof(err) };
224
225 if (common->stage_store.len == 0) {
226 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
227 "brix_stage on requires brix_stage_store");
228 return NGX_ERROR;
229 }
230 if (brix_tier_parse_store(&tp, &common->stage_store,
231 common->stage_store_args, BRIX_TIER_STAGE) != NGX_OK)
232 {
233 return NGX_ERROR;
234 }
235 ngx_memzero(&spol, sizeof(spol));
236 spol.enabled = 1;
237 spol.flush_mode = common->stage_flush_async ? BRIX_WT_MODE_ASYNC
238 : BRIX_WT_MODE_SYNC;
239 brix_vfs_backend_config_stage_store(common->root_canon, &cfg, &spol);
240 }
241
242 return NGX_OK;
243}
