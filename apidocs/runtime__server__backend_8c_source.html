23/* brix_pblock_write_opts_sidecar — persist a pblock `?tail` query string as the
24 * one-line <root>/pblock.opts sidecar the pblock driver parses at instance init
25 * (Phase-83 static opts). Best-effort at config finalise: the root dir already
26 * exists (mkdir ran first). Kept here (not in the sqlite-gated pblock driver) so
27 * it links regardless of BRIX_HAVE_SQLITE. */
28static void
29brix_pblock_write_opts_sidecar(const char *root, const char *tail)
30{
31 char path[4096];
32 FILE *f;
33
34 (void) snprintf(path, sizeof(path), "%s/pblock.opts", root);
35 f = fopen(path, "we");
36 if (f == NULL) {
37 return;
38 }
39 (void) fputs(tail != NULL ? tail : "", f);
40 (void) fputc('\n', f);
41 (void) fclose(f);
42}

44/* Directive setter for a tier store-URL directive: arg[1] = the store URL (into the
45 * ngx_str_t at cmd->offset); args[2..] = trailing credential=/block_size= params
46 * (into the ngx_array_t* at the field offset carried in cmd->post). See header. */
47char *
48brix_conf_set_store_slot(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
49{
50 char *p = conf;
51 ngx_str_t *url = (ngx_str_t *) (p + cmd->offset);
52 ngx_array_t **args = (ngx_array_t **) (p + (uintptr_t) cmd->post);
53 ngx_str_t *value = cf->args->elts;
54 ngx_uint_t i;
55
56 if (url->data != NULL) {
57 return "is duplicate";
58 }
59 *url = value[1]; /* the store URL token */
60
61 if (cf->args->nelts > 2) {
62 *args = ngx_array_create(cf->pool, cf->args->nelts - 2, sizeof(ngx_str_t));
63 if (*args == NULL) {
64 return NGX_CONF_ERROR;
65 }
66 for (i = 2; i < cf->args->nelts; i++) {
67 ngx_str_t *a = ngx_array_push(*args);
68
69 if (a == NULL) {
70 return NGX_CONF_ERROR;
71 }
72 *a = value[i]; /* "credential=..." / "block_size=..." */
73 }
74 }
75 return NGX_CONF_OK;
76}

78/* A LOCAL storage backend NAMES THE EXPORT TREE — the fully composable replacement
79 * for brix_root. Rewrites common->root from the backend URL and anchors root_canon
80 * there. No-op for a remote/non-local backend (root://, tape://, http://, a cache
81 * origin, or none). Shared by all three protocol finalisers; called BEFORE the
82 * export-root prep. Two forms:
83 * posix:<path> → root = <path>; CLEAR the backend (default POSIX driver).
84 * pblock://<path> → root = /<path> (one leading '/' guaranteed: "pblock://x"→
85 * "/x", "pblock:///abs"→"//abs"→/abs); KEEP the "pblock"
86 * driver; create the block-store directory on init if needed. */
87void
88brix_storage_backend_posix_root(ngx_http_brix_shared_conf_t *common)
89{
90 ngx_str_t *sb = &common->storage_backend;
91
92 if (sb->len > sizeof("posix:") - 1
93 && ngx_strncmp(sb->data, "posix:", sizeof("posix:") - 1) == 0)
94 {
95 u_char *p = sb->data + (sizeof("posix:") - 1);
96 size_t plen = sb->len - (sizeof("posix:") - 1);
97
98 /* Accept both posix:<path> and posix://<path>: collapse leading "//" runs to
99 * one '/' so the root is a single canonical path (a stray double slash
100 * propagates into common->root and breaks raw prefix-strip path uses).
101 * "posix://abs" with an absolute path yields "///abs" → "/abs". */
102 while (plen >= 2 && p[0] == '/' && p[1] == '/') {
103 p++;
104 plen--;
105 }
106 common->root.data = p;
107 common->root.len = plen;
108 sb->len = 0; /* default POSIX driver */
109 return;
110 }
111
112 if (sb->len > sizeof("pblock://") - 1
113 && ngx_strncmp(sb->data, "pblock://", sizeof("pblock://") - 1) == 0)
114 {
115 size_t base = sizeof("pblock://") - 1; /* past "pblock://" */
116
117 /* Yield EXACTLY one leading '/': "pblock:///abs" already has it (offset
118 * `base`); "pblock://rel" gains it by keeping the prior '/' (offset base-1).
119 * A double slash would break the write-through's prefix-strip path derivation. */
120 if (sb->data[base] == '/') {
121 common->root.data = sb->data + base;
122 common->root.len = sb->len - base;
123 } else {
124 common->root.data = sb->data + base - 1;
125 common->root.len = sb->len - base + 1;
126 }
127 sb->len = sizeof("pblock") - 1; /* the bare "pblock" driver */
128
129 /* Phase-83 static opts: a `?tail` query string on the root
130 * (pblock:///srv/x?lab=1&caps=-sendfile) selects the lab gate + caps/mem
131 * knobs. Strip it BEFORE mkdir/root_canon (which would otherwise create a
132 * literal '?' directory / canonicalise it away) by NUL-terminating the
133 * root at '?', then stash the tail as the <root>/pblock.opts sidecar the
134 * driver reads at init. No tail ⇒ no sidecar ⇒ lab OFF (production path). */
135 {
136 u_char *q = memchr(common->root.data, '?', common->root.len);
137
138 if (q != NULL) {
139 const char *tail = (const char *) (q + 1);
140
141 *q = '\0'; /* truncate root at '?' */
142 common->root.len = (size_t) (q - common->root.data);
143 (void) brix_mkdir_recursive((const char *) common->root.data,
144 0755);
145 brix_pblock_write_opts_sidecar(
146 (const char *) common->root.data, tail);
147 } else {
148 (void) brix_mkdir_recursive((const char *) common->root.data,
149 0755);
150 }
151 }
152 }
153}

155/* 1 iff the storage backend is REMOTE (the export's bytes live off-box: a root://
156 * origin, http(s)://, s3://, tape://, ceph). For such an export the LOCAL root_canon
157 * is only a namespace anchor — never written — so its export-root prep must not
158 * demand W_OK (a pure remote-backed node defaults root_canon to "/", which is not
159 * writable). Local backends (posix/pblock, or none) return 0 and keep the W_OK
160 * check. Called after brix_storage_backend_posix_root (posix:/pblock:// already
161 * rewritten away). */
162int
163brix_storage_backend_is_remote(const ngx_http_brix_shared_conf_t *common)
164{
165 static const char *const schemes[] = {
166 "root://", "roots://", "http://", "https://", "s3://",
167 "tape://", "frm://", "rados://", "ceph:", "cephfsro:", NULL
168 };
169 const ngx_str_t *sb = &common->storage_backend;
170 int i;
171
172 for (i = 0; schemes[i] != NULL; i++) {
173 size_t n = ngx_strlen(schemes[i]);
174
175 if (sb->len >= n && ngx_strncmp(sb->data, schemes[i], n) == 0) {
176 return 1;
177 }
178 }
179 return 0;
180}

182/* 1 iff the storage backend is a WHOLE-OBJECT remote gateway (WebDAV or S3:
183 * an upload replaces the entire object, so the remote driver cannot honour
184 * random-offset writes and rejects them at the cap). root://, rados and ceph
185 * remotes are byte-addressable and never need a stage tier for random writes;
186 * tape:// / frm:// are nearline and already require a cache tier instead. */
187static int
188brix_storage_backend_is_whole_object(const ngx_http_brix_shared_conf_t *common)
189{
190 static const char *const schemes[] = { "http://", "https://", "s3://",
191 NULL };
192 const ngx_str_t *sb = &common->storage_backend;
193 int i;
194
195 for (i = 0; schemes[i] != NULL; i++) {
196 size_t n = ngx_strlen(schemes[i]);
197
198 if (sb->len >= n && ngx_strncmp(sb->data, schemes[i], n) == 0) {
199 return 1;
200 }
201 }
202 return 0;
203}

205/* Register the export's phase-64 composable cache/stage tiers (additive over the
206 * storage backend). Parses the cache_store / stage_store URLs (operator errors are
207 * [emerg], failing nginx -t) and records the tier cfg + policy on the backend
208 * registry, which composes the sd_cache / sd_stage decorators per worker. Shared by
209 * all three protocol finalisers (§4.4) - it reads only the common preamble. */
210ngx_int_t
211brix_tier_register_stores(ngx_conf_t *cf, ngx_http_brix_shared_conf_t *common)
212{
213 char err[256];
214
215 /* G8 (P4/§9.4): a nearline (tape) backend is unservable without a cache tier
216 * as the recall target - reject at config time. "frm://" is the tape:// alias. */
217 {
218 const ngx_str_t *sb = &common->storage_backend;
219 int is_nearline =
220 (sb->len > sizeof("tape://") - 1
221 && ngx_strncmp(sb->data, "tape://", sizeof("tape://") - 1) == 0)
222 || (sb->len > sizeof("frm://") - 1
223 && ngx_strncmp(sb->data, "frm://", sizeof("frm://") - 1) == 0);
224
225 if (is_nearline && common->cache_store.len == 0) {
226 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
227 "brix: a \"tape://\"/\"frm://\" backend is nearline and requires "
228 "brix_cache_store (the recall target); add a cache tier");
229 return NGX_ERROR;
230 }
231 }
232
233 /* Phase-85 F7: a cold tier is meaningless without the hot cache it sits
234 * under — reject at config time rather than silently ignoring it. */
235 if (common->cache_cold_store.len > 0 && common->cache_store.len == 0) {
236 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
237 "brix_cache_cold_store requires brix_cache_store (the hot tier)");
238 return NGX_ERROR;
239 }
240
241 /* Phase-85 F8: the sibling mesh fills INTO the cache tier — meaningless
242 * without one. */
243 if (common->cache_peers != NULL && common->cache_store.len == 0) {
244 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
245 "brix_cache_peers requires brix_cache_store (the mesh fills "
246 "the cache tier)");
247 return NGX_ERROR;
248 }
249
250 if (common->cache_store.len > 0
251 && brix_tier_register_cache_store(cf, common) != NGX_OK)
252 {
253 return NGX_ERROR;
254 }
255
256 /* Zero-config gateway staging: a whole-object remote backend (WebDAV/S3)
257 * rejects random-offset writes at the cap, so a WRITABLE export with no
258 * stage tier configured at all gets a brix-managed default store under
259 * /tmp/staging (loud [warn] banner; opt out with "brix_stage off"). An
260 * explicit "brix_stage off" merges to 0, not UNSET, and skips this. */
261 if (common->stage_enable == NGX_CONF_UNSET
262 && common->stage_store.len == 0
263 && common->allow_write == 1
264 && brix_storage_backend_is_whole_object(common))
265 {
266 brix_tier_default_stage_store(cf, common);
267 }
268
269 /* PrivateTmp posture check: any stage/cache store under a systemd-private
270 * /tmp (including the default just provisioned) is wiped on restart. */
271 brix_tier_warn_private_tmp(cf, common);
272
273 if (common->stage_enable == 1) {
274 brix_tier_cfg_t cfg;
275 brix_stage_policy_t spol;
276 brix_tier_parse_t tp = { cf, &cfg, err, sizeof(err) };
277
278 if (common->stage_store.len == 0) {
279 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
280 "brix_stage on requires brix_stage_store");
281 return NGX_ERROR;
282 }
283 if (brix_tier_parse_store(&tp, &common->stage_store,
284 common->stage_store_args, BRIX_TIER_STAGE) != NGX_OK)
285 {
286 return NGX_ERROR;
287 }
288 ngx_memzero(&spol, sizeof(spol));
289 spol.enabled = 1;
290 spol.flush_mode = common->stage_flush_async ? BRIX_WT_MODE_ASYNC
291 : BRIX_WT_MODE_SYNC;
292 brix_vfs_backend_config_stage_store(common->root_canon, &cfg, &spol);
293 }
294
295 return NGX_OK;
296}
