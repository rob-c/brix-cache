1/*
2 * xrdcinfo.c — dump a file's unified xmeta metadata record as JSON.
3 *
4 * WHAT: Reads the "user.xrd.cinfo" xattr (or the "<path>.cinfo" sidecar /
5 * an explicit record file) and prints the record as one JSON object:
6 * geometry, present bitmap, dirty/verify state, block-CRC coverage and
7 * cached digests. WHY: operators/tests inspect exactly what a node
8 * recorded for a file without any nginx dependency. HOW: a minimal
9 * self-contained parser of the xmeta layout — stock XrdPfc cinfo v4
10 * prefix (version i32, 48-byte Store POD, crc, bitmap, AStat[], crc)
11 * followed by "XCX1" TLV sections (STATE/DIGEST/BLOCKCRC/ORIGIN).
12 */

39static uint32_t
40rd_u32(const uint8_t *b, size_t off)
41{
42 return (uint32_t) b[off] | ((uint32_t) b[off + 1] << 8)
43 | ((uint32_t) b[off + 2] << 16) | ((uint32_t) b[off + 3] << 24);
44}

46static uint64_t
47rd_u64(const uint8_t *b, size_t off)
48{
49 uint64_t v = 0;
50 int i;
51
52 for (i = 0; i < 8; i++) {
53 v |= (uint64_t) b[off + i] << (8 * i);
54 }
55 return v;
56}

58/* Read the whole record into *buf (malloc'd). Returns byte count, or -1. */
59static ssize_t
60slurp_file(const char *path, uint8_t **buf)
61{
62 FILE *f = fopen(path, "rb");
63 long n;
64 uint8_t *p;
65 size_t got;
66
67 if (f == NULL) {
68 return -1;
69 }
70 if (fseek(f, 0, SEEK_END) != 0) {
71 fclose(f);
72 return -1;
73 }
74 n = ftell(f);
75 if (n < 0) {
76 fclose(f);
77 return -1;
78 }
79 rewind(f);
80 p = malloc((size_t) n + 1);
81 if (p == NULL) {
82 fclose(f);
83 return -1;
84 }
85 got = fread(p, 1, (size_t) n, f);
86 fclose(f);
87 if (got != (size_t) n) {
88 free(p);
89 return -1;
90 }
91 *buf = p;
92 return n;
93}

98{

100 uint8_t *p;

117/*
118 * xmeta_hdr_t — the decoded stock-cinfo v4 prefix (geometry + bitmap window).
119 * WHAT: geometry fields plus the byte offsets the JSON dump needs. WHY: the
120 * header parse and the emitters share exactly this state; passing one struct
121 * keeps the data flow explicit and each emitter pure. HOW: filled once by
122 * parse_stock_header(), then read by the present/flags emitters.
123 */
124typedef struct {
125 int64_t bsize; /* block size */
126 int64_t size; /* logical file size */
127 int64_t no_cksum; /* count of blocks lacking a verified checksum */
128 uint64_t nblocks; /* derived block count */
129 const uint8_t *bm; /* present bitmap (nblocks bits) */
130 size_t ext_off; /* offset of the first extension section */
131} xmeta_hdr_t;
132
133/*
134 * xmeta_state_t — the STATE-section fields the flags emitter consumes.
135 * WHAT: dirty window, POSIX mode, state flag bits, and a presence marker.
136 * WHY: the STATE section is optional; the flags line needs its values (or their
137 * zeroed absence) resolved before it runs. HOW: zero-initialised by the caller,
138 * populated by emit_ext_state() when a STATE section is present.
139 */
140typedef struct {
141 uint64_t dirty_lo;
142 uint64_t dirty_hi;
143 uint32_t mode;
144 uint32_t flags;
145 int have_state;
146} xmeta_state_t;

148/*
149 * parse_stock_header — validate and decode the stock v4 cinfo prefix.
150 * WHAT: checks magic/geometry bounds and fills *h with block geometry, the
151 * bitmap pointer and the extension offset. WHY: isolates all bounds checks in
152 * one pure step so the dumper orchestrator stays flat. HOW: returns 0 on
153 * success, or the process exit code (2) on any malformed field — the caller
154 * emits the frozen "bad_magic" object on non-zero.
155 */
156static int
157parse_stock_header(const uint8_t *buf, size_t n, xmeta_hdr_t *h)
158{
159 int32_t astatn;
160 size_t bmlen, off;
161
162 if (n < 56 || (int32_t) rd_u32(buf, 0) != XMETA_STOCK_VERSION) {
163 return 2;
164 }
165 h->bsize = (int64_t) rd_u64(buf, 4);
166 h->size = (int64_t) rd_u64(buf, 12);
167 h->no_cksum = (int64_t) rd_u64(buf, 28);
168 astatn = (int32_t) rd_u32(buf, 48);
169 if (h->bsize <= 0 || h->size < 0 || astatn < 0) {
170 return 2;
171 }
172 h->nblocks = (h->size > 0)
173 ? ((uint64_t) h->size + (uint64_t) h->bsize - 1) / (uint64_t) h->bsize
174 : 0;
175 bmlen = (size_t) ((h->nblocks + 7) / 8);
176 off = 4 + 48 + 4; /* version + Store + crc */
177 if (off + bmlen + (size_t) astatn * 56 + 4 > n) {
178 return 2;
179 }
180 h->bm = buf + off;
181 h->ext_off = off + bmlen + (size_t) astatn * 56 + 4; /* + AStat[] + crc */
182 return 0;
183}

185/*
186 * emit_geometry — print the leading geometry keys of the JSON object.
187 * WHAT: opens the object and emits version/block_size/size/nblocks. WHY: the
188 * fixed prefix is a single concern separate from the variable bitmap/sections.
189 * HOW: byte-identical to the original leading printf; side effect only.
190 */
191static void
192emit_geometry(const xmeta_hdr_t *h)
193{
194 printf("{\"version\":4,\"block_size\":%lld,\"size\":%lld,"
195 "\"nblocks\":%llu,", (long long) h->bsize, (long long) h->size,
196 (unsigned long long) h->nblocks);
197}

199/*
200 * emit_present_bitmap — print the present_blocks array and completeness keys.
201 * WHAT: walks the bitmap, emits each present block index, then present_count
202 * and complete. WHY: the present-bitmap emit is one self-contained section.
203 * HOW: returns the present-block count so the flags emitter can classify
204 * complete/partial without rescanning; output bytes are frozen.
205 */
206static uint64_t
207emit_present_bitmap(const xmeta_hdr_t *h)
208{
209 uint64_t present = 0, b;
210 const char *sep = "";
211
212 printf("\"present_blocks\":[");
213 for (b = 0; b < h->nblocks; b++) {
214 if (h->bm[b / 8] & (1u << (b % 8))) {
215 printf("%s%llu", sep, (unsigned long long) b);
216 sep = ",";
217 present++;
218 }
219 }
220 printf("],\"present_count\":%llu,\"complete\":%s",
221 (unsigned long long) present,
222 (h->nblocks == 0 || present == h->nblocks) ? "true" : "false");
223 return present;
224}

226/*
227 * emit_ext_state — handle a STATE (SEC_STATE) extension section.
228 * WHAT: records dirty/mode/flags into *st and emits origin_mtime/flush_gen.
229 * WHY: STATE feeds both the JSON stream (mtime/gen) and the later flags line
230 * (dirty/verified/mode). HOW: pure record + one frozen printf; only acts when
231 * the payload is long enough, matching the original guard.
232 */
233static void
234emit_ext_state(const uint8_t *buf, size_t pay, uint32_t plen,
235 xmeta_state_t *st)
236{
237 if (plen < 80) {
238 return;
239 }
240 st->dirty_lo = rd_u64(buf, pay + 8);
241 st->dirty_hi = rd_u64(buf, pay + 16);
242 st->mode = rd_u32(buf, pay + 72);
243 st->flags = rd_u32(buf, pay + 76);
244 st->have_state = 1;
245 printf(",\"origin_mtime\":%llu,\"flush_gen\":%llu",
246 (unsigned long long) rd_u64(buf, pay),
247 (unsigned long long) rd_u64(buf, pay + 24));
248}

250/*
251 * emit_ext_blockcrc — handle a BLOCKCRC (SEC_BLOCKCRC) extension section.
252 * WHAT: counts the non-zero per-block CRC slots and emits the blockcrc object.
253 * WHY: block-CRC coverage is its own JSON key. HOW: bounds-checked scan against
254 * n, then one frozen printf; only acts on a payload of the minimum size.
255 */
256static void
257emit_ext_blockcrc(const uint8_t *buf, size_t n, size_t pay, uint32_t plen)
258{
259 uint64_t nb, set = 0, k;
260
261 if (plen < 16) {
262 return;
263 }
264 nb = rd_u64(buf, pay + 8);
265 for (k = 0; k < nb && pay + 16 + k * 4 + 4 <= n; k++) {
266 if (rd_u32(buf, pay + 16 + (size_t) k * 4) != 0) {
267 set++;
268 }
269 }
270 printf(",\"blockcrc\":{\"granule\":%u,\"nblocks\":%llu,"
271 "\"computed\":%llu}", rd_u32(buf, pay),
272 (unsigned long long) nb, (unsigned long long) set);
273}

275/*
276 * emit_digest_value — print one digest value as an escaped JSON string body.
277 * WHAT: walks vl bytes at buf[base] emitting printable ASCII verbatim and
278 * escaping everything else as \uXXXX. WHY: the escape loop is the reusable core
279 * of the digest emitter. HOW: putchar/printf side effects only; bytes frozen.
280 */
281static void
282emit_digest_value(const uint8_t *buf, size_t base, uint16_t vl)
283{
284 size_t j;
285
286 for (j = 0; j < vl; j++) {
287 uint8_t ch = buf[base + j];
288
289 if (ch >= 0x20 && ch < 0x7f && ch != '"' && ch != '\\') {
290 putchar(ch);
291 } else {
292 printf("\\u%04x", ch);
293 }
294 }
295}

297/*
298 * emit_ext_digest — handle a DIGEST (SEC_DIGEST) extension section.
299 * WHAT: emits the digests array, one {alg,value} object per TLV entry. WHY:
300 * cached digests are their own JSON key with a variable-length entry list.
301 * HOW: iterates the alg/len/value triples within the payload window, delegating
302 * value escaping to emit_digest_value; output bytes are frozen.
303 */
304static void
305emit_ext_digest(const uint8_t *buf, size_t pay, uint32_t plen)
306{
307 size_t d = pay, dend = pay + plen;
308 const char *sep = "";
309
310 printf(",\"digests\":[");
311 while (d + 4 <= dend) {
312 uint16_t alg = rd_u16(buf, d);
313 uint16_t vl = rd_u16(buf, d + 2);
314
315 if (d + 4 + vl > dend) {
316 break;
317 }
318 printf("%s{\"alg\":%u,\"value\":\"", sep, alg);
319 emit_digest_value(buf, d + 4, vl);
320 printf("\"}");
321 sep = ",";
322 d += 4 + (size_t) vl;
323 }
324 printf("]");
325}

327/*
328 * emit_ext_sections — walk the "XCX1" TLV extension block, if present.
329 * WHAT: dispatches each section to its type-specific emitter and collects STATE
330 * fields into *st. WHY: the section loop is a distinct concern from the header
331 * and flags emitters, and its per-type dispatch reads cleanly as a switch.
332 * HOW: bounds-checked TLV walk identical to the original; no output when the
333 * block is absent.
334 */
335static void
336emit_ext_sections(const uint8_t *buf, size_t n, const xmeta_hdr_t *h,
337 xmeta_state_t *st)
338{
339 size_t off = h->ext_off;
340 uint16_t nsec, i;
341
342 if (off + 8 > n || rd_u32(buf, off) != XMETA_EXT_MAGIC) {
343 return;
344 }
345 nsec = rd_u16(buf, off + 6);
346 off += 8;
347 for (i = 0; i < nsec && off + 12 <= n; i++) {
348 uint16_t type = rd_u16(buf, off);
349 uint32_t plen = rd_u32(buf, off + 4);
350 size_t pay = off + 8;
351
352 if (pay + plen + 4 > n) {
353 break;
354 }
355 switch (type) {
356 case SEC_STATE:
357 emit_ext_state(buf, pay, plen, st);
358 break;
359 case SEC_BLOCKCRC:
360 emit_ext_blockcrc(buf, n, pay, plen);
361 break;
362 case SEC_DIGEST:
363 emit_ext_digest(buf, pay, plen);
364 break;
365 default:
366 break;
367 }
368 off = pay + plen + 4;
369 }
370}

372/*
373 * emit_flags — print the trailing flags array and mode, closing the object.
374 * WHAT: derives COMPLETE/PARTIAL/DIRTY/VERIFIED/UNVERIFIED_BLOCKS from the
375 * decoded header + state and emits them with the mode. WHY: the flags line is
376 * the final derived summary and belongs in one place. HOW: same guard order and
377 * separator handling as the original; output bytes frozen.
378 */
379static void
380emit_flags(const xmeta_hdr_t *h, const xmeta_state_t *st, uint64_t present)
381{
382 const char *sep = "";
383
384 printf(",\"flags\":[");
385 if (h->nblocks == 0 || present == h->nblocks) {
386 printf("%s\"COMPLETE\"", sep); sep = ",";
387 } else if (present > 0) {
388 printf("%s\"PARTIAL\"", sep); sep = ",";
389 }
390 if (st->have_state && st->dirty_lo < st->dirty_hi) {
391 printf("%s\"DIRTY\"", sep); sep = ",";
392 }
393 if (st->flags & F_VERIFIED) {
394 printf("%s\"VERIFIED\"", sep); sep = ",";
395 }
396 if (h->no_cksum != 0) {
397 printf("%s\"UNVERIFIED_BLOCKS\"", sep); sep = ",";
398 }
399 printf("],\"mode\":%u}\n", st->mode);
400}

402/*
403 * dump_record — emit the JSON dump for a slurped record; return the exit code.
404 * WHAT: orchestrates header parse → geometry → present bitmap → extension
405 * sections → flags. WHY: keeps the top-level dumper a flat, readable sequence of
406 * named steps. HOW: on a malformed header emits the frozen "bad_magic" object
407 * and returns 2; otherwise composes the emitters and returns 0.
408 */
409static int
410dump_record(const uint8_t *buf, size_t n)
411{
412 xmeta_hdr_t h = {0};
413 xmeta_state_t st = {0};
414 uint64_t present;
415
416 if (parse_stock_header(buf, n, &h) != 0) {
417 printf("{\"error\":\"bad_magic\"}\n");
418 return 2;
419 }
420 emit_geometry(&h);
421 present = emit_present_bitmap(&h);
422 emit_ext_sections(buf, n, &h, &st);
423 emit_flags(&h, &st, present);
424 return 0;
425}

428int
429brix_xrdcinfo_main(int argc, char **argv)
430{
431 const char *path = NULL;
432 int use_xattr = 0;
433 int i;
434 uint8_t *buf = NULL;
435 ssize_t n;
436 int rc;
437
438 /* xrdcinfo does not include brix.h — bring in version.h directly. */
439 if (argc >= 2) {
440 if (strcmp(argv[1], "--version") == 0) {
441 /* version.h included at file top (see include block below) */
442 printf("%s (BriX-Cache client) %s\n", brix_prog_base(argv[0]),
443 brix_client_version());
444 return 0;
445 }
446 if (strcmp(argv[1], "--help") == 0 || strcmp(argv[1], "-h") == 0) {
447 printf("usage: %s [--xattr] <path>\n"
448 " dump a proxy-cache .cinfo / xmeta xattr as JSON\n",
449 brix_prog_base(argv[0]));
450 brix_usage_footer(stdout, argv[0]);
451 return 0;
452 }
453 }
454
455 for (i = 1; i < argc; i++) {
456 if (strcmp(argv[i], "--xattr") == 0) {
457 use_xattr = 1;
458 } else {
459 path = argv[i];
460 }
461 }
462 if (path == NULL) {
463 fprintf(stderr, "usage: %s [--xattr] <path>\n", brix_prog_base(argv[0]));
464 return 4;
465 }
466
467 /* carrier chain: the data file's xattr, then its "<path>.cinfo"
468 * sidecar, then the path itself as a raw record file. --xattr forces
469 * the first form only. */
470 n = slurp_xattr(path, &buf);
471 if (n < 0 && !use_xattr) {
472 char sc[4096];
473 int k = snprintf(sc, sizeof(sc), "%s.cinfo", path);
474
475 if (k > 0 && (size_t) k < sizeof(sc)) {
476 n = slurp_file(sc, &buf);
477 }
478 if (n < 0) {
479 n = slurp_file(path, &buf);
480 }
481 }
482 if (n < 0) {
483 printf("{\"absent\":true}\n");
484 return 3;
485 }
486 rc = dump_record(buf, (size_t) n);
487 free(buf);
488 return rc;
489}
