1/*
2 * xrdcinfo.c — dump a file's unified xmeta metadata record as JSON.
3 *
4 * WHAT: Reads the "user.xrd.cinfo" xattr (or the "<path>.cinfo" sidecar /
5 * an explicit record file) and prints the record as one JSON object:
6 * geometry, present bitmap, dirty/verify state, block-CRC coverage and
7 * cached digests. WHY: operators/tests inspect exactly what a node
8 * recorded for a file without any nginx dependency. HOW: a minimal
9 * self-contained parser of the xmeta layout — stock XrdPfc cinfo v4
10 * prefix (version i32, 48-byte Store POD, crc, bitmap, AStat[], crc)
11 * followed by "XCX1" TLV sections (STATE/DIGEST/BLOCKCRC/ORIGIN).
12 */

32static uint16_t
33rd_u16(const uint8_t *b, size_t off)
34{
35 return (uint16_t) (b[off] | (b[off + 1] << 8));
36}

38static uint32_t
39rd_u32(const uint8_t *b, size_t off)
40{
41 return (uint32_t) b[off] | ((uint32_t) b[off + 1] << 8)
42 | ((uint32_t) b[off + 2] << 16) | ((uint32_t) b[off + 3] << 24);
43}

45static uint64_t
46rd_u64(const uint8_t *b, size_t off)
47{
48 uint64_t v = 0;
49 int i;
50
51 for (i = 0; i < 8; i++) {
52 v |= (uint64_t) b[off + i] << (8 * i);
53 }
54 return v;
55}

57/* Read the whole record into *buf (malloc'd). Returns byte count, or -1. */
58static ssize_t
59slurp_file(const char *path, uint8_t **buf)
60{
61 FILE *f = fopen(path, "rb");
62 long n;
63 uint8_t *p;
64 size_t got;
65
66 if (f == NULL) {
67 return -1;
68 }
69 if (fseek(f, 0, SEEK_END) != 0) {
70 fclose(f);
71 return -1;
72 }
73 n = ftell(f);
74 if (n < 0) {
75 fclose(f);
76 return -1;
77 }
78 rewind(f);
79 p = malloc((size_t) n + 1);
80 if (p == NULL) {
81 fclose(f);
82 return -1;
83 }
84 got = fread(p, 1, (size_t) n, f);
85 fclose(f);
86 if (got != (size_t) n) {
87 free(p);
88 return -1;
89 }
90 *buf = p;
91 return n;
92}

95static ssize_t
96slurp_xattr(const char *path, uint8_t **buf)
97{
98 ssize_t n = getxattr(path, "user.xrd.cinfo", NULL, 0);
99 uint8_t *p;
100
101 if (n < 0) {
102 return -1;
103 }
104 p = malloc((size_t) n + 1);
105 if (p == NULL) {
106 return -1;
107 }
108 if (getxattr(path, "user.xrd.cinfo", p, (size_t) n) != n) {
109 free(p);
110 return -1;
111 }
112 *buf = p;
113 return n;
114}

116/* Emit the JSON dump for a slurped record; return the process exit code. */
117static int
118dump_record(const uint8_t *buf, size_t n)
119{
120 int64_t bsize, size, no_cksum;
121 int32_t astatn;
122 uint64_t nblocks, present = 0, b;
123 size_t bmlen, off;
124 const uint8_t *bm;
125 const char *sep;
126 uint64_t dirty_lo = 0, dirty_hi = 0;
127 uint32_t mode = 0, state_flags = 0;
128 int have_state = 0;
129
130 if (n < 56 || (int32_t) rd_u32(buf, 0) != XMETA_STOCK_VERSION) {
131 printf("{\"error\":\"bad_magic\"}\n");
132 return 2;
133 }
134 bsize = (int64_t) rd_u64(buf, 4);
135 size = (int64_t) rd_u64(buf, 12);
136 no_cksum = (int64_t) rd_u64(buf, 28);
137 astatn = (int32_t) rd_u32(buf, 48);
138 if (bsize <= 0 || size < 0 || astatn < 0) {
139 printf("{\"error\":\"bad_magic\"}\n");
140 return 2;
141 }
142 nblocks = (size > 0) ? ((uint64_t) size + (uint64_t) bsize - 1)
143 / (uint64_t) bsize : 0;
144 bmlen = (size_t) ((nblocks + 7) / 8);
145 off = 4 + 48 + 4; /* version + Store + crc */
146 if (off + bmlen + (size_t) astatn * 56 + 4 > n) {
147 printf("{\"error\":\"bad_magic\"}\n");
148 return 2;
149 }
150 bm = buf + off;
151 off += bmlen + (size_t) astatn * 56 + 4; /* bitmap + AStat[] + crc */
152
153 printf("{\"version\":4,\"block_size\":%lld,\"size\":%lld,"
154 "\"nblocks\":%llu,", (long long) bsize, (long long) size,
155 (unsigned long long) nblocks);
156
157 printf("\"present_blocks\":[");
158 sep = "";
159 for (b = 0; b < nblocks; b++) {
160 if (bm[b / 8] & (1u << (b % 8))) {
161 printf("%s%llu", sep, (unsigned long long) b);
162 sep = ",";
163 present++;
164 }
165 }
166 printf("],\"present_count\":%llu,\"complete\":%s",
167 (unsigned long long) present,
168 (nblocks == 0 || present == nblocks) ? "true" : "false");
169
170 /* extension sections */
171 if (off + 8 <= n && rd_u32(buf, off) == XMETA_EXT_MAGIC) {
172 uint16_t nsec = rd_u16(buf, off + 6);
173 uint16_t i;
174
175 off += 8;
176 for (i = 0; i < nsec && off + 12 <= n; i++) {
177 uint16_t type = rd_u16(buf, off);
178 uint32_t plen = rd_u32(buf, off + 4);
179 size_t pay = off + 8;
180
181 if (pay + plen + 4 > n) {
182 break;
183 }
184 if (type == SEC_STATE && plen >= 80) {
185 dirty_lo = rd_u64(buf, pay + 8);
186 dirty_hi = rd_u64(buf, pay + 16);
187 mode = rd_u32(buf, pay + 72);
188 state_flags = rd_u32(buf, pay + 76);
189 have_state = 1;
190 printf(",\"origin_mtime\":%llu,\"flush_gen\":%llu",
191 (unsigned long long) rd_u64(buf, pay),
192 (unsigned long long) rd_u64(buf, pay + 24));
193 } else if (type == SEC_BLOCKCRC && plen >= 16) {
194 uint64_t nb = rd_u64(buf, pay + 8), set = 0, k;
195
196 for (k = 0; k < nb && pay + 16 + k * 4 + 4 <= n; k++) {
197 if (rd_u32(buf, pay + 16 + (size_t) k * 4) != 0) {
198 set++;
199 }
200 }
201 printf(",\"blockcrc\":{\"granule\":%u,\"nblocks\":%llu,"
202 "\"computed\":%llu}", rd_u32(buf, pay),
203 (unsigned long long) nb, (unsigned long long) set);
204 } else if (type == SEC_DIGEST) {
205 size_t d = pay, dend = pay + plen;
206
207 printf(",\"digests\":[");
208 sep = "";
209 while (d + 4 <= dend) {
210 uint16_t alg = rd_u16(buf, d);
211 uint16_t vl = rd_u16(buf, d + 2);
212 size_t j;
213
214 if (d + 4 + vl > dend) {
215 break;
216 }
217 printf("%s{\"alg\":%u,\"value\":\"", sep, alg);
218 for (j = 0; j < vl; j++) {
219 uint8_t ch = buf[d + 4 + j];
220 if (ch >= 0x20 && ch < 0x7f && ch != '"'
221 && ch != '\\')
222 {
223 putchar(ch);
224 } else {
225 printf("\\u%04x", ch);
226 }
227 }
228 printf("\"}");
229 sep = ",";
230 d += 4 + (size_t) vl;
231 }
232 printf("]");
233 }
234 off = pay + plen + 4;
235 }
236 }
237
238 printf(",\"flags\":[");
239 sep = "";
240 if (nblocks == 0 || present == nblocks) {
241 printf("%s\"COMPLETE\"", sep); sep = ",";
242 } else if (present > 0) {
243 printf("%s\"PARTIAL\"", sep); sep = ",";
244 }
245 if (have_state && dirty_lo < dirty_hi) {
246 printf("%s\"DIRTY\"", sep); sep = ",";
247 }
248 if (state_flags & F_VERIFIED) {
249 printf("%s\"VERIFIED\"", sep); sep = ",";
250 }
251 if (no_cksum != 0) {
252 printf("%s\"UNVERIFIED_BLOCKS\"", sep); sep = ",";
253 }
254 printf("],\"mode\":%u}\n", mode);
255 return 0;
256}

258/* Real main; dispatched from the xrdcksum multi-call binary (xrdcksum.c). */
259int
260brix_xrdcinfo_main(int argc, char **argv)
261{
262 const char *path = NULL;
263 int use_xattr = 0;
264 int i;
265 uint8_t *buf = NULL;
266 ssize_t n;
267 int rc;
268
269 /* xrdcinfo does not include brix.h — bring in version.h directly. */
270 if (argc >= 2) {
271 if (strcmp(argv[1], "--version") == 0) {
272 /* version.h included at file top (see include block below) */
273 printf("xrdcinfo (BriX-Cache client) %s\n", brix_client_version());
274 return 0;
275 }
276 if (strcmp(argv[1], "--help") == 0 || strcmp(argv[1], "-h") == 0) {
277 printf("usage: xrdcinfo [--xattr] <path>\n"
278 " dump a proxy-cache .cinfo / xmeta xattr as JSON\n"
279 BRIX_USAGE_FOOTER("xrdcinfo"));
280 return 0;
281 }
282 }
283
284 for (i = 1; i < argc; i++) {
285 if (strcmp(argv[i], "--xattr") == 0) {
286 use_xattr = 1;
287 } else {
288 path = argv[i];
289 }
290 }
291 if (path == NULL) {
292 fprintf(stderr, "usage: xrdcinfo [--xattr] <path>\n");
293 return 4;
294 }
295
296 /* carrier chain: the data file's xattr, then its "<path>.cinfo"
297 * sidecar, then the path itself as a raw record file. --xattr forces
298 * the first form only. */
299 n = slurp_xattr(path, &buf);
300 if (n < 0 && !use_xattr) {
301 char sc[4096];
302 int k = snprintf(sc, sizeof(sc), "%s.cinfo", path);
303
304 if (k > 0 && (size_t) k < sizeof(sc)) {
305 n = slurp_file(sc, &buf);
306 }
307 if (n < 0) {
308 n = slurp_file(path, &buf);
309 }
310 }
311 if (n < 0) {
312 printf("{\"absent\":true}\n");
313 return 3;
314 }
315 rc = dump_record(buf, (size_t) n);
316 free(buf);
317 return rc;
318}
