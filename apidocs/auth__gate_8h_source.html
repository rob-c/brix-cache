5/*
6 * brix_auth_gate — three-tier auth check for path-based operations.
7 *
8 * Centralises the repeated authdb → VO ACL → token-scope sequence that
9 * appears in every namespace handler. On the first failing tier the gate
10 * sends a kXR_NotAuthorized wire response, stores the nginx return code in
11 * ctx->write_rc, and returns NGX_DONE. The caller must return ctx->write_rc
12 * immediately:
13 *
14 * if (brix_auth_gate(ctx, c, BRIX_OP_MKDIR, "MKDIR",
15 * reqpath, resolved, conf,
16 * BRIX_AUTH_MKDIR, 1) != NGX_OK) {
17 * return ctx->write_rc;
18 * }
19 *
20 * Parameters:
21 * op_id — BRIX_OP_* constant for metric tracking
22 * op_name — verb for the access log ("MKDIR", "STAT", etc.)
23 * reqpath — client-supplied path; used for token scope check
24 * resolved — canonical path; used for authdb and VO ACL checks
25 * conf — server config (carries vo_rules, authdb pointer)
26 * auth_level — BRIX_AUTH_READ / _LOOKUP / _UPDATE / _DELETE / _MKDIR
27 * need_write — 1 if write token scope is required, 0 for read
28 *
29 * Returns NGX_OK when all tiers pass, NGX_DONE when one denies.
30 */

33 const char *reqpath, const char *resolved,
