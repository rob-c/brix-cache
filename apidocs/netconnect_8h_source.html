4/*
5 * netconnect.h — shared OUTBOUND connect/I/O hardening for blocking-thread paths.
6 *
7 * WHAT: two header-only helpers used by every subsystem that opens an outbound
8 * TCP connection from a worker thread (not the event loop):
9 * - brix_apply_socket_io_timeouts() — SO_RCVTIMEO + SO_SNDTIMEO on a fd.
10 * - brix_connect_fd_deadline() — a non-blocking connect(2) bounded by
11 * poll(POLLOUT) + getsockopt(SO_ERROR).
12 *
13 * WHY: native TPC (src/tpc/outbound/connect.c), the read-through cache origin fill
14 * (src/cache/origin_connection.c) and OCSP fetching (src/crypto/ocsp.c) each
15 * grew their OWN copy of the same hardening dance, with the same Linux caveat:
16 * SO_SNDTIMEO does not reliably bound connect(2), so an unreachable/black-holed
17 * peer must be bounded with a non-blocking connect + poll() deadline or it stalls
18 * the worker thread for a full TCP retransmit window (~2 min). Factoring the
19 * shared logic here keeps ONE audited implementation across protocols and lets
20 * the raw-fd connectors (TPC, cache) drop their near-identical private copies.
21 *
22 * HOW: header-only `static ngx_inline`, compiled into each translation unit (the
23 * netopt.h precedent — no new .c, so the `config` source list is unchanged).
24 * Every setsockopt is best-effort/non-fatal; the connect helper restores the
25 * fd's original blocking mode on success and sets errno to the SO_ERROR cause on
26 * a connect-time failure. OpenSSL/BIO connectors keep their own connect path but
27 * can still share the I/O-timeout helper.
28 */

45typedef enum {

47 BRIX_RESOLVE_ERR_DNS, /* getaddrinfo() failed for host:port */
48 BRIX_RESOLVE_ERR_SOCKET /* no resolved family yielded a usable socket */

51/*
52 * Resolve host:port (SOCK_STREAM) and create the first non-blocking socket whose
53 * address family succeeds; copy that sockaddr into *addr_out / *addrlen_out.
54 * af_policy constrains getaddrinfo's family — BRIX_AF_AUTO (AF_UNSPEC) tries all,
55 * BRIX_AF_INET / _INET6 force IPv4 / IPv6 only. Returns the connected-ready
56 * socket fd (caller owns it) with
57 * *status_out = BRIX_RESOLVE_OK, or NGX_INVALID_FILE with *status_out set to
58 * the failure reason (DNS vs no-usable-socket) so the caller logs its own
59 * message. Does no logging itself and leaves no fd open on failure.
60 *
61 * Shared by the event-driven outbound connectors (proxy upstream, root://
62 * upstream) which then wrap the fd in an nginx connection and connect() under
63 * the event loop — so this helper deliberately stops at a non-blocking socket
64 * + chosen address and does NOT call connect().
65 */
66static ngx_inline int
67brix_resolve_connect_socket(const char *host, unsigned port,
68 brix_af_policy_t af_policy,
69 struct sockaddr_storage *addr_out, socklen_t *addrlen_out,
70 brix_resolve_status_t *status_out)
71{
72 struct addrinfo hints;
73 struct addrinfo *res;
74 struct addrinfo *rp;
75 char port_str[16];
76 int fd = (int) NGX_INVALID_FILE;
77
78 ngx_memzero(&hints, sizeof(hints));
79 hints.ai_socktype = SOCK_STREAM;
80 hints.ai_family = (int) af_policy; /* AUTO==AF_UNSPEC keeps legacy behaviour */
81 snprintf(port_str, sizeof(port_str), "%u", port);
82
83 if (getaddrinfo(host, port_str, &hints, &res) != 0) {
84 *status_out = BRIX_RESOLVE_ERR_DNS;
85 return (int) NGX_INVALID_FILE;
86 }
87
88 for (rp = res; rp != NULL; rp = rp->ai_next) {
89 fd = ngx_socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
90 if (fd == (int) NGX_INVALID_FILE) {
91 continue;
92 }
93 if (ngx_nonblocking(fd) == NGX_ERROR) {
94 ngx_close_socket(fd);
95 fd = (int) NGX_INVALID_FILE;
96 continue;
97 }
98 ngx_memcpy(addr_out, rp->ai_addr, rp->ai_addrlen);
99 *addrlen_out = rp->ai_addrlen;
100 break;
101 }
102 freeaddrinfo(res);
103
104 if (fd == (int) NGX_INVALID_FILE) {
105 *status_out = BRIX_RESOLVE_ERR_SOCKET;
106 return (int) NGX_INVALID_FILE;
107 }
108
109 *status_out = BRIX_RESOLVE_OK;
110 return fd;
111}

128/*
129 * Connect fd to addr with an explicit poll() deadline (timeout_ms), so an
130 * unreachable peer cannot stall the calling thread for a full TCP retransmit
131 * window — SO_SNDTIMEO does not reliably bound connect(2) on Linux.
132 *
133 * Sets O_NONBLOCK, issues connect(2); on EINPROGRESS waits for POLLOUT up to
134 * timeout_ms then checks SO_ERROR. Returns 0 on success (fd restored to its
135 * original blocking mode), -1 on failure (on a connect-time failure errno is set
136 * to the SO_ERROR cause). On the error paths the fd is left non-blocking — the
137 * caller closes it on failure, so this is harmless.
138 */
139static ngx_inline int
140brix_connect_fd_deadline(int fd, const struct sockaddr *addr,
141 socklen_t addrlen, int timeout_ms)
142{
143 struct pollfd pfd;
144 int original_flags;
145 int rc;
146 int socket_error = 0;
147 socklen_t socket_error_len = sizeof(socket_error);
148
149 original_flags = fcntl(fd, F_GETFL, 0);
150 if (original_flags < 0) {
151 return -1;
152 }
153
154 if (fcntl(fd, F_SETFL, original_flags | O_NONBLOCK) != 0) {
155 return -1;
156 }
157
158 rc = connect(fd, addr, addrlen);
159 if (rc == 0) {
160 (void) fcntl(fd, F_SETFL, original_flags);
161 return 0;
162 }
163
164 if (errno != EINPROGRESS) {
165 return -1;
166 }
167
168 pfd.fd = fd;
169 pfd.events = POLLOUT;
170 pfd.revents = 0;
171
172 rc = poll(&pfd, 1, timeout_ms);
173 if (rc <= 0) {
174 return -1;
175 }
176
177 if (getsockopt(fd, SOL_SOCKET, SO_ERROR, &socket_error,
178 &socket_error_len) != 0)
179 {
180 return -1;
181 }
182
183 if (socket_error != 0) {
184 errno = socket_error;
185 return -1;
186 }
187
188 (void) fcntl(fd, F_SETFL, original_flags);
189 return 0;
190}
