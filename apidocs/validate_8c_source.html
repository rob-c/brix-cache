1/*
2 * validate.c — WLCG/SciToken JWT bearer-token validation and claim extraction.
3 *
4 * Validates JWTs (RS256/ES256) against a JWKS key set, enforces
5 * issuer/audience/exp/nbf, extracts claims (scopes, groups, iss/sub/aud), and
6 * handles macaroons as an alternative auth path. WLCG storage tokens are HEP's
7 * primary bearer mechanism (scope-granted paths + VO group membership), so this is
8 * the authoritative path a connection trusts before granting filesystem access.
9 * brix_token_validate() runs one deterministic pipeline: structural (3 segments)
10 * → alg check → key select (by kid, else single-key) → EVP signature → claim
11 * extraction → time window. Returns 0 (claims populated) / -1 (reason logged).
12 */

27/*
28 * Escape control characters and non-printable bytes in a string before it is
29 * written to the nginx error log. This prevents log-injection attacks where a
30 * malicious token embeds newlines or other control codes to forge log entries.
31 *
32 * Output format: printable ASCII is kept verbatim; anything < 0x20, 0x7f, or
33 * DEL is replaced with \xHH. The output is always NUL-terminated.
34 */
35/*
36 *
37 * WHAT: Escapes control characters and non-printable bytes in a string before it is written to the nginx error log. Printable ASCII (0x20-0x7e) passes through verbatim; anything below 0x20, at 0x7f, or DEL is replaced with \xHH hex notation. This prevents log-injection attacks where malicious tokens could embed newlines or other control codes to forge or corrupt log entries.
38 *
39 * WHY: JWT tokens are untrusted input that may contain arbitrary bytes (base64url decoded payloads). Without sanitization, an attacker could inject newline characters into a token's issuer or subject claim and make it appear as if separate log entries were generated — this is a log forgery attack vector. This function follows the AGENTS.md FAQ rule for "Log strings from wire" which mandates using brix_sanitize_log_string(). */
40static void
41token_sanitize_for_log(const char *in, char *out, size_t outsz)
42{
43 static const char hex[] = "0123456789abcdef";
44 size_t i = 0;
45
46 if (out == NULL || outsz == 0) {
47 return;
48 }
49 if (in == NULL) {
50 out[0] = '\0';
51 return;
52 }
53
54 while (*in && i + 1 < outsz) {
55 unsigned char c = (unsigned char) *in++;
56
57 if (c >= 0x20 && c < 0x7f) {
58 out[i++] = (char) c;
59 } else {
60 if (i + 4 >= outsz) {
61 break;
62 }
63 out[i++] = '\\';
64 out[i++] = 'x';
65 out[i++] = hex[c >> 4];
66 out[i++] = hex[c & 0x0f];
67 }
68 }
69 out[i] = '\0';
70}

87/*
88 *
89 * WHAT: Extracts the WLCG groups array from the JWT payload JSON into a comma-separated string stored in claims->groups. Reads up to 16 group names from the "wlcg.groups" claim and concatenates them with commas, truncating any individual group name that would exceed remaining buffer space. Empty result (no groups) produces an empty string.
90 *
91 * WHY: WLCG tokens include a groups array for VO membership attribution but nginx needs a flat comma-separated representation for logging and ACL matching. This helper converts the JSON array into a format compatible with existing path/acl.c logic which checks group memberships against configured VO ACLs without requiring additional JSON parsing at access time.
92 * HOW: Reads up to 16 group names from JSON "wlcg.groups" array, concatenates with commas into claims->groups buffer, truncating individual groups that exceed remaining space. */
93static void
94brix_token_extract_groups(const char *pay_json, size_t pay_len,
95 brix_token_claims_t *claims)
96{
97 char groups[16][256];
98 int i, gcount;
99
100 gcount = json_get_string_array(pay_json, pay_len, "wlcg.groups",
101 groups, 16);
102
103 claims->groups[0] = '\0';
104 for (i = 0; i < gcount; i++) {
105 size_t cur, rem, gl;
106
107 if (i > 0) {
108 cur = strlen(claims->groups);
109 if (cur + 1 < sizeof(claims->groups)) {
110 claims->groups[cur] = ',';
111 claims->groups[cur + 1] = '\0';
112 }
113 }
114
115 cur = strlen(claims->groups);
116 rem = sizeof(claims->groups) - cur - 1;
117 if (rem == 0) {
118 continue;
119 }
120
121 gl = strlen(groups[i]);
122 if (gl > rem) {
123 gl = rem;
124 }
125
126 ngx_memcpy(claims->groups + cur, groups[i], gl);
127 claims->groups[cur + gl] = '\0';
128 }
129}

132 * token_sig_ok — dispatch signature verification by algorithm.
133 *
134 * WHAT: Calls brix_token_verify_es256() for ES256 or brix_token_verify_rs256()
135 * for RS256 (the only two accepted algorithms at this call site).
136 * WHY: DRY helper shared by the kid-present single-key path and the kid-absent

138 * HOW: strcmp on alg (already validated non-NULL before this point).

144{
145 if (strcmp(alg, "ES256") == 0) {
146 return brix_token_verify_es256(token, signed_len, sig_bin, sig_len, pkey);

152/*
153 * brix_token_validate — verify a WLCG/SciToken JWT bearer token.
154 *
155 * Validates the token according to the WLCG token profile spec:
156 * https://zenodo.org/record/3992838
157 *
158 * Checks performed, in order:
159 * 1. Structural: exactly three base64url-encoded segments separated by '.'.
160 * 2. Algorithm: only "RS256" is accepted.
161 * alg:"none" bypass — a token that declares alg:"none" or any other
162 * algorithm is rejected before signature verification, preventing an
163 * attacker from forging an unsigned token.
164 * 3. Key selection: by "kid" header claim (exact match); when "kid" is
165 * absent all JWKS keys are tried in order (rotation grace, §3.3).
166 * 4. Signature: RSA PKCS#1v1.5 SHA-256 over header.payload.
167 * The signature MUST be verified before claims are trusted.
168 * 5. Issuer: if expected_issuer is non-empty, "iss" must match exactly.
169 * Issuer confusion attack — without this check an attacker can present a
170 * valid token from a different trusted issuer.
171 * 6. Audience: if expected_audience is non-empty, "aud" must match exactly.
172 * 7. Expiry ("exp") and not-before ("nbf") against the server clock.
173 * Clock-skew tolerance is deliberately absent; configure NTP.
174 *
175 * Preconditions:
176 * - keys/key_count contain loaded, trusted RSA public keys (JWKS).
177 * - claims is an output parameter; the caller need not initialise it.
178 *
179 * Postconditions on success (return 0):
180 * - claims->scopes and claims->scope_count contain the parsed scope list.
181 * - claims->iss/sub/aud/exp/nbf are populated from the payload.
182 * - The caller must check brix_token_check_read() / _write() before
183 * granting access to any specific path.
184 *
185 * Ownership: claims points into caller-provided storage; no heap allocation.
186 *
187 * Returns: 0 on success, -1 on any validation failure (reason logged).
188 */
189int
190brix_token_validate(ngx_log_t *log,
191 const char *token, size_t token_len,
192 const brix_jwks_key_t *keys, int key_count,
193 const char *expected_issuer, const char *expected_audience,
194 const u_char *macaroon_secret, size_t secret_len,
195 int clock_skew,
196 brix_token_claims_t *claims)
197{
198 xrdjwt_seg seg[3];
199 u_char hdr_json[2048], pay_json[4096], sig_bin[512];
200 ssize_t hdr_len, pay_len, sig_len;
201 char alg[16], kid[128];
202 int i;
203 time_t now;
204 size_t hdr_b64_len, pay_b64_len, sig_b64_len, signed_len;
205
206 ngx_memzero(claims, sizeof(*claims));
207
208 if (brix_token_is_macaroon(token, token_len)) {
209 if (macaroon_secret == NULL || secret_len == 0) {
210 ngx_log_error(NGX_LOG_WARN, log, 0,
211 "brix_token: Macaroon detected but no secret configured");
212 return -1;
213 }
214 if (brix_macaroon_validate(log, token, token_len,
215 macaroon_secret, secret_len, claims) != 0)
216 {
217 return -1;
218 }
219 /*
220 * Issuer pinning applies to macaroons too (fail-closed). The macaroon
221 * "location" packet is recorded in claims->iss; when the operator has
222 * configured an expected issuer, a macaroon whose location does not match
223 * — including one with no location packet (claims->iss == "") — is
224 * rejected, closing the same issuer-confusion gap the JWT path guards
225 * against below. Macaroons carry no audience claim, so expected_audience
226 * is intentionally not applied here.
227 */
228 if (expected_issuer != NULL && expected_issuer[0]
229 && strcmp(claims->iss, expected_issuer) != 0)
230 {
231 char safe_iss[512];
232 token_sanitize_for_log(claims->iss, safe_iss, sizeof(safe_iss));
233 ngx_log_error(NGX_LOG_WARN, log, 0,
234 "brix_macaroon: issuer/location mismatch: got \"%s\" "
235 "expected \"%s\"", safe_iss, expected_issuer);
236 return -1;
237 }
238 return 0;
239 }
240
241 if (token_len == 0 || token_len > 8192) {
242 ngx_log_error(NGX_LOG_WARN, log, 0,
243 "brix_token: token length invalid: %uz", token_len);
244 return -1;
245 }
246
247 /* Split "header.payload.signature" via the shared splitter (libxrdproto —
248 * the same xrdjwt_split() the native client's token introspection uses), so
249 * the two-dot scan is single-sourced across both trees. */
250 if (xrdjwt_split((const char *) token, token_len, seg) != 0) {
251 return brix_token_malformed(log); /* fewer than 2 dots */
252 }
253
254 /* Compact JWS is EXACTLY 3 segments; xrdjwt_split folds any extra dots into
255 * the signature slice, so reject a dot inside it here — keeping this gate's
256 * strict structural check (defence-in-depth) and its "malformed" log path. */
257 if (memchr(seg[2].p, '.', seg[2].n) != NULL) {
258 return brix_token_malformed(log);
259 }
260
261 hdr_b64_len = seg[0].n;
262 pay_b64_len = seg[1].n;
263 sig_b64_len = seg[2].n;
264
265 hdr_len = b64url_decode(token, hdr_b64_len, hdr_json,
266 sizeof(hdr_json) - 1);
267 if (hdr_len < 0) {
268 return brix_token_malformed(log);
269 }
270 hdr_json[hdr_len] = '\0';
271
272 ngx_memzero(alg, sizeof(alg));
273 ngx_memzero(kid, sizeof(kid));
274 json_get_string((char *) hdr_json, (size_t) hdr_len, "alg", alg,
275 sizeof(alg));
276 /* alg:"none" bypass: reject any algorithm other than RS256/ES256 before
277 * touching the signature. An unsigned token (alg:"none") must be
278 * rejected here, not later, or the signature step is skipped. */
279 if (strcmp(alg, "RS256") != 0 && strcmp(alg, "ES256") != 0) {
280 char safe_alg[64];
281 token_sanitize_for_log(alg, safe_alg, sizeof(safe_alg));
282 ngx_log_error(NGX_LOG_WARN, log, 0,
283 "brix_token: unsupported JWT algorithm \"%s\" "
284 "(only RS256 and ES256 accepted)", safe_alg);
285 return -1;
286 }
287
288 json_get_string((char *) hdr_json, (size_t) hdr_len, "kid", kid,
289 sizeof(kid));
290
291 /* RFC 7515 §4.1.11: we implement no `crit` extension parameters, so any
292 * token asserting critical headers we do not understand MUST be rejected. */
293 if (json_has_member((char *) hdr_json, (size_t) hdr_len, "crit")) {
294 ngx_log_error(NGX_LOG_WARN, log, 0,
295 "brix_token: JWS 'crit' header not supported — rejecting");
296 return -1;
297 }
298
299 sig_len = b64url_decode(seg[2].p, sig_b64_len, sig_bin, sizeof(sig_bin));
300 if (sig_len < 0) {
301 ngx_log_error(NGX_LOG_WARN, log, 0,
302 "brix_token: cannot decode JWT signature");
303 return -1;
304 }
305
306 signed_len = hdr_b64_len + 1 + pay_b64_len; /* "header.payload" */
307 {
308 int sig_ok = 0;
309 if (kid[0] != '\0') {
310 /* kid asserted: exact-match the named key only. */
311 EVP_PKEY *pkey = NULL;
312 for (i = 0; i < key_count; i++) {
313 if (strcmp(keys[i].kid, kid) == 0) {
314 pkey = keys[i].pkey;
315 break;
316 }
317 }
318 if (pkey == NULL && key_count == 1) {
319 pkey = keys[0].pkey; /* preserve legacy single-key leniency */
320 }
321 if (pkey == NULL) {
322 char safe_kid[256];
323 token_sanitize_for_log(kid, safe_kid, sizeof(safe_kid));
324 ngx_log_error(NGX_LOG_WARN, log, 0,
325 "brix_token: no JWKS key matching kid=\"%s\"",
326 safe_kid);
327 return -1;
328 }
329 sig_ok = token_sig_ok(alg, (const u_char *) token, signed_len,
330 sig_bin, (size_t) sig_len, pkey);
331 } else {
332 /* kid absent: try every JWKS key in order (rotation grace, §3.3). */
333 for (i = 0; i < key_count && !sig_ok; i++) {
334 sig_ok = token_sig_ok(alg, (const u_char *) token, signed_len,
335 sig_bin, (size_t) sig_len, keys[i].pkey);
336 }
337 }
338 if (!sig_ok) {
339 ngx_log_error(NGX_LOG_WARN, log, 0,
340 "brix_token: JWT signature verification failed");
341 return -1;
342 }
343 }
344
345 pay_len = b64url_decode(seg[1].p, pay_b64_len, pay_json,
346 sizeof(pay_json) - 1);
347 if (pay_len < 0) {
348 return brix_token_malformed(log);
349 }
350 pay_json[pay_len] = '\0';
351
352 json_get_string((char *) pay_json, (size_t) pay_len, "iss",
353 claims->iss, sizeof(claims->iss));
354 json_get_string((char *) pay_json, (size_t) pay_len, "sub",
355 claims->sub, sizeof(claims->sub));
356 if (json_get_string((char *) pay_json, (size_t) pay_len, "aud",
357 claims->aud, sizeof(claims->aud)) < 0) {
358 /* RFC 7519 §4.1.3: "aud" MAY be an array of strings (WLCG/OIDC commonly
359 * emit it that way). json_get_string() only accepts a JSON string, so
360 * record the first array entry for logging/audit; the membership check
361 * below accepts the string OR array form. */
362 char aud_arr[1][256];
363 if (json_get_string_array((char *) pay_json, (size_t) pay_len, "aud",
364 aud_arr, 1) > 0) {
365 ngx_cpystrn((u_char *) claims->aud, (u_char *) aud_arr[0],
366 sizeof(claims->aud));
367 }
368 }
369 json_get_string((char *) pay_json, (size_t) pay_len, "scope",
370 claims->scope_raw, sizeof(claims->scope_raw));
371
372 if (json_get_int64((char *) pay_json, (size_t) pay_len, "exp",
373 &claims->exp) != 0
374 || claims->exp <= 0)
375 {
376 ngx_log_error(NGX_LOG_WARN, log, 0,
377 "brix_token: token missing valid positive exp");
378 return -1;
379 }
380 json_get_int64((char *) pay_json, (size_t) pay_len, "nbf",
381 &claims->nbf);
382 json_get_int64((char *) pay_json, (size_t) pay_len, "iat",
383 &claims->iat);
384
385 brix_token_extract_groups((char *) pay_json, (size_t) pay_len, claims);
386
387 /* Issuer-confusion attack prevention: without this check an attacker
388 * could present a valid token from a different trusted issuer. */
389 if (expected_issuer != NULL && expected_issuer[0]) {
390 if (strcmp(claims->iss, expected_issuer) != 0) {
391 char safe_iss[512];
392 token_sanitize_for_log(claims->iss, safe_iss, sizeof(safe_iss));
393 ngx_log_error(NGX_LOG_WARN, log, 0,
394 "brix_token: issuer mismatch: got \"%s\" "
395 "expected \"%s\"", safe_iss, expected_issuer);
396 return -1;
397 }
398 }
399
400 if (expected_audience != NULL && expected_audience[0] != '\0') {
401 /* Accept the expected audience whether "aud" is a single string or an
402 * array of strings containing it (RFC 7519 §4.1.3). Also accept the
403 * WLCG-profile wildcard audience (rules 104/105): a token whose aud is
404 * 'https://wlcg.cern.ch/jwt/v1/any' MUST be accepted by any WLCG
405 * endpoint regardless of its locally configured audience. */
406 int aud_ok =
407 json_string_or_array_contains((char *) pay_json, (size_t) pay_len,
408 "aud", expected_audience)
409 || json_string_or_array_contains((char *) pay_json, (size_t) pay_len,
410 "aud",
411 "https://wlcg.cern.ch/jwt/v1/any");
412 if (!aud_ok) {
413 char safe_aud[512];
414 token_sanitize_for_log(claims->aud, safe_aud, sizeof(safe_aud));
415 ngx_log_error(NGX_LOG_WARN, log, 0,
416 "brix_token: audience mismatch: got \"%s\" "
417 "expected \"%s\"", safe_aud,
418 expected_audience);
419 return -1;
420 }
421 }
422
423 now = time(NULL);
424
425 /* Apply a clock-skew window so that tokens from systems whose clock differs
426 * from ours by a few seconds are not spuriously rejected. The skew widens
427 * the acceptance window on both sides without permanently extending validity.
428 *
429 * Guard against int64 overflow: json_get_int64() clamps a far-future NumericDate
430 * (e.g. 99999999999999999999) to INT64_MAX, so claims->exp + clock_skew can
431 * overflow. Use saturating addition — if the sum would overflow, the expiry is
432 * effectively infinite and the token is always valid. */
433 {
434 int64_t exp_limit;
435
436 if (clock_skew > 0
437 && claims->exp > (int64_t) INT64_MAX - (int64_t) clock_skew)
438 {
439 exp_limit = INT64_MAX; /* saturate: far-future, never expired */
440 } else {
441 exp_limit = claims->exp + (int64_t) clock_skew;
442 }
443 if (now > (time_t) exp_limit) {
444 ngx_log_error(NGX_LOG_WARN, log, 0,
445 "brix_token: token expired at %L (now=%L skew=%d)",
446 (long long) claims->exp, (long long) now,
447 clock_skew);
448 return -1;
449 }
450 }
451
452 if (claims->nbf > 0 && now < (time_t) claims->nbf)
453 {
454 ngx_log_error(NGX_LOG_WARN, log, 0,
455 "brix_token: token not yet valid (nbf=%L now=%L)",
456 (long long) claims->nbf, (long long) now);
457 return -1;
458 }
459
460 claims->scope_count = brix_token_parse_scopes(
461 claims->scope_raw, claims->scopes, BRIX_MAX_TOKEN_SCOPES);
462
463 /* Only build the ~2.5 KB of sanitized log strings when INFO logging is
464 * actually enabled — otherwise this is pure wasted work on every validation
465 * (the hot path under token-auth load). */
466 if (log->log_level >= NGX_LOG_INFO) {
467 char safe_sub[512], safe_iss[512], safe_scope[1024], safe_groups[512];
468 token_sanitize_for_log(claims->sub, safe_sub, sizeof(safe_sub));
469 token_sanitize_for_log(claims->iss, safe_iss, sizeof(safe_iss));
470 token_sanitize_for_log(claims->scope_raw, safe_scope, sizeof(safe_scope));
471 token_sanitize_for_log(claims->groups, safe_groups, sizeof(safe_groups));
472 ngx_log_error(NGX_LOG_INFO, log, 0,
473 "brix_token: valid token sub=\"%s\" iss=\"%s\" "
474 "scope=\"%s\" groups=\"%s\" scopes=%d",
475 safe_sub, safe_iss, safe_scope, safe_groups,
476 claims->scope_count);
477 }
478 return 0;
479}

481/* brix_token_peek_iss — read the "iss" claim WITHOUT trusting the signature
482 * (xrdjwt_split + b64url_decode + json_get_string): the registry must pick an
483 * issuer, and thus its verification keys, before it can trust anything, so this
484 * read is explicitly untrusted and re-derived from verified claims afterwards. */
485int
486brix_token_peek_iss(const char *token, size_t token_len,
487 char *out, size_t outsz)
488{
489 xrdjwt_seg seg[3];
490 u_char pay[4096];
491 ssize_t n;
492
493 out[0] = '\0';
494
495 if (xrdjwt_split(token, token_len, seg) != 0) {
496 return -1; /* not a compact JWS */
497 }
498 n = b64url_decode(seg[1].p, seg[1].n, pay, sizeof(pay) - 1);
499 if (n < 0) {
500 return -1;
501 }
502 pay[n] = '\0';
503 if (json_get_string((char *) pay, (size_t) n, "iss", out, outsz) < 0) {
504 return -1;
505 }
506 return 0;
507}

509/* brix_token_validate_registry_authn — registry authN with no path gate: peek
510 * iss → registry_find → validate() with THAT issuer's keys → multi-audience accept.
511 * Stream kXR_auth happens before any path is known, so issuer-keyed authentication
512 * and per-path authorization are split; this is the authN half (also reused by the
513 * combined HTTP entry point). On success returns 0, fills *claims, sets *out_issuer. */
514int
515brix_token_validate_registry_authn(ngx_log_t *log,
516 const char *token, size_t token_len,
517 const brix_token_registry_t *reg,
518 const u_char *macaroon_secret, size_t secret_len,
519 int clock_skew,
520 brix_token_claims_t *claims, const brix_token_issuer_t **out_issuer)
521{
522 char iss[256];
523 const brix_token_issuer_t *is;
524 const char *expected_aud;
525 int i;
526
527 *out_issuer = NULL;
528
529 if (brix_token_peek_iss(token, token_len, iss, sizeof(iss)) != 0) {
530 ngx_log_error(NGX_LOG_WARN, log, 0,
531 "brix_token: cannot read iss for issuer selection");
532 return -1;
533 }
534 is = brix_token_registry_find(reg, iss);
535 if (is == NULL) {
536 char safe[512];
537 token_sanitize_for_log(iss, safe, sizeof(safe));
538 ngx_log_error(NGX_LOG_WARN, log, 0,
539 "brix_token: unknown issuer \"%s\"", safe);
540 return -1;
541 }
542
543 /* Verify with THIS issuer's keys; validate() re-checks iss after the
544 * signature is trusted. A single declared audience is handed to validate()
545 * for correct string-or-array membership; multiple audiences are accepted
546 * best-effort below (full multi-audience array membership = W1b). */
547 expected_aud = (is->audience_count == 1) ? is->audiences[0] : NULL;
548 if (brix_token_validate(log, token, token_len,
549 is->jwks_key_count ? is->jwks_keys : NULL, is->jwks_key_count,
550 is->issuer, expected_aud, macaroon_secret, secret_len,
551 clock_skew, claims) != 0)
552 {
553 return -1;
554 }
555
556 if (expected_aud == NULL && is->audience_count > 0) {
557 int ok = 0;
558 for (i = 0; i < is->audience_count && !ok; i++) {
559 ok = (strcmp(claims->aud, is->audiences[i]) == 0);

576/* brix_token_authz_strategy — the per-path authorization ladder: enforce the
577 * issuer base_path/restricted_path gate, then run the authorization_strategy
578 * (capability / group / mapping) for (req_path, op). Shared by the HTTP combined
579 * path and the stream per-path identity check. Returns 1 = ALLOW, 0 = DENY. */
580int
581brix_token_authz_strategy(const brix_token_issuer_t *is,
582 const brix_token_claims_t *claims, const char *req_path,
583 brix_token_op_e op)
584{
585 if (!brix_token_issuer_path_ok(is, req_path)) {
586 return 0;
587 }
588
589 /* capability — the token's own WLCG scopes must cover (path, op). */
590 if (is->strategy & BRIX_AUTHZ_CAPABILITY) {
591 int ok = (op == BRIX_TOKEN_OP_WRITE)
592 ? brix_token_check_write(claims->scopes, claims->scope_count,
593 req_path)
594 : brix_token_check_read(claims->scopes, claims->scope_count,
595 req_path);
596 if (ok) {
597 return 1;
598 }
599 }
600
601 /* group — the issuer vouches for its base_path for any token bearing a
602 * group claim; the per-VO/group ACL layer refines the actual access using
603 * the identity's groups (claims->groups → identity vo_list). */
604 if (is->strategy & BRIX_AUTHZ_GROUP) {
605 if (claims->groups[0] != '\0') {
606 return 1;
607 }
608 }
609
610 /* mapping — the subject must resolve to a local user (map_subject +
611 * name_mapfile, with onmissing/default policy); the per-user ACL/POSIX
612 * layer then enforces that user's permissions within base_path. */
613 if (is->strategy & BRIX_AUTHZ_MAPPING) {
614 char user[64];
615
616 if (!is->map_subject) {
617 return 1; /* trust the issuer's subject as-is */
618 }
619 if (is->name_mapfile[0] != '\0'
620 && brix_subject_mapfile_lookup(is->name_mapfile, claims->sub,
621 user, sizeof(user)) == 0)
622 {
623 return 1;
624 }
625 if (!is->onmissing_fail && is->default_user[0] != '\0') {
626 return 1; /* fall back to default_user */
627 }
628 }
629
630 return 0;
631}
