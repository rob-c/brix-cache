1/*
2 * bootstrap_pack.h — XRootD outbound session-bootstrap request packers
3 * (single source of truth for the handshake/login wire layout).
4 *
5 * WHAT: header-only static inlines that fill the three fixed-layout request
6 * structs every outbound XRootD session sends, in order, before any file op:
7 * - xrd_pack_handshake: ClientInitHandShake (20B init hello)
8 * - xrd_pack_protocol_request: ClientProtocolRequest(24B kXR_protocol)
9 * - xrd_pack_login_request: ClientLoginRequest (24B kXR_login)
10 * WHY: this exact byte layout was hand-assembled in FOUR places — the native
11 * client (client/lib/conn.c), the proxy upstream (src/upstream/bootstrap.c,
12 * twice: bootstrap + TLS-resend), and native TPC pull (src/tpc/outbound/bootstrap.c).
13 * It is security-relevant (protocol version, TLS-capability flags, login
14 * capver) and was therefore audited 4× and could drift between roles. One
15 * header keeps the wire facts in a single place; the per-role policy that
16 * genuinely differs (streamid ownership, TLS flags, username, pid, async
17 * capability) stays at the call site as explicit parameters.
18 * HOW: header-only static inlines over the packed wire structs — no ngx, no
19 * allocation, no OpenSSL — so the same code compiles into both the nginx
20 * module and the ngx-free libxrdproto archive the native client links.
21 * Each packer pre-zeroes its struct (so reserved/padding bytes and dlen
22 * are 0) then writes only the fields it owns. Callers that manage their own
23 * streamid (the client's xrdc_send stamps it post-pack) pass {0,0}.
24 *
25 * Clean-room: layouts from src/protocol/wire_core_requests.h (vs XProtocol.hh).
26 */

38/*
39 * ClientInitHandShake = {0,0,0, htonl(4), htonl(ROOTD_PQ)} (20 bytes). The first
40 * three words are zero; the server validates fourth==htonl(4) and
41 * fifth==htonl(2012=ROOTD_PQ) to accept the client as protocol-compatible.
42 * Identical for every role — no parameters.
43 */
44static inline void
45xrd_pack_handshake(ClientInitHandShake *hs)
46{
47 memset(hs, 0, sizeof(*hs));
48 hs->fourth = htonl(4);
49 hs->fifth = htonl(ROOTD_PQ);
50}

52/*
53 * ClientProtocolRequest (kXR_protocol, 24 bytes). `flags` is the per-role
54 * capability byte (0 for a server-internal connector; kXR_secreqs|kXR_ableTLS
55 * [|kXR_wantTLS] for the user-facing client requesting the security trailer and
56 * advertising/forcing TLS). `streamid` is the 2-byte id the caller owns (a
57 * server connector uses a fixed {0,1}; the client passes its protocol streamid).
58 * clientpv, requestid and expect (kXR_ExpLogin=0x03) are fixed wire facts.
59 */
60static inline void
61xrd_pack_protocol_request(ClientProtocolRequest *pr,
62 const uint8_t streamid[2], uint8_t flags)
63{
64 xrdw_protocol_req_t b = { .clientpv = kXR_PROTOCOLVERSION,
65 .flags = flags, .expect = kXR_ExpLogin };
66 memset(pr, 0, sizeof(*pr));
67 pr->streamid[0] = streamid[0];
68 pr->streamid[1] = streamid[1];
69 pr->requestid = htons(kXR_protocol);
70 xrdw_protocol_req_pack(&b, ((ClientRequestHdr *) pr)->body); /* clientpv/flags/expect */
71 pr->dlen = 0;
72}
73
74/*
75 * ClientLoginRequest (kXR_login, 24 bytes). The auth-relevant policy is the
76 * caller's: `username` (a NUL-terminated string, copied NUL-padded into the
77 * 8-byte wire field and truncated at 8 — matching XRootD's strnlen contract),
78 * `pid` (informational), and `capver` (kXR_ver005, optionally |kXR_asyncap).
79 * `streamid` is {0,1} for a server connector; the client passes {0,0} because
80 * its xrdc_send stamps the real streamid (and dlen) after packing. ability and
81 * ability2 stay 0 (pre-zeroed); dlen=0 means anonymous (no credential payload).
82 */
83static inline void
84xrd_pack_login_request(ClientLoginRequest *lr, const uint8_t streamid[2],
85 int32_t pid, const char *username, uint8_t capver)
86{
87 size_t n = (username != NULL) ? strlen(username) : 0;
88 xrdw_login_req_t b = { .pid = pid, .capver = capver };
89
90 if (n > sizeof(b.username)) {
91 n = sizeof(b.username);
92 }
93 if (n > 0) {
94 memcpy(b.username, username, n); /* NUL-padded; b pre-zeroed by init */
95 }
96
97 memset(lr, 0, sizeof(*lr));
98 lr->streamid[0] = streamid[0];
99 lr->streamid[1] = streamid[1];
100 lr->requestid = htons(kXR_login);
101 xrdw_login_req_pack(&b, ((ClientRequestHdr *) lr)->body); /* pid/username/capver */
102 lr->dlen = 0;
103}
