19/* ---- deleg_proxy_leaf_dn ----------------------------------------------------
20 *
21 * WHAT: Extract the leaf (first) certificate's subject DN from a proxy PEM into
22 * `out` (NUL-terminated), using the same normalisation as the front-door
23 * DN (brix_x509_oneline). Returns 1 on success, 0 if the bytes do not
24 * parse as a PEM certificate.
25 *
26 * WHY: §5.1 forbids a privilege swap: the supplied proxy's identity must equal
27 * the authenticated identity. Comparing the leaf DN to identity->dn with
28 * matching normalisation is the string-level identity binding (the
29 * cryptographic chain-trust check is enforced later in the VFS gate).
30 *
31 * HOW: A BIO over the bytes → PEM_read_bio_X509 reads the first cert (the leaf,
32 * as written by the client's proxy chain); brix_x509_oneline renders its
33 * subject. Frees the cert + BIO. */
34static int
35deleg_proxy_leaf_dn(const u_char *pem, size_t len, char *out, size_t outsz)
36{
37 BIO *bio;
38 X509 *leaf;
39
40 bio = BIO_new_mem_buf(pem, (int) len);
41 if (bio == NULL) {
42 return 0;
43 }
44
45 leaf = PEM_read_bio_X509(bio, NULL, NULL, NULL);
46 BIO_free(bio);
47
48 if (leaf == NULL) {
49 return 0;
50 }
51
52 brix_x509_oneline(X509_get_subject_name(leaf), out, outsz);
53 X509_free(leaf);
54 return 1;
55}

96 }
97

101 return NGX_HTTP_FORBIDDEN;
102 }

108 "brix: " BRIX_DELEG_PROXY_HEADER
109 " does not decode to a PEM certificate");
110 return NGX_HTTP_FORBIDDEN;
111 }
112
113 if (identity == NULL || identity->dn.len == 0
114 || identity->dn.len != ngx_strlen(leaf_dn)
115 || ngx_strncmp(identity->dn.data, leaf_dn, identity->dn.len) != 0)
116 {
117 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
118 "brix: rejecting " BRIX_DELEG_PROXY_HEADER
119 " — proxy leaf DN does not match authenticated identity");
120 return NGX_HTTP_FORBIDDEN;
121 }
122
123 *out = decoded;
124 return NGX_OK;
125}
